use std::cmp::Ordering;
use std::convert::TryFrom;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    value: i128,
    index: usize,
}

struct Scanner {
    buf: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, pos: 0 }
    }

    fn skip_ws(&mut self) {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
    }

    fn next_i128(&mut self) -> i128 {
        self.skip_ws();
        let mut sign = 1i128;
        if self.pos < self.buf.len() && self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.pos < self.buf.len() && self.buf[self.pos] == b'+' {
            self.pos += 1;
        }
        let mut v = 0i128;
        while self.pos < self.buf.len() {
            let c = self.buf[self.pos];
            if !c.is_ascii_digit() {
                break;
            }
            v = v * 10 + i128::from(c - b'0');
            self.pos += 1;
        }
        v * sign
    }

    fn next_usize(&mut self) -> usize {
        self.next_i128() as usize
    }
}

fn gcd_i128(mut a: i128, mut b: i128) -> i128 {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    if a < 0 { -a } else { a }
}

fn push_subset(indices: &mut Vec<usize>, mask: u64, base: usize, count: usize) {
    for bit in 0..count {
        if (mask >> bit) & 1 == 1 {
            indices.push(base + bit);
        }
    }
}

fn enumerate_sums(items: &[Item]) -> Vec<(i128, u64)> {
    let mut sums: Vec<(i128, u64)> = vec![(0, 0)];
    for (bit, item) in items.iter().enumerate() {
        let bit_mask = 1u64 << bit;
        let len = sums.len();
        sums.reserve(len);
        for i in 0..len {
            let (s, m) = sums[i];
            sums.push((s + item.value, m | bit_mask));
        }
    }
    sums
}

fn solve_mitm(items: &[Item], target: i128) -> Option<Vec<usize>> {
    let mid = items.len() / 2;
    let left = &items[..mid];
    let right = &items[mid..];
    let mut left_sums = enumerate_sums(left);
    let mut right_sums = enumerate_sums(right);
    left_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));
    right_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    let mut i = 0usize;
    let mut j = right_sums.len();
    while i < left_sums.len() && j > 0 {
        let rhs = right_sums[j - 1].0;
        let sum = left_sums[i].0 + rhs;
        match sum.cmp(&target) {
            Ordering::Less => i += 1,
            Ordering::Greater => j -= 1,
            Ordering::Equal => {
                let mut ans = Vec::new();
                push_subset(&mut ans, left_sums[i].1, 0, left.len());
                push_subset(&mut ans, right_sums[j - 1].1, mid, right.len());
                ans.sort_unstable();
                return Some(ans);
            }
        }
    }
    None
}

fn shift_left(src: &[u64], shift: usize, dst: &mut [u64]) {
    dst.fill(0);
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    if word_shift >= src.len() {
        return;
    }
    for i in 0..src.len() {
        let v = src[i];
        if v == 0 {
            continue;
        }
        let j = i + word_shift;
        if j < dst.len() {
            dst[j] |= v << bit_shift;
        }
        if bit_shift != 0 {
            let k = j + 1;
            if k < dst.len() {
                dst[k] |= v >> (64 - bit_shift);
            }
        }
    }
}

fn shift_right(src: &[u64], shift: usize, dst: &mut [u64]) {
    dst.fill(0);
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    if word_shift >= src.len() {
        return;
    }
    for i in word_shift..src.len() {
        let v = src[i];
        if v == 0 {
            continue;
        }
        let j = i - word_shift;
        dst[j] |= v >> bit_shift;
        if bit_shift != 0 && j > 0 {
            dst[j - 1] |= v << (64 - bit_shift);
        }
    }
}

fn reconstruct_from_parents(
    mut idx: usize,
    offset: usize,
    parent_sum: &[i32],
    parent_item: &[i32],
    items: &[Item],
) -> Option<Vec<usize>> {
    let mut out = Vec::new();
    while idx != offset {
        let item = parent_item[idx];
        let prev = parent_sum[idx];
        if item < 0 || prev < 0 {
            return None;
        }
        out.push(items[item as usize].index);
        idx = prev as usize;
    }
    out.sort_unstable();
    Some(out)
}

fn solve_bitset(items: &[Item], target: i128) -> Option<Vec<usize>> {
    if items.is_empty() {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let mut min_sum = 0i128;
    let mut max_sum = 0i128;
    for item in items {
        if item.value < 0 {
            min_sum += item.value;
        } else {
            max_sum += item.value;
        }
    }
    if target < min_sum || target > max_sum {
        return None;
    }

    let range_i128 = max_sum - min_sum;
    let range = usize::try_from(range_i128).ok()?;
    let offset = usize::try_from(-min_sum).ok()?;
    let target_idx = usize::try_from(target - min_sum).ok()?;

    let words = (range + 64) / 64;
    let mut bits = vec![0u64; words];
    let mut temp = vec![0u64; words];
    let mut parent_sum = vec![-1i32; range + 1];
    let mut parent_item = vec![-1i32; range + 1];
    bits[offset / 64] |= 1u64 << (offset % 64);
    parent_sum[offset] = offset as i32;
    parent_item[offset] = -2;

    for (item_idx, item) in items.iter().enumerate() {
        if item.value >= 0 {
            let shift = usize::try_from(item.value).ok()?;
            shift_left(&bits, shift, &mut temp);
        } else {
            let shift = usize::try_from(-item.value).ok()?;
            shift_right(&bits, shift, &mut temp);
        }

        let mut any_new = false;
        for w in 0..words {
            let new_bits = temp[w] & !bits[w];
            if new_bits != 0 {
                any_new = true;
                let mut x = new_bits;
                while x != 0 {
                    let bit = x.trailing_zeros() as usize;
                    let s = w * 64 + bit;
                    if s <= range && parent_item[s] == -1 {
                        let prev = if item.value >= 0 {
                            s as i128 - item.value
                        } else {
                            s as i128 + (-item.value)
                        };
                        if prev >= 0 && prev <= range as i128 {
                            parent_sum[s] = prev as i32;
                            parent_item[s] = item_idx as i32;
                        }
                    }
                    x &= x - 1;
                }
            }
            bits[w] |= temp[w];
        }

        if any_new {
            let word = target_idx / 64;
            let bit = target_idx % 64;
            if ((bits[word] >> bit) & 1) == 1 {
                return reconstruct_from_parents(target_idx, offset, &parent_sum, &parent_item, items);
            }
        }
    }

    let word = target_idx / 64;
    let bit = target_idx % 64;
    if ((bits[word] >> bit) & 1) == 0 {
        return None;
    }
    reconstruct_from_parents(target_idx, offset, &parent_sum, &parent_item, items)
}

fn dfs_branch_and_bound(
    idx: usize,
    current: i128,
    target: i128,
    items: &[Item],
    suffix_min: &[i128],
    suffix_max: &[i128],
    chosen: &mut Vec<usize>,
    answer: &mut Option<Vec<usize>>,
) {
    if answer.is_some() {
        return;
    }
    if current + suffix_min[idx] > target || current + suffix_max[idx] < target {
        return;
    }
    if idx == items.len() {
        if current == target {
            let mut out = chosen.clone();
            out.sort_unstable();
            *answer = Some(out);
        }
        return;
    }

    let item = items[idx];
    let include_first = (target - current - item.value).abs() <= (target - current).abs();

    let try_include = |chosen: &mut Vec<usize>,
                       answer: &mut Option<Vec<usize>>,
                       idx: usize,
                       current: i128,
                       target: i128,
                       items: &[Item],
                       suffix_min: &[i128],
                       suffix_max: &[i128]| {
        chosen.push(items[idx].index);
        dfs_branch_and_bound(
            idx + 1,
            current + items[idx].value,
            target,
            items,
            suffix_min,
            suffix_max,
            chosen,
            answer,
        );
        chosen.pop();
    };

    let try_exclude = |chosen: &mut Vec<usize>,
                       answer: &mut Option<Vec<usize>>,
                       idx: usize,
                       current: i128,
                       target: i128,
                       items: &[Item],
                       suffix_min: &[i128],
                       suffix_max: &[i128]| {
        dfs_branch_and_bound(
            idx + 1,
            current,
            target,
            items,
            suffix_min,
            suffix_max,
            chosen,
            answer,
        );
    };

    if include_first {
        try_include(chosen, answer, idx, current, target, items, suffix_min, suffix_max);
        if answer.is_none() {
            try_exclude(chosen, answer, idx, current, target, items, suffix_min, suffix_max);
        }
    } else {
        try_exclude(chosen, answer, idx, current, target, items, suffix_min, suffix_max);
        if answer.is_none() {
            try_include(chosen, answer, idx, current, target, items, suffix_min, suffix_max);
        }
    }
}

fn solve_dfs(items: &[Item], target: i128) -> Option<Vec<usize>> {
    let mut sorted = items.to_vec();
    sorted.sort_unstable_by(|a, b| {
        let av = a.value.unsigned_abs();
        let bv = b.value.unsigned_abs();
        bv.cmp(&av).then_with(|| a.index.cmp(&b.index))
    });
    let n = sorted.len();
    let mut suffix_min = vec![0i128; n + 1];
    let mut suffix_max = vec![0i128; n + 1];
    for i in (0..n).rev() {
        let v = sorted[i].value;
        suffix_min[i] = suffix_min[i + 1] + if v < 0 { v } else { 0 };
        suffix_max[i] = suffix_max[i + 1] + if v > 0 { v } else { 0 };
    }
    let mut chosen = Vec::new();
    let mut answer = None;
    dfs_branch_and_bound(
        0,
        0,
        target,
        &sorted,
        &suffix_min,
        &suffix_max,
        &mut chosen,
        &mut answer,
    );
    answer
}

fn solve_positive_only(items: &[Item], target: i128) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if target < 0 {
        return None;
    }
    let mut filtered: Vec<Item> = items.iter().copied().filter(|x| x.value > 0 && x.value <= target).collect();
    for item in items {
        if item.value == target {
            return Some(vec![item.index]);
        }
    }
    if filtered.is_empty() {
        return None;
    }
    filtered.sort_unstable_by(|a, b| b.value.cmp(&a.value).then_with(|| a.index.cmp(&b.index)));

    let max_sum = usize::try_from(target).ok()?;
    let words = (max_sum + 64) / 64;
    let mut bits = vec![0u64; words];
    let mut temp = vec![0u64; words];
    let mut parent_sum = vec![-1i32; max_sum + 1];
    let mut parent_item = vec![-1i32; max_sum + 1];
    bits[0] |= 1;
    parent_sum[0] = 0;
    parent_item[0] = -2;

    for (item_idx, item) in filtered.iter().enumerate() {
        let shift = usize::try_from(item.value).ok()?;
        shift_left(&bits, shift, &mut temp);
        let mut any_new = false;
        for w in 0..words {
            let new_bits = temp[w] & !bits[w];
            if new_bits != 0 {
                any_new = true;
                let mut x = new_bits;
                while x != 0 {
                    let bit = x.trailing_zeros() as usize;
                    let s = w * 64 + bit;
                    if s <= max_sum && parent_item[s] == -1 {
                        let prev = s as i128 - item.value;
                        if prev >= 0 && prev <= max_sum as i128 {
                            parent_sum[s] = prev as i32;
                            parent_item[s] = item_idx as i32;
                        }
                    }
                    x &= x - 1;
                }
            }
            bits[w] |= temp[w];
        }
        if any_new {
            let word = max_sum / 64;
            let bit = max_sum % 64;
            if ((bits[word] >> bit) & 1) == 1 {
                let mut ans = reconstruct_from_parents(max_sum, 0, &parent_sum, &parent_item, &filtered)?;
                ans.sort_unstable();
                return Some(ans);
            }
        }
    }

    let word = max_sum / 64;
    let bit = max_sum % 64;
    if ((bits[word] >> bit) & 1) == 0 {
        return None;
    }
    let mut ans = reconstruct_from_parents(max_sum, 0, &parent_sum, &parent_item, &filtered)?;
    ans.sort_unstable();
    Some(ans)
}

fn solve_negative_only(items: &[Item], target: i128) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if target > 0 {
        return None;
    }
    let negated: Vec<Item> = items
        .iter()
        .copied()
        .map(|x| Item {
            value: -x.value,
            index: x.index,
        })
        .collect();
    let res = solve_positive_only(&negated, -target)?;
    Some(res)
}

fn main() {
    let mut sc = Scanner::new();
    if sc.buf.is_empty() {
        return;
    }

    let n = sc.next_usize();
    let target_in = sc.next_i128();
    let mut items = Vec::with_capacity(n);
    for i in 0..n {
        let v = sc.next_i128();
        if v != 0 {
            items.push(Item { value: v, index: i });
        }
    }

    if target_in == 0 {
        println!("YES");
        println!();
        return;
    }

    if items.is_empty() {
        println!("NO");
        return;
    }

    let mut sum_pos = 0i128;
    let mut sum_neg = 0i128;
    let mut gcd = 0i128;
    let mut has_pos = false;
    let mut has_neg = false;

    for item in &items {
        if item.value > 0 {
            has_pos = true;
            sum_pos += item.value;
        } else {
            has_neg = true;
            sum_neg += item.value;
        }
        gcd = if gcd == 0 { item.value.unsigned_abs() as i128 } else { gcd_i128(gcd, item.value.unsigned_abs() as i128) };
    }

    if target_in < sum_neg || target_in > sum_pos {
        println!("NO");
        return;
    }

    if gcd > 1 && target_in % gcd != 0 {
        println!("NO");
        return;
    }

    let mut target = target_in;
    if gcd > 1 {
        for item in &mut items {
            item.value /= gcd;
        }
        target /= gcd;
    }

    if let Some(item) = items.iter().find(|x| x.value == target) {
        println!("YES");
        println!("{}", item.index);
        return;
    }

    let n2 = items.len();
    if n2 <= 46 {
        if let Some(ans) = solve_mitm(&items, target) {
            println!("YES");
            if ans.is_empty() {
                println!();
            } else {
                let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
                println!("{line}");
            }
            return;
        }
        println!("NO");
        return;
    }

    if !has_neg {
        let limit_ok = target >= 0 && target <= 4_000_000i128;
        if limit_ok {
            if let Some(ans) = solve_positive_only(&items, target) {
                println!("YES");
                if ans.is_empty() {
                    println!();
                } else {
                    let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
                    println!("{line}");
                }
                return;
            }
        }
        if let Some(ans) = solve_dfs(&items, target) {
            println!("YES");
            if ans.is_empty() {
                println!();
            } else {
                let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
                println!("{line}");
            }
            return;
        }
        println!("NO");
        return;
    }

    if !has_pos {
        if target <= 0 && -target <= 4_000_000i128 {
            if let Some(ans) = solve_negative_only(&items, target) {
                println!("YES");
                if ans.is_empty() {
                    println!();
                } else {
                    let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
                    println!("{line}");
                }
                return;
            }
        }
        if let Some(ans) = solve_dfs(&items, target) {
            println!("YES");
            if ans.is_empty() {
                println!();
            } else {
                let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
                println!("{line}");
            }
            return;
        }
        println!("NO");
        return;
    }

    let range = sum_pos - sum_neg;
    let can_bitset = if range <= 4_000_000i128 {
        if let Ok(range_usize) = usize::try_from(range) {
            let words = (range_usize + 64) / 64;
            items
                .len()
                .checked_mul(words)
                .map(|prod| prod <= 80_000_000)
                .unwrap_or(false)
        } else {
            false
        }
    } else {
        false
    };
    if can_bitset {
        if let Some(ans) = solve_bitset(&items, target) {
            println!("YES");
            if ans.is_empty() {
                println!();
            } else {
                let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
                println!("{line}");
            }
            return;
        }
    }

    if let Some(ans) = solve_dfs(&items, target) {
        println!("YES");
        if ans.is_empty() {
            println!();
        } else {
            let line = ans.iter().map(|x| x.to_string()).collect::<Vec<_>>().join(" ");
            println!("{line}");
        }
        return;
    }

    println!("NO");
}
