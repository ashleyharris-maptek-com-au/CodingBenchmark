use std::io::{self, BufRead, Write};

#[derive(Clone, Copy, PartialEq, Eq)]
enum Phase {
    Climb,
    Transit,
    Approach,
    Land,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn parse_pair<T: std::str::FromStr>(line: &str) -> (T, T)
where
    <T as std::str::FromStr>::Err: std::fmt::Debug,
{
    let mut it = line.split_whitespace();
    let a = it.next().unwrap().parse::<T>().unwrap();
    let b = it.next().unwrap().parse::<T>().unwrap();
    (a, b)
}

fn read_nonempty_line(input: &mut impl BufRead, buf: &mut String) -> io::Result<usize> {
    loop {
        buf.clear();
        let n = input.read_line(buf)?;
        if n == 0 || !buf.trim().is_empty() {
            return Ok(n);
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut input = stdin.lock();
    let mut line = String::new();

    read_nonempty_line(&mut input, &mut line).unwrap();
    let mut it = line.split_whitespace();
    let width: usize = it.next().unwrap().parse().unwrap();
    let height: usize = it.next().unwrap().parse().unwrap();
    let gravity: f64 = it.next().unwrap().parse().unwrap();
    let max_thrust: f64 = it.next().unwrap().parse().unwrap();
    let _fuel: f64 = it.next().unwrap().parse().unwrap();

    read_nonempty_line(&mut input, &mut line).unwrap();
    let (start_x, start_y): (f64, f64) = parse_pair(&line);
    let mut it2 = line.split_whitespace();
    let _ = it2.next();
    let _ = it2.next();
    let target_x: f64 = it2.next().unwrap().parse().unwrap();
    let target_y: f64 = it2.next().unwrap().parse().unwrap();

    let mut ground = vec![-1_i32; width];
    let mut pad_left: i32 = i32::MAX;
    let mut pad_right: i32 = i32::MIN;
    let mut pad_y: i32 = -1;

    for y in 0..height {
        read_nonempty_line(&mut input, &mut line).unwrap();
        let row = line.trim_end_matches(&['\r', '\n'][..]).as_bytes();
        let limit = width.min(row.len());
        for x in 0..limit {
            match row[x] {
                b'.' => {}
                b'L' | b'T' => {
                    ground[x] = y as i32;
                    pad_left = pad_left.min(x as i32);
                    pad_right = pad_right.max(x as i32);
                    if pad_y < 0 {
                        pad_y = y as i32;
                    }
                }
                _ => {
                    ground[x] = y as i32;
                }
            }
        }
    }

    if pad_left == i32::MAX {
        pad_left = target_x.floor() as i32;
        pad_right = target_x.floor() as i32;
        pad_y = target_y.floor() as i32;
    }

    let pad_center = (pad_left as f64 + pad_right as f64 + 1.0) * 0.5;
    let pad_yf = pad_y as f64;

    let sx = start_x.floor().max(0.0) as usize;
    let tx = target_x.floor().max(0.0) as usize;
    let lo = sx.min(tx);
    let hi = sx.max(tx).min(width.saturating_sub(1));
    let mut route_min_ground = i32::MAX;
    for x in lo..=hi {
        let g = ground[x];
        if g >= 0 {
            route_min_ground = route_min_ground.min(g);
        }
    }

    let cruise_y = if route_min_ground == i32::MAX {
        (start_y.min(target_y) - 50.0).max(5.0)
    } else {
        ((route_min_ground as f64) - 25.0).max(5.0)
    };

    let mut out = io::BufWriter::new(io::stdout().lock());
    let mut state = String::new();

    loop {
        state.clear();
        if input.read_line(&mut state).unwrap() == 0 {
            break;
        }
        if state.trim().is_empty() {
            continue;
        }

        let mut it = state.split_whitespace();
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        let vx: f64 = it.next().unwrap().parse().unwrap();
        let vy: f64 = it.next().unwrap().parse().unwrap();
        let angle: f64 = it.next().unwrap().parse().unwrap();
        let angular_vel: f64 = it.next().unwrap().parse().unwrap();
        let _fuel_remaining: f64 = it.next().unwrap().parse().unwrap();

        let dx = pad_center - x;
        let alt_to_pad = pad_yf - y;

        let phase = if y > cruise_y + 4.0 {
            Phase::Climb
        } else if alt_to_pad > 80.0 {
            Phase::Transit
        } else if alt_to_pad > 15.0 {
            Phase::Approach
        } else {
            Phase::Land
        };

        let (ax_des, vy_des, y_target, max_tilt) = match phase {
            Phase::Climb => {
                let vy_target = clamp((cruise_y - y) * 0.18, -6.0, 2.0);
                let ax = clamp(dx * 0.015 - vx * 0.45, -0.45, 0.45);
                (ax, vy_target, cruise_y, 0.28)
            }
            Phase::Transit => {
                let vy_target = clamp((cruise_y - y) * 0.12, -4.0, 4.0);
                let ax = clamp(dx * 0.030 - vx * 0.70, -1.4, 1.4);
                (ax, vy_target, cruise_y, 0.50)
            }
            Phase::Approach => {
                let landing_alt = (alt_to_pad - 3.0).max(0.0);
                let brake = (max_thrust - gravity).max(0.5);
                let safe_v = (2.0 * brake * landing_alt).sqrt();
                let vy_target = safe_v.min(2.0);
                let ax = clamp(dx * 0.040 - vx * 0.95, -1.1, 1.1);
                (ax, vy_target, (pad_yf - 10.0).max(5.0), 0.35)
            }
            Phase::Land => {
                let landing_alt = (alt_to_pad - 1.0).max(0.0);
                let brake = (max_thrust - gravity).max(0.5);
                let safe_v = (2.0 * brake * landing_alt).sqrt();
                let vy_target = safe_v.min(0.8);
                let ax = clamp(dx * 0.060 - vx * 1.10, -0.75, 0.75);
                (ax, vy_target, (pad_yf - 2.0).max(5.0), 0.22)
            }
        };

        let ay_des = clamp((vy_des - vy) * 1.35 + (y_target - y) * 0.035, -gravity, max_thrust);
        let required_vertical = (gravity - ay_des).clamp(0.0, max_thrust);

        let mut angle_des = clamp((ax_des / max_thrust).clamp(-1.0, 1.0).asin(), -max_tilt, max_tilt);
        let angle_cap = if required_vertical >= max_thrust {
            0.0
        } else {
            ((required_vertical / max_thrust).clamp(0.0, 1.0)).acos()
        };
        let allowed = angle_cap.min(max_tilt);
        angle_des = clamp(angle_des, -allowed, allowed);

        let cos_a = angle_des.cos().max(0.18);
        let thrust = if required_vertical <= 0.0 {
            0.0
        } else {
            clamp(required_vertical / (max_thrust * cos_a), 0.0, 1.0)
        };

        let turn = clamp((angle_des - angle) * 2.4 - angular_vel * 0.55, -1.0, 1.0);

        writeln!(out, "{:.2} {:.2}", thrust, turn).unwrap();
        out.flush().unwrap();
    }
}
