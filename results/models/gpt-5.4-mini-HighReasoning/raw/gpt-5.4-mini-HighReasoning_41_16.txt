; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func_void = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Function_v2float = OpTypePointer Function %v2float
%_ptr_Function_bool = OpTypePointer Function %bool
%_ptr_Function_int = OpTypePointer Function %int

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%float_20 = OpConstant %float 20.0
%true = OpConstantTrue %bool
%false = OpConstantFalse %bool

%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output

%main = OpFunction %void None %func_void
%entry = OpLabel
%c = OpVariable %_ptr_Function_v2float Function
%z = OpVariable %_ptr_Function_v2float Function
%iter = OpVariable %_ptr_Function_int Function
%active = OpVariable %_ptr_Function_bool Function

%uv_val = OpLoad %v2float %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%scaled_x = OpFMul %float %uv_x %float_3
%c_x = OpFSub %float %scaled_x %float_2
%scaled_y = OpFMul %float %uv_y %float_2
%c_y = OpFSub %float %scaled_y %float_1
%c_val = OpCompositeConstruct %v2float %c_x %c_y
OpStore %c %c_val
%z_init = OpCompositeConstruct %v2float %float_0 %float_0
OpStore %z %z_init
OpStore %iter %int_0
OpStore %active %true
OpBranch %loop_header

%loop_header = OpLabel
%iter_val = OpLoad %int %iter
%loop_cond = OpSLessThan %bool %iter_val %int_20
OpLoopMerge %loop_merge %loop_continue None
OpBranchConditional %loop_cond %loop_body %loop_merge

%loop_body = OpLabel
%active_val = OpLoad %bool %active
OpSelectionMerge %after_active None
OpBranchConditional %active_val %do_iter %after_active

%do_iter = OpLabel
%z_val = OpLoad %v2float %z
%z_x = OpCompositeExtract %float %z_val 0
%z_y = OpCompositeExtract %float %z_val 1
%z_x2 = OpFMul %float %z_x %z_x
%z_y2 = OpFMul %float %z_y %z_y
%z_xy = OpFMul %float %z_x %z_y
%two_xy = OpFMul %float %z_xy %float_2
%new_x_base = OpFSub %float %z_x2 %z_y2
%c_loaded = OpLoad %v2float %c
%c_comp_x = OpCompositeExtract %float %c_loaded 0
%c_comp_y = OpCompositeExtract %float %c_loaded 1
%new_x = OpFAdd %float %new_x_base %c_comp_x
%new_y_base = OpFAdd %float %two_xy %c_comp_y
%new_z = OpCompositeConstruct %v2float %new_x %new_y_base
OpStore %z %new_z
%iter_cur = OpLoad %int %iter
%iter_next = OpIAdd %int %iter_cur %int_1
OpStore %iter %iter_next
%new_x2 = OpFMul %float %new_x %new_x
%new_y2 = OpFMul %float %new_y_base %new_y_base
%r2 = OpFAdd %float %new_x2 %new_y2
%escaped = OpFOrdGreaterThan %bool %r2 %float_4
OpSelectionMerge %after_escape None
OpBranchConditional %escaped %set_inactive %after_escape

%set_inactive = OpLabel
OpStore %active %false
OpBranch %after_escape

%after_escape = OpLabel
OpBranch %after_active

%after_active = OpLabel
OpBranch %loop_continue

%loop_continue = OpLabel
OpBranch %loop_header

%loop_merge = OpLabel
%final_iter = OpLoad %int %iter
%final_t_int = OpConvertSToF %float %final_iter
%t = OpFDiv %float %final_t_int %float_20
%green = OpFMul %float %t %float_0_5
%blue = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %v4float %t %green %blue %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
