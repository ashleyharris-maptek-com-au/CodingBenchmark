using namespace std;

struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    array<char, BUFSIZE> buf{};
    size_t idx = 0, size = 0;

    inline char getChar() {
        if (idx >= size) {
            size = fread(buf.data(), 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        T sign = 1;
        T val = 0;
        c = getChar();
        if (!c) return false;
        while (c != '-' && (c < '0' || c > '9')) {
            c = getChar();
            if (!c) return false;
        }
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        for (; c >= '0' && c <= '9'; c = getChar()) {
            val = val * 10 + (c - '0');
        }
        out = val * sign;
        return true;
    }
};

struct Edge {
    int u, v;
};

struct BridgeResult {
    bool found = false;
    int edgeId = -1;
};

struct CutResult {
    long long value = numeric_limits<long long>::max();
    vector<int> side;
};

struct Frame {
    int v;
    int parentEdge;
    size_t it;
};

static BridgeResult findBridge(const vector<Edge>& edges, int n) {
    vector<vector<int>> adj(n);
    adj.reserve(n);
    for (int i = 0; i < (int)edges.size(); ++i) {
        int u = edges[i].u, v = edges[i].v;
        if (u == v) continue;
        adj[u].push_back(i);
        adj[v].push_back(i);
    }

    vector<int> disc(n, 0), low(n, 0), parent(n, -1), parentEdge(n, -1);
    int timer = 0;
    vector<Frame> st;
    st.reserve(n);

    for (int s = 0; s < n; ++s) {
        if (disc[s] != 0) continue;
        st.push_back({s, -1, 0});
        parent[s] = -1;
        parentEdge[s] = -1;
        while (!st.empty()) {
            Frame& f = st.back();
            int v = f.v;
            if (disc[v] == 0) {
                disc[v] = low[v] = ++timer;
            }
            if (f.it < adj[v].size()) {
                int eid = adj[v][f.it++];
                if (eid == f.parentEdge) continue;
                const Edge& e = edges[eid];
                int to = (e.u == v ? e.v : e.u);
                if (disc[to] == 0) {
                    parent[to] = v;
                    parentEdge[to] = eid;
                    st.push_back({to, eid, 0});
                } else {
                    low[v] = min(low[v], disc[to]);
                }
            } else {
                st.pop_back();
                int p = parent[v];
                if (p != -1) {
                    low[p] = min(low[p], low[v]);
                    if (low[v] > disc[p]) {
                        return {true, parentEdge[v]};
                    }
                }
            }
        }
    }

    return {};
}

static vector<int> buildSide(const vector<vector<int>>& members, const vector<int>& order) {
    size_t total = 0;
    for (size_t i = 0; i + 1 < order.size(); ++i) {
        total += members[order[i]].size();
    }
    vector<int> side;
    side.reserve(total);
    for (size_t i = 0; i + 1 < order.size(); ++i) {
        const vector<int>& part = members[order[i]];
        side.insert(side.end(), part.begin(), part.end());
    }
    return side;
}

static CutResult stoerWagnerMatrix(int n, const vector<Edge>& edges) {
    vector<vector<long long>> g(n, vector<long long>(n, 0));
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++g[e.u][e.v];
        ++g[e.v][e.u];
    }

    vector<vector<int>> members(n);
    for (int i = 0; i < n; ++i) members[i] = {i};
    vector<char> alive(n, 1), added(n, 0);
    vector<long long> weight(n, 0);
    int aliveCount = n;

    CutResult best;

    while (aliveCount > 1) {
        fill(weight.begin(), weight.end(), 0);
        fill(added.begin(), added.end(), 0);

        vector<int> order;
        order.reserve(aliveCount);

        int prev = -1;
        int last = -1;

        for (int step = 0; step < aliveCount; ++step) {
            int sel = -1;
            long long bestW = -1;
            for (int v = 0; v < n; ++v) {
                if (!alive[v] || added[v]) continue;
                if (sel == -1 || weight[v] > bestW) {
                    sel = v;
                    bestW = weight[v];
                }
            }

            if (step == aliveCount - 1) {
                last = sel;
                if (bestW < best.value) {
                    best.value = bestW;
                    best.side = buildSide(members, order);
                }
                break;
            }

            added[sel] = 1;
            order.push_back(sel);
            prev = sel;

            for (int u = 0; u < n; ++u) {
                if (!alive[u] || added[u] || u == sel) continue;
                weight[u] += g[sel][u];
            }
        }

        if (aliveCount == 1) break;
        int s = prev;
        int t = last;

        if (members[s].size() + g[s].size() < members[t].size() + g[t].size()) {
            swap(s, t);
        }

        for (int i = 0; i < n; ++i) {
            if (!alive[i] || i == s || i == t) continue;
            g[s][i] += g[t][i];
            g[i][s] = g[s][i];
        }
        for (int i = 0; i < n; ++i) {
            g[t][i] = 0;
            g[i][t] = 0;
        }

        members[s].reserve(members[s].size() + members[t].size());
        members[s].insert(members[s].end(), members[t].begin(), members[t].end());
        members[t].clear();
        alive[t] = 0;
        --aliveCount;
    }

    return best;
}

static CutResult stoerWagnerSparse(int n, const vector<Edge>& edges) {
    vector<int> deg(n, 0);
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++deg[e.u];
        ++deg[e.v];
    }

    vector<unordered_map<int, long long>> adj(n);
    for (int i = 0; i < n; ++i) {
        adj[i].reserve(static_cast<size_t>(deg[i]) * 2 + 1);
    }
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        ++adj[e.u][e.v];
        ++adj[e.v][e.u];
    }

    vector<vector<int>> members(n);
    for (int i = 0; i < n; ++i) members[i] = {i};

    vector<char> alive(n, 1), added(n, 0);
    vector<long long> weight(n, 0);
    int aliveCount = n;

    CutResult best;

    auto score = [&](int v) -> size_t {
        return adj[v].size() + members[v].size();
    };

    while (aliveCount > 1) {
        fill(weight.begin(), weight.end(), 0);
        fill(added.begin(), added.end(), 0);

        priority_queue<pair<long long, int>> pq;
        for (int v = 0; v < n; ++v) {
            if (alive[v]) pq.emplace(0, v);
        }

        vector<int> order;
        order.reserve(aliveCount);

        int prev = -1;
        int last = -1;

        for (int step = 0; step < aliveCount; ++step) {
            int sel = -1;
            while (!pq.empty()) {
                auto [w, v] = pq.top();
                if (!alive[v] || added[v] || w != weight[v]) {
                    pq.pop();
                    continue;
                }
                sel = v;
                pq.pop();
                break;
            }

            if (sel == -1) {
                for (int v = 0; v < n; ++v) {
                    if (alive[v] && !added[v]) {
                        sel = v;
                        break;
                    }
                }
            }

            if (step == aliveCount - 1) {
                last = sel;
                if (weight[sel] < best.value) {
                    best.value = weight[sel];
                    best.side = buildSide(members, order);
                }
                break;
            }

            added[sel] = 1;
            order.push_back(sel);
            prev = sel;

            for (const auto& kv : adj[sel]) {
                int u = kv.first;
                long long w = kv.second;
                if (!alive[u] || added[u]) continue;
                weight[u] += w;
                pq.emplace(weight[u], u);
            }
        }

        if (aliveCount == 1) break;
        int s = prev;
        int t = last;

        int keep = s;
        int kill = t;
        if (score(t) > score(s)) {
            keep = t;
            kill = s;
        }

        adj[keep].reserve(adj[keep].size() + adj[kill].size() + 1);
        vector<pair<int, long long>> neigh;
        neigh.reserve(adj[kill].size());
        for (const auto& kv : adj[kill]) neigh.push_back(kv);

        for (const auto& kv : neigh) {
            int u = kv.first;
            long long w = kv.second;
            if (u == keep || !alive[u]) continue;
            adj[keep][u] += w;
            adj[u].erase(kill);
            adj[u][keep] += w;
        }
        adj[keep].erase(kill);
        adj[kill].clear();

        members[keep].reserve(members[keep].size() + members[kill].size());
        members[keep].insert(members[keep].end(), members[kill].begin(), members[kill].end());
        members[kill].clear();

        alive[kill] = 0;
        --aliveCount;
    }

    return best;
}

int main() {
    FastScanner fs;
    int n, m;
    if (!fs.readInt(n)) return 0;
    fs.readInt(m);

    vector<Edge> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        fs.readInt(u);
        fs.readInt(v);
        edges.push_back({u, v});
    }

    if (n <= 1) {
        cout << 0 << '\n';
        return 0;
    }

    BridgeResult br = findBridge(edges, n);
    if (br.found) {
        const Edge& e = edges[br.edgeId];
        cout << 1 << '\n' << e.u << ' ' << e.v << '\n';
        return 0;
    }

    CutResult res;
    const int SMALL_MATRIX_LIMIT = 450;
    if (n <= SMALL_MATRIX_LIMIT) {
        res = stoerWagnerMatrix(n, edges);
    } else {
        res = stoerWagnerSparse(n, edges);
    }

    vector<char> inSide(n, 0);
    for (int v : res.side) {
        if (0 <= v && v < n) inSide[v] = 1;
    }

    vector<Edge> cutEdges;
    cutEdges.reserve(edges.size());
    for (const auto& e : edges) {
        if (e.u == e.v) continue;
        if (inSide[e.u] != inSide[e.v]) {
            cutEdges.push_back(e);
        }
    }

    cout << cutEdges.size() << '\n';
    for (const auto& e : cutEdges) {
        cout << e.u << ' ' << e.v << '\n';
    }
    return 0;
}
