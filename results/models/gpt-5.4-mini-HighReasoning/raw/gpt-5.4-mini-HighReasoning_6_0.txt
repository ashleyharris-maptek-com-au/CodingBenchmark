using System;
using System.Collections.Generic;
using System.Globalization;
using System.Numerics;

public class Program
{
    private const double Eps = 1e-12;

    public static void Main()
    {
        var input = Console.In.ReadToEnd();
        var fs = new FastScanner(input);

        int n = fs.NextInt();
        double mu = fs.NextDouble();

        var startR = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        var startV = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
        var startState = MakeState(startR, startV, mu);

        var stationStates = new OrbitState[n];
        for (int i = 0; i < n; i++)
        {
            var r = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            var v = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());
            stationStates[i] = MakeState(r, v, mu);
        }

        var allStates = new OrbitState[n + 1];
        allStates[0] = startState;
        for (int i = 0; i < n; i++) allStates[i + 1] = stationStates[i];

        double posScale = 0, velScale = 0, hScale = 0, eScale = 0;
        int pairs = 0;
        for (int i = 0; i < allStates.Length; i++)
        {
            for (int j = i + 1; j < allStates.Length; j++)
            {
                posScale += Vec3.Distance(allStates[i].R, allStates[j].R);
                velScale += Vec3.Distance(allStates[i].V, allStates[j].V);
                hScale += Vec3.Distance(allStates[i].H, allStates[j].H);
                eScale += Math.Abs(allStates[i].Energy - allStates[j].Energy);
                pairs++;
            }
        }

        if (pairs == 0) pairs = 1;
        posScale = NormalizeScale(posScale / pairs);
        velScale = NormalizeScale(velScale / pairs);
        hScale = NormalizeScale(hScale / pairs);
        eScale = NormalizeScale(eScale / pairs);

        var cost = new double[n][];
        var startCost = new double[n];
        for (int i = 0; i < n; i++)
        {
            cost[i] = new double[n];
        }

        for (int i = 0; i < n; i++)
        {
            startCost[i] = Distance(startState, stationStates[i], posScale, velScale, hScale, eScale);
            for (int j = i + 1; j < n; j++)
            {
                double d = Distance(stationStates[i], stationStates[j], posScale, velScale, hScale, eScale);
                cost[i][j] = d;
                cost[j][i] = d;
            }
        }

        int[] route = n <= 16
            ? SolveExact(startCost, cost)
            : SolveHeuristic(startCost, cost);

        Console.WriteLine(string.Join(" ", route));
    }

    private static OrbitState MakeState(in Vec3 r, in Vec3 v, double mu)
    {
        var h = Vec3.Cross(r, v);
        double rMag = r.Length();
        double vMag2 = Vec3.Dot(v, v);
        double energy = 0.5 * vMag2 - mu / Math.Max(rMag, 1e-9);
        return new OrbitState(r, v, h, energy);
    }

    private static double Distance(in OrbitState a, in OrbitState b, double posScale, double velScale, double hScale, double eScale)
    {
        double dr = Vec3.Distance(a.R, b.R) / posScale;
        double dv = Vec3.Distance(a.V, b.V) / velScale;
        double dh = Vec3.Distance(a.H, b.H) / hScale;
        double de = Math.Abs(a.Energy - b.Energy) / eScale;
        return Math.Sqrt(dr * dr + dv * dv + dh * dh + de * de);
    }

    private static double NormalizeScale(double value)
    {
        if (double.IsNaN(value) || double.IsInfinity(value) || value < 1e-9) return 1.0;
        return value;
    }

    private static int[] SolveExact(double[] startCost, double[][] cost)
    {
        int n = startCost.Length;
        int full = 1 << n;
        int stateCount = full * n;
        var dp = new double[stateCount];
        var parent = new int[stateCount];

        for (int i = 0; i < stateCount; i++)
        {
            dp[i] = double.PositiveInfinity;
            parent[i] = -1;
        }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            int idx = mask * n + i;
            dp[idx] = startCost[i];
        }

        for (int mask = 1; mask < full; mask++)
        {
            for (int j = 0; j < n; j++)
            {
                if ((mask & (1 << j)) == 0) continue;

                int idx = mask * n + j;
                double cur = dp[idx];
                if (double.IsInfinity(cur)) continue;

                int remain = (full - 1) ^ mask;
                while (remain != 0)
                {
                    int bit = remain & -remain;
                    int k = BitOperations.TrailingZeroCount(bit);
                    int nextMask = mask | bit;
                    int nextIdx = nextMask * n + k;
                    double nextCost = cur + cost[j][k];
                    if (nextCost < dp[nextIdx])
                    {
                        dp[nextIdx] = nextCost;
                        parent[nextIdx] = j;
                    }
                    remain ^= bit;
                }
            }
        }

        int fullMask = full - 1;
        int bestEnd = 0;
        double bestCost = double.PositiveInfinity;
        for (int j = 0; j < n; j++)
        {
            int idx = fullMask * n + j;
            if (dp[idx] < bestCost)
            {
                bestCost = dp[idx];
                bestEnd = j;
            }
        }

        var route = new int[n];
        int mask2 = fullMask;
        int curNode = bestEnd;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curNode;
            int idx = mask2 * n + curNode;
            int prev = parent[idx];
            mask2 ^= 1 << curNode;
            curNode = prev;
        }

        return route;
    }

    private static int[] SolveHeuristic(double[] startCost, double[][] cost)
    {
        int n = startCost.Length;
        double[] minOut = new double[n];
        for (int i = 0; i < n; i++)
        {
            double best = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                double d = cost[i][j];
                if (d < best) best = d;
            }
            minOut[i] = double.IsInfinity(best) ? 0.0 : best;
        }

        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => startCost[a].CompareTo(startCost[b]));

        var seeds = new List<int>();
        var seen = new bool[n];
        void AddSeed(int idx)
        {
            if (idx < 0 || idx >= n || seen[idx]) return;
            seen[idx] = true;
            seeds.Add(idx);
        }

        int q = Math.Min(4, n);
        for (int i = 0; i < q; i++) AddSeed(order[i]);
        AddSeed(order[n / 2]);
        AddSeed(order[Math.Max(0, n - 1)]);
        AddSeed(order[Math.Max(0, n - 2)]);
        AddSeed(order[Math.Max(0, n - 3)]);

        int[] bestRoute = null;
        double bestCost = double.PositiveInfinity;

        foreach (int seed in seeds)
        {
            int[] route = BuildGreedyRoute(seed, cost, minOut);
            ImproveTwoOpt(route, startCost, cost);
            double routeCost = RouteCost(route, startCost, cost);
            if (routeCost < bestCost)
            {
                bestCost = routeCost;
                bestRoute = route;
            }
        }

        if (bestRoute == null)
        {
            bestRoute = BuildGreedyRoute(order[0], cost, minOut);
        }

        ImproveTwoOpt(bestRoute, startCost, cost);
        return bestRoute;
    }

    private static int[] BuildGreedyRoute(int first, double[][] cost, double[] minOut)
    {
        int n = cost.Length;
        var route = new int[n];
        var used = new bool[n];
        route[0] = first;
        used[first] = true;

        int current = first;
        for (int pos = 1; pos < n; pos++)
        {
            int best = -1;
            double bestScore = double.PositiveInfinity;

            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double score = cost[current][j] + 0.15 * minOut[j];
                if (score < bestScore)
                {
                    bestScore = score;
                    best = j;
                }
            }

            route[pos] = best;
            used[best] = true;
            current = best;
        }

        return route;
    }

    private static void ImproveTwoOpt(int[] route, double[] startCost, double[][] cost)
    {
        int n = route.Length;
        if (n <= 2) return;

        bool improved = true;
        int pass = 0;
        while (improved && pass++ < 64)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                int prev = i == 0 ? -1 : route[i - 1];
                int a = route[i];
                double left = i == 0 ? startCost[a] : cost[prev][a];

                for (int j = i + 1; j < n; j++)
                {
                    int b = route[j];
                    int next = j == n - 1 ? -1 : route[j + 1];
                    double oldCost = left + (next == -1 ? 0.0 : cost[b][next]);
                    double newCost = (i == 0 ? startCost[b] : cost[prev][b]) + (next == -1 ? 0.0 : cost[a][next]);
                    if (newCost + Eps < oldCost)
                    {
                        Array.Reverse(route, i, j - i + 1);
                        improved = true;
                        goto NextPass;
                    }
                }
            }
        NextPass:
            ;
        }
    }

    private static double RouteCost(int[] route, double[] startCost, double[][] cost)
    {
        double total = startCost[route[0]];
        for (int i = 1; i < route.Length; i++) total += cost[route[i - 1]][route[i]];
        return total;
    }

    private sealed class FastScanner
    {
        private readonly string _data;
        private int _index;

        public FastScanner(string data)
        {
            _data = data ?? string.Empty;
        }

        public string Next()
        {
            while (_index < _data.Length && char.IsWhiteSpace(_data[_index])) _index++;
            int start = _index;
            while (_index < _data.Length && !char.IsWhiteSpace(_data[_index])) _index++;
            return _data.Substring(start, _index - start);
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);

        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    private readonly struct OrbitState
    {
        public readonly Vec3 R;
        public readonly Vec3 V;
        public readonly Vec3 H;
        public readonly double Energy;

        public OrbitState(Vec3 r, Vec3 v, Vec3 h, double energy)
        {
            R = r;
            V = v;
            H = h;
            Energy = energy;
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator -(in Vec3 a, in Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);

        public static Vec3 Cross(in Vec3 a, in Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X);
        }

        public static double Dot(in Vec3 a, in Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public static double Distance(in Vec3 a, in Vec3 b) => (a - b).Length();
    }
}
