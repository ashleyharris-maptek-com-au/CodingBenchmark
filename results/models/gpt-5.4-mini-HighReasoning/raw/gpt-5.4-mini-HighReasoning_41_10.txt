; SPIR-V
; Version: 1.0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%glsl450 = OpExtInstImport "GLSL.std.450"

%float_0 = OpConstant %float 0.0
%float_01 = OpConstant %float 0.1
%float_04 = OpConstant %float 0.4
%float_06 = OpConstant %float 0.6
%float_09 = OpConstant %float 0.9
%float_1 = OpConstant %float 1.0
%float_pi = OpConstant %float 3.14159
%float_16 = OpConstant %float 16.0
%int_3 = OpConstant %int 3

%red = OpConstantComposite %v3float %float_09 %float_01 %float_01
%blue = OpConstantComposite %v3float %float_01 %float_01 %float_09

%main = OpFunction %void None %funct
%label = OpLabel

%uv_val = OpLoad %v2float %uv
%uv_y = OpCompositeExtract %float %uv_val 1
%angle0 = OpFMul %float %uv_y %float_pi
%angle = OpFMul %float %angle0 %float_16
%s = OpExtInst %float %glsl450 Sin %angle
%is_red = OpFOrdGreaterThan %bool %s %float_0
%stripe = OpSelect %v3float %is_red %red %blue

%world_pos = OpLoad %v3float %worldPos
%normal_val = OpLoad %v3float %normal
%normal_n = OpExtInst %v3float %glsl450 Normalize %normal_val
%light_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light_pos4 = OpLoad %v4float %light_ptr
%light_pos = OpVectorShuffle %v3float %light_pos4 %light_pos4 0 1 2
%light_dir0 = OpFSub %v3float %light_pos %world_pos
%light_dir = OpExtInst %v3float %glsl450 Normalize %light_dir0
%diff = OpDot %float %normal_n %light_dir
%diff_clamped = OpExtInst %float %glsl450 FMax %diff %float_0
%shade_mul = OpFMul %float %float_06 %diff_clamped
%shade = OpFAdd %float %float_04 %shade_mul
%rgb = OpVectorTimesScalar %v3float %stripe %shade
%color = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
