const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const data = JSON.parse(input);
const W = data.width | 0;
const H = data.height | 0;
const N = W * H;
const grid = data.grid || [];

const burnable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  const base = y * W;
  for (let x = 0; x < W; x++) {
    burnable[base + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const townValueByCell = new Float64Array(N);
const isTownCell = new Uint8Array(N);
const towns = Array.isArray(data.towns) ? data.towns : [];
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  const idx = y * W + x;
  townValueByCell[idx] += Number(t.value) || 0;
  isTownCell[idx] = 1;
}

const isIgnition = new Uint8Array(N);
const ignitionList = [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
for (let i = 0; i < ignitions.length; i++) {
  const p = ignitions[i];
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  const idx = y * W + x;
  if (!burnable[idx] || isIgnition[idx]) continue;
  isIgnition[idx] = 1;
  ignitionList.push(idx);
}

const cutBudget = Math.max(0, data.cutBudget | 0);
if (cutBudget === 0 || ignitionList.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

// Multi-source BFS from ignitions to get the reachable fire region,
// distances for path scoring, and one shortest-path parent per cell.
const dist = new Int32Array(N);
dist.fill(-1);
const bfsParent = new Int32Array(N);
bfsParent.fill(-1);
const queue = new Int32Array(N);
let qh = 0;
let qt = 0;
for (let i = 0; i < ignitionList.length; i++) {
  const s = ignitionList[i];
  if (dist[s] !== -1) continue;
  dist[s] = 0;
  queue[qt++] = s;
}

while (qh < qt) {
  const v = queue[qh++];
  const nd = dist[v] + 1;
  const x = v % W;
  const y = (v / W) | 0;

  if (x > 0) {
    const nb = v - 1;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = nd;
      bfsParent[nb] = v;
      queue[qt++] = nb;
    }
  }
  if (x + 1 < W) {
    const nb = v + 1;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = nd;
      bfsParent[nb] = v;
      queue[qt++] = nb;
    }
  }
  if (y > 0) {
    const nb = v - W;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = nd;
      bfsParent[nb] = v;
      queue[qt++] = nb;
    }
  }
  if (y + 1 < H) {
    const nb = v + W;
    if (burnable[nb] && dist[nb] === -1) {
      dist[nb] = nd;
      bfsParent[nb] = v;
      queue[qt++] = nb;
    }
  }
}

const reachable = dist;
const degree = new Uint8Array(N);
for (let idx = 0; idx < N; idx++) {
  if (reachable[idx] === -1) continue;
  let d = 0;
  const x = idx % W;
  const y = (idx / W) | 0;
  if (x > 0 && reachable[idx - 1] !== -1) d++;
  if (x + 1 < W && reachable[idx + 1] !== -1) d++;
  if (y > 0 && reachable[idx - W] !== -1) d++;
  if (y + 1 < H && reachable[idx + W] !== -1) d++;
  degree[idx] = d;
}

// Iterative Tarjan DFS on the reachable graph, with a virtual root linked
// to all ignition cells.
const ROOT = N;
const totalNodes = N + 1;
const disc = new Int32Array(totalNodes);
const low = new Int32Array(totalNodes);
const dfsParent = new Int32Array(totalNodes);
dfsParent.fill(-1);
const iter = new Int32Array(totalNodes);
const tin = new Int32Array(totalNodes);
const order = new Int32Array(N);
let orderLen = 0;
let timer = 0;

const stack = new Int32Array(totalNodes);
let sp = 0;
stack[sp++] = ROOT;
disc[ROOT] = low[ROOT] = ++timer;
iter[ROOT] = 0;

while (sp > 0) {
  const v = stack[sp - 1];
  if (v === ROOT) {
    if (iter[ROOT] < ignitionList.length) {
      const to = ignitionList[iter[ROOT]++];
      if (disc[to] === 0) {
        dfsParent[to] = ROOT;
        disc[to] = low[to] = ++timer;
        iter[to] = 0;
        tin[to] = orderLen;
        order[orderLen++] = to;
        stack[sp++] = to;
      } else {
        if (disc[to] < low[ROOT]) low[ROOT] = disc[to];
      }
    } else {
      sp--;
    }
    continue;
  }

  if (iter[v] < 4) {
    const dir = iter[v]++;
    const x = v % W;
    const y = (v / W) | 0;
    let nb = -1;
    if (dir === 0) {
      if (x > 0) nb = v - 1;
    } else if (dir === 1) {
      if (x + 1 < W) nb = v + 1;
    } else if (dir === 2) {
      if (y > 0) nb = v - W;
    } else {
      if (y + 1 < H) nb = v + W;
    }

    if (nb === -1 || reachable[nb] === -1) continue;
    if (nb === dfsParent[v]) continue;
    if (disc[nb] === 0) {
      dfsParent[nb] = v;
      disc[nb] = low[nb] = ++timer;
      iter[nb] = 0;
      tin[nb] = orderLen;
      order[orderLen++] = nb;
      stack[sp++] = nb;
    } else if (disc[nb] < low[v]) {
      low[v] = disc[nb];
    }
  } else {
    sp--;
    const p = dfsParent[v];
    if (p !== -1) {
      if (low[v] < low[p]) low[p] = low[v];
    }
  }
}

// Subtree aggregation.
const subtreeValue = new Float64Array(totalNodes);
const subtreeSources = new Int32Array(totalNodes);
const subtreeSize = new Int32Array(totalNodes);
for (let i = 0; i < orderLen; i++) {
  const v = order[i];
  subtreeValue[v] = townValueByCell[v];
  subtreeSources[v] = isIgnition[v] ? 1 : 0;
  subtreeSize[v] = 1;
}
for (let i = orderLen - 1; i >= 0; i--) {
  const v = order[i];
  const p = dfsParent[v];
  if (p !== -1) {
    subtreeValue[p] += subtreeValue[v];
    subtreeSources[p] += subtreeSources[v];
    subtreeSize[p] += subtreeSize[v];
  }
}

// For each potential cut cell, accumulate exact saved value from source-free
// child subtrees that become disconnected when the cell is removed.
const gain = new Float64Array(N);
const intervalHead = new Int32Array(N);
intervalHead.fill(-1);
const intervalL = [];
const intervalR = [];
const intervalNext = [];

for (let i = 0; i < orderLen; i++) {
  const v = order[i];
  const p = dfsParent[v];
  if (p <= ROOT || p === -1) continue;
  if (low[v] >= disc[p] && subtreeSources[v] === 0) {
    gain[p] += subtreeValue[v];
    const k = intervalL.length;
    intervalL.push(tin[v]);
    intervalR.push(tin[v] + subtreeSize[v]);
    intervalNext.push(intervalHead[p]);
    intervalHead[p] = k;
  }
}

// Heuristic scores from exact chokepoints, town frontiers, and shortest-path
// corridors. Higher score = better candidate cut.
const score = new Float64Array(N);
for (let i = 0; i < N; i++) {
  if (reachable[i] === -1 || !burnable[i] || isIgnition[i] || isTownCell[i]) continue;
  if (gain[i] > 0) score[i] += gain[i];
}

for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0;
  const y = t.y | 0;
  if (x < 0 || x >= W || y < 0 || y >= H) continue;
  const idx = y * W + x;
  const value = Number(t.value) || 0;
  if (value <= 0 || reachable[idx] === -1) continue;

  const frontier = [];
  if (x > 0) {
    const nb = idx - 1;
    if (burnable[nb] && reachable[nb] !== -1 && !isIgnition[nb] && !isTownCell[nb]) frontier.push(nb);
  }
  if (x + 1 < W) {
    const nb = idx + 1;
    if (burnable[nb] && reachable[nb] !== -1 && !isIgnition[nb] && !isTownCell[nb]) frontier.push(nb);
  }
  if (y > 0) {
    const nb = idx - W;
    if (burnable[nb] && reachable[nb] !== -1 && !isIgnition[nb] && !isTownCell[nb]) frontier.push(nb);
  }
  if (y + 1 < H) {
    const nb = idx + W;
    if (burnable[nb] && reachable[nb] !== -1 && !isIgnition[nb] && !isTownCell[nb]) frontier.push(nb);
  }

  if (frontier.length > 0) {
    const add = value / frontier.length;
    for (let j = 0; j < frontier.length; j++) {
      score[frontier[j]] += add;
    }
  }

  let v = bfsParent[idx];
  while (v !== -1) {
    if (!isIgnition[v] && !isTownCell[v]) {
      const d = degree[v] + 1;
      score[v] += value / ((dist[v] + 1) * d);
    }
    v = bfsParent[v];
  }
}

// Candidate list.
const candidates = [];
for (let i = 0; i < N; i++) {
  if (reachable[i] === -1 || !burnable[i] || isIgnition[i] || isTownCell[i]) continue;
  if (score[i] > 0) candidates.push(i);
}

if (candidates.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

candidates.sort((a, b) => score[b] - score[a]);

// Fenwick tree for range blocking of protected source-free subtrees.
class Fenwick {
  constructor(n) {
    this.n = n;
    this.bit = new Int32Array(n + 2);
  }
  add(i, delta) {
    for (let x = i + 1; x <= this.n + 1; x += x & -x) this.bit[x] += delta;
  }
  rangeAdd(l, r, delta) {
    if (l >= r) return;
    this.add(l, delta);
    this.add(r, -delta);
  }
  pointQuery(i) {
    let s = 0;
    for (let x = i + 1; x > 0; x -= x & -x) s += this.bit[x];
    return s;
  }
}

const bit = new Fenwick(orderLen + 2);
const chosen = [];
const chosenSet = new Uint8Array(N);

for (let i = 0; i < candidates.length && chosen.length < cutBudget; i++) {
  const v = candidates[i];
  if (chosenSet[v]) continue;
  if (bit.pointQuery(tin[v]) > 0) continue;

  chosenSet[v] = 1;
  chosen.push([v % W, (v / W) | 0]);

  // Block all source-free child subtrees that this articulation safely detaches.
  for (let e = intervalHead[v]; e !== -1; e = intervalNext[e]) {
    bit.rangeAdd(intervalL[e], intervalR[e], 1);
  }
}

process.stdout.write(JSON.stringify({ firebreaks: chosen }));
