use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::f64::consts::{PI, SQRT_2};
use std::io::Read;

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug, Default)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    #[inline]
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
}

use std::ops::{Add, Div, Mul, Sub};

impl Add for Point {
    type Output = Self;
    fn add(self, rhs: Self) -> Self::Output {
        Self::new(self.x + rhs.x, self.y + rhs.y)
    }
}

impl Sub for Point {
    type Output = Self;
    fn sub(self, rhs: Self) -> Self::Output {
        Self::new(self.x - rhs.x, self.y - rhs.y)
    }
}

impl Mul<f64> for Point {
    type Output = Self;
    fn mul(self, rhs: f64) -> Self::Output {
        Self::new(self.x * rhs, self.y * rhs)
    }
}

impl Div<f64> for Point {
    type Output = Self;
    fn div(self, rhs: f64) -> Self::Output {
        Self::new(self.x / rhs, self.y / rhs)
    }
}

#[derive(Clone, Copy, Debug)]
struct Circle {
    c: Point,
    r: f64,
}

impl Circle {
    #[inline]
    fn contains(&self, p: Point) -> bool {
        dist2(self.c, p) <= (self.r + 1e-9) * (self.r + 1e-9)
    }
}

#[inline]
fn dist2(a: Point, b: Point) -> f64 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    dx * dx + dy * dy
}

#[inline]
fn dist(a: Point, b: Point) -> f64 {
    dist2(a, b).sqrt()
}

#[inline]
fn cross(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

#[inline]
fn dot(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.x - a.x) + (b.y - a.y) * (c.y - a.y)
}

#[inline]
fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    cross(a, b, p).abs() <= 1e-10 && dot(p, a, b) <= 1e-10
}

#[inline]
fn point_segment_distance(p: Point, a: Point, b: Point) -> f64 {
    let ab = b - a;
    let ap = p - a;
    let len2 = ab.x * ab.x + ab.y * ab.y;
    if len2 <= EPS {
        return dist(p, a);
    }
    let t = (ap.x * ab.x + ap.y * ab.y) / len2;
    let t = t.clamp(0.0, 1.0);
    let proj = Point::new(a.x + ab.x * t, a.y + ab.y * t);
    dist(p, proj)
}

#[inline]
fn polygon_bbox(poly: &[Point]) -> (f64, f64, f64, f64) {
    let mut minx = poly[0].x;
    let mut maxx = poly[0].x;
    let mut miny = poly[0].y;
    let mut maxy = poly[0].y;
    for &p in poly.iter().skip(1) {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }
    (minx, maxx, miny, maxy)
}

fn polygon_area2(poly: &[Point]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        s += a.x * b.y - a.y * b.x;
    }
    s
}

fn polygon_centroid(poly: &[Point]) -> Point {
    let n = poly.len();
    let area2 = polygon_area2(poly);
    if area2.abs() <= EPS {
        let mut sx = 0.0;
        let mut sy = 0.0;
        for &p in poly {
            sx += p.x;
            sy += p.y;
        }
        return Point::new(sx / n as f64, sy / n as f64);
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let k = a.x * b.y - b.x * a.y;
        cx += (a.x + b.x) * k;
        cy += (a.y + b.y) * k;
    }
    let inv = 1.0 / (3.0 * area2);
    Point::new(cx * inv, cy * inv)
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let x = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x > p.x + EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn signed_distance_to_polygon(p: Point, poly: &[Point]) -> f64 {
    let n = poly.len();
    let mut d = f64::INFINITY;
    for i in 0..n {
        d = d.min(point_segment_distance(p, poly[i], poly[(i + 1) % n]));
    }
    if point_in_polygon(p, poly) {
        d
    } else {
        -d
    }
}

#[derive(Clone, Copy, Debug)]
struct Cell {
    x: f64,
    y: f64,
    h: f64,
    d: f64,
    max: f64,
}

impl Cell {
    fn new(x: f64, y: f64, h: f64, poly: &[Point]) -> Self {
        let d = signed_distance_to_polygon(Point::new(x, y), poly);
        let max = d + h * SQRT_2;
        Self { x, y, h, d, max }
    }
}

impl PartialEq for Cell {
    fn eq(&self, other: &Self) -> bool {
        self.max.to_bits() == other.max.to_bits()
    }
}
impl Eq for Cell {}

impl PartialOrd for Cell {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        self.max.partial_cmp(&other.max)
    }
}

impl Ord for Cell {
    fn cmp(&self, other: &Self) -> Ordering {
        self.partial_cmp(other).unwrap_or(Ordering::Equal)
    }
}

fn polylabel(poly: &[Point], precision: f64) -> (Point, f64) {
    let (minx, maxx, miny, maxy) = polygon_bbox(poly);
    let width = maxx - minx;
    let height = maxy - miny;
    if width.abs() <= EPS && height.abs() <= EPS {
        return (poly[0], 0.0);
    }

    let cell_size = width.min(height);
    let mut heap = BinaryHeap::new();

    let mut x = minx;
    while x <= maxx {
        let mut y = miny;
        while y <= maxy {
            heap.push(Cell::new(x + cell_size * 0.5, y + cell_size * 0.5, cell_size * 0.5, poly));
            y += cell_size;
        }
        x += cell_size;
    }

    let centroid = polygon_centroid(poly);
    let mut best = Cell::new(centroid.x, centroid.y, 0.0, poly);
    if best.d < 0.0 {
        best = Cell::new((minx + maxx) * 0.5, (miny + maxy) * 0.5, 0.0, poly);
    }

    while let Some(cell) = heap.pop() {
        if cell.d > best.d {
            best = cell;
        }
        if cell.max - best.d <= precision {
            continue;
        }
        let h = cell.h * 0.5;
        heap.push(Cell::new(cell.x - h, cell.y - h, h, poly));
        heap.push(Cell::new(cell.x + h, cell.y - h, h, poly));
        heap.push(Cell::new(cell.x - h, cell.y + h, h, poly));
        heap.push(Cell::new(cell.x + h, cell.y + h, h, poly));
    }

    (Point::new(best.x, best.y), best.d.max(0.0))
}

fn shuffle(points: &mut [Point]) {
    let mut seed = 0x9E37_79B9_7F4A_7C15u64 ^ (points.len() as u64);
    for i in (1..points.len()).rev() {
        seed ^= seed << 7;
        seed ^= seed >> 9;
        seed ^= seed << 8;
        let j = (seed % ((i + 1) as u64)) as usize;
        points.swap(i, j);
    }
}

fn circle_from2(a: Point, b: Point) -> Circle {
    let c = Point::new((a.x + b.x) * 0.5, (a.y + b.y) * 0.5);
    Circle { c, r: dist(a, b) * 0.5 }
}

fn circle_from3(a: Point, b: Point, c: Point) -> Circle {
    let d = 2.0 * (a.x * (b.y - c.y) + b.x * (c.y - a.y) + c.x * (a.y - b.y));
    if d.abs() <= 1e-12 {
        let ab = dist2(a, b);
        let ac = dist2(a, c);
        let bc = dist2(b, c);
        if ab >= ac && ab >= bc {
            return circle_from2(a, b);
        }
        if ac >= ab && ac >= bc {
            return circle_from2(a, c);
        }
        return circle_from2(b, c);
    }

    let a2 = a.x * a.x + a.y * a.y;
    let b2 = b.x * b.x + b.y * b.y;
    let c2 = c.x * c.x + c.y * c.y;
    let ux = (a2 * (b.y - c.y) + b2 * (c.y - a.y) + c2 * (a.y - b.y)) / d;
    let uy = (a2 * (c.x - b.x) + b2 * (a.x - c.x) + c2 * (b.x - a.x)) / d;
    let center = Point::new(ux, uy);
    Circle { c: center, r: dist(center, a) }
}

fn min_enclosing_circle(points: &[Point]) -> Circle {
    if points.is_empty() {
        return Circle { c: Point::new(0.0, 0.0), r: 0.0 };
    }
    if points.len() == 1 {
        return Circle { c: points[0], r: 0.0 };
    }
    let mut pts = points.to_vec();
    shuffle(&mut pts);

    let mut c = Circle { c: pts[0], r: 0.0 };
    for i in 1..pts.len() {
        if !c.contains(pts[i]) {
            c = Circle { c: pts[i], r: 0.0 };
            for j in 0..i {
                if !c.contains(pts[j]) {
                    c = circle_from2(pts[i], pts[j]);
                    for k in 0..j {
                        if !c.contains(pts[k]) {
                            c = circle_from3(pts[i], pts[j], pts[k]);
                        }
                    }
                }
            }
        }
    }
    c
}

fn rotate_about(p: Point, anchor: Point, angle: f64) -> Point {
    let s = angle.sin();
    let c = angle.cos();
    let dx = p.x - anchor.x;
    let dy = p.y - anchor.y;
    Point::new(anchor.x + dx * c - dy * s, anchor.y + dx * s + dy * c)
}

#[inline]
fn orient(a: Point, b: Point, c: Point) -> f64 {
    cross(a, b, c)
}

#[inline]
fn proper_segment_intersection(a: Point, b: Point, c: Point, d: Point) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    let s1 = o1 > EPS;
    let s2 = o2 > EPS;
    let s3 = o3 > EPS;
    let s4 = o4 > EPS;
    (s1 != s2) && (s3 != s4)
}

fn fits_in_stock(
    piece: &[Point],
    stock: &[Point],
    stock_edges: &[(Point, Point)],
    translation: Point,
    angle: f64,
    scratch: &mut Vec<Point>,
) -> bool {
    scratch.clear();
    let s = angle.sin();
    let c = angle.cos();
    for &p in piece {
        let q = Point::new(p.x * c - p.y * s + translation.x, p.x * s + p.y * c + translation.y);
        if !point_in_polygon(q, stock) {
            return false;
        }
        scratch.push(q);
    }

    let n = scratch.len();
    for i in 0..n {
        let a = scratch[i];
        let b = scratch[(i + 1) % n];
        for &(c1, c2) in stock_edges {
            if proper_segment_intersection(a, b, c1, c2) {
                return false;
            }
        }
    }
    true
}

#[derive(Clone, Copy)]
struct Candidate {
    p: Point,
    clearance: f64,
}

fn stock_candidates(stock: &[Point]) -> Vec<Candidate> {
    let (minx, maxx, miny, maxy) = polygon_bbox(stock);
    let (best_p, best_d) = polylabel(stock, 1e-6);
    let mut candidates = Vec::new();
    candidates.push(Candidate {
        p: best_p,
        clearance: best_d,
    });

    let steps = 64usize;
    let dx = if (maxx - minx).abs() <= EPS {
        0.0
    } else {
        (maxx - minx) / steps as f64
    };
    let dy = if (maxy - miny).abs() <= EPS {
        0.0
    } else {
        (maxy - miny) / steps as f64
    };
    for ix in 0..=steps {
        let x = minx + dx * ix as f64;
        for iy in 0..=steps {
            let y = miny + dy * iy as f64;
            let p = Point::new(x, y);
            let d = signed_distance_to_polygon(p, stock);
            if d > 0.0 {
                candidates.push(Candidate { p, clearance: d });
            }
        }
    }

    candidates.sort_by(|a, b| {
        b.clearance
            .partial_cmp(&a.clearance)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.p.x.partial_cmp(&b.p.x).unwrap_or(Ordering::Equal))
            .then_with(|| a.p.y.partial_cmp(&b.p.y).unwrap_or(Ordering::Equal))
    });
    let mut dedup = Vec::new();
    for c in candidates {
        if dedup.iter().all(|d: &Candidate| {
            (d.p.x - c.p.x).abs() > 1e-9 || (d.p.y - c.p.y).abs() > 1e-9
        }) {
            dedup.push(c);
        }
        if dedup.len() >= 256 {
            break;
        }
    }
    dedup
}

fn parse_input() -> (Vec<Point>, Vec<Vec<Point>>) {
    let mut input = String::new();
    std::io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Point::new(x, y));
    }

    let mut pieces = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut piece = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            piece.push(Point::new(x, y));
        }
        pieces.push(piece);
    }
    (stock, pieces)
}

fn main() {
    let (stock, pieces) = parse_input();
    let num_stocks = pieces.len();
    let candidates = stock_candidates(&stock);
    let stock_edges: Vec<(Point, Point)> = (0..stock.len())
        .map(|i| (stock[i], stock[(i + 1) % stock.len()]))
        .collect();

    let fallback_angles = [
        0.0,
        PI / 6.0,
        PI / 3.0,
        PI / 2.0,
        2.0 * PI / 3.0,
        5.0 * PI / 6.0,
        PI,
        7.0 * PI / 6.0,
        4.0 * PI / 3.0,
        3.0 * PI / 2.0,
        5.0 * PI / 3.0,
        11.0 * PI / 6.0,
    ];

    let mut outputs: Vec<(usize, f64, f64, f64)> = Vec::with_capacity(num_stocks);
    let mut scratch = Vec::new();

    for (i, piece) in pieces.iter().enumerate() {
        let mec = min_enclosing_circle(piece);
        let piece_anchor = polygon_centroid(piece);

        let mut placed = false;

        for cand in &candidates {
            if cand.clearance + 1e-8 >= mec.r {
                let angle = 0.0;
                let rot_anchor = mec.c;
                let rot_c = rotate_about(rot_anchor, Point::new(0.0, 0.0), angle);
                let translation = Point::new(cand.p.x - rot_c.x, cand.p.y - rot_c.y);
                outputs.push((i, translation.x, translation.y, angle));
                placed = true;
                break;
            }
        }
        if placed {
            continue;
        }

        'search: for cand in &candidates {
            for &angle in &fallback_angles {
                let translation = Point::new(
                    cand.p.x - (piece_anchor.x * angle.cos() - piece_anchor.y * angle.sin()),
                    cand.p.y - (piece_anchor.x * angle.sin() + piece_anchor.y * angle.cos()),
                );
                if fits_in_stock(
                    piece,
                    &stock,
                    &stock_edges,
                    translation,
                    angle,
                    &mut scratch,
                ) {
                    outputs.push((i, translation.x, translation.y, angle));
                    placed = true;
                    break 'search;
                }
            }
        }

        if !placed {
            let cand = candidates.first().copied().unwrap_or(Candidate {
                p: Point::new(0.0, 0.0),
                clearance: 0.0,
            });
            let angle = 0.0;
            let translation = Point::new(cand.p.x - piece_anchor.x, cand.p.y - piece_anchor.y);
            outputs.push((i, translation.x, translation.y, angle));
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", num_stocks));
    for (idx, x, y, rot) in outputs {
        out.push_str(&format!("{} {:.12} {:.12} {:.12}\n", idx, x, y, rot));
    }
    print!("{}", out);
}
