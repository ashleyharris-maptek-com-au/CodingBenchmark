using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

    static List<int>[] adj;      // adjacency list (sorted)
    static HashSet<int>[] adjSet; // adjacency as hash sets for fast membership (large graphs)
    static int[] degree;

    static List<int> bestClique = new List<int>();
    static int bestSize = 0;
    static object bestLock = new object();

    // Time management
    static Stopwatch sw = new Stopwatch();
    // Hard time limit (ms). Use 290000 (~4.8 min) to be safe under 5 min.
    const int HardTimeLimitMs = 290000;

    static volatile bool timeExceeded = false;

    static void Main()
    {
        sw.Start();
        ReadInput();

        if (n == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        double avgDeg = (n == 0) ? 0.0 : (2.0 * m) / n;

        // Heuristic threshold: above this vertex count, we do only heuristic.
        const int ExactVertexLimit = 2000;

        if (n <= ExactVertexLimit)
        {
            RunExactOrHybrid(avgDeg);
        }
        else
        {
            // Large n: heuristic only
            RunHeuristicOnly();
        }

        // Output best clique found
        bestClique.Sort();
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static void ReadInput()
    {
        var input = Console.In;
        string? first = input.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            n = 0; m = 0;
            adj = new List<int>[0];
            adjSet = new HashSet<int>[0];
            degree = new int[0];
            return;
        }

        var parts = first.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(parts[0]);
        m = int.Parse(parts[1]);

        adj = new List<int>[n];
        adjSet = new HashSet<int>[n];
        degree = new int[n];

        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            string? line = input.ReadLine();
            if (line == null)
                break;
            var sp = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            if (sp.Length < 2) { i--; continue; }
            int u = int.Parse(sp[0]);
            int v = int.Parse(sp[1]);
            if (u == v) continue; // ignore self-loop
            adj[u].Add(v);
            adj[v].Add(u);
        }

        for (int i = 0; i < n; i++)
        {
            adj[i].Sort();
            degree[i] = adj[i].Count;
        }

        // For large graphs, build HashSet adjacency for quick membership
        if (n > 2000)
        {
            for (int i = 0; i < n; i++)
            {
                adjSet[i] = new HashSet<int>(adj[i]);
            }
        }
        else
        {
            // For smaller graphs, do not allocate heavy hashsets
            for (int i = 0; i < n; i++)
            {
                adjSet[i] = null!; // not used
            }
        }
    }

    static void RunExactOrHybrid(double avgDeg)
    {
        // For small graphs, we try an exact Bron-Kerbosch with bounds.
        // But if time is near limit we stop and then run heuristic seeded by best.

        // Initial heuristic to get a starting lower bound
        HeuristicGreedySequential(limitMs: 5000); // 5s quick heuristic

        // Decide complexity: if avgDeg is very high, exact search may still be expensive.
        // But we attempt anyway until a soft budget, say half the time limit.

        int softLimitMs = HardTimeLimitMs / 2;

        // Prepare candidate set: all vertices sorted by degree descending
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();

        List<int> R = new List<int>();
        List<int> P = new List<int>(order);
        List<int> X = new List<int>();

        // Exact search
        BronKerboschWithPivot_ColorBound(R, P, X, softLimitMs);

        // If still time, run some multi-start heuristic to potentially improve
        if (!timeExceeded && sw.ElapsedMilliseconds < HardTimeLimitMs)
        {
            int remaining = HardTimeLimitMs - (int)sw.ElapsedMilliseconds;
            if (remaining > 2000)
            {
                HeuristicMultiStartParallel(remaining - 1000); // leave some slack
            }
        }
    }

    static void RunHeuristicOnly()
    {
        // For large graphs we run parallel multi-start heuristics directly.
        // Possibly start with a quick greedy.
        HeuristicGreedySequential(limitMs: 2000);
        if (sw.ElapsedMilliseconds < HardTimeLimitMs)
        {
            int remaining = HardTimeLimitMs - (int)sw.ElapsedMilliseconds;
            HeuristicMultiStartParallel(remaining - 500);
        }
    }

    // ==================== Exact algorithm: Bron-Kerbosch with pivoting and coloring bound ===================

    static void BronKerboschWithPivot_ColorBound(List<int> R, List<int> P, List<int> X, int softLimitMs)
    {
        if (timeExceeded) return;
        long elapsed = sw.ElapsedMilliseconds;
        if (elapsed >= HardTimeLimitMs)
        {
            timeExceeded = true;
            return;
        }
        if (elapsed >= softLimitMs)
        {
            // Soft limit reached for exact search
            return;
        }

        if (P.Count == 0 && X.Count == 0)
        {
            // Maximal clique
            int size = R.Count;
            if (size > bestSize)
            {
                lock (bestLock)
                {
                    if (size > bestSize)
                    {
                        bestSize = size;
                        bestClique = new List<int>(R);
                    }
                }
            }
            return;
        }

        // Upper bound via greedy coloring on P
        int ubound = R.Count + ColorBound(P);
        if (ubound <= bestSize) return;

        // Choose pivot: vertex in P ∪ X with max degree into P
        int pivot = -1;
        int maxDeg = -1;
        // Reuse small arrays to avoid allocations? For simplicity, just compute.
        // Consider union of P and X
        foreach (var u in P)
        {
            int d = CountNeighborsInSet(u, P);
            if (d > maxDeg)
            {
                maxDeg = d;
                pivot = u;
            }
        }
        foreach (var u in X)
        {
            int d = CountNeighborsInSet(u, P);
            if (d > maxDeg)
            {
                maxDeg = d;
                pivot = u;
            }
        }

        // Candidates to branch on: P \ N(pivot)
        List<int> pivotNeighbors = (pivot == -1) ? new List<int>() : IntersectListWithNeighbors(P, pivot);

        // Instead iterate over a snapshot, because P will be modified
        List<int> branchVertices = new List<int>();
        if (pivot == -1)
        {
            branchVertices.AddRange(P);
        }
        else
        {
            // P \ pivotNeighbors
            var isNeighbor = new HashSet<int>(pivotNeighbors);
            foreach (var v in P)
            {
                if (!isNeighbor.Contains(v)) branchVertices.Add(v);
            }
        }

        foreach (var v in branchVertices)
        {
            if (timeExceeded) return;

            // R' = R ∪ {v}
            R.Add(v);

            // P' = P ∩ N(v)
            List<int> P2 = IntersectListWithNeighbors(P, v);
            // X' = X ∩ N(v)
            List<int> X2 = IntersectListWithNeighbors(X, v);

            if (R.Count + P2.Count > bestSize) // trivial bound
            {
                BronKerboschWithPivot_ColorBound(R, P2, X2, softLimitMs);
            }

            R.RemoveAt(R.Count - 1);

            // Move v from P to X
            P.Remove(v);
            X.Add(v);

            if (sw.ElapsedMilliseconds >= softLimitMs) return;
        }
    }

    static int ColorBound(List<int> vertices)
    {
        // Simple greedy sequential coloring for upper bound: order by descending degree inside the set.
        // For small n this is fine.
        int len = vertices.Count;
        if (len == 0) return 0;

        // Order vertices by degree in original graph (cheap heuristic)
        int[] vs = new int[len];
        for (int i = 0; i < len; i++) vs[i] = vertices[i];
        Array.Sort(vs, (a, b) => degree[b].CompareTo(degree[a]));

        int[] colorOf = new int[len]; // index -> color
        for (int i = 0; i < len; i++) colorOf[i] = -1;

        int nColors = 0;
        // Map vertex -> index in vs array
        // For performance we build a dictionary
        Dictionary<int, int> indexOf = new Dictionary<int, int>(len);
        for (int i = 0; i < len; i++) indexOf[vs[i]] = i;

        for (int i = 0; i < len; i++)
        {
            int v = vs[i];
            bool[] forbidden = new bool[nColors];
            // Check neighbors within the set already colored
            foreach (int u in adj[v])
            {
                int idx;
                if (indexOf.TryGetValue(u, out idx))
                {
                    if (idx < i)
                    {
                        int c = colorOf[idx];
                        if (c >= 0 && c < nColors) forbidden[c] = true;
                    }
                }
            }
            int color;
            for (color = 0; color < nColors; color++)
            {
                if (!forbidden[color]) break;
            }
            if (color == nColors) nColors++;
            colorOf[i] = color;
        }
        return nColors;
    }

    static int CountNeighborsInSet(int v, List<int> set)
    {
        // set is small; intersection with adjacency list
        int count = 0;
        var neigh = adj[v];
        // Both lists sorted
        int i = 0, j = 0;
        int n1 = neigh.Count, n2 = set.Count;
        while (i < n1 && j < n2)
        {
            int a = neigh[i];
            int b = set[j];
            if (a == b) { count++; i++; j++; }
            else if (a < b) i++; else j++;
        }
        return count;
    }

    static List<int> IntersectListWithNeighbors(List<int> list, int v)
    {
        // list and adj[v] are sorted; perform intersection
        List<int> res = new List<int>();
        var neigh = adj[v];
        int i = 0, j = 0;
        int n1 = list.Count, n2 = neigh.Count;
        while (i < n1 && j < n2)
        {
            int a = list[i];
            int b = neigh[j];
            if (a == b) { res.Add(a); i++; j++; }
            else if (a < b) i++; else j++;
        }
        return res;
    }

    static void HeuristicGreedySequential(int limitMs)
    {
        long start = sw.ElapsedMilliseconds;
        if (timeExceeded) return;

        // Simple greedy: sort vertices by degree descending, and for each try a greedy clique expansion.
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();

        foreach (int startV in order)
        {
            if (sw.ElapsedMilliseconds - start > limitMs) break;

            List<int> clique = BuildGreedyCliqueFrom(startV);
            if (clique.Count > bestSize)
            {
                lock (bestLock)
                {
                    if (clique.Count > bestSize)
                    {
                        bestSize = clique.Count;
                        bestClique = clique;
                    }
                }
            }
        }
    }

    static List<int> BuildGreedyCliqueFrom(int startV)
    {
        List<int> clique = new List<int> { startV };

        // Candidate vertices that are neighbors of all in the clique so far
        // Start with neighbors of startV
        List<int> candidates = new List<int>(adj[startV]);
        // Sort by degree descending (heuristic)
        candidates.Sort((a, b) => degree[b].CompareTo(degree[a]));

        foreach (int v in candidates)
        {
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }

        return clique;
    }

    static bool AreNeighbors(int u, int v)
    {
        if (n > 2000)
        {
            // Use hashset
            return adjSet[u].Contains(v);
        }
        else
        {
            // Binary search in sorted list
            var list = adj[u];
            int lo = 0, hi = list.Count - 1;
            while (lo <= hi)
            {
                int mid = (lo + hi) >> 1;
                int x = list[mid];
                if (x == v) return true;
                if (x < v) lo = mid + 1; else hi = mid - 1;
            }
            return false;
        }
    }

    static void HeuristicMultiStartParallel(int limitMs)
    {
        if (limitMs <= 0) return;
        long start = sw.ElapsedMilliseconds;
        int remaining = (int)(HardTimeLimitMs - start);
        if (remaining <= 0) return;
        limitMs = Math.Min(limitMs, remaining);

        int maxTasks = Environment.ProcessorCount;

        // We will pick seeds from high-degree vertices
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
        int seedsToUse = Math.Min(order.Length, maxTasks * 8); // some multiple of cores

        CancellationTokenSource cts = new CancellationTokenSource();
        CancellationToken token = cts.Token;

        List<Task> tasks = new List<Task>();

        for (int t = 0; t < maxTasks; t++)
        {
            int taskId = t;
            Task task = Task.Run(() =>
            {
                Random rnd = new Random(unchecked(Environment.TickCount * 31 + taskId * 997));
                int idx = taskId;
                while (!token.IsCancellationRequested)
                {
                    long now = sw.ElapsedMilliseconds;
                    if (now - start >= limitMs) break;

                    if (idx >= seedsToUse) idx = rnd.Next(seedsToUse);
                    int seed = order[idx++];

                    List<int> clique = LocalSearchClique(seed, rnd, (int)(limitMs - (now - start)), token);

                    if (clique.Count > bestSize)
                    {
                        lock (bestLock)
                        {
                            if (clique.Count > bestSize)
                            {
                                bestSize = clique.Count;
                                bestClique = clique;
                            }
                        }
                    }
                }
            }, token);
            tasks.Add(task);
        }

        // Wait up to limitMs
        Task.WaitAll(tasks.ToArray(), limitMs);
        cts.Cancel();
    }

    static List<int> LocalSearchClique(int seed, Random rnd, int localLimitMs, CancellationToken token)
    {
        long start = sw.ElapsedMilliseconds;

        // Start from a greedy clique
        List<int> clique = BuildGreedyCliqueFrom(seed);

        // Make a candidate pool: all vertices that connect to the clique
        HashSet<int> inClique = new HashSet<int>(clique);
        List<int> candidates = new List<int>();
        foreach (int v in Enumerable.Range(0, n))
        {
            if (inClique.Contains(v)) continue;
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);
        }

        // Basic local improvement: random permutations of candidates; try to add
        for (;;)
        {
            if (token.IsCancellationRequested) break;
            if (sw.ElapsedMilliseconds - start >= localLimitMs) break;

            if (candidates.Count == 0) break;

            int idx = rnd.Next(candidates.Count);
            int v = candidates[idx];
            candidates.RemoveAt(idx);

            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok)
            {
                clique.Add(v);
                inClique.Add(v);

                // Filter candidates to those adjacent to new v
                candidates = candidates.Where(x => AreNeighbors(x, v)).ToList();
            }
        }

        return clique;
    }

    static bool AreNeighbors(int u, int v)
    {

    static void HeuristicMultiStartParallel(int limitMs)
    {
        if (limitMs <= 0) return;
        long start = sw.ElapsedMilliseconds;
        int remaining = (int)(HardTimeLimitMs - start);
        if (remaining <= 0) return;
        limitMs = Math.Min(limitMs, remaining);

        int maxTasks = Environment.ProcessorCount;

        // We will pick seeds from high-degree vertices
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
        int seedsToUse = Math.Min(order.Length, maxTasks * 8); // some multiple of cores

        CancellationTokenSource cts = new CancellationTokenSource();
        CancellationToken token = cts.Token;

        List<Task> tasks = new List<Task>();

        for (int t = 0; t < maxTasks; t++)
        {
            int taskId = t;
            Task task = Task.Run(() =>
            {
                Random rnd = new Random(unchecked(Environment.TickCount * 31 + taskId * 997));
                int idx = taskId;
                while (!token.IsCancellationRequested)
                {
                    long now = sw.ElapsedMilliseconds;
                    if (now - start >= limitMs) break;

                    if (idx >= seedsToUse) idx = rnd.Next(seedsToUse);
                    int seed = order[idx++];

                    List<int> clique = LocalSearchClique(seed, rnd, (int)(limitMs - (now - start)), token);

                    if (clique.Count > bestSize)
                    {
                        lock (bestLock)
                        {
                            if (clique.Count > bestSize)
                            {
                                bestSize = clique.Count;
                                bestClique = clique;
                            }
                        }
                    }
                }
            }, token);
            tasks.Add(task);
        }

        // Wait up to limitMs
        Task.WaitAll(tasks.ToArray(), limitMs);
        cts.Cancel();
    }

    static List<int> LocalSearchClique(int seed, Random rnd, int localLimitMs, CancellationToken token)
    {
        long start = sw.ElapsedMilliseconds;

        // Start from a greedy clique
        List<int> clique = BuildGreedyCliqueFrom(seed);

        // Make a candidate pool: all vertices that connect to the clique
        HashSet<int> inClique = new HashSet<int>(clique);
        List<int> candidates = new List<int>();
        foreach (int v in Enumerable.Range(0, n))
        {
            if (inClique.Contains(v)) continue;
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);
        }

        // Basic local improvement: random permutations of candidates; try to add
        for (;;)
        {
            if (token.IsCancellationRequested) break;
            if (sw.ElapsedMilliseconds - start >= localLimitMs) break;

            if (candidates.Count == 0) break;

            int idx = rnd.Next(candidates.Count);
            int v = candidates[idx];
            candidates.RemoveAt(idx);

            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok)
            {
                clique.Add(v);
                inClique.Add(v);

                // Filter candidates to those adjacent to new v
                candidates = candidates.Where(x => AreNeighbors(x, v)).ToList();
            }
        }

        return clique;
    }

    static void BronKerboschWithPivot_ColorBound(List<int> R, List<int> P, List<int> X, int softLimitMs)
    {

        long elapsed = sw.ElapsedMilliseconds;
        if (elapsed >= HardTimeLimitMs)

        for (int i = 0; i < len; i++) colorOf[i] = -1;

        // Map vertex -> index in vs array

        Dictionary<int, int> indexOf = new Dictionary<int, int>(len);

            foreach (int u in adj[v])

                    {

                    }
                }

            for (color = 0; color < nColors; color++)
            {
                if (!forbidden[color]) break;
            }
            if (color == nColors) nColors++;
            colorOf[i] = color;
        }
        return nColors;

    static void HeuristicMultiStartParallel(int limitMs)
    {
        if (limitMs <= 0) return;

        if (remaining <= 0) return;

        int seedsToUse = Math.Min(order.Length, maxTasks * 8); // some multiple of cores

        {

            {
                Random rnd = new Random(unchecked(Environment.TickCount * 31 + taskId * 997));
                int idx = taskId;
                while (!token.IsCancellationRequested)
                {
                    long now = sw.ElapsedMilliseconds;
                    if (now - start >= limitMs) break;

                    if (idx >= seedsToUse) idx = rnd.Next(seedsToUse);
                    int seed = order[idx++];

                    List<int> clique = LocalSearchClique(seed, rnd, (int)(limitMs - (now - start)), token);

                    if (clique.Count > bestSize)
                    {
                        lock (bestLock)
                        {
                            if (clique.Count > bestSize)
                            {
                                bestSize = clique.Count;
                                bestClique = clique;
                            }
                        }
                    }
                }
            }, token);
            tasks.Add(task);
        }

        // Wait up to limitMs
        Task.WaitAll(tasks.ToArray(), limitMs);
        cts.Cancel();
    }

    static List<int> LocalSearchClique(int seed, Random rnd, int localLimitMs, CancellationToken token)
    {
        long start = sw.ElapsedMilliseconds;

        // Start from a greedy clique
        List<int> clique = BuildGreedyCliqueFrom(seed);

        // Make a candidate pool: all vertices that connect to the clique
        HashSet<int> inClique = new HashSet<int>(clique);
        List<int> candidates = new List<int>();
        foreach (int v in Enumerable.Range(0, n))
        {
            if (inClique.Contains(v)) continue;
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);
        }

        // Basic local improvement: random permutations of candidates; try to add
        for (;;)
        {
            if (token.IsCancellationRequested) break;
            if (sw.ElapsedMilliseconds - start >= localLimitMs) break;

            if (candidates.Count == 0) break;

            int idx = rnd.Next(candidates.Count);
            int v = candidates[idx];
            candidates.RemoveAt(idx);

            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok)
            {
                clique.Add(v);
                inClique.Add(v);

                // Filter candidates to those adjacent to new v
                candidates = candidates.Where(x => AreNeighbors(x, v)).ToList();
            }
        }

        return clique;
    }

    static int ColorBound(List<int> vertices)

        // Order vertices by degree in original graph (cheap heuristic)
        int[] vs = new int[len];
        for (int i = 0; i < len; i++) vs[i] = vertices[i];
        Array.Sort(vs, (a, b) => degree[b].CompareTo(degree[a]));

        int[] colorOf = new int[len]; // index -> color
        for (int i = 0; i < len; i++) colorOf[i] = -1;

        int nColors = 0;
        // Map vertex -> index in vs array
        // For performance we build a dictionary
        Dictionary<int, int> indexOf = new Dictionary<int, int>(len);
        for (int i = 0; i < len; i++) indexOf[vs[i]] = i;

        for (int i = 0; i < len; i++)
        {
            int v = vs[i];
            bool[] forbidden = new bool[nColors];
            // Check neighbors within the set already colored
            foreach (int u in adj[v])
            {
                int idx;
                if (indexOf.TryGetValue(u, out idx))
                {
                    if (idx < i)
                    {
                        int c = colorOf[idx];
                        if (c >= 0 && c < nColors) forbidden[c] = true;
                    }
                }
            }
            int color;
            for (color = 0; color < nColors; color++)
            {
                if (!forbidden[color]) break;
            }
            if (color == nColors) nColors++;
            colorOf[i] = color;
        }
        return nColors;
    }

    static int CountNeighborsInSet(int v, List<int> set)
    {
        // set is small; intersection with adjacency list
        int count = 0;
        var neigh = adj[v];

        int i = 0, j = 0;
        int n1 = neigh.Count, n2 = set.Count;
        while (i < n1 && j < n2)
        {
            int a = neigh[i];
            int b = set[j];
            if (a == b) { count++; i++; j++; }
            else if (a < b) i++; else j++;
        }
        return count;
    }

    static List<int> IntersectListWithNeighbors(List<int> list, int v)
    {
        // list and adj[v] are sorted; perform intersection
        List<int> res = new List<int>();
        var neigh = adj[v];

            int b = neigh[j];
            if (a == b) { res.Add(a); i++; j++; }
            else if (a < b) i++; else j++;
        }
        return res;
    }

                if (x < v) lo = mid + 1; else hi = mid - 1;

        }

        HashSet<int> inClique = new HashSet<int>(clique);
        List<int> candidates = new List<int>();

            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);

        for (;;)
        {
            if (token.IsCancellationRequested) break;
            if (sw.ElapsedMilliseconds - start >= localLimitMs) break;

            if (candidates.Count == 0) break;

            int idx = rnd.Next(candidates.Count);
            int v = candidates[idx];
            candidates.RemoveAt(idx);

            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok)
            {
                clique.Add(v);
                inClique.Add(v);

                // Filter candidates to those adjacent to new v
                candidates = candidates.Where(x => AreNeighbors(x, v)).ToList();
            }
        }

        return clique;
    }

            for (color = 0; color < nColors; color++)
            {
                if (!forbidden[color]) break;
            }
            if (color == nColors) nColors++;
            colorOf[i] = color;
        }
        return nColors;

            int b = set[j];
            if (a == b) { count++; i++; j++; }

        }

    }

    {

    static List<int> BuildGreedyCliqueFrom(int startV)
    {
        List<int> clique = new List<int> { startV };

        // Candidate vertices that are neighbors of all in the clique so far
        // Start with neighbors of startV
        List<int> candidates = new List<int>(adj[startV]);
        // Sort by degree descending (heuristic)
        candidates.Sort((a, b) => degree[b].CompareTo(degree[a]));

        {

            foreach (int u in clique)

            if (ok) clique.Add(v);

    static bool AreNeighbors(int u, int v)
    {
        if (n > 2000)
        {
            // Use hashset
            return adjSet[u].Contains(v);
        }
        else
        {
            // Binary search in sorted list

            int lo = 0, hi = list.Count - 1;
            while (lo <= hi)
            {
                int mid = (lo + hi) >> 1;
                int x = list[mid];
                if (x == v) return true;
                if (x < v) lo = mid + 1; else hi = mid - 1;
            }
            return false;
        }
    }

    {
        if (limitMs <= 0) return;

        // We will pick seeds from high-degree vertices

        int seedsToUse = Math.Min(order.Length, maxTasks * 8); // some multiple of cores

        CancellationToken token = cts.Token;

        List<Task> tasks = new List<Task>();

        {
            int taskId = t;

            {

                int idx = taskId;
                while (!token.IsCancellationRequested)
                {

                        {
                            if (clique.Count > bestSize)
                            {
                                bestSize = clique.Count;
                                bestClique = clique;
                            }
                        }
                    }
                }
            }, token);
            tasks.Add(task);
        }

        // Wait up to limitMs
        Task.WaitAll(tasks.ToArray(), limitMs);
        cts.Cancel();
    }

    static List<int> LocalSearchClique(int seed, Random rnd, int localLimitMs, CancellationToken token)

        List<int> clique = BuildGreedyCliqueFrom(seed);

        // Make a candidate pool: all vertices that connect to the clique
        HashSet<int> inClique = new HashSet<int>(clique);
        List<int> candidates = new List<int>();

        {
            if (inClique.Contains(v)) continue;
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);
        }

        for (;;)

        }

        return clique;
    }

using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static int n, m;
    static List<int>[] adj;      // adjacency list (sorted)
    static HashSet<int>[] adjSet; // adjacency as hash sets for fast membership (large graphs)
    static int[] degree;

    static List<int> bestClique = new List<int>();
    static int bestSize = 0;
    static object bestLock = new object();

    // Time management
    static Stopwatch sw = new Stopwatch();
    // Hard time limit (ms). Use 290000 (~4.8 min) to be safe under 5 min.
    const int HardTimeLimitMs = 290000;

    static volatile bool timeExceeded = false;

    static void Main()
    {
        sw.Start();
        ReadInput();

        if (n == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        double avgDeg = (n == 0) ? 0.0 : (2.0 * m) / n;

        // Heuristic threshold: above this vertex count, we do only heuristic.
        const int ExactVertexLimit = 2000;

        if (n <= ExactVertexLimit)
        {
            RunExactOrHybrid(avgDeg);
        }
        else
        {
            // Large n: heuristic only
            RunHeuristicOnly();
        }

        // Output best clique found
        bestClique.Sort();
        Console.WriteLine(bestClique.Count);
        if (bestClique.Count > 0)
        {
            Console.WriteLine(string.Join(" ", bestClique));
        }
        else
        {
            Console.WriteLine();
        }
    }

    static void ReadInput()
    {
        var input = Console.In;
        string? first = input.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            n = 0; m = 0;
            adj = new List<int>[0];
            adjSet = new HashSet<int>[0];
            degree = new int[0];
            return;
        }

        var parts = first.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        n = int.Parse(parts[0]);
        m = int.Parse(parts[1]);

        adj = new List<int>[n];
        adjSet = new HashSet<int>[n];
        degree = new int[n];

        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            string? line = input.ReadLine();
            if (line == null)
                break;
            var sp = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            if (sp.Length < 2) { i--; continue; }
            int u = int.Parse(sp[0]);
            int v = int.Parse(sp[1]);
            if (u == v) continue; // ignore self-loop
            adj[u].Add(v);
            adj[v].Add(u);
        }

        for (int i = 0; i < n; i++)
        {
            adj[i].Sort();
            degree[i] = adj[i].Count;
        }

        // For large graphs, build HashSet adjacency for quick membership
        if (n > 2000)
        {
            for (int i = 0; i < n; i++)
            {
                adjSet[i] = new HashSet<int>(adj[i]);
            }
        }
        else
        {
            // For smaller graphs, do not allocate heavy hashsets
            for (int i = 0; i < n; i++)
            {
                adjSet[i] = null!; // not used
            }
        }
    }

    static void RunExactOrHybrid(double avgDeg)
    {
        // For small graphs, we try an exact Bron-Kerbosch with bounds.
        // But if time is near limit we stop and then run heuristic seeded by best.

        // Initial heuristic to get a starting lower bound
        HeuristicGreedySequential(limitMs: 5000); // 5s quick heuristic

        // Decide complexity: if avgDeg is very high, exact search may still be expensive.
        // But we attempt anyway until a soft budget, say half the time limit.

        int softLimitMs = HardTimeLimitMs / 2;

        // Prepare candidate set: all vertices sorted by degree descending
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();

        List<int> R = new List<int>();
        List<int> P = new List<int>(order);
        List<int> X = new List<int>();

        // Exact search
        BronKerboschWithPivot_ColorBound(R, P, X, softLimitMs);

        // If still time, run some multi-start heuristic to potentially improve
        if (!timeExceeded && sw.ElapsedMilliseconds < HardTimeLimitMs)
        {
            int remaining = HardTimeLimitMs - (int)sw.ElapsedMilliseconds;
            if (remaining > 2000)
            {
                HeuristicMultiStartParallel(remaining - 1000); // leave some slack
            }
        }
    }

    static void RunHeuristicOnly()
    {
        // For large graphs we run parallel multi-start heuristics directly.
        // Possibly start with a quick greedy.
        HeuristicGreedySequential(limitMs: 2000);
        if (sw.ElapsedMilliseconds < HardTimeLimitMs)
        {
            int remaining = HardTimeLimitMs - (int)sw.ElapsedMilliseconds;
            HeuristicMultiStartParallel(remaining - 500);
        }
    }

    // ==================== Exact algorithm: Bron-Kerbosch with pivoting and coloring bound ===================

    static void BronKerboschWithPivot_ColorBound(List<int> R, List<int> P, List<int> X, int softLimitMs)
    {
        if (timeExceeded) return;
        long elapsed = sw.ElapsedMilliseconds;
        if (elapsed >= HardTimeLimitMs)
        {
            timeExceeded = true;
            return;
        }
        if (elapsed >= softLimitMs)
        {
            // Soft limit reached for exact search
            return;
        }

        if (P.Count == 0 && X.Count == 0)
        {
            // Maximal clique
            int size = R.Count;
            if (size > bestSize)
            {
                lock (bestLock)
                {
                    if (size > bestSize)
                    {
                        bestSize = size;
                        bestClique = new List<int>(R);
                    }
                }
            }
            return;
        }

        // Upper bound via greedy coloring on P
        int ubound = R.Count + ColorBound(P);
        if (ubound <= bestSize) return;

        // Choose pivot: vertex in P ∪ X with max degree into P
        int pivot = -1;
        int maxDeg = -1;
        // Reuse small arrays to avoid allocations? For simplicity, just compute.
        // Consider union of P and X
        foreach (var u in P)
        {
            int d = CountNeighborsInSet(u, P);
            if (d > maxDeg)
            {
                maxDeg = d;
                pivot = u;
            }
        }
        foreach (var u in X)
        {
            int d = CountNeighborsInSet(u, P);
            if (d > maxDeg)
            {
                maxDeg = d;
                pivot = u;
            }
        }

        // Candidates to branch on: P \ N(pivot)
        List<int> pivotNeighbors = (pivot == -1) ? new List<int>() : IntersectListWithNeighbors(P, pivot);

        // Instead iterate over a snapshot, because P will be modified
        List<int> branchVertices = new List<int>();
        if (pivot == -1)
        {
            branchVertices.AddRange(P);
        }
        else
        {
            // P \ pivotNeighbors
            var isNeighbor = new HashSet<int>(pivotNeighbors);
            foreach (var v in P)
            {
                if (!isNeighbor.Contains(v)) branchVertices.Add(v);
            }
        }

        foreach (var v in branchVertices)
        {
            if (timeExceeded) return;

            // R' = R ∪ {v}
            R.Add(v);

            // P' = P ∩ N(v)
            List<int> P2 = IntersectListWithNeighbors(P, v);
            // X' = X ∩ N(v)
            List<int> X2 = IntersectListWithNeighbors(X, v);

            if (R.Count + P2.Count > bestSize) // trivial bound
            {
                BronKerboschWithPivot_ColorBound(R, P2, X2, softLimitMs);
            }

            R.RemoveAt(R.Count - 1);

            // Move v from P to X
            P.Remove(v);
            X.Add(v);

            if (sw.ElapsedMilliseconds >= softLimitMs) return;
        }
    }

    static int ColorBound(List<int> vertices)
    {
        // Simple greedy sequential coloring for upper bound: order by descending degree inside the set.
        // For small n this is fine.
        int len = vertices.Count;
        if (len == 0) return 0;

        // Order vertices by degree in original graph (cheap heuristic)
        int[] vs = new int[len];
        for (int i = 0; i < len; i++) vs[i] = vertices[i];
        Array.Sort(vs, (a, b) => degree[b].CompareTo(degree[a]));

        int[] colorOf = new int[len]; // index -> color
        for (int i = 0; i < len; i++) colorOf[i] = -1;

        int nColors = 0;
        // Map vertex -> index in vs array
        // For performance we build a dictionary
        Dictionary<int, int> indexOf = new Dictionary<int, int>(len);
        for (int i = 0; i < len; i++) indexOf[vs[i]] = i;

        for (int i = 0; i < len; i++)
        {
            int v = vs[i];
            bool[] forbidden = new bool[nColors];
            // Check neighbors within the set already colored
            foreach (int u in adj[v])
            {
                int idx;
                if (indexOf.TryGetValue(u, out idx))
                {
                    if (idx < i)
                    {
                        int c = colorOf[idx];
                        if (c >= 0 && c < nColors) forbidden[c] = true;
                    }
                }
            }
            int color;
            for (color = 0; color < nColors; color++)
            {
                if (!forbidden[color]) break;
            }
            if (color == nColors) nColors++;
            colorOf[i] = color;
        }
        return nColors;
    }

    static int CountNeighborsInSet(int v, List<int> set)
    {
        // set is small; intersection with adjacency list
        int count = 0;
        var neigh = adj[v];
        // Both lists sorted
        int i = 0, j = 0;
        int n1 = neigh.Count, n2 = set.Count;
        while (i < n1 && j < n2)
        {
            int a = neigh[i];
            int b = set[j];
            if (a == b) { count++; i++; j++; }
            else if (a < b) i++; else j++;
        }
        return count;
    }

    static List<int> IntersectListWithNeighbors(List<int> list, int v)
    {
        // list and adj[v] are sorted; perform intersection
        List<int> res = new List<int>();
        var neigh = adj[v];
        int i = 0, j = 0;
        int n1 = list.Count, n2 = neigh.Count;
        while (i < n1 && j < n2)
        {
            int a = list[i];
            int b = neigh[j];
            if (a == b) { res.Add(a); i++; j++; }
            else if (a < b) i++; else j++;
        }
        return res;
    }

    // ==================== Heuristic algorithms ===================

    static void HeuristicGreedySequential(int limitMs)
    {
        long start = sw.ElapsedMilliseconds;
        if (timeExceeded) return;

        // Simple greedy: sort vertices by degree descending, and for each try a greedy clique expansion.
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();

        foreach (int startV in order)
        {
            if (sw.ElapsedMilliseconds - start > limitMs) break;

            List<int> clique = BuildGreedyCliqueFrom(startV);
            if (clique.Count > bestSize)
            {
                lock (bestLock)
                {
                    if (clique.Count > bestSize)
                    {
                        bestSize = clique.Count;
                        bestClique = clique;
                    }
                }
            }
        }
    }

    static List<int> BuildGreedyCliqueFrom(int startV)
    {
        List<int> clique = new List<int> { startV };

        // Candidate vertices that are neighbors of all in the clique so far
        // Start with neighbors of startV
        List<int> candidates = new List<int>(adj[startV]);
        // Sort by degree descending (heuristic)
        candidates.Sort((a, b) => degree[b].CompareTo(degree[a]));

        foreach (int v in candidates)
        {
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) clique.Add(v);
        }

        return clique;
    }

    static bool AreNeighbors(int u, int v)
    {
        if (n > 2000)
        {
            // Use hashset
            return adjSet[u].Contains(v);
        }
        else
        {
            // Binary search in sorted list
            var list = adj[u];
            int lo = 0, hi = list.Count - 1;
            while (lo <= hi)
            {
                int mid = (lo + hi) >> 1;
                int x = list[mid];
                if (x == v) return true;
                if (x < v) lo = mid + 1; else hi = mid - 1;
            }
            return false;
        }
    }

    static void HeuristicMultiStartParallel(int limitMs)
    {
        if (limitMs <= 0) return;
        long start = sw.ElapsedMilliseconds;
        int remaining = (int)(HardTimeLimitMs - start);
        if (remaining <= 0) return;
        limitMs = Math.Min(limitMs, remaining);

        int maxTasks = Environment.ProcessorCount;

        // We will pick seeds from high-degree vertices
        int[] order = Enumerable.Range(0, n).OrderByDescending(v => degree[v]).ToArray();
        int seedsToUse = Math.Min(order.Length, maxTasks * 8); // some multiple of cores

        CancellationTokenSource cts = new CancellationTokenSource();
        CancellationToken token = cts.Token;

        List<Task> tasks = new List<Task>();

        for (int t = 0; t < maxTasks; t++)
        {
            int taskId = t;
            Task task = Task.Run(() =>
            {
                Random rnd = new Random(unchecked(Environment.TickCount * 31 + taskId * 997));
                int idx = taskId;
                while (!token.IsCancellationRequested)
                {
                    long now = sw.ElapsedMilliseconds;
                    if (now - start >= limitMs) break;

                    if (idx >= seedsToUse) idx = rnd.Next(seedsToUse);
                    int seed = order[idx++];

                    List<int> clique = LocalSearchClique(seed, rnd, (int)(limitMs - (now - start)), token);

                    if (clique.Count > bestSize)
                    {
                        lock (bestLock)
                        {
                            if (clique.Count > bestSize)
                            {
                                bestSize = clique.Count;
                                bestClique = clique;
                            }
                        }
                    }
                }
            }, token);
            tasks.Add(task);
        }

        // Wait up to limitMs
        Task.WaitAll(tasks.ToArray(), limitMs);
        cts.Cancel();
    }

    static List<int> LocalSearchClique(int seed, Random rnd, int localLimitMs, CancellationToken token)
    {
        long start = sw.ElapsedMilliseconds;

        // Start from a greedy clique
        List<int> clique = BuildGreedyCliqueFrom(seed);

        // Make a candidate pool: all vertices that connect to the clique
        HashSet<int> inClique = new HashSet<int>(clique);
        List<int> candidates = new List<int>();
        foreach (int v in Enumerable.Range(0, n))
        {
            if (inClique.Contains(v)) continue;
            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok) candidates.Add(v);
        }

        // Basic local improvement: random permutations of candidates; try to add
        for (;;)
        {
            if (token.IsCancellationRequested) break;
            if (sw.ElapsedMilliseconds - start >= localLimitMs) break;

            if (candidates.Count == 0) break;

            int idx = rnd.Next(candidates.Count);
            int v = candidates[idx];
            candidates.RemoveAt(idx);

            bool ok = true;
            foreach (int u in clique)
            {
                if (!AreNeighbors(u, v)) { ok = false; break; }
            }
            if (ok)
            {
                clique.Add(v);
                inClique.Add(v);

                // Filter candidates to those adjacent to new v
                candidates = candidates.Where(x => AreNeighbors(x, v)).ToList();
            }
        }

        return clique;
    }
