            // No input

        }
    };

        Some(x) => x.parse().unwrap_or(0),
        None => 0,
    };

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];

    for _ in 0..m {
        let u: usize = match it.next() { Some(x) => x.parse().unwrap(), None => break };
        let v: usize = match it.next() { Some(x) => x.parse().unwrap(), None => break };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v as u32);
        adj[v].push(u as u32);
    }

    // Greedy dominating set construction
    let dominating_set = greedy_dominating_set(&adj);

    // Local improvement
    let improved_set = improve_dominating_set(&adj, &dominating_set);

    // Sanity check: ensure dominating
    let final_set = if is_dominating_set(&adj, &improved_set) {
        improved_set
    } else if is_dominating_set(&adj, &dominating_set) {
        dominating_set
    } else {
        // Fallback: all vertices
        (0..n as u32).collect::<Vec<u32>>()
    };

    // Output result
    println!("{}", final_set.len());
    if !final_set.is_empty() {
        for (i, v) in final_set.iter().enumerate() {
            if i > 0 { print!(" "); }
            print!("{}", v);
        }
    }
    println!();
}

fn greedy_dominating_set(adj: &Vec<Vec<u32>>) -> Vec<u32> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let mut covered = vec![false; n];
    let mut dominating_set: Vec<u32> = Vec::new();

    // Approximate count of uncovered vertices in closed neighborhood
    let mut score: Vec<i32> = Vec::with_capacity(n);
    for v in 0..n {
        // closed neighborhood size = deg + 1
        score.push((adj[v].len() + 1) as i32);
    }

    let mut uncovered_count = n; // number of uncovered vertices

    // Quick lambda to check if vertex can still contribute
    let mut iteration_guard = 0usize; // safety against infinite loops

    while uncovered_count > 0 {
        iteration_guard += 1;
        if iteration_guard > n * 5 {
            // Safety fallback, avoid any unexpected infinite loop
            break;
        }

        // Select best vertex by score among those that can cover someone uncovered
        let mut best: Option<usize> = None;
        let mut best_score: i32 = -1;

        for v in 0..n {
            if score[v] <= 0 {
                continue;
            }
            if score[v] > best_score {
                best_score = score[v];
                best = Some(v);
            }
        }

        let chosen = match best {
            Some(v) => v,
            None => {
                // As a fallback, pick any uncovered vertex
                let mut any_uncovered = None;
                for v in 0..n {
                    if !covered[v] {
                        any_uncovered = Some(v);
                        break;
                    }
                }
                if let Some(v) = any_uncovered {
                    v
                } else {
                    break;
                }
            }
        };

        // Add chosen to dominating set
        dominating_set.push(chosen as u32);

        // For chosen and its neighbors: mark covered and update scores
        // We first collect all vertices newly covered by this choice
        let mut newly_covered: Vec<usize> = Vec::new();

        if !covered[chosen] {
            covered[chosen] = true;
            uncovered_count -= 1;
            newly_covered.push(chosen);
        }

        for &nb_u32 in &adj[chosen] {
            let u = nb_u32 as usize;
            if !covered[u] {
                covered[u] = true;
                uncovered_count -= 1;
                newly_covered.push(u);
            }
        }

        // Decrease scores for vertices whose neighborhoods lost uncovered vertices
        // For each newly covered u, its neighbors (and itself) lose one potential uncovered
        for &u in &newly_covered {
            // Self
            if score[u] > 0 {
                score[u] -= 1;
            }
            // Neighbors
            for &w_u32 in &adj[u] {
                let w = w_u32 as usize;
                if score[w] > 0 {
                    score[w] -= 1;
                }
            }
        }
    }

    // In case some vertices remain uncovered due to approximate scores, cover them directly
    for v in 0..n {
        if !covered[v] {
            dominating_set.push(v as u32);
            covered[v] = true;
            for &u32 in &adj[v] {
                covered[u32 as usize] = true;
            }
        }
    }

    dominating_set
}

fn improve_dominating_set(adj: &Vec<Vec<u32>>, dom_set: &Vec<u32>) -> Vec<u32> {
    let n = adj.len();
    if n == 0 || dom_set.is_empty() {
        return dom_set.clone();
    }

    let mut in_set = vec![false; n];
    for &v_u32 in dom_set {
        in_set[v_u32 as usize] = true;
    }

    // cover_count[u] = how many vertices in current dominating set dominate u
    let mut cover_count: Vec<u32> = vec![0; n];

    for &v_u32 in dom_set {
        let v = v_u32 as usize;
        cover_count[v] += 1;
        for &u32 in &adj[v] {
            cover_count[u32 as usize] += 1;
        }
    }

    let mut current_set: Vec<u32> = dom_set.clone();

    // We'll attempt a couple of passes
    let passes = 2usize;
    for _ in 0..passes {
        let mut i = 0;
        while i < current_set.len() {
            let v_u32 = current_set[i];
            let v = v_u32 as usize;

            // Check if v is removable: all u in N[v] ∪ {v} must remain covered by at least one other vertex
            let mut removable = true;

            // Check self
            if cover_count[v] <= 1 {
                removable = false;
            }

            if removable {
                for &u32 in &adj[v] {
                    let u = u32 as usize;
                    if cover_count[u] <= 1 {
                        removable = false;
                        break;
                    }
                }
            }

            if removable {
                // Remove v from dominating set
                in_set[v] = false;

                // Decrease cover counts
                cover_count[v] -= 1;
                for &u32 in &adj[v] {
                    let u = u32 as usize;
                    cover_count[u] -= 1;
                }

                // Remove v from current_set by swapping with last
                current_set.swap_remove(i);
                // Do not increment i; new element at i must be checked too
            } else {
                i += 1;
            }
        }
    }

    current_set
}

            break;

    let mut iteration_guard = 0usize; // safety against infinite loops

            // Safety fallback, avoid any unexpected infinite loop
            break;

        // Select best vertex by score among those that can cover someone uncovered
        let mut best: Option<usize> = None;
        let mut best_score: i32 = -1;

            if score[v] <= 0 {
                continue;

                best = Some(v);
            }

            None => {

                        any_uncovered = Some(v);
                        break;
                    }
                }
                if let Some(v) = any_uncovered {
                    v
                } else {
                    break;
                }

fn greedy_dominating_set(adj: &Vec<Vec<u32>>) -> Vec<u32> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let mut covered = vec![false; n];
    let mut dominating_set: Vec<u32> = Vec::new();

    // Approximate count of uncovered vertices in closed neighborhood
    let mut score: Vec<i32> = Vec::with_capacity(n);
    for v in 0..n {
        // closed neighborhood size = deg + 1
        score.push((adj[v].len() + 1) as i32);
    }

    let mut uncovered_count = n; // number of uncovered vertices

    // Quick lambda to check if vertex can still contribute
    let mut iteration_guard = 0usize; // safety against infinite loops

    while uncovered_count > 0 {
        iteration_guard += 1;
        if iteration_guard > n * 5 {
            // Safety fallback, avoid any unexpected infinite loop
            break;
        }

        // Select best vertex by score among those that can cover someone uncovered
        let mut best: Option<usize> = None;
        let mut best_score: i32 = -1;

        for v in 0..n {
            if score[v] <= 0 {
                continue;
            }
            if score[v] > best_score {
                best_score = score[v];
                best = Some(v);
            }
        }

        let chosen = match best {
            Some(v) => v,
            None => {
                // As a fallback, pick any uncovered vertex
                let mut any_uncovered = None;
                for v in 0..n {
                    if !covered[v] {
                        any_uncovered = Some(v);
                        break;
                    }
                }
                if let Some(v) = any_uncovered {
                    v
                } else {
                    break;
                }
            }
        };

        // Add chosen to dominating set
        dominating_set.push(chosen as u32);

        // For chosen and its neighbors: mark covered and update scores
        // We first collect all vertices newly covered by this choice
        let mut newly_covered: Vec<usize> = Vec::new();

        if !covered[chosen] {
            covered[chosen] = true;
            uncovered_count -= 1;
            newly_covered.push(chosen);
        }

        for &nb_u32 in &adj[chosen] {
            let u = nb_u32 as usize;
            if !covered[u] {
                covered[u] = true;
                uncovered_count -= 1;
                newly_covered.push(u);
            }
        }

        // Decrease scores for vertices whose neighborhoods lost uncovered vertices
        // For each newly covered u, its neighbors (and itself) lose one potential uncovered
        for &u in &newly_covered {
            // Self
            if score[u] > 0 {
                score[u] -= 1;
            }
            // Neighbors
            for &w_u32 in &adj[u] {
                let w = w_u32 as usize;
                if score[w] > 0 {
                    score[w] -= 1;
                }
            }
        }
    }

    // In case some vertices remain uncovered due to approximate scores, cover them directly
    for v in 0..n {
        if !covered[v] {
            dominating_set.push(v as u32);
            covered[v] = true;
            for &u32 in &adj[v] {
                covered[u32 as usize] = true;
            }
        }
    }

    dominating_set
}

fn improve_dominating_set(adj: &Vec<Vec<u32>>, dom_set: &Vec<u32>) -> Vec<u32> {
    let n = adj.len();
    if n == 0 || dom_set.is_empty() {
        return dom_set.clone();
    }

    let mut in_set = vec![false; n];
    for &v_u32 in dom_set {
        in_set[v_u32 as usize] = true;
    }

    // cover_count[u] = how many vertices in current dominating set dominate u
    let mut cover_count: Vec<u32> = vec![0; n];

    for &v_u32 in dom_set {
        let v = v_u32 as usize;
        cover_count[v] += 1;
        for &u32 in &adj[v] {
            cover_count[u32 as usize] += 1;
        }
    }

    let mut current_set: Vec<u32> = dom_set.clone();

    // We'll attempt a couple of passes
    let passes = 2usize;
    for _ in 0..passes {
        let mut i = 0;
        while i < current_set.len() {
            let v_u32 = current_set[i];
            let v = v_u32 as usize;

            // Check if v is removable: all u in N[v] ∪ {v} must remain covered by at least one other vertex
            let mut removable = true;

            // Check self
            if cover_count[v] <= 1 {
                removable = false;
            }

            if removable {
                for &u32 in &adj[v] {
                    let u = u32 as usize;
                    if cover_count[u] <= 1 {
                        removable = false;
                        break;
                    }
                }
            }

            if removable {
                // Remove v from dominating set
                in_set[v] = false;

                // Decrease cover counts
                cover_count[v] -= 1;
                for &u32 in &adj[v] {
                    let u = u32 as usize;
                    cover_count[u] -= 1;
                }

                // Remove v from current_set by swapping with last
                current_set.swap_remove(i);
                // Do not increment i; new element at i must be checked too
            } else {
                i += 1;
            }
        }
    }

    current_set
}
