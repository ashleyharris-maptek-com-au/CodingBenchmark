#include <algorithm>

using namespace std;

struct Point {
    float x, y;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = double(a.x) - double(b.x);
    double dy = double(a.y) - double(b.y);
    return dx*dx + dy*dy;
}

static inline double dist(const Point& a, const Point& b) {
    return sqrt(dist2(a,b));
}

// Simple time limiter helper
struct Timer {
    chrono::steady_clock::time_point start;
    Timer() : start(chrono::steady_clock::now()) {}
    double elapsed() const {
        using namespace chrono;
        return duration_cast<duration<double>>(steady_clock::now() - start).count();
    }
};

/// ---------- Strategy 1: brute-force / full 2-opt for very small N ----------

void solve_small(const vector<Point>& pts, vector<int>& tour, double time_limit_sec) {
    int n = (int)pts.size();
    tour.resize(n);
    iota(tour.begin(), tour.end(), 0);

    if (n <= 12) {
        // brute force via next_permutation, fixing 0 at first position
        vector<int> best = tour;
        double best_len = numeric_limits<double>::infinity();

        vector<int> perm(n-1);
        iota(perm.begin(), perm.end(), 1);

        do {
            double len = 0.0;
            int prev = 0;
            for (int i = 0; i < n-1; ++i) {
                int v = perm[i];
                len += dist(pts[prev], pts[v]);
                prev = v;
            }
            if (len < best_len) {
                best_len = len;
                best[0] = 0;
                for (int i = 0; i < n-1; ++i) best[i+1] = perm[i];
            }
        } while (next_permutation(perm.begin(), perm.end()));

        tour = move(best);
        return;
    }

    // nearest neighbor greedy from 0
    vector<char> used(n, 0);
    tour[0] = 0;
    used[0] = 1;
    for (int i = 1; i < n; ++i) {
        int last = tour[i-1];
        int best = -1;
        double best_d2 = numeric_limits<double>::infinity();
        for (int j = 0; j < n; ++j) if (!used[j]) {
            double d2 = dist2(pts[last], pts[j]);
            if (d2 < best_d2) {
                best_d2 = d2;
                best = j;
            }
        }
        tour[i] = best;
        used[best] = 1;
    }

    // full 2-opt until no improvement or time limit
    Timer timer;
    bool improved = true;
    while (improved && timer.elapsed() < time_limit_sec) {
        improved = false;
        for (int i = 1; i < n-2; ++i) {
            for (int k = i+1; k < n-1; ++k) {
                if (timer.elapsed() > time_limit_sec) return;

                int a = tour[i-1];
                int b = tour[i];
                int c = tour[k];
                int d = tour[k+1];

                double before = dist(pts[a], pts[b]) + dist(pts[c], pts[d]);
                double after  = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
                if (after + 1e-9 < before) {
                    reverse(tour.begin()+i, tour.begin()+k+1);
                    improved = true;
                }
            }
        }
    }
}

/// ---------- Strategy 2: nearest-neighbor + limited 2-opt for medium N ----------

void solve_medium(const vector<Point>& pts, vector<int>& tour, double time_limit_sec) {
    int n = (int)pts.size();
    tour.resize(n);
    vector<char> used(n, 0);
    tour[0] = 0;
    used[0] = 1;

    // simple nearest neighbor heuristic
    for (int i = 1; i < n; ++i) {
        int last = tour[i-1];
        int best = -1;
        double best_d2 = numeric_limits<double>::infinity();
        for (int j = 0; j < n; ++j) if (!used[j]) {
            double d2 = dist2(pts[last], pts[j]);
            if (d2 < best_d2) {
                best_d2 = d2;
                best = j;
            }
        }
        tour[i] = best;
        used[best] = 1;
    }

    // limited 2-opt: only some random pairs, time bounded
    Timer timer;
    mt19937_64 rng(1234567);
    uniform_int_distribution<int> dist_i(1, n-3);
    uniform_int_distribution<int> dist_k(2, n-2);

    const int max_trials = n * 50; // heuristic
    for (int t = 0; t < max_trials; ++t) {
        if (timer.elapsed() > time_limit_sec) break;
        int i = dist_i(rng);
        int k = i + dist_k(rng) % (n-1-i);
        if (k <= i) continue;

        int a = tour[i-1];
        int b = tour[i];
        int c = tour[k];
        int d = tour[(k+1)%n];

        double before = dist(pts[a], pts[b]) + dist(pts[c], pts[d]);
        double after  = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
        if (after + 1e-9 < before) {
            reverse(tour.begin()+i, tour.begin()+k+1);
        }
    }
}

/// ---------- Strategy 3: space-filling curve (Hilbert-like) for large+ ----------

// A simple, fast integer hash for 32-bit key
static inline uint32_t splitmix32(uint32_t x) {
    x += 0x9e3779b9;
    x = (x ^ (x >> 16)) * 0x85ebca6b;
    x = (x ^ (x >> 13)) * 0xc2b2ae35;
    x = x ^ (x >> 16);
    return x;
}

// Map [0,1]x[0,1] to a 64-bit Morton-like code.
// This is not a true Hilbert curve, but similar locality behavior.
// It avoids floats beyond initial normalization.
static inline uint64_t morton_code(float x, float y) {
    // clamp and scale
    double xx = x, yy = y;
    if (xx < 0) xx = 0; if (xx > 1) xx = 1;
    if (yy < 0) yy = 0; if (yy > 1) yy = 1;
    uint32_t ix = (uint32_t)(xx * 4294967295.0); // 2^32-1
    uint32_t iy = (uint32_t)(yy * 4294967295.0);

    auto part1by1 = [](uint32_t n) {
        uint64_t x = n;
        x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
        x = (x | (x << 8))  & 0x00FF00FF00FF00FFULL;
        x = (x | (x << 4))  & 0x0F0F0F0F0F0F0F0FULL;
        x = (x | (x << 2))  & 0x3333333333333333ULL;
        x = (x | (x << 1))  & 0x5555555555555555ULL;
        return x;
    };

    uint64_t mx = part1by1(ix);
    uint64_t my = part1by1(iy) << 1;
    return mx | my;
}

void solve_large_space_filling(const vector<Point>& pts, vector<int>& tour) {
    int n = (int)pts.size();
    tour.resize(n);

    // find bounding box
    float minx = numeric_limits<float>::infinity();
    float miny = numeric_limits<float>::infinity();
    float maxx = -numeric_limits<float>::infinity();
    float maxy = -numeric_limits<float>::infinity();
    for (const auto& p : pts) {
        if (p.x < minx) minx = p.x;
        if (p.y < miny) miny = p.y;
        if (p.x > maxx) maxx = p.x;
        if (p.y > maxy) maxy = p.y;
    }
    float rangex = maxx - minx;
    float rangey = maxy - miny;
    if (rangex <= 0) rangex = 1;
    if (rangey <= 0) rangey = 1;

    struct Node {
        uint64_t key;
        int idx;
    };
    vector<Node> nodes;
    nodes.reserve(n);

    for (int i = 0; i < n; ++i) {
        float nx = (pts[i].x - minx) / rangex;
        float ny = (pts[i].y - miny) / rangey;
        uint64_t key = morton_code(nx, ny);
        // Add small hashed perturbation to reduce ties
        key ^= (uint64_t)splitmix32((uint32_t)i);
        nodes.push_back({key, i});
    }

    sort(nodes.begin(), nodes.end(),
         [](const Node& a, const Node& b){ return a.key < b.key; });

    // Ensure 0 is first: rotate so that index 0 appears at position 0,
    // preserving relative order (still a valid SFC traversal)
    int pos0 = 0;
    for (int i = 0; i < n; ++i) {
        if (nodes[i].idx == 0) { pos0 = i; break; }
    }

    // Build in rotated order
    int t = 0;
    for (int i = pos0; i < n; ++i) tour[t++] = nodes[i].idx;
    for (int i = 0; i < pos0; ++i) tour[t++] = nodes[i].idx;
}

/// ----------- Main dispatcher: choose strategy based on N ----------------------

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) {
        return 0;
    }
    vector<Point> pts(N);
    for (int i = 0; i < N; ++i) {
        cin >> pts[i].x >> pts[i].y;
    }

    vector<int> tour;

    // Global (very generous) time budget in seconds.
    // In contest/system this would be much lower than 300 for 1e9, but
    // we keep the internal algorithms lightweight anyway.
    const double global_time_limit = 280.0;

    if (N <= 40) {
        // brute force + full 2-opt
        solve_small(pts, tour, min(30.0, global_time_limit));
    } else if (N <= 2000) {
        // NN + limited 2-opt
        // Time scales mildly with N
        double tl = min(global_time_limit, 0.1 * N);
        solve_medium(pts, tour, tl);
    } else {
        // Large and up: space-filling curve ordering
        solve_large_space_filling(pts, tour);
    }

    // Safety: ensure valid permutation starting with 0
    // If not, fix trivially (should not happen).
    if ((int)tour.size() != N) {
        tour.resize(N);
        iota(tour.begin(), tour.end(), 0);
    } else {
        vector<char> seen(N, 0);
        for (int i = 0; i < N; ++i) {
            int v = tour[i];
            if (v < 0 || v >= N || seen[v]) {
                // fallback to identity
                iota(tour.begin(), tour.end(), 0);
                break;
            }
            seen[v] = 1;
        }
    }
    if (tour[0] != 0) {
        int pos0 = -1;
        for (int i = 0; i < N; ++i) if (tour[i] == 0) { pos0 = i; break; }
        if (pos0 == -1) {
            iota(tour.begin(), tour.end(), 0);
        } else if (pos0 != 0) {
            rotate(tour.begin(), tour.begin()+pos0, tour.end());
        }
    }

    // Output
    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
