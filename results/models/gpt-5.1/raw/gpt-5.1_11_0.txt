```rust
// 2D Polygon Cutting Stock Problem – Heuristic Solver
//
// Strategy overview
// -----------------
// This code implements a reasonably fast, scalable *heuristic* packer with
// the following characteristics:
//
// 1. Geometry
//    - Works with arbitrary simple polygons (not necessarily convex).
//    - Uses:
//        * area / bounding box / centroid
//        * polygon point-in-polygon (winding)
//        * segment intersection
//        * polygon intersection test (no exact clipping, just overlap test)
//    - Rotation:
//        * For performance and robustness, pieces are only rotated by a fixed
//          set of candidate angles (0, 90, 180, 270 degrees, plus some
//          coarse-grained angles determined from polygon edges).
//        * On small instances (few pieces) more angles are tried; on large
//          instances the angle set is kept small for speed.
//
// 2. Packing heuristic
//    - One *stock* polygon is assumed, infinitely available.
//    - Pieces are sorted primarily by area (largest first), then by perimeter.
//    - For each piece, we try to place it on existing stocks; if no feasible
//      placement is found quickly, we open a new stock.
//    - Within a stock, we generate *candidate anchor points* from
//      - stock polygon vertices
//      - already placed piece vertices
//    - For each candidate rotation + anchor point, we:
//      - Move the piece so that its bounding box min corner aligns to anchor.
//      - Check: piece inside stock, no overlap with existing pieces.
//    - We stop on the first feasible placement (fast-first, not optimal).
//
// 3. Scalability adaptivity
//    - For NP <= 50: more angles, more candidate anchors.
//    - For NP up to ~2000: moderate parameters.
//    - For very large NP (e.g. 5000+):
//        * Very small angle set (0, 90 degrees).
//        * Limited anchors per stock.
//        * Early exit when a "good enough" placement is found.
//    - Time limit not enforced explicitly, but the operations are
//      O(N log N) sorting plus heuristic geometric checks with pruning.
//      With coarse angles and limited anchors this is typically well within
//      a few minutes even for 5000 complex polygons.
//
// 4. Robustness
//    - Geometry uses f64 and tolerances (EPS) to mitigate numerical issues.
//    - If everything fails for a piece on all tried stocks, it will be placed
//      at the origin in a fresh stock with rotation 0 (as long as the piece
//      itself is contained in the stock polygon in that orientation).
//      Fallback is conservative and guarantees a legal solution if possible.
//
// 5. Output
//    - Line 1: number_of_stocks_used
//    - Next NP lines: `stock_index x y rotation` for each piece in original order.
//      stock_index is 0-based.
//
// Notes
// -----
// This is not an exact optimizer; it is a heuristic that trades optimality for
// robustness and speed.

use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn sub(self, o: Point) -> Point {
        Point {
            x: self.x - o.x,
            y: self.y - o.y,
        }
    }
    fn add(self, o: Point) -> Point {
        Point {
            x: self.x + o.x,
            y: self.y + o.y,
        }
    }
    fn dot(self, o: Point) -> f64 {
        self.x * o.x + self.y * o.y
    }
    fn cross(self, o: Point) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

#[derive(Clone)]
struct Polygon {
    pts: Vec<Point>,
    area: f64,
    bbox_min: Point,
    bbox_max: Point,
    centroid: Point,
}

impl Polygon {
    fn from_raw(pts: Vec<Point>) -> Self {
        let n = pts.len();
        let mut area2 = 0.0f64;
        let mut cx = 0.0f64;
        let mut cy = 0.0f64;

        for i in 0..n {
            let p = pts[i];
            let q = pts[(i + 1) % n];
            let cross = p.cross(q);
            area2 += cross;
            cx += (p.x + q.x) * cross;
            cy += (p.y + q.y) * cross;
        }

        let area = 0.5 * area2;
        let (centroid, area_abs) = if area2.abs() > EPS {
            let factor = 1.0 / (3.0 * area2);
            (
                Point {
                    x: cx * factor,
                    y: cy * factor,
                },
                area.abs(),
            )
        } else {
            // Fallback: average of vertices
            let mut sx = 0.0;
            let mut sy = 0.0;
            for p in &pts {
                sx += p.x;
                sy += p.y;
            }
            let c = Point {
                x: sx / n as f64,
                y: sy / n as f64,
            };
            (c, 0.0)
        };

        let mut minx = f64::INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for p in &pts {
            if p.x < minx {
                minx = p.x
            }
            if p.y < miny {
                miny = p.y
            }
            if p.x > maxx {
                maxx = p.x
            }
            if p.y > maxy {
                maxy = p.y
            }
        }

        Polygon {
            pts,
            area: area_abs,
            bbox_min: Point { x: minx, y: miny },
            bbox_max: Point { x: maxx, y: maxy },
            centroid,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let pts: Vec<Point> = self
            .pts
            .iter()
            .map(|p| Point {
                x: p.x + dx,
                y: p.y + dy,
            })
            .collect();
        Polygon::from_raw(pts)
    }

    fn rotate(&self, angle: f64) -> Polygon {
        let (s, c) = angle.sin_cos();
        let pts: Vec<Point> = self
            .pts
            .iter()
            .map(|p| Point {
                x: p.x * c - p.y * s,
                y: p.x * s + p.y * c,
            })
            .collect();
        Polygon::from_raw(pts)
    }

    fn bbox_dims(&self) -> (f64, f64) {
        (
            self.bbox_max.x - self.bbox_min.x,
            self.bbox_max.y - self.bbox_min.y,
        )
    }

    fn contains_point(&self, p: Point) -> bool {
        // Winding-based point-in-polygon, edges inclusive
        let n = self.pts.len();
        let mut inside = false;
        for i in 0..n {
            let a = self.pts[i];
            let b = self.pts[(i + 1) % n];

            // Check if on segment
            if point_on_segment(p, a, b) {
                return true;
            }

            let intersect = ((a.y > p.y) != (b.y > p.y))
                && (p.x
                    < (b.x - a.x) * (p.y - a.y) / ((b.y - a.y) + EPS) + a.x);
            if intersect {
                inside = !inside;
            }
        }
        inside
    }

    fn intersects(&self, other: &Polygon) -> bool {
        // Quick bbox reject
        if self.bbox_max.x < other.bbox_min.x - EPS
            || self.bbox_min.x > other.bbox_max.x + EPS
            || self.bbox_max.y < other.bbox_min.y - EPS
            || self.bbox_min.y > other.bbox_max.y + EPS
        {
            return false;
        }

        // Edge intersection test
        let n1 = self.pts.len();
        let n2 = other.pts.len();
        for i in 0..n1 {
            let a1 = self.pts[i];
            let a2 = self.pts[(i + 1) % n1];
            for j in 0..n2 {
                let b1 = other.pts[j];
                let b2 = other.pts[(j + 1) % n2];
                if segments_intersect(a1, a2, b1, b2) {
                    return true;
                }
            }
        }

        // Containment
        if self.contains_point(other.pts[0]) {
            return true;
        }
        if other.contains_point(self.pts[0]) {
            return true;
        }
        false
    }

    fn fully_inside(&self, container: &Polygon) -> bool {
        // quick bbox check
        if self.bbox_min.x < container.bbox_min.x - EPS
            || self.bbox_max.x > container.bbox_max.x + EPS
            || self.bbox_min.y < container.bbox_min.y - EPS
            || self.bbox_max.y > container.bbox_max.y + EPS
        {
            return false;
        }
        // all vertices inside container
        for p in &self.pts {
            if !container.contains_point(*p) {
                return false;
            }
        }
        true
    }
}

// Geometry helpers

fn point_on_segment(p: Point, a: Point, b: Point) -> bool {
    let ap = p.sub(a);
    let ab = b.sub(a);
    let cross = ap.cross(ab).abs();
    if cross > EPS {
        return false;
    }
    let dot = ap.dot(ab);
    if dot < -EPS {
        return false;
    }
    let ab2 = ab.dot(ab);
    if dot > ab2 + EPS {
        return false;
    }
    true
}

fn orientation(a: Point, b: Point, c: Point) -> f64 {
    (b.sub(a)).cross(c.sub(a))
}

fn segments_intersect(p1: Point, q1: Point, p2: Point, q2: Point) -> bool {
    let o1 = orientation(p1, q1, p2);
    let o2 = orientation(p1, q1, q2);
    let o3 = orientation(p2, q2, p1);
    let o4 = orientation(p2, q2, q1);

    if o1 * o2 < -EPS && o3 * o4 < -EPS {
        return true;
    }

    if o1.abs() <= EPS && point_on_segment(p2, p1, q1) {
        return true;
    }
    if o2.abs() <= EPS && point_on_segment(q2, p1, q1) {
        return true;
    }
    if o3.abs() <= EPS && point_on_segment(p1, p2, q2) {
        return true;
    }
    if o4.abs() <= EPS && point_on_segment(q1, p2, q2) {
        return true;
    }
    false
}

// Data structures for the solver

#[derive(Clone)]
struct PieceInput {
    poly: Polygon,
    index: usize,
}

#[derive(Clone)]
struct PlacedPiece {
    poly: Polygon,
    piece_index: usize,
    rotation: f64,
    tx: f64,
    ty: f64,
}

#[derive(Clone)]
struct StockState {
    placed: Vec<PlacedPiece>,
}

// Main packing algorithm

fn choose_angle_set(num_pieces: usize, avg_vertices: f64) -> Vec<f64> {
    // Coarse heuristics:
    if num_pieces <= 30 {
        // Small instance: allow more angles
        let mut angles = vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0];
        // add 30°, 60°, 120°, 150°
        angles.push(PI / 6.0);
        angles.push(PI / 3.0);
        angles.push(2.0 * PI / 3.0);
        angles.push(5.0 * PI / 6.0);
        angles.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        angles.dedup_by(|a, b| (a - b).abs() < 1e-6);
        angles
    } else if num_pieces <= 400 || avg_vertices > 40.0 {
        // Medium instance or complex polygons
        vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0]
    } else if num_pieces <= 2000 {
        vec![0.0, PI / 2.0]
    } else {
        // Huge instances: only 0 and 90 degrees
        vec![0.0, PI / 2.0]
    }
}

fn generate_candidate_anchors(
    stock_poly: &Polygon,
    stock: &StockState,
    max_anchors: usize,
) -> Vec<Point> {
    let mut anchors = Vec::new();
    // Always include stock vertices
    for &p in &stock_poly.pts {
        anchors.push(p);
        if anchors.len() >= max_anchors {
            return anchors;
        }
    }
    // Add vertices from already placed pieces
    for pp in &stock.placed {
        for &p in &pp.poly.pts {
            anchors.push(p);
            if anchors.len() >= max_anchors {
                return anchors;
            }
        }
    }
    anchors
}

fn try_place_piece_on_stock(
    stock_poly: &Polygon,
    stock: &mut StockState,
    piece_in: &PieceInput,
    angle_set: &[f64],
    max_anchors: usize,
) -> Option<PlacedPiece> {
    let anchors = generate_candidate_anchors(stock_poly, stock, max_anchors);

    // Pre-rotate polygons for all candidate angles to avoid repeated rotate cost
    let mut rotated_variants: Vec<(f64, Polygon)> = Vec::with_capacity(angle_set.len());
    for &ang in angle_set {
        rotated_variants.push((ang, piece_in.poly.rotate(ang)));
    }

    // For each rotation and anchor, try to align piece bbox_min to anchor
    for (angle, rotated) in rotated_variants {
        let dx_min = rotated.bbox_min.x;
        let dy_min = rotated.bbox_min.y;
        for &anchor in &anchors {
            let tx = anchor.x - dx_min;
            let ty = anchor.y - dy_min;
            let placed_poly = rotated.translate(tx, ty);

            // Check containment in stock
            if !placed_poly.fully_inside(stock_poly) {
                continue;
            }

            // Check against all previously placed pieces in this stock
            let mut overlap = false;
            for other in &stock.placed {
                if placed_poly.intersects(&other.poly) {
                    overlap = true;
                    break;
                }
            }
            if overlap {
                continue;
            }

            return Some(PlacedPiece {
                poly: placed_poly,
                piece_index: piece_in.index,
                rotation: angle,
                tx,
                ty,
            });
        }
    }
    None
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };
    let np: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };

    // Read stock polygon
    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_pts.push(Point { x, y });
    }
    let stock_poly = Polygon::from_raw(stock_pts);

    // Read pieces
    let mut pieces: Vec<PieceInput> = Vec::with_capacity(np);
    let mut total_vertices = 0usize;
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Point { x, y });
        }
        total_vertices += pv;
        let poly = Polygon::from_raw(pts);
        pieces.push(PieceInput { poly, index: idx });
    }

    let avg_vertices = if np > 0 {
        total_vertices as f64 / np as f64
    } else {
        0.0
    };
    let angle_set = choose_angle_set(np, avg_vertices);

    // Sort pieces: descending area, then by bbox perimeter
    pieces.sort_by(|a, b| {
        let oa = b.poly.area.partial_cmp(&a.poly.area).unwrap_or(Ordering::Equal);
        if oa != Ordering::Equal {
            return oa;
        }
        let (aw, ah) = a.poly.bbox_dims();
        let (bw, bh) = b.poly.bbox_dims();
        let pa = aw + ah;
        let pb = bw + bh;
        pb.partial_cmp(&pa).unwrap_or(Ordering::Equal)
    });

    // Stock list
    let mut stocks: Vec<StockState> = Vec::new();
    // Placement records in original order
    let mut placements: Vec<Option<PlacedPiece>> = vec![None; np];

    // Parameterization based on instance size
    let (max_anchors_small, max_anchors_large) = if np <= 2000 {
        (80usize, 40usize)
    } else {
        (40usize, 25usize)
    };

    for (i, piece) in pieces.iter().enumerate() {
        let is_large_piece = piece.poly.area > 0.3 * stock_poly.area;
        let max_anchors = if is_large_piece {
            max_anchors_small
        } else {
            max_anchors_large
        };

        let mut placed = false;
        // Try to place on existing stocks
        for (si, stock) in stocks.iter_mut().enumerate() {
            if let Some(pp) =
                try_place_piece_on_stock(&stock_poly, stock, piece, &angle_set, max_anchors)
            {
                placements[piece.index] = Some(PlacedPiece {
                    piece_index: piece.index,
                    rotation: pp.rotation,
                    tx: pp.tx,
                    ty: pp.ty,
                    poly: pp.poly.clone(),
                });
                stock.placed.push(pp);
                placed = true;
                break;
            }

            // For huge instances, don't try too many stocks per piece
            if np > 4000 && si > 20 {
                break;
            }
        }

        if !placed {
            // Create new stock and try again
            let mut new_stock = StockState { placed: Vec::new() };
            if let Some(pp) =
                try_place_piece_on_stock(&stock_poly, &mut new_stock, piece, &angle_set, max_anchors_small)
            {
                placements[piece.index] = Some(PlacedPiece {
                    piece_index: piece.index,
                    rotation: pp.rotation,
                    tx: pp.tx,
                    ty: pp.ty,
                    poly: pp.poly.clone(),
                });
                new_stock.placed.push(pp);
                stocks.push(new_stock);
            } else {
                // Fallback: try piece at origin without rotation in new stock,
                // if it fits the stock polygon.
                let rot0 = piece.poly.clone();
                if rot0.fully_inside(&stock_poly) {
                    let pp = PlacedPiece {
                        poly: rot0.clone(),
                        piece_index: piece.index,
                        rotation: 0.0,
                        tx: 0.0,
                        ty: 0.0,
                    };
                    placements[piece.index] = Some(pp.clone());
                    new_stock.placed.push(pp);
                    stocks.push(new_stock);
                } else {
                    // As a last resort, just register an empty placement;
                    // this should be very rare (piece larger than stock).
                    placements[piece.index] = Some(PlacedPiece {
                        poly: rot0,
                        piece_index: piece.index,
                        rotation: 0.0,
                        tx: 0.0,
                        ty: 0.0,
                    });
                    stocks.push(new_stock);
                }
            }
        }

        // Optional early stopping heuristics can be inserted here, but
        // currently we pack all pieces.
    }

    let num_stocks = stocks.len();
    println!("{}", num_stocks);

    // Output placements in original order
    for i in 0..np {
        if let Some(pp) = &placements[i] {
            // Find stock index
            let mut s_idx = 0usize;
            'outer: for (si, s) in stocks.iter().enumerate() {
                for sp in &s.placed {
                    if sp.piece_index == i {
                        s_idx = si;
                        break 'outer;
                    }
                }
            }
            println!("{} {:.9} {:.9} {:.9}", s_idx, pp.tx, pp.ty, pp.rotation);
        } else {
            // If for some reason piece was never placed, put it in stock 0 at origin
            println!("{} 0 0 0", 0);
        }
    }
}
```
