#include <bits/stdc++.h>
using namespace std;

/*
 Simple iterative deepening alpha-beta chess bot with FEN input and
 disambiguated algebraic move output as specified in the problem.

 This is not a full professional engine, but it:
 - Parses FEN
 - Generates legal moves (including castling, promotion, en passant)
 - Does alpha-beta with iterative deepening and basic move ordering
 - Detects check, checkmate, stalemate
 - Produces the required move notation (including x, +, #, ep, promotions, O-O/O-O-O)

 It respects a given time limit in seconds.
*/

// -------------------------------------------------------------
// Time control
// -------------------------------------------------------------
atomic<bool> timeUp(false);
chrono::steady_clock::time_point endTime;

bool isTimeUp() {
    return timeUp.load(memory_order_relaxed) ||
           chrono::steady_clock::now() >= endTime;
}

// -------------------------------------------------------------
// Basic types
// -------------------------------------------------------------
enum Piece {
    EMPTY = 0,
    WP, WN, WB, WR, WQ, WK,
    BP, BN, BB, BR, BQ, BK
};

enum Color { WHITE = 0, BLACK = 1 };

struct Move {
    int from = -1;
    int to = -1;
    int piece = EMPTY;
    int captured = EMPTY;
    bool isCapture = false;
    bool isPromotion = false;
    int promoPiece = EMPTY;
    bool isEnPassant = false;
    bool isCastleKS = false;
    bool isCastleQS = false;
};

struct Board {
    int board[64];
    Color sideToMove;
    bool whiteCastleKS, whiteCastleQS;
    bool blackCastleKS, blackCastleQS;
    int epSquare;            // square index or -1
    int halfmoveClock;
    int fullmoveNumber;

    // Precomputed king locations for speed
    int whiteKingSq;
    int blackKingSq;

    Board() {
        memset(board, 0, sizeof(board));
        sideToMove = WHITE;
        whiteCastleKS = whiteCastleQS = false;
        blackCastleKS = blackCastleQS = false;
        epSquare = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        whiteKingSq = blackKingSq = -1;
    }
};

// Helpers
inline int sq(int file, int rank) { return rank * 8 + file; }
inline int fileOf(int sq) { return sq & 7; }
inline int rankOf(int sq) { return sq >> 3; }

inline bool onBoard(int sq) {
    return sq >= 0 && sq < 64;
}

inline Color pieceColor(int p) {
    if (p >= WP && p <= WK) return WHITE;
    if (p >= BP && p <= BK) return BLACK;
    return WHITE; // default
}

// -------------------------------------------------------------
// FEN parsing
// -------------------------------------------------------------
bool parseFEN(const string &fen, Board &b) {
    b = Board();

    stringstream ss(fen);
    string boardPart, sidePart, castlePart, epPart;
    int halfmove, fullmove;

    if (!(ss >> boardPart >> sidePart >> castlePart >> epPart >> halfmove >> fullmove)) {
        // allow missing halfmove/fullmove (not critical for search)
        ss.clear();
        ss.str(fen);
        if (!(ss >> boardPart >> sidePart >> castlePart >> epPart)) return false;
        halfmove = 0;
        fullmove = 1;
    }

    int rank = 7;
    int file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            rank--;
            file = 0;
            if (rank < 0) return false;
            continue;
        }
        if (isdigit((unsigned char)c)) {
            file += c - '0';
        } else {
            int sqi = sq(file, rank);
            int p = EMPTY;
            switch (c) {
                case 'P': p = WP; break;
                case 'N': p = WN; break;
                case 'B': p = WB; break;
                case 'R': p = WR; break;
                case 'Q': p = WQ; break;
                case 'K': p = WK; b.whiteKingSq = sqi; break;
                case 'p': p = BP; break;
                case 'n': p = BN; break;
                case 'b': p = BB; break;
                case 'r': p = BR; break;
                case 'q': p = BQ; break;
                case 'k': p = BK; b.blackKingSq = sqi; break;
                default: return false;
            }
            b.board[sqi] = p;
            file++;
        }
    }

    b.sideToMove = (sidePart == "w" ? WHITE : BLACK);

    b.whiteCastleKS = b.whiteCastleQS = b.blackCastleKS = b.blackCastleQS = false;
    if (castlePart != "-") {
        for (char c : castlePart) {
            if (c == 'K') b.whiteCastleKS = true;
            else if (c == 'Q') b.whiteCastleQS = true;
            else if (c == 'k') b.blackCastleKS = true;
            else if (c == 'q') b.blackCastleQS = true;
        }
    }

    if (epPart == "-") b.epSquare = -1;
    else {
        if (epPart.size() != 2) return false;
        int f = epPart[0] - 'a';
        int r = epPart[1] - '1';
        if (f < 0 || f > 7 || r < 0 || r > 7) return false;
        b.epSquare = sq(f, r);
    }

    b.halfmoveClock = halfmove;
    b.fullmoveNumber = fullmove;

    return true;
}

// Move generation utilities
// -------------------------------------------------------------
const int knightDirs[8] = { 17, 15, 10, 6, -17, -15, -10, -6 };
const int bishopDirs[4] = { 9, 7, -9, -7 };
const int rookDirs[4] = { 8, -8, 1, -1 };
const int kingDirs[8] = { 8, -8, 1, -1, 9, 7, -9, -7 };

// -------------------------------------------------------------
// Attack detection
// -------------------------------------------------------------
bool isSquareAttacked(const Board &b, int sqr, Color bySide);

// -------------------------------------------------------------
// Move generation (pseudo-legal)
// -------------------------------------------------------------
void genPawnMoves(const Board &b, vector<Move> &moves, Color side) {
    int dir = (side == WHITE ? 1 : -1);
    int startRank = (side == WHITE ? 1 : 6);
    int promoRank = (side == WHITE ? 6 : 1);
    int epSq = b.epSquare;

    for (int s = 0; s < 64; ++s) {
        int p = b.board[s];
        if (p == EMPTY) continue;
        if (side == WHITE && p != WP) continue;
        if (side == BLACK && p != BP) continue;

        int r = rankOf(s);
        int f = fileOf(s);

        int oneStep = s + dir * 8;
        if (onBoard(oneStep) && b.board[oneStep] == EMPTY) {
            // promotion?
            if (r == promoRank) {
                int promoPieces[4] = { (side==WHITE?WQ:BQ), (side==WHITE?WR:BR),
                                       (side==WHITE?WB:BB), (side==WHITE?WN:BN) };
                for (int pp: promoPieces) {
                    Move m;
                    m.from = s; m.to = oneStep; m.piece = p;
                    m.isPromotion = true; m.promoPiece = pp;
                    moves.push_back(m);
                }
            } else {
                Move m;
                m.from = s; m.to = oneStep; m.piece = p;
                moves.push_back(m);

                // two-step
                if (r == startRank) {
                    int twoStep = s + dir * 16;
                    if (onBoard(twoStep) && b.board[twoStep] == EMPTY) {
                        Move m2;
                        m2.from = s; m2.to = twoStep; m2.piece = p;
                        moves.push_back(m2);
                    }
                }
            }
        }

        // captures
        for (int df : {-1, 1}) {
            int nf = f + df;
            if (nf < 0 || nf > 7) continue;
            int to = s + dir * 8 + df;
            if (!onBoard(to)) continue;
            int target = b.board[to];
            if (target != EMPTY && pieceColor(target) != side) {
                // capture
                if (r == promoRank) {
                    int promoPieces[4] = { (side==WHITE?WQ:BQ), (side==WHITE?WR:BR),
                                           (side==WHITE?WB:BB), (side==WHITE?WN:BN) };
                    for (int pp: promoPieces) {
                        Move m;
                        m.from = s; m.to = to; m.piece = p;
                        m.captured = target;
                        m.isCapture = true;
                        m.isPromotion = true;
                        m.promoPiece = pp;
                        moves.push_back(m);
                    }
                } else {
                    Move m;
                    m.from = s; m.to = to; m.piece = p;
                    m.captured = target;
                    m.isCapture = true;
                    moves.push_back(m);
                }
            } else if (to == epSq) {
                // en passant
                Move m;
                m.from = s; m.to = to; m.piece = p;
                m.isCapture = true;
                m.isEnPassant = true;
                m.captured = (side == WHITE ? BP : WP);
                moves.push_back(m);
            }
        }
    }
}

void genKnightMoves(const Board &b, vector<Move> &moves, Color side) {
    for (int s = 0; s < 64; ++s) {
        int p = b.board[s];
        if (p == EMPTY) continue;
        if (side == WHITE && p != WN) continue;
        if (side == BLACK && p != BN) continue;

        int rf = rankOf(s), ff = fileOf(s);
        for (int d : knightDirs) {
            int to = s + d;
            if (!onBoard(to)) continue;
            int tr = rankOf(to), tf = fileOf(to);
            if (max(abs(tr - rf), abs(tf - ff)) != 2 ||
                abs(tr - rf) == 0 || abs(tf - ff) == 0) continue;

            int target = b.board[to];
            if (target == EMPTY || pieceColor(target) != side) {
                Move m;
                m.from = s; m.to = to; m.piece = p;
                if (target != EMPTY) {
                    m.captured = target;
                    m.isCapture = true;
                }
                moves.push_back(m);
            }
        }
    }
}

void genSlidingMoves(const Board &b, vector<Move> &moves, Color side,
                     const int *dirs, int dirCount, const int *pieceTypes, int pieceTypeCount) {
    for (int s = 0; s < 64; ++s) {
        int p = b.board[s];
        if (p == EMPTY) continue;
        bool match = false;
        for (int i = 0; i < pieceTypeCount; ++i) {
            if (p == pieceTypes[i]) {
                match = true; break;
            }
        }
        if (!match) continue;
        if (pieceColor(p) != side) continue;

        for (int i = 0; i < dirCount; ++i) {
            int d = dirs[i];
            int to = s;
            while (true) {
                int r0 = rankOf(to), f0 = fileOf(to);
                to += d;
                if (!onBoard(to)) break;
                int r1 = rankOf(to), f1 = fileOf(to);
                if (abs(r1 - r0) > 1 || abs(f1 - f0) > 1) break; // crossed edge

                int target = b.board[to];
                if (target == EMPTY) {
                    Move m;
                    m.from = s; m.to = to; m.piece = p;
                    moves.push_back(m);
                } else {
                    if (pieceColor(target) != side) {
                        Move m;
                        m.from = s; m.to = to; m.piece = p;
                        m.captured = target;
                        m.isCapture = true;
                        moves.push_back(m);
                    }
                    break;
                }
            }
        }
    }
}

void genKingMoves(const Board &b, vector<Move> &moves, Color side) {
    int kingSq = (side == WHITE ? b.whiteKingSq : b.blackKingSq);
    if (kingSq < 0) return;
    int p = (side == WHITE ? WK : BK);

    int r0 = rankOf(kingSq), f0 = fileOf(kingSq);
    for (int d : kingDirs) {
        int to = kingSq + d;
        if (!onBoard(to)) continue;
        int r1 = rankOf(to), f1 = fileOf(to);
        if (abs(r1 - r0) > 1 || abs(f1 - f0) > 1) continue;
        int target = b.board[to];
        if (target == EMPTY || pieceColor(target) != side) {
            Move m;
            m.from = kingSq; m.to = to; m.piece = p;
            if (target != EMPTY) {
                m.captured = target;
                m.isCapture = true;
            }
            moves.push_back(m);
        }
    }

    // Castling
    if (side == WHITE) {
        if (b.whiteCastleKS) {
            // squares e1,f1,g1 empty; not in check on e1,f1,g1
            if (b.board[sq(4,0)] == WK &&
                b.board[sq(5,0)] == EMPTY &&
                b.board[sq(6,0)] == EMPTY &&
                !isSquareAttacked(b, sq(4,0), BLACK) &&
                !isSquareAttacked(b, sq(5,0), BLACK) &&
                !isSquareAttacked(b, sq(6,0), BLACK)) {
                Move m;
                m.from = sq(4,0); m.to = sq(6,0); m.piece = WK;
                m.isCastleKS = true;
                moves.push_back(m);
            }
        }
        if (b.whiteCastleQS) {
            if (b.board[sq(4,0)] == WK &&
                b.board[sq(3,0)] == EMPTY &&
                b.board[sq(2,0)] == EMPTY &&
                b.board[sq(1,0)] == EMPTY &&
                !isSquareAttacked(b, sq(4,0), BLACK) &&
                !isSquareAttacked(b, sq(3,0), BLACK) &&
                !isSquareAttacked(b, sq(2,0), BLACK)) {
                Move m;
                m.from = sq(4,0); m.to = sq(2,0); m.piece = WK;
                m.isCastleQS = true;
                moves.push_back(m);
            }
        }
    } else {
        if (b.blackCastleKS) {
            if (b.board[sq(4,7)] == BK &&
                b.board[sq(5,7)] == EMPTY &&
                b.board[sq(6,7)] == EMPTY &&
                !isSquareAttacked(b, sq(4,7), WHITE) &&
                !isSquareAttacked(b, sq(5,7), WHITE) &&
                !isSquareAttacked(b, sq(6,7), WHITE)) {
                Move m;
                m.from = sq(4,7); m.to = sq(6,7); m.piece = BK;
                m.isCastleKS = true;
                moves.push_back(m);
            }
        }
        if (b.blackCastleQS) {
            if (b.board[sq(4,7)] == BK &&
                b.board[sq(3,7)] == EMPTY &&
                b.board[sq(2,7)] == EMPTY &&
                b.board[sq(1,7)] == EMPTY &&
                !isSquareAttacked(b, sq(4,7), WHITE) &&
                !isSquareAttacked(b, sq(3,7), WHITE) &&
                !isSquareAttacked(b, sq(2,7), WHITE)) {
                Move m;
                m.from = sq(4,7); m.to = sq(2,7); m.piece = BK;
                m.isCastleQS = true;
                moves.push_back(m);
            }
        }
    }
}

void genPseudoLegalMoves(const Board &b, vector<Move> &moves) {
    Color side = b.sideToMove;
    genPawnMoves(b, moves, side);
    genKnightMoves(b, moves, side);
    int bishops[2] = { (side==WHITE?WB:BB), (side==WHITE?WQ:BQ) };
    int rooks[2]   = { (side==WHITE?WR:BR), (side==WHITE?WQ:BQ) };
    genSlidingMoves(b, moves, side, bishopDirs, 4, bishops, 2);
    genSlidingMoves(b, moves, side, rookDirs, 4, rooks, 2);
    genKingMoves(b, moves, side);
}

// -------------------------------------------------------------
// Attack detection implementation
// -------------------------------------------------------------
bool isSquareAttacked(const Board &b, int sqr, Color bySide) {
    // Pawns
    int dir = (bySide == WHITE ? 1 : -1);
    int r = rankOf(sqr), f = fileOf(sqr);
    for (int df : {-1,1}) {
        int nf = f + df;
        int nr = r - dir; // because if attacker pawn is on (nr, nf) it attacks sqr
        if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
        int s = sq(nf, nr);
        int p = b.board[s];
        if (bySide == WHITE && p == WP) return true;
        if (bySide == BLACK && p == BP) return true;
    }

    // Knights
    for (int d : knightDirs) {
        int to = sqr + d;
        if (!onBoard(to)) continue;
        int rf0 = rankOf(sqr), ff0 = fileOf(sqr);
        int rf1 = rankOf(to), ff1 = fileOf(to);
        if (max(abs(rf1 - rf0), abs(ff1 - ff0)) != 2 ||
            abs(rf1 - rf0) == 0 || abs(ff1 - ff0) == 0) continue;
        int p = b.board[to];
        if (bySide == WHITE && p == WN) return true;
        if (bySide == BLACK && p == BN) return true;
    }

    // Bishops / Queens (diagonals)
    for (int d : bishopDirs) {
        int to = sqr;
        while (true) {
            int r0 = rankOf(to), f0 = fileOf(to);
            to += d;
            if (!onBoard(to)) break;
            int r1 = rankOf(to), f1 = fileOf(to);
            if (abs(r1 - r0) > 1 || abs(f1 - f0) > 1) break;
            int p = b.board[to];
            if (p != EMPTY) {
                if (bySide == WHITE && (p == WB || p == WQ)) return true;
                if (bySide == BLACK && (p == BB || p == BQ)) return true;
                break;
            }
        }
    }

    // Rooks / Queens (straight)
    for (int d : rookDirs) {
        int to = sqr;
        while (true) {
            int r0 = rankOf(to), f0 = fileOf(to);
            to += d;
            if (!onBoard(to)) break;
            int r1 = rankOf(to), f1 = fileOf(to);
            if (abs(r1 - r0) > 1 || abs(f1 - f0) > 1) break;
            int p = b.board[to];
            if (p != EMPTY) {
                if (bySide == WHITE && (p == WR || p == WQ)) return true;
                if (bySide == BLACK && (p == BR || p == BQ)) return true;
                break;
            }
        }
    }

    // King
    for (int d : kingDirs) {
        int to = sqr + d;
        if (!onBoard(to)) continue;
        int r0 = rankOf(sqr), f0 = fileOf(sqr);
        int r1 = rankOf(to), f1 = fileOf(to);
        if (abs(r1 - r0) > 1 || abs(f1 - f0) > 1) continue;
        int p = b.board[to];
        if (bySide == WHITE && p == WK) return true;
        if (bySide == BLACK && p == BK) return true;
    }
    return false;
}

// -------------------------------------------------------------
// Make / unmake move
// -------------------------------------------------------------
struct Undo {
    int captured;
    int epSquare;
    bool wcks, wcqs, bcks, bcqs;
    int halfmoveClock;
    int fullmoveNumber;
    int whiteKingSq, blackKingSq;
};

void makeMove(Board &b, const Move &m, Undo &u) {
    u.captured = m.captured;
    u.epSquare = b.epSquare;
    u.wcks = b.whiteCastleKS; u.wcqs = b.whiteCastleQS;
    u.bcks = b.blackCastleKS; u.bcqs = b.blackCastleQS;
    u.halfmoveClock = b.halfmoveClock;
    u.fullmoveNumber = b.fullmoveNumber;
    u.whiteKingSq = b.whiteKingSq;
    u.blackKingSq = b.blackKingSq;

    // halfmove clock
    if (m.piece == WP || m.piece == BP || m.isCapture) b.halfmoveClock = 0;
    else b.halfmoveClock++;

    if (b.sideToMove == BLACK) b.fullmoveNumber++;

    b.epSquare = -1;

    int from = m.from, to = m.to;
    int piece = m.piece;

    // Move piece
    b.board[from] = EMPTY;

    // En passant capture removal
    if (m.isEnPassant) {
        int capSq = to + (b.sideToMove == WHITE ? -8 : 8);
        b.board[capSq] = EMPTY;
    }

    // Castling rook move
    if (m.isCastleKS) {
        if (b.sideToMove == WHITE) {
            b.board[sq(7,0)] = EMPTY;
            b.board[sq(5,0)] = WR;
        } else {
            b.board[sq(7,7)] = EMPTY;
            b.board[sq(5,7)] = BR;
        }
    } else if (m.isCastleQS) {
        if (b.sideToMove == WHITE) {
            b.board[sq(0,0)] = EMPTY;
            b.board[sq(3,0)] = WR;
        } else {
            b.board[sq(0,7)] = EMPTY;
            b.board[sq(3,7)] = BR;
        }
    }

    // Promotion or normal placement
    int placedPiece = piece;
    if (m.isPromotion) placedPiece = m.promoPiece;
    b.board[to] = placedPiece;

    // Update king squares
    if (piece == WK || placedPiece == WK) b.whiteKingSq = to;
    if (piece == BK || placedPiece == BK) b.blackKingSq = to;

    // Update castling rights
    if (piece == WK) { b.whiteCastleKS = b.whiteCastleQS = false; }
    if (piece == BK) { b.blackCastleKS = b.blackCastleQS = false; }
    if (from == sq(0,0) || to == sq(0,0)) b.whiteCastleQS = false;
    if (from == sq(7,0) || to == sq(7,0)) b.whiteCastleKS = false;
    if (from == sq(0,7) || to == sq(0,7)) b.blackCastleQS = false;
    if (from == sq(7,7) || to == sq(7,7)) b.blackCastleKS = false;

    // Set epSquare if double pawn push
    if (piece == WP && rankOf(from) == 1 && rankOf(to) == 3) {
        b.epSquare = from + 8;
    } else if (piece == BP && rankOf(from) == 6 && rankOf(to) == 4) {
        b.epSquare = from - 8;
    }

    // Switch side
    b.sideToMove = (b.sideToMove == WHITE ? BLACK : WHITE);
}

void unmakeMove(Board &b, const Move &m, const Undo &u) {
    b.sideToMove = (b.sideToMove == WHITE ? BLACK : WHITE);

    int from = m.from, to = m.to;
    int piece = m.piece;

    // Undo castling rook moves
    if (m.isCastleKS) {
        if (b.sideToMove == WHITE) {
            b.board[sq(7,0)] = WR;
            b.board[sq(5,0)] = EMPTY;
        } else {
            b.board[sq(7,7)] = BR;
            b.board[sq(5,7)] = EMPTY;
        }
    } else if (m.isCastleQS) {
        if (b.sideToMove == WHITE) {
            b.board[sq(0,0)] = WR;
            b.board[sq(3,0)] = EMPTY;
        } else {
            b.board[sq(0,7)] = BR;
            b.board[sq(3,7)] = EMPTY;
        }
    }

    // Remove piece from target
    b.board[to] = EMPTY;

    // Restore captured piece
    if (m.isEnPassant) {
        int capSq = to + (b.sideToMove == WHITE ? -8 : 8);
        b.board[capSq] = u.captured;
    } else if (m.isCapture) {
        b.board[to] = u.captured;
    }

    // Put moving piece back
    int restorePiece = piece;
    b.board[from] = restorePiece;

    // Restore state
    b.epSquare = u.epSquare;
    b.whiteCastleKS = u.wcks; b.whiteCastleQS = u.wcqs;
    b.blackCastleKS = u.bcks; b.blackCastleQS = u.bcqs;
    b.halfmoveClock = u.halfmoveClock;
    b.fullmoveNumber = u.fullmoveNumber;
    b.whiteKingSq = u.whiteKingSq;
    b.blackKingSq = u.blackKingSq;
}

// -------------------------------------------------------------
// Legal move filtering
// -------------------------------------------------------------
void genLegalMoves(const Board &b, vector<Move> &legalMoves) {
    vector<Move> pseudo;
    genPseudoLegalMoves(b, pseudo);
    legalMoves.clear();
    legalMoves.reserve(pseudo.size());

    for (const Move &m : pseudo) {
        Board bb = b;
        Undo u;
        makeMove(bb, m, u);
        int kingSq = (b.sideToMove == WHITE ? bb.whiteKingSq : bb.blackKingSq);
        if (!isSquareAttacked(bb, kingSq, (Color)(1 - b.sideToMove))) {
            legalMoves.push_back(m);
        }
    }
}

// -------------------------------------------------------------
// Evaluation (simple material + piece-square)
// -------------------------------------------------------------
const int pieceValue[13] = {
    0,
    100, 320, 330, 500, 900, 20000,  // white
    100, 320, 330, 500, 900, 20000   // black
};

int evaluate(const Board &b) {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        int p = b.board[i];
        if (p == EMPTY) continue;
        int val = pieceValue[p];
        if (p >= WP && p <= WK) score += val;
        else score -= val;
    }
    return (b.sideToMove == WHITE ? score : -score);
}

// -------------------------------------------------------------
// Alpha-beta search with iterative deepening
// -------------------------------------------------------------
int quiescence(Board &b, int alpha, int beta) {
    int standPat = evaluate(b);
    if (standPat >= beta) return beta;
    if (standPat > alpha) alpha = standPat;

    // generate captures only
    vector<Move> moves;
    genPseudoLegalMoves(b, moves);
    // filter to captures
    vector<Move> caps;
    for (auto &m: moves) if (m.isCapture) caps.push_back(m);

    for (const Move &m : caps) {
        if (isTimeUp()) break;
        Undo u;
        makeMove(b, m, u);
        int kingSq = (b.sideToMove == WHITE ? b.blackKingSq : b.whiteKingSq);
        // ensure legality: previous side's king not in check (already ensured earlier)
        int score = -quiescence(b, -beta, -alpha);
        unmakeMove(b, m, u);
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }

    return alpha;
}

int alphabeta(Board &b, int depth, int alpha, int beta, Move &bestMoveOut) {
    if (depth == 0) {
        return quiescence(b, alpha, beta);
    }
    if (isTimeUp()) return evaluate(b);

    vector<Move> moves;
    genLegalMoves(b, moves);

    if (moves.empty()) {
        // checkmate or stalemate
        Color us = b.sideToMove;
        int kingSq = (us == WHITE ? b.whiteKingSq : b.blackKingSq);
        if (isSquareAttacked(b, kingSq, (Color)(1 - us))) {
            // checkmate
            return -1000000 + (5 - depth);
        } else {
            // stalemate
            return 0;
        }
    }

    // simple move ordering: captures first
    sort(moves.begin(), moves.end(), [](const Move &a, const Move &b){
        if (a.isCapture != b.isCapture) return a.isCapture > b.isCapture;
        return false;
    });

    int bestScore = -10000000;
    Move localBest = moves[0];

    for (const Move &m : moves) {
        if (isTimeUp()) break;
        Undo u;
        makeMove(b, m, u);
        Move replyBest;
        int score = -alphabeta(b, depth - 1, -beta, -alpha, replyBest);
        unmakeMove(b, m, u);

        if (score > bestScore) {
            bestScore = score;
            localBest = m;
        }
        if (score > alpha) {
            alpha = score;
            if (alpha >= beta) break; // beta cutoff
        }
    }

    bestMoveOut = localBest;
    return bestScore;
}

Move iterativeDeepening(Board &b, double timeLimitSeconds) {
    endTime = chrono::steady_clock::now() + chrono::duration<double>(timeLimitSeconds * 0.95);
    timeUp.store(false);

    Move bestMove;
    vector<Move> rootMoves;
    genLegalMoves(b, rootMoves);
    if (!rootMoves.empty()) bestMove = rootMoves[0];

    for (int depth = 1; depth <= 64; ++depth) {
        if (isTimeUp()) break;
        Move mv;
        Board copy = b;
        alphabeta(copy, depth, -10000000, 10000000, mv);
        if (!isTimeUp()) {
            bestMove = mv;
        } else {
            break;
        }
    }
    return bestMove;
}

// -------------------------------------------------------------
// Move to notation
// -------------------------------------------------------------
string squareToStr(int sqr) {
    string s;
    s += char('a' + fileOf(sqr));
    s += char('1' + rankOf(sqr));
    return s;
}

char pieceChar(int p) {
    switch (p) {
        case WP: return 'p';
        case BP: return 'p';
        case WN: return 'N';
        case BN: return 'N';
        case WB: return 'B';
        case BB: return 'B';
        case WR: return 'R';
        case BR: return 'R';
        case WQ: return 'Q';
        case BQ: return 'Q';
        case WK: return 'K';
        case BK: return 'K';
        default: return 'p';
    }
}

string moveToNotation(const Board &b, const Move &m) {
    // Special case castling
    if (m.isCastleKS) return "O-O";
    if (m.isCastleQS) return "O-O-O";

    string s;
    char pc = pieceChar(m.piece);
    s += pc;
    s += squareToStr(m.from);
    if (m.isCapture) s += "x";
    s += squareToStr(m.to);
    if (m.isEnPassant) s += "ep";
    if (m.isPromotion) {
        s += "/";
        s += pieceChar(m.promoPiece);
    }

    // Determine check or mate
    Board bb = b;
    Undo u;
    makeMove(bb, m, u);
    vector<Move> replies;
    genLegalMoves(bb, replies);
    int oppKingSq = (bb.sideToMove == WHITE ? bb.whiteKingSq : bb.blackKingSq);
    bool inCheck = isSquareAttacked(bb, oppKingSq, (Color)(1 - bb.sideToMove));
    bool hasLegal = !replies.empty();
    unmakeMove(bb, m, u);

    if (inCheck) {
        if (!hasLegal) s += "#";
        else s += "+";
    }

    return s;
}

    genLegalMoves(b, legal);
    if (legal.empty()) {
        // no legal moves: output any valid-looking move (protocol not defined for this)
        cout << "O-O\n";
        return 0;
    }

    Move best = iterativeDeepening(b, timeout);
    cout << moveToNotation(b, best) << "\n";
    return 0;
}
