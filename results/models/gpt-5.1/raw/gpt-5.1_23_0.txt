```rust
// Simple real-time Lunar Lander controller.
// Strategy:
// - Parse initial map quickly, then stream state lines.
// - Use a dedicated stdin-reading thread that parses each state update
//   and stores the latest state in an atomic slot.
// - Main thread runs a fixed 50 Hz control loop (20+ Hz required),
//   reading the latest state and outputting commands.
//
// Controller logic (very simple, robust-ish):
// - While high above target: keep roughly over target X, manage vertical speed.
// - Near ground: aim almost straight up and reduce vertical velocity.
// - Use PID-like proportional control on angle and vertical speed.
// This is NOT optimal but should be safe and stable in many cases.

use std::f64::consts::PI;
use std::io::{self, Read};
use std::sync::{
    atomic::{AtomicBool, Ordering},
    Arc, Mutex,
};
use std::thread;
use std::time::{Duration, Instant};

// Shared state from reader to controller
#[derive(Clone, Copy, Debug)]
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
    valid: bool,
}

fn main() {
    // Read initial configuration and map in a blocking way once.
    // We must be careful to not slurp arbitrary GB maps into memory.
    // However, we *must* read all map rows, but can immediately discard them
    // after reading each line, since we only use target coordinates.
    let mut input = String::new();
    let mut stdin = io::stdin();

    // Read the first two lines fully (sizes + start/target)
    // To simplify, read line-by-line using read_line.
    use std::io::BufRead;
    let mut reader = io::BufReader::new(stdin);

    let mut line = String::new();

    // Line 1: width height gravity max_thrust fuel
    line.clear();
    if reader.read_line(&mut line).unwrap_or(0) == 0 {
        return;
    }
    let mut it = line.split_whitespace();
    let width: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let height: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let gravity: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let max_thrust: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let _fuel_total: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);

    // Line 2: start_x start_y target_x target_y
    line.clear();
    if reader.read_line(&mut line).unwrap_or(0) == 0 {
        return;
    }
    let mut it = line.split_whitespace();
    let _start_x: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let _start_y: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let target_x: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);
    let target_y: f64 = it.next().unwrap_or("0").parse().unwrap_or(0.0);

    // Read map rows and discard (to avoid huge memory usage).
    // Map rows: each is a line of characters.
    for _ in 0..height {
        line.clear();
        if reader.read_line(&mut line).unwrap_or(0) == 0 {
            break;
        }
        // We could parse obstacles here if needed; for now we only
        // ensure we drain the input for the map.
    }

    // At this point, 'reader' is positioned at the start of the
    // continuous state updates.

    // Shared state between reader thread and control loop
    let state = Arc::new(Mutex::new(LanderState {
        x: 0.0,
        y: 0.0,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel_remaining: 0.0,
        valid: false,
    }));
    let running = Arc::new(AtomicBool::new(true));

    // Spawn reader thread: constantly parse numbers from stdin.
    // We re-use the existing BufRead 'reader' by moving it into the thread.
    let state_reader = Arc::clone(&state);
    let running_reader = Arc::clone(&running);

    // NOTE: We must move 'reader' into the thread; to do that, we need to
    // reconstruct from stdin again, as we can’t move 'reader' easily
    // after partial use. To keep things simple, we now switch to raw
    // Read for the remaining data, since we've consumed the text map.
    // We'll parse whitespace-separated floats from here on.

    // Extract the underlying stdin handle again
    // SAFETY: We create a new reader from stdin; previously 'reader'
    // was only used for initial lines & map.
    let mut raw_stdin = io::stdin();

    thread::spawn(move || {
        // We'll read into a buffer and parse floats with a minimal tokenizer.
        // To keep it straightforward and robust, use a small internal buffer
        // and accumulate into a string, then split_whitespace.
        let mut buf = [0u8; 8192];
        let mut leftover = String::new();

        // We expect updates in blocks of 7 floats: x y vx vy angle angular_vel fuel_remaining
        loop {
            if !running_reader.load(Ordering::Relaxed) {
                break;
            }

            let n = match raw_stdin.read(&mut buf) {
                Ok(0) => break, // EOF
                Ok(n) => n,
                Err(_) => break,
            };

            leftover.push_str(std::str::from_utf8(&buf[..n]).unwrap_or(""));

            // Try to parse as many complete states as possible
            let mut parts = leftover.split_whitespace().peekable();
            let mut consumed = 0usize;
            let mut count = 0usize;

            while let (Some(_), Some(_), Some(_), Some(_), Some(_), Some(_), Some(_)) = (
                parts.peek(),
                { parts.nth(0) },
                parts.nth(0),
                parts.nth(0),
                parts.nth(0),
                parts.nth(0),
                parts.nth(0),
            ) {
                // The peek consumes one; nth(0) consumes one each time.
                // But this is messy to track consumed bytes, so instead
                // we'll re-implement a simpler parser below.
                count += 1;
                break;
            }

            // The above attempt is cumbersome. Instead, implement a more direct tokenizer:
            // Re-split leftover and try to parse chunks of 7 tokens, but retain unused tail.

            let tokens: Vec<&str> = leftover.split_whitespace().collect();
            let num_states = tokens.len() / 7;
            if num_states == 0 {
                continue;
            }

            for i in 0..num_states {
                let base = i * 7;
                let parse = |s: &str| s.parse::<f64>().unwrap_or(0.0);
                let sx = parse(tokens[base]);
                let sy = parse(tokens[base + 1]);
                let svx = parse(tokens[base + 2]);
                let svy = parse(tokens[base + 3]);
                let sangle = parse(tokens[base + 4]);
                let sangvel = parse(tokens[base + 5]);
                let sfuel = parse(tokens[base + 6]);

                {
                    let mut st = state_reader.lock().unwrap();
                    st.x = sx;
                    st.y = sy;
                    st.vx = svx;
                    st.vy = svy;
                    st.angle = sangle;
                    st.angular_vel = sangvel;
                    st.fuel_remaining = sfuel;
                    st.valid = true;
                }
            }

            // Rebuild leftover from any unused tokens (tail)
            let tail_start = num_states * 7;
            if tail_start < tokens.len() {
                let mut new_leftover = String::new();
                for i in tail_start..tokens.len() {
                    new_leftover.push_str(tokens[i]);
                    new_leftover.push(' ');
                }
                leftover = new_leftover;
            } else {
                leftover.clear();
            }
        }
    });

    // Controller loop at ~50 Hz
    let mut stdout = io::BufWriter::new(io::stdout());
    let tick = Duration::from_millis(20); // 50 Hz
    let mut last = Instant::now();

    loop {
        if !running.load(Ordering::Relaxed) {
            break;
        }

        // Busy-wait / sleep hybrid to maintain timing
        let now = Instant::now();
        if let Some(remaining) = tick.checked_sub(now.duration_since(last)) {
            // Sleep for most of the remaining duration
            if remaining > Duration::from_micros(200) {
                thread::sleep(remaining - Duration::from_micros(200));
            }
            // Brief spin to get closer to exact tick (optional)
            while last + tick > Instant::now() {}
        }
        last = Instant::now();

        // Read latest state
        let st = {
            let st_guard = state.lock().unwrap();
            *st_guard
        };

        if !st.valid {
            // No state yet; send zero commands to be safe
            use std::io::Write;
            writeln!(stdout, "0 0").unwrap();
            stdout.flush().unwrap();
            continue;
        }

        // Basic controller

        // Height above target (y increases downward)
        let height = st.y - target_y;

        // Desired horizontal position: target_x
        let dx = target_x - st.x;

        // Angle: 0 is up, positive is clockwise. We want angle ~0 for vertical thrust.
        // We'll use small tilts to correct x when high, but get upright when close.
        let distance_to_target = (dx * dx + height * height).sqrt();

        // Vertical velocity limits become more strict near ground
        let safe_vy_far = 20.0;
        let safe_vy_near = 2.0;

        // We define a "near" threshold
        let near_threshold = 20.0;
        let vy_limit = if height > near_threshold {
            safe_vy_far
        } else {
            safe_vy_near
        };

        // Compute desired angle (tilt) to move horizontally toward target when high
        let mut desired_angle = 0.0;
        if height > 5.0 {
            // Proportional control to move over the target
            let kx = 0.02; // horizontal correction factor
            let vx_target = kx * dx; // desired horizontal velocity
            let kvx = 0.2;
            let vx_error = vx_target - st.vx;

            // Map vx_error to angle: small tilt, limited
            let angle_for_vx = -vx_error * kvx; // negative error => tilt left, etc.
            let max_tilt = 20.0_f64.to_radians();
            desired_angle = angle_for_vx.clamp(-max_tilt, max_tilt);
        } else {
            // Very close: stay upright
            desired_angle = 0.0;
        }

        // Angle control: proportional on angle error
        let mut angle_error = desired_angle - st.angle;
        // Normalize angle error to [-pi, pi] to avoid weird wrap behaviors
        while angle_error > PI {
            angle_error -= 2.0 * PI;
        }
        while angle_error < -PI {
            angle_error += 2.0 * PI;
        }
        let k_angle = 1.5; // rad/s per rad
        let desired_turn_rate = k_angle * angle_error;
        let turn = desired_turn_rate.clamp(-1.0, 1.0);

        // Vertical speed control:
        // We want vy small (negative upward, positive downward, but spec says + is downward).
        // Gravity is +g, thrust is opposite angle; but we approximate thrust as vertical
        // component only (we always keep angle small).
        let mut desired_vy = 0.0;
        if height > 5.0 {
            // When high, allow some downward speed proportional to height
            let max_down_speed = 10.0;
            desired_vy = (height * 0.1).clamp(0.0, max_down_speed);
        } else {
            // Near ground: very small downward speed
            desired_vy = 0.5;
        }

        let vy_error = desired_vy - st.vy;
        let kvy = 0.1;
        // Base thrust fraction to counter gravity: thrust = g / max_thrust
        let hover_thrust = if max_thrust > 0.0 {
            gravity / max_thrust
        } else {
            0.0
        };

        // PD-like control on vertical speed
        let thrust_correction = kvy * vy_error;
        let mut thrust = (hover_thrust + thrust_correction).clamp(0.0, 1.0);

        // Extra safety: if near ground and falling too fast, max thrust
        if height <= near_threshold && st.vy > vy_limit {
            thrust = 1.0;
        }

        // If almost landed: height small and vy small, cut thrust
        if height <= 1.0 && st.vy <= safe_vy_near && st.vy >= -safe_vy_near {
            thrust = 0.0;
        }

        // If out of fuel, no thrust
        if st.fuel_remaining <= 0.0 {
            thrust = 0.0;
        }

        // Simple fuel conservation: if high and slow, allow more free-fall
        if height > 50.0 && st.vy < safe_vy_far * 0.5 {
            thrust *= 0.5;
        }

        // Output commands
        use std::io::Write;
        writeln!(stdout, "{} {}", thrust, turn).unwrap();
        stdout.flush().unwrap();
    }
}
```
