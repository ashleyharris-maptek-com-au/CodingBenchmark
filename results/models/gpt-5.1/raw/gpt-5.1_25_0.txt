// Compile with: csc /optimize+ /unsafe+ AsteroidInterceptor.cs
// Target: .NET Framework compatible with csc in host description (no external deps).

using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class AsteroidInterceptor
{
    // ---------------------------- Constants ---------------------------------

    const double G = 6.67430e-11;              // m^3 kg^-1 s^-2
    const double DaySec = 86400.0;             // s
    const double Km = 1000.0;                  // m
    const double ImpactProximity = 1000.0 * Km; // 1000 km
    const double EarthRadius = 6371.0 * Km;

    // ---------------------------- Data Types --------------------------------

    class Body
    {
        public string Name;
        public double Mass;
        public double[] R = new double[3]; // m
        public double[] V = new double[3]; // m/s

        // Ephemeris sampled at 1-day steps in DE421 time span for quick lookup
        // In this simplified solver we compute ephemeris via N-body integration
        // rather than reading detailed DE421; the provided path is ignored.
    }

    class State
    {
        public double[] R = new double[3];
        public double[] V = new double[3];
    }

    // Spacecraft trajectory representation for simple patched-conic-like search
    class SpacecraftTrajectory
    {
        public double LaunchTimeDays;
        public double[] LaunchDV = new double[3];
        public List<(double tDays, double[] dv)> MidCourseBurns = new List<(double, double[])>();
        public double ImpactTimeDays;
        public double[] ImpactDV = new double[3];
        public double TotalDV;
        public bool Feasible;
        public double MissDistanceToAsteroid;   // at intercept
        public double PostDeflectionMissToEarth;
    }

    // ---------------------------- Vector Helpers ----------------------------

    static double[] Vec(double x, double y, double z) => new[] { x, y, z };

    static double[] Add(double[] a, double[] b) =>
        new[] { a[0] + b[0], a[1] + b[1], a[2] + b[2] };

    static double[] Sub(double[] a, double[] b) =>
        new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };

    static double[] Mul(double[] a, double s) =>
        new[] { a[0] * s, a[1] * s, a[2] * s };

    static double Dot(double[] a, double[] b) =>
        a[0] * b[0] + a[1] * b[1] + a[2] * b[2];

    static double Norm(double[] a) =>
        Math.Sqrt(Dot(a, a));

    static double[] Normalize(double[] a)
    {
        double n = Norm(a);
        if (n == 0) return Vec(0, 0, 0);
        return Mul(a, 1.0 / n);
    }

    // ---------------------------- Simple N-body Integrator ------------------

    // We implement a simple Kick-Drift-Kick (KDK) leapfrog integrator.
    // This is not as accurate as real DE421 but suffices for this solver.

    static void AccelNBody(List<Body> bodies, double[][] acc)
    {
        int n = bodies.Count;
        for (int i = 0; i < n; i++)
        {
            acc[i][0] = acc[i][1] = acc[i][2] = 0.0;
        }

        for (int i = 0; i < n; i++)
        {
            var bi = bodies[i];
            for (int j = i + 1; j < n; j++)
            {
                var bj = bodies[j];
                double[] rij = Sub(bj.R, bi.R);
                double r2 = Dot(rij, rij) + 1e-9; // simple softening
                double r = Math.Sqrt(r2);
                double invR3 = 1.0 / (r2 * r);
                double f = G * bi.Mass * bj.Mass * invR3;
                double ax = f * rij[0] / bi.Mass;
                double ay = f * rij[1] / bi.Mass;
                double az = f * rij[2] / bi.Mass;
                acc[i][0] += ax;
                acc[i][1] += ay;
                acc[i][2] += az;
                acc[j][0] -= ax; // equal and opposite
                acc[j][1] -= ay;
                acc[j][2] -= az;
            }
        }
    }

    static void IntegrateNBody(List<Body> bodies, double dt)
    {
        int n = bodies.Count;
        var acc = new double[n][];
        for (int i = 0; i < n; i++) acc[i] = new double[3];

        // Initial acceleration
        AccelNBody(bodies, acc);

        // Kick 1/2
        for (int i = 0; i < n; i++)
        {
            bodies[i].V[0] += 0.5 * dt * acc[i][0];
            bodies[i].V[1] += 0.5 * dt * acc[i][1];
            bodies[i].V[2] += 0.5 * dt * acc[i][2];
        }

        // Drift
        for (int i = 0; i < n; i++)
        {
            bodies[i].R[0] += dt * bodies[i].V[0];
            bodies[i].R[1] += dt * bodies[i].V[1];
            bodies[i].R[2] += dt * bodies[i].V[2];
        }

        // Acceleration at new positions
        AccelNBody(bodies, acc);

        // Kick 2
        for (int i = 0; i < n; i++)
        {
            bodies[i].V[0] += 0.5 * dt * acc[i][0];
            bodies[i].V[1] += 0.5 * dt * acc[i][1];
            bodies[i].V[2] += 0.5 * dt * acc[i][2];
        }
    }

    // ---------------------------- Ephemeris / Propagation -------------------

    // Clone bodies list (deep copy positions/velocities, same masses/names)
    static List<Body> CloneBodies(List<Body> src)
    {
        var list = new List<Body>(src.Count);
        foreach (var b in src)
        {
            var nb = new Body
            {
                Name = b.Name,
                Mass = b.Mass,
                R = new[] { b.R[0], b.R[1], b.R[2] },
                V = new[] { b.V[0], b.V[1], b.V[2] }
            };
            list.Add(nb);
        }
        return list;
    }

    // Propagate system from t=0 to t = targetDays, using stepDays
    static List<Body> PropagateTo(List<Body> initial, double targetDays, double stepDays)
    {
        double t = 0.0;
        double dtSec = stepDays * DaySec;
        var bodies = CloneBodies(initial);

        int steps = (int)Math.Floor(Math.Abs(targetDays) / stepDays);
        double remainingDays = Math.Abs(targetDays) - steps * stepDays;
        double sign = targetDays >= 0 ? 1.0 : -1.0;

        for (int i = 0; i < steps; i++)
        {
            IntegrateNBody(bodies, sign * dtSec);
            t += sign * stepDays;
        }

        if (remainingDays > 1e-6)
        {
            IntegrateNBody(bodies, sign * remainingDays * DaySec);
            t += sign * remainingDays;
        }

        return bodies;
    }

    // Get state of a named body at given time (days from t0) by propagating
    static State GetBodyStateAt(List<Body> initial, string name, double tDays)
    {
        var bodies = PropagateTo(initial, tDays, 0.5); // 0.5 day steps
        foreach (var b in bodies)
        {
            if (b.Name == name)
            {
                return new State { R = (double[])b.R.Clone(), V = (double[])b.V.Clone() };
            }
        }
        return null;
    }

    // ---------------------------- Simple Mission Logic ----------------------

    // We attempt a coarse search:
    // - Sample launch times within window
    // - For each, sample intercept times between launch and warning_days
    // - Assume impulsive burn at launch to match asteroid position at intercept
    // - Optionally 1 mid-course correction at t_mid (fraction between launch and intercept)
    // - Choose solution with lowest DV that intercepts asteroid and achieves small
    //   post-impact deviation relative to Earth.

    static SpacecraftTrajectory PlanMission(
        List<Body> initialBodies,
        double warningDays,
        double impactorMass,
        double dvBudget,
        double asteroidMass)
    {
        // Identify Earth and Asteroid
        var earth0 = initialBodies.FirstOrDefault(b => b.Name == "Earth");
        var asteroid0 = initialBodies.FirstOrDefault(b => b.Name == "Asteroid");
        if (earth0 == null || asteroid0 == null)
        {
            return new SpacecraftTrajectory { Feasible = false };
        }

        // Estimate asteroid impact time with Earth (very rough: when distance minima)
        double approxImpactTime = EstimateImpactTime(initialBodies, "Earth", "Asteroid", warningDays);

        // Upper search limit: min(warningDays, approxImpactTime)
        double maxInterceptTime = Math.Min(warningDays, approxImpactTime);

        // Sampling strategy (coarse but fast)
        int launchSamples = 10;
        int interceptSamples = 20;

        SpacecraftTrajectory best = null;

        for (int li = 0; li < launchSamples; li++)
        {
            double launchTime = warningDays * li / Math.Max(1, launchSamples - 1);

            // Earth state at launch
            State earthLaunch = GetBodyStateAt(initialBodies, "Earth", launchTime);
            if (earthLaunch == null) continue;

            for (int ii = 0; ii < interceptSamples; ii++)
            {
                double tIntercept = launchTime + (maxInterceptTime - launchTime) * (ii + 1) / (interceptSamples + 1);
                if (tIntercept <= launchTime) continue;

                // Asteroid state at intercept
                State astAtIntercept = GetBodyStateAt(initialBodies, "Asteroid", tIntercept);
                if (astAtIntercept == null) continue;

                // Earth at launch again (ensure consistency)
                earthLaunch = GetBodyStateAt(initialBodies, "Earth", launchTime);
                if (earthLaunch == null) continue;

                // Simple two-point Lambert-like approximation: straight-line transfer
                double dtSec = (tIntercept - launchTime) * DaySec;
                if (dtSec <= 0) continue;

                double[] r0 = earthLaunch.R;
                double[] r1 = astAtIntercept.R;

                double[] vReq = Mul(Sub(r1, r0), 1.0 / dtSec); // constant velocity needed
                double[] dvLaunch = Sub(vReq, earthLaunch.V);
                double dvLaunchMag = Norm(dvLaunch);

                if (dvLaunchMag > dvBudget) continue; // can't even launch

                // Assume no mid-course correction for baseline
                double totalDV = dvLaunchMag;

                // Check intercept distance by propagating relative motion with constant vReq
                double[] scRAtIntercept = Add(r0, Mul(vReq, dtSec));
                double missToAsteroid = Norm(Sub(scRAtIntercept, r1));

                if (missToAsteroid > ImpactProximity) continue; // not close enough

                // Relative velocity at impact:
                double[] astV = astAtIntercept.V;
                double[] relV = Sub(vReq, astV);
                double relSpeed = Norm(relV);
                if (relSpeed < 1.0) continue; // degenerate

                // Momentum transfer: one-dimensional along relative velocity
                // New asteroid velocity: v' = v + (m_imp/m_ast)*Δv_eff
                // Here Δv_eff ~ projection along line of centers; approximate full relV
                double mu = impactorMass / asteroidMass;
                double[] deltaVast = Mul(relV, mu); // very simplified

                // Estimate deflection at encounter with Earth:
                // propagate asteroid with modified v from tIntercept to approxImpactTime
                double deflectionMiss = EstimatePostImpactMissDistance(
                    initialBodies,
                    "Earth",
                    "Asteroid",
                    tIntercept,
                    approxImpactTime,
                    deltaVast);

                // Construct trajectory candidate
                var traj = new SpacecraftTrajectory
                {
                    LaunchTimeDays = launchTime,
                    LaunchDV = dvLaunch,
                    ImpactTimeDays = tIntercept,
                    ImpactDV = Vec(0, 0, 0), // impact is ballistic
                    TotalDV = totalDV,
                    Feasible = deflectionMiss > EarthRadius, // must at least miss Earth radius
                    MissDistanceToAsteroid = missToAsteroid,
                    PostDeflectionMissToEarth = deflectionMiss
                };

                if (!traj.Feasible) continue;

                if (best == null ||
                    traj.PostDeflectionMissToEarth > best.PostDeflectionMissToEarth + 10 * Km ||
                    (Math.Abs(traj.PostDeflectionMissToEarth - (best?.PostDeflectionMissToEarth ?? 0)) < 10 * Km
                     && traj.TotalDV < (best?.TotalDV ?? double.MaxValue)))
                {
                    best = traj;
                }
            }
        }

        // If we found something, optionally add a small mid-course correction placeholder
        if (best != null)
        {
            // Budget some 5% of DV as a generic mid-course correction
            double dvCorrMag = 0.05 * best.TotalDV;
            double tMid = (best.LaunchTimeDays + best.ImpactTimeDays) * 0.5;
            double[] dvCorr = Vec(0, 0, 0);
            if (dvCorrMag > 0)
            {
                // Direction: roughly toward asteroid at mid-time
                State astMid = GetBodyStateAt(initialBodies, "Asteroid", tMid);
                State earthMid = GetBodyStateAt(initialBodies, "Earth", tMid);
                if (astMid != null && earthMid != null)
                {
                    double[] dir = Normalize(Sub(astMid.R, earthMid.R));
                    dvCorr = Mul(dir, dvCorrMag);
                }
            }
            best.MidCourseBurns.Add((tMid, dvCorr));
            best.TotalDV += dvCorrMag;

            if (best.TotalDV > dvBudget)
            {
                // If we busted the budget, drop correction burn
                best.TotalDV -= dvCorrMag;
                best.MidCourseBurns.Clear();
            }

            return best;
        }

        return new SpacecraftTrajectory { Feasible = false };
    }

    static double EstimateImpactTime(
        List<Body> initialBodies,
        string earthName,
        string asteroidName,
        double maxDays)
    {
        // Very crude: sample distance every day and find minimum
        double bestT = maxDays;
        double bestD = double.MaxValue;
        int samples = (int)Math.Ceiling(maxDays);

        for (int i = 0; i <= samples; i++)
        {
            double t = i;
            var bodies = PropagateTo(initialBodies, t, 0.5);
            Body earth = null, asteroid = null;
            foreach (var b in bodies)
            {
                if (b.Name == earthName) earth = b;
                if (b.Name == asteroidName) asteroid = b;
            }
            if (earth == null || asteroid == null) continue;
            double d = Norm(Sub(earth.R, asteroid.R));
            if (d < bestD)
            {
                bestD = d;
                bestT = t;
            }
        }

        return bestT;
    }

    static double EstimatePostImpactMissDistance(
        List<Body> initialBodies,
        string earthName,
        string asteroidName,
        double impactTime,
        double approxEncounterTime,
        double[] deltaVast)
    {
        // Clone system and apply deltaV to asteroid at impactTime,
        // then propagate to approxEncounterTime and measure distance.
        var bodiesAtImpact = PropagateTo(initialBodies, impactTime, 0.5);
        Body earth = null, asteroid = null;
        foreach (var b in bodiesAtImpact)
        {
            if (b.Name == earthName) earth = b;
            if (b.Name == asteroidName) asteroid = b;
        }
        if (earth == null || asteroid == null) return 0.0;

        asteroid.V[0] += deltaVast[0];
        asteroid.V[1] += deltaVast[1];
        asteroid.V[2] += deltaVast[2];

        double dt = approxEncounterTime - impactTime;
        if (dt <= 0) dt = 1.0;

        var bodiesAtEncounter = PropagateTo(bodiesAtImpact, dt, 0.5);
        earth = null;
        asteroid = null;
        foreach (var b in bodiesAtEncounter)
        {
            if (b.Name == earthName) earth = b;
            if (b.Name == asteroidName) asteroid = b;
        }
        if (earth == null || asteroid == null) return 0.0;

        return Norm(Sub(earth.R, asteroid.R));
    }

    // ---------------------------- IO / Main ---------------------------------

    static double ReadDouble(string s)
    {
        return double.Parse(s, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);
    }

    public static void Main()
    {
        // Read all tokens from stdin
        var tokens = new List<string>();
        using (var sr = new StreamReader(Console.OpenStandardInput()))
        {
            string line;
            while ((line = sr.ReadLine()) != null)
            {
                if (line.Length == 0) continue;
                var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                tokens.AddRange(parts);
            }
        }

        if (tokens.Count < 6)
        {
            // Minimal default output: no burns
            Console.WriteLine("0");
            return;
        }

        int idx = 0;
        int numBodies = (int)ReadDouble(tokens[idx++]);
        double warningDays = ReadDouble(tokens[idx++]);
        double impactorMass = ReadDouble(tokens[idx++]);
        double dvBudget = ReadDouble(tokens[idx++]);
        double asteroidMass = ReadDouble(tokens[idx++]);
        double t0UnixUtc = ReadDouble(tokens[idx++]); // not used explicitly

        var bodies = new List<Body>();

        for (int i = 0; i < numBodies; i++)
        {
            if (idx + 8 >= tokens.Count) break;
            string name = tokens[idx++];
            double mass = ReadDouble(tokens[idx++]);
            double x = ReadDouble(tokens[idx++]);
            double y = ReadDouble(tokens[idx++]);
            double z = ReadDouble(tokens[idx++]);
            double vx = ReadDouble(tokens[idx++]);
            double vy = ReadDouble(tokens[idx++]);
            double vz = ReadDouble(tokens[idx++]);
            var b = new Body
            {
                Name = name,
                Mass = mass,
                R = Vec(x, y, z),
                V = Vec(vx, vy, vz)
            };
            bodies.Add(b);
        }

        // Remaining token should be DE421 path (ignored in this approximate solver)
        // string de421Path = idx < tokens.Count ? tokens[idx] : null;

        var traj = PlanMission(bodies, warningDays, impactorMass, dvBudget, asteroidMass);

        if (traj == null || !traj.Feasible)
        {
            // Fallback: output no burns
            Console.WriteLine("0");
            return;
        }

        // Burns: launch + mid-course burns + "impact burn" (zero) as format requires.
        int numBurns = 1 + traj.MidCourseBurns.Count + 1;
        Console.WriteLine(numBurns.ToString(CultureInfo.InvariantCulture));

        // 1. Launch burn
        PrintBurn(traj.LaunchTimeDays, traj.LaunchDV);

        // 2. Mid-course burns (if any)
        foreach (var (t, dv) in traj.MidCourseBurns)
        {
            PrintBurn(t, dv);
        }

        // 3. Impact "burn" (usually zero DV)
        PrintBurn(traj.ImpactTimeDays, traj.ImpactDV);
    }

    static void PrintBurn(double tDays, double[] dv)
    {
        Console.WriteLine(
            string.Format(CultureInfo.InvariantCulture,
                "{0} {1} {2} {3}",
                tDays,
                dv[0],
                dv[1],
                dv[2]));
    }
}
