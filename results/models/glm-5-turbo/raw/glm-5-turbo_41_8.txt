; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpDecorate %ubo_0 Offset 0
OpDecorate %ubo_1 Offset 64
OpDecorate %ubo_2 Offset 128
OpDecorate %ubo_3 Offset 192
OpDecorate %ubo_4 Offset 208
OpDecorate %ubo_5 Offset 224

; Types and Constants
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v2 = OpTypePointer Input %v2float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float

%c_0 = OpConstant %float 0
%c_0_75 = OpConstant %float 0.75
%c_0_5 = OpConstant %float 0.5
%c_0_25 = OpConstant %float 0.25
%c_1 = OpConstant %float 1
%c_0_7 = OpConstant %float 0.7
%c_0_4 = OpConstant %float 0.4
%c_0_2 = OpConstant %float 0.2
%c_1_0 = OpConstant %float 1.0
%c_vec4_alpha = OpConstantComposite %v4float %c_1_0 %c_1_0 %c_1_0 %c_1_0
%c_vec3_0 = OpConstantComposite %v3float %c_0 %c_0 %c_0
%c_bool_true = OpConstantTrue %bool
%bool = OpTypeBool
%v2float = OpTypeVector %float 2

; Variables
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%uv = OpVariable %ptr_in_v2 Input
%tangent = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input
%outColor = OpVariable %ptr_out_v4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Main Function
%main = OpFunction %void None %func
%entry = OpLabel

; Load Normal and normalize
%n = OpLoad %v3float %normal
%N = OpExtInst %v3float %glsl450 Normalize %n

; Load light position
%ptr_light = OpAccessChain %ptr_uniform_ubo %ubo %c_0
%light4 = OpLoad %v4float %ptr_light
%L3 = OpVectorShuffle %v3float %light4 %light4 0 1 2

; Calculate L vector
%P = OpLoad %v3float %worldPos
%L_vec = OpFSub %v3float %L3 %P
%L = OpExtInst %v3float %glsl450 Normalize %L_vec

; Calculate diffuse term
%dot_nl = OpDot %float %N %L
%d = OpExtInst %float %glsl450 FMax %dot_nl %c_0

; Quantize to bands
%cmp1 = OpFOrdGreaterThan %bool %d %c_0_75
%band = OpSelect %float %cmp1 %c_1 %c_0_4
%cmp2 = OpFOrdGreaterThan %bool %d %c_0_5
%band2 = OpSelect %float %cmp2 %c_0_7 %band
%cmp3 = OpFOrdGreaterThan %bool %d %c_0_25
%final_band = OpSelect %float %cmp3 %band2 %c_0_2

; Load vertex color
%v_color = OpLoad %v3float %color

; Multiply color by band
%scaled_color = OpVectorTimesScalar %v3float %v_color %final_band

; Output RGBA with alpha 1.0
%final_color = OpVectorShuffle %v4float %scaled_color %c_vec4_alpha 0 1 2 3
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
