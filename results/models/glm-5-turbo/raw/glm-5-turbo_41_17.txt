               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450
               OpName %main "main"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %uv "uv"
               OpName %tangent "tangent"
               OpName %color "color"
               OpName %outColor "outColor"
               OpName %UBO "UBO"
               OpName %ubo "ubo"
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
      %void = OpTypeVoid
          %9 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
     %v2float = OpTypeVector %float 2
%ptr_Output_v4float = OpTypePointer Output %v4float
       %UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
  %worldPos = OpVariable %ptr_Input_v3float Input
    %normal = OpVariable %ptr_Input_v3float Input
        %uv = OpVariable %ptr_Input_v2float Input
    %tangent = OpVariable %ptr_Input_v3float Input
     %color = OpVariable %ptr_Input_v3float Input
  %outColor = OpVariable %ptr_Output_v4float Output
       %ubo = OpVariable %ptr_Uniform_UBO Uniform
        %fp = OpTypePointer Uniform %float
       %int = OpTypeInt 32 1
      %int3 = OpConstant %int 3
      %int4 = OpConstant %int 4
   %float_0 = OpConstant %float 0
     %bool = OpTypeBool
    %float_1 = OpConstant %float 1
     %v3one = OpConstantComposite %v3float %float_1 %float_1 %float_1
  %float_8 = OpConstant %float 8
 %float_16 = OpConstant %float 16
%float_0_05 = OpConstant %float 0.05
 %float_0_1 = OpConstant %float 0.1
%float_0_7 = OpConstant %float 0.7
%float_0_2 = OpConstant %float 0.2
%float_0_1b = OpConstant %float 0.1
%v3mortar = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7
%v3brick = OpConstantComposite %v3float %float_0_7 %float_0_2 %float_0_1b
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
       %main = OpFunction %void None %9
         %50 = OpLabel
      %uv_v = OpLoad %v2float %uv
       %su = OpVectorTimesScalar %v2float %uv_v %float_8
       %suy = OpCompositeExtract %float %su 1
       %row = OpFloor %float %suy
       %half = OpFMod %float %row %float_1
  %cmp_half = OpFOrdGreaterThan %float %half %float_0_5
       %off = OpSelect %float %cmp_half %float_1 %float_0
     %suxm = OpCompositeExtract %float %su 0
        %su = OpFAdd %float %suxm %off
       %fx = OpFract %float %su
       %fy = OpFract %float %suy
  %mx_cmp = OpFOrdLessThan %float %fx %float_0_05
  %my_cmp = OpFOrdLessThan %float %fy %float_0_1
     %m_or = OpLogicalOr %bool %mx_cmp %my_cmp
     %bcol = OpSelect %v3float %m_or %v3mortar %v3brick
       %N = OpLoad %v3float %normal
       %Nn = OpExtInst %v3float %1 Normalize %N
   %l_ptr = OpAccessChain %fp %ubo %int3
        %l = OpLoad %float %l_ptr
   %lx_ptr = OpAccessChain %fp %ubo %int3 %int3
       %lx = OpLoad %float %lx_ptr
   %ly_ptr = OpAccessChain %fp %ubo %int3 %int4
       %ly = OpLoad %float %ly_ptr
 %lightPos = OpCompositeConstruct %v3float %l %lx %ly
       %L = OpExtInst %v3float %1 Normalize %lightPos
    %ndotl = OpDot %float %Nn %L
      %diff = OpFClamp %float %ndotl %float_0 %float_1
       %sf = OpFMul %float %float_0_6 %diff
        %a = OpFAdd %float %float_0_4 %sf
      %scal = OpCompositeConstruct %v3float %a %a %a
     %lit = OpVectorTimesScalar %v3float %bcol %a
   %out_rgb = OpCompositeConstruct %v4float %lit %float_1
               OpStore %outColor %out_rgb
               OpReturn
               OpFunctionEnd
