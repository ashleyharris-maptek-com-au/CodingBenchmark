; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Block
OpDecorate %ubo_struct MemberOffset 0
OpDecorate %ubo_struct MemberOffset 64
OpDecorate %ubo_struct MemberOffset 128
OpDecorate %ubo_struct MemberOffset 192
OpDecorate %ubo_struct MemberOffset 208
OpDecorate %ubo_struct MemberOffset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types and Constants
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%c_16 = OpConstant %float 16.0
%c_pi = OpConstant %float 3.14159
%c_0_0 = OpConstant %float 0.0
%c_0_5 = OpConstant %float 0.5
%c_0_4 = OpConstant %float 0.4
%c_0_6 = OpConstant %float 0.6
%c_1_0 = OpConstant %float 1.0
%c_0_1 = OpConstant %float 0.1
%c_0_9 = OpConstant %float 0.9
%c_red = OpConstantComposite %vec3 %c_0_9 %c_0_1 %c_0_1
%c_blue = OpConstantComposite %vec3 %c_0_1 %c_0_1 %c_0_9
%c_one_vec4 = OpConstantComposite %vec4 %c_1_0 %c_1_0 %c_1_0 %c_1_0
%c_zero_vec3 = OpConstantComposite %vec3 %c_0_0 %c_0_0 %c_0_0
%c_i32_4 = OpConstant %int 4

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_ubo Uniform

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; Load inputs
%uv_val = OpLoad %vec2 %uv
%normal_val = OpLoad %vec3 %normal

; Compute sin(uv.y * 3.14159 * 16.0)
%uv_y = OpCompositeExtract %float %uv_val 1
%mul1 = OpFMul %float %uv_y %c_pi
%mul2 = OpFMul %float %mul1 %c_16
%sin_val = OpSin %float %mul2

; Determine stripe color: sin > 0.0 -> Red, else -> Blue
%is_pos = OpFOrdGreaterThan %bool %sin_val %c_0_0
%stripe_color = OpSelect %vec3 %is_pos %c_red %c_blue

; Calculate Lambertian shading
%N = OpNormalize %vec3 %normal_val
%light_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %c_i32_4
%light_pos = OpLoad %vec4 %light_ptr
%L_vec3 = OpVectorShuffle %vec3 %light_pos %light_pos 0 1 2
%L = OpNormalize %vec3 %L_vec3

%dot_NL = OpDot %float %N %L
%diffuse = OpFClamp %float %dot_NL %c_0_0 %c_1_0
%factor = OpFMul %float %c_0_6 %diffuse
%factor_plus = OpFAdd %float %c_0_4 %factor
%factor_vec3 = OpCompositeConstruct %vec3 %factor_plus %factor_plus %factor_plus

%final_rgb = OpFMul %vec3 %stripe_color %factor_vec3
%final_color = OpVectorShuffle %vec4 %final_rgb %c_one_vec4 0 1 2 4

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
