; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types and Constants
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%func_type = OpTypeFunction %void
%input_ptr_v3 = OpTypePointer Input %v3float
%input_ptr_v2 = OpTypePointer Input %v3float
%output_ptr_v4 = OpTypePointer Output %v4float
%uniform_ptr_ubo = OpTypePointer Uniform %UBO
%ubo_type = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%uniform_ptr_v4 = OpTypePointer Uniform %v4float
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1.0
%sky_color = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1_0
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_15 = OpConstant %float 0.15
%float_0_05 = OpConstant %float 0.05
%ground_color = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05
%zero_vec3 = OpConstantNull %v3float

; Variables
%worldPos = OpVariable %input_ptr_v3 Input
%normal = OpVariable %input_ptr_v3 Input
%uv = OpVariable %input_ptr_v3 Input
%tangent = OpVariable %input_ptr_v3 Input
%color = OpVariable %input_ptr_v3 Input
%outColor = OpVariable %output_ptr_v4 Output
%ubo = OpVariable %uniform_ptr_ubo Uniform

; Main Function
%main = OpFunction %void None %func_type
%label = OpLabel

; Load and normalize normal
%n_raw = OpLoad %v3float %normal
%n_len2 = OpDot %v3float %n_raw %n_raw
%n_len = OpSqrt %float %n_len2
%n = OpCompositeConstruct %v3float %n_len %n_len %n_len
%n_norm = OpFDiv %v3float %n_raw %n

; Calculate blend factor: N.y * 0.5 + 0.5
%n_y = OpCompositeExtract %float %n_norm 1
%n_y_half = OpFMul %float %n_y %float_0_5
%blend = OpFAdd %float %n_y_half %float_0_5

; Blend sky and ground colors
%sky_blended = OpVectorTimesScalar %v3float %sky_color %blend
%ground_blend = OpFSub %float %float_1_0 %blend
%ground_blended = OpVectorTimesScalar %v3float %ground_color %ground_blend
%ambient = OpFAdd %v3float %sky_blended %ground_blended

; Calculate diffuse term
%light_pos_ptr = OpAccessChain %uniform_ptr_v4 %ubo %int_3
%light_pos = OpLoad %v4float %light_pos_ptr
%light_vec3 = OpVectorShuffle %v3float %light_pos %light_pos 0 1 2
%to_light = OpFSub %v3float %light_vec3 %worldPos_val
%to_light_len = OpLength %v3float %to_light
%l = OpFDiv %v3float %to_light %to_light_len
%d = OpDot %v3float %n_norm %l
%d_clamped = OpFMax %float %d %float_0_0
%diffuse = OpVectorTimesScalar %v3float %n_norm %d_clamped
%diffuse_term = OpVectorTimesScalar %v3float %diffuse %float_0_3

; Combine lighting and output
%final_color = OpFAdd %v3float %ambient %diffuse_term
%out_vec4 = OpCompositeConstruct %v4float %final_color %float_1_0
OpStore %outColor %out_vec4

OpReturn
OpFunctionEnd
