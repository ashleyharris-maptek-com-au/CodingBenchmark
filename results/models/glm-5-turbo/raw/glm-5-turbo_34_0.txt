```python
import sys
import math
import random

def solve():
    # Read all input from stdin
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    # Parse input
    N = int(input_data[0])
    K = int(input_data[1])
    CAPACITY = int(input_data[2])
    
    depot_x = float(input_data[3])
    depot_y = float(input_data[4])
    depot = (depot_x, depot_y)
    
    coords = []
    demands = []
    idx = 5
    for i in range(N):
        x = float(input_data[idx])
        y = float(input_data[idx+1])
        d = int(input_data[idx+2])
        coords.append((x, y))
        demands.append(d)
        idx += 3
    
    # Precompute distance matrix (Euclidean)
    dist = [[0.0] * (N + 1) for _ in range(N + 1)]
    for i in range(N + 1):
        for j in range(N + 1):
            if i == 0:
                pi = depot
            else:
                pi = coords[i-1]
            if j == 0:
                pj = depot
            else:
                pj = coords[j-1]
            dist[i][j] = math.hypot(pi[0] - pj[0], pi[1] - pj[1])
    
    def route_distance(route):
        """Calculate total distance of a route starting and ending at depot."""
        d = dist[0][route[0] + 1]
        for i in range(len(route) - 1):
            d += dist[route[i] + 1][route[i+1] + 1]
        d += dist[route[-1] + 1][0]
        return d
    
    def nearest_neighbor_starting_from(start_node, unvisited, remaining_capacity, demands):
        """Build a route greedily using nearest neighbor heuristic."""
        route = [start_node]
        unvisited.remove(start_node)
        remaining_capacity -= demands[start_node]
        
        while unvisited:
            last = route[-1]
            best_next = -1
            best_dist = float('inf')
            for u in unvisited:
                if demands[u] <= remaining_capacity:
                    d = dist[last + 1][u + 1]
                    if d < best_dist:
                        best_dist = d
                        best_next = u
            
            if best_next == -1:
                break  # No more customers fit in this route
            
            route.append(best_next)
            unvisited.remove(best_next)
            remaining_capacity -= demands[best_next]
        
        return route
    
    # --- Heuristic Construction ---
    # Multiple randomized nearest-neighbor runs to escape local optima
    best_routes = None
    best_total_dist = float('inf')
    
    # Deterministic pass: start from farthest nodes from depot
    unvisited = set(range(N))
    temp_routes = []
    while unvisited:
        # Pick the unvisited node farthest from the depot
        start_node = max(unvisited, key=lambda u: dist[0][u + 1])
        r = nearest_neighbor_starting_from(start_node, unvisited, CAPACITY, demands)
        temp_routes.append(r)
    
    temp_total = sum(route_distance(r) for r in temp_routes)
    if temp_total < best_total_dist:
        best_total_dist = temp_total
        best_routes = temp_routes
    
    # Randomized passes
    for _ in range(50):
        unvisited = set(range(N))
        temp_routes = []
        while unvisited:
            start_node = random.choice(list(unvisited))
            r = nearest_neighbor_starting_from(start_node, unvisited, CAPACITY, demands)
            temp_routes.append(r)
        
        temp_total = sum(route_distance(r) for r in temp_routes)
        if temp_total < best_total_dist:
            best_total_dist = temp_total
            best_routes = temp_routes
    
    # Prune empty routes (edge case if N=0)
    routes = [r for r in best_routes if r]
    
    # --- Local Search Improvement ---
    # 2-opt intra-route improvement
    def two_opt(r):
        improved = True
        while improved:
            improved = False
            for i in range(len(r) - 1):
                for j in range(i + 2, len(r)):
                    # Current edges: (i, i+1) and (j, j+1)
                    # New edges: (i, j) and (i+1, j+1)
                    d1 = dist[r[i]+1][r[i+1]+1]
                    d2 = dist[r[j]+1][r[j+1]+1] if j + 1 < len(r) else dist[r[j]+1][0]
                    
                    nd1 = dist[r[i]+1][r[j]+1]
                    nd2 = dist[r[i+1]+1][r[j+1]+1] if j + 1 < len(r) else dist[r[i+1]+1][0]
                    
                    if nd1 + nd2 < d1 + d2 - 1e-9:
                        r[i+1:j+1] = reversed(r[i+1:j+1])
                        improved = True
        return r
    
    for i in range(len(routes)):
        routes[i] = two_opt(routes[i])
    
    # Inter-route 2-opt* and Relocate improvements
    improved = True
    while improved:
        improved = False
        # Cache route demands
        route_demands = [sum(demands[c] for c in r) for r in routes]
        
        for i in range(len(routes)):
            for j in range(len(routes)):
                if i == j:
                    continue
                    
                ri = routes[i]
                rj = routes[j]
                
                if not ri and not rj:
                    continue
                
                # --- Relocate single node ---
                # Try moving a node from ri to rj
                for k in range(len(ri)):
                    c = ri[k]
                    if route_demands[j] + demands[c] > CAPACITY:
                        continue
                    
                    # Cost of removing c from ri
                    prev_i = ri[k-1] if k > 0 else -1
                    next_i = ri[k+1] if k < len(ri) - 1 else -1
                    
                    rem_cost = 0
                    add_cost = 0
                    if prev_i == -1 and next_i == -1:
                        # ri becomes empty
                        rem_cost = dist[0][c+1] + dist[c+1][0]
                        add_cost = 0
                    elif prev_i == -1:
                        rem_cost = dist[0][c+1] + dist[c+1][next_i+1] - dist[0][next_i+1]
                        add_cost = 0
                    elif next_i == -1:
                        rem_cost = dist[prev_i+1][c+1] + dist[c+1][0] - dist[prev_i+1][0]
                        add_cost = 0
                    else:
                        rem_cost = dist[prev_i+1][c+1] + dist[c+1][next_i+1] - dist[prev_i+1][next_i+1]
                        add_cost = 0
                    
                    # Best insertion position in rj
                    best_ins_cost = float('inf')
                    for m in range(len(rj) + 1):
                        prev_j = rj[m-1] if m > 0 else -1
                        next_j = rj[m] if m < len(rj) else -1
                        
                        if prev_j == -1 and next_j == -1:
                            ins_c = dist[0][c+1] + dist[c+1][0]
                        elif prev_j == -1:
                            ins_c = dist[0][c+1] + dist[c+1][next_j+1] - dist[0][next_j+1]
                        elif next_j == -1:
                            ins_c = dist[prev_j+1][c+1] + dist[c+1][0] - dist[prev_j+1][0]
                        else:
                            ins_c = dist[prev_j+1][c+1] + dist[c+1][next_j+1] - dist[prev_j+1][next_j+1]
                        
                        if ins_c < best_ins_cost:
                            best_ins_cost = ins_c
                    
                    if best_ins_cost + rem_cost < -1e-9:
                        # Apply move
                        ri.pop(k)
                        if not ri:
                            routes.pop(i)
                            route_demands.pop(i)
                            # Re-optimize the shortened ri (now removed)
                            j_idx = j if j < i else j - 1
                            rj.insert(0, c) # Dummy, will find best below
                            # Actually find best insert pos
                            # Remove dummy
                            rj.pop(0)
                            best_m = 0
                            best_ins_c2 = float('inf')
                            for m in range(len(rj) + 1):
                                prev_j = rj[m-1] if m > 0 else -1
                                next_j = rj[m] if m < len(rj) else -1
                                if prev_j == -1 and next_j == -1:
                                    ins_c = dist[0][c+1] + dist[c+1][0]
                                elif prev_j == -1:
                                    ins_c = dist[0][c+1] + dist[c+1][next_j+1] - dist[0][next_j+1]
                                elif next_j == -1:
                                    ins_c = dist[prev_j+1][c+1] + dist[c+1][0] - dist[prev_j+1][0]
                                else:
                                    ins_c = dist[prev_j+1][c+1] + dist[c+1][next_j+1] - dist[prev_j+1][next_j+1]
                                if ins_c < best_ins_c2:
                                    best_ins_c2 = ins_c
                                    best_m = m
                            rj.insert(best_m, c)
                            route_demands[j_idx] += demands[c]
                        else:
                            best_m = 0
                            best_ins_c2 = float('inf')
                            for m in range(len(rj) + 1):
                                prev_j = rj[m-1] if m > 0 else -1
                                next_j = rj[m] if m < len(rj) else -1
                                if prev_j == -1 and next_j == -1:
                                    ins_c = dist[0][c+1] + dist[c+1][0]
                                elif prev_j == -1:
                                    ins_c = dist[0][c+1] + dist[c+1][next_j+1] - dist[0][next_j+1]
                                elif next_j == -1:
                                    ins_c = dist[prev_j+1][c+1] + dist[c+1][0] - dist[prev_j+1][0]
                                else:
                                    ins_c = dist[prev_j+1][c+1] + dist[c+1][next_j+1] - dist[prev_j+1][next_j+1]
                                if ins_c < best_ins_c2:
                                    best_ins_c2 = ins_c
                                    best_m = m
                            rj.insert(best_m, c)
                            route_demands[i] -= demands[c]
                            route_demands[j] += demands[c]
                        
                        improved = True
                        break
                
                if improved:
                    break
            if improved:
                break
        
        # After any move, re-run intra-route 2-opt on affected routes
        if improved:
            for r in routes:
                if r:
                    two_opt(r)
    
    # Final recalculation of distances
    routes = [r for r in routes if r]
    total_distance = sum(route_distance(r) for r in routes)
    num_routes = len(routes)
    
    # Output
    print("{:.6f}".format(total_distance))
    print(num_routes)
    for r in routes:
        print(" ".join(map(str, r)))

if __name__ == '__main__':
    solve()
```
