; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

; Types and Constants
%void = OpTypeVoid
%func_type = OpTypeFunction %void
%f32 = OpTypeFloat 32
%v3f32 = OpTypeVector %f32 3
%v4f32 = OpTypeVector %f32 4
%ptr_input_v3f32 = OpTypePointer Input %v3f32
%ptr_output_v4f32 = OpTypePointer Output %v4f32
%f32_2 = OpConstant %f32 2.0
%f32_0_5 = OpConstant %f32 0.5
%f32_1 = OpConstant %f32 1.0
%v3f32_0_5 = OpConstantComposite %v3f32 %f32_0_5 %f32_0_5 %f32_0_5
%v3f32_1 = OpConstantComposite %v3f32 %f32_1 %f32_1 %f32_1

; Variables
%worldPos = OpVariable %ptr_input_v3f32 Input
%normal = OpVariable %ptr_input_v3f32 Input
%uv = OpVariable %ptr_input_v3f32 Input
%tangent = OpVariable %ptr_input_v3f32 Input
%color = OpVariable %ptr_input_v3f32 Input
%outColor = OpVariable %ptr_output_v4f32 Output

; Main Function
%main = OpFunction %void None %func_type
%label = OpLabel
%normal_val = OpLoad %v3f32 %normal
%normal_norm = OpNormalize %v3f32 %normal_val
%normal_scaled = OpFMul %v3f32 %normal_norm %f32_0_5
%normal_mapped = OpFAdd %v3f32 %normal_scaled %v3f32_0_5
%out_color = OpCompositeExtract %v4f32 %normal_mapped 0 1 2 %f32_1
OpStore %outColor %out_color
OpReturn
OpFunctionEnd
