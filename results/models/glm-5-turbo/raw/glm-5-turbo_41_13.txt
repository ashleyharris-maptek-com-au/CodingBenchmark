; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%ptr_input_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_input_v3 Input

%ptr_output_v4 = OpTypePointer Output %v4float
%outColor = OpVariable %ptr_output_v4 Output

%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_0 = OpConstant %float 0.0

%vec_bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%vec_top = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0
%vec_alpha = OpConstantComposite %v4float %float_0_0 %float_0_0 %float_0_0 %float_1_0

%main = OpFunction %void None %func
%label = OpLabel

%pos = OpLoad %v3float %worldPos
%y = OpCompositeExtract %float %pos 0

%mul = OpFMul %float %y %float_0_5
%add = OpFAdd %float %mul %float_0_5

%clamp_low = OpFMax %float %add %float_0_0
%t = OpFMin %float %clamp_low %float_1_0

%one_minus_t = OpFSub %float %float_1_0 %t
%scaled_bottom = OpVectorTimesScalar %v3float %vec_bottom %one_minus_t
%scaled_top = OpVectorTimesScalar %v3float %vec_top %t
%mixed_rgb = OpFAdd %v3float %scaled_bottom %scaled_top

%final_color = OpCompositeInsert %v4float %mixed_rgb %vec_alpha 0
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
