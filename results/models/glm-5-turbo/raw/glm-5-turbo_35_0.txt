use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    if n == 0 {
        writeln!(out, "0\n").unwrap();
        return;
    }

    // adj[i] stores neighbors of vertex i, always with i < j for edge (i, j)
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut it = line.split_whitespace();
        let mut u: usize = it.next().unwrap().parse().unwrap();
        let mut v: usize = it.next().unwrap().parse().unwrap();
        if u > v {
            std::mem::swap(&mut u, &mut v);
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut solution: Vec<usize> = Vec::new();

    let mut degree_order: Vec<usize> = (0..n).collect();
    degree_order.sort_unstable_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));

    // Phase 1: Greedy selection - repeatedly pick the vertex that dominates
    // the most currently undominated vertices.
    loop {
        let mut best_u = usize::MAX;
        let mut best_gain = 0i32;

        for &u in &degree_order {
            if in_set[u] {
                continue;
            }
            let mut gain: i32 = 0;
            if !dominated[u] {
                gain += 1;
            }
            for &v in &adj[u] {
                if !dominated[v] {
                    gain += 1;
                }
            }
            if gain > best_gain {
                best_gain = gain;
                best_u = u;
            }
        }

        if best_gain <= 0 {
            break;
        }

        solution.push(best_u);
        in_set[best_u] = true;
        dominated[best_u] = true;
        for &v in &adj[best_u] {
            dominated[v] = true;
        }
    }

    // Phase 2: Redundancy elimination - try removing vertices from the
    // current solution if all their dominated neighbors remain covered.
    // Multiple passes to catch cascading removals.
    let mut changed = true;
    while changed {
        changed = false;
        let mut i = 0;
        while i < solution.len() {
            let u = solution[i];
            let mut redundant = true;
            for &v in &adj[u] {
                if !in_set[v] {
                    // v is only dominated by u in the current solution
                    redundant = false;
                    break;
                }
            }
            // u itself is in the set, so it's always covered
            if redundant {
                in_set[u] = false;
                solution.swap_remove(i);
                changed = true;
            } else {
                i += 1;
            }
        }
    }

    // Phase 3: Local search (VNS-like) - iteratively try replacing a
    // vertex in the solution with two vertices outside the solution
    // to reduce total set size.
    let mut improved = true;
    let max_iters = 80;
    let mut iters = 0;
    while improved && iters < max_iters {
        improved = false;
        iters += 1;

        let mut i = 0;
        while i < solution.len() && !improved {
            let u = solution[i];

            // Find the set P of vertices that would become uncovered if u is removed
            let mut p_vec = Vec::new();
            for &v in &adj[u] {
                if !in_set[v] {
                    p_vec.push(v);
                }
            }
            // u itself is in_set, so removing it doesn't uncover u

            if p_vec.is_empty() {
                // Redundant (shouldn't happen after Phase 2, but safe check)
                in_set[u] = false;
                solution.swap_remove(i);
                improved = true;
                break;
            }

            // For small |P|, find all subsets of P of size |P|-1.
            // Adding those vertices exactly covers P. We just need to check
            // if we can add <= 1 vertex to cover P (net reduction of 0 or better).
            if p_vec.len() <= 12 {
                let k = p_vec.len() - 1;
                let mut subset = vec![false; p_vec.len()];

                'outer: loop {
                    // Find the next k-subset using a bitmask-like increment
                    let mut count_ones = 0;
                    let mut pos = 0;
                    for j in 0..p_vec.len() {
                        if subset[j] {
                            count_ones += 1;
                        }
                    }
                    if count_ones == k {
                        // Check if adding these k vertices covers P
                        // A vertex w covers P if every p in P is either w itself or adjacent to w
                        for &w in &p_vec {
                            if subset.iter().position(|&b| b).map_or(false, |idx| p_vec[idx] == w) {
                                continue; // w is one of the vertices being added
                            }
                            let covers_all = p_vec.iter().all(|&p| p == w || adj[w].contains(&p));
                            if covers_all {
                                // We can replace u with just w!
                                in_set[u] = false;
                                solution.swap_remove(i);
                                if !in_set[w] {
                                    in_set[w] = true;
                                    solution.push(w);
                                }
                                improved = true;
                                break 'outer;
                            }
                        }
                    }

                    // Increment subset bitmask
                    let mut j = 0;
                    while j < p_vec.len() {
                        if !subset[j] {
                            subset[j] = true;
                            // Clear all bits before j
                            for x in 0..j {
                                subset[x] = false;
                            }
                            break;
                        }
                        subset[j] = false;
                        j += 1;
                    }
                    if j == p_vec.len() {
                        break;
                    }
                }
            }

            // General 2-for-1 swap: try removing u, adding any w outside solution
            // that covers P (if w doesn't cover all of P, it's not a strict 2-for-1,
            // but we check it anyway as a fallback)
            if !improved {
                'outer2: for w in 0..n {
                    if in_set[w] {
                        continue;
                    }
                    let covers_all = p_vec.iter().all(|&p| p == w || adj[w].contains(&p));
                    if covers_all {
                        in_set[u] = false;
                        solution.swap_remove(i);
                        in_set[w] = true;
                        solution.push(w);
                        improved = true;
                        break 'outer2;
                    }
                }
            }

            if !improved {
                i += 1;
            }
        }

        if improved {
            // After a successful replacement, run redundancy elimination again
            // to clean up any newly redundant vertices
            let mut changed = true;
            while changed {
                changed = false;
                let mut i2 = 0;
                while i2 < solution.len() {
                    let u2 = solution[i2];
                    let mut redundant = true;
                    for &v in &adj[u2] {
                        if !in_set[v] {
                            redundant = false;
                            break;
                        }
                    }
                    if redundant {
                        in_set[u2] = false;
                        solution.swap_remove(i2);
                        changed = true;
                    } else {
                        i2 += 1;
                    }
                }
            }
        }
    }

    // Phase 4: 1-for-1 swaps that improve future potential.
    // Swap u in S with w not in S if:
    //   1. w covers all vertices that only u covers (valid swap)
    //   2. w has higher degree, OR w strictly increases coverage of non-S vertices
    let mut improved_11 = true;
    let max_11_iters = 3;
    let mut iters_11 = 0;
    while improved_11 && iters_11 < max_11_iters {
        improved_11 = false;
        iters_11 += 1;

        'outer11: for i in 0..solution.len() {
            let u = solution[i];
            let mut p_vec = Vec::new();
            for &v in &adj[u] {
                if !in_set[v] {
                    p_vec.push(v);
                }
            }

            for w in 0..n {
                if in_set[w] || w == u {
                    continue;
                }
                let covers_all = p_vec.iter().all(|&p| p == w || adj[w].contains(&p));
                if !covers_all {
                    continue;
                }

                // Check if this swap is strictly beneficial
                let deg_u = adj[u].len();
                let deg_w = adj[w].len();

                // Count how many non-S vertices u covers vs w covers
                let mut u_out = 0;
                for &v in &adj[u] {
                    if !in_set[v] && v != w {
                        u_out += 1;
                    }
                }
                let mut w_out = 0;
                for &v in &adj[w] {
                    if !in_set[v] && v != u {
                        w_out += 1;
                    }
                }

                if deg_w > deg_u || w_out > u_out {
                    in_set[u] = false;
                    in_set[w] = true;
                    solution[i] = w;
                    improved_11 = true;
                    break 'outer11;
                }
            }
        }
    }

    // Sort output for deterministic results
    solution.sort_unstable();

    writeln!(out, "{}", solution.len()).unwrap();
    if !solution.is_empty() {
        let s: Vec<String> = solution.iter().map(|x| x.to_string()).collect();
        writeln!(out, "{}", s.join(" ")).unwrap();
    }
}
