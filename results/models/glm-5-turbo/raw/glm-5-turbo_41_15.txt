; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types and Constants
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptrInputVec3 = OpTypePointer Input %vec3
%ptrInputVec2 = OpTypePointer Input %vec3
%ptrOutputVec4 = OpTypePointer Output %vec4
%ptrUniformUBO = OpTypePointer Uniform %UBO
%UBO = OpTypeStruct %vec4 %vec4 %vec4 %vec4 %vec4 %vec4

%worldPos = OpVariable %ptrInputVec3 Input
%normal = OpVariable %ptrInputVec3 Input
%uv = OpVariable %ptrInputVec3 Input
%tangent = OpVariable %ptrInputVec3 Input
%color = OpVariable %ptrInputVec3 Input
%outColor = OpVariable %ptrOutputVec4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%f0 = OpConstant %float 0.0
%f05 = OpConstant %float 0.5
%f1 = OpConstant %float 1.0
%f05_mul = OpConstant %float 0.25

%cool_color = OpConstantComposite %vec3 %f05_mul %f05_mul %f0375
%f0375 = OpConstant %float 0.375
%f035 = OpConstant %float 0.35
%f02 = OpConstant %float 0.2
%warm_color = OpConstantComposite %vec3 %f035 %f035 %f02
%alpha_one = OpConstantComposite %vec4 %f1 %f1 %f1 %f1

; Main Function
%main = OpFunction %void None %func
%entry = OpLabel

; Load normal and normalize
%N = OpLoad %vec3 %normal
%N_norm = OpExtInst %vec3 %glslstd450 Normalize %N

; Load light position
%light_ptr = OpAccessChain %ptrUniformVec4 %ubo %int3
%int3 = OpConstant %int 3
%ptrUniformVec4 = OpTypePointer Uniform %vec4
%light_vec4 = OpLoad %vec4 %light_ptr
%L = OpVectorShuffle %vec3 %light_vec4 %light_vec4 0 1 2

; Calculate L direction (from surface to light)
%P = OpLoad %vec3 %worldPos
%diff = OpFSub %vec3 %L %P
%L_norm = OpExtInst %vec3 %glslstd450 Normalize %diff

; Calculate N dot L
%NdotL = OpDot %float %N_norm %L_norm

; Calculate t = dot(N,L) * 0.5 + 0.5
%t_half = OpFMul %float %NdotL %f05
%t = OpFAdd %float %t_half %f05

; Calculate mix
%one_minus_t = OpFSub %float %f1 %t
%cool_scaled = OpVectorTimesScalar %vec3 %cool_color %one_minus_t
%warm_scaled = OpVectorTimesScalar %vec3 %warm_color %t
%shaded_color = OpFAdd %vec3 %cool_scaled %warm_scaled

; Output with alpha 1.0
%final_color = OpVectorShuffle %vec4 %shaded_color %alpha_one 0 1 2 4
OpStore %outColor %final_color

OpReturn
OpFunctionEnd
