```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

const DIRS = [[1, 0], [-1, 0], [0, 1], [0, -1]];

function idx(x, y) { return y * width + x; }

function simulateFire(breakSet) {
    const burned = new Uint8Array(width * height);
    const inQ = new Uint8Array(width * height);
    const q = [];

    for (const [x, y] of ignitions) {
        const i = idx(x, y);
        if (grid[y][x] !== '~' && !breakSet.has(i)) {
            burned[i] = 1;
            inQ[i] = 1;
            q.push(x, y);
        }
    }

    let head = 0;
    while (head < q.length) {
        const x = q[head++];
        const y = q[head++];
        for (const [dx, dy] of DIRS) {
            const nx = x + dx;
            const ny = y + dy;
            if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
            const ni = idx(nx, ny);
            if (inQ[ni]) continue;
            if (grid[ny][nx] === '~') continue;
            if (breakSet.has(ni)) continue;
            inQ[ni] = 1;
            burned[ni] = 1;
            q.push(nx, ny);
        }
    }
    return burned;
}

const emptySet = new Set();
const burnedNoBreak = simulateFire(emptySet);

const burnableTowns = [];
let totalBurnableValue = 0;
for (const t of towns) {
    if (burnedNoBreak[idx(t.x, t.y)]) {
        burnableTowns.push(t);
        totalBurnableValue += t.value;
    }
}

if (burnableTowns.length === 0 || cutBudget === 0) {
    console.log(JSON.stringify({ firebreaks: [] }));
    process.exit(0);
}

function computeDistances(breakSet) {
    const dist = new Int32Array(width * height).fill(-1);
    const q = [];

    for (const [x, y] of ignitions) {
        const i = idx(x, y);
        if (grid[y][x] === '~') continue;
        if (breakSet.has(i)) continue;
        dist[i] = 0;
        q.push(x, y);
    }

    let head = 0;
    while (head < q.length) {
        const x = q[head++];
        const y = q[head++];
        const d = dist[idx(x, y)] + 1;
        for (const [dx, dy] of DIRS) {
            const nx = x + dx;
            const ny = y + dy;
            if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
            const ni = idx(nx, ny);
            if (dist[ni] >= 0) continue;
            if (grid[ny][nx] === '~') continue;
            if (breakSet.has(ni)) continue;
            dist[ni] = d;
            q.push(nx, ny);
        }
    }
    return dist;
}

function findBestCuts(dist) {
    const neighbors = new Int32Array(width * height * 4).fill(-1);
    const nCount = new Uint8Array(width * height);

    for (let y = 0; y < height; y++) {
        for (let x = 0; x < width; x++) {
            const i = idx(x, y);
            if (dist[i] < 0) continue;
            let cnt = 0;
            for (const [dx, dy] of DIRS) {
                const nx = x + dx;
                const ny = y + dy;
                if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;
                const ni = idx(nx, ny);
                if (dist[ni] > dist[i]) {
                    neighbors[i * 4 + cnt++] = ni;
                }
            }
            nCount[i] = cnt;
        }
    }

    const totalBlocks = new Float64Array(width * height);
    const parent = new Int32Array(width * height).fill(-1);

    for (const t of burnableTowns) {
        const ti = idx(t.x, t.y);
        if (dist[ti] < 0) continue;

        let cur = ti;
        while (cur >= 0) {
            const cnt = nCount[cur];
            const share = t.value / cnt;
            const base = cur * 4;
            for (let j = 0; j < cnt; j++) {
                const nIdx = neighbors[base + j];
                totalBlocks[nIdx] += share;
                parent[nIdx] = cur;
            }
            cur = parent[cur];
        }
    }

    const candidates = [];
    for (let i = 0; i < width * height; i++) {
        if (totalBlocks[i] > 0) {
            candidates.push({ i, score: totalBlocks[i] });
        }
    }
    candidates.sort((a, b) => b.score - a.score);

    const selected = new Set();
    const used = new Uint8Array(width * height);
    let k = 0;
    while (selected.size < cutBudget && k < candidates.length) {
        const { i } = candidates[k++];
        if (used[i]) continue;
        selected.add(i);
        used[i] = 1;
        // Invalidate immediate neighbors from greedy selection
        const bx = i % width;
        const by = (i - bx) / width;
        for (const [dx, dy] of DIRS) {
            const nx = bx + dx;
            const ny = by + dy;
            if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                used[idx(nx, ny)] = 1;
            }
        }
    }

    return selected;
}

const dist0 = computeDistances(emptySet);
let bestBreaks = findBestCuts(dist0);
let bestSaved = 0;

const burned0 = simulateFire(bestBreaks);
for (const t of burnableTowns) {
    if (!burned0[idx(t.x, t.y)]) bestSaved += t.value;
}

for (let iter = 0; iter < 2; iter++) {
    const distI = computeDistances(bestBreaks);
    const candidateBreaks = findBestCuts(distI);
    if (candidateBreaks.size === 0) break;

    const burnedI = simulateFire(candidateBreaks);
    let savedI = 0;
    for (const t of burnableTowns) {
        if (!burnedI[idx(t.x, t.y)]) savedI += t.value;
    }

    if (savedI > bestSaved) {
        bestSaved = savedI;
        bestBreaks = candidateBreaks;
    } else {
        break;
    }
}

const firebreaks = [];
for (const i of bestBreaks) {
    firebreaks.push([i % width, Math.floor(i / width)]);
}

console.log(JSON.stringify({ firebreaks }));
```
