; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo Binding 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Types and Constants
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec3 ; Reusing vec3 to avoid unneeded vec2 type if possible, but strictly using vec3 for uv as per input type is fine
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_32 = OpConstant %float 32.0
%vec4_1_0 = OpConstantComposite %vec4 %float_1 %float_1 %float_1 %float_1

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%uv = OpVariable %ptr_input_vec3 Input
%tangent = OpVariable %ptr_input_vec3 Input
%color = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform

; Main Function
%main = OpFunction %void None %func
%label = OpLabel

; Load interpolated normal and normalize it
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %GLSLstd450 Normalize %n_raw

; Load lightPos (vec4) and cameraPos (vec4) from UBO
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%lightPos = OpLoad %vec4 %lightPos_ptr
%cameraPos = OpLoad %vec4 %cameraPos_ptr

; Extract vec3 world positions for L and V
%lightPosVec3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%cameraPosVec3 = OpVectorShuffle %vec3 %cameraPos %cameraPos 0 1 2

; Load worldPos
%wp = OpLoad %vec3 %worldPos

; Compute L = normalize(lightPos - worldPos)
%L_tmp = OpFSub %vec3 %lightPosVec3 %wp
%L = OpExtInst %vec3 %GLSLstd450 Normalize %L_tmp

; Compute V = normalize(cameraPos - worldPos)
%V_tmp = OpFSub %vec3 %cameraPosVec3 %wp
%V = OpExtInst %vec3 %GLSLstd450 Normalize %V_tmp

; Compute diffuse = max(dot(N, L), 0.0)
%dot_nl = OpDot %float %n %L
%diffuse = OpExtInst %float %GLSLstd450 FMax %dot_nl %float_0

; Compute R = reflect(-L, N)
%neg_L = OpFNegate %vec3 %L
%R = OpExtInst %vec3 %GLSLstd450 Reflect %neg_L %n

; Compute specular = pow(max(dot(R, V), 0.0), 32.0)
%dot_rv = OpDot %float %R %V
%dot_rv_max = OpExtInst %float %GLSLstd450 FMax %dot_rv %float_0
%specular = OpExtInst %float %GLSLstd450 Pow %dot_rv_max %float_32

; Compute grayscale intensity: 0.2 + 0.5 * diffuse + 0.8 * specular
%diff_scale = OpFMul %float %float_0_5 %diffuse
%spec_scale = OpFMul %float %float_0_8 %specular
%intensity = OpFAdd %float %float_0_2 %diff_scale
%intensity = OpFAdd %float %intensity %spec_scale

; Output vec4(intensity, intensity, intensity, 1.0)
%out_vec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %out_vec

OpReturn
OpFunctionEnd
