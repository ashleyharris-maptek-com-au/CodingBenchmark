; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %uv Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func_type = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v2float = OpTypeVector %float 2
%ptr_input_v2 = OpTypePointer Input %v2float
%ptr_output_v4 = OpTypePointer Output %v4float

%uv = OpVariable %ptr_input_v2 Input
%outColor = OpVariable %ptr_output_v4 Output

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%vec2_01 = OpConstantComposite %v2float %float_0 %float_1
%vec4_color = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

%main = OpFunction %void None %func_type
%label = OpLabel
%uv_val = OpLoad %v2float %uv
%r = OpCompositeExtract %float %uv_val 0
%g = OpCompositeExtract %float %uv_val 1
%b = OpCompositeExtract %float %vec4_color 2
%a = OpCompositeExtract %float %vec4_color 3
%color = OpCompositeConstruct %v4float %r %g %b %a
OpStore %outColor %color
OpReturn
OpFunctionEnd
