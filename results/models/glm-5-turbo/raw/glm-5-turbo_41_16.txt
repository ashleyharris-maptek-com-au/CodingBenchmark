; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %uv %outColor
               OpExecutionMode %main OriginUpperLeft
               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0
               OpDecorate %ubo Block
               OpDecorate %ubo Binding 0
               OpDecorate %ubo DescriptorSet 0
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
       %void = OpTypeVoid
      %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v4float = OpTypeVector %float 4
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
        %uv = OpVariable %ptr_Input_v2float Input
   %outColor = OpVariable %ptr_Output_v4float Output
       %UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_Uniform_UBO Uniform
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_20 = OpConstant %int 20
      %float_0 = OpConstant %float 0
      %float_1 = OpConstant %float 1
      %float_2 = OpConstant %float 2
      %float_3 = OpConstant %float 3
      %float_4 = OpConstant %float 4
      %float_5 = OpConstant %float 5
      %float_05 = OpConstant %float 0.5
    %v2zero = OpConstantComposite %v2float %float_0 %float_0
    %v2scale = OpConstantComposite %v2float %float_3 %float_2
    %v2sub = OpConstantComposite %v2float %float_2 %float_1
       %main = OpFunction %void None %func
      %label = OpLabel
        %uv_val = OpLoad %v2float %uv
        %c_scaled = OpFMul %v2float %uv_val %v2scale
        %c = OpFSub %v2float %c_scaled %v2sub
               OpBranch %loop_header
      %loop_header = OpLabel
        %i = OpPhi %int %int_0 %label %i_next %loop_body
        %z = OpPhi %v2float %v2zero %label %z_new %loop_body
        %cmp = OpSLessThan %bool %i %int_20
               OpLoopMerge %loop_merge %loop_body None
               OpBranchConditional %cmp %loop_body %loop_merge
      %loop_body = OpLabel
        %z_x = OpCompositeExtract %float %z 0
        %z_y = OpCompositeExtract %float %z 1
        %x2 = OpFMul %float %z_x %z_x
        %y2 = OpFMul %float %z_y %z_y
        %mag2 = OpFAdd %float %x2 %y2
        %escape = OpFOrdGreaterThan %bool %mag2 %float_4
        %i_next = OpIAdd %int %i %int_1
        %xy = OpFMul %v2float %z %z
        %c_sub = OpFSub %v2float %xy %c
        %z_new = OpVectorShuffle %v2float %c_sub %z 0 1
               OpBranchConditional %escape %loop_merge %loop_header
      %loop_merge = OpLabel
        %final_i = OpPhi %int %i %loop_header %i %loop_body
        %final_i_f = OpConvertSToF %float %final_i
        %t = OpFMul %float %final_i_f %float_05
        %t05 = OpFMul %float %t %float_05
        %one_minus_t = OpFSub %float %float_1 %t
        %color = OpCompositeConstruct %v4float %t %t05 %one_minus_t %float_1
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
