// Adaptive TSP Solver
// Uses different strategies based on problem size:
// - Small (N <= 20): Exact brute-force via DP
// - Medium (N <= 400): Nearest Neighbor + 2-opt
// - Large (N <= 50000): Strip-based Nearest Neighbor + 2-opt
// - Extreme/Epic: Space-filling curve (Morton/Z-order) approximation

#ifdef _MSC_VER
#include <intrin.h>
#endif

// Fast approximate integer square root for distance comparisons
inline unsigned int isqrt32(unsigned int x) {
    if (x <= 1) return x;
    unsigned int r = x;
    unsigned int y = (x + 1) >> 1;
    while (y < r) {
        r = y;
        y = (y + x / y) >> 1;
    }
    return r;
}

// 32-bit count leading zeros
inline int clz32(unsigned int x) {
#ifdef _MSC_VER
    unsigned long idx;
    _BitScanReverse(&idx, x);
    return 31 - (int)idx;
#else
    return __builtin_clz(x);
#endif
}

// Compute 64-bit Morton code (Z-order curve) from normalized coordinates
inline unsigned long long compute_morton(double x, double y) {
    const unsigned int MAX_MORT = 0x3FFFFFFF; // 30 bits
    unsigned int ix = (unsigned int)(x * MAX_MORT);
    unsigned int iy = (unsigned int)(y * MAX_MORT);
    if (ix > MAX_MORT) ix = MAX_MORT;
    if (iy > MAX_MORT) iy = MAX_MORT;
    
    unsigned long long morton = 0;
    for (int i = 0; i < 15; i++) {
        morton |= ((unsigned long long)(ix & 1) << (2 * i));
        morton |= ((unsigned long long)(iy & 1) << (2 * i + 1));
        ix >>= 1;
        iy >>= 1;
    }
    return morton;
}

// Reverse a subarray
template<typename T>
void reverse_range(T* arr, int start, int end) {
    while (start < end) {
        std::swap(arr[start], arr[end]);
        start++;
        end--;
    }
}

// ============================================================
// EXACT SOLVER (N <= 20) - Bitmask DP
// ============================================================
void solve_exact(const std::vector<double>& x, const std::vector<double>& y, 
                 std::vector<int>& tour) {
    int n = (int)x.size();
    
    // Precompute squared distances as integers
    std::vector<std::vector<unsigned int>> d(n, std::vector<unsigned int>(n));
    for (int i = 0; i < n; i++) {
        for (int j = 0; j < n; j++) {
            double dx = x[i] - x[j];
            double dy = y[i] - y[j];
            d[i][j] = (unsigned int)(dx * dx + dy * dy + 0.5);
        }
    }
    
    int full_mask = (1 << n) - 1;
    const unsigned int INF = 0xFFFFFFFF;
    
    // DP[mask][i] = min cost to reach city i with visited set = mask
    std::vector<std::vector<unsigned int>> dp(1 << n, std::vector<unsigned int>(n, INF));
    std::vector<std::vector<int>> parent(1 << n, std::vector<int>(n, -1));
    
    dp[1][0] = 0; // Start at city 0, only city 0 visited
    
    for (int mask = 1; mask <= full_mask; mask++) {
        if (!(mask & 1)) continue; // Must include city 0
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            if (dp[mask][u] == INF) continue;
            
            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;
                int new_mask = mask | (1 << v);
                unsigned int new_cost = dp[mask][u] + d[u][v];
                if (new_cost < dp[new_mask][v]) {
                    dp[new_mask][v] = new_cost;
                    parent[new_mask][v] = u;
                }
            }
        }
    }
    
    // Find best ending city
    unsigned int best_cost = INF;
    int best_end = -1;
    for (int i = 1; i < n; i++) {
        unsigned int cost = dp[full_mask][i] + d[i][0];
        if (cost < best_cost) {
            best_cost = cost;
            best_end = i;
        }
    }
    
    // Reconstruct path
    tour.resize(n);
    tour[0] = 0;
    int mask = full_mask;
    int cur = best_end;
    int idx = n - 1;
    
    while (cur != 0) {
        tour[idx--] = cur;
        int prev = parent[mask][cur];
        mask ^= (1 << cur);
        cur = prev;
    }
}

// ============================================================
// NEAREST NEIGHBOR with spatial indexing
// ============================================================
void nearest_neighbor(const std::vector<double>& x, const std::vector<double>& y,
                      std::vector<int>& tour, int strip_width = 64) {
    int n = (int)x.size();
    tour.resize(n);
    
    // Normalize coordinates to [0, 1]
    double min_x = x[0], max_x = x[0];
    double min_y = y[0], max_y = y[0];
    for (int i = 1; i < n; i++) {
        if (x[i] < min_x) min_x = x[i];
        if (x[i] > max_x) max_x = x[i];
        if (y[i] < min_y) min_y = y[i];
        if (y[i] > max_y) max_y = y[i];
    }
    double range_x = max_x - min_x;
    double range_y = max_y - min_y;
    if (range_x < 1e-10) range_x = 1e-10;
    if (range_y < 1e-10) range_y = 1e-10;
    
    // Build strip-based spatial index
    int num_strips = (int)std::ceil(range_x / (range_x / n * strip_width));
    if (num_strips < 1) num_strips = 1;
    if (num_strips > n) num_strips = n;
    
    std::vector<std::vector<int>> strips(num_strips);
    for (int i = 0; i < n; i++) {
        double nx = (x[i] - min_x) / range_x;
        int s = (int)(nx * num_strips);
        if (s >= num_strips) s = num_strips - 1;
        strips[s].push_back(i);
    }
    
    // Visited flags
    std::vector<bool> visited(n, false);
    visited[0] = true;
    tour[0] = 0;
    
    double cur_x = x[0], cur_y = y[0];
    
    for (int pos = 1; pos < n; pos++) {
        double nx_cur = (cur_x - min_x) / range_x;
        int cur_strip = (int)(nx_cur * num_strips);
        if (cur_strip >= num_strips) cur_strip = num_strips - 1;
        
        int best = -1;
        double best_dist = 1e300;
        int search_radius = 0;
        const int max_radius = 32;
        
        // Expand search radius until we find candidates
        while (best == -1 && search_radius <= max_radius) {
            int s_start = cur_strip - search_radius;
            int s_end = cur_strip + search_radius;
            if (s_start < 0) s_start = 0;
            if (s_end >= num_strips) s_end = num_strips - 1;
            
            for (int s = s_start; s <= s_end; s++) {
                for (int idx : strips[s]) {
                    if (visited[idx]) continue;
                    double dx = x[idx] - cur_x;
                    double dy = y[idx] - cur_y;
                    double dist = dx * dx + dy * dy; // Squared distance for comparison
                    if (dist < best_dist) {
                        best_dist = dist;
                        best = idx;
                    }
                }
            }
            search_radius++;
        }
        
        // Fallback: linear scan (should rarely happen)
        if (best == -1) {
            for (int i = 0; i < n; i++) {
                if (visited[i]) continue;
                double dx = x[i] - cur_x;
                double dy = y[i] - cur_y;
                double dist = dx * dx + dy * dy;
                if (dist < best_dist) {
                    best_dist = dist;
                    best = i;
                }
            }
        }
        
        tour[pos] = best;
        visited[best] = true;
        cur_x = x[best];
        cur_y = y[best];
    }
}

// ============================================================
// 2-OPT LOCAL SEARCH
// ============================================================
void two_opt(const std::vector<double>& x, const std::vector<double>& y,
             std::vector<int>& tour, int max_iterations = -1) {
    int n = (int)tour.size();
    if (n < 4) return;
    
    // Precompute edge lengths (squared)
    auto edge_len = [&](int i, int j) -> double {
        double dx = x[tour[i]] - x[tour[j]];
        double dy = y[tour[i]] - y[tour[j]];
        return dx * dx + dy * dy;
    };
    
    bool improved = true;
    int iteration = 0;
    
    while (improved) {
        improved = false;
        
        for (int i = 0; i < n - 2; i++) {
            int i_next = i + 1;
            double d1 = edge_len(i, i_next);
            
            for (int j = i + 2; j < n; j++) {
                int j_next = (j + 1) % n;
                
                // Skip if adjacent (would create degenerate edge)
                if (i == 0 && j_next == n - 1) continue;
                
                double d2 = edge_len(j, j_next);
                double d3 = edge_len(i, j);
                double d4 = edge_len(i_next, j_next);
                
                if (d1 + d2 > d3 + d4 + 1e-10) {
                    reverse_range(tour.data(), i_next, j);
                    improved = true;
                }
            }
        }
        
        iteration++;
        if (max_iterations > 0 && iteration >= max_iterations) break;
    }
}

// ============================================================
// SPACE-FILLING CURVE (Morton/Z-order) - For massive instances
// ============================================================
void solve_morton(const std::vector<double>& x, const std::vector<double>& y,
                  std::vector<int>& tour) {
    int n = (int)x.size();
    tour.resize(n);
    
    // Find bounds
    double min_x = x[0], max_x = x[0];
    double min_y = y[0], max_y = y[0];
    for (int i = 1; i < n; i++) {
        if (x[i] < min_x) min_x = x[i];
        if (x[i] > max_x) max_x = x[i];
        if (y[i] < min_y) min_y = y[i];
        if (y[i] > max_y) max_y = y[i];
    }
    double range_x = max_x - min_x;
    double range_y = max_y - min_y;
    if (range_x < 1e-10) range_x = 1e-10;
    if (range_y < 1e-10) range_y = 1e-10;
    
    // Compute Morton codes for all cities
    std::vector<unsigned long long> morton(n);
    for (int i = 0; i < n; i++) {
        double nx = (x[i] - min_x) / range_x;
        double ny = (y[i] - min_y) / range_y;
        morton[i] = compute_morton(nx, ny);
    }
    
    // Create index array and sort by Morton code
    std::vector<int> idx(n);
    std::iota(idx.begin(), idx.end(), 0);
    std::sort(idx.begin(), idx.end(), [&](int a, int b) {
        return morton[a] < morton[b];
    });
    
    // Find position of city 0
    auto it = std::find(idx.begin(), idx.end(), 0);
    int pos0 = (int)(it - idx.begin());
    
    // Rotate so city 0 is first
    std::rotate(idx.begin(), idx.begin() + pos0, idx.end());
    
    // Copy to tour
    for (int i = 0; i < n; i++) {
        tour[i] = idx[i];
    }
}

// ============================================================
// BATCHED 2-OPT for large instances (processes segments)
// ============================================================
void two_opt_batched(const std::vector<double>& x, const std::vector<double>& y,
                     std::vector<int>& tour, int segment_size = 500, int passes = 3) {
    int n = (int)tour.size();
    if (n < 4) return;
    
    auto edge_len_sq = [&](int i, int j) -> double {
        double dx = x[tour[i]] - x[tour[j]];
        double dy = y[tour[i]] - y[tour[j]];
        return dx * dx + dy * dy;
    };
    
    for (int pass = 0; pass < passes; pass++) {
        for (int start = 0; start < n - 2; start += segment_size) {
            int end = std::min(start + segment_size, n - 2);
            
            for (int i = start; i < end; i++) {
                int i_next = i + 1;
                double d1 = edge_len_sq(i, i_next);
                
                // Limit j range for performance
                int j_max = std::min(i + 200, n - 1);
                for (int j = i + 2; j < j_max; j++) {
                    int j_next = (j + 1) % n;
                    if (i == 0 && j_next == n - 1) continue;
                    
                    double d2 = edge_len_sq(j, j_next);
                    double d3 = edge_len_sq(i, j);
                    double d4 = edge_len_sq(i_next, j_next);
                    
                    if (d1 + d2 > d3 + d4 + 1e-10) {
                        reverse_range(tour.data(), i_next, j);
                        d1 = edge_len_sq(i, i_next); // Update after reversal
                    }
                }
            }
        }
    }
}

// ============================================================
// MAIN SOLVER WITH ADAPTIVE STRATEGY SELECTION
// ============================================================
int main() {
    // Fast I/O
    std::setvbuf(stdin, nullptr, _IOFBF, 1 << 16);
    std::setvbuf(stdout, nullptr, _IOFBF, 1 << 16);
    
    int n;
    if (scanf("%d", &n) != 1 || n < 2) {
        return 0;
    }
    
    // Read coordinates
    std::vector<double> x(n), y(n);
    for (int i = 0; i < n; i++) {
        scanf("%lf %lf", &x[i], &y[i]);
    }
    
    std::vector<int> tour;
    
    if (n <= 20) {
        // EXACT: Brute-force DP
        solve_exact(x, y, tour);
    }
    else if (n <= 400) {
        // MEDIUM: Nearest Neighbor + Full 2-opt
        nearest_neighbor(x, y, tour, 32);
        two_opt(x, y, tour);
    }
    else if (n <= 50000) {
        // LARGE: Nearest Neighbor + Batched 2-opt
        int strip_w = (n < 5000) ? 48 : 64;
        nearest_neighbor(x, y, tour, strip_w);
        int seg = (n < 5000) ? 800 : 500;
        int passes = (n < 5000) ? 4 : 3;
        two_opt_batched(x, y, tour, seg, passes);
    }
    else {
        // EXTREME/EPIC: Morton curve (O(n log n))
        solve_morton(x, y, tour);
    }
    
    // Output tour
    char buf[32];
    for (int i = 0; i < n; i++) {
        if (i > 0) putchar(' ');
        int len = sprintf(buf, "%d", tour[i]);
        fwrite(buf, 1, len, stdout);
    }
    putchar('\n');
    
    return 0;
}
