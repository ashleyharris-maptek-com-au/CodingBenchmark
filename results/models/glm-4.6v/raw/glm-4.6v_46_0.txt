```glsl
#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds access
    if (i >= N) {
        return;
    }
    
    // Read position and velocity of body i
    vec4 pos_i = inData[i * 2];
    vec4 vel_i = inData[i * 2 + 1];
    float mass_i = pos_i.w;
    
    // Initialize acceleration to zero
    vec3 acc = vec3(0.0);
    
    // Compute gravitational acceleration from all other bodies
    for (uint j = 0; j < N; ++j) {
        if (i == j) continue; // Skip self-interaction
        
        vec4 pos_j = inData[j * 2];
        float mass_j = pos_j.w;
        
        // Compute distance vector
        vec3 r = pos_j.xyz - pos_i.xyz;
        
        // Apply softening to avoid singularities
        float r_squared = dot(r, r) + physics.y * physics.y;
        float r_cubed = r_squared * sqrt(r_squared);
        
        // Add contribution to acceleration
        acc += physics.z * mass_j * r / r_cubed;
    }
    
    // Update velocity (Euler integration)
    vec3 new_vel = vel_i.xyz + acc * physics.x;
    
    // Update position
    vec3 new_pos = pos_i.xyz + new_vel * physics.x;
    
    // Write updated values to output
    outData[i * 2] = vec4(new_pos, mass_i);
    outData[i * 2 + 1] = vec4(new_vel, 0.0);
}
```
