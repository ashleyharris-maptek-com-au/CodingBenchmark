```wgsl
@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let nonce = params.y + gid.x;
    if (gid.x >= params.z) {
        return;
    }

    var state : array<u32, 16>;
    for (var i : u32 = 0u; i < 16u; i++) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing
    for (var round : u32 = 0u; round < 8u; round++) {
        // Group 0: state[0..3]
        {
            var a = state[0];
            var b = state[1];
            var c = state[2];
            var d = state[3];
            a += b;  d ^= a;  d = (d << 16u) | (d >> (32u - 16u));
            c += d;  b ^= c;  b = (b << 12u) | (b >> (32u - 12u));
            a += b;  d ^= a;  d = (d << 8u) | (d >> (32u - 8u));
            c += d;  b ^= c;  b = (b << 7u) | (b >> (32u - 7u));
            state[0] = a;
            state[1] = b;
            state[2] = c;
            state[3] = d;
        }

        // Group 1: state[4..7]
        {
            var a = state[4];
            var b = state[5];
            var c = state[6];
            var d = state[7];
            a += b;  d ^= a;  d = (d << 16u) | (d >> (32u - 16u));
            c += d;  b ^= c;  b = (b << 12u) | (b >> (32u - 12u));
            a += b;  d ^= a;  d = (d << 8u) | (d >> (32u - 8u));
            c += d;  b ^= c;  b = (b << 7u) | (b >> (32u - 7u));
            state[4] = a;
            state[5] = b;
            state[6] = c;
            state[7] = d;
        }

        // Group 2: state[8..11]
        {
            var a = state[8];
            var b = state[9];
            var c = state[10];
            var d = state[11];
            a += b;  d ^= a;  d = (d << 16u) | (d >> (32u - 16u));
            c += d;  b ^= c;  b = (b << 12u) | (b >> (32u - 12u));
            a += b;  d ^= a;  d = (d << 8u) | (d >> (32u - 8u));
            c += d;  b ^= c;  b = (b << 7u) | (b >> (32u - 7u));
            state[8] = a;
            state[9] = b;
            state[10] = c;
            state[11] = d;
        }

        // Group 3: state[12..15]
        {
            var a = state[12];
            var b = state[13];
            var c = state[14];
            var d = state[15];
            a += b;  d ^= a;  d = (d << 16u) | (d >> (32u - 16u));
            c += d;  b ^= c;  b = (b << 12u) | (b >> (32u - 12u));
            a += b;  d ^= a;  d = (d << 8u) | (d >> (32u - 8u));
            c += d;  b ^= c;  b = (b << 7u) | (b >> (32u - 7u));
            state[12] = a;
            state[13] = b;
            state[14] = c;
            state[15] = d;
        }
    }

    // Fold 16 words to 2
    var h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    var h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    // Count trailing zeros
    var trailing_zeros : u32 = 0u;
    if (h0 == 0u) {
        trailing_zeros += 32u;
        h0 = h1;
    }
    if (h0 == 0u) {
        trailing_zeros += 32u;
    } else {
        trailing_zeros += countTrailingZeros(h0);
    }

    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}

// WGSL doesn't have a built-in for countTrailingZeros, so we implement it
fn countTrailingZeros(x : u32) -> u32 {
    var n = 0u;
    if ((x & 0x0000FFFFu) == 0u) { n += 16u; x >>= 16u; }
    if ((x & 0x000000FFu) == 0u) { n += 8u;  x >>= 8u;  }
    if ((x & 0x0000000Fu) == 0u) { n += 4u;  x >>= 4u;  }
    if ((x & 0x00000003u) == 0u) { n += 2u;  x >>= 2u;  }
    if ((x & 0x00000001u) == 0u) { n += 1u;  x >>= 1u;  }
    return n;
}
```
