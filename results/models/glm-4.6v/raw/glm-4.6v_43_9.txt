```hlsl
[[vk::location(0)]] float3 worldPos : TEXCOORD0;
[[vk::location(1)]] float3 normal   : TEXCOORD1;
[[vk::location(2)]] float2 uv       : TEXCOORD2;
[[vk::location(3)]] float3 tangent  : TEXCOORD3;
[[vk::location(4)]] float3 color    : TEXCOORD4;

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float2 uv = input.uv * 8.0;
    float checker = (fmod(uv.x, 2.0) < 1.0) ^ (fmod(uv.y, 2.0) < 1.0) ? 0.9 : 0.2;
    
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    float3 H = normalize(L + V);
    
    float diffuse = max(dot(N, L), 0.0);
    float specular = pow(max(dot(N, H), 0.0), 32.0);
    
    float3 baseColor = float3(checker, checker, checker);
    float3 ambient = 0.1 * baseColor;
    float3 diffuseColor = baseColor * diffuse;
    float3 specularColor = float3(1.0, 1.0, 1.0) * specular;
    
    float3 finalColor = ambient + diffuseColor + specularColor;
    finalColor *= (0.5 + 0.5 * diffuse);
    
    return float4(finalColor, 1.0);
}
```
