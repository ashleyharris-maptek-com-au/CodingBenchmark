const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Calculate Euclidean distance between two points
function distance(p1, p2) {
    return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
}

// Calculate total distance of a route
function routeDistance(route, depot, packages) {
    let total = 0;
    let current = depot;
    for (const id of route) {
        const pkg = packages.find(p => p.id === id);
        total += distance(current, [pkg.x, pkg.y]);
        current = [pkg.x, pkg.y];
    }
    total += distance(current, depot);
    return total;
}

// Simple greedy algorithm to assign packages to routes
function planRoutes(depot, truckCount, packages) {
    const routes = [];
    const remainingPackages = [...packages];
    
    // Sort packages by distance from depot (closest first)
    remainingPackages.sort((a, b) => {
        const distA = distance(depot, [a.x, a.y]);
        const distB = distance(depot, [b.x, b.y]);
        return distA - distB;
    });
    
    // Assign packages to routes in a round-robin fashion
    while (remainingPackages.length > 0) {
        const route = [];
        let currentRouteIndex = routes.length % truckCount;
        
        if (routes.length < truckCount) {
            routes.push(route);
        } else {
            currentRouteIndex = routes.length % truckCount;
        }
        
        // Add packages to the current route until we run out or reach truck limit
        while (remainingPackages.length > 0 && routes[currentRouteIndex].length < Math.ceil(packages.length / truckCount)) {
            const pkg = remainingPackages.shift();
            routes[currentRouteIndex].push(pkg.id);
        }
    }
    
    return routes;
}

// Main execution
const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;

const routes = planRoutes(depot, truckCount, packages);

// Output the result
console.log(JSON.stringify({ routes }));
