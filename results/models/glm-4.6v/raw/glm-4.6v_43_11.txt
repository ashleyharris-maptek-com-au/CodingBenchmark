```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the normal vector
    float3 N = normalize(input.normal);
    
    // Calculate light direction (normalized)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate diffuse lighting (Lambertian)
    float diffuse = max(dot(N, L), 0.0);
    
    // Polka dot pattern
    float2 cell = input.uv * 6.0; // 6x6 grid
    float2 center = floor(cell) + 0.5; // Center of each cell
    float2 toCenter = input.uv * 6.0 - center;
    float dist = length(toCenter);
    float radius = 0.3;
    
    // Determine if we're in a dot
    bool inDot = dist < radius;
    
    // Base colors
    float3 dotColor = float3(1.0, 0.8, 0.0); // Yellow
    float3 bgColor = float3(0.1, 0.1, 0.3);  // Dark blue
    
    // Select color based on pattern
    float3 baseColor = inDot ? dotColor : bgColor;
    
    // Apply Lambertian shading factor
    float shading = 0.3 + 0.7 * diffuse;
    float3 finalColor = baseColor * shading;
    
    return float4(finalColor, 1.0);
}
```
