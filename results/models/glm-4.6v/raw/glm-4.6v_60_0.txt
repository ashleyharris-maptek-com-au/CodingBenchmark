```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Create a map of tasks by ID for quick lookup
const taskMap = {};
tasks.forEach(task => {
  taskMap[task.id] = task;
});

// Build adjacency list for dependencies
const adj = {};
const inDegree = {};
tasks.forEach(task => {
  inDegree[task.id] = task.deps.length;
  adj[task.id] = [];
});
tasks.forEach(task => {
  task.deps.forEach(depId => {
    adj[depId].push(task.id);
  });
});

// Topological sort using Kahn's algorithm
const queue = [];
const order = [];
for (const taskId in inDegree) {
  if (inDegree[taskId] === 0) {
    queue.push(taskId);
  }
}

while (queue.length > 0) {
  const current = queue.shift();
  order.push(current);
  adj[current].forEach(neighbor => {
    inDegree[neighbor]--;
    if (inDegree[neighbor] === 0) {
      queue.push(neighbor);
    }
  });
}

// If there's a cycle (shouldn't happen in valid input), we can't schedule
if (order.length !== tasks.length) {
  console.log(JSON.stringify({ order }));
  process.exit(0);
}

// Greedy scheduling based on topological order
const schedule = [];
const machineEndTimes = new Array(machineCount).fill(0);
const taskStartTimes = {};

for (const taskId of order) {
  const task = taskMap[taskId];
  let earliestStart = 0;
  
  // Find the maximum finish time of all dependencies
  for (const depId of task.deps) {
    const depEndTime = taskStartTimes[depId] + taskMap[depId].duration;
    if (depEndTime > earliestStart) {
      earliestStart = depEndTime;
    }
  }
  
  // Find the machine with the earliest available time
  let minMachine = 0;
  let minTime = machineEndTimes[0];
  for (let i = 1; i < machineCount; i++) {
    if (machineEndTimes[i] < minTime) {
      minTime = machineEndTimes[i];
      minMachine = i;
    }
  }
  
  // Schedule the task
  const start = Math.max(earliestStart, minTime);
  schedule.push({
    id: taskId,
    start: start,
    machine: minMachine
  });
  
  // Update machine end time and task start time
  machineEndTimes[minMachine] = start + task.duration;
  taskStartTimes[taskId] = start;
}

console.log(JSON.stringify({ schedule }));
```
