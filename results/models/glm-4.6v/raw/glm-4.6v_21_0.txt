# N-Dimensional Snake Game Solver

I'll create a C++ solver that can handle snake games in any number of dimensions efficiently. The solution will use a breadth-first search (BFS) approach with heuristic guidance to find the shortest path to food while avoiding obstacles and the snake's own body.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <unordered_set>
#include <algorithm>
#include <cmath>
#include <limits>
#include <tuple>
#include <functional>

using namespace std;

// Define a point in N-dimensional space
using Point = vector<int>;

// Define a state in the game (position + direction)
struct State {
    Point pos;
    int dir; // direction index (0 to D-1)
    int sign; // +1 or -1
    int steps; // steps taken to reach this state
    int heuristic; // estimated distance to food

    // Comparison operator for priority queue
    bool operator<(const State& other) const {
        return heuristic > other.heuristic; // min-heap
    }
};

// Hash function for Point
struct PointHash {
    size_t operator()(const Point& p) const {
        size_t hash = 0;
        for (int coord : p) {
            hash ^= hash << 13;
            hash ^= hash >> 7;
            hash ^= coord;
        }
        return hash;
    }
};

// Equality operator for Point
struct PointEqual {
    bool operator()(const Point& a, const Point& b) const {
        return a == b;
    }
};

class SnakeGame {
private:
    int D; // dimensions
    vector<int> bounds; // bounds for each dimension
    int max_turns;
    int snake_length;
    Point head_pos;
    vector<Point> food;
    vector<Point> obstacles;
    unordered_set<Point, PointHash, PointEqual> obstacle_set;
    unordered_set<Point, PointHash, PointEqual> snake_body;

    // Check if a position is valid (within bounds and not an obstacle)
    bool is_valid(const Point& pos) const {
        for (int i = 0; i < D; ++i) {
            if (pos[i] < 0 || pos[i] >= bounds[i]) {
                return false;
            }
        }
        return obstacle_set.find(pos) == obstacle_set.end();
    }

    // Check if a position is occupied by the snake (excluding head)
    bool is_snake_body(const Point& pos) const {
        return snake_body.find(pos) != snake_body.end();
    }

    // Manhattan distance heuristic
    int manhattan_distance(const Point& a, const Point& b) const {
        int dist = 0;
        for (int i = 0; i < D; ++i) {
            dist += abs(a[i] - b[i]);
        }
        return dist;
    }

    // Generate next possible moves from current state
    vector<State> get_neighbors(const State& current) const {
        vector<State> neighbors;
        for (int dir = 0; dir < D; ++dir) {
            for (int sign : {-1, 1}) {
                if (dir == current.dir && sign == -current.sign) {
                    continue; // avoid going back immediately
                }
                Point new_pos = current.pos;
                new_pos[dir] += sign;
                
                if (is_valid(new_pos) && !is_snake_body(new_pos)) {
                    int heuristic = manhattan_distance(new_pos, food[0]);
                    neighbors.emplace_back(State{new_pos, dir, sign, current.steps + 1, heuristic});
                }
            }
        }
        return neighbors;
    }

public:
    SnakeGame() {
        // Read input
        cin >> D >> max_turns;
        bounds.resize(D);
        for (int i = 0; i < D; ++i) {
            cin >> bounds[i];
        }
        
        cin >> snake_length;
        head_pos.resize(D);
        for (int i = 0; i < D; ++i) {
            cin >> head_pos[i];
        }
        
        int F;
        cin >> F;
        food.resize(F);
        for (int i = 0; i < F; ++i) {
            food[i].resize(D);
            for (int j = 0; j < D; ++j) {
                cin >> food[i][j];
            }
        }
        
        int O;
        cin >> O;
        obstacles.resize(O);
        for (int i = 0; i < O; ++i) {
            obstacles[i].resize(D);
            for (int j = 0; j < D; ++j) {
                cin >> obstacles[i][j];
            }
            obstacle_set.insert(obstacles[i]);
        }
        
        // Initialize snake body (just the head for now)
        snake_body.insert(head_pos);
    }

    // Main game loop
    void play() {
        int turn = 0;
        Point current_head = head_pos;
        int food_index = 0;
        
        while (turn < max_turns) {
            if (food_index >= food.size()) {
                break; // all food eaten
            }
            
            // A* search to find path to next food
            priority_queue<State> open_set;
            unordered_set<Point, PointHash, PointEqual> closed_set;
            
            int initial_heuristic = manhattan_distance(current_head, food[food_index]);
            open_set.emplace(State{current_head, -1, 0, 0, initial_heuristic});
            
            Point target = food[food_index];
            bool found = false;
            State best_state;
            
            while (!open_set.empty()) {
                State current = open_set.top();
                open_set.pop();
                
                if (closed_set.find(current.pos) != closed_set.end()) {
                    continue;
                }
                closed_set.insert(current.pos);
                
                if (current.pos == target) {
                    found = true;
                    best_state = current;
                    break;
                }
                
                for (const State& neighbor : get_neighbors(current)) {
                    if (closed_set.find(neighbor.pos) == closed_set.end()) {
                        open_set.push(neighbor);
                    }
                }
            }
            
            if (!found) {
                // No path found, game over
                break;
            }
            
            // Move towards the target
            if (best_state.dir != -1) {
                cout << best_state.dir << " " << best_state.sign << endl;
                cout.flush();
                
                // Update snake position
                Point new_head = current_head;
                new_head[best_state.dir] += best_state.sign;
                current_head = new_head;
                
                // Check if we ate food
                if (current_head == target) {
                    food_index++;
                    // Snake grows, so we don't remove the tail
                } else {
                    // Remove tail (snake moves forward)
                    snake_body.erase(snake_body.begin());
                    snake_body.insert(current_head);
                }
                
                turn++;
            }
        }
    }
};

int main() {
    SnakeGame game;
    game.play();
    return 0;
}
```

### Explanation:

1. **Data Structures**:
   - `Point` represents a position in N-dimensional space
   - `State` represents a game state (position, direction, steps taken, and heuristic)
   - `PointHash` and `PointEqual` are custom hash and equality functions for points

2. **Game Logic**:
   - The game reads input parameters including dimensions, bounds, snake position, food, and obstacles
   - The main game loop uses A* search to find the shortest path to the next food item
   - The heuristic used is Manhattan distance, which is admissible for grid-based pathfinding
   - The snake moves one step at a time, updating its position and checking for collisions

3. **Efficiency Considerations**:
   - The A* search is bounded by the number of possible moves (2*D per state)
   - Obstacles and snake body are stored in hash sets for O(1) lookup
   - The algorithm works for any number of dimensions by using vector-based data structures

4. **Output**:
   - For each turn, the solver outputs the axis (0 to D-1) and direction (-1 or +1) to move
   - The game ends when all food is eaten, a collision occurs, or the turn limit is reached

This solution should handle games from 2D to 20D efficiently, though performance will naturally degrade with higher dimensions due to the exponential growth of the search space. The A* algorithm with Manhattan distance heuristic provides a good balance between optimality and performance for this problem.
