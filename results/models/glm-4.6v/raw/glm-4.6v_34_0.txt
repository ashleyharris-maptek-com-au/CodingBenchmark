To solve the Capacitated Vehicle Routing Problem (CVRP), we'll use a heuristic approach that combines the Clarke-Wright savings algorithm with a nearest neighbor method to construct feasible routes. The solution involves the following steps:

1. **Read Input**: Parse the number of customers (N), number of vehicles (K), vehicle capacity, depot coordinates, and customer details.
2. **Calculate Distances**: Compute the Euclidean distance between all pairs of nodes (depot and customers).
3. **Clarke-Wright Savings Algorithm**: Compute savings for merging routes and prioritize merges that yield the highest savings while respecting vehicle capacity.
4. **Construct Routes**: Use the savings to build initial routes, ensuring each route's total demand does not exceed the vehicle capacity.
5. **Output Results**: Print the total distance, number of routes, and the routes themselves.

Here's the complete Python code implementing this approach:

```python
import sys
import math
from collections import defaultdict

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    CAPACITY = int(data[idx]); idx += 1
    depot_x = float(data[idx]); idx += 1
    depot_y = float(data[idx]); idx += 1
    customers = []
    for _ in range(N):
        x = float(data[idx]); idx += 1
        y = float(data[idx]); idx += 1
        demand = int(data[idx]); idx += 1
        customers.append((x, y, demand))
    return N, K, CAPACITY, (depot_x, depot_y), customers

def compute_distance_matrix(depot, customers):
    nodes = [depot] + customers
    n = len(nodes)
    dist = [[0.0] * n for _ in range(n)]
    for i in range(n):
        for j in range(i + 1, n):
            dx = nodes[i][0] - nodes[j][0]
            dy = nodes[i][1] - nodes[j][1]
            dist[i][j] = dist[j][i] = math.sqrt(dx*dx + dy*dy)
    return dist

def clarke_wright_savings(depot, customers, capacity, dist):
    n = len(customers) + 1  # +1 for depot
    savings = []
    for i in range(1, n):
        for j in range(i + 1, n):
            s = dist[0][i] + dist[0][j] - dist[i][j]
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])
    
    routes = []
    # Each route is a list of customer indices (1-based for customers, 0 is depot)
    # Initially, each customer is its own route
    customer_routes = {}
    for i in range(1, n):
        customer_routes[i] = [i]
    
    for s, i, j in savings:
        if i in customer_routes and j in customer_routes:
            route_i = customer_routes[i]
            route_j = customer_routes[j]
            if route_i is not route_j:
                # Check if merging is possible
                total_demand = sum(customers[idx-1][2] for idx in route_i + route_j)
                if total_demand <= capacity:
                    # Merge routes
                    new_route = route_i + route_j
                    # Remove old routes
                    del customer_routes[i]
                    del customer_routes[j]
                    # Assign new route to the smallest index
                    min_idx = min(i, j)
                    customer_routes[min_idx] = new_route
    # Collect all routes
    routes = []
    for route in customer_routes.values():
        if route:
            routes.append(route)
    return routes

def calculate_total_distance(routes, dist):
    total_dist = 0.0
    for route in routes:
        if not route:
            continue
        # Start from depot (0), go through each customer, return to depot
        prev = 0
        for customer in route:
            total_dist += dist[prev][customer]
            prev = customer
        total_dist += dist[prev][0]  # return to depot
    return total_dist

def main():
    input_data = read_input()
    if not input_data:
        return
    N, K, CAPACITY, depot, customers = input_data
    dist = compute_distance_matrix(depot, customers)
    routes = clarke_wright_savings(depot, customers, CAPACITY, dist)
    # Ensure all customers are covered (some may be left out in savings)
    covered = set()
    for route in routes:
        covered.update(route)
    uncovered = set(range(1, N+1)) - covered
    # Assign uncovered customers to new routes if possible
    for customer in uncovered:
        added = False
        for route in routes:
            current_demand = sum(customers[idx-1][2] for idx in route)
            if current_demand + customers[customer-1][2] <= CAPACITY:
                route.append(customer)
                added = True
                break
        if not added:
            routes.append([customer])
    # Limit to K routes (if more, split the longest routes)
    while len(routes) > K:
        # Find the route with the maximum distance
        max_dist = -1
        max_idx = -1
        for i, route in enumerate(routes):
            route_dist = 0.0
            prev = 0
            for customer in route:
                route_dist += dist[prev][customer]
                prev = customer
            route_dist += dist[prev][0]
            if route_dist > max_dist:
                max_dist = route_dist
                max_idx = i
        # Split the route into two
        route = routes.pop(max_idx)
        if len(route) < 2:
            continue  # cannot split
        # Split at the middle
        mid = len(route) // 2
        routes.append(route[:mid])
        routes.append(route[mid:])
    # Ensure all customers are covered
    covered = set()
    for route in routes:
        covered.update(route)
    if len(covered) != N:
        # Assign remaining customers to any route that can take them
        for customer in range(1, N+1):
            if customer not in covered:
                for route in routes:
                    current_demand = sum(customers[idx-1][2] for idx in route)
                    if current_demand + customers[customer-1][2] <= CAPACITY:
                        route.append(customer)
                        covered.add(customer)
                        break
    total_dist = calculate_total_distance(routes, dist)
    # Output
    print("{:.6f}".format(total_dist))
    print(len(routes))
    for route in routes:
        # Convert to 0-based indices
        print(' '.join(map(str, [idx-1 for idx in route])))

if __name__ == "__main__":
    main()
```

### Explanation:
1. **Reading Input**: The `read_input` function parses the input data into the required variables.
2. **Distance Matrix**: The `compute_distance_matrix` function calculates the Euclidean distances between all nodes (depot and customers).
3. **Clarke-Wright Savings**: The `clarke_wright_savings` function computes savings for merging routes and constructs initial routes by merging those with the highest savings, ensuring capacity constraints are met.
4. **Total Distance Calculation**: The `calculate_total_distance` function computes the total travel distance for all routes.
5. **Main Function**: The `main` function orchestrates the process, handles uncovered customers, limits the number of routes to K, and outputs the results in the specified format.

This approach efficiently constructs feasible routes using a heuristic method, ensuring all constraints are respected while minimizing the total travel distance.
