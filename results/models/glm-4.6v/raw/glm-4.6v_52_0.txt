import math
import numpy as np

# Module-level variables for state persistence
class AutopilotState:
    def __init__(self):
        # PID controllers
        self.speed_pid = PIDController(Kp=0.5, Ki=0.1, Kd=0.2, setpoint=0)
        self.lane_pid = PIDController(Kp=0.8, Ki=0.05, Kd=0.1, setpoint=0)
        
        # Sensor fusion buffers
        self.wheel_speeds = []  # Buffer for valid wheel speeds
        self.gyro_rates = []    # Buffer for valid yaw rates
        self.accel_data = []    # Buffer for valid accel data
        self.gps_speeds = []    # Buffer for valid GPS speeds
        self.steering_angles = []  # Buffer for valid steering angles
        
        # Filtering
        self.lidar_ranges = {angle: [] for angle in [-90, -60, -45, -30, -15, 0, 15, 30, 45, 60, 90]}
        self.vision_lane_offsets = []
        self.vision_obstacles = []
        
        # Previous outputs
        self.prev_steering = 0.0
        self.prev_throttle = 0.0
        self.prev_brake = 0.0
        
        # Obstacle tracking
        self.obstacle_detected = False
        self.obstacle_distance = float('inf')
        self.obstacle_relative_speed = 0.0
        
        # Traction management
        self.low_mu = False
        self.mu_threshold = 0.4

# PID Controller class
class PIDController:
    def __init__(self, Kp, Ki, Kd, setpoint=0):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.setpoint = setpoint
        self.integral = 0
        self.prev_error = 0
        self.prev_time = None
    
    def update(self, current_value, dt):
        error = self.setpoint - current_value
        self.integral += error * dt
        
        # Anti-windup
        integral_limit = 10.0
        self.integral = max(-integral_limit, min(integral_limit, self.integral))
        
        if self.prev_time is not None:
            derivative = (error - self.prev_error) / dt
        else:
            derivative = 0
        
        output = self.Kp * error + self.Ki * self.integral + self.Kd * derivative
        
        self.prev_error = error
        self.prev_time = dt
        
        return output

# Initialize global state
autopilot_state = AutopilotState()

def autopilot_step(state: dict, dt: float) -> dict:
    # Update module-level state with new data
    update_sensors(state, dt)
    
    # Sensor fusion and data processing
    vehicle_speed = get_fused_speed()
    lane_offset = get_fused_lane_offset()
    steering_angle = get_fused_steering_angle()
    obstacle_info = get_fused_obstacle_info()
    
    # Update obstacle tracking
    autopilot_state.obstacle_detected = obstacle_info['detected']
    autopilot_state.obstacle_distance = obstacle_info['distance']
    autopilot_state.obstacle_relative_speed = obstacle_info['rel_speed']
    
    # Check for low friction
    autopilot_state.low_mu = state.get('estimated_mu', 1.0) < autopilot_state.mu_threshold
    
    # Determine target speed
    target_speed = state.get('target_speed_kmh', 100)
    speed_limit = get_fused_speed_limit()
    if speed_limit is not None and speed_limit < target_speed:
        target_speed = speed_limit
    
    # Speed control
    speed_error = target_speed - vehicle_speed
    throttle = autopilot_state.speed_pid.update(speed_error, dt)
    
    # Lane keeping
    lane_error = lane_offset
    steering = autopilot_state.lane_pid.update(lane_error, dt)
    
    # Obstacle avoidance
    brake = 0.0
    if autopilot_state.obstacle_detected and autopilot_state.obstacle_distance < 50:
        # Emergency braking
        if autopilot_state.obstacle_distance < 20:
            brake = 1.0
        else:
            # Proportional braking based on distance
            brake = max(0, 1.0 - (autopilot_state.obstacle_distance / 50))
            brake = min(1.0, brake * 2)  # Amplify for safety
    
    # Traction management
    if autopilot_state.low_mu:
        # Reduce speed and make inputs gentler
        throttle *= 0.5
        steering *= 0.7
        brake *= 0.8
    
    # Apply limits
    throttle = max(0.0, min(1.0, throttle))
    steering = max(-1.0, min(1.0, steering))
    brake = max(0.0, min(1.0, brake))
    
    # Return control outputs
    return {
        'steering': steering,
        'throttle': throttle,
        'brake': brake
    }

def update_sensors(state: dict, dt: float):
    # Update wheel speeds
    wheel_speeds = []
    for i in range(1, 5):
        key = f'wheel_speed_{["fl", "fr", "rl", "rr"][i-1]}_kmh'
        valid_key = f'wheel_speed_{["fl", "fr", "rl", "rr"][i-1]}_valid'
        if state.get(valid_key, False):
            wheel_speeds.append(state.get(key, 0))
    autopilot_state.wheel_speeds.extend(wheel_speeds[-5:])  # Keep last 5 valid readings
    
    # Update gyro rates
    gyro_rates = []
    for i in range(1, 4):
        key = f'gyro_{i}_yaw_rate_dps'
        valid_key = f'gyro_{i}_valid'
        if state.get(valid_key, False):
            gyro_rates.append(state.get(key, 0))
    autopilot_state.gyro_rates.extend(gyro_rates[-3:])  # Keep last 3 valid readings
    
    # Update GPS speeds
    gps_speeds = []
    for i in range(1, 3):
        key = f'gps_{i}_speed_kmh'
        valid_key = f'gps_{i}_valid'
        if state.get(valid_key, False):
            gps_speeds.append(state.get(key, 0))
    autopilot_state.gps_speeds.extend(gps_speeds[-2:])  # Keep last 2 valid readings
    
    # Update steering angles
    steering_angles = []
    for i in range(1, 3):
        key = f'steering_angle_{i}_deg'
        valid_key = f'steering_angle_{i}_valid'
        if state.get(valid_key, False):
            steering_angles.append(state.get(key, 0))
    autopilot_state.steering_angles.extend(steering_angles[-2:])  # Keep last 2 valid readings
    
    # Update lidar ranges
    for angle in autopilot_state.lidar_ranges.keys():
        key = f'lidar_{angle:+03d}_deg_range_m'
        valid_key = f'lidar_{angle:+03d}_deg_valid'
        if state.get(valid_key, False):
            autopilot_state.lidar_ranges[angle].append(state.get(key, 0))
        if len(autopilot_state.lidar_ranges[angle]) > 10:
            autopilot_state.lidar_ranges[angle].pop(0)
    
    # Update vision systems
    for i in range(1, 4):
        vision_prefix = f'vision_{i}_'
        offset = state.get(vision_prefix + 'lane_offset_m', 0)
        valid = state.get(vision_prefix + 'left_line_valid', False) and state.get(vision_prefix + 'right_line_valid', False)
        if valid:
            autopilot_state.vision_lane_offsets.append(offset)
        if len(autopilot_state.vision_lane_offsets) > 5:
            autopilot_state.vision_lane_offsets.pop(0)
        
        # Update obstacle info
        if state.get(vision_prefix + 'obstacle_detected', False) and state.get(vision_prefix + 'obstacle_valid', False):
            autopilot_state.vision_obstacles.append({
                'distance': state.get(vision_prefix + 'obstacle_dist_m', float('inf')),
                'rel_speed': state.get(vision_prefix + 'obstacle_rel_speed_kmh', 0)
            })
        if len(autopilot_state.vision_obstacles) > 5:
            autopilot_state.vision_obstacles.pop(0)

def get_fused_speed() -> float:
    # Use median of all valid speed sources
    speeds = []
    
    # Wheel speeds
    if autopilot_state.wheel_speeds:
        speeds.extend(autopilot_state.wheel_speeds[-3:])
    
    # GPS speeds
    if autopilot_state.gps_speeds:
        speeds.extend(autopilot_state.gps_speeds[-2:])
    
    # Vehicle speed sensors
    speeds.append(state.get('vehicle_speed_1_kmh', 0))
    speeds.append(state.get('vehicle_speed_2_kmh', 0))
    
    if not speeds:
        return 0.0
    
    return np.median(speeds)

def get_fused_lane_offset() -> float:
    # Use median of vision lane offsets
    if autopilot_state.vision_lane_offsets:
        return np.median(autopilot_state.vision_lane_offsets[-3:])
    return 0.0

def get_fused_steering_angle() -> float:
    # Use median of steering angle sensors
    if autopilot_state.steering_angles:
        return np.median(autopilot_state.steering_angles[-2:])
    return 0.0

def get_fused_obstacle_info() -> dict:
    # Use the closest obstacle from vision systems
    if not autopilot_state.vision_obstacles:
        return {'detected': False, 'distance': float('inf'), 'rel_speed': 0}
    
    # Find the closest obstacle
    closest = min(autopilot_state.vision_obstacles, key=lambda o: o['distance'])
    return {
        'detected': True,
        'distance': closest['distance'],
        'rel_speed': closest['rel_speed']
    }

def get_fused_speed_limit() -> float:
    # Use median of valid speed limits from vision systems
    speed_limits = []
    for i in range(1, 4):
        vision_prefix = f'vision_{i}_'
        if state.get(vision_prefix + 'speed_limit_valid', False):
            speed_limits.append(state.get(vision_prefix + 'speed_limit_kmh', 0))
    
    if speed_limits:
        return np.median(speed_limits)
    return None
