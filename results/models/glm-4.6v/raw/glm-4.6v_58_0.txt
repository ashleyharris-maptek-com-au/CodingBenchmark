const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Helper function to calculate Manhattan distance
function manhattanDistance(x1, y1, x2, y2) {
    return Math.abs(x1 - x2) + Math.abs(y1 - y2);
}

// Main assignment function
function assignJobsToRobots(grid, robots, jobs) {
    const numRobots = robots.length;
    const numJobs = jobs.length;
    
    // If there are more robots than jobs, some robots will get no jobs
    const assignments = new Array(numRobots).fill().map(() => []);
    
    // Create a list of jobs sorted by priority (higher priority first)
    const sortedJobs = [...jobs].sort((a, b) => b.priority - a.priority);
    
    // Assign jobs to robots using a greedy approach
    for (const job of sortedJobs) {
        let bestRobot = -1;
        let minDistance = Infinity;
        
        for (let i = 0; i < numRobots; i++) {
            const robot = robots[i];
            const distance = manhattanDistance(robot.x, robot.y, job.x, job.y);
            
            if (distance < minDistance) {
                minDistance = distance;
                bestRobot = i;
            }
        }
        
        assignments[bestRobot].push(job.id);
    }
    
    return assignments;
}

// Execute the assignment
const result = assignJobsToRobots(data.grid, data.robots, data.jobs);

// Output the result
console.log(JSON.stringify({ assignments: result }));
