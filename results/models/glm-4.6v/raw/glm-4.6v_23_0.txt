# Lunar Lander Controller in Rust

I'll implement a real-time lunar lander controller that reads from stdin and writes to stdout. The controller will use a simple PID controller for both vertical and horizontal positioning, with additional logic to ensure safe landing.

```rust
use std::io::{self, BufRead, BufReader, Write};
use std::sync::mpsc;
use std::thread;
use std::time::{Duration, Instant};

// Constants
const UPDATE_INTERVAL: Duration = Duration::from_millis(100); // 10 Hz updates
const OUTPUT_INTERVAL: Duration = Duration::from_millis(50);  // 20 Hz output
const MAX_THRUST: f64 = 1.0;
const MAX_TURN: f64 = 1.0;
const SAFE_LANDING_VELOCITY: f64 = 2.0;
const TARGET_PAD_SIZE: f64 = 2.0; // Assuming 2x2m target pad

// PID Controller parameters
const VERTICAL_KP: f64 = 0.8;
const VERTICAL_KI: f64 = 0.1;
const VERTICAL_KD: f64 = 0.3;
const HORIZONTAL_KP: f64 = 0.5;
const HORIZONTAL_KI: f64 = 0.05;
const HORIZONTAL_KD: f64 = 0.2;

// PID Controller struct
struct PIDController {
    kp: f64,
    ki: f64,
    kd: f64,
    integral: f64,
    prev_error: f64,
    prev_time: Instant,
}

impl PIDController {
    fn new(kp: f64, ki: f64, kd: f64) -> Self {
        PIDController {
            kp,
            ki,
            kd,
            integral: 0.0,
            prev_error: 0.0,
            prev_time: Instant::now(),
        }
    }

    fn update(&mut self, error: f64) -> f64 {
        let now = Instant::now();
        let dt = now.duration_since(self.prev_time).as_secs_f64();
        self.prev_time = now;

        if dt <= 0.0 {
            return 0.0;
        }

        self.integral += error * dt;
        let derivative = (error - self.prev_error) / dt;
        self.prev_error = error;

        self.kp * error + self.ki * self.integral + self.kd * derivative
    }

    fn reset(&mut self) {
        self.integral = 0.0;
        self.prev_error = 0.0;
        self.prev_time = Instant::now();
    }
}

// Lander state
struct LanderState {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

// Target state
struct TargetState {
    x: f64,
    y: f64,
    width: f64,
    height: f64,
}

fn main() -> io::Result<()> {
    // Create channels for communication between threads
    let (tx, rx) = mpsc::channel();
    let (output_tx, output_rx) = mpsc::channel();

    // Spawn input thread
    let input_thread = thread::spawn(move || {
        let stdin = io::stdin();
        let mut reader = BufReader::new(stdin.lock());

        // Read initial map
        let mut line = String::new();
        reader.read_line(&mut line)?;
        let mut parts = line.split_whitespace();
        let width: usize = parts.next().unwrap().parse().unwrap();
        let height: usize = parts.next().unwrap().parse().unwrap();
        let gravity: f64 = parts.next().unwrap().parse().unwrap();
        let max_thrust: f64 = parts.next().unwrap().parse().unwrap();
        let fuel: f64 = parts.next().unwrap().parse().unwrap();

        reader.read_line(&mut line)?;
        parts = line.split_whitespace();
        let start_x: f64 = parts.next().unwrap().parse().unwrap();
        let start_y: f64 = parts.next().unwrap().parse().unwrap();
        let target_x: f64 = parts.next().unwrap().parse().unwrap();
        let target_y: f64 = parts.next().unwrap().parse().unwrap();

        // Read map (we don't actually need to store it for this controller)
        for _ in 0..height {
            reader.read_line(&mut line)?;
        }

        // Initialize controllers
        let mut vertical_pid = PIDController::new(VERTICAL_KP, VERTICAL_KI, VERTICAL_KD);
        let mut horizontal_pid = PIDController::new(HORIZONTAL_KP, HORIZONTAL_KI, HORIZONTAL_KD);

        // Initialize target
        let target = TargetState {
            x: target_x,
            y: target_y,
            width: TARGET_PAD_SIZE,
            height: TARGET_PAD_SIZE,
        };

        // Main control loop
        loop {
            // Read state update
            line.clear();
            if reader.read_line(&mut line).unwrap_or(0) == 0 {
                break;
            }

            parts = line.split_whitespace();
            let x: f64 = parts.next().unwrap().parse().unwrap();
            let y: f64 = parts.next().unwrap().parse().unwrap();
            let vx: f64 = parts.next().unwrap().parse().unwrap();
            let vy: f64 = parts.next().unwrap().parse().unwrap();
            let angle: f64 = parts.next().unwrap().parse().unwrap();
            let angular_vel: f64 = parts.next().unwrap().parse().unwrap();
            let fuel_remaining: f64 = parts.next().unwrap().parse().unwrap();

            let state = LanderState {
                x,
                y,
                vx,
                vy,
                angle,
                angular_vel,
                fuel_remaining,
            };

            // Calculate errors
            let horizontal_error = target.x - x;
            let vertical_error = target.y - y;

            // Calculate desired velocities (simple approach: maintain small velocity towards target)
            let desired_vx = horizontal_error * 0.5;
            let desired_vy = vertical_error * 0.5;

            // Calculate thrust and turn
            let mut thrust = 0.0;
            let mut turn = 0.0;

            // Vertical control (landing)
            if y < target.y + target.height + 5.0 {
                // Close to landing zone - focus on vertical velocity
                let vertical_velocity_error = desired_vy - vy;
                let vertical_control = vertical_pid.update(vertical_velocity_error);
                thrust = (vertical_control).clamp(0.0, MAX_THRUST);
                
                // If we're close to the target and velocity is low, reduce thrust
                if y > target.y - 2.0 && vy.abs() < 1.0 {
                    thrust *= 0.5;
                }
            } else {
                // Far from landing zone - use more thrust
                let vertical_control = vertical_pid.update(vertical_error);
                thrust = (vertical_control + gravity).clamp(0.0, MAX_THRUST);
            }

            // Horizontal control
            let horizontal_velocity_error = desired_vx - vx;
            let horizontal_control = horizontal_pid.update(horizontal_velocity_error);
            turn = horizontal_control.clamp(-MAX_TURN, MAX_TURN);

            // Adjust angle to face downward (for landing)
            let angle_error = -angle; // We want to face down (angle = 0 is up, so -angle makes us face down)
            turn += angle_error * 0.5;

            // Clamp turn
            turn = turn.clamp(-MAX_TURN, MAX_TURN);

            // Send control output
            if output_tx.send((thrust, turn)).is_err() {
                break;
            }
        }

        Ok(())
    });

    // Spawn output thread
    let output_thread = thread::spawn(move || {
        let stdout = io::stdout();
        let mut handle = stdout.lock();

        loop {
            match output_rx.recv_timeout(OUTPUT_INTERVAL) {
                Ok((thrust, turn)) => {
                    writeln!(handle, "{} {}", thrust, turn).unwrap();
                    handle.flush().unwrap();
                }
                Err(_) => {
                    // Timeout - send zero thrust and turn
                    writeln!(handle, "0.0 0.0").unwrap();
                    handle.flush().unwrap();
                }
            }
        }
    });

    // Main thread - just waits for input thread to finish
    input_thread.join().unwrap().unwrap();
    output_thread.join().unwrap();

    Ok(())
}
```

This implementation:

1. Uses two threads for input and output to avoid blocking issues
2. Implements PID controllers for both vertical and horizontal positioning
3. Adjusts thrust based on distance to target and current velocity
4. Adjusts turn to face the target and maintain proper orientation
5. Outputs control commands at 20Hz (every 50ms)
6. Handles edge cases like running out of fuel or reaching the target

The controller focuses on:
- Maintaining a safe vertical velocity when approaching the landing zone
- Adjusting horizontal position to align with the target
- Rotating to face downward for landing
- Gradually reducing thrust as it approaches the target

The code is optimized for real-time performance and should handle the input/output requirements efficiently.
