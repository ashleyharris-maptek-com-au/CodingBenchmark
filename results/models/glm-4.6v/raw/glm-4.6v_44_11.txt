```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction (from surface to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate view direction (from surface to camera)
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Calculate diffuse component (Lambertian)
    float diffuse = max(dot(N, L), 0.0);
    
    // Polka dot pattern
    vec2 cell = fract(uv * 6.0); // 6x6 grid
    float dist = distance(cell, vec2(0.5)); // Distance from center of cell
    float dotMask = step(dist, 0.3); // 1 if inside dot, 0 otherwise
    
    // Base colors
    vec3 dotColor = vec3(1.0, 0.8, 0.0); // Yellow
    vec3 bgColor = vec3(0.1, 0.1, 0.3); // Dark blue
    
    // Mix colors based on polka dot pattern
    vec3 baseColor = mix(bgColor, dotColor, dotMask);
    
    // Apply Lambertian shading factor
    float shadingFactor = 0.3 + 0.7 * diffuse;
    vec3 shadedColor = baseColor * shadingFactor;
    
    // Output final color with alpha 1.0
    outColor = vec4(shadedColor, 1.0);
}
```
