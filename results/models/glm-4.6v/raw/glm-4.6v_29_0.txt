# Maximum Clique Finder in C#

I'll implement a solution that uses a branch-and-bound algorithm with heuristics to efficiently find the maximum clique in graphs of varying sizes and densities. The solution will include:
1. A Bron-Kerbosch algorithm with pivot selection for efficient clique enumeration
2. Degree-based ordering and pruning
3. Parallel processing for larger graphs
4. Time limit handling with fallback heuristics

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public class MaximumCliqueFinder
{
    private static int[] vertices;
    private static int[] degrees;
    private static int[,] adjacency;
    private static int numVertices;
    private static int numEdges;
    private static int maxCliqueSize;
    private static List<int> maxClique;
    private static Stopwatch timer;
    private static int timeLimit = 5 * 60 * 1000; // 5 minutes in milliseconds

    public static void Main()
    {
        // Read input
        var input = Console.In.ReadToEnd().Trim().Split();
        numVertices = int.Parse(input[0]);
        numEdges = int.Parse(input[1]);

        // Initialize adjacency matrix
        adjacency = new int[numVertices, numVertices];
        vertices = new int[numVertices];
        degrees = new int[numVertices];

        // Read edges
        int index = 2;
        for (int i = 0; i < numEdges; i++)
        {
            int u = int.Parse(input[index++]);
            int v = int.Parse(input[index++]);
            adjacency[u, v] = 1;
            adjacency[v, u] = 1;
            degrees[u]++;
            degrees[v]++;
        }

        // Order vertices by degree (descending)
        vertices = Enumerable.Range(0, numVertices)
            .OrderByDescending(v => degrees[v])
            .ToArray();

        // Initialize
        maxCliqueSize = 0;
        maxClique = new List<int>();
        timer = Stopwatch.StartNew();

        // Try to find maximum clique
        FindMaximumClique();

        // Output result
        Console.WriteLine(maxCliqueSize);
        Console.WriteLine(string.Join(" ", maxClique));
    }

    private static void FindMaximumClique()
    {
        // Try exact algorithm first
        var exactTask = Task.Run(() => BronKerboschWithPivot());
        
        // If time limit is approaching, start heuristic in parallel
        var heuristicTask = Task.Run(() => HeuristicClique());
        
        // Wait for either task to complete
        Task.WaitAny(exactTask, heuristicTask);
        
        // If exact algorithm finished, use its result
        if (exactTask.IsCompleted)
        {
            return;
        }
        
        // Otherwise, use heuristic result
        maxClique = HeuristicClique();
    }

    private static void BronKerboschWithPivot()
    {
        var candidates = new List<int>(vertices);
        var excluded = new List<int>();
        var currentClique = new List<int>();
        
        BronKerbosch(candidates, excluded, currentClique);
    }

    private static void BronKerbosch(List<int> candidates, List<int> excluded, List<int> currentClique)
    {
        // Check time limit
        if (timer.ElapsedMilliseconds > timeLimit * 0.8) // 80% of time for exact algorithm
        {
            return;
        }

        // Prune if current clique + candidates can't beat max
        if (currentClique.Count + candidates.Count <= maxCliqueSize)
        {
            return;
        }

        // If no candidates left, check if this is maximal clique
        if (candidates.Count == 0)
        {
            if (currentClique.Count > maxCliqueSize)
            {
                maxCliqueSize = currentClique.Count;
                maxClique = new List<int>(currentClique);
            }
            return;
        }

        // Select pivot
        int pivot = SelectPivot(candidates, excluded);
        var nonNeighbors = new List<int>();
        
        // Find non-neighbors of pivot
        foreach (var v in candidates)
        {
            if (!IsAdjacent(v, pivot))
            {
                nonNeighbors.Add(v);
            }
        }

        // Iterate through non-neighbors
        foreach (var v in nonNeighbors)
        {
            // Add v to current clique
            currentClique.Add(v);
            
            // Get new candidates (neighbors of v)
            var newCandidates = new List<int>();
            foreach (var u in candidates)
            {
                if (IsAdjacent(u, v))
                {
                    newCandidates.Add(u);
                }
            }
            
            // Get new excluded (neighbors of v in excluded)
            var newExcluded = new List<int>();
            foreach (var u in excluded)
            {
                if (IsAdjacent(u, v))
                {
                    newExcluded.Add(u);
                }
            }
            
            BronKerbosch(newCandidates, newExcluded, currentClique);
            
            // Remove v from current clique
            currentClique.RemoveAt(currentClique.Count - 1);
            
            // Move v from candidates to excluded
            candidates.Remove(v);
            excluded.Add(v);
        }
    }

    private static int SelectPivot(List<int> candidates, List<int> excluded)
    {
        // Select pivot with maximum degree in candidates ∪ excluded
        int maxDegree = -1;
        int pivot = -1;
        
        foreach (var v in candidates.Concat(excluded))
        {
            int degree = 0;
            foreach (var u in candidates)
            {
                if (IsAdjacent(v, u))
                {
                    degree++;
                }
            }
            if (degree > maxDegree)
            {
                maxDegree = degree;
                pivot = v;
            }
        }
        
        return pivot;
    }

    private static bool IsAdjacent(int u, int v)
    {
        return adjacency[u, v] == 1;
    }

    private static List<int> HeuristicClique()
    {
        // Greedy heuristic: start with highest degree vertex and add neighbors
        var clique = new List<int>();
        var available = new HashSet<int>(vertices);
        
        if (available.Count == 0)
            return clique;
            
        // Start with highest degree vertex
        int current = available.First();
        clique.Add(current);
        available.Remove(current);
        
        while (available.Count > 0)
        {
            // Find vertex with maximum common neighbors with current clique
            int bestVertex = -1;
            int maxCommon = -1;
            
            foreach (var v in available)
            {
                int common = 0;
                foreach (var u in clique)
                {
                    if (IsAdjacent(v, u))
                    {
                        common++;
                    }
                }
                
                if (common > maxCommon)
                {
                    maxCommon = common;
                    bestVertex = v;
                }
            }
            
            if (maxCommon == clique.Count) // All vertices in clique are neighbors
            {
                clique.Add(bestVertex);
                available.Remove(bestVertex);
            }
            else
            {
                break;
            }
        }
        
        return clique;
    }
}
```

## Explanation:

1. **Input Handling**: The program reads the graph from standard input, storing it in an adjacency matrix.

2. **Vertex Ordering**: Vertices are ordered by degree in descending order to improve pruning efficiency.

3. **Bron-Kerbosch Algorithm**: 
   - Uses the pivot selection strategy to reduce the number of recursive calls
   - Implements pruning when the potential clique size can't exceed the current maximum
   - Uses a time limit check to prevent excessive computation

4. **Heuristic Fallback**: 
   - If the exact algorithm doesn't complete in time, a greedy heuristic is used
   - The heuristic starts with the highest degree vertex and adds vertices with maximum common neighbors

5. **Parallel Processing**: 
   - Both exact and heuristic algorithms run in parallel
   - The first to complete determines the result

6. **Output**: The size of the maximum clique and its vertices are printed to standard output.

This solution balances between exact computation (for smaller graphs) and heuristic approximation (for larger graphs) to ensure it completes within the time limit while providing the best possible result.
