```typescript
import { performance } from 'perf_hooks';

interface Package {
    id: string;
    x: number;
    y: number;
}

interface Input {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

interface Output {
    routes: string[][];
}

function calculateDistance(p1: [number, number], p2: [number, number]): number {
    const dx = p1[0] - p2[0];
    const dy = p1[1] - p2[1];
    return Math.sqrt(dx * dx + dy * dy);
}

function nearestNeighbor(depot: [number, number], packages: Package[]): string[] {
    const route: string[] = [];
    const unvisited = [...packages];
    let current = depot;

    while (unvisited.length > 0) {
        let nearestIndex = 0;
        let minDistance = Infinity;

        for (let i = 0; i < unvisited.length; i++) {
            const distance = calculateDistance(current, [unvisited[i].x, unvisited[i].y]);
            if (distance < minDistance) {
                minDistance = distance;
                nearestIndex = i;
            }
        }

        const nearestPackage = unvisited[nearestIndex];
        route.push(nearestPackage.id);
        current = [nearestPackage.x, nearestPackage.y];
        unvisited.splice(nearestIndex, 1);
    }

    return route;
}

function solve(input: Input): Output {
    const { depot, truckCount, packages } = input;
    
    // Sort packages by distance from depot (for better initial distribution)
    const packagesWithDistance = packages.map(pkg => ({
        ...pkg,
        distance: calculateDistance(depot, [pkg.x, pkg.y])
    }));
    packagesWithDistance.sort((a, b) => a.distance - b.distance);

    const routes: string[][] = [];
    const assignedPackages = new Set<string>();

    // Distribute packages to trucks using a simple heuristic
    for (let i = 0; i < truckCount && packagesWithDistance.length > 0; i++) {
        const route: string[] = [];
        let current = depot;
        let remainingCapacity = Math.floor(packages.length / truckCount) + (i < packages.length % truckCount ? 1 : 0);

        while (remainingCapacity > 0 && packagesWithDistance.length > 0) {
            // Find nearest unassigned package
            let nearestIndex = 0;
            let minDistance = Infinity;

            for (let j = 0; j < packagesWithDistance.length; j++) {
                if (!assignedPackages.has(packagesWithDistance[j].id)) {
                    const distance = calculateDistance(current, [packagesWithDistance[j].x, packagesWithDistance[j].y]);
                    if (distance < minDistance) {
                        minDistance = distance;
                        nearestIndex = j;
                    }
                }
            }

            const nearestPackage = packagesWithDistance[nearestIndex];
            route.push(nearestPackage.id);
            assignedPackages.add(nearestPackage.id);
            current = [nearestPackage.x, nearestPackage.y];
            packagesWithDistance.splice(nearestIndex, 1);
            remainingCapacity--;
        }

        routes.push(route);
    }

    // Add any remaining packages to the first route
    while (packagesWithDistance.length > 0) {
        routes[0].push(packagesWithDistance[0].id);
        packagesWithDistance.shift();
    }

    return { routes };
}

// Read input from stdin
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Solve the problem
const start = performance.now();
const result = solve(data);
const end = performance.now();

// Output the result
console.log(JSON.stringify(result));
```
