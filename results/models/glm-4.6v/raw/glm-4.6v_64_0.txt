import { Readable } from 'stream';

interface Input {
  probability: number[][];
  drones: { id: number; x: number; y: number }[];
  maxStepsPerDrone: number;
}

interface Output {
  paths: number[][][];
}

class GridPathPlanner {
  private grid: number[][];
  private rows: number;
  private cols: number;
  private drones: { id: number; x: number; y: number }[];
  private maxSteps: number;

  constructor(input: Input) {
    this.grid = input.probability;
    this.rows = this.grid.length;
    this.cols = this.grid[0].length;
    this.drones = input.drones;
    this.maxSteps = input.maxStepsPerDrone;
  }

  private isValid(x: number, y: number): boolean {
    return x >= 0 && x < this.rows && y >= 0 && y < this.cols && this.grid[x][y] !== -1;
  }

  private getNeighbors(x: number, y: number): [number, number][] {
    const neighbors: [number, number][] = [];
    const moves = [
      [0, 0], // stay in place
      [-1, 0], // up
      [1, 0], // down
      [0, -1], // left
      [0, 1], // right
    ];

    for (const [dx, dy] of moves) {
      const nx = x + dx;
      const ny = y + dy;
      if (this.isValid(nx, ny)) {
        neighbors.push([nx, ny]);
      }
    }

    return neighbors;
  }

  private calculateGreedyScore(path: number[][]): number {
    const visited = new Set<string>();
    let score = 0;

    for (const [x, y] of path) {
      const key = `${x},${y}`;
      if (!visited.has(key)) {
        visited.add(key);
        score += this.grid[x][y];
      }
    }

    return score;
  }

  private findPath(startX: number, startY: number): number[][] {
    const path: number[][] = [[startX, startY]];
    const visited = new Set<string>();
    visited.add(`${startX},${startY}`);

    let currentX = startX;
    let currentY = startY;

    for (let step = 1; step < this.maxSteps; step++) {
      const neighbors = this.getNeighbors(currentX, currentY);
      let bestNeighbor: [number, number] | null = null;
      let bestValue = -1;

      for (const [nx, ny] of neighbors) {
        const key = `${nx},${ny}`;
        if (!visited.has(key)) {
          const value = this.grid[nx][ny];
          if (value > bestValue) {
            bestValue = value;
            bestNeighbor = [nx, ny];
          }
        }
      }

      if (bestNeighbor) {
        const [nx, ny] = bestNeighbor;
        path.push([nx, ny]);
        visited.add(`${nx},${ny}`);
        currentX = nx;
        currentY = ny;
      } else {
        break;
      }
    }

    return path;
  }

  public planPaths(): Output {
    const paths: number[][][] = [];

    for (const drone of this.drones) {
      const path = this.findPath(drone.x, drone.y);
      paths.push(path);
    }

    return { paths };
  }
}
