struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 viewNormal = normalize((view * mul(model, float4(input.normal, 1.0))).xyz);

    // Light direction (world space)
    float3 lightDirWorld = normalize(lightPos.xyz - input.worldPos);

    // Compute t
    float t = dot(viewNormal, lightDirWorld);
    t = saturate(t); // clamp to [0,1]

    // Gooch color
    float3 cool = float3(0.2, 0.2, 0.75);
    float3 warm = float3(0.7, 0.7, 0.4);
    float3 goochColor = lerp(cool, warm, t);

    // Per-vertex color based on position
    float3 baseColor = 0.5 + 0.5 * input.worldPos;

    // Mix Gooch and base color
    float3 finalColor = lerp(goochColor, baseColor, 0.5);

    // Output
    return float4(finalColor, 1.0);
}
