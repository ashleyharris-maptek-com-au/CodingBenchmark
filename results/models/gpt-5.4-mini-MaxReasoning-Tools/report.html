static inline double sqDist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist(const Point& a, const Point& b) {
    return std::sqrt(sqDist(a, b));
}

static inline uint64_t doubleToSortable(double v) {
    uint64_t u = std::bit_cast<uint64_t>(v);
    return (u & 0x8000000000000000ULL) ? ~u : (u ^ 0x8000000000000000ULL);
}

static inline uint32_t part1By1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}

static inline uint64_t morton16(uint32_t x, uint32_t y) {
    return static_cast<uint64_t>(part1By1(x)) | (static_cast<uint64_t>(part1By1(y)) << 1);
}

static void radixSortByKey(vector<int>& ord, const vector<uint64_t>& key) {
    const int BITS = 16;
    const int BUCKETS = 1 << BITS;
    const uint64_t MASK = BUCKETS - 1;
    vector<int> tmp(ord.size());
    vector<int> cnt(BUCKETS);

    for (int shift = 0; shift < 64; shift += BITS) {
        std::fill(cnt.begin(), cnt.end(), 0);
        for (int idx : ord) {
            ++cnt[(key[idx] >> shift) & MASK];
        }
        int sum = 0;
        for (int i = 0; i < BUCKETS; ++i) {
            int c = cnt[i];
            cnt[i] = sum;
            sum += c;
        }
        for (int idx : ord) {
            int b = (key[idx] >> shift) & MASK;
            tmp[cnt[b]++] = idx;
        }
        ord.swap(tmp);
    }
}

static inline double routeLength(const vector<int>& ord, const vector<Point>& pts) {
    int n = (int)ord.size();

    for (int i = 0; i < n; ++i) {

    }
    return total;
}

static void orientShortFirst(vector<int>& ord, const vector<Point>& pts) {
    int n = (int)ord.size();
    if (n <= 2) return;
    double d0 = sqDist(pts[ord[0]], pts[ord[1]]);
    double d1 = sqDist(pts[ord[0]], pts[ord[n - 1]]);
    if (d1 < d0) {
        std::reverse(ord.begin() + 1, ord.end());
    }
}

static void twoOptLimited(vector<int>& ord, const vector<Point>& pts, int window, int passes) {

        }
        if (!improved) break;
    }

static vector<int> buildNearestNeighborOrder(const vector<Point>& pts) {

static vector<int> buildProjectionOrder(const vector<Point>& pts, double ux, double uy) {

}

static std::pair<std::pair<double, double>, double> principalAxis(const Stats& st, int n) {
    double mx = st.sx / n;
    double my = st.sy / n;
    double cxx = st.sxx / n - mx * mx;
    double cyy = st.syy / n - my * my;
    double cxy = st.sxy / n - mx * my;

    double disc = std::sqrt(std::max(0.0, (cxx - cyy) * (cxx - cyy) + 4.0 * cxy * cxy));
    double lambda1 = 0.5 * (cxx + cyy + disc);
    double lambda2 = 0.5 * (cxx + cyy - disc);
    double elong = (lambda2 > 1e-18) ? (lambda1 / lambda2) : ((lambda1 > 1e-18) ? 1e300 : 1.0);

    double ux = 1.0;
    double uy = 0.0;
    if (std::abs(cxx - cyy) > 1e-18 || std::abs(cxy) > 1e-18) {
        double theta = 0.5 * std::atan2(2.0 * cxy, cxx - cyy);
        ux = std::cos(theta);
        uy = std::sin(theta);
    }
    return {{ux, uy}, elong};
}

static vector<int> exactHeldKarp(const vector<Point>& pts) {
    int n = (int)pts.size();
    if (n <= 1) return {0};
    if (n == 2) return {0, 1};

    int m = n - 1;
    int totalMasks = 1 << m;
    const double INF = 1e300;

    vector<double> distMat((size_t)n * n, 0.0);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            distMat[(size_t)i * n + j] = dist(pts[i], pts[j]);
        }
    }

    vector<double> dp((size_t)totalMasks * m, INF);
    vector<uint8_t> parent((size_t)totalMasks * m, 255);
    auto id = [m](int mask, int last) -> size_t {
        return (size_t)mask * m + (size_t)last;
    };

    for (int j = 0; j < m; ++j) {
        dp[id(1 << j, j)] = distMat[(size_t)0 * n + (j + 1)];
        parent[id(1 << j, j)] = 255;
    }

    for (int mask = 1; mask < totalMasks; ++mask) {
        for (int last = 0; last < m; ++last) {
            if ((mask & (1 << last)) == 0) continue;
            double cur = dp[id(mask, last)];
            if (cur >= INF / 2) continue;
            int from = last + 1;
            for (int nxt = 0; nxt < m; ++nxt) {
                if (mask & (1 << nxt)) continue;
                int nmask = mask | (1 << nxt);
                double cand = cur + distMat[(size_t)from * n + (nxt + 1)];
                size_t pos = id(nmask, nxt);
                if (cand < dp[pos]) {
                    dp[pos] = cand;
                    parent[pos] = (uint8_t)last;
                }
            }
        }
    }

    int full = totalMasks - 1;
    double best = INF;
    int bestLast = -1;
    for (int last = 0; last < m; ++last) {
        double cand = dp[id(full, last)] + distMat[(size_t)(last + 1) * n + 0];
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }

    vector<int> rev;
    int mask = full;
    int last = bestLast;
    while (true) {
        rev.push_back(last + 1);
        size_t pos = id(mask, last);
        uint8_t prev = parent[pos];
        mask ^= (1 << last);
        if (prev == 255) break;
        last = (int)prev;
    }
    std::reverse(rev.begin(), rev.end());

    vector<int> order;
    order.reserve(n);
    order.push_back(0);
    order.insert(order.end(), rev.begin(), rev.end());
    return order;
}

static vector<int> solveHeuristic(const vector<Point>& pts, const Stats& st) {
    int n = (int)pts.size();
    auto [axis, elong] = principalAxis(st, n);
    double ux = axis.first;
    double uy = axis.second;

    int window = 0;
    int passes = 0;
    if (n <= 1000) {
        window = 64;
        passes = 3;
    } else if (n <= 5000) {
        window = 48;
        passes = 2;
    } else if (n <= 200000) {
        window = 12;
        passes = 1;
    }

    auto evaluate = [&](vector<int> ord, bool doImprove) -> std::pair<vector<int>, double> {
        rotateToZero(ord);
        orientShortFirst(ord, pts);
        if (doImprove && window > 0) {
            twoOptLimited(ord, pts, window, passes);
        }
        double len = routeLength(ord, pts);
        return {std::move(ord), len};
    };

    if (n <= 5000) {
        vector<int> nn = buildNearestNeighborOrder(pts);
        auto best = evaluate(std::move(nn), true);

        auto cand1 = evaluate(buildProjectionOrder(pts, ux, uy), true);
        if (cand1.second < best.second) best = std::move(cand1);

        auto cand2 = evaluate(buildMortonOrder(pts, st), true);
        if (cand2.second < best.second) best = std::move(cand2);

        return best.first;
    }

    auto candidateProjection = buildProjectionOrder(pts, ux, uy);
    auto candidateMorton = buildMortonOrder(pts, st);

    if (n <= 200000) {
        auto best = evaluate(std::move(candidateProjection), true);
        auto cand2 = evaluate(std::move(candidateMorton), true);
        if (cand2.second < best.second) best = std::move(cand2);
        return best.first;
    }

    bool useProjection = (elong > 4.0);
    if (useProjection) {
        return evaluate(std::move(candidateProjection), false).first;
    }
    return evaluate(std::move(candidateMorton), false).first;
}

            p.y = y;

fn build_piece_plans(
    pieces: Vec<Piece>,
    cell: f64,
    gw: usize,
    gh: usize,
) -> Vec<PiecePlan> {
    let mut plans = Vec::with_capacity(pieces.len());
    for piece in pieces {
        let mut op = Vec::new();
        let mut best = usize::MAX;
        for &o in &piece.orients {
            let cw = ((o.w / cell).ceil() as usize).max(1);
            let ch = ((o.h / cell).ceil() as usize).max(1);
            if cw > gw || ch > gh {
                continue;
            }
            let cell_area = cw * ch;
            best = best.min(cell_area);
            op.push(OrientPlan {
                rot: o.rot,
                minx: o.minx,
                miny: o.miny,
                w: o.w,
                h: o.h,
                cw,
                ch,
                cell_area,
            });
        }
        if op.is_empty() {
            // Keep at least one orientation so the solver can still attempt exact fallback.
            let o = piece.orients[0];
            let cw = ((o.w / cell).ceil() as usize).max(1);
            let ch = ((o.h / cell).ceil() as usize).max(1);
            best = cw.saturating_mul(ch);
            op.push(OrientPlan {
                rot: o.rot,
                minx: o.minx,
                miny: o.miny,
                w: o.w,
                h: o.h,
                cw,
                ch,
                cell_area: cw * ch,
            });
        }
        op.sort_by(|a, b| {
            a.cell_area
                .cmp(&b.cell_area)
                .then_with(|| a.h.partial_cmp(&b.h).unwrap_or(Ordering::Equal))
                .then_with(|| a.w.partial_cmp(&b.w).unwrap_or(Ordering::Equal))
                .then_with(|| a.rot.partial_cmp(&b.rot).unwrap_or(Ordering::Equal))
        });
        plans.push(PiecePlan {
            idx: piece.idx,
            area: piece.area,
            best_cell_area: best,
            plans: op,
        });
    }
    plans
}

        let x_limit = grid.gw - plan.cw;
        let y_limit = grid.gh - plan.ch;
        for y in 0..=y_limit {
            let mut inter = row_mask_all;
            for r in 0..plan.ch {
                inter &= bin.free_rows[y + r];
                if inter == 0 {
                    break;
                }
            }
            if inter == 0 {
                continue;
            }
            if let Some(x) = first_run(inter, grid.gw, plan.cw, x_limit) {

            }
        }
    }

fn first_run(mask: u128, width: usize, run_len: usize, x_limit: usize) -> Option<usize> {
    let mut count = 0usize;
    let mut bits = mask;
    for x in 0..width {
        if bits & 1 != 0 {
            count += 1;
            if count >= run_len {
                let start = x + 1 - run_len;
                if start <= x_limit {
                    return Some(start);
                }
            }
        } else {
            count = 0;
        }
        bits >>= 1;
    }
    None
}

fn try_place_exact(
    bin: &Bin,
    piece: &PiecePlan,
    grid: &StockGrid,
    poly: &[Pt],
    edges: &[Edge],
    step: f64,
) -> Option<(Rect, OrientPlan)> {
    let max_x = grid.maxx;
    let max_y = grid.maxy;
    let phases = [0.0, step * 0.5];
    for plan in &piece.plans {
        for &oy in &phases {
            let mut y = grid.miny + oy;
            while y + plan.h <= max_y + grid.geom_eps {
                for &ox in &phases {
                    let mut x = grid.minx + ox;
                    while x + plan.w <= max_x + grid.geom_eps {
                        let rect = Rect {
                            x,
                            y,
                            w: plan.w,
                            h: plan.h,
                        };
                        if rect_inside_poly_exact(rect, poly, edges, grid.geom_eps)
                            && rect_cells_free(bin, grid, rect)
                        {
                            return Some((rect, plan.clone()));
                        }
                        x += step;
                    }
                }
                y += step;
            }
        }
    }
    None
}

fn rect_cells_free(bin: &Bin, grid: &StockGrid, rect: Rect) -> bool {
    let (x0, x1, y0, y1) = rect_cell_bounds(grid, rect);
    if x0 >= x1 || y0 >= y1 {
        return false;
    }
    let mask = range_mask(x0, x1);
    for row in y0..y1 {
        if bin.free_rows[row] & mask != mask {
            return false;
        }
    }
    true
}

fn mark_rect_cells(bin_rows: &mut [u128], grid: &StockGrid, rect: Rect) {
    let (x0, x1, y0, y1) = rect_cell_bounds(grid, rect);
    if x0 >= x1 || y0 >= y1 {
        return;
    }
    let mask = range_mask(x0, x1);
    for row in y0..y1 {
        bin_rows[row] &= !mask;
    }
}

fn rect_cell_bounds(grid: &StockGrid, rect: Rect) -> (usize, usize, usize, usize) {
    let mut x0 = ((rect.x - grid.minx) / grid.cell).floor() as isize;
    let mut x1 = ((rect.x + rect.w - grid.minx) / grid.cell).ceil() as isize;
    let mut y0 = ((rect.y - grid.miny) / grid.cell).floor() as isize;
    let mut y1 = ((rect.y + rect.h - grid.miny) / grid.cell).ceil() as isize;

    x0 = x0.max(0).min(grid.gw as isize);
    x1 = x1.max(0).min(grid.gw as isize);
    y0 = y0.max(0).min(grid.gh as isize);
    y1 = y1.max(0).min(grid.gh as isize);

    (x0 as usize, x1 as usize, y0 as usize, y1 as usize)
}

fn range_mask(x0: usize, x1: usize) -> u128 {
    if x0 >= x1 {
        0
    } else {
        let len = x1 - x0;
        let base = if len >= 128 {
            u128::MAX
        } else {
            (1u128 << len) - 1
        };
        base << x0
    }
}

fn build_stock_grid(
    poly: &[Pt],
    edges: &[Edge],
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
    geom_eps: f64,
) -> StockGrid {
    let bbox_w = (maxx - minx).max(1e-12);
    let bbox_h = (maxy - miny).max(1e-12);
    let cell = (bbox_w.max(bbox_h) / GRID_DIM as f64).max(1e-12);
    let gw = ((bbox_w / cell).floor() as usize).max(1).min(128);
    let gh = ((bbox_h / cell).floor() as usize).max(1).min(128);

    let mut buckets = vec![Vec::<usize>::new(); gw * gh];
    for (ei, e) in edges.iter().enumerate() {
        let mut ix0 = ((e.minx - minx) / cell).floor() as isize;
        let mut ix1 = ((e.maxx - minx) / cell).floor() as isize;
        let mut iy0 = ((e.miny - miny) / cell).floor() as isize;
        let mut iy1 = ((e.maxy - miny) / cell).floor() as isize;
        ix0 = ix0.max(0).min((gw - 1) as isize);
        ix1 = ix1.max(0).min((gw - 1) as isize);
        iy0 = iy0.max(0).min((gh - 1) as isize);
        iy1 = iy1.max(0).min((gh - 1) as isize);
        for y in iy0..=iy1 {
            for x in ix0..=ix1 {
                buckets[(y as usize) * gw + (x as usize)].push(ei);
            }
        }
    }

    let mut safe_rows = vec![0u128; gh];
    for j in 0..gh {
        for i in 0..gw {
            let x0 = minx + (i as f64) * cell;
            let y0 = miny + (j as f64) * cell;
            let x1 = x0 + cell;
            let y1 = y0 + cell;
            let corners = [
                Pt { x: x0, y: y0 },
                Pt { x: x1, y: y0 },
                Pt { x: x1, y: y1 },
                Pt { x: x0, y: y1 },
            ];
            if corners.iter().any(|&p| !point_in_poly(p, poly, geom_eps)) {
                continue;
            }
            let mut safe = true;
            for &ei in &buckets[j * gw + i] {
                let e = &edges[ei];
                if edge_intersects_rect_proper(e.a, e.b, x0, y0, x1, y1, geom_eps) {
                    safe = false;
                    break;
                }
            }
            if safe {
                safe_rows[j] |= 1u128 << i;
            }
        }
    }

    StockGrid {
        minx,
        miny,
        maxx,
        maxy,
        cell,
        gw,
        gh,
        safe_rows,
        edges: edges.to_vec(),
        geom_eps,
    }
}

fn rect_inside_poly_exact(rect: Rect, poly: &[Pt], edges: &[Edge], eps: f64) -> bool {
    let corners = [
        Pt {
            x: rect.x,
            y: rect.y,
        },
        Pt {
            x: rect.x + rect.w,
            y: rect.y,
        },
        Pt {
            x: rect.x + rect.w,
            y: rect.y + rect.h,
        },
        Pt {
            x: rect.x,
            y: rect.y + rect.h,
        },
    ];
    if corners.iter().any(|&p| !point_in_poly(p, poly, eps)) {
        return false;
    }

    let y0 = rect.y;
    let x1 = rect.x + rect.w;
    let y1 = rect.y + rect.h;
    for e in edges {
        if edge_intersects_rect_proper(e.a, e.b, x0, y0, x1, y1, eps) {
            return false;
        }
    }
    true
}

fn edge_intersects_rect_proper(a: Pt, b: Pt, x0: f64, y0: f64, x1: f64, y1: f64, eps: f64) -> bool {
    let r1 = Pt { x: x0, y: y0 };
    let r2 = Pt { x: x1, y: y0 };
    let r3 = Pt { x: x1, y: y1 };
    let r4 = Pt { x: x0, y: y1 };
    proper_segment_intersection(a, b, r1, r2, eps)
        || proper_segment_intersection(a, b, r2, r3, eps)
        || proper_segment_intersection(a, b, r3, r4, eps)
        || proper_segment_intersection(a, b, r4, r1, eps)
}

fn proper_segment_intersection(a: Pt, b: Pt, c: Pt, d: Pt, eps: f64) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() <= eps || o2.abs() <= eps || o3.abs() <= eps || o4.abs() <= eps {
        return false;
    }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly(p: Pt, poly: &[Pt], eps: f64) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b, eps) {
            return true;
        }
        let yi = a.y > p.y;
        let yj = b.y > p.y;
        if yi != yj {
            let x_int = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x_int >= p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_on_segment(p: Pt, a: Pt, b: Pt, eps: f64) -> bool {
    let cross = orient(a, b, p).abs();
    if cross > eps {
        return false;
    }
    let minx = a.x.min(b.x) - eps;
    let maxx = a.x.max(b.x) + eps;
    let miny = a.y.min(b.y) - eps;
    let maxy = a.y.max(b.y) + eps;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn build_edges(poly: &[Pt]) -> Vec<Edge> {
    let mut edges = Vec::with_capacity(poly.len());
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        edges.push(Edge {
            a,
            b,
            minx: a.x.min(b.x),
            maxx: a.x.max(b.x),
            miny: a.y.min(b.y),
            maxy: a.y.max(b.y),
        });
    }
    edges
}

fn polygon_area_signed(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut acc = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        acc += a.x * b.y - a.y * b.x;
    }
    acc * 0.5
}

fn polygon_bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn rotated_bbox(poly: &[Pt], angle: f64) -> (f64, f64, f64, f64) {
    let c = angle.cos();
    let s = angle.sin();
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        let rx = p.x * c - p.y * s;
        let ry = p.x * s + p.y * c;
        minx = minx.min(rx);
        miny = miny.min(ry);
        maxx = maxx.max(rx);
        maxy = maxy.max(ry);
    }
    (minx, miny, maxx - minx, maxy - miny)
}

            });
        }
        if op.is_empty() {
            // Keep at least one orientation so the solver can still attempt exact fallback.
            let o = piece.orients[0];
            let cw = ((o.w / cell).ceil() as usize).max(1);
            let ch = ((o.h / cell).ceil() as usize).max(1);
            best = cw.saturating_mul(ch);
            op.push(OrientPlan {
                rot: o.rot,
                minx: o.minx,
                miny: o.miny,
                w: o.w,
                h: o.h,
                cw,
                ch,
                cell_area: cw * ch,
            });
        }
        op.sort_by(|a, b| {
            a.cell_area
                .cmp(&b.cell_area)
                .then_with(|| a.h.partial_cmp(&b.h).unwrap_or(Ordering::Equal))
                .then_with(|| a.w.partial_cmp(&b.w).unwrap_or(Ordering::Equal))
                .then_with(|| a.rot.partial_cmp(&b.rot).unwrap_or(Ordering::Equal))
        });
        plans.push(PiecePlan {
            idx: piece.idx,
            area: piece.area,
            best_cell_area: best,
            plans: op,
        });
    }
    plans
}

fn try_place_grid(
    bin: &Bin,
    piece: &PiecePlan,
    grid: &StockGrid,
    row_mask_all: u128,
) -> Option<(usize, usize, OrientPlan)> {
    for plan in &piece.plans {
        if plan.cw > grid.gw || plan.ch > grid.gh {
            continue;
        }
        let x_limit = grid.gw - plan.cw;
        let y_limit = grid.gh - plan.ch;
        for y in 0..=y_limit {
            let mut inter = row_mask_all;
            for r in 0..plan.ch {
                inter &= bin.free_rows[y + r];
                if inter == 0 {
                    break;
                }
            }
            if inter == 0 {
                continue;
            }
            if let Some(x) = first_run(inter, grid.gw, plan.cw, x_limit) {
                return Some((x, y, plan.clone()));
            }
        }
    }
    None
}

fn first_run(mask: u128, width: usize, run_len: usize, x_limit: usize) -> Option<usize> {
    let mut count = 0usize;
    let mut bits = mask;
    for x in 0..width {
        if bits & 1 != 0 {
            count += 1;
            if count >= run_len {
                let start = x + 1 - run_len;
                if start <= x_limit {
                    return Some(start);
                }
            }
        } else {
            count = 0;
        }
        bits >>= 1;
    }
    None
}

fn try_place_exact(
    bin: &Bin,
    piece: &PiecePlan,
    grid: &StockGrid,
    poly: &[Pt],
    edges: &[Edge],
    step: f64,
) -> Option<(Rect, OrientPlan)> {
    let max_x = grid.maxx;
    let max_y = grid.maxy;
    let phases = [0.0, step * 0.5];
    for plan in &piece.plans {
        for &oy in &phases {
            let mut y = grid.miny + oy;
            while y + plan.h <= max_y + grid.geom_eps {
                for &ox in &phases {
                    let mut x = grid.minx + ox;
                    while x + plan.w <= max_x + grid.geom_eps {
                        let rect = Rect {
                            x,
                            y,
                            w: plan.w,
                            h: plan.h,
                        };
                        if rect_inside_poly_exact(rect, poly, edges, grid.geom_eps)
                            && rect_cells_free(bin, grid, rect)
                        {
                            return Some((rect, plan.clone()));
                        }
                        x += step;
                    }
                }
                y += step;
            }
        }
    }
    None
}

fn rect_cells_free(bin: &Bin, grid: &StockGrid, rect: Rect) -> bool {
    let (x0, x1, y0, y1) = rect_cell_bounds(grid, rect);
    if x0 >= x1 || y0 >= y1 {
        return false;
    }
    let mask = range_mask(x0, x1);
    for row in y0..y1 {
        if bin.free_rows[row] & mask != mask {
            return false;
        }
    }
    true
}

fn mark_placement_cells(bin_rows: &mut [u128], grid: &StockGrid, rect_x: f64, rect_y: f64, w: f64, h: f64) {
    let rect = Rect {
        x: rect_x,
        y: rect_y,
        w,
        h,
    };
    mark_rect_cells(bin_rows, grid, rect);
}

fn mark_rect_cells(bin_rows: &mut [u128], grid: &StockGrid, rect: Rect) {
    let (x0, x1, y0, y1) = rect_cell_bounds(grid, rect);
    if x0 >= x1 || y0 >= y1 {
        return;
    }
    let mask = range_mask(x0, x1);
    for row in y0..y1 {
        bin_rows[row] &= !mask;
    }
}

fn rect_cell_bounds(grid: &StockGrid, rect: Rect) -> (usize, usize, usize, usize) {
    let mut x0 = ((rect.x - grid.minx) / grid.cell).floor() as isize;
    let mut x1 = ((rect.x + rect.w - grid.minx) / grid.cell).ceil() as isize;
    let mut y0 = ((rect.y - grid.miny) / grid.cell).floor() as isize;
    let mut y1 = ((rect.y + rect.h - grid.miny) / grid.cell).ceil() as isize;

    x0 = x0.max(0).min(grid.gw as isize);
    x1 = x1.max(0).min(grid.gw as isize);
    y0 = y0.max(0).min(grid.gh as isize);
    y1 = y1.max(0).min(grid.gh as isize);

    (x0 as usize, x1 as usize, y0 as usize, y1 as usize)
}

fn range_mask(x0: usize, x1: usize) -> u128 {
    if x0 >= x1 {
        0
    } else {
        let len = x1 - x0;
        let base = if len >= 128 {
            u128::MAX
        } else {
            (1u128 << len) - 1
        };
        base << x0
    }
}

fn build_stock_grid(
    poly: &[Pt],
    edges: &[Edge],
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
    geom_eps: f64,
) -> StockGrid {
    let bbox_w = (maxx - minx).max(1e-12);
    let bbox_h = (maxy - miny).max(1e-12);
    let cell = (bbox_w.max(bbox_h) / GRID_DIM as f64).max(1e-12);
    let gw = ((bbox_w / cell).floor() as usize).max(1).min(128);
    let gh = ((bbox_h / cell).floor() as usize).max(1).min(128);

    let mut buckets = vec![Vec::<usize>::new(); gw * gh];
    for (ei, e) in edges.iter().enumerate() {
        let mut ix0 = ((e.minx - minx) / cell).floor() as isize;
        let mut ix1 = ((e.maxx - minx) / cell).floor() as isize;
        let mut iy0 = ((e.miny - miny) / cell).floor() as isize;
        let mut iy1 = ((e.maxy - miny) / cell).floor() as isize;
        ix0 = ix0.max(0).min((gw - 1) as isize);
        ix1 = ix1.max(0).min((gw - 1) as isize);
        iy0 = iy0.max(0).min((gh - 1) as isize);
        iy1 = iy1.max(0).min((gh - 1) as isize);
        for y in iy0..=iy1 {
            for x in ix0..=ix1 {
                buckets[(y as usize) * gw + (x as usize)].push(ei);
            }
        }
    }

    let mut safe_rows = vec![0u128; gh];
    for j in 0..gh {
        for i in 0..gw {
            let x0 = minx + (i as f64) * cell;
            let y0 = miny + (j as f64) * cell;
            let x1 = x0 + cell;
            let y1 = y0 + cell;
            let corners = [
                Pt { x: x0, y: y0 },
                Pt { x: x1, y: y0 },
                Pt { x: x1, y: y1 },
                Pt { x: x0, y: y1 },
            ];
            if corners.iter().any(|&p| !point_in_poly(p, poly, geom_eps)) {
                continue;
            }
            let mut safe = true;
            for &ei in &buckets[j * gw + i] {
                let e = &edges[ei];
                if edge_intersects_rect_proper(e.a, e.b, x0, y0, x1, y1, geom_eps) {
                    safe = false;
                    break;
                }
            }
            if safe {
                safe_rows[j] |= 1u128 << i;
            }
        }
    }

    StockGrid {
        minx,
        miny,
        maxx,
        maxy,
        cell,
        gw,
        gh,
        safe_rows,
        edges: edges.to_vec(),
        geom_eps,
    }
}

fn rect_inside_poly_exact(rect: Rect, poly: &[Pt], edges: &[Edge], eps: f64) -> bool {
    let corners = [
        Pt {
            x: rect.x,
            y: rect.y,
        },
        Pt {
            x: rect.x + rect.w,
            y: rect.y,
        },
        Pt {
            x: rect.x + rect.w,
            y: rect.y + rect.h,
        },
        Pt {
            x: rect.x,
            y: rect.y + rect.h,
        },
    ];
    if corners.iter().any(|&p| !point_in_poly(p, poly, eps)) {
        return false;
    }
    let x0 = rect.x;
    let y0 = rect.y;
    let x1 = rect.x + rect.w;
    let y1 = rect.y + rect.h;
    for e in edges {
        if edge_intersects_rect_proper(e.a, e.b, x0, y0, x1, y1, eps) {
            return false;
        }
    }
    true
}

fn edge_intersects_rect_proper(a: Pt, b: Pt, x0: f64, y0: f64, x1: f64, y1: f64, eps: f64) -> bool {
    let r1 = Pt { x: x0, y: y0 };
    let r2 = Pt { x: x1, y: y0 };
    let r3 = Pt { x: x1, y: y1 };
    let r4 = Pt { x: x0, y: y1 };
    proper_segment_intersection(a, b, r1, r2, eps)
        || proper_segment_intersection(a, b, r2, r3, eps)
        || proper_segment_intersection(a, b, r3, r4, eps)
        || proper_segment_intersection(a, b, r4, r1, eps)
}

fn proper_segment_intersection(a: Pt, b: Pt, c: Pt, d: Pt, eps: f64) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() <= eps || o2.abs() <= eps || o3.abs() <= eps || o4.abs() <= eps {
        return false;
    }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly(p: Pt, poly: &[Pt], eps: f64) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b, eps) {
            return true;
        }
        let yi = a.y > p.y;
        let yj = b.y > p.y;
        if yi != yj {
            let x_int = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x_int >= p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_on_segment(p: Pt, a: Pt, b: Pt, eps: f64) -> bool {
    let cross = orient(a, b, p).abs();
    if cross > eps {
        return false;
    }
    let minx = a.x.min(b.x) - eps;
    let maxx = a.x.max(b.x) + eps;
    let miny = a.y.min(b.y) - eps;
    let maxy = a.y.max(b.y) + eps;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn build_edges(poly: &[Pt]) -> Vec<Edge> {
    let mut edges = Vec::with_capacity(poly.len());
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        edges.push(Edge {
            a,
            b,
            minx: a.x.min(b.x),
            maxx: a.x.max(b.x),
            miny: a.y.min(b.y),
            maxy: a.y.max(b.y),
        });
    }
    edges
}

fn polygon_area_signed(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut acc = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        acc += a.x * b.y - a.y * b.x;
    }
    acc * 0.5
}

fn polygon_bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn rotated_bbox(poly: &[Pt], angle: f64) -> (f64, f64, f64, f64) {
    let c = angle.cos();
    let s = angle.sin();
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        let rx = p.x * c - p.y * s;
        let ry = p.x * s + p.y * c;
        minx = minx.min(rx);
        miny = miny.min(ry);
        maxx = maxx.max(rx);
        maxy = maxy.max(ry);
    }
    (minx, miny, maxx - minx, maxy - miny)
}
