using System;
using System.Collections.Generic;
using System.Globalization;
using System.Numerics;
using System.Text;

public static class Program
{
    private const double Eps = 1e-12;
    private const double TwoPi = Math.PI * 2.0;

    public static void Main()
    {
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            return;
        }

        input = input.TrimStart('\uFEFF');
        string[] tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;

        int n = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        double mu = double.Parse(tokens[p++], CultureInfo.InvariantCulture);

        var nodes = new OrbitInfo[n + 1];
        nodes[0] = BuildOrbit(
            double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            double.Parse(tokens[p++], CultureInfo.InvariantCulture),
            mu);

        for (int i = 1; i <= n; i++)
        {
            nodes[i] = BuildOrbit(
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                double.Parse(tokens[p++], CultureInfo.InvariantCulture),
                mu);
        }

        double[][] cost = BuildCostMatrix(nodes, mu);
        int[] answer = n <= 20 ? SolveExact(cost, n) : SolveHeuristic(cost, n);

        var sb = new StringBuilder();
        for (int i = 0; i < answer.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(answer[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static int[] SolveExact(double[][] cost, int n)
    {
        if (n == 0)
        {
            return Array.Empty<int>();
        }

        int states = 1 << n;
        int total = states * n;
        double[] dp = new double[total];
        int[] parent = new int[total];
        Array.Fill(dp, double.PositiveInfinity);
        Array.Fill(parent, -1);

        for (int j = 0; j < n; j++)
        {
            int mask = 1 << j;
            dp[mask * n + j] = cost[0][j + 1];
        }

        for (int mask = 1; mask < states; mask++)
        {
            if ((mask & (mask - 1)) == 0)
            {
                continue;
            }

            int baseIndex = mask * n;
            int bits = mask;
            while (bits != 0)
            {
                int j = BitOperations.TrailingZeroCount((uint)bits);
                bits &= bits - 1;

                int prevMask = mask ^ (1 << j);
                int prevBase = prevMask * n;
                int pBits = prevMask;
                double best = double.PositiveInfinity;
                int bestPrev = -1;

                while (pBits != 0)
                {
                    int k = BitOperations.TrailingZeroCount((uint)pBits);
                    pBits &= pBits - 1;

                    double cand = dp[prevBase + k] + cost[k + 1][j + 1];
                    if (cand < best)
                    {
                        best = cand;
                        bestPrev = k;
                    }
                }

                dp[baseIndex + j] = best;
                parent[baseIndex + j] = bestPrev;
            }
        }

        int fullMask = states - 1;
        int fullBase = fullMask * n;
        double bestCost = double.PositiveInfinity;
        int end = -1;

        for (int j = 0; j < n; j++)
        {
            double c = dp[fullBase + j];
            if (c < bestCost)
            {
                bestCost = c;
                end = j;
            }
        }

        int[] order = new int[n];
        int mask2 = fullMask;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            order[pos] = end;
            int prev = parent[mask2 * n + end];
            mask2 ^= 1 << end;
            end = prev;
        }

        return order;
    }

    private static int[] SolveHeuristic(double[][] cost, int n)
    {
        var rng = new Random(123456789);
        var starters = new List<List<int>>
        {
            BuildNearestNeighbor(cost, n, false),
            BuildNearestNeighbor(cost, n, true),
            BuildCheapestInsertion(cost, n),
            BuildFarthestInsertion(cost, n),
            BuildRandomRoute(n, rng),
            BuildRandomRoute(n, rng),
        };

        List<int> bestRoute = null;
        double bestCost = double.PositiveInfinity;

        foreach (var route in starters)
        {
            ImproveRoute(route, cost);
            double c = RouteCost(route, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestRoute = new List<int>(route);
            }
        }

        for (int iter = 0; iter < 10; iter++)
        {
            var candidate = new List<int>(bestRoute);
            Shake(candidate, rng);
            ImproveRoute(candidate, cost);
            double c = RouteCost(candidate, cost);
            if (c < bestCost)
            {
                bestCost = c;
                bestRoute = candidate;
            }
        }

        return bestRoute.ToArray();
    }

    private static void ImproveRoute(List<int> route, double[][] cost)
    {
        if (route.Count < 2)
        {
            return;
        }

        double current = RouteCost(route, cost);
        for (int pass = 0; pass < 12; pass++)
        {
            bool changed = false;
            if (TryTwoOpt(route, cost, ref current)) changed = true;
            if (TryRelocate(route, cost, ref current)) changed = true;
            if (TrySwap(route, cost, ref current)) changed = true;
            if (!changed)
            {
                break;
            }
        }
    }

    private static bool TryTwoOpt(List<int> route, double[][] cost, ref double currentCost)
    {
        int n = route.Count;
        for (int i = 0; i < n - 1; i++)
        {
            int prev = i == 0 ? -1 : route[i - 1];
            int a = route[i];
            for (int j = i + 1; j < n; j++)
            {
                int b = route[j];
                int next = j + 1 < n ? route[j + 1] : -1;

                double oldCost = EdgeCost(cost, prev, a) + EdgeCost(cost, b, next);
                double newCost = EdgeCost(cost, prev, b) + EdgeCost(cost, a, next);

                if (newCost + Eps < oldCost)
                {
                    route.Reverse(i, j - i + 1);
                    currentCost += newCost - oldCost;
                    return true;
                }
            }
        }
        return false;
    }

    private static bool TrySwap(List<int> route, double[][] cost, ref double currentCost)
    {
        int n = route.Count;
        for (int i = 0; i < n - 1; i++)
        {
            for (int j = i + 1; j < n; j++)
            {
                (route[i], route[j]) = (route[j], route[i]);
                double candidate = RouteCost(route, cost);
                if (candidate + Eps < currentCost)
                {
                    currentCost = candidate;
                    return true;
                }
                (route[i], route[j]) = (route[j], route[i]);
            }
        }
        return false;
    }

    private static bool TryRelocate(List<int> route, double[][] cost, ref double currentCost)
    {
        int n = route.Count;
        for (int i = 0; i < n; i++)
        {
            int node = route[i];
            route.RemoveAt(i);
            int limit = n - 1;
            for (int pos = 0; pos <= limit; pos++)
            {
                if (pos == i)
                {
                    continue;
                }

                route.Insert(pos, node);
                double candidate = RouteCost(route, cost);
                if (candidate + Eps < currentCost)
                {
                    currentCost = candidate;
                    return true;
                }
                route.RemoveAt(pos);
            }
            route.Insert(i, node);
        }
        return false;
    }

    private static void Shake(List<int> route, Random rng)
    {
        int n = route.Count;
        if (n < 2)
        {
            return;
        }

        for (int step = 0; step < 3; step++)
        {
            int op = rng.Next(3);
            if (op == 0)
            {
                int i = rng.Next(n);
                int j = rng.Next(n);
                if (i != j)
                {
                    (route[i], route[j]) = (route[j], route[i]);
                }
            }
            else if (op == 1)
            {
                int i = rng.Next(n);
                int j = rng.Next(n);
                if (i > j)
                {
                    (i, j) = (j, i);
                }
                if (i < j)
                {
                    route.Reverse(i, j - i + 1);
                }
            }
            else
            {
                int i = rng.Next(n);
                int node = route[i];
                route.RemoveAt(i);
                int pos = rng.Next(n);
                route.Insert(pos, node);
            }
        }
    }

    private static List<int> BuildNearestNeighbor(double[][] cost, int n, bool farthestFirst)
    {
        var used = new bool[n];
        int first = 0;
        double best = farthestFirst ? double.NegativeInfinity : double.PositiveInfinity;

        for (int i = 0; i < n; i++)
        {
            double c = cost[0][i + 1];
            if ((!farthestFirst && c < best) || (farthestFirst && c > best))
            {
                best = c;
                first = i;
            }
        }

        var route = new List<int>(n) { first };
        used[first] = true;

        while (route.Count < n)
        {
            int last = route[route.Count - 1];
            int pick = -1;
            double pickCost = double.PositiveInfinity;

            for (int i = 0; i < n; i++)
            {
                if (!used[i])
                {
                    double c = cost[last + 1][i + 1];
                    if (c < pickCost)
                    {
                        pickCost = c;
                        pick = i;
                    }
                }
            }

            route.Add(pick);
            used[pick] = true;
        }

        return route;
    }

    private static List<int> BuildCheapestInsertion(double[][] cost, int n)
    {
        if (n == 1)
        {
            return new List<int> { 0 };
        }

        var used = new bool[n];
        int first = -1;
        int second = -1;
        double best = double.PositiveInfinity;

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                double c = cost[0][i + 1] + cost[i + 1][j + 1];
                if (c < best)
                {
                    best = c;
                    first = i;
                    second = j;
                }
            }
        }

        var route = new List<int>(n) { first, second };
        used[first] = true;
        used[second] = true;

        while (route.Count < n)
        {
            int bestNode = -1;
            int bestPos = -1;
            double bestDelta = double.PositiveInfinity;

            for (int node = 0; node < n; node++)
            {
                if (used[node]) continue;
                for (int pos = 0; pos <= route.Count; pos++)
                {
                    double delta = InsertionDelta(route, pos, node, cost);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestNode = node;
                        bestPos = pos;
                    }
                }
            }

            route.Insert(bestPos, bestNode);
            used[bestNode] = true;
        }

        return route;
    }

    private static List<int> BuildFarthestInsertion(double[][] cost, int n)
    {
        if (n == 1)
        {
            return new List<int> { 0 };
        }

        var used = new bool[n];
        int first = 0;
        double firstCost = double.NegativeInfinity;

        for (int i = 0; i < n; i++)
        {
            double c = cost[0][i + 1];
            if (c > firstCost)
            {
                firstCost = c;
                first = i;
            }
        }

        int second = -1;
        double secondCost = double.NegativeInfinity;
        for (int j = 0; j < n; j++)
        {
            if (j == first) continue;
            double c = cost[first + 1][j + 1];
            if (c > secondCost)
            {
                secondCost = c;
                second = j;
            }
        }

        var route = new List<int>(n) { first, second };
        used[first] = true;
        used[second] = true;

        while (route.Count < n)
        {
            int chosen = -1;
            double farthest = double.NegativeInfinity;

            for (int node = 0; node < n; node++)
            {
                if (used[node]) continue;
                double nearest = cost[0][node + 1];
                for (int i = 0; i < route.Count; i++)
                {
                    double c = cost[route[i] + 1][node + 1];
                    if (c < nearest)
                    {
                        nearest = c;
                    }
                }

                if (nearest > farthest)
                {
                    farthest = nearest;
                    chosen = node;
                }
            }

            int bestPos = 0;
            double bestDelta = double.PositiveInfinity;
            for (int pos = 0; pos <= route.Count; pos++)
            {
                double delta = InsertionDelta(route, pos, chosen, cost);
                if (delta < bestDelta)
                {
                    bestDelta = delta;
                    bestPos = pos;
                }
            }

            route.Insert(bestPos, chosen);
            used[chosen] = true;
        }

        return route;
    }

    private static List<int> BuildRandomRoute(int n, Random rng)
    {
        var route = new List<int>(n);
        for (int i = 0; i < n; i++)
        {
            route.Add(i);
        }

        for (int i = n - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            (route[i], route[j]) = (route[j], route[i]);
        }

        return route;
    }

    private static double InsertionDelta(List<int> route, int pos, int node, double[][] cost)
    {
        int prev = pos == 0 ? -1 : route[pos - 1];
        int next = pos == route.Count ? -1 : route[pos];
        return EdgeCost(cost, prev, node) + EdgeCost(cost, node, next) - EdgeCost(cost, prev, next);
    }

    private static double RouteCost(List<int> route, double[][] cost)
    {
        if (route.Count == 0)
        {
            return 0.0;
        }

        double total = cost[0][route[0] + 1];
        for (int i = 1; i < route.Count; i++)
        {
            total += cost[route[i - 1] + 1][route[i] + 1];
        }
        return total;
    }

    private static double EdgeCost(double[][] cost, int from, int to)
    {
        if (to < 0)
        {
            return 0.0;
        }

        int fi = from < 0 ? 0 : from + 1;
        int ti = to + 1;
        return cost[fi][ti];
    }

    private static double[][] BuildCostMatrix(OrbitInfo[] nodes, double mu)
    {
        int total = nodes.Length;
        double posSum = 0.0;
        double velSum = 0.0;
        double energySum = 0.0;
        double planeSum = 0.0;
        double eccSum = 0.0;
        double phaseSum = 0.0;
        int pairs = 0;

        for (int i = 0; i < total; i++)
        {
            for (int j = i + 1; j < total; j++)
            {
                double dr = Distance(nodes[i].R, nodes[j].R);
                double dv = Distance(nodes[i].V, nodes[j].V);
                double de = Math.Abs(nodes[i].E - nodes[j].E);
                double dplane = AngleBetween(nodes[i].Hhat, nodes[j].Hhat);
                double dphase = Math.Abs(WrapPi(nodes[i].MeanLongitude - nodes[j].MeanLongitude));
                double dvEnergy = Math.Abs(VCirc(mu, nodes[i].A) - VCirc(mu, nodes[j].A));

                posSum += dr;
                velSum += dv;
                energySum += dvEnergy;
                planeSum += dplane;
                eccSum += de;
                phaseSum += dphase;
                pairs++;
            }
        }

        double posScale = pairs > 0 ? posSum / pairs : 1.0;
        double velScale = pairs > 0 ? velSum / pairs : 1.0;
        double energyScale = pairs > 0 ? energySum / pairs : 1.0;
        double planeScale = pairs > 0 ? planeSum / pairs : 1.0;
        double eccScale = pairs > 0 ? eccSum / pairs : 1.0;
        double phaseScale = pairs > 0 ? phaseSum / pairs : 1.0;

        if (posScale < Eps) posScale = 1.0;
        if (velScale < Eps) velScale = 1.0;
        if (energyScale < Eps) energyScale = 1.0;
        if (planeScale < Eps) planeScale = 1.0;
        if (eccScale < Eps) eccScale = 1.0;
        if (phaseScale < Eps) phaseScale = 1.0;

        var cost = new double[total][];
        for (int i = 0; i < total; i++)
        {
            cost[i] = new double[total];
        }

        for (int i = 0; i < total; i++)
        {
            cost[i][i] = 0.0;
            for (int j = i + 1; j < total; j++)
            {
                double dr = Distance(nodes[i].R, nodes[j].R);
                double dv = Distance(nodes[i].V, nodes[j].V);
                double de = Math.Abs(nodes[i].E - nodes[j].E);
                double dplane = AngleBetween(nodes[i].Hhat, nodes[j].Hhat);
                double dphase = Math.Abs(WrapPi(nodes[i].MeanLongitude - nodes[j].MeanLongitude));
                double dvEnergy = Math.Abs(VCirc(mu, nodes[i].A) - VCirc(mu, nodes[j].A));

                double statePart = Math.Sqrt(Square(dr / posScale) + Square(dv / velScale));
                double orbitPart = Math.Sqrt(
                    Square(dvEnergy / energyScale) +
                    Square(de / eccScale) +
                    Square(1.5 * dplane / planeScale) +
                    Square(0.7 * dphase / phaseScale));

                double c = 0.35 * statePart + 0.65 * orbitPart;
                cost[i][j] = c;
                cost[j][i] = c;
            }
        }

        return cost;
    }

    private static OrbitInfo BuildOrbit(double x, double y, double z, double vx, double vy, double vz, double mu)
    {
        Vec3 r = new Vec3(x, y, z);
        Vec3 v = new Vec3(vx, vy, vz);

        double rMag = Length(r);
        double vMag = Length(v);
        Vec3 h = Cross(r, v);
        double hMag = Length(h);
        Vec3 hHat = hMag > Eps ? h / hMag : new Vec3(0.0, 0.0, 1.0);

        Vec3 n = new Vec3(-h.Y, h.X, 0.0);
        double nMag = Length(n);
        Vec3 eVec = Cross(v, h) / mu - r / rMag;
        double e = Length(eVec);

        double energy = 0.5 * vMag * vMag - mu / rMag;
        double a = Math.Abs(energy) > Eps ? -mu / (2.0 * energy) : rMag;

        double raan = nMag > Eps ? Math.Atan2(n.Y, n.X) : 0.0;
        double argPeri = 0.0;
        double trueAnom;

        if (e > 1e-10 && nMag > Eps)
        {
            argPeri = Math.Atan2(Dot(Cross(n, eVec), hHat), Dot(n, eVec));
            trueAnom = Math.Atan2(Dot(Cross(eVec, r), hHat), Dot(eVec, r));
        }
        else if (e > 1e-10)
        {
            trueAnom = Math.Atan2(r.Y, r.X);
        }
        else if (nMag > Eps)
        {
            trueAnom = Math.Atan2(Dot(Cross(n, r), hHat), Dot(n, r));
        }
        else
        {
            trueAnom = Math.Atan2(r.Y, r.X);
        }

        double meanAnom = trueAnom;
        if (e < 1.0 - 1e-10)
        {
            double sinNu = Math.Sin(trueAnom);
            double cosNu = Math.Cos(trueAnom);
            double E = Math.Atan2(Math.Sqrt(Math.Max(0.0, 1.0 - e * e)) * sinNu, e + cosNu);
            meanAnom = E - e * Math.Sin(E);
        }

        double meanLongitude;
        if (e > 1e-10 && nMag > Eps)
        {
            meanLongitude = Wrap2Pi(raan + argPeri + meanAnom);
        }
        else if (nMag > Eps)
        {
            meanLongitude = Wrap2Pi(raan + trueAnom);
        }
        else
        {
            meanLongitude = Wrap2Pi(Math.Atan2(r.Y, r.X));
        }

        return new OrbitInfo
        {
            R = r,
            V = v,
            Hhat = hHat,
            A = a,
            E = e,
            MeanLongitude = meanLongitude
        };
    }

    private static double VCirc(double mu, double a)
    {
        double aa = Math.Max(Math.Abs(a), 1.0);
        return Math.Sqrt(mu / aa);
    }

    private static double Distance(Vec3 a, Vec3 b)
    {
        return Math.Sqrt(Square(a.X - b.X) + Square(a.Y - b.Y) + Square(a.Z - b.Z));
    }

    private static double AngleBetween(Vec3 a, Vec3 b)
    {
        double d = Dot(a, b);
        if (d > 1.0) d = 1.0;
        else if (d < -1.0) d = -1.0;
        return Math.Acos(d);
    }

    private static double WrapPi(double x)
    {
        x %= TwoPi;
        if (x <= -Math.PI)
        {
            x += TwoPi;
        }
        else if (x > Math.PI)
        {
            x -= TwoPi;
        }
        return x;
    }

    private static double Wrap2Pi(double x)
    {
        x %= TwoPi;
        if (x < 0.0)
        {
            x += TwoPi;
        }
        return x;
    }

    private static double Square(double x) => x * x;

    private static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

    private static Vec3 Cross(Vec3 a, Vec3 b)
    {
        return new Vec3(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    private static double Length(Vec3 a) => Math.Sqrt(Dot(a, a));

    private struct OrbitInfo
    {
        public Vec3 R;
        public Vec3 V;
        public Vec3 Hhat;
        public double A;
        public double E;
        public double MeanLongitude;
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator *(Vec3 a, double s)
        {
            return new Vec3(a.X * s, a.Y * s, a.Z * s);
        }

        public static Vec3 operator /(Vec3 a, double s)
        {
            return new Vec3(a.X / s, a.Y / s, a.Z / s);
        }
    }
}
