#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct Task {
    int machine = 0;
    std::int64_t duration = 0;
};

struct Job {
    std::vector<Task> tasks;
    std::int64_t totalDuration = 0;
};

static void scheduleGreedy(const std::vector<Job>& jobs,
                           int machineCount,
                           std::vector<std::vector<std::int64_t>>& starts) {
    const int jobCount = static_cast<int>(jobs.size());
    std::vector<std::int64_t> jobReady(jobCount, 0);
    std::vector<std::int64_t> machineReady(machineCount, 0);
    std::vector<int> nextIndex(jobCount, 0);

    std::int64_t remaining = 0;
    for (const auto& job : jobs) {
        remaining += static_cast<std::int64_t>(job.tasks.size());
    }

    while (remaining > 0) {
        int bestJob = -1;
        std::int64_t bestStart = 0;
        std::int64_t bestFinish = std::numeric_limits<std::int64_t>::max();
        std::int64_t bestDuration = -1;

        for (int j = 0; j < jobCount; ++j) {
            const int idx = nextIndex[j];
            if (idx >= static_cast<int>(jobs[j].tasks.size())) {
                continue;
            }

            const Task& task = jobs[j].tasks[idx];
            const std::int64_t start = std::max(jobReady[j], machineReady[task.machine]);
            const std::int64_t finish = start + task.duration;

            if (finish < bestFinish ||
                (finish == bestFinish && (start < bestStart ||
                 (start == bestStart && task.duration > bestDuration)))) {
                bestJob = j;
                bestStart = start;
                bestFinish = finish;
                bestDuration = task.duration;
            }
        }

        const Task& task = jobs[bestJob].tasks[nextIndex[bestJob]];
        starts[bestJob][nextIndex[bestJob]] = bestStart;
        jobReady[bestJob] = bestFinish;
        machineReady[task.machine] = bestFinish;
        ++nextIndex[bestJob];
        --remaining;
    }
}

static void scheduleSerial(const std::vector<Job>& jobs,
                           int machineCount,
                           std::vector<std::vector<std::int64_t>>& starts) {
    const int jobCount = static_cast<int>(jobs.size());
    std::vector<int> order(jobCount);
    std::iota(order.begin(), order.end(), 0);

    std::sort(order.begin(), order.end(), [&](int a, int b) {
        if (jobs[a].totalDuration != jobs[b].totalDuration) {
            return jobs[a].totalDuration > jobs[b].totalDuration;
        }
        if (jobs[a].tasks.size() != jobs[b].tasks.size()) {
            return jobs[a].tasks.size() > jobs[b].tasks.size();
        }
        return a < b;
    });

    std::vector<std::int64_t> machineReady(machineCount, 0);
    for (int j : order) {
        std::int64_t current = 0;
        const auto& tasks = jobs[j].tasks;
        for (std::size_t i = 0; i < tasks.size(); ++i) {
            const Task& task = tasks[i];
            current = std::max(current, machineReady[task.machine]);
            starts[j][i] = current;
            current += task.duration;
            machineReady[task.machine] = current;
        }
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int J = 0;
    int M = 0;
    if (!(std::cin >> J >> M)) {
        return 0;
    }

    std::vector<Job> jobs(J);
    std::int64_t totalTasks = 0;

    for (int j = 0; j < J; ++j) {
        int T = 0;
        std::cin >> T;
        jobs[j].tasks.resize(T);
        totalTasks += T;

        for (int i = 0; i < T; ++i) {
            int machine = 0;
            std::int64_t duration = 0;
            std::cin >> machine >> duration;
            jobs[j].tasks[i] = Task{machine, duration};
            jobs[j].totalDuration += duration;
        }
    }

    std::vector<std::vector<std::int64_t>> starts(J);
    for (int j = 0; j < J; ++j) {
        starts[j].resize(jobs[j].tasks.size());
    }

    // Small and medium instances: greedy interleaving.
    // Large instances: fast serialized schedule with a good job order.
    const bool useGreedy = (totalTasks <= 12000 && J <= 4000);
    if (useGreedy) {
        scheduleGreedy(jobs, M, starts);
    } else {
        scheduleSerial(jobs, M, starts);
    }

    for (int j = 0; j < J; ++j) {
        for (std::size_t i = 0; i < jobs[j].tasks.size(); ++i) {
            if (i > 0) {
                std::cout << ' ';
            }
            std::cout << starts[j][i] << ' ' << jobs[j].tasks[i].machine;
        }
        std::cout << '\n';
    }

    return 0;
}
