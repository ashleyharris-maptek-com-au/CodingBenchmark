// Minimum Edge Cut solver for undirected connected graphs.
// Hybrid strategy:
// 1) exact bridge detection
// 2) exact cut-preserving sparsification via maximum-adjacency ordering
// 3) Stoer-Wagner on small reduced graphs
// 4) randomized contraction on larger reduced graphs, with recursive
//    exact reduction when a valid upper bound shrinks the graph again.

#include <algorithm>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

using std::int64_t;

struct Edge {
    int u;
    int v;
};

struct Graph {
    int n = 0;
    std::vector<Edge> edges;
    std::vector<std::vector<int>> members; // original input vertices per graph vertex
};

struct CutResult {
    int64_t weight = (std::numeric_limits<int64_t>::max)();
    std::vector<char> side; // side on the current graph's vertices
};

struct SolveResult {
    Graph graph;
    CutResult cut;
};

struct FastRNG {
    uint64_t state;

    explicit FastRNG(uint64_t seed) : state(seed ? seed : 0x9e3779b97f4a7c15ULL) {}

    uint64_t next() {
        uint64_t z = (state += 0x9e3779b97f4a7c15ULL);
        z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
        z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
        return z ^ (z >> 31);
    }
};

struct DSU {
    std::vector<int> p;
    std::vector<int> r;

    explicit DSU(int n = 0) { init(n); }

    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        std::iota(p.begin(), p.end(), 0);
    }

    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }

    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) std::swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static int g_original_n = 0;
static std::vector<Edge> g_original_edges;
static std::vector<int> g_stamp;
static int g_cur_stamp = 0;

static std::vector<std::vector<std::pair<int, int>>> buildAdjWithIds(const Graph& g) {
    std::vector<int> deg(g.n, 0);
    for (const auto& e : g.edges) {
        ++deg[e.u];
        ++deg[e.v];
    }
    std::vector<std::vector<std::pair<int, int>>> adj(g.n);
    for (int i = 0; i < g.n; ++i) adj[i].reserve(deg[i]);
    for (int id = 0; id < (int)g.edges.size(); ++id) {
        const auto& e = g.edges[id];
        adj[e.u].push_back({e.v, id});
        adj[e.v].push_back({e.u, id});
    }
    return adj;
}

static std::vector<std::vector<int>> buildAdjSimple(const Graph& g) {
    std::vector<int> deg(g.n, 0);
    for (const auto& e : g.edges) {
        ++deg[e.u];
        ++deg[e.v];
    }
    std::vector<std::vector<int>> adj(g.n);
    for (int i = 0; i < g.n; ++i) adj[i].reserve(deg[i]);
    for (const auto& e : g.edges) {
        adj[e.u].push_back(e.v);
        adj[e.v].push_back(e.u);
    }
    return adj;
}

static int findBridgeEdgeId(const Graph& g) {
    if (g.n <= 1) return -1;
    auto adj = buildAdjWithIds(g);

    std::vector<int> tin(g.n, -1), low(g.n, 0), parentV(g.n, -1), parentE(g.n, -1), it(g.n, 0);
    int timer = 0;
    std::vector<int> st;
    st.reserve(g.n);

    for (int root = 0; root < g.n; ++root) {
        if (tin[root] != -1) continue;
        st.push_back(root);
        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) tin[v] = low[v] = timer++;
            if (it[v] < (int)adj[v].size()) {
                auto [to, eid] = adj[v][it[v]++];
                if (eid == parentE[v]) continue;
                if (tin[to] == -1) {
                    parentV[to] = v;
                    parentE[to] = eid;
                    st.push_back(to);
                } else {
                    low[v] = std::min(low[v], tin[to]);
                }
            } else {
                st.pop_back();
                if (parentV[v] != -1) {
                    int p = parentV[v];
                    low[p] = std::min(low[p], low[v]);
                    if (low[v] > tin[p]) return parentE[v];
                }
            }
        }
    }
    return -1;
}

static int64_t computeMinDegree(const Graph& g) {
    if (g.n <= 1) return 0;
    std::vector<int64_t> deg(g.n, 0);
    for (const auto& e : g.edges) {
        ++deg[e.u];
        ++deg[e.v];
    }
    return *std::min_element(deg.begin(), deg.end());
}

static void normalizeSide(std::vector<char>& side) {
    int cnt = 0;
    for (char c : side) cnt += (c != 0);
    if (cnt > 0 && cnt * 2 > (int)side.size()) {
        for (char& c : side) c = !c;
    }
}

static int64_t cutWeight(const Graph& g, const std::vector<char>& side) {
    int64_t w = 0;
    for (const auto& e : g.edges) {
        if (side[e.u] != side[e.v]) ++w;
    }
    return w;
}

static std::vector<char> expandToOriginal(const Graph& g, const std::vector<char>& side) {
    std::vector<char> orig(g_original_n, 0);
    for (int v = 0; v < g.n; ++v) {
        if (!side[v]) continue;
        for (int x : g.members[v]) orig[x] = 1;
    }
    return orig;
}

static std::vector<char> currentSideFromOriginal(const Graph& g, const std::vector<char>& origSide) {
    std::vector<char> side(g.n, 0);
    for (int v = 0; v < g.n; ++v) {
        if (!g.members[v].empty() && origSide[g.members[v].front()]) side[v] = 1;
    }
    return side;
}

static CutResult evaluateCandidate(const Graph& parent, const Graph& child, const CutResult& childRes) {
    std::vector<char> origSide = expandToOriginal(child, childRes.side);
    std::vector<char> parentSide = currentSideFromOriginal(parent, origSide);
    normalizeSide(parentSide);
    CutResult out;
    out.side = std::move(parentSide);
    out.weight = cutWeight(parent, out.side);
    return out;
}

static Graph buildCertificate(const Graph& g, int64_t tau) {
    if (g.n <= 1) return g;
    auto adj = buildAdjSimple(g);

    std::vector<int64_t> deg(g.n, 0);
    int start = 0;
    int64_t bestDeg = -1;
    for (int v = 0; v < g.n; ++v) {
        deg[v] = (int64_t)adj[v].size();
        if (deg[v] > bestDeg) {
            bestDeg = deg[v];
            start = v;
        }
    }

    std::vector<char> used(g.n, 0);
    std::vector<int> order;
    order.reserve(g.n);
    std::vector<int64_t> score(g.n, 0);
    std::vector<int64_t> prefixAdj(g.n, 0);
    std::priority_queue<std::pair<int64_t, int>> pq;

    used[start] = 1;
    order.push_back(start);
    for (int to : adj[start]) {
        if (!used[to]) {
            ++score[to];
            pq.push({score[to], to});
        }
    }

    while ((int)order.size() < g.n) {
        int v = -1;
        while (!pq.empty()) {
            auto [sc, u] = pq.top();
            pq.pop();
            if (!used[u] && sc == score[u]) {
                v = u;
                break;
            }
        }
        if (v == -1) {
            for (int u = 0; u < g.n; ++u) {
                if (!used[u]) {
                    v = u;
                    break;
                }
            }
        }
        used[v] = 1;
        prefixAdj[order.size()] = score[v];
        order.push_back(v);
        for (int to : adj[v]) {
            if (!used[to]) {
                ++score[to];
                pq.push({score[to], to});
            }
        }
    }

    std::vector<int> blockOf(g.n, 0);
    int blocks = 1;
    blockOf[order[0]] = 0;
    for (int i = 1; i < g.n; ++i) {
        if (prefixAdj[i] >= tau) blockOf[order[i]] = blocks - 1;
        else blockOf[order[i]] = blocks++;
    }

    Graph h;
    h.n = blocks;
    h.members.assign(blocks, {});

    std::vector<std::size_t> memberCount(blocks, 0);
    for (int v = 0; v < g.n; ++v) memberCount[blockOf[v]] += g.members[v].size();
    for (int b = 0; b < blocks; ++b) h.members[b].reserve(memberCount[b]);

    for (int v = 0; v < g.n; ++v) {
        int b = blockOf[v];
        const auto& src = g.members[v];
        auto& dst = h.members[b];
        dst.insert(dst.end(), src.begin(), src.end());
    }

    h.edges.reserve(g.edges.size());
    for (const auto& e : g.edges) {
        int a = blockOf[e.u];
        int b = blockOf[e.v];
        if (a != b) h.edges.push_back({a, b});
    }
    return h;
}

static CutResult normalizeCutResult(const Graph& g, CutResult res) {
    normalizeSide(res.side);
    res.weight = cutWeight(g, res.side);
    return res;
}

static CutResult stoerWagner(const Graph& g) {
    int n = g.n;
    if (n == 0) return {};
    if (n == 1) {
        CutResult r;
        r.weight = 0;
        r.side = {1};
        return r;
    }

    std::vector<std::vector<int64_t>> w(n, std::vector<int64_t>(n, 0));
    for (const auto& e : g.edges) {
        ++w[e.u][e.v];
        ++w[e.v][e.u];
    }

    std::vector<int> active(n);
    std::iota(active.begin(), active.end(), 0);
    std::vector<std::vector<int>> groups(n);
    for (int i = 0; i < n; ++i) groups[i].push_back(i);

    CutResult best;
    best.weight = (std::numeric_limits<int64_t>::max)();
    best.side.assign(n, 0);

    while (active.size() > 1) {
        std::vector<int64_t> weights(n, 0);
        std::vector<char> added(n, 0);
        int prev = -1, last = -1;

        for (int step = 0; step < (int)active.size(); ++step) {
            int sel = -1;
            int64_t bestW = -1;
            for (int v : active) {
                if (!added[v] && weights[v] > bestW) {
                    bestW = weights[v];
                    sel = v;
                }
            }
            added[sel] = 1;
            prev = last;
            last = sel;
            if (step + 1 < (int)active.size()) {
                for (int v : active) {
                    if (!added[v]) weights[v] += w[sel][v];
                }
            }
        }

        if (weights[last] < best.weight) {
            best.weight = weights[last];
            best.side.assign(n, 0);
            for (int v : active) {
                if (v == last) continue;
                for (int x : groups[v]) best.side[x] = 1;
            }
        }

        if (prev == -1) break;

        for (int v : active) {
            if (v == prev || v == last) continue;
            w[prev][v] += w[last][v];
            w[v][prev] = w[prev][v];
        }
        groups[prev].insert(groups[prev].end(), groups[last].begin(), groups[last].end());

        auto it = std::find(active.begin(), active.end(), last);
        if (it != active.end()) {
            *it = active.back();
            active.pop_back();
        } else {
            break;
        }
    }

    return normalizeCutResult(g, best);
}

static CutResult kargerTrial(const Graph& g, FastRNG& rng) {
    if (g.n <= 1) return {};
    if (g.n == 2) {
        std::vector<char> side = {1, 0};
        CutResult r;
        r.side = std::move(side);
        r.weight = cutWeight(g, r.side);
        return normalizeCutResult(g, r);
    }

    DSU dsu(g.n);
    int comps = g.n;
    while (comps > 2) {
        const Edge& e = g.edges[rng.next() % g.edges.size()];
        if (dsu.unite(e.u, e.v)) --comps;
    }

    std::vector<char> side(g.n, 0);
    int root = dsu.find(0);
    for (int v = 0; v < g.n; ++v) {
        if (dsu.find(v) == root) side[v] = 1;
    }

    CutResult r;
    r.side = std::move(side);
    r.weight = cutWeight(g, r.side);
    return normalizeCutResult(g, r);
}

static CutResult randomUpperBound(const Graph& g, FastRNG& rng, int trials) {
    CutResult best;
    best.weight = (std::numeric_limits<int64_t>::max)();
    for (int i = 0; i < trials; ++i) {
        CutResult cand = kargerTrial(g, rng);
        if (cand.weight < best.weight) {
            best = std::move(cand);
            if (best.weight <= 2) break;
        }
    }
    if (best.weight == (std::numeric_limits<int64_t>::max)()) {
        best.weight = 0;
        best.side.assign(g.n, 0);
    }
    return best;
}

static Graph contractTo(const Graph& g, int target, FastRNG& rng) {
    if (g.n <= target) return g;
    DSU dsu(g.n);
    int comps = g.n;
    while (comps > target) {
        const Edge& e = g.edges[rng.next() % g.edges.size()];
        if (dsu.unite(e.u, e.v)) --comps;
    }

    std::vector<int> rootToNew(g.n, -1);
    int nextId = 0;
    for (int v = 0; v < g.n; ++v) {
        int r = dsu.find(v);
        if (rootToNew[r] == -1) rootToNew[r] = nextId++;
    }

    Graph h;
    h.n = nextId;
    h.members.assign(h.n, {});
    std::vector<std::size_t> memberCount(h.n, 0);

    for (int v = 0; v < g.n; ++v) {
        int id = rootToNew[dsu.find(v)];
        memberCount[id] += g.members[v].size();
    }
    for (int i = 0; i < h.n; ++i) h.members[i].reserve(memberCount[i]);

    for (int v = 0; v < g.n; ++v) {
        int id = rootToNew[dsu.find(v)];
        const auto& src = g.members[v];
        auto& dst = h.members[id];
        dst.insert(dst.end(), src.begin(), src.end());
    }

    h.edges.reserve(g.edges.size());
    for (const auto& e : g.edges) {
        int a = rootToNew[dsu.find(e.u)];
        int b = rootToNew[dsu.find(e.v)];
        if (a != b) h.edges.push_back({a, b});
    }
    return h;
}

static CutResult kargerSteinOnce(const Graph& g, FastRNG& rng);

static CutResult kargerSteinRepeated(const Graph& g, FastRNG& rng, int trials) {
    CutResult best;
    best.weight = (std::numeric_limits<int64_t>::max)();
    for (int i = 0; i < trials; ++i) {
        CutResult cand = kargerSteinOnce(g, rng);
        if (cand.weight < best.weight) {
            best = std::move(cand);
            if (best.weight <= 2) break;
        }
    }
    if (best.weight == (std::numeric_limits<int64_t>::max)()) {
        best.weight = 0;
        best.side.assign(g.n, 0);
    }
    return best;
}

static CutResult kargerSteinOnce(const Graph& g, FastRNG& rng) {
    if (g.n <= 40) return stoerWagner(g);

    int target = (int)std::ceil(g.n / std::sqrt(2.0));
    if (target < 2) target = 2;

    Graph g1 = contractTo(g, target, rng);
    Graph g2 = contractTo(g, target, rng);

    CutResult r1 = kargerSteinOnce(g1, rng);
    CutResult r2 = kargerSteinOnce(g2, rng);

    CutResult e1 = evaluateCandidate(g, g1, r1);
    CutResult e2 = evaluateCandidate(g, g2, r2);

    return (e1.weight <= e2.weight) ? e1 : e2;
}

static SolveResult solveCurrentGraph(Graph g, FastRNG& rng) {
    // Exact cut-preserving reductions using the current min degree as an upper bound.
    for (int iter = 0; iter < 3 && g.n > 1; ++iter) {
        if (g.n <= 40) break;
        int64_t tau = computeMinDegree(g);
        if (tau <= 1) break;
        Graph h = buildCertificate(g, tau);
        if (h.n == g.n && h.edges.size() == g.edges.size()) break;
        g = std::move(h);
    }

    if (g.n <= 40) {
        CutResult sw = stoerWagner(g);
        return {std::move(g), std::move(sw)};
    }

    // A few random trials often tighten the upper bound enough to shrink again.
    {
        int trials = 8;
        if (g.n > 500) trials = 12;
        CutResult ub = randomUpperBound(g, rng, trials);
        if (ub.weight == 2) return {std::move(g), ub};

        int64_t tau = computeMinDegree(g);
        if (ub.weight < tau) tau = ub.weight;
        if (tau > 1) {
            Graph h = buildCertificate(g, tau);
            if (h.n < g.n || h.edges.size() < g.edges.size()) {
                SolveResult sub = solveCurrentGraph(std::move(h), rng);
                CutResult mapped = evaluateCandidate(g, sub.graph, sub.cut);
                CutResult best = (mapped.weight <= ub.weight) ? mapped : ub;
                return {std::move(g), std::move(best)};
            }
        }
    }

    // Larger residual core: randomized contraction with recursive reduction.
    int logn = 0;
    while ((1ULL << logn) < (unsigned long long)g.n && logn < 30) ++logn;
    int ksTrials = std::min(20, std::max(8, 2 * logn + 2));
    CutResult approx = kargerSteinRepeated(g, rng, ksTrials);
    if (approx.weight == 2) return {std::move(g), approx};

    Graph r = buildCertificate(g, approx.weight);
    if (r.n < g.n || r.edges.size() < g.edges.size()) {
        SolveResult sub = solveCurrentGraph(std::move(r), rng);
        CutResult mapped = evaluateCandidate(g, sub.graph, sub.cut);
        CutResult best = (mapped.weight <= approx.weight) ? mapped : approx;
        return {std::move(g), std::move(best)};
    }

    return {std::move(g), approx};
}

static std::vector<char> expandFinalSide(const Graph& g, const std::vector<char>& side) {
    std::vector<char> orig(g_original_n, 0);
    for (int v = 0; v < g.n; ++v) {
        if (!side[v]) continue;
        for (int x : g.members[v]) orig[x] = 1;
    }
    return orig;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m;
    if (!(std::cin >> n >> m)) return 0;

    g_original_n = n;
    g_original_edges.clear();
    g_original_edges.reserve(m);

    Graph initial;
    initial.n = n;
    initial.members.resize(n);
    for (int i = 0; i < n; ++i) initial.members[i].push_back(i);

    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u == v) continue; // self-loops do not affect edge cuts
        g_original_edges.push_back({u, v});
        initial.edges.push_back({u, v});
    }

    if (n <= 1) {
        std::cout << 0 << '\n';
        return 0;
    }

    // Exact bridge check first: if any bridge exists, the minimum cut is 1.
    int bridgeId = findBridgeEdgeId(initial);
    if (bridgeId != -1) {
        const auto& e = initial.edges[bridgeId];
        std::cout << 1 << '\n' << e.u << ' ' << e.v << '\n';
        return 0;
    }

    uint64_t seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
    seed ^= (uint64_t)n << 32;
    seed ^= (uint64_t)g_original_edges.size();
    FastRNG rng(seed);

    SolveResult res = solveCurrentGraph(std::move(initial), rng);
    normalizeSide(res.cut.side);

    std::vector<char> origSide = expandFinalSide(res.graph, res.cut.side);
    std::vector<Edge> cut;
    cut.reserve(g_original_edges.size());
    for (const auto& e : g_original_edges) {
        if (origSide[e.u] != origSide[e.v]) cut.push_back(e);
    }

    std::cout << cut.size() << '\n';
    for (const auto& e : cut) {
        std::cout << e.u << ' ' << e.v << '\n';
    }
    return 0;
}
