using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Box
    {
        public int Index;
        public int A;
        public int B;
        public int C;
        public long Volume;
    }

    private struct Space
    {
        public int X;
        public int Y;
        public int Z;
        public int W;
        public int H;
        public int D;
        public int A;
        public int B;
        public int C;
        public byte Ax0;
        public byte Ax1;
        public byte Ax2;
        public long Volume;
    }

    private struct Placement
    {
        public int Index;
        public int X;
        public int Y;
        public int Z;
    }

    private sealed class PackResult
    {
        public List<Placement> Placements = new List<Placement>();
        public long PackedVolume;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int w = fs.NextInt();
        int h = fs.NextInt();
        int d = fs.NextInt();

        if (n <= 0 || w <= 0 || h <= 0 || d <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();
            Sort3Descending(ref a, ref b, ref c);
            boxes[i] = new Box
            {
                Index = i,
                A = a,
                B = b,
                C = c,
                Volume = (long)a * b * c
            };
        }

        var strategies = BuildStrategies(n);
        PackResult best = null;

        foreach (var comparison in strategies)
        {
            var ordered = (Box[])boxes.Clone();
            Array.Sort(ordered, comparison);

            var result = Pack(ordered, w, h, d);
            if (best == null ||
                result.Placements.Count > best.Placements.Count ||
                (result.Placements.Count == best.Placements.Count && result.PackedVolume > best.PackedVolume))
            {
                best = result;
            }
        }

        var sb = new StringBuilder(best.Placements.Count * 24 + 32);
        sb.AppendLine(best.Placements.Count.ToString());
        for (int i = 0; i < best.Placements.Count; i++)
        {
            var p = best.Placements[i];
            sb.Append(p.Index).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).AppendLine();
        }

        Console.Write(sb.ToString());
    }

    private static Comparison<Box>[] BuildStrategies(int n)
    {
        return new Comparison<Box>[]
        {
            CompareByMinSide,
            CompareByVolume,
            CompareByMaxSide
        };
    }

    private static PackResult Pack(Box[] orderedBoxes, int containerW, int containerH, int containerD)
    {
        var spaces = new List<Space>(orderedBoxes.Length * 2 + 1);
        spaces.Add(MakeSpace(0, 0, 0, containerW, containerH, containerD));

        var placements = new List<Placement>(orderedBoxes.Length);
        long packedVolume = 0;

        for (int i = 0; i < orderedBoxes.Length; i++)
        {
            Box box = orderedBoxes[i];

            int bestSpaceIndex = -1;
            long bestSpaceVolume = long.MaxValue;
            int bestSpaceC = int.MaxValue;
            int bestSpaceB = int.MaxValue;
            int bestSpaceA = int.MaxValue;

            for (int s = 0; s < spaces.Count; s++)
            {
                Space space = spaces[s];
                if (space.C < box.C || space.B < box.B || space.A < box.A)
                {
                    continue;
                }

                if (space.Volume < bestSpaceVolume ||
                    (space.Volume == bestSpaceVolume &&
                     (space.C < bestSpaceC ||
                      (space.C == bestSpaceC &&
                       (space.B < bestSpaceB ||
                        (space.B == bestSpaceB && space.A < bestSpaceA))))))
                {
                    bestSpaceIndex = s;
                    bestSpaceVolume = space.Volume;
                    bestSpaceC = space.C;
                    bestSpaceB = space.B;
                    bestSpaceA = space.A;
                }
            }

            if (bestSpaceIndex < 0)
            {
                continue;
            }

            Space chosen = spaces[bestSpaceIndex];
            RemoveAtSwapBack(spaces, bestSpaceIndex);

            GetOrientedBoxExtents(in chosen, in box, out int bx, out int by, out int bz);

            placements.Add(new Placement
            {
                Index = box.Index,
                X = chosen.X,
                Y = chosen.Y,
                Z = chosen.Z
            });
            packedVolume += box.Volume;

            AddChildSpaces(spaces, in chosen, bx, by, bz);
        }

        return new PackResult
        {
            Placements = placements,
            PackedVolume = packedVolume
        };
    }

    private static void AddChildSpaces(List<Space> spaces, in Space parent, int bx, int by, int bz)
    {
        int rightW = parent.W - bx;
        if (rightW > 0)
        {
            spaces.Add(MakeSpace(parent.X + bx, parent.Y, parent.Z, rightW, parent.H, parent.D));
        }

        int frontH = parent.H - by;
        if (frontH > 0)
        {
            spaces.Add(MakeSpace(parent.X, parent.Y + by, parent.Z, bx, frontH, parent.D));
        }

        int topD = parent.D - bz;
        if (topD > 0)
        {
            spaces.Add(MakeSpace(parent.X, parent.Y, parent.Z + bz, bx, by, topD));
        }
    }

    private static void GetOrientedBoxExtents(in Space space, in Box box, out int bx, out int by, out int bz)
    {
        bx = 0;
        by = 0;
        bz = 0;

        AssignAxis(ref bx, ref by, ref bz, space.Ax0, box.A);
        AssignAxis(ref bx, ref by, ref bz, space.Ax1, box.B);
        AssignAxis(ref bx, ref by, ref bz, space.Ax2, box.C);
    }

    private static void AssignAxis(ref int bx, ref int by, ref int bz, byte axis, int value)
    {
        if (axis == 0)
        {
            bx = value;
        }
        else if (axis == 1)
        {
            by = value;
        }
        else
        {
            bz = value;
        }
    }

    private static Space MakeSpace(int x, int y, int z, int w, int h, int d)
    {
        int ax0 = 0;
        int ax1 = 1;
        int ax2 = 2;

        int s0 = w;
        int s1 = h;
        int s2 = d;

        if (s0 < s1)
        {
            Swap(ref s0, ref s1);
            Swap(ref ax0, ref ax1);
        }
        if (s1 < s2)
        {
            Swap(ref s1, ref s2);
            Swap(ref ax1, ref ax2);
        }
        if (s0 < s1)
        {
            Swap(ref s0, ref s1);
            Swap(ref ax0, ref ax1);
        }

        return new Space
        {
            X = x,
            Y = y,
            Z = z,
            W = w,
            H = h,
            D = d,
            A = s0,
            B = s1,
            C = s2,
            Ax0 = (byte)ax0,
            Ax1 = (byte)ax1,
            Ax2 = (byte)ax2,
            Volume = (long)w * h * d
        };
    }

    private static void RemoveAtSwapBack<T>(List<T> list, int index)
    {
        int last = list.Count - 1;
        list[index] = list[last];
        list.RemoveAt(last);
    }

    private static void Sort3Descending(ref int a, ref int b, ref int c)
    {
        if (a < b)
        {
            Swap(ref a, ref b);
        }

        if (b < c)
        {
            Swap(ref b, ref c);
        }

        if (a < b)
        {
            Swap(ref a, ref b);
        }
    }

    private static void Swap(ref int a, ref int b)
    {
        int t = a;
        a = b;
        b = t;
    }

    private static int CompareByMinSide(Box x, Box y)
    {
        int cmp = y.C.CompareTo(x.C);
        if (cmp != 0) return cmp;
        cmp = y.B.CompareTo(x.B);
        if (cmp != 0) return cmp;
        cmp = y.A.CompareTo(x.A);
        if (cmp != 0) return cmp;
        cmp = y.Volume.CompareTo(x.Volume);
        if (cmp != 0) return cmp;
        return x.Index.CompareTo(y.Index);
    }

    private static int CompareByVolume(Box x, Box y)
    {
        int cmp = y.Volume.CompareTo(x.Volume);
        if (cmp != 0) return cmp;
        cmp = y.C.CompareTo(x.C);
        if (cmp != 0) return cmp;
        cmp = y.B.CompareTo(x.B);
        if (cmp != 0) return cmp;
        cmp = y.A.CompareTo(x.A);
        if (cmp != 0) return cmp;
        return x.Index.CompareTo(y.Index);
    }

    private static int CompareByMaxSide(Box x, Box y)
    {
        int cmp = y.A.CompareTo(x.A);
        if (cmp != 0) return cmp;
        cmp = y.B.CompareTo(x.B);
        if (cmp != 0) return cmp;
        cmp = y.C.CompareTo(x.C);
        if (cmp != 0) return cmp;
        cmp = y.Volume.CompareTo(x.Volume);
        if (cmp != 0) return cmp;
        return x.Index.CompareTo(y.Index);
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_ptr++];
        }
    }
}
