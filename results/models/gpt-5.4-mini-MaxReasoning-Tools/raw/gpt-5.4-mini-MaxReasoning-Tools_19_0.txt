using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;

public static class Program
{
    private const int FullAreaThreshold = 1_500_000;
    private const long FullVolumeThreshold = 20_000_000;
    private const int MaxSelectedOperations = 10_000;
    private const int Inf = int.MaxValue / 4;

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        string header = input.ReadHeaderLine();
        if (string.IsNullOrWhiteSpace(header))
        {
            return;
        }

        string[] parts = header.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 3)
        {
            return;
        }

        int xCount = int.Parse(parts[0], CultureInfo.InvariantCulture);
        int yCount = int.Parse(parts[1], CultureInfo.InvariantCulture);
        int zCount = int.Parse(parts[2], CultureInfo.InvariantCulture);

        long volume = (long)xCount * yCount * zCount;
        int area = xCount * yCount;

        if (area <= FullAreaThreshold && volume <= FullVolumeThreshold)
        {
            SolveFull(input, xCount, yCount, zCount, area);
        }
        else
        {
            SolveFallback(input, xCount, yCount, zCount);
        }
    }

    private static void SolveFull(FastInput input, int xCount, int yCount, int zCount, int area)
    {
        var height = new short[area];
        Array.Fill(height, (short)-1);
        var topVoxel = new byte[area];

        for (int x = 0; x < xCount; x++)
        {
            int rowBase = x * yCount;
            for (int y = 0; y < yCount; y++)
            {
                int idx = rowBase + y;
                for (int z = 0; z < zCount; z++)
                {
                    int d = input.ReadByte();
                    if (d < 0)
                    {
                        throw new EndOfStreamException("Unexpected end of voxel stream.");
                    }

                    if ((d & 15) != 0)
                    {
                        height[idx] = (short)z;
                        topVoxel[idx] = (byte)d;
                    }
                }
            }
        }

        // Give the plant a stable origin even if the origin column is empty.
        if (height[0] < 0)
        {
            height[0] = 0;
        }

        var dist = new int[area];
        Array.Fill(dist, Inf);
        var parent = new int[area];
        Array.Fill(parent, -1);
        var order = new int[area];
        int orderCount = 0;

        var heap = new IntMinHeap(Math.Min(1024, Math.Max(4, area)));
        dist[0] = 0;
        heap.Push(0, 0);

        while (heap.Count > 0)
        {
            heap.Pop(out int u, out int du);
            if (du != dist[u])
            {
                continue;
            }

            order[orderCount++] = u;

            int hu = height[u];
            int ux = u / yCount;
            int uy = u - ux * yCount;

            // Reversed-graph Dijkstra:
            // moving from current node to a neighbor uses the cost of the original
            // move from the neighbor back to the current node.

            if (ux > 0)
            {
                int v = u - yCount;
                int hv = height[v];
                if (hv >= 0)
                {
                    int diff = hv - hu;
                    if (diff >= -1 && diff <= 1)
                    {
                        int w = diff == 1 ? 0 : diff == 0 ? 1 : 5;
                        int nd = du + w;
                        if (nd < dist[v])
                        {
                            dist[v] = nd;
                            parent[v] = u;
                            heap.Push(v, nd);
                        }
                    }
                }
            }

            if (ux + 1 < xCount)
            {
                int v = u + yCount;
                int hv = height[v];
                if (hv >= 0)
                {
                    int diff = hv - hu;
                    if (diff >= -1 && diff <= 1)
                    {
                        int w = diff == 1 ? 0 : diff == 0 ? 1 : 5;
                        int nd = du + w;
                        if (nd < dist[v])
                        {
                            dist[v] = nd;
                            parent[v] = u;
                            heap.Push(v, nd);
                        }
                    }
                }
            }

            if (uy > 0)
            {
                int v = u - 1;
                int hv = height[v];
                if (hv >= 0)
                {
                    int diff = hv - hu;
                    if (diff >= -1 && diff <= 1)
                    {
                        int w = diff == 1 ? 0 : diff == 0 ? 1 : 5;
                        int nd = du + w;
                        if (nd < dist[v])
                        {
                            dist[v] = nd;
                            parent[v] = u;
                            heap.Push(v, nd);
                        }
                    }
                }
            }

            if (uy + 1 < yCount)
            {
                int v = u + 1;
                int hv = height[v];
                if (hv >= 0)
                {
                    int diff = hv - hu;
                    if (diff >= -1 && diff <= 1)
                    {
                        int w = diff == 1 ? 0 : diff == 0 ? 1 : 5;
                        int nd = du + w;
                        if (nd < dist[v])
                        {
                            dist[v] = nd;
                            parent[v] = u;
                            heap.Push(v, nd);
                        }
                    }
                }
            }
        }

        if (orderCount == 0)
        {
            WriteNoOps();
            return;
        }

        var depth = new int[area];
        Array.Fill(depth, -1);
        depth[0] = 0;
        for (int i = 1; i < orderCount; i++)
        {
            int u = order[i];
            int p = parent[u];
            depth[u] = p >= 0 ? depth[p] + 1 : 0;
        }

        var selector = new TopCandidateHeap(MaxSelectedOperations);
        for (int idx = 0; idx < area; idx++)
        {
            if (dist[idx] >= Inf || depth[idx] < 0)
            {
                continue;
            }

            byte b = topVoxel[idx];
            int mass = b & 15;
            int value = b >> 4;
            if (mass == 0 || value == 0)
            {
                continue;
            }

            long net = (long)mass * ((long)value * 1000 - dist[idx]);
            if (net <= 0)
            {
                continue;
            }

            selector.Add(new Candidate(idx, net, depth[idx]));
        }

        Candidate[] selected = selector.ToArray();
        if (selected.Length == 0)
        {
            WriteNoOps();
            return;
        }

        Array.Sort(selected, CandidateOutputComparer.Instance);

        var output = new FastOutput(Console.OpenStandardOutput());
        output.WriteInt(selected.Length);
        output.WriteLine();

        var path = new List<int>(64);
        for (int i = 0; i < selected.Length; i++)
        {
            int idx = selected[i].Index;
            path.Clear();

            int cur = idx;
            while (true)
            {
                path.Add(cur);
                if (cur == 0)
                {
                    break;
                }

                cur = parent[cur];
                if (cur < 0)
                {
                    break;
                }
            }

            int stepCount = path.Count >= 2 ? path.Count - 2 : 0;
            output.WriteInt(stepCount);
            output.WriteSpace();
            output.WriteInt(idx / yCount);
            output.WriteSpace();
            output.WriteInt(idx % yCount);
            output.WriteSpace();
            output.WriteInt(height[idx]);

            for (int p = 1; p < path.Count - 1; p++)
            {
                int node = path[p];
                output.WriteSpace();
                output.WriteInt(node / yCount);
                output.WriteSpace();
                output.WriteInt(node % yCount);
            }

            output.WriteSpace();
            output.WriteInt(0);
            output.WriteSpace();
            output.WriteInt(0);
            output.WriteLine();
        }

        output.Flush();
    }

    private static void SolveFallback(FastInput input, int xCount, int yCount, int zCount)
    {
        // Large-world fallback:
        // read only the x=0 slice and mine directly along the y-axis to the plant.
        var height = new short[yCount];
        Array.Fill(height, (short)-1);
        var topVoxel = new byte[yCount];

        int xLimit = Math.Min(1, xCount);
        for (int x = 0; x < xLimit; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int d = input.ReadByte();
                    if (d < 0)
                    {
                        throw new EndOfStreamException("Unexpected end of voxel stream.");
                    }

                    if ((d & 15) != 0)
                    {
                        height[y] = (short)z;
                        topVoxel[y] = (byte)d;
                    }
                }
            }
        }

        if (height[0] < 0)
        {
            height[0] = 0;
        }

        var cost = new int[yCount];
        var reachable = new bool[yCount];
        reachable[0] = true;
        cost[0] = 0;

        for (int y = 1; y < yCount; y++)
        {
            if (!reachable[y - 1] || height[y] < 0)
            {
                reachable[y] = false;
                cost[y] = Inf;
                continue;
            }

            int diff = height[y - 1] - height[y];
            if (diff < -1 || diff > 1)
            {
                reachable[y] = false;
                cost[y] = Inf;
                continue;
            }

            int w = diff == 1 ? 0 : diff == 0 ? 1 : 5;
            cost[y] = cost[y - 1] + w;
            reachable[y] = true;
        }

        var selected = new List<Candidate>(yCount);
        for (int y = 0; y < yCount; y++)
        {
            if (!reachable[y])
            {
                continue;
            }

            byte b = topVoxel[y];
            int mass = b & 15;
            int value = b >> 4;
            if (mass == 0 || value == 0)
            {
                continue;
            }

            long net = (long)mass * ((long)value * 1000 - cost[y]);
            if (net > 0)
            {
                selected.Add(new Candidate(y, net, y));
            }
        }

        if (selected.Count == 0)
        {
            WriteNoOps();
            return;
        }

        selected.Sort((a, b) =>
        {
            int cmp = b.Depth.CompareTo(a.Depth);
            if (cmp != 0) return cmp;
            cmp = b.Net.CompareTo(a.Net);
            if (cmp != 0) return cmp;
            return a.Index.CompareTo(b.Index);
        });

        var output = new FastOutput(Console.OpenStandardOutput());
        output.WriteInt(selected.Count);
        output.WriteLine();

        for (int i = 0; i < selected.Count; i++)
        {
            int y = selected[i].Index;
            int stepCount = y > 0 ? y - 1 : 0;
            output.WriteInt(stepCount);
            output.WriteSpace();
            output.WriteInt(0);
            output.WriteSpace();
            output.WriteInt(y);
            output.WriteSpace();
            output.WriteInt(height[y]);

            for (int s = y - 1; s >= 1; s--)
            {
                output.WriteSpace();
                output.WriteInt(0);
                output.WriteSpace();
                output.WriteInt(s);
            }

            output.WriteSpace();
            output.WriteInt(0);
            output.WriteSpace();
            output.WriteInt(0);
            output.WriteLine();
        }

        output.Flush();
    }

    private static void WriteNoOps()
    {
        var output = new FastOutput(Console.OpenStandardOutput());
        output.WriteInt(0);
        output.WriteLine();
        output.Flush();
    }

    private readonly struct Candidate
    {
        public readonly int Index;
        public readonly long Net;
        public readonly int Depth;

        public Candidate(int index, long net, int depth)
        {
            Index = index;
            Net = net;
            Depth = depth;
        }
    }

    private sealed class CandidateOutputComparer : IComparer<Candidate>
    {
        public static readonly CandidateOutputComparer Instance = new CandidateOutputComparer();

        public int Compare(Candidate a, Candidate b)
        {
            int cmp = b.Depth.CompareTo(a.Depth);
            if (cmp != 0)
            {
                return cmp;
            }

            cmp = b.Net.CompareTo(a.Net);
            if (cmp != 0)
            {
                return cmp;
            }

            return a.Index.CompareTo(b.Index);
        }
    }

    private sealed class TopCandidateHeap
    {
        private Candidate[] _data;
        private int _count;

        public TopCandidateHeap(int capacity)
        {
            _data = new Candidate[Math.Max(4, capacity)];
        }

        public int Count => _count;

        public void Add(Candidate value)
        {
            if (_count < _data.Length)
            {
                SiftUp(_count++, value);
                return;
            }

            if (Better(value, _data[0]))
            {
                _data[0] = value;
                SiftDown(0);
            }
        }

        public Candidate[] ToArray()
        {
            var result = new Candidate[_count];
            Array.Copy(_data, result, _count);
            return result;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static bool Worse(in Candidate a, in Candidate b)
        {
            if (a.Net != b.Net)
            {
                return a.Net < b.Net;
            }

            if (a.Depth != b.Depth)
            {
                return a.Depth < b.Depth;
            }

            return a.Index < b.Index;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private static bool Better(in Candidate a, in Candidate b)
        {
            return Worse(b, a);
        }

        private void SiftUp(int index, Candidate value)
        {
            int i = index;
            while (i > 0)
            {
                int parent = (i - 1) >> 1;
                if (!Worse(value, _data[parent]))
                {
                    break;
                }

                _data[i] = _data[parent];
                i = parent;
            }

            _data[i] = value;
        }

        private void SiftDown(int index)
        {
            int i = index;
            Candidate value = _data[i];
            int count = _count;

            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= count)
                {
                    break;
                }

                int right = left + 1;
                int child = left;
                if (right < count && Worse(_data[right], _data[left]))
                {
                    child = right;
                }

                if (!Worse(_data[child], value))
                {
                    break;
                }

                _data[i] = _data[child];
                i = child;
            }

            _data[i] = value;
        }
    }

    private sealed class IntMinHeap
    {
        private int[] _nodes;
        private int[] _costs;
        private int _count;

        public IntMinHeap(int capacity)
        {
            _nodes = new int[capacity];
            _costs = new int[capacity];
        }

        public int Count => _count;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Push(int node, int cost)
        {
            if (_count == _nodes.Length)
            {
                Grow();
            }

            int i = _count++;
            while (i > 0)
            {
                int parent = (i - 1) >> 1;
                if (_costs[parent] <= cost)
                {
                    break;
                }

                _nodes[i] = _nodes[parent];
                _costs[i] = _costs[parent];
                i = parent;
            }

            _nodes[i] = node;
            _costs[i] = cost;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Pop(out int node, out int cost)
        {
            node = _nodes[0];
            cost = _costs[0];

            int n = --_count;
            if (n == 0)
            {
                return;
            }

            int lastNode = _nodes[n];
            int lastCost = _costs[n];
            int i = 0;

            while (true)
            {
                int left = (i << 1) + 1;
                if (left >= n)
                {
                    break;
                }

                int right = left + 1;
                int child = left;
                if (right < n && _costs[right] < _costs[left])
                {
                    child = right;
                }

                if (_costs[child] >= lastCost)
                {
                    break;
                }

                _nodes[i] = _nodes[child];
                _costs[i] = _costs[child];
                i = child;
            }

            _nodes[i] = lastNode;
            _costs[i] = lastCost;
        }

        private void Grow()
        {
            int newSize = _nodes.Length == 0 ? 4 : _nodes.Length * 2;
            Array.Resize(ref _nodes, newSize);
            Array.Resize(ref _costs, newSize);
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _length;
        private int _position;

        public FastInput(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int ReadByte()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length <= 0)
                {
                    return -1;
                }
            }

            return _buffer[_position++];
        }

        public string ReadHeaderLine()
        {
            var sb = new StringBuilder();
            int b;
            while ((b = ReadByte()) != -1)
            {
                if (b == '\n')
                {
                    break;
                }

                if (b != '\r')
                {
                    sb.Append((char)b);
                }
            }

            return sb.ToString();
        }
    }

    private sealed class FastOutput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _position;

        public FastOutput(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                if (value == int.MinValue)
                {
                    WriteString("2147483648");
                    return;
                }

                value = -value;
            }

            Span<byte> tmp = stackalloc byte[12];
            int len = 0;
            while (value > 0)
            {
                int digit = value % 10;
                value /= 10;
                tmp[len++] = (byte)('0' + digit);
            }

            for (int i = len - 1; i >= 0; i--)
            {
                WriteByte(tmp[i]);
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteLong(long value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                if (value == long.MinValue)
                {
                    WriteString("9223372036854775808");
                    return;
                }

                value = -value;
            }

            Span<byte> tmp = stackalloc byte[24];
            int len = 0;
            while (value > 0)
            {
                int digit = (int)(value % 10);
                value /= 10;
                tmp[len++] = (byte)('0' + digit);
            }

            for (int i = len - 1; i >= 0; i--)
            {
                WriteByte(tmp[i]);
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace()
        {
            WriteByte((byte)' ');
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteLine()
        {
            WriteByte((byte)'\n');
        }

        public void WriteString(string text)
        {
            for (int i = 0; i < text.Length; i++)
            {
                WriteByte((byte)text[i]);
            }
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void WriteByte(byte b)
        {
            if (_position >= _buffer.Length)
            {
                _stream.Write(_buffer, 0, _position);
                _position = 0;
            }

            _buffer[_position++] = b;
        }

        public void Flush()
        {
            if (_position > 0)
            {
                _stream.Write(_buffer, 0, _position);
                _position = 0;
            }

            _stream.Flush();
        }
    }
}
