using System;
using System.IO;
using System.Text;

public static class Program
{
    private struct Cut : IComparable<Cut>
    {
        public long Length;
        public int Index;

        public int CompareTo(Cut other)
        {
            int cmp = other.Length.CompareTo(Length); // descending
            if (cmp != 0) return cmp;
            return Index.CompareTo(other.Index);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            while (true)
            {
                c = Read();
                if (c < 0) return 0;
                if (c == '-' || (uint)(c - '0') <= 9) break;
            }

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    private sealed class FastOutput : IDisposable
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;

        public FastOutput(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public void WriteChar(char c)
        {
            if (_ptr >= _buffer.Length) Flush();
            _buffer[_ptr++] = (byte)c;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteChar('0');
                return;
            }

            if (value < 0)
            {
                WriteChar('-');
                value = -value;
            }

            Span<char> tmp = stackalloc char[12];
            int n = 0;
            while (value > 0)
            {
                int digit = value % 10;
                value /= 10;
                tmp[n++] = (char)('0' + digit);
            }
            while (n-- > 0) WriteChar(tmp[n]);
        }

        public void WriteLong(long value)
        {
            if (value == 0)
            {
                WriteChar('0');
                return;
            }

            if (value < 0)
            {
                WriteChar('-');
                value = -value;
            }

            Span<char> tmp = stackalloc char[24];
            int n = 0;
            while (value > 0)
            {
                long digit = value % 10;
                value /= 10;
                tmp[n++] = (char)('0' + digit);
            }
            while (n-- > 0) WriteChar(tmp[n]);
        }

        public void WriteLine()
        {
            WriteChar('\n');
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }

        public void Dispose()
        {
            Flush();
        }
    }

    private static int[] _nodeLeft = Array.Empty<int>();
    private static int[] _nodeRight = Array.Empty<int>();
    private static long[] _nodeKey = Array.Empty<long>();
    private static int[] _nodePriority = Array.Empty<int>();
    private static int[] _nodeBinHead = Array.Empty<int>();
    private static int[] _binNext = Array.Empty<int>();

    private static int _nodeCount;
    private static int _root;
    private static uint _rng = 2463534242u;

    private static int NextRand()
    {
        _rng ^= _rng << 13;
        _rng ^= _rng >> 17;
        _rng ^= _rng << 5;
        return (int)_rng;
    }

    private static int NewNode(long key, int binId)
    {
        int node = ++_nodeCount;
        _nodeKey[node] = key;
        _nodePriority[node] = NextRand();
        _nodeLeft[node] = 0;
        _nodeRight[node] = 0;
        _nodeBinHead[node] = binId;
        _binNext[binId] = -1;
        return node;
    }

    private static int RotateRight(int node)
    {
        int left = _nodeLeft[node];
        _nodeLeft[node] = _nodeRight[left];
        _nodeRight[left] = node;
        return left;
    }

    private static int RotateLeft(int node)
    {
        int right = _nodeRight[node];
        _nodeRight[node] = _nodeLeft[right];
        _nodeLeft[right] = node;
        return right;
    }

    private static int Merge(int a, int b)
    {
        if (a == 0) return b;
        if (b == 0) return a;
        if (_nodePriority[a] > _nodePriority[b])
        {
            _nodeRight[a] = Merge(_nodeRight[a], b);
            return a;
        }
        else
        {
            _nodeLeft[b] = Merge(a, _nodeLeft[b]);
            return b;
        }
    }

    private static int InsertBin(int node, long key, int binId)
    {
        if (node == 0) return NewNode(key, binId);

        if (key == _nodeKey[node])
        {
            _binNext[binId] = _nodeBinHead[node];
            _nodeBinHead[node] = binId;
            return node;
        }

        if (key < _nodeKey[node])
        {
            _nodeLeft[node] = InsertBin(_nodeLeft[node], key, binId);
            if (_nodePriority[_nodeLeft[node]] > _nodePriority[node]) node = RotateRight(node);
        }
        else
        {
            _nodeRight[node] = InsertBin(_nodeRight[node], key, binId);
            if (_nodePriority[_nodeRight[node]] > _nodePriority[node]) node = RotateLeft(node);
        }

        return node;
    }

    private static int DeleteKey(int node, long key)
    {
        if (node == 0) return 0;

        if (key < _nodeKey[node])
        {
            _nodeLeft[node] = DeleteKey(_nodeLeft[node], key);
            return node;
        }

        if (key > _nodeKey[node])
        {
            _nodeRight[node] = DeleteKey(_nodeRight[node], key);
            return node;
        }

        return Merge(_nodeLeft[node], _nodeRight[node]);
    }

    private static int FindCeiling(int node, long key)
    {
        int best = 0;
        while (node != 0)
        {
            if (_nodeKey[node] >= key)
            {
                best = node;
                node = _nodeLeft[node];
            }
            else
            {
                node = _nodeRight[node];
            }
        }
        return best;
    }

    public static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        int n = (int)input.NextLong();
        long stockLength = input.NextLong();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i].Length = input.NextLong();
            cuts[i].Index = i;
        }

        Array.Sort(cuts);

        _nodeLeft = new int[n + 1];
        _nodeRight = new int[n + 1];
        _nodeKey = new long[n + 1];
        _nodePriority = new int[n + 1];
        _nodeBinHead = new int[n + 1];
        _binNext = new int[n + 1];
        Array.Fill(_nodeBinHead, -1);

        var binCutHead = new int[n];
        Array.Fill(binCutHead, -1);
        var cutNext = new int[n];

        int binCount = 0;

        for (int i = 0; i < n; i++)
        {
            long len = cuts[i].Length;
            int cutIndex = cuts[i].Index;

            int node = FindCeiling(_root, len);
            if (node != 0)
            {
                long remaining = _nodeKey[node];
                int binId = _nodeBinHead[node];
                _nodeBinHead[node] = _binNext[binId];
                if (_nodeBinHead[node] == -1)
                {
                    _root = DeleteKey(_root, remaining);
                }

                long newRemaining = remaining - len;
                cutNext[cutIndex] = binCutHead[binId];
                binCutHead[binId] = cutIndex;

                if (newRemaining > 0)
                {
                    _root = InsertBin(_root, newRemaining, binId);
                }
            }
            else
            {
                int binId = binCount++;
                binCutHead[binId] = -1;
                long newRemaining = stockLength - len;
                cutNext[cutIndex] = binCutHead[binId];
                binCutHead[binId] = cutIndex;

                if (newRemaining > 0)
                {
                    _root = InsertBin(_root, newRemaining, binId);
                }
            }
        }

        using var output = new FastOutput(Console.OpenStandardOutput());
        output.WriteInt(binCount);
        output.WriteLine();

        for (int binId = 0; binId < binCount; binId++)
        {
            int cur = binCutHead[binId];
            bool first = true;
            while (cur != -1)
            {
                if (!first) output.WriteChar(' ');
                output.WriteInt(cur);
                first = false;
                cur = cutNext[cur];
            }
            output.WriteLine();
        }
    }
}
