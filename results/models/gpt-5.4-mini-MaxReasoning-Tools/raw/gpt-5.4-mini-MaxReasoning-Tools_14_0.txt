#nullable disable
using System;
using System.Collections.Generic;

public class Program
{
    private sealed class Constraint
    {
        public int Need;
        public int[] Vars = Array.Empty<int>();
    }

    private static int H;
    private static int W;
    private static string[] Board = Array.Empty<string>();

    private static readonly Dictionary<long, int> VarByCell = new();
    private static readonly List<int> VarX = new();
    private static readonly List<int> VarY = new();
    private static readonly List<List<int>> VarConsTemp = new();

    private static readonly List<Constraint> Constraints = new();

    private static byte[] VarState = Array.Empty<byte>(); // 0 unknown, 1 safe, 2 mine
    private static int[] ConNeed = Array.Empty<int>();
    private static int[] ConUnassigned = Array.Empty<int>();
    private static int[][] ConVars = Array.Empty<int[]>();
    private static int[][] VarCons = Array.Empty<int[]>();

    private static Queue<int> DirtyConstraints = new();
    private static bool[] InQueue = Array.Empty<bool>();
    private static bool Contradiction;

    public static void Main()
    {
        ReadInput();
        BuildGraph();
        GlobalDeterministicClosure();

        var safeIds = new List<int>();
        for (int i = 0; i < VarState.Length; i++)
        {
            if (VarState[i] == 1)
                safeIds.Add(i);
        }

        ProcessExactComponents(safeIds);

        if (safeIds.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        safeIds.Sort((a, b) =>
        {
            int by = VarY[a].CompareTo(VarY[b]);
            if (by != 0) return by;
            return VarX[a].CompareTo(VarX[b]);
        });

        Console.WriteLine(safeIds.Count);
        foreach (int id in safeIds)
        {
            Console.Write(VarX[id]);
            Console.Write(' ');
            Console.WriteLine(VarY[id]);
        }
    }

    private static void ReadInput()
    {
        string header = Console.ReadLine();
        while (header != null && header.Length == 0)
            header = Console.ReadLine();

        if (header == null)
        {
            H = 0;
            W = 0;
            Board = Array.Empty<string>();
            return;
        }

        var parts = header.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        Board = new string[H];
        for (int r = 0; r < H; r++)
        {
            string line = Console.ReadLine();
            if (line == null)
                line = string.Empty;
            if (line.Length < W)
                line = line.PadRight(W, ' ');
            else if (line.Length > W)
                line = line.Substring(0, W);
            Board[r] = line;
        }
    }

    private static void BuildGraph()
    {
        VarByCell.Clear();
        VarX.Clear();
        VarY.Clear();
        VarConsTemp.Clear();
        Constraints.Clear();

        if (H == 0 || W == 0)
            return;

        for (int r = 0; r < H; r++)
        {
            string row = Board[r];
            for (int c = 0; c < W; c++)
            {
                char ch = row[c];
                if (ch < '0' || ch > '8')
                    continue;

                int need = ch - '0';
                int flagged = 0;
                var localVars = new List<int>(8);

                for (int dr = -1; dr <= 1; dr++)
                {
                    int nr = r + dr;
                    if ((uint)nr >= (uint)H) continue;
                    string nrow = Board[nr];
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        int nc = c + dc;
                        if ((uint)nc >= (uint)W) continue;

                        char nch = nrow[nc];
                        if (nch == '*')
                        {
                            flagged++;
                        }
                        else if (nch == ' ')
                        {
                            long key = (((long)nr) << 32) | (uint)nc;
                            if (!VarByCell.TryGetValue(key, out int id))
                            {
                                id = VarX.Count;
                                VarByCell[key] = id;
                                VarX.Add(nc);
                                VarY.Add(nr);
                                VarConsTemp.Add(new List<int>(4));
                            }
                            localVars.Add(id);
                        }
                    }
                }

                if (localVars.Count == 0 && need == flagged)
                    continue;

                var constraint = new Constraint
                {
                    Need = need - flagged,
                    Vars = localVars.DistinctSorted()
                };
                int conId = Constraints.Count;
                Constraints.Add(constraint);
                foreach (int v in constraint.Vars)
                    VarConsTemp[v].Add(conId);
            }
        }

        VarState = new byte[VarX.Count];
        InQueue = new bool[Constraints.Count];
        ConNeed = new int[Constraints.Count];
        ConUnassigned = new int[Constraints.Count];
        ConVars = new int[Constraints.Count][];
        for (int i = 0; i < Constraints.Count; i++)
        {
            ConNeed[i] = Constraints[i].Need;
            ConVars[i] = Constraints[i].Vars;
            ConUnassigned[i] = ConVars[i].Length;
        }

        VarCons = new int[VarX.Count][];
        for (int v = 0; v < VarX.Count; v++)
            VarCons[v] = VarConsTemp[v].ToArray();
    }

    private static void GlobalDeterministicClosure()
    {
        DirtyConstraints = new Queue<int>(Constraints.Count);
        for (int i = 0; i < Constraints.Count; i++)
        {
            if (ConUnassigned[i] > 0)
            {
                DirtyConstraints.Enqueue(i);
                InQueue[i] = true;
            }
        }

        while (DirtyConstraints.Count > 0 && !Contradiction)
        {
            int c = DirtyConstraints.Dequeue();
            InQueue[c] = false;
            if (ConUnassigned[c] == 0)
                continue;

            if (ConNeed[c] < 0 || ConNeed[c] > ConUnassigned[c])
            {
                Contradiction = true;
                break;
            }

            if (ConNeed[c] == 0)
            {
                foreach (int v in ConVars[c])
                    AssignGlobal(v, 1);
            }
            else if (ConNeed[c] == ConUnassigned[c])
            {
                foreach (int v in ConVars[c])
                    AssignGlobal(v, 2);
            }

            if (Contradiction)
                break;

            foreach (int v in ConVars[c])
            {
                if (VarState[v] != 0)
                    continue;
                foreach (int other in VarCons[v])
                {
                    if (other == c || ConUnassigned[other] == 0)
                        continue;
                    CheckSubsetGlobal(c, other);
                    if (Contradiction)
                        break;
                }
                if (Contradiction)
                    break;
            }
        }
    }

    private static void AssignGlobal(int v, byte value)
    {
        byte cur = VarState[v];
        if (cur == value)
            return;
        if (cur != 0)
        {
            Contradiction = true;
            return;
        }

        VarState[v] = value;
        foreach (int c in VarCons[v])
        {
            ConUnassigned[c]--;
            if (value == 2)
                ConNeed[c]--;
            if (!InQueue[c])
            {
                InQueue[c] = true;
                DirtyConstraints.Enqueue(c);
            }
        }
    }

    private static void CheckSubsetGlobal(int a, int b)
    {
        if (a == b || Contradiction)
            return;

        int[] av = ConVars[a];
        int[] bv = ConVars[b];

        bool aSubsetB = true;
        int diffAB = 0;

        for (int i = 0; i < av.Length; i++)
        {
            int v = av[i];
            if (VarState[v] != 0)
                continue;
            if (!ContainsUnassigned(bv, v))
            {
                aSubsetB = false;
                break;
            }
        }

        if (aSubsetB)
        {
            for (int i = 0; i < bv.Length; i++)
            {
                int v = bv[i];
                if (VarState[v] != 0)
                    continue;
                if (!ContainsUnassigned(av, v))
                    diffAB++;
            }

            if (ConNeed[a] == ConNeed[b])
            {
                for (int i = 0; i < bv.Length; i++)
                {
                    int v = bv[i];
                    if (VarState[v] == 0 && !ContainsUnassigned(av, v))
                        AssignGlobal(v, 1);
                    if (Contradiction)
                        return;
                }
            }
            else if (ConNeed[b] - ConNeed[a] == diffAB)
            {
                for (int i = 0; i < bv.Length; i++)
                {
                    int v = bv[i];
                    if (VarState[v] == 0 && !ContainsUnassigned(av, v))
                        AssignGlobal(v, 2);
                    if (Contradiction)
                        return;
                }
            }
        }

        bool bSubsetA = true;
        int diffBA = 0;
        for (int i = 0; i < bv.Length; i++)
        {
            int v = bv[i];
            if (VarState[v] != 0)
                continue;
            if (!ContainsUnassigned(av, v))
            {
                bSubsetA = false;
                break;
            }
        }

        if (bSubsetA)
        {
            for (int i = 0; i < av.Length; i++)
            {
                int v = av[i];
                if (VarState[v] != 0)
                    continue;
                if (!ContainsUnassigned(bv, v))
                    diffBA++;
            }

            if (ConNeed[a] == ConNeed[b])
            {
                for (int i = 0; i < av.Length; i++)
                {
                    int v = av[i];
                    if (VarState[v] == 0 && !ContainsUnassigned(bv, v))
                        AssignGlobal(v, 1);
                    if (Contradiction)
                        return;
                }
            }
            else if (ConNeed[a] - ConNeed[b] == diffBA)
            {
                for (int i = 0; i < av.Length; i++)
                {
                    int v = av[i];
                    if (VarState[v] == 0 && !ContainsUnassigned(bv, v))
                        AssignGlobal(v, 2);
                    if (Contradiction)
                        return;
                }
            }
        }
    }

    private static bool ContainsUnassigned(int[] vars, int v)
    {
        for (int i = 0; i < vars.Length; i++)
        {
            int t = vars[i];
            if (t == v)
                return VarState[t] == 0;
        }
        return false;
    }

    private static void ProcessExactComponents(List<int> safeIds)
    {
        bool[] seenVar = new bool[VarState.Length];
        bool[] seenCon = new bool[Constraints.Count];
        var queueVars = new Queue<int>();

        int totalVars = 0;
        int totalCons = 0;
        for (int i = 0; i < VarState.Length; i++)
            if (VarState[i] == 0) totalVars++;
        for (int i = 0; i < Constraints.Count; i++)
            if (ConUnassigned[i] > 0) totalCons++;

        int exactThreshold = totalVars <= 400 ? 60 : totalVars <= 1500 ? 50 : 40;
        int exactConThreshold = totalCons <= 400 ? 60 : totalCons <= 1500 ? 50 : 40;

        for (int start = 0; start < VarState.Length; start++)
        {
            if (VarState[start] != 0 || seenVar[start])
                continue;

            if (VarCons[start].Length == 0)
                continue;

            var compVars = new List<int>();
            var compCons = new List<int>();

            seenVar[start] = true;
            queueVars.Enqueue(start);

            while (queueVars.Count > 0)
            {
                int v = queueVars.Dequeue();
                compVars.Add(v);
                foreach (int c in VarCons[v])
                {
                    if (ConUnassigned[c] == 0 || seenCon[c])
                        continue;
                    seenCon[c] = true;
                    compCons.Add(c);
                    foreach (int nv in ConVars[c])
                    {
                        if (VarState[nv] != 0 || seenVar[nv])
                            continue;
                        seenVar[nv] = true;
                        queueVars.Enqueue(nv);
                    }
                }
            }

            if (compVars.Count == 0 || compCons.Count == 0)
                continue;

            if (compVars.Count > exactThreshold || compCons.Count > exactConThreshold)
                continue;

            if (SolveComponentExactly(compVars, compCons, out bool[] minePossible))
            {
                for (int i = 0; i < compVars.Count; i++)
                {
                    int v = compVars[i];
                    if (!minePossible[i])
                    {
                        VarState[v] = 1;
                        safeIds.Add(v);
                    }
                }
            }
        }
    }

    private static bool SolveComponentExactly(List<int> compVars, List<int> compCons, out bool[] minePossible)
    {
        minePossible = Array.Empty<bool>();
        int n = compVars.Count;
        int m = compCons.Count;

        var localIndex = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++)
            localIndex[compVars[i]] = i;

        int[][] consVars = new int[m][];
        int[] need = new int[m];
        for (int i = 0; i < m; i++)
        {
            int c = compCons[i];
            need[i] = ConNeed[c];
            var arr = ConVars[c];
            var list = new List<int>(arr.Length);
            for (int j = 0; j < arr.Length; j++)
            {
                int v = arr[j];
                if (VarState[v] != 0)
                    continue;
                if (localIndex.TryGetValue(v, out int lid))
                    list.Add(lid);
            }
            consVars[i] = list.ToArray();
            if (need[i] < 0 || need[i] > consVars[i].Length)
                return false;
        }

        bool[] canBeMine = new bool[n];
        byte[] assign = new byte[n];
        int nodeLimit = 3000000 + 20000 * n;
        int nodes = 0;
        bool foundSolution = false;
        bool completed = SearchAll(assign, consVars, need, canBeMine, ref foundSolution, ref nodes, nodeLimit);
        if (!completed || !foundSolution)
            return false;

        minePossible = canBeMine;
        return true;
    }

    private static bool SearchAll(byte[] assign, int[][] consVars, int[] need, bool[] canBeMine, ref bool foundSolution, ref int nodes, int nodeLimit)
    {
        if (++nodes > nodeLimit)
            return false;

        if (!Propagate(ref assign, consVars, need))
            return true;

        int unassigned = -1;
        for (int i = 0; i < assign.Length; i++)
        {
            if (assign[i] == 0)
            {
                unassigned = i;
                break;
            }
        }

        if (unassigned < 0)
        {
            foundSolution = true;
            for (int i = 0; i < assign.Length; i++)
                if (assign[i] == 2) canBeMine[i] = true;
            return true;
        }

        if (AllVarsAlreadyPossibleMine(canBeMine))
            return true;

        int score = -1;
        int branchVar = unassigned;
        for (int i = 0; i < assign.Length; i++)
        {
            if (assign[i] != 0) continue;
            int s = 0;
            for (int c = 0; c < consVars.Length; c++)
            {
                var cv = consVars[c];
                for (int j = 0; j < cv.Length; j++)
                {
                    if (cv[j] == i)
                    {
                        s += 8 - cv.Length;
                        break;
                    }
                }
            }
            if (s > score)
            {
                score = s;
                branchVar = i;
            }
        }

        bool complete = true;

        var assignSafe = (byte[])assign.Clone();
        assignSafe[branchVar] = 1;
        if (!SearchAll(assignSafe, consVars, need, canBeMine, ref foundSolution, ref nodes, nodeLimit))
            complete = false;

        var assignMine = (byte[])assign.Clone();
        assignMine[branchVar] = 2;
        if (!SearchAll(assignMine, consVars, need, canBeMine, ref foundSolution, ref nodes, nodeLimit))
            complete = false;

        return complete;
    }

    private static bool AllVarsAlreadyPossibleMine(bool[] canBeMine)
    {
        for (int i = 0; i < canBeMine.Length; i++)
            if (!canBeMine[i])
                return false;
        return true;
    }

    private static bool Propagate(ref byte[] assign, int[][] consVars, int[] need)
    {
        bool changed;
        do
        {
            changed = false;
            for (int c = 0; c < consVars.Length; c++)
            {
                int remaining = need[c];
                int unassigned = 0;
                int[] vars = consVars[c];
                for (int i = 0; i < vars.Length; i++)
                {
                    byte a = assign[vars[i]];
                    if (a == 2) remaining--;
                    else if (a == 0) unassigned++;
                }

                if (remaining < 0 || remaining > unassigned)
                    return false;

                if (unassigned == 0)
                    continue;

                if (remaining == 0)
                {
                    for (int i = 0; i < vars.Length; i++)
                    {
                        int v = vars[i];
                        if (assign[v] == 0)
                        {
                            assign[v] = 1;
                            changed = true;
                        }
                        else if (assign[v] == 2)
                        {
                            return false;
                        }
                    }
                }
                else if (remaining == unassigned)
                {
                    for (int i = 0; i < vars.Length; i++)
                    {
                        int v = vars[i];
                        if (assign[v] == 0)
                        {
                            assign[v] = 2;
                            changed = true;
                        }
                        else if (assign[v] == 1)
                        {
                            return false;
                        }
                    }
                }
            }

            for (int a = 0; a < consVars.Length; a++)
            {
                int[] av = consVars[a];
                int remA = need[a];
                int unA = 0;
                for (int i = 0; i < av.Length; i++)
                {
                    byte st = assign[av[i]];
                    if (st == 2) remA--;
                    else if (st == 0) unA++;
                }
                if (remA < 0 || remA > unA)
                    return false;

                for (int b = a + 1; b < consVars.Length; b++)
                {
                    int[] bv = consVars[b];
                    int remB = need[b];
                    int unB = 0;
                    for (int i = 0; i < bv.Length; i++)
                    {
                        byte st = assign[bv[i]];
                        if (st == 2) remB--;
                        else if (st == 0) unB++;
                    }
                    if (remB < 0 || remB > unB)
                        return false;

                    bool aSubsetB = true;
                    int diffAB = 0;
                    for (int i = 0; i < av.Length; i++)
                    {
                        int v = av[i];
                        if (assign[v] != 0)
                            continue;
                        if (!ContainsUnassignedLocal(bv, assign, v))
                        {
                            aSubsetB = false;
                            break;
                        }
                    }
                    if (aSubsetB)
                    {
                        for (int i = 0; i < bv.Length; i++)
                        {
                            int v = bv[i];
                            if (assign[v] != 0)
                                continue;
                            if (!ContainsUnassignedLocal(av, assign, v))
                                diffAB++;
                        }
                        if (remA == remB)
                        {
                            for (int i = 0; i < bv.Length; i++)
                            {
                                int v = bv[i];
                                if (assign[v] == 0 && !ContainsUnassignedLocal(av, assign, v))
                                {
                                    assign[v] = 1;
                                    changed = true;
                                }
                            }
                        }
                        else if (remB - remA == diffAB)
                        {
                            for (int i = 0; i < bv.Length; i++)
                            {
                                int v = bv[i];
                                if (assign[v] == 0 && !ContainsUnassignedLocal(av, assign, v))
                                {
                                    assign[v] = 2;
                                    changed = true;
                                }
                            }
                        }
                    }

                    bool bSubsetA = true;
                    int diffBA = 0;
                    for (int i = 0; i < bv.Length; i++)
                    {
                        int v = bv[i];
                        if (assign[v] != 0)
                            continue;
                        if (!ContainsUnassignedLocal(av, assign, v))
                        {
                            bSubsetA = false;
                            break;
                        }
                    }
                    if (bSubsetA)
                    {
                        for (int i = 0; i < av.Length; i++)
                        {
                            int v = av[i];
                            if (assign[v] != 0)
                                continue;
                            if (!ContainsUnassignedLocal(bv, assign, v))
                                diffBA++;
                        }
                        if (remA == remB)
                        {
                            for (int i = 0; i < av.Length; i++)
                            {
                                int v = av[i];
                                if (assign[v] == 0 && !ContainsUnassignedLocal(bv, assign, v))
                                {
                                    assign[v] = 1;
                                    changed = true;
                                }
                            }
                        }
                        else if (remA - remB == diffBA)
                        {
                            for (int i = 0; i < av.Length; i++)
                            {
                                int v = av[i];
                                if (assign[v] == 0 && !ContainsUnassignedLocal(bv, assign, v))
                                {
                                    assign[v] = 2;
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
        }
        while (changed);

        return true;
    }

    private static bool ContainsUnassignedLocal(int[] vars, byte[] assign, int v)
    {
        for (int i = 0; i < vars.Length; i++)
        {
            if (vars[i] == v)
                return assign[v] == 0;
        }
        return false;
    }
}

internal static class Extensions
{
    public static int[] DistinctSorted(this List<int> values)
    {
        if (values.Count == 0)
            return Array.Empty<int>();
        values.Sort();
        int write = 1;
        for (int read = 1; read < values.Count; read++)
        {
            if (values[read] != values[write - 1])
                values[write++] = values[read];
        }
        var arr = new int[write];
        for (int i = 0; i < write; i++)
            arr[i] = values[i];
        return arr;
    }
}
