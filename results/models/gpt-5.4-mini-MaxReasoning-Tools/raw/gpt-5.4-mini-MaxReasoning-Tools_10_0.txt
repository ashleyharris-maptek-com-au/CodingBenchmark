// 2D cutting stock / rectangle bin packing heuristic solver.
// Packs rectangles with a best-fit decreasing shelf strategy and compares
// horizontal/vertical variants plus several orientation rules.

#include <algorithm>
#include <iostream>
#include <limits>
#include <set>
#include <utility>
#include <vector>

struct Rect {
    int w;
    int h;
};

struct Candidate {
    int w = 0;
    int h = 0;
    int rot = 0;
    bool valid = false;
};

struct PackedItem {
    int idx = 0;
    int w = 0;
    int h = 0;
    int rot = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rot = 0;
};

struct BoardState {
    int usedH = 0;
    int remainingH = 0;
};

struct ShelfState {
    int board = -1;
    int y = 0;
    int usedW = 0;
    int remainingW = 0;
};

struct VariantConfig {
    bool swapAxes = false;
    int rule = 0;
};

struct VariantResult {
    bool valid = false;
    int boardCount = 0;
    std::vector<Placement> placements;
};

static long long absll(long long x) {
    return x < 0 ? -x : x;
}

static std::pair<int, int> orientDims(const Rect& r, bool swapAxes, int rot) {
    if (!swapAxes) {
        return rot == 0 ? std::make_pair(r.w, r.h) : std::make_pair(r.h, r.w);
    }
    return rot == 0 ? std::make_pair(r.h, r.w) : std::make_pair(r.w, r.h);
}

static bool betterCandidate(const Candidate& a, const Candidate& b, int rule, int boardW, int boardH) {
    if (rule == 0) {
        if (a.h != b.h) return a.h < b.h;
        if (a.w != b.w) return a.w < b.w;
        return a.rot < b.rot;
    }
    if (rule == 1) {
        if (a.w != b.w) return a.w < b.w;
        if (a.h != b.h) return a.h < b.h;
        return a.rot < b.rot;
    }
    long long da = absll(static_cast<long long>(a.w) * boardH - static_cast<long long>(a.h) * boardW);
    long long db = absll(static_cast<long long>(b.w) * boardH - static_cast<long long>(b.h) * boardW);
    if (da != db) return da < db;
    if (a.h != b.h) return a.h < b.h;
    if (a.w != b.w) return a.w < b.w;
    return a.rot < b.rot;
}

static Candidate chooseCandidate(const Rect& r, bool swapAxes, int rule, int boardW, int boardH) {
    Candidate c0;
    Candidate c1;
    {
        auto [w, h] = orientDims(r, swapAxes, 0);
        c0.w = w;
        c0.h = h;
        c0.rot = 0;
        c0.valid = (w <= boardW && h <= boardH);
    }
    {
        auto [w, h] = orientDims(r, swapAxes, 1);
        c1.w = w;
        c1.h = h;
        c1.rot = 1;
        c1.valid = (w <= boardW && h <= boardH);
    }

    if (!c0.valid && !c1.valid) {
        return Candidate{};
    }
    if (c0.valid && !c1.valid) {
        return c0;
    }
    if (!c0.valid && c1.valid) {
        return c1;
    }
    return betterCandidate(c1, c0, rule, boardW, boardH) ? c1 : c0;
}

static VariantResult packVariant(const std::vector<Rect>& rects, int boardW, int boardH, bool swapAxes, int rule) {
    const int n = static_cast<int>(rects.size());
    std::vector<PackedItem> items;
    items.reserve(n);

    for (int i = 0; i < n; ++i) {
        Candidate c = chooseCandidate(rects[i], swapAxes, rule, boardW, boardH);
        if (!c.valid) {
            return VariantResult{};
        }
        items.push_back(PackedItem{i, c.w, c.h, c.rot});
    }

    std::sort(items.begin(), items.end(), [](const PackedItem& a, const PackedItem& b) {
        if (a.h != b.h) return a.h > b.h;
        if (a.w != b.w) return a.w > b.w;
        return a.idx < b.idx;
    });

    std::vector<BoardState> boards;
    std::vector<ShelfState> shelves;
    boards.reserve(n);
    shelves.reserve(n);

    std::set<std::pair<int, int>> boardSet;
    std::set<std::pair<int, int>> shelfSet;

    std::vector<Placement> placements(n);

    for (const PackedItem& item : items) {
        const int w = item.w;
        const int h = item.h;

        auto shelfIt = shelfSet.lower_bound(std::make_pair(w, std::numeric_limits<int>::min()));
        if (shelfIt != shelfSet.end()) {
            const int sid = shelfIt->second;
            shelfSet.erase(shelfIt);

            ShelfState& shelf = shelves[sid];
            const int x = shelf.usedW;
            const int y = shelf.y;
            shelf.usedW += w;
            shelf.remainingW -= w;
            if (shelf.remainingW > 0) {
                shelfSet.insert(std::make_pair(shelf.remainingW, sid));
            }

            Placement p;
            p.board = shelf.board;
            p.rot = item.rot;
            if (!swapAxes) {
                p.x = x;
                p.y = y;
            } else {
                p.x = y;
                p.y = x;
            }
            placements[item.idx] = p;
            continue;
        }

        int bid = -1;
        auto boardIt = boardSet.lower_bound(std::make_pair(h, std::numeric_limits<int>::min()));
        if (boardIt != boardSet.end()) {
            bid = boardIt->second;
            boardSet.erase(boardIt);
        } else {
            bid = static_cast<int>(boards.size());
            boards.push_back(BoardState{0, boardH});
        }

        BoardState& board = boards[bid];
        const int y = board.usedH;
        board.usedH += h;
        board.remainingH -= h;
        if (board.remainingH > 0) {
            boardSet.insert(std::make_pair(board.remainingH, bid));
        }

        const int sid = static_cast<int>(shelves.size());
        shelves.push_back(ShelfState{bid, y, w, boardW - w});
        if (shelves[sid].remainingW > 0) {
            shelfSet.insert(std::make_pair(shelves[sid].remainingW, sid));
        }

        Placement p;
        p.board = bid;
        p.rot = item.rot;
        if (!swapAxes) {
            p.x = 0;
            p.y = y;
        } else {
            p.x = y;
            p.y = 0;
        }
        placements[item.idx] = p;
    }

    VariantResult result;
    result.valid = true;
    result.boardCount = static_cast<int>(boards.size());
    result.placements = std::move(placements);
    return result;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    int boardW = 0;
    int boardH = 0;
    if (!(std::cin >> n >> boardW >> boardH)) {
        return 0;
    }

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> rects[i].w >> rects[i].h;
    }

    std::vector<VariantConfig> variants;
    variants.reserve(6);
    if (boardW >= boardH) {
        variants.push_back(VariantConfig{false, 0});
        variants.push_back(VariantConfig{false, 2});
        variants.push_back(VariantConfig{false, 1});
        variants.push_back(VariantConfig{true, 0});
        variants.push_back(VariantConfig{true, 2});
        variants.push_back(VariantConfig{true, 1});
    } else {
        variants.push_back(VariantConfig{true, 0});
        variants.push_back(VariantConfig{true, 2});
        variants.push_back(VariantConfig{true, 1});
        variants.push_back(VariantConfig{false, 0});
        variants.push_back(VariantConfig{false, 2});
        variants.push_back(VariantConfig{false, 1});
    }

    VariantResult best;
    for (const VariantConfig& cfg : variants) {
        VariantResult cur = packVariant(rects, cfg.swapAxes ? boardH : boardW, cfg.swapAxes ? boardW : boardH, cfg.swapAxes, cfg.rule);
        if (!cur.valid) {
            continue;
        }
        if (!best.valid || cur.boardCount < best.boardCount) {
            best = std::move(cur);
        }
    }

    if (!best.valid) {
        // Fallback for pathological impossible instances: place each rectangle on its own board
        // using the first orientation that fits if possible.
        best.valid = true;
        best.boardCount = n;
        best.placements.resize(n);
        for (int i = 0; i < n; ++i) {
            int rot = 0;
            if (!(rects[i].w <= boardW && rects[i].h <= boardH) && (rects[i].h <= boardW && rects[i].w <= boardH)) {
                rot = 1;
            }
            best.placements[i] = Placement{i, 0, 0, rot};
        }
    }

    std::cout << best.boardCount << '\n';
    for (int i = 0; i < n; ++i) {
        const Placement& p = best.placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rot << '\n';
    }
    return 0;
}
