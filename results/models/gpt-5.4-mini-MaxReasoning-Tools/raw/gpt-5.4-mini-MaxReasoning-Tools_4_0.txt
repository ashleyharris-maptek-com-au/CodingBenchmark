#include <algorithm>
#include <array>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <unordered_map>
#include <vector>

using namespace std;

struct Pt {
    long double x = 0;
    long double y = 0;
    long double z = 0;
};

static Pt operator+(const Pt& a, const Pt& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Pt operator-(const Pt& a, const Pt& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Pt operator*(const Pt& a, long double s) { return {a.x * s, a.y * s, a.z * s}; }
static Pt operator/(const Pt& a, long double s) { return {a.x / s, a.y / s, a.z / s}; }

static long double dot(const Pt& a, const Pt& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static Pt cross(const Pt& a, const Pt& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}
static long double norm2(const Pt& a) { return dot(a, a); }
static long double dist2(const Pt& a, const Pt& b) { return norm2(a - b); }

struct Face {
    int a = -1, b = -1, c = -1;
    long double nx = 0, ny = 0, nz = 0, d = 0;
};

struct Placement {
    long double cx = 0, cy = 0, cz = 0;
    long double rx = 0, ry = 0, rz = 0;
};

struct EdgeRec {
    int cnt = 0;
    int u = 0, v = 0;
};

static uint64_t edgeKey(int a, int b) {
    if (a > b) swap(a, b);
    return (uint64_t(uint32_t(a)) << 32) | uint32_t(b);
}

static long double positiveMod(long double x, long double m) {
    long double r = fmodl(x, m);
    if (r < 0) r += m;
    if (r >= m) r -= m;
    return r;
}

static Face makeFace(int a, int b, int c, const Pt& inside, const vector<Pt>& pts) {
    Face f;
    f.a = a;
    f.b = b;
    f.c = c;

    Pt n = cross(pts[b] - pts[a], pts[c] - pts[a]);
    long double len2 = norm2(n);
    if (len2 <= 0) {
        f.a = f.b = f.c = -1;
        return f;
    }
    long double len = sqrtl(len2);
    n = n / len;
    long double d = -dot(n, pts[a]);
    if (dot(n, inside) + d > 0) {
        swap(f.b, f.c);
        n = n * -1;
        d = -d;
    }

    f.nx = n.x;
    f.ny = n.y;
    f.nz = n.z;
    f.d = d;
    return f;
}

static long double signedDist(const Face& f, const Pt& p) {
    return f.nx * p.x + f.ny * p.y + f.nz * p.z + f.d;
}

static bool pointInsideHull(const Pt& p, const vector<Face>& faces, long double eps) {
    for (const Face& f : faces) {
        if (signedDist(f, p) > eps) return false;
    }
    return true;
}

static bool pointInsideAABB(const Pt& p, const Pt& mn, const Pt& mx, long double eps) {
    return p.x >= mn.x - eps && p.x <= mx.x + eps &&
           p.y >= mn.y - eps && p.y <= mx.y + eps &&
           p.z >= mn.z - eps && p.z <= mx.z + eps;
}

static int lexMinIndex(const vector<Pt>& pts) {
    int id = 0;
    for (int i = 1; i < (int)pts.size(); ++i) {
        if (pts[i].x < pts[id].x ||
            (pts[i].x == pts[id].x && (pts[i].y < pts[id].y ||
            (pts[i].y == pts[id].y && pts[i].z < pts[id].z)))) {
            id = i;
        }
    }
    return id;
}

static vector<Face> buildConvexHull(const vector<Pt>& inputPts, Pt& bboxMin, Pt& bboxMax, bool& ok) {
    ok = false;
    vector<Pt> pts = inputPts;
    int n = (int)pts.size();
    vector<Face> empty;
    if (n < 4) return empty;

    bboxMin = bboxMax = pts[0];
    for (const Pt& p : pts) {
        bboxMin.x = min(bboxMin.x, p.x);
        bboxMin.y = min(bboxMin.y, p.y);
        bboxMin.z = min(bboxMin.z, p.z);
        bboxMax.x = max(bboxMax.x, p.x);
        bboxMax.y = max(bboxMax.y, p.y);
        bboxMax.z = max(bboxMax.z, p.z);
    }
    long double scale = max({1.0L,
                             bboxMax.x - bboxMin.x,
                             bboxMax.y - bboxMin.y,
                             bboxMax.z - bboxMin.z});
    long double distEps = scale * 1e-12L + 1e-15L;
    long double areaEps = scale * scale * 1e-12L + 1e-15L;
    long double volEps = scale * scale * scale * 1e-15L + 1e-18L;
    long double planeEps = scale * 1e-12L + 1e-15L;

    int i0 = lexMinIndex(pts);
    int i1 = -1;
    long double bestD2 = -1;
    for (int i = 0; i < n; ++i) {
        if (i == i0) continue;
        long double d2 = dist2(pts[i0], pts[i]);
        if (d2 > bestD2) {
            bestD2 = d2;
            i1 = i;
        }
    }
    if (i1 < 0 || bestD2 <= distEps * distEps) return empty;

    Pt ab = pts[i1] - pts[i0];
    int i2 = -1;
    long double bestArea2 = -1;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1) continue;
        Pt ap = pts[i] - pts[i0];
        long double a2 = norm2(cross(ab, ap));
        if (a2 > bestArea2) {
            bestArea2 = a2;
            i2 = i;
        }
    }
    if (i2 < 0 || bestArea2 <= areaEps * areaEps) return empty;

    Pt n012 = cross(pts[i1] - pts[i0], pts[i2] - pts[i0]);
    int i3 = -1;
    long double bestVol = -1;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1 || i == i2) continue;
        long double v6 = fabsl(dot(n012, pts[i] - pts[i0]));
        if (v6 > bestVol) {
            bestVol = v6;
            i3 = i;
        }
    }
    if (i3 < 0 || bestVol <= volEps) return empty;

    Pt interior = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0L;

    vector<Face> faces;
    faces.reserve(max(8, 4 * n));
    auto addInitial = [&](int a, int b, int c) {
        Face f = makeFace(a, b, c, interior, pts);
        if (f.a >= 0) faces.push_back(f);
    };
    addInitial(i0, i1, i2);
    addInitial(i0, i3, i1);
    addInitial(i0, i2, i3);
    addInitial(i1, i3, i2);

    for (int pi = 0; pi < n; ++pi) {
        if (pi == i0 || pi == i1 || pi == i2 || pi == i3) continue;

        vector<char> visible(faces.size(), 0);
        vector<int> visList;
        visList.reserve(faces.size() / 4 + 1);
        for (int fi = 0; fi < (int)faces.size(); ++fi) {
            if (signedDist(faces[fi], pts[pi]) > planeEps) {
                visible[fi] = 1;
                visList.push_back(fi);
            }
        }
        if (visList.empty()) continue;

        unordered_map<uint64_t, EdgeRec> edgeMap;
        edgeMap.reserve(visList.size() * 4 + 8);
        vector<Face> nextFaces;
        nextFaces.reserve(faces.size() + visList.size() * 2 + 8);

        for (int fi = 0; fi < (int)faces.size(); ++fi) {
            if (!visible[fi]) {
                nextFaces.push_back(faces[fi]);
                continue;
            }
            const Face& f = faces[fi];
            int e0u = f.a, e0v = f.b;
            int e1u = f.b, e1v = f.c;
            int e2u = f.c, e2v = f.a;
            uint64_t k0 = edgeKey(e0u, e0v);
            uint64_t k1 = edgeKey(e1u, e1v);
            uint64_t k2 = edgeKey(e2u, e2v);
            auto& r0 = edgeMap[k0];
            if (r0.cnt == 0) { r0.u = e0u; r0.v = e0v; }
            ++r0.cnt;
            auto& r1 = edgeMap[k1];
            if (r1.cnt == 0) { r1.u = e1u; r1.v = e1v; }
            ++r1.cnt;
            auto& r2 = edgeMap[k2];
            if (r2.cnt == 0) { r2.u = e2u; r2.v = e2v; }
            ++r2.cnt;
        }

        pts.push_back(pts[pi]);
        int newIdx = (int)pts.size() - 1;
        for (const auto& kv : edgeMap) {
            if (kv.second.cnt != 1) continue;
            Face nf = makeFace(kv.second.u, kv.second.v, newIdx, interior, pts);
            if (nf.a >= 0) nextFaces.push_back(nf);
        }
        faces.swap(nextFaces);
    }

    ok = !faces.empty();
    return faces;
}

static bool insideCandidate(const Pt& p, const vector<Face>& faces, long double eps, const Pt& bboxMin, const Pt& bboxMax) {
    if (!pointInsideAABB(p, bboxMin, bboxMax, eps)) return false;
    return pointInsideHull(p, faces, eps);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    long double edgeLength;
    if (!(cin >> V >> edgeLength)) return 0;

    vector<Pt> pts;
    pts.reserve(max(0, V));
    for (int i = 0; i < V; ++i) {
        Pt p;
        cin >> p.x >> p.y >> p.z;
        pts.push_back(p);
    }

    cout.setf(std::ios::fmtflags(0), std::ios::floatfield);
    cout << setprecision(17);

    if (pts.size() < 4 || edgeLength <= 0) {
        cout << 0 << '\n';
        return 0;
    }

    Pt bboxMin, bboxMax;
    bool hullOk = false;
    vector<Face> hull = buildConvexHull(pts, bboxMin, bboxMax, hullOk);
    if (!hullOk) {
        cout << 0 << '\n';
        return 0;
    }

    long double s = edgeLength / sqrtl(2.0L);
    long double half = s * 0.5L;
    long double pi = acosl(-1.0L);
    long double halfPi = pi * 0.5L;

    long double scale = max({1.0L,
                             bboxMax.x - bboxMin.x,
                             bboxMax.y - bboxMin.y,
                             bboxMax.z - bboxMin.z});
    long double eps = scale * 1e-12L + 1e-15L;

    auto modBase = [&](long double mn) {
        return positiveMod(mn, s);
    };

    Pt baseOff{modBase(bboxMin.x), modBase(bboxMin.y), modBase(bboxMin.z)};
    array<Pt, 8> offsets{};
    int offsetCount = 1;
    long double targetCells = 50000.0L;
    long double xSpan = max(0.0L, bboxMax.x - bboxMin.x - s);
    long double ySpan = max(0.0L, bboxMax.y - bboxMin.y - s);
    long double zSpan = max(0.0L, bboxMax.z - bboxMin.z - s);
    long double estCells = 0;
    if (xSpan >= 0 && ySpan >= 0 && zSpan >= 0) {
        long long nx = (long long)floorl(xSpan / s) + 1;
        long long ny = (long long)floorl(ySpan / s) + 1;
        long long nz = (long long)floorl(zSpan / s) + 1;
        if (nx > 0 && ny > 0 && nz > 0) estCells = (long double)nx * (long double)ny * (long double)nz;
    }
    if (estCells > 0 && estCells <= 50000.0L) {
        offsetCount = 8;
        int idx = 0;
        for (int mask = 0; mask < 8; ++mask) {
            Pt o = baseOff;
            if (mask & 1) o.x += half;
            if (mask & 2) o.y += half;
            if (mask & 4) o.z += half;
            if (o.x >= s) o.x -= s;
            if (o.y >= s) o.y -= s;
            if (o.z >= s) o.z -= s;
            offsets[idx++] = o;
        }
    } else {
        offsets[0] = baseOff;
    }

    auto estimateForOffset = [&](const Pt& off, long long step) -> long double {
        long double ix0d = ceill((bboxMin.x - off.x) / s - 1e-15L);
        long double ix1d = floorl((bboxMax.x - s - off.x) / s + 1e-15L);
        long double iy0d = ceill((bboxMin.y - off.y) / s - 1e-15L);
        long double iy1d = floorl((bboxMax.y - s - off.y) / s + 1e-15L);
        long double iz0d = ceill((bboxMin.z - off.z) / s - 1e-15L);
        long double iz1d = floorl((bboxMax.z - s - off.z) / s + 1e-15L);
        if (ix0d > ix1d || iy0d > iy1d || iz0d > iz1d) return 0;
        long long nx = (long long)((ix1d - ix0d) / step) + 1;
        long long ny = (long long)((iy1d - iy0d) / step) + 1;
        long long nz = (long long)((iz1d - iz0d) / step) + 1;
        if (nx <= 0 || ny <= 0 || nz <= 0) return 0;
        return (long double)nx * (long double)ny * (long double)nz;
    };

    long long stride = 1;
    long double baseEst = estimateForOffset(offsets[0], 1);
    if (baseEst > targetCells) {
        stride = (long long)ceill(powl(baseEst / targetCells, 1.0L / 3.0L));
        if (stride < 1) stride = 1;
    }

    auto scanOffset = [&](const Pt& off) -> vector<Placement> {
        vector<Placement> out;
        long double est = estimateForOffset(off, stride);
        if (est > 0) out.reserve((size_t)min<long double>(150000.0L, est * 2.0L + 16.0L));

        long long ix0 = (long long)ceill((bboxMin.x - off.x) / s - 1e-15L);
        long long ix1 = (long long)floorl((bboxMax.x - s - off.x) / s + 1e-15L);
        long long iy0 = (long long)ceill((bboxMin.y - off.y) / s - 1e-15L);
        long long iy1 = (long long)floorl((bboxMax.y - s - off.y) / s + 1e-15L);
        long long iz0 = (long long)ceill((bboxMin.z - off.z) / s - 1e-15L);
        long long iz1 = (long long)floorl((bboxMax.z - s - off.z) / s + 1e-15L);
        if (ix0 > ix1 || iy0 > iy1 || iz0 > iz1) return out;

        auto start = chrono::steady_clock::now();
        const long long maxPlacements = 150000;
        const auto timeLimit = chrono::milliseconds(25000);
        long long cellVisits = 0;

        for (long long ix = ix0; ix <= ix1; ix += stride) {
            long double ox = off.x + (long double)ix * s;
            for (long long iy = iy0; iy <= iy1; iy += stride) {
                long double oy = off.y + (long double)iy * s;
                for (long long iz = iz0; iz <= iz1; iz += stride) {
                    ++cellVisits;
                    if ((cellVisits & 8191LL) == 0) {
                        if (chrono::steady_clock::now() - start > timeLimit) return out;
                    }

                    long double oz = off.z + (long double)iz * s;
                    Pt center{ox + half, oy + half, oz + half};

                    if (!insideCandidate(center, hull, eps, bboxMin, bboxMax)) continue;

                    Pt t0[4] = {
                        {ox, oy, oz},
                        {ox + s, oy + s, oz},
                        {ox + s, oy, oz + s},
                        {ox, oy + s, oz + s}
                    };
                    bool ok0 = true;
                    for (const Pt& q : t0) {
                        if (!insideCandidate(q, hull, eps, bboxMin, bboxMax)) {
                            ok0 = false;
                            break;
                        }
                    }
                    if (ok0) {
                        out.push_back({center.x, center.y, center.z, 0.0L, 0.0L, 0.0L});
                        if ((long long)out.size() >= maxPlacements) return out;
                    }

                    Pt t1[4] = {
                        {ox + s, oy, oz},
                        {ox, oy + s, oz},
                        {ox, oy, oz + s},
                        {ox + s, oy + s, oz + s}
                    };
                    bool ok1 = true;
                    for (const Pt& q : t1) {
                        if (!insideCandidate(q, hull, eps, bboxMin, bboxMax)) {
                            ok1 = false;
                            break;
                        }
                    }
                    if (ok1) {
                        out.push_back({center.x, center.y, center.z, 0.0L, -halfPi, 0.0L});
                        if ((long long)out.size() >= maxPlacements) return out;
                    }
                }
            }
        }
        return out;
    };

    vector<Placement> best;
    if (offsetCount == 8) {
        for (int i = 0; i < 8; ++i) {
            vector<Placement> cur = scanOffset(offsets[i]);
            if (cur.size() > best.size()) best.swap(cur);
            if ((long long)best.size() >= 150000) break;
        }
    } else {
        best = scanOffset(offsets[0]);
    }

    cout << best.size() << '\n';
    for (const auto& p : best) {
        cout << p.cx << ' ' << p.cy << ' ' << p.cz << ' '
             << p.rx << ' ' << p.ry << ' ' << p.rz << '\n';
    }
    return 0;
}
