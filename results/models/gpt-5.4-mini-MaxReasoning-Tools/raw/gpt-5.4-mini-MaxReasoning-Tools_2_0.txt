using System;
using System.IO;

public static class Program
{
    public static void Main()
    {
        var input = Console.OpenStandardInput();
        var fs = new FastScanner(input);

        int n = (int)fs.NextLong();
        int m = (int)fs.NextLong();

        int edgeCopies = m << 1;
        int arcCount = edgeCopies << 1;

        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[arcCount];
        int[] next = new int[arcCount];
        bool[] used = new bool[edgeCopies];
        int arcPtr = 0;

        for (int i = 0; i < m; i++)
        {
            int u = (int)fs.NextLong();
            int v = (int)fs.NextLong();
            fs.NextLong(); // weight is irrelevant for validity

            int a = arcPtr;
            to[a] = v;
            next[a] = head[u];
            head[u] = a;
            a++;
            to[a] = u;
            next[a] = head[v];
            head[v] = a;
            arcPtr = a + 1;

            a = arcPtr;
            to[a] = v;
            next[a] = head[u];
            head[u] = a;
            a++;
            to[a] = u;
            next[a] = head[v];
            head[v] = a;
            arcPtr = a + 1;
        }

        int totalCopies = arcPtr >> 1;
        int[] it = new int[n];
        Array.Copy(head, it, n);

        int[] stack = new int[totalCopies + 1];
        int[] route = new int[totalCopies + 1];
        int sp = 0;
        int rp = 0;
        stack[sp++] = 0;

        // Duplicate every edge so the multigraph is Eulerian, then extract one closed walk.
        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = it[v];
            while (e != -1 && used[e >> 1]) e = next[e];
            it[v] = e;

            if (e == -1)
            {
                route[rp++] = v;
                sp--;
            }
            else
            {
                used[e >> 1] = true;
                it[v] = next[e];
                stack[sp++] = to[e];
            }
        }

        var output = new FastWriter(Console.OpenStandardOutput());
        for (int i = rp - 1; i >= 0; i--)
        {
            output.WriteInt(route[i]);
            if (i > 0) output.WriteSpace();
        }
        output.Flush();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c >= 0 && c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }

            return value * sign;
        }
    }

    private sealed class FastWriter
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;

        public FastWriter(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 20];
        }

        public void WriteSpace()
        {
            WriteByte((byte)' ');
        }

        private void WriteByte(byte b)
        {
            if (_ptr == _buffer.Length) Flush();
            _buffer[_ptr++] = b;
        }

        public void WriteInt(int value)
        {
            if (_ptr > _buffer.Length - 12) Flush();

            if (value == 0)
            {
                _buffer[_ptr++] = (byte)'0';
                return;
            }

            if (value < 0)
            {
                _buffer[_ptr++] = (byte)'-';
                value = -value;
            }

            int start = _ptr;
            while (value > 0)
            {
                _buffer[_ptr++] = (byte)('0' + (value % 10));
                value /= 10;
            }

            int i = start;
            int j = _ptr - 1;
            while (i < j)
            {
                byte t = _buffer[i];
                _buffer[i] = _buffer[j];
                _buffer[j] = t;
                i++;
                j--;
            }
        }

        public void Flush()
        {
            if (_ptr > 0)
            {
                _stream.Write(_buffer, 0, _ptr);
                _ptr = 0;
            }
        }
    }
}
