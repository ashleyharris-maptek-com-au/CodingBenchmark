using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = (int)scanner.NextLong();
        int m = (int)scanner.NextLong();

        using var writer = new FastWriter(Console.OpenStandardOutput());

        if (m == 0)
        {
            writer.WriteInt(0);
            writer.WriteByte((byte)'\n');
            return;
        }

        int[] head = new int[n];
        Array.Fill(head, -1);

        int totalCopies = m << 1;
        int totalHalfEdges = m << 2;

        int[] to = new int[totalHalfEdges];
        int[] next = new int[totalHalfEdges];
        int[] edgeId = new int[totalHalfEdges];

        int ptr = 0;
        for (int i = 0; i < m; i++)
        {
            int u = (int)scanner.NextLong();
            int v = (int)scanner.NextLong();
            scanner.NextLong(); // weight is irrelevant for feasibility.

            int id = i << 1;
            AddCopy(u, v, id, head, to, next, edgeId, ref ptr);
            AddCopy(u, v, id + 1, head, to, next, edgeId, ref ptr);
        }

        int[] cur = new int[n];
        Array.Copy(head, cur, n);

        bool[] used = new bool[totalCopies];
        int[] stack = new int[totalCopies + 1];
        int[] route = new int[totalCopies + 1];

        int sp = 0;
        int rp = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = cur[v];

            while (e != -1 && used[edgeId[e]])
                e = next[e];

            cur[v] = e;

            if (e == -1)
            {
                route[rp++] = v;
                sp--;
            }
            else
            {
                used[edgeId[e]] = true;
                cur[v] = next[e];
                stack[sp++] = to[e];
            }
        }

        for (int i = 0; i < rp; i++)
        {
            if (i != 0) writer.WriteByte((byte)' ');
            writer.WriteInt(route[i]);
        }
        writer.WriteByte((byte)'\n');
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void AddCopy(int u, int v, int id, int[] head, int[] to, int[] next, int[] edgeId, ref int ptr)
    {
        to[ptr] = v;
        next[ptr] = head[u];
        edgeId[ptr] = id;
        head[u] = ptr++;

        to[ptr] = u;
        next[ptr] = head[v];
        edgeId[ptr] = id;
        head[v] = ptr++;
    }
}

internal sealed class FastScanner
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private int _len;
    private int _ptr;

    public FastScanner(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 20];
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len <= 0) return -1;
        }

        return _buffer[_ptr++];
    }

    public long NextLong()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= 32 && c >= 0);

        long sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        long value = 0;
        while (c > 32)
        {
            value = value * 10 + (c - '0');
            c = Read();
        }

        return value * sign;
    }
}

internal sealed class FastWriter : IDisposable
{
    private readonly Stream _stream;
    private readonly byte[] _buffer;
    private readonly byte[] _digits;
    private int _ptr;

    public FastWriter(Stream stream)
    {
        _stream = stream;
        _buffer = new byte[1 << 20];
        _digits = new byte[12];
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteByte(byte value)
    {
        if (_ptr >= _buffer.Length)
        {
            Flush();
        }

        _buffer[_ptr++] = value;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    public void WriteInt(int value)
    {
        if (value == 0)
        {
            WriteByte((byte)'0');
            return;
        }

        if (value < 0)
        {
            WriteByte((byte)'-');
            value = -value;
        }

        int len = 0;
        while (value > 0)
        {
            _digits[len++] = (byte)('0' + (value % 10));
            value /= 10;
        }

        while (len > 0)
        {
            WriteByte(_digits[--len]);
        }
    }

    public void Flush()
    {
        if (_ptr > 0)
        {
            _stream.Write(_buffer, 0, _ptr);
            _ptr = 0;
        }
    }

    public void Dispose()
    {
        Flush();
    }
}
