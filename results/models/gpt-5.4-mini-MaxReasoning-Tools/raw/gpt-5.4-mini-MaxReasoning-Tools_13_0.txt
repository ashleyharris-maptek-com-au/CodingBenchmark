static inline uint64_t splitmix64(uint64_t x) {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

static inline uint64_t combine_hash(uint64_t a, uint64_t b) {
    uint64_t x = splitmix64(a ^ 0x9e3779b97f4a7c15ULL);
    uint64_t y = splitmix64(b + 0x632be59bd9b4e019ULL);
    return splitmix64(x ^ (y << 1));
}

static inline uint64_t pow_u64(uint64_t base, size_t exp) {
    uint64_t result = 1;
    while (exp) {
        if (exp & 1) result *= base;
        base *= base;
        exp >>= 1;
    }
    return result;
}

static bool contains_with_pi(const string& text, string_view pat, const vector<int>& pi) {
    if (pat.empty()) return true;
    if (pat.size() > text.size()) return false;
    int j = 0;
    for (char ch : text) {
        while (j > 0 && pat[j] != ch) j = pi[j - 1];
        if (pat[j] == ch) {
            ++j;
            if (j == static_cast<int>(pat.size())) return true;
        }
    }
    return false;
}

struct SuffixAutomaton {
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    vector<int> head;
    vector<int> link;
    vector<int> len;
    vector<int> firstPos;
    vector<Edge> edges;
    int last = 0;

    void reserve(size_t stateCap, size_t edgeCap) {
        head.reserve(stateCap);
        link.reserve(stateCap);
        len.reserve(stateCap);
        firstPos.reserve(stateCap);
        edges.reserve(edgeCap);
    }

    int new_state(int l, int pos) {
        head.push_back(-1);
        link.push_back(-1);
        len.push_back(l);
        firstPos.push_back(pos);
        return static_cast<int>(head.size()) - 1;
    }

    int find_edge(int v, unsigned char c) const {
        for (int e = head[v]; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    void replace_edge(int v, unsigned char c, int to) {
        for (int e = head[v]; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
    }

    void add_edge(int v, unsigned char c, int to) {
        edges.push_back({to, head[v], c});
        head[v] = static_cast<int>(edges.size()) - 1;
    }

    void extend(unsigned char c, int pos) {
        int cur = new_state(len[last] + 1, pos);
        int p = last;
        while (p != -1 && find_edge(p, c) == -1) {
            add_edge(p, c, cur);
            p = link[p];
        }

        if (p == -1) {
            link[cur] = 0;
            last = cur;
            return;
        }

        int q = find_edge(p, c);
        if (len[p] + 1 == len[q]) {
            link[cur] = q;
            last = cur;
            return;
        }

        int clone = new_state(len[p] + 1, firstPos[q]);
        for (int e = head[q]; e != -1; e = edges[e].next) {
            add_edge(clone, edges[e].ch, edges[e].to);
        }
        link[clone] = link[q];

        while (p != -1) {
            int t = find_edge(p, c);
            if (t != q) break;
            replace_edge(p, c, clone);
            p = link[p];
        }

        link[q] = link[cur] = clone;
        last = cur;
    }

    void build(const string& s) {
        reserve(2 * s.size() + 1, 3 * s.size() + 5);
        new_state(0, -1);  // root
        link[0] = -1;
        last = 0;
        for (int i = 0; i < static_cast<int>(s.size()); ++i) {
            extend(static_cast<unsigned char>(s[i]), i);
        }
    }
};

struct Key {
    uint64_t h1;
    uint64_t h2;

    bool operator==(const Key& other) const {
        return h1 == other.h1 && h2 == other.h2;
    }
};

struct Entry {
    Key key;
    int pos;
};

static inline uint64_t key_hash(const Key& key) {
    return combine_hash(key.h1, key.h2);
}

static bool sam_two_strings(const string& a, const string& b, string& answer) {
    if (a.empty() || b.empty()) {
        answer.clear();
        return true;
    }

    SuffixAutomaton sam;
    sam.build(a);

    int v = 0;
    int l = 0;
    int bestLen = 0;
    int bestState = 0;

    for (int i = 0; i < static_cast<int>(b.size()); ++i) {
        unsigned char c = static_cast<unsigned char>(b[i]);
        while (v != 0 && sam.find_edge(v, c) == -1) {
            v = sam.link[v];
            l = sam.len[v];
        }
        int nxt = sam.find_edge(v, c);
        if (nxt == -1) {
            v = 0;
            l = 0;
            continue;
        }
        v = nxt;
        ++l;
        if (l > bestLen) {
            bestLen = l;
            bestState = v;
        }
    }

    if (bestLen == 0) {
        answer.clear();
    } else {
        int endPos = sam.firstPos[bestState];
        answer = a.substr(static_cast<size_t>(endPos - bestLen + 1), static_cast<size_t>(bestLen));
    }
    return true;
}

static string sam_multi_strings(const vector<string>& s) {
    const string& base = s[0];
    if (base.empty()) return "";

    SuffixAutomaton sam;
    sam.build(base);
    const int stateCount = static_cast<int>(sam.head.size());

    vector<int> common(stateCount);
    for (int v = 0; v < stateCount; ++v) common[v] = sam.len[v];

    vector<int> best(stateCount, 0);
    vector<int> order(stateCount);
    vector<int> cnt(base.size() + 1, 0);
    for (int v = 0; v < stateCount; ++v) {
        ++cnt[sam.len[v]];
    }
    for (size_t i = 1; i < cnt.size(); ++i) {
        cnt[i] += cnt[i - 1];
    }
    for (int v = 0; v < stateCount; ++v) {
        order[--cnt[sam.len[v]]] = v;
    }

    for (size_t idx = 1; idx < s.size(); ++idx) {
        fill(best.begin(), best.end(), 0);
        int v = 0;
        int l = 0;
        const string& t = s[idx];

        for (char ch : t) {
            unsigned char c = static_cast<unsigned char>(ch);
            while (v != 0 && sam.find_edge(v, c) == -1) {
                v = sam.link[v];
                l = sam.len[v];
            }
            int nxt = sam.find_edge(v, c);
            if (nxt == -1) {
                v = 0;
                l = 0;
                continue;
            }
            v = nxt;
            ++l;
            if (l > best[v]) best[v] = l;
        }

        for (int i = stateCount - 1; i >= 0; --i) {
            int x = order[i];
            int p = sam.link[x];
            if (p != -1 && best[x] > 0) {
                int up = best[x];
                if (up > sam.len[p]) up = sam.len[p];
                if (up > best[p]) best[p] = up;
            }
        }

        int currentMax = 0;
        for (int vtx = 0; vtx < stateCount; ++vtx) {
            if (best[vtx] < common[vtx]) common[vtx] = best[vtx];
            if (common[vtx] > currentMax) currentMax = common[vtx];
        }

        if (currentMax == 0) return "";
    }

    int bestLen = 0;
    int bestState = 0;
    for (int v = 0; v < stateCount; ++v) {
        if (common[v] > bestLen) {
            bestLen = common[v];
            bestState = v;
        }
    }

    if (bestLen == 0) return "";
    int endPos = sam.firstPos[bestState];
    return base.substr(static_cast<size_t>(endPos - bestLen + 1), static_cast<size_t>(bestLen));
}

static bool contains_all_base(const vector<string>& s, const string& base, const vector<int>& piBase) {
    for (size_t i = 1; i < s.size(); ++i) {
        if (!contains_with_pi(s[i], base, piBase)) return false;
    }
    return true;
}

static bool verify_candidate(const vector<string>& s, string_view pat) {
    if (pat.empty()) return true;
    vector<int> pi(pat.size(), 0);
    for (size_t i = 1, j = 0; i < pat.size(); ++i) {
        while (j > 0 && pat[i] != pat[j]) j = static_cast<size_t>(pi[j - 1]);
        if (pat[i] == pat[j]) ++j;
        pi[i] = static_cast<int>(j);
    }
    for (const string& text : s) {
        if (!contains_with_pi(text, pat, pi)) return false;
    }
    return true;
}

static string hash_solve(const vector<string>& s) {
    const string& base = s[0];
    const size_t baseLen = base.size();
    if (baseLen == 0) return "";

    vector<int> piBase(baseLen, 0);
    for (size_t i = 1, j = 0; i < baseLen; ++i) {
        while (j > 0 && base[i] != base[j]) j = static_cast<size_t>(piBase[j - 1]);
        if (base[i] == base[j]) ++j;
        piBase[i] = static_cast<int>(j);
    }

    if (contains_all_base(s, base, piBase)) {
        return base;
    }

    const uint64_t B1 = 11400714819323198485ULL;
    const uint64_t B2 = 14029467366897019727ULL;

    auto check = [&](size_t L, string& out) -> bool {
        if (L == 0) {
            out.clear();
            return true;
        }
        if (L > baseLen) return false;

        uint64_t pow1 = pow_u64(B1, L - 1);
        uint64_t pow2 = pow_u64(B2, L - 1);

        vector<Entry> cur;
        cur.reserve(baseLen - L + 1);

        size_t cap = 1;
        size_t need = (baseLen >= L ? baseLen - L + 1 : 0);
        while (cap < (need * 2 + 1)) cap <<= 1;
        vector<int> table(cap, -1);

        auto insert_key = [&](const Key& key, int pos) {
            size_t slot = key_hash(key) & (cap - 1);
            while (true) {
                int idx = table[slot];
                if (idx == -1) {
                    table[slot] = static_cast<int>(cur.size());
                    cur.push_back({key, pos});
                    return;
                }
                if (cur[idx].key == key) return;
                slot = (slot + 1) & (cap - 1);
            }
        };

        uint64_t h1 = 0, h2 = 0;
        for (size_t i = 0; i < L; ++i) {
            uint64_t x = static_cast<unsigned char>(base[i]) + 1ULL;
            h1 = h1 * B1 + x;
            h2 = h2 * B2 + x;
        }
        insert_key({h1, h2}, 0);
        for (size_t i = L; i < baseLen; ++i) {
            uint64_t out = static_cast<unsigned char>(base[i - L]) + 1ULL;
            uint64_t in = static_cast<unsigned char>(base[i]) + 1ULL;
            h1 = h1 - out * pow1;
            h1 = h1 * B1 + in;
            h2 = h2 - out * pow2;
            h2 = h2 * B2 + in;
            insert_key({h1, h2}, static_cast<int>(i - L + 1));
        }

        vector<Entry> next;
        vector<unsigned char> seen(cur.size(), 0);

        for (size_t si = 1; si < s.size(); ++si) {
            const string& text = s[si];
            if (text.size() < L) {
                cur.clear();
                return false;
            }

            fill(seen.begin(), seen.end(), 0);
            vector<int> hit;
            hit.reserve(min(cur.size(), text.size()));

            auto mark = [&](const Key& key) {
                size_t slot = key_hash(key) & (cap - 1);
                while (true) {
                    int idx = table[slot];
                    if (idx == -1) return;
                    if (cur[idx].key == key) {
                        if (!seen[idx]) {
                            seen[idx] = 1;
                            hit.push_back(idx);
                        }
                        return;
                    }
                    slot = (slot + 1) & (cap - 1);
                }
            };

            uint64_t th1 = 0, th2 = 0;
            for (size_t i = 0; i < L; ++i) {
                uint64_t x = static_cast<unsigned char>(text[i]) + 1ULL;
                th1 = th1 * B1 + x;
                th2 = th2 * B2 + x;
            }
            mark({th1, th2});
            for (size_t i = L; i < text.size(); ++i) {
                uint64_t out = static_cast<unsigned char>(text[i - L]) + 1ULL;
                uint64_t in = static_cast<unsigned char>(text[i]) + 1ULL;
                th1 = th1 - out * pow1;
                th1 = th1 * B1 + in;
                th2 = th2 - out * pow2;
                th2 = th2 * B2 + in;
                mark({th1, th2});
            }

            if (hit.empty()) {
                cur.clear();
                return false;
            }

            next.clear();
            next.reserve(hit.size());
            for (size_t i = 0; i < cur.size(); ++i) {
                if (seen[i]) next.push_back(cur[i]);
            }
            cur.swap(next);

            if (cur.empty()) return false;
            if (cur.size() == 1) {
                const Entry& e = cur[0];
                string_view pat(base.data() + e.pos, L);
                if (verify_candidate(s, pat)) {
                    out.assign(pat);
                    return true;
                }
                return false;
            }

            size_t newCap = 1;
            while (newCap < (cur.size() * 2 + 1)) newCap <<= 1;
            cap = newCap;
            table.assign(cap, -1);
            for (size_t i = 0; i < cur.size(); ++i) {
                size_t slot = key_hash(cur[i].key) & (cap - 1);
                while (table[slot] != -1) slot = (slot + 1) & (cap - 1);
                table[slot] = static_cast<int>(i);
            }
            seen.assign(cur.size(), 0);
        }

        for (const Entry& e : cur) {
            string_view pat(base.data() + e.pos, L);
            if (verify_candidate(s, pat)) {
                out.assign(pat);
                return true;
            }
        }
        return false;
    };

    size_t lo = 0, hi = baseLen;
    string bestAnswer;
    while (lo < hi) {
        size_t mid = (lo + hi + 1) >> 1;
        string candidate;
        if (check(mid, candidate)) {
            lo = mid;
            bestAnswer = std::move(candidate);
        } else {
            hi = mid - 1;
        }
    }

    if (lo == 0) return "";
    if (bestAnswer.size() != lo) {
        string candidate;
        if (check(lo, candidate)) bestAnswer = std::move(candidate);
    }
    return bestAnswer;
}

static bool use_sam_solver(size_t baseLen, int n) {
    if (n == 2) return true;
    long long stateEstimate = static_cast<long long>(2 * baseLen + 1);
    long long work = stateEstimate * static_cast<long long>(max(0, n - 1));
    return baseLen <= 3000000ULL && work <= 120000000LL;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);

    vector<string> strings(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, strings[i]);
        if (!strings[i].empty() && strings[i].back() == '\r') strings[i].pop_back();
    }

    sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });

    if (strings.empty()) {
        cout << '\n';
        return 0;
    }

    const size_t baseLen = strings[0].size();
    if (baseLen == 0) {
        cout << '\n';
        return 0;
    }

    if (N == 1) {
        cout << strings[0] << '\n';
        return 0;
    }

    if (use_sam_solver(baseLen, N)) {
        string ans;
        if (N == 2) {
            sam_two_strings(strings[0], strings[1], ans);
        } else {
            ans = sam_multi_strings(strings);
        }
        cout << ans << '\n';
        return 0;
    }

    string ans = hash_solve(strings);
    cout << ans << '\n';
    return 0;
}
