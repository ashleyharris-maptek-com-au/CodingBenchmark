use std::io::{self, BufRead, Write};

#[inline]
fn parse_two_usize(line: &[u8]) -> (usize, usize) {
    let mut i = 0;
    while i < line.len() && line[i].is_ascii_whitespace() {
        i += 1;
    }

    let mut a = 0usize;
    while i < line.len() && line[i].is_ascii_digit() {
        a = a * 10 + (line[i] - b'0') as usize;
        i += 1;
    }

    while i < line.len() && line[i].is_ascii_whitespace() {
        i += 1;
    }

    let mut b = 0usize;
    while i < line.len() && line[i].is_ascii_digit() {
        b = b * 10 + (line[i] - b'0') as usize;
        i += 1;
    }

    (a, b)
}

#[inline]
fn pack_pos(row: u32, col: u32) -> u32 {
    (row << 16) | col
}

#[inline]
fn unpack_row(pos: u32) -> u32 {
    pos >> 16
}

#[inline]
fn unpack_col(pos: u32) -> u32 {
    pos & 0xFFFF
}

#[inline]
fn is_marked(bits: &[u64], idx: usize) -> bool {
    ((bits[idx >> 6] >> (idx & 63)) & 1) != 0
}

#[inline]
fn mark(bits: &mut [u64], idx: usize) {
    bits[idx >> 6] |= 1u64 << (idx & 63);
}

#[inline]
fn direction_order(row: u32, col: u32, goal_row: u32, goal_col: u32) -> [u8; 4] {
    const UP: u8 = 0;
    const DOWN: u8 = 1;
    const LEFT: u8 = 2;
    const RIGHT: u8 = 3;

    let dr = goal_row as i32 - row as i32;
    let dc = goal_col as i32 - col as i32;
    let abs_dr = dr.abs();
    let abs_dc = dc.abs();

    let vert_toward = if dr >= 0 { DOWN } else { UP };
    let vert_away = if dr >= 0 { UP } else { DOWN };
    let horiz_toward = if dc >= 0 { RIGHT } else { LEFT };
    let horiz_away = if dc >= 0 { LEFT } else { RIGHT };

    if abs_dr >= abs_dc {
        [vert_toward, horiz_toward, vert_away, horiz_away]
    } else {
        [horiz_toward, vert_toward, horiz_away, vert_away]
    }
}

fn main() {
    let mut input = io::stdin().lock();
    let mut first = Vec::new();
    if input.read_until(b'\n', &mut first).unwrap() == 0 {
        return;
    }
    if first.last() == Some(&b'\n') {
        first.pop();
    }
    if first.last() == Some(&b'\r') {
        first.pop();
    }

    let (h, w) = parse_two_usize(&first);
    let total = h.checked_mul(w).unwrap_or(0);
    let mut blocked = vec![0u64; (total + 63) >> 6];

    let mut row_buf = Vec::with_capacity(w + 2);
    let mut start_pos: Option<u32> = None;
    let mut goal_pos: Option<u32> = None;

    for r in 0..h {
        row_buf.clear();
        if input.read_until(b'\n', &mut row_buf).unwrap() == 0 {
            break;
        }
        if row_buf.last() == Some(&b'\n') {
            row_buf.pop();
        }
        if row_buf.last() == Some(&b'\r') {
            row_buf.pop();
        }

        let base = r * w;
        let row = &row_buf[..w];
        for (c, &b) in row.iter().enumerate() {
            let idx = base + c;
            match b {
                b'#' => mark(&mut blocked, idx),
                b'A' => start_pos = Some(pack_pos(r as u32, c as u32)),
                b'B' => goal_pos = Some(pack_pos(r as u32, c as u32)),
                _ => {}
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    let (start, goal) = match (start_pos, goal_pos) {
        (Some(s), Some(g)) => (s, g),
        _ => {
            let _ = writeln!(out, "0");
            return;
        }
    };

    if start == goal {
        let row = unpack_row(start);
        let col = unpack_col(start);
        let _ = writeln!(out, "1");
        let _ = writeln!(out, "{} {}", col, row);
        return;
    }

    let start_row = unpack_row(start);
    let start_col = unpack_col(start);
    let goal_row = unpack_row(goal);
    let goal_col = unpack_col(goal);

    let start_idx = start_row as usize * w + start_col as usize;
    mark(&mut blocked, start_idx);

    let mut path = Vec::<u32>::new();
    let mut next_dir = Vec::<u8>::new();
    path.push(start);
    next_dir.push(0);

    let mut found = false;
    'search: while let Some(&cur) = path.last() {
        if cur == goal {
            found = true;
            break;
        }

        let row = unpack_row(cur);
        let col = unpack_col(cur);
        let order = direction_order(row, col, goal_row, goal_col);
        let top = next_dir.len() - 1;
        let mut advanced = false;

        while next_dir[top] < 4 {
            let d = order[next_dir[top] as usize];
            next_dir[top] += 1;

            let (nr, nc) = match d {
                0 if row > 0 => (row - 1, col),
                1 if row + 1 < h as u32 => (row + 1, col),
                2 if col > 0 => (row, col - 1),
                3 if col + 1 < w as u32 => (row, col + 1),
                _ => continue,
            };

            let nidx = nr as usize * w + nc as usize;
            if !is_marked(&blocked, nidx) {
                mark(&mut blocked, nidx);
                let np = pack_pos(nr, nc);
                path.push(np);
                next_dir.push(0);
                advanced = true;
                if np == goal {
                    found = true;
                    break 'search;
                }
                break;
            }
        }

        if !advanced {
            path.pop();
            next_dir.pop();
        }
    }

    if !found {
        let _ = writeln!(out, "0");
        return;
    }

    let _ = writeln!(out, "{}", path.len());
    for &pos in &path {
        let row = unpack_row(pos);
        let col = unpack_col(pos);
        let _ = writeln!(out, "{} {}", col, row);
    }
}
