template <std::size_t I, typename T>
struct nth {
    inline static typename std::tuple_element<I, T>::type get(const T& t) { return std::get<I>(t); };
};

namespace detail {

template <typename N = uint32_t>
class Earcut {
public:
    std::vector<N> indices;
    std::size_t vertices = 0;

    template <typename Polygon>
    void operator()(const Polygon& points);

private:
    struct Node {
        Node(N index, double x_, double y_) : x(x_), y(y_), i(index), steiner(0) {}
        Node(const Node&) = delete;
        Node& operator=(const Node&) = delete;
        Node(Node&&) = delete;
        Node& operator=(Node&&) = delete;

        const double x;
        const double y;

        // previous and next vertice nodes in a polygon ring
        Node* prev = nullptr;
        Node* next = nullptr;

        // z-order curve value
        int32_t z = 0;

        // original index in polygon
        const N i : (sizeof(N) * 8 - 1);

        // indicates whether this is a steiner point
        N steiner : 1;

        // previous and next nodes in z-order
        Node* prevZ = nullptr;
        Node* nextZ = nullptr;
    };

    // Cache-optimized Triangle structure for repeated geometric tests
    struct Triangle {
        const double ax, ay;
        const double bx, by;
        const double cx, cy;

        Triangle(const Node* a, const Node* b, const Node* c)
            : ax(a->x), ay(a->y), bx(b->x), by(b->y), cx(c->x), cy(c->y) {}

        inline double area() const { return (by - ay) * (cx - bx) - (bx - ax) * (cy - by); }

        inline bool containsPoint(double px, double py) const {
            return (cx - px) * (ay - py) >= (ax - px) * (cy - py) && (ax - px) * (by - py) >= (bx - px) * (ay - py) &&
                   (bx - px) * (cy - py) >= (cx - px) * (by - py);
        }
    };

    template <typename Ring>
    Node* linkedList(const Ring& points, const bool clockwise);
    Node* filterPoints(Node* start, Node* end = nullptr);
    void earcutLinked(Node* ear, int pass = 0);
    bool isEar(Node* ear);
    bool isEarHashed(Node* ear);
    Node* cureLocalIntersections(Node* start);
    void splitEarcut(Node* start);
    template <typename Polygon>
    Node* eliminateHoles(const Polygon& points, Node* outerNode);
    Node* eliminateHole(Node* hole, Node* outerNode);
    Node* findHoleBridge(Node* hole, Node* outerNode);
    bool sectorContainsSector(const Node* m, const Node* p);
    void indexCurve(Node* start);
    Node* sortLinked(Node* list);
    int32_t zOrder(const double x_, const double y_);
    Node* getLeftmost(Node* start);
    bool pointInTriangle(double ax, double ay, double bx, double by, double cx, double cy, double px, double py) const;
    bool isValidDiagonal(Node* a, Node* b);
    double area(const Node* p, const Node* q, const Node* r) const;
    bool equals(const Node* p1, const Node* p2);
    bool intersects(const Node* p1, const Node* q1, const Node* p2, const Node* q2);
    bool onSegment(const Node* p, const Node* q, const Node* r);
    int sign(double val);
    bool intersectsPolygon(const Node* a, const Node* b);
    bool locallyInside(const Node* a, const Node* b);
    bool middleInside(const Node* a, const Node* b);
    Node* splitPolygon(Node* a, Node* b);
    template <typename Point>
    Node* insertNode(std::size_t i, const Point& p, Node* last);
    void removeNode(Node* p);

    bool hashing;
    double minX, maxX;
    double minY, maxY;
    double inv_size = 0;

    template <typename T, typename Alloc = std::allocator<T>>
    class ObjectPool {
    public:
        ObjectPool() { allocateNewBlock(256); }
        ObjectPool(std::size_t blockSize_) : baseBlockSize(blockSize_) {
            allocateNewBlock(std::max<std::size_t>(blockSize_, 256));
        }
        ~ObjectPool() { clear(); }
        template <typename... Args>
        T* construct(Args&&... args) {
            // If current block is full, move to next block or allocate new one
            if (currentIndex >= baseBlockSize) {
                currentBlockIndex++;
                if (currentBlockIndex < memoryBlocks.size()) {
                    // Reuse existing block
                    currentIndex = 0;
                } else {
                    // Allocate a new one
                    allocateNewBlock(baseBlockSize);
                }
            }

            T* object = memoryBlocks[currentBlockIndex].get() + currentIndex;
            alloc_traits::construct(alloc, object, std::forward<Args>(args)...);
            totalObjects++;
            currentIndex++;
            return object;
        }
        void reset() { clear(); }
        void clear() {
            // Destroy all objects, but keep blocks allocated for reuse
            std::size_t objectsDestroyed = 0;
            for (std::size_t blockIdx = 0; blockIdx < memoryBlocks.size() && objectsDestroyed < totalObjects;
                 ++blockIdx) {
                // check if we are in the last block
                std::size_t objectsInThisBlock = std::min(baseBlockSize, totalObjects - objectsDestroyed);
                for (std::size_t i = 0; i < objectsInThisBlock; ++i) {
                    T* object = memoryBlocks[blockIdx].get() + i;
                    alloc_traits::destroy(alloc, object);
                }
                objectsDestroyed += objectsInThisBlock;
            }
            // Reset to start from first block again
            currentBlockIndex = 0;
            currentIndex = 0;
            totalObjects = 0;
        }

    private:
        Alloc alloc;
        typedef typename std::allocator_traits<Alloc> alloc_traits;

        // Custom deleter that uses the allocator
        struct AllocDeleter {
            Alloc alloc;
            std::size_t capacity;
            void operator()(T* ptr) { alloc_traits::deallocate(alloc, ptr, capacity); }
        };

        std::vector<std::unique_ptr<T[], AllocDeleter>> memoryBlocks;
        std::vector<std::size_t> blockCapacities;
        std::size_t currentBlockIndex = 0;
        std::size_t currentIndex = 0;
        std::size_t totalObjects = 0;
        std::size_t baseBlockSize = 256;

        void allocateNewBlock(std::size_t capacity) {
            T* rawMemory = alloc_traits::allocate(alloc, capacity);
            auto newBlock = std::unique_ptr<T[], AllocDeleter>(rawMemory, AllocDeleter{alloc, capacity});
            memoryBlocks.push_back(std::move(newBlock));
            blockCapacities.push_back(capacity);
            currentBlockIndex = memoryBlocks.size() - 1;
            currentIndex = 0;
        }
    };

    std::unique_ptr<ObjectPool<Node>> nodes;
    std::vector<Node*> holeQueue;
};

template <typename N>
template <typename Polygon>
void Earcut<N>::operator()(const Polygon& points) {
    // reset
    indices.clear();
    vertices = 0;

    if (points.empty()) return;

    double x;
    double y;
    int threshold = 80;
    std::size_t len = 0;

    for (size_t i = 0; threshold >= 0 && i < points.size(); i++) {
        threshold -= static_cast<int>(points[i].size());
        len += points[i].size();
    }

    // estimate size of nodes and indices
    if (!nodes) {
        std::size_t estimatedNodes = len * 3 / 2;
        nodes = std::make_unique<ObjectPool<Node>>(std::max<std::size_t>(estimatedNodes, 256));
    }
    indices.reserve(len + points[0].size());

    Node* outerNode = linkedList(points[0], true);
    if (!outerNode || outerNode->prev == outerNode->next) return;

    if (points.size() > 1) outerNode = eliminateHoles(points, outerNode);

    // if the shape is not too simple, we'll use z-order curve hash later; calculate polygon bbox
    hashing = threshold < 0;
    if (hashing) {
        Node* p = outerNode->next;
        minX = maxX = outerNode->x;
        minY = maxY = outerNode->y;
        do {
            x = p->x;
            y = p->y;
            minX = std::min<double>(minX, x);
            minY = std::min<double>(minY, y);
            maxX = std::max<double>(maxX, x);
            maxY = std::max<double>(maxY, y);
            p = p->next;
        } while (p != outerNode);

        // minX, minY and inv_size are later used to transform coords into integers for z-order calculation
        inv_size = std::max<double>(maxX - minX, maxY - minY);
        inv_size = inv_size != .0 ? (32767. / inv_size) : .0;
    }

    earcutLinked(outerNode);

    nodes->clear();
    holeQueue.clear();
}

// create a circular doubly linked list from polygon points in the specified winding order
template <typename N>
template <typename Ring>
typename Earcut<N>::Node* Earcut<N>::linkedList(const Ring& points, const bool clockwise) {
    using Point = typename Ring::value_type;
    double sum = 0;
    const std::size_t len = points.size();
    std::size_t i, j;
    Node* last = nullptr;

    // calculate original winding order of a polygon ring
    for (i = 0, j = len > 0 ? len - 1 : 0; i < len; j = i++) {
        const auto& p1 = points[i];
        const auto& p2 = points[j];
        const double p20 = util::nth<0, Point>::get(p2);
        const double p10 = util::nth<0, Point>::get(p1);
        const double p11 = util::nth<1, Point>::get(p1);
        const double p21 = util::nth<1, Point>::get(p2);
        sum += (p20 - p10) * (p11 + p21);
    }

    // link points into circular doubly-linked list in the specified winding order
    if (clockwise == (sum > 0)) {
        for (i = 0; i < len; i++) last = insertNode(vertices + i, points[i], last);
    } else {
        for (i = len; i-- > 0;) last = insertNode(vertices + i, points[i], last);
    }

    if (last && equals(last, last->next)) {
        removeNode(last);
        last = last->next;
    }

    vertices += len;

    return last;
}

// eliminate colinear or duplicate points
template <typename N>
typename Earcut<N>::Node* Earcut<N>::filterPoints(Node* start, Node* end) {
    if (!end) end = start;

    Node* p = start;
    bool again;
    do {
        again = false;

        if (!p->steiner && (equals(p, p->next) || area(p->prev, p, p->next) == 0)) {
            removeNode(p);
            p = end = p->prev;

            if (p == p->next) break;
            again = true;

        } else {
            p = p->next;
        }
    } while (again || p != end);

    return end;
}

// main ear slicing loop which triangulates a polygon (given as a linked list)
template <typename N>
void Earcut<N>::earcutLinked(Node* ear, int pass) {
    if (!ear) return;

    // interlink polygon nodes in z-order
    if (!pass && hashing) indexCurve(ear);

    Node* stop = ear;
    Node* prev;
    Node* next;

    // iterate through ears, slicing them one by one
    while (ear->prev != ear->next) {
        prev = ear->prev;
        next = ear->next;

        if (hashing ? isEarHashed(ear) : isEar(ear)) {
            // cut off the triangle
            indices.emplace_back(prev->i);
            indices.emplace_back(ear->i);
            indices.emplace_back(next->i);

            removeNode(ear);

            // skipping the next vertice leads to less sliver triangles
            ear = next->next;
            stop = next->next;

            continue;
        }

        ear = next;

        // if we looped through the whole remaining polygon and can't find any more ears
        if (ear == stop) {
            // try filtering points and slicing again
            if (!pass) earcutLinked(filterPoints(ear), 1);

            // if this didn't work, try curing all small self-intersections locally
            else if (pass == 1) {
                ear = cureLocalIntersections(filterPoints(ear));
                earcutLinked(ear, 2);

                // as a last resort, try splitting the remaining polygon into two
            } else if (pass == 2)
                splitEarcut(ear);

            break;
        }
    }
}

// check whether a polygon node forms a valid ear with adjacent nodes
template <typename N>
bool Earcut<N>::isEar(Node* ear) {
    const Node* a = ear->prev;
    const Node* b = ear;
    const Node* c = ear->next;

    // Create triangle with cached coordinates and bounding box
    const Triangle tri(a, b, c);
    if (tri.area() >= 0) return false; // reflex, can't be an ear

    // now make sure we don't have other points inside the potential ear
    Node* p = ear->next->next;

    while (p != ear->prev) {
        if (tri.containsPoint(p->x, p->y) && area(p->prev, p, p->next) >= 0) return false;
        p = p->next;
    }

    return true;
}

template <typename N>
bool Earcut<N>::isEarHashed(Node* ear) {
    const Node* a = ear->prev;
    const Node* b = ear;
    const Node* c = ear->next;

    // Create triangle with cached coordinates and bounding box
    const Triangle tri(a, b, c);
    if (tri.area() >= 0) return false; // reflex, can't be an ear

    // triangle bbox; min & max are calculated like this for speed
    const double minTX = std::min<double>(tri.ax, std::min<double>(tri.bx, tri.cx));
    const double minTY = std::min<double>(tri.ay, std::min<double>(tri.by, tri.cy));
    const double maxTX = std::max<double>(tri.ax, std::max<double>(tri.bx, tri.cx));
    const double maxTY = std::max<double>(tri.ay, std::max<double>(tri.by, tri.cy));

    // z-order range for the current triangle bbox;
    const int32_t minZ = zOrder(minTX, minTY);
    const int32_t maxZ = zOrder(maxTX, maxTY);

    // first look for points inside the triangle in increasing z-order
    Node* p = ear->nextZ;

    while (p && p->z <= maxZ) {
        if (p != ear->prev && p != ear->next && tri.containsPoint(p->x, p->y) && area(p->prev, p, p->next) >= 0)
            return false;
        p = p->nextZ;
    }

    // then look for points in decreasing z-order
    p = ear->prevZ;

    while (p && p->z >= minZ) {
        if (p != ear->prev && p != ear->next && tri.containsPoint(p->x, p->y) && area(p->prev, p, p->next) >= 0)
            return false;
        p = p->prevZ;
    }

    return true;
}

// go through all polygon nodes and cure small local self-intersections
template <typename N>
typename Earcut<N>::Node* Earcut<N>::cureLocalIntersections(Node* start) {
    Node* p = start;
    do {
        Node* a = p->prev;
        Node* b = p->next->next;

        // a self-intersection where edge (v[i-1],v[i]) intersects (v[i+1],v[i+2])
        if (!equals(a, b) && intersects(a, p, p->next, b) && locallyInside(a, b) && locallyInside(b, a)) {
            indices.emplace_back(a->i);
            indices.emplace_back(p->i);
            indices.emplace_back(b->i);

            // remove two nodes involved
            removeNode(p);
            removeNode(p->next);

            p = start = b;
        }
        p = p->next;
    } while (p != start);

    return filterPoints(p);
}

// try splitting polygon into two and triangulate them independently
template <typename N>
void Earcut<N>::splitEarcut(Node* start) {
    // look for a valid diagonal that divides the polygon into two
    Node* a = start;
    do {
        Node* b = a->next->next;
        while (b != a->prev) {
            if (a->i != b->i && isValidDiagonal(a, b)) {
                // split the polygon in two by the diagonal
                Node* c = splitPolygon(a, b);

                // filter colinear points around the cuts
                a = filterPoints(a, a->next);
                c = filterPoints(c, c->next);

                // run earcut on each half
                earcutLinked(a);
                earcutLinked(c);
                return;
            }
            b = b->next;
        }
        a = a->next;
    } while (a != start);
}

// link every hole into the outer loop, producing a single-ring polygon without holes
template <typename N>
template <typename Polygon>
typename Earcut<N>::Node* Earcut<N>::eliminateHoles(const Polygon& points, Node* outerNode) {
    const size_t len = points.size();

    holeQueue.clear();
    for (size_t i = 1; i < len; i++) {
        Node* list = linkedList(points[i], false);
        if (list) {
            if (list == list->next) list->steiner = true;
            holeQueue.push_back(getLeftmost(list));
        }
    }
    std::sort(holeQueue.begin(), holeQueue.end(), [](const Node* a, const Node* b) { return a->x < b->x; });

    // process holes from left to right
    for (size_t i = 0; i < holeQueue.size(); i++) {
        outerNode = eliminateHole(holeQueue[i], outerNode);
    }

    return outerNode;
}

// find a bridge between vertices that connects hole with an outer ring and and link it
template <typename N>
typename Earcut<N>::Node* Earcut<N>::eliminateHole(Node* hole, Node* outerNode) {
    Node* bridge = findHoleBridge(hole, outerNode);
    if (!bridge) {
        return outerNode;
    }

    Node* bridgeReverse = splitPolygon(bridge, hole);

    // filter collinear points around the cuts
    filterPoints(bridgeReverse, bridgeReverse->next);

    // Check if input node was removed by the filtering
    return filterPoints(bridge, bridge->next);
}

// David Eberly's algorithm for finding a bridge between hole and outer polygon
template <typename N>
typename Earcut<N>::Node* Earcut<N>::findHoleBridge(Node* hole, Node* outerNode) {
    Node* p = outerNode;
    double hx = hole->x;
    double hy = hole->y;
    double qx = -std::numeric_limits<double>::max();
    Node* m = nullptr;

    // find a segment intersected by a ray from the hole's leftmost Vertex to the left;
    // segment's endpoint with lesser x will be potential connection Vertex
    do {
        if (hy <= p->y && hy >= p->next->y && p->next->y != p->y) {
            double x = p->x + (hy - p->y) * (p->next->x - p->x) / (p->next->y - p->y);
            if (x <= hx && x > qx) {
                qx = x;
                m = p->x < p->next->x ? p : p->next;
                if (x == hx) return m; // hole touches outer segment; pick leftmost endpoint
            }
        }
        p = p->next;
    } while (p != outerNode);

    if (!m) return 0;

    // look for points inside the triangle of hole Vertex, segment intersection and endpoint;
    // if there are no points found, we have a valid connection;
    // otherwise choose the Vertex of the minimum angle with the ray as connection Vertex

    const Node* stop = m;
    double tanMin = std::numeric_limits<double>::max();
    double tanCur = 0;

    p = m;
    double mx = m->x;
    double my = m->y;

    do {
        if (hx >= p->x && p->x >= mx && hx != p->x &&
            pointInTriangle(hy < my ? hx : qx, hy, mx, my, hy < my ? qx : hx, hy, p->x, p->y)) {
            tanCur = std::abs(hy - p->y) / (hx - p->x); // tangential

            if (locallyInside(p, hole) &&
                (tanCur < tanMin || (tanCur == tanMin && (p->x > m->x || sectorContainsSector(m, p))))) {
                m = p;
                tanMin = tanCur;
            }
        }

        p = p->next;
    } while (p != stop);

    return m;
}

// whether sector in vertex m contains sector in vertex p in the same coordinates
template <typename N>
bool Earcut<N>::sectorContainsSector(const Node* m, const Node* p) {
    return area(m->prev, m, p->prev) < 0 && area(p->next, m, m->next) < 0;
}

// interlink polygon nodes in z-order
template <typename N>
void Earcut<N>::indexCurve(Node* start) {
    assert(start);
    Node* p = start;

    do {
        p->z = p->z ? p->z : zOrder(p->x, p->y);
        p->prevZ = p->prev;
        p->nextZ = p->next;
        p = p->next;
    } while (p != start);

    p->prevZ->nextZ = nullptr;
    p->prevZ = nullptr;

    sortLinked(p);
}

// Simon Tatham's linked list merge sort algorithm
// http://www.chiark.greenend.org.uk/~sgtatham/algorithms/listsort.html
template <typename N>
typename Earcut<N>::Node* Earcut<N>::sortLinked(Node* list) {
    assert(list);
    Node* p;
    Node* q;
    Node* e;
    Node* tail;
    int i, numMerges, pSize, qSize;
    int inSize = 1;

    for (;;) {
        p = list;
        list = nullptr;
        tail = nullptr;
        numMerges = 0;

        while (p) {
            numMerges++;
            q = p;
            pSize = 0;
            for (i = 0; i < inSize; i++) {
                pSize++;
                q = q->nextZ;
                if (!q) break;
            }

            qSize = inSize;

            while (pSize > 0 || (qSize > 0 && q)) {
                if (pSize == 0) {
                    e = q;
                    q = q->nextZ;
                    qSize--;
                } else if (qSize == 0 || !q) {
                    e = p;
                    p = p->nextZ;
                    pSize--;
                } else if (p->z <= q->z) {
                    e = p;
                    p = p->nextZ;
                    pSize--;
                } else {
                    e = q;
                    q = q->nextZ;
                    qSize--;
                }

                if (tail)
                    tail->nextZ = e;
                else
                    list = e;

                e->prevZ = tail;
                tail = e;
            }

            p = q;
        }

        tail->nextZ = nullptr;

        if (numMerges <= 1) return list;

        inSize *= 2;
    }
}

// z-order of a Vertex given coords and size of the data bounding box
template <typename N>
int32_t Earcut<N>::zOrder(const double x_, const double y_) {
    // coords are transformed into non-negative 15-bit integer range
    int32_t x = static_cast<int32_t>((x_ - minX) * inv_size);
    int32_t y = static_cast<int32_t>((y_ - minY) * inv_size);

    x = (x | (x << 8)) & 0x00FF00FF;
    x = (x | (x << 4)) & 0x0F0F0F0F;
    x = (x | (x << 2)) & 0x33333333;
    x = (x | (x << 1)) & 0x55555555;

    y = (y | (y << 8)) & 0x00FF00FF;
    y = (y | (y << 4)) & 0x0F0F0F0F;
    y = (y | (y << 2)) & 0x33333333;
    y = (y | (y << 1)) & 0x55555555;

    return x | (y << 1);
}

// find the leftmost node of a polygon ring
template <typename N>
typename Earcut<N>::Node* Earcut<N>::getLeftmost(Node* start) {
    Node* p = start;
    Node* leftmost = start;
    do {
        if (p->x < leftmost->x || (p->x == leftmost->x && p->y < leftmost->y)) leftmost = p;
        p = p->next;
    } while (p != start);

    return leftmost;
}

// check if a point lies within a convex triangle
template <typename N>
bool Earcut<N>::pointInTriangle(
    double ax, double ay, double bx, double by, double cx, double cy, double px, double py) const {
    return (cx - px) * (ay - py) >= (ax - px) * (cy - py) && (ax - px) * (by - py) >= (bx - px) * (ay - py) &&
           (bx - px) * (cy - py) >= (cx - px) * (by - py);
}

// check if a diagonal between two polygon nodes is valid (lies in polygon interior)
template <typename N>
bool Earcut<N>::isValidDiagonal(Node* a, Node* b) {
    return a->next->i != b->i && a->prev->i != b->i && !intersectsPolygon(a, b) && // dones't intersect other edges
           ((locallyInside(a, b) && locallyInside(b, a) && middleInside(a, b) &&   // locally visible
             (area(a->prev, a, b->prev) != 0.0 ||
              area(a, b->prev, b) != 0.0)) || // does not create opposite-facing sectors
            (equals(a, b) && area(a->prev, a, a->next) > 0 &&
             area(b->prev, b, b->next) > 0)); // special zero-length case
}

// signed area of a triangle
template <typename N>
double Earcut<N>::area(const Node* p, const Node* q, const Node* r) const {
    return (q->y - p->y) * (r->x - q->x) - (q->x - p->x) * (r->y - q->y);
}

// check if two points are equal
template <typename N>
bool Earcut<N>::equals(const Node* p1, const Node* p2) {
    return p1->x == p2->x && p1->y == p2->y;
}

// check if two segments intersect
template <typename N>
bool Earcut<N>::intersects(const Node* p1, const Node* q1, const Node* p2, const Node* q2) {
    int o1 = sign(area(p1, q1, p2));
    int o2 = sign(area(p1, q1, q2));
    int o3 = sign(area(p2, q2, p1));
    int o4 = sign(area(p2, q2, q1));

    if (o1 != o2 && o3 != o4) return true; // general case

    if (o1 == 0 && onSegment(p1, p2, q1)) return true; // p1, q1 and p2 are collinear and p2 lies on p1q1
    if (o2 == 0 && onSegment(p1, q2, q1)) return true; // p1, q1 and q2 are collinear and q2 lies on p1q1
    if (o3 == 0 && onSegment(p2, p1, q2)) return true; // p2, q2 and p1 are collinear and p1 lies on p2q2
    if (o4 == 0 && onSegment(p2, q1, q2)) return true; // p2, q2 and q1 are collinear and q1 lies on p2q2

    return false;
}

// for collinear points p, q, r, check if point q lies on segment pr
template <typename N>
bool Earcut<N>::onSegment(const Node* p, const Node* q, const Node* r) {
    return q->x <= std::max<double>(p->x, r->x) && q->x >= std::min<double>(p->x, r->x) &&
           q->y <= std::max<double>(p->y, r->y) && q->y >= std::min<double>(p->y, r->y);
}

template <typename N>
int Earcut<N>::sign(double val) {
    return (0.0 < val) - (val < 0.0);
}

// check if a polygon diagonal intersects any polygon segments
template <typename N>
bool Earcut<N>::intersectsPolygon(const Node* a, const Node* b) {
    const Node* p = a;
    do {
        if (p->i != a->i && p->next->i != a->i && p->i != b->i && p->next->i != b->i && intersects(p, p->next, a, b))
            return true;
        p = p->next;
    } while (p != a);

    return false;
}

// check if a polygon diagonal is locally inside the polygon
template <typename N>
bool Earcut<N>::locallyInside(const Node* a, const Node* b) {
    return area(a->prev, a, a->next) < 0 ? area(a, b, a->next) >= 0 && area(a, a->prev, b) >= 0
                                         : area(a, b, a->prev) < 0 || area(a, a->next, b) < 0;
}

// check if the middle Vertex of a polygon diagonal is inside the polygon
template <typename N>
bool Earcut<N>::middleInside(const Node* a, const Node* b) {
    const Node* p = a;
    bool inside = false;
    double px = (a->x + b->x) / 2;
    double py = (a->y + b->y) / 2;
    do {
        if (((p->y > py) != (p->next->y > py)) && p->next->y != p->y &&
            (px < (p->next->x - p->x) * (py - p->y) / (p->next->y - p->y) + p->x))
            inside = !inside;
        p = p->next;
    } while (p != a);

    return inside;
}

// link two polygon vertices with a bridge; if the vertices belong to the same ring, it splits
// polygon into two; if one belongs to the outer ring and another to a hole, it merges it into a
// single ring
template <typename N>
typename Earcut<N>::Node* Earcut<N>::splitPolygon(Node* a, Node* b) {
    Node* a2 = nodes->construct(a->i, a->x, a->y);
    Node* b2 = nodes->construct(b->i, b->x, b->y);
    Node* an = a->next;
    Node* bp = b->prev;

    a->next = b;
    b->prev = a;

    a2->next = an;
    an->prev = a2;

    b2->next = a2;
    a2->prev = b2;

    bp->next = b2;
    b2->prev = bp;

    return b2;
}

// create a node and util::optionally link it with previous one (in a circular doubly linked list)
template <typename N>
template <typename Point>
typename Earcut<N>::Node* Earcut<N>::insertNode(std::size_t i, const Point& pt, Node* last) {
    Node* p = nodes->construct(static_cast<N>(i), util::nth<0, Point>::get(pt), util::nth<1, Point>::get(pt));

    if (!last) {
        p->prev = p;
        p->next = p;

    } else {
        assert(last);
        p->next = last->next;
        p->prev = last;
        last->next->prev = p;
        last->next = p;
    }
    return p;
}

template <typename N>
void Earcut<N>::removeNode(Node* p) {
    p->next->prev = p->prev;
    p->prev->next = p->next;

    if (p->prevZ) p->prevZ->nextZ = p->nextZ;
    if (p->nextZ) p->nextZ->prevZ = p->prevZ;
}

}

struct Vec2 {

struct Vec3 {
    double x = 0.0;
    double y = 0.0;
    double z = 0.0;

static inline Vec2 operator+(const Vec2& a, const Vec2& b) { return {a.x + b.x, a.y + b.y}; }
static inline Vec2 operator-(const Vec2& a, const Vec2& b) { return {a.x - b.x, a.y - b.y}; }
static inline Vec2 operator*(const Vec2& a, double s) { return {a.x * s, a.y * s}; }

static inline Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static inline Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static inline Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static inline Vec3 operator/(const Vec3& a, double s) { return {a.x / s, a.y / s, a.z / s}; }

static inline Vec2& operator+=(Vec2& a, const Vec2& b) {
    a.x += b.x;
    a.y += b.y;
    return a;
}

static inline Vec3& operator+=(Vec3& a, const Vec3& b) {
    a.x += b.x;
    a.y += b.y;
    a.z += b.z;
    return a;
}

static inline double dot(const Vec2& a, const Vec2& b) { return a.x * b.x + a.y * b.y; }
static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }

static inline double cross2(const Vec2& a, const Vec2& b) { return a.x * b.y - a.y * b.x; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

static inline double norm2(const Vec2& a) { return dot(a, a); }
static inline double norm2(const Vec3& a) { return dot(a, a); }
static inline double norm(const Vec3& a) { return std::sqrt(norm2(a)); }

static inline Vec3 normalize(const Vec3& a) {
    double n = norm(a);
    if (n == 0.0) return a;
    return a / n;
}

static inline double orient2d(const Vec2& a, const Vec2& b, const Vec2& c) {
    return cross2(b - a, c - a);
}

struct BBox3 {
    Vec3 mn;
    Vec3 mx;
};

static inline BBox3 makeEmptyBox() {
    double inf = std::numeric_limits<double>::infinity();
    return {{inf, inf, inf}, {-inf, -inf, -inf}};
}

static inline void expand(BBox3& b, const Vec3& p) {
    b.mn.x = std::min(b.mn.x, p.x);
    b.mn.y = std::min(b.mn.y, p.y);
    b.mn.z = std::min(b.mn.z, p.z);
    b.mx.x = std::max(b.mx.x, p.x);
    b.mx.y = std::max(b.mx.y, p.y);
    b.mx.z = std::max(b.mx.z, p.z);
}

static inline BBox3 unite(const BBox3& a, const BBox3& b) {
    BBox3 r = a;
    expand(r, b.mn);
    expand(r, b.mx);
    return r;
}

static inline bool overlap(const BBox3& a, const BBox3& b) {
    if (a.mx.x < b.mn.x - kGeomEps || b.mx.x < a.mn.x - kGeomEps) return false;
    if (a.mx.y < b.mn.y - kGeomEps || b.mx.y < a.mn.y - kGeomEps) return false;
    if (a.mx.z < b.mn.z - kGeomEps || b.mx.z < a.mn.z - kGeomEps) return false;
    return true;
}

static inline double volume(const BBox3& b) {
    return std::max(0.0, b.mx.x - b.mn.x) * std::max(0.0, b.mx.y - b.mn.y) *
           std::max(0.0, b.mx.z - b.mn.z);
}

struct TriData {
    int idx[3]{};
    Vec3 p[3];
    Vec2 q[3];
    Vec3 origin;
    Vec3 u;
    Vec3 v;
    Vec3 nUnit;
    double planeD = 0.0;
    double scale = 0.0;
    double area2 = 0.0;
    Vec3 centroid;
    BBox3 box;

    Vec2 project(const Vec3& pnt) const {
        Vec3 d = pnt - origin;
        return {dot(d, u), dot(d, v)};
    }

    Vec3 lift(const Vec2& pt) const {
        return origin + u * pt.x + v * pt.y;
    }
};

static inline double signedDistanceToPlane(const TriData& t, const Vec3& p) {
    return dot(t.nUnit, p) - t.planeD;
}

static inline bool pointInTri2D(const TriData& t, const Vec2& p, double eps) {
    double a = orient2d(t.q[0], t.q[1], p);
    double b = orient2d(t.q[1], t.q[2], p);
    double c = orient2d(t.q[2], t.q[0], p);
    return a >= -eps && b >= -eps && c >= -eps;
}

static inline bool pointOnEdge2D(const TriData& t, const Vec2& p, int e, double eps, double* outT = nullptr) {
    const Vec2& a = t.q[e];
    const Vec2& b = t.q[(e + 1) % 3];
    Vec2 ab = b - a;
    double len2 = norm2(ab);
    if (len2 == 0.0) return false;
    double o = orient2d(a, b, p);
    if (std::fabs(o) > eps) return false;
    double tt = dot(p - a, ab) / len2;
    if (tt < -eps || tt > 1.0 + eps) return false;
    if (outT) *outT = tt;
    return true;
}

static inline bool segOnBoundary(const TriData& t, const Vec2& a, const Vec2& b, double eps) {
    for (int e = 0; e < 3; ++e) {
        double ta = 0.0, tb = 0.0;
        if (pointOnEdge2D(t, a, e, eps, &ta) && pointOnEdge2D(t, b, e, eps, &tb)) return true;
    }
    return false;
}

static inline bool pointInPolygon(const vector<Vec2>& poly, const Vec2& p, double eps = 1e-12) {
    if (poly.size() < 3) return false;
    bool inside = false;
    for (size_t i = 0, j = poly.size() - 1; i < poly.size(); j = i++) {
        const Vec2& a = poly[i];
        const Vec2& b = poly[j];
        Vec2 ab = b - a;
        Vec2 ap = p - a;
        double o = cross2(ab, ap);
        double minx = std::min(a.x, b.x) - eps;
        double maxx = std::max(a.x, b.x) + eps;
        double miny = std::min(a.y, b.y) - eps;
        double maxy = std::max(a.y, b.y) + eps;
        if (std::fabs(o) <= eps && p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy) return true;
        bool intersect = ((a.y > p.y) != (b.y > p.y)) &&
                         (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0) + a.x);
        if (intersect) inside = !inside;
    }
    return inside;
}

struct Cell2 {
    long long x = 0;
    long long y = 0;
    bool operator==(const Cell2& o) const { return x == o.x && y == o.y; }
};

struct Cell3 {
    long long x = 0;
    long long y = 0;
    long long z = 0;
    bool operator==(const Cell3& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct Cell2Hash {
    size_t operator()(const Cell2& c) const noexcept {
        size_t h1 = std::hash<long long>{}(c.x);
        size_t h2 = std::hash<long long>{}(c.y);
        return h1 ^ (h2 + 0x9e3779b97f4a7c15ULL + (h1 << 6) + (h1 >> 2));
    }
};

struct Cell3Hash {
    size_t operator()(const Cell3& c) const noexcept {
        size_t h1 = std::hash<long long>{}(c.x);
        size_t h2 = std::hash<long long>{}(c.y);
        size_t h3 = std::hash<long long>{}(c.z);
        size_t h = h1 ^ (h2 + 0x9e3779b97f4a7c15ULL + (h1 << 6) + (h1 >> 2));
        return h ^ (h3 + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2));
    }
};

struct LocalPointMap {
    double cell = 1e-9;
    double tol2 = 1e-18;
    vector<Vec2> pts2;
    vector<Vec3> pts3;
    std::unordered_map<Cell2, vector<int>, Cell2Hash> buckets;

    explicit LocalPointMap(double c, double tol) : cell(c), tol2(tol * tol) {}

    int add(const Vec2& p2, const Vec3& p3) {
        long long cx = (long long)std::floor(p2.x / cell);
        long long cy = (long long)std::floor(p2.y / cell);
        for (long long dx = -1; dx <= 1; ++dx) {
            for (long long dy = -1; dy <= 1; ++dy) {
                Cell2 key{cx + dx, cy + dy};
                auto it = buckets.find(key);
                if (it == buckets.end()) continue;
                for (int id : it->second) {
                    Vec2 d = pts2[id] - p2;
                    if (norm2(d) <= tol2) return id;
                }
            }
        }
        int id = (int)pts2.size();
        pts2.push_back(p2);
        pts3.push_back(p3);
        buckets[Cell2{cx, cy}].push_back(id);
        return id;
    }
};

struct GlobalVertexMap {
    double cell = 1e-8;
    double tol2 = 1e-16;
    vector<Vec3> pts;
    std::unordered_map<Cell3, vector<int>, Cell3Hash> buckets;

    explicit GlobalVertexMap(double c, double tol) : cell(c), tol2(tol * tol) {}

    int add(const Vec3& p) {
        long long cx = (long long)std::floor(p.x / cell);
        long long cy = (long long)std::floor(p.y / cell);
        long long cz = (long long)std::floor(p.z / cell);
        for (long long dx = -1; dx <= 1; ++dx) {
            for (long long dy = -1; dy <= 1; ++dy) {
                for (long long dz = -1; dz <= 1; ++dz) {
                    Cell3 key{cx + dx, cy + dy, cz + dz};
                    auto it = buckets.find(key);
                    if (it == buckets.end()) continue;
                    for (int id : it->second) {
                        Vec3 d = pts[id] - p;
                        if (norm2(d) <= tol2) return id;
                    }
                }
            }
        }
        int id = (int)pts.size();
        pts.push_back(p);
        buckets[Cell3{cx, cy, cz}].push_back(id);
        return id;
    }
};

static inline bool rayBoxHit(const Vec3& o, const Vec3& d, const BBox3& b) {
    double tmin = 0.0;
    double tmax = std::numeric_limits<double>::infinity();
    auto step = [&](double o1, double d1, double mn, double mx) -> bool {
        if (std::fabs(d1) < kRayEps) {
            return o1 >= mn - kRayEps && o1 <= mx + kRayEps;
        }
        double inv = 1.0 / d1;
        double t1 = (mn - o1) * inv;
        double t2 = (mx - o1) * inv;
        if (t1 > t2) std::swap(t1, t2);
        tmin = std::max(tmin, t1);
        tmax = std::min(tmax, t2);
        return tmin <= tmax;
    };
    if (!step(o.x, d.x, b.mn.x, b.mx.x)) return false;
    if (!step(o.y, d.y, b.mn.y, b.mx.y)) return false;
    if (!step(o.z, d.z, b.mn.z, b.mx.z)) return false;
    return tmax >= 0.0;
}

static inline bool rayTriangleHit(const Vec3& o, const Vec3& d, const TriData& t, double& thit) {
    Vec3 e1 = t.p[1] - t.p[0];
    Vec3 e2 = t.p[2] - t.p[0];
    Vec3 pvec = cross(d, e2);
    double det = dot(e1, pvec);
    if (std::fabs(det) < kRayEps) return false;
    double invDet = 1.0 / det;
    Vec3 tvec = o - t.p[0];
    double u = dot(tvec, pvec) * invDet;
    if (u <= kRayEps || u >= 1.0 - kRayEps) return false;
    Vec3 qvec = cross(tvec, e1);
    double v = dot(d, qvec) * invDet;
    if (v <= kRayEps || u + v >= 1.0 - kRayEps) return false;
    double tt = dot(e2, qvec) * invDet;
    if (tt <= kRayEps) return false;
    thit = tt;
    return true;
}

struct MeshAccel {
    vector<TriData> tris;
    vector<int> order;
    vector<BVHNode> nodes;
    int root = -1;

    void build(const vector<TriData>& t) {
        tris = t;
        order.resize(tris.size());
        std::iota(order.begin(), order.end(), 0);
        nodes.clear();
        if (!tris.empty()) root = buildRec(0, (int)tris.size());
    }

    int buildRec(int l, int r) {
        int id = (int)nodes.size();
        nodes.push_back({});
        BBox3 box = makeEmptyBox();
        for (int i = l; i < r; ++i) {
            const auto& tb = tris[order[i]].box;
            box = unite(box, tb);
        }
        nodes[id].box = box;
        nodes[id].start = l;
        nodes[id].count = r - l;
        if (r - l <= 8) {
            nodes[id].leaf = true;
            return id;
        }

        BBox3 cb = makeEmptyBox();
        for (int i = l; i < r; ++i) expand(cb, tris[order[i]].centroid);
        Vec3 ext = cb.mx - cb.mn;
        int axis = 0;
        if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z >= ext.y) axis = 2;

        int m = (l + r) / 2;
        std::nth_element(order.begin() + l, order.begin() + m, order.begin() + r, [&](int a, int b) {
            const Vec3& ca = tris[a].centroid;
            const Vec3& cb = tris[b].centroid;
            if (axis == 0) return ca.x < cb.x;
            if (axis == 1) return ca.y < cb.y;
            return ca.z < cb.z;
        });
        if (m == l || m == r) {
            nodes[id].leaf = true;
            return id;
        }
        nodes[id].left = buildRec(l, m);
        nodes[id].right = buildRec(m, r);
        nodes[id].box = unite(nodes[nodes[id].left].box, nodes[nodes[id].right].box);
        return id;
    }

    bool pointInside(const Vec3& p, const Vec3& rayDir) const {
        if (root < 0) return false;
        if (p.x < nodes[root].box.mn.x - 1e-8 || p.x > nodes[root].box.mx.x + 1e-8 ||
            p.y < nodes[root].box.mn.y - 1e-8 || p.y > nodes[root].box.mx.y + 1e-8 ||
            p.z < nodes[root].box.mn.z - 1e-8 || p.z > nodes[root].box.mx.z + 1e-8) {
            return false;
        }

        int hits = 0;
        vector<int> st;
        st.push_back(root);
        while (!st.empty()) {
            int nid = st.back();
            st.pop_back();
            const BVHNode& node = nodes[nid];
            if (!rayBoxHit(p, rayDir, node.box)) continue;
            if (node.leaf) {
                for (int i = node.start; i < node.start + node.count; ++i) {
                    const TriData& t = tris[order[i]];
                    if (!rayBoxHit(p, rayDir, t.box)) continue;
                    double thit = 0.0;
                    if (rayTriangleHit(p, rayDir, t, thit)) ++hits;
                }
            } else {
                st.push_back(node.left);
                st.push_back(node.right);
            }
        }
        return (hits & 1) != 0;
    }
};

static inline double polygonArea2D(const vector<Vec2>& poly) {
    double a = 0.0;
    for (size_t i = 0, j = poly.size() - 1; i < poly.size(); j = i++) a += cross2(poly[j], poly[i]);
    return 0.5 * a;
}

static inline BBox3 boxFromPoints3D(const vector<Vec3>& pts) {
    BBox3 b = makeEmptyBox();
    for (const auto& p : pts) expand(b, p);
    return b;
}

static inline bool box2ContainsPoint(const vector<Vec2>& poly, const Vec2& p, double eps) {
    double minx = std::numeric_limits<double>::infinity();
    double miny = std::numeric_limits<double>::infinity();
    double maxx = -std::numeric_limits<double>::infinity();
    double maxy = -std::numeric_limits<double>::infinity();
    for (const auto& q : poly) {
        minx = std::min(minx, q.x);
        miny = std::min(miny, q.y);
        maxx = std::max(maxx, q.x);
        maxy = std::max(maxy, q.y);
    }
    return p.x >= minx - eps && p.x <= maxx + eps && p.y >= miny - eps && p.y <= maxy + eps;
}

static bool triangulateRegion(
    const TriData& tri,
    const Cycle& outer,
    const vector<int>& childIds,
    const vector<Cycle>& cycles,
    const vector<Vec2>& local2,
    const vector<Vec3>& local3,
    const MeshAccel& other,
    const Vec3& rayDir,
    double sampleEps,
    RegionTriangulation& out
) {
    vector<vector<Point2>> polygon;
    vector<int> flatIds;
    polygon.emplace_back();
    polygon.back().reserve(outer.verts.size());
    for (int id : outer.verts) {
        polygon.back().push_back({local2[id].x, local2[id].y});
        flatIds.push_back(id);
    }
    for (int cid : childIds) {
        polygon.emplace_back();
        polygon.back().reserve(cycles[cid].verts.size());
        for (int id : cycles[cid].verts) {
            polygon.back().push_back({local2[id].x, local2[id].y});
            flatIds.push_back(id);
        }
    }

    std::vector<uint32_t> idx = mapbox::earcut<uint32_t>(polygon);
    if (idx.size() < 3) return false;

    vector<array<int, 3>> tris;
    tris.reserve(idx.size() / 3);
    for (size_t i = 0; i + 2 < idx.size(); i += 3) {
        int a = flatIds[idx[i + 0]];
        int b = flatIds[idx[i + 1]];
        int c = flatIds[idx[i + 2]];
        Vec2 pa = local2[a], pb = local2[b], pc = local2[c];
        double ar = orient2d(pa, pb, pc);
        if (std::fabs(ar) < 1e-14) continue;
        if (ar < 0) std::swap(b, c);
        tris.push_back({a, b, c});
    }
    if (tris.empty()) return false;

    Vec3 sample = {};
    bool found = false;
    for (const auto& t : tris) {
        Vec2 c2{(local2[t[0]].x + local2[t[1]].x + local2[t[2]].x) / 3.0, (local2[t[0]].y + local2[t[1]].y + local2[t[2]].y) / 3.0};
        bool insideChild = false;
        for (int cid : childIds) {
            vector<Vec2> ring;
            ring.reserve(cycles[cid].verts.size());
            for (int id : cycles[cid].verts) ring.push_back(local2[id]);
            if (pointInPolygon(ring, c2, 1e-12)) {
                insideChild = true;
                break;
            }
        }
        if (insideChild) continue;
        sample = (local3[t[0]] + local3[t[1]] + local3[t[2]]) / 3.0 + tri.nUnit * sampleEps;
        found = true;
        break;
    }
    if (!found) {
        const auto& t = tris[0];
        sample = (local3[t[0]] + local3[t[1]] + local3[t[2]]) / 3.0 + tri.nUnit * sampleEps;
    }

    out.tris = std::move(tris);
    out.sample = sample;
    out.valid = true;
    return true;
}

static inline bool candidateExists(const vector<Vec3>& cand, const Vec3& p, double eps2) {
    for (const auto& q : cand) {
        Vec3 d = q - p;
        if (norm2(d) <= eps2) return true;
    }
    return false;
}

static inline void addCandidate(vector<Vec3>& cand, const Vec3& p, double eps2) {
    if (!candidateExists(cand, p, eps2)) cand.push_back(p);
}

static void intersectCoplanar(
    const TriData& A,
    const TriData& B,
    vector<RawSeg>& outA,
    vector<RawSeg>& outB,
    double planeEps
) {
    vector<Vec2> poly = {A.q[0], A.q[1], A.q[2]};
    Vec2 bq[3] = {A.project(B.p[0]), A.project(B.p[1]), A.project(B.p[2])};
    double sign = polygonArea2D(vector<Vec2>{bq[0], bq[1], bq[2]}) >= 0.0 ? 1.0 : -1.0;

    auto clip = [&](const Vec2& s, const Vec2& e) {
        if (poly.empty()) return;
        vector<Vec2> res;
        Vec2 prev = poly.back();
        bool prevIn = orient2d(s, e, prev) * sign >= -planeEps;
        for (const Vec2& cur : poly) {
            bool curIn = orient2d(s, e, cur) * sign >= -planeEps;
            if (curIn != prevIn) {
                Vec2 d1 = cur - prev;
                Vec2 d2 = e - s;
                double denom = cross2(d1, d2);
                double t = 0.0;
                if (std::fabs(denom) > 1e-18) {
                    t = cross2(s - prev, d2) / denom;
                }
                res.push_back(prev + d1 * t);
            }
            if (curIn) res.push_back(cur);
            prev = cur;
            prevIn = curIn;
        }
        poly = std::move(res);
    };

    clip(bq[0], bq[1]);
    clip(bq[1], bq[2]);
    clip(bq[2], bq[0]);
    if (poly.size() < 3) return;
    double area = polygonArea2D(poly);
    if (std::fabs(area) < 1e-14) return;

    double segEps = planeEps * 10.0;
    for (size_t i = 0; i < poly.size(); ++i) {
        Vec2 a2 = poly[i];
        Vec2 b2 = poly[(i + 1) % poly.size()];
        Vec3 a3 = A.lift(a2);
        Vec3 b3 = A.lift(b2);

        if (!segOnBoundary(A, a2, b2, segEps)) {
            outA.push_back({a3, b3});
        }
        Vec2 paB = B.project(a3);
        Vec2 pbB = B.project(b3);
        if (!segOnBoundary(B, paB, pbB, segEps)) {
            outB.push_back({a3, b3});
        }
    }
}

static void intersectTrianglePair(
    const TriData& A,
    const TriData& B,
    vector<RawSeg>& outA,
    vector<RawSeg>& outB,
    double planeEps
) {
    double dA[3];
    double dB[3];
    for (int i = 0; i < 3; ++i) dA[i] = signedDistanceToPlane(B, A.p[i]);
    for (int i = 0; i < 3; ++i) dB[i] = signedDistanceToPlane(A, B.p[i]);

    Vec3 nCross = cross(A.nUnit, B.nUnit);
    double nCross2 = norm2(nCross);

    double maxA = std::max({std::fabs(dA[0]), std::fabs(dA[1]), std::fabs(dA[2])});
    double maxB = std::max({std::fabs(dB[0]), std::fabs(dB[1]), std::fabs(dB[2])});

    if (nCross2 <= 1e-16) {
        if (maxA <= planeEps && maxB <= planeEps) {
            intersectCoplanar(A, B, outA, outB, planeEps);
        }
        return;
    }

    if ((dA[0] > planeEps && dA[1] > planeEps && dA[2] > planeEps) ||
        (dA[0] < -planeEps && dA[1] < -planeEps && dA[2] < -planeEps) ||
        (dB[0] > planeEps && dB[1] > planeEps && dB[2] > planeEps) ||
        (dB[0] < -planeEps && dB[1] < -planeEps && dB[2] < -planeEps)) {
        return;
    }

    vector<Vec3> cand;
    auto addFromEdge = [&](const Vec3& p0, const Vec3& p1, double s0, double s1, const TriData& other) {
        if (std::fabs(s0) <= planeEps) {
            Vec2 q = other.project(p0);
            if (pointInTri2D(other, q, planeEps)) addCandidate(cand, p0, planeEps * planeEps * 4.0);
        }
        if (std::fabs(s1) <= planeEps) {
            Vec2 q = other.project(p1);
            if (pointInTri2D(other, q, planeEps)) addCandidate(cand, p1, planeEps * planeEps * 4.0);
        }
        if ((s0 < -planeEps && s1 > planeEps) || (s0 > planeEps && s1 < -planeEps)) {
            double t = s0 / (s0 - s1);
            Vec3 p = p0 + (p1 - p0) * t;
            Vec2 q = other.project(p);
            if (pointInTri2D(other, q, planeEps)) addCandidate(cand, p, planeEps * planeEps * 4.0);
        }
    };

    addFromEdge(A.p[0], A.p[1], dA[0], dA[1], B);
    addFromEdge(A.p[1], A.p[2], dA[1], dA[2], B);
    addFromEdge(A.p[2], A.p[0], dA[2], dA[0], B);
    addFromEdge(B.p[0], B.p[1], dB[0], dB[1], A);
    addFromEdge(B.p[1], B.p[2], dB[1], dB[2], A);
    addFromEdge(B.p[2], B.p[0], dB[2], dB[0], A);

    if (cand.size() < 2) return;
    std::sort(cand.begin(), cand.end(), [&](const Vec3& x, const Vec3& y) { return dot(x, nCross) < dot(y, nCross); });
    Vec3 a = cand.front();
    Vec3 b = cand.back();
    Vec3 d = b - a;
    if (norm2(d) <= planeEps * planeEps) return;
    outA.push_back({a, b});
    outB.push_back({a, b});
}

static void traversePairs(
    const MeshAccel& A,
    int na,
    const MeshAccel& B,
    int nb,
    vector<vector<RawSeg>>& segA,
    vector<vector<RawSeg>>& segB,
    double planeEps
) {
    const BVHNode& a = A.nodes[na];
    const BVHNode& b = B.nodes[nb];
    if (!overlap(a.box, b.box)) return;
    if (a.leaf && b.leaf) {
        for (int i = a.start; i < a.start + a.count; ++i) {
            int ia = A.order[i];
            for (int j = b.start; j < b.start + b.count; ++j) {
                int ib = B.order[j];
                if (!overlap(A.tris[ia].box, B.tris[ib].box)) continue;
                intersectTrianglePair(A.tris[ia], B.tris[ib], segA[ia], segB[ib], planeEps);
            }
        }
        return;
    }
    if (a.leaf || (!b.leaf && volume(a.box) < volume(b.box))) {
        traversePairs(A, na, B, b.left, segA, segB, planeEps);
        traversePairs(A, na, B, b.right, segA, segB, planeEps);
    } else {
        traversePairs(A, a.left, B, nb, segA, segB, planeEps);
        traversePairs(A, a.right, B, nb, segA, segB, planeEps);
    }
}

static void buildMeshTris(Mesh& mesh) {
    mesh.tris.clear();
    mesh.tris.reserve(mesh.faces.size());
    for (const auto& f : mesh.faces) {
        TriData t;
        t.idx[0] = f[0];
        t.idx[1] = f[1];
        t.idx[2] = f[2];
        t.p[0] = mesh.verts[f[0]];
        t.p[1] = mesh.verts[f[1]];
        t.p[2] = mesh.verts[f[2]];
        t.box = makeEmptyBox();
        expand(t.box, t.p[0]);
        expand(t.box, t.p[1]);
        expand(t.box, t.p[2]);
        t.centroid = (t.p[0] + t.p[1] + t.p[2]) / 3.0;
        Vec3 e0 = t.p[1] - t.p[0];
        Vec3 e1 = t.p[2] - t.p[0];
        Vec3 n = cross(e0, e1);
        double nlen = norm(n);
        if (nlen == 0.0) continue;
        t.nUnit = n / nlen;
        t.planeD = dot(t.nUnit, t.p[0]);
        t.origin = t.p[0];
        t.u = normalize(e0);
        t.v = cross(t.nUnit, t.u);
        t.q[0] = {0.0, 0.0};
        t.q[1] = t.project(t.p[1]);
        t.q[2] = t.project(t.p[2]);
        if (orient2d(t.q[0], t.q[1], t.q[2]) < 0.0) {
            t.v = t.v * -1.0;
            t.q[1].y = -t.q[1].y;
            t.q[2].y = -t.q[2].y;
        }
        t.area2 = orient2d(t.q[0], t.q[1], t.q[2]);
        double l01 = norm(t.p[1] - t.p[0]);
        double l12 = norm(t.p[2] - t.p[1]);
        double l20 = norm(t.p[0] - t.p[2]);
        t.scale = std::max({l01, l12, l20});
        mesh.tris.push_back(t);
    }
}

static bool pointInsideMesh(const MeshAccel& mesh, const Vec3& p, const Vec3& rayDir) {
    return mesh.pointInside(p, rayDir);
}

struct CycleSampleResult {
    Vec3 sample{};
    bool valid = false;
};

struct TriKey {
    int a = 0;
    int b = 0;
    int c = 0;
    bool operator==(const TriKey& o) const { return a == o.a && b == o.b && c == o.c; }
};

struct TriKeyHash {
    size_t operator()(const TriKey& k) const noexcept {
        size_t h1 = std::hash<int>{}(k.a);
        size_t h2 = std::hash<int>{}(k.b);
        size_t h3 = std::hash<int>{}(k.c);
        size_t h = h1 ^ (h2 + 0x9e3779b97f4a7c15ULL + (h1 << 6) + (h1 >> 2));
        return h ^ (h3 + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2));
    }
};

static CycleSampleResult samplePointForCycle(
    const Cycle& cyc,
    const vector<Cycle>& cycles,
    const vector<Vec2>& all2,
    const vector<Vec3>& all3
) {
    vector<vector<Point2>> poly(1);
    poly[0].reserve(cyc.verts.size());
    for (int id : cyc.verts) poly[0].push_back({all2[id].x, all2[id].y});
    std::vector<uint32_t> idx = mapbox::earcut<uint32_t>(poly);
    CycleSampleResult res;
    if (idx.size() < 3) return res;
    vector<Vec2> cycPoly;
    cycPoly.reserve(cyc.verts.size());
    for (int id : cyc.verts) cycPoly.push_back(all2[id]);

    for (size_t i = 0; i + 2 < idx.size(); i += 3) {
        int a = cyc.verts[idx[i + 0]];
        int b = cyc.verts[idx[i + 1]];
        int c = cyc.verts[idx[i + 2]];
        Vec2 p2{(all2[a].x + all2[b].x + all2[c].x) / 3.0, (all2[a].y + all2[b].y + all2[c].y) / 3.0};
        if (!pointInPolygon(cycPoly, p2, 1e-12)) continue;
        bool insideAnySmaller = false;
        for (const auto& other : cycles) {
            if (other.area >= cyc.area - 1e-14) continue;
            vector<Vec2> poly2;
            poly2.reserve(other.verts.size());
            for (int id : other.verts) poly2.push_back(all2[id]);
            if (pointInPolygon(poly2, p2, 1e-12)) {
                insideAnySmaller = true;
                break;
            }
        }
        if (!insideAnySmaller) {
            res.sample = (all3[a] + all3[b] + all3[c]) / 3.0;
            res.valid = true;
            return res;
        }
    }

    int a = cyc.verts[0];
    int b = cyc.verts[1];
    int c = cyc.verts[2];
    res.sample = (all3[a] + all3[b] + all3[c]) / 3.0;
    res.valid = true;
    return res;
}

static void processMeshTriangle(
    const TriData& tri,
    const vector<RawSeg>& rawSegs,
    const MeshAccel& other,
    const Vec3& rayDir,
    double sampleEps,
    vector<array<Vec3, 3>>& outTris
) {
    vector<Vec3> rawPts3;
    vector<Vec2> rawPts2;
    rawPts3.reserve(3 + rawSegs.size() * 2);
    rawPts2.reserve(3 + rawSegs.size() * 2);

    for (int i = 0; i < 3; ++i) {
        rawPts3.push_back(tri.p[i]);
        rawPts2.push_back(tri.q[i]);
    }
    for (const auto& s : rawSegs) {
        rawPts3.push_back(s.a);
        rawPts2.push_back(tri.project(s.a));
        rawPts3.push_back(s.b);
        rawPts2.push_back(tri.project(s.b));
    }

    double localTol = std::max({tri.scale * 1e-9, 1e-12});
    double cell = std::max(localTol * 4.0, 1e-15);
    LocalPointMap pointMap(cell, localTol);
    vector<int> idOfRaw(rawPts2.size(), -1);
    for (size_t i = 0; i < rawPts2.size(); ++i) {
        idOfRaw[i] = pointMap.add(rawPts2[i], rawPts3[i]);
    }

    vector<RawSeg> interiorSegs;
    interiorSegs.reserve(rawSegs.size());
    auto edgeIdOfPoint = [&](const Vec2& p, double* outT = nullptr) -> int {
        for (int e = 0; e < 3; ++e) {
            double tt = 0.0;
            if (pointOnEdge2D(tri, p, e, localTol * 8.0, &tt)) {
                if (outT) *outT = tt;
                return e;
            }
        }
        return -1;
    };

    for (size_t i = 0; i < rawSegs.size(); ++i) {
        Vec2 a2 = tri.project(rawSegs[i].a);
        Vec2 b2 = tri.project(rawSegs[i].b);
        if (segOnBoundary(tri, a2, b2, localTol * 8.0)) continue;
        int ia = pointMap.add(a2, rawSegs[i].a);
        int ib = pointMap.add(b2, rawSegs[i].b);
        if (ia == ib) continue;
        interiorSegs.push_back(rawSegs[i]);
    }

    if ((int)pointMap.pts2.size() == 3 && interiorSegs.empty()) {
        Vec3 sample = tri.centroid + tri.nUnit * sampleEps;
        if (!pointInsideMesh(other, sample, rayDir)) {
            outTris.push_back({tri.p[0], tri.p[1], tri.p[2]});
        }
        return;
    }

    vector<Vec2> local2 = pointMap.pts2;
    vector<Vec3> local3 = pointMap.pts3;

    vector<std::pair<double, int>> edgePts[3];
    for (int id = 0; id < (int)local2.size(); ++id) {
        for (int e = 0; e < 3; ++e) {
            double t = 0.0;
            if (pointOnEdge2D(tri, local2[id], e, localTol * 16.0, &t)) {
                edgePts[e].push_back({t, id});
            }
        }
    }
    vector<std::pair<int, int>> undirected;
    std::unordered_set<uint64_t> edgeSet;
    auto addEdge = [&](int a, int b) {
        if (a == b) return;
        if (a > b) std::swap(a, b);
        uint64_t key = (uint64_t)(uint32_t)a << 32 | (uint32_t)b;
        if (!edgeSet.insert(key).second) return;
        undirected.push_back({a, b});
    };

    for (int e = 0; e < 3; ++e) {
        auto& v = edgePts[e];
        std::sort(v.begin(), v.end(), [](const auto& A, const auto& B) {
            if (std::fabs(A.first - B.first) > 1e-12) return A.first < B.first;
            return A.second < B.second;
        });
        vector<std::pair<double, int>> uniq;
        for (const auto& p : v) {
            if (uniq.empty() || std::fabs(p.first - uniq.back().first) > 1e-12 || p.second != uniq.back().second) {
                uniq.push_back(p);
            }
        }
        v.swap(uniq);
        for (size_t i = 1; i < v.size(); ++i) addEdge(v[i - 1].second, v[i].second);
    }

    for (const auto& s : rawSegs) {
        Vec2 a2 = tri.project(s.a);
        Vec2 b2 = tri.project(s.b);
        if (segOnBoundary(tri, a2, b2, localTol * 8.0)) continue;
        int ia = pointMap.add(a2, s.a);
        int ib = pointMap.add(b2, s.b);
        addEdge(ia, ib);
    }

    if (undirected.empty()) {
        Vec3 sample = tri.centroid + tri.nUnit * sampleEps;
        if (!pointInsideMesh(other, sample, rayDir)) {
            outTris.push_back({tri.p[0], tri.p[1], tri.p[2]});
        }
        return;
    }

    vector<vector<int>> adj(local2.size());
    for (const auto& e : undirected) {
        adj[e.first].push_back(e.second);
        adj[e.second].push_back(e.first);
    }

    for (size_t i = 0; i < adj.size(); ++i) {
        auto& v = adj[i];
        std::sort(v.begin(), v.end(), [&](int a, int b) {
            Vec2 da = local2[a] - local2[i];
            Vec2 db = local2[b] - local2[i];
            double aa = std::atan2(da.y, da.x);
            double ab = std::atan2(db.y, db.x);
            if (aa != ab) return aa < ab;
            double da2 = norm2(da);
            double db2 = norm2(db);
            return da2 < db2;
        });
        v.erase(std::unique(v.begin(), v.end()), v.end());
    }

    std::unordered_set<uint64_t> visited;
    vector<Cycle> cycles;
    auto enc = [](int a, int b) -> uint64_t {
        return (uint64_t)(uint32_t)a << 32 | (uint32_t)b;
    };

    for (int u = 0; u < (int)adj.size(); ++u) {
        for (int v : adj[u]) {
            if (visited.count(enc(u, v))) continue;
            vector<int> cyc;
            int a = u, b = v;
            while (!visited.count(enc(a, b))) {
                visited.insert(enc(a, b));
                cyc.push_back(a);
                const auto& nb = adj[b];
                int pos = -1;
                for (int i = 0; i < (int)nb.size(); ++i) {
                    if (nb[i] == a) {
                        pos = i;
                        break;
                    }
                }
                if (pos < 0 || nb.empty()) break;
                int nxt = nb[(pos - 1 + (int)nb.size()) % (int)nb.size()];
                a = b;
                b = nxt;
            }
            while (cyc.size() >= 2 && cyc.front() == cyc.back()) cyc.pop_back();
            if (cyc.size() < 3) continue;
            vector<Vec2> poly;
            poly.reserve(cyc.size());
            for (int id : cyc) poly.push_back(local2[id]);
            double ar = polygonArea2D(poly);
            if (ar <= 1e-14) continue;
            Cycle c;
            c.verts = std::move(cyc);
            c.area = ar;
            c.bbox = makeEmptyBox();
            for (int id : c.verts) expand(c.bbox, local3[id]);
            cycles.push_back(std::move(c));
        }
    }

    if (cycles.empty()) {
        Vec3 sample = tri.centroid + tri.nUnit * sampleEps;
        if (!pointInsideMesh(other, sample, rayDir)) {
            outTris.push_back({tri.p[0], tri.p[1], tri.p[2]});
        }
        return;
    }

    std::sort(cycles.begin(), cycles.end(), [](const Cycle& a, const Cycle& b) {
        if (std::fabs(a.area - b.area) > 1e-12) return a.area < b.area;
        return a.verts.size() < b.verts.size();
    });

    vector<Vec2> cyclePolyScratch;
    vector<CycleSampleResult> cycleSamples(cycles.size());
    for (size_t i = 0; i < cycles.size(); ++i) {
        cycleSamples[i] = samplePointForCycle(cycles[i], vector<Cycle>(cycles.begin(), cycles.begin() + (int)i), local2, local3);
        if (!cycleSamples[i].valid) {
            Vec2 fallback2{0.0, 0.0};
            Vec3 fallback3{0.0, 0.0, 0.0};
            for (int id : cycles[i].verts) {
                fallback2 += local2[id];
                fallback3 += local3[id];
            }
            fallback2 = fallback2 * (1.0 / cycles[i].verts.size());
            fallback3 = fallback3 * (1.0 / cycles[i].verts.size());
            cycleSamples[i].sample = fallback3;
            cycleSamples[i].valid = true;
        }
        cycles[i].sample3 = cycleSamples[i].sample;
    }

    for (size_t i = 0; i < cycles.size(); ++i) {
        int parent = -1;
        double bestArea = std::numeric_limits<double>::infinity();
        vector<Vec2> p2;
        p2.reserve(cycles[i].verts.size());
        for (int id : cycles[i].verts) p2.push_back(local2[id]);
        Vec2 sp2 = tri.project(cycleSamples[i].sample);

        for (size_t j = 0; j < cycles.size(); ++j) {
            if (cycles[j].area <= cycles[i].area + 1e-12) continue;
            vector<Vec2> jp2;
            jp2.reserve(cycles[j].verts.size());
            for (int id : cycles[j].verts) jp2.push_back(local2[id]);
            if (!box2ContainsPoint(jp2, sp2, 1e-12)) continue;
            if (!pointInPolygon(jp2, sp2, 1e-12)) continue;
            if (cycles[j].area < bestArea) {
                bestArea = cycles[j].area;
                parent = (int)j;
            }
        }
        cycles[i].parent = parent;
        if (parent >= 0) cycles[parent].children.push_back((int)i);
    }

    for (size_t i = 0; i < cycles.size(); ++i) {
        if (cycles[i].parent >= 0) continue;
        // Outer cycles should be handled here, but process all roots in case of disjoint components.
        vector<int> stackNodes = {(int)i};
        while (!stackNodes.empty()) {
            int cur = stackNodes.back();
            stackNodes.pop_back();

            vector<vector<Point2>> polygon;
            vector<int> flatIds;
            polygon.emplace_back();
            polygon.back().reserve(cycles[cur].verts.size());
            for (int id : cycles[cur].verts) {
                polygon.back().push_back({local2[id].x, local2[id].y});
                flatIds.push_back(id);
            }
            for (int ch : cycles[cur].children) {
                polygon.emplace_back();
                polygon.back().reserve(cycles[ch].verts.size());
                for (int id : cycles[ch].verts) {
                    polygon.back().push_back({local2[id].x, local2[id].y});
                    flatIds.push_back(id);
                }
            }

            std::vector<uint32_t> idx = mapbox::earcut<uint32_t>(polygon);
            if (idx.size() >= 3) {
                vector<array<int, 3>> regionTris;
                regionTris.reserve(idx.size() / 3);
                for (size_t k = 0; k + 2 < idx.size(); k += 3) {
                    int a = flatIds[idx[k + 0]];
                    int b = flatIds[idx[k + 1]];
                    int c = flatIds[idx[k + 2]];
                    Vec2 pa = local2[a], pb = local2[b], pc = local2[c];
                    double ar = orient2d(pa, pb, pc);
                    if (std::fabs(ar) < 1e-14) continue;
                    if (ar < 0) std::swap(b, c);
                    regionTris.push_back({a, b, c});
                }

                if (!regionTris.empty()) {
                    Vec3 sample = (local3[regionTris[0][0]] + local3[regionTris[0][1]] + local3[regionTris[0][2]]) / 3.0 + tri.nUnit * sampleEps;
                    if (!pointInsideMesh(other, sample, rayDir)) {
                        for (const auto& t : regionTris) {
                            outTris.push_back({local3[t[0]], local3[t[1]], local3[t[2]]});
                        }
                    }
                }
            }

            for (int ch : cycles[cur].children) stackNodes.push_back(ch);
        }
    }
}

static TriData makeTri(const vector<Vec3>& verts, const array<int, 3>& f) {
    TriData t;
    t.idx[0] = f[0];
    t.idx[1] = f[1];
    t.idx[2] = f[2];
    t.p[0] = verts[f[0]];
    t.p[1] = verts[f[1]];
    t.p[2] = verts[f[2]];
    t.box = makeEmptyBox();
    expand(t.box, t.p[0]);
    expand(t.box, t.p[1]);
    expand(t.box, t.p[2]);
    t.centroid = (t.p[0] + t.p[1] + t.p[2]) / 3.0;
    Vec3 e0 = t.p[1] - t.p[0];
    Vec3 e1 = t.p[2] - t.p[0];
    Vec3 n = cross(e0, e1);
    double nlen = norm(n);
    if (nlen == 0.0) {
        t.nUnit = {0, 0, 1};
        t.origin = t.p[0];
        t.u = {1, 0, 0};
        t.v = {0, 1, 0};
        t.q[0] = {0, 0};
        t.q[1] = {dot(e0, t.u), dot(e0, t.v)};
        t.q[2] = {dot(e1, t.u), dot(e1, t.v)};
        t.scale = 0.0;
        return t;
    }
    t.nUnit = n / nlen;
    t.planeD = dot(t.nUnit, t.p[0]);
    t.origin = t.p[0];
    t.u = normalize(e0);
    t.v = cross(t.nUnit, t.u);
    t.q[0] = {0.0, 0.0};
    t.q[1] = t.project(t.p[1]);
    t.q[2] = t.project(t.p[2]);
    if (orient2d(t.q[0], t.q[1], t.q[2]) < 0.0) {
        t.v = t.v * -1.0;
        t.q[1].y = -t.q[1].y;
        t.q[2].y = -t.q[2].y;
    }
    t.area2 = orient2d(t.q[0], t.q[1], t.q[2]);
    double l01 = norm(t.p[1] - t.p[0]);
    double l12 = norm(t.p[2] - t.p[1]);
    double l20 = norm(t.p[0] - t.p[2]);
    t.scale = std::max({l01, l12, l20});
    return t;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.verts.resize(VA);
    B.verts.resize(VB);
    for (int i = 0; i < VA; ++i) std::cin >> A.verts[i].x >> A.verts[i].y >> A.verts[i].z;
    for (int i = 0; i < FA; ++i) {
        array<int, 3> f{};
        std::cin >> f[0] >> f[1] >> f[2];
        A.faces.push_back(f);
    }
    for (int i = 0; i < VB; ++i) std::cin >> B.verts[i].x >> B.verts[i].y >> B.verts[i].z;
    for (int i = 0; i < FB; ++i) {
        array<int, 3> f{};
        std::cin >> f[0] >> f[1] >> f[2];
        B.faces.push_back(f);
    }

    buildMeshTris(A);
    buildMeshTris(B);

    BBox3 globalBox = makeEmptyBox();
    for (const auto& p : A.verts) expand(globalBox, p);
    for (const auto& p : B.verts) expand(globalBox, p);
    double diag = norm(globalBox.mx - globalBox.mn);
    double planeEps = std::max(1e-9 * diag, 1e-12);
    double sampleEps = std::max(1e-9 * diag, 1e-12);
    Vec3 rayDir = normalize(Vec3{0.731, 0.527, 0.437});

    MeshAccel accelA, accelB;
    accelA.build(A.tris);
    accelB.build(B.tris);

    vector<vector<RawSeg>> segA(A.tris.size()), segB(B.tris.size());
    if (accelA.root >= 0 && accelB.root >= 0) {
        traversePairs(accelA, accelA.root, accelB, accelB.root, segA, segB, planeEps);
    }

    vector<array<Vec3, 3>> outTris3;
    outTris3.reserve((size_t)(FA + FB) * 2);

    for (size_t i = 0; i < A.tris.size(); ++i) {
        processMeshTriangle(A.tris[i], segA[i], accelB, rayDir, sampleEps, outTris3);
    }
    for (size_t i = 0; i < B.tris.size(); ++i) {
        processMeshTriangle(B.tris[i], segB[i], accelA, rayDir, sampleEps, outTris3);
    }

    GlobalVertexMap gv(std::max(1e-8 * diag, 1e-12), std::max(1e-8 * diag, 1e-12));
    vector<array<int, 3>> facesOut;
    facesOut.reserve(outTris3.size());
    std::unordered_set<TriKey, TriKeyHash> triSet;

    for (const auto& t : outTris3) {
        int a = gv.add(t[0]);
        int b = gv.add(t[1]);
        int c = gv.add(t[2]);
        if (a == b || b == c || a == c) continue;
        Vec3 ab = gv.pts[b] - gv.pts[a];
        Vec3 ac = gv.pts[c] - gv.pts[a];
        if (norm2(cross(ab, ac)) <= 1e-20 * diag * diag) continue;
        array<int, 3> keyArr{a, b, c};
        std::sort(keyArr.begin(), keyArr.end());
        TriKey key{keyArr[0], keyArr[1], keyArr[2]};
        if (!triSet.insert(key).second) continue;
        facesOut.push_back({a, b, c});
    }

    std::cout.setf(std::ios::fixed);
    std::cout.precision(17);
    std::cout << gv.pts.size() << ' ' << facesOut.size() << '\n';
    for (const auto& p : gv.pts) {
        std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const auto& f : facesOut) {
        std::cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }
    return 0;
}
