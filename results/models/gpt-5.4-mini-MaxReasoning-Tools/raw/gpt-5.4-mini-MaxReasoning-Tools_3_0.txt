use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read, Write};

const SMALL_LIMIT: usize = 1200;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        if looks_like_utf16le(&buf) {
            let mut ascii = Vec::with_capacity(buf.len() / 2);
            let mut i = if buf.len() >= 2 && buf[0] == 0xFF && buf[1] == 0xFE {
                2
            } else {
                0
            };
            while i + 1 < buf.len() {
                ascii.push(buf[i]);
                i += 2;
            }
            buf = ascii;
        }
        Self { buf, idx: 0 }
    }

    #[inline]
    fn next_usize(&mut self) -> usize {
        while self.idx < self.buf.len() && is_sep(self.buf[self.idx]) {
            self.idx += 1;
        }
        let mut n = 0usize;
        while self.idx < self.buf.len() && !is_sep(self.buf[self.idx]) {
            n = n * 10 + (self.buf[self.idx] - b'0') as usize;
            self.idx += 1;
        }
        n
    }
}

#[inline]
fn is_sep(b: u8) -> bool {
    b == 0 || b.is_ascii_whitespace()
}

fn looks_like_utf16le(buf: &[u8]) -> bool {
    if buf.len() < 4 {
        return false;
    }
    let pairs = buf.len() / 2;
    let mut zero_odd = 0usize;
    let mut zero_even = 0usize;
    for i in 0..pairs {
        if buf[2 * i] == 0 {
            zero_even += 1;
        }
        if buf[2 * i + 1] == 0 {
            zero_odd += 1;
        }
    }
    zero_odd > pairs / 2 && zero_even < pairs / 4
}

fn bfs_farthest(
    start: usize,
    cid: usize,
    comp_id: &[usize],
    adj: &[Vec<usize>],
    seen: &mut [u32],
    stamp: &mut u32,
    dist: &mut [i32],
    parent: &mut [usize],
) -> usize {
    *stamp = stamp.wrapping_add(1);
    let cur = *stamp;
    let mut q = VecDeque::new();
    q.push_back(start);
    seen[start] = cur;
    dist[start] = 0;
    parent[start] = usize::MAX;
    let mut far = start;

    while let Some(v) = q.pop_front() {
        let dv = dist[v];
        if dv > dist[far] || (dv == dist[far] && v < far) {
            far = v;
        }
        for &to in &adj[v] {
            if comp_id[to] != cid || seen[to] == cur {
                continue;
            }
            seen[to] = cur;
            dist[to] = dv + 1;
            parent[to] = v;
            q.push_back(to);
        }
    }

    far
}

fn choose_root(
    nodes: &[usize],
    cid: usize,
    comp_id: &[usize],
    adj: &[Vec<usize>],
    seen: &mut [u32],
    stamp: &mut u32,
    dist: &mut [i32],
    parent: &mut [usize],
) -> usize {
    let mut start = nodes[0];
    let mut best_deg = adj[start].len();
    for &v in nodes.iter().skip(1) {
        let d = adj[v].len();
        if d > best_deg || (d == best_deg && v < start) {
            start = v;
            best_deg = d;
        }
    }

    let a = bfs_farthest(start, cid, comp_id, adj, seen, stamp, dist, parent);
    let b = bfs_farthest(a, cid, comp_id, adj, seen, stamp, dist, parent);

    let mut path = Vec::new();
    let mut cur = b;
    loop {
        path.push(cur);
        if cur == a {
            break;
        }
        cur = parent[cur];
    }
    path[path.len() / 2]
}

fn normalize_component(
    nodes: &[usize],
    x: &mut [f64],
    y: &mut [f64],
    target: f64,
) -> (f64, f64) {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for &u in nodes {
        let xu = x[u];
        let yu = y[u];
        if xu < minx {
            minx = xu;
        }
        if xu > maxx {
            maxx = xu;
        }
        if yu < miny {
            miny = yu;
        }
        if yu > maxy {
            maxy = yu;
        }
    }

    let w = maxx - minx;
    let h = maxy - miny;
    let max_side = w.max(h);
    if !max_side.is_finite() || max_side <= 1e-12 {
        for &u in nodes {
            x[u] = 0.0;
            y[u] = 0.0;
        }
        return (0.0, 0.0);
    }

    let scale = target / max_side;
    for &u in nodes {
        x[u] = (x[u] - minx) * scale;
        y[u] = (y[u] - miny) * scale;
    }
    (w * scale, h * scale)
}

fn layout_small_component(
    nodes: &[usize],
    adj: &[Vec<usize>],
    x: &mut [f64],
    y: &mut [f64],
    local_of: &mut [usize],
) -> (f64, f64) {
    let k = nodes.len();
    if k == 1 {
        x[nodes[0]] = 0.0;
        y[nodes[0]] = 0.0;
        return (0.0, 0.0);
    }

    for (i, &u) in nodes.iter().enumerate() {
        local_of[u] = i;
    }

    let mut edges: Vec<(usize, usize)> = Vec::new();
    for &u in nodes {
        let lu = local_of[u];
        for &v in &adj[u] {
            let lv = local_of[v];
            if lv != usize::MAX && lu < lv {
                edges.push((lu, lv));
            }
        }
    }
    edges.sort_unstable();
    edges.dedup();

    let radius = (k as f64).sqrt().max(2.0);
    let ideal = radius;
    let mut px = vec![0.0; k];
    let mut py = vec![0.0; k];
    for i in 0..k {
        let ang = 2.0 * PI * (i as f64) / (k as f64);
        px[i] = radius * ang.cos();
        py[i] = radius * ang.sin();
    }

    let mut dx = vec![0.0; k];
    let mut dy = vec![0.0; k];
    let iters = if k <= 50 {
        120
    } else if k <= 200 {
        80
    } else if k <= 500 {
        50
    } else if k <= 800 {
        35
    } else {
        25
    };
    let mut temp = radius * 0.5;

    for _ in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);

        for i in 0..k {
            for j in (i + 1)..k {
                let vx = px[i] - px[j];
                let vy = py[i] - py[j];
                let dist2 = vx * vx + vy * vy + 1e-9;
                let dist = dist2.sqrt();
                let force = ideal * ideal / dist;
                let fx = vx / dist * force;
                let fy = vy / dist * force;
                dx[i] += fx;
                dy[i] += fy;
                dx[j] -= fx;
                dy[j] -= fy;
            }
        }

        for &(a, b) in &edges {
            let vx = px[a] - px[b];
            let vy = py[a] - py[b];
            let dist2 = vx * vx + vy * vy + 1e-9;
            let dist = dist2.sqrt();
            let force = dist2 / ideal;
            let fx = vx / dist * force;
            let fy = vy / dist * force;
            dx[a] -= fx;
            dy[a] -= fy;
            dx[b] += fx;
            dy[b] += fy;
        }

        for i in 0..k {
            dx[i] -= px[i] * 0.02;
            dy[i] -= py[i] * 0.02;
            let disp = (dx[i] * dx[i] + dy[i] * dy[i]).sqrt();
            if disp > 0.0 {
                let step = temp.min(disp);
                px[i] += dx[i] / disp * step;
                py[i] += dy[i] / disp * step;
            }
        }

        temp *= 0.92;
    }

    let target = 10.0 * (k as f64).sqrt().max(1.0) + 8.0;
    let dims = normalize_component(nodes, &mut px, &mut py, target);

    for (i, &u) in nodes.iter().enumerate() {
        x[u] = px[i];
        y[u] = py[i];
        local_of[u] = usize::MAX;
    }

    dims
}

fn layout_large_component(
    nodes: &[usize],
    cid: usize,
    comp_id: &[usize],
    adj: &[Vec<usize>],
    x: &mut [f64],
    y: &mut [f64],
    seen: &mut [u32],
    stamp: &mut u32,
    dist: &mut [i32],
    parent: &mut [usize],
    rank: &mut [usize],
    score: &mut [f64],
) -> (f64, f64) {
    let root = choose_root(nodes, cid, comp_id, adj, seen, stamp, dist, parent);

    *stamp = stamp.wrapping_add(1);
    let cur = *stamp;
    let mut q = VecDeque::new();
    q.push_back(root);
    seen[root] = cur;
    dist[root] = 0;

    let mut layers: Vec<Vec<usize>> = vec![vec![root]];
    while let Some(v) = q.pop_front() {
        let dv = dist[v] as usize;
        for &to in &adj[v] {
            if comp_id[to] != cid || seen[to] == cur {
                continue;
            }
            seen[to] = cur;
            dist[to] = dv as i32 + 1;
            if layers.len() <= dv + 1 {
                layers.resize_with(dv + 2, Vec::new);
            }
            layers[dv + 1].push(to);
            q.push_back(to);
        }
    }

    for layer in &layers {
        for (i, &u) in layer.iter().enumerate() {
            rank[u] = i;
        }
    }

    let k = nodes.len();
    let sweeps = if k > 80_000 {
        2
    } else if k > 10_000 {
        3
    } else {
        4
    };

    for _ in 0..sweeps {
        for l in 1..layers.len() {
            let prev_depth = (l as i32) - 1;
            for &u in &layers[l] {
                let mut sum = 0.0;
                let mut cnt = 0usize;
                for &v in &adj[u] {
                    if comp_id[v] == cid && dist[v] == prev_depth {
                        sum += rank[v] as f64;
                        cnt += 1;
                    }
                }
                score[u] = if cnt > 0 {
                    sum / cnt as f64
                } else {
                    rank[u] as f64
                };
            }
            layers[l].sort_unstable_by(|&a, &b| {
                score[a]
                    .total_cmp(&score[b])
                    .then(rank[a].cmp(&rank[b]))
                    .then(a.cmp(&b))
            });
            for (i, &u) in layers[l].iter().enumerate() {
                rank[u] = i;
            }
        }

        if layers.len() >= 2 {
            for l in (0..layers.len() - 1).rev() {
                let next_depth = (l as i32) + 1;
                for &u in &layers[l] {
                    let mut sum = 0.0;
                    let mut cnt = 0usize;
                    for &v in &adj[u] {
                        if comp_id[v] == cid && dist[v] == next_depth {
                            sum += rank[v] as f64;
                            cnt += 1;
                        }
                    }
                    score[u] = if cnt > 0 {
                        sum / cnt as f64
                    } else {
                        rank[u] as f64
                    };
                }
                layers[l].sort_unstable_by(|&a, &b| {
                    score[a]
                        .total_cmp(&score[b])
                        .then(rank[a].cmp(&rank[b]))
                        .then(a.cmp(&b))
                });
                for (i, &u) in layers[l].iter().enumerate() {
                    rank[u] = i;
                }
            }
        }
    }

    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;

    for (l, layer) in layers.iter().enumerate() {
        let len = layer.len();
        let center = (len as f64 - 1.0) * 0.5;
        let yy = l as f64 * 2.0;
        for (i, &u) in layer.iter().enumerate() {
            let xx = i as f64 - center;
            x[u] = xx;
            y[u] = yy;
            if xx < minx {
                minx = xx;
            }
            if xx > maxx {
                maxx = xx;
            }
            if yy < miny {
                miny = yy;
            }
            if yy > maxy {
                maxy = yy;
            }
        }
    }

    let w = maxx - minx;
    let h = maxy - miny;
    let max_side = w.max(h);
    let target = 10.0 * (k as f64).sqrt().max(1.0) + 8.0;
    if max_side > 1e-12 && max_side.is_finite() {
        let scale = target / max_side;
        for &u in nodes {
            x[u] = (x[u] - minx) * scale;
            y[u] = (y[u] - miny) * scale;
        }
        (w * scale, h * scale)
    } else {
        for &u in nodes {
            x[u] = 0.0;
            y[u] = 0.0;
        }
        (0.0, 0.0)
    }
}

fn main() {
    let mut sc = Scanner::new();
    if sc.buf.is_empty() {
        return;
    }

    let n = sc.next_usize();
    let m = sc.next_usize();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    if n == 0 {
        return;
    }

    let mut comp_id = vec![usize::MAX; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    let mut q = VecDeque::new();

    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let cid = components.len();
        comp_id[s] = cid;
        q.clear();
        q.push_back(s);
        let mut nodes = Vec::new();
        while let Some(v) = q.pop_front() {
            nodes.push(v);
            for &to in &adj[v] {
                if comp_id[to] == usize::MAX {
                    comp_id[to] = cid;
                    q.push_back(to);
                }
            }
        }
        components.push(nodes);
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];
    let mut comp_w = vec![0.0f64; components.len()];
    let mut comp_h = vec![0.0f64; components.len()];

    let mut seen = vec![0u32; n];
    let mut dist = vec![0i32; n];
    let mut parent = vec![usize::MAX; n];
    let mut rank = vec![0usize; n];
    let mut score = vec![0.0f64; n];
    let mut stamp: u32 = 1;
    let mut local_of = vec![usize::MAX; n];

    for (cid, nodes) in components.iter().enumerate() {
        let (w, h) = if nodes.len() <= SMALL_LIMIT {
            layout_small_component(nodes, &adj, &mut x, &mut y, &mut local_of)
        } else {
            layout_large_component(
                nodes,
                cid,
                &comp_id,
                &adj,
                &mut x,
                &mut y,
                &mut seen,
                &mut stamp,
                &mut dist,
                &mut parent,
                &mut rank,
                &mut score,
            )
        };
        comp_w[cid] = w;
        comp_h[cid] = h;
    }

    let mut order: Vec<usize> = (0..components.len()).collect();
    order.sort_unstable_by(|&a, &b| {
        let area_a = comp_w[a] * comp_h[a];
        let area_b = comp_w[b] * comp_h[b];
        area_b
            .total_cmp(&area_a)
            .then(components[b].len().cmp(&components[a].len()))
            .then(a.cmp(&b))
    });

    let total_area: f64 = order
        .iter()
        .map(|&cid| (comp_w[cid] + 12.0) * (comp_h[cid] + 12.0))
        .sum();
    let row_limit = (total_area.sqrt() * 1.25).max(1000.0);
    let gap = 12.0;
    let mut cur_x = 0.0;
    let mut cur_y = 0.0;
    let mut row_h = 0.0;

    for &cid in &order {
        let w = comp_w[cid];
        let h = comp_h[cid];
        let place_w = w + gap;
        let place_h = h + gap;
        if cur_x > 0.0 && cur_x + place_w > row_limit {
            cur_x = 0.0;
            cur_y += row_h;
            row_h = 0.0;
        }
        for &u in &components[cid] {
            x[u] += cur_x;
            y[u] += cur_y;
        }
        cur_x += place_w;
        if place_h > row_h {
            row_h = place_h;
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    for i in 0..n {
        writeln!(out, "{:.9} {:.9}", x[i], y[i]).unwrap();
    }
}
