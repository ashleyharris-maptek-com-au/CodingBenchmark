use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const ANGLE_COUNT: usize = 12;
const GRID_DIM: usize = 128;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Rect {
    x: f64,
    y: f64,
    w: f64,
    h: f64,
}

#[derive(Clone, Debug)]
struct Edge {
    a: Pt,
    b: Pt,
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
}

#[derive(Clone, Copy, Debug)]
struct Orient {
    rot: f64,
    minx: f64,
    miny: f64,
    w: f64,
    h: f64,
}

#[derive(Clone, Copy, Debug)]
struct OrientPlan {
    rot: f64,
    minx: f64,
    miny: f64,
    w: f64,
    h: f64,
    cw: usize,
    ch: usize,
    cell_area: usize,
}

#[derive(Clone, Debug)]
struct Piece {
    idx: usize,
    area: f64,
    orients: Vec<Orient>,
}

#[derive(Clone, Debug)]
struct PiecePlan {
    idx: usize,
    area: f64,
    best_cell_area: usize,
    plans: Vec<OrientPlan>,
}

#[derive(Clone, Debug)]
struct StockGrid {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
    cell: f64,
    gw: usize,
    gh: usize,
    safe_rows: Vec<u128>,
    edges: Vec<Edge>,
    geom_eps: f64,
}

#[derive(Clone, Debug)]
struct Bin {
    free_rows: Vec<u128>,
}

#[derive(Clone, Copy, Debug)]
struct Placement {
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut sc = Scanner::new(&input);

    let sv: usize = sc.next();
    let np: usize = sc.next();

    let mut stock_poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = sc.next();
        let y: f64 = sc.next();
        stock_poly.push(Pt { x, y });
    }

    let _stock_area = polygon_area_signed(&stock_poly).abs();
    let (minx, miny, maxx, maxy) = polygon_bbox(&stock_poly);
    let bbox_w = (maxx - minx).max(1e-12);
    let bbox_h = (maxy - miny).max(1e-12);
    let bbox_max = bbox_w.max(bbox_h).max(1e-12);

    let edges = build_edges(&stock_poly);
    let geom_eps = bbox_max * 1e-10 + 1e-12;

    let mut raw_pieces = Vec::with_capacity(np);
    let angles: Vec<f64> = (0..ANGLE_COUNT)
        .map(|k| (k as f64) * PI / 24.0)
        .collect();

    for idx in 0..np {
        let pv: usize = sc.next();
        let mut verts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = sc.next();
            let y: f64 = sc.next();
            verts.push(Pt { x, y });
        }
        let area = polygon_area_signed(&verts).abs();
        let mut orients = Vec::with_capacity(angles.len());
        for &rot in &angles {
            let (minx_r, miny_r, w, h) = rotated_bbox(&verts, rot);
            orients.push(Orient {
                rot,
                minx: minx_r,
                miny: miny_r,
                w,
                h,
            });
        }
        raw_pieces.push(Piece { idx, area, orients });
    }

    let grid = build_stock_grid(&stock_poly, &edges, minx, miny, maxx, maxy, geom_eps);
    let piece_plans = build_piece_plans(raw_pieces, grid.cell, grid.gw, grid.gh);

    let mut sorted = piece_plans;
    sorted.sort_by(|a, b| {
        b.best_cell_area
            .cmp(&a.best_cell_area)
            .then_with(|| b.area.partial_cmp(&a.area).unwrap_or(Ordering::Equal))
            .then_with(|| a.idx.cmp(&b.idx))
    });

    let mut bins: Vec<Bin> = Vec::new();
    let mut placements = vec![
        Placement {
            stock: 0,
            x: 0.0,
            y: 0.0,
            rot: 0.0,
        };
        np
    ];

    let row_mask_all = if grid.gw >= 128 {
        u128::MAX
    } else if grid.gw == 0 {
        0
    } else {
        (1u128 << grid.gw) - 1
    };

    for piece in sorted {
        let mut placed = false;

        for bin_idx in 0..bins.len() {
            if let Some((x_cell, y_cell, plan)) = try_place_grid(
                &bins[bin_idx],
                &piece,
                &grid,
                row_mask_all,
            ) {
                let rect_x = grid.minx + (x_cell as f64) * grid.cell;
                let rect_y = grid.miny + (y_cell as f64) * grid.cell;
                let tx = rect_x - plan.minx;
                let ty = rect_y - plan.miny;
                placements[piece.idx] = Placement {
                    stock: bin_idx,
                    x: tx,
                    y: ty,
                    rot: plan.rot,
                };
                mark_placement_cells(
                    &mut bins[bin_idx].free_rows,
                    &grid,
                    rect_x,
                    rect_y,
                    plan.w,
                    plan.h,
                );
                placed = true;
                break;
            }
        }

        if !placed {
            let exact_steps = [grid.cell / 4.0, grid.cell / 8.0];
            'exact_existing: for bin_idx in 0..bins.len() {
                for &step in &exact_steps {
                    if let Some((rect, plan)) = try_place_exact(
                        &bins[bin_idx],
                        &piece,
                        &grid,
                        &stock_poly,
                        &grid.edges,
                        step.max(1e-9),
                    ) {
                        let bin_idx_out = bin_idx;
                        placements[piece.idx] = Placement {
                            stock: bin_idx_out,
                            x: rect.x - plan.minx,
                            y: rect.y - plan.miny,
                            rot: plan.rot,
                        };
                        mark_rect_cells(&mut bins[bin_idx_out].free_rows, &grid, rect);
                        placed = true;
                        break 'exact_existing;
                    }
                }
            }
        }

        if !placed {
            let mut new_bin = Some(Bin {
                free_rows: grid.safe_rows.clone(),
            });

            if let Some((x_cell, y_cell, plan)) =
                try_place_grid(new_bin.as_ref().unwrap(), &piece, &grid, row_mask_all)
            {
                let rect_x = grid.minx + (x_cell as f64) * grid.cell;
                let rect_y = grid.miny + (y_cell as f64) * grid.cell;
                let tx = rect_x - plan.minx;
                let ty = rect_y - plan.miny;
                let bin_idx = bins.len();
                placements[piece.idx] = Placement {
                    stock: bin_idx,
                    x: tx,
                    y: ty,
                    rot: plan.rot,
                };
                if let Some(mut nb) = new_bin.take() {
                    mark_placement_cells(
                        &mut nb.free_rows,
                        &grid,
                        rect_x,
                        rect_y,
                        plan.w,
                        plan.h,
                    );
                    bins.push(nb);
                }
                placed = true;
            } else {
                let exact_steps = [
                    grid.cell / 4.0,
                    grid.cell / 8.0,
                    grid.cell / 16.0,
                    grid.cell / 32.0,
                ];
                let mut done = false;
                for &step in &exact_steps {
                    if let Some((rect, plan)) = try_place_exact(
                        new_bin.as_ref().unwrap(),
                        &piece,
                        &grid,
                        &stock_poly,
                        &grid.edges,
                        step.max(1e-9),
                    )
                    {
                        let bin_idx = bins.len();
                        placements[piece.idx] = Placement {
                            stock: bin_idx,
                            x: rect.x - plan.minx,
                            y: rect.y - plan.miny,
                            rot: plan.rot,
                        };
                        if let Some(mut nb) = new_bin.take() {
                            mark_rect_cells(&mut nb.free_rows, &grid, rect);
                            bins.push(nb);
                        }
                        placed = true;
                        done = true;
                        break;
                    }
                }
                if !done {
                    if let Some(nb) = new_bin.take() {
                        bins.push(nb);
                    }
                    let plan = &piece.plans[0];
                    let bin_idx = bins.len() - 1;
                    let rect = Rect {
                        x: grid.minx,
                        y: grid.miny,
                        w: plan.w,
                        h: plan.h,
                    };
                    placements[piece.idx] = Placement {
                        stock: bin_idx,
                        x: rect.x - plan.minx,
                        y: rect.y - plan.miny,
                        rot: plan.rot,
                    };
                    placed = true;
                }
            }
        }

        if !placed {
            unreachable!();
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", bins.len()));
    for p in placements {
        out.push_str(&format!(
            "{} {:.10} {:.10} {:.10}\n",
            p.stock, p.x, p.y, p.rot
        ));
    }
    print!("{}", out);
}

fn build_piece_plans(
    pieces: Vec<Piece>,
    cell: f64,
    gw: usize,
    gh: usize,
) -> Vec<PiecePlan> {
    let mut plans = Vec::with_capacity(pieces.len());
    for piece in pieces {
        let mut op = Vec::new();
        let mut best = usize::MAX;
        for &o in &piece.orients {
            let cw = ((o.w / cell).ceil() as usize).max(1);
            let ch = ((o.h / cell).ceil() as usize).max(1);
            if cw > gw || ch > gh {
                continue;
            }
            let cell_area = cw * ch;
            best = best.min(cell_area);
            op.push(OrientPlan {
                rot: o.rot,
                minx: o.minx,
                miny: o.miny,
                w: o.w,
                h: o.h,
                cw,
                ch,
                cell_area,
            });
        }
        if op.is_empty() {
            // Keep at least one orientation so the solver can still attempt exact fallback.
            let o = piece.orients[0];
            let cw = ((o.w / cell).ceil() as usize).max(1);
            let ch = ((o.h / cell).ceil() as usize).max(1);
            best = cw.saturating_mul(ch);
            op.push(OrientPlan {
                rot: o.rot,
                minx: o.minx,
                miny: o.miny,
                w: o.w,
                h: o.h,
                cw,
                ch,
                cell_area: cw * ch,
            });
        }
        op.sort_by(|a, b| {
            a.cell_area
                .cmp(&b.cell_area)
                .then_with(|| a.h.partial_cmp(&b.h).unwrap_or(Ordering::Equal))
                .then_with(|| a.w.partial_cmp(&b.w).unwrap_or(Ordering::Equal))
                .then_with(|| a.rot.partial_cmp(&b.rot).unwrap_or(Ordering::Equal))
        });
        plans.push(PiecePlan {
            idx: piece.idx,
            area: piece.area,
            best_cell_area: best,
            plans: op,
        });
    }
    plans
}

fn try_place_grid(
    bin: &Bin,
    piece: &PiecePlan,
    grid: &StockGrid,
    row_mask_all: u128,
) -> Option<(usize, usize, OrientPlan)> {
    for plan in &piece.plans {
        if plan.cw > grid.gw || plan.ch > grid.gh {
            continue;
        }
        let x_limit = grid.gw - plan.cw;
        let y_limit = grid.gh - plan.ch;
        for y in 0..=y_limit {
            let mut inter = row_mask_all;
            for r in 0..plan.ch {
                inter &= bin.free_rows[y + r];
                if inter == 0 {
                    break;
                }
            }
            if inter == 0 {
                continue;
            }
            if let Some(x) = first_run(inter, grid.gw, plan.cw, x_limit) {
                return Some((x, y, plan.clone()));
            }
        }
    }
    None
}

fn first_run(mask: u128, width: usize, run_len: usize, x_limit: usize) -> Option<usize> {
    let mut count = 0usize;
    let mut bits = mask;
    for x in 0..width {
        if bits & 1 != 0 {
            count += 1;
            if count >= run_len {
                let start = x + 1 - run_len;
                if start <= x_limit {
                    return Some(start);
                }
            }
        } else {
            count = 0;
        }
        bits >>= 1;
    }
    None
}

fn try_place_exact(
    bin: &Bin,
    piece: &PiecePlan,
    grid: &StockGrid,
    poly: &[Pt],
    edges: &[Edge],
    step: f64,
) -> Option<(Rect, OrientPlan)> {
    let max_x = grid.maxx;
    let max_y = grid.maxy;
    let phases = [0.0, step * 0.5];
    for plan in &piece.plans {
        for &oy in &phases {
            let mut y = grid.miny + oy;
            while y + plan.h <= max_y + grid.geom_eps {
                for &ox in &phases {
                    let mut x = grid.minx + ox;
                    while x + plan.w <= max_x + grid.geom_eps {
                        let rect = Rect {
                            x,
                            y,
                            w: plan.w,
                            h: plan.h,
                        };
                        if rect_inside_poly_exact(rect, poly, edges, grid.geom_eps)
                            && rect_cells_free(bin, grid, rect)
                        {
                            return Some((rect, plan.clone()));
                        }
                        x += step;
                    }
                }
                y += step;
            }
        }
    }
    None
}

fn rect_cells_free(bin: &Bin, grid: &StockGrid, rect: Rect) -> bool {
    let (x0, x1, y0, y1) = rect_cell_bounds(grid, rect);
    if x0 >= x1 || y0 >= y1 {
        return false;
    }
    let mask = range_mask(x0, x1);
    for row in y0..y1 {
        if bin.free_rows[row] & mask != mask {
            return false;
        }
    }
    true
}

fn mark_placement_cells(bin_rows: &mut [u128], grid: &StockGrid, rect_x: f64, rect_y: f64, w: f64, h: f64) {
    let rect = Rect {
        x: rect_x,
        y: rect_y,
        w,
        h,
    };
    mark_rect_cells(bin_rows, grid, rect);
}

fn mark_rect_cells(bin_rows: &mut [u128], grid: &StockGrid, rect: Rect) {
    let (x0, x1, y0, y1) = rect_cell_bounds(grid, rect);
    if x0 >= x1 || y0 >= y1 {
        return;
    }
    let mask = range_mask(x0, x1);
    for row in y0..y1 {
        bin_rows[row] &= !mask;
    }
}

fn rect_cell_bounds(grid: &StockGrid, rect: Rect) -> (usize, usize, usize, usize) {
    let mut x0 = ((rect.x - grid.minx) / grid.cell).floor() as isize;
    let mut x1 = ((rect.x + rect.w - grid.minx) / grid.cell).ceil() as isize;
    let mut y0 = ((rect.y - grid.miny) / grid.cell).floor() as isize;
    let mut y1 = ((rect.y + rect.h - grid.miny) / grid.cell).ceil() as isize;

    x0 = x0.max(0).min(grid.gw as isize);
    x1 = x1.max(0).min(grid.gw as isize);
    y0 = y0.max(0).min(grid.gh as isize);
    y1 = y1.max(0).min(grid.gh as isize);

    (x0 as usize, x1 as usize, y0 as usize, y1 as usize)
}

fn range_mask(x0: usize, x1: usize) -> u128 {
    if x0 >= x1 {
        0
    } else {
        let len = x1 - x0;
        let base = if len >= 128 {
            u128::MAX
        } else {
            (1u128 << len) - 1
        };
        base << x0
    }
}

fn build_stock_grid(
    poly: &[Pt],
    edges: &[Edge],
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
    geom_eps: f64,
) -> StockGrid {
    let bbox_w = (maxx - minx).max(1e-12);
    let bbox_h = (maxy - miny).max(1e-12);
    let cell = (bbox_w.max(bbox_h) / GRID_DIM as f64).max(1e-12);
    let gw = ((bbox_w / cell).floor() as usize).max(1).min(128);
    let gh = ((bbox_h / cell).floor() as usize).max(1).min(128);

    let mut buckets = vec![Vec::<usize>::new(); gw * gh];
    for (ei, e) in edges.iter().enumerate() {
        let mut ix0 = ((e.minx - minx) / cell).floor() as isize;
        let mut ix1 = ((e.maxx - minx) / cell).floor() as isize;
        let mut iy0 = ((e.miny - miny) / cell).floor() as isize;
        let mut iy1 = ((e.maxy - miny) / cell).floor() as isize;
        ix0 = ix0.max(0).min((gw - 1) as isize);
        ix1 = ix1.max(0).min((gw - 1) as isize);
        iy0 = iy0.max(0).min((gh - 1) as isize);
        iy1 = iy1.max(0).min((gh - 1) as isize);
        for y in iy0..=iy1 {
            for x in ix0..=ix1 {
                buckets[(y as usize) * gw + (x as usize)].push(ei);
            }
        }
    }

    let mut safe_rows = vec![0u128; gh];
    for j in 0..gh {
        for i in 0..gw {
            let x0 = minx + (i as f64) * cell;
            let y0 = miny + (j as f64) * cell;
            let x1 = x0 + cell;
            let y1 = y0 + cell;
            let corners = [
                Pt { x: x0, y: y0 },
                Pt { x: x1, y: y0 },
                Pt { x: x1, y: y1 },
                Pt { x: x0, y: y1 },
            ];
            if corners.iter().any(|&p| !point_in_poly(p, poly, geom_eps)) {
                continue;
            }
            let mut safe = true;
            for &ei in &buckets[j * gw + i] {
                let e = &edges[ei];
                if edge_intersects_rect_proper(e.a, e.b, x0, y0, x1, y1, geom_eps) {
                    safe = false;
                    break;
                }
            }
            if safe {
                safe_rows[j] |= 1u128 << i;
            }
        }
    }

    StockGrid {
        minx,
        miny,
        maxx,
        maxy,
        cell,
        gw,
        gh,
        safe_rows,
        edges: edges.to_vec(),
        geom_eps,
    }
}

fn rect_inside_poly_exact(rect: Rect, poly: &[Pt], edges: &[Edge], eps: f64) -> bool {
    let corners = [
        Pt {
            x: rect.x,
            y: rect.y,
        },
        Pt {
            x: rect.x + rect.w,
            y: rect.y,
        },
        Pt {
            x: rect.x + rect.w,
            y: rect.y + rect.h,
        },
        Pt {
            x: rect.x,
            y: rect.y + rect.h,
        },
    ];
    if corners.iter().any(|&p| !point_in_poly(p, poly, eps)) {
        return false;
    }
    let x0 = rect.x;
    let y0 = rect.y;
    let x1 = rect.x + rect.w;
    let y1 = rect.y + rect.h;
    for e in edges {
        if edge_intersects_rect_proper(e.a, e.b, x0, y0, x1, y1, eps) {
            return false;
        }
    }
    true
}

fn edge_intersects_rect_proper(a: Pt, b: Pt, x0: f64, y0: f64, x1: f64, y1: f64, eps: f64) -> bool {
    let r1 = Pt { x: x0, y: y0 };
    let r2 = Pt { x: x1, y: y0 };
    let r3 = Pt { x: x1, y: y1 };
    let r4 = Pt { x: x0, y: y1 };
    proper_segment_intersection(a, b, r1, r2, eps)
        || proper_segment_intersection(a, b, r2, r3, eps)
        || proper_segment_intersection(a, b, r3, r4, eps)
        || proper_segment_intersection(a, b, r4, r1, eps)
}

fn proper_segment_intersection(a: Pt, b: Pt, c: Pt, d: Pt, eps: f64) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() <= eps || o2.abs() <= eps || o3.abs() <= eps || o4.abs() <= eps {
        return false;
    }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly(p: Pt, poly: &[Pt], eps: f64) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b, eps) {
            return true;
        }
        let yi = a.y > p.y;
        let yj = b.y > p.y;
        if yi != yj {
            let x_int = a.x + (b.x - a.x) * (p.y - a.y) / (b.y - a.y);
            if x_int >= p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_on_segment(p: Pt, a: Pt, b: Pt, eps: f64) -> bool {
    let cross = orient(a, b, p).abs();
    if cross > eps {
        return false;
    }
    let minx = a.x.min(b.x) - eps;
    let maxx = a.x.max(b.x) + eps;
    let miny = a.y.min(b.y) - eps;
    let maxy = a.y.max(b.y) + eps;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn build_edges(poly: &[Pt]) -> Vec<Edge> {
    let mut edges = Vec::with_capacity(poly.len());
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        edges.push(Edge {
            a,
            b,
            minx: a.x.min(b.x),
            maxx: a.x.max(b.x),
            miny: a.y.min(b.y),
            maxy: a.y.max(b.y),
        });
    }
    edges
}

fn polygon_area_signed(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut acc = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        acc += a.x * b.y - a.y * b.x;
    }
    acc * 0.5
}

fn polygon_bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn rotated_bbox(poly: &[Pt], angle: f64) -> (f64, f64, f64, f64) {
    let c = angle.cos();
    let s = angle.sin();
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        let rx = p.x * c - p.y * s;
        let ry = p.x * s + p.y * c;
        minx = minx.min(rx);
        miny = miny.min(ry);
        maxx = maxx.max(rx);
        maxy = maxy.max(ry);
    }
    (minx, miny, maxx - minx, maxy - miny)
}

struct Scanner<'a> {
    it: std::str::SplitWhitespace<'a>,
}

impl<'a> Scanner<'a> {
    fn new(s: &'a str) -> Self {
        let s = s.strip_prefix('\u{feff}').unwrap_or(s);
        Self {
            it: s.split_whitespace(),
        }
    }

    fn next<T: std::str::FromStr>(&mut self) -> T {
        self.it
            .next()
            .expect("expected token")
            .parse()
            .ok()
            .expect("failed to parse token")
    }
}
