use std::collections::HashSet;
use std::io::{self, Read, Write};

const EPS: f64 = 1e-12;
const SCALE: f64 = 10.0;
const CELL: f64 = 2.0 * SCALE;
const WORK_THRESHOLD: u128 = 25_000_000;

#[derive(Clone, Copy, Debug)]
struct Pt2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Pt3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Copy, Debug)]
struct Geometry {
    e1: Pt3,
    e2: Pt3,
    d: Pt3,
    t1: Pt2,
    t2: Pt2,
    y_scale: f64,
}

impl Geometry {
    fn new(sx: f64, sy: f64, sz: f64) -> Self {
        let norm = (sx * sx + sy * sy + sz * sz).sqrt();
        let d = Pt3 {
            x: sx / norm,
            y: sy / norm,
            z: sz / norm,
        };

        let horiz = (d.x * d.x + d.y * d.y).sqrt();
        if horiz > EPS {
            let e1 = Pt3 {
                x: -d.y / horiz,
                y: d.x / horiz,
                z: 0.0,
            };
            let e2 = cross3(d, e1);
            let t1 = Pt2 { x: e1.x, y: e1.y };
            let t2 = Pt2 {
                x: -d.x / horiz,
                y: -d.y / horiz,
            };
            Self {
                e1,
                e2,
                d,
                t1,
                t2,
                y_scale: d.z,
            }
        } else {
            let sign_z = if d.z >= 0.0 { 1.0 } else { -1.0 };
            let e1 = Pt3 {
                x: 1.0,
                y: 0.0,
                z: 0.0,
            };
            let e2 = Pt3 {
                x: 0.0,
                y: sign_z,
                z: 0.0,
            };
            let t1 = Pt2 { x: 1.0, y: 0.0 };
            let t2 = Pt2 {
                x: 0.0,
                y: sign_z,
            };
            Self {
                e1,
                e2,
                d,
                t1,
                t2,
                y_scale: 1.0,
            }
        }
    }

    fn ground_to_q(&self, p: Pt2) -> Pt2 {
        let x = dot2(p, self.t1);
        let y = dot2(p, self.t2);
        let u = x;
        let v = self.y_scale * y;
        Pt2 {
            x: (u + v + SCALE) / CELL,
            y: (u - v + SCALE) / CELL,
        }
    }

    fn cell_center_to_world(&self, i: i64, j: i64) -> Pt3 {
        let u = SCALE * ((i + j) as f64);
        let v = SCALE * ((i - j) as f64);
        Pt3 {
            x: u * self.e1.x + v * self.e2.x,
            y: u * self.e1.y + v * self.e2.y,
            z: u * self.e1.z + v * self.e2.z,
        }
    }

    fn quaternion(&self) -> [f64; 4] {
        let m00 = self.e1.x;
        let m01 = self.e2.x;
        let m02 = self.d.x;
        let m10 = self.e1.y;
        let m11 = self.e2.y;
        let m12 = self.d.y;
        let m20 = self.e1.z;
        let m21 = self.e2.z;
        let m22 = self.d.z;

        let (w, x, y, z) = if m00 + m11 + m22 > 0.0 {
            let s = (m00 + m11 + m22 + 1.0).sqrt() * 2.0;
            (
                0.25 * s,
                (m21 - m12) / s,
                (m02 - m20) / s,
                (m10 - m01) / s,
            )
        } else if m00 > m11 && m00 > m22 {
            let s = (1.0 + m00 - m11 - m22).sqrt() * 2.0;
            (
                (m21 - m12) / s,
                0.25 * s,
                (m01 + m10) / s,
                (m02 + m20) / s,
            )
        } else if m11 > m22 {
            let s = (1.0 + m11 - m00 - m22).sqrt() * 2.0;
            (
                (m02 - m20) / s,
                (m01 + m10) / s,
                0.25 * s,
                (m12 + m21) / s,
            )
        } else {
            let s = (1.0 + m22 - m00 - m11).sqrt() * 2.0;
            (
                (m10 - m01) / s,
                (m02 + m20) / s,
                (m12 + m21) / s,
                0.25 * s,
            )
        };

        let len = (w * w + x * x + y * y + z * z).sqrt();
        [w / len, x / len, y / len, z / len]
    }
}

fn dot2(a: Pt2, b: Pt2) -> f64 {
    a.x * b.x + a.y * b.y
}

fn cross3(a: Pt3, b: Pt3) -> Pt3 {
    Pt3 {
        x: a.y * b.z - a.z * b.y,
        y: a.z * b.x - a.x * b.z,
        z: a.x * b.y - a.y * b.x,
    }
}

fn point_on_segment(p: Pt2, a: Pt2, b: Pt2) -> bool {
    let cross = (b.x - a.x) * (p.y - a.y) - (b.y - a.y) * (p.x - a.x);
    if cross.abs() > 1e-10 {
        return false;
    }
    let dot = (p.x - a.x) * (p.x - b.x) + (p.y - a.y) * (p.y - b.y);
    dot <= 1e-10
}

fn point_in_polygon(p: Pt2, poly: &[Pt2]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(p, a, b) {
            return true;
        }
        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x
                < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn clip(p: f64, q: f64, t0: &mut f64, t1: &mut f64) -> bool {
    if p.abs() < EPS {
        return q >= -EPS;
    }
    let r = q / p;
    if p < 0.0 {
        if r > *t1 + EPS {
            return false;
        }
        if r > *t0 {
            *t0 = r;
        }
    } else {
        if r < *t0 - EPS {
            return false;
        }
        if r < *t1 {
            *t1 = r;
        }
    }
    true
}

fn segment_intersects_square(a: Pt2, b: Pt2, i: i64, j: i64) -> bool {
    let min_x = i as f64;
    let max_x = min_x + 1.0;
    let min_y = j as f64;
    let max_y = min_y + 1.0;
    let dx = b.x - a.x;
    let dy = b.y - a.y;
    let mut t0 = 0.0;
    let mut t1 = 1.0;
    if !clip(-dx, a.x - min_x, &mut t0, &mut t1) {
        return false;
    }
    if !clip(dx, max_x - a.x, &mut t0, &mut t1) {
        return false;
    }
    if !clip(-dy, a.y - min_y, &mut t0, &mut t1) {
        return false;
    }
    if !clip(dy, max_y - a.y, &mut t0, &mut t1) {
        return false;
    }
    t0 <= t1 + EPS
}

fn cell_intersects_polygon(i: i64, j: i64, poly: &[Pt2]) -> bool {
    let min_x = i as f64;
    let max_x = min_x + 1.0;
    let min_y = j as f64;
    let max_y = min_y + 1.0;

    for &p in poly {
        if p.x >= min_x - EPS && p.x <= max_x + EPS && p.y >= min_y - EPS && p.y <= max_y + EPS
        {
            return true;
        }
    }

    let corners = [
        Pt2 { x: min_x, y: min_y },
        Pt2 { x: max_x, y: min_y },
        Pt2 { x: max_x, y: max_y },
        Pt2 { x: min_x, y: max_y },
    ];
    for &c in &corners {
        if point_in_polygon(c, poly) {
            return true;
        }
    }

    for k in 0..poly.len() {
        let a = poly[k];
        let b = poly[(k + 1) % poly.len()];
        if segment_intersects_square(a, b, i, j) {
            return true;
        }
    }
    false
}

fn collect_exact_cells(poly: &[Pt2], imin: i64, imax: i64, jmin: i64, jmax: i64) -> Vec<(i64, i64)> {
    let mut set = HashSet::new();
    for j in jmin..=jmax {
        for i in imin..=imax {
            if cell_intersects_polygon(i, j, poly) {
                set.insert((i, j));
            }
        }
    }
    let mut cells: Vec<(i64, i64)> = set.into_iter().collect();
    cells.sort_unstable_by(|a, b| a.1.cmp(&b.1).then(a.0.cmp(&b.0)));
    cells
}

fn fill_bbox_count(imin: i64, imax: i64, jmin: i64, jmax: i64) -> u128 {
    let width = (imax - imin + 1) as u128;
    let height = (jmax - jmin + 1) as u128;
    width * height
}

fn parse_bbox(poly: &[Pt2]) -> (i64, i64, i64, i64) {
    let mut min_x = poly[0].x;
    let mut max_x = poly[0].x;
    let mut min_y = poly[0].y;
    let mut max_y = poly[0].y;
    for &p in &poly[1..] {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    let imin = min_x.floor() as i64 - 1;
    let imax = max_x.floor() as i64 + 1;
    let jmin = min_y.floor() as i64 - 1;
    let jmax = max_y.floor() as i64 + 1;
    (imin, imax, jmin, jmax)
}

struct Scanner<'a> {
    bytes: &'a [u8],
    idx: usize,
}

impl<'a> Scanner<'a> {
    fn new(s: &'a str) -> Self {
        Self {
            bytes: s.as_bytes(),
            idx: 0,
        }
    }

    fn next<T: std::str::FromStr>(&mut self) -> T {
        while self.idx < self.bytes.len() && self.bytes[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.bytes.len() && !self.bytes[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        std::str::from_utf8(&self.bytes[start..self.idx])
            .ok()
            .and_then(|s| s.parse().ok())
            .expect("failed to parse input token")
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut sc = Scanner::new(&input);

    let n: usize = sc.next();
    let mut poly_world = Vec::with_capacity(n);
    for _ in 0..n {
        let x: f64 = sc.next();
        let y: f64 = sc.next();
        poly_world.push(Pt2 { x, y });
    }
    let sx: f64 = sc.next();
    let sy: f64 = sc.next();
    let sz: f64 = sc.next();

    let geom = Geometry::new(sx, sy, sz);
    let quat = geom.quaternion();

    let mut poly_q = Vec::with_capacity(n);
    for &p in &poly_world {
        poly_q.push(geom.ground_to_q(p));
    }

    let (imin, imax, jmin, jmax) = parse_bbox(&poly_q);
    let width = (imax - imin + 1) as u128;
    let height = (jmax - jmin + 1) as u128;
    let work = width * height * (n as u128);

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    if work <= WORK_THRESHOLD {
        let mut cells = collect_exact_cells(&poly_q, imin, imax, jmin, jmax);
        if cells.is_empty() {
            let count = fill_bbox_count(imin, imax, jmin, jmax);
            writeln!(out, "{}", count).unwrap();
            for j in jmin..=jmax {
                for i in imin..=imax {
                    let p = geom.cell_center_to_world(i, j);
                    writeln!(
                        out,
                        "{} {} {} {} {} {} {} {}",
                        p.x, p.y, p.z, quat[0], quat[1], quat[2], quat[3], SCALE
                    )
                    .unwrap();
                }
            }
            return;
        }

        writeln!(out, "{}", cells.len()).unwrap();
        for (i, j) in cells.drain(..) {
            let p = geom.cell_center_to_world(i, j);
            writeln!(
                out,
                "{} {} {} {} {} {} {} {}",
                p.x, p.y, p.z, quat[0], quat[1], quat[2], quat[3], SCALE
            )
            .unwrap();
        }
    } else {
        let count = fill_bbox_count(imin, imax, jmin, jmax);
        writeln!(out, "{}", count).unwrap();
        for j in jmin..=jmax {
            for i in imin..=imax {
                let p = geom.cell_center_to_world(i, j);
                writeln!(
                    out,
                    "{} {} {} {} {} {} {} {}",
                    p.x, p.y, p.z, quat[0], quat[1], quat[2], quat[3], SCALE
                )
                .unwrap();
            }
        }
    }
}
