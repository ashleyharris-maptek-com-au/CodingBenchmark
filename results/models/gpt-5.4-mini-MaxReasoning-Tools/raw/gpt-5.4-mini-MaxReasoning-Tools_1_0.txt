#include <algorithm>
#include <bit>
#include <cmath>
#include <cstdint>
#include <cstdio>

#include <numeric>

using std::vector;

struct Point {
    double x = 0.0;
    double y = 0.0;
};

struct Stats {
    double minx = 0.0;
    double maxx = 0.0;
    double miny = 0.0;
    double maxy = 0.0;
    double sx = 0.0;
    double sy = 0.0;
    double sxx = 0.0;
    double syy = 0.0;
    double sxy = 0.0;
};

static inline double sqDist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist(const Point& a, const Point& b) {
    return std::sqrt(sqDist(a, b));
}

static inline uint64_t doubleToSortable(double v) {
    uint64_t u = std::bit_cast<uint64_t>(v);
    return (u & 0x8000000000000000ULL) ? ~u : (u ^ 0x8000000000000000ULL);
}

static inline uint32_t part1By1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}

static inline uint64_t morton16(uint32_t x, uint32_t y) {
    return static_cast<uint64_t>(part1By1(x)) | (static_cast<uint64_t>(part1By1(y)) << 1);
}

static inline int clampU16(int v) {
    if (v < 0) return 0;
    if (v > 65535) return 65535;
    return v;
}

class FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    std::vector<char> buf;
    size_t idx = 0;
    size_t size = 0;

    inline char getChar() {
        if (idx >= size) {
            size = std::fread(buf.data(), 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

public:
    FastScanner() : buf(BUFSIZE) {}

    bool readInt(int& out) {
        char c;
        do {
            c = getChar();
            if (!c) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        }

        int val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = getChar();
        }
        out = val * sign;
        return true;
    }

    bool readDouble(double& out) {
        char c;
        do {
            c = getChar();
            if (!c) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '+' || c == '-') {
            sign = (c == '-') ? -1 : 1;
            c = getChar();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + double(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double frac = 0.0;
            double base = 1.0;
            c = getChar();
            while (c >= '0' && c <= '9') {
                frac = frac * 10.0 + double(c - '0');
                base *= 10.0;
                c = getChar();
            }
            val += frac / base;
        }

        if (c == 'e' || c == 'E') {
            c = getChar();
            int esign = 1;
            if (c == '+' || c == '-') {
                esign = (c == '-') ? -1 : 1;
                c = getChar();
            }
            int exp = 0;
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = getChar();
            }
            if (exp != 0) {
                val *= std::pow(10.0, double(esign * exp));
            }
        }

        out = (sign < 0) ? -val : val;
        return true;
    }
};

class FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    std::vector<char> buf;
    size_t idx = 0;

public:
    FastOutput() : buf(BUFSIZE) {}

    ~FastOutput() {
        flush();
    }

    inline void pushChar(char c) {
        if (idx == BUFSIZE) flush();
        buf[idx++] = c;
    }

    void writeInt(int x) {
        if (x == 0) {
            pushChar('0');
            return;
        }
        if (x < 0) {
            pushChar('-');
            x = -x;
        }
        char s[16];
        int n = 0;
        while (x > 0) {
            s[n++] = char('0' + (x % 10));
            x /= 10;
        }
        while (n--) pushChar(s[n]);
    }

    void flush() {
        if (idx) {
            std::fwrite(buf.data(), 1, idx, stdout);
            idx = 0;
        }
    }
};

static void radixSortByKey(vector<int>& ord, const vector<uint64_t>& key) {
    const int BITS = 16;
    const int BUCKETS = 1 << BITS;
    const uint64_t MASK = BUCKETS - 1;
    vector<int> tmp(ord.size());
    vector<int> cnt(BUCKETS);

    for (int shift = 0; shift < 64; shift += BITS) {
        std::fill(cnt.begin(), cnt.end(), 0);
        for (int idx : ord) {
            ++cnt[(key[idx] >> shift) & MASK];
        }
        int sum = 0;
        for (int i = 0; i < BUCKETS; ++i) {
            int c = cnt[i];
            cnt[i] = sum;
            sum += c;
        }
        for (int idx : ord) {
            int b = (key[idx] >> shift) & MASK;
            tmp[cnt[b]++] = idx;
        }
        ord.swap(tmp);
    }
}

static inline double routeLength(const vector<int>& ord, const vector<Point>& pts) {
    int n = (int)ord.size();
    if (n <= 1) return 0.0;
    double total = 0.0;
    for (int i = 0; i < n; ++i) {
        total += dist(pts[ord[i]], pts[ord[(i + 1) % n]]);
    }
    return total;
}

static void rotateToZero(vector<int>& ord) {
    auto it = std::find(ord.begin(), ord.end(), 0);
    if (it != ord.end()) {
        std::rotate(ord.begin(), it, ord.end());
    }
}

static void orientShortFirst(vector<int>& ord, const vector<Point>& pts) {
    int n = (int)ord.size();
    if (n <= 2) return;
    double d0 = sqDist(pts[ord[0]], pts[ord[1]]);
    double d1 = sqDist(pts[ord[0]], pts[ord[n - 1]]);
    if (d1 < d0) {
        std::reverse(ord.begin() + 1, ord.end());
    }
}

static void twoOptLimited(vector<int>& ord, const vector<Point>& pts, int window, int passes) {
    int n = (int)ord.size();
    if (n < 4 || window <= 0 || passes <= 0) return;
    window = std::min(window, n - 2);
    if (window <= 0) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 1; i + 1 < n; ++i) {
            int a = ord[i - 1];
            int b = ord[i];
            double ab = dist(pts[a], pts[b]);
            int limit = std::min(n - 1, i + window);
            bool changedHere = false;
            for (int j = i + 1; j <= limit; ++j) {
                int c = ord[j];
                int d = ord[(j + 1) % n];
                double current = ab + dist(pts[c], pts[d]);
                double proposed = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
                if (proposed + 1e-12 < current) {
                    std::reverse(ord.begin() + i, ord.begin() + j + 1);
                    improved = true;
                    changedHere = true;
                    break;
                }
            }
            if (changedHere) continue;
        }
        if (!improved) break;
    }
}

static vector<int> buildNearestNeighborOrder(const vector<Point>& pts) {
    int n = (int)pts.size();
    vector<int> ord(n);
    vector<char> used(n, 0);
    ord[0] = 0;
    used[0] = 1;
    for (int step = 1; step < n; ++step) {
        int prev = ord[step - 1];
        const Point& a = pts[prev];
        int best = -1;
        double bestD = std::numeric_limits<double>::infinity();
        for (int j = 0; j < n; ++j) {
            if (used[j]) continue;
            double dx = a.x - pts[j].x;
            double dy = a.y - pts[j].y;
            double d2 = dx * dx + dy * dy;
            if (d2 < bestD) {
                bestD = d2;
                best = j;
            }
        }
        ord[step] = best;
        used[best] = 1;
    }
    return ord;
}

static vector<int> buildProjectionOrder(const vector<Point>& pts, double ux, double uy) {
    int n = (int)pts.size();
    vector<uint64_t> key(n);
    for (int i = 0; i < n; ++i) {
        double proj = pts[i].x * ux + pts[i].y * uy;
        key[i] = doubleToSortable(proj);
    }
    vector<int> ord(n);
    std::iota(ord.begin(), ord.end(), 0);
    radixSortByKey(ord, key);
    return ord;
}

static vector<int> buildMortonOrder(const vector<Point>& pts, const Stats& st) {
    int n = (int)pts.size();
    vector<uint64_t> key(n);
    double w = st.maxx - st.minx;
    double h = st.maxy - st.miny;
    double invx = (w > 0.0) ? (65535.0 / w) : 0.0;
    double invy = (h > 0.0) ? (65535.0 / h) : 0.0;
    for (int i = 0; i < n; ++i) {
        int xi = 0;
        int yi = 0;
        if (w > 0.0) {
            xi = clampU16((int)((pts[i].x - st.minx) * invx + 0.5));
        }
        if (h > 0.0) {
            yi = clampU16((int)((pts[i].y - st.miny) * invy + 0.5));
        }
        key[i] = morton16((uint32_t)xi, (uint32_t)yi);
    }
    vector<int> ord(n);
    std::iota(ord.begin(), ord.end(), 0);
    radixSortByKey(ord, key);
    return ord;
}

static std::pair<std::pair<double, double>, double> principalAxis(const Stats& st, int n) {
    double mx = st.sx / n;
    double my = st.sy / n;
    double cxx = st.sxx / n - mx * mx;
    double cyy = st.syy / n - my * my;
    double cxy = st.sxy / n - mx * my;

    double disc = std::sqrt(std::max(0.0, (cxx - cyy) * (cxx - cyy) + 4.0 * cxy * cxy));
    double lambda1 = 0.5 * (cxx + cyy + disc);
    double lambda2 = 0.5 * (cxx + cyy - disc);
    double elong = (lambda2 > 1e-18) ? (lambda1 / lambda2) : ((lambda1 > 1e-18) ? 1e300 : 1.0);

    double ux = 1.0;
    double uy = 0.0;
    if (std::abs(cxx - cyy) > 1e-18 || std::abs(cxy) > 1e-18) {
        double theta = 0.5 * std::atan2(2.0 * cxy, cxx - cyy);
        ux = std::cos(theta);
        uy = std::sin(theta);
    }
    return {{ux, uy}, elong};
}

static vector<int> exactHeldKarp(const vector<Point>& pts) {
    int n = (int)pts.size();
    if (n <= 1) return {0};
    if (n == 2) return {0, 1};

    int m = n - 1;
    int totalMasks = 1 << m;
    const double INF = 1e300;

    vector<double> distMat((size_t)n * n, 0.0);
    for (int i = 0; i < n; ++i) {
        for (int j = 0; j < n; ++j) {
            distMat[(size_t)i * n + j] = dist(pts[i], pts[j]);
        }
    }

    vector<double> dp((size_t)totalMasks * m, INF);
    vector<uint8_t> parent((size_t)totalMasks * m, 255);
    auto id = [m](int mask, int last) -> size_t {
        return (size_t)mask * m + (size_t)last;
    };

    for (int j = 0; j < m; ++j) {
        dp[id(1 << j, j)] = distMat[(size_t)0 * n + (j + 1)];
        parent[id(1 << j, j)] = 255;
    }

    for (int mask = 1; mask < totalMasks; ++mask) {
        for (int last = 0; last < m; ++last) {
            if ((mask & (1 << last)) == 0) continue;
            double cur = dp[id(mask, last)];
            if (cur >= INF / 2) continue;
            int from = last + 1;
            for (int nxt = 0; nxt < m; ++nxt) {
                if (mask & (1 << nxt)) continue;
                int nmask = mask | (1 << nxt);
                double cand = cur + distMat[(size_t)from * n + (nxt + 1)];
                size_t pos = id(nmask, nxt);
                if (cand < dp[pos]) {
                    dp[pos] = cand;
                    parent[pos] = (uint8_t)last;
                }
            }
        }
    }

    int full = totalMasks - 1;
    double best = INF;
    int bestLast = -1;
    for (int last = 0; last < m; ++last) {
        double cand = dp[id(full, last)] + distMat[(size_t)(last + 1) * n + 0];
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }

    vector<int> rev;
    int mask = full;
    int last = bestLast;
    while (true) {
        rev.push_back(last + 1);
        size_t pos = id(mask, last);
        uint8_t prev = parent[pos];
        mask ^= (1 << last);
        if (prev == 255) break;
        last = (int)prev;
    }
    std::reverse(rev.begin(), rev.end());

    vector<int> order;
    order.reserve(n);
    order.push_back(0);
    order.insert(order.end(), rev.begin(), rev.end());
    return order;
}

static vector<int> solveHeuristic(const vector<Point>& pts, const Stats& st) {
    int n = (int)pts.size();
    auto [axis, elong] = principalAxis(st, n);
    double ux = axis.first;
    double uy = axis.second;

    int window = 0;
    int passes = 0;
    if (n <= 1000) {
        window = 64;
        passes = 3;
    } else if (n <= 5000) {
        window = 48;
        passes = 2;
    } else if (n <= 200000) {
        window = 12;
        passes = 1;
    }

    auto evaluate = [&](vector<int> ord, bool doImprove) -> std::pair<vector<int>, double> {
        rotateToZero(ord);
        orientShortFirst(ord, pts);
        if (doImprove && window > 0) {
            twoOptLimited(ord, pts, window, passes);
        }
        double len = routeLength(ord, pts);
        return {std::move(ord), len};
    };

    if (n <= 5000) {
        vector<int> nn = buildNearestNeighborOrder(pts);
        auto best = evaluate(std::move(nn), true);

        auto cand1 = evaluate(buildProjectionOrder(pts, ux, uy), true);
        if (cand1.second < best.second) best = std::move(cand1);

        auto cand2 = evaluate(buildMortonOrder(pts, st), true);
        if (cand2.second < best.second) best = std::move(cand2);

        return best.first;
    }

    auto candidateProjection = buildProjectionOrder(pts, ux, uy);
    auto candidateMorton = buildMortonOrder(pts, st);

    if (n <= 200000) {
        auto best = evaluate(std::move(candidateProjection), true);
        auto cand2 = evaluate(std::move(candidateMorton), true);
        if (cand2.second < best.second) best = std::move(cand2);
        return best.first;
    }

    bool useProjection = (elong > 4.0);
    if (useProjection) {
        return evaluate(std::move(candidateProjection), false).first;
    }
    return evaluate(std::move(candidateMorton), false).first;
}

int main() {
    FastScanner fs;
    FastOutput fo;

    int n = 0;
    if (!fs.readInt(n)) return 0;
    if (n <= 0) return 0;

    vector<Point> pts(n);
    Stats st;
    st.minx = st.miny = std::numeric_limits<double>::infinity();
    st.maxx = st.maxy = -std::numeric_limits<double>::infinity();

    for (int i = 0; i < n; ++i) {
        double x, y;
        fs.readDouble(x);
        fs.readDouble(y);
        pts[i] = {x, y};

        st.minx = std::min(st.minx, x);
        st.maxx = std::max(st.maxx, x);
        st.miny = std::min(st.miny, y);
        st.maxy = std::max(st.maxy, y);
        st.sx += x;
        st.sy += y;
        st.sxx += x * x;
        st.syy += y * y;
        st.sxy += x * y;
    }

    vector<int> order;
    if (n <= 20) {
        order = exactHeldKarp(pts);
    } else {
        order = solveHeuristic(pts, st);
    }

    for (int i = 0; i < n; ++i) {
        if (i) fo.pushChar(' ');
        fo.writeInt(order[i]);
    }
    fo.pushChar('\n');
    fo.flush();
    return 0;
}
