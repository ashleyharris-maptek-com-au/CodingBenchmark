// Travelling Salesman heuristic solver with adaptive strategies.
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <cstring>
#include <limits>
#include <numeric>
#include <memory>
#include <vector>

struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 20;
    std::unique_ptr<char[]> buf;
    size_t pos = 0;
    size_t len = 0;

    FastScanner() : buf(new char[BUFSIZE]) {}

    inline int getChar() {
        if (pos >= len) {
            len = std::fread(buf.get(), 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return -1;
        }
        return static_cast<unsigned char>(buf[pos++]);
    }

    inline void unread() {
        --pos;
    }

    bool readU64(uint64_t& out) {
        int c;
        do {
            c = getChar();
            if (c == -1) return false;
        } while (c <= ' ');

        uint64_t value = 0;
        while (c > ' ') {
            value = value * 10 + static_cast<uint64_t>(c - '0');
            c = getChar();
        }
        if (c != -1) unread();
        out = value;
        return true;
    }

    static inline double pow10i(int exp) {
        if (exp == 0) return 1.0;
        bool neg = exp < 0;
        unsigned int e = static_cast<unsigned int>(neg ? -exp : exp);
        double result = 1.0;
        double base = 10.0;
        while (e) {
            if (e & 1u) result *= base;
            base *= base;
            e >>= 1u;
        }
        return neg ? 1.0 / result : result;
    }

    bool readDouble(double& out) {
        int c;
        do {
            c = getChar();
            if (c == -1) return false;
        } while (c <= ' ');

        int sign = 1;
        if (c == '+' || c == '-') {
            if (c == '-') sign = -1;
            c = getChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double frac = 0.0;
            double base = 1.0;
            c = getChar();
            while (c >= '0' && c <= '9') {
                frac = frac * 10.0 + static_cast<double>(c - '0');
                base *= 10.0;
                c = getChar();
            }
            value += frac / base;
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int exp = 0;
            c = getChar();
            if (c == '+' || c == '-') {
                if (c == '-') expSign = -1;
                c = getChar();
            }
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = getChar();
            }
            value *= pow10i(expSign * exp);
        }

        if (c != -1) unread();
        out = sign * value;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    std::unique_ptr<char[]> buf;
    size_t pos = 0;

    FastOutput() : buf(new char[BUFSIZE]) {}

    ~FastOutput() {
        flush();
    }

    inline void flush() {
        if (pos) {
            std::fwrite(buf.get(), 1, pos, stdout);
            pos = 0;
        }
    }

    inline void putChar(char c) {
        if (pos >= BUFSIZE) flush();
        buf.get()[pos++] = c;
    }

    inline void writeU32(uint32_t value) {
        char tmp[16];
        int n = 0;
        do {
            tmp[n++] = static_cast<char>('0' + (value % 10u));
            value /= 10u;
        } while (value != 0u);
        while (n--) putChar(tmp[n]);
    }
};

static inline double euclidDist(const double* x, const double* y, uint32_t a, uint32_t b) {
    double dx = x[a] - x[b];
    double dy = y[a] - y[b];
    return std::sqrt(dx * dx + dy * dy);
}

static inline double sqDist(const double* x, const double* y, uint32_t a, uint32_t b) {
    double dx = x[a] - x[b];
    double dy = y[a] - y[b];
    return dx * dx + dy * dy;
}

static std::vector<uint32_t> exactRoute(const std::vector<double>& x, const std::vector<double>& y) {
    const size_t n = x.size();
    std::vector<uint32_t> bestRoute(n);
    bestRoute[0] = 0u;
    if (n == 1) return bestRoute;

    std::vector<double> dist(n * n, 0.0);
    for (size_t i = 0; i < n; ++i) {
        for (size_t j = i + 1; j < n; ++j) {
            double d = euclidDist(x.data(), y.data(), static_cast<uint32_t>(i), static_cast<uint32_t>(j));
            dist[i * n + j] = d;
            dist[j * n + i] = d;
        }
    }

    std::vector<uint32_t> perm(n - 1);
    std::iota(perm.begin(), perm.end(), 1u);

    double bestLen = std::numeric_limits<double>::infinity();
    do {
        double len = dist[0 * n + perm[0]];
        for (size_t i = 0; i + 1 < perm.size(); ++i) {
            len += dist[perm[i] * n + perm[i + 1]];
        }
        len += dist[perm.back() * n + 0];
        if (len < bestLen) {
            bestLen = len;
            bestRoute[0] = 0u;
            std::copy(perm.begin(), perm.end(), bestRoute.begin() + 1);
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    return bestRoute;
}

static std::vector<uint32_t> nearestNeighborRoute(const std::vector<double>& x, const std::vector<double>& y) {
    const size_t n = x.size();
    std::vector<uint32_t> route(n);
    route[0] = 0u;
    if (n == 1) return route;

    std::vector<uint32_t> remaining(n - 1);
    std::iota(remaining.begin(), remaining.end(), 1u);
    size_t remainingSize = remaining.size();
    uint32_t cur = 0u;
    const double* px = x.data();
    const double* py = y.data();

    for (size_t pos = 1; pos < n; ++pos) {
        size_t bestPos = 0;
        double bestD = std::numeric_limits<double>::infinity();
        const double cx = px[cur];
        const double cy = py[cur];
        for (size_t i = 0; i < remainingSize; ++i) {
            uint32_t nxt = remaining[i];
            double dx = cx - px[nxt];
            double dy = cy - py[nxt];
            double d = dx * dx + dy * dy;
            if (d < bestD) {
                bestD = d;
                bestPos = i;
            }
        }
        uint32_t chosen = remaining[bestPos];
        route[pos] = chosen;
        remaining[bestPos] = remaining[remainingSize - 1];
        --remainingSize;
        cur = chosen;
    }

    return route;
}

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static void radixSortIndices(std::vector<uint32_t>& order, std::vector<uint32_t>& tmp, const std::vector<uint64_t>& keys) {
    constexpr int BITS = 16;
    constexpr uint32_t MASK = (1u << BITS) - 1u;
    std::vector<uint32_t> cnt(1u << BITS);

    for (int shift = 0; shift < 64; shift += BITS) {
        std::fill(cnt.begin(), cnt.end(), 0u);
        for (uint32_t id : order) {
            ++cnt[static_cast<uint32_t>((keys[id] >> shift) & MASK)];
        }

        uint32_t sum = 0;
        for (size_t i = 0; i < cnt.size(); ++i) {
            uint32_t c = cnt[i];
            cnt[i] = sum;
            sum += c;
        }

        for (uint32_t id : order) {
            tmp[cnt[static_cast<uint32_t>((keys[id] >> shift) & MASK)]++] = id;
        }
        order.swap(tmp);
    }
}

static std::vector<uint32_t> mortonRoute(std::vector<double>& x, std::vector<double>& y) {
    const size_t n = x.size();
    std::vector<uint64_t> keys(n);

    double minx = x[0], maxx = x[0];
    double miny = y[0], maxy = y[0];
    for (size_t i = 1; i < n; ++i) {
        if (x[i] < minx) minx = x[i];
        if (x[i] > maxx) maxx = x[i];
        if (y[i] < miny) miny = y[i];
        if (y[i] > maxy) maxy = y[i];
    }

    constexpr uint32_t GRID_MAX = 0x7FFFFFFFu;
    const double scaleX = (maxx > minx) ? (static_cast<double>(GRID_MAX) / (maxx - minx)) : 0.0;
    const double scaleY = (maxy > miny) ? (static_cast<double>(GRID_MAX) / (maxy - miny)) : 0.0;

    for (size_t i = 0; i < n; ++i) {
        uint32_t gx = 0u;
        uint32_t gy = 0u;
        if (scaleX > 0.0) {
            double tx = (x[i] - minx) * scaleX;
            if (tx > 0.0) {
                gx = (tx >= static_cast<double>(GRID_MAX)) ? GRID_MAX : static_cast<uint32_t>(tx);
            }
        }
        if (scaleY > 0.0) {
            double ty = (y[i] - miny) * scaleY;
            if (ty > 0.0) {
                gy = (ty >= static_cast<double>(GRID_MAX)) ? GRID_MAX : static_cast<uint32_t>(ty);
            }
        }
        keys[i] = part1by1(gx) | (part1by1(gy) << 1);
    }

    std::vector<double>().swap(x);
    std::vector<double>().swap(y);

    std::vector<uint32_t> order(n), tmp(n);
    std::iota(order.begin(), order.end(), 0u);
    radixSortIndices(order, tmp, keys);
    return order;
}

static void writeRoute(const std::vector<uint32_t>& route, FastOutput& out) {
    const size_t n = route.size();
    if (n == 0) {
        out.putChar('\n');
        return;
    }

    size_t start = 0;
    while (start < n && route[start] != 0u) ++start;
    if (start == n) start = 0;

    size_t printed = 0;
    for (size_t i = start; i < n; ++i) {
        if (printed++) out.putChar(' ');
        out.writeU32(route[i]);
    }
    for (size_t i = 0; i < start; ++i) {
        if (printed++) out.putChar(' ');
        out.writeU32(route[i]);
    }
    out.putChar('\n');
}

int main() {
    FastScanner in;
    FastOutput out;

    uint64_t n64 = 0;
    if (!in.readU64(n64)) return 0;
    const size_t n = static_cast<size_t>(n64);

    if (n == 0) {
        out.putChar('\n');
        out.flush();
        return 0;
    }

    std::vector<double> x(n), y(n);
    for (size_t i = 0; i < n; ++i) {
        in.readDouble(x[i]);
        in.readDouble(y[i]);
    }

    std::vector<uint32_t> route;
    if (n <= 11) {
        route = exactRoute(x, y);
    } else if (n <= 10000) {
        route = nearestNeighborRoute(x, y);
    } else {
        route = mortonRoute(x, y);
    }

    writeRoute(route, out);
    out.flush();
    return 0;
}
