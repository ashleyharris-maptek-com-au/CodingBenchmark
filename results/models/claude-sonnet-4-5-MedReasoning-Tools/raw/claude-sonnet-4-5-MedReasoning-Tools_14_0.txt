using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int H = int.Parse(input[0]);
        int W = int.Parse(input[1]);

        char[,] board = new char[H, W];
        for (int i = 0; i < H; i++)
        {
            string line = Console.ReadLine();
            for (int j = 0; j < W; j++)
            {
                board[i, j] = j < line.Length ? line[j] : ' ';
            }
        }

        var solver = new Solver(board, H, W);
        var safeCells = solver.FindSafeCells();

        Console.WriteLine(safeCells.Count);
        foreach (var cell in safeCells.OrderBy(c => c.Item2).ThenBy(c => c.Item1))
        {
            Console.WriteLine($"{cell.Item1} {cell.Item2}");
        }
    }
}

class Solver
{
    private char[,] board;
    private int H, W;
    private HashSet<(int, int)> knownMines;
    private HashSet<(int, int)> knownSafe;
    private bool[,] isUnknown;

    public Solver(char[,] board, int H, int W)
    {
        this.board = board;
        this.H = H;
        this.W = W;
        this.knownMines = new HashSet<(int, int)>();
        this.knownSafe = new HashSet<(int, int)>();
        this.isUnknown = new bool[H, W];

        for (int i = 0; i < H; i++)
        {
            for (int j = 0; j < W; j++)
            {
                if (board[i, j] == '*')
                    knownMines.Add((j, i));
                else if (board[i, j] == ' ')
                    isUnknown[i, j] = true;
            }
        }
    }

    public List<(int, int)> FindSafeCells()
    {
        bool changed = true;
        int iterations = 0;
        int maxIterations = Math.Min(1000, H * W);

        while (changed && iterations < maxIterations)
        {
            changed = false;
            iterations++;

            // Simple rule-based deduction
            for (int i = 0; i < H; i++)
            {
                for (int j = 0; j < W; j++)
                {
                    if (char.IsDigit(board[i, j]))
                    {
                        if (ApplySimpleRules(j, i))
                            changed = true;
                    }
                }
            }

            // For larger boards or after simple rules plateau, try constraint satisfaction
            if (!changed && iterations < 5 && (H * W > 100 || knownSafe.Count > 0))
            {
                if (ApplyConstraintSatisfaction())
                    changed = true;
            }
        }

        return knownSafe.ToList();
    }

    private bool ApplySimpleRules(int x, int y)
    {
        int mineCount = board[y, x] - '0';
        var neighbors = GetNeighbors(x, y);

        var unknownNeighbors = neighbors.Where(n => isUnknown[n.Item2, n.Item1]).ToList();
        var flaggedNeighbors = neighbors.Where(n => knownMines.Contains(n)).ToList();

        bool changed = false;

        // Rule 1: If all mines are accounted for, remaining unknowns are safe
        if (flaggedNeighbors.Count == mineCount)
        {
            foreach (var unknown in unknownNeighbors)
            {
                if (knownSafe.Add(unknown))
                {
                    isUnknown[unknown.Item2, unknown.Item1] = false;
                    changed = true;
                }
            }
        }

        // Rule 2: If unknown + flagged = mine count, all unknowns are mines
        if (unknownNeighbors.Count + flaggedNeighbors.Count == mineCount)
        {
            foreach (var unknown in unknownNeighbors)
            {
                if (knownMines.Add(unknown))
                {
                    isUnknown[unknown.Item2, unknown.Item1] = false;
                    changed = true;
                }
            }
        }

        return changed;
    }

    private bool ApplyConstraintSatisfaction()
    {
        // Build constraints from numbered cells
        var constraints = new List<Constraint>();
        var allUnknownCells = new HashSet<(int, int)>();

        for (int i = 0; i < H; i++)
        {
            for (int j = 0; j < W; j++)
            {
                if (char.IsDigit(board[i, j]))
                {
                    int mineCount = board[i, j] - '0';
                    var neighbors = GetNeighbors(j, i);
                    var unknownNeighbors = neighbors.Where(n => isUnknown[n.Item2, n.Item1]).ToHashSet();
                    var flaggedCount = neighbors.Count(n => knownMines.Contains(n));

                    if (unknownNeighbors.Count > 0)
                    {
                        int remainingMines = mineCount - flaggedCount;
                        constraints.Add(new Constraint(unknownNeighbors, remainingMines));
                        allUnknownCells.UnionWith(unknownNeighbors);
                    }
                }
            }
        }

        if (constraints.Count == 0 || allUnknownCells.Count == 0)
            return false;

        // Subset reduction: find implications
        bool changed = false;
        for (int i = 0; i < constraints.Count; i++)
        {
            for (int j = 0; j < constraints.Count; j++)
            {
                if (i == j) continue;

                var c1 = constraints[i];
                var c2 = constraints[j];

                // If c1 is subset of c2
                if (c1.Cells.IsSubsetOf(c2.Cells))
                {
                    var diff = new HashSet<(int, int)>(c2.Cells);
                    diff.ExceptWith(c1.Cells);
                    int mineDiff = c2.MineCount - c1.MineCount;

                    if (diff.Count > 0)
                    {
                        // All cells in diff are safe
                        if (mineDiff == 0)
                        {
                            foreach (var cell in diff)
                            {
                                if (knownSafe.Add(cell))
                                {
                                    isUnknown[cell.Item2, cell.Item1] = false;
                                    changed = true;
                                }
                            }
                        }
                        // All cells in diff are mines
                        else if (mineDiff == diff.Count)
                        {
                            foreach (var cell in diff)
                            {
                                if (knownMines.Add(cell))
                                {
                                    isUnknown[cell.Item2, cell.Item1] = false;
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
        }

        // Check cells that appear in only one constraint
        var cellConstraintCount = new Dictionary<(int, int), List<Constraint>>();
        foreach (var c in constraints)
        {
            foreach (var cell in c.Cells)
            {
                if (!cellConstraintCount.ContainsKey(cell))
                    cellConstraintCount[cell] = new List<Constraint>();
                cellConstraintCount[cell].Add(c);
            }
        }

        foreach (var kvp in cellConstraintCount)
        {
            if (kvp.Value.Count == 1)
            {
                var constraint = kvp.Value[0];
                // If only one cell and must have mine
                if (constraint.Cells.Count == 1 && constraint.MineCount == 1)
                {
                    if (knownMines.Add(kvp.Key))
                    {
                        isUnknown[kvp.Key.Item2, kvp.Key.Item1] = false;
                        changed = true;
                    }
                }
                else if (constraint.Cells.Count == 1 && constraint.MineCount == 0)
                {
                    if (knownSafe.Add(kvp.Key))
                    {
                        isUnknown[kvp.Key.Item2, kvp.Key.Item1] = false;
                        changed = true;
                    }
                }
            }
        }

        return changed;
    }

    private List<(int, int)> GetNeighbors(int x, int y)
    {
        var neighbors = new List<(int, int)>();
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                int ny = y + dy;
                if (nx >= 0 && nx < W && ny >= 0 && ny < H)
                    neighbors.Add((nx, ny));
            }
        }
        return neighbors;
    }
}

class Constraint
{
    public HashSet<(int, int)> Cells { get; set; }
    public int MineCount { get; set; }

    public Constraint(HashSet<(int, int)> cells, int mineCount)
    {
        Cells = cells;
        MineCount = mineCount;
    }
}
