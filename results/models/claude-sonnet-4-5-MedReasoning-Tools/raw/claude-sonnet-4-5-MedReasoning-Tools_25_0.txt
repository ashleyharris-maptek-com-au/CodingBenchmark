using System;
using System.Linq;
using System.Collections.Generic;

class AsteroidInterceptor
{
    static void Main()
    {
        // Parse input
        var line1 = Console.ReadLine().Split();
        int numBodies = int.Parse(line1[0]);
        double warningDays = double.Parse(line1[1]);
        double impactorMass = double.Parse(line1[2]);
        double deltaVBudget = double.Parse(line1[3]);
        double asteroidMass = double.Parse(line1[4]);
        double t0Unix = double.Parse(line1[5]);

        var bodies = new Dictionary<string, Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var parts = Console.ReadLine().Split();
            var body = new Body
            {
                Name = parts[0],
                Mass = double.Parse(parts[1]),
                Pos = new Vec3(double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4])),
                Vel = new Vec3(double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]))
            };
            bodies[body.Name] = body;
        }
        string bspPath = Console.ReadLine().Trim();

        var earth = bodies["Earth"];
        var asteroid = bodies["Asteroid"];

        // Constants
        const double G = 6.67430e-11;
        const double SunMass = 1.989e30;
        const double AU = 1.496e11;

        // Estimate asteroid-Earth collision time by propagating forward
        double collisionTime = EstimateCollisionTime(earth, asteroid, warningDays, SunMass, G);

        // Search for optimal launch window
        double bestLaunchTime = 0;
        double bestTotalDV = double.MaxValue;
        Vec3 bestLaunchDV = new Vec3();
        Vec3 bestImpactDV = new Vec3();
        double bestImpactTime = 0;

        int samples = Math.Min(200, (int)(warningDays * 2));
        for (int i = 0; i < samples; i++)
        {
            double launchTime = i * warningDays / samples;
            double interceptTime = launchTime + (collisionTime - launchTime) * 0.7; // Intercept before collision

            if (interceptTime <= launchTime || interceptTime > collisionTime - 10) continue;

            // Propagate bodies to launch and intercept times
            var earthLaunch = Propagate(earth, launchTime, SunMass, G);
            var earthIntercept = Propagate(earth, interceptTime, SunMass, G);
            var asteroidIntercept = Propagate(asteroid, interceptTime, SunMass, G);

            // Solve Lambert's problem for interception trajectory
            double tof = (interceptTime - launchTime) * 86400; // Convert to seconds
            var result = SolveLambert(earthLaunch.Pos, asteroidIntercept.Pos, tof, SunMass, G);

            if (result == null) continue;

            // Calculate required delta-v at launch and impact
            var launchDV = result.V1 - earthLaunch.Vel;
            var arrivalVel = result.V2;
            var impactDV = arrivalVel - asteroidIntercept.Vel;

            double totalDV = launchDV.Magnitude() + impactDV.Magnitude();

            if (totalDV < bestTotalDV && totalDV < deltaVBudget)
            {
                bestTotalDV = totalDV;
                bestLaunchTime = launchTime;
                bestLaunchDV = launchDV;
                bestImpactDV = impactDV;
                bestImpactTime = interceptTime;
            }
        }

        // Refine solution with mid-course correction if budget allows
        int numBurns = 2;
        Console.WriteLine(numBurns);
        Console.WriteLine($"{bestLaunchTime:F6} {bestLaunchDV.X:E15} {bestLaunchDV.Y:E15} {bestLaunchDV.Z:E15}");
        Console.WriteLine($"{bestImpactTime:F6} {bestImpactDV.X:E15} {bestImpactDV.Y:E15} {bestImpactDV.Z:E15}");
    }

    static double EstimateCollisionTime(Body earth, Body asteroid, double maxDays, double sunMass, double G)
    {
        double minDist = double.MaxValue;
        double minTime = maxDays;

        for (double t = 0; t <= maxDays; t += 0.5)
        {
            var e = Propagate(earth, t, sunMass, G);
            var a = Propagate(asteroid, t, sunMass, G);
            double dist = (e.Pos - a.Pos).Magnitude();
            if (dist < minDist)
            {
                minDist = dist;
                minTime = t;
            }
        }
        return minTime;
    }

    static Body Propagate(Body body, double days, double sunMass, double G)
    {
        const int steps = 1000;
        double dt = days * 86400 / steps;
        var pos = body.Pos;
        var vel = body.Vel;

        for (int i = 0; i < steps; i++)
        {
            var r = pos.Magnitude();
            var acc = pos * (-G * sunMass / (r * r * r));
            vel = vel + acc * dt;
            pos = pos + vel * dt;
        }

        return new Body { Pos = pos, Vel = vel };
    }

    static LambertResult SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu, double G)
    {
        mu = G * mu;
        double r1mag = r1.Magnitude();
        double r2mag = r2.Magnitude();
        double cosDeltaNu = Vec3.Dot(r1, r2) / (r1mag * r2mag);
        double deltaNu = Math.Acos(Math.Max(-1, Math.Min(1, cosDeltaNu)));

        double A = Math.Sin(deltaNu) * Math.Sqrt(r1mag * r2mag / (1 - cosDeltaNu));
        if (A < 1e-6) return null;

        // Universal variable formulation
        double psi = 0;
        double c2 = 0.5, c3 = 1.0 / 6.0;

        for (int iter = 0; iter < 50; iter++)
        {
            double y = r1mag + r2mag + A * (psi * c3 - 1) / Math.Sqrt(c2);
            if (y < 0) y = 0.1;

            double chi = Math.Sqrt(y / c2);
            double t = (Math.Pow(chi, 3) * c3 + A * Math.Sqrt(y)) / Math.Sqrt(mu);

            if (Math.Abs(t - tof) < 0.01) break;

            double dtdpsi = (Math.Pow(chi, 3) * (0.5 / psi * (c3 - 3 * c2 / (2 * psi)) + 3 * c3 / (4 * psi)) + A / 8 * (3 * c3 / c2 * Math.Sqrt(y) + A / chi)) / Math.Sqrt(mu);
            if (Math.Abs(dtdpsi) < 1e-10) break;

            psi = psi + (tof - t) / dtdpsi;

            if (psi > 1e-6)
            {
                c2 = (1 - Math.Cos(Math.Sqrt(psi))) / psi;
                c3 = (Math.Sqrt(psi) - Math.Sin(Math.Sqrt(psi))) / Math.Sqrt(psi * psi * psi);
            }
            else if (psi < -1e-6)
            {
                c2 = (1 - Math.Cosh(Math.Sqrt(-psi))) / psi;
                c3 = (Math.Sinh(Math.Sqrt(-psi)) - Math.Sqrt(-psi)) / Math.Sqrt(-psi * -psi * -psi);
            }
            else
            {
                c2 = 0.5;
                c3 = 1.0 / 6.0;
            }
        }

        double y_final = r1mag + r2mag + A * (psi * c3 - 1) / Math.Sqrt(c2);
        double f = 1 - y_final / r1mag;
        double g = A * Math.Sqrt(y_final / mu);
        double gdot = 1 - y_final / r2mag;

        var v1 = (r2 - r1 * f) * (1.0 / g);
        var v2 = (r2 * gdot - r1) * (1.0 / g);

        return new LambertResult { V1 = v1, V2 = v2 };
    }
}

class Body
{
    public string Name;
    public double Mass;
    public Vec3 Pos;
    public Vec3 Vel;
}

struct Vec3
{
    public double X, Y, Z;
    public Vec3(double x = 0, double y = 0, double z = 0) { X = x; Y = y; Z = z; }
    public double Magnitude() => Math.Sqrt(X * X + Y * Y + Z * Z);
    public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
    public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
}

class LambertResult
{
    public Vec3 V1, V2;
}
