using namespace std;

constexpr double EPSILON = 1e-9;

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalized() const { double len = length(); return len > EPSILON ? *this / len : *this; }
};

struct Triangle {
    int v[3];
    int meshId;
};

struct AABB {
    Vec3 min, max;

    AABB() : min(Vec3(1e30, 1e30, 1e30)), max(Vec3(-1e30, -1e30, -1e30)) {}

    void expand(const Vec3& p) {
        min.x = std::min(min.x, p.x);
        min.y = std::min(min.y, p.y);
        min.z = std::min(min.z, p.z);
        max.x = std::max(max.x, p.x);
        max.y = std::max(max.y, p.y);
        max.z = std::max(max.z, p.z);
    }

    void expand(const AABB& box) {
        expand(box.min);
        expand(box.max);
    }

    bool intersects(const AABB& box) const {
        return !(max.x < box.min.x || min.x > box.max.x ||
                 max.y < box.min.y || min.y > box.max.y ||
                 max.z < box.min.z || min.z > box.max.z);
    }

    Vec3 center() const { return (min + max) * 0.5; }
};

struct BVHNode {
    AABB box;
    int left, right;
    int triIndex;

    bool isLeaf() const { return triIndex >= 0; }
};

class BVH {
    vector<BVHNode> nodes;
    const vector<Vec3>* vertices;
    const vector<Triangle>* triangles;

    AABB computeTriangleBounds(int triIdx) const {
        AABB box;
        const Triangle& tri = (*triangles)[triIdx];
        box.expand((*vertices)[tri.v[0]]);
        box.expand((*vertices)[tri.v[1]]);
        box.expand((*vertices)[tri.v[2]]);
        return box;
    }

    int buildRecursive(vector<int>& indices, int start, int end) {
        int nodeIdx = nodes.size();
        nodes.push_back(BVHNode());

        AABB box;
        for (int i = start; i < end; i++) {
            box.expand(computeTriangleBounds(indices[i]));
        }
        nodes[nodeIdx].box = box;

        if (end - start <= 1) {
            nodes[nodeIdx].triIndex = indices[start];
            nodes[nodeIdx].left = nodes[nodeIdx].right = -1;
            return nodeIdx;
        }

        nodes[nodeIdx].triIndex = -1;

        Vec3 extent = box.max - box.min;
        int axis = extent.x > extent.y ? (extent.x > extent.z ? 0 : 2) : (extent.y > extent.z ? 1 : 2);

        sort(indices.begin() + start, indices.begin() + end, [&](int a, int b) {
            Vec3 ca = computeTriangleBounds(a).center();
            Vec3 cb = computeTriangleBounds(b).center();
            return (axis == 0 ? ca.x : (axis == 1 ? ca.y : ca.z)) <
                   (axis == 0 ? cb.x : (axis == 1 ? cb.y : cb.z));
        });

        int mid = (start + end) / 2;
        nodes[nodeIdx].left = buildRecursive(indices, start, mid);
        nodes[nodeIdx].right = buildRecursive(indices, mid, end);

        return nodeIdx;
    }

public:
    void build(const vector<Vec3>& verts, const vector<Triangle>& tris) {
        vertices = &verts;
        triangles = &tris;
        nodes.clear();

        if (tris.empty()) return;

        vector<int> indices(tris.size());
        for (size_t i = 0; i < tris.size(); i++) indices[i] = i;
        buildRecursive(indices, 0, indices.size());
    }

    void query(const AABB& queryBox, vector<int>& results) const {
        if (nodes.empty()) return;
        vector<int> stack = {0};

        while (!stack.empty()) {
            int nodeIdx = stack.back();
            stack.pop_back();

            const BVHNode& node = nodes[nodeIdx];
            if (!node.box.intersects(queryBox)) continue;

            if (node.isLeaf()) {
                results.push_back(node.triIndex);
            } else {
                stack.push_back(node.left);
                stack.push_back(node.right);
            }
        }
    }
};

bool rayTriangleIntersect(const Vec3& origin, const Vec3& dir,
                          const Vec3& v0, const Vec3& v1, const Vec3& v2,
                          double& t) {
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 pvec = dir.cross(e2);
    double det = e1.dot(pvec);

    if (abs(det) < EPSILON) return false;

    double invDet = 1.0 / det;
    Vec3 tvec = origin - v0;
    double u = tvec.dot(pvec) * invDet;
    if (u < 0 || u > 1) return false;

    Vec3 qvec = tvec.cross(e1);
    double v = dir.dot(qvec) * invDet;
    if (v < 0 || u + v > 1) return false;

    t = e2.dot(qvec) * invDet;
    return t > EPSILON;
}

bool isPointInsideMesh(const Vec3& point, const vector<Vec3>& vertices,
                       const vector<Triangle>& triangles, const BVH& bvh) {
    Vec3 rayDir(0.123456, 0.789012, 0.345678);
    rayDir = rayDir.normalized();

    int intersectionCount = 0;

    for (const auto& tri : triangles) {
        const Vec3& v0 = vertices[tri.v[0]];
        const Vec3& v1 = vertices[tri.v[1]];
        const Vec3& v2 = vertices[tri.v[2]];

        double t;
        if (rayTriangleIntersect(point, rayDir, v0, v1, v2, t)) {
            intersectionCount++;
        }
    }

    return (intersectionCount % 2) == 1;
}

struct VertexHash {
    size_t operator()(const Vec3& v) const {
        auto h1 = hash<long long>{}(llround(v.x / EPSILON));
        auto h2 = hash<long long>{}(llround(v.y / EPSILON));
        auto h3 = hash<long long>{}(llround(v.z / EPSILON));
        return h1 ^ (h2 << 1) ^ (h3 << 2);
    }
};

struct VertexEqual {
    bool operator()(const Vec3& a, const Vec3& b) const {
        return abs(a.x - b.x) < EPSILON &&
               abs(a.y - b.y) < EPSILON &&
               abs(a.z - b.z) < EPSILON;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    vector<Vec3> verticesA(VA);
    for (int i = 0; i < VA; i++) {
        cin >> verticesA[i].x >> verticesA[i].y >> verticesA[i].z;
    }

    vector<Triangle> trianglesA(FA);
    for (int i = 0; i < FA; i++) {
        cin >> trianglesA[i].v[0] >> trianglesA[i].v[1] >> trianglesA[i].v[2];
        trianglesA[i].meshId = 0;
    }

    vector<Vec3> verticesB(VB);
    for (int i = 0; i < VB; i++) {
        cin >> verticesB[i].x >> verticesB[i].y >> verticesB[i].z;
    }

    vector<Triangle> trianglesB(FB);
    for (int i = 0; i < FB; i++) {
        cin >> trianglesB[i].v[0] >> trianglesB[i].v[1] >> trianglesB[i].v[2];
        trianglesB[i].meshId = 1;
    }

    // Build BVH for both meshes
    BVH bvhA, bvhB;
    bvhA.build(verticesA, trianglesA);
    bvhB.build(verticesB, trianglesB);

    // For union: keep faces from A that are outside B, and faces from B that are outside A
    vector<Vec3> resultVertices;
    vector<Triangle> resultTriangles;
    unordered_map<Vec3, int, VertexHash, VertexEqual> vertexMap;

    auto addVertex = [&](const Vec3& v) -> int {
        auto it = vertexMap.find(v);
        if (it != vertexMap.end()) {
            return it->second;
        }
        int idx = resultVertices.size();
        resultVertices.push_back(v);
        vertexMap[v] = idx;
        return idx;
    };

    auto addTriangle = [&](const Vec3& v0, const Vec3& v1, const Vec3& v2) {
        Triangle tri;
        tri.v[0] = addVertex(v0);
        tri.v[1] = addVertex(v1);
        tri.v[2] = addVertex(v2);
        tri.meshId = -1;
        resultTriangles.push_back(tri);
    };

    // Process triangles from mesh A - keep those outside B
    for (const auto& tri : trianglesA) {
        Vec3 v0 = verticesA[tri.v[0]];
        Vec3 v1 = verticesA[tri.v[1]];
        Vec3 v2 = verticesA[tri.v[2]];

        Vec3 centroid = (v0 + v1 + v2) / 3.0;

        if (!isPointInsideMesh(centroid, verticesB, trianglesB, bvhB)) {
            addTriangle(v0, v1, v2);
        }
    }

    // Process triangles from mesh B - keep those outside A
    for (const auto& tri : trianglesB) {
        Vec3 v0 = verticesB[tri.v[0]];
        Vec3 v1 = verticesB[tri.v[1]];
        Vec3 v2 = verticesB[tri.v[2]];

        Vec3 centroid = (v0 + v1 + v2) / 3.0;

        if (!isPointInsideMesh(centroid, verticesA, trianglesA, bvhA)) {
            addTriangle(v0, v1, v2);
        }
    }

    // Output result
    cout << resultVertices.size() << " " << resultTriangles.size() << "\n";

    cout.precision(17);
    for (const auto& v : resultVertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& tri : resultTriangles) {
        cout << tri.v[0] << " " << tri.v[1] << " " << tri.v[2] << "\n";
    }

    return 0;
}
