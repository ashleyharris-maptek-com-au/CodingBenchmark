using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Voxel
{
    public int Mass;
    public int Value;
    public bool Exists;

    public Voxel(int mass, int value)
    {
        Mass = mass;
        Value = value;
        Exists = mass > 0;
    }
}

class Operation
{
    public int DigX, DigY, DigZ;
    public List<(int x, int y)> Steps = new List<(int x, int y)>();
    public int DestX, DestY;
    public double Profit;
}

class MiningProgram
{
    static int XCount, YCount, ZCount;
    static Voxel[,,] grid;
    static int[,] groundLevel;
    static List<Operation> operations = new List<Operation>();

    static void Main()
    {
        ReadInput();
        ProcessMining();
        WriteOutput();
    }

    static void ReadInput()
    {
        using (var stdin = Console.OpenStandardInput())
        using (var reader = new StreamReader(stdin))
        {
            var header = reader.ReadLine().Split(' ');
            XCount = int.Parse(header[0]);
            YCount = int.Parse(header[1]);
            ZCount = int.Parse(header[2]);

            grid = new Voxel[XCount, YCount, ZCount];
            groundLevel = new int[XCount, YCount];

            // Initialize ground levels to -1 (no ground)
            for (int x = 0; x < XCount; x++)
                for (int y = 0; y < YCount; y++)
                    groundLevel[x, y] = -1;

            var baseStream = reader.BaseStream;
            for (int x = 0; x < XCount; x++)
            {
                for (int y = 0; y < YCount; y++)
                {
                    for (int z = 0; z < ZCount; z++)
                    {
                        int d = baseStream.ReadByte();
                        if (d == -1) d = 0;

                        int mass = d % 16;
                        int value = d / 16;
                        grid[x, y, z] = new Voxel(mass, value);

                        if (mass > 0)
                        {
                            groundLevel[x, y] = z;
                        }
                    }
                }
            }
        }
    }

    static void ProcessMining()
    {
        long totalVoxels = (long)XCount * YCount * ZCount;

        // Collect all valuable voxels with profit estimates
        var candidates = new List<(int x, int y, int z, double profitEstimate)>();

        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                for (int z = 0; z <= groundLevel[x, y] && z < ZCount; z++)
                {
                    var voxel = grid[x, y, z];
                    if (voxel.Exists && voxel.Value > 0)
                    {
                        double revenue = voxel.Mass * voxel.Value * 1000.0;
                        int manhattan = Math.Abs(x) + Math.Abs(y);
                        double estCost = voxel.Mass * manhattan * 1.5; // Estimate with some uphill
                        double profit = revenue - estCost;

                        if (profit > 0)
                        {
                            candidates.Add((x, y, z, profit));
                        }
                    }
                }
            }
        }

        // Sort by profit estimate (highest first)
        candidates.Sort((a, b) => b.profitEstimate.CompareTo(a.profitEstimate));

        // Process based on problem scale
        int maxOps = totalVoxels < 1000000 ? candidates.Count : Math.Min(candidates.Count, 50000);

        for (int i = 0; i < maxOps; i++)
        {
            var (x, y, z, _) = candidates[i];
            if (CanDig(x, y, z))
            {
                TryDigAndTransport(x, y, z);
            }
        }
    }

    static bool CanDig(int x, int y, int z)
    {
        var voxel = grid[x, y, z];
        if (!voxel.Exists) return false;

        // Exposed from above
        if (z >= groundLevel[x, y]) return true;

        // Exposed from sides
        if ((x > 0 && !grid[x - 1, y, z].Exists) ||
            (x < XCount - 1 && !grid[x + 1, y, z].Exists) ||
            (y > 0 && !grid[x, y - 1, z].Exists) ||
            (y < YCount - 1 && !grid[x, y + 1, z].Exists))
        {
            return true;
        }

        return false;
    }

    static void TryDigAndTransport(int digX, int digY, int digZ)
    {
        var voxel = grid[digX, digY, digZ];
        if (!voxel.Exists) return;

        // Always transport valuable ore to plant at (0, 0)
        var path = FindPath(digX, digY, digZ, 0, 0);

        if (path != null)
        {
            var op = new Operation
            {
                DigX = digX,
                DigY = digY,
                DigZ = digZ,
                Steps = path,
                DestX = 0,
                DestY = 0
            };

            operations.Add(op);

            // Update state
            grid[digX, digY, digZ].Exists = false;
            UpdateGroundLevel(digX, digY);
        }
    }

    static List<(int x, int y)> FindPath(int startX, int startY, int startZ, int destX, int destY)
    {
        var path = new List<(int x, int y)>();
        int curX = startX;
        int curY = startY;
        int curZ = startZ;

        while (curX != destX || curY != destY)
        {
            int nextX = curX;
            int nextY = curY;

            // Prefer moving toward destination
            if (curX < destX) nextX++;
            else if (curX > destX) nextX--;
            else if (curY < destY) nextY++;
            else if (curY > destY) nextY--;

            int nextGroundZ = GetGroundLevel(nextX, nextY);
            int deltaZ = nextGroundZ - curZ;

            // Check slope constraint (max 1 Z per X/Y step)
            // For 10x more precision in Z, 1 Z per X/Y = 10% slope
            if (Math.Abs(deltaZ) > 1)
            {
                // Path blocked by terrain
                return null;
            }

            path.Add((nextX, nextY));
            curX = nextX;
            curY = nextY;
            curZ = nextGroundZ;
        }

        return path;
    }

    static int GetGroundLevel(int x, int y)
    {
        if (x < 0 || x >= XCount || y < 0 || y >= YCount)
            return 0;
        return Math.Max(0, groundLevel[x, y]);
    }

    static void UpdateGroundLevel(int x, int y)
    {
        int newLevel = -1;
        for (int z = ZCount - 1; z >= 0; z--)
        {
            if (grid[x, y, z].Exists)
            {
                newLevel = z;
                break;
            }
        }
        groundLevel[x, y] = newLevel;
    }

    static void WriteOutput()
    {
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.Steps.Count} {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var step in op.Steps)
            {
                Console.Write($" {step.x} {step.y}");
            }
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }
}
