using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
        public int Volume;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
            Volume = w * h * d;
        }
    }

    struct PlacedBox
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;

        public PlacedBox(int index, int x, int y, int z, int w, int h, int d)
        {
            Index = index;
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    struct Space
    {
        public int X, Y, Z;
        public int W, H, D;

        public Space(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int N = int.Parse(firstLine[0]);
        int containerW = int.Parse(firstLine[1]);
        int containerH = int.Parse(firstLine[2]);
        int containerD = int.Parse(firstLine[3]);

        List<Box> boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            string[] dims = Console.ReadLine().Split();
            int w = int.Parse(dims[0]);
            int h = int.Parse(dims[1]);
            int d = int.Parse(dims[2]);
            boxes.Add(new Box(i, w, h, d));
        }

        // Sort boxes by volume (largest first)
        boxes.Sort((a, b) => b.Volume.CompareTo(a.Volume));

        List<PlacedBox> placedBoxes = new List<PlacedBox>();
        List<Space> spaces = new List<Space>();
        spaces.Add(new Space(0, 0, 0, containerW, containerH, containerD));

        foreach (var box in boxes)
        {
            bool placed = false;

            // Try all 6 rotations
            int[][] rotations = new int[][]
            {
                new int[] { box.W, box.H, box.D },
                new int[] { box.W, box.D, box.H },
                new int[] { box.H, box.W, box.D },
                new int[] { box.H, box.D, box.W },
                new int[] { box.D, box.W, box.H },
                new int[] { box.D, box.H, box.W }
            };

            // Remove duplicate rotations
            var uniqueRotations = new List<int[]>();
            var seen = new HashSet<string>();
            foreach (var rot in rotations)
            {
                string key = $"{rot[0]}_{rot[1]}_{rot[2]}";
                if (seen.Add(key))
                {
                    uniqueRotations.Add(rot);
                }
            }

            int bestSpaceIdx = -1;
            int[] bestRotation = null;
            int bestScore = int.MaxValue;

            // Find best space and rotation
            for (int i = 0; i < spaces.Count; i++)
            {
                var space = spaces[i];

                foreach (var rotation in uniqueRotations)
                {
                    int w = rotation[0];
                    int h = rotation[1];
                    int d = rotation[2];

                    if (w <= space.W && h <= space.H && d <= space.D)
                    {
                        // Calculate score - prefer spaces with tighter fit
                        int wastedSpace = (space.W - w) + (space.H - h) + (space.D - d);
                        int score = wastedSpace * 1000 + space.X + space.Y + space.Z;

                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestSpaceIdx = i;
                            bestRotation = rotation;
                        }
                    }
                }
            }

            if (bestSpaceIdx >= 0)
            {
                var space = spaces[bestSpaceIdx];
                int w = bestRotation[0];
                int h = bestRotation[1];
                int d = bestRotation[2];

                placedBoxes.Add(new PlacedBox(box.Index, space.X, space.Y, space.Z, w, h, d));

                // Remove used space
                spaces.RemoveAt(bestSpaceIdx);

                // Generate new spaces
                List<Space> newSpaces = new List<Space>();

                // Space to the right
                if (space.X + w < space.X + space.W)
                {
                    newSpaces.Add(new Space(
                        space.X + w, space.Y, space.Z,
                        space.W - w, h, d));
                }

                // Space above
                if (space.Y + h < space.Y + space.H)
                {
                    newSpaces.Add(new Space(
                        space.X, space.Y + h, space.Z,
                        w, space.H - h, d));
                }

                // Space in front
                if (space.Z + d < space.Z + space.D)
                {
                    newSpaces.Add(new Space(
                        space.X, space.Y, space.Z + d,
                        w, h, space.D - d));
                }

                // Space to the right of the entire column
                if (space.X + w < space.X + space.W)
                {
                    newSpaces.Add(new Space(
                        space.X + w, space.Y, space.Z,
                        space.W - w, space.H, space.D));
                }

                // Space above the entire row
                if (space.Y + h < space.Y + space.H)
                {
                    newSpaces.Add(new Space(
                        space.X, space.Y + h, space.Z,
                        space.W, space.H - h, space.D));
                }

                // Space in front of the entire layer
                if (space.Z + d < space.Z + space.D)
                {
                    newSpaces.Add(new Space(
                        space.X, space.Y, space.Z + d,
                        space.W, space.H, space.D - d));
                }

                // Add new spaces and remove contained/invalid ones
                foreach (var newSpace in newSpaces)
                {
                    if (newSpace.W > 0 && newSpace.H > 0 && newSpace.D > 0 &&
                        newSpace.X + newSpace.W <= containerW &&
                        newSpace.Y + newSpace.H <= containerH &&
                        newSpace.Z + newSpace.D <= containerD)
                    {
                        bool isContained = false;
                        foreach (var existingSpace in spaces)
                        {
                            if (IsContained(newSpace, existingSpace))
                            {
                                isContained = true;
                                break;
                            }
                        }

                        if (!isContained && !OverlapsWithPlaced(newSpace, placedBoxes))
                        {
                            spaces.Add(newSpace);
                        }
                    }
                }

                // Remove spaces that overlap with the placed box
                spaces.RemoveAll(s => OverlapsWithBox(s, placedBoxes[placedBoxes.Count - 1]));

                placed = true;
            }
        }

        // Output results
        Console.WriteLine(placedBoxes.Count);
        foreach (var box in placedBoxes)
        {
            Console.WriteLine($"{box.Index} {box.X} {box.Y} {box.Z}");
        }
    }

    static bool IsContained(Space s1, Space s2)
    {
        return s1.X >= s2.X && s1.Y >= s2.Y && s1.Z >= s2.Z &&
               s1.X + s1.W <= s2.X + s2.W &&
               s1.Y + s1.H <= s2.Y + s2.H &&
               s1.Z + s1.D <= s2.Z + s2.D;
    }

    static bool OverlapsWithPlaced(Space space, List<PlacedBox> placed)
    {
        foreach (var box in placed)
        {
            if (!(space.X >= box.X + box.W || space.X + space.W <= box.X ||
                  space.Y >= box.Y + box.H || space.Y + space.H <= box.Y ||
                  space.Z >= box.Z + box.D || space.Z + space.D <= box.Z))
            {
                return true;
            }
        }
        return false;
    }

    static bool OverlapsWithBox(Space space, PlacedBox box)
    {
        return !(space.X >= box.X + box.W || space.X + space.W <= box.X ||
                 space.Y >= box.Y + box.H || space.Y + space.H <= box.Y ||
                 space.Z >= box.Z + box.D || space.Z + space.D <= box.Z);
    }
}
