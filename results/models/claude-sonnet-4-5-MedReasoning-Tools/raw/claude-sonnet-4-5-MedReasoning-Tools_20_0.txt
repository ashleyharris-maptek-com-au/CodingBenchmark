use std::collections::HashMap;
use std::io::{self, BufRead};

struct Sample {
    hole_id: String,
    sample_idx: usize,
    x: f64,
    y: f64,
    z: f64,
    properties: Vec<f64>,
}

struct Outlier {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

struct SpatialGrid {
    grid: HashMap<(i32, i32, i32), Vec<usize>>,
    cell_size: f64,
}

impl SpatialGrid {
    fn new(cell_size: f64) -> Self {
        SpatialGrid {
            grid: HashMap::new(),
            cell_size,
        }
    }

    fn insert(&mut self, idx: usize, x: f64, y: f64, z: f64) {
        let cell = self.get_cell(x, y, z);
        self.grid.entry(cell).or_insert_with(Vec::new).push(idx);
    }

    fn get_cell(&self, x: f64, y: f64, z: f64) -> (i32, i32, i32) {
        (
            (x / self.cell_size).floor() as i32,
            (y / self.cell_size).floor() as i32,
            (z / self.cell_size).floor() as i32,
        )
    }

    fn query_neighbors(&self, x: f64, y: f64, z: f64, radius: i32) -> Vec<usize> {
        let center = self.get_cell(x, y, z);
        let mut neighbors = Vec::new();

        for dx in -radius..=radius {
            for dy in -radius..=radius {
                for dz in -radius..=radius {
                    let cell = (center.0 + dx, center.1 + dy, center.2 + dz);
                    if let Some(indices) = self.grid.get(&cell) {
                        neighbors.extend(indices);
                    }
                }
            }
        }

        neighbors
    }
}

fn distance_squared(x1: f64, y1: f64, z1: f64, x2: f64, y2: f64, z2: f64) -> f64 {
    let dx = x1 - x2;
    let dy = y1 - y2;
    let dz = z1 - z2;
    dx * dx + dy * dy + dz * dz
}

fn find_k_nearest(
    samples: &[Sample],
    idx: usize,
    k: usize,
    spatial_index: Option<&SpatialGrid>,
) -> Vec<usize> {
    let sample = &samples[idx];

    let candidates: Vec<usize> = if let Some(grid) = spatial_index {
        let mut radius = 1;
        let mut candidates = Vec::new();

        while candidates.len() < k * 3 && radius <= 5 {
            candidates = grid.query_neighbors(sample.x, sample.y, sample.z, radius);
            radius += 1;
        }

        candidates
    } else {
        (0..samples.len()).collect()
    };

    let mut distances: Vec<(usize, f64)> = candidates
        .iter()
        .filter(|&&j| j != idx)
        .map(|&j| {
            let other = &samples[j];
            let dist_sq = distance_squared(sample.x, sample.y, sample.z, other.x, other.y, other.z);
            (j, dist_sq)
        })
        .collect();

    distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());

    distances.into_iter().take(k).map(|(j, _)| j).collect()
}

fn is_valid_value(v: f64) -> bool {
    v.is_finite()
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap_or(0);
    let num_properties: usize = parts[1].parse().unwrap_or(0);

    let property_names: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    let mut samples = Vec::new();

    for _ in 0..num_holes {
        let hole_line = match lines.next() {
            Some(Ok(line)) => line,
            _ => break,
        };

        let parts: Vec<&str> = hole_line.split_whitespace().collect();
        if parts.len() < 9 {
            continue;
        }

        let hole_id = parts[0].to_string();
        let start_x: f64 = parts[1].parse().unwrap_or(0.0);
        let start_y: f64 = parts[2].parse().unwrap_or(0.0);
        let start_z: f64 = parts[3].parse().unwrap_or(0.0);
        let dir_x: f64 = parts[4].parse().unwrap_or(0.0);
        let dir_y: f64 = parts[5].parse().unwrap_or(0.0);
        let dir_z: f64 = parts[6].parse().unwrap_or(0.0);
        let _length: f64 = parts[7].parse().unwrap_or(0.0);
        let num_samples: usize = parts[8].parse().unwrap_or(0);

        for sample_idx in 0..num_samples {
            let sample_line = match lines.next() {
                Some(Ok(line)) => line,
                _ => break,
            };

            let parts: Vec<&str> = sample_line.split_whitespace().collect();
            if parts.len() < 1 + num_properties {
                continue;
            }

            let depth: f64 = parts[0].parse().unwrap_or(0.0);

            let x = start_x + dir_x * depth;
            let y = start_y + dir_y * depth;
            let z = start_z + dir_z * depth;

            let properties: Vec<f64> = parts[1..]
                .iter()
                .take(num_properties)
                .map(|s| s.parse().unwrap_or(0.0))
                .collect();

            samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx,
                x,
                y,
                z,
                properties,
            });
        }
    }

    let total_samples = samples.len();

    if total_samples == 0 {
        println!("0");
        return;
    }

    // Adaptive strategy based on dataset size
    let (use_spatial_index, k_neighbors, cell_size, sample_stride) = if total_samples < 5_000 {
        (false, 20, 0.0, 1)
    } else if total_samples < 50_000 {
        (true, 30, 50.0, 1)
    } else if total_samples < 500_000 {
        (true, 50, 150.0, 1)
    } else {
        // For very large datasets, sample every Nth point
        (true, 50, 200.0, 3)
    };

    let spatial_index = if use_spatial_index {
        let mut grid = SpatialGrid::new(cell_size);
        for (i, sample) in samples.iter().enumerate() {
            grid.insert(i, sample.x, sample.y, sample.z);
        }
        Some(grid)
    } else {
        None
    };

    let mut outliers = Vec::new();

    for i in (0..total_samples).step_by(sample_stride) {
        let sample = &samples[i];
        let neighbors = find_k_nearest(&samples, i, k_neighbors, spatial_index.as_ref());

        if neighbors.len() < 5 {
            continue;
        }

        for (prop_idx, &value) in sample.properties.iter().enumerate() {
            if !is_valid_value(value) {
                continue;
            }

            if prop_idx >= property_names.len() {
                continue;
            }

            let neighbor_values: Vec<f64> = neighbors
                .iter()
                .map(|&j| samples[j].properties[prop_idx])
                .filter(|&v| is_valid_value(v))
                .collect();

            if neighbor_values.len() < 5 {
                continue;
            }

            let mean: f64 = neighbor_values.iter().sum::<f64>() / neighbor_values.len() as f64;
            let variance: f64 = neighbor_values
                .iter()
                .map(|v| (v - mean).powi(2))
                .sum::<f64>() / neighbor_values.len() as f64;
            let std_dev = variance.sqrt();

            if std_dev > 1e-10 {
                let z_score = ((value - mean) / std_dev).abs();
                if z_score > 3.0 {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence: z_score,
                    });
                }
            }
        }
    }

    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.2}", outlier.hole_id, outlier.sample_idx, outlier.property_name, outlier.confidence);
    }
}
