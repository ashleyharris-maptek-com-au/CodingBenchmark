using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class Program
{
    // ====== Basic graph storage for input edges ======
    struct OriginalEdge
    {
        public int u, v, w;
    }

    // For Euler multigraph
    struct EulerEdge
    {
        public int to;
        public int next;
        public int copyId;
    }

    // For Dijkstra on small graphs
    struct WeightedEdge
    {
        public int to;
        public int w;
        public int next;
    }

    // For MST adjacency
    struct MSTArc
    {
        public int to;
        public int next;
        public int edgeIndex; // index in edges[]
    }

    // For Kruskal sort
    struct EdgeComparer : IComparer<OriginalEdge>
    {
        public int Compare(OriginalEdge a, OriginalEdge b)
        {
            return a.w.CompareTo(b.w);
        }
    }

    // ====== Fast scanner ======
    sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return 0;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            for (; c > ' '; c = Read())
                val = val * 10 + c - '0';

            return val * sign;
        }
    }

    // ====== Simple binary heap for Dijkstra (min-heap by distance) ======
    struct HeapItem
    {
        public long dist;
        public int v;
    }

    sealed class BinaryHeap
    {
        private HeapItem[] _data;
        private int _size;

        public BinaryHeap(int capacity)
        {
            _data = new HeapItem[capacity];
            _size = 0;
        }

        public bool Empty => _size == 0;

        public void Clear()
        {
            _size = 0;
        }

        public void Push(long dist, int v)
        {
            int i = _size++;
            // sift up
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (_data[p].dist <= dist) break;
                _data[i] = _data[p];
                i = p;
            }
            _data[i].dist = dist;
            _data[i].v = v;
        }

        public HeapItem Pop()
        {
            HeapItem ret = _data[0];
            HeapItem x = _data[--_size];
            int i = 0;
            while (true)
            {
                int l = (i << 1) + 1;
                if (l >= _size) break;
                int r = l + 1;
                int smallest = (r < _size && _data[r].dist < _data[l].dist) ? r : l;
                if (_data[smallest].dist >= x.dist) break;
                _data[i] = _data[smallest];
                i = smallest;
            }
            _data[i] = x;
            return ret;
        }
    }

    // ====== Static fields for global access ======
    static int N, M;
    static OriginalEdge[] edges;
    static int[] deg;

    // For Euler multigraph
    static EulerEdge[] eulerEdges;
    static int[] eulerHead;
    static bool[] usedCopy;
    static int maxCopies;
    static int copyCnt;
    static int eulerArcCnt;

    // Odd vertices
    static int[] oddVertices;
    static int oddCount;

    // ====== Add one undirected copy-edge to Euler multigraph ======
    static void AddCopyEdge(int u, int v)
    {
        int id = copyCnt++;
        // u -> v
        eulerEdges[eulerArcCnt].to = v;
        eulerEdges[eulerArcCnt].copyId = id;
        eulerEdges[eulerArcCnt].next = eulerHead[u];
        eulerHead[u] = eulerArcCnt++;
        // v -> u
        eulerEdges[eulerArcCnt].to = u;
        eulerEdges[eulerArcCnt].copyId = id;
        eulerEdges[eulerArcCnt].next = eulerHead[v];
        eulerHead[v] = eulerArcCnt++;
    }

    // ====== Exact solver for small graphs using DP over odd vertices ======
    static void RunExactDp()
    {
        int n = N;
        int m = M;
        int k = oddCount;
        if (k == 0) return;

        int[] odd = oddVertices;

        // Build adjacency for Dijkstra
        int[] headW = new int[n];
        for (int i = 0; i < n; i++) headW[i] = -1;

        WeightedEdge[] wEdges = new WeightedEdge[2 * m];
        int wArcCnt = 0;
        for (int i = 0; i < m; i++)
        {
            var e = edges[i];
            int u = e.u, v = e.v, w = e.w;

            wEdges[wArcCnt].to = v;
            wEdges[wArcCnt].w = w;
            wEdges[wArcCnt].next = headW[u];
            headW[u] = wArcCnt++;

            wEdges[wArcCnt].to = u;
            wEdges[wArcCnt].w = w;
            wEdges[wArcCnt].next = headW[v];
            headW[v] = wArcCnt++;
        }

        const long INF = long.MaxValue / 4;
        long[,] distMat = new long[k, k];
        int[,] prevAll = new int[k, n];
        long[] dist = new long[n];

        var heap = new BinaryHeap(Math.Max(m * 2 + 1, n * 4));

        // Dijkstra from each odd vertex
        for (int idx = 0; idx < k; idx++)
        {
            int s = odd[idx];

            for (int v = 0; v < n; v++)
            {
                dist[v] = INF;
                prevAll[idx, v] = -1;
            }

            dist[s] = 0;
            heap.Clear();
            heap.Push(0, s);

            while (!heap.Empty)
            {
                var item = heap.Pop();
                long d = item.dist;
                int v = item.v;
                if (d != dist[v]) continue;

                for (int eid = headW[v]; eid != -1; eid = wEdges[eid].next)
                {
                    int to = wEdges[eid].to;
                    long nd = d + wEdges[eid].w;
                    if (nd < dist[to])
                    {
                        dist[to] = nd;
                        prevAll[idx, to] = v;
                        heap.Push(nd, to);
                    }
                }
            }

            for (int j = 0; j < k; j++)
            {
                distMat[idx, j] = dist[odd[j]];
            }
        }

        // DP over subsets of odd vertices
        int fullMask = (1 << k) - 1;
        long[] dp = new long[fullMask + 1];
        int[] pair = new int[fullMask + 1];

        for (int i = 1; i <= fullMask; i++)
        {
            dp[i] = INF;
            pair[i] = -1;
        }
        dp[0] = 0;
        pair[0] = -1;

        for (int S = 1; S <= fullMask; S++)
        {
            // Find first set bit i
            int tmp = S;
            int i = 0;
            while ((tmp & 1) == 0)
            {
                i++;
                tmp >>= 1;
            }

            long best = INF;
            int bestJ = -1;
            for (int j = i + 1; j < k; j++)
            {
                if ((S & (1 << j)) == 0) continue;
                int S2 = S ^ (1 << i) ^ (1 << j);
                long cand = distMat[i, j] + dp[S2];
                if (cand < best)
                {
                    best = cand;
                    bestJ = j;
                }
            }
            dp[S] = best;
            pair[S] = bestJ;
        }

        // Reconstruct pairings and paths, add duplicate edges
        int curMask = fullMask;
        while (curMask != 0)
        {
            int tmp = curMask;
            int iIdx = 0;
            while ((tmp & 1) == 0)
            {
                iIdx++;
                tmp >>= 1;
            }
            int jIdx = pair[curMask];
            if (jIdx == -1) break; // should not happen if graph connected

            int start = odd[iIdx];
            int end = odd[jIdx];

            int cur = end;
            // Follow predecessors from Dijkstra starting at 'start'
            while (cur != start)
            {
                int p = prevAll[iIdx, cur];
                // Duplicate edge (p, cur)
                AddCopyEdge(p, cur);
                cur = p;
            }

            curMask ^= (1 << iIdx) ^ (1 << jIdx);
        }
    }

    // ====== Approximate large-graph solver using MST + tree parity DP ======
    static void RunMstParity()
    {
        int n = N;
        int m = M;
        if (oddCount == 0 || m == 0) return;

        // Kruskal MST (on all edges)
        int[] dsuParent = new int[n];
        int[] dsuRank = new int[n];
        for (int i = 0; i < n; i++)
        {
            dsuParent[i] = i;
            dsuRank[i] = 0;
        }

        int Find(int x)
        {
            while (dsuParent[x] != x)
            {
                dsuParent[x] = dsuParent[dsuParent[x]];
                x = dsuParent[x];
            }
            return x;
        }

        bool Union(int a, int b)
        {
            int ra = Find(a);
            int rb = Find(b);
            if (ra == rb) return false;
            if (dsuRank[ra] < dsuRank[rb])
            {
                dsuParent[ra] = rb;
            }
            else if (dsuRank[ra] > dsuRank[rb])
            {
                dsuParent[rb] = ra;
            }
            else
            {
                dsuParent[rb] = ra;
                dsuRank[ra]++;
            }
            return true;
        }

        // Sort edges by weight
        Array.Sort(edges, new EdgeComparer());

        int maxMstEdges = Math.Min(n - 1, m);
        MSTArc[] mstArcs = new MSTArc[2 * maxMstEdges];
        int[] mstHead = new int[n];
        for (int i = 0; i < n; i++) mstHead[i] = -1;
        int mstArcCnt = 0;
        int mstEdgeUsed = 0;

        for (int i = 0; i < m && mstEdgeUsed < n - 1; i++)
        {
            var e = edges[i];
            int u = e.u, v = e.v;
            if (u == v) continue; // ignore loops for MST
            if (Union(u, v))
            {
                // u -> v
                mstArcs[mstArcCnt].to = v;
                mstArcs[mstArcCnt].edgeIndex = i;
                mstArcs[mstArcCnt].next = mstHead[u];
                mstHead[u] = mstArcCnt++;
                // v -> u
                mstArcs[mstArcCnt].to = u;
                mstArcs[mstArcCnt].edgeIndex = i;
                mstArcs[mstArcCnt].next = mstHead[v];
                mstHead[v] = mstArcCnt++;

                mstEdgeUsed++;
            }
        }

        // BFS (or multi-BFS) over MST forest to get parent relationships
        int[] parent = new int[n];
        int[] parentEdgeIdx = new int[n];
        bool[] visited = new bool[n];
        int[] order = new int[n];
        int orderCnt = 0;
        int[] queue = new int[n];

        for (int i = 0; i < n; i++)
        {
            parent[i] = -1;
            parentEdgeIdx[i] = -1;
            visited[i] = false;
        }

        for (int start = 0; start < n; start++)
        {
            if (visited[start]) continue;
            if (mstHead[start] == -1) continue; // isolated or only loops

            int qh = 0, qt = 0;
            visited[start] = true;
            parent[start] = -1;
            queue[qt++] = start;

            while (qh < qt)
            {
                int v = queue[qh++];
                order[orderCnt++] = v;

                for (int eid = mstHead[v]; eid != -1; eid = mstArcs[eid].next)
                {
                    int to = mstArcs[eid].to;
                    if (!visited[to])
                    {
                        visited[to] = true;
                        parent[to] = v;
                        parentEdgeIdx[to] = mstArcs[eid].edgeIndex;
                        queue[qt++] = to;
                    }
                }
            }
        }

        // Parity array from original degrees
        int[] oddFlag = new int[n];
        for (int i = 0; i < n; i++)
            oddFlag[i] = deg[i] & 1;

        // Process vertices bottom-up; toggle parities via MST edges
        for (int idx = orderCnt - 1; idx >= 0; idx--)
        {
            int v = order[idx];
            int p = parent[v];
            if (p != -1 && oddFlag[v] == 1)
            {
                int eidx = parentEdgeIdx[v];
                var e = edges[eidx];
                AddCopyEdge(e.u, e.v);
                oddFlag[p] ^= 1;
                // oddFlag[v] becomes even implicitly
            }
        }
    }

    // ====== Build Eulerian tour (Hierholzer) on the multigraph ======
    static int[] BuildEulerTour()
    {
        int n = N;
        int totalCopies = copyCnt; // number of undirected copies

        int[] iter = new int[n];
        for (int i = 0; i < n; i++)
            iter[i] = eulerHead[i];

        int[] stack = new int[totalCopies + 1];
        int[] result = new int[totalCopies + 1];

        int sp = 0;
        int resLen = 0;

        // start at node 0 (as required)
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int eid = iter[v];

            // Skip used edges
            while (eid != -1 && usedCopy[eulerEdges[eid].copyId])
            {
                eid = eulerEdges[eid].next;
            }
            iter[v] = eid;

            if (eid == -1)
            {
                // no more unused edges out of v
                result[resLen++] = v;
                sp--;
            }
            else
            {
                EulerEdge e = eulerEdges[eid];
                usedCopy[e.copyId] = true;
                iter[v] = eulerEdges[eid].next; // advance iterator
                stack[sp++] = e.to;
            }
        }

        // result is in reverse order; reverse when printing
        // resLen should be totalCopies + 1
        if (resLen == 0)
        {
            // no edges, just stay at 0
            result[0] = 0;
            resLen = 1;
        }

        // Return array with actual length encoded in first element? Simpler: store length separately.
        int[] tour = new int[resLen];
        for (int i = 0; i < resLen; i++)
            tour[i] = result[resLen - 1 - i];
        return tour;
    }

    // ====== Main ======
    static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = fs.NextInt();
        M = fs.NextInt();

        edges = new OriginalEdge[M];
        deg = new int[N];

        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();

            edges[i].u = u;
            edges[i].v = v;
            edges[i].w = w;

            if (u == v)
            {
                // loop contributes 2 to degree
                deg[u] += 2;
            }
            else
            {
                deg[u]++;
                deg[v]++;
            }
        }

        // Collect odd-degree vertices
        var oddList = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if ((deg[i] & 1) != 0)
                oddList.Add(i);
        }
        oddVertices = oddList.ToArray();
        oddCount = oddVertices.Length;

        // Prepare Euler multigraph storage
        maxCopies = 2 * M + 5;                 // enough for all original edges + all duplicates
        eulerEdges = new EulerEdge[2 * maxCopies];
        eulerHead = new int[N];
        for (int i = 0; i < N; i++) eulerHead[i] = -1;
        usedCopy = new bool[maxCopies];
        copyCnt = 0;
        eulerArcCnt = 0;

        // Choose algorithm:
        // - Small graph with few odd vertices: exact DP
        // - Otherwise: MST-based parity fix
        bool useExactDp =
            (oddCount > 0 &&
             N <= 500 &&
             M <= 5000 &&
             oddCount <= 20);

        if (useExactDp)
        {
            RunExactDp();
        }
        else if (oddCount > 0)
        {
            RunMstParity();
        }
        // if oddCount == 0, graph is already Eulerian; no extra edges needed

        // Add all original edges (each once)
        for (int i = 0; i < M; i++)
        {
            var e = edges[i];
            AddCopyEdge(e.u, e.v);
        }

        // Build Euler tour starting/ending at 0
        int[] tour = BuildEulerTour();

        // Output tour as sequence of node indices
        var sb = new StringBuilder(tour.Length * 4);
        for (int i = 0; i < tour.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(tour[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
