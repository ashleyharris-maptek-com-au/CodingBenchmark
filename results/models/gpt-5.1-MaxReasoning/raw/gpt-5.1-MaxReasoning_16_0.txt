#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

struct Task {
    int machine;       // machine index
    int duration;      // processing time
    long long start;   // computed start time
};

struct Interval {
    long long start;
    long long end;
};

// ---------- Small-instance helpers (gap-aware scheduling) ----------

static inline long long findEarliestOnMachine(const vector<Interval>& intervals,
                                              long long release,
                                              int dur) {
    long long t = release;
    for (const auto& iv : intervals) {
        // If the operation fits before this interval, we are done
        if (t + dur <= iv.start) {
            break;
        }
        // Otherwise, if it overlaps, push t to after this interval
        if (t < iv.end) {
            t = iv.end;
        }
    }
    return t;
}

static inline void insertInterval(vector<Interval>& intervals,
                                  long long s,
                                  long long e) {
    // Keep intervals sorted by start time
    auto it = intervals.begin();
    while (it != intervals.end() && it->start <= s) {
        ++it;
    }
    intervals.insert(it, Interval{s, e});
}

void scheduleSmall(vector<vector<Task>>& jobs, int J, int M, long long totalTasks) {
    const long long INF = std::numeric_limits<long long>::max() / 4;

    vector<long long> jobReady(J, 0);     // completion time of last scheduled task in each job
    vector<int> jobNext(J, 0);           // index of next task to schedule in each job
    vector<vector<Interval>> machineSched(M); // intervals on each machine

    long long remaining = totalTasks;

    while (remaining > 0) {
        int bestJob = -1;
        int bestMachine = -1;
        long long bestStart = 0;
        long long bestFinish = INF;

        // Select the next operation to schedule
        for (int j = 0; j < J; ++j) {
            int idx = jobNext[j];
            if (idx >= (int)jobs[j].size()) continue; // job finished

            Task& t = jobs[j][idx];
            int m = t.machine;
            int dur = t.duration;
            long long release = jobReady[j];

            long long startCand = findEarliestOnMachine(machineSched[m], release, dur);
            long long finishCand = startCand + dur;

            if (finishCand < bestFinish ||
               (finishCand == bestFinish && startCand < bestStart) ||
               (finishCand == bestFinish && startCand == bestStart && j < bestJob)) {
                bestFinish = finishCand;
                bestStart = startCand;
                bestJob = j;
                bestMachine = m;
            }
        }

        // Schedule the chosen operation
        if (bestJob == -1) {
            // Should not happen in valid input, but guard against it
            break;
        }

        int idx = jobNext[bestJob];
        Task& bt = jobs[bestJob][idx];
        bt.start = bestStart;
        long long finish = bestStart + bt.duration;

        jobReady[bestJob] = finish;
        jobNext[bestJob]++;
        remaining--;

        insertInterval(machineSched[bestMachine], bestStart, finish);
    }
}

// ---------- Large-instance scheduler (fast, append-only, PQ-based) ----------

void scheduleLarge(vector<vector<Task>>& jobs, int J, int M, long long totalTasks) {
    vector<long long> jobReady(J, 0);    // completion time of last scheduled task in each job
    vector<int> jobNext(J, 0);          // index of next task to schedule in each job
    vector<long long> machineReady(M, 0); // time when each machine becomes free

    struct PQItem {
        long long est;  // estimated earliest start (key for ordering)
        int job;        // job index
    };

    struct Cmp {
        bool operator()(const PQItem& a, const PQItem& b) const {
            if (a.est != b.est) return a.est > b.est; // min-heap on est
            return a.job > b.job;                     // tie-break by job index
        }
    };

    priority_queue<PQItem, vector<PQItem>, Cmp> pq;

    // Initialize: first task of each job
    for (int j = 0; j < J; ++j) {
        if (jobs[j].empty()) continue;
        Task& t = jobs[j][0];
        int m = t.machine;
        long long est = max(jobReady[j], machineReady[m]); // here both zeros
        pq.push(PQItem{est, j});
    }

    long long scheduledCount = 0;

    while (scheduledCount < totalTasks && !pq.empty()) {
        PQItem item = pq.top();
        pq.pop();
        int j = item.job;

        int idx = jobNext[j];
        if (idx >= (int)jobs[j].size()) {
            // This job is already finished (stale PQ entry), skip
            continue;
        }

        Task& t = jobs[j][idx];
        int m = t.machine;
        int dur = t.duration;

        long long actualStart = max(jobReady[j], machineReady[m]);
        long long finish = actualStart + (long long)dur;

        t.start = actualStart;
        jobReady[j] = finish;
        machineReady[m] = finish;

        jobNext[j]++;
        scheduledCount++;

        if (jobNext[j] < (int)jobs[j].size()) {
            Task& nextT = jobs[j][jobNext[j]];
            int nextM = nextT.machine;
            long long nextEst = max(jobReady[j], machineReady[nextM]);
            pq.push(PQItem{nextEst, j});
        }
    }
}

// ---------- Main ----------

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) {
        return 0; // no input
    }

    vector<vector<Task>> jobs(J);
    long long totalTasks = 0;

    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].reserve(T);
        totalTasks += T;
        for (int k = 0; k < T; ++k) {
            int m, d;
            cin >> m >> d;
            Task t;
            t.machine = m;
            t.duration = d;
            t.start = 0;
            jobs[j].push_back(t);
        }
    }

    if (totalTasks == 0) {
        // No tasks: print one empty line per job (or nothing if J==0)
        for (int j = 0; j < J; ++j) {
            cout << '\n';
        }
        return 0;
    }

    // Adaptive strategy selection based on problem scale
    const long long SMALL_MAX_TASKS = 3000;
    const int      SMALL_MAX_JOBS = 50;
    const int      SMALL_MAX_MACHINES = 20;

    bool useSmall =
        (totalTasks <= SMALL_MAX_TASKS &&
         J <= SMALL_MAX_JOBS &&
         M <= SMALL_MAX_MACHINES);

    if (useSmall) {
        scheduleSmall(jobs, J, M, totalTasks);
    } else {
        scheduleLarge(jobs, J, M, totalTasks);
    }

    // Output: for each job, all tasks: start_time machine start_time machine ...
    for (int j = 0; j < J; ++j) {
        int T = (int)jobs[j].size();
        for (int k = 0; k < T; ++k) {
            if (k > 0) cout << ' ';
            cout << jobs[j][k].start << ' ' << jobs[j][k].machine;
        }
        cout << '\n';
    }

    return 0;
}
