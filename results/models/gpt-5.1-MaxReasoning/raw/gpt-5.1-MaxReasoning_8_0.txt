```rust
use std::collections::VecDeque;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

const WALL: u8 = 0;
const OPEN: u8 = 1;
// For a visited cell, these encode where its parent is relative to this cell.
const FROM_UP: u8 = 2;    // parent is above (idx - W)
const FROM_DOWN: u8 = 3;  // parent is below (idx + W)
const FROM_LEFT: u8 = 4;  // parent is left  (idx - 1)
const FROM_RIGHT: u8 = 5; // parent is right (idx + 1)
const START: u8 = 6;      // start cell (A), treated as visited root

fn main() {
    // ----- Input -----
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    // Read H and W
    let mut first_line = String::new();
    if reader.read_line(&mut first_line).unwrap() == 0 {
        // No input
        return;
    }
    let mut parts = first_line.split_whitespace();
    let h: usize = parts
        .next()
        .expect("Missing H")
        .parse()
        .expect("H must be an integer");
    let w: usize = parts
        .next()
        .expect("Missing W")
        .parse()
        .expect("W must be an integer");

    let total_cells = h
        .checked_mul(w)
        .expect("Grid too large (overflow computing H*W)");

    // Single flat grid, 1 byte per cell.
    let mut grid = vec![0u8; total_cells];

    let mut start_idx: Option<usize> = None;
    let mut goal_idx: Option<usize> = None;

    let mut line = String::with_capacity(w + 4);

    for y in 0..h {
        line.clear();
        let mut read_any = false;
        while !read_any {
            let n = reader.read_line(&mut line).unwrap();
            if n == 0 {
                panic!("Unexpected EOF while reading maze rows");
            }
            if n > 0 {
                read_any = true;
            }
        }

        let bytes = line.as_bytes();
        if bytes.len() < w {
            panic!("Row {} has fewer than W characters", y);
        }

        for x in 0..w {
            let ch = bytes[x] as char;
            let idx = y * w + x;
            match ch {
                '#' => grid[idx] = WALL,
                ' ' => grid[idx] = OPEN,
                'A' => {
                    if start_idx.is_some() {
                        panic!("Multiple 'A' cells found");
                    }
                    start_idx = Some(idx);
                    grid[idx] = START;
                }
                'B' => {
                    if goal_idx.is_some() {
                        panic!("Multiple 'B' cells found");
                    }
                    goal_idx = Some(idx);
                    grid[idx] = OPEN; // treat as open
                }
                _ => {
                    // Treat any unexpected character as wall for safety.
                    // Adjust if you want to be more permissive.
                    grid[idx] = WALL;
                }
            }
        }
    }

    let start_idx = start_idx.expect("Start 'A' not found in maze");
    let goal_idx = goal_idx.expect("Goal 'B' not found in maze");

    let start_x = (start_idx % w) as u32;
    let start_y = (start_idx / w) as u32;
    let goal_x = (goal_idx % w) as u32;
    let goal_y = (goal_idx / w) as u32;

    // ----- BFS -----
    let mut queue: VecDeque<(u32, u32)> = VecDeque::new();
    queue.push_back((start_x, start_y));

    let mut found = false;

    while let Some((x, y)) = queue.pop_front() {
        if x == goal_x && y == goal_y {
            found = true;
            break;
        }

        let x_us = x as usize;
        let y_us = y as usize;
        let idx = y_us * w + x_us;

        // Up
        if y_us > 0 {
            let nidx = idx - w;
            if grid[nidx] == OPEN {
                grid[nidx] = FROM_DOWN; // parent is below this cell
                queue.push_back((x, y - 1));
            }
        }

        // Down
        if y_us + 1 < h {
            let nidx = idx + w;
            if grid[nidx] == OPEN {
                grid[nidx] = FROM_UP; // parent is above this cell
                queue.push_back((x, y + 1));
            }
        }

        // Left
        if x_us > 0 {
            let nidx = idx - 1;
            if grid[nidx] == OPEN {
                grid[nidx] = FROM_RIGHT; // parent is right of this cell
                queue.push_back((x - 1, y));
            }
        }

        // Right
        if x_us + 1 < w {
            let nidx = idx + 1;
            if grid[nidx] == OPEN {
                grid[nidx] = FROM_LEFT; // parent is left of this cell
                queue.push_back((x + 1, y));
            }
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if !found {
        // No path
        writeln!(out, "0").unwrap();
        return;
    }

    // ----- Reconstruct path -----
    // First pass: compute path length in edges (number of moves).
    let mut length_edges: usize = 0;
    let mut idx = goal_idx;
    while idx != start_idx {
        let code = grid[idx];
        let parent_idx = match code {
            FROM_UP => idx - w,
            FROM_DOWN => idx + w,
            FROM_LEFT => idx - 1,
            FROM_RIGHT => idx + 1,
            _ => panic!("Broken parent chain during length computation"),
        };
        length_edges += 1;
        idx = parent_idx;
    }

    // Second pass: collect directions from A to B (as 0..3).
    // We'll gather them in reverse (B->A), then iterate reversed.
    // Direction encoding for moves:
    // 0 = Up, 1 = Down, 2 = Left, 3 = Right
    let mut dirs: Vec<u8> = Vec::with_capacity(length_edges);
    idx = goal_idx;
    while idx != start_idx {
        let code = grid[idx];
        let parent_idx = match code {
            FROM_UP => idx - w,
            FROM_DOWN => idx + w,
            FROM_LEFT => idx - 1,
            FROM_RIGHT => idx + 1,
            _ => panic!("Broken parent chain during direction reconstruction"),
        };

        // Convert parent-relative direction to move from parent to this cell
        let move_dir = match code {
            FROM_UP => 1,    // parent above -> move Down
            FROM_DOWN => 0,  // parent below -> move Up
            FROM_LEFT => 3,  // parent left  -> move Right
            FROM_RIGHT => 2, // parent right -> move Left
            _ => unreachable!(),
        };

        dirs.push(move_dir);
        idx = parent_idx;
    }

    // Number of positions includes both start and end.
    let total_positions = length_edges + 1;
    writeln!(out, "{}", total_positions).unwrap();

    // Walk from A applying directions in reverse order to print coordinates.
    let mut x = start_x as i32;
    let mut y = start_y as i32;

    writeln!(out, "{} {}", x, y).unwrap(); // start position

    for &dir in dirs.iter().rev() {
        match dir {
            0 => y -= 1, // Up
            1 => y += 1, // Down
            2 => x -= 1, // Left
            3 => x += 1, // Right
            _ => unreachable!(),
        }
        writeln!(out, "{} {}", x, y).unwrap();
    }

    out.flush().unwrap();
}
```
