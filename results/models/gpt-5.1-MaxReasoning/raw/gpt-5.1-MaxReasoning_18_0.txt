#include <iostream>
#include <vector>
#include <cstdint>
#include <limits>
#include <queue>
#include <deque>
#include <algorithm>

using std::int64_t;
using std::uint32_t;
using std::uint64_t;

// ===================== Disjoint Set Union (Union-Find) =====================

struct DSU {
    std::vector<int> parent;
    std::vector<int> rankv;

    DSU() = default;
    explicit DSU(size_t n) { init(n); }

    void init(size_t n) {
        parent.resize(n);
        rankv.assign(n, 0);
        for (size_t i = 0; i < n; ++i) parent[i] = (int)i;
    }

    int find(int x) {
        while (parent[x] != x) {
            parent[x] = parent[parent[x]];
            x = parent[x];
        }
        return x;
    }

    void unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return;
        if (rankv[a] < rankv[b]) std::swap(a, b);
        parent[b] = a;
        if (rankv[a] == rankv[b]) ++rankv[a];
    }
};

// ========================== Dinic Max-Flow ==========================

struct Dinic {
    struct Edge {
        int to;
        int rev;
        int64_t cap;
    };

    int N;
    std::vector<std::vector<Edge>> g;
    std::vector<int> level;
    std::vector<int> it;

    Dinic() : N(0) {}
    explicit Dinic(int n) { init(n); }

    void init(int n) {
        N = n;
        g.assign(n, {});
    }

    void addEdge(int fr, int to, int64_t cap) {
        Edge a{to, (int)g[to].size(), cap};
        Edge b{fr, (int)g[fr].size(), 0};
        g[fr].push_back(a);
        g[to].push_back(b);
    }

    bool bfs(int s, int t) {
        level.assign(N, -1);
        std::deque<int> q;
        level[s] = 0;
        q.push_back(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop_front();
            for (const auto &e : g[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    if (e.to == t) return true;
                    q.push_back(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int64_t dfs(int v, int t, int64_t f) {
        if (v == t) return f;
        for (int &i = it[v]; i < (int)g[v].size(); ++i) {
            Edge &e = g[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                int64_t d = dfs(e.to, t, std::min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    g[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }

    int64_t maxflow(int s, int t) {
        const int64_t INF = std::numeric_limits<int64_t>::max() / 4;
        int64_t flow = 0;
        while (bfs(s, t)) {
            it.assign(N, 0);
            int64_t f;
            while ((f = dfs(s, t, INF)) > 0) {
                flow += f;
            }
        }
        return flow;
    }
};

// ====================== Stoer–Wagner: matrix variant ======================

/*
 * This variant is used for small graphs (N <= MATRIX_N_THRESHOLD).
 * It works on an adjacency matrix 'cap' (undirected, non-negative weights),
 * and logs all contractions (merges) in 'merges'.
 * It returns the minimum cut value in 'best_value' and the pair of
 * supernodes (best_s_super, best_t_super) at the phase where the best
 * cut was first observed, plus the merge prefix length at that time.
 */

void stoer_wagner_matrix(
    uint32_t N,
    std::vector<std::vector<uint64_t>> &cap,
    std::vector<std::pair<uint32_t, uint32_t>> &merges,
    uint32_t &best_s_super,
    uint32_t &best_t_super,
    size_t &best_merge_prefix,
    uint64_t &best_value
) {
    std::vector<uint32_t> verts(N);
    for (uint32_t i = 0; i < N; ++i) verts[i] = i;

    best_value = std::numeric_limits<uint64_t>::max();
    best_s_super = 0;
    best_t_super = (N > 1 ? 1 : 0);
    best_merge_prefix = 0;
    size_t merges_so_far = 0;

    uint32_t cur_n = N;
    while (cur_n > 1) {
        std::vector<uint64_t> w(cur_n, 0);
        std::vector<char> added(cur_n, 0);

        int prev_idx = -1;
        int sel_idx = -1;

        for (uint32_t i = 0; i < cur_n; ++i) {
            sel_idx = -1;
            for (uint32_t j = 0; j < cur_n; ++j) {
                if (!added[j] && (sel_idx == -1 || w[j] > w[sel_idx])) {
                    sel_idx = (int)j;
                }
            }
            if (sel_idx == -1) break;

            added[sel_idx] = 1;
            uint32_t v_sel = verts[sel_idx];

            if (i == cur_n - 1) {
                uint64_t cut_val = w[sel_idx];
                if (prev_idx < 0) prev_idx = sel_idx; // should not happen for cur_n >= 2
                uint32_t s_vert = verts[prev_idx];
                uint32_t t_vert = v_sel;

                if (cut_val < best_value) {
                    best_value = cut_val;
                    best_s_super = s_vert;
                    best_t_super = t_vert;
                    best_merge_prefix = merges_so_far;
                }

                merges.emplace_back(s_vert, t_vert);
                ++merges_so_far;

                // Contract t_vert into s_vert
                for (uint32_t j = 0; j < cur_n; ++j) {
                    uint32_t vj = verts[j];
                    if (vj == s_vert || vj == t_vert) continue;
                    cap[s_vert][vj] += cap[t_vert][vj];
                    cap[vj][s_vert] = cap[s_vert][vj];
                }

                // Remove t_vert from verts
                if ((uint32_t)sel_idx != cur_n - 1)
                    std::swap(verts[sel_idx], verts[cur_n - 1]);
                --cur_n;
                break;
            }

            prev_idx = sel_idx;
            // Update weights to un-added vertices
            for (uint32_t j = 0; j < cur_n; ++j) {
                if (!added[j]) {
                    w[j] += cap[v_sel][verts[j]];
                }
            }
        }
    }
}

// ====================== Stoer–Wagner: edge-list variant ======================

/*
 * Large-graph variant using a single edge list plus incidence lists.
 * Each undirected input edge is stored exactly once.
 * The algorithm runs in O(n * m) time in the worst case, but uses
 * only O(n + m) memory (apart from logging).
 */

struct SW_Edge {
    uint32_t u, v;       // current endpoints (change during contractions)
    uint32_t orig_u, orig_v; // original endpoints (never change)
    bool active;         // false if this edge became a self-loop or dead
};

void stoer_wagner_edgelist(
    uint32_t N,
    std::vector<SW_Edge> &edges,
    std::vector<std::vector<uint32_t>> &incident,
    std::vector<std::pair<uint32_t, uint32_t>> &merges,
    uint32_t &best_s_super,
    uint32_t &best_t_super,
    size_t &best_merge_prefix,
    uint64_t &best_value
) {
    std::vector<uint32_t> verts;
    verts.reserve(N);
    for (uint32_t i = 0; i < N; ++i) verts.push_back(i);

    std::vector<char> isActiveVertex(N, 1);

    std::vector<int32_t> posIndex(N, -1);
    std::vector<uint32_t> posStamp(N, 0);
    uint32_t curStamp = 1;

    best_value = std::numeric_limits<uint64_t>::max();
    best_s_super = 0;
    best_t_super = (N > 1 ? 1 : 0);
    best_merge_prefix = 0;
    size_t merges_so_far = 0;

    auto contract = [&](uint32_t keep, uint32_t rem) {
        if (keep == rem) return;
        isActiveVertex[rem] = 0;
        for (uint32_t idx : incident[rem]) {
            SW_Edge &e = edges[idx];
            if (!e.active) continue;
            uint32_t other = (e.u == rem ? e.v : e.u);
            if (!isActiveVertex[other] || other == keep) {
                // Edge becomes self-loop or connects to an already removed vertex
                e.active = false;
                continue;
            }
            // Redirect rem to keep
            if (e.u == rem) e.u = keep;
            else           e.v = keep;
            incident[keep].push_back(idx);
        }
        incident[rem].clear();
    };

    while (verts.size() > 1) {
        size_t k = verts.size();
        std::vector<uint64_t> w(k, 0);
        std::vector<char> added(k, 0);

        ++curStamp;
        for (size_t i = 0; i < k; ++i) {
            uint32_t v = verts[i];
            posIndex[v] = (int32_t)i;
            posStamp[v] = curStamp;
        }

        int prev_idx = -1;
        int sel_idx = -1;

        for (size_t step = 0; step < k; ++step) {
            sel_idx = -1;
            for (size_t j = 0; j < k; ++j) {
                if (!added[j] && (sel_idx == -1 || w[j] > w[sel_idx])) {
                    sel_idx = (int)j;
                }
            }
            if (sel_idx == -1) break;

            uint32_t vSel = verts[sel_idx];

            if (step == k - 1) {
                uint64_t cut_val = w[sel_idx];
                if (prev_idx < 0) prev_idx = sel_idx; // cur_k>=2 so prev_idx should be >=0
                uint32_t s_vert = verts[prev_idx];
                uint32_t t_vert = vSel;

                if (cut_val < best_value) {
                    best_value = cut_val;
                    best_s_super = s_vert;
                    best_t_super = t_vert;
                    best_merge_prefix = merges_so_far;
                }

                merges.emplace_back(s_vert, t_vert);
                ++merges_so_far;
                contract(s_vert, t_vert);

                if ((size_t)sel_idx != k - 1)
                    std::swap(verts[sel_idx], verts[k - 1]);
                verts.pop_back();
                break;
            }

            added[sel_idx] = 1;
            prev_idx = sel_idx;

            // Update weights for neighbors
            for (uint32_t idx : incident[vSel]) {
                SW_Edge &e = edges[idx];
                if (!e.active) continue;
                uint32_t other = (e.u == vSel ? e.v : e.u);
                if (!isActiveVertex[other]) continue;
                if (posStamp[other] != curStamp) continue; // vertex not in current verts
                int j = posIndex[other];
                if (added[j]) continue;
                ++w[j]; // unweighted edges, weight 1 per edge
            }
        }
    }
}

// ====================== Reconstruct cut via max-flow ======================

/*
 * Given:
 *  - N: number of original vertices
 *  - edgesOrig: list of original undirected edges (u,v) for matrix case
 *  - edgesSW:   list of SW_Edge (only .orig_u,.orig_v used) for edge-list case
 *  - merges:    list of (keep,rem) contractions in chronological order
 *  - best_s_super, best_t_super: supernodes at best phase
 *  - best_merge_prefix: number of merges BEFORE the phase producing best cut
 *
 * This function:
 *  1. Replays the first best_merge_prefix merges in a DSU to reconstruct the
 *     supernode partition at the start of the best phase.
 *  2. Identifies clusters containing best_s_super and best_t_super.
 *  3. Builds a flow network with:
 *        - original vertices 0..N-1
 *        - supersource SS = N
 *        - supersink  TT = N+1
 *     with:
 *        - capacity 1 on each undirected edge (as two directed arcs)
 *        - "infinite" capacity from SS to all vertices in s-cluster
 *        - "infinite" capacity from all vertices in t-cluster to TT
 *  4. Runs Dinic max-flow from SS to TT.
 *  5. From residual graph, takes set of vertices reachable from SS as S,
 *     and outputs all original edges crossing S vs V\S.
 *
 * It uses either edgesOrig (matrix case) or edgesSW (edge-list case),
 * depending on which vector is non-empty.
 */

void build_and_output_cut(
    uint32_t N,
    const std::vector<std::pair<uint32_t, uint32_t>> &edgesOrig,
    const std::vector<SW_Edge> &edgesSW,
    const std::vector<std::pair<uint32_t, uint32_t>> &merges,
    uint32_t best_s_super,
    uint32_t best_t_super,
    size_t best_merge_prefix
) {
    // Reconstruct clustering at the start of the best phase
    DSU dsu(N);
    for (size_t i = 0; i < best_merge_prefix; ++i) {
        dsu.unite((int)merges[i].first, (int)merges[i].second);
    }
    int compS = dsu.find((int)best_s_super);
    int compT = dsu.find((int)best_t_super);

    // Build flow graph
    int totalNodes = (int)N + 2;
    int SS = (int)N;
    int TT = (int)N + 1;
    Dinic dinic(totalNodes);

    // Add original undirected edges with capacity 1 in each direction
    if (!edgesOrig.empty()) {
        for (const auto &e : edgesOrig) {
            uint32_t u = e.first;
            uint32_t v = e.second;
            if (u == v) continue; // ignore self-loops if any slipped through
            dinic.addEdge((int)u, (int)v, 1);
            dinic.addEdge((int)v, (int)u, 1);
        }
    } else {
        for (const auto &e : edgesSW) {
            uint32_t u = e.orig_u;
            uint32_t v = e.orig_v;
            if (u == v) continue;
            dinic.addEdge((int)u, (int)v, 1);
            dinic.addEdge((int)v, (int)u, 1);
        }
    }

    // Choose a large enough "infinite" capacity.
    // We don't actually need the exact min-cut value here, so we can safely
    // use a large constant.
    int64_t INF = (int64_t)1e15;

    for (uint32_t i = 0; i < N; ++i) {
        int root = dsu.find((int)i);
        if (root == compS) {
            dinic.addEdge(SS, (int)i, INF);
        } else if (root == compT) {
            dinic.addEdge((int)i, TT, INF);
        }
    }

    (void)dinic.maxflow(SS, TT); // we only need the partition

    // Find vertices reachable from SS in residual graph
    std::vector<char> side(totalNodes, 0);
    std::deque<int> dq;
    dq.push_back(SS);
    side[SS] = 1;
    while (!dq.empty()) {
        int v = dq.front();
        dq.pop_front();
        for (const auto &fe : dinic.g[v]) {
            if (fe.cap > 0 && !side[fe.to]) {
                side[fe.to] = 1;
                dq.push_back(fe.to);
            }
        }
    }

    // Collect cut edges
    std::vector<std::pair<uint32_t, uint32_t>> cutEdges;
    if (!edgesOrig.empty()) {
        cutEdges.reserve(edgesOrig.size());
        for (const auto &e : edgesOrig) {
            uint32_t u = e.first;
            uint32_t v = e.second;
            if (u == v) continue;
            if (side[u] != side[v]) {
                cutEdges.emplace_back(u, v);
            }
        }
    } else {
        cutEdges.reserve(edgesSW.size());
        for (const auto &e : edgesSW) {
            uint32_t u = e.orig_u;
            uint32_t v = e.orig_v;
            if (u == v) continue;
            if (side[u] != side[v]) {
                cutEdges.emplace_back(u, v);
            }
        }
    }

    std::cout << cutEdges.size() << '\n';
    for (const auto &p : cutEdges) {
        std::cout << p.first << ' ' << p.second << '\n';
    }
}

// ================================ Main ================================

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint32_t N;
    uint64_t M;
    if (!(std::cin >> N >> M)) {
        return 0;
    }

    if (N <= 1 || M == 0) {
        // Trivial / degenerate cases. For a connected graph, this shouldn't occur.
        std::cout << 0 << '\n';
        return 0;
    }

    // Adaptive choice: for small N, use matrix-based Stoer–Wagner;
    // for larger N, use edge-list-based Stoer–Wagner.
    constexpr uint32_t MATRIX_N_THRESHOLD = 1000;

    if (N <= MATRIX_N_THRESHOLD) {
        // -------- Small graph: matrix-based Stoer–Wagner --------
        std::vector<std::vector<uint64_t>> cap(N, std::vector<uint64_t>(N, 0));
        std::vector<std::pair<uint32_t, uint32_t>> origEdges;
        origEdges.reserve((size_t)M);

        for (uint64_t i = 0; i < M; ++i) {
            uint64_t uu, vv;
            if (!(std::cin >> uu >> vv)) break;
            uint32_t u = (uint32_t)uu;
            uint32_t v = (uint32_t)vv;
            if (u == v) continue; // ignore self-loops; they don't affect edge-connectivity
            if (u >= N || v >= N) continue; // defensive, though input should be valid

            cap[u][v] += 1;
            cap[v][u] += 1;
            origEdges.emplace_back(u, v);
        }

        std::vector<std::pair<uint32_t, uint32_t>> merges;
        merges.reserve(N - 1);
        uint32_t best_s_super = 0, best_t_super = 1;
        size_t best_merge_prefix = 0;
        uint64_t best_value = 0;

        stoer_wagner_matrix(N, cap, merges, best_s_super,
                            best_t_super, best_merge_prefix, best_value);

        // Reconstruct and output cut
        std::vector<SW_Edge> dummySW; // unused
        build_and_output_cut(N, origEdges, dummySW, merges,
                             best_s_super, best_t_super, best_merge_prefix);
    } else {
        // -------- Large graph: edge-list-based Stoer–Wagner --------
        std::vector<SW_Edge> edges;
        edges.reserve((size_t)M);
        std::vector<std::vector<uint32_t>> incident(N);

        for (uint64_t i = 0; i < M; ++i) {
            uint64_t uu, vv;
            if (!(std::cin >> uu >> vv)) break;
            uint32_t u = (uint32_t)uu;
            uint32_t v = (uint32_t)vv;
            if (u == v) continue; // ignore self-loops
            if (u >= N || v >= N) continue; // defensive

            SW_Edge e;
            e.u = u;
            e.v = v;
            e.orig_u = u;
            e.orig_v = v;
            e.active = true;
            edges.push_back(e);
            uint32_t idx = (uint32_t)(edges.size() - 1);
            incident[u].push_back(idx);
            incident[v].push_back(idx);
        }

        std::vector<std::pair<uint32_t, uint32_t>> merges;
        merges.reserve(N - 1);
        uint32_t best_s_super = 0, best_t_super = 1;
        size_t best_merge_prefix = 0;
        uint64_t best_value = 0;

        stoer_wagner_edgelist(N, edges, incident, merges,
                              best_s_super, best_t_super,
                              best_merge_prefix, best_value);

        std::vector<std::pair<uint32_t, uint32_t>> dummyOrig; // unused
        build_and_output_cut(N, dummyOrig, edges, merges,
                             best_s_super, best_t_super, best_merge_prefix);
    }

    return 0;
}
