```csharp
using System;
using System.Collections.Generic;

class Program
{
    // Internal representation:
    // '0'-'8' : revealed numbers
    // '*'     : known/flagged mine
    // ' '     : unknown (may be mine or safe)
    // '.'     : deduced safe (used only internally by the solver)

    static void Main()
    {
        string? firstLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(firstLine))
        {
            Console.WriteLine("0");
            return;
        }

        var parts = firstLine.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
        {
            Console.WriteLine("0");
            return;
        }

        int height = int.Parse(parts[0]);
        int width = int.Parse(parts[1]);

        var grid = new char[height, width];

        for (int y = 0; y < height; y++)
        {
            string? line = Console.ReadLine() ?? string.Empty;
            if (line.Length < width)
                line = line.PadRight(width, ' ');
            for (int x = 0; x < width; x++)
                grid[y, x] = line[x];
        }

        Solve(grid, height, width);

        // Collect all deduced safe cells ('.')
        var safeCells = new List<(int x, int y)>();
        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                if (grid[y, x] == '.')
                    safeCells.Add((x, y));
            }
        }

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
            Console.WriteLine($"{x} {y}");
    }

    static void Solve(char[,] grid, int height, int width)
    {
        // 1. Fast, purely local logic (works well on any size)
        SimplePropagation(grid, height, width);

        // 2. Build global constraints, but only up to some scale limits
        //    to stay efficient on huge boards.
        const int MaxVarsForConstraints = 200_000;
        const int MaxEqsForConstraints = 200_000;

        BuildConstraints(
            grid, height, width,
            MaxVarsForConstraints, MaxEqsForConstraints,
            out var equations,
            out var varPositions,
            out var varIndexMap,
            out bool truncated);

        // If there are no frontier variables or no equations, nothing more to do.
        if (equations.Count == 0 || varPositions.Count == 0)
            return;

        // 3. Advanced logic: per-component exact solving via CSP search,
        //    but only for small components (adaptive).
        ApplyAdvancedSolver(grid, equations, varPositions);

        // 4. New mines found by advanced solver can unlock more trivial deductions.
        SimplePropagation(grid, height, width);
    }

    // ---------------------------------------------------------------------
    // 1. Simple local propagation: classic Minesweeper rules
    // ---------------------------------------------------------------------

    static void SimplePropagation(char[,] grid, int height, int width)
    {
        bool changed;
        do
        {
            changed = false;

            for (int y = 0; y < height; y++)
            {
                for (int x = 0; x < width; x++)
                {
                    char ch = grid[y, x];
                    if (ch < '0' || ch > '8')
                        continue;

                    int n = ch - '0';
                    int flagged = 0;
                    int unknownCount = 0;

                    // Up to 8 neighbors; store coordinates of unknowns on stack
                    Span<int> unksX = stackalloc int[8];
                    Span<int> unksY = stackalloc int[8];

                    for (int dy = -1; dy <= 1; dy++)
                    {
                        int ny = y + dy;
                        if (ny < 0 || ny >= height) continue;

                        for (int dx = -1; dx <= 1; dx++)
                        {
                            int nx = x + dx;
                            if (nx < 0 || nx >= width) continue;
                            if (dx == 0 && dy == 0) continue;

                            char nch = grid[ny, nx];
                            if (nch == '*')
                            {
                                flagged++;
                            }
                            else if (nch == ' ')
                            {
                                unksX[unknownCount] = nx;
                                unksY[unknownCount] = ny;
                                unknownCount++;
                            }
                            // '.' and digits are known safe and do not count as unknown.
                        }
                    }

                    int minesRemaining = n - flagged;
                    if (minesRemaining < 0 || minesRemaining > unknownCount)
                        continue; // inconsistent or not useful

                    if (minesRemaining == 0 && unknownCount > 0)
                    {
                        // All unknown neighbors are safe
                        for (int i = 0; i < unknownCount; i++)
                        {
                            int ux = unksX[i];
                            int uy = unksY[i];
                            if (grid[uy, ux] == ' ')
                            {
                                grid[uy, ux] = '.';
                                changed = true;
                            }
                        }
                    }
                    else if (minesRemaining == unknownCount && unknownCount > 0)
                    {
                        // All unknown neighbors are mines
                        for (int i = 0; i < unknownCount; i++)
                        {
                            int ux = unksX[i];
                            int uy = unksY[i];
                            if (grid[uy, ux] == ' ')
                            {
                                grid[uy, ux] = '*';
                                changed = true;
                            }
                        }
                    }
                }
            }
        } while (changed);
    }

    // ---------------------------------------------------------------------
    // 2. Constraint building
    // ---------------------------------------------------------------------

    struct Equation
    {
        public int MinesRequired; // rhs of equation
        public int[] Vars;        // indices into varPositions
    }

    static void BuildConstraints(
        char[,] grid,
        int height,
        int width,
        int maxVars,
        int maxEqs,
        out List<Equation> equations,
        out List<(int y, int x)> varPositions,
        out Dictionary<int, int> varIndexMap,
        out bool truncated)
    {
        equations = new List<Equation>();
        varPositions = new List<(int y, int x)>();
        varIndexMap = new Dictionary<int, int>();
        truncated = false;

        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                char ch = grid[y, x];
                if (ch < '0' || ch > '8')
                    continue;

                int n = ch - '0';
                int flagged = 0;
                int unknownCount = 0;
                Span<int> localVarIds = stackalloc int[8];

                for (int dy = -1; dy <= 1; dy++)
                {
                    int ny = y + dy;
                    if (ny < 0 || ny >= height) continue;

                    for (int dx = -1; dx <= 1; dx++)
                    {
                        int nx = x + dx;
                        if (nx < 0 || nx >= width) continue;
                        if (dx == 0 && dy == 0) continue;

                        char nch = grid[ny, nx];
                        if (nch == '*')
                        {
                            flagged++;
                        }
                        else if (nch == ' ')
                        {
                            int key = ny * width + nx;
                            if (!varIndexMap.TryGetValue(key, out int varIndex))
                            {
                                varIndex = varPositions.Count;
                                if (varIndex >= maxVars)
                                {
                                    truncated = true;
                                    goto Done;
                                }

                                varIndexMap[key] = varIndex;
                                varPositions.Add((ny, nx));
                            }

                            localVarIds[unknownCount] = varIndex;
                            unknownCount++;
                        }
                    }
                }

                int minesRemaining = n - flagged;
                // Only keep non-trivial equations (0 < minesRemaining < unknownCount)
                if (unknownCount > 0 &&
                    minesRemaining > 0 &&
                    minesRemaining < unknownCount)
                {
                    if (equations.Count >= maxEqs)
                    {
                        truncated = true;
                        goto Done;
                    }

                    var vars = new int[unknownCount];
                    for (int i = 0; i < unknownCount; i++)
                        vars[i] = localVarIds[i];

                    equations.Add(new Equation
                    {
                        MinesRequired = minesRemaining,
                        Vars = vars
                    });
                }
            }
        }

    Done:
        // If truncated, we keep the partial set of equations/variables we already built.
        // Ignoring some constraints never makes our deductions unsound; it only
        // makes them weaker (we consider a superset of possible mine placements).
        ;
    }

    // ---------------------------------------------------------------------
    // 3. Advanced CSP-based solver on small constraint components
    // ---------------------------------------------------------------------

    static void ApplyAdvancedSolver(
        char[,] grid,
        List<Equation> equations,
        List<(int y, int x)> varPositions)
    {
        int varCount = varPositions.Count;
        int eqCount = equations.Count;
        if (varCount == 0 || eqCount == 0)
            return;

        // We will only do full CSP search on components with at most this many variables.
        const int MaxComponentVarsForSearch = 18;

        // Build var -> equations adjacency
        var varToEqs = new List<int>[varCount];
        for (int eqId = 0; eqId < eqCount; eqId++)
        {
            var vars = equations[eqId].Vars;
            for (int i = 0; i < vars.Length; i++)
            {
                int v = vars[i];
                var list = varToEqs[v];
                if (list == null)
                {
                    list = new List<int>();
                    varToEqs[v] = list;
                }
                list.Add(eqId);
            }
        }

        var varVisited = new bool[varCount];
        var eqVisited = new bool[eqCount];
        var queue = new Queue<int>();

        for (int startVar = 0; startVar < varCount; startVar++)
        {
            // Skip variables that don't appear in any equation
            if (varVisited[startVar] || varToEqs[startVar] == null)
                continue;

            var compVars = new List<int>();
            var compEqs = new List<int>();

            queue.Clear();
            queue.Enqueue(startVar);
            varVisited[startVar] = true;
            compVars.Add(startVar);

            // BFS over bipartite graph (variables <-> equations)
            while (queue.Count > 0)
            {
                int v = queue.Dequeue();
                var eqList = varToEqs[v];
                if (eqList == null) continue;

                foreach (int eqId in eqList)
                {
                    if (eqVisited[eqId])
                        continue;

                    eqVisited[eqId] = true;
                    compEqs.Add(eqId);

                    var eqVars = equations[eqId].Vars;
                    for (int i = 0; i < eqVars.Length; i++)
                    {
                        int nv = eqVars[i];
                        if (!varVisited[nv])
                        {
                            varVisited[nv] = true;
                            queue.Enqueue(nv);
                            compVars.Add(nv);
                        }
                    }
                }
            }

            if (compVars.Count == 0 || compEqs.Count == 0)
                continue;

            // Only solve small components exactly; larger ones use only local logic.
            if (compVars.Count > MaxComponentVarsForSearch)
                continue;

            SolveComponent(grid, compVars, compEqs, equations, varPositions, varToEqs);
        }
    }

    static void SolveComponent(
        char[,] grid,
        List<int> compVarsGlobal,
        List<int> compEqsGlobal,
        List<Equation> equations,
        List<(int y, int x)> varPositions,
        List<int>[] varToEqsGlobal)
    {
        int nVars = compVarsGlobal.Count;
        int nEqs = compEqsGlobal.Count;

        if (nVars == 0 || nEqs == 0)
            return;

        // Map global variable id -> local id in [0..nVars)
        var g2lVar = new Dictionary<int, int>(nVars);
        for (int i = 0; i < nVars; i++)
            g2lVar[compVarsGlobal[i]] = i;

        // Build local equation structures
        var eqVarsLocal = new int[nEqs][];
        var eqRequired = new int[nEqs];
        var varToEqLocal = new List<int>[nVars];

        for (int leq = 0; leq < nEqs; leq++)
        {
            int geq = compEqsGlobal[leq];
            Equation eqG = equations[geq];
            int len = eqG.Vars.Length;
            var localVars = new int[len];

            for (int i = 0; i < len; i++)
            {
                int gVar = eqG.Vars[i];
                if (!g2lVar.TryGetValue(gVar, out int lVar))
                    continue; // Should not happen within a component, but safe.

                localVars[i] = lVar;
                var list = varToEqLocal[lVar];
                if (list == null)
                {
                    list = new List<int>();
                    varToEqLocal[lVar] = list;
                }
                list.Add(leq);
            }

            eqVarsLocal[leq] = localVars;
            eqRequired[leq] = eqG.MinesRequired;
        }

        var solver = new ComponentSolver(nVars, nEqs, eqVarsLocal, eqRequired, varToEqLocal);
        solver.Solve();

        if (!solver.HasSolution)
            return; // Inconsistent or no information.

        // Apply forced results back to global grid
        for (int lv = 0; lv < nVars; lv++)
        {
            bool alwaysSafe = solver.CanBeSafe[lv] && !solver.CanBeMine[lv];
            bool alwaysMine = solver.CanBeMine[lv] && !solver.CanBeSafe[lv];

            if (!alwaysSafe && !alwaysMine)
                continue;

            int gv = compVarsGlobal[lv];
            var (y, x) = varPositions[gv];

            if (alwaysSafe)
            {
                if (grid[y, x] == ' ')
                    grid[y, x] = '.';
            }
            else if (alwaysMine)
            {
                if (grid[y, x] == ' ')
                    grid[y, x] = '*';
            }
        }
    }

    // ---------------------------------------------------------------------
    // 3a. Per-component CSP solver with full search (for small components)
    // ---------------------------------------------------------------------

    sealed class ComponentSolver
    {
        private readonly int nVars;
        private readonly int nEqs;
        private readonly int[][] eqVars;
        private readonly int[] eqRequired;
        private readonly List<int>[] varToEq;

        private readonly sbyte[] varState;   // 0 = unknown, 1 = mine, -1 = safe
        private readonly int[] eqRemaining;  // remaining mines to place in each eq
        private readonly int[] eqUnassigned; // remaining unassigned vars in each eq

        private readonly int[] variableOrder; // heuristic ordering

        public bool[] CanBeSafe { get; }
        public bool[] CanBeMine { get; }
        public bool HasSolution { get; private set; }

        private readonly Stack<VarChange> varChangeStack;
        private readonly Stack<EqChange> eqChangeStack;
        private readonly Queue<int> assignQueue;

        private const sbyte Unknown = 0;
        private const sbyte Mine = 1;
        private const sbyte Safe = -1;

        private struct VarChange
        {
            public int VarId;
            public VarChange(int id) { VarId = id; }
        }

        private struct EqChange
        {
            public int EqId;
            public int PrevRem;
            public int PrevUnassigned;
            public EqChange(int id, int rem, int unass)
            {
                EqId = id;
                PrevRem = rem;
                PrevUnassigned = unass;
            }
        }

        public ComponentSolver(
            int nVars,
            int nEqs,
            int[][] eqVars,
            int[] eqRequired,
            List<int>[] varToEq)
        {
            this.nVars = nVars;
            this.nEqs = nEqs;
            this.eqVars = eqVars;
            this.eqRequired = eqRequired;
            this.varToEq = varToEq;

            varState = new sbyte[nVars];
            eqRemaining = new int[nEqs];
            eqUnassigned = new int[nEqs];

            variableOrder = new int[nVars];

            CanBeSafe = new bool[nVars];
            CanBeMine = new bool[nVars];

            varChangeStack = new Stack<VarChange>();
            eqChangeStack = new Stack<EqChange>();
            assignQueue = new Queue<int>();
        }

        public void Solve()
        {
            // Initialize equations
            for (int e = 0; e < nEqs; e++)
            {
                eqRemaining[e] = eqRequired[e];
                eqUnassigned[e] = eqVars[e].Length;
            }

            // Degree heuristic: variables appearing in many equations first
            var degrees = new int[nVars];
            for (int v = 0; v < nVars; v++)
            {
                variableOrder[v] = v;
                if (varToEq[v] != null)
                    degrees[v] = varToEq[v].Count;
            }
            Array.Sort(variableOrder, (a, b) => degrees[b].CompareTo(degrees[a]));

            HasSolution = false;
            DFS();
        }

        private void DFS()
        {
            // Find first unassigned variable
            int varId = -1;
            for (int i = 0; i < nVars; i++)
            {
                int v = variableOrder[i];
                if (varState[v] == Unknown)
                {
                    varId = v;
                    break;
                }
            }

            if (varId == -1)
            {
                // All variables assigned: record a solution
                HasSolution = true;
                for (int i = 0; i < nVars; i++)
                {
                    if (varState[i] == Mine)
                        CanBeMine[i] = true;
                    else if (varState[i] == Safe)
                        CanBeSafe[i] = true;
                }
                return;
            }

            // Branch 1: varId is safe
            int varStackStart = varChangeStack.Count;
            int eqStackStart = eqChangeStack.Count;

            if (AddAssignmentAndPropagate(varId, isMine: false))
                DFS();
            UndoTo(varStackStart, eqStackStart);

            // Branch 2: varId is a mine
            varStackStart = varChangeStack.Count;
            eqStackStart = eqChangeStack.Count;

            if (AddAssignmentAndPropagate(varId, isMine: true))
                DFS();
            UndoTo(varStackStart, eqStackStart);
        }

        private void UndoTo(int varStackStart, int eqStackStart)
        {
            while (varChangeStack.Count > varStackStart)
            {
                var vc = varChangeStack.Pop();
                varState[vc.VarId] = Unknown;
            }
            while (eqChangeStack.Count > eqStackStart)
            {
                var ec = eqChangeStack.Pop();
                eqRemaining[ec.EqId] = ec.PrevRem;
                eqUnassigned[ec.EqId] = ec.PrevUnassigned;
            }
            assignQueue.Clear();
        }

        private bool AddAssignmentAndPropagate(int varId, bool isMine)
        {
            assignQueue.Clear();
            if (!EnqueueAssignment(varId, isMine))
                return false;

            while (assignQueue.Count > 0)
            {
                int v = assignQueue.Dequeue();
                bool vIsMine = varState[v] == Mine;

                var eqList = varToEq[v];
                if (eqList == null)
                    continue;

                foreach (int eqId in eqList)
                {
                    // Save previous state for backtracking
                    eqChangeStack.Push(new EqChange(eqId, eqRemaining[eqId], eqUnassigned[eqId]));

                    eqUnassigned[eqId]--;
                    if (vIsMine)
                        eqRemaining[eqId]--;

                    int rem = eqRemaining[eqId];
                    int unassigned = eqUnassigned[eqId];

                    if (rem < 0 || rem > unassigned)
                        return false; // contradiction

                    if (rem == 0)
                    {
                        // All remaining variables in this equation must be safe
                        var vars = eqVars[eqId];
                        for (int i = 0; i < vars.Length; i++)
                        {
                            int other = vars[i];
                            if (varState[other] == Unknown)
                            {
                                if (!EnqueueAssignment(other, isMine: false))
                                    return false;
                            }
                        }
                    }
                    else if (rem == unassigned)
                    {
                        // All remaining variables in this equation must be mines
                        var vars = eqVars[eqId];
                        for (int i = 0; i < vars.Length; i++)
                        {
                            int other = vars[i];
                            if (varState[other] == Unknown)
                            {
                                if (!EnqueueAssignment(other, isMine: true))
                                    return false;
                            }
                        }
                    }
                }
            }

            return true;
        }

        private bool EnqueueAssignment(int varId, bool isMine)
        {
            sbyte newState = isMine ? Mine : Safe;
            sbyte current = varState[varId];

            if (current != Unknown)
            {
                // Already assigned; check consistency
                if (current == newState)
                    return true;
                return false;
            }

            varChangeStack.Push(new VarChange(varId));
            varState[varId] = newState;
            assignQueue.Enqueue(varId);
            return true;
        }
    }
}
```
