#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <cstdint>
#include <limits>

using namespace std;

/*
 * Suffix Automaton based solver for multi-string Longest Common Substring.
 * Build SAM on the shortest string, then for each other string
 * compute longest substring ending in each state, propagate along
 * suffix links, and keep the minimum over all strings.
 */
class MultiLCS_SAM {
public:
    struct State {
        int link;   // suffix link
        int len;    // length of the longest string in this state
        int head;   // head of adjacency list of outgoing edges
        int pos;    // position of the (rightmost) occurrence in base string
    };

    struct Edge {
        int to;
        int next;
        unsigned char c;
    };

    vector<State> st;
    vector<Edge> edges;
    int last;
    int curPos;

    vector<int> order; // states ordered by len ascending
    vector<int> best;  // per-string best match length
    vector<int> ans;   // minimum over all strings

    MultiLCS_SAM() : last(0), curPos(-1) {}

    void init(size_t maxLen) {
        st.clear();
        edges.clear();
        st.reserve(2 * maxLen + 1);
        edges.reserve(4 * maxLen + 1);

        // root state
        State root;
        root.link = -1;
        root.len  = 0;
        root.head = -1;
        root.pos  = -1;
        st.push_back(root);
        last = 0;
        curPos = -1;
    }

    inline int addState() {
        State s;
        s.link = 0;
        s.len  = 0;
        s.head = -1;
        s.pos  = -1;
        st.push_back(s);
        return static_cast<int>(st.size()) - 1;
    }

    inline void addEdge(int from, unsigned char c, int to) {
        Edge e;
        e.to   = to;
        e.next = st[from].head;
        e.c    = c;
        edges.push_back(e);
        st[from].head = static_cast<int>(edges.size()) - 1;
    }

    inline int getNext(int from, unsigned char c) const {
        for (int e = st[from].head; e != -1; e = edges[e].next) {
            if (edges[e].c == c) return edges[e].to;
        }
        return -1;
    }

    inline void setEdgeTarget(int from, unsigned char c, int newTo) {
        for (int e = st[from].head; e != -1; e = edges[e].next) {
            if (edges[e].c == c) {
                edges[e].to = newTo;
                return;
            }
        }
    }

    void extend(unsigned char ch) {
        ++curPos;

        int cur = addState();
        st[cur].len = st[last].len + 1;
        st[cur].pos = curPos;

        int p = last;

        // 1. Add transitions for ch from states that don't have it yet.
        while (p != -1 && getNext(p, ch) == -1) {
            addEdge(p, ch, cur);
            p = st[p].link;
        }

        if (p == -1) {
            // No state had transition on ch; link to root.
            st[cur].link = 0;
        } else {
            int q = getNext(p, ch);
            if (st[p].len + 1 == st[q].len) {
                // Directly attach.
                st[cur].link = q;
            } else {
                // Need clone.
                int clone = addState();
                st[clone].len = st[p].len + 1;
                st[clone].pos = st[q].pos;

                // Copy transitions of q to clone.
                for (int e = st[q].head; e != -1; e = edges[e].next) {
                    addEdge(clone, edges[e].c, edges[e].to);
                }

                st[clone].link = st[q].link;

                // Redirect transitions that used to go to q.
                while (p != -1 && getNext(p, ch) == q) {
                    setEdgeTarget(p, ch, clone);
                    p = st[p].link;
                }

                st[q].link   = clone;
                st[cur].link = clone;
            }
        }

        last = cur;
    }

    void build(const string &s) {
        init(s.size());
        for (unsigned char ch : s) {
            extend(ch);
        }
    }

    void prepare() {
        int sz = static_cast<int>(st.size());
        order.assign(sz, 0);
        ans.assign(sz, 0);
        best.assign(sz, 0);

        int maxLen = 0;
        for (int i = 0; i < sz; ++i) {
            if (st[i].len > maxLen) maxLen = st[i].len;
        }

        vector<int> cnt(maxLen + 1, 0);
        for (int i = 0; i < sz; ++i) {
            ++cnt[st[i].len];
        }
        for (int i = 1; i <= maxLen; ++i) {
            cnt[i] += cnt[i - 1];
        }
        for (int i = sz - 1; i >= 0; --i) {
            order[--cnt[st[i].len]] = i;
        }

        // Initial minimal length for each state is its own len (from base string).
        for (int i = 0; i < sz; ++i) {
            ans[i] = st[i].len;
        }
    }

    void match(const string &t) {
        int sz = static_cast<int>(st.size());
        std::fill(best.begin(), best.end(), 0);

        int v = 0;
        int l = 0;

        for (unsigned char ch : t) {
            while (v && getNext(v, ch) == -1) {
                v = st[v].link;
                l = st[v].len;
            }
            int to = getNext(v, ch);
            if (to != -1) {
                v = to;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (best[v] < l) best[v] = l;
        }

        // Propagate 'best' values up the suffix link tree.
        for (int i = sz - 1; i > 0; --i) {
            int state  = order[i];
            int parent = st[state].link;
            int val    = best[state];
            if (val > 0) {
                if (val > st[parent].len) val = st[parent].len;
                if (best[parent] < val) best[parent] = val;
            }
        }

        // Update global minimums.
        for (int i = 0; i < sz; ++i) {
            if (ans[i] > best[i]) ans[i] = best[i];
        }
    }

    string getLCS(const string &base) const {
        int sz = static_cast<int>(st.size());
        int bestLen = 0;
        int bestState = 0;
        for (int i = 0; i < sz; ++i) {
            if (ans[i] > bestLen) {
                bestLen = ans[i];
                bestState = i;
            }
        }
        if (bestLen <= 0) return string();

        int endPos   = st[bestState].pos;
        int startPos = endPos - bestLen + 1;
        if (startPos < 0) startPos = 0;
        return base.substr(static_cast<size_t>(startPos),
                           static_cast<size_t>(bestLen));
    }
};

/*
 * Rolling-hash + binary search solver for moderate sizes.
 * Uses 64-bit hash modulo 2^64. Collision probability is negligible.
 */
string solveWithRollingHash(const vector<string> &strs,
                            int baseIdx,
                            size_t minLen) {
    const string &base = strs[baseIdx];
    if (minLen == 0) return string();

    const uint64_t BASE = 1315423911ULL;

    // Precompute powers up to minLen
    vector<uint64_t> powBase(minLen + 1);
    powBase[0] = 1;
    for (size_t i = 1; i <= minLen; ++i) {
        powBase[i] = powBase[i - 1] * BASE;
    }

    auto existsLen = [&](size_t L, string &example) -> bool {
        if (L == 0) {
            example.clear();
            return true;
        }

        const string &s0 = base;
        size_t n0 = s0.size();
        if (n0 < L) return false;

        // Prefix hash for base
        vector<uint64_t> pref0(n0 + 1);
        pref0[0] = 0;
        for (size_t i = 0; i < n0; ++i) {
            pref0[i + 1] = pref0[i] * BASE + static_cast<unsigned char>(s0[i]);
        }

        unordered_set<uint64_t> cur;
        cur.reserve((n0 - L + 1) * 2 + 1);

        for (size_t i = 0; i + L <= n0; ++i) {
            uint64_t h =
                pref0[i + L] - pref0[i] * powBase[L]; // mod 2^64 by overflow
            cur.insert(h);
        }
        if (cur.empty()) return false;

        int N = static_cast<int>(strs.size());
        for (int idx = 0; idx < N; ++idx) {
            if (idx == baseIdx) continue;

            const string &s = strs[idx];
            size_t m = s.size();
            if (m < L) return false;

            vector<uint64_t> pref(m + 1);
            pref[0] = 0;
            for (size_t i = 0; i < m; ++i) {
                pref[i + 1] =
                    pref[i] * BASE + static_cast<unsigned char>(s[i]);
            }

            unordered_set<uint64_t> next;
            size_t maxSub = m - L + 1;
            size_t reserveSize =
                std::min(cur.size(), maxSub) * 2 + 1;
            next.reserve(reserveSize);

            for (size_t i = 0; i + L <= m; ++i) {
                uint64_t h =
                    pref[i + L] - pref[i] * powBase[L];
                if (cur.find(h) != cur.end()) {
                    next.insert(h);
                }
            }

            if (next.empty()) return false;
            cur.swap(next);
        }

        if (cur.empty()) return false;

        // Pick one hash and reconstruct the substring from base.
        uint64_t target = *cur.begin();
        for (size_t i = 0; i + L <= n0; ++i) {
            uint64_t h =
                pref0[i + L] - pref0[i] * powBase[L];
            if (h == target) {
                example = s0.substr(i, L);
                return true;
            }
        }
        // Should not happen, but be safe:
        return false;
    };

    string best;
    size_t low = 1, high = minLen;
    while (low <= high) {
        size_t mid = (low + high) / 2;
        string cand;
        if (existsLen(mid, cand)) {
            best.swap(cand);
            low = mid + 1;
        } else {
            if (mid == 0) break;
            high = mid - 1;
        }
    }
    return best;
}

/*
 * SAM-based solver for large / extreme cases.
 */
string solveWithSAM(const vector<string> &strs, int baseIdx) {
    const string &base = strs[baseIdx];

    MultiLCS_SAM sam;
    sam.build(base);
    sam.prepare();

    int N = static_cast<int>(strs.size());
    for (int i = 0; i < N; ++i) {
        if (i == baseIdx) continue;
        sam.match(strs[i]);
    }

    return sam.getLCS(base);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy); // consume end of line after N

    if (N <= 0) {
        cout << '\n';
        return 0;
    }

    vector<string> strs(N);
    size_t totalLen = 0;
    size_t minLen   = std::numeric_limits<size_t>::max();
    int    minIdx   = 0;

    for (int i = 0; i < N; ++i) {
        std::getline(cin, strs[i]);
        totalLen += strs[i].size();
        if (strs[i].size() < minLen) {
            minLen = strs[i].size();
            minIdx = i;
        }
    }

    // Edge cases
    if (N == 1) {
        cout << strs[0] << '\n';
        return 0;
    }
    if (minLen == 0) {
        // At least one empty string -> longest common substring is empty.
        cout << '\n';
        return 0;
    }

    // Heuristic: choose algorithm based on scale.
    bool useSAM = false;

    // Many strings or very long strings or large total length -> SAM.
    if (N > 10) useSAM = true;
    if (minLen > 200000) useSAM = true;
    if (totalLen > 20000000ULL) useSAM = true;

    string result;
    if (useSAM) {
        result = solveWithSAM(strs, minIdx);
    } else {
        result = solveWithRollingHash(strs, minIdx, minLen);
    }

    cout << result << '\n';
    return 0;
}
