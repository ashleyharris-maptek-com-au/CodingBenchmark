using System;
using System.Globalization;

namespace OrbitalTSP
{
    struct Vector3
    {
        public double X, Y, Z;

        public Vector3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vector3 operator +(Vector3 a, Vector3 b)
        {
            return new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vector3 operator -(Vector3 a, Vector3 b)
        {
            return new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vector3 operator *(Vector3 a, double s)
        {
            return new Vector3(a.X * s, a.Y * s, a.Z * s);
        }

        public static Vector3 operator *(double s, Vector3 a)
        {
            return new Vector3(a.X * s, a.Y * s, a.Z * s);
        }

        public double Norm()
        {
            return Math.Sqrt(X * X + Y * Y + Z * Z);
        }

        public static double Dot(Vector3 a, Vector3 b)
        {
            return a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        }

        public static Vector3 Cross(Vector3 a, Vector3 b)
        {
            return new Vector3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X
            );
        }
    }

    struct Orbit
    {
        public Vector3 R;
        public Vector3 V;
        public Vector3 Hhat;  // unit angular momentum vector
        public double Rmag;
        public double Vmag;
        public double CharRadius;
        public double CharSpeed;
        public double Eccentricity;

        public static Orbit FromState(double x, double y, double z, double vx, double vy, double vz, double mu)
        {
            Vector3 r = new Vector3(x, y, z);
            Vector3 v = new Vector3(vx, vy, vz);
            double rMag = r.Norm();
            double vMag = v.Norm();

            Vector3 h = Vector3.Cross(r, v);
            double hMag = h.Norm();
            Vector3 hHat;
            if (hMag > 0)
            {
                double invH = 1.0 / hMag;
                hHat = h * invH;
            }
            else
            {
                // Degenerate case, choose arbitrary normal
                hHat = new Vector3(0, 0, 1);
            }

            double energy = 0.5 * vMag * vMag - mu / rMag;
            double a;
            if (Math.Abs(energy) < 1e-9)
                a = 1e9; // very high semi-major axis
            else
                a = -mu / (2.0 * energy);

            Vector3 eVec = Vector3.Cross(v, h) * (1.0 / mu) - r * (1.0 / rMag);
            double ecc = eVec.Norm();

            double charRadius;
            if (a > 0.0 && ecc < 0.99)
                charRadius = a;           // bound elliptic orbit
            else
                charRadius = rMag;        // fallback to instantaneous radius

            if (charRadius < 1.0)
                charRadius = 1.0;

            double charSpeed = Math.Sqrt(mu / charRadius);

            Orbit o = new Orbit
            {
                R = r,
                V = v,
                Hhat = hHat,
                Rmag = rMag,
                Vmag = vMag,
                CharRadius = charRadius,
                CharSpeed = charSpeed,
                Eccentricity = ecc
            };
            return o;
        }
    }

    class Program
    {
        static void Main()
        {
            string input = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(input))
                return;

            string[] tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            int t = 0;
            var ci = CultureInfo.InvariantCulture;

            int nStations = int.Parse(tokens[t++], ci);
            double mu = double.Parse(tokens[t++], ci);

            Orbit[] orbits = new Orbit[nStations + 1];

            // Start orbit
            double sx = double.Parse(tokens[t++], ci);
            double sy = double.Parse(tokens[t++], ci);
            double sz = double.Parse(tokens[t++], ci);
            double svx = double.Parse(tokens[t++], ci);
            double svy = double.Parse(tokens[t++], ci);
            double svz = double.Parse(tokens[t++], ci);
            orbits[0] = Orbit.FromState(sx, sy, sz, svx, svy, svz, mu);

            // Station orbits
            for (int i = 0; i < nStations; i++)
            {
                double x = double.Parse(tokens[t++], ci);
                double y = double.Parse(tokens[t++], ci);
                double z = double.Parse(tokens[t++], ci);
                double vx = double.Parse(tokens[t++], ci);
                double vy = double.Parse(tokens[t++], ci);
                double vz = double.Parse(tokens[t++], ci);
                orbits[i + 1] = Orbit.FromState(x, y, z, vx, vy, vz, mu);
            }

            int totalNodes = nStations + 1;

            // Precompute pairwise delta-V estimates
            double[][] cost = new double[totalNodes][];
            for (int i = 0; i < totalNodes; i++)
                cost[i] = new double[totalNodes];

            for (int i = 0; i < totalNodes; i++)
            {
                cost[i][i] = 0.0;
                for (int j = i + 1; j < totalNodes; j++)
                {
                    double dv = EstimateDeltaV(orbits[i], orbits[j], mu);
                    cost[i][j] = dv;
                    cost[j][i] = dv;
                }
            }

            int[] order;
            const int ExactThreshold = 16;
            if (nStations <= ExactThreshold)
            {
                order = SolveExactDP(nStations, cost);
            }
            else
            {
                order = SolveHeuristic(nStations, cost);
            }

            // Output station visitation order (indices 0..N-1)
            if (order.Length > 0)
            {
                Console.Write(order[0].ToString(ci));
                for (int i = 1; i < order.Length; i++)
                {
                    Console.Write(' ');
                    Console.Write(order[i].ToString(ci));
                }
            }
            Console.WriteLine();
        }

        static double EstimateDeltaV(Orbit a, Orbit b, double mu)
        {
            // Plane change component between orbital planes
            double dot = Vector3.Dot(a.Hhat, b.Hhat);
            if (dot > 1.0) dot = 1.0;
            if (dot < -1.0) dot = -1.0;
            double dPlane = Math.Acos(dot); // radians
            double vPlane = Math.Min(a.CharSpeed, b.CharSpeed);
            double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * dPlane);

            // Altitude / energy change via Hohmann-like transfer between circularized orbits
            double r1 = a.CharRadius;
            double r2 = b.CharRadius;
            double dvOrbit;

            if (r1 <= 0.0 || r2 <= 0.0)
            {
                dvOrbit = Math.Abs(a.Vmag - b.Vmag);
            }
            else
            {
                double relDiff = Math.Abs(r1 - r2) / ((r1 + r2) * 0.5);
                if (relDiff < 1e-6)
                {
                    dvOrbit = 0.0;
                }
                else
                {
                    double v1 = Math.Sqrt(mu / r1);
                    double v2 = Math.Sqrt(mu / r2);
                    double aTrans = 0.5 * (r1 + r2);
                    double vPeriTrans = Math.Sqrt(mu * (2.0 / r1 - 1.0 / aTrans));
                    double vApoTrans = Math.Sqrt(mu * (2.0 / r2 - 1.0 / aTrans));
                    double dv1 = Math.Abs(vPeriTrans - v1);
                    double dv2 = Math.Abs(v2 - vApoTrans);
                    dvOrbit = dv1 + dv2;
                }
            }

            double totalDV = dvPlane + dvOrbit;
            return totalDV;
        }

        // Exact Held-Karp DP for Hamiltonian path starting at node 0
        static int[] SolveExactDP(int nStations, double[][] cost)
        {
            int m = nStations; // stations indexed 0..m-1, global nodes 1..m
            int fullMask = 1 << m;

            double[,] dp = new double[fullMask, m];
            int[,] parent = new int[fullMask, m];

            for (int mask = 0; mask < fullMask; mask++)
            {
                for (int j = 0; j < m; j++)
                {
                    dp[mask, j] = double.PositiveInfinity;
                    parent[mask, j] = -1;
                }
            }

            // Base: paths that go directly from start (0) to station j
            for (int j = 0; j < m; j++)
            {
                int mask = 1 << j;
                dp[mask, j] = cost[0][j + 1];
                parent[mask, j] = -1;
            }

            for (int mask = 1; mask < fullMask; mask++)
            {
                for (int j = 0; j < m; j++)
                {
                    if ((mask & (1 << j)) == 0)
                        continue;

                    int prevMask = mask ^ (1 << j);
                    if (prevMask == 0)
                        continue; // base case already set

                    double bestVal = dp[mask, j];
                    int bestK = parent[mask, j];

                    for (int k = 0; k < m; k++)
                    {
                        if ((prevMask & (1 << k)) == 0)
                            continue;

                        double prevCost = dp[prevMask, k];
                        if (double.IsPositiveInfinity(prevCost))
                            continue;

                        double cand = prevCost + cost[k + 1][j + 1];
                        if (cand < bestVal)
                        {
                            bestVal = cand;
                            bestK = k;
                        }
                    }

                    dp[mask, j] = bestVal;
                    parent[mask, j] = bestK;
                }
            }

            // Choose best ending station
            int finalMask = fullMask - 1;
            double bestPath = double.PositiveInfinity;
            int endIndex = 0;
            for (int j = 0; j < m; j++)
            {
                double c = dp[finalMask, j];
                if (c < bestPath)
                {
                    bestPath = c;
                    endIndex = j;
                }
            }

            // Reconstruct station order (indices 0..m-1)
            int[] order = new int[m];
            int idx = m - 1;
            int maskReconstruct = finalMask;
            int current = endIndex;
            while (maskReconstruct != 0)
            {
                order[idx--] = current;
                int prev = parent[maskReconstruct, current];
                maskReconstruct ^= 1 << current;
                current = prev;
            }

            return order;
        }

        // Heuristic TSP solver for larger instances: multi-start nearest neighbor + 2-opt
        static int[] SolveHeuristic(int nStations, double[][] cost)
        {
            int totalNodes = nStations + 1; // node 0 is start
            int m = nStations;

            int maxSeeds = Math.Min(12, m);
            double bestCost = double.PositiveInfinity;
            int[] bestRoute = null;

            bool[] chosenFirst = new bool[totalNodes];
            chosenFirst[0] = true;

            for (int s = 0; s < maxSeeds; s++)
            {
                int firstNode = -1;
                double bestFirstCost = double.PositiveInfinity;

                for (int j = 1; j <= m; j++)
                {
                    if (chosenFirst[j])
                        continue;

                    double c = cost[0][j];
                    if (c < bestFirstCost)
                    {
                        bestFirstCost = c;
                        firstNode = j;
                    }
                }

                if (firstNode == -1)
                    break;

                chosenFirst[firstNode] = true;

                int[] route = BuildNearestNeighborRoute(totalNodes, firstNode, cost);
                TwoOpt(route, cost);
                double routeCost = PathCost(route, cost);

                if (routeCost < bestCost)
                {
                    bestCost = routeCost;
                    bestRoute = route;
                }
            }

            if (bestRoute == null)
            {
                // Fallback: simple sequential order if something went wrong
                bestRoute = new int[totalNodes];
                for (int i = 0; i < totalNodes; i++)
                    bestRoute[i] = i;
            }

            // Convert route (nodes 0..m) to station indices 0..nStations-1
            int[] order = new int[m];
            for (int i = 1; i < bestRoute.Length; i++)
            {
                order[i - 1] = bestRoute[i] - 1;
            }
            return order;
        }

        static int[] BuildNearestNeighborRoute(int totalNodes, int firstNode, double[][] cost)
        {
            int[] route = new int[totalNodes];
            bool[] used = new bool[totalNodes];

            route[0] = 0;
            used[0] = true;

            route[1] = firstNode;
            used[firstNode] = true;

            int current = firstNode;
            for (int pos = 2; pos < totalNodes; pos++)
            {
                int best = -1;
                double bestC = double.PositiveInfinity;

                for (int j = 1; j < totalNodes; j++)
                {
                    if (used[j])
                        continue;

                    double c = cost[current][j];
                    if (c < bestC)
                    {
                        bestC = c;
                        best = j;
                    }
                }

                if (best == -1)
                {
                    // Fallback in pathological case
                    for (int j = 1; j < totalNodes; j++)
                    {
                        if (!used[j])
                        {
                            best = j;
                            break;
                        }
                    }
                }

                route[pos] = best;
                used[best] = true;
                current = best;
            }

            return route;
        }

        static void TwoOpt(int[] route, double[][] cost)
        {
            int n = route.Length;
            if (n <= 3)
                return;

            bool improved = true;
            while (improved)
            {
                improved = false;
                // i from 1 to n-3 (never move the starting node at index 0)
                for (int i = 1; i < n - 2; i++)
                {
                    int a = route[i - 1];
                    int b = route[i];

                    for (int j = i + 1; j < n - 1; j++)
                    {
                        int c = route[j];
                        int d = route[j + 1];

                        double delta = (cost[a][c] + cost[b][d]) - (cost[a][b] + cost[c][d]);
                        if (delta < -1e-9)
                        {
                            Array.Reverse(route, i, j - i + 1);
                            improved = true;
                        }
                    }
                }
            }
        }

        static double PathCost(int[] route, double[][] cost)
        {
            double c = 0.0;
            for (int i = 0; i < route.Length - 1; i++)
            {
                c += cost[route[i]][route[i + 1]];
            }
            return c;
        }
    }
}
