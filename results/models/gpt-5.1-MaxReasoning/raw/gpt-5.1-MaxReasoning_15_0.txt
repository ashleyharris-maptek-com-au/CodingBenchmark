use std::f64::consts::SQRT_2;
use std::io::Read;

#[derive(Clone, Copy)]
struct Point2 {
    x: f64,
    y: f64,
}

fn cross(o: Point2, a: Point2, b: Point2) -> f64 {
    (a.x - o.x) * (b.y - o.y) - (a.y - o.y) * (b.x - o.x)
}

fn convex_hull(mut pts: Vec<Point2>) -> Vec<Point2> {
    use std::cmp::Ordering;

    if pts.len() <= 1 {
        return pts;
    }

    // Sort by x, then y (with small epsilon to avoid issues)
    pts.sort_by(|a, b| {
        let dx = a.x - b.x;
        if dx.abs() > 1e-9 {
            if dx < 0.0 {
                Ordering::Less
            } else {
                Ordering::Greater
            }
        } else {
            let dy = a.y - b.y;
            if dy.abs() > 1e-9 {
                if dy < 0.0 {
                    Ordering::Less
                } else {
                    Ordering::Greater
                }
            } else {
                Ordering::Equal
            }
        }
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1e-9 && (a.y - b.y).abs() < 1e-9);

    if pts.len() <= 2 {
        return pts;
    }

    let mut lower: Vec<Point2> = Vec::with_capacity(pts.len());
    for &p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            if cross(lower[l - 2], lower[l - 1], p) <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }

    let mut upper: Vec<Point2> = Vec::with_capacity(pts.len());
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            if cross(upper[l - 2], upper[l - 1], p) <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }

    upper.pop();
    lower.pop();
    lower.extend(upper);
    lower
}

fn polygon_centroid_and_area(poly: &[Point2]) -> (Point2, f64) {
    let n = poly.len();
    if n == 0 {
        return (Point2 { x: 0.0, y: 0.0 }, 0.0);
    }
    if n == 1 {
        return (poly[0], 0.0);
    }

    let mut a = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        let cross = p.x * q.y - q.x * p.y;
        a += cross;
        cx += (p.x + q.x) * cross;
        cy += (p.y + q.y) * cross;
    }
    if a.abs() < 1e-12 {
        // Degenerate polygon: fall back to average of vertices
        let mut sx = 0.0;
        let mut sy = 0.0;
        for p in poly {
            sx += p.x;
            sy += p.y;
        }
        let nf = n as f64;
        return (
            Point2 {
                x: sx / nf,
                y: sy / nf,
            },
            0.0,
        );
    }
    a *= 0.5;
    cx /= 6.0 * a;
    cy /= 6.0 * a;
    (
        Point2 {
            x: cx,
            y: cy,
        },
        a.abs(),
    )
}

fn polygon_inradius(poly: &[Point2], center: Point2) -> f64 {
    let n = poly.len();
    if n < 2 {
        return 1.0;
    }
    let mut r = std::f64::MAX;
    for i in 0..n {
        let p = poly[i];
        let q = poly[(i + 1) % n];
        let vx = q.x - p.x;
        let vy = q.y - p.y;
        let wx = center.x - p.x;
        let wy = center.y - p.y;
        let area2 = (vx * wy - vy * wx).abs();
        let len = (vx * vx + vy * vy).sqrt();
        if len > 1e-12 {
            let dist = area2 / len;
            if dist < r {
                r = dist;
            }
        }
    }
    if r.is_finite() && r > 1e-9 {
        r
    } else {
        1e-3
    }
}

fn project_ref_tetra(sx: f64, sy: f64, sz: f64) -> Vec<Point2> {
    // Standard reference tetrahedron vertices
    const REF_VERTS: [(f64, f64, f64); 4] = [
        (1.0, 0.0, -0.707),
        (-1.0, 0.0, -0.707),
        (0.0, 1.0, 0.707),
        (0.0, -1.0, 0.707),
    ];

    let mut pts: Vec<Point2> = Vec::with_capacity(4);
    for &(x, y, z) in REF_VERTS.iter() {
        // Project along sun direction (sx, sy, sz) to z=0 plane
        let t = -z / sz;
        let px = x + t * sx;
        let py = y + t * sy;
        pts.push(Point2 { x: px, y: py });
    }
    convex_hull(pts)
}

fn ref_radius() -> f64 {
    // Bounding sphere radius of reference tetrahedron
    const REF_VERTS: [(f64, f64, f64); 4] = [
        (1.0, 0.0, -0.707),
        (-1.0, 0.0, -0.707),
        (0.0, 1.0, 0.707),
        (0.0, -1.0, 0.707),
    ];
    let mut r0 = 0.0;
    for &(x, y, z) in REF_VERTS.iter() {
        let d = (x * x + y * y + z * z).sqrt();
        if d > r0 {
            r0 = d;
        }
    }
    r0
}

struct Tetra {
    px: f64,
    py: f64,
    pz: f64,
    qw: f64,
    qx: f64,
    qy: f64,
    qz: f64,
    scale: f64,
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    std::io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(t) => t.parse().unwrap_or(0usize),
        None => 0usize,
    };
    if nv == 0 {
        println!("0");
        return;
    }

    let mut poly: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push((x, y));
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();

    // Sun direction must not be parallel to ground
    if sz.abs() < 1e-12 {
        // Degenerate, but avoid division by zero; no reasonable shadow possible
        println!("0");
        return;
    }

    // Polygon bounding box
    let mut min_x = poly[0].0;
    let mut max_x = poly[0].0;
    let mut min_y = poly[0].1;
    let mut max_y = poly[0].1;
    for &(x, y) in &poly {
        if x < min_x {
            min_x = x;
        }
        if x > max_x {
            max_x = x;
        }
        if y < min_y {
            min_y = y;
        }
        if y > max_y {
            max_y = y;
        }
    }
    let w = max_x - min_x;
    let h = max_y - min_y;
    let diag_half = ((w * w + h * h).sqrt()) * 0.5;

    // Shadow of one reference tetrahedron (scale=1, no rotation, centered at origin)
    let hull = project_ref_tetra(sx, sy, sz);
    let (center2d, _area) = polygon_centroid_and_area(&hull);
    let inradius = polygon_inradius(&hull, center2d);

    // Strategy decision: small polygons -> single tetra; larger -> grid covering.
    let max_scale = 10.0_f64;
    let can_single = diag_half <= inradius * max_scale;

    let mut tetras: Vec<Tetra> = Vec::new();

    if can_single {
        // --- Small-scale strategy: single (or minimal) tetrahedron ---

        // Choose scale just large enough to cover bounding box, but <= max_scale
        let mut needed = if inradius > 1e-9 {
            diag_half / inradius
        } else {
            1.0
        };
        // Inflate slightly for numerical safety
        needed *= 1.05;
        if needed < 1.0 {
            needed = 1.0;
        }
        if needed > max_scale {
            needed = max_scale;
        }
        let scale = needed;

        let bx = 0.5 * (min_x + max_x);
        let by = 0.5 * (min_y + max_y);
        let target_center = Point2 { x: bx, y: by };

        // We want the inscribed-disc center of the shadow at 'target_center'
        // For P=(px,py,pz) with pz=0, π(P)=(px,py). The shadow is
        // π(P) + scale * S0, whose inscribed-disc center is:
        //   π(P) + scale * center2d
        // So choose px,py so that this equals target_center.
        let px0 = target_center.x - scale * center2d.x;
        let py0 = target_center.y - scale * center2d.y;
        let pz0 = 0.0;

        tetras.push(Tetra {
            px: px0,
            py: py0,
            pz: pz0,
            qw: 1.0,
            qx: 0.0,
            qy: 0.0,
            qz: 0.0,
            scale,
        });
    } else {
        // --- Large-scale strategy: grid of tetrahedrons covering the bounding box ---

        let scale = max_scale;
        let r_scaled = inradius * scale;

        // Square grid of centers with spacing that guarantees coverage by discs
        // of radius r_scaled, then each disc is contained in a tetrahedron shadow.
        let step = r_scaled * SQRT_2;

        // Expand the grid slightly beyond the bounding box for edge coverage
        let x0 = min_x - step;
        let x1 = max_x + step;
        let y0 = min_y - step;
        let y1 = max_y + step;

        let nx = ((x1 - x0) / step).ceil() as usize;
        let ny = ((y1 - y0) / step).ceil() as usize;

        // Non-intersection: separate tetrahedrons along sun direction
        let len_s = (sx * sx + sy * sy + sz * sz).sqrt();
        let (ux, uy, uz) = if len_s > 0.0 {
            (sx / len_s, sy / len_s, sz / len_s)
        } else {
            (0.0, 0.0, 1.0)
        };
        let r0 = ref_radius();
        let r_max = scale * r0;
        let shift_mag = 10.0 * r_max; // > 2*r_max ensures disjoint bounding spheres
        let shift_x = ux * shift_mag;
        let shift_y = uy * shift_mag;
        let shift_z = uz * shift_mag;

        let mut idx: usize = 0;
        for iy in 0..=ny {
            let cy = y0 + (iy as f64) * step;
            for ix in 0..=nx {
                let cx = x0 + (ix as f64) * step;
                let target_center = Point2 { x: cx, y: cy };

                // Base center for this tetra before separating along sun direction
                let px_base = target_center.x - scale * center2d.x;
                let py_base = target_center.y - scale * center2d.y;
                let pz_base = 0.0;

                // Shift along sun direction to avoid any 3D intersections.
                let fidx = idx as f64;
                let px = px_base + shift_x * fidx;
                let py = py_base + shift_y * fidx;
                let pz = pz_base + shift_z * fidx;

                tetras.push(Tetra {
                    px,
                    py,
                    pz,
                    qw: 1.0,
                    qx: 0.0,
                    qy: 0.0,
                    qz: 0.0,
                    scale,
                });
                idx += 1;
            }
        }
    }

    // Output
    println!("{}", tetras.len());
    for t in tetras {
        println!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            t.px, t.py, t.pz, t.qw, t.qx, t.qy, t.qz, t.scale
        );
    }
}
