```rust
// 2D Polygon Cutting Stock heuristic solver
// ----------------------------------------
// This solution is designed to be:
//   * Fast and scalable for up to 5000+ pieces (large-mode)
//   * Geometrically more careful for small instances (small-mode)
// It uses simple rectangle-based packing and only two orientations (0 and 90°).
// For large instances, it prioritizes speed over exact geometric validity.

use std::f64::consts::FRAC_PI_2;
use std::io::{self, Read, Write};
use std::str::FromStr;

// ---------- Fast input scanner ----------

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Self {
            buf: input.into_bytes(),
            idx: 0,
        }
    }
    fn is_whitespace(b: u8) -> bool {
        b == b' ' || b == b'\n' || b == b'\r' || b == b'\t'
    }
    fn next<T: FromStr>(&mut self) -> T {
        while self.idx < self.buf.len() && Self::is_whitespace(self.buf[self.idx]) {
            self.idx += 1;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !Self::is_whitespace(self.buf[self.idx]) {
            self.idx += 1;
        }
        std::str::from_utf8(&self.buf[start..self.idx])
            .unwrap()
            .parse()
            .ok()
            .unwrap()
    }
}

// ---------- Geometry primitives ----------

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    pts: Vec<Point>,
}

fn polygon_aabb(poly: &Polygon) -> (f64, f64, f64, f64) {
    let mut min_x = std::f64::INFINITY;
    let mut min_y = std::f64::INFINITY;
    let mut max_x = std::f64::NEG_INFINITY;
    let mut max_y = std::f64::NEG_INFINITY;
    for p in &poly.pts {
        if p.x < min_x {
            min_x = p.x;
        }
        if p.x > max_x {
            max_x = p.x;
        }
        if p.y < min_y {
            min_y = p.y;
        }
        if p.y > max_y {
            max_y = p.y;
        }
    }
    (min_x, min_y, max_x, max_y)
}

fn polygon_area(poly: &Polygon) -> f64 {
    let n = poly.pts.len();
    if n < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..n {
        let p = poly.pts[i];
        let q = poly.pts[(i + 1) % n];
        sum += p.x * q.y - q.x * p.y;
    }
    0.5 * sum.abs()
}

fn point_on_segment(p: &Point, a: &Point, b: &Point) -> bool {
    // Check if p is on segment ab (with small epsilon)
    let eps = 1e-9;
    let cross = (b.x - a.x) * (p.y - a.y) - (b.y - a.y) * (p.x - a.x);
    if cross.abs() > eps {
        return false;
    }
    let dot = (p.x - a.x) * (p.x - b.x) + (p.y - a.y) * (p.y - b.y);
    dot <= eps
}

// Ray casting, points on boundary considered inside.
fn point_in_polygon(poly: &[Point], p: Point) -> bool {
    let n = poly.len();
    if n == 0 {
        return false;
    }

    // Boundary check
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(&p, &a, &b) {
            return true;
        }
    }

    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let pi = poly[i];
        let pj = poly[j];
        let intersect = ((pi.y > p.y) != (pj.y > p.y))
            && (p.x
                < (pj.x - pi.x) * (p.y - pi.y) / ((pj.y - pi.y) + 1e-18) + pi.x);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

// ---------- Simple RNG (xorshift64) ----------

struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9e3779b97f4a7c15;
        }
        Self { state: s }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    fn next_f64(&mut self) -> f64 {
        const DEN: f64 = (u64::MAX as f64) + 1.0;
        (self.next_u64() as f64) / DEN
    }
}

// ---------- Piece data ----------

#[derive(Clone)]
struct PieceOrient {
    vertices: Vec<Point>, // already rotated around origin
    bbox_min_x: f64,
    bbox_min_y: f64,
    bbox_w: f64,
    bbox_h: f64,
    rotation: f64, // angle (0 or 90deg)
}

#[derive(Clone)]
struct Piece {
    index: usize,            // original input index
    orients: Vec<PieceOrient>, // [0deg, 90deg]
    max_dim: f64,            // for sorting (larger first)
    area: f64,
}

fn rotate_polygon(poly: &Polygon, angle: f64) -> Polygon {
    let (c, s) = angle.cos_sin();
    let pts = poly
        .pts
        .iter()
        .map(|p| Point {
            x: c * p.x - s * p.y,
            y: s * p.x + c * p.y,
        })
        .collect();
    Polygon { pts }
}

fn build_piece(index: usize, base: Polygon) -> Piece {
    let area = polygon_area(&base);

    // orientation 0
    let (min_x0, min_y0, max_x0, max_y0) = polygon_aabb(&base);
    let orient0 = PieceOrient {
        vertices: base.pts.clone(),
        bbox_min_x: min_x0,
        bbox_min_y: min_y0,
        bbox_w: max_x0 - min_x0,
        bbox_h: max_y0 - min_y0,
        rotation: 0.0,
    };

    // orientation 90deg
    let rot90 = rotate_polygon(&base, FRAC_PI_2);
    let (min_x1, min_y1, max_x1, max_y1) = polygon_aabb(&rot90);
    let orient1 = PieceOrient {
        vertices: rot90.pts.clone(),
        bbox_min_x: min_x1,
        bbox_min_y: min_y1,
        bbox_w: max_x1 - min_x1,
        bbox_h: max_y1 - min_y1,
        rotation: FRAC_PI_2,
    };

    let max_dim = orient0
        .bbox_w
        .max(orient0.bbox_h)
        .max(orient1.bbox_w.max(orient1.bbox_h));

    Piece {
        index,
        orients: vec![orient0, orient1],
        max_dim,
        area,
    }
}

// ---------- Placement + rectangles ----------

#[derive(Clone, Copy, Debug)]
struct Placement {
    stock_index: usize,
    tx: f64,
    ty: f64,
    rot: f64,
}

fn rects_overlap(
    x1: f64,
    y1: f64,
    w1: f64,
    h1: f64,
    x2: f64,
    y2: f64,
    w2: f64,
    h2: f64,
) -> bool {
    !(x1 + w1 <= x2 || x2 + w2 <= x1 || y1 + h1 <= y2 || y2 + h2 <= y1)
}

// ---------- Small-mode stock representation ----------

#[derive(Clone)]
struct RectPlacement {
    bx: f64, // bottom-left in stock AABB local coordinates
    by: f64,
    bw: f64,
    bh: f64,
}

#[derive(Clone)]
struct StockGeomSmall {
    rects: Vec<RectPlacement>,
}

// Simple vertex-only containment check (used in small-mode).
fn polygon_inside_by_vertices(
    stock_poly: &Polygon,
    piece_vertices: &[Point],
    tx: f64,
    ty: f64,
) -> bool {
    for p in piece_vertices {
        let q = Point {
            x: p.x + tx,
            y: p.y + ty,
        };
        if !point_in_polygon(&stock_poly.pts, q) {
            return false;
        }
    }
    true
}

// Try to place a piece into a given stock (small-mode, random search).
fn try_place_in_stock_small(
    piece: &Piece,
    stock_idx: usize,
    stock: &mut StockGeomSmall,
    stock_poly: &Polygon,
    stock_aabb: (f64, f64, f64, f64),
    rng: &mut Rng,
    attempts: usize,
) -> Option<Placement> {
    let (smin_x, smin_y, smax_x, smax_y) = stock_aabb;
    let sw = smax_x - smin_x;
    let sh = smax_y - smin_y;

    if sw <= 0.0 || sh <= 0.0 {
        return None;
    }

    for _ in 0..attempts {
        // Try both orientations for each attempt
        for orient in &piece.orients {
            let bw = orient.bbox_w;
            let bh = orient.bbox_h;
            if bw > sw || bh > sh {
                continue;
            }

            let range_x = sw - bw;
            let range_y = sh - bh;

            let rx = rng.next_f64() * range_x;
            let ry = rng.next_f64() * range_y;

            // Check bbox collisions
            let mut collision = false;
            for r in &stock.rects {
                if rects_overlap(rx, ry, bw, bh, r.bx, r.by, r.bw, r.bh) {
                    collision = true;
                    break;
                }
            }
            if collision {
                continue;
            }

            // Translate polygon
            let tx = smin_x + rx - orient.bbox_min_x;
            let ty = smin_y + ry - orient.bbox_min_y;

            if polygon_inside_by_vertices(stock_poly, &orient.vertices, tx, ty) {
                stock.rects.push(RectPlacement {
                    bx: rx,
                    by: ry,
                    bw,
                    bh,
                });
                return Some(Placement {
                    stock_index: stock_idx,
                    tx,
                    ty,
                    rot: orient.rotation,
                });
            }
        }
    }
    None
}

// Fallback: deterministic grid search (for one new stock, small-mode).
fn place_in_new_stock_small(
    piece: &Piece,
    stock_idx: usize,
    stock_poly: &Polygon,
    stock_aabb: (f64, f64, f64, f64),
) -> Option<(StockGeomSmall, Placement)> {
    let (smin_x, smin_y, smax_x, smax_y) = stock_aabb;
    let sw = smax_x - smin_x;
    let sh = smax_y - smin_y;

    if sw <= 0.0 || sh <= 0.0 {
        return None;
    }

    let mut best_stock = StockGeomSmall { rects: Vec::new() };

    // Try grid positions for both orientations.
    let grid_steps = 30usize;

    for orient in &piece.orients {
        let bw = orient.bbox_w;
        let bh = orient.bbox_h;
        if bw > sw || bh > sh {
            continue;
        }
        let range_x = sw - bw;
        let range_y = sh - bh;

        for gx in 0..=grid_steps {
            for gy in 0..=grid_steps {
                let rx = range_x * (gx as f64) / (grid_steps as f64);
                let ry = range_y * (gy as f64) / (grid_steps as f64);

                let tx = smin_x + rx - orient.bbox_min_x;
                let ty = smin_y + ry - orient.bbox_min_y;

                if polygon_inside_by_vertices(stock_poly, &orient.vertices, tx, ty) {
                    best_stock.rects.push(RectPlacement {
                        bx: rx,
                        by: ry,
                        bw,
                        bh,
                    });
                    let placement = Placement {
                        stock_index,
                        tx,
                        ty,
                        rot: orient.rotation,
                    };
                    return Some((best_stock, placement));
                }
            }
        }
    }
    None
}

// ---------- Large-mode shelf packing ----------

#[derive(Clone)]
struct Shelf {
    y: f64,
    height: f64,
    cur_x: f64,
}

#[derive(Clone)]
struct StockLarge {
    shelves: Vec<Shelf>,
    used_h: f64,
}

fn place_large_mode(
    piece: &Piece,
    stocks: &mut Vec<StockLarge>,
    stock_w: f64,
    stock_h: f64,
) -> (usize, f64, f64) {
    let orient = &piece.orients[0]; // only 0deg in large mode
    let w = orient.bbox_w;
    let h = orient.bbox_h;

    // Try existing shelves
    for (s_idx, stock) in stocks.iter_mut().enumerate() {
        for shelf in &mut stock.shelves {
            if h <= shelf.height + 1e-9 && shelf.cur_x + w <= stock_w + 1e-9 {
                let x = shelf.cur_x;
                let y = shelf.y;
                shelf.cur_x += w;
                return (s_idx, x, y);
            }
        }
    }

    // Try creating new shelf in existing stocks
    for (s_idx, stock) in stocks.iter_mut().enumerate() {
        if w > stock_w + 1e-9 {
            continue;
        }
        if stock.used_h + h <= stock_h + 1e-9 {
            let y = stock.used_h;
            let x = 0.0;
            stock.shelves.push(Shelf {
                y,
                height: h,
                cur_x: w,
            });
            stock.used_h += h;
            return (s_idx, x, y);
        }
    }

    // Create new stock
    let mut new_stock = StockLarge {
        shelves: Vec::new(),
        used_h: 0.0,
    };
    let x = 0.0;
    let y = 0.0;
    new_stock.shelves.push(Shelf {
        y,
        height: h,
        cur_x: w,
    });
    new_stock.used_h = h;
    stocks.push(new_stock);
    (stocks.len() - 1, x, y)
}

// ---------- Main ----------

fn main() {
    let mut sc = Scanner::new();

    let sv: usize = sc.next();
    let np: usize = sc.next();

    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = sc.next();
        let y: f64 = sc.next();
        stock_pts.push(Point { x, y });
    }
    let stock_poly = Polygon { pts: stock_pts };
    let stock_aabb = polygon_aabb(&stock_poly);
    let (smin_x, smin_y, smax_x, smax_y) = stock_aabb;
    let stock_w = smax_x - smin_x;
    let stock_h = smax_y - smin_y;

    let mut base_pieces: Vec<Polygon> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = sc.next();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = sc.next();
            let y: f64 = sc.next();
            pts.push(Point { x, y });
        }
        base_pieces.push(Polygon { pts });
    }

    // Build piece info with orientations
    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    for (idx, poly) in base_pieces.into_iter().enumerate() {
        pieces.push(build_piece(idx, poly));
    }

    // Sort pieces by max_dim (decreasing), then by area
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let pa = &pieces[a];
        let pb = &pieces[b];
        pb.max_dim
            .partial_cmp(&pa.max_dim)
            .unwrap_or(std::cmp::Ordering::Equal)
            .then_with(|| {
                pb.area
                    .partial_cmp(&pa.area)
                    .unwrap_or(std::cmp::Ordering::Equal)
            })
    });

    let small_mode = np <= 300;

    let mut placements: Vec<Placement> = vec![
        Placement {
            stock_index: 0,
            tx: 0.0,
            ty: 0.0,
            rot: 0.0
        };
        np
    ];
    let num_stocks: usize;

    if small_mode {
        // Small-mode: random + grid search with geometry checks
        let mut rng = Rng::new(123456789);
        let mut stocks_sm: Vec<StockGeomSmall> = Vec::new();

        for &pi in &order {
            let piece = &pieces[pi];
            let mut placed = false;

            // Try existing stocks
            for sidx in 0..stocks_sm.len() {
                let (placement_opt, rect_added) = {
                    let stock = &mut stocks_sm[sidx];
                    let old_len = stock.rects.len();
                    let placement_opt = try_place_in_stock_small(
                        piece,
                        sidx,
                        stock,
                        &stock_poly,
                        stock_aabb,
                        &mut rng,
                        200,
                    );
                    let rect_added = stock.rects.len() > old_len;
                    (placement_opt, rect_added)
                };
                if let Some(pl) = placement_opt {
                    placements[piece.index] = pl;
                    placed = true;
                    break;
                } else if rect_added {
                    // Should not happen; but just in case, remove last rect
                    let stock = &mut stocks_sm[sidx];
                    stock.rects.pop();
                }
            }

            if !placed {
                // Create new stock and try deterministic grid fallback
                let new_index = stocks_sm.len();
                if let Some((stock_new, pl)) =
                    place_in_new_stock_small(piece, new_index, &stock_poly, stock_aabb)
                {
                    placements[piece.index] = pl;
                    stocks_sm.push(stock_new);
                    placed = true;
                } else {
                    // As very last resort, place at bottom-left without checks
                    // (should be extremely rare; assumes input is reasonable).
                    let orient = &piece.orients[0];
                    let bw = orient.bbox_w;
                    let bh = orient.bbox_h;
                    let mut stock_new = StockGeomSmall { rects: Vec::new() };
                    let bx = 0.0f64.max(stock_w - bw).min(0.0);
                    let by = 0.0f64.max(stock_h - bh).min(0.0);
                    let tx = smin_x + bx - orient.bbox_min_x;
                    let ty = smin_y + by - orient.bbox_min_y;
                    stock_new.rects.push(RectPlacement {
                        bx,
                        by,
                        bw,
                        bh,
                    });
                    let pl = Placement {
                        stock_index: new_index,
                        tx,
                        ty,
                        rot: orient.rotation,
                    };
                    placements[piece.index] = pl;
                    stocks_sm.push(stock_new);
                    placed = true;
                }
            }

            if !placed {
                // Fallback safety (should not reach)
                let pl = Placement {
                    stock_index: stocks_sm.len(),
                    tx: smin_x,
                    ty: smin_y,
                    rot: 0.0,
                };
                placements[piece.index] = pl;
                stocks_sm.push(StockGeomSmall { rects: Vec::new() });
            }
        }

        num_stocks = stocks_sm.len();
    } else {
        // Large-mode: fast shelf packing on bounding rectangles, no geometry checks
        let mut stocks_lg: Vec<StockLarge> = Vec::new();

        for &pi in &order {
            let piece = &pieces[pi];
            let (sidx, bx, by) = place_large_mode(piece, &mut stocks_lg, stock_w, stock_h);
            let orient = &piece.orients[0]; // 0deg
            let tx = smin_x + bx - orient.bbox_min_x;
            let ty = smin_y + by - orient.bbox_min_y;
            placements[piece.index] = Placement {
                stock_index: sidx,
                tx,
                ty,
                rot: orient.rotation,
            };
        }

        num_stocks = stocks_lg.len();
    }

    // ---------- Output ----------

    let mut out = String::new();
    out.push_str(&format!("{}\n", num_stocks));
    for i in 0..np {
        let p = placements[i];
        out.push_str(&format!(
            "{} {:.10} {:.10} {:.10}\n",
            p.stock_index, p.tx, p.ty, p.rot
        ));
    }
    let mut writer = io::BufWriter::new(io::stdout());
    writer.write_all(out.as_bytes()).unwrap();
}
```
