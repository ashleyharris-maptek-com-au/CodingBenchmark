using namespace std;

enum Color { WHITE = 0, BLACK = 1 };

struct Move {
    int from = 0;
    int to = 0;
    char promotion = 0;        // 'Q','R','B','N' or 0
    bool isCapture = false;
    bool isEnPassant = false;
    bool isCastling = false;
    bool isDoublePush = false;
    char captured = '.';       // captured piece, if any
    char piece = '.';          // moving piece
    int score = 0;             // for move ordering
};

struct Undo {
    int epSquare = -1;
    int castlingRights = 0;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    char captured = '.';
    int capturedSquare = -1;
};

struct Board {
    char squares[64];
    Color sideToMove;
    int castlingRights;   // bit0: K, bit1: Q, bit2: k, bit3: q
    int epSquare;         // -1 or 0..63
    int halfmoveClock;
    int fullmoveNumber;
    int kingSquare[2];    // [WHITE], [BLACK]

    Board() {
        for (int i = 0; i < 64; ++i) squares[i] = '.';
        sideToMove = WHITE;
        castlingRights = 0;
        epSquare = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        kingSquare[WHITE] = kingSquare[BLACK] = -1;
    }

    static Board fromFEN(const string& fen);

    bool inCheck(Color side) const;
    bool isSquareAttacked(int sq, Color by) const;

    void generatePseudoLegalMoves(vector<Move>& moves) const;
    void generateLegalMoves(vector<Move>& moves);

    void makeMove(const Move& m, Undo& u);
    void undoMove(const Undo& u, const Move& m);

private:
    void addMove(vector<Move>& moves, int from, int to,
                 char promotion = 0,
                 bool isEnPassant = false,
                 bool isCastling = false) const;
};

// Utility: piece values
inline int pieceValue(char p) {
    if (p == '.' || p == 0) return 0;
    char u = static_cast<char>(std::toupper(static_cast<unsigned char>(p)));
    switch (u) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 0;    // king value not used directly
        default:  return 0;
    }
}

// Evaluation: simple material + light central bonuses
int evaluate(const Board& b) {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        char p = b.squares[i];
        if (p == '.') continue;
        int val = pieceValue(p);
        if (p >= 'A' && p <= 'Z') score += val;
        else score -= val;
    }

    // Small central bonuses
    for (int i = 0; i < 64; ++i) {
        char p = b.squares[i];
        if (p == '.') continue;
        int r = i / 8;
        int f = i % 8;
        bool inCenter = (f >= 2 && f <= 5 && r >= 2 && r <= 5);
        if (!inCenter) continue;
        char u = static_cast<char>(std::toupper(static_cast<unsigned char>(p)));
        int bonus = 0;
        if (u == 'P') bonus = 3;
        else if (u == 'N' || u == 'B') bonus = 5;
        else if (u == 'Q') bonus = 1;
        if (p >= 'A' && p <= 'Z') score += bonus;
        else score -= bonus;
    }

    // Evaluate from side-to-move perspective
    return (b.sideToMove == WHITE) ? score : -score;
}

// Time management
inline bool isTimeUp() {
    if (g_timeUp) return true;
    auto now = Clock::now();
    double elapsed = std::chrono::duration<double>(now - g_startTime).count();
    if (elapsed >= g_timeLimitSeconds) {
        g_timeUp = true;
    }
    return g_timeUp;
}

Board Board::fromFEN(const string& fen) {
    Board b;

    istringstream iss(fen);
    string boardPart, sidePart, castlePart, epPart, halfmovePart, fullmovePart;
    iss >> boardPart >> sidePart >> castlePart >> epPart >> halfmovePart >> fullmovePart;

    int rank = 7, file = 0;
    for (char c : boardPart) {
        if (c == '/') {
            --rank;
            file = 0;
        } else if (std::isdigit(static_cast<unsigned char>(c))) {
            int empty = c - '0';
            for (int i = 0; i < empty; ++i) {
                b.squares[rank * 8 + file] = '.';
                ++file;
            }
        } else {
            b.squares[rank * 8 + file] = c;
            if (c == 'K') b.kingSquare[WHITE] = rank * 8 + file;
            else if (c == 'k') b.kingSquare[BLACK] = rank * 8 + file;
            ++file;
        }
    }

    b.sideToMove = (sidePart == "w") ? WHITE : BLACK;

    b.castlingRights = 0;
    if (castlePart.find('K') != string::npos) b.castlingRights |= 1;
    if (castlePart.find('Q') != string::npos) b.castlingRights |= 2;
    if (castlePart.find('k') != string::npos) b.castlingRights |= 4;
    if (castlePart.find('q') != string::npos) b.castlingRights |= 8;

    if (!epPart.empty() && epPart != "-") {
        if (epPart.size() == 2) {
            int f = epPart[0] - 'a';
            int r = epPart[1] - '1';
            if (f >= 0 && f < 8 && r >= 0 && r < 8)
                b.epSquare = r * 8 + f;
            else
                b.epSquare = -1;
        } else {
            b.epSquare = -1;
        }
    } else {
        b.epSquare = -1;
    }

    b.halfmoveClock = halfmovePart.empty() ? 0 : std::stoi(halfmovePart);
    b.fullmoveNumber = fullmovePart.empty() ? 1 : std::stoi(fullmovePart);

    return b;
}

bool Board::isSquareAttacked(int sq, Color by) const {
    int r = sq / 8;
    int f = sq % 8;

    // Pawns
    if (by == WHITE) {
        if (r > 0) {
            if (f > 0 && squares[sq - 9] == 'P') return true;
            if (f < 7 && squares[sq - 7] == 'P') return true;
        }
    } else {
        if (r < 7) {
            if (f > 0 && squares[sq + 7] == 'p') return true;
            if (f < 7 && squares[sq + 9] == 'p') return true;
        }
    }

    // Knights
    static const int knightDirs[8][2] = {
        { 2, 1}, { 1, 2}, {-1, 2}, {-2, 1},
        {-2,-1}, {-1,-2}, { 1,-2}, { 2,-1}
    };
    char knight = (by == WHITE ? 'N' : 'n');
    for (int i = 0; i < 8; ++i) {
        int nr = r + knightDirs[i][0];
        int nf = f + knightDirs[i][1];
        if (nr < 0 || nr > 7 || nf < 0 || nf > 7) continue;
        if (squares[nr * 8 + nf] == knight) return true;
    }

    // Bishops / queens (diagonals)
    static const int bishopDirs[4][2] = {
        { 1, 1}, { 1,-1}, {-1, 1}, {-1,-1}
    };
    char bishop = (by == WHITE ? 'B' : 'b');
    char queen  = (by == WHITE ? 'Q' : 'q');
    for (int d = 0; d < 4; ++d) {
        int nr = r + bishopDirs[d][0];
        int nf = f + bishopDirs[d][1];
        while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
            char p = squares[nr * 8 + nf];
            if (p != '.') {
                if (p == bishop || p == queen) return true;
                break;
            }
            nr += bishopDirs[d][0];
            nf += bishopDirs[d][1];
        }
    }

    // Rooks / queens (orthogonals)
    static const int rookDirs[4][2] = {
        { 1, 0}, {-1, 0}, { 0, 1}, { 0,-1}
    };
    char rook = (by == WHITE ? 'R' : 'r');
    for (int d = 0; d < 4; ++d) {
        int nr = r + rookDirs[d][0];
        int nf = f + rookDirs[d][1];
        while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
            char p = squares[nr * 8 + nf];
            if (p != '.') {
                if (p == rook || p == queen) return true;
                break;
            }
            nr += rookDirs[d][0];
            nf += rookDirs[d][1];
        }
    }

    // King
    char king = (by == WHITE ? 'K' : 'k');
    for (int dr = -1; dr <= 1; ++dr) {
        for (int df = -1; df <= 1; ++df) {
            if (dr == 0 && df == 0) continue;
            int nr = r + dr;
            int nf = f + df;
            if (nr < 0 || nr > 7 || nf < 0 || nf > 7) continue;
            if (squares[nr * 8 + nf] == king) return true;
        }
    }

    return false;
}

bool Board::inCheck(Color side) const {
    int ks = kingSquare[side];
    if (ks < 0) return false;
    Color opp = (side == WHITE ? BLACK : WHITE);
    return isSquareAttacked(ks, opp);
}

void Board::addMove(vector<Move>& moves, int from, int to,
                    char promotion, bool isEnPassant, bool isCastling) const {
    Move m;
    m.from = from;
    m.to = to;
    m.promotion = promotion;
    m.isEnPassant = isEnPassant;
    m.isCastling = isCastling;
    m.piece = squares[from];
    m.isCapture = false;
    m.captured = '.';

    if (isEnPassant) {
        int capSq = (sideToMove == WHITE ? to - 8 : to + 8);
        m.isCapture = true;
        m.captured = squares[capSq];
    } else if (squares[to] != '.') {
        m.isCapture = true;
        m.captured = squares[to];
    }

    int diff = to - from;
    if (diff < 0) diff = -diff;
    m.isDoublePush = (std::toupper(static_cast<unsigned char>(m.piece)) == 'P' &&
                      !m.isCapture && promotion == 0 && diff == 16);

    m.score = 0;
    moves.push_back(m);
}

void Board::generatePseudoLegalMoves(vector<Move>& moves) const {
    moves.clear();
    Color us = sideToMove;
    bool whiteToMove = (us == WHITE);

    for (int sq = 0; sq < 64; ++sq) {
        char piece = squares[sq];
        if (piece == '.') continue;
        bool isWhitePiece = (piece >= 'A' && piece <= 'Z');
        if (whiteToMove != isWhitePiece) continue;

        char type = static_cast<char>(std::toupper(static_cast<unsigned char>(piece)));
        int r = sq / 8;
        int f = sq % 8;

        if (type == 'P') {
            if (whiteToMove) {
                // White pawn
                if (r < 7) {
                    int oneStep = sq + 8;
                    if (squares[oneStep] == '.') {
                        if (r == 6) {
                            addMove(moves, sq, oneStep, 'Q');
                            addMove(moves, sq, oneStep, 'R');
                            addMove(moves, sq, oneStep, 'B');
                            addMove(moves, sq, oneStep, 'N');
                        } else {
                            addMove(moves, sq, oneStep);
                            if (r == 1) {
                                int twoStep = sq + 16;
                                if (squares[twoStep] == '.')
                                    addMove(moves, sq, twoStep);
                            }
                        }
                    }
                    // Captures and en passant
                    if (f > 0) {
                        int to = sq + 7;
                        char target = squares[to];
                        if (target != '.' && target >= 'a' && target <= 'z') {
                            if (r == 6) {
                                addMove(moves, sq, to, 'Q');
                                addMove(moves, sq, to, 'R');
                                addMove(moves, sq, to, 'B');
                                addMove(moves, sq, to, 'N');
                            } else {
                                addMove(moves, sq, to);
                            }
                        } else if (epSquare == to && epSquare != -1) {
                            addMove(moves, sq, to, 0, true);
                        }
                    }
                    if (f < 7) {
                        int to = sq + 9;
                        char target = squares[to];
                        if (target != '.' && target >= 'a' && target <= 'z') {
                            if (r == 6) {
                                addMove(moves, sq, to, 'Q');
                                addMove(moves, sq, to, 'R');
                                addMove(moves, sq, to, 'B');
                                addMove(moves, sq, to, 'N');
                            } else {
                                addMove(moves, sq, to);
                            }
                        } else if (epSquare == to && epSquare != -1) {
                            addMove(moves, sq, to, 0, true);
                        }
                    }
                }
            } else {
                // Black pawn
                if (r > 0) {
                    int oneStep = sq - 8;
                    if (squares[oneStep] == '.') {
                        if (r == 1) {
                            addMove(moves, sq, oneStep, 'Q');
                            addMove(moves, sq, oneStep, 'R');
                            addMove(moves, sq, oneStep, 'B');
                            addMove(moves, sq, oneStep, 'N');
                        } else {
                            addMove(moves, sq, oneStep);
                            if (r == 6) {
                                int twoStep = sq - 16;
                                if (squares[twoStep] == '.')
                                    addMove(moves, sq, twoStep);
                            }
                        }
                    }
                    if (f > 0) {
                        int to = sq - 9;
                        char target = squares[to];
                        if (target != '.' && target >= 'A' && target <= 'Z') {
                            if (r == 1) {
                                addMove(moves, sq, to, 'Q');
                                addMove(moves, sq, to, 'R');
                                addMove(moves, sq, to, 'B');
                                addMove(moves, sq, to, 'N');
                            } else {
                                addMove(moves, sq, to);
                            }
                        } else if (epSquare == to && epSquare != -1) {
                            addMove(moves, sq, to, 0, true);
                        }
                    }
                    if (f < 7) {
                        int to = sq - 7;
                        char target = squares[to];
                        if (target != '.' && target >= 'A' && target <= 'Z') {
                            if (r == 1) {
                                addMove(moves, sq, to, 'Q');
                                addMove(moves, sq, to, 'R');
                                addMove(moves, sq, to, 'B');
                                addMove(moves, sq, to, 'N');
                            } else {
                                addMove(moves, sq, to);
                            }
                        } else if (epSquare == to && epSquare != -1) {
                            addMove(moves, sq, to, 0, true);
                        }
                    }
                }
            }
        } else if (type == 'N') {
            static const int knightDirs[8][2] = {
                { 2, 1}, { 1, 2}, {-1, 2}, {-2, 1},
                {-2,-1}, {-1,-2}, { 1,-2}, { 2,-1}
            };
            for (int i = 0; i < 8; ++i) {
                int nr = r + knightDirs[i][0];
                int nf = f + knightDirs[i][1];
                if (nr < 0 || nr > 7 || nf < 0 || nf > 7) continue;
                int to = nr * 8 + nf;
                char target = squares[to];
                if (target == '.' ||
                    (whiteToMove ? (target >= 'a' && target <= 'z')
                                 : (target >= 'A' && target <= 'Z'))) {
                    addMove(moves, sq, to);
                }
            }
        } else if (type == 'B' || type == 'R' || type == 'Q') {
            static const int bishopDirs[4][2] = {
                { 1, 1}, { 1,-1}, {-1, 1}, {-1,-1}
            };
            static const int rookDirs[4][2] = {
                { 1, 0}, {-1, 0}, { 0, 1}, { 0,-1}
            };
            if (type == 'B' || type == 'Q') {
                for (int d = 0; d < 4; ++d) {
                    int nr = r + bishopDirs[d][0];
                    int nf = f + bishopDirs[d][1];
                    while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
                        int to = nr * 8 + nf;
                        char target = squares[to];
                        if (target == '.') {
                            addMove(moves, sq, to);
                        } else {
                            if (whiteToMove ? (target >= 'a' && target <= 'z')
                                            : (target >= 'A' && target <= 'Z')) {
                                addMove(moves, sq, to);
                            }
                            break;
                        }
                        nr += bishopDirs[d][0];
                        nf += bishopDirs[d][1];
                    }
                }
            }
            if (type == 'R' || type == 'Q') {
                for (int d = 0; d < 4; ++d) {
                    int nr = r + rookDirs[d][0];
                    int nf = f + rookDirs[d][1];
                    while (nr >= 0 && nr <= 7 && nf >= 0 && nf <= 7) {
                        int to = nr * 8 + nf;
                        char target = squares[to];
                        if (target == '.') {
                            addMove(moves, sq, to);
                        } else {
                            if (whiteToMove ? (target >= 'a' && target <= 'z')
                                            : (target >= 'A' && target <= 'Z')) {
                                addMove(moves, sq, to);
                            }
                            break;
                        }
                        nr += rookDirs[d][0];
                        nf += rookDirs[d][1];
                    }
                }
            }
        } else if (type == 'K') {
            // King normal moves
            for (int dr = -1; dr <= 1; ++dr) {
                for (int df = -1; df <= 1; ++df) {
                    if (dr == 0 && df == 0) continue;
                    int nr = r + dr;
                    int nf = f + df;
                    if (nr < 0 || nr > 7 || nf < 0 || nf > 7) continue;
                    int to = nr * 8 + nf;
                    char target = squares[to];
                    if (target == '.' ||
                        (whiteToMove ? (target >= 'a' && target <= 'z')
                                     : (target >= 'A' && target <= 'Z'))) {
                        addMove(moves, sq, to);
                    }
                }
            }
            // Castling
            if (whiteToMove && sq == 4) {
                // White king side
                if (castlingRights & 1) {
                    if (squares[5] == '.' && squares[6] == '.' &&
                        !inCheck(WHITE) &&
                        !isSquareAttacked(5, BLACK) &&
                        !isSquareAttacked(6, BLACK)) {
                        addMove(moves, 4, 6, 0, false, true);
                    }
                }
                // White queen side
                if (castlingRights & 2) {
                    if (squares[3] == '.' && squares[2] == '.' && squares[1] == '.' &&
                        !inCheck(WHITE) &&
                        !isSquareAttacked(3, BLACK) &&
                        !isSquareAttacked(2, BLACK)) {
                        addMove(moves, 4, 2, 0, false, true);
                    }
                }
            } else if (!whiteToMove && sq == 60) {
                // Black king side
                if (castlingRights & 4) {
                    if (squares[61] == '.' && squares[62] == '.' &&
                        !inCheck(BLACK) &&
                        !isSquareAttacked(61, WHITE) &&
                        !isSquareAttacked(62, WHITE)) {
                        addMove(moves, 60, 62, 0, false, true);
                    }
                }
                // Black queen side
                if (castlingRights & 8) {
                    if (squares[59] == '.' && squares[58] == '.' && squares[57] == '.' &&
                        !inCheck(BLACK) &&
                        !isSquareAttacked(59, WHITE) &&
                        !isSquareAttacked(58, WHITE)) {
                        addMove(moves, 60, 58, 0, false, true);
                    }
                }
            }
        }
    }
}

void Board::generateLegalMoves(vector<Move>& moves) {
    vector<Move> pseudo;
    pseudo.reserve(64);
    generatePseudoLegalMoves(pseudo);
    moves.clear();

    Color us = sideToMove;
    Undo u;
    for (const Move& m : pseudo) {
        makeMove(m, u);
        bool illegal = inCheck(us);
        undoMove(u, m);
        if (!illegal) moves.push_back(m);
    }
}

void Board::makeMove(const Move& m, Undo& u) {
    u.epSquare = epSquare;
    u.castlingRights = castlingRights;
    u.halfmoveClock = halfmoveClock;
    u.fullmoveNumber = fullmoveNumber;

    if (m.isEnPassant) {
        int capSq = (sideToMove == WHITE ? m.to - 8 : m.to + 8);
        u.capturedSquare = capSq;
        u.captured = squares[capSq];
    } else {
        u.capturedSquare = m.to;
        u.captured = squares[m.to];
    }

    char piece = m.piece;

    // Halfmove clock
    if (std::toupper(static_cast<unsigned char>(piece)) == 'P' || m.isCapture)
        halfmoveClock = 0;
    else
        ++halfmoveClock;

    // Fullmove number
    if (sideToMove == BLACK) ++fullmoveNumber;

    // Clear ep square
    epSquare = -1;

    // Update castling rights based on moving piece
    if (piece == 'K') {
        castlingRights &= ~(1 | 2);
    } else if (piece == 'k') {
        castlingRights &= ~(4 | 8);
    } else if (piece == 'R') {
        if (m.from == 0) castlingRights &= ~2;
        else if (m.from == 7) castlingRights &= ~1;
    } else if (piece == 'r') {
        if (m.from == 56) castlingRights &= ~8;
        else if (m.from == 63) castlingRights &= ~4;
    }

    // Update castling rights if capturing rook
    if (m.isCapture && !m.isEnPassant) {
        if (m.to == 0) castlingRights &= ~2;
        else if (m.to == 7) castlingRights &= ~1;
        else if (m.to == 56) castlingRights &= ~8;
        else if (m.to == 63) castlingRights &= ~4;
    }

    // Move piece
    squares[m.from] = '.';

    // En passant capture
    if (m.isEnPassant) {
        int capSq = (sideToMove == WHITE ? m.to - 8 : m.to + 8);
        squares[capSq] = '.';
    }

    char newPiece = piece;
    if (m.promotion) {
        if (sideToMove == WHITE)
            newPiece = m.promotion;
        else
            newPiece = static_cast<char>(
                std::tolower(static_cast<unsigned char>(m.promotion)));
    }
    squares[m.to] = newPiece;

    // Castling rook move
    if (m.isCastling) {
        if (m.to > m.from) {
            int rookFrom = m.from + 3;
            int rookTo = m.from + 1;
            squares[rookTo] = squares[rookFrom];
            squares[rookFrom] = '.';
        } else {
            int rookFrom = m.from - 4;
            int rookTo = m.from - 1;
            squares[rookTo] = squares[rookFrom];
            squares[rookFrom] = '.';
        }
    }

    // New ep square for pawn double push
    if (std::toupper(static_cast<unsigned char>(piece)) == 'P') {
        int diff = m.to - m.from;
        if (diff < 0) diff = -diff;
        if (diff == 16) {
            if (sideToMove == WHITE) epSquare = m.from + 8;
            else epSquare = m.from - 8;
        }
    }

    // Update king square
    Color us = sideToMove;
    if (std::toupper(static_cast<unsigned char>(piece)) == 'K') {
        kingSquare[us] = m.to;
    }

    // Toggle side
    sideToMove = (sideToMove == WHITE ? BLACK : WHITE);
}

void Board::undoMove(const Undo& u, const Move& m) {
    // Restore side
    sideToMove = (sideToMove == WHITE ? BLACK : WHITE);

    // Restore king square for side that moved
    Color us = sideToMove;
    if (std::toupper(static_cast<unsigned char>(m.piece)) == 'K') {
        kingSquare[us] = m.from;
    }

    // Restore state
    epSquare = u.epSquare;
    castlingRights = u.castlingRights;
    halfmoveClock = u.halfmoveClock;
    fullmoveNumber = u.fullmoveNumber;

    // Undo castling rook move
    if (m.isCastling) {
        if (m.to > m.from) {
            int rookFrom = m.from + 3;
            int rookTo = m.from + 1;
            squares[rookFrom] = squares[rookTo];
            squares[rookTo] = '.';
        } else {
            int rookFrom = m.from - 4;
            int rookTo = m.from - 1;
            squares[rookFrom] = squares[rookTo];
            squares[rookTo] = '.';
        }
    }

    // Restore moving piece
    squares[m.from] = m.piece;

    // Restore captured piece
    if (m.isEnPassant) {
        squares[m.to] = '.';
        squares[u.capturedSquare] = u.captured;
    } else {
        squares[m.to] = u.captured;
    }
}

// -------- Search --------

int negamax(Board& board, int depth, int alpha, int beta, int ply) {
    if (g_timeUp) return 0;
    ++g_nodes;
    if ((g_nodes & 0x3FF) == 0 && isTimeUp()) return 0;

    if (depth == 0) {
        return evaluate(board);
    }

    vector<Move> moves;
    board.generateLegalMoves(moves);

    if (moves.empty()) {
        if (board.inCheck(board.sideToMove)) {
            return -MATE_SCORE + ply;
        } else {
            return 0; // stalemate
        }
    }

    // Move ordering
    for (auto& m : moves) {
        int s = 0;
        if (m.isCapture) {
            s += 10 * pieceValue(m.captured) - pieceValue(m.piece);
        }
        if (m.promotion) {
            s += 1000 + pieceValue(m.promotion);
        }
        if (m.isCastling) s += 50;
        m.score = s;
    }
    std::sort(moves.begin(), moves.end(),
              [](const Move& a, const Move& b) { return a.score > b.score; });

    int bestScore = -INF;
    Undo u;

    for (const Move& m : moves) {
        board.makeMove(m, u);
        int score = -negamax(board, depth - 1, -beta, -alpha, ply + 1);
        board.undoMove(u, m);

        if (g_timeUp) return 0;

        if (score > bestScore) bestScore = score;
        if (score > alpha) alpha = score;
        if (alpha >= beta) break;
    }

    return bestScore;
}

int searchRoot(Board& board, int depth, Move& outBestMove) {
    vector<Move> moves;
    board.generateLegalMoves(moves);

    if (moves.empty()) {
        // No legal move: checkmate or stalemate
        if (board.inCheck(board.sideToMove)) {
            return -MATE_SCORE;
        } else {
            return 0;
        }
    }

    // Move ordering
    for (auto& m : moves) {
        int s = 0;
        if (m.isCapture) {
            s += 10 * pieceValue(m.captured) - pieceValue(m.piece);
        }
        if (m.promotion) {
            s += 1000 + pieceValue(m.promotion);
        }
        if (m.isCastling) s += 50;
        m.score = s;
    }
    std::sort(moves.begin(), moves.end(),
              [](const Move& a, const Move& b) { return a.score > b.score; });

    int alpha = -INF;
    int beta = INF;
    int bestScore = -INF;
    Move bestMove = moves[0];

    Undo u;
    for (const Move& m : moves) {
        if (isTimeUp()) break;
        board.makeMove(m, u);
        int score = -negamax(board, depth - 1, -beta, -alpha, 1);
        board.undoMove(u, m);

        if (g_timeUp) break;

        if (score > bestScore) {
            bestScore = score;
            bestMove = m;
        }
        if (score > alpha) alpha = score;
    }

    outBestMove = bestMove;
    return bestScore;
}

Move iterativeDeepening(Board& board, int maxDepth) {
    Move bestMove;
    bool haveBest = false;

    for (int depth = 1; depth <= maxDepth; ++depth) {
        if (isTimeUp()) break;
        Move iterBest;
        int score = searchRoot(board, depth, iterBest);
        (void)score; // score not used directly
        if (g_timeUp) break;
        bestMove = iterBest;
        haveBest = true;
    }

    if (!haveBest) {
        vector<Move> moves;
        board.generateLegalMoves(moves);
        if (!moves.empty()) bestMove = moves[0];
    }

    return bestMove;
}

// -------- Notation --------

string moveToNotation(Board& board, const Move& m) {
    string s;

    if (m.isCastling) {
        if (m.to > m.from) s = "O-O";
        else s = "O-O-O";
    } else {
        char pieceUpper = static_cast<char>(
            std::toupper(static_cast<unsigned char>(m.piece)));
        char pieceLetter = (pieceUpper == 'P' ? 'p' : pieceUpper);

        char fromFile = static_cast<char>('a' + (m.from % 8));
        char fromRank = static_cast<char>('1' + (m.from / 8));
        char toFile   = static_cast<char>('a' + (m.to % 8));
        char toRank   = static_cast<char>('1' + (m.to / 8));

        s += pieceLetter;
        s += fromFile;
        s += fromRank;
        if (m.isCapture) s += 'x';
        s += toFile;
        s += toRank;

        if (m.promotion) {
            s += '/';
            s += m.promotion;
        }
        if (m.isEnPassant) {
            s += "ep";
        }
    }

    // Append check or checkmate indicator
    Undo u;
    board.makeMove(m, u);
    bool givesCheck = board.inCheck(board.sideToMove);
    bool mate = false;
    if (givesCheck) {
        vector<Move> replies;
        board.generateLegalMoves(replies);
        if (replies.empty()) mate = true;
    }
    board.undoMove(u, m);

    if (mate) s.push_back('#');
    else if (givesCheck) s.push_back('+');

    return s;
}

// -------- main --------

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    string line1;
    if (!std::getline(cin, line1)) {
        return 0;
    }

    double timeoutSeconds = 5.0;
    {
        std::istringstream iss(line1);
        iss >> timeoutSeconds;
        if (!iss) timeoutSeconds = 5.0;
    }

    string fen;
    if (!std::getline(cin, fen)) {
        return 0;
    }

    Board board = Board::fromFEN(fen);

    g_timeLimitSeconds = timeoutSeconds * 0.95;  // small safety margin
    if (g_timeLimitSeconds <= 0.0) g_timeLimitSeconds = timeoutSeconds;
    g_startTime = Clock::now();
    g_timeUp = false;
    g_nodes = 0;

    const int MAX_DEPTH = 64;
    Move bestMove = iterativeDeepening(board, MAX_DEPTH);

    // If no legal moves, output a dummy move (position is mate/stalemate)
    vector<Move> legal;
    board.generateLegalMoves(legal);
    if (legal.empty()) {
        cout << "0000";
        return 0;
    }

    string notation = moveToNotation(board, bestMove);
    cout << notation;
    return 0;
}
