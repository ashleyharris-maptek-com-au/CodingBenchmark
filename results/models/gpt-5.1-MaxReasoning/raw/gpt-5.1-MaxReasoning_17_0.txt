```csharp
using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class Program
{
    struct Box
    {
        public int Index;    // original index from input (0-based)
        public int W, H, D;
        public long Volume;
        public int MaxDim;
        public int MinDim;
    }

    struct FreeSpace
    {
        public int X, Y, Z;
        public int W, H, D;
    }

    struct Placement
    {
        public int BoxIndex;
        public int X, Y, Z;
    }

    enum BoxOrder
    {
        VolumeAsc,
        VolumeDesc,
        MaxDimAsc,
        MinDimAsc
    }

    static void Main()
    {
        var scanner = new FastScanner();

        int n;
        try
        {
            n = scanner.NextInt();
        }
        catch
        {
            // No input
            return;
        }

        int contW = scanner.NextInt();
        int contH = scanner.NextInt();
        int contD = scanner.NextInt();

        // Pre-sort container dimensions for quick "can it ever fit?" test
        int[] contDims = new int[3] { contW, contH, contD };
        Array.Sort(contDims);

        var boxes = new List<Box>(n);

        for (int i = 0; i < n; i++)
        {
            int w = scanner.NextInt();
            int h = scanner.NextInt();
            int d = scanner.NextInt();

            if (w <= 0 || h <= 0 || d <= 0) continue;

            // Quick fit test: can this box fit into the container in some orientation?
            int[] bd = new int[3] { w, h, d };
            Array.Sort(bd);
            if (!(bd[0] <= contDims[0] && bd[1] <= contDims[1] && bd[2] <= contDims[2]))
            {
                // Cannot fit in container at all
                continue;
            }

            long vol = (long)w * h * d;
            int maxDim = Math.Max(w, Math.Max(h, d));
            int minDim = Math.Min(w, Math.Min(h, d));

            boxes.Add(new Box
            {
                Index = i,
                W = w,
                H = h,
                D = d,
                Volume = vol,
                MaxDim = maxDim,
                MinDim = minDim
            });
        }

        if (boxes.Count == 0)
        {
            Console.WriteLine(0);
            return;
        }

        List<Placement> bestPlacements;

        int count = boxes.Count;

        // Adapt strategy based on problem size
        if (count <= 128)
        {
            // Small instances: try several orderings with best-fit
            var r1 = PackWithOrder(boxes, contW, contH, contD, BoxOrder.VolumeAsc, useBestFit: true);
            var r2 = PackWithOrder(boxes, contW, contH, contD, BoxOrder.VolumeDesc, useBestFit: true);
            var r3 = PackWithOrder(boxes, contW, contH, contD, BoxOrder.MaxDimAsc, useBestFit: true);

            bestPlacements = r1;
            if (r2.Count > bestPlacements.Count) bestPlacements = r2;
            if (r3.Count > bestPlacements.Count) bestPlacements = r3;
        }
        else if (count <= 1000)
        {
            // Medium instances: one ordering, still using best-fit
            bestPlacements = PackWithOrder(boxes, contW, contH, contD, BoxOrder.VolumeAsc, useBestFit: true);
        }
        else
        {
            // Large instances: simpler, faster first-fit with small-first ordering
            bestPlacements = PackWithOrder(boxes, contW, contH, contD, BoxOrder.MinDimAsc, useBestFit: false);
        }

        var sb = new StringBuilder();
        sb.AppendLine(bestPlacements.Count.ToString());
        foreach (var p in bestPlacements)
        {
            sb.Append(p.BoxIndex);
            sb.Append(' ');
            sb.Append(p.X);
            sb.Append(' ');
            sb.Append(p.Y);
            sb.Append(' ');
            sb.Append(p.Z);
            sb.AppendLine();
        }

        Console.Write(sb.ToString());
    }

    static List<Placement> PackWithOrder(List<Box> inputBoxes, int contW, int contH, int contD, BoxOrder order, bool useBestFit)
    {
        var boxes = new List<Box>(inputBoxes);

        switch (order)
        {
            case BoxOrder.VolumeAsc:
                boxes.Sort((a, b) =>
                {
                    int cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    cmp = a.MaxDim.CompareTo(b.MaxDim);
                    if (cmp != 0) return cmp;
                    return a.MinDim.CompareTo(b.MinDim);
                });
                break;

            case BoxOrder.VolumeDesc:
                boxes.Sort((a, b) =>
                {
                    int cmp = b.Volume.CompareTo(a.Volume);
                    if (cmp != 0) return cmp;
                    cmp = b.MaxDim.CompareTo(a.MaxDim);
                    if (cmp != 0) return cmp;
                    return b.MinDim.CompareTo(a.MinDim);
                });
                break;

            case BoxOrder.MaxDimAsc:
                boxes.Sort((a, b) =>
                {
                    int cmp = a.MaxDim.CompareTo(b.MaxDim);
                    if (cmp != 0) return cmp;
                    cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    return a.MinDim.CompareTo(b.MinDim);
                });
                break;

            case BoxOrder.MinDimAsc:
                boxes.Sort((a, b) =>
                {
                    int cmp = a.MinDim.CompareTo(b.MinDim);
                    if (cmp != 0) return cmp;
                    cmp = a.Volume.CompareTo(b.Volume);
                    if (cmp != 0) return cmp;
                    return a.MaxDim.CompareTo(b.MaxDim);
                });
                break;
        }

        return PackGreedy(boxes, contW, contH, contD, useBestFit);
    }

    static List<Placement> PackGreedy(List<Box> boxes, int contW, int contH, int contD, bool useBestFit)
    {
        var spaces = new List<FreeSpace>();
        spaces.Add(new FreeSpace { X = 0, Y = 0, Z = 0, W = contW, H = contH, D = contD });

        var placements = new List<Placement>(boxes.Count);

        foreach (var box in boxes)
        {
            int selectedSpace = -1;
            int selBw = 0, selBh = 0, selBd = 0;
            long bestScore = long.MaxValue;

            if (useBestFit)
            {
                // Best-fit: scan all spaces and orientations, choose minimal slack
                for (int si = 0; si < spaces.Count; si++)
                {
                    FreeSpace s = spaces[si];

                    // Try all 6 orientations
                    TryOrientationBestFit(box.W, box.H, box.D, s, si, ref selectedSpace, ref selBw, ref selBh, ref selBd, ref bestScore);
                    TryOrientationBestFit(box.W, box.D, box.H, s, si, ref selectedSpace, ref selBw, ref selBh, ref selBd, ref bestScore);
                    TryOrientationBestFit(box.H, box.W, box.D, s, si, ref selectedSpace, ref selBw, ref selBh, ref selBd, ref bestScore);
                    TryOrientationBestFit(box.H, box.D, box.W, s, si, ref selectedSpace, ref selBw, ref selBh, ref selBd, ref bestScore);
                    TryOrientationBestFit(box.D, box.W, box.H, s, si, ref selectedSpace, ref selBw, ref selBh, ref selBd, ref bestScore);
                    TryOrientationBestFit(box.D, box.H, box.W, s, si, ref selectedSpace, ref selBw, ref selBh, ref selBd, ref bestScore);
                }
            }
            else
            {
                // First-fit: stop as soon as we find any space where it fits (best orientation within that space)
                bool found = false;
                for (int si = 0; si < spaces.Count && !found; si++)
                {
                    FreeSpace s = spaces[si];
                    long bestLocalScore = long.MaxValue;
                    int localBw = 0, localBh = 0, localBd = 0;

                    TryOrientationLocal(box.W, box.H, box.D, s, ref bestLocalScore, ref localBw, ref localBh, ref localBd);
                    TryOrientationLocal(box.W, box.D, box.H, s, ref bestLocalScore, ref localBw, ref localBh, ref localBd);
                    TryOrientationLocal(box.H, box.W, box.D, s, ref bestLocalScore, ref localBw, ref localBh, ref localBd);
                    TryOrientationLocal(box.H, box.D, box.W, s, ref bestLocalScore, ref localBw, ref localBh, ref localBd);
                    TryOrientationLocal(box.D, box.W, box.H, s, ref bestLocalScore, ref localBw, ref localBh, ref localBd);
                    TryOrientationLocal(box.D, box.H, box.W, s, ref bestLocalScore, ref localBw, ref localBh, ref localBd);

                    if (bestLocalScore != long.MaxValue)
                    {
                        selectedSpace = si;
                        selBw = localBw;
                        selBh = localBh;
                        selBd = localBd;
                        found = true;
                    }
                }
            }

            if (selectedSpace != -1)
            {
                // Place the box in selectedSpace at its origin (x,y,z)
                FreeSpace used = spaces[selectedSpace];
                int x = used.X;
                int y = used.Y;
                int z = used.Z;

                // Remove used space by swapping with last to keep O(1) removal
                int lastIndex = spaces.Count - 1;
                spaces[selectedSpace] = spaces[lastIndex];
                spaces.RemoveAt(lastIndex);

                int remW = used.W - selBw;
                int remH = used.H - selBh;
                int remD = used.D - selBd;

                // Split used space into up to 3 non-overlapping spaces:
                // 1) Right of box
                if (remW > 0)
                {
                    spaces.Add(new FreeSpace
                    {
                        X = x + selBw,
                        Y = y,
                        Z = z,
                        W = remW,
                        H = used.H,
                        D = used.D
                    });
                }

                // 2) Above box (over its footprint in X)
                if (remH > 0)
                {
                    spaces.Add(new FreeSpace
                    {
                        X = x,
                        Y = y + selBh,
                        Z = z,
                        W = selBw,
                        H = remH,
                        D = used.D
                    });
                }

                // 3) In front of box (over its footprint in X and Y)
                if (remD > 0)
                {
                    spaces.Add(new FreeSpace
                    {
                        X = x,
                        Y = y,
                        Z = z + selBd,
                        W = selBw,
                        H = selBh,
                        D = remD
                    });
                }

                placements.Add(new Placement
                {
                    BoxIndex = box.Index,
                    X = x,
                    Y = y,
                    Z = z
                });
            }
        }

        return placements;
    }

    // Best-fit helper: update global best candidate
    static void TryOrientationBestFit(
        int bw, int bh, int bd,
        FreeSpace s, int spaceIndex,
        ref int selectedSpace, ref int selBw, ref int selBh, ref int selBd, ref long bestScore)
    {
        if (bw <= s.W && bh <= s.H && bd <= s.D)
        {
            int dx = s.W - bw;
            int dy = s.H - bh;
            int dz = s.D - bd;
            long score = (long)dx * dx + (long)dy * dy + (long)dz * dz;

            if (score < bestScore)
            {
                bestScore = score;
                selectedSpace = spaceIndex;
                selBw = bw;
                selBh = bh;
                selBd = bd;
            }
        }
    }

    // First-fit helper: update best orientation for a single space
    static void TryOrientationLocal(
        int bw, int bh, int bd,
        FreeSpace s,
        ref long bestLocalScore, ref int localBw, ref int localBh, ref int localBd)
    {
        if (bw <= s.W && bh <= s.H && bd <= s.D)
        {
            int dx = s.W - bw;
            int dy = s.H - bh;
            int dz = s.D - bd;
            long score = (long)dx * dx + (long)dy * dy + (long)dz * dz;

            if (score < bestLocalScore)
            {
                bestLocalScore = score;
                localBw = bw;
                localBh = bh;
                localBd = bd;
            }
        }
    }

    // Simple fast scanner for large inputs
    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len, _ptr;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
            _buffer = new byte[1 << 16];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) throw new EndOfStreamException();
            } while (c == ' ' || c == '\n' || c == '\r' || c == '\t');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return val * sign;
        }
    }
}
```
