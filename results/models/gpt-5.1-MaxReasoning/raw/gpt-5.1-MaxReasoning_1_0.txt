#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <limits>

using std::size_t;

struct Point {
    double x;
    double y;
};

inline double dist2(const Point &a, const Point &b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

// ---------- Tiny N: exact Held-Karp DP (N <= 18) ----------

std::vector<uint32_t> solve_tiny_held_karp(const std::vector<Point> &pts) {
    size_t N = pts.size();
    if (N <= 1) {
        return {0};
    }
    if (N == 2) {
        return {0, 1};
    }

    const double INF = std::numeric_limits<double>::infinity();

    // Precompute full distance matrix with true Euclidean distances
    std::vector<double> dist(N * N);
    for (size_t i = 0; i < N; ++i) {
        for (size_t j = 0; j < N; ++j) {
            double dx = pts[i].x - pts[j].x;
            double dy = pts[i].y - pts[j].y;
            dist[i * N + j] = std::sqrt(dx * dx + dy * dy);
        }
    }

    size_t M = N - 1;              // cities 1..N-1
    uint32_t states = 1u << M;     // M <= 17 (because N <= 18)

    std::vector<double> dp(static_cast<size_t>(states) * M, INF);
    std::vector<int16_t> parent(static_cast<size_t>(states) * M, -1);

    // Initial transitions: 0 -> i
    for (size_t i = 1; i < N; ++i) {
        uint32_t mask = 1u << (i - 1);
        size_t idx = static_cast<size_t>(mask) * M + (i - 1);
        dp[idx] = dist[0 * N + i];
        parent[idx] = -1;
    }

    // DP over subsets of {1..N-1}
    for (uint32_t mask = 1; mask < states; ++mask) {
        for (size_t i = 0; i < M; ++i) {
            if (!(mask & (1u << i))) continue; // city (i+1) not in subset
            double cur = dp[static_cast<size_t>(mask) * M + i];
            if (!std::isfinite(cur)) continue;
            size_t city_i = i + 1; // actual city index

            uint32_t remaining = (~mask) & (states - 1);
            while (remaining) {
                uint32_t bit = remaining & -remaining;
                size_t j = __builtin_ctz(bit); // index in 0..M-1
                remaining ^= bit;

                size_t city_j = j + 1;
                uint32_t newMask = mask | static_cast<uint32_t>(bit);
                size_t idxNew = static_cast<size_t>(newMask) * M + j;
                double cand = cur + dist[city_i * N + city_j];
                if (cand < dp[idxNew]) {
                    dp[idxNew] = cand;
                    parent[idxNew] = static_cast<int16_t>(i);
                }
            }
        }
    }

    // Find best ending city (no return to 0)
    uint32_t fullMask = states - 1;
    double bestCost = INF;
    size_t bestLast = 0;
    for (size_t i = 0; i < M; ++i) {
        double v = dp[static_cast<size_t>(fullMask) * M + i];
        if (v < bestCost) {
            bestCost = v;
            bestLast = i;
        }
    }

    // Reconstruct path: 0 -> ... -> (bestLast+1)
    std::vector<uint32_t> rev;
    rev.reserve(M);
    uint32_t mask = fullMask;
    size_t last = bestLast;

    while (true) {
        uint32_t city = static_cast<uint32_t>(last + 1);
        rev.push_back(city);
        uint32_t prevMask = mask ^ (1u << last);
        if (prevMask == 0) {
            break;
        }
        int16_t prev = parent[static_cast<size_t>(mask) * M + last];
        last = static_cast<size_t>(prev);
        mask = prevMask;
    }

    std::vector<uint32_t> tour;
    tour.reserve(N);
    tour.push_back(0);
    for (auto it = rev.rbegin(); it != rev.rend(); ++it) {
        tour.push_back(*it);
    }
    return tour;
}

// ---------- Small/Medium N: Nearest Neighbor + full 2-opt (N up to ~8000) ----------

void twoOptFull(std::vector<uint32_t> &tour, const std::vector<Point> &pts, int maxIterations) {
    size_t n = tour.size();
    if (n < 4 || maxIterations <= 0) return;
    const double EPS = 1e-9;

    for (int iter = 0; iter < maxIterations; ++iter) {
        double bestDelta = -EPS;
        size_t best_i = 0, best_j = 0;
        bool found = false;

        for (size_t i = 0; i + 3 < n; ++i) {
            uint32_t a = tour[i];
            uint32_t b = tour[i + 1];
            double ab = dist2(pts[a], pts[b]);

            for (size_t j = i + 2; j + 1 < n; ++j) {
                uint32_t c = tour[j];
                uint32_t d = tour[j + 1];

                double cd = dist2(pts[c], pts[d]);
                double ac = dist2(pts[a], pts[c]);
                double bd = dist2(pts[b], pts[d]);

                double delta = (ac + bd) - (ab + cd);
                if (delta < bestDelta) {
                    bestDelta = delta;
                    best_i = i;
                    best_j = j;
                    found = true;
                }
            }
        }
        if (!found) break;
        std::reverse(tour.begin() + best_i + 1, tour.begin() + best_j + 1);
    }
}

std::vector<uint32_t> solve_small_nn_2opt(const std::vector<Point> &pts) {
    size_t N = pts.size();
    std::vector<uint32_t> tour;
    tour.reserve(N);
    std::vector<char> used(N, 0);

    // Nearest neighbor starting at city 0
    tour.push_back(0);
    used[0] = 1;

    for (size_t step = 1; step < N; ++step) {
        uint32_t last = tour.back();
        double bestD = std::numeric_limits<double>::infinity();
        size_t bestIdx = 0;
        for (size_t j = 0; j < N; ++j) {
            if (used[j]) continue;
            double d = dist2(pts[last], pts[j]);
            if (d < bestD) {
                bestD = d;
                bestIdx = j;
            }
        }
        used[bestIdx] = 1;
        tour.push_back(static_cast<uint32_t>(bestIdx));
    }

    int maxIterations;
    if (N <= 100) maxIterations = 200;
    else if (N <= 500) maxIterations = 100;
    else if (N <= 2000) maxIterations = 50;
    else if (N <= 4000) maxIterations = 20;
    else maxIterations = 10; // up to ~8000

    twoOptFull(tour, pts, maxIterations);
    return tour;
}

// ---------- Large N: Morton (Z-order) + windowed 2-opt ----------

struct MortonEntry {
    uint64_t key;
    uint32_t idx;
};

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return (part1by1(y) << 1) | part1by1(x);
}

void twoOptWindow(std::vector<uint32_t> &tour, const std::vector<Point> &pts,
                  int window, int maxIterations) {
    size_t n = tour.size();
    if (n < 4 || window <= 0 || maxIterations <= 0) return;
    const double EPS = 1e-9;

    for (int iter = 0; iter < maxIterations; ++iter) {
        double bestDelta = -EPS;
        size_t best_i = 0, best_j = 0;
        bool found = false;

        for (size_t i = 0; i + 3 < n; ++i) {
            uint32_t a = tour[i];
            uint32_t b = tour[i + 1];
            double ab = dist2(pts[a], pts[b]);

            size_t jStart = i + 2;
            size_t jEnd = i + 1 + static_cast<size_t>(window);
            if (jEnd > n - 2) jEnd = n - 2;
            if (jStart > jEnd) continue;

            for (size_t j = jStart; j <= jEnd; ++j) {
                uint32_t c = tour[j];
                uint32_t d = tour[j + 1];
                double cd = dist2(pts[c], pts[d]);
                double ac = dist2(pts[a], pts[c]);
                double bd = dist2(pts[b], pts[d]);
                double delta = (ac + bd) - (ab + cd);
                if (delta < bestDelta) {
                    bestDelta = delta;
                    best_i = i;
                    best_j = j;
                    found = true;
                }
            }
        }
        if (!found) break;
        std::reverse(tour.begin() + best_i + 1, tour.begin() + best_j + 1);
    }
}

std::vector<uint32_t> solve_morton(const std::vector<Point> &pts) {
    size_t N = pts.size();
    std::vector<uint32_t> tour(N);
    if (N == 0) return tour;
    if (N == 1) {
        tour[0] = 0;
        return tour;
    }

    // Compute bounding box
    double min_x = pts[0].x, max_x = pts[0].x;
    double min_y = pts[0].y, max_y = pts[0].y;
    for (size_t i = 1; i < N; ++i) {
        if (pts[i].x < min_x) min_x = pts[i].x;
        if (pts[i].x > max_x) max_x = pts[i].x;
        if (pts[i].y < min_y) min_y = pts[i].y;
        if (pts[i].y > max_y) max_y = pts[i].y;
    }

    const uint32_t MAX_COORD_INT = (1u << 21) - 1; // 21 bits per coord

    double range_x = max_x - min_x;
    double range_y = max_y - min_y;
    double scale_x = (range_x > 0) ? (double)MAX_COORD_INT / range_x : 0.0;
    double scale_y = (range_y > 0) ? (double)MAX_COORD_INT / range_y : 0.0;

    std::vector<MortonEntry> entries(N);
    for (size_t i = 0; i < N; ++i) {
        uint32_t xi = 0, yi = 0;
        if (scale_x > 0) {
            double fx = (pts[i].x - min_x) * scale_x;
            if (fx < 0.0) fx = 0.0;
            if (fx > MAX_COORD_INT) fx = MAX_COORD_INT;
            xi = static_cast<uint32_t>(fx);
        }
        if (scale_y > 0) {
            double fy = (pts[i].y - min_y) * scale_y;
            if (fy < 0.0) fy = 0.0;
            if (fy > MAX_COORD_INT) fy = MAX_COORD_INT;
            yi = static_cast<uint32_t>(fy);
        }
        entries[i].idx = static_cast<uint32_t>(i);
        entries[i].key = morton2D(xi, yi);
    }

    std::sort(entries.begin(), entries.end(),
              [](const MortonEntry &a, const MortonEntry &b) {
                  if (a.key != b.key) return a.key < b.key;
                  return a.idx < b.idx;
              });

    size_t pos0 = 0;
    for (size_t i = 0; i < N; ++i) {
        tour[i] = entries[i].idx;
        if (tour[i] == 0) pos0 = i;
    }

    // Rotate so that city 0 is first
    std::rotate(tour.begin(), tour.begin() + pos0, tour.end());

    // Light local optimization for moderately large N
    if (N <= 200000) {
        int window = (N <= 50000) ? 40 : 20;
        int maxIter = (N <= 50000) ? 8 : 5;
        twoOptWindow(tour, pts, window, maxIter);
    } else if (N <= 2000000) {
        twoOptWindow(tour, pts, 10, 3);
    }
    // For >2M cities, skip local optimization to keep runtime linearithmic.

    return tour;
}

// ---------- main ----------

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint64_t N_in;
    if (!(std::cin >> N_in)) {
        return 0;
    }
    size_t N = static_cast<size_t>(N_in);

    std::vector<Point> pts;
    pts.reserve(N);
    for (size_t i = 0; i < N; ++i) {
        double x, y;
        if (!(std::cin >> x >> y)) {
            // If input is malformed/short, just stop reading.
            break;
        }
        pts.push_back({x, y});
    }
    N = pts.size(); // in case fewer lines than claimed

    if (N == 0) {
        return 0;
    }
    if (N == 1) {
        std::cout << 0 << '\n';
        return 0;
    }

    const size_t TINY_N_MAX = 18;
    const size_t SMALL_N_MAX = 8000;

    std::vector<uint32_t> tour;

    if (N <= TINY_N_MAX) {
        tour = solve_tiny_held_karp(pts);
    } else if (N <= SMALL_N_MAX) {
        tour = solve_small_nn_2opt(pts);
    } else {
        tour = solve_morton(pts);
    }

    // Ensure valid size; fall back to identity if something went wrong
    if (tour.size() != N) {
        tour.resize(N);
        for (size_t i = 0; i < N; ++i) tour[i] = static_cast<uint32_t>(i);
    }
    if (tour[0] != 0) {
        auto it = std::find(tour.begin(), tour.end(), 0u);
        if (it != tour.end()) {
            std::rotate(tour.begin(), it, tour.end());
        } else {
            tour[0] = 0; // ensure start at 0; remaining still a permutation in fallback case
        }
    }

    for (size_t i = 0; i < N; ++i) {
        if (i) std::cout << ' ';
        std::cout << tour[i];
    }
    std::cout << '\n';

    return 0;
}
