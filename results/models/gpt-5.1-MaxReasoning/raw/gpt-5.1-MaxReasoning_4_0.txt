// Regular tetrahedron packing in a 3D polyhedral container (convex hull of input points).
// Strategy:
//  - Interpret container as convex hull of given vertices if vertex count is moderate.
//  - Otherwise approximate container as its axis-aligned bounding box (AABB).
//  - Pack *bounding spheres* of regular tetrahedra (radius = circumradius) on a 3D grid.
//  - Grid spacing >= 2R guarantees no overlap of tetrahedra.
//  - Grid spacing is scaled so that total number of placements remains manageable
//    even for huge containers (scales with container volume).
//  - All tetrahedra use the same orientation (Euler angles = 0,0,0).

#include <iostream>
#include <vector>
#include <cmath>
#include <limits>
#include <algorithm>
#include <unordered_map>
#include <cstdint>
#include <iomanip>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}

    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }

    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    Vec3 cross(const Vec3& o) const {
        return Vec3(
            y * o.z - z * o.y,
            z * o.x - x * o.z,
            x * o.y - y * o.x
        );
    }
    double norm2() const { return x * x + y * y + z * z; }
    double norm() const { return std::sqrt(norm2()); }
};

struct Face {
    int a, b, c;      // indices of vertices in the point array
    Vec3 normal;      // outward unit normal
    double offset;    // plane offset: normal·p + offset = 0
    bool removed;     // face removed during hull construction
};

// Build convex hull with a simple incremental algorithm.
// Returns true on success; on any degeneracy, returns false.
bool buildConvexHull3D(const vector<Vec3>& pts,
                       vector<Face>& faces,
                       Vec3& interiorPoint,
                       double diagLen)
{
    const int n = static_cast<int>(pts.size());
    faces.clear();
    if (n < 4) return false;

    // Tolerances based on overall scale
    const double eps = max(1e-12, 1e-9 * diagLen);
    const double eps2 = eps * eps;

    // 1) Pick first point with minimal x as i0
    int i0 = 0;
    for (int i = 1; i < n; ++i) {
        if (pts[i].x < pts[i0].x) i0 = i;
    }

    // 2) Pick farthest from i0 as i1
    int i1 = (i0 == 0 ? 1 : 0);
    double maxDist2 = 0.0;
    for (int i = 0; i < n; ++i) {
        if (i == i0) continue;
        double d2 = (pts[i] - pts[i0]).norm2();
        if (d2 > maxDist2) {
            maxDist2 = d2;
            i1 = i;
        }
    }
    if (maxDist2 < eps2) {
        // All points nearly coincident
        return false;
    }

    // 3) Pick i2 farthest from line i0-i1
    Vec3 d01 = pts[i1] - pts[i0];
    double maxArea2 = 0.0;
    int i2 = -1;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1) continue;
        Vec3 v = pts[i] - pts[i0];
        Vec3 cr = d01.cross(v);
        double a2 = cr.norm2();
        if (a2 > maxArea2) {
            maxArea2 = a2;
            i2 = i;
        }
    }
    if (i2 == -1 || maxArea2 < eps2) {
        // Points are nearly collinear
        return false;
    }

    // 4) Pick i3 farthest from plane (i0,i1,i2)
    Vec3 n012 = (pts[i1] - pts[i0]).cross(pts[i2] - pts[i0]);
    double maxDistPlane = 0.0;
    int i3 = -1;
    for (int i = 0; i < n; ++i) {
        if (i == i0 || i == i1 || i == i2) continue;
        double dist = fabs((pts[i] - pts[i0]).dot(n012));
        if (dist > maxDistPlane) {
            maxDistPlane = dist;
            i3 = i;
        }
    }
    if (i3 == -1 || maxDistPlane < eps * diagLen * 1e-3) {
        // Points nearly coplanar
        return false;
    }

    // Interior point = centroid of initial tetra
    interiorPoint = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) * 0.25;

    auto make_face = [&](int ia, int ib, int ic) -> Face {
        Face f;
        f.a = ia; f.b = ib; f.c = ic;
        Vec3 a = pts[ia];
        Vec3 b = pts[ib];
        Vec3 c = pts[ic];
        Vec3 n = (b - a).cross(c - a);
        double len = n.norm();
        if (len == 0.0) {
            // Degenerate; give a dummy normal (will be ignored because face never used)
            n = Vec3(0, 0, 0);
            f.normal = n;
            f.offset = 0.0;
            f.removed = true;
            return f;
        }
        n = n / len;
        // Ensure normal points *away* from interior
        if (n.dot(interiorPoint - a) > 0.0) {
            // Flip orientation
            std::swap(ib, ic);
            b = pts[ib];
            c = pts[ic];
            n = (b - a).cross(c - a);
            len = n.norm();
            if (len == 0.0) {
                n = Vec3(0, 0, 0);
                f.normal = n;
                f.offset = 0.0;
                f.removed = true;
                return f;
            }
            n = n / len;
        }
        f.a = ia; f.b = ib; f.c = ic;
        f.normal = n;
        f.offset = -n.dot(a);
        f.removed = false;
        return f;
    };

    // Initial 4 faces of the tetrahedron
    faces.reserve(64);
    faces.push_back(make_face(i0, i1, i2));
    faces.push_back(make_face(i0, i3, i1));
    faces.push_back(make_face(i0, i2, i3));
    faces.push_back(make_face(i1, i3, i2));

    // Incremental insertion of remaining points
    vector<bool> isInitial(n, false);
    isInitial[i0] = isInitial[i1] = isInitial[i2] = isInitial[i3] = true;

    for (int pi = 0; pi < n; ++pi) {
        if (isInitial[pi]) continue;
        const Vec3& p = pts[pi];

        // Find faces visible from p
        vector<int> visibleFaces;
        visibleFaces.reserve(faces.size());
        const int faceCountBefore = static_cast<int>(faces.size());
        bool outside = false;
        for (int fi = 0; fi < faceCountBefore; ++fi) {
            Face& f = faces[fi];
            if (f.removed) continue;
            double dist = f.normal.dot(p) + f.offset;
            if (dist > eps) {
                outside = true;
                visibleFaces.push_back(fi);
            }
        }
        if (!outside) {
            // Point inside or on hull; ignore
            continue;
        }

        // Collect horizon edges: edges of visible faces that are not shared by two visible faces.
        struct EdgeData {
            int count;
            int u, v; // oriented as in the face that contributed it
        };
        unordered_map<uint64_t, EdgeData> edgeMap;
        edgeMap.reserve(visibleFaces.size() * 3 * 2);

        auto add_edge = [&](int u, int v) {
            int a = std::min(u, v);
            int b = std::max(u, v);
            uint64_t key = (static_cast<uint64_t>(static_cast<uint32_t>(a)) << 32)
                         | static_cast<uint32_t>(b);
            auto it = edgeMap.find(key);
            if (it == edgeMap.end()) {
                EdgeData ed;
                ed.count = 1;
                ed.u = u;
                ed.v = v;
                edgeMap.emplace(key, ed);
            } else {
                it->second.count += 1;
            }
        };

        for (int fi : visibleFaces) {
            Face& f = faces[fi];
            f.removed = true;
            int ia = f.a, ib = f.b, ic = f.c;
            add_edge(ia, ib);
            add_edge(ib, ic);
            add_edge(ic, ia);
        }

        // Create new faces from horizon edges and point pi
        for (auto& kv : edgeMap) {
            EdgeData& ed = kv.second;
            if (ed.count != 1) continue; // interior edge
            int u = ed.u;
            int v = ed.v;

            // Candidate face (u, v, pi)
            Vec3 a = pts[u];
            Vec3 b = pts[v];
            Vec3 c = pts[pi];
            Vec3 n = (b - a).cross(c - a);
            double len = n.norm();
            if (len == 0.0) continue; // degenerate; skip
            n = n / len;
            // Ensure outward orientation (interiorPoint must be inside negative half-space)
            if (n.dot(interiorPoint - a) > 0.0) {
                // Flip to (u, pi, v)
                b = pts[pi];
                c = pts[v];
                n = (b - a).cross(c - a);
                len = n.norm();
                if (len == 0.0) continue;
                n = n / len;
                // After flip, it should point outward
            }

            Face newF;
            newF.a = u;
            newF.b = v;
            newF.c = pi;
            newF.normal = n;
            newF.offset = -n.dot(a);
            newF.removed = false;
            faces.push_back(newF);
        }
    }

    // Final sanity: if too few valid faces, consider hull build failed
    int validCount = 0;
    for (const auto& f : faces) if (!f.removed) ++validCount;
    if (validCount < 4) {
        faces.clear();
        return false;
    }
    return true;
}

// Test if a sphere with center c and radius R lies entirely inside a convex polyhedron
// defined by faces (normals are outward and normalized).
bool sphereInsideHull(const Vec3& c, double R,
                      const vector<Face>& faces,
                      double eps)
{
    for (const auto& f : faces) {
        if (f.removed) continue;
        double dist = f.normal.dot(c) + f.offset; // signed distance
        // To keep full sphere inside, center must be at most -R from plane
        if (dist > -R + eps) {
            return false;
        }
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edgeLength;
    if (!(cin >> V >> edgeLength)) {
        return 0;
    }

    vector<Vec3> vertices;
    vertices.reserve(max(V, 0));
    for (int i = 0; i < V; ++i) {
        double x, y, z;
        cin >> x >> y >> z;
        vertices.emplace_back(x, y, z);
    }

    // If edge length is non-positive, nothing to pack
    if (edgeLength <= 0.0 || vertices.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    // Compute bounding box
    double minX = numeric_limits<double>::infinity();
    double minY = numeric_limits<double>::infinity();
    double minZ = numeric_limits<double>::infinity();
    double maxX = -numeric_limits<double>::infinity();
    double maxY = -numeric_limits<double>::infinity();
    double maxZ = -numeric_limits<double>::infinity();

    for (const auto& p : vertices) {
        minX = std::min(minX, p.x);
        minY = std::min(minY, p.y);
        minZ = std::min(minZ, p.z);
        maxX = std::max(maxX, p.x);
        maxY = std::max(maxY, p.y);
        maxZ = std::max(maxZ, p.z);
    }

    double dx = maxX - minX;
    double dy = maxY - minY;
    double dz = maxZ - minZ;

    // Degenerate container
    if (dx <= 0.0 || dy <= 0.0 || dz <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    double diag = std::sqrt(dx * dx + dy * dy + dz * dz);

    // Tetrahedron key metrics
    const double e = edgeLength;
    const double R = e * std::sqrt(3.0 / 8.0);     // circumradius
    // const double r = e / std::sqrt(24.0);       // inradius (unused but documented)

    // If even one tetrahedron cannot fit inside the AABB, exit
    if (2.0 * R > dx || 2.0 * R > dy || 2.0 * R > dz) {
        cout << 0 << '\n';
        return 0;
    }

    // Decide whether to build full convex hull or just use AABB approximation.
    const int MAX_HULL_POINTS = 256;   // limit for hull construction (performance / robustness)
    bool useHull = false;
    vector<Face> hullFaces;
    Vec3 interiorPoint;

    if (V >= 4 && V <= MAX_HULL_POINTS) {
        if (buildConvexHull3D(vertices, hullFaces, interiorPoint, diag)) {
            useHull = true;
        }
    }

    // Grid packing setup
    // Target maximum number of grid positions (controls runtime / output size)
    const double TARGET_POSITIONS = 150000.0;

    double boundingVolume = dx * dy * dz;
    // Base grid spacing for non-overlapping bounding spheres
    double baseSpacing = 2.0 * R;

    // Adjust spacing so that estimated number of grid cells ~ TARGET_POSITIONS
    // Number of grid cells ~ boundingVolume / spacing^3
    double spacingFactor = 1.0;
    if (boundingVolume > 0.0) {
        double denom = TARGET_POSITIONS * 8.0 * R * R * R;
        if (denom > 0.0) {
            double ratio = boundingVolume / denom;
            if (ratio > 1.0) {
                spacingFactor = std::cbrt(ratio);
            }
        }
    }
    double spacing = baseSpacing * spacingFactor;
    if (spacing <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    // Effective bounds for centers: shrink by R from each side to keep sphere in AABB
    double startX = minX + R;
    double endX   = maxX - R;
    double startY = minY + R;
    double endY   = maxY - R;
    double startZ = minZ + R;
    double endZ   = maxZ - R;

    if (startX > endX || startY > endY || startZ > endZ) {
        cout << 0 << '\n';
        return 0;
    }

    // Tolerance for inside checks
    double planeEps = max(1e-12, 1e-9 * diag);

    struct Placement {
        double cx, cy, cz;
        double rx, ry, rz; // Euler angles
    };
    vector<Placement> placements;
    placements.reserve(static_cast<size_t>(TARGET_POSITIONS * 1.1));

    // Generate grid positions
    for (double x = startX; x <= endX + 1e-9 * fabs(endX); x += spacing) {
        if (x > endX + 1e-6) break; // numerical safety
        for (double y = startY; y <= endY + 1e-9 * fabs(endY); y += spacing) {
            if (y > endY + 1e-6) break;
            for (double z = startZ; z <= endZ + 1e-9 * fabs(endZ); z += spacing) {
                if (z > endZ + 1e-6) break;
                Vec3 c(x, y, z);

                if (useHull) {
                    if (!sphereInsideHull(c, R, hullFaces, planeEps)) {
                        continue;
                    }
                }
                // On AABB-only mode, the loop bounds already ensure the sphere fits.

                Placement pl;
                pl.cx = x; pl.cy = y; pl.cz = z;
                pl.rx = 0.0; pl.ry = 0.0; pl.rz = 0.0; // all tetrahedra share same orientation
                placements.push_back(pl);
            }
        }
    }

    // Output
    cout << placements.size() << '\n';
    cout.setf(std::ios::fixed);
    cout << setprecision(10);
    for (const auto& pl : placements) {
        cout << pl.cx << ' ' << pl.cy << ' ' << pl.cz << ' '
             << pl.rx << ' ' << pl.ry << ' ' << pl.rz << '\n';
    }

    return 0;
}
