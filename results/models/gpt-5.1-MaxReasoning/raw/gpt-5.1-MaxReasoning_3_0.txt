// Graph layout / edge-crossing minimization
// Single-file, compilable with: rustc 1.92, edition 2024

use std::collections::VecDeque;
use std::io::{self, Read, Write};

/// Fast scanner for whitespace-separated unsigned integers.
struct FastScanner {
    buf: Vec<u8>,
    idx: usize,
}

impl FastScanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let n = self.buf.len();
        let mut i = self.idx;

        // skip whitespace
        while i < n && (self.buf[i] == b' ' || self.buf[i] == b'\n' ||
                        self.buf[i] == b'\r' || self.buf[i] == b'\t') {
            i += 1;
        }
        if i >= n {
            self.idx = i;
            return None;
        }

        let mut val: usize = 0;
        while i < n && self.buf[i].is_ascii_digit() {
            val = val * 10 + (self.buf[i] - b'0') as usize;
            i += 1;
        }
        self.idx = i;
        Some(val)
    }
}

/// Very small, fast PRNG (xorshift-like).
struct Rng {
    state: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        // xorshift*
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }

    fn next_f32(&mut self) -> f32 {
        let v = (self.next_u64() >> 32) as u32;
        (v as f32) / (u32::MAX as f32 + 1.0)
    }
}

/// Iteration schedule depending on component size.
fn iterations_for_comp_size(n: usize) -> usize {
    if n <= 10 {
        300
    } else if n <= 30 {
        250
    } else if n <= 100 {
        200
    } else if n <= 300 {
        150
    } else if n <= 1_000 {
        120
    } else if n <= 5_000 {
        90
    } else if n <= 20_000 {
        70
    } else if n <= 100_000 {
        50
    } else {
        40
    }
}

/// Fruchterman-Reingold layout for a single connected component.
/// - `nodes`: global node indices belonging to this component
/// - `adj`: global adjacency list
/// - `pos`: global positions (modified in-place)
/// - `index_in_comp`: mapping from global node id -> local index in `nodes`
/// - `iterations`: max number of iterations
/// - `use_all_pairs`: if true, use O(n^2) repulsion; otherwise, grid-based approximation
fn fr_layout_component(
    nodes: &[usize],
    adj: &Vec<Vec<usize>>,
    pos: &mut Vec<[f32; 2]>,
    index_in_comp: &Vec<usize>,
    iterations: usize,
    use_all_pairs: bool,
) {
    let n = nodes.len();
    if n <= 2 || iterations == 0 {
        return;
    }

    // Basic FR parameters tuned for reasonable scale.
    let side: f32 = 100.0;
    let area: f32 = side * side; // 10,000
    let k: f32 = (area / n as f32).sqrt(); // ideal edge length
    let k2: f32 = k * k;
    let mut t: f32 = side / 10.0; // initial "temperature" ~= 10
    let cooling: f32 = t / (iterations as f32 + 1.0);
    let eps: f32 = 0.001;

    // Displacement accumulator.
    let mut disp = vec![[0.0f32; 2]; n];

    // Grid structures for approximate repulsion when needed.
    let (mut grid, mut row_idx, mut col_idx, rows, cols) = if use_all_pairs {
        (
            Vec::<Vec<usize>>::new(),
            Vec::<usize>::new(),
            Vec::<usize>::new(),
            0usize,
            0usize,
        )
    } else {
        let grid_side = ((n as f32).sqrt().ceil() as usize).clamp(2, 2000);
        let rows = grid_side;
        let cols = grid_side;
        let cells = rows * cols;
        let grid = vec![Vec::<usize>::new(); cells];
        let row_idx = vec![0usize; n];
        let col_idx = vec![0usize; n];
        (grid, row_idx, col_idx, rows, cols)
    };

    for _iter in 0..iterations {
        // Reset displacements.
        for d in &mut disp {
            d[0] = 0.0;
            d[1] = 0.0;
        }

        // ---------------- Repulsive forces ----------------
        if use_all_pairs {
            for i in 0..n {
                let vi = nodes[i];
                let xi = pos[vi][0];
                let yi = pos[vi][1];
                for j in (i + 1)..n {
                    let vj = nodes[j];
                    let xj = pos[vj][0];
                    let yj = pos[vj][1];

                    let dx = xi - xj;
                    let dy = yi - yj;
                    let dist2 = dx * dx + dy * dy + eps;
                    let dist = dist2.sqrt();
                    let force = k2 / dist;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;

                    disp[i][0] += fx;
                    disp[i][1] += fy;
                    disp[j][0] -= fx;
                    disp[j][1] -= fy;
                }
            }
        } else {
            // Grid-based approximation

            // 1) bounding box
            let mut min_x = f32::INFINITY;
            let mut max_x = f32::NEG_INFINITY;
            let mut min_y = f32::INFINITY;
            let mut max_y = f32::NEG_INFINITY;
            for &v in nodes {
                let x = pos[v][0];
                let y = pos[v][1];
                if x < min_x {
                    min_x = x;
                }
                if x > max_x {
                    max_x = x;
                }
                if y < min_y {
                    min_y = y;
                }
                if y > max_y {
                    max_y = y;
                }
            }
            let width = (max_x - min_x).max(1e-3);
            let height = (max_y - min_y).max(1e-3);

            let cols_f = cols as f32;
            let rows_f = rows as f32;
            let cell_w = width / cols_f + 1e-6;
            let cell_h = height / rows_f + 1e-6;

            // 2) clear grid
            for cell in &mut grid {
                cell.clear();
            }

            // 3) assign nodes to cells
            for (idx, &v) in nodes.iter().enumerate() {
                let x = pos[v][0];
                let y = pos[v][1];
                let mut cx = ((x - min_x) / cell_w).floor() as isize;
                let mut cy = ((y - min_y) / cell_h).floor() as isize;
                if cx < 0 {
                    cx = 0;
                } else if cx >= cols as isize {
                    cx = cols as isize - 1;
                }
                if cy < 0 {
                    cy = 0;
                } else if cy >= rows as isize {
                    cy = rows as isize - 1;
                }
                let col = cx as usize;
                let row = cy as usize;
                let cell_index = row * cols + col;
                grid[cell_index].push(idx);
                col_idx[idx] = col;
                row_idx[idx] = row;
            }

            // 4) local neighborhood repulsion
            for i in 0..n {
                let vi = nodes[i];
                let xi = pos[vi][0];
                let yi = pos[vi][1];
                let row = row_idx[i] as isize;
                let col = col_idx[i] as isize;

                for dr in -1..=1 {
                    let r = row + dr;
                    if r < 0 || r >= rows as isize {
                        continue;
                    }
                    for dc in -1..=1 {
                        let c = col + dc;
                        if c < 0 || c >= cols as isize {
                            continue;
                        }
                        let cell_index = (r as usize) * cols + c as usize;
                        let cell = &grid[cell_index];
                        for &j in cell {
                            if j <= i {
                                continue;
                            }
                            let vj = nodes[j];
                            let xj = pos[vj][0];
                            let yj = pos[vj][1];

                            let dx = xi - xj;
                            let dy = yi - yj;
                            let dist2 = dx * dx + dy * dy + eps;
                            let dist = dist2.sqrt();
                            let force = k2 / dist;
                            let fx = dx / dist * force;
                            let fy = dy / dist * force;

                            disp[i][0] += fx;
                            disp[i][1] += fy;
                            disp[j][0] -= fx;
                            disp[j][1] -= fy;
                        }
                    }
                }
            }
        }

        // ---------------- Attractive forces (edges) ----------------
        for &v in nodes {
            let i = index_in_comp[v];
            let xv = pos[v][0];
            let yv = pos[v][1];
            for &u in &adj[v] {
                // process each undirected edge once
                if u <= v {
                    continue;
                }
                let j = index_in_comp[u];
                let xu = pos[u][0];
                let yu = pos[u][1];

                let dx = xv - xu;
                let dy = yv - yu;
                let dist2 = dx * dx + dy * dy + eps;
                let dist = dist2.sqrt();
                let force = dist2 / k;
                let fx = dx / dist * force;
                let fy = dy / dist * force;

                // v is pulled towards u
                disp[i][0] -= fx;
                disp[i][1] -= fy;
                // u is pulled towards v
                disp[j][0] += fx;
                disp[j][1] += fy;
            }
        }

        // ---------------- Position update with temperature ----------------
        let temp = t;
        if temp <= 0.0 {
            break;
        }
        for (idx, &v) in nodes.iter().enumerate() {
            let dx = disp[idx][0];
            let dy = disp[idx][1];
            let disp_len2 = dx * dx + dy * dy;
            if disp_len2 > 0.0 {
                let disp_len = disp_len2.sqrt();
                let scale = (disp_len.min(temp)) / disp_len;
                pos[v][0] += dx * scale;
                pos[v][1] += dy * scale;
            }
        }

        t -= cooling;
        if t <= 0.01 {
            break;
        }
    }
}

fn main() {
    let mut scanner = FastScanner::new();

    let n = match scanner.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = scanner.next_usize().unwrap_or(0);

    // Build adjacency list; skip self-loops (they don't change connectivity
    // and can cause numerical issues).
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u_opt = scanner.next_usize();
        let v_opt = scanner.next_usize();
        if u_opt.is_none() || v_opt.is_none() {
            break;
        }
        let u = u_opt.unwrap();
        let v = v_opt.unwrap();
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Find connected components.
    let mut comp_id = vec![usize::MAX; n];
    let mut components: Vec<Vec<usize>> = Vec::new();
    let mut queue: VecDeque<usize> = VecDeque::new();

    for start in 0..n {
        if comp_id[start] != usize::MAX {
            continue;
        }
        let cid = components.len();
        comp_id[start] = cid;
        queue.clear();
        queue.push_back(start);
        let mut nodes = Vec::new();

        while let Some(v) = queue.pop_front() {
            nodes.push(v);
            for &u in &adj[v] {
                if comp_id[u] == usize::MAX {
                    comp_id[u] = cid;
                    queue.push_back(u);
                }
            }
        }
        components.push(nodes);
    }

    // Global positions.
    let mut pos: Vec<[f32; 2]> = vec![[0.0, 0.0]; n];

    // Mapping from global node id to index within its component.
    let mut index_in_comp: Vec<usize> = vec![usize::MAX; n];

    let mut rng = Rng::new(0x1234_5678_9ABC_DEF0u64);
    let comp_count = components.len();
    let mut comp_max_dim: Vec<f32> = vec![1.0; comp_count];

    for (cid, nodes) in components.iter().enumerate() {
        let csize = nodes.len();
        if csize == 0 {
            comp_max_dim[cid] = 1.0;
            continue;
        }
        if csize == 1 {
            let v = nodes[0];
            pos[v][0] = 0.0;
            pos[v][1] = 0.0;
            comp_max_dim[cid] = 1.0;
            continue;
        }
        if csize == 2 {
            let v0 = nodes[0];
            let v1 = nodes[1];
            pos[v0][0] = -0.5;
            pos[v0][1] = 0.0;
            pos[v1][0] = 0.5;
            pos[v1][1] = 0.0;
            comp_max_dim[cid] = 1.0;
            continue;
        }

        // Map global ids to local indices and initialize positions randomly.
        for (i, &v) in nodes.iter().enumerate() {
            index_in_comp[v] = i;
            let rx = rng.next_f32() - 0.5;
            let ry = rng.next_f32() - 0.5;
            pos[v][0] = rx * 100.0; // within [-50, 50]
            pos[v][1] = ry * 100.0;
        }

        let iterations = iterations_for_comp_size(csize);
        let use_all_pairs = csize <= 500;

        fr_layout_component(
            nodes,
            &adj,
            &mut pos,
            &index_in_comp,
            iterations,
            use_all_pairs,
        );

        // Recentre this component around (0,0) and record its size.
        let mut min_x = f32::INFINITY;
        let mut max_x = f32::NEG_INFINITY;
        let mut min_y = f32::INFINITY;
        let mut max_y = f32::NEG_INFINITY;
        for &v in nodes {
            let x = pos[v][0];
            let y = pos[v][1];
            if x < min_x {
                min_x = x;
            }
            if x > max_x {
                max_x = x;
            }
            if y < min_y {
                min_y = y;
            }
            if y > max_y {
                max_y = y;
            }
        }
        let center_x = 0.5 * (min_x + max_x);
        let center_y = 0.5 * (min_y + max_y);
        for &v in nodes {
            pos[v][0] -= center_x;
            pos[v][1] -= center_y;
        }
        let dim_x = max_x - min_x;
        let dim_y = max_y - min_y;
        let d = dim_x.max(dim_y).max(1.0);
        comp_max_dim[cid] = d;
    }

    // Pack components so they don't overlap.
    if comp_count > 1 {
        let mut max_dim_global = 1.0f32;
        for &d in &comp_max_dim {
            if d > max_dim_global {
                max_dim_global = d;
            }
        }
        let gap_factor: f32 = 1.5;
        let cell_size: f32 = max_dim_global * gap_factor + 10.0;

        let cols = ((comp_count as f32).sqrt().ceil() as usize).max(1);
        let rows = (comp_count + cols - 1) / cols;

        for (cid, nodes) in components.iter().enumerate() {
            let row = cid / cols;
            let col = cid % cols;
            let offset_x =
                (col as f32 - (cols as f32 - 1.0) / 2.0) * cell_size;
            let offset_y =
                (row as f32 - (rows as f32 - 1.0) / 2.0) * cell_size;
            for &v in nodes {
                pos[v][0] += offset_x;
                pos[v][1] += offset_y;
            }
        }
    }

    // Output coordinates.
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    for v in 0..n {
        // 6 decimal places is plenty; adjust if needed.
        writeln!(out, "{:.6} {:.6}", pos[v][0], pos[v][1]).unwrap();
    }
}
