// 3D Bin Packing Solver for identical polyhedra in an axis-aligned box.
// Strategy:
//   - Read a single polyhedron, ignore faces (we use only geometry of vertices).
//   - Try multiple orientations (24 axis-aligned cube symmetries + some random rotations),
//     compute oriented bounding boxes (AABB in container coordinates).
//   - For each orientation, estimate how many copies fit in a 3D grid (AABB packing).
//   - Choose the orientation with the highest theoretical count.
//   - For very large counts, coarsen the grid to cap output size (adaptive to problem scale).
//   - Output grid placements with a single global quaternion per orientation.
//
// This is a heuristic AABB packing, but guaranteed non-intersection: disjoint AABBs ⇒ disjoint polyhedra.
//
// Compile with: csc Program.cs
using System;
using System.IO;
using System.Globalization;
using System.Collections.Generic;

namespace BinPacking3DSolver
{
    struct Vec3d
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3d(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }
    }

    struct Matrix3x3
    {
        public double M00, M01, M02;
        public double M10, M11, M12;
        public double M20, M21, M22;

        public Vec3d Multiply(Vec3d v)
        {
            return new Vec3d(
                M00 * v.X + M01 * v.Y + M02 * v.Z,
                M10 * v.X + M11 * v.Y + M12 * v.Z,
                M20 * v.X + M21 * v.Y + M22 * v.Z
            );
        }

        public static Matrix3x3 FromQuaternion(Quaternion q)
        {
            // Assumes q is normalized
            double w = q.W, x = q.X, y = q.Y, z = q.Z;

            double xx = x * x;
            double yy = y * y;
            double zz = z * z;
            double xy = x * y;
            double xz = x * z;
            double yz = y * z;
            double wx = w * x;
            double wy = w * y;
            double wz = w * z;

            Matrix3x3 m;
            m.M00 = 1.0 - 2.0 * (yy + zz);
            m.M11 = 1.0 - 2.0 * (xx + zz);
            m.M22 = 1.0 - 2.0 * (xx + yy);

            m.M01 = 2.0 * (xy - wz);
            m.M10 = 2.0 * (xy + wz);

            m.M02 = 2.0 * (xz + wy);
            m.M20 = 2.0 * (xz - wy);

            m.M12 = 2.0 * (yz - wx);
            m.M21 = 2.0 * (yz + wx);

            return m;
        }
    }

    struct Quaternion
    {
        public double W;
        public double X;
        public double Y;
        public double Z;

        public Quaternion(double w, double x, double y, double z)
        {
            W = w; X = x; Y = y; Z = z;
        }

        public void Normalize()
        {
            double norm = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
            if (norm <= 0.0) { W = 1.0; X = Y = Z = 0.0; }
            else
            {
                double inv = 1.0 / norm;
                W *= inv; X *= inv; Y *= inv; Z *= inv;
            }
        }

        public static Quaternion FromMatrix(Matrix3x3 m)
        {
            double trace = m.M00 + m.M11 + m.M22;
            double w, x, y, z;

            if (trace > 0.0)
            {
                double s = Math.Sqrt(trace + 1.0) * 2.0; // s = 4 * w
                w = 0.25 * s;
                x = (m.M21 - m.M12) / s;
                y = (m.M02 - m.M20) / s;
                z = (m.M10 - m.M01) / s;
            }
            else if (m.M00 > m.M11 && m.M00 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0; // s = 4 * x
                w = (m.M21 - m.M12) / s;
                x = 0.25 * s;
                y = (m.M01 + m.M10) / s;
                z = (m.M02 + m.M20) / s;
            }
            else if (m.M11 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0; // s = 4 * y
                w = (m.M02 - m.M20) / s;
                x = (m.M01 + m.M10) / s;
                y = 0.25 * s;
                z = (m.M12 + m.M21) / s;
            }
            else
            {
                double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0; // s = 4 * z
                w = (m.M10 - m.M01) / s;
                x = (m.M02 + m.M20) / s;
                y = (m.M12 + m.M21) / s;
                z = 0.25 * s;
            }

            Quaternion q = new Quaternion(w, x, y, z);
            q.Normalize();
            return q;
        }

        public static Quaternion RandomUniform(Random rng)
        {
            // Uniform random quaternion on S^3
            double u1 = rng.NextDouble();
            double u2 = rng.NextDouble();
            double u3 = rng.NextDouble();

            double sqrt1MinusU1 = Math.Sqrt(1.0 - u1);
            double sqrtU1 = Math.Sqrt(u1);
            double twoPiU2 = 2.0 * Math.PI * u2;
            double twoPiU3 = 2.0 * Math.PI * u3;

            double x = sqrt1MinusU1 * Math.Sin(twoPiU2);
            double y = sqrt1MinusU1 * Math.Cos(twoPiU2);
            double z = sqrtU1 * Math.Sin(twoPiU3);
            double w = sqrtU1 * Math.Cos(twoPiU3);

            Quaternion q = new Quaternion(w, x, y, z);
            q.Normalize();
            return q;
        }
    }

    enum ScaleCategory
    {
        Small,
        Medium,
        Large,
        Extreme
    }

    struct OrientationResult
    {
        public Matrix3x3 R;
        public Quaternion Q;
        public Vec3d Min;
        public Vec3d Max;
        public Vec3d Size;
        public long Nx;
        public long Ny;
        public long Nz;
        public double RawCount; // theoretical grid count (Nx*Ny*Nz as double)
    }

    class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c = Read();
            while (c <= ' ' && c >= 0) c = Read();
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return sign * val;
        }

        public double NextDouble()
        {
            int c = Read();
            while (c <= ' ' && c >= 0) c = Read();
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            long intPart = 0;
            while (c >= '0' && c <= '9')
            {
                intPart = intPart * 10 + (c - '0');
                c = Read();
            }
            double result = intPart;
            if (c == '.')
            {
                c = Read();
                double place = 0.1;
                while (c >= '0' && c <= '9')
                {
                    result += (c - '0') * place;
                    place *= 0.1;
                    c = Read();
                }
            }
            return sign * result;
        }
    }

    class Program
    {
        static void Main(string[] args)
        {
            var scanner = new FastScanner(Console.OpenStandardInput());

            // Read polyhedron
            int V, F;
            try
            {
                V = scanner.NextInt();
            }
            catch
            {
                // No input
                return;
            }
            F = scanner.NextInt();
            if (V <= 0)
            {
                Console.WriteLine("0");
                return;
            }

            Vec3d[] vertices = new Vec3d[V];
            for (int i = 0; i < V; i++)
            {
                double x = scanner.NextDouble();
                double y = scanner.NextDouble();
                double z = scanner.NextDouble();
                vertices[i] = new Vec3d(x, y, z);
            }

            // Faces are not needed for AABB packing; read and ignore
            for (int i = 0; i < F; i++)
            {
                // a b c
                scanner.NextInt();
                scanner.NextInt();
                scanner.NextInt();
            }

            // Container dimensions
            double W = scanner.NextDouble();
            double H = scanner.NextDouble();
            double D = scanner.NextDouble();

            if (W <= 0 || H <= 0 || D <= 0)
            {
                Console.WriteLine("0");
                return;
            }

            double maxDim = Math.Max(W, Math.Max(H, D));
            ScaleCategory scale;
            if (maxDim <= 20.0) scale = ScaleCategory.Small;
            else if (maxDim <= 60.0) scale = ScaleCategory.Medium;
            else if (maxDim <= 200.0) scale = ScaleCategory.Large;
            else scale = ScaleCategory.Extreme;

            OrientationResult best = FindBestOrientation(vertices, W, H, D, scale);

            // If nothing fits
            if (best.RawCount <= 0.0 || best.Nx <= 0 || best.Ny <= 0 || best.Nz <= 0)
            {
                Console.WriteLine("0");
                return;
            }

            // Adaptive output limit to keep runtime reasonable even for massive containers
            // Adjust this if needed; it caps the number of packed copies we actually output.
            const long targetMaxPlacements = 1_000_000L;

            long nx = best.Nx;
            long ny = best.Ny;
            long nz = best.Nz;

            double rawCount = best.RawCount;
            long step = 1;

            if (rawCount > targetMaxPlacements)
            {
                double factor = rawCount / targetMaxPlacements;
                double stepDouble = Math.Ceiling(Math.Pow(factor, 1.0 / 3.0));
                if (stepDouble < 1.0) stepDouble = 1.0;
                if (stepDouble > long.MaxValue) stepDouble = long.MaxValue;
                step = (long)stepDouble;
                if (step < 1) step = 1;

                // Ensure we can place at least one along each dimension
                while (step > 1 && ((nx / step) == 0 || (ny / step) == 0 || (nz / step) == 0))
                {
                    step--;
                }
                if (step < 1) step = 1;
            }

            long effNx = nx / step;
            long effNy = ny / step;
            long effNz = nz / step;

            if (effNx <= 0 || effNy <= 0 || effNz <= 0)
            {
                Console.WriteLine("0");
                return;
            }

            // Final effective count is exact
            long effCount = effNx * effNy * effNz;
            if (effCount > targetMaxPlacements)
            {
                // In very pathological numeric cases we may still exceed target: increase step slightly.
                while (effCount > targetMaxPlacements && step < long.MaxValue)
                {
                    step++;
                    effNx = nx / step;
                    effNy = ny / step;
                    effNz = nz / step;
                    if (effNx <= 0 || effNy <= 0 || effNz <= 0)
                    {
                        break;
                    }
                    effCount = effNx * effNy * effNz;
                }
            }

            if (effNx <= 0 || effNy <= 0 || effNz <= 0)
            {
                Console.WriteLine("0");
                return;
            }

            effCount = effNx * effNy * effNz;

            var culture = CultureInfo.InvariantCulture;
            using (var writer = new StreamWriter(Console.OpenStandardOutput()))
            {
                writer.NewLine = "\n";
                writer.WriteLine(effCount.ToString(culture));

                Quaternion q = best.Q;
                Vec3d min = best.Min;
                Vec3d size = best.Size;

                // Output placements: tx ty tz qw qx qy qz
                // tx,ty,tz chosen so that rotated poly's bounding box min corner
                // is at (ix*step*size.X, iy*step*size.Y, iz*step*size.Z).
                for (long ix = 0; ix < effNx; ix++)
                {
                    double x0 = ix * step * size.X;
                    for (long iy = 0; iy < effNy; iy++)
                    {
                        double y0 = iy * step * size.Y;
                        for (long iz = 0; iz < effNz; iz++)
                        {
                            double z0 = iz * step * size.Z;

                            double tx = x0 - min.X;
                            double ty = y0 - min.Y;
                            double tz = z0 - min.Z;

                            writer.Write(tx.ToString("R", culture));
                            writer.Write(' ');
                            writer.Write(ty.ToString("R", culture));
                            writer.Write(' ');
                            writer.Write(tz.ToString("R", culture));
                            writer.Write(' ');
                            writer.Write(q.W.ToString("R", culture));
                            writer.Write(' ');
                            writer.Write(q.X.ToString("R", culture));
                            writer.Write(' ');
                            writer.Write(q.Y.ToString("R", culture));
                            writer.Write(' ');
                            writer.WriteLine(q.Z.ToString("R", culture));
                        }
                    }
                }
            }
        }

        static OrientationResult FindBestOrientation(
            Vec3d[] vertices,
            double W, double H, double D,
            ScaleCategory scale)
        {
            int vCount = vertices.Length;
            const double minSizeEps = 1e-9;

            // Orientation search budget adapts to polyhedron complexity and container scale.
            long baseBudget;
            int maxRandom;
            switch (scale)
            {
                case ScaleCategory.Small:
                    baseBudget = 50_000_000L;
                    maxRandom = 256;
                    break;
                case ScaleCategory.Medium:
                    baseBudget = 20_000_000L;
                    maxRandom = 128;
                    break;
                case ScaleCategory.Large:
                    baseBudget = 10_000_000L;
                    maxRandom = 64;
                    break;
                default:
                    baseBudget = 5_000_000L;
                    maxRandom = 32;
                    break;
            }

            // Generate 24 axis-aligned cube-symmetry rotations
            List<Matrix3x3> mats = new List<Matrix3x3>(24 + maxRandom);
            List<Quaternion> quats = new List<Quaternion>(24 + maxRandom);

            int[][] perms = new int[][]
            {
                new int[]{0,1,2},
                new int[]{0,2,1},
                new int[]{1,0,2},
                new int[]{1,2,0},
                new int[]{2,0,1},
                new int[]{2,1,0}
            };

            for (int pi = 0; pi < perms.Length; pi++)
            {
                int[] p = perms[pi];
                int paritySign = PermutationParitySign(p[0], p[1], p[2]);

                for (int mask = 0; mask < 8; mask++)
                {
                    int sx = ((mask & 1) == 0) ? 1 : -1;
                    int sy = ((mask & 2) == 0) ? 1 : -1;
                    int sz = ((mask & 4) == 0) ? 1 : -1;
                    int signProduct = sx * sy * sz;
                    if (signProduct != paritySign) continue; // ensure det = +1

                    Matrix3x3 m = BuildMatrixFromPermAndSigns(p, sx, sy, sz);
                    Quaternion q = Quaternion.FromMatrix(m);
                    mats.Add(m);
                    quats.Add(q);
                }
            }

            int baseOrientationCount = mats.Count; // should be 24

            long remainingBudget = baseBudget - (long)baseOrientationCount * vCount;
            if (remainingBudget < 0) remainingBudget = 0;

            int randomOrientations = 0;
            if (vCount > 0)
            {
                randomOrientations = (int)(remainingBudget / vCount);
                if (randomOrientations > maxRandom) randomOrientations = maxRandom;
                if (randomOrientations < 0) randomOrientations = 0;
            }

            if (randomOrientations > 0)
            {
                Random rng = new Random(1);
                for (int i = 0; i < randomOrientations; i++)
                {
                    Quaternion qRand = Quaternion.RandomUniform(rng);
                    Matrix3x3 mRand = Matrix3x3.FromQuaternion(qRand);
                    mats.Add(mRand);
                    quats.Add(qRand);
                }
            }

            OrientationResult best = default(OrientationResult);
            double bestCount = -1.0;
            bool hasBest = false;

            for (int idx = 0; idx < mats.Count; idx++)
            {
                Matrix3x3 R = mats[idx];
                Quaternion Q = quats[idx];

                OrientationResult cand = EvaluateOrientation(vertices, R, Q, W, H, D, minSizeEps);
                if (cand.RawCount > bestCount)
                {
                    bestCount = cand.RawCount;
                    best = cand;
                    hasBest = true;
                }
            }

            if (!hasBest)
            {
                // Nothing fits
                best = default(OrientationResult);
            }

            return best;
        }

        static OrientationResult EvaluateOrientation(
            Vec3d[] vertices,
            Matrix3x3 R,
            Quaternion q,
            double W, double H, double D,
            double minSizeEps)
        {
            Vec3d min = new Vec3d(double.PositiveInfinity, double.PositiveInfinity, double.PositiveInfinity);
            Vec3d max = new Vec3d(double.NegativeInfinity, double.NegativeInfinity, double.NegativeInfinity);

            for (int i = 0; i < vertices.Length; i++)
            {
                Vec3d v = vertices[i];
                Vec3d r = R.Multiply(v);

                if (r.X < min.X) min.X = r.X;
                if (r.Y < min.Y) min.Y = r.Y;
                if (r.Z < min.Z) min.Z = r.Z;

                if (r.X > max.X) max.X = r.X;
                if (r.Y > max.Y) max.Y = r.Y;
                if (r.Z > max.Z) max.Z = r.Z;
            }

            Vec3d size = new Vec3d(max.X - min.X, max.Y - min.Y, max.Z - min.Z);

            if (size.X < minSizeEps || size.Y < minSizeEps || size.Z < minSizeEps)
            {
                // Degenerate orientation; can't pack
                OrientationResult resDeg;
                resDeg.R = R;
                resDeg.Q = q;
                resDeg.Min = min;
                resDeg.Max = max;
                resDeg.Size = size;
                resDeg.Nx = resDeg.Ny = resDeg.Nz = 0;
                resDeg.RawCount = 0.0;
                return resDeg;
            }

            long nx = (long)Math.Floor(W / size.X);
            long ny = (long)Math.Floor(H / size.Y);
            long nz = (long)Math.Floor(D / size.Z);

            if (nx <= 0 || ny <= 0 || nz <= 0)
            {
                OrientationResult resZero;
                resZero.R = R;
                resZero.Q = q;
                resZero.Min = min;
                resZero.Max = max;
                resZero.Size = size;
                resZero.Nx = resZero.Ny = resZero.Nz = 0;
                resZero.RawCount = 0.0;
                return resZero;
            }

            double rawCount = (double)nx * (double)ny;
            // Prevent double overflow in extreme pathological cases
            if (rawCount > 1e300) rawCount = 1e300;
            rawCount *= (double)nz;
            if (rawCount > 1e300) rawCount = 1e300;

            OrientationResult res;
            res.R = R;
            res.Q = q;
            res.Min = min;
            res.Max = max;
            res.Size = size;
            res.Nx = nx;
            res.Ny = ny;
            res.Nz = nz;
            res.RawCount = rawCount;
            return res;
        }

        static int PermutationParitySign(int a, int b, int c)
        {
            // Only six possible permutations of {0,1,2}
            if ((a == 0 && b == 1 && c == 2) ||
                (a == 1 && b == 2 && c == 0) ||
                (a == 2 && b == 0 && c == 1))
                return +1; // even
            else
                return -1; // odd
        }

        static Matrix3x3 BuildMatrixFromPermAndSigns(int[] perm, int sx, int sy, int sz)
        {
            // Column j is the image of basis axis j.
            // Column j has a single non-zero entry at row perm[j] with sign s[j].
            Matrix3x3 m;
            m.M00 = m.M01 = m.M02 = 0.0;
            m.M10 = m.M11 = m.M12 = 0.0;
            m.M20 = m.M21 = m.M22 = 0.0;

            int[] s = { sx, sy, sz };

            for (int col = 0; col < 3; col++)
            {
                int row = perm[col];
                double val = s[col];

                if (row == 0)
                {
                    if (col == 0) m.M00 = val;
                    else if (col == 1) m.M01 = val;
                    else m.M02 = val;
                }
                else if (row == 1)
                {
                    if (col == 0) m.M10 = val;
                    else if (col == 1) m.M11 = val;
                    else m.M12 = val;
                }
                else // row == 2
                {
                    if (col == 0) m.M20 = val;
                    else if (col == 1) m.M21 = val;
                    else m.M22 = val;
                }
            }

            return m;
        }
    }
