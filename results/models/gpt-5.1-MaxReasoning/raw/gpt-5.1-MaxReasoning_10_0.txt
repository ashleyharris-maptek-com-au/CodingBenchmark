#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>
#include <cstdint>

using namespace std;

struct RectInput {
    int idx;
    int w;
    int h;
};

struct Placement {
    int board;
    int x;
    int y;
    int rotated; // 0 = not rotated, 1 = rotated 90 degrees
};

/**************************** Small-N solver: Guillotine packing ****************************/

struct FreeRect {
    int x, y, w, h;
};

struct BoardGuillotine {
    vector<FreeRect> freeRects;
};

static void splitFreeRectGuillotine(vector<FreeRect> &freeRects,
                                    const FreeRect &free,
                                    const FreeRect &placed)
{
    // Assumes placed.x == free.x and placed.y == free.y and placed is fully inside free.
    int wRem = free.w - placed.w;
    int hRem = free.h - placed.h;

    // Choose split direction: shorter leftover side first.
    bool splitVertical = (wRem <= hRem);

    if (splitVertical) {
        // Right column
        if (wRem > 0) {
            FreeRect right;
            right.x = free.x + placed.w;
            right.y = free.y;
            right.w = wRem;
            right.h = free.h;
            freeRects.push_back(right);
        }
        // Bottom strip under placed rectangle
        if (hRem > 0) {
            FreeRect bottom;
            bottom.x = free.x;
            bottom.y = free.y + placed.h;
            bottom.w = placed.w;
            bottom.h = hRem;
            freeRects.push_back(bottom);
        }
    } else {
        // Bottom strip
        if (hRem > 0) {
            FreeRect bottom;
            bottom.x = free.x;
            bottom.y = free.y + placed.h;
            bottom.w = free.w;
            bottom.h = hRem;
            freeRects.push_back(bottom);
        }
        // Right strip next to placed rectangle
        if (wRem > 0) {
            FreeRect right;
            right.x = free.x + placed.w;
            right.y = free.y;
            right.w = wRem;
            right.h = placed.h;
            freeRects.push_back(right);
        }
    }
}

// Choose orientation for placing rect into an empty bin.
static void chooseOrientationNewBin(int rw, int rh, int binW, int binH,
                                    bool &rotate, int &outW, int &outH)
{
    bool can0 = (rw <= binW && rh <= binH);
    bool can1 = (rh <= binW && rw <= binH);

    if (!can0 && !can1) {
        // Should not happen for valid input; fallback: clamp
        rotate = false;
        outW = min(rw, binW);
        outH = min(rh, binH);
        return;
    }

    if (can0 && can1) {
        // Choose orientation minimizing the smaller leftover side.
        int lw0 = binW - rw;
        int lh0 = binH - rh;
        int shortRem0 = min(lw0, lh0);

        int lw1 = binW - rh;
        int lh1 = binH - rw;
        int shortRem1 = min(lw1, lh1);

        if (shortRem0 <= shortRem1) {
            rotate = false;
            outW = rw;
            outH = rh;
        } else {
            rotate = true;
            outW = rh;
            outH = rw;
        }
    } else if (can0) {
        rotate = false;
        outW = rw;
        outH = rh;
    } else {
        rotate = true;
        outW = rh;
        outH = rw;
    }
}

static void solve_guillotine_small(const vector<RectInput> &rectsIn,
                                   int binW, int binH,
                                   vector<Placement> &placements,
                                   int &numBoards)
{
    int N = (int)rectsIn.size();
    placements.assign(N, Placement{0, 0, 0, 0});

    // Sort rectangles by decreasing max side, then area (descending).
    vector<RectInput> rects = rectsIn;
    sort(rects.begin(), rects.end(), [](const RectInput &a, const RectInput &b) {
        int ma = max(a.w, a.h);
        int mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        int64_t aa = (int64_t)a.w * a.h;
        int64_t ab = (int64_t)b.w * b.h;
        return aa > ab;
    });

    vector<BoardGuillotine> boards;
    boards.reserve(max(1, N / 10));

    const int INF = numeric_limits<int>::max();

    for (const auto &r : rects) {
        int rw0 = r.w;
        int rh0 = r.h;

        int bestBoard = -1;
        int bestFreeIdx = -1;
        bool bestRot = false;
        int bestAreaFit = INF;
        int bestShortSide = INF;

        // Try to place into existing boards.
        for (int b = 0; b < (int)boards.size(); ++b) {
            auto &freeRects = boards[b].freeRects;
            for (int i = 0; i < (int)freeRects.size(); ++i) {
                const FreeRect &f = freeRects[i];

                // Not rotated
                if (rw0 <= f.w && rh0 <= f.h) {
                    int areaFit = f.w * f.h - rw0 * rh0;
                    int shortSide = min(f.w - rw0, f.h - rh0);
                    if (areaFit < bestAreaFit ||
                        (areaFit == bestAreaFit && shortSide < bestShortSide)) {
                        bestAreaFit = areaFit;
                        bestShortSide = shortSide;
                        bestBoard = b;
                        bestFreeIdx = i;
                        bestRot = false;
                    }
                }

                // Rotated
                if (rh0 <= f.w && rw0 <= f.h) {
                    int rw = rh0;
                    int rh = rw0;
                    int areaFit = f.w * f.h - rw * rh;
                    int shortSide = min(f.w - rw, f.h - rh);
                    if (areaFit < bestAreaFit ||
                        (areaFit == bestAreaFit && shortSide < bestShortSide)) {
                        bestAreaFit = areaFit;
                        bestShortSide = shortSide;
                        bestBoard = b;
                        bestFreeIdx = i;
                        bestRot = true;
                    }
                }
            }
        }

        if (bestBoard == -1) {
            // Need a new board.
            BoardGuillotine nb;
            nb.freeRects.clear();
            nb.freeRects.push_back(FreeRect{0, 0, binW, binH});
            boards.push_back(nb);
            int b = (int)boards.size() - 1;

            bool rot;
            int pw, ph;
            chooseOrientationNewBin(rw0, rh0, binW, binH, rot, pw, ph);

            // Place at (0,0)
            placements[r.idx] = Placement{b, 0, 0, rot ? 1 : 0};

            FreeRect base = boards[b].freeRects[0];
            boards[b].freeRects.clear();
            FreeRect placed{base.x, base.y, pw, ph};
            splitFreeRectGuillotine(boards[b].freeRects, base, placed);
        } else {
            // Place into existing board at chosen free rect.
            BoardGuillotine &B = boards[bestBoard];
            FreeRect f = B.freeRects[bestFreeIdx];
            // Remove used free rect (swap with back).
            B.freeRects[bestFreeIdx] = B.freeRects.back();
            B.freeRects.pop_back();

            int pw = bestRot ? rh0 : rw0;
            int ph = bestRot ? rw0 : rh0;

            placements[r.idx] = Placement{bestBoard, f.x, f.y, bestRot ? 1 : 0};

            FreeRect placed{f.x, f.y, pw, ph};
            splitFreeRectGuillotine(B.freeRects, f, placed);
        }
    }

    numBoards = (int)boards.size();
}

/**************************** Large-N solver: Shelf-based packing ****************************/

struct Shelf {
    int y;
    int height;
    int used_width;
    bool active;
};

struct BoardShelf {
    vector<Shelf> shelves;
    int used_height;
};

static void solve_shelf_large(const vector<RectInput> &rectsIn,
                              int binW, int binH,
                              vector<Placement> &placements,
                              int &numBoards)
{
    int N = (int)rectsIn.size();
    placements.assign(N, Placement{0, 0, 0, 0});

    // Sort rectangles by decreasing max side, then area.
    vector<RectInput> rects = rectsIn;
    sort(rects.begin(), rects.end(), [](const RectInput &a, const RectInput &b) {
        int ma = max(a.w, a.h);
        int mb = max(b.w, b.h);
        if (ma != mb) return ma > mb;
        int64_t aa = (int64_t)a.w * a.h;
        int64_t ab = (int64_t)b.w * b.h;
        return aa > ab;
    });

    // Precompute future minimum possible side (min(w,h)) from each position to end.
    vector<int> futureMinSide(N + 1);
    const int INF = numeric_limits<int>::max();
    futureMinSide[N] = INF;
    for (int i = N - 1; i >= 0; --i) {
        int side = min(rects[i].w, rects[i].h);
        futureMinSide[i] = std::min(futureMinSide[i + 1], side);
    }

    vector<BoardShelf> boards;
    boards.reserve(max(1, N / 10));

    // Each entry: (boardIndex, shelfIndex)
    vector<pair<int, int>> shelfRefs;
    shelfRefs.reserve(N);

    const int MAX_SHELF_SCAN = 200;
    const int MAX_BOARD_SCAN = 200;

    for (int cur = 0; cur < N; ++cur) {
        const RectInput &r = rects[cur];
        int w0 = r.w;
        int h0 = r.h;

        bool placed = false;

        // 1) Try existing shelves.
        int bestRefIdx = -1;
        bool bestRot = false;
        int bestWidthWaste = INF;
        int bestHeightWaste = INF;

        int neededMinSide = futureMinSide[cur];

        size_t activeScanned = 0;
        for (size_t refIdx = 0; refIdx < shelfRefs.size() && activeScanned < (size_t)MAX_SHELF_SCAN; ++refIdx) {
            int bIdx = shelfRefs[refIdx].first;
            int sIdx = shelfRefs[refIdx].second;

            BoardShelf &B = boards[bIdx];
            Shelf &S = B.shelves[sIdx];

            if (!S.active) continue;

            int remWidth = binW - S.used_width;
            if (remWidth <= 0) {
                S.active = false;
                continue;
            }
            if (remWidth < neededMinSide) {
                S.active = false;
                continue;
            }

            bool can0 = (w0 <= remWidth && h0 <= S.height);
            bool can1 = (h0 <= remWidth && w0 <= S.height);
            if (!can0 && !can1) continue;

            ++activeScanned;

            bool curRot;
            int dw, dh;
            if (can0 && can1) {
                int slackH0 = S.height - h0;
                int slackH1 = S.height - w0;
                int slackW0 = remWidth - w0;
                int slackW1 = remWidth - h0;
                if (slackH0 < slackH1 || (slackH0 == slackH1 && slackW0 <= slackW1)) {
                    curRot = false;
                    dw = w0;
                    dh = h0;
                } else {
                    curRot = true;
                    dw = h0;
                    dh = w0;
                }
            } else if (can0) {
                curRot = false;
                dw = w0;
                dh = h0;
            } else { // can1
                curRot = true;
                dw = h0;
                dh = w0;
            }

            int newUsedWidth = S.used_width + dw;
            int wWaste = binW - newUsedWidth;
            int hWaste = S.height - dh;

            if (!placed || wWaste < bestWidthWaste ||
                (wWaste == bestWidthWaste && hWaste < bestHeightWaste)) {
                placed = true;
                bestRefIdx = (int)refIdx;
                bestRot = curRot;
                bestWidthWaste = wWaste;
                bestHeightWaste = hWaste;
            }
        }

        if (placed) {
            int bIdx = shelfRefs[bestRefIdx].first;
            int sIdx = shelfRefs[bestRefIdx].second;

            BoardShelf &B = boards[bIdx];
            Shelf &S = B.shelves[sIdx];

            int dw = bestRot ? h0 : w0;
            int dh = bestRot ? w0 : h0;

            int x = S.used_width;
            int y = S.y;

            S.used_width += dw;

            int remWidth = binW - S.used_width;
            int nextMinSide = (cur + 1 <= N) ? futureMinSide[cur + 1] : INF;
            if (remWidth <= 0 || remWidth < nextMinSide)
                S.active = false;

            placements[r.idx] = Placement{bIdx, x, y, bestRot ? 1 : 0};
            continue;
        }

        // 2) Need a new shelf. Try existing boards first.
        int bestBoard = -1;
        bool bestBoardRot = false;
        int bestVerticalWaste = INF;

        size_t boardsToScan = min((size_t)MAX_BOARD_SCAN, boards.size());
        for (size_t bIdx = 0; bIdx < boardsToScan; ++bIdx) {
            BoardShelf &B = boards[bIdx];
            int remHeight = binH - B.used_height;
            if (remHeight <= 0) continue;

            bool can0 = (w0 <= binW && h0 <= remHeight);
            bool can1 = (h0 <= binW && w0 <= remHeight);
            if (!can0 && !can1) continue;

            bool curRot;
            int dh;
            if (can0 && can1) {
                // choose orientation with smaller shelf height
                if (h0 <= w0) {
                    curRot = false;
                    dh = h0;
                } else {
                    curRot = true;
                    dh = w0;
                }
            } else if (can0) {
                curRot = false;
                dh = h0;
            } else { // can1
                curRot = true;
                dh = w0;
            }

            int vWaste = remHeight - dh;
            if (vWaste < bestVerticalWaste) {
                bestVerticalWaste = vWaste;
                bestBoard = (int)bIdx;
                bestBoardRot = curRot;
            }
        }

        if (bestBoard == -1) {
            // 3) Create a new board.
            BoardShelf nb;
            nb.used_height = 0;
            boards.push_back(nb);
            bestBoard = (int)boards.size() - 1;
            BoardShelf &B = boards[bestBoard];

            int remHeight = binH - B.used_height;

            bool can0 = (w0 <= binW && h0 <= remHeight);
            bool can1 = (h0 <= binW && w0 <= remHeight);

            bool rot;
            int dw, dh;
            if (!can0 && !can1) {
                // Should not happen for valid input; clamp.
                rot = false;
                dw = min(w0, binW);
                dh = min(h0, binH);
            } else if (can0 && !can1) {
                rot = false;
                dw = w0;
                dh = h0;
            } else if (!can0 && can1) {
                rot = true;
                dw = h0;
                dh = w0;
            } else {
                if (h0 <= w0) {
                    rot = false;
                    dw = w0;
                    dh = h0;
                } else {
                    rot = true;
                    dw = h0;
                    dh = w0;
                }
            }

            Shelf s;
            s.y = B.used_height;
            s.height = dh;
            s.used_width = 0;
            s.active = true;
            B.shelves.push_back(s);
            int sIdx = (int)B.shelves.size() - 1;
            B.used_height += dh;
            shelfRefs.emplace_back(bestBoard, sIdx);

            int x = 0;
            int y = s.y;

            B.shelves[sIdx].used_width += dw;
            int remWidth = binW - B.shelves[sIdx].used_width;
            int nextMinSide = (cur + 1 <= N) ? futureMinSide[cur + 1] : INF;
            if (remWidth <= 0 || remWidth < nextMinSide)
                B.shelves[sIdx].active = false;

            placements[r.idx] = Placement{bestBoard, x, y, rot ? 1 : 0};
        } else {
            // Create a new shelf in existing bestBoard.
            BoardShelf &B = boards[bestBoard];
            int remHeight = binH - B.used_height;

            bool can0 = (w0 <= binW && h0 <= remHeight);
            bool can1 = (h0 <= binW && w0 <= remHeight);

            bool rot;
            int dw, dh;
            if (!can0 && !can1) {
                // Should not happen; fallback (clamp).
                rot = false;
                dw = min(w0, binW);
                dh = min(h0, remHeight);
            } else if (can0 && !can1) {
                rot = false;
                dw = w0;
                dh = h0;
            } else if (!can0 && can1) {
                rot = true;
                dw = h0;
                dh = w0;
            } else {
                // Use previously chosen bestBoardRot.
                rot = bestBoardRot;
                if (!rot) {
                    dw = w0;
                    dh = h0;
                } else {
                    dw = h0;
                    dh = w0;
                }
            }

            Shelf s;
            s.y = B.used_height;
            s.height = dh;
            s.used_width = 0;
            s.active = true;
            B.shelves.push_back(s);
            int sIdx = (int)B.shelves.size() - 1;
            B.used_height += dh;
            shelfRefs.emplace_back(bestBoard, sIdx);

            int x = 0;
            int y = s.y;

            B.shelves[sIdx].used_width += dw;
            int remWidth = binW - B.shelves[sIdx].used_width;
            int nextMinSide = (cur + 1 <= N) ? futureMinSide[cur + 1] : INF;
            if (remWidth <= 0 || remWidth < nextMinSide)
                B.shelves[sIdx].active = false;

            placements[r.idx] = Placement{bestBoard, x, y, rot ? 1 : 0};
        }
    }

    numBoards = (int)boards.size();
}

/**************************** Main ****************************/

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    int binW, binH;
    if (!(cin >> N >> binW >> binH)) {
        return 0;
    }

    vector<RectInput> rects(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects[i] = RectInput{i, w, h};
    }

    vector<Placement> placements;
    int numBoards = 0;

    // Adaptive choice: use higher-quality guillotine method for small N,
    // faster shelf-based heuristic for larger N.
    const int SMALL_N_THRESHOLD = 2000;
    if (N <= SMALL_N_THRESHOLD) {
        solve_guillotine_small(rects, binW, binH, placements, numBoards);
    } else {
        solve_shelf_large(rects, binW, binH, placements, numBoards);
    }

    // Output: number of boards, then one line per rectangle in original order.
    cout << numBoards << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement &p = placements[i];
        cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
