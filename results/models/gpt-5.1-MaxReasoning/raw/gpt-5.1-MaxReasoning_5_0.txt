```rust
use std::cmp::Ordering;
use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Debug)]
struct Solver {
    graph: Vec<Vec<usize>>,
    coord: Vec<(usize, usize)>,
    start_id: usize,
    total: usize,
    require_cycle: bool,
    visited: Vec<bool>,
    path: Vec<usize>,
    bfs_mark: Vec<u32>,
    bfs_mark_counter: u32,
    queue: VecDeque<usize>,
}

impl Solver {
    #[inline]
    fn count_unvisited_neighbors(&self, v: usize) -> u8 {
        let mut c = 0u8;
        for &n in &self.graph[v] {
            if !self.visited[n] {
                c += 1;
            }
        }
        c
    }

    /// Check that every unvisited vertex is reachable from `endpoint`
    /// using only unvisited vertices (plus the endpoint itself).
    #[inline]
    fn remaining_connected(&mut self, endpoint: usize, visited_count: usize) -> bool {
        let remaining = self.total - visited_count;
        if remaining == 0 {
            return true;
        }

        self.bfs_mark_counter = self.bfs_mark_counter.wrapping_add(1);
        let cur_mark = self.bfs_mark_counter;

        self.queue.clear();
        let mut seen = 0usize;

        // Start BFS from all unvisited neighbors of the endpoint.
        for &u in &self.graph[endpoint] {
            if !self.visited[u] && self.bfs_mark[u] != cur_mark {
                self.bfs_mark[u] = cur_mark;
                self.queue.push_back(u);
                seen += 1;
            }
        }

        while let Some(v) = self.queue.pop_front() {
            for &n in &self.graph[v] {
                if !self.visited[n] && self.bfs_mark[n] != cur_mark {
                    self.bfs_mark[n] = cur_mark;
                    self.queue.push_back(n);
                    seen += 1;
                }
            }
        }

        seen == remaining
    }

    fn dfs(&mut self, current: usize, visited_count: usize) -> bool {
        if visited_count == self.total {
            if self.require_cycle {
                // Last vertex must be adjacent to start
                for &n in &self.graph[current] {
                    if n == self.start_id {
                        return true;
                    }
                }
                return false;
            } else {
                return true;
            }
        }

        // Collect candidate neighbors (unvisited) and order them by Warnsdorff-style heuristic
        let mut cand = [usize::MAX; 4];
        let mut score = [0u8; 4];
        let mut len = 0usize;

        for &u in &self.graph[current] {
            if !self.visited[u] {
                cand[len] = u;
                score[len] = self.count_unvisited_neighbors(u);
                len += 1;
            }
        }

        if len == 0 {
            return false;
        }

        // Insertion sort by score (small len <= 4)
        for i in 1..len {
            let key_v = cand[i];
            let key_s = score[i];
            let mut j = i;
            while j > 0 && score[j - 1] > key_s {
                cand[j] = cand[j - 1];
                score[j] = score[j - 1];
                j -= 1;
            }
            cand[j] = key_v;
            score[j] = key_s;
        }

        for idx in 0..len {
            let nxt = cand[idx];

            // Move to nxt
            self.visited[nxt] = true;
            self.path.push(nxt);

            // Prune if remaining unvisited vertices are disconnected from nxt
            if self.remaining_connected(nxt, visited_count + 1) {
                if self.dfs(nxt, visited_count + 1) {
                    return true;
                }
            }

            // Backtrack
            self.visited[nxt] = false;
            self.path.pop();
        }

        false
    }
}

fn main() {
    // -------- Read input --------
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();

    let w: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let h: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let obstacle_count: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let require_cycle_flag: i32 = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let require_cycle = require_cycle_flag != 0;

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; w * h];
    for _ in 0..obstacle_count {
        let x: usize = match it.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => 0,
        };
        let y: usize = match it.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => 0,
        };
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    // Start must be (0,0)
    if blocked[0] {
        println!("0");
        return;
    }

    let total_cells = w * h;
    let open_cells: usize = blocked.iter().filter(|&&b| !b).count();

    if open_cells == 0 {
        println!("0");
        return;
    }

    // Fast path for completely open grids when cycle is not required:
    // simple row-wise snake Hamiltonian path (always exists).
    if obstacle_count == 0 && !require_cycle {
        let mut out = String::new();
        out.push_str(&(total_cells.to_string()));
        out.push('\n');
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    out.push_str(&format!("{} {}\n", x, y));
                }
            } else {
                for x in (0..w).rev() {
                    out.push_str(&format!("{} {}\n", x, y));
                }
            }
        }
        print!("{}", out);
        return;
    }

    // -------- Build compressed vertex indexing --------
    const INVALID: usize = usize::MAX;
    let mut cell_id = vec![INVALID; w * h];
    let mut coord = Vec::<(usize, usize)>::with_capacity(open_cells);

    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if !blocked[idx] {
                let id = coord.len();
                coord.push((x, y));
                cell_id[idx] = id;
            }
        }
    }

    let total_vertices = coord.len();
    if total_vertices == 0 {
        println!("0");
        return;
    }

    let start_id = cell_id[0];
    if start_id == INVALID {
        // (0,0) is blocked or not in the reachable component
        println!("0");
        return;
    }

    // -------- Build adjacency (grid-graph) --------
    let mut graph = vec![Vec::<usize>::new(); total_vertices];

    let dirs = [(1isize, 0isize), (-1, 0), (0, 1), (0, -1)];
    for y in 0..h {
        for x in 0..w {
            let idx = y * w + x;
            if blocked[idx] {
                continue;
            }
            let id = cell_id[idx];
            for (dx, dy) in dirs.iter() {
                let nx = x as isize + dx;
                let ny = y as isize + dy;
                if nx >= 0 && nx < w as isize && ny >= 0 && ny < h as isize {
                    let nidx = ny as usize * w + nx as usize;
                    if !blocked[nidx] {
                        let nid = cell_id[nidx];
                        graph[id].push(nid);
                    }
                }
            }
        }
    }

    // -------- Connectivity check (must reach all open cells) --------
    let mut seen = vec![false; total_vertices];
    let mut q = VecDeque::new();
    seen[start_id] = true;
    q.push_back(start_id);
    let mut reachable = 1usize;

    while let Some(v) = q.pop_front() {
        for &n in &graph[v] {
            if !seen[n] {
                seen[n] = true;
                reachable += 1;
                q.push_back(n);
            }
        }
    }

    if reachable != total_vertices {
        println!("0");
        return;
    }

    // -------- Handle trivial single-vertex case --------
    if total_vertices == 1 {
        if require_cycle {
            // Cannot form a cycle on a single vertex with degree 0
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    // -------- Parity / bipartite checks --------
    // Grid graph is bipartite by (x + y) % 2
    let mut color = vec![0u8; total_vertices];
    let mut cnt = [0usize; 2];

    for (id, &(x, y)) in coord.iter().enumerate() {
        let c = ((x + y) & 1) as u8;
        color[id] = c;
        cnt[c as usize] += 1;
    }

    let total = total_vertices;
    let white = cnt[0];
    let black = cnt[1];
    let start_color = color[start_id];

    if require_cycle {
        // Hamiltonian cycle in bipartite graph => equal color classes and even total
        if total % 2 != 0 || white != black {
            println!("0");
            return;
        }
    } else {
        // Hamiltonian path parity conditions:
        // For a path covering all vertices in a bipartite graph:
        //   * if total is even, color counts must be equal
        //   * if total is odd, counts differ by 1 and both endpoints are in majority color
        if total % 2 == 0 {
            if white != black {
                println!("0");
                return;
            }
        } else {
            let diff = if white > black { white - black } else { black - white };
            if diff != 1 {
                println!("0");
                return;
            }
            let majority_color: u8 = if white > black { 0 } else { 1 };
            if start_color != majority_color {
                println!("0");
                return;
            }
        }
    }

    // -------- Degree / leaf checks (necessary conditions) --------
    let mut deg = vec![0usize; total_vertices];
    for (i, adj) in graph.iter().enumerate() {
        deg[i] = adj.len();
    }

    // No isolated vertices (except the trivial N==1 already handled)
    for d in &deg {
        if *d == 0 {
            println!("0");
            return;
        }
    }

    if require_cycle {
        // Every vertex must have degree >= 2 for a Hamiltonian cycle
        if deg.iter().any(|&d| d < 2) {
            println!("0");
            return;
        }
    } else {
        // For a Hamiltonian path:
        // number of leaves (degree==1) must be 0,1, or 2.
        // If exactly 2, start must be one endpoint.
        let leaves: Vec<usize> = (0..total_vertices)
            .filter(|&i| deg[i] == 1)
            .collect();
        if leaves.len() > 2 {
            println!("0");
            return;
        }
        if leaves.len() == 2 && !leaves.contains(&start_id) {
            println!("0");
            return;
        }
    }

    // -------- Set up solver --------
    let mut solver = Solver {
        graph,
        coord,
        start_id,
        total: total_vertices,
        require_cycle,
        visited: vec![false; total_vertices],
        path: Vec::with_capacity(total_vertices),
        bfs_mark: vec![0u32; total_vertices],
        bfs_mark_counter: 0,
        queue: VecDeque::new(),
    };

    solver.visited[start_id] = true;
    solver.path.push(start_id);

    let success = solver.dfs(start_id, 1);

    if !success {
        println!("0");
        return;
    }

    // -------- Output solution --------
    let mut out = String::new();
    out.push_str(&format!("{}\n", solver.path.len()));
    for &id in &solver.path {
        let (x, y) = solver.coord[id];
        out.push_str(&format!("{} {}\n", x, y));
    }
    print!("{}", out);
}
```
