import sys

def solve():
    lines = sys.stdin.read().strip().split('\n')
    U, S = map(int, lines[0].split())

    sets = []
    for i in range(1, S + 1):
        if i < len(lines) and lines[i].strip():
            sets.append(set(map(int, lines[i].split())))
        else:
            sets.append(set())

    universe = set(range(U))

    def backtrack(covered, idx, chosen):
        if covered == universe:
            return chosen[:]
        if idx == S:
            return None

        if sets[idx] and sets[idx].isdisjoint(covered):
            new_covered = covered | sets[idx]
            result = backtrack(new_covered, idx + 1, chosen + [idx])
            if result is not None:
                return result

        return backtrack(covered, idx + 1, chosen)

    result = backtrack(set(), 0, [])

    if result is not None:
        print("SOLUTION")
        print(' '.join(map(str, result)))
    else:
        print("NO SOLUTION")

solve()
