#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;
vector<int> best_sol;
long long best_val;

void solve(int idx, vector<int>& cur, long long val) {
    if (idx == n) {
        if (val > best_val) {
            best_val = val;
            best_sol = cur;
        }
        return;
    }
    solve(idx + 1, cur, val);
    bool ok = true;
    for (int v : adj[idx]) {
        if (v < idx && cur[v] == 1) {
            ok = false;
            break;
        }
    }
    if (ok) {
        cur[idx] = 1;
        solve(idx + 1, cur, val + c[idx]);
        cur[idx] = 0;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    c.resize(n);
    adj.resize(n);
    for (int i = 0; i < n; i++) cin >> c[i];
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    best_val = LLONG_MIN;
    vector<int> cur(n, 0);
    solve(0, cur, 0);

    cout << best_val << "\n";
    for (int i = 0; i < n; i++) {
        cout << best_sol[i] << (i + 1 < n ? " " : "\n");
    }
    return 0;
}
