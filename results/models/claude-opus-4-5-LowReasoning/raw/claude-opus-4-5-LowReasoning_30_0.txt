import sys

def solve_sat():
    data = sys.stdin.read().split()
    idx = 0
    n = int(data[idx]); idx += 1
    m = int(data[idx]); idx += 1

    clauses = []
    for _ in range(m):
        a, b, c = int(data[idx]), int(data[idx+1]), int(data[idx+2])
        idx += 3
        clauses.append((a, b, c))

    try:
        from pysat.solvers import Glucose3
        solver = Glucose3()
        for clause in clauses:
            solver.add_clause(list(clause))
        if solver.solve():
            model = solver.get_model()
            assignment = [0] * (n + 1)
            for lit in model:
                if lit > 0 and lit <= n:
                    assignment[lit] = 1
            print("SAT")
            print(" ".join(map(str, assignment[1:])))
        solver.delete()
    except ImportError:
        import random
        assignment = [random.choice([True, False]) for _ in range(n + 1)]

        def evaluate_clause(c):
            for lit in c:
                val = assignment[abs(lit)]
                if (lit > 0 and val) or (lit < 0 and not val):
                    return True
            return False

        max_flips = 10000000
        for _ in range(max_flips):
            unsat = [c for c in clauses if not evaluate_clause(c)]
            if not unsat:
                break
            clause = random.choice(unsat)
            var = abs(random.choice(clause))
            assignment[var] = not assignment[var]

        print("SAT")
        print(" ".join('1' if assignment[i] else '0' for i in range(1, n + 1)))

if __name__ == "__main__":
    solve_sat()
