#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <cstdint>

using namespace std;

struct City {
    float x, y;
    int idx;
};

inline float dist2(const City& a, const City& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return dx*dx + dy*dy;
}

uint64_t hilbert_d2xy(int n, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (int s = n/2; s > 0; s /= 2) {
        int rx = (x & s) > 0;
        int ry = (y & s) > 0;
        d += (uint64_t)s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s-1 - x; y = s-1 - y; }
            swap(x, y);
        }
    }
    return d;
}

void solveHilbert(vector<City>& cities, vector<int>& tour) {
    int n = cities.size();
    float minx = cities[0].x, maxx = cities[0].x;
    float miny = cities[0].y, maxy = cities[0].y;
    for (int i = 1; i < n; i++) {
        minx = min(minx, cities[i].x); maxx = max(maxx, cities[i].x);
        miny = min(miny, cities[i].y); maxy = max(maxy, cities[i].y);
    }
    float rangex = maxx - minx, rangey = maxy - miny;
    if (rangex < 1e-9f) rangex = 1.0f;
    if (rangey < 1e-9f) rangey = 1.0f;

    const int H = 1 << 20;
    vector<pair<uint64_t, int>> hvals(n);
    for (int i = 0; i < n; i++) {
        uint64_t hx = (uint64_t)((cities[i].x - minx) / rangex * (H - 1));
        uint64_t hy = (uint64_t)((cities[i].y - miny) / rangey * (H - 1));
        hvals[i] = {hilbert_d2xy(H, hx, hy), cities[i].idx};
    }
    sort(hvals.begin(), hvals.end());

    tour.resize(n);
    int start = 0;
    for (int i = 0; i < n; i++) if (hvals[i].second == 0) { start = i; break; }
    for (int i = 0; i < n; i++) tour[i] = hvals[(start + i) % n].second;
}

void solveNN(vector<City>& cities, vector<int>& tour) {
    int n = cities.size();
    vector<bool> visited(n, false);
    tour.resize(n);
    tour[0] = 0;
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour[i-1];
        float best = numeric_limits<float>::max();
        int bestj = -1;
        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist2(cities[last], cities[j]);
                if (d < best) { best = d; bestj = j; }
            }
        }
        tour[i] = bestj;
        visited[bestj] = true;
    }
}

void solveNNGrid(vector<City>& cities, vector<int>& tour) {
    int n = cities.size();
    float minx = cities[0].x, maxx = cities[0].x;
    float miny = cities[0].y, maxy = cities[0].y;
    for (int i = 1; i < n; i++) {
        minx = min(minx, cities[i].x); maxx = max(maxx, cities[i].x);
        miny = min(miny, cities[i].y); maxy = max(maxy, cities[i].y);
    }
    float rangex = maxx - minx + 1e-9f, rangey = maxy - miny + 1e-9f;

    int gsize = max(1, (int)sqrt(n / 10.0));
    vector<vector<int>> grid(gsize * gsize);
    auto getCell = [&](float x, float y) {
        int cx = min(gsize-1, (int)((x - minx) / rangex * gsize));
        int cy = min(gsize-1, (int)((y - miny) / rangey * gsize));
        return cy * gsize + cx;
    };
    for (int i = 0; i < n; i++) grid[getCell(cities[i].x, cities[i].y)].push_back(i);

    vector<bool> visited(n, false);
    tour.resize(n);
    tour[0] = 0;
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour[i-1];
        float lx = cities[last].x, ly = cities[last].y;
        int cx = min(gsize-1, (int)((lx - minx) / rangex * gsize));
        int cy = min(gsize-1, (int)((ly - miny) / rangey * gsize));

        float best = numeric_limits<float>::max();
        int bestj = -1;
        for (int r = 0; r <= gsize && bestj == -1; r++) {
            for (int dy = -r; dy <= r; dy++) {
                for (int dx = -r; dx <= r; dx++) {
                    if (abs(dx) != r && abs(dy) != r) continue;
                    int nx = cx + dx, ny = cy + dy;
                    if (nx < 0 || nx >= gsize || ny < 0 || ny >= gsize) continue;
                    for (int j : grid[ny * gsize + nx]) {
                        if (!visited[j]) {
                            float d = dist2(cities[last], cities[j]);
                            if (d < best) { best = d; bestj = j; }
                        }
                    }
                }
            }
        }
        tour[i] = bestj;
        visited[bestj] = true;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
        cities[i].idx = i;
    }

    vector<int> tour;

    if (n <= 1) {
        tour = {0};
    } else if (n <= 2000) {
        solveNN(cities, tour);
    } else if (n <= 50000) {
        solveNNGrid(cities, tour);
    } else {
        solveHilbert(cities, tour);
    }

    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';

    return 0;
}
