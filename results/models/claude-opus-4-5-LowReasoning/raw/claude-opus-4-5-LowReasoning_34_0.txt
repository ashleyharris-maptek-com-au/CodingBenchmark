import sys
import math

def dist(a, b):
    return math.sqrt((a[0]-b[0])**2 + (a[1]-b[1])**2)

def main():
    data = sys.stdin.read().split()
    idx = 0
    N, K, CAP = int(data[idx]), int(data[idx+1]), int(data[idx+2])
    idx += 3
    depot = (float(data[idx]), float(data[idx+1]))
    idx += 2
    customers = []
    for i in range(N):
        x, y, d = float(data[idx]), float(data[idx+1]), int(data[idx+2])
        customers.append((x, y, d, i))
        idx += 3

    # Greedy nearest neighbor with capacity constraint
    visited = [False] * N
    routes = []

    for _ in range(K):
        if all(visited):
            break
        route = []
        load = 0
        pos = depot
        while True:
            best_i, best_d = -1, float('inf')
            for i, c in enumerate(customers):
                if not visited[i] and load + c[2] <= CAP:
                    d = dist(pos, (c[0], c[1]))
                    if d < best_d:
                        best_d, best_i = d, i
            if best_i == -1:
                break
            visited[best_i] = True
            route.append(best_i)
            load += customers[best_i][2]
            pos = (customers[best_i][0], customers[best_i][1])
        if route:
            routes.append(route)

    # Handle any remaining (shouldn't happen if K is sufficient)
    for i in range(N):
        if not visited[i]:
            routes.append([i])

    # 2-opt improvement for each route
    def two_opt(route):
        if len(route) < 2:
            return route
        improved = True
        while improved:
            improved = False
            for i in range(len(route)-1):
                for j in range(i+2, len(route)):
                    c1 = customers[route[i]]
                    c2 = customers[route[i+1]] if i+1 < len(route) else None
                    c3 = customers[route[j]]
                    c4_idx = j+1
                    p1 = (c1[0], c1[1])
                    p2 = (c2[0], c2[1]) if c2 else depot
                    p3 = (c3[0], c3[1])
                    prev = depot if i == 0 else (customers[route[i-1]][0], customers[route[i-1]][1])
                    nxt = depot if c4_idx >= len(route) else (customers[route[c4_idx]][0], customers[route[c4_idx]][1])

                    old = dist(prev, p1) + dist(p3, nxt)
                    new = dist(prev, p3) + dist(p1, nxt)
                    if new < old - 1e-9:
                        route[i:j+1] = route[i:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break
        return route

    routes = [two_opt(r) for r in routes]

    # Calculate total distance
    total = 0.0
    for route in routes:
        pos = depot
        for i in route:
            c = customers[i]
            total += dist(pos, (c[0], c[1]))
            pos = (c[0], c[1])
        total += dist(pos, depot)

    print(f"{total:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    main()
