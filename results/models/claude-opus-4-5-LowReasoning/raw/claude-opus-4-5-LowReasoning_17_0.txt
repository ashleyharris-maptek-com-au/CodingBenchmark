using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int W = int.Parse(firstLine[1]);
        int H = int.Parse(firstLine[2]);
        int D = int.Parse(firstLine[3]);

        var boxes = new (int w, int h, int d, int idx)[n];
        for (int i = 0; i < n; i++)
        {
            var parts = Console.ReadLine().Split();
            int bw = int.Parse(parts[0]);
            int bh = int.Parse(parts[1]);
            int bd = int.Parse(parts[2]);
            boxes[i] = (bw, bh, bd, i);
        }

        var sorted = boxes.OrderByDescending(b => b.w * b.h * b.d).ToArray();
        var result = new List<(int idx, int x, int y, int z)>();
        var spaces = new List<(int x, int y, int z, int w, int h, int d)> { (0, 0, 0, W, H, D) };

        foreach (var box in sorted)
        {
            int bestSpace = -1;
            int bestVol = int.MaxValue;
            int bw = 0, bh = 0, bd = 0;

            var dims = new[] { box.w, box.h, box.d };
            Array.Sort(dims);
            var orientations = new (int w, int h, int d)[]
            {
                (dims[0], dims[1], dims[2]), (dims[0], dims[2], dims[1]),
                (dims[1], dims[0], dims[2]), (dims[1], dims[2], dims[0]),
                (dims[2], dims[0], dims[1]), (dims[2], dims[1], dims[0])
            };

            for (int si = 0; si < spaces.Count; si++)
            {
                var s = spaces[si];
                foreach (var o in orientations)
                {
                    if (o.w <= s.w && o.h <= s.h && o.d <= s.d)
                    {
                        int vol = s.w * s.h * s.d;
                        if (vol < bestVol)
                        {
                            bestVol = vol;
                            bestSpace = si;
                            bw = o.w; bh = o.h; bd = o.d;
                        }
                    }
                }
            }

            if (bestSpace >= 0)
            {
                var s = spaces[bestSpace];
                result.Add((box.idx, s.x, s.y, s.z));
                spaces.RemoveAt(bestSpace);

                if (s.w - bw > 0) spaces.Add((s.x + bw, s.y, s.z, s.w - bw, bh, bd));
                if (s.h - bh > 0) spaces.Add((s.x, s.y + bh, s.z, s.w, s.h - bh, bd));
                if (s.d - bd > 0) spaces.Add((s.x, s.y, s.z + bd, s.w, s.h, s.d - bd));

                if (spaces.Count > 5000)
                    spaces = spaces.OrderBy(sp => sp.x + sp.y + sp.z).Take(3000).ToList();
            }
        }

        Console.WriteLine(result.Count);
        foreach (var r in result)
            Console.WriteLine($"{r.idx} {r.x} {r.y} {r.z}");
    }
}
