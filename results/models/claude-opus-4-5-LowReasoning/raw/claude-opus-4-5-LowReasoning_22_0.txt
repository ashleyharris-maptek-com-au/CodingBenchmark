using System;
using System.Collections.Generic;

class TetrisPlayer
{
    static int width, height;
    static int[,] board;

    static int[][,] pieces = new int[][,]
    {
        new int[,] {{1,1,1,1}},
        new int[,] {{1,1},{1,1}},
        new int[,] {{0,1,0},{1,1,1}},
        new int[,] {{0,1,1},{1,1,0}},
        new int[,] {{1,1,0},{0,1,1}},
        new int[,] {{1,0,0},{1,1,1}},
        new int[,] {{0,0,1},{1,1,1}}
    };

    static Dictionary<char, int> pieceIndex = new Dictionary<char, int>
    {
        {'I', 0}, {'O', 1}, {'T', 2}, {'S', 3}, {'Z', 4}, {'J', 5}, {'L', 6}
    };

    static int[,] Rotate(int[,] p)
    {
        int h = p.GetLength(0), w = p.GetLength(1);
        int[,] r = new int[w, h];
        for (int y = 0; y < h; y++)
            for (int x = 0; x < w; x++)
                r[x, h - 1 - y] = p[y, x];
        return r;
    }

    static bool CanPlace(int[,] p, int col, int row)
    {
        int ph = p.GetLength(0), pw = p.GetLength(1);
        if (col < 0 || col + pw > width || row < 0 || row + ph > height) return false;
        for (int y = 0; y < ph; y++)
            for (int x = 0; x < pw; x++)
                if (p[y, x] == 1 && board[row + y, col + x] == 1) return false;
        return true;
    }

    static int DropRow(int[,] p, int col)
    {
        int ph = p.GetLength(0);
        for (int row = height - ph; row >= 0; row--)
            if (CanPlace(p, col, row)) return row;
        return -1;
    }

    static void Place(int[,] p, int col, int row)
    {
        int ph = p.GetLength(0), pw = p.GetLength(1);
        for (int y = 0; y < ph; y++)
            for (int x = 0; x < pw; x++)
                if (p[y, x] == 1) board[row + y, col + x] = 1;
    }

    static int ClearLines()
    {
        int cleared = 0;
        for (int y = 0; y < height; y++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
                if (board[y, x] == 0) { full = false; break; }
            if (full)
            {
                cleared++;
                for (int yy = y; yy > 0; yy--)
                    for (int x = 0; x < width; x++)
                        board[yy, x] = board[yy - 1, x];
                for (int x = 0; x < width; x++) board[0, x] = 0;
            }
        }
        return cleared;
    }

    static double Evaluate()
    {
        int holes = 0, bumpiness = 0, aggregateHeight = 0;
        int[] colHeights = new int[width];

        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (board[y, x] == 1) { colHeights[x] = height - y; break; }
            }
            aggregateHeight += colHeights[x];
        }

        for (int x = 0; x < width; x++)
        {
            bool found = false;
            for (int y = 0; y < height; y++)
            {
                if (board[y, x] == 1) found = true;
                else if (found) holes++;
            }
        }

        for (int x = 0; x < width - 1; x++)
            bumpiness += Math.Abs(colHeights[x] - colHeights[x + 1]);

        return -0.51 * aggregateHeight - 0.36 * holes - 0.18 * bumpiness;
    }

    static (int rot, int col) FindBest(char pieceType)
    {
        int[,] p = pieces[pieceIndex[pieceType]];
        double bestScore = double.MinValue;
        int bestRot = 0, bestCol = 0;

        for (int rot = 0; rot < 4; rot++)
        {
            int pw = p.GetLength(1);
            for (int col = 0; col <= width - pw; col++)
            {
                int row = DropRow(p, col);
                if (row < 0) continue;

                int[,] backup = (int[,])board.Clone();
                Place(p, col, row);
                int lines = ClearLines();
                double score = Evaluate() + lines * 10;
                board = backup;

                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
            p = Rotate(p);
        }
        return (bestRot, bestCol);
    }

    static void Main()
    {
        string[] init = Console.ReadLine().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        width = int.Parse(init[0]);
        height = int.Parse(init[1]);
        int numPieces = int.Parse(init[2]);
        board = new int[height, width];

        for (int i = 0; i < numPieces; i++)
        {
            string line = Console.ReadLine();
            if (string.IsNullOrEmpty(line)) { i--; continue; }
            char pieceType = line.Trim()[0];

            var (rot, col) = FindBest(pieceType);
            Console.WriteLine($"{rot} {col}");
            Console.Out.Flush();

            int[,] p = pieces[pieceIndex[pieceType]];
            for (int r = 0; r < rot; r++) p = Rotate(p);
            int row = DropRow(p, col);
            if (row >= 0)
            {
                Place(p, col, row);
                ClearLines();
            }

            string result = Console.ReadLine();
            if (result != null && result.Contains("gameover")) break;
        }
    }
}
