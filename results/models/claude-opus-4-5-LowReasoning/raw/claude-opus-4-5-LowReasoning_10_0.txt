#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

struct Rect {
    int w, h, idx;
    int board, x, y, rotated;
};

struct Shelf {
    int y, h, x_used;
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, W, H;
    std::cin >> N >> W >> H;

    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    // Sort by max dimension descending (FFDH heuristic)
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return std::max(a.w, a.h) > std::max(b.w, b.h);
    });

    std::vector<std::vector<Shelf>> boards;

    for (auto& r : rects) {
        int rw = r.w, rh = r.h;
        bool rot = false;

        // Prefer orientation where width <= height for shelf packing
        if (rw > rh && rh <= W && rw <= H) {
            std::swap(rw, rh);
            rot = true;
        }

        // Ensure fits on board
        if (rw > W || rh > H) {
            if (rh <= W && rw <= H) {
                std::swap(rw, rh);
                rot = !rot;
            }
        }

        int best_board = -1, best_shelf = -1, best_waste = INT32_MAX;
        bool best_rot = rot;

        // Try both orientations
        for (int try_rot = 0; try_rot < 2; try_rot++) {
            int tw = try_rot ? rh : rw;
            int th = try_rot ? rw : rh;
            if (tw > W || th > H) continue;

            for (int b = 0; b < (int)boards.size(); b++) {
                // Try existing shelves
                for (int s = 0; s < (int)boards[b].size(); s++) {
                    if (boards[b][s].h >= th && boards[b][s].x_used + tw <= W) {
                        int waste = boards[b][s].h - th;
                        if (waste < best_waste) {
                            best_waste = waste;
                            best_board = b;
                            best_shelf = s;
                            best_rot = (try_rot != 0) != rot;
                        }
                    }
                }
                // Try new shelf
                int y_used = boards[b].empty() ? 0 : boards[b].back().y + boards[b].back().h;
                if (y_used + th <= H && tw <= W) {
                    if (0 < best_waste) {
                        best_waste = 0;
                        best_board = b;
                        best_shelf = -1;
                        best_rot = (try_rot != 0) != rot;
                    }
                }
            }
        }

        if (best_board == -1) {
            // New board
            boards.push_back({});
            best_board = (int)boards.size() - 1;
            best_shelf = -1;
            // Choose best orientation for new shelf
            if (rh > rw && rw <= H && rh <= W) {
                std::swap(rw, rh);
                rot = !rot;
            }
            best_rot = rot;
        }

        int tw = best_rot != (r.w > r.h && r.h <= W && r.w <= H) ? r.h : r.w;
        int th = best_rot != (r.w > r.h && r.h <= W && r.w <= H) ? r.w : r.h;

        // Recalculate actual dimensions
        tw = r.w; th = r.h;
        if (best_rot) std::swap(tw, th);

        if (best_shelf == -1) {
            int y = boards[best_board].empty() ? 0 :
                    boards[best_board].back().y + boards[best_board].back().h;
            boards[best_board].push_back({y, th, tw});
            r.board = best_board;
            r.x = 0;
            r.y = y;
            r.rotated = best_rot ? 1 : 0;
        } else {
            r.board = best_board;
            r.x = boards[best_board][best_shelf].x_used;
            r.y = boards[best_board][best_shelf].y;
            r.rotated = best_rot ? 1 : 0;
            boards[best_board][best_shelf].x_used += tw;
        }
    }

    // Sort back by original index
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return a.idx < b.idx;
    });

    std::cout << boards.size() << '\n';
    for (const auto& r : rects) {
        std::cout << r.board << ' ' << r.x << ' ' << r.y << ' ' << r.rotated << '\n';
    }

    return 0;
}
