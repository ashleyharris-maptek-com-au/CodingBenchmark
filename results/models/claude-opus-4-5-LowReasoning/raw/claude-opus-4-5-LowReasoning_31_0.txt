use std::io::{self, BufRead, BufWriter, Write};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let target: i64 = parts.next().unwrap().parse().unwrap();

    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for line in lines {
        let line = line.unwrap();
        for tok in line.split_whitespace() {
            nums.push(tok.parse().unwrap());
            if nums.len() >= n { break; }
        }
        if nums.len() >= n { break; }
    }

    if target == 0 {
        writeln!(out, "YES").unwrap();
        writeln!(out, "[]").unwrap();
        return;
    }

    if n == 0 {
        writeln!(out, "NO").unwrap();
        return;
    }

    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            writeln!(out, "YES").unwrap();
            writeln!(out, "[{}]", i).unwrap();
            return;
        }
    }

    if n <= 40 {
        if let Some(indices) = meet_in_middle(&nums, target) {
            writeln!(out, "YES").unwrap();
            write!(out, "[").unwrap();
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 { write!(out, " ").unwrap(); }
                write!(out, "{}", idx).unwrap();
            }
            writeln!(out, "]").unwrap();
        } else {
            writeln!(out, "NO").unwrap();
        }
        return;
    }

    let max_val = nums.iter().map(|x| x.abs()).max().unwrap_or(0);
    if n <= 10000 && target.abs() <= 2_000_000 && max_val <= 200_000 {
        if let Some(indices) = dp_subset_sum(&nums, target) {
            writeln!(out, "YES").unwrap();
            write!(out, "[").unwrap();
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 { write!(out, " ").unwrap(); }
                write!(out, "{}", idx).unwrap();
            }
            writeln!(out, "]").unwrap();
        } else {
            writeln!(out, "NO").unwrap();
        }
        return;
    }

    if let Some(indices) = large_subset_sum(&nums, target) {
        writeln!(out, "YES").unwrap();
        write!(out, "[").unwrap();
        for (i, idx) in indices.iter().enumerate() {
            if i > 0 { write!(out, " ").unwrap(); }
            write!(out, "{}", idx).unwrap();
        }
        writeln!(out, "]").unwrap();
    } else {
        writeln!(out, "NO").unwrap();
    }
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;

    let mut left_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    left_sums.insert(0, vec![]);

    for i in 0..mid {
        let mut new_sums: Vec<(i64, Vec<usize>)> = Vec::new();
        for (sum, indices) in &left_sums {
            let new_sum = sum + nums[i];
            let mut new_indices = indices.clone();
            new_indices.push(i);
            new_sums.push((new_sum, new_indices));
        }
        for (s, idx) in new_sums {
            left_sums.entry(s).or_insert(idx);
        }
    }

    let mut right_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    right_sums.insert(0, vec![]);

    for i in mid..n {
        let mut new_sums: Vec<(i64, Vec<usize>)> = Vec::new();
        for (sum, indices) in &right_sums {
            let new_sum = sum + nums[i];
            let mut new_indices = indices.clone();
            new_indices.push(i);
            new_sums.push((new_sum, new_indices));
        }
        for (s, idx) in new_sums {
            right_sums.entry(s).or_insert(idx);
        }
    }

    for (left_sum, left_indices) in &left_sums {
        let need = target - left_sum;
        if let Some(right_indices) = right_sums.get(&need) {
            if left_indices.is_empty() && right_indices.is_empty() {
                continue;
            }
            let mut result = left_indices.clone();
            result.extend(right_indices);
            result.sort();
            return Some(result);
        }
    }
    None
}

fn dp_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let positives: Vec<(usize, i64)> = nums.iter().enumerate()
        .filter(|(_, &v)| v > 0).map(|(i, &v)| (i, v)).collect();
    let negatives: Vec<(usize, i64)> = nums.iter().enumerate()
        .filter(|(_, &v)| v < 0).map(|(i, &v)| (i, -v)).collect();
    let zeros: Vec<usize> = nums.iter().enumerate()
        .filter(|(_, &v)| v == 0).map(|(i, _)| i).collect();

    if target == 0 && !zeros.is_empty() {
        return Some(vec![zeros[0]]);
    }

    let pos_sum: i64 = positives.iter().map(|(_, v)| v).sum();
    let neg_sum: i64 = negatives.iter().map(|(_, v)| v).sum();

    if target > pos_sum || target < -neg_sum {
        return None;
    }

    let offset = neg_sum as usize;
    let size = (pos_sum + neg_sum + 1) as usize;

    if size > 50_000_000 {
        return large_subset_sum(nums, target);
    }

    let mut dp: Vec<i32> = vec![-1; size];
    dp[offset] = 0;

    let all_nums: Vec<(usize, i64)> = nums.iter().enumerate()
        .filter(|(_, &v)| v != 0).map(|(i, &v)| (i, v)).collect();

    for (idx, (_orig_idx, val)) in all_nums.iter().enumerate() {
        let idx_i32 = (idx + 1) as i32;
        if *val > 0 {
            for s in ((*val as usize)..size).rev() {
                let prev = s - (*val as usize);
                if dp[prev] >= 0 && dp[s] < 0 {
                    dp[s] = idx_i32;
                }
            }
        } else {
            let abs_val = (-*val) as usize;
            for s in 0..(size - abs_val) {
                let next = s + abs_val;
                if dp[s] >= 0 && dp[next] < 0 {
                    dp[next] = idx_i32;
                }
            }
        }
    }

    let target_idx = (target + neg_sum as i64) as usize;
    if target_idx >= size || dp[target_idx] < 0 {
        return None;
    }

    let mut result = Vec::new();
    let mut current = target_idx;
    while dp[current] > 0 {
        let idx = (dp[current] - 1) as usize;
        let (orig_idx, val) = all_nums[idx];
        result.push(orig_idx);
        if val > 0 {
            current -= val as usize;
        } else {
            current += (-val) as usize;
        }
    }

    if result.is_empty() {
        return None;
    }

    result.sort();
    Some(result)
}

fn large_subset_sum(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut indexed: Vec<(usize, i64)> = nums.iter().enumerate().map(|(i, &v)| (i, v)).collect();
    indexed.sort_by(|a, b| b.1.abs().cmp(&a.1.abs()));

    let mut current_sum: i64 = 0;
    let mut selected: Vec<usize> = Vec::new();

    for &(idx, val) in &indexed {
        if (target - current_sum).signum() == val.signum() {
            if (current_sum + val - target).abs() <= (current_sum - target).abs() {
                current_sum += val;
                selected.push(idx);
            }
        }
        if current_sum == target {
            selected.sort();
            return Some(selected);
        }
    }

    for seed in 0..200 {
        current_sum = 0;
        selected.clear();

        let start = (seed * 997) % nums.len();
        for i in 0..nums.len() {
            let idx = (start + i) % nums.len();
            let val = nums[idx];
            if val == 0 { continue; }
            if (target - current_sum).signum() == val.signum() {
                if (current_sum + val - target).abs() < (current_sum - target).abs() {
                    current_sum += val;
                    selected.push(idx);
                }
            }
            if current_sum == target {
                selected.sort();
                return Some(selected);
            }
        }
    }

    None
}
