using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int XCount, YCount, ZCount;
    static double XSize, YSize, ZSize;
    static byte[,,] grid;
    static int[,] surface;

    static void Main()
    {
        var header = Console.ReadLine().Split(' ');
        XCount = int.Parse(header[0]);
        YCount = int.Parse(header[1]);
        ZCount = int.Parse(header[2]);
        XSize = double.Parse(header[3]);
        YSize = double.Parse(header[4]);
        ZSize = double.Parse(header[5]);

        grid = new byte[XCount, YCount, ZCount];
        surface = new int[XCount, YCount];

        using var stdin = Console.OpenStandardInput();
        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                surface[x, y] = -1;
                for (int z = 0; z < ZCount; z++)
                {
                    int b = stdin.ReadByte();
                    grid[x, y, z] = (byte)b;
                    if (b > 0 && surface[x, y] < z) surface[x, y] = z;
                }
            }
        }

        var operations = new List<string>();
        bool changed = true;

        while (changed)
        {
            changed = false;
            var candidates = new List<(int x, int y, int z, int mass, int value, double profit)>();

            for (int x = 0; x < XCount; x++)
            {
                for (int y = 0; y < YCount; y++)
                {
                    int z = surface[x, y];
                    if (z < 0) continue;

                    byte d = grid[x, y, z];
                    if (d == 0) continue;

                    int mass = d % 16;
                    int value = d / 16;
                    if (mass == 0) continue;

                    if (value > 0 && CanReachPlant(x, y, z, out var path, out int cost))
                    {
                        double profit = 1000.0 * mass * value - cost * mass;
                        if (profit > 0)
                            candidates.Add((x, y, z, mass, value, profit));
                    }
                }
            }

            if (candidates.Count > 0)
            {
                var best = candidates.OrderByDescending(c => c.profit).First();
                CanReachPlant(best.x, best.y, best.z, out var path, out _);

                grid[best.x, best.y, best.z] = 0;
                UpdateSurface(best.x, best.y);

                string steps = string.Join(" ", path.Select(p => $"{p.dx} {p.dy}"));
                operations.Add($"{path.Count} {best.x} {best.y} {best.z} {steps} 0 0");
                changed = true;
            }
        }

        Console.WriteLine(operations.Count);
        foreach (var op in operations)
            Console.WriteLine(op);
    }

    static void UpdateSurface(int x, int y)
    {
        surface[x, y] = -1;
        for (int z = ZCount - 1; z >= 0; z--)
        {
            if (grid[x, y, z] > 0) { surface[x, y] = z; break; }
        }
    }

    static bool CanReachPlant(int sx, int sy, int sz, out List<(int dx, int dy)> path, out int cost)
    {
        path = new List<(int, int)>();
        cost = 0;

        int x = sx, y = sy, z = sz;
        int[] dx = { -1, 1, 0, 0 };
        int[] dy = { 0, 0, -1, 1 };

        while (x != 0 || y != 0)
        {
            int bestDir = -1, bestCost = int.MaxValue;

            for (int d = 0; d < 4; d++)
            {
                int nx = x + dx[d], ny = y + dy[d];
                if (nx < 0 || ny < 0 || nx >= XCount || ny >= YCount) continue;

                int nz = surface[nx, ny] + 1;
                if (nz - z > 1) continue;

                int dist = Math.Abs(nx) + Math.Abs(ny);
                int moveCost = nz > z ? 5 : 1;

                if (dist < Math.Abs(x) + Math.Abs(y) && moveCost < bestCost)
                {
                    bestDir = d;
                    bestCost = moveCost;
                }
            }

            if (bestDir < 0)
            {
                for (int d = 0; d < 4; d++)
                {
                    int nx = x + dx[d], ny = y + dy[d];
                    if (nx < 0 || ny < 0 || nx >= XCount || ny >= YCount) continue;

                    int nz = surface[nx, ny] + 1;
                    if (nz - z > 1) continue;

                    int moveCost = nz > z ? 5 : 1;
                    if (moveCost < bestCost)
                    {
                        bestDir = d;
                        bestCost = moveCost;
                    }
                }
            }

            if (bestDir < 0) return false;

            path.Add((dx[bestDir], dy[bestDir]));
            cost += bestCost;
            int newZ = surface[x + dx[bestDir], y + dy[bestDir]] + 1;
            x += dx[bestDir];
            y += dy[bestDir];
            z = Math.Max(z, newZ);
        }

        return true;
    }
}
