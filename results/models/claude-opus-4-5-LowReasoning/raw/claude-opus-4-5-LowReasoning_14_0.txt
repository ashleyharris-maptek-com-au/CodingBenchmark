using System;
using System.Collections.Generic;

class Minesweeper
{
    static int H, W;
    static char[,] board;
    static int[] dx = {-1, -1, -1, 0, 0, 1, 1, 1};
    static int[] dy = {-1, 0, 1, -1, 1, -1, 0, 1};

    static void Main()
    {
        var line = Console.ReadLine().Split();
        H = int.Parse(line[0]);
        W = int.Parse(line[1]);
        board = new char[H, W];

        for (int r = 0; r < H; r++)
        {
            string row = Console.ReadLine();
            for (int c = 0; c < W; c++)
                board[r, c] = c < row.Length ? row[c] : ' ';
        }

        var safe = new HashSet<(int, int)>();
        bool changed = true;

        while (changed)
        {
            changed = false;
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    if (board[r, c] >= '0' && board[r, c] <= '8')
                    {
                        int num = board[r, c] - '0';
                        var unknowns = new List<(int, int)>();
                        int mines = 0;

                        for (int i = 0; i < 8; i++)
                        {
                            int nr = r + dy[i], nc = c + dx[i];
                            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                            {
                                if (board[nr, nc] == '*') mines++;
                                else if (board[nr, nc] == ' ') unknowns.Add((nc, nr));
                            }
                        }

                        if (mines == num)
                        {
                            foreach (var (x, y) in unknowns)
                                safe.Add((x, y));
                        }
                        else if (mines + unknowns.Count == num)
                        {
                            foreach (var (x, y) in unknowns)
                            {
                                board[y, x] = '*';
                                changed = true;
                            }
                        }
                    }
                }
            }
        }

        // CSP for harder cases - subset constraint propagation
        var frontier = new List<(int r, int c, int need, List<(int, int)> unknowns)>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] >= '1' && board[r, c] <= '8')
                {
                    int num = board[r, c] - '0';
                    var unknowns = new List<(int, int)>();
                    int mines = 0;

                    for (int i = 0; i < 8; i++)
                    {
                        int nr = r + dy[i], nc = c + dx[i];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            if (board[nr, nc] == '*') mines++;
                            else if (board[nr, nc] == ' ') unknowns.Add((nc, nr));
                        }
                    }

                    if (unknowns.Count > 0)
                        frontier.Add((r, c, num - mines, unknowns));
                }
            }
        }

        // Subset checking between constraint pairs
        foreach (var (r1, c1, need1, u1) in frontier)
        {
            foreach (var (r2, c2, need2, u2) in frontier)
            {
                if (r1 == r2 && c1 == c2) continue;
                var s1 = new HashSet<(int, int)>(u1);
                var s2 = new HashSet<(int, int)>(u2);
                if (s1.IsSubsetOf(s2))
                {
                    var diff = new HashSet<(int, int)>(s2);
                    diff.ExceptWith(s1);
                    int diffNeed = need2 - need1;
                    if (diffNeed == 0)
                    {
                        foreach (var cell in diff)
                            safe.Add(cell);
                    }
                }
            }
        }

        // Remove any that got flagged
        safe.RemoveWhere(p => board[p.Item2, p.Item1] != ' ');

        Console.WriteLine(safe.Count);
        foreach (var (x, y) in safe)
            Console.WriteLine($"{x} {y}");
    }
}
