use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    let first_line: Vec<usize> = lines.next().unwrap()
        .split_whitespace().map(|x| x.parse().unwrap()).collect();
    let sv = first_line[0];
    let np = first_line[1];

    let mut stock: Vec<(f64, f64)> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let coords: Vec<f64> = lines.next().unwrap()
            .split_whitespace().map(|x| x.parse().unwrap()).collect();
        stock.push((coords[0], coords[1]));
    }

    let (stock_min_x, stock_max_x, stock_min_y, stock_max_y) = bounding_box(&stock);
    let stock_width = stock_max_x - stock_min_x;
    let stock_height = stock_max_y - stock_min_y;

    let mut pieces: Vec<Vec<(f64, f64)>> = Vec::with_capacity(np);
    let mut piece_areas: Vec<f64> = Vec::with_capacity(np);

    for _ in 0..np {
        let pv: usize = lines.next().unwrap().trim().parse().unwrap();
        let mut piece: Vec<(f64, f64)> = Vec::with_capacity(pv);
        for _ in 0..pv {
            let coords: Vec<f64> = lines.next().unwrap()
                .split_whitespace().map(|x| x.parse().unwrap()).collect();
            piece.push((coords[0], coords[1]));
        }
        piece_areas.push(polygon_area(&piece));
        pieces.push(piece);
    }

    let mut indices: Vec<usize> = (0..np).collect();
    indices.sort_by(|&a, &b| piece_areas[b].partial_cmp(&piece_areas[a]).unwrap());

    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];
    let mut stocks_used: Vec<Vec<(f64, f64, f64, f64)>> = vec![Vec::new()];

    let rotations: Vec<f64> = if np > 1000 {
        vec![0.0, std::f64::consts::FRAC_PI_2]
    } else if np > 100 {
        vec![0.0, std::f64::consts::FRAC_PI_4, std::f64::consts::FRAC_PI_2, 3.0 * std::f64::consts::FRAC_PI_4]
    } else {
        (0..8).map(|i| i as f64 * std::f64::consts::FRAC_PI_4).collect()
    };

    let grid_step = if np > 1000 {
        (stock_width.min(stock_height) / 10.0).max(1.0)
    } else if np > 100 {
        (stock_width.min(stock_height) / 20.0).max(0.5)
    } else {
        (stock_width.min(stock_height) / 30.0).max(0.25)
    };

    for &piece_idx in &indices {
        let piece = &pieces[piece_idx];
        let mut placed = false;

        'stock_loop: for stock_idx in 0..stocks_used.len() {
            for &rot in &rotations {
                let rotated = rotate_polygon(piece, rot);
                let (pmin_x, pmax_x, pmin_y, pmax_y) = bounding_box(&rotated);
                let pw = pmax_x - pmin_x;
                let ph = pmax_y - pmin_y;

                if pw > stock_width || ph > stock_height {
                    continue;
                }

                let mut x = stock_min_x - pmin_x;
                while x <= stock_max_x - pmax_x + 0.001 {
                    let mut y = stock_min_y - pmin_y;
                    while y <= stock_max_y - pmax_y + 0.001 {
                        let translated = translate_polygon(&rotated, x, y);
                        let (tx_min, tx_max, ty_min, ty_max) = bounding_box(&translated);

                        if polygon_inside_stock(&translated, &stock) {
                            let mut overlaps = false;
                            for &(ox, oy, ow, oh) in &stocks_used[stock_idx] {
                                if tx_min < ox + ow && tx_max > ox && ty_min < oy + oh && ty_max > oy {
                                    overlaps = true;
                                    break;
                                }
                            }

                            if !overlaps {
                                placements[piece_idx] = (stock_idx, x, y, rot);
                                stocks_used[stock_idx].push((tx_min, ty_min, tx_max - tx_min, ty_max - ty_min));
                                placed = true;
                                break 'stock_loop;
                            }
                        }
                        y += grid_step;
                    }
                    x += grid_step;
                }
            }
        }

        if !placed {
            let new_stock_idx = stocks_used.len();
            stocks_used.push(Vec::new());

            for &rot in &rotations {
                let rotated = rotate_polygon(piece, rot);
                let (pmin_x, pmax_x, pmin_y, pmax_y) = bounding_box(&rotated);
                let pw = pmax_x - pmin_x;
                let ph = pmax_y - pmin_y;

                if pw > stock_width || ph > stock_height {
                    continue;
                }

                let x = stock_min_x - pmin_x;
                let y = stock_min_y - pmin_y;
                let translated = translate_polygon(&rotated, x, y);

                if polygon_inside_stock(&translated, &stock) {
                    let (tx_min, tx_max, ty_min, ty_max) = bounding_box(&translated);
                    placements[piece_idx] = (new_stock_idx, x, y, rot);
                    stocks_used[new_stock_idx].push((tx_min, ty_min, tx_max - tx_min, ty_max - ty_min));
                    placed = true;
                    break;
                }
            }

            if !placed {
                placements[piece_idx] = (new_stock_idx, 0.0, 0.0, 0.0);
            }
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", stocks_used.len()).unwrap();
    for (stock_idx, x, y, rot) in placements {
        writeln!(out, "{} {} {} {}", stock_idx, x, y, rot).unwrap();
    }
}

fn bounding_box(poly: &[(f64, f64)]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    for &(x, y) in poly {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }
    (min_x, max_x, min_y, max_y)
}

fn polygon_area(poly: &[(f64, f64)]) -> f64 {
    let n = poly.len();
    let mut area = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        area += poly[i].0 * poly[j].1;
        area -= poly[j].0 * poly[i].1;
    }
    area.abs() / 2.0
}

fn rotate_polygon(poly: &[(f64, f64)], angle: f64) -> Vec<(f64, f64)> {
    let cos_a = angle.cos();
    let sin_a = angle.sin();
    poly.iter().map(|&(x, y)| (x * cos_a - y * sin_a, x * sin_a + y * cos_a)).collect()
}

fn translate_polygon(poly: &[(f64, f64)], dx: f64, dy: f64) -> Vec<(f64, f64)> {
    poly.iter().map(|&(x, y)| (x + dx, y + dy)).collect()
}

fn polygon_inside_stock(piece: &[(f64, f64)], stock: &[(f64, f64)]) -> bool {
    for &p in piece {
        if !point_in_polygon(p, stock) {
            return false;
        }
    }
    true
}

fn point_in_polygon(point: (f64, f64), polygon: &[(f64, f64)]) -> bool {
    let (px, py) = point;
    let n = polygon.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = polygon[i];
        let (xj, yj) = polygon[j];
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}
