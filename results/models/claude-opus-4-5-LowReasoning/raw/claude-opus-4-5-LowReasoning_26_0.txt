use std::io::{self, BufReader, BufWriter, Read, Write, BufRead};

const BATCH_SIZE: usize = 100_000;
const MAX_ITERATIONS: usize = 50;
const CONVERGENCE_THRESHOLD: f64 = 0.0001;

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    let mut header = String::new();
    reader.read_line(&mut header)?;
    let parts: Vec<usize> = header.trim().split_whitespace()
        .map(|s| s.parse().unwrap()).collect();
    let num_points = parts[0];
    let num_clusters = parts[1];

    if num_points <= 10_000_000 {
        cluster_in_memory(&mut reader, &mut writer, num_points, num_clusters)?;
    } else {
        cluster_streaming(&mut reader, &mut writer, num_points, num_clusters)?;
    }
    writer.flush()?;
    Ok(())
}

fn cluster_in_memory<R: Read, W: Write>(
    reader: &mut R, writer: &mut W, num_points: usize, num_clusters: usize
) -> io::Result<()> {
    let mut points = vec![[0.0f32; 3]; num_points];
    let mut buf = vec![0u8; num_points * 12];
    reader.read_exact(&mut buf)?;

    for i in 0..num_points {
        let offset = i * 12;
        points[i][0] = f32::from_le_bytes(buf[offset..offset+4].try_into().unwrap());
        points[i][1] = f32::from_le_bytes(buf[offset+4..offset+8].try_into().unwrap());
        points[i][2] = f32::from_le_bytes(buf[offset+8..offset+12].try_into().unwrap());
    }
    drop(buf);

    let mut centroids: Vec<[f64; 3]> = Vec::with_capacity(num_clusters);
    let step = num_points.max(1) / num_clusters.max(1);
    for i in 0..num_clusters {
        let idx = (i * step).min(num_points - 1);
        centroids.push([points[idx][0] as f64, points[idx][1] as f64, points[idx][2] as f64]);
    }

    let mut assignments = vec![0u32; num_points];

    for _ in 0..MAX_ITERATIONS {
        let mut new_centroids = vec![[0.0f64; 3]; num_clusters];
        let mut counts = vec![0usize; num_clusters];
        let mut changed = 0usize;

        for (i, p) in points.iter().enumerate() {
            let mut best = 0;
            let mut best_dist = f64::MAX;
            for (j, c) in centroids.iter().enumerate() {
                let dx = p[0] as f64 - c[0];
                let dy = p[1] as f64 - c[1];
                let dz = p[2] as f64 - c[2];
                let d = dx*dx + dy*dy + dz*dz;
                if d < best_dist { best_dist = d; best = j; }
            }
            if assignments[i] != best as u32 { changed += 1; }
            assignments[i] = best as u32;
            new_centroids[best][0] += p[0] as f64;
            new_centroids[best][1] += p[1] as f64;
            new_centroids[best][2] += p[2] as f64;
            counts[best] += 1;
        }

        for j in 0..num_clusters {
            if counts[j] > 0 {
                centroids[j][0] = new_centroids[j][0] / counts[j] as f64;
                centroids[j][1] = new_centroids[j][1] / counts[j] as f64;
                centroids[j][2] = new_centroids[j][2] / counts[j] as f64;
            }
        }

        if (changed as f64) / (num_points as f64) < CONVERGENCE_THRESHOLD { break; }
    }

    for &a in &assignments {
        writer.write_all(&a.to_le_bytes())?;
    }
    Ok(())
}

fn cluster_streaming<R: Read, W: Write>(
    reader: &mut R, writer: &mut W, num_points: usize, num_clusters: usize
) -> io::Result<()> {
    let init_sample = BATCH_SIZE.min(num_points);
    let mut init_buf = vec![0u8; init_sample * 12];
    reader.read_exact(&mut init_buf)?;

    let mut centroids: Vec<[f64; 3]> = Vec::with_capacity(num_clusters);
    let step = init_sample.max(1) / num_clusters.max(1);
    for i in 0..num_clusters {
        let idx = (i * step).min(init_sample - 1);
        let offset = idx * 12;
        let x = f32::from_le_bytes(init_buf[offset..offset+4].try_into().unwrap()) as f64;
        let y = f32::from_le_bytes(init_buf[offset+4..offset+8].try_into().unwrap()) as f64;
        let z = f32::from_le_bytes(init_buf[offset+8..offset+12].try_into().unwrap()) as f64;
        centroids.push([x, y, z]);
    }

    let remaining = num_points - init_sample;
    let mut rem_buf = vec![0u8; remaining * 12];
    reader.read_exact(&mut rem_buf)?;

    let all_buf: Vec<u8> = [init_buf, rem_buf].concat();

    for iter in 0..MAX_ITERATIONS.min(10) {
        let mut new_centroids = vec![[0.0f64; 3]; num_clusters];
        let mut new_counts = vec![0usize; num_clusters];
        let lr = 1.0 / (iter + 1) as f64;

        for batch_start in (0..num_points).step_by(BATCH_SIZE) {
            let batch_end = (batch_start + BATCH_SIZE).min(num_points);
            for i in batch_start..batch_end {
                let offset = i * 12;
                let x = f32::from_le_bytes(all_buf[offset..offset+4].try_into().unwrap()) as f64;
                let y = f32::from_le_bytes(all_buf[offset+4..offset+8].try_into().unwrap()) as f64;
                let z = f32::from_le_bytes(all_buf[offset+8..offset+12].try_into().unwrap()) as f64;

                let mut best = 0;
                let mut best_dist = f64::MAX;
                for (j, c) in centroids.iter().enumerate() {
                    let dx = x - c[0]; let dy = y - c[1]; let dz = z - c[2];
                    let d = dx*dx + dy*dy + dz*dz;
                    if d < best_dist { best_dist = d; best = j; }
                }
                new_centroids[best][0] += x;
                new_centroids[best][1] += y;
                new_centroids[best][2] += z;
                new_counts[best] += 1;
            }
        }

        for j in 0..num_clusters {
            if new_counts[j] > 0 {
                let new_c = [
                    new_centroids[j][0] / new_counts[j] as f64,
                    new_centroids[j][1] / new_counts[j] as f64,
                    new_centroids[j][2] / new_counts[j] as f64,
                ];
                centroids[j][0] += lr * (new_c[0] - centroids[j][0]);
                centroids[j][1] += lr * (new_c[1] - centroids[j][1]);
                centroids[j][2] += lr * (new_c[2] - centroids[j][2]);
            }
        }
    }

    for i in 0..num_points {
        let offset = i * 12;
        let x = f32::from_le_bytes(all_buf[offset..offset+4].try_into().unwrap()) as f64;
        let y = f32::from_le_bytes(all_buf[offset+4..offset+8].try_into().unwrap()) as f64;
        let z = f32::from_le_bytes(all_buf[offset+8..offset+12].try_into().unwrap()) as f64;

        let mut best = 0u32;
        let mut best_dist = f64::MAX;
        for (j, c) in centroids.iter().enumerate() {
            let dx = x - c[0]; let dy = y - c[1]; let dz = z - c[2];
            let d = dx*dx + dy*dy + dz*dz;
            if d < best_dist { best_dist = d; best = j as u32; }
        }
        writer.write_all(&best.to_le_bytes())?;
    }
    Ok(())
}
