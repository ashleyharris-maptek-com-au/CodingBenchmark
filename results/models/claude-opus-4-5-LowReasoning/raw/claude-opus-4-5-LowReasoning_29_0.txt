using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using System.Diagnostics;

class MaxClique
{
    static int n;
    static HashSet<int>[] adj;
    static int[] bestClique;
    static int bestSize;
    static object lockObj = new object();
    static Stopwatch sw;
    static readonly long TimeLimit = 270000;

    static void Main()
    {
        sw = Stopwatch.StartNew();
        var line = Console.ReadLine().Split();
        n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            adj[v].Add(u);
        }

        bestClique = new int[0];
        bestSize = 0;

        int[] degrees = new int[n];
        for (int i = 0; i < n; i++) degrees[i] = adj[i].Count;
        int[] order = Enumerable.Range(0, n).OrderByDescending(x => degrees[x]).ToArray();

        if (n <= 150 && m < 5000)
        {
            BronKerbosch(new List<int>(), new HashSet<int>(Enumerable.Range(0, n)), new HashSet<int>());
        }
        else
        {
            GreedyClique(order);
            if (sw.ElapsedMilliseconds < TimeLimit - 10000)
            {
                ParallelLocalSearch(order, degrees);
            }
        }

        Console.WriteLine(bestSize);
        Console.WriteLine(string.Join(" ", bestClique));
    }

    static void BronKerbosch(List<int> R, HashSet<int> P, HashSet<int> X)
    {
        if (sw.ElapsedMilliseconds > TimeLimit) return;

        if (P.Count == 0 && X.Count == 0)
        {
            lock (lockObj)
            {
                if (R.Count > bestSize)
                {
                    bestSize = R.Count;
                    bestClique = R.ToArray();
                }
            }
            return;
        }

        int upperBound = R.Count + P.Count;
        if (upperBound <= bestSize) return;

        int pivot = P.Concat(X).OrderByDescending(v => P.Count(p => adj[v].Contains(p))).FirstOrDefault();
        var candidates = P.Where(v => !adj[pivot].Contains(v)).ToList();

        foreach (int v in candidates)
        {
            if (sw.ElapsedMilliseconds > TimeLimit) return;
            var newR = new List<int>(R) { v };
            var newP = new HashSet<int>(P.Where(u => adj[v].Contains(u)));
            var newX = new HashSet<int>(X.Where(u => adj[v].Contains(u)));
            BronKerbosch(newR, newP, newX);
            P.Remove(v);
            X.Add(v);
        }
    }

    static void GreedyClique(int[] order)
    {
        var clique = new List<int>();
        foreach (int v in order)
        {
            if (clique.All(u => adj[v].Contains(u)))
            {
                clique.Add(v);
            }
        }
        lock (lockObj)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                bestClique = clique.ToArray();
            }
        }
    }

    static void ParallelLocalSearch(int[] order, int[] degrees)
    {
        int threads = Environment.ProcessorCount;
        Parallel.For(0, threads, new ParallelOptions { MaxDegreeOfParallelism = threads }, tid =>
        {
            var rnd = new Random(tid * 12345);
            int iterations = 0;
            while (sw.ElapsedMilliseconds < TimeLimit && iterations < 50000)
            {
                iterations++;
                int startIdx = rnd.Next(Math.Min(100, n));
                int start = order[startIdx];
                var clique = new List<int> { start };
                var candidates = new List<int>(adj[start]);
                candidates.Sort((a, b) => degrees[b].CompareTo(degrees[a]));

                while (candidates.Count > 0)
                {
                    int best = -1;
                    int bestDeg = -1;
                    int limit = Math.Min(candidates.Count, 20);
                    for (int i = 0; i < limit; i++)
                    {
                        int c = candidates[i];
                        if (degrees[c] > bestDeg)
                        {
                            bestDeg = degrees[c];
                            best = c;
                        }
                    }
                    if (best == -1) break;
                    clique.Add(best);
                    candidates = candidates.Where(c => c != best && adj[best].Contains(c)).ToList();
                }

                lock (lockObj)
                {
                    if (clique.Count > bestSize)
                    {
                        bestSize = clique.Count;
                        bestClique = clique.ToArray();
                    }
                }
            }
        });
    }
}
