#include <iostream>
#include <vector>
#include <algorithm>
#include <random>
#include <chrono>

using namespace std;

int n, m, k;
vector<vector<int>> adj;
vector<int> color;
vector<int> degree;

bool isValid(int v, int c) {
    for (int u : adj[v]) {
        if (color[u] == c) return false;
    }
    return true;
}

bool backtrack(int v) {
    if (v == n) return true;
    for (int c = 0; c < k; c++) {
        if (isValid(v, c)) {
            color[v] = c;
            if (backtrack(v + 1)) return true;
            color[v] = -1;
        }
    }
    return false;
}

void greedyColor(vector<int>& order) {
    for (int v : order) {
        vector<bool> used(k, false);
        for (int u : adj[v]) {
            if (color[u] >= 0) used[color[u]] = true;
        }
        for (int c = 0; c < k; c++) {
            if (!used[c]) {
                color[v] = c;
                break;
            }
        }
    }
}

void dsatur() {
    vector<int> saturation(n, 0);
    vector<bool> colored(n, false);

    for (int i = 0; i < n; i++) {
        int best = -1;
        for (int v = 0; v < n; v++) {
            if (colored[v]) continue;
            if (best == -1 || saturation[v] > saturation[best] ||
                (saturation[v] == saturation[best] && degree[v] > degree[best])) {
                best = v;
            }
        }

        vector<bool> used(k, false);
        for (int u : adj[best]) {
            if (color[u] >= 0) used[color[u]] = true;
        }

        for (int c = 0; c < k; c++) {
            if (!used[c]) {
                color[best] = c;
                break;
            }
        }

        colored[best] = true;
        for (int u : adj[best]) {
            if (!colored[u]) {
                vector<bool> neighborColors(k, false);
                for (int w : adj[u]) {
                    if (color[w] >= 0) neighborColors[color[w]] = true;
                }
                saturation[u] = count(neighborColors.begin(), neighborColors.end(), true);
            }
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;

    adj.resize(n);
    color.assign(n, -1);
    degree.resize(n, 0);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }

    if (n <= 50 && m <= 200) {
        vector<int> order(n);
        for (int i = 0; i < n; i++) order[i] = i;
        sort(order.begin(), order.end(), [](int a, int b) {
            return degree[a] > degree[b];
        });

        vector<int> reorder(n);
        for (int i = 0; i < n; i++) reorder[order[i]] = i;

        backtrack(0);
    } else if (n <= 10000) {
        dsatur();
    } else {
        vector<int> order(n);
        for (int i = 0; i < n; i++) order[i] = i;
        sort(order.begin(), order.end(), [](int a, int b) {
            return degree[a] > degree[b];
        });
        greedyColor(order);
    }

    for (int i = 0; i < n; i++) {
        if (i > 0) cout << ' ';
        cout << (color[i] >= 0 ? color[i] : 0);
    }
    cout << '\n';

    return 0;
}
