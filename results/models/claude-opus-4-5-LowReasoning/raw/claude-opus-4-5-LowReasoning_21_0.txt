#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <queue>
#include <algorithm>

using namespace std;

int D;
vector<int> bounds;
deque<vector<int>> snake;
set<vector<int>> food;
set<vector<int>> obstacles;

bool inBounds(const vector<int>& pos) {
    for (int i = 0; i < D; i++)
        if (pos[i] < 0 || pos[i] >= bounds[i]) return false;
    return true;
}

bool isSafe(const vector<int>& pos, const set<vector<int>>& body) {
    if (!inBounds(pos)) return false;
    if (obstacles.count(pos)) return false;
    if (body.count(pos)) return false;
    return true;
}

pair<int,int> bfs(const vector<int>& start, const set<vector<int>>& body) {
    if (food.empty()) return {0, 1};

    queue<pair<vector<int>, pair<int,int>>> q;
    set<vector<int>> visited;
    visited.insert(start);

    for (int axis = 0; axis < D; axis++) {
        for (int dir : {-1, 1}) {
            vector<int> next = start;
            next[axis] += dir;
            if (isSafe(next, body)) {
                if (food.count(next)) return {axis, dir};
                q.push({next, {axis, dir}});
                visited.insert(next);
            }
        }
    }

    while (!q.empty()) {
        auto [pos, firstMove] = q.front();
        q.pop();

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> next = pos;
                next[axis] += dir;
                if (isSafe(next, body) && !visited.count(next)) {
                    if (food.count(next)) return firstMove;
                    visited.insert(next);
                    q.push({next, firstMove});
                }
            }
        }
    }

    for (int axis = 0; axis < D; axis++) {
        for (int dir : {-1, 1}) {
            vector<int> next = start;
            next[axis] += dir;
            if (isSafe(next, body)) return {axis, dir};
        }
    }
    return {0, 1};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int max_turns;
    cin >> D >> max_turns;

    bounds.resize(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    int snake_len;
    cin >> snake_len;
    vector<int> head(D);
    for (int i = 0; i < D; i++) cin >> head[i];
    snake.push_back(head);
    for (int i = 1; i < snake_len; i++) snake.push_back(head);

    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        vector<int> pos(D);
        for (int j = 0; j < D; j++) cin >> pos[j];
        food.insert(pos);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        vector<int> pos(D);
        for (int j = 0; j < D; j++) cin >> pos[j];
        obstacles.insert(pos);
    }

    for (int turn = 0; turn < max_turns && !food.empty(); turn++) {
        set<vector<int>> body(snake.begin(), snake.end());
        body.erase(snake.back());

        auto [axis, dir] = bfs(snake.front(), body);
        cout << axis << " " << dir << "\n" << flush;

        vector<int> newHead = snake.front();
        newHead[axis] += dir;
        snake.push_front(newHead);

        if (food.count(newHead)) {
            food.erase(newHead);
        } else {
            snake.pop_back();
        }
    }

    return 0;
}
