using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int n;
    static List<int>[] adj;
    static List<int>[] radj;
    static int bestSize;
    static HashSet<int> bestSet;

    static void Main()
    {
        var line = Console.ReadLine().Split();
        n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        adj = new List<int>[n];
        radj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
            radj[v].Add(u);
        }

        bestSize = n;
        bestSet = new HashSet<int>();

        var removed = new HashSet<int>();
        BranchAndBound(removed, 0);

        Console.WriteLine(bestSet.Count);
        Console.WriteLine(string.Join(" ", bestSet));
    }

    static List<int> FindCycle(HashSet<int> removed)
    {
        var color = new int[n];
        var parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        for (int i = 0; i < n; i++)
        {
            if (removed.Contains(i) || color[i] != 0) continue;
            var cycle = DfsFindCycle(i, color, parent, removed);
            if (cycle != null) return cycle;
        }
        return null;
    }

    static List<int> DfsFindCycle(int u, int[] color, int[] parent, HashSet<int> removed)
    {
        color[u] = 1;
        foreach (var v in adj[u])
        {
            if (removed.Contains(v)) continue;
            if (color[v] == 1)
            {
                var cycle = new List<int> { v };
                int cur = u;
                while (cur != v)
                {
                    cycle.Add(cur);
                    cur = parent[cur];
                }
                return cycle;
            }
            if (color[v] == 0)
            {
                parent[v] = u;
                var cycle = DfsFindCycle(v, color, parent, removed);
                if (cycle != null) return cycle;
            }
        }
        color[u] = 2;
        return null;
    }

    static int LowerBound(HashSet<int> removed)
    {
        var inDeg = new int[n];
        var outDeg = new int[n];
        var active = new bool[n];
        int activeCount = 0;

        for (int i = 0; i < n; i++)
        {
            if (!removed.Contains(i))
            {
                active[i] = true;
                activeCount++;
            }
        }

        for (int i = 0; i < n; i++)
        {
            if (!active[i]) continue;
            foreach (var v in adj[i])
            {
                if (active[v])
                {
                    outDeg[i]++;
                    inDeg[v]++;
                }
            }
        }

        var queue = new Queue<int>();
        for (int i = 0; i < n; i++)
        {
            if (active[i] && (inDeg[i] == 0 || outDeg[i] == 0))
                queue.Enqueue(i);
        }

        while (queue.Count > 0)
        {
            int u = queue.Dequeue();
            if (!active[u]) continue;
            active[u] = false;
            activeCount--;

            foreach (var v in adj[u])
            {
                if (active[v])
                {
                    inDeg[v]--;
                    if (inDeg[v] == 0) queue.Enqueue(v);
                }
            }
            foreach (var v in radj[u])
            {
                if (active[v])
                {
                    outDeg[v]--;
                    if (outDeg[v] == 0) queue.Enqueue(v);
                }
            }
        }

        if (activeCount == 0) return 0;
        return 1;
    }

    static void BranchAndBound(HashSet<int> removed, int depth)
    {
        if (removed.Count >= bestSize) return;

        int lb = LowerBound(removed);
        if (removed.Count + lb >= bestSize) return;

        var cycle = FindCycle(removed);
        if (cycle == null)
        {
            bestSize = removed.Count;
            bestSet = new HashSet<int>(removed);
            return;
        }

        var sorted = cycle.OrderByDescending(v =>
        {
            int score = 0;
            foreach (var u in adj[v]) if (!removed.Contains(u)) score++;
            foreach (var u in radj[v]) if (!removed.Contains(u)) score++;
            return score;
        }).ToList();

        foreach (var v in sorted)
        {
            removed.Add(v);
            BranchAndBound(removed, depth + 1);
            removed.Remove(v);
            if (removed.Count + 1 >= bestSize) break;
        }
    }
}
