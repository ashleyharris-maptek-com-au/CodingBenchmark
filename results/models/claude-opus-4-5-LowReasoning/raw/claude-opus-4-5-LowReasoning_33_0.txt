using System;
using System.Linq;

class Program
{
    static int N;
    static int[][] flow, dist;

    static void Main()
    {
        N = int.Parse(Console.ReadLine());
        flow = new int[N][];
        dist = new int[N][];

        for (int i = 0; i < N; i++)
            flow[i] = Console.ReadLine().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries).Select(int.Parse).ToArray();
        for (int i = 0; i < N; i++)
            dist[i] = Console.ReadLine().Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries).Select(int.Parse).ToArray();

        int[] perm = Enumerable.Range(0, N).ToArray();

        // Greedy construction
        bool[] usedLoc = new bool[N];
        int[] greedy = new int[N];
        int[] flowSum = new int[N];
        int[] distSum = new int[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += flow[i][j];
                distSum[i] += dist[i][j];
            }
        }
        var facOrder = Enumerable.Range(0, N).OrderByDescending(i => flowSum[i]).ToArray();
        var locOrder = Enumerable.Range(0, N).OrderBy(i => distSum[i]).ToArray();
        for (int i = 0; i < N; i++)
        {
            int fac = facOrder[i];
            for (int j = 0; j < N; j++)
            {
                int loc = locOrder[j];
                if (!usedLoc[loc]) { greedy[fac] = loc; usedLoc[loc] = true; break; }
            }
        }

        if (Cost(greedy) < Cost(perm)) Array.Copy(greedy, perm, N);

        // 2-opt local search
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    long delta = SwapDelta(perm, i, j);
                    if (delta < 0)
                    {
                        int t = perm[i]; perm[i] = perm[j]; perm[j] = t;
                        improved = true;
                    }
                }
            }
        }

        Console.WriteLine(Cost(perm));
        Console.WriteLine(string.Join(" ", perm));
    }

    static long Cost(int[] p)
    {
        long c = 0;
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                c += (long)flow[i][j] * dist[p[i]][p[j]];
        return c;
    }

    static long SwapDelta(int[] p, int i, int j)
    {
        long delta = 0;
        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            delta -= (long)flow[i][k] * dist[p[i]][p[k]];
            delta -= (long)flow[k][i] * dist[p[k]][p[i]];
            delta -= (long)flow[j][k] * dist[p[j]][p[k]];
            delta -= (long)flow[k][j] * dist[p[k]][p[j]];
            delta += (long)flow[i][k] * dist[p[j]][p[k]];
            delta += (long)flow[k][i] * dist[p[k]][p[j]];
            delta += (long)flow[j][k] * dist[p[i]][p[k]];
            delta += (long)flow[k][j] * dist[p[k]][p[i]];
        }
        delta -= (long)flow[i][j] * dist[p[i]][p[j]];
        delta -= (long)flow[j][i] * dist[p[j]][p[i]];
        delta += (long)flow[i][j] * dist[p[j]][p[i]];
        delta += (long)flow[j][i] * dist[p[i]][p[j]];
        return delta;
    }
}
