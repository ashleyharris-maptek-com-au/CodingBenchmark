#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<int> adj[1001];
bool edge[1001][1001];
int best;
vector<int> bestSet, currentSet;

void dfs(int v, vector<int>& candidates) {
    if (candidates.empty()) {
        if ((int)currentSet.size() > best) {
            best = currentSet.size();
            bestSet = currentSet;
        }
        return;
    }
    if ((int)(currentSet.size() + candidates.size()) <= best) return;

    int u = candidates.back();
    candidates.pop_back();

    // Branch: exclude u
    dfs(v, candidates);

    // Branch: include u
    vector<int> newCand;
    for (int w : candidates) {
        if (!edge[u][w]) newCand.push_back(w);
    }
    currentSet.push_back(u);
    dfs(v, newCand);
    currentSet.pop_back();

    candidates.push_back(u);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    memset(edge, false, sizeof(edge));

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        edge[u][v] = edge[v][u] = true;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> candidates;
    for (int i = 0; i < n; i++) candidates.push_back(i);

    // Sort by degree (ascending) for better pruning
    sort(candidates.begin(), candidates.end(), [](int a, int b) {
        return adj[a].size() < adj[b].size();
    });

    best = 0;
    dfs(0, candidates);

    cout << best << "\n";
    for (int i = 0; i < best; i++) {
        if (i > 0) cout << " ";
        cout << bestSet[i];
    }
    cout << "\n";

    return 0;
}
