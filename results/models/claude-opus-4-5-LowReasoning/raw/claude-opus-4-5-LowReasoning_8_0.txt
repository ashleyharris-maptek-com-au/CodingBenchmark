use std::io::{self, BufRead, BufWriter, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut dims = first_line.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();

    let mut maze: Vec<Vec<u8>> = Vec::with_capacity(h);
    let mut start = (0usize, 0usize);
    let mut end = (0usize, 0usize);

    for y in 0..h {
        let row = lines.next().unwrap().unwrap().into_bytes();
        for (x, &c) in row.iter().enumerate() {
            if c == b'A' { start = (y, x); }
            else if c == b'B' { end = (y, x); }
        }
        maze.push(row);
    }

    let mut visited = vec![vec![false; w]; h];
    let mut parent: Vec<Vec<(usize, usize)>> = vec![vec![(usize::MAX, usize::MAX); w]; h];
    let mut queue = VecDeque::new();

    queue.push_back(start);
    visited[start.0][start.1] = true;

    let dirs: [(i32, i32); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];
    let mut found = false;

    while let Some((y, x)) = queue.pop_front() {
        if (y, x) == end {
            found = true;
            break;
        }
        for (dy, dx) in dirs {
            let ny = y as i32 + dy;
            let nx = x as i32 + dx;
            if ny >= 0 && ny < h as i32 && nx >= 0 && nx < w as i32 {
                let (ny, nx) = (ny as usize, nx as usize);
                if !visited[ny][nx] && maze[ny][nx] != b'#' {
                    visited[ny][nx] = true;
                    parent[ny][nx] = (y, x);
                    queue.push_back((ny, nx));
                }
            }
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path = Vec::new();
    let mut cur = end;
    while cur != start {
        path.push(cur);
        cur = parent[cur.0][cur.1];
    }
    path.push(start);
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for (y, x) in path {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
