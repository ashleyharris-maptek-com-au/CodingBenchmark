#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
using namespace std;

typedef long long ll;
typedef pair<ll, int> pli;
const ll INF = LLONG_MAX / 2;

int n, m, k;
vector<pair<int, int>> adj[5005];
vector<int> terminals;
ll dist[5005][5005];
ll dp[5005][1 << 11];
int parent[5005][1 << 11];
int parentMask[5005][1 << 11];

void dijkstra(int src) {
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    for (int i = 0; i < n; i++) dist[src][i] = INF;
    dist[src][src] = 0;
    pq.push({0, src});
    while (!pq.empty()) {
        auto [d, u] = pq.top(); pq.pop();
        if (d > dist[src][u]) continue;
        for (auto [v, w] : adj[u]) {
            if (dist[src][u] + w < dist[src][v]) {
                dist[src][v] = dist[src][u] + w;
                pq.push({dist[src][v], v});
            }
        }
    }
}

vector<pair<int,int>> reconstructPath(int u, int v) {
    vector<pair<int,int>> edges;
    if (u == v) return edges;
    vector<int> prev(n, -1);
    vector<ll> d(n, INF);
    priority_queue<pli, vector<pli>, greater<pli>> pq;
    d[u] = 0;
    pq.push({0, u});
    while (!pq.empty()) {
        auto [dd, x] = pq.top(); pq.pop();
        if (dd > d[x]) continue;
        for (auto [y, w] : adj[x]) {
            if (d[x] + w < d[y]) {
                d[y] = d[x] + w;
                prev[y] = x;
                pq.push({d[y], y});
            }
        }
    }
    int cur = v;
    while (cur != u) {
        edges.push_back({prev[cur], cur});
        cur = prev[cur];
    }
    return edges;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;

    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    terminals.resize(k);
    for (int i = 0; i < k; i++) cin >> terminals[i];

    if (k == 1) {
        cout << "0 0\n";
        return 0;
    }

    if (k == 2) {
        dijkstra(terminals[0]);
        auto path = reconstructPath(terminals[0], terminals[1]);
        ll total = dist[terminals[0]][terminals[1]];
        cout << total << " " << path.size() << "\n";
        for (auto [u, v] : path) cout << u << " " << v << "\n";
        return 0;
    }

    for (int i = 0; i < k; i++) dijkstra(terminals[i]);

    if (k > 11) {
        vector<tuple<ll,int,int>> edges;
        for (int i = 0; i < k; i++)
            for (int j = i+1; j < k; j++)
                edges.push_back({dist[terminals[i]][terminals[j]], i, j});
        sort(edges.begin(), edges.end());

        vector<int> par(k);
        for (int i = 0; i < k; i++) par[i] = i;
        function<int(int)> find = [&](int x) { return par[x] == x ? x : par[x] = find(par[x]); };

        vector<pair<int,int>> mstEdges;
        for (auto [w, i, j] : edges) {
            if (find(i) != find(j)) {
                par[find(i)] = find(j);
                mstEdges.push_back({i, j});
            }
        }

        vector<pair<int,int>> result;
        for (auto [i, j] : mstEdges) {
            auto path = reconstructPath(terminals[i], terminals[j]);
            for (auto e : path) result.push_back(e);
        }

        vector<tuple<int,int,int>> allEdges;
        for (auto [u, v] : result) {
            if (u > v) swap(u, v);
            int w = 0;
            for (auto [x, ww] : adj[u]) if (x == v) { w = ww; break; }
            allEdges.push_back({w, u, v});
        }
        sort(allEdges.begin(), allEdges.end());
        allEdges.erase(unique(allEdges.begin(), allEdges.end()), allEdges.end());

        vector<int> par2(n);
        for (int i = 0; i < n; i++) par2[i] = i;
        function<int(int)> find2 = [&](int x) { return par2[x] == x ? x : par2[x] = find2(par2[x]); };

        vector<pair<int,int>> finalEdges;
        ll total = 0;
        for (auto [w, u, v] : allEdges) {
            if (find2(u) != find2(v)) {
                par2[find2(u)] = find2(v);
                finalEdges.push_back({u, v});
                total += w;
            }
        }

        cout << total << " " << finalEdges.size() << "\n";
        for (auto [u, v] : finalEdges) cout << u << " " << v << "\n";
        return 0;
    }

    for (int v = 0; v < n; v++)
        for (int S = 0; S < (1 << k); S++)
            dp[v][S] = INF;

    for (int i = 0; i < k; i++)
        dp[terminals[i]][1 << i] = 0;

    for (int S = 1; S < (1 << k); S++) {
        for (int v = 0; v < n; v++) {
            for (int T = (S - 1) & S; T > 0; T = (T - 1) & S) {
                if (dp[v][T] + dp[v][S ^ T] < dp[v][S]) {
                    dp[v][S] = dp[v][T] + dp[v][S ^ T];
                    parent[v][S] = v;
                    parentMask[v][S] = T;
                }
            }
        }
        priority_queue<pli, vector<pli>, greater<pli>> pq;
        for (int v = 0; v < n; v++)
            if (dp[v][S] < INF) pq.push({dp[v][S], v});

        while (!pq.empty()) {
            auto [d, u] = pq.top(); pq.pop();
            if (d > dp[u][S]) continue;
            for (auto [v, w] : adj[u]) {
                if (dp[u][S] + w < dp[v][S]) {
                    dp[v][S] = dp[u][S] + w;
                    parent[v][S] = u;
                    parentMask[v][S] = S;
                    pq.push({dp[v][S], v});
                }
            }
        }
    }

    int fullMask = (1 << k) - 1;
    int root = 0;
    for (int v = 0; v < n; v++)
        if (dp[v][fullMask] < dp[root][fullMask]) root = v;

    vector<pair<int,int>> result;
    function<void(int, int)> reconstruct = [&](int v, int S) {
        if (S == 0 || (S & (S - 1)) == 0) return;
        int pv = parent[v][S];
        int pm = parentMask[v][S];
        if (pv == v) {
            reconstruct(v, pm);
            reconstruct(v, S ^ pm);
        } else {
            result.push_back({pv, v});
            reconstruct(pv, S);
        }
    };

    reconstruct(root, fullMask);

    ll total = dp[root][fullMask];
    cout << total << " " << result.size() << "\n";
    for (auto [u, v] : result) cout << u << " " << v << "\n";

    return 0;
}
