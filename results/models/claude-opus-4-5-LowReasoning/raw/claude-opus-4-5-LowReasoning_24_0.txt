#include <iostream>
#include <fstream>
#include <sstream>
#include <cmath>
#include <thread>
#include <mutex>
#include <atomic>
#include <chrono>

struct State {
    double x=0, y=0, z=0;
    double vx=0, vy=0, vz=0;
    double pitch=0, yaw=0, roll=0;
    double pitch_rate=0, yaw_rate=0, roll_rate=0;
    double fuel=0;
};

std::mutex state_mutex;
State current_state;
std::atomic<bool> running{true};

double max_thrust = 10.0;
double target_x, target_y, target_z;

void input_thread() {
    std::string line;
    while (running && std::getline(std::cin, line)) {
        if (line.empty()) continue;
        std::istringstream iss(line);
        State s;
        if (iss >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
               >> s.pitch >> s.yaw >> s.roll
               >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel) {
            std::lock_guard<std::mutex> lock(state_mutex);
            current_state = s;
        }
    }
    running = false;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    double width, depth, height, gravity, fuel_init;
    std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel_init;
    double start_x, start_y, start_z;
    std::cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z;
    std::cin.ignore();

    std::string terrain_path;
    std::getline(std::cin, terrain_path);

    std::string state_marker;
    std::cin >> state_marker;
    std::cin.ignore();

    current_state.x = start_x;
    current_state.y = start_y;
    current_state.z = start_z;

    std::thread reader(input_thread);

    const double G = 1.62;
    const double Kp = 0.02;
    const double Kd = 0.35;
    const double Ka = 2.5;
    const double Kad = 1.8;

    while (running) {
        State s;
        {
            std::lock_guard<std::mutex> lock(state_mutex);
            s = current_state;
        }

        double dx = target_x - s.x;
        double dy = target_y - s.y;
        double dz = target_z - s.z;
        double dist_h = std::sqrt(dx*dx + dy*dy);
        double speed = std::sqrt(s.vx*s.vx + s.vy*s.vy + s.vz*s.vz);

        double ax = Kp * dx - Kd * s.vx;
        double ay = Kp * dy - Kd * s.vy;

        double vz_target = -std::min(2.8, std::sqrt(2.0 * G * std::max(0.0, dz - 3.0)));
        if (dz < 8 && dist_h < 8) vz_target = std::max(vz_target, -2.0);
        double az = G + 0.6 * (vz_target - s.vz);

        double amag = std::sqrt(ax*ax + ay*ay + az*az);
        double thrust = std::min(1.0, std::max(0.0, amag / max_thrust));

        double pitch_t = 0, yaw_t = s.yaw;
        if (amag > 0.05) {
            pitch_t = std::asin(std::min(1.0, std::max(-1.0, -ax / amag)));
            double proj = std::sqrt(ay*ay + az*az);
            if (proj > 0.01) yaw_t = std::atan2(ay, az);
        }
        pitch_t = std::min(0.6, std::max(-0.6, pitch_t));

        double pe = pitch_t - s.pitch;
        double ye = yaw_t - s.yaw;
        while (ye > 3.14159) ye -= 6.28318;
        while (ye < -3.14159) ye += 6.28318;

        double pc = std::min(1.0, std::max(-1.0, Ka * pe - Kad * s.pitch_rate));
        double yc = std::min(1.0, std::max(-1.0, Ka * ye - Kad * s.yaw_rate));
        double rc = std::min(1.0, std::max(-1.0, -Ka * s.roll - Kad * s.roll_rate));

        if (dz < 4 && dist_h < 4 && speed < 4) {
            thrust = std::max(0.0, std::min(1.0, (G - 0.4 * (s.vz + 2.0)) / max_thrust));
            pc = -Ka * s.pitch - Kad * s.pitch_rate;
            yc = -Kad * s.yaw_rate;
        }

        std::cout << thrust << " " << pc << " " << yc << " " << rc << "\n";
        std::cout.flush();

        std::this_thread::sleep_for(std::chrono::milliseconds(50));
    }

    reader.detach();
    return 0;
}
