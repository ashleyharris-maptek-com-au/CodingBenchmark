using System;
using System.Collections.Generic;
using System.Globalization;

class Program
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;

    static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var tokens = new Queue<string>(Console.In.ReadToEnd().Split((char[])null, StringSplitOptions.RemoveEmptyEntries));

        int numBodies = int.Parse(tokens.Dequeue());
        double warningDays = double.Parse(tokens.Dequeue());
        double impactorMass = double.Parse(tokens.Dequeue());
        double dvBudget = double.Parse(tokens.Dequeue());
        double asteroidMass = double.Parse(tokens.Dequeue());
        double t0Unix = double.Parse(tokens.Dequeue());

        var bodies = new Dictionary<string, Body>();
        for (int i = 0; i < numBodies; i++)
        {
            string name = tokens.Dequeue();
            double mass = double.Parse(tokens.Dequeue());
            double x = double.Parse(tokens.Dequeue());
            double y = double.Parse(tokens.Dequeue());
            double z = double.Parse(tokens.Dequeue());
            double vx = double.Parse(tokens.Dequeue());
            double vy = double.Parse(tokens.Dequeue());
            double vz = double.Parse(tokens.Dequeue());
            bodies[name] = new Body { Mass = mass, Pos = new Vec3(x, y, z), Vel = new Vec3(vx, vy, vz) };
        }
        string bspPath = tokens.Dequeue();

        var earth = bodies["Earth"];
        var asteroid = bodies["Asteroid"];
        double mu = bodies.ContainsKey("Sun") ? G * bodies["Sun"].Mass : 1.32712440018e20;

        double impactTime = FindClosestApproach(earth, asteroid, mu, warningDays);
        double interceptTime = impactTime * 0.65;
        double launchTime = Math.Max(1, Math.Min(warningDays * 0.02, 20));

        var earthL = Propagate(earth.Pos, earth.Vel, mu, launchTime * DAY);
        var astI = Propagate(asteroid.Pos, asteroid.Vel, mu, interceptTime * DAY);

        double tof = (interceptTime - launchTime) * DAY;
        var (v1, v2) = SolveLambert(earthL.pos, astI.pos, tof, mu);

        Vec3 launchDv = v1 - earthL.vel;
        double launchMag = launchDv.Mag();

        if (launchMag > dvBudget * 0.9)
        {
            launchDv = launchDv * (dvBudget * 0.85 / launchMag);
            launchMag = launchDv.Mag();
        }

        double remaining = dvBudget - launchMag;
        Vec3 impactDv = Vec3.Zero;
        if (remaining > 50)
        {
            Vec3 relVel = v2 - astI.vel;
            if (relVel.Mag() > 1)
                impactDv = relVel.Norm() * Math.Min(remaining * 0.3, 200);
        }

        var burns = new List<(double t, Vec3 dv)> { (launchTime, launchDv) };

        double midT = (launchTime + interceptTime) * 0.5;
        var scMid = Propagate(earthL.pos, earthL.vel + launchDv, mu, (midT - launchTime) * DAY);
        double tofMid = (interceptTime - midT) * DAY;
        if (tofMid > DAY * 5 && remaining > 100)
        {
            var (vm1, vm2) = SolveLambert(scMid.pos, astI.pos, tofMid, mu);
            Vec3 corr = vm1 - scMid.vel;
            double corrMag = corr.Mag();
            if (corrMag > 5 && corrMag < remaining * 0.4)
            {
                burns.Add((midT, corr));
                remaining -= corrMag;
            }
        }

        burns.Add((interceptTime, impactDv));

        Console.WriteLine(burns.Count);
        foreach (var b in burns)
            Console.WriteLine($"{b.t:G17} {b.dv.X:G17} {b.dv.Y:G17} {b.dv.Z:G17}");
    }

    static double FindClosestApproach(Body earth, Body ast, double mu, double maxDays)
    {
        double minD = double.MaxValue, tMin = maxDays * 0.9;
        for (double t = 10; t <= maxDays; t += Math.Max(1, maxDays / 500))
        {
            var e = Propagate(earth.Pos, earth.Vel, mu, t * DAY);
            var a = Propagate(ast.Pos, ast.Vel, mu, t * DAY);
            double d = (e.pos - a.pos).Mag();
            if (d < minD) { minD = d; tMin = t; }
        }
        return tMin;
    }

    static (Vec3 pos, Vec3 vel) Propagate(Vec3 r0, Vec3 v0, double mu, double dt)
    {
        if (Math.Abs(dt) < 1) return (r0, v0);
        double r0m = r0.Mag(), v0m = v0.Mag();
        double xi = v0m * v0m / 2 - mu / r0m;
        double a = -mu / (2 * xi);
        if (a < 0 || double.IsNaN(a) || double.IsInfinity(a)) a = r0m * 2;

        double sqrtMu = Math.Sqrt(mu);
        double chi = sqrtMu * dt / a;
        double psi, c2, c3;

        for (int i = 0; i < 50; i++)
        {
            psi = chi * chi / a;
            c2 = StumpffC2(psi);
            c3 = StumpffC3(psi);
            double r = chi * chi * c2 + Vec3.Dot(r0, v0) / sqrtMu * chi * (1 - psi * c3) + r0m * (1 - psi * c2);
            double dchi = (sqrtMu * dt - chi * chi * chi * c3 - Vec3.Dot(r0, v0) / sqrtMu * chi * chi * c2 - r0m * chi * (1 - psi * c3)) / r;
            chi += dchi;
            if (Math.Abs(dchi) < 1e-10) break;
        }

        psi = chi * chi / a;
        c2 = StumpffC2(psi);
        c3 = StumpffC3(psi);

        double f = 1 - chi * chi / r0m * c2;
        double g = dt - chi * chi * chi / sqrtMu * c3;
        Vec3 r = r0 * f + v0 * g;
        double rm = r.Mag();
        double fdot = sqrtMu / (rm * r0m) * chi * (psi * c3 - 1);
        double gdot = 1 - chi * chi / rm * c2;
        Vec3 v = r0 * fdot + v0 * gdot;

        return (r, v);
    }

    static double StumpffC2(double psi)
    {
        if (Math.Abs(psi) < 1e-6) return 0.5;
        if (psi > 0) return (1 - Math.Cos(Math.Sqrt(psi))) / psi;
        return (Math.Cosh(Math.Sqrt(-psi)) - 1) / (-psi);
    }

    static double StumpffC3(double psi)
    {
        if (Math.Abs(psi) < 1e-6) return 1.0 / 6.0;
        if (psi > 0) { double sq = Math.Sqrt(psi); return (sq - Math.Sin(sq)) / (psi * sq); }
        double sqn = Math.Sqrt(-psi); return (Math.Sinh(sqn) - sqn) / (-psi * sqn);
    }

    static (Vec3 v1, Vec3 v2) SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu)
    {
        double r1m = r1.Mag(), r2m = r2.Mag();
        double cosTA = Vec3.Dot(r1, r2) / (r1m * r2m);
        cosTA = Math.Max(-1, Math.Min(1, cosTA));
        Vec3 h = Vec3.Cross(r1, r2);
        bool prograde = h.Z >= 0;
        double sinTA = (prograde ? 1 : -1) * Math.Sqrt(1 - cosTA * cosTA);

        double A = Math.Sqrt(r1m * r2m * (1 + cosTA));
        if (Math.Abs(A) < 1e-10) A = 1e-10;
        if (!prograde) A = -A;

        double psi = 0, c2 = 0.5, c3 = 1.0 / 6.0;
        double psiLow = -4 * Math.PI * Math.PI, psiHigh = 4 * Math.PI * Math.PI;
        double sqrtMu = Math.Sqrt(mu);

        for (int i = 0; i < 100; i++)
        {
            double y = r1m + r2m + A * (psi * c3 - 1) / Math.Sqrt(c2);
            if (y < 0) { psiLow = psi; psi = (psi + psiHigh) / 2; continue; }
            double chi = Math.Sqrt(y / c2);
            double tCalc = (chi * chi * chi * c3 + A * Math.Sqrt(y)) / sqrtMu;
            if (tCalc < tof) psiLow = psi; else psiHigh = psi;
            psi = (psiLow + psiHigh) / 2;
            c2 = StumpffC2(psi);
            c3 = StumpffC3(psi);
            if (Math.Abs(tCalc - tof) < 1e-6) break;
        }

        double yf = r1m + r2m + A * (psi * c3 - 1) / Math.Sqrt(c2);
        double f = 1 - yf / r1m;
        double g = A * Math.Sqrt(yf / mu);
        double gdot = 1 - yf / r2m;

        Vec3 v1 = (r2 - r1 * f) * (1 / g);
        Vec3 v2 = (r2 * gdot - r1) * (1 / g);
        return (v1, v2);
    }

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 Zero => new Vec3(0, 0, 0);
        public double Mag() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Norm() { double m = Mag(); return m > 0 ? this * (1 / m) : Zero; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    class Body { public double Mass; public Vec3 Pos, Vel; }
}
