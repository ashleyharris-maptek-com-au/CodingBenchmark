using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static List<(int to, int weight, int idx)>[] adj;
    static int n, m;
    static int[] degree;
    static List<(int u, int v, int w)> edges;

    static void Main()
    {
        var line = Console.ReadLine().Split();
        n = int.Parse(line[0]);
        m = int.Parse(line[1]);

        adj = new List<(int, int, int)>[n];
        degree = new int[n];
        edges = new List<(int, int, int)>();

        for (int i = 0; i < n; i++)
            adj[i] = new List<(int, int, int)>();

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            edges.Add((u, v, w));
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
            degree[u]++;
            degree[v]++;
        }

        var oddVerts = new List<int>();
        for (int i = 0; i < n; i++)
            if (degree[i] % 2 == 1)
                oddVerts.Add(i);

        if (oddVerts.Count > 0)
        {
            int k = oddVerts.Count;
            var dist = new long[k, k];
            var path = new List<int>[k, k];

            for (int i = 0; i < k; i++)
            {
                var d = Dijkstra(oddVerts[i], oddVerts);
                for (int j = 0; j < k; j++)
                {
                    dist[i, j] = d[oddVerts[j]].dist;
                    path[i, j] = d[oddVerts[j]].path;
                }
            }

            var matched = new bool[k];
            var matchPairs = new List<(int, int)>();

            if (k <= 20)
            {
                var dp = new long[1 << k];
                var parent = new int[1 << k];
                Array.Fill(dp, long.MaxValue);
                dp[0] = 0;

                for (int mask = 0; mask < (1 << k); mask++)
                {
                    if (dp[mask] == long.MaxValue) continue;
                    int first = -1;
                    for (int i = 0; i < k; i++)
                        if ((mask & (1 << i)) == 0) { first = i; break; }
                    if (first == -1) continue;

                    for (int second = first + 1; second < k; second++)
                    {
                        if ((mask & (1 << second)) != 0) continue;
                        int newMask = mask | (1 << first) | (1 << second);
                        long newDist = dp[mask] + dist[first, second];
                        if (newDist < dp[newMask])
                        {
                            dp[newMask] = newDist;
                            parent[newMask] = mask | (first << 16) | (second << 24);
                        }
                    }
                }

                int cur = (1 << k) - 1;
                while (cur != 0)
                {
                    int p = parent[cur];
                    int i = (p >> 16) & 0xFF;
                    int j = (p >> 24) & 0xFF;
                    matchPairs.Add((i, j));
                    cur = p & 0xFFFF;
                }
            }
            else
            {
                var pq = new SortedSet<(long d, int i, int j)>();
                for (int i = 0; i < k; i++)
                    for (int j = i + 1; j < k; j++)
                        pq.Add((dist[i, j], i, j));

                while (matchPairs.Count < k / 2)
                {
                    var best = pq.Min;
                    pq.Remove(best);
                    if (!matched[best.i] && !matched[best.j])
                    {
                        matched[best.i] = matched[best.j] = true;
                        matchPairs.Add((best.i, best.j));
                    }
                }
            }

            foreach (var (i, j) in matchPairs)
            {
                var p = path[i, j];
                if (p != null)
                {
                    for (int x = 0; x < p.Count - 1; x++)
                    {
                        int u = p[x], v = p[x + 1];
                        foreach (var e in adj[u])
                        {
                            if (e.to == v)
                            {
                                adj[u].Add((v, e.weight, m + edges.Count));
                                adj[v].Add((u, e.weight, m + edges.Count));
                                edges.Add((u, v, e.weight));
                                degree[u]++;
                                degree[v]++;
                                break;
                            }
                        }
                    }
                }
            }
        }

        var circuit = new List<int>();
        var edgeIdx = new int[n];
        var usedEdge = new HashSet<int>();
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            while (edgeIdx[u] < adj[u].Count)
            {
                var (v, w, idx) = adj[u][edgeIdx[u]];
                edgeIdx[u]++;
                if (!usedEdge.Contains(idx))
                {
                    usedEdge.Add(idx);
                    stack.Push(v);
                    found = true;
                    break;
                }
            }
            if (!found)
                circuit.Add(stack.Pop());
        }

        circuit.Reverse();
        Console.WriteLine(string.Join(" ", circuit));
    }

    static Dictionary<int, (long dist, List<int> path)> Dijkstra(int src, List<int> targets)
    {
        var dist = new long[n];
        var prev = new int[n];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(prev, -1);
        dist[src] = 0;

        var pq = new SortedSet<(long d, int v)> { (0, src) };
        var targetSet = new HashSet<int>(targets);
        int found = 0;

        while (pq.Count > 0 && found < targets.Count)
        {
            var (d, u) = pq.Min;
            pq.Remove(pq.Min);

            if (d > dist[u]) continue;
            if (targetSet.Contains(u)) found++;

            foreach (var (v, w, _) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    pq.Remove((dist[v], v));
                    dist[v] = nd;
                    prev[v] = u;
                    pq.Add((nd, v));
                }
            }
        }

        var result = new Dictionary<int, (long, List<int>)>();
        foreach (int t in targets)
        {
            var path = new List<int>();
            for (int v = t; v != -1; v = prev[v])
                path.Add(v);
            path.Reverse();
            result[t] = (dist[t], path);
        }
        return result;
    }
}
