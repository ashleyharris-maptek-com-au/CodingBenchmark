use std::io::{self, BufRead, Write, BufWriter};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        adj[u].insert(v);
        adj[v].insert(u);
    }

    let mut dominated: Vec<bool> = vec![false; n];
    let mut in_set: Vec<bool> = vec![false; n];
    let mut dominating_set: Vec<usize> = Vec::new();
    let mut undominated_count = n;

    while undominated_count > 0 {
        let mut best_v = 0;
        let mut best_score = 0;

        for v in 0..n {
            if in_set[v] { continue; }
            let mut score = if !dominated[v] { 1 } else { 0 };
            for &u in &adj[v] {
                if !dominated[u] { score += 1; }
            }
            if score > best_score {
                best_score = score;
                best_v = v;
            }
        }

        in_set[best_v] = true;
        dominating_set.push(best_v);

        if !dominated[best_v] {
            dominated[best_v] = true;
            undominated_count -= 1;
        }
        for &u in &adj[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                undominated_count -= 1;
            }
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", dominating_set.len()).unwrap();
    let result: Vec<String> = dominating_set.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", result.join(" ")).unwrap();
}
