#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>

using namespace std;

struct Edge {
    int u, v;
};

class MaxFlow {
    struct FlowEdge {
        int to, rev;
        int cap, flow;
    };

    int n;
    vector<vector<FlowEdge>> g;
    vector<int> level, iter;

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front(); q.pop();
            for (auto& e : g[v]) {
                if (e.cap > e.flow && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)g[v].size(); i++) {
            FlowEdge& e = g[v][i];
            if (e.cap > e.flow && level[v] < level[e.to]) {
                int d = dfs(e.to, t, min(f, e.cap - e.flow));
                if (d > 0) {
                    e.flow += d;
                    g[e.to][e.rev].flow -= d;
                    return d;
                }
            }
        }
        return 0;
    }

public:
    MaxFlow(int n) : n(n), g(n), level(n), iter(n) {}

    void addEdge(int from, int to, int cap) {
        g[from].push_back({to, (int)g[to].size(), cap, 0});
        g[to].push_back({from, (int)g[from].size() - 1, cap, 0});
    }

    int maxflow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, numeric_limits<int>::max())) > 0) {
                flow += f;
            }
        }
        return flow;
    }

    void reset() {
        for (auto& adj : g) {
            for (auto& e : adj) {
                e.flow = 0;
            }
        }
    }

    vector<int> getMinCutSide(int s) {
        vector<int> side;
        for (int i = 0; i < n; i++) {
            if (level[i] >= 0) side.push_back(i);
        }
        return side;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    if (n <= 1) {
        cout << 0 << "\n";
        return 0;
    }

    vector<Edge> edges(m);
    vector<vector<pair<int,int>>> adj(n);

    for (int i = 0; i < m; i++) {
        cin >> edges[i].u >> edges[i].v;
        adj[edges[i].u].push_back({edges[i].v, i});
        adj[edges[i].v].push_back({edges[i].u, i});
    }

    int minDeg = m + 1, minDegV = 0;
    for (int i = 0; i < n; i++) {
        if ((int)adj[i].size() < minDeg) {
            minDeg = adj[i].size();
            minDegV = i;
        }
    }

    if (minDeg == 1) {
        auto& e = edges[adj[minDegV][0].second];
        cout << 1 << "\n" << e.u << " " << e.v << "\n";
        return 0;
    }

    int bestCut = minDeg;
    int bestT = -1;
    vector<int> bestSide;

    MaxFlow mf(n);
    for (int i = 0; i < m; i++) {
        mf.addEdge(edges[i].u, edges[i].v, 1);
    }

    for (int t = 1; t < n && bestCut > 1; t++) {
        mf.reset();
        int cut = mf.maxflow(0, t);
        if (cut < bestCut) {
            bestCut = cut;
            bestT = t;
            bestSide = mf.getMinCutSide(0);
        }
    }

    vector<bool> inSide(n, false);
    for (int v : bestSide) inSide[v] = true;

    vector<Edge> cutEdges;
    for (auto& e : edges) {
        if (inSide[e.u] != inSide[e.v]) {
            cutEdges.push_back(e);
        }
    }

    cout << cutEdges.size() << "\n";
    for (auto& e : cutEdges) {
        cout << e.u << " " << e.v << "\n";
    }

    return 0;
}
