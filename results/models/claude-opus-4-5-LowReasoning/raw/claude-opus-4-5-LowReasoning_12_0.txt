using System;
using System.Collections.Generic;
using System.Globalization;

class Program
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int V = int.Parse(line[0]), F = int.Parse(line[1]);

        double minX = double.MaxValue, minY = double.MaxValue, minZ = double.MaxValue;
        double maxX = double.MinValue, maxY = double.MinValue, maxZ = double.MinValue;

        for (int i = 0; i < V; i++)
        {
            var v = Console.ReadLine().Split();
            double x = double.Parse(v[0], CultureInfo.InvariantCulture);
            double y = double.Parse(v[1], CultureInfo.InvariantCulture);
            double z = double.Parse(v[2], CultureInfo.InvariantCulture);
            minX = Math.Min(minX, x); maxX = Math.Max(maxX, x);
            minY = Math.Min(minY, y); maxY = Math.Max(maxY, y);
            minZ = Math.Min(minZ, z); maxZ = Math.Max(maxZ, z);
        }

        for (int i = 0; i < F; i++) Console.ReadLine();

        var last = Console.ReadLine().Split();
        double W = double.Parse(last[0], CultureInfo.InvariantCulture);
        double H = double.Parse(last[1], CultureInfo.InvariantCulture);
        double D = double.Parse(last[2], CultureInfo.InvariantCulture);

        double dx = maxX - minX, dy = maxY - minY, dz = maxZ - minZ;
        double[] dims = {dx, dy, dz};
        Array.Sort(dims);

        double[][] rotations = {
            new[] {1.0, 0, 0, 0, dx, dy, dz},
            new[] {0.7071067811865476, 0.7071067811865476, 0, 0, dx, dz, dy},
            new[] {0.7071067811865476, 0, 0.7071067811865476, 0, dz, dy, dx},
            new[] {0.7071067811865476, 0, 0, 0.7071067811865476, dy, dx, dz},
            new[] {0.5, 0.5, 0.5, 0.5, dy, dz, dx},
            new[] {0.5, 0.5, -0.5, 0.5, dz, dx, dy}
        };

        int bestCount = 0;
        double[] bestRot = rotations[0];
        double bestSx = dx, bestSy = dy, bestSz = dz;

        foreach (var rot in rotations)
        {
            double sx = rot[4], sy = rot[5], sz = rot[6];
            if (sx < 1e-9 || sy < 1e-9 || sz < 1e-9) continue;
            int nx = (int)(W / sx), ny = (int)(H / sy), nz = (int)(D / sz);
            int count = nx * ny * nz;
            if (count > bestCount)
            {
                bestCount = count;
                bestRot = rot;
                bestSx = sx; bestSy = sy; bestSz = sz;
            }
        }

        int countX = (int)(W / bestSx), countY = (int)(H / bestSy), countZ = (int)(D / bestSz);

        var results = new List<string>();
        for (int ix = 0; ix < countX; ix++)
        {
            for (int iy = 0; iy < countY; iy++)
            {
                for (int iz = 0; iz < countZ; iz++)
                {
                    double tx = ix * bestSx - minX;
                    double ty = iy * bestSy - minY;
                    double tz = iz * bestSz - minZ;

                    results.Add(string.Format(CultureInfo.InvariantCulture, "{0} {1} {2} {3} {4} {5} {6}",
                        tx, ty, tz, bestRot[0], bestRot[1], bestRot[2], bestRot[3]));
                }
            }
        }

        Console.WriteLine(results.Count);
        foreach (var r in results) Console.WriteLine(r);
    }
}
