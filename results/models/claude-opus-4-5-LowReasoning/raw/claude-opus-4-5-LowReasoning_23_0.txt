use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::Duration;

#[derive(Clone, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    target_x: f64,
    target_y: f64,
    gravity: f64,
    max_thrust: f64,
    initialized: bool,
}

fn main() {
    let state = Arc::new(Mutex::new(State::default()));
    let state_writer = Arc::clone(&state);

    // Reader thread
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut lines = stdin.lock().lines();

        // Parse initial map header
        if let Some(Ok(line)) = lines.next() {
            let vals: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok()).collect();
            if vals.len() >= 5 {
                let mut s = state_writer.lock().unwrap();
                s.gravity = vals[2];
                s.max_thrust = vals[3];
                s.fuel = vals[4];
            }
        }

        // Parse start/target line
        if let Some(Ok(line)) = lines.next() {
            let vals: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok()).collect();
            if vals.len() >= 4 {
                let mut s = state_writer.lock().unwrap();
                s.x = vals[0];
                s.y = vals[1];
                s.target_x = vals[2];
                s.target_y = vals[3];
                s.initialized = true;
            }
        }

        // Skip map rows until we get state updates (7 numbers)
        for line in lines {
            if let Ok(line) = line {
                let vals: Vec<f64> = line.split_whitespace()
                    .filter_map(|s| s.parse().ok()).collect();
                if vals.len() == 7 {
                    let mut s = state_writer.lock().unwrap();
                    s.x = vals[0];
                    s.y = vals[1];
                    s.vx = vals[2];
                    s.vy = vals[3];
                    s.angle = vals[4];
                    s.angular_vel = vals[5];
                    s.fuel = vals[6];
                }
            }
        }
    });

    let stdout = io::stdout();
    let mut out = stdout.lock();

    loop {
        thread::sleep(Duration::from_millis(20));

        let s = state.lock().unwrap().clone();
        if !s.initialized {
            let _ = writeln!(out, "0.0 0.0");
            let _ = out.flush();
            continue;
        }

        let dx = s.target_x - s.x;
        let dy = s.target_y - s.y;
        let dist = (dx * dx + dy * dy).sqrt().max(0.1);

        // Desired angle to point toward target (accounting for gravity)
        let desired_angle = if dy > 10.0 {
            // Far above target: point somewhat toward it
            (dx / dist).asin().clamp(-0.8, 0.8)
        } else {
            // Close to target: point up for soft landing
            0.0
        };

        // Turn control (PD controller)
        let angle_error = desired_angle - s.angle;
        let turn = (angle_error * 3.0 - s.angular_vel * 2.0).clamp(-1.0, 1.0);

        // Thrust control
        let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
        let thrust = if dy > 5.0 {
            // Above target: minimal thrust, let gravity work
            if s.vy > 3.0 { 0.3 } else { 0.0 }
        } else if speed > 2.0 || s.vy > 1.5 {
            // Need to slow down
            ((speed - 1.0) / 3.0).clamp(0.3, 1.0)
        } else if s.vy > 0.5 {
            // Gentle descent
            0.4
        } else {
            // Hovering/ascending - reduce thrust
            0.1
        };

        // Only thrust if roughly pointing up
        let effective_thrust = if s.angle.abs() < 0.5 { thrust } else { thrust * 0.3 };

        let _ = writeln!(out, "{:.3} {:.3}", effective_thrust, turn);
        let _ = out.flush();
    }
}
