using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var line1 = Console.ReadLine().Split(' ');
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        var cuts = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries)
                         .Select(int.Parse).ToArray();

        var indexed = new (int len, int idx)[n];
        for (int i = 0; i < n; i++)
            indexed[i] = (cuts[i], i);

        Array.Sort(indexed, (a, b) => b.len.CompareTo(a.len));

        var stocks = new List<(int remaining, List<int> indices)>();

        foreach (var (len, idx) in indexed)
        {
            int bestFit = -1;
            int bestRemaining = int.MaxValue;

            for (int i = 0; i < stocks.Count; i++)
            {
                int rem = stocks[i].remaining;
                if (rem >= len && rem - len < bestRemaining)
                {
                    bestRemaining = rem - len;
                    bestFit = i;
                    if (bestRemaining == 0) break;
                }
            }

            if (bestFit >= 0)
            {
                var s = stocks[bestFit];
                s.remaining -= len;
                s.indices.Add(idx);
                stocks[bestFit] = s;
            }
            else
            {
                stocks.Add((stockLength - len, new List<int> { idx }));
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var s in stocks)
            Console.WriteLine(string.Join(" ", s.indices));
    }
}
