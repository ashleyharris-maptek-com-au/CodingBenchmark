use std::io::{self, BufRead, Write, BufWriter};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<i32> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let width = parts[0] as usize;
    let height = parts[1] as usize;
    let obstacle_count = parts[2] as usize;
    let require_cycle = parts[3] == 1;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.insert((coords[0], coords[1]));
    }

    let total_cells = width * height - obstacle_count;

    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }

    let mut visited = vec![vec![false; height]; width];
    let mut path = Vec::with_capacity(total_cells);

    for &(x, y) in &obstacles {
        visited[x][y] = true;
    }

    visited[0][0] = true;
    path.push((0usize, 0usize));

    if solve(width, height, total_cells, require_cycle, &mut visited, &mut path) {
        let stdout = io::stdout();
        let mut out = BufWriter::new(stdout.lock());
        writeln!(out, "{}", path.len()).unwrap();
        for &(x, y) in &path {
            writeln!(out, "{} {}", x, y).unwrap();
        }
    } else {
        println!("0");
    }
}

fn solve(
    width: usize,
    height: usize,
    target: usize,
    require_cycle: bool,
    visited: &mut Vec<Vec<bool>>,
    path: &mut Vec<(usize, usize)>,
) -> bool {
    if path.len() == target {
        if require_cycle {
            let (x, y) = path[path.len() - 1];
            return (x == 0 && y == 1) || (x == 1 && y == 0);
        }
        return true;
    }

    let (cx, cy) = path[path.len() - 1];

    let mut neighbors: Vec<(usize, usize, usize)> = Vec::with_capacity(4);

    for (dx, dy) in [(0i32, 1i32), (1, 0), (0, -1), (-1, 0)] {
        let nx = cx as i32 + dx;
        let ny = cy as i32 + dy;
        if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;
            if !visited[nx][ny] {
                let degree = count_neighbors(width, height, nx, ny, visited);
                neighbors.push((nx, ny, degree));
            }
        }
    }

    neighbors.sort_by_key(|&(_, _, d)| d);

    for (nx, ny, _) in neighbors {
        visited[nx][ny] = true;
        path.push((nx, ny));

        if solve(width, height, target, require_cycle, visited, path) {
            return true;
        }

        path.pop();
        visited[nx][ny] = false;
    }

    false
}

fn count_neighbors(width: usize, height: usize, x: usize, y: usize, visited: &Vec<Vec<bool>>) -> usize {
    let mut count = 0;
    for (dx, dy) in [(0i32, 1i32), (1, 0), (0, -1), (-1, 0)] {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;
        if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
            if !visited[nx as usize][ny as usize] {
                count += 1;
            }
        }
    }
    count
}
