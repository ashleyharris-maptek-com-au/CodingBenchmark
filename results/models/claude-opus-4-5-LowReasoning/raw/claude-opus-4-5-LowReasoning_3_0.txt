use std::io::{self, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let mut p = line.split_whitespace();
            let u: usize = p.next().unwrap().parse().unwrap();
            let v: usize = p.next().unwrap().parse().unwrap();
            if u != v && u < n && v < n {
                adj[u].push(v);
                adj[v].push(u);
            }
        }
    }

    let mut pos: Vec<(f64, f64)> = Vec::with_capacity(n);

    if n <= 1 {
        for i in 0..n {
            pos.push((i as f64, 0.0));
        }
    } else if n <= 500 {
        let scale = (n as f64).sqrt() * 10.0;
        for i in 0..n {
            let angle = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
            pos.push((scale * angle.cos(), scale * angle.sin()));
        }

        let k = scale / (n as f64).sqrt();
        let iterations = if n < 100 { 200 } else { 100 };

        for iter in 0..iterations {
            let temp = 0.1 * scale * (1.0 - iter as f64 / iterations as f64);
            let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

            for i in 0..n {
                for j in (i+1)..n {
                    let dx = pos[i].0 - pos[j].0;
                    let dy = pos[i].1 - pos[j].1;
                    let dist = (dx*dx + dy*dy).sqrt().max(0.01);
                    let force = k * k / dist;
                    let fx = dx / dist * force;
                    let fy = dy / dist * force;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }

            for u in 0..n {
                for &v in &adj[u] {
                    if u < v {
                        let dx = pos[u].0 - pos[v].0;
                        let dy = pos[u].1 - pos[v].1;
                        let dist = (dx*dx + dy*dy).sqrt().max(0.01);
                        let force = dist / k;
                        let fx = dx / dist * force;
                        let fy = dy / dist * force;
                        disp[u].0 -= fx;
                        disp[u].1 -= fy;
                        disp[v].0 += fx;
                        disp[v].1 += fy;
                    }
                }
            }

            for i in 0..n {
                let len = (disp[i].0*disp[i].0 + disp[i].1*disp[i].1).sqrt().max(0.01);
                pos[i].0 += disp[i].0 / len * temp.min(len);
                pos[i].1 += disp[i].1 / len * temp.min(len);
            }
        }
    } else {
        let cols = (n as f64).sqrt().ceil() as usize;
        let spacing = 10.0;
        for i in 0..n {
            let row = i / cols;
            let col = i % cols;
            pos.push((col as f64 * spacing, row as f64 * spacing));
        }

        let k = spacing;
        let iterations = 20.min(5000 / n.max(1));

        for iter in 0..iterations {
            let temp = spacing * 0.5 * (1.0 - iter as f64 / iterations as f64);
            let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

            for u in 0..n {
                for &v in &adj[u] {
                    if u < v {
                        let dx = pos[u].0 - pos[v].0;
                        let dy = pos[u].1 - pos[v].1;
                        let dist = (dx*dx + dy*dy).sqrt().max(0.01);
                        let force = dist / k * 0.1;
                        let fx = dx / dist * force;
                        let fy = dy / dist * force;
                        disp[u].0 -= fx;
                        disp[u].1 -= fy;
                        disp[v].0 += fx;
                        disp[v].1 += fy;
                    }
                }
            }

            for i in 0..n {
                let len = (disp[i].0*disp[i].0 + disp[i].1*disp[i].1).sqrt().max(0.01);
                pos[i].0 += disp[i].0 / len * temp.min(len);
                pos[i].1 += disp[i].1 / len * temp.min(len);
            }
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    for (x, y) in pos {
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
