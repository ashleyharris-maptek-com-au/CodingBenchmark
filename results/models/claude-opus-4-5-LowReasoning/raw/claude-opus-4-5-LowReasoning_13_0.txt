#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>

using namespace std;

typedef unsigned long long ull;
const ull BASE = 31;
const ull MOD1 = 1000000007ULL;
const ull MOD2 = 1000000009ULL;

vector<ull> pw1, pw2;

void precompute_powers(size_t maxLen) {
    pw1.resize(maxLen + 1);
    pw2.resize(maxLen + 1);
    pw1[0] = pw2[0] = 1;
    for (size_t i = 1; i <= maxLen; i++) {
        pw1[i] = (pw1[i-1] * BASE) % MOD1;
        pw2[i] = (pw2[i-1] * BASE) % MOD2;
    }
}

pair<ull, ull> getHash(const string& s, int start, int len, const vector<ull>& h1, const vector<ull>& h2) {
    ull hash1 = (h1[start + len] - h1[start] * pw1[len] % MOD1 + MOD1) % MOD1;
    ull hash2 = (h2[start + len] - h2[start] * pw2[len] % MOD2 + MOD2) % MOD2;
    return {hash1, hash2};
}

void computePrefixHash(const string& s, vector<ull>& h1, vector<ull>& h2) {
    int n = (int)s.size();
    h1.resize(n + 1);
    h2.resize(n + 1);
    h1[0] = h2[0] = 0;
    for (int i = 0; i < n; i++) {
        h1[i+1] = (h1[i] * BASE + (unsigned char)s[i]) % MOD1;
        h2[i+1] = (h2[i] * BASE + (unsigned char)s[i]) % MOD2;
    }
}

string checkLength(int len, const vector<string>& strs, int minIdx,
                   const vector<vector<ull>>& allH1, const vector<vector<ull>>& allH2) {
    if (len == 0) return "";

    int n = (int)strs.size();
    const string& shortest = strs[minIdx];

    unordered_map<ull, vector<int>> candidates;

    for (int i = 0; i + len <= (int)shortest.size(); i++) {
        auto h = getHash(shortest, i, len, allH1[minIdx], allH2[minIdx]);
        ull combined = h.first * MOD2 + h.second;
        candidates[combined].push_back(i);
    }

    for (int s = 0; s < n; s++) {
        if (s == minIdx) continue;

        unordered_set<ull> found;
        for (int i = 0; i + len <= (int)strs[s].size(); i++) {
            auto h = getHash(strs[s], i, len, allH1[s], allH2[s]);
            ull combined = h.first * MOD2 + h.second;
            if (candidates.count(combined)) {
                found.insert(combined);
            }
        }

        for (auto it = candidates.begin(); it != candidates.end(); ) {
            if (!found.count(it->first)) {
                it = candidates.erase(it);
            } else {
                ++it;
            }
        }

        if (candidates.empty()) return "";
    }

    if (!candidates.empty()) {
        int pos = candidates.begin()->second[0];
        return shortest.substr(pos, len);
    }
    return "";
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strs(n);
    size_t minLen = SIZE_MAX;
    int minIdx = 0;
    size_t maxLen = 0;

    for (int i = 0; i < n; i++) {
        getline(cin, strs[i]);
        if (strs[i].size() < minLen) {
            minLen = strs[i].size();
            minIdx = i;
        }
        maxLen = max(maxLen, strs[i].size());
    }

    if (n == 0 || minLen == 0) {
        cout << "" << endl;
        return 0;
    }

    precompute_powers(maxLen);

    vector<vector<ull>> allH1(n), allH2(n);
    for (int i = 0; i < n; i++) {
        computePrefixHash(strs[i], allH1[i], allH2[i]);
    }

    int lo = 0, hi = (int)minLen;
    string result = "";

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        string found = checkLength(mid, strs, minIdx, allH1, allH2);
        if (!found.empty() || mid == 0) {
            if (mid > 0) result = found;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    cout << result << endl;
    return 0;
}
