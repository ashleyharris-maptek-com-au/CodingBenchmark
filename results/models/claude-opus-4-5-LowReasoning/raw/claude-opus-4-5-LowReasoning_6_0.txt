using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    static double mu;

    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        mu = double.Parse(line1[1]);

        var start = ParseOrbit(Console.ReadLine());
        var stations = new double[n][];
        for (int i = 0; i < n; i++)
            stations[i] = ParseOrbit(Console.ReadLine());

        double[,] cost = new double[n + 1, n + 1];
        for (int i = 0; i <= n; i++)
        {
            var from = i == 0 ? start : stations[i - 1];
            for (int j = 0; j <= n; j++)
            {
                if (i == j) { cost[i, j] = 0; continue; }
                var to = j == 0 ? start : stations[j - 1];
                cost[i, j] = EstimateDeltaV(from, to);
            }
        }

        int[] tour;
        if (n <= 8)
            tour = BruteForce(n, cost);
        else
            tour = GreedyWith2Opt(n, cost);

        Console.WriteLine(string.Join(" ", tour));
    }

    static double[] ParseOrbit(string line)
    {
        return line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries)
                   .Select(double.Parse).ToArray();
    }

    static double EstimateDeltaV(double[] from, double[] to)
    {
        double r1 = Math.Sqrt(from[0]*from[0] + from[1]*from[1] + from[2]*from[2]);
        double r2 = Math.Sqrt(to[0]*to[0] + to[1]*to[1] + to[2]*to[2]);
        double v1 = Math.Sqrt(from[3]*from[3] + from[4]*from[4] + from[5]*from[5]);
        double v2 = Math.Sqrt(to[3]*to[3] + to[4]*to[4] + to[5]*to[5]);

        double a = (r1 + r2) / 2;
        double vt1 = Math.Sqrt(mu * (2/r1 - 1/a));
        double vt2 = Math.Sqrt(mu * (2/r2 - 1/a));

        double dv1 = Math.Abs(vt1 - v1);
        double dv2 = Math.Abs(v2 - vt2);

        double dx = to[0] - from[0], dy = to[1] - from[1], dz = to[2] - from[2];
        double posDiff = Math.Sqrt(dx*dx + dy*dy + dz*dz);
        double planePenalty = posDiff * 0.0001;

        return dv1 + dv2 + planePenalty;
    }

    static int[] BruteForce(int n, double[,] cost)
    {
        var indices = Enumerable.Range(0, n).ToArray();
        int[] best = null;
        double bestCost = double.MaxValue;

        foreach (var perm in Permutations(indices))
        {
            double c = cost[0, perm[0] + 1];
            for (int i = 0; i < n - 1; i++)
                c += cost[perm[i] + 1, perm[i + 1] + 1];
            if (c < bestCost) { bestCost = c; best = perm.ToArray(); }
        }
        return best;
    }

    static IEnumerable<int[]> Permutations(int[] arr)
    {
        if (arr.Length == 1) { yield return arr; yield break; }
        for (int i = 0; i < arr.Length; i++)
        {
            var rest = arr.Where((_, idx) => idx != i).ToArray();
            foreach (var p in Permutations(rest))
                yield return new[] { arr[i] }.Concat(p).ToArray();
        }
    }

    static int[] GreedyWith2Opt(int n, double[,] cost)
    {
        var tour = new List<int>();
        var visited = new bool[n];
        int cur = 0;

        for (int i = 0; i < n; i++)
        {
            double best = double.MaxValue;
            int next = -1;
            for (int j = 0; j < n; j++)
            {
                if (visited[j]) continue;
                double c = cost[cur == 0 ? 0 : tour[tour.Count - 1] + 1, j + 1];
                if (c < best) { best = c; next = j; }
            }
            tour.Add(next);
            visited[next] = true;
            cur = next + 1;
        }

        bool improved = true;
        int iter = 0, maxIter = n > 50 ? 1000 : 5000;
        while (improved && iter++ < maxIter)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double oldC = GetEdgeCost(tour, i, cost) + GetEdgeCost(tour, j, cost);
                    Reverse(tour, i + 1, j);
                    double newC = GetEdgeCost(tour, i, cost) + GetEdgeCost(tour, j, cost);
                    if (newC < oldC - 1e-9) improved = true;
                    else Reverse(tour, i + 1, j);
                }
            }
        }
        return tour.ToArray();
    }

    static double GetEdgeCost(List<int> tour, int i, double[,] cost)
    {
        int from = i == 0 ? 0 : tour[i - 1] + 1;
        int to = tour[i] + 1;
        return cost[from, to];
    }

    static void Reverse(List<int> tour, int i, int j)
    {
        while (i < j) { int t = tour[i]; tour[i] = tour[j]; tour[j] = t; i++; j--; }
    }
}
