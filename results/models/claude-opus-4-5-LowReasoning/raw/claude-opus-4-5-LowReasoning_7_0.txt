using namespace std;

const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double t) const { return {x*t, y*t, z*t}; }
    Vec3 operator/(double t) const { return {x/t, y/t, z/t}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const { return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x}; }
    double len() const { return sqrt(x*x + y*y + z*z); }
    Vec3 norm() const { double l = len(); return l > EPS ? *this/l : Vec3(0,0,0); }
    bool operator<(const Vec3& o) const {
        if (abs(x-o.x) > EPS) return x < o.x;
        if (abs(y-o.y) > EPS) return y < o.y;
        return z < o.z - EPS;
    }
    bool operator==(const Vec3& o) const {
        return abs(x-o.x) < EPS && abs(y-o.y) < EPS && abs(z-o.z) < EPS;
    }
};

struct Triangle {
    int v[3];
    int meshId;
};

struct Mesh {
    vector<Vec3> verts;
    vector<Triangle> tris;
};

struct AABB {
    Vec3 mn, mx;
    AABB() : mn(1e18,1e18,1e18), mx(-1e18,-1e18,-1e18) {}
    void expand(const Vec3& p) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    bool intersects(const AABB& o) const {
        return mn.x <= o.mx.x + EPS && mx.x >= o.mn.x - EPS &&
               mn.y <= o.mx.y + EPS && mx.y >= o.mn.y - EPS &&
               mn.z <= o.mx.z + EPS && mx.z >= o.mn.z - EPS;
    }
};

AABB triAABB(const Vec3& a, const Vec3& b, const Vec3& c) {
    AABB box;
    box.expand(a); box.expand(b); box.expand(c);
    return box;
}

int rayTriIntersect(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = dir.cross(e2);
    double a = e1.dot(h);
    if (abs(a) < EPS) return 0;
    double f = 1.0 / a;
    Vec3 s = orig - v0;
    double u = f * s.dot(h);
    if (u < -EPS || u > 1.0 + EPS) return 0;
    Vec3 q = s.cross(e1);
    double v = f * dir.dot(q);
    if (v < -EPS || u + v > 1.0 + EPS) return 0;
    double t = f * e2.dot(q);
    if (t > EPS) return (a > 0) ? 1 : -1;
    return 0;
}

bool triTriIntersect(const Vec3& a0, const Vec3& a1, const Vec3& a2,
                     const Vec3& b0, const Vec3& b1, const Vec3& b2) {
    auto testSep = [](const Vec3& n, const Vec3& a0, const Vec3& a1, const Vec3& a2,
                      const Vec3& b0, const Vec3& b1, const Vec3& b2) -> bool {
        if (n.len() < EPS) return false;
        double amin = min({n.dot(a0), n.dot(a1), n.dot(a2)});
        double amax = max({n.dot(a0), n.dot(a1), n.dot(a2)});
        double bmin = min({n.dot(b0), n.dot(b1), n.dot(b2)});
        double bmax = max({n.dot(b0), n.dot(b1), n.dot(b2)});
        return amax < bmin - EPS || bmax < amin - EPS;
    };

    Vec3 na = (a1-a0).cross(a2-a0);
    Vec3 nb = (b1-b0).cross(b2-b0);
    if (testSep(na, a0,a1,a2, b0,b1,b2)) return false;
    if (testSep(nb, a0,a1,a2, b0,b1,b2)) return false;

    Vec3 ea[3] = {a1-a0, a2-a1, a0-a2};
    Vec3 eb[3] = {b1-b0, b2-b1, b0-b2};
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++) {
            Vec3 axis = ea[i].cross(eb[j]);
            if (testSep(axis, a0,a1,a2, b0,b1,b2)) return false;
        }
    }
    return true;
}

struct BVHNode {
    AABB box;
    int left, right;
    int triIdx;
};

class BVH {
public:
    vector<BVHNode> nodes;
    const Mesh* mesh;

    void build(const Mesh* m) {
        mesh = m;
        if (m->tris.empty()) return;
        vector<int> indices(m->tris.size());
        for (int i = 0; i < (int)m->tris.size(); i++) indices[i] = i;
        buildNode(indices, 0, (int)indices.size());
    }

    int buildNode(vector<int>& indices, int start, int end) {
        BVHNode node;
        node.left = node.right = -1;
        node.triIdx = -1;

        for (int i = start; i < end; i++) {
            const auto& t = mesh->tris[indices[i]];
            node.box.expand(mesh->verts[t.v[0]]);
            node.box.expand(mesh->verts[t.v[1]]);
            node.box.expand(mesh->verts[t.v[2]]);
        }

        if (end - start <= 4) {
            node.triIdx = indices[start];
            int idx = (int)nodes.size();
            nodes.push_back(node);
            return idx;
        }

        int axis = 0;
        Vec3 ext = node.box.mx - node.box.mn;
        if (ext.y > ext.x && ext.y > ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z > ext.y) axis = 2;

        int mid = (start + end) / 2;
        nth_element(indices.begin() + start, indices.begin() + mid, indices.begin() + end,
            [&](int a, int b) {
                auto ca = (mesh->verts[mesh->tris[a].v[0]] + mesh->verts[mesh->tris[a].v[1]] + mesh->verts[mesh->tris[a].v[2]]) / 3.0;
                auto cb = (mesh->verts[mesh->tris[b].v[0]] + mesh->verts[mesh->tris[b].v[1]] + mesh->verts[mesh->tris[b].v[2]]) / 3.0;
                return (axis == 0 ? ca.x : (axis == 1 ? ca.y : ca.z)) < (axis == 0 ? cb.x : (axis == 1 ? cb.y : cb.z));
            });

        int idx = (int)nodes.size();
        nodes.push_back(node);
        nodes[idx].left = buildNode(indices, start, mid);
        nodes[idx].right = buildNode(indices, mid, end);
        return idx;
    }

    bool anyIntersection(const Mesh* other, const BVH& otherBVH) const {
        if (nodes.empty() || otherBVH.nodes.empty()) return false;
        return intersectNodes(0, otherBVH, 0, other);
    }

    bool intersectNodes(int a, const BVH& ob, int b, const Mesh* other) const {
        if (!nodes[a].box.intersects(ob.nodes[b].box)) return false;

        bool aLeaf = nodes[a].left == -1;
        bool bLeaf = ob.nodes[b].left == -1;

        if (aLeaf && bLeaf) {
            const auto& ta = mesh->tris[nodes[a].triIdx];
            const auto& tb = other->tris[ob.nodes[b].triIdx];
            return triTriIntersect(
                mesh->verts[ta.v[0]], mesh->verts[ta.v[1]], mesh->verts[ta.v[2]],
                other->verts[tb.v[0]], other->verts[tb.v[1]], other->verts[tb.v[2]]);
        }

        if (aLeaf) {
            return intersectNodes(a, ob, ob.nodes[b].left, other) ||
                   intersectNodes(a, ob, ob.nodes[b].right, other);
        }
        if (bLeaf) {
            return intersectNodes(nodes[a].left, ob, b, other) ||
                   intersectNodes(nodes[a].right, ob, b, other);
        }

        return intersectNodes(nodes[a].left, ob, ob.nodes[b].left, other) ||
               intersectNodes(nodes[a].left, ob, ob.nodes[b].right, other) ||
               intersectNodes(nodes[a].right, ob, ob.nodes[b].left, other) ||
               intersectNodes(nodes[a].right, ob, ob.nodes[b].right, other);
    }

    int countRayHits(const Vec3& orig, const Vec3& dir) const {
        if (nodes.empty()) return 0;
        return countRayHitsNode(0, orig, dir);
    }

    int countRayHitsNode(int idx, const Vec3& orig, const Vec3& dir) const {
        const auto& node = nodes[idx];

        Vec3 invDir(1.0/dir.x, 1.0/dir.y, 1.0/dir.z);
        double t1 = (node.box.mn.x - orig.x) * invDir.x;
        double t2 = (node.box.mx.x - orig.x) * invDir.x;
        double t3 = (node.box.mn.y - orig.y) * invDir.y;
        double t4 = (node.box.mx.y - orig.y) * invDir.y;
        double t5 = (node.box.mn.z - orig.z) * invDir.z;
        double t6 = (node.box.mx.z - orig.z) * invDir.z;
        double tmin = max({min(t1,t2), min(t3,t4), min(t5,t6)});
        double tmax = min({max(t1,t2), max(t3,t4), max(t5,t6)});
        if (tmax < -EPS || tmin > tmax + EPS) return 0;

        if (node.left == -1) {
            const auto& tri = mesh->tris[node.triIdx];
            int hit = rayTriIntersect(orig, dir, mesh->verts[tri.v[0]], mesh->verts[tri.v[1]], mesh->verts[tri.v[2]]);
            return (hit != 0) ? 1 : 0;
        }

        return countRayHitsNode(node.left, orig, dir) + countRayHitsNode(node.right, orig, dir);
    }
};

bool pointInsideMeshBVH(const Vec3& p, const Mesh& mesh, const BVH& bvh) {
    Vec3 dir(1.0000123, 0.0000456, 0.0000789);
    int count = bvh.countRayHits(p, dir);
    return (count % 2) == 1;
}

Mesh readMesh(istream& in, int nv, int nf, int meshId) {
    Mesh m;
    m.verts.resize(nv);
    m.tris.resize(nf);
    for (int i = 0; i < nv; i++) {
        in >> m.verts[i].x >> m.verts[i].y >> m.verts[i].z;
    }
    for (int i = 0; i < nf; i++) {
        in >> m.tris[i].v[0] >> m.tris[i].v[1] >> m.tris[i].v[2];
        m.tris[i].meshId = meshId;
    }
    return m;
}

void writeMesh(ostream& out, const Mesh& m) {
    out << m.verts.size() << " " << m.tris.size() << "\n";
    out << fixed;
    out.precision(9);
    for (const auto& v : m.verts) {
        out << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& t : m.tris) {
        out << t.v[0] << " " << t.v[1] << " " << t.v[2] << "\n";
    }
}

bool meshesIntersect(const Mesh& a, const Mesh& b, const BVH& bvhA, const BVH& bvhB) {
    return bvhA.anyIntersection(&b, bvhB);
}

bool meshContains(const Mesh& container, const BVH& bvh, const Mesh& contained) {
    if (contained.verts.empty()) return false;
    return pointInsideMeshBVH(contained.verts[0], container, bvh);
}

Mesh combineMeshes(const Mesh& a, const Mesh& b) {
    Mesh result;
    result.verts = a.verts;
    result.tris = a.tris;
    int offset = (int)a.verts.size();
    for (const auto& v : b.verts) {
        result.verts.push_back(v);
    }
    for (auto t : b.tris) {
        t.v[0] += offset;
        t.v[1] += offset;
        t.v[2] += offset;
        result.tris.push_back(t);
    }
    return result;
}

Mesh filterMeshOutside(const Mesh& src, const Mesh& other, const BVH& otherBVH) {
    Mesh result;
    unordered_map<int, int> vertMap;

    for (const auto& tri : src.tris) {
        Vec3 c = (src.verts[tri.v[0]] + src.verts[tri.v[1]] + src.verts[tri.v[2]]) / 3.0;
        if (!pointInsideMeshBVH(c, other, otherBVH)) {
            Triangle newTri;
            for (int i = 0; i < 3; i++) {
                int oldIdx = tri.v[i];
                if (vertMap.find(oldIdx) == vertMap.end()) {
                    vertMap[oldIdx] = (int)result.verts.size();
                    result.verts.push_back(src.verts[oldIdx]);
                }
                newTri.v[i] = vertMap[oldIdx];
            }
            newTri.meshId = tri.meshId;
            result.tris.push_back(newTri);
        }
    }
    return result;
}

Mesh computeUnion(Mesh& a, Mesh& b) {
    BVH bvhA, bvhB;
    bvhA.build(&a);
    bvhB.build(&b);

    if (!meshesIntersect(a, b, bvhA, bvhB)) {
        if (meshContains(a, bvhA, b)) {
            return a;
        }
        if (meshContains(b, bvhB, a)) {
            return b;
        }
        return combineMeshes(a, b);
    }

    Mesh filteredA = filterMeshOutside(a, b, bvhB);
    Mesh filteredB = filterMeshOutside(b, a, bvhA);

    return combineMeshes(filteredA, filteredB);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int va, fa, vb, fb;
    cin >> va >> fa >> vb >> fb;

    Mesh a = readMesh(cin, va, fa, 0);
    Mesh b = readMesh(cin, vb, fb, 1);

    Mesh result = computeUnion(a, b);

    writeMesh(cout, result);

    return 0;
}
