use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut vertices: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        vertices.push((parts[0], parts[1]));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let sun: Vec<f64> = sun_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let (sx, sy, sz) = (sun[0], sun[1], sun[2]);

    let (min_x, max_x, min_y, max_y) = vertices.iter().fold(
        (f64::MAX, f64::MIN, f64::MAX, f64::MIN),
        |(mnx, mxx, mny, mxy), &(x, y)| (mnx.min(x), mxx.max(x), mny.min(y), mxy.max(y)),
    );

    let width = max_x - min_x;
    let height = max_y - min_y;
    let center_x = (min_x + max_x) / 2.0;
    let center_y = (min_y + max_y) / 2.0;

    let base_shadow_radius = 1.0_f64.hypot(0.707);
    let needed_radius = (width / 2.0).hypot(height / 2.0) * 1.2;
    let single_scale = (needed_radius / base_shadow_radius).min(10.0).max(0.1);

    let sun_len = (sx * sx + sy * sy + sz * sz).sqrt();
    let (nsx, nsy, nsz) = (sx / sun_len, sy / sun_len, sz / sun_len);

    let (qw, qx, qy, qz) = if nsz.abs() > 0.999 {
        (1.0, 0.0, 0.0, 0.0)
    } else {
        let tn_len = (nsx * nsx + nsy * nsy).sqrt();
        let angle = nsz.acos();
        let axis_x = -nsy / tn_len;
        let axis_y = nsx / tn_len;
        let half = angle / 2.0;
        (half.cos(), axis_x * half.sin(), axis_y * half.sin(), 0.0)
    };

    let stdout = io::stdout();
    let mut out = stdout.lock();

    let can_cover_with_one = single_scale <= 10.0 && needed_radius <= base_shadow_radius * 10.0;

    if can_cover_with_one {
        let height_above = single_scale * 2.0;
        let offset_x = if nsz.abs() > 1e-9 { -nsx / nsz * height_above } else { 0.0 };
        let offset_y = if nsz.abs() > 1e-9 { -nsy / nsz * height_above } else { 0.0 };
        let px = center_x - offset_x;
        let py = center_y - offset_y;

        writeln!(out, "1").unwrap();
        writeln!(out, "{} {} {} {} {} {} {} {}", px, py, height_above, qw, qx, qy, qz, single_scale).unwrap();
    } else {
        let scale = 9.5;
        let cell_coverage = base_shadow_radius * scale * 0.8;
        let nx = ((width / cell_coverage).ceil() as usize).max(1);
        let ny = ((height / cell_coverage).ceil() as usize).max(1);

        let mut tetras = Vec::new();
        let step_x = if nx > 1 { width / (nx - 1) as f64 } else { 0.0 };
        let step_y = if ny > 1 { height / (ny - 1) as f64 } else { 0.0 };

        for i in 0..nx {
            for j in 0..ny {
                let cx = if nx > 1 { min_x + i as f64 * step_x } else { center_x };
                let cy = if ny > 1 { min_y + j as f64 * step_y } else { center_y };
                let z = scale * 2.0 + ((i + j) as f64) * scale * 0.6;
                let ox = if nsz.abs() > 1e-9 { -nsx / nsz * z } else { 0.0 };
                let oy = if nsz.abs() > 1e-9 { -nsy / nsz * z } else { 0.0 };
                tetras.push((cx - ox, cy - oy, z, qw, qx, qy, qz, scale));
            }
        }

        writeln!(out, "{}", tetras.len()).unwrap();
        for t in &tetras {
            writeln!(out, "{} {} {} {} {} {} {} {}", t.0, t.1, t.2, t.3, t.4, t.5, t.6, t.7).unwrap();
        }
    }
}
