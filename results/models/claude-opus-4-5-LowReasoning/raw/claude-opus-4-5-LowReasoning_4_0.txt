#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3(double x=0, double y=0, double z=0): x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x+v.x, y+v.y, z+v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x-v.x, y-v.y, z-v.z); }
    Vec3 operator*(double s) const { return Vec3(x*s, y*s, z*s); }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y*v.z-z*v.y, z*v.x-x*v.z, x*v.y-y*v.x); }
    double len() const { return sqrt(x*x+y*y+z*z); }
    Vec3 norm() const { double l=len(); return l>1e-12? Vec3(x/l,y/l,z/l): Vec3(); }
};

vector<Vec3> verts;
Vec3 minB, maxB;

bool pointInConvexHull(const Vec3& p) {
    if (p.x < minB.x || p.x > maxB.x || p.y < minB.y || p.y > maxB.y || p.z < minB.z || p.z > maxB.z)
        return false;
    int n = (int)verts.size();
    if (n < 4) return false;
    Vec3 c(0,0,0);
    for (auto& v : verts) c = c + v;
    c = c * (1.0/n);
    for (int i = 0; i < n; i++) {
        for (int j = i+1; j < n; j++) {
            for (int k = j+1; k < n; k++) {
                Vec3 a = verts[i], b = verts[j], cc = verts[k];
                Vec3 nm = (b-a).cross(cc-a);
                if (nm.len() < 1e-12) continue;
                nm = nm.norm();
                double dc = nm.dot(c-a);
                double dp = nm.dot(p-a);
                if (dc * dp < -1e-9) return false;
            }
        }
    }
    return true;
}

Vec3 tetraVerts[4];
void initTetra(double e) {
    double h = e * sqrt(2.0/3.0);
    double r = e / sqrt(3.0);
    tetraVerts[0] = Vec3(0, 0, h * 0.75);
    tetraVerts[1] = Vec3(r, 0, -h * 0.25);
    tetraVerts[2] = Vec3(-r/2, r*sqrt(3.0)/2, -h * 0.25);
    tetraVerts[3] = Vec3(-r/2, -r*sqrt(3.0)/2, -h * 0.25);
}

Vec3 rotateEuler(Vec3 p, double rx, double ry, double rz) {
    double cx=cos(rx), sx=sin(rx), cy=cos(ry), sy=sin(ry), cz=cos(rz), sz=sin(rz);
    double x1 = p.x, y1 = p.y*cx - p.z*sx, z1 = p.y*sx + p.z*cx;
    double x2 = x1*cy + z1*sy, y2 = y1, z2 = -x1*sy + z1*cy;
    return Vec3(x2*cz - y2*sz, x2*sz + y2*cz, z2);
}

bool canPlace(Vec3 center, double rx, double ry, double rz) {
    for (int i = 0; i < 4; i++) {
        Vec3 v = center + rotateEuler(tetraVerts[i], rx, ry, rz);
        if (!pointInConvexHull(v)) return false;
    }
    return true;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    cin >> V >> e;

    verts.resize(V);
    minB = Vec3(1e18, 1e18, 1e18);
    maxB = Vec3(-1e18, -1e18, -1e18);

    for (int i = 0; i < V; i++) {
        cin >> verts[i].x >> verts[i].y >> verts[i].z;
        minB.x = min(minB.x, verts[i].x);
        minB.y = min(minB.y, verts[i].y);
        minB.z = min(minB.z, verts[i].z);
        maxB.x = max(maxB.x, verts[i].x);
        maxB.y = max(maxB.y, verts[i].y);
        maxB.z = max(maxB.z, verts[i].z);
    }

    initTetra(e);

    double R = e * sqrt(3.0/8.0);
    double step = e * 0.9;

    vector<tuple<double,double,double,double,double,double>> results;

    double vol = (maxB.x-minB.x)*(maxB.y-minB.y)*(maxB.z-minB.z);
    if (vol > 1e9) step = e * 1.5;

    for (double x = minB.x + R; x <= maxB.x - R; x += step) {
        for (double y = minB.y + R; y <= maxB.y - R; y += step) {
            for (double z = minB.z + R; z <= maxB.z - R; z += step) {
                if (canPlace(Vec3(x,y,z), 0, 0, 0)) {
                    results.push_back({x, y, z, 0, 0, 0});
                }
            }
        }
    }

    cout << results.size() << "\n";
    cout.precision(10);
    for (auto& [cx,cy,cz,rx,ry,rz] : results) {
        cout << cx << " " << cy << " " << cz << " " << rx << " " << ry << " " << rz << "\n";
    }

    return 0;
}
