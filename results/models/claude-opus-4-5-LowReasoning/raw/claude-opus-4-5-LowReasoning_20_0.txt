use std::io::{self, BufRead, Write, BufWriter};
use std::collections::HashMap;

#[derive(Clone)]
struct Sample {
    hole_idx: usize,
    sample_idx: usize,
    x: f64,
    y: f64,
    z: f64,
    values: Vec<f64>,
}

struct Suspect {
    hole_id: String,
    sample_idx: usize,
    property: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    let prop_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = prop_line.split_whitespace().map(|s| s.to_string()).collect();

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut all_samples: Vec<Sample> = Vec::new();

    for hole_idx in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = hole_line.split_whitespace().collect();

        let hole_id = parts[0].to_string();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let _length: f64 = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        hole_ids.push(hole_id);

        for sample_idx in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let vals: Vec<&str> = sample_line.split_whitespace().collect();
            let depth: f64 = vals[0].parse().unwrap();

            let x = start_x + dir_x * depth;
            let y = start_y + dir_y * depth;
            let z = start_z + dir_z * depth;

            let mut values = Vec::with_capacity(num_properties);
            for i in 0..num_properties {
                let v: f64 = vals.get(i + 1).and_then(|s| s.parse().ok()).unwrap_or(f64::NAN);
                values.push(v);
            }

            all_samples.push(Sample { hole_idx, sample_idx, x, y, z, values });
        }
    }

    let total_samples = all_samples.len();
    let k = if total_samples < 100 { 5 } else if total_samples < 1000 { 10 } else { 15 };

    let mut suspects: Vec<Suspect> = Vec::new();

    // Build spatial index using grid cells
    let mut grid: HashMap<(i64, i64, i64), Vec<usize>> = HashMap::new();
    let cell_size = estimate_cell_size(&all_samples);

    for (i, s) in all_samples.iter().enumerate() {
        let cx = (s.x / cell_size).floor() as i64;
        let cy = (s.y / cell_size).floor() as i64;
        let cz = (s.z / cell_size).floor() as i64;
        grid.entry((cx, cy, cz)).or_default().push(i);
    }

    for prop_idx in 0..num_properties {
        let valid_values: Vec<f64> = all_samples.iter()
            .map(|s| s.values[prop_idx])
            .filter(|v| v.is_finite())
            .collect();

        if valid_values.len() < 10 { continue; }

        let global_mean = valid_values.iter().sum::<f64>() / valid_values.len() as f64;
        let global_var = valid_values.iter().map(|v| (v - global_mean).powi(2)).sum::<f64>() / valid_values.len() as f64;
        let global_std = global_var.sqrt().max(1e-10);

        for (i, sample) in all_samples.iter().enumerate() {
            let val = sample.values[prop_idx];
            if !val.is_finite() { continue; }

            let neighbors = find_neighbors(&all_samples, &grid, cell_size, i, k);
            if neighbors.len() < 3 { continue; }

            let neighbor_vals: Vec<f64> = neighbors.iter()
                .map(|&j| all_samples[j].values[prop_idx])
                .filter(|v| v.is_finite())
                .collect();

            if neighbor_vals.is_empty() { continue; }

            let local_mean = neighbor_vals.iter().sum::<f64>() / neighbor_vals.len() as f64;
            let local_var = neighbor_vals.iter().map(|v| (v - local_mean).powi(2)).sum::<f64>() / neighbor_vals.len() as f64;
            let local_std = local_var.sqrt().max(global_std * 0.1);

            let local_z = (val - local_mean).abs() / local_std;
            let global_z = (val - global_mean).abs() / global_std;

            let confidence = (local_z * 0.7 + global_z * 0.3) / 3.0;

            if local_z > 3.0 && confidence > 0.5 {
                suspects.push(Suspect {
                    hole_id: hole_ids[sample.hole_idx].clone(),
                    sample_idx: sample.sample_idx,
                    property: property_names[prop_idx].clone(),
                    confidence: confidence.min(1.0),
                });
            }
        }
    }

    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", suspects.len()).unwrap();
    for s in &suspects {
        writeln!(out, "{} {} {} {:.4}", s.hole_id, s.sample_idx, s.property, s.confidence).unwrap();
    }
}

fn estimate_cell_size(samples: &[Sample]) -> f64 {
    if samples.len() < 2 { return 100.0; }

    let n = samples.len().min(1000);
    let step = samples.len() / n;

    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;

    for i in (0..samples.len()).step_by(step.max(1)) {
        let s = &samples[i];
        min_x = min_x.min(s.x);
        max_x = max_x.max(s.x);
        min_y = min_y.min(s.y);
        max_y = max_y.max(s.y);
    }

    let extent = ((max_x - min_x).max(max_y - min_y)).max(1.0);
    extent / (samples.len() as f64).sqrt().max(10.0)
}

fn find_neighbors(
    samples: &[Sample],
    grid: &HashMap<(i64, i64, i64), Vec<usize>>,
    cell_size: f64,
    idx: usize,
    k: usize,
) -> Vec<usize> {
    let s = &samples[idx];
    let cx = (s.x / cell_size).floor() as i64;
    let cy = (s.y / cell_size).floor() as i64;
    let cz = (s.z / cell_size).floor() as i64;

    let mut candidates: Vec<(usize, f64)> = Vec::new();

    for dx in -2..=2 {
        for dy in -2..=2 {
            for dz in -2..=2 {
                if let Some(indices) = grid.get(&(cx + dx, cy + dy, cz + dz)) {
                    for &j in indices {
                        if j != idx {
                            let o = &samples[j];
                            let dist = (s.x - o.x).powi(2) + (s.y - o.y).powi(2) + (s.z - o.z).powi(2);
                            candidates.push((j, dist));
                        }
                    }
                }
            }
        }
    }

    candidates.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
    candidates.truncate(k);
    candidates.into_iter().map(|(j, _)| j).collect()
}
