#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int,int>>> jobs(J);

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].first >> jobs[j][t].second;
        }
    }

    vector<vector<int>> startTimes(J);
    for (int j = 0; j < J; j++) {
        startTimes[j].resize(jobs[j].size());
    }

    vector<int> machineEnd(M, 0);
    vector<int> jobEnd(J, 0);
    vector<int> taskIdx(J, 0);

    auto cmp = [](pair<int,int>& a, pair<int,int>& b) { return a.first > b.first; };
    priority_queue<pair<int,int>, vector<pair<int,int>>, decltype(cmp)> pq(cmp);

    for (int j = 0; j < J; j++) {
        if (!jobs[j].empty()) {
            pq.push({0, j});
        }
    }

    int scheduled = 0;
    int totalTasks = 0;
    for (int j = 0; j < J; j++) totalTasks += (int)jobs[j].size();

    while (scheduled < totalTasks) {
        vector<pair<int,int>> pending;
        int bestJob = -1;
        int bestStart = INT_MAX;

        while (!pq.empty()) {
            auto [est, j] = pq.top();
            pq.pop();

            int t = taskIdx[j];
            if (t >= (int)jobs[j].size()) continue;

            int m = jobs[j][t].first;
            int actualStart = max(jobEnd[j], machineEnd[m]);

            if (actualStart < bestStart) {
                if (bestJob != -1) pending.push_back({bestStart, bestJob});
                bestStart = actualStart;
                bestJob = j;
            } else {
                pending.push_back({actualStart, j});
            }
        }

        if (bestJob != -1) {
            int j = bestJob;
            int t = taskIdx[j];
            int m = jobs[j][t].first;
            int dur = jobs[j][t].second;

            startTimes[j][t] = bestStart;
            machineEnd[m] = bestStart + dur;
            jobEnd[j] = bestStart + dur;
            taskIdx[j]++;
            scheduled++;

            if (taskIdx[j] < (int)jobs[j].size()) {
                pending.push_back({jobEnd[j], j});
            }
        }

        for (auto& p : pending) {
            pq.push(p);
        }
    }

    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)jobs[j].size(); t++) {
            if (t > 0) cout << ' ';
            cout << startTimes[j][t] << ' ' << jobs[j][t].first;
        }
        cout << '\n';
    }

    return 0;
}
