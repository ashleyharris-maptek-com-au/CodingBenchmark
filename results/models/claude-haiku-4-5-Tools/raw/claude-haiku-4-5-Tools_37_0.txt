using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        var adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adj[u].Add(v);
        }

        var fvs = FindMinimumFVS(n, adj);

        Console.WriteLine(fvs.Count);
        if (fvs.Count > 0)
            Console.WriteLine(string.Join(" ", fvs));
    }

    static List<int> FindMinimumFVS(int n, List<int>[] adj)
    {
        // Check if already acyclic
        if (IsAcyclic(n, adj, new HashSet<int>()))
            return new List<int>();

        // Find vertices involved in cycles for faster search
        var candidates = FindCyclicVertices(n, adj);
        if (candidates.Count == 0)
            return new List<int>();

        // Try removing k vertices for k = 1, 2, 3, ... until we find a solution
        for (int k = 1; k <= candidates.Count; k++)
        {
            var result = TryRemoveKVertices(candidates.ToList(), k, adj, n);
            if (result != null)
                return result;
        }

        return new List<int>();
    }

    static HashSet<int> FindCyclicVertices(int n, List<int>[] adj)
    {
        var cyclic = new HashSet<int>();
        var visited = new bool[n];
        var inStack = new bool[n];

        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
                DFSForCycles(i, adj, visited, inStack, cyclic);
        }

        return cyclic;
    }

    static void DFSForCycles(int u, List<int>[] adj, bool[] visited, bool[] inStack, HashSet<int> cyclic)
    {
        visited[u] = true;
        inStack[u] = true;

        foreach (int v in adj[u])
        {
            if (!visited[v])
            {
                DFSForCycles(v, adj, visited, inStack, cyclic);
            }
            else if (inStack[v])
            {
                cyclic.Add(u);
                cyclic.Add(v);
            }
        }

        inStack[u] = false;
    }

    static List<int> TryRemoveKVertices(List<int> candidates, int k, List<int>[] adj, int n)
    {
        // Sort candidates by degree (heuristic: remove high-degree vertices first)
        var sorted = candidates.OrderByDescending(v => adj[v].Count).ToList();

        return TryCombinations(sorted, k, 0, new List<int>(), adj, n);
    }

    static List<int> TryCombinations(List<int> candidates, int k, int start, List<int> current, List<int>[] adj, int n)
    {
        if (current.Count == k)
        {
            var removed = new HashSet<int>(current);
            if (IsAcyclic(n, adj, removed))
                return new List<int>(current);
            return null;
        }

        for (int i = start; i < candidates.Count; i++)
        {
            current.Add(candidates[i]);
            var result = TryCombinations(candidates, k, i + 1, current, adj, n);
            if (result != null)
                return result;
            current.RemoveAt(current.Count - 1);
        }

        return null;
    }

    static bool IsAcyclic(int n, List<int>[] adj, HashSet<int> removed)
    {
        var visited = new bool[n];
        var inStack = new bool[n];

        for (int i = 0; i < n; i++)
        {
            if (!removed.Contains(i) && !visited[i])
            {
                if (HasCycleDFS(i, adj, visited, inStack, removed))
                    return false;
            }
        }

        return true;
    }

    static bool HasCycleDFS(int u, List<int>[] adj, bool[] visited, bool[] inStack, HashSet<int> removed)
    {
        visited[u] = true;
        inStack[u] = true;

        foreach (int v in adj[u])
        {
            if (removed.Contains(v))
                continue;

            if (!visited[v])
            {
                if (HasCycleDFS(v, adj, visited, inStack, removed))
                    return true;
            }
            else if (inStack[v])
                return true;
        }

        inStack[u] = false;
        return false;
    }
}
