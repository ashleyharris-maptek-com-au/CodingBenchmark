use std::io::{self, Read, Write, BufReader, BufWriter};

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    // Read num_points and num_clusters as binary u32
    let mut header = [0u8; 8];
    reader.read_exact(&mut header)?;

    let num_points = u32::from_le_bytes([header[0], header[1], header[2], header[3]]) as usize;
    let num_clusters = u32::from_le_bytes([header[4], header[5], header[6], header[7]]) as usize;

    // Adaptive algorithm selection based on dataset size
    if num_points <= 10_000_000 {
        kmeans_full(&mut reader, &mut writer, num_points, num_clusters)?;
    } else if num_points <= 200_000_000 {
        kmeans_minibatch(&mut reader, &mut writer, num_points, num_clusters)?;
    } else {
        kmeans_streaming(&mut reader, &mut writer, num_points, num_clusters)?;
    }

    writer.flush()?;
    Ok(())
}

fn kmeans_full(
    reader: &mut BufReader<io::StdinLock>,
    writer: &mut BufWriter<io::StdoutLock>,
    num_points: usize,
    num_clusters: usize,
) -> io::Result<()> {
    // Load all points into memory
    let mut points = Vec::with_capacity(num_points * 3);
    let mut buf = [0u8; 12];

    for _ in 0..num_points {
        reader.read_exact(&mut buf)?;
        points.extend_from_slice(&buf);
    }

    // Initialize centroids from random points
    let mut centroids = vec![[0f32; 3]; num_clusters];
    let mut rng = 12345u64;
    for i in 0..num_clusters {
        rng = rng.wrapping_mul(1103515245).wrapping_add(12345);
        let idx = ((rng / 65536) as usize % num_points) * 12;
        centroids[i][0] = f32::from_le_bytes([points[idx], points[idx + 1], points[idx + 2], points[idx + 3]]);
        centroids[i][1] = f32::from_le_bytes([points[idx + 4], points[idx + 5], points[idx + 6], points[idx + 7]]);
        centroids[i][2] = f32::from_le_bytes([points[idx + 8], points[idx + 9], points[idx + 10], points[idx + 11]]);
    }

    let mut assignments = vec![0u32; num_points];

    // K-means iterations
    for _ in 0..20 {
        // Assign points to nearest centroid
        for i in 0..num_points {
            let idx = i * 12;
            let px = f32::from_le_bytes([points[idx], points[idx + 1], points[idx + 2], points[idx + 3]]);
            let py = f32::from_le_bytes([points[idx + 4], points[idx + 5], points[idx + 6], points[idx + 7]]);
            let pz = f32::from_le_bytes([points[idx + 8], points[idx + 9], points[idx + 10], points[idx + 11]]);

            let mut best_dist = f32::MAX;
            let mut best_cluster = 0u32;

            for (c, centroid) in centroids.iter().enumerate() {
                let dx = px - centroid[0];
                let dy = py - centroid[1];
                let dz = pz - centroid[2];
                let dist = dx * dx + dy * dy + dz * dz;
                if dist < best_dist {
                    best_dist = dist;
                    best_cluster = c as u32;
                }
            }
            assignments[i] = best_cluster;
        }

        // Update centroids
        let mut counts = vec![0u32; num_clusters];
        let mut sums = vec![[0f32; 3]; num_clusters];

        for i in 0..num_points {
            let idx = i * 12;
            let c = assignments[i] as usize;
            sums[c][0] += f32::from_le_bytes([points[idx], points[idx + 1], points[idx + 2], points[idx + 3]]);
            sums[c][1] += f32::from_le_bytes([points[idx + 4], points[idx + 5], points[idx + 6], points[idx + 7]]);
            sums[c][2] += f32::from_le_bytes([points[idx + 8], points[idx + 9], points[idx + 10], points[idx + 11]]);
            counts[c] += 1;
        }

        for c in 0..num_clusters {
            if counts[c] > 0 {
                centroids[c][0] = sums[c][0] / counts[c] as f32;
                centroids[c][1] = sums[c][1] / counts[c] as f32;
                centroids[c][2] = sums[c][2] / counts[c] as f32;
            }
        }
    }

    // Write output
    for assignment in assignments {
        writer.write_all(&assignment.to_le_bytes())?;
    }

    Ok(())
}

fn kmeans_minibatch(
    reader: &mut BufReader<io::StdinLock>,
    writer: &mut BufWriter<io::StdoutLock>,
    num_points: usize,
    num_clusters: usize,
) -> io::Result<()> {
    let batch_size = 1_000_000;
    let mut centroids = vec![[0f32; 3]; num_clusters];
    let mut centroids_count = vec![0u32; num_clusters];
    let mut all_assignments = Vec::with_capacity(num_points);
    let mut remaining = num_points;

    while remaining > 0 {
        let current_batch = std::cmp::min(batch_size, remaining);
        let mut buf = [0u8; 12];

        for _ in 0..current_batch {
            reader.read_exact(&mut buf)?;

            let px = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
            let py = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
            let pz = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);

            let mut best_dist = f32::MAX;
            let mut best_cluster = 0u32;

            for (c, centroid) in centroids.iter().enumerate() {
                let dx = px - centroid[0];
                let dy = py - centroid[1];
                let dz = pz - centroid[2];
                let dist = dx * dx + dy * dy + dz * dz;
                if dist < best_dist {
                    best_dist = dist;
                    best_cluster = c as u32;
                }
            }

            all_assignments.push(best_cluster);

            // Update centroid incrementally
            let c = best_cluster as usize;
            let count = centroids_count[c];
            if count == 0 {
                centroids[c] = [px, py, pz];
            } else {
                centroids[c][0] = (centroids[c][0] * count as f32 + px) / (count as f32 + 1.0);
                centroids[c][1] = (centroids[c][1] * count as f32 + py) / (count as f32 + 1.0);
                centroids[c][2] = (centroids[c][2] * count as f32 + pz) / (count as f32 + 1.0);
            }
            centroids_count[c] += 1;
        }

        remaining -= current_batch;
    }

    // Write output
    for assignment in all_assignments {
        writer.write_all(&assignment.to_le_bytes())?;
    }

    Ok(())
}

fn kmeans_streaming(
    reader: &mut BufReader<io::StdinLock>,
    writer: &mut BufWriter<io::StdoutLock>,
    num_points: usize,
    num_clusters: usize,
) -> io::Result<()> {
    let mut centroids = vec![[0f32; 3]; num_clusters];
    let mut centroids_count = vec![0u32; num_clusters];
    let mut output_buffer = Vec::with_capacity(500_000);
    let mut buf = [0u8; 12];

    for _ in 0..num_points {
        reader.read_exact(&mut buf)?;

        let px = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
        let py = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
        let pz = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);

        let mut best_dist = f32::MAX;
        let mut best_cluster = 0u32;

        for (c, centroid) in centroids.iter().enumerate() {
            let dx = px - centroid[0];
            let dy = py - centroid[1];
            let dz = pz - centroid[2];
            let dist = dx * dx + dy * dy + dz * dz;
            if dist < best_dist {
                best_dist = dist;
                best_cluster = c as u32;
            }
        }

        let c = best_cluster as usize;
        let count = centroids_count[c];
        if count == 0 {
            centroids[c] = [px, py, pz];
        } else {
            centroids[c][0] = (centroids[c][0] * count as f32 + px) / (count as f32 + 1.0);
            centroids[c][1] = (centroids[c][1] * count as f32 + py) / (count as f32 + 1.0);
            centroids[c][2] = (centroids[c][2] * count as f32 + pz) / (count as f32 + 1.0);
        }
        centroids_count[c] += 1;

        output_buffer.push(best_cluster);

        if output_buffer.len() >= 500_000 {
            for assignment in &output_buffer {
                writer.write_all(&assignment.to_le_bytes())?;
            }
            output_buffer.clear();
        }
    }

    // Write remaining buffer
    for assignment in &output_buffer {
        writer.write_all(&assignment.to_le_bytes())?;
    }

    Ok(())
}
