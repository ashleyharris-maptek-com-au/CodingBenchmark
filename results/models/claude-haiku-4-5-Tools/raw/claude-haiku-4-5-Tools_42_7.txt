OpCapability Geometry
               OpCapability Float64
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %outColor %gl_out %inColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in BuiltIn SampleId
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out BuiltIn SampleId
               OpDecorate %outColor Location 0
               OpDecorate %inColor Location 2

       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
    %uint_3 = OpConstant %uint 3
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
 %float_0_08 = OpConstant %float 0.08
  %float_0 = OpConstant %float 0.0
  %float_1 = OpConstant %float 1.0
  %float_3 = OpConstant %float 3.0

%gl_PerVertex_in = OpTypeStruct %v4float
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%arr_color_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %inColor = OpVariable %ptr_in_arr_color Input

%ptr_in_v4f = OpTypePointer Input %v4float
%ptr_in_v3f = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v4f = OpTypePointer Output %v4float
%ptr_out_v3f = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3f Output

       %main = OpFunction %void None %func
      %entry = OpLabel

      ; Read the 3 vertex positions from gl_in array
      %p0_ptr = OpAccessChain %ptr_in_v4f %gl_in %int_0 %int_0
      %p0 = OpLoad %v4float %p0_ptr

      %p1_ptr = OpAccessChain %ptr_in_v4f %gl_in %int_1 %int_0
      %p1 = OpLoad %v4float %p1_ptr

      %p2_ptr = OpAccessChain %ptr_in_v4f %gl_in %int_2 %int_0
      %p2 = OpLoad %v4float %p2_ptr

      ; Read the 3 vertex colors from inColor array
      %c0_ptr = OpAccessChain %ptr_in_v3f %inColor %int_0
      %c0 = OpLoad %v3float %c0_ptr

      %c1_ptr = OpAccessChain %ptr_in_v3f %inColor %int_1
      %c1 = OpLoad %v3float %c1_ptr

      %c2_ptr = OpAccessChain %ptr_in_v3f %inColor %int_2
      %c2 = OpLoad %v3float %c2_ptr

      ; Extract xyz components as v3float
      %p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
      %p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
      %p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

      ; Compute e1 = p1 - p0
      %e1 = OpFSub %v3float %p1_xyz %p0_xyz

      ; Compute e2 = p2 - p0
      %e2 = OpFSub %v3float %p2_xyz %p0_xyz

      ; Compute face normal: N = normalize(cross(e1, e2))
      %cross_prod = OpExtInst %v3float %1 Cross %e1 %e2
      %normal = OpExtInst %v3float %1 Normalize %cross_prod

      ; Compute offset = normal * 0.08
      %offset = OpVectorTimesScalar %v3float %normal %float_0_08

      ; Emit vertex 0
      %v0_pos = OpFAdd %v3float %p0_xyz %offset
      %v0_pos_ext = OpCompositeConstruct %v4float %v0_pos %float_1
      %gl_out_ptr_v0 = OpAccessChain %ptr_out_v4f %gl_out %int_0
      OpStore %gl_out_ptr_v0 %v0_pos_ext
      OpStore %outColor %c0
      OpEmitVertex

      ; Emit vertex 1
      %v1_pos = OpFAdd %v3float %p1_xyz %offset
      %v1_pos_ext = OpCompositeConstruct %v4float %v1_pos %float_1
      %gl_out_ptr_v1 = OpAccessChain %ptr_out_v4f %gl_out %int_0
      OpStore %gl_out_ptr_v1 %v1_pos_ext
      OpStore %outColor %c1
      OpEmitVertex

      ; Emit vertex 2
      %v2_pos = OpFAdd %v3float %p2_xyz %offset
      %v2_pos_ext = OpCompositeConstruct %v4float %v2_pos %float_1
      %gl_out_ptr_v2 = OpAccessChain %ptr_out_v4f %gl_out %int_0
      OpStore %gl_out_ptr_v2 %v2_pos_ext
      OpStore %outColor %c2
      OpEmitVertex

      OpEndPrimitive
      OpReturn
               OpFunctionEnd
