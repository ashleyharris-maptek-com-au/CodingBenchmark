using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static void Main()
    {
        string[] line1 = Console.ReadLine().Split();
        int N = int.Parse(line1[0]);
        int M = int.Parse(line1[1]);

        List<(int to, int weight)>[] adj = new List<(int, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int)>();

        for (int i = 0; i < M; i++)
        {
            string[] parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            adj[u].Add((v, w));
            adj[v].Add((u, w));
        }

        // Find vertices with odd degree
        var oddVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (adj[i].Count % 2 == 1)
                oddVertices.Add(i);
        }

        // Handle odd vertices by matching them optimally
        if (oddVertices.Count > 0)
        {
            int k = oddVertices.Count;
            long[][] dist = new long[k][];
            int[][][] next = new int[k][][];

            // Compute shortest paths between all odd vertices
            for (int i = 0; i < k; i++)
            {
                var (d, parent) = Dijkstra(adj, N, oddVertices[i]);
                dist[i] = new long[k];
                next[i] = new int[k][];

                for (int j = 0; j < k; j++)
                {
                    dist[i][j] = d[oddVertices[j]];
                    next[i][j] = ReconstructPath(parent, oddVertices[i], oddVertices[j]).ToArray();
                }
            }

            // Find minimum weight perfect matching
            var matching = FindMatching(dist);

            // Add edges from matching to make graph Eulerian
            for (int i = 0; i < matching.Count; i++)
            {
                int u = matching[i].u;
                int v = matching[i].v;
                var path = next[u][v];

                for (int j = 0; j < path.Length - 1; j++)
                {
                    int a = path[j];
                    int b = path[j + 1];

                    int weight = 0;
                    foreach (var (to, w) in adj[a])
                    {
                        if (to == b)
                        {
                            weight = w;
                            break;
                        }
                    }

                    adj[a].Add((b, weight));
                    adj[b].Add((a, weight));
                }
            }
        }

        // Find Eulerian circuit using Hierholzer's algorithm
        var circuit = EulerianCircuit(adj, N);

        Console.WriteLine(circuit.Count + " " + string.Join(" ", circuit));
    }

    static (long[], int[]) Dijkstra(List<(int, int)>[] adj, int N, int src)
    {
        long[] dist = new long[N];
        int[] parent = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(parent, -1);
        dist[src] = 0;

        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);

        while (pq.Count > 0)
        {
            int u = pq.Dequeue();

            foreach (var (v, w) in adj[u])
            {
                if (dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    parent[v] = u;
                    pq.Enqueue(v, dist[v]);
                }
            }
        }

        return (dist, parent);
    }

    static List<int> ReconstructPath(int[] parent, int src, int dst)
    {
        var path = new List<int>();
        int current = dst;
        while (current != -1)
        {
            path.Add(current);
            current = parent[current];
        }
        path.Reverse();
        return path;
    }

    static List<(int u, int v)> FindMatching(long[][] dist)
    {
        int n = dist.Length;
        var result = new List<(int, int)>();
        var used = new bool[n];

        if (n <= 24)
        {
            // Use bitmask DP for optimal matching
            return MinWeightMatchingDP(dist);
        }
        else
        {
            // Use greedy for large odd sets
            return GreedyMatching(dist);
        }
    }

    static List<(int, int)> MinWeightMatchingDP(long[][] dist)
    {
        int n = dist.Length;
        long[] dp = new long[1 << n];
        int[] match_with = new int[1 << n];
        Array.Fill(dp, long.MaxValue);
        Array.Fill(match_with, -1);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << n); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;

            int first = -1;
            for (int i = 0; i < n; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    first = i;
                    break;
                }
            }

            if (first == -1) continue;

            for (int second = first + 1; second < n; second++)
            {
                if ((mask & (1 << second)) == 0)
                {
                    int newMask = mask | (1 << first) | (1 << second);
                    long cost = dp[mask] + dist[first][second];
                    if (cost < dp[newMask])
                    {
                        dp[newMask] = cost;
                        match_with[newMask] = second;
                    }
                }
            }
        }

        var result = new List<(int, int)>();
        int fullMask = (1 << n) - 1;
        int currentMask = fullMask;

        while (currentMask > 0)
        {
            int second = match_with[currentMask];
            int first = -1;
            for (int i = 0; i < second; i++)
            {
                if ((currentMask & (1 << i)) != 0)
                {
                    if (dp[currentMask ^ (1 << i) ^ (1 << second)] + dist[i][second] == dp[currentMask])
                    {
                        first = i;
                        break;
                    }
                }
            }

            result.Add((first, second));
            currentMask ^= (1 << first) | (1 << second);
        }

        return result;
    }

    static List<(int, int)> GreedyMatching(long[][] dist)
    {
        int n = dist.Length;
        var result = new List<(int, int)>();
        var used = new bool[n];

        var edges = new List<(long w, int u, int v)>();
        for (int i = 0; i < n; i++)
            for (int j = i + 1; j < n; j++)
                edges.Add((dist[i][j], i, j));

        edges.Sort();

        foreach (var (w, u, v) in edges)
        {
            if (!used[u] && !used[v])
            {
                result.Add((u, v));
                used[u] = used[v] = true;
            }
        }

        return result;
    }

    static List<int> EulerianCircuit(List<(int, int)>[] adj, int N)
    {
        var result = new List<int>();
        var stack = new Stack<int>();
        var curr_adj = new List<(int, int)>[N];
        for (int i = 0; i < N; i++) curr_adj[i] = new List<(int, int)>(adj[i]);

        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (curr_adj[u].Count > 0)
            {
                var (v, w) = curr_adj[u][curr_adj[u].Count - 1];
                curr_adj[u].RemoveAt(curr_adj[u].Count - 1);
                stack.Push(v);
            }
            else
            {
                result.Add(u);
                stack.Pop();
            }
        }

        result.Reverse();
        return result;
    }
}
