const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

const gridArray = grid.map(row => row.split(''));

function simulateFire(gridArray, ignitions, cutCells, width, height) {
  const burned = new Set();
  const queue = [];
  const cut = new Set(cutCells.map(c => `${c[0]},${c[1]}`));

  for (const [x, y] of ignitions) {
    queue.push([x, y]);
    burned.add(`${x},${y}`);
  }

  while (queue.length > 0) {
    const [x, y] = queue.shift();

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
        const key = `${nx},${ny}`;

        if (!burned.has(key) && gridArray[ny][nx] === '.' && !cut.has(key)) {
          burned.add(key);
          queue.push([nx, ny]);
        }
      }
    }
  }

  return burned;
}

function calculateTownValueBurned(towns, burned) {
  let value = 0;
  for (const town of towns) {
    if (burned.has(`${town.x},${town.y}`)) {
      value += town.value;
    }
  }
  return value;
}

function getBoundary(burned, gridArray, width, height) {
  const boundary = new Set();

  for (const key of burned) {
    const [x, y] = key.split(',').map(Number);

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx >= 0 && nx < width && ny >= 0 && ny < height && gridArray[ny][nx] === '.') {
        if (!burned.has(`${nx},${ny}`)) {
          boundary.add(`${nx},${ny}`);
        }
      }
    }
  }

  return boundary;
}

let selectedCuts = [];
let currentBurned = simulateFire(gridArray, ignitions, [], width, height);
let valueBurnedNow = calculateTownValueBurned(towns, currentBurned);

if (valueBurnedNow === 0) {
  console.log(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

for (let i = 0; i < cutBudget; i++) {
  const boundary = getBoundary(currentBurned, gridArray, width, height);
  const candidates = Array.from(boundary).map(k => k.split(',').map(Number));

  if (candidates.length === 0) break;

  let bestCell = null;
  let bestImprovement = 0;

  for (const cell of candidates) {
    const key = `${cell[0]},${cell[1]}`;
    if (selectedCuts.some(c => `${c[0]},${c[1]}` === key)) continue;

    const testCuts = [...selectedCuts, cell];
    const burned = simulateFire(gridArray, ignitions, testCuts, width, height);
    const burnedValue = calculateTownValueBurned(towns, burned);
    const improvement = valueBurnedNow - burnedValue;

    if (improvement > bestImprovement) {
      bestImprovement = improvement;
      bestCell = cell;
    }
  }

  if (bestCell && bestImprovement > 0) {
    selectedCuts.push(bestCell);

    const burned = simulateFire(gridArray, ignitions, selectedCuts, width, height);
    valueBurnedNow = calculateTownValueBurned(towns, burned);
    currentBurned = burned;
  } else {
    break;
  }
}

console.log(JSON.stringify({ firebreaks: selectedCuts }));
