#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <cstring>

using namespace std;

struct Edge {
    int to, rev;
    long long cap;
};

class MaxFlow {
public:
    vector<vector<Edge>> graph;
    vector<int> level, iter;
    int n;

    MaxFlow(int n) : n(n), graph(n), level(n), iter(n) {}

    void addEdge(int from, int to, long long cap) {
        graph[from].push_back({to, (int)graph[to].size(), cap});
        graph[to].push_back({from, (int)graph[from].size() - 1, 0});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (e.cap > 0 && level[e.to] < 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }

        return level[t] >= 0;
    }

    long long dfs(int v, int t, long long f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (e.cap > 0 && level[v] < level[e.to]) {
                long long d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    graph[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }

    long long maxFlow(int s, int t) {
        long long flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            long long f;
            while ((f = dfs(s, t, LLONG_MAX)) > 0) {
                flow += f;
            }
        }
        return flow;
    }

    vector<pair<int, int>> getCut(int s, int t, vector<pair<int, int>>& originalEdges) {
        vector<pair<int, int>> cut;
        for (auto& edge : originalEdges) {
            int u = edge.first;
            int v = edge.second;
            for (auto& e : graph[u]) {
                if (e.to == v && e.cap == 0) {
                    cut.push_back({u, v});
                    break;
                }
            }
        }
        return cut;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<pair<int, int>> edges;
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        edges.push_back({u, v});
    }

    if (n <= 1 || m == 0) {
        cout << 0 << "\n";
        return 0;
    }

    long long minFlow = LLONG_MAX;
    int bestSource = 0, bestSink = 1;

    for (int source = 0; source < n; source++) {
        for (int sink = source + 1; sink < n; sink++) {
            MaxFlow mf(n);
            for (auto& e : edges) {
                mf.addEdge(e.first, e.second, 1);
                mf.addEdge(e.second, e.first, 1);
            }

            long long flow = mf.maxFlow(source, sink);
            if (flow < minFlow) {
                minFlow = flow;
                bestSource = source;
                bestSink = sink;
            }

            if (minFlow == 1) break;
        }
        if (minFlow == 1) break;
    }

    MaxFlow finalMf(n);
    for (auto& e : edges) {
        finalMf.addEdge(e.first, e.second, 1);
        finalMf.addEdge(e.second, e.first, 1);
    }

    finalMf.maxFlow(bestSource, bestSink);
    vector<pair<int, int>> cut = finalMf.getCut(bestSource, bestSink, edges);

    cout << cut.size() << "\n";
    for (auto& e : cut) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
