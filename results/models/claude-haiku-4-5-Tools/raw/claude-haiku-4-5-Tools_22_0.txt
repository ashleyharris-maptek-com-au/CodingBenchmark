using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    private int boardWidth;
    private int boardHeight;
    private bool[,] board;
    private int[] columnHeights;

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int width = int.Parse(firstLine[0]);
        int height = int.Parse(firstLine[1]);
        int numPieces = int.Parse(firstLine[2]);

        var player = new TetrisPlayer(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine().Trim();
            var (bestRotation, bestColumn) = player.FindOptimalPlacement(pieceType);
            Console.WriteLine($"{bestRotation} {bestColumn}");
            Console.Out.Flush();

            string result = Console.ReadLine().Trim();
            player.PlacePieceAndUpdate(pieceType, bestRotation, bestColumn, result);
        }
    }

    public TetrisPlayer(int width, int height)
    {
        boardWidth = width;
        boardHeight = height;
        board = new bool[height, width];
        columnHeights = new int[width];
    }

    private (int rotation, int column) FindOptimalPlacement(string pieceType)
    {
        double bestScore = double.MinValue;
        int bestRotation = 0;
        int bestColumn = 0;

        var rotations = GetPieceRotations(pieceType);

        for (int rot = 0; rot < rotations.Count; rot++)
        {
            var shape = rotations[rot];
            int pieceWidth = GetPieceWidth(shape);

            for (int col = 0; col <= boardWidth - pieceWidth; col++)
            {
                if (CanPlace(shape, col))
                {
                    double score = EvaluatePlacement(shape, col);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestRotation = rot;
                        bestColumn = col;
                    }
                }
            }
        }

        return (bestRotation, bestColumn);
    }

    private bool CanPlace(List<(int r, int c)> shape, int col)
    {
        int minCol = shape.Min(x => x.c);
        int maxCol = shape.Max(x => x.c);

        if (col + minCol < 0 || col + maxCol >= boardWidth)
            return false;

        foreach (var (r, c) in shape)
        {
            int boardCol = col + c;
            if (boardCol < 0 || boardCol >= boardWidth)
                return false;
        }

        return true;
    }

    private int GetLandingHeight(List<(int r, int c)> shape, int col)
    {
        int minCol = shape.Min(x => x.c);
        int maxRow = shape.Max(x => x.r);
        int landingHeight = 0;

        foreach (var (r, c) in shape)
        {
            int boardCol = col + c - minCol;
            landingHeight = Math.Max(landingHeight, columnHeights[boardCol] + (maxRow - r + 1));
        }

        return landingHeight;
    }

    private double EvaluatePlacement(List<(int r, int c)> shape, int col)
    {
        int minCol = shape.Min(x => x.c);
        int maxRow = shape.Max(x => x.r);
        int landingHeight = GetLandingHeight(shape, col);

        if (landingHeight > boardHeight)
            return double.MinValue;

        // Simulate placement
        var tempHeights = (int[])columnHeights.Clone();

        foreach (var (r, c) in shape)
        {
            int boardCol = col + c - minCol;
            int newHeight = landingHeight - (maxRow - r);
            tempHeights[boardCol] = Math.Max(tempHeights[boardCol], newHeight);
        }

        // Score: lower maximum height is better, smoother surface is better
        double score = -tempHeights.Max() * 2.0;

        // Penalize height variance (uneven surface)
        double avgHeight = tempHeights.Average();
        double variance = tempHeights.Sum(h => Math.Abs(h - avgHeight));
        score -= variance * 0.5;

        // Bonus for placing in middle columns (more flexibility)
        double centerDistance = Math.Abs((double)(col + minCol + GetPieceWidth(shape) / 2) - boardWidth / 2.0);
        score += (boardWidth / 2.0 - centerDistance) * 0.1;

        return score;
    }

    private int GetPieceWidth(List<(int r, int c)> shape)
    {
        return shape.Max(x => x.c) - shape.Min(x => x.c) + 1;
    }

    private void PlacePieceAndUpdate(string pieceType, int rotation, int column, string result)
    {
        var rotations = GetPieceRotations(pieceType);
        var shape = rotations[rotation];
        int minCol = shape.Min(x => x.c);
        int maxRow = shape.Max(x => x.r);
        int landingHeight = GetLandingHeight(shape, column);

        // Place the piece on the board
        foreach (var (r, c) in shape)
        {
            int boardCol = column + c - minCol;
            int boardRow = boardHeight - (landingHeight - (maxRow - r));
            if (boardRow >= 0 && boardRow < boardHeight && boardCol >= 0 && boardCol < boardWidth)
                board[boardRow, boardCol] = true;
        }

        // Update column heights
        for (int col = 0; col < boardWidth; col++)
        {
            columnHeights[col] = 0;
            for (int row = boardHeight - 1; row >= 0; row--)
            {
                if (board[row, col])
                {
                    columnHeights[col] = boardHeight - row;
                    break;
                }
            }
        }

        // Handle line clears if result indicates them
        if (result.StartsWith("cleared_"))
        {
            int linesCleared = int.Parse(result.Substring("cleared_".Length));
            ClearCompletedLines(linesCleared);
        }
    }

    private void ClearCompletedLines(int expectedClears)
    {
        var rowsToClear = new List<int>();

        for (int row = 0; row < boardHeight; row++)
        {
            bool complete = true;
            for (int col = 0; col < boardWidth; col++)
            {
                if (!board[row, col])
                {
                    complete = false;
                    break;
                }
            }
            if (complete)
                rowsToClear.Add(row);
        }

        // Remove cleared rows from top to bottom
        foreach (var row in rowsToClear.OrderByDescending(x => x))
        {
            for (int r = row; r < boardHeight - 1; r++)
            {
                for (int col = 0; col < boardWidth; col++)
                {
                    board[r, col] = board[r + 1, col];
                }
            }
            for (int col = 0; col < boardWidth; col++)
            {
                board[boardHeight - 1, col] = false;
            }
        }

        // Recalculate heights
        for (int col = 0; col < boardWidth; col++)
        {
            columnHeights[col] = 0;
            for (int row = boardHeight - 1; row >= 0; row--)
            {
                if (board[row, col])
                {
                    columnHeights[col] = boardHeight - row;
                    break;
                }
            }
        }
    }

    private List<List<(int r, int c)>> GetPieceRotations(string type)
    {
        return type switch
        {
            "I" => new List<List<(int, int)>>
            {
                new() { (0, 0), (0, 1), (0, 2), (0, 3) },
                new() { (0, 0), (1, 0), (2, 0), (3, 0) },
                new() { (0, 0), (0, 1), (0, 2), (0, 3) },
                new() { (0, 0), (1, 0), (2, 0), (3, 0) }
            },
            "O" => new List<List<(int, int)>>
            {
                new() { (0, 0), (0, 1), (1, 0), (1, 1) },
                new() { (0, 0), (0, 1), (1, 0), (1, 1) },
                new() { (0, 0), (0, 1), (1, 0), (1, 1) },
                new() { (0, 0), (0, 1), (1, 0), (1, 1) }
            },
            "T" => new List<List<(int, int)>>
            {
                new() { (0, 1), (1, 0), (1, 1), (1, 2) },
                new() { (0, 0), (1, 0), (1, 1), (2, 0) },
                new() { (1, 0), (1, 1), (1, 2), (2, 1) },
                new() { (0, 1), (1, 1), (2, 0), (2, 1) }
            },
            "S" => new List<List<(int, int)>>
            {
                new() { (0, 1), (0, 2), (1, 0), (1, 1) },
                new() { (0, 0), (1, 0), (1, 1), (2, 1) },
                new() { (0, 1), (0, 2), (1, 0), (1, 1) },
                new() { (0, 0), (1, 0), (1, 1), (2, 1) }
            },
            "Z" => new List<List<(int, int)>>
            {
                new() { (0, 0), (0, 1), (1, 1), (1, 2) },
                new() { (0, 1), (1, 0), (1, 1), (2, 0) },
                new() { (0, 0), (0, 1), (1, 1), (1, 2) },
                new() { (0, 1), (1, 0), (1, 1), (2, 0) }
            },
            "J" => new List<List<(int, int)>>
            {
                new() { (0, 0), (1, 0), (1, 1), (1, 2) },
                new() { (0, 0), (0, 1), (1, 0), (2, 0) },
                new() { (1, 0), (1, 1), (1, 2), (2, 2) },
                new() { (0, 2), (1, 0), (1, 1), (1, 2) }
            },
            "L" => new List<List<(int, int)>>
            {
                new() { (0, 2), (1, 0), (1, 1), (1, 2) },
                new() { (0, 0), (1, 0), (2, 0), (2, 1) },
                new() { (1, 0), (1, 1), (1, 2), (2, 0) },
                new() { (0, 0), (0, 1), (1, 1), (2, 1) }
            },
            _ => throw new ArgumentException($"Unknown piece: {type}")
        };
    }
}
