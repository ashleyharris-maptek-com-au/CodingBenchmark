#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <cstring>
#include <algorithm>
#include <set>

using namespace std;

const long long INF = 1e18;

int n, m, k;
vector<pair<int, int>> adj[505];
int terminals[20];
long long dp[1 << 18][505];
int parent_v[1 << 18][505];
int parent_mask[1 << 18][505];
set<pair<int, int>> edges_used;

void dijkstra(int mask) {
    priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<>> pq;

    vector<long long> dist(n, INF);
    for (int v = 0; v < n; v++) {
        dist[v] = dp[mask][v];
        if (dist[v] < INF) {
            pq.push({dist[v], v});
            parent_v[mask][v] = -1;
            parent_mask[mask][v] = -1;
        }
    }

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto [v, w] : adj[u]) {
            long long new_dist = dist[u] + w;
            if (new_dist < dist[v]) {
                dist[v] = new_dist;
                dp[mask][v] = new_dist;
                parent_v[mask][v] = u;
                parent_mask[mask][v] = mask;
                pq.push({new_dist, v});
            }
        }
    }
}

void reconstruct(int mask, int v) {
    if (parent_v[mask][v] == -1 && parent_mask[mask][v] == -1) {
        return;
    }

    if (parent_mask[mask][v] == mask) {
        int u = parent_v[mask][v];
        int a = min(u, v), b = max(u, v);
        edges_used.insert({a, b});
        reconstruct(mask, u);
    } else {
        int submask = parent_mask[mask][v];
        int complement = mask ^ submask;
        reconstruct(submask, v);
        reconstruct(complement, v);
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> m >> k;

    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    memset(dp, 0x3f, sizeof(dp));
    memset(parent_v, -1, sizeof(parent_v));
    memset(parent_mask, -1, sizeof(parent_mask));

    for (int i = 0; i < n; i++) {
        dp[0][i] = 0;
    }

    for (int mask = 1; mask < (1 << k); mask++) {
        for (int submask = (mask - 1) & mask; submask > 0; submask = (submask - 1) & mask) {
            int complement = mask ^ submask;
            for (int v = 0; v < n; v++) {
                if (dp[submask][v] + dp[complement][v] < dp[mask][v]) {
                    dp[mask][v] = dp[submask][v] + dp[complement][v];
                    parent_v[mask][v] = v;
                    parent_mask[mask][v] = submask;
                }
            }
        }

        dijkstra(mask);
    }

    int full_mask = (1 << k) - 1;
    long long ans = INF;
    int best_root = -1;
    for (int i = 0; i < n; i++) {
        if (dp[full_mask][i] < ans) {
            ans = dp[full_mask][i];
            best_root = i;
        }
    }

    reconstruct(full_mask, best_root);

    cout << ans << " " << edges_used.size() << "\n";
    for (auto [u, v] : edges_used) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
