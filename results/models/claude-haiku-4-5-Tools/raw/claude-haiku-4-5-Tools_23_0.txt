use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut lines = stdin.lock().lines();
    let mut stdout = stdout.lock();

    // Read header: width height gravity max_thrust fuel
    let header = parse_floats(&lines.next().unwrap().unwrap());
    let map_height = header[1] as usize;
    let gravity = header[2];
    let max_thrust = header[3];

    // Read position: start_x start_y target_x target_y
    let positions = parse_floats(&lines.next().unwrap().unwrap());
    let target_x = positions[2];
    let target_y = positions[3];

    // Skip map lines
    for _ in 0..map_height {
        let _ = lines.next();
    }

    // Process state updates
    for line_result in lines {
        let line = line_result.unwrap();
        let state = parse_floats(&line);

        if state.len() >= 7 {
            let x = state[0];
            let y = state[1];
            let vx = state[2];
            let vy = state[3];
            let angle = state[4];
            let fuel = state[6];

            let (thrust, turn) = get_control(x, y, vx, vy, angle, fuel, gravity, max_thrust, target_x, target_y);
            let _ = writeln!(stdout, "{:.2} {:.2}", thrust, turn);
            let _ = stdout.flush();
        }
    }
}

fn parse_floats(line: &str) -> Vec<f64> {
    line.split_whitespace()
        .filter_map(|s| s.parse::<f64>().ok())
        .collect()
}

fn get_control(x: f64, y: f64, vx: f64, vy: f64, angle: f64, fuel: f64, gravity: f64, max_thrust: f64, target_x: f64, target_y: f64) -> (f64, f64) {
    if fuel < 0.001 {
        return (0.0, 0.0);
    }

    // Vector to target
    let dx = target_x - x;
    let dy = target_y - y;
    let distance = (dx * dx + dy * dy).sqrt();

    // Target angle (where to point for landing)
    let target_angle = dy.atan2(dx);

    // Normalize angle error to [-π, π]
    let mut angle_error = target_angle - angle;
    while angle_error > std::f64::consts::PI {
        angle_error -= 2.0 * std::f64::consts::PI;
    }
    while angle_error < -std::f64::consts::PI {
        angle_error += 2.0 * std::f64::consts::PI;
    }

    // Turn command: scale angle error proportionally
    let turn = (angle_error / (std::f64::consts::PI / 2.0)).max(-1.0).min(1.0);

    // Thrust command: velocity-based control with distance-adaptive thresholds
    let thrust = if distance > 50.0 {
        // Far from target - focus on descent control
        if vy > 3.0 {
            1.0
        } else if vy > 1.0 {
            0.8
        } else if vy > -0.5 {
            (gravity / max_thrust).min(1.0)
        } else {
            0.3
        }
    } else if distance > 20.0 {
        // Medium range - balanced control
        if vy > 2.0 {
            1.0
        } else if vy > 0.5 {
            0.7
        } else if vy > -0.5 {
            (gravity / max_thrust).min(1.0)
        } else {
            0.4
        }
    } else {
        // Close to target - gentle landing
        if vy > 1.5 {
            1.0
        } else if vy > 0.3 {
            0.6
        } else if vy > -0.3 {
            (gravity / max_thrust * 0.9).min(1.0)
        } else {
            0.2
        }
    };

    (thrust, turn)
}
