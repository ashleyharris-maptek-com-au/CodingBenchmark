# Read input
line = input().split()
U = int(line[0])
S = int(line[1])

sets = []
for _ in range(S):
    elements = list(map(int, input().split()))
    sets.append(set(elements))

# Find exact cover using backtracking
def backtrack(covered, selected_indices):
    # If all elements are covered, we found a solution
    if len(covered) == U:
        return selected_indices[:]

    # Find an uncovered element
    uncovered = None
    for i in range(U):
        if i not in covered:
            uncovered = i
            break

    # Try each set that covers this uncovered element
    for idx in range(S):
        if idx in selected_indices:
            continue
        if uncovered in sets[idx]:
            # Check if this set only covers uncovered elements (no conflicts)
            if sets[idx].isdisjoint(covered):
                selected_indices.append(idx)
                new_covered = covered | sets[idx]
                result = backtrack(new_covered, selected_indices)
                if result is not None:
                    return result
                selected_indices.pop()

    return None

solution = backtrack(set(), [])

if solution is None:
    print("NO SOLUTION")
else:
    print("SOLUTION")
    print(" ".join(map(str, solution)))
