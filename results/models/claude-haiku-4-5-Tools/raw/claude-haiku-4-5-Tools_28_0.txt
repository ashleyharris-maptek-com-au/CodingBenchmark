use std::collections::HashSet;
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_vertices = parts[0].parse::<usize>().unwrap();
    let num_edges = parts[1].parse::<usize>().unwrap();

    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];
    let mut edges = Vec::new();

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u = parts[0].parse::<usize>().unwrap();
        let v = parts[1].parse::<usize>().unwrap();
        adj[u].insert(v);
        adj[v].insert(u);
        edges.push((u, v));
    }

    let mut cover = HashSet::new();
    let mut remaining_edges: HashSet<(usize, usize)> = edges
        .iter()
        .map(|&(u, v)| if u < v { (u, v) } else { (v, u) })
        .collect();

    while !remaining_edges.is_empty() {
        let (u, v) = *remaining_edges.iter().next().unwrap();

        let u_degree = remaining_edges.iter().filter(|&&(a, b)| a == u || b == u).count();
        let v_degree = remaining_edges.iter().filter(|&&(a, b)| a == v || b == v).count();

        let chosen = if u_degree >= v_degree { u } else { v };
        cover.insert(chosen);

        remaining_edges.retain(|&(a, b)| a != chosen && b != chosen);
    }

    let mut vertices: Vec<usize> = cover.iter().cloned().collect();
    vertices.sort();

    println!("{}", vertices.len());
    if !vertices.is_empty() {
        println!("{}", vertices.iter().map(|v| v.to_string()).collect::<Vec<_>>().join(" "));
    }
}
