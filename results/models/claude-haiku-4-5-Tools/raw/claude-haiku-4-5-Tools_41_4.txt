OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0 ColMajor MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64 ColMajor MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128 ColMajor MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %uboBuffer DescriptorSet 0
OpDecorate %uboBuffer Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%uboPtr = OpTypePointer Uniform %UBO
%vec4UPtr = OpTypePointer Uniform %vec4
%inputPtr = OpTypePointer Input %vec3
%outputPtr = OpTypePointer Output %vec4
%voidFunc = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

%glsl = OpExtInstImport "GLSL.std.450"

%uboBuffer = OpVariable %uboPtr Uniform
%worldPos = OpVariable %inputPtr Input
%normal = OpVariable %inputPtr Input
%outColor = OpVariable %outputPtr Output

%main = OpFunction %void None %voidFunc
%entry = OpLabel

%wpos = OpLoad %vec3 %worldPos
%norm = OpLoad %vec3 %normal

%lightPosAddr = OpAccessChain %vec4UPtr %uboBuffer %int_0 %int_3
%lightPosVec4 = OpLoad %vec4 %lightPosAddr
%lightPos = OpVectorShuffle %vec3 %lightPosVec4 %lightPosVec4 0 1 2

%Lvec = OpFSub %vec3 %lightPos %wpos
%L = OpExtInst %vec3 %glsl Normalize %Lvec
%N = OpExtInst %vec3 %glsl Normalize %norm

%d = OpDot %float %N %L
%intensity = OpExtInst %float %glsl FMax %d %float_0

%outColorVec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
OpStore %outColor %outColorVec

OpReturn
OpFunctionEnd
