#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    // Scale UV for brick pattern (8 wide, 16 tall)
    vec2 brickUV = uv * vec2(8.0, 16.0);

    // Get the row and determine if it's staggered
    float row = floor(brickUV.y);
    float isStaggered = mod(row, 2.0);

    // Offset X by 0.5 for staggered rows
    float offsetX = brickUV.x + (isStaggered * 0.5);

    // Get the brick coordinates within the current brick
    float brickX = fract(offsetX);
    float brickY = fract(brickUV.y);

    // Mortar thickness
    float mortarThickX = 0.05;
    float mortarThickY = 0.1;

    // Determine if we're in mortar region
    bool inMortarX = (brickX < mortarThickX) || (brickX > (1.0 - mortarThickX));
    bool inMortarY = (brickY < mortarThickY) || (brickY > (1.0 - mortarThickY));
    bool inMortar = inMortarX || inMortarY;

    // Colors
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    vec3 brickColor = vec3(0.7, 0.2, 0.1);

    vec3 finalColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    finalColor *= shadingFactor;

    outColor = vec4(finalColor, 1.0);
}
