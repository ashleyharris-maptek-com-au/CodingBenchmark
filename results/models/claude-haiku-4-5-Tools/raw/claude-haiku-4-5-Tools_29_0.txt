using System;
using System.Collections.Generic;
using System.Linq;
using System.Diagnostics;
using System.Threading;
using System.Threading.Tasks;

class MaximumCliqueFinder
{
    static void Main()
    {
        var input = Console.ReadLine().Split(' ');
        int numVertices = int.Parse(input[0]);
        int numEdges = int.Parse(input[1]);

        var adjList = new List<int>[numVertices];
        for (int i = 0; i < numVertices; i++)
            adjList[i] = new List<int>();

        for (int i = 0; i < numEdges; i++)
        {
            var edge = Console.ReadLine().Split(' ');
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adjList[u].Add(v);
            adjList[v].Add(u);
        }

        var sw = Stopwatch.StartNew();
        const long timeLimit = 4800; // 4.8 seconds

        List<int> maxClique;
        if (numVertices <= 100)
            maxClique = FindMaxCliqueExact(adjList, numVertices, timeLimit - sw.ElapsedMilliseconds);
        else
            maxClique = FindMaxCliqueHeuristic(adjList, numVertices, timeLimit - sw.ElapsedMilliseconds);

        sw.Stop();

        Console.WriteLine(maxClique.Count);
        if (maxClique.Count > 0)
            Console.WriteLine(string.Join(" ", maxClique.OrderBy(x => x)));
    }

    static List<int> FindMaxCliqueExact(List<int>[] adjList, int numVertices, long timeLimit)
    {
        var maxClique = new List<int>();
        var sw = Stopwatch.StartNew();

        for (int v = numVertices - 1; v >= 0; v--)
        {
            if (sw.ElapsedMilliseconds > timeLimit) break;

            var candidates = new List<int> { v };
            var possibleCandidates = new List<int>(adjList[v]);
            var clique = BronKerboschPivot(candidates, possibleCandidates, new List<int>(), adjList, sw, timeLimit);

            if (clique.Count > maxClique.Count)
                maxClique = clique;

            if (sw.ElapsedMilliseconds > timeLimit) break;
        }

        if (maxClique.Count == 0 && numVertices > 0)
            maxClique.Add(0);

        return maxClique;
    }

    static List<int> BronKerboschPivot(List<int> r, List<int> p, List<int> x, List<int>[] adjList, Stopwatch sw, long timeLimit)
    {
        if (sw.ElapsedMilliseconds > timeLimit)
            return r;

        if (p.Count == 0 && x.Count == 0)
            return new List<int>(r);

        var maxClique = new List<int>(r);
        var pivot = GetPivot(p, x, adjList);
        var pCopy = new List<int>(p);

        foreach (var v in pCopy)
        {
            if (sw.ElapsedMilliseconds > timeLimit)
                break;

            if (pivot >= 0 && !adjList[v].Contains(pivot))
            {
                var newR = new List<int>(r) { v };
                var neighbors = new HashSet<int>(adjList[v]);
                var newP = p.Where(u => neighbors.Contains(u)).ToList();
                var newX = x.Where(u => neighbors.Contains(u)).ToList();

                var clique = BronKerboschPivot(newR, newP, newX, adjList, sw, timeLimit);
                if (clique.Count > maxClique.Count)
                    maxClique = clique;
            }

            p.Remove(v);
            x.Add(v);
        }

        return maxClique;
    }

    static int GetPivot(List<int> p, List<int> x, List<int>[] adjList)
    {
        var candidates = p.Count > 0 ? p : x;
        if (candidates.Count == 0) return -1;
        return candidates.OrderByDescending(v => adjList[v].Count).First();
    }

    static List<int> FindMaxCliqueHeuristic(List<int>[] adjList, int numVertices, long timeLimit)
    {
        var sw = Stopwatch.StartNew();
        var maxClique = new List<int>();
        var random = new Random(42);

        // Sort vertices by degree (descending)
        var verticesByDegree = Enumerable.Range(0, numVertices)
            .OrderByDescending(v => adjList[v].Count)
            .ToList();

        // Try greedy clique construction from top degree vertices
        int tryCount = Math.Min(100, numVertices / 10 + 1);
        for (int i = 0; i < tryCount && sw.ElapsedMilliseconds < timeLimit; i++)
        {
            var clique = GreedyCliqueConstruction(verticesByDegree[i], adjList, timeLimit - sw.ElapsedMilliseconds);
            if (clique.Count > maxClique.Count)
                maxClique = clique;
        }

        // Local search improvements
        if (sw.ElapsedMilliseconds < timeLimit)
            maxClique = LocalSearchImprovement(maxClique, adjList, timeLimit - sw.ElapsedMilliseconds);

        if (maxClique.Count == 0 && numVertices > 0)
            maxClique.Add(0);

        return maxClique;
    }

    static List<int> GreedyCliqueConstruction(int startVertex, List<int>[] adjList, long timeLimit)
    {
        var clique = new List<int> { startVertex };
        var candidates = new HashSet<int>(adjList[startVertex]);
        var sw = Stopwatch.StartNew();

        while (candidates.Count > 0 && sw.ElapsedMilliseconds < timeLimit)
        {
            var best = candidates.OrderByDescending(v => candidates.Count(u => adjList[v].Contains(u))).First();
            clique.Add(best);

            var newCandidates = new List<int>();
            foreach (var c in candidates)
            {
                if (c != best && adjList[best].Contains(c))
                    newCandidates.Add(c);
            }
            candidates = new HashSet<int>(newCandidates);
        }

        return clique;
    }

    static List<int> LocalSearchImprovement(List<int> clique, List<int>[] adjList, long timeLimit)
    {
        var sw = Stopwatch.StartNew();
        var current = new List<int>(clique);

        while (sw.ElapsedMilliseconds < timeLimit)
        {
            var improved = false;

            // Try removing and replacing vertices
            for (int i = 0; i < current.Count && sw.ElapsedMilliseconds < timeLimit; i++)
            {
                var v = current[i];
                var temp = new List<int>(current);
                temp.RemoveAt(i);

                // Find best vertex to add
                var cliqueSet = new HashSet<int>(temp);
                var candidates = Enumerable.Range(0, adjList.Length)
                    .Where(u => !cliqueSet.Contains(u) && temp.All(x => adjList[u].Contains(x)))
                    .ToList();

                if (candidates.Count > 0)
                {
                    temp.Add(candidates[0]);
                    current = temp;
                    improved = true;
                    break;
                }
            }

            if (!improved) break;
        }

        return current;
    }
}
