use std::io::{self, BufRead};
use std::collections::{HashSet, VecDeque};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
struct Pos {
    x: i32,
    y: i32,
}

struct Grid {
    width: i32,
    height: i32,
    obstacles: HashSet<Pos>,
    total_cells: usize,
    require_cycle: bool,
}

impl Grid {
    fn is_valid(&self, p: Pos) -> bool {
        p.x >= 0 && p.x < self.width && p.y >= 0 && p.y < self.height && !self.obstacles.contains(&p)
    }

    fn neighbors(&self, p: Pos) -> Vec<Pos> {
        let mut result = Vec::new();
        for &(dx, dy) in &[(0, 1), (1, 0), (0, -1), (-1, 0)] {
            let next = Pos {
                x: p.x + dx,
                y: p.y + dy,
            };
            if self.is_valid(next) {
                result.push(next);
            }
        }
        result
    }

    fn count_unvisited_neighbors(&self, p: Pos, visited: &HashSet<Pos>) -> usize {
        self.neighbors(p)
            .iter()
            .filter(|n| !visited.contains(n))
            .count()
    }

    fn is_connected(&self, from: Pos, visited: &HashSet<Pos>) -> bool {
        if visited.len() == self.total_cells {
            return true;
        }

        let mut queue = VecDeque::new();
        let mut checked = HashSet::new();
        queue.push_back(from);
        checked.insert(from);

        let unvisited_count = self.total_cells - visited.len();
        let mut connected_count = 1;

        while let Some(current) = queue.pop_front() {
            for next in self.neighbors(current) {
                if !visited.contains(&next) && !checked.contains(&next) {
                    checked.insert(next);
                    queue.push_back(next);
                    connected_count += 1;
                    if connected_count == unvisited_count {
                        return true;
                    }
                }
            }
        }

        connected_count == unvisited_count
    }

    fn has_articulation_point(&self, p: Pos, visited: &HashSet<Pos>) -> bool {
        if visited.len() == self.total_cells {
            return false;
        }

        let neighbors = self.neighbors(p);
        if neighbors.len() <= 1 {
            return false;
        }

        let unvisited: Vec<_> = neighbors
            .iter()
            .filter(|n| !visited.contains(n))
            .copied()
            .collect();

        if unvisited.len() <= 1 {
            return false;
        }

        if unvisited.is_empty() {
            return false;
        }

        let mut visited_check = visited.clone();
        visited_check.insert(p);
        let start = unvisited[0];

        let mut queue = VecDeque::new();
        let mut checked = HashSet::new();
        queue.push_back(start);
        checked.insert(start);

        while let Some(current) = queue.pop_front() {
            for next in self.neighbors(current) {
                if !visited_check.contains(&next) && !checked.contains(&next) {
                    checked.insert(next);
                    queue.push_back(next);
                }
            }
        }

        checked.len() < unvisited.len()
    }
}

fn solve(grid: &Grid) -> Option<Vec<Pos>> {
    if grid.total_cells == 0 {
        return Some(vec![]);
    }

    let start = Pos { x: 0, y: 0 };
    if !grid.is_valid(start) {
        return None;
    }

    let mut visited = HashSet::new();
    let mut path = vec![start];
    visited.insert(start);

    let mut call_count = 0;
    let max_calls = 50_000_000;

    if dfs(grid, start, &mut visited, &mut path, &mut call_count, max_calls) {
        Some(path)
    } else {
        None
    }
}

fn dfs(
    grid: &Grid,
    current: Pos,
    visited: &mut HashSet<Pos>,
    path: &mut Vec<Pos>,
    call_count: &mut i32,
    max_calls: i32,
) -> bool {
    *call_count += 1;
    if *call_count > max_calls {
        return false;
    }

    if visited.len() == grid.total_cells {
        if grid.require_cycle {
            let neighbors = grid.neighbors(current);
            return neighbors.contains(&Pos { x: 0, y: 0 });
        }
        return true;
    }

    // Pruning: check if remaining cells are still connected
    if !grid.is_connected(current, visited) {
        return false;
    }

    // Pruning: check for articulation points in unvisited region
    if grid.has_articulation_point(current, visited) {
        return false;
    }

    let mut neighbors = grid.neighbors(current);

    // Heuristic: prioritize cells with fewer unvisited neighbors (most constrained first)
    neighbors.sort_by_key(|n| {
        if visited.contains(n) {
            i32::MAX
        } else {
            grid.count_unvisited_neighbors(*n, visited) as i32
        }
    });

    for next in neighbors {
        if !visited.contains(&next) {
            visited.insert(next);
            path.push(next);

            if dfs(grid, next, visited, path, call_count, max_calls) {
                return true;
            }

            path.pop();
            visited.remove(&next);
        }
    }

    false
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: i32 = parts[0].parse().unwrap();
    let height: i32 = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<i32>().unwrap() != 0;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<&str> = line.split_whitespace().collect();
        let x: i32 = coords[0].parse().unwrap();
        let y: i32 = coords[1].parse().unwrap();
        obstacles.insert(Pos { x, y });
    }

    let total_cells = ((width * height) as usize) - obstacles.len();

    let grid = Grid {
        width,
        height,
        obstacles,
        total_cells,
        require_cycle,
    };

    if let Some(path) = solve(&grid) {
        println!("{}", path.len());
        for pos in path {
            println!("{} {}", pos.x, pos.y);
        }
    } else {
        println!("0");
    }
}
