using namespace std;

const double EPS = 1e-9;
const double INF = 1e18;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& v) const { return Vec3(x+v.x, y+v.y, z+v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x-v.x, y-v.y, z-v.z); }
    Vec3 operator*(double t) const { return Vec3(x*t, y*t, z*t); }
    Vec3 operator/(double t) const { return Vec3(x/t, y/t, z/t); }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y*v.z-z*v.y, z*v.x-x*v.z, x*v.y-y*v.x); }
    double len() const { return sqrt(x*x+y*y+z*z); }
    Vec3 norm() const { double l = len(); if(l < EPS) return Vec3(1,0,0); return *this * (1.0/l); }
    double& operator[](int i) { return (&x)[i]; }
};

struct Face { int v[3]; };

struct Mesh {
    vector<Vec3> verts;
    vector<Face> faces;
};

struct AABB {
    Vec3 minp, maxp;
    AABB() : minp(INF, INF, INF), maxp(-INF, -INF, -INF) {}
    void add(Vec3 p) {
        minp.x = min(minp.x, p.x); minp.y = min(minp.y, p.y); minp.z = min(minp.z, p.z);
        maxp.x = max(maxp.x, p.x); maxp.y = max(maxp.y, p.y); maxp.z = max(maxp.z, p.z);
    }
};

struct BVHNode {
    AABB box;
    int face_idx;
    BVHNode* left;
    BVHNode* right;
    BVHNode() : face_idx(-1), left(nullptr), right(nullptr) {}
};

BVHNode* buildBVH(const Mesh& mesh, vector<int>& indices, int start, int end) {
    BVHNode* node = new BVHNode();

    for (int i = start; i < end; i++) {
        int fi = indices[i];
        for (int j = 0; j < 3; j++) {
            node->box.add(mesh.verts[mesh.faces[fi].v[j]]);
        }
    }

    if (end - start <= 4) {
        node->face_idx = (end - start == 1) ? indices[start] : -1;
        return node;
    }

    int axis = 0;
    if (node->box.maxp.y - node->box.minp.y > node->box.maxp.x - node->box.minp.x) axis = 1;
    if (node->box.maxp.z - node->box.minp.z > node->box.maxp.y - node->box.minp.y) axis = 2;

    double mid = (node->box.minp[axis] + node->box.maxp[axis]) * 0.5;
    int split = start;
    for (int i = start; i < end; i++) {
        double c = (mesh.verts[mesh.faces[indices[i]].v[0]][axis] +
                    mesh.verts[mesh.faces[indices[i]].v[1]][axis] +
                    mesh.verts[mesh.faces[indices[i]].v[2]][axis]) / 3.0;
        if (c < mid) swap(indices[split++], indices[i]);
    }

    if (split == start || split == end) split = (start + end) / 2;

    node->left = buildBVH(mesh, indices, start, split);
    node->right = buildBVH(mesh, indices, split, end);
    return node;
}

bool rayTriangleIntersect(Vec3 o, Vec3 d, Vec3 v0, Vec3 v1, Vec3 v2) {
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = d.cross(e2);
    double a = e1.dot(h);
    if (abs(a) < EPS) return false;

    double f = 1.0 / a;
    Vec3 s = o - v0;
    double u = f * s.dot(h);
    if (u < -EPS || u > 1.0 + EPS) return false;

    Vec3 q = s.cross(e1);
    double v = f * d.dot(q);
    if (v < -EPS || u + v > 1.0 + EPS) return false;

    double t = f * e2.dot(q);
    return t > EPS;
}

bool pointInMesh(Vec3 p, const Mesh& mesh) {
    Vec3 d(0.73856093, 0.32921938, 0.10659975);
    int hits = 0;
    for (const auto& f : mesh.faces) {
        if (rayTriangleIntersect(p, d, mesh.verts[f.v[0]], mesh.verts[f.v[1]], mesh.verts[f.v[2]])) {
            hits++;
        }
    }
    return (hits & 1) == 1;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int va, fa, vb, fb;
    cin >> va >> fa >> vb >> fb;

    Mesh a, b;
    a.verts.resize(va);
    a.faces.resize(fa);
    b.verts.resize(vb);
    b.faces.resize(fb);

    for (int i = 0; i < va; i++) {
        cin >> a.verts[i].x >> a.verts[i].y >> a.verts[i].z;
    }
    for (int i = 0; i < fa; i++) {
        cin >> a.faces[i].v[0] >> a.faces[i].v[1] >> a.faces[i].v[2];
    }
    for (int i = 0; i < vb; i++) {
        cin >> b.verts[i].x >> b.verts[i].y >> b.verts[i].z;
    }
    for (int i = 0; i < fb; i++) {
        cin >> b.faces[i].v[0] >> b.faces[i].v[1] >> b.faces[i].v[2];
    }

    Mesh result;
    map<int, int> vert_remap_a, vert_remap_b;

    // Process faces from A - keep those outside B
    for (int i = 0; i < fa; i++) {
        Face f = a.faces[i];
        Vec3 c = (a.verts[f.v[0]] + a.verts[f.v[1]] + a.verts[f.v[2]]) * (1.0/3.0);
        Vec3 n = (a.verts[f.v[1]] - a.verts[f.v[0]]).cross(a.verts[f.v[2]] - a.verts[f.v[0]]).norm();
        Vec3 p = c + n * (EPS * 10);

        if (!pointInMesh(p, b)) {
            Face nf;
            for (int j = 0; j < 3; j++) {
                if (vert_remap_a.find(f.v[j]) == vert_remap_a.end()) {
                    vert_remap_a[f.v[j]] = result.verts.size();
                    result.verts.push_back(a.verts[f.v[j]]);
                }
                nf.v[j] = vert_remap_a[f.v[j]];
            }
            result.faces.push_back(nf);
        }
    }

    // Process faces from B - keep those outside A
    for (int i = 0; i < fb; i++) {
        Face f = b.faces[i];
        Vec3 c = (b.verts[f.v[0]] + b.verts[f.v[1]] + b.verts[f.v[2]]) * (1.0/3.0);
        Vec3 n = (b.verts[f.v[1]] - b.verts[f.v[0]]).cross(b.verts[f.v[2]] - b.verts[f.v[0]]).norm();
        Vec3 p = c + n * (EPS * 10);

        if (!pointInMesh(p, a)) {
            Face nf;
            for (int j = 0; j < 3; j++) {
                if (vert_remap_b.find(f.v[j]) == vert_remap_b.end()) {
                    vert_remap_b[f.v[j]] = result.verts.size();
                    result.verts.push_back(b.verts[f.v[j]]);
                }
                nf.v[j] = vert_remap_b[f.v[j]];
            }
            result.faces.push_back(nf);
        }
    }

    cout << result.verts.size() << " " << result.faces.size() << "\n";
    cout.precision(17);
    for (const auto& v : result.verts) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }
    for (const auto& f : result.faces) {
        cout << f.v[0] << " " << f.v[1] << " " << f.v[2] << "\n";
    }

    return 0;
}
