def autopilot_step(state: dict, dt: float) -> dict:
    """Twin-engine airliner autopilot with sensor fusion and flight control."""

    # Initialize persistent state
    if not hasattr(autopilot_step, 'alt_integral'):
        autopilot_step.alt_integral = 0.0
        autopilot_step.hdg_integral = 0.0
        autopilot_step.ias_integral = 0.0
        autopilot_step.roll_integral = 0.0
        autopilot_step.pitch_integral = 0.0
        autopilot_step.prev_alt = 0.0
        autopilot_step.prev_hdg = 0.0
        autopilot_step.prev_ias = 0.0
        autopilot_step.trim_offset = 0.0

    # Sensor fusion: get best estimate from redundant sensors
    def fuse_altitude():
        """Fuse barometric, radar, and GPS altitude using voting."""
        altitudes = []

        # Barometric altitude (most reliable at altitude)
        for i in range(1, 4):
            valid = state.get(f'baro_alt_{i}_valid', False)
            alt = state.get(f'baro_alt_{i}_ft', 0.0)
            if valid and -500 < alt < 50000:
                altitudes.append(('baro', alt, i))

        # Radar altitude (accurate near ground, unreliable above ~2500 ft)
        radar_alts = []
        for i in range(1, 4):
            valid = state.get(f'radar_alt_{i}_valid', False)
            alt = state.get(f'radar_alt_{i}_ft', 0.0)
            if valid and 0 <= alt < 2500:
                radar_alts.append(alt)

        if radar_alts:
            radar_alt = sorted(radar_alts)[len(radar_alts) // 2]
            return radar_alt, 'radar'

        # Use GPS altitude as secondary
        for i in range(1, 4):
            valid = state.get(f'gps_{i}_valid', False)
            alt = state.get(f'gps_{i}_alt_ft', 0.0)
            if valid and -500 < alt < 50000:
                altitudes.append(('gps', alt, i))

        if altitudes:
            altitudes.sort(key=lambda x: x[1])
            return altitudes[len(altitudes) // 2][1], altitudes[len(altitudes) // 2][0]

        return state.get('baro_alt_1_ft', 0.0), 'baro'

    def fuse_heading():
        """Fuse magnetic heading from IRS units."""
        headings = []
        for i in range(1, 4):
            valid = state.get(f'irs_{i}_valid', False)
            hdg = state.get(f'irs_{i}_heading_deg', 0.0)
            if valid:
                headings.append(hdg)

        if not headings:
            return state.get('magnetic_heading_deg', 0.0)

        headings.sort()
        return headings[len(headings) // 2]

    def fuse_airspeed():
        """Fuse indicated airspeed from ADCs."""
        speeds = []
        for i in range(1, 4):
            valid = state.get(f'ias_{i}_valid', False)
            ias = state.get(f'ias_{i}_kt', 0.0)
            if valid and 0 <= ias < 500:
                speeds.append(ias)

        if not speeds:
            return state.get('ias_1_kt', 100.0)

        speeds.sort()
        return speeds[len(speeds) // 2]

    def fuse_vs():
        """Fuse vertical speed from IRS units."""
        vss = []
        for i in range(1, 4):
            valid = state.get(f'irs_{i}_valid', False)
            vs = state.get(f'irs_{i}_vs_fpm', 0.0)
            if valid:
                vss.append(vs)

        if not vss:
            return state.get('vertical_speed_fpm', 0.0)

        vss.sort()
        return vss[len(vss) // 2]

    def fuse_pitch():
        """Fuse pitch from IRS units."""
        pitches = []
        for i in range(1, 4):
            valid = state.get(f'irs_{i}_valid', False)
            pitch = state.get(f'irs_{i}_pitch_deg', 0.0)
            if valid:
                pitches.append(pitch)

        if not pitches:
            return 0.0

        pitches.sort()
        return pitches[len(pitches) // 2]

    def fuse_roll():
        """Fuse roll from IRS units."""
        rolls = []
        for i in range(1, 4):
            valid = state.get(f'irs_{i}_valid', False)
            roll = state.get(f'irs_{i}_roll_deg', 0.0)
            if valid:
                rolls.append(roll)

        if not rolls:
            return 0.0

        rolls.sort()
        return rolls[len(rolls) // 2]

    # Get fused sensor values
    current_alt, alt_source = fuse_altitude()
    current_hdg = fuse_heading()
    current_ias = fuse_airspeed()
    current_vs = fuse_vs()
    current_pitch = fuse_pitch()
    current_roll = fuse_roll()

    target_alt = state.get('target_alt_ft', current_alt)
    target_hdg = state.get('target_hdg_deg', current_hdg)
    target_ias = state.get('target_ias_kt', current_ias)
    target_vs = state.get('target_vs_fpm', 0.0)

    # Safety limits
    Vne = 365.0
    Vs = 115.0
    max_pitch = 25.0

    # Altitude control (main vertical mode)
    alt_error = target_alt - current_alt
    autopilot_step.alt_integral += alt_error * dt
    autopilot_step.alt_integral = max(-5000, min(5000, autopilot_step.alt_integral))

    alt_kp = 0.001
    alt_ki = 0.0001
    alt_kd = 0.05

    alt_derivative = (alt_error - (target_alt - autopilot_step.prev_alt)) / dt if dt > 0 else 0
    alt_command = alt_kp * alt_error + alt_ki * autopilot_step.alt_integral + alt_kd * alt_derivative

    # Convert altitude command to pitch command
    pitch_cmd = max(-max_pitch, min(max_pitch, alt_command * 0.1))

    # Vertical speed dampening
    vs_error = current_vs - target_vs
    if abs(vs_error) > 100:
        vs_correction = -vs_error * 0.001
        pitch_cmd += vs_correction

    # Pitch control (elevator)
    pitch_error = current_pitch - pitch_cmd
    elevator_cmd = -pitch_error * 0.05

    # Add trim adjustment for stick force relief
    autopilot_step.trim_offset += elevator_cmd * dt * 0.1
    autopilot_step.trim_offset = max(-1.0, min(1.0, autopilot_step.trim_offset))

    elevator_out = max(-1.0, min(1.0, elevator_cmd))

    # Heading/roll control
    hdg_error = target_hdg - current_hdg
    while hdg_error > 180:
        hdg_error -= 360
    while hdg_error < -180:
        hdg_error += 360

    autopilot_step.hdg_integral += hdg_error * dt
    autopilot_step.hdg_integral = max(-360, min(360, autopilot_step.hdg_integral))

    hdg_kp = 0.01
    hdg_ki = 0.0005

    roll_cmd = max(-25.0, min(25.0, hdg_kp * hdg_error + hdg_ki * autopilot_step.hdg_integral))

    # Roll control (aileron)
    roll_error = current_roll - roll_cmd
    aileron_cmd = -roll_error * 0.02
    aileron_out = max(-1.0, min(1.0, aileron_cmd))

    # Yaw damping from gyro
    yaw_rate = state.get('gyro_1_yaw_rate_dps', 0.0)
    if abs(yaw_rate) > 1.0:
        rudder_cmd = -yaw_rate * 0.02
    else:
        rudder_cmd = 0.0

    rudder_out = max(-1.0, min(1.0, rudder_cmd))

    # Airspeed control (throttle)
    ias_error = target_ias - current_ias
    autopilot_step.ias_integral += ias_error * dt
    autopilot_step.ias_integral = max(-100, min(100, autopilot_step.ias_integral))

    ias_kp = 0.002
    ias_ki = 0.0001

    throttle_cmd = ias_kp * ias_error + ias_ki * autopilot_step.ias_integral
    throttle_base = state.get('throttle_1_pos', 0.5)
    throttle_1_out = max(0.0, min(1.0, throttle_base + throttle_cmd))

    throttle_base_2 = state.get('throttle_2_pos', 0.5)
    throttle_2_out = max(0.0, min(1.0, throttle_base_2 + throttle_cmd))

    # Stall protection
    if current_ias < Vs + 10:
        pitch_cmd = max(pitch_cmd, -5.0)
        throttle_1_out = max(throttle_1_out, 0.7)
        throttle_2_out = max(throttle_2_out, 0.7)

    # Overspeed protection
    if current_ias > Vne - 10:
        pitch_cmd = min(pitch_cmd, 5.0)
        throttle_1_out = min(throttle_1_out, 0.3)
        throttle_2_out = min(throttle_2_out, 0.3)

    # Flap management
    on_ground = (state.get('weight_on_wheels_left', False) or
                 state.get('weight_on_wheels_nose', False) or
                 state.get('weight_on_wheels_right', False))

    if on_ground:
        flaps = 40
        gear = 1
    else:
        if current_ias < 200:
            flaps = 15
        elif current_ias < 250:
            flaps = 5
        else:
            flaps = 0
        gear = 0

    # Update persistent values
    autopilot_step.prev_alt = target_alt
    autopilot_step.prev_hdg = target_hdg
    autopilot_step.prev_ias = target_ias

    return {
        'elevator': elevator_out,
        'aileron': aileron_out,
        'rudder': rudder_out,
        'throttle_1': throttle_1_out,
        'throttle_2': throttle_2_out,
        'elevator_trim': autopilot_step.trim_offset,
        'flaps': flaps,
        'speedbrake': 0.0,
        'gear': gear,
    }
