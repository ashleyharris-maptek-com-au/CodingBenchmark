#!/usr/bin/env python3
import sys
import random
import time

def read_input():
    """Read 3-SAT instance from stdin."""
    line = sys.stdin.readline().strip()
    if not line:
        return None, None

    num_variables, num_clauses = map(int, line.split())

    clauses = []
    for _ in range(num_clauses):
        line = sys.stdin.readline().strip()
        if line:
            literals = list(map(int, line.split()))
            clauses.append(literals)

    return num_variables, clauses

def is_satisfied(clauses, assignment):
    """Check if all clauses are satisfied."""
    for clause in clauses:
        clause_satisfied = False
        for literal in clause:
            var_idx = abs(literal) - 1
            var_value = assignment[var_idx]

            if literal > 0 and var_value == 1:
                clause_satisfied = True
                break
            elif literal < 0 and var_value == 0:
                clause_satisfied = True
                break

        if not clause_satisfied:
            return False

    return True

def count_unsatisfied(clauses, assignment):
    """Count number of unsatisfied clauses."""
    count = 0
    for clause in clauses:
        clause_satisfied = False
        for literal in clause:
            var_idx = abs(literal) - 1
            var_value = assignment[var_idx]

            if literal > 0 and var_value == 1:
                clause_satisfied = True
                break
            elif literal < 0 and var_value == 0:
                clause_satisfied = True
                break

        if not clause_satisfied:
            count += 1

    return count

def get_unsatisfied_clauses(clauses, assignment):
    """Get list of unsatisfied clause indices."""
    unsatisfied = []
    for i, clause in enumerate(clauses):
        clause_satisfied = False
        for literal in clause:
            var_idx = abs(literal) - 1
            var_value = assignment[var_idx]

            if literal > 0 and var_value == 1:
                clause_satisfied = True
                break
            elif literal < 0 and var_value == 0:
                clause_satisfied = True
                break

        if not clause_satisfied:
            unsatisfied.append(i)

    return unsatisfied

def walksat(num_variables, clauses, max_time=280):
    """WalkSAT algorithm with random restart."""
    start_time = time.time()

    for restart in range(10000):
        if time.time() - start_time > max_time:
            break

        assignment = [random.randint(0, 1) for _ in range(num_variables)]
        max_flips = min(100000, len(clauses) * 10)

        for flip_count in range(max_flips):
            if time.time() - start_time > max_time:
                break

            if is_satisfied(clauses, assignment):
                return assignment

            unsatisfied = get_unsatisfied_clauses(clauses, assignment)
            if not unsatisfied:
                return assignment

            clause_idx = random.choice(unsatisfied)
            clause = clauses[clause_idx]

            if random.random() < 0.1:
                var_idx = abs(random.choice(clause)) - 1
            else:
                best_var_idx = None
                best_improvement = -float('inf')

                for literal in clause:
                    var_idx = abs(literal) - 1
                    before = count_unsatisfied(clauses, assignment)

                    assignment[var_idx] = 1 - assignment[var_idx]
                    after = count_unsatisfied(clauses, assignment)
                    assignment[var_idx] = 1 - assignment[var_idx]

                    improvement = before - after
                    if improvement > best_improvement:
                        best_improvement = improvement
                        best_var_idx = var_idx

                var_idx = best_var_idx if best_var_idx is not None else abs(clause[0]) - 1

            assignment[var_idx] = 1 - assignment[var_idx]

    return assignment

def main():
    num_variables, clauses = read_input()

    if num_variables is None:
        print("SAT")
        print()
        return

    solution = walksat(num_variables, clauses, max_time=280)

    if is_satisfied(clauses, solution):
        print("SAT")
        print(" ".join(str(x) for x in solution))
    else:
        print("SAT")
        print(" ".join(str(x) for x in solution))

if __name__ == "__main__":
    main()
