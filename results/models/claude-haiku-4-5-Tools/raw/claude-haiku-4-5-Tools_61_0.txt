declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: string;
  x: number;
  y: number;
}

const depot = data.depot as [number, number];
const packages: Package[] = data.packages;
const maxTrucks = data.truckCount;

function dist(a: [number, number] | Package, b: [number, number] | Package): number {
  const ax = 'x' in a ? a.x : a[0];
  const ay = 'y' in a ? a.y : a[1];
  const bx = 'x' in b ? b.x : b[0];
  const by = 'y' in b ? b.y : b[1];
  return Math.hypot(ax - bx, ay - by);
}

function buildNearestNeighborRoutes(): string[][] {
  const unvisited = new Set(packages.map(p => p.id));
  const routes: string[][] = [];

  while (unvisited.size > 0) {
    const route: string[] = [];
    let currentPos = depot;

    while (unvisited.size > 0 && routes.length < maxTrucks) {
      let nearestId = '';
      let nearestDist = Infinity;

      for (const id of unvisited) {
        const pkg = packages.find(p => p.id === id)!;
        const d = dist(currentPos, pkg);
        if (d < nearestDist) {
          nearestDist = d;
          nearestId = id;
        }
      }

      if (!nearestId) break;

      route.push(nearestId);
      unvisited.delete(nearestId);
      currentPos = packages.find(p => p.id === nearestId)!;
    }

    if (route.length > 0) {
      routes.push(route);
    }
  }

  return routes;
}

function twoOptImprovement(routes: string[][]): string[][] {
  let improved = true;

  while (improved) {
    improved = false;

    for (let routeIdx = 0; routeIdx < routes.length; routeIdx++) {
      const route = routes[routeIdx];

      for (let i = 0; i < route.length - 1; i++) {
        for (let k = i + 2; k < route.length; k++) {
          const p_i = packages.find(p => p.id === route[i])!;
          const p_i1 = packages.find(p => p.id === route[i + 1])!;
          const p_k = packages.find(p => p.id === route[k])!;
          const p_k1 = k + 1 < route.length ? packages.find(p => p.id === route[k + 1])! : null;

          // Current cost of edges (i->i+1) and (k->k+1 or depot)
          const currentCost = dist(p_i, p_i1) + (p_k1 ? dist(p_k, p_k1) : dist(p_k, depot));

          // Proposed cost after 2-opt swap: (i->k) and (i+1->k+1 or depot)
          const proposedCost = dist(p_i, p_k) + (p_k1 ? dist(p_i1, p_k1) : dist(p_i1, depot));

          if (proposedCost < currentCost) {
            // Reverse segment from i+1 to k
            const newRoute = route.slice(0, i + 1)
              .concat(route.slice(i + 1, k + 1).reverse())
              .concat(route.slice(k + 1));
            routes[routeIdx] = newRoute;
            improved = true;
          }
        }
      }
    }
  }

  return routes;
}

const solution = twoOptImprovement(buildNearestNeighborRoutes());
