#include <iostream>
#include <fstream>
#include <thread>
#include <mutex>
#include <atomic>
#include <queue>
#include <cmath>
#include <vector>
#include <chrono>
#include <iomanip>

using namespace std;

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct TerrainCell {
    uint16_t ground_top;
    uint16_t float_bottom;
    uint16_t float_top;
};

int terrain_width, terrain_depth, terrain_height;
double gravity, max_thrust;
double target_x, target_y, target_z;
vector<vector<TerrainCell>> terrain;

mutex state_mutex;
State current_state;
bool state_received = false;
bool shutdown_flag = false;

void read_terrain(const string& path) {
    ifstream file(path, ios::binary);
    if (!file.is_open()) return;

    terrain.assign(terrain_width, vector<TerrainCell>(terrain_depth));

    for (int x = 0; x < terrain_width; x++) {
        for (int y = 0; y < terrain_depth; y++) {
            uint16_t a, b, c;
            file.read((char*)&a, 2);
            file.read((char*)&b, 2);
            file.read((char*)&c, 2);
            terrain[x][y] = {a, b, c};
        }
    }
}

void input_thread_func() {
    string line;

    // Read header
    cin >> terrain_width >> terrain_depth >> terrain_height >> gravity >> max_thrust;
    double fuel_initial;
    cin >> fuel_initial;

    double sx, sy, sz;
    cin >> sx >> sy >> sz;
    cin >> target_x >> target_y >> target_z;

    getline(cin, line);  // consume newline
    getline(cin, line);  // terrain path
    read_terrain(line);

    // Wait for STATE header
    while (getline(cin, line)) {
        if (line.find("STATE") != string::npos) break;
    }

    // Read state updates
    while (!shutdown_flag) {
        State s;
        if (!(cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                  >> s.pitch >> s.yaw >> s.roll
                  >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel)) {
            shutdown_flag = true;
            break;
        }

        {
            lock_guard<mutex> lock(state_mutex);
            current_state = s;
            state_received = true;
        }
    }
}

double compute_distance_to_landing_zone() {
    lock_guard<mutex> lock(state_mutex);
    double dx = current_state.x - target_x;
    double dy = current_state.y - target_y;
    return sqrt(dx*dx + dy*dy);
}

double compute_descent_rate() {
    lock_guard<mutex> lock(state_mutex);
    return current_state.vz;
}

double compute_horizontal_speed() {
    lock_guard<mutex> lock(state_mutex);
    return sqrt(current_state.vx*current_state.vx + current_state.vy*current_state.vy);
}

double compute_altitude() {
    lock_guard<mutex> lock(state_mutex);
    return current_state.z - target_z;
}

pair<double, double> compute_target_pitch_yaw() {
    lock_guard<mutex> lock(state_mutex);

    double dx = target_x - current_state.x;
    double dy = target_y - current_state.y;
    double dz = target_z - current_state.z;

    double alt = current_state.z - target_z;
    double horiz_dist = sqrt(dx*dx + dy*dy);

    if (horiz_dist < 1.0) {
        return {0.0, 0.0};
    }

    // Guidance: tilt towards target based on altitude and distance
    double target_pitch_magnitude = min(1.0, horiz_dist / (alt + 10.0));

    double target_yaw_angle = atan2(dx, dy);
    double yaw_error = target_yaw_angle - current_state.yaw;

    // Normalize yaw error
    while (yaw_error > M_PI) yaw_error -= 2*M_PI;
    while (yaw_error < -M_PI) yaw_error += 2*M_PI;

    double pitch_direction = atan2(dx, dy);
    double pitch_component = target_pitch_magnitude * cos(pitch_direction - current_state.yaw);

    return {pitch_component, yaw_error * 0.5};
}

double compute_thrust() {
    lock_guard<mutex> lock(state_mutex);

    double alt = current_state.z - target_z;
    double descent_rate = -current_state.vz;  // positive = falling
    double horiz_speed = sqrt(current_state.vx*current_state.vx + current_state.vy*current_state.vy);

    // Vertical control: use gravity as baseline, add damping
    double base_accel = gravity;
    double damping = 0.0;

    if (alt < 1000.0) {
        // Near landing zone: careful descent control
        double target_descent = min(2.0, alt / 20.0);
        damping = (descent_rate - target_descent) * 0.3;
    } else if (alt < 10000.0) {
        // Mid altitude: moderate descent
        damping = descent_rate * 0.1;
    } else {
        // High altitude: gentle descent control
        damping = descent_rate * 0.05;
    }

    double accel_needed = base_accel + damping;

    // Lateral tilt reduces vertical thrust component
    double cos_pitch = cos(current_state.pitch);
    double vertical_thrust_factor = cos_pitch;

    double thrust = 0.0;
    if (vertical_thrust_factor > 0.01) {
        thrust = accel_needed / vertical_thrust_factor;
    } else {
        thrust = max_thrust;
    }

    thrust = max(0.0, min(1.0, thrust / max_thrust));

    // Safety: ensure enough thrust when near ground
    if (alt < 500.0 && thrust < 0.5) {
        thrust = 0.5;
    }

    return thrust;
}

int main() {
    cout << fixed << setprecision(6);
    cerr << fixed << setprecision(6);

    // Start input reading thread
    thread input_thread(input_thread_func);

    // Wait for first state
    auto start_time = chrono::steady_clock::now();
    while (!state_received && !shutdown_flag) {
        this_thread::sleep_for(chrono::milliseconds(5));
        auto elapsed = chrono::duration_cast<chrono::seconds>(
            chrono::steady_clock::now() - start_time).count();
        if (elapsed > 120) {
            shutdown_flag = true;
            break;
        }
    }

    // Control loop
    auto last_output = chrono::steady_clock::now();
    const auto output_period = chrono::milliseconds(100);  // 10 Hz minimum

    int missed_updates = 0;

    while (!shutdown_flag) {
        auto now = chrono::steady_clock::now();

        if (now - last_output >= output_period) {
            if (!state_received) {
                missed_updates++;
                if (missed_updates >= 2) {
                    cout << "0 0 0 0\n" << flush;
                    missed_updates = 0;
                }
                this_thread::sleep_for(chrono::milliseconds(10));
                continue;
            }

            missed_updates = 0;

            // Compute control outputs
            auto [target_pitch, target_yaw] = compute_target_pitch_yaw();
            double thrust = compute_thrust();

            // Simple proportional control for pitch/yaw
            double pitch_cmd = 0.0, yaw_cmd = 0.0;

            {
                lock_guard<mutex> lock(state_mutex);
                double pitch_error = target_pitch - current_state.pitch;
                double yaw_error = target_yaw;  // already computed relative

                pitch_cmd = max(-1.0, min(1.0, pitch_error * 1.5));
                yaw_cmd = max(-1.0, min(1.0, yaw_error * 1.5));
            }

            double roll_cmd = 0.0;

            cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n" << flush;
            last_output = now;
        }

        this_thread::sleep_for(chrono::milliseconds(5));
    }

    input_thread.join();
    return 0;
}
