OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_color %out_color
OpExecutionMode %main OriginUpperLeft

OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%func_type = OpTypeFunction %void

%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

%main = OpFunction %void None %func_type
%entry = OpLabel
%loaded_color = OpLoad %vec3 %in_color
%x = OpCompositeExtract %float %loaded_color 0
%y = OpCompositeExtract %float %loaded_color 1
%z = OpCompositeExtract %float %loaded_color 2
%one = OpConstant %float 1.0
%result = OpCompositeConstruct %vec4 %x %y %z %one
OpStore %out_color %result
OpReturn
OpFunctionEnd
