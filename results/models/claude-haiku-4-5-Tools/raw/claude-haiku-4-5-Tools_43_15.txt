float4 main(PSInput input) : SV_Target0
{
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(float3(5.0, 5.0, 5.0) - input.worldPos);

    float t = dot(normal, lightDir) * 0.5 + 0.5;

    float3 cool = float3(0.2, 0.2, 0.75);
    float3 warm = float3(0.7, 0.7, 0.4);

    float3 color = lerp(cool, warm, t);

    return float4(color, 1.0);
}
