use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read the first line with number of vertices and edges
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let num_vertices: usize = parts[0].parse().unwrap();
    let num_edges: usize = parts[1].parse().unwrap();

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); num_vertices];

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let u: usize = parts[0].parse().unwrap();
        let v: usize = parts[1].parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
    }

    // Greedy dominating set: repeatedly pick the vertex that covers the most uncovered vertices
    let mut dominating_set = Vec::new();
    let mut covered = vec![false; num_vertices];

    while covered.iter().any(|&x| !x) {
        // Find the vertex not yet in dominating set that covers the most uncovered vertices
        let mut best_vertex = 0;
        let mut best_count = 0;

        for v in 0..num_vertices {
            if dominating_set.contains(&v) {
                continue;
            }

            let mut count = 0;
            // Count v itself if not covered
            if !covered[v] {
                count += 1;
            }
            // Count neighbors if not covered
            for &u in &adj[v] {
                if !covered[u] {
                    count += 1;
                }
            }

            if count > best_count {
                best_count = count;
                best_vertex = v;
            }
        }

        // Add the best vertex to the dominating set
        dominating_set.push(best_vertex);
        covered[best_vertex] = true;
        for &u in &adj[best_vertex] {
            covered[u] = true;
        }
    }

    // Output
    println!("{}", dominating_set.len());
    println!("{}", dominating_set.iter()
        .map(|v| v.to_string())
        .collect::<Vec<_>>()
        .join(" "));
}
