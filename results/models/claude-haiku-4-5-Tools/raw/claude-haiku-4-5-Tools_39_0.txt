use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read n and m
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    // Read edges
    let mut edges = Vec::new();
    let mut adj = vec![Vec::new(); n];

    for _ in 0..m {
        let edge_line = lines.next().unwrap().unwrap();
        let mut parts = edge_line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        edges.push((u, v));
        adj[u].push(v);
        adj[v].push(u);
    }

    // Find minimum bisection
    let (cut_size, partition) = find_min_bisection(n, &edges, &adj);

    // Output
    println!("{}", cut_size);
    let partition_str = partition.iter()
        .map(|x| x.to_string())
        .collect::<Vec<_>>()
        .join(" ");
    println!("{}", partition_str);
}

fn find_min_bisection(n: usize, edges: &[(usize, usize)], adj: &[Vec<usize>]) -> (usize, Vec<usize>) {
    if n <= 18 {
        exhaustive_search(n, edges)
    } else {
        kernighan_lin_bisection(n, edges, adj)
    }
}

fn exhaustive_search(n: usize, edges: &[(usize, usize)]) -> (usize, Vec<usize>) {
    let target_size = n / 2;
    let mut best_cut = usize::MAX;
    let mut best_partition = Vec::new();

    let mut partition = (0..target_size).collect::<Vec<_>>();

    loop {
        let partition_set: HashSet<_> = partition.iter().copied().collect();
        let mut cut = 0;

        for &(u, v) in edges {
            let u_in = partition_set.contains(&u);
            let v_in = partition_set.contains(&v);
            if u_in != v_in {
                cut += 1;
            }
        }

        if cut < best_cut {
            best_cut = cut;
            best_partition = partition.clone();
        }

        if !next_combination(&mut partition, n) {
            break;
        }
    }

    best_partition.sort();
    (best_cut, best_partition)
}

fn next_combination(combo: &mut [usize], n: usize) -> bool {
    let len = combo.len();

    for i in (0..len).rev() {
        if combo[i] < n - (len - i) {
            combo[i] += 1;

            for j in i + 1..len {
                combo[j] = combo[j - 1] + 1;
            }
            return true;
        }
    }

    false
}

fn kernighan_lin_bisection(n: usize, edges: &[(usize, usize)], adj: &[Vec<usize>]) -> (usize, Vec<usize>) {
    let mut partition = initial_partition(n, adj);
    let mut improved = true;

    while improved {
        improved = false;
        partition = kernighan_lin_pass(&partition, n, edges, adj, &mut improved);
    }

    let cut_size = calculate_cut(n, &partition, edges);
    let mut result: Vec<usize> = partition.iter()
        .enumerate()
        .filter_map(|(i, &in_part)| if in_part { Some(i) } else { None })
        .collect();
    result.sort();
    (cut_size, result)
}

fn initial_partition(n: usize, adj: &[Vec<usize>]) -> Vec<bool> {
    let mut partition = vec![false; n];
    let mut count = 0;

    let mut added = vec![false; n];
    let mut queue = vec![0];
    added[0] = true;
    partition[0] = true;
    count = 1;

    while !queue.is_empty() && count < n / 2 {
        let v = queue.remove(0);
        for &u in &adj[v] {
            if !added[u] {
                added[u] = true;
                if count < n / 2 {
                    partition[u] = true;
                    count += 1;
                    queue.push(u);
                }
            }
        }
    }

    partition
}

fn kernighan_lin_pass(
    current: &[bool],
    n: usize,
    edges: &[(usize, usize)],
    adj: &[Vec<usize>],
    improved: &mut bool,
) -> Vec<bool> {
    let mut partition = current.to_vec();
    let mut locked = vec![false; n];
    let mut best_partition = partition.clone();
    let mut best_cut = calculate_cut(n, &best_partition, edges);

    for _ in 0..n / 2 {
        let (_max_gain_pair, _max_gain) = find_best_swap(&partition, &locked, n, adj);

        if let Some((u, v)) = _max_gain_pair {
            partition[u] = !partition[u];
            partition[v] = !partition[v];
            locked[u] = true;
            locked[v] = true;

            let cut = calculate_cut(n, &partition, edges);
            if cut < best_cut {
                best_cut = cut;
                best_partition = partition.clone();
                *improved = true;
            }
        } else {
            break;
        }
    }

    best_partition
}

fn find_best_swap(
    partition: &[bool],
    locked: &[bool],
    n: usize,
    adj: &[Vec<usize>],
) -> (Option<(usize, usize)>, i32) {
    let mut best_gain = i32::MIN;
    let mut best_pair = None;

    for u in 0..n {
        if locked[u] {
            continue;
        }

        let mut du = 0i32;
        for &v in &adj[u] {
            if partition[v] == partition[u] {
                du -= 1;
            } else {
                du += 1;
            }
        }

        for v in u + 1..n {
            if locked[v] || partition[u] == partition[v] {
                continue;
            }

            let mut dv = 0i32;
            for &w in &adj[v] {
                if partition[w] == partition[v] {
                    dv -= 1;
                } else {
                    dv += 1;
                }
            }

            let gain = du + dv - 2 * if adj[u].contains(&v) { 1 } else { 0 };

            if gain > best_gain {
                best_gain = gain;
                best_pair = Some((u, v));
            }
        }
    }

    (best_pair, best_gain)
}

fn calculate_cut(n: usize, partition: &[bool], edges: &[(usize, usize)]) -> usize {
    let mut cut = 0;
    for &(u, v) in edges {
        if partition[u] != partition[v] {
            cut += 1;
        }
    }
    cut
}
