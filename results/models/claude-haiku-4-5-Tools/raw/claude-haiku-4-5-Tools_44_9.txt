#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Create 8x8 checkerboard pattern
    vec2 scaledUV = uv * 8.0;
    ivec2 checkerPos = ivec2(floor(scaledUV));

    // Determine parity: alternates 0.9 and 0.2
    int parity = (checkerPos.x + checkerPos.y) & 1;
    float checkerColor = mix(0.9, 0.2, float(parity));

    // Calculate Lambertian shading
    vec3 normalDir = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(0.0, dot(normalDir, lightDir));

    // Apply shading factor: 0.5 + 0.5 * diffuse
    float shadingFactor = 0.5 + 0.5 * diffuse;
    float finalColor = checkerColor * shadingFactor;

    outColor = vec4(vec3(finalColor), 1.0);
}
