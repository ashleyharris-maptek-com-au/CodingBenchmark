OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in BuiltIn

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out BuiltIn

OpDecorate %worldPos Location 0
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%int_6 = OpConstant %int 6
%float_0_008 = OpConstant %float 0.008
%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1.0
%float_0_0 = OpConstant %float 0.0
%float_m1_0 = OpConstant %float -1.0

; Input gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input per-vertex arrays
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

; Function type
%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

; Edge 0: vertices 0, 2
%pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %pos0
%pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %pos2
%col0 = OpAccessChain %ptr_in_arr_v3 %color %int_0
%c0 = OpLoad %v3float %col0
%col2 = OpAccessChain %ptr_in_arr_v3 %color %int_2
%c2 = OpLoad %v3float %col2
%c_avg_0 = OpVectorTimesScalar %v3float %c0 %float_0_5
%c_avg_1 = OpVectorTimesScalar %v3float %c2 %float_0_5
%c_edge0 = OpFAdd %v3float %c_avg_0 %c_avg_1

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%edge_dir0 = OpFSub %v3float %p2_xyz %p0_xyz
%edge_len_sq0 = OpDot %float %edge_dir0 %edge_dir0
%edge_len0 = OpSqrt %float %edge_len_sq0
%inv_len0 = OpFDiv %float %float_1_0 %edge_len0
%edge_norm0 = OpVectorTimesScalar %v3float %edge_dir0 %inv_len0
%perp0_raw = OpVectorShuffle %v3float %edge_norm0 %edge_norm0 1 0 2
%perp0_x = OpVectorExtractDynamic %float %perp0_raw %int_0
%perp0_x_neg = OpFMul %float %perp0_x %float_m1_0
%perp0 = OpCompositeConstruct %v3float %perp0_x_neg %float_0_0 %float_0_0
%offset0 = OpVectorTimesScalar %v3float %perp0 %float_0_008

%v0_0 = OpFAdd %v3float %p0_xyz %offset0
%v0_0_4 = OpCompositeConstruct %v4float %v0_0 %float_1_0
%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos %v0_0_4
OpStore %outColor %c_edge0
OpEmitVertex

%v0_1 = OpFSub %v3float %p0_xyz %offset0
%v0_1_4 = OpCompositeConstruct %v4float %v0_1 %float_1_0
OpStore %out_pos %v0_1_4
OpStore %outColor %c_edge0
OpEmitVertex

%v0_2 = OpFAdd %v3float %p2_xyz %offset0
%v0_2_4 = OpCompositeConstruct %v4float %v0_2 %float_1_0
OpStore %out_pos %v0_2_4
OpStore %outColor %c_edge0
OpEmitVertex

%v0_3 = OpFSub %v3float %p2_xyz %offset0
%v0_3_4 = OpCompositeConstruct %v4float %v0_3 %float_1_0
OpStore %out_pos %v0_3_4
OpStore %outColor %c_edge0
OpEmitVertex

OpEndPrimitive

; Edge 1: vertices 2, 4
%pos4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %pos4
%col4 = OpAccessChain %ptr_in_arr_v3 %color %int_4
%c4 = OpLoad %v3float %col4
%c_avg_1_0 = OpVectorTimesScalar %v3float %c2 %float_0_5
%c_avg_1_1 = OpVectorTimesScalar %v3float %c4 %float_0_5
%c_edge1 = OpFAdd %v3float %c_avg_1_0 %c_avg_1_1

%p4_xyz = OpVectorShuffle %v3float %p4 %p4 0 1 2
%edge_dir1 = OpFSub %v3float %p4_xyz %p2_xyz
%edge_len_sq1 = OpDot %float %edge_dir1 %edge_dir1
%edge_len1 = OpSqrt %float %edge_len_sq1
%inv_len1 = OpFDiv %float %float_1_0 %edge_len1
%edge_norm1 = OpVectorTimesScalar %v3float %edge_dir1 %inv_len1
%perp1_raw = OpVectorShuffle %v3float %edge_norm1 %edge_norm1 1 0 2
%perp1_x = OpVectorExtractDynamic %float %perp1_raw %int_0
%perp1_x_neg = OpFMul %float %perp1_x %float_m1_0
%perp1 = OpCompositeConstruct %v3float %perp1_x_neg %float_0_0 %float_0_0
%offset1 = OpVectorTimesScalar %v3float %perp1 %float_0_008

%v1_0 = OpFAdd %v3float %p2_xyz %offset1
%v1_0_4 = OpCompositeConstruct %v4float %v1_0 %float_1_0
OpStore %out_pos %v1_0_4
OpStore %outColor %c_edge1
OpEmitVertex

%v1_1 = OpFSub %v3float %p2_xyz %offset1
%v1_1_4 = OpCompositeConstruct %v4float %v1_1 %float_1_0
OpStore %out_pos %v1_1_4
OpStore %outColor %c_edge1
OpEmitVertex

%v1_2 = OpFAdd %v3float %p4_xyz %offset1
%v1_2_4 = OpCompositeConstruct %v4float %v1_2 %float_1_0
OpStore %out_pos %v1_2_4
OpStore %outColor %c_edge1
OpEmitVertex

%v1_3 = OpFSub %v3float %p4_xyz %offset1
%v1_3_4 = OpCompositeConstruct %v4float %v1_3 %float_1_0
OpStore %out_pos %v1_3_4
OpStore %outColor %c_edge1
OpEmitVertex

OpEndPrimitive

; Edge 2: vertices 4, 0
%c_avg_2_0 = OpVectorTimesScalar %v3float %c4 %float_0_5
%c_avg_2_1 = OpVectorTimesScalar %v3float %c0 %float_0_5
%c_edge2 = OpFAdd %v3float %c_avg_2_0 %c_avg_2_1

%edge_dir2 = OpFSub %v3float %p0_xyz %p4_xyz
%edge_len_sq2 = OpDot %float %edge_dir2 %edge_dir2
%edge_len2 = OpSqrt %float %edge_len_sq2
%inv_len2 = OpFDiv %float %float_1_0 %edge_len2
%edge_norm2 = OpVectorTimesScalar %v3float %edge_dir2 %inv_len2
%perp2_raw = OpVectorShuffle %v3float %edge_norm2 %edge_norm2 1 0 2
%perp2_x = OpVectorExtractDynamic %float %perp2_raw %int_0
%perp2_x_neg = OpFMul %float %perp2_x %float_m1_0
%perp2 = OpCompositeConstruct %v3float %perp2_x_neg %float_0_0 %float_0_0
%offset2 = OpVectorTimesScalar %v3float %perp2 %float_0_008

%v2_0 = OpFAdd %v3float %p4_xyz %offset2
%v2_0_4 = OpCompositeConstruct %v4float %v2_0 %float_1_0
OpStore %out_pos %v2_0_4
OpStore %outColor %c_edge2
OpEmitVertex

%v2_1 = OpFSub %v3float %p4_xyz %offset2
%v2_1_4 = OpCompositeConstruct %v4float %v2_1 %float_1_0
OpStore %out_pos %v2_1_4
OpStore %outColor %c_edge2
OpEmitVertex

%v2_2 = OpFAdd %v3float %p0_xyz %offset2
%v2_2_4 = OpCompositeConstruct %v4float %v2_2 %float_1_0
OpStore %out_pos %v2_2_4
OpStore %outColor %c_edge2
OpEmitVertex

%v2_3 = OpFSub %v3float %p0_xyz %offset2
%v2_3_4 = OpCompositeConstruct %v4float %v2_3 %float_1_0
OpStore %out_pos %v2_3_4
OpStore %outColor %c_edge2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
