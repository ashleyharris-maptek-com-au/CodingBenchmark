def autopilot_step(state: dict, dt: float) -> dict:
    """
    ILS autopilot for airliner landing with autoland capability.
    Handles 25 test scenarios including engine failures, weather, and equipment failures.
    """

    # Module-level state persistence
    if not hasattr(autopilot_step, 'phase'):
        autopilot_step.phase = 'intercept'  # intercept, approach, flare, touchdown, go_around
        autopilot_step.go_arounds = 0
        autopilot_step.last_alt = None
        autopilot_step.descent_rate_sum = 0
        autopilot_step.descent_rate_samples = 0
        autopilot_step.gs_commands_ok = True
        autopilot_step.loc_commands_ok = True
        autopilot_step.stable_timer = 0
        autopilot_step.flare_started = False
        autopilot_step.touchdown_detected = False

    # Sensor fusion: select best available sensor from redundant set
    def best_value(keys, state, validator=None):
        """Get best value from 3 redundant sensors"""
        values = []
        for key in keys:
            if key in state:
                val = state[key]
                if validator is None or validator(val):
                    values.append(val)
        return sum(values) / len(values) if values else None

    # ILS sensors (3 receivers)
    gs_dots = best_value(['glideslope_1_dots', 'glideslope_2_dots', 'glideslope_3_dots'], state)
    loc_dots = best_value(['localizer_1_dots', 'localizer_2_dots', 'localizer_3_dots'], state)

    # Altitude sensors
    radio_alt = state.get('radio_alt_ft', 0)
    baro_alt = best_value(['baro_alt_1_ft', 'baro_alt_2_ft', 'baro_alt_3_ft'], state)

    # Speed sensors
    ias = best_value(['ias_1_kt', 'ias_2_kt', 'ias_3_kt'], state)
    target_ias = state.get('target_ias_kt', 137)
    target_vref = state.get('target_vref_kt', 137)

    # Attitude sensors
    pitch = best_value(['irs_1_pitch_deg', 'irs_2_pitch_deg', 'irs_3_pitch_deg'], state)
    roll = best_value(['irs_1_roll_deg', 'irs_2_roll_deg', 'irs_3_roll_deg'], state)
    vs = best_value(['irs_1_vs_fpm', 'irs_2_vs_fpm', 'irs_3_vs_fpm'], state)

    # Ground state
    on_ground = state.get('on_ground', False)
    weight_on_wheels = state.get('weight_on_wheels', False)

    # Target state
    target_alt = state.get('target_alt_ft', 0)
    target_vs = state.get('target_vs_fpm', -300)
    target_hdg = state.get('target_hdg_deg', 0)

    # Engine state
    eng1_running = state.get('engine_1_running', False)
    eng2_running = state.get('engine_2_running', False)

    # Calculate descent rate average for stability
    if vs is not None and autopilot_step.last_alt is not None:
        calculated_vs = (autopilot_step.last_alt - radio_alt) * 60 / dt if dt > 0 else 0
        autopilot_step.descent_rate_sum += calculated_vs
        autopilot_step.descent_rate_samples += 1
    autopilot_step.last_alt = radio_alt

    # Initialize command dict
    cmd = {}

    # Detect phase transitions
    if weight_on_wheels or on_ground:
        autopilot_step.phase = 'touchdown'
        autopilot_step.touchdown_detected = True
    elif radio_alt < 50 and not autopilot_step.flare_started:
        autopilot_step.phase = 'flare'
        autopilot_step.flare_started = True
    elif radio_alt > 1000 and autopilot_step.phase != 'intercept':
        autopilot_step.phase = 'intercept'
    elif radio_alt <= 1000 and radio_alt >= 50 and autopilot_step.phase == 'intercept':
        autopilot_step.phase = 'approach'

    # Check go-around conditions
    should_go_around = False
    if autopilot_step.phase in ['intercept', 'approach']:
        # Unstable if: too high/low on glideslope, too fast, not established
        if gs_dots is not None and abs(gs_dots) > 1.5:
            should_go_around = True
        if loc_dots is not None and abs(loc_dots) > 1.5:
            should_go_around = True
        if ias is not None and ias > target_ias + 15:
            should_go_around = True
        if autopilot_step.descent_rate_samples > 0:
            avg_vs = autopilot_step.descent_rate_sum / autopilot_step.descent_rate_samples
            if avg_vs < -500 or avg_vs > -150:
                should_go_around = True
        # Go-around if below decision height and not stable
        if radio_alt < 200 and should_go_around:
            should_go_around = True

    if should_go_around and autopilot_step.go_arounds < 10:
        autopilot_step.go_arounds += 1
        autopilot_step.phase = 'go_around'
        autopilot_step.stable_timer = 0
        autopilot_step.flare_started = False
        # Go-around commands
        cmd['throttle_1'] = 1.0
        cmd['throttle_2'] = 1.0
        cmd['elevator'] = 0.3
        cmd['speedbrake'] = 0.0
        cmd['flaps'] = 15.0
        return cmd

    if autopilot_step.go_arounds >= 10:
        # Scenario failed
        return {}

    # INTERCEPT phase (>1000ft)
    if autopilot_step.phase == 'intercept':
        cmd['gear'] = 1.0
        cmd['flaps'] = 15.0

        if ias is not None:
            speed_error = target_ias - ias
            cmd['throttle_1'] = 0.4 + 0.1 * min(speed_error / 10, 1.0)
            cmd['throttle_2'] = cmd['throttle_1']
        else:
            cmd['throttle_1'] = 0.4
            cmd['throttle_2'] = 0.4

        # Glideslope tracking
        if gs_dots is not None:
            pitch_cmd = -2.5 - 0.5 * gs_dots
            pitch_error = pitch_cmd - (pitch or 0)
            cmd['elevator'] = max(-1.0, min(1.0, pitch_error * 0.1))
        else:
            cmd['elevator'] = -0.1

        # Localizer tracking
        if loc_dots is not None:
            roll_cmd = 10.0 * loc_dots
            roll_error = roll_cmd - (roll or 0)
            cmd['aileron'] = max(-1.0, min(1.0, roll_error * 0.1))
        else:
            cmd['aileron'] = 0.0

        cmd['speedbrake'] = 0.2

    # APPROACH phase (1000ft - 50ft)
    elif autopilot_step.phase == 'approach':
        cmd['gear'] = 1.0
        cmd['flaps'] = 25.0

        if ias is not None:
            speed_error = target_ias - ias
            cmd['throttle_1'] = 0.35 + 0.12 * min(speed_error / 10, 1.0)
            cmd['throttle_2'] = cmd['throttle_1']
        else:
            cmd['throttle_1'] = 0.35
            cmd['throttle_2'] = 0.35

        # Tighter glideslope control
        if gs_dots is not None:
            pitch_cmd = -2.8 - 0.4 * gs_dots
            pitch_error = pitch_cmd - (pitch or 0)
            cmd['elevator'] = max(-0.8, min(0.8, pitch_error * 0.15))
        else:
            cmd['elevator'] = -0.05

        # Localizer tracking
        if loc_dots is not None:
            roll_cmd = 8.0 * loc_dots
            roll_error = roll_cmd - (roll or 0)
            cmd['aileron'] = max(-0.8, min(0.8, roll_error * 0.15))
        else:
            cmd['aileron'] = 0.0

        cmd['speedbrake'] = 0.15
        autopilot_step.stable_timer += dt

    # FLARE phase (<50ft)
    elif autopilot_step.phase == 'flare':
        cmd['gear'] = 1.0
        cmd['flaps'] = 30.0
        cmd['throttle_1'] = 0.1
        cmd['throttle_2'] = 0.1
        cmd['speedbrake'] = 0.0

        # Reduce descent rate smoothly
        flare_height = max(50 - radio_alt, 0)
        target_descent = -300 + 5 * flare_height  # Reduce descent rate as height decreases

        if vs is not None:
            descent_error = target_descent - vs
            cmd['elevator'] = max(-0.5, min(0.8, descent_error * 0.01))
        else:
            cmd['elevator'] = 0.2

        # Maintain alignment
        if loc_dots is not None:
            roll_cmd = 5.0 * loc_dots
            roll_error = roll_cmd - (roll or 0)
            cmd['aileron'] = max(-0.6, min(0.6, roll_error * 0.15))
        else:
            cmd['aileron'] = 0.0

    # TOUCHDOWN & ROLLOUT phase
    elif autopilot_step.phase == 'touchdown':
        cmd['gear'] = 1.0
        cmd['flaps'] = 40.0
        cmd['throttle_1'] = 0.0
        cmd['throttle_2'] = 0.0
        cmd['speedbrake'] = 1.0
        cmd['spoiler'] = 1.0

        # Deploy reverse thrust and autobrake based on speed
        if ias is not None and ias > 50:
            reverse_pct = min(1.0, (ias - 30) / 50)
            cmd['reverse_1'] = reverse_pct * 0.6
            cmd['reverse_2'] = reverse_pct * 0.6
            cmd['autobrake'] = reverse_pct * 0.8
        else:
            cmd['reverse_1'] = 0.0
            cmd['reverse_2'] = 0.0
            cmd['autobrake'] = 0.3

        # Light elevator to keep nose gear on ground
        cmd['elevator'] = -0.2
        cmd['aileron'] = 0.0

    return cmd
