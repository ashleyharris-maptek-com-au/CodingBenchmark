#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <cmath>

using namespace std;

struct Rect {
    int x, y, w, h;
};

struct Item {
    int w, h, id;
};

struct Placement {
    int board, x, y, rotated;
};

class MaxRectPacker {
private:
    int board_w, board_h;
    vector<Item> items;
    vector<Placement> placements;
    vector<vector<Rect>> free_rects;

public:
    MaxRectPacker(int w, int h) : board_w(w), board_h(h) {}

    void add_item(int w, int h, int id) {
        items.push_back({w, h, id});
    }

    Rect score_rect(const vector<Rect>& rects, int w, int h, int& best_score) {
        best_score = INT_MAX;
        Rect best = {-1, -1, 0, 0};

        for (const auto& r : rects) {
            if (r.w >= w && r.h >= h) {
                int leftoverx = r.w - w;
                int leftovery = r.h - h;
                int score = leftoverx + leftovery;

                if (score < best_score) {
                    best_score = score;
                    best = {r.x, r.y, w, h};
                }
            }
        }
        return best;
    }

    void split_rects(vector<Rect>& rects, const Rect& placed) {
        vector<Rect> new_rects;

        for (const auto& r : rects) {
            if (placed.x + placed.w <= r.x || r.x + r.w <= placed.x ||
                placed.y + placed.h <= r.y || r.y + r.h <= placed.y) {
                new_rects.push_back(r);
            } else {
                if (r.x < placed.x && placed.x - r.x > 0 && r.h > 0) {
                    new_rects.push_back({r.x, r.y, placed.x - r.x, r.h});
                }
                if (r.x + r.w > placed.x + placed.w && r.x + r.w - placed.x - placed.w > 0 && r.h > 0) {
                    new_rects.push_back({placed.x + placed.w, r.y,
                                        r.x + r.w - placed.x - placed.w, r.h});
                }
                if (r.y < placed.y && placed.y - r.y > 0 && placed.w > 0) {
                    new_rects.push_back({placed.x, r.y, placed.w, placed.y - r.y});
                }
                if (r.y + r.h > placed.y + placed.h && r.y + r.h - placed.y - placed.h > 0 && placed.w > 0) {
                    new_rects.push_back({placed.x, placed.y + placed.h, placed.w,
                                        r.y + r.h - placed.y - placed.h});
                }
            }
        }

        rects = new_rects;
    }

    bool pack() {
        vector<pair<int, int>> order;
        for (int i = 0; i < items.size(); i++) {
            order.push_back({items[i].w * items[i].h, i});
        }

        if (items.size() < 2000) {
            sort(order.rbegin(), order.rend());
        }

        free_rects.push_back({{0, 0, board_w, board_h}});

        for (auto [area, idx] : order) {
            int w = items[idx].w;
            int h = items[idx].h;

            bool placed = false;
            for (int b = 0; b < free_rects.size(); b++) {
                int score = INT_MAX;
                Rect rect = score_rect(free_rects[b], w, h, score);

                if (score != INT_MAX) {
                    placements.push_back({b, rect.x, rect.y, 0});
                    split_rects(free_rects[b], rect);
                    placed = true;
                    break;
                }

                if (w != h) {
                    rect = score_rect(free_rects[b], h, w, score);
                    if (score != INT_MAX) {
                        placements.push_back({b, rect.x, rect.y, 1});
                        split_rects(free_rects[b], rect);
                        placed = true;
                        break;
                    }
                }
            }

            if (!placed) {
                free_rects.push_back({{0, 0, board_w, board_h}});

                int score = INT_MAX;
                Rect rect = score_rect(free_rects.back(), w, h, score);
                if (score != INT_MAX) {
                    placements.push_back({(int)free_rects.size() - 1, rect.x, rect.y, 0});
                    split_rects(free_rects.back(), rect);
                } else if (w != h) {
                    rect = score_rect(free_rects.back(), h, w, score);
                    if (score != INT_MAX) {
                        placements.push_back({(int)free_rects.size() - 1, rect.x, rect.y, 1});
                        split_rects(free_rects.back(), rect);
                    }
                }
            }
        }

        return true;
    }

    void output() {
        cout << free_rects.size() << "\n";
        for (const auto& p : placements) {
            cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, bw, bh;
    cin >> n >> bw >> bh;

    MaxRectPacker packer(bw, bh);

    for (int i = 0; i < n; i++) {
        int w, h;
        cin >> w >> h;
        packer.add_item(w, h, i);
    }

    packer.pack();
    packer.output();

    return 0;
}
