#include <bits/stdc++.h>
using namespace std;

int n, m;
vector<vector<int>> adj;
vector<int> best;

void search(vector<int> current, vector<bool> excluded) {
    // Find first non-excluded vertex
    int next = -1;
    int remaining = 0;
    for (int i = 0; i < n; i++) {
        if (!excluded[i]) {
            if (next == -1) next = i;
            remaining++;
        }
    }

    if (next == -1) {
        // All vertices decided
        if (current.size() > best.size()) {
            best = current;
        }
        return;
    }

    // Pruning: if even including all remaining vertices can't beat best, skip
    if ((int)current.size() + remaining <= (int)best.size()) {
        return;
    }

    // Branch 1: include next vertex
    current.push_back(next);
    vector<bool> excluded1 = excluded;
    excluded1[next] = true;
    for (int v : adj[next]) {
        excluded1[v] = true;
    }
    search(current, excluded1);

    // Branch 2: exclude next vertex
    current.pop_back();
    vector<bool> excluded2 = excluded;
    excluded2[next] = true;
    search(current, excluded2);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m;
    adj.resize(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> current;
    vector<bool> excluded(n, false);
    search(current, excluded);

    cout << best.size() << "\n";
    for (int i = 0; i < best.size(); i++) {
        if (i > 0) cout << " ";
        cout << best[i];
    }
    cout << "\n";

    return 0;
}
