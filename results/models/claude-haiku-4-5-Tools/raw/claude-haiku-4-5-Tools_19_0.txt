using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

class MiningOptimizer
{
    struct Voxel
    {
        public byte mass;
        public byte value;
    }

    struct Pos3D
    {
        public int x, y, z;
        public override bool Equals(object obj) => obj is Pos3D p && p.x == x && p.y == y && p.z == z;
        public override int GetHashCode() => x ^ (y << 10) ^ (z << 20);
        public static bool operator ==(Pos3D a, Pos3D b) => a.x == b.x && a.y == b.y && a.z == b.z;
        public static bool operator !=(Pos3D a, Pos3D b) => !(a == b);
    }

    int xCount, yCount, zCount;
    double xBlockSize, yBlockSize, zBlockSize;
    Voxel[,,] voxels;
    HashSet<Pos3D> dug = new();
    Dictionary<Pos3D, int> groundHeight;
    List<string> operations = new();

    void ReadInput()
    {
        var header = Console.ReadLine().Split(' ');
        xCount = int.Parse(header[0]);
        yCount = int.Parse(header[1]);
        zCount = int.Parse(header[2]);
        xBlockSize = double.Parse(header[3]);
        yBlockSize = double.Parse(header[4]);
        zBlockSize = double.Parse(header[5]);

        voxels = new Voxel[xCount, yCount, zCount];
        groundHeight = new Dictionary<Pos3D, int>();

        using (var stdin = Console.OpenStandardInput())
        {
            for (int x = 0; x < xCount; x++)
            {
                for (int y = 0; y < yCount; y++)
                {
                    for (int z = 0; z < zCount; z++)
                    {
                        int b = stdin.ReadByte();
                        voxels[x, y, z].mass = (byte)(b % 16);
                        voxels[x, y, z].value = (byte)(b / 16);
                    }
                }
            }
        }
    }

    int GetHeightAt(int x, int y)
    {
        int maxZ = 0;
        for (int z = 0; z < zCount; z++)
            if (voxels[x, y, z].mass > 0 && !dug.Contains(new Pos3D { x = x, y = y, z = z }))
                maxZ = z + 1;
        return maxZ;
    }

    bool IsAccessible(Pos3D p)
    {
        if (dug.Contains(p)) return false;
        if (p.x < 0 || p.x >= xCount || p.y < 0 || p.y >= yCount || p.z < 0 || p.z >= zCount) return false;
        if (voxels[p.x, p.y, p.z].mass == 0) return false;

        if (p.z == zCount - 1) return true;
        if (dug.Contains(new Pos3D { x = p.x, y = p.y, z = p.z + 1 })) return true;
        if (voxels[p.x, p.y, p.z + 1].mass == 0) return true;

        var sides = new[] { (1, 0), (-1, 0), (0, 1), (0, -1) };
        foreach (var (dx, dy) in sides)
        {
            int nx = p.x + dx, ny = p.y + dy;
            if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) return true;
            if (dug.Contains(new Pos3D { x = nx, y = ny, z = p.z })) return true;
            if (voxels[nx, ny, p.z].mass == 0) return true;
        }
        return false;
    }

    List<(int, int)> FindPathToPlant(int sx, int sy, int sz)
    {
        var path = new List<(int, int)>();
        int cx = sx, cy = sy;
        int targetX = 0, targetY = 0;

        while (cx != targetX || cy != targetY)
        {
            int dx = 0, dy = 0;
            int minDist = int.MaxValue;

            var moves = new[] { (1, 0), (-1, 0), (0, 1), (0, -1) };
            foreach (var (mx, my) in moves)
            {
                int nx = cx + mx, ny = cy + my;
                if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) continue;

                int currentHeight = GetHeightAt(cx, cy);
                int nextHeight = GetHeightAt(nx, ny);
                int slope = nextHeight - currentHeight;

                if (slope > 1) continue;

                int dist = Math.Abs(nx - targetX) + Math.Abs(ny - targetY);
                if (dist < minDist)
                {
                    minDist = dist;
                    dx = mx;
                    dy = my;
                }
            }

            if (dx == 0 && dy == 0) break;
            cx += dx;
            cy += dy;
            path.Add((dx, dy));
        }

        return path;
    }

    void Solve()
    {
        var candidates = new List<(int x, int y, int z, int profitScore)>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    var pos = new Pos3D { x = x, y = y, z = z };
                    if (voxels[x, y, z].mass > 0 && voxels[x, y, z].value > 0 && IsAccessible(pos))
                    {
                        int score = voxels[x, y, z].value * voxels[x, y, z].mass;
                        candidates.Add((x, y, z, score));
                    }
                }
            }
        }

        candidates = candidates.OrderByDescending(c => c.profitScore).ToList();

        foreach (var (ox, oy, oz, _) in candidates.Take(Math.Min(200, candidates.Count)))
        {
            var pos = new Pos3D { x = ox, y = oy, z = oz };
            if (dug.Contains(pos) || !IsAccessible(pos)) continue;

            var path = FindPathToPlant(ox, oy, oz);
            if (path.Count == 0) continue;

            var opStr = $"1 {ox} {oy} {oz}";
            foreach (var (dx, dy) in path)
                opStr += $" {dx} {dy}";
            opStr += " 0 0";

            operations.Add(opStr);
            dug.Add(pos);
        }

        Console.WriteLine(operations.Count);
        foreach (var op in operations)
            Console.WriteLine(op);
    }

    static void Main()
    {
        var solver = new MiningOptimizer();
        solver.ReadInput();
        solver.Solve();
    }
}
