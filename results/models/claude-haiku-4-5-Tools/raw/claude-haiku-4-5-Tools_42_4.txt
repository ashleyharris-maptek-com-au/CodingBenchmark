OpCapability Geometry
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor

OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inColor Location 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%main_type = OpTypeFunction %void

%main = OpFunction %void None %main_type
%entry = OpLabel

%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr

%col0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%col0 = OpLoad %v3float %col0_ptr
%col1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%col1 = OpLoad %v3float %col1_ptr

%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1

%diff = OpFSub %v2float %pos1_xy %pos0_xy

%diff_x = OpVectorExtractDynamic %float %diff %int_0
%diff_y = OpVectorExtractDynamic %float %diff %int_1
%neg_diff_y = OpFNegate %float %diff_y
%perp = OpCompositeConstruct %v2float %neg_diff_y %diff_x

%perp_dot = OpDot %float %perp %perp
%perp_len = OpSqrt %float %perp_dot
%perp_len_vec = OpCompositeConstruct %v2float %perp_len %perp_len
%perp_norm = OpFDiv %v2float %perp %perp_len_vec

%offset_start = OpVectorTimesScalar %v2float %perp_norm %float_0_04
%offset_end = OpVectorTimesScalar %v2float %perp_norm %float_0_005

%offset_start_neg = OpFNegate %v2float %offset_start
%offset_end_neg = OpFNegate %v2float %offset_end

%v1_xy = OpFAdd %v2float %pos0_xy %offset_start
%v1 = OpVectorShuffle %v4float %pos0 %v1_xy 4 5 2 3
%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos %v1
%out_col = OpAccessChain %ptr_out_v3 %outColor %int_0
OpStore %out_col %col0
OpEmitVertex

%v2_xy = OpFAdd %v2float %pos0_xy %offset_start_neg
%v2 = OpVectorShuffle %v4float %pos0 %v2_xy 4 5 2 3
OpStore %out_pos %v2
OpStore %out_col %col0
OpEmitVertex

%v3_xy = OpFAdd %v2float %pos1_xy %offset_end
%v3 = OpVectorShuffle %v4float %pos1 %v3_xy 4 5 2 3
OpStore %out_pos %v3
OpStore %out_col %col1
OpEmitVertex

%v4_xy = OpFAdd %v2float %pos1_xy %offset_end_neg
%v4 = OpVectorShuffle %v4float %pos1 %v4_xy 4 5 2 3
OpStore %out_pos %v4
OpStore %out_col %col1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
