OpCapability Geometry

OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %color_in %gl_out %outColor

%outColor = OpVariable %ptr_out_v3 Output

%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_Function_int = OpTypePointer Function %int

%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%entry = OpLabel

%pos_chain = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %pos_chain

%pos_x = OpCompositeExtract %float %in_pos 0
%pos_y = OpCompositeExtract %float %in_pos 1
%pos_z = OpCompositeExtract %float %in_pos 2
%pos_w = OpCompositeExtract %float %in_pos 3

%i_var = OpVariable %ptr_Function_int Function %int_0

OpBranch %loop_header

%loop_header = OpLabel
%i_load = OpLoad %int %i_var
%cond = OpSLessThan %bool %i_load %int_6
OpLoopMerge %loop_exit %loop_cont None
OpBranchConditional %cond %loop_body %loop_exit

%loop_body = OpLabel
%i_cur = OpLoad %int %i_var

%out_pos_center = OpCompositeConstruct %v4float %pos_x %pos_y %pos_z %pos_w
%center_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %center_ptr %out_pos_center
OpStore %outColor %in_color
OpEmitVertex

%edge0_case0_x = %float_0_04
%edge0_case0_y = %float_0

%case_eq_0 = OpIEqual %bool %i_cur %int_0
%case_eq_1 = OpIEqual %bool %i_cur %int_1
%case_eq_2 = OpIEqual %bool %i_cur %int_2
%case_eq_3 = OpIEqual %bool %i_cur %int_3
%case_eq_4 = OpIEqual %bool %i_cur %int_4

%edge0_x = OpSelect %float %case_eq_0 %edge0_case0_x %float_0
%edge0_y = OpSelect %float %case_eq_0 %edge0_case0_y %float_0
%temp_x = OpSelect %float %case_eq_1 %edge0_case1_x %edge0_x
%temp_y = OpSelect %float %case_eq_1 %edge0_case1_y %edge0_y
%edge0_x_2 = OpSelect %float %case_eq_2 %edge0_case2_x %temp_x
%edge0_y_2 = OpSelect %float %case_eq_2 %edge0_case2_y %temp_y
%temp2_x = OpSelect %float %case_eq_3 %edge0_case3_x %edge0_x_2
%temp2_y = OpSelect %float %case_eq_3 %edge0_case3_y %edge0_y_2
%edge0_final_x = OpSelect %float %case_eq_4 %edge0_case4_x %temp2_x
%edge0_final_y = OpSelect %float %case_eq_4 %edge0_case4_y %temp2_y

%final_edge0_x = OpFAdd %float %pos_x %edge0_final_x
%final_edge0_y = OpFAdd %float %pos_y %edge0_final_y
%e0_pos = OpCompositeConstruct %v4float %final_edge0_x %final_edge0_y %pos_z %pos_w
%e0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %e0_ptr %e0_pos
OpStore %outColor %darker
OpEmitVertex

%i_next = OpIAdd %int %i_cur %int_1
%i_next_mod = OpUMod %int %i_next %int_6

%edge1_case0_x = %float_0_02
%edge1_case0_y = %float_0_03464
%edge1_case1_x = %float_neg0_02
%edge1_case1_y = %float_0_03464
%edge1_case2_x = %float_neg0_04
%edge1_case2_y = %float_0
%edge1_case3_x = %float_neg0_02
%edge1_case3_y = %float_neg0_03464
%edge1_case4_x = %float_0_02
%edge1_case4_y = %float_neg0_03464
%edge1_case5_x = %float_0_04
%edge1_case5_y = %float_0

%case1_eq_0 = OpIEqual %bool %i_next_mod %int_0
%case1_eq_1 = OpIEqual %bool %i_next_mod %int_1
%case1_eq_2 = OpIEqual %bool %i_next_mod %int_2
%case1_eq_3 = OpIEqual %bool %i_next_mod %int_3
%case1_eq_4 = OpIEqual %bool %i_next_mod %int_4

%edge1_x = OpSelect %float %case1_eq_0 %edge1_case0_x %float_0
%edge1_y = OpSelect %float %case1_eq_0 %edge1_case0_y %float_0
%temp1_x = OpSelect %float %case1_eq_1 %edge1_case1_x %edge1_x
%temp1_y = OpSelect %float %case1_eq_1 %edge1_case1_y %edge1_y
%edge1_x_2 = OpSelect %float %case1_eq_2 %edge1_case2_x %temp1_x

%edge1_final_x = OpSelect %float %case1_eq_4 %edge1_case4_x %temp3_x

%final_edge1_x = OpFAdd %float %pos_x %edge1_final_x
