const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { probability, drones, maxStepsPerDrone } = data;

const height = probability.length;
const width = probability[0].length;

const visited = new Set();

function isValid(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height && probability[y][x] !== -1;
}

function getNeighbors(x, y) {
  return [
    [x, y],
    [x + 1, y],
    [x - 1, y],
    [x, y + 1],
    [x, y - 1]
  ].filter(([nx, ny]) => isValid(nx, ny));
}

function getCellValue(x, y) {
  const key = `${x},${y}`;
  if (visited.has(key)) {
    return 0;
  }
  return probability[y][x] || 0;
}

const paths = drones.map(drone => [[drone.x, drone.y]]);

// Mark starting positions as visited
for (const drone of drones) {
  const key = `${drone.x},${drone.y}`;
  if (probability[drone.y][drone.x] > 0) {
    visited.add(key);
  }
}

// Simulate movement steps
for (let step = 1; step < maxStepsPerDrone; step++) {
  for (let i = 0; i < drones.length; i++) {
    const [currentX, currentY] = paths[i][paths[i].length - 1];

    const neighbors = getNeighbors(currentX, currentY);

    let bestX = currentX;
    let bestY = currentY;
    let bestValue = 0;

    for (const [nx, ny] of neighbors) {
      const value = getCellValue(nx, ny);
      if (value > bestValue) {
        bestValue = value;
        bestX = nx;
        bestY = ny;
      }
    }

    paths[i].push([bestX, bestY]);

    const key = `${bestX},${bestY}`;
    if (probability[bestY][bestX] > 0) {
      visited.add(key);
    }
  }
}
