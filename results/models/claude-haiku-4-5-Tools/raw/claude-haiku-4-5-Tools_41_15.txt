%v3_cool = OpConstantComposite %v3float %float_0p2 %float_0p2 %float_0p75

; Main function

; Load inputs
%normal = OpLoad %v3float %in_normal
%worldPos = OpLoad %v3float %in_worldPos

; Normalize normal
%normal_normalized = OpExtInst %v3float %ext Normalize %normal

; Get light position from UBO
%light_ptr = OpAccessChain %ptr_uniform_v4float %ubo_var %int_3
%light_pos_vec4 = OpLoad %v4float %light_ptr
%light_pos = OpVectorShuffle %v3float %light_pos_vec4 %light_pos_vec4 0 1 2

; Calculate light direction
%light_dir_unnorm = OpFSub %v3float %light_pos %worldPos
%light_dir = OpExtInst %v3float %ext Normalize %light_dir_unnorm

; Calculate dot product N·L
%dot_product = OpDot %float %normal_normalized %light_dir

; Calculate t = dot(N,L) * 0.5 + 0.5
%t_scaled = OpFMul %float %dot_product %float_0p5
%t = OpFAdd %float %t_scaled %float_0p5

; Clamp t to [0, 1]
%t_clamped = OpExtInst %float %ext FClamp %t %float_0 %float_1

; Blend: color = cool * (1 - t) + warm * t
%one_minus_t = OpFSub %float %float_1 %t_clamped
%cool_scaled = OpVectorTimesScalar %v3float %v3_cool %one_minus_t
%warm_scaled = OpVectorTimesScalar %v3float %v3_warm %t_clamped
%final_color_rgb = OpFAdd %v3float %cool_scaled %warm_scaled

; Construct output vec4 with alpha = 1.0
%result = OpCompositeConstruct %v4float %final_color_rgb %float_1

; Store to output
OpStore %out_color %result

OpReturn
