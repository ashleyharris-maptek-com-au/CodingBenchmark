@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    // Guard: if out of range, return
    if (gid.x >= params.z) {
        return;
    }

    let my_nonce = params.y + gid.x;

    // Initialize state with baseData XOR nonce
    var state : array<u32, 16>;
    for (var i = 0u; i < 16u; i++) {
        state[i] = baseData[i] ^ (my_nonce + i);
    }

    // 8 rounds of quarter-round mixing
    for (var round = 0u; round < 8u; round++) {
        // Group 0: state[0..3]
        state[0] += state[1];
        state[3] ^= state[0];
        state[3] = (state[3] << 16u) | (state[3] >> 16u);

        state[2] += state[3];
        state[1] ^= state[2];
        state[1] = (state[1] << 12u) | (state[1] >> 20u);

        state[0] += state[1];
        state[3] ^= state[0];
        state[3] = (state[3] << 8u) | (state[3] >> 24u);

        state[2] += state[3];
        state[1] ^= state[2];
        state[1] = (state[1] << 7u) | (state[1] >> 25u);

        // Group 1: state[4..7]
        state[4] += state[5];
        state[7] ^= state[4];
        state[7] = (state[7] << 16u) | (state[7] >> 16u);

        state[6] += state[7];
        state[5] ^= state[6];
        state[5] = (state[5] << 12u) | (state[5] >> 20u);

        state[4] += state[5];
        state[7] ^= state[4];
        state[7] = (state[7] << 8u) | (state[7] >> 24u);

        state[6] += state[7];
        state[5] ^= state[6];
        state[5] = (state[5] << 7u) | (state[5] >> 25u);

        // Group 2: state[8..11]
        state[8] += state[9];
        state[11] ^= state[8];
        state[11] = (state[11] << 16u) | (state[11] >> 16u);

        state[10] += state[11];
        state[9] ^= state[10];
        state[9] = (state[9] << 12u) | (state[9] >> 20u);

        state[8] += state[9];
        state[11] ^= state[8];
        state[11] = (state[11] << 8u) | (state[11] >> 24u);

        state[10] += state[11];
        state[9] ^= state[10];
        state[9] = (state[9] << 7u) | (state[9] >> 25u);

        // Group 3: state[12..15]
        state[12] += state[13];
        state[15] ^= state[12];
        state[15] = (state[15] << 16u) | (state[15] >> 16u);

        state[14] += state[15];
        state[13] ^= state[14];
        state[13] = (state[13] << 12u) | (state[13] >> 20u);

        state[12] += state[13];
        state[15] ^= state[12];
        state[15] = (state[15] << 8u) | (state[15] >> 24u);

        state[14] += state[15];
        state[13] ^= state[14];
        state[13] = (state[13] << 7u) | (state[13] >> 25u);
    }

    // Fold 16 words to 2
    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    // Count trailing zeros in 64-bit value [h0 | (h1 << 32)]
    var trailing_zeros = 0u;
    if (h0 == 0u) {
        trailing_zeros = 32u + countTrailingZeros(h1);
    } else {
        trailing_zeros = countTrailingZeros(h0);
    }

    // Check if trailing_zeros >= difficulty
    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = my_nonce;
        result[2] = h0;
        result[3] = h1;
    }
}

fn countTrailingZeros(x : u32) -> u32 {
    if (x == 0u) {
        return 32u;
    }

    var count = 0u;
    var val = x;

    if ((val & 0xFFFFu) == 0u) {
        count += 16u;
        val >>= 16u;
    }
    if ((val & 0xFFu) == 0u) {
        count += 8u;
        val >>= 8u;
    }
    if ((val & 0xFu) == 0u) {
        count += 4u;
        val >>= 4u;
    }
    if ((val & 0x3u) == 0u) {
        count += 2u;
        val >>= 2u;
    }
    if ((val & 0x1u) == 0u) {
        count += 1u;
    }

    return count;
}
