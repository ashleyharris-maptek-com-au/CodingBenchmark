using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    private int H, W;
    private char[][] board;
    private bool[][] isMine;
    private bool[][] isSafe;
    private bool[][] isUnknown;

    static void Main()
    {
        var solver = new MinesweeperSolver();
        solver.Solve();
    }

    void Solve()
    {
        ReadInput();
        PropagateConstraints();
        OutputSafeCells();
    }

    void ReadInput()
    {
        var line = Console.ReadLine();
        var parts = line.Split(' ');
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        board = new char[H][];
        isMine = new bool[H][];
        isSafe = new bool[H][];
        isUnknown = new bool[H][];

        for (int i = 0; i < H; i++)
        {
            board[i] = Console.ReadLine().ToCharArray();
            isMine[i] = new bool[W];
            isSafe[i] = new bool[W];
            isUnknown[i] = new bool[W];

            for (int j = 0; j < W; j++)
            {
                if (board[i][j] == '*')
                    isMine[i][j] = true;
                else if (board[i][j] == ' ')
                    isUnknown[i][j] = true;
                else
                    isSafe[i][j] = true;
            }
        }
    }

    void PropagateConstraints()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;

            for (int i = 0; i < H; i++)
            {
                for (int j = 0; j < W; j++)
                {
                    if (char.IsDigit(board[i][j]))
                    {
                        if (ProcessNumberedCell(i, j))
                            changed = true;
                    }
                }
            }
        }
    }

    bool ProcessNumberedCell(int row, int col)
    {
        int mineCount = board[row][col] - '0';
        var neighbors = GetNeighbors(row, col);

        int flaggedCount = neighbors.Count(pos => isMine[pos.r][pos.c]);
        int unknownCount = neighbors.Count(pos => isUnknown[pos.r][pos.c]);

        bool changed = false;

        // All mines found - mark unknowns as safe
        if (flaggedCount == mineCount)
        {
            foreach (var (r, c) in neighbors)
            {
                if (isUnknown[r][c])
                {
                    isUnknown[r][c] = false;
                    isSafe[r][c] = true;
                    changed = true;
                }
            }
        }
        // Need all remaining unknowns as mines
        else if (flaggedCount + unknownCount == mineCount)
        {
            foreach (var (r, c) in neighbors)
            {
                if (isUnknown[r][c])
                {
                    isUnknown[r][c] = false;
                    isMine[r][c] = true;
                    changed = true;
                }
            }
        }

        return changed;
    }

    List<(int r, int c)> GetNeighbors(int row, int col)
    {
        var result = new List<(int, int)>();
        for (int dr = -1; dr <= 1; dr++)
        {
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int nr = row + dr;
                int nc = col + dc;
                if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                    result.Add((nr, nc));
            }
        }
        return result;
    }

    void OutputSafeCells()
    {
        var safeCells = new List<(int x, int y)>();

        for (int i = 0; i < H; i++)
        {
            for (int j = 0; j < W; j++)
            {
                if (isSafe[i][j] && board[i][j] == ' ')
                    safeCells.Add((j, i));
            }
        }

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
            Console.WriteLine($"{x} {y}");
    }
}
