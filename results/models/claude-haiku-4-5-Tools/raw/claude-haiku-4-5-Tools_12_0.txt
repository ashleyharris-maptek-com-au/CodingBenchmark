using System;
using System.Collections.Generic;
using System.Linq;
using System.Diagnostics;

class BinPackingSolver
{
    struct Vector3
    {
        public double X, Y, Z;
        public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vector3 operator+(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator-(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator*(double s, Vector3 v) => new Vector3(s * v.X, s * v.Y, s * v.Z);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Dot(Vector3 v) => X * v.X + Y * v.Y + Z * v.Z;
        public Vector3 Cross(Vector3 v) => new Vector3(Y * v.Z - Z * v.Y, Z * v.X - X * v.Z, X * v.Y - Y * v.X);
        public Vector3 Normalize() { double len = Length(); return len > 0 ? new Vector3(X/len, Y/len, Z/len) : this; }
    }

    struct Quaternion
    {
        public double W, X, Y, Z;
        public Quaternion(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        public static Quaternion Identity() => new Quaternion(1, 0, 0, 0);
        public Vector3 RotatePoint(Vector3 p)
        {
            double xx = X * X, yy = Y * Y, zz = Z * Z, xy = X * Y, xz = X * Z, yz = Y * Z, wx = W * X, wy = W * Y, wz = W * Z;
            return new Vector3(
                (1 - 2 * (yy + zz)) * p.X + 2 * (xy - wz) * p.Y + 2 * (xz + wy) * p.Z,
                2 * (xy + wz) * p.X + (1 - 2 * (xx + zz)) * p.Y + 2 * (yz - wx) * p.Z,
                2 * (xz - wy) * p.X + 2 * (yz + wx) * p.Y + (1 - 2 * (xx + yy)) * p.Z
            );
        }
    }

    class Polyhedron
    {
        public Vector3[] Vertices;
        public int[][] Faces;
        public Vector3 Center;
        public double Radius;

        public Polyhedron(Vector3[] verts, int[][] faces)
        {
            Vertices = verts;
            Faces = faces;
            Center = new Vector3(
                verts.Average(v => v.X),
                verts.Average(v => v.Y),
                verts.Average(v => v.Z)
            );
            Radius = verts.Max(v => (v - Center).Length());
        }
    }

    class Placement
    {
        public Vector3 Translation;
        public Quaternion Rotation;
        public Vector3[] TransformedVertices;
    }

    static Vector3[] ReadVertices(int count)
    {
        var verts = new Vector3[count];
        for (int i = 0; i < count; i++)
        {
            var parts = Console.ReadLine().Split();
            verts[i] = new Vector3(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2]));
        }
        return verts;
    }

    static int[][] ReadFaces(int count)
    {
        var faces = new int[count][];
        for (int i = 0; i < count; i++)
        {
            var parts = Console.ReadLine().Split();
            faces[i] = new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) };
        }
        return faces;
    }

    static Vector3[] TransformVertices(Vector3[] verts, Vector3 translation, Quaternion rotation)
    {
        return verts.Select(v => translation + rotation.RotatePoint(v)).ToArray();
    }

    static bool IsInsideContainer(Vector3[] verts, double w, double h, double d)
    {
        return verts.All(v => v.X >= 0 && v.X <= w && v.Y >= 0 && v.Y <= h && v.Z >= 0 && v.Z <= d);
    }

    static bool CheckCollision(Vector3[] v1, Vector3[] v2)
    {
        if (v1.Length == 0 || v2.Length == 0) return false;
        var aabb1Min = new Vector3(v1.Min(v => v.X), v1.Min(v => v.Y), v1.Min(v => v.Z));
        var aabb1Max = new Vector3(v1.Max(v => v.X), v1.Max(v => v.Y), v1.Max(v => v.Z));
        var aabb2Min = new Vector3(v2.Min(v => v.X), v2.Min(v => v.Y), v2.Min(v => v.Z));
        var aabb2Max = new Vector3(v2.Max(v => v.X), v2.Max(v => v.Y), v2.Max(v => v.Z));

        return aabb1Min.X <= aabb2Max.X && aabb1Max.X >= aabb2Min.X &&
               aabb1Min.Y <= aabb2Max.Y && aabb1Max.Y >= aabb2Min.Y &&
               aabb1Min.Z <= aabb2Max.Z && aabb1Max.Z >= aabb2Min.Z;
    }

    static List<Placement> PackGreedy(Polyhedron poly, double w, double h, double d, double timeLimit)
    {
        var placements = new List<Placement>();
        var sw = Stopwatch.StartNew();
        int maxAttempts = 1000;

        if (w > 100) maxAttempts = 50;
        if (w > 200) maxAttempts = 20;

        for (int attempt = 0; attempt < maxAttempts && sw.Elapsed.TotalSeconds < timeLimit; attempt++)
        {
            var rand = new Random(attempt);
            var rotation = new Quaternion(1, 0, 0, 0);

            if (attempt > 0 && w <= 100)
            {
                double angle = (attempt % 4) * Math.PI / 2;
                rotation = new Quaternion(Math.Cos(angle / 2), 0, 0, Math.Sin(angle / 2));
            }

            var xCandidates = GenerateCandidates(w, rand, 10);
            var yCandidates = GenerateCandidates(h, rand, 5);
            var zCandidates = GenerateCandidates(d, rand, 5);

            foreach (var tx in xCandidates)
            {
                foreach (var ty in yCandidates)
                {
                    foreach (var tz in zCandidates)
                    {
                        var translation = new Vector3(tx, ty, tz);
                        var transformed = TransformVertices(poly.Vertices, translation, rotation);

                        if (!IsInsideContainer(transformed, w, h, d)) continue;
                        if (placements.Any(p => CheckCollision(p.TransformedVertices, transformed))) continue;

                        placements.Add(new Placement { Translation = translation, Rotation = rotation, TransformedVertices = transformed });
                        break;
                    }
                    if (placements.Count > attempt / 5 + 1) break;
                }
                if (placements.Count > attempt / 5 + 1) break;
            }
        }

        return placements;
    }

    static List<double> GenerateCandidates(double max, Random rand, int count)
    {
        var candidates = new List<double> { 0, max / 2, Math.Max(0, max - 1) };
        for (int i = 0; i < count; i++)
            candidates.Add(rand.NextDouble() * max);
        return candidates.Distinct().ToList();
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int V = int.Parse(firstLine[0]);
        int F = int.Parse(firstLine[1]);

        var vertices = ReadVertices(V);
        var faces = ReadFaces(F);

        string[] lastLine = Console.ReadLine().Split();
        double W = double.Parse(lastLine[0]);
        double H = double.Parse(lastLine[1]);
        double D = double.Parse(lastLine[2]);

        var poly = new Polyhedron(vertices, faces);
        double timeLimit = 300;

        var placements = PackGreedy(poly, W, H, D, timeLimit);

        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X:F6} {p.Translation.Y:F6} {p.Translation.Z:F6} {p.Rotation.W:F6} {p.Rotation.X:F6} {p.Rotation.Y:F6} {p.Rotation.Z:F6}");
        }
    }
}
