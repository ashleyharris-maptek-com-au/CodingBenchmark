using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int N = int.Parse(line[0]);
        int W = int.Parse(line[1]);
        int H = int.Parse(line[2]);
        int D = int.Parse(line[3]);

        var boxes = new List<(int index, int w, int h, int d)>();
        for (int i = 0; i < N; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes.Add((i, int.Parse(dims[0]), int.Parse(dims[1]), int.Parse(dims[2])));
        }

        // Sort by volume descending for better packing
        boxes = boxes.OrderByDescending(b => (long)b.w * b.h * b.d).ToList();

        var packed = new List<(int, int, int, int)>();
        var spaces = new List<(int x, int y, int z, int w, int h, int d)>
        {
            (0, 0, 0, W, H, D)
        };

        foreach (var (idx, bw, bh, bd) in boxes)
        {
            bool placed = false;

            // Try all 6 permutations of box dimensions
            var perms = new[]
            {
                (bw, bh, bd), (bw, bd, bh),
                (bh, bw, bd), (bh, bd, bw),
                (bd, bw, bh), (bd, bh, bw)
            };

            foreach (var (pw, ph, pd) in perms)
            {
                if (placed) break;

                // Find first space that fits
                for (int sIdx = 0; sIdx < spaces.Count; sIdx++)
                {
                    var (sx, sy, sz, sw, sh, sd) = spaces[sIdx];

                    if (pw <= sw && ph <= sh && pd <= sd)
                    {
                        // Place box
                        packed.Add((idx, sx, sy, sz));
                        spaces.RemoveAt(sIdx);

                        // Guillotine split: create new spaces from the remainder
                        if (sw - pw > 0)
                            spaces.Add((sx + pw, sy, sz, sw - pw, sh, sd));
                        if (sh - ph > 0)
                            spaces.Add((sx, sy + ph, sz, pw, sh - ph, sd));
                        if (sd - pd > 0)
                            spaces.Add((sx, sy, sz + pd, sw, sh, sd - pd));

                        placed = true;
                        break;
                    }
                }
            }
        }

        Console.WriteLine(packed.Count);
        foreach (var (idx, x, y, z) in packed)
            Console.WriteLine($"{idx} {x} {y} {z}");
    }
}
