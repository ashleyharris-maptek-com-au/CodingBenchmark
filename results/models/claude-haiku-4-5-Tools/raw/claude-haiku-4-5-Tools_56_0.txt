const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

function distance(p1, p2) {
  const dx = p1[0] - p2[0];
  const dy = p1[1] - p2[1];
  return Math.sqrt(dx * dx + dy * dy);
}

function routeDistance(route, depot, packages) {
  let total = distance(depot, packages[route[0]]);
  for (let i = 1; i < route.length; i++) {
    total += distance(packages[route[i - 1]], packages[route[i]]);
  }
  total += distance(packages[route[route.length - 1]], depot);
  return total;
}

function solveVRP(depot, packages, maxVehicles) {
  if (packages.length === 0) return [];

  const n = packages.length;
  const packageIds = Array.from({length: n}, (_, i) => i);

  // Nearest neighbor starting from depot
  let routes = [];
  let remaining = new Set(packageIds);

  while (remaining.size > 0 && routes.length < maxVehicles) {
    let route = [];
    let current = depot;

    while (remaining.size > 0) {
      let nearest = null;
      let nearestDist = Infinity;

      for (const pkg of remaining) {
        const d = distance(current, packages[pkg]);
        if (d < nearestDist) {
          nearestDist = d;
          nearest = pkg;
        }
      }

      if (nearest === null) break;

      route.push(nearest);
      remaining.delete(nearest);
      current = packages[nearest];

      // Estimate cost of going back to depot
      const returnCost = distance(current, depot);

      // If remaining packages are too far, start new route
      if (remaining.size > 0 && routes.length < maxVehicles - 1) {
        let minNextDist = Infinity;
        for (const pkg of remaining) {
          const d = distance(current, packages[pkg]);
          if (d < minNextDist) minNextDist = d;
        }
        if (returnCost < minNextDist * 0.3) break;
      }
    }

    if (route.length > 0) {
      routes.push(route);
    }
  }

  // 2-opt improvement
  let improved = true;
  let iterations = 0;
  while (improved && iterations < 100) {
    improved = false;
    iterations++;

    for (let i = 0; i < routes.length; i++) {
      for (let j = i; j < routes.length; j++) {
        if (routes[i].length < 2 && routes[j].length < 2) continue;

        // Try 2-opt within route
        if (i === j && routes[i].length > 3) {
          for (let a = 0; a < routes[i].length - 1; a++) {
            for (let b = a + 2; b < routes[i].length; b++) {
              const old = distance(packages[routes[i][a]], packages[routes[i][a+1]]) +
                         distance(packages[routes[i][b]], packages[routes[i][(b+1) % routes[i].length]]);
              const next = distance(packages[routes[i][a]], packages[routes[i][b]]) +
                          distance(packages[routes[i][a+1]], packages[routes[i][(b+1) % routes[i].length]]);

              if (next < old - 1e-6) {
                let newRoute = routes[i].slice(0, a + 1);
                for (let k = b; k > a; k--) {
                  newRoute.push(routes[i][k]);
                }
                for (let k = b + 1; k < routes[i].length; k++) {
                  newRoute.push(routes[i][k]);
                }
                routes[i] = newRoute;
                improved = true;
              }
            }
          }
        }
      }
    }
  }

  return routes;
}

const depot = data.depot;
const packages = data.packages;
const truckCount = data.truckCount;

const routes = solveVRP(depot, packages, truckCount);
const result = { routes: routes };
