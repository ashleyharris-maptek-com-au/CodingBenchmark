using System;
using System.Collections.Generic;
using System.Linq;

class CuttingStockSolver
{
    static void Main()
    {
        var line1 = Console.ReadLine().Split(' ');
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        var cuts = Console.ReadLine().Split(' ').Select(int.Parse).ToArray();

        // Create indexed cuts and sort by length descending
        var indexedCuts = Enumerable.Range(0, n)
            .Select(i => (index: i, length: cuts[i]))
            .OrderByDescending(x => x.length)
            .ToList();

        var stocks = new List<List<int>>();
        var stockSpaceLeft = new List<int>();

        // First Fit Decreasing algorithm
        foreach (var (index, length) in indexedCuts)
        {
            bool placed = false;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockSpaceLeft[i] >= length)
                {
                    stocks[i].Add(index);
                    stockSpaceLeft[i] -= length;
                    placed = true;
                    break;
                }
            }

            if (!placed)
            {
                stocks.Add(new List<int> { index });
                stockSpaceLeft.Add(stockLength - length);
            }
        }

        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
