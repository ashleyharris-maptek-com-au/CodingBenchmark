def autopilot_step(state: dict, dt: float) -> dict:
    import statistics

    # Module-level state for persistence
    if not hasattr(autopilot_step, 'initialized'):
        autopilot_step.initialized = True
        autopilot_step.steering_integral = 0.0
        autopilot_step.speed_integral = 0.0
        autopilot_step.lane_history = []
        autopilot_step.speed_history = []
        autopilot_step.heading_history = []
        autopilot_step.last_valid_steering = 0.0
        autopilot_step.last_valid_throttle = 0.0
        autopilot_step.last_valid_brake = 0.0

    # Sensor fusion helpers
    def fuse_values(val1, valid1, val2, valid2, val3=None, valid3=None):
        """Fuse redundant sensor values using median of valid readings."""
        values = []
        if valid1 and val1 is not None:
            values.append(val1)
        if valid2 and val2 is not None:
            values.append(val2)
        if val3 is not None and valid3 and val3 is not None:
            values.append(val3)
        return statistics.median(values) if values else None

    def fuse_speed_kmh():
        """Fuse wheel speeds and GPS speeds."""
        wheel_speeds = [
            state.get('wheel_speed_fl_kmh', 0) if state.get('wheel_speed_fl_valid') else None,
            state.get('wheel_speed_fr_kmh', 0) if state.get('wheel_speed_fr_valid') else None,
            state.get('wheel_speed_rl_kmh', 0) if state.get('wheel_speed_rl_valid') else None,
            state.get('wheel_speed_rr_kmh', 0) if state.get('wheel_speed_rr_valid') else None,
        ]
        wheel_speeds = [s for s in wheel_speeds if s is not None]

        gps_speeds = []
        if state.get('gps_1_valid'):
            gps_speeds.append(state.get('gps_1_speed_kmh', 0))
        if state.get('gps_2_valid'):
            gps_speeds.append(state.get('gps_2_speed_kmh', 0))

        all_speeds = wheel_speeds + gps_speeds
        if all_speeds:
            return statistics.median(all_speeds)
        return state.get('vehicle_speed_1_kmh', 0)

    def fuse_heading():
        """Fuse gyro yaw rate and GPS heading."""
        gyro_rates = []
        if state.get('gyro_1_valid'):
            gyro_rates.append(state.get('gyro_1_yaw_rate_dps', 0))
        if state.get('gyro_2_valid'):
            gyro_rates.append(state.get('gyro_2_yaw_rate_dps', 0))
        if state.get('gyro_3_valid'):
            gyro_rates.append(state.get('gyro_3_yaw_rate_dps', 0))

        if gyro_rates:
            autopilot_step.heading_history.append(statistics.median(gyro_rates) * dt)

        gps_heading = None
        if state.get('gps_1_valid') and state.get('gps_2_valid'):
            gps_heading = (state.get('gps_1_heading_deg', 0) + state.get('gps_2_heading_deg', 0)) / 2
        elif state.get('gps_1_valid'):
            gps_heading = state.get('gps_1_heading_deg', 0)

        return gps_heading

    def fuse_lateral_accel():
        """Fuse accelerometer lateral readings."""
        accels = []
        if state.get('accel_1_valid'):
            accels.append(state.get('accel_1_lateral_ms2', 0))
        if state.get('accel_2_valid'):
            accels.append(state.get('accel_2_lateral_ms2', 0))
        if state.get('accel_3_valid'):
            accels.append(state.get('accel_3_lateral_ms2', 0))

        return statistics.median(accels) if accels else 0

    # Get current speed
    current_speed_kmh = fuse_speed_kmh()
    target_speed_kmh = state.get('target_speed_kmh', 100)

    # Check speed limits from vision
    vision_speed_limits = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_speed_limit_valid'):
            limit = state.get(f'vision_{i}_speed_limit_kmh', target_speed_kmh)
            if limit > 0:
                vision_speed_limits.append(limit)

    if vision_speed_limits:
        target_speed_kmh = min(target_speed_kmh, statistics.median(vision_speed_limits))

    # Get friction coefficient
    mu = state.get('estimated_mu', 1.0)

    # Adjust target speed for low traction
    if mu < 0.5:
        target_speed_kmh = min(target_speed_kmh, 50)
    elif mu < 0.7:
        target_speed_kmh = min(target_speed_kmh, 80)

    # Obstacle detection and avoidance
    obstacle_dist = float('inf')
    obstacle_rel_speed = 0
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_obstacle_valid') and state.get(f'vision_{i}_obstacle_detected'):
            dist = state.get(f'vision_{i}_obstacle_dist_m', float('inf'))
            if dist < obstacle_dist:
                obstacle_dist = dist
                obstacle_rel_speed = state.get(f'vision_{i}_obstacle_rel_speed_kmh', 0)

    # Check lidar for obstacles
    lidar_min_dist = float('inf')
    for angle in [5, 15, 30, 45, 60, 90, -5, -15, -30, -45, -60, -90]:
        key_range = f'lidar_{angle:+d}_deg_range_m'
        key_valid = f'lidar_{angle:+d}_deg_valid'
        if state.get(key_valid):
            dist = state.get(key_range, float('inf'))
            if abs(angle) <= 45:  # Forward-facing beams
                lidar_min_dist = min(lidar_min_dist, dist)

    obstacle_dist = min(obstacle_dist, lidar_min_dist)

    # STEERING CONTROL - Lane keeping
    steering = 0.0

    # Get lane offset from vision
    lane_offsets = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_left_line_valid') and state.get(f'vision_{i}_right_line_valid'):
            offset = state.get(f'vision_{i}_lane_offset_m', 0)
            lane_offsets.append(offset)

    if lane_offsets:
        lane_offset = statistics.median(lane_offsets)
    else:
        lane_offset = 0.0

    # Get road curvature
    curvatures = []
    for i in [1, 2, 3]:
        if state.get(f'vision_{i}_left_line_valid') and state.get(f'vision_{i}_right_line_valid'):
            curv = state.get(f'vision_{i}_road_curvature_1pm', 0)
            curvatures.append(curv)

    road_curvature = statistics.median(curvatures) if curvatures else 0.0

    # Feed-forward steering for road curvature
    if road_curvature != 0:
        steering = min(1.0, max(-1.0, road_curvature * 0.1))

    # PID control for lane offset
    lane_error = lane_offset
    autopilot_step.steering_integral += lane_error * dt
    autopilot_step.steering_integral = max(-0.5, min(0.5, autopilot_step.steering_integral))

    kp_steering = 0.3 if mu > 0.7 else 0.15
    ki_steering = 0.05
    kd_steering = 0.15

    steering_correction = kp_steering * lane_error + ki_steering * autopilot_step.steering_integral
    steering = min(1.0, max(-1.0, steering + steering_correction))

    # Check steering sensor agreement
    steering_valid1 = state.get('steering_angle_1_valid')
    steering_valid2 = state.get('steering_angle_2_valid')
    steering_angle_1 = state.get('steering_angle_1_deg', 0) if steering_valid1 else None
    steering_angle_2 = state.get('steering_angle_2_deg', 0) if steering_valid2 else None

    if steering_valid1 and steering_valid2:
        if abs(steering_angle_1 - steering_angle_2) > 10:  # Disagree
            steering = autopilot_step.last_valid_steering

    # THROTTLE/BRAKE CONTROL - Speed management
    throttle = 0.0
    brake = 0.0

    speed_error = target_speed_kmh - current_speed_kmh
    autopilot_step.speed_integral += speed_error * dt
    autopilot_step.speed_integral = max(-100, min(100, autopilot_step.speed_integral))

    kp_speed = 0.015 if mu > 0.7 else 0.008
    ki_speed = 0.002

    control_signal = kp_speed * speed_error + ki_speed * autopilot_step.speed_integral

    # Emergency braking for obstacles
    emergency_brake = False
    if obstacle_dist < 10:
        brake = 1.0
        throttle = 0.0
        emergency_brake = True
    elif obstacle_dist < 20:
        brake = 0.7
        throttle = 0.0
        emergency_brake = True
    elif obstacle_dist < 30 and obstacle_rel_speed < -10:  # Approaching fast
        brake = 0.4
        throttle = 0.0
        emergency_brake = True

    if not emergency_brake:
        if control_signal > 0:
            throttle = min(1.0, control_signal)
            brake = 0.0
        else:
            brake = min(1.0, -control_signal * 0.5)
            throttle = 0.0

    # Check brake pressure sensors
    brake_valid1 = state.get('brake_pressure_1_valid')
    brake_valid2 = state.get('brake_pressure_2_valid')

    if brake_valid1 and brake_valid2:
        p1 = state.get('brake_pressure_1_pct', 0)
        p2 = state.get('brake_pressure_2_pct', 0)
        if abs(p1 - p2) > 30 and brake > 0:  # Brake system failure
            brake = 0.0

    # Clamp outputs
    steering = max(-1.0, min(1.0, steering))
    throttle = max(0.0, min(1.0, throttle))
    brake = max(0.0, min(1.0, brake))

    # Apply traction management
    if mu < 0.3:  # Ice
        if abs(steering) > 0.3:
            steering = 0.3 if steering > 0 else -0.3
        throttle = min(throttle, 0.3)
    elif mu < 0.5:
        throttle = min(throttle, 0.6)

    # Store last valid values
    autopilot_step.last_valid_steering = steering
    autopilot_step.last_valid_throttle = throttle
    autopilot_step.last_valid_brake = brake

    return {
        'steering': steering,
        'throttle': throttle,
        'brake': brake
    }
