[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    float3 normal = normalize(input.normal);

    // Scale UV coordinates: 8 bricks wide, 16 bricks tall
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Determine row and apply staggering
    float row = floor(scaledUV.y);
    float rowFrac = frac(scaledUV.y);
    float staggerOffset = mod(row, 2.0) * 0.5;
    float adjustedX = scaledUV.x + staggerOffset;
    float colFrac = frac(adjustedX);

    // Check if we're in mortar
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;

    bool inMortarX = (colFrac < mortarThicknessX) || (colFrac > (1.0 - mortarThicknessX));
    bool inMortarY = (rowFrac < mortarThicknessY) || (rowFrac > (1.0 - mortarThicknessY));
    bool inMortar = inMortarX || inMortarY;

    // Select color
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 baseColor = inMortar ? mortarColor : brickColor;

    // Lambertian shading
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(normal, lightDir));
    float lambertian = 0.4 + 0.6 * diffuse;

    // Apply shading to base color
    float3 finalColor = baseColor * lambertian;

    return float4(finalColor, 1.0);
}
