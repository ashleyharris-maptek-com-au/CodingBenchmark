const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

function bfsDistance(grid, start, end) {
  if (start.x === end.x && start.y === end.y) return 0;

  const rows = grid.length;
  const cols = grid[0].length;
  const visited = new Set();
  const queue = [[start.x, start.y, 0]];
  visited.add(`${start.x},${start.y}`);

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift();

    const neighbors = [[x+1, y], [x-1, y], [x, y+1], [x, y-1]];

    for (const [nx, ny] of neighbors) {
      if (nx >= 0 && nx < cols && ny >= 0 && ny < rows &&
          grid[ny][nx] === '.' && !visited.has(`${nx},${ny}`)) {

        if (nx === end.x && ny === end.y) {
          return dist + 1;
        }

        visited.add(`${nx},${ny}`);
        queue.push([nx, ny, dist + 1]);
      }
    }
  }

  return Infinity;
}

const robots = data.robots;
const jobs = data.jobs;
const grid = data.grid;

const numRobots = robots.length;
const numJobs = jobs.length;

const assignment = Array.from({ length: numRobots }, () => []);
const assignedJobs = new Set();
const robotPos = robots.map(r => ({ x: r.x, y: r.y }));

while (assignedJobs.size < numJobs) {
  let bestJob = -1;
  let bestRobot = -1;
  let minDist = Infinity;

  for (let jobId = 0; jobId < numJobs; jobId++) {
    if (assignedJobs.has(jobId)) continue;

    for (let robotId = 0; robotId < numRobots; robotId++) {
      const dist = bfsDistance(grid, robotPos[robotId], jobs[jobId]);

      if (dist < minDist) {
        minDist = dist;
        bestJob = jobId;
        bestRobot = robotId;
      }
    }
  }

  if (bestJob === -1) break;

  assignment[bestRobot].push(bestJob);
  robotPos[bestRobot] = jobs[bestJob];
  assignedJobs.add(bestJob);
}

console.log(JSON.stringify({ assignments: assignment }));
