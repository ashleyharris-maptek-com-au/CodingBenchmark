#include <iostream>
#include <vector>
#include <algorithm>

int n, m;
std::vector<long long> c;
std::vector<std::pair<int, int>> constraints;
long long best_obj = 0;
std::vector<int> best_x;

void solve(int idx, std::vector<int>& x, long long current_obj) {
    // Pruning: if current can't beat best, skip
    if (idx < n) {
        long long upper_bound = current_obj;
        for (int i = idx; i < n; i++) {
            if (c[i] > 0) upper_bound += c[i];
        }
        if (upper_bound <= best_obj) {
            return;
        }
    }

    if (idx == n) {
        if (current_obj > best_obj) {
            best_obj = current_obj;
            best_x = x;
        }
        return;
    }

    // Try x[idx] = 1
    bool can_set_one = true;
    for (const auto& [u, v] : constraints) {
        if ((u == idx && x[v] == 1) || (v == idx && x[u] == 1)) {
            can_set_one = false;
            break;
        }
    }

    if (can_set_one) {
        x[idx] = 1;
        solve(idx + 1, x, current_obj + c[idx]);
        x[idx] = 0;
    }

    // Try x[idx] = 0
    solve(idx + 1, x, current_obj);
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::cin >> n >> m;

    c.resize(n);
    for (int i = 0; i < n; i++) {
        std::cin >> c[i];
    }

    constraints.resize(m);
    for (int i = 0; i < m; i++) {
        std::cin >> constraints[i].first >> constraints[i].second;
    }

    best_x.resize(n, 0);
    std::vector<int> x(n, 0);
    solve(0, x, 0);

    // Output
    std::cout << best_obj << "\n";
    for (int i = 0; i < n; i++) {
        if (i > 0) std::cout << " ";
        std::cout << best_x[i];
    }
    std::cout << "\n";

    return 0;
}
