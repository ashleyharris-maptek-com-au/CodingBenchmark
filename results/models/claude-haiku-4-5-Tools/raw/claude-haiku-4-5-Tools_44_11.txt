#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Get grid coordinates for 6x6 pattern
    vec2 gridPos = uv * 6.0;

    // Get position within current cell (0 to 1)
    vec2 cellPos = fract(gridPos);

    // Distance from cell center (0.5, 0.5)
    vec2 fromCenter = cellPos - vec2(0.5);
    float dist = length(fromCenter);

    // Determine if we're in a circular dot (radius 0.3)
    vec3 baseColor;
    if (dist <= 0.3) {
        baseColor = vec3(1.0, 0.8, 0.0);  // yellow
    } else {
        baseColor = vec3(0.1, 0.1, 0.3);  // dark blue
    }

    // Lambertian shading
    vec3 norm = normalize(normal);
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(0.0, dot(norm, lightDir));
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Apply shading and output
    vec3 shadedColor = baseColor * shadingFactor;
    outColor = vec4(shadedColor, 1.0);
}
