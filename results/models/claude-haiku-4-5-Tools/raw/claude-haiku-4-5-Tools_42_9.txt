OpCapability Geometry
OpCapability Float32
OpCapability Vector16

OpExtInstImport "GLSL.std.450"

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor

OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %outColor Location 0

; Basic types
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_025 = OpConstant %float 0.025

; GLSL std450 ext set
%glsl = OpExtInstImport "GLSL.std.450"

; Input gl_PerVertex structure
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

; Input vertex arrays
%arr_v3 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Output gl_PerVertex structure
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Function type
%func_type = OpTypeFunction %void

; Input/Output variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func_type
%bb_entry = OpLabel

; Load 4 positions from gl_in
%pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr

; Load colors [1] and [2]
%color1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%color2 = OpLoad %v3float %color2_ptr

; Extract xyz components
%pos0_xyz = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
%pos1_xyz = OpVectorShuffle %v3float %pos1 %pos1 0 1 2
%pos2_xyz = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
%pos3_xyz = OpVectorShuffle %v3float %pos3 %pos3 0 1 2

; Compute tangent at start: t0 = normalize(pos2 - pos0)
%diff0 = OpFSub %v3float %pos2_xyz %pos0_xyz
%t0 = OpExtInst %v3float %glsl Normalize %diff0

; Compute tangent at end: t1 = normalize(pos3 - pos1)
%diff1 = OpFSub %v3float %pos3_xyz %pos1_xyz
%t1 = OpExtInst %v3float %glsl Normalize %diff1

; Extract xy components for perpendicular
%t0_xy = OpVectorShuffle %v2float %t0 %t0 0 1
%t1_xy = OpVectorShuffle %v2float %t1 %t1 0 1

; Extract x, y components
%t0_x = OpCompositeExtract %float %t0_xy 0
%t0_y = OpCompositeExtract %float %t0_xy 1
%t1_x = OpCompositeExtract %float %t1_xy 0
%t1_y = OpCompositeExtract %float %t1_xy 1

; Compute perpendiculars: perp = vec2(-y, x)
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; Scale by thickness (0.025)
%perp0_scaled = OpVectorTimesScalar %v2float %perp0 %float_025
%perp1_scaled = OpVectorTimesScalar %v2float %perp1 %float_025

; Extract position components
%pos1_xy = OpVectorShuffle %v2float %pos1_xyz %pos1_xyz 0 1
%pos1_z = OpCompositeExtract %float %pos1_xyz 2
%pos2_xy = OpVectorShuffle %v2float %pos2_xyz %pos2_xyz 0 1
%pos2_z = OpCompositeExtract %float %pos2_xyz 2

; Vertex 0: start + perp0*thickness
%v0_xy = OpFAdd %v2float %pos1_xy %perp0_scaled
%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %pos1_z %float_1

%out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos %v0_pos
OpStore %outColor %color1
OpEmitVertex

; Vertex 1: start - perp0*thickness
%v1_xy = OpFSub %v2float %pos1_xy %perp0_scaled
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %pos1_z %float_1

OpStore %out_pos %v1_pos
OpStore %outColor %color1
OpEmitVertex

; Vertex 2: end + perp1*thickness
%v2_xy = OpFAdd %v2float %pos2_xy %perp1_scaled
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %pos2_z %float_1

OpStore %out_pos %v2_pos
OpStore %outColor %color2
OpEmitVertex

; Vertex 3: end - perp1*thickness
%v3_xy = OpFSub %v2float %pos2_xy %perp1_scaled
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %pos2_z %float_1

OpStore %out_pos %v3_pos
OpStore %outColor %color2
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
