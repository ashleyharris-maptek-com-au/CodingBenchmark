#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>
#include <iomanip>

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    double length() const { return std::sqrt(x * x + y * y + z * z); }
    Vec3 normalized() const { double len = length(); return len > 1e-9 ? *this * (1.0 / len) : *this; }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz;
    double circumradius;

    Tetrahedron() : circumradius(0.6124) {}
};

struct Face {
    Vec3 p0, p1, p2;

    Face(Vec3 p0, Vec3 p1, Vec3 p2) : p0(p0), p1(p1), p2(p2) {}

    Vec3 normal() const {
        Vec3 e1 = p1 - p0;
        Vec3 e2 = p2 - p0;
        double nx = e1.y * e2.z - e1.z * e2.y;
        double ny = e1.z * e2.x - e1.x * e2.z;
        double nz = e1.x * e2.y - e1.y * e2.x;
        return Vec3(nx, ny, nz).normalized();
    }

    double signedDistance(const Vec3& p) const {
        Vec3 n = normal();
        return n.dot(p - p0);
    }
};

class Container {
public:
    std::vector<Vec3> vertices;
    std::vector<Face> faces;
    Vec3 center;
    double minX, maxX, minY, maxY, minZ, maxZ;

    void computeProperties() {
        if (vertices.empty()) return;

        minX = maxX = vertices[0].x;
        minY = maxY = vertices[0].y;
        minZ = maxZ = vertices[0].z;

        double cx = 0, cy = 0, cz = 0;
        for (const auto& v : vertices) {
            cx += v.x; cy += v.y; cz += v.z;
            minX = std::min(minX, v.x); maxX = std::max(maxX, v.x);
            minY = std::min(minY, v.y); maxY = std::max(maxY, v.y);
            minZ = std::min(minZ, v.z); maxZ = std::max(maxZ, v.z);
        }
        center = Vec3(cx / vertices.size(), cy / vertices.size(), cz / vertices.size());
    }

    bool isInside(const Vec3& p) const {
        for (const auto& face : faces) {
            if (face.signedDistance(p) < -1e-6) {
                return false;
            }
        }
        return true;
    }
};

bool checkCollision(const Tetrahedron& t1, const Tetrahedron& t2) {
    double dist = (t1.center - t2.center).length();
    return dist < (t1.circumradius + t2.circumradius) * 2.0 - 1e-6;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edgeLength;
    std::cin >> V >> edgeLength;

    Container container;
    for (int i = 0; i < V; i++) {
        double x, y, z;
        std::cin >> x >> y >> z;
        container.vertices.push_back(Vec3(x, y, z));
    }

    container.computeProperties();

    if (V == 4) {
        for (int i = 0; i < 4; i++) {
            for (int j = i + 1; j < 4; j++) {
                for (int k = j + 1; k < 4; k++) {
                    container.faces.push_back(Face(
                        container.vertices[i],
                        container.vertices[j],
                        container.vertices[k]
                    ));
                }
            }
        }
    } else if (V == 8) {
        std::vector<std::vector<int>> faceIndices = {
            {0, 1, 2, 3}, {4, 5, 6, 7}, {0, 1, 5, 4},
            {2, 3, 7, 6}, {0, 3, 7, 4}, {1, 2, 6, 5}
        };
        for (auto& fi : faceIndices) {
            if (fi.size() >= 3) {
                container.faces.push_back(Face(
                    container.vertices[fi[0]],
                    container.vertices[fi[1]],
                    container.vertices[fi[2]]
                ));
            }
        }
    } else {
        int n = std::min((int)container.vertices.size(), 12);
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                for (int k = j + 1; k < n; k++) {
                    container.faces.push_back(Face(
                        container.vertices[i],
                        container.vertices[j],
                        container.vertices[k]
                    ));
                }
            }
        }
    }

    std::vector<Tetrahedron> placed;
    double spacing = 0.6124 * 2.5;

    for (double x = container.minX; x <= container.maxX; x += spacing) {
        for (double y = container.minY; y <= container.maxY; y += spacing) {
            for (double z = container.minZ; z <= container.maxZ; z += spacing) {
                Tetrahedron t;
                t.center = Vec3(x, y, z);
                t.rx = 0.0; t.ry = 0.0; t.rz = 0.0;

                if (container.isInside(t.center)) {
                    bool collision = false;
                    for (const auto& p : placed) {
                        if (checkCollision(t, p)) {
                            collision = true;
                            break;
                        }
                    }
                    if (!collision) {
                        placed.push_back(t);
                    }
                }
            }
        }
    }

    std::mt19937 gen(123);
    std::uniform_real_distribution<> dis(0, 1);
    int maxAttempts = std::min(10000, (int)std::pow(container.vertices.size(), 2) * 100);

    for (int attempt = 0; attempt < maxAttempts && placed.size() < 100000; attempt++) {
        Tetrahedron t;
        t.center = Vec3(
            container.minX + dis(gen) * (container.maxX - container.minX),
            container.minY + dis(gen) * (container.maxY - container.minY),
            container.minZ + dis(gen) * (container.maxZ - container.minZ)
        );
        t.rx = dis(gen) * 2 * M_PI;
        t.ry = dis(gen) * 2 * M_PI;
        t.rz = dis(gen) * 2 * M_PI;

        if (container.isInside(t.center)) {
            bool collision = false;
            for (const auto& p : placed) {
                if (checkCollision(t, p)) {
                    collision = true;
                    break;
                }
            }
            if (!collision) {
                placed.push_back(t);
            }
        }
    }

    std::cout << placed.size() << "\n";
    std::cout << std::fixed << std::setprecision(6);
    for (const auto& t : placed) {
        std::cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
                  << t.rx << " " << t.ry << " " << t.rz << "\n";
    }

    return 0;
}
