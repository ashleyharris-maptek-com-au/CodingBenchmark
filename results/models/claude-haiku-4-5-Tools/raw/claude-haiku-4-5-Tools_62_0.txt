declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
}

const data: InputData = JSON.parse(input);

function getCell(grid: string[], x: number, y: number): string {
  if (y < 0 || y >= grid.length || x < 0 || x >= grid[0].length) {
    return '~';
  }
  return grid[y][x];
}

function isBurnable(grid: string[], x: number, y: number): boolean {
  return getCell(grid, x, y) === '.';
}

function simulateFire(
  grid: string[],
  ignitions: [number, number][],
  cutCells: Set<string>
): Set<string> {
  const burned = new Set<string>();
  const queue: Array<[number, number]> = [];

  // Initialize with ignition points that aren't cut
  for (const [x, y] of ignitions) {
    const key = `${x},${y}`;
    if (!cutCells.has(key)) {
      queue.push([x, y]);
      burned.add(key);
    }
  }

  // BFS fire spread
  while (queue.length > 0) {
    const [x, y] = queue.shift()!;

    // Spread to 4 adjacent cells
    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;
      const key = `${nx},${ny}`;

      if (!burned.has(key) && !cutCells.has(key) && isBurnable(grid, nx, ny)) {
        burned.add(key);
        queue.push([nx, ny]);
      }
    }
  }

  return burned;
}

function calculateLoss(towns: Town[], burned: Set<string>): number {
  let loss = 0;
  for (const town of towns) {
    const key = `${town.x},${town.y}`;
    if (burned.has(key)) {
      loss += town.value;
    }
  }
  return loss;
}

function findFrontier(burned: Set<string>, cuts: Set<string>, grid: string[]): Set<string> {
  const frontier = new Set<string>();

  for (const key of burned) {
    const [x, y] = key.split(',').map(Number);

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;
      const nkey = `${nx},${ny}`;

      if (!cuts.has(nkey) && !burned.has(nkey) && isBurnable(grid, nx, ny)) {
        frontier.add(nkey);
      }
    }
  }

  return frontier;
}

const cuts = new Set<string>();

// Greedy algorithm: iteratively place cuts that save the most value
for (let iteration = 0; iteration < data.cutBudget; iteration++) {
  const currentBurned = simulateFire(data.grid, data.ignitions, cuts);
  const currentLoss = calculateLoss(data.towns, currentBurned);

  if (currentLoss === 0) break;

  const frontier = findFrontier(currentBurned, cuts, data.grid);

  if (frontier.size === 0) break;

  let bestCell: string | null = null;
  let bestSavings = 0;

  // Evaluate each frontier cell
  for (const cell of frontier) {
    const testCuts = new Set(cuts);
    testCuts.add(cell);
    const newBurned = simulateFire(data.grid, data.ignitions, testCuts);
    const newLoss = calculateLoss(data.towns, newBurned);
    const savings = currentLoss - newLoss;

    if (savings > bestSavings) {
      bestSavings = savings;
      bestCell = cell;
    }
  }

  if (bestCell === null || bestSavings === 0) break;
  cuts.add(bestCell);
}

const result = {
  firebreaks: Array.from(cuts).map(key => {
    const [x, y] = key.split(',').map(Number);
    return [x, y];
  })
};
