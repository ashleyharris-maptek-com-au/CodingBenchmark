; Bound: 60

OpCapability Shader
OpExtInstImport "GLSL.std.450" %glsl_ext
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %out_color
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_normal Location 1
OpDecorate %out_color Location 0

; Type declarations
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4

%float_0 = OpConstant %float 0.0
%float_0p5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0

%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4

%in_normal = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

%main_type = OpTypeFunction %void

; Main function
%main = OpFunction %void None %main_type
%entry = OpLabel

; Load the normal
%normal = OpLoad %vec3 %in_normal

; Normalize the normal using GLSL extended instruction (opcode 1 is Normalize)
%normalized = OpExtInst %vec3 %glsl_ext 1 %normal

; Map from [-1, 1] to [0, 1]: (normal + 1) / 2
%one_vec = OpConstantComposite %vec3 %float_1 %float_1 %float_1
%added = OpFAdd %vec3 %normalized %one_vec
%two_vec = OpConstantComposite %vec3 %float_2 %float_2 %float_2
%mapped = OpFDiv %vec3 %added %two_vec

; Extract components for vec4 construction
%x = OpCompositeExtract %float %mapped 0
%y = OpCompositeExtract %float %mapped 1
%z = OpCompositeExtract %float %mapped 2

; Create output color with alpha = 1.0
%color = OpCompositeConstruct %vec4 %x %y %z %float_1

OpStore %out_color %color

OpReturn
OpFunctionEnd
