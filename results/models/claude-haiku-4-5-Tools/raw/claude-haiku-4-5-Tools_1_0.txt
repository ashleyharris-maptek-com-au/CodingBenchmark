#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <limits>
#include <cstring>

using namespace std;

struct Point {
    float x, y;
};

inline float distance(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return sqrt(dx * dx + dy * dy);
}

// Nearest neighbor heuristic - O(n²)
vector<int> nearestNeighbor(const vector<Point>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour.back();
        int nearest = -1;
        float minDist = numeric_limits<float>::max();

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = distance(cities[last], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// 2-opt local search - improves tour quality
void twoOpt(const vector<Point>& cities, vector<int>& tour, int maxIterations) {
    int n = tour.size();
    bool improved = true;
    int iter = 0;

    while (improved && iter < maxIterations) {
        improved = false;
        iter++;

        for (int i = 0; i < n - 2 && !improved; i++) {
            for (int j = i + 2; j < n; j++) {
                float d1 = distance(cities[tour[i]], cities[tour[i + 1]]) +
                          distance(cities[tour[j]], cities[tour[(j + 1) % n]]);
                float d2 = distance(cities[tour[i]], cities[tour[j]]) +
                          distance(cities[tour[i + 1]], cities[tour[(j + 1) % n]]);

                if (d2 < d1 - 1e-9f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    break;
                }
            }
        }
    }
}

// Hilbert curve mapping for space-filling curve
uint32_t xy2d(uint32_t x, uint32_t y) {
    uint32_t d = 0;
    for (uint32_t s = 1 << 15; s > 0; s >>= 1) {
        uint32_t rx = (x & s) > 0;
        uint32_t ry = (y & s) > 0;
        d += s * s * ((3 * rx) ^ ry);

        if (ry == 0) {
            if (rx == 1) {
                x = (1 << 16) - 1 - x;
                y = (1 << 16) - 1 - y;
            }
            swap(x, y);
        }
    }
    return d;
}

// Space-filling curve approximation for huge N
vector<int> spaceFilling(const vector<Point>& cities) {
    int n = cities.size();

    // Find bounding box
    float minX = cities[0].x, maxX = cities[0].x;
    float minY = cities[0].y, maxY = cities[0].y;

    for (const auto& p : cities) {
        minX = min(minX, p.x);
        maxX = max(maxX, p.x);
        minY = min(minY, p.y);
        maxY = max(maxY, p.y);
    }

    float rangeX = max(maxX - minX, 1e-6f);
    float rangeY = max(maxY - minY, 1e-6f);

    // Compute Hilbert curve indices and sort
    vector<pair<uint32_t, int>> indices;
    indices.reserve(n);

    for (int i = 0; i < n; i++) {
        uint32_t x = (uint32_t)((cities[i].x - minX) / rangeX * 65535.0f);
        uint32_t y = (uint32_t)((cities[i].y - minY) / rangeY * 65535.0f);
        uint32_t d = xy2d(x, y);
        indices.emplace_back(d, i);
    }

    sort(indices.begin(), indices.end());

    vector<int> tour;
    tour.reserve(n);
    for (const auto& [d, idx] : indices) {
        tour.push_back(idx);
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> tour;

    if (n <= 40) {
        // Trivial: Nearest neighbor + aggressive 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(cities, tour, 5000);
    } else if (n <= 500) {
        // Small-medium: Nearest neighbor + strong 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(cities, tour, 1000);
    } else if (n <= 5000) {
        // Medium: Nearest neighbor + moderate 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(cities, tour, 200);
    } else if (n <= 50000) {
        // Large: Nearest neighbor + light 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(cities, tour, 50);
    } else {
        // Extreme/Epic: Space-filling curve (Hilbert)
        tour = spaceFilling(cities);
    }

    // Output
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
