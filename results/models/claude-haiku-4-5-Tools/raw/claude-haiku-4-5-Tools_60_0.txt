const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Build dependency graph and task map
const taskMap = {};
const dependents = {};
const inDegree = {};

tasks.forEach(task => {
  taskMap[task.id] = task;
  inDegree[task.id] = (task.deps || []).length;
  if (!dependents[task.id]) {
    dependents[task.id] = [];
  }
  (task.deps || []).forEach(dep => {
    if (!dependents[dep]) {
      dependents[dep] = [];
    }
    dependents[dep].push(task.id);
  });
});

// Initialize machine states
const machines = Array(machineCount).fill(0).map(() => ({
  availableAt: 0,
  tasks: []
}));

// Track scheduled tasks
const scheduled = new Set();
const scheduledTasks = [];
const finishTimes = {};

// Priority queue based on deadline and weight heuristic
function getPriority(task) {
  // Lower deadline = higher priority
  // Higher weight = higher priority
  // Combine as: deadline / weight (lower is better)
  const deadline = task.deadline || Infinity;
  const weight = task.weight || 1;
  return deadline / Math.max(weight, 1);
}

// Greedy scheduling
while (scheduled.size < tasks.length) {
  // Find all tasks with satisfied dependencies
  const available = [];

  for (const taskId in taskMap) {
    if (scheduled.has(taskId)) continue;

    const task = taskMap[taskId];
    const deps = task.deps || [];

    // Check if all dependencies are scheduled
    if (deps.every(depId => scheduled.has(depId))) {
      available.push(task);
    }
  }

  if (available.length === 0) break;

  // Sort by priority (deadline/weight ratio)
  available.sort((a, b) => getPriority(a) - getPriority(b));

  // For each available task, find best machine to schedule it on
  let bestTask = null;
  let bestMachine = -1;
  let bestCost = Infinity;

  for (const task of available) {
    // Calculate earliest start time (after all dependencies finish)
    let earliestStart = 0;
    if (task.deps && task.deps.length > 0) {
      earliestStart = Math.max(...task.deps.map(depId => finishTimes[depId] || 0));
    }

    // Try each machine
    for (let machineIdx = 0; machineIdx < machineCount; machineIdx++) {
      const machine = machines[machineIdx];
      const startTime = Math.max(earliestStart, machine.availableAt);
      const finishTime = startTime + task.duration;

      // Cost: weighted lateness if deadline exceeded, plus makespan
      let cost = finishTime;
      if (task.deadline && finishTime > task.deadline) {
        cost += (task.weight || 1) * (finishTime - task.deadline);
      }

      // Prefer machine that finishes earlier (lower makespan impact)
      if (cost < bestCost) {
        bestCost = cost;
        bestTask = task;
        bestMachine = machineIdx;
      }
    }
  }

  if (bestTask === null) break;

  // Schedule the best task on the best machine
  const task = bestTask;
  const machine = machines[bestMachine];

  // Calculate actual start time
  let startTime = machine.availableAt;
  if (task.deps && task.deps.length > 0) {
    startTime = Math.max(startTime, ...task.deps.map(depId => finishTimes[depId] || 0));
  }

  const finishTime = startTime + task.duration;

  scheduledTasks.push({
    id: task.id,
    start: startTime,
    machine: bestMachine
  });

  scheduled.add(task.id);
  finishTimes[task.id] = finishTime;
  machine.availableAt = finishTime;
  machine.tasks.push(task.id);
}

// Output the schedule
console.log(JSON.stringify({ schedule: scheduledTasks }));
