import sys
import math
from typing import List, Tuple

def read_input():
    lines = sys.stdin.read().strip().split('\n')
    n, k, capacity = map(int, lines[0].split())
    depot_x, depot_y = map(int, lines[1].split())

    customers = []
    for i in range(2, 2 + n):
        x, y, demand = map(int, lines[i].split())
        customers.append((x, y, demand))

    return n, k, capacity, (depot_x, depot_y), customers

def distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def route_distance(route, depot, customers):
    if not route:
        return 0
    dist = distance(depot, customers[route[0]])
    for i in range(len(route) - 1):
        dist += distance(customers[route[i]], customers[route[i+1]])
    dist += distance(customers[route[-1]], depot)
    return dist

def route_demand(route, customers):
    return sum(customers[i][2] for i in route)

def nearest_neighbor_routes(n, capacity, depot, customers):
    unvisited = set(range(n))
    routes = []

    while unvisited:
        current = depot
        route = []
        load = 0

        while unvisited:
            nearest = None
            nearest_dist = float('inf')

            for customer_id in unvisited:
                demand = customers[customer_id][2]
                if load + demand <= capacity:
                    d = distance(current, customers[customer_id])
                    if d < nearest_dist:
                        nearest = customer_id
                        nearest_dist = d

            if nearest is None:
                break

            route.append(nearest)
            load += customers[nearest][2]
            unvisited.remove(nearest)
            current = customers[nearest]

        if route:
            routes.append(route)

    return routes

def two_opt_route(route, depot, customers):
    improved = True
    while improved:
        improved = False
        for i in range(len(route) - 1):
            for j in range(i + 2, len(route)):
                if j == len(route) - 1:
                    j_end = len(route)
                else:
                    j_end = j + 1

                old_dist = (distance(customers[route[i]], customers[route[i+1]]) +
                           distance(customers[route[j]], customers[route[j%len(route)]]))
                new_dist = (distance(customers[route[i]], customers[route[j]]) +
                           distance(customers[route[i+1]], customers[route[j%len(route)]]))

                if new_dist < old_dist:
                    route[i+1:j+1] = reversed(route[i+1:j+1])
                    improved = True
                    break
            if improved:
                break
    return route

def optimize_routes(routes, depot, customers):
    for i in range(len(routes)):
        routes[i] = two_opt_route(routes[i], depot, customers)
    return routes

def calculate_total_distance(routes, depot, customers):
    total = 0
    for route in routes:
        total += route_distance(route, depot, customers)
    return total

def main():
    n, k, capacity, depot, customers = read_input()

    routes = nearest_neighbor_routes(n, capacity, depot, customers)
    routes = optimize_routes(routes, depot, customers)

    total_distance = calculate_total_distance(routes, depot, customers)

    print(total_distance)
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    main()
