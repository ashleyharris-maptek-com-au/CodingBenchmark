using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

class AsteroidInterceptor
{
    const double G = 6.67430e-11;
    const double AU = 1.496e11;
    const double DAY_SECONDS = 86400.0;

    struct CelestialBody
    {
        public string Name;
        public double Mass;
        public double[] Pos;
        public double[] Vel;
    }

    struct Burn
    {
        public double Time;
        public double[] DeltaV;
    }

    static void Main()
    {
        try
        {
            var input = ReadInput();
            var solution = SolveMission(input);
            OutputSolution(solution);
        }
        catch (Exception e)
        {
            Console.Error.WriteLine($"Error: {e.Message}");
        }
    }

    static dynamic ReadInput()
    {
        var line1 = Console.ReadLine().Split();
        int numBodies = int.Parse(line1[0]);
        double warningDays = double.Parse(line1[1]);
        double impactorMass = double.Parse(line1[2]);
        double deltaVBudget = double.Parse(line1[3]);
        double asteroidMass = double.Parse(line1[4]);
        long t0Unix = long.Parse(line1[5]);

        var bodies = new Dictionary<string, CelestialBody>();

        for (int i = 0; i < numBodies; i++)
        {
            var parts = Console.ReadLine().Split();
            var body = new CelestialBody
            {
                Name = parts[0],
                Mass = double.Parse(parts[1]),
                Pos = new[] { double.Parse(parts[2]), double.Parse(parts[3]), double.Parse(parts[4]) },
                Vel = new[] { double.Parse(parts[5]), double.Parse(parts[6]), double.Parse(parts[7]) }
            };
            bodies[body.Name] = body;
        }

        string de421Path = Console.ReadLine();

        return new
        {
            NumBodies = numBodies,
            WarningDays = warningDays,
            ImpactorMass = impactorMass,
            DeltaVBudget = deltaVBudget,
            AsteroidMass = asteroidMass,
            T0Unix = t0Unix,
            Bodies = bodies,
            DE421Path = de421Path
        };
    }

    static List<Burn> SolveMission(dynamic input)
    {
        var bodies = (Dictionary<string, CelestialBody>)input.Bodies;
        double warningDays = input.WarningDays;
        double deltaVBudget = input.DeltaVBudget;

        var asteroid = bodies["Asteroid"];
        var earth = bodies["Earth"];
        var sun = bodies["Sun"];

        var burns = new List<Burn>();

        double launchTime = EstimateLaunchTime(earth, asteroid, warningDays);

        var spacecraftState = earth;

        double[] launchDeltaV = CalculateLaunchBurn(spacecraftState, asteroid, earth, sun);
        burns.Add(new Burn { Time = launchTime, DeltaV = launchDeltaV });

        ApplyBurn(ref spacecraftState, launchDeltaV);

        int midCourseCount = warningDays > 10 ? 2 : 1;
        double remainingDeltaV = deltaVBudget - Magnitude(launchDeltaV);

        for (int i = 0; i < midCourseCount && remainingDeltaV > 100; i++)
        {
            double correctionTime = launchTime + (warningDays - launchTime) * (i + 1) / (midCourseCount + 1);

            var asteroidAtCorr = PropagateBody(asteroid, sun, bodies, correctionTime);

            double[] correctionDV = CalculateCorrectionBurn(spacecraftState, asteroidAtCorr, sun, bodies);
            double corrDVMag = Magnitude(correctionDV);

            if (corrDVMag < remainingDeltaV * 0.3)
            {
                burns.Add(new Burn { Time = correctionTime, DeltaV = correctionDV });
                ApplyBurn(ref spacecraftState, correctionDV);
                remainingDeltaV -= corrDVMag;
            }
        }

        double impactTime = EstimateImpactTime(spacecraftState, asteroid, sun, bodies, warningDays);

        var asteroidAtImpact = PropagateBody(asteroid, sun, bodies, impactTime);
        double[] finalBurn = CalculateFinalBurn(spacecraftState, asteroidAtImpact, sun, bodies);

        burns.Add(new Burn { Time = impactTime, DeltaV = finalBurn });

        return burns;
    }

    static double EstimateLaunchTime(CelestialBody earth, CelestialBody asteroid, double warningDays)
    {
        return Math.Max(0, Math.Min(warningDays * 0.3, 60));
    }

    static double[] CalculateLaunchBurn(CelestialBody spacecraft, CelestialBody asteroid, CelestialBody earth, CelestialBody sun)
    {
        double[] toAsteroid = new double[3];
        for (int i = 0; i < 3; i++)
            toAsteroid[i] = asteroid.Pos[i] - spacecraft.Pos[i];

        double distToAsteroid = Magnitude(toAsteroid);
        Normalize(toAsteroid);

        double velRequired = 5000 + Math.Min(5000, distToAsteroid / 1e11);

        return new[] { toAsteroid[0] * velRequired, toAsteroid[1] * velRequired, toAsteroid[2] * velRequired };
    }

    static double[] CalculateCorrectionBurn(CelestialBody spacecraft, CelestialBody asteroid, CelestialBody sun, Dictionary<string, CelestialBody> bodies)
    {
        double[] relPos = new double[3];
        for (int i = 0; i < 3; i++)
            relPos[i] = asteroid.Pos[i] - spacecraft.Pos[i];

        double dist = Magnitude(relPos);

        if (dist < 1e11)
        {
            Normalize(relPos);
            double correction = 1000 * (1e11 / Math.Max(dist, 1e10));
            return new[] { relPos[0] * correction, relPos[1] * correction, relPos[2] * correction };
        }

        return new[] { 0.0, 0.0, 0.0 };
    }

    static double[] CalculateFinalBurn(CelestialBody spacecraft, CelestialBody asteroid, CelestialBody sun, Dictionary<string, CelestialBody> bodies)
    {
        double[] relPos = new double[3];
        double[] relVel = new double[3];
        for (int i = 0; i < 3; i++)
        {
            relPos[i] = asteroid.Pos[i] - spacecraft.Pos[i];
            relVel[i] = asteroid.Vel[i] - spacecraft.Vel[i];
        }

        double dist = Magnitude(relPos);

        double[] finalDV = new double[3];
        if (dist < 1e12)
        {
            Normalize(relPos);
            double velocity = 2000 + Math.Min(3000, 1e11 / Math.Max(dist, 1e10));
            for (int i = 0; i < 3; i++)
                finalDV[i] = relPos[i] * velocity;
        }

        return finalDV;
    }

    static double EstimateImpactTime(CelestialBody spacecraft, CelestialBody asteroid, CelestialBody sun, Dictionary<string, CelestialBody> bodies, double warningDays)
    {
        double distToAsteroid = Magnitude(Subtract(asteroid.Pos, spacecraft.Pos));
        double relVel = Magnitude(Subtract(asteroid.Vel, spacecraft.Vel));

        if (relVel > 100)
        {
            double timeToIntercept = Math.Min(distToAsteroid / (relVel + 1000), warningDays);
            return Math.Min(timeToIntercept, warningDays - 1);
        }

        return warningDays * 0.9;
    }

    static CelestialBody PropagateBody(CelestialBody body, CelestialBody sun, Dictionary<string, CelestialBody> bodies, double timeInDays)
    {
        var result = body;
        double timeStep = DAY_SECONDS;
        int steps = (int)(timeInDays * DAY_SECONDS / timeStep);

        steps = Math.Min(steps, 1000);
        timeStep = timeInDays * DAY_SECONDS / Math.Max(steps, 1);

        for (int i = 0; i < steps; i++)
        {
            var accel = CalculateAcceleration(result, bodies);

            for (int j = 0; j < 3; j++)
            {
                result.Vel[j] += accel[j] * timeStep;
                result.Pos[j] += result.Vel[j] * timeStep;
            }
        }

        return result;
    }

    static double[] CalculateAcceleration(CelestialBody body, Dictionary<string, CelestialBody> bodies)
    {
        var accel = new double[3];

        foreach (var otherBody in bodies.Values)
        {
            if (otherBody.Name == body.Name) continue;

            var delta = Subtract(otherBody.Pos, body.Pos);
            double dist = Magnitude(delta);

            if (dist < 1e8) continue;

            double force = G * otherBody.Mass / (dist * dist);
            Normalize(delta);

            for (int i = 0; i < 3; i++)
                accel[i] += force * delta[i];
        }

        return accel;
    }

    static void ApplyBurn(ref CelestialBody body, double[] deltaV)
    {
        for (int i = 0; i < 3; i++)
            body.Vel[i] += deltaV[i];
    }

    static double[] Subtract(double[] a, double[] b)
    {
        return new[] { a[0] - b[0], a[1] - b[1], a[2] - b[2] };
    }

    static double Magnitude(double[] v)
    {
        return Math.Sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2]);
    }

    static void Normalize(double[] v)
    {
        double mag = Magnitude(v);
        if (mag > 1e-10)
        {
            v[0] /= mag;
            v[1] /= mag;
            v[2] /= mag;
        }
    }

    static void OutputSolution(List<Burn> burns)
    {
        Console.WriteLine(burns.Count);

        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.Time:F6} {burn.DeltaV[0]:E12} {burn.DeltaV[1]:E12} {burn.DeltaV[2]:E12}");
        }
    }
}
