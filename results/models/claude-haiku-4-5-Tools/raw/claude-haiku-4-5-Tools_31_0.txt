use std::io::{self, BufRead, BufReader};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut reader = BufReader::new(stdin.lock());

    let mut first_line = String::new();
    reader.read_line(&mut first_line).unwrap();

    let parts: Vec<&str> = first_line.trim().split_whitespace().collect();
    let target: i64 = parts[1].parse().unwrap();

    let mut second_line = String::new();
    reader.read_line(&mut second_line).unwrap();

    let numbers: Vec<i64> = second_line.trim().split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();

    if let Some(indices) = find_subset(&numbers, target) {
        println!("YES");
        let output = indices.iter().map(|i| i.to_string()).collect::<Vec<_>>().join(" ");
        println!("{}", output);
    } else {
        println!("NO");
    }
}

fn find_subset(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(vec![]);
    }

    if numbers.is_empty() {
        return None;
    }

    let n = numbers.len();

    if n <= 22 {
        brute_force(numbers, target)
    } else {
        dynamic_programming(numbers, target)
    }
}

fn brute_force(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    for mask in 0..(1 << n) {
        let mut sum = 0i64;
        let mut indices = Vec::new();
        for i in 0..n {
            if (mask & (1 << i)) != 0 {
                sum += numbers[i];
                indices.push(i);
            }
        }
        if sum == target {
            return Some(indices);
        }
    }
    None
}

fn dynamic_programming(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for (idx, &num) in numbers.iter().enumerate() {
        let mut new_entries: Vec<_> = Vec::new();

        for (sum, indices) in dp.iter() {
            let new_sum = sum + num;
            if !dp.contains_key(&new_sum) {
                let mut new_indices = indices.clone();
                new_indices.push(idx);
                new_entries.push((new_sum, new_indices));
            }
        }

        for (sum, indices) in new_entries {
            dp.insert(sum, indices);
        }
    }

    dp.get(&target).cloned()
}
