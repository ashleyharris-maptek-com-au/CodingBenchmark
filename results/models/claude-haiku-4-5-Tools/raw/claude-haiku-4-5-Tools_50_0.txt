#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds invocations
    if (idx >= N) return;

    // Read current boid data
    vec3 pos = inData[idx * 2].xyz;
    vec3 vel = inData[idx * 2 + 1].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Accumulate forces
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Iterate through all other boids
    for (uint j = 0; j < N; j++) {
        if (j == idx) continue;  // Skip self

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = otherPos - pos;
        float dist = length(diff);

        if (dist < 0.001) continue;  // Skip zero-distance neighbors

        // Separation rule
        if (dist < sepRadius) {
            sepForce -= normalize(diff) * (sepRadius - dist) / sepRadius;
            sepCount++;
        }

        // Alignment rule
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }

        // Cohesion rule
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }

    // Average and apply weights
    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        accel += (sepForce / float(sepCount)) * sepWeight;
    }

    if (alignCount > 0) {
        accel += (alignForce / float(alignCount) - vel) * alignWeight;
    }

    if (cohCount > 0) {
        accel += (cohForce / float(cohCount) - pos) * cohWeight;
    }

    // Apply boundary force
    float boundLimit = boundSize * 0.4;
    float boundCenter = boundSize * 0.5;

    if (abs(pos.x) > boundLimit) {
        accel.x += boundForce * sign(-pos.x);
    }
    if (abs(pos.y) > boundLimit) {
        accel.y += boundForce * sign(-pos.y);
    }
    if (abs(pos.z) > boundLimit) {
        accel.z += boundForce * sign(-pos.z);
    }

    // Update velocity
    vel += accel * dt;

    // Clamp velocity magnitude to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write updated boid data
    outData[idx * 2] = vec4(pos, 0.0);
    outData[idx * 2 + 1] = vec4(vel, 0.0);
}
