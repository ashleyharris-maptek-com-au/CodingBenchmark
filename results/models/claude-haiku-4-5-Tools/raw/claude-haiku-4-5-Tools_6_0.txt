using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSPSolver
{
    const double MU = 398600.4418; // Earth gravitational parameter km³/s²
    const double EARTH_RADIUS = 6371; // km

    struct State
    {
        public double X, Y, Z, Vx, Vy, Vz;
    }

    struct Edge
    {
        public int From, To;
        public double Cost;
    }

    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);

        var states = new State[n + 1];

        // Read start orbit
        var startLine = Console.ReadLine().Split();
        states[0] = new State
        {
            X = double.Parse(startLine[0]),
            Y = double.Parse(startLine[1]),
            Z = double.Parse(startLine[2]),
            Vx = double.Parse(startLine[3]),
            Vy = double.Parse(startLine[4]),
            Vz = double.Parse(startLine[5])
        };

        // Read station orbits
        for (int i = 1; i <= n; i++)
        {
            var stationLine = Console.ReadLine().Split();
            states[i] = new State
            {
                X = double.Parse(stationLine[0]),
                Y = double.Parse(stationLine[1]),
                Z = double.Parse(stationLine[2]),
                Vx = double.Parse(stationLine[3]),
                Vy = double.Parse(stationLine[4]),
                Vz = double.Parse(stationLine[5])
            };
        }

        // Build cost matrix
        double[,] costs = new double[n + 1, n + 1];
        for (int i = 0; i <= n; i++)
        {
            for (int j = 0; j <= n; j++)
            {
                if (i != j)
                {
                    costs[i, j] = CalculateDeltaV(states[i], states[j]);
                }
            }
        }

        int[] solution = SolveTSP(n, costs);
        Console.WriteLine(string.Join(" ", solution));
    }

    static double CalculateDeltaV(State from, State to)
    {
        double dx = to.X - from.X;
        double dy = to.Y - from.Y;
        double dz = to.Z - from.Z;
        double dvx = to.Vx - from.Vx;
        double dvy = to.Vy - from.Vy;
        double dvz = to.Vz - from.Vz;

        // Simplified delta-V: use Euclidean distance in phase space
        double posDist = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double velDist = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);

        // Normalize by orbital velocity scale
        double orbitalVel = Math.Sqrt(MU / Math.Sqrt(from.X * from.X + from.Y * from.Y + from.Z * from.Z));
        double normalizedPosDist = posDist / (EARTH_RADIUS + 400); // Normalize to typical LEO scale

        return velDist + normalizedPosDist * 0.1;
    }

    static int[] SolveTSP(int n, double[,] costs)
    {
        if (n <= 8)
        {
            return SolveBruteForce(n, costs);
        }
        else if (n <= 15)
        {
            return SolveNearestNeighborWith2Opt(n, costs, 5000);
        }
        else
        {
            return SolveNearestNeighborWith2Opt(n, costs, 2000);
        }
    }

    static int[] SolveBruteForce(int n, double[,] costs)
    {
        var stations = Enumerable.Range(1, n).ToList();
        int[] bestRoute = stations.ToArray();
        double bestCost = CalculateRouteCost(0, bestRoute, costs);

        var permutations = GeneratePermutations(stations);
        foreach (var perm in permutations)
        {
            double routeCost = CalculateRouteCost(0, perm.ToArray(), costs);
            if (routeCost < bestCost)
            {
                bestCost = routeCost;
                bestRoute = perm.ToArray();
            }
        }

        return bestRoute;
    }

    static int[] SolveNearestNeighborWith2Opt(int n, double[,] costs, int iterations)
    {
        // Try multiple nearest neighbor starts
        int[] bestRoute = null;
        double bestCost = double.MaxValue;

        int tryCount = Math.Min(n, 5);
        for (int start = 0; start < tryCount; start++)
        {
            int[] route = NearestNeighbor(start, n, costs);
            route = LocalSearch2Opt(route, costs, iterations);
            double routeCost = CalculateRouteCost(0, route, costs);

            if (routeCost < bestCost)
            {
                bestCost = routeCost;
                bestRoute = route;
            }
        }

        return bestRoute;
    }

    static int[] NearestNeighbor(int start, int n, double[,] costs)
    {
        var visited = new bool[n + 1];
        var route = new List<int>();

        int current = start == 0 ? 1 : start;
        route.Add(current);
        visited[0] = true;
        visited[current] = true;

        while (route.Count < n)
        {
            int nearest = -1;
            double nearestCost = double.MaxValue;

            for (int next = 1; next <= n; next++)
            {
                if (!visited[next] && costs[current, next] < nearestCost)
                {
                    nearestCost = costs[current, next];
                    nearest = next;
                }
            }

            if (nearest != -1)
            {
                route.Add(nearest);
                visited[nearest] = true;
                current = nearest;
            }
        }

        return route.ToArray();
    }

    static int[] LocalSearch2Opt(int[] route, double[,] costs, int maxIterations)
    {
        int[] best = (int[])route.Clone();
        bool improved = true;
        int iterations = 0;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < route.Length - 1; i++)
            {
                for (int j = i + 2; j < route.Length; j++)
                {
                    if (j == route.Length - 1 && i == 0) continue;

                    double delta = costs[route[i], route[j]] + costs[route[i + 1], route[(j + 1) % route.Length]]
                                 - costs[route[i], route[i + 1]] - costs[route[j], route[(j + 1) % route.Length]];

                    if (delta < -1e-9)
                    {
                        Array.Reverse(route, i + 1, j - i);
                        best = (int[])route.Clone();
                        improved = true;
                    }
                }
            }
        }

        return best;
    }

    static double CalculateRouteCost(int start, int[] route, double[,] costs)
    {
        double total = 0;
        int current = start;

        foreach (int next in route)
        {
            total += costs[current, next];
            current = next;
        }

        total += costs[current, start];
        return total;
    }

    static IEnumerable<List<int>> GeneratePermutations(List<int> items)
    {
        if (items.Count == 1)
            yield return items;
        else
        {
            foreach (var item in items)
            {
                var remaining = new List<int>(items);
                remaining.Remove(item);
                foreach (var perm in GeneratePermutations(remaining))
                {
                    var result = new List<int> { item };
                    result.AddRange(perm);
                    yield return result;
                }
            }
        }
    }
}
