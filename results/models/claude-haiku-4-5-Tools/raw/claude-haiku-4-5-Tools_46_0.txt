#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 physics;
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    if (idx >= N) {
        return;
    }

    vec3 pos_i = inData[idx * 2].xyz;
    float mass_i = inData[idx * 2].w;
    vec3 vel_i = inData[idx * 2 + 1].xyz;

    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; j++) {
        if (j == idx) {
            continue;
        }

        vec3 pos_j = inData[j * 2].xyz;
        float mass_j = inData[j * 2].w;

        vec3 delta = pos_j - pos_i;
        float distSq = dot(delta, delta);

        float denom = pow(distSq + softening * softening, 1.5);

        acc += G * mass_j * delta / denom;
    }

    vec3 new_vel = vel_i + acc * dt;

    vec3 new_pos = pos_i + vel_i * dt;

    outData[idx * 2] = vec4(new_pos, mass_i);
    outData[idx * 2 + 1] = vec4(new_vel, 0.0);
}
