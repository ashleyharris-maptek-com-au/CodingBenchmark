%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

%ubo_var = OpVariable %ptr_Uniform_UBO Uniform
%in_uv = OpVariable %ptr_Input_vec2 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

%func_main_type = OpTypeFunction %void

%main = OpFunction %void None %func_main_type
%entry = OpLabel

%uv = OpLoad %vec2 %in_uv
%normal = OpLoad %vec3 %in_normal
%worldPos = OpLoad %vec3 %in_worldPos

%light_idx = OpConstant %int 3
%light_gep = OpAccessChain %ptr_Uniform_vec4 %ubo_var %light_idx
%lightPos = OpLoad %vec4 %light_gep
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%light_dir = OpFSub %vec3 %lightPos_xyz %worldPos
%light_dir_normalized = OpNormalize %vec3 %light_dir
%normal_normalized = OpNormalize %vec3 %normal
%diffuse = OpDot %float %normal_normalized %light_dir_normalized
%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%diffuse_clamped = OpFMax %float %diffuse %zero
%diffuse_clamped_final = OpFMin %float %diffuse_clamped %one

%eight = OpConstant %float 8.0
%scaled_uv = OpVectorTimesScalar %vec2 %uv %eight
%floor_uv = OpFloor %vec2 %scaled_uv
%fract_uv = OpFSub %vec2 %scaled_uv %floor_uv

%half = OpConstant %float 0.5
%fract_x = OpCompositeExtract %float %fract_uv 0
%fract_y = OpCompositeExtract %float %fract_uv 1

%check_x = OpFOrdLessThan %bool %fract_x %half
%check_y = OpFOrdLessThan %bool %fract_y %half

%xor_temp = OpLogicalNotEqual %bool %check_x %check_y

%color_light = OpConstant %float 0.9
%color_dark = OpConstant %float 0.2
%checkerboard = OpSelect %float %xor_temp %color_light %color_dark

%half_const = OpConstant %float 0.5
%shading_mult = OpFMul %float %diffuse_clamped_final %half_const
%shading_factor = OpFAdd %float %half_const %shading_mult

%final_gray = OpFMul %float %checkerboard %shading_factor
%alpha = OpConstant %float 1.0
%final_color = OpCompositeConstruct %vec4 %final_gray %final_gray %final_gray %alpha

OpStore %out_color %final_color
OpReturn
