declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const probability = data.probability;
const drones = data.drones;
const maxStepsPerDrone = data.maxStepsPerDrone;

const height = probability.length;
const width = probability[0].length;

// Track visited cells globally to maximize unique coverage across all drones
const visited = new Set<string>();

interface Drone {
  id: number;
  x: number;
  y: number;
}

function cellKey(x: number, y: number): string {
  return `${x},${y}`;
}

function isValid(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height && probability[y][x] !== -1;
}

function getNeighbors(x: number, y: number): Array<[number, number, number]> {
  const neighbors: Array<[number, number, number]> = [];

  const directions = [
    { dx: 0, dy: -1 }, // up
    { dx: 0, dy: 1 },  // down
    { dx: -1, dy: 0 }, // left
    { dx: 1, dy: 0 }   // right
  ];

  for (const dir of directions) {
    const nx = x + dir.dx;
    const ny = y + dir.dy;

    if (isValid(nx, ny)) {
      neighbors.push([nx, ny, probability[ny][nx]]);
    }
  }

  // Also allow staying in place
  neighbors.push([x, y, probability[y][x]]);

  return neighbors;
}

function planPath(drone: Drone, maxSteps: number): Array<[number, number]> {
  const path: Array<[number, number]> = [[drone.x, drone.y]];
  let currentX = drone.x;
  let currentY = drone.y;

  visited.add(cellKey(currentX, currentY));

  for (let step = 0; step < maxSteps; step++) {
    const neighbors = getNeighbors(currentX, currentY);

    let bestScore = -Infinity;
    let bestX = currentX;
    let bestY = currentY;

    for (const [nx, ny, prob] of neighbors) {
      let score = 0;

      // Heavily prioritize unvisited cells to maximize unique coverage
      const isUnvisited = !visited.has(cellKey(nx, ny));
      if (isUnvisited) {
        score = 10000 + prob;
      } else {
        score = prob;
      }

      if (score > bestScore) {
        bestScore = score;
        bestX = nx;
        bestY = ny;
      }
    }

    if (bestX !== currentX || bestY !== currentY) {
      currentX = bestX;
      currentY = bestY;
    }

    visited.add(cellKey(currentX, currentY));
    path.push([currentX, currentY]);
  }

  return path;
}

const paths = drones.map((drone: Drone) => planPath(drone, maxStepsPerDrone));

const output = { paths };
console.log(JSON.stringify(output));
