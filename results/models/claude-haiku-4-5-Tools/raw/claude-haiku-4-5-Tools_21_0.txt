#include <iostream>
#include <vector>
#include <set>
#include <algorithm>
#include <cmath>

using namespace std;

int main() {
    int D, max_turns;
    cin >> D >> max_turns;

    vector<int> bounds(D);
    for (int i = 0; i < D; i++) {
        cin >> bounds[i];
    }

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; i++) {
        cin >> head[i];
    }

    vector<vector<int>> snake_body;
    for (int i = 0; i < snake_length; i++) {
        snake_body.push_back(head);
    }

    int F;
    cin >> F;
    set<vector<int>> food_set;
    for (int i = 0; i < F; i++) {
        vector<int> pos(D);
        for (int j = 0; j < D; j++) {
            cin >> pos[j];
        }
        food_set.insert(pos);
    }

    int O;
    cin >> O;
    set<vector<int>> obstacles;
    for (int i = 0; i < O; i++) {
        vector<int> obs(D);
        for (int j = 0; j < D; j++) {
            cin >> obs[j];
        }
        obstacles.insert(obs);
    }

    auto manhattan = [](const vector<int>& a, const vector<int>& b) {
        int dist = 0;
        for (int i = 0; i < a.size(); i++) {
            dist += abs(a[i] - b[i]);
        }
        return dist;
    };

    for (int turn = 0; turn < max_turns && !food_set.empty(); turn++) {
        vector<int> nearest_food;
        int nearest_dist = INT_MAX;
        for (const auto& f : food_set) {
            int dist = manhattan(head, f);
            if (dist < nearest_dist) {
                nearest_dist = dist;
                nearest_food = f;
            }
        }

        int best_axis = -1;
        int best_direction = 0;
        int best_dist = nearest_dist;

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> next = head;
                next[axis] += dir;

                if (next[axis] < 0 || next[axis] >= bounds[axis]) {
                    continue;
                }

                if (obstacles.count(next)) {
                    continue;
                }

                bool self_collision = false;
                for (int i = 0; i < (int)snake_body.size() - 1; i++) {
                    if (snake_body[i] == next) {
                        self_collision = true;
                        break;
                    }
                }
                if (self_collision) continue;

                int new_dist = manhattan(next, nearest_food);
                if (best_axis == -1 || new_dist < best_dist) {
                    best_axis = axis;
                    best_direction = dir;
                    best_dist = new_dist;
                }
            }
        }

        if (best_axis == -1) {
            break;
        }

        head[best_axis] += best_direction;
        snake_body.insert(snake_body.begin(), head);

        if (food_set.count(head)) {
            food_set.erase(head);
        } else {
            snake_body.pop_back();
        }

        cout << best_axis << " " << best_direction << "\n";
    }

    return 0;
}
