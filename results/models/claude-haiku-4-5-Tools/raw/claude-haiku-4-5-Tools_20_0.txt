use std::io::{self, BufRead};
use std::collections::HashMap;

fn main() {
    let stdin = io::stdin();
    let mut reader = stdin.lock();
    let mut line = String::new();

    // Read header
    reader.read_line(&mut line).unwrap();
    let parts: Vec<&str> = line.trim().split_whitespace().collect();
    let num_holes: usize = parts[0].parse().unwrap();
    let num_properties: usize = parts[1].parse().unwrap();

    line.clear();
    reader.read_line(&mut line).unwrap();
    let property_names: Vec<&str> = line.trim().split_whitespace().collect();

    // Parse holes and samples
    let mut holes: Vec<Hole> = Vec::new();
    let mut hole_map: HashMap<String, usize> = HashMap::new();

    for _ in 0..num_holes {
        line.clear();
        reader.read_line(&mut line).unwrap();
        let parts: Vec<&str> = line.trim().split_whitespace().collect();

        let hole_id = parts[0].to_string();
        let num_samples: usize = parts[8].parse().unwrap();

        let mut samples = Vec::new();
        for _ in 0..num_samples {
            line.clear();
            reader.read_line(&mut line).unwrap();
            let sample_parts: Vec<&str> = line.trim().split_whitespace().collect();

            let depth: f64 = sample_parts[0].parse().unwrap();
            let mut properties = Vec::new();
            for i in 1..=num_properties {
                properties.push(sample_parts[i].parse::<f64>().unwrap_or(f64::NAN));
            }
            samples.push(Sample { depth, properties });
        }

        hole_map.insert(hole_id.clone(), holes.len());
        holes.push(Hole { id: hole_id, samples });
    }

    // Detect outliers
    let mut outliers: Vec<Outlier> = Vec::new();

    for (hole_idx, hole) in holes.iter().enumerate() {
        let num_samples = hole.samples.len();

        for sample_idx in 0..num_samples {
            let sample = &hole.samples[sample_idx];

            for prop_idx in 0..num_properties {
                if sample.properties[prop_idx].is_nan() {
                    continue;
                }

                // Find neighbors within depth window
                let mut neighbor_values = Vec::new();
                let window_size = if num_samples > 1000 { 5.0 } else { 3.0 };

                for other_idx in 0..num_samples {
                    if other_idx == sample_idx {
                        continue;
                    }
                    let other = &hole.samples[other_idx];
                    let depth_diff = (sample.depth - other.depth).abs();

                    if depth_diff <= window_size && !other.properties[prop_idx].is_nan() {
                        neighbor_values.push(other.properties[prop_idx]);
                    }
                }

                if neighbor_values.is_empty() {
                    continue;
                }

                // Calculate statistics
                neighbor_values.sort_by(|a, b| a.partial_cmp(b).unwrap());
                let median = if neighbor_values.len() % 2 == 0 {
                    (neighbor_values[neighbor_values.len() / 2 - 1] + neighbor_values[neighbor_values.len() / 2]) / 2.0
                } else {
                    neighbor_values[neighbor_values.len() / 2]
                };

                let mad = calculate_mad(&neighbor_values, median);
                let modified_z_score = if mad > 0.0 {
                    0.6745 * (sample.properties[prop_idx] - median) / mad
                } else {
                    0.0
                };

                let confidence = modified_z_score.abs();

                if confidence > 3.5 {
                    outliers.push(Outlier {
                        hole_id: hole.id.clone(),
                        sample_idx,
                        property_name: property_names[prop_idx].to_string(),
                        confidence,
                    });
                }
            }
        }
    }

    // Sort by confidence descending
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Output
    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.2}", outlier.hole_id, outlier.sample_idx, outlier.property_name, outlier.confidence);
    }
}

struct Hole {
    id: String,
    samples: Vec<Sample>,
}

struct Sample {
    depth: f64,
    properties: Vec<f64>,
}

struct Outlier {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn calculate_mad(sorted_values: &[f64], median: f64) -> f64 {
    let mut deviations: Vec<f64> = sorted_values
        .iter()
        .map(|&v| (v - median).abs())
        .collect();
    deviations.sort_by(|a, b| a.partial_cmp(b).unwrap());

    if deviations.len() % 2 == 0 {
        (deviations[deviations.len() / 2 - 1] + deviations[deviations.len() / 2]) / 2.0
    } else {
        deviations[deviations.len() / 2]
    }
}
