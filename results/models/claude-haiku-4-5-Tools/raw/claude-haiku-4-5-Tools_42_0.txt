OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %inColor Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%f_0 = OpConstant %float 0.0
%f_1 = OpConstant %float 1.0
%f_h = OpConstant %float 0.04
%i_0 = OpConstant %int 0
%u_1 = OpConstant %uint 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %u_1
%ptr_in_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_gl_PerVertex Input

%arr_v3 = OpTypeArray %v3float %u_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_gl_PerVertex Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%fn_type = OpTypeFunction %void

%main = OpFunction %void None %fn_type
%bb_entry = OpLabel

%ptr_in_v4 = OpTypePointer Input %v4float
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %i_0 %i_0
%pos = OpLoad %v4float %pos_ptr

%ptr_in_v3 = OpTypePointer Input %v3float
%color_ptr = OpAccessChain %ptr_in_v3 %inColor %i_0
%col = OpLoad %v3float %color_ptr

%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1

%x_m = OpFSub %float %x %f_h
%x_p = OpFAdd %float %x %f_h
%y_m = OpFSub %float %y %f_h
%y_p = OpFAdd %float %y %f_h

%v0 = OpCompositeConstruct %v4float %x_m %y_m %f_0 %f_1
%ptr_out_v4 = OpTypePointer Output %v4float
%pos_out = OpAccessChain %ptr_out_v4 %gl_out %i_0
OpStore %pos_out %v0
OpStore %outColor %col
OpEmitVertex

%v1 = OpCompositeConstruct %v4float %x_p %y_m %f_0 %f_1
OpStore %pos_out %v1
OpStore %outColor %col
OpEmitVertex

%v2 = OpCompositeConstruct %v4float %x_m %y_p %f_0 %f_1
OpStore %pos_out %v2
OpStore %outColor %col
OpEmitVertex

%v3 = OpCompositeConstruct %v4float %x_p %y_p %f_0 %f_1
OpStore %pos_out %v3
OpStore %outColor %col
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
