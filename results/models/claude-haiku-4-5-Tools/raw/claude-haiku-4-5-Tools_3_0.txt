use std::io::{self, BufRead};

const PI: f64 = std::f64::consts::PI;

#[derive(Clone)]
struct Node {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
}

struct Graph {
    nodes: Vec<Node>,
    edges: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
}

impl Graph {
    fn new(n: usize) -> Self {
        Graph {
            nodes: (0..n).map(|_| Node {
                x: 0.0,
                y: 0.0,
                vx: 0.0,
                vy: 0.0,
            }).collect(),
            edges: Vec::new(),
            adj: vec![Vec::new(); n],
        }
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        if u != v {
            self.edges.push((u, v));
            self.adj[u].push(v);
            if u != v {
                self.adj[v].push(u);
            }
        }
    }

    fn layout(&mut self) {
        let n = self.nodes.len();
        if n == 0 {
            return;
        }

        // Initialize positions in a circle
        for i in 0..n {
            let angle = 2.0 * PI * (i as f64) / (n as f64);
            let radius = 100.0 + (n as f64).sqrt() * 10.0;
            self.nodes[i].x = radius * angle.cos();
            self.nodes[i].y = radius * angle.sin();
        }

        // Choose algorithm based on graph size
        if n <= 100 {
            self.fruchterman_reingold(100, 1.0);
        } else if n <= 500 {
            self.fruchterman_reingold(50, 0.5);
        } else if n <= 5000 {
            self.fruchterman_reingold(20, 0.2);
        } else {
            self.fruchterman_reingold(10, 0.1);
        }
    }

    fn fruchterman_reingold(&mut self, iterations: usize, damping: f64) {
        let n = self.nodes.len();
        if n == 0 {
            return;
        }

        let k = 100.0 / (n as f64).sqrt();
        let max_dist = k * 4.0;

        for iter in 0..iterations {
            // Reset forces
            for node in &mut self.nodes {
                node.vx = 0.0;
                node.vy = 0.0;
            }

            // Repulsive forces
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = self.nodes[j].x - self.nodes[i].x;
                    let dy = self.nodes[j].y - self.nodes[i].y;
                    let dist = (dx * dx + dy * dy).sqrt() + 0.01;
                    let force = k * k / dist;

                    let fx = force * dx / dist;
                    let fy = force * dy / dist;

                    self.nodes[i].vx -= fx;
                    self.nodes[i].vy -= fy;
                    self.nodes[j].vx += fx;
                    self.nodes[j].vy += fy;
                }
            }

            // Attractive forces (only for edges)
            for &(u, v) in &self.edges {
                let dx = self.nodes[v].x - self.nodes[u].x;
                let dy = self.nodes[v].y - self.nodes[u].y;
                let dist = (dx * dx + dy * dy).sqrt() + 0.01;
                let force = dist * dist / k;

                let fx = force * dx / dist;
                let fy = force * dy / dist;

                self.nodes[u].vx += fx;
                self.nodes[u].vy += fy;
                self.nodes[v].vx -= fx;
                self.nodes[v].vy -= fy;
            }

            // Apply velocity with temperature cooling
            let temp = 1.0 / ((iter as f64 / iterations as f64) + 1.0);
            for node in &mut self.nodes {
                let vel = (node.vx * node.vx + node.vy * node.vy).sqrt();
                if vel > 0.0 {
                    let max_vel = max_dist * temp;
                    let scale = max_vel.min(vel) / vel;
                    node.x += node.vx * scale * damping;
                    node.y += node.vy * scale * damping;
                }
            }
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    let mut graph = Graph::new(n);

    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let parts: Vec<&str> = line.split_whitespace().collect();
            let u: usize = parts[0].parse().unwrap();
            let v: usize = parts[1].parse().unwrap();
            if u < n && v < n {
                graph.add_edge(u, v);
            }
        }
    }

    graph.layout();

    // Output
    println!("{}", n);
    for node in &graph.nodes {
        println!("{} {}", node.x, node.y);
    }
}
