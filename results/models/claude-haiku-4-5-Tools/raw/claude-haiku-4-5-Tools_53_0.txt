import math

_state_estimate = {
    'pos_x': 0.0, 'pos_y': 0.0, 'pos_z': 0.0,
    'vel_x': 0.0, 'vel_y': 0.0, 'vel_z': 0.0,
}
_sensor_history = {'range': [], 'radar': [], 'lidar': []}
_approach_phase = 'far'
_fuel_emergency = False

def autopilot_step(state: dict, dt: float) -> dict:
    global _state_estimate, _approach_phase, _fuel_emergency

    rel_pos, rel_vel = fuse_sensors(state)
    range_val = estimate_range(rel_pos)

    if range_val > 200:
        _approach_phase = 'far'
    elif range_val > 50:
        _approach_phase = 'medium'
    elif range_val > 10:
        _approach_phase = 'close'
    else:
        _approach_phase = 'final'

    fuel_pct = state.get('fuel_remaining_pct', 100)
    _fuel_emergency = fuel_pct < 15

    thrust_x, thrust_y, thrust_z = compute_thrust(
        rel_pos, rel_vel, range_val, _approach_phase, _fuel_emergency
    )

    att_roll, att_pitch, att_yaw = compute_attitude_rates(
        rel_pos, rel_vel, thrust_x, thrust_y, thrust_z, state
    )

    _state_estimate['pos_x'] = rel_pos[0]
    _state_estimate['pos_y'] = rel_pos[1]
    _state_estimate['pos_z'] = rel_pos[2]
    _state_estimate['vel_x'] = rel_vel[0]
    _state_estimate['vel_y'] = rel_vel[1]
    _state_estimate['vel_z'] = rel_vel[2]

    return {
        'thrust_x': clamp(thrust_x, -1.0, 1.0),
        'thrust_y': clamp(thrust_y, -1.0, 1.0),
        'thrust_z': clamp(thrust_z, -1.0, 1.0),
        'att_rate_roll': clamp(att_roll, -0.035, 0.035),
        'att_rate_pitch': clamp(att_pitch, -0.035, 0.035),
        'att_rate_yaw': clamp(att_yaw, -0.035, 0.035),
    }

def fuse_sensors(state):
    """Fuse radar, lidar, docking camera with redundancy checks."""
    if is_valid(state, ['lidar_rel_x_valid', 'lidar_rel_y_valid', 'lidar_rel_z_valid']):
        pos = (
            state.get('lidar_rel_pos_x_m', 0),
            state.get('lidar_rel_pos_y_m', 0),
            state.get('lidar_rel_pos_z_m', 0),
        )
        vel = (
            state.get('lidar_rel_vel_x_ms', 0),
            state.get('lidar_rel_vel_y_ms', 0),
            state.get('lidar_rel_vel_z_ms', 0),
        )
    elif is_valid(state, ['radar_rel_x_valid', 'radar_rel_y_valid', 'radar_rel_z_valid']):
        pos = (
            state.get('radar_rel_pos_x_m', 0),
            state.get('radar_rel_pos_y_m', 0),
            state.get('radar_rel_pos_z_m', 0),
        )
        vel = (
            state.get('radar_rel_vel_x_ms', 0),
            state.get('radar_rel_vel_y_ms', 0),
            state.get('radar_rel_vel_z_ms', 0),
        )
    else:
        pos = (0, 0, 0)
        vel = (0, 0, 0)

    if state.get('dock_camera_active', False) and is_valid(state, ['dock_camera_bearing_x_valid']):
        bearing = (
            state.get('dock_camera_bearing_x', 0),
            state.get('dock_camera_bearing_y', 0),
            state.get('dock_camera_bearing_z', 0),
        )
        cam_range = state.get('dock_camera_range_m', 0)
        if cam_range > 0:
            pos = tuple(b * cam_range for b in bearing)

    return pos, vel

def is_valid(state, keys):
    """Check if all given keys are valid."""
    return all(state.get(k, False) for k in keys)

def estimate_range(rel_pos):
    """Compute distance to target."""
    return math.sqrt(sum(x**2 for x in rel_pos))

def compute_thrust(rel_pos, rel_vel, range_val, phase, fuel_emergency):
    """Compute thrust vector with orbital mechanics awareness."""
    x, y, z = rel_pos
    vx, vy, vz = rel_vel

    Kp_pos = 0.015 if phase == 'far' else (0.03 if phase == 'medium' else 0.08)
    Kd_vel = 0.05 if phase == 'far' else (0.1 if phase == 'medium' else 0.2)

    if fuel_emergency:
        Kp_pos *= 0.5
        Kd_vel *= 0.5

    ex, ey, ez = x, y, z
    evx, evy, evz = vx, vy, vz

    if phase == 'far' or phase == 'medium':
        if ey < -50 and evy < 0.2:
            ey_correction = -0.3
        elif ey > 50 and evy > -0.2:
            ey_correction = 0.2
        else:
            ey_correction = ey
    else:
        ey_correction = ey

    thrust_x = -Kp_pos * ex - Kd_vel * evx
    thrust_y = -Kp_pos * ey_correction - Kd_vel * evy
    thrust_z = -Kp_pos * ez - Kd_vel * evz

    return thrust_x, thrust_y, thrust_z

def compute_attitude_rates(rel_pos, rel_vel, thrust_x, thrust_y, thrust_z, state):
    """Compute attitude rates to align with thrust vector."""
    thrust_mag = math.sqrt(thrust_x**2 + thrust_y**2 + thrust_z**2)
    if thrust_mag < 0.01:
        return 0, 0, 0

    norm_thrust_x = thrust_x / thrust_mag
    norm_thrust_y = thrust_y / thrust_mag
    norm_thrust_z = thrust_z / thrust_mag

    Kp_att = 0.02

    att_pitch = Kp_att * math.atan2(norm_thrust_z, norm_thrust_x)
    att_roll = Kp_att * math.atan2(norm_thrust_y, norm_thrust_x)
    att_yaw = 0

    return att_roll, att_pitch, att_yaw

def clamp(value, min_val, max_val):
    """Clamp value between min and max."""
    return max(min_val, min(max_val, value))
