OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0p05 = OpConstant %float 0.05

; Input gl_PerVertex
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input arrays
%arr_v3 = OpTypeArray %v3float %int_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

; Pointer types
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

%main_func = OpTypeFunction %void
%main = OpFunction %void None %main_func
%entry = OpLabel

; Load input position: gl_in[0].gl_Position
%pos_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr

; Load input color: color[0]
%color_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%color_val = OpLoad %v3float %color_ptr

; Extract position components
%x = OpCompositeExtract %float %pos 0
%y = OpCompositeExtract %float %pos 1
%z = OpCompositeExtract %float %pos 2
%w = OpCompositeExtract %float %pos 3

; Compute diamond vertices
; Top: (x, y+h, z, w)
%top_y = OpFAdd %float %y %float_0p05
%top = OpCompositeConstruct %v4float %x %top_y %z %w

; Right: (x+h, y, z, w)
%right_x = OpFAdd %float %x %float_0p05
%right = OpCompositeConstruct %v4float %right_x %y %z %w

; Left: (x-h, y, z, w)
%left_x = OpFSub %float %x %float_0p05
%left = OpCompositeConstruct %v4float %left_x %y %z %w

; Bottom: (x, y-h, z, w)
%bottom_y = OpFSub %float %y %float_0p05
%bottom = OpCompositeConstruct %v4float %x %bottom_y %z %w

; Access chain for output position
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0

; Emit top vertex
OpStore %out_pos_ptr %top
OpStore %outColor %color_val
OpEmitVertex

; Emit right vertex
OpStore %out_pos_ptr %right
OpStore %outColor %color_val
OpEmitVertex

; Emit left vertex
OpStore %out_pos_ptr %left
OpStore %outColor %color_val
OpEmitVertex

; Emit bottom vertex
OpStore %out_pos_ptr %bottom
OpStore %outColor %color_val
OpEmitVertex

; End primitive
OpEndPrimitive

OpReturn
OpFunctionEnd
