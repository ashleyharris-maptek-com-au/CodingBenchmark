[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0 {
    // UV checkerboard at 8x8 frequency
    float2 checkerUV = input.uv * 8.0;
    int checkX = int(checkerUV.x) % 2;
    int checkY = int(checkerUV.y) % 2;
    float checkerValue = (checkX ^ checkY) == 0 ? 0.9 : 0.2;

    // Normalize the normal
    float3 normal = normalize(input.normal);

    // Lambertian shading
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(normal, lightDir), 0.0);
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Apply shading to checkerboard
    float3 finalColor = checkerValue * shadingFactor;

    return float4(finalColor, 1.0);
}
