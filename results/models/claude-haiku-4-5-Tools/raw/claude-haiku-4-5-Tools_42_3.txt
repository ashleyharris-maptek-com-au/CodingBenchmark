OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %inColor "inColor"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Uniform
OpDecorate %inColor Location 2

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_03 = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_2_pv = OpTypeArray %gl_PerVertex_in %int_2
%arr_2_v3 = OpTypeArray %v3float %int_2

%ptr_in_pv_arr = OpTypePointer Input %arr_2_pv
%ptr_in_v3_arr = OpTypePointer Input %arr_2_v3
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%func_type = OpTypeFunction %void

%gl_in = OpVariable %ptr_in_pv_arr Input
%gl_out = OpVariable %ptr_out_pv Output
%inColor = OpVariable %ptr_in_v3_arr Input
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func_type
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2

%dir = OpFSub %v3float %p1_xyz %p0_xyz

%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y

%perp_base = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0

%perp_x = OpCompositeExtract %float %perp_base 0
%perp_y = OpCompositeExtract %float %perp_base 1
%sq_x = OpFMul %float %perp_x %perp_x
%sq_y = OpFMul %float %perp_y %perp_y
%perp_len_sq = OpFAdd %float %sq_x %sq_y
%perp_len = OpExtInst %float %GLSL450 Sqrt %perp_len_sq

%scale = OpFDiv %float %float_0_03 %perp_len
%perp = OpVectorTimesScalar %v3float %perp_base %scale

%v0_pos = OpFAdd %v3float %p0_xyz %perp
%v0_x = OpCompositeExtract %float %v0_pos 0
%v0_y = OpCompositeExtract %float %v0_pos 1
%v0_z = OpCompositeExtract %float %v0_pos 2
%v0_final = OpCompositeConstruct %v4float %v0_x %v0_y %v0_z %float_1

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %v0_final
OpStore %outColor %c0
OpEmitVertex

%v1_pos = OpFSub %v3float %p0_xyz %perp
%v1_x = OpCompositeExtract %float %v1_pos 0
%v1_y = OpCompositeExtract %float %v1_pos 1
%v1_z = OpCompositeExtract %float %v1_pos 2
%v1_final = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1

OpStore %out_pos_ptr %v1_final
OpStore %outColor %c0
OpEmitVertex

%v2_pos = OpFAdd %v3float %p1_xyz %perp
%v2_x = OpCompositeExtract %float %v2_pos 0
%v2_y = OpCompositeExtract %float %v2_pos 1
%v2_z = OpCompositeExtract %float %v2_pos 2
%v2_final = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1

OpStore %out_pos_ptr %v2_final
OpStore %outColor %c1
OpEmitVertex

%v3_pos = OpFSub %v3float %p1_xyz %perp
%v3_x = OpCompositeExtract %float %v3_pos 0
%v3_y = OpCompositeExtract %float %v3_pos 1
%v3_z = OpCompositeExtract %float %v3_pos 2
%v3_final = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1

OpStore %out_pos_ptr %v3_final
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpFunctionEnd
