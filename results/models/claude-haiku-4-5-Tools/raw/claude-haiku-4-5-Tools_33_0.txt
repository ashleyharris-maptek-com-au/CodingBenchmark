using System;
using System.Collections.Generic;
using System.Linq;

class QAP {
    static void Main() {
        int n = int.Parse(Console.ReadLine());

        // Read flow matrix
        double[][] flow = new double[n][];
        for (int i = 0; i < n; i++) {
            flow[i] = Console.ReadLine().Split().Select(double.Parse).ToArray();
        }

        // Read distance matrix
        double[][] dist = new double[n][];
        for (int i = 0; i < n; i++) {
            dist[i] = Console.ReadLine().Split().Select(double.Parse).ToArray();
        }

        // Solve QAP
        int[] bestPerm = SolveQAP(n, flow, dist);
        double bestCost = CalculateCost(bestPerm, flow, dist);

        Console.WriteLine((long)bestCost);
        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static int[] SolveQAP(int n, double[][] flow, double[][] dist) {
        int[] bestPerm = null;
        double bestCost = double.MaxValue;

        // For small N, try multiple random starts
        int numStarts = n < 50 ? Math.Min(10, n) : (n < 200 ? 3 : 1);
        Random rand = new Random(42);

        for (int start = 0; start < numStarts; start++) {
            int[] perm = new int[n];

            if (start == 0) {
                // First try: identity permutation
                for (int i = 0; i < n; i++) perm[i] = i;
            } else {
                // Other tries: random permutation
                for (int i = 0; i < n; i++) perm[i] = i;
                for (int i = n - 1; i > 0; i--) {
                    int j = rand.Next(i + 1);
                    int temp = perm[i];
                    perm[i] = perm[j];
                    perm[j] = temp;
                }
            }

            // Improve with 2-opt
            TwoOpt(perm, flow, dist);

            double cost = CalculateCost(perm, flow, dist);
            if (cost < bestCost) {
                bestCost = cost;
                bestPerm = (int[])perm.Clone();
            }
        }

        return bestPerm;
    }

    static void TwoOpt(int[] perm, double[][] flow, double[][] dist) {
        int n = perm.Length;
        bool improved = true;
        int iterations = 0;
        int maxIterations = n < 100 ? 100 : (n < 500 ? 20 : 5);

        while (improved && iterations < maxIterations) {
            improved = false;
            iterations++;

            for (int i = 0; i < n - 1; i++) {
                for (int j = i + 2; j < n; j++) {
                    // Calculate cost delta for swapping positions i+1 and j
                    double delta = CalculateDelta(perm, i, j, flow, dist);

                    if (delta < -1e-9) {
                        // Reverse the segment
                        Array.Reverse(perm, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
    }

    static double CalculateDelta(int[] perm, int i, int j, double[][] flow, double[][] dist) {
        int n = perm.Length;
        double delta = 0;

        // Cost contribution of perm[i+1]..perm[j] in current order
        double oldCost = 0, newCost = 0;

        // Edges from i to i+1 and j to j+1 (if exists)
        if (i >= 0 && i + 1 < n) {
            int f1 = i, f2 = i + 1;
            int loc1 = perm[f1], loc2 = perm[f2];
            oldCost += flow[f1][f2] * dist[loc1][loc2];
            oldCost += flow[f2][f1] * dist[loc2][loc1];
            newCost += flow[f1][j] * dist[loc1][perm[j]];
            newCost += flow[j][f1] * dist[perm[j]][loc1];
        }

        if (j + 1 < n) {
            int f1 = j, f2 = j + 1;
            int loc1 = perm[f1], loc2 = perm[f2];
            oldCost += flow[f1][f2] * dist[loc1][loc2];
            oldCost += flow[f2][f1] * dist[loc2][loc1];
            newCost += flow[i + 1][f2] * dist[perm[i + 1]][loc2];
            newCost += flow[f2][i + 1] * dist[loc2][perm[i + 1]];
        }

        return newCost - oldCost;
    }

    static double CalculateCost(int[] perm, double[][] flow, double[][] dist) {
        double cost = 0;
        int n = perm.Length;

        for (int i = 0; i < n; i++) {
            for (int j = 0; j < n; j++) {
                cost += flow[i][j] * dist[perm[i]][perm[j]];
            }
        }

        return cost;
    }
}
