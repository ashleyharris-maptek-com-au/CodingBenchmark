use std::collections::VecDeque;
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let h: usize = parts[0].parse().unwrap();
    let w: usize = parts[1].parse().unwrap();

    let mut maze = Vec::new();
    let mut start = (0, 0);
    let mut end = (0, 0);

    for i in 0..h {
        let row = lines.next().unwrap().unwrap();
        let bytes = row.as_bytes();

        for (j, &byte) in bytes.iter().enumerate() {
            if j < w {
                if byte == b'A' {
                    start = (i, j);
                } else if byte == b'B' {
                    end = (i, j);
                }
            }
        }

        maze.push(row);
    }

    let path = solve_maze(&maze, start, end, h, w);

    if path.is_empty() {
        println!("0");
    } else {
        println!("{}", path.len());
        for (row, col) in path {
            println!("{} {}", col, row);
        }
    }
}

fn solve_maze(
    maze: &[String],
    start: (usize, usize),
    end: (usize, usize),
    h: usize,
    w: usize,
) -> Vec<(usize, usize)> {
    let mut visited = vec![vec![false; w]; h];
    let mut parent = vec![vec![None; w]; h];
    let mut queue = VecDeque::new();

    queue.push_back(start);
    visited[start.0][start.1] = true;

    while let Some((row, col)) = queue.pop_front() {
        if (row, col) == end {
            let mut path = vec![(row, col)];
            let mut current = (row, col);

            while let Some(prev) = parent[current.0][current.1] {
                path.push(prev);
                current = prev;
            }

            path.reverse();
            return path;
        }

        let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];
        for (dr, dc) in directions.iter() {
            let nr = row as i32 + dr;
            let nc = col as i32 + dc;

            if nr >= 0 && nr < h as i32 && nc >= 0 && nc < w as i32 {
                let nr = nr as usize;
                let nc = nc as usize;

                if !visited[nr][nc] {
                    let cell = maze[nr].as_bytes()[nc] as char;
                    if cell != '#' {
                        visited[nr][nc] = true;
                        parent[nr][nc] = Some((row, col));
                        queue.push_back((nr, nc));
                    }
                }
            }
        }
    }

    vec![]
}
