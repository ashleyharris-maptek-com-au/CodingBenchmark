#include <iostream>
#include <vector>
#include <algorithm>
#include <cstring>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    vector<vector<pair<int, int>>> jobs(J);

    // Read input: for each job, read its tasks (machine, duration)
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].first >> jobs[j][t].second;
        }
    }

    // Track which task is next for each job
    vector<int> next_task(J, 0);

    // Track when each job and machine will be free
    vector<int> job_finish(J, 0);
    vector<int> machine_free(M, 0);

    // Store scheduling results: result[job][task] = (start_time, machine)
    vector<vector<pair<int, int>>> result(J);
    for (int j = 0; j < J; j++) {
        result[j].resize(jobs[j].size());
    }

    // Count total tasks to schedule
    int total_tasks = 0;
    for (int j = 0; j < J; j++) {
        total_tasks += jobs[j].size();
    }

    // Greedy scheduling: repeatedly pick the available task that can start earliest
    for (int i = 0; i < total_tasks; i++) {
        int best_job = -1;
        int best_start_time = INT_MAX;

        // Find available task that can start earliest
        for (int j = 0; j < J; j++) {
            if (next_task[j] < (int)jobs[j].size()) {
                int machine = jobs[j][next_task[j]].first;
                int duration = jobs[j][next_task[j]].second;

                // Task can start when both job is ready and machine is free
                int start_time = max(job_finish[j], machine_free[machine]);

                if (start_time < best_start_time) {
                    best_start_time = start_time;
                    best_job = j;
                }
            }
        }

        if (best_job == -1) break;

        // Schedule the selected task
        int j = best_job;
        int t = next_task[j];
        int machine = jobs[j][t].first;
        int duration = jobs[j][t].second;
        int start_time = best_start_time;
        int end_time = start_time + duration;

        result[j][t] = {start_time, machine};
        job_finish[j] = end_time;
        machine_free[machine] = end_time;
        next_task[j]++;
    }

    // Output: for each job, print all (start_time, machine) pairs
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)result[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << result[j][t].first << " " << result[j][t].second;
        }
        cout << "\n";
    }

    return 0;
}
