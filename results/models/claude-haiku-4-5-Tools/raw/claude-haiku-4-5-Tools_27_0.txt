#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <algorithm>
#include <cstring>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<int>> adj(n);
    vector<int> degree(n, 0);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }

    vector<int> color(n, -1);
    vector<bool> available(k);

    // Adaptive strategy based on graph size and density
    if (n <= 10000 && m / (double)n <= 10) {
        // Small to medium graph: use saturation-based ordering (DSatur)
        vector<int> order;
        vector<int> saturation(n, 0);
        vector<bool> used(n, false);

        // Start with highest degree vertex
        int start = 0;
        for (int i = 1; i < n; i++) {
            if (degree[i] > degree[start]) start = i;
        }

        order.push_back(start);
        used[start] = true;
        color[start] = 0;

        // Update saturation for neighbors
        for (int neighbor : adj[start]) {
            saturation[neighbor]++;
        }

        // DSatur ordering
        for (int iter = 1; iter < n; iter++) {
            int next = -1;
            int maxSat = -1;

            for (int i = 0; i < n; i++) {
                if (!used[i] && saturation[i] > maxSat) {
                    maxSat = saturation[i];
                    next = i;
                } else if (!used[i] && saturation[i] == maxSat && degree[i] > degree[next]) {
                    next = i;
                }
            }

            if (next == -1) {
                // Find any uncolored vertex
                for (int i = 0; i < n; i++) {
                    if (!used[i]) {
                        next = i;
                        break;
                    }
                }
            }

            used[next] = true;
            order.push_back(next);

            // Color the vertex
            fill(available.begin(), available.end(), false);
            for (int neighbor : adj[next]) {
                if (color[neighbor] != -1) {
                    available[color[neighbor]] = true;
                }
            }

            int assignedColor = 0;
            while (assignedColor < k && available[assignedColor]) {
                assignedColor++;
            }

            color[next] = assignedColor;

            // Update saturation for neighbors
            for (int neighbor : adj[next]) {
                if (!used[neighbor] && color[next] < k) {
                    if (!available[assignedColor]) {
                        saturation[neighbor]++;
                    }
                }
            }
        }
    } else {
        // Large graph: use Welsh-Powell (greedy by degree order)
        vector<pair<int, int>> degreeOrder;
        for (int i = 0; i < n; i++) {
            degreeOrder.push_back({degree[i], i});
        }
        sort(degreeOrder.rbegin(), degreeOrder.rend());

        for (auto [deg, u] : degreeOrder) {
            fill(available.begin(), available.end(), false);

            for (int neighbor : adj[u]) {
                if (color[neighbor] != -1) {
                    available[color[neighbor]] = true;
                }
            }

            int assignedColor = 0;
            while (assignedColor < k && available[assignedColor]) {
                assignedColor++;
            }

            color[u] = assignedColor;
        }
    }

    // Output colors
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << color[i];
    }
    cout << "\n";

    return 0;
}
