OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %color %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpExtInstImport "GLSL.std.450"
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
     %uint_6 = OpConstant %uint 6
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %int_2 = OpConstant %int 2
     %int_4 = OpConstant %int 4
   %float_0 = OpConstant %float 0.0
 %float_0p2 = OpConstant %float 0.2
 %float_0p8 = OpConstant %float 0.8
   %float_1 = OpConstant %float 1.0
 %float_1_3 = OpConstant %float 0.333333333
%gl_PerVertex_in = OpTypeStruct %v4float
   %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %arr_v3 = OpTypeArray %v3float %uint_6
  %ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_out_v3 = OpTypePointer Output %v3float
   %func_void = OpTypeFunction %void
      %gl_in = OpVariable %ptr_in_arr_pv Input
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %color = OpVariable %ptr_in_arr_v3 Input
     %gl_out = OpVariable %ptr_out_pv Output
   %outColor = OpVariable %ptr_out_v3 Output
%ext_glsl_450 = OpExtInstImport "GLSL.std.450"
       %main = OpFunction %void None %func_void
      %entry = OpLabel
  %pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %pos0 = OpLoad %v4float %pos0_ptr
  %pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
       %pos2 = OpLoad %v4float %pos2_ptr
  %pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
       %pos4 = OpLoad %v4float %pos4_ptr
   %pos0_v3 = OpVectorShuffle %v3float %pos0 %pos0 0 1 2
   %pos2_v3 = OpVectorShuffle %v3float %pos2 %pos2 0 1 2
   %pos4_v3 = OpVectorShuffle %v3float %pos4 %pos4 0 1 2
        %v1 = OpFSub %v3float %pos2_v3 %pos0_v3
        %v2 = OpFSub %v3float %pos4_v3 %pos0_v3
     %cross = OpExtInst %v3float %ext_glsl_450 Cross %v1 %v2
     %normal = OpExtInst %v3float %ext_glsl_450 Normalize %cross
 %light_vec = OpConstantComposite %v3float %float_1 %float_1 %float_1
     %light = OpExtInst %v3float %ext_glsl_450 Normalize %light_vec
   %dot_val = OpDot %float %normal %light
   %diffuse = OpExtInst %float %ext_glsl_450 FMax %dot_val %float_0
 %color0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
     %color0 = OpLoad %v3float %color0_ptr
 %color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
     %color2 = OpLoad %v3float %color2_ptr
 %color4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
     %color4 = OpLoad %v3float %color4_ptr
   %sum_01 = OpFAdd %v3float %color0 %color2
    %sum_all = OpFAdd %v3float %sum_01 %color4
  %avg_color = OpVectorTimesScalar %v3float %sum_all %float_1_3
   %scale = OpFMul %float %float_0p8 %diffuse
   %final_scale = OpFAdd %float %float_0p2 %scale
 %final_color = OpVectorTimesScalar %v3float %avg_color %final_scale
  %out_ac0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ac0 %pos0
               OpStore %outColor %final_color
               OpEmitVertex
  %out_ac1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ac1 %pos2
               OpStore %outColor %final_color
               OpEmitVertex
  %out_ac2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ac2 %pos4
               OpStore %outColor %final_color
               OpEmitVertex
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
