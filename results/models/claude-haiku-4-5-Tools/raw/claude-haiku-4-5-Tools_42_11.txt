OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo Block
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

OpExtInstImport %glsl "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0p333 = OpConstant %float 0.333333
%float_0p667 = OpConstant %float 0.666667
%float_0p5 = OpConstant %float 0.5
%float_0p02 = OpConstant %float 0.02

%v3float_up = OpConstantComposite %v3float %float_0 %float_0 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr4_gl_PerVertex = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr = OpTypePointer Input %arr4_gl_PerVertex
%gl_in = OpVariable %ptr_Input_arr Input

%arr4_v3 = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3 = OpTypePointer Input %arr4_v3
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_Output_gl_PerVertex Output

%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%ubo = OpTypeStruct %mat4float %mat4float %mat4float %v3float %float %v3float %float %v3float %float
%ptr_Uniform_ubo = OpTypePointer Uniform %ubo
%uboVar = OpVariable %ptr_Uniform_ubo Uniform

%func_type = OpTypeFunction %void

%main = OpFunction %void None %func_type
%entry = OpLabel

%p0_chain = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_chain
%p1_chain = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_chain
%p2_chain = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_chain
%p3_chain = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_chain

%c1_chain = OpAccessChain %ptr_Input_v3 %color %int_1
%c1 = OpLoad %v3float %c1_chain
%c2_chain = OpAccessChain %ptr_Input_v3 %color %int_2
%c2 = OpLoad %v3float %c2_chain
%col_avg = OpVectorTimesScalar %v3float %c1 %float_0p5
%col_avg2 = OpVectorTimesScalar %v3float %c2 %float_0p5
%col_final = OpFAdd %v3float %col_avg %col_avg2

%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_v3 = OpVectorShuffle %v3float %p3 %p3 0 1 2

%seg = OpFSub %v3float %p2_v3 %p1_v3

%perp = OpExtInst %v3float %glsl Cross %seg %v3float_up
%perp_norm = OpExtInst %v3float %glsl Normalize %perp
%offset = OpVectorTimesScalar %v3float %perp_norm %float_0p02

%seg_1_3 = OpVectorTimesScalar %v3float %seg %float_0p333
%seg_2_3 = OpVectorTimesScalar %v3float %seg %float_0p667

%part1_start = %p1_v3
%part1_end = OpFAdd %v3float %p1_v3 %seg_1_3

%part3_start = OpFAdd %v3float %p1_v3 %seg_2_3
%part3_end = %p2_v3

%e0_pos = OpFSub %v3float %part1_start %offset
%e0_x = OpCompositeExtract %float %e0_pos 0
%e0_y = OpCompositeExtract %float %e0_pos 1
%e0_z = OpCompositeExtract %float %e0_pos 2
%e0_v4 = OpCompositeConstruct %v4float %e0_x %e0_y %e0_z %float_1
%e0_out = OpAccessChain %ptr_Output_v4 %gl_out %int_0
OpStore %e0_out %e0_v4
OpStore %outColor %col_final
OpEmitVertex

%e1_pos = OpFAdd %v3float %part1_start %offset
%e1_x = OpCompositeExtract %float %e1_pos 0
%e1_y = OpCompositeExtract %float %e1_pos 1
%e1_z = OpCompositeExtract %float %e1_pos 2
%e1_v4 = OpCompositeConstruct %v4float %e1_x %e1_y %e1_z %float_1
OpStore %e0_out %e1_v4
OpStore %outColor %col_final
OpEmitVertex

%e2_pos = OpFSub %v3float %part1_end %offset
%e2_x = OpCompositeExtract %float %e2_pos 0
%e2_y = OpCompositeExtract %float %e2_pos 1
%e2_z = OpCompositeExtract %float %e2_pos 2
%e2_v4 = OpCompositeConstruct %v4float %e2_x %e2_y %e2_z %float_1
OpStore %e0_out %e2_v4
OpStore %outColor %col_final
OpEmitVertex

%e3_pos = OpFAdd %v3float %part1_end %offset
%e3_x = OpCompositeExtract %float %e3_pos 0
%e3_y = OpCompositeExtract %float %e3_pos 1
%e3_z = OpCompositeExtract %float %e3_pos 2
%e3_v4 = OpCompositeConstruct %v4float %e3_x %e3_y %e3_z %float_1
OpStore %e0_out %e3_v4
OpStore %outColor %col_final
OpEmitVertex

OpEndPrimitive

%e4_pos = OpFSub %v3float %part3_start %offset
%e4_x = OpCompositeExtract %float %e4_pos 0
%e4_y = OpCompositeExtract %float %e4_pos 1
%e4_z = OpCompositeExtract %float %e4_pos 2
%e4_v4 = OpCompositeConstruct %v4float %e4_x %e4_y %e4_z %float_1
OpStore %e0_out %e4_v4
OpStore %outColor %col_final
OpEmitVertex

%e5_pos = OpFAdd %v3float %part3_start %offset
%e5_x = OpCompositeExtract %float %e5_pos 0
%e5_y = OpCompositeExtract %float %e5_pos 1
%e5_z = OpCompositeExtract %float %e5_pos 2
%e5_v4 = OpCompositeConstruct %v4float %e5_x %e5_y %e5_z %float_1
OpStore %e0_out %e5_v4
OpStore %outColor %col_final
OpEmitVertex

%e6_pos = OpFSub %v3float %part3_end %offset
%e6_x = OpCompositeExtract %float %e6_pos 0
%e6_y = OpCompositeExtract %float %e6_pos 1
%e6_z = OpCompositeExtract %float %e6_pos 2
%e6_v4 = OpCompositeConstruct %v4float %e6_x %e6_y %e6_z %float_1
OpStore %e0_out %e6_v4
OpStore %outColor %col_final
OpEmitVertex

%e7_pos = OpFAdd %v3float %part3_end %offset
%e7_x = OpCompositeExtract %float %e7_pos 0
%e7_y = OpCompositeExtract %float %e7_pos 1
%e7_z = OpCompositeExtract %float %e7_pos 2
%e7_v4 = OpCompositeConstruct %v4float %e7_x %e7_y %e7_z %float_1
OpStore %e0_out %e7_v4
OpStore %outColor %col_final
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
