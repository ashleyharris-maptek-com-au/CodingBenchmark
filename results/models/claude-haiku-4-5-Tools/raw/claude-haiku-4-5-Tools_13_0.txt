#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <algorithm>
#include <cstring>

using namespace std;

const unsigned long long BASE = 31;
const unsigned long long MOD = 1e9 + 7;

struct RollingHash {
    vector<unsigned long long> hash, pow;
    size_t len;

    RollingHash(const string& s) : len(s.size()) {
        hash.resize(len + 1);
        pow.resize(len + 1);
        pow[0] = 1;
        for (size_t i = 0; i < len; i++) {
            hash[i + 1] = (hash[i] * BASE + (s[i] - 'a' + 1)) % MOD;
            pow[i + 1] = (pow[i] * BASE) % MOD;
        }
    }

    unsigned long long getHash(size_t l, size_t r) const {
        return (hash[r] - hash[l] * pow[r - l] % MOD + MOD) % MOD;
    }
};

bool hasCommonSubstring(const vector<string>& strings, size_t targetLen) {
    if (strings.empty() || targetLen == 0) return false;

    vector<RollingHash> rollers;
    for (const auto& s : strings) {
        rollers.emplace_back(s);
    }

    unordered_set<unsigned long long> candidates;

    // Collect all substring hashes of targetLen from first string
    for (size_t i = 0; i + targetLen <= strings[0].size(); i++) {
        candidates.insert(rollers[0].getHash(i, i + targetLen));
    }

    if (candidates.empty()) return false;

    // Check if hashes exist in all other strings
    for (size_t i = 1; i < strings.size() && !candidates.empty(); i++) {
        unordered_set<unsigned long long> nextCandidates;
        for (size_t j = 0; j + targetLen <= strings[i].size(); j++) {
            unsigned long long h = rollers[i].getHash(j, j + targetLen);
            if (candidates.count(h)) {
                nextCandidates.insert(h);
            }
        }
        candidates = nextCandidates;
    }

    return !candidates.empty();
}

string findActualSubstring(const vector<string>& strings, size_t len) {
    if (strings.empty() || len == 0) return "";

    // Get all substrings of length len from first string
    unordered_set<string> candidates;
    for (size_t i = 0; i + len <= strings[0].size(); i++) {
        candidates.insert(strings[0].substr(i, len));
    }

    // Filter by checking existence in all other strings
    for (size_t i = 1; i < strings.size(); i++) {
        unordered_set<string> nextCandidates;
        for (const auto& sub : candidates) {
            if (strings[i].find(sub) != string::npos) {
                nextCandidates.insert(sub);
            }
        }
        candidates = nextCandidates;
        if (candidates.empty()) return "";
    }

    return candidates.empty() ? "" : *candidates.begin();
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    size_t minLen = SIZE_MAX;

    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
        minLen = min(minLen, strings[i].size());
    }

    if (n == 0 || minLen == 0) {
        cout << "\n";
        return 0;
    }

    // Binary search for longest common substring length
    size_t left = 1, right = minLen;
    size_t bestLen = 0;

    while (left <= right) {
        size_t mid = (left + right) / 2;
        if (hasCommonSubstring(strings, mid)) {
            bestLen = mid;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    if (bestLen == 0) {
        cout << "\n";
    } else {
        string result = findActualSubstring(strings, bestLen);
        cout << result << "\n";
    }

    return 0;
}
