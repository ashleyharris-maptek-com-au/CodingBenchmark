use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn rotate(&self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.rotate(angle)).collect(),
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.translate(dx, dy)).collect(),
        }
    }

    fn get_bounds(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for v in &self.vertices {
            min_x = min_x.min(v.x);
            max_x = max_x.max(v.x);
            min_y = min_y.min(v.y);
            max_y = max_y.max(v.y);
        }

        (min_x, max_x, min_y, max_y)
    }

    fn area(&self) -> f64 {
        if self.vertices.len() < 3 {
            return 0.0;
        }

        let mut area = 0.0;
        for i in 0..self.vertices.len() {
            let j = (i + 1) % self.vertices.len();
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        (area / 2.0).abs()
    }

    fn contains_point(&self, p: &Point) -> bool {
        let mut inside = false;
        let mut j = self.vertices.len() - 1;

        for i in 0..self.vertices.len() {
            let xi = self.vertices[i].x;
            let yi = self.vertices[i].y;
            let xj = self.vertices[j].x;
            let yj = self.vertices[j].y;

            if ((yi > p.y) != (yj > p.y)) && (p.x < (xj - xi) * (p.y - yi) / (yj - yi) + xi) {
                inside = !inside;
            }
            j = i;
        }

        inside
    }

    fn contains_polygon(&self, other: &Polygon) -> bool {
        for v in &other.vertices {
            if !self.contains_point(v) {
                return false;
            }
        }
        true
    }

    fn aabb_overlap(&self, other: &Polygon) -> bool {
        let (min_x1, max_x1, min_y1, max_y1) = self.get_bounds();
        let (min_x2, max_x2, min_y2, max_y2) = other.get_bounds();

        !(max_x1 < min_x2 || max_x2 < min_x1 || max_y1 < min_y2 || max_y2 < min_y1)
    }
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        stock_vertices.push(Point::new(x, y));
    }
    let stock_polygon = Polygon::new(stock_vertices);

    let mut pieces = Vec::new();
    let mut piece_areas = Vec::new();
    for _ in 0..np {
        let pv_line = lines.next().unwrap().unwrap();
        let pv: usize = pv_line.parse().unwrap();
        let mut vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let parts: Vec<&str> = line.split_whitespace().collect();
            let x: f64 = parts[0].parse().unwrap();
            let y: f64 = parts[1].parse().unwrap();
            vertices.push(Point::new(x, y));
        }
        let poly = Polygon::new(vertices);
        piece_areas.push(poly.area());
        pieces.push(poly);
    }

    let mut piece_indices: Vec<usize> = (0..pieces.len()).collect();
    piece_indices.sort_by(|&a, &b| piece_areas[b].partial_cmp(&piece_areas[a]).unwrap());

    let (sx_min, sx_max, sy_min, sy_max) = stock_polygon.get_bounds();
    let stock_w = sx_max - sx_min;
    let stock_h = sy_max - sy_min;

    let mut stocks: Vec<Vec<Polygon>> = vec![Vec::new()];
    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];

    let num_rotations = if np > 2000 { 4 } else if np > 500 { 8 } else if np > 100 { 16 } else { 24 };
    let rotation_step = PI / num_rotations as f64;

    for &piece_idx in &piece_indices {
        let piece = &pieces[piece_idx];
        let mut placed = false;

        'outer: for rot_i in 0..num_rotations {
            if placed { break; }
            let rotation = rot_i as f64 * rotation_step;
            let rotated = piece.rotate(rotation);
            let (px_min, px_max, py_min, py_max) = rotated.get_bounds();
            let piece_w = px_max - px_min;
            let piece_h = py_max - py_min;

            for stock_idx in 0..stocks.len() {
                if placed { break 'outer; }

                let grid_size = if num_rotations <= 8 { 6 } else { 4 };

                for gx in 0..grid_size {
                    if placed { break 'outer; }
                    for gy in 0..grid_size {
                        let x = sx_min + (gx as f64 + 0.5) / grid_size as f64 * stock_w;
                        let y = sy_min + (gy as f64 + 0.5) / grid_size as f64 * stock_h;

                        let placed_piece = rotated.translate(x, y);

                        if !stock_polygon.contains_polygon(&placed_piece) {
                            continue;
                        }

                        let mut overlaps = false;
                        for existing in &stocks[stock_idx] {
                            if placed_piece.aabb_overlap(existing) {
                                overlaps = true;
                                break;
                            }
                        }

                        if !overlaps {
                            stocks[stock_idx].push(placed_piece);
                            placements[piece_idx] = (stock_idx, x, y, rotation);
                            placed = true;
                            break 'outer;
                        }
                    }
                }
            }
        }

        if !placed {
            let (px_min, px_max, py_min, py_max) = piece.get_bounds();
            let x = sx_min + (stock_w - (px_max - px_min)) / 2.0;
            let y = sy_min + (stock_h - (py_max - py_min)) / 2.0;
            let placed_piece = piece.translate(x, y);

            if stock_polygon.contains_polygon(&placed_piece) {
                stocks.push(vec![placed_piece]);
                placements[piece_idx] = (stocks.len() - 1, x, y, 0.0);
            } else {
                stocks.push(vec![piece.translate(sx_min, sy_min).clone()]);
                placements[piece_idx] = (stocks.len() - 1, sx_min, sy_min, 0.0);
            }
        }
    }

    println!("{}", stocks.len());
    for (stock_idx, x, y, rotation) in placements {
        println!("{} {} {} {}", stock_idx, x, y, rotation);
    }
}
