#include <iostream>
#include <string>
#include <vector>
#include <cstring>
#include <algorithm>
#include <chrono>
#include <thread>
#include <atomic>
#include <limits>
#include <cctype>
#include <sstream>

using namespace std;

constexpr int INF = 1000000;

struct Move {
    int from, to;
    int piece;
    bool capture, castle, promotion;
    int promoted_to;
    bool en_passant, check, checkmate;

    string to_notation() const {
        string result;
        result += tolower("PNBRQK"[piece]);
        result += char('a' + (from % 8));
        result += char('8' - (from / 8));
        if (capture) result += 'x';
        result += char('a' + (to % 8));
        result += char('8' - (to / 8));
        if (promotion) {
            result += '/';
            result += "NBRQ"[promoted_to - 1];
        }
        if (en_passant) result += "ep";
        if (checkmate) result += '#';
        else if (check) result += '+';
        return result;
    }
};

class Board {
public:
    uint64_t pieces[12];
    bool white_to_move;
    int en_passant_sq;
    bool white_castle_k, white_castle_q, black_castle_k, black_castle_q;

    Board() {
        memset(pieces, 0, sizeof(pieces));
        white_to_move = true;
        en_passant_sq = -1;
    }

    void load_fen(const string& fen) {
        memset(pieces, 0, sizeof(pieces));
        size_t pos = 0;
        int rank = 7, file = 0;

        while (pos < fen.size() && fen[pos] != ' ') {
            if (fen[pos] == '/') {
                rank--;
                file = 0;
            } else if (isdigit(fen[pos])) {
                file += fen[pos] - '0';
            } else {
                int sq = rank * 8 + file;
                int piece = -1;
                int color = isupper(fen[pos]) ? 0 : 1;
                switch (tolower(fen[pos])) {
                    case 'p': piece = 0; break;
                    case 'n': piece = 1; break;
                    case 'b': piece = 2; break;
                    case 'r': piece = 3; break;
                    case 'q': piece = 4; break;
                    case 'k': piece = 5; break;
                }
                if (piece >= 0) {
                    pieces[color * 6 + piece] |= (1ULL << sq);
                }
                file++;
            }
            pos++;
        }

        while (pos < fen.size() && fen[pos] == ' ') pos++;
        white_to_move = (fen[pos] == 'w');

        while (pos < fen.size() && fen[pos] != ' ') pos++;
        while (pos < fen.size() && fen[pos] == ' ') pos++;

        white_castle_k = white_castle_q = black_castle_k = black_castle_q = false;
        while (pos < fen.size() && fen[pos] != ' ') {
            if (fen[pos] == 'K') white_castle_k = true;
            if (fen[pos] == 'Q') white_castle_q = true;
            if (fen[pos] == 'k') black_castle_k = true;
            if (fen[pos] == 'q') black_castle_q = true;
            pos++;
        }

        while (pos < fen.size() && fen[pos] == ' ') pos++;
        en_passant_sq = -1;
        if (pos < fen.size() && fen[pos] != '-') {
            en_passant_sq = (fen[pos] - 'a') + 8 * (8 - (fen[pos + 1] - '0'));
        }
    }

    uint64_t all_pieces() const {
        uint64_t all = 0;
        for (int i = 0; i < 12; i++) all |= pieces[i];
        return all;
    }

    uint64_t white_pieces() const {
        uint64_t w = 0;
        for (int i = 0; i < 6; i++) w |= pieces[i];
        return w;
    }

    uint64_t black_pieces() const {
        uint64_t b = 0;
        for (int i = 6; i < 12; i++) b |= pieces[i];
        return b;
    }

    bool is_attacked(int sq, bool by_white) {
        for (int i = 0; i < 12; i++) {
            if (by_white && i >= 6) continue;
            if (!by_white && i < 6) continue;
            int piece = i % 6;
            uint64_t mask = pieces[i];
            while (mask) {
                int from = __builtin_ctzll(mask);
                mask &= mask - 1;
                if (can_attack(from, sq, piece, i / 6 == 0)) return true;
            }
        }
        return false;
    }

    bool can_attack(int from, int to, int piece, bool is_white) {
        int df = to % 8 - from % 8;
        int dr = to / 8 - from / 8;
        uint64_t all = all_pieces();

        switch (piece) {
            case 0: // pawn
                return (is_white ? dr == 1 : dr == -1) && abs(df) == 1;
            case 1: // knight
                return (abs(dr) == 2 && abs(df) == 1) || (abs(dr) == 1 && abs(df) == 2);
            case 2: // bishop
                if (abs(dr) != abs(df)) return false;
                return path_clear(from, to, all);
            case 3: // rook
                if (dr != 0 && df != 0) return false;
                return path_clear(from, to, all);
            case 4: // queen
                if (dr != 0 && df != 0 && abs(dr) != abs(df)) return false;
                return path_clear(from, to, all);
            case 5: // king
                return abs(dr) <= 1 && abs(df) <= 1;
        }
        return false;
    }

    bool path_clear(int from, int to, uint64_t all) {
        int df = (to % 8 > from % 8) ? 1 : (to % 8 < from % 8) ? -1 : 0;
        int dr = (to / 8 > from / 8) ? 1 : (to / 8 < from / 8) ? -1 : 0;
        int sq = from + dr * 8 + df;
        while (sq != to) {
            if (all & (1ULL << sq)) return false;
            sq += dr * 8 + df;
        }
        return true;
    }

    void generate_moves(vector<Move>& moves) {
        moves.clear();
        bool is_white = white_to_move;
        uint64_t own = is_white ? white_pieces() : black_pieces();
        uint64_t enemy = is_white ? black_pieces() : white_pieces();
        uint64_t all = own | enemy;

        for (int piece = 0; piece < 6; piece++) {
            uint64_t mask = pieces[is_white ? piece : piece + 6];
            while (mask) {
                int from = __builtin_ctzll(mask);
                mask &= mask - 1;
                generate_piece_moves(from, piece, is_white, own, enemy, all, moves);
            }
        }
    }

    void generate_piece_moves(int from, int piece, bool is_white, uint64_t own, uint64_t enemy, uint64_t all, vector<Move>& moves) {
        if (piece == 0) { // pawn
            int dir = is_white ? 1 : -1;
            int to = from + dir * 8;
            if (to >= 0 && to < 64 && !(all & (1ULL << to))) {
                if ((is_white && to >= 56) || (!is_white && to < 8)) {
                    for (int p = 1; p <= 3; p++) {
                        Move m = {from, to, 0, false, false, true, p, false, false, false};
                        moves.push_back(m);
                    }
                } else {
                    Move m = {from, to, 0, false, false, false, 0, false, false, false};
                    moves.push_back(m);
                }
            }
            for (int df : {-1, 1}) {
                int to = from + dir * 8 + df;
                if (to >= 0 && to < 64 && (enemy & (1ULL << to))) {
                    if ((is_white && to >= 56) || (!is_white && to < 8)) {
                        for (int p = 1; p <= 3; p++) {
                            Move m = {from, to, 0, true, false, true, p, false, false, false};
                            moves.push_back(m);
                        }
                    } else {
                        Move m = {from, to, 0, true, false, false, 0, false, false, false};
                        moves.push_back(m);
                    }
                }
            }
        } else if (piece == 5) { // king
            for (int dr = -1; dr <= 1; dr++) {
                for (int df = -1; df <= 1; df++) {
                    if (dr == 0 && df == 0) continue;
                    int to = from + dr * 8 + df;
                    if (to >= 0 && to < 64 && to / 8 >= 0 && to / 8 <= 7) {
                        if (!(own & (1ULL << to))) {
                            Move m = {from, to, piece, bool(enemy & (1ULL << to)), false, false, 0, false, false, false};
                            moves.push_back(m);
                        }
                    }
                }
            }
            if (is_white && white_castle_k && !(all & (1ULL << 5)) && !(all & (1ULL << 6))) {
                Move m = {4, 6, 5, false, true, false, 0, false, false, false};
                moves.push_back(m);
            }
            if (is_white && white_castle_q && !(all & (1ULL << 3)) && !(all & (1ULL << 2)) && !(all & (1ULL << 1))) {
                Move m = {4, 2, 5, false, true, false, 0, false, false, false};
                moves.push_back(m);
            }
            if (!is_white && black_castle_k && !(all & (1ULL << 61)) && !(all & (1ULL << 62))) {
                Move m = {60, 62, 5, false, true, false, 0, false, false, false};
                moves.push_back(m);
            }
            if (!is_white && black_castle_q && !(all & (1ULL << 59)) && !(all & (1ULL << 58)) && !(all & (1ULL << 57))) {
                Move m = {60, 58, 5, false, true, false, 0, false, false, false};
                moves.push_back(m);
            }
        } else {
            vector<pair<int, int>> directions;
            if (piece == 1) { // knight
                for (int dr = -2; dr <= 2; dr++) {
                    for (int df = -2; df <= 2; df++) {
                        if (abs(dr) + abs(df) == 3) directions.push_back({dr, df});
                    }
                }
            } else if (piece == 2 || piece == 4) { // bishop or queen
                for (int dr = -1; dr <= 1; dr++) {
                    for (int df = -1; df <= 1; df++) {
                        if (abs(dr) == abs(df) && !(dr == 0 && df == 0)) {
                            directions.push_back({dr, df});
                        }
                    }
                }
            }
            if (piece == 3 || piece == 4) { // rook or queen
                for (int dr = -1; dr <= 1; dr++) {
                    for (int df = -1; df <= 1; df++) {
                        if ((dr == 0 || df == 0) && !(dr == 0 && df == 0)) {
                            directions.push_back({dr, df});
                        }
                    }
                }
            }

            for (auto [dr, df] : directions) {
                int to = from + dr * 8 + df;
                while (to >= 0 && to < 64) {
                    if (own & (1ULL << to)) break;
                    Move m = {from, to, piece, bool(enemy & (1ULL << to)), false, false, 0, false, false, false};
                    moves.push_back(m);
                    if (enemy & (1ULL << to)) break;
                    if (piece == 1) break; // knight
                    to += dr * 8 + df;
                }
            }
        }
    }

    void apply_move(const Move& m) {
        int p = m.piece;
        int idx = white_to_move ? p : p + 6;
        pieces[idx] &= ~(1ULL << m.from);
        pieces[idx] |= (1ULL << m.to);

        if (m.capture) {
            for (int i = 0; i < 12; i++) {
                if ((white_to_move && i < 6 && i != idx) || (!white_to_move && i >= 6 && i != idx)) {
                    if (pieces[i] & (1ULL << m.to)) {
                        pieces[i] &= ~(1ULL << m.to);
                        break;
                    }
                }
            }
        }

        if (m.promotion) {
            pieces[idx] &= ~(1ULL << m.to);
            int pidx = white_to_move ? m.promoted_to : m.promoted_to + 6;
            pieces[pidx] |= (1ULL << m.to);
        }

        if (m.castle) {
            if (m.to == 6) {
                pieces[0] &= ~(1ULL << 7);
                pieces[0] |= (1ULL << 5);
            } else if (m.to == 2) {
                pieces[0] &= ~(1ULL << 0);
                pieces[0] |= (1ULL << 3);
            } else if (m.to == 62) {
                pieces[9] &= ~(1ULL << 63);
                pieces[9] |= (1ULL << 61);
            } else if (m.to == 58) {
                pieces[9] &= ~(1ULL << 56);
                pieces[9] |= (1ULL << 59);
            }
        }

        white_to_move = !white_to_move;
    }

    int evaluate() {
        int score = 0;
        int pval[] = {1, 3, 3, 5, 9, 0};
        for (int i = 0; i < 6; i++) {
            uint64_t mask = pieces[i];
            while (mask) {
                int sq = __builtin_ctzll(mask);
                mask &= mask - 1;
                score += pval[i];
            }
        }
        for (int i = 6; i < 12; i++) {
            uint64_t mask = pieces[i];
            while (mask) {
                int sq = __builtin_ctzll(mask);
                mask &= mask - 1;
                score -= pval[i - 6];
            }
        }
        return score;
    }
};

atomic<bool> timeout_flag(false);

int alphabeta(Board& board, int depth, int alpha, int beta, bool maximizing, chrono::high_resolution_clock::time_point deadline) {
    if (timeout_flag || chrono::high_resolution_clock::now() > deadline) {
        timeout_flag = true;
        return board.evaluate();
    }

    if (depth == 0) return board.evaluate();

    vector<Move> moves;
    board.generate_moves(moves);

    if (moves.empty()) {
        return board.white_to_move ? -INF : INF;
    }

    if (maximizing) {
        int value = -INF;
        for (const auto& m : moves) {
            board.apply_move(m);
            int eval = alphabeta(board, depth - 1, alpha, beta, false, deadline);
            board.apply_move(m);
            board.apply_move(m);
            value = max(value, eval);
            alpha = max(alpha, value);
            if (beta <= alpha) break;
        }
        return value;
    } else {
        int value = INF;
        for (const auto& m : moves) {
            board.apply_move(m);
            int eval = alphabeta(board, depth - 1, alpha, beta, true, deadline);
            board.apply_move(m);
            board.apply_move(m);
            value = min(value, eval);
            beta = min(beta, value);
            if (beta <= alpha) break;
        }
        return value;
    }
}

int main() {
    int timeout_s;
    string fen;
    getline(cin, fen);
    timeout_s = stoi(fen);
    getline(cin, fen);

    Board board;
    board.load_fen(fen);

    auto deadline = chrono::high_resolution_clock::now() + chrono::seconds(timeout_s - 1);

    vector<Move> moves;
    board.generate_moves(moves);

    if (moves.empty()) {
        return 1;
    }

    Move best_move = moves[0];
    int best_value = -INF;

    for (int depth = 1; depth <= 20 && !timeout_flag; depth++) {
        for (const auto& m : moves) {
            if (timeout_flag) break;
            board.apply_move(m);
            int value = -alphabeta(board, depth - 1, -INF, INF, false, deadline);
            board.apply_move(m);
            board.apply_move(m);
            if (value > best_value) {
                best_value = value;
                best_move = m;
            }
        }
        if (timeout_flag) break;
    }

    cout << best_move.to_notation() << endl;

    return 0;
}
