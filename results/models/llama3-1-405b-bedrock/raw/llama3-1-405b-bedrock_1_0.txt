#include <iostream>
#include <vector>
#include <cmath>
#include <random>
#include <algorithm>

using namespace std;

struct City {
    double x, y;
};

double distance(const City& a, const City& b) {
    return sqrt(pow(a.x - b.x, 2) + pow(a.y - b.y, 2));
}

vector<int> nearest_neighbor(const vector<City>& cities) {
    vector<int> path;
    path.push_back(0);
    vector<bool> visited(cities.size(), false);
    visited[0] = true;

    for (int i = 0; i < cities.size() - 1; ++i) {
        double min_dist = numeric_limits<double>::max();
        int next_city = -1;
        for (int j = 0; j < cities.size(); ++j) {
            if (!visited[j] && distance(cities[path.back()], cities[j]) < min_dist) {
                min_dist = distance(cities[path.back()], cities[j]);
                next_city = j;
            }
        }
        path.push_back(next_city);
        visited[next_city] = true;
    }

    return path;
}

vector<int> two_opt(const vector<City>& cities, const vector<int>& path) {
    vector<int> new_path = path;
    random_device rd;
    mt19937 g(rd());
    bool improved = true;

    while (improved) {
        improved = false;
        shuffle(new_path.begin() + 1, new_path.end(), g);
        for (int i = 1; i < new_path.size() - 2; ++i) {
            for (int j = i + 1; j < new_path.size() - 1; ++j) {
                double current_dist = distance(cities[new_path[i - 1]], cities[new_path[i]]) +
                                      distance(cities[new_path[j]], cities[new_path[j + 1]]);
                double new_dist = distance(cities[new_path[i - 1]], cities[new_path[j]]) +
                                  distance(cities[new_path[i]], cities[new_path[j + 1]]);
                if (new_dist < current_dist) {
                    reverse(new_path.begin() + i, new_path.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }

    return new_path;
}

vector<int> christofides(const vector<City>& cities) {
    vector<int> path = nearest_neighbor(cities);
    vector<bool> visited(cities.size(), false);
    for (int i : path) {
        visited[i] = true;
    }

    vector<int> odd_vertices;
    for (int i = 0; i < cities.size(); ++i) {
        if (visited[i] && (path.size() - (find(path.begin(), path.end(), i) - path.begin())) % 2 != 0) {
            odd_vertices.push_back(i);
        }
    }

    vector<pair<int, int>> perfect_matching;
    for (int i = 0; i < odd_vertices.size(); ++i) {
        double min_dist = numeric_limits<double>::max();
        int closest_vertex = -1;
        for (int j = i + 1; j < odd_vertices.size(); ++j) {
            double dist = distance(cities[odd_vertices[i]], cities[odd_vertices[j]]);
            if (dist < min_dist) {
                min_dist = dist;
                closest_vertex = j;
            }
        }
        perfect_matching.push_back({odd_vertices[i], odd_vertices[closest_vertex]});
    }

    for (const auto& pair : perfect_matching) {
        path.insert(find(path.begin(), path.end(), pair.first) + 1, pair.second);
    }

    return path;
}

int main() {
    int N;
    cin >> N;

    vector<City> cities(N);
    for (int i = 0; i < N; ++i) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> path;
    if (N < 100) {
        path = nearest_neighbor(cities);
    } else if (N < 1000) {
        path = two_opt(cities, nearest_neighbor(cities));
    } else if (N < 5000) {
        path = christofides(cities);
    } else {
        path = nearest_neighbor(cities);
    }

    for (int i : path) {
        cout << i << " ";
    }
    cout << endl;

    return 0;
}
