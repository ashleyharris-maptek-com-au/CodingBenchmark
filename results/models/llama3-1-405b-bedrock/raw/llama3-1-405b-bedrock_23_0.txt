

Here is a simple implementation of a lunar lander controller in Rust. This code reads input from stdin, processes the state updates, and outputs thrust/turn commands to stdout.

```rust
use std::io::{BufRead, Write};

struct Lander {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

fn main() {
    let mut lines = std::io::stdin().lock().lines();

    // Read initial map
    let map_info: Vec<f64> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    let start_info: Vec<f64> = lines
        .next()
        .unwrap()
        .unwrap()
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    let map: Vec<String> = lines
        .by_ref()
        .take(map_info[1] as usize)
        .map(|line| line.unwrap())
        .collect();

    let mut lander = Lander {
        x: start_info[0],
        y: start_info[1],
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel_remaining: map_info[4],
    };

    let target_x = start_info[2];
    let target_y = start_info[3];

    let gravity = map_info[2];
    let max_thrust = map_info[3];

    let mut stdout = std::io::stdout().lock();

    loop {
        // Read state update
        let state: Vec<f64> = lines
            .next()
            .unwrap()
            .unwrap()
            .split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();

        lander.x = state[0];
        lander.y = state[1];
        lander.vx = state[2];
        lander.vy = state[3];
        lander.angle = state[4];
        lander.angular_vel = state[5];
        lander.fuel_remaining = state[6];

        // Calculate thrust and turn
        let thrust = calculate_thrust(&lander, target_x, target_y, gravity, max_thrust);
        let turn = calculate_turn(&lander, target_x, target_y);

        // Output thrust and turn commands
        writeln!(stdout, "{} {}", thrust, turn).unwrap();

        // Check if landed
        if is_landed(&lander, target_x, target_y) {
            break;
        }
    }
}

fn calculate_thrust(
    lander: &Lander,
    target_x: f64,
    target_y: f64,
    gravity: f64,
    max_thrust: f64,
) -> f64 {
    let distance_to_target = ((lander.x - target_x).powi(2) + (lander.y - target_y).powi(2)).sqrt();
    let velocity = (lander.vx.powi(2) + lander.vy.powi(2)).sqrt();

    if distance_to_target < 10.0 && velocity < 2.0 {
        0.0
    } else {
        (gravity / max_thrust).min(1.0)
    }
}

fn calculate_turn(lander: &Lander, target_x: f64, target_y: f64) -> f64 {
    let angle_to_target = (target_y - lander.y).atan2(target_x - lander.x);
    let angle_diff = angle_to_target - lander.angle;

    if angle_diff > std::f64::consts::PI {
        -1.0
    } else if angle_diff < -std::f64::consts::PI {
        1.0
    } else {
        angle_diff / std::f64::consts::PI
    }
}

fn is_landed(lander: &Lander, target_x: f64, target_y: f64) -> bool {
    let distance_to_target = ((lander.x - target_x).powi(2) + (lander.y - target_y).powi(2)).sqrt();
    let velocity = (lander.vx.powi(2) + lander.vy.powi(2)).sqrt();

    distance_to_target < 1.0 && velocity < 2.0
}
```

This code uses a simple control strategy to guide the lander to the target. The `calculate_thrust` function adjusts the thrust to counteract gravity and slow down the lander when it's close to the target. The `calculate_turn` function adjusts the turn to point the lander towards the target.

Note that this is a very basic implementation and may not work well for all scenarios. You may need to adjust the control strategy and add more complexity to the code to handle different situations.

Also, this code assumes that the input is well-formed and does not handle errors. You may want to add error handling to make the code more robust.
