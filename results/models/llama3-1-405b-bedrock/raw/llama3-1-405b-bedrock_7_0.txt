// Function to check if a point is inside a mesh
bool isPointInsideMesh(const Vertex& point, const vector<Vertex>& vertices, const vector<Face>& faces) {
    int intersections = 0;
    for (const auto& face : faces) {
        Vertex v1 = vertices[face.a];
        Vertex v2 = vertices[face.b];
        Vertex v3 = vertices[face.c];

        // Check if point is on the same plane as the face
        float planeNormalX = (v2.y - v1.y) * (v3.z - v1.z) - (v2.z - v1.z) * (v3.y - v1.y);
        float planeNormalY = (v2.z - v1.z) * (v3.x - v1.x) - (v2.x - v1.x) * (v3.z - v1.z);
        float planeNormalZ = (v2.x - v1.x) * (v3.y - v1.y) - (v2.y - v1.y) * (v3.x - v1.x);
        float dotProduct = planeNormalX * (point.x - v1.x) + planeNormalY * (point.y - v1.y) + planeNormalZ * (point.z - v1.z);
        if (dotProduct == 0.0f) {
            // Point is on the plane, check if it's inside the face
            float area = 0.5f * sqrt(planeNormalX * planeNormalX + planeNormalY * planeNormalY + planeNormalZ * planeNormalZ);
            float area1 = 0.5f * sqrt((v2.y - point.y) * (v3.z - point.z) - (v2.z - point.z) * (v3.y - point.y));
            float area2 = 0.5f * sqrt((v3.y - point.y) * (v1.z - point.z) - (v3.z - point.z) * (v1.y - point.y));
            float area3 = 0.5f * sqrt((v1.y - point.y) * (v2.z - point.z) - (v1.z - point.z) * (v2.y - point.y));
            if (area == area1 + area2 + area3) {
                return true;
            }
        } else {
            // Check if the ray from the point to the face intersects with the face
            float t = -dotProduct / (planeNormalX * planeNormalX + planeNormalY * planeNormalY + planeNormalZ * planeNormalZ);
            Vertex intersectionPoint = {point.x + t * planeNormalX, point.y + t * planeNormalY, point.z + t * planeNormalZ};
            if (intersectionPoint.x >= min(v1.x, min(v2.x, v3.x)) && intersectionPoint.x <= max(v1.x, max(v2.x, v3.x)) &&
                intersectionPoint.y >= min(v1.y, min(v2.y, v3.y)) && intersectionPoint.y <= max(v1.y, max(v2.y, v3.y)) &&
                intersectionPoint.z >= min(v1.z, min(v2.z, v3.z)) && intersectionPoint.z <= max(v1.z, max(v2.z, v3.z))) {
                intersections++;
            }
        }
    }
    return intersections % 2 == 1;
}

// Function to check if an edge intersects with a mesh
bool doesEdgeIntersectMesh(const Vertex& v1, const Vertex& v2, const vector<Vertex>& vertices, const vector<Face>& faces) {
    for (const auto& face : faces) {
        Vertex faceV1 = vertices[face.a];
        Vertex faceV2 = vertices[face.b];
        Vertex faceV3 = vertices[face.c];

        // Check if the edge intersects with the face
        float planeNormalX = (faceV2.y - faceV1.y) * (faceV3.z - faceV1.z) - (faceV2.z - faceV1.z) * (faceV3.y - faceV1.y);
        float planeNormalY = (faceV2.z - faceV1.z) * (faceV3.x - faceV1.x) - (faceV2.x - faceV1.x) * (faceV3.z - faceV1.z);
        float planeNormalZ = (faceV2.x - faceV1.x) * (faceV3.y - faceV1.y) - (faceV2.y - faceV1.y) * (faceV3.x - faceV1.x);
        float dotProduct1 = planeNormalX * (v1.x - faceV1.x) + planeNormalY * (v1.y - faceV1.y) + planeNormalZ * (v1.z - faceV1.z);
        float dotProduct2 = planeNormalX * (v2.x - faceV1.x) + planeNormalY * (v2.y - faceV1.y) + planeNormalZ * (v2.z - faceV1.z);
        if (dotProduct1 * dotProduct2 < 0.0f) {
            // Edge intersects with the face, check if the intersection point is inside the face
            float t = -dotProduct1 / (dotProduct2 - dotProduct1);
            Vertex intersectionPoint = {v1.x + t * (v2.x - v1.x), v1.y + t * (v2.y - v1.y), v1.z + t * (v2.z - v1.z)};
            float area = 0.5f * sqrt(planeNormalX * planeNormalX + planeNormalY * planeNormalY + planeNormalZ * planeNormalZ);
            float area1 = 0.5f * sqrt((faceV2.y - intersectionPoint.y) * (faceV3.z - intersectionPoint.z) - (faceV2.z - intersectionPoint.z) * (faceV3.y - intersectionPoint.y));
            float area2 = 0.5f * sqrt((faceV3.y - intersectionPoint.y) * (faceV1.z - intersectionPoint.z) - (faceV3.z - intersectionPoint.z) * (faceV1.y - intersectionPoint.y));
            float area3 = 0.5f * sqrt((faceV1.y - intersectionPoint.y) * (faceV2.z - intersectionPoint.z) - (faceV1.z - intersectionPoint.z) * (faceV2.y - intersectionPoint.y));
            if (area == area1 + area2 + area3) {
                return true;
            }
        }
    }
    return false;
}
