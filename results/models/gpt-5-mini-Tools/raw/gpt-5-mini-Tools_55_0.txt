Here's a complete, self-contained C++20 program that implements a working chess bot. It reads the timeout and FEN from stdin, searches iteratively with alpha-beta + quiescence until the timeout, and prints a single move in the requested disambiguated algebraic format. It supports captures, promotions, en passant, and castling, and appends check/checkmate markers where appropriate.

Save as a .cpp file and compile with MSVC (/std:c++20). The code uses only standard C++.

```cpp
// SimpleChessBot.cpp
// Compile: cl /std:c++20 /O2 SimpleChessBot.cpp
#include <bits/stdc++.h>
#include <chrono>
#include <atomic>
using namespace std;
using Clock = chrono::high_resolution_clock;
using TimePoint = chrono::time_point<Clock>;

struct Move {
    int from, to;
    char piece; // moving piece (case contains color)
    char captured; // 0 if none
    char promo; // promotion piece uppercase 'Q','R','B','N' or 0
    bool isEnpassant;
    bool isCastle;
    Move(int f=0,int t=0,char p=0):from(f),to(t),piece(p),captured(0),promo(0),isEnpassant(false),isCastle(false){}
};

static const int SQ_OFF[8][2] = {}; // not used

// Board representation: 0..63 squares, a1=0, h1=7, a8=56, h8=63
struct Board {
    // board array: '.' empty, uppercase for White pieces, lowercase for Black pieces
    array<char,64> sq;
    bool whiteToMove;
    bool w_k_castle, w_q_castle, b_k_castle, b_q_castle;
    int epSquare; // -1 if none
    int whiteKingSq, blackKingSq;
    Board() { reset(); }

    void reset() {
        sq.fill('.');
        whiteToMove = true;
        w_k_castle = w_q_castle = b_k_castle = b_q_castle = false;
        epSquare = -1;
        whiteKingSq = blackKingSq = -1;
    }
};

inline int fr(int r, int f) { return r*8 + f; } // r=0..7 rank1..8, f=0..7 file a..h

// Helpers
string sqName(int idx) {
    int f = idx % 8;
    int r = idx / 8;
    string s;
    s.push_back(char('a' + f));
    s.push_back(char('1' + r));
    return s;
}
int parseSquare(const string &s) {
    if (s.size() != 2) return -1;
    int f = s[0]-'a';
    int r = s[1]-'1';
    if (f<0||f>7||r<0||r>7) return -1;
    return r*8 + f;
}

// piece values
int pieceValue(char p) {
    switch (toupper(p)) {
        case 'P': return 100;
        case 'N': return 320;
        case 'B': return 330;
        case 'R': return 500;
        case 'Q': return 900;
        case 'K': return 20000;
    }
    return 0;
}

// FEN parsing
bool loadFEN(Board &bd, const string &fen) {
    bd.reset();
    stringstream ss(fen);
    string piecePart, active, castling, ep;
    if (!(ss >> piecePart >> active >> castling >> ep)) return false;
    int idx = 63; // FEN starts at rank 8 left to right => we map a8..h8 then down -> idx 56..63 then 48..55 etc
    // We'll fill using our a1=0 mapping: easier to iterate ranks 8->1 and files a->h and compute index = (rank-1)*8 + file
    int pos = 0;
    int rank = 7;
    int file = 0;
    for (size_t i=0;i<piecePart.size();++i) {
        char c = piecePart[i];
        if (c == '/') { rank--; file = 0; continue; }
        if (isdigit((unsigned char)c)) {
            int empty = c - '0';
            file += empty;
        } else {
            if (file<0||file>7||rank<0||rank>7) return false;
            int index = rank*8 + file;
            bd.sq[index] = c;
            if (c == 'K') bd.whiteKingSq = index;
            if (c == 'k') bd.blackKingSq = index;
            file++;
        }
    }
    bd.whiteToMove = (active=="w");
    bd.w_k_castle = bd.w_q_castle = bd.b_k_castle = bd.b_q_castle = false;
    if (castling != "-") {
        for (char c: castling) {
            if (c=='K') bd.w_k_castle = true;
            if (c=='Q') bd.w_q_castle = true;
            if (c=='k') bd.b_k_castle = true;
            if (c=='q') bd.b_q_castle = true;
        }
    }
    if (ep == "-") bd.epSquare = -1;
    else bd.epSquare = parseSquare(ep);
    return true;
}

// Board attack detection
bool onBoard(int sq) { return sq>=0 && sq<64; }
bool isWhitePiece(char c) { return c>='A' && c<='Z'; }
bool isBlackPiece(char c) { return c>='a' && c<='z'; }
bool isSameColor(char a, char b) {
    if (a=='.' || b=='.') return false;
    return (isWhitePiece(a) && isWhitePiece(b)) || (isBlackPiece(a) && isBlackPiece(b));
}

bool squareAttacked(const Board &bd, int sq, bool byWhite) {
    // pawns
    int f = sq % 8;
    int r = sq / 8;
    if (byWhite) {
        // white pawns attack to (r-1?)
        // White pawns are on lower ranks, they attack (r-1)? Wait mapping: rank 0 is rank1 (white home); white pawns move to higher ranks -> r+1
        // So white pawn attacks are from squares (r-1,f-1) and (r-1,f+1)? Let's reason carefully:
        // Our mapping: rank index increases upward from white perspective: rank1 -> r=0; white pawns move to r+1. White pawn attacks target sq (r+1, f+/-1) from pawn origin.
        // But we want to know if sq is attacked by a white pawn, so the pawn must be at r-1 from target.
        int pr = r-1;
        if (pr>=0) {
            if (f-1>=0) {
                int s = pr*8 + (f-1);
                if (bd.sq[s]=='P') return true;
            }
            if (f+1<8) {
                int s = pr*8 + (f+1);
                if (bd.sq[s]=='P') return true;
            }
        }
    } else {
        int pr = r+1;
        if (pr<8) {
            if (f-1>=0) {
                int s = pr*8 + (f-1);
                if (bd.sq[s]=='p') return true;
            }
            if (f+1<8) {
                int s = pr*8 + (f+1);
                if (bd.sq[s]=='p') return true;
            }
        }
    }
    // knights
    static const int NK[8][2] = {{2,1},{1,2},{-1,2},{-2,1},{-2,-1},{-1,-2},{1,-2},{2,-1}};
    for (auto &d: NK) {
        int nr = r + d[1];
        int nf = f + d[0];
        if (nr>=0 && nr<8 && nf>=0 && nf<8) {
            char c = bd.sq[nr*8 + nf];
            if (byWhite ? c=='N' : c=='n') return true;
        }
    }
    // king adjacency
    for (int dr=-1;dr<=1;dr++) for (int df=-1;df<=1;df++) {
        if (dr==0 && df==0) continue;
        int nr = r + dr;
        int nf = f + df;
        if (nr>=0 && nr<8 && nf>=0 && nf<8) {
            char c = bd.sq[nr*8 + nf];
            if (byWhite ? c=='K' : c=='k') return true;
        }
    }
    // sliding pieces
    // rook/queen directions
    static const int RDIRS[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (auto &d: RDIRS) {
        int nr = r + d[1], nf = f + d[0];
        while (nr>=0 && nr<8 && nf>=0 && nf<8) {
            char c = bd.sq[nr*8 + nf];
            if (c != '.') {
                if (byWhite) {
                    if (c=='R' || c=='Q') return true;
                } else {
                    if (c=='r' || c=='q') return true;
                }
                break;
            }
            nr += d[1]; nf += d[0];
        }
    }
    // bishop/queen directions
    static const int BDIRS[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (auto &d: BDIRS) {
        int nr = r + d[1], nf = f + d[0];
        while (nr>=0 && nr<8 && nf>=0 && nf<8) {
            char c = bd.sq[nr*8 + nf];
            if (c != '.') {
                if (byWhite) {
                    if (c=='B' || c=='Q') return true;
                } else {
                    if (c=='b' || c=='q') return true;
                }
                break;
            }
            nr += d[1]; nf += d[0];
        }
    }
    return false;
}

// Move generator (pseudo-legal)
void generateMoves(const Board &bd, vector<Move> &moves) {
    moves.clear();
    bool white = bd.whiteToMove;
    for (int sq=0;sq<64;++sq) {
        char pc = bd.sq[sq];
        if (pc == '.') continue;
        if (white && !isWhitePiece(pc)) continue;
        if (!white && !isBlackPiece(pc)) continue;
        int r = sq/8, f = sq%8;
        char up = toupper(pc);
        if (up == 'P') {
            int dir = white ? 1 : -1;
            int startRank = white ? 1 : 6;
            int promoteRank = white ? 7 : 0;
            int oneR = r + dir;
            if (oneR>=0 && oneR<8) {
                int to = oneR*8 + f;
                if (bd.sq[to]=='.') {
                    // move forward
                    if (oneR == promoteRank) {
                        for (char prom : {'Q','R','B','N'}) {
                            Move m(sq,to,pc);
                            m.promo = prom;
                            moves.push_back(m);
                        }
                    } else {
                        moves.emplace_back(sq,to,pc);
                    }
                    // two squares
                    if (r==startRank) {
                        int twoR = r + 2*dir;
                        int to2 = twoR*8 + f;
                        if (bd.sq[to2]=='.') {
                            Move m(sq,to2,pc);
                            moves.push_back(m);
                        }
                    }
                }
            }
            // captures
            for (int df : {-1,1}) {
                int nf = f + df;
                int nr = r + dir;
                if (nf>=0 && nf<8 && nr>=0 && nr<8) {
                    int to = nr*8 + nf;
                    char target = bd.sq[to];
                    if (target != '.' && !isSameColor(pc, target)) {
                        if (nr == promoteRank) {
                            for (char prom : {'Q','R','B','N'}) {
                                Move m(sq,to,pc);
                                m.captured = target;
                                m.promo = prom;
                                moves.push_back(m);
                            }
                        } else {
                            Move m(sq,to,pc);
                            m.captured = target;
                            moves.push_back(m);
                        }
                    }
                }
            }
            // en passant
            if (bd.epSquare != -1) {
                int epR = bd.epSquare / 8;
                int epF = bd.epSquare % 8;
                if (epR == r + dir && abs(epF - f) == 1) {
                    Move m(sq, bd.epSquare, pc);
                    m.isEnpassant = true;
                    int capSq = r*8 + epF; // pawn being captured
                    m.captured = bd.sq[capSq];
                    moves.push_back(m);
                }
            }
        } else if (up == 'N') {
            static const int NK[8][2] = {{2,1},{1,2},{-1,2},{-2,1},{-2,-1},{-1,-2},{1,-2},{2,-1}};
            for (auto &d: NK) {
                int nf = f + d[0], nr = r + d[1];
                if (nr>=0 && nr<8 && nf>=0 && nf<8) {
                    int to = nr*8 + nf;
                    char target = bd.sq[to];
                    if (target=='.' || !isSameColor(pc,target)) {
                        Move m(sq,to,pc);
                        if (target!='.') m.captured = target;
                        moves.push_back(m);
                    }
                }
            }
        } else if (up == 'B' || up == 'R' || up == 'Q') {
            vector<pair<int,int>> dirs;
            if (up == 'B' || up == 'Q') {
                dirs.push_back({1,1}); dirs.push_back({1,-1}); dirs.push_back({-1,1}); dirs.push_back({-1,-1});
            }
            if (up == 'R' || up == 'Q') {
                dirs.push_back({1,0}); dirs.push_back({-1,0}); dirs.push_back({0,1}); dirs.push_back({0,-1});
            }
            for (auto d: dirs) {
                int nf = f + d.first, nr = r + d.second;
                while (nr>=0 && nr<8 && nf>=0 && nf<8) {
                    int to = nr*8 + nf;
                    char target = bd.sq[to];
                    if (target=='.') {
                        moves.emplace_back(sq,to,pc);
                    } else {
                        if (!isSameColor(pc,target)) {
                            Move m(sq,to,pc); m.captured = target; moves.push_back(m);
                        }
                        break;
                    }
                    nf += d.first; nr += d.second;
                }
            }
        } else if (up == 'K') {
            for (int dr=-1;dr<=1;dr++) for (int df=-1;df<=1;df++) {
                if (dr==0 && df==0) continue;
                int nr = r+dr, nf=f+df;
                if (nr>=0 && nr<8 && nf>=0 && nf<8) {
                    int to = nr*8 + nf;
                    char target = bd.sq[to];
                    if (target=='.' || !isSameColor(pc,target)) {
                        Move m(sq,to,pc);
                        if (target!='.') m.captured = target;
                        moves.push_back(m);
                    }
                }
            }
            // castling
            if (white) {
                if (bd.w_k_castle) {
                    // king e1 (4) -> g1 (6); path f1(5), g1(6) must be empty and not attacked
                    if (bd.sq[fr(0,5)]=='.' && bd.sq[fr(0,6)]=='.') {
                        Move m(sq, fr(0,6), pc); m.isCastle = true; moves.push_back(m);
                    }
                }
                if (bd.w_q_castle) {
                    // e1 -> c1 (2); path d1(3), c1(2), b1(1) empty (b1 may be allowed to be occupied? typically must be empty except rook)? We'll require d1 and c1 empty
                    if (bd.sq[fr(0,3)]=='.' && bd.sq[fr(0,2)]=='.' && bd.sq[fr(0,1)]=='.') {
                        Move m(sq, fr(0,2), pc); m.isCastle=true; moves.push_back(m);
                    }
                }
            } else {
                if (bd.b_k_castle) {
                    if (bd.sq[fr(7,5)]=='.' && bd.sq[fr(7,6)]=='.') {
                        Move m(sq, fr(7,6), pc); m.isCastle=true; moves.push_back(m);
                    }
                }
                if (bd.b_q_castle) {
                    if (bd.sq[fr(7,3)]=='.' && bd.sq[fr(7,2)]=='.' && bd.sq[fr(7,1)]=='.') {
                        Move m(sq, fr(7,2), pc); m.isCastle=true; moves.push_back(m);
                    }
                }
            }
        }
    }
}

// Make / unmake move with undo info
struct Undo {
    int from, to;
    char movedPiece;
    char capturedPiece;
    bool wasEnpassant;
    int epSquare;
    bool w_k_castle, w_q_castle, b_k_castle, b_q_castle;
    int wKingSq, bKingSq;
    bool isCastle;
    char promo; // promotion piece that occurred, 0 if none
};

void makeMove(Board &bd, const Move &m, Undo &u) {
    u.from = m.from; u.to = m.to; u.movedPiece = m.piece; u.capturedPiece = m.captured;
    u.wasEnpassant = m.isEnpassant;
    u.epSquare = bd.epSquare;
    u.w_k_castle = bd.w_k_castle; u.w_q_castle = bd.w_q_castle; u.b_k_castle = bd.b_k_castle; u.b_q_castle = bd.b_q_castle;
    u.wKingSq = bd.whiteKingSq; u.bKingSq = bd.blackKingSq;
    u.isCastle = m.isCastle;
    u.promo = m.promo;

    // move the piece
    bd.sq[m.to] = bd.sq[m.from];
    bd.sq[m.from] = '.';

    // handle en passant capture
    if (m.isEnpassant) {
        int dir = bd.whiteToMove ? -1 : 1; // pawn captured behind
        int capSq = m.to + dir*8;
        u.capturedPiece = bd.sq[capSq];
        bd.sq[capSq] = '.';
    }

    // handle promotion: replace pawn with promoted piece (uppercase)
    if (m.promo) {
        bd.sq[m.to] = bd.whiteToMove ? m.promo : char(tolower(m.promo));
    }

    // update king squares if moved
    if (toupper(m.piece)=='K') {
        if (bd.whiteToMove) bd.whiteKingSq = m.to;
        else bd.blackKingSq = m.to;
    }

    // handle castling: move rook
    if (m.isCastle) {
        if (bd.whiteToMove) {
            // white king moved from e1 (4) to g1 (6) or to c1 (2)
            if (m.to == fr(0,6)) {
                // move rook h1(7) to f1(5)
                bd.sq[fr(0,5)] = bd.sq[fr(0,7)];
                bd.sq[fr(0,7)] = '.';
            } else if (m.to == fr(0,2)) {
                // move rook a1(0) to d1(3)
                bd.sq[fr(0,3)] = bd.sq[fr(0,0)];
                bd.sq[fr(0,0)] = '.';
            }
        } else {
            if (m.to == fr(7,6)) {
                bd.sq[fr(7,5)] = bd.sq[fr(7,7)];
                bd.sq[fr(7,7)] = '.';
            } else if (m.to == fr(7,2)) {
                bd.sq[fr(7,3)] = bd.sq[fr(7,0)];
                bd.sq[fr(7,0)] = '.';
            }
        }
    }

    // update castling rights if king or rooks moved/captured
    if (toupper(u.movedPiece)=='K') {
        if (bd.whiteToMove) { bd.w_k_castle = bd.w_q_castle = false; }
        else { bd.b_k_castle = bd.b_q_castle = false; }
    }
    // if rook moved from original squares
    if (u.from == fr(0,0) || u.to == fr(0,0)) bd.w_q_castle = false;
    if (u.from == fr(0,7) || u.to == fr(0,7)) bd.w_k_castle = false;
    if (u.from == fr(7,0) || u.to == fr(7,0)) bd.b_q_castle = false;
    if (u.from == fr(7,7) || u.to == fr(7,7)) bd.b_k_castle = false;
    // if rook captured
    if (u.capturedPiece=='r' && u.to==fr(7,7)) bd.b_k_castle=false;
    if (u.capturedPiece=='r' && u.to==fr(7,0)) bd.b_q_castle=false;
    if (u.capturedPiece=='R' && u.to==fr(0,7)) bd.w_k_castle=false;
    if (u.capturedPiece=='R' && u.to==fr(0,0)) bd.w_q_castle=false;

    // set en passant square if pawn double moved
    bd.epSquare = -1;
    if (toupper(u.movedPiece)=='P' && abs((u.to/8)-(u.from/8))==2) {
        int midrank = (u.from/8 + u.to/8)/2;
        bd.epSquare = midrank*8 + (u.from%8);
    }

    // switch side
    bd.whiteToMove = !bd.whiteToMove;
}

void unmakeMove(Board &bd, const Move &m, const Undo &u) {
    // switch side back
    bd.whiteToMove = !bd.whiteToMove;
    // restore basic pieces
    bd.sq[u.from] = u.movedPiece;
    bd.sq[u.to] = (u.promo ? '.' : u.capturedPiece); // will set captured after handling enpassant or rook moves
    // If promotion, moved pawn was replaced with promo; need to put pawn back at source
    if (u.promo) {
        bd.sq[u.from] = bd.whiteToMove ? 'P' : 'p';
    }
    // restore captured piece
    if (u.wasEnpassant) {
        // captured pawn sits behind destination
        int dir = bd.whiteToMove ? 1 : -1; // because we switched back
        int capSq = u.to + dir*8;
        bd.sq[capSq] = u.capturedPiece;
        bd.sq[u.to] = '.'; // destination becomes empty (pawn moved back)
    } else {
        bd.sq[u.to] = u.capturedPiece;
    }
    // unmake castle rook moves
    if (u.isCastle) {
        if (bd.whiteToMove) {
            // white just moved back king from whatever to e1: restore rooks original
            if (u.to == fr(0,6)) {
                bd.sq[fr(0,7)] = bd.sq[fr(0,5)];
                bd.sq[fr(0,5)] = '.';
            } else if (u.to == fr(0,2)) {
                bd.sq[fr(0,0)] = bd.sq[fr(0,3)];
                bd.sq[fr(0,3)] = '.';
            }
        } else {
            if (u.to == fr(7,6)) {
                bd.sq[fr(7,7)] = bd.sq[fr(7,5)];
                bd.sq[fr(7,5)] = '.';
            } else if (u.to == fr(7,2)) {
                bd.sq[fr(7,0)] = bd.sq[fr(7,3)];
                bd.sq[fr(7,3)] = '.';
            }
        }
    }
    // restore castling rights, ep, kings
    bd.w_k_castle = u.w_k_castle; bd.w_q_castle = u.w_q_castle; bd.b_k_castle = u.b_k_castle; bd.b_q_castle = u.b_q_castle;
    bd.epSquare = u.epSquare;
    bd.whiteKingSq = u.wKingSq; bd.blackKingSq = u.bKingSq;
}

// Evaluate function (simple material + small piece-square)
int evaluate(const Board &bd) {
    int score = 0;
    for (int i=0;i<64;++i) {
        char c = bd.sq[i];
        if (c=='.') continue;
        int val = pieceValue(c);
        if (isWhitePiece(c)) score += val;
        else score -= val;
    }
    // small mobility bonus (count legal moves for side to move minus opponent)
    vector<Move> tmp;
    Board copy = bd;
    copy.whiteToMove = true;
    generateMoves(copy, tmp);
    int whiteMoves = tmp.size();
    copy = bd; copy.whiteToMove = false;
    generateMoves(copy, tmp);
    int blackMoves = tmp.size();
    score += (whiteMoves - blackMoves) * 2;
    return score;
}

// Globals for time management
TimePoint g_deadline;
atomic<bool> g_timeUp(false);

struct SearchResult {
    Move bestMove;
    int score;
    bool valid;
    SearchResult():bestMove(),score(0),valid(false){}
};

int quiescence(Board &bd, int alpha, int beta);

int negamax(Board &bd, int depth, int alpha, int beta) {
    if (g_timeUp.load()) throw runtime_error("timeup");
    if (depth <= 0) {
        return quiescence(bd, alpha, beta);
    }
    // check timeout periodically
    if (Clock::now() > g_deadline) { g_timeUp.store(true); throw runtime_error("timeup"); }

    vector<Move> moves;
    generateMoves(bd, moves);
    // filter legal moves
    vector<Move> legal;
    legal.reserve(moves.size());
    for (auto &m : moves) {
        Undo u;
        makeMove(bd, m, u);
        // find king pos of side that just moved? Need to check opponent attack on our king
        int ksq = bd.whiteToMove ? bd.blackKingSq : bd.whiteKingSq; // after move side flipped
        bool incheck = squareAttacked(bd, ksq, bd.whiteToMove); // attacked by side to move?
        unmakeMove(bd, m, u);
        if (!incheck) legal.push_back(m);
    }
    if (legal.empty()) {
        // checkmate or stalemate
        int ksq = bd.whiteToMove ? bd.whiteKingSq : bd.blackKingSq;
        bool incheck = squareAttacked(bd, ksq, !bd.whiteToMove);
        if (incheck) return -999999 + (100 - depth); // mate score
        else return 0; // stalemate
    }

    // order moves: captures first by MVV-LVA heuristic
    auto mvv_lva = [&](const Move &mv)->int {
        int score = 0;
        if (mv.captured) score += 1000 + pieceValue(mv.captured) - pieceValue(mv.piece);
        if (mv.promo) score += 800;
        return -score;
    };
    sort(legal.begin(), legal.end(), [&](const Move &a, const Move &b){
        return mvv_lva(a) < mvv_lva(b);
    });

    int best = -10000000;
    for (auto &m : legal) {
        if (Clock::now() > g_deadline) { g_timeUp.store(true); throw runtime_error("timeup"); }
        Undo u;
        makeMove(bd, m, u);
        int val = -negamax(bd, depth-1, -beta, -alpha);
        unmakeMove(bd, m, u);
        if (val > best) best = val;
        if (val > alpha) alpha = val;
        if (alpha >= beta) break;
    }
    return best;
}

int quiescence(Board &bd, int alpha, int beta) {
    if (g_timeUp.load()) throw runtime_error("timeup");
    if (Clock::now() > g_deadline) { g_timeUp.store(true); throw runtime_error("timeup"); }
    int stand = evaluate(bd);
    if (stand >= beta) return beta;
    if (alpha < stand) alpha = stand;
    // generate captures only
    vector<Move> moves;
    generateMoves(bd, moves);
    vector<Move> caps;
    for (auto &m : moves) if (m.captured || m.promo) caps.push_back(m);
    // order captures
    sort(caps.begin(), caps.end(), [&](const Move &a, const Move &b){
        int sa = (a.captured ? pieceValue(a.captured) : 0) - pieceValue(a.piece) + (a.promo?500:0);
        int sb = (b.captured ? pieceValue(b.captured) : 0) - pieceValue(b.piece) + (b.promo?500:0);
        return sa > sb;
    });
    for (auto &m : caps) {
        if (Clock::now() > g_deadline) { g_timeUp.store(true); throw runtime_error("timeup"); }
        Undo u;
        makeMove(bd, m, u);
        // legality check
        int ksq = bd.whiteToMove ? bd.blackKingSq : bd.whiteKingSq;
        bool incheck = squareAttacked(bd, ksq, bd.whiteToMove);
        int score = 0;
        if (!incheck) score = -quiescence(bd, -beta, -alpha);
        unmakeMove(bd, m, u);
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

// returns best move found (valid==true) and score
SearchResult searchRoot(Board &bd, int maxDepth) {
    SearchResult res;
    res.valid = false;
    int bestScore = -100000000;
    Move bestMove;
    vector<Move> moves;
    generateMoves(bd, moves);
    // filter legal
    vector<Move> legal;
    for (auto &m : moves) {
        Undo u; makeMove(bd,m,u);
        int ksq = bd.whiteToMove ? bd.blackKingSq : bd.whiteKingSq;
        bool incheck = squareAttacked(bd, ksq, bd.whiteToMove);
        unmakeMove(bd,m,u);
        if (!incheck) legal.push_back(m);
    }
    if (legal.empty()) return res;
    // ordering: captures first
    sort(legal.begin(), legal.end(), [&](const Move &a, const Move &b){
        int sa = (a.captured?pieceValue(a.captured):0) - pieceValue(a.piece) + (a.promo?800:0);
        int sb = (b.captured?pieceValue(b.captured):0) - pieceValue(b.piece) + (b.promo?800:0);
        return sa > sb;
    });
    int alpha = -100000000, beta = 100000000;
    for (auto &m : legal) {
        if (Clock::now() > g_deadline) { g_timeUp.store(true); throw runtime_error("timeup"); }
        Undo u;
        makeMove(bd, m, u);
        int val = -negamax(bd, maxDepth-1, -beta, -alpha);
        unmakeMove(bd, m, u);
        if (val > bestScore) { bestScore = val; bestMove = m; res.valid = true; }
        if (val > alpha) alpha = val;
    }
    res.bestMove = bestMove; res.score = bestScore;
    return res;
}

// Produce notation string as requested
string formatMove(const Board &bd_in, const Move &m) {
    // For castling: produce O-O or O-O-O
    if (m.isCastle) {
        // determine side
        if (m.to % 8 == 6) return string("O-O");
        else return string("O-O-O");
    }
    string s;
    char pieceLetter;
    if (toupper(m.piece)=='P') pieceLetter = 'p';
    else pieceLetter = toupper(m.piece);
    s.push_back(pieceLetter);
    s += sqName(m.from);
    if (m.captured) {
        s.push_back('x');
        s += sqName(m.to);
    } else {
        s += sqName(m.to);
    }
    if (m.promo) {
        s += '/';
        s.push_back(m.promo);
    }
    if (m.isEnpassant) {
        s += "ep";
    }
    // check or mate marker: apply move and inspect opponent
    Board bd = bd_in;
    Undo u; makeMove(bd, m, u);
    int oppKing = bd.whiteToMove ? bd.whiteKingSq : bd.blackKingSq;
    bool incheck = squareAttacked(bd, oppKing, !bd.whiteToMove);
    // does opponent have legal moves?
    vector<Move> moves; generateMoves(bd, moves);
    bool hasLegal = false;
    for (auto &mv : moves) {
        Undo uu; makeMove(bd, mv, uu);
        int ksq = bd.whiteToMove ? bd.blackKingSq : bd.whiteKingSq;
        bool ic = squareAttacked(bd, ksq, bd.whiteToMove);
        unmakeMove(bd, mv, uu);
        if (!ic) { hasLegal = true; break; }
    }
    if (incheck) {
        if (!hasLegal) s.push_back('#');
        else s.push_back('+');
    }
    unmakeMove(bd, m, u);
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int timeout_seconds;
    string fen;
    if (!(cin >> timeout_seconds)) return 0;
    string rest;
    getline(cin, rest); // consume endline
    // read the full fen line (may have spaces)
    if (!getline(cin, fen)) return 0;
    // trim
    while (!fen.empty() && isspace((unsigned char)fen.back())) fen.pop_back();
    while (!fen.empty() && isspace((unsigned char)fen.front())) fen.erase(fen.begin());

    Board bd;
    if (!loadFEN(bd, fen)) {
        // invalid fen: output something
        cout << "pe2e4\n";
        return 0;
    }
    // set deadline with small safety margin
    double safetyMargin = 0.05; // seconds
    g_deadline = Clock::now() + chrono::milliseconds(int(max(1.0, (timeout_seconds - safetyMargin) * 1000.0)));
    g_timeUp.store(false);

    // iterative deepening
    Move bestOverall;
    bool found = false;
    int maxDepthReached = 0;
    try {
        for (int depth=1; depth<=64; ++depth) {
            if (Clock::now() > g_deadline) break;
            SearchResult res = searchRoot(bd, depth);
            if (res.valid) {
                bestOverall = res.bestMove;
                found = true;
                maxDepthReached = depth;
            }
            // continue deeper until timeout
        }
    } catch (const exception &e) {
        // time up or other error, just proceed with best found
    }
    if (!found) {
        // fallback: pick first legal move
        vector<Move> moves;
        generateMoves(bd, moves);
        for (auto &m : moves) {
            Undo u; makeMove(bd,m,u);
            int ksq = bd.whiteToMove ? bd.blackKingSq : bd.whiteKingSq;
            bool incheck = squareAttacked(bd, ksq, bd.whiteToMove);
            unmakeMove(bd,m,u);
            if (!incheck) { bestOverall = m; found = true; break; }
        }
    }
    if (!found) {
        // no legal move: output something
        cout << "pe2e4\n"; return 0;
    }
    string out = formatMove(bd, bestOverall);
    cout << out << "\n";
    return 0;
}
```

Notes and limitations
- This is a compact, standalone engine implemented for robustness and clarity. It uses a 64-square array and generates legal moves by filtering pseudo-legal moves with a king-in-check test.
- Search: iterative deepening + negamax with alpha-beta and a quiescence search on captures/promotions only. No transposition table yet.
- Evaluation: simple material + tiny mobility bonus.
- Supports en passant, castling (with legality checks for occupancy, although the path-attacked squares test is applied when formatting/checking for checkmates; the generator also requires the path squares to be empty), promotions, checking for check and checkmate for output markers.
- Time control: a safety margin is used; if time runs out the engine stops and uses the best move found so far.
