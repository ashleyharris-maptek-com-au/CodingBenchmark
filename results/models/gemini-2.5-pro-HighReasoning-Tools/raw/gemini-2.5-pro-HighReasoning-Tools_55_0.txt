#include "movegen.h"
#include <vector>

void generatePawnMoves(const Board& board, int row, int col, std::vector<Move>& moves);
void generateKnightMoves(const Board& board, int row, int col, std::vector<Move>& moves);
void generateSlidingMoves(const Board& board, int row, int col, int dr[], int dc[], int num_directions, std::vector<Move>& moves);
void generateRookMoves(const Board& board, int row, int col, std::vector<Move>& moves);
void generateBishopMoves(const Board& board, int row, int col, std::vector<Move>& moves);
void generateQueenMoves(const Board& board, int row, int col, std::vector<Move>& moves);
void generateKingMoves(const Board& board, int row, int col, std::vector<Move>& moves);


void generateMoves(const Board& board, std::vector<Move>& moves) {
    Color side = board.getSideToMove();
    for (int i = 0; i < 8; ++i) {
        for (int j = 0; j < 8; ++j) {
            Piece piece = board.getPiece(i, j);
            if (piece == EMPTY) continue;
            Color pieceColor = (piece >= B_PAWN) ? BLACK : WHITE;
            if (pieceColor != side) continue;

            switch (piece) {
                case W_PAWN:
                case B_PAWN:
                    generatePawnMoves(board, i, j, moves);
                    break;
                case W_KNIGHT:
                case B_KNIGHT:
                    generateKnightMoves(board, i, j, moves);
                    break;
                case W_ROOK:
                case B_ROOK:
                    generateRookMoves(board, i, j, moves);
                    break;
                case W_BISHOP:
                case B_BISHOP:
                    generateBishopMoves(board, i, j, moves);
                    break;
                case W_QUEEN:
                case B_QUEEN:
                    generateQueenMoves(board, i, j, moves);
                    break;
                case W_KING:
                case B_KING:
                    generateKingMoves(board, i, j, moves);
                    break;
                default:
                    break;
            }
        }
    }
}

void generatePawnMoves(const Board& board, int row, int col, std::vector<Move>& moves) {
    Piece pawn = board.getPiece(row, col);
    int direction = (pawn == W_PAWN) ? 1 : -1;
    int startRow = (pawn == W_PAWN) ? 1 : 6;

    // Single push
    if (board.getPiece(row + direction, col) == EMPTY) {
        moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)(row + direction), (uint8_t)col});
        // Double push
        if (row == startRow && board.getPiece(row + 2 * direction, col) == EMPTY) {
            moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)(row + 2 * direction), (uint8_t)col});
        }
    }

    // Captures
    for (int dcol = -1; dcol <= 1; dcol += 2) {
        if (col + dcol >= 0 && col + dcol < 8) {
            Piece target = board.getPiece(row + direction, col + dcol);
            if (target != EMPTY && ((pawn == W_PAWN && target >= B_PAWN) || (pawn == B_PAWN && target <= W_KING))) {
                moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)(row + direction), (uint8_t)(col + dcol)});
            }
        }
    }
}

void generateKnightMoves(const Board& board, int row, int col, std::vector<Move>& moves) {
    int dr[] = {-2, -2, -1, -1, 1, 1, 2, 2};
    int dc[] = {-1, 1, -2, 2, -2, 2, -1, 1};

    for (int i = 0; i < 8; ++i) {
        int new_row = row + dr[i];
        int new_col = col + dc[i];

        if (new_row >= 0 && new_row < 8 && new_col >= 0 && new_col < 8) {
            Piece target = board.getPiece(new_row, new_col);
            if (target == EMPTY || (board.getSideToMove() == WHITE && target >= B_PAWN) || (board.getSideToMove() == BLACK && target <= W_KING)) {
                moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)new_row, (uint8_t)new_col});
            }
        }
    }
}

void generateSlidingMoves(const Board& board, int row, int col, int dr[], int dc[], int num_directions, std::vector<Move>& moves) {
    for (int i = 0; i < num_directions; ++i) {
        for (int j = 1; j < 8; ++j) {
            int new_row = row + j * dr[i];
            int new_col = col + j * dc[i];

            if (new_row >= 0 && new_row < 8 && new_col >= 0 && new_col < 8) {
                Piece target = board.getPiece(new_row, new_col);
                if (target == EMPTY) {
                    moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)new_row, (uint8_t)new_col});
                } else {
                    if ((board.getSideToMove() == WHITE && target >= B_PAWN) || (board.getSideToMove() == BLACK && target <= W_KING)) {
                        moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)new_row, (uint8_t)new_col});
                    }
                    break;
                }
            } else {
                break;
            }
        }
    }
}

void generateRookMoves(const Board& board, int row, int col, std::vector<Move>& moves) {
    int dr[] = {-1, 1, 0, 0};
    int dc[] = {0, 0, -1, 1};
    generateSlidingMoves(board, row, col, dr, dc, 4, moves);
}

void generateBishopMoves(const Board& board, int row, int col, std::vector<Move>& moves) {
    int dr[] = {-1, -1, 1, 1};
    int dc[] = {-1, 1, -1, 1};
    generateSlidingMoves(board, row, col, dr, dc, 4, moves);
}

void generateQueenMoves(const Board& board, int row, int col, std::vector<Move>& moves) {
    generateRookMoves(board, row, col, moves);
    generateBishopMoves(board, row, col, moves);
}

void generateKingMoves(const Board& board, int row, int col, std::vector<Move>& moves) {
    int dr[] = {-1, -1, -1, 0, 0, 1, 1, 1};
    int dc[] = {-1, 0, 1, -1, 1, -1, 0, 1};

    for (int i = 0; i < 8; ++i) {
        int new_row = row + dr[i];
        int new_col = col + dc[i];

        if (new_row >= 0 && new_row < 8 && new_col >= 0 && new_col < 8) {
            Piece target = board.getPiece(new_row, new_col);
            if (target == EMPTY || (board.getSideToMove() == WHITE && target >= B_PAWN) || (board.getSideToMove() == BLACK && target <= W_KING)) {
                moves.push_back({(uint8_t)row, (uint8_t)col, (uint8_t)new_row, (uint8_t)new_col});
            }
        }
