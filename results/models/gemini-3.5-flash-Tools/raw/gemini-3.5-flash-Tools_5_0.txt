use std::io::{self, BufRead};
use std::time::{Instant, Duration};

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
struct Point {
    x: i32,
    y: i32,
}

struct Grid {
    width: i32,
    height: i32,
    blocked: Vec<Vec<bool>>,
    visited: Vec<Vec<bool>>,
    total_nodes: usize,
    require_cycle: bool,
}

impl Grid {
    #[inline]
    fn in_bounds(&self, p: Point) -> bool {
        p.x >= 0 && p.x < self.width && p.y >= 0 && p.y < self.height
    }

    #[inline]
    fn is_free(&self, p: Point) -> bool {
        self.in_bounds(p) && !self.blocked[p.y as usize][p.x as usize] && !self.visited[p.y as usize][p.x as usize]
    }

    fn get_free_neighbors(&self, p: Point) -> Vec<Point> {
        let mut neighbors = Vec::with_capacity(4);
        let dirs = [Point { x: 0, y: 1 }, Point { x: 0, y: -1 }, Point { x: 1, y: 0 }, Point { x: -1, y: 0 }];
        for d in dirs {
            let next = Point { x: p.x + d.x, y: p.y + d.y };
            if self.is_free(next) {
                neighbors.push(next);
            }
        }
        neighbors
    }

    fn get_degree(&self, p: Point) -> usize {
        let dirs = [Point { x: 0, y: 1 }, Point { x: 0, y: -1 }, Point { x: 1, y: 0 }, Point { x: -1, y: 0 }];
        let mut deg = 0;
        for d in dirs {
            let next = Point { x: p.x + d.x, y: p.y + d.y };
            if self.is_free(next) {
                deg += 1;
            }
        }
        deg
    }
}

// Highly optimized BFS check that stops as soon as target_count is reached
fn check_connectivity(grid: &Grid, start: Point, target_count: usize) -> bool {
    let mut visited = vec![vec![false; grid.width as usize]; grid.height as usize];
    let mut queue = std::collections::VecDeque::new();
    queue.push_back(start);
    visited[start.y as usize][start.x as usize] = true;
    let mut count = 0;

    while let Some(p) = queue.pop_front() {
        count += 1;
        if count == target_count {
            return true;
        }
        let dirs = [Point { x: 0, y: 1 }, Point { x: 0, y: -1 }, Point { x: 1, y: 0 }, Point { x: -1, y: 0 }];
        for d in dirs {
            let next = Point { x: p.x + d.x, y: p.y + d.y };
            if grid.is_free(next) && !visited[next.y as usize][next.x as usize] {
                visited[next.y as usize][next.x as usize] = true;
                queue.push_back(next);
            }
        }
    }
    count == target_count
}

fn solve(
    current: Point,
    grid: &mut Grid,
    path: &mut Vec<Point>,
    start_time: &Instant,
    timeout: &Duration,
    step_count: &mut usize,
) -> bool {
    *step_count += 1;
    if *step_count % 2048 == 0 && start_time.elapsed() > *timeout {
        return false;
    }

    let remaining_unvisited_count = grid.total_nodes - path.len();
    if remaining_unvisited_count == 0 {
        if grid.require_cycle {
            let dist = (current.x - 0).abs() + (current.y - 0).abs();
            return dist == 1;
        }
        return true;
    }

    let neighbors = grid.get_free_neighbors(current);
    if neighbors.is_empty() {
        return false;
    }

    let mut candidates = Vec::with_capacity(neighbors.len());
    for n in neighbors {
        let deg = grid.get_degree(n);
        candidates.push((n, deg));
    }
    
    // Sort candidates using Warnsdorff's rule.
    // TIE BREAKER:
    // When multiple candidates have the same degree, pick the one furthest from the center (on the boundaries).
    let cx = (grid.width - 1) as f64 / 2.0;
    let cy = (grid.height - 1) as f64 / 2.0;
    
    candidates.sort_by(|a, b| {
        let cmp = a.1.cmp(&b.1);
        if cmp == std::cmp::Ordering::Equal {
            let dist_a = ((a.0.x as f64 - cx).powi(2) + (a.0.y as f64 - cy).powi(2)).sqrt();
            let dist_b = ((b.0.x as f64 - cx).powi(2) + (b.0.y as f64 - cy).powi(2)).sqrt();
            dist_b.partial_cmp(&dist_a).unwrap_or(std::cmp::Ordering::Equal)
        } else {
            cmp
        }
    });

    for &(next, _) in &candidates {
        grid.visited[next.y as usize][next.x as usize] = true;
        path.push(next);

        let mut ok = true;
        let mut degree_0_count = 0;
        let mut degree_1_count = 0;
        
        let next_neighbors = grid.get_free_neighbors(next);
        for nn in &next_neighbors {
            let d = grid.get_degree(*nn);
            if d == 0 {
                degree_0_count += 1;
            } else if d == 1 {
                degree_1_count += 1;
            }
        }
        
        if remaining_unvisited_count > 1 && degree_0_count > 0 {
            ok = false;
        }

        if ok {
            if grid.require_cycle {
                let mut adj_0_0_free_count = 0;
                let dirs = [Point { x: 0, y: 1 }, Point { x: 0, y: -1 }, Point { x: 1, y: 0 }, Point { x: -1, y: 0 }];
                for d in dirs {
                    let adj = Point { x: 0 + d.x, y: 0 + d.y };
                    if grid.is_free(adj) {
                        adj_0_0_free_count += 1;
                    }
                }
                
                if remaining_unvisited_count > 1 && adj_0_0_free_count == 0 {
                    ok = false;
                }

                if ok && remaining_unvisited_count > 1 {
                    for nn in &next_neighbors {
                        if grid.get_degree(*nn) == 1 {
                            let is_adj_to_start = (nn.x - 0).abs() + (nn.y - 0).abs() == 1;
                            if !is_adj_to_start {
                                ok = false;
                                break;
                            }
                        }
                    }
                }
            } else {
                if remaining_unvisited_count > 1 && degree_1_count > 1 {
                    ok = false;
                }
            }
        }

        // To make the search scale for large grids (like 1000+ cells), we use a dynamic interval for connectivity checks.
        // For large grids, checking connectivity at every step or every 100 steps can still be slow due to O(V^2) overall.
        // We only check if degree_1_count > 0, or periodically based on grid size:
        if ok && remaining_unvisited_count > 1 {
            let check_interval = if grid.total_nodes > 1000 {
                500
            } else if grid.total_nodes > 500 {
                250
            } else {
                50
            };
            if degree_1_count > 0 || remaining_unvisited_count % check_interval == 0 {
                if let Some(&first_free) = next_neighbors.first() {
                    if !check_connectivity(grid, first_free, remaining_unvisited_count - 1) {
                        ok = false;
                    }
                }
            }
        }

        if ok {
            if solve(next, grid, path, start_time, timeout, step_count) {
                return true;
            }
        }

        path.pop();
        grid.visited[next.y as usize][next.x as usize] = false;
    }

    false
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = match lines.next() {
        Some(Ok(line)) => line,
        _ => return,
    };

    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() < 4 {
        println!("0");
        return;
    }

    let width: i32 = parts[0].parse().unwrap();
    let height: i32 = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<i32>().unwrap() == 1;

    let mut blocked = vec![vec![false; width as usize]; height as usize];
    for _ in 0..obstacle_count {
        if let Some(Ok(line)) = lines.next() {
            let coords: Vec<i32> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
            if coords.len() == 2 {
                blocked[coords[1] as usize][coords[0] as usize] = true;
            }
        }
    }

    let mut total_nodes = 0;
    for y in 0..height {
        for x in 0..width {
            if !blocked[y as usize][x as usize] {
                total_nodes += 1;
            }
        }
    }

    if total_nodes == 0 {
        println!("0");
        return;
    }

    if blocked[0][0] {
        println!("0");
        return;
    }

    let mut visited = vec![vec![false; width as usize]; height as usize];
    visited[0][0] = true;

    let mut path = Vec::with_capacity(total_nodes);
    path.push(Point { x: 0, y: 0 });

    let mut grid = Grid {
        width,
        height,
        blocked,
        visited,
        total_nodes,
        require_cycle,
    };

    let start_time = Instant::now();
    let timeout = Duration::from_secs(290);
    let mut step_count = 0;

    if total_nodes == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    if solve(Point { x: 0, y: 0 }, &mut grid, &mut path, &start_time, &timeout, &mut step_count) {
        println!("{}", path.len());
        for p in path {
            println!("{} {}", p.x, p.y);
        }
    } else {
        println!("0");
    }
}
