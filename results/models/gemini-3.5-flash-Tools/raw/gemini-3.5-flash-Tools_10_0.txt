#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>
#include <cstdint>

using namespace std;

struct Rectangle {
    int id;
    int w, h;
};

struct Placement {
    int board_idx;
    int x, y;
    int rotated; // 0 or 1
};

struct Segment {
    int x;
    int y;
    int width;
};

struct PlacementScore {
    int y_level;
    int top_y;
    int x;
    
    bool operator<(const PlacementScore& other) const {
        if (y_level != other.y_level) return y_level < other.y_level;
        if (top_y != other.top_y) return top_y < other.top_y;
        return x < other.x;
    }
};

// Pack rectangles using the Skyline heuristic for one board
int pack_board_skyline(
    int board_idx,
    int W, int H,
    const vector<Rectangle>& rects,
    const vector<int>& remaining_indices,
    vector<Placement>& placements,
    vector<bool>& placed,
    int min_remaining_dim
) {
    vector<Segment> skyline = { {0, 0, W} };
    int placed_count = 0;

    for (int idx : remaining_indices) {
        if (placed[idx]) continue;

        const auto& r = rects[idx];
        
        // Try orientations
        bool best_found = false;
        PlacementScore best_score = { 1000000000, 1000000000, 1000000000 };
        int best_rw = 0, best_rh = 0;

        // We can place it in 2 orientations: normal (w, h) and rotated (h, w)
        int orientations[2][2] = { {r.w, r.h}, {r.h, r.w} };
        int num_orientations = (r.w == r.h) ? 1 : 2;

        for (int o = 0; o < num_orientations; ++o) {
            int rw = orientations[o][0];
            int rh = orientations[o][1];

            if (rw > W || rh > H) continue;

            for (size_t i = 0; i < skyline.size(); ++i) {
                int x_start = skyline[i].x;
                if (x_start + rw > W) continue;

                // Find max y-level in the range [x_start, x_start + rw]
                int y_max = 0;
                for (size_t j = i; j < skyline.size(); ++j) {
                    if (skyline[j].x >= x_start + rw) break;
                    y_max = max(y_max, skyline[j].y);
                }

                if (y_max + rh <= H) {
                    PlacementScore score = { y_max, y_max + rh, x_start };
                    if (score < best_score) {
                        best_score = score;
                        best_rw = rw;
                        best_rh = rh;
                        best_found = true;
                    }
                }
            }
        }

        if (best_found) {
            int best_x = best_score.x;
            int best_y = best_score.y_level;

            // Record placement
            placements[r.id] = { board_idx, best_x, best_y, (best_rw == r.h && r.w != r.h) ? 1 : 0 };
            placed[idx] = true;
            placed_count++;

            // Update skyline
            int x1 = best_x;
            int x2 = best_x + best_rw;

            vector<Segment> next_skyline;
            next_skyline.reserve(skyline.size() + 2);

            // Add segments before x1
            for (const auto& seg : skyline) {
                if (seg.x + seg.width <= x1) {
                    next_skyline.push_back(seg);
                } else if (seg.x < x1) {
                    next_skyline.push_back({seg.x, seg.y, x1 - seg.x});
                }
            }

            // Add the new segment
            next_skyline.push_back({x1, best_y + best_rh, best_rw});

            // Add segments after x2
            for (const auto& seg : skyline) {
                if (seg.x >= x2) {
                    next_skyline.push_back(seg);
                } else if (seg.x + seg.width > x2) {
                    next_skyline.push_back({x2, seg.y, (seg.x + seg.width) - x2});
                }
            }

            // Merge adjacent segments with the same y
            skyline.clear();
            for (const auto& seg : next_skyline) {
                if (!skyline.empty() && skyline.back().y == seg.y) {
                    skyline.back().width += seg.width;
                } else {
                    skyline.push_back(seg);
                }
            }

            // Perform gap filling if applicable
            if (min_remaining_dim > 0) {
                bool raised = false;
                for (size_t i = 0; i < skyline.size(); ++i) {
                    if (skyline[i].width < min_remaining_dim) {
                        int left_y = (i > 0) ? skyline[i-1].y : 1000000000;
                        int right_y = (i + 1 < skyline.size()) ? skyline[i+1].y : 1000000000;
                        int next_y = min(left_y, right_y);
                        if (next_y != 1000000000) {
                            skyline[i].y = max(skyline[i].y, next_y);
                            raised = true;
                        }
                    }
                }
                if (raised) {
                    vector<Segment> merged;
                    for (const auto& seg : skyline) {
                        if (!merged.empty() && merged.back().y == seg.y) {
                            merged.back().width += seg.width;
                        } else {
                            merged.push_back(seg);
                        }
                    }
                    skyline = move(merged);
                }
            }
        }
    }

    return placed_count;
}

// Function to solve the bin packing with a given ordering of rectangles
vector<Placement> solve_with_order(
    int W, int H,
    const vector<Rectangle>& rects,
    const vector<int>& order,
    int& total_boards_used
) {
    int N = rects.size();
    vector<Placement> placements(N);
    vector<bool> placed(N, false);
    int remaining = N;

    int board_idx = 0;
    while (remaining > 0) {
        // Calculate min remaining dimension
        int min_remaining_dim = 1000000000;
        for (int idx : order) {
            if (!placed[idx]) {
                min_remaining_dim = min(min_remaining_dim, min(rects[idx].w, rects[idx].h));
            }
        }
        if (min_remaining_dim == 1000000000) min_remaining_dim = 0;

        int placed_on_board = pack_board_skyline(board_idx, W, H, rects, order, placements, placed, min_remaining_dim);
        if (placed_on_board == 0) {
            // Unpackable item fallback
            for (int idx : order) {
                if (!placed[idx]) {
                    placements[rects[idx].id] = { board_idx, 0, 0, 0 };
                    placed[idx] = true;
                    remaining--;
                    break;
                }
            }
            board_idx++;
        } else {
            remaining -= placed_on_board;
            board_idx++;
        }
    }

    total_boards_used = board_idx;
    return placements;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N, W, H;
    if (!(cin >> N >> W >> H)) return 0;

    vector<Rectangle> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        cin >> rects[i].w >> rects[i].h;
    }

    auto start_time = chrono::high_resolution_clock::now();

    // Generate different orderings
    vector<vector<int>> orderings;

    // Ordering 0: Sort by max(w, h) desc, then min(w, h) desc
    {
        vector<int> order(N);
        for (int i = 0; i < N; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            int max_a = max(rects[a].w, rects[a].h);
            int max_b = max(rects[b].w, rects[b].h);
            if (max_a != max_b) return max_a > max_b;
            int min_a = min(rects[a].w, rects[a].h);
            int min_b = min(rects[b].w, rects[b].h);
            return min_a > min_b;
        });
        orderings.push_back(move(order));
    }

    // Ordering 1: Sort by area desc
    {
        vector<int> order(N);
        for (int i = 0; i < N; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            long long area_a = (long long)rects[a].w * rects[a].h;
            long long area_b = (long long)rects[b].w * rects[b].h;
            if (area_a != area_b) return area_a > area_b;
            return max(rects[a].w, rects[a].h) > max(rects[b].w, rects[b].h);
        });
        orderings.push_back(move(order));
    }

    // Ordering 2: Sort by max(w, h) desc, then min(w, h) asc
    {
        vector<int> order(N);
        for (int i = 0; i < N; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            int max_a = max(rects[a].w, rects[a].h);
            int max_b = max(rects[b].w, rects[b].h);
            if (max_a != max_b) return max_a > max_b;
            int min_a = min(rects[a].w, rects[a].h);
            int min_b = min(rects[b].w, rects[b].h);
            return min_a < min_b;
        });
        orderings.push_back(move(order));
    }

    // Ordering 3: Sort by min(w, h) desc, then max(w, h) desc
    {
        vector<int> order(N);
        for (int i = 0; i < N; ++i) order[i] = i;
        sort(order.begin(), order.end(), [&](int a, int b) {
            int min_a = min(rects[a].w, rects[a].h);
            int min_b = min(rects[b].w, rects[b].h);
            if (min_a != min_b) return min_a > min_b;
            return max(rects[a].w, rects[a].h) > max(rects[b].w, rects[b].h);
        });
        orderings.push_back(move(order));
    }

    int best_boards = 1000000000;
    vector<Placement> best_placements;

    for (size_t k = 0; k < orderings.size(); ++k) {
        auto current_time = chrono::high_resolution_clock::now();
        double elapsed = chrono::duration<double>(current_time - start_time).count();
        if (elapsed > 20.0 && k > 0) {
            break;
        }

        int boards_used = 0;
        auto placements = solve_with_order(W, H, rects, orderings[k], boards_used);
        if (boards_used < best_boards) {
            best_boards = boards_used;
            best_placements = move(placements);
        }
    }

    // Output results
    cout << best_boards << "\n";
    for (int i = 0; i < N; ++i) {
        const auto& p = best_placements[i];
        cout << p.board_idx << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }

    return 0;
}
