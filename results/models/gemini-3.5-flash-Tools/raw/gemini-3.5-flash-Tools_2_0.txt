using System;
using System.Collections.Generic;
using System.IO;

class Program {
    struct Edge {
        public int Target;
        public int Weight;
        public int Id;
    }

    struct CandidateEdge : IComparable<CandidateEdge> {
        public int OddU;
        public int OddV;
        public int MeetX;
        public int MeetY;
        public long Weight;

        public int CompareTo(CandidateEdge other) {
            return Weight.CompareTo(other.Weight);
        }
    }

    class MinHeap {
        private struct HeapNode {
            public int Vertex;
            public long Dist;
        }
        private HeapNode[] nodes;
        private int count;

        public MinHeap(int capacity) {
            nodes = new HeapNode[capacity];
            count = 0;
        }

        public int Count => count;

        public void Push(int vertex, long dist) {
            if (count == nodes.Length) {
                Array.Resize(ref nodes, nodes.Length * 2);
            }
            nodes[count] = new HeapNode { Vertex = vertex, Dist = dist };
            SiftUp(count);
            count++;
        }

        public (int Vertex, long Dist) Pop() {
            var result = nodes[0];
            count--;
            if (count > 0) {
                nodes[0] = nodes[count];
                SiftDown(0);
            }
            return (result.Vertex, result.Dist);
        }

        private void SiftUp(int i) {
            while (i > 0) {
                int p = (i - 1) / 2;
                if (nodes[i].Dist >= nodes[p].Dist) break;
                var tmp = nodes[i];
                nodes[i] = nodes[p];
                nodes[p] = tmp;
                i = p;
            }
        }

        private void SiftDown(int i) {
            int left = 2 * i + 1;
            while (left < count) {
                int best = left;
                int right = left + 1;
                if (right < count && nodes[right].Dist < nodes[left].Dist) {
                    best = right;
                }
                if (nodes[i].Dist <= nodes[best].Dist) break;
                var tmp = nodes[i];
                nodes[i] = nodes[best];
                nodes[best] = tmp;
                i = best;
                left = 2 * i + 1;
            }
        }
    }

    class FastScanner {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[65536];
        private int bufferLength = 0;
        private int bufferPointer = 0;

        public FastScanner() {
            stream = Console.OpenStandardInput();
        }

        private byte Read() {
            if (bufferPointer >= bufferLength) {
                bufferLength = stream.Read(buffer, 0, buffer.Length);
                bufferPointer = 0;
                if (bufferLength <= 0) return 0;
            }
            return buffer[bufferPointer++];
        }

        public int NextInt() {
            int next = 0;
            byte b = Read();
            while (b <= 32) {
                if (b == 0) return -1;
                b = Read();
            }
            bool neg = false;
            if (b == '-') {
                neg = true;
                b = Read();
            }
            while (b >= '0' && b <= '9') {
                next = next * 10 + (b - '0');
                b = Read();
            }
            return neg ? -next : next;
        }
    }

    class FastWriter {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[65536];
        private int bufferPointer = 0;

        public FastWriter() {
            stream = Console.OpenStandardOutput();
        }

        public void WriteChar(char c) {
            if (bufferPointer >= buffer.Length) {
                Flush();
            }
            buffer[bufferPointer++] = (byte)c;
        }

        public void WriteInt(int val) {
            if (val == 0) {
                WriteChar('0');
                return;
            }
            if (val < 0) {
                WriteChar('-');
                val = -val;
            }
            int temp = val;
            int len = 0;
            byte[] digits = new byte[12];
            while (temp > 0) {
                digits[len++] = (byte)('0' + (temp % 10));
                temp /= 10;
            }
            for (int i = len - 1; i >= 0; i--) {
                if (bufferPointer >= buffer.Length) Flush();
                buffer[bufferPointer++] = digits[i];
            }
        }

        public void Flush() {
            if (bufferPointer > 0) {
                stream.Write(buffer, 0, bufferPointer);
                bufferPointer = 0;
            }
        }
    }

    static void Main() {
        var scanner = new FastScanner();
        int N = scanner.NextInt();
        if (N <= 0) return;
        int M = scanner.NextInt();
        if (M < 0) return;

        int[] uArr = new int[M];
        int[] vArr = new int[M];
        int[] wArr = new int[M];
        int[] deg = new int[N];
        for (int i = 0; i < M; i++) {
            uArr[i] = scanner.NextInt();
            vArr[i] = scanner.NextInt();
            wArr[i] = scanner.NextInt();
            deg[uArr[i]]++;
            deg[vArr[i]]++;
        }

        var adj = new List<Edge>[N];
        for (int i = 0; i < N; i++) {
            adj[i] = new List<Edge>(deg[i]);
        }

        List<int> edgeU = new List<int>(M + 1000);
        List<int> edgeV = new List<int>(M + 1000);
        List<int> edgeW = new List<int>(M + 1000);

        for (int i = 0; i < M; i++) {
            int u = uArr[i];
            int v = vArr[i];
            int w = wArr[i];
            edgeU.Add(u);
            edgeV.Add(v);
            edgeW.Add(w);
            adj[u].Add(new Edge { Target = v, Weight = w, Id = i });
            adj[v].Add(new Edge { Target = u, Weight = w, Id = i });
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++) {
            if (deg[i] % 2 != 0) {
                oddNodes.Add(i);
            }
        }

        bool[] matched = new bool[N];

        if (oddNodes.Count > 0) {
            long[] dist = new long[N];
            int[] owner = new int[N];
            int[] parent = new int[N];
            int[] parentEdgeId = new int[N];

            for (int i = 0; i < N; i++) {
                dist[i] = long.MaxValue;
                owner[i] = -1;
                parent[i] = -1;
                parentEdgeId[i] = -1;
            }

            var pq = new MinHeap(M);
            foreach (int u in oddNodes) {
                dist[u] = 0;
                owner[u] = u;
                pq.Push(u, 0);
            }

            var candidateEdges = new List<CandidateEdge>();

            while (pq.Count > 0) {
                var curr = pq.Pop();
                int x = curr.Vertex;
                long d = curr.Dist;
                if (d > dist[x]) continue;

                foreach (var edge in adj[x]) {
                    int y = edge.Target;
                    long w = edge.Weight;
                    if (dist[y] == long.MaxValue) {
                        dist[y] = d + w;
                        owner[y] = owner[x];
                        parent[y] = x;
                        parentEdgeId[y] = edge.Id;
                        pq.Push(y, dist[y]);
                    } else if (owner[y] != owner[x]) {
                        candidateEdges.Add(new CandidateEdge {
                            OddU = owner[x],
                            OddV = owner[y],
                            MeetX = x,
                            MeetY = y,
                            Weight = dist[x] + dist[y] + w
                        });
                    }
                }
            }

            candidateEdges.Sort();

            Action<int> duplicateEdge = (edgeId) => {
                int u = edgeU[edgeId];
                int v = edgeV[edgeId];
                int w = edgeW[edgeId];
                int dupId = edgeU.Count;
                edgeU.Add(u);
                edgeV.Add(v);
                edgeW.Add(w);
                adj[u].Add(new Edge { Target = v, Weight = w, Id = dupId });
                adj[v].Add(new Edge { Target = u, Weight = w, Id = dupId });
            };

            foreach (var edge in candidateEdges) {
                if (!matched[edge.OddU] && !matched[edge.OddV]) {
                    matched[edge.OddU] = true;
                    matched[edge.OddV] = true;

                    // Duplicate path
                    // Path from MeetX back to OddU
                    int curr = edge.MeetX;
                    while (curr != edge.OddU) {
                        duplicateEdge(parentEdgeId[curr]);
                        curr = parent[curr];
                    }
                    // Meeting edge
                    int meetEdgeId = -1;
                    foreach (var e in adj[edge.MeetX]) {
                        if (e.Target == edge.MeetY) {
                            meetEdgeId = e.Id;
                            break;
                        }
                    }
                    if (meetEdgeId != -1) {
                        duplicateEdge(meetEdgeId);
                    }
                    // Path from MeetY back to OddV
                    curr = edge.MeetY;
                    while (curr != edge.OddV) {
                        duplicateEdge(parentEdgeId[curr]);
                        curr = parent[curr];
                    }
                }
            }

            // Fallback for unmatched odd nodes
            List<int> unmatched = new List<int>();
            foreach (int u in oddNodes) {
                if (!matched[u]) unmatched.Add(u);
            }

            if (unmatched.Count > 0) {
                bool[] isUnmatched = new bool[N];
                foreach (int u in unmatched) isUnmatched[u] = true;

                long[] fallbackDist = new long[N];
                int[] fallbackParent = new int[N];
                int[] fallbackParentEdgeId = new int[N];
                bool[] fallbackVisited = new bool[N];
                List<int> visitedNodes = new List<int>();

                foreach (int u in unmatched) {
                    if (!isUnmatched[u]) continue;

                    var fpq = new MinHeap(N);
                    fallbackDist[u] = 0;
                    fallbackParent[u] = -1;
                    fallbackParentEdgeId[u] = -1;
                    fallbackVisited[u] = true;
                    fpq.Push(u, 0);
                    visitedNodes.Add(u);

                    int target = -1;
                    while (fpq.Count > 0) {
                        var curr = fpq.Pop();
                        int x = curr.Vertex;
                        long d = curr.Dist;
                        if (d > fallbackDist[x]) continue;

                        if (x != u && isUnmatched[x]) {
                            target = x;
                            break;
                        }

                        foreach (var edge in adj[x]) {
                            int y = edge.Target;
                            long w = edge.Weight;
                            if (!fallbackVisited[y] || d + w < fallbackDist[y]) {
                                if (!fallbackVisited[y]) {
                                    fallbackVisited[y] = true;
                                    visitedNodes.Add(y);
                                }
                                fallbackDist[y] = d + w;
                                fallbackParent[y] = x;
                                fallbackParentEdgeId[y] = edge.Id;
                                fpq.Push(y, fallbackDist[y]);
                            }
                        }
                    }

                    if (target != -1) {
                        isUnmatched[u] = false;
                        isUnmatched[target] = false;
                        matched[u] = true;
                        matched[target] = true;

                        int curr = target;
                        while (curr != u) {
                            duplicateEdge(fallbackParentEdgeId[curr]);
                            curr = fallbackParent[curr];
                        }
                    }

                    // Reset visited arrays
                    foreach (int node in visitedNodes) {
                        fallbackVisited[node] = false;
                        fallbackDist[node] = 0;
                        fallbackParent[node] = 0;
                        fallbackParentEdgeId[node] = 0;
                    }
                    visitedNodes.Clear();
                }
            }
        }

        // Run Hierholzer's algorithm
        int totalEdges = edgeU.Count;
        bool[] used = new bool[totalEdges];
        int[] head = new int[N];

        var route = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0) {
            int u = stack.Peek();
            bool found = false;
            while (head[u] < adj[u].Count) {
                var edge = adj[u][head[u]];
                if (!used[edge.Id]) {
                    used[edge.Id] = true;
                    stack.Push(edge.Target);
                    found = true;
                    head[u]++;
                    break;
                }
                head[u]++;
            }
            if (!found) {
                route.Add(u);
                stack.Pop();
            }
        }
        route.Reverse();

        var writer = new FastWriter();
        writer.WriteInt(route.Count);
        writer.WriteChar('\n');
        writer.WriteInt(route[0]);
        for (int i = 1; i < route.Count; i++) {
            writer.WriteChar(' ');
            writer.WriteInt(route[i]);
        }
        writer.WriteChar('\n');
        writer.Flush();
    }
}
