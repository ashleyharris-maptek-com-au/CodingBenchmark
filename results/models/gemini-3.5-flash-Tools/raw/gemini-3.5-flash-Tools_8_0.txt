use std::io::{self, BufRead, Write};

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
#[repr(u8)]
enum Dir {
    None = 0,
    Up = 1,
    Down = 2,
    Left = 3,
    Right = 4,
    StartOrEnd = 5,
}

impl Dir {
    fn from_u8(v: u8) -> Self {
        match v {
            1 => Dir::Up,
            2 => Dir::Down,
            3 => Dir::Left,
            4 => Dir::Right,
            5 => Dir::StartOrEnd,
            _ => Dir::None,
        }
    }
}

#[inline(always)]
fn is_wall(val: u8) -> bool {
    (val & 1) != 0
}

#[inline(always)]
fn get_fwd_dir(val: u8) -> Dir {
    Dir::from_u8((val >> 1) & 7)
}

#[inline(always)]
fn set_fwd_dir(val: &mut u8, dir: Dir) {
    *val = (*val & !(7 << 1)) | ((dir as u8) << 1);
}

#[inline(always)]
fn get_bwd_dir(val: u8) -> Dir {
    Dir::from_u8((val >> 4) & 7)
}

#[inline(always)]
fn set_bwd_dir(val: &mut u8, dir: Dir) {
    *val = (*val & !(7 << 4)) | ((dir as u8) << 4);
}

fn reconstruct_and_print(grid: &[u8], w: usize, _h: usize, meeting_point: (u32, u32)) {
    // Count forward path length
    let mut fwd_len = 1;
    let mut curr = meeting_point;
    loop {
        let idx = (curr.1 as usize) * w + (curr.0 as usize);
        let dir = get_fwd_dir(grid[idx]);
        if dir == Dir::StartOrEnd {
            break;
        }
        curr = match dir {
            Dir::Up => (curr.0, curr.1 - 1),
            Dir::Down => (curr.0, curr.1 + 1),
            Dir::Left => (curr.0 - 1, curr.1),
            Dir::Right => (curr.0 + 1, curr.1),
            _ => unreachable!(),
        };
        fwd_len += 1;
    }

    // Count backward path length
    let mut bwd_len = 0;
    curr = meeting_point;
    loop {
        let idx = (curr.1 as usize) * w + (curr.0 as usize);
        let dir = get_bwd_dir(grid[idx]);
        if dir == Dir::StartOrEnd {
            break;
        }
        curr = match dir {
            Dir::Up => (curr.0, curr.1 - 1),
            Dir::Down => (curr.0, curr.1 + 1),
            Dir::Left => (curr.0 - 1, curr.1),
            Dir::Right => (curr.0 + 1, curr.1),
            _ => unreachable!(),
        };
        bwd_len += 1;
    }

    let total_len = fwd_len + bwd_len;
    let mut path = vec![(0u32, 0u32); total_len];
    
    // Populate forward path
    curr = meeting_point;
    let mut idx_fwd = fwd_len;
    loop {
        idx_fwd -= 1;
        path[idx_fwd] = curr;
        let idx = (curr.1 as usize) * w + (curr.0 as usize);
        let dir = get_fwd_dir(grid[idx]);
        if dir == Dir::StartOrEnd {
            break;
        }
        curr = match dir {
            Dir::Up => (curr.0, curr.1 - 1),
            Dir::Down => (curr.0, curr.1 + 1),
            Dir::Left => (curr.0 - 1, curr.1),
            Dir::Right => (curr.0 + 1, curr.1),
            _ => unreachable!(),
        };
    }
    
    // Populate backward path
    curr = meeting_point;
    let mut idx_bwd = fwd_len;
    loop {
        let idx = (curr.1 as usize) * w + (curr.0 as usize);
        let dir = get_bwd_dir(grid[idx]);
        if dir == Dir::StartOrEnd {
            break;
        }
        curr = match dir {
            Dir::Up => (curr.0, curr.1 - 1),
            Dir::Down => (curr.0, curr.1 + 1),
            Dir::Left => (curr.0 - 1, curr.1),
            Dir::Right => (curr.0 + 1, curr.1),
            _ => unreachable!(),
        };
        path[idx_bwd] = curr;
        idx_bwd += 1;
    }

    let stdout = io::stdout();
    let mut writer = io::BufWriter::with_capacity(16 * 1024 * 1024, stdout.lock());
    
    if let Err(_) = writeln!(writer, "{}", total_len) {
        return;
    }
    for pos in &path {
        if let Err(_) = writeln!(writer, "{} {}", pos.0, pos.1) {
            return;
        }
    }
}

fn main() {
    let stdin = io::stdin();
    let locked = stdin.lock();
    let mut reader = io::BufReader::with_capacity(4 * 1024 * 1024, locked);
    
    let mut line = String::new();
    if reader.read_line(&mut line).is_err() {
        println!("0");
        return;
    }
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 2 {
        println!("0");
        return;
    }
    let h: usize = match parts[0].parse() {
        Ok(val) => val,
        Err(_) => {
            println!("0");
            return;
        }
    };
    let w: usize = match parts[1].parse() {
        Ok(val) => val,
        Err(_) => {
            println!("0");
            return;
        }
    };
    
    let mut grid = vec![0u8; h * w];
    let mut start_pos = None;
    let mut end_pos = None;
    
    let mut row_buf = Vec::new();
    for r in 0..h {
        row_buf.clear();
        if reader.read_until(b'\n', &mut row_buf).is_err() {
            println!("0");
            return;
        }
        while let Some(&last) = row_buf.last() {
            if last == b'\n' || last == b'\r' {
                row_buf.pop();
            } else {
                break;
            }
        }
        let offset = r * w;
        for c in 0..w {
            if c < row_buf.len() {
                let val = row_buf[c];
                if val == b'#' {
                    grid[offset + c] = 1;
                } else if val == b'A' {
                    start_pos = Some((c as u32, r as u32));
                } else if val == b'B' {
                    end_pos = Some((c as u32, r as u32));
                }
            } else {
                grid[offset + c] = 1;
            }
        }
    }
    
    let (sx, sy) = match start_pos {
        Some(pos) => pos,
        None => {
            println!("0");
            return;
        }
    };
    let (ex, ey) = match end_pos {
        Some(pos) => pos,
        None => {
            println!("0");
            return;
        }
    };
    
    if sx == ex && sy == ey {
        println!("1\n{} {}", sx, sy);
        return;
    }
    
    let s_idx = (sy as usize) * w + (sx as usize);
    set_fwd_dir(&mut grid[s_idx], Dir::StartOrEnd);
    
    let e_idx = (ey as usize) * w + (ex as usize);
    set_bwd_dir(&mut grid[e_idx], Dir::StartOrEnd);
    
    let mut fwd_front = Vec::new();
    let mut fwd_next = Vec::new();
    let mut bwd_front = Vec::new();
    let mut bwd_next = Vec::new();
    
    fwd_front.push((sx, sy));
    bwd_front.push((ex, ey));
    
    while !fwd_front.is_empty() && !bwd_front.is_empty() {
        if fwd_front.len() <= bwd_front.len() {
            fwd_next.clear();
            for &(x, y) in &fwd_front {
                for (dx, dy, dir_to_parent) in [
                    (0, -1, Dir::Down),
                    (0, 1, Dir::Up),
                    (-1, 0, Dir::Right),
                    (1, 0, Dir::Left),
                ] {
                    let nx = x as i32 + dx;
                    let ny = y as i32 + dy;
                    if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                        let nx = nx as u32;
                        let ny = ny as u32;
                        let n_idx = (ny as usize) * w + (nx as usize);
                        let val = grid[n_idx];
                        if is_wall(val) {
                            continue;
                        }
                        if get_fwd_dir(val) != Dir::None {
                            continue;
                        }
                        if get_bwd_dir(val) != Dir::None {
                            set_fwd_dir(&mut grid[n_idx], dir_to_parent);
                            reconstruct_and_print(&grid, w, h, (nx, ny));
                            return;
                        }
                        set_fwd_dir(&mut grid[n_idx], dir_to_parent);
                        fwd_next.push((nx, ny));
                    }
                }
            }
            std::mem::swap(&mut fwd_front, &mut fwd_next);
        } else {
            bwd_next.clear();
            for &(x, y) in &bwd_front {
                for (dx, dy, dir_to_parent) in [
                    (0, -1, Dir::Down),
                    (0, 1, Dir::Up),
                    (-1, 0, Dir::Right),
                    (1, 0, Dir::Left),
                ] {
                    let nx = x as i32 + dx;
                    let ny = y as i32 + dy;
                    if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                        let nx = nx as u32;
                        let ny = ny as u32;
                        let n_idx = (ny as usize) * w + (nx as usize);
                        let val = grid[n_idx];
                        if is_wall(val) {
                            continue;
                        }
                        if get_bwd_dir(val) != Dir::None {
                            continue;
                        }
                        if get_fwd_dir(val) != Dir::None {
                            set_bwd_dir(&mut grid[n_idx], dir_to_parent);
                            reconstruct_and_print(&grid, w, h, (nx, ny));
                            return;
                        }
                        set_bwd_dir(&mut grid[n_idx], dir_to_parent);
                        bwd_next.push((nx, ny));
                    }
                }
            }
            std::mem::swap(&mut bwd_front, &mut bwd_next);
        }
    }
    
    println!("0");
}
