using System;
using System.IO;
using System.Runtime.CompilerServices;

public struct CutItem : IComparable<CutItem>
{
    public int Length;
    public int OriginalIndex;

    public int CompareTo(CutItem other)
    {
        // Descending order of Length
        int comp = other.Length.CompareTo(this.Length);
        if (comp != 0) return comp;
        return this.OriginalIndex.CompareTo(other.OriginalIndex);
    }
}

public class Program
{
    // Segment tree representing the remaining capacities of the stocks.
    // Each leaf represents a stock piece (initially empty with stock_length capacity).
    // An internal node stores the maximum remaining capacity in its subtree.
    private int[] tree;
    private int leavesCount;
    private int stockLength;

    public static void Main(string[] args)
    {
        // Setup fast custom I/O reader
        FastReader reader = new FastReader(Console.OpenStandardInput());
        
        int n = reader.NextInt();
        if (n <= 0) return;
        
        int stockLen = reader.NextInt();
        
        CutItem[] cuts = new CutItem[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = new CutItem { Length = reader.NextInt(), OriginalIndex = i };
        }

        // Sort cuts in descending order of length
        Array.Sort(cuts);

        // We can have at most n stocks.
        // Let's implement FFD with a Segment Tree.
        Program p = new Program();
        p.stockLength = stockLen;
        
        // Find next power of 2 for segment tree leaves to make it simpler and faster.
        int power = 1;
        while (power < n)
        {
            power <<= 1;
        }
        p.leavesCount = power;
        
        // Segment tree size is 2 * leavesCount
        p.tree = new int[2 * p.leavesCount];
        // Initialize all stocks (leaves) with capacity stockLen
        for (int i = 0; i < p.leavesCount; i++)
        {
            p.tree[p.leavesCount + i] = stockLen;
        }
        // Build the segment tree (propagate maximum values up)
        for (int i = p.leavesCount - 1; i > 0; i--)
        {
            p.tree[i] = stockLen; // Since all leaves are stockLen, parents are also stockLen
        }

        // Head/Tail pointers for storing indices assigned to each stock.
        // We will store all outputs in a single flat array to avoid lists and overhead.
        int[] nextAssigned = new int[n];
        int[] headAssigned = new int[n]; // maps stockIndex -> first cut in linked list
        for (int i = 0; i < n; i++) headAssigned[i] = -1;
        
        // Track the count of active stocks and the maximum stock index used
        int maxStockUsed = 0;

        for (int i = 0; i < n; i++)
        {
            int cutLen = cuts[i].Length;
            
            // Query the segment tree to find the first stock index that can hold cutLen
            int stockIdx = p.FindFirstFitStock(1, 0, p.leavesCount - 1, cutLen);
            
            if (stockIdx > maxStockUsed)
            {
                maxStockUsed = stockIdx;
            }

            // Update the segment tree: reduce capacity of the chosen stock
            p.UpdateStockCapacity(stockIdx, cutLen);

            // Add the cut's original index to this stock's linked list (stored in flat arrays)
            nextAssigned[i] = headAssigned[stockIdx];
            headAssigned[stockIdx] = i;
        }

        // Total number of stocks used is maxStockUsed + 1
        int numStocks = maxStockUsed + 1;

        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.UTF8, 65536))
        {
            writer.WriteLine(numStocks);
            for (int s = 0; s < numStocks; s++)
            {
                // We traversed the cuts in descending order, and prepended them.
                // We should output the cuts assigned to this stock.
                // The order of output within a stock line doesn't matter as long as indices are correct.
                int curr = headAssigned[s];
                bool first = true;
                while (curr != -1)
                {
                    if (!first) writer.Write(' ');
                    first = false;
                    writer.Write(cuts[curr].OriginalIndex);
                    curr = nextAssigned[curr];
                }
                writer.WriteLine();
            }
        }
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private int FindFirstFitStock(int node, int start, int end, int val)
    {
        while (start != end)
        {
            int mid = (start + end) >> 1;
            int leftChild = node << 1;
            int rightChild = leftChild | 1;

            if (tree[leftChild] >= val)
            {
                node = leftChild;
                end = mid;
            }
            else
            {
                node = rightChild;
                start = mid + 1;
            }
        }
        return start;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private void UpdateStockCapacity(int stockIdx, int val)
    {
        int node = leavesCount + stockIdx;
        tree[node] -= val;
        while (node > 1)
        {
            node >>= 1;
            int leftVal = tree[node << 1];
            int rightVal = tree[(node << 1) | 1];
            tree[node] = leftVal > rightVal ? leftVal : rightVal;
        }
    }
}

public class FastReader
{
    private readonly Stream _stream;
    private readonly byte[] _buffer = new byte[1024 * 128];
    private int _ptr = 0;
    private int _len = 0;

    public FastReader(Stream stream)
    {
        _stream = stream;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private byte Read()
    {
        if (_ptr >= _len)
        {
            _len = _stream.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0) return 0; // EOF
        }
        return _buffer[_ptr++];
    }

    public int NextInt()
    {
        int val = 0;
        byte b = Read();
        while (b > 0 && b <= 32) // Skip all whitespace control chars
        {
            b = Read();
        }
        if (b == 0) return -1; // EOF
        while (b >= 48 && b <= 57) // '0' is 48, '9' is 57
        {
            val = val * 10 + (b - 48);
            b = Read();
        }
        return val;
    }
}
