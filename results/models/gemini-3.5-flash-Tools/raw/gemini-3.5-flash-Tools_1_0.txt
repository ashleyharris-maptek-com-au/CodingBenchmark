#include <vector>
#include <algorithm>
#include <cstdint>
#include <cmath>
#include <chrono>
#include <random>
#include <cstdio>

// Custom fast buffered I/O
const size_t BUFFER_SIZE = 1 << 20; // 1MB buffer
char io_buffer[BUFFER_SIZE];
size_t io_head = 0;
size_t io_tail = 0;

inline char nextChar() {
    if (io_head >= io_tail) {
        io_head = 0;
        io_tail = fread(io_buffer, 1, BUFFER_SIZE, stdin);
        if (io_tail == 0) return EOF;
    }
    return io_buffer[io_head++];
}

inline double readDouble() {
    double val = 0;
    char c = nextChar();
    while (c <= 32 && c != EOF) {
        c = nextChar();
    }
    if (c == EOF) return 0;
    bool neg = false;
    if (c == '-') {
        neg = true;
        c = nextChar();
    }
    while (c >= '0' && c <= '9') {
        val = val * 10.0 + (c - '0');
        c = nextChar();
    }
    if (c == '.') {
        double div = 1.0;
        c = nextChar();
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + (c - '0');
            div *= 10.0;
            c = nextChar();
        }
        val /= div;
    }
    return neg ? -val : val;
}

inline int readInt() {
    int val = 0;
    char c = nextChar();
    while (c <= 32 && c != EOF) {
        c = nextChar();
    }
    if (c == EOF) return 0;
    bool neg = false;
    if (c == '-') {
        neg = true;
        c = nextChar();
    }
    while (c >= '0' && c <= '9') {
        val = val * 10 + (c - '0');
        c = nextChar();
    }
    return neg ? -val : val;
}

char out_buffer[BUFFER_SIZE];
size_t out_head = 0;

inline void flushOut() {
    if (out_head > 0) {
        fwrite(out_buffer, 1, out_head, stdout);
        out_head = 0;
    }
}

inline void writeChar(char c) {
    if (out_head >= BUFFER_SIZE) {
        flushOut();
    }
    out_buffer[out_head++] = c;
}

inline void writeInt(int n) {
    if (n == 0) {
        writeChar('0');
        return;
    }
    char buf[12];
    int len = 0;
    while (n > 0) {
        buf[len++] = (n % 10) + '0';
        n /= 10;
    }
    for (int i = len - 1; i >= 0; --i) {
        writeChar(buf[i]);
    }
}

struct City {
    float x, y;
    int id;
    uint64_t h_val;
};

// Convert (x, y) coordinates to 2D Hilbert curve index.
uint64_t hilbert_xy2d(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 30; s >= 0; s--) {
        int rx = (x >> s) & 1;
        int ry = (y >> s) & 1;
        d = (d << 2) | ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            std::swap(x, y);
        }
    }
    return d;
}

inline float dist(const City& a, const City& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return std::sqrt(dx * dx + dy * dy);
}

// Compute total tour distance
double get_tour_distance(const std::vector<City>& tour) {
    double total = 0;
    int n = tour.size();
    for (int i = 0; i < n; ++i) {
        total += dist(tour[i], tour[(i + 1) % n]);
    }
    return total;
}

// Full 2-opt local search
void run_full_2opt(std::vector<City>& tour) {
    int n = tour.size();
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 2; j < n; ++j) {
                int next_i = i + 1;
                int next_j = (j + 1 == n) ? 0 : j + 1;
                float change = dist(tour[i], tour[j]) + dist(tour[next_i], tour[next_j])
                               - dist(tour[i], tour[next_i]) - dist(tour[j], tour[next_j]);
                if (change < -1e-5f) {
                    std::reverse(tour.begin() + next_i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Perturbation for Iterated Local Search (Double Bridge move)
void double_bridge(std::vector<City>& tour, std::mt19937& rng) {
    int n = tour.size();
    if (n < 8) return;
    std::uniform_int_distribution<int> dist_pos(1, n / 4);
    int pos1 = dist_pos(rng);
    int pos2 = pos1 + dist_pos(rng);
    int pos3 = pos2 + dist_pos(rng);
    
    std::vector<City> new_tour;
    new_tour.reserve(n);
    new_tour.insert(new_tour.end(), tour.begin(), tour.begin() + pos1);
    new_tour.insert(new_tour.end(), tour.begin() + pos3, tour.end());
    new_tour.insert(new_tour.end(), tour.begin() + pos2, tour.begin() + pos3);
    new_tour.insert(new_tour.end(), tour.begin() + pos1, tour.begin() + pos2);
    tour = std::move(new_tour);
}

int main() {
    int N = readInt();
    if (N <= 0) return 0;
    
    std::vector<City> cities(N);
    float min_x = 1e18f, max_x = -1e18f, min_y = 1e18f, max_y = -1e18f;
    for (int i = 0; i < N; ++i) {
        cities[i].x = (float)readDouble();
        cities[i].y = (float)readDouble();
        cities[i].id = i;
        if (cities[i].x < min_x) min_x = cities[i].x;
        if (cities[i].x > max_x) max_x = cities[i].x;
        if (cities[i].y < min_y) min_y = cities[i].y;
        if (cities[i].y > max_y) max_y = cities[i].y;
    }
    
    std::vector<City> best_tour;
    
    if (N <= 1000) {
        // Small/Medium: Multi-start Nearest Neighbor + Iterated Local Search (ILS)
        std::mt19937 rng(42);
        double best_dist = 1e18;
        
        int starts = std::min(N, 100);
        for (int s = 0; s < starts; ++s) {
            std::vector<City> current_tour;
            current_tour.reserve(N);
            std::vector<bool> visited(N, false);
            
            // Start from city s
            int curr = s;
            current_tour.push_back(cities[curr]);
            visited[curr] = true;
            
            for (int step = 1; step < N; ++step) {
                int next_city = -1;
                float min_d = 1e18f;
                for (int next = 0; next < N; ++next) {
                    if (!visited[next]) {
                        float d = dist(cities[curr], cities[next]);
                        if (d < min_d) {
                            min_d = d;
                            next_city = next;
                        }
                    }
                }
                curr = next_city;
                current_tour.push_back(cities[curr]);
                visited[curr] = true;
            }
            
            // Run 2-opt
            run_full_2opt(current_tour);
            
            // Run standard ILS for up to 30 iterations to escape local minima
            std::vector<City> ils_tour = current_tour;
            double ils_dist = get_tour_distance(ils_tour);
            
            for (int iter = 0; iter < 30; ++iter) {
                std::vector<City> perturbed = ils_tour;
                double_bridge(perturbed, rng);
                run_full_2opt(perturbed);
                double d = get_tour_distance(perturbed);
                if (d < ils_dist) {
                    ils_dist = d;
                    ils_tour = std::move(perturbed);
                }
            }
            
            if (ils_dist < best_dist) {
                best_dist = ils_dist;
                best_tour = std::move(ils_tour);
            }
        }
    } else {
        // Large/Extreme/Epic: Normalize coordinates to Hilbert Curve space-filling curve
        float dx = max_x - min_x;
        float dy = max_y - min_y;
        float max_d = std::max(dx, dy);
        if (max_d < 1e-9f) max_d = 1e-9f;
        double scale = (double)((1ULL << 30) - 1) / max_d;
        
        for (int i = 0; i < N; ++i) {
            uint32_t ix = (uint32_t)((cities[i].x - min_x) * scale);
            uint32_t iy = (uint32_t)((cities[i].y - min_y) * scale);
            cities[i].h_val = hilbert_xy2d(ix, iy);
        }
        
        // Initial tour from Hilbert Sort
        std::sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
            return a.h_val < b.h_val;
        });
        
        best_tour = std::move(cities);
        
        // Windowed 2-opt local search adaptive settings based on N
        int W = 15;
        int passes = 2;
        if (N <= 50000) {
            W = 200;
            passes = 8;
        } else if (N <= 500000) {
            W = 50;
            passes = 4;
        } else if (N <= 5000000) {
            W = 20;
            passes = 2;
        } else {
            W = 10;
            passes = 1;
        }
        
        for (int p = 0; p < passes; ++p) {
            bool improved = false;
            for (int i = 0; i < N - 1; ++i) {
                int limit = std::min(N, i + W);
                for (int j = i + 2; j < limit; ++j) {
                    int next_i = i + 1;
                    int next_j = (j + 1 == N) ? 0 : j + 1;
                    float change = dist(best_tour[i], best_tour[j]) + dist(best_tour[next_i], best_tour[next_j])
                                   - dist(best_tour[i], best_tour[next_i]) - dist(best_tour[j], best_tour[next_j]);
                    if (change < -1e-4f) {
                        std::reverse(best_tour.begin() + next_i, best_tour.begin() + j + 1);
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }
    }
    
    // Output must start with city index 0
    int start_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (best_tour[i].id == 0) {
            start_idx = i;
            break;
        }
    }
    
    for (int i = 0; i < N; ++i) {
        int idx = (start_idx + i) % N;
        writeInt(best_tour[idx].id);
        writeChar(i == N - 1 ? '\n' : ' ');
    }
    
    flushOut();
    return 0;
}
