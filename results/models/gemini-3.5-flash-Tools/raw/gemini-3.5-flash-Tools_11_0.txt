use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Debug, Clone, Copy)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn rotate(&self, theta: f64) -> Point {
        let cos_t = theta.cos();
        let sin_t = theta.sin();
        Point {
            x: self.x * cos_t - self.y * sin_t,
            y: self.x * sin_t + self.y * cos_t,
        }
    }
    fn translate(&self, tx: f64, ty: f64) -> Point {
        Point {
            x: self.x + tx,
            y: self.y + ty,
        }
    }
}

#[derive(Debug, Clone)]
struct Polygon {
    vertices: Vec<Point>,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for p in &vertices {
            if p.x < min_x { min_x = p.x; }
            if p.x > max_x { max_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.y > max_y { max_y = p.y; }
        }
        Polygon {
            vertices,
            min_x,
            max_x,
            min_y,
            max_y,
        }
    }

    fn transform(&self, tx: f64, ty: f64, theta: f64) -> Polygon {
        let new_verts = self.vertices.iter()
            .map(|p| p.rotate(theta).translate(tx, ty))
            .collect();
        Polygon::new(new_verts)
    }

    fn area(&self) -> f64 {
        let n = self.vertices.len();
        if n < 3 {
            return 0.0;
        }
        let mut area = 0.0;
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() * 0.5
    }
}

fn ccw(a: Point, b: Point, c: Point) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn intersect_proper(a1: Point, a2: Point, b1: Point, b2: Point) -> bool {
    let eps = 1e-9;
    let ccw1 = ccw(a1, a2, b1);
    let ccw2 = ccw(a1, a2, b2);
    let ccw3 = ccw(b1, b2, a1);
    let ccw4 = ccw(b1, b2, a2);
    
    (ccw1 * ccw2 < -eps) && (ccw3 * ccw4 < -eps)
}

fn point_in_polygon(p: Point, poly: &Polygon, include_boundary: bool) -> bool {
    let mut inside = false;
    let n = poly.vertices.len();
    let mut on_boundary = false;
    
    for i in 0..n {
        let v1 = poly.vertices[i];
        let v2 = poly.vertices[(i + 1) % n];
        
        let min_x = f64::min(v1.x, v2.x);
        let max_x = f64::max(v1.x, v2.x);
        let min_y = f64::min(v1.y, v2.y);
        let max_y = f64::max(v1.y, v2.y);
        
        if p.x >= min_x - 1e-9 && p.x <= max_x + 1e-9 && p.y >= min_y - 1e-9 && p.y <= max_y + 1e-9 {
            if ccw(v1, v2, p).abs() < 1e-9 {
                on_boundary = true;
                break;
            }
        }
        
        if (v1.y > p.y) != (v2.y > p.y) {
            let x_int = v1.x + (p.y - v1.y) * (v2.x - v1.x) / (v2.y - v1.y);
            if p.x < x_int {
                inside = !inside;
            }
        }
    }
    
    if on_boundary {
        include_boundary
    } else {
        inside
    }
}

fn polygons_overlap(a: &Polygon, b: &Polygon) -> bool {
    if a.max_x < b.min_x - 1e-9 || a.min_x > b.max_x + 1e-9 ||
       a.max_y < b.min_y - 1e-9 || a.min_y > b.max_y + 1e-9 {
        return false;
    }
    
    let na = a.vertices.len();
    let nb = b.vertices.len();
    for i in 0..na {
        let a1 = a.vertices[i];
        let a2 = a.vertices[(i + 1) % na];
        for j in 0..nb {
            let b1 = b.vertices[j];
            let b2 = b.vertices[(j + 1) % nb];
            if intersect_proper(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    
    for &va in &a.vertices {
        if point_in_polygon(va, b, false) {
            return true;
        }
    }
    for &vb in &b.vertices {
        if point_in_polygon(vb, a, false) {
            return true;
        }
    }
    
    for i in 0..na {
        let a1 = a.vertices[i];
        let a2 = a.vertices[(i + 1) % na];
        let mid = Point { x: 0.5 * (a1.x + a2.x), y: 0.5 * (a1.y + a2.y) };
        if point_in_polygon(mid, b, false) {
            return true;
        }
    }
    for j in 0..nb {
        let b1 = b.vertices[j];
        let b2 = b.vertices[(j + 1) % nb];
        let mid = Point { x: 0.5 * (b1.x + b2.x), y: 0.5 * (b1.y + b2.y) };
        if point_in_polygon(mid, a, false) {
            return true;
        }
    }
    
    false
}

fn polygon_inside_stock(p: &Polygon, s: &Polygon) -> bool {
    if p.min_x < s.min_x - 1e-9 || p.max_x > s.max_x + 1e-9 ||
       p.min_y < s.min_y - 1e-9 || p.max_y > s.max_y + 1e-9 {
        return false;
    }
    
    for &vp in &p.vertices {
        if !point_in_polygon(vp, s, true) {
            return false;
        }
    }
    
    let np = p.vertices.len();
    for i in 0..np {
        let p1 = p.vertices[i];
        let p2 = p.vertices[(i + 1) % np];
        let mid = Point { x: 0.5 * (p1.x + p2.x), y: 0.5 * (p1.y + p2.y) };
        if !point_in_polygon(mid, s, true) {
            return false;
        }
    }
    
    let ns = s.vertices.len();
    for i in 0..np {
        let p1 = p.vertices[i];
        let p2 = p.vertices[(i + 1) % np];
        for j in 0..ns {
            let s1 = s.vertices[j];
            let s2 = s.vertices[(j + 1) % ns];
            if intersect_proper(p1, p2, s1, s2) {
                return false;
            }
        }
    }
    
    for &vs in &s.vertices {
        if point_in_polygon(vs, p, false) {
            return false;
        }
    }
    
    true
}

#[derive(Clone)]
struct SpatialGrid {
    min_x: f64,
    min_y: f64,
    cell_w: f64,
    cell_h: f64,
    cols: usize,
    rows: usize,
    cells: Vec<Vec<usize>>,
}

impl SpatialGrid {
    fn new(min_x: f64, min_y: f64, max_x: f64, max_y: f64, pieces_count: usize) -> Self {
        let grid_size = if pieces_count > 1000 {
            64
        } else if pieces_count > 100 {
            32
        } else {
            16
        };
        let cell_w = (max_x - min_x) / grid_size as f64;
        let cell_h = (max_y - min_y) / grid_size as f64;
        SpatialGrid {
            min_x,
            min_y,
            cell_w,
            cell_h,
            cols: grid_size,
            rows: grid_size,
            cells: vec![Vec::new(); grid_size * grid_size],
        }
    }

    fn get_cell_range(&self, min_x: f64, max_x: f64, min_y: f64, max_y: f64) -> (usize, usize, usize, usize) {
        let c1 = ((min_x - self.min_x) / self.cell_w).floor() as isize;
        let c2 = ((max_x - self.min_x) / self.cell_w).floor() as isize;
        let r1 = ((min_y - self.min_y) / self.cell_h).floor() as isize;
        let r2 = ((max_y - self.min_y) / self.cell_h).floor() as isize;
        
        let c_start = c1.clamp(0, self.cols as isize - 1) as usize;
        let c_end = c2.clamp(0, self.cols as isize - 1) as usize;
        let r_start = r1.clamp(0, self.rows as isize - 1) as usize;
        let r_end = r2.clamp(0, self.rows as isize - 1) as usize;
        
        (c_start, c_end, r_start, r_end)
    }

    fn insert(&mut self, idx: usize, poly: &Polygon) {
        let (c_start, c_end, r_start, r_end) = self.get_cell_range(poly.min_x, poly.max_x, poly.min_y, poly.max_y);
        for r in r_start..=r_end {
            for c in c_start..=c_end {
                self.cells[r * self.cols + c].push(idx);
            }
        }
    }

    fn query_overlap(&self, poly: &Polygon, placed_polys: &[Polygon], checked: &mut [bool]) -> bool {
        let (c_start, c_end, r_start, r_end) = self.get_cell_range(poly.min_x, poly.max_x, poly.min_y, poly.max_y);
        
        let mut candidates = Vec::with_capacity(32);
        for r in r_start..=r_end {
            for c in c_start..=c_end {
                for &idx in &self.cells[r * self.cols + c] {
                    if !checked[idx] {
                        checked[idx] = true;
                        candidates.push(idx);
                    }
                }
            }
        }
        
        let mut overlap = false;
        for &idx in &candidates {
            if polygons_overlap(poly, &placed_polys[idx]) {
                overlap = true;
                break;
            }
        }
        
        for &idx in &candidates {
            checked[idx] = false;
        }
        
        overlap
    }
}

fn generate_candidates(
    rotated_piece: &Polygon,
    stock: &Polygon,
    placed_polys: &[Polygon],
    max_targets: usize,
) -> Vec<Point> {
    let mut candidates = Vec::new();
    let mut targets = Vec::new();
    
    for &v in &stock.vertices {
        targets.push(v);
    }
    for poly in placed_polys {
        for &v in &poly.vertices {
            targets.push(v);
        }
    }
    
    // Sort targets to prioritize bottom-left area of the stock
    targets.sort_by(|a, b| (a.x + a.y).partial_cmp(&(b.x + b.y)).unwrap());
    if targets.len() > max_targets {
        targets.truncate(max_targets);
    }
    
    let mut sources = Vec::new();
    let p_min_x = rotated_piece.vertices.iter().min_by(|a, b| a.x.partial_cmp(&b.x).unwrap()).cloned().unwrap();
    let p_max_x = rotated_piece.vertices.iter().max_by(|a, b| a.x.partial_cmp(&b.x).unwrap()).cloned().unwrap();
    let p_min_y = rotated_piece.vertices.iter().min_by(|a, b| a.y.partial_cmp(&b.y).unwrap()).cloned().unwrap();
    let p_max_y = rotated_piece.vertices.iter().max_by(|a, b| a.y.partial_cmp(&b.y).unwrap()).cloned().unwrap();
    
    sources.push(p_min_x);
    sources.push(p_max_x);
    sources.push(p_min_y);
    sources.push(p_max_y);
    
    sources.sort_by(|a, b| {
        if (a.x - b.x).abs() > 1e-9 {
            a.x.partial_cmp(&b.x).unwrap()
        } else {
            a.y.partial_cmp(&b.y).unwrap()
        }
    });
    sources.dedup_by(|a, b| (a.x - b.x).abs() < 1e-9 && (a.y - b.y).abs() < 1e-9);
    
    for &t in &targets {
        for &s in &sources {
            let tx = t.x - s.x;
            let ty = t.y - s.y;
            candidates.push(Point { x: tx, y: ty });
        }
    }
    
    let tx1 = stock.min_x - p_min_x.x;
    let ty1 = stock.min_y - p_min_y.y;
    candidates.push(Point { x: tx1, y: ty1 });
    
    candidates.sort_by(|a, b| {
        if (a.x - b.x).abs() > 1e-9 {
            a.x.partial_cmp(&b.x).unwrap()
        } else {
            a.y.partial_cmp(&b.y).unwrap()
        }
    });
    candidates.dedup_by(|a, b| (a.x - b.x).abs() < 1e-9 && (a.y - b.y).abs() < 1e-9);
    
    candidates
}

#[derive(Debug, Clone)]
struct PieceInfo {
    original_idx: usize,
    poly: Polygon,
}

#[derive(Debug, Clone)]
struct Placement {
    stock_idx: usize,
    tx: f64,
    ty: f64,
    rotation: f64,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return Ok(()),
    };
    let mut parts = first_line.split_whitespace();
    let sv: usize = parts.next().unwrap().parse().unwrap();
    let np: usize = parts.next().unwrap().parse().unwrap();
    
    let mut stock_verts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = lines.next().unwrap()?;
        let mut p = line.split_whitespace();
        let x: f64 = p.next().unwrap().parse().unwrap();
        let y: f64 = p.next().unwrap().parse().unwrap();
        stock_verts.push(Point { x, y });
    }
    let stock = Polygon::new(stock_verts);
    
    let mut pieces = Vec::with_capacity(np);
    for i in 0..np {
        let line = lines.next().unwrap()?;
        let pv: usize = line.trim().parse().unwrap();
        let mut verts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = lines.next().unwrap()?;
            let mut p = line.split_whitespace();
            let x: f64 = p.next().unwrap().parse().unwrap();
            let y: f64 = p.next().unwrap().parse().unwrap();
            verts.push(Point { x, y });
        }
        pieces.push(PieceInfo {
            original_idx: i,
            poly: Polygon::new(verts),
        });
    }
    
    // Sort pieces by area descending
    pieces.sort_by(|a, b| b.poly.area().partial_cmp(&a.poly.area()).unwrap());
    
    let mut placements = vec![None; np];
    let mut stock_index = 0;
    
    let mut remaining = pieces;
    
    let num_threads = std::thread::available_parallelism().map(|n| n.get()).unwrap_or(64);
    
    while !remaining.is_empty() {
        let mut placed_polys: Vec<Polygon> = Vec::new();
        let mut grid = SpatialGrid::new(stock.min_x, stock.min_y, stock.max_x, stock.max_y, np);
        let mut next_remaining = Vec::new();
        
        let max_targets = if remaining.len() > 1000 {
            10
        } else if remaining.len() > 100 {
            30
        } else {
            60
        };
        
        let rotations = if remaining.len() > 1000 {
            vec![0.0, PI / 2.0]
        } else if remaining.len() > 100 {
            vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0]
        } else {
            vec![
                0.0, PI / 4.0, PI / 2.0, 3.0 * PI / 4.0,
                PI, 5.0 * PI / 4.0, 3.0 * PI / 2.0, 7.0 * PI / 4.0,
            ]
        };
        
        for piece_info in remaining {
            let piece = &piece_info.poly;
            
            let mut candidates = Vec::new();
            for &theta in &rotations {
                let rot_p = piece.transform(0.0, 0.0, theta);
                let trans_pts = generate_candidates(&rot_p, &stock, &placed_polys, max_targets);
                for pt in trans_pts {
                    candidates.push((theta, pt));
                }
            }
            
            let chunk_size = (candidates.len() + num_threads - 1) / num_threads;
            let mut best_placement = None;
            let mut best_score = f64::INFINITY;
            
            if !candidates.is_empty() {
                let placed_polys_ref = &placed_polys;
                let grid_ref = &grid;
                let stock_ref = &stock;
                let piece_ref = piece;
                
                let results: Vec<Option<(f64, Point, f64)>> = std::thread::scope(|s| {
                    let mut handles = Vec::new();
                    for chunk in candidates.chunks(chunk_size) {
                        let handle = s.spawn(move || {
                            let mut local_best = None;
                            let mut local_best_score = f64::INFINITY;
                            let mut checked = vec![false; placed_polys_ref.len() + 1];
                            
                            for &(theta, pt) in chunk {
                                let candidate_poly = piece_ref.transform(pt.x, pt.y, theta);
                                
                                if !polygon_inside_stock(&candidate_poly, stock_ref) {
                                    continue;
                                }
                                
                                if grid_ref.query_overlap(&candidate_poly, placed_polys_ref, &mut checked) {
                                    continue;
                                }
                                
                                let score = (candidate_poly.min_x - stock_ref.min_x) + (candidate_poly.min_y - stock_ref.min_y);
                                if score < local_best_score {
                                    local_best_score = score;
                                    local_best = Some((score, pt, theta));
                                }
                            }
                            local_best
                        });
                        handles.push(handle);
                    }
                    handles.into_iter().map(|h| h.join().unwrap()).collect()
                });
                
                for res in results {
                    if let Some((score, pt, theta)) = res {
                        if score < best_score {
                            best_score = score;
                            best_placement = Some((pt, theta));
                        }
                    }
                }
            }
            
            if let Some((pt, theta)) = best_placement {
                let final_poly = piece.transform(pt.x, pt.y, theta);
                let idx = placed_polys.len();
                placed_polys.push(final_poly.clone());
                grid.insert(idx, &final_poly);
                
                placements[piece_info.original_idx] = Some(Placement {
                    stock_idx: stock_index,
                    tx: pt.x,
                    ty: pt.y,
                    rotation: theta,
                });
            } else {
                next_remaining.push(piece_info);
            }
        }
        
        remaining = next_remaining;
        stock_index += 1;
    }
    
    println!("{}", stock_index);
    for opt in placements {
        if let Some(p) = opt {
            println!("{} {} {} {}", p.stock_idx, p.tx, p.ty, p.rotation);
        }
    }
    
    Ok(())
}
