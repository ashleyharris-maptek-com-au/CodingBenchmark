#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <chrono>
#include <random>

using namespace std;

struct hash_t {
    uint64_t h1;
    uint64_t h2;
    bool operator==(const hash_t& o) const {
        return h1 == o.h1 && h2 == o.h2;
    }
};

uint64_t P1, P2;

uint64_t power(uint64_t base, int exp) {
    uint64_t res = 1;
    while(exp > 0) {
        if(exp % 2 == 1) res *= base;
        base *= base;
        exp /= 2;
    }
    return res;
}

vector<hash_t> current_V;

struct HashTable {
    vector<hash_t> keys;
    vector<uint32_t> status;
    uint32_t mask;
    
    vector<uint64_t> bloom;
    uint32_t bloom_mask;
    
    vector<uint32_t> seen_indices;
    
    void build(const vector<hash_t>& V) {
        uint32_t size = 16;
        while(size < V.size() * 3) size *= 2;
        mask = size - 1;
        
        keys.assign(size, 0);
        status.assign(size, 0);
        
        uint64_t target_bits = V.size() * 10;
        uint32_t b_size = 16; 
        while (b_size * 64ULL < target_bits) {
            b_size *= 2;
            if (b_size >= (1 << 21)) { // Cap at 16MB
                b_size = 1 << 21;
                break;
            }
        }
        
        bloom_mask = (b_size * 64) - 1;
        bloom.assign(b_size, 0);
        
        seen_indices.clear();
        seen_indices.reserve(V.size());
        
        for(auto h : V) {
            uint32_t idx = (uint32_t)((h ^ (h >> 32)) & mask);
            while(status[idx] != 0) {
                if (keys[idx] == h) break;
                idx = (idx + 1) & mask;
            }
            if (status[idx] == 0) {
                keys[idx] = h;
                status[idx] = 1;
            }
            
            uint32_t b1 = (uint32_t)(h & bloom_mask);
            uint32_t b2 = (uint32_t)((h >> 32) & bloom_mask);
            bloom[b1 >> 6] |= (1ULL << (b1 & 63));
            bloom[b2 >> 6] |= (1ULL << (b2 & 63));
        }
    }
    
    inline bool check_and_mark(hash_t h) {
        uint32_t b1 = (uint32_t)(h.h1 & bloom_mask);
        if (!(bloom[b1 >> 6] & (1ULL << (b1 & 63)))) return false;
        
        uint32_t b2 = (uint32_t)(h.h2 & bloom_mask);
        if (!(bloom[b2 >> 6] & (1ULL << (b2 & 63)))) return false;
        
        uint32_t idx = (uint32_t)((h.h1 ^ h.h2) & mask);
        while(status[idx] != 0) {
            if (keys[idx] == h) {
                if (status[idx] == 1) {
                    status[idx] = 2;
                    seen_indices.push_back(idx);
                    return true;
                }
                return false;
            }
            idx = (idx + 1) & mask;
        }
        return false;
    }
    
    void extract_seen(vector<hash_t>& new_V) {
        for (uint32_t idx : seen_indices) {
            new_V.push_back(keys[idx]);
        }
        seen_indices.clear();
    }
} global_ht;

void compute_hashes(const string& S, int L, vector<hash_t>& hashes) {
    hashes.clear();
    if (S.length() < (size_t)L) return;
    
    uint64_t h1 = 0, h2 = 0;
    uint64_t p1_L = power(P1, L);
    uint64_t p2_L = power(P2, L);
    
    for (int i = 0; i < L; ++i) {
        h1 = h1 * P1 + S[i];
        h2 = h2 * P2 + S[i];
    }
    
    hashes.push_back(hash_t{h1, h2});
    
    for (size_t i = L; i < S.length(); ++i) {
        h1 = h1 * P1 + S[i] - S[i - L] * p1_L;
        h2 = h2 * P2 + S[i] - S[i - L] * p2_L;
        hashes.push_back(hash_t{h1, h2});
    }
}

bool check(int L, int& match_start_in_s0, const vector<string>& strings) {
    if (L == 0) return true;
    
    compute_hashes(strings[0], L, current_V);
    if (current_V.empty()) return false;
    
    for (size_t i = 1; i < strings.size(); ++i) {
        global_ht.build(current_V);
        current_V.clear();
        
        uint64_t h1 = 0, h2 = 0;
        uint64_t p1_L = power(P1, L);
        uint64_t p2_L = power(P2, L);
        
        const string& S = strings[i];
        if (S.length() < (size_t)L) return false;
        
        for (int j = 0; j < L; ++j) {
            h1 = h1 * P1 + S[j];
            h2 = h2 * P2 + S[j];
        }
        
        global_ht.check_and_mark(hash_t{h1, h2});
        
        for (size_t j = L; j < S.length(); ++j) {
            h1 = h1 * P1 + S[j] - S[j - L] * p1_L;
            h2 = h2 * P2 + S[j] - S[j - L] * p2_L;
            global_ht.check_and_mark(hash_t{h1, h2});
        }
        
        global_ht.extract_seen(current_V);
        if (current_V.empty()) return false;
    }
    
    hash_t target = current_V[0];
    uint64_t h1 = 0, h2 = 0;
    uint64_t p1_L = power(P1, L);
    uint64_t p2_L = power(P2, L);
    const string& S0 = strings[0];
    
    for (int j = 0; j < L; ++j) {
        h1 = h1 * P1 + S0[j];
        h2 = h2 * P2 + S0[j];
    }
    
    if (hash_t{h1, h2} == target) {
        match_start_in_s0 = 0;
        return true;
    }
    
    for (size_t j = L; j < S0.length(); ++j) {
        h1 = h1 * P1 + S0[j] - S0[j - L] * p1_L;
        h2 = h2 * P2 + S0[j] - S0[j - L] * p2_L;
        if (hash_t{h1, h2} == target) {
            match_start_in_s0 = j - L + 1;
            return true;
        }
    }
    
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N;
    if (!(cin >> N)) return 0;
    
    string line;
    getline(cin, line);
    
    vector<string> strings(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, strings[i]);
        if (!strings[i].empty() && strings[i].back() == '\r') {
            strings[i].pop_back();
        }
    }
    
    if (N == 0) {
        cout << "\n";
        return 0;
    }
    
    sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.length() < b.length();
    });
    
    mt19937_64 rng(chrono::steady_clock::now().time_since_epoch().count());
    P1 = rng() | 1;
    P2 = rng() | 1;
    while(P1 < 256) P1 = rng() | 1;
    while(P2 < 256) P2 = rng() | 1;
    if (P1 == P2) P2 += 2;
    
    int low = 0, high = strings[0].length();
    int best_L = 0;
    int best_start = 0;
    
    while (low <= high) {
        int mid = low + (high - low) / 2;
        int start = -1;
        if (check(mid, start, strings)) {
            best_L = mid;
            best_start = start;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }
    
    if (best_L == 0) {
        cout << "\n";
    } else {
        cout << strings[0].substr(best_start, best_L) << "\n";
    }
    
    return 0;
}
