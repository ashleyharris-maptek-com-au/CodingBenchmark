#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <algorithm>
#include <cmath>

using namespace std;

int D;

struct Point {
    int c[20];
    Point() {
        for (int i = 0; i < 20; ++i) c[i] = 0;
    }
    bool operator==(const Point& o) const {
        for (int i = 0; i < D; ++i) {
            if (c[i] != o.c[i]) return false;
        }
        return true;
    }
    bool operator!=(const Point& o) const {
        return !(*this == o);
    }
};

struct HashPoint {
    size_t operator()(const Point& p) const {
        size_t res = 0;
        for (int i = 0; i < D; ++i) {
            res ^= std::hash<int>()(p.c[i]) + 0x9e3779b9 + (res << 6) + (res >> 2);
        }
        return res;
    }
};

int manhattan(const Point& a, const Point& b) {
    int dist = 0;
    for (int i = 0; i < D; ++i) dist += abs(a.c[i] - b.c[i]);
    return dist;
}

vector<int> bounds;
unordered_set<Point, HashPoint> obstacles;
unordered_set<Point, HashPoint> current_foods;
deque<Point> snake;

struct AStarNode {
    Point p;
    int g;
    int f;
    bool operator>(const AStarNode& other) const {
        if (f != other.f) return f > other.f;
        return g < other.g;
    }
};

vector<Point> find_path(Point start, Point target) {
    priority_queue<AStarNode, vector<AStarNode>, greater<AStarNode>> pq;
    unordered_map<Point, int, HashPoint> g_score;
    unordered_map<Point, Point, HashPoint> parent;

    pq.push({start, 0, manhattan(start, target)});
    g_score[start] = 0;

    int L = snake.size();

    while (!pq.empty()) {
        AStarNode curr = pq.top();
        pq.pop();

        if (curr.g > g_score[curr.p]) continue;

        if (curr.p == target) {
            vector<Point> path;
            Point p = target;
            while (p != start) {
                path.push_back(p);
                p = parent[p];
            }
            reverse(path.begin(), path.end());
            return path;
        }

        for (int i = 0; i < D; ++i) {
            for (int dir : {-1, 1}) {
                Point v = curr.p;
                v.c[i] += dir;

                if (v.c[i] < 0 || v.c[i] >= bounds[i]) continue;
                if (obstacles.count(v)) continue;
                if (v != target && current_foods.count(v)) continue;

                int next_g = curr.g + 1;

                bool collision = false;
                for (int j = 0; j < L - next_g; ++j) {
                    if (v == snake[j]) {
                        collision = true;
                        break;
                    }
                }
                if (collision) continue;

                if (g_score.find(v) == g_score.end() || next_g < g_score[v]) {
                    g_score[v] = next_g;
                    parent[v] = curr.p;
                    pq.push({v, next_g, next_g + manhattan(v, target)});
                }
            }
        }
    }
    return {};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    cin >> snake_length;

    Point start_pos;
    for (int i = 0; i < D; ++i) cin >> start_pos.c[i];

    int F;
    cin >> F;
    vector<Point> remaining_foods(F);
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) cin >> remaining_foods[i].c[j];
        current_foods.insert(remaining_foods[i]);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        Point obs;
        for (int j = 0; j < D; ++j) cin >> obs.c[j];
        obstacles.insert(obs);
    }

    for (int i = 0; i < snake_length; ++i) {
        snake.push_back(start_pos);
    }

    Point current_target;
    bool has_target = false;

    for (int turn = 0; turn < max_turns; ++turn) {
        if (remaining_foods.empty()) break;

        vector<Point> path;
        if (has_target) {
            path = find_path(snake.front(), current_target);
        }

        if (path.empty()) {
            vector<Point> sorted_foods = remaining_foods;
            sort(sorted_foods.begin(), sorted_foods.end(), [&](const Point& a, const Point& b) {
                return manhattan(snake.front(), a) < manhattan(snake.front(), b);
            });

            for (const Point& target : sorted_foods) {
                path = find_path(snake.front(), target);
                if (!path.empty()) {
                    current_target = target;
                    has_target = true;
                    break;
                }
            }
        }

        Point next_pos;
        if (!path.empty()) {
            next_pos = path.front();
        } else {
            vector<pair<Point, int>> valid_moves;
            for (int i = 0; i < D; ++i) {
                for (int dir : {-1, 1}) {
                    Point v = snake.front();
                    v.c[i] += dir;
                    if (v.c[i] < 0 || v.c[i] >= bounds[i]) continue;
                    if (obstacles.count(v)) continue;

                    bool collision = false;
                    int limit = current_foods.count(v) ? snake.size() : snake.size() - 1;
                    for (int j = 0; j < limit; ++j) {
                        if (v == snake[j]) {
                            collision = true;
                            break;
                        }
                    }
                    if (!collision) {
                        int min_dist = 1e9;
                        for (const Point& f : remaining_foods) {
                            min_dist = min(min_dist, manhattan(v, f));
                        }
                        valid_moves.push_back({v, min_dist});
                    }
                }
            }
            if (!valid_moves.empty()) {
                sort(valid_moves.begin(), valid_moves.end(), [](const pair<Point, int>& a, const pair<Point, int>& b) {
                    return a.second < b.second;
                });
                next_pos = valid_moves.front().first;
            } else {
                break; // No valid moves, game over
            }
        }

        int move_axis = -1;
        int move_dir = 0;
        for (int i = 0; i < D; ++i) {
            if (next_pos.c[i] != snake.front().c[i]) {
                move_axis = i;
                move_dir = next_pos.c[i] - snake.front().c[i];
                break;
            }
        }

        cout << move_axis << " " << move_dir << endl;

        snake.push_front(next_pos);

        auto it = find(remaining_foods.begin(), remaining_foods.end(), next_pos);
        if (it != remaining_foods.end()) {
            remaining_foods.erase(it);
            current_foods.erase(next_pos);
            has_target = false;
        } else {
            snake.pop_back();
        }
    }

    return 0;
}
