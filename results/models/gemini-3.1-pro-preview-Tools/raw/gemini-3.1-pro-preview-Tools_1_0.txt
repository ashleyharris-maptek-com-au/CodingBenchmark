#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <cstdint>

using namespace std;

struct Point {
    float x, y;
};

inline float dist(const Point& a, const Point& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return std::sqrt(dx * dx + dy * dy);
}

uint64_t hilbert(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int i = 31; i >= 0; i--) {
        uint32_t rx = (x >> i) & 1;
        uint32_t ry = (y >> i) & 1;
        d = (d << 2) | ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    auto start_time = std::chrono::steady_clock::now();
    
    int N;
    if (!(cin >> N)) return 0;
    
    if (N <= 0) return 0;
    
    vector<Point> pts(N);
    for (int i = 0; i < N; i++) {
        cin >> pts[i].x >> pts[i].y;
    }
    
    if (N == 1) {
        cout << 0 << "\n";
        return 0;
    }
    
    vector<int> tour(N);
    
    if (N <= 5000) {
        vector<bool> visited(N, false);
        tour[0] = 0;
        visited[0] = true;
        int curr = 0;
        for (int i = 1; i < N; i++) {
            int best_next = -1;
            float best_dist = 1e30f;
            for (int j = 0; j < N; j++) {
                if (!visited[j]) {
                    float d = dist(pts[curr], pts[j]);
                    if (d < best_dist) {
                        best_dist = d;
                        best_next = j;
                    }
                }
            }
            tour[i] = best_next;
            visited[best_next] = true;
            curr = best_next;
        }
    } else {
        float min_x = pts[0].x, max_x = pts[0].x;
        float min_y = pts[0].y, max_y = pts[0].y;
        for(int i = 1; i < N; i++) {
            if(pts[i].x < min_x) min_x = pts[i].x;
            if(pts[i].x > max_x) max_x = pts[i].x;
            if(pts[i].y < min_y) min_y = pts[i].y;
            if(pts[i].y > max_y) max_y = pts[i].y;
        }
        double range_x = max_x - min_x;
        double range_y = max_y - min_y;
        double range = max(range_x, range_y);
        if (range == 0.0) range = 1.0;
        
        vector<uint64_t> hcodes(N);
        double max_val = 4294967295.0;
        for (int i = 0; i < N; i++) {
            double nx = (pts[i].x - min_x) / range;
            double ny = (pts[i].y - min_y) / range;
            uint32_t ix = (uint32_t)std::min(max_val, std::max(0.0, nx * max_val));
            uint32_t iy = (uint32_t)std::min(max_val, std::max(0.0, ny * max_val));
            hcodes[i] = hilbert(ix, iy);
        }
        
        for(int i = 0; i < N; i++) tour[i] = i;
        std::sort(tour.begin(), tour.end(), [&](int a, int b) {
            return hcodes[a] < hcodes[b];
        });
        
        auto it = std::find(tour.begin(), tour.end(), 0);
        std::rotate(tour.begin(), it, tour.end());
    }
    
    int window = N;
    if (N > 5000) {
        long long max_ops = 500000000LL;
        window = (int)(max_ops * 2 / N);
        if (window < 20) window = 20;
        if (window > N) window = N;
    }
    
    bool improved = true;
    int passes = 0;
    while (improved && passes < 30) {
        improved = false;
        passes++;
        for (int i = 0; i < N; i++) {
            if ((i & 1023) == 0) {
                auto now = std::chrono::steady_clock::now();
                if (std::chrono::duration_cast<std::chrono::seconds>(now - start_time).count() > 280) {
                    improved = false;
                    break;
                }
            }
            
            int limit = min(N, i + window);
            for (int j = i + 2; j < limit; j++) {
                if (i == 0 && j == N - 1) continue; 
                
                int u1 = tour[i];
                int v1 = tour[i + 1];
                int u2 = tour[j];
                int v2 = tour[(j + 1 == N) ? 0 : j + 1];
                
                float d1 = dist(pts[u1], pts[v1]) + dist(pts[u2], pts[v2]);
                float d2 = dist(pts[u1], pts[u2]) + dist(pts[v1], pts[v2]);
                
                if (d2 < d1 - 1e-6f) {
                    int L = i + 1;
                    int R = j;
                    while (L < R) {
                        int temp = tour[L]; tour[L] = tour[R]; tour[R] = temp;
                        L++; R--;
                    }
                    improved = true;
                }
            }
        }
    }
    
    for (int i = 0; i < N; i++) {
        cout << tour[i] << (i == N - 1 ? "" : " ");
    }
    cout << "\n";
    
    return 0;
}
