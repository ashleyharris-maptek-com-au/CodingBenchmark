#include <iostream>
#include <vector>
#include <string>
#include <cctype>
#include <cmath>
#include <algorithm>
#include <chrono>

using namespace std;

// --- Constants & Types ---
const int EMPTY = 0;
const int PAWN = 1, KNIGHT = 2, BISHOP = 3, ROOK = 4, QUEEN = 5, KING = 6;
const int WHITE = 8, BLACK = 16;
const int COLOR_MASK = 24, PIECE_MASK = 7;

// Directions
const int dirs[8] = {-9, -8, -7, -1, 1, 7, 8, 9};
const int knight_dirs[8] = {-17, -15, -10, -6, 6, 10, 15, 17};

// Values for evaluation
const int piece_values[7] = {0, 100, 300, 310, 500, 900, 20000};

struct Move {
    int from;
    int to;
    int promotion; // 0 or piece type
    int capture;   // piece captured (for unmake)
    int moved;     // piece moved

    bool operator==(const Move& other) const {
        return from == other.from && to == other.to && promotion == other.promotion;
    }

    string to_uci() const {
        string s = "";
        s += char('a' + (from % 8));
        s += char('1' + (from / 8));
        s += char('a' + (to % 8));
        s += char('1' + (to / 8));
        if (promotion) {
            if (promotion == QUEEN) s += 'q';
            else if (promotion == ROOK) s += 'r';
            else if (promotion == BISHOP) s += 'b';
            else if (promotion == KNIGHT) s += 'n';
        }
        return s;
    }
};

class Board {
public:
    int squares[64];
    int turn; // WHITE or BLACK
    
    // Castling rights and En Passant can be added here for a fully compliant engine,
    // but for this basic bot we focus on standard piece movements.
    
    Board() {
        fill(squares, squares + 64, EMPTY);
        turn = WHITE;
    }

    void load_fen(string fen) {
        fill(squares, squares + 64, EMPTY);
        int row = 7, col = 0;
        int i = 0;
        for (; i < fen.size() && fen[i] != ' '; i++) {
            char c = fen[i];
            if (c == '/') {
                row--;
                col = 0;
            } else if (isdigit(c)) {
                col += (c - '0');
            } else {
                int color = isupper(c) ? WHITE : BLACK;
                int piece = EMPTY;
                c = tolower(c);
                if (c == 'p') piece = PAWN;
                else if (c == 'n') piece = KNIGHT;
                else if (c == 'b') piece = BISHOP;
                else if (c == 'r') piece = ROOK;
                else if (c == 'q') piece = QUEEN;
                else if (c == 'k') piece = KING;
                
                squares[row * 8 + col] = color | piece;
                col++;
            }
        }
        
        if (i < fen.size()) {
            i++; // skip space
            turn = (fen[i] == 'w') ? WHITE : BLACK;
        }
    }

    void make_move(const Move& m) {
        squares[m.to] = m.promotion ? (turn | m.promotion) : squares[m.from];
        squares[m.from] = EMPTY;
        turn = (turn == WHITE) ? BLACK : WHITE;
    }

    void unmake_move(const Move& m) {
        turn = (turn == WHITE) ? BLACK : WHITE;
        squares[m.from] = m.moved;
        squares[m.to] = m.capture;
    }

    bool in_bounds(int r, int c) const {
        return r >= 0 && r < 8 && c >= 0 && c < 8;
    }

    // Pseudo-legal move generator
    vector<Move> generate_moves() const {
        vector<Move> moves;
        int opp_color = (turn == WHITE) ? BLACK : WHITE;
        
        for (int i = 0; i < 64; i++) {
            if ((squares[i] & COLOR_MASK) != turn) continue;
            
            int piece = squares[i] & PIECE_MASK;
            int r = i / 8;
            int c = i % 8;

            if (piece == PAWN) {
                int dir = (turn == WHITE) ? 1 : -1;
                int start_rank = (turn == WHITE) ? 1 : 6;
                int prom_rank = (turn == WHITE) ? 7 : 0;
                
                // Push
                int next_sq = i + dir * 8;
                if (in_bounds(r + dir, c) && squares[next_sq] == EMPTY) {
                    add_pawn_moves(moves, i, next_sq, r + dir == prom_rank);
                    // Double push
                    if (r == start_rank) {
                        int double_sq = i + dir * 16;
                        if (squares[double_sq] == EMPTY) {
                            moves.push_back({i, double_sq, 0, EMPTY, squares[i]});
                        }
                    }
                }
                
                // Captures
                for (int dc : {-1, 1}) {
                    if (in_bounds(r + dir, c + dc)) {
                        int cap_sq = (r + dir) * 8 + (c + dc);
                        if (squares[cap_sq] != EMPTY && (squares[cap_sq] & COLOR_MASK) == opp_color) {
                            add_pawn_moves(moves, i, cap_sq, r + dir == prom_rank);
                        }
                    }
                }
            } 
            else if (piece == KNIGHT) {
                int kr[8] = {-2, -2, -1, -1, 1, 1, 2, 2};
                int kc[8] = {-1, 1, -2, 2, -2, 2, -1, 1};
                for (int d = 0; d < 8; d++) {
                    int nr = r + kr[d], nc = c + kc[d];
                    if (in_bounds(nr, nc)) {
                        int target = nr * 8 + nc;
                        if ((squares[target] & COLOR_MASK) != turn) {
                            moves.push_back({i, target, 0, squares[target], squares[i]});
                        }
                    }
                }
            }
            else if (piece == KING) {
                for (int dr = -1; dr <= 1; dr++) {
                    for (int dc = -1; dc <= 1; dc++) {
                        if (dr == 0 && dc == 0) continue;
                        int nr = r + dr, nc = c + dc;
                        if (in_bounds(nr, nc)) {
                            int target = nr * 8 + nc;
                            if ((squares[target] & COLOR_MASK) != turn) {
                                moves.push_back({i, target, 0, squares[target], squares[i]});
                            }
                        }
                    }
                }
            }
            // Sliding pieces: BISHOP, ROOK, QUEEN
            else {
                int start_idx = (piece == ROOK) ? 0 : 4;
                int end_idx = (piece == BISHOP) ? 4 : 8;
                
                int dr[8] = {-1, 1, 0, 0, -1, -1, 1, 1};
                int dc[8] = {0, 0, -1, 1, -1, 1, -1, 1};
                
                for (int d = start_idx; d < end_idx; d++) {
                    int nr = r + dr[d], nc = c + dc[d];
                    while (in_bounds(nr, nc)) {
                        int target = nr * 8 + nc;
                        if (squares[target] != EMPTY) {
                            if ((squares[target] & COLOR_MASK) != turn) {
                                moves.push_back({i, target, 0, squares[target], squares[i]});
                            }
                            break; // blocked
                        }
                        moves.push_back({i, target, 0, EMPTY, squares[i]});
                        nr += dr[d];
                        nc += dc[d];
                    }
                }
            }
        }
        return moves;
    }

    // Helper for pawn promotion
    void add_pawn_moves(vector<Move>& moves, int from, int to, bool promote) const {
        if (promote) {
            moves.push_back({from, to, QUEEN, squares[to], squares[from]});
            moves.push_back({from, to, ROOK, squares[to], squares[from]});
            moves.push_back({from, to, BISHOP, squares[to], squares[from]});
            moves.push_back({from, to, KNIGHT, squares[to], squares[from]});
        } else {
            moves.push_back({from, to, 0, squares[to], squares[from]});
        }
    }

    bool is_in_check(int color) const {
        // Find king
        int king_sq = -1;
        for (int i = 0; i < 64; i++) {
            if (squares[i] == (color | KING)) {
                king_sq = i;
                break;
            }
        }
        if (king_sq == -1) return false; // Should not happen in valid positions

        int r = king_sq / 8;
        int c = king_sq % 8;
        int opp_color = (color == WHITE) ? BLACK : WHITE;

        // Check knights
        int kr[8] = {-2, -2, -1, -1, 1, 1, 2, 2};
        int kc[8] = {-1, 1, -2, 2, -2, 2, -1, 1};
        for (int d = 0; d < 8; d++) {
            int nr = r + kr[d], nc = c + kc[d];
            if (in_bounds(nr, nc)) {
                if (squares[nr * 8 + nc] == (opp_color | KNIGHT)) return true;
            }
        }

        // Check pawns
        int pawn_dir = (color == WHITE) ? 1 : -1; // Opponent pawns attack from opposite direction
        for (int dc : {-1, 1}) {
            if (in_bounds(r + pawn_dir, c + dc)) {
                if (squares[(r + pawn_dir) * 8 + (c + dc)] == (opp_color | PAWN)) return true;
            }
        }

        // Check sliding pieces & kings
        int dr[8] = {-1, 1, 0, 0, -1, -1, 1, 1};
        int dc[8] = {0, 0, -1, 1, -1, 1, -1, 1};
        
        for (int d = 0; d < 8; d++) {
            int nr = r + dr[d], nc = c + dc[d];
            int dist = 1;
            while (in_bounds(nr, nc)) {
                int target = nr * 8 + nc;
                int p = squares[target];
                if (p != EMPTY) {
                    if ((p & COLOR_MASK) == opp_color) {
                        int type = p & PIECE_MASK;
                        if (dist == 1 && type == KING) return true;
                        if (d < 4 && (type == ROOK || type == QUEEN)) return true;
                        if (d >= 4 && (type == BISHOP || type == QUEEN)) return true;
                    }
                    break;
                }
                nr += dr[d];
                nc += dc[d];
                dist++;
            }
        }
        return false;
    }
    
    // Returns strictly legal moves
    vector<Move> get_legal_moves() {
        vector<Move> pseudo = generate_moves();
        vector<Move> legal;
        for (const Move& m : pseudo) {
            make_move(m);
            // After making a move, check if the color that just moved (now opposite of turn) is in check
            int moved_color = (turn == WHITE) ? BLACK : WHITE; 
            if (!is_in_check(moved_color)) {
                legal.push_back(m);
            }
            unmake_move(m);
        }
        return legal;
    }
};

// --- Evaluation ---
int evaluate(const Board& board) {
    int score = 0;
    for (int i = 0; i < 64; i++) {
        int p = board.squares[i];
        if (p != EMPTY) {
            int val = piece_values[p & PIECE_MASK];
            // Basic positional bonus to encourage center control
            int r = i / 8, c = i % 8;
            int dist_center = abs(3.5 - r) + abs(3.5 - c);
            val += (10 - dist_center); // Max +10 in center

            if ((p & COLOR_MASK) == WHITE) score += val;
            else score -= val;
        }
    }
    return score;
}

// --- Search ---
const int INF = 1000000;
int nodes = 0;

int minimax(Board& board, int depth, int alpha, int beta) {
    nodes++;
    if (depth == 0) {
        return evaluate(board);
    }

    vector<Move> moves = board.get_legal_moves();
    
    if (moves.empty()) {
        if (board.is_in_check(board.turn)) {
             return (board.turn == WHITE) ? -INF + (100 - depth) : INF - (100 - depth);
        }
        return 0; // Stalemate
    }

    if (board.turn == WHITE) {
        int max_eval = -INF;
        for (const Move& m : moves) {
            board.make_move(m);
            int eval = minimax(board, depth - 1, alpha, beta);
            board.unmake_move(m);
            max_eval = max(max_eval, eval);
            alpha = max(alpha, eval);
            if (beta <= alpha) break;
        }
        return max_eval;
    } else {
        int min_eval = INF;
        for (const Move& m : moves) {
            board.make_move(m);
            int eval = minimax(board, depth - 1, alpha, beta);
            board.unmake_move(m);
            min_eval = min(min_eval, eval);
            beta = min(beta, eval);
            if (beta <= alpha) break;
        }
        return min_eval;
    }
}

Move find_best_move(Board& board, int depth) {
    nodes = 0;
    vector<Move> moves = board.get_legal_moves();
    if (moves.empty()) return {-1, -1, 0, 0, 0}; // No moves

    Move best_move = moves[0];
    
    if (board.turn == WHITE) {
        int max_eval = -INF;
        for (const Move& m : moves) {
            board.make_move(m);
            int eval = minimax(board, depth - 1, -INF, INF);
            board.unmake_move(m);
            if (eval > max_eval) {
                max_eval = eval;
                best_move = m;
            }
        }
    } else {
        int min_eval = INF;
        for (const Move& m : moves) {
            board.make_move(m);
            int eval = minimax(board, depth - 1, -INF, INF);
            board.unmake_move(m);
            if (eval < min_eval) {
                min_eval = eval;
                best_move = m;
            }
        }
    }
    
    return best_move;
}

int main(int argc, char* argv[]) {
    Board board;
    string fen = "rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1"; // Default starting position
    int depth = 4; // Search depth

    // Parse command line arguments if provided
    for(int i = 1; i < argc; ++i) {
        string arg = argv[i];
        if (arg == "--fen" && i + 1 < argc) {
            fen = argv[++i];
            // FEN strings often have spaces, so we might need to stitch them back together
            // if passed as separate arguments, but standard use quotes around it: --fen "..."
        }
        else if (arg == "--depth" && i + 1 < argc) {
            depth = stoi(argv[++i]);
        }
    }

    board.load_fen(fen);
    
    auto start_time = chrono::high_resolution_clock::now();
    
    Move best_move = find_best_move(board, depth);
    
    auto end_time = chrono::high_resolution_clock::now();
    auto duration = chrono::duration_cast<chrono::milliseconds>(end_time - start_time);

    if (best_move.from == -1) {
        cout << "No valid moves (Checkmate or Stalemate)" << endl;
    } else {
        cout << "Best Move: " << best_move.to_uci() << endl;
        cout << "Nodes searched: " << nodes << endl;
        cout << "Time taken: " << duration.count() << " ms" << endl;
    }

    return 0;
}
