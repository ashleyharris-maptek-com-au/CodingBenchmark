using System;
using System.IO;
using System.Diagnostics;
using System.Collections.Generic;

class Program {
    struct Cut : IComparable<Cut> {
        public int length;
        public int index;
        public Cut(int l, int i) { length = l; index = i; }
        public int CompareTo(Cut other) {
            int c = other.length.CompareTo(this.length);
            if (c == 0) return this.index.CompareTo(other.index);
            return c;
        }
    }

    static void Main() {
        Stream stream = Console.OpenStandardInput();
        byte[] buffer = new byte[65536];
        int head = 0, tail = 0;

        int ReadInt() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = Read();
            }
            int res = 0;
            while (c > 32) {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res;
        }

        int Read() {
            if (head >= tail) {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }

        int n = ReadInt();
        if (n <= 0) return;
        int stock_length = ReadInt();

        Cut[] cuts = new Cut[n];
        long sum = 0;
        for (int i = 0; i < n; i++) {
            int len = ReadInt();
            cuts[i] = new Cut(len, i);
            sum += len;
        }

        long lower_bound = (sum + stock_length - 1) / stock_length;

        Array.Sort(cuts);
        Cut[] original_cuts = new Cut[n];
        Array.Copy(cuts, original_cuts, n);

        int nextPow2 = 1;
        while (nextPow2 < n) nextPow2 *= 2;
        int[] tree = new int[2 * nextPow2];

        int[] best_bin_assignment = new int[n];
        int best_num_bins = int.MaxValue;

        Stopwatch sw = Stopwatch.StartNew();
        Random rnd = new Random(42);

        for (int iter = 0; ; iter++) {
            Array.Fill(tree, stock_length);
            int num_bins = 0;
            int[] current_assignment = new int[n];
            bool valid = true;

            for (int i = 0; i < n; i++) {
                int len = cuts[i].length;
                if (len > tree[1]) {
                    valid = false;
                    break;
                }
                
                int node = 1;
                int left = 0, right = nextPow2 - 1;
                while (left < right) {
                    int mid = (left + right) / 2;
                    if (tree[2 * node] >= len) {
                        node = 2 * node;
                        right = mid;
                    } else {
                        node = 2 * node + 1;
                        left = mid + 1;
                    }
                }
                
                int bin_idx = left;
                current_assignment[cuts[i].index] = bin_idx;
                if (bin_idx >= num_bins) num_bins = bin_idx + 1;

                tree[node] -= len;
                node /= 2;
                while (node > 0) {
                    tree[node] = Math.Max(tree[2 * node], tree[2 * node + 1]);
                    node /= 2;
                }
            }

            if (valid && num_bins < best_num_bins) {
                best_num_bins = num_bins;
                Array.Copy(current_assignment, best_bin_assignment, n);
            }

            if (best_num_bins == lower_bound) break;
            
            // Limit to 4.5 minutes
            if (sw.ElapsedMilliseconds > 270000) break;
            
            // Limit iterations to avoid endless loops on impossible cases
            if (iter > 50000) break;
            
            if (iter % 10 == 0) {
                Array.Copy(original_cuts, cuts, n);
            }
            for (int i = 0; i < n - 1; i++) {
                if (rnd.NextDouble() < 0.3) {
                    var tmp = cuts[i];
                    cuts[i] = cuts[i+1];
                    cuts[i+1] = tmp;
                }
            }
        }

        int[] head_bin = new int[best_num_bins];
        Array.Fill(head_bin, -1);
        int[] next_cut = new int[n];

        for (int i = n - 1; i >= 0; i--) {
            int b = best_bin_assignment[i];
            next_cut[i] = head_bin[b];
            head_bin[b] = i;
        }

        Stream outStream = Console.OpenStandardOutput();
        byte[] outBuf = new byte[65536];
        int outPos = 0;

        void Flush() {
            if (outPos > 0) {
                outStream.Write(outBuf, 0, outPos);
                outPos = 0;
            }
        }

        void WriteInt(int val, byte terminator) {
            if (val == 0) {
                if (outPos == outBuf.Length) Flush();
                outBuf[outPos++] = (byte)'0';
            } else {
                int temp = val;
                int len = 0;
                while (temp > 0) { temp /= 10; len++; }
                if (outPos + len >= outBuf.Length) Flush();
                int end = outPos + len - 1;
                temp = val;
                while (temp > 0) {
                    outBuf[end--] = (byte)('0' + (temp % 10));
                    temp /= 10;
                }
                outPos += len;
            }
            if (terminator != 0) {
                if (outPos == outBuf.Length) Flush();
                outBuf[outPos++] = terminator;
            }
        }

        WriteInt(best_num_bins, (byte)'\n');
        for (int b = 0; b < best_num_bins; b++) {
            int curr = head_bin[b];
            while (curr != -1) {
                int next = next_cut[curr];
                WriteInt(curr, next == -1 ? (byte)'\n' : (byte)' ');
                curr = next;
            }
        }
        Flush();
    }
}
