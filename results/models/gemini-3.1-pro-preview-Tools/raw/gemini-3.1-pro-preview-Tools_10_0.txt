#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;

struct Rect {
    int id;
    int w, h;
};

struct Placement {
    int board_index;
    int x, y;
    int rotated;
};

struct FreeRect {
    int board_index;
    int x, y, w, h;
};

enum SortType {
    SORT_AREA_DESC,
    SORT_MAX_DIM_DESC,
    SORT_PERIMETER_DESC,
    SORT_W_THEN_H_DESC
};

enum AlgoType {
    ALGO_GUILLOTINE,
    ALGO_MAXRECTS
};

enum ScoreType {
    SCORE_BAF,
    SCORE_BSSF,
    SCORE_BLSF,
    SCORE_BLF
};

enum SplitType {
    SPLIT_MAXAS,
    SPLIT_MINAS,
    SPLIT_SLAS,
    SPLIT_LLAS
};

struct Heuristic {
    int algo;
    int sort_type;
    int score_type;
    int split_type;
};

struct MaxRectsBoard {
    std::vector<FreeRect> free_rects;
    int max_w;
    int max_h;
};

auto get_sort_lambda = [](int sort_type) {
    return [sort_type](const Rect& a, const Rect& b) {
        long long area_a = (long long)a.w * a.h;
        long long area_b = (long long)b.w * b.h;
        int max_a = std::max(a.w, a.h);
        int max_b = std::max(b.w, b.h);
        int min_a = std::min(a.w, a.h);
        int min_b = std::min(b.w, b.h);
        long long perim_a = (long long)a.w + a.h;
        long long perim_b = (long long)b.w + b.h;
        
        if (sort_type == SORT_AREA_DESC) {
            if (area_a != area_b) return area_a > area_b;
            if (max_a != max_b) return max_a > max_b;
            return a.id < b.id;
        } else if (sort_type == SORT_MAX_DIM_DESC) {
            if (max_a != max_b) return max_a > max_b;
            if (min_a != min_b) return min_a > min_b;
            return a.id < b.id;
        } else if (sort_type == SORT_PERIMETER_DESC) {
            if (perim_a != perim_b) return perim_a > perim_b;
            if (max_a != max_b) return max_a > max_b;
            return a.id < b.id;
        } else if (sort_type == SORT_W_THEN_H_DESC) {
            if (max_a != max_b) return max_a > max_b;
            if (min_a != min_b) return min_a > min_b;
            return a.id < b.id;
        }
        return a.id < b.id;
    };
};

long long get_score(int score_type, const FreeRect& fr, int pw, int ph) {
    if (score_type == SCORE_BAF) {
        return (long long)fr.w * fr.h;
    } else if (score_type == SCORE_BSSF) {
        return std::min(fr.w - pw, fr.h - ph);
    } else if (score_type == SCORE_BLSF) {
        return std::max(fr.w - pw, fr.h - ph);
    } else if (score_type == SCORE_BLF) {
        return ((long long)fr.y << 32) | (unsigned int)fr.x;
    }
    return 2e18;
}

void split_and_prune_maxrects(MaxRectsBoard& brd, int px, int py, int pw, int ph, int min_req_side, long long min_req_area) {
    std::vector<FreeRect> new_rects;
    for (int i = 0; i < (int)brd.free_rects.size(); ) {
        const FreeRect& F = brd.free_rects[i];
        if (px < F.x + F.w && px + pw > F.x && py < F.y + F.h && py + ph > F.y) {
            if (py + ph < F.y + F.h) {
                int nw = F.w;
                int nh = F.y + F.h - (py + ph);
                if (std::max(nw, nh) >= min_req_side && (long long)nw * nh >= min_req_area)
                    new_rects.push_back({0, F.x, py + ph, nw, nh});
            }
            if (py > F.y) {
                int nw = F.w;
                int nh = py - F.y;
                if (std::max(nw, nh) >= min_req_side && (long long)nw * nh >= min_req_area)
                    new_rects.push_back({0, F.x, F.y, nw, nh});
            }
            if (px > F.x) {
                int nw = px - F.x;
                int nh = F.h;
                if (std::max(nw, nh) >= min_req_side && (long long)nw * nh >= min_req_area)
                    new_rects.push_back({0, F.x, F.y, nw, nh});
            }
            if (px + pw < F.x + F.w) {
                int nw = F.x + F.w - (px + pw);
                int nh = F.h;
                if (std::max(nw, nh) >= min_req_side && (long long)nw * nh >= min_req_area)
                    new_rects.push_back({0, px + pw, F.y, nw, nh});
            }
            brd.free_rects[i] = brd.free_rects.back();
            brd.free_rects.pop_back();
        } else {
            i++;
        }
    }
    
    brd.free_rects.insert(brd.free_rects.end(), new_rects.begin(), new_rects.end());
    
    for (int i = 0; i < (int)brd.free_rects.size(); ) {
        bool contained = false;
        for (int j = 0; j < (int)brd.free_rects.size(); j++) {
            if (i == j) continue;
            const FreeRect& A = brd.free_rects[i];
            const FreeRect& B = brd.free_rects[j];
            if (B.x <= A.x && B.y <= A.y && B.x + B.w >= A.x + A.w && B.y + B.h >= A.y + A.h) {
                contained = true;
                break;
            }
        }
        if (contained) {
            brd.free_rects[i] = brd.free_rects.back();
            brd.free_rects.pop_back();
        } else {
            i++;
        }
    }
    
    if (brd.free_rects.size() > 100) {
        std::sort(brd.free_rects.begin(), brd.free_rects.end(), [](const FreeRect& a, const FreeRect& b) {
            return (long long)a.w * a.h > (long long)b.w * b.h;
        });
        brd.free_rects.resize(100);
    }
    
    brd.max_w = 0;
    brd.max_h = 0;
    for (const auto& fr : brd.free_rects) {
        brd.max_w = std::max(brd.max_w, fr.w);
        brd.max_h = std::max(brd.max_h, fr.h);
    }
}

std::pair<int, std::vector<Placement>> solve_maxrects(
    int N, int W, int H, const std::vector<Rect>& original_rects,
    const Heuristic& heur) 
{
    std::vector<Rect> rects = original_rects;
    std::sort(rects.begin(), rects.end(), get_sort_lambda(heur.sort_type));
    
    std::vector<int> suffix_min_side(N);
    std::vector<long long> suffix_min_area(N);
    int current_min_side = 2e9;
    long long current_min_area = 2e18;
    for (int i = N - 1; i >= 0; i--) {
        current_min_side = std::min(current_min_side, std::min(rects[i].w, rects[i].h));
        current_min_area = std::min(current_min_area, (long long)rects[i].w * rects[i].h);
        suffix_min_side[i] = current_min_side;
        suffix_min_area[i] = current_min_area;
    }
    
    std::vector<MaxRectsBoard> boards;
    std::vector<Placement> placements(N);
    
    for (int i = 0; i < N; i++) {
        const Rect& R = rects[i];
        long long best_score = 2e18;
        int best_board = -1;
        FreeRect best_fr = {0,0,0,0,0};
        bool best_rotated = false;
        
        int req_side = (i + 1 < N) ? suffix_min_side[i + 1] : 0;
        long long req_area = (i + 1 < N) ? suffix_min_area[i + 1] : 0;
        
        for (int b = 0; b < (int)boards.size(); b++) {
            const MaxRectsBoard& brd = boards[b];
            if (!((brd.max_w >= R.w && brd.max_h >= R.h) || (brd.max_w >= R.h && brd.max_h >= R.w))) {
                continue;
            }
            for (const auto& fr : brd.free_rects) {
                if (R.w <= fr.w && R.h <= fr.h) {
                    long long score = get_score(heur.score_type, fr, R.w, R.h);
                    if (score < best_score) {
                        best_score = score;
                        best_board = b;
                        best_fr = fr;
                        best_rotated = false;
                    }
                }
                if (R.h <= fr.w && R.w <= fr.h) {
                    long long score = get_score(heur.score_type, fr, R.h, R.w);
                    if (score < best_score) {
                        best_score = score;
                        best_board = b;
                        best_fr = fr;
                        best_rotated = true;
                    }
                }
            }
        }
        
        if (best_board == -1) {
            MaxRectsBoard nb;
            nb.free_rects.push_back({(int)boards.size(), 0, 0, W, H});
            nb.max_w = W;
            nb.max_h = H;
            boards.push_back(nb);
            best_board = (int)boards.size() - 1;
            best_fr = boards.back().free_rects[0];
            
            long long s_norm = (R.w <= W && R.h <= H) ? get_score(heur.score_type, best_fr, R.w, R.h) : 2e18;
            long long s_rot = (R.h <= W && R.w <= H) ? get_score(heur.score_type, best_fr, R.h, R.w) : 2e18;
            best_rotated = s_rot < s_norm;
        }
        
        int pw = best_rotated ? R.h : R.w;
        int ph = best_rotated ? R.w : R.h;
        
        placements[R.id] = {best_board, best_fr.x, best_fr.y, best_rotated ? 1 : 0};
        
        split_and_prune_maxrects(boards[best_board], best_fr.x, best_fr.y, pw, ph, req_side, req_area);
    }
    
    return {boards.size(), placements};
}

std::pair<int, std::vector<Placement>> solve_guillotine(
    int N, int W, int H, const std::vector<Rect>& original_rects,
    const Heuristic& heur) 
{
    std::vector<Rect> rects = original_rects;
    std::sort(rects.begin(), rects.end(), get_sort_lambda(heur.sort_type));
    
    std::vector<int> suffix_min_side(N);
    std::vector<long long> suffix_min_area(N);
    int current_min_side = 2e9;
    long long current_min_area = 2e18;
    for (int i = N - 1; i >= 0; i--) {
        current_min_side = std::min(current_min_side, std::min(rects[i].w, rects[i].h));
        current_min_area = std::min(current_min_area, (long long)rects[i].w * rects[i].h);
        suffix_min_side[i] = current_min_side;
        suffix_min_area[i] = current_min_area;
    }
    
    std::vector<FreeRect> free_rects;
    std::vector<Placement> placements(N);
    int num_boards = 0;
    
    for (int i = 0; i < N; i++) {
        const Rect& R = rects[i];
        long long best_score = 2e18;
        int best_fr_idx = -1;
        bool best_rotated = false;
        
        int req_side = (i + 1 < N) ? suffix_min_side[i + 1] : 0;
        long long req_area = (i + 1 < N) ? suffix_min_area[i + 1] : 0;
        
        for (int j = 0; j < (int)free_rects.size(); j++) {
            const FreeRect& fr = free_rects[j];
            if (R.w <= fr.w && R.h <= fr.h) {
                long long score = get_score(heur.score_type, fr, R.w, R.h);
                if (score < best_score) {
                    best_score = score;
                    best_fr_idx = j;
                    best_rotated = false;
                }
            }
            if (R.h <= fr.w && R.w <= fr.h) {
                long long score = get_score(heur.score_type, fr, R.h, R.w);
                if (score < best_score) {
                    best_score = score;
                    best_fr_idx = j;
                    best_rotated = true;
                }
            }
        }
        
        auto add_free_rect = [&](int board, int x, int y, int w, int h) {
            if (w <= 0 || h <= 0) return;
            if (std::max(w, h) < req_side) return;
            if ((long long)w * h < req_area) return;
            free_rects.push_back({board, x, y, w, h});
        };
        
        if (best_fr_idx == -1) {
            int new_board = num_boards++;
            FreeRect fr = {new_board, 0, 0, W, H};
            
            long long s_norm = (R.w <= W && R.h <= H) ? get_score(heur.score_type, fr, R.w, R.h) : 2e18;
            long long s_rot = (R.h <= W && R.w <= H) ? get_score(heur.score_type, fr, R.h, R.w) : 2e18;
            best_rotated = s_rot < s_norm;
            
            int pw = best_rotated ? R.h : R.w;
            int ph = best_rotated ? R.w : R.h;
            
            placements[R.id] = {new_board, 0, 0, best_rotated ? 1 : 0};
            
            int split_type = heur.split_type;
            bool split_horiz = false;
            int fw = W;
            int fh = H;
            
            if (split_type == SPLIT_MAXAS) {
                long long horiz_max = std::max((long long)fw * (fh - ph), (long long)(fw - pw) * ph);
                long long vert_max = std::max((long long)pw * (fh - ph), (long long)(fw - pw) * fh);
                split_horiz = horiz_max > vert_max;
            } else if (split_type == SPLIT_MINAS) {
                long long horiz_max = std::max((long long)fw * (fh - ph), (long long)(fw - pw) * ph);
                long long vert_max = std::max((long long)pw * (fh - ph), (long long)(fw - pw) * fh);
                split_horiz = horiz_max < vert_max;
            } else if (split_type == SPLIT_SLAS) {
                split_horiz = (fw - pw) < (fh - ph);
            } else if (split_type == SPLIT_LLAS) {
                split_horiz = (fw - pw) > (fh - ph);
            }
            
            if (split_horiz) {
                add_free_rect(new_board, 0, ph, fw, fh - ph);
                add_free_rect(new_board, pw, 0, fw - pw, ph);
            } else {
                add_free_rect(new_board, 0, ph, pw, fh - ph);
                add_free_rect(new_board, pw, 0, fw - pw, fh);
            }
        } else {
            FreeRect fr = free_rects[best_fr_idx];
            free_rects[best_fr_idx] = free_rects.back();
            free_rects.pop_back();
            
            int pw = best_rotated ? R.h : R.w;
            int ph = best_rotated ? R.w : R.h;
            
            placements[R.id] = {fr.board_index, fr.x, fr.y, best_rotated ? 1 : 0};
            
            int split_type = heur.split_type;
            bool split_horiz = false;
            int fw = fr.w;
            int fh = fr.h;
            
            if (split_type == SPLIT_MAXAS) {
                long long horiz_max = std::max((long long)fw * (fh - ph), (long long)(fw - pw) * ph);
                long long vert_max = std::max((long long)pw * (fh - ph), (long long)(fw - pw) * fh);
                split_horiz = horiz_max > vert_max;
            } else if (split_type == SPLIT_MINAS) {
                long long horiz_max = std::max((long long)fw * (fh - ph), (long long)(fw - pw) * ph);
                long long vert_max = std::max((long long)pw * (fh - ph), (long long)(fw - pw) * fh);
                split_horiz = horiz_max < vert_max;
            } else if (split_type == SPLIT_SLAS) {
                split_horiz = (fw - pw) < (fh - ph);
            } else if (split_type == SPLIT_LLAS) {
                split_horiz = (fw - pw) > (fh - ph);
            }
            
            if (split_horiz) {
                add_free_rect(fr.board_index, fr.x, fr.y + ph, fw, fh - ph);
                add_free_rect(fr.board_index, fr.x + pw, fr.y, fw - pw, ph);
            } else {
                add_free_rect(fr.board_index, fr.x, fr.y + ph, pw, fh - ph);
                add_free_rect(fr.board_index, fr.x + pw, fr.y, fw - pw, fh);
            }
        }
    }
    
    return {num_boards, placements};
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    auto start_time = std::chrono::steady_clock::now();
    
    int N, W, H;
    if (!(std::cin >> N >> W >> H)) return 0;
    
    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        rects[i].id = i;
        std::cin >> rects[i].w >> rects[i].h;
    }
    
    std::vector<Heuristic> all_heuristics;
    auto add_heur = [&](int algo, int sort, int score, int split) {
        for (const auto& h : all_heuristics) {
            if (h.algo == algo && h.sort_type == sort && h.score_type == score && h.split_type == split) return;
        }
        all_heuristics.push_back({algo, sort, score, split});
    };

    add_heur(ALGO_MAXRECTS, SORT_AREA_DESC, SCORE_BAF, 0);
    add_heur(ALGO_MAXRECTS, SORT_AREA_DESC, SCORE_BSSF, 0);
    add_heur(ALGO_MAXRECTS, SORT_MAX_DIM_DESC, SCORE_BAF, 0);
    add_heur(ALGO_GUILLOTINE, SORT_AREA_DESC, SCORE_BAF, SPLIT_MAXAS);
    add_heur(ALGO_GUILLOTINE, SORT_MAX_DIM_DESC, SCORE_BSSF, SPLIT_SLAS);
    add_heur(ALGO_MAXRECTS, SORT_AREA_DESC, SCORE_BLF, 0);

    for (int algo : {ALGO_MAXRECTS, ALGO_GUILLOTINE}) {
        for (int sort_type : {SORT_AREA_DESC, SORT_MAX_DIM_DESC, SORT_PERIMETER_DESC, SORT_W_THEN_H_DESC}) {
            for (int score_type : {SCORE_BAF, SCORE_BSSF, SCORE_BLSF, SCORE_BLF}) {
                if (algo == ALGO_GUILLOTINE && score_type == SCORE_BLF) continue;
                if (algo == ALGO_MAXRECTS) {
                    add_heur(ALGO_MAXRECTS, sort_type, score_type, 0);
                } else {
                    for (int split_type : {SPLIT_MAXAS, SPLIT_MINAS, SPLIT_SLAS, SPLIT_LLAS}) {
                        add_heur(ALGO_GUILLOTINE, sort_type, score_type, split_type);
                    }
                }
            }
        }
    }

    int best_boards = 2e9;
    std::vector<Placement> best_placements;

    for (const auto& heur : all_heuristics) {
        std::pair<int, std::vector<Placement>> res;
        if (heur.algo == ALGO_MAXRECTS) {
            res = solve_maxrects(N, W, H, rects, heur);
        } else {
            res = solve_guillotine(N, W, H, rects, heur);
        }
        
        if (res.first < best_boards) {
            best_boards = res.first;
            best_placements = std::move(res.second);
        }
        
        auto now = std::chrono::steady_clock::now();
        if (std::chrono::duration<double>(now - start_time).count() > 28.0) {
            break;
        }
    }

    cout << best_boards << "\n";
    for (int i = 0; i < N; i++) {
        cout << best_placements[i].board_index << " "
             << best_placements[i].x << " "
             << best_placements[i].y << " "
             << best_placements[i].rotated << "\n";
    }

    return 0;
}
