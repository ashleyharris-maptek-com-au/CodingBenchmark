#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>
#include <set>
#include <iomanip>

struct Point {
    double x, y, z;
    Point operator-(const Point& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Point operator+(const Point& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Point operator*(double s) const { return {x * s, y * s, z * s}; }
};

double dot(const Point& a, const Point& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
Point cross(const Point& a, const Point& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
double norm(const Point& a) { return std::sqrt(dot(a, a)); }
Point normalize(const Point& a) { double n = norm(a); return {a.x/n, a.y/n, a.z/n}; }

struct Face {
    int a, b, c;
    Point n;
    double d;
};

std::vector<Face> build_hull(std::vector<Point>& P) {
    int n = P.size();
    if(n < 4) return {};
    
    std::mt19937 rng(42);
    std::shuffle(P.begin(), P.end(), rng);
    
    std::vector<Face> hull;
    int a=0, b=1, c=2, d_idx=3;
    while(c < n && norm(cross(P[b]-P[a], P[c]-P[a])) < 1e-5) c++;
    if(c == n) return {}; 
    std::swap(P[2], P[c]);
    
    Point n_abc = cross(P[1]-P[0], P[2]-P[0]);
    while(d_idx < n && std::abs(dot(n_abc, P[d_idx]-P[0])) < 1e-5) d_idx++;
    if(d_idx == n) return {}; 
    std::swap(P[3], P[d_idx]);
    
    if (dot(n_abc, P[3]-P[0]) > 0) {
        std::swap(P[1], P[2]);
    }
    
    auto make_face = [&](int v1, int v2, int v3) -> Face {
        Point normal = cross(P[v2]-P[v1], P[v3]-P[v1]);
        double len = norm(normal);
        if(len > 1e-12) normal = normal * (1.0 / len);
        return {v1, v2, v3, normal, dot(normal, P[v1])};
    };
    
    hull.push_back(make_face(0, 1, 2));
    hull.push_back(make_face(0, 2, 3));
    hull.push_back(make_face(0, 3, 1));
    hull.push_back(make_face(1, 3, 2));
    
    for (int i = 4; i < n; ++i) {
        std::vector<bool> visible(hull.size(), false);
        bool any_visible = false;
        for(size_t j=0; j<hull.size(); ++j) {
            if (dot(hull[j].n, P[i]) > hull[j].d + 1e-5) {
                visible[j] = true;
                any_visible = true;
            }
        }
        if (!any_visible) continue;
        
        std::set<std::pair<int,int>> vis_edges;
        for(size_t j=0; j<hull.size(); ++j) {
            if (visible[j]) {
                vis_edges.insert({hull[j].a, hull[j].b});
                vis_edges.insert({hull[j].b, hull[j].c});
                vis_edges.insert({hull[j].c, hull[j].a});
            }
        }
        
        std::vector<Face> next_hull;
        for(size_t j=0; j<hull.size(); ++j) {
            if (!visible[j]) {
                next_hull.push_back(hull[j]);
            }
        }
        
        for (auto e : vis_edges) {
            if (vis_edges.find({e.second, e.first}) == vis_edges.end()) {
                next_hull.push_back(make_face(e.first, e.second, i));
            }
        }
        hull = next_hull;
    }
    return hull;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    int V;
    double e;
    if (!(std::cin >> V >> e)) return 0;
    
    std::vector<Point> pts(V);
    for(int i=0; i<V; ++i) {
        std::cin >> pts[i].x >> pts[i].y >> pts[i].z;
    }
    
    std::vector<Face> hull = build_hull(pts);
    if(hull.empty()) {
        std::cout << "0\n";
        return 0;
    }
    
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;
    for(auto& p : pts) {
        min_x = std::min(min_x, p.x); max_x = std::max(max_x, p.x);
        min_y = std::min(min_y, p.y); max_y = std::max(max_y, p.y);
        min_z = std::min(min_z, p.z); max_z = std::max(max_z, p.z);
    }
    
    double R = e * std::sqrt(3.0 / 8.0);
    double S = e * 1.000001 / std::sqrt(2.0);
    
    int i_min = std::floor(min_x / S) - 1;
    int i_max = std::ceil(max_x / S) + 1;
    int j_min = std::floor(min_y / S) - 1;
    int j_max = std::ceil(max_y / S) + 1;
    int k_min = std::floor(min_z / S) - 1;
    int k_max = std::ceil(max_z / S) + 1;
    
    long long dx = std::max(0, i_max - i_min);
    long long dy = std::max(0, j_max - j_min);
    long long dz = std::max(0, k_max - k_min);
    long long est_points = (dx * dy * dz) / 2;
    
    int num_attempts = 1;
    if (est_points < 1000) num_attempts = 200;
    else if (est_points < 10000) num_attempts = 50;
    else if (est_points < 100000) num_attempts = 10;
    else num_attempts = 1;

    int max_K = -1;
    std::vector<Point> best_centers;

    std::mt19937 rng(1337);
    std::uniform_real_distribution<double> dist(0.0, 2.0 * S);

    for(int attempt = 0; attempt < num_attempts; ++attempt) {
        double ox = dist(rng);
        double oy = dist(rng);
        double oz = dist(rng);
        
        std::vector<Point> current;
        for(int i = i_min; i <= i_max; ++i) {
            for(int j = j_min; j <= j_max; ++j) {
                for(int k = k_min; k <= k_max; ++k) {
                    if ( std::abs(i + j + k) % 2 != 0 ) continue;
                    Point C = {i * S + ox, j * S + oy, k * S + oz};
                    
                    bool ok = true;
                    for(const auto& face : hull) {
                        if (dot(face.n, C) > face.d - R - 1e-7) {
                            ok = false;
                            break;
                        }
                    }
                    if (ok) {
                        current.push_back(C);
                    }
                }
            }
        }
        if ((int)current.size() > max_K) {
            max_K = current.size();
            best_centers = current;
        }
    }
    
    std::cout << best_centers.size() << "\n";
    for(auto& c : best_centers) {
        std::cout << std::fixed << std::setprecision(6) 
                  << c.x << " " << c.y << " " << c.z << " 0.0 0.0 0.0\n";
    }
    
    return 0;
}
