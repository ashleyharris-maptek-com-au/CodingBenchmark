use std::io::{self, Read, Write};
use std::collections::BinaryHeap;
use std::cmp::Ordering;

struct Scanner<R: Read> {
    reader: R,
    buffer: Vec<u8>,
    pos: usize,
    len: usize,
}

impl<R: Read> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buffer: vec![0; 1024 * 1024],
            pos: 0,
            len: 0,
        }
    }
    
    fn fill_buffer(&mut self) {
        self.len = self.reader.read(&mut self.buffer).unwrap_or(0);
        self.pos = 0;
    }

    fn skip_whitespace(&mut self) {
        loop {
            while self.pos < self.len && self.buffer[self.pos].is_ascii_whitespace() {
                self.pos += 1;
            }
            if self.pos < self.len {
                break;
            }
            self.fill_buffer();
            if self.pos >= self.len {
                break;
            }
        }
    }

    fn next_bytes(&mut self) -> Option<Vec<u8>> {
        self.skip_whitespace();
        if self.pos >= self.len {
            return None;
        }
        let mut res = Vec::new();
        loop {
            let start = self.pos;
            while self.pos < self.len && !self.buffer[self.pos].is_ascii_whitespace() {
                self.pos += 1;
            }
            res.extend_from_slice(&self.buffer[start..self.pos]);
            if self.pos < self.len {
                break;
            } else {
                self.fill_buffer();
                if self.pos >= self.len {
                    break;
                }
            }
        }
        Some(res)
    }
    
    fn next_string(&mut self) -> Option<String> {
        let bytes = self.next_bytes()?;
        Some(String::from_utf8_lossy(&bytes).into_owned())
    }
    
    fn next_f64(&mut self) -> Option<f64> {
        let bytes = self.next_bytes()?;
        let s = std::str::from_utf8(&bytes).ok()?;
        s.parse().ok()
    }

    fn next_usize(&mut self) -> Option<usize> {
        let bytes = self.next_bytes()?;
        let s = std::str::from_utf8(&bytes).ok()?;
        s.parse().ok()
    }
}

#[derive(Clone)]
struct Sample {
    hole_idx: u32,
    sample_idx: u32,
    pos: [f64; 3],
    props: Vec<f64>,
}

#[derive(Clone, Copy)]
struct PosNode {
    sample_id: u32,
    pos: [f64; 3],
}

#[derive(Clone, Copy, Debug)]
struct KnnResult {
    dist2: f64,
    sample_id: u32,
}

impl PartialEq for KnnResult {
    fn eq(&self, other: &Self) -> bool {
        self.dist2 == other.dist2
    }
}
impl Eq for KnnResult {}
impl PartialOrd for KnnResult {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        self.dist2.partial_cmp(&other.dist2)
    }
}
impl Ord for KnnResult {
    fn cmp(&self, other: &Self) -> Ordering {
        self.partial_cmp(other).unwrap_or(Ordering::Equal)
    }
}

struct Anomaly {
    sample_id: usize,
    prop_idx: usize,
    confidence: f64,
}

fn build_implicit_kdtree_parallel(points: &mut [PosNode], depth: usize, max_threads: usize) {
    if points.len() <= 1 {
        return;
    }
    let axis = depth % 3;
    let mid = points.len() / 2;
    
    points.select_nth_unstable_by(mid, |a, b| a.pos[axis].partial_cmp(&b.pos[axis]).unwrap_or(Ordering::Equal));
    
    let (left, right) = points.split_at_mut(mid);
    let right = &mut right[1..];
    
    if max_threads > 1 && points.len() > 10_000 {
        std::thread::scope(|s| {
            s.spawn(|| build_implicit_kdtree_parallel(left, depth + 1, max_threads / 2));
            s.spawn(|| build_implicit_kdtree_parallel(right, depth + 1, max_threads - max_threads / 2));
        });
    } else {
        build_implicit_kdtree(left, depth + 1);
        build_implicit_kdtree(right, depth + 1);
    }
}

fn build_implicit_kdtree(points: &mut [PosNode], depth: usize) {
    if points.len() <= 1 {
        return;
    }
    let axis = depth % 3;
    let mid = points.len() / 2;
    
    points.select_nth_unstable_by(mid, |a, b| a.pos[axis].partial_cmp(&b.pos[axis]).unwrap_or(Ordering::Equal));
    
    let (left, right) = points.split_at_mut(mid);
    let right = &mut right[1..];
    
    build_implicit_kdtree(left, depth + 1);
    build_implicit_kdtree(right, depth + 1);
}

fn search_knn(
    points: &[PosNode],
    target: &[f64; 3],
    target_id: u32,
    k: usize,
    depth: usize,
    heap: &mut BinaryHeap<KnnResult>,
) {
    if points.is_empty() {
        return;
    }
    let axis = depth % 3;
    let mid = points.len() / 2;
    let point = &points[mid];
    
    if point.sample_id != target_id {
        let dx = point.pos[0] - target[0];
        let dy = point.pos[1] - target[1];
        let dz = point.pos[2] - target[2];
        let d2 = dx*dx + dy*dy + dz*dz;
        
        if heap.len() < k {
            heap.push(KnnResult { dist2: d2, sample_id: point.sample_id });
        } else if d2 < heap.peek().unwrap().dist2 {
            heap.pop();
            heap.push(KnnResult { dist2: d2, sample_id: point.sample_id });
        }
    }
    
    let diff = target[axis] - point.pos[axis];
    
    let (first, second) = if diff < 0.0 {
        (&points[..mid], &points[mid + 1..])
    } else {
        (&points[mid + 1..], &points[..mid])
    };
    
    search_knn(first, target, target_id, k, depth + 1, heap);
    
    if heap.len() < k || diff * diff < heap.peek().unwrap().dist2 {
        search_knn(second, target, target_id, k, depth + 1, heap);
    }
}

fn is_valid(v: f64) -> bool {
    !v.is_nan() && v > -900.0 && (v + 99.0).abs() > 1e-4 && (v + 99.99).abs() > 1e-4 && (v + 999.0).abs() > 1e-4
}

fn main() {
    let stdin = io::stdin();
    let mut scanner = Scanner::new(stdin.lock());
    
    let num_holes = match scanner.next_usize() {
        Some(n) => n,
        None => return,
    };
    let num_properties = match scanner.next_usize() {
        Some(n) => n,
        None => return,
    };
    
    let mut property_names = Vec::with_capacity(num_properties);
    for _ in 0..num_properties {
        property_names.push(scanner.next_string().unwrap_or_default());
    }
    
    let mut holes = Vec::with_capacity(num_holes);
    let mut samples = Vec::new();
    
    for hole_idx in 0..num_holes {
        let hole_id = match scanner.next_string() {
            Some(id) => id,
            None => break,
        };
        holes.push(hole_id);
        
        let start_x = scanner.next_f64().unwrap_or(0.0);
        let start_y = scanner.next_f64().unwrap_or(0.0);
        let start_z = scanner.next_f64().unwrap_or(0.0);
        let dir_x = scanner.next_f64().unwrap_or(0.0);
        let dir_y = scanner.next_f64().unwrap_or(0.0);
        let dir_z = scanner.next_f64().unwrap_or(1.0);
        let _length = scanner.next_f64().unwrap_or(0.0);
        let num_samples = scanner.next_usize().unwrap_or(0);
        
        let len = (dir_x*dir_x + dir_y*dir_y + dir_z*dir_z).sqrt();
        let (dx, dy, dz) = if len > 1e-6 {
            (dir_x/len, dir_y/len, dir_z/len)
        } else {
            (dir_x, dir_y, dir_z)
        };
        
        for sample_idx in 0..num_samples {
            let depth = scanner.next_f64().unwrap_or(0.0);
            let mut props = Vec::with_capacity(num_properties);
            for _ in 0..num_properties {
                let v = scanner.next_f64().unwrap_or(f64::NAN);
                props.push(v);
            }
            
            samples.push(Sample {
                hole_idx: hole_idx as u32,
                sample_idx: sample_idx as u32,
                pos: [
                    start_x + dx * depth,
                    start_y + dy * depth,
                    start_z + dz * depth,
                ],
                props,
            });
        }
    }
    
    if samples.is_empty() {
        println!("0");
        return;
    }
    
    let mut pos_nodes: Vec<PosNode> = samples.iter().enumerate().map(|(i, s)| PosNode {
        sample_id: i as u32,
        pos: s.pos,
    }).collect();
    
    let num_threads = std::thread::available_parallelism().map(|n| n.get()).unwrap_or(12);
    
    build_implicit_kdtree_parallel(&mut pos_nodes, 0, num_threads);
    
    let mut global_mads = Vec::with_capacity(num_properties);
    for p in 0..num_properties {
        let mut valid_values = Vec::with_capacity(samples.len());
        for s in &samples {
            let v = s.props[p];
            if is_valid(v) {
                valid_values.push(v);
            }
        }
        if valid_values.is_empty() {
            global_mads.push(1.0);
            continue;
        }
        valid_values.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        let median = valid_values[valid_values.len() / 2];
        
        let mut abs_devs: Vec<f64> = valid_values.iter().map(|&v| (v - median).abs()).collect();
        abs_devs.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
        let mad = abs_devs[abs_devs.len() / 2];
        
        global_mads.push(mad);
    }
    
    let num_points = samples.len();
    let chunk_size = (num_points + num_threads - 1) / num_threads;
    
    let mut all_anomalies = std::thread::scope(|s| {
        let mut handles = Vec::new();
        for chunk_start in (0..num_points).step_by(chunk_size) {
            let chunk_end = (chunk_start + chunk_size).min(num_points);
            let pos_nodes_ref = &pos_nodes;
            let samples_ref = &samples;
            let global_mads_ref = &global_mads;
            
            let handle = s.spawn(move || {
                let mut local_anomalies = Vec::new();
                let mut heap = BinaryHeap::with_capacity(20);
                
                for i in chunk_start..chunk_end {
                    let target_pos = samples_ref[i].pos;
                    heap.clear();
                    
                    search_knn(pos_nodes_ref, &target_pos, i as u32, 15, 0, &mut heap);
                    
                    let mut neighbors = Vec::with_capacity(heap.len());
                    for res in heap.iter() {
                        neighbors.push(res.sample_id as usize);
                    }
                    
                    for p in 0..num_properties {
                        let v_i = samples_ref[i].props[p];
                        if !is_valid(v_i) {
                            continue;
                        }
                        
                        let mut valid_neighbors = Vec::with_capacity(neighbors.len());
                        for &n_idx in &neighbors {
                            let v_n = samples_ref[n_idx].props[p];
                            if is_valid(v_n) {
                                valid_neighbors.push(v_n);
                            }
                        }
                        
                        if valid_neighbors.len() < 5 {
                            continue;
                        }
                        
                        valid_neighbors.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
                        let median = valid_neighbors[valid_neighbors.len() / 2];
                        
                        let mut abs_devs: Vec<f64> = valid_neighbors.iter().map(|&v| (v - median).abs()).collect();
                        abs_devs.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
                        let mad = abs_devs[abs_devs.len() / 2];
                        
                        let local_robust_std = mad * 1.4826 + 0.02 * global_mads_ref[p] + 1e-5;
                        let confidence = (v_i - median).abs() / local_robust_std;
                        
                        if confidence > 5.0 {
                            local_anomalies.push(Anomaly {
                                sample_id: i,
                                prop_idx: p,
                                confidence,
                            });
                        }
                    }
                }
                local_anomalies
            });
            handles.push(handle);
        }
        
        let mut combined = Vec::new();
        for handle in handles {
            combined.extend(handle.join().unwrap());
        }
        combined
    });
    
    all_anomalies.sort_unstable_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal));
    
    let max_out = (samples.len() / 100).max(100).min(10000);
    let m = all_anomalies.len().min(max_out);
    
    let stdout = io::stdout();
    let mut handle = io::BufWriter::new(stdout.lock());
    
    writeln!(handle, "{}", m).unwrap();
    for i in 0..m {
        let a = &all_anomalies[i];
        let sample = &samples[a.sample_id];
        let hole_id = &holes[sample.hole_idx as usize];
        let prop_name = &property_names[a.prop_idx];
        writeln!(handle, "{} {} {} {:.2}", hole_id, sample.sample_idx, prop_name, a.confidence).unwrap();
    }
}
