using namespace std;

// Fast I/O scanner for fast reading of large inputs
class FastScanner {
    static const int BUFFER_SIZE = 1 << 16;
    char buffer[BUFFER_SIZE];
    int pos = 0, len = 0;
    
    char next_char() {
        if (pos >= len) {
            pos = 0;
            len = (int)std::fread(buffer, 1, BUFFER_SIZE, stdin);
            if (len == 0) return EOF;
        }
        return buffer[pos++];
    }
public:
    int next_int() {
        char c = next_char();
        while (c <= 32) {
            if (c == EOF) return -1;
            c = next_char();
        }
        int res = 0;
        while (c > 32) {
            res = res * 10 + (c - '0');
            c = next_char();
        }
        return res;
    }
};

// Fast I/O writer
char out_buf[1 << 16];
int out_pos = 0;
void write_char(char c) {
    if (out_pos == (1 << 16)) {
        std::fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
    out_buf[out_pos++] = c;
}
void write_int(int x) {
    if (x == 0) {
        write_char('0');
        return;
    }
    char temp[12];
    int len = 0;
    while (x > 0) {
        temp[len++] = (x % 10) + '0';
        x /= 10;
    }
    for (int i = len - 1; i >= 0; --i) {
        write_char(temp[i]);
    }
}
void flush_out() {
    if (out_pos > 0) {
        std::fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

// Memory-efficient bitset for tracking used colors per vertex (supports up to 128 colors)
struct ColorSet {
    uint64_t lo = 0;
    uint64_t hi = 0;
    
    void set(int c) {
        if (c < 64) lo |= (1ULL << c);
        else hi |= (1ULL << (c - 64));
    }
    
    bool test(int c) const {
        if (c < 64) return (lo & (1ULL << c)) != 0;
        return (hi & (1ULL << (c - 64))) != 0;
    }
    
    int popcount() const {
        return std::popcount(lo) + std::popcount(hi);
    }
    
    int get_lowest_available(int K) const {
        uint64_t avail_lo = ~lo;
        if (avail_lo != 0) {
            int c = std::countr_zero(avail_lo);
            if (c < K && c < 64) {
                return c;
            }
        }
        if (K > 64) {
            uint64_t avail_hi = ~hi;
            if (avail_hi != 0) {
                int c = std::countr_zero(avail_hi) + 64;
                if (c < K) {
                    return c;
                }
            }
        }
        return -1; // All K colors are used
    }
};

// Priority Queue node for DSatur
struct Node {
    int sat;
    int deg;
    int v;
    bool operator<(const Node& other) const {
        if (sat != other.sat) return sat < other.sat;
        if (deg != other.deg) return deg < other.deg;
        return v < other.v; // Stable tie-breaking
    }
};

int main() {
    auto start_time = chrono::steady_clock::now();
    
    FastScanner scanner;
    int V = scanner.next_int();
    if (V == -1) return 0; // Empty input
    int E = scanner.next_int();
    int K = scanner.next_int();
    
    // Read edges and remove duplicates/self-loops
    vector<pair<int, int>> edges;
    edges.reserve(E);
    for (int i = 0; i < E; ++i) {
        int u = scanner.next_int();
        int v = scanner.next_int();
        if (u > v) swap(u, v);
        if (u != v) {
            edges.push_back({u, v});
        }
    }
    
    sort(edges.begin(), edges.end());
    edges.erase(unique(edges.begin(), edges.end()), edges.end());
    int unique_E = edges.size();
    
    // Build CSR format adjacency list
    vector<int> deg(V, 0);
    for (const auto& e : edges) {
        deg[e.first]++;
        deg[e.second]++;
    }
    
    vector<int> head(V + 1, 0);
    for (int i = 0; i < V; ++i) {
        head[i + 1] = head[i] + deg[i];
    }
    
    vector<int> to(2 * unique_E);
    vector<int> cur_head = head;
    for (const auto& e : edges) {
        to[cur_head[e.first]++] = e.second;
        to[cur_head[e.second]++] = e.first;
    }
    
    edges.clear();
    edges.shrink_to_fit();
    
    mt19937 rng(1337);
    
    // --- Phase 1: DSatur Heuristic ---
    vector<ColorSet> used_colors(V);
    vector<int> color(V, -1);
    priority_queue<Node> pq;
    
    for (int i = 0; i < V; ++i) {
        pq.push({0, deg[i], i});
    }
    
    int colored_count = 0;
    while (colored_count < V) {
        Node node = pq.top();
        pq.pop();
        
        int u = node.v;
        if (color[u] != -1) continue;
        
        int actual_sat = used_colors[u].popcount();
        if (node.sat != actual_sat) continue; // Lazy deletion
        
        int c = used_colors[u].get_lowest_available(K);
        
        if (c == -1) {
            // Failed to find a valid color under K. Pick the one that minimizes conflicts.
            vector<int> conflict_count(K, 0);
            for (int i = head[u]; i < head[u+1]; ++i) {
                int v = to[i];
                if (color[v] != -1) {
                    conflict_count[color[v]]++;
                }
            }
            int min_conf = 1e9;
            vector<int> best_c_list;
            for (int i = 0; i < K; ++i) {
                if (conflict_count[i] < min_conf) {
                    min_conf = conflict_count[i];
                    best_c_list.clear();
                    best_c_list.push_back(i);
                } else if (conflict_count[i] == min_conf) {
                    best_c_list.push_back(i);
                }
            }
            c = best_c_list[rng() % best_c_list.size()];
        }
        
        color[u] = c;
        colored_count++;
        
        // Update neighbors
        for (int i = head[u]; i < head[u+1]; ++i) {
            int v = to[i];
            if (color[v] == -1) {
                if (!used_colors[v].test(c)) {
                    used_colors[v].set(c);
                    pq.push({used_colors[v].popcount(), deg[v], v});
                }
            }
        }
    }
    
    // --- Phase 2: Local Search (TabuCol) ---
    int total_conflicts = 0;
    vector<int> gamma((size_t)V * K, 0);
    vector<int> conflicting_vertices;
    vector<int> conflicting_pos(V, -1);

    auto add_conflict = [&](int u) {
        if (conflicting_pos[u] == -1) {
            conflicting_pos[u] = conflicting_vertices.size();
            conflicting_vertices.push_back(u);
        }
    };

    auto remove_conflict = [&](int u) {
        if (conflicting_pos[u] != -1) {
            int pos = conflicting_pos[u];
            int last = conflicting_vertices.back();
            conflicting_vertices[pos] = last;
            conflicting_pos[last] = pos;
            conflicting_vertices.pop_back();
            conflicting_pos[u] = -1;
        }
    };

    // Initialize conflict structures
    for (int u = 0; u < V; ++u) {
        int cu = color[u];
        for (int i = head[u]; i < head[u+1]; ++i) {
            int v = to[i];
            int cv = color[v];
            gamma[(size_t)u * K + cv]++;
            if (u < v && cu == cv) {
                total_conflicts++;
            }
        }
    }

    for (int u = 0; u < V; ++u) {
        if (gamma[(size_t)u * K + color[u]] > 0) {
            add_conflict(u);
        }
    }
    
    int best_total_conflicts = total_conflicts;
    vector<int> best_color_assignment = color;
    
    if (total_conflicts > 0) {
        vector<int> tabu_tenure((size_t)V * K, 0);
        int iteration = 0;
        int no_improve = 0;
        int restart_threshold = max(500000, min(5000000, V * 5));
        
        while (total_conflicts > 0) {
            iteration++;
            
            // Check time limits occasionally
            if ((iteration & 1023) == 0) {
                auto now = chrono::steady_clock::now();
                double elapsed = chrono::duration<double>(now - start_time).count();
                if (elapsed > 290.0) { // 4.83 minutes
                    break;
                }
            }
            
            if (conflicting_vertices.empty()) break;
            
            // Pick a random conflicting vertex
            int rand_idx = rng() % conflicting_vertices.size();
            int v = conflicting_vertices[rand_idx];
            
            int best_c = -1;
            int best_delta = 1e9;
            vector<int> best_colors;
            int current_conflict = gamma[(size_t)v * K + color[v]];
            
            // Evaluate all available color moves
            for (int c = 0; c < K; ++c) {
                if (c == color[v]) continue;
                int delta = gamma[(size_t)v * K + c] - current_conflict;
                
                bool is_tabu = (iteration <= tabu_tenure[(size_t)v * K + c]);
                bool aspiration = (total_conflicts + delta < best_total_conflicts);
                
                if (!is_tabu || aspiration) {
                    if (delta < best_delta) {
                        best_delta = delta;
                        best_colors.clear();
                        best_colors.push_back(c);
                    } else if (delta == best_delta) {
                        best_colors.push_back(c);
                    }
                }
            }
            
            // If all moves are tabu, pick the best among all regardless of tabu status
            if (best_colors.empty()) {
                best_delta = 1e9;
                for (int c = 0; c < K; ++c) {
                    if (c == color[v]) continue;
                    int delta = gamma[(size_t)v * K + c] - current_conflict;
                    if (delta < best_delta) {
                        best_delta = delta;
                        best_colors.clear();
                        best_colors.push_back(c);
                    } else if (delta == best_delta) {
                        best_colors.push_back(c);
                    }
                }
            }
            
            // Perform the move
            int c_new = best_colors[rng() % best_colors.size()];
            int c_old = color[v];
            
            color[v] = c_new;
            
            for (int i = head[v]; i < head[v+1]; ++i) {
                int u = to[i];
                
                gamma[(size_t)u * K + c_old]--;
                if (color[u] == c_old) {
                    if (gamma[(size_t)u * K + c_old] == 0) remove_conflict(u);
                    total_conflicts--;
                }
                
                gamma[(size_t)u * K + c_new]++;
                if (color[u] == c_new) {
                    if (gamma[(size_t)u * K + c_new] == 1) add_conflict(u);
                    total_conflicts++;
                }
            }
            
            if (gamma[(size_t)v * K + c_new] > 0) {
                add_conflict(v);
            } else {
                remove_conflict(v);
            }
            
            // Update Tabu Tenure
            int A = (total_conflicts * 6) / 10;
            tabu_tenure[(size_t)v * K + c_old] = iteration + (rng() % 10) + A + 2;
            
            if (total_conflicts < best_total_conflicts) {
                best_total_conflicts = total_conflicts;
                best_color_assignment = color;
                no_improve = 0;
            } else {
                no_improve++;
            }
            
            // Soft restart if completely stuck
            if (no_improve > restart_threshold) {
                no_improve = 0;
                for (int u : conflicting_vertices) {
                    color[u] = rng() % K;
                }
                
                fill(gamma.begin(), gamma.end(), 0);
                fill(tabu_tenure.begin(), tabu_tenure.end(), 0);
                total_conflicts = 0;
                conflicting_vertices.clear();
                fill(conflicting_pos.begin(), conflicting_pos.end(), -1);
                
                for (int u = 0; u < V; ++u) {
                    int cu = color[u];
                    for (int i = head[u]; i < head[u+1]; ++i) {
                        int v = to[i];
                        int cv = color[v];
                        gamma[(size_t)u * K + cv]++;
                        if (u < v && cu == cv) {
                            total_conflicts++;
                        }
                    }
                }
                for (int u = 0; u < V; ++u) {
                    if (gamma[(size_t)u * K + color[u]] > 0) {
                        add_conflict(u);
                    }
                }
            }
        }
    }
    
    // --- Output Phase ---
    for (int i = 0; i < V; ++i) {
        write_int(best_color_assignment[i]);
        if (i < V - 1) write_char(' ');
        else write_char('\n');
    }
    flush_out();
    
    return 0;
}
