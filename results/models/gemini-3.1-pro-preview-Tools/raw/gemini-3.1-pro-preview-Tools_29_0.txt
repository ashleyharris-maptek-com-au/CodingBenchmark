using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static int num_vertices;
    static int num_edges;
    public static int[][] adj;

    public static int max_size = 0;
    public static int[] best_clique = new int[0];
    public static object lockObj = new object();

    public static Stopwatch sw = new Stopwatch();
    public static TimeSpan timeLimit = TimeSpan.FromSeconds(290);

    static void Main()
    {
        sw.Start();
        if (!ReadInput()) return;

        if (num_vertices == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int[] vert = CoreDecomposition();
        
        int[] rank = new int[num_vertices];
        for (int i = 0; i < num_vertices; i++) rank[vert[i]] = i;

        int[][] forward_adj = new int[num_vertices][];
        
        for (int i = 0; i < num_vertices; i++)
        {
            int v = vert[i];
            int f_deg = 0;
            for (int j = 0; j < adj[v].Length; j++)
            {
                if (rank[adj[v][j]] > i) f_deg++;
            }
            
            forward_adj[v] = new int[f_deg];
            int idx = 0;
            for (int j = 0; j < adj[v].Length; j++)
            {
                int u = adj[v][j];
                if (rank[u] > i) forward_adj[v][idx++] = u;
            }
        }

        RunHeuristic(vert, forward_adj);

        int[] process_order = new int[num_vertices];
        for (int i = 0; i < num_vertices; i++) process_order[i] = vert[num_vertices - 1 - i];

        ParallelOptions po = new ParallelOptions();
        po.MaxDegreeOfParallelism = Environment.ProcessorCount;
        
        CancellationTokenSource cts = new CancellationTokenSource();
        po.CancellationToken = cts.Token;
        
        Task.Run(() => {
            while (sw.Elapsed < timeLimit) {
                Thread.Sleep(50);
            }
            try { cts.Cancel(); } catch {}
        });

        try
        {
            Parallel.ForEach(process_order, po, 
                () => new SearchWorker(num_vertices),
                (v, state, worker) =>
                {
                    if (sw.Elapsed > timeLimit || state.ShouldExitCurrentIteration) 
                    {
                        state.Stop();
                        return worker;
                    }
                    
                    int[] P_initial = forward_adj[v];
                    int initial_size = P_initial.Length;
                    
                    int global_max = Volatile.Read(ref max_size);
                    if (initial_size + 1 <= global_max) return worker; 

                    worker.EnsureCapacity(initial_size);
                    
                    new Span<int>(P_initial).CopyTo(new Span<int>(worker.P_pool, 0, initial_size));
                    worker.C[0] = v;
                    
                    worker.Expand(0, 1, initial_size, 0);
                    
                    return worker;
                },
                (worker) => {}
            );
        }
        catch (OperationCanceledException) { }

        OutputResult();
    }

    static void RunHeuristic(int[] vert, int[][] forward_adj)
    {
        int limit = Math.Min(num_vertices, 2000);
        for (int i = num_vertices - 1; i >= num_vertices - limit; i--)
        {
            if (sw.Elapsed > timeLimit) break;
            
            int v = vert[i];
            int[] C = new int[forward_adj[v].Length + 1];
            C[0] = v;
            int c_size = 1;
            
            int[] candidates = forward_adj[v];
            int cands_size = candidates.Length;
            
            if (cands_size + 1 <= max_size) continue;
            
            for (int j = 0; j < cands_size; j++)
            {
                int u = candidates[j];
                bool ok = true;
                for (int k = 1; k < c_size; k++)
                {
                    if (Array.BinarySearch(adj[u], C[k]) < 0)
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok)
                {
                    C[c_size++] = u;
                }
            }
            
            lock (lockObj)
            {
                if (c_size > max_size)
                {
                    max_size = c_size;
                    best_clique = new int[c_size];
                    Array.Copy(C, best_clique, c_size);
                }
            }
        }
        
        // Also if max_size is still 0, we can set it to 1 and pick vert[0]
        if (max_size == 0 && num_vertices > 0)
        {
            max_size = 1;
            best_clique = new int[] { vert[0] };
        }
    }

    static int[] CoreDecomposition()
    {
        int n = num_vertices;
        int[] deg = new int[n];
        int max_deg = 0;
        for (int i = 0; i < n; i++)
        {
            deg[i] = adj[i].Length;
            if (deg[i] > max_deg) max_deg = deg[i];
        }

        int[] bin = new int[max_deg + 1];
        for (int i = 0; i < n; i++) bin[deg[i]]++;

        int start = 0;
        for (int d = 0; d <= max_deg; d++)
        {
            int num = bin[d];
            bin[d] = start;
            start += num;
        }

        int[] pos = new int[n];
        int[] vert = new int[n];
        for (int i = 0; i < n; i++)
        {
            pos[i] = bin[deg[i]];
            vert[pos[i]] = i;
            bin[deg[i]]++;
        }
        for (int d = max_deg; d > 0; d--) bin[d] = bin[d - 1];
        bin[0] = 0;

        for (int i = 0; i < n; i++)
        {
            int v = vert[i];
            for (int j = 0; j < adj[v].Length; j++)
            {
                int u = adj[v][j];
                if (deg[u] > deg[v])
                {
                    int du = deg[u];
                    int pu = pos[u];
                    int pw = bin[du];
                    int w = vert[pw];

                    if (u != w)
                    {
                        pos[u] = pw;
                        pos[w] = pu;
                        vert[pu] = w;
                        vert[pw] = u;
                    }
                    bin[du]++;
                    deg[u]--;
                }
            }
        }
        return vert;
    }

    class Scanner
    {
        Stream s;
        byte[] buf = new byte[8192 * 16];
        int head = 0;
        int tail = 0;

        public Scanner(Stream s) { this.s = s; }

        public int NextInt()
        {
            int res = 0;
            int c = Read();
            while (c <= 32)
            {
                if (c == -1) return -1;
                c = Read();
            }
            while (c > 32)
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res;
        }

        int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = s.Read(buf, 0, buf.Length);
                if (tail == 0) return -1;
            }
            return buf[head++];
        }
    }

    static bool ReadInput()
    {
        Scanner sc = new Scanner(Console.OpenStandardInput());
        num_vertices = sc.NextInt();
        if (num_vertices == -1) return false;
        num_edges = sc.NextInt();

        int[] deg = new int[num_vertices];
        int[] u = new int[num_edges];
        int[] v = new int[num_edges];

        for (int i = 0; i < num_edges; i++)
        {
            u[i] = sc.NextInt();
            v[i] = sc.NextInt();
            deg[u[i]]++;
            deg[v[i]]++;
        }

        adj = new int[num_vertices][];
        for (int i = 0; i < num_vertices; i++)
        {
            adj[i] = new int[deg[i]];
            deg[i] = 0;
        }

        for (int i = 0; i < num_edges; i++)
        {
            int u_i = u[i], v_i = v[i];
            // Don't add self-loops
            if (u_i != v_i) 
            {
                adj[u_i][deg[u_i]++] = v_i;
                adj[v_i][deg[v_i]++] = u_i;
            }
        }

        for (int i = 0; i < num_vertices; i++)
        {
            if (deg[i] < adj[i].Length) Array.Resize(ref adj[i], deg[i]);
            Array.Sort(adj[i]);
            // Remove duplicates
            int unique = 0;
            for(int j = 0; j < adj[i].Length; j++)
            {
                if (j == 0 || adj[i][j] != adj[i][j-1])
                {
                    adj[i][unique++] = adj[i][j];
                }
            }
            if (unique < adj[i].Length) Array.Resize(ref adj[i], unique);
        }
        return true;
    }

    static void OutputResult()
    {
        Console.WriteLine(max_size);
        Console.WriteLine(string.Join(" ", best_clique));
    }

    class SearchWorker
    {
        public int[] P_pool;
        public int[] color_pool;
        public int[] indices_pool;
        public int[] C;
        
        public int current_K = 0;
        public int num_vertices;
        
        public int[] color_used_gen;
        public int color_gen = 0;
        
        public int[] active_gen_array;
        public int active_gen = 0;
        
        public SearchWorker(int n)
        {
            num_vertices = n;
            P_pool = new int[0];
            color_pool = new int[0];
            indices_pool = new int[0];
            C = new int[0];
            
            color_used_gen = new int[n + 2];
            active_gen_array = new int[n];
        }
        
        public void EnsureCapacity(int K)
        {
            if (K >= current_K)
            {
                int new_K = K + 10;
                long required = (long)new_K * (new_K + 1) / 2 + new_K + 10;
                
                if (required > 200_000_000) 
                {
                    required = 200_000_000; 
                }
                
                if (P_pool.Length < required)
                {
                    P_pool = new int[required];
                    color_pool = new int[required];
                    indices_pool = new int[required];
                }
                if (C.Length < new_K + 1)
                {
                    C = new int[new_K + 1];
                }
                current_K = new_K;
            }
        }

        public void Expand(int depth, int C_size, int P_size, int pool_offset)
        {
            if (Program.sw.Elapsed > Program.timeLimit) return;
            
            if (P_size == 0)
            {
                lock (Program.lockObj)
                {
                    if (C_size > Program.max_size)
                    {
                        Program.max_size = C_size;
                        Program.best_clique = new int[C_size];
                        Array.Copy(C, Program.best_clique, C_size);
                    }
                }
                return;
            }
            
            int global_max = Volatile.Read(ref Program.max_size);
            if (C_size + P_size <= global_max) return;
            
            Span<int> P = new Span<int>(P_pool, pool_offset, P_size);
            Span<int> color = new Span<int>(color_pool, pool_offset, P_size);
            Span<int> indices = new Span<int>(indices_pool, pool_offset, P_size);
            
            int max_color = 0;
            for (int i = 0; i < P_size; i++)
            {
                int v = P[i];
                color_gen++;
                if (color_gen >= 2_000_000_000)
                {
                    Array.Clear(color_used_gen, 0, color_used_gen.Length);
                    color_gen = 1;
                }
                
                int[] adj_v = Program.adj[v];
                if (i * 20 < adj_v.Length)
                {
                    for (int k = 0; k < i; k++)
                    {
                        if (Array.BinarySearch(adj_v, P[k]) >= 0)
                            color_used_gen[color[k]] = color_gen;
                    }
                }
                else
                {
                    int idx_P = 0, idx_adj = 0;
                    while (idx_P < i && idx_adj < adj_v.Length)
                    {
                        if (P[idx_P] < adj_v[idx_adj]) idx_P++;
                        else if (P[idx_P] > adj_v[idx_adj]) idx_adj++;
                        else
                        {
                            color_used_gen[color[idx_P]] = color_gen;
                            idx_P++; idx_adj++;
                        }
                    }
                }
                
                int c = 1;
                while (color_used_gen[c] == color_gen) c++;
                color[i] = c;
                if (c > max_color) max_color = c;
                
                indices[i] = i;
            }
            
            if (C_size + max_color <= global_max) return;
            
            int next_pool_offset = pool_offset + P_size;
            Span<int> bucket_start = new Span<int>(color_pool, next_pool_offset, max_color + 2);
            bucket_start.Clear();
            
            for (int i = 0; i < P_size; i++) bucket_start[color[i]]++;
            
            int curr_start = 0;
            for (int c = 1; c <= max_color; c++)
            {
                int amount = bucket_start[c];
                bucket_start[c] = curr_start;
                curr_start += amount;
            }
            
            for (int i = 0; i < P_size; i++)
            {
                int c = color[i];
                int pos = bucket_start[c]++;
                indices[pos] = i;
            }
            
            active_gen++;
            if (active_gen >= 2_000_000_000)
            {
                Array.Clear(active_gen_array, 0, active_gen_array.Length);
                active_gen = 1;
            }
            for (int i = 0; i < P_size; i++) active_gen_array[P[i]] = active_gen;
            
            Span<int> next_P = new Span<int>(P_pool, next_pool_offset, P_size);
            
            for (int j = P_size - 1; j >= 0; j--)
            {
                int idx = indices[j];
                int v = P[idx];
                int c = color[idx];
                
                active_gen_array[v] = active_gen - 1;
                
                global_max = Volatile.Read(ref Program.max_size);
                if (C_size + c <= global_max) continue;
                
                int next_P_size = 0;
                int[] adj_v = Program.adj[v];
                
                if (P_size * 20 < adj_v.Length)
                {
                    for (int k = 0; k < P_size; k++)
                    {
                        int u = P[k];
                        if (active_gen_array[u] == active_gen)
                        {
                            if (Array.BinarySearch(adj_v, u) >= 0)
                                next_P[next_P_size++] = u;
                        }
                    }
                }
                else
                {
                    for (int k = 0; k < adj_v.Length; k++)
                    {
                        int u = adj_v[k];
                        if (active_gen_array[u] == active_gen)
                        {
                            next_P[next_P_size++] = u;
                        }
                    }
                }
                
                C[C_size] = v;
                Expand(depth + 1, C_size + 1, next_P_size, next_pool_offset);
            }
        }
    }
}
