using System;
using System.Diagnostics;
using System.Globalization;

class Program
{
    class Scanner
    {
        string[] tokens = new string[0];
        int idx = 0;
        public string Next()
        {
            while (idx >= tokens.Length)
            {
                string line = Console.ReadLine();
                if (line == null) return null;
                tokens = line.Split(new char[] { ' ', '\t', ',', ':' }, StringSplitOptions.RemoveEmptyEntries);
                idx = 0;
            }
            return tokens[idx++];
        }
        public double NextDouble()
        {
            while (true)
            {
                string s = Next();
                if (s == null) return double.NaN;
                if (double.TryParse(s, NumberStyles.Any, CultureInfo.InvariantCulture, out double v))
                {
                    return v;
                }
            }
        }
        public int NextInt()
        {
            return (int)Math.Round(NextDouble());
        }
    }

    class Orbit
    {
        public double[] r = new double[3];
        public double[] v = new double[3];
        public double[] h = new double[3];
        public double[] e_vec = new double[3];
        public double a;
        public double e;
        public double v_c;

        public Orbit(double[] state, double mu)
        {
            for (int i = 0; i < 3; i++)
            {
                r[i] = state[i];
                v[i] = state[i + 3];
            }

            h[0] = r[1] * v[2] - r[2] * v[1];
            h[1] = r[2] * v[0] - r[0] * v[2];
            h[2] = r[0] * v[1] - r[1] * v[0];

            double r_mag = Math.Sqrt(r[0] * r[0] + r[1] * r[1] + r[2] * r[2]);
            double v_sq = v[0] * v[0] + v[1] * v[1] + v[2] * v[2];

            double E = v_sq / 2.0 - mu / r_mag;
            a = -mu / (2.0 * E);

            double[] vxh = new double[3];
            vxh[0] = v[1] * h[2] - v[2] * h[1];
            vxh[1] = v[2] * h[0] - v[0] * h[2];
            vxh[2] = v[0] * h[1] - v[1] * h[0];

            e_vec[0] = vxh[0] / mu - r[0] / r_mag;
            e_vec[1] = vxh[1] / mu - r[1] / r_mag;
            e_vec[2] = vxh[2] / mu - r[2] / r_mag;

            e = Math.Sqrt(e_vec[0] * e_vec[0] + e_vec[1] * e_vec[1] + e_vec[2] * e_vec[2]);
            v_c = Math.Sqrt(mu / Math.Abs(a));
        }
    }

    static double GetCost(Orbit o1, Orbit o2)
    {
        double h1_mag = Math.Sqrt(o1.h[0] * o1.h[0] + o1.h[1] * o1.h[1] + o1.h[2] * o1.h[2]);
        double h2_mag = Math.Sqrt(o2.h[0] * o2.h[0] + o2.h[1] * o2.h[1] + o2.h[2] * o2.h[2]);

        double dot_h = (o1.h[0] * o2.h[0] + o1.h[1] * o2.h[1] + o1.h[2] * o2.h[2]) / (h1_mag * h2_mag);
        dot_h = Math.Max(-1.0, Math.Min(1.0, dot_h));
        double theta = Math.Acos(dot_h);

        double v_c_avg = (o1.v_c + o2.v_c) / 2.0;
        double dv_plane = 2.0 * v_c_avg * Math.Sin(theta / 2.0);

        double dv_a = Math.Abs(o1.v_c - o2.v_c);

        double de_x = o1.e_vec[0] - o2.e_vec[0];
        double de_y = o1.e_vec[1] - o2.e_vec[1];
        double de_z = o1.e_vec[2] - o2.e_vec[2];
        double de_mag = Math.Sqrt(de_x * de_x + de_y * de_y + de_z * de_z);
        double dv_e = v_c_avg * de_mag;

        return dv_a + dv_plane + dv_e;
    }

    static double CalcCost(int[] path, double[,] dist, int N)
    {
        double c = 0;
        for (int i = 0; i < N; i++)
        {
            c += dist[path[i], path[i + 1]];
        }
        return c;
    }

    static void Main(string[] args)
    {
        Scanner scanner = new Scanner();
        
        double nVal = scanner.NextDouble();
        if (double.IsNaN(nVal)) return;
        int N = (int)Math.Round(nVal);
        double mu = scanner.NextDouble();

        double[] startState = new double[6];
        for (int i = 0; i < 6; i++) startState[i] = scanner.NextDouble();

        double[][] stations = new double[N][];
        for (int i = 0; i < N; i++)
        {
            stations[i] = new double[6];
            for (int j = 0; j < 6; j++) stations[i][j] = scanner.NextDouble();
        }

        Orbit[] orbits = new Orbit[N + 1];
        orbits[0] = new Orbit(startState, mu);
        for (int i = 0; i < N; i++)
        {
            orbits[i + 1] = new Orbit(stations[i], mu);
        }

        double[,] dist = new double[N + 1, N + 1];
        for (int i = 0; i <= N; i++)
        {
            for (int j = 0; j <= N; j++)
            {
                dist[i, j] = GetCost(orbits[i], orbits[j]);
            }
        }

        int[] bestPath = new int[N + 1];
        bool[] visited = new bool[N + 1];
        visited[0] = true;
        bestPath[0] = 0;
        int curr = 0;
        for (int i = 1; i <= N; i++)
        {
            double minDist = double.MaxValue;
            int next = -1;
            for (int j = 1; j <= N; j++)
            {
                if (!visited[j] && dist[curr, j] < minDist)
                {
                    minDist = dist[curr, j];
                    next = j;
                }
            }
            bestPath[i] = next;
            visited[next] = true;
            curr = next;
        }

        double bestCost = CalcCost(bestPath, dist, N);
        int[] currentPath = (int[])bestPath.Clone();
        double currentCost = bestCost;

        Random rng = new Random(42);
        int[] newPath = new int[N + 1];

        Stopwatch sw = Stopwatch.StartNew();
        long maxTimeMs = 15000; // 15 seconds limit
        int iter = 0;
        double initialTemp = 100.0;
        double temp = initialTemp;

        while (true)
        {
            if (iter % 1000 == 0)
            {
                long elapsed = sw.ElapsedMilliseconds;
                if (elapsed >= maxTimeMs) break;
                double progress = (double)elapsed / maxTimeMs;
                temp = initialTemp * Math.Exp(-10.0 * progress);
            }
            iter++;

            Array.Copy(currentPath, newPath, N + 1);

            int type = rng.Next(3);
            if (type == 0)
            {
                int i = rng.Next(1, N);
                int j = rng.Next(i + 1, N + 1);
                Array.Reverse(newPath, i, j - i + 1);
            }
            else if (type == 1)
            {
                int i = rng.Next(1, N + 1);
                int j = rng.Next(1, N + 1);
                int tmp = newPath[i];
                newPath[i] = newPath[j];
                newPath[j] = tmp;
            }
            else
            {
                int i = rng.Next(1, N + 1);
                int j = rng.Next(1, N + 1);
                if (i != j)
                {
                    int val = newPath[i];
                    if (i < j)
                    {
                        for (int k = i; k < j; k++) newPath[k] = newPath[k + 1];
                    }
                    else
                    {
                        for (int k = i; k > j; k--) newPath[k] = newPath[k - 1];
                    }
                    newPath[j] = val;
                }
            }

            double newCost = CalcCost(newPath, dist, N);
            double delta = newCost - currentCost;

            if (delta < 0 || Math.Exp(-delta / temp) > rng.NextDouble())
            {
                Array.Copy(newPath, currentPath, N + 1);
                currentCost = newCost;
                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    Array.Copy(currentPath, bestPath, N + 1);
                }
            }
        }

        int[] result = new int[N];
        for (int i = 0; i < N; i++)
        {
            result[i] = bestPath[i + 1] - 1;
        }
        Console.WriteLine(string.Join(" ", result));
    }
}
