using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

struct Box {
    public int id;
    public int w, h, d;
    public long vol;
}

struct EMS {
    public int x1, y1, z1, x2, y2, z2;
    public EMS(int x1, int y1, int z1, int x2, int y2, int z2) {
        this.x1 = x1; this.y1 = y1; this.z1 = z1;
        this.x2 = x2; this.y2 = y2; this.z2 = z2;
    }
}

class FastList {
    public EMS[] array;
    public int count;
    public FastList(int capacity = 1024) {
        array = new EMS[capacity];
        count = 0;
    }
    public void Add(EMS e) {
        if (count == array.Length) {
            Array.Resize(ref array, array.Length * 2);
        }
        array[count++] = e;
    }
    public void RemoveAt(int i) {
        array[i] = array[--count];
    }
    public void Clear() {
        count = 0;
    }
}

class Scanner {
    Stream s;
    byte[] buf = new byte[8192];
    int head = 0, tail = 0;
    public Scanner(Stream s) { this.s = s; }
    public int NextInt() {
        int c = Read();
        while (c < '0' || c > '9') {
            if (c == -1) return -1;
            c = Read();
        }
        int res = 0;
        while (c >= '0' && c <= '9') {
            res = res * 10 + (c - '0');
            c = Read();
        }
        return res;
    }
    int Read() {
        if (head >= tail) {
            head = 0;
            tail = s.Read(buf, 0, buf.Length);
            if (tail == 0) return -1;
        }
        return buf[head++];
    }
}

struct EmsComparer : IComparer<EMS> {
    public int Compare(EMS a, EMS b) {
        long volA = (long)(a.x2 - a.x1) * (a.y2 - a.y1) * (a.z2 - a.z1);
        long volB = (long)(b.x2 - b.x1) * (b.y2 - b.y1) * (b.z2 - b.z1);
        return volB.CompareTo(volA);
    }
}

struct Result {
    public int id;
    public int x, y, z;
}

class Program {
    static void AddRot(int w, int h, int d, ref int count, int[] rw, int[] rh, int[] rd) {
        for (int i = 0; i < count; i++) {
            if (rw[i] == w && rh[i] == h && rd[i] == d) return;
        }
        rw[count] = w;
        rh[count] = h;
        rd[count] = d;
        count++;
    }

    static void GetRotations(int w, int h, int d, out int count, int[] rw, int[] rh, int[] rd) {
        count = 0;
        AddRot(w, h, d, ref count, rw, rh, rd);
        AddRot(w, d, h, ref count, rw, rh, rd);
        AddRot(h, w, d, ref count, rw, rh, rd);
        AddRot(h, d, w, ref count, rw, rh, rd);
        AddRot(d, w, h, ref count, rw, rh, rd);
        AddRot(d, h, w, ref count, rw, rh, rd);
    }

    static void Main() {
        Scanner sc = new Scanner(Console.OpenStandardInput());
        int N = sc.NextInt();
        if (N == -1) return;
        int W = sc.NextInt();
        int H = sc.NextInt();
        int D = sc.NextInt();

        Box[] boxes = new Box[N];
        int globalMinDim = int.MaxValue;
        for (int i = 0; i < N; i++) {
            int w = sc.NextInt();
            int h = sc.NextInt();
            int d = sc.NextInt();
            boxes[i] = new Box { id = i, w = w, h = h, d = d, vol = (long)w * h * d };
            int minDim = Math.Min(w, Math.Min(h, d));
            if (minDim < globalMinDim) globalMinDim = minDim;
        }

        Array.Sort(boxes, (a, b) => {
            int c = a.vol.CompareTo(b.vol);
            if (c != 0) return c;
            int aMin = Math.Min(a.w, Math.Min(a.h, a.d));
            int bMin = Math.Min(b.w, Math.Min(b.h, b.d));
            return aMin.CompareTo(bMin);
        });

        FastList emsList = new FastList(16384);
        emsList.Add(new EMS(0, 0, 0, W, H, D));
        
        FastList newEmsList = new FastList(256);
        List<Result> results = new List<Result>(N);

        int[] rw = new int[6];
        int[] rh = new int[6];
        int[] rd = new int[6];
        int rotCount = 0;
        
        EmsComparer emsComparer = new EmsComparer();

        for (int bIdx = 0; bIdx < N; bIdx++) {
            var box = boxes[bIdx];
            GetRotations(box.w, box.h, box.d, out rotCount, rw, rh, rd);

            int bestEmsIndex = -1;
            int bestRot = -1;
            long bestScore1 = long.MaxValue; // z1
            long bestScore2 = long.MaxValue; // y1
            long bestScore3 = long.MaxValue; // x1
            long bestScore4 = long.MaxValue; // fitness

            for (int i = 0; i < emsList.count; i++) {
                var e = emsList.array[i];
                int ew = e.x2 - e.x1;
                int eh = e.y2 - e.y1;
                int ed = e.z2 - e.z1;

                if (ew < box.w && eh < box.w && ed < box.w) continue; 

                for (int r = 0; r < rotCount; r++) {
                    int bw = rw[r], bh = rh[r], bd = rd[r];
                    if (bw <= ew && bh <= eh && bd <= ed) {
                        long s1 = e.z1;
                        long s2 = e.y1;
                        long s3 = e.x1;
                        long s4 = Math.Min(ew - bw, Math.Min(eh - bh, ed - bd));

                        bool better = false;
                        if (s1 < bestScore1) better = true;
                        else if (s1 == bestScore1) {
                            if (s2 < bestScore2) better = true;
                            else if (s2 == bestScore2) {
                                if (s3 < bestScore3) better = true;
                                else if (s3 == bestScore3) {
                                    if (s4 < bestScore4) better = true;
                                }
                            }
                        }

                        if (better) {
                            bestScore1 = s1;
                            bestScore2 = s2;
                            bestScore3 = s3;
                            bestScore4 = s4;
                            bestEmsIndex = i;
                            bestRot = r;
                        }
                    }
                }
            }

            if (bestEmsIndex != -1) {
                var bestE = emsList.array[bestEmsIndex];
                int bx1 = bestE.x1;
                int by1 = bestE.y1;
                int bz1 = bestE.z1;
                int bx2 = bx1 + rw[bestRot];
                int by2 = by1 + rh[bestRot];
                int bz2 = bz1 + rd[bestRot];

                results.Add(new Result { id = box.id, x = bx1, y = by1, z = bz1 });

                newEmsList.Clear();
                for (int i = emsList.count - 1; i >= 0; i--) {
                    var e = emsList.array[i];
                    if (bx1 < e.x2 && bx2 > e.x1 &&
                        by1 < e.y2 && by2 > e.y1 &&
                        bz1 < e.z2 && bz2 > e.z1) {
                        
                        emsList.RemoveAt(i);

                        if (bx1 > e.x1) {
                            var n = new EMS(e.x1, e.y1, e.z1, bx1, e.y2, e.z2);
                            if (n.x2 - n.x1 >= globalMinDim && n.y2 - n.y1 >= globalMinDim && n.z2 - n.z1 >= globalMinDim)
                                newEmsList.Add(n);
                        }
                        if (bx2 < e.x2) {
                            var n = new EMS(bx2, e.y1, e.z1, e.x2, e.y2, e.z2);
                            if (n.x2 - n.x1 >= globalMinDim && n.y2 - n.y1 >= globalMinDim && n.z2 - n.z1 >= globalMinDim)
                                newEmsList.Add(n);
                        }
                        if (by1 > e.y1) {
                            var n = new EMS(e.x1, e.y1, e.z1, e.x2, by1, e.z2);
                            if (n.x2 - n.x1 >= globalMinDim && n.y2 - n.y1 >= globalMinDim && n.z2 - n.z1 >= globalMinDim)
                                newEmsList.Add(n);
                        }
                        if (by2 < e.y2) {
                            var n = new EMS(e.x1, by2, e.z1, e.x2, e.y2, e.z2);
                            if (n.x2 - n.x1 >= globalMinDim && n.y2 - n.y1 >= globalMinDim && n.z2 - n.z1 >= globalMinDim)
                                newEmsList.Add(n);
                        }
                        if (bz1 > e.z1) {
                            var n = new EMS(e.x1, e.y1, e.z1, e.x2, e.y2, bz1);
                            if (n.x2 - n.x1 >= globalMinDim && n.y2 - n.y1 >= globalMinDim && n.z2 - n.z1 >= globalMinDim)
                                newEmsList.Add(n);
                        }
                        if (bz2 < e.z2) {
                            var n = new EMS(e.x1, e.y1, bz2, e.x2, e.y2, e.z2);
                            if (n.x2 - n.x1 >= globalMinDim && n.y2 - n.y1 >= globalMinDim && n.z2 - n.z1 >= globalMinDim)
                                newEmsList.Add(n);
                        }
                    }
                }

                for (int i = newEmsList.count - 1; i >= 0; i--) {
                    var e1 = newEmsList.array[i];
                    bool contained = false;
                    for (int j = 0; j < newEmsList.count; j++) {
                        if (i == j) continue;
                        var e2 = newEmsList.array[j];
                        if (e1.x1 >= e2.x1 && e1.y1 >= e2.y1 && e1.z1 >= e2.z1 &&
                            e1.x2 <= e2.x2 && e1.y2 <= e2.y2 && e1.z2 <= e2.z2) {
                            contained = true;
                            break;
                        }
                    }
                    if (contained) newEmsList.RemoveAt(i);
                }

                for (int i = newEmsList.count - 1; i >= 0; i--) {
                    var e1 = newEmsList.array[i];
                    bool contained = false;
                    for (int j = 0; j < emsList.count; j++) {
                        var e2 = emsList.array[j];
                        if (e1.x1 >= e2.x1 && e1.y1 >= e2.y1 && e1.z1 >= e2.z1 &&
                            e1.x2 <= e2.x2 && e1.y2 <= e2.y2 && e1.z2 <= e2.z2) {
                            contained = true;
                            break;
                        }
                    }
                    if (contained) newEmsList.RemoveAt(i);
                }

                for (int i = 0; i < newEmsList.count; i++) {
                    emsList.Add(newEmsList.array[i]);
                }
                
                if (emsList.count > 15000) {
                    Array.Sort(emsList.array, 0, emsList.count, emsComparer);
                    emsList.count = 10000;
                }
            }
        }

        using (var sw = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 65536)) {
            sw.WriteLine(results.Count);
            foreach (var r in results) {
                sw.WriteLine($"{r.id} {r.x} {r.y} {r.z}");
            }
        }
    }
}
