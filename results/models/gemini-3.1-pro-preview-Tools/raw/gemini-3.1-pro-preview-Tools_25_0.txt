using System;
using System.Collections.Generic;
using System.Globalization;

class Program
{
    static void Main(string[] args)
    {
        string[] tokens = ReadAllTokens();
        if (tokens.Length < 6) return;

        int numBodies = int.Parse(tokens[0]);
        double warningDays = double.Parse(tokens[1], CultureInfo.InvariantCulture);
        double impactorMass = double.Parse(tokens[2], CultureInfo.InvariantCulture);
        double deltaVBudget = double.Parse(tokens[3], CultureInfo.InvariantCulture);
        double asteroidMass = double.Parse(tokens[4], CultureInfo.InvariantCulture);
        double t0UnixUtc = double.Parse(tokens[5], CultureInfo.InvariantCulture);

        double earthX = 0, earthY = 0, earthZ = 0, earthVx = 0, earthVy = 0, earthVz = 0;
        double astX = 0, astY = 0, astZ = 0, astVx = 0, astVy = 0, astVz = 0;

        int tokenIndex = 6;
        for (int i = 0; i < numBodies; i++)
        {
            if (tokenIndex >= tokens.Length) break;
            string bodyName = tokens[tokenIndex++];
            double mass = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);
            double x = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);
            double y = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);
            double z = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);
            double vx = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);
            double vy = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);
            double vz = double.Parse(tokens[tokenIndex++], CultureInfo.InvariantCulture);

            if (bodyName == "Earth")
            {
                earthX = x; earthY = y; earthZ = z;
                earthVx = vx; earthVy = vy; earthVz = vz;
            }
            else if (bodyName == "Asteroid")
            {
                astX = x; astY = y; astZ = z;
                astVx = vx; astVy = vy; astVz = vz;
            }
        }

        string bspFilePath = tokenIndex < tokens.Length ? tokens[tokenIndex] : "";

        // Use a linear approximation for interception trajectory
        double interceptDays = warningDays * 0.75; 
        double interceptSeconds = interceptDays * 86400.0;

        // Predict asteroid position at intercept time
        double targetX = astX + astVx * interceptSeconds;
        double targetY = astY + astVy * interceptSeconds;
        double targetZ = astZ + astVz * interceptSeconds;

        // Required average velocity to intercept
        double reqVx = (targetX - earthX) / interceptSeconds;
        double reqVy = (targetY - earthY) / interceptSeconds;
        double reqVz = (targetZ - earthZ) / interceptSeconds;

        // Delta-V required for launch
        double dvX = reqVx - earthVx;
        double dvY = reqVy - earthVy;
        double dvZ = reqVz - earthVz;

        double dvMag = Math.Sqrt(dvX * dvX + dvY * dvY + dvZ * dvZ);
        
        // Ensure we stay within budget
        double launchDvMag = Math.Min(dvMag, deltaVBudget * 0.9);
        if (dvMag > 0)
        {
            dvX = (dvX / dvMag) * launchDvMag;
            dvY = (dvY / dvMag) * launchDvMag;
            dvZ = (dvZ / dvMag) * launchDvMag;
        }

        double remainingDv = deltaVBudget - launchDvMag;
        
        // Impact terminal maneuver: oppose asteroid velocity to maximize orbital deflection
        double astVMag = Math.Sqrt(astVx * astVx + astVy * astVy + astVz * astVz);
        double impDvX = 0, impDvY = 0, impDvZ = 0;
        if (astVMag > 0)
        {
            impDvX = -(astVx / astVMag) * remainingDv;
            impDvY = -(astVy / astVMag) * remainingDv;
            impDvZ = -(astVz / astVMag) * remainingDv;
        }

        Console.WriteLine("2");
        Console.WriteLine($"0.000000 {dvX:F6} {dvY:F6} {dvZ:F6}");
        Console.WriteLine($"{interceptDays:F6} {impDvX:F6} {impDvY:F6} {impDvZ:F6}");
    }

    static string[] ReadAllTokens()
    {
        var tokens = new List<string>();
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            var parts = line.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
            tokens.AddRange(parts);
        }
        return tokens.ToArray();
    }
}
