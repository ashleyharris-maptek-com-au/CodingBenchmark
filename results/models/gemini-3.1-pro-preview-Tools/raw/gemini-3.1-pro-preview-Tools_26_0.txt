use std::io::{self, Read, Write, BufRead, BufReader, BufWriter};

#[derive(Clone, Copy, Default)]
#[repr(C)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[inline(always)]
fn dist2(a: &Point, b: &Point) -> f32 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

fn xor_shift(mut state: u32) -> u32 {
    state ^= state << 13;
    state ^= state >> 17;
    state ^= state << 5;
    state
}

struct CentroidIndex {
    centroids: Vec<Point>,
    use_index: bool,
    num_supers: usize,
    super_centroids: Vec<Point>,
    super_members: Vec<Vec<usize>>,
}

impl CentroidIndex {
    fn new(centroids: Vec<Point>) -> Self {
        let k = centroids.len();
        let use_index = k > 100;
        let num_supers = if use_index { (k as f64).sqrt() as usize } else { 0 };
        
        let mut idx = CentroidIndex {
            centroids,
            use_index,
            num_supers,
            super_centroids: vec![Point::default(); num_supers],
            super_members: vec![Vec::new(); num_supers],
        };
        
        if use_index {
            idx.rebuild_index();
        }
        idx
    }
    
    fn rebuild_index(&mut self) {
        if !self.use_index { return; }
        
        let mut uninit = true;
        for sc in &self.super_centroids {
            if sc.x != 0.0 || sc.y != 0.0 || sc.z != 0.0 {
                uninit = false;
                break;
            }
        }
        if uninit {
            for i in 0..self.num_supers {
                self.super_centroids[i] = self.centroids[i];
            }
        }
        
        for s in 0..self.num_supers {
            self.super_members[s].clear();
        }
        
        for (i, c) in self.centroids.iter().enumerate() {
            let mut best_s = 0;
            let mut best_dist = f32::MAX;
            for s in 0..self.num_supers {
                let d = dist2(c, &self.super_centroids[s]);
                if d < best_dist {
                    best_dist = d;
                    best_s = s;
                }
            }
            self.super_members[best_s].push(i);
        }
        
        let mut s_sums = vec![Point::default(); self.num_supers];
        let mut s_counts = vec![0; self.num_supers];
        for s in 0..self.num_supers {
            for &c_idx in &self.super_members[s] {
                s_sums[s].x += self.centroids[c_idx].x;
                s_sums[s].y += self.centroids[c_idx].y;
                s_sums[s].z += self.centroids[c_idx].z;
                s_counts[s] += 1;
            }
            if s_counts[s] > 0 {
                self.super_centroids[s].x = s_sums[s].x / s_counts[s] as f32;
                self.super_centroids[s].y = s_sums[s].y / s_counts[s] as f32;
                self.super_centroids[s].z = s_sums[s].z / s_counts[s] as f32;
            }
        }
    }
    
    fn nearest(&self, p: &Point) -> usize {
        let mut best_c = 0;
        let mut best_dist = f32::MAX;
        
        if self.use_index {
            let mut s_best = 0;
            let mut s_best_dist = f32::MAX;
            let mut s_second = 0;
            let mut s_second_dist = f32::MAX;
            
            for s in 0..self.num_supers {
                let d = dist2(p, &self.super_centroids[s]);
                if d < s_best_dist {
                    s_second_dist = s_best_dist;
                    s_second = s_best;
                    s_best_dist = d;
                    s_best = s;
                } else if d < s_second_dist {
                    s_second_dist = d;
                    s_second = s;
                }
            }
            
            for &c in &self.super_members[s_best] {
                let d = dist2(p, &self.centroids[c]);
                if d < best_dist {
                    best_dist = d;
                    best_c = c;
                }
            }
            for &c in &self.super_members[s_second] {
                let d = dist2(p, &self.centroids[c]);
                if d < best_dist {
                    best_dist = d;
                    best_c = c;
                }
            }
            
            if best_dist != f32::MAX {
                return best_c;
            }
        }
        
        for (c_idx, c) in self.centroids.iter().enumerate() {
            let d = dist2(p, c);
            if d < best_dist {
                best_dist = d;
                best_c = c_idx;
            }
        }
        best_c
    }
}

fn memory_kmeans(num_points: usize, num_clusters: usize, reader: &mut impl Read, writer: &mut impl Write) -> io::Result<()> {
    let mut points = Vec::with_capacity(num_points);
    let mut batch_bytes = vec![0u8; 120_000];
    let mut points_read = 0;
    while points_read < num_points {
        let to_read = (num_points - points_read).min(10_000);
        let bytes_to_read = to_read * 12;
        reader.read_exact(&mut batch_bytes[..bytes_to_read])?;
        for i in 0..to_read {
            let offset = i * 12;
            let x = f32::from_le_bytes(batch_bytes[offset..offset+4].try_into().unwrap());
            let y = f32::from_le_bytes(batch_bytes[offset+4..offset+8].try_into().unwrap());
            let z = f32::from_le_bytes(batch_bytes[offset+8..offset+12].try_into().unwrap());
            points.push(Point { x, y, z });
        }
        points_read += to_read;
    }
    
    let mut initial_centroids = Vec::with_capacity(num_clusters);
    let mut rng = 12345u32;
    for _ in 0..num_clusters {
        rng = xor_shift(rng);
        let idx = (rng as usize) % num_points;
        initial_centroids.push(points[idx]);
    }
    
    let mut index = CentroidIndex::new(initial_centroids);
    let mut assignments = vec![0u32; num_points];
    
    let max_iters = 15;
    for _iter in 0..max_iters {
        let mut sums = vec![Point::default(); num_clusters];
        let mut counts = vec![0u32; num_clusters];
        
        let mut changed = 0;
        for (i, p) in points.iter().enumerate() {
            let c = index.nearest(p);
            if assignments[i] != c as u32 {
                changed += 1;
                assignments[i] = c as u32;
            }
            sums[c].x += p.x;
            sums[c].y += p.y;
            sums[c].z += p.z;
            counts[c] += 1;
        }
        
        for c in 0..num_clusters {
            if counts[c] > 0 {
                index.centroids[c].x = sums[c].x / counts[c] as f32;
                index.centroids[c].y = sums[c].y / counts[c] as f32;
                index.centroids[c].z = sums[c].z / counts[c] as f32;
            } else {
                rng = xor_shift(rng);
                let idx = (rng as usize) % num_points;
                index.centroids[c] = points[idx];
            }
        }
        index.rebuild_index();
        
        if changed as f32 / num_points as f32 < 0.0001 {
            break;
        }
    }
    
    for a in assignments {
        writer.write_all(&a.to_le_bytes())?;
    }
    
    Ok(())
}

fn streaming_kmeans(num_points: usize, num_clusters: usize, reader: &mut impl Read, writer: &mut impl Write) -> io::Result<()> {
    let mut initial_centroids = Vec::with_capacity(num_clusters);
    let mut initial_bytes = vec![0u8; num_clusters * 12];
    reader.read_exact(&mut initial_bytes)?;
    for i in 0..num_clusters {
        let x = f32::from_le_bytes(initial_bytes[i*12..i*12+4].try_into().unwrap());
        let y = f32::from_le_bytes(initial_bytes[i*12+4..i*12+8].try_into().unwrap());
        let z = f32::from_le_bytes(initial_bytes[i*12+8..i*12+12].try_into().unwrap());
        initial_centroids.push(Point { x, y, z });
        writer.write_all(&(i as u32).to_le_bytes())?;
    }
    
    let mut index = CentroidIndex::new(initial_centroids);
    let mut centroid_weights = vec![1.0f64; num_clusters];
    
    let batch_size = 100_000;
    let mut batch_bytes = vec![0u8; batch_size * 12];
    
    let mut points_read = num_clusters;
    while points_read < num_points {
        let current_batch_size = (num_points - points_read).min(batch_size);
        let bytes_to_read = current_batch_size * 12;
        reader.read_exact(&mut batch_bytes[..bytes_to_read])?;
        
        let mut batch_sums = vec![Point::default(); num_clusters];
        let mut batch_counts = vec![0u32; num_clusters];
        
        for i in 0..current_batch_size {
            let offset = i * 12;
            let x = f32::from_le_bytes(batch_bytes[offset..offset+4].try_into().unwrap());
            let y = f32::from_le_bytes(batch_bytes[offset+4..offset+8].try_into().unwrap());
            let z = f32::from_le_bytes(batch_bytes[offset+8..offset+12].try_into().unwrap());
            let p = Point { x, y, z };
            
            let c = index.nearest(&p);
            batch_sums[c].x += p.x;
            batch_sums[c].y += p.y;
            batch_sums[c].z += p.z;
            batch_counts[c] += 1;
            
            writer.write_all(&(c as u32).to_le_bytes())?;
        }
        
        for c in 0..num_clusters {
            if batch_counts[c] > 0 {
                let bc = batch_counts[c] as f64;
                let total_w = centroid_weights[c] + bc;
                let alpha = bc / total_w;
                
                index.centroids[c].x = (index.centroids[c].x as f64 * (1.0 - alpha) + (batch_sums[c].x as f64 / bc) * alpha) as f32;
                index.centroids[c].y = (index.centroids[c].y as f64 * (1.0 - alpha) + (batch_sums[c].y as f64 / bc) * alpha) as f32;
                index.centroids[c].z = (index.centroids[c].z as f64 * (1.0 - alpha) + (batch_sums[c].z as f64 / bc) * alpha) as f32;
                
                centroid_weights[c] = total_w;
            }
        }
        
        index.rebuild_index();
        points_read += current_batch_size;
    }
    
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::with_capacity(1024 * 1024, stdin.lock());
    let mut writer = BufWriter::with_capacity(1024 * 1024, stdout.lock());
    
    let mut header = String::new();
    reader.read_line(&mut header)?;
    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    if parts.len() < 2 {
        return Ok(());
    }
    
    let num_points: usize = parts[0].parse().unwrap();
    let num_clusters: usize = parts[1].parse().unwrap();
    
    if num_clusters == 0 {
        return Ok(());
    }
    
    if num_clusters >= num_points {
        let mut pt_bytes = [0u8; 12];
        for i in 0..num_points {
            reader.read_exact(&mut pt_bytes)?;
            writer.write_all(&(i as u32).to_le_bytes())?;
        }
        return Ok(());
    }
    
    if num_points <= 10_000_000 {
        memory_kmeans(num_points, num_clusters, &mut reader, &mut writer)?;
    } else {
        streaming_kmeans(num_points, num_clusters, &mut reader, &mut writer)?;
    }
    
    writer.flush()?;
    Ok(())
}
