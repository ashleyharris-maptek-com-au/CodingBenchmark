#include <iostream>
#include <string>
#include <sstream>
#include <thread>
#include <atomic>
#include <mutex>
#include <vector>
#include <cmath>
#include <windows.h>
#include <algorithm>
#include <condition_variable>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double prate, yrate, rrate;
    double fuel;
    bool valid = false;
};

std::mutex state_mutex;
std::condition_variable state_cv;
State current_state;
std::atomic<bool> new_state{false};
std::atomic<bool> is_running{true};

void stdin_thread_func() {
    std::string line;
    while (is_running && std::getline(std::cin, line)) {
        if (line.empty()) continue;
        std::istringstream iss(line);
        State s;
        if (iss >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz 
                >> s.pitch >> s.yaw >> s.roll >> s.prate >> s.yrate >> s.rrate >> s.fuel) {
            s.valid = true;
            {
                std::lock_guard<std::mutex> lock(state_mutex);
                current_state = s;
                new_state = true;
            }
            state_cv.notify_one();
        }
    }
    is_running = false;
    state_cv.notify_all();
}

struct Vec3 {
    double x, y, z;
    Vec3 operator+(Vec3 o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(Vec3 o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double length() const { return std::sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { 
        double l = length(); 
        return l > 0.0001 ? Vec3{x/l, y/l, z/l} : Vec3{0,0,0}; 
    }
};

double shortest_angle(double target, double current) {
    double diff = std::fmod(target - current, 2.0 * 3.141592653589793);
    if (diff > 3.141592653589793) diff -= 2.0 * 3.141592653589793;
    if (diff < -3.141592653589793) diff += 2.0 * 3.141592653589793;
    return diff;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    
    double width, depth, height, gravity, max_thrust, fuel_init;
    double sx, sy, sz, tx, ty, tz;
    if (!(std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel_init
                   >> sx >> sy >> sz >> tx >> ty >> tz)) {
        return 1;
    }

    std::string terrain_path;
    std::getline(std::cin, terrain_path);
    while (terrain_path.empty() || terrain_path.find_first_not_of(" \r\n\t") == std::string::npos) {
        if (!std::getline(std::cin, terrain_path)) break;
    }
    if (!terrain_path.empty() && terrain_path.back() == '\r') terrain_path.pop_back();

    std::string state_line;
    while (std::getline(std::cin, state_line)) {
        if (!state_line.empty() && state_line.back() == '\r') state_line.pop_back();
        if (state_line == "STATE") break;
    }

    std::thread reader(stdin_thread_func);

    double safe_z = std::max(sz, tz) + 200.0;
    HANDLE hFile = CreateFileA(terrain_path.c_str(), GENERIC_READ, FILE_SHARE_READ, NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
    if (hFile != INVALID_HANDLE_VALUE) {
        HANDLE hMapping = CreateFileMappingA(hFile, NULL, PAGE_READONLY, 0, 0, NULL);
        if (hMapping) {
            uint16_t* terrain_data = (uint16_t*)MapViewOfFile(hMapping, FILE_MAP_READ, 0, 0, 0);
            if (terrain_data) {
                int steps = std::max(100, (int)std::hypot(tx - sx, ty - sy));
                int max_h = 0;
                size_t w = width;
                size_t d = depth;
                for (int i = 0; i <= steps; ++i) {
                    double t = (double)i / steps;
                    double cx = sx + t * (tx - sx);
                    double cy = sy + t * (ty - sy);
                    int icx = (int)std::round(cx);
                    int icy = (int)std::round(cy);
                    for (int dx = -50; dx <= 50; dx+=10) {
                        for (int dy = -50; dy <= 50; dy+=10) {
                            int ix = icx + dx;
                            int iy = icy + dy;
                            if (ix >= 0 && ix < w && iy >= 0 && iy < d) {
                                size_t index = (size_t)ix * d + (size_t)iy;
                                int a = terrain_data[index * 3];
                                int c = terrain_data[index * 3 + 2];
                                int h = a > c ? a : c;
                                if (h > max_h) max_h = h;
                            }
                        }
                    }
                }
                safe_z = (double)max_h + 100.0;
                UnmapViewOfFile(terrain_data);
            }
            CloseHandle(hMapping);
        }
        CloseHandle(hFile);
    }

    enum Phase { ASCENT, CRUISE, DESCENT };
    Phase phase = ASCENT;

    while (is_running) {
        State state;
        {
            std::unique_lock<std::mutex> lock(state_mutex);
            state_cv.wait(lock, []{ return new_state.load() || !is_running.load(); });
            if (!is_running) break;
            state = current_state;
            new_state = false;
        }

        Vec3 pos = {state.x, state.y, state.z};
        Vec3 vel = {state.vx, state.vy, state.vz};

        double dist_xy = std::hypot(tx - state.x, ty - state.y);
        double v_xy = std::hypot(state.vx, state.vy);

        if (phase == ASCENT) {
            if (state.z > safe_z - 5.0 && std::abs(state.vz) < 2.0) phase = CRUISE;
        } else if (phase == CRUISE) {
            if (dist_xy < 10.0 && v_xy < 5.0) phase = DESCENT;
        }

        Vec3 target_pos;
        if (phase == ASCENT) {
            target_pos = {sx, sy, safe_z};
        } else if (phase == CRUISE) {
            target_pos = {tx, ty, safe_z};
        } else {
            target_pos = {tx, ty, tz - 1.0};
        }

        Vec3 pos_err = target_pos - pos;
        double dist = pos_err.length();
        
        double avail_accel = std::sqrt(std::max(0.1, max_thrust*max_thrust - gravity*gravity));
        double a_brake = std::max(0.2, avail_accel * 0.4); 

        double max_v_for_dist = std::sqrt(2.0 * a_brake * std::max(0.0, dist));
        
        double P_pos = 0.2;
        if (phase == DESCENT && dist < 50.0) P_pos = 0.1;
        
        Vec3 desired_vel;
        if (dist > 0.001) {
            double v_mag = std::min(dist * P_pos, max_v_for_dist);
            double absolute_max = (phase == DESCENT) ? 15.0 : 150.0;
            v_mag = std::min(v_mag, absolute_max);
            desired_vel = pos_err.normalized() * v_mag;
        } else {
            desired_vel = {0, 0, 0};
        }

        Vec3 vel_err = desired_vel - vel;
        double P_vel = 1.0;
        Vec3 a_des = vel_err * P_vel;
        
        a_des.z += gravity;

        double required_thrust = a_des.length();
        Vec3 U_des = required_thrust > 0.0001 ? a_des.normalized() : Vec3{0, 0, 1.0};

        if (required_thrust > max_thrust) {
            double max_z_accel = max_thrust; 
            if (a_des.z > max_z_accel) {
                U_des = {0, 0, 1.0}; 
                required_thrust = max_thrust;
            } else {
                double required_xy_sq = max_thrust * max_thrust - a_des.z * a_des.z;
                double current_xy_sq = a_des.x * a_des.x + a_des.y * a_des.y;
                if (current_xy_sq > required_xy_sq && current_xy_sq > 0.0001) {
                    double scale = std::sqrt(required_xy_sq / current_xy_sq);
                    a_des.x *= scale;
                    a_des.y *= scale;
                }
                U_des = a_des.normalized();
                required_thrust = max_thrust;
            }
        }

        double thrust_cmd = required_thrust / max_thrust;
        if (thrust_cmd > 1.0) thrust_cmd = 1.0;
        if (thrust_cmd < 0.0) thrust_cmd = 0.0;

        if (U_des.z < 0.01) {
            U_des.z = 0.01;
            U_des = U_des.normalized();
        }
        if (U_des.z > 1.0) U_des.z = 1.0;

        double U_horiz = std::hypot(U_des.x, U_des.y);
        double target_yaw = state.yaw;
        double target_pitch = 0.0;

        if (U_horiz > 0.01) {
            target_yaw = std::atan2(-U_des.y, -U_des.x);
            double yaw_err1 = shortest_angle(target_yaw, state.yaw);
            double yaw_err2 = shortest_angle(target_yaw + 3.141592653589793, state.yaw);
            if (std::abs(yaw_err2) < std::abs(yaw_err1)) {
                target_yaw += 3.141592653589793;
                target_pitch = -std::acos(U_des.z);
            } else {
                target_pitch = std::acos(U_des.z);
            }
        } else {
            target_pitch = std::acos(U_des.z);
        }

        double yaw_err = shortest_angle(target_yaw, state.yaw);
        double yaw_cmd = 2.0 * yaw_err - 2.0 * state.yrate;

        double pitch_err = target_pitch - state.pitch;
        double pitch_cmd = 2.0 * pitch_err - 2.0 * state.prate;

        double roll_err = 0.0 - state.roll;
        double roll_cmd = 2.0 * roll_err - 2.0 * state.rrate;

        auto clamp = [](double v) { return v > 1.0 ? 1.0 : (v < -1.0 ? -1.0 : v); };
        pitch_cmd = clamp(pitch_cmd);
        yaw_cmd = clamp(yaw_cmd);
        roll_cmd = clamp(roll_cmd);

        std::cout << thrust_cmd << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();
    }

    if (reader.joinable()) reader.join();
    return 0;
}
