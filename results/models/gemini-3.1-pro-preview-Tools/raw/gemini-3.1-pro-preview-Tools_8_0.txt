use std::io::{self, Read, BufRead, Write, BufWriter};

fn main() {
    let stdin = io::stdin();
    let mut lock = stdin.lock();
    
    let mut first_line = String::new();
    lock.read_line(&mut first_line).unwrap();
    let mut parts = first_line.trim().split_whitespace();
    let h: u32 = parts.next().unwrap().parse().unwrap();
    let w: u32 = parts.next().unwrap().parse().unwrap();
    
    // Read the entire maze into memory.
    // Pre-allocate slightly more to account for newlines without reallocating.
    let mut maze = Vec::with_capacity(h as usize * w as usize + h as usize * 3 + 128);
    lock.read_to_end(&mut maze).unwrap();
    
    // Strip all newlines in-place extremely quickly
    maze.retain(|&c| c != b'\n' && c != b'\r');
    
    // Ensure the size is exactly w * h to guarantee memory safety during unsafe accesses
    let expected_len = h as usize * w as usize;
    if maze.len() > expected_len {
        maze.truncate(expected_len);
    } else if maze.len() < expected_len {
        panic!("Maze data too small");
    }
    
    // Find Start and Goal.
    let start_idx = maze.iter().position(|&c| c == b'A').unwrap() as u32;
    let end_idx = maze.iter().position(|&c| c == b'B').unwrap() as u32;
    
    let mut found = false;
    
    if start_idx == end_idx {
        found = true;
    } else {
        // Prevent re-visiting the start node
        maze[start_idx as usize] = b'S';
        
        // Two-queue BFS avoids allocating per-node parent pointers or using VecDeque wrapping logic.
        let mut current_level: Vec<u32> = Vec::with_capacity(w as usize * 2);
        let mut next_level: Vec<u32> = Vec::with_capacity(w as usize * 2);
        
        current_level.push(start_idx);
        
        let cell_ptr = maze.as_mut_ptr();
        
        'search: while !current_level.is_empty() {
            for &idx in &current_level {
                let x = idx % w;
                let y = idx / w;
                
                // Move UP
                if y > 0 {
                    let n_idx = idx - w;
                    unsafe {
                        let ptr = cell_ptr.add(n_idx as usize);
                        let cell = *ptr;
                        if cell == b' ' || cell == b'B' {
                            *ptr = 2; // Store direction we came from (parent is DOWN)
                            next_level.push(n_idx);
                            if cell == b'B' { found = true; break 'search; }
                        }
                    }
                }
                
                // Move DOWN
                if y + 1 < h {
                    let n_idx = idx + w;
                    unsafe {
                        let ptr = cell_ptr.add(n_idx as usize);
                        let cell = *ptr;
                        if cell == b' ' || cell == b'B' {
                            *ptr = 0; // Store direction we came from (parent is UP)
                            next_level.push(n_idx);
                            if cell == b'B' { found = true; break 'search; }
                        }
                    }
                }
                
                // Move LEFT
                if x > 0 {
                    let n_idx = idx - 1;
                    unsafe {
                        let ptr = cell_ptr.add(n_idx as usize);
                        let cell = *ptr;
                        if cell == b' ' || cell == b'B' {
                            *ptr = 1; // Store direction we came from (parent is RIGHT)
                            next_level.push(n_idx);
                            if cell == b'B' { found = true; break 'search; }
                        }
                    }
                }
                
                // Move RIGHT
                if x + 1 < w {
                    let n_idx = idx + 1;
                    unsafe {
                        let ptr = cell_ptr.add(n_idx as usize);
                        let cell = *ptr;
                        if cell == b' ' || cell == b'B' {
                            *ptr = 3; // Store direction we came from (parent is LEFT)
                            next_level.push(n_idx);
                            if cell == b'B' { found = true; break 'search; }
                        }
                    }
                }
            }
            current_level.clear();
            std::mem::swap(&mut current_level, &mut next_level);
        }
    }
    
    let stdout = io::stdout();
    let mut out = BufWriter::with_capacity(1024 * 1024, stdout.lock());
    
    if !found {
        writeln!(out, "0").unwrap();
    } else {
        // Trace back the shortest path by following the directional markers left during BFS
        let mut path = Vec::new();
        let mut curr_idx = end_idx;
        path.push(curr_idx);
        
        while curr_idx != start_idx {
            let dir = maze[curr_idx as usize];
            match dir {
                0 => curr_idx -= w,
                1 => curr_idx += 1,
                2 => curr_idx += w,
                3 => curr_idx -= 1,
                _ => panic!("Corrupted path state at index {}: {}", curr_idx, dir),
            }
            path.push(curr_idx);
        }
        path.reverse();
        
        writeln!(out, "{}", path.len()).unwrap();
        for &idx in &path {
            let x = idx % w;
            let y = idx / w;
            writeln!(out, "{} {}", x, y).unwrap();
        }
    }
}
