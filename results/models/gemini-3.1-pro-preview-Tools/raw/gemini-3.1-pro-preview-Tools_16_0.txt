#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>

using namespace std;

struct Task {
    int id;
    int job;
    int machine;
    int duration;
};

struct Job {
    int id;
    vector<Task> tasks;
    int total_work;
};

struct ScheduleBuilder {
    int J, M;
    const vector<Job>& jobs;
    vector<int> next_task_idx;
    vector<int> job_ready;
    vector<int> machine_ready;
    vector<int> job_rem_work;
    
    vector<vector<int>> start_times;
    
    int completed_tasks;
    int total_tasks;
    
    ScheduleBuilder(int J, int M, const vector<Job>& jobs, int total_tasks) 
        : J(J), M(M), jobs(jobs), total_tasks(total_tasks) {
        next_task_idx.assign(J, 0);
        job_ready.assign(J, 0);
        machine_ready.assign(M, 0);
        job_rem_work.assign(J, 0);
        start_times.resize(J);
        for(int i=0; i<J; ++i) {
            start_times[i].resize(jobs[i].tasks.size(), -1);
            job_rem_work[i] = jobs[i].total_work;
        }
        completed_tasks = 0;
    }
    
    int build(mt19937& rng) {
        while (completed_tasks < total_tasks) {
            int min_ect = 2e9;
            int best_m = -1;
            int min_ect_job = -1;
            
            for (int j = 0; j < J; ++j) {
                if (next_task_idx[j] < jobs[j].tasks.size()) {
                    const Task& t = jobs[j].tasks[next_task_idx[j]];
                    int est = max(job_ready[j], machine_ready[t.machine]);
                    int ect = est + t.duration;
                    if (ect < min_ect) {
                        min_ect = ect;
                        best_m = t.machine;
                        min_ect_job = j;
                    }
                }
            }
            
            vector<int> G;
            G.reserve(J);
            for (int j = 0; j < J; ++j) {
                if (next_task_idx[j] < jobs[j].tasks.size()) {
                    const Task& t = jobs[j].tasks[next_task_idx[j]];
                    if (t.machine == best_m) {
                        int est = max(job_ready[j], machine_ready[t.machine]);
                        if (t.duration == 0) {
                            if (est <= min_ect) G.push_back(j);
                        } else {
                            if (est < min_ect) G.push_back(j);
                        }
                    }
                }
            }
            
            if (G.empty()) {
                G.push_back(min_ect_job);
            }
            
            int chosen_j = -1;
            if (G.size() == 1) {
                chosen_j = G[0];
            } else {
                int rule = rng() % 100;
                if (rule < 30) {
                    chosen_j = G[rng() % G.size()];
                } else if (rule < 70) {
                    int max_rem = -1;
                    int count = 0;
                    for (int j : G) {
                        if (job_rem_work[j] > max_rem) {
                            max_rem = job_rem_work[j];
                            chosen_j = j;
                            count = 1;
                        } else if (job_rem_work[j] == max_rem) {
                            count++;
                            if (rng() % count == 0) chosen_j = j;
                        }
                    }
                } else {
                    int min_dur = 2e9;
                    int count = 0;
                    for (int j : G) {
                        int dur = jobs[j].tasks[next_task_idx[j]].duration;
                        if (dur < min_dur) {
                            min_dur = dur;
                            chosen_j = j;
                            count = 1;
                        } else if (dur == min_dur) {
                            count++;
                            if (rng() % count == 0) chosen_j = j;
                        }
                    }
                }
            }
            
            const Task& t = jobs[chosen_j].tasks[next_task_idx[chosen_j]];
            int start = max(job_ready[chosen_j], machine_ready[t.machine]);
            
            start_times[chosen_j][next_task_idx[chosen_j]] = start;
            job_ready[chosen_j] = start + t.duration;
            machine_ready[t.machine] = start + t.duration;
            job_rem_work[chosen_j] -= t.duration;
            
            next_task_idx[chosen_j]++;
            completed_tasks++;
        }
        
        int makespan = 0;
        for (int m_time : machine_ready) makespan = max(makespan, m_time);
        return makespan;
    }
};

int best_makespan = 2e9;
vector<vector<int>> best_start_times;

int dfs_nodes = 0;
auto dfs_start = chrono::high_resolution_clock::now();
bool dfs_timeout = false;

void dfs(ScheduleBuilder& sb, int current_makespan) {
    if (dfs_timeout) return;
    if (current_makespan >= best_makespan) return;
    
    dfs_nodes++;
    if (dfs_nodes % 10000 == 0) {
        auto now = chrono::high_resolution_clock::now();
        if (chrono::duration<double>(now - dfs_start).count() > 1.8) {
            dfs_timeout = true;
            return;
        }
    }
    
    if (sb.completed_tasks == sb.total_tasks) {
        best_makespan = current_makespan;
        best_start_times = sb.start_times;
        return;
    }
    
    int min_ect = 2e9;
    int best_m = -1;
    for (int j = 0; j < sb.J; ++j) {
        if (sb.next_task_idx[j] < sb.jobs[j].tasks.size()) {
            const Task& t = sb.jobs[j].tasks[sb.next_task_idx[j]];
            int est = max(sb.job_ready[j], sb.machine_ready[t.machine]);
            int ect = est + t.duration;
            if (ect < min_ect) {
                min_ect = ect;
                best_m = t.machine;
            }
        }
    }
    
    vector<int> G;
    G.reserve(sb.J);
    for (int j = 0; j < sb.J; ++j) {
        if (sb.next_task_idx[j] < sb.jobs[j].tasks.size()) {
            const Task& t = sb.jobs[j].tasks[sb.next_task_idx[j]];
            if (t.machine == best_m) {
                int est = max(sb.job_ready[j], sb.machine_ready[t.machine]);
                if (t.duration == 0) {
                    if (est <= min_ect) G.push_back(j);
                } else {
                    if (est < min_ect) G.push_back(j);
                }
            }
        }
    }
    
    // Safety
    if (G.empty()) {
        for (int j = 0; j < sb.J; ++j) {
            if (sb.next_task_idx[j] < sb.jobs[j].tasks.size()) {
                const Task& t = sb.jobs[j].tasks[sb.next_task_idx[j]];
                int est = max(sb.job_ready[j], sb.machine_ready[t.machine]);
                if (est + t.duration == min_ect && t.machine == best_m) {
                    G.push_back(j);
                    break;
                }
            }
        }
    }
    
    for (int j : G) {
        const Task& t = sb.jobs[j].tasks[sb.next_task_idx[j]];
        int start = max(sb.job_ready[j], sb.machine_ready[t.machine]);
        
        int prev_jr = sb.job_ready[j];
        int prev_mr = sb.machine_ready[t.machine];
        int prev_rem = sb.job_rem_work[j];
        
        sb.start_times[j][sb.next_task_idx[j]] = start;
        sb.job_ready[j] = start + t.duration;
        sb.machine_ready[t.machine] = start + t.duration;
        sb.job_rem_work[j] -= t.duration;
        sb.next_task_idx[j]++;
        sb.completed_tasks++;
        
        int next_makespan = max(current_makespan, start + t.duration);
        dfs(sb, next_makespan);
        
        sb.completed_tasks--;
        sb.next_task_idx[j]--;
        sb.job_rem_work[j] = prev_rem;
        sb.machine_ready[t.machine] = prev_mr;
        sb.job_ready[j] = prev_jr;
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int J, M;
    if (!(cin >> J >> M)) return 0;
    
    vector<Job> jobs(J);
    int total_tasks = 0;
    for (int i = 0; i < J; ++i) {
        jobs[i].id = i;
        int T;
        cin >> T;
        jobs[i].total_work = 0;
        for (int k = 0; k < T; ++k) {
            Task t;
            t.id = k;
            t.job = i;
            cin >> t.machine >> t.duration;
            jobs[i].tasks.push_back(t);
            jobs[i].total_work += t.duration;
        }
        total_tasks += T;
    }
    
    mt19937 rng(1337);
    
    // Baseline upper bound
    ScheduleBuilder initial_sb(J, M, jobs, total_tasks);
    best_makespan = initial_sb.build(rng);
    best_start_times = initial_sb.start_times;
    
    if (total_tasks <= 25) {
        ScheduleBuilder sb(J, M, jobs, total_tasks);
        dfs_start = chrono::high_resolution_clock::now();
        dfs(sb, 0);
    } else {
        auto start_time = chrono::high_resolution_clock::now();
        int iterations = 0;
        int check_freq = (total_tasks > 10000) ? 1 : 10;
        
        while (true) {
            ScheduleBuilder sb(J, M, jobs, total_tasks);
            int makespan = sb.build(rng);
            if (makespan < best_makespan) {
                best_makespan = makespan;
                best_start_times = sb.start_times;
            }
            iterations++;
            
            if (iterations % check_freq == 0) {
                auto now = chrono::high_resolution_clock::now();
                if (chrono::duration<double>(now - start_time).count() > 1.8) {
                    break;
                }
            }
        }
    }
    
    for (int i = 0; i < J; ++i) {
        for (int k = 0; k < jobs[i].tasks.size(); ++k) {
            cout << best_start_times[i][k] << " " << jobs[i].tasks[k].machine;
            if (k + 1 < jobs[i].tasks.size()) cout << " ";
        }
        cout << "\n";
    }
    
    return 0;
}
