// Maximum Independent Set - exact solver
// Strategy:
//   - Build adjacency bitsets.
//   - Split into connected components, solve each.
//   - Branch-and-bound with reductions:
//       * isolated vertex (deg 0): take it
//       * degree-1 vertex: take it, drop its neighbor
//       * otherwise branch on max-degree vertex (include / exclude)
//   - Simple upper bound prune: current + |R| <= best -> cut.

#include <cstdio>
#include <cstdint>
#include <vector>
#include <algorithm>
#include <queue>
#include <climits>

using namespace std;
typedef uint64_t U64;

static int W = 0;

struct BS {
    vector<U64> a;
    BS() {}
    BS(int w) : a(w, 0) {}
    inline void set(int i)   { a[i>>6] |=  (U64(1) << (i & 63)); }
    inline void clr(int i)   { a[i>>6] &= ~(U64(1) << (i & 63)); }
    int first_bit() const {
        for (size_t i = 0; i < a.size(); ++i)
            if (a[i]) return int(i)*64 + __builtin_ctzll(a[i]);
        return -1;
    }
};

static inline void bs_and_not(BS& dst, const BS& src) {
    for (int i = 0; i < W; ++i) dst.a[i] &= ~src.a[i];
}
static inline int bs_and_popcount(const BS& a, const BS& b) {
    int s = 0;
    for (int i = 0; i < W; ++i) s += __builtin_popcountll(a.a[i] & b.a[i]);
    return s;
}

static vector<BS> adj;
static vector<BS> closed_nb;

static int best_size_global;
static vector<int> best_set_global;
static vector<int> current_set;

static void solve(BS R, int currentSize) {
    while (true) {
        int first = R.first_bit();
        if (first < 0) {
            if (currentSize > best_size_global) {
                best_size_global = currentSize;
                best_set_global = current_set;
            }
            return;
        }
        int min_v = -1, min_d = INT_MAX;
        int max_v = -1, max_d = -1;
        int countR = 0;
        for (int wi = 0; wi < W; ++wi) {
            U64 x = R.a[wi];
            while (x) {
                int b = __builtin_ctzll(x);
                x &= x - 1;
                int u = wi*64 + b;
                ++countR;
                int d = bs_and_popcount(adj[u], R);
                if (d < min_d) { min_d = d; min_v = u; }
                if (d > max_d) { max_d = d; max_v = u; }
            }
        }
        if (currentSize + countR <= best_size_global) return;

        if (min_d == 0) {
            current_set.push_back(min_v);
            R.clr(min_v);
            ++currentSize;
            continue;
        }
        if (min_d == 1) {
            current_set.push_back(min_v);
            int nb = -1;
            for (int wi = 0; wi < W; ++wi) {
                U64 x = adj[min_v].a[wi] & R.a[wi];
                if (x) { nb = wi*64 + __builtin_ctzll(x); break; }
            }
            R.clr(min_v);
            if (nb >= 0) R.clr(nb);
            ++currentSize;
            continue;
        }

        int v_pick = max_v;
        {
            BS R2 = R;
            bs_and_not(R2, closed_nb[v_pick]);
            current_set.push_back(v_pick);
            solve(R2, currentSize + 1);
            current_set.pop_back();
        }
        {
            BS R2 = R;
            R2.clr(v_pick);
            solve(R2, currentSize);
        }
        return;
    }
}

int main() {
    int n, m;
    if (scanf("%d %d", &n, &m) != 2) return 0;
    W = (n + 63) / 64;
    if (W == 0) W = 1;

    adj.assign(n, BS(W));
    closed_nb.assign(n, BS(W));
    for (int i = 0; i < m; ++i) {
        int u, v;
        if (scanf("%d %d", &u, &v) != 2) break;
        if (u == v) continue;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        adj[u].set(v);
        adj[v].set(u);
    }
    for (int v = 0; v < n; ++v) {
        closed_nb[v] = adj[v];
        closed_nb[v].set(v);
    }

    vector<int> comp(n, -1);
    vector<vector<int>> comps;
    for (int s = 0; s < n; ++s) {
        if (comp[s] != -1) continue;
        comp[s] = (int)comps.size();
        vector<int> nodes;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int u = q.front(); q.pop();
            nodes.push_back(u);
            for (int wi = 0; wi < W; ++wi) {
                U64 x = adj[u].a[wi];
                while (x) {
                    int b = __builtin_ctzll(x);
                    x &= x - 1;
                    int w = wi*64 + b;
                    if (comp[w] == -1) {
                        comp[w] = comp[s];
                        q.push(w);
                    }
                }
            }
        }
        comps.push_back(std::move(nodes));
    }

    vector<int> final_set;
    for (auto& nodes : comps) {
        BS R(W);
        for (int v : nodes) R.set(v);
        best_size_global = 0;
        best_set_global.clear();
        current_set.clear();
        solve(R, 0);
        for (int v : best_set_global) final_set.push_back(v);
    }

    printf("%d\n", (int)final_set.size());
    for (size_t i = 0; i < final_set.size(); ++i) {
        if (i) printf(" ");
        printf("%d", final_set[i]);
    }
    printf("\n");
    return 0;
}
