#include <cstdlib>

using namespace std;
using Clock = chrono::steady_clock;

static double seconds_since(Clock::time_point t0) {
    return chrono::duration<double>(Clock::now() - t0).count();
}

struct Pt { float x, y; };

static inline float dist(const Pt& a, const Pt& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}
static inline double distd(const Pt& a, const Pt& b) {
    double dx = a.x - b.x, dy = a.y - b.y;
    return sqrt(dx*dx + dy*dy);
}

// ---------------- Fast stdin reader ----------------
struct Reader {
    static const size_t BUF = 1 << 20;
    char buf[BUF];
    size_t p = 0, e = 0;
    FILE* f;
    Reader(FILE* fp): f(fp) {}
    inline int gc() {
        if (p >= e) {
            e = fread(buf, 1, BUF, f);
            p = 0;
            if (e == 0) return -1;
        }
        return (unsigned char)buf[p++];
    }
    bool read_int(long long& v) {
        int c; bool neg = false;
        do { c = gc(); if (c < 0) return false; } while (c != '-' && (c < '0' || c > '9'));
        if (c == '-') { neg = true; c = gc(); }
        v = 0;
        while (c >= '0' && c <= '9') { v = v*10 + (c-'0'); c = gc(); }
        if (neg) v = -v;
        return true;
    }
    bool read_double(double& v) {
        int c; bool neg=false;
        do { c = gc(); if (c < 0) return false; } while (c != '-' && c != '+' && c != '.' && (c < '0' || c > '9'));
        if (c == '-') { neg=true; c=gc(); }
        else if (c == '+') { c=gc(); }
        double x = 0;
        while (c >= '0' && c <= '9') { x = x*10 + (c-'0'); c = gc(); }
        if (c == '.') {
            c = gc();
            double f = 1.0;
            while (c >= '0' && c <= '9') { f *= 0.1; x += (c-'0')*f; c = gc(); }
        }
        if (c == 'e' || c == 'E') {
            c = gc(); bool en=false;
            if (c == '-') { en=true; c=gc(); } else if (c=='+') c=gc();
            int ex = 0;
            while (c >= '0' && c <= '9') { ex = ex*10 + (c-'0'); c = gc(); }
            double m = 1.0;
            for (int i = 0; i < ex; ++i) m *= 10.0;
            if (en) x /= m; else x *= m;
        }
        v = neg ? -x : x;
        return true;
    }
};

// ---------------- Output writer ----------------
struct Writer {
    static const size_t BUF = 1 << 20;
    char buf[BUF];
    size_t p = 0;
    FILE* f;
    Writer(FILE* fp): f(fp) {}
    ~Writer() { flush(); }
    void flush() { if (p) { fwrite(buf,1,p,f); p=0; } }
    inline void put(char c) { if (p>=BUF) flush(); buf[p++]=c; }
    void write_int(int v) {
        if (v == 0) { put('0'); return; }
        if (v < 0) { put('-'); v = -v; }
        char tmp[12]; int n=0;
        while (v) { tmp[n++] = '0' + (v%10); v/=10; }
        while (n--) put(tmp[n]);
    }
};

// ---------------- Hilbert curve order for huge N ----------------
static uint64_t hilbert_d(uint32_t order, uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = order/2; s > 0; s >>= 1) {
        uint32_t rx = (x & s) ? 1u : 0u;
        uint32_t ry = (y & s) ? 1u : 0u;
        d += (uint64_t)s * (uint64_t)s * ((3u * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) { x = s - 1 - x; y = s - 1 - y; }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

// ---------------- Grid for nearest-neighbor search ----------------
struct Grid {
    int gx, gy;
    float minx, miny, cell;
    vector<vector<int>> cells;
    vector<char> used;
    const vector<Pt>* pts;

    void build(const vector<Pt>& P, float targetPerCell = 4.0f) {
        pts = &P;
        int n = (int)P.size();
        minx = P[0].x; miny = P[0].y;
        float maxx = minx, maxy = miny;
        for (auto& p : P) {
            if (p.x < minx) minx = p.x; if (p.x > maxx) maxx = p.x;
            if (p.y < miny) miny = p.y; if (p.y > maxy) maxy = p.y;
        }
        float w = maxx - minx, h = maxy - miny;
        if (w < 1e-9f) w = 1e-9f; if (h < 1e-9f) h = 1e-9f;
        double area = (double)w * (double)h;
        double cellsWanted = max(1.0, n / (double)targetPerCell);
        double side = sqrt(area / cellsWanted);
        if (side < 1e-9) side = 1e-9;
        gx = max(1, (int)ceil(w / side));
        gy = max(1, (int)ceil(h / side));
        while ((long long)gx * gy > 8LL * n + 1024) {
            gx = max(1, gx*3/4);
            gy = max(1, gy*3/4);
            if (gx == 1 && gy == 1) break;
        }
        cell = max(w/gx, h/gy);
        cells.assign((size_t)gx*gy, {});
        used.assign(n, 0);
        for (int i = 0; i < n; ++i) {
            int cx = min(gx-1, max(0, (int)((P[i].x - minx) / cell)));
            int cy = min(gy-1, max(0, (int)((P[i].y - miny) / cell)));
            cells[(size_t)cy*gx + cx].push_back(i);
        }
    }
    inline void cell_of(const Pt& p, int& cx, int& cy) const {
        cx = min(gx-1, max(0, (int)((p.x - minx) / cell)));
        cy = min(gy-1, max(0, (int)((p.y - miny) / cell)));
    }
    void remove(int idx) {
        used[idx] = 1;
        int cx, cy; cell_of((*pts)[idx], cx, cy);
        auto& v = cells[(size_t)cy*gx + cx];
        for (size_t i = 0; i < v.size(); ++i) {
            if (v[i] == idx) { v[i] = v.back(); v.pop_back(); return; }
        }
    }
    int nearest(const Pt& p) {
        int cx, cy; cell_of(p, cx, cy);
        int best = -1;
        float bestD = 1e38f;
        int r = 0;
        int maxR = max(gx, gy);
        int foundR = -1;
        while (r <= maxR) {
            int x0 = max(0, cx - r), x1 = min(gx-1, cx + r);
            int y0 = max(0, cy - r), y1 = min(gy-1, cy + r);
            for (int y = y0; y <= y1; ++y) {
                for (int x = x0; x <= x1; ++x) {
                    if (r > 0 && abs(x-cx) != r && abs(y-cy) != r) continue;
                    auto& v = cells[(size_t)y*gx + x];
                    for (int id : v) {
                        float dx = (*pts)[id].x - p.x;
                        float dy = (*pts)[id].y - p.y;
                        float d2 = dx*dx + dy*dy;
                        if (d2 < bestD) { bestD = d2; best = id; }
                    }
                }
            }
            if (best != -1) {
                if (foundR == -1) foundR = r;
                if (r >= foundR + 1) break;
            }
            ++r;
        }
        return best;
    }
};

// ---------------- 2-opt with neighbor lists ----------------
static void two_opt_neighbors(vector<int>& tour, const vector<Pt>& P,
                              const vector<vector<int>>& nbr, double timeLimitAbs, Clock::time_point t0) {
    int n = (int)tour.size();
    vector<int> pos(n);
    for (int i = 0; i < n; ++i) pos[tour[i]] = i;

    auto dseg = [&](int a, int b) { return dist(P[a], P[b]); };

    bool improved = true;
    int iter = 0;
    int passes = 0;
    while (improved && passes < 50) {
        improved = false;
        ++passes;
        for (int i = 0; i < n; ++i) {
            if ((iter & 1023) == 0 && seconds_since(t0) > timeLimitAbs) return;
            ++iter;
            int a = tour[i];
            int b = tour[(i+1) % n];
            float dab = dseg(a, b);
            for (int c : nbr[a]) {
                if (c == a || c == b) continue;
                int j = pos[c];
                int d = tour[(j+1) % n];
                if (d == a) continue;
                float dac = dseg(a, c);
                if (dac >= dab) break;
                float dcd = dseg(c, d);
                float dbd = dseg(b, d);
                if (dab + dcd > dac + dbd + 1e-7f) {
                    int l = (i+1) % n;
                    int r = j;
                    int len;
                    if (r >= l) len = r - l + 1;
                    else len = n - l + r + 1;
                    int half = len / 2;
                    int li = l, ri = r;
                    for (int k = 0; k < half; ++k) {
                        swap(tour[li], tour[ri]);
                        pos[tour[li]] = li;
                        pos[tour[ri]] = ri;
                        li = (li + 1) % n;
                        ri = (ri - 1 + n) % n;
                    }
                    improved = true;
                    break;
                }
            }
        }
    }
}

// ---------------- Or-opt (segments of len 1,2,3) ----------------
static void or_opt(vector<int>& tour, const vector<Pt>& P,
                   const vector<vector<int>>& nbr, double timeLimitAbs, Clock::time_point t0) {
    int n = (int)tour.size();
    vector<int> nxt(n), prv(n);
    for (int i = 0; i < n; ++i) {
        nxt[tour[i]] = tour[(i+1) % n];
        prv[tour[i]] = tour[(i-1+n) % n];
    }
    auto D = [&](int a, int b) { return dist(P[a], P[b]); };
    bool improved = true;
    int safety = 0;
    while (improved && safety++ < 10) {
        improved = false;
        for (int seglen = 1; seglen <= 3; ++seglen) {
            for (int a = 0; a < n; ++a) {
                if ((a & 4095) == 0 && seconds_since(t0) > timeLimitAbs) goto done;
                int s0 = a;
                int s1 = s0;
                for (int k = 1; k < seglen; ++k) s1 = nxt[s1];
                int p = prv[s0];
                int q = nxt[s1];
                if (p == s1 || q == s0 || p == s0) continue;
                float removeCost = D(p, s0) + D(s1, q) - D(p, q);
                if (removeCost <= 1e-7f) continue;
                float bestGain = 0;
                int bestC = -1, bestD = -1;
                bool bestRev = false;
                for (int c : nbr[s0]) {
                    if (c == s0 || c == s1 || c == p) continue;
                    bool inside = false;
                    int x = s0;
                    for (int k = 0; k < seglen; ++k) { if (x == c) { inside = true; break; } x = nxt[x]; }
                    if (inside) continue;
                    int d = nxt[c];
                    if (d == s0) continue;
                    float addCost = D(c, s0) + D(s1, d) - D(c, d);
                    float gain = removeCost - addCost;
                    if (gain > bestGain + 1e-7f) {
                        bestGain = gain; bestC = c; bestD = d; bestRev = false;
                    }
                    float addCostR = D(c, s1) + D(s0, d) - D(c, d);
                    float gainR = removeCost - addCostR;
                    if (gainR > bestGain + 1e-7f) {
                        bestGain = gainR; bestC = c; bestD = d; bestRev = true;
                    }
                }
                if (bestC != -1) {
                    nxt[p] = q; prv[q] = p;
                    int c = bestC;
                    int dd = bestD;
                    if (bestRev) {
                        vector<int> seg; seg.reserve(seglen);
                        int y = s0;
                        for (int k = 0; k < seglen; ++k) { seg.push_back(y); y = nxt[y]; }
                        for (int k = 0; k < seglen-1; ++k) {
                            int u = seg[seglen-1-k];
                            int v = seg[seglen-2-k];
                            nxt[u] = v; prv[v] = u;
                        }
                        int head = seg[seglen-1];
                        int tail = seg[0];
                        nxt[c] = head; prv[head] = c;
                        nxt[tail] = dd; prv[dd] = tail;
                    } else {
                        nxt[c] = s0; prv[s0] = c;
                        nxt[s1] = dd; prv[dd] = s1;
                    }
                    improved = true;
                }
            }
        }
    }
done:
    int cur = 0;
    for (int i = 0; i < n; ++i) { tour[i] = cur; cur = nxt[cur]; }
}

// ---------------- Build k-NN neighbor lists ----------------
static vector<vector<int>> build_knn(const vector<Pt>& P, int k) {
    int n = (int)P.size();
    Grid g; g.build(P, 4.0f);
    vector<vector<int>> nbr(n);
    vector<pair<float,int>> cand;
    cand.reserve(64);
    for (int i = 0; i < n; ++i) {
        cand.clear();
        int cx, cy; g.cell_of(P[i], cx, cy);
        int r = 0;
        while (r <= max(g.gx, g.gy)) {
            int x0 = max(0, cx - r), x1 = min(g.gx-1, cx + r);
            int y0 = max(0, cy - r), y1 = min(g.gy-1, cy + r);
            for (int y = y0; y <= y1; ++y) {
                for (int x = x0; x <= x1; ++x) {
                    if (r > 0 && abs(x-cx) != r && abs(y-cy) != r) continue;
                    auto& v = g.cells[(size_t)y*g.gx + x];
                    for (int id : v) {
                        if (id == i) continue;
                        float dx = P[id].x - P[i].x;
                        float dy = P[id].y - P[i].y;
                        cand.push_back({dx*dx+dy*dy, id});
                    }
                }
            }
            if ((int)cand.size() >= k && r >= 1) {
                if (r >= 2 || (int)cand.size() >= k*2) break;
            }
            ++r;
        }
        sort(cand.begin(), cand.end());
        int kk = min((int)cand.size(), k);
        nbr[i].resize(kk);
        for (int j = 0; j < kk; ++j) nbr[i][j] = cand[j].second;
    }
    return nbr;
}

// ---------------- Nearest-neighbor tour ----------------
static vector<int> nn_tour(const vector<Pt>& P) {
    int n = (int)P.size();
    vector<int> tour; tour.reserve(n);
    Grid g; g.build(P, 4.0f);
    int cur = 0;
    tour.push_back(0);
    g.remove(0);
    for (int i = 1; i < n; ++i) {
        int nx = g.nearest(P[cur]);
        if (nx < 0) {
            float bd = 1e38f; int bi = -1;
            for (int j = 0; j < n; ++j) if (!g.used[j]) {
                float dx = P[j].x - P[cur].x, dy = P[j].y - P[cur].y;
                float d2 = dx*dx+dy*dy;
                if (d2 < bd) { bd = d2; bi = j; }
            }
            nx = bi;
        }
        tour.push_back(nx);
        g.remove(nx);
        cur = nx;
    }
    return tour;
}

// ---------------- Hilbert tour for huge N ----------------
static vector<int> hilbert_tour(const vector<Pt>& P) {
    int n = (int)P.size();
    double minx = P[0].x, miny = P[0].y, maxx = minx, maxy = miny;
    for (auto& p : P) {
        if (p.x < minx) minx = p.x; if (p.x > maxx) maxx = p.x;
        if (p.y < miny) miny = p.y; if (p.y > maxy) maxy = p.y;
    }
    double w = maxx - minx, h = maxy - miny;
    if (w < 1e-9) w = 1e-9; if (h < 1e-9) h = 1e-9;
    uint32_t order = 1u << 16;
    vector<pair<uint64_t,int>> keys(n);
    for (int i = 0; i < n; ++i) {
        uint32_t x = (uint32_t)((P[i].x - minx) / w * (order - 1));
        uint32_t y = (uint32_t)((P[i].y - miny) / h * (order - 1));
        if (x >= order) x = order-1;
        if (y >= order) y = order-1;
        keys[i] = { hilbert_d(order, x, y), i };
    }
    sort(keys.begin(), keys.end());
    vector<int> tour(n);
    for (int i = 0; i < n; ++i) tour[i] = keys[i].second;
    int idx0 = -1;
    for (int i = 0; i < n; ++i) if (tour[i] == 0) { idx0 = i; break; }
    if (idx0 > 0) rotate(tour.begin(), tour.begin()+idx0, tour.end());
    return tour;
}

// ---------------- Brute force for very small N ----------------
static vector<int> brute_force(const vector<Pt>& P) {
    int n = (int)P.size();
    vector<int> perm(n-1);
    for (int i = 0; i < n-1; ++i) perm[i] = i+1;
    vector<int> best = perm;
    double bestD = 1e300;
    do {
        double tot = distd(P[0], P[perm[0]]);
        for (int i = 0; i+1 < n-1; ++i) tot += distd(P[perm[i]], P[perm[i+1]]);
        tot += distd(P[perm[n-2]], P[0]);
        if (tot < bestD) { bestD = tot; best = perm; }
    } while (next_permutation(perm.begin(), perm.end()));
    vector<int> out(n);
    out[0] = 0;
    for (int i = 0; i < n-1; ++i) out[i+1] = best[i];
    return out;
}

// ---------------- Held-Karp DP for small N ----------------
static vector<int> held_karp(const vector<Pt>& P) {
    int n = (int)P.size();
    vector<vector<double>> D(n, vector<double>(n));
    for (int i = 0; i < n; ++i) for (int j = 0; j < n; ++j) D[i][j] = distd(P[i], P[j]);
    int FS = 1 << n;
    vector<vector<double>> dp(FS, vector<double>(n, 1e300));
    vector<vector<int>> par(FS, vector<int>(n, -1));
    dp[1][0] = 0.0;
    for (int mask = 1; mask < FS; ++mask) {
        if (!(mask & 1)) continue;
        for (int u = 0; u < n; ++u) {
            if (!(mask & (1<<u))) continue;
            double cur = dp[mask][u];
            if (cur >= 1e299) continue;
            for (int v = 1; v < n; ++v) {
                if (mask & (1<<v)) continue;
                int nm = mask | (1<<v);
                double nd = cur + D[u][v];
                if (nd < dp[nm][v]) { dp[nm][v] = nd; par[nm][v] = u; }
            }
        }
    }
    int full = FS - 1;
    int end = 0; double bestD = 1e300;
    for (int u = 1; u < n; ++u) {
        double d = dp[full][u] + D[u][0];
        if (d < bestD) { bestD = d; end = u; }
    }
    vector<int> tour;
    int m = full, c = end;
    while (c != -1) { tour.push_back(c); int p = par[m][c]; m ^= (1<<c); c = p; }
    reverse(tour.begin(), tour.end());
    return tour;
}

int main() {
    auto t0 = Clock::now();
    const double TIME_BUDGET = 270.0;

    Reader R(stdin);
    long long Nll;
    if (!R.read_int(Nll)) return 0;
    int N = (int)Nll;
    if (N <= 0) return 0;
    if (N == 1) { Writer W(stdout); W.put('0'); W.put('\n'); return 0; }

    vector<Pt> P((size_t)N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        if (!R.read_double(x)) x = 0;
        if (!R.read_double(y)) y = 0;
        P[i].x = (float)x; P[i].y = (float)y;
    }

    vector<int> tour;

    if (N <= 11) {
        tour = brute_force(P);
    } else if (N <= 18) {
        tour = held_karp(P);
    } else if (N <= 3000) {
        tour = nn_tour(P);
        int k = min(N-1, 20);
        auto nbr = build_knn(P, k);
        double now = seconds_since(t0);
        double total = TIME_BUDGET - now - 5.0;
        if (total < 5.0) total = 5.0;
        two_opt_neighbors(tour, P, nbr, now + total * 0.5, t0);
        or_opt(tour, P, nbr, seconds_since(t0) + total * 0.3, t0);
        two_opt_neighbors(tour, P, nbr, seconds_since(t0) + total * 0.15, t0);
    } else if (N <= 200000) {
        tour = nn_tour(P);
        if (seconds_since(t0) < TIME_BUDGET - 30.0) {
            int k = (N <= 20000) ? 15 : 10;
            auto nbr = build_knn(P, k);
            double remaining = TIME_BUDGET - seconds_since(t0) - 10.0;
            two_opt_neighbors(tour, P, nbr, seconds_since(t0) + remaining * 0.6, t0);
            if (seconds_since(t0) < TIME_BUDGET - 15.0)
                or_opt(tour, P, nbr, seconds_since(t0) + max(2.0, TIME_BUDGET - seconds_since(t0) - 8.0), t0);
        }
    } else if (N <= 2000000) {
        tour = nn_tour(P);
        if (seconds_since(t0) < TIME_BUDGET - 60.0 && N <= 800000) {
            int k = 6;
            auto nbr = build_knn(P, k);
            double remaining = TIME_BUDGET - seconds_since(t0) - 20.0;
            if (remaining > 5.0) {
                two_opt_neighbors(tour, P, nbr, seconds_since(t0) + remaining, t0);
            }
        }
    } else {
        tour = hilbert_tour(P);
    }

    if (!tour.empty() && tour[0] != 0) {
        int idx0 = -1;
        for (int i = 0; i < (int)tour.size(); ++i) if (tour[i] == 0) { idx0 = i; break; }
        if (idx0 > 0) rotate(tour.begin(), tour.begin()+idx0, tour.end());
    }

    Writer W(stdout);
    for (size_t i = 0; i < tour.size(); ++i) {
        if (i) W.put(' ');
        W.write_int(tour[i]);
    }
    W.put('\n');
    return 0;
}
