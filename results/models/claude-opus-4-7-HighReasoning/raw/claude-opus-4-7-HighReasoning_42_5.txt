OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 7

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_2 = OpConstant %uint 2
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
 %float_0_05 = OpConstant %float 0.05
 %float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

       %p_p0 = OpAccessChain %ptr_in_v3 %worldPos %int_0
         %p0 = OpLoad %v3float %p_p0
       %p_p1 = OpAccessChain %ptr_in_v3 %worldPos %int_1
         %p1 = OpLoad %v3float %p_p1

       %p_c0 = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %p_c0
       %p_c1 = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %p_c1

        %dir = OpFSub %v3float %p1 %p0

         %dx = OpCompositeExtract %float %dir 0
         %dy = OpCompositeExtract %float %dir 1
     %neg_dy = OpFNegate %float %dy
   %perp_raw = OpCompositeConstruct %v2float %neg_dy %dx
     %perp_n = OpExtInst %v2float %1 Normalize %perp_raw
     %perp_x = OpCompositeExtract %float %perp_n 0
     %perp_y = OpCompositeExtract %float %perp_n 1

     %dir075 = OpVectorTimesScalar %v3float %dir %float_0_75
        %mid = OpFAdd %v3float %p0 %dir075

   %ox_shaft = OpFMul %float %perp_x %float_0_015
   %oy_shaft = OpFMul %float %perp_y %float_0_015
    %ox_head = OpFMul %float %perp_x %float_0_05
    %oy_head = OpFMul %float %perp_y %float_0_05

        %p0x = OpCompositeExtract %float %p0 0
        %p0y = OpCompositeExtract %float %p0 1
        %p0z = OpCompositeExtract %float %p0 2
         %mx = OpCompositeExtract %float %mid 0
         %my = OpCompositeExtract %float %mid 1
         %mz = OpCompositeExtract %float %mid 2
        %p1x = OpCompositeExtract %float %p1 0
        %p1y = OpCompositeExtract %float %p1 1
        %p1z = OpCompositeExtract %float %p1 2

        %v1x = OpFAdd %float %p0x %ox_shaft
        %v1y = OpFAdd %float %p0y %oy_shaft
         %v1 = OpCompositeConstruct %v4float %v1x %v1y %p0z %float_1

        %v2x = OpFSub %float %p0x %ox_shaft
        %v2y = OpFSub %float %p0y %oy_shaft
         %v2 = OpCompositeConstruct %v4float %v2x %v2y %p0z %float_1

        %v3x = OpFAdd %float %mx %ox_shaft
        %v3y = OpFAdd %float %my %oy_shaft
         %v3 = OpCompositeConstruct %v4float %v3x %v3y %mz %float_1

        %v4x = OpFSub %float %mx %ox_shaft
        %v4y = OpFSub %float %my %oy_shaft
         %v4 = OpCompositeConstruct %v4float %v4x %v4y %mz %float_1

   %pos_out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out1 %v1
               OpStore %outColor %c0
               OpEmitVertex

   %pos_out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out2 %v2
               OpStore %outColor %c0
               OpEmitVertex

   %pos_out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out3 %v3
               OpStore %outColor %c0
               OpEmitVertex

   %pos_out4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out4 %v4
               OpStore %outColor %c0
               OpEmitVertex
               OpEndPrimitive

        %v5x = OpFAdd %float %mx %ox_head
        %v5y = OpFAdd %float %my %oy_head
         %v5 = OpCompositeConstruct %v4float %v5x %v5y %mz %float_1

        %v6x = OpFSub %float %mx %ox_head
        %v6y = OpFSub %float %my %oy_head
         %v6 = OpCompositeConstruct %v4float %v6x %v6y %mz %float_1

         %v7 = OpCompositeConstruct %v4float %p1x %p1y %p1z %float_1

   %pos_out5 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out5 %v5
               OpStore %outColor %c1
               OpEmitVertex

   %pos_out6 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out6 %v6
               OpStore %outColor %c1
               OpEmitVertex

   %pos_out7 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pos_out7 %v7
               OpStore %outColor %c1
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
