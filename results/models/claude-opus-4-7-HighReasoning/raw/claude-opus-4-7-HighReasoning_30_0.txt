import sys
import numpy as np


def main():
    data = sys.stdin.buffer.read().split()
    n = int(data[0])
    m = int(data[1])

    if m == 0:
        sys.stdout.buffer.write(b"SAT\n" + b" ".join([b"1"] * n) + b"\n")
        return

    lits = np.array(data[2:2 + 3 * m], dtype=np.int64)
    vars_idx = (np.abs(lits) - 1).astype(np.int64)
    signs = (lits > 0)  # True means literal is positive

    # Initial assignment via majority polarity vote.
    pos_count = np.bincount(vars_idx[signs], minlength=n)
    neg_count = np.bincount(vars_idx[~signs], minlength=n)
    assign = pos_count >= neg_count  # bool array of length n

    vars_r = vars_idx.reshape(m, 3)
    signs_r = signs.reshape(m, 3)

    rng = np.random.default_rng(12345)

    # Local search: repeatedly find unsatisfied clauses and flip a variable in each.
    max_iter = 200
    for it in range(max_iter):
        lit_val = assign[vars_r] == signs_r  # (m, 3) bool
        clause_sat = lit_val.any(axis=1)
        unsat_mask = ~clause_sat
        num_unsat = int(unsat_mask.sum())
        if num_unsat == 0:
            break

        unsat_vars = vars_r[unsat_mask]  # (k, 3)
        k = unsat_vars.shape[0]

        if it % 3 == 0:
            # Greedy-ish: prefer flipping the variable that currently appears
            # least often with its current polarity (a cheap proxy).
            cur_score = np.where(assign[unsat_vars], pos_count[unsat_vars], neg_count[unsat_vars])
            cols = cur_score.argmin(axis=1)
        else:
            cols = rng.integers(0, 3, size=k)

        flip_vars = unsat_vars[np.arange(k), cols]
        flip_counts = np.bincount(flip_vars, minlength=n)
        assign = assign ^ (flip_counts % 2 == 1).astype(bool)

    out = np.where(assign, b"1", b"0")
    sys.stdout.buffer.write(b"SAT\n" + b" ".join(out.tolist()) + b"\n")


main()
