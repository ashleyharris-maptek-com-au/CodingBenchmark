using System;
using System.Collections.Generic;
using System.IO;
using System.Text;
using System.Diagnostics;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
        public long Vol;
    }

    struct Placed
    {
        public int Idx;
        public int X, Y, Z;
        public int W, H, DD;
        public int X2 { get { return X + W; } }
        public int Y2 { get { return Y + H; } }
        public int Z2 { get { return Z + DD; } }
    }

    struct EP
    {
        public int X, Y, Z;
    }

    static int CW, CH, CD;
    static Stopwatch globalSw;

    static void Main()
    {
        globalSw = Stopwatch.StartNew();
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        int N = int.Parse(tokens[p++]);
        CW = int.Parse(tokens[p++]);
        CH = int.Parse(tokens[p++]);
        CD = int.Parse(tokens[p++]);

        var boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            int w = int.Parse(tokens[p++]);
            int h = int.Parse(tokens[p++]);
            int d = int.Parse(tokens[p++]);
            boxes[i] = new Box { Index = i, W = w, H = h, D = d, Vol = (long)w * h * d };
        }

        var strategies = new int[] { 0, 1, 2, 3 };
        List<Placed> best = null;
        double totalBudget = 27.0;

        for (int s = 0; s < strategies.Length; s++)
        {
            double elapsed = globalSw.Elapsed.TotalSeconds;
            if (elapsed > totalBudget - 1.0) break;
            double perRemain = (totalBudget - elapsed) / (strategies.Length - s);
            double deadline = elapsed + perRemain;
            var copy = (Box[])boxes.Clone();
            var result = Pack(copy, strategies[s], deadline);
            if (best == null || result.Count > best.Count) best = result;
            if (best.Count == N) break;
        }

        if (best == null) best = new List<Placed>();

        var sb = new StringBuilder();
        sb.Append(best.Count).Append('\n');
        foreach (var pl in best)
        {
            sb.Append(pl.Idx).Append(' ').Append(pl.X).Append(' ').Append(pl.Y).Append(' ').Append(pl.Z).Append('\n');
        }
        Console.Out.Write(sb.ToString());
    }

    static List<Placed> Pack(Box[] boxes, int sortMode, double deadline)
    {
        Array.Sort(boxes, (a, b) =>
        {
            switch (sortMode)
            {
                case 0: return b.Vol.CompareTo(a.Vol);
                case 1:
                    int ma = Math.Max(Math.Max(a.W, a.H), a.D);
                    int mb = Math.Max(Math.Max(b.W, b.H), b.D);
                    return mb.CompareTo(ma);
                case 2: return b.H.CompareTo(a.H);
                case 3: return ((long)b.W * b.D).CompareTo((long)a.W * a.D);
                default: return b.Vol.CompareTo(a.Vol);
            }
        });

        var result = new List<Placed>();
        var eps = new List<EP> { new EP { X = 0, Y = 0, Z = 0 } };

        int N = boxes.Length;
        bool timeUp = false;

        for (int i = 0; i < N; i++)
        {
            if ((i & 63) == 0)
            {
                if (globalSw.Elapsed.TotalSeconds > deadline) { timeUp = true; }
            }

            var box = boxes[i];

            int bestEpIdx = -1;
            int bestRotW = 0, bestRotH = 0, bestRotD = 0;
            long bestScore = long.MaxValue;

            int[][] rots = new int[][]
            {
                new[]{box.W, box.H, box.D},
                new[]{box.W, box.D, box.H},
                new[]{box.H, box.W, box.D},
                new[]{box.H, box.D, box.W},
                new[]{box.D, box.W, box.H},
                new[]{box.D, box.H, box.W},
            };

            // If time is up, only try base orientation at first feasible EP (fast finish)
            int rotLimit = timeUp ? 1 : 6;

            for (int e = 0; e < eps.Count; e++)
            {
                var ep = eps[e];
                for (int rIdx = 0; rIdx < rotLimit; rIdx++)
                {
                    var r = rots[rIdx];
                    int rw = r[0], rh = r[1], rd = r[2];
                    if (ep.X + rw > CW || ep.Y + rh > CH || ep.Z + rd > CD) continue;
                    if (Overlaps(ep.X, ep.Y, ep.Z, rw, rh, rd, result)) continue;

                    long score = (long)ep.Y * (long)CW * (long)CD + (long)ep.Z * (long)CW + ep.X;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestEpIdx = e;
                        bestRotW = rw; bestRotH = rh; bestRotD = rd;
                    }
                    if (timeUp) break;
                }
                if (timeUp && bestEpIdx >= 0) break;
            }

            if (bestEpIdx >= 0)
            {
                var ep = eps[bestEpIdx];
                var pl = new Placed { Idx = box.Index, X = ep.X, Y = ep.Y, Z = ep.Z, W = bestRotW, H = bestRotH, DD = bestRotD };
                result.Add(pl);
                eps.RemoveAt(bestEpIdx);

                AddEP(eps, ep.X + bestRotW, ep.Y, ep.Z);
                AddEP(eps, ep.X, ep.Y + bestRotH, ep.Z);
                AddEP(eps, ep.X, ep.Y, ep.Z + bestRotD);

                for (int k = eps.Count - 1; k >= 0; k--)
                {
                    if (PointInside(eps[k], result))
                    {
                        eps.RemoveAt(k);
                    }
                }

                // Cap EP list size if it grows too large
                if (eps.Count > 5000)
                {
                    eps.Sort((a, b) =>
                    {
                        int c = a.Y.CompareTo(b.Y);
                        if (c != 0) return c;
                        c = a.Z.CompareTo(b.Z);
                        if (c != 0) return c;
                        return a.X.CompareTo(b.X);
                    });
                    eps.RemoveRange(3000, eps.Count - 3000);
                }
            }
        }

        return result;
    }

    static void AddEP(List<EP> eps, int x, int y, int z)
    {
        if (x >= CW || y >= CH || z >= CD) return;
        for (int i = 0; i < eps.Count; i++)
        {
            if (eps[i].X == x && eps[i].Y == y && eps[i].Z == z) return;
        }
        eps.Add(new EP { X = x, Y = y, Z = z });
    }

    static bool PointInside(EP ep, List<Placed> placed)
    {
        for (int i = 0; i < placed.Count; i++)
        {
            var p = placed[i];
            if (ep.X >= p.X && ep.X < p.X2 &&
                ep.Y >= p.Y && ep.Y < p.Y2 &&
                ep.Z >= p.Z && ep.Z < p.Z2) return true;
        }
        return false;
    }

    static bool Overlaps(int x, int y, int z, int w, int h, int d, List<Placed> placed)
    {
        int x2 = x + w, y2 = y + h, z2 = z + d;
        for (int i = 0; i < placed.Count; i++)
        {
            var p = placed[i];
            if (x < p.X2 && x2 > p.X &&
                y < p.Y2 && y2 > p.Y &&
                z < p.Z2 && z2 > p.Z) return true;
        }
        return false;
    }
}
