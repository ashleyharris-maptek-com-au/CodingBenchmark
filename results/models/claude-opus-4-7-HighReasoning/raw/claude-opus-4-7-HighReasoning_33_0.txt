using System;
using System.Diagnostics;
using System.IO;
using System.Text;

class QAP
{
    static int N;
    static int[][] F;
    static int[][] D;
    static Random rng = new Random(12345);
    static Stopwatch sw = new Stopwatch();
    static long timeLimitMs = 270000;

    static void Main()
    {
        sw.Start();
        ReadInput();

        int[] best = GreedyConstruct();
        long bestCost = ComputeCost(best);

        int[] cur = (int[])best.Clone();
        long curCost = LocalSearch(cur, bestCost);
        if (curCost < bestCost) { bestCost = curCost; best = (int[])cur.Clone(); }

        int restarts;
        if (N <= 30) restarts = 1000;
        else if (N <= 80) restarts = 200;
        else if (N <= 200) restarts = 50;
        else if (N <= 500) restarts = 10;
        else if (N <= 1000) restarts = 3;
        else restarts = 2;

        for (int r = 0; r < restarts; r++)
        {
            if (sw.ElapsedMilliseconds > timeLimitMs) break;
            cur = (int[])best.Clone();
            int perturbCount = Math.Max(2, N / 10);
            for (int k = 0; k < perturbCount; k++)
            {
                int a = rng.Next(N), b = rng.Next(N);
                int t = cur[a]; cur[a] = cur[b]; cur[b] = t;
            }
            long c = ComputeCost(cur);
            c = LocalSearch(cur, c);
            if (c < bestCost) { bestCost = c; best = (int[])cur.Clone(); }
        }

        var sb = new StringBuilder();
        sb.Append(bestCost).Append('\n');
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(best[i]);
        }
        sb.Append('\n');
        Console.Out.Write(sb.ToString());
    }

    static void ReadInput()
    {
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(tokens[idx++]);
        F = new int[N][];
        D = new int[N][];
        for (int i = 0; i < N; i++)
        {
            F[i] = new int[N];
            for (int j = 0; j < N; j++) F[i][j] = int.Parse(tokens[idx++]);
        }
        for (int i = 0; i < N; i++)
        {
            D[i] = new int[N];
            for (int j = 0; j < N; j++) D[i][j] = int.Parse(tokens[idx++]);
        }
    }

    static long ComputeCost(int[] p)
    {
        long c = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            var Fi = F[i];
            var Dpi = D[pi];
            for (int j = 0; j < N; j++)
            {
                c += (long)Fi[j] * Dpi[p[j]];
            }
        }
        return c;
    }

    static int[] GreedyConstruct()
    {
        long[] flowSum = new long[N];
        long[] distSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += F[i][j] + F[j][i];
                distSum[i] += D[i][j] + D[j][i];
            }
        }
        int[] facOrder = new int[N];
        int[] locOrder = new int[N];
        for (int i = 0; i < N; i++) { facOrder[i] = i; locOrder[i] = i; }
        Array.Sort(facOrder, (a, b) => flowSum[b].CompareTo(flowSum[a]));
        Array.Sort(locOrder, (a, b) => distSum[a].CompareTo(distSum[b]));
        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[facOrder[i]] = locOrder[i];
        return p;
    }

    static long LocalSearch(int[] p, long curCost)
    {
        bool improved = true;
        while (improved)
        {
            improved = false;
            if (sw.ElapsedMilliseconds > timeLimitMs) break;
            for (int i = 0; i < N - 1; i++)
            {
                if (sw.ElapsedMilliseconds > timeLimitMs) break;
                for (int j = i + 1; j < N; j++)
                {
                    long delta = SwapDelta(p, i, j);
                    if (delta < 0)
                    {
                        int t = p[i]; p[i] = p[j]; p[j] = t;
                        curCost += delta;
                        improved = true;
                    }
                }
            }
        }
        return curCost;
    }

    static long SwapDelta(int[] p, int i, int j)
    {
        int pi = p[i], pj = p[j];
        long delta = 0;
        var Fi = F[i]; var Fj = F[j];
        var Dpi = D[pi]; var Dpj = D[pj];

        delta += (long)Fi[i] * (Dpj[pj] - Dpi[pi]);
        delta += (long)Fj[j] * (Dpi[pi] - Dpj[pj]);
        delta += (long)Fi[j] * (Dpj[pi] - Dpi[pj]);
        delta += (long)Fj[i] * (Dpi[pj] - Dpj[pi]);

        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];
            delta += (long)Fi[k] * (Dpj[pk] - Dpi[pk]);
            delta += (long)F[k][i] * (D[pk][pj] - D[pk][pi]);
            delta += (long)Fj[k] * (Dpi[pk] - Dpj[pk]);
            delta += (long)F[k][j] * (D[pk][pi] - D[pk][pj]);
        }
        return delta;
    }
}
