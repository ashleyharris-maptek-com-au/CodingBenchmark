OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_out %gl_in %outColor %inColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
    %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_1 = OpConstant %uint 1
 %float_0_05 = OpConstant %float 0.05

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_float = OpTypePointer Input %float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
    %inColor = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
      %entry = OpLabel

      %pxPtr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_0
          %x = OpLoad %float %pxPtr
      %pyPtr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_1
          %y = OpLoad %float %pyPtr
      %pzPtr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_2
          %z = OpLoad %float %pzPtr
      %pwPtr = OpAccessChain %ptr_in_float %gl_in %int_0 %int_0 %int_3
          %w = OpLoad %float %pwPtr

     %pColIn = OpAccessChain %ptr_in_v3 %inColor %int_0
      %colIn = OpLoad %v3float %pColIn

        %xph = OpFAdd %float %x %float_0_05
        %xmh = OpFSub %float %x %float_0_05
        %yph = OpFAdd %float %y %float_0_05
        %ymh = OpFSub %float %y %float_0_05

       %vTop = OpCompositeConstruct %v4float %x %yph %z %w
     %vRight = OpCompositeConstruct %v4float %xph %y %z %w
      %vLeft = OpCompositeConstruct %v4float %xmh %y %z %w
       %vBot = OpCompositeConstruct %v4float %x %ymh %z %w

    %ptopOut = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ptopOut %vTop
               OpStore %outColor %colIn
               OpEmitVertex

  %prightOut = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %prightOut %vRight
               OpStore %outColor %colIn
               OpEmitVertex

   %pleftOut = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pleftOut %vLeft
               OpStore %outColor %colIn
               OpEmitVertex

    %pbotOut = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %pbotOut %vBot
               OpStore %outColor %colIn
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
