use std::io::{self, Read, Write, BufWriter, BufReader};
use std::fs::File;

const POINT_BYTES: usize = 12;
const IDX_BYTES: usize = 4;

#[inline(always)]
fn dist2(a: [f32; 3], b: [f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline(always)]
fn nearest(centroids: &[[f32; 3]], p: [f32; 3]) -> u32 {
    let mut best = 0u32;
    let mut best_d = f32::INFINITY;
    for (i, c) in centroids.iter().enumerate() {
        let d = dist2(*c, p);
        if d < best_d {
            best_d = d;
            best = i as u32;
        }
    }
    best
}

struct Rng(u64);
impl Rng {
    fn new(seed: u64) -> Self { Rng(seed | 1) }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.0;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.0 = x;
        x
    }
    fn next_f32(&mut self) -> f32 {
        (self.next_u64() >> 40) as f32 / (1u64 << 24) as f32
    }
    fn gen_range(&mut self, n: usize) -> usize {
        if n == 0 { return 0; }
        (self.next_u64() as usize) % n
    }
}

fn read_header<R: Read>(r: &mut R) -> io::Result<(u64, u32)> {
    let mut buf = String::new();
    let mut byte = [0u8; 1];
    loop {
        let n = r.read(&mut byte)?;
        if n == 0 { break; }
        if byte[0] == b'\n' { break; }
        buf.push(byte[0] as char);
    }
    let parts: Vec<&str> = buf.split_whitespace().collect();
    if parts.len() < 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "bad header"));
    }
    let np: u64 = parts[0].parse().unwrap_or(0);
    let nc: u32 = parts[1].parse().unwrap_or(1);
    Ok((np, nc))
}

fn read_points_batch<R: Read>(r: &mut R, max: usize) -> io::Result<Vec<[f32; 3]>> {
    let mut bytes = vec![0u8; max * POINT_BYTES];
    let mut filled = 0;
    while filled < bytes.len() {
        let n = r.read(&mut bytes[filled..])?;
        if n == 0 { break; }
        filled += n;
    }
    let count = filled / POINT_BYTES;
    let mut out = Vec::with_capacity(count);
    for i in 0..count {
        let off = i * POINT_BYTES;
        let x = f32::from_le_bytes([bytes[off], bytes[off+1], bytes[off+2], bytes[off+3]]);
        let y = f32::from_le_bytes([bytes[off+4], bytes[off+5], bytes[off+6], bytes[off+7]]);
        let z = f32::from_le_bytes([bytes[off+8], bytes[off+9], bytes[off+10], bytes[off+11]]);
        out.push([x, y, z]);
    }
    Ok(out)
}

fn run_kmeans_in_memory(points: &[[f32; 3]], k: usize, max_iter: usize) -> Vec<[f32; 3]> {
    if points.is_empty() || k == 0 {
        return vec![[0.0; 3]; k];
    }
    let mut rng = Rng::new(0xA5A5_5A5A);
    let mut centroids: Vec<[f32; 3]> = Vec::with_capacity(k);
    let first = rng.gen_range(points.len());
    centroids.push(points[first]);
    let sample_size = points.len().min(4096);
    let mut sample: Vec<[f32; 3]> = Vec::with_capacity(sample_size);
    for _ in 0..sample_size {
        sample.push(points[rng.gen_range(points.len())]);
    }
    let mut dists: Vec<f32> = sample.iter().map(|p| dist2(centroids[0], *p)).collect();
    while centroids.len() < k {
        let mut sum = 0.0f64;
        for &d in &dists { sum += d as f64; }
        if sum <= 0.0 {
            centroids.push(points[rng.gen_range(points.len())]);
        } else {
            let r = rng.next_f32() as f64 * sum;
            let mut acc = 0.0f64;
            let mut chosen = 0usize;
            for (i, &d) in dists.iter().enumerate() {
                acc += d as f64;
                if acc >= r { chosen = i; break; }
            }
            centroids.push(sample[chosen]);
        }
        let last = *centroids.last().unwrap();
        for i in 0..sample.len() {
            let d = dist2(last, sample[i]);
            if d < dists[i] { dists[i] = d; }
        }
    }

    let mut sums: Vec<[f64; 3]> = vec![[0.0; 3]; k];
    let mut counts: Vec<u64> = vec![0; k];
    for _ in 0..max_iter {
        for s in sums.iter_mut() { *s = [0.0; 3]; }
        for c in counts.iter_mut() { *c = 0; }
        for p in points {
            let i = nearest(&centroids, *p) as usize;
            sums[i][0] += p[0] as f64;
            sums[i][1] += p[1] as f64;
            sums[i][2] += p[2] as f64;
            counts[i] += 1;
        }
        let mut moved = 0.0f64;
        for i in 0..k {
            if counts[i] > 0 {
                let nc = [
                    (sums[i][0] / counts[i] as f64) as f32,
                    (sums[i][1] / counts[i] as f64) as f32,
                    (sums[i][2] / counts[i] as f64) as f32,
                ];
                moved += dist2(centroids[i], nc) as f64;
                centroids[i] = nc;
            } else {
                centroids[i] = points[rng.gen_range(points.len())];
            }
        }
        if moved < 1e-6 { break; }
    }
    centroids
}

fn run_minibatch<R: Read>(r: &mut R, centroids: &mut Vec<[f32; 3]>, k: usize, batch_size: usize, max_batches: usize) -> io::Result<()> {
    let mut counts: Vec<u64> = vec![1; k];
    let mut batches = 0;
    loop {
        let batch = read_points_batch(r, batch_size)?;
        if batch.is_empty() { break; }
        for p in &batch {
            let i = nearest(centroids, *p) as usize;
            counts[i] += 1;
            let eta = 1.0f32 / counts[i] as f32;
            centroids[i][0] = (1.0 - eta) * centroids[i][0] + eta * p[0];
            centroids[i][1] = (1.0 - eta) * centroids[i][1] + eta * p[1];
            centroids[i][2] = (1.0 - eta) * centroids[i][2] + eta * p[2];
        }
        batches += 1;
        if batches >= max_batches { break; }
    }
    Ok(())
}

fn assign_pass<R: Read, W: Write>(r: &mut R, w: &mut W, centroids: &[[f32; 3]]) -> io::Result<()> {
    let batch_size = 1 << 16;
    let mut out_buf: Vec<u8> = Vec::with_capacity(batch_size * IDX_BYTES);
    loop {
        let batch = read_points_batch(r, batch_size)?;
        if batch.is_empty() { break; }
        out_buf.clear();
        for p in &batch {
            let idx = nearest(centroids, *p);
            out_buf.extend_from_slice(&idx.to_le_bytes());
        }
        w.write_all(&out_buf)?;
    }
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut stdin_lock = stdin.lock();
    let (num_points, num_clusters) = read_header(&mut stdin_lock)?;
    let k = num_clusters as usize;
    if k == 0 { return Ok(()); }

    let stdout = io::stdout();
    let mut out = BufWriter::with_capacity(1 << 20, stdout.lock());

    let small_threshold: u64 = 10_000_000;

    if num_points <= small_threshold {
        let mut points: Vec<[f32; 3]> = Vec::with_capacity(num_points as usize);
        let batch_size = 1 << 16;
        loop {
            let batch = read_points_batch(&mut stdin_lock, batch_size)?;
            if batch.is_empty() { break; }
            points.extend_from_slice(&batch);
            if points.len() as u64 >= num_points { break; }
        }
        let max_iter = if num_points <= 1_000_000 { 25 } else { 15 };
        let centroids = run_kmeans_in_memory(&points, k, max_iter);
        let mut out_buf: Vec<u8> = Vec::with_capacity(points.len() * IDX_BYTES);
        for p in &points {
            let idx = nearest(&centroids, *p);
            out_buf.extend_from_slice(&idx.to_le_bytes());
        }
        out.write_all(&out_buf)?;
        out.flush()?;
        return Ok(());
    }

    // Larger: stream to temp file so we can do multiple passes
    let tmp_path = std::env::temp_dir().join(format!("kmeans_points_{}.bin", std::process::id()));
    {
        let f = File::create(&tmp_path)?;
        let mut bw = BufWriter::with_capacity(1 << 20, f);
        let total_bytes = num_points.saturating_mul(POINT_BYTES as u64);
        let mut written: u64 = 0;
        let mut buf = vec![0u8; 1 << 16];
        while written < total_bytes {
            let want = ((total_bytes - written) as usize).min(buf.len());
            let n = stdin_lock.read(&mut buf[..want])?;
            if n == 0 { break; }
            bw.write_all(&buf[..n])?;
            written += n as u64;
        }
        bw.flush()?;
    }

    // Pass 1: reservoir sample for init
    let sample_k = (k * 4).max(2048).min(50_000);
    let mut sample: Vec<[f32; 3]> = Vec::with_capacity(sample_k);
    {
        let f = File::open(&tmp_path)?;
        let mut br = BufReader::with_capacity(1 << 20, f);
        let mut rng = Rng::new(0xDEAD_BEEF);
        let mut idx: u64 = 0;
        let bs = 1 << 16;
        loop {
            let batch = read_points_batch(&mut br, bs)?;
            if batch.is_empty() { break; }
            for p in batch {
                if (idx as usize) < sample_k {
                    sample.push(p);
                } else {
                    let j = (rng.next_u64() % (idx + 1)) as usize;
                    if j < sample_k { sample[j] = p; }
                }
                idx += 1;
            }
        }
    }

    let mut centroids = run_kmeans_in_memory(&sample, k, 15);

    // Pass 2: mini-batch refinement over the full data
    {
        let f = File::open(&tmp_path)?;
        let mut br = BufReader::with_capacity(1 << 20, f);
        let batch_size = 1 << 14;
        let max_batches = if num_points <= 100_000_000 { usize::MAX } else { 2048 };
        run_minibatch(&mut br, &mut centroids, k, batch_size, max_batches)?;
    }

    // Pass 3: final assignment
    {
        let f = File::open(&tmp_path)?;
        let mut br = BufReader::with_capacity(1 << 20, f);
        assign_pass(&mut br, &mut out, &centroids)?;
    }
    out.flush()?;

    let _ = std::fs::remove_file(&tmp_path);
    Ok(())
}
