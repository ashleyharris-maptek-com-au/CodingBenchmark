use std::io::{self, Read, Write, BufWriter};

fn solve(
    idx: usize,
    order: &[usize],
    adj: &[Vec<usize>],
    assign: &mut [u8],
    a_count: usize,
    b_count: usize,
    half: usize,
    n: usize,
    cur_cut: i32,
    best_cut: &mut i32,
    best_a: &mut Vec<usize>,
) {
    if cur_cut >= *best_cut {
        return;
    }
    if idx == order.len() {
        if a_count == half {
            *best_cut = cur_cut;
            best_a.clear();
            for v in 0..n {
                if assign[v] == 0 {
                    best_a.push(v);
                }
            }
        }
        return;
    }
    let remaining = order.len() - idx;
    if a_count + remaining < half || b_count + remaining < half {
        return;
    }
    let v = order[idx];

    if a_count < half {
        let mut added: i32 = 0;
        for &u in &adj[v] {
            if assign[u] == 1 {
                added += 1;
            }
        }
        assign[v] = 0;
        solve(
            idx + 1,
            order,
            adj,
            assign,
            a_count + 1,
            b_count,
            half,
            n,
            cur_cut + added,
            best_cut,
            best_a,
        );
        assign[v] = 2;
    }

    if b_count < half {
        let mut added: i32 = 0;
        for &u in &adj[v] {
            if assign[u] == 0 {
                added += 1;
            }
        }
        assign[v] = 1;
        solve(
            idx + 1,
            order,
            adj,
            assign,
            a_count,
            b_count + 1,
            half,
            n,
            cur_cut + added,
            best_cut,
            best_a,
        );
        assign[v] = 2;
    }
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut it = s.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];
    let mut seen: std::collections::HashSet<(usize, usize)> = std::collections::HashSet::new();
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        let key = if u < v { (u, v) } else { (v, u) };
        if seen.insert(key) {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let half = n / 2;

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by_key(|&v| std::cmp::Reverse(adj[v].len()));

    let mut assign = vec![2u8; n];
    let mut best_cut: i32 = i32::MAX;
    let mut best_a: Vec<usize> = (0..half).collect();

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }

    let v0 = order[0];
    assign[v0] = 0;
    solve(
        1,
        &order,
        &adj,
        &mut assign,
        1,
        0,
        half,
        n,
        0,
        &mut best_cut,
        &mut best_a,
    );

    if best_cut == i32::MAX {
        best_cut = 0;
        best_a = (0..half).collect();
    }

    best_a.sort();
    writeln!(out, "{}", best_cut).unwrap();
    let strs: Vec<String> = best_a.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", strs.join(" ")).unwrap();
}
