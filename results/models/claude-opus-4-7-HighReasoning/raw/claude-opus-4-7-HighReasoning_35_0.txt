use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

#[derive(Eq, PartialEq)]
struct Entry {
    gain: usize,
    v: usize,
}
impl Ord for Entry {
    fn cmp(&self, other: &Self) -> std::cmp::Ordering {
        self.gain.cmp(&other.gain).then_with(|| other.v.cmp(&self.v))
    }
}
impl PartialOrd for Entry {
    fn partial_cmp(&self, other: &Self) -> Option<std::cmp::Ordering> {
        Some(self.cmp(other))
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => 0,
    };
    let m: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);

    if n == 0 {
        writeln!(out, "0").unwrap();
        writeln!(out, "").unwrap();
        return;
    }

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        let v: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut result: Vec<usize> = Vec::new();
    let mut num_dominated = 0usize;

    let mut heap: BinaryHeap<Entry> = BinaryHeap::with_capacity(n);
    for v in 0..n {
        heap.push(Entry { gain: adj[v].len() + 1, v });
    }

    while num_dominated < n {
        let best = loop {
            match heap.pop() {
                None => break None,
                Some(e) => {
                    if in_set[e.v] {
                        continue;
                    }
                    let mut g = if !dominated[e.v] { 1usize } else { 0 };
                    for &u in &adj[e.v] {
                        if !dominated[u] {
                            g += 1;
                        }
                    }
                    if g == e.gain {
                        break Some(e.v);
                    } else if g > 0 {
                        heap.push(Entry { gain: g, v: e.v });
                    }
                }
            }
        };

        let v = match best {
            Some(v) => v,
            None => {
                let mut found = None;
                for i in 0..n {
                    if !dominated[i] {
                        found = Some(i);
                        break;
                    }
                }
                match found {
                    Some(i) => i,
                    None => break,
                }
            }
        };

        in_set[v] = true;
        result.push(v);
        if !dominated[v] {
            dominated[v] = true;
            num_dominated += 1;
        }
        let neighbors = adj[v].clone();
        for &u in &neighbors {
            if !dominated[u] {
                dominated[u] = true;
                num_dominated += 1;
            }
        }
    }

    // Pruning pass: remove redundant vertices
    let mut dom_count = vec![0u32; n];
    for &s in &result {
        dom_count[s] += 1;
        for &u in &adj[s] {
            dom_count[u] += 1;
        }
    }

    let mut order: Vec<usize> = (0..result.len()).collect();
    order.sort_by_key(|&i| adj[result[i]].len());

    let mut removed = vec![false; result.len()];
    for &idx in &order {
        let v = result[idx];
        let mut can_remove = dom_count[v] >= 2;
        if can_remove {
            for &u in &adj[v] {
                if dom_count[u] < 2 {
                    can_remove = false;
                    break;
                }
            }
        }
        if can_remove {
            removed[idx] = true;
            dom_count[v] -= 1;
            for &u in &adj[v] {
                dom_count[u] -= 1;
            }
        }
    }

    let final_set: Vec<usize> = result
        .iter()
        .enumerate()
        .filter_map(|(i, &v)| if !removed[i] { Some(v) } else { None })
        .collect();

    writeln!(out, "{}", final_set.len()).unwrap();
    let s: Vec<String> = final_set.iter().map(|x| x.to_string()).collect();
    writeln!(out, "{}", s.join(" ")).unwrap();
}
