using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

class Solver
{
    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int V = int.Parse(tokens[idx++], ci);
        int F = int.Parse(tokens[idx++], ci);

        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        for (int i = 0; i < V; i++)
        {
            double x = double.Parse(tokens[idx++], ci);
            double y = double.Parse(tokens[idx++], ci);
            double z = double.Parse(tokens[idx++], ci);
            if (x < minX) minX = x;
            if (y < minY) minY = y;
            if (z < minZ) minZ = z;
            if (x > maxX) maxX = x;
            if (y > maxY) maxY = y;
            if (z > maxZ) maxZ = z;
        }

        // Skip face indices
        idx += F * 3;

        double W = double.Parse(tokens[idx++], ci);
        double H = double.Parse(tokens[idx++], ci);
        double D = double.Parse(tokens[idx++], ci);

        double bx = maxX - minX;
        double by = maxY - minY;
        double bz = maxZ - minZ;

        if (bx <= 0) bx = 1e-9;
        if (by <= 0) by = 1e-9;
        if (bz <= 0) bz = 1e-9;

        // Tiny epsilon to avoid floating point overlap on shared faces
        double eps = 1e-6;
        double cellX = bx + eps;
        double cellY = by + eps;
        double cellZ = bz + eps;

        long nx = (long)Math.Floor(W / cellX);
        long ny = (long)Math.Floor(H / cellY);
        long nz = (long)Math.Floor(D / cellZ);

        if (nx < 0) nx = 0;
        if (ny < 0) ny = 0;
        if (nz < 0) nz = 0;

        long total = nx * ny * nz;

        // Cap total to keep output size and runtime reasonable
        long cap = 2000000;
        if (total > cap)
        {
            double ratio = Math.Pow((double)cap / total, 1.0 / 3.0);
            nx = Math.Max(1, (long)(nx * ratio));
            ny = Math.Max(1, (long)(ny * ratio));
            nz = Math.Max(1, (long)(nz * ratio));
            total = nx * ny * nz;
        }

        var sb = new StringBuilder();
        sb.Append(total.ToString(ci));
        sb.Append('\n');

        // Identity rotation: shift so polyhedron local bbox-min sits at cell origin.
        double ox = -minX;
        double oy = -minY;
        double oz = -minZ;

        for (long ix = 0; ix < nx; ix++)
        {
            double tx = ix * cellX + ox;
            for (long iy = 0; iy < ny; iy++)
            {
                double ty = iy * cellY + oy;
                for (long iz = 0; iz < nz; iz++)
                {
                    double tz = iz * cellZ + oz;
                    sb.Append(tx.ToString("G17", ci));
                    sb.Append(' ');
                    sb.Append(ty.ToString("G17", ci));
                    sb.Append(' ');
                    sb.Append(tz.ToString("G17", ci));
                    sb.Append(" 1 0 0 0\n");
                }
            }
        }

        Console.Out.Write(sb.ToString());
    }
}
