using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class AsteroidInterceptor
{
    const double G = 6.67430e-11;
    const double DAY = 86400.0;

    class Body
    {
        public string Name;
        public double Mass;
        public double[] R = new double[3];
        public double[] V = new double[3];
    }

    static double Dot(double[] a, double[] b) => a[0]*b[0]+a[1]*b[1]+a[2]*b[2];
    static double Norm(double[] a) => Math.Sqrt(Dot(a,a));

    static void ComputeAccel(List<Body> bodies, double[][] R, double[][] A)
    {
        int n = bodies.Count;
        for (int i = 0; i < n; i++) { A[i][0]=0; A[i][1]=0; A[i][2]=0; }
        for (int i = 0; i < n; i++)
        {
            for (int j = i+1; j < n; j++)
            {
                double dx = R[j][0]-R[i][0];
                double dy = R[j][1]-R[i][1];
                double dz = R[j][2]-R[i][2];
                double r2 = dx*dx+dy*dy+dz*dz;
                double r = Math.Sqrt(r2);
                double inv = 1.0/(r2*r);
                double fi = G*bodies[j].Mass*inv;
                double fj = G*bodies[i].Mass*inv;
                A[i][0]+=fi*dx; A[i][1]+=fi*dy; A[i][2]+=fi*dz;
                A[j][0]-=fj*dx; A[j][1]-=fj*dy; A[j][2]-=fj*dz;
            }
        }
    }

    static void PropagateAll(List<Body> bodies, double[][] R, double[][] V, double tEnd, double dt)
    {
        int n = bodies.Count;
        double[][] A = new double[n][];
        for (int i = 0; i < n; i++) A[i] = new double[3];
        ComputeAccel(bodies, R, A);
        double t = 0;
        while (t < tEnd - 1e-9)
        {
            double step = Math.Min(dt, tEnd - t);
            for (int i = 0; i < n; i++)
            {
                V[i][0] += 0.5*step*A[i][0];
                V[i][1] += 0.5*step*A[i][1];
                V[i][2] += 0.5*step*A[i][2];
                R[i][0] += step*V[i][0];
                R[i][1] += step*V[i][1];
                R[i][2] += step*V[i][2];
            }
            ComputeAccel(bodies, R, A);
            for (int i = 0; i < n; i++)
            {
                V[i][0] += 0.5*step*A[i][0];
                V[i][1] += 0.5*step*A[i][1];
                V[i][2] += 0.5*step*A[i][2];
            }
            t += step;
        }
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var tokens = new List<string>();
        string s;
        while ((s = Console.In.ReadLine()) != null)
        {
            tokens.AddRange(s.Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries));
        }
        int idx = 0;
        int numBodies = int.Parse(tokens[idx++], ci);
        double warningDays = double.Parse(tokens[idx++], ci);
        double impactorMass = double.Parse(tokens[idx++], ci);
        double dvBudget = double.Parse(tokens[idx++], ci);
        double asteroidMass = double.Parse(tokens[idx++], ci);
        double t0Unix = double.Parse(tokens[idx++], ci);

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[idx++];
            b.Mass = double.Parse(tokens[idx++], ci);
            b.R[0] = double.Parse(tokens[idx++], ci);
            b.R[1] = double.Parse(tokens[idx++], ci);
            b.R[2] = double.Parse(tokens[idx++], ci);
            b.V[0] = double.Parse(tokens[idx++], ci);
            b.V[1] = double.Parse(tokens[idx++], ci);
            b.V[2] = double.Parse(tokens[idx++], ci);
            bodies.Add(b);
        }

        int earthIdx = bodies.FindIndex(b => b.Name == "Earth");
        int astIdx = bodies.FindIndex(b => b.Name == "Asteroid");

        double launchDay = Math.Min(1.0, warningDays * 0.02);
        double impactDay = warningDays * 0.5;
        if (impactDay <= launchDay + 1) impactDay = Math.Max(launchDay + 1, warningDays - 1);

        int n = bodies.Count;
        double[][] R0 = new double[n][];
        double[][] V0 = new double[n][];
        for (int i = 0; i < n; i++)
        {
            R0[i] = (double[])bodies[i].R.Clone();
            V0[i] = (double[])bodies[i].V.Clone();
        }

        double dt = 3600.0;

        double[][] Rl = new double[n][];
        double[][] Vl = new double[n][];
        for (int i = 0; i < n; i++) { Rl[i] = (double[])R0[i].Clone(); Vl[i] = (double[])V0[i].Clone(); }
        if (launchDay > 0) PropagateAll(bodies, Rl, Vl, launchDay * DAY, dt);

        double[][] Ri = new double[n][];
        double[][] Vi = new double[n][];
        for (int i = 0; i < n; i++) { Ri[i] = (double[])R0[i].Clone(); Vi[i] = (double[])V0[i].Clone(); }
        PropagateAll(bodies, Ri, Vi, impactDay * DAY, dt);

        double tof = (impactDay - launchDay) * DAY;
        double[] earthRL = Rl[earthIdx];
        double[] earthVL = Vl[earthIdx];
        double[] astRI = Ri[astIdx];

        double[] reqV = new double[3];
        for (int k = 0; k < 3; k++) reqV[k] = (astRI[k] - earthRL[k]) / tof;
        double[] dv = new double[3];
        for (int k = 0; k < 3; k++) dv[k] = reqV[k] - earthVL[k];

        double dvMag = Norm(dv);
        double cap = dvBudget * 0.8;
        if (dvMag > cap)
        {
            for (int k = 0; k < 3; k++) dv[k] *= cap / dvMag;
            dvMag = cap;
        }

        double[] astVI = Vi[astIdx];
        double avMag = Norm(astVI);
        double remaining = Math.Max(0.0, dvBudget - dvMag);
        double[] impactDv = new double[3];
        if (avMag > 0)
        {
            for (int k = 0; k < 3; k++) impactDv[k] = -astVI[k] / avMag * remaining;
        }

        var w = Console.Out;
        w.WriteLine("3");
        w.WriteLine($"{launchDay.ToString("G17",ci)} {dv[0].ToString("G17",ci)} {dv[1].ToString("G17",ci)} {dv[2].ToString("G17",ci)}");
        double midDay = (launchDay + impactDay) * 0.5;
        w.WriteLine($"{midDay.ToString("G17",ci)} 0 0 0");
        w.WriteLine($"{impactDay.ToString("G17",ci)} {impactDv[0].ToString("G17",ci)} {impactDv[1].ToString("G17",ci)} {impactDv[2].ToString("G17",ci)}");
    }
}
