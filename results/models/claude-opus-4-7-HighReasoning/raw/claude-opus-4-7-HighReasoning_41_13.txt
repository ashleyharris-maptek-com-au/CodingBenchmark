; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
   %voidFunc = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
    %ptr_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %ptr_UBO Uniform

  %ptr_in_v3 = OpTypePointer Input %v3f
  %ptr_in_v2 = OpTypePointer Input %v2f
 %ptr_out_v4 = OpTypePointer Output %v4f
   %ptr_in_f = OpTypePointer Input %float

   %worldPos = OpVariable %ptr_in_v3 Input
     %normal = OpVariable %ptr_in_v3 Input
         %uv = OpVariable %ptr_in_v2 Input
    %tangent = OpVariable %ptr_in_v3 Input
      %color = OpVariable %ptr_in_v3 Input
   %outColor = OpVariable %ptr_out_v4 Output

      %f_0_5 = OpConstant %float 0.5
      %f_0_0 = OpConstant %float 0.0
      %f_1_0 = OpConstant %float 1.0
      %f_0_1 = OpConstant %float 0.1
      %f_0_4 = OpConstant %float 0.4
     %bottom = OpConstantComposite %v3f %f_0_1 %f_0_4 %f_0_1
        %top = OpConstantComposite %v3f %f_1_0 %f_1_0 %f_1_0

       %uint = OpTypeInt 32 0
     %uint_1 = OpConstant %uint 1

       %main = OpFunction %void None %voidFunc
      %label = OpLabel
       %pY_p = OpAccessChain %ptr_in_f %worldPos %uint_1
         %pY = OpLoad %float %pY_p
         %hY = OpFMul %float %pY %f_0_5
         %t0 = OpFAdd %float %hY %f_0_5
          %t = OpExtInst %float %glsl FClamp %t0 %f_0_0 %f_1_0
        %tv3 = OpCompositeConstruct %v3f %t %t %t
         %mx = OpExtInst %v3f %glsl FMix %bottom %top %tv3
         %rx = OpCompositeExtract %float %mx 0
         %gx = OpCompositeExtract %float %mx 1
         %bx = OpCompositeExtract %float %mx 2
        %col = OpCompositeConstruct %v4f %rx %gx %bx %f_1_0
               OpStore %outColor %col
               OpReturn
               OpFunctionEnd
