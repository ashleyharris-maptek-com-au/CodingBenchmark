using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var input = Console.In.ReadToEnd();
        int pos = 0;
        int len = input.Length;

        int ReadInt()
        {
            while (pos < len && (input[pos] == ' ' || input[pos] == '\r' || input[pos] == '\n' || input[pos] == '\t')) pos++;
            long v = 0;
            bool neg = false;
            if (pos < len && input[pos] == '-') { neg = true; pos++; }
            while (pos < len && input[pos] >= '0' && input[pos] <= '9')
            {
                v = v * 10 + (input[pos] - '0');
                pos++;
            }
            return (int)(neg ? -v : v);
        }

        int N = ReadInt();
        int stockLen = ReadInt();

        int[] cuts = new int[N];
        for (int i = 0; i < N; i++) cuts[i] = ReadInt();

        // Sort indices by cut length descending (First/Best Fit Decreasing)
        int[] idx = new int[N];
        for (int i = 0; i < N; i++) idx[i] = i;
        Array.Sort(idx, (a, b) => cuts[b].CompareTo(cuts[a]));

        // Best Fit Decreasing using SortedSet<(rem, binId)>.
        var sset = new SortedSet<(int rem, int binId)>();
        var binAssignments = new List<List<int>>();
        var binRem = new List<int>();

        for (int k = 0; k < N; k++)
        {
            int ci = idx[k];
            int c = cuts[ci];

            if (c > stockLen)
            {
                // Shouldn't happen per constraints; place alone.
                int nb = binRem.Count;
                binRem.Add(0);
                binAssignments.Add(new List<int> { ci });
                continue;
            }

            // Find smallest (rem, binId) with rem >= c
            var view = sset.GetViewBetween((c, int.MinValue), (int.MaxValue, int.MaxValue));
            if (view.Count > 0)
            {
                var found = view.Min;
                sset.Remove(found);
                int newRem = found.rem - c;
                binRem[found.binId] = newRem;
                binAssignments[found.binId].Add(ci);
                sset.Add((newRem, found.binId));
            }
            else
            {
                int newBinId = binRem.Count;
                int rem = stockLen - c;
                binRem.Add(rem);
                binAssignments.Add(new List<int> { ci });
                sset.Add((rem, newBinId));
            }
        }

        var sb = new StringBuilder();
        sb.Append(binAssignments.Count).Append('\n');
        foreach (var bin in binAssignments)
        {
            for (int i = 0; i < bin.Count; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(bin[i]);
            }
            sb.Append('\n');
        }

        var stdout = Console.OpenStandardOutput();
        using (var bw = new BufferedStream(stdout, 1 << 16))
        {
            var bytes = Encoding.ASCII.GetBytes(sb.ToString());
            bw.Write(bytes, 0, bytes.Length);
        }
    }
}
