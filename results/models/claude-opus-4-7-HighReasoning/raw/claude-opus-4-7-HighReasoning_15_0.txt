use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut tokens = input.split_ascii_whitespace();
    let nv: usize = tokens.next().unwrap().parse().unwrap();
    let mut poly: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = tokens.next().unwrap().parse().unwrap();
        let y: f64 = tokens.next().unwrap().parse().unwrap();
        poly.push((x, y));
    }
    let sx: f64 = tokens.next().unwrap().parse().unwrap();
    let sy: f64 = tokens.next().unwrap().parse().unwrap();
    let sz: f64 = tokens.next().unwrap().parse().unwrap();

    // shadow(x,y,z) -> (x - z*shear_x, y - z*shear_y)
    let sz_safe = if sz.abs() < 1e-9 { -1.0 } else { sz };
    let shear_x = sx / sz_safe;
    let shear_y = sy / sz_safe;

    let tris = triangulate(&poly);

    let mut min_x = f64::INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for &(x, y) in &poly {
        if x < min_x { min_x = x; }
        if y < min_y { min_y = y; }
        if x > max_x { max_x = x; }
        if y > max_y { max_y = y; }
    }
    let span = (max_x - min_x).max(max_y - min_y);

    let shear_norm = (1.0 + shear_x.abs() + shear_y.abs()).max(1.0);

    let mut tetras: Vec<(f64, f64, f64, f64, f64, f64, f64, f64)> = Vec::new();
    let z_gap = 0.01_f64;
    let mut z_cursor: f64 = 0.0;

    let cx_all = 0.5 * (min_x + max_x);
    let cy_all = 0.5 * (min_y + max_y);
    let mut max_r_all = 0.0_f64;
    for &(x, y) in &poly {
        let dx = x - cx_all;
        let dy = y - cy_all;
        let r = (dx * dx + dy * dy).sqrt();
        if r > max_r_all { max_r_all = r; }
    }
    let needed_scale_single = max_r_all * std::f64::consts::SQRT_2 * shear_norm * 1.15;

    if needed_scale_single <= 10.0 && !tris.is_empty() && span <= 30.0 {
        let mut s = needed_scale_single.max(0.1);
        if s > 10.0 { s = 10.0; }
        let cz = 0.707 * s + z_gap;
        let cx = cx_all + cz * shear_x;
        let cy = cy_all + cz * shear_y;
        tetras.push((cx, cy, cz, 1.0, 0.0, 0.0, 0.0, s));
    } else {
        let mut work: Vec<[(f64, f64); 3]> = tris.clone();
        let mut processed: Vec<[(f64, f64); 3]> = Vec::new();
        let max_radius_per_tetra = 10.0 / (std::f64::consts::SQRT_2 * shear_norm * 1.15);

        while let Some(tri) = work.pop() {
            let cx_t = (tri[0].0 + tri[1].0 + tri[2].0) / 3.0;
            let cy_t = (tri[0].1 + tri[1].1 + tri[2].1) / 3.0;
            let mut r = 0.0_f64;
            for v in &tri {
                let dx = v.0 - cx_t;
                let dy = v.1 - cy_t;
                let d = (dx * dx + dy * dy).sqrt();
                if d > r { r = d; }
            }
            if r <= max_radius_per_tetra || r < 0.01 {
                processed.push(tri);
            } else {
                let m01 = midpoint(tri[0], tri[1]);
                let m12 = midpoint(tri[1], tri[2]);
                let m20 = midpoint(tri[2], tri[0]);
                work.push([tri[0], m01, m20]);
                work.push([tri[1], m12, m01]);
                work.push([tri[2], m20, m12]);
                work.push([m01, m12, m20]);
            }
        }

        processed.sort_by(|a, b| {
            let aa = tri_area(a);
            let bb = tri_area(b);
            bb.partial_cmp(&aa).unwrap_or(std::cmp::Ordering::Equal)
        });

        for tri in &processed {
            let cx_t = (tri[0].0 + tri[1].0 + tri[2].0) / 3.0;
            let cy_t = (tri[0].1 + tri[1].1 + tri[2].1) / 3.0;
            let mut r = 0.0_f64;
            for v in tri {
                let dx = v.0 - cx_t;
                let dy = v.1 - cy_t;
                let d = (dx * dx + dy * dy).sqrt();
                if d > r { r = d; }
            }
            let mut s = (r * std::f64::consts::SQRT_2 * shear_norm * 1.15).max(0.05);
            if s > 10.0 { s = 10.0; }

            let cz = z_cursor + 0.707 * s + z_gap;
            z_cursor = cz + 0.707 * s + z_gap;
            let cx = cx_t + cz * shear_x;
            let cy = cy_t + cz * shear_y;
            tetras.push((cx, cy, cz, 1.0, 0.0, 0.0, 0.0, s));
        }
    }

    writeln!(out, "{}", tetras.len()).unwrap();
    for t in &tetras {
        writeln!(
            out,
            "{} {} {} {} {} {} {} {}",
            t.0, t.1, t.2, t.3, t.4, t.5, t.6, t.7
        )
        .unwrap();
    }
}

fn midpoint(a: (f64, f64), b: (f64, f64)) -> (f64, f64) {
    (0.5 * (a.0 + b.0), 0.5 * (a.1 + b.1))
}

fn tri_area(t: &[(f64, f64); 3]) -> f64 {
    0.5 * ((t[1].0 - t[0].0) * (t[2].1 - t[0].1)
        - (t[2].0 - t[0].0) * (t[1].1 - t[0].1))
        .abs()
}

fn signed_area(p: &[(f64, f64)]) -> f64 {
    let n = p.len();
    let mut s = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        s += p[i].0 * p[j].1 - p[j].0 * p[i].1;
    }
    s * 0.5
}

fn cross2(a: (f64, f64), b: (f64, f64), c: (f64, f64)) -> f64 {
    (b.0 - a.0) * (c.1 - a.1) - (b.1 - a.1) * (c.0 - a.0)
}

fn point_in_tri(p: (f64, f64), a: (f64, f64), b: (f64, f64), c: (f64, f64)) -> bool {
    let d1 = cross2(a, b, p);
    let d2 = cross2(b, c, p);
    let d3 = cross2(c, a, p);
    let has_neg = d1 < 0.0 || d2 < 0.0 || d3 < 0.0;
    let has_pos = d1 > 0.0 || d2 > 0.0 || d3 > 0.0;
    !(has_neg && has_pos)
}

fn triangulate(poly: &[(f64, f64)]) -> Vec<[(f64, f64); 3]> {
    let n = poly.len();
    if n < 3 {
        return Vec::new();
    }
    if n == 3 {
        return vec![[poly[0], poly[1], poly[2]]];
    }
    let mut p: Vec<(f64, f64)> = poly.to_vec();
    if signed_area(&p) < 0.0 {
        p.reverse();
    }
    let mut indices: Vec<usize> = (0..p.len()).collect();
    let mut result: Vec<[(f64, f64); 3]> = Vec::new();
    let mut guard = 0_usize;
    let max_guard = p.len() * p.len() + 100;
    while indices.len() > 3 && guard < max_guard {
        guard += 1;
        let m = indices.len();
        let mut found = false;
        for i in 0..m {
            let i0 = indices[(i + m - 1) % m];
            let i1 = indices[i];
            let i2 = indices[(i + 1) % m];
            let va = p[i0];
            let vb = p[i1];
            let vc = p[i2];
            let crs = (vb.0 - va.0) * (vc.1 - va.1) - (vb.1 - va.1) * (vc.0 - va.0);
            if crs <= 0.0 {
                continue;
            }
            let mut ok = true;
            for &j in &indices {
                if j == i0 || j == i1 || j == i2 {
                    continue;
                }
                if point_in_tri(p[j], va, vb, vc) {
                    ok = false;
                    break;
                }
            }
            if ok {
                result.push([va, vb, vc]);
                indices.remove(i);
                found = true;
                break;
            }
        }
        if !found {
            for k in 1..(indices.len() - 1) {
                result.push([p[indices[0]], p[indices[k]], p[indices[k + 1]]]);
            }
            return result;
        }
    }
    if indices.len() == 3 {
        result.push([p[indices[0]], p[indices[1]], p[indices[2]]]);
    }
    result
}
