#include <cstdio>
#include <cstdlib>
#include <cstring>

#include <algorithm>

using namespace std;

typedef long long ll;
const ll INF = (ll)1e18;

struct Edge {
    int to;
    ll w;
    int id;
};

int N, M, K;
vector<vector<Edge>> adj;
vector<int> terms;
vector<int> termIdx;
vector<tuple<int,int,ll>> edgeList;

void dijkstra(int src, vector<ll>& dist, vector<int>& parent, vector<int>& parentEdge) {
    dist.assign(N, INF);
    parent.assign(N, -1);
    parentEdge.assign(N, -1);
    dist[src] = 0;
    priority_queue<pair<ll,int>, vector<pair<ll,int>>, greater<pair<ll,int>>> pq;
    pq.push({0, src});
    while (!pq.empty()) {
        auto top = pq.top(); pq.pop();
        ll d = top.first;
        int u = top.second;
        if (d > dist[u]) continue;
        for (auto& e : adj[u]) {
            ll nd = d + e.w;
            if (nd < dist[e.to]) {
                dist[e.to] = nd;
                parent[e.to] = u;
                parentEdge[e.to] = e.id;
                pq.push({nd, e.to});
            }
        }
    }
}

struct DSU {
    vector<int> p, r;
    DSU(int n) : p(n), r(n, 0) { for (int i = 0; i < n; i++) p[i] = i; }
    int find(int x) { while (p[x] != x) { p[x] = p[p[x]]; x = p[x]; } return x; }
    bool unite(int a, int b) {
        a = find(a); b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

vector<pair<int,int>> solve(ll& totalWeight) {
    int Keff = K;
    vector<vector<ll>> dist(Keff);
    vector<vector<int>> parent(Keff);
    vector<vector<int>> parentEdge(Keff);
    for (int i = 0; i < Keff; i++) {
        dijkstra(terms[i], dist[i], parent[i], parentEdge[i]);
    }

    vector<tuple<ll,int,int>> termEdges;
    for (int i = 0; i < Keff; i++) {
        for (int j = i+1; j < Keff; j++) {
            ll d = dist[i][terms[j]];
            if (d < INF) termEdges.push_back(make_tuple(d, i, j));
        }
    }
    sort(termEdges.begin(), termEdges.end());

    DSU dsu1(Keff);
    vector<tuple<int,int,ll>> mstTermEdges;
    for (auto& t : termEdges) {
        ll w = get<0>(t);
        int i = get<1>(t);
        int j = get<2>(t);
        if (dsu1.unite(i, j)) {
            mstTermEdges.push_back(make_tuple(i, j, w));
        }
    }

    unordered_set<int> usedEdgeIds;
    for (auto& t : mstTermEdges) {
        int i = get<0>(t);
        int j = get<1>(t);
        int cur = terms[j];
        while (cur != terms[i] && parent[i][cur] != -1) {
            usedEdgeIds.insert(parentEdge[i][cur]);
            cur = parent[i][cur];
        }
    }

    unordered_set<int> subVerts;
    vector<tuple<ll,int,int,int>> subEdges;
    for (int eid : usedEdgeIds) {
        int u = get<0>(edgeList[eid]);
        int v = get<1>(edgeList[eid]);
        ll w = get<2>(edgeList[eid]);
        subVerts.insert(u);
        subVerts.insert(v);
        subEdges.push_back(make_tuple(w, u, v, eid));
    }
    for (int tt : terms) subVerts.insert(tt);

    sort(subEdges.begin(), subEdges.end());
    DSU dsu2(N);
    vector<tuple<int,int,int>> finalEdges;
    vector<vector<pair<int,int>>> finalAdj(N);
    for (auto& e : subEdges) {
        int u = get<1>(e);
        int v = get<2>(e);
        int eid = get<3>(e);
        if (dsu2.unite(u, v)) {
            finalEdges.push_back(make_tuple(u, v, eid));
            finalAdj[u].push_back({v, eid});
            finalAdj[v].push_back({u, eid});
        }
    }

    vector<int> degree(N, 0);
    for (int v : subVerts) degree[v] = (int)finalAdj[v].size();

    vector<bool> isTerm(N, false);
    for (int tt : terms) isTerm[tt] = true;

    vector<bool> edgeRemoved(edgeList.size(), false);

    queue<int> leafQ;
    for (int v : subVerts) {
        if (degree[v] == 1 && !isTerm[v]) leafQ.push(v);
    }
    while (!leafQ.empty()) {
        int u = leafQ.front(); leafQ.pop();
        if (degree[u] != 1 || isTerm[u]) continue;
        for (auto& pr : finalAdj[u]) {
            int nb = pr.first;
            int eid = pr.second;
            if (!edgeRemoved[eid]) {
                edgeRemoved[eid] = true;
                degree[u]--;
                degree[nb]--;
                if (degree[nb] == 1 && !isTerm[nb]) leafQ.push(nb);
                break;
            }
        }
    }

    vector<pair<int,int>> result;
    totalWeight = 0;
    for (auto& fe : finalEdges) {
        int u = get<0>(fe);
        int v = get<1>(fe);
        int eid = get<2>(fe);
        if (!edgeRemoved[eid]) {
            ll ew = get<2>(edgeList[eid]);
            result.push_back({u, v});
            totalWeight += ew;
        }
    }
    return result;
}

int main() {
    if (scanf("%d %d %d", &N, &M, &K) != 3) return 0;

    adj.assign(N, {});
    edgeList.reserve(M);
    for (int i = 0; i < M; i++) {
        int u, v; ll w;
        scanf("%d %d %lld", &u, &v, &w);
        edgeList.push_back(make_tuple(u, v, w));
        adj[u].push_back({v, w, i});
        adj[v].push_back({u, w, i});
    }

    terms.resize(K);
    termIdx.assign(N, -1);
    for (int i = 0; i < K; i++) {
        scanf("%d", &terms[i]);
        termIdx[terms[i]] = i;
    }

    if (K <= 1) {
        printf("0 0\n");
        return 0;
    }

    ll totalWeight = 0;
    vector<pair<int,int>> edges = solve(totalWeight);

    printf("%lld %zu\n", totalWeight, edges.size());
    for (auto& pr : edges) {
        printf("%d %d\n", pr.first, pr.second);
    }
    return 0;
}
