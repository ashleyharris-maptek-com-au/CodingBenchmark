; SPIR-V
; Version: 1.0
               OpCapability Shader
        %ext = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnvoid = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
        %m4f = OpTypeMatrix %v4f 4

        %UBO = OpTypeStruct %m4f %m4f %m4f %v4f %v4f %v4f
  %p_uni_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %p_uni_UBO Uniform
   %p_uni_v4 = OpTypePointer Uniform %v4f

     %p_in_v3 = OpTypePointer Input %v3f
     %p_in_v2 = OpTypePointer Input %v2f
    %p_out_v4 = OpTypePointer Output %v4f

   %worldPos = OpVariable %p_in_v3 Input
     %normal = OpVariable %p_in_v3 Input
         %uv = OpVariable %p_in_v2 Input
    %tangent = OpVariable %p_in_v3 Input
      %color = OpVariable %p_in_v3 Input
   %outColor = OpVariable %p_out_v4 Output

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
       %f_ax = OpConstant %float 0.1
       %f_ay = OpConstant %float 0.05
       %f_az = OpConstant %float 0.05
       %f_32 = OpConstant %float 32.0
    %ambient = OpConstantComposite %v3f %f_ax %f_ay %f_az
    %specCol = OpConstantComposite %v3f %f_1 %f_1 %f_1
      %zeroV = OpConstantComposite %v3f %f_0 %f_0 %f_0
       %oneV = OpConstantComposite %v3f %f_1 %f_1 %f_1

       %main = OpFunction %void None %fnvoid
      %entry = OpLabel

         %wp = OpLoad %v3f %worldPos
         %nr = OpLoad %v3f %normal
         %vc = OpLoad %v3f %color
          %N = OpExtInst %v3f %ext Normalize %nr

       %p_lp = OpAccessChain %p_uni_v4 %ubo %int_3
        %lp4 = OpLoad %v4f %p_lp
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2
       %ldir = OpFSub %v3f %lp3 %wp
          %L = OpExtInst %v3f %ext Normalize %ldir

       %ndl0 = OpDot %float %N %L
        %ndl = OpExtInst %float %ext FMax %ndl0 %f_0
       %diff = OpVectorTimesScalar %v3f %vc %ndl

       %negL = OpFNegate %v3f %L
          %R = OpExtInst %v3f %ext Reflect %negL %N

       %p_cp = OpAccessChain %p_uni_v4 %ubo %int_4
        %cp4 = OpLoad %v4f %p_cp
        %cp3 = OpVectorShuffle %v3f %cp4 %cp4 0 1 2
       %vdir = OpFSub %v3f %cp3 %wp
          %V = OpExtInst %v3f %ext Normalize %vdir

       %rdv0 = OpDot %float %R %V
        %rdv = OpExtInst %float %ext FMax %rdv0 %f_0
       %spow = OpExtInst %float %ext Pow %rdv %f_32
       %spec = OpVectorTimesScalar %v3f %specCol %spow

         %s1 = OpFAdd %v3f %ambient %diff
        %sum = OpFAdd %v3f %s1 %spec
         %cl = OpExtInst %v3f %ext FClamp %sum %zeroV %oneV

          %r = OpCompositeExtract %float %cl 0
          %g = OpCompositeExtract %float %cl 1
          %b = OpCompositeExtract %float %cl 2
       %out4 = OpCompositeConstruct %v4f %r %g %b %f_1
               OpStore %outColor %out4
               OpReturn
               OpFunctionEnd
