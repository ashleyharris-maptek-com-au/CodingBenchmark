; SPIR-V
; Version: 1.0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
      %fn_ty = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
        %f_0 = OpConstant %float 0.0
      %f_025 = OpConstant %float 0.25
       %f_05 = OpConstant %float 0.5
      %f_075 = OpConstant %float 0.75
        %f_1 = OpConstant %float 1.0
    %f_thick = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
      %color = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_ty
      %entry = OpLabel

       %p0p = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %P0 = OpLoad %v4float %p0p
       %p1p = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %P1 = OpLoad %v4float %p1p
       %p2p = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %P2 = OpLoad %v4float %p2p
       %p3p = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %P3 = OpLoad %v4float %p3p

       %c1p = OpAccessChain %ptr_in_v3 %color %int_1
        %C1 = OpLoad %v3float %c1p
       %c2p = OpAccessChain %ptr_in_v3 %color %int_2
        %C2 = OpLoad %v3float %c2p

    %T0_full = OpFSub %v4float %P2 %P0
    %T1_full = OpFSub %v4float %P3 %P1
         %T0 = OpVectorShuffle %v2float %T0_full %T0_full 0 1
         %T1 = OpVectorShuffle %v2float %T1_full %T1_full 0 1

          %D = OpFSub %v4float %P2 %P1
         %DT = OpFSub %v2float %T1 %T0
         %DC = OpFSub %v3float %C2 %C1

%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ===== t = 0.0 =====
       %tD0 = OpVectorTimesScalar %v4float %D %f_0
      %pos0 = OpFAdd %v4float %P1 %tD0
      %tDT0 = OpVectorTimesScalar %v2float %DT %f_0
     %tang0 = OpFAdd %v2float %T0 %tDT0
      %tDC0 = OpVectorTimesScalar %v3float %DC %f_0
      %col0 = OpFAdd %v3float %C1 %tDC0
    %tangx0 = OpCompositeExtract %float %tang0 0
    %tangy0 = OpCompositeExtract %float %tang0 1
   %ntangy0 = OpFNegate %float %tangy0
  %perpraw0 = OpCompositeConstruct %v2float %ntangy0 %tangx0
 %perpnorm0 = OpExtInst %v2float %1 Normalize %perpraw0
     %perp0 = OpVectorTimesScalar %v2float %perpnorm0 %f_thick
    %perpx0 = OpCompositeExtract %float %perp0 0
    %perpy0 = OpCompositeExtract %float %perp0 1
    %perp40 = OpCompositeConstruct %v4float %perpx0 %perpy0 %f_0 %f_0
  %posplus0 = OpFAdd %v4float %pos0 %perp40
 %posminus0 = OpFSub %v4float %pos0 %perp40
               OpStore %gl_pos_ptr %posplus0
               OpStore %outColor %col0
               OpEmitVertex
               OpStore %gl_pos_ptr %posminus0
               OpStore %outColor %col0
               OpEmitVertex
