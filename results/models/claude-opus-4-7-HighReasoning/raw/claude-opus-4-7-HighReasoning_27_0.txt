#include <cstdlib>

using namespace std;
using Clock = chrono::steady_clock;

static double elapsed(Clock::time_point t0) {
    return chrono::duration<double>(Clock::now() - t0).count();
}

int N, M, K;
vector<vector<int>> adj;
vector<int> color;

static int readInt() {
    int c = getchar();
    while (c != EOF && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = getchar();
    int sign = 1;
    if (c == '-') { sign = -1; c = getchar(); }
    int x = 0;
    while (c >= '0' && c <= '9') { x = x*10 + (c - '0'); c = getchar(); }
    return x * sign;
}

bool greedyDSatur(double timeLimit, Clock::time_point t0) {
    vector<int> col(N, -1);
    vector<int> deg(N);
    for (int i = 0; i < N; i++) deg[i] = (int)adj[i].size();

    vector<vector<uint8_t>> nbColor(N);
    vector<int> sat(N, 0);

    int start = 0;
    for (int i = 1; i < N; i++) if (deg[i] > deg[start]) start = i;

    auto assign = [&](int v, int c) {
        col[v] = c;
        for (int u : adj[v]) {
            if (col[u] == -1) {
                if ((int)nbColor[u].size() <= c) nbColor[u].resize(c+1, 0);
                if (!nbColor[u][c]) {
                    nbColor[u][c] = 1;
                    sat[u]++;
                }
            }
        }
    };

    assign(start, 0);
    int colored = 1;

    while (colored < N) {
        if ((colored & 1023) == 0 && elapsed(t0) > timeLimit) return false;
        int best = -1;
        int bestSat = -1, bestDeg = -1;
        for (int i = 0; i < N; i++) {
            if (col[i] != -1) continue;
            if (sat[i] > bestSat || (sat[i] == bestSat && deg[i] > bestDeg)) {
                bestSat = sat[i];
                bestDeg = deg[i];
                best = i;
            }
        }
        if (best == -1) break;
        int c = 0;
        while (c < (int)nbColor[best].size() && nbColor[best][c]) c++;
        if (c >= K) return false;
        assign(best, c);
        colored++;
    }
    color = col;
    return true;
}

bool greedyRandom(double timeLimit, Clock::time_point t0, uint64_t seed) {
    mt19937_64 rng(seed);
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    if (seed == 0) {
        sort(order.begin(), order.end(), [](int a, int b){
            return adj[a].size() > adj[b].size();
        });
    } else {
        shuffle(order.begin(), order.end(), rng);
    }

    vector<int> col(N, -1);
    vector<int> used(K+1, 0);
    int stamp = 0;

    for (int idx = 0; idx < N; idx++) {
        if ((idx & 0xFFFF) == 0 && elapsed(t0) > timeLimit) return false;
        int v = order[idx];
        stamp++;
        for (int u : adj[v]) {
            if (col[u] != -1) used[col[u]] = stamp;
        }
        int c = -1;
        for (int i = 0; i < K; i++) {
            if (used[i] != stamp) { c = i; break; }
        }
        if (c == -1) return false;
        col[v] = c;
    }
    color = col;
    return true;
}

bool tabuColor(double timeLimit, Clock::time_point t0, uint64_t seed) {
    mt19937_64 rng(seed);
    vector<int> col(N);
    for (int i = 0; i < N; i++) col[i] = (int)(rng() % K);

    if ((long long)N * K > 50'000'000LL) return false;

    vector<vector<int>> ncc(N, vector<int>(K, 0));
    for (int v = 0; v < N; v++) {
        for (int u : adj[v]) {
            ncc[v][col[u]]++;
        }
    }

    long long conflicts = 0;
    for (int v = 0; v < N; v++) conflicts += ncc[v][col[v]];
    conflicts /= 2;

    if (conflicts == 0) { color = col; return true; }

    vector<vector<int>> tabu(N, vector<int>(K, 0));
    int iter = 0;
    int maxIter = 200000;

    while (conflicts > 0 && iter < maxIter) {
        if ((iter & 1023) == 0 && elapsed(t0) > timeLimit) return false;
        iter++;

        long long bestDelta = LLONG_MAX;
        long long bestNonTabuDelta = LLONG_MAX;
        int bestV = -1, bestC = -1;
        int bestNonTabuV = -1, bestNonTabuC = -1;

        for (int v = 0; v < N; v++) {
            int cv = col[v];
            if (ncc[v][cv] == 0) continue;
            for (int c = 0; c < K; c++) {
                if (c == cv) continue;
                long long delta = (long long)ncc[v][c] - ncc[v][cv];
                if (delta < bestDelta) {
                    bestDelta = delta;
                    bestV = v; bestC = c;
                }
                if (tabu[v][c] <= iter && delta < bestNonTabuDelta) {
                    bestNonTabuDelta = delta;
                    bestNonTabuV = v; bestNonTabuC = c;
                }
            }
        }

        int useV, useC;
        long long useDelta;
        if (bestNonTabuV != -1) {
            useV = bestNonTabuV; useC = bestNonTabuC; useDelta = bestNonTabuDelta;
        } else if (bestV != -1) {
            useV = bestV; useC = bestC; useDelta = bestDelta;
        } else {
            break;
        }

        int oldC = col[useV];
        col[useV] = useC;
        for (int u : adj[useV]) {
            ncc[u][oldC]--;
            ncc[u][useC]++;
        }
        conflicts += useDelta;

        tabu[useV][oldC] = iter + 10 + (int)(rng() % 10);
    }

    if (conflicts == 0) { color = col; return true; }
    return false;
}

int main() {
    auto t0 = Clock::now();
    double timeBudget = 270.0;

    N = readInt();
    M = readInt();
    K = readInt();

    adj.assign(N, {});
    vector<int> degCount(N, 0);
    vector<pair<int,int>> edges;
    edges.reserve(M);
    for (int i = 0; i < M; i++) {
        int u = readInt(), v = readInt();
        if (u < 0 || v < 0 || u >= N || v >= N || u == v) continue;
        edges.push_back({u, v});
        degCount[u]++;
        degCount[v]++;
    }
    for (int i = 0; i < N; i++) adj[i].reserve(degCount[i]);
    for (auto& e : edges) {
        adj[e.first].push_back(e.second);
        adj[e.second].push_back(e.first);
    }
    edges.clear(); edges.shrink_to_fit();

    if ((long long)N <= 200000) {
        for (int i = 0; i < N; i++) {
            auto& a = adj[i];
            sort(a.begin(), a.end());
            a.erase(unique(a.begin(), a.end()), a.end());
        }
    }

    bool ok = false;

    if (!ok) {
        ok = greedyRandom(min(30.0, timeBudget * 0.2), t0, 0);
    }

    if (!ok && N <= 50000) {
        ok = greedyDSatur(min(60.0, timeBudget * 0.3), t0);
    }

    if (!ok) {
        for (int attempt = 1; attempt < 30 && !ok; attempt++) {
            if (elapsed(t0) > timeBudget * 0.6) break;
            ok = greedyRandom(min(8.0, timeBudget * 0.05), t0, (uint64_t)attempt * 7919ULL + 1);
        }
    }

    if (!ok) {
        double remaining = timeBudget - elapsed(t0) - 5.0;
        if (remaining > 5.0) {
            ok = tabuColor(remaining, t0, 12345);
        }
    }

    if (!ok || (int)color.size() != N) {
        color.assign(N, -1);
        vector<int> used(K+1, 0);
        int stamp = 0;
        vector<int> order(N);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [](int a, int b){
            return adj[a].size() > adj[b].size();
        });
        for (int idx = 0; idx < N; idx++) {
            int v = order[idx];
            stamp++;
            for (int u : adj[v]) {
                if (color[u] != -1) used[color[u]] = stamp;
            }
            int c = 0;
            for (int i = 0; i < K; i++) {
                if (used[i] != stamp) { c = i; break; }
            }
            if (c >= K) c = 0;
            color[v] = c;
        }
    }

    for (int i = 0; i < N; i++) {
        if (i) putchar(' ');
        int x = color[i];
        if (x < 0) x = 0;
        char buf[16]; int len = 0;
        if (x == 0) buf[len++] = '0';
        else { while (x) { buf[len++] = '0' + (x % 10); x /= 10; } }
        for (int j = len - 1; j >= 0; j--) putchar(buf[j]);
    }
    putchar('\n');
    return 0;
}
