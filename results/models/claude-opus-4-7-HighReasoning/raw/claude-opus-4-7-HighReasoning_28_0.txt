use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<u32>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: u32 = it.next().unwrap().parse().unwrap();
        let v: u32 = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u as usize].push(v);
        adj[v as usize].push(u);
    }

    for a in adj.iter_mut() {
        a.sort_unstable();
        a.dedup();
    }

    let mut degree: Vec<u32> = adj.iter().map(|a| a.len() as u32).collect();
    let mut alive = vec![true; n];
    let mut in_cover = vec![false; n];

    let mut heap: BinaryHeap<(u32, u32)> = BinaryHeap::with_capacity(n);
    for v in 0..n {
        if degree[v] > 0 {
            heap.push((degree[v], v as u32));
        }
    }

    while let Some((d, v)) = heap.pop() {
        let vu = v as usize;
        if !alive[vu] || degree[vu] != d || d == 0 {
            continue;
        }

        // Degree-1 reduction: if vu has exactly one remaining neighbor,
        // putting that neighbor in the cover is optimal.
        let pick = if d == 1 {
            let mut nb = vu;
            for &w in adj[vu].iter() {
                if alive[w as usize] {
                    nb = w as usize;
                    break;
                }
            }
            nb
        } else {
            vu
        };

        in_cover[pick] = true;
        alive[pick] = false;
        let neighbors: Vec<u32> = std::mem::take(&mut adj[pick]);
        for &w in neighbors.iter() {
            let wu = w as usize;
            if alive[wu] && degree[wu] > 0 {
                degree[wu] -= 1;
                if degree[wu] > 0 {
                    heap.push((degree[wu], w));
                }
            }
        }
    }

    let cover: Vec<usize> = (0..n).filter(|&i| in_cover[i]).collect();
    writeln!(out, "{}", cover.len()).unwrap();
    let mut first = true;
    for &c in cover.iter() {
        if !first {
            write!(out, " ").unwrap();
        }
        write!(out, "{}", c).unwrap();
        first = false;
    }
    writeln!(out).unwrap();
}
