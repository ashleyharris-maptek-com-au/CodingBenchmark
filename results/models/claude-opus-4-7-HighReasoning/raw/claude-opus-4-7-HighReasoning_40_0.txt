#include <bits/stdc++.h>
using namespace std;

// 0/1 ILP with constraints x_u + x_v <= 1  is Maximum Weight Independent Set.
// Variables with c_i <= 0 should be set to 0.

struct Solver {
    vector<int> sub;              // local -> original
    vector<long long> w;
    vector<vector<int>> g;
    int n;
    vector<int> bestPick;
    long long bestVal;

    long long upperBound(const vector<char>& alive) {
        long long s = 0;
        for (int i = 0; i < n; i++) if (alive[i] && w[i] > 0) s += w[i];
        return s;
    }

    void branch(vector<char> alive, long long curVal, vector<int> picked) {
        bool changed = true;
        while (changed) {
            changed = false;
            for (int i = 0; i < n; i++) {
                if (!alive[i]) continue;
                if (w[i] <= 0) { alive[i] = 0; changed = true; continue; }
                bool hasNbr = false;
                for (int j : g[i]) if (alive[j]) { hasNbr = true; break; }
                if (!hasNbr) {
                    curVal += w[i];
                    picked.push_back(i);
                    alive[i] = 0;
                    changed = true;
                }
            }
            // Domination-ish: if w[i] >= sum of positive neighbor weights, take it.
            for (int i = 0; i < n; i++) {
                if (!alive[i] || w[i] <= 0) continue;
                long long sumNbr = 0;
                for (int j : g[i]) if (alive[j] && w[j] > 0) sumNbr += w[j];
                if (w[i] >= sumNbr) {
                    curVal += w[i];
                    picked.push_back(i);
                    alive[i] = 0;
                    for (int j : g[i]) if (alive[j]) alive[j] = 0;
                    changed = true;
                }
            }
        }

        long long ub = curVal + upperBound(alive);
        if (ub <= bestVal) return;

        int pivot = -1;
        int bestDeg = -1;
        for (int i = 0; i < n; i++) {
            if (!alive[i] || w[i] <= 0) continue;
            int d = 0;
            for (int j : g[i]) if (alive[j]) d++;
            if (d > bestDeg) { bestDeg = d; pivot = i; }
        }

        if (pivot == -1) {
            if (curVal > bestVal) {
                bestVal = curVal;
                bestPick = picked;
            }
            return;
        }

        // Include pivot
        {
            vector<char> a2 = alive;
            vector<int> p2 = picked;
            long long v2 = curVal + w[pivot];
            a2[pivot] = 0;
            for (int j : g[pivot]) if (a2[j]) a2[j] = 0;
            p2.push_back(pivot);
            if (v2 + upperBound(a2) > bestVal)
                branch(a2, v2, p2);
        }
        // Exclude pivot
        {
            vector<char> a2 = alive;
            a2[pivot] = 0;
            if (curVal + upperBound(a2) > bestVal)
                branch(a2, curVal, picked);
        }
    }

    pair<long long, vector<int>> solve() {
        n = (int)sub.size();
        bestVal = 0;
        bestPick.clear();
        vector<char> alive(n, 1);
        branch(alive, 0LL, {});
        return {bestVal, bestPick};
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    vector<long long> weight(n, 0);
    vector<vector<int>> adj(n);
    for (int i = 0; i < n; i++) cin >> weight[i];
    for (int i = 0; i < m; i++) {
        int u, v; cin >> u >> v;
        if (u == v) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    for (int i = 0; i < n; i++) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    vector<int> x(n, 0);
    long long totalObj = 0;

    vector<int> comp(n, -1);
    for (int s = 0; s < n; s++) {
        if (comp[s] != -1) continue;
        vector<int> nodes;
        comp[s] = s;
        queue<int> q;
        q.push(s);
        while (!q.empty()) {
            int u = q.front(); q.pop();
            nodes.push_back(u);
            for (int v : adj[u]) {
                if (comp[v] == -1) { comp[v] = s; q.push(v); }
            }
        }
        unordered_map<int,int> loc;
        loc.reserve(nodes.size()*2);
        for (int i = 0; i < (int)nodes.size(); i++) loc[nodes[i]] = i;

        Solver S;
        S.sub = nodes;
        S.w.resize(nodes.size());
        S.g.assign(nodes.size(), {});
        for (int i = 0; i < (int)nodes.size(); i++) {
            S.w[i] = weight[nodes[i]];
            for (int v : adj[nodes[i]]) {
                auto it = loc.find(v);
                if (it != loc.end()) S.g[i].push_back(it->second);
            }
        }
        auto res = S.solve();
        totalObj += res.first;
        for (int li : res.second) x[nodes[li]] = 1;
    }

    cout << totalObj << "\n";
    for (int i = 0; i < n; i++) {
        cout << x[i];
        cout << (i+1==n ? '\n' : ' ');
    }
    return 0;
}
