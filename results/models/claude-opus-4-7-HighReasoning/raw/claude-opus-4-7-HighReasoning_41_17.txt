; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnVoid = OpTypeFunction %void
        %flt = OpTypeFloat 32
         %v2 = OpTypeVector %flt 2
         %v3 = OpTypeVector %flt 3
         %v4 = OpTypeVector %flt 4
       %mat4 = OpTypeMatrix %v4 4
        %int = OpTypeInt 32 1
       %bool = OpTypeBool

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
     %pU_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %pU_UBO Uniform
      %pU_v4 = OpTypePointer Uniform %v4

     %pIn_v3 = OpTypePointer Input %v3
     %pIn_v2 = OpTypePointer Input %v2
    %pOut_v4 = OpTypePointer Output %v4

   %worldPos = OpVariable %pIn_v3 Input
     %normal = OpVariable %pIn_v3 Input
         %uv = OpVariable %pIn_v2 Input
   %outColor = OpVariable %pOut_v4 Output

        %f0  = OpConstant %flt 0.0
        %f1  = OpConstant %flt 1.0
        %f2  = OpConstant %flt 2.0
        %f8  = OpConstant %flt 8.0
        %f16 = OpConstant %flt 16.0
        %f05 = OpConstant %flt 0.5
        %f005= OpConstant %flt 0.05
        %f01 = OpConstant %flt 0.1
        %f04 = OpConstant %flt 0.4
        %f06 = OpConstant %flt 0.6
        %f07 = OpConstant %flt 0.7
        %f02 = OpConstant %flt 0.2
        %fbB = OpConstant %flt 0.1
        %i3  = OpConstant %int 3

     %mortar = OpConstantComposite %v3 %f07 %f07 %f07
      %brick = OpConstantComposite %v3 %f07 %f02 %fbB

       %main = OpFunction %void None %fnVoid
      %entry = OpLabel
       %uvL  = OpLoad %v2 %uv
       %ux   = OpCompositeExtract %flt %uvL 0
       %uy   = OpCompositeExtract %flt %uvL 1
       %sy   = OpFMul %flt %uy %f16
       %fly  = OpExtInst %flt %glsl Floor %sy
       %div2 = OpFMul %flt %fly %f05
       %fld2 = OpExtInst %flt %glsl Floor %div2
       %mul2 = OpFMul %flt %fld2 %f2
       %modR = OpFSub %flt %fly %mul2
       %off  = OpFMul %flt %modR %f05
       %sx   = OpFMul %flt %ux %f8
       %sxOff= OpFAdd %flt %sx %off
       %lx   = OpExtInst %flt %glsl Fract %sxOff
       %ly   = OpExtInst %flt %glsl Fract %sy
       %isMx = OpFOrdLessThan %bool %lx %f005
       %isMy = OpFOrdLessThan %bool %ly %f01
       %isM  = OpLogicalOr %bool %isMx %isMy
       %base = OpSelect %v3 %isM %mortar %brick

       %nL   = OpLoad %v3 %normal
       %N    = OpExtInst %v3 %glsl Normalize %nL
       %wp   = OpLoad %v3 %worldPos
       %lpP  = OpAccessChain %pU_v4 %ubo %i3
       %lp4  = OpLoad %v4 %lpP
       %lpx  = OpCompositeExtract %flt %lp4 0
       %lpy  = OpCompositeExtract %flt %lp4 1
       %lpz  = OpCompositeExtract %flt %lp4 2
       %lp3  = OpCompositeConstruct %v3 %lpx %lpy %lpz
       %ldir = OpFSub %v3 %lp3 %wp
       %L    = OpExtInst %v3 %glsl Normalize %ldir
       %d    = OpDot %flt %N %L
       %dmax = OpExtInst %flt %glsl FMax %d %f0
       %fac1 = OpFMul %flt %dmax %f06
       %fac  = OpFAdd %flt %f04 %fac1
       %col3 = OpVectorTimesScalar %v3 %base %fac
       %cx   = OpCompositeExtract %flt %col3 0
       %cy   = OpCompositeExtract %flt %col3 1
       %cz   = OpCompositeExtract %flt %col3 2
       %outV = OpCompositeConstruct %v4 %cx %cy %cz %f1
               OpStore %outColor %outV
               OpReturn
               OpFunctionEnd
