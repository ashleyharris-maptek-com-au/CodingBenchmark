import sys
import math

def main():
    data = sys.stdin.read().split()
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    CAP = float(data[idx]); idx += 1
    dx = float(data[idx]); idx += 1
    dy = float(data[idx]); idx += 1
    xs = [dx]
    ys = [dy]
    dem = [0.0]
    for _ in range(N):
        xs.append(float(data[idx])); idx += 1
        ys.append(float(data[idx])); idx += 1
        dem.append(float(data[idx])); idx += 1

    n = N + 1
    D = [[0.0] * n for _ in range(n)]
    for i in range(n):
        for j in range(i + 1, n):
            d = math.hypot(xs[i] - xs[j], ys[i] - ys[j])
            D[i][j] = d
            D[j][i] = d

    # Clarke-Wright savings
    route_of = [0] * (N + 1)
    routes = {}
    route_load = {}
    route_left = {}
    route_right = {}

    for i in range(1, N + 1):
        routes[i] = [i]
        route_load[i] = dem[i]
        route_of[i] = i
        route_left[i] = i
        route_right[i] = i

    savings = []
    for i in range(1, N + 1):
        for j in range(i + 1, N + 1):
            s = D[0][i] + D[0][j] - D[i][j]
            if s > 0:
                savings.append((s, i, j))
    savings.sort(reverse=True)

    for s, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            continue
        if route_load[ri] + route_load[rj] > CAP + 1e-9:
            continue
        i_end = (i == route_left[ri] or i == route_right[ri])
        j_end = (j == route_left[rj] or j == route_right[rj])
        if not (i_end and j_end):
            continue
        ri_route = routes[ri]
        rj_route = routes[rj]
        if ri_route[0] == i:
            ri_route.reverse()
        if rj_route[-1] == j:
            rj_route.reverse()
        merged = ri_route + rj_route
        new_load = route_load[ri] + route_load[rj]
        routes[ri] = merged
        route_load[ri] = new_load
        route_left[ri] = merged[0]
        route_right[ri] = merged[-1]
        for c in rj_route:
            route_of[c] = ri
        del routes[rj]
        del route_load[rj]
        del route_left[rj]
        del route_right[rj]

    route_list = list(routes.values())

    # Force route count <= K by merging cheapest pairs that fit
    while len(route_list) > K:
        best = None
        for a in range(len(route_list)):
            for b in range(a + 1, len(route_list)):
                la = sum(dem[c] for c in route_list[a])
                lb = sum(dem[c] for c in route_list[b])
                if la + lb > CAP + 1e-9:
                    continue
                ra = route_list[a]
                rb = route_list[b]
                options = [
                    (D[ra[-1]][rb[0]] - D[ra[-1]][0] - D[0][rb[0]], ra + rb),
                    (D[ra[-1]][rb[-1]] - D[ra[-1]][0] - D[0][rb[-1]], ra + rb[::-1]),
                    (D[ra[0]][rb[0]] - D[ra[0]][0] - D[0][rb[0]], ra[::-1] + rb),
                    (D[ra[0]][rb[-1]] - D[ra[0]][0] - D[0][rb[-1]], ra[::-1] + rb[::-1]),
                ]
                cost, merged = min(options, key=lambda x: x[0])
                if best is None or cost < best[0]:
                    best = (cost, a, b, merged)
        if best is None:
            break
        _, a, b, merged = best
        new_list = [r for k, r in enumerate(route_list) if k != a and k != b]
        new_list.append(merged)
        route_list = new_list

    # 2-opt per route
    def two_opt(route):
        if len(route) < 3:
            return route
        path = [0] + route + [0]
        improved = True
        passes = 0
        max_passes = 50 if len(route) <= 50 else 5
        while improved and passes < max_passes:
            improved = False
            passes += 1
            for i in range(1, len(path) - 2):
                for j in range(i + 1, len(path) - 1):
                    a = path[i - 1]
                    b = path[i]
                    c = path[j]
                    d = path[j + 1]
                    if a == c or b == d:
                        continue
                    delta = D[a][c] + D[b][d] - D[a][b] - D[c][d]
                    if delta < -1e-12:
                        path[i:j + 1] = path[i:j + 1][::-1]
                        improved = True
        return path[1:-1]

    route_list = [two_opt(r) for r in route_list]

    total = 0.0
    for r in route_list:
        prev = 0
        for c in r:
            total += D[prev][c]
            prev = c
        total += D[prev][0]

    out = []
    out.append(f"{total:.6f}")
    out.append(str(len(route_list)))
    for r in route_list:
        out.append(" ".join(str(c - 1) for c in r))
    sys.stdout.write("\n".join(out) + "\n")

if __name__ == "__main__":
    main()
