use std::io::{self, Read, Write, BufWriter};

struct Solver {
    w: usize,
    h: usize,
    n: usize,
    blocked: Vec<bool>,
    visited: Vec<bool>,
    target_count: usize,
    require_cycle: bool,
    path: Vec<usize>,
    deadline: std::time::Instant,
    found: bool,
}

impl Solver {
    #[inline]
    fn neighbors(&self, i: usize) -> [Option<usize>; 4] {
        let x = i % self.w;
        let y = i / self.w;
        let mut r = [None; 4];
        if x > 0 { r[0] = Some(i - 1); }
        if x + 1 < self.w { r[1] = Some(i + 1); }
        if y > 0 { r[2] = Some(i - self.w); }
        if y + 1 < self.h { r[3] = Some(i + self.w); }
        r
    }

    #[inline]
    fn free_degree(&self, i: usize) -> u32 {
        let mut c = 0u32;
        for n in self.neighbors(i).iter().flatten() {
            if !self.blocked[*n] && !self.visited[*n] { c += 1; }
        }
        c
    }

    // Connectivity + dead-end pruning.
    fn connectivity_ok(&self, current: usize, remaining: usize) -> bool {
        if remaining == 0 { return true; }
        let nbrs = self.neighbors(current);
        let mut start = usize::MAX;
        for n in nbrs.iter().flatten() {
            if !self.blocked[*n] && !self.visited[*n] {
                start = *n;
                break;
            }
        }
        if start == usize::MAX { return false; }

        let mut seen = vec![false; self.n];
        let mut stack: Vec<usize> = Vec::with_capacity(remaining);
        stack.push(start);
        seen[start] = true;
        let mut count = 0usize;
        let mut deg1_count = 0u32;
        // For cycle require: also need to allow start(0) connection. We'll check
        // that at least one remaining cell is adjacent to (0,0) if cycle required
        // and remaining > 0.
        let mut adj_to_start_count = 0u32;

        while let Some(u) = stack.pop() {
            count += 1;
            let mut deg = 0u32;
            for nn in self.neighbors(u).iter().flatten() {
                let v = *nn;
                if !self.blocked[v] && !self.visited[v] {
                    deg += 1;
                    if !seen[v] {
                        seen[v] = true;
                        stack.push(v);
                    }
                }
            }
            if deg <= 1 { deg1_count += 1; }
            if self.require_cycle {
                // check if u is adjacent to cell 0
                for nn in self.neighbors(u).iter().flatten() {
                    if *nn == 0 { adj_to_start_count += 1; break; }
                }
            }
        }
        if count != remaining { return false; }

        if self.require_cycle {
            if adj_to_start_count == 0 { return false; }
            if deg1_count > 1 { return false; }
        } else if deg1_count > 2 {
            return false;
        }

        true
    }

    fn solve(&mut self, current: usize, depth: usize) -> bool {
        if self.found { return true; }
        if depth & 4095 == 0 {
            if std::time::Instant::now() >= self.deadline {
                return false;
            }
        }

        if depth == self.target_count {
            if self.require_cycle {
                for n in self.neighbors(current).iter().flatten() {
                    if *n == 0 { self.found = true; return true; }
                }
                return false;
            } else {
                self.found = true;
                return true;
            }
        }

        let remaining = self.target_count - depth;

        if !self.connectivity_ok(current, remaining) {
            return false;
        }

        let nbrs = self.neighbors(current);
        let mut cands: [(u32, usize); 4] = [(u32::MAX, usize::MAX); 4];
        let mut nc = 0usize;
        for n in nbrs.iter().flatten() {
            let v = *n;
            if !self.blocked[v] && !self.visited[v] {
                if self.require_cycle && v == 0 { continue; }
                self.visited[v] = true;
                let d = self.free_degree(v);
                self.visited[v] = false;
                cands[nc] = (d, v);
                nc += 1;
            }
        }
        if nc == 0 { return false; }

        cands[..nc].sort_by_key(|x| x.0);

        if cands[0].0 == 0 && remaining > 1 {
            return false;
        }

        for k in 0..nc {
            let v = cands[k].1;
            self.visited[v] = true;
            self.path.push(v);
            if self.solve(v, depth + 1) { return true; }
            self.path.pop();
            self.visited[v] = false;
            if self.found { return true; }
        }
        false
    }
}

fn parity_color(x: usize, y: usize) -> u32 { ((x + y) & 1) as u32 }

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let w: usize = it.next().unwrap().parse().unwrap();
    let h: usize = it.next().unwrap().parse().unwrap();
    let obs: usize = it.next().unwrap().parse().unwrap();
    let rc_flag: u32 = it.next().unwrap().parse().unwrap();
    let require_cycle = rc_flag != 0;

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obs {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    if blocked[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let target_count: usize = blocked.iter().filter(|b| !**b).count();
    if target_count == 1 {
        if require_cycle {
            writeln!(out, "0").unwrap();
        } else {
            writeln!(out, "1").unwrap();
            writeln!(out, "0 0").unwrap();
        }
        return;
    }

    // Bipartite parity feasibility
    let mut b0 = 0i64; let mut b1 = 0i64;
    for y in 0..h {
        for x in 0..w {
            if !blocked[y * w + x] {
                if parity_color(x, y) == 0 { b0 += 1; } else { b1 += 1; }
            }
        }
    }
    let t = target_count as i64;
    let need0 = (t + 1) / 2;
    let need1 = t / 2;
    if b0 != need0 || b1 != need1 {
        writeln!(out, "0").unwrap();
        return;
    }
    if require_cycle && t % 2 != 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let deadline = std::time::Instant::now() + std::time::Duration::from_secs(270);

    let mut solver = Solver {
        w, h, n,
        blocked: blocked.clone(),
        visited: vec![false; n],
        target_count,
        require_cycle,
        path: Vec::with_capacity(target_count),
        deadline,
        found: false,
    };
    solver.visited[0] = true;
    solver.path.push(0);

    let _ = solver.solve(0, 1);

    if !solver.found {
        writeln!(out, "0").unwrap();
        return;
    }

    writeln!(out, "{}", solver.path.len()).unwrap();
    for &i in &solver.path {
        let x = i % w;
        let y = i / w;
        writeln!(out, "{} {}", x, y).unwrap();
    }
}
