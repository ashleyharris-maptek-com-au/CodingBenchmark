OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor %worldPos %normal
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
       %glsl = OpExtInstImport "GLSL.std.450"

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
       %fnty = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
        %f_0 = OpConstant %float 0.0
        %f_1 = OpConstant %float 1.0
    %f_third = OpConstant %float 0.33333333
   %f_2third = OpConstant %float 0.66666667
    %f_thick = OpConstant %float 0.02
   %v2_thick = OpConstantComposite %v2float %f_thick %f_thick
   %v2_third = OpConstantComposite %v2float %f_third %f_third
  %v2_2third = OpConstantComposite %v2float %f_2third %f_2third
   %v3_third = OpConstantComposite %v3float %f_third %f_third %f_third
  %v3_2third = OpConstantComposite %v3float %f_2third %f_2third %f_2third
   %v4_third = OpConstantComposite %v4float %f_third %f_third %f_third %f_third
  %v4_2third = OpConstantComposite %v4float %f_2third %f_2third %f_2third %f_2third

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
  %ptr_in_v4 = OpTypePointer Input %v4float

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
 %ptr_out_v4 = OpTypePointer Output %v4float

  %arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input
  %ptr_in_v3 = OpTypePointer Input %v3float
 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fnty
      %entry = OpLabel

        %pp0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %pp0
        %pp1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %pp1
        %pp2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %pp2
        %pp3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %pp3

        %pc1 = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %pc1
        %pc2 = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %pc2

       %p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
       %p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
       %p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
       %p3xy = OpVectorShuffle %v2float %p3 %p3 0 1

         %ts = OpFSub %v2float %p2xy %p0xy
        %tsN = OpExtInst %v2float %glsl Normalize %ts
        %tsx = OpCompositeExtract %float %tsN 0
        %tsy = OpCompositeExtract %float %tsN 1
       %ntsy = OpFNegate %float %tsy
     %ps_raw = OpCompositeConstruct %v2float %ntsy %tsx
     %perp_s = OpFMul %v2float %ps_raw %v2_thick

         %te = OpFSub %v2float %p3xy %p1xy
        %teN = OpExtInst %v2float %glsl Normalize %te
        %tex = OpCompositeExtract %float %teN 0
        %tey = OpCompositeExtract %float %teN 1
       %ntey = OpFNegate %float %tey
     %pe_raw = OpCompositeConstruct %v2float %ntey %tex
     %perp_e = OpFMul %v2float %pe_raw %v2_thick

      %pos_0 = OpCopyObject %v4float %p1
     %perp_0 = OpCopyObject %v2float %perp_s
      %col_0 = OpCopyObject %v3float %c1

     %pos_13 = OpExtInst %v4float %glsl FMix %p1 %p2 %v4_third
    %perp_13 = OpExtInst %v2float %glsl FMix %perp_s %perp_e %v2_third
     %col_13 = OpExtInst %v3float %glsl FMix %c1 %c2 %v3_third

     %pos_23 = OpExtInst %v4float %glsl FMix %p1 %p2 %v4_2third
    %perp_23 = OpExtInst %v2float %glsl FMix %perp_s %perp_e %v2_2third
     %col_23 = OpExtInst %v3float %glsl FMix %c1 %c2 %v3_2third

      %pos_1 = OpCopyObject %v4float %p2
     %perp_1 = OpCopyObject %v2float %perp_e
      %col_1 = OpCopyObject %v3float %c2

; ---------------- Quad 1: t = 0 .. 1/3 ----------------
       %q1v0_px = OpCompositeExtract %float %pos_0 0
       %q1v0_py = OpCompositeExtract %float %pos_0 1
       %q1v0_pz = OpCompositeExtract %float %pos_0 2
       %q1v0_pw = OpCompositeExtract %float %pos_0 3
       %q1v0_dx = OpCompositeExtract %float %perp_0 0
       %q1v0_dy = OpCompositeExtract %float %perp_0 1
       %q1v0_nx = OpFAdd %float %q1v0_px %q1v0_dx
       %q1v0_ny = OpFAdd %float %q1v0_py %q1v0_dy
       %q1v0_p  = OpCompositeConstruct %v4float %q1v0_nx %q1v0_ny %q1v0_pz %q1v0_pw
       %q1v0_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q1v0_op %q1v0_p
                  OpStore %outColor %col_0
                  OpEmitVertex

       %q1v1_nx = OpFSub %float %q1v0_px %q1v0_dx
       %q1v1_ny = OpFSub %float %q1v0_py %q1v0_dy
       %q1v1_p  = OpCompositeConstruct %v4float %q1v1_nx %q1v1_ny %q1v0_pz %q1v0_pw
       %q1v1_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q1v1_op %q1v1_p
                  OpStore %outColor %col_0
                  OpEmitVertex

       %q1v2_px = OpCompositeExtract %float %pos_13 0
       %q1v2_py = OpCompositeExtract %float %pos_13 1
       %q1v2_pz = OpCompositeExtract %float %pos_13 2
       %q1v2_pw = OpCompositeExtract %float %pos_13 3
       %q1v2_dx = OpCompositeExtract %float %perp_13 0
       %q1v2_dy = OpCompositeExtract %float %perp_13 1
       %q1v2_nx = OpFAdd %float %q1v2_px %q1v2_dx
       %q1v2_ny = OpFAdd %float %q1v2_py %q1v2_dy
       %q1v2_p  = OpCompositeConstruct %v4float %q1v2_nx %q1v2_ny %q1v2_pz %q1v2_pw
       %q1v2_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q1v2_op %q1v2_p
                  OpStore %outColor %col_13
                  OpEmitVertex

       %q1v3_nx = OpFSub %float %q1v2_px %q1v2_dx
       %q1v3_ny = OpFSub %float %q1v2_py %q1v2_dy
       %q1v3_p  = OpCompositeConstruct %v4float %q1v3_nx %q1v3_ny %q1v2_pz %q1v2_pw
       %q1v3_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q1v3_op %q1v3_p
                  OpStore %outColor %col_13
                  OpEmitVertex

                  OpEndPrimitive

; ---------------- Quad 2: t = 2/3 .. 1 ----------------
       %q2v0_px = OpCompositeExtract %float %pos_23 0
       %q2v0_py = OpCompositeExtract %float %pos_23 1
       %q2v0_pz = OpCompositeExtract %float %pos_23 2
       %q2v0_pw = OpCompositeExtract %float %pos_23 3
       %q2v0_dx = OpCompositeExtract %float %perp_23 0
       %q2v0_dy = OpCompositeExtract %float %perp_23 1
       %q2v0_nx = OpFAdd %float %q2v0_px %q2v0_dx
       %q2v0_ny = OpFAdd %float %q2v0_py %q2v0_dy
       %q2v0_p  = OpCompositeConstruct %v4float %q2v0_nx %q2v0_ny %q2v0_pz %q2v0_pw
       %q2v0_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q2v0_op %q2v0_p
                  OpStore %outColor %col_23
                  OpEmitVertex

       %q2v1_nx = OpFSub %float %q2v0_px %q2v0_dx
       %q2v1_ny = OpFSub %float %q2v0_py %q2v0_dy
       %q2v1_p  = OpCompositeConstruct %v4float %q2v1_nx %q2v1_ny %q2v0_pz %q2v0_pw
       %q2v1_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q2v1_op %q2v1_p
                  OpStore %outColor %col_23
                  OpEmitVertex

       %q2v2_px = OpCompositeExtract %float %pos_1 0
       %q2v2_py = OpCompositeExtract %float %pos_1 1
       %q2v2_pz = OpCompositeExtract %float %pos_1 2
       %q2v2_pw = OpCompositeExtract %float %pos_1 3
       %q2v2_dx = OpCompositeExtract %float %perp_1 0
       %q2v2_dy = OpCompositeExtract %float %perp_1 1
       %q2v2_nx = OpFAdd %float %q2v2_px %q2v2_dx
       %q2v2_ny = OpFAdd %float %q2v2_py %q2v2_dy
       %q2v2_p  = OpCompositeConstruct %v4float %q2v2_nx %q2v2_ny %q2v2_pz %q2v2_pw
       %q2v2_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q2v2_op %q2v2_p
                  OpStore %outColor %col_1
                  OpEmitVertex

       %q2v3_nx = OpFSub %float %q2v2_px %q2v2_dx
       %q2v3_ny = OpFSub %float %q2v2_py %q2v2_dy
       %q2v3_p  = OpCompositeConstruct %v4float %q2v3_nx %q2v3_ny %q2v2_pz %q2v2_pw
       %q2v3_op = OpAccessChain %ptr_out_v4 %gl_out %int_0
                  OpStore %q2v3_op %q2v3_p
                  OpStore %outColor %col_1
                  OpEmitVertex

                  OpEndPrimitive

                  OpReturn
                  OpFunctionEnd
