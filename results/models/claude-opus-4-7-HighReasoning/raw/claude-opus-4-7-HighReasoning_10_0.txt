#include <cstdio>
#include <cstdlib>
#include <cstdint>
#include <climits>
#include <vector>
#include <algorithm>
#include <chrono>

using namespace std;

struct Rect { int w, h; };

struct Placement { int board, x, y, rotated; };

struct SkyNode { int x, y, width; };

struct Board {
    int W, H;
    vector<SkyNode> sky;
    Board(int w, int h) : W(w), H(h) { sky.push_back({0, 0, w}); }

    // Bottom-left find: returns true if fits, sets bestX, bestY
    bool find(int rw, int rh, int &bestX, int &bestY) const {
        bestY = INT_MAX;
        bestX = INT_MAX;
        int n = (int)sky.size();
        for (int i = 0; i < n; i++) {
            int x = sky[i].x;
            if (x + rw > W) continue;
            int y = sky[i].y;
            int widthLeft = rw;
            int j = i;
            bool fits = true;
            while (widthLeft > 0) {
                if (j >= n) { fits = false; break; }
                if (sky[j].y > y) y = sky[j].y;
                if (y + rh > H) { fits = false; break; }
                int take = min(widthLeft, sky[j].width - (j == i ? 0 : 0));
                widthLeft -= sky[j].width;
                j++;
            }
            if (!fits) continue;
            if (y < bestY || (y == bestY && x < bestX)) {
                bestY = y; bestX = x;
            }
        }
        return bestY != INT_MAX;
    }

    void place(int x, int y, int rw, int rh) {
        SkyNode nn = {x, y + rh, rw};
        int xe = x + rw;
        vector<SkyNode> ns;
        ns.reserve(sky.size() + 2);
        for (auto &s : sky) {
            int sx = s.x, sw = s.width, sxe = sx + sw;
            if (sxe <= x || sx >= xe) {
                ns.push_back(s);
            } else {
                if (sx < x) ns.push_back({sx, s.y, x - sx});
                if (sxe > xe) ns.push_back({xe, s.y, sxe - xe});
            }
        }
        ns.push_back(nn);
        sort(ns.begin(), ns.end(), [](const SkyNode &a, const SkyNode &b){ return a.x < b.x; });
        sky.clear();
        for (auto &s : ns) {
            if (!sky.empty() && sky.back().y == s.y && sky.back().x + sky.back().width == s.x) {
                sky.back().width += s.width;
            } else {
                sky.push_back(s);
            }
        }
    }
};

int main() {
    int N, BW, BH;
    if (scanf("%d %d %d", &N, &BW, &BH) != 3) return 0;
    vector<Rect> rects(N);
    for (int i = 0; i < N; i++) scanf("%d %d", &rects[i].w, &rects[i].h);

    vector<int> order(N);
    for (int i = 0; i < N; i++) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b){
        int ma = max(rects[a].w, rects[a].h);
        int mb = max(rects[b].w, rects[b].h);
        if (ma != mb) return ma > mb;
        return rects[a].w * rects[a].h > rects[b].w * rects[b].h;
    });

    bool firstFit = N > 3000;

    vector<Placement> placements(N);
    vector<Board> boards;
    auto t0 = chrono::steady_clock::now();

    for (int k = 0; k < N; k++) {
        int ri = order[k];
        int rw = rects[ri].w, rh = rects[ri].h;
        bool fits0 = (rw <= BW && rh <= BH);
        bool fits1 = (rh <= BW && rw <= BH);

        int bestBoard = -1, bestX = 0, bestY = INT_MAX, bestRot = 0;
        int B = (int)boards.size();
        for (int b = 0; b < B; b++) {
            if (fits0) {
                int ix, iy;
                if (boards[b].find(rw, rh, ix, iy)) {
                    if (iy < bestY || (iy == bestY && ix < bestX)) {
                        bestY = iy; bestX = ix; bestBoard = b; bestRot = 0;
                    }
                }
            }
            if (fits1 && rw != rh) {
                int ix, iy;
                if (boards[b].find(rh, rw, ix, iy)) {
                    if (iy < bestY || (iy == bestY && ix < bestX)) {
                        bestY = iy; bestX = ix; bestBoard = b; bestRot = 1;
                    }
                }
            }
            if (firstFit && bestBoard != -1) break;
        }

        if (bestBoard == -1) {
            boards.emplace_back(BW, BH);
            bestBoard = (int)boards.size() - 1;
            bestX = 0; bestY = 0;
            bestRot = fits0 ? 0 : 1;
        }

        int pw = bestRot ? rh : rw;
        int ph = bestRot ? rw : rh;
        boards[bestBoard].place(bestX, bestY, pw, ph);
        placements[ri] = {bestBoard, bestX, bestY, bestRot};

        if ((k & 511) == 0) {
            double elapsed = chrono::duration<double>(chrono::steady_clock::now() - t0).count();
            if (elapsed > 25.0) firstFit = true;
        }
    }

    printf("%d\n", (int)boards.size());
    for (int i = 0; i < N; i++) {
        printf("%d %d %d %d\n", placements[i].board, placements[i].x, placements[i].y, placements[i].rotated);
    }
    return 0;
}
