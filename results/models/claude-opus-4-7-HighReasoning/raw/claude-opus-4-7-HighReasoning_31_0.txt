use std::io::{self, Read, Write, BufWriter};

fn parse_i64(bytes: &[u8], pos: &mut usize) -> Option<i64> {
    let n = bytes.len();
    while *pos < n && (bytes[*pos] == b' ' || bytes[*pos] == b'\n' || bytes[*pos] == b'\r' || bytes[*pos] == b'\t') {
        *pos += 1;
    }
    if *pos >= n {
        return None;
    }
    let mut neg = false;
    if bytes[*pos] == b'-' {
        neg = true;
        *pos += 1;
    } else if bytes[*pos] == b'+' {
        *pos += 1;
    }
    let mut val: i64 = 0;
    let mut any = false;
    while *pos < n {
        let c = bytes[*pos];
        if c >= b'0' && c <= b'9' {
            val = val.wrapping_mul(10).wrapping_add((c - b'0') as i64);
            *pos += 1;
            any = true;
        } else {
            break;
        }
    }
    if !any {
        return None;
    }
    Some(if neg { -val } else { val })
}

fn output_yes(indices: &[usize]) {
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "YES").ok();
    let mut s = String::new();
    s.push('[');
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            s.push(' ');
        }
        s.push_str(&idx.to_string());
    }
    s.push(']');
    writeln!(out, "{}", s).ok();
}

fn output_no() {
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "NO").ok();
}

fn brute_force(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let total: u64 = 1u64 << n;
    for mask in 0..total {
        let mut s: i64 = 0;
        for i in 0..n {
            if (mask >> i) & 1 == 1 {
                s = s.wrapping_add(nums[i]);
            }
        }
        if s == target {
            let mut indices = Vec::new();
            for i in 0..n {
                if (mask >> i) & 1 == 1 {
                    indices.push(i);
                }
            }
            return Some(indices);
        }
    }
    None
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let half = n / 2;
    let left = &nums[..half];
    let right = &nums[half..];

    let mut left_sums: Vec<(i64, u64)> = Vec::with_capacity(1usize << left.len());
    for mask in 0u64..(1u64 << left.len()) {
        let mut s: i64 = 0;
        for i in 0..left.len() {
            if (mask >> i) & 1 == 1 {
                s = s.wrapping_add(left[i]);
            }
        }
        left_sums.push((s, mask));
    }
    left_sums.sort_by_key(|x| x.0);

    for mask in 0u64..(1u64 << right.len()) {
        let mut s: i64 = 0;
        for i in 0..right.len() {
            if (mask >> i) & 1 == 1 {
                s = s.wrapping_add(right[i]);
            }
        }
        let need = target.wrapping_sub(s);
        let mut lo = 0usize;
        let mut hi = left_sums.len();
        while lo < hi {
            let mid = (lo + hi) / 2;
            if left_sums[mid].0 < need {
                lo = mid + 1;
            } else {
                hi = mid;
            }
        }
        if lo < left_sums.len() && left_sums[lo].0 == need {
            let lmask = left_sums[lo].1;
            let mut indices = Vec::new();
            for i in 0..left.len() {
                if (lmask >> i) & 1 == 1 {
                    indices.push(i);
                }
            }
            for i in 0..right.len() {
                if (mask >> i) & 1 == 1 {
                    indices.push(half + i);
                }
            }
            return Some(indices);
        }
    }
    None
}

fn bitset_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = target as usize;
    let words = t / 64 + 1;
    let mut dp: Vec<u64> = vec![0u64; words];
    dp[0] = 1;
    let mut parent: Vec<u32> = vec![u32::MAX; t + 1];

    for (idx, &num) in nums.iter().enumerate() {
        if num <= 0 || num as usize > t {
            continue;
        }
        let v = num as usize;
        let word_shift = v / 64;
        let bit_shift = v % 64;

        let mut newly_set: Vec<usize> = Vec::new();
        if bit_shift == 0 {
            for i in (word_shift..words).rev() {
                let shifted = dp[i - word_shift];
                let new_word = dp[i] | shifted;
                let diff = new_word & !dp[i];
                if diff != 0 {
                    let mut d = diff;
                    while d != 0 {
                        let b = d.trailing_zeros() as usize;
                        let s = i * 64 + b;
                        if s <= t {
                            newly_set.push(s);
                        }
                        d &= d - 1;
                    }
                }
                dp[i] = new_word;
            }
        } else {
            for i in (word_shift..words).rev() {
                let lo = dp[i - word_shift] << bit_shift;
                let hi = if i > word_shift {
                    dp[i - word_shift - 1] >> (64 - bit_shift)
                } else {
                    0
                };
                let shifted = lo | hi;
                let new_word = dp[i] | shifted;
                let diff = new_word & !dp[i];
                if diff != 0 {
                    let mut d = diff;
                    while d != 0 {
                        let b = d.trailing_zeros() as usize;
                        let s = i * 64 + b;
                        if s <= t {
                            newly_set.push(s);
                        }
                        d &= d - 1;
                    }
                }
                dp[i] = new_word;
            }
        }

        for s in newly_set {
            if s <= t && parent[s] == u32::MAX {
                parent[s] = idx as u32;
            }
        }

        if (dp[t / 64] >> (t % 64)) & 1 == 1 {
            let mut indices = Vec::new();
            let mut cur = t;
            while cur > 0 {
                let p = parent[cur];
                if p == u32::MAX {
                    return None;
                }
                let pu = p as usize;
                indices.push(pu);
                let v = nums[pu] as usize;
                if v > cur {
                    return None;
                }
                cur -= v;
            }
            indices.sort();
            return Some(indices);
        }
    }
    None
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).ok();
    let mut pos = 0usize;

    let n = match parse_i64(&input, &mut pos) {
        Some(v) => v as usize,
        None => {
            output_no();
            return;
        }
    };
    let target = match parse_i64(&input, &mut pos) {
        Some(v) => v,
        None => {
            output_no();
            return;
        }
    };

    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for _ in 0..n {
        match parse_i64(&input, &mut pos) {
            Some(v) => nums.push(v),
            None => break,
        }
    }

    if target == 0 {
        output_yes(&[]);
        return;
    }
    if nums.is_empty() {
        output_no();
        return;
    }

    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            output_yes(&[i]);
            return;
        }
    }

    let total: i64 = nums.iter().fold(0i64, |a, b| a.wrapping_add(*b));
    if total == target {
        let indices: Vec<usize> = (0..nums.len()).collect();
        output_yes(&indices);
        return;
    }

    let nn = nums.len();

    if nn <= 22 {
        if let Some(idx) = brute_force(&nums, target) {
            output_yes(&idx);
            return;
        }
        output_no();
        return;
    }

    if nn <= 38 {
        if let Some(idx) = meet_in_middle(&nums, target) {
            output_yes(&idx);
            return;
        }
        output_no();
        return;
    }

    let all_non_neg = nums.iter().all(|&x| x >= 0);
    if all_non_neg && target > 0 {
        let t = target as u64;
        if t <= 200_000_000 {
            if let Some(idx) = bitset_dp(&nums, target) {
                output_yes(&idx);
                return;
            }
            output_no();
            return;
        }
    }

    if all_non_neg {
        let mut indexed: Vec<(i64, usize)> = nums.iter().copied().enumerate().map(|(i, v)| (v, i)).collect();
        indexed.sort_by(|a, b| b.0.cmp(&a.0));
        let mut remaining = target;
        let mut chosen = Vec::new();
        for &(v, i) in &indexed {
            if v > 0 && v <= remaining {
                remaining -= v;
                chosen.push(i);
                if remaining == 0 {
                    chosen.sort();
                    output_yes(&chosen);
                    return;
                }
            }
        }
    }

    let mut state: u64 = 0x9E3779B97F4A7C15;
    let mut rng = || {
        state ^= state << 13;
        state ^= state >> 7;
        state ^= state << 17;
        state
    };

    let best_attempts = if nn > 100_000 { 5_000 } else { 50_000 };
    for _ in 0..best_attempts {
        let r = rng();
        let take = ((r as usize) % nn.min(64)).max(1);
        let mut sum: i64 = 0;
        let mut chosen: Vec<usize> = Vec::with_capacity(take);
        let mut seen = std::collections::HashSet::new();
        for _ in 0..take {
            let r2 = rng();
            let idx = (r2 as usize) % nn;
            if !seen.insert(idx) {
                continue;
            }
            sum = sum.wrapping_add(nums[idx]);
            chosen.push(idx);
            if sum == target {
                chosen.sort();
                output_yes(&chosen);
                return;
            }
        }
    }

    output_no();
}
