using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Text;
using System.Threading;

class Program
{
    static int N;
    static int[][] outAdj, inAdj;
    static bool[] alive;
    static int bestSize;
    static List<int> bestSet;
    static List<int> forced;

    static void Main()
    {
        var t = new Thread(Run, 128 * 1024 * 1024);
        t.Start();
        t.Join();
    }

    static void Run()
    {
        var sr = new StreamReader(Console.OpenStandardInput());
        var all = sr.ReadToEnd().Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        N = int.Parse(all[idx++]);
        int m = int.Parse(all[idx++]);
        var outL = new List<int>[N];
        var inL = new List<int>[N];
        for (int i = 0; i < N; i++) { outL[i] = new List<int>(); inL[i] = new List<int>(); }
        forced = new List<int>();
        var forcedSet = new bool[N];
        var edgeSet = new HashSet<long>();
        for (int i = 0; i < m; i++)
        {
            int u = int.Parse(all[idx++]);
            int v = int.Parse(all[idx++]);
            if (u == v)
            {
                if (!forcedSet[u]) { forcedSet[u] = true; forced.Add(u); }
                continue;
            }
            long key = (long)u * N + v;
            if (edgeSet.Add(key))
            {
                outL[u].Add(v);
                inL[v].Add(u);
            }
        }
        outAdj = new int[N][];
        inAdj = new int[N][];
        for (int i = 0; i < N; i++) { outAdj[i] = outL[i].ToArray(); inAdj[i] = inL[i].ToArray(); }

        alive = new bool[N];
        for (int i = 0; i < N; i++) alive[i] = !forcedSet[i];

        ReduceAll();

        var components = GetSCCs();
        var result = new List<int>(forced);
        foreach (var comp in components)
        {
            if (comp.Count < 2) continue;
            var saveAlive = (bool[])alive.Clone();
            for (int i = 0; i < N; i++)
                if (alive[i] && !comp.Contains(i)) alive[i] = false;

            var heur = GreedyFVS(comp);
            bestSize = heur.Count;
            bestSet = heur;

            var partial = new List<int>();
            BnB(partial);

            result.AddRange(bestSet);
            alive = saveAlive;
            foreach (var v in bestSet) alive[v] = false;
        }

        var sb = new StringBuilder();
        sb.Append(result.Count).Append('\n');
        for (int i = 0; i < result.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(result[i]);
        }
        Console.Out.Write(sb.ToString());
    }

    static void ReduceAll()
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int v = 0; v < N; v++)
            {
                if (!alive[v]) continue;
                bool hi = false, ho = false;
                foreach (var w in inAdj[v]) if (alive[w]) { hi = true; break; }
                if (!hi) { alive[v] = false; changed = true; continue; }
                foreach (var w in outAdj[v]) if (alive[w]) { ho = true; break; }
                if (!ho) { alive[v] = false; changed = true; }
            }
        }
    }

    static List<HashSet<int>> GetSCCs()
    {
        var index = new int[N];
        var lowlink = new int[N];
        var onStack = new bool[N];
        for (int i = 0; i < N; i++) index[i] = -1;
        int idx = 0;
        var sccStack = new Stack<int>();
        var result = new List<HashSet<int>>();

        for (int start = 0; start < N; start++)
        {
            if (!alive[start] || index[start] != -1) continue;
            var work = new Stack<(int v, int i)>();
            index[start] = idx; lowlink[start] = idx; idx++;
            sccStack.Push(start); onStack[start] = true;
            work.Push((start, 0));

            while (work.Count > 0)
            {
                var (v, i) = work.Pop();
                bool descended = false;
                while (i < outAdj[v].Length)
                {
                    int w = outAdj[v][i];
                    i++;
                    if (!alive[w]) continue;
                    if (index[w] == -1)
                    {
                        index[w] = idx; lowlink[w] = idx; idx++;
                        sccStack.Push(w); onStack[w] = true;
                        work.Push((v, i));
                        work.Push((w, 0));
                        descended = true;
                        break;
                    }
                    else if (onStack[w])
                    {
                        if (index[w] < lowlink[v]) lowlink[v] = index[w];
                    }
                }
                if (descended) continue;
                if (lowlink[v] == index[v])
                {
                    var comp = new HashSet<int>();
                    int w;
                    do
                    {
                        w = sccStack.Pop();
                        onStack[w] = false;
                        comp.Add(w);
                    } while (w != v);
                    result.Add(comp);
                }
                if (work.Count > 0)
                {
                    var top = work.Pop();
                    int pv = top.v;
                    if (lowlink[v] < lowlink[pv]) lowlink[pv] = lowlink[v];
                    work.Push(top);
                }
            }
        }
        return result;
    }

    static List<int> GreedyFVS(HashSet<int> comp)
    {
        var la = new bool[N];
        foreach (var v in comp) la[v] = true;
        var removed = new List<int>();
        while (true)
        {
            ReduceLocal(la);
            var cyc = FindCycle(la);
            if (cyc == null) break;
            int best = cyc[0]; long bestScore = -1;
            foreach (var v in cyc)
            {
                int ic = 0, oc = 0;
                foreach (var w in inAdj[v]) if (la[w]) ic++;
                foreach (var w in outAdj[v]) if (la[w]) oc++;
                long sc = (long)ic * oc;
                if (sc > bestScore) { bestScore = sc; best = v; }
            }
            la[best] = false;
            removed.Add(best);
        }
        return removed;
    }

    static void ReduceLocal(bool[] la)
    {
        bool ch = true;
        while (ch)
        {
            ch = false;
            for (int v = 0; v < N; v++)
            {
                if (!la[v]) continue;
                bool hi = false, ho = false;
                foreach (var w in inAdj[v]) if (la[w]) { hi = true; break; }
                if (!hi) { la[v] = false; ch = true; continue; }
                foreach (var w in outAdj[v]) if (la[w]) { ho = true; break; }
                if (!ho) { la[v] = false; ch = true; }
            }
        }
    }

    static int[] cycState;
    static int[] cycParent;

    static List<int> FindCycle(bool[] la)
    {
        cycState = new int[N];
        cycParent = new int[N];
        for (int s = 0; s < N; s++)
        {
            if (!la[s] || cycState[s] != 0) continue;
            cycParent[s] = -1;
            var c = IterDFS(s, la);
            if (c != null) return c;
        }
        return null;
    }

    static List<int> IterDFS(int start, bool[] la)
    {
        var stk = new Stack<(int v, int i)>();
        stk.Push((start, 0));
        cycState[start] = 1;
        while (stk.Count > 0)
        {
            var (v, i) = stk.Pop();
            bool descended = false;
            while (i < outAdj[v].Length)
            {
                int w = outAdj[v][i];
                i++;
                if (!la[w]) continue;
                if (cycState[w] == 0)
                {
                    cycParent[w] = v;
                    cycState[w] = 1;
                    stk.Push((v, i));
                    stk.Push((w, 0));
                    descended = true;
                    break;
                }
                else if (cycState[w] == 1)
                {
                    var c = new List<int> { w };
                    int x = v;
                    while (x != w) { c.Add(x); x = cycParent[x]; }
                    return c;
                }
            }
            if (descended) continue;
            cycState[v] = 2;
        }
        return null;
    }

    static void BnB(List<int> partial)
    {
        if (partial.Count >= bestSize) return;

        var disabled = new List<int>();
        bool ch = true;
        while (ch)
        {
            ch = false;
            for (int v = 0; v < N; v++)
            {
                if (!alive[v]) continue;
                bool hi = false, ho = false;
                foreach (var w in inAdj[v]) if (alive[w]) { hi = true; break; }
                if (!hi) { alive[v] = false; disabled.Add(v); ch = true; continue; }
                foreach (var w in outAdj[v]) if (alive[w]) { ho = true; break; }
                if (!ho) { alive[v] = false; disabled.Add(v); ch = true; }
            }
        }

        var cyc = FindCycle(alive);
        if (cyc == null)
        {
            if (partial.Count < bestSize)
            {
                bestSize = partial.Count;
                bestSet = new List<int>(partial);
            }
        }
        else
        {
            cyc.Sort((a, b) =>
            {
                int da = 0, db = 0;
                foreach (var w in inAdj[a]) if (alive[w]) da++;
                foreach (var w in outAdj[a]) if (alive[w]) da++;
                foreach (var w in inAdj[b]) if (alive[w]) db++;
                foreach (var w in outAdj[b]) if (alive[w]) db++;
                return db.CompareTo(da);
            });
            foreach (var v in cyc)
            {
                if (partial.Count + 1 >= bestSize) continue;
                alive[v] = false;
                partial.Add(v);
                BnB(partial);
                partial.RemoveAt(partial.Count - 1);
                alive[v] = true;
            }
        }

        foreach (var v in disabled) alive[v] = true;
    }
}
