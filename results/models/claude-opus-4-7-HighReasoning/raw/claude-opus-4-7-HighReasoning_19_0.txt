using System;
using System.IO;
using System.Collections.Generic;
using System.Text;
using System.Globalization;

class MiningSolver
{
    static int XC, YC, ZC;
    static double BX, BY, BZ;
    static int[,] topZ;

    static void Main()
    {
        var bin = new BufferedStream(Console.OpenStandardInput(), 1 << 20);
        var sb = new StringBuilder();
        int c;
        while ((c = bin.ReadByte()) != -1 && c != '\n')
        {
            if (c != '\r') sb.Append((char)c);
        }
        var parts = sb.ToString().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        XC = int.Parse(parts[0]);
        YC = int.Parse(parts[1]);
        ZC = int.Parse(parts[2]);
        BX = double.Parse(parts[3], CultureInfo.InvariantCulture);
        BY = double.Parse(parts[4], CultureInfo.InvariantCulture);
        BZ = double.Parse(parts[5], CultureInfo.InvariantCulture);

        topZ = new int[XC, YC];
        for (int x = 0; x < XC; x++)
            for (int y = 0; y < YC; y++)
                topZ[x, y] = -1;

        // Read voxels streamed; collect surface-layer ore candidates only.
        // Z-resolution is 10x finer than X/Y. So one X/Y step = 10 Z voxels for slope-1.
        // Slope rule "1 Z voxel per X/Y voxel" - we interpret literally per problem text.
        var candidates = new List<(int x, int y, int z, int mass, int val)>();
        byte[] buf = new byte[ZC];
        long totalVoxels = (long)XC * YC * ZC;
        bool storeAll = totalVoxels <= 20_000_000L;
        byte[,,] allVox = null;
        if (storeAll) allVox = new byte[XC, YC, ZC];

        for (int x = 0; x < XC; x++)
        {
            for (int y = 0; y < YC; y++)
            {
                int read = 0;
                while (read < ZC)
                {
                    int n = bin.Read(buf, read, ZC - read);
                    if (n <= 0) break;
                    read += n;
                }
                int topFound = -1;
                for (int z = 0; z < ZC; z++)
                {
                    byte d = buf[z];
                    if (storeAll) allVox[x, y, z] = d;
                    if (d != 0) topFound = z;
                }
                topZ[x, y] = topFound;
                int zMin = storeAll ? 0 : Math.Max(0, topFound - 5);
                for (int z = zMin; z <= topFound; z++)
                {
                    byte d = buf[z];
                    int val = d >> 4;
                    int mass = d & 15;
                    if (val > 0 && mass > 0)
                        candidates.Add((x, y, z, mass, val));
                }
            }
        }

        // Sort: top-down (highest z first), then highest value, then nearest to (0,0).
        candidates.Sort((a, b) =>
        {
            int cmp = b.z.CompareTo(a.z);
            if (cmp != 0) return cmp;
            cmp = b.val.CompareTo(a.val);
            if (cmp != 0) return cmp;
            int da = a.x + a.y, db = b.x + b.y;
            return da.CompareTo(db);
        });

        var ops = new List<string>();
        foreach (var cand in candidates)
        {
            int cx = cand.x, cy = cand.y, cz = cand.z;
            if (topZ[cx, cy] != cz) continue;

            // After digging, stand height at (cx,cy) is cz-1.
            int standZ = cz - 1;
            int curX = cx, curY = cy;
            var steps = new List<(int x, int y)>();
            bool ok = true;
            while (curX != 0 || curY != 0)
            {
                int nx = curX, ny = curY;
                if (curX != 0) nx = curX - Math.Sign(curX);
                else ny = curY - Math.Sign(curY);
                int nz = topZ[nx, ny];
                if (Math.Abs(nz - standZ) > 1) { ok = false; break; }
                steps.Add((nx, ny));
                curX = nx; curY = ny; standZ = nz;
            }
            if (!ok) continue;

            var sbOp = new StringBuilder();
            sbOp.Append(steps.Count);
            sbOp.Append(' ').Append(cx).Append(' ').Append(cy).Append(' ').Append(cz);
            foreach (var s in steps)
                sbOp.Append(' ').Append(s.x).Append(' ').Append(s.y);
            sbOp.Append(' ').Append(0).Append(' ').Append(0);
            ops.Add(sbOp.ToString());

            topZ[cx, cy] = cz - 1;
        }

        var outStream = new BufferedStream(Console.OpenStandardOutput(), 1 << 20);
        var w = new StreamWriter(outStream);
        w.WriteLine(ops.Count);
        foreach (var op in ops) w.WriteLine(op);
        w.Flush();
    }
}
