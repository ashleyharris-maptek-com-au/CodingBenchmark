using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class MinesweeperSolver
{
    static int H, W;
    static byte[] grid;
    const byte UNK = 9;
    const byte MINE = 10;

    static int Idx(int x, int y) => y * W + x;

    static void Main()
    {
        using var sin = new StreamReader(Console.OpenStandardInput(), Encoding.ASCII, false, 1 << 16);
        string first = sin.ReadLine();
        if (first == null) { Console.WriteLine(0); return; }
        var parts = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        grid = new byte[H * W];
        for (int y = 0; y < H; y++)
        {
            string line = sin.ReadLine() ?? "";
            if (line.Length < W) line = line.PadRight(W, ' ');
            for (int x = 0; x < W; x++)
            {
                char c = line[x];
                if (c == ' ') grid[Idx(x, y)] = UNK;
                else if (c == '*') grid[Idx(x, y)] = MINE;
                else if (c >= '0' && c <= '8') grid[Idx(x, y)] = (byte)(c - '0');
                else grid[Idx(x, y)] = UNK;
            }
        }

        byte[] safe = new byte[H * W];
        int N = H * W;

        int[] dxs = { -1, 0, 1, -1, 1, -1, 0, 1 };
        int[] dys = { -1, -1, -1, 0, 0, 1, 1, 1 };

        // Phase 1: trivial constraint propagation to fixed point.
        var queue = new Queue<int>();
        var inQueue = new bool[N];
        for (int i = 0; i < N; i++)
            if (grid[i] <= 8) { queue.Enqueue(i); inQueue[i] = true; }

        while (queue.Count > 0)
        {
            int idx = queue.Dequeue();
            inQueue[idx] = false;
            byte v = grid[idx];
            if (v > 8) continue;
            int cx = idx % W;
            int cy = idx / W;
            int mines = 0, unk = 0;
            Span<int> ubuf = stackalloc int[8];
            for (int k = 0; k < 8; k++)
            {
                int nx = cx + dxs[k]; int ny = cy + dys[k];
                if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                byte g = grid[Idx(nx, ny)];
                if (g == MINE) mines++;
                else if (g == UNK) { ubuf[unk++] = Idx(nx, ny); }
            }
            int remaining = v - mines;
            if (remaining < 0 || unk == 0) continue;
            if (remaining == 0)
            {
                for (int k = 0; k < unk; k++)
                {
                    int u = ubuf[k];
                    if (safe[u] == 0) safe[u] = 1;
                }
            }
            else if (remaining == unk)
            {
                for (int k = 0; k < unk; k++)
                {
                    int u = ubuf[k];
                    grid[u] = MINE;
                    int ux = u % W; int uy = u / W;
                    for (int kk = 0; kk < 8; kk++)
                    {
                        int nx = ux + dxs[kk]; int ny = uy + dys[kk];
                        if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                        int ni = Idx(nx, ny);
                        if (grid[ni] <= 8 && !inQueue[ni])
                        {
                            inQueue[ni] = true;
                            queue.Enqueue(ni);
                        }
                    }
                }
            }
        }

        // Phase 2: subset deduction (pairwise). Adaptive: skip on extremely large boards
        // or restart phase 1 whenever phase 2 finds new mines.
        long bigThreshold = 2_000_000L;
        int maxRounds = 6;

        var consVars = new List<int[]>();
        var consCount = new List<int>();
        var cellToCons = new Dictionary<int, List<int>>();

        void Recompute()
        {
            consVars.Clear(); consCount.Clear(); cellToCons.Clear();
            for (int i = 0; i < N; i++)
            {
                if (grid[i] > 8) continue;
                int cx = i % W, cy = i / W;
                int mines = 0;
                var ulist = new List<int>(8);
                for (int k = 0; k < 8; k++)
                {
                    int nx = cx + dxs[k]; int ny = cy + dys[k];
                    if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                    int ni = Idx(nx, ny);
                    byte g = grid[ni];
                    if (g == MINE) mines++;
                    else if (g == UNK && safe[ni] == 0) ulist.Add(ni);
                }
                if (ulist.Count == 0) continue;
                int rem = grid[i] - mines;
                if (rem < 0) continue;
                ulist.Sort();
                int cid = consVars.Count;
                consVars.Add(ulist.ToArray());
                consCount.Add(rem);
                foreach (var u in ulist)
                {
                    if (!cellToCons.TryGetValue(u, out var lst))
                    {
                        lst = new List<int>();
                        cellToCons[u] = lst;
                    }
                    lst.Add(cid);
                }
            }
        }

        for (int round = 0; round < maxRounds; round++)
        {
            Recompute();
            if (consVars.Count == 0) break;
            if (consVars.Count > bigThreshold) break;

            bool progress = false;

            // Apply trivial again on the current (possibly reduced) constraints.
            for (int i = 0; i < consVars.Count; i++)
            {
                var V = consVars[i]; int c = consCount[i];
                if (c == 0)
                {
                    foreach (var u in V)
                        if (grid[u] == UNK && safe[u] == 0) { safe[u] = 1; progress = true; }
                }
                else if (c == V.Length)
                {
                    foreach (var u in V)
                        if (grid[u] == UNK) { grid[u] = MINE; progress = true; }
                }
            }
            if (progress) continue;

            int cc = consVars.Count;
            for (int a = 0; a < cc; a++)
            {
                var A = consVars[a];
                int ac = consCount[a];
                if (A.Length == 0) continue;
                var candidates = new HashSet<int>();
                foreach (var v in A)
                {
                    if (cellToCons.TryGetValue(v, out var lst))
                        foreach (var bb in lst)
                            if (bb != a) candidates.Add(bb);
                }
                foreach (var b in candidates)
                {
                    var B = consVars[b];
                    int bc = consCount[b];
                    if (A.Length > B.Length) continue;
                    int ia = 0, ib = 0;
                    bool subset = true;
                    var diffList = new List<int>(B.Length);
                    while (ib < B.Length)
                    {
                        if (ia < A.Length && A[ia] == B[ib]) { ia++; ib++; }
                        else if (ia < A.Length && A[ia] < B[ib]) { subset = false; break; }
                        else { diffList.Add(B[ib]); ib++; }
                    }
                    if (!subset || ia < A.Length) continue;
                    int newCount = bc - ac;
                    if (newCount < 0 || newCount > diffList.Count) continue;
                    if (newCount == 0)
                    {
                        foreach (var u in diffList)
                            if (grid[u] == UNK && safe[u] == 0) { safe[u] = 1; progress = true; }
                    }
                    else if (newCount == diffList.Count)
                    {
                        foreach (var u in diffList)
                            if (grid[u] == UNK) { grid[u] = MINE; progress = true; }
                    }
                }
            }
            if (!progress) break;
        }

        // Phase 3: small connected-component exact CSP enumeration.
        // Only on boards where total unknown frontier is manageable; per-component cap on size.
        // Build frontier graph: unknown cells adjacent to a number, linked through shared numbered cells.
        try
        {
            Recompute();
            if (consVars.Count > 0 && consVars.Count <= 200_000)
            {
                // Map of unknown frontier cells to id
                var cellId = new Dictionary<int, int>();
                foreach (var V in consVars)
                    foreach (var u in V)
                        if (!cellId.ContainsKey(u)) cellId[u] = cellId.Count;
                int U = cellId.Count;
                if (U > 0 && U <= 50_000)
                {
                    // Convert constraints to int[] of cell ids
                    int M = consVars.Count;
                    int[][] cV = new int[M][];
                    int[] cC = new int[M];
                    for (int i = 0; i < M; i++)
                    {
                        var V = consVars[i];
                        var arr = new int[V.Length];
                        for (int j = 0; j < V.Length; j++) arr[j] = cellId[V[j]];
                        cV[i] = arr;
                        cC[i] = consCount[i];
                    }
                    // cell -> constraints
                    var cellCons = new List<int>[U];
                    for (int i = 0; i < U; i++) cellCons[i] = new List<int>();
                    for (int i = 0; i < M; i++)
                        foreach (var v in cV[i]) cellCons[v].Add(i);

                    // Union-find on cells (link cells sharing a constraint)
                    int[] uf = new int[U];
                    for (int i = 0; i < U; i++) uf[i] = i;
                    int Find(int x) { while (uf[x] != x) { uf[x] = uf[uf[x]]; x = uf[x]; } return x; }
                    void Union(int a, int b) { int ra = Find(a), rb = Find(b); if (ra != rb) uf[ra] = rb; }
                    for (int i = 0; i < M; i++)
                    {
                        var V = cV[i];
                        for (int j = 1; j < V.Length; j++) Union(V[0], V[j]);
                    }
                    var compCells = new Dictionary<int, List<int>>();
                    for (int i = 0; i < U; i++)
                    {
                        int r = Find(i);
                        if (!compCells.TryGetValue(r, out var lst)) { lst = new List<int>(); compCells[r] = lst; }
                        lst.Add(i);
                    }
                    // Reverse: cell -> original idx
                    int[] cellIdToIdx = new int[U];
                    foreach (var kv in cellId) cellIdToIdx[kv.Value] = kv.Key;

                    foreach (var kv in compCells)
                    {
                        var cells = kv.Value;
                        if (cells.Count > 24) continue; // cap CSP size
                        // collect constraints touching these cells
                        var consSet = new HashSet<int>();
                        foreach (var c in cells) foreach (var ci in cellCons[c]) consSet.Add(ci);
                        var consList = new List<int>(consSet);
                        // Map cell id -> local bit index
                        var localIdx = new Dictionary<int, int>();
                        for (int i = 0; i < cells.Count; i++) localIdx[cells[i]] = i;
                        // Prepare local constraint masks/counts
                        int CC = consList.Count;
                        ulong[] cmask = new ulong[CC];
                        int[] cval = new int[CC];
                        for (int i = 0; i < CC; i++)
                        {
                            var V = cV[consList[i]];
                            ulong m = 0;
                            foreach (var v in V) m |= 1UL << localIdx[v];
                            cmask[i] = m;
                            cval[i] = cC[consList[i]];
                        }
                        int n = cells.Count;
                        // Enumerate all 2^n assignments and count which cells must be 0 (safe) across solutions.
                        // We'll track: count of solutions; for each cell, count of solutions where bit=1.
                        long totalSolutions = 0;
                        int[] oneCount = new int[n];
                        ulong end = 1UL << n;
                        for (ulong a = 0; a < end; a++)
                        {
                            bool ok = true;
                            for (int i = 0; i < CC; i++)
                            {
                                int p = System.Numerics.BitOperations.PopCount(a & cmask[i]);
                                if (p != cval[i]) { ok = false; break; }
                            }
                            if (!ok) continue;
                            totalSolutions++;
                            ulong t = a;
                            while (t != 0)
                            {
                                int b = System.Numerics.BitOperations.TrailingZeroCount(t);
                                oneCount[b]++;
                                t &= t - 1;
                            }
                        }
                        if (totalSolutions == 0) continue;
                        for (int i = 0; i < n; i++)
                        {
                            int origCell = cellIdToIdx[cells[i]];
                            if (oneCount[i] == 0)
                            {
                                if (grid[origCell] == UNK && safe[origCell] == 0) safe[origCell] = 1;
                            }
                            else if (oneCount[i] == totalSolutions)
                            {
                                if (grid[origCell] == UNK) grid[origCell] = MINE;
                            }
                        }
                    }
                }
            }
        }
        catch { /* CSP is opportunistic; ignore failures */ }

        // Output
        var lines = new List<string>();
        for (int y = 0; y < H; y++)
        {
            for (int x = 0; x < W; x++)
            {
                int i = Idx(x, y);
                if (safe[i] == 1 && grid[i] == UNK)
                    lines.Add(x.ToString() + " " + y.ToString());
            }
        }
        var outW = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 16);
        outW.WriteLine(lines.Count);
        foreach (var ln in lines) outW.WriteLine(ln);
        outW.Flush();
    }
}
