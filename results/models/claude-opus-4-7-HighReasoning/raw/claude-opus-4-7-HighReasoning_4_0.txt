#include <cstdio>
#include <cstdlib>
#include <cmath>
#include <vector>
#include <algorithm>
#include <limits>
#include <chrono>

#ifndef M_PI
#define M_PI 3.14159265358979323846
#endif

// Regular tetrahedron packing into a polyhedral container.
// Strategy: compute container AABB, shrink by circumradius R so any tetra centered
// inside the shrunken box has all 4 vertices inside the AABB. Place centers on a
// cubic grid with spacing 2R + epsilon so circumscribed spheres do not overlap,
// guaranteeing no tetrahedron overlap regardless of orientation.
// This adapts from trivial to huge containers, with a hard cap on count and a
// wall-clock time guard.

int main() {
    int V;
    double e;
    if (scanf("%d %lf", &V, &e) != 2) return 0;
    if (e <= 0.0) { printf("0\n"); return 0; }

    double minx =  std::numeric_limits<double>::infinity();
    double miny =  std::numeric_limits<double>::infinity();
    double minz =  std::numeric_limits<double>::infinity();
    double maxx = -std::numeric_limits<double>::infinity();
    double maxy = -std::numeric_limits<double>::infinity();
    double maxz = -std::numeric_limits<double>::infinity();

    for (int i = 0; i < V; ++i) {
        double x, y, z;
        if (scanf("%lf %lf %lf", &x, &y, &z) != 3) return 0;
        if (x < minx) minx = x; if (x > maxx) maxx = x;
        if (y < miny) miny = y; if (y > maxy) maxy = y;
        if (z < minz) minz = z; if (z > maxz) maxz = z;
    }

    double R = e * std::sqrt(3.0 / 8.0);   // circumradius

    double lx = minx + R, hx = maxx - R;
    double ly = miny + R, hy = maxy - R;
    double lz = minz + R, hz = maxz - R;

    if (lx > hx || ly > hy || lz > hz) {
        printf("0\n");
        return 0;
    }

    double spacing = 2.0 * R + 1e-9;

    long long nx = (long long)std::floor((hx - lx) / spacing) + 1;
    long long ny = (long long)std::floor((hy - ly) / spacing) + 1;
    long long nz = (long long)std::floor((hz - lz) / spacing) + 1;
    if (nx < 1) nx = 1; if (ny < 1) ny = 1; if (nz < 1) nz = 1;

    const long long MAX_COUNT = 2'000'000LL;
    long long total = nx * ny * nz;
    if (total > MAX_COUNT) {
        double scale = std::cbrt((double)total / (double)MAX_COUNT);
        spacing *= scale;
        nx = (long long)std::floor((hx - lx) / spacing) + 1;
        ny = (long long)std::floor((hy - ly) / spacing) + 1;
        nz = (long long)std::floor((hz - lz) / spacing) + 1;
        if (nx < 1) nx = 1; if (ny < 1) ny = 1; if (nz < 1) nz = 1;
    }

    auto t0 = std::chrono::steady_clock::now();
    const double TIME_LIMIT = 25.0;

    std::vector<char> buf;
    buf.reserve(1 << 20);
    auto append_double = [&](double v) {
        char tmp[64];
        int n = std::snprintf(tmp, sizeof(tmp), "%.6f", v);
        buf.insert(buf.end(), tmp, tmp + n);
    };

    long long K = 0;
    bool stop = false;
    for (long long iz = 0; iz < nz && !stop; ++iz) {
        double cz = lz + iz * spacing;
        if (cz > hz + 1e-12) break;
        for (long long iy = 0; iy < ny && !stop; ++iy) {
            double cy = ly + iy * spacing;
            if (cy > hy + 1e-12) break;
            for (long long ix = 0; ix < nx; ++ix) {
                double cx = lx + ix * spacing;
                if (cx > hx + 1e-12) break;

                double rx = 0.0, ry = 0.0, rz = 0.0;
                if (((ix + iy + iz) & 1) != 0) rx = M_PI;

                append_double(cx); buf.push_back(' ');
                append_double(cy); buf.push_back(' ');
                append_double(cz); buf.push_back(' ');
                append_double(rx); buf.push_back(' ');
                append_double(ry); buf.push_back(' ');
                append_double(rz); buf.push_back('\n');
                ++K;

                if ((K & 0xFFFF) == 0) {
                    auto t = std::chrono::steady_clock::now();
                    double el = std::chrono::duration<double>(t - t0).count();
                    if (el > TIME_LIMIT) { stop = true; break; }
                }
            }
        }
    }

    printf("%lld\n", K);
    if (!buf.empty()) {
        fwrite(buf.data(), 1, buf.size(), stdout);
    }
    return 0;
}
