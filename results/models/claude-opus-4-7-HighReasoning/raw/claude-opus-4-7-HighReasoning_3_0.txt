use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut it = input.split_ascii_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u: u32 = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        let v: u32 = match it.next() { Some(s) => s.parse().unwrap_or(0), None => break };
        if (u as usize) >= n || (v as usize) >= n || u == v {
            continue;
        }
        edges.push((u, v));
    }

    // Random init
    let mut rng_state: u64 = 0x9E3779B97F4A7C15;
    let mut next_rand = || -> f64 {
        rng_state = rng_state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        ((rng_state >> 33) as u32 as f64) / (u32::MAX as f64)
    };

    let mut pos: Vec<(f64, f64)> = vec![(0.0, 0.0); n];
    let scale = (n as f64).sqrt();
    for i in 0..n {
        let a = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64).max(1.0);
        let r = scale * (0.3 + 0.7 * next_rand());
        pos[i] = (r * a.cos() + (next_rand() - 0.5) * 0.5, r * a.sin() + (next_rand() - 0.5) * 0.5);
    }

    let iterations: usize = if n <= 100 {
        400
    } else if n <= 1000 {
        250
    } else if n <= 10000 {
        120
    } else if n <= 50000 {
        60
    } else {
        35
    };

    let area = (n as f64) * 10.0;
    let k = (area / (n as f64).max(1.0)).sqrt();
    let k2 = k * k;

    let mut t = scale * 0.5;
    let cool = t / (iterations as f64 + 1.0);

    if n <= 2000 {
        // Exact O(n^2) Fruchterman-Reingold
        for _iter in 0..iterations {
            let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

            for i in 0..n {
                let (xi, yi) = pos[i];
                for j in (i + 1)..n {
                    let (xj, yj) = pos[j];
                    let dx = xi - xj;
                    let dy = yi - yj;
                    let mut d2 = dx * dx + dy * dy;
                    if d2 < 1e-6 { d2 = 1e-6; }
                    let d = d2.sqrt();
                    let f = k2 / d;
                    let fx = dx / d * f;
                    let fy = dy / d * f;
                    disp[i].0 += fx;
                    disp[i].1 += fy;
                    disp[j].0 -= fx;
                    disp[j].1 -= fy;
                }
            }

            for &(u, v) in &edges {
                let (xu, yu) = pos[u as usize];
                let (xv, yv) = pos[v as usize];
                let dx = xu - xv;
                let dy = yu - yv;
                let mut d = (dx * dx + dy * dy).sqrt();
                if d < 1e-6 { d = 1e-6; }
                let f = d * d / k;
                let fx = dx / d * f;
                let fy = dy / d * f;
                disp[u as usize].0 -= fx;
                disp[u as usize].1 -= fy;
                disp[v as usize].0 += fx;
                disp[v as usize].1 += fy;
            }

            for i in 0..n {
                let (dx, dy) = disp[i];
                let d = (dx * dx + dy * dy).sqrt();
                if d > 1e-9 {
                    let lim = d.min(t);
                    pos[i].0 += dx / d * lim;
                    pos[i].1 += dy / d * lim;
                }
            }

            t = (t - cool).max(cool * 0.5);
        }
    } else {
        // Large graph: grid-based approximate repulsion
        for _iter in 0..iterations {
            let mut minx = f64::INFINITY;
            let mut miny = f64::INFINITY;
            let mut maxx = f64::NEG_INFINITY;
            let mut maxy = f64::NEG_INFINITY;
            for i in 0..n {
                let (x, y) = pos[i];
                if x < minx { minx = x; }
                if y < miny { miny = y; }
                if x > maxx { maxx = x; }
                if y > maxy { maxy = y; }
            }
            let bw = (maxx - minx).max(1.0);
            let bh = (maxy - miny).max(1.0);
            let cell = k * 2.0;
            let gw = (((bw / cell).ceil() as usize).max(1)).min(2000);
            let gh = (((bh / cell).ceil() as usize).max(1)).min(2000);
            let cw = bw / (gw as f64);
            let ch = bh / (gh as f64);

            let mut grid: Vec<Vec<u32>> = vec![Vec::new(); gw * gh];
            for i in 0..n {
                let (x, y) = pos[i];
                let gx = (((x - minx) / cw) as isize).max(0).min(gw as isize - 1) as usize;
                let gy = (((y - miny) / ch) as isize).max(0).min(gh as isize - 1) as usize;
                grid[gy * gw + gx].push(i as u32);
            }

            let mut disp: Vec<(f64, f64)> = vec![(0.0, 0.0); n];

            for i in 0..n {
                let (xi, yi) = pos[i];
                let gx = (((xi - minx) / cw) as isize).max(0).min(gw as isize - 1) as usize;
                let gy = (((yi - miny) / ch) as isize).max(0).min(gh as isize - 1) as usize;
                for dyc in -1i32..=1 {
                    for dxc in -1i32..=1 {
                        let nx = gx as i32 + dxc;
                        let ny = gy as i32 + dyc;
                        if nx < 0 || ny < 0 || nx >= gw as i32 || ny >= gh as i32 {
                            continue;
                        }
                        let cell_idx = (ny as usize) * gw + (nx as usize);
                        for &j32 in &grid[cell_idx] {
                            let j = j32 as usize;
                            if j == i { continue; }
                            let (xj, yj) = pos[j];
                            let ddx = xi - xj;
                            let ddy = yi - yj;
                            let mut d2 = ddx * ddx + ddy * ddy;
                            if d2 < 1e-6 { d2 = 1e-6; }
                            let d = d2.sqrt();
                            let f = k2 / d;
                            disp[i].0 += ddx / d * f;
                            disp[i].1 += ddy / d * f;
                        }
                    }
                }
            }

            for &(u, v) in &edges {
                let (xu, yu) = pos[u as usize];
                let (xv, yv) = pos[v as usize];
                let dx = xu - xv;
                let dy = yu - yv;
                let mut d = (dx * dx + dy * dy).sqrt();
                if d < 1e-6 { d = 1e-6; }
                let f = d * d / k;
                let fx = dx / d * f;
                let fy = dy / d * f;
                disp[u as usize].0 -= fx;
                disp[u as usize].1 -= fy;
                disp[v as usize].0 += fx;
                disp[v as usize].1 += fy;
            }

            for i in 0..n {
                let (dx, dy) = disp[i];
                let d = (dx * dx + dy * dy).sqrt();
                if d > 1e-9 {
                    let lim = d.min(t);
                    pos[i].0 += dx / d * lim;
                    pos[i].1 += dy / d * lim;
                }
            }

            t = (t - cool).max(cool * 0.5);
        }
    }

    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", pos[i].0, pos[i].1).unwrap();
    }
}
