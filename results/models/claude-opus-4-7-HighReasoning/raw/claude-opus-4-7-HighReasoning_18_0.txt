// Minimum Edge Cut solver.
// Strategy (adaptive by graph scale):
//   1. Stream input with fast buffered IO into a compact CSR.
//   2. Iterative Tarjan bridge search:
//        - if any bridge exists, the answer is 1 (output it).
//   3. Otherwise compute min-degree d (an upper bound on edge-connectivity)
//      and the cut "isolate the min-degree vertex".
//   4. For small/dense graphs (N small) run Stoer-Wagner for an EXACT
//      minimum cut. (Stoer-Wagner is O(V^3); fine while V is moderate.)
//      Otherwise emit the min-degree cut as a valid (often optimal) cut.
//
// Compiles with MSVC /std:c++20.

#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <cstdint>
#include <vector>
#include <algorithm>
#include <numeric>
#include <climits>

using namespace std;

typedef int32_t  i32;
typedef int64_t  i64;
typedef uint32_t u32;
typedef uint64_t u64;

// ---------------- fast IO ----------------
static const int IBUFSZ = 1 << 20;
static char gibuf[IBUFSZ];
static int  gipos = 0, gilen = 0;
static inline int gch() {
    if (gipos >= gilen) {
        gilen = (int)fread(gibuf, 1, IBUFSZ, stdin);
        gipos = 0;
        if (gilen <= 0) return -1;
    }
    return (unsigned char)gibuf[gipos++];
}
static inline i64 rdI() {
    int c = gch();
    while (c != -1 && (c < '0' || c > '9') && c != '-') c = gch();
    int s = 1;
    if (c == '-') { s = -1; c = gch(); }
    i64 x = 0;
    while (c >= '0' && c <= '9') { x = x * 10 + (c - '0'); c = gch(); }
    return x * s;
}

static char gobuf[1 << 20];
static int  gopos = 0;
static inline void oflush() {
    if (gopos) { fwrite(gobuf, 1, gopos, stdout); gopos = 0; }
}
static inline void oputc(char c) {
    if (gopos == (int)sizeof(gobuf)) oflush();
    gobuf[gopos++] = c;
}
static inline void oputI(i64 x) {
    if (x < 0) { oputc('-'); x = -x; }
    char tmp[32]; int n = 0;
    if (x == 0) tmp[n++] = '0';
    while (x) { tmp[n++] = char('0' + x % 10); x /= 10; }
    while (n--) oputc(tmp[n]);
}

// ---------------- main ----------------
int main() {
    i64 N = rdI();
    i64 M = rdI();
    if (N <= 0) {
        printf("0\n");
        return 0;
    }

    // Edge list.
    vector<i32> eu((size_t)M);
    vector<i32> ev((size_t)M);
    vector<i32> deg((size_t)N, 0);

    for (i64 i = 0; i < M; ++i) {
        i32 u = (i32)rdI();
        i32 v = (i32)rdI();
        eu[i] = u;
        ev[i] = v;
        if (u != v) {
            ++deg[u];
            ++deg[v];
        }
    }

    // Build CSR. adj[] stores neighbor vertex, eix[] stores edge index in eu/ev.
    vector<i64> head((size_t)N + 1, 0);
    for (i64 i = 0; i < N; ++i) head[i + 1] = head[i] + deg[i];
    i64 totalDeg = head[N];

    vector<i32> adj((size_t)totalDeg);
    vector<i32> eix((size_t)totalDeg);   // edge id (< M, fits i32 as long as M < 2^31)
    {
        vector<i64> pos((size_t)N);
        for (i64 i = 0; i < N; ++i) pos[i] = head[i];
        for (i64 i = 0; i < M; ++i) {
            i32 u = eu[i], v = ev[i];
            if (u == v) continue;
            adj[pos[u]] = v; eix[pos[u]] = (i32)i; ++pos[u];
            adj[pos[v]] = u; eix[pos[v]] = (i32)i; ++pos[v];
        }
    }

    // ----- 1) Iterative Tarjan bridge detection -----
    vector<i32> disc((size_t)N, -1);
    vector<i32> low ((size_t)N, 0);
    vector<i32> parentEdge((size_t)N, -1);
    vector<i64> itp ((size_t)N);
    for (i64 i = 0; i < N; ++i) itp[i] = head[i];

    vector<i32> stk;
    stk.reserve((size_t)min<i64>(N, 1 << 20));
    i32 timer = 0;
    i64 bridgeId = -1;

    for (i32 s = 0; s < (i32)N && bridgeId == -1; ++s) {
        if (disc[s] != -1) continue;
        disc[s] = low[s] = timer++;
        stk.push_back(s);
        while (!stk.empty()) {
            i32 u = stk.back();
            if (itp[u] < head[u + 1]) {
                i64 idx = itp[u]++;
                i32 v   = adj[idx];
                i32 e   = eix[idx];
                if (disc[v] == -1) {
                    disc[v] = low[v] = timer++;
                    parentEdge[v] = e;
                    stk.push_back(v);
                } else if (e != parentEdge[u]) {
                    if (disc[v] < low[u]) low[u] = disc[v];
                }
            } else {
                stk.pop_back();
                if (!stk.empty()) {
                    i32 p = stk.back();
                    if (low[u] < low[p]) low[p] = low[u];
                    if (low[u] > disc[p]) {
                        bridgeId = parentEdge[u];
                        break;
                    }
                }
            }
        }
    }

    if (bridgeId != -1) {
        oputI(1); oputc('\n');
        oputI(eu[bridgeId]); oputc(' '); oputI(ev[bridgeId]); oputc('\n');
        oflush();
        return 0;
    }

    // Free bridge bookkeeping (not needed below).
    disc.clear(); disc.shrink_to_fit();
    low.clear();  low.shrink_to_fit();
    parentEdge.clear(); parentEdge.shrink_to_fit();
    itp.clear(); itp.shrink_to_fit();
    stk.clear(); stk.shrink_to_fit();

    // ----- 2) Min-degree upper bound + isolation cut -----
    i32 mv = 0;
    for (i32 i = 1; i < (i32)N; ++i) if (deg[i] < deg[mv]) mv = i;
    i32 mind = deg[mv];

    // Edge case: trivial graph.
    if (N <= 1 || mind == 0) {
        oputI(0); oputc('\n');
        oflush();
        return 0;
    }

    // ----- 3) Exact min cut for small graphs (Stoer-Wagner) -----
    // O(V^3) time, O(V^2) memory; safe for V up to a few hundred.
    bool tryExact = (N <= 350);

    if (tryExact) {
        const i32 n = (i32)N;
        vector<i32> W((size_t)n * n, 0);
        for (i64 i = 0; i < M; ++i) {
            i32 a = eu[i], b = ev[i];
            if (a == b) continue;
            W[(size_t)a * n + b]++;
            W[(size_t)b * n + a]++;
        }

        vector<char> alive(n, 1);
        vector<vector<i32>> members(n);
        for (i32 i = 0; i < n; ++i) members[i].push_back(i);

        i64 best = LLONG_MAX;
        vector<i32> bestSet;

        vector<i32> w(n, 0);
        vector<char> added(n, 0);

        i32 nAlive = n;
        while (nAlive >= 2) {
            i32 start = -1;
            for (i32 v = 0; v < n; ++v) if (alive[v]) { start = v; break; }
            if (start < 0) break;

            fill(w.begin(), w.end(), 0);
            fill(added.begin(), added.end(), 0);

            i32 prev = -1, last = -1;
            for (i32 step = 0; step < nAlive; ++step) {
                i32 sel;
                if (step == 0) {
                    sel = start;
                } else {
                    sel = -1;
                    i32 maxw = -1;
                    for (i32 v = 0; v < n; ++v) {
                        if (alive[v] && !added[v] && w[v] > maxw) {
                            maxw = w[v];
                            sel = v;
                        }
                    }
                    if (sel == -1) {
                        for (i32 v = 0; v < n; ++v) if (alive[v] && !added[v]) { sel = v; break; }
                    }
                }
                added[sel] = 1;
                prev = last;
                last = sel;

                if (step == nAlive - 1) {
                    i64 cutw = w[last];
                    if (cutw < best) {
                        best = cutw;
                        bestSet = members[last];
                    }
                    if (prev >= 0 && prev != last) {
                        for (i32 v = 0; v < n; ++v) {
                            if (alive[v] && v != last && v != prev) {
                                i32 add = W[(size_t)last * n + v];
                                W[(size_t)prev * n + v] += add;
                                W[(size_t)v * n + prev] += add;
                            }
                        }
                        members[prev].insert(members[prev].end(),
                                             members[last].begin(),
                                             members[last].end());
                        members[last].clear();
                        alive[last] = 0;
                        --nAlive;
                    } else {
                        break;
                    }
                } else {
                    const i32* row = &W[(size_t)sel * n];
                    for (i32 v = 0; v < n; ++v) {
                        if (alive[v] && !added[v]) {
                            w[v] += row[v];
                        }
                    }
                }
            }
        }

        if (best == LLONG_MAX) best = mind;

        // Use min-degree isolation cut if it's at least as good.
        if ((i64)mind <= best) {
            oputI(mind); oputc('\n');
            for (i64 ei = head[mv]; ei < head[mv + 1]; ++ei) {
                i32 v = adj[ei];
                oputI(mv); oputc(' '); oputI(v); oputc('\n');
            }
            oflush();
            return 0;
        }

        // Emit SW cut: edges with exactly one endpoint in bestSet.
        vector<char> inS((size_t)n, 0);
        for (i32 x : bestSet) inS[x] = 1;

        oputI(best); oputc('\n');
        i64 emitted = 0;
        for (i64 i = 0; i < M && emitted < best; ++i) {
            i32 a = eu[i], b = ev[i];
            if (a == b) continue;
            if (inS[a] != inS[b]) {
                oputI(a); oputc(' '); oputI(b); oputc('\n');
                ++emitted;
            }
        }
        oflush();
        return 0;
    }

    // ----- 4) Large graph: min-degree isolation cut (valid upper bound) -----
    oputI(mind); oputc('\n');
    for (i64 ei = head[mv]; ei < head[mv + 1]; ++ei) {
        i32 v = adj[ei];
        oputI(mv); oputc(' '); oputI(v); oputc('\n');
    }
    oflush();
    return 0;
}
