#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>
#include <climits>

using namespace std;
using ll = long long;

struct Task { int machine; int duration; };

static int J, M;
static vector<vector<Task>> jobs;
static vector<int> jobOpCount;
static int totalOps = 0;

// ---------------- Fast input ----------------
static vector<char> g_inbuf;
static size_t g_inpos = 0;

static void readAll() {
    g_inbuf.clear();
    g_inbuf.reserve(1 << 16);
    char tmp[1 << 16];
    size_t r;
    while ((r = fread(tmp, 1, sizeof(tmp), stdin)) > 0) {
        g_inbuf.insert(g_inbuf.end(), tmp, tmp + r);
    }
    g_inbuf.push_back(0);
}
static inline int rd() {
    while (g_inpos < g_inbuf.size()) {
        char c = g_inbuf[g_inpos];
        if (c == ' ' || c == '\n' || c == '\r' || c == '\t') g_inpos++;
        else break;
    }
    int sign = 1, v = 0;
    if (g_inpos < g_inbuf.size() && g_inbuf[g_inpos] == '-') { sign = -1; g_inpos++; }
    while (g_inpos < g_inbuf.size()) {
        char c = g_inbuf[g_inpos];
        if (c >= '0' && c <= '9') { v = v * 10 + (c - '0'); g_inpos++; }
        else break;
    }
    return v * sign;
}

// ---------------- Output buffer ----------------
static vector<char> outbuf;
static inline void wInt(int x) {
    if (x == 0) { outbuf.push_back('0'); return; }
    if (x < 0) { outbuf.push_back('-'); x = -x; }
    char tmp[12]; int n = 0;
    while (x) { tmp[n++] = (char)('0' + (x % 10)); x /= 10; }
    while (n--) outbuf.push_back(tmp[n]);
}
static inline void wChar(char c) { outbuf.push_back(c); }

// ---------------- Decoders ----------------
static int decode(const vector<int>& seq,
                  const vector<int>& offsets,
                  vector<int>& flatStart)
{
    vector<int> jobReady(J, 0);
    vector<int> jobNext(J, 0);
    vector<int> machReady(M, 0);
    int makespan = 0;
    for (int op : seq) {
        int t = jobNext[op];
        const Task& tk = jobs[op][t];
        int s = std::max(jobReady[op], machReady[tk.machine]);
        int e = s + tk.duration;
        flatStart[offsets[op] + t] = s;
        jobReady[op] = e;
        machReady[tk.machine] = e;
        jobNext[op] = t + 1;
        if (e > makespan) makespan = e;
    }
    return makespan;
}

static int decodeMakespan(const vector<int>& seq,
                          vector<int>& jobReady,
                          vector<int>& jobNext,
                          vector<int>& machReady)
{
    std::fill(jobReady.begin(), jobReady.end(), 0);
    std::fill(jobNext.begin(), jobNext.end(), 0);
    std::fill(machReady.begin(), machReady.end(), 0);
    int makespan = 0;
    for (int op : seq) {
        int t = jobNext[op];
        const Task& tk = jobs[op][t];
        int s = std::max(jobReady[op], machReady[tk.machine]);
        int e = s + tk.duration;
        jobReady[op] = e;
        machReady[tk.machine] = e;
        jobNext[op] = t + 1;
        if (e > makespan) makespan = e;
    }
    return makespan;
}

// ---------------- Dispatch rule builder ----------------
// rule: 0 SPT (shortest op), 1 LPT, 2 MWR (most work remaining), 3 LWR,
//       4 FIFO (lowest job idx), 5 MOR (most ops remaining), 6 EST (earliest possible start)
static vector<int> dispatchBuild(int rule) {
    vector<int> jobReady(J, 0);
    vector<int> jobNext(J, 0);
    vector<int> machReady(M, 0);
    vector<ll> remWork(J, 0);
    for (int j = 0; j < J; ++j) for (auto& t : jobs[j]) remWork[j] += t.duration;

    vector<int> seq;
    seq.reserve(totalOps);

    while ((int)seq.size() < totalOps) {
        int best = -1;
        ll bestKey = 0;
        int bestTie = INT_MAX;
        for (int j = 0; j < J; ++j) {
            if (jobNext[j] >= (int)jobs[j].size()) continue;
            const Task& tk = jobs[j][jobNext[j]];
            int s = std::max(jobReady[j], machReady[tk.machine]);
            ll key;
            switch (rule) {
                case 0: key = -(ll)tk.duration; break;
                case 1: key = (ll)tk.duration; break;
                case 2: key = remWork[j]; break;
                case 3: key = -remWork[j]; break;
                case 4: key = -(ll)j; break;
                case 5: key = (ll)((int)jobs[j].size() - jobNext[j]); break;
                case 6: key = -(ll)s; break;
                default: key = 0;
            }
            if (best < 0 || key > bestKey || (key == bestKey && s < bestTie)) {
                best = j; bestKey = key; bestTie = s;
            }
        }
        int j = best;
        const Task& tk = jobs[j][jobNext[j]];
        int s = std::max(jobReady[j], machReady[tk.machine]);
        int e = s + tk.duration;
        jobReady[j] = e;
        machReady[tk.machine] = e;
        remWork[j] -= tk.duration;
        seq.push_back(j);
        jobNext[j] = jobNext[j] + 1;
    }
    return seq;
}

int main() {
    auto tStart = chrono::steady_clock::now();
    readAll();

    J = rd();
    M = rd();
    jobs.assign(J, {});
    jobOpCount.assign(J, 0);
    totalOps = 0;
    for (int j = 0; j < J; ++j) {
        int T = rd();
        jobOpCount[j] = T;
        jobs[j].resize(T);
        for (int t = 0; t < T; ++t) {
            int mch = rd();
            int dur = rd();
            jobs[j][t] = {mch, dur};
        }
        totalOps += T;
    }

    vector<int> offsets(J, 0);
    for (int j = 1; j < J; ++j) offsets[j] = offsets[j-1] + jobOpCount[j-1];
    vector<int> flatStart(totalOps, 0);

    vector<int> bestSeq;
    int bestMakespan = INT_MAX;

    vector<int> jobReady(J), jobNext(J), machReady(M);

    // ----- Try multiple dispatch rules; keep best -----
    const int numRules = 7;
    for (int r = 0; r < numRules; ++r) {
        auto seq = dispatchBuild(r);
        int ms = decodeMakespan(seq, jobReady, jobNext, machReady);
        if (ms < bestMakespan) {
            bestMakespan = ms;
            bestSeq = seq;
        }
    }

    // ----- Adaptive time budget for local search -----
    double timeBudgetSec;
    if (totalOps <= 50)         timeBudgetSec = 1.0;
    else if (totalOps <= 500)   timeBudgetSec = 2.0;
    else if (totalOps <= 5000)  timeBudgetSec = 3.0;
    else if (totalOps <= 20000) timeBudgetSec = 3.0;
    else if (totalOps <= 50000) timeBudgetSec = 2.0;
    else                        timeBudgetSec = 1.0;

    auto elapsed = [&]() {
        return chrono::duration<double>(
            chrono::steady_clock::now() - tStart).count();
    };

    // ----- Local search: adjacent-swap hill climb with restarts -----
    std::mt19937 rng(0xC0FFEEu ^ (uint32_t)totalOps
                     ^ ((uint32_t)J << 7) ^ ((uint32_t)M << 13));

    int seqLen = totalOps;
    vector<int> curSeq = bestSeq;
    int curMakespan = bestMakespan;

    if (seqLen >= 2) {
        long long iterCap;
        if (totalOps <= 100)         iterCap = 500000;
        else if (totalOps <= 1000)   iterCap = 300000;
        else if (totalOps <= 10000)  iterCap = 80000;
        else if (totalOps <= 50000)  iterCap = 15000;
        else                         iterCap = 3000;

        long long iter = 0;
        int sinceImprove = 0;
        while (iter < iterCap) {
            if ((iter & 63) == 0 && elapsed() > timeBudgetSec) break;
            int i = (int)(rng() % (uint32_t)(seqLen - 1));
            if (curSeq[i] == curSeq[i+1]) { iter++; continue; }
            std::swap(curSeq[i], curSeq[i+1]);
            int ms = decodeMakespan(curSeq, jobReady, jobNext, machReady);
            if (ms <= curMakespan) {
                curMakespan = ms;
                if (ms < bestMakespan) {
                    bestMakespan = ms;
                    bestSeq = curSeq;
                    sinceImprove = 0;
                } else {
                    sinceImprove++;
                }
            } else {
                std::swap(curSeq[i], curSeq[i+1]);
                sinceImprove++;
            }
            // Restart from best every so often for diversification
            if (sinceImprove > 5000 && totalOps <= 5000) {
                curSeq = bestSeq;
                curMakespan = bestMakespan;
                // perform a few random shuffles between adjacent pairs
                for (int k = 0; k < 20; ++k) {
                    int p = (int)(rng() % (uint32_t)(seqLen - 1));
                    if (curSeq[p] != curSeq[p+1]) std::swap(curSeq[p], curSeq[p+1]);
                }
                curMakespan = decodeMakespan(curSeq, jobReady, jobNext, machReady);
                sinceImprove = 0;
            }
            iter++;
        }
    }

    // ----- Decode final best sequence -----
    decode(bestSeq, offsets, flatStart);

    // ----- Output -----
    outbuf.reserve((size_t)totalOps * 12);
    for (int j = 0; j < J; ++j) {
        int base = offsets[j];
        int T = jobOpCount[j];
        for (int t = 0; t < T; ++t) {
            if (t) wChar(' ');
            wInt(flatStart[base + t]);
            wChar(' ');
            wInt(jobs[j][t].machine);
        }
        wChar('\n');
    }
    fwrite(outbuf.data(), 1, outbuf.size(), stdout);
    return 0;
}
