OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
      %fn_ty = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

      %int_0 = OpConstant %int 0
     %uint_1 = OpConstant %uint 1

    %float_dark  = OpConstant %float 0.6

   %dx0 = OpConstant %float 0.04
   %dy0 = OpConstant %float 0.0
   %dx1 = OpConstant %float 0.02
   %dy1 = OpConstant %float 0.034641016
   %dx2 = OpConstant %float -0.02
   %dy2 = OpConstant %float 0.034641016
   %dx3 = OpConstant %float -0.04
   %dy3 = OpConstant %float 0.0
   %dx4 = OpConstant %float -0.02
   %dy4 = OpConstant %float -0.034641016
   %dx5 = OpConstant %float 0.02
   %dy5 = OpConstant %float -0.034641016

%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%gl_in            = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4        = OpTypePointer Input %v4float

%arr_v3_in        = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3_in
%inColor          = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3        = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%gl_out           = OpVariable %ptr_out_pv Output
%ptr_out_v4       = OpTypePointer Output %v4float

%ptr_out_v3       = OpTypePointer Output %v3float
%outColor         = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_ty
     %entry = OpLabel

   %p_in_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
   %center_pos = OpLoad %v4float %p_in_ptr

   %c_in_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
   %center_color = OpLoad %v3float %c_in_ptr

   %dark_color = OpVectorTimesScalar %v3float %center_color %float_dark

   %cx = OpCompositeExtract %float %center_pos 0
   %cy = OpCompositeExtract %float %center_pos 1
   %cz = OpCompositeExtract %float %center_pos 2
   %cw = OpCompositeExtract %float %center_pos 3

   %e0x = OpFAdd %float %cx %dx0
   %e0y = OpFAdd %float %cy %dy0
   %edge0 = OpCompositeConstruct %v4float %e0x %e0y %cz %cw

   %e1x = OpFAdd %float %cx %dx1
   %e1y = OpFAdd %float %cy %dy1
   %edge1 = OpCompositeConstruct %v4float %e1x %e1y %cz %cw

   %e2x = OpFAdd %float %cx %dx2
   %e2y = OpFAdd %float %cy %dy2
   %edge2 = OpCompositeConstruct %v4float %e2x %e2y %cz %cw

   %e3x = OpFAdd %float %cx %dx3
   %e3y = OpFAdd %float %cy %dy3
   %edge3 = OpCompositeConstruct %v4float %e3x %e3y %cz %cw

   %e4x = OpFAdd %float %cx %dx4
   %e4y = OpFAdd %float %cy %dy4
   %edge4 = OpCompositeConstruct %v4float %e4x %e4y %cz %cw

   %e5x = OpFAdd %float %cx %dx5
   %e5y = OpFAdd %float %cy %dy5
   %edge5 = OpCompositeConstruct %v4float %e5x %e5y %cz %cw

   %pos_out = OpAccessChain %ptr_out_v4 %gl_out %int_0

   OpStore %pos_out %center_pos
   OpStore %outColor %center_color
   OpEmitVertex
   OpStore %pos_out %edge0
   OpStore %outColor %dark_color
   OpEmitVertex
   OpStore %pos_out %edge1
   OpStore %outColor %dark_color
   OpEmitVertex
   OpEndPrimitive

   OpStore %pos_out %center_pos
   OpStore %outColor %center_color
   OpEmitVertex
   OpStore %pos_out %edge1
   OpStore %outColor %dark_color
   OpEmitVertex
   OpStore %pos_out %edge2
   OpStore %outColor %dark_color
   OpEmitVertex
   OpEndPrimitive

   OpStore %pos_out %center_pos
   OpStore %outColor %center_color
   OpEmitVertex
   OpStore %pos_out %edge2
   OpStore %outColor %dark_color
   OpEmitVertex
   OpStore %pos_out %edge3
   OpStore %outColor %dark_color
   OpEmitVertex
   OpEndPrimitive

   OpStore %pos_out %center_pos
   OpStore %outColor %center_color
   OpEmitVertex
   OpStore %pos_out %edge3
   OpStore %outColor %dark_color
   OpEmitVertex
   OpStore %pos_out %edge4
   OpStore %outColor %dark_color
   OpEmitVertex
   OpEndPrimitive

   OpStore %pos_out %center_pos
   OpStore %outColor %center_color
   OpEmitVertex
   OpStore %pos_out %edge4
   OpStore %outColor %dark_color
   OpEmitVertex
   OpStore %pos_out %edge5
   OpStore %outColor %dark_color
   OpEmitVertex
   OpEndPrimitive

   OpStore %pos_out %center_pos
   OpStore %outColor %center_color
   OpEmitVertex
   OpStore %pos_out %edge5
   OpStore %outColor %dark_color
   OpEmitVertex
   OpStore %pos_out %edge0
   OpStore %outColor %dark_color
   OpEmitVertex
   OpEndPrimitive

   OpReturn
   OpFunctionEnd
