use std::io::{self, Read, Write, BufWriter};

struct Hole {
    id: String,
    sx: f64, sy: f64, sz: f64,
    dx: f64, dy: f64, dz: f64,
    depths: Vec<f64>,
    values: Vec<Vec<f64>>, // values[prop_idx][sample_idx]
}

fn parse_f64(s: &str) -> f64 {
    s.parse::<f64>().unwrap_or(f64::NAN)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut tokens = input.split_ascii_whitespace();

    let num_holes: usize = match tokens.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => { writeln!(out, "0").ok(); return; }
    };
    let num_props: usize = tokens.next().and_then(|s| s.parse().ok()).unwrap_or(0);

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(t) = tokens.next() { prop_names.push(t.to_string()); }
        else { prop_names.push(String::from("?")); }
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);

    for _ in 0..num_holes {
        let hid = match tokens.next() { Some(s) => s.to_string(), None => break };
        let sx = parse_f64(tokens.next().unwrap_or("0"));
        let sy = parse_f64(tokens.next().unwrap_or("0"));
        let sz = parse_f64(tokens.next().unwrap_or("0"));
        let dx = parse_f64(tokens.next().unwrap_or("0"));
        let dy = parse_f64(tokens.next().unwrap_or("0"));
        let dz = parse_f64(tokens.next().unwrap_or("0"));
        let _length = parse_f64(tokens.next().unwrap_or("0"));
        let nsamp: usize = tokens.next().and_then(|s| s.parse().ok()).unwrap_or(0);

        let mut ndx = dx; let mut ndy = dy; let mut ndz = dz;
        let mag = (dx*dx + dy*dy + dz*dz).sqrt();
        if mag > 1e-12 { ndx /= mag; ndy /= mag; ndz /= mag; }
        else { ndz = -1.0; }

        let mut depths = Vec::with_capacity(nsamp);
        let mut values: Vec<Vec<f64>> = (0..num_props).map(|_| Vec::with_capacity(nsamp)).collect();

        for _ in 0..nsamp {
            let d = parse_f64(tokens.next().unwrap_or("0"));
            depths.push(d);
            for p in 0..num_props {
                let v = parse_f64(tokens.next().unwrap_or("nan"));
                values[p].push(v);
            }
        }

        holes.push(Hole { id: hid, sx, sy, sz, dx: ndx, dy: ndy, dz: ndz, depths, values });
    }

    // Per-property global stats
    let mut global_count = vec![0usize; num_props];
    let mut global_sum = vec![0.0f64; num_props];
    let mut global_sum_sq = vec![0.0f64; num_props];

    let mut total_samples: usize = 0;
    for h in &holes {
        total_samples += h.depths.len();
        for p in 0..num_props {
            for &v in &h.values[p] {
                if v.is_finite() {
                    global_count[p] += 1;
                    global_sum[p] += v;
                    global_sum_sq[p] += v*v;
                }
            }
        }
    }

    let mut global_std = vec![0.0f64; num_props];
    for p in 0..num_props {
        if global_count[p] > 0 {
            let m = global_sum[p] / global_count[p] as f64;
            let var = global_sum_sq[p] / global_count[p] as f64 - m*m;
            global_std[p] = if var > 0.0 { var.sqrt() } else { 0.0 };
        }
    }

    // Robust median & MAD via sampling
    let mut global_mad = vec![0.0f64; num_props];
    for p in 0..num_props {
        let mut buf: Vec<f64> = Vec::new();
        let sample_cap = 200_000usize;
        if global_count[p] <= sample_cap {
            buf.reserve(global_count[p]);
            for h in &holes { for &v in &h.values[p] { if v.is_finite() { buf.push(v); } } }
        } else {
            let stride = (global_count[p] / sample_cap).max(1);
            let mut idx = 0usize;
            for h in &holes {
                for &v in &h.values[p] {
                    if v.is_finite() {
                        if idx % stride == 0 { buf.push(v); }
                        idx += 1;
                    }
                }
            }
        }
        if !buf.is_empty() {
            buf.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
            let med = buf[buf.len()/2];
            let mut dev: Vec<f64> = buf.iter().map(|x| (x - med).abs()).collect();
            dev.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
            global_mad[p] = dev[dev.len()/2];
        }
    }

    if total_samples == 0 { writeln!(out, "0").ok(); return; }

    // 3D coordinates for every sample
    let mut sample_coords: Vec<(f32, f32, f32)> = Vec::with_capacity(total_samples);
    let mut sample_owner: Vec<(u32, u32)> = Vec::with_capacity(total_samples);

    for (hi, h) in holes.iter().enumerate() {
        for (si, &d) in h.depths.iter().enumerate() {
            let x = h.sx + h.dx * d;
            let y = h.sy + h.dy * d;
            let z = h.sz + h.dz * d;
            sample_coords.push((x as f32, y as f32, z as f32));
            sample_owner.push((hi as u32, si as u32));
        }
    }

    // Bounding box
    let mut min_x = f32::INFINITY; let mut min_y = f32::INFINITY; let mut min_z = f32::INFINITY;
    let mut max_x = f32::NEG_INFINITY; let mut max_y = f32::NEG_INFINITY; let mut max_z = f32::NEG_INFINITY;
    for &(x,y,z) in &sample_coords {
        if x < min_x { min_x = x; } if x > max_x { max_x = x; }
        if y < min_y { min_y = y; } if y > max_y { max_y = y; }
        if z < min_z { min_z = z; } if z > max_z { max_z = z; }
    }
    let extent_x = (max_x - min_x).max(1.0);
    let extent_y = (max_y - min_y).max(1.0);
    let extent_z = (max_z - min_z).max(1.0);
    let volume = (extent_x as f64) * (extent_y as f64) * (extent_z as f64);
    let target_per_cell = 8.0;
    let cell_vol = volume * target_per_cell / (total_samples as f64).max(1.0);
    let mut cell_size = (cell_vol.cbrt() as f32).max(1e-3);

    let mut nx = ((extent_x / cell_size).ceil() as i64).max(1) as usize;
    let mut ny = ((extent_y / cell_size).ceil() as i64).max(1) as usize;
    let mut nz = ((extent_z / cell_size).ceil() as i64).max(1) as usize;

    let max_cells = 8_000_000usize;
    if nx.saturating_mul(ny).saturating_mul(nz) > max_cells {
        let scale = ((nx as f64 * ny as f64 * nz as f64) / max_cells as f64).cbrt();
        cell_size *= scale as f32;
        nx = ((extent_x / cell_size).ceil() as i64).max(1) as usize;
        ny = ((extent_y / cell_size).ceil() as i64).max(1) as usize;
        nz = ((extent_z / cell_size).ceil() as i64).max(1) as usize;
    }

    let inv_cs = 1.0 / cell_size;
    let cell_idx = |x: f32, y: f32, z: f32| -> (usize, usize, usize) {
        let cx = (((x - min_x) * inv_cs) as i64).max(0).min(nx as i64 - 1) as usize;
        let cy = (((y - min_y) * inv_cs) as i64).max(0).min(ny as i64 - 1) as usize;
        let cz = (((z - min_z) * inv_cs) as i64).max(0).min(nz as i64 - 1) as usize;
        (cx, cy, cz)
    };
    let flat = |cx: usize, cy: usize, cz: usize| -> usize { (cz * ny + cy) * nx + cx };
    let ncells = nx * ny * nz;

    // CSR cell-data
    let mut counts = vec![0u32; ncells];
    for &(x, y, z) in &sample_coords {
        let (cx, cy, cz) = cell_idx(x, y, z);
        counts[flat(cx, cy, cz)] += 1;
    }
    let mut offsets = vec![0u32; ncells + 1];
    for i in 0..ncells { offsets[i+1] = offsets[i] + counts[i]; }
    let mut cell_data = vec![0u32; total_samples];
    let mut cursor = vec![0u32; ncells];
    for (i, &(x,y,z)) in sample_coords.iter().enumerate() {
        let (cx, cy, cz) = cell_idx(x, y, z);
        let cell = flat(cx, cy, cz);
        let pos = offsets[cell] + cursor[cell];
        cell_data[pos as usize] = i as u32;
        cursor[cell] += 1;
    }

    // K neighbors based on scale
    let k_neighbors: usize = if total_samples < 500 { 6 }
        else if total_samples < 5_000 { 10 }
        else if total_samples < 50_000 { 12 }
        else if total_samples < 500_000 { 10 }
        else { 8 };

    let max_shell: i32 = if total_samples < 10_000 { 4 } else { 3 };

    let mut suspects: Vec<(f64, u32, u32, u32)> = Vec::new();

    // Flat per-property value arrays for fast access
    let mut prop_vals: Vec<Vec<f64>> = vec![Vec::with_capacity(total_samples); num_props];
    for p in 0..num_props {
        for &(hi, si) in &sample_owner {
            prop_vals[p].push(holes[hi as usize].values[p][si as usize]);
        }
    }

    let mut neighbor_vals: Vec<f64> = Vec::with_capacity(128);
    let mut neighbor_dists: Vec<f64> = Vec::with_capacity(128);
    let mut pairs: Vec<(f64, f64)> = Vec::with_capacity(128);

    for p in 0..num_props {
        if global_count[p] < 3 { continue; }
        let gstd = global_std[p];
        let gmad = global_mad[p];
        let robust_scale = if gmad > 1e-12 { 1.4826 * gmad } else { gstd.max(1e-12) };
        let vals_p = &prop_vals[p];

        for i in 0..total_samples {
            let v = vals_p[i];
            if !v.is_finite() { continue; }
            let (sx, sy, sz) = sample_coords[i];
            let (cx0, cy0, cz0) = cell_idx(sx, sy, sz);
            let cx0i = cx0 as i32; let cy0i = cy0 as i32; let cz0i = cz0 as i32;

            neighbor_vals.clear();
            neighbor_dists.clear();

            let mut shell: i32 = 0;
            while neighbor_vals.len() < k_neighbors && shell <= max_shell {
                let s = shell;
                let prev = s - 1;
                for dz in -s..=s {
                    let cz = cz0i + dz;
                    if cz < 0 || cz >= nz as i32 { continue; }
                    for dy in -s..=s {
                        let cy = cy0i + dy;
                        if cy < 0 || cy >= ny as i32 { continue; }
                        for dx in -s..=s {
                            let cx = cx0i + dx;
                            if cx < 0 || cx >= nx as i32 { continue; }
                            if s > 0 && dx.abs() <= prev && dy.abs() <= prev && dz.abs() <= prev { continue; }
                            let cell = flat(cx as usize, cy as usize, cz as usize);
                            let start = offsets[cell] as usize;
                            let end = offsets[cell+1] as usize;
                            for idx_ in start..end {
                                let j = cell_data[idx_] as usize;
                                if j == i { continue; }
                                let jv = vals_p[j];
                                if !jv.is_finite() { continue; }
                                let (jx, jy, jz) = sample_coords[j];
                                let dxv = (jx - sx) as f64;
                                let dyv = (jy - sy) as f64;
                                let dzv = (jz - sz) as f64;
                                let d2 = dxv*dxv + dyv*dyv + dzv*dzv;
                                neighbor_vals.push(jv);
                                neighbor_dists.push(d2);
                            }
                        }
                    }
                }
                shell += 1;
            }

            if neighbor_vals.len() < 3 { continue; }

            pairs.clear();
            for (d2, jv) in neighbor_dists.iter().zip(neighbor_vals.iter()) {
                pairs.push((*d2, *jv));
            }
            pairs.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal));
            let k = k_neighbors.min(pairs.len());
            pairs.truncate(k);

            let mut wsum = 0.0; let mut vsum = 0.0;
            let mut vals_only: Vec<f64> = Vec::with_capacity(k);
            for &(d2, vv) in pairs.iter() {
                let w = 1.0 / (d2 + 1e-6);
                wsum += w;
                vsum += w * vv;
                vals_only.push(vv);
            }
            if wsum <= 0.0 { continue; }
            let predicted = vsum / wsum;

            vals_only.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
            let local_med = vals_only[vals_only.len()/2];
            let mut local_dev: Vec<f64> = vals_only.iter().map(|x| (x - local_med).abs()).collect();
            local_dev.sort_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
            let local_mad = local_dev[local_dev.len()/2];
            let local_scale = if local_mad > 1e-9 { 1.4826 * local_mad } else { robust_scale };
            let scale = local_scale.max(robust_scale * 0.1);

            let z = (v - predicted).abs() / scale;
            if z > 3.0 && z.is_finite() {
                let (hi, si) = sample_owner[i];
                suspects.push((z, hi, si, p as u32));
            }
        }
    }

    suspects.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));

    let max_out = 1_000_000usize.min(suspects.len());
    writeln!(out, "{}", max_out).ok();
    for &(conf, hi, si, pi) in suspects.iter().take(max_out) {
        writeln!(out, "{} {} {} {:.4}", holes[hi as usize].id, si, prop_names[pi as usize], conf).ok();
    }
}
