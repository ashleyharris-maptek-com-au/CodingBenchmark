using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Text;

class OrbitalTSP
{
    const double MU = 398600.4418;

    struct State
    {
        public double x, y, z, vx, vy, vz;
    }

    static State ParseState(string line)
    {
        var parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        return new State
        {
            x = double.Parse(parts[0], CultureInfo.InvariantCulture),
            y = double.Parse(parts[1], CultureInfo.InvariantCulture),
            z = double.Parse(parts[2], CultureInfo.InvariantCulture),
            vx = double.Parse(parts[3], CultureInfo.InvariantCulture),
            vy = double.Parse(parts[4], CultureInfo.InvariantCulture),
            vz = double.Parse(parts[5], CultureInfo.InvariantCulture),
        };
    }

    // Heuristic cost: position distance + weighted velocity mismatch.
    // Cheap proxy for transfer delta-V that gives reasonable ordering at all scales.
    static double Cost(in State a, in State b)
    {
        double dx = b.x - a.x, dy = b.y - a.y, dz = b.z - a.z;
        double dist = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double dvx = b.vx - a.vx, dvy = b.vy - a.vy, dvz = b.vz - a.vz;
        double dv = Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);
        return dist + 500.0 * dv;
    }

    // 0 = start, 1..N = stations. order holds station indices (0..N-1).
    static double TourCost(int[] order, double[,] C)
    {
        double total = C[0, order[0] + 1];
        for (int i = 0; i < order.Length - 1; i++)
            total += C[order[i] + 1, order[i + 1] + 1];
        return total;
    }

    static int[] NearestNeighbor(double[,] C, int N, int firstChoice = -1)
    {
        var order = new int[N];
        var visited = new bool[N];
        int cur;
        int startStep = 0;
        if (firstChoice >= 0)
        {
            order[0] = firstChoice; visited[firstChoice] = true;
            cur = firstChoice + 1; startStep = 1;
        }
        else cur = 0;

        for (int step = startStep; step < N; step++)
        {
            double best = double.PositiveInfinity;
            int bestJ = -1;
            for (int j = 0; j < N; j++)
            {
                if (visited[j]) continue;
                double c = C[cur, j + 1];
                if (c < best) { best = c; bestJ = j; }
            }
            order[step] = bestJ;
            visited[bestJ] = true;
            cur = bestJ + 1;
        }
        return order;
    }

    static void TwoOpt(int[] order, double[,] C, Stopwatch sw, long deadlineMs)
    {
        int N = order.Length;
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                if (sw.ElapsedMilliseconds > deadlineMs) return;
                for (int j = i + 1; j < N; j++)
                {
                    int aPrev = i == 0 ? 0 : order[i - 1] + 1;
                    int a = order[i] + 1;
                    int b = order[j] + 1;
                    int bNext = j == N - 1 ? -1 : order[j + 1] + 1;

                    double before = C[aPrev, a];
                    double after = C[aPrev, b];
                    if (bNext >= 0)
                    {
                        before += C[b, bNext];
                        after += C[a, bNext];
                    }
                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(order, i, j - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static void OrOpt(int[] order, double[,] C, Stopwatch sw, long deadlineMs)
    {
        int N = order.Length;
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int segLen = 1; segLen <= 3 && segLen < N; segLen++)
            {
                for (int i = 0; i <= N - segLen; i++)
                {
                    if (sw.ElapsedMilliseconds > deadlineMs) return;
                    int prev = i == 0 ? 0 : order[i - 1] + 1;
                    int segStart = order[i] + 1;
                    int segEnd = order[i + segLen - 1] + 1;
                    int next = (i + segLen >= N) ? -1 : order[i + segLen] + 1;

                    double removeCost = C[prev, segStart];
                    if (next >= 0) removeCost += C[segEnd, next];
                    double bridgeCost = (next >= 0) ? C[prev, next] : 0.0;

                    int bestJ = -1;
                    double bestDelta = -1e-9;

                    for (int j = 0; j <= N - segLen; j++)
                    {
                        if (j >= i && j <= i + segLen) continue;

                        int insPrev, insNext;
                        if (j == 0) { insPrev = 0; insNext = order[0] + 1; }
                        else if (j == N - segLen)
                        {
                            int lastIdx = (i + segLen == N) ? N - segLen - 1 : N - 1;
                            if (lastIdx < 0) continue;
                            insPrev = order[lastIdx] + 1; insNext = -1;
                        }
                        else
                        {
                            int prevIdx = j - 1;
                            int nextIdx = j;
                            if (prevIdx >= i && prevIdx <= i + segLen - 1) continue;
                            if (nextIdx >= i && nextIdx <= i + segLen - 1) continue;
                            insPrev = order[prevIdx] + 1;
                            insNext = order[nextIdx] + 1;
                        }

                        double insertOld = (insNext >= 0) ? C[insPrev, insNext] : 0.0;
                        double insertNew = C[insPrev, segStart] + (insNext >= 0 ? C[segEnd, insNext] : 0.0);
                        double delta = (bridgeCost - removeCost) + (insertNew - insertOld);
                        if (delta < bestDelta) { bestDelta = delta; bestJ = j; }
                    }

                    if (bestJ >= 0)
                    {
                        var seg = new int[segLen];
                        for (int k = 0; k < segLen; k++) seg[k] = order[i + k];
                        var list = new List<int>(order);
                        list.RemoveRange(i, segLen);
                        int insertAt = bestJ;
                        if (bestJ > i) insertAt -= segLen;
                        list.InsertRange(insertAt, seg);
                        for (int k = 0; k < N; k++) order[k] = list[k];
                        improved = true;
                    }
                }
            }
        }
    }

    static int[] BruteForce(double[,] C, int N)
    {
        var perm = new int[N];
        for (int i = 0; i < N; i++) perm[i] = i;
        int[] best = (int[])perm.Clone();
        double bestCost = TourCost(perm, C);
        Permute(perm, 0, C, ref bestCost, ref best);
        return best;
    }

    static void Permute(int[] arr, int k, double[,] C, ref double bestCost, ref int[] best)
    {
        if (k == arr.Length - 1)
        {
            double c = TourCost(arr, C);
            if (c < bestCost) { bestCost = c; best = (int[])arr.Clone(); }
            return;
        }
        for (int i = k; i < arr.Length; i++)
        {
            (arr[k], arr[i]) = (arr[i], arr[k]);
            Permute(arr, k + 1, C, ref bestCost, ref best);
            (arr[k], arr[i]) = (arr[i], arr[k]);
        }
    }

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        var input = Console.In.ReadToEnd();
        var lines = input.Split(new[] { '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);

        var hdr = lines[0].Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(hdr[0], CultureInfo.InvariantCulture);

        var states = new State[N + 1];
        states[0] = ParseState(lines[1]);
        for (int i = 0; i < N; i++)
            states[i + 1] = ParseState(lines[2 + i]);

        var C = new double[N + 1, N + 1];
        for (int i = 0; i <= N; i++)
            for (int j = 0; j <= N; j++)
                if (i != j) C[i, j] = Cost(states[i], states[j]);

        int[] order;

        if (N == 0)
        {
            Console.WriteLine();
            return;
        }
        else if (N == 1)
        {
            order = new[] { 0 };
        }
        else if (N <= 8)
        {
            order = BruteForce(C, N);
        }
        else
        {
            long totalBudget = 240000L;
            int starts;
            long perStart;
            if (N <= 20) { starts = Math.Min(N, 10); }
            else if (N <= 50) { starts = 5; }
            else { starts = 2; }
            perStart = totalBudget / starts;

            int[] best = NearestNeighbor(C, N);
            long d1 = sw.ElapsedMilliseconds + perStart;
            TwoOpt(best, C, sw, d1);
            OrOpt(best, C, sw, d1);
            double bestCost = TourCost(best, C);

            var rng = new Random(12345);
            var triedFirst = new HashSet<int>();
            for (int s = 1; s < starts && sw.ElapsedMilliseconds < totalBudget - 5000; s++)
            {
                int first;
                int attempts = 0;
                do { first = rng.Next(N); attempts++; } while (triedFirst.Contains(first) && attempts < 20);
                triedFirst.Add(first);

                var cand = NearestNeighbor(C, N, first);
                long deadline = Math.Min(totalBudget, sw.ElapsedMilliseconds + perStart);
                TwoOpt(cand, C, sw, deadline);
                OrOpt(cand, C, sw, deadline);
                double cc = TourCost(cand, C);
                if (cc < bestCost) { bestCost = cc; best = cand; }
            }

            order = best;
        }

        var sbOut = new StringBuilder();
        for (int i = 0; i < order.Length; i++)
        {
            if (i > 0) sbOut.Append(' ');
            sbOut.Append(order[i]);
        }
        Console.WriteLine(sbOut.ToString());
    }
}
