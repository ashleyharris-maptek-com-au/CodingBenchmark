using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

class Program {
    static byte[] buf;
    static int bpos, blen;

    static int ReadInt() {
        while (bpos < blen && (buf[bpos] < (byte)'0' || buf[bpos] > (byte)'9') && buf[bpos] != (byte)'-') bpos++;
        int sign = 1;
        if (bpos < blen && buf[bpos] == (byte)'-') { sign = -1; bpos++; }
        int v = 0;
        while (bpos < blen && buf[bpos] >= (byte)'0' && buf[bpos] <= (byte)'9') {
            v = v * 10 + (buf[bpos] - (byte)'0');
            bpos++;
        }
        return v * sign;
    }

    static void Main() {
        using (var ms = new MemoryStream()) {
            var stdin = Console.OpenStandardInput();
            stdin.CopyTo(ms);
            buf = ms.ToArray();
            blen = buf.Length;
        }
        bpos = 0;

        int N = ReadInt();
        int M = ReadInt();

        var sw = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);

        if (N <= 0) { sw.WriteLine(0); sw.Flush(); return; }

        var adjTo = new List<int>[N];
        var adjE = new List<int>[N];
        for (int i = 0; i < N; i++) { adjTo[i] = new List<int>(); adjE[i] = new List<int>(); }

        for (int i = 0; i < M; i++) {
            int u = ReadInt();
            int v = ReadInt();
            ReadInt(); // weight (not needed for tree-based pairing heuristic)
            adjTo[u].Add(v); adjE[u].Add(i);
            if (u != v) { adjTo[v].Add(u); adjE[v].Add(i); }
        }

        if (M == 0) { sw.WriteLine(0); sw.Flush(); return; }

        // BFS spanning tree from node 0
        var parent = new int[N];
        var parentE = new int[N];
        var order = new int[N];
        var visited = new bool[N];
        for (int i = 0; i < N; i++) { parent[i] = -1; parentE[i] = -1; }
        int oi = 0;
        var bfsQ = new int[N];
        int qh = 0, qt = 0;
        bfsQ[qt++] = 0;
        visited[0] = true;
        while (qh < qt) {
            int u = bfsQ[qh++];
            order[oi++] = u;
            var aTo = adjTo[u];
            var aE = adjE[u];
            int cnt = aTo.Count;
            for (int j = 0; j < cnt; j++) {
                int v = aTo[j];
                if (!visited[v]) {
                    visited[v] = true;
                    parent[v] = u;
                    parentE[v] = aE[j];
                    bfsQ[qt++] = v;
                }
            }
        }

        // Determine odd-degree vertices (self-loop contributes 2 to degree)
        var oddV = new bool[N];
        for (int u = 0; u < N; u++) {
            var aTo = adjTo[u];
            int cnt = aTo.Count;
            int sl = 0;
            for (int j = 0; j < cnt; j++) if (aTo[j] == u) sl++;
            int actualDeg = cnt + sl;
            oddV[u] = (actualDeg & 1) == 1;
        }

        // T-join on spanning tree: in reverse BFS order, if subtree has odd #odd, duplicate parent edge
        int totalEdges = M;
        var oddInSubtree = new int[N];
        for (int u = 0; u < N; u++) oddInSubtree[u] = oddV[u] ? 1 : 0;

        for (int i = oi - 1; i > 0; i--) {
            int u = order[i];
            int par = parent[u];
            if ((oddInSubtree[u] & 1) == 1) {
                int newId = totalEdges++;
                adjTo[u].Add(par); adjE[u].Add(newId);
                adjTo[par].Add(u); adjE[par].Add(newId);
            }
            oddInSubtree[par] += oddInSubtree[u];
        }

        // Hierholzer's algorithm (iterative)
        var used = new bool[totalEdges];
        var iter = new int[N];
        int stackCap = totalEdges + 8;
        var stack = new int[stackCap];
        int sp = 0;
        stack[sp++] = 0;
        var circuit = new int[stackCap];
        int cp = 0;

        while (sp > 0) {
            int u = stack[sp - 1];
            var aTo = adjTo[u];
            var aE = adjE[u];
            int aCount = aTo.Count;
            bool advanced = false;
            int it = iter[u];
            while (it < aCount) {
                int eid = aE[it];
                int nxt = aTo[it];
                it++;
                if (used[eid]) continue;
                used[eid] = true;
                iter[u] = it;
                stack[sp++] = nxt;
                advanced = true;
                break;
            }
            if (!advanced) {
                iter[u] = it;
                circuit[cp++] = stack[--sp];
            }
        }

        // circuit is reversed Euler tour
        var outSb = new StringBuilder(cp * 7);
        for (int i = cp - 1; i >= 0; i--) {
            if (i < cp - 1) outSb.Append(' ');
            outSb.Append(circuit[i]);
        }
        sw.WriteLine(outSb.ToString());
        sw.Flush();
    }
}
