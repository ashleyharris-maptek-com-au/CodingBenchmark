using namespace std;

typedef vector<int> Pos;

struct PosHash {
    size_t operator()(const Pos& p) const noexcept {
        size_t h = 1469598103934665603ULL;
        for (int v : p) {
            h ^= (size_t)v + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        }
        return h;
    }
};

static int D;
static long long MAXT;
static vector<int> bounds_;
static unordered_set<Pos, PosHash> obstacles;

inline bool inBounds(const Pos& p) {
    for (int i = 0; i < D; ++i) if (p[i] < 0 || p[i] >= bounds_[i]) return false;
    return true;
}

inline long long manhattan(const Pos& a, const Pos& b) {
    long long s = 0;
    for (int i = 0; i < D; ++i) s += abs(a[i] - b[i]);
    return s;
}

vector<pair<int,int>> astar(const Pos& start, const Pos& goal,
                             const unordered_set<Pos, PosHash>& blocked,
                             int nodeLimit)
{
    if (start == goal) return {};
    struct Node {
        long long f;
        long long g;
        int idx;
    };
    struct Cmp { bool operator()(const Node& a, const Node& b) const { return a.f > b.f; } };
    priority_queue<Node, vector<Node>, Cmp> pq;
    vector<Pos> positions;
    vector<int> parent;
    vector<int> parentAxis;
    vector<int> parentDir;
    unordered_map<Pos, int, PosHash> bestG;

    positions.push_back(start);
    parent.push_back(-1);
    parentAxis.push_back(-1);
    parentDir.push_back(0);
    bestG[start] = 0;
    pq.push({manhattan(start, goal), 0, 0});

    int explored = 0;
    while (!pq.empty() && explored < nodeLimit) {
        Node cur = pq.top(); pq.pop();
        ++explored;
        Pos here = positions[cur.idx];
        if (here == goal) {
            vector<pair<int,int>> moves;
            int i = cur.idx;
            while (parent[i] != -1) {
                moves.push_back({parentAxis[i], parentDir[i]});
                i = parent[i];
            }
            reverse(moves.begin(), moves.end());
            return moves;
        }
        auto itG = bestG.find(here);
        if (itG != bestG.end() && cur.g > itG->second) continue;

        for (int ax = 0; ax < D; ++ax) {
            for (int dr = -1; dr <= 1; dr += 2) {
                Pos np = here;
                np[ax] += dr;
                if (!inBounds(np)) continue;
                if (blocked.find(np) != blocked.end()) continue;
                long long ng = cur.g + 1;
                auto it = bestG.find(np);
                if (it != bestG.end() && (long long)it->second <= ng) continue;
                int nidx = (int)positions.size();
                positions.push_back(np);
                parent.push_back(cur.idx);
                parentAxis.push_back(ax);
                parentDir.push_back(dr);
                bestG[np] = (int)ng;
                long long h = manhattan(np, goal);
                pq.push({ng + h, ng, nidx});
            }
        }
    }
    return {};
}

pair<int,int> greedyStep(const Pos& cur, const Pos& goal,
                         const unordered_set<Pos, PosHash>& blocked)
{
    int bestAx = -1, bestDr = 0;
    long long bestDist = 0;
    bool found = false;
    for (int ax = 0; ax < D; ++ax) {
        for (int dr = -1; dr <= 1; dr += 2) {
            Pos np = cur; np[ax] += dr;
            if (!inBounds(np)) continue;
            if (blocked.find(np) != blocked.end()) continue;
            long long d = manhattan(np, goal);
            if (!found || d < bestDist) { bestDist = d; bestAx = ax; bestDr = dr; found = true; }
        }
    }
    if (!found) {
        for (int ax = 0; ax < D; ++ax) {
            for (int dr = -1; dr <= 1; dr += 2) {
                Pos np = cur; np[ax] += dr;
                if (!inBounds(np)) continue;
                if (blocked.find(np) != blocked.end()) continue;
                return {ax, dr};
            }
        }
        return {0, 1};
    }
    return {bestAx, bestDr};
}

int main() {
    if (!(scanf("%d %lld", &D, &MAXT) == 2)) return 0;
    bounds_.assign(D, 0);
    for (int i = 0; i < D; ++i) scanf("%d", &bounds_[i]);

    int snakeLen;
    scanf("%d", &snakeLen);
    Pos head(D);
    for (int i = 0; i < D; ++i) scanf("%d", &head[i]);

    deque<Pos> body;
    for (int i = 0; i < snakeLen; ++i) body.push_back(head);

    int F; scanf("%d", &F);
    vector<Pos> foods(F, Pos(D));
    for (int i = 0; i < F; ++i)
        for (int j = 0; j < D; ++j) scanf("%d", &foods[i][j]);

    int O; scanf("%d", &O);
    for (int i = 0; i < O; ++i) {
        Pos p(D);
        for (int j = 0; j < D; ++j) scanf("%d", &p[j]);
        obstacles.insert(p);
    }

    unordered_set<Pos, PosHash> foodSet;
    for (auto& f : foods) foodSet.insert(f);

    unordered_map<Pos, int, PosHash> bodyCount;
    for (auto& b : body) bodyCount[b]++;

    auto buildBlocked = [&]() {
        unordered_set<Pos, PosHash> s = obstacles;
        Pos tail = body.back();
        for (auto& kv : bodyCount) {
            if (kv.first == tail && kv.second == 1) continue;
            s.insert(kv.first);
        }
        return s;
    };

    long long turn = 0;

    while (turn < MAXT && !foodSet.empty()) {
        Pos cur = body.front();
        Pos target;
        long long bestD = -1;
        for (auto& f : foodSet) {
            long long d = manhattan(cur, f);
            if (bestD < 0 || d < bestD) { bestD = d; target = f; }
        }

        unordered_set<Pos, PosHash> blocked = buildBlocked();

        int nodeLimit;
        if (D <= 3) nodeLimit = 200000;
        else if (D <= 5) nodeLimit = 60000;
        else if (D <= 8) nodeLimit = 20000;
        else nodeLimit = 6000;

        vector<pair<int,int>> path = astar(cur, target, blocked, nodeLimit);

        pair<int,int> move;
        if (!path.empty()) {
            move = path.front();
        } else {
            move = greedyStep(cur, target, blocked);
        }

        Pos newHead = cur;
        newHead[move.first] += move.second;

        bool dies = false;
        if (!inBounds(newHead)) dies = true;
        else if (obstacles.count(newHead)) dies = true;
        else {
            auto it = bodyCount.find(newHead);
            if (it != bodyCount.end() && it->second > 0) {
                Pos tail = body.back();
                if (!(newHead == tail && it->second == 1)) dies = true;
            }
        }

        if (dies) {
            bool any = false;
            for (int ax = 0; ax < D && !any; ++ax) {
                for (int dr = -1; dr <= 1 && !any; dr += 2) {
                    Pos np = cur; np[ax] += dr;
                    if (!inBounds(np)) continue;
                    if (obstacles.count(np)) continue;
                    auto it = bodyCount.find(np);
                    if (it != bodyCount.end() && it->second > 0) {
                        Pos tail = body.back();
                        if (!(np == tail && it->second == 1)) continue;
                    }
                    move = {ax, dr};
                    newHead = np;
                    any = true;
                }
            }
            if (!any) {
                printf("%d %d\n", move.first, move.second);
                break;
            }
        }

        bool ate = foodSet.count(newHead) > 0;
        body.push_front(newHead);
        bodyCount[newHead]++;
        if (ate) {
            foodSet.erase(newHead);
        } else {
            Pos tail = body.back(); body.pop_back();
            auto it = bodyCount.find(tail);
            if (it != bodyCount.end()) {
                if (--it->second == 0) bodyCount.erase(it);
            }
        }

        printf("%d %d\n", move.first, move.second);
        fflush(stdout);
        ++turn;
    }
    return 0;
}
