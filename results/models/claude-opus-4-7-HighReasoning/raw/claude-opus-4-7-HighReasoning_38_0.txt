import sys

def solve():
    data = sys.stdin.read().split('\n')
    first = data[0].split()
    U, S = int(first[0]), int(first[1])
    masks = [0] * S
    for i in range(S):
        line = data[1 + i].strip() if 1 + i < len(data) else ''
        m = 0
        if line:
            for tok in line.split():
                m |= (1 << int(tok))
        masks[i] = m

    full = (1 << U) - 1

    elem_sets = [[] for _ in range(U)]
    for i, m in enumerate(masks):
        for e in range(U):
            if m & (1 << e):
                elem_sets[e].append(i)

    chosen = []
    result = [None]

    def backtrack(covered):
        if covered == full:
            result[0] = list(chosen)
            return True
        remaining = full & ~covered
        best_e = -1
        best_count = None
        best_candidates = None
        rem = remaining
        while rem:
            low = rem & -rem
            e = low.bit_length() - 1
            cands = [si for si in elem_sets[e] if not (masks[si] & covered)]
            cnt = len(cands)
            if best_count is None or cnt < best_count:
                best_count = cnt
                best_e = e
                best_candidates = cands
                if cnt <= 1:
                    break
            rem ^= low
        if best_count == 0:
            return False
        for si in best_candidates:
            chosen.append(si)
            if backtrack(covered | masks[si]):
                return True
            chosen.pop()
        return False

    if backtrack(0):
        sol = sorted(result[0])
        print("SOLUTION")
        print(" ".join(map(str, sol)))
    else:
        print("NO SOLUTION")

solve()
