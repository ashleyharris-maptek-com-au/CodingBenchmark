use std::io::{self, Read, Write, BufWriter};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let bytes = input.as_bytes();
    let mut i = 0usize;
    let len = bytes.len();

    while i < len && (bytes[i] == b' ' || bytes[i] == b'\r' || bytes[i] == b'\n' || bytes[i] == b'\t') { i += 1; }
    let mut h: usize = 0;
    while i < len && bytes[i].is_ascii_digit() {
        h = h * 10 + (bytes[i] - b'0') as usize;
        i += 1;
    }
    while i < len && (bytes[i] == b' ' || bytes[i] == b'\t') { i += 1; }
    let mut w: usize = 0;
    while i < len && bytes[i].is_ascii_digit() {
        w = w * 10 + (bytes[i] - b'0') as usize;
        i += 1;
    }
    while i < len && bytes[i] != b'\n' { i += 1; }
    if i < len { i += 1; }

    if h == 0 || w == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    let n = h.checked_mul(w).expect("maze too large");
    let mut grid: Vec<u8> = vec![b'#'; n];

    let mut start: usize = usize::MAX;
    let mut goal: usize = usize::MAX;

    for r in 0..h {
        let row_start = r * w;
        let mut c = 0usize;
        while c < w && i < len && bytes[i] != b'\n' && bytes[i] != b'\r' {
            let ch = bytes[i];
            let idx = row_start + c;
            grid[idx] = ch;
            if ch == b'A' { start = idx; }
            else if ch == b'B' { goal = idx; }
            c += 1;
            i += 1;
        }
        while i < len && bytes[i] != b'\n' { i += 1; }
        if i < len { i += 1; }
    }

    if start == usize::MAX || goal == usize::MAX {
        writeln!(out, "0").unwrap();
        return;
    }

    if start == goal {
        let sx = start % w;
        let sy = start / w;
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", sx, sy).unwrap();
        return;
    }

    // BFS. parent_dir: 0=unvisited, 1=came from up, 2=from down, 3=from left, 4=from right, 5=start
    let mut parent_dir: Vec<u8> = vec![0u8; n];
    parent_dir[start] = 5;

    let mut queue: Vec<usize> = Vec::with_capacity(1024);
    queue.push(start);
    let mut head = 0usize;

    let mut found = false;
    'outer: while head < queue.len() {
        let cur = queue[head];
        head += 1;
        let x = cur % w;
        let y = cur / w;

        if y > 0 {
            let nx = cur - w;
            if parent_dir[nx] == 0 && grid[nx] != b'#' {
                parent_dir[nx] = 2;
                if nx == goal { found = true; break 'outer; }
                queue.push(nx);
            }
        }
        if y + 1 < h {
            let nx = cur + w;
            if parent_dir[nx] == 0 && grid[nx] != b'#' {
                parent_dir[nx] = 1;
                if nx == goal { found = true; break 'outer; }
                queue.push(nx);
            }
        }
        if x > 0 {
            let nx = cur - 1;
            if parent_dir[nx] == 0 && grid[nx] != b'#' {
                parent_dir[nx] = 4;
                if nx == goal { found = true; break 'outer; }
                queue.push(nx);
            }
        }
        if x + 1 < w {
            let nx = cur + 1;
            if parent_dir[nx] == 0 && grid[nx] != b'#' {
                parent_dir[nx] = 3;
                if nx == goal { found = true; break 'outer; }
                queue.push(nx);
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut path: Vec<usize> = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start { break; }
        let d = parent_dir[cur];
        cur = match d {
            1 => cur - w,
            2 => cur + w,
            3 => cur - 1,
            4 => cur + 1,
            _ => unreachable!(),
        };
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    let mut buf = [0u8; 24];
    for p in path {
        let x = p % w;
        let y = p / w;
        write_usize(&mut out, &mut buf, x);
        out.write_all(b" ").unwrap();
        write_usize(&mut out, &mut buf, y);
        out.write_all(b"\n").unwrap();
    }
}

fn write_usize<W: Write>(w: &mut W, buf: &mut [u8; 24], mut n: usize) {
    if n == 0 {
        w.write_all(b"0").unwrap();
        return;
    }
    let mut i = buf.len();
    while n > 0 {
        i -= 1;
        buf[i] = b'0' + (n % 10) as u8;
        n /= 10;
    }
    w.write_all(&buf[i..]).unwrap();
}
