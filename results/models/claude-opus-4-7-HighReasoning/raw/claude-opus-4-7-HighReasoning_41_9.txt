; SPIR-V
; Version: 1.0
               OpCapability Shader
       %glsl = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
     %fnType = OpTypeFunction %void
      %float = OpTypeFloat 32
        %v2f = OpTypeVector %float 2
        %v3f = OpTypeVector %float 3
        %v4f = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4f 4
       %bool = OpTypeBool

        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4f %v4f %v4f
   %_ptr_ubo = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_ubo Uniform

 %_ptr_in_v3 = OpTypePointer Input %v3f
 %_ptr_in_v2 = OpTypePointer Input %v2f
%_ptr_out_v4 = OpTypePointer Output %v4f
  %_ptr_u_v4 = OpTypePointer Uniform %v4f

   %worldPos = OpVariable %_ptr_in_v3 Input
     %normal = OpVariable %_ptr_in_v3 Input
         %uv = OpVariable %_ptr_in_v2 Input
    %tangent = OpVariable %_ptr_in_v3 Input
      %color = OpVariable %_ptr_in_v3 Input
   %outColor = OpVariable %_ptr_out_v4 Output

        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_3 = OpConstant %int 3
       %f_0  = OpConstant %float 0.0
       %f_1  = OpConstant %float 1.0
      %f_0_5 = OpConstant %float 0.5
      %f_0_9 = OpConstant %float 0.9
      %f_0_2 = OpConstant %float 0.2
       %f_8  = OpConstant %float 8.0
       %f_2  = OpConstant %float 2.0

       %main = OpFunction %void None %fnType
      %label = OpLabel

       %uvL  = OpLoad %v2f %uv
       %u_   = OpCompositeExtract %float %uvL 0
       %v_   = OpCompositeExtract %float %uvL 1
       %u8   = OpFMul %float %u_ %f_8
       %v8   = OpFMul %float %v_ %f_8
       %uf   = OpExtInst %float %glsl Floor %u8
       %vf   = OpExtInst %float %glsl Floor %v8
       %sum  = OpFAdd %float %uf %vf
       %half = OpFMul %float %sum %f_0_5
      %halfF = OpExtInst %float %glsl Floor %half
       %dbl  = OpFMul %float %halfF %f_2
       %mod2 = OpFSub %float %sum %dbl
      %isOdd = OpFOrdGreaterThan %bool %mod2 %f_0_5

         %cb = OpSelect %float %isOdd %f_0_2 %f_0_9

      %lpPtr = OpAccessChain %_ptr_u_v4 %ubo %int_3
        %lp4 = OpLoad %v4f %lpPtr
        %lp3 = OpVectorShuffle %v3f %lp4 %lp4 0 1 2
         %wp = OpLoad %v3f %worldPos
       %ldir = OpFSub %v3f %lp3 %wp
      %lndir = OpExtInst %v3f %glsl Normalize %ldir
          %n = OpLoad %v3f %normal
         %nn = OpExtInst %v3f %glsl Normalize %n
      %dotNL = OpDot %float %nn %lndir
       %diff = OpExtInst %float %glsl FClamp %dotNL %f_0 %f_1
      %half2 = OpFMul %float %diff %f_0_5
       %lamb = OpFAdd %float %half2 %f_0_5

     %shaded = OpFMul %float %cb %lamb
       %outv = OpCompositeConstruct %v4f %shaded %shaded %shaded %f_1
                OpStore %outColor %outv
                OpReturn
                OpFunctionEnd
