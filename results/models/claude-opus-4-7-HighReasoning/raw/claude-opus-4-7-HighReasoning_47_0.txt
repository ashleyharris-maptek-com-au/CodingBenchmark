@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn ctz32(v : u32) -> u32 {
    var x : u32 = v;
    var n : u32 = 0u;
    for (var i : u32 = 0u; i < 32u; i = i + 1u) {
        if ((x & 1u) != 0u) {
            return n;
        }
        x = x >> 1u;
        n = n + 1u;
    }
    return 32u;
}

fn count_trailing_zeros_64(h0 : u32, h1 : u32) -> u32 {
    if (h0 != 0u) {
        return ctz32(h0);
    }
    if (h1 != 0u) {
        return 32u + ctz32(h1);
    }
    return 64u;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let my_nonce : u32 = params.y + gid.x;

    var state : array<u32, 16>;
    state[0]  = baseData[0]  ^ (my_nonce + 0u);
    state[1]  = baseData[1]  ^ (my_nonce + 1u);
    state[2]  = baseData[2]  ^ (my_nonce + 2u);
    state[3]  = baseData[3]  ^ (my_nonce + 3u);
    state[4]  = baseData[4]  ^ (my_nonce + 4u);
    state[5]  = baseData[5]  ^ (my_nonce + 5u);
    state[6]  = baseData[6]  ^ (my_nonce + 6u);
    state[7]  = baseData[7]  ^ (my_nonce + 7u);
    state[8]  = baseData[8]  ^ (my_nonce + 8u);
    state[9]  = baseData[9]  ^ (my_nonce + 9u);
    state[10] = baseData[10] ^ (my_nonce + 10u);
    state[11] = baseData[11] ^ (my_nonce + 11u);
    state[12] = baseData[12] ^ (my_nonce + 12u);
    state[13] = baseData[13] ^ (my_nonce + 13u);
    state[14] = baseData[14] ^ (my_nonce + 14u);
    state[15] = baseData[15] ^ (my_nonce + 15u);

    for (var r : u32 = 0u; r < 8u; r = r + 1u) {
        // Group 0: state[0..3]
        state[0] = state[0] + state[1]; state[3] = state[3] ^ state[0]; state[3] = rotl(state[3], 16u);
        state[2] = state[2] + state[3]; state[1] = state[1] ^ state[2]; state[1] = rotl(state[1], 12u);
        state[0] = state[0] + state[1]; state[3] = state[3] ^ state[0]; state[3] = rotl(state[3], 8u);
        state[2] = state[2] + state[3]; state[1] = state[1] ^ state[2]; state[1] = rotl(state[1], 7u);

        // Group 1: state[4..7]
        state[4] = state[4] + state[5]; state[7] = state[7] ^ state[4]; state[7] = rotl(state[7], 16u);
        state[6] = state[6] + state[7]; state[5] = state[5] ^ state[6]; state[5] = rotl(state[5], 12u);
        state[4] = state[4] + state[5]; state[7] = state[7] ^ state[4]; state[7] = rotl(state[7], 8u);
        state[6] = state[6] + state[7]; state[5] = state[5] ^ state[6]; state[5] = rotl(state[5], 7u);

        // Group 2: state[8..11]
        state[8]  = state[8]  + state[9];  state[11] = state[11] ^ state[8];  state[11] = rotl(state[11], 16u);
        state[10] = state[10] + state[11]; state[9]  = state[9]  ^ state[10]; state[9]  = rotl(state[9],  12u);
        state[8]  = state[8]  + state[9];  state[11] = state[11] ^ state[8];  state[11] = rotl(state[11], 8u);
        state[10] = state[10] + state[11]; state[9]  = state[9]  ^ state[10]; state[9]  = rotl(state[9],  7u);

        // Group 3: state[12..15]
        state[12] = state[12] + state[13]; state[15] = state[15] ^ state[12]; state[15] = rotl(state[15], 16u);
        state[14] = state[14] + state[15]; state[13] = state[13] ^ state[14]; state[13] = rotl(state[13], 12u);
        state[12] = state[12] + state[13]; state[15] = state[15] ^ state[12]; state[15] = rotl(state[15], 8u);
        state[14] = state[14] + state[15]; state[13] = state[13] ^ state[14]; state[13] = rotl(state[13], 7u);
    }

    let h0 : u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 : u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    let tz : u32 = count_trailing_zeros_64(h0, h1);

    if (tz >= params.x) {
        result[0] = 1u;
        result[1] = my_nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
