use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy)]
struct Pt { x: f64, y: f64 }

fn bbox(pts: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in pts {
        if p.x < minx { minx = p.x; }
        if p.y < miny { miny = p.y; }
        if p.x > maxx { maxx = p.x; }
        if p.y > maxy { maxy = p.y; }
    }
    (minx, miny, maxx, maxy)
}

fn rotate(pts: &[Pt], ang: f64) -> Vec<Pt> {
    let c = ang.cos();
    let s = ang.sin();
    pts.iter().map(|p| Pt { x: p.x * c - p.y * s, y: p.x * s + p.y * c }).collect()
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_ascii_whitespace();
    let mut next_f = || -> f64 { it.next().unwrap().parse().unwrap() };

    let sv: usize = next_f() as usize;
    let np: usize = next_f() as usize;

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = next_f();
        let y = next_f();
        stock.push(Pt { x, y });
    }
    let (sminx, sminy, smaxx, smaxy) = bbox(&stock);
    let sw = smaxx - sminx;
    let sh = smaxy - sminy;

    let mut pieces: Vec<Vec<Pt>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = next_f() as usize;
        let mut p = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = next_f();
            let y = next_f();
            p.push(Pt { x, y });
        }
        pieces.push(p);
    }

    // Adapt rotation search based on scale
    let rot_count: usize = if np <= 50 { 12 } else if np <= 500 { 6 } else if np <= 2000 { 3 } else { 1 };

    struct PInfo {
        idx: usize,
        rot: f64,
        w: f64,
        h: f64,
        minx: f64,
        miny: f64,
    }

    let mut infos: Vec<PInfo> = Vec::with_capacity(np);
    for (i, piece) in pieces.iter().enumerate() {
        let mut best_area = f64::INFINITY;
        let mut best_rot = 0.0;
        let mut best_w = 0.0;
        let mut best_h = 0.0;
        let mut best_mx = 0.0;
        let mut best_my = 0.0;
        let rc = rot_count.max(1);
        for r in 0..rc {
            let ang = std::f64::consts::PI * (r as f64) / (rc as f64);
            let rp = rotate(piece, ang);
            let (mnx, mny, mxx, mxy) = bbox(&rp);
            let w = mxx - mnx;
            let h = mxy - mny;
            let a = w * h;
            if a < best_area {
                best_area = a;
                best_rot = ang;
                best_w = w;
                best_h = h;
                best_mx = mnx;
                best_my = mny;
            }
        }
        // Also try orienting longer axis horizontally for better shelf packing
        if best_w < best_h {
            // rotate by 90 degrees additionally
            let ang = best_rot + std::f64::consts::FRAC_PI_2;
            let rp = rotate(piece, ang);
            let (mnx, mny, mxx, mxy) = bbox(&rp);
            best_rot = ang;
            best_w = mxx - mnx;
            best_h = mxy - mny;
            best_mx = mnx;
            best_my = mny;
        }
        infos.push(PInfo {
            idx: i,
            rot: best_rot,
            w: best_w,
            h: best_h,
            minx: best_mx,
            miny: best_my,
        });
    }

    // Sort by height descending for shelf packing
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| infos[b].h.partial_cmp(&infos[a].h).unwrap_or(std::cmp::Ordering::Equal));

    struct Place { stock: usize, tx: f64, ty: f64, rot: f64 }
    let mut placements: Vec<Place> = (0..np).map(|_| Place { stock: 0, tx: 0.0, ty: 0.0, rot: 0.0 }).collect();

    struct StockState { shelf_y: f64, shelf_h: f64, cursor_x: f64 }
    let mut stocks: Vec<StockState> = Vec::new();
    let mut num_stocks: usize = 0;

    for &oi in &order {
        let info = &infos[oi];
        let w = info.w;
        let h = info.h;
        let eps = 1e-9;

        let mut placed = false;
        for (si, st) in stocks.iter_mut().enumerate() {
            if st.cursor_x + w <= sw + eps && h <= st.shelf_h + eps && st.shelf_y + h <= sh + eps {
                let tx = sminx + st.cursor_x - info.minx;
                let ty = sminy + st.shelf_y - info.miny;
                placements[info.idx] = Place { stock: si, tx, ty, rot: info.rot };
                st.cursor_x += w;
                placed = true;
                break;
            }
            let new_shelf_y = st.shelf_y + st.shelf_h;
            if w <= sw + eps && new_shelf_y + h <= sh + eps {
                let tx = sminx + 0.0 - info.minx;
                let ty = sminy + new_shelf_y - info.miny;
                placements[info.idx] = Place { stock: si, tx, ty, rot: info.rot };
                st.shelf_y = new_shelf_y;
                st.shelf_h = h;
                st.cursor_x = w;
                placed = true;
                break;
            }
        }
        if !placed {
            let si = num_stocks;
            num_stocks += 1;
            let tx = sminx + 0.0 - info.minx;
            let ty = sminy + 0.0 - info.miny;
            placements[info.idx] = Place { stock: si, tx, ty, rot: info.rot };
            stocks.push(StockState { shelf_y: 0.0, shelf_h: h, cursor_x: w });
        }
    }

    if num_stocks == 0 { num_stocks = 1; }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", num_stocks).unwrap();
    for p in &placements {
        writeln!(out, "{} {} {} {}", p.stock, p.tx, p.ty, p.rot).unwrap();
    }
}
