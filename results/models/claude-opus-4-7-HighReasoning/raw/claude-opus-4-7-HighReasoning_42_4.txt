; SPIR-V
; Version: 1.0
               OpCapability Geometry
       %ext = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %inNormal %inWorldPos %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal   Location 1
               OpDecorate %inColor    Location 2

               OpDecorate %outColor Location 0

       %void = OpTypeVoid
     %fn_void = OpTypeFunction %void
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

       %uint_2 = OpConstant %uint 2

       %int_0 = OpConstant %int 0
       %int_1 = OpConstant %int 1

       %f_thick_start = OpConstant %float 0.04
       %f_thick_end   = OpConstant %float 0.005

       %gl_PerVertex_in = OpTypeStruct %v4float
       %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
       %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
       %gl_in = OpVariable %ptr_in_arr_pv Input
       %ptr_in_v4 = OpTypePointer Input %v4float

       %gl_PerVertex_out = OpTypeStruct %v4float
       %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
       %gl_out = OpVariable %ptr_out_pv Output
       %ptr_out_v4 = OpTypePointer Output %v4float

       %arr_v3 = OpTypeArray %v3float %uint_2
       %ptr_in_arr_v3 = OpTypePointer Input %arr_v3
       %ptr_in_v3 = OpTypePointer Input %v3float

       %inWorldPos = OpVariable %ptr_in_arr_v3 Input
       %inNormal   = OpVariable %ptr_in_arr_v3 Input
       %inColor    = OpVariable %ptr_in_arr_v3 Input

       %ptr_out_v3 = OpTypePointer Output %v3float
       %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %fn_void
       %entry = OpLabel

       %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
       %p0 = OpLoad %v4float %p0_ptr
       %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
       %p1 = OpLoad %v4float %p1_ptr

       %c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
       %c0 = OpLoad %v3float %c0_ptr
       %c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
       %c1 = OpLoad %v3float %c1_ptr

       %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
       %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
       %dir = OpFSub %v2float %p1_xy %p0_xy
       %dir_norm = OpExtInst %v2float %ext Normalize %dir
       %dx = OpCompositeExtract %float %dir_norm 0
       %dy = OpCompositeExtract %float %dir_norm 1
       %neg_dy = OpFNegate %float %dy
       %perp = OpCompositeConstruct %v2float %neg_dy %dx

       %off0 = OpVectorTimesScalar %v2float %perp %f_thick_start
       %off1 = OpVectorTimesScalar %v2float %perp %f_thick_end

       %off0_x = OpCompositeExtract %float %off0 0
       %off0_y = OpCompositeExtract %float %off0 1
       %off1_x = OpCompositeExtract %float %off1 0
       %off1_y = OpCompositeExtract %float %off1 1

       %p0_x = OpCompositeExtract %float %p0 0
       %p0_y = OpCompositeExtract %float %p0 1
       %p0_z = OpCompositeExtract %float %p0 2
       %p0_w = OpCompositeExtract %float %p0 3
       %p1_x = OpCompositeExtract %float %p1 0
       %p1_y = OpCompositeExtract %float %p1 1
       %p1_z = OpCompositeExtract %float %p1 2
       %p1_w = OpCompositeExtract %float %p1 3

       %v0x = OpFAdd %float %p0_x %off0_x
       %v0y = OpFAdd %float %p0_y %off0_y
       %v0 = OpCompositeConstruct %v4float %v0x %v0y %p0_z %p0_w

       %v1x = OpFSub %float %p0_x %off0_x
       %v1y = OpFSub %float %p0_y %off0_y
       %v1 = OpCompositeConstruct %v4float %v1x %v1y %p0_z %p0_w

       %v2x = OpFAdd %float %p1_x %off1_x
       %v2y = OpFAdd %float %p1_y %off1_y
       %v2v = OpCompositeConstruct %v4float %v2x %v2y %p1_z %p1_w

       %v3x = OpFSub %float %p1_x %off1_x
       %v3y = OpFSub %float %p1_y %off1_y
       %v3v = OpCompositeConstruct %v4float %v3x %v3y %p1_z %p1_w

       %gl_out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %gl_out_pos0 %v0
       OpStore %outColor %c0
       OpEmitVertex

       %gl_out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %gl_out_pos1 %v1
       OpStore %outColor %c0
       OpEmitVertex

       %gl_out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %gl_out_pos2 %v2v
       OpStore %outColor %c1
       OpEmitVertex

       %gl_out_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
       OpStore %gl_out_pos3 %v3v
       OpStore %outColor %c1
       OpEmitVertex

       OpEndPrimitive
       OpReturn
       OpFunctionEnd
