#include <iostream>
#include <vector>
#include <string>
#include <string_view>
#include <algorithm>
#include <unordered_set>
#include <unordered_map>
#include <cstdint>
#include <chrono>

using namespace std;

// Two 32-bit prime mods so multiplications fit in 64-bit cleanly.
static constexpr uint64_t MOD1 = 1000000007ULL;
static constexpr uint64_t MOD2 = 998244353ULL;

struct HashPair {
    uint32_t h1, h2;
    bool operator==(const HashPair& o) const { return h1 == o.h1 && h2 == o.h2; }
};
struct HashPairHasher {
    size_t operator()(const HashPair& p) const noexcept {
        return ((size_t)p.h1 * 1469598103934665603ULL) ^ ((size_t)p.h2 * 11400714819323198485ULL);
    }
};

// Compute hash table: hash -> earliest index in s, for substrings of length L.
static void compute_hashes_map(const string& s, int L,
                                unordered_map<HashPair, int, HashPairHasher>& out,
                                uint64_t B1, uint64_t B2) {
    out.clear();
    int n = (int)s.size();
    if (L <= 0 || L > n) return;
    out.reserve((size_t)(n - L + 1) * 2);

    uint64_t p1 = 1, p2 = 1;
    for (int i = 0; i < L; ++i) {
        p1 = (p1 * B1) % MOD1;
        p2 = (p2 * B2) % MOD2;
    }
    uint64_t h1 = 0, h2 = 0;
    for (int i = 0; i < L; ++i) {
        h1 = (h1 * B1 + (uint8_t)s[i] + 1) % MOD1;
        h2 = (h2 * B2 + (uint8_t)s[i] + 1) % MOD2;
    }
    out.emplace(HashPair{(uint32_t)h1, (uint32_t)h2}, 0);
    for (int i = L; i < n; ++i) {
        h1 = (h1 * B1 + (uint8_t)s[i] + 1) % MOD1;
        uint64_t sub1 = (p1 * ((uint8_t)s[i - L] + 1)) % MOD1;
        h1 = (h1 + MOD1 - sub1) % MOD1;
        h2 = (h2 * B2 + (uint8_t)s[i] + 1) % MOD2;
        uint64_t sub2 = (p2 * ((uint8_t)s[i - L] + 1)) % MOD2;
        h2 = (h2 + MOD2 - sub2) % MOD2;
        int start = i - L + 1;
        HashPair key{(uint32_t)h1, (uint32_t)h2};
        auto it = out.find(key);
        if (it == out.end()) out.emplace(key, start);
    }
}

static void compute_hashes_set(const string& s, int L,
                                unordered_set<HashPair, HashPairHasher>& out,
                                uint64_t B1, uint64_t B2) {
    out.clear();
    int n = (int)s.size();
    if (L <= 0 || L > n) return;
    out.reserve((size_t)(n - L + 1) * 2);
    uint64_t p1 = 1, p2 = 1;
    for (int i = 0; i < L; ++i) {
        p1 = (p1 * B1) % MOD1;
        p2 = (p2 * B2) % MOD2;
    }
    uint64_t h1 = 0, h2 = 0;
    for (int i = 0; i < L; ++i) {
        h1 = (h1 * B1 + (uint8_t)s[i] + 1) % MOD1;
        h2 = (h2 * B2 + (uint8_t)s[i] + 1) % MOD2;
    }
    out.insert(HashPair{(uint32_t)h1, (uint32_t)h2});
    for (int i = L; i < n; ++i) {
        h1 = (h1 * B1 + (uint8_t)s[i] + 1) % MOD1;
        uint64_t sub1 = (p1 * ((uint8_t)s[i - L] + 1)) % MOD1;
        h1 = (h1 + MOD1 - sub1) % MOD1;
        h2 = (h2 * B2 + (uint8_t)s[i] + 1) % MOD2;
        uint64_t sub2 = (p2 * ((uint8_t)s[i - L] + 1)) % MOD2;
        h2 = (h2 + MOD2 - sub2) % MOD2;
        out.insert(HashPair{(uint32_t)h1, (uint32_t)h2});
    }
}

// Check whether a common substring of length L exists across all strings.
// Returns position in strs[0] if found; -1 otherwise.
static int has_common_length(const vector<string>& strs, int L,
                              uint64_t B1, uint64_t B2) {
    int n = (int)strs.size();
    // Pick shortest as base for candidate enumeration
    int baseIdx = 0;
    for (int i = 1; i < n; ++i) if (strs[i].size() < strs[baseIdx].size()) baseIdx = i;

    unordered_map<HashPair, int, HashPairHasher> baseMap;
    compute_hashes_map(strs[baseIdx], L, baseMap, B1, B2);
    if (baseMap.empty()) return -1;

    for (int i = 0; i < n; ++i) {
        if (i == baseIdx) continue;
        if ((int)strs[i].size() < L) return -1;
        unordered_set<HashPair, HashPairHasher> other;
        compute_hashes_set(strs[i], L, other, B1, B2);
        for (auto it = baseMap.begin(); it != baseMap.end(); ) {
            if (other.find(it->first) == other.end()) it = baseMap.erase(it);
            else ++it;
        }
        if (baseMap.empty()) return -1;
    }

    // Verify candidates via direct find (avoid hash collisions).
    for (auto& kv : baseMap) {
        int start = kv.second;
        string_view cand(strs[baseIdx].data() + start, L);
        bool ok = true;
        size_t posIn0 = string::npos;
        for (int i = 0; i < n; ++i) {
            if (i == baseIdx) { if (i == 0) posIn0 = start; continue; }
            size_t p = strs[i].find(cand);
            if (p == string::npos) { ok = false; break; }
            if (i == 0) posIn0 = p;
        }
        if (ok) {
            if (posIn0 == string::npos) posIn0 = strs[0].find(cand);
            if (posIn0 != string::npos) return (int)posIn0;
        }
    }
    return -1;
}

// Brute solver for small inputs.
static string solve_brute(const vector<string>& strs) {
    int n = (int)strs.size();
    int baseIdx = 0;
    for (int i = 1; i < n; ++i) if (strs[i].size() < strs[baseIdx].size()) baseIdx = i;
    const string& base = strs[baseIdx];
    int blen = (int)base.size();
    string best;
    for (int L = blen; L > (int)best.size(); --L) {
        bool foundAny = false;
        for (int i = 0; i + L <= blen; ++i) {
            string_view cand(base.data() + i, L);
            bool ok = true;
            for (int k = 0; k < n; ++k) {
                if (k == baseIdx) continue;
                if (strs[k].find(cand) == string::npos) { ok = false; break; }
            }
            if (ok) { best = string(cand); foundAny = true; break; }
        }
        if (foundAny) break;
    }
    return best;
}

// DP for two strings (longest common substring).
static string lcs_two_dp(const string& a, const string& b) {
    int n = (int)a.size(), m = (int)b.size();
    if (n == 0 || m == 0) return "";
    vector<int> prev(m + 1, 0), cur(m + 1, 0);
    int best = 0, bestEnd = 0;
    for (int i = 1; i <= n; ++i) {
        for (int j = 1; j <= m; ++j) {
            if (a[i-1] == b[j-1]) {
                cur[j] = prev[j-1] + 1;
                if (cur[j] > best) { best = cur[j]; bestEnd = i; }
            } else cur[j] = 0;
        }
        swap(prev, cur);
        fill(cur.begin(), cur.end(), 0);
    }
    return a.substr(bestEnd - best, best);
}

string solve(const vector<string>& strs) {
    int n = (int)strs.size();
    if (n == 0) return "";
    if (n == 1) return strs[0];

    size_t minLen = strs[0].size();
    size_t totalSize = 0;
    for (auto& s : strs) { minLen = min(minLen, s.size()); totalSize += s.size(); }
    if (minLen == 0) return "";

    // Small problem: brute force
    if (minLen <= 100 && totalSize <= 3000) {
        return solve_brute(strs);
    }

    // Two-string medium: DP
    if (n == 2 && (size_t)strs[0].size() * (size_t)strs[1].size() <= 100000000ULL) {
        // O(n*m) memory-light DP
        return lcs_two_dp(strs[0], strs[1]);
    }

    // Pick bases
    uint64_t B1 = 131, B2 = 137;
    {
        uint64_t seed = 0x9E3779B97F4A7C15ULL ^ (uint64_t)totalSize ^ ((uint64_t)n * 1000003ULL);
        B1 = 1000 + (seed % 100000);
        B2 = 1000 + ((seed >> 17) % 100000);
        if (B1 == B2) B2 ^= 31;
    }

    int lo = 1, hi = (int)minLen;
    int bestL = 0;
    int bestStart = 0;

    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        int pos = has_common_length(strs, mid, B1, B2);
        if (pos >= 0) {
            bestL = mid;
            bestStart = pos;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    if (bestL == 0) return "";
    return strs[0].substr(bestStart, bestL);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) { cout << "\n"; return 0; }
    cin.ignore();
    vector<string> strs(N);
    for (int i = 0; i < N; ++i) {
        getline(cin, strs[i]);
    }

    string ans = solve(strs);
    cout << ans << "\n";
    return 0;
}
