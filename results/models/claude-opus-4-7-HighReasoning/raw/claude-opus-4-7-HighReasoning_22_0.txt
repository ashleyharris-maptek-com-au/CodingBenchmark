using System;
using System.IO;
using System.Text;

class TetrisPlayer
{
    static int W, H;
    static int[] board;
    static int fullMask;
    static int[][][] pieces;

    static void Main()
    {
        InitPieces();
        var input = Console.In.ReadToEnd();
        var tokens = input.Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int ti = 0;
        W = int.Parse(tokens[ti++]);
        H = int.Parse(tokens[ti++]);
        int N = int.Parse(tokens[ti++]);

        board = new int[H];
        fullMask = (1 << W) - 1;

        var stdout = Console.Out;

        for (int p = 0; p < N; p++)
        {
            if (ti >= tokens.Length) break;
            string pieceStr = tokens[ti++];
            char pc = pieceStr[0];
            int pi = PieceIndex(pc);

            int fill = CountFill();
            double fillRatio = (double)fill / (W * H);

            var (bestRot, bestCol, bestLanding) = ChooseBest(pi, fillRatio);

            stdout.Write(bestRot);
            stdout.Write(' ');
            stdout.Write(bestCol);
            stdout.Write('\n');
            stdout.Flush();

            int cleared = 0;
            if (bestLanding != null)
            {
                ApplyLanding(bestLanding, out cleared);
            }

            if (ti < tokens.Length)
            {
                string result = tokens[ti++];
                if (result.StartsWith("gameover")) break;
            }
        }
    }

    static int CountFill()
    {
        int c = 0;
        for (int r = 0; r < H; r++)
        {
            int x = board[r];
            while (x != 0) { x &= x - 1; c++; }
        }
        return c;
    }

    static int PieceIndex(char c)
    {
        switch (c)
        {
            case 'I': return 0;
            case 'O': return 1;
            case 'T': return 2;
            case 'S': return 3;
            case 'Z': return 4;
            case 'J': return 5;
            case 'L': return 6;
        }
        return 0;
    }

    static void InitPieces()
    {
        pieces = new int[7][][];

        var I0 = new int[][] { new[] { 0, 0 }, new[] { 0, 1 }, new[] { 0, 2 }, new[] { 0, 3 } };
        pieces[0] = BuildRotations(I0);

        var O0 = new int[][] { new[] { 0, 0 }, new[] { 0, 1 }, new[] { 1, 0 }, new[] { 1, 1 } };
        pieces[1] = BuildRotations(O0);

        var T0 = new int[][] { new[] { 0, 0 }, new[] { 0, 1 }, new[] { 0, 2 }, new[] { 1, 1 } };
        pieces[2] = BuildRotations(T0);

        var S0 = new int[][] { new[] { 0, 1 }, new[] { 0, 2 }, new[] { 1, 0 }, new[] { 1, 1 } };
        pieces[3] = BuildRotations(S0);

        var Z0 = new int[][] { new[] { 0, 0 }, new[] { 0, 1 }, new[] { 1, 1 }, new[] { 1, 2 } };
        pieces[4] = BuildRotations(Z0);

        var J0 = new int[][] { new[] { 0, 0 }, new[] { 1, 0 }, new[] { 1, 1 }, new[] { 1, 2 } };
        pieces[5] = BuildRotations(J0);

        var L0 = new int[][] { new[] { 0, 2 }, new[] { 1, 0 }, new[] { 1, 1 }, new[] { 1, 2 } };
        pieces[6] = BuildRotations(L0);
    }

    static int[][] BuildRotations(int[][] cells0)
    {
        var rots = new int[4][];
        var cur = cells0;
        for (int i = 0; i < 4; i++)
        {
            cur = Normalize(cur);
            var flat = new int[cur.Length * 2];
            for (int j = 0; j < cur.Length; j++)
            {
                flat[j * 2] = cur[j][0];
                flat[j * 2 + 1] = cur[j][1];
            }
            rots[i] = flat;
            var next = new int[cur.Length][];
            for (int j = 0; j < cur.Length; j++)
            {
                next[j] = new int[] { cur[j][1], -cur[j][0] };
            }
            cur = next;
        }
        return rots;
    }

    static int[][] Normalize(int[][] cells)
    {
        int minR = int.MaxValue, minC = int.MaxValue;
        for (int i = 0; i < cells.Length; i++)
        {
            if (cells[i][0] < minR) minR = cells[i][0];
            if (cells[i][1] < minC) minC = cells[i][1];
        }
        var result = new int[cells.Length][];
        for (int i = 0; i < cells.Length; i++)
        {
            result[i] = new int[] { cells[i][0] - minR, cells[i][1] - minC };
        }
        return result;
    }

    static (int rot, int col, int[][] landing) ChooseBest(int pi, double fillRatio)
    {
        int bestRot = 0, bestCol = 0;
        double bestScore = double.NegativeInfinity;
        int[][] bestLanding = null;

        var rots = pieces[pi];
        int numRots = 4;

        for (int r = 0; r < numRots; r++)
        {
            var flat = rots[r];
            int n = flat.Length / 2;
            int maxC = 0;
            for (int i = 0; i < n; i++) if (flat[i * 2 + 1] > maxC) maxC = flat[i * 2 + 1];
            int pw = maxC + 1;

            for (int col = 0; col + pw <= W; col++)
            {
                int landingTop = FindLanding(flat, n, col);
                if (landingTop < 0) continue;

                var newBoard = (int[])board.Clone();
                var landingCells = new int[n][];
                bool overflow = false;
                for (int i = 0; i < n; i++)
                {
                    int rr = landingTop + flat[i * 2];
                    int cc = col + flat[i * 2 + 1];
                    if (rr < 0) { overflow = true; break; }
                    newBoard[rr] |= (1 << cc);
                    landingCells[i] = new int[] { rr, cc };
                }
                if (overflow) continue;

                int cleared = 0;
                var clearedBoard = ClearLines(newBoard, out cleared);

                double score = Evaluate(clearedBoard, cleared, fillRatio);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = r;
                    bestCol = col;
                    bestLanding = landingCells;
                }
            }
        }

        if (bestLanding == null)
        {
            bestRot = 0;
            bestCol = 0;
        }

        return (bestRot, bestCol, bestLanding);
    }

    static int FindLanding(int[] flat, int n, int col)
    {
        int maxDr = 0;
        for (int i = 0; i < n; i++) if (flat[i * 2] > maxDr) maxDr = flat[i * 2];

        int maxTopR = int.MinValue;
        for (int topR = -maxDr; topR <= H - 1 - maxDr; topR++)
        {
            bool ok = true;
            for (int i = 0; i < n; i++)
            {
                int rr = topR + flat[i * 2];
                int cc = col + flat[i * 2 + 1];
                if (rr < 0) continue;
                if ((board[rr] & (1 << cc)) != 0) { ok = false; break; }
            }
            if (!ok) continue;
            bool blocked = false;
            for (int i = 0; i < n; i++)
            {
                int rr = topR + flat[i * 2] + 1;
                int cc = col + flat[i * 2 + 1];
                if (rr >= H) { blocked = true; break; }
                if (rr < 0) continue;
                if ((board[rr] & (1 << cc)) != 0) { blocked = true; break; }
            }
            if (blocked) maxTopR = topR;
        }

        if (maxTopR == int.MinValue) return -1;
        for (int i = 0; i < n; i++)
        {
            int rr = maxTopR + flat[i * 2];
            if (rr < 0) return -1;
        }
        return maxTopR;
    }

    static int[] ClearLines(int[] b, out int cleared)
    {
        cleared = 0;
        var result = new int[H];
        int writeRow = H - 1;
        for (int r = H - 1; r >= 0; r--)
        {
            if (b[r] == fullMask) cleared++;
            else { if (writeRow >= 0) result[writeRow--] = b[r]; }
        }
        while (writeRow >= 0) result[writeRow--] = 0;
        return result;
    }

    static void ApplyLanding(int[][] landing, out int cleared)
    {
        for (int i = 0; i < landing.Length; i++)
        {
            board[landing[i][0]] |= (1 << landing[i][1]);
        }
        board = ClearLines(board, out cleared);
    }

    static double Evaluate(int[] b, int cleared, double fillRatio)
    {
        var heights = new int[W];
        for (int c = 0; c < W; c++)
        {
            heights[c] = 0;
            for (int r = 0; r < H; r++)
            {
                if ((b[r] & (1 << c)) != 0) { heights[c] = H - r; break; }
            }
        }

        int aggregate = 0, maxH = 0;
        for (int c = 0; c < W; c++) { aggregate += heights[c]; if (heights[c] > maxH) maxH = heights[c]; }

        int holes = 0;
        for (int c = 0; c < W; c++)
        {
            bool seen = false;
            for (int r = 0; r < H; r++)
            {
                bool filled = (b[r] & (1 << c)) != 0;
                if (filled) seen = true;
                else if (seen) holes++;
            }
        }

        int bumpiness = 0;
        for (int c = 0; c < W - 1; c++) bumpiness += Math.Abs(heights[c] - heights[c + 1]);

        int rowTrans = 0;
        for (int r = 0; r < H; r++)
        {
            int row = b[r];
            int prev = 1;
            for (int c = 0; c < W; c++)
            {
                int cell = (row >> c) & 1;
                if (cell != prev) rowTrans++;
                prev = cell;
            }
            if (prev != 1) rowTrans++;
        }

        int colTrans = 0;
        for (int c = 0; c < W; c++)
        {
            int prev = 0;
            for (int r = 0; r < H; r++)
            {
                int cell = (b[r] >> c) & 1;
                if (cell != prev) colTrans++;
                prev = cell;
            }
            if (prev != 1) colTrans++;
        }

        int wells = 0;
        for (int c = 0; c < W; c++)
        {
            int left = c == 0 ? H : heights[c - 1];
            int right = c == W - 1 ? H : heights[c + 1];
            int minNeighbor = Math.Min(left, right);
            if (minNeighbor > heights[c])
            {
                int depth = minNeighbor - heights[c];
                wells += depth * (depth + 1) / 2;
            }
        }

        double score = 0;
        score += cleared * 3.4181;
        score -= aggregate * 0.510066;
        score -= holes * 2.5;
        score -= bumpiness * 0.184483;
        score -= rowTrans * 0.32;
        score -= colTrans * 0.93;
        score -= wells * 0.35;

        if (maxH > H * 0.75) score -= (maxH - H * 0.75) * 2.0;

        return score;
    }
}
