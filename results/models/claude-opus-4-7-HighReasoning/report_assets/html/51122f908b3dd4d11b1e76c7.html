using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

class MaxClique
{
    static int N;
    static ulong[][] adj;
    static int W;
    static int[] degree;
    static int[] coreNum;
    static int[] rowStart;
    static int[] adjList;

    static volatile int bestSize = 0;
    static int[] bestClique = Array.Empty<int>();
    static object bestLock = new object();

    static Stopwatch sw = new Stopwatch();
    static long timeLimitMs = 4 * 60 * 1000 + 30 * 1000;
    static volatile bool stopFlag = false;

    static void UpdateBest(int[] clique, int sz)
    {
        if (sz > bestSize)
        {
            lock (bestLock)
            {
                if (sz > bestSize)
                {
                    bestSize = sz;
                    var copy = new int[sz];
                    Array.Copy(clique, copy, sz);
                    bestClique = copy;
                }
            }
        }
    }

    static void Main()
    {
        sw.Start();
        var input = Console.In.ReadToEnd();
        int idx = 0;
        int ReadInt()
        {
            while (idx < input.Length && (input[idx] < '0' || input[idx] > '9') && input[idx] != '-') idx++;
            int sign = 1;
            if (idx < input.Length && input[idx] == '-') { sign = -1; idx++; }
            int v = 0;
            while (idx < input.Length && input[idx] >= '0' && input[idx] <= '9')
            { v = v * 10 + (input[idx] - '0'); idx++; }
            return v * sign;
        }

        N = ReadInt();
        int M = ReadInt();
        if (N <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        W = (N + 63) >> 6;

        var degCount = new int[N];
        var edgesU = new int[M];
        var edgesV = new int[M];
        for (int i = 0; i < M; i++)
        {
            int u = ReadInt();
            int v = ReadInt();
            if (u == v || u < 0 || v < 0 || u >= N || v >= N) { edgesU[i] = -1; continue; }
            edgesU[i] = u;
            edgesV[i] = v;
            degCount[u]++;
            degCount[v]++;
        }

        rowStart = new int[N + 1];
        for (int i = 0; i < N; i++) rowStart[i + 1] = rowStart[i] + degCount[i];
        var rowPos = new int[N];
        Array.Copy(rowStart, rowPos, N);
        adjList = new int[rowStart[N]];
        for (int i = 0; i < M; i++)
        {
            int u = edgesU[i];
            if (u < 0) continue;
            int v = edgesV[i];
            adjList[rowPos[u]++] = v;
            adjList[rowPos[v]++] = u;
        }
        for (int i = 0; i < N; i++)
        {
            int s = rowStart[i], e = rowStart[i + 1];
            if (e - s > 1)
            {
                Array.Sort(adjList, s, e - s);
                int w = s;
                int prev = -1;
                for (int r = s; r < e; r++)
                {
                    int x = adjList[r];
                    if (x != prev && x != i) { adjList[w++] = x; prev = x; }
                }
                degCount[i] = w - s;
                for (int k = w; k < e; k++) adjList[k] = -1;
            }
            else if (e - s == 1)
            {
                if (adjList[s] == i) { adjList[s] = -1; degCount[i] = 0; }
            }
        }
        var rs2 = new int[N + 1];
        for (int i = 0; i < N; i++) rs2[i + 1] = rs2[i] + degCount[i];
        var adj2 = new int[rs2[N]];
        for (int i = 0; i < N; i++)
        {
            int s = rowStart[i], e = rowStart[i + 1];
            int w = rs2[i];
            for (int k = s; k < e; k++)
            {
                if (adjList[k] >= 0) adj2[w++] = adjList[k];
            }
        }
        rowStart = rs2;
        adjList = adj2;
        degree = degCount;

        var order = new int[N];
        var pos = new int[N];
        var degCur = new int[N];
        Array.Copy(degree, degCur, N);
        int maxDeg = 0;
        for (int i = 0; i < N; i++) if (degCur[i] > maxDeg) maxDeg = degCur[i];

        var bucketStart = new int[maxDeg + 2];
        for (int i = 0; i < N; i++) bucketStart[degCur[i] + 1]++;
        for (int i = 1; i <= maxDeg + 1; i++) bucketStart[i] += bucketStart[i - 1];
        var vert = new int[N];
        var bptr = new int[maxDeg + 2];
        Array.Copy(bucketStart, bptr, maxDeg + 2);
        for (int i = 0; i < N; i++)
        {
            vert[bptr[degCur[i]]] = i;
            pos[i] = bptr[degCur[i]];
            bptr[degCur[i]]++;
        }

        coreNum = new int[N];
        int degeneracy = 0;
        for (int i = 0; i < N; i++)
        {
            int v = vert[i];
            coreNum[v] = degCur[v];
            if (degCur[v] > degeneracy) degeneracy = degCur[v];
            order[i] = v;
            int s = rowStart[v], e = rowStart[v + 1];
            for (int k = s; k < e; k++)
            {
                int u = adjList[k];
                if (degCur[u] > degCur[v])
                {
                    int du = degCur[u];
                    int pu = pos[u];
                    int pw = bucketStart[du];
                    int w = vert[pw];
                    if (u != w)
                    {
                        vert[pu] = w; pos[w] = pu;
                        vert[pw] = u; pos[u] = pw;
                    }
                    bucketStart[du]++;
                    degCur[u]--;
                }
            }
        }

        GreedyHeuristic();

        long bitsetMem = (long)N * W * 8L;
        bool useBitset = bitsetMem < 1_500_000_000L;

        var posInOrder = new int[N];
        for (int i = 0; i < N; i++) posInOrder[order[i]] = i;

        if (useBitset)
        {
            adj = new ulong[N][];
            for (int i = 0; i < N; i++) adj[i] = new ulong[W];
            for (int u = 0; u < N; u++)
            {
                int s = rowStart[u], e = rowStart[u + 1];
                var row = adj[u];
                for (int k = s; k < e; k++)
                {
                    int v = adjList[k];
                    row[v >> 6] |= 1UL << (v & 63);
                }
            }

            int threads = Math.Max(1, Environment.ProcessorCount - 1);
            var workOrder = new int[N];
            for (int i = 0; i < N; i++) workOrder[i] = order[N - 1 - i];

            var po = new ParallelOptions { MaxDegreeOfParallelism = threads };

            try
            {
                Parallel.ForEach(workOrder, po, v =>
                {
                    if (stopFlag) return;
                    if (sw.ElapsedMilliseconds > timeLimitMs) { stopFlag = true; return; }
                    if (coreNum[v] + 1 <= bestSize) return;

                    int pv = posInOrder[v];
                    var P = new ulong[W];
                    int s2 = rowStart[v], e2 = rowStart[v + 1];
                    for (int k = s2; k < e2; k++)
                    {
                        int u = adjList[k];
                        if (posInOrder[u] > pv)
                            P[u >> 6] |= 1UL << (u & 63);
                    }
                    var R = new int[degeneracy + 2];
                    R[0] = v;
                    BronKerbosch(R, 1, P);
                });
            }
            catch { }
        }
        else
        {
            int threads = Math.Max(1, Environment.ProcessorCount - 1);
            var workOrder = new int[N];
            for (int i = 0; i < N; i++) workOrder[i] = order[N - 1 - i];

            var po = new ParallelOptions { MaxDegreeOfParallelism = threads };
            try
            {
                Parallel.ForEach(workOrder, po, v =>
                {
                    if (stopFlag) return;
                    if (sw.ElapsedMilliseconds > timeLimitMs) { stopFlag = true; return; }
                    if (coreNum[v] + 1 <= bestSize) return;

                    int pv = posInOrder[v];
                    int s2 = rowStart[v], e2 = rowStart[v + 1];
                    var cand = new List<int>();
                    for (int k = s2; k < e2; k++)
                    {
                        int u = adjList[k];
                        if (posInOrder[u] > pv) cand.Add(u);
                    }
                    cand.Sort();
                    var R = new List<int>();
                    R.Add(v);
                    BKList(R, cand);
                });
            }
            catch { }
        }

        var sb = new StringBuilder();
        sb.Append(bestSize).Append('\n');
        if (bestSize > 0)
        {
            for (int i = 0; i < bestSize; i++)
            {
                if (i > 0) sb.Append(' ');
                sb.Append(bestClique[i]);
            }
            sb.Append('\n');
        }
        else
        {
            sb.Append('\n');
        }
        Console.Write(sb.ToString());
    }

    static void BronKerbosch(int[] R, int rSize, ulong[] P)
    {
        if (stopFlag) return;

        int pCount = 0;
        for (int i = 0; i < W; i++) pCount += System.Numerics.BitOperations.PopCount(P[i]);
        if (pCount == 0)
        {
            UpdateBest(R, rSize);
            return;
        }

        if (rSize + pCount <= bestSize) return;

        if (sw.ElapsedMilliseconds > timeLimitMs) { stopFlag = true; return; }

        int pivot = -1;
        int pivotCount = -1;
        for (int wi = 0; wi < W; wi++)
        {
            ulong bits = P[wi];
            while (bits != 0)
            {
                int b = System.Numerics.BitOperations.TrailingZeroCount(bits);
                bits &= bits - 1;
                int u = (wi << 6) | b;
                var nu = adj[u];
                int c = 0;
                for (int i = 0; i < W; i++) c += System.Numerics.BitOperations.PopCount(P[i] & nu[i]);
                if (c > pivotCount) { pivotCount = c; pivot = u; }
            }
        }

        var cand = new ulong[W];
        var pivAdj = adj[pivot];
        for (int i = 0; i < W; i++) cand[i] = P[i] & ~pivAdj[i];

        for (int wi = 0; wi < W; wi++)
        {
            ulong bits = cand[wi];
            while (bits != 0)
            {
                if (stopFlag) return;
                int b = System.Numerics.BitOperations.TrailingZeroCount(bits);
                bits &= bits - 1;
                int v = (wi << 6) | b;

                var newP = new ulong[W];
                var nv = adj[v];
                for (int i = 0; i < W; i++) newP[i] = P[i] & nv[i];

                R[rSize] = v;
                BronKerbosch(R, rSize + 1, newP);

                P[wi] &= ~(1UL << b);
            }
        }
    }

    static void BKList(List<int> R, List<int> P)
    {
        if (stopFlag) return;
        if (P.Count == 0)
        {
            UpdateBest(R.ToArray(), R.Count);
            return;
        }
        if (R.Count + P.Count <= bestSize) return;

        if (sw.ElapsedMilliseconds > timeLimitMs) { stopFlag = true; return; }

        int pivot = P[0];
        int pivotCount = -1;
        foreach (var u in P)
        {
            int c = IntersectCount(P, u);
            if (c > pivotCount) { pivotCount = c; pivot = u; }
        }

        var cand = new List<int>();
        foreach (var v in P) if (v != pivot && !IsAdjacent(pivot, v)) cand.Add(v);
        if (cand.Count == 0) cand.Add(pivot);

        foreach (var v in cand)
        {
            if (stopFlag) return;
            var newP = IntersectWithNeighbors(P, v);
            R.Add(v);
            BKList(R, newP);
            R.RemoveAt(R.Count - 1);
            P.Remove(v);
        }
    }

    static bool IsAdjacent(int u, int v)
    {
        int s = rowStart[u], e = rowStart[u + 1];
        int lo = s, hi = e - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = adjList[mid];
            if (x == v) return true;
            if (x < v) lo = mid + 1; else hi = mid - 1;
        }
        return false;
    }

    static int IntersectCount(List<int> P, int u)
    {
        int s = rowStart[u], e = rowStart[u + 1];
        int c = 0;
        int i = 0, j = s;
        while (i < P.Count && j < e)
        {
            int a = P[i], b = adjList[j];
            if (a == b) { c++; i++; j++; }
            else if (a < b) i++;
            else j++;
        }
        return c;
    }

    static List<int> IntersectWithNeighbors(List<int> P, int u)
    {
        var res = new List<int>();
        int s = rowStart[u], e = rowStart[u + 1];
        int i = 0, j = s;
        while (i < P.Count && j < e)
        {
            int a = P[i], b = adjList[j];
            if (a == b) { res.Add(a); i++; j++; }
            else if (a < b) i++;
            else j++;
        }
        return res;
    }

    static void GreedyHeuristic()
    {
        var starts = Enumerable.Range(0, N).OrderByDescending(v => coreNum[v]).Take(Math.Min(N, 30)).ToArray();
        foreach (var start in starts)
        {
            var clique = new List<int> { start };
            var cand = new List<int>();
            int s = rowStart[start], e = rowStart[start + 1];
            for (int k = s; k < e; k++) cand.Add(adjList[k]);
            while (cand.Count > 0)
            {
                int best = -1, bestD = -1;
                foreach (var c in cand)
                {
                    if (coreNum[c] > bestD) { bestD = coreNum[c]; best = c; }
                }
                clique.Add(best);
                var newCand = new List<int>();
                foreach (var c in cand)
                {
                    if (c != best && IsAdjacent(best, c)) newCand.Add(c);
                }
                cand = newCand;
            }
            UpdateBest(clique.ToArray(), clique.Count);
        }
    }
