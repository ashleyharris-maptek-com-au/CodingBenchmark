using namespace std;

// Fast I/O buffers
const int BUF_SIZE = 1 << 24;
char in_buf[BUF_SIZE];
char* in_ptr = in_buf;
char* in_end = in_buf;

inline char next_char() {
    if (in_ptr == in_end) {
        size_t res = fread(in_buf, 1, BUF_SIZE, stdin);
        in_ptr = in_buf;
        in_end = in_buf + res;
        if (res == 0) return EOF;
    }
    return *in_ptr++;
}

inline float read_float() {
    char c = next_char();
    while (c <= 32 && c != EOF) c = next_char();
    if (c == EOF) return 0;
    
    bool neg = false;
    if (c == '-') { neg = true; c = next_char(); }
    else if (c == '+') { c = next_char(); }
    
    double res = 0;
    while (c >= '0' && c <= '9') {
        res = res * 10.0 + (c - '0');
        c = next_char();
    }
    if (c == '.') {
        c = next_char();
        double frac = 1.0;
        while (c >= '0' && c <= '9') {
            frac *= 0.1;
            res += (c - '0') * frac;
            c = next_char();
        }
    }
    if (c == 'e' || c == 'E') {
        c = next_char();
        bool exp_neg = false;
        if (c == '-') { exp_neg = true; c = next_char(); }
        else if (c == '+') { c = next_char(); }
        int exp = 0;
        while (c >= '0' && c <= '9') {
            exp = exp * 10 + (c - '0');
            c = next_char();
        }
        res *= std::pow(10.0, exp_neg ? -exp : exp);
    }
    return (float)(neg ? -res : res);
}

inline int read_int() {
    char c = next_char();
    while (c <= 32 && c != EOF) c = next_char();
    if (c == EOF) return 0;
    
    bool neg = false;
    if (c == '-') { neg = true; c = next_char(); }
    else if (c == '+') { c = next_char(); }
    
    int res = 0;
    while (c >= '0' && c <= '9') {
        res = res * 10 + (c - '0');
        c = next_char();
    }
    return neg ? -res : res;
}

char out_buf[1 << 24];
int out_pos = 0;

inline void write_int(int x) {
    if (x == 0) {
        out_buf[out_pos++] = '0';
        out_buf[out_pos++] = ' ';
        if (out_pos > (1<<24) - 32) {
            fwrite(out_buf, 1, out_pos, stdout);
            out_pos = 0;
        }
        return;
    }
    char rev[12];
    int len = 0;
    while (x > 0) {
        rev[len++] = (x % 10) + '0';
        x /= 10;
    }
    for (int i = len - 1; i >= 0; --i) {
        out_buf[out_pos++] = rev[i];
    }
    out_buf[out_pos++] = ' ';
    
    if (out_pos > (1<<24) - 32) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

vector<float> X;
vector<float> Y;

inline float distSq(int i, int j) {
    float dx = X[i] - X[j];
    float dy = Y[i] - Y[j];
    return dx * dx + dy * dy;
}
inline float dist(int i, int j) {
    return std::sqrt(distSq(i, j));
}

uint64_t hilbert_index(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int i = 30; i >= 0; i--) {
        uint32_t rx = (x >> i) & 1;
        uint32_t ry = (y >> i) & 1;
        uint32_t quad = (3 * rx) ^ ry;
        d = (d << 2) | quad;
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

struct SortItem {
    uint64_t h;
    int id;
    bool operator<(const SortItem& other) const {
        return h < other.h;
    }
};

bool two_opt_pass(vector<int>& path, int window_size, auto start_time) {
    bool improved = false;
    int n = (int)path.size();
    for (int i = 0; i < n - 2; ++i) {
        if ((i & 1023) == 0) {
            if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > 290.0) 
                return false;
        }
        
        int city_i = path[i];
        int city_next_i = path[i+1];
        float d1 = dist(city_i, city_next_i);
        
        int max_j = min(n - 1, i + window_size);
        for (int j = i + 2; j <= max_j; ++j) {
            if (i == 0 && j == n - 1) continue;
            
            int city_j = path[j];
            int j_next = (j + 1 == n) ? 0 : j + 1;
            int city_next_j = path[j_next];
            
            float d2 = dist(city_j, city_next_j);
            float d3 = dist(city_i, city_j);
            float d4 = dist(city_next_i, city_next_j);
            
            if (d3 + d4 < d1 + d2 - 1e-5f) {
                reverse(path.begin() + i + 1, path.begin() + j + 1);
                improved = true;
                city_next_i = path[i+1];
                d1 = dist(city_i, city_next_i);
            }
        }
    }
    return improved;
}

vector<int> nearest_neighbor(int start, int N) {
    vector<int> p;
    p.reserve(N);
    vector<bool> vis(N, false);
    p.push_back(start);
    vis[start] = true;
    for(int i = 1; i < N; ++i) {
        int last = p.back();
        float best_d = 1e30f;
        int best_j = -1;
        for(int j = 0; j < N; ++j) {
            if(!vis[j]) {
                float d = distSq(last, j);
                if(d < best_d) {
                    best_d = d;
                    best_j = j;
                }
            }
        }
        p.push_back(best_j);
        vis[best_j] = true;
    }
    return p;
}

int main() {
    auto start_time = chrono::steady_clock::now();
    
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int N = read_int();
    if (N <= 0) return 0;
    if (N == 1) {
        write_int(0);
        if (out_pos > 0) fwrite(out_buf, 1, out_pos, stdout);
        return 0;
    }
    
    X.resize(N);
    Y.resize(N);
    
    float min_x = 1e30f, max_x = -1e30f;
    float min_y = 1e30f, max_y = -1e30f;
    for (int i = 0; i < N; ++i) {
        float x = read_float();
        float y = read_float();
        X[i] = x;
        Y[i] = y;
        if (x < min_x) min_x = x;
        if (x > max_x) max_x = x;
        if (y < min_y) min_y = y;
        if (y > max_y) max_y = y;
    }
    
    vector<SortItem> items(N);
    double range_x = max_x - min_x;
    double range_y = max_y - min_y;
    double scale_x = range_x > 0 ? 2147483647.0 / range_x : 0;
    double scale_y = range_y > 0 ? 2147483647.0 / range_y : 0;
    
    for (int i = 0; i < N; ++i) {
        uint32_t ix = (uint32_t)((X[i] - min_x) * scale_x);
        uint32_t iy = (uint32_t)((Y[i] - min_y) * scale_y);
        items[i].h = hilbert_index(ix, iy);
        items[i].id = i;
    }
    
    sort(items.begin(), items.end());
    
    vector<int> original_id(N);
    for (int i = 0; i < N; ++i) {
        original_id[i] = items[i].id;
    }
    
    items.clear();
    items.shrink_to_fit();
    
    vector<float> newX(N), newY(N);
    for (int i = 0; i < N; ++i) {
        int old_id = original_id[i];
        newX[i] = X[old_id];
        newY[i] = Y[old_id];
    }
    X = std::move(newX);
    Y = std::move(newY);
    
    vector<int> path(N);
    iota(path.begin(), path.end(), 0);
    
    auto calc_len = [&](const vector<int>& p) {
        float l = 0;
        for(int i = 0; i < N; ++i) l += dist(p[i], p[(i+1)%N]);
        return l;
    };
    
    if (N <= 2000) {
        vector<int> best_path = path; 
        float best_len = calc_len(best_path);
        
        bool imp = true;
        while(imp) {
            imp = two_opt_pass(best_path, N, start_time);
            if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > 290.0) break;
        }
        best_len = calc_len(best_path);
        
        int max_starts = min(N, 100000 / N);
        if (max_starts < 1) max_starts = 1;
        
        for(int start = 0; start < max_starts; ++start) {
            if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > 290.0) break;
            vector<int> p = nearest_neighbor(start, N);
            imp = true;
            while(imp) {
                imp = two_opt_pass(p, N, start_time);
                if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > 290.0) break;
            }
            float l = calc_len(p);
            if (l < best_len) {
                best_len = l;
                best_path = p;
            }
        }
        path = best_path;
    } else {
        int max_passes = 50;
        long long target_checks = 200'000'000LL;
        int window_size = (int)min((long long)N, target_checks / N);
        if (window_size < 1) window_size = 1;
        if (window_size > 500) window_size = 500; 
        
        for (int pass = 0; pass < max_passes; ++pass) {
            if (chrono::duration<double>(chrono::steady_clock::now() - start_time).count() > 290.0) break;
            if (!two_opt_pass(path, window_size, start_time)) break;
        }
    }
    
    int start_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (original_id[path[i]] == 0) {
            start_idx = i;
            break;
        }
    }
    rotate(path.begin(), path.begin() + start_idx, path.end());
    
    for (int i = 0; i < N; ++i) {
        write_int(original_id[path[i]]);
    }
    if (out_pos > 0) {
        fwrite(out_buf, 1, out_pos, stdout);
    }
    
    return 0;
}
