#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>
#include <random>
#include <cstdio>

using namespace std;

const double EPS = 1e-9;
const double PI = 3.14159265358979323846;

struct Pt {
    double x, y, z;
    Pt operator-(Pt o) const { return {x - o.x, y - o.y, z - o.z}; }
    Pt operator+(Pt o) const { return {x + o.x, y + o.y, z + o.z}; }
    Pt operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(Pt o) const { return x * o.x + y * o.y + z * o.z; }
    Pt cross(Pt o) const { return {y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x}; }
    double norm() const { return sqrt(dot(*this)); }
};

struct Face {
    int a, b, c;
    Pt n;
    double d;
};

struct Tetra {
    double cx, cy, cz, rx, ry, rz;
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int V;
    double e;
    if (!(cin >> V >> e)) return 0;
    
    vector<Pt> pts(V);
    double min_x = 1e18, max_x = -1e18;
    double min_y = 1e18, max_y = -1e18;
    double min_z = 1e18, max_z = -1e18;
    for(int i=0; i<V; ++i) {
        cin >> pts[i].x >> pts[i].y >> pts[i].z;
        min_x = min(min_x, pts[i].x); max_x = max(max_x, pts[i].x);
        min_y = min(min_y, pts[i].y); max_y = max(max_y, pts[i].y);
        min_z = min(min_z, pts[i].z); max_z = max(max_z, pts[i].z);
    }
    
    mt19937 rng(42);
    uniform_real_distribution<double> dist(-1e-8, 1e-8);
    vector<Pt> hull_pts = pts;
    for(auto& p : hull_pts) {
        p.x += dist(rng);
        p.y += dist(rng);
        p.z += dist(rng);
    }
    
    int v1 = 0; double d1 = -1;
    for(int i=1; i<V; ++i) {
        double d = (hull_pts[i] - hull_pts[0]).norm();
        if(d > d1) { d1 = d; v1 = i; }
    }
    swap(hull_pts[1], hull_pts[v1]);

    int v2 = 0; double d2 = -1;
    for(int i=2; i<V; ++i) {
        Pt v_vec = hull_pts[1] - hull_pts[0];
        Pt w_vec = hull_pts[i] - hull_pts[0];
        double d = v_vec.cross(w_vec).norm();
        if(d > d2) { d2 = d; v2 = i; }
    }
    swap(hull_pts[2], hull_pts[v2]);

    int v3 = 0; double d3 = -1;
    for(int i=3; i<V; ++i) {
        Pt v_vec = hull_pts[1] - hull_pts[0];
        Pt w_vec = hull_pts[2] - hull_pts[0];
        Pt n = v_vec.cross(w_vec);
        double d = abs(n.dot(hull_pts[i] - hull_pts[0]));
        if(d > d3) { d3 = d; v3 = i; }
    }
    swap(hull_pts[3], hull_pts[v3]);

    vector<Face> faces;
    Pt center0 = (hull_pts[0] + hull_pts[1] + hull_pts[2] + hull_pts[3]) * 0.25;

    auto add_face = [&](int a, int b, int c) {
        Pt n = (hull_pts[b] - hull_pts[a]).cross(hull_pts[c] - hull_pts[a]);
        double len = n.norm();
        if (len < 1e-9) return;
        n = n * (1.0 / len);
        double d = n.dot(hull_pts[a]);
        if (n.dot(center0) > d) {
            swap(b, c);
            n = n * -1.0;
            d = -d;
        }
        faces.push_back({a, b, c, n, d});
    };

    add_face(0, 1, 2);
    add_face(0, 1, 3);
    add_face(0, 2, 3);
    add_face(1, 2, 3);

    for(int i=4; i<V; ++i) {
        vector<Face> next_faces;
        vector<pair<int, int>> edges;
        for(auto& f : faces) {
            if (f.n.dot(hull_pts[i]) > f.d + EPS) {
                edges.push_back({f.a, f.b});
                edges.push_back({f.b, f.c});
                edges.push_back({f.c, f.a});
            } else {
                next_faces.push_back(f);
            }
        }
        if (edges.empty()) continue;

        vector<pair<int,int>> sorted_edges = edges;
        for(auto& edge : sorted_edges) {
            if(edge.first > edge.second) swap(edge.first, edge.second);
        }
        sort(sorted_edges.begin(), sorted_edges.end());
        
        for(auto& edge : edges) {
            pair<int,int> canonical = {min(edge.first, edge.second), max(edge.first, edge.second)};
            auto it1 = lower_bound(sorted_edges.begin(), sorted_edges.end(), canonical);
            auto it2 = upper_bound(sorted_edges.begin(), sorted_edges.end(), canonical);
            if (it2 - it1 == 1) { 
                int a = edge.first, b = edge.second;
                Pt n = (hull_pts[a] - hull_pts[b]).cross(hull_pts[i] - hull_pts[b]);
                double len = n.norm();
                if (len > 1e-9) {
                    n = n * (1.0 / len);
                    double d = n.dot(hull_pts[b]);
                    next_faces.push_back({b, a, i, n, d});
                }
            }
        }
        faces = next_faces;
    }

    Pt u = {0, e, 0};
    Pt v = {e * sqrt(3.0)/2.0, e / 2.0, 0};
    Pt w = {e * sqrt(3.0)/6.0, e / 2.0, e * sqrt(6.0)/3.0};

    Pt V1[4] = {
        {e * sqrt(3.0)/3.0, 0, -e * sqrt(6.0)/12.0},
        {-e * sqrt(3.0)/6.0, e / 2.0, -e * sqrt(6.0)/12.0},
        {-e * sqrt(3.0)/6.0, -e / 2.0, -e * sqrt(6.0)/12.0},
        {0, 0, e * sqrt(6.0)/4.0}
    };
    
    Pt V2[4] = {
        {e * sqrt(3.0)/3.0, 0, -e * sqrt(6.0)/12.0},
        {-e * sqrt(3.0)/6.0, -e / 2.0, -e * sqrt(6.0)/12.0},
        {-e * sqrt(3.0)/6.0, e / 2.0, -e * sqrt(6.0)/12.0},
        {0, 0, -5.0 * e * sqrt(6.0)/12.0}
    };

    double max_Vz1 = max({V1[0].z, V1[1].z, V1[2].z, V1[3].z});
    double min_Vz1 = min({V1[0].z, V1[1].z, V1[2].z, V1[3].z});
    double max_Vz2 = max({V2[0].z, V2[1].z, V2[2].z, V2[3].z});
    double min_Vz2 = min({V2[0].z, V2[1].z, V2[2].z, V2[3].z});
    double max_Vz = max(max_Vz1, max_Vz2);
    double min_Vz = min(min_Vz1, min_Vz2);

    int k_min = floor((min_z - max_Vz) / w.z) - 3;
    int k_max = ceil((max_z - min_Vz) / w.z) + 3;

    double max_Vx = max({V1[0].x, V1[1].x, V1[2].x, V1[3].x, V2[0].x, V2[1].x, V2[2].x, V2[3].x});
    double min_Vx = min({V1[0].x, V1[1].x, V1[2].x, V1[3].x, V2[0].x, V2[1].x, V2[2].x, V2[3].x});

    vector<Tetra> out;
    auto start_time = chrono::steady_clock::now();
    bool time_up = false;

    for(int k = k_min; k <= k_max && !time_up; ++k) {
        int j_min = floor((min_x - max_Vx - k * w.x) / v.x) - 3;
        int j_max = ceil((max_x - min_Vx - k * w.x) / v.x) + 3;
        for(int j = j_min; j <= j_max && !time_up; ++j) {
            
            if ((j & 63) == 0) {
                auto now = chrono::steady_clock::now();
                if (chrono::duration_cast<chrono::milliseconds>(now - start_time).count() > 28000) {
                    time_up = true;
                    break;
                }
            }

            double I_min1 = -1e18, I_max1 = 1e18;
            for(auto& f : faces) {
                double A = f.n.y * e;
                Pt O_pt = {j * v.x + k * w.x, j * v.y + k * w.y, k * w.z};
                for(int m=0; m<4; ++m) {
                    double B = f.d - f.n.dot(O_pt + V1[m]);
                    if (abs(A) < 1e-9) {
                        if (B < -1e-9) I_max1 = -1e18;
                    } else if (A > 0) {
                        I_max1 = min(I_max1, B / A);
                    } else {
                        I_min1 = max(I_min1, B / A);
                    }
                }
            }
            if (I_min1 <= I_max1 + 1e-9) {
                int start_i = ceil(I_min1 - 1e-9);
                int end_i = floor(I_max1 + 1e-9);
                for(int i = start_i; i <= end_i; ++i) {
                    Pt C = {j * v.x + k * w.x, i * u.y + j * v.y + k * w.y, k * w.z};
                    out.push_back({C.x, C.y, C.z, 0, 0, 0});
                    if(out.size() >= 10000000) { time_up = true; break; }
                }
            }
            if(time_up) break;

            double I_min2 = -1e18, I_max2 = 1e18;
            for(auto& f : faces) {
                double A = f.n.y * e;
                Pt O_pt = {j * v.x + k * w.x, j * v.y + k * w.y, k * w.z};
                for(int m=0; m<4; ++m) {
                    double B = f.d - f.n.dot(O_pt + V2[m]);
                    if (abs(A) < 1e-9) {
                        if (B < -1e-9) I_max2 = -1e18;
                    } else if (A > 0) {
                        I_max2 = min(I_max2, B / A);
                    } else {
                        I_min2 = max(I_min2, B / A);
                    }
                }
            }
            if (I_min2 <= I_max2 + 1e-9) {
                int start_i = ceil(I_min2 - 1e-9);
                int end_i = floor(I_max2 + 1e-9);
                for(int i = start_i; i <= end_i; ++i) {
                    Pt C = {j * v.x + k * w.x, i * u.y + j * v.y + k * w.y, k * w.z};
                    out.push_back({C.x, C.y, C.z - e * sqrt(6.0)/6.0, PI, 0, 0});
                    if(out.size() >= 10000000) { time_up = true; break; }
                }
            }
        }
    }

    printf("%zu\n", out.size());
    for(auto& t : out) {
        printf("%.6f %.6f %.6f %.6f %.6f %.6f\n", t.cx, t.cy, t.cz, t.rx, t.ry, t.rz);
    }

    return 0;
}
