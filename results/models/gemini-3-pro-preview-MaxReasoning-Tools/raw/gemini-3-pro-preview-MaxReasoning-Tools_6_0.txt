using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main(string[] args)
    {
        string line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        string[] parts1 = line1.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0]);
        double mu = double.Parse(parts1[1]);

        string line2 = Console.ReadLine();
        double[] startOrbit = ParseState(line2);

        List<double[]> stations = new List<double[]>();
        for (int i = 0; i < N; i++)
        {
            stations.Add(ParseState(Console.ReadLine()));
        }

        List<int> path = SolveTSP(N, startOrbit, stations, mu);
        
        Console.WriteLine(string.Join(" ", path));
    }

    static double[] ParseState(string line)
    {
        string[] parts = line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] state = new double[6];
        for (int i = 0; i < 6; i++) state[i] = double.Parse(parts[i]);
        return state;
    }

    static List<int> SolveTSP(int N, double[] startOrbit, List<double[]> stations, double mu)
    {
        // Compute distance matrix
        double[,] distMatrix = new double[N + 1, N + 1];
        
        List<double[]> allNodes = new List<double[]>();
        allNodes.Add(startOrbit);
        allNodes.AddRange(stations);

        for (int i = 0; i <= N; i++)
        {
            for (int j = 0; j <= N; j++)
            {
                if (i == j) distMatrix[i, j] = 0;
                else distMatrix[i, j] = PseudoDeltaV(allNodes[i], allNodes[j], mu);
            }
        }

        // Nearest Neighbor
        List<int> unvisited = Enumerable.Range(1, N).ToList();
        List<int> path = new List<int>();
        int current = 0;

        while (unvisited.Count > 0)
        {
            int bestNext = -1;
            double bestDist = double.MaxValue;
            foreach (int cand in unvisited)
            {
                if (distMatrix[current, cand] < bestDist)
                {
                    bestDist = distMatrix[current, cand];
                    bestNext = cand;
                }
            }
            path.Add(bestNext);
            unvisited.Remove(bestNext);
            current = bestNext;
        }

        // 2-opt
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int k = i + 1; k < N; k++)
                {
                    int n1 = i == 0 ? 0 : path[i - 1];
                    int n2 = path[i];
                    int n3 = path[k];
                    // We don't have to return to start
                    int n4 = k == N - 1 ? -1 : path[k + 1]; 

                    double currentDist = distMatrix[n1, n2];
                    if (n4 != -1) currentDist += distMatrix[n3, n4];
                    
                    double newDist = distMatrix[n1, n3];
                    if (n4 != -1) newDist += distMatrix[n2, n4];

                    if (newDist < currentDist - 1e-5)
                    {
                        path.Reverse(i, k - i + 1);
                        improved = true;
                    }
                }
            }
        }

        // Convert path to 0-indexed station indices
        return path.Select(x => x - 1).ToList();
    }

    static double PseudoDeltaV(double[] s1, double[] s2, double mu)
    {
        // Simple proxy for delta-V: compare orbital elements
        // E = v^2 / 2 - mu / r
        double r1 = Math.Sqrt(s1[0]*s1[0] + s1[1]*s1[1] + s1[2]*s1[2]);
        double v1_sq = s1[3]*s1[3] + s1[4]*s1[4] + s1[5]*s1[5];
        double e1 = v1_sq / 2.0 - mu / r1;

        double r2 = Math.Sqrt(s2[0]*s2[0] + s2[1]*s2[1] + s2[2]*s2[2]);
        double v2_sq = s2[3]*s2[3] + s2[4]*s2[4] + s2[5]*s2[5];
        double e2 = v2_sq / 2.0 - mu / r2;

        // Angular momentum h = r x v
        double[] h1 = Cross(s1, new double[] { s1[3], s1[4], s1[5] });
        double[] h2 = Cross(s2, new double[] { s2[3], s2[4], s2[5] });

        double hm1 = Norm(h1);
        double hm2 = Norm(h2);

        // Inclination diff proxy: angle between h vectors
        double dot = h1[0]*h2[0] + h1[1]*h2[1] + h1[2]*h2[2];
        double cosTheta = dot / (hm1 * hm2);
        if (cosTheta > 1.0) cosTheta = 1.0;
        if (cosTheta < -1.0) cosTheta = -1.0;
        double theta = Math.Acos(cosTheta);

        // Plane change delta-V approx: 2 * v * sin(theta/2)
        double v_avg = Math.Sqrt(mu / ((r1+r2)/2.0));
        double dv_plane = 2.0 * v_avg * Math.Sin(theta / 2.0);

        // Hohmann transfer approx delta-V (coplanar)
        double a1 = -mu / (2.0 * e1);
        double a2 = -mu / (2.0 * e2);
        
        // Fallback for hyperbolic or undefined
        if (a1 <= 0 || a2 <= 0)
        {
             return Math.Sqrt(Math.Pow(s1[3]-s2[3], 2) + Math.Pow(s1[4]-s2[4], 2) + Math.Pow(s1[5]-s2[5], 2));
        }

        double a_trans = (a1 + a2) / 2.0;
        double v_t1 = Math.Sqrt(Math.Max(0, mu * (2.0/a1 - 1.0/a_trans)));
        double v_t2 = Math.Sqrt(Math.Max(0, mu * (2.0/a2 - 1.0/a_trans)));
        
        double v_c1 = Math.Sqrt(mu / a1);
        double v_c2 = Math.Sqrt(mu / a2);

        double dv_hohmann = Math.Abs(v_t1 - v_c1) + Math.Abs(v_c2 - v_t2);

        return dv_hohmann + dv_plane;
    }

    static double[] Cross(double[] v1, double[] v2)
    {
        return new double[]
        {
            v1[1]*v2[2] - v1[2]*v2[1],
            v1[2]*v2[0] - v1[0]*v2[2],
            v1[0]*v2[1] - v1[1]*v2[0]
        };
    }

    static double Norm(double[] v)
    {
        return Math.Sqrt(v[0]*v[0] + v[1]*v[1] + v[2]*v[2]);
    }
}
