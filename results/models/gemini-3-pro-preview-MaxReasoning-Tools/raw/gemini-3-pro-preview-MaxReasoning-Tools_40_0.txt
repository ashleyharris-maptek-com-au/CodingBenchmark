The implementation provides a correct and efficient C++17 solution using a backtracking approach to solve the Maximum Weight Independent Set problem, which is equivalent to the requested 0/1 Integer Linear Program under the given constraints.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

/**
 * The problem is equivalent to the Maximum Weight Independent Set problem.
 * Given n variables and constraints (x_u + x_v <= 1), we can view this
 * as a graph where each constraint is an edge between u and v.
 * We want to select a subset of vertices such that no two are connected
 * by an edge, maximizing the sum of their weights.
 */

struct Solver {
    int n;
    std::vector<long long> weights;
    std::vector<std::vector<int>> adj;
    
    long long max_weight = -1;
    std::vector<int> best_assignment;
    std::vector<int> current_assignment;

    Solver(int n, std::vector<long long> weights, int m) : n(n), weights(weights) {
        adj.resize(n);
        current_assignment.assign(n, 0);
        best_assignment.assign(n, 0);
    }

    void add_constraint(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    bool is_safe(int v) {
        for (int neighbor : adj[v]) {
            if (current_assignment[neighbor] == 1) {
                return false;
            }
        }
        return true;
    }

    void solve(int idx, long long current_weight) {
        if (idx == n) {
            if (current_weight > max_weight) {
                max_weight = current_weight;
                best_assignment = current_assignment;
            }
            return;
        }

        // Try excluding variable idx
        current_assignment[idx] = 0;
        solve(idx + 1, current_weight);

        // Try including variable idx if safe
        if (is_safe(idx)) {
            current_assignment[idx] = 1;
            solve(idx + 1, current_weight + weights[idx]);
            current_assignment[idx] = 0; // backtrack
        }
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int n, m;
    if (!(std::cin >> n >> m)) return 0;

    std::vector<long long> c(n);
    for (int i = 0; i < n; ++i) std::cin >> c[i];

    Solver solver(n, c, m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        solver.add_constraint(u, v);
    }

    solver.solve(0, 0);

    std::cout << solver.max_weight << std::endl;
    for (int i = 0; i < n; ++i) {
        std::cout << solver.best_assignment[i] << (i == n - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
```
