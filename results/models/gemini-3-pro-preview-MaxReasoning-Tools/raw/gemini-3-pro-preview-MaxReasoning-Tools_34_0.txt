import sys
import math

def calculate_distance(p1, p2):
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    ptr = 0
    N = int(input_data[ptr])
    K = int(input_data[ptr+1])
    CAPACITY = int(input_data[ptr+2])
    ptr += 3
    
    depot = (float(input_data[ptr]), float(input_data[ptr+1]))
    ptr += 2
    
    customers = []
    for i in range(N):
        customers.append({
            'id': i,
            'pos': (float(input_data[ptr]), float(input_data[ptr+1])),
            'demand': int(input_data[ptr+2])
        })
        ptr += 3
        
    savings = []
    for i in range(N):
        for j in range(i + 1, N):
            dist_i0 = calculate_distance(customers[i]['pos'], depot)
            dist_0j = calculate_distance(depot, customers[j]['pos'])
            dist_ij = calculate_distance(customers[i]['pos'], customers[j]['pos'])
            s = dist_i0 + dist_0j - dist_ij
            savings.append((s, i, j))
    
    savings.sort(key=lambda x: x[0], reverse=True)
    
    routes = [[i] for i in range(N)]
    route_demands = [customers[i]['demand'] for i in range(N)]
    route_map = {i: i for i in range(N)}
    route_ends = {i: (i, i) for i in range(N)}
    
    for s, i, j in savings:
        r1 = route_map[i]
        r2 = route_map[j]
        
        if r1 == r2:
            continue
            
        if route_demands[r1] + route_demands[r2] > CAPACITY:
            continue
            
        start1, end1 = route_ends[r1]
        start2, end2 = route_ends[r2]
        
        merged = False
        if end1 == i and start2 == j:
            new_route = routes[r1] + routes[r2]
            new_ends = (start1, end2)
            merged = True
        elif end1 == i and end2 == j:
            new_route = routes[r1] + routes[r2][::-1]
            new_ends = (start1, start2)
            merged = True
        elif start1 == i and start2 == j:
            new_route = routes[r1][::-1] + routes[r2]
            new_ends = (end1, end2)
            merged = True
        elif start1 == i and end2 == j:
            new_route = routes[r2] + routes[r1]
            new_ends = (start2, end1)
            merged = True
            
        if merged:
            routes[r1] = new_route
            route_demands[r1] += route_demands[r2]
            route_ends[r1] = new_ends
            routes[r2] = []
            route_demands[r2] = 0
            for node in new_route:
                route_map[node] = r1
            
    final_routes = [r for r in routes if r]
    
    total_dist = 0
    for route in final_routes:
        total_dist += calculate_distance(depot, customers[route[0]]['pos'])
        for k in range(len(route) - 1):
            total_dist += calculate_distance(customers[route[k]]['pos'], customers[route[k+1]]['pos'])
        total_dist += calculate_distance(customers[route[-1]]['pos'], depot)
        
    print(f"{total_dist:.4f}")
    print(len(final_routes))
    for route in final_routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    solve()
