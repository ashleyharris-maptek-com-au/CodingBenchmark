use std::io::{self, Read, Write};

#[derive(Clone, Copy)]
struct QuadNode {
    mass: usize,
    cx: f32,
    cy: f32,
    min_x: f32,
    min_y: f32,
    max_x: f32,
    max_y: f32,
    first_child: usize,
}

impl QuadNode {
    fn new(min_x: f32, min_y: f32, max_x: f32, max_y: f32) -> Self {
        Self {
            mass: 0,
            cx: 0.0,
            cy: 0.0,
            min_x,
            min_y,
            max_x,
            max_y,
            first_child: !0,
        }
    }
}

fn partition<F>(arr: &mut [usize], mut pred: F) -> usize
where
    F: FnMut(usize) -> bool,
{
    let mut left = 0;
    let mut right = arr.len();
    while left < right {
        if pred(arr[left]) {
            left += 1;
        } else {
            right -= 1;
            arr.swap(left, right);
        }
    }
    left
}

fn build_quadtree(
    tree: &mut Vec<QuadNode>,
    node_idx: usize,
    indices: &mut [usize],
    pos: &[(f32, f32)],
    depth: usize,
) {
    if indices.is_empty() {
        return;
    }

    let mut cx = 0.0;
    let mut cy = 0.0;
    for &idx in indices.iter() {
        cx += pos[idx].0;
        cy += pos[idx].1;
    }
    let mass = indices.len();
    cx /= mass as f32;
    cy /= mass as f32;

    tree[node_idx].cx = cx;
    tree[node_idx].cy = cy;
    tree[node_idx].mass = mass;

    if mass <= 1 || depth >= 20 {
        return;
    }

    let mid_x = (tree[node_idx].min_x + tree[node_idx].max_x) / 2.0;
    let mid_y = (tree[node_idx].min_y + tree[node_idx].max_y) / 2.0;

    let p_x = partition(indices, |idx| pos[idx].0 < mid_x);
    let (left_x, right_x) = indices.split_at_mut(p_x);

    let p_y1 = partition(left_x, |idx| pos[idx].1 < mid_y);
    let (q0, q1) = left_x.split_at_mut(p_y1);

    let p_y2 = partition(right_x, |idx| pos[idx].1 < mid_y);
    let (q2, q3) = right_x.split_at_mut(p_y2);

    let first_child = tree.len();
    let min_x = tree[node_idx].min_x;
    let max_x = tree[node_idx].max_x;
    let min_y = tree[node_idx].min_y;
    let max_y = tree[node_idx].max_y;

    tree.push(QuadNode::new(min_x, min_y, mid_x, mid_y));
    tree.push(QuadNode::new(min_x, mid_y, mid_x, max_y));
    tree.push(QuadNode::new(mid_x, min_y, max_x, mid_y));
    tree.push(QuadNode::new(mid_x, mid_y, max_x, max_y));

    tree[node_idx].first_child = first_child;

    build_quadtree(tree, first_child, q0, pos, depth + 1);
    build_quadtree(tree, first_child + 1, q1, pos, depth + 1);
    build_quadtree(tree, first_child + 2, q2, pos, depth + 1);
    build_quadtree(tree, first_child + 3, q3, pos, depth + 1);
}

fn compute_repulsion(tree: &[QuadNode], pos_i: (f32, f32), theta: f32) -> (f32, f32) {
    let mut force_x = 0.0;
    let mut force_y = 0.0;

    let mut stack = [0; 128];
    let mut stack_ptr = 0;
    stack[stack_ptr] = 0;
    stack_ptr += 1;

    while stack_ptr > 0 {
        stack_ptr -= 1;
        let curr = &tree[stack[stack_ptr]];

        if curr.mass == 0 {
            continue;
        }

        let dx = curr.cx - pos_i.0;
        let dy = curr.cy - pos_i.1;
        let dist_sq = dx * dx + dy * dy;
        let dist = dist_sq.sqrt();

        let width = curr.max_x - curr.min_x;

        if curr.first_child == !0 || width / dist < theta {
            if dist > 0.01 {
                let f = (curr.mass as f32) / dist;
                force_x -= (dx / dist) * f;
                force_y -= (dy / dist) * f;
            }
        } else {
            stack[stack_ptr] = curr.first_child;
            stack[stack_ptr + 1] = curr.first_child + 1;
            stack[stack_ptr + 2] = curr.first_child + 2;
            stack[stack_ptr + 3] = curr.first_child + 3;
            stack_ptr += 4;
        }
    }

    (force_x, force_y)
}

fn main() {
    let start_time = std::time::Instant::now();
    let max_duration = std::time::Duration::from_millis(28500);

    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();

    let mut tokens = input.split_whitespace();
    let n: usize = match tokens.next() {
        Some(s) => match s.parse() {
            Ok(val) => val,
            Err(_) => return,
        },
        None => return,
    };
    let m: usize = match tokens.next() {
        Some(s) => match s.parse() {
            Ok(val) => val,
            Err(_) => return,
        },
        None => return,
    };

    if n == 0 {
        return;
    }

    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = match tokens.next() {
            Some(s) => match s.parse() {
                Ok(val) => val,
                Err(_) => break,
            },
            None => break,
        };
        let v: usize = match tokens.next() {
            Some(s) => match s.parse() {
                Ok(val) => val,
                Err(_) => break,
            },
            None => break,
        };
        if u != v {
            let (u, v) = if u < v { (u, v) } else { (v, u) };
            edges.push((u, v));
        }
    }

    edges.sort_unstable();
    edges.dedup();

    let mut adj = vec![Vec::new(); n];
    for &(u, v) in &edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut pos = vec![(0.0, 0.0); n];
    let mut visited = vec![false; n];
    let mut current_center_x = 0.0;

    let mut nodes_by_degree: Vec<usize> = (0..n).collect();
    nodes_by_degree.sort_by_key(|&u| std::cmp::Reverse(adj[u].len()));

    for &start_node in &nodes_by_degree {
        if visited[start_node] {
            continue;
        }

        let mut queue = std::collections::VecDeque::new();
        queue.push_back(start_node);
        visited[start_node] = true;

        let mut current_radius = 0.0;
        let mut layer_nodes = Vec::new();
        let mut next_layer = std::collections::VecDeque::new();

        pos[start_node] = (current_center_x, 0.0);
        let mut max_radius_in_comp = 0.0;
        let mut comp_size = 1;

        while !queue.is_empty() {
            while let Some(u) = queue.pop_front() {
                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        next_layer.push_back(v);
                        layer_nodes.push(v);
                        comp_size += 1;
                    }
                }
            }

            if !layer_nodes.is_empty() {
                let l = layer_nodes.len() as f32;
                let min_r_for_spacing = l * 1.0 / (2.0 * std::f32::consts::PI);
                current_radius += 1.0;
                if current_radius < min_r_for_spacing {
                    current_radius = min_r_for_spacing;
                }
                max_radius_in_comp = current_radius;

                let angle_step = 2.0 * std::f32::consts::PI / l;
                for (i, &v) in layer_nodes.iter().enumerate() {
                    let angle = i as f32 * angle_step;
                    pos[v] = (
                        current_center_x + current_radius * angle.cos(),
                        current_radius * angle.sin(),
                    );
                }
                layer_nodes.clear();
            }
            std::mem::swap(&mut queue, &mut next_layer);
        }

        let shift = if comp_size == 1 {
            2.0
        } else {
            max_radius_in_comp * 2.0 + 10.0
        };
        current_center_x += shift;
    }

    for i in 0..n {
        let noise_x = ((i * 12345) % 100) as f32 * 0.001;
        let noise_y = ((i * 54321) % 100) as f32 * 0.001;
        pos[i].0 += noise_x;
        pos[i].1 += noise_y;
    }

    let initial_temp = (n as f32).sqrt().max(10.0);
    let mut max_iters = 400;
    let mut iter = 0;

    let mut indices: Vec<usize> = (0..n).collect();
    let mut tree = Vec::with_capacity(n * 4 + 128);
    let mut forces = vec![(0.0, 0.0); n];

    let num_threads = std::thread::available_parallelism()
        .map(|num| num.get())
        .unwrap_or(12);
    let chunk_size = (n + num_threads - 1) / num_threads;

    while iter < max_iters && start_time.elapsed() < max_duration {
        if iter == 10 {
            let elapsed = start_time.elapsed().as_secs_f32();
            if elapsed > 0.01 {
                let iters_per_sec = 10.0 / elapsed;
                let possible_iters = (28.0 * iters_per_sec) as usize;
                max_iters = max_iters.min(possible_iters);
                max_iters = max_iters.max(15);
            }
        }

        let mut progress = iter as f32 / max_iters as f32;
        if progress > 1.0 {
            progress = 1.0;
        }
        let temp = initial_temp * (1.0 - progress).max(0.01);

        tree.clear();
        let mut min_x = f32::INFINITY;
        let mut min_y = f32::INFINITY;
        let mut max_x = f32::NEG_INFINITY;
        let mut max_y = f32::NEG_INFINITY;

        for p in pos.iter() {
            if p.0 < min_x { min_x = p.0; }
            if p.0 > max_x { max_x = p.0; }
            if p.1 < min_y { min_y = p.1; }
            if p.1 > max_y { max_y = p.1; }
        }

        min_x -= 1.0;
        min_y -= 1.0;
        max_x += 1.0;
        max_y += 1.0;

        let size = (max_x - min_x).max(max_y - min_y);
        max_x = min_x + size;
        max_y = min_y + size;

        tree.push(QuadNode::new(min_x, min_y, max_x, max_y));

        for (i, x) in indices.iter_mut().enumerate() {
            *x = i;
        }

        build_quadtree(&mut tree, 0, &mut indices, &pos, 0);

        std::thread::scope(|s| {
            for (i, chunk) in forces.chunks_mut(chunk_size).enumerate() {
                let tree_ref = &tree;
                let pos_ref = &pos;
                let start_idx = i * chunk_size;
                s.spawn(move || {
                    for (j, f) in chunk.iter_mut().enumerate() {
                        let node_idx = start_idx + j;
                        if node_idx < pos_ref.len() {
                            *f = compute_repulsion(tree_ref, pos_ref[node_idx], 1.2);
                        }
                    }
                });
            }
        });

        for &(u, v) in &edges {
            let dx = pos[v].0 - pos[u].0;
            let dy = pos[v].1 - pos[u].1;
            let dist_sq = dx * dx + dy * dy;
            let dist = dist_sq.sqrt().max(0.01);
            
            let f = dist_sq;
            let fx = (dx / dist) * f;
            let fy = (dy / dist) * f;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

        for i in 0..n {
            let fx = forces[i].0;
            let fy = forces[i].1;
            let mag = (fx * fx + fy * fy).sqrt().max(0.001);

            let move_x = (fx / mag) * mag.min(temp);
            let move_y = (fy / mag) * mag.min(temp);

            pos[i].0 += move_x;
            pos[i].1 += move_y;
        }

        iter += 1;
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    for p in pos {
        writeln!(out, "{:.4} {:.4}", p.0, p.1).unwrap();
    }
}
