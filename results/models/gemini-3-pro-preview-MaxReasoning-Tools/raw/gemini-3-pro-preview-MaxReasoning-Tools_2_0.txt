using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    struct EdgeInfo 
    {
        public int u, v, id;
        public long w;
    }

    struct Edge 
    {
        public int to;
        public int id;
        public Edge(int to, int id) 
        {
            this.to = to;
            this.id = id;
        }
    }

    class Scanner 
    {
        Stream stream;
        byte[] buffer = new byte[1024 * 64];
        int head = 0;
        int tail = 0;

        public Scanner(Stream stream) 
        {
            this.stream = stream;
        }

        public long NextLong() 
        {
            int c = Read();
            while (c <= 32) 
            {
                if (c == -1) return -1;
                c = Read();
            }
            long res = 0;
            bool sign = false;
            if (c == '-') 
            {
                sign = true;
                c = Read();
            }
            while (c > 32) 
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return sign ? -res : res;
        }

        int Read() 
        {
            if (head >= tail) 
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }

    static void Main(string[] args)
    {
        Scanner sc = new Scanner(Console.OpenStandardInput());
        long n_long = sc.NextLong();
        if (n_long == -1) return;
        int N = (int)n_long;
        int M = (int)sc.NextLong();

        List<EdgeInfo> allEdges = new List<EdgeInfo>(M);
        List<Edge>[] adj = new List<Edge>[N];
        for (int i=0; i<N; i++) adj[i] = new List<Edge>();
        
        int[] originalDegree = new int[N];

        for (int i=0; i<M; i++) 
        {
            int u = (int)sc.NextLong();
            int v = (int)sc.NextLong();
            long w = sc.NextLong();
            
            allEdges.Add(new EdgeInfo { u = u, v = v, w = w, id = i });
            adj[u].Add(new Edge(v, i));
            adj[v].Add(new Edge(u, i));
            originalDegree[u]++;
            originalDegree[v]++;
        }

        List<int> oddNodes = new List<int>();
        for (int i=0; i<N; i++) 
        {
            if (originalDegree[i] % 2 != 0) 
            {
                oddNodes.Add(i);
            }
        }

        int nextEdgeId = M;

        if (oddNodes.Count > 0) 
        {
            if (oddNodes.Count <= 18) 
            {
                SolveExactDP(N, oddNodes, allEdges, adj, ref nextEdgeId);
            }
            else if (oddNodes.Count <= 400 && (long)oddNodes.Count * M <= 20000000L) 
            {
                SolveGreedy(N, oddNodes, allEdges, adj, ref nextEdgeId);
            }
            else 
            {
                SolveMST(N, oddNodes, allEdges, adj, originalDegree, ref nextEdgeId);
            }
        }

        bool[] usedEdge = new bool[nextEdgeId];
        int[] edgeIndex = new int[N];
        
        Stack<int> stack = new Stack<int>();
        List<int> result = new List<int>();
        stack.Push(0);

        while (stack.Count > 0) 
        {
            int u = stack.Peek();
            bool found = false;
            while (edgeIndex[u] < adj[u].Count) 
            {
                Edge e = adj[u][edgeIndex[u]++];
                if (!usedEdge[e.id]) 
                {
                    usedEdge[e.id] = true;
                    stack.Push(e.to);
                    found = true;
                    break;
                }
            }
            if (!found) 
            {
                result.Add(stack.Pop());
            }
        }

        result.Reverse();

        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536)) 
        {
            writer.WriteLine(result.Count);
            for (int i = 0; i < result.Count; i++) 
            {
                writer.Write(result[i]);
                if (i < result.Count - 1) writer.Write(' ');
            }
            writer.WriteLine();
        }
    }

    class DijkstraResult 
    {
        public long[] dist;
        public int[] prevNode;
    }

    static DijkstraResult RunDijkstra(int N, int start, List<Edge>[] adj, List<EdgeInfo> allEdges) 
    {
        long[] dist = new long[N];
        int[] prevNode = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(prevNode, -1);
        
        dist[start] = 0;
        PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
        pq.Enqueue(start, 0);

        while (pq.Count > 0) 
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            
            foreach (var e in adj[u]) 
            {
                long w = allEdges[e.id].w;
                if (dist[u] + w < dist[e.to]) 
                {
                    dist[e.to] = dist[u] + w;
                    prevNode[e.to] = u;
                    pq.Enqueue(e.to, dist[e.to]);
                }
            }
        }
        return new DijkstraResult { dist = dist, prevNode = prevNode };
    }

    static void SolveExactDP(int N, List<int> oddNodes, List<EdgeInfo> allEdges, List<Edge>[] adj, ref int nextEdgeId) 
    {
        int O = oddNodes.Count;
        DijkstraResult[] results = new DijkstraResult[O];
        for (int i = 0; i < O; i++) 
        {
            results[i] = RunDijkstra(N, oddNodes[i], adj, allEdges);
        }

        long[] dp = new long[1 << O];
        int[] best_prev_mask = new int[1 << O];
        Array.Fill(dp, long.MaxValue / 2);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << O); mask++) 
        {
            if (dp[mask] >= long.MaxValue / 2) continue;

            int i = -1;
            for (int b = 0; b < O; b++) 
            {
                if ((mask & (1 << b)) == 0) 
                {
                    i = b;
                    break;
                }
            }
            if (i == -1) continue;

            for (int j = i + 1; j < O; j++) 
            {
                if ((mask & (1 << j)) == 0) 
                {
                    long w = results[i].dist[oddNodes[j]];
                    if (w == long.MaxValue) continue;
                    
                    int nmask = mask | (1 << i) | (1 << j);
                    long ncost = dp[mask] + w;
                    if (ncost < dp[nmask]) 
                    {
                        dp[nmask] = ncost;
                        best_prev_mask[nmask] = mask;
                    }
                }
            }
        }

        int currMask = (1 << O) - 1;
        while (currMask > 0) 
        {
            int prev = best_prev_mask[currMask];
            int diff = currMask ^ prev;
            int u = -1, v = -1;
            for (int b = 0; b < O; b++) 
            {
                if ((diff & (1 << b)) != 0) 
                {
                    if (u == -1) u = b;
                    else v = b;
                }
            }

            int currNode = oddNodes[v];
            int srcNode = oddNodes[u];
            while (currNode != srcNode) 
            {
                int pNode = results[u].prevNode[currNode];
                adj[currNode].Add(new Edge(pNode, nextEdgeId));
                adj[pNode].Add(new Edge(currNode, nextEdgeId));
                nextEdgeId++;
                currNode = pNode;
            }
            currMask = prev;
        }
    }

    struct Pair 
    {
        public int i, j;
        public long w;
    }

    static void SolveGreedy(int N, List<int> oddNodes, List<EdgeInfo> allEdges, List<Edge>[] adj, ref int nextEdgeId) 
    {
        int O = oddNodes.Count;
        long[,] distMatrix = new long[O, O];
        for (int i = 0; i < O; i++) 
        {
            DijkstraResult res = RunDijkstra(N, oddNodes[i], adj, allEdges);
            for (int j = 0; j < O; j++) 
            {
                distMatrix[i, j] = res.dist[oddNodes[j]];
            }
        }

        List<Pair> pairs = new List<Pair>();
        for (int i = 0; i < O; i++) 
        {
            for (int j = i + 1; j < O; j++) 
            {
                if (distMatrix[i, j] != long.MaxValue) 
                {
                    pairs.Add(new Pair { i = i, j = j, w = distMatrix[i, j] });
                }
            }
        }

        pairs.Sort((a, b) => a.w.CompareTo(b.w));
        bool[] matched = new bool[O];

        foreach (var p in pairs) 
        {
            if (!matched[p.i] && !matched[p.j]) 
            {
                matched[p.i] = true;
                matched[p.j] = true;

                // Rerun Dijkstra to get the path
                DijkstraResult pathRes = RunDijkstra(N, oddNodes[p.i], adj, allEdges);

                int curr = oddNodes[p.j];
                int src = oddNodes[p.i];
                while (curr != src) 
                {
                    int pNode = pathRes.prevNode[curr];
                    adj[curr].Add(new Edge(pNode, nextEdgeId));
                    adj[pNode].Add(new Edge(curr, nextEdgeId));
                    nextEdgeId++;
                    curr = pNode;
                }
            }
        }
    }

    class DSU 
    {
        int[] parent;
        public DSU(int n) 
        {
            parent = new int[n];
            for (int i=0; i<n; i++) parent[i] = i;
        }
        public int Find(int i) 
        {
            if (parent[i] == i)
                return i;
            return parent[i] = Find(parent[i]);
        }
        public void Union(int i, int j) 
        {
            int root_i = Find(i);
            int root_j = Find(j);
            if (root_i != root_j) 
            {
                parent[root_i] = root_j;
            }
        }
    }

    static void SolveMST(int N, List<int> oddNodes, List<EdgeInfo> allEdges, List<Edge>[] adj, int[] originalDegree, ref int nextEdgeId) 
    {
        DSU dsu = new DSU(N);
        List<EdgeInfo>[] mstAdj = new List<EdgeInfo>[N];
        for (int i=0; i<N; i++) mstAdj[i] = new List<EdgeInfo>();

        allEdges.Sort((a, b) => a.w.CompareTo(b.w));
        foreach (var e in allEdges) 
        {
            if (dsu.Find(e.u) != dsu.Find(e.v)) 
            {
                dsu.Union(e.u, e.v);
                mstAdj[e.u].Add(e);
                mstAdj[e.v].Add(e);
            }
        }

        int[] parity = new int[N];
        for (int i=0; i<N; i++) parity[i] = originalDegree[i] % 2;

        int[] parent = new int[N];
        Array.Fill(parent, -1);
        bool[] vis = new bool[N];
        List<int> order = new List<int>();

        Queue<int> q = new Queue<int>();
        
        for (int i=0; i<N; i++) 
        {
            if (!vis[i] && originalDegree[i] > 0) 
            {
                q.Enqueue(i);
                vis[i] = true;
                parent[i] = i; 
                while (q.Count > 0) 
                {
                    int u = q.Dequeue();
                    order.Add(u);
                    foreach (var e in mstAdj[u]) 
                    {
                        int v = (e.u == u) ? e.v : e.u;
                        if (!vis[v]) 
                        {
                            vis[v] = true;
                            parent[v] = u;
                            q.Enqueue(v);
                        }
                    }
                }
            }
        }

        order.Reverse();

        foreach (int u in order) 
        {
            if (parent[u] == u) continue; 
            
            if (parity[u] == 1) 
            {
                parity[u] = 0;
                parity[parent[u]] ^= 1;
                
                adj[u].Add(new Edge(parent[u], nextEdgeId));
                adj[parent[u]].Add(new Edge(u, nextEdgeId));
                nextEdgeId++;
            }
        }
    }
}
