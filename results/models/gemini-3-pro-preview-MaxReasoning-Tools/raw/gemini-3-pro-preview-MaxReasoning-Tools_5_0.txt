use std::io::{self, Read};
use std::collections::VecDeque;

struct Solver {
    w: usize,
    h: usize,
    target: usize,
    req_cycle: bool,
    transpose: bool,
    start_x: usize,
    start_y: usize,
    obstacles: Vec<u64>,
    visited: Vec<u64>,
    path: Vec<(usize, usize)>,
}

impl Solver {
    fn solve(&mut self) -> bool {
        self.visited = self.obstacles.clone();
        self.visited[self.start_y] |= 1 << self.start_x;
        self.path.push((self.start_x, self.start_y));
        self.dfs(self.start_x, self.start_y)
    }

    fn dfs(&mut self, x: usize, y: usize) -> bool {
        let rem = self.target - self.path.len();
        if rem == 0 {
            if !self.req_cycle {
                return true;
            }
            let dx = (x as isize - self.start_x as isize).abs();
            let dy = (y as isize - self.start_y as isize).abs();
            return dx + dy == 1;
        }

        if !self.check_connected_bitmask(x, y, rem) {
            return false;
        }

        let neighbors = self.get_neighbors(x, y);

        for (nx, ny) in neighbors {
            self.visited[ny] |= 1 << nx;
            self.path.push((nx, ny));
            if self.dfs(nx, ny) {
                return true;
            }
            self.path.pop();
            self.visited[ny] &= !(1 << nx);
        }
        false
    }

    fn check_connected_bitmask(&self, x: usize, y: usize, rem: usize) -> bool {
        let mut seen = self.visited.clone();
        let mut active = vec![0u64; self.h];
        
        let mut found = false;
        let mut start_ny = 0;
        
        let dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)];
        for &(dx, dy) in &dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && nx < self.w as isize && ny >= 0 && ny < self.h as isize {
                let nx = nx as usize;
                let ny = ny as usize;
                if (seen[ny] & (1 << nx)) == 0 {
                    active[ny] |= 1 << nx;
                    seen[ny] |= 1 << nx;
                    start_ny = ny;
                    found = true;
                    break;
                }
            }
        }
        
        if !found {
            return false;
        }
        
        let mut count = 1;
        let w_mask = if self.w == 64 { !0 } else { (1u64 << self.w) - 1 };
        
        let mut min_y = start_ny;
        let mut max_y = start_ny;
        
        loop {
            let mut next_active = vec![0u64; self.h];
            let mut any = false;
            
            let mut next_min = self.h;
            let mut next_max = 0;
            
            for i in min_y..=max_y {
                let a = active[i];
                if a == 0 { continue; }
                
                let mut na = (a << 1) & w_mask;
                na |= a >> 1;
                if i > 0 { na |= active[i - 1]; }
                if i + 1 < self.h { na |= active[i + 1]; }
                
                na &= !seen[i];
                
                if na != 0 {
                    next_active[i] = na;
                    seen[i] |= na;
                    count += na.count_ones() as usize;
                    any = true;
                    
                    if i < next_min { next_min = i; }
                    if i > next_max { next_max = i; }
                }
            }
            
            if !any {
                break;
            }
            active = next_active;
            min_y = next_min;
            max_y = next_max;
        }
        
        count == rem
    }

    fn get_neighbors(&self, x: usize, y: usize) -> Vec<(usize, usize)> {
        let mut neighbors = Vec::new();
        let dirs = [(0, 1), (1, 0), (0, -1), (-1, 0)];
        for &(dx, dy) in &dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && nx < self.w as isize && ny >= 0 && ny < self.h as isize {
                let nx = nx as usize;
                let ny = ny as usize;
                if (self.visited[ny] & (1 << nx)) == 0 {
                    let mut deg = 0;
                    for &(ddx, ddy) in &dirs {
                        let nnx = nx as isize + ddx;
                        let nny = ny as isize + ddy;
                        if nnx >= 0 && nnx < self.w as isize && nny >= 0 && nny < self.h as isize {
                            let nnx = nnx as usize;
                            let nny = nny as usize;
                            if (self.visited[nny] & (1 << nnx)) == 0 {
                                deg += 1;
                            }
                        }
                    }
                    neighbors.push((deg, nx, ny));
                }
            }
        }
        neighbors.sort_by_key(|k| k.0);
        neighbors.into_iter().map(|(_, nx, ny)| (nx, ny)).collect()
    }
}

fn check_parity(w: usize, h: usize, obstacles: &Vec<(usize, usize)>, target: usize, req_cycle: bool) -> bool {
    let mut white = 0;
    let mut black = 0;
    for y in 0..h {
        for x in 0..w {
            let mut is_obs = false;
            for &(ox, oy) in obstacles {
                if ox == x && oy == y {
                    is_obs = true;
                    break;
                }
            }
            if !is_obs {
                if (x + y) % 2 == 0 {
                    white += 1;
                } else {
                    black += 1;
                }
            }
        }
    }
    
    let req_white = (target + 1) / 2;
    let req_black = target / 2;
    
    if white != req_white || black != req_black {
        return false;
    }
    
    if req_cycle {
        if target % 2 != 0 {
            return false;
        }
        if white != black {
            return false;
        }
    }
    
    true
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut tokens = input.split_whitespace();
    
    let w_str = match tokens.next() {
        Some(s) => s,
        None => return,
    };
    let w = w_str.parse::<usize>().unwrap();
    let h = tokens.next().unwrap().parse::<usize>().unwrap();
    let obs_count = tokens.next().unwrap().parse::<usize>().unwrap();
    let req_cycle = tokens.next().unwrap().parse::<i32>().unwrap() != 0;
    
    let mut obstacles = Vec::new();
    for _ in 0..obs_count {
        let x = tokens.next().unwrap().parse::<usize>().unwrap();
        let y = tokens.next().unwrap().parse::<usize>().unwrap();
        obstacles.push((x, y));
    }
    
    let target = w * h - obs_count;
    
    if target == 0 {
        println!("0");
        return;
    }
    
    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }
    
    if !check_parity(w, h, &obstacles, target, req_cycle) {
        println!("0");
        return;
    }
    
    let mut transpose = false;
    let mut eff_w = w;
    let mut eff_h = h;
    let mut eff_obstacles = obstacles.clone();
    
    if w > h {
        transpose = true;
        eff_w = h;
        eff_h = w;
        for obs in &mut eff_obstacles {
            let temp = obs.0;
            obs.0 = obs.1;
            obs.1 = temp;
        }
    }
    
    if eff_w > 64 {
        // Fallback or panic, but min(w,h) <= 50 for 2500 cells.
    }
    
    let mut grid_mask = vec![0u64; eff_h];
    for &(x, y) in &eff_obstacles {
        grid_mask[y] |= 1 << x;
    }
    
    let mut solver = Solver {
        w: eff_w,
        h: eff_h,
        target,
        req_cycle,
        transpose,
        start_x: 0,
        start_y: 0,
        obstacles: grid_mask.clone(),
        visited: grid_mask,
        path: Vec::with_capacity(target),
    };
    
    if solver.solve() {
        println!("{}", solver.path.len());
        for &(x, y) in &solver.path {
            if transpose {
                println!("{} {}", y, x);
            } else {
                println!("{} {}", x, y);
            }
        }
    } else {
        println!("0");
    }
}
