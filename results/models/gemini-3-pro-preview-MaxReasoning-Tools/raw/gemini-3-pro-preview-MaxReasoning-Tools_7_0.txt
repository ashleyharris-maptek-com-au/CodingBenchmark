using namespace std;

struct vec3 {
    double x, y, z;
    double operator[](int i) const { return i==0 ? x : (i==1 ? y : z); }
    double& operator[](int i) { return i==0 ? x : (i==1 ? y : z); }
};

vec3 operator-(vec3 a, vec3 b) { return {a.x-b.x, a.y-b.y, a.z-b.z}; }
vec3 operator+(vec3 a, vec3 b) { return {a.x+b.x, a.y+b.y, a.z+b.z}; }
vec3 operator*(vec3 a, double s) { return {a.x*s, a.y*s, a.z*s}; }
vec3 operator/(vec3 a, double s) { return {a.x/s, a.y/s, a.z/s}; }
double dot(vec3 a, vec3 b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
vec3 cross(vec3 a, vec3 b) { return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x}; }
double length(vec3 a) { return sqrt(dot(a, a)); }
vec3 normalize(vec3 a) { double l = length(a); return l > 0 ? a / l : vec3{0,0,0}; }

using vec3i = array<int, 3>;

struct Mesh {
    vector<vec3> verts;
    vector<vec3i> faces;
    vec3 get_normal(int i) const {
        vec3 v0 = verts[faces[i][0]];
        vec3 v1 = verts[faces[i][1]];
        vec3 v2 = verts[faces[i][2]];
        return normalize(cross(v1 - v0, v2 - v0));
    }
    vec3 get_centroid(int i) const {
        return (verts[faces[i][0]] + verts[faces[i][1]] + verts[faces[i][2]]) / 3.0;
    }
};

struct AABB {
    vec3 bmin{1e30, 1e30, 1e30};
    vec3 bmax{-1e30, -1e30, -1e30};
    void expand(vec3 p) {
        bmin.x = min(bmin.x, p.x); bmin.y = min(bmin.y, p.y); bmin.z = min(bmin.z, p.z);
        bmax.x = max(bmax.x, p.x); bmax.y = max(bmax.y, p.y); bmax.z = max(bmax.z, p.z);
    }
    vec3 center() const { return (bmin + bmax) * 0.5; }
};

bool intersect(const AABB& a, const AABB& b) {
    if (a.bmax.x < b.bmin.x || a.bmin.x > b.bmax.x) return false;
    if (a.bmax.y < b.bmin.y || a.bmin.y > b.bmax.y) return false;
    if (a.bmax.z < b.bmin.z || a.bmin.z > b.bmax.z) return false;
    return true;
}

struct BVHNode {
    AABB box;
    int left, right;
    int first_tri, num_tris;
};

struct BVH {
    vector<BVHNode> nodes;
    vector<int> tri_indices;
};

int build_node(BVH& bvh, const Mesh& mesh, int first, int num) {
    int node_idx = bvh.nodes.size();
    bvh.nodes.push_back(BVHNode());
    
    AABB box;
    for(int i=0; i<num; ++i) {
        int t_idx = bvh.tri_indices[first + i];
        box.expand(mesh.verts[mesh.faces[t_idx][0]]);
        box.expand(mesh.verts[mesh.faces[t_idx][1]]);
        box.expand(mesh.verts[mesh.faces[t_idx][2]]);
    }
    bvh.nodes[node_idx].box = box;
    
    if (num <= 4) {
        bvh.nodes[node_idx].left = -1;
        bvh.nodes[node_idx].right = -1;
        bvh.nodes[node_idx].first_tri = first;
        bvh.nodes[node_idx].num_tris = num;
        return node_idx;
    }
    
    vec3 extent = box.bmax - box.bmin;
    int axis = 0;
    if (extent.y > extent.x) axis = 1;
    if (extent.z > extent[axis]) axis = 2;
    
    double split_pos = box.center()[axis];
    
    int mid = partition(bvh.tri_indices.begin() + first, bvh.tri_indices.begin() + first + num, [&](int t_idx) {
        return mesh.get_centroid(t_idx)[axis] < split_pos;
    }) - bvh.tri_indices.begin();
    
    if (mid == first || mid == first + num) {
        mid = first + num / 2;
    }
    
    int left_child = build_node(bvh, mesh, first, mid - first);
    int right_child = build_node(bvh, mesh, mid, first + num - mid);
    
    bvh.nodes[node_idx].left = left_child;
    bvh.nodes[node_idx].right = right_child;
    
    return node_idx;
}

void build_bvh(BVH& bvh, const Mesh& mesh) {
    bvh.tri_indices.resize(mesh.faces.size());
    for(size_t i=0; i<mesh.faces.size(); ++i) bvh.tri_indices[i] = i;
    bvh.nodes.reserve(mesh.faces.size() * 2);
    build_node(bvh, mesh, 0, mesh.faces.size());
}

void query_bvh(int node_idx, const BVHNode* bvh, const AABB& box, vector<int>& out_tris) {
    if (!intersect(bvh[node_idx].box, box)) return;
    if (bvh[node_idx].left == -1) {
        for(int i=0; i<bvh[node_idx].num_tris; ++i) {
            out_tris.push_back(bvh[node_idx].first_tri + i);
        }
    } else {
        query_bvh(bvh[node_idx].left, bvh, box, out_tris);
        query_bvh(bvh[node_idx].right, bvh, box, out_tris);
    }
}

double safe_inv(double x) {
    if (abs(x) < 1e-12) return x >= 0 ? 1e30 : -1e30;
    return 1.0 / x;
}

bool ray_aabb(const AABB& box, vec3 orig, vec3 inv_dir, double& tmin, double& tmax) {
    double tx1 = (box.bmin.x - orig.x) * inv_dir.x;
    double tx2 = (box.bmax.x - orig.x) * inv_dir.x;
    tmin = min(tx1, tx2); tmax = max(tx1, tx2);
    
    double ty1 = (box.bmin.y - orig.y) * inv_dir.y;
    double ty2 = (box.bmax.y - orig.y) * inv_dir.y;
    tmin = max(tmin, min(ty1, ty2)); tmax = min(tmax, max(ty1, ty2));
    
    double tz1 = (box.bmin.z - orig.z) * inv_dir.z;
    double tz2 = (box.bmax.z - orig.z) * inv_dir.z;
    tmin = max(tmin, min(tz1, tz2)); tmax = min(tmax, max(tz1, tz2));
    
    return tmax >= max(0.0, tmin) && tmin <= tmax;
}

bool ray_triangle(vec3 orig, vec3 dir, vec3 v0, vec3 v1, vec3 v2, double& t, double& u, double& v) {
    vec3 edge1 = v1 - v0;
    vec3 edge2 = v2 - v0;
    vec3 pvec = cross(dir, edge2);
    double det = dot(edge1, pvec);
    if (abs(det) < 1e-12) return false;
    double inv_det = 1.0 / det;
    vec3 tvec = orig - v0;
    u = dot(tvec, pvec) * inv_det;
    if (u < -1e-7 || u > 1.0 + 1e-7) return false;
    vec3 qvec = cross(tvec, edge1);
    v = dot(dir, qvec) * inv_det;
    if (v < -1e-7 || u + v > 1.0 + 1e-7) return false;
    t = dot(edge2, qvec) * inv_det;
    return t > -1e-7;
}

void ray_bvh(int node_idx, const BVHNode* bvh, const Mesh& mesh, vec3 orig, vec3 dir, vec3 inv_dir, 
             double& min_t, int& best_face, double& best_u, double& best_v) {
    double tmin, tmax;
    if (!ray_aabb(bvh[node_idx].box, orig, inv_dir, tmin, tmax)) return;
    if (tmin > min_t) return;
    
    if (bvh[node_idx].left == -1) {
        for(int i=0; i<bvh[node_idx].num_tris; ++i) {
            int t_idx = bvh[node_idx].first_tri + i;
            double t, u, v;
            if (ray_triangle(orig, dir, mesh.verts[mesh.faces[t_idx][0]], mesh.verts[mesh.faces[t_idx][1]], mesh.verts[mesh.faces[t_idx][2]], t, u, v)) {
                if (t < min_t) { min_t = t; best_face = t_idx; best_u = u; best_v = v; }
            }
        }
    } else {
        ray_bvh(bvh[node_idx].left, bvh, mesh, orig, dir, inv_dir, min_t, best_face, best_u, best_v);
        ray_bvh(bvh[node_idx].right, bvh, mesh, orig, dir, inv_dir, min_t, best_face, best_u, best_v);
    }
}

enum Class { INSIDE, OUTSIDE, ON, UNKNOWN };

Class classify(vec3 origin, vec3 ray_dir, const BVH& bvh, const Mesh& mesh, vec3& hit_normal, double EPS) {
    double min_t = 1e30;
    int best_face = -1;
    double best_u = 0, best_v = 0;
    vec3 inv_dir = {safe_inv(ray_dir.x), safe_inv(ray_dir.y), safe_inv(ray_dir.z)};
    
    ray_bvh(0, bvh.nodes.data(), mesh, origin, ray_dir, inv_dir, min_t, best_face, best_u, best_v);
    
    if (best_face == -1) return OUTSIDE;
    
    if (abs(min_t) < EPS) {
        hit_normal = mesh.get_normal(best_face);
        return ON;
    }
    
    if (best_u < 1e-4 || best_v < 1e-4 || (1.0 - best_u - best_v) < 1e-4) return UNKNOWN;
    
    hit_normal = mesh.get_normal(best_face);
    double dot_val = dot(ray_dir, hit_normal);
    if (abs(dot_val) < 1e-4) return UNKNOWN;
    
    return dot_val > 0 ? INSIDE : OUTSIDE;
}

double get_rand() {
    static thread_local mt19937 gen(1337);
    uniform_real_distribution<double> dis(0.0, 1.0);
    return dis(gen);
}

Class robust_classify(vec3 origin, vec3 normal_src, const BVH& bvh, const Mesh& mesh, vec3& hit_normal, double EPS) {
    Class c = classify(origin, normal_src, bvh, mesh, hit_normal, EPS);
    if (c != UNKNOWN) return c;
    
    for(int i=0; i<15; ++i) {
        vec3 rand_dir = normalize(vec3{get_rand()-0.5, get_rand()-0.5, get_rand()-0.5});
        c = classify(origin, rand_dir, bvh, mesh, hit_normal, EPS);
        if (c != UNKNOWN && c != ON) return c;
    }
    return OUTSIDE;
}

struct Polygon {
    vector<vec3> verts;
};

void split_polygon(const Polygon& poly, const vec3& plane_p, const vec3& plane_n, 
                   vector<Polygon>& out_front, vector<Polygon>& out_back, double EPS) {
    vector<vec3> front_verts, back_verts;
    int n = poly.verts.size();
    vector<double> dists(n);
    bool all_front = true, all_back = true;
    
    for (int i = 0; i < n; ++i) {
        dists[i] = dot(poly.verts[i] - plane_p, plane_n);
        if (dists[i] < -EPS) all_front = false;
        if (dists[i] > EPS) all_back = false;
    }
    
    if (all_front && all_back) { out_front.push_back(poly); return; }
    if (all_front) { out_front.push_back(poly); return; }
    if (all_back) { out_back.push_back(poly); return; }
    
    for (int i = 0; i < n; ++i) {
        int j = (i + 1) % n;
        vec3 u = poly.verts[i];
        vec3 v = poly.verts[j];
        double d_u = dists[i];
        double d_v = dists[j];

        if (d_u > EPS) front_verts.push_back(u);
        else if (d_u < -EPS) back_verts.push_back(u);
        else { front_verts.push_back(u); back_verts.push_back(u); }

        if ((d_u > EPS && d_v < -EPS) || (d_u < -EPS && d_v > EPS)) {
            double t = d_u / (d_u - d_v);
            vec3 w = u + (v - u) * t;
            front_verts.push_back(w);
            back_verts.push_back(w);
        }
    }
    
    if (front_verts.size() >= 3) out_front.push_back({front_verts});
    if (back_verts.size() >= 3) out_back.push_back({back_verts});
}

bool tri_tri_overlap(vec3 V0, vec3 V1, vec3 V2, vec3 U0, vec3 U1, vec3 U2, double EPS) {
    vec3 N_U = cross(U1 - U0, U2 - U0);
    double d_U0 = dot(N_U, V0 - U0);
    double d_U1 = dot(N_U, V1 - U0);
    double d_U2 = dot(N_U, V2 - U0);
    
    if ((d_U0 > EPS && d_U1 > EPS && d_U2 > EPS) || 
        (d_U0 < -EPS && d_U1 < -EPS && d_U2 < -EPS)) return false;
    
    vec3 N_V = cross(V1 - V0, V2 - V0);
    double d_V0 = dot(N_V, U0 - V0);
    double d_V1 = dot(N_V, U1 - V0);
    double d_V2 = dot(N_V, U2 - V0);
    
    if ((d_V0 > EPS && d_V1 > EPS && d_V2 > EPS) || 
        (d_V0 < -EPS && d_V1 < -EPS && d_V2 < -EPS)) return false;
    
    vec3 D = cross(N_U, N_V);
    if (length(D) < 1e-12) return true;
    
    auto get_interval = [&](double d0, double d1, double d2, vec3 p0, vec3 p1, vec3 p2, double& out_min, double& out_max) {
        double p[3]; int idx = 0;
        auto add = [&](double da, double db, vec3 pa, vec3 pb) {
            if (abs(da - db) < 1e-12) {
                if (abs(da) < 1e-12) { p[idx++] = dot(D, pa); p[idx++] = dot(D, pb); }
            } else if (da * db <= 0) {
                p[idx++] = dot(D, pa + (pb - pa) * (da / (da - db)));
            }
        };
        add(d0, d1, p0, p1); add(d1, d2, p1, p2); add(d2, d0, p2, p0);
        if (idx >= 2) {
            out_min = p[0]; out_max = p[0];
            for (int i=1; i<idx; ++i) { out_min = min(out_min, p[i]); out_max = max(out_max, p[i]); }
        } else { out_min = 0; out_max = -1; }
    };
    
    double min_V, max_V, min_U, max_U;
    get_interval(d_U0, d_U1, d_U2, V0, V1, V2, min_V, max_V);
    get_interval(d_V0, d_V1, d_V2, U0, U1, U2, min_U, max_U);
    
    if (min_V > max_V || min_U > max_U) return true;
    if (max_V < min_U - EPS || max_U < min_V - EPS) return false;
    return true;
}

double triangle_area(vec3 v0, vec3 v1, vec3 v2) {
    return length(cross(v1 - v0, v2 - v0)) * 0.5;
}

struct TaskOutput {
    vector<vec3> verts;
    vector<vec3i> faces;
};

struct Plane { vec3 p, n; };

void process_triangle(const Mesh& A, const Mesh& B, const BVH& bvh_B, int t_idx, TaskOutput& out, bool is_A, double EPS, double AREA_EPS) {
    vec3 V0 = A.verts[A.faces[t_idx][0]];
    vec3 V1 = A.verts[A.faces[t_idx][1]];
    vec3 V2 = A.verts[A.faces[t_idx][2]];
    vec3 n_A = normalize(cross(V1 - V0, V2 - V0));
    
    AABB box;
    box.expand(V0); box.expand(V1); box.expand(V2);
    box.bmin = box.bmin - vec3{EPS, EPS, EPS};
    box.bmax = box.bmax + vec3{EPS, EPS, EPS};
    
    vector<int> candidate_tris;
    query_bvh(0, bvh_B.nodes.data(), box, candidate_tris);
    
    vector<Plane> splitting_planes;
    for (int tb_idx : candidate_tris) {
        vec3 U0 = B.verts[B.faces[tb_idx][0]];
        vec3 U1 = B.verts[B.faces[tb_idx][1]];
        vec3 U2 = B.verts[B.faces[tb_idx][2]];
        vec3 n_B = normalize(cross(U1 - U0, U2 - U0));
        
        double d0 = dot(n_B, V0 - U0);
        double d1 = dot(n_B, V1 - U0);
        double d2 = dot(n_B, V2 - U0);
        
        if (abs(d0) < EPS && abs(d1) < EPS && abs(d2) < EPS) {
            auto add_edge = [&](vec3 u, vec3 v) {
                vec3 edge_dir = v - u;
                if (length(edge_dir) > 1e-12) {
                    splitting_planes.push_back({u, normalize(cross(edge_dir, n_A))});
                }
            };
            add_edge(U0, U1); add_edge(U1, U2); add_edge(U2, U0);
        } else {
            if (tri_tri_overlap(V0, V1, V2, U0, U1, U2, EPS)) {
                splitting_planes.push_back({U0, n_B});
            }
        }
    }
    
    vector<Polygon> polys = { { {V0, V1, V2} } };
    for (const auto& plane : splitting_planes) {
        vector<Polygon> next_polys;
        for (const auto& poly : polys) {
            split_polygon(poly, plane.p, plane.n, next_polys, next_polys, EPS);
        }
        polys = next_polys;
    }
    
    for (const auto& poly : polys) {
        if (poly.verts.size() < 3) continue;
        for (size_t i = 1; i < poly.verts.size() - 1; ++i) {
            vec3 t0 = poly.verts[0];
            vec3 t1 = poly.verts[i];
            vec3 t2 = poly.verts[i+1];
            
            if (triangle_area(t0, t1, t2) < AREA_EPS) continue;
            
            vec3 centroid = (t0 + t1 + t2) / 3.0;
            vec3 hit_normal;
            Class c = robust_classify(centroid, n_A, bvh_B, B, hit_normal, EPS);
            
            if (c == OUTSIDE) {
                out.verts.push_back(t0); out.verts.push_back(t1); out.verts.push_back(t2);
                out.faces.push_back({(int)out.verts.size()-3, (int)out.verts.size()-2, (int)out.verts.size()-1});
            } else if (c == ON) {
                if (is_A) {
                    if (dot(n_A, hit_normal) > 0) {
                        out.verts.push_back(t0); out.verts.push_back(t1); out.verts.push_back(t2);
                        out.faces.push_back({(int)out.verts.size()-3, (int)out.verts.size()-2, (int)out.verts.size()-1});
                    }
                }
            }
        }
    }
}

struct VertRec {
    vec3 v;
    int idx;
    bool operator<(const VertRec& o) const { return v.x < o.v.x; }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    
    Mesh A, B;
    A.verts.resize(VA); A.faces.resize(FA);
    for(int i=0; i<VA; ++i) cin >> A.verts[i].x >> A.verts[i].y >> A.verts[i].z;
    for(int i=0; i<FA; ++i) cin >> A.faces[i][0] >> A.faces[i][1] >> A.faces[i][2];
    
    B.verts.resize(VB); B.faces.resize(FB);
    for(int i=0; i<VB; ++i) cin >> B.verts[i].x >> B.verts[i].y >> B.verts[i].z;
    for(int i=0; i<FB; ++i) cin >> B.faces[i][0] >> B.faces[i][1] >> B.faces[i][2];
    
    AABB scene_box;
    for(auto v : A.verts) scene_box.expand(v);
    for(auto v : B.verts) scene_box.expand(v);
    vec3 ext = scene_box.bmax - scene_box.bmin;
    double max_ext = max({ext.x, ext.y, ext.z});
    double EPS = max_ext * 1e-7;
    double AREA_EPS = max_ext * max_ext * 1e-14;
    
    BVH bvh_A, bvh_B;
    build_bvh(bvh_A, A);
    build_bvh(bvh_B, B);
    
    vector<TaskOutput> outputs_A(FA);
    for_each(execution::par, A.faces.begin(), A.faces.end(), [&](const vec3i& f) {
        int idx = &f - A.faces.data();
        process_triangle(A, B, bvh_B, idx, outputs_A[idx], true, EPS, AREA_EPS);
    });
    
    vector<TaskOutput> outputs_B(FB);
    for_each(execution::par, B.faces.begin(), B.faces.end(), [&](const vec3i& f) {
        int idx = &f - B.faces.data();
        process_triangle(B, A, bvh_A, idx, outputs_B[idx], false, EPS, AREA_EPS);
    });
    
    int total_verts = 0;
    for(auto& out : outputs_A) total_verts += out.verts.size();
    for(auto& out : outputs_B) total_verts += out.verts.size();
    
    vector<vec3> all_verts; all_verts.reserve(total_verts);
    vector<vec3i> all_faces;
    
    int current_v_idx = 0;
    for(auto& out : outputs_A) {
        all_verts.insert(all_verts.end(), out.verts.begin(), out.verts.end());
        for(auto f : out.faces) all_faces.push_back({f[0]+current_v_idx, f[1]+current_v_idx, f[2]+current_v_idx});
        current_v_idx += out.verts.size();
    }
    for(auto& out : outputs_B) {
        all_verts.insert(all_verts.end(), out.verts.begin(), out.verts.end());
        for(auto f : out.faces) all_faces.push_back({f[0]+current_v_idx, f[1]+current_v_idx, f[2]+current_v_idx});
        current_v_idx += out.verts.size();
    }
    
    vector<VertRec> vrec(total_verts);
    for(int i=0; i<total_verts; ++i) vrec[i] = {all_verts[i], i};
    sort(execution::par, vrec.begin(), vrec.end());
    
    vector<int> remap(total_verts, -1);
    vector<vec3> final_verts;
    
    for (int i = 0; i < total_verts; ++i) {
        if (remap[vrec[i].idx] != -1) continue;
        int new_idx = final_verts.size();
        final_verts.push_back(vrec[i].v);
        remap[vrec[i].idx] = new_idx;
        
        for (int j = i + 1; j < total_verts; ++j) {
            if (vrec[j].v.x - vrec[i].v.x > EPS) break;
            if (remap[vrec[j].idx] == -1) {
                if (abs(vrec[j].v.y - vrec[i].v.y) <= EPS && abs(vrec[j].v.z - vrec[i].v.z) <= EPS) {
                    remap[vrec[j].idx] = new_idx;
                }
            }
        }
    }
    
    vector<vec3i> final_faces;
    for (auto f : all_faces) {
        int i0 = remap[f[0]], i1 = remap[f[1]], i2 = remap[f[2]];
        if (i0 == i1 || i1 == i2 || i2 == i0) continue;
        if (triangle_area(final_verts[i0], final_verts[i1], final_verts[i2]) < AREA_EPS) continue;
        final_faces.push_back({i0, i1, i2});
    }
    
    cout << final_verts.size() << " " << final_faces.size() << "\n";
    for(auto v : final_verts) cout << v.x << " " << v.y << " " << v.z << "\n";
    for(auto f : final_faces) cout << f[0] << " " << f[1] << " " << f[2] << "\n";
    
    return 0;
}
