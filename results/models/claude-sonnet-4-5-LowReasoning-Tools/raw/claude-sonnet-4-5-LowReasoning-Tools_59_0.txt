const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { probability, drones, maxStepsPerDrone } = data;
const rows = probability.length;
const cols = probability[0].length;

// Current positions and paths
const positions = drones.map(d => [d.x, d.y]);
const paths = drones.map(d => [[d.x, d.y]]);
const covered = new Set();

// Mark starting positions as covered
positions.forEach(([x, y]) => covered.add(`${x},${y}`));

// Helper function to check if position is valid
function isValid(x, y) {
    return x >= 0 && x < cols && y >= 0 && y < rows && probability[y][x] !== -1;
}

// Calculate adaptive lookahead radius based on grid size
const radius = Math.min(6, Math.max(3, Math.ceil(Math.sqrt(rows * cols) / 30)));

// Score a position based on coverage potential
function score(x, y, covered, others) {
    let s = 0;
    const key = `${x},${y}`;

    // Immediate value (high weight for uncovered cells)
    if (!covered.has(key) && probability[y][x] > 0) {
        s += probability[y][x] * 100;
    }

    // Future potential (nearby uncovered cells within radius)
    for (let dy = -radius; dy <= radius; dy++) {
        for (let dx = -radius; dx <= radius; dx++) {
            if (dx === 0 && dy === 0) continue;
            const ny = y + dy;
            const nx = x + dx;
            if (isValid(nx, ny)) {
                const nkey = `${nx},${ny}`;
                if (!covered.has(nkey) && probability[ny][nx] > 0) {
                    const dist = Math.abs(dx) + Math.abs(dy);
                    s += probability[ny][nx] / (dist * 0.5);
                }
            }
        }
    }

    // Avoid clustering with other drones (encourage spread)
    for (const [ox, oy] of others) {
        const d = Math.abs(x - ox) + Math.abs(y - oy);
        if (d > 0 && d < radius * 1.5) {
            s -= 10 / d;
        }
    }

    return s;
}

// Execute steps
for (let step = 0; step < maxStepsPerDrone; step++) {
    const next = [];

    // Calculate best move for each drone
    for (let i = 0; i < drones.length; i++) {
        const [x, y] = positions[i];
        const otherPos = positions.filter((_, j) => j !== i);

        let best = [x, y];
        let bestScore = -Infinity;

        // Try all possible moves: stay, left, right, up, down
        for (const [dx, dy] of [[0, 0], [-1, 0], [1, 0], [0, -1], [0, 1]]) {
            const nx = x + dx;
            const ny = y + dy;

            if (isValid(nx, ny)) {
                const s = score(nx, ny, covered, otherPos);
                if (s > bestScore) {
                    bestScore = s;
                    best = [nx, ny];
                }
            }
        }

        next.push(best);
    }

    // Apply moves and update coverage
    for (let i = 0; i < drones.length; i++) {
        positions[i] = next[i];
        paths[i].push(next[i]);
        covered.add(`${next[i][0]},${next[i][1]}`);
    }
}
