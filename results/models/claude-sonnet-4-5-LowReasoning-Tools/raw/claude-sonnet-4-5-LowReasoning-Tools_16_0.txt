#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <limits>

using namespace std;

struct Task {
    int job;
    int task_idx;
    int machine;
    int duration;
    int start_time;
};

struct Job {
    vector<int> machines;
    vector<int> durations;
    vector<int> start_times;
    int next_task;
};

// Event for priority queue: when a task can be scheduled
struct Event {
    int time;
    int job;

    bool operator>(const Event& other) const {
        if (time != other.time) return time > other.time;
        return job > other.job;
    }
};

class JobShopScheduler {
private:
    int num_jobs;
    int num_machines;
    vector<Job> jobs;
    vector<int> machine_available;

public:
    JobShopScheduler(int j, int m) : num_jobs(j), num_machines(m) {
        jobs.resize(j);
        machine_available.resize(m, 0);
    }

    void add_job(int job_id, const vector<int>& machines, const vector<int>& durations) {
        jobs[job_id].machines = machines;
        jobs[job_id].durations = durations;
        jobs[job_id].start_times.resize(machines.size());
        jobs[job_id].next_task = 0;
    }

    void solve() {
        // Use greedy dispatch rule for scalability
        // Tracks when each job can schedule its next task
        vector<int> job_available(num_jobs, 0);
        vector<bool> job_done(num_jobs, false);

        int completed_jobs = 0;

        while (completed_jobs < num_jobs) {
            // Find the next task to schedule
            int best_job = -1;
            int best_time = numeric_limits<int>::max();

            for (int j = 0; j < num_jobs; j++) {
                if (job_done[j]) continue;

                int task_idx = jobs[j].next_task;
                if (task_idx >= jobs[j].machines.size()) {
                    job_done[j] = true;
                    completed_jobs++;
                    continue;
                }

                int machine = jobs[j].machines[task_idx];
                int earliest_start = max(job_available[j], machine_available[machine]);

                // Use earliest start time as priority (greedy)
                if (earliest_start < best_time ||
                    (earliest_start == best_time && j < best_job)) {
                    best_time = earliest_start;
                    best_job = j;
                }
            }

            if (best_job == -1) break;

            // Schedule the selected task
            int task_idx = jobs[best_job].next_task;
            int machine = jobs[best_job].machines[task_idx];
            int duration = jobs[best_job].durations[task_idx];
            int start_time = max(job_available[best_job], machine_available[machine]);

            jobs[best_job].start_times[task_idx] = start_time;
            jobs[best_job].next_task++;
            job_available[best_job] = start_time + duration;
            machine_available[machine] = start_time + duration;
        }
    }

    void print_solution() {
        for (int j = 0; j < num_jobs; j++) {
            for (size_t t = 0; t < jobs[j].machines.size(); t++) {
                if (t > 0) cout << " ";
                cout << jobs[j].start_times[t] << " " << jobs[j].machines[t];
            }
            cout << "\n";
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    JobShopScheduler scheduler(J, M);

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;

        vector<int> machines(T);
        vector<int> durations(T);

        for (int t = 0; t < T; t++) {
            cin >> machines[t] >> durations[t];
        }

        scheduler.add_job(j, machines, durations);
    }

    scheduler.solve();
    scheduler.print_solution();

    return 0;
}
