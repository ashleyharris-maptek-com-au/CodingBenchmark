OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %out_color
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 10
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4
     %uint_5 = OpConstant %uint 5

  %float_0_0 = OpConstant %float 0.0
%float_0_25 = OpConstant %float 0.25
  %float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
  %float_1_0 = OpConstant %float 1.0
 %float_0_02 = OpConstant %float 0.02

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
  %out_color = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_func_v4 = OpTypePointer Function %v4float
%ptr_func_v3 = OpTypePointer Function %v3float

       %main = OpFunction %void None %3
      %label = OpLabel

; Load 4 input positions
  %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %p0 = OpLoad %v4float %p0_ptr
  %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
      %p1 = OpLoad %v4float %p1_ptr
  %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
      %p2 = OpLoad %v4float %p2_ptr
  %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
      %p3 = OpLoad %v4float %p3_ptr

; Load colors at indices 1 and 2
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
     %c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
     %c2 = OpLoad %v3float %c2_ptr

; Compute tangent at start: (P2 - P0) for direction
 %tan_s = OpFSub %v4float %p2 %p0
; Compute tangent at end: (P3 - P1) for direction
 %tan_e = OpFSub %v4float %p3 %p1

; Average tangent for perpendicular computation
%tan_avg_tmp = OpFAdd %v4float %tan_s %tan_e
%tan_avg = OpVectorTimesScalar %v4float %tan_avg_tmp %float_0_5

; Extract xy components for perpendicular (assuming 2D in xy plane)
%tan_x = OpCompositeExtract %float %tan_avg 0
%tan_y = OpCompositeExtract %float %tan_avg 1

; Perpendicular: rotate 90 degrees: (-y, x, 0, 0)
%perp_x = OpFNegate %float %tan_y
%perp_y = %tan_x
; Normalize perpendicular and scale by thickness
%perp_unnorm = OpCompositeConstruct %v4float %perp_x %perp_y %float_0_0 %float_0_0

; Compute length for normalization
%perp_x_sq = OpFMul %float %perp_x %perp_x
%perp_y_sq = OpFMul %float %perp_y %perp_y
%perp_len_sq = OpFAdd %float %perp_x_sq %perp_y_sq
%perp_len = OpExtInst %float %1 Sqrt %perp_len_sq
%perp_scale = OpFDiv %float %float_0_02 %perp_len
%perp_norm = OpVectorTimesScalar %v4float %perp_unnorm %perp_scale

; Point 0 (t=0.0): interpolate P1 and P2
     %t0 = %float_0_0
 %inv_t0 = OpFSub %float %float_1_0 %t0
 %pos0_a = OpVectorTimesScalar %v4float %p1 %inv_t0
 %pos0_b = OpVectorTimesScalar %v4float %p2 %t0
   %pos0 = OpFAdd %v4float %pos0_a %pos0_b
%col0_a = OpVectorTimesScalar %v3float %c1 %inv_t0
%col0_b = OpVectorTimesScalar %v3float %c2 %t0
   %col0 = OpFAdd %v3float %col0_a %col0_b

; Emit vertex 0 top (pos + perp)
%pos0_top = OpFAdd %v4float %pos0 %perp_norm
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
          OpStore %out_pos_ptr %pos0_top
          OpStore %out_color %col0
          OpEmitVertex

; Emit vertex 0 bottom (pos - perp)
%pos0_bot = OpFSub %v4float %pos0 %perp_norm
          OpStore %out_pos_ptr %pos0_bot
          OpStore %out_color %col0
          OpEmitVertex

; Point 1 (t=0.25)
     %t1 = %float_0_25
 %inv_t1 = OpFSub %float %float_1_0 %t1
 %pos1_a = OpVectorTimesScalar %v4float %p1 %inv_t1
 %pos1_b = OpVectorTimesScalar %v4float %p2 %t1
   %pos1 = OpFAdd %v4float %pos1_a %pos1_b
%col1_a = OpVectorTimesScalar %v3float %c1 %inv_t1
%col1_b = OpVectorTimesScalar %v3float %c2 %t1
   %col1 = OpFAdd %v3float %col1_a %col1_b

%pos1_top = OpFAdd %v4float %pos1 %perp_norm
          OpStore %out_pos_ptr %pos1_top
          OpStore %out_color %col1
          OpEmitVertex

%pos1_bot = OpFSub %v4float %pos1 %perp_norm
          OpStore %out_pos_ptr %pos1_bot
          OpStore %out_color %col1
          OpEmitVertex

; Point 2 (t=0.5)
     %t2 = %float_0_5
 %inv_t2 = OpFSub %float %float_1_0 %t2
 %pos2_a = OpVectorTimesScalar %v4float %p1 %inv_t2
 %pos2_b = OpVectorTimesScalar %v4float %p2 %t2
   %pos2 = OpFAdd %v4float %pos2_a %pos2_b
%col2_a = OpVectorTimesScalar %v3float %c1 %inv_t2
%col2_b = OpVectorTimesScalar %v3float %c2 %t2
   %col2 = OpFAdd %v3float %col2_a %col2_b

%pos2_top = OpFAdd %v4float %pos2 %perp_norm
          OpStore %out_pos_ptr %pos2_top
          OpStore %out_color %col2
          OpEmitVertex

%pos2_bot = OpFSub %v4float %pos2 %perp_norm
          OpStore %out_pos_ptr %pos2_bot
          OpStore %out_color %col2
          OpEmitVertex

; Point 3 (t=0.75)
     %t3 = %float_0_75
 %inv_t3 = OpFSub %float %float_1_0 %t3
 %pos3_a = OpVectorTimesScalar %v4float %p1 %inv_t3
 %pos3_b = OpVectorTimesScalar %v4float %p2 %t3
   %pos3 = OpFAdd %v4float %pos3_a %pos3_b
%col3_a = OpVectorTimesScalar %v3float %c1 %inv_t3
%col3_b = OpVectorTimesScalar %v3float %c2 %t3
   %col3 = OpFAdd %v3float %col3_a %col3_b

%pos3_top = OpFAdd %v4float %pos3 %perp_norm
          OpStore %out_pos_ptr %pos3_top
          OpStore %out_color %col3
          OpEmitVertex

%pos3_bot = OpFSub %v4float %pos3 %perp_norm
          OpStore %out_pos_ptr %pos3_bot
          OpStore %out_color %col3
          OpEmitVertex

; Point 4 (t=1.0)
     %t4 = %float_1_0
 %inv_t4 = OpFSub %float %float_1_0 %t4
 %pos4_a = OpVectorTimesScalar %v4float %p1 %inv_t4
 %pos4_b = OpVectorTimesScalar %v4float %p2 %t4
   %pos4 = OpFAdd %v4float %pos4_a %pos4_b
%col4_a = OpVectorTimesScalar %v3float %c1 %inv_t4
%col4_b = OpVectorTimesScalar %v3float %c2 %t4
   %col4 = OpFAdd %v3float %col4_a %col4_b

%pos4_top = OpFAdd %v4float %pos4 %perp_norm
          OpStore %out_pos_ptr %pos4_top
          OpStore %out_color %col4
          OpEmitVertex

%pos4_bot = OpFSub %v4float %pos4 %perp_norm
          OpStore %out_pos_ptr %pos4_bot
          OpStore %out_color %col4
          OpEmitVertex

          OpEndPrimitive
          OpReturn
          OpFunctionEnd
