OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1
               OpSource GLSL 450
               OpName %main "main"
               OpName %gl_PerVertex_in "gl_PerVertex"
               OpMemberName %gl_PerVertex_in 0 "gl_Position"
               OpName %gl_in "gl_in"
               OpName %color_in "color"
               OpName %gl_PerVertex_out "gl_PerVertex"
               OpMemberName %gl_PerVertex_out 0 "gl_Position"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_color_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_color Input
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output
  %float_1_3 = OpConstant %float 0.333333343
 %float_0_08 = OpConstant %float 0.08
    %float_0 = OpConstant %float 0
       %main = OpFunction %void None %3
          %5 = OpLabel
               ; Load positions from gl_in[0..2].gl_Position
         %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
         %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr
         %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr
               ; Compute center = (p0 + p1 + p2) / 3 (xyz components only)
         %p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
         %p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
         %p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
         %sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
         %sum = OpFAdd %v3float %sum01 %p2_xyz
         %center = OpVectorTimesScalar %v3float %sum %float_1_3
               ; Compute edge vectors: e1 = p1 - p0, e2 = p2 - p0
         %e1 = OpFSub %v3float %p1_xyz %p0_xyz
         %e2 = OpFSub %v3float %p2_xyz %p0_xyz
               ; Compute cross product: N = cross(e1, e2)
         %N_raw = OpExtInst %v3float %1 Cross %e1 %e2
               ; Normalize N
         %N = OpExtInst %v3float %1 Normalize %N_raw
               ; Compute offset = N * 0.08
         %offset = OpVectorTimesScalar %v3float %N %float_0_08
               ; Emit vertex 0
         %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr
         %offset_v4_0 = OpCompositeConstruct %v4float %offset %float_0
         %new_p0 = OpFAdd %v4float %p0 %offset_v4_0
         %out_p0_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_p0_ptr %new_p0
               OpStore %outColor %c0
               OpEmitVertex
               ; Emit vertex 1
         %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr
         %offset_v4_1 = OpCompositeConstruct %v4float %offset %float_0
         %new_p1 = OpFAdd %v4float %p1 %offset_v4_1
         %out_p1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_p1_ptr %new_p1
               OpStore %outColor %c1
               OpEmitVertex
               ; Emit vertex 2
         %c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
         %c2 = OpLoad %v3float %c2_ptr
         %offset_v4_2 = OpCompositeConstruct %v4float %offset %float_0
         %new_p2 = OpFAdd %v4float %p2 %offset_v4_2
         %out_p2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_p2_ptr %new_p2
               OpStore %outColor %c2
               OpEmitVertex
               ; End primitive
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
