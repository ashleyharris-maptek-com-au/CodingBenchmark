OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_2 = OpConstant %uint 2
     %uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_color Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
  %float_0 = OpConstant %float 0.0

       %main = OpFunction %void None %3
          %5 = OpLabel

; Load p0 = gl_in[0].gl_Position
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr

; Load p1 = gl_in[1].gl_Position
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr

; Load c0 = color_in[0]
     %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr

; Load c1 = color_in[1]
     %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr

; Compute direction: dir = p1 - p0 (vec3 part)
    %p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
    %p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
        %dir = OpFSub %v3float %p1_xyz %p0_xyz

; Extract dir.x and dir.y
      %dir_x = OpCompositeExtract %float %dir 0
      %dir_y = OpCompositeExtract %float %dir 1

; Compute perpendicular: perp_unnorm = vec3(-dir.y, dir.x, 0.0)
   %neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0

; Normalize perpendicular
       %perp = OpExtInst %v3float %1 Normalize %perp_unnorm

; Compute thick_start and thick_end offsets
%offset_start = OpVectorTimesScalar %v3float %perp %float_0_04
 %offset_end = OpVectorTimesScalar %v3float %perp %float_0_005

; Vertex 0: p0 + perp*thick_start
     %v0_xyz = OpFAdd %v3float %p0_xyz %offset_start
         %v0 = OpVectorShuffle %v4float %v0_xyz %p0 0 1 2 3
    %out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_ptr %v0
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 1: p0 - perp*thick_start
     %v1_xyz = OpFSub %v3float %p0_xyz %offset_start
         %v1 = OpVectorShuffle %v4float %v1_xyz %p0 0 1 2 3
               OpStore %out_ptr %v1
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 2: p1 + perp*thick_end
     %v2_xyz = OpFAdd %v3float %p1_xyz %offset_end
         %v2 = OpVectorShuffle %v4float %v2_xyz %p1 0 1 2 3
               OpStore %out_ptr %v2
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 3: p1 - perp*thick_end
     %v3_xyz = OpFSub %v3float %p1_xyz %offset_end
         %v3 = OpVectorShuffle %v4float %v3_xyz %p1 0 1 2 3
               OpStore %out_ptr %v3
               OpStore %outColor %c1
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
