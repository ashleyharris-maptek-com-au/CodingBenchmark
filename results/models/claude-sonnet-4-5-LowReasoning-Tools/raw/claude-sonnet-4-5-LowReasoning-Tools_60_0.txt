const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { machineCount, tasks } = data;

// Build data structures
const taskMap = new Map(tasks.map(t => [t.id, t]));
const children = new Map(tasks.map(t => [t.id, []]));

for (const task of tasks) {
    for (const depId of (task.deps || [])) {
        if (children.has(depId)) {
            children.get(depId).push(task.id);
        }
    }
}

// Compute earliest finish times (forward pass through dependencies)
const earliestStart = new Map();
const earliestFinish = new Map();

function computeEarliest(taskId, memo = new Map()) {
    if (memo.has(taskId)) return memo.get(taskId);

    const task = taskMap.get(taskId);
    let maxDepFinish = 0;

    for (const depId of (task.deps || [])) {
        maxDepFinish = Math.max(maxDepFinish, computeEarliest(depId, memo));
    }

    const start = maxDepFinish;
    const finish = start + task.duration;

    memo.set(taskId, finish);
    earliestStart.set(taskId, start);
    earliestFinish.set(taskId, finish);

    return finish;
}

for (const task of tasks) {
    computeEarliest(task.id);
}

// Compute critical path length (longest path from each task to any end)
const cpLength = new Map();

function computeCP(taskId, memo = new Map()) {
    if (memo.has(taskId)) return memo.get(taskId);

    const task = taskMap.get(taskId);
    let maxPath = 0;

    for (const childId of children.get(taskId)) {
        maxPath = Math.max(maxPath, computeCP(childId, memo));
    }

    const result = task.duration + maxPath;
    memo.set(taskId, result);
    return result;
}

for (const task of tasks) {
    cpLength.set(task.id, computeCP(task.id));
}

// Priority function: balances weighted lateness, makespan, and efficiency
function getPriority(task) {
    const ef = earliestFinish.get(task.id);
    const slack = task.deadline - ef;
    const cp = cpLength.get(task.id);

    let priority = 0;

    // Weighted urgency: higher weight and tighter deadline = higher priority
    priority += task.weight * 1000 / Math.max(1, slack + task.duration);

    // Critical path: longer critical path = higher priority (reduces makespan)
    priority += cp;

    // Weighted shortest processing time: high value per time unit
    priority += (task.weight / task.duration) * 10;

    return priority;
}

// Build topological order using priority-based selection
const order = [];
const scheduled = new Set();
const ready = [];

// Initialize with tasks that have no dependencies
for (const task of tasks) {
    if (!task.deps || task.deps.length === 0) {
        ready.push(task.id);
    }
}

// Greedily schedule highest priority ready task
while (ready.length > 0) {
    // Sort by priority (descending) and pick the best
    ready.sort((a, b) => getPriority(taskMap.get(b)) - getPriority(taskMap.get(a)));
    const taskId = ready.shift();

    order.push(taskId);
    scheduled.add(taskId);

    // Add newly ready children whose dependencies are all satisfied
    for (const childId of children.get(taskId)) {
        const child = taskMap.get(childId);
        if (child.deps.every(d => scheduled.has(d))) {
            ready.push(childId);
        }
    }
}
