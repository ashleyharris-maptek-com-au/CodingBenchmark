#include <iostream>
#include <fstream>
#include <string>
#include <vector>
#include <thread>
#include <mutex>
#include <atomic>
#include <cmath>
#include <algorithm>
#include <sstream>
#include <chrono>

struct Terrain {
    int width, depth, height;
    std::vector<uint16_t> data; // stored as triplets: ground_top, float_bottom, float_top

    bool isLoaded = false;

    void load(const std::string& path, int w, int d, int h) {
        width = w;
        depth = d;
        height = h;

        std::ifstream file(path, std::ios::binary);
        if (!file) {
            std::cerr << "Failed to open terrain file: " << path << std::endl;
            return;
        }

        size_t expectedSize = static_cast<size_t>(width) * depth * 3;
        data.resize(expectedSize);

        for (size_t i = 0; i < expectedSize; i++) {
            uint16_t val;
            file.read(reinterpret_cast<char*>(&val), sizeof(uint16_t));
            data[i] = val;
        }

        isLoaded = true;
    }

    bool isSolid(double x, double y, double z) const {
        if (!isLoaded) return false;

        int ix = static_cast<int>(x);
        int iy = static_cast<int>(y);
        int iz = static_cast<int>(z);

        // Out of bounds check
        if (ix < 0 || ix >= width || iy < 0 || iy >= depth || iz < 0 || iz >= height) {
            return true;
        }

        size_t idx = (static_cast<size_t>(ix) * depth + iy) * 3;
        if (idx + 2 >= data.size()) return true;

        uint16_t ground_top = data[idx];
        uint16_t float_bottom = data[idx + 1];
        uint16_t float_top = data[idx + 2];

        // z < ground_top: solid ground
        if (iz < ground_top) return true;

        // float_bottom <= z < float_top: solid floating rock
        if (iz >= float_bottom && iz < float_top) return true;

        return false;
    }
};

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    std::chrono::steady_clock::time_point timestamp;
};

struct GameParams {
    int width, depth, height;
    double gravity;
    double max_thrust;
    double fuel;
    double start_x, start_y, start_z;
    double target_x, target_y, target_z;
    std::string terrain_path;
};

class LunarLanderController {
private:
    GameParams params;
    Terrain terrain;
    State currentState;
    std::mutex stateMutex;
    std::atomic<bool> running{true};
    std::atomic<bool> stateReady{false};

    // PID gains
    const double kp_pos = 0.15;
    const double kd_pos = 0.8;
    const double kp_vel = 0.3;

    const double kp_angle = 2.0;
    const double kd_angle = 1.5;

public:
    void readInput() {
        // Read header
        std::cin >> params.width >> params.depth >> params.height;
        std::cin >> params.gravity >> params.max_thrust >> params.fuel;
        std::cin >> params.start_x >> params.start_y >> params.start_z;
        std::cin >> params.target_x >> params.target_y >> params.target_z;
        std::cin.ignore(); // consume newline

        std::getline(std::cin, params.terrain_path);

        std::string stateLine;
        std::getline(std::cin, stateLine); // Read "STATE" line

        // Start terrain loading in background
        std::thread terrainThread([this]() {
            terrain.load(params.terrain_path, params.width, params.depth, params.height);
        });
        terrainThread.detach();

        // Read state updates
        while (running) {
            State newState;
            if (!(std::cin >> newState.x >> newState.y >> newState.z
                  >> newState.vx >> newState.vy >> newState.vz
                  >> newState.pitch >> newState.yaw >> newState.roll
                  >> newState.pitch_rate >> newState.yaw_rate >> newState.roll_rate
                  >> newState.fuel)) {
                break;
            }

            newState.timestamp = std::chrono::steady_clock::now();

            {
                std::lock_guard<std::mutex> lock(stateMutex);
                currentState = newState;
                stateReady = true;
            }
        }
    }

    void controlLoop() {
        // Wait for first state
        while (!stateReady && running) {
            std::this_thread::sleep_for(std::chrono::milliseconds(10));
        }

        auto lastOutput = std::chrono::steady_clock::now();
        const auto outputInterval = std::chrono::milliseconds(50); // 20Hz

        while (running) {
            auto now = std::chrono::steady_clock::now();

            if (now - lastOutput >= outputInterval) {
                State state;
                {
                    std::lock_guard<std::mutex> lock(stateMutex);
                    state = currentState;
                }

                // Compute control commands
                double thrust, pitch_cmd, yaw_cmd, roll_cmd;
                computeControl(state, thrust, pitch_cmd, yaw_cmd, roll_cmd);

                // Output commands
                std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << std::endl;
                std::cout.flush();

                lastOutput = now;
            }

            std::this_thread::sleep_for(std::chrono::milliseconds(5));
        }
    }

    void computeControl(const State& state, double& thrust, double& pitch_cmd, double& yaw_cmd, double& roll_cmd) {
        // Calculate errors
        double dx = params.target_x - state.x;
        double dy = params.target_y - state.y;
        double dz = params.target_z - state.z;

        double dist_horizontal = std::sqrt(dx*dx + dy*dy);
        double dist_total = std::sqrt(dx*dx + dy*dy + dz*dz);

        // Target velocity based on distance (approach speed)
        double target_vz = 0.0;
        if (dz > 0) {
            // Descend slowly
            target_vz = std::max(-2.0, -dz * 0.3);
        } else {
            // Ascend if below target
            target_vz = std::min(2.0, -dz * 0.5);
        }

        // Horizontal target velocity
        double target_vx = 0.0, target_vy = 0.0;
        if (dist_horizontal > 1.0) {
            double approach_speed = std::min(3.0, dist_horizontal * 0.2);
            target_vx = (dx / dist_horizontal) * approach_speed;
            target_vy = (dy / dist_horizontal) * approach_speed;
        }

        // Velocity errors
        double vx_error = target_vx - state.vx;
        double vy_error = target_vy - state.vy;
        double vz_error = target_vz - state.vz;

        // Compute desired acceleration
        double ax_desired = kp_pos * dx + kd_pos * vx_error;
        double ay_desired = kp_pos * dy + kd_pos * vy_error;
        double az_desired = kp_pos * dz + kd_pos * vz_error + params.gravity;

        // Limit acceleration magnitude
        double a_horiz = std::sqrt(ax_desired*ax_desired + ay_desired*ay_desired);
        double max_horiz_accel = 3.0;
        if (a_horiz > max_horiz_accel) {
            ax_desired *= max_horiz_accel / a_horiz;
            ay_desired *= max_horiz_accel / a_horiz;
        }

        // Compute desired pitch and yaw from acceleration
        double desired_pitch = 0.0;
        double desired_yaw = state.yaw;

        double total_accel = std::sqrt(ax_desired*ax_desired + ay_desired*ay_desired + az_desired*az_desired);
        thrust = std::min(1.0, std::max(0.0, total_accel / params.max_thrust));

        // Adjust thrust for fuel conservation when close
        if (state.fuel < 20.0 && dist_total < 50.0) {
            thrust *= 0.7;
        }

        if (total_accel > 0.01) {
            // Compute pitch: tilt to redirect thrust
            double horiz_component = std::sqrt(ax_desired*ax_desired + ay_desired*ay_desired);
            desired_pitch = -std::atan2(horiz_component, az_desired);
            desired_pitch = std::max(-1.5, std::min(1.5, desired_pitch));

            // Compute yaw: direction of horizontal acceleration
            if (horiz_component > 0.01) {
                // In standard coords: X=East, Y=North
                // Yaw=0 points along +X axis
                // We need to point thrust direction towards (ax_desired, ay_desired)
                // But pitch already tilts us, so yaw determines which direction we tilt
                double angle_to_target = std::atan2(dy, dx);
                desired_yaw = angle_to_target;
            }
        }

        // Angle control
        double pitch_error = desired_pitch - state.pitch;
        double yaw_error = desired_yaw - state.yaw;

        // Normalize yaw error to [-pi, pi]
        while (yaw_error > M_PI) yaw_error -= 2*M_PI;
        while (yaw_error < -M_PI) yaw_error += 2*M_PI;

        pitch_cmd = kp_angle * pitch_error - kd_angle * state.pitch_rate;
        yaw_cmd = kp_angle * yaw_error - kd_angle * state.yaw_rate;
        roll_cmd = -kd_angle * state.roll_rate; // Dampen roll

        // Clamp commands
        pitch_cmd = std::max(-1.0, std::min(1.0, pitch_cmd));
        yaw_cmd = std::max(-1.0, std::min(1.0, yaw_cmd));
        roll_cmd = std::max(-1.0, std::min(1.0, roll_cmd));

        // Emergency handling
        double speed = std::sqrt(state.vx*state.vx + state.vy*state.vy + state.vz*state.vz);

        // If moving down too fast and close to ground, emergency thrust
        if (state.vz < -5.0 && state.z < 100.0) {
            thrust = 1.0;
            pitch_cmd = -kp_angle * state.pitch - kd_angle * state.pitch_rate;
            yaw_cmd = -kd_angle * state.yaw_rate;
        }

        // If very close to target, reduce thrust and level out
        if (dist_total < 10.0 && speed < 5.0) {
            thrust = 0.4 + 0.2 * (state.vz < 0 ? -state.vz : 0);
            thrust = std::min(1.0, thrust);
        }

        // No fuel = no thrust
        if (state.fuel <= 0) {
            thrust = 0.0;
        }
    }

    void run() {
        std::thread inputThread(&LunarLanderController::readInput, this);
        std::thread controlThread(&LunarLanderController::controlLoop, this);

        inputThread.join();
        running = false;
        controlThread.join();
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    LunarLanderController controller;
    controller.run();

    return 0;
}
