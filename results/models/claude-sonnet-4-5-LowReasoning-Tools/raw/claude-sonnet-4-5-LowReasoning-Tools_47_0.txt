@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;

fn rotl(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

fn countTrailingZeros64(h0: u32, h1: u32) -> u32 {
    if (h0 != 0u) {
        return countTrailingZeros(h0);
    } else {
        return 32u + countTrailingZeros(h1);
    }
}

fn custom_hash(nonce: u32) -> vec2<u32> {
    var state: array<u32, 16>;

    // Initialize state with baseData XOR (nonce + i)
    for (var i = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    // 8 rounds of quarter-round mixing
    for (var round = 0u; round < 8u; round = round + 1u) {
        // Group 0: state[0..3]
        state[0] = state[0] + state[1];
        state[3] = state[3] ^ state[0];
        state[3] = rotl(state[3], 16u);
        state[2] = state[2] + state[3];
        state[1] = state[1] ^ state[2];
        state[1] = rotl(state[1], 12u);
        state[0] = state[0] + state[1];
        state[3] = state[3] ^ state[0];
        state[3] = rotl(state[3], 8u);
        state[2] = state[2] + state[3];
        state[1] = state[1] ^ state[2];
        state[1] = rotl(state[1], 7u);

        // Group 1: state[4..7]
        state[4] = state[4] + state[5];
        state[7] = state[7] ^ state[4];
        state[7] = rotl(state[7], 16u);
        state[6] = state[6] + state[7];
        state[5] = state[5] ^ state[6];
        state[5] = rotl(state[5], 12u);
        state[4] = state[4] + state[5];
        state[7] = state[7] ^ state[4];
        state[7] = rotl(state[7], 8u);
        state[6] = state[6] + state[7];
        state[5] = state[5] ^ state[6];
        state[5] = rotl(state[5], 7u);

        // Group 2: state[8..11]
        state[8] = state[8] + state[9];
        state[11] = state[11] ^ state[8];
        state[11] = rotl(state[11], 16u);
        state[10] = state[10] + state[11];
        state[9] = state[9] ^ state[10];
        state[9] = rotl(state[9], 12u);
        state[8] = state[8] + state[9];
        state[11] = state[11] ^ state[8];
        state[11] = rotl(state[11], 8u);
        state[10] = state[10] + state[11];
        state[9] = state[9] ^ state[10];
        state[9] = rotl(state[9], 7u);

        // Group 3: state[12..15]
        state[12] = state[12] + state[13];
        state[15] = state[15] ^ state[12];
        state[15] = rotl(state[15], 16u);
        state[14] = state[14] + state[15];
        state[13] = state[13] ^ state[14];
        state[13] = rotl(state[13], 12u);
        state[12] = state[12] + state[13];
        state[15] = state[15] ^ state[12];
        state[15] = rotl(state[15], 8u);
        state[14] = state[14] + state[15];
        state[13] = state[13] ^ state[14];
        state[13] = rotl(state[13], 7u);
    }

    // Fold 16 words to 2
    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    return vec2<u32>(h0, h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    // Guard: check if thread is in range
    if (gid.x >= params.z) {
        return;
    }

    // Compute nonce for this thread
    let my_nonce = params.y + gid.x;

    // Compute hash
    let hash = custom_hash(my_nonce);
    let h0 = hash.x;
    let h1 = hash.y;

    // Count trailing zeros
    let trailing_zeros = countTrailingZeros64(h0, h1);

    // Check if we found a valid nonce
    if (trailing_zeros >= params.x) {
        result[0] = 1u;
        result[1] = my_nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
