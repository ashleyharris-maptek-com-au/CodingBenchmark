OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %inColor %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0

%gl_PerVertex_in = OpTypeStruct %v4float
     %uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
    %inColor = OpVariable %ptr_in_arr_color Input
%ptr_in_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float

%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output

  %float_0_05 = OpConstant %float 0.05
    %float_0 = OpConstant %float 0.0

       %main = OpFunction %void None %3
          %5 = OpLabel

; Load input position
         %20 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
         %21 = OpLoad %v4float %20

; Load input color
         %30 = OpAccessChain %ptr_in_v3float %inColor %int_0
         %31 = OpLoad %v3float %30

; Extract x and y from position
         %40 = OpCompositeExtract %float %21 0
         %41 = OpCompositeExtract %float %21 1
         %42 = OpCompositeExtract %float %21 2
         %43 = OpCompositeExtract %float %21 3

; Calculate offsets
         %50 = OpFAdd %float %41 %float_0_05  ; y + h (top)
         %51 = OpFAdd %float %40 %float_0_05  ; x + h (right)
         %52 = OpFSub %float %41 %float_0_05  ; y - h (bottom)
         %53 = OpFSub %float %40 %float_0_05  ; x - h (left)

; Emit vertex 0: Top (x, y+h)
         %60 = OpCompositeConstruct %v4float %40 %50 %42 %43
         %61 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %61 %60
               OpStore %outColor %31
               OpEmitVertex

; Emit vertex 1: Right (x+h, y)
         %70 = OpCompositeConstruct %v4float %51 %41 %42 %43
         %71 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %71 %70
               OpStore %outColor %31
               OpEmitVertex

; Emit vertex 2: Left (x-h, y)
         %80 = OpCompositeConstruct %v4float %53 %41 %42 %43
         %81 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %81 %80
               OpStore %outColor %31
               OpEmitVertex

; Emit vertex 3: Bottom (x, y-h)
         %90 = OpCompositeConstruct %v4float %40 %52 %42 %43
         %91 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %91 %90
               OpStore %outColor %31
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
