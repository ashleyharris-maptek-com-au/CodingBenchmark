using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    const double MU = 398600.4418; // km³/s²

    struct Orbit
    {
        public double x, y, z;
        public double vx, vy, vz;

        public Orbit(double x, double y, double z, double vx, double vy, vz)
        {
            this.x = x; this.y = y; this.z = z;
            this.vx = vx; this.vy = vy; this.vz = vz;
        }
    }

    struct OrbitalElements
    {
        public double semiMajorAxis;
        public double eccentricity;
        public double inclination;
        public double energy;
        public double angularMomentum;
    }

    static OrbitalElements ComputeOrbitalElements(Orbit orbit)
    {
        double r = Math.Sqrt(orbit.x * orbit.x + orbit.y * orbit.y + orbit.z * orbit.z);
        double v = Math.Sqrt(orbit.vx * orbit.vx + orbit.vy * orbit.vy + orbit.vz * orbit.vz);

        // Specific orbital energy
        double energy = v * v / 2.0 - MU / r;

        // Angular momentum vector
        double hx = orbit.y * orbit.vz - orbit.z * orbit.vy;
        double hy = orbit.z * orbit.vx - orbit.x * orbit.vz;
        double hz = orbit.x * orbit.vy - orbit.y * orbit.vx;
        double h = Math.Sqrt(hx * hx + hy * hy + hz * hz);

        // Semi-major axis
        double a = -MU / (2.0 * energy);

        // Eccentricity vector
        double ex = (orbit.vy * hz - orbit.vz * hy) / MU - orbit.x / r;
        double ey = (orbit.vz * hx - orbit.vx * hz) / MU - orbit.y / r;
        double ez = (orbit.vx * hy - orbit.vy * hx) / MU - orbit.z / r;
        double e = Math.Sqrt(ex * ex + ey * ey + ez * ez);

        // Inclination
        double inc = Math.Acos(hz / h);

        return new OrbitalElements
        {
            semiMajorAxis = a,
            eccentricity = e,
            inclination = inc,
            energy = energy,
            angularMomentum = h
        };
    }

    static double EstimateDeltaV(Orbit from, Orbit to)
    {
        var elem1 = ComputeOrbitalElements(from);
        var elem2 = ComputeOrbitalElements(to);

        // Simple delta-V estimate based on orbital elements
        double r1 = Math.Sqrt(from.x * from.x + from.y * from.y + from.z * from.z);
        double r2 = Math.Sqrt(to.x * to.x + to.y * to.y + to.z * to.z);

        double v1 = Math.Sqrt(from.vx * from.vx + from.vy * from.vy + from.vz * from.vz);
        double v2 = Math.Sqrt(to.vx * to.vx + to.vy * to.vy + to.vz * to.vz);

        // Energy difference component
        double energyDelta = Math.Abs(elem2.energy - elem1.energy);
        double deltaVEnergy = Math.Sqrt(2.0 * energyDelta);

        // Inclination change component
        double incChange = Math.Abs(elem2.inclination - elem1.inclination);
        double vCirc = Math.Sqrt(MU / ((r1 + r2) / 2.0));
        double deltaVInc = 2.0 * vCirc * Math.Sin(incChange / 2.0);

        // Plane change component (angular momentum difference)
        double hDiff = Math.Abs(elem2.angularMomentum - elem1.angularMomentum);
        double deltaVPlane = hDiff / Math.Max(r1, r2);

        // Eccentricity change component
        double eDiff = Math.Abs(elem2.eccentricity - elem1.eccentricity);
        double deltaVEcc = eDiff * vCirc * 0.5;

        // Combined estimate
        double totalDeltaV = deltaVEnergy + deltaVInc + deltaVPlane + deltaVEcc;

        // Add position-based penalty for physical distance
        double dx = to.x - from.x;
        double dy = to.y - from.y;
        double dz = to.z - from.z;
        double distance = Math.Sqrt(dx * dx + dy * dy + dz * dz);
        double distancePenalty = distance / 10000.0; // Normalized

        return totalDeltaV + distancePenalty;
    }

    static List<int> NearestNeighbor(int start, double[,] costs, int n)
    {
        var visited = new bool[n];
        var tour = new List<int>();
        int current = start;

        tour.Add(current);
        visited[current] = true;

        for (int i = 1; i < n; i++)
        {
            int nearest = -1;
            double minCost = double.MaxValue;

            for (int j = 0; j < n; j++)
            {
                if (!visited[j] && costs[current, j] < minCost)
                {
                    minCost = costs[current, j];
                    nearest = j;
                }
            }

            tour.Add(nearest);
            visited[nearest] = true;
            current = nearest;
        }

        return tour;
    }

    static double TourCost(List<int> tour, double[,] costs, Orbit start)
    {
        double total = costs[tour.Count, tour[0]]; // Start to first station
        for (int i = 0; i < tour.Count - 1; i++)
        {
            total += costs[tour[i], tour[i + 1]];
        }
        return total;
    }

    static void TwoOpt(List<int> tour, double[,] costs, Orbit start, int maxIterations = 1000)
    {
        int n = tour.Count;
        bool improved = true;
        int iterations = 0;

        while (improved && iterations < maxIterations)
        {
            improved = false;
            iterations++;

            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    double delta = TwoOptDelta(tour, costs, start, i, j);

                    if (delta < -0.001) // Improvement threshold
                    {
                        TwoOptReverse(tour, i + 1, j);
                        improved = true;
                    }
                }
            }
        }
    }

    static double TwoOptDelta(List<int> tour, double[,] costs, Orbit start, int i, int j)
    {
        int n = tour.Count;

        // Current edges
        int a = (i == 0) ? tour.Count : tour[i - 1];
        int b = tour[i];
        int c = tour[j];
        int d = (j == n - 1) ? -1 : tour[j + 1];

        double oldCost, newCost;

        if (i == 0)
        {
            oldCost = costs[n, b];
            if (d == -1)
            {
                oldCost += costs[c, -1];
                newCost = costs[n, c] + costs[b, -1];
            }
            else
            {
                oldCost += costs[c, d];
                newCost = costs[n, c] + costs[b, d];
            }
        }
        else
        {
            oldCost = costs[a, b];
            if (d == -1)
            {
                oldCost += costs[c, -1];
                newCost = costs[a, c] + costs[b, -1];
            }
            else
            {
                oldCost += costs[c, d];
                newCost = costs[a, c] + costs[b, d];
            }
        }

        return newCost - oldCost;
    }

    static void TwoOptReverse(List<int> tour, int i, int j)
    {
        while (i < j)
        {
            int temp = tour[i];
            tour[i] = tour[j];
            tour[j] = temp;
            i++;
            j--;
        }
    }

    static List<int> SolveTSP(Orbit start, List<Orbit> stations)
    {
        int n = stations.Count;

        // Build cost matrix
        // Index n is the starting orbit
        var costs = new double[n + 1, n + 1];

        for (int i = 0; i < n; i++)
        {
            costs[n, i] = EstimateDeltaV(start, stations[i]);
            costs[i, n] = costs[n, i];
        }

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                if (i != j)
                {
                    costs[i, j] = EstimateDeltaV(stations[i], stations[j]);
                }
            }
        }

        List<int> bestTour = null;
        double bestCost = double.MaxValue;

        // Adaptive strategy based on problem size
        int numStarts = 1;
        int maxIterations = 1000;

        if (n <= 10)
        {
            numStarts = n;
            maxIterations = 2000;
        }
        else if (n <= 20)
        {
            numStarts = Math.Min(10, n);
            maxIterations = 1000;
        }
        else if (n <= 50)
        {
            numStarts = 5;
            maxIterations = 500;
        }
        else
        {
            numStarts = 3;
            maxIterations = 200;
        }

        // Try multiple starting points
        for (int startIdx = 0; startIdx < numStarts; startIdx++)
        {
            var tour = NearestNeighbor(startIdx, costs, n);
            TwoOpt(tour, costs, start, maxIterations);

            double cost = TourCost(tour, costs, start);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestTour = new List<int>(tour);
            }
        }

        return bestTour;
    }

    static void Main()
    {
        string[] firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);

        string[] startLine = Console.ReadLine().Split();
        var startOrbit = new Orbit(
            double.Parse(startLine[0]), double.Parse(startLine[1]), double.Parse(startLine[2]),
            double.Parse(startLine[3]), double.Parse(startLine[4]), double.Parse(startLine[5])
        );

        var stations = new List<Orbit>();
        for (int i = 0; i < n; i++)
        {
            string[] line = Console.ReadLine().Split();
            stations.Add(new Orbit(
                double.Parse(line[0]), double.Parse(line[1]), double.Parse(line[2]),
                double.Parse(line[3]), double.Parse(line[4]), double.Parse(line[5])
            ));
        }

        var solution = SolveTSP(startOrbit, stations);
        Console.WriteLine(string.Join(" ", solution));
    }
}
