OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
     %uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %3
          %5 = OpLabel

               ; Vertex 0
         %20 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %21 = OpLoad %v4float %20
         %22 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %22 %21
         %23 = OpAccessChain %ptr_in_v3 %color_in %int_0
         %24 = OpLoad %v3float %23
               OpStore %outColor %24
               OpEmitVertex

               ; Vertex 1
         %30 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %31 = OpLoad %v4float %30
         %32 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %32 %31
         %33 = OpAccessChain %ptr_in_v3 %color_in %int_1
         %34 = OpLoad %v3float %33
               OpStore %outColor %34
               OpEmitVertex

               ; Vertex 2
         %40 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %41 = OpLoad %v4float %40
         %42 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %42 %41
         %43 = OpAccessChain %ptr_in_v3 %color_in %int_2
         %44 = OpLoad %v3float %43
               OpStore %outColor %44
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
