const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { depot, truckCount, packages } = data;

// Calculate Euclidean distance
function distance(p1, p2) {
  const dx = p1.x - p2.x;
  const dy = p1.y - p2.y;
  return Math.sqrt(dx * dx + dy * dy);
}

// Calculate angle from depot
function angle(p) {
  return Math.atan2(p.y - depot[1], p.x - depot[0]);
}

// Calculate total route distance including depot
function routeDistance(route, pkgMap) {
  if (route.length === 0) return 0;

  let dist = distance(depot, pkgMap[route[0]]);
  for (let i = 0; i < route.length - 1; i++) {
    dist += distance(pkgMap[route[i]], pkgMap[route[i + 1]]);
  }
  dist += distance(pkgMap[route[route.length - 1]], depot);
  return dist;
}

// 2-opt improvement for a single route
function twoOpt(route, pkgMap, maxIterations = 100) {
  if (route.length < 4) return route;

  let improved = true;
  let iterations = 0;

  while (improved && iterations < maxIterations) {
    improved = false;
    iterations++;

    for (let i = 0; i < route.length - 2; i++) {
      for (let j = i + 2; j < route.length; j++) {
        // Calculate current distance
        const p1 = i === 0 ? depot : pkgMap[route[i - 1]];
        const p2 = pkgMap[route[i]];
        const p3 = pkgMap[route[j]];
        const p4 = j === route.length - 1 ? depot : pkgMap[route[j + 1]];

        const currentDist = distance(p1, p2) + distance(p3, p4);
        const newDist = distance(p1, p3) + distance(p2, p4);

        if (newDist < currentDist - 0.001) {
          // Reverse the segment between i and j
          const newRoute = route.slice(0, i).concat(
            route.slice(i, j + 1).reverse(),
            route.slice(j + 1)
          );
          route = newRoute;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }

  return route;
}

// Create package map
const pkgMap = {};
packages.forEach(p => {
  pkgMap[p.id] = { x: p.x, y: p.y };
});

// Sweep algorithm: sort by angle and partition
const sortedPackages = packages.slice().sort((a, b) => angle(a) - angle(b));

// Partition packages into routes
const routes = [];
const packagesPerTruck = Math.ceil(packages.length / truckCount);

for (let i = 0; i < truckCount && i * packagesPerTruck < sortedPackages.length; i++) {
  const start = i * packagesPerTruck;
  const end = Math.min(start + packagesPerTruck, sortedPackages.length);
  const routePackages = sortedPackages.slice(start, end);

  if (routePackages.length > 0) {
    // Build initial route with nearest neighbor from first package
    const route = [];
    const remaining = new Set(routePackages.map(p => p.id));

    // Start with first package in sweep order
    let current = routePackages[0];
    route.push(current.id);
    remaining.delete(current.id);

    // Nearest neighbor construction
    while (remaining.size > 0) {
      let nearest = null;
      let minDist = Infinity;

      for (const id of remaining) {
        const pkg = pkgMap[id];
        const d = distance(current, pkg);
        if (d < minDist) {
          minDist = d;
          nearest = id;
        }
      }

      route.push(nearest);
      current = pkgMap[nearest];
      remaining.delete(nearest);
    }

    // Apply 2-opt improvement
    const optimizedRoute = twoOpt(route, pkgMap);
    routes.push(optimizedRoute);
  }
}

// Output result
console.log(JSON.stringify({ routes }));
