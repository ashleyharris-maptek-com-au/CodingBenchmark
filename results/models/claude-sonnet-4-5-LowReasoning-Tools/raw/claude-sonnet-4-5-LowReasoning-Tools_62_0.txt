declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

const { width, height, grid, ignitions, towns, cutBudget }: {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
} = data;

// Simulate fire spread with given cuts
function simulateFire(cuts: Set<string>): Set<string> {
  const burned = new Set<string>();
  const queue: [number, number][] = [];

  for (const [x, y] of ignitions) {
    const key = `${x},${y}`;
    burned.add(key);
    queue.push([x, y]);
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;

      const key = `${nx},${ny}`;
      if (burned.has(key)) continue;
      if (cuts.has(key)) continue;
      if (grid[ny][nx] !== '.') continue;

      burned.add(key);
      queue.push([nx, ny]);
    }
  }

  return burned;
}

// Calculate total town value saved
function calculateSavedValue(burned: Set<string>): number {
  let saved = 0;
  for (const town of towns) {
    const key = `${town.x},${town.y}`;
    if (!burned.has(key)) {
      saved += town.value;
    }
  }
  return saved;
}

// Calculate distance from fire to all reachable cells
function distanceFromFire(): Map<string, number> {
  const dist = new Map<string, number>();
  const queue: [number, number, number][] = [];

  for (const [x, y] of ignitions) {
    const key = `${x},${y}`;
    dist.set(key, 0);
    queue.push([x, y, 0]);
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y, d] = queue[head++];

    for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
      const nx = x + dx;
      const ny = y + dy;

      if (nx < 0 || nx >= width || ny < 0 || ny >= height) continue;

      const key = `${nx},${ny}`;
      if (dist.has(key)) continue;
      if (grid[ny][nx] !== '.') continue;

      dist.set(key, d + 1);
      queue.push([nx, ny, d + 1]);
    }
  }

  return dist;
}

const currentCuts = new Set<string>();

// For smaller budgets or grids, use greedy approach
// For larger instances, use distance-based heuristic
const useGreedy = cutBudget <= 50 || (cutBudget < Math.sqrt(width * height));

if (useGreedy) {
  // Greedy: iteratively pick the cut with best marginal value
  const baselineBurned = simulateFire(currentCuts);
  const candidates: string[] = [];

  // Only consider cells that would burn in baseline
  for (const key of baselineBurned) {
    const [x, y] = key.split(',').map(Number);
    if (grid[y][x] === '.') {
      candidates.push(key);
    }
  }

  for (let i = 0; i < cutBudget; i++) {
    let bestCut: string | null = null;
    let bestValue = calculateSavedValue(simulateFire(currentCuts));

    for (const candidate of candidates) {
      if (currentCuts.has(candidate)) continue;

      currentCuts.add(candidate);
      const value = calculateSavedValue(simulateFire(currentCuts));
      currentCuts.delete(candidate);

      if (value > bestValue) {
        bestValue = value;
        bestCut = candidate;
      }
    }

    if (bestCut === null) break;
    currentCuts.add(bestCut);
  }
} else {
  // Heuristic: prioritize cells on paths to high-value towns
  const dist = distanceFromFire();
  const baselineBurned = simulateFire(new Set());

  // Score each cell by its potential impact
  const cellScore = new Map<string, number>();

  for (const town of towns) {
    const townKey = `${town.x},${town.y}`;

    // Skip towns that are already safe or unreachable
    if (!baselineBurned.has(townKey)) continue;

    const townDist = dist.get(townKey);
    if (townDist === undefined || townDist === 0) continue;

    // Cells closer to the fire on paths to this town are valuable
    for (const [key, d] of dist.entries()) {
      const [x, y] = key.split(',').map(Number);
      if (grid[y][x] !== '.') continue;

      // Prioritize cells that are:
      // 1. On the path (distance < town distance)
      // 2. Closer to the fire (lower distance)
      // 3. Protecting high-value towns
      if (d > 0 && d <= townDist) {
        const proximity = 1 / (d + 1); // Favor cells closer to fire
        const impact = town.value * proximity;
        cellScore.set(key, (cellScore.get(key) || 0) + impact);
      }
    }
  }

  // Select top scoring cells
  const sortedCells = Array.from(cellScore.entries())
    .sort((a, b) => b[1] - a[1])
    .slice(0, cutBudget);

  for (const [key, _] of sortedCells) {
    currentCuts.add(key);
  }
}

// Convert to output format
const firebreaks: [number, number][] = Array.from(currentCuts).map(key => {
  const [x, y] = key.split(',').map(Number);
  return [x, y];
});
