use std::collections::{VecDeque, HashSet, HashMap};
use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let dims: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (h, w) = (dims[0], dims[1]);

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    // Read maze
    let mut maze = Vec::with_capacity(h);
    let mut start = None;
    let mut end = None;

    for i in 0..h {
        let row = lines.next().unwrap().unwrap();
        let chars: Vec<char> = row.chars().collect();

        for (j, &ch) in chars.iter().enumerate() {
            if ch == 'A' {
                start = Some((i, j));
            } else if ch == 'B' {
                end = Some((i, j));
            }
        }

        maze.push(chars);
    }

    // Check if start and end exist
    if start.is_none() || end.is_none() {
        println!("0");
        return;
    }

    let start = start.unwrap();
    let end = end.unwrap();

    // BFS to find path
    if let Some(path) = bfs(&maze, h, w, start, end) {
        println!("{}", path.len());
        for (row, col) in path {
            println!("{} {}", col, row);
        }
    } else {
        println!("0");
    }
}

fn bfs(maze: &[Vec<char>], h: usize, w: usize, start: (usize, usize), end: (usize, usize)) -> Option<Vec<(usize, usize)>> {
    let mut queue = VecDeque::new();
    let mut visited = HashSet::new();
    let mut parent = HashMap::new();

    queue.push_back(start);
    visited.insert(start);

    let directions = [(0, 1), (1, 0), (0, -1), (-1, 0)];

    while let Some(pos) = queue.pop_front() {
        if pos == end {
            // Reconstruct path
            let mut path = Vec::new();
            let mut current = end;

            loop {
                path.push(current);

                if current == start {
                    break;
                }

                current = parent[&current];
            }

            path.reverse();
            return Some(path);
        }

        let (row, col) = pos;

        for &(dy, dx) in &directions {
            let ny = row as isize + dy;
            let nx = col as isize + dx;

            if ny >= 0 && ny < h as isize && nx >= 0 && nx < w as isize {
                let next = (ny as usize, nx as usize);

                if !visited.contains(&next) && maze[next.0][next.1] != '#' {
                    visited.insert(next);
                    parent.insert(next, pos);
                    queue.push_back(next);
                }
            }
        }
    }

    None
}
