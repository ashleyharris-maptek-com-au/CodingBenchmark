using System;
using System.Collections.Generic;

class Program
{
    static void Main()
    {
        // Read input
        var line1 = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        var line2 = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries);
        var cuts = new (int length, int index)[n];
        for (int i = 0; i < n; i++)
        {
            cuts[i] = (int.Parse(line2[i]), i);
        }

        // Sort by length descending (for Best Fit Decreasing)
        Array.Sort(cuts, (a, b) => b.length.CompareTo(a.length));

        // Best Fit Decreasing with SortedSet for efficient lookup
        var binsBySpace = new SortedSet<(int space, int binId)>();
        var bins = new List<List<int>>();
        int nextBinId = 0;

        foreach (var cut in cuts)
        {
            // Find bin with minimum space that still fits this cut
            var candidateBins = binsBySpace.GetViewBetween(
                (cut.length, int.MinValue),
                (int.MaxValue, int.MaxValue)
            );

            if (candidateBins.Count > 0)
            {
                // Use the bin with minimum sufficient space (best fit)
                var selectedBin = candidateBins.Min;
                binsBySpace.Remove(selectedBin);
                bins[selectedBin.binId].Add(cut.index);

                int remainingSpace = selectedBin.space - cut.length;
                if (remainingSpace > 0)
                {
                    binsBySpace.Add((remainingSpace, selectedBin.binId));
                }
            }
            else
            {
                // No existing bin fits, create new one
                bins.Add(new List<int> { cut.index });
                int remainingSpace = stockLength - cut.length;
                binsBySpace.Add((remainingSpace, nextBinId));
                nextBinId++;
            }
        }

        // Output result
        Console.WriteLine(bins.Count);
        foreach (var bin in bins)
        {
            Console.WriteLine(string.Join(" ", bin));
        }
    }
}
