OpCapability Geometry
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
    %float_0 = OpConstant %float 0
 %float_0_02 = OpConstant %float 0.02
%float_0_333 = OpConstant %float 0.333333343
%float_0_667 = OpConstant %float 0.666666687
    %float_1 = OpConstant %float 1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
  %color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

       %main = OpFunction %void None %3
          %5 = OpLabel

; Load positions gl_in[0..3].gl_Position
        %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
            %p0 = OpLoad %v4float %p0_ptr
        %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
            %p1 = OpLoad %v4float %p1_ptr
        %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
            %p2 = OpLoad %v4float %p2_ptr
        %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
            %p3 = OpLoad %v4float %p3_ptr

; Load colors
     %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
          %c1 = OpLoad %v3float %c1_ptr
     %c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
          %c2 = OpLoad %v3float %c2_ptr

; Extract xyz from positions
       %p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
       %p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
       %p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
       %p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

; Compute tangents for smooth perpendiculars using adjacency
      %delta0 = OpFSub %v3float %p1_xyz %p0_xyz
   %tangent_start = OpExtInst %v3float %1 Normalize %delta0
      %delta1 = OpFSub %v3float %p3_xyz %p2_xyz
   %tangent_end = OpExtInst %v3float %1 Normalize %delta1

; Perpendiculars (cross with up vector)
          %up = OpCompositeConstruct %v3float %float_0 %float_0 %float_1
   %perp_start_raw = OpExtInst %v3float %1 Cross %tangent_start %up
   %perp_start = OpExtInst %v3float %1 Normalize %perp_start_raw
   %perp_start_scaled = OpVectorTimesScalar %v3float %perp_start %float_0_02
   %perp_end_raw = OpExtInst %v3float %1 Cross %tangent_end %up
   %perp_end = OpExtInst %v3float %1 Normalize %perp_end_raw
   %perp_end_scaled = OpVectorTimesScalar %v3float %perp_end %float_0_02

; Segment direction
    %seg_dir = OpFSub %v3float %p2_xyz %p1_xyz

; === First dash: t in [0, 1/3] ===
; Start position at t=0
    %dash1_start = %p1_xyz
; End position at t=1/3
     %offset_1 = OpVectorTimesScalar %v3float %seg_dir %float_0_333
    %dash1_end = OpFAdd %v3float %p1_xyz %offset_1

; Perpendicular at t=1/3
     %perp_diff = OpFSub %v3float %perp_end_scaled %perp_start_scaled
   %perp_interp_1 = OpVectorTimesScalar %v3float %perp_diff %float_0_333
    %perp_1 = OpFAdd %v3float %perp_start_scaled %perp_interp_1

; Color at t=1/3
    %color_diff = OpFSub %v3float %c2 %c1
  %color_interp_1 = OpVectorTimesScalar %v3float %color_diff %float_0_333
    %color_1 = OpFAdd %v3float %c1 %color_interp_1

; Emit first quad (triangle strip)
; Vertex 1: dash1_start - perp_start_scaled
       %v1_pos3 = OpFSub %v3float %dash1_start %perp_start_scaled
         %v1_x = OpCompositeExtract %float %v1_pos3 0
         %v1_y = OpCompositeExtract %float %v1_pos3 1
         %v1_z = OpCompositeExtract %float %v1_pos3 2
        %v1_p = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1
    %gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %gl_pos_ptr %v1_p
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 2: dash1_start + perp_start_scaled
       %v2_pos3 = OpFAdd %v3float %dash1_start %perp_start_scaled
         %v2_x = OpCompositeExtract %float %v2_pos3 0
         %v2_y = OpCompositeExtract %float %v2_pos3 1
         %v2_z = OpCompositeExtract %float %v2_pos3 2
        %v2_p = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1
               OpStore %gl_pos_ptr %v2_p
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 3: dash1_end - perp_1
       %v3_pos3 = OpFSub %v3float %dash1_end %perp_1
         %v3_x = OpCompositeExtract %float %v3_pos3 0
         %v3_y = OpCompositeExtract %float %v3_pos3 1
         %v3_z = OpCompositeExtract %float %v3_pos3 2
        %v3_p = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1
               OpStore %gl_pos_ptr %v3_p
               OpStore %outColor %color_1
               OpEmitVertex

; Vertex 4: dash1_end + perp_1
       %v4_pos3 = OpFAdd %v3float %dash1_end %perp_1
         %v4_x = OpCompositeExtract %float %v4_pos3 0
         %v4_y = OpCompositeExtract %float %v4_pos3 1
         %v4_z = OpCompositeExtract %float %v4_pos3 2
        %v4_p = OpCompositeConstruct %v4float %v4_x %v4_y %v4_z %float_1
               OpStore %gl_pos_ptr %v4_p
               OpStore %outColor %color_1
               OpEmitVertex

               OpEndPrimitive

; === Second dash: t in [2/3, 1] ===
; Start position at t=2/3
     %offset_2 = OpVectorTimesScalar %v3float %seg_dir %float_0_667
    %dash2_start = OpFAdd %v3float %p1_xyz %offset_2
; End position at t=1
    %dash2_end = %p2_xyz

; Perpendicular at t=2/3
   %perp_interp_2 = OpVectorTimesScalar %v3float %perp_diff %float_0_667
    %perp_2 = OpFAdd %v3float %perp_start_scaled %perp_interp_2

; Color at t=2/3
  %color_interp_2 = OpVectorTimesScalar %v3float %color_diff %float_0_667
    %color_2 = OpFAdd %v3float %c1 %color_interp_2

; Emit second quad (triangle strip)
; Vertex 5: dash2_start - perp_2
       %v5_pos3 = OpFSub %v3float %dash2_start %perp_2
         %v5_x = OpCompositeExtract %float %v5_pos3 0
         %v5_y = OpCompositeExtract %float %v5_pos3 1
         %v5_z = OpCompositeExtract %float %v5_pos3 2
        %v5_p = OpCompositeConstruct %v4float %v5_x %v5_y %v5_z %float_1
               OpStore %gl_pos_ptr %v5_p
               OpStore %outColor %color_2
               OpEmitVertex

; Vertex 6: dash2_start + perp_2
       %v6_pos3 = OpFAdd %v3float %dash2_start %perp_2
         %v6_x = OpCompositeExtract %float %v6_pos3 0
         %v6_y = OpCompositeExtract %float %v6_pos3 1
         %v6_z = OpCompositeExtract %float %v6_pos3 2
        %v6_p = OpCompositeConstruct %v4float %v6_x %v6_y %v6_z %float_1
               OpStore %gl_pos_ptr %v6_p
               OpStore %outColor %color_2
               OpEmitVertex

; Vertex 7: dash2_end - perp_end_scaled
       %v7_pos3 = OpFSub %v3float %dash2_end %perp_end_scaled
         %v7_x = OpCompositeExtract %float %v7_pos3 0
         %v7_y = OpCompositeExtract %float %v7_pos3 1
         %v7_z = OpCompositeExtract %float %v7_pos3 2
        %v7_p = OpCompositeConstruct %v4float %v7_x %v7_y %v7_z %float_1
               OpStore %gl_pos_ptr %v7_p
               OpStore %outColor %c2
               OpEmitVertex

; Vertex 8: dash2_end + perp_end_scaled
       %v8_pos3 = OpFAdd %v3float %dash2_end %perp_end_scaled
         %v8_x = OpCompositeExtract %float %v8_pos3 0
         %v8_y = OpCompositeExtract %float %v8_pos3 1
         %v8_z = OpCompositeExtract %float %v8_pos3 2
        %v8_p = OpCompositeConstruct %v4float %v8_x %v8_y %v8_z %float_1
               OpStore %gl_pos_ptr %v8_p
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
