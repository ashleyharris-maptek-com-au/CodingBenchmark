OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

%gl_PerVertex_in = OpTypeStruct %v4float
     %uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%arr_color_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_color = OpTypePointer Input %arr_color_in
   %color_in = OpVariable %ptr_in_arr_color Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_func_v4 = OpTypePointer Function %v4float
%ptr_func_v3 = OpTypePointer Function %v3float

      %int_0 = OpConstant %int 0
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
  %float_004 = OpConstant %float 0.04

       %main = OpFunction %void None %3
          %5 = OpLabel
   %position = OpVariable %ptr_func_v4 Function
      %color = OpVariable %ptr_func_v3 Function
     %vertex = OpVariable %ptr_func_v4 Function

         %20 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %21 = OpLoad %v4float %20
               OpStore %position %21

         %25 = OpAccessChain %ptr_in_v3 %color_in %int_0
         %26 = OpLoad %v3float %25
               OpStore %color %26

         %30 = OpLoad %v4float %position
         %31 = OpCompositeExtract %float %30 0
         %32 = OpCompositeExtract %float %30 1

         %40 = OpFSub %float %31 %float_004
         %41 = OpFSub %float %32 %float_004
         %42 = OpCompositeConstruct %v4float %40 %41 %float_0 %float_1
               OpStore %vertex %42
         %43 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %43 %42
         %44 = OpLoad %v3float %color
               OpStore %outColor %44
               OpEmitVertex

         %50 = OpFAdd %float %31 %float_004
         %51 = OpFSub %float %32 %float_004
         %52 = OpCompositeConstruct %v4float %50 %51 %float_0 %float_1
               OpStore %vertex %52
         %53 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %53 %52
         %54 = OpLoad %v3float %color
               OpStore %outColor %54
               OpEmitVertex

         %60 = OpFSub %float %31 %float_004
         %61 = OpFAdd %float %32 %float_004
         %62 = OpCompositeConstruct %v4float %60 %61 %float_0 %float_1
               OpStore %vertex %62
         %63 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %63 %62
         %64 = OpLoad %v3float %color
               OpStore %outColor %64
               OpEmitVertex

         %70 = OpFAdd %float %31 %float_004
         %71 = OpFAdd %float %32 %float_004
         %72 = OpCompositeConstruct %v4float %70 %71 %float_0 %float_1
               OpStore %vertex %72
         %73 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %73 %72
         %74 = OpLoad %v3float %color
               OpStore %outColor %74
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
