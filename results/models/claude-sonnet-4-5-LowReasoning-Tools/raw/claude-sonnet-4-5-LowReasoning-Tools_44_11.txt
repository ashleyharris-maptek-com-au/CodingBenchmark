#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 6x6 polka dot pattern
    vec2 cellUV = fract(uv * 6.0);

    // Distance from cell center (0.5, 0.5)
    vec2 cellCenter = vec2(0.5, 0.5);
    float dist = distance(cellUV, cellCenter);

    // Choose color based on whether we're inside a dot (radius 0.3)
    vec3 baseColor;
    if (dist < 0.3) {
        baseColor = vec3(1.0, 0.8, 0.0); // Yellow dot
    } else {
        baseColor = vec3(0.1, 0.1, 0.3); // Dark blue background
    }

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(0.0, dot(N, L));
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Apply shading and output
    outColor = vec4(baseColor * shadingFactor, 1.0);
}
