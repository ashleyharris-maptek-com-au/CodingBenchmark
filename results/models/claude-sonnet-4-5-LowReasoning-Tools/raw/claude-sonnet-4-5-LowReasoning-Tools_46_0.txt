#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out-of-bounds access
    if (i >= N) {
        return;
    }

    // Extract parameters
    float dt = physics.x;
    float softening = physics.y;
    float G = physics.z;

    // Read current body's position and velocity
    vec4 positionMass = inData[i * 2];
    vec4 velocityPad = inData[i * 2 + 1];

    vec3 pos_i = positionMass.xyz;
    float mass_i = positionMass.w;
    vec3 vel_i = velocityPad.xyz;

    // Calculate acceleration from all other bodies
    vec3 acc = vec3(0.0);

    for (uint j = 0; j < N; j++) {
        // Skip self-interaction
        if (i == j) {
            continue;
        }

        // Read body j
        vec4 positionMass_j = inData[j * 2];
        vec3 pos_j = positionMass_j.xyz;
        float mass_j = positionMass_j.w;

        // Calculate displacement
        vec3 r = pos_j - pos_i;
        float distSq = dot(r, r) + softening * softening;
        float invDist = 1.0 / sqrt(distSq);
        float invDistCube = invDist * invDist * invDist;

        // acc += G * mass_j * (pos_j - pos_i) / (|pos_j - pos_i|^2 + softening^2)^(3/2)
        acc += G * mass_j * r * invDistCube;
    }

    // Update velocity and position using Euler integration
    vel_i += acc * dt;
    pos_i += vel_i * dt;

    // Write updated position+mass and velocity to output
    outData[i * 2] = vec4(pos_i, mass_i);
    outData[i * 2 + 1] = vec4(vel_i, 0.0);
}
