declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
  id: number;
  x: number;
  y: number;
}

const depot: [number, number] = data.depot;
const truckCount: number = data.truckCount;
const packages: Package[] = data.packages;

function dist(x1: number, y1: number, x2: number, y2: number): number {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return Math.sqrt(dx * dx + dy * dy);
}

function clusterPackages(packages: Package[], k: number): Package[][] {
  if (packages.length === 0) return [];
  if (k >= packages.length) return packages.map(p => [p]);

  // Sweep algorithm: sort by polar angle from depot
  const withAngles = packages.map(p => ({
    pkg: p,
    angle: Math.atan2(p.y - depot[1], p.x - depot[0]),
    distFromDepot: dist(p.x, p.y, depot[0], depot[1])
  }));

  withAngles.sort((a, b) => a.angle !== b.angle ? a.angle - b.angle : a.distFromDepot - b.distFromDepot);

  // Distribute packages into k clusters
  const clusters: Package[][] = Array.from({ length: k }, () => []);
  for (let i = 0; i < withAngles.length; i++) {
    const clusterIdx = Math.floor(i * k / withAngles.length);
    clusters[clusterIdx].push(withAngles[i].pkg);
  }

  return clusters.filter(c => c.length > 0);
}

function buildRoute(cluster: Package[]): number[] {
  if (cluster.length === 0) return [];
  if (cluster.length === 1) return [cluster[0].id];

  // Nearest neighbor construction from depot
  const route: Package[] = [];
  const remaining = new Set<Package>(cluster);
  let currX = depot[0];
  let currY = depot[1];

  while (remaining.size > 0) {
    let nearest: Package | null = null;
    let minDist = Infinity;

    for (const pkg of remaining) {
      const d = dist(currX, currY, pkg.x, pkg.y);
      if (d < minDist) {
        minDist = d;
        nearest = pkg;
      }
    }

    route.push(nearest!);
    remaining.delete(nearest!);
    currX = nearest!.x;
    currY = nearest!.y;
  }

  // 2-opt improvement
  let improved = true;
  while (improved) {
    improved = false;

    for (let i = 0; i < route.length - 1; i++) {
      for (let j = i + 2; j < route.length; j++) {
        // Calculate positions for edge swap
        const p0 = i === 0 ? depot : [route[i - 1].x, route[i - 1].y];
        const p1 = [route[i].x, route[i].y];
        const p2 = [route[j].x, route[j].y];
        const p3 = j === route.length - 1 ? depot : [route[j + 1].x, route[j + 1].y];

        const oldDist = dist(p0[0], p0[1], p1[0], p1[1]) + dist(p2[0], p2[1], p3[0], p3[1]);
        const newDist = dist(p0[0], p0[1], p2[0], p2[1]) + dist(p1[0], p1[1], p3[0], p3[1]);

        if (newDist < oldDist - 1e-9) {
          // Reverse segment between i and j
          const segment = route.slice(i, j + 1).reverse();
          route.splice(i, j - i + 1, ...segment);
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }

  return route.map(p => p.id);
}

const numTrucks = Math.min(truckCount, packages.length);
const clusters = clusterPackages(packages, numTrucks);
const routes = clusters.map(buildRoute);
