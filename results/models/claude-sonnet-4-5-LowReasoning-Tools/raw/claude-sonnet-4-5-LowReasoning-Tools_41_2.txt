; SPIR-V
; Version: 1.0
; Generator: Khronos; 0
; Bound: 20
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %uv %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Variables
%uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Main function
%main = OpFunction %void None %void_func
%entry = OpLabel
%uv_val = OpLoad %vec2 %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%color = OpCompositeConstruct %vec4 %uv_x %uv_y %float_0 %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
