#include <iostream>
#include <string>
#include <vector>
#include <algorithm>
#include <chrono>
#include <sstream>
#include <cctype>
#include <limits>
#include <thread>
#include <atomic>

using namespace std;

// Piece values for evaluation
constexpr int PAWN_VALUE = 100;
constexpr int KNIGHT_VALUE = 320;
constexpr int BISHOP_VALUE = 330;
constexpr int ROOK_VALUE = 500;
constexpr int QUEEN_VALUE = 900;
constexpr int KING_VALUE = 20000;

// Search constants
constexpr int INF = 1000000;
constexpr int MAX_DEPTH = 50;

enum Piece { EMPTY = 0, PAWN = 1, KNIGHT = 2, BISHOP = 3, ROOK = 4, QUEEN = 5, KING = 6 };
enum Color { WHITE = 8, BLACK = 16 };

struct Move {
    int from, to;
    int piece, captured;
    int promotion = 0;
    bool isEnPassant = false;
    bool isCastling = false;
    int score = 0;
};

class ChessBoard {
public:
    int board[64];
    bool whiteToMove;
    bool whiteCastleK, whiteCastleQ, blackCastleK, blackCastleQ;
    int enPassantSquare;
    int halfmoveClock;
    int fullmoveNumber;

    ChessBoard() { clear(); }

    void clear() {
        for (int i = 0; i < 64; i++) board[i] = EMPTY;
        whiteToMove = true;
        whiteCastleK = whiteCastleQ = blackCastleK = blackCastleQ = false;
        enPassantSquare = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
    }

    bool parseFEN(const string& fen) {
        clear();
        istringstream ss(fen);
        string position, turn, castling, enpassant;
        ss >> position >> turn >> castling >> enpassant >> halfmoveClock >> fullmoveNumber;

        int rank = 7, file = 0;
        for (char c : position) {
            if (c == '/') { rank--; file = 0; }
            else if (isdigit(c)) { file += c - '0'; }
            else {
                int piece = EMPTY;
                int color = isupper(c) ? WHITE : BLACK;
                c = tolower(c);
                if (c == 'p') piece = PAWN;
                else if (c == 'n') piece = KNIGHT;
                else if (c == 'b') piece = BISHOP;
                else if (c == 'r') piece = ROOK;
                else if (c == 'q') piece = QUEEN;
                else if (c == 'k') piece = KING;
                board[rank * 8 + file] = piece | color;
                file++;
            }
        }

        whiteToMove = (turn == "w");
        whiteCastleK = castling.find('K') != string::npos;
        whiteCastleQ = castling.find('Q') != string::npos;
        blackCastleK = castling.find('k') != string::npos;
        blackCastleQ = castling.find('q') != string::npos;

        if (enpassant != "-") {
            enPassantSquare = (enpassant[1] - '1') * 8 + (enpassant[0] - 'a');
        }

        return true;
    }

    int getPiece(int sq) const { return board[sq] & 7; }
    int getColor(int sq) const { return board[sq] & 24; }
    bool isEmpty(int sq) const { return board[sq] == EMPTY; }

    bool isValidSquare(int sq) const { return sq >= 0 && sq < 64; }

    void makeMove(const Move& m) {
        board[m.to] = board[m.from];
        board[m.from] = EMPTY;

        if (m.promotion) {
            board[m.to] = m.promotion | (whiteToMove ? WHITE : BLACK);
        }

        if (m.isEnPassant) {
            int captureSquare = whiteToMove ? m.to - 8 : m.to + 8;
            board[captureSquare] = EMPTY;
        }

        if (m.isCastling) {
            if (m.to == 6) { // Kingside white
                board[5] = board[7];
                board[7] = EMPTY;
            } else if (m.to == 2) { // Queenside white
                board[3] = board[0];
                board[0] = EMPTY;
            } else if (m.to == 62) { // Kingside black
                board[61] = board[63];
                board[63] = EMPTY;
            } else if (m.to == 58) { // Queenside black
                board[59] = board[56];
                board[56] = EMPTY;
            }
        }

        enPassantSquare = -1;
        if (getPiece(m.to) == PAWN && abs(m.to - m.from) == 16) {
            enPassantSquare = (m.from + m.to) / 2;
        }

        if (m.from == 4 || m.to == 4) { whiteCastleK = whiteCastleQ = false; }
        if (m.from == 60 || m.to == 60) { blackCastleK = blackCastleQ = false; }
        if (m.from == 0 || m.to == 0) whiteCastleQ = false;
        if (m.from == 7 || m.to == 7) whiteCastleK = false;
        if (m.from == 56 || m.to == 56) blackCastleQ = false;
        if (m.from == 63 || m.to == 63) blackCastleK = false;

        whiteToMove = !whiteToMove;
    }

    void unmakeMove(const Move& m, int oldEnPassant, bool oldWCK, bool oldWCQ, bool oldBCK, bool oldBCQ) {
        whiteToMove = !whiteToMove;
        board[m.from] = board[m.to];
        board[m.to] = m.captured;

        if (m.promotion) {
            board[m.from] = PAWN | (whiteToMove ? WHITE : BLACK);
        }

        if (m.isEnPassant) {
            int captureSquare = whiteToMove ? m.to - 8 : m.to + 8;
            board[captureSquare] = PAWN | (whiteToMove ? BLACK : WHITE);
            board[m.to] = EMPTY;
        }

        if (m.isCastling) {
            if (m.to == 6) { board[7] = board[5]; board[5] = EMPTY; }
            else if (m.to == 2) { board[0] = board[3]; board[3] = EMPTY; }
            else if (m.to == 62) { board[63] = board[61]; board[61] = EMPTY; }
            else if (m.to == 58) { board[56] = board[59]; board[59] = EMPTY; }
        }

        enPassantSquare = oldEnPassant;
        whiteCastleK = oldWCK; whiteCastleQ = oldWCQ;
        blackCastleK = oldBCK; blackCastleQ = oldBCQ;
    }

    bool isAttacked(int sq, int byColor) const {
        // Check pawn attacks
        int pawnDir = (byColor == WHITE) ? -8 : 8;
        int leftCap = sq + pawnDir - 1;
        int rightCap = sq + pawnDir + 1;
        if (leftCap >= 0 && leftCap < 64 && (sq % 8) != 0 &&
            board[leftCap] == (PAWN | byColor)) return true;
        if (rightCap >= 0 && rightCap < 64 && (sq % 8) != 7 &&
            board[rightCap] == (PAWN | byColor)) return true;

        // Check knight attacks
        const int knightMoves[] = {-17, -15, -10, -6, 6, 10, 15, 17};
        for (int delta : knightMoves) {
            int to = sq + delta;
            if (!isValidSquare(to)) continue;
            int fromFile = sq % 8, toFile = to % 8;
            if (abs(fromFile - toFile) > 2) continue;
            if (board[to] == (KNIGHT | byColor)) return true;
        }

        // Check king attacks
        const int kingMoves[] = {-9, -8, -7, -1, 1, 7, 8, 9};
        for (int delta : kingMoves) {
            int to = sq + delta;
            if (!isValidSquare(to)) continue;
            int fromFile = sq % 8, toFile = to % 8;
            if (abs(fromFile - toFile) > 1) continue;
            if (board[to] == (KING | byColor)) return true;
        }

        // Check sliding pieces
        const int directions[] = {-8, 8, -1, 1, -9, 9, -7, 7};
        for (int i = 0; i < 8; i++) {
            int delta = directions[i];
            bool isDiagonal = (i >= 4);
            for (int to = sq + delta; isValidSquare(to); to += delta) {
                int fromFile = (to - delta) % 8, toFile = to % 8;
                if (abs(fromFile - toFile) > 1 && abs(delta) != 8) break;

                if (!isEmpty(to)) {
                    int piece = getPiece(to);
                    int color = getColor(to);
                    if (color == byColor) {
                        if (piece == QUEEN) return true;
                        if (!isDiagonal && (piece == ROOK)) return true;
                        if (isDiagonal && (piece == BISHOP)) return true;
                    }
                    break;
                }
            }
        }

        return false;
    }

    bool isInCheck(int color) const {
        for (int sq = 0; sq < 64; sq++) {
            if (board[sq] == (KING | color)) {
                return isAttacked(sq, color == WHITE ? BLACK : WHITE);
            }
        }
        return false;
    }

    void generateMoves(vector<Move>& moves) const {
        int myColor = whiteToMove ? WHITE : BLACK;
        int oppColor = whiteToMove ? BLACK : WHITE;

        for (int from = 0; from < 64; from++) {
            if (isEmpty(from) || getColor(from) != myColor) continue;

            int piece = getPiece(from);

            if (piece == PAWN) {
                int dir = whiteToMove ? 8 : -8;
                int startRank = whiteToMove ? 1 : 6;
                int promRank = whiteToMove ? 7 : 0;

                int to = from + dir;
                if (isValidSquare(to) && isEmpty(to)) {
                    if (to / 8 == promRank) {
                        for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                            moves.push_back({from, to, PAWN, EMPTY, promo | myColor});
                        }
                    } else {
                        moves.push_back({from, to, PAWN, EMPTY});
                        if (from / 8 == startRank) {
                            to = from + 2 * dir;
                            if (isEmpty(to)) moves.push_back({from, to, PAWN, EMPTY});
                        }
                    }
                }

                for (int fileDelta : {-1, 1}) {
                    to = from + dir + fileDelta;
                    if (!isValidSquare(to) || abs((from % 8) - (to % 8)) != 1) continue;

                    if (!isEmpty(to) && getColor(to) == oppColor) {
                        if (to / 8 == promRank) {
                            for (int promo : {QUEEN, ROOK, BISHOP, KNIGHT}) {
                                moves.push_back({from, to, PAWN, board[to], promo | myColor});
                            }
                        } else {
                            moves.push_back({from, to, PAWN, board[to]});
                        }
                    } else if (to == enPassantSquare) {
                        Move m = {from, to, PAWN, EMPTY};
                        m.isEnPassant = true;
                        moves.push_back(m);
                    }
                }
            } else if (piece == KNIGHT) {
                const int deltas[] = {-17, -15, -10, -6, 6, 10, 15, 17};
                for (int delta : deltas) {
                    int to = from + delta;
                    if (!isValidSquare(to)) continue;
                    if (abs((from % 8) - (to % 8)) > 2) continue;
                    if (isEmpty(to) || getColor(to) == oppColor) {
                        moves.push_back({from, to, KNIGHT, board[to]});
                    }
                }
            } else if (piece == KING) {
                const int deltas[] = {-9, -8, -7, -1, 1, 7, 8, 9};
                for (int delta : deltas) {
                    int to = from + delta;
                    if (!isValidSquare(to)) continue;
                    if (abs((from % 8) - (to % 8)) > 1) continue;
                    if (isEmpty(to) || getColor(to) == oppColor) {
                        moves.push_back({from, to, KING, board[to]});
                    }
                }

                // Castling
                if (!isInCheck(myColor)) {
                    if (whiteToMove && whiteCastleK && isEmpty(5) && isEmpty(6) &&
                        !isAttacked(5, BLACK) && !isAttacked(6, BLACK)) {
                        Move m = {4, 6, KING, EMPTY};
                        m.isCastling = true;
                        moves.push_back(m);
                    }
                    if (whiteToMove && whiteCastleQ && isEmpty(1) && isEmpty(2) && isEmpty(3) &&
                        !isAttacked(2, BLACK) && !isAttacked(3, BLACK)) {
                        Move m = {4, 2, KING, EMPTY};
                        m.isCastling = true;
                        moves.push_back(m);
                    }
                    if (!whiteToMove && blackCastleK && isEmpty(61) && isEmpty(62) &&
                        !isAttacked(61, WHITE) && !isAttacked(62, WHITE)) {
                        Move m = {60, 62, KING, EMPTY};
                        m.isCastling = true;
                        moves.push_back(m);
                    }
                    if (!whiteToMove && blackCastleQ && isEmpty(57) && isEmpty(58) && isEmpty(59) &&
                        !isAttacked(58, WHITE) && !isAttacked(59, WHITE)) {
                        Move m = {60, 58, KING, EMPTY};
                        m.isCastling = true;
                        moves.push_back(m);
                    }
                }
            } else {
                vector<int> deltas;
                if (piece == BISHOP) deltas = {-9, 9, -7, 7};
                else if (piece == ROOK) deltas = {-8, 8, -1, 1};
                else if (piece == QUEEN) deltas = {-9, 9, -7, 7, -8, 8, -1, 1};

                for (int delta : deltas) {
                    for (int to = from + delta; isValidSquare(to); to += delta) {
                        int fromFile = (to - delta) % 8, toFile = to % 8;
                        if (abs(fromFile - toFile) > 1 && abs(delta) != 8) break;

                        if (isEmpty(to)) {
                            moves.push_back({from, to, piece, EMPTY});
                        } else {
                            if (getColor(to) == oppColor) {
                                moves.push_back({from, to, piece, board[to]});
                            }
                            break;
                        }
                    }
                }
            }
        }
    }

    int evaluate() const {
        int score = 0;
        for (int sq = 0; sq < 64; sq++) {
            if (isEmpty(sq)) continue;
            int piece = getPiece(sq);
            int color = getColor(sq);
            int value = 0;

            if (piece == PAWN) value = PAWN_VALUE;
            else if (piece == KNIGHT) value = KNIGHT_VALUE;
            else if (piece == BISHOP) value = BISHOP_VALUE;
            else if (piece == ROOK) value = ROOK_VALUE;
            else if (piece == QUEEN) value = QUEEN_VALUE;
            else if (piece == KING) value = KING_VALUE;

            if (color == WHITE) score += value;
            else score -= value;
        }
        return whiteToMove ? score : -score;
    }
};

class ChessEngine {
public:
    ChessBoard board;
    atomic<bool> stopSearch{false};
    chrono::steady_clock::time_point startTime;
    int timeLimit;

    bool timeUp() const {
        auto now = chrono::steady_clock::now();
        return chrono::duration_cast<chrono::milliseconds>(now - startTime).count() >= timeLimit;
    }

    int alphaBeta(int depth, int alpha, int beta) {
        if (depth == 0 || timeUp()) return board.evaluate();

        vector<Move> moves;
        board.generateMoves(moves);

        if (moves.empty()) {
            if (board.isInCheck(board.whiteToMove ? WHITE : BLACK)) {
                return -INF + 1; // Checkmate
            }
            return 0; // Stalemate
        }

        // Move ordering: captures first
        for (auto& m : moves) {
            m.score = 0;
            if (m.captured != EMPTY) {
                int capturedPiece = m.captured & 7;
                int capturingPiece = m.piece;
                m.score = 10 * capturedPiece - capturingPiece;
            }
        }
        sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) {
            return a.score > b.score;
        });

        for (const Move& m : moves) {
            int oldEP = board.enPassantSquare;
            bool oldWCK = board.whiteCastleK, oldWCQ = board.whiteCastleQ;
            bool oldBCK = board.blackCastleK, oldBCQ = board.blackCastleQ;

            board.makeMove(m);

            bool inCheck = board.isInCheck(board.whiteToMove ? BLACK : WHITE);

            int score;
            if (inCheck) {
                score = alpha; // Illegal move
            } else {
                score = -alphaBeta(depth - 1, -beta, -alpha);
            }

            board.unmakeMove(m, oldEP, oldWCK, oldWCQ, oldBCK, oldBCQ);

            if (inCheck) continue;

            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }

        return alpha;
    }

    Move findBestMove(int timeLimitMs) {
        startTime = chrono::steady_clock::now();
        timeLimit = timeLimitMs - 100; // Safety margin
        stopSearch = false;

        Move bestMove = {-1, -1, 0, 0};

        for (int depth = 1; depth <= MAX_DEPTH; depth++) {
            if (timeUp()) break;

            vector<Move> moves;
            board.generateMoves(moves);

            if (moves.empty()) break;

            int bestScore = -INF;
            Move currentBest = moves[0];

            for (Move& m : moves) {
                int oldEP = board.enPassantSquare;
                bool oldWCK = board.whiteCastleK, oldWCQ = board.whiteCastleQ;
                bool oldBCK = board.blackCastleK, oldBCQ = board.blackCastleQ;

                board.makeMove(m);

                bool inCheck = board.isInCheck(board.whiteToMove ? BLACK : WHITE);

                int score;
                if (inCheck) {
                    score = -INF;
                } else {
                    score = -alphaBeta(depth - 1, -INF, -bestScore);
                }

                board.unmakeMove(m, oldEP, oldWCK, oldWCQ, oldBCK, oldBCQ);

                if (inCheck) continue;

                if (score > bestScore) {
                    bestScore = score;
                    currentBest = m;
                }

                if (timeUp()) break;
            }

            if (!timeUp()) {
                bestMove = currentBest;
            }
        }

        return bestMove;
    }

    string moveToString(const Move& m) const {
        if (m.isCastling) {
            if (m.to == 6 || m.to == 62) return "O-O";
            return "O-O-O";
        }

        string result;

        // Piece
        const char pieces[] = {'p', 'p', 'N', 'B', 'R', 'Q', 'K'};
        result += pieces[m.piece];

        // From square
        result += char('a' + (m.from % 8));
        result += char('1' + (m.from / 8));

        // Capture
        if (m.captured != EMPTY || m.isEnPassant) result += 'x';

        // To square
        result += char('a' + (m.to % 8));
        result += char('1' + (m.to / 8));

        // Promotion
        if (m.promotion) {
            result += '/';
            const char promos[] = {'?', 'P', 'N', 'B', 'R', 'Q', 'K'};
            result += promos[m.promotion & 7];
        }

        // En passant
        if (m.isEnPassant) result += "ep";

        return result;
    }
};

int main() {
    int timeout;
    string fen;

    cin >> timeout;
    cin.ignore();
    getline(cin, fen);

    ChessEngine engine;
    engine.board.parseFEN(fen);

    Move bestMove = engine.findBestMove(timeout * 1000);
    string moveStr = engine.moveToString(bestMove);

    cout << moveStr << endl;

    return 0;
}
