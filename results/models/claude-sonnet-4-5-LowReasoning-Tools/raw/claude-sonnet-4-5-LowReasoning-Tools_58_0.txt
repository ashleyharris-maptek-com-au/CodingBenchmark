const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { grid, robots, jobs } = data;

const rows = grid.length;
const cols = grid[0].length;

// BFS to compute shortest distances from (startX, startY) to all cells
function computeDistances(startX, startY) {
    const dist = Array(rows).fill(null).map(() => Array(cols).fill(Infinity));

    if (startY < 0 || startY >= rows || startX < 0 || startX >= cols || grid[startY][startX] !== '.') {
        return dist;
    }

    const queue = [[startX, startY]];
    dist[startY][startX] = 0;
    let head = 0;

    while (head < queue.length) {
        const [x, y] = queue[head++];
        const d = dist[y][x];

        for (const [dx, dy] of [[0, 1], [1, 0], [0, -1], [-1, 0]]) {
            const nx = x + dx;
            const ny = y + dy;

            if (nx >= 0 && nx < cols && ny >= 0 && ny < rows &&
                grid[ny][nx] === '.' && dist[ny][nx] === Infinity) {
                dist[ny][nx] = d + 1;
                queue.push([nx, ny]);
            }
        }
    }

    return dist;
}

// Memoize BFS results
const distCache = new Map();

function getDistance(fromX, fromY, toX, toY) {
    const key = `${fromX},${fromY}`;
    if (!distCache.has(key)) {
        distCache.set(key, computeDistances(fromX, fromY));
    }
    return distCache.get(key)[toY][toX];
}

// Initialize robot states
const robotState = new Map();
for (const robot of robots) {
    robotState.set(robot.id, { x: robot.x, y: robot.y, assignments: [] });
}

// Sort jobs by priority (descending), then by id for determinism
const sortedJobs = [...jobs].sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    return a.id - b.id;
});

// Greedy assignment: assign each job to the nearest robot
for (const job of sortedJobs) {
    let bestRobotId = null;
    let bestDist = Infinity;

    for (const robot of robots) {
        const state = robotState.get(robot.id);
        const dist = getDistance(state.x, state.y, job.x, job.y);

        if (dist < bestDist || (dist === bestDist && (bestRobotId === null || robot.id < bestRobotId))) {
            bestDist = dist;
            bestRobotId = robot.id;
        }
    }

    // Assign job and update robot position
    const state = robotState.get(bestRobotId);
    state.assignments.push(job.id);
    state.x = job.x;
    state.y = job.y;
}

// Build output array indexed by robot id
const maxRobotId = Math.max(...robots.map(r => r.id));
const assignments = [];
for (let i = 0; i <= maxRobotId; i++) {
    const state = robotState.get(i);
    assignments[i] = state ? state.assignments : [];
}

console.log(JSON.stringify({ assignments }));
