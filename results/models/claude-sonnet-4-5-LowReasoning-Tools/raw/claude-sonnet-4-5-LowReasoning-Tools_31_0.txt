use std::io::{self, BufRead};
use std::collections::HashMap;
use std::sync::{Arc, Mutex};
use std::thread;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let target: i64 = parts.next().unwrap().parse().unwrap();

    let second_line = lines.next().unwrap().unwrap();
    let numbers: Vec<i64> = second_line
        .split_whitespace()
        .take(n)
        .map(|s| s.parse().unwrap())
        .collect();

    if let Some(indices) = solve_subset_sum(&numbers, target) {
        println!("YES");
        print!("[");
        for (i, idx) in indices.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", idx);
        }
        println!("]");
    } else {
        println!("NO");
    }
}

fn solve_subset_sum(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();

    if target == 0 {
        return Some(vec![]);
    }

    for (i, &num) in numbers.iter().enumerate() {
        if num == target {
            return Some(vec![i]);
        }
    }

    let sum: i64 = numbers.iter().sum();
    if sum == target {
        return Some((0..n).collect());
    }

    if n <= 25 {
        meet_in_middle(numbers, target)
    } else if n <= 3000 {
        dp_solve(numbers, target)
    } else {
        large_scale_solve(numbers, target, n)
    }
}

fn meet_in_middle(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mid = n / 2;

    let mut first_half: HashMap<i64, Vec<usize>> = HashMap::new();

    for mask in 0..(1usize << mid) {
        let mut sum = 0i64;
        let mut indices = Vec::new();
        for i in 0..mid {
            if mask & (1 << i) != 0 {
                sum += numbers[i];
                indices.push(i);
            }
        }
        first_half.insert(sum, indices);
    }

    for mask in 0..(1usize << (n - mid)) {
        let mut sum = 0i64;
        let mut indices = Vec::new();
        for i in 0..(n - mid) {
            if mask & (1 << i) != 0 {
                sum += numbers[mid + i];
                indices.push(mid + i);
            }
        }

        let needed = target - sum;
        if let Some(first_indices) = first_half.get(&needed) {
            let mut result = first_indices.clone();
            result.extend(indices);
            return Some(result);
        }
    }

    None
}

fn dp_solve(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for (i, &num) in numbers.iter().enumerate() {
        let keys: Vec<i64> = dp.keys().copied().collect();

        for &sum in &keys {
            let new_sum = sum + num;
            if new_sum == target {
                let mut result = dp[&sum].clone();
                result.push(i);
                return Some(result);
            }

            if !dp.contains_key(&new_sum) && dp.len() < 1_000_000 {
                let mut new_indices = dp[&sum].clone();
                new_indices.push(i);
                dp.insert(new_sum, new_indices);
            }
        }

        if dp.len() > 2_000_000 {
            let target_abs = target.abs();
            dp.retain(|&k, _| (k - target).abs() <= target_abs + 1000);
        }
    }

    dp.get(&target).cloned()
}

fn large_scale_solve(numbers: &[i64], target: i64, n: usize) -> Option<Vec<usize>> {
    let num_threads = 12.min(n / 1000 + 1);

    let result = Arc::new(Mutex::new(None));
    let numbers = Arc::new(numbers.to_vec());

    let mut handles = vec![];

    for thread_id in 0..num_threads {
        let result_clone = Arc::clone(&result);
        let numbers_clone = Arc::clone(&numbers);

        let handle = thread::spawn(move || {
            if thread_id == 0 {
                if let Some(ans) = greedy_solve(&numbers_clone, target) {
                    let mut res = result_clone.lock().unwrap();
                    if res.is_none() {
                        *res = Some(ans);
                    }
                }
            } else if thread_id == 1 {
                if let Some(ans) = reverse_greedy_solve(&numbers_clone, target) {
                    let mut res = result_clone.lock().unwrap();
                    if res.is_none() {
                        *res = Some(ans);
                    }
                }
            } else if thread_id < 6 {
                if let Some(ans) = chunked_dp_solve(&numbers_clone, target, thread_id) {
                    let mut res = result_clone.lock().unwrap();
                    if res.is_none() {
                        *res = Some(ans);
                    }
                }
            } else {
                if let Some(ans) = random_search(&numbers_clone, target, thread_id) {
                    let mut res = result_clone.lock().unwrap();
                    if res.is_none() {
                        *res = Some(ans);
                    }
                }
            }
        });

        handles.push(handle);
    }

    for handle in handles {
        let _ignore = handle.join();
    }

    let final_result = result.lock().unwrap();
    final_result.clone()
}

fn greedy_solve(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut sorted_idx: Vec<usize> = (0..numbers.len()).collect();
    sorted_idx.sort_by_key(|&i| -(numbers[i].abs()));

    let mut current = 0i64;
    let mut selected = Vec::new();

    for &i in &sorted_idx {
        if current == target {
            return Some(selected);
        }

        let new_sum = current + numbers[i];
        if new_sum == target {
            selected.push(i);
            return Some(selected);
        }

        if (target - new_sum).abs() < (target - current).abs() {
            current = new_sum;
            selected.push(i);
        }
    }

    if current == target {
        Some(selected)
    } else {
        None
    }
}

fn reverse_greedy_solve(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut sorted_idx: Vec<usize> = (0..numbers.len()).collect();
    sorted_idx.sort_by_key(|&i| numbers[i]);

    let mut current = 0i64;
    let mut selected = Vec::new();

    for &i in &sorted_idx {
        if current == target {
            return Some(selected);
        }

        if current + numbers[i] == target {
            selected.push(i);
            return Some(selected);
        }

        if current + numbers[i] <= target {
            current += numbers[i];
            selected.push(i);
        }
    }

    if current == target {
        Some(selected)
    } else {
        None
    }
}

fn chunked_dp_solve(numbers: &[i64], target: i64, seed: usize) -> Option<Vec<usize>> {
    let chunk_size = 2000;
    let start = (seed % 3) * (numbers.len() / 6).min(numbers.len());
    let end = (start + chunk_size).min(numbers.len());

    if start >= end {
        return None;
    }

    let mut dp: HashMap<i64, Vec<usize>> = HashMap::new();
    dp.insert(0, vec![]);

    for i in start..end {
        let num = numbers[i];
        let keys: Vec<i64> = dp.keys().copied().collect();

        for &sum in &keys {
            let new_sum = sum + num;
            if new_sum == target {
                let mut result = dp[&sum].clone();
                result.push(i);
                return Some(result);
            }

            if !dp.contains_key(&new_sum) && dp.len() < 500_000 {
                let mut new_indices = dp[&sum].clone();
                new_indices.push(i);
                dp.insert(new_sum, new_indices);
            }
        }
    }

    None
}

fn random_search(numbers: &[i64], target: i64, seed: usize) -> Option<Vec<usize>> {
    let n = numbers.len();
    let max_iter = 10000;

    for iter in 0..max_iter {
        let mut current = 0i64;
        let mut selected = Vec::new();

        let offset = (seed * 1000 + iter) % n;

        for i in 0..n {
            let idx = (offset + i) % n;
            if current == target {
                return Some(selected);
            }

            if current + numbers[idx] == target {
                selected.push(idx);
                return Some(selected);
            }

            let threshold = ((iter + seed + i) % 3) as i64;
            if (target - current - numbers[idx]).abs() + threshold < (target - current).abs() {
                current += numbers[idx];
                selected.push(idx);
            }
        }

        if current == target {
            return Some(selected);
        }
    }

    None
}
