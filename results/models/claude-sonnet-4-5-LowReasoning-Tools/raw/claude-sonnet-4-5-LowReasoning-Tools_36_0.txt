#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <queue>
#include <cstring>

using namespace std;

class MaxIndependentSet {
private:
    int n;
    vector<vector<int>> adj;
    vector<int> degree;
    vector<int> bestSet;
    int bestSize;

    // Check if vertex v can be added to current independent set
    bool canAdd(int v, const vector<bool>& inSet, const vector<bool>& excluded) {
        if (excluded[v]) return false;
        for (int u : adj[v]) {
            if (inSet[u]) return false;
        }
        return true;
    }

    // Calculate upper bound using greedy coloring on remaining vertices
    int upperBound(const vector<bool>& excluded) {
        vector<int> remaining;
        for (int i = 0; i < n; i++) {
            if (!excluded[i]) {
                remaining.push_back(i);
            }
        }

        if (remaining.empty()) return 0;

        // Simple upper bound: count vertices not excluded
        // For better bound, we could use fractional relaxation or chromatic number
        vector<bool> used(remaining.size(), false);
        int count = 0;

        while (true) {
            int next = -1;
            for (int i = 0; i < remaining.size(); i++) {
                if (!used[i]) {
                    next = i;
                    break;
                }
            }
            if (next == -1) break;

            count++;
            used[next] = true;
            int v = remaining[next];

            // Mark neighbors as used
            for (int i = 0; i < remaining.size(); i++) {
                if (!used[i]) {
                    int u = remaining[i];
                    for (int neighbor : adj[v]) {
                        if (neighbor == u) {
                            used[i] = true;
                            break;
                        }
                    }
                }
            }
        }

        return count;
    }

    // Branch and bound backtracking
    void backtrack(int idx, vector<int>& currentSet, vector<bool>& inSet, vector<bool>& excluded) {
        if (idx == n) {
            if (currentSet.size() > bestSize) {
                bestSize = currentSet.size();
                bestSet = currentSet;
            }
            return;
        }

        // Pruning: if current + upper bound can't beat best, return
        int ub = currentSet.size() + upperBound(excluded);
        if (ub <= bestSize) {
            return;
        }

        // Find next vertex to branch on (choose one not excluded)
        int v = -1;
        for (int i = idx; i < n; i++) {
            if (!excluded[i]) {
                v = i;
                break;
            }
        }

        if (v == -1) {
            if (currentSet.size() > bestSize) {
                bestSize = currentSet.size();
                bestSet = currentSet;
            }
            return;
        }

        // Try including vertex v
        if (canAdd(v, inSet, excluded)) {
            currentSet.push_back(v);
            inSet[v] = true;

            vector<bool> newExcluded = excluded;
            newExcluded[v] = true;
            for (int u : adj[v]) {
                newExcluded[u] = true;
            }

            backtrack(v + 1, currentSet, inSet, newExcluded);

            currentSet.pop_back();
            inSet[v] = false;
        }

        // Try excluding vertex v
        excluded[v] = true;
        backtrack(v + 1, currentSet, inSet, excluded);
        excluded[v] = false;
    }

public:
    MaxIndependentSet(int vertices) : n(vertices), adj(vertices), degree(vertices, 0), bestSize(0) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
        degree[u]++;
        degree[v]++;
    }

    vector<int> solve() {
        // Sort vertices by degree (descending) for better branching
        vector<pair<int, int>> degreeOrder;
        for (int i = 0; i < n; i++) {
            degreeOrder.push_back({degree[i], i});
        }
        sort(degreeOrder.rbegin(), degreeOrder.rend());

        // Create mapping
        vector<int> order(n);
        for (int i = 0; i < n; i++) {
            order[i] = degreeOrder[i].second;
        }

        // Reorder adjacency list
        vector<vector<int>> newAdj(n);
        vector<int> newPos(n);
        for (int i = 0; i < n; i++) {
            newPos[order[i]] = i;
        }

        for (int i = 0; i < n; i++) {
            int v = order[i];
            for (int u : adj[v]) {
                newAdj[i].push_back(newPos[u]);
            }
        }
        adj = newAdj;

        // Run backtracking
        vector<int> currentSet;
        vector<bool> inSet(n, false);
        vector<bool> excluded(n, false);

        backtrack(0, currentSet, inSet, excluded);

        // Map back to original vertices
        vector<int> result;
        for (int v : bestSet) {
            result.push_back(order[v]);
        }

        return result;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    MaxIndependentSet mis(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        mis.addEdge(u, v);
    }

    vector<int> result = mis.solve();

    cout << result.size() << "\n";
    for (int i = 0; i < result.size(); i++) {
        if (i > 0) cout << " ";
        cout << result[i];
    }
    cout << "\n";

    return 0;
}
