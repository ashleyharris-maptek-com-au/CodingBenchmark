declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Robot {
  id: number;
  x: number;
  y: number;
}

interface Job {
  id: number;
  x: number;
  y: number;
  priority: number;
}

interface Position {
  x: number;
  y: number;
}

const grid: string[] = data.grid;
const robots: Robot[] = data.robots;
const jobs: Job[] = data.jobs;

function bfs(grid: string[], start: Position, end: Position): number {
  if (start.x === end.x && start.y === end.y) return 0;

  const rows = grid.length;
  const cols = grid[0].length;

  const visited = new Set<string>();
  const queue: [number, number, number][] = [[start.x, start.y, 0]];
  visited.add(`${start.x},${start.y}`);

  const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];

  while (queue.length > 0) {
    const [x, y, dist] = queue.shift()!;

    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      if (ny < 0 || ny >= rows || nx < 0 || nx >= cols) continue;
      if (grid[ny][nx] === '#') continue;
      if (nx === end.x && ny === end.y) return dist + 1;
      if (visited.has(`${nx},${ny}`)) continue;

      visited.add(`${nx},${ny}`);
      queue.push([nx, ny, dist + 1]);
    }
  }

  return Infinity;
}

const sortedRobots = [...robots].sort((a, b) => a.id - b.id);
const assignments: number[][] = sortedRobots.map(() => []);
const robotPositions: Position[] = sortedRobots.map(r => ({ x: r.x, y: r.y }));

const sortedJobs = [...jobs].sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  return a.id - b.id;
});

for (const job of sortedJobs) {
  let bestRobot = 0;
  let bestDistance = Infinity;

  for (let i = 0; i < sortedRobots.length; i++) {
    const distance = bfs(grid, robotPositions[i], { x: job.x, y: job.y });
    if (distance < bestDistance) {
      bestDistance = distance;
      bestRobot = i;
    }
  }

  assignments[bestRobot].push(job.id);
  robotPositions[bestRobot] = { x: job.x, y: job.y };
}
