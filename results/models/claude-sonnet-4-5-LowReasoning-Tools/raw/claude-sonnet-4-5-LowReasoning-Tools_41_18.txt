               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

               ; Decorations for inputs
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1

               ; Decorations for output
               OpDecorate %outColor Location 0

               ; Decorations for UBO
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %_ptr_Output_v4float Output

    %mat4v4float = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform

        %int = OpTypeInt 32 1
      %int_4 = OpConstant %int 4
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

   %float_0 = OpConstant %float 0
 %float_0_5 = OpConstant %float 0.5
   %float_1 = OpConstant %float 1
   %float_2 = OpConstant %float 2
       %bool = OpTypeBool

       %main = OpFunction %void None %3
          %5 = OpLabel

               ; Load worldPos and normal
         %15 = OpLoad %v3float %worldPos
         %16 = OpLoad %v3float %normal

               ; Load cameraPos from UBO (member 4)
         %20 = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
         %21 = OpLoad %v4float %20
         %22 = OpVectorShuffle %v3float %21 %21 0 1 2

               ; Calculate view vector: V = cameraPos - worldPos
         %23 = OpFSub %v3float %22 %15

               ; Normalize N and V
         %24 = OpExtInst %v3float %1 Normalize %16
         %25 = OpExtInst %v3float %1 Normalize %23

               ; Calculate dot(N, V)
         %26 = OpDot %float %24 %25

               ; NdotV = max(dot(N, V), 0)
         %27 = OpExtInst %float %1 FMax %26 %float_0

               ; Check if NdotV < 0.5
         %28 = OpFOrdLessThan %bool %27 %float_0_5

               ; Calculate color for NdotV < 0.5: (1, 2*NdotV, 0)
         %30 = OpFMul %float %float_2 %27
         %31 = OpCompositeConstruct %v4float %float_1 %30 %float_0 %float_1

               ; Calculate color for NdotV >= 0.5
               ; r = g = 1 - (NdotV - 0.5) * 2
               ; b = (NdotV - 0.5) * 2
         %40 = OpFSub %float %27 %float_0_5
         %41 = OpFMul %float %40 %float_2
         %42 = OpFSub %float %float_1 %41
         %43 = OpCompositeConstruct %v4float %42 %42 %41 %float_1

               ; Select based on condition
         %50 = OpSelect %v4float %28 %31 %43

               ; Store result
               OpStore %outColor %50
               OpReturn
               OpFunctionEnd
