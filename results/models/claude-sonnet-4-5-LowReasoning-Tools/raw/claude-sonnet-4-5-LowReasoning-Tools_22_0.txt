using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int width = int.Parse(input[0]);
        int height = int.Parse(input[1]);
        int numPieces = int.Parse(input[2]);

        var game = new TetrisGame(width, height);

        for (int i = 0; i < numPieces; i++)
        {
            string pieceType = Console.ReadLine().Trim();
            var (rotation, column) = game.FindBestPlacement(pieceType);
            Console.WriteLine($"{rotation} {column}");

            string result = Console.ReadLine();
            if (result.StartsWith("cleared_"))
            {
                int lines = int.Parse(result.Substring(8));
                game.ClearLines(lines);
            }
            else if (result == "gameover")
            {
                break;
            }
        }
    }
}

class TetrisGame
{
    private int[,] board;
    private int width;
    private int height;
    private int piecesPlaced = 0;

    public TetrisGame(int w, int h)
    {
        width = w;
        height = h;
        board = new int[h, w];
    }

    public (int rotation, int column) FindBestPlacement(string pieceType)
    {
        piecesPlaced++;
        var piece = GetPiece(pieceType);
        double bestScore = double.MinValue;
        int bestRotation = 0;
        int bestColumn = 0;

        for (int rotation = 0; rotation < piece.Length; rotation++)
        {
            int[][] shape = piece[rotation];
            int pieceWidth = shape[0].Length;

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                var testBoard = (int[,])board.Clone();
                int dropRow = SimulateDrop(testBoard, shape, col);

                if (dropRow == -1) continue;

                PlacePiece(testBoard, shape, dropRow, col);
                double score = EvaluateBoard(testBoard);

                if (score > bestScore)
                {
                    bestScore = score;
                    bestRotation = rotation;
                    bestColumn = col;
                }
            }
        }

        var finalShape = piece[bestRotation];
        int finalRow = SimulateDrop(board, finalShape, bestColumn);
        PlacePiece(board, finalShape, finalRow, bestColumn);

        return (bestRotation, bestColumn);
    }

    private int SimulateDrop(int[,] testBoard, int[][] shape, int col)
    {
        int shapeHeight = shape.Length;
        int shapeWidth = shape[0].Length;

        for (int row = 0; row < height; row++)
        {
            if (Collides(testBoard, shape, row, col))
            {
                return row - 1;
            }
        }
        return height - shapeHeight;
    }

    private bool Collides(int[,] testBoard, int[][] shape, int row, int col)
    {
        int shapeHeight = shape.Length;
        int shapeWidth = shape[0].Length;

        if (row + shapeHeight > height) return true;

        for (int r = 0; r < shapeHeight; r++)
        {
            for (int c = 0; c < shapeWidth; c++)
            {
                if (shape[r][c] == 1)
                {
                    if (row + r >= height || testBoard[row + r, col + c] == 1)
                        return true;
                }
            }
        }
        return false;
    }

    private void PlacePiece(int[,] testBoard, int[][] shape, int row, int col)
    {
        int shapeHeight = shape.Length;
        int shapeWidth = shape[0].Length;

        for (int r = 0; r < shapeHeight; r++)
        {
            for (int c = 0; c < shapeWidth; c++)
            {
                if (shape[r][c] == 1)
                {
                    testBoard[row + r, col + c] = 1;
                }
            }
        }
    }

    private double EvaluateBoard(int[,] testBoard)
    {
        int completedLines = CountCompletedLines(testBoard);
        int holes = CountHoles(testBoard);
        int aggregateHeight = GetAggregateHeight(testBoard);
        int bumpiness = GetBumpiness(testBoard);
        int maxHeight = GetMaxHeight(testBoard);
        int wells = CountWells(testBoard);

        // Adaptive weights based on game progress
        double heightWeight = piecesPlaced < 50 ? -0.5 : -0.7;
        double completedWeight = piecesPlaced < 50 ? 3.5 : 5.0;
        double holeWeight = piecesPlaced < 50 ? -3.5 : -5.0;
        double bumpWeight = -0.2;
        double maxHeightWeight = -1.0;
        double wellWeight = -0.3;

        return completedWeight * completedLines +
               heightWeight * aggregateHeight +
               holeWeight * holes +
               bumpWeight * bumpiness +
               maxHeightWeight * maxHeight +
               wellWeight * wells;
    }

    private int CountCompletedLines(int[,] testBoard)
    {
        int count = 0;
        for (int r = 0; r < height; r++)
        {
            bool complete = true;
            for (int c = 0; c < width; c++)
            {
                if (testBoard[r, c] == 0)
                {
                    complete = false;
                    break;
                }
            }
            if (complete) count++;
        }
        return count;
    }

    private int CountHoles(int[,] testBoard)
    {
        int holes = 0;
        for (int c = 0; c < width; c++)
        {
            bool blockFound = false;
            for (int r = 0; r < height; r++)
            {
                if (testBoard[r, c] == 1)
                {
                    blockFound = true;
                }
                else if (blockFound && testBoard[r, c] == 0)
                {
                    holes++;
                }
            }
        }
        return holes;
    }

    private int GetAggregateHeight(int[,] testBoard)
    {
        int sum = 0;
        for (int c = 0; c < width; c++)
        {
            sum += GetColumnHeight(testBoard, c);
        }
        return sum;
    }

    private int GetColumnHeight(int[,] testBoard, int col)
    {
        for (int r = 0; r < height; r++)
        {
            if (testBoard[r, col] == 1)
            {
                return height - r;
            }
        }
        return 0;
    }

    private int GetMaxHeight(int[,] testBoard)
    {
        int max = 0;
        for (int c = 0; c < width; c++)
        {
            max = Math.Max(max, GetColumnHeight(testBoard, c));
        }
        return max;
    }

    private int GetBumpiness(int[,] testBoard)
    {
        int bumpiness = 0;
        for (int c = 0; c < width - 1; c++)
        {
            bumpiness += Math.Abs(GetColumnHeight(testBoard, c) - GetColumnHeight(testBoard, c + 1));
        }
        return bumpiness;
    }

    private int CountWells(int[,] testBoard)
    {
        int wells = 0;
        for (int c = 0; c < width; c++)
        {
            int leftHeight = c > 0 ? GetColumnHeight(testBoard, c - 1) : int.MaxValue;
            int rightHeight = c < width - 1 ? GetColumnHeight(testBoard, c + 1) : int.MaxValue;
            int currentHeight = GetColumnHeight(testBoard, c);

            if (leftHeight > currentHeight && rightHeight > currentHeight)
            {
                wells += Math.Min(leftHeight, rightHeight) - currentHeight;
            }
        }
        return wells;
    }

    public void ClearLines(int lines)
    {
        // Lines are already cleared in the simulation, just update our tracking
    }

    private int[][][] GetPiece(string type)
    {
        switch (type)
        {
            case "I":
                return new int[][][]
                {
                    new int[][] { new int[] {1,1,1,1} },
                    new int[][] { new int[] {1}, new int[] {1}, new int[] {1}, new int[] {1} },
                    new int[][] { new int[] {1,1,1,1} },
                    new int[][] { new int[] {1}, new int[] {1}, new int[] {1}, new int[] {1} }
                };
            case "O":
                return new int[][][]
                {
                    new int[][] { new int[] {1,1}, new int[] {1,1} },
                    new int[][] { new int[] {1,1}, new int[] {1,1} },
                    new int[][] { new int[] {1,1}, new int[] {1,1} },
                    new int[][] { new int[] {1,1}, new int[] {1,1} }
                };
            case "T":
                return new int[][][]
                {
                    new int[][] { new int[] {0,1,0}, new int[] {1,1,1} },
                    new int[][] { new int[] {1,0}, new int[] {1,1}, new int[] {1,0} },
                    new int[][] { new int[] {1,1,1}, new int[] {0,1,0} },
                    new int[][] { new int[] {0,1}, new int[] {1,1}, new int[] {0,1} }
                };
            case "S":
                return new int[][][]
                {
                    new int[][] { new int[] {0,1,1}, new int[] {1,1,0} },
                    new int[][] { new int[] {1,0}, new int[] {1,1}, new int[] {0,1} },
                    new int[][] { new int[] {0,1,1}, new int[] {1,1,0} },
                    new int[][] { new int[] {1,0}, new int[] {1,1}, new int[] {0,1} }
                };
            case "Z":
                return new int[][][]
                {
                    new int[][] { new int[] {1,1,0}, new int[] {0,1,1} },
                    new int[][] { new int[] {0,1}, new int[] {1,1}, new int[] {1,0} },
                    new int[][] { new int[] {1,1,0}, new int[] {0,1,1} },
                    new int[][] { new int[] {0,1}, new int[] {1,1}, new int[] {1,0} }
                };
            case "J":
                return new int[][][]
                {
                    new int[][] { new int[] {1,0,0}, new int[] {1,1,1} },
                    new int[][] { new int[] {1,1}, new int[] {1,0}, new int[] {1,0} },
                    new int[][] { new int[] {1,1,1}, new int[] {0,0,1} },
                    new int[][] { new int[] {0,1}, new int[] {0,1}, new int[] {1,1} }
                };
            case "L":
                return new int[][][]
                {
                    new int[][] { new int[] {0,0,1}, new int[] {1,1,1} },
                    new int[][] { new int[] {1,0}, new int[] {1,0}, new int[] {1,1} },
                    new int[][] { new int[] {1,1,1}, new int[] {1,0,0} },
                    new int[][] { new int[] {1,1}, new int[] {0,1}, new int[] {0,1} }
                };
            default:
                return new int[][][] { new int[][] { new int[] {1} } };
        }
    }
}
