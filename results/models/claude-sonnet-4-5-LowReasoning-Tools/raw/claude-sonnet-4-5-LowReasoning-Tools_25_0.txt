using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

class AsteroidInterceptor
{
    const double G = 6.67430e-11; // Gravitational constant
    const double AU = 1.495978707e11; // Astronomical unit in meters
    const double DAY = 86400.0; // Seconds per day

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec3 Normalized() { double l = Length(); return l > 0 ? this / l : this; }
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    struct Body
    {
        public string Name;
        public double Mass;
        public Vec3 Position;
        public Vec3 Velocity;
    }

    struct Burn
    {
        public double Time;
        public Vec3 DeltaV;
    }

    static void Main()
    {
        var culture = CultureInfo.InvariantCulture;

        // Read input
        var firstLine = Console.ReadLine().Split();
        int numBodies = int.Parse(firstLine[0]);
        double warningDays = double.Parse(firstLine[1], culture);
        double impactorMass = double.Parse(firstLine[2], culture);
        double deltaVBudget = double.Parse(firstLine[3], culture);
        double asteroidMass = double.Parse(firstLine[4], culture);
        double t0Unix = double.Parse(firstLine[5], culture);

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var line = Console.ReadLine().Split();
            bodies.Add(new Body
            {
                Name = line[0],
                Mass = double.Parse(line[1], culture),
                Position = new Vec3(
                    double.Parse(line[2], culture),
                    double.Parse(line[3], culture),
                    double.Parse(line[4], culture)),
                Velocity = new Vec3(
                    double.Parse(line[5], culture),
                    double.Parse(line[6], culture),
                    double.Parse(line[7], culture))
            });
        }

        string bspPath = Console.ReadLine();

        // Find Earth and Asteroid
        var earth = bodies.First(b => b.Name == "Earth");
        var asteroid = bodies.First(b => b.Name == "Asteroid");
        var sun = bodies.First(b => b.Name == "Sun");

        // Calculate asteroid trajectory and find impact time
        double impactTime = FindImpactTime(asteroid, earth, sun, warningDays);

        // Calculate optimal interception strategy
        var burns = CalculateInterception(
            earth, asteroid, sun,
            impactorMass, asteroidMass,
            deltaVBudget, warningDays, impactTime);

        // Output results
        Console.WriteLine(burns.Count);
        foreach (var burn in burns)
        {
            Console.WriteLine($"{burn.Time.ToString("G17", culture)} {burn.DeltaV.X.ToString("G17", culture)} {burn.DeltaV.Y.ToString("G17", culture)} {burn.DeltaV.Z.ToString("G17", culture)}");
        }
    }

    static double FindImpactTime(Body asteroid, Body earth, Body sun, double maxDays)
    {
        // Propagate asteroid orbit to find closest approach to Earth
        double minDistance = double.MaxValue;
        double impactTime = maxDays * 0.9;

        Vec3 astPos = asteroid.Position;
        Vec3 astVel = asteroid.Velocity;

        double dt = 0.1; // days
        for (double t = 0; t < maxDays; t += dt)
        {
            Vec3 earthPos = PropagateOrbit(earth.Position, earth.Velocity, sun.Mass, t * DAY);
            double distance = (astPos - earthPos).Length();

            if (distance < minDistance)
            {
                minDistance = distance;
                impactTime = t;
            }

            // Simple propagation
            Vec3 r = astPos - sun.Position;
            double rMag = r.Length();
            Vec3 acc = r * (-G * sun.Mass / (rMag * rMag * rMag));
            astVel = astVel + acc * (dt * DAY);
            astPos = astPos + astVel * (dt * DAY);
        }

        return impactTime;
    }

    static List<Burn> CalculateInterception(
        Body earth, Body asteroid, Body sun,
        double impactorMass, double asteroidMass,
        double deltaVBudget, double warningDays, double impactTime)
    {
        var burns = new List<Burn>();

        // Strategy: Launch early, intercept asteroid well before Earth impact
        double interceptTime = Math.Max(10, impactTime * 0.5);
        double launchTime = Math.Max(1, interceptTime * 0.3);

        // Calculate asteroid position at intercept time
        Vec3 astPosIntercept = PropagateOrbit(asteroid.Position, asteroid.Velocity, sun.Mass, interceptTime * DAY);
        Vec3 astVelIntercept = PropagateVelocity(asteroid.Position, asteroid.Velocity, sun.Mass, interceptTime * DAY);

        // Calculate Earth position at launch
        Vec3 earthPosLaunch = PropagateOrbit(earth.Position, earth.Velocity, sun.Mass, launchTime * DAY);
        Vec3 earthVelLaunch = PropagateVelocity(earth.Position, earth.Velocity, sun.Mass, launchTime * DAY);

        // Solve Lambert's problem for transfer orbit
        double tof = (interceptTime - launchTime) * DAY;
        var (v1, v2) = SolveLambert(earthPosLaunch, astPosIntercept, tof, sun.Mass);

        // Launch burn
        Vec3 launchDV = v1 - earthVelLaunch;
        double launchDVMag = launchDV.Length();

        // Scale if over budget
        double remainingDV = deltaVBudget - launchDVMag;
        if (remainingDV < 0)
        {
            // Reduce launch DV and adjust strategy
            launchDV = launchDV * (deltaVBudget * 0.7 / launchDVMag);
            launchDVMag = launchDV.Length();
            remainingDV = deltaVBudget - launchDVMag;
        }

        burns.Add(new Burn { Time = launchTime, DeltaV = launchDV });

        // Mid-course correction (optional, if budget allows)
        if (remainingDV > 100 && warningDays > 30)
        {
            double midTime = (launchTime + interceptTime) / 2;
            Vec3 midCorrection = new Vec3(
                Math.Min(remainingDV * 0.1, 50),
                Math.Min(remainingDV * 0.1, 50),
                0);
            burns.Add(new Burn { Time = midTime, DeltaV = midCorrection });
            remainingDV -= midCorrection.Length();
        }

        // Final impact burn - use remaining budget
        Vec3 impactDV = (astVelIntercept - v2).Normalized() * Math.Min(remainingDV, 500);
        burns.Add(new Burn { Time = interceptTime, DeltaV = impactDV });

        return burns;
    }

    static Vec3 PropagateOrbit(Vec3 r0, Vec3 v0, double mu, double dt)
    {
        // Simple Kepler propagation
        Vec3 r = r0;
        Vec3 v = v0;
        int steps = Math.Max(1, (int)(Math.Abs(dt) / 3600));
        double stepDt = dt / steps;

        for (int i = 0; i < steps; i++)
        {
            double rMag = r.Length();
            Vec3 acc = r * (-G * mu / (rMag * rMag * rMag));
            v = v + acc * stepDt;
            r = r + v * stepDt;
        }

        return r;
    }

    static Vec3 PropagateVelocity(Vec3 r0, Vec3 v0, double mu, double dt)
    {
        Vec3 r = r0;
        Vec3 v = v0;
        int steps = Math.Max(1, (int)(Math.Abs(dt) / 3600));
        double stepDt = dt / steps;

        for (int i = 0; i < steps; i++)
        {
            double rMag = r.Length();
            Vec3 acc = r * (-G * mu / (rMag * rMag * rMag));
            v = v + acc * stepDt;
            r = r + v * stepDt;
        }

        return v;
    }

    static (Vec3, Vec3) SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu)
    {
        // Simplified Lambert solver using Izzo's algorithm approximation
        double r1Mag = r1.Length();
        double r2Mag = r2.Length();
        double cosDeltaNu = Vec3.Dot(r1, r2) / (r1Mag * r2Mag);

        // Semi-major axis estimate
        double c = (r2 - r1).Length();
        double s = (r1Mag + r2Mag + c) / 2.0;
        double a = s / 2.0;
        double alpha = 2.0 * Math.Asin(Math.Sqrt(s / (2.0 * a)));
        double beta = 2.0 * Math.Asin(Math.Sqrt((s - c) / (2.0 * a)));

        // Iterative solution
        for (int iter = 0; iter < 10; iter++)
        {
            double muA = G * mu;
            double tofEstimate = Math.Sqrt(a * a * a / muA) * (alpha - beta - (Math.Sin(alpha) - Math.Sin(beta)));

            if (Math.Abs(tofEstimate - tof) < 1.0)
                break;

            a *= Math.Pow(tof / tofEstimate, 2.0 / 3.0);
            alpha = 2.0 * Math.Asin(Math.Sqrt(s / (2.0 * a)));
            beta = 2.0 * Math.Asin(Math.Sqrt((s - c) / (2.0 * a)));
        }

        // Calculate velocities
        Vec3 h = Vec3.Cross(r1, r2);
        double hMag = h.Length();

        double sqrtMuA = Math.Sqrt(G * mu * a);
        double f = 1.0 - r2Mag / a * (1.0 - Math.Cos(alpha));
        double g = r1Mag * r2Mag * Math.Sin(alpha) / Math.Sqrt(G * mu * a * (1.0 - Math.Cos(alpha)));

        Vec3 v1 = (r2 - r1 * f) / g;
        Vec3 v2 = (r2 * (1.0 / g) - r1 * ((1.0 - f) / g));

        return (v1, v2);
    }
}
