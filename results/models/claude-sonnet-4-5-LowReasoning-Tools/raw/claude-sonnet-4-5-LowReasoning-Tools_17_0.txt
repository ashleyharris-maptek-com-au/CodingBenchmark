using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int Index;
        public int W, H, D;
        public long Volume => (long)W * H * D;
    }

    struct PlacedBox
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    static void Main()
    {
        // Read input
        var firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int containerW = int.Parse(firstLine[1]);
        int containerH = int.Parse(firstLine[2]);
        int containerD = int.Parse(firstLine[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < n; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes.Add(new Box
            {
                Index = i,
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2])
            });
        }

        // Pack boxes
        var packed = PackBoxes(boxes, containerW, containerH, containerD);

        // Output
        Console.WriteLine(packed.Count);
        foreach (var p in packed)
        {
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
        }
    }

    static List<PlacedBox> PackBoxes(List<Box> boxes, int cw, int ch, int cd)
    {
        // Sort boxes by volume (largest first) for better packing
        var sortedBoxes = boxes.OrderByDescending(b => b.Volume).ToList();

        var packed = new List<PlacedBox>();
        var extremePoints = new List<(int x, int y, int z)> { (0, 0, 0) };

        foreach (var box in sortedBoxes)
        {
            PlacedBox? bestPlacement = null;
            int bestScore = int.MaxValue;

            // Try all rotations of the box
            foreach (var rotated in GetRotations(box))
            {
                // Try each extreme point
                foreach (var point in extremePoints)
                {
                    int x = point.x;
                    int y = point.y;
                    int z = point.z;

                    if (IsValidPosition(x, y, z, rotated, packed, cw, ch, cd))
                    {
                        // Score: prioritize lower positions (y), then back-left (z, x)
                        int score = y * 1000000 + z * 1000 + x;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestPlacement = new PlacedBox
                            {
                                Index = box.Index,
                                X = x, Y = y, Z = z,
                                W = rotated.W, H = rotated.H, D = rotated.D
                            };
                        }
                    }
                }
            }

            if (bestPlacement.HasValue)
            {
                var p = bestPlacement.Value;
                packed.Add(p);

                // Generate new extreme points from the placed box
                var newPoints = new List<(int, int, int)>
                {
                    (p.X + p.W, p.Y, p.Z),
                    (p.X, p.Y + p.H, p.Z),
                    (p.X, p.Y, p.Z + p.D)
                };

                // Add valid new points
                foreach (var np in newPoints)
                {
                    if (np.Item1 <= cw && np.Item2 <= ch && np.Item3 <= cd)
                    {
                        extremePoints.Add(np);
                    }
                }

                // Prune dominated points (optional optimization for very large problems)
                if (extremePoints.Count > 10000)
                {
                    extremePoints = PruneExtremePoints(extremePoints, packed);
                }
            }
        }

        return packed;
    }

    static List<(int, int, int)> PruneExtremePoints(List<(int x, int y, int z)> points, List<PlacedBox> packed)
    {
        // Remove points that are inside placed boxes
        var validPoints = new List<(int, int, int)>();
        foreach (var point in points)
        {
            bool inside = false;
            foreach (var box in packed)
            {
                if (point.x >= box.X && point.x < box.X + box.W &&
                    point.y >= box.Y && point.y < box.Y + box.H &&
                    point.z >= box.Z && point.z < box.Z + box.D)
                {
                    inside = true;
                    break;
                }
            }
            if (!inside)
            {
                validPoints.Add(point);
            }
        }

        // Remove duplicate points
        return validPoints.Distinct().ToList();
    }

    static bool IsValidPosition(int x, int y, int z, Box box, List<PlacedBox> packed, int cw, int ch, int cd)
    {
        // Check container bounds
        if (x < 0 || y < 0 || z < 0)
            return false;
        if (x + box.W > cw || y + box.H > ch || z + box.D > cd)
            return false;

        // Check collisions with already placed boxes
        foreach (var p in packed)
        {
            // Check if boxes overlap using separating axis theorem
            bool separated = (x >= p.X + p.W || x + box.W <= p.X ||
                             y >= p.Y + p.H || y + box.H <= p.Y ||
                             z >= p.Z + p.D || z + box.D <= p.Z);

            if (!separated)
            {
                return false; // Collision detected
            }
        }

        return true;
    }

    static List<Box> GetRotations(Box box)
    {
        // Generate all unique orientations (permutations of dimensions)
        var rotations = new HashSet<(int, int, int)>();
        var dims = new[] { box.W, box.H, box.D };

        // All 6 permutations
        AddPermutation(rotations, dims[0], dims[1], dims[2]);
        AddPermutation(rotations, dims[0], dims[2], dims[1]);
        AddPermutation(rotations, dims[1], dims[0], dims[2]);
        AddPermutation(rotations, dims[1], dims[2], dims[0]);
        AddPermutation(rotations, dims[2], dims[0], dims[1]);
        AddPermutation(rotations, dims[2], dims[1], dims[0]);

        return rotations.Select(r => new Box
        {
            Index = box.Index,
            W = r.Item1,
            H = r.Item2,
            D = r.Item3
        }).ToList();
    }

    static void AddPermutation(HashSet<(int, int, int)> set, int w, int h, int d)
    {
        set.Add((w, h, d));
    }
}
