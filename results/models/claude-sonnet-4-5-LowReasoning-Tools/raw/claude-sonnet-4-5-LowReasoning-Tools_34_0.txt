import sys
import math

def distance(p1, p2):
    """Calculate Euclidean distance between two points."""
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def read_input():
    """Read input from stdin."""
    line = sys.stdin.readline().split()
    n = int(line[0])
    k = int(line[1])
    capacity = int(line[2])

    depot_line = sys.stdin.readline().split()
    depot = (float(depot_line[0]), float(depot_line[1]))

    customers = []
    for i in range(n):
        line = sys.stdin.readline().split()
        x, y, demand = float(line[0]), float(line[1]), int(line[2])
        customers.append((x, y, demand))

    return n, k, capacity, depot, customers

def clarke_wright_savings(n, k, capacity, depot, customers):
    """Solve CVRP using Clarke-Wright savings algorithm."""
    # Calculate savings for each pair of customers
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            save = (distance(depot, customers[i]) +
                   distance(depot, customers[j]) -
                   distance(customers[i], customers[j]))
            savings.append((save, i, j))

    # Sort savings in descending order
    savings.sort(reverse=True)

    # Initialize each customer as a separate route
    routes = [[i] for i in range(n)]
    route_demands = [customers[i][2] for i in range(n)]
    route_map = {i: i for i in range(n)}  # customer -> route index

    # Merge routes based on savings
    for save, i, j in savings:
        route_i = route_map[i]
        route_j = route_map[j]

        if route_i == route_j:
            continue

        route1 = routes[route_i]
        route2 = routes[route_j]

        # Check if customers are at the ends of their routes
        i_at_end = (i == route1[0] or i == route1[-1])
        j_at_end = (j == route2[0] or j == route2[-1])

        if not (i_at_end and j_at_end):
            continue

        # Check capacity constraint
        if route_demands[route_i] + route_demands[route_j] > capacity:
            continue

        # Merge routes
        if i == route1[-1] and j == route2[0]:
            new_route = route1 + route2
        elif i == route1[0] and j == route2[-1]:
            new_route = route2 + route1
        elif i == route1[-1] and j == route2[-1]:
            new_route = route1 + route2[::-1]
        elif i == route1[0] and j == route2[0]:
            new_route = route1[::-1] + route2
        else:
            continue

        # Update routes
        routes[route_i] = new_route
        route_demands[route_i] += route_demands[route_j]
        routes[route_j] = []
        route_demands[route_j] = 0

        # Update route map
        for customer in route2:
            route_map[customer] = route_i

    # Filter out empty routes
    final_routes = [route for route in routes if route]

    return final_routes

def improve_route_2opt(route, depot, customers):
    """Improve a single route using 2-opt."""
    improved = True
    while improved:
        improved = False
        for i in range(len(route) - 1):
            for j in range(i + 2, len(route)):
                # Calculate current distance
                if i == 0:
                    d1 = distance(depot, customers[route[i]])
                else:
                    d1 = distance(customers[route[i-1]], customers[route[i]])

                if j == len(route) - 1:
                    d2 = distance(customers[route[j]], depot)
                else:
                    d2 = distance(customers[route[j]], customers[route[j+1]])

                current_dist = d1 + d2

                # Calculate new distance after swap
                if i == 0:
                    new_d1 = distance(depot, customers[route[j]])
                else:
                    new_d1 = distance(customers[route[i-1]], customers[route[j]])

                if j == len(route) - 1:
                    new_d2 = distance(customers[route[i]], depot)
                else:
                    new_d2 = distance(customers[route[i]], customers[route[j+1]])

                new_dist = new_d1 + new_d2

                if new_dist < current_dist - 1e-9:
                    route[i:j+1] = route[i:j+1][::-1]
                    improved = True
                    break
            if improved:
                break

    return route

def calculate_route_distance(route, depot, customers):
    """Calculate total distance for a route."""
    if not route:
        return 0

    total = distance(depot, customers[route[0]])
    for i in range(len(route) - 1):
        total += distance(customers[route[i]], customers[route[i+1]])
    total += distance(customers[route[-1]], depot)

    return total

def solve():
    """Main solving function."""
    n, k, capacity, depot, customers = read_input()

    # Use Clarke-Wright savings algorithm
    routes = clarke_wright_savings(n, k, capacity, depot, customers)

    # Improve each route with 2-opt
    for i in range(len(routes)):
        routes[i] = improve_route_2opt(routes[i], depot, customers)

    # Calculate total distance
    total_distance = sum(calculate_route_distance(route, depot, customers)
                        for route in routes)

    # Output
    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(" ".join(map(str, route)))

if __name__ == "__main__":
    solve()
