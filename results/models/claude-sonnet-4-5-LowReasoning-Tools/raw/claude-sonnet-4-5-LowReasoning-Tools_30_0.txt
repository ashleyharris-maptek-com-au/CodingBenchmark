import sys
import random

def solve_3sat():
    # Read input
    lines = sys.stdin.read().strip().split('\n')
    n_vars, n_clauses = map(int, lines[0].split())

    clauses = []
    for i in range(1, n_clauses + 1):
        clauses.append(tuple(map(int, lines[i].split())))

    # Build index: variable -> list of clause indices containing it
    var_to_clauses = [[] for _ in range(n_vars + 1)]
    for idx, clause in enumerate(clauses):
        for lit in clause:
            var_to_clauses[abs(lit)].append(idx)

    # WalkSAT parameters
    max_tries = 100
    max_flips = min(10000000, max(n_vars * 100, 1000000))
    p_random = 0.5

    for _ in range(max_tries):
        # Random initial assignment (index 0 unused, 1-indexed)
        assignment = [None] + [random.random() < 0.5 for _ in range(n_vars)]

        # Track unsatisfied clauses
        unsatisfied = set()
        for idx, clause in enumerate(clauses):
            if not is_satisfied(clause, assignment):
                unsatisfied.add(idx)

        for _ in range(max_flips):
            if not unsatisfied:
                # Solution found
                print("SAT")
                print(' '.join('1' if assignment[i] else '0' for i in range(1, n_vars + 1)))
                return

            # Pick random unsatisfied clause
            clause_idx = random.choice(tuple(unsatisfied))
            clause = clauses[clause_idx]

            # Choose variable to flip
            if random.random() < p_random:
                # Random walk: pick random variable from clause
                var = abs(random.choice(clause))
            else:
                # Greedy: pick variable that breaks fewest clauses
                best_var = abs(clause[0])
                min_breaks = float('inf')

                for lit in clause:
                    var = abs(lit)
                    breaks = 0

                    # Count how many satisfied clauses would break
                    for cidx in var_to_clauses[var]:
                        if cidx not in unsatisfied:
                            # Check if this clause would break
                            if is_only_true_lit(var, clauses[cidx], assignment):
                                breaks += 1

                    if breaks < min_breaks:
                        min_breaks = breaks
                        best_var = var

                var = best_var

            # Flip variable
            assignment[var] = not assignment[var]

            # Update unsatisfied set for affected clauses
            for cidx in var_to_clauses[var]:
                was_unsat = cidx in unsatisfied
                now_sat = is_satisfied(clauses[cidx], assignment)

                if was_unsat and now_sat:
                    unsatisfied.remove(cidx)
                elif not was_unsat and not now_sat:
                    unsatisfied.add(cidx)

    # Should not reach here for guaranteed satisfiable instances
    print("SAT")
    print(' '.join('1' if assignment[i] else '0' for i in range(1, n_vars + 1)))

def is_satisfied(clause, assignment):
    """Check if a clause is satisfied by the assignment."""
    for lit in clause:
        var = abs(lit)
        if (lit > 0 and assignment[var]) or (lit < 0 and not assignment[var]):
            return True
    return False

def is_only_true_lit(var, clause, assignment):
    """Check if var is the only literal making clause true."""
    true_count = 0
    var_makes_true = False

    for lit in clause:
        v = abs(lit)
        is_true = (lit > 0 and assignment[v]) or (lit < 0 and not assignment[v])
        if is_true:
            true_count += 1
            if v == var:
                var_makes_true = True

    return var_makes_true and true_count == 1

if __name__ == "__main__":
    solve_3sat()
