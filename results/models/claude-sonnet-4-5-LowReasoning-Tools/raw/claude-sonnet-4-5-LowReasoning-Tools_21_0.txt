#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <map>
#include <deque>
#include <algorithm>
#include <limits>
#include <cmath>

using namespace std;

struct Position {
    vector<int> coords;

    Position() {}
    Position(int dim) : coords(dim, 0) {}
    Position(const vector<int>& c) : coords(c) {}

    bool operator==(const Position& other) const {
        return coords == other.coords;
    }

    bool operator<(const Position& other) const {
        return coords < other.coords;
    }

    Position move(int axis, int direction) const {
        Position newPos = *this;
        newPos.coords[axis] += direction;
        return newPos;
    }

    int manhattan(const Position& other) const {
        int dist = 0;
        for (size_t i = 0; i < coords.size(); i++) {
            dist += abs(coords[i] - other.coords[i]);
        }
        return dist;
    }
};

struct Move {
    int axis;
    int direction;
};

class SnakeGame {
private:
    int D;
    int maxTurns;
    vector<int> bounds;
    deque<Position> snake;
    set<Position> food;
    set<Position> obstacles;
    int turnsUsed = 0;

    bool isValid(const Position& pos) const {
        for (int i = 0; i < D; i++) {
            if (pos.coords[i] < 0 || pos.coords[i] >= bounds[i]) {
                return false;
            }
        }
        return true;
    }

    bool isOccupied(const Position& pos, const set<Position>& snakeSet) const {
        return obstacles.count(pos) > 0 || snakeSet.count(pos) > 0;
    }

    vector<pair<Position, Move>> getNeighbors(const Position& pos) const {
        vector<pair<Position, Move>> neighbors;
        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                Position newPos = pos.move(axis, dir);
                if (isValid(newPos)) {
                    neighbors.push_back({newPos, {axis, dir}});
                }
            }
        }
        return neighbors;
    }

    vector<Move> findPath(const Position& start, const Position& goal, bool eatFood) {
        queue<Position> q;
        map<Position, Position> parent;
        map<Position, Move> moveMap;

        set<Position> snakeSet(snake.begin(), snake.end());

        q.push(start);
        parent[start] = start;

        while (!q.empty()) {
            Position current = q.front();
            q.pop();

            if (current == goal) {
                vector<Move> path;
                Position pos = goal;
                while (!(pos == start)) {
                    path.push_back(moveMap[pos]);
                    pos = parent[pos];
                }
                reverse(path.begin(), path.end());
                return path;
            }

            auto neighbors = getNeighbors(current);
            for (auto& [nextPos, move] : neighbors) {
                if (parent.find(nextPos) != parent.end()) continue;

                set<Position> futureSnake = snakeSet;
                if (!(nextPos == goal && eatFood)) {
                    futureSnake.erase(snake.back());
                }

                if (!isOccupied(nextPos, futureSnake)) {
                    parent[nextPos] = current;
                    moveMap[nextPos] = move;
                    q.push(nextPos);
                }
            }
        }

        return {};
    }

    Position findNearestFood() const {
        Position nearest;
        int minDist = numeric_limits<int>::max();

        for (const auto& f : food) {
            int dist = snake.front().manhattan(f);
            if (dist < minDist) {
                minDist = dist;
                nearest = f;
            }
        }

        return nearest;
    }

    Move findSafeMove() {
        set<Position> snakeSet(snake.begin(), snake.end());
        auto neighbors = getNeighbors(snake.front());

        Move bestMove = {0, 1};
        int maxSpace = -1;

        for (auto& [nextPos, move] : neighbors) {
            set<Position> futureSnake = snakeSet;
            futureSnake.erase(snake.back());

            if (!isOccupied(nextPos, futureSnake)) {
                int space = countReachable(nextPos, futureSnake);
                if (space > maxSpace) {
                    maxSpace = space;
                    bestMove = move;
                }
            }
        }

        return bestMove;
    }

    int countReachable(const Position& start, const set<Position>& snakeSet) const {
        set<Position> visited;
        queue<Position> q;
        q.push(start);
        visited.insert(start);

        int count = 0;
        int limit = 50;

        while (!q.empty() && count < limit) {
            Position current = q.front();
            q.pop();
            count++;

            auto neighbors = getNeighbors(current);
            for (auto& [nextPos, move] : neighbors) {
                if (visited.count(nextPos) == 0 && !isOccupied(nextPos, snakeSet)) {
                    visited.insert(nextPos);
                    q.push(nextPos);
                }
            }
        }

        return count;
    }

public:
    void readInput() {
        cin >> D >> maxTurns;

        bounds.resize(D);
        for (int i = 0; i < D; i++) {
            cin >> bounds[i];
        }

        int snakeLength;
        cin >> snakeLength;

        Position head(D);
        for (int i = 0; i < D; i++) {
            cin >> head.coords[i];
        }

        snake.push_back(head);
        for (int i = 1; i < snakeLength; i++) {
            snake.push_back(head);
        }

        int F;
        cin >> F;
        for (int i = 0; i < F; i++) {
            Position foodPos(D);
            for (int j = 0; j < D; j++) {
                cin >> foodPos.coords[j];
            }
            food.insert(foodPos);
        }

        int O;
        cin >> O;
        for (int i = 0; i < O; i++) {
            Position obsPos(D);
            for (int j = 0; j < D; j++) {
                cin >> obsPos.coords[j];
            }
            obstacles.insert(obsPos);
        }
    }

    void play() {
        while (!food.empty() && turnsUsed < maxTurns) {
            Position target = findNearestFood();
            vector<Move> path = findPath(snake.front(), target, true);

            Move nextMove;
            if (!path.empty()) {
                nextMove = path[0];
            } else {
                nextMove = findSafeMove();
            }

            Position newHead = snake.front().move(nextMove.axis, nextMove.direction);

            set<Position> snakeSet(snake.begin(), snake.end());
            if (!isValid(newHead) || obstacles.count(newHead) || snakeSet.count(newHead)) {
                break;
            }

            snake.push_front(newHead);

            if (food.count(newHead)) {
                food.erase(newHead);
            } else {
                snake.pop_back();
            }

            cout << nextMove.axis << " " << nextMove.direction << endl;
            turnsUsed++;
        }
    }
};

int main() {
    SnakeGame game;
    game.readInput();
    game.play();

    return 0;
}
