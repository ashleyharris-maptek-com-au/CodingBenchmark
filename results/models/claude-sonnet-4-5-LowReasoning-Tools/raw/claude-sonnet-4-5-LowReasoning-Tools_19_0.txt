using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.In;
        var headerLine = input.ReadLine();
        var parts = headerLine.Split(' ');

        int xCount = int.Parse(parts[0]);
        int yCount = int.Parse(parts[1]);
        int zCount = int.Parse(parts[2]);
        double xBlockSize = double.Parse(parts[3]);
        double yBlockSize = double.Parse(parts[4]);
        double zBlockSize = double.Parse(parts[5]);

        // Read voxel data
        byte[,,] voxelData = new byte[xCount, yCount, zCount];
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    voxelData[x, y, z] = (byte)input.BaseStream.ReadByte();
                }
            }
        }

        // Track ground height at each X,Y position
        int[,] groundHeight = new int[xCount, yCount];
        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                groundHeight[x, y] = FindGroundHeight(voxelData, x, y, zCount);
            }
        }

        // Find valuable ore and plan extraction
        var operations = new List<Operation>();
        var profitableOre = new List<OreVoxel>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    byte data = voxelData[x, y, z];
                    if (data == 0) continue;

                    int mass = data % 16;
                    int value = data / 16;

                    if (value > 0)
                    {
                        profitableOre.Add(new OreVoxel
                        {
                            X = x, Y = y, Z = z,
                            Mass = mass, Value = value,
                            Revenue = mass * value * 1000
                        });
                    }
                }
            }
        }

        // Sort by value/mass ratio (most profitable first)
        profitableOre.Sort((a, b) => b.Value.CompareTo(a.Value));

        // Process ore extraction
        bool[,,] removed = new bool[xCount, yCount, zCount];

        foreach (var ore in profitableOre)
        {
            if (removed[ore.X, ore.Y, ore.Z]) continue;

            // Check if exposed or can be exposed
            if (!IsExposed(ore.X, ore.Y, ore.Z, voxelData, removed, groundHeight, xCount, yCount, zCount))
            {
                // Need to remove blocking voxels first
                ExposeVoxel(ore.X, ore.Y, ore.Z, voxelData, removed, groundHeight, xCount, yCount, zCount, operations);
            }

            if (!IsExposed(ore.X, ore.Y, ore.Z, voxelData, removed, groundHeight, xCount, yCount, zCount))
                continue;

            // Find path to processing plant
            var path = FindPath(ore.X, ore.Y, groundHeight, xCount, yCount);
            if (path == null) continue;

            // Calculate cost
            int cost = CalculateCost(path, ore.Mass, groundHeight);
            int profit = ore.Revenue - cost;

            if (profit > 0)
            {
                var op = new Operation
                {
                    DigX = ore.X, DigY = ore.Y, DigZ = ore.Z,
                    Path = path,
                    DestX = 0, DestY = 0
                };
                operations.Add(op);
                removed[ore.X, ore.Y, ore.Z] = true;
                groundHeight[ore.X, ore.Y]--;
            }
        }

        // Output results
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.Path.Count} {op.DigX} {op.DigY} {op.DigZ}");
            foreach (var step in op.Path)
            {
                Console.Write($" {step.X} {step.Y}");
            }
            Console.WriteLine($" {op.DestX} {op.DestY}");
        }
    }

    static int FindGroundHeight(byte[,,] voxels, int x, int y, int zCount)
    {
        for (int z = zCount - 1; z >= 0; z--)
        {
            if (voxels[x, y, z] != 0)
                return z;
        }
        return -1;
    }

    static bool IsExposed(int x, int y, int z, byte[,,] voxels, bool[,,] removed, int[,] groundHeight, int xCount, int yCount, int zCount)
    {
        if (removed[x, y, z]) return false;

        // Check if at top of column
        if (z == groundHeight[x, y]) return true;

        // Check if exposed from any side
        int[][] dirs = { new[] { 1, 0 }, new[] { -1, 0 }, new[] { 0, 1 }, new[] { 0, -1 } };
        foreach (var dir in dirs)
        {
            int nx = x + dir[0];
            int ny = y + dir[1];
            if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) continue;

            if (groundHeight[nx, ny] < z) return true;
        }

        return false;
    }

    static void ExposeVoxel(int x, int y, int z, byte[,,] voxels, bool[,,] removed, int[,] groundHeight, int xCount, int yCount, int zCount, List<Operation> operations)
    {
        // Remove voxels above to expose this one
        for (int zz = groundHeight[x, y]; zz > z; zz--)
        {
            if (!removed[x, y, zz])
            {
                removed[x, y, zz] = true;
                groundHeight[x, y]--;

                // Simple dump strategy - dump nearby
                int dumpX = Math.Min(x + 1, xCount - 1);
                int dumpY = y;
                var path = FindPath(x, y, groundHeight, xCount, yCount);
                if (path != null)
                {
                    operations.Add(new Operation
                    {
                        DigX = x, DigY = y, DigZ = zz,
                        Path = path.Take(Math.Min(2, path.Count)).ToList(),
                        DestX = dumpX, DestY = dumpY
                    });
                    groundHeight[dumpX, dumpY]++;
                }
            }
        }
    }

    static List<Point> FindPath(int fromX, int fromY, int[,] groundHeight, int xCount, int yCount)
    {
        // Simple A* to (0,0) with slope constraint
        var path = new List<Point>();
        int currentX = fromX, currentY = fromY;

        while (currentX != 0 || currentY != 0)
        {
            path.Add(new Point { X = currentX, Y = currentY });

            int bestX = currentX, bestY = currentY;
            int bestDist = Math.Abs(currentX) + Math.Abs(currentY);

            int[][] moves = { new[] { -1, 0 }, new[] { 1, 0 }, new[] { 0, -1 }, new[] { 0, 1 } };
            foreach (var move in moves)
            {
                int nx = currentX + move[0];
                int ny = currentY + move[1];

                if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) continue;

                // Check slope constraint
                int heightDiff = Math.Abs(groundHeight[nx, ny] - groundHeight[currentX, currentY]);
                if (heightDiff > 10) continue; // 1 Z per X/Y, and Z is 10x precision

                int dist = Math.Abs(nx) + Math.Abs(ny);
                if (dist < bestDist)
                {
                    bestDist = dist;
                    bestX = nx;
                    bestY = ny;
                }
            }

            if (bestX == currentX && bestY == currentY) break;
            currentX = bestX;
            currentY = bestY;
        }

        path.Add(new Point { X = 0, Y = 0 });
        return path;
    }

    static int CalculateCost(List<Point> path, int mass, int[,] groundHeight)
    {
        int cost = 0;
        for (int i = 0; i < path.Count - 1; i++)
        {
            int h1 = groundHeight[path[i].X, path[i].Y];
            int h2 = groundHeight[path[i + 1].X, path[i + 1].Y];

            if (h2 > h1)
            {
                cost += mass * 5; // Uphill
            }
            else
            {
                cost += mass * 1; // Flat or downhill
            }
        }
        return cost;
    }

    class OreVoxel
    {
        public int X, Y, Z;
        public int Mass, Value;
        public int Revenue;
    }

    class Operation
    {
        public int DigX, DigY, DigZ;
        public List<Point> Path;
        public int DestX, DestY;
    }

    class Point
    {
        public int X, Y;
    }
}
