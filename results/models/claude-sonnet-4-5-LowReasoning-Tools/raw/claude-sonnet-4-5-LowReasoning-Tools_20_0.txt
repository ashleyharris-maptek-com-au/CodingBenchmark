use std::io::{self, BufRead};
use std::collections::HashMap;

#[derive(Clone, Debug)]
struct Sample {
    hole_id: String,
    sample_idx: usize,
    position: [f64; 3],
    properties: Vec<f64>,
}

#[derive(Clone, Debug)]
struct Outlier {
    hole_id: String,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read header
    let first_line = lines.next().unwrap().unwrap();
    let header: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (num_holes, num_properties) = (header[0], header[1]);

    // Read property names
    let property_names: Vec<String> = lines.next().unwrap().unwrap()
        .split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // Read all samples
    let mut samples = Vec::new();
    let mut sample_count = 0;

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = hole_line.split_whitespace().collect();

        let hole_id = parts[0].to_string();
        let start_x: f64 = parts[1].parse().unwrap();
        let start_y: f64 = parts[2].parse().unwrap();
        let start_z: f64 = parts[3].parse().unwrap();
        let dir_x: f64 = parts[4].parse().unwrap();
        let dir_y: f64 = parts[5].parse().unwrap();
        let dir_z: f64 = parts[6].parse().unwrap();
        let _length: f64 = parts[7].parse().unwrap();
        let num_samples: usize = parts[8].parse().unwrap();

        // Normalize direction vector
        let dir_mag = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let dir = [dir_x / dir_mag, dir_y / dir_mag, dir_z / dir_mag];

        for idx in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let values: Vec<f64> = sample_line.split_whitespace()
                .map(|s| s.parse().unwrap_or(f64::NAN))
                .collect();

            let depth = values[0];
            let properties: Vec<f64> = values[1..].to_vec();

            // Calculate 3D position
            let position = [
                start_x + dir[0] * depth,
                start_y + dir[1] * depth,
                start_z + dir[2] * depth,
            ];

            samples.push(Sample {
                hole_id: hole_id.clone(),
                sample_idx: idx,
                position,
                properties,
            });
            sample_count += 1;
        }
    }

    // Determine strategy based on dataset size
    let use_full_search = sample_count < 50000;
    let max_neighbors = if sample_count < 1000 { 100 }
                       else if sample_count < 10000 { 50 }
                       else { 30 };
    let max_distance = estimate_max_distance(&samples, sample_count);

    // Find outliers
    let mut outliers = Vec::new();

    for prop_idx in 0..num_properties {
        // Collect all valid values for this property
        let mut all_values: Vec<f64> = samples.iter()
            .map(|s| s.properties[prop_idx])
            .filter(|v| v.is_finite())
            .collect();

        if all_values.len() < 3 {
            continue;
        }

        all_values.sort_by(|a, b| a.partial_cmp(b).unwrap());
        let global_median = median(&all_values);
        let global_mad = mad(&all_values, global_median);

        if global_mad < 1e-10 {
            continue; // Skip properties with no variation
        }

        // Check each sample
        for (i, sample) in samples.iter().enumerate() {
            let value = sample.properties[prop_idx];

            if !value.is_finite() {
                continue;
            }

            // Find neighbors
            let neighbors = find_neighbors(
                &samples, i, max_neighbors, max_distance, use_full_search
            );

            if neighbors.len() < 3 {
                continue;
            }

            // Get neighbor values for this property
            let neighbor_values: Vec<f64> = neighbors.iter()
                .map(|&idx| samples[idx].properties[prop_idx])
                .filter(|v| v.is_finite())
                .collect();

            if neighbor_values.len() < 3 {
                continue;
            }

            // Calculate local statistics
            let local_median = median(&neighbor_values);
            let local_mad = mad(&neighbor_values, local_median);

            if local_mad < 1e-10 {
                // Use global statistics if local has no variation
                let z_score = ((value - global_median) / global_mad).abs();
                if z_score > 3.5 {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence: z_score,
                    });
                }
            } else {
                // Modified Z-score using MAD
                let modified_z = 0.6745 * (value - local_median).abs() / local_mad;

                if modified_z > 3.5 {
                    outliers.push(Outlier {
                        hole_id: sample.hole_id.clone(),
                        sample_idx: sample.sample_idx,
                        property_name: property_names[prop_idx].clone(),
                        confidence: modified_z,
                    });
                }
            }
        }
    }

    // Sort by confidence descending
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Output results
    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.6}",
                 outlier.hole_id,
                 outlier.sample_idx,
                 outlier.property_name,
                 outlier.confidence);
    }
}

fn distance_squared(a: &[f64; 3], b: &[f64; 3]) -> f64 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

fn find_neighbors(
    samples: &[Sample],
    index: usize,
    max_neighbors: usize,
    max_distance: f64,
    use_full_search: bool,
) -> Vec<usize> {
    let pos = samples[index].position;
    let max_dist_sq = max_distance * max_distance;

    let mut distances: Vec<(usize, f64)> = Vec::new();

    if use_full_search {
        // Full search for smaller datasets
        for (i, sample) in samples.iter().enumerate() {
            if i == index {
                continue;
            }
            let dist_sq = distance_squared(&pos, &sample.position);
            if dist_sq <= max_dist_sq {
                distances.push((i, dist_sq));
            }
        }
    } else {
        // Sampled search for larger datasets
        let step = (samples.len() / 10000).max(1);
        for i in (0..samples.len()).step_by(step) {
            if i == index {
                continue;
            }
            let dist_sq = distance_squared(&pos, &samples[i].position);
            if dist_sq <= max_dist_sq {
                distances.push((i, dist_sq));
            }
        }

        // Also check immediate neighbors in the array
        let start = index.saturating_sub(100);
        let end = (index + 100).min(samples.len());
        for i in start..end {
            if i == index {
                continue;
            }
            let dist_sq = distance_squared(&pos, &samples[i].position);
            if dist_sq <= max_dist_sq {
                if !distances.iter().any(|(idx, _)| *idx == i) {
                    distances.push((i, dist_sq));
                }
            }
        }
    }

    // Sort by distance and take closest neighbors
    distances.sort_by(|a, b| a.1.partial_cmp(&b.1).unwrap());
    distances.truncate(max_neighbors);

    distances.into_iter().map(|(idx, _)| idx).collect()
}

fn estimate_max_distance(samples: &[Sample], sample_count: usize) -> f64 {
    if sample_count < 2 {
        return 1000.0;
    }

    // Sample a subset to estimate typical distances
    let sample_size = sample_count.min(1000);
    let step = sample_count / sample_size;

    let mut distances = Vec::new();
    for i in (0..sample_count).step_by(step.max(1)).take(sample_size) {
        if i + 1 < sample_count {
            let d = distance_squared(&samples[i].position, &samples[i + 1].position).sqrt();
            if d.is_finite() && d > 0.0 {
                distances.push(d);
            }
        }
    }

    if distances.is_empty() {
        return 1000.0;
    }

    distances.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let median_dist = distances[distances.len() / 2];

    // Use 50x median distance as search radius
    (median_dist * 50.0).max(10.0).min(10000.0)
}

fn median(values: &[f64]) -> f64 {
    if values.is_empty() {
        return 0.0;
    }
    let mut sorted = values.to_vec();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap());
    let mid = sorted.len() / 2;
    if sorted.len() % 2 == 0 {
        (sorted[mid - 1] + sorted[mid]) / 2.0
    } else {
        sorted[mid]
    }
}

fn mad(values: &[f64], median_val: f64) -> f64 {
    let mut deviations: Vec<f64> = values.iter()
        .map(|v| (v - median_val).abs())
        .collect();
    median(&deviations)
}
