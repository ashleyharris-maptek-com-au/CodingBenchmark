OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%in_color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
%out_color = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %void_func
%entry = OpLabel

; Load p0 = gl_in[0].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr

; Load p1 = gl_in[1].gl_Position
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load c0 = in_color[0]
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr

; Load c1 = in_color[1]
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Extract xy components
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1

; dir = p1.xy - p0.xy
%dir = OpFSub %v2float %p1_xy %p0_xy

; perp_unnorm = vec2(-dir.y, dir.x)
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v2float %neg_dir_y %dir_x

; normalize perp
%perp_len_sq = OpDot %float %perp_unnorm %perp_unnorm
%perp_len = OpExtInst %float %1 Sqrt %perp_len_sq
%perp_norm = OpFDiv %v2float %perp_unnorm %perp_len

; mid = p0 + 0.75 * (p1 - p0)
%p1_minus_p0 = OpFSub %v4float %p1 %p0
%offset = OpVectorTimesScalar %v4float %p1_minus_p0 %float_0_75
%mid = OpFAdd %v4float %p0 %offset
%mid_xy = OpVectorShuffle %v2float %mid %mid 0 1
%mid_zw = OpVectorShuffle %v2float %mid %mid 2 3

; perp_shaft = perp_norm * 0.015
%perp_shaft = OpVectorTimesScalar %v2float %perp_norm %float_0_015

; perp_arrow = perp_norm * 0.05
%perp_arrow = OpVectorTimesScalar %v2float %perp_norm %float_0_05

; Extract zw from p0 for shaft start vertices
%p0_zw = OpVectorShuffle %v2float %p0 %p0 2 3

; --- Emit shaft (4 vertices as triangle strip) ---
; Vertex 0: p0 - perp_shaft, color c0
%v0_xy = OpFSub %v2float %p0_xy %perp_shaft
%v0_pos = OpVectorShuffle %v4float %v0_xy %p0_zw 0 1 2 3
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %v0_pos
OpStore %out_color %c0
OpEmitVertex

; Vertex 1: p0 + perp_shaft, color c0
%v1_xy = OpFAdd %v2float %p0_xy %perp_shaft
%v1_pos = OpVectorShuffle %v4float %v1_xy %p0_zw 0 1 2 3
OpStore %out_pos_ptr %v1_pos
OpStore %out_color %c0
OpEmitVertex

; Vertex 2: mid - perp_shaft, color c0
%v2_xy = OpFSub %v2float %mid_xy %perp_shaft
%v2_pos = OpVectorShuffle %v4float %v2_xy %mid_zw 0 1 2 3
OpStore %out_pos_ptr %v2_pos
OpStore %out_color %c0
OpEmitVertex

; Vertex 3: mid + perp_shaft, color c0
%v3_xy = OpFAdd %v2float %mid_xy %perp_shaft
%v3_pos = OpVectorShuffle %v4float %v3_xy %mid_zw 0 1 2 3
OpStore %out_pos_ptr %v3_pos
OpStore %out_color %c0
OpEmitVertex

OpEndPrimitive

; --- Emit arrowhead (3 vertices as triangle strip) ---
; Vertex 4: mid + perp_arrow, color c1
%v4_xy = OpFAdd %v2float %mid_xy %perp_arrow
%v4_pos = OpVectorShuffle %v4float %v4_xy %mid_zw 0 1 2 3
OpStore %out_pos_ptr %v4_pos
OpStore %out_color %c1
OpEmitVertex

; Vertex 5: mid - perp_arrow, color c1
%v5_xy = OpFSub %v2float %mid_xy %perp_arrow
%v5_pos = OpVectorShuffle %v4float %v5_xy %mid_zw 0 1 2 3
OpStore %out_pos_ptr %v5_pos
OpStore %out_color %c1
OpEmitVertex

; Vertex 6: p1 (tip), color c1
OpStore %out_pos_ptr %p1
OpStore %out_color %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
