using System;
using System.Linq;

class QAP
{
    static int N;
    static int[][] flow;
    static int[][] dist;

    static void Main()
    {
        ReadInput();

        int[] solution;
        long bestCost = long.MaxValue;

        // Try multiple strategies for robustness
        // Strategy 1: Greedy construction + local search
        solution = GreedyConstruction();
        LocalSearch(solution);
        long cost1 = CalculateCost(solution);
        if (cost1 < bestCost)
        {
            bestCost = cost1;
        }

        // Strategy 2: Simple construction + intensive local search
        int[] solution2 = Enumerable.Range(0, N).ToArray();
        LocalSearch(solution2);
        long cost2 = CalculateCost(solution2);
        if (cost2 < bestCost)
        {
            bestCost = cost2;
            solution = solution2;
        }

        // Strategy 3: Reverse construction + local search
        int[] solution3 = Enumerable.Range(0, N).Reverse().ToArray();
        LocalSearch(solution3);
        long cost3 = CalculateCost(solution3);
        if (cost3 < bestCost)
        {
            bestCost = cost3;
            solution = solution3;
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", solution));
    }

    static void ReadInput()
    {
        N = int.Parse(Console.ReadLine());
        flow = new int[N][];
        dist = new int[N][];

        for (int i = 0; i < N; i++)
        {
            flow[i] = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries)
                .Select(int.Parse).ToArray();
        }

        for (int i = 0; i < N; i++)
        {
            dist[i] = Console.ReadLine().Split(' ', StringSplitOptions.RemoveEmptyEntries)
                .Select(int.Parse).ToArray();
        }
    }

    static long CalculateCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += (long)flow[i][j] * dist[p[i]][p[j]];
            }
        }
        return cost;
    }

    static long CalculateDelta(int[] p, int i, int j)
    {
        if (i == j) return 0;

        long delta = 0;
        int pi = p[i], pj = p[j];

        for (int k = 0; k < N; k++)
        {
            if (k == i || k == j) continue;
            int pk = p[k];

            // Remove old costs
            delta -= (long)flow[i][k] * dist[pi][pk];
            delta -= (long)flow[k][i] * dist[pk][pi];
            delta -= (long)flow[j][k] * dist[pj][pk];
            delta -= (long)flow[k][j] * dist[pk][pj];

            // Add new costs
            delta += (long)flow[i][k] * dist[pj][pk];
            delta += (long)flow[k][i] * dist[pk][pj];
            delta += (long)flow[j][k] * dist[pi][pk];
            delta += (long)flow[k][j] * dist[pk][pi];
        }

        // Adjust for i-j interaction
        delta -= (long)flow[i][j] * dist[pi][pj];
        delta -= (long)flow[j][i] * dist[pj][pi];
        delta += (long)flow[i][j] * dist[pj][pi];
        delta += (long)flow[j][i] * dist[pi][pj];

        return delta;
    }

    static void LocalSearch(int[] p)
    {
        bool improved = true;
        int maxIterations = N > 100 ? 100 : 1000;
        int iteration = 0;

        while (improved && iteration < maxIterations)
        {
            improved = false;
            iteration++;

            for (int i = 0; i < N - 1; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    long delta = CalculateDelta(p, i, j);

                    if (delta < 0)
                    {
                        // Swap
                        int temp = p[i];
                        p[i] = p[j];
                        p[j] = temp;
                        improved = true;
                    }
                }
            }
        }
    }

    static int[] GreedyConstruction()
    {
        // Calculate total flow and distance for each facility/location
        long[] totalFlow = new long[N];
        long[] totalDist = new long[N];

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                totalFlow[i] += flow[i][j];
                totalDist[i] += dist[i][j];
            }
        }

        // Sort facilities by decreasing total flow
        var facilities = Enumerable.Range(0, N)
            .OrderByDescending(i => totalFlow[i])
            .ToArray();

        // Sort locations by increasing total distance
        var locations = Enumerable.Range(0, N)
            .OrderBy(i => totalDist[i])
            .ToArray();

        // Assign high-flow facilities to low-distance locations
        int[] result = new int[N];
        for (int i = 0; i < N; i++)
        {
            result[facilities[i]] = locations[i];
        }

        return result;
    }
}
