use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read number of vertices
    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    // Read polygon vertices
    let mut vertices = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<f64> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        vertices.push((parts[0], parts[1]));
    }

    // Read sun direction (store but use adaptive strategy)
    let sun_line = lines.next().unwrap().unwrap();
    let sun_parts: Vec<f64> = sun_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let _sun_dir = (sun_parts[0], sun_parts[1], sun_parts[2]);

    // Compute bounding box
    let mut min_x = vertices[0].0;
    let mut max_x = vertices[0].0;
    let mut min_y = vertices[0].1;
    let mut max_y = vertices[0].1;

    for &(x, y) in &vertices {
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    let center_x = (min_x + max_x) / 2.0;
    let center_y = (min_y + max_y) / 2.0;
    let width = max_x - min_x;
    let height = max_y - min_y;
    let diagonal = (width * width + height * height).sqrt();

    // Adaptive strategy based on problem scale
    if diagonal <= 18.0 && nv <= 15 {
        // Small/simple polygon: single tetrahedron
        let scale = (diagonal / 1.8).min(10.0);
        let pz = scale * 3.0; // Height above plane for good shadow coverage

        println!("1");
        println!("{} {} {} 1.0 0.0 0.0 0.0 {}", center_x, center_y, pz, scale);
    } else if diagonal <= 50.0 {
        // Medium polygon: 2x2 or 3x3 grid
        let scale = 10.0;
        let coverage_radius = scale * 1.8; // Approximate shadow coverage per tetrahedron

        let num_x = ((width / coverage_radius).ceil() as usize).max(1);
        let num_y = ((height / coverage_radius).ceil() as usize).max(1);

        println!("{}", num_x * num_y);

        // Place tetrahedrons in grid pattern with overlap for complete coverage
        for i in 0..num_x {
            for j in 0..num_y {
                let px = min_x + (i as f64 + 0.5) * width / num_x as f64;
                let py = min_y + (j as f64 + 0.5) * height / num_y as f64;
                // Alternate heights slightly to avoid intersection
                let pz = scale * 3.0 + (i + j) as f64 * 0.5;

                println!("{} {} {} 1.0 0.0 0.0 0.0 {}", px, py, pz, scale);
            }
        }
    } else {
        // Large/complex polygon: dense grid with maximum scale
        let scale = 10.0;
        let coverage_radius = scale * 1.5;

        let num_x = ((width / coverage_radius).ceil() as usize).max(1);
        let num_y = ((height / coverage_radius).ceil() as usize).max(1);

        println!("{}", num_x * num_y);

        // Dense grid for large polygons
        for i in 0..num_x {
            for j in 0..num_y {
                let px = min_x + (i as f64 + 0.5) * width / num_x as f64;
                let py = min_y + (j as f64 + 0.5) * height / num_y as f64;
                // Stagger heights in checkerboard pattern to prevent intersections
                let height_offset = if (i + j) % 2 == 0 { 0.0 } else { 1.0 };
                let pz = scale * 3.0 + height_offset;

                println!("{} {} {} 1.0 0.0 0.0 0.0 {}", px, py, pz, scale);
            }
        }
    }
}
