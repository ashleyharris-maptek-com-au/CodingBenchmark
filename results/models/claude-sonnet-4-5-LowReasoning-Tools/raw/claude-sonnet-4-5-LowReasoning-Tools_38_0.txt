def solve_exact_cover():
    # Read input
    first_line = input().split()
    U = int(first_line[0])
    S = int(first_line[1])

    sets = []
    for i in range(S):
        line = input().strip()
        if line:
            elements = list(map(int, line.split()))
            sets.append(set(elements))
        else:
            sets.append(set())

    universe = set(range(U))

    # Backtracking to find exact cover
    def backtrack(idx, covered, solution):
        # If all elements are covered, we found a solution
        if covered == universe:
            return solution

        # If we've gone through all sets without covering all elements
        if idx >= S:
            return None

        # Try including current set
        current_set = sets[idx]
        # Check if there's no overlap with already covered elements
        if not (current_set & covered):
            new_covered = covered | current_set
            result = backtrack(idx + 1, new_covered, solution + [idx])
            if result is not None:
                return result

        # Try not including current set
        result = backtrack(idx + 1, covered, solution)
        if result is not None:
            return result

        return None

    solution = backtrack(0, set(), [])

    if solution is not None:
        print("SOLUTION")
        print(' '.join(map(str, solution)))
    else:
        print("NO SOLUTION")

solve_exact_cover()
