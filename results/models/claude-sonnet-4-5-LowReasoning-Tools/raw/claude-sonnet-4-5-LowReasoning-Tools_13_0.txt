#include <iostream>
#include <vector>
#include <string>
#include <unordered_set>
#include <algorithm>
#include <climits>

using namespace std;

typedef unsigned long long ull;

const ull BASE1 = 31;
const ull BASE2 = 37;
const ull MOD1 = 1e9 + 7;
const ull MOD2 = 1e9 + 9;

struct Hash {
    ull h1, h2;

    bool operator==(const Hash& other) const {
        return h1 == other.h1 && h2 == other.h2;
    }
};

struct HashHash {
    size_t operator()(const Hash& h) const {
        return h.h1 ^ (h.h2 << 1);
    }
};

ull computePower(ull base, ull mod, int len) {
    ull result = 1;
    for (int i = 0; i < len; i++) {
        result = (result * base) % mod;
    }
    return result;
}

string checkLength(const vector<string>& strings, int len) {
    if (len == 0) return "";

    int n = strings.size();
    if (n == 0) return "";

    const string& first = strings[0];
    if ((int)first.length() < len) return "";

    unordered_set<Hash, HashHash> hashes;
    ull pow1 = computePower(BASE1, MOD1, len - 1);
    ull pow2 = computePower(BASE2, MOD2, len - 1);

    ull h1 = 0, h2 = 0;

    for (int i = 0; i < len; i++) {
        h1 = (h1 * BASE1 + first[i]) % MOD1;
        h2 = (h2 * BASE2 + first[i]) % MOD2;
    }
    hashes.insert({h1, h2});

    for (size_t i = len; i < first.length(); i++) {
        h1 = (h1 - ((ull)first[i - len] * pow1) % MOD1 + MOD1) % MOD1;
        h1 = (h1 * BASE1 + first[i]) % MOD1;
        h2 = (h2 - ((ull)first[i - len] * pow2) % MOD2 + MOD2) % MOD2;
        h2 = (h2 * BASE2 + first[i]) % MOD2;
        hashes.insert({h1, h2});
    }

    for (int s = 1; s < n; s++) {
        unordered_set<Hash, HashHash> currentHashes;
        const string& str = strings[s];

        if ((int)str.length() < len) return "";

        h1 = 0;
        h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE1 + str[i]) % MOD1;
            h2 = (h2 * BASE2 + str[i]) % MOD2;
        }
        Hash currentHash = {h1, h2};
        if (hashes.count(currentHash)) {
            currentHashes.insert(currentHash);
        }

        for (size_t i = len; i < str.length(); i++) {
            h1 = (h1 - ((ull)str[i - len] * pow1) % MOD1 + MOD1) % MOD1;
            h1 = (h1 * BASE1 + str[i]) % MOD1;
            h2 = (h2 - ((ull)str[i - len] * pow2) % MOD2 + MOD2) % MOD2;
            h2 = (h2 * BASE2 + str[i]) % MOD2;
            currentHash = {h1, h2};
            if (hashes.count(currentHash)) {
                currentHashes.insert(currentHash);
            }
        }

        hashes = move(currentHashes);
        if (hashes.empty()) return "";
    }

    if (!hashes.empty()) {
        Hash targetHash = *hashes.begin();

        h1 = 0;
        h2 = 0;
        for (int i = 0; i < len; i++) {
            h1 = (h1 * BASE1 + first[i]) % MOD1;
            h2 = (h2 * BASE2 + first[i]) % MOD2;
        }
        if (h1 == targetHash.h1 && h2 == targetHash.h2) {
            return first.substr(0, len);
        }

        for (size_t i = len; i < first.length(); i++) {
            h1 = (h1 - ((ull)first[i - len] * pow1) % MOD1 + MOD1) % MOD1;
            h1 = (h1 * BASE1 + first[i]) % MOD1;
            h2 = (h2 - ((ull)first[i - len] * pow2) % MOD2 + MOD2) % MOD2;
            h2 = (h2 * BASE2 + first[i]) % MOD2;
            if (h1 == targetHash.h1 && h2 == targetHash.h2) {
                return first.substr(i - len + 1, len);
            }
        }
    }

    return "";
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    int minLen = INT_MAX;

    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
        if (!strings[i].empty()) {
            minLen = min(minLen, (int)strings[i].length());
        }
    }

    if (n == 0 || minLen == 0 || minLen == INT_MAX) {
        cout << endl;
        return 0;
    }

    // Sort strings by length (shortest first) for potential early termination
    sort(strings.begin(), strings.end(), [](const string& a, const string& b) {
        return a.length() < b.length();
    });

    int left = 0, right = minLen;
    string result = "";

    while (left <= right) {
        int mid = (left + right) / 2;
        string common = checkLength(strings, mid);

        if (!common.empty()) {
            result = common;
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    cout << result << endl;
    return 0;
}
