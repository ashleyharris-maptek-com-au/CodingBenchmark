using System;
using System.Collections.Generic;
using System.Linq;

public class MinesweeperSolver
{
    private char[,] board;
    private int height, width;
    private HashSet<(int, int)> safeCells;
    private HashSet<(int, int)> mines;

    static void Main()
    {
        var solver = new MinesweeperSolver();
        solver.Solve();
    }

    public void Solve()
    {
        ReadBoard();
        safeCells = new HashSet<(int, int)>();
        mines = new HashSet<(int, int)>();

        // Mark known mines
        for (int r = 0; r < height; r++)
        {
            for (int c = 0; c < width; c++)
            {
                if (board[r, c] == '*')
                    mines.Add((c, r));
            }
        }

        // Adaptive strategy based on board size
        int totalCells = height * width;

        if (totalCells <= 10000) // Small to medium boards
        {
            // Run multiple iterations of constraint propagation
            bool changed = true;
            int iterations = 0;
            while (changed && iterations < 100)
            {
                changed = ApplyBasicRules();
                changed |= ApplyAdvancedConstraints();
                iterations++;
            }
        }
        else // Large boards - optimize for speed
        {
            // Single pass with basic rules only
            ApplyBasicRules();
            ApplyAdvancedConstraints();
        }

        OutputResults();
    }

    private void ReadBoard()
    {
        var line = Console.ReadLine().Split();
        height = int.Parse(line[0]);
        width = int.Parse(line[1]);
        board = new char[height, width];

        for (int r = 0; r < height; r++)
        {
            string row = Console.ReadLine();
            for (int c = 0; c < width; c++)
            {
                board[r, c] = c < row.Length ? row[c] : ' ';
            }
        }
    }

    private bool ApplyBasicRules()
    {
        bool foundNew = false;

        for (int r = 0; r < height; r++)
        {
            for (int c = 0; c < width; c++)
            {
                char cell = board[r, c];
                if (cell >= '0' && cell <= '8')
                {
                    int count = cell - '0';
                    var neighbors = GetNeighbors(c, r);
                    var unknowns = neighbors.Where(n => board[n.Item2, n.Item1] == ' ' && !safeCells.Contains(n)).ToList();
                    var knownMines = neighbors.Where(n => board[n.Item2, n.Item1] == '*' || mines.Contains(n)).ToList();

                    // Rule 1: If mine count equals known mines, rest are safe
                    if (knownMines.Count == count)
                    {
                        foreach (var pos in unknowns)
                        {
                            if (safeCells.Add(pos))
                                foundNew = true;
                        }
                    }

                    // Rule 2: If mine count equals known mines + unknowns, all unknowns are mines
                    if (knownMines.Count + unknowns.Count == count)
                    {
                        foreach (var pos in unknowns)
                        {
                            if (mines.Add(pos))
                                foundNew = true;
                        }
                    }
                }
            }
        }

        return foundNew;
    }

    private bool ApplyAdvancedConstraints()
    {
        bool foundNew = false;
        var constraints = new List<Constraint>();

        // Build constraints from numbered cells
        for (int r = 0; r < height; r++)
        {
            for (int c = 0; c < width; c++)
            {
                char cell = board[r, c];
                if (cell >= '0' && cell <= '8')
                {
                    int count = cell - '0';
                    var neighbors = GetNeighbors(c, r);
                    var unknowns = neighbors.Where(n => board[n.Item2, n.Item1] == ' ' && !safeCells.Contains(n) && !mines.Contains(n)).ToList();
                    var knownMines = neighbors.Where(n => board[n.Item2, n.Item1] == '*' || mines.Contains(n)).Count();

                    if (unknowns.Count > 0)
                    {
                        constraints.Add(new Constraint
                        {
                            Cells = new HashSet<(int, int)>(unknowns),
                            MineCount = count - knownMines
                        });
                    }
                }
            }
        }

        // Try subset elimination
        for (int i = 0; i < constraints.Count; i++)
        {
            for (int j = 0; j < constraints.Count; j++)
            {
                if (i == j) continue;

                var c1 = constraints[i];
                var c2 = constraints[j];

                // If c1 is subset of c2
                if (c1.Cells.IsSubsetOf(c2.Cells))
                {
                    var difference = new HashSet<(int, int)>(c2.Cells);
                    difference.ExceptWith(c1.Cells);
                    int diffMines = c2.MineCount - c1.MineCount;

                    // All cells in difference are safe
                    if (diffMines == 0)
                    {
                        foreach (var pos in difference)
                        {
                            if (safeCells.Add(pos))
                                foundNew = true;
                        }
                    }
                    // All cells in difference are mines
                    else if (diffMines == difference.Count)
                    {
                        foreach (var pos in difference)
                        {
                            if (mines.Add(pos))
                                foundNew = true;
                        }
                    }
                }
            }
        }

        return foundNew;
    }

    private List<(int, int)> GetNeighbors(int x, int y)
    {
        var neighbors = new List<(int, int)>();
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                int ny = y + dy;
                if (nx >= 0 && nx < width && ny >= 0 && ny < height)
                {
                    neighbors.Add((nx, ny));
                }
            }
        }
        return neighbors;
    }

    private void OutputResults()
    {
        var result = safeCells.Where(pos => board[pos.Item2, pos.Item1] == ' ').OrderBy(p => p.Item2).ThenBy(p => p.Item1).ToList();
        Console.WriteLine(result.Count);
        foreach (var (x, y) in result)
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    private class Constraint
    {
        public HashSet<(int, int)> Cells { get; set; }
        public int MineCount { get; set; }
    }
}
