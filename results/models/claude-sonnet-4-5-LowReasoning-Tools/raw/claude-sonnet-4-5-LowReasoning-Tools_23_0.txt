use std::io::{self, BufRead, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy)]
struct MapInfo {
    width: f64,
    height: f64,
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse initial map data
    let first_line = lines.next().unwrap().unwrap();
    let params: Vec<f64> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let map_info = MapInfo {
        width: params[0],
        height: params[1],
        gravity: params[2],
        max_thrust: params[3],
        initial_fuel: params[4],
        start_x: 0.0,
        start_y: 0.0,
        target_x: 0.0,
        target_y: 0.0,
    };

    let second_line = lines.next().unwrap().unwrap();
    let positions: Vec<f64> = second_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();

    let mut map_info = map_info;
    map_info.start_x = positions[0];
    map_info.start_y = positions[1];
    map_info.target_x = positions[2];
    map_info.target_y = positions[3];

    // Skip map rows (read until we get numeric data)
    loop {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() == 7 {
            // This is state data, parse it
            let state = State {
                x: parts[0].parse().unwrap(),
                y: parts[1].parse().unwrap(),
                vx: parts[2].parse().unwrap(),
                vy: parts[3].parse().unwrap(),
                angle: parts[4].parse().unwrap(),
                angular_vel: parts[5].parse().unwrap(),
                fuel: parts[6].parse().unwrap(),
            };

            // Start the control system with this first state
            run_control_system(state, map_info, lines);
            break;
        }
    }
}

fn run_control_system(initial_state: State, map_info: MapInfo, mut lines: io::Lines<io::StdinLock>) {
    let current_state = Arc::new(Mutex::new(initial_state));
    let state_for_reader = Arc::clone(&current_state);

    // Spawn thread to read state updates
    thread::spawn(move || {
        for line in lines {
            if let Ok(line) = line {
                let parts: Vec<&str> = line.split_whitespace().collect();
                if parts.len() == 7 {
                    let state = State {
                        x: parts[0].parse().unwrap_or(0.0),
                        y: parts[1].parse().unwrap_or(0.0),
                        vx: parts[2].parse().unwrap_or(0.0),
                        vy: parts[3].parse().unwrap_or(0.0),
                        angle: parts[4].parse().unwrap_or(0.0),
                        angular_vel: parts[5].parse().unwrap_or(0.0),
                        fuel: parts[6].parse().unwrap_or(0.0),
                    };

                    if let Ok(mut s) = state_for_reader.lock() {
                        *s = state;
                    }
                }
            }
        }
    });

    // Main control loop - output at 50Hz (20ms intervals)
    let mut stdout = io::stdout();
    let interval = Duration::from_millis(20);

    loop {
        let start = Instant::now();

        let state = {
            current_state.lock().unwrap().clone()
        };

        // Calculate control commands
        let (thrust, turn) = calculate_control(&state, &map_info);

        // Output commands
        writeln!(stdout, "{:.6} {:.6}", thrust, turn).unwrap();
        stdout.flush().unwrap();

        // Sleep to maintain 50Hz rate
        let elapsed = start.elapsed();
        if elapsed < interval {
            thread::sleep(interval - elapsed);
        }
    }
}

fn calculate_control(state: &State, map_info: &MapInfo) -> (f64, f64) {
    // Calculate distance to target
    let dx = map_info.target_x - state.x;
    let dy = map_info.target_y - state.y;
    let dist = (dx * dx + dy * dy).sqrt();

    // Desired angle to face target (0 = up, positive = clockwise)
    let target_angle = if dist > 5.0 {
        // Point toward target when far away
        dy.atan2(dx) - std::f64::consts::PI / 2.0
    } else {
        // Point upward when close (for landing)
        0.0
    };

    // Normalize angles to [-PI, PI]
    let mut angle_error = target_angle - state.angle;
    while angle_error > std::f64::consts::PI {
        angle_error -= 2.0 * std::f64::consts::PI;
    }
    while angle_error < -std::f64::consts::PI {
        angle_error += 2.0 * std::f64::consts::PI;
    }

    // PD control for rotation
    let kp_angle = 3.0;
    let kd_angle = 2.0;
    let turn = (kp_angle * angle_error - kd_angle * state.angular_vel).clamp(-1.0, 1.0);

    // Thrust control
    let speed = (state.vx * state.vx + state.vy * state.vy).sqrt();

    // Calculate if we're in landing phase
    let landing_phase = dist < 10.0 && dy > -5.0;

    let thrust = if state.fuel < 0.1 {
        // Out of fuel
        0.0
    } else if landing_phase {
        // Landing: gentle descent
        let desired_vy = 0.5; // Slow descent
        let vy_error = state.vy - desired_vy;

        // Compensate for gravity and adjust velocity
        let base_thrust = map_info.gravity / map_info.max_thrust;
        let correction = -vy_error * 0.3;

        (base_thrust + correction).clamp(0.0, 1.0)
    } else {
        // Navigation phase
        // If falling too fast or moving away from target
        if state.vy > 3.0 || (dy < 0.0 && state.vy > 0.5) {
            // Slow down vertical speed
            let thrust_needed = (map_info.gravity + 2.0) / map_info.max_thrust;
            thrust_needed.clamp(0.0, 1.0)
        } else if dist > 20.0 && state.angle.abs() < 0.5 {
            // Accelerate toward target when pointing roughly right direction
            0.8
        } else if speed > 5.0 {
            // Slow down if moving too fast
            0.6
        } else {
            // Maintain altitude and controlled descent
            let thrust_needed = map_info.gravity / map_info.max_thrust;
            (thrust_needed * 0.9).clamp(0.0, 1.0)
        }
    };

    (thrust, turn)
}
