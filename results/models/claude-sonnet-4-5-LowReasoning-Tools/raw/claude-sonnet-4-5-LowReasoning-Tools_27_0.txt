#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <queue>
#include <random>
#include <chrono>
#include <cstring>

using namespace std;

class GraphColoring {
private:
    int n, m, k;
    vector<vector<int>> adj;
    vector<int> color;
    vector<set<int>> neighbor_colors;

    // DSATUR heuristic: choose vertex with highest saturation degree
    // (most colors used by neighbors), ties broken by highest degree
    int selectVertexDSATUR(const vector<bool>& colored) {
        int best = -1;
        int maxSaturation = -1;
        int maxDegree = -1;

        for (int i = 0; i < n; i++) {
            if (colored[i]) continue;

            int saturation = neighbor_colors[i].size();
            int degree = adj[i].size();

            if (saturation > maxSaturation ||
                (saturation == maxSaturation && degree > maxDegree)) {
                best = i;
                maxSaturation = saturation;
                maxDegree = degree;
            }
        }

        return best;
    }

    // Find smallest available color for vertex v
    int findSmallestColor(int v) {
        for (int c = 0; c < k; c++) {
            if (neighbor_colors[v].find(c) == neighbor_colors[v].end()) {
                return c;
            }
        }
        return -1; // No valid color found
    }

    // DSATUR algorithm
    bool colorDSATUR() {
        color.assign(n, -1);
        neighbor_colors.assign(n, set<int>());
        vector<bool> colored(n, false);

        // Find vertex with maximum degree to start
        int maxDeg = -1, startVertex = 0;
        for (int i = 0; i < n; i++) {
            if ((int)adj[i].size() > maxDeg) {
                maxDeg = adj[i].size();
                startVertex = i;
            }
        }

        // Color the starting vertex
        color[startVertex] = 0;
        colored[startVertex] = true;
        for (int u : adj[startVertex]) {
            neighbor_colors[u].insert(0);
        }

        // Color remaining vertices
        for (int count = 1; count < n; count++) {
            int v = selectVertexDSATUR(colored);
            if (v == -1) break;

            int c = findSmallestColor(v);
            if (c == -1) {
                return false; // Failed to find valid coloring
            }

            color[v] = c;
            colored[v] = true;

            // Update neighbor colors
            for (int u : adj[v]) {
                if (!colored[u]) {
                    neighbor_colors[u].insert(c);
                }
            }
        }

        return true;
    }

    // Simple greedy coloring as fallback
    bool colorGreedy() {
        color.assign(n, -1);

        // Create ordering by degree (descending)
        vector<pair<int, int>> vertices;
        for (int i = 0; i < n; i++) {
            vertices.push_back({adj[i].size(), i});
        }
        sort(vertices.rbegin(), vertices.rend());

        for (auto [deg, v] : vertices) {
            vector<bool> used(k, false);

            for (int u : adj[v]) {
                if (color[u] != -1) {
                    used[color[u]] = true;
                }
            }

            for (int c = 0; c < k; c++) {
                if (!used[c]) {
                    color[v] = c;
                    break;
                }
            }

            if (color[v] == -1) {
                return false;
            }
        }

        return true;
    }

    // Verify the coloring is valid
    bool verifyColoring() {
        for (int v = 0; v < n; v++) {
            if (color[v] < 0 || color[v] >= k) return false;

            for (int u : adj[v]) {
                if (color[v] == color[u]) return false;
            }
        }
        return true;
    }

public:
    GraphColoring(int vertices, int edges, int colors)
        : n(vertices), m(edges), k(colors), adj(vertices), color(vertices, -1) {}

    void addEdge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    bool solve() {
        // Try DSATUR first (generally best quality)
        if (colorDSATUR() && verifyColoring()) {
            return true;
        }

        // Fallback to greedy if DSATUR fails
        if (colorGreedy() && verifyColoring()) {
            return true;
        }

        return false;
    }

    void printSolution() {
        for (int i = 0; i < n; i++) {
            if (i > 0) cout << " ";
            cout << color[i];
        }
        cout << "\n";
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    GraphColoring gc(n, m, k);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        gc.addEdge(u, v);
    }

    if (gc.solve()) {
        gc.printSolution();
    } else {
        // If no solution found, output something (shouldn't happen with valid input)
        for (int i = 0; i < n; i++) {
            if (i > 0) cout << " ";
            cout << 0;
        }
        cout << "\n";
    }

    return 0;
}
