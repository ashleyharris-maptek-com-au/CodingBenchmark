; SPIR-V Geometry Shader - Point to Hexagon Expansion
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %out_color
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
     %uint_1 = OpConstant %uint 1
     %uint_6 = OpConstant %uint 6

; Constants for hexagon
 %float_0_04 = OpConstant %float 0.04
  %float_0_9 = OpConstant %float 0.9
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1

; Hexagon angle constants (in radians)
  %angle_0 = OpConstant %float 0.0
 %angle_60 = OpConstant %float 1.047197551
%angle_120 = OpConstant %float 2.094395102
%angle_180 = OpConstant %float 3.141592654
%angle_240 = OpConstant %float 4.188790205
%angle_300 = OpConstant %float 5.235987756

; Input types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr Input
%ptr_in_v4 = OpTypePointer Input %v4float

%arr_color_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_col = OpTypePointer Input %arr_color_in
   %in_color = OpVariable %ptr_in_arr_col Input
%ptr_in_v3 = OpTypePointer Input %v3float

; Output types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_struct = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_struct Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
  %out_color = OpVariable %ptr_out_v3 Output

; Main function
       %main = OpFunction %void None %3
      %label = OpLabel

; Load input position
 %pos_ptr_in = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %in_pos4 = OpLoad %v4float %pos_ptr_in

; Load input color
%color_ptr_in = OpAccessChain %ptr_in_v3 %in_color %int_0
%in_color_v3 = OpLoad %v3float %color_ptr_in

; Extract center position (x, y from vec4)
  %center_x = OpCompositeExtract %float %in_pos4 0
  %center_y = OpCompositeExtract %float %in_pos4 1
  %center_z = OpCompositeExtract %float %in_pos4 2
  %center_w = OpCompositeExtract %float %in_pos4 3

; Darken color for edges
   %color_r = OpCompositeExtract %float %in_color_v3 0
   %color_g = OpCompositeExtract %float %in_color_v3 1
   %color_b = OpCompositeExtract %float %in_color_v3 2
    %edge_r = OpFMul %float %color_r %float_0_9
    %edge_g = OpFMul %float %color_g %float_0_9
    %edge_b = OpFMul %float %color_b %float_0_9
%edge_color = OpCompositeConstruct %v3float %edge_r %edge_g %edge_b

; Compute edge positions (6 vertices)
; Edge 0 (0°)
   %cos_0 = OpExtInst %float %1 Cos %angle_0
   %sin_0 = OpExtInst %float %1 Sin %angle_0
 %off_x_0 = OpFMul %float %cos_0 %float_0_04
 %off_y_0 = OpFMul %float %sin_0 %float_0_04
   %e0_x = OpFAdd %float %center_x %off_x_0
   %e0_y = OpFAdd %float %center_y %off_y_0
 %edge_0 = OpCompositeConstruct %v4float %e0_x %e0_y %center_z %center_w

; Edge 1 (60°)
  %cos_60 = OpExtInst %float %1 Cos %angle_60
  %sin_60 = OpExtInst %float %1 Sin %angle_60
 %off_x_1 = OpFMul %float %cos_60 %float_0_04
 %off_y_1 = OpFMul %float %sin_60 %float_0_04
   %e1_x = OpFAdd %float %center_x %off_x_1
   %e1_y = OpFAdd %float %center_y %off_y_1
 %edge_1 = OpCompositeConstruct %v4float %e1_x %e1_y %center_z %center_w

; Edge 2 (120°)
 %cos_120 = OpExtInst %float %1 Cos %angle_120
 %sin_120 = OpExtInst %float %1 Sin %angle_120
 %off_x_2 = OpFMul %float %cos_120 %float_0_04
 %off_y_2 = OpFMul %float %sin_120 %float_0_04
   %e2_x = OpFAdd %float %center_x %off_x_2
   %e2_y = OpFAdd %float %center_y %off_y_2
 %edge_2 = OpCompositeConstruct %v4float %e2_x %e2_y %center_z %center_w

; Edge 3 (180°)
 %cos_180 = OpExtInst %float %1 Cos %angle_180
 %sin_180 = OpExtInst %float %1 Sin %angle_180
 %off_x_3 = OpFMul %float %cos_180 %float_0_04
 %off_y_3 = OpFMul %float %sin_180 %float_0_04
   %e3_x = OpFAdd %float %center_x %off_x_3
   %e3_y = OpFAdd %float %center_y %off_y_3
 %edge_3 = OpCompositeConstruct %v4float %e3_x %e3_y %center_z %center_w

; Edge 4 (240°)
 %cos_240 = OpExtInst %float %1 Cos %angle_240
 %sin_240 = OpExtInst %float %1 Sin %angle_240
 %off_x_4 = OpFMul %float %cos_240 %float_0_04
 %off_y_4 = OpFMul %float %sin_240 %float_0_04
   %e4_x = OpFAdd %float %center_x %off_x_4
   %e4_y = OpFAdd %float %center_y %off_y_4
 %edge_4 = OpCompositeConstruct %v4float %e4_x %e4_y %center_z %center_w

; Edge 5 (300°)
 %cos_300 = OpExtInst %float %1 Cos %angle_300
 %sin_300 = OpExtInst %float %1 Sin %angle_300
 %off_x_5 = OpFMul %float %cos_300 %float_0_04
 %off_y_5 = OpFMul %float %sin_300 %float_0_04
   %e5_x = OpFAdd %float %center_x %off_x_5
   %e5_y = OpFAdd %float %center_y %off_y_5
 %edge_5 = OpCompositeConstruct %v4float %e5_x %e5_y %center_z %center_w

; Emit Triangle 1: center, edge0, edge1
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos_ptr %in_pos4
               OpStore %out_color %in_color_v3
               OpEmitVertex
               OpStore %out_pos_ptr %edge_0
               OpStore %out_color %edge_color
               OpEmitVertex
               OpStore %out_pos_ptr %edge_1
               OpStore %out_color %edge_color
               OpEmitVertex
               OpEndPrimitive

; Emit Triangle 2: center, edge1, edge2
               OpStore %out_pos_ptr %in_pos4
               OpStore %out_color %in_color_v3
               OpEmitVertex
               OpStore %out_pos_ptr %edge_1
               OpStore %out_color %edge_color
               OpEmitVertex
               OpStore %out_pos_ptr %edge_2
               OpStore %out_color %edge_color
               OpEmitVertex
               OpEndPrimitive

; Emit Triangle 3: center, edge2, edge3
               OpStore %out_pos_ptr %in_pos4
               OpStore %out_color %in_color_v3
               OpEmitVertex
               OpStore %out_pos_ptr %edge_2
               OpStore %out_color %edge_color
               OpEmitVertex
               OpStore %out_pos_ptr %edge_3
               OpStore %out_color %edge_color
               OpEmitVertex
               OpEndPrimitive

; Emit Triangle 4: center, edge3, edge4
               OpStore %out_pos_ptr %in_pos4
               OpStore %out_color %in_color_v3
               OpEmitVertex
               OpStore %out_pos_ptr %edge_3
               OpStore %out_color %edge_color
               OpEmitVertex
               OpStore %out_pos_ptr %edge_4
               OpStore %out_color %edge_color
               OpEmitVertex
               OpEndPrimitive

; Emit Triangle 5: center, edge4, edge5
               OpStore %out_pos_ptr %in_pos4
               OpStore %out_color %in_color_v3
               OpEmitVertex
               OpStore %out_pos_ptr %edge_4
               OpStore %out_color %edge_color
               OpEmitVertex
               OpStore %out_pos_ptr %edge_5
               OpStore %out_color %edge_color
               OpEmitVertex
               OpEndPrimitive

; Emit Triangle 6: center, edge5, edge0
               OpStore %out_pos_ptr %in_pos4
               OpStore %out_color %in_color_v3
               OpEmitVertex
               OpStore %out_pos_ptr %edge_5
               OpStore %out_color %edge_color
               OpEmitVertex
               OpStore %out_pos_ptr %edge_0
               OpStore %out_color %edge_color
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd
