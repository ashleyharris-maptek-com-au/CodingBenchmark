#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <limits>

using namespace std;

struct Rect {
    int w, h, idx;
};

struct Placement {
    int board, x, y, rotated;
};

struct FreeRect {
    int x, y, w, h;
};

class Board {
public:
    int width, height;
    vector<FreeRect> freeRects;

    Board(int w, int h) : width(w), height(h) {
        freeRects.push_back({0, 0, w, h});
    }

    bool tryPlace(int rw, int rh, int& outX, int& outY) {
        int bestIdx = -1;
        int bestShortSideFit = numeric_limits<int>::max();
        int bestLongSideFit = numeric_limits<int>::max();

        for (int i = 0; i < (int)freeRects.size(); i++) {
            const auto& fr = freeRects[i];
            if (fr.w >= rw && fr.h >= rh) {
                int leftoverHoriz = fr.w - rw;
                int leftoverVert = fr.h - rh;
                int shortSideFit = min(leftoverHoriz, leftoverVert);
                int longSideFit = max(leftoverHoriz, leftoverVert);

                if (shortSideFit < bestShortSideFit ||
                    (shortSideFit == bestShortSideFit && longSideFit < bestLongSideFit)) {
                    bestIdx = i;
                    bestShortSideFit = shortSideFit;
                    bestLongSideFit = longSideFit;
                    outX = fr.x;
                    outY = fr.y;
                }
            }
        }

        if (bestIdx == -1) return false;

        splitFreeRect(bestIdx, outX, outY, rw, rh);
        return true;
    }

private:
    void splitFreeRect(int idx, int x, int y, int w, int h) {
        FreeRect used = freeRects[idx];
        freeRects.erase(freeRects.begin() + idx);

        // Guillotine split
        if (used.w > w) {
            freeRects.push_back({x + w, y, used.w - w, h});
        }
        if (used.h > h) {
            freeRects.push_back({x, y + h, used.w, used.h - h});
        }

        pruneFreeRects();
    }

    void pruneFreeRects() {
        for (int i = 0; i < (int)freeRects.size(); i++) {
            for (int j = i + 1; j < (int)freeRects.size(); ) {
                if (isContainedIn(freeRects[i], freeRects[j])) {
                    freeRects.erase(freeRects.begin() + i);
                    i--;
                    break;
                } else if (isContainedIn(freeRects[j], freeRects[i])) {
                    freeRects.erase(freeRects.begin() + j);
                } else {
                    j++;
                }
            }
        }
    }

    bool isContainedIn(const FreeRect& a, const FreeRect& b) {
        return a.x >= b.x && a.y >= b.y &&
               a.x + a.w <= b.x + b.w && a.y + a.h <= b.y + b.h;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, boardW, boardH;
    cin >> n >> boardW >> boardH;

    vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    // Sort by area (largest first) for better packing
    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        int64_t areaA = (int64_t)a.w * a.h;
        int64_t areaB = (int64_t)b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        return max(a.w, a.h) > max(b.w, b.h);
    });

    vector<Board> boards;
    vector<Placement> placements(n);

    for (const auto& rect : rects) {
        bool placed = false;

        // Try existing boards
        for (int b = 0; b < (int)boards.size(); b++) {
            int x, y;

            // Try without rotation
            if (rect.w <= boardW && rect.h <= boardH) {
                Board tempBoard = boards[b];
                if (tempBoard.tryPlace(rect.w, rect.h, x, y)) {
                    boards[b] = tempBoard;
                    placements[rect.idx] = {b, x, y, 0};
                    placed = true;
                    break;
                }
            }

            // Try with rotation
            if (!placed && rect.h <= boardW && rect.w <= boardH) {
                Board tempBoard = boards[b];
                if (tempBoard.tryPlace(rect.h, rect.w, x, y)) {
                    boards[b] = tempBoard;
                    placements[rect.idx] = {b, x, y, 1};
                    placed = true;
                    break;
                }
            }
        }

        // Create new board if needed
        if (!placed) {
            Board newBoard(boardW, boardH);
            int x, y;

            if (rect.w <= boardW && rect.h <= boardH && newBoard.tryPlace(rect.w, rect.h, x, y)) {
                placements[rect.idx] = {(int)boards.size(), x, y, 0};
                boards.push_back(newBoard);
            } else if (rect.h <= boardW && rect.w <= boardH) {
                newBoard = Board(boardW, boardH);
                if (newBoard.tryPlace(rect.h, rect.w, x, y)) {
                    placements[rect.idx] = {(int)boards.size(), x, y, 1};
                    boards.push_back(newBoard);
                }
            }
        }
    }

    cout << boards.size() << "\n";
    for (const auto& p : placements) {
        cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }

    return 0;
}
