using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        var input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        List<int>[] graph = new List<int>[n];
        for (int i = 0; i < n; i++)
            graph[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            graph[u].Add(v);
        }

        var fvs = FindMinimumFVS(graph, n);

        Console.WriteLine(fvs.Count);
        if (fvs.Count > 0)
            Console.WriteLine(string.Join(" ", fvs.OrderBy(x => x)));
        else
            Console.WriteLine();
    }

    static HashSet<int> FindMinimumFVS(List<int>[] graph, int n)
    {
        var sccs = FindSCCs(graph, n);
        HashSet<int> result = new HashSet<int>();

        foreach (var scc in sccs)
        {
            // Check if this SCC has cycles
            bool hasCycle = false;
            if (scc.Count > 1)
            {
                hasCycle = true;
            }
            else if (scc.Count == 1)
            {
                // Check for self-loop
                if (graph[scc[0]].Contains(scc[0]))
                    hasCycle = true;
            }

            if (!hasCycle)
                continue;

            var sccFvs = FindSCCMinimumFVS(graph, scc);
            foreach (var v in sccFvs)
                result.Add(v);
        }

        return result;
    }

    static HashSet<int> FindSCCMinimumFVS(List<int>[] graph, List<int> scc)
    {
        // For small SCCs, use exact algorithm
        if (scc.Count <= 20)
        {
            return ExactFVS(graph, scc);
        }
        else
        {
            return GreedyFVS(graph, scc);
        }
    }

    static HashSet<int> ExactFVS(List<int>[] graph, List<int> scc)
    {
        var sccSet = new HashSet<int>(scc);

        // Try increasing sizes of FVS
        for (int size = 0; size <= scc.Count; size++)
        {
            var fvs = new HashSet<int>();
            if (TryFindFVS(graph, scc, sccSet, 0, size, fvs))
                return fvs;
        }

        return new HashSet<int>(scc);
    }

    static bool TryFindFVS(List<int>[] graph, List<int> scc, HashSet<int> sccSet,
                           int start, int remaining, HashSet<int> current)
    {
        if (remaining == 0)
        {
            return IsAcyclic(graph, sccSet, current);
        }

        for (int i = start; i < scc.Count; i++)
        {
            current.Add(scc[i]);
            if (TryFindFVS(graph, scc, sccSet, i + 1, remaining - 1, current))
                return true;
            current.Remove(scc[i]);
        }

        return false;
    }

    static bool IsAcyclic(List<int>[] graph, HashSet<int> vertices, HashSet<int> removed)
    {
        var color = new int[graph.Length];

        foreach (var v in vertices)
        {
            if (removed.Contains(v))
                continue;

            if (color[v] == 0)
            {
                if (HasCycleDFS(v, graph, vertices, removed, color))
                    return false;
            }
        }

        return true;
    }

    static bool HasCycleDFS(int u, List<int>[] graph, HashSet<int> vertices,
                            HashSet<int> removed, int[] color)
    {
        color[u] = 1;

        foreach (int v in graph[u])
        {
            if (!vertices.Contains(v) || removed.Contains(v))
                continue;

            if (color[v] == 1)
                return true;

            if (color[v] == 0)
            {
                if (HasCycleDFS(v, graph, vertices, removed, color))
                    return true;
            }
        }

        color[u] = 2;
        return false;
    }

    static HashSet<int> GreedyFVS(List<int>[] graph, List<int> scc)
    {
        var removed = new HashSet<int>();
        var sccSet = new HashSet<int>(scc);

        while (!IsAcyclic(graph, sccSet, removed))
        {
            int bestVertex = -1;
            int bestScore = -1;

            foreach (var v in scc)
            {
                if (removed.Contains(v))
                    continue;

                int score = 0;
                foreach (var u in graph[v])
                {
                    if (sccSet.Contains(u) && !removed.Contains(u))
                        score++;
                }

                foreach (var u in scc)
                {
                    if (removed.Contains(u))
                        continue;
                    if (graph[u].Contains(v))
                        score++;
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    bestVertex = v;
                }
            }

            if (bestVertex == -1)
                break;

            removed.Add(bestVertex);
        }

        return removed;
    }

    static List<List<int>> FindSCCs(List<int>[] graph, int n)
    {
        var sccs = new List<List<int>>();
        var visited = new bool[n];
        var stack = new Stack<int>();

        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
                FillOrder(i, graph, visited, stack);
        }

        var reverseGraph = new List<int>[n];
        for (int i = 0; i < n; i++)
            reverseGraph[i] = new List<int>();

        for (int u = 0; u < n; u++)
        {
            foreach (int v in graph[u])
                reverseGraph[v].Add(u);
        }

        visited = new bool[n];
        while (stack.Count > 0)
        {
            int v = stack.Pop();
            if (!visited[v])
            {
                var scc = new List<int>();
                CollectSCC(v, reverseGraph, visited, scc);
                sccs.Add(scc);
            }
        }

        return sccs;
    }

    static void FillOrder(int u, List<int>[] graph, bool[] visited, Stack<int> stack)
    {
        visited[u] = true;
        foreach (int v in graph[u])
        {
            if (!visited[v])
                FillOrder(v, graph, visited, stack);
        }
        stack.Push(u);
    }

    static void CollectSCC(int u, List<int>[] graph, bool[] visited, List<int> scc)
    {
        visited[u] = true;
        scc.Add(u);
        foreach (int v in graph[u])
        {
            if (!visited[v])
                CollectSCC(v, graph, visited, scc);
        }
    }
}
