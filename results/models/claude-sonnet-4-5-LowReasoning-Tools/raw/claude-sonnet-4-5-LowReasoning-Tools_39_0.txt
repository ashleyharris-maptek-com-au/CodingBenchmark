use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let (u, v) = (edge[0], edge[1]);
        adj[u].push(v);
        adj[v].push(u);
    }

    let half = n / 2;
    let (min_cut, partition) = if n <= 24 {
        brute_force_bisection(n, &adj, half)
    } else {
        heuristic_bisection(n, &adj, half)
    };

    println!("{}", min_cut);
    for (i, &v) in partition.iter().enumerate() {
        if i > 0 { print!(" "); }
        print!("{}", v);
    }
    println!();
}

fn brute_force_bisection(n: usize, adj: &[Vec<usize>], half: usize) -> (usize, Vec<usize>) {
    let mut min_cut = usize::MAX;
    let mut best_partition = Vec::new();
    let mut current = (0..half).collect::<Vec<_>>();

    loop {
        let cut = calculate_cut(&current, adj);
        if cut < min_cut {
            min_cut = cut;
            best_partition = current.clone();
        }
        if !next_combination(&mut current, n) { break; }
    }

    (min_cut, best_partition)
}

fn next_combination(current: &mut [usize], n: usize) -> bool {
    let k = current.len();
    for i in (0..k).rev() {
        if current[i] < n - k + i {
            current[i] += 1;
            for j in i + 1..k {
                current[j] = current[j - 1] + 1;
            }
            return true;
        }
    }
    false
}

fn calculate_cut(partition_a: &[usize], adj: &[Vec<usize>]) -> usize {
    let set_a: HashSet<usize> = partition_a.iter().cloned().collect();
    partition_a.iter()
        .flat_map(|&u| &adj[u])
        .filter(|v| !set_a.contains(v))
        .count()
}

fn heuristic_bisection(n: usize, adj: &[Vec<usize>], half: usize) -> (usize, Vec<usize>) {
    let mut best_cut = usize::MAX;
    let mut best_partition = Vec::new();

    // Try greedy degree-based approach
    let mut vertices: Vec<(usize, usize)> = (0..n).map(|i| (i, adj[i].len())).collect();
    vertices.sort_by_key(|&(_, deg)| deg);

    for start in 0..3 {
        let mut partition: Vec<usize> = if start == 0 {
            vertices.iter().take(half).map(|&(v, _)| v).collect()
        } else if start == 1 {
            vertices.iter().skip(n - half).map(|&(v, _)| v).collect()
        } else {
            (0..half).collect()
        };

        let (cut, refined) = kernighan_lin(n, adj, partition);
        if cut < best_cut {
            best_cut = cut;
            best_partition = refined;
        }
    }

    (best_cut, best_partition)
}

fn kernighan_lin(n: usize, adj: &[Vec<usize>], mut partition_a: Vec<usize>) -> (usize, Vec<usize>) {
    let mut improved = true;

    while improved {
        improved = false;
        let mut set_a: HashSet<usize> = partition_a.iter().cloned().collect();
        let partition_b: Vec<usize> = (0..n).filter(|v| !set_a.contains(v)).collect();
        let current_cut = calculate_cut(&partition_a, adj);

        for &a in &partition_a.clone() {
            for &b in &partition_b {
                set_a.remove(&a);
                set_a.insert(b);
                let new_partition_a: Vec<usize> = set_a.iter().cloned().collect();
                let new_cut = calculate_cut(&new_partition_a, adj);

                if new_cut < current_cut {
                    partition_a = new_partition_a;
                    improved = true;
                    break;
                }

                set_a.insert(a);
                set_a.remove(&b);
            }
            if improved { break; }
        }
    }

    let final_cut = calculate_cut(&partition_a, adj);
    (final_cut, partition_a)
}
