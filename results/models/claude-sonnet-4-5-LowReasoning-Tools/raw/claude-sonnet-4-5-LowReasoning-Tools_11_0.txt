use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Debug)]
struct Point {
    x: f64,
    y: f64,
}

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

#[derive(Clone, Debug)]
struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

impl Polygon {
    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }

    fn centroid(&self) -> Point {
        let mut cx = 0.0;
        let mut cy = 0.0;
        for v in &self.vertices {
            cx += v.x;
            cy += v.y;
        }
        let n = self.vertices.len() as f64;
        Point { x: cx / n, y: cy / n }
    }

    fn transform(&self, dx: f64, dy: f64, rotation: f64) -> Polygon {
        let cos_r = rotation.cos();
        let sin_r = rotation.sin();

        let vertices = self.vertices.iter().map(|p| {
            let x_rot = p.x * cos_r - p.y * sin_r;
            let y_rot = p.x * sin_r + p.y * cos_r;
            Point {
                x: x_rot + dx,
                y: y_rot + dy,
            }
        }).collect();

        Polygon { vertices }
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for p in &self.vertices {
            min_x = min_x.min(p.x);
            max_x = max_x.max(p.x);
            min_y = min_y.min(p.y);
            max_y = max_y.max(p.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn point_inside(&self, point: &Point) -> bool {
        let mut inside = false;
        let n = self.vertices.len();
        let eps = 1e-9;

        for i in 0..n {
            let j = (i + 1) % n;
            let vi = &self.vertices[i];
            let vj = &self.vertices[j];

            if ((vi.y > point.y - eps) != (vj.y > point.y - eps)) &&
               (point.x < (vj.x - vi.x) * (point.y - vi.y) / (vj.y - vi.y) + vi.x + eps) {
                inside = !inside;
            }
        }

        inside
    }

    fn contains_polygon(&self, other: &Polygon) -> bool {
        // Check if all vertices of other are inside self
        for vertex in &other.vertices {
            if !self.point_inside(vertex) {
                return false;
            }
        }

        // Also check edge midpoints for robustness
        let n = other.vertices.len();
        for i in 0..n.min(8) {  // Check subset for performance
            let j = (i + 1) % n;
            let mid = Point {
                x: (other.vertices[i].x + other.vertices[j].x) / 2.0,
                y: (other.vertices[i].y + other.vertices[j].y) / 2.0,
            };
            if !self.point_inside(&mid) {
                return false;
            }
        }

        true
    }

    fn intersects(&self, other: &Polygon) -> bool {
        // Quick bounding box check
        let bb1 = self.bounding_box();
        let bb2 = other.bounding_box();

        if bb1.2 < bb2.0 || bb2.2 < bb1.0 || bb1.3 < bb2.1 || bb2.3 < bb1.1 {
            return false;
        }

        // Check if any edge of self intersects with any edge of other
        let n1 = self.vertices.len();
        let n2 = other.vertices.len();

        for i in 0..n1 {
            let j = (i + 1) % n1;
            for k in 0..n2 {
                let l = (k + 1) % n2;
                if segments_intersect(
                    &self.vertices[i], &self.vertices[j],
                    &other.vertices[k], &other.vertices[l]
                ) {
                    return true;
                }
            }
        }

        // Check if one polygon is inside the other
        if self.point_inside(&other.vertices[0]) || other.point_inside(&self.vertices[0]) {
            return true;
        }

        false
    }
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    let eps = 1e-9;

    let d1 = direction(p3, p4, p1);
    let d2 = direction(p3, p4, p2);
    let d3 = direction(p1, p2, p3);
    let d4 = direction(p1, p2, p4);

    if ((d1 > eps && d2 < -eps) || (d1 < -eps && d2 > eps)) &&
       ((d3 > eps && d4 < -eps) || (d3 < -eps && d4 > eps)) {
        return true;
    }

    if d1.abs() < eps && on_segment(p3, p1, p4) { return true; }
    if d2.abs() < eps && on_segment(p3, p2, p4) { return true; }
    if d3.abs() < eps && on_segment(p1, p3, p2) { return true; }
    if d4.abs() < eps && on_segment(p1, p4, p2) { return true; }

    false
}

fn direction(p1: &Point, p2: &Point, p3: &Point) -> f64 {
    (p3.x - p1.x) * (p2.y - p1.y) - (p2.x - p1.x) * (p3.y - p1.y)
}

fn on_segment(p1: &Point, p2: &Point, p3: &Point) -> bool {
    let eps = 1e-9;
    p2.x >= p1.x.min(p3.x) - eps && p2.x <= p1.x.max(p3.x) + eps &&
    p2.y >= p1.y.min(p3.y) - eps && p2.y <= p1.y.max(p3.y) + eps
}

struct Stock {
    polygon: Polygon,
    placed_pieces: Vec<Polygon>,
}

impl Stock {
    fn new(polygon: Polygon) -> Self {
        Stock {
            polygon,
            placed_pieces: Vec::new(),
        }
    }

    fn can_place(&self, piece: &Polygon) -> bool {
        // Check if piece fits in stock
        if !self.polygon.contains_polygon(piece) {
            return false;
        }

        // Check if piece overlaps with any placed pieces
        for placed in &self.placed_pieces {
            if piece.intersects(placed) {
                return false;
            }
        }

        true
    }

    fn place(&mut self, piece: Polygon) {
        self.placed_pieces.push(piece);
    }
}

fn solve(stock_polygon: Polygon, pieces: Vec<Polygon>) -> Vec<Placement> {
    let num_pieces = pieces.len();
    let mut placements = vec![Placement { stock_index: 0, x: 0.0, y: 0.0, rotation: 0.0 }; num_pieces];
    let mut stocks = Vec::new();

    // Create sorted indices by area (largest first)
    let mut piece_order: Vec<usize> = (0..num_pieces).collect();
    piece_order.sort_by(|&a, &b| {
        pieces[b].area().partial_cmp(&pieces[a].area()).unwrap()
    });

    // Determine rotation angles based on problem size
    let rotations = if num_pieces < 50 {
        vec![0.0, PI/6.0, PI/4.0, PI/3.0, PI/2.0, 2.0*PI/3.0, 3.0*PI/4.0, 5.0*PI/6.0, PI]
    } else if num_pieces < 500 {
        vec![0.0, PI/4.0, PI/2.0, 3.0*PI/4.0, PI]
    } else if num_pieces < 2000 {
        vec![0.0, PI/2.0, PI]
    } else {
        vec![0.0, PI/2.0]
    };

    // Determine grid resolution based on problem size
    let stock_bb = stock_polygon.bounding_box();
    let stock_width = stock_bb.2 - stock_bb.0;
    let stock_height = stock_bb.3 - stock_bb.1;

    let grid_step = if num_pieces < 50 {
        stock_width.max(stock_height) / 40.0
    } else if num_pieces < 500 {
        stock_width.max(stock_height) / 20.0
    } else if num_pieces < 2000 {
        stock_width.max(stock_height) / 10.0
    } else {
        stock_width.max(stock_height) / 5.0
    };

    for &piece_idx in &piece_order {
        let piece = &pieces[piece_idx];
        let mut placed = false;

        // Try to place on existing stocks
        for (stock_idx, stock) in stocks.iter_mut().enumerate() {
            if placed { break; }

            for &rotation in &rotations {
                if placed { break; }

                // Get rotated piece bounding box for offset calculation
                let rotated_base = piece.transform(0.0, 0.0, rotation);
                let rotated_bb = rotated_base.bounding_box();
                let offset_x = -rotated_bb.0;
                let offset_y = -rotated_bb.1;

                // Try grid positions
                let mut y = stock_bb.1;
                while y <= stock_bb.3 && !placed {
                    let mut x = stock_bb.0;
                    while x <= stock_bb.2 && !placed {
                        let transformed = piece.transform(x + offset_x, y + offset_y, rotation);

                        if stock.can_place(&transformed) {
                            stock.place(transformed.clone());
                            placements[piece_idx] = Placement {
                                stock_index: stock_idx,
                                x: x + offset_x,
                                y: y + offset_y,
                                rotation,
                            };
                            placed = true;
                        }

                        x += grid_step;
                    }
                    y += grid_step;
                }
            }
        }

        // If not placed, create new stock
        if !placed {
            let mut new_stock = Stock::new(stock_polygon.clone());
            let stock_idx = stocks.len();

            for &rotation in &rotations {
                if placed { break; }

                let rotated_base = piece.transform(0.0, 0.0, rotation);
                let rotated_bb = rotated_base.bounding_box();
                let offset_x = -rotated_bb.0;
                let offset_y = -rotated_bb.1;

                let mut y = stock_bb.1;
                while y <= stock_bb.3 && !placed {
                    let mut x = stock_bb.0;
                    while x <= stock_bb.2 && !placed {
                        let transformed = piece.transform(x + offset_x, y + offset_y, rotation);

                        if new_stock.can_place(&transformed) {
                            new_stock.place(transformed.clone());
                            placements[piece_idx] = Placement {
                                stock_index: stock_idx,
                                x: x + offset_x,
                                y: y + offset_y,
                                rotation,
                            };
                            placed = true;
                        }

                        x += grid_step;
                    }
                    y += grid_step;
                }
            }

            stocks.push(new_stock);

            // Fallback: place at origin if still not placed
            if !placed {
                let transformed = piece.transform(stock_bb.0, stock_bb.1, 0.0);
                placements[piece_idx] = Placement {
                    stock_index: stock_idx,
                    x: stock_bb.0,
                    y: stock_bb.1,
                    rotation: 0.0,
                };
            }
        }
    }

    placements
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read stock polygon
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let (sv, np) = (parts[0], parts[1]);

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        stock_vertices.push(Point { x: coords[0], y: coords[1] });
    }
    let stock_polygon = Polygon { vertices: stock_vertices };

    // Read pieces
    let mut pieces = Vec::new();
    for _ in 0..np {
        let line = lines.next().unwrap().unwrap();
        let pv: usize = line.trim().parse().unwrap();

        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let coords: Vec<f64> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            piece_vertices.push(Point { x: coords[0], y: coords[1] });
        }
        pieces.push(Polygon { vertices: piece_vertices });
    }

    // Solve
    let placements = solve(stock_polygon, pieces);

    // Output
    let num_stocks = placements.iter().map(|p| p.stock_index).max().unwrap_or(0) + 1;
    println!("{}", num_stocks);
    for placement in placements {
        println!("{} {} {} {}", placement.stock_index, placement.x, placement.y, placement.rotation);
    }
}
