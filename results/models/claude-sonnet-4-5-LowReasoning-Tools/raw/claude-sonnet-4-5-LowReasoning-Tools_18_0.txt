#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <cstring>
#include <unordered_set>
#include <unordered_map>

using namespace std;

struct Edge {
    int u, v;
};

class MinimumCut {
private:
    int n;
    vector<vector<int>> capacity;
    vector<Edge> originalEdges;
    vector<bool> merged;

    // Stoer-Wagner algorithm for global minimum cut
    pair<int, vector<Edge>> stoerWagner() {
        vector<vector<int>> cap = capacity;
        int minCut = numeric_limits<int>::max();
        vector<int> bestPartition;
        vector<int> nodes(n);
        for (int i = 0; i < n; i++) nodes[i] = i;

        vector<vector<int>> groups(n);
        for (int i = 0; i < n; i++) {
            groups[i].push_back(i);
        }

        for (int phase = 0; phase < n - 1; phase++) {
            vector<int> weight(n, 0);
            vector<bool> inA(n, false);
            int prev = -1, last = -1;

            for (int i = 0; i < n - phase; i++) {
                int maxWeight = -1;
                int next = -1;

                for (int j = 0; j < n; j++) {
                    if (nodes[j] >= 0 && !inA[j] && weight[j] > maxWeight) {
                        maxWeight = weight[j];
                        next = j;
                    }
                }

                if (next == -1) break;

                inA[next] = true;
                prev = last;
                last = next;

                for (int j = 0; j < n; j++) {
                    if (nodes[j] >= 0 && !inA[j]) {
                        weight[j] += cap[next][j];
                    }
                }
            }

            if (last == -1 || prev == -1) break;

            int cutWeight = 0;
            for (int j = 0; j < n; j++) {
                if (nodes[j] >= 0) {
                    cutWeight += cap[last][j];
                }
            }

            if (cutWeight < minCut) {
                minCut = cutWeight;
                bestPartition = groups[last];
            }

            // Merge last into prev
            for (int j = 0; j < n; j++) {
                if (nodes[j] >= 0 && j != last) {
                    cap[prev][j] += cap[last][j];
                    cap[j][prev] = cap[prev][j];
                }
            }

            groups[prev].insert(groups[prev].end(), groups[last].begin(), groups[last].end());
            nodes[last] = -1;
        }

        // Extract cut edges
        unordered_set<int> partition(bestPartition.begin(), bestPartition.end());
        vector<Edge> cutEdges;

        for (const auto& edge : originalEdges) {
            bool uInPartition = partition.count(edge.u) > 0;
            bool vInPartition = partition.count(edge.v) > 0;
            if (uInPartition != vInPartition) {
                cutEdges.push_back(edge);
            }
        }

        return {minCut, cutEdges};
    }

    // For very large graphs, use max-flow based approach with heuristic source/sink
    pair<int, vector<Edge>> maxFlowApproach() {
        // Use node with max degree as source
        int source = 0;
        int maxDegree = 0;
        for (int i = 0; i < n; i++) {
            int degree = 0;
            for (int j = 0; j < n; j++) {
                degree += capacity[i][j];
            }
            if (degree > maxDegree) {
                maxDegree = degree;
                source = i;
            }
        }

        int bestCut = numeric_limits<int>::max();
        vector<int> bestReachable;

        // Try a few different sinks
        for (int sinkIdx = 0; sinkIdx < min(10, n); sinkIdx++) {
            int sink = (source + sinkIdx + 1) % n;
            if (sink == source) continue;

            // Run max flow using BFS (Edmonds-Karp variant)
            vector<vector<int>> flow(n, vector<int>(n, 0));
            int totalFlow = 0;

            while (true) {
                vector<int> parent(n, -1);
                queue<pair<int, int>> q;
                q.push({source, numeric_limits<int>::max()});
                parent[source] = source;

                while (!q.empty()) {
                    auto [u, minCap] = q.front();
                    q.pop();

                    if (u == sink) break;

                    for (int v = 0; v < n; v++) {
                        if (parent[v] == -1 && capacity[u][v] - flow[u][v] > 0) {
                            parent[v] = u;
                            int newCap = min(minCap, capacity[u][v] - flow[u][v]);
                            q.push({v, newCap});
                        }
                    }
                }

                if (parent[sink] == -1) break;

                int pathFlow = numeric_limits<int>::max();
                for (int v = sink; v != source; v = parent[v]) {
                    int u = parent[v];
                    pathFlow = min(pathFlow, capacity[u][v] - flow[u][v]);
                }

                for (int v = sink; v != source; v = parent[v]) {
                    int u = parent[v];
                    flow[u][v] += pathFlow;
                    flow[v][u] -= pathFlow;
                }

                totalFlow += pathFlow;
            }

            if (totalFlow < bestCut) {
                bestCut = totalFlow;

                // Find reachable nodes from source in residual graph
                vector<bool> visited(n, false);
                queue<int> bfs;
                bfs.push(source);
                visited[source] = true;
                bestReachable.clear();

                while (!bfs.empty()) {
                    int u = bfs.front();
                    bfs.pop();
                    bestReachable.push_back(u);

                    for (int v = 0; v < n; v++) {
                        if (!visited[v] && capacity[u][v] - flow[u][v] > 0) {
                            visited[v] = true;
                            bfs.push(v);
                        }
                    }
                }
            }
        }

        // Extract cut edges
        unordered_set<int> partition(bestReachable.begin(), bestReachable.end());
        vector<Edge> cutEdges;

        for (const auto& edge : originalEdges) {
            bool uInPartition = partition.count(edge.u) > 0;
            bool vInPartition = partition.count(edge.v) > 0;
            if (uInPartition != vInPartition) {
                cutEdges.push_back(edge);
            }
        }

        return {bestCut, cutEdges};
    }

public:
    MinimumCut(int numNodes) : n(numNodes), capacity(numNodes, vector<int>(numNodes, 0)), merged(numNodes, false) {}

    void addEdge(int u, int v) {
        originalEdges.push_back({u, v});
        capacity[u][v]++;
        capacity[v][u]++;
    }

    pair<int, vector<Edge>> solve() {
        // Adaptive algorithm selection based on graph size
        if (n <= 500) {
            // Use Stoer-Wagner for smaller graphs
            return stoerWagner();
        } else {
            // Use max-flow approach for larger graphs
            return maxFlowApproach();
        }
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    MinimumCut mc(n);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        mc.addEdge(u, v);
    }

    auto [cutSize, cutEdges] = mc.solve();

    cout << cutSize << "\n";
    for (const auto& edge : cutEdges) {
        cout << edge.u << " " << edge.v << "\n";
    }

    return 0;
}
