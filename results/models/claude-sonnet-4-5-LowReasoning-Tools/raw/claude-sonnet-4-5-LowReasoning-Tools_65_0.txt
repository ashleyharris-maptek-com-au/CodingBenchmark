declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Task {
  id: string;
  duration: number;
  deps: string[];
  deadline: number;
  weight: number;
}

interface ScheduleEntry {
  id: string;
  start: number;
  machine: number;
}

interface TaskInfo {
  task: Task;
  finishTime: number;
  criticalPathLength: number;
}

const machineCount: number = data.machineCount;
const tasks: Task[] = data.tasks;

// Build dependency graph
const taskMap = new Map<string, Task>();
const dependents = new Map<string, Set<string>>();
const inDegree = new Map<string, number>();

for (const task of tasks) {
  taskMap.set(task.id, task);
  if (!dependents.has(task.id)) {
    dependents.set(task.id, new Set());
  }
  inDegree.set(task.id, task.deps.length);

  for (const dep of task.deps) {
    if (!dependents.has(dep)) {
      dependents.set(dep, new Set());
    }
    dependents.get(dep)!.add(task.id);
  }
}

// Calculate critical path length for each task (longest path to any leaf)
const criticalPath = new Map<string, number>();

function calculateCriticalPath(taskId: string): number {
  if (criticalPath.has(taskId)) {
    return criticalPath.get(taskId)!;
  }

  const task = taskMap.get(taskId)!;
  let maxPath = task.duration;

  const deps = dependents.get(taskId);
  if (deps && deps.size > 0) {
    let maxChildPath = 0;
    for (const depId of deps) {
      maxChildPath = Math.max(maxChildPath, calculateCriticalPath(depId));
    }
    maxPath += maxChildPath;
  }

  criticalPath.set(taskId, maxPath);
  return maxPath;
}

for (const task of tasks) {
  calculateCriticalPath(task.id);
}

// Schedule tasks using priority-based approach
const schedule: ScheduleEntry[] = [];
const taskFinishTimes = new Map<string, number>();
const machineAvailable: number[] = new Array(machineCount).fill(0);

// Track remaining dependencies
const remainingDeps = new Map<string, number>();
for (const task of tasks) {
  remainingDeps.set(task.id, task.deps.length);
}

// Priority queue of ready tasks
const readyTasks: Task[] = [];

// Initialize with tasks that have no dependencies
for (const task of tasks) {
  if (task.deps.length === 0) {
    readyTasks.push(task);
  }
}

function getPriority(task: Task, currentTime: number): number {
  // Calculate earliest possible start time based on dependencies
  let earliestStart = currentTime;
  for (const depId of task.deps) {
    const depFinish = taskFinishTimes.get(depId) || 0;
    earliestStart = Math.max(earliestStart, depFinish);
  }

  // Slack time: how much buffer before deadline
  const slack = task.deadline - earliestStart - task.duration;

  // Priority factors (higher = more urgent):
  // 1. Negative slack (past deadline) - highest priority
  // 2. Weight (important tasks)
  // 3. Critical path length (tasks on critical path)
  // 4. Low slack (tight deadline)

  let priority = 0;

  // Heavily penalize tasks that will miss deadline
  if (slack < 0) {
    priority += 1000000 - slack * 10000;
  }

  // Weight factor (scaled)
  priority += task.weight * 1000;

  // Critical path factor
  const cpLength = criticalPath.get(task.id) || 0;
  priority += cpLength * 10;

  // Urgency factor (inverse of slack, but only if positive)
  if (slack >= 0) {
    priority += 10000 / (slack + 1);
  }

  return priority;
}

// Schedule all tasks
while (readyTasks.length > 0 || schedule.length < tasks.length) {
  if (readyTasks.length === 0) {
    break; // Should not happen if input is valid
  }

  // Find earliest available machine time
  const currentTime = Math.min(...machineAvailable);

  // Sort ready tasks by priority
  readyTasks.sort((a, b) => getPriority(b, currentTime) - getPriority(a, currentTime));

  // Take highest priority task
  const task = readyTasks.shift()!;

  // Find earliest time this task can start (after all deps finish)
  let earliestStart = 0;
  for (const depId of task.deps) {
    const depFinish = taskFinishTimes.get(depId) || 0;
    earliestStart = Math.max(earliestStart, depFinish);
  }

  // Find best machine (earliest available that can start at or after earliestStart)
  let bestMachine = 0;
  let bestTime = Math.max(machineAvailable[0], earliestStart);

  for (let m = 1; m < machineCount; m++) {
    const startTime = Math.max(machineAvailable[m], earliestStart);
    if (startTime < bestTime) {
      bestMachine = m;
      bestTime = startTime;
    }
  }

  // Schedule the task
  const startTime = bestTime;
  const finishTime = startTime + task.duration;

  schedule.push({
    id: task.id,
    start: startTime,
    machine: bestMachine
  });

  taskFinishTimes.set(task.id, finishTime);
  machineAvailable[bestMachine] = finishTime;

  // Update dependents - add newly ready tasks
  const deps = dependents.get(task.id);
  if (deps) {
    for (const depId of deps) {
      const remaining = remainingDeps.get(depId)! - 1;
      remainingDeps.set(depId, remaining);

      if (remaining === 0) {
        readyTasks.push(taskMap.get(depId)!);
      }
    }
  }
}

// Output the schedule
console.log(JSON.stringify({ schedule }));
