   %float_pi = OpConstant %float 3.14159
  %float_16 = OpConstant %float 16.0
  %float_0_4 = OpConstant %float 0.4
  %float_0_6 = OpConstant %float 0.6
  %float_0_9 = OpConstant %float 0.9
  %float_0_1 = OpConstant %float 0.1
   %red_color = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
  %blue_color = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9

      %label = OpLabel

               ; Calculate diffuse: max(dot(normal, lightDir), 0.0)

    %diffuse = OpExtInst %float %1 FMax %dot_prod %float_0

               ; Calculate shading factor: 0.4 + 0.6 * diffuse

    %shading = OpFAdd %float %float_0_4 %diffuse_scaled

%shading_vec = OpCompositeConstruct %v3float %shading %shading %shading
