declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Drone {
  id: number;
  x: number;
  y: number;
}

interface Input {
  probability: number[][];
  drones: Drone[];
  maxStepsPerDrone: number;
}

const { probability, drones, maxStepsPerDrone } = data as Input;

const rows = probability.length;
const cols = probability[0]?.length || 0;

// Global coverage tracking - ensures drones don't redundantly cover same cells
const covered = new Set<string>();

// Initialize paths with starting positions
const paths: number[][][] = [];

for (const drone of drones) {
  const key = `${drone.x},${drone.y}`;
  if (probability[drone.y][drone.x] > 0) {
    covered.add(key);
  }
  paths.push([[drone.x, drone.y]]);
}

// Possible moves: stay, up, down, left, right
const directions = [
  [0, 0],   // stay
  [0, -1],  // up
  [0, 1],   // down
  [-1, 0],  // left
  [1, 0]    // right
];

// Adaptive search radius based on grid size
const searchRadius = Math.min(10, Math.max(5, Math.floor(Math.sqrt(rows * cols) / 15)));

// Build paths step by step with coordinated greedy selection
for (let step = 1; step < maxStepsPerDrone; step++) {
  // Process each drone sequentially to coordinate coverage
  for (let droneIdx = 0; droneIdx < drones.length; droneIdx++) {
    const currentPath = paths[droneIdx];
    const [x, y] = currentPath[step - 1];

    let bestPosition = [x, y];
    let bestScore = -Infinity;

    // Evaluate each possible move
    for (const [dx, dy] of directions) {
      const nx = x + dx;
      const ny = y + dy;

      // Validate move: check bounds and obstacles
      if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
      if (probability[ny][nx] === -1) continue;

      // Score this position
      const score = evaluatePosition(nx, ny);

      if (score > bestScore) {
        bestScore = score;
        bestPosition = [nx, ny];
      }
    }

    // Add best move to path
    currentPath.push(bestPosition);

    // Mark cell as covered
    const key = `${bestPosition[0]},${bestPosition[1]}`;
    if (probability[bestPosition[1]][bestPosition[0]] > 0) {
      covered.add(key);
    }
  }
}

function evaluatePosition(x: number, y: number): number {
  const key = `${x},${y}`;
  let score = 0;

  // High weight for immediate uncovered cell value
  if (!covered.has(key) && probability[y][x] > 0) {
    score += probability[y][x] * 100;
  }

  // Look-ahead: value of nearby uncovered cells
  // This helps drones move toward valuable regions
  for (let dy = -searchRadius; dy <= searchRadius; dy++) {
    for (let dx = -searchRadius; dx <= searchRadius; dx++) {
      const ny = y + dy;
      const nx = x + dx;

      if (nx < 0 || nx >= cols || ny < 0 || ny >= rows) continue;
      if (probability[ny][nx] <= 0) continue;

      const neighborKey = `${nx},${ny}`;
      if (!covered.has(neighborKey)) {
        const distance = Math.abs(dx) + Math.abs(dy);
        // Weight by distance - closer cells more valuable
        score += probability[ny][nx] / (1 + distance * 0.25);
      }
    }
  }

  return score;
}
