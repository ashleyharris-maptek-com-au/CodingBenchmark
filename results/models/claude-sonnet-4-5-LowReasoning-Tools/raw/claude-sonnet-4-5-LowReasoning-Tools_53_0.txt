import math

# Module-level state for integration and filtering
_prev_pos = None
_prev_vel = None
_integrated_vel = None
_vel_history = []
_pos_history = []
_gyro_bias = [0.0, 0.0, 0.0]
_total_time = 0.0

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autonomous docking autopilot for spacecraft.
    Handles sensor fusion, orbital mechanics, and robust control.
    """
    global _prev_pos, _prev_vel, _integrated_vel, _vel_history, _pos_history
    global _gyro_bias, _total_time

    _total_time += dt

    # ========== SENSOR FUSION ==========
    # Fuse position estimates from multiple sources
    rel_pos, rel_pos_confidence = fuse_position(state)

    # Fuse velocity estimates
    rel_vel, rel_vel_confidence = fuse_velocity(state, rel_pos, dt)

    # Get range estimate
    range_m = estimate_range(state, rel_pos)

    # Get relative speed
    rel_speed = math.sqrt(rel_vel[0]**2 + rel_vel[1]**2 + rel_vel[2]**2) if rel_vel else 0.0

    # Update history for filtering
    if rel_pos:
        _pos_history.append(rel_pos)
        if len(_pos_history) > 10:
            _pos_history.pop(0)

    if rel_vel:
        _vel_history.append(rel_vel)
        if len(_vel_history) > 10:
            _vel_history.pop(0)

    # Apply smoothing if we have history
    if len(_pos_history) >= 3:
        rel_pos = median_filter(_pos_history[-5:])

    if len(_vel_history) >= 3:
        rel_vel = median_filter(_vel_history[-5:])

    # ========== CONTROL STRATEGY SELECTION ==========
    fuel_pct = state.get('fuel_remaining_pct', 100.0)
    fuel_kg = state.get('fuel_remaining_kg', 500.0)
    low_fuel = fuel_kg < 100.0 or fuel_pct < 20.0

    # Default no thrust
    thrust_x, thrust_y, thrust_z = 0.0, 0.0, 0.0

    if rel_pos and rel_vel:
        # Determine control regime based on range
        if range_m > 200.0:
            # Far range: Orbital mechanics regime
            thrust_x, thrust_y, thrust_z = far_range_control(
                rel_pos, rel_vel, range_m, low_fuel
            )
        elif range_m > 50.0:
            # Medium range: Transition regime
            thrust_x, thrust_y, thrust_z = medium_range_control(
                rel_pos, rel_vel, range_m, low_fuel
            )
        elif range_m > 10.0:
            # Close range: Linear regime
            thrust_x, thrust_y, thrust_z = close_range_control(
                rel_pos, rel_vel, range_m, low_fuel
            )
        else:
            # Final approach: Very gentle
            thrust_x, thrust_y, thrust_z = final_approach_control(
                rel_pos, rel_vel, range_m
            )

    # Apply fuel conservation if low
    if low_fuel:
        thrust_x *= 0.5
        thrust_y *= 0.5
        thrust_z *= 0.5

    # Clamp thrusts
    thrust_x = max(-1.0, min(1.0, thrust_x))
    thrust_y = max(-1.0, min(1.0, thrust_y))
    thrust_z = max(-1.0, min(1.0, thrust_z))

    # ========== ATTITUDE CONTROL ==========
    # Estimate angular rates from gyros
    roll_rate, pitch_rate, yaw_rate = fuse_gyros(state)

    # Simple damping control
    att_rate_roll = -roll_rate * 0.3
    att_rate_pitch = -pitch_rate * 0.3
    att_rate_yaw = -yaw_rate * 0.3

    # Clamp attitude rates
    max_rate = 0.035
    att_rate_roll = max(-max_rate, min(max_rate, att_rate_roll))
    att_rate_pitch = max(-max_rate, min(max_rate, att_rate_pitch))
    att_rate_yaw = max(-max_rate, min(max_rate, att_rate_yaw))

    _prev_pos = rel_pos
    _prev_vel = rel_vel

    return {
        'thrust_x': thrust_x,
        'thrust_y': thrust_y,
        'thrust_z': thrust_z,
        'att_rate_roll': att_rate_roll,
        'att_rate_pitch': att_rate_pitch,
        'att_rate_yaw': att_rate_yaw
    }


def fuse_position(state):
    """Fuse position estimates from radar, lidar, and GPS."""
    positions = []
    weights = []

    # Radar relative position
    if (state.get('radar_rel_x_valid') and
        state.get('radar_rel_y_valid') and
        state.get('radar_rel_z_valid')):
        pos = [
            state.get('radar_rel_pos_x_m', 0.0),
            state.get('radar_rel_pos_y_m', 0.0),
            state.get('radar_rel_pos_z_m', 0.0)
        ]
        positions.append(pos)
        weights.append(1.0)

    # Lidar relative position
    if (state.get('lidar_rel_x_valid') and
        state.get('lidar_rel_y_valid') and
        state.get('lidar_rel_z_valid')):
        pos = [
            state.get('lidar_rel_pos_x_m', 0.0),
            state.get('lidar_rel_pos_y_m', 0.0),
            state.get('lidar_rel_pos_z_m', 0.0)
        ]
        positions.append(pos)
        weights.append(1.0)

    # GPS differential (if both valid, compute relative position)
    if (state.get('gps_1_x_valid') and state.get('gps_1_y_valid') and
        state.get('gps_1_z_valid') and state.get('gps_2_x_valid') and
        state.get('gps_2_y_valid') and state.get('gps_2_z_valid')):
        # Assume GPS2 is station position, GPS1 is spacecraft
        pos = [
            state.get('gps_2_pos_x_m', 0.0) - state.get('gps_1_pos_x_m', 0.0),
            state.get('gps_2_pos_y_m', 0.0) - state.get('gps_1_pos_y_m', 0.0),
            state.get('gps_2_pos_z_m', 0.0) - state.get('gps_1_pos_z_m', 0.0)
        ]
        positions.append(pos)
        weights.append(0.8)  # Lower weight for GPS

    if not positions:
        return None, 0.0

    # Weighted average
    fused = [0.0, 0.0, 0.0]
    total_weight = sum(weights)
    for i in range(3):
        fused[i] = sum(p[i] * w for p, w in zip(positions, weights)) / total_weight

    confidence = min(1.0, len(positions) / 3.0)
    return fused, confidence


def fuse_velocity(state, current_pos, dt):
    """Fuse velocity estimates from radar, lidar, and differentiation."""
    global _prev_pos

    velocities = []
    weights = []

    # Radar relative velocity
    if (state.get('radar_relv_x_valid') and
        state.get('radar_relv_y_valid') and
        state.get('radar_relv_z_valid')):
        vel = [
            state.get('radar_rel_vel_x_ms', 0.0),
            state.get('radar_rel_vel_y_ms', 0.0),
            state.get('radar_rel_vel_z_ms', 0.0)
        ]
        velocities.append(vel)
        weights.append(1.0)

    # Lidar relative velocity
    if (state.get('lidar_relv_x_valid') and
        state.get('lidar_relv_y_valid') and
        state.get('lidar_relv_z_valid')):
        vel = [
            state.get('lidar_rel_vel_x_ms', 0.0),
            state.get('lidar_rel_vel_y_ms', 0.0),
            state.get('lidar_rel_vel_z_ms', 0.0)
        ]
        velocities.append(vel)
        weights.append(1.0)

    # Differentiate position if available
    if current_pos and _prev_pos and dt > 0.001:
        vel = [
            (current_pos[0] - _prev_pos[0]) / dt,
            (current_pos[1] - _prev_pos[1]) / dt,
            (current_pos[2] - _prev_pos[2]) / dt
        ]
        velocities.append(vel)
        weights.append(0.5)  # Lower weight for numerical differentiation

    if not velocities:
        return None, 0.0

    # Weighted average
    fused = [0.0, 0.0, 0.0]
    total_weight = sum(weights)
    for i in range(3):
        fused[i] = sum(v[i] * w for v, w in zip(velocities, weights)) / total_weight

    confidence = min(1.0, len(velocities) / 3.0)
    return fused, confidence


def estimate_range(state, rel_pos):
    """Estimate range using all available sensors."""
    ranges = []

    # Range finders
    if state.get('range_1_valid'):
        ranges.append(state.get('range_1_m', 0.0))
    if state.get('range_2_valid'):
        ranges.append(state.get('range_2_m', 0.0))
    if state.get('range_3_valid'):
        ranges.append(state.get('range_3_m', 0.0))

    # Docking camera
    if state.get('dock_camera_active'):
        ranges.append(state.get('dock_camera_range_m', 0.0))

    # Target range
    if state.get('target_dock_range_m', 0.0) > 0:
        ranges.append(state.get('target_dock_range_m'))

    # Computed from position
    if rel_pos:
        computed_range = math.sqrt(rel_pos[0]**2 + rel_pos[1]**2 + rel_pos[2]**2)
        ranges.append(computed_range)

    if ranges:
        # Use median to reject outliers
        ranges_sorted = sorted(ranges)
        return ranges_sorted[len(ranges_sorted) // 2]

    return 1000.0  # Default large range


def far_range_control(rel_pos, rel_vel, range_m, low_fuel):
    """
    Far range control (>200m): Use orbital mechanics.
    Key insight: thrust retrograde to catch up (lower orbit = faster).
    """
    x, y, z = rel_pos
    vx, vy, vz = rel_vel

    # Target relative position (0, 0, 0)
    # In LVLH: y is along-track, negative y means we're behind

    # Proportional-Derivative gains (tuned for orbital mechanics)
    kp_y = 0.0008  # Along-track position gain
    kd_y = 0.015   # Along-track velocity gain
    kp_x = 0.0005  # Radial position gain
    kd_x = 0.01    # Radial velocity gain
    kp_z = 0.0008  # Cross-track position gain
    kd_z = 0.012   # Cross-track velocity gain

    # Along-track control (orbital mechanics dominant)
    # If behind (y < 0), thrust retrograde (negative) to catch up
    thrust_y = -kp_y * y - kd_y * vy

    # Radial control (with orbital coupling)
    thrust_x = -kp_x * x - kd_x * vx

    # Cross-track control
    thrust_z = -kp_z * z - kd_z * vz

    # Limit thrust magnitude for fuel efficiency
    if low_fuel:
        max_thrust = 0.3
    else:
        max_thrust = 0.6

    thrust_mag = math.sqrt(thrust_x**2 + thrust_y**2 + thrust_z**2)
    if thrust_mag > max_thrust:
        scale = max_thrust / thrust_mag
        thrust_x *= scale
        thrust_y *= scale
        thrust_z *= scale

    return thrust_x, thrust_y, thrust_z


def medium_range_control(rel_pos, rel_vel, range_m, low_fuel):
    """
    Medium range control (50-200m): Transition regime.
    Gentle corrections with V-bar approach preference.
    """
    x, y, z = rel_pos
    vx, vy, vz = rel_vel

    # Increased gains for tighter control
    kp_y = 0.002
    kd_y = 0.025
    kp_x = 0.0015
    kd_x = 0.020
    kp_z = 0.002
    kd_z = 0.025

    # PD control for all axes
    thrust_x = -kp_x * x - kd_x * vx
    thrust_y = -kp_y * y - kd_y * vy
    thrust_z = -kp_z * z - kd_z * vz

    # Limit thrust
    max_thrust = 0.4 if low_fuel else 0.7
    thrust_mag = math.sqrt(thrust_x**2 + thrust_y**2 + thrust_z**2)
    if thrust_mag > max_thrust:
        scale = max_thrust / thrust_mag
        thrust_x *= scale
        thrust_y *= scale
        thrust_z *= scale

    return thrust_x, thrust_y, thrust_z


def close_range_control(rel_pos, rel_vel, range_m, low_fuel):
    """
    Close range control (10-50m): Linear regime.
    Direct thrust corrections work well here.
    """
    x, y, z = rel_pos
    vx, vy, vz = rel_vel

    # Higher gains for responsive control
    kp = 0.005
    kd = 0.08

    thrust_x = -kp * x - kd * vx
    thrust_y = -kp * y - kd * vy
    thrust_z = -kp * z - kd * vz

    # Limit approach speed
    desired_speed = 0.3  # m/s
    current_speed = math.sqrt(vx**2 + vy**2 + vz**2)
    if current_speed > desired_speed and range_m < 30:
        # Add extra damping
        thrust_x -= 0.15 * vx
        thrust_y -= 0.15 * vy
        thrust_z -= 0.15 * vz

    max_thrust = 0.5 if low_fuel else 0.8
    thrust_mag = math.sqrt(thrust_x**2 + thrust_y**2 + thrust_z**2)
    if thrust_mag > max_thrust:
        scale = max_thrust / thrust_mag
        thrust_x *= scale
        thrust_y *= scale
        thrust_z *= scale

    return thrust_x, thrust_y, thrust_z


def final_approach_control(rel_pos, rel_vel, range_m):
    """
    Final approach (<10m): Very gentle, target < 0.1 m/s.
    """
    x, y, z = rel_pos
    vx, vy, vz = rel_vel

    # Very gentle control
    kp = 0.008
    kd = 0.15

    thrust_x = -kp * x - kd * vx
    thrust_y = -kp * y - kd * vy
    thrust_z = -kp * z - kd * vz

    # Strict speed limit for docking
    target_speed = 0.05 if range_m < 5 else 0.1
    current_speed = math.sqrt(vx**2 + vy**2 + vz**2)
    if current_speed > target_speed:
        # Strong damping
        thrust_x -= 0.3 * vx
        thrust_y -= 0.3 * vy
        thrust_z -= 0.3 * vz

    # Very limited thrust for safety
    max_thrust = 0.3
    thrust_mag = math.sqrt(thrust_x**2 + thrust_y**2 + thrust_z**2)
    if thrust_mag > max_thrust:
        scale = max_thrust / thrust_mag
        thrust_x *= scale
        thrust_y *= scale
        thrust_z *= scale

    return thrust_x, thrust_y, thrust_z


def fuse_gyros(state):
    """Fuse gyroscope measurements with voting."""
    roll_rates = []
    pitch_rates = []
    yaw_rates = []

    # Gyro 1
    if state.get('gyro_1_roll_valid'):
        roll_rates.append(math.radians(state.get('gyro_1_roll_dps', 0.0)))
    if state.get('gyro_1_pitch_valid'):
        pitch_rates.append(math.radians(state.get('gyro_1_pitch_dps', 0.0)))
    if state.get('gyro_1_yaw_valid'):
        yaw_rates.append(math.radians(state.get('gyro_1_yaw_dps', 0.0)))

    # Gyro 2
    if state.get('gyro_2_roll_valid'):
        roll_rates.append(math.radians(state.get('gyro_2_roll_dps', 0.0)))
    if state.get('gyro_2_pitch_valid'):
        pitch_rates.append(math.radians(state.get('gyro_2_pitch_dps', 0.0)))
    if state.get('gyro_2_yaw_valid'):
        yaw_rates.append(math.radians(state.get('gyro_2_yaw_dps', 0.0)))

    # Gyro 3
    if state.get('gyro_3_roll_valid'):
        roll_rates.append(math.radians(state.get('gyro_3_roll_dps', 0.0)))
    if state.get('gyro_3_pitch_valid'):
        pitch_rates.append(math.radians(state.get('gyro_3_pitch_dps', 0.0)))
    if state.get('gyro_3_yaw_valid'):
        yaw_rates.append(math.radians(state.get('gyro_3_yaw_dps', 0.0)))

    # Use median to reject outliers
    roll_rate = median(roll_rates) if roll_rates else 0.0
    pitch_rate = median(pitch_rates) if pitch_rates else 0.0
    yaw_rate = median(yaw_rates) if yaw_rates else 0.0

    return roll_rate, pitch_rate, yaw_rate


def median(values):
    """Compute median of a list."""
    if not values:
        return 0.0
    sorted_values = sorted(values)
    n = len(sorted_values)
    if n % 2 == 0:
        return (sorted_values[n//2 - 1] + sorted_values[n//2]) / 2.0
    else:
        return sorted_values[n//2]


def median_filter(position_list):
    """Apply median filter to list of [x, y, z] positions."""
    if not position_list:
        return [0.0, 0.0, 0.0]

    x_vals = [p[0] for p in position_list]
    y_vals = [p[1] for p in position_list]
    z_vals = [p[2] for p in position_list]

    return [median(x_vals), median(y_vals), median(z_vals)]
