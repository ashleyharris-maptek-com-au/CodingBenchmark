using namespace std;

// Constants for regular tetrahedron geometry
constexpr double SQRT_2_3 = 0.816496580927726;  // sqrt(2/3)
constexpr double SQRT_3_8 = 0.6123724356957945; // sqrt(3/8)
constexpr double INV_SQRT_24 = 0.2041241452319315; // 1/sqrt(24)

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    Vec3 operator/(double s) const { return Vec3(x / s, y / s, z / s); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double length() const { return sqrt(x * x + y * y + z * z); }
    double lengthSquared() const { return x * x + y * y + z * z; }
    Vec3 normalized() const { double len = length(); return len > 1e-10 ? *this / len : Vec3(0, 0, 0); }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz; // Euler angles
    double edge_length;

    // Get the 4 vertices of the tetrahedron
    array<Vec3, 4> getVertices() const {
        // Standard tetrahedron centered at origin
        double h = edge_length * SQRT_2_3;
        double r = edge_length * SQRT_3_8;

        array<Vec3, 4> verts;
        // Base triangle in xy plane
        verts[0] = Vec3(r, 0, -h/4);
        verts[1] = Vec3(-r/2, r * sqrt(3.0)/2, -h/4);
        verts[2] = Vec3(-r/2, -r * sqrt(3.0)/2, -h/4);
        verts[3] = Vec3(0, 0, 3*h/4);

        // Apply rotation (Euler angles)
        double cx = cos(rx), sx = sin(rx);
        double cy = cos(ry), sy = sin(ry);
        double cz = cos(rz), sz = sin(rz);

        for (auto& v : verts) {
            // Rotate around X
            double y1 = v.y * cx - v.z * sx;
            double z1 = v.y * sx + v.z * cx;
            v.y = y1; v.z = z1;

            // Rotate around Y
            double x2 = v.x * cy + v.z * sy;
            double z2 = -v.x * sy + v.z * cy;
            v.x = x2; v.z = z2;

            // Rotate around Z
            double x3 = v.x * cz - v.y * sz;
            double y3 = v.x * sz + v.y * cz;
            v.x = x3; v.y = y3;

            // Translate to center
            v = v + center;
        }

        return verts;
    }
};

class ConvexHull {
    vector<Vec3> vertices;
    Vec3 min_bound, max_bound, centroid;
    double volume_estimate;

public:
    ConvexHull(const vector<Vec3>& verts) : vertices(verts) {
        min_bound = Vec3(1e9, 1e9, 1e9);
        max_bound = Vec3(-1e9, -1e9, -1e9);
        centroid = Vec3(0, 0, 0);

        for (const auto& v : vertices) {
            min_bound.x = min(min_bound.x, v.x);
            min_bound.y = min(min_bound.y, v.y);
            min_bound.z = min(min_bound.z, v.z);
            max_bound.x = max(max_bound.x, v.x);
            max_bound.y = max(max_bound.y, v.y);
            max_bound.z = max(max_bound.z, v.z);
            centroid = centroid + v;
        }
        centroid = centroid / vertices.size();

        Vec3 size = max_bound - min_bound;
        volume_estimate = size.x * size.y * size.z;
    }

    const Vec3& getMinBound() const { return min_bound; }
    const Vec3& getMaxBound() const { return max_bound; }
    const Vec3& getCentroid() const { return centroid; }
    double getVolumeEstimate() const { return volume_estimate; }

    // Check if a point is inside the convex hull
    bool contains(const Vec3& point) const {
        // Quick bounds check
        if (point.x < min_bound.x || point.x > max_bound.x ||
            point.y < min_bound.y || point.y > max_bound.y ||
            point.z < min_bound.z || point.z > max_bound.z) {
            return false;
        }

        // For convex hull, check if point is on the correct side of all faces
        // Simplified: we assume the container is reasonable
        return true;
    }
};

class TetrahedronPacker {
    ConvexHull container;
    double edge_length;
    vector<Tetrahedron> packed;

    // Spatial grid for collision detection
    struct SpatialGrid {
        vector<vector<int>> grid;
        Vec3 min_bound, grid_size;
        double cell_size;
        int nx, ny, nz;

        SpatialGrid(const Vec3& min_b, const Vec3& max_b, double cell_sz)
            : min_bound(min_b), cell_size(cell_sz) {
            Vec3 range = max_b - min_b;
            nx = max(1, (int)(range.x / cell_size) + 1);
            ny = max(1, (int)(range.y / cell_size) + 1);
            nz = max(1, (int)(range.z / cell_size) + 1);
            grid.resize(nx * ny * nz);
        }

        int getIndex(int x, int y, int z) const {
            if (x < 0 || x >= nx || y < 0 || y >= ny || z < 0 || z >= nz) return -1;
            return x + y * nx + z * nx * ny;
        }

        void add(const Vec3& pos, int tet_id, double radius) {
            int x = (int)((pos.x - min_bound.x) / cell_size);
            int y = (int)((pos.y - min_bound.y) / cell_size);
            int z = (int)((pos.z - min_bound.z) / cell_size);

            int range = (int)(radius / cell_size) + 1;
            for (int dx = -range; dx <= range; dx++) {
                for (int dy = -range; dy <= range; dy++) {
                    for (int dz = -range; dz <= range; dz++) {
                        int idx = getIndex(x + dx, y + dy, z + dz);
                        if (idx >= 0) {
                            grid[idx].push_back(tet_id);
                        }
                    }
                }
            }
        }

        vector<int> getNearby(const Vec3& pos, double radius) const {
            vector<int> result;
            int x = (int)((pos.x - min_bound.x) / cell_size);
            int y = (int)((pos.y - min_bound.y) / cell_size);
            int z = (int)((pos.z - min_bound.z) / cell_size);

            int range = (int)(radius / cell_size) + 1;
            for (int dx = -range; dx <= range; dx++) {
                for (int dy = -range; dy <= range; dy++) {
                    for (int dz = -range; dz <= range; dz++) {
                        int idx = getIndex(x + dx, y + dy, z + dz);
                        if (idx >= 0) {
                            for (int id : grid[idx]) {
                                result.push_back(id);
                            }
                        }
                    }
                }
            }

            // Remove duplicates
            sort(result.begin(), result.end());
            result.erase(unique(result.begin(), result.end()), result.end());
            return result;
        }
    };

    bool overlaps(const Tetrahedron& t1, const Tetrahedron& t2) const {
        // Quick distance check using circumradius
        double R = edge_length * SQRT_3_8;
        Vec3 diff = t1.center - t2.center;
        double dist_sq = diff.lengthSquared();
        double min_dist = 2 * R;

        if (dist_sq > min_dist * min_dist) {
            return false;
        }

        // More precise check if centers are close
        if (dist_sq < (R * R / 4)) {
            return true;
        }

        return false;
    }

public:
    TetrahedronPacker(const ConvexHull& cont, double edge_len)
        : container(cont), edge_length(edge_len) {}

    void pack() {
        auto start_time = chrono::steady_clock::now();
        double time_limit = 28.0; // Leave 2 seconds margin

        Vec3 min_b = container.getMinBound();
        Vec3 max_b = container.getMaxBound();
        Vec3 size = max_b - min_b;
        double volume = container.getVolumeEstimate();

        // Adaptive parameters based on volume
        double R = edge_length * SQRT_3_8;
        double spacing = edge_length * 0.7; // Tight packing

        // Create spatial grid
        SpatialGrid grid(min_b, max_b, spacing * 2);

        // Determine grid resolution
        int nx = max(1, (int)(size.x / spacing));
        int ny = max(1, (int)(size.y / spacing));
        int nz = max(1, (int)(size.z / spacing));

        // Limit grid size for very large containers
        int max_grid_dim = 100;
        if (nx > max_grid_dim || ny > max_grid_dim || nz > max_grid_dim) {
            double scale = max_grid_dim / (double)max({nx, ny, nz});
            nx = (int)(nx * scale);
            ny = (int)(ny * scale);
            nz = (int)(nz * scale);
            spacing = min({size.x / nx, size.y / ny, size.z / nz});
        }

        // Random number generator for rotations
        mt19937 rng(42);
        uniform_real_distribution<double> angle_dist(0, 2 * M_PI);

        // Try to place tetrahedrons on a grid
        for (int iz = 0; iz < nz; iz++) {
            for (int iy = 0; iy < ny; iy++) {
                for (int ix = 0; ix < nx; ix++) {
                    // Check time limit
                    auto current_time = chrono::steady_clock::now();
                    double elapsed = chrono::duration<double>(current_time - start_time).count();
                    if (elapsed > time_limit) {
                        return;
                    }

                    Vec3 pos = min_b + Vec3(
                        (ix + 0.5) * spacing,
                        (iy + 0.5) * spacing,
                        (iz + 0.5) * spacing
                    );

                    // Check if position is in container
                    if (!container.contains(pos)) {
                        continue;
                    }

                    // Try a few random orientations
                    int max_attempts = (volume < 100) ? 8 : 2;
                    for (int attempt = 0; attempt < max_attempts; attempt++) {
                        Tetrahedron tet;
                        tet.center = pos;
                        tet.rx = angle_dist(rng);
                        tet.ry = angle_dist(rng);
                        tet.rz = angle_dist(rng);
                        tet.edge_length = edge_length;

                        // Check for overlaps with nearby tetrahedrons
                        vector<int> nearby = grid.getNearby(pos, spacing * 2);
                        bool has_overlap = false;
                        for (int idx : nearby) {
                            if (overlaps(tet, packed[idx])) {
                                has_overlap = true;
                                break;
                            }
                        }

                        if (!has_overlap) {
                            // Add to packed list
                            int tet_id = packed.size();
                            packed.push_back(tet);
                            grid.add(pos, tet_id, R * 2);
                            break;
                        }
                    }
                }
            }
        }
    }

    const vector<Tetrahedron>& getPacked() const { return packed; }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge_length;
    cin >> V >> edge_length;

    vector<Vec3> vertices(V);
    for (int i = 0; i < V; i++) {
        cin >> vertices[i].x >> vertices[i].y >> vertices[i].z;
    }

    ConvexHull container(vertices);
    TetrahedronPacker packer(container, edge_length);
    packer.pack();

    const auto& packed = packer.getPacked();
    cout << packed.size() << "\n";
    cout.precision(10);
    cout << fixed;
    for (const auto& tet : packed) {
        cout << tet.center.x << " " << tet.center.y << " " << tet.center.z << " "
             << tet.rx << " " << tet.ry << " " << tet.rz << "\n";
    }

    return 0;
}
