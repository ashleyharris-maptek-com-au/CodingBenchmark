OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color_in %gl_out %outColor
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
     %uint_4 = OpConstant %uint 4
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output
 %thickness = OpConstant %float 0.025

       %main = OpFunction %void None %3
          %5 = OpLabel

; Read positions
        %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %p0_ptr
        %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %p1_ptr
        %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2 = OpLoad %v4float %p2_ptr
        %p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
        %p3 = OpLoad %v4float %p3_ptr

; Read colors
        %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
        %c1 = OpLoad %v3float %c1_ptr
        %c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
        %c2 = OpLoad %v3float %c2_ptr

; Compute tangent at start: t0 = normalize(p2 - p0)
        %t0_4 = OpFSub %v4float %p2 %p0
        %t0_xy = OpVectorShuffle %v2float %t0_4 %t0_4 0 1
        %t0_len = OpDot %float %t0_xy %t0_xy
        %t0_len_sqrt = OpExtInst %float %1 Sqrt %t0_len
        %t0_norm = OpFDiv %v2float %t0_xy %t0_len_sqrt

; Compute tangent at end: t1 = normalize(p3 - p1)
        %t1_4 = OpFSub %v4float %p3 %p1
        %t1_xy = OpVectorShuffle %v2float %t1_4 %t1_4 0 1
        %t1_len = OpDot %float %t1_xy %t1_xy
        %t1_len_sqrt = OpExtInst %float %1 Sqrt %t1_len
        %t1_norm = OpFDiv %v2float %t1_xy %t1_len_sqrt

; Compute perpendiculars: perp0 = vec2(-t0.y, t0.x)
        %t0_x = OpCompositeExtract %float %t0_norm 0
        %t0_y = OpCompositeExtract %float %t0_norm 1
        %t0_y_neg = OpFNegate %float %t0_y
        %perp0 = OpCompositeConstruct %v2float %t0_y_neg %t0_x

; perp1 = vec2(-t1.y, t1.x)
        %t1_x = OpCompositeExtract %float %t1_norm 0
        %t1_y = OpCompositeExtract %float %t1_norm 1
        %t1_y_neg = OpFNegate %float %t1_y
        %perp1 = OpCompositeConstruct %v2float %t1_y_neg %t1_x

; Scale perpendiculars by thickness
        %perp0_t = OpVectorTimesScalar %v2float %perp0 %thickness
        %perp1_t = OpVectorTimesScalar %v2float %perp1 %thickness

; Extract xy from p1 and p2
        %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
        %p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1

; Vertex 1: p1 + perp0*thickness
        %v1_xy = OpFAdd %v2float %p1_xy %perp0_t
        %p1_z = OpCompositeExtract %float %p1 2
        %p1_w = OpCompositeExtract %float %p1 3
        %v1_pos = OpCompositeConstruct %v4float %v1_xy %p1_z %p1_w
        %out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
        OpStore %out_ptr1 %v1_pos
        OpStore %outColor %c1
        OpEmitVertex

; Vertex 2: p1 - perp0*thickness
        %perp0_t_neg = OpFNegate %v2float %perp0_t
        %v2_xy = OpFAdd %v2float %p1_xy %perp0_t_neg
        %v2_pos = OpCompositeConstruct %v4float %v2_xy %p1_z %p1_w
        OpStore %out_ptr1 %v2_pos
        OpStore %outColor %c1
        OpEmitVertex

; Vertex 3: p2 + perp1*thickness
        %v3_xy = OpFAdd %v2float %p2_xy %perp1_t
        %p2_z = OpCompositeExtract %float %p2 2
        %p2_w = OpCompositeExtract %float %p2 3
        %v3_pos = OpCompositeConstruct %v4float %v3_xy %p2_z %p2_w
        OpStore %out_ptr1 %v3_pos
        OpStore %outColor %c2
        OpEmitVertex

; Vertex 4: p2 - perp1*thickness
        %perp1_t_neg = OpFNegate %v2float %perp1_t
        %v4_xy = OpFAdd %v2float %p2_xy %perp1_t_neg
        %v4_pos = OpCompositeConstruct %v4float %v4_xy %p2_z %p2_w
        OpStore %out_ptr1 %v4_pos
        OpStore %outColor %c2
        OpEmitVertex

        OpEndPrimitive
        OpReturn
        OpFunctionEnd
