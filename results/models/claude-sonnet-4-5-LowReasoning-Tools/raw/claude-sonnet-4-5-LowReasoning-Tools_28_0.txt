use std::io::{self, BufRead};
use std::collections::{HashSet, HashMap};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Build adjacency list
    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];
    let mut edges = Vec::with_capacity(num_edges);

    for _ in 0..num_edges {
        if let Some(Ok(line)) = lines.next() {
            let edge: Vec<usize> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            let u = edge[0];
            let v = edge[1];
            adj[u].insert(v);
            adj[v].insert(u);
            edges.push((u, v));
        }
    }

    // Greedy algorithm: repeatedly pick vertex with highest degree
    let mut cover = HashSet::new();
    let mut uncovered_edges: HashSet<(usize, usize)> = edges.iter()
        .map(|&(u, v)| if u < v { (u, v) } else { (v, u) })
        .collect();

    let mut degrees: HashMap<usize, usize> = HashMap::new();
    for v in 0..num_vertices {
        if !adj[v].is_empty() {
            degrees.insert(v, adj[v].len());
        }
    }

    while !uncovered_edges.is_empty() {
        // Find vertex with maximum degree among vertices that cover uncovered edges
        let mut max_vertex = None;
        let mut max_coverage = 0;

        for &v in degrees.keys() {
            if cover.contains(&v) {
                continue;
            }

            // Count how many uncovered edges this vertex covers
            let coverage = uncovered_edges.iter()
                .filter(|&&(u, w)| u == v || w == v)
                .count();

            if coverage > max_coverage {
                max_coverage = coverage;
                max_vertex = Some(v);
            }
        }

        if let Some(v) = max_vertex {
            cover.insert(v);

            // Remove covered edges
            uncovered_edges.retain(|&(u, w)| u != v && w != v);
        } else {
            // Fallback: pick any vertex from remaining edges
            if let Some(&(u, _)) = uncovered_edges.iter().next() {
                cover.insert(u);
                uncovered_edges.retain(|&(a, b)| a != u && b != u);
            }
        }
    }

    // Try to remove vertices that are redundant
    let mut final_cover: Vec<usize> = cover.into_iter().collect();
    final_cover.sort_unstable();

    let cover_set: HashSet<usize> = final_cover.iter().copied().collect();
    let mut to_remove = Vec::new();

    for &v in &final_cover {
        // A vertex is redundant if all its neighbors are in the cover
        let all_neighbors_covered = adj[v].iter().all(|&u| cover_set.contains(&u));

        if all_neighbors_covered {
            to_remove.push(v);
        }
    }

    final_cover.retain(|v| !to_remove.contains(v));

    // Output
    println!("{}", final_cover.len());
    if !final_cover.is_empty() {
        let output: Vec<String> = final_cover.iter().map(|v| v.to_string()).collect();
        println!("{}", output.join(" "));
    }
}
