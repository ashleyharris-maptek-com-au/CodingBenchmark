use std::io::{self, BufRead};
use std::f64::consts::PI;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read N and M
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    // Read edges
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let parts: Vec<usize> = line.split_whitespace()
                .map(|s| s.parse().unwrap())
                .collect();
            if parts.len() >= 2 && parts[0] < n && parts[1] < n {
                edges.push((parts[0], parts[1]));
            }
        }
    }

    // Compute layout using adaptive strategy
    let positions = compute_layout(n, &edges);

    // Output results
    println!("{}", n);
    for (x, y) in positions {
        println!("{} {}", x, y);
    }
}

fn compute_layout(n: usize, edges: &[(usize, usize)]) -> Vec<(f64, f64)> {
    // Adaptive strategy based on graph size
    if n <= 100 {
        // Small graphs: Full force-directed with many iterations
        force_directed_layout(n, edges, 500, true)
    } else if n <= 1000 {
        // Medium graphs: Force-directed with moderate iterations
        force_directed_layout(n, edges, 200, true)
    } else if n <= 10000 {
        // Large graphs: Reduced iterations, limited repulsion
        force_directed_layout(n, edges, 50, false)
    } else {
        // Very large graphs: Spring-only layout (O(m) complexity)
        spring_only_layout(n, edges, 20)
    }
}

fn force_directed_layout(n: usize, edges: &[(usize, usize)], iterations: usize, full_repulsion: bool) -> Vec<(f64, f64)> {
    // Initialize positions in a circle to break symmetry
    let mut positions: Vec<(f64, f64)> = (0..n)
        .map(|i| {
            let angle = 2.0 * PI * i as f64 / n as f64;
            let radius = 100.0 * (n as f64).sqrt();
            (radius * angle.cos(), radius * angle.sin())
        })
        .collect();

    let area = 1000.0 * (n as f64);
    let k = (area / n as f64).sqrt();

    for iter in 0..iterations {
        // Temperature decreases over time
        let t = 50.0 * (1.0 - iter as f64 / iterations as f64);

        let mut forces = vec![(0.0, 0.0); n];

        // Repulsive forces between nodes
        if full_repulsion {
            // O(n^2) - only for small graphs
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = positions[j].0 - positions[i].0;
                    let dy = positions[j].1 - positions[i].1;
                    let dist_sq = dx * dx + dy * dy;
                    let dist = dist_sq.sqrt().max(0.1);

                    let repulsion = k * k / dist;
                    let fx = repulsion * dx / dist;
                    let fy = repulsion * dy / dist;

                    forces[i].0 -= fx;
                    forces[i].1 -= fy;
                    forces[j].0 += fx;
                    forces[j].1 += fy;
                }
            }
        } else {
            // Sampled repulsion for medium graphs - only check nearby nodes
            let sample_rate = (500.0 / n as f64).min(1.0).max(0.05);
            for i in 0..n {
                for j in (i + 1)..n {
                    if (i * 31 + j * 17) % 100 < (sample_rate * 100.0) as usize {
                        let dx = positions[j].0 - positions[i].0;
                        let dy = positions[j].1 - positions[i].1;
                        let dist = (dx * dx + dy * dy).sqrt().max(0.1);

                        let repulsion = k * k / dist;
                        let fx = repulsion * dx / dist;
                        let fy = repulsion * dy / dist;

                        forces[i].0 -= fx;
                        forces[i].1 -= fy;
                        forces[j].0 += fx;
                        forces[j].1 += fy;
                    }
                }
            }
        }

        // Attractive forces along edges
        for &(u, v) in edges {
            let dx = positions[v].0 - positions[u].0;
            let dy = positions[v].1 - positions[u].1;
            let dist = (dx * dx + dy * dy).sqrt().max(0.1);

            let attraction = dist * dist / k;
            let fx = attraction * dx / dist;
            let fy = attraction * dy / dist;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

        // Apply forces with cooling
        for i in 0..n {
            let force_mag = (forces[i].0 * forces[i].0 + forces[i].1 * forces[i].1).sqrt();
            if force_mag > 0.01 {
                let displacement = t.min(force_mag);
                positions[i].0 += forces[i].0 / force_mag * displacement;
                positions[i].1 += forces[i].1 / force_mag * displacement;
            }
        }
    }

    positions
}

fn spring_only_layout(n: usize, edges: &[(usize, usize)], iterations: usize) -> Vec<(f64, f64)> {
    // For very large graphs: O(m) spring embedding without global repulsion
    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for &(u, v) in edges {
        adj[u].push(v);
        adj[v].push(u);
    }

    // Initialize with BFS-based layout for better structure
    let mut positions = vec![(0.0, 0.0); n];
    let mut visited = vec![false; n];
    let mut component = 0;

    for start in 0..n {
        if !visited[start] {
            // BFS to lay out connected component
            let mut queue = vec![start];
            let mut level = vec![0; n];
            visited[start] = true;
            level[start] = 0;

            let mut qidx = 0;
            let mut max_level = 0;
            while qidx < queue.len() {
                let u = queue[qidx];
                qidx += 1;

                for &v in &adj[u] {
                    if !visited[v] {
                        visited[v] = true;
                        level[v] = level[u] + 1;
                        max_level = max_level.max(level[v]);
                        queue.push(v);
                    }
                }
            }

            // Position nodes by level
            let mut level_counts = vec![0; max_level + 1];
            for &node in &queue {
                level_counts[level[node]] += 1;
            }

            let mut level_idx = vec![0; max_level + 1];
            let comp_offset_x = (component % 10) as f64 * 500.0;
            let comp_offset_y = (component / 10) as f64 * 500.0;

            for &node in &queue {
                let lvl = level[node];
                let count = level_counts[lvl];
                let idx = level_idx[lvl];
                level_idx[lvl] += 1;

                let x = lvl as f64 * 50.0 + comp_offset_x;
                let y = if count > 1 {
                    (idx as f64 - (count as f64 - 1.0) / 2.0) * 30.0 + comp_offset_y
                } else {
                    comp_offset_y
                };

                positions[node] = (x, y);
            }

            component += 1;
        }
    }

    // Refine with spring forces only
    let k = 50.0;
    for iter in 0..iterations {
        let t = 20.0 * (1.0 - iter as f64 / iterations as f64);
        let mut forces = vec![(0.0, 0.0); n];

        for &(u, v) in edges {
            let dx = positions[v].0 - positions[u].0;
            let dy = positions[v].1 - positions[u].1;
            let dist = (dx * dx + dy * dy).sqrt().max(1.0);

            let attraction = (dist - k) * 0.1;
            let fx = attraction * dx / dist;
            let fy = attraction * dy / dist;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

        // Local repulsion among neighbors
        for i in 0..n {
            for &j in &adj[i] {
                if i < j {
                    let dx = positions[j].0 - positions[i].0;
                    let dy = positions[j].1 - positions[i].1;
                    let dist = (dx * dx + dy * dy).sqrt().max(0.1);

                    if dist < k * 2.0 {
                        let repulsion = k * 0.5 / dist;
                        let fx = repulsion * dx / dist;
                        let fy = repulsion * dy / dist;

                        forces[i].0 -= fx;
                        forces[i].1 -= fy;
                        forces[j].0 += fx;
                        forces[j].1 += fy;
                    }
                }
            }
        }

        for i in 0..n {
            let force_mag = (forces[i].0 * forces[i].0 + forces[i].1 * forces[i].1).sqrt();
            if force_mag > 0.01 {
                let displacement = t.min(force_mag);
                positions[i].0 += forces[i].0 / force_mag * displacement;
                positions[i].1 += forces[i].1 / force_mag * displacement;
            }
        }
    }

    positions
}
