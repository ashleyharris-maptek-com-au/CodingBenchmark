use std::io::{self, Read, Write, BufReader, BufWriter};

#[derive(Clone, Copy, Debug)]
struct Point3D {
    x: f32,
    y: f32,
    z: f32,
}

impl Point3D {
    fn distance_squared(&self, other: &Point3D) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    fn zero() -> Self {
        Point3D { x: 0.0, y: 0.0, z: 0.0 }
    }

    fn add(&mut self, other: &Point3D) {
        self.x += other.x;
        self.y += other.y;
        self.z += other.z;
    }

    fn scale(&mut self, factor: f32) {
        self.x *= factor;
        self.y *= factor;
        self.z *= factor;
    }
}

fn read_binary_point<R: Read>(reader: &mut R) -> io::Result<Point3D> {
    let mut buf = [0u8; 12];
    reader.read_exact(&mut buf)?;

    let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
    let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
    let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);

    Ok(Point3D { x, y, z })
}

fn write_binary_cluster<W: Write>(writer: &mut W, cluster: u32) -> io::Result<()> {
    writer.write_all(&cluster.to_le_bytes())
}

// Initialize centroids using k-means++ inspired approach
fn initialize_centroids(points: &[Point3D], k: usize) -> Vec<Point3D> {
    if points.is_empty() || k == 0 {
        return vec![Point3D::zero(); k.max(1)];
    }

    let mut centroids = Vec::with_capacity(k);
    let mut rng_state = 12345u64;

    let mut next_random = || {
        rng_state = rng_state.wrapping_mul(1664525).wrapping_add(1013904223);
        ((rng_state >> 16) as usize) % points.len()
    };

    // First centroid: random point
    centroids.push(points[next_random()]);

    // Remaining centroids: k-means++ style
    for _ in 1..k {
        let mut best_idx = 0;
        let mut best_dist = 0.0f32;

        let sample_size = points.len().min(5000);
        for _ in 0..sample_size {
            let idx = next_random();
            let point = &points[idx];

            let min_dist = centroids.iter()
                .map(|c| point.distance_squared(c))
                .min_by(|a, b| a.partial_cmp(b).unwrap())
                .unwrap_or(0.0);

            if min_dist > best_dist {
                best_dist = min_dist;
                best_idx = idx;
            }
        }

        centroids.push(points[best_idx]);
    }

    centroids
}

// Standard k-means for datasets that fit in memory
fn kmeans_in_memory(points: Vec<Point3D>, k: usize) -> Vec<u32> {
    let n = points.len();
    if n == 0 {
        return Vec::new();
    }
    if k == 0 {
        return vec![0u32; n];
    }

    let mut centroids = initialize_centroids(&points, k);
    let mut assignments = vec![0u32; n];
    let max_iterations = 100;

    for iteration in 0..max_iterations {
        // Assignment step
        let mut changed = false;
        for (i, point) in points.iter().enumerate() {
            let mut min_dist = f32::INFINITY;
            let mut best_cluster = 0u32;

            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    best_cluster = j as u32;
                }
            }

            if assignments[i] != best_cluster {
                changed = true;
                assignments[i] = best_cluster;
            }
        }

        if !changed && iteration > 5 {
            break;
        }

        // Update step
        let mut new_centroids = vec![Point3D::zero(); k];
        let mut counts = vec![0u32; k];

        for (i, point) in points.iter().enumerate() {
            let cluster = assignments[i] as usize;
            if cluster < k {
                new_centroids[cluster].add(point);
                counts[cluster] += 1;
            }
        }

        for i in 0..k {
            if counts[i] > 0 {
                new_centroids[i].scale(1.0 / counts[i] as f32);
                centroids[i] = new_centroids[i];
            }
        }

        // Handle empty clusters
        for i in 0..k {
            if counts[i] == 0 {
                let mut rng_state = (i as u64 + 1) * 54321;
                let idx = ((rng_state >> 16) as usize) % n;
                centroids[i] = points[idx];
            }
        }
    }

    assignments
}

// Mini-batch streaming k-means for large datasets
fn kmeans_streaming<R: Read>(reader: &mut R, num_points: usize, k: usize) -> io::Result<Vec<u32>> {
    if num_points == 0 || k == 0 {
        return Ok(vec![0u32; num_points]);
    }

    // Initialize centroids from first batch
    let init_size = k.max(1000).min(num_points);
    let mut init_points = Vec::with_capacity(init_size);
    for _ in 0..init_size {
        init_points.push(read_binary_point(reader)?);
    }

    let mut centroids = initialize_centroids(&init_points, k);
    let mut assignments = Vec::with_capacity(num_points);

    // Process initialization points
    let mut centroid_sums = vec![Point3D::zero(); k];
    let mut centroid_counts = vec![0u32; k];

    for point in &init_points {
        let mut min_dist = f32::INFINITY;
        let mut best_cluster = 0u32;

        for (j, centroid) in centroids.iter().enumerate() {
            let dist = point.distance_squared(centroid);
            if dist < min_dist {
                min_dist = dist;
                best_cluster = j as u32;
            }
        }

        assignments.push(best_cluster);
        let idx = best_cluster as usize;
        if idx < k {
            centroid_sums[idx].add(point);
            centroid_counts[idx] += 1;
        }
    }

    // Process remaining points in batches
    let batch_size = 50000;
    let mut points_read = init_size;
    let update_frequency = 500000;

    while points_read < num_points {
        let to_read = (num_points - points_read).min(batch_size);
        let mut batch = Vec::with_capacity(to_read);

        for _ in 0..to_read {
            match read_binary_point(reader) {
                Ok(point) => batch.push(point),
                Err(_) => break,
            }
        }

        if batch.is_empty() {
            break;
        }

        // Assign and update
        for point in &batch {
            let mut min_dist = f32::INFINITY;
            let mut best_cluster = 0u32;

            for (j, centroid) in centroids.iter().enumerate() {
                let dist = point.distance_squared(centroid);
                if dist < min_dist {
                    min_dist = dist;
                    best_cluster = j as u32;
                }
            }

            assignments.push(best_cluster);
            let idx = best_cluster as usize;
            if idx < k {
                centroid_sums[idx].add(point);
                centroid_counts[idx] += 1;
            }
        }

        points_read += batch.len();

        // Update centroids periodically
        if points_read % update_frequency == 0 || points_read == num_points {
            for i in 0..k {
                if centroid_counts[i] > 0 {
                    centroids[i] = centroid_sums[i];
                    centroids[i].scale(1.0 / centroid_counts[i] as f32);
                }
            }
        }
    }

    Ok(assignments)
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1024 * 1024, stdin.lock());

    // Read header (text format: "num_points num_clusters\n")
    let mut header = String::new();
    let mut byte = [0u8; 1];

    loop {
        reader.read_exact(&mut byte)?;
        if byte[0] == b'\n' {
            break;
        }
        header.push(byte[0] as char);
    }

    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    if parts.len() < 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidData, "Invalid header"));
    }

    let num_points: usize = parts[0].parse().unwrap_or(0);
    let num_clusters: usize = parts[1].parse().unwrap_or(1);

    // Adaptive strategy based on dataset size
    let assignments = if num_points <= 5_000_000 {
        // Small dataset: load into memory for better quality
        let mut points = Vec::with_capacity(num_points);
        for _ in 0..num_points {
            match read_binary_point(&mut reader) {
                Ok(p) => points.push(p),
                Err(_) => break,
            }
        }
        kmeans_in_memory(points, num_clusters)
    } else {
        // Large dataset: streaming approach
        kmeans_streaming(&mut reader, num_points, num_clusters)?
    };

    // Write binary output
    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(1024 * 1024, stdout.lock());

    for &assignment in &assignments {
        write_binary_cluster(&mut writer, assignment)?;
    }

    writer.flush()?;

    Ok(())
}
