const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { width, height, grid, ignitions, towns, cutBudget } = data;

function getKey(x, y) {
    return y * width + x;
}

function simulateFire(cutSet) {
    const burned = new Set();
    const queue = [];

    for (const [x, y] of ignitions) {
        const key = getKey(x, y);
        if (!cutSet.has(key) && grid[y][x] === '.') {
            burned.add(key);
            queue.push([x, y]);
        }
    }

    for (let i = 0; i < queue.length; i++) {
        const [x, y] = queue[i];

        for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
            const nx = x + dx;
            const ny = y + dy;

            if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                const key = getKey(nx, ny);
                if (!burned.has(key) && grid[ny][nx] === '.' && !cutSet.has(key)) {
                    burned.add(key);
                    queue.push([nx, ny]);
                }
            }
        }
    }

    return burned;
}

function calculateDistances(startPoints) {
    const dist = new Map();
    const queue = [];

    for (const [x, y] of startPoints) {
        if (grid[y][x] === '.') {
            const key = getKey(x, y);
            dist.set(key, 0);
            queue.push([x, y, 0]);
        }
    }

    for (let i = 0; i < queue.length; i++) {
        const [x, y, d] = queue[i];

        for (const [dx, dy] of [[0, 1], [0, -1], [1, 0], [-1, 0]]) {
            const nx = x + dx;
            const ny = y + dy;

            if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.') {
                const key = getKey(nx, ny);
                if (!dist.has(key)) {
                    dist.set(key, d + 1);
                    queue.push([nx, ny, d + 1]);
                }
            }
        }
    }

    return dist;
}

function findFirebreaks() {
    // Calculate distances from all ignition points
    const distFromIgnition = calculateDistances(ignitions);

    // Find which towns are at risk without any cuts
    const baselineBurned = simulateFire(new Set());
    const atRiskTowns = towns.filter(t => baselineBurned.has(getKey(t.x, t.y)));

    if (atRiskTowns.length === 0) {
        return [];
    }

    // Calculate impact score for each cell based on its position between ignitions and towns
    const impact = new Map();

    for (const town of atRiskTowns) {
        const distFromTown = calculateDistances([[town.x, town.y]]);

        for (const [key, dIgn] of distFromIgnition) {
            if (distFromTown.has(key)) {
                const dTown = distFromTown.get(key);
                // Cells that are on paths between ignition and high-value towns get higher scores
                const score = town.value / (1 + dIgn + dTown);
                impact.set(key, (impact.get(key) || 0) + score);
            }
        }
    }

    // Sort candidates by impact score and limit search space
    const candidates = Array.from(impact.entries())
        .sort((a, b) => b[1] - a[1])
        .slice(0, Math.min(cutBudget * 5, impact.size));

    // Greedy selection: iteratively choose the cut that saves the most value
    const cutSet = new Set();
    const firebreaks = [];
    let currentBurnedValue = atRiskTowns.reduce((sum, t) => sum + t.value, 0);

    for (let i = 0; i < cutBudget && candidates.length > 0; i++) {
        let bestIdx = -1;
        let bestValue = currentBurnedValue;

        // Test top candidates (limit to keep it scalable)
        const testLimit = Math.min(candidates.length, 100);
        for (let j = 0; j < testLimit; j++) {
            const [key] = candidates[j];

            cutSet.add(key);
            const burned = simulateFire(cutSet);
            const value = towns.reduce((sum, t) =>
                sum + (burned.has(getKey(t.x, t.y)) ? t.value : 0), 0);
            cutSet.delete(key);

            if (value < bestValue) {
                bestValue = value;
                bestIdx = j;
            }
        }

        if (bestIdx === -1) {
            break;
        }

        const [key] = candidates[bestIdx];
        const y = Math.floor(key / width);
        const x = key % width;
        firebreaks.push([x, y]);
        cutSet.add(key);
        candidates.splice(bestIdx, 1);
        currentBurnedValue = bestValue;

        if (bestValue === 0) {
            break;
        }
    }

    return firebreaks;
}

const firebreaks = findFirebreaks();
