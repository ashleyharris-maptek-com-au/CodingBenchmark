use std::io::{self, BufRead};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3] == "1";

    let mut grid = vec![vec![false; height]; width];
    let mut total_cells = width * height;

    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        grid[coords[0]][coords[1]] = true;
        total_cells -= 1;
    }

    if grid[0][0] {
        println!("0");
        return;
    }

    let mut solver = HamiltonianSolver {
        width,
        height,
        grid,
        require_cycle,
        total_cells,
        path: Vec::with_capacity(total_cells),
        visited: vec![vec![false; height]; width],
    };

    if solver.solve() {
        println!("{}", solver.path.len());
        for (x, y) in solver.path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

struct HamiltonianSolver {
    width: usize,
    height: usize,
    grid: Vec<Vec<bool>>,
    require_cycle: bool,
    total_cells: usize,
    path: Vec<(usize, usize)>,
    visited: Vec<Vec<bool>>,
}

impl HamiltonianSolver {
    fn solve(&mut self) -> bool {
        // Early checks
        if !self.is_connected() {
            return false;
        }

        self.visited[0][0] = true;
        self.path.push((0, 0));

        let result = self.backtrack();

        if !result {
            self.visited[0][0] = false;
            self.path.pop();
        }

        result
    }

    fn is_connected(&self) -> bool {
        let mut visited = vec![vec![false; self.height]; self.width];
        let mut stack = vec![(0, 0)];
        visited[0][0] = true;
        let mut count = 1;

        while let Some((x, y)) = stack.pop() {
            for (nx, ny) in self.get_neighbors(x, y) {
                if !visited[nx][ny] && !self.grid[nx][ny] {
                    visited[nx][ny] = true;
                    stack.push((nx, ny));
                    count += 1;
                }
            }
        }

        count == self.total_cells
    }

    fn backtrack(&mut self) -> bool {
        if self.path.len() == self.total_cells {
            if self.require_cycle {
                let (last_x, last_y) = self.path[self.path.len() - 1];
                return self.are_adjacent(last_x, last_y, 0, 0);
            }
            return true;
        }

        // Get current position
        let (curr_x, curr_y) = self.path[self.path.len() - 1];

        // Use Warnsdorff's heuristic: try neighbors with fewest unvisited neighbors first
        let mut neighbors: Vec<(usize, usize, usize)> = self.get_neighbors(curr_x, curr_y)
            .into_iter()
            .filter(|&(nx, ny)| !self.visited[nx][ny] && !self.grid[nx][ny])
            .map(|(nx, ny)| {
                let degree = self.count_unvisited_neighbors(nx, ny);
                (nx, ny, degree)
            })
            .collect();

        // Early pruning: if we're not at the last step and a neighbor has degree 0, it's a dead end
        if self.path.len() < self.total_cells - 1 {
            for &(_, _, degree) in &neighbors {
                if degree == 0 {
                    return false;
                }
            }
        }

        // Sort by degree (Warnsdorff's heuristic)
        neighbors.sort_by_key(|&(_, _, degree)| degree);

        for (nx, ny, _) in neighbors {
            self.visited[nx][ny] = true;
            self.path.push((nx, ny));

            // Check if remaining unvisited cells are still connected
            if self.are_remaining_connected() {
                if self.backtrack() {
                    return true;
                }
            }

            self.path.pop();
            self.visited[nx][ny] = false;
        }

        false
    }

    fn get_neighbors(&self, x: usize, y: usize) -> Vec<(usize, usize)> {
        let mut neighbors = Vec::with_capacity(4);
        let directions = [(0, -1), (0, 1), (-1, 0), (1, 0)];

        for (dx, dy) in directions {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;

            if nx >= 0 && nx < self.width as i32 && ny >= 0 && ny < self.height as i32 {
                neighbors.push((nx as usize, ny as usize));
            }
        }

        neighbors
    }

    fn count_unvisited_neighbors(&self, x: usize, y: usize) -> usize {
        self.get_neighbors(x, y)
            .into_iter()
            .filter(|&(nx, ny)| !self.visited[nx][ny] && !self.grid[nx][ny])
            .count()
    }

    fn are_adjacent(&self, x1: usize, y1: usize, x2: usize, y2: usize) -> bool {
        let dx = (x1 as i32 - x2 as i32).abs();
        let dy = (y1 as i32 - y2 as i32).abs();
        dx + dy == 1
    }

    fn are_remaining_connected(&self) -> bool {
        // Find first unvisited cell
        let mut start = None;
        'outer: for x in 0..self.width {
            for y in 0..self.height {
                if !self.visited[x][y] && !self.grid[x][y] {
                    start = Some((x, y));
                    break 'outer;
                }
            }
        }

        let start = match start {
            Some(s) => s,
            None => return true, // No unvisited cells
        };

        // BFS to check if all unvisited cells are connected
        let mut temp_visited = vec![vec![false; self.height]; self.width];
        let mut stack = vec![start];
        temp_visited[start.0][start.1] = true;
        let mut count = 1;

        // Count unvisited cells
        let unvisited_total: usize = (0..self.width)
            .flat_map(|x| (0..self.height).map(move |y| (x, y)))
            .filter(|&(x, y)| !self.visited[x][y] && !self.grid[x][y])
            .count();

        while let Some((x, y)) = stack.pop() {
            for (nx, ny) in self.get_neighbors(x, y) {
                if !self.visited[nx][ny] && !self.grid[nx][ny] && !temp_visited[nx][ny] {
                    temp_visited[nx][ny] = true;
                    stack.push((nx, ny));
                    count += 1;
                }
            }
        }

        count == unvisited_total
    }
}
