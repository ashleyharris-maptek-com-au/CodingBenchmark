#include <iostream>
#include <vector>
#include <climits>

using namespace std;

int n, m;
vector<long long> c;
vector<vector<int>> adj;
vector<int> best_solution;
long long best_value;

void backtrack(int idx, long long current_value, vector<int>& solution, vector<bool>& forbidden) {
    if (idx == n) {
        if (current_value > best_value) {
            best_value = current_value;
            best_solution = solution;
        }
        return;
    }

    // Calculate upper bound (current value + sum of remaining positive coefficients)
    long long upper_bound = current_value;
    for (int i = idx; i < n; i++) {
        if (!forbidden[i] && c[i] > 0) {
            upper_bound += c[i];
        }
    }

    // Prune if we can't improve best solution
    if (upper_bound <= best_value) {
        return;
    }

    // Try x[idx] = 0
    solution[idx] = 0;
    backtrack(idx + 1, current_value, solution, forbidden);

    // Try x[idx] = 1 (if not forbidden by previous selections)
    if (!forbidden[idx]) {
        solution[idx] = 1;

        // Mark future neighbors as forbidden
        vector<int> marked;
        for (int neighbor : adj[idx]) {
            if (neighbor > idx && !forbidden[neighbor]) {
                forbidden[neighbor] = true;
                marked.push_back(neighbor);
            }
        }

        backtrack(idx + 1, current_value + c[idx], solution, forbidden);

        // Restore forbidden status
        for (int neighbor : marked) {
            forbidden[neighbor] = false;
        }
    }
}

int main() {
    cin >> n >> m;
    c.resize(n);
    adj.resize(n);
    best_solution.resize(n, 0);
    best_value = LLONG_MIN;

    for (int i = 0; i < n; i++) {
        cin >> c[i];
    }

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> solution(n, 0);
    vector<bool> forbidden(n, false);
    backtrack(0, 0, solution, forbidden);

    cout << best_value << endl;
    for (int i = 0; i < n; i++) {
        cout << best_solution[i];
        if (i < n - 1) cout << " ";
    }
    cout << endl;

    return 0;
}
