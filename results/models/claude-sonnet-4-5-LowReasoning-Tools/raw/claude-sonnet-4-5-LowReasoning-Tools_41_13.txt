OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

; Constants
%float_0_5 = OpConstant %float 0.5
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%vec3_bottom = OpConstantComposite %vec3 %float_0_1 %float_0_4 %float_0_1
%vec3_top = OpConstantComposite %vec3 %float_1 %float_1 %float_1

; Input/Output variables
%worldPos = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Main function
%main = OpFunction %void None %func
%entry = OpLabel

; Load worldPos
%worldPos_val = OpLoad %vec3 %worldPos

; Extract Y component (index 1)
%y = OpCompositeExtract %float %worldPos_val 1

; Compute t = y * 0.5 + 0.5
%y_scaled = OpFMul %float %y %float_0_5
%t_unclamped = OpFAdd %float %y_scaled %float_0_5

; Clamp t to [0, 1]
%t = OpExtInst %float %glsl450 FClamp %t_unclamped %float_0 %float_1

; Mix bottom and top colors: bottom + t * (top - bottom)
%diff = OpFSub %vec3 %vec3_top %vec3_bottom
%t_vec3 = OpCompositeConstruct %vec3 %t %t %t
%scaled_diff = OpFMul %vec3 %t_vec3 %diff
%rgb = OpFAdd %vec3 %vec3_bottom %scaled_diff

; Extract RGB components
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2

; Construct output vec4 with alpha = 1.0
%result = OpCompositeConstruct %vec4 %r %g %b %float_1

; Store result
OpStore %outColor %result

OpReturn
OpFunctionEnd
