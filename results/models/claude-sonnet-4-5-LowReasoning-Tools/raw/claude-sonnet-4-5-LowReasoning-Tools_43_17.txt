[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Scale UV coordinates
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Determine brick row
    float row = floor(scaledUV.y);

    // Apply half-brick offset for odd rows
    float offset = (row - 2.0 * floor(row * 0.5)) * 0.5;
    float2 brickUV = float2(scaledUV.x + offset, scaledUV.y);

    // Get position within brick
    float2 brickPos = frac(brickUV);

    // Determine if we're in mortar
    float mortarX = step(brickPos.x, 0.05) + step(1.0 - 0.05, brickPos.x);
    float mortarY = step(brickPos.y, 0.1) + step(1.0 - 0.1, brickPos.y);
    float inMortar = clamp(mortarX + mortarY, 0.0, 1.0);

    // Brick and mortar colors
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);

    // Select base color
    float3 baseColor = lerp(brickColor, mortarColor, inMortar);

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;

    // Final color
    float3 finalColor = baseColor * shadingFactor;

    return float4(finalColor, 1.0);
}
