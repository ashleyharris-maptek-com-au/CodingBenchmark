using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;

class BinPacking3D
{
    static void Main()
    {
        // Parse input
        var input = Console.In.ReadToEnd().Split(new[] { '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int lineIdx = 0;

        var firstLine = input[lineIdx++].Split();
        int V = int.Parse(firstLine[0]);
        int F = int.Parse(firstLine[1]);

        Vector3[] vertices = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            var parts = input[lineIdx++].Split();
            vertices[i] = new Vector3(float.Parse(parts[0]), float.Parse(parts[1]), float.Parse(parts[2]));
        }

        int[][] faces = new int[F][];
        for (int i = 0; i < F; i++)
        {
            var parts = input[lineIdx++].Split();
            faces[i] = new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) };
        }

        var containerParts = input[lineIdx++].Split();
        float W = float.Parse(containerParts[0]);
        float H = float.Parse(containerParts[1]);
        float D = float.Parse(containerParts[2]);

        // Create solver and pack
        var solver = new Solver(vertices, faces, W, H, D);
        var placements = solver.Pack();

        // Output
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X} {p.Translation.Y} {p.Translation.Z} {p.Rotation.W} {p.Rotation.X} {p.Rotation.Y} {p.Rotation.Z}");
        }
    }
}

struct Placement
{
    public Vector3 Translation;
    public Quaternion Rotation;

    public Placement(Vector3 t, Quaternion r)
    {
        Translation = t;
        Rotation = r;
    }
}

class Solver
{
    Vector3[] baseVertices;
    int[][] faces;
    float W, H, D;
    Vector3 polyMin, polyMax;
    float polyDiameter;

    public Solver(Vector3[] vertices, int[][] faces, float w, float h, float d)
    {
        this.baseVertices = vertices;
        this.faces = faces;
        this.W = w;
        this.H = h;
        this.D = d;

        // Compute bounding box of base polyhedron
        polyMin = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
        polyMax = new Vector3(float.MinValue, float.MinValue, float.MinValue);
        foreach (var v in vertices)
        {
            polyMin = Vector3.Min(polyMin, v);
            polyMax = Vector3.Max(polyMax, v);
        }
        polyDiameter = Vector3.Distance(polyMin, polyMax);
    }

    public List<Placement> Pack()
    {
        var placements = new List<Placement>();
        float containerVolume = W * H * D;

        // Adaptive parameters based on container size
        float stepSize;
        List<Quaternion> rotations;

        if (containerVolume < 1000) // Trivial
        {
            stepSize = Math.Max(0.5f, polyDiameter * 0.3f);
            rotations = GenerateRotations(24);
        }
        else if (containerVolume < 10000) // Medium
        {
            stepSize = Math.Max(1.0f, polyDiameter * 0.5f);
            rotations = GenerateRotations(16);
        }
        else if (containerVolume < 1000000) // Large
        {
            stepSize = Math.Max(2.0f, polyDiameter * 0.7f);
            rotations = GenerateRotations(8);
        }
        else // Extreme
        {
            stepSize = Math.Max(5.0f, polyDiameter);
            rotations = GenerateRotations(4);
        }

        // Spatial grid for collision detection
        var grid = new SpatialGrid(W, H, D, stepSize * 2);

        // Try placements in a grid pattern
        for (float z = 0; z < D; z += stepSize)
        {
            for (float y = 0; y < H; y += stepSize)
            {
                for (float x = 0; x < W; x += stepSize)
                {
                    Vector3 pos = new Vector3(x, y, z);

                    foreach (var rot in rotations)
                    {
                        var transformed = TransformVertices(baseVertices, pos, rot);

                        if (IsInsideContainer(transformed) && !grid.Collides(transformed))
                        {
                            placements.Add(new Placement(pos, rot));
                            grid.Add(transformed);
                            goto nextPosition; // Found valid placement at this position
                        }
                    }
                    nextPosition:;
                }
            }
        }

        return placements;
    }

    Vector3[] TransformVertices(Vector3[] verts, Vector3 translation, Quaternion rotation)
    {
        var result = new Vector3[verts.Length];
        for (int i = 0; i < verts.Length; i++)
        {
            result[i] = Vector3.Transform(verts[i], rotation) + translation;
        }
        return result;
    }

    bool IsInsideContainer(Vector3[] verts)
    {
        foreach (var v in verts)
        {
            if (v.X < 0 || v.X > W || v.Y < 0 || v.Y > H || v.Z < 0 || v.Z > D)
                return false;
        }
        return true;
    }

    List<Quaternion> GenerateRotations(int count)
    {
        var rotations = new List<Quaternion>();

        if (count >= 24)
        {
            // Full set of 24 cube rotations
            for (int i = 0; i < 4; i++)
            {
                float angle = i * MathF.PI / 2;
                rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, angle));
                rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, angle));
                rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, angle));
            }
            // Add diagonal rotations
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.Normalize(new Vector3(1, 1, 0)), MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.Normalize(new Vector3(1, 0, 1)), MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.Normalize(new Vector3(0, 1, 1)), MathF.PI));
        }
        else if (count >= 16)
        {
            // 90-degree rotations around each axis
            for (int i = 0; i < 4; i++)
            {
                float angle = i * MathF.PI / 2;
                rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, angle));
                rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, angle));
                rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, angle));
            }
        }
        else if (count >= 8)
        {
            // Basic rotations
            rotations.Add(Quaternion.Identity);
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, 3 * MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI));
        }
        else
        {
            // Minimal rotations
            rotations.Add(Quaternion.Identity);
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, MathF.PI / 2));
            rotations.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, MathF.PI / 2));
        }

        return rotations.Take(count).ToList();
    }
}

class SpatialGrid
{
    Dictionary<(int, int, int), List<Vector3[]>> cells;
    float cellSize;
    float W, H, D;

    public SpatialGrid(float w, float h, float d, float cellSize)
    {
        this.W = w;
        this.H = h;
        this.D = d;
        this.cellSize = Math.Max(1.0f, cellSize);
        this.cells = new Dictionary<(int, int, int), List<Vector3[]>>();
    }

    public void Add(Vector3[] verts)
    {
        var bounds = GetBounds(verts);
        var cellsToAdd = GetCellsInBounds(bounds.min, bounds.max);

        foreach (var cell in cellsToAdd)
        {
            if (!cells.ContainsKey(cell))
                cells[cell] = new List<Vector3[]>();
            cells[cell].Add(verts);
        }
    }

    public bool Collides(Vector3[] verts)
    {
        var bounds = GetBounds(verts);
        var cellsToCheck = GetCellsInBounds(bounds.min, bounds.max);

        foreach (var cell in cellsToCheck)
        {
            if (cells.TryGetValue(cell, out var polys))
            {
                foreach (var poly in polys)
                {
                    if (PolygonsCollide(verts, poly))
                        return true;
                }
            }
        }
        return false;
    }

    (Vector3 min, Vector3 max) GetBounds(Vector3[] verts)
    {
        Vector3 min = new Vector3(float.MaxValue, float.MaxValue, float.MaxValue);
        Vector3 max = new Vector3(float.MinValue, float.MinValue, float.MinValue);
        foreach (var v in verts)
        {
            min = Vector3.Min(min, v);
            max = Vector3.Max(max, v);
        }
        return (min, max);
    }

    List<(int, int, int)> GetCellsInBounds(Vector3 min, Vector3 max)
    {
        var result = new List<(int, int, int)>();
        int minX = Math.Max(0, (int)(min.X / cellSize));
        int maxX = Math.Min((int)(W / cellSize), (int)(max.X / cellSize) + 1);
        int minY = Math.Max(0, (int)(min.Y / cellSize));
        int maxY = Math.Min((int)(H / cellSize), (int)(max.Y / cellSize) + 1);
        int minZ = Math.Max(0, (int)(min.Z / cellSize));
        int maxZ = Math.Min((int)(D / cellSize), (int)(max.Z / cellSize) + 1);

        for (int x = minX; x <= maxX; x++)
            for (int y = minY; y <= maxY; y++)
                for (int z = minZ; z <= maxZ; z++)
                    result.Add((x, y, z));

        return result;
    }

    bool PolygonsCollide(Vector3[] poly1, Vector3[] poly2)
    {
        // Simple bounding box check
        var b1 = GetBounds(poly1);
        var b2 = GetBounds(poly2);

        // Add small margin for numerical stability
        float margin = 0.01f;

        return !(b1.max.X + margin < b2.min.X || b1.min.X > b2.max.X + margin ||
                 b1.max.Y + margin < b2.min.Y || b1.min.Y > b2.max.Y + margin ||
                 b1.max.Z + margin < b2.min.Z || b1.min.Z > b2.max.Z + margin);
    }
}
