#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;

    // Guard against out of bounds
    if (i >= N) {
        return;
    }

    // Read current boid position and velocity
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Flocking forces
    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Check all other boids
    for (uint j = 0; j < N; j++) {
        // Skip self
        if (i == j) {
            continue;
        }

        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;

        vec3 diff = pos - otherPos;
        float dist = length(diff);

        // Skip zero-distance neighbors
        if (dist < 0.0001) {
            continue;
        }

        // Separation: steer away from nearby boids
        if (dist < sepRadius) {
            separation += diff / dist;  // Normalized direction away
            sepCount++;
        }

        // Alignment: match velocity of nearby boids
        if (dist < alignRadius) {
            alignment += otherVel;
            alignCount++;
        }

        // Cohesion: steer towards center of mass of nearby boids
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }

    // Average the forces by neighbor count (avoid division by zero)
    if (sepCount > 0) {
        separation /= float(sepCount);
    }
    if (alignCount > 0) {
        alignment /= float(alignCount);
        alignment = alignment - vel;  // Steering force
    }
    if (cohCount > 0) {
        cohesion /= float(cohCount);
        cohesion = cohesion - pos;  // Steering force towards center
    }

    // Apply weights
    vec3 accel = separation * sepWeight + alignment * alignWeight + cohesion * cohWeight;

    // Boundary force: push towards center when exceeding bounds
    vec3 boundAccel = vec3(0.0);
    if (abs(pos.x) > boundSize * 0.4) {
        boundAccel.x = -sign(pos.x) * boundForce * boundSize * 0.5;
    }
    if (abs(pos.y) > boundSize * 0.4) {
        boundAccel.y = -sign(pos.y) * boundForce * boundSize * 0.5;
    }
    if (abs(pos.z) > boundSize * 0.4) {
        boundAccel.z = -sign(pos.z) * boundForce * boundSize * 0.5;
    }
    accel += boundAccel;

    // Update velocity
    vel += accel * dt;

    // Clamp velocity to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = (vel / speed) * maxSpeed;
    }

    // Update position
    pos += vel * dt;

    // Write output
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
