%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4

%ptr_Function_int = OpTypePointer Function %int

%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0

%float_20 = OpConstant %float 20.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

%uv = OpVariable %ptr_Input_vec2 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%void_func = OpTypeFunction %void

%main = OpFunction %void None %void_func
%entry = OpLabel
%zr = OpVariable %ptr_Function_float Function
%zi = OpVariable %ptr_Function_float Function
%cr = OpVariable %ptr_Function_float Function
%ci = OpVariable %ptr_Function_float Function
%iter = OpVariable %ptr_Function_int Function

%uv_val = OpLoad %vec2 %uv
%u = OpCompositeExtract %float %uv_val 0
%v = OpCompositeExtract %float %uv_val 1

%u_times_3 = OpFMul %float %u %float_3
%cr_val = OpFSub %float %u_times_3 %float_2
%v_times_2 = OpFMul %float %v %float_2
%ci_val = OpFSub %float %v_times_2 %float_1

OpStore %cr %cr_val
OpStore %ci %ci_val
OpStore %zr %float_0
OpStore %zi %float_0
OpStore %iter %int_0

OpBranch %loop_header

%loop_header = OpLabel
%current_iter = OpLoad %int %iter
%continue_cond = OpSLessThan %bool %current_iter %int_20
OpLoopMerge %loop_merge %loop_continue None
OpBranchConditional %continue_cond %loop_body %loop_merge

%loop_body = OpLabel
%zr_val = OpLoad %float %zr
%zi_val = OpLoad %float %zi

%zr2_val = OpFMul %float %zr_val %zr_val
%zi2_val = OpFMul %float %zi_val %zi_val
%mag2_val = OpFAdd %float %zr2_val %zi2_val
%escaped = OpFOrdGreaterThan %bool %mag2_val %float_4
OpSelectionMerge %check_merge None
OpBranchConditional %escaped %loop_merge %check_merge

%check_merge = OpLabel
%zr_sq_minus_zi_sq = OpFSub %float %zr2_val %zi2_val
%cr_load = OpLoad %float %cr
%zr_new = OpFAdd %float %zr_sq_minus_zi_sq %cr_load

%two_zr_zi = OpFMul %float %zr_val %zi_val
%two_zr_zi_times_2 = OpFMul %float %float_2 %two_zr_zi
%ci_load = OpLoad %float %ci
%zi_new = OpFAdd %float %two_zr_zi_times_2 %ci_load

OpStore %zr %zr_new
OpStore %zi %zi_new

OpBranch %loop_continue

%loop_continue = OpLabel
%iter_val = OpLoad %int %iter
%iter_inc = OpIAdd %int %iter_val %int_1
OpStore %iter %iter_inc
OpBranch %loop_header

%loop_merge = OpLabel
%final_iter = OpLoad %int %iter
%final_iter_float = OpConvertSToF %float %final_iter
%t = OpFDiv %float %final_iter_float %float_20

%t_times_0_5 = OpFMul %float %t %float_0_5
%one_minus_t = OpFSub %float %float_1 %t

%color = OpCompositeConstruct %vec4 %t %t_times_0_5 %one_minus_t %float_1

OpStore %outColor %color

OpReturn
