#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <random>
#include <chrono>
#include <immintrin.h>

using namespace std;

struct City {
    float x, y;
};

// Fast distance calculation
inline float dist(const City& a, const City& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return sqrtf(dx * dx + dy * dy);
}

// Nearest neighbor construction
vector<int> nearestNeighbor(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int current = tour.back();
        int nearest = -1;
        float minDist = numeric_limits<float>::max();

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                float d = dist(cities[current], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// 2-opt improvement
void twoOpt(vector<int>& tour, const vector<City>& cities, int maxIterations) {
    int n = tour.size();
    bool improved = true;
    int iterations = 0;

    while (improved && iterations < maxIterations) {
        improved = false;
        iterations++;

        for (int i = 1; i < n - 2; i++) {
            for (int j = i + 1; j < n - 1; j++) {
                float d1 = dist(cities[tour[i-1]], cities[tour[i]]) +
                          dist(cities[tour[j]], cities[tour[j+1]]);
                float d2 = dist(cities[tour[i-1]], cities[tour[j]]) +
                          dist(cities[tour[i]], cities[tour[j+1]]);

                if (d2 < d1) {
                    reverse(tour.begin() + i, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

// Fast greedy construction for very large instances
vector<int> fastGreedy(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour;
    tour.reserve(n);
    vector<bool> visited(n, false);

    tour.push_back(0);
    visited[0] = true;

    // For very large instances, only check a subset of unvisited cities
    int checkLimit = min(1000, n);

    for (int i = 1; i < n; i++) {
        int current = tour.back();
        int nearest = -1;
        float minDist = numeric_limits<float>::max();

        int checked = 0;
        for (int j = 0; j < n && checked < checkLimit; j++) {
            if (!visited[j]) {
                float d = dist(cities[current], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
                checked++;
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// Spatial sorting for extreme cases
vector<int> spatialSort(const vector<City>& cities) {
    int n = cities.size();
    vector<int> tour(n);
    for (int i = 0; i < n; i++) tour[i] = i;

    // Find bounding box
    float minX = cities[0].x, maxX = cities[0].x;
    float minY = cities[0].y, maxY = cities[0].y;
    for (const auto& c : cities) {
        minX = min(minX, c.x);
        maxX = max(maxX, c.x);
        minY = min(minY, c.y);
        maxY = max(maxY, c.y);
    }

    // Space-filling curve approximation: sort by Hilbert-like order
    auto hilbertCmp = [&](int a, int b) {
        const City& ca = cities[a];
        const City& cb = cities[b];

        // Normalize coordinates
        float nx_a = (ca.x - minX) / (maxX - minX + 1e-9f);
        float ny_a = (ca.y - minY) / (maxY - minY + 1e-9f);
        float nx_b = (cb.x - minX) / (maxX - minX + 1e-9f);
        float ny_b = (cb.y - minY) / (maxY - minY + 1e-9f);

        // Simple angle-based sorting from center
        float angle_a = atan2f(ny_a - 0.5f, nx_a - 0.5f);
        float angle_b = atan2f(ny_b - 0.5f, nx_b - 0.5f);
        return angle_a < angle_b;
    };

    sort(tour.begin(), tour.end(), hilbertCmp);

    // Ensure tour starts with 0
    auto it = find(tour.begin(), tour.end(), 0);
    rotate(tour.begin(), it, tour.end());

    return tour;
}

// Adaptive TSP solver
vector<int> solveTSP(const vector<City>& cities) {
    int n = cities.size();

    if (n <= 1) {
        return vector<int>{0};
    }

    // Adaptive strategy based on problem size
    if (n <= 10) {
        // Trivial: use nearest neighbor with full 2-opt
        auto tour = nearestNeighbor(cities);
        twoOpt(tour, cities, 100);
        return tour;
    }
    else if (n <= 1000) {
        // Small-medium: nearest neighbor with 2-opt
        auto tour = nearestNeighbor(cities);
        int maxIter = max(1, 50 - n / 50);
        twoOpt(tour, cities, maxIter);
        return tour;
    }
    else if (n <= 10000) {
        // Medium: nearest neighbor with limited 2-opt
        auto tour = nearestNeighbor(cities);
        twoOpt(tour, cities, 5);
        return tour;
    }
    else if (n <= 100000) {
        // Large: nearest neighbor only
        return nearestNeighbor(cities);
    }
    else if (n <= 1000000) {
        // Very large: fast greedy
        return fastGreedy(cities);
    }
    else {
        // Extreme: spatial sorting
        return spatialSort(cities);
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<City> cities(n);
    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> tour = solveTSP(cities);

    // Output tour
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
