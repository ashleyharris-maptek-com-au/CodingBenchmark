use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read first line: num_vertices num_edges
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().unwrap())
        .collect();
    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Build adjacency list
    let mut adj: Vec<HashSet<usize>> = vec![HashSet::new(); num_vertices];

    for _ in 0..num_edges {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line
            .split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        let u = edge[0];
        let v = edge[1];
        adj[u].insert(v);
        adj[v].insert(u);
    }

    // Greedy algorithm for minimum dominating set
    let mut dominating_set: Vec<usize> = Vec::new();
    let mut covered: Vec<bool> = vec![false; num_vertices];

    while covered.iter().any(|&c| !c) {
        // Find vertex that covers the most uncovered vertices
        let mut best_vertex = 0;
        let mut best_count = 0;

        for v in 0..num_vertices {
            let mut count = 0;

            // Count how many uncovered vertices this would cover
            if !covered[v] {
                count += 1;
            }
            for &neighbor in &adj[v] {
                if !covered[neighbor] {
                    count += 1;
                }
            }

            if count > best_count {
                best_count = count;
                best_vertex = v;
            }
        }

        // Add best vertex to dominating set
        dominating_set.push(best_vertex);

        // Mark vertex and its neighbors as covered
        covered[best_vertex] = true;
        for &neighbor in &adj[best_vertex] {
            covered[neighbor] = true;
        }
    }

    // Output: set_size on first line, vertices on second line
    println!("{}", dominating_set.len());
    for (i, &v) in dominating_set.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
