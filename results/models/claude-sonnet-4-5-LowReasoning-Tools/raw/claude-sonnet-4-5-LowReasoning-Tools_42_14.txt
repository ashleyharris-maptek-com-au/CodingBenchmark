OpCapability Geometry
               OpExtension "SPV_KHR_storage_buffer_storage_class"
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

               ; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
     %uint_6 = OpConstant %uint 6

  %float_0_2 = OpConstant %float 0.2
  %float_0_8 = OpConstant %float 0.8
    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
  %float_1_3 = OpConstant %float 0.333333343

      %v3f_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1

               ; Input types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in_6 = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in_6
       %gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_v4float = OpTypePointer Input %v4float

%arr_v3f_in_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3f_in_6
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
 %normal_in = OpVariable %ptr_in_arr_v3 Input
  %color_in = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3float = OpTypePointer Input %v3float

               ; Output types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
      %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output

               ; Function types
%ptr_func_v3float = OpTypePointer Function %v3float

               ; Main function
       %main = OpFunction %void None %3
          %5 = OpLabel

               ; Load positions for main triangle (indices 0, 2, 4)
       %p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
       %p0_v4 = OpLoad %v4float %p0_ptr
       %p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2

       %p2_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
       %p2_v4 = OpLoad %v4float %p2_ptr
       %p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

       %p4_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
       %p4_v4 = OpLoad %v4float %p4_ptr
       %p4 = OpVectorShuffle %v3float %p4_v4 %p4_v4 0 1 2

               ; Compute edge vectors: e1 = p2 - p0, e2 = p4 - p0
        %e1 = OpFSub %v3float %p2 %p0
        %e2 = OpFSub %v3float %p4 %p0

               ; Extract components for cross product
      %e1_x = OpCompositeExtract %float %e1 0
      %e1_y = OpCompositeExtract %float %e1 1
      %e1_z = OpCompositeExtract %float %e1 2
      %e2_x = OpCompositeExtract %float %e2 0
      %e2_y = OpCompositeExtract %float %e2 1
      %e2_z = OpCompositeExtract %float %e2 2

               ; Cross product: cross(e1, e2) = (e1.y*e2.z - e1.z*e2.y, e1.z*e2.x - e1.x*e2.z, e1.x*e2.y - e1.y*e2.x)
   %cross_x_1 = OpFMul %float %e1_y %e2_z
   %cross_x_2 = OpFMul %float %e1_z %e2_y
     %cross_x = OpFSub %float %cross_x_1 %cross_x_2

   %cross_y_1 = OpFMul %float %e1_z %e2_x
   %cross_y_2 = OpFMul %float %e1_x %e2_z
     %cross_y = OpFSub %float %cross_y_1 %cross_y_2

   %cross_z_1 = OpFMul %float %e1_x %e2_y
   %cross_z_2 = OpFMul %float %e1_y %e2_x
     %cross_z = OpFSub %float %cross_z_1 %cross_z_2

 %face_normal_unnorm = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

               ; Normalize face normal
%face_normal = OpExtInst %v3float %1 Normalize %face_normal_unnorm

               ; Light direction (normalized)
   %light_dir = OpExtInst %v3float %1 Normalize %v3f_1

               ; Compute diffuse: max(dot(N, L), 0.0)
        %dot = OpDot %float %face_normal %light_dir
    %diffuse = OpExtInst %float %1 FMax %dot %float_0

               ; Load colors from indices 0, 2, 4
   %c0_ptr = OpAccessChain %ptr_in_v3float %color_in %int_0
       %c0 = OpLoad %v3float %c0_ptr
   %c2_ptr = OpAccessChain %ptr_in_v3float %color_in %int_2
       %c2 = OpLoad %v3float %c2_ptr
   %c4_ptr = OpAccessChain %ptr_in_v3float %color_in %int_4
       %c4 = OpLoad %v3float %c4_ptr

               ; Average color: (c0 + c2 + c4) / 3
  %c_sum_01 = OpFAdd %v3float %c0 %c2
  %c_sum_all = OpFAdd %v3float %c_sum_01 %c4
 %base_color = OpVectorTimesScalar %v3float %c_sum_all %float_1_3

               ; Final color: base * (0.2 + 0.8 * diffuse)
%diffuse_scaled = OpFMul %float %float_0_8 %diffuse
  %intensity = OpFAdd %float %float_0_2 %diffuse_scaled
%final_color = OpVectorTimesScalar %v3float %base_color %intensity

               ; Emit vertex 0
  %out_pos_0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_pos_0 %p0_v4
               OpStore %outColor %final_color
               OpEmitVertex

               ; Emit vertex 2
  %out_pos_2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_pos_2 %p2_v4
               OpStore %outColor %final_color
               OpEmitVertex

               ; Emit vertex 4
  %out_pos_4 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %out_pos_4 %p4_v4
               OpStore %outColor %final_color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
