#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <limits>
#include <unordered_set>

using namespace std;

const int INF = 1e9;

struct Edge {
    int to, weight;
};

int n, m, k;
vector<vector<Edge>> graph;
vector<int> terminals;

// Dijkstra's algorithm from a single source
vector<pair<int, int>> dijkstra(int start) {
    vector<int> dist(n, INF);
    vector<int> parent(n, -1);
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;

    dist[start] = 0;
    pq.push({0, start});

    while (!pq.empty()) {
        auto [d, u] = pq.top();
        pq.pop();

        if (d > dist[u]) continue;

        for (auto& e : graph[u]) {
            if (dist[u] + e.weight < dist[e.to]) {
                dist[e.to] = dist[u] + e.weight;
                parent[e.to] = u;
                pq.push({dist[e.to], e.to});
            }
        }
    }

    vector<pair<int, int>> result(n);
    for (int i = 0; i < n; i++) {
        result[i] = {dist[i], parent[i]};
    }
    return result;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> n >> m >> k;

    graph.resize(n);

    for (int i = 0; i < m; i++) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].push_back({v, w});
        graph[v].push_back({u, w});
    }

    terminals.resize(k);
    for (int i = 0; i < k; i++) {
        cin >> terminals[i];
    }

    // Compute shortest paths from all terminals
    vector<vector<pair<int, int>>> terminalPaths(k);
    for (int i = 0; i < k; i++) {
        terminalPaths[i] = dijkstra(terminals[i]);
    }

    // Build MST on terminal complete graph using Prim's algorithm
    vector<bool> inMST(k, false);
    vector<int> minDist(k, INF);
    vector<int> mstParent(k, -1);

    minDist[0] = 0;

    for (int count = 0; count < k; count++) {
        int u = -1;
        for (int v = 0; v < k; v++) {
            if (!inMST[v] && (u == -1 || minDist[v] < minDist[u])) {
                u = v;
            }
        }

        inMST[u] = true;

        for (int v = 0; v < k; v++) {
            if (!inMST[v]) {
                int dist = terminalPaths[u][terminals[v]].first;
                if (dist < minDist[v]) {
                    minDist[v] = dist;
                    mstParent[v] = u;
                }
            }
        }
    }

    // Collect all edges from shortest paths in the MST
    unordered_set<long long> edgeSet;

    for (int i = 1; i < k; i++) {
        int termU = terminals[mstParent[i]];
        int termV = terminals[i];

        // Trace back shortest path from termV to termU
        int curr = termV;
        while (curr != termU && terminalPaths[mstParent[i]][curr].second != -1) {
            int prev = terminalPaths[mstParent[i]][curr].second;

            long long edge = ((long long)min(prev, curr) << 32) | max(prev, curr);
            edgeSet.insert(edge);

            curr = prev;
        }
    }

    // Convert edge set to vector and compute total weight
    vector<pair<int, int>> resultEdges;
    int totalWeight = 0;

    for (long long e : edgeSet) {
        int u = (int)(e >> 32);
        int v = (int)(e & 0xFFFFFFFF);
        resultEdges.push_back({u, v});

        // Find edge weight
        for (auto& edge : graph[u]) {
            if (edge.to == v) {
                totalWeight += edge.weight;
                break;
            }
        }
    }

    // Output result
    cout << totalWeight << " " << resultEdges.size() << "\n";
    for (auto [u, v] : resultEdges) {
        cout << u << " " << v << "\n";
    }

    return 0;
}
