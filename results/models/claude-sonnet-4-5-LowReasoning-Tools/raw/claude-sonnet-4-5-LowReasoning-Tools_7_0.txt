const double EPS = 1e-9;

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double t) const { return Vec3(x * t, y * t, z * t); }
    Vec3 operator/(double t) const { return Vec3(x / t, y / t, z / t); }

    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const {
        return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x);
    }

    double length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const {
        double len = length();
        return len > EPS ? (*this / len) : Vec3(0, 0, 0);
    }
};

struct Triangle {
    int v[3];
    int meshId;
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Triangle> faces;
    int id;
};

Vec3 computeNormal(const Vec3& a, const Vec3& b, const Vec3& c) {
    return (b - a).cross(c - a).normalize();
}

double signedDistance(const Vec3& point, const Vec3& planePoint, const Vec3& normal) {
    return (point - planePoint).dot(normal);
}

bool isPointInTriangle(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c, const Vec3& normal) {
    Vec3 v0 = c - a, v1 = b - a, v2 = p - a;
    double d00 = v0.dot(v0);
    double d01 = v0.dot(v1);
    double d11 = v1.dot(v1);
    double d20 = v2.dot(v0);
    double d21 = v2.dot(v1);
    double denom = d00 * d11 - d01 * d01;
    if (abs(denom) < EPS) return false;
    double v = (d11 * d20 - d01 * d21) / denom;
    double w = (d00 * d21 - d01 * d20) / denom;
    double u = 1.0 - v - w;
    return (u >= -EPS && v >= -EPS && w >= -EPS);
}

bool rayIntersectsTriangle(const Vec3& origin, const Vec3& dir,
                          const Vec3& a, const Vec3& b, const Vec3& c, double& t) {
    Vec3 edge1 = b - a;
    Vec3 edge2 = c - a;
    Vec3 h = dir.cross(edge2);
    double det = edge1.dot(h);

    if (abs(det) < EPS) return false;

    double invDet = 1.0 / det;
    Vec3 s = origin - a;
    double u = invDet * s.dot(h);

    if (u < -EPS || u > 1.0 + EPS) return false;

    Vec3 q = s.cross(edge1);
    double v = invDet * dir.dot(q);

    if (v < -EPS || u + v > 1.0 + EPS) return false;

    t = invDet * edge2.dot(q);
    return t > EPS;
}

bool isPointInsideMesh(const Vec3& point, const Mesh& mesh) {
    Vec3 rayDir(0.7071, 0.5773, 0.4082);
    int intersections = 0;

    for (const auto& face : mesh.faces) {
        const Vec3& a = mesh.vertices[face.v[0]];
        const Vec3& b = mesh.vertices[face.v[1]];
        const Vec3& c = mesh.vertices[face.v[2]];

        double t;
        if (rayIntersectsTriangle(point, rayDir, a, b, c, t)) {
            intersections++;
        }
    }

    return (intersections % 2) == 1;
}

Vec3 triangleCenter(const Vec3& a, const Vec3& b, const Vec3& c) {
    return (a + b + c) / 3.0;
}

void computeCSGUnion(const Mesh& meshA, const Mesh& meshB, Mesh& result) {
    result.id = 2;

    map<array<double, 3>, int> vertexMap;
    auto addVertex = [&](const Vec3& v) {
        array<double, 3> key = {v.x, v.y, v.z};
        auto it = vertexMap.find(key);
        if (it != vertexMap.end()) {
            return it->second;
        }
        int idx = result.vertices.size();
        result.vertices.push_back(v);
        vertexMap[key] = idx;
        return idx;
    };

    for (const auto& face : meshA.faces) {
        const Vec3& a = meshA.vertices[face.v[0]];
        const Vec3& b = meshA.vertices[face.v[1]];
        const Vec3& c = meshA.vertices[face.v[2]];
        Vec3 center = triangleCenter(a, b, c);

        if (!isPointInsideMesh(center, meshB)) {
            Triangle newFace;
            newFace.v[0] = addVertex(a);
            newFace.v[1] = addVertex(b);
            newFace.v[2] = addVertex(c);
            newFace.meshId = 0;
            result.faces.push_back(newFace);
        }
    }

    for (const auto& face : meshB.faces) {
        const Vec3& a = meshB.vertices[face.v[0]];
        const Vec3& b = meshB.vertices[face.v[1]];
        const Vec3& c = meshB.vertices[face.v[2]];
        Vec3 center = triangleCenter(a, b, c);

        if (!isPointInsideMesh(center, meshA)) {
            Triangle newFace;
            newFace.v[0] = addVertex(a);
            newFace.v[1] = addVertex(b);
            newFace.v[2] = addVertex(c);
            newFace.meshId = 1;
            result.faces.push_back(newFace);
        }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;
    meshA.id = 0;
    meshB.id = 1;

    meshA.vertices.resize(VA);
    for (int i = 0; i < VA; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }

    meshA.faces.resize(FA);
    for (int i = 0; i < FA; i++) {
        cin >> meshA.faces[i].v[0] >> meshA.faces[i].v[1] >> meshA.faces[i].v[2];
        meshA.faces[i].meshId = 0;
    }

    meshB.vertices.resize(VB);
    for (int i = 0; i < VB; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }

    meshB.faces.resize(FB);
    for (int i = 0; i < FB; i++) {
        cin >> meshB.faces[i].v[0] >> meshB.faces[i].v[1] >> meshB.faces[i].v[2];
        meshB.faces[i].meshId = 1;
    }

    Mesh result;
    computeCSGUnion(meshA, meshB, result);

    cout << result.vertices.size() << " " << result.faces.size() << "\n";

    cout.precision(10);
    cout << fixed;
    for (const auto& v : result.vertices) {
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : result.faces) {
        cout << f.v[0] << " " << f.v[1] << " " << f.v[2] << "\n";
    }

    return 0;
}
