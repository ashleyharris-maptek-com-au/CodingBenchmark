using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using System.Diagnostics;

class MaximumClique
{
    static int numVertices;
    static bool[][] adjMatrix;
    static int[] degrees;
    static List<int> bestClique;
    static object lockObj = new object();
    static Stopwatch timer;
    static readonly int TIMEOUT_MS = 280000;

    static void Main()
    {
        timer = Stopwatch.StartNew();

        string[] firstLine = Console.ReadLine().Split();
        numVertices = int.Parse(firstLine[0]);
        int numEdges = int.Parse(firstLine[1]);

        adjMatrix = new bool[numVertices][];
        for (int i = 0; i < numVertices; i++)
            adjMatrix[i] = new bool[numVertices];

        degrees = new int[numVertices];

        for (int i = 0; i < numEdges; i++)
        {
            string[] edge = Console.ReadLine().Split();
            int u = int.Parse(edge[0]);
            int v = int.Parse(edge[1]);
            adjMatrix[u][v] = true;
            adjMatrix[v][u] = true;
            degrees[u]++;
            degrees[v]++;
        }

        bestClique = new List<int>();

        if (numVertices <= 10000)
        {
            SolveExact();
        }
        else
        {
            SolveHeuristic();
        }

        Console.WriteLine(bestClique.Count);
        Console.WriteLine(string.Join(" ", bestClique.OrderBy(x => x)));
    }

    static void SolveExact()
    {
        var vertices = Enumerable.Range(0, numVertices)
            .OrderByDescending(v => degrees[v])
            .ToArray();

        var candidates = new HashSet<int>(vertices);
        var currentClique = new List<int>();

        if (numVertices <= 1000)
        {
            int threadCount = Math.Min(12, Math.Max(1, vertices.Length / 10));
            var tasks = new List<Task>();

            for (int t = 0; t < threadCount && t < vertices.Length; t++)
            {
                int startVertex = vertices[t];
                tasks.Add(Task.Run(() =>
                {
                    var localClique = new List<int> { startVertex };
                    var localCandidates = new HashSet<int>(
                        Enumerable.Range(0, numVertices)
                            .Where(v => v != startVertex && adjMatrix[startVertex][v])
                    );
                    BranchAndBound(localClique, localCandidates);
                }));
            }

            Task.WaitAll(tasks.ToArray());
        }
        else
        {
            BranchAndBound(currentClique, candidates);
        }

        if (bestClique.Count <= 2)
        {
            SolveHeuristic();
        }
    }

    static void BranchAndBound(List<int> currentClique, HashSet<int> candidates)
    {
        if (timer.ElapsedMilliseconds > TIMEOUT_MS)
            return;

        lock (lockObj)
        {
            if (currentClique.Count > bestClique.Count)
            {
                bestClique = new List<int>(currentClique);
            }
        }

        int currentBest;
        lock (lockObj) { currentBest = bestClique.Count; }

        if (currentClique.Count + candidates.Count <= currentBest)
            return;

        var candidateList = candidates.OrderByDescending(v => degrees[v]).ToList();

        foreach (var v in candidateList)
        {
            if (timer.ElapsedMilliseconds > TIMEOUT_MS)
                return;

            bool isConnected = currentClique.All(u => adjMatrix[u][v]);

            if (isConnected)
            {
                var newClique = new List<int>(currentClique) { v };

                var newCandidates = new HashSet<int>(
                    candidates.Where(c => c != v && adjMatrix[v][c] &&
                                     currentClique.All(u => adjMatrix[u][c]))
                );

                BranchAndBound(newClique, newCandidates);
            }
        }
    }

    static void SolveHeuristic()
    {
        var vertices = Enumerable.Range(0, numVertices)
            .OrderByDescending(v => degrees[v])
            .ToArray();

        var clique = new List<int>();

        foreach (var v in vertices)
        {
            if (clique.All(u => adjMatrix[u][v]))
            {
                clique.Add(v);
            }
        }

        lock (lockObj)
        {
            if (clique.Count > bestClique.Count)
            {
                bestClique = clique;
            }
        }

        int attempts = Math.Min(100, numVertices);
        Parallel.For(0, attempts, i =>
        {
            if (timer.ElapsedMilliseconds > TIMEOUT_MS)
                return;

            var localClique = new List<int> { vertices[i % vertices.Length] };
            var candidates = Enumerable.Range(0, numVertices)
                .Where(v => v != localClique[0] && adjMatrix[localClique[0]][v])
                .OrderByDescending(v => degrees[v])
                .ToList();

            foreach (var v in candidates)
            {
                if (localClique.All(u => adjMatrix[u][v]))
                {
                    localClique.Add(v);
                }
            }

            lock (lockObj)
            {
                if (localClique.Count > bestClique.Count)
                {
                    bestClique = new List<int>(localClique);
                }
            }
        });
    }
}
