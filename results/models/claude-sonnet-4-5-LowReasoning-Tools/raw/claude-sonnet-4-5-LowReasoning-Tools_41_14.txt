; SPIR-V
; Version: 1.0
; Generator: Khronos; 0
; Bound: 100
; Schema: 0

OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_vec4 = OpTypePointer Uniform %vec4

%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4

%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3

%baseColor = OpConstantComposite %vec3 %float_0_8 %float_0_2 %float_0_2
%fogColor = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_8

; Variables
%worldPos = OpVariable %ptr_input_vec3 Input
%normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

; Main function
%main = OpFunction %void None %void_func
%entry = OpLabel

; Load worldPos
%worldPos_val = OpLoad %vec3 %worldPos

; Load cameraPos from UBO (member 4, offset 208)
%cameraPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_4
%cameraPos_vec4 = OpLoad %vec4 %cameraPos_ptr
%cameraPos_x = OpCompositeExtract %float %cameraPos_vec4 0
%cameraPos_y = OpCompositeExtract %float %cameraPos_vec4 1
%cameraPos_z = OpCompositeExtract %float %cameraPos_vec4 2
%cameraPos = OpCompositeConstruct %vec3 %cameraPos_x %cameraPos_y %cameraPos_z

; Calculate distance from camera to worldPos
%toCamera = OpFSub %vec3 %worldPos_val %cameraPos
%dist = OpExtInst %float %1 Length %toCamera

; Calculate fog factor: clamp((dist - 1.5) / 3.0, 0, 1)
%dist_minus_1_5 = OpFSub %float %dist %float_1_5
%fog_raw = OpFDiv %float %dist_minus_1_5 %float_3
%fogFactor = OpExtInst %float %1 FClamp %fog_raw %float_0 %float_1

; Load and normalize normal
%normal_val = OpLoad %vec3 %normal
%normal_norm = OpExtInst %vec3 %1 Normalize %normal_val

; Load lightPos from UBO (member 3, offset 192)
%lightPos_ptr = OpAccessChain %ptr_uniform_vec4 %ubo %int_3
%lightPos_vec4 = OpLoad %vec4 %lightPos_ptr
%lightPos_x = OpCompositeExtract %float %lightPos_vec4 0
%lightPos_y = OpCompositeExtract %float %lightPos_vec4 1
%lightPos_z = OpCompositeExtract %float %lightPos_vec4 2
%lightPos = OpCompositeConstruct %vec3 %lightPos_x %lightPos_y %lightPos_z

; Calculate light direction
%toLight = OpFSub %vec3 %lightPos %worldPos_val
%lightDir = OpExtInst %vec3 %1 Normalize %toLight

; Calculate diffuse lighting: max(dot(normal, lightDir), 0)
%diffuse_raw = OpDot %float %normal_norm %lightDir
%diffuse = OpExtInst %float %1 FMax %diffuse_raw %float_0

; Calculate Lambertian factor: 0.3 + 0.7 * diffuse
%diffuse_scaled = OpFMul %float %float_0_7 %diffuse
%lambertian = OpFAdd %float %float_0_3 %diffuse_scaled

; Apply Lambertian factor to base color
%baseColor_lit = OpVectorTimesScalar %vec3 %baseColor %lambertian

; Mix base color with fog color using fog factor
; result = baseColor_lit * (1 - fogFactor) + fogColor * fogFactor
%one_minus_fog = OpFSub %float %float_1 %fogFactor
%baseColor_weighted = OpVectorTimesScalar %vec3 %baseColor_lit %one_minus_fog
%fogColor_weighted = OpVectorTimesScalar %vec3 %fogColor %fogFactor
%finalColor_rgb = OpFAdd %vec3 %baseColor_weighted %fogColor_weighted

; Extract RGB components and construct vec4 with alpha 1.0
%final_r = OpCompositeExtract %float %finalColor_rgb 0
%final_g = OpCompositeExtract %float %finalColor_rgb 1
%final_b = OpCompositeExtract %float %finalColor_rgb 2
%finalColor = OpCompositeConstruct %vec4 %final_r %final_g %final_b %float_1

; Store output
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
