using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Solution
{
    static int N;
    static List<int> edgeU, edgeV;
    static List<long> edgeW;
    static List<int>[] adj;
    static long[] tmpDist;
    static int[] tmpPred;
    static List<int> touched;
    static PriorityQueue<int, long> sharedPQ;

    public static void Main()
    {
        // Fast read all stdin into a byte buffer
        var sin = Console.OpenStandardInput();
        var ms = new MemoryStream();
        sin.CopyTo(ms);
        byte[] data = ms.ToArray();
        int pos = 0;
        int dlen = data.Length;

        int ReadInt()
        {
            while (pos < dlen && (data[pos] < (byte)'0' || data[pos] > (byte)'9') && data[pos] != (byte)'-') pos++;
            bool neg = false;
            if (pos < dlen && data[pos] == (byte)'-') { neg = true; pos++; }
            int v = 0;
            while (pos < dlen && data[pos] >= (byte)'0' && data[pos] <= (byte)'9')
            {
                v = v * 10 + (data[pos] - (byte)'0');
                pos++;
            }
            return neg ? -v : v;
        }

        long ReadLong()
        {
            while (pos < dlen && (data[pos] < (byte)'0' || data[pos] > (byte)'9') && data[pos] != (byte)'-') pos++;
            bool neg = false;
            if (pos < dlen && data[pos] == (byte)'-') { neg = true; pos++; }
            long v = 0;
            while (pos < dlen && data[pos] >= (byte)'0' && data[pos] <= (byte)'9')
            {
                v = v * 10L + (data[pos] - (byte)'0');
                pos++;
            }
            return neg ? -v : v;
        }

        N = ReadInt();
        int M = ReadInt();

        int cap = Math.Max(M * 2 + 16, 16);
        edgeU = new List<int>(cap);
        edgeV = new List<int>(cap);
        edgeW = new List<long>(cap);
        adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();

        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            int u = ReadInt();
            int v = ReadInt();
            long w = ReadLong();
            edgeU.Add(u);
            edgeV.Add(v);
            edgeW.Add(w);
            adj[u].Add(i);
            adj[v].Add(i);
            deg[u]++;
            deg[v]++;
        }

        // Find odd-degree vertices
        var oddVerts = new List<int>();
        for (int i = 0; i < N; i++) if ((deg[i] & 1) == 1) oddVerts.Add(i);

        int k = oddVerts.Count;
        if (k > 0)
        {
            tmpDist = new long[N];
            tmpPred = new int[N];
            for (int i = 0; i < N; i++) { tmpDist[i] = long.MaxValue; tmpPred[i] = -1; }
            touched = new List<int>(Math.Min(N, 1024));
            sharedPQ = new PriorityQueue<int, long>();

            if (k <= 18)
                ExactMatching(oddVerts);
            else
                GreedyMatching(oddVerts);
        }

        // Hierholzer's algorithm to find Euler circuit starting from node 0
        int totalEdges = edgeU.Count;
        bool[] used = new bool[totalEdges];
        int[] iter = new int[N];

        var stk = new Stack<int>();
        var circuit = new List<int>(totalEdges + 2);
        stk.Push(0);
        while (stk.Count > 0)
        {
            int v = stk.Peek();
            var av = adj[v];
            int it = iter[v];
            while (it < av.Count && used[av[it]]) it++;
            iter[v] = it;
            if (it < av.Count)
            {
                int eid = av[it];
                used[eid] = true;
                int next = (edgeU[eid] == v) ? edgeV[eid] : edgeU[eid];
                stk.Push(next);
            }
            else
            {
                circuit.Add(v);
                stk.Pop();
            }
        }
        circuit.Reverse();

        // Output the route
        var outSb = new StringBuilder(circuit.Count * 7 + 8);
        for (int i = 0; i < circuit.Count; i++)
        {
            if (i > 0) outSb.Append(' ');
            outSb.Append(circuit[i]);
        }
        outSb.Append('\n');

        var stdout = new BufferedStream(Console.OpenStandardOutput(), 1 << 16);
        byte[] outBytes = Encoding.ASCII.GetBytes(outSb.ToString());
        stdout.Write(outBytes, 0, outBytes.Length);
        stdout.Flush();
    }

    static void ExactMatching(List<int> oddVerts)
    {
        int k = oddVerts.Count;
        long[][] dist = new long[k][];
        int[][] preds = new int[k][];
        for (int i = 0; i < k; i++)
        {
            var r = FullDijkstra(oddVerts[i]);
            dist[i] = r.Item1;
            preds[i] = r.Item2;
        }

        int totalMasks = 1 << k;
        long[] dp = new long[totalMasks];
        int[] dpFrom = new int[totalMasks];
        for (int i = 0; i < totalMasks; i++) dp[i] = long.MaxValue;
        dp[0] = 0;

        for (int mask = 0; mask < totalMasks; mask++)
        {
            long cur = dp[mask];
            if (cur == long.MaxValue) continue;
            int i = 0;
            while (i < k && ((mask >> i) & 1) == 1) i++;
            if (i >= k) continue;
            long[] di = dist[i];
            for (int j = i + 1; j < k; j++)
            {
                if (((mask >> j) & 1) != 0) continue;
                long pd = di[oddVerts[j]];
                if (pd == long.MaxValue) continue;
                int newMask = mask | (1 << i) | (1 << j);
                long newCost = cur + pd;
                if (newCost < dp[newMask])
                {
                    dp[newMask] = newCost;
                    dpFrom[newMask] = i * k + j;
                }
            }
        }

        int curMask = totalMasks - 1;
        while (curMask != 0)
        {
            int packed = dpFrom[curMask];
            int i = packed / k, j = packed % k;
            DuplicatePath(oddVerts[i], oddVerts[j], preds[i]);
            curMask ^= (1 << i) | (1 << j);
        }
    }

    static void GreedyMatching(List<int> oddVerts)
    {
        int k = oddVerts.Count;
        int[] vToOddIdx = new int[N];
        for (int i = 0; i < N; i++) vToOddIdx[i] = -1;
        for (int i = 0; i < k; i++) vToOddIdx[oddVerts[i]] = i;

        bool[] paired = new bool[k];

        for (int i = 0; i < k; i++)
        {
            if (paired[i]) continue;
            int target = LimitedDijkstra(oddVerts[i], vToOddIdx, paired);
            if (target == -1) continue;
            int tIdx = vToOddIdx[target];
            paired[i] = true;
            paired[tIdx] = true;
            DuplicatePath(oddVerts[i], target, tmpPred);
        }
    }

    static Tuple<long[], int[]> FullDijkstra(int src)
    {
        long[] dist = new long[N];
        int[] pred = new int[N];
        for (int i = 0; i < N; i++) { dist[i] = long.MaxValue; pred[i] = -1; }
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        int v;
        long d;
        while (pq.TryDequeue(out v, out d))
        {
            if (d > dist[v]) continue;
            var av = adj[v];
            for (int idx = 0; idx < av.Count; idx++)
            {
                int eid = av[idx];
                int u = (edgeU[eid] == v) ? edgeV[eid] : edgeU[eid];
                long nd = d + edgeW[eid];
                if (nd < dist[u])
                {
                    dist[u] = nd;
                    pred[u] = eid;
                    pq.Enqueue(u, nd);
                }
            }
        }
        return Tuple.Create(dist, pred);
    }

    static int LimitedDijkstra(int src, int[] vToOddIdx, bool[] paired)
    {
        // Reset previously touched vertices
        for (int i = 0; i < touched.Count; i++)
        {
            int v = touched[i];
            tmpDist[v] = long.MaxValue;
            tmpPred[v] = -1;
        }
        touched.Clear();
        sharedPQ.Clear();

        tmpDist[src] = 0;
        tmpPred[src] = -1;
        touched.Add(src);
        sharedPQ.Enqueue(src, 0);

        int v;
        long d;
        while (sharedPQ.TryDequeue(out v, out d))
        {
            if (d > tmpDist[v]) continue;
            if (v != src && vToOddIdx[v] >= 0 && !paired[vToOddIdx[v]])
            {
                return v;
            }
            var av = adj[v];
            for (int idx = 0; idx < av.Count; idx++)
            {
                int eid = av[idx];
                int u = (edgeU[eid] == v) ? edgeV[eid] : edgeU[eid];
                long nd = d + edgeW[eid];
                if (nd < tmpDist[u])
                {
                    if (tmpDist[u] == long.MaxValue) touched.Add(u);
                    tmpDist[u] = nd;
                    tmpPred[u] = eid;
                    sharedPQ.Enqueue(u, nd);
                }
            }
        }
        return -1;
    }

    static void DuplicatePath(int from, int to, int[] pred)
    {
        int cur = to;
        int safety = 0;
        int maxSteps = edgeU.Count + N + 16;
        while (cur != from && pred[cur] != -1)
        {
            if (++safety > maxSteps) break;
            int eid = pred[cur];
            int u = edgeU[eid], v = edgeV[eid];
            long w = edgeW[eid];
            int newEid = edgeU.Count;
            edgeU.Add(u);
            edgeV.Add(v);
            edgeW.Add(w);
            adj[u].Add(newEid);
            adj[v].Add(newEid);
            cur = (cur == v) ? u : v;
        }
    }
}
