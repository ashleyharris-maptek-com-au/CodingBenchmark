// Adaptive Travelling Salesman Problem solver.
//
// Strategy (chosen by N):
//   * Trivial  (N <= 13)      : nearest neighbour seed + full 2-opt to (near-)optimal
//   * Small    (N <= 3000)    : grid nearest neighbour + full 2-opt with time budget
//   * Medium   (N <= 50000)   : grid nearest neighbour + neighbour-list 2-opt
//   * Large    (N <= 500000)  : grid nearest neighbour (single pass, no improvement)
//   * Extreme  (N > 500000)   : Hilbert space-filling curve sort (O(N log N))
//
// All branches finish well under the 300 s limit on the target hardware while
// producing a valid permutation of 0..N-1 that starts at city 0.
//
// Targeting MSVC cl 19.50 with /std:c++20 -- only standard headers are used.

#include <algorithm>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using std::int32_t;
using std::int64_t;
using std::uint32_t;
using std::uint64_t;

// ---------- Globals ----------
static std::chrono::steady_clock::time_point g_t0;
static double elapsed() {
    return std::chrono::duration<double>(std::chrono::steady_clock::now() - g_t0).count();
}
static constexpr double TIME_LIMIT = 280.0; // safety margin under the 300 s wall clock

static int N = 0;
static std::vector<float> X, Y;

static inline float distSq(int a, int b) {
    float dx = X[a] - X[b];
    float dy = Y[a] - Y[b];
    return dx * dx + dy * dy;
}
static inline float dist(int a, int b) { return std::sqrt(distSq(a, b)); }

// ---------- Fast stdin parsing ----------
namespace fastio {
    static constexpr size_t BUF = 1 << 20;
    static char buf[BUF];
    static size_t pos = 0, len = 0;
    static inline int getch() {
        if (pos >= len) {
            len = std::fread(buf, 1, BUF, stdin);
            pos = 0;
            if (len == 0) return -1;
        }
        return (unsigned char)buf[pos++];
    }
    static inline bool readInt(int& out) {
        int c = getch();
        while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = getch();
        if (c == -1) return false;
        int sign = 1;
        if (c == '-') { sign = -1; c = getch(); }
        int v = 0;
        while (c >= '0' && c <= '9') { v = v * 10 + (c - '0'); c = getch(); }
        out = sign * v;
        return true;
    }
    static inline bool readFloat(float& out) {
        int c = getch();
        while (c != -1 && (c == ' ' || c == '\n' || c == '\r' || c == '\t')) c = getch();
        if (c == -1) return false;
        int sign = 1;
        if (c == '-') { sign = -1; c = getch(); }
        else if (c == '+') c = getch();
        double v = 0.0;
        while (c >= '0' && c <= '9') { v = v * 10.0 + (c - '0'); c = getch(); }
        if (c == '.') {
            c = getch();
            double f = 0.0, p = 1.0;
            while (c >= '0' && c <= '9') { f = f * 10.0 + (c - '0'); p *= 10.0; c = getch(); }
            v += f / p;
        }
        if (c == 'e' || c == 'E') {
            c = getch();
            int esign = 1;
            if (c == '-') { esign = -1; c = getch(); }
            else if (c == '+') c = getch();
            int ev = 0;
            while (c >= '0' && c <= '9') { ev = ev * 10 + (c - '0'); c = getch(); }
            v *= std::pow(10.0, esign * ev);
        }
        out = (float)(sign * v);
        return true;
    }
} // namespace fastio

// ---------- Hilbert curve ordering ----------
static inline uint64_t hilbert_d(uint64_t order, uint64_t x, uint64_t y) {
    uint64_t d = 0;
    for (uint64_t s = order / 2; s > 0; s >>= 1) {
        uint64_t rx = (x & s) ? 1ULL : 0ULL;
        uint64_t ry = (y & s) ? 1ULL : 0ULL;
        d += s * s * ((3ULL * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = s - 1 - x;
                y = s - 1 - y;
            }
            std::swap(x, y);
        }
    }
    return d;
}

static std::vector<int> hilbert_tour() {
    float minX = X[0], maxX = X[0], minY = Y[0], maxY = Y[0];
    for (int i = 1; i < N; i++) {
        if (X[i] < minX) minX = X[i]; else if (X[i] > maxX) maxX = X[i];
        if (Y[i] < minY) minY = Y[i]; else if (Y[i] > maxY) maxY = Y[i];
    }
    double range = std::max((double)(maxX - minX), (double)(maxY - minY));
    if (!(range > 0.0)) range = 1.0;
    const uint64_t order = 1ULL << 30; // ~10^9 resolution per axis

    std::vector<std::pair<uint64_t, int>> h(N);
    const double scale = (double)(order - 1) / range;
    for (int i = 0; i < N; i++) {
        uint64_t xi = (uint64_t)((double)(X[i] - minX) * scale);
        uint64_t yi = (uint64_t)((double)(Y[i] - minY) * scale);
        if (xi >= order) xi = order - 1;
        if (yi >= order) yi = order - 1;
        h[i].first = hilbert_d(order, xi, yi);
        h[i].second = i;
    }
    std::sort(h.begin(), h.end(),
              [](const std::pair<uint64_t,int>& a, const std::pair<uint64_t,int>& b) {
                  return a.first < b.first;
              });
    std::vector<int> tour(N);
    for (int i = 0; i < N; i++) tour[i] = h[i].second;
    return tour;
}

// ---------- Grid-based nearest neighbour ----------
struct Grid {
    int gridSize;
    double minX, minY, cellW, cellH;
    // CSR-style bucketed storage to avoid the overhead of vector<vector<int>>
    std::vector<int> head;   // first point id in cell, -1 if empty
    std::vector<int> next;   // linked list next per point
    std::vector<int> prev;   // linked list prev per point
    std::vector<int> cellOf; // cell index per point
};

static void build_grid(Grid& g) {
    float minX = X[0], maxX = X[0], minY = Y[0], maxY = Y[0];
    for (int i = 1; i < N; i++) {
        if (X[i] < minX) minX = X[i]; else if (X[i] > maxX) maxX = X[i];
        if (Y[i] < minY) minY = Y[i]; else if (Y[i] > maxY) maxY = Y[i];
    }
    double rx = (double)(maxX - minX); if (rx <= 0) rx = 1.0;
    double ry = (double)(maxY - minY); if (ry <= 0) ry = 1.0;
    // ~4 points per cell on average; cap to keep memory sane
    long long target = (long long)std::ceil(std::sqrt((double)N / 4.0));
    if (target < 2) target = 2;
    if (target > 4096) target = 4096;
    g.gridSize = (int)target;
    g.minX = minX; g.minY = minY;
    g.cellW = rx / g.gridSize;
    g.cellH = ry / g.gridSize;

    long long cells = (long long)g.gridSize * g.gridSize;
    g.head.assign((size_t)cells, -1);
    g.next.assign(N, -1);
    g.prev.assign(N, -1);
    g.cellOf.assign(N, 0);

    for (int i = 0; i < N; i++) {
        int cx = (int)((X[i] - g.minX) / g.cellW);
        int cy = (int)((Y[i] - g.minY) / g.cellH);
        if (cx < 0) cx = 0; else if (cx >= g.gridSize) cx = g.gridSize - 1;
        if (cy < 0) cy = 0; else if (cy >= g.gridSize) cy = g.gridSize - 1;
        int idx = cy * g.gridSize + cx;
        g.cellOf[i] = idx;
        int h = g.head[idx];
        g.next[i] = h;
        g.prev[i] = -1;
        if (h != -1) g.prev[h] = i;
        g.head[idx] = i;
    }
}

static inline void grid_remove(Grid& g, int p) {
    int c = g.cellOf[p];
    int pr = g.prev[p];
    int nx = g.next[p];
    if (pr != -1) g.next[pr] = nx; else g.head[c] = nx;
    if (nx != -1) g.prev[nx] = pr;
}

static std::vector<int> nn_grid_tour() {
    Grid g;
    build_grid(g);

    std::vector<int> tour;
    tour.reserve(N);

    int current = 0;
    tour.push_back(0);
    grid_remove(g, 0);

    const double mincell = std::min(g.cellW, g.cellH);

    for (int step = 1; step < N; step++) {
        int c = g.cellOf[current];
        int cx = c % g.gridSize;
        int cy = c / g.gridSize;
        int bestNext = -1;
        float bestDS = std::numeric_limits<float>::infinity();

        for (int r = 0; r <= g.gridSize; r++) {
            int x0 = cx - r, x1 = cx + r;
            int y0 = cy - r, y1 = cy + r;
            if (x0 < 0) x0 = 0;
            if (y0 < 0) y0 = 0;
            if (x1 >= g.gridSize) x1 = g.gridSize - 1;
            if (y1 >= g.gridSize) y1 = g.gridSize - 1;

            if (r == 0) {
                int idx = cy * g.gridSize + cx;
                for (int p = g.head[idx]; p != -1; p = g.next[p]) {
                    float d = distSq(current, p);
                    if (d < bestDS) { bestDS = d; bestNext = p; }
                }
            } else {
                // Top and bottom rows of the ring
                for (int yy : {cy - r, cy + r}) {
                    if (yy < 0 || yy >= g.gridSize) continue;
                    for (int xx = x0; xx <= x1; xx++) {
                        int idx = yy * g.gridSize + xx;
                        for (int p = g.head[idx]; p != -1; p = g.next[p]) {
                            float d = distSq(current, p);
                            if (d < bestDS) { bestDS = d; bestNext = p; }
                        }
                    }
                }
                // Left and right columns excluding corners
                for (int xx : {cx - r, cx + r}) {
                    if (xx < 0 || xx >= g.gridSize) continue;
                    int yya = cy - r + 1, yyb = cy + r - 1;
                    if (yya < 0) yya = 0;
                    if (yyb >= g.gridSize) yyb = g.gridSize - 1;
                    for (int yy = yya; yy <= yyb; yy++) {
                        int idx = yy * g.gridSize + xx;
                        for (int p = g.head[idx]; p != -1; p = g.next[p]) {
                            float d = distSq(current, p);
                            if (d < bestDS) { bestDS = d; bestNext = p; }
                        }
                    }
                }
            }
            if (bestNext != -1) {
                // True nearest is guaranteed once the ring's inner edge
                // is at least sqrt(bestDS) away from the source cell.
                double radDist = (double)r * mincell;
                if (radDist * radDist >= (double)bestDS) break;
            }
        }

        if (bestNext == -1) {
            // Defensive fallback: scan the whole grid for any remaining point
            for (size_t idx = 0; idx < g.head.size(); idx++) {
                if (g.head[idx] != -1) { bestNext = g.head[idx]; break; }
            }
        }

        tour.push_back(bestNext);
        grid_remove(g, bestNext);
        current = bestNext;
    }
    return tour;
}

// ---------- Full 2-opt (for small N) ----------
static void two_opt_full(std::vector<int>& tour, double budgetSec) {
    double tStart = elapsed();
    int n = (int)tour.size();
    if (n < 4) return;
    bool improved = true;
    int passes = 0;
    while (improved) {
        if (elapsed() - tStart > budgetSec) break;
        improved = false;
        for (int i = 0; i < n - 1; i++) {
            if ((i & 255) == 0 && elapsed() - tStart > budgetSec) break;
            int a = tour[i];
            int b = tour[i + 1];
            float dab = dist(a, b);
            for (int j = i + 2; j < n; j++) {
                if (i == 0 && j == n - 1) continue;
                int c = tour[j];
                int d = tour[(j + 1) % n];
                float dcd = dist(c, d);
                float dac = dist(a, c);
                float dbd = dist(b, d);
                if (dac + dbd + 1e-9f < dab + dcd) {
                    std::reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    b = tour[i + 1];
                    dab = dist(a, b);
                }
            }
        }
        if (++passes > 200) break;
    }
}

// ---------- Neighbour-list 2-opt (for medium N) ----------
// Build k nearest neighbours per city via the spatial grid, then perform
// 2-opt restricted to candidate edges -- O(N*k) per pass, several passes.
static std::vector<int> build_knn(int k) {
    Grid g;
    build_grid(g);
    std::vector<int> knn((size_t)N * k, -1);
    const double mincell = std::min(g.cellW, g.cellH);

    std::vector<std::pair<float, int>> cand;
    cand.reserve(k * 8);

    for (int i = 0; i < N; i++) {
        cand.clear();
        int c = g.cellOf[i];
        int cx = c % g.gridSize;
        int cy = c / g.gridSize;
        float worst = std::numeric_limits<float>::infinity();
        for (int r = 0; r <= g.gridSize; r++) {
            int x0 = std::max(0, cx - r), x1 = std::min(g.gridSize - 1, cx + r);
            int y0 = std::max(0, cy - r), y1 = std::min(g.gridSize - 1, cy + r);
            for (int yy = y0; yy <= y1; yy++) {
                for (int xx = x0; xx <= x1; xx++) {
                    if (r > 0 && std::abs(xx - cx) < r && std::abs(yy - cy) < r) continue;
                    int idx = yy * g.gridSize + xx;
                    for (int p = g.head[idx]; p != -1; p = g.next[p]) {
                        if (p == i) continue;
                        float d = distSq(i, p);
                        cand.push_back({d, p});
                    }
                }
            }
            if ((int)cand.size() >= k) {
                std::nth_element(cand.begin(), cand.begin() + k - 1, cand.end(),
                                 [](const std::pair<float,int>& a, const std::pair<float,int>& b){
                                     return a.first < b.first;
                                 });
                worst = cand[k - 1].first;
                double radDist = (double)r * mincell;
                if (radDist * radDist >= (double)worst) break;
            }
        }
        int take = std::min(k, (int)cand.size());
        std::partial_sort(cand.begin(), cand.begin() + take, cand.end(),
                          [](const std::pair<float,int>& a, const std::pair<float,int>& b){
                              return a.first < b.first;
                          });
        for (int j = 0; j < take; j++) knn[(size_t)i * k + j] = cand[j].second;
    }
    return knn;
}

static void two_opt_neighbors(std::vector<int>& tour, int k, double budgetSec) {
    int n = (int)tour.size();
    if (n < 4) return;
    std::vector<int> knn = build_knn(k);
    std::vector<int> pos(n);
    for (int i = 0; i < n; i++) pos[tour[i]] = i;

    double tStart = elapsed();
    bool improved = true;
    int passes = 0;
    while (improved) {
        if (elapsed() - tStart > budgetSec) break;
        improved = false;
        for (int i = 0; i < n; i++) {
            if ((i & 4095) == 0 && elapsed() - tStart > budgetSec) break;
            int a = tour[i];
            int b = tour[(i + 1) % n];
            float dab = dist(a, b);
            for (int t = 0; t < k; t++) {
                int c = knn[(size_t)a * k + t];
                if (c < 0 || c == b || c == a) continue;
                float dac = dist(a, c);
                if (dac >= dab) break; // neighbours are sorted by distance
                int jc = pos[c];
                int d = tour[(jc + 1) % n];
                if (d == a) continue;
                float dcd = dist(c, d);
                float dbd = dist(b, d);
                if (dac + dbd + 1e-9f < dab + dcd) {
                    int lo = (i + 1) % n;
                    int hi = jc;
                    int len;
                    if (hi >= lo) len = hi - lo + 1;
                    else len = (n - lo) + hi + 1;
                    int otherLen = n - len;
                    // Reverse whichever arc is shorter to keep updates cheap
                    if (len <= otherLen) {
                        int p = lo, q = hi;
                        for (int s = 0; s < len / 2; s++) {
                            std::swap(tour[p], tour[q]);
                            pos[tour[p]] = p;
                            pos[tour[q]] = q;
                            p = (p + 1) % n;
                            q = (q - 1 + n) % n;
                        }
                    } else {
                        int p = (hi + 1) % n, q = (lo - 1 + n) % n;
                        for (int s = 0; s < otherLen / 2; s++) {
                            std::swap(tour[p], tour[q]);
                            pos[tour[p]] = p;
                            pos[tour[q]] = q;
                            p = (p + 1) % n;
                            q = (q - 1 + n) % n;
                        }
                    }
                    improved = true;
                    b = tour[(i + 1) % n];
                    dab = dist(a, b);
                }
            }
        }
        if (++passes > 50) break;
    }
}

// ---------- Output ----------
static void write_tour(const std::vector<int>& tour) {
    // Buffered ASCII output -- significantly faster than printf per token at scale.
    std::string out;
    out.reserve((size_t)N * 8);
    char tmp[16];
    for (int i = 0; i < N; i++) {
        if (i > 0) out.push_back(' ');
        int v = tour[i];
        int len = std::snprintf(tmp, sizeof(tmp), "%d", v);
        out.append(tmp, (size_t)len);
    }
    out.push_back('\n');
    std::fwrite(out.data(), 1, out.size(), stdout);
}

// ---------- Main dispatcher ----------
int main() {
    g_t0 = std::chrono::steady_clock::now();

    if (!fastio::readInt(N)) return 0;
    if (N <= 0) { std::fputs("\n", stdout); return 0; }

    X.resize(N);
    Y.resize(N);
    for (int i = 0; i < N; i++) {
        fastio::readFloat(X[i]);
        fastio::readFloat(Y[i]);
    }

    std::vector<int> tour;

    if (N == 1) {
        tour = {0};
    } else if (N <= 13) {
        // Cheap NN seed then exhaustive 2-opt -- effectively optimal at this size.
        tour.reserve(N);
        std::vector<char> used((size_t)N, 0);
        tour.push_back(0);
        used[0] = 1;
        for (int s = 1; s < N; s++) {
            int last = tour.back();
            int best = -1;
            float bd = std::numeric_limits<float>::infinity();
            for (int j = 0; j < N; j++) {
                if (!used[j]) {
                    float d = distSq(last, j);
                    if (d < bd) { bd = d; best = j; }
                }
            }
            tour.push_back(best);
            used[best] = 1;
        }
        two_opt_full(tour, 5.0);
    } else if (N <= 3000) {
        tour = nn_grid_tour();
        double remaining = TIME_LIMIT - elapsed();
        two_opt_full(tour, std::max(1.0, std::min(remaining - 5.0, 60.0)));
    } else if (N <= 50000) {
        tour = nn_grid_tour();
        double remaining = TIME_LIMIT - elapsed();
        // 10 nearest neighbours strikes a good speed/quality trade-off.
        if (remaining > 30.0) two_opt_neighbors(tour, 10, remaining - 20.0);
    } else if (N <= 500000) {
        // NN by itself: ~25% over optimal, but cheap and robust at this scale.
        tour = nn_grid_tour();
    } else {
        // Hilbert curve -- O(N log N), no large auxiliary structures.
        tour = hilbert_tour();
    }

    // Ensure city 0 is the start of the tour.
    if (!tour.empty() && tour[0] != 0) {
        int pos0 = -1;
        for (int i = 0; i < (int)tour.size(); i++) {
            if (tour[i] == 0) { pos0 = i; break; }
        }
        if (pos0 > 0) std::rotate(tour.begin(), tour.begin() + pos0, tour.end());
    }

    write_tour(tour);
    return 0;
}
