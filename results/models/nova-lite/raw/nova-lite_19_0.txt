using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class MiningProblemSolver
{
    private static int XVoxelCount, YVoxelCount, ZVoxelCount;
    private static double XBlockSize, YBlockSize, ZBlockSize;
    private static int[,,] grid;
    private static int[,,] heights;
    private static List<(int, int, int, int, int, int, int, int, int, int)> operations = new List<(int, int, int, int, int, int, int, int, int, int)>();

    public static void Main(string[] args)
    {
        var input = Console.In.ReadToEnd().Split('\n');
        var dimensions = input[0].Split(' ').Select(int.Parse).ToArray();
        XVoxelCount = dimensions[0];
        YVoxelCount = dimensions[1];
        ZVoxelCount = dimensions[2];
        XBlockSize = dimensions[3];
        YBlockSize = dimensions[4];
        ZBlockSize = dimensions[5];

        grid = new int[XVoxelCount, YVoxelCount, ZVoxelCount];
        heights = new int[XVoxelCount, YVoxelCount];

        for (int x = 0; x < XVoxelCount; x++)
        {
            for (int y = 0; y < YVoxelCount; y++)
            {
                for (int z = 0; z < ZVoxelCount; z++)
                {
                    int massInTonnes = input[x * ZVoxelCount + y * XVoxelCount + z + 1][0] % 16;
                    int valuePerTonne = input[x * ZVoxelCount + y * XVoxelCount + z + 1][0] / 16;
                    grid[x, y, z] = valuePerTonne;
                    if (z == 0) heights[x, y] = z;
                }
            }
        }

        Solve();
    }

    private static void Solve()
    {
        int totalOperations = 0;
        for (int x = 0; x < XVoxelCount; x++)
        {
            for (int y = 0; y < YVoxelCount; y++)
            {
                for (int z = ZVoxelCount - 1; z >= 0; z--)
                {
                    if (grid[x, y, z] > 0)
                    {
                        int valuePerTonne = grid[x, y, z];
                        int mass = 1; // Assuming 1 ton for simplicity
                        int targetX = 0, targetY = 0, targetZ = 0;
                        int cost = 0;

                        // Find best dumping location
                        for (int dz = 0; dz < ZVoxelCount; dz++)
                        {
                            for (int dx = 0; dx < XVoxelCount; dx++)
                            {
                                for (int dy = 0; dy < YVoxelCount; dy++)
                                {
                                    if (grid[dx, dy, dz] == 0 && heights[dx, dy] + 1 <= dz)
                                    {
                                        int pathCost = CalculatePathCost(x, y, z, dx, dy, dz);
                                        if (pathCost < int.MaxValue && (targetZ == 0 || pathCost + dz < targetZ))
                                        {
                                            targetX = dx;
                                            targetY = dy;
                                            targetZ = dz;
                                            cost = pathCost;
                                        }
                                    }
                                }
                            }
                        }

                        if (targetX == 0 && targetY == 0)
                        {
                            // Dump at processing plant
                            totalOperations += MoveOre(x, y, z, 0, 0, targetZ, valuePerTonne, mass, cost);
                        }
                        else
                        {
                            // Dump at target location
                            totalOperations += MoveOre(x, y, z, targetX, targetY, targetZ, valuePerTonne, mass, cost);
                        }
                    }
                }
            }
        }

        Console.WriteLine(totalOperations);
        foreach (var operation in operations)
        {
            Console.WriteLine($"{operation.Item1} {operation.Item2} {operation.Item3} {operation.Item4} {operation.Item5} {operation.Item6} {operation.Item7} {operation.Item8} {operation.Item9} {operation.Item10}");
        }
    }

    private static int CalculatePathCost(int sx, int sy, int sz, int tx, int ty, int tz)
    {
        var queue = new Queue<(int, int, int, int)>();
        var visited = new bool[XVoxelCount, YVoxelCount, ZVoxelCount];
        queue.Enqueue((sx, sy, sz, 0));
        visited[sx, sy, sz] = true;

        while (queue.Count > 0)
        {
            var (cx, cy, cz, ccost) = queue.Dequeue();
            if (cx == tx && cy == ty && cz == tz)
                return ccost;

            foreach (var (dx, dy) in new[] { (1, 0), (-1, 0), (0, 1), (0, -1) })
            {
                int nx = cx + dx, ny = cy + dy;
                if (nx >= 0 && nx < XVoxelCount && ny >= 0 && ny < YVoxelCount && !visited[nx, ny, cz])
                {
                    if (Math.Abs(heights[nx, ny] - heights[cx, cy]) <= 1)
                    {
                        visited[nx, ny, cz] = true;
                        queue.Enqueue((nx, ny, cz, ccost + (Math.Abs(heights[nx, ny] - heights[cx, cy]) > 0? 5 : 1)));
                    }
                }

                if (cz > 0 && !visited[cx, cy, cz - 1])
                {
                    visited[cx, cy, cz - 1] = true;
                    queue.Enqueue((cx, cy, cz - 1, ccost + 1));
                }
            }
        }

        return int.MaxValue;
    }

    private static int MoveOre(int sx, int sy, int sz, int tx, int ty, int tz, int valuePerTonne, int mass, int cost)
    {
        grid[sx, sy, sz] = 0;
        grid[tx, ty, tz + 1] = valuePerTonne;
        heights[tx, ty] = Math.Max(heights[tx, ty], tz + 1);
        int totalCost = cost * mass;
        if (tx == 0 && ty == 0)
            totalCost += valuePerTonne * mass * 1000;
        else
            totalCost -= valuePerTonne * mass * 1000;

        operations.Add((1, sx, sy, sz, tx, ty, tz, 0, 0, 0));
        return 1;
    }
}
