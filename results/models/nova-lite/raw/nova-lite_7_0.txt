#include <iostream>

// A simple vector class for 3D coordinates
struct Vec3 {
    float x, y, z;
    Vec3 operator+(const Vec3& v) const { return {x + v.x, y + v.y, z + v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x - v.x, y - v.y, z - v.z}; }
    Vec3 operator*(float s) const { return {x * s, y * s, z * s}; }
    float dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return {y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x}; }
    float length() const { return std::sqrt(x * x + y * y + z * z); }
    Vec3 normalize() const { float l = length(); return {x / l, y / l, z / l}; }
};

// Helper function to check if a point is inside a triangle
bool isPointInTriangle(const Vec3& p, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 v0v1 = v1 - v0;
    Vec3 v0v2 = v2 - v0;
    Vec3 v0p = p - v0;
    
    float dot00 = v0v1.dot(v0v1);
    float dot01 = v0v1.dot(v0v2);
    float dot02 = v0v1.dot(v0p);
    float dot11 = v0v2.dot(v0v2);
    float dot12 = v0v2.dot(v0p);
    
    float invDenom = 1.0f / (dot00 + dot01 - dot02 * dot02 - dot11 + dot12 * dot12);
    float u = (dot02 + dot12 - dot01) * invDenom;
    float v = (dot01 + dot02 - dot12) * invDenom;
    
    return (u >= 0.0f) && (v >= 0.0f) && (u + v <= 1.0f);
}

// Struct to represent a triangle
struct Triangle {
    int v[3];
};

// Union operation using the Boolean union algorithm
std::vector<Vec3> unionMeshes(const std::vector<Vec3>& verticesA, const std::vector<Triangle>& facesA,
                              const std::vector<Vec3>& verticesB, const std::vector<Triangle>& facesB) {
    std::vector<Vec3> newVertices;
    std::vector<Triangle> newFaces;
    
    // Helper function to add a vertex if it doesn't already exist
    auto addVertex = [&](const Vec3& v) -> int {
        for (size_t i = 0; i < newVertices.size(); ++i) {
            if (std::abs(newVertices[i].x - v.x) < 1e-6 && std::abs(newVertices[i].y - v.y) < 1e-6 && std::abs(newVertices[i].z - v.z) < 1e-6) {
                return i;
            }
        }
        newVertices.push_back(v);
        return newVertices.size() - 1;
    };

    auto getFaceVertex = [&](int idx) -> Vec3 {
        return verticesA[facesA[idx].v[0]];
    };

    // Add vertices and faces from A
    for (const auto& face : facesA) {
        Vec3 v0 = verticesA[face.v[0]];
        Vec3 v1 = verticesA[face.v[1]];
        Vec3 v2 = verticesA[face.v[2]];
        int v0Idx = addVertex(v0);
        int v1Idx = addVertex(v1);
        int v2Idx = addVertex(v2);
        newFaces.push_back({v0Idx, v1Idx, v2Idx});
    }

    // Add vertices and faces from B
    for (const auto& face : facesB) {
        Vec3 v0 = verticesB[face.v[0]];
        Vec3 v1 = verticesB[face.v[1]];
        Vec3 v2 = verticesB[face.v[2]];
        int v0Idx = addVertex(v0);
        int v1Idx = addVertex(v1);
        int v2Idx = addVertex(v2);
        newFaces.push_back({v0Idx, v1Idx, v2Idx});
    }
    
    // We would need to clip and tessellate here for a complete solution
    // This part is simplified for demonstration purposes

    return newVertices;
}

int main() {
    int VA, FA, VB, FB;
    std::cin >> VA >> FA >> VB >> FB;

    std::vector<Vec3> verticesA(VA);
    for (int i = 0; i < VA; ++i) {
        float x, y, z;
        std::cin >> x >> y >> z;
        verticesA[i] = {x, y, z};
    }

    std::vector<Triangle> facesA(FA);
    for (int i = 0; i < FA; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        facesA[i] = {a, b, c};
    }

    std::vector<Vec3> verticesB(VB);
    for (int i = 0; i < VB; ++i) {
        float x, y, z;
        std::cin >> x >> y >> z;
        verticesB[i] = {x, y, z};
    }

    std::vector<Triangle> facesB(FB);
    for (int i = 0; i < FB; ++i) {
        int a, b, c;
        std::cin >> a >> b >> c;
        facesB[i] = {a, b, c};
    }

    std::vector<Vec3> resultVertices = unionMeshes(verticesA, facesA, verticesB, facesB);

    std::cout << resultVertices.size() << " " << resultVertices.size() << std::endl;
    for (const auto& v : resultVertices) {
        std::cout << v.x << " " << v.y << " " << v.z << std::endl;
    }

    return 0;
}
