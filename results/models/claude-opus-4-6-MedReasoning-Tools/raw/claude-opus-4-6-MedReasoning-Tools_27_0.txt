#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <random>
#include <chrono>
#include <set>
#include <tuple>
#include <climits>
#include <cstdio>

using namespace std;
using ll = long long;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    cin >> n >> m >> k;

    vector<vector<int>> adj(n);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }

    vector<int> color(n, -1);
    auto start = chrono::steady_clock::now();
    auto elapsed_ms = [&]() -> ll {
        return chrono::duration_cast<chrono::milliseconds>(
            chrono::steady_clock::now() - start).count();
    };

    auto output = [&]() {
        for (int i = 0; i < n; i++) {
            if (i) putchar(' ');
            printf("%d", color[i]);
        }
        putchar('\n');
    };

    auto greedy = [&](const vector<int>& order) -> bool {
        fill(color.begin(), color.end(), -1);
        vector<bool> used(k, false);
        for (int v : order) {
            fill(used.begin(), used.end(), false);
            for (int u : adj[v]) {
                if (color[u] >= 0) used[color[u]] = true;
            }
            color[v] = -1;
            for (int i = 0; i < k; i++) {
                if (!used[i]) { color[v] = i; break; }
            }
            if (color[v] == -1) return false;
        }
        return true;
    };

    // 1. Greedy with largest-degree-first ordering
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return adj[a].size() > adj[b].size();
    });

    if (greedy(order)) { output(); return 0; }

    ll nk = (ll)n * k;

    // 2. DSatur with priority queue
    if (nk <= 60000000LL) {
        fill(color.begin(), color.end(), -1);
        vector<int> cc(nk, 0);
        vector<int> sat(n, 0);
        vector<int> deg(n);
        for (int i = 0; i < n; i++) deg[i] = (int)adj[i].size();

        set<tuple<int,int,int>> pq;
        for (int i = 0; i < n; i++) {
            pq.insert({0, -deg[i], i});
        }

        bool ok = true;
        for (int step = 0; step < n && ok; step++) {
            auto it = pq.begin();
            auto [ns, nd, v] = *it;
            pq.erase(it);

            vector<bool> used(k, false);
            for (int u : adj[v]) {
                if (color[u] >= 0) used[color[u]] = true;
            }
            int c = -1;
            for (int i = 0; i < k; i++) {
                if (!used[i]) { c = i; break; }
            }
            if (c == -1) { ok = false; break; }

            color[v] = c;
            for (int u : adj[v]) {
                if (color[u] == -1) {
                    ll idx = (ll)u * k + c;
                    if (cc[idx] == 0) {
                        pq.erase({-sat[u], -deg[u], u});
                        sat[u]++;
                        pq.insert({-sat[u], -deg[u], u});
                    }
                    cc[idx]++;
                }
            }
        }

        if (ok) { output(); return 0; }
        cc.clear();
        cc.shrink_to_fit();
    }

    // 3. TabuCol local search
    if (nk <= 40000000LL) {
        mt19937 rng(42);
        int best_total = INT_MAX;
        vector<int> best_color(n);

        for (int restart = 0; restart < 100 && elapsed_ms() < 270000; restart++) {
            for (int i = 0; i < n; i++) color[i] = rng() % k;

            vector<int> cc(nk, 0);
            for (int v = 0; v < n; v++) {
                for (int u : adj[v]) {
                    cc[(ll)v * k + color[u]]++;
                }
            }

            int total = 0;
            vector<int> conf;
            for (int v = 0; v < n; v++) {
                int c = cc[(ll)v * k + color[v]];
                total += c;
                if (c > 0) conf.push_back(v);
            }
            total /= 2;

            if (total == 0) { output(); return 0; }

            vector<int> tabu(nk, 0);
            int tenure_base = k + 5;
            int no_improve = 0;
            int prev_total = total;

            for (int iter = 1; total > 0 && elapsed_ms() < 270000; iter++) {
                if (conf.empty()) break;

                if (iter % 10000 == 0) {
                    if (total >= prev_total) no_improve += 10000;
                    else no_improve = 0;
                    prev_total = total;
                    if (no_improve > 500000) break;
                }

                if (iter % 50000 == 0) {
                    conf.clear();
                    for (int v = 0; v < n; v++) {
                        if (cc[(ll)v * k + color[v]] > 0) conf.push_back(v);
                    }
                    if (conf.empty()) break;
                }

                int idx = rng() % (int)conf.size();
                int v = conf[idx];
                ll vk = (ll)v * k;

                if (cc[vk + color[v]] == 0) {
                    conf[idx] = conf.back();
                    conf.pop_back();
                    continue;
                }

                int old_c = color[v];
                int best_c = -1;
                int best_delta = INT_MAX;

                for (int c = 0; c < k; c++) {
                    if (c == old_c) continue;
                    int delta = cc[vk + c] - cc[vk + old_c];
                    if (delta < best_delta) {
                        if (tabu[vk + c] <= iter || total + delta == 0) {
                            best_delta = delta;
                            best_c = c;
                        }
                    }
                }

                if (best_c == -1) continue;

                color[v] = best_c;
                total += best_delta;
                tabu[vk + old_c] = iter + tenure_base + (int)(rng() % 10);

                for (int u : adj[v]) {
                    ll uk = (ll)u * k;
                    bool was_conf = (cc[uk + color[u]] > 0);
                    cc[uk + old_c]--;
                    cc[uk + best_c]++;
                    bool is_conf = (cc[uk + color[u]] > 0);
                    if (!was_conf && is_conf) conf.push_back(u);
                }

                if (cc[vk + best_c] == 0) {
                    conf[idx] = conf.back();
                    conf.pop_back();
                }

                if (total < best_total) {
                    best_total = total;
                    best_color = color;
                }
            }

            if (total == 0) { output(); return 0; }
        }

        if (best_total < INT_MAX) {
            color = best_color;
        }
    }

    // 4. Smallest-last ordering
    if (n <= 500000) {
        vector<int> deg(n);
        for (int i = 0; i < n; i++) deg[i] = (int)adj[i].size();
        vector<bool> removed(n, false);
        vector<int> sl_order;
        sl_order.reserve(n);

        for (int step = 0; step < n; step++) {
            int best = -1;
            for (int i = 0; i < n; i++) {
                if (!removed[i] && (best == -1 || deg[i] < deg[best])) {
                    best = i;
                }
            }
            sl_order.push_back(best);
            removed[best] = true;
            for (int u : adj[best]) {
                if (!removed[u]) deg[u]--;
            }
        }
        reverse(sl_order.begin(), sl_order.end());
        if (greedy(sl_order)) { output(); return 0; }
    }

    // 5. Random shuffle retries
    {
        mt19937 rng(12345);
        while (elapsed_ms() < 280000) {
            shuffle(order.begin(), order.end(), rng);
            if (greedy(order)) { output(); return 0; }
        }
    }

    // 6. Fallback
    for (int i = 0; i < n; i++) {
        if (color[i] < 0 || color[i] >= k) color[i] = 0;
    }
    output();
    return 0;
}
