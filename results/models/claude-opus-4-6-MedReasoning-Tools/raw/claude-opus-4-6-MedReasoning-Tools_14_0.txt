using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class MinesweeperSolver
{
    static int H, W;
    static char[,] board;
    static bool[,] knownSafe;
    static bool[,] knownMine;

    static readonly int[] dx = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] dy = { -1, 0, 1, -1, 1, -1, 0, 1 };

    static void Main()
    {
        var input = Console.In.ReadToEnd().Replace("\r\n", "\n").Replace("\r", "\n");
        var lines = input.Split('\n');
        var hw = lines[0].Split(' ');
        H = int.Parse(hw[0]);
        W = int.Parse(hw[1]);
        board = new char[H, W];
        knownSafe = new bool[H, W];
        knownMine = new bool[H, W];

        for (int r = 0; r < H; r++)
        {
            string row = r + 1 < lines.Length ? lines[r + 1] : "";
            for (int c = 0; c < W; c++)
            {
                char ch = c < row.Length ? row[c] : ' ';
                board[r, c] = ch;
                if (ch >= '0' && ch <= '8') knownSafe[r, c] = true;
                else if (ch == '*') knownMine[r, c] = true;
            }
        }

        // Phase 1: Simple constraint propagation (works for all board sizes)
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    if (board[r, c] < '0' || board[r, c] > '8') continue;
                    int num = board[r, c] - '0';
                    var unknowns = new List<(int, int)>();
                    int mineCount = 0;
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dx[d], nc = c + dy[d];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        if (knownMine[nr, nc]) mineCount++;
                        else if (!knownSafe[nr, nc]) unknowns.Add((nr, nc));
                    }
                    if (unknowns.Count == 0) continue;
                    int remaining = num - mineCount;
                    if (remaining == 0)
                    {
                        foreach (var (ur, uc) in unknowns)
                            if (!knownSafe[ur, uc]) { knownSafe[ur, uc] = true; changed = true; }
                    }
                    else if (remaining == unknowns.Count)
                    {
                        foreach (var (ur, uc) in unknowns)
                            if (!knownMine[ur, uc]) { knownMine[ur, uc] = true; changed = true; }
                    }
                }
            }
        }

        // Phase 2: Constraint subset reduction
        var frontierSet = new HashSet<(int, int)>();
        var constraints = BuildConstraints(frontierSet);

        if (frontierSet.Count > 0 && frontierSet.Count <= 2000)
        {
            bool cChanged = true;
            int iterations = 0;
            while (cChanged && iterations < 20)
            {
                cChanged = false;
                iterations++;
                int count = constraints.Count;
                for (int i = 0; i < count && constraints.Count < 100000; i++)
                {
                    for (int j = 0; j < count && constraints.Count < 100000; j++)
                    {
                        if (i == j) continue;
                        var (ci, mi) = constraints[i];
                        var (cj, mj) = constraints[j];
                        if (ci.Count == 0 || cj.Count == 0 || ci.Count >= cj.Count) continue;
                        if (ci.IsSubsetOf(cj))
                        {
                            var diff = new HashSet<(int, int)>(cj);
                            diff.ExceptWith(ci);
                            int diffMines = mj - mi;
                            if (diff.Count > 0 && diffMines >= 0)
                            {
                                bool isNew = true;
                                foreach (var (ec, em) in constraints)
                                {
                                    if (em == diffMines && ec.Count == diff.Count && ec.SetEquals(diff))
                                    { isNew = false; break; }
                                }
                                if (isNew)
                                {
                                    constraints.Add((diff, diffMines));
                                    cChanged = true;
                                }
                            }
                        }
                    }
                }

                // Apply deductions
                bool deduced = true;
                while (deduced)
                {
                    deduced = false;
                    foreach (var (cells, mines) in constraints)
                    {
                        if (cells.Count == 0) continue;
                        if (mines == 0)
                        {
                            foreach (var (cr, cc) in cells)
                                if (!knownSafe[cr, cc]) { knownSafe[cr, cc] = true; deduced = true; }
                        }
                        else if (mines == cells.Count)
                        {
                            foreach (var (cr, cc) in cells)
                                if (!knownMine[cr, cc]) { knownMine[cr, cc] = true; deduced = true; }
                        }
                    }
                    if (deduced)
                    {
                        for (int i = 0; i < constraints.Count; i++)
                        {
                            var (cells, mines) = constraints[i];
                            int removedMines = 0;
                            var toRemove = new List<(int, int)>();
                            foreach (var cell in cells)
                            {
                                if (knownSafe[cell.Item1, cell.Item2]) toRemove.Add(cell);
                                else if (knownMine[cell.Item1, cell.Item2]) { toRemove.Add(cell); removedMines++; }
                            }
                            foreach (var cell in toRemove) cells.Remove(cell);
                            constraints[i] = (cells, mines - removedMines);
                        }
                    }
                }
            }

            // Re-run simple propagation after subset deductions
            changed = true;
            while (changed)
            {
                changed = false;
                for (int r = 0; r < H; r++)
                {
                    for (int c = 0; c < W; c++)
                    {
                        if (board[r, c] < '0' || board[r, c] > '8') continue;
                        int num = board[r, c] - '0';
                        var unknowns = new List<(int, int)>();
                        int mineCount = 0;
                        for (int d = 0; d < 8; d++)
                        {
                            int nr = r + dx[d], nc = c + dy[d];
                            if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                            if (knownMine[nr, nc]) mineCount++;
                            else if (!knownSafe[nr, nc]) unknowns.Add((nr, nc));
                        }
                        if (unknowns.Count == 0) continue;
                        int remaining = num - mineCount;
                        if (remaining == 0)
                        {
                            foreach (var (ur, uc) in unknowns)
                                if (!knownSafe[ur, uc]) { knownSafe[ur, uc] = true; changed = true; }
                        }
                        else if (remaining == unknowns.Count)
                        {
                            foreach (var (ur, uc) in unknowns)
                                if (!knownMine[ur, uc]) { knownMine[ur, uc] = true; changed = true; }
                        }
                    }
                }
            }
        }

        // Phase 3: Backtracking enumeration on connected components
        frontierSet.Clear();
        var activeConstraints = BuildConstraints(frontierSet);

        // Remove already-resolved cells
        var remainingFrontier = new HashSet<(int, int)>();
        foreach (var f in frontierSet)
            if (!knownSafe[f.Item1, f.Item2] && !knownMine[f.Item1, f.Item2])
                remainingFrontier.Add(f);

        if (remainingFrontier.Count > 0 && remainingFrontier.Count <= 500)
        {
            // Rebuild constraints for remaining frontier only
            activeConstraints.Clear();
            for (int r = 0; r < H; r++)
            {
                for (int c = 0; c < W; c++)
                {
                    if (board[r, c] < '0' || board[r, c] > '8') continue;
                    int num = board[r, c] - '0';
                    var unknowns = new HashSet<(int, int)>();
                    int mineCount = 0;
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dx[d], nc = c + dy[d];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        if (knownMine[nr, nc]) mineCount++;
                        else if (!knownSafe[nr, nc] && remainingFrontier.Contains((nr, nc)))
                            unknowns.Add((nr, nc));
                    }
                    int remaining = num - mineCount;
                    if (unknowns.Count > 0 && remaining >= 0)
                        activeConstraints.Add((unknowns, remaining));
                }
            }

            var cellToIdx = new Dictionary<(int, int), int>();
            var idxToCell = new List<(int, int)>();
            foreach (var f in remainingFrontier)
            {
                cellToIdx[f] = idxToCell.Count;
                idxToCell.Add(f);
            }

            int n = idxToCell.Count;
            if (n > 0)
            {
                // Find connected components via union-find
                var parent = new int[n];
                for (int i = 0; i < n; i++) parent[i] = i;
                int Find(int x) { while (parent[x] != x) { parent[x] = parent[parent[x]]; x = parent[x]; } return x; }
                void Union(int a, int b) { parent[Find(a)] = Find(b); }

                foreach (var (cells, _) in activeConstraints)
                {
                    int first = -1;
                    foreach (var cell in cells)
                    {
                        if (cellToIdx.TryGetValue(cell, out int idx))
                        {
                            if (first == -1) first = idx;
                            else Union(first, idx);
                        }
                    }
                }

                var components = new Dictionary<int, List<int>>();
                for (int i = 0; i < n; i++)
                {
                    int root = Find(i);
                    if (!components.ContainsKey(root)) components[root] = new List<int>();
                    components[root].Add(i);
                }

                foreach (var comp in components.Values)
                {
                    if (comp.Count > 30) continue;
                    var compCells = comp.Select(i => idxToCell[i]).ToList();
                    var compCellSet = new HashSet<(int, int)>(compCells);
                    var compCellToIdx = new Dictionary<(int, int), int>();
                    for (int i = 0; i < compCells.Count; i++)
                        compCellToIdx[compCells[i]] = i;

                    var compConstraints = new List<(HashSet<(int, int)> cells, int mines)>();
                    foreach (var (cells, mines) in activeConstraints)
                    {
                        var inter = new HashSet<(int, int)>(cells);
                        inter.IntersectWith(compCellSet);
                        if (inter.Count > 0)
                        {
                            int interMines = mines;
                            foreach (var cell in cells)
                                if (!compCellSet.Contains(cell) && knownMine[cell.Item1, cell.Item2])
                                    interMines--;
                            compConstraints.Add((inter, mines));
                        }
                    }

                    var canBeSafe = new bool[compCells.Count];
                    var canBeMine = new bool[compCells.Count];
                    EnumerateAll(compCells, compCellToIdx, compConstraints, canBeSafe, canBeMine);
                    for (int i = 0; i < compCells.Count; i++)
                    {
                        if (canBeSafe[i] && !canBeMine[i])
                            knownSafe[compCells[i].Item1, compCells[i].Item2] = true;
                        if (canBeMine[i] && !canBeSafe[i])
                            knownMine[compCells[i].Item1, compCells[i].Item2] = true;
                    }
                }
            }
        }

        var results = new List<(int, int)>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == ' ' && knownSafe[r, c])
                    results.Add((c, r));

        var sb = new System.Text.StringBuilder();
        sb.AppendLine(results.Count.ToString());
        foreach (var (x, y) in results)
            sb.AppendLine(x + " " + y);
        Console.Write(sb);
    }

    static List<(HashSet<(int, int)> cells, int mines)> BuildConstraints(HashSet<(int, int)> frontierSet)
    {
        var constraints = new List<(HashSet<(int, int)> cells, int mines)>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] < '0' || board[r, c] > '8') continue;
                int num = board[r, c] - '0';
                var unknowns = new HashSet<(int, int)>();
                int mineCount = 0;
                for (int d = 0; d < 8; d++)
                {
                    int nr = r + dx[d], nc = c + dy[d];
                    if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                    if (knownMine[nr, nc]) mineCount++;
                    else if (!knownSafe[nr, nc]) unknowns.Add((nr, nc));
                }
                int remaining = num - mineCount;
                if (unknowns.Count > 0 && remaining >= 0)
                {
                    constraints.Add((unknowns, remaining));
                    foreach (var u in unknowns) frontierSet.Add(u);
                }
            }
        }
        return constraints;
    }

    static void EnumerateAll(List<(int, int)> cells, Dictionary<(int, int), int> cellToIdx,
        List<(HashSet<(int, int)> cells, int mines)> constraints, bool[] canBeSafe, bool[] canBeMine)
    {
        int n = cells.Count;
        var idxConstraints = new List<(int[] indices, int mines)>();
        foreach (var (cCells, mines) in constraints)
        {
            var indices = new List<int>();
            foreach (var c in cCells)
                if (cellToIdx.TryGetValue(c, out int idx))
                    indices.Add(idx);
            if (indices.Count > 0)
                idxConstraints.Add((indices.ToArray(), mines));
        }

        var assignment = new int[n]; // 0=unassigned, 1=safe, 2=mine

        // Precompute which constraints involve each cell for faster pruning
        var cellConstraints = new List<int>[n];
        for (int i = 0; i < n; i++) cellConstraints[i] = new List<int>();
        for (int ci = 0; ci < idxConstraints.Count; ci++)
            foreach (int idx in idxConstraints[ci].indices)
                cellConstraints[idx].Add(ci);

        void Backtrack(int pos)
        {
            if (pos == n)
            {
                for (int i = 0; i < n; i++)
                {
                    if (assignment[i] == 1) canBeSafe[i] = true;
                    else canBeMine[i] = true;
                }
                return;
            }

            for (int val = 1; val <= 2; val++)
            {
                assignment[pos] = val;
                bool ok = true;
                foreach (int ci in cellConstraints[pos])
                {
                    var (indices, mines) = idxConstraints[ci];
                    int mc = 0, unassigned = 0;
                    foreach (int idx in indices)
                    {
                        if (assignment[idx] == 2) mc++;
                        else if (assignment[idx] == 0) unassigned++;
                    }
                    if (mc > mines || mc + unassigned < mines) { ok = false; break; }
                }
                if (ok) Backtrack(pos + 1);
                if (canBeSafe[pos] && canBeMine[pos])
                {
                    // Early termination: this cell already determined to be both
                    // Check if ALL cells already have both set
                    bool allDetermined = true;
                    for (int i = pos + 1; i < n; i++)
                        if (!canBeSafe[i] || !canBeMine[i]) { allDetermined = false; break; }
                    if (allDetermined) { assignment[pos] = 0; return; }
                }
            }
            assignment[pos] = 0;
        }

        Backtrack(0);
    }
}
