using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static byte[] _buf = new byte[1 << 16];
    static int _bp, _bl;
    static Stream _sin;

    static int RB()
    {
        if (_bp >= _bl) { _bl = _sin.Read(_buf, 0, _buf.Length); _bp = 0; if (_bl <= 0) return -1; }
        return _buf[_bp++];
    }

    static int RI()
    {
        int c = RB();
        while (c == ' ' || c == '\n' || c == '\r' || c == '\t') c = RB();
        bool neg = c == '-';
        if (neg) c = RB();
        int v = 0;
        while (c >= '0' && c <= '9') { v = v * 10 + c - '0'; c = RB(); }
        return neg ? -v : v;
    }

    static void Main()
    {
        _sin = Console.OpenStandardInput();
        int N = RI(), M = RI();

        if (M == 0) { Console.WriteLine("0"); return; }

        var eu = new int[M];
        var ev = new int[M];
        var ew = new int[M];
        var deg = new int[N];
        var adj = new List<(int v, int w)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int)>();

        for (int i = 0; i < M; i++)
        {
            eu[i] = RI(); ev[i] = RI(); ew[i] = RI();
            deg[eu[i]]++;
            deg[ev[i]]++;
            adj[eu[i]].Add((ev[i], ew[i]));
            adj[ev[i]].Add((eu[i], ew[i]));
        }

        var odd = new List<int>();
        for (int i = 0; i < N; i++)
            if (deg[i] % 2 == 1) odd.Add(i);

        var extra = new List<(int u, int v)>();
        int K = odd.Count;
        if (K > 0)
        {
            if (K <= 20)
                ExactMatch(N, adj, odd, K, extra);
            else
                GreedyMatch(N, adj, odd, K, extra);
        }

        int TE = M + extra.Count;
        var ea = new List<(int to, int idx)>[N];
        for (int i = 0; i < N; i++) ea[i] = new List<(int, int)>();
        for (int i = 0; i < M; i++)
        {
            ea[eu[i]].Add((ev[i], i));
            ea[ev[i]].Add((eu[i], i));
        }
        for (int i = 0; i < extra.Count; i++)
        {
            int idx = M + i;
            ea[extra[i].u].Add((extra[i].v, idx));
            ea[extra[i].v].Add((extra[i].u, idx));
        }

        var used = new bool[TE];
        var ptr = new int[N];
        var stk = new Stack<int>();
        var cir = new List<int>(TE + 1);
        stk.Push(0);

        while (stk.Count > 0)
        {
            int u = stk.Peek();
            bool found = false;
            while (ptr[u] < ea[u].Count)
            {
                var (to, idx) = ea[u][ptr[u]];
                ptr[u]++;
                if (!used[idx])
                {
                    used[idx] = true;
                    stk.Push(to);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                stk.Pop();
                cir.Add(u);
            }
        }

        var sb = new StringBuilder(cir.Count * 5);
        for (int i = 0; i < cir.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(cir[i]);
        }
        using var sw = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 16);
        sw.WriteLine(sb.ToString());
    }

    static void ExactMatch(int N, List<(int v, int w)>[] adj, List<int> odd, int K, List<(int u, int v)> extra)
    {
        var ds = new long[K][];
        var ps = new int[K][];
        for (int i = 0; i < K; i++)
            (ds[i], ps[i]) = DijkstraFull(N, adj, odd[i]);

        int full = (1 << K) - 1;
        var dp = new long[1 << K];
        var match = new int[1 << K];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;

        for (int mask = 0; mask < (1 << K); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;
            int lo = 0;
            while ((mask & (1 << lo)) != 0) lo++;
            for (int j = lo + 1; j < K; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int nm = mask | (1 << lo) | (1 << j);
                long c = dp[mask] + ds[lo][odd[j]];
                if (c < dp[nm])
                {
                    dp[nm] = c;
                    match[nm] = (lo << 16) | j;
                }
            }
        }

        int cm = full;
        while (cm != 0)
        {
            int pk = match[cm];
            int ii = pk >> 16, jj = pk & 0xFFFF;
            cm ^= (1 << ii) | (1 << jj);
            TracePath(odd[ii], odd[jj], ps[ii], extra);
        }
    }

    static void GreedyMatch(int N, List<(int v, int w)>[] adj, List<int> odd, int K, List<(int u, int v)> extra)
    {
        var um = new bool[N];
        foreach (var v in odd) um[v] = true;
        int rem = K;

        var dist = new long[N];
        var prev = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(prev, -1);

        var vis = new List<int>(1024);
        var pq = new PriorityQueue<int, long>();
        int oi = 0;

        while (rem > 1)
        {
            while (oi < odd.Count && !um[odd[oi]]) oi++;
            if (oi >= odd.Count) break;
            int src = odd[oi++];
            um[src] = false;

            vis.Clear();
            pq.Clear();
            dist[src] = 0;
            vis.Add(src);
            pq.Enqueue(src, 0);
            int near = -1;

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[u]) continue;
                if (u != src && um[u]) { near = u; break; }
                foreach (var (v, w) in adj[u])
                {
                    long nd = d + w;
                    if (nd < dist[v])
                    {
                        if (dist[v] == long.MaxValue) vis.Add(v);
                        dist[v] = nd;
                        prev[v] = u;
                        pq.Enqueue(v, nd);
                    }
                }
            }

            if (near != -1)
            {
                um[near] = false;
                rem -= 2;
                TracePath(src, near, prev, extra);
            }
            else
            {
                rem--;
            }

            foreach (var v in vis) { dist[v] = long.MaxValue; prev[v] = -1; }
        }
    }

    static (long[], int[]) DijkstraFull(int N, List<(int v, int w)>[] adj, int src)
    {
        var dist = new long[N];
        var prev = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(prev, -1);
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            foreach (var (v, w) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    prev[v] = u;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, prev);
    }

    static void TracePath(int from, int to, int[] prev, List<(int u, int v)> extra)
    {
        int cur = to;
        while (cur != from && cur != -1)
        {
            int p = prev[cur];
            if (p >= 0) extra.Add((p, cur));
            cur = p;
        }
    }
}
