using System;
using System.Collections.Generic;
using System.Linq;

class TetrisPlayer
{
    static int Width, Height;
    static int[] board;

    static readonly int[][][][] Pieces = new int[][][][]
    {
        // I piece (index 0)
        new int[][][]{
            new int[][]{new[]{0,0},new[]{1,0},new[]{2,0},new[]{3,0}},       // rot 0: horizontal
            new int[][]{new[]{0,0},new[]{0,1},new[]{0,2},new[]{0,3}},       // rot 1: vertical
            new int[][]{new[]{0,0},new[]{1,0},new[]{2,0},new[]{3,0}},       // rot 2: horizontal
            new int[][]{new[]{0,0},new[]{0,1},new[]{0,2},new[]{0,3}}        // rot 3: vertical
        },
        // O piece (index 1)
        new int[][][]{
            new int[][]{new[]{0,0},new[]{1,0},new[]{0,1},new[]{1,1}},
            new int[][]{new[]{0,0},new[]{1,0},new[]{0,1},new[]{1,1}},
            new int[][]{new[]{0,0},new[]{1,0},new[]{0,1},new[]{1,1}},
            new int[][]{new[]{0,0},new[]{1,0},new[]{0,1},new[]{1,1}}
        },
        // T piece (index 2)
        new int[][][]{
            new int[][]{new[]{0,0},new[]{1,0},new[]{2,0},new[]{1,1}},       // rot 0: T pointing down
            new int[][]{new[]{0,0},new[]{0,1},new[]{0,2},new[]{1,1}},       // rot 1: T pointing right
            new int[][]{new[]{1,0},new[]{0,1},new[]{1,1},new[]{2,1}},       // rot 2: T pointing up
            new int[][]{new[]{1,0},new[]{1,1},new[]{1,2},new[]{0,1}}        // rot 3: T pointing left
        },
        // S piece (index 3)
        new int[][][]{
            new int[][]{new[]{1,0},new[]{2,0},new[]{0,1},new[]{1,1}},       // rot 0: horizontal
            new int[][]{new[]{0,0},new[]{0,1},new[]{1,1},new[]{1,2}},       // rot 1: vertical
            new int[][]{new[]{1,0},new[]{2,0},new[]{0,1},new[]{1,1}},       // rot 2
            new int[][]{new[]{0,0},new[]{0,1},new[]{1,1},new[]{1,2}}        // rot 3
        },
        // Z piece (index 4)
        new int[][][]{
            new int[][]{new[]{0,0},new[]{1,0},new[]{1,1},new[]{2,1}},       // rot 0: horizontal
            new int[][]{new[]{1,0},new[]{0,1},new[]{1,1},new[]{0,2}},       // rot 1: vertical
            new int[][]{new[]{0,0},new[]{1,0},new[]{1,1},new[]{2,1}},       // rot 2
            new int[][]{new[]{1,0},new[]{0,1},new[]{1,1},new[]{0,2}}        // rot 3
        },
        // J piece (index 5)
        new int[][][]{
            new int[][]{new[]{0,0},new[]{0,1},new[]{1,1},new[]{2,1}},       // rot 0
            new int[][]{new[]{0,0},new[]{1,0},new[]{0,1},new[]{0,2}},       // rot 1
            new int[][]{new[]{0,0},new[]{1,0},new[]{2,0},new[]{2,1}},       // rot 2
            new int[][]{new[]{1,0},new[]{1,1},new[]{0,2},new[]{1,2}}        // rot 3
        },
        // L piece (index 6)
        new int[][][]{
            new int[][]{new[]{2,0},new[]{0,1},new[]{1,1},new[]{2,1}},       // rot 0
            new int[][]{new[]{0,0},new[]{0,1},new[]{0,2},new[]{1,2}},       // rot 1
            new int[][]{new[]{0,0},new[]{1,0},new[]{2,0},new[]{0,1}},       // rot 2
            new int[][]{new[]{0,0},new[]{1,0},new[]{1,1},new[]{1,2}}        // rot 3
        }
    };

    static readonly Dictionary<string, int> PieceIndex = new Dictionary<string, int>
    {
        {"I",0},{"O",1},{"T",2},{"S",3},{"Z",4},{"J",5},{"L",6}
    };

    static int DropPiece(int[] b, int[][] cells, int col)
    {
        int maxDrop = Height;
        foreach (var c in cells)
        {
            int cx = col + c[0];
            if (cx < 0 || cx >= Width) return -1;
            int drop = -1;
            for (int r = c[1]; r < Height; r++)
            {
                if (b[r * Width + cx] != 0) break;
                drop = r - c[1];
            }
            if (drop < 0) return -1;
            maxDrop = Math.Min(maxDrop, drop);
        }
        foreach (var c in cells)
        {
            int cy = maxDrop + c[1];
            if (cy < 0 || cy >= Height) return -1;
        }
        return maxDrop;
    }

    static bool PlacePiece(int[] b, int[][] cells, int col, int dropRow)
    {
        foreach (var c in cells)
        {
            int cx = col + c[0];
            int cy = dropRow + c[1];
            if (cy < 0 || cy >= Height || cx < 0 || cx >= Width) return false;
            if (b[cy * Width + cx] != 0) return false;
        }
        foreach (var c in cells)
        {
            b[(dropRow + c[1]) * Width + (col + c[0])] = 1;
        }
        return true;
    }

    static int ClearLines(int[] b)
    {
        int cleared = 0;
        int writeRow = Height - 1;
        for (int r = Height - 1; r >= 0; r--)
        {
            bool full = true;
            for (int c = 0; c < Width; c++)
            {
                if (b[r * Width + c] == 0) { full = false; break; }
            }
            if (!full)
            {
                if (writeRow != r)
                    Array.Copy(b, r * Width, b, writeRow * Width, Width);
                writeRow--;
            }
            else cleared++;
        }
        for (int r = 0; r <= writeRow; r++)
            for (int c = 0; c < Width; c++)
                b[r * Width + c] = 0;
        return cleared;
    }

    static double Evaluate(int[] b, int linesCleared)
    {
        double score = linesCleared * 760.0;

        int aggregateHeight = 0;
        int holes = 0;
        int bumpiness = 0;
        int[] colHeights = new int[Width];

        for (int c = 0; c < Width; c++)
        {
            for (int r = 0; r < Height; r++)
            {
                if (b[r * Width + c] != 0)
                {
                    colHeights[c] = Height - r;
                    break;
                }
            }
            aggregateHeight += colHeights[c];
        }

        for (int c = 0; c < Width; c++)
        {
            bool blockFound = false;
            for (int r = 0; r < Height; r++)
            {
                if (b[r * Width + c] != 0) blockFound = true;
                else if (blockFound) holes++;
            }
        }

        for (int c = 0; c < Width - 1; c++)
            bumpiness += Math.Abs(colHeights[c] - colHeights[c + 1]);

        int maxHeight = 0;
        for (int c = 0; c < Width; c++)
            if (colHeights[c] > maxHeight) maxHeight = colHeights[c];

        int coveredDepth = 0;
        for (int c = 0; c < Width; c++)
        {
            int above = 0;
            for (int r = 0; r < Height; r++)
            {
                if (b[r * Width + c] != 0) above++;
                else if (above > 0) coveredDepth += above;
            }
        }

        int rowTrans = 0;
        for (int r = Height - maxHeight; r < Height; r++)
        {
            for (int c = 0; c < Width - 1; c++)
            {
                if (b[r * Width + c] != b[r * Width + c + 1])
                    rowTrans++;
            }
        }

        int colTrans = 0;
        for (int c = 0; c < Width; c++)
        {
            for (int r = Height - maxHeight; r < Height - 1; r++)
            {
                if (b[r * Width + c] != b[(r + 1) * Width + c])
                    colTrans++;
            }
        }

        score -= aggregateHeight * 4.5;
        score -= holes * 35.0;
        score -= coveredDepth * 8.0;
        score -= bumpiness * 3.2;
        score -= maxHeight * 1.5;
        score -= rowTrans * 3.2;
        score -= colTrans * 3.2;

        if (maxHeight > Height - 4)
            score -= (maxHeight - (Height - 4)) * 80.0;

        return score;
    }

    static void Main()
    {
        string[] tokens = ReadTokens(3);
        Width = int.Parse(tokens[0]);
        Height = int.Parse(tokens[1]);
        int numPieces = int.Parse(tokens[2]);

        board = new int[Width * Height];

        for (int p = 0; p < numPieces; p++)
        {
            tokens = ReadTokens(1);
            string pieceType = tokens[0];
            int pi = PieceIndex[pieceType];

            int bestRot = 0, bestCol = 0;
            double bestScore = double.MinValue;

            for (int rot = 0; rot < 4; rot++)
            {
                int[][] cells = Pieces[pi][rot];
                int maxX = 0, maxY = 0;
                foreach (var c in cells)
                {
                    if (c[0] > maxX) maxX = c[0];
                    if (c[1] > maxY) maxY = c[1];
                }

                for (int col = 0; col <= Width - 1 - maxX; col++)
                {
                    int dropRow = DropPiece(board, cells, col);
                    if (dropRow < 0) continue;
                    if (dropRow + maxY >= Height) continue;

                    int[] temp = (int[])board.Clone();
                    if (!PlacePiece(temp, cells, col, dropRow)) continue;

                    int cleared = ClearLines(temp);
                    double sc = Evaluate(temp, cleared);

                    if (sc > bestScore)
                    {
                        bestScore = sc;
                        bestRot = rot;
                        bestCol = col;
                    }
                }
            }

            Console.WriteLine(bestRot + " " + bestCol);
            Console.Out.Flush();

            tokens = ReadTokens(1);
            if (tokens[0] == null || tokens[0].StartsWith("gameover")) return;

            int[][] bestCells = Pieces[pi][bestRot];
            int dr = DropPiece(board, bestCells, bestCol);
            if (dr >= 0)
            {
                PlacePiece(board, bestCells, bestCol, dr);
                ClearLines(board);
            }
        }
    }

    static List<string> tokenBuffer = new List<string>();
    static string[] ReadTokens(int count)
    {
        while (tokenBuffer.Count < count)
        {
            string line = Console.ReadLine();
            if (line == null) return new string[count];
            line = line.Trim();
            if (line.Length == 0) continue;
            tokenBuffer.AddRange(line.Split(new char[]{' ','\t','\r','\n'}, StringSplitOptions.RemoveEmptyEntries));
        }
        string[] result = new string[count];
        for (int i = 0; i < count; i++)
            result[i] = tokenBuffer[i];
        tokenBuffer.RemoveRange(0, Math.Min(count, tokenBuffer.Count));
        return result;
    }
}
