#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
#include <chrono>

struct Rect { int w, h, idx; };
struct Placement { int board, x, y, rotated; };
struct FreeRect { int x, y, w, h; };

class MaxRectsBin {
    int binW, binH;
    std::vector<FreeRect> free_;

public:
    MaxRectsBin(int w, int h) : binW(w), binH(h) {
        free_.push_back({0, 0, w, h});
    }

    bool place(int rw, int rh, int &px, int &py, bool &rot, int method) {
        int bs1 = INT_MAX, bs2 = INT_MAX;
        int bx = 0, by = 0;
        bool brot = false;
        bool found = false;

        for (auto &fr : free_) {
            for (int r = 0; r < 2; r++) {
                int tw = r ? rh : rw, th = r ? rw : rh;
                if (tw > fr.w || th > fr.h) continue;
                int s1, s2;
                calcScore(method, tw, th, fr, s1, s2);
                if (s1 < bs1 || (s1 == bs1 && s2 < bs2)) {
                    bs1 = s1; bs2 = s2;
                    bx = fr.x; by = fr.y;
                    brot = (r == 1);
                    found = true;
                }
            }
        }
        if (!found) return false;

        px = bx; py = by; rot = brot;
        int pw = rot ? rh : rw, ph = rot ? rw : rh;
        splitAndPrune(px, py, pw, ph);
        return true;
    }

private:
    void calcScore(int m, int rw, int rh, const FreeRect &fr, int &s1, int &s2) {
        int dw = fr.w - rw, dh = fr.h - rh;
        switch (m) {
            case 0: s1 = std::min(dw, dh); s2 = std::max(dw, dh); break;
            case 1: s1 = std::max(dw, dh); s2 = std::min(dw, dh); break;
            case 2: s1 = fr.w * fr.h - rw * rh; s2 = std::min(dw, dh); break;
            default: s1 = fr.y; s2 = fr.x; break;
        }
    }

    void splitAndPrune(int px, int py, int pw, int ph) {
        int n = (int)free_.size();
        for (int i = n - 1; i >= 0; i--) {
            FreeRect &fr = free_[i];
            if (px >= fr.x + fr.w || px + pw <= fr.x ||
                py >= fr.y + fr.h || py + ph <= fr.y)
                continue;

            if (px > fr.x)
                free_.push_back({fr.x, fr.y, px - fr.x, fr.h});
            if (px + pw < fr.x + fr.w)
                free_.push_back({px + pw, fr.y, fr.x + fr.w - px - pw, fr.h});
            if (py > fr.y)
                free_.push_back({fr.x, fr.y, fr.w, py - fr.y});
            if (py + ph < fr.y + fr.h)
                free_.push_back({fr.x, py + ph, fr.w, fr.y + fr.h - py - ph});

            free_[i] = free_.back();
            free_.pop_back();
        }

        for (int i = 0; i < (int)free_.size(); i++) {
            for (int j = i + 1; j < (int)free_.size(); ) {
                if (contains(free_[i], free_[j])) {
                    free_[j] = free_.back(); free_.pop_back();
                } else if (contains(free_[j], free_[i])) {
                    free_[i] = free_[j];
                    free_[j] = free_.back(); free_.pop_back();
                    j = i + 1;
                } else {
                    j++;
                }
            }
        }

        if (free_.size() > 600) {
            std::sort(free_.begin(), free_.end(),
                [](const FreeRect &a, const FreeRect &b) {
                    return (long long)a.w * a.h > (long long)b.w * b.h;
                });
            free_.resize(600);
        }
    }

    static bool contains(const FreeRect &a, const FreeRect &b) {
        return b.x >= a.x && b.y >= a.y &&
               b.x + b.w <= a.x + a.w && b.y + b.h <= a.y + a.h;
    }
};

struct Result {
    int boards;
    std::vector<Placement> pl;
};

Result solve(const std::vector<Rect> &rects, int N, int W, int H,
             const std::vector<int> &order, int method) {
    std::vector<MaxRectsBin> bins;
    std::vector<Placement> pl(N);

    for (int oi = 0; oi < N; oi++) {
        int i = order[oi];
        int rw = rects[i].w, rh = rects[i].h;
        bool placed = false;

        for (int b = 0; b < (int)bins.size(); b++) {
            int px, py; bool rot;
            if (bins[b].place(rw, rh, px, py, rot, method)) {
                pl[i] = {b, px, py, rot ? 1 : 0};
                placed = true;
                break;
            }
        }
        if (!placed) {
            bins.emplace_back(W, H);
            int px, py; bool rot;
            if (bins.back().place(rw, rh, px, py, rot, method)) {
                pl[i] = {(int)bins.size() - 1, px, py, rot ? 1 : 0};
            } else {
                pl[i] = {(int)bins.size() - 1, 0, 0, 0};
            }
        }
    }
    return {(int)bins.size(), std::move(pl)};
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, W, H;
    std::cin >> N >> W >> H;

    std::vector<Rect> rects(N);
    for (int i = 0; i < N; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].idx = i;
    }

    auto t0 = std::chrono::steady_clock::now();
    auto elapsed = [&]() {
        return std::chrono::duration_cast<std::chrono::milliseconds>(
            std::chrono::steady_clock::now() - t0).count();
    };

    auto makeOrder = [&](auto cmp) {
        std::vector<int> ord(N);
        for (int i = 0; i < N; i++) ord[i] = i;
        std::sort(ord.begin(), ord.end(), cmp);
        return ord;
    };

    std::vector<int> orderArea = makeOrder([&](int a, int b) {
        long long aa = (long long)rects[a].w * rects[a].h;
        long long ab = (long long)rects[b].w * rects[b].h;
        if (aa != ab) return aa > ab;
        return std::max(rects[a].w, rects[a].h) > std::max(rects[b].w, rects[b].h);
    });

    std::vector<int> orderMaxSide = makeOrder([&](int a, int b) {
        int ma = std::max(rects[a].w, rects[a].h);
        int mb = std::max(rects[b].w, rects[b].h);
        if (ma != mb) return ma > mb;
        return (long long)rects[a].w * rects[a].h > (long long)rects[b].w * rects[b].h;
    });

    std::vector<int> orderHeight = makeOrder([&](int a, int b) {
        int ha = std::max(rects[a].h, rects[a].w);
        int hb = std::max(rects[b].h, rects[b].w);
        if (ha != hb) return ha > hb;
        return std::min(rects[a].w, rects[a].h) > std::min(rects[b].w, rects[b].h);
    });

    std::vector<int> orderPerim = makeOrder([&](int a, int b) {
        int pa = rects[a].w + rects[a].h;
        int pb = rects[b].w + rects[b].h;
        if (pa != pb) return pa > pb;
        return (long long)rects[a].w * rects[a].h > (long long)rects[b].w * rects[b].h;
    });

    std::vector<std::vector<int>*> sorts = {&orderArea, &orderMaxSide, &orderHeight, &orderPerim};
    int numMethods = 4;

    int bestBoards = INT_MAX;
    std::vector<Placement> bestPl;

    int timeBudget = (N > 5000) ? 15000 : 25000;

    for (auto *srt : sorts) {
        for (int m = 0; m < numMethods; m++) {
            if (elapsed() > timeBudget) break;
            auto res = solve(rects, N, W, H, *srt, m);
            if (res.boards < bestBoards) {
                bestBoards = res.boards;
                bestPl = std::move(res.pl);
            }
        }
        if (elapsed() > timeBudget) break;
    }

    std::cout << bestBoards << "\n";
    for (int i = 0; i < N; i++) {
        std::cout << bestPl[i].board << " " << bestPl[i].x << " "
                  << bestPl[i].y << " " << bestPl[i].rotated << "\n";
    }
    return 0;
}
