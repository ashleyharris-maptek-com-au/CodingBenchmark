#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>
#include <chrono>

using namespace std;

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double a, double b, double c) : x(a), y(b), z(c) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double norm() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { double n = norm(); return n>1e-15 ? Vec3{x/n,y/n,z/n} : Vec3{0,0,0}; }
};

struct Mat3 {
    double m[3][3];
    Vec3 operator*(const Vec3& v) const {
        return {m[0][0]*v.x+m[0][1]*v.y+m[0][2]*v.z,
                m[1][0]*v.x+m[1][1]*v.y+m[1][2]*v.z,
                m[2][0]*v.x+m[2][1]*v.y+m[2][2]*v.z};
    }
};

Mat3 eulerToMat(double rx, double ry, double rz) {
    double cx=cos(rx),sx=sin(rx),cy=cos(ry),sy=sin(ry),cz=cos(rz),sz=sin(rz);
    Mat3 r;
    r.m[0][0]=cy*cz; r.m[0][1]=-cy*sz; r.m[0][2]=sy;
    r.m[1][0]=sx*sy*cz+cx*sz; r.m[1][1]=-sx*sy*sz+cx*cz; r.m[1][2]=-sx*cy;
    r.m[2][0]=-cx*sy*cz+sx*sz; r.m[2][1]=cx*sy*sz+sx*cz; r.m[2][2]=cx*cy;
    return r;
}

array<Vec3,4> baseTetraVerts() {
    double s = 1.0/sqrt(8.0);
    return {Vec3(s,s,s), Vec3(s,-s,-s), Vec3(-s,s,-s), Vec3(-s,-s,s)};
}

array<Vec3,4> getTetraVerts(Vec3 center, double rx, double ry, double rz) {
    static auto base = baseTetraVerts();
    Mat3 rot = eulerToMat(rx,ry,rz);
    array<Vec3,4> v;
    for(int i=0;i<4;i++) v[i] = center + rot*base[i];
    return v;
}

struct Plane { Vec3 normal; double d; };

vector<Plane> containerPlanes;

void buildConvexHullPlanes(const vector<Vec3>& verts) {
    containerPlanes.clear();
    int n=(int)verts.size();
    if(n<4) return;
    for(int i=0;i<n;i++) {
        for(int j=i+1;j<n;j++) {
            for(int k=j+1;k<n;k++) {
                Vec3 e1=verts[j]-verts[i], e2=verts[k]-verts[i];
                Vec3 nm=e1.cross(e2);
                double len=nm.norm();
                if(len<1e-12) continue;
                nm=nm*(1.0/len);
                double d=nm.dot(verts[i]);
                bool allBelow=true, allAbove=true;
                for(int l=0;l<n;l++) {
                    if(l==i||l==j||l==k) continue;
                    double val=nm.dot(verts[l])-d;
                    if(val>1e-9) allBelow=false;
                    if(val<-1e-9) allAbove=false;
                }
                if(allBelow) containerPlanes.push_back({nm,d});
                else if(allAbove) containerPlanes.push_back({nm*-1.0,-d});
            }
        }
    }
    vector<Plane> unique;
    for(auto& p:containerPlanes) {
        bool dup=false;
        for(auto& q:unique)
            if(abs(p.normal.dot(q.normal)-1.0)<1e-6 && abs(p.d-q.d)<1e-6) {dup=true;break;}
        if(!dup) unique.push_back(p);
    }
    containerPlanes=unique;
}

bool pointInContainer(const Vec3& p) {
    for(auto& pl:containerPlanes)
        if(pl.normal.dot(p)>pl.d+1e-9) return false;
    return true;
}

bool tetraInContainer(const array<Vec3,4>& v) {
    for(auto& p:v) if(!pointInContainer(p)) return false;
    return true;
}

array<double,3> findEulerAngles(const array<Vec3,4>& target) {
    auto base=baseTetraVerts();
    double B[3][3]={{base[0].x,base[1].x,base[2].x},{base[0].y,base[1].y,base[2].y},{base[0].z,base[1].z,base[2].z}};
    double A[3][3]={{target[0].x,target[1].x,target[2].x},{target[0].y,target[1].y,target[2].y},{target[0].z,target[1].z,target[2].z}};
    double det=B[0][0]*(B[1][1]*B[2][2]-B[1][2]*B[2][1])-B[0][1]*(B[1][0]*B[2][2]-B[1][2]*B[2][0])+B[0][2]*(B[1][0]*B[2][1]-B[1][1]*B[2][0]);
    if(abs(det)<1e-15) return{0,0,0};
    double Bi[3][3];
    Bi[0][0]=(B[1][1]*B[2][2]-B[1][2]*B[2][1])/det; Bi[0][1]=(B[0][2]*B[2][1]-B[0][1]*B[2][2])/det; Bi[0][2]=(B[0][1]*B[1][2]-B[0][2]*B[1][1])/det;
    Bi[1][0]=(B[1][2]*B[2][0]-B[1][0]*B[2][2])/det; Bi[1][1]=(B[0][0]*B[2][2]-B[0][2]*B[2][0])/det; Bi[1][2]=(B[0][2]*B[1][0]-B[0][0]*B[1][2])/det;
    Bi[2][0]=(B[1][0]*B[2][1]-B[1][1]*B[2][0])/det; Bi[2][1]=(B[0][1]*B[2][0]-B[0][0]*B[2][1])/det; Bi[2][2]=(B[0][0]*B[1][1]-B[0][1]*B[1][0])/det;
    double R[3][3];
    for(int i=0;i<3;i++) for(int j=0;j<3;j++){R[i][j]=0;for(int k=0;k<3;k++) R[i][j]+=A[i][k]*Bi[k][j];}
    double ry=asin(max(-1.0,min(1.0,R[0][2])));
    double cy2=cos(ry);
    double rx,rz;
    if(abs(cy2)>1e-6){rx=atan2(-R[1][2],R[2][2]);rz=atan2(-R[0][1],R[0][0]);}
    else{rx=atan2(R[2][1],R[1][1]);rz=0;}
    return{rx,ry,rz};
}

int main(){
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    auto t0=chrono::steady_clock::now();
    auto elapsed=[&](){return chrono::duration<double>(chrono::steady_clock::now()-t0).count();};

    int V; double edgeLen;
    cin>>V>>edgeLen;
    vector<Vec3> verts(V);
    for(int i=0;i<V;i++) cin>>verts[i].x>>verts[i].y>>verts[i].z;
    for(auto& v:verts) v=v*(1.0/edgeLen);

    buildConvexHullPlanes(verts);

    Vec3 lo(1e18,1e18,1e18), hi(-1e18,-1e18,-1e18);
    for(auto& v:verts){
        lo.x=min(lo.x,v.x);lo.y=min(lo.y,v.y);lo.z=min(lo.z,v.z);
        hi.x=max(hi.x,v.x);hi.y=max(hi.y,v.y);hi.z=max(hi.z,v.z);
    }

    double a=sqrt(2.0); // FCC lattice constant for unit edge tetrahedra

    // FCC tetrahedra: 8 per unit cell
    // Up-type centroids: (a/4,a/4,a/4), (3a/4,3a/4,a/4), (3a/4,a/4,3a/4), (a/4,3a/4,3a/4)
    // Down-type centroids: (3a/4,a/4,a/4), (a/4,3a/4,a/4), (a/4,a/4,3a/4), (3a/4,3a/4,3a/4)

    // Up tetra relative vertices (centered at origin):
    double q=a/4.0;
    array<Vec3,4> upRel={Vec3(-q,-q,-q),Vec3(q,q,-q),Vec3(q,-q,q),Vec3(-q,q,q)};
    // Down tetra: e.g. centroid at (3a/4,a/4,a/4), vertices (a/2,a/2,0),(a,0,0),(a,a/2,a/2),(a/2,0,a/2)
    // Relative: (-q,q,-q),(q,-q,-q),(q,q,q),(-q,-q,q)
    array<Vec3,4> downRel={Vec3(-q,q,-q),Vec3(q,-q,-q),Vec3(q,q,q),Vec3(-q,-q,q)};

    auto upAng=findEulerAngles(upRel);
    auto downAng=findEulerAngles(downRel);

    struct Off{double ox,oy,oz;int t;};
    vector<Off> offsets={
        {q,q,q,0},{3*q,3*q,q,0},{3*q,q,3*q,0},{q,3*q,3*q,0},
        {3*q,q,q,1},{q,3*q,q,1},{q,q,3*q,1},{3*q,3*q,3*q,1}
    };

    int ni=(int)ceil((hi.x-lo.x)/a)+1;
    int nj=(int)ceil((hi.y-lo.y)/a)+1;
    int nk=(int)ceil((hi.z-lo.z)/a)+1;

    // Cap grid
    long long gs=(long long)ni*nj*nk*8;
    if(gs>30000000){
        double sc=cbrt((double)gs/30000000.0);
        ni=max(1,(int)(ni/sc));nj=max(1,(int)(nj/sc));nk=max(1,(int)(nk/sc));
    }

    struct Result{Vec3 c;double rx,ry,rz;};

    auto tryShift=[&](double sx,double sy,double sz)->vector<Result>{
        vector<Result> res;
        for(int k=-1;k<=nk&&elapsed()<24.0;k++){
            for(int j=-1;j<=nj;j++){
                for(int i=-1;i<=ni;i++){
                    for(auto& off:offsets){
                        Vec3 c(lo.x+i*a+off.ox+sx, lo.y+j*a+off.oy+sy, lo.z+k*a+off.oz+sz);
                        auto& ang=(off.t==0)?upAng:downAng;
                        auto v=getTetraVerts(c,ang[0],ang[1],ang[2]);
                        if(tetraInContainer(v)) res.push_back({c,ang[0],ang[1],ang[2]});
                    }
                }
            }
        }
        return res;
    };

    auto best=tryShift(0,0,0);

    // Try some shifted lattices to find better packing
    if(elapsed()<18.0){
        double step=a/4.0;
        for(int si=0;si<4&&elapsed()<18.0;si++){
            for(int sj=0;sj<4&&elapsed()<18.0;sj++){
                for(int sk=0;sk<4&&elapsed()<18.0;sk++){
                    if(si==0&&sj==0&&sk==0) continue;
                    auto trial=tryShift(si*step,sj*step,sk*step);
                    if(trial.size()>best.size()) best=trial;
                }
            }
        }
    }

    cout<<best.size()<<"\n";
    cout<<fixed;
    cout.precision(10);
    for(auto& r:best){
        cout<<r.c.x*edgeLen<<" "<<r.c.y*edgeLen<<" "<<r.c.z*edgeLen
            <<" "<<r.rx<<" "<<r.ry<<" "<<r.rz<<"\n";
    }
    return 0;
}
