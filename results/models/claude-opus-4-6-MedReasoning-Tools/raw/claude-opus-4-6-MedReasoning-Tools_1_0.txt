#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>
#include <chrono>

using namespace std;

struct Point { float x, y; };

static inline float dist(const Point& a, const Point& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

struct GridIndex {
    int gw, gh;
    float minx, miny, cellw, cellh;
    vector<vector<int>> cells;

    void build(const vector<Point>& pts, int n) {
        int gridSize = max(1, (int)sqrt((double)n / 4.0));
        gw = gh = gridSize;
        minx = miny = 1e18f;
        float maxx = -1e18f, maxy = -1e18f;
        for (int i = 0; i < n; i++) {
            if (pts[i].x < minx) minx = pts[i].x;
            if (pts[i].y < miny) miny = pts[i].y;
            if (pts[i].x > maxx) maxx = pts[i].x;
            if (pts[i].y > maxy) maxy = pts[i].y;
        }
        float pad = 1e-6f;
        cellw = (maxx - minx + pad) / gw;
        cellh = (maxy - miny + pad) / gh;
        if (cellw < 1e-12f) cellw = 1.0f;
        if (cellh < 1e-12f) cellh = 1.0f;
        cells.resize((size_t)gw * gh);
        for (int i = 0; i < n; i++) {
            int cx = min((int)((pts[i].x - minx) / cellw), gw - 1);
            int cy = min((int)((pts[i].y - miny) / cellh), gh - 1);
            cells[(size_t)cy * gw + cx].push_back(i);
        }
    }

    int findNearest(const vector<Point>& pts, int qi, const vector<bool>& used) const {
        float qx = pts[qi].x, qy = pts[qi].y;
        int cx = min((int)((qx - minx) / cellw), gw - 1);
        int cy = min((int)((qy - miny) / cellh), gh - 1);
        float bestD = 1e30f;
        int bestIdx = -1;
        for (int radius = 0; radius <= max(gw, gh); radius++) {
            if (bestIdx >= 0 && radius > 1) {
                float minPossible = (float)(radius - 1) * min(cellw, cellh);
                if (minPossible > bestD) break;
            }
            int x0 = max(0, cx - radius), x1 = min(gw - 1, cx + radius);
            int y0 = max(0, cy - radius), y1 = min(gh - 1, cy + radius);
            for (int yy = y0; yy <= y1; yy++) {
                for (int xx = x0; xx <= x1; xx++) {
                    if (radius > 0 && xx > x0 && xx < x1 && yy > y0 && yy < y1) continue;
                    for (int idx : cells[(size_t)yy * gw + xx]) {
                        if (used[idx]) continue;
                        float d = dist(pts[qi], pts[idx]);
                        if (d < bestD) { bestD = d; bestIdx = idx; }
                    }
                }
            }
            if (bestIdx >= 0 && radius >= 1) break;
        }
        return bestIdx;
    }
};

vector<int> greedyNNSimple(const vector<Point>& pts, int n) {
    vector<int> tour;
    tour.reserve(n);
    vector<bool> used(n, false);
    tour.push_back(0);
    used[0] = true;
    for (int i = 1; i < n; i++) {
        int last = tour.back();
        float bestD = 1e30f;
        int bestIdx = -1;
        for (int j = 0; j < n; j++) {
            if (!used[j]) {
                float d = dist(pts[last], pts[j]);
                if (d < bestD) { bestD = d; bestIdx = j; }
            }
        }
        tour.push_back(bestIdx);
        used[bestIdx] = true;
    }
    return tour;
}

vector<int> greedyNN(const vector<Point>& pts, int n, const GridIndex& grid) {
    vector<int> tour;
    tour.reserve(n);
    vector<bool> used(n, false);
    tour.push_back(0);
    used[0] = true;
    for (int i = 1; i < n; i++) {
        int nxt = grid.findNearest(pts, tour.back(), used);
        tour.push_back(nxt);
        used[nxt] = true;
    }
    return tour;
}

void twoOpt(const vector<Point>& pts, vector<int>& tour, int n, double timeLimit) {
    auto start = chrono::steady_clock::now();
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < n - 2; i++) {
            for (int j = i + 2; j < n; j++) {
                int a = tour[i], b = tour[i+1];
                int c = tour[j], d = tour[(j+1) % n];
                float oldD = dist(pts[a], pts[b]) + dist(pts[c], pts[d]);
                float newD = dist(pts[a], pts[c]) + dist(pts[b], pts[d]);
                if (newD < oldD - 1e-10f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
            if (i % 100 == 0) {
                double elapsed = chrono::duration<double>(chrono::steady_clock::now() - start).count();
                if (elapsed > timeLimit) return;
            }
        }
    }
}

void twoOptLarge(const vector<Point>& pts, vector<int>& tour, int n, const GridIndex& grid, double timeLimit) {
    auto start = chrono::steady_clock::now();
    vector<int> pos(n);
    for (int i = 0; i < n; i++) pos[tour[i]] = i;

    bool improved = true;
    int passCount = 0;
    while (improved) {
        improved = false;
        passCount++;
        for (int idx = 0; idx < n; idx++) {
            int i = idx;
            int a = tour[i];
            int inext = (i + 1) % n;
            int b = tour[inext];
            float dab = dist(pts[a], pts[b]);

            int cx = min((int)((pts[a].x - grid.minx) / grid.cellw), grid.gw - 1);
            int cy = min((int)((pts[a].y - grid.miny) / grid.cellh), grid.gh - 1);

            int maxRad = min(5, max(grid.gw, grid.gh));
            bool found = false;
            for (int radius = 0; radius <= maxRad && !found; radius++) {
                int x0 = max(0, cx - radius), x1 = min(grid.gw - 1, cx + radius);
                int y0 = max(0, cy - radius), y1 = min(grid.gh - 1, cy + radius);
                for (int yy = y0; yy <= y1 && !found; yy++) {
                    for (int xx = x0; xx <= x1 && !found; xx++) {
                        if (radius > 0 && xx > x0 && xx < x1 && yy > y0 && yy < y1) continue;
                        for (int c : grid.cells[(size_t)yy * grid.gw + xx]) {
                            int j = pos[c];
                            if (abs(j - i) <= 1 || (i == 0 && j == n-1) || (j == 0 && i == n-1)) continue;
                            int jj = j, ii = i;
                            if (jj < ii) { swap(ii, jj); }
                            int inxt = ii + 1;
                            int jnxt = (jj + 1) % n;
                            int aa = tour[ii], bb = tour[inxt];
                            int cc = tour[jj], dd = tour[jnxt];
                            float oldCost = dist(pts[aa], pts[bb]) + dist(pts[cc], pts[dd]);
                            float newCost = dist(pts[aa], pts[cc]) + dist(pts[bb], pts[dd]);
                            if (newCost < oldCost - 1e-10f) {
                                reverse(tour.begin() + inxt, tour.begin() + jj + 1);
                                for (int k = inxt; k <= jj; k++) pos[tour[k]] = k;
                                improved = true;
                                found = true;
                                break;
                            }
                        }
                    }
                }
            }

            if (idx % 5000 == 0) {
                double elapsed = chrono::duration<double>(chrono::steady_clock::now() - start).count();
                if (elapsed > timeLimit) return;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    vector<Point> pts(n);
    for (int i = 0; i < n; i++) cin >> pts[i].x >> pts[i].y;

    vector<int> tour;

    if (n <= 1) {
        tour.resize(n);
        iota(tour.begin(), tour.end(), 0);
    } else if (n <= 3000) {
        tour = greedyNNSimple(pts, n);
        double tl = 250.0;
        if (n <= 500) tl = 60.0;
        twoOpt(pts, tour, n, tl);
    } else {
        GridIndex grid;
        grid.build(pts, n);
        tour = greedyNN(pts, n, grid);

        if (n <= 500000) {
            twoOptLarge(pts, tour, n, grid, 200.0);
        } else if (n <= 5000000) {
            twoOptLarge(pts, tour, n, grid, 150.0);
        }
    }

    if (tour[0] != 0) {
        int zp = 0;
        for (int i = 0; i < n; i++) if (tour[i] == 0) { zp = i; break; }
        rotate(tour.begin(), tour.begin() + zp, tour.end());
    }

    for (int i = 0; i < n; i++) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
