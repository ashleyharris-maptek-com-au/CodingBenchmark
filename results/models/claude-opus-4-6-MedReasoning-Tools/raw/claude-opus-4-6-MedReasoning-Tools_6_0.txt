using System;
using System.Diagnostics;
using System.Globalization;

class Program
{
    static double MU;

    static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        var sw = Stopwatch.StartNew();

        string[] parts = Console.ReadLine().Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts[0]);
        MU = double.Parse(parts[1], ci);

        double[] startState = ReadState(ci);
        double[][] stationStates = new double[N][];
        for (int i = 0; i < N; i++)
            stationStates[i] = ReadState(ci);

        if (N == 1) { Console.WriteLine("0"); return; }

        double[] startOE = ToOE(startState);
        double[][] oe = new double[N][];
        for (int i = 0; i < N; i++)
            oe[i] = ToOE(stationStates[i]);

        double[] fromStart = new double[N];
        for (int i = 0; i < N; i++)
            fromStart[i] = TransferCost(startOE, oe[i]);

        double[,] cost = new double[N, N];
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                if (i != j)
                    cost[i, j] = TransferCost(oe[i], oe[j]);

        int[] bestPerm;
        if (N <= 10)
            bestPerm = BruteForce(fromStart, cost, N);
        else
            bestPerm = IteratedLocalSearch(fromStart, cost, N, sw, 240000);

        Console.WriteLine(string.Join(" ", bestPerm));
    }

    static double[] ReadState(CultureInfo ci)
    {
        string[] p = Console.ReadLine().Trim().Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        double[] r = new double[6];
        for (int i = 0; i < 6; i++) r[i] = double.Parse(p[i], ci);
        return r;
    }

    static double Clamp(double v) { return v < -1.0 ? -1.0 : (v > 1.0 ? 1.0 : v); }

    static double[] ToOE(double[] s)
    {
        double x = s[0], y = s[1], z = s[2], vx = s[3], vy = s[4], vz = s[5];
        double r = Math.Sqrt(x * x + y * y + z * z);
        double v = Math.Sqrt(vx * vx + vy * vy + vz * vz);

        double hx = y * vz - z * vy, hy = z * vx - x * vz, hz = x * vy - y * vx;
        double h = Math.Sqrt(hx * hx + hy * hy + hz * hz);

        double nx = -hy, ny = hx;
        double nmag = Math.Sqrt(nx * nx + ny * ny);

        double rdv = x * vx + y * vy + z * vz;
        double c = v * v - MU / r;
        double ex = (c * x - rdv * vx) / MU;
        double ey = (c * y - rdv * vy) / MU;
        double ez = (c * z - rdv * vz) / MU;
        double e = Math.Sqrt(ex * ex + ey * ey + ez * ez);

        double energy = v * v / 2.0 - MU / r;
        double a;
        if (Math.Abs(e - 1.0) > 1e-8)
            a = -MU / (2.0 * energy);
        else
            a = h * h / MU;

        double inc = Math.Acos(Clamp(hz / h));

        double raan = 0;
        if (nmag > 1e-10)
        {
            raan = Math.Acos(Clamp(nx / nmag));
            if (ny < 0) raan = 2 * Math.PI - raan;
        }

        double argp = 0;
        if (nmag > 1e-10 && e > 1e-10)
        {
            argp = Math.Acos(Clamp((nx * ex + ny * ey) / (nmag * e)));
            if (ez < 0) argp = 2 * Math.PI - argp;
        }

        return new double[] { a, e, inc, raan, argp };
    }

    static double TransferCost(double[] from, double[] to)
    {
        double a1 = from[0], e1 = from[1], i1 = from[2], raan1 = from[3];
        double a2 = to[0], e2 = to[1], i2 = to[2], raan2 = to[3];

        double absA1 = Math.Abs(a1);
        double absA2 = Math.Abs(a2);
        if (absA1 < 1.0) absA1 = 1.0;
        if (absA2 < 1.0) absA2 = 1.0;

        double v1 = Math.Sqrt(MU / absA1);
        double v2 = Math.Sqrt(MU / absA2);

        double dvA = Math.Abs(v1 - v2);

        double dRaan = raan1 - raan2;
        double cosP = Math.Cos(i1) * Math.Cos(i2) + Math.Sin(i1) * Math.Sin(i2) * Math.Cos(dRaan);
        double planeAngle = Math.Acos(Clamp(cosP));
        double vAvg = (v1 + v2) / 2.0;
        double dvP = 2.0 * vAvg * Math.Sin(planeAngle / 2.0);

        double dvE = Math.Abs(e1 - e2) * vAvg * 0.5;

        double dArgp = Math.Abs(from[4] - to[4]);
        if (dArgp > Math.PI) dArgp = 2 * Math.PI - dArgp;
        double dvW = dArgp / Math.PI * vAvg * Math.Max(e1, e2) * 0.2;

        return Math.Sqrt(dvA * dvA + dvP * dvP) + dvE + dvW;
    }

    static double PathCost(double[] fromStart, double[,] cost, int[] perm)
    {
        double total = fromStart[perm[0]];
        for (int i = 0; i < perm.Length - 1; i++)
            total += cost[perm[i], perm[i + 1]];
        return total;
    }

    static int[] BruteForce(double[] fromStart, double[,] cost, int N)
    {
        int[] perm = new int[N];
        for (int i = 0; i < N; i++) perm[i] = i;
        int[] best = (int[])perm.Clone();
        double bestCost = PathCost(fromStart, cost, perm);

        int[] c = new int[N];
        int idx = 0;
        while (idx < N)
        {
            if (c[idx] < idx)
            {
                int swapIdx = (idx % 2 == 0) ? 0 : c[idx];
                int tmp = perm[swapIdx];
                perm[swapIdx] = perm[idx];
                perm[idx] = tmp;

                double pc = PathCost(fromStart, cost, perm);
                if (pc < bestCost)
                {
                    bestCost = pc;
                    best = (int[])perm.Clone();
                }

                c[idx]++;
                idx = 0;
            }
            else
            {
                c[idx] = 0;
                idx++;
            }
        }
        return best;
    }

    static int[] NearestNeighbor(double[] fromStart, double[,] cost, int N, int firstStation)
    {
        int[] perm = new int[N];
        bool[] visited = new bool[N];
        perm[0] = firstStation;
        visited[firstStation] = true;

        for (int step = 1; step < N; step++)
        {
            int best = -1;
            double bc = double.MaxValue;
            for (int j = 0; j < N; j++)
            {
                if (!visited[j] && cost[perm[step - 1], j] < bc)
                {
                    bc = cost[perm[step - 1], j];
                    best = j;
                }
            }
            perm[step] = best;
            visited[best] = true;
        }
        return perm;
    }

    static void TwoOptTimed(double[] fromStart, double[,] cost, int[] perm, int N, Stopwatch sw, int timeLimit)
    {
        bool improved = true;
        while (improved && sw.ElapsedMilliseconds < timeLimit)
        {
            improved = false;
            for (int i = 0; i < N - 1; i++)
            {
                if (sw.ElapsedMilliseconds >= timeLimit) break;
                for (int j = i + 1; j < N; j++)
                {
                    double oldCost, newCost;

                    if (i == 0)
                    {
                        oldCost = fromStart[perm[0]];
                        newCost = fromStart[perm[j]];
                    }
                    else
                    {
                        oldCost = cost[perm[i - 1], perm[i]];
                        newCost = cost[perm[i - 1], perm[j]];
                    }

                    if (j + 1 < N)
                    {
                        oldCost += cost[perm[j], perm[j + 1]];
                        newCost += cost[perm[i], perm[j + 1]];
                    }

                    if (newCost < oldCost - 1e-12)
                    {
                        Array.Reverse(perm, i, j - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static int[] IteratedLocalSearch(double[] fromStart, double[,] cost, int N, Stopwatch sw, int timeLimit)
    {
        int[] bestPerm = null;
        double bestCost = double.MaxValue;

        int maxStarts = N <= 30 ? N : 30;
        for (int s = 0; s < maxStarts && sw.ElapsedMilliseconds < timeLimit / 3; s++)
        {
            int[] perm = NearestNeighbor(fromStart, cost, N, s);
            TwoOptTimed(fromStart, cost, perm, N, sw, timeLimit);
            double pc = PathCost(fromStart, cost, perm);
            if (pc < bestCost)
            {
                bestCost = pc;
                bestPerm = (int[])perm.Clone();
            }
        }

        if (bestPerm == null)
        {
            bestPerm = new int[N];
            for (int i = 0; i < N; i++) bestPerm[i] = i;
        }

        var rng = new Random(42);
        while (sw.ElapsedMilliseconds < timeLimit)
        {
            int[] perm = (int[])bestPerm.Clone();

            if (N >= 8)
            {
                int c0, c1, c2;
                do
                {
                    c0 = rng.Next(1, N);
                    c1 = rng.Next(1, N);
                    c2 = rng.Next(1, N);
                } while (c0 == c1 || c1 == c2 || c0 == c2);

                if (c0 > c1) { int t = c0; c0 = c1; c1 = t; }
                if (c1 > c2) { int t = c1; c1 = c2; c2 = t; }
                if (c0 > c1) { int t = c0; c0 = c1; c1 = t; }

                int[] np = new int[N];
                int pos = 0;
                for (int i = 0; i < c0; i++) np[pos++] = perm[i];
                for (int i = c1; i < c2; i++) np[pos++] = perm[i];
                for (int i = c0; i < c1; i++) np[pos++] = perm[i];
                for (int i = c2; i < N; i++) np[pos++] = perm[i];
                perm = np;
            }
            else
            {
                int a = rng.Next(N), b = rng.Next(N);
                int tmp = perm[a]; perm[a] = perm[b]; perm[b] = tmp;
            }

            TwoOptTimed(fromStart, cost, perm, N, sw, timeLimit);
            double pc = PathCost(fromStart, cost, perm);
            if (pc < bestCost)
            {
                bestCost = pc;
                bestPerm = (int[])perm.Clone();
            }
        }

        return bestPerm;
    }
}
