#define NOMINMAX
#define _CRT_SECURE_NO_WARNINGS
#include <windows.h>
#include <iostream>
#include <cmath>
#include <thread>
#include <mutex>
#include <atomic>
#include <string>
#include <algorithm>
#include <chrono>
#include <cstdint>
#include <cstdio>
#include <limits>

constexpr double PI = 3.14159265358979323846;

struct LanderState {
    double x, y, z, vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
    bool valid = false;
};

std::mutex g_mtx;
LanderState g_state;
std::atomic<bool> g_running{true};
std::atomic<bool> g_ready{false};

double g_width, g_depth, g_height;
double g_gravity, g_max_thrust, g_fuel_init;
double g_target_x, g_target_y, g_target_z;

const uint16_t* g_terrain = nullptr;
int g_tw = 0, g_td = 0;

inline uint16_t terrain_val(int x, int y, int idx) {
    if (!g_terrain || x < 0 || x >= g_tw || y < 0 || y >= g_td) return 0;
    return g_terrain[((size_t)x * g_td + y) * 3 + idx];
}

double ground_height(double px, double py) {
    int x = (int)std::floor(px);
    int y = (int)std::floor(py);
    return (double)terrain_val(x, y, 0);
}

bool is_solid(double px, double py, double pz) {
    int x = (int)std::floor(px);
    int y = (int)std::floor(py);
    int z = (int)std::floor(pz);
    if (x < 0 || x >= g_tw || y < 0 || y >= g_td || z < 0) return true;
    if (!g_terrain) return false;
    size_t base = ((size_t)x * g_td + y) * 3;
    uint16_t a = g_terrain[base];
    uint16_t b = g_terrain[base + 1];
    uint16_t c = g_terrain[base + 2];
    return (z < (int)a) || (b != c && z >= (int)b && z < (int)c);
}

double normalize_angle(double a) {
    while (a > PI) a -= 2 * PI;
    while (a < -PI) a += 2 * PI;
    return a;
}

void reader_func() {
    double sx, sy, sz;
    std::cin >> g_width >> g_depth >> g_height >> g_gravity >> g_max_thrust >> g_fuel_init;
    std::cin >> sx >> sy >> sz >> g_target_x >> g_target_y >> g_target_z;
    g_tw = (int)g_width;
    g_td = (int)g_depth;

    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');
    std::string terrain_path;
    std::getline(std::cin, terrain_path);
    while (!terrain_path.empty() && (terrain_path.back() == '\r' || terrain_path.back() == '\n'))
        terrain_path.pop_back();

    std::string marker;
    std::getline(std::cin, marker);

    HANDLE hFile = CreateFileA(terrain_path.c_str(), GENERIC_READ, FILE_SHARE_READ,
                               NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
    if (hFile != INVALID_HANDLE_VALUE) {
        HANDLE hMap = CreateFileMappingA(hFile, NULL, PAGE_READONLY, 0, 0, NULL);
        if (hMap) {
            void* ptr = MapViewOfFile(hMap, FILE_MAP_READ, 0, 0, 0);
            if (ptr) g_terrain = reinterpret_cast<const uint16_t*>(ptr);
        }
    }

    g_ready = true;

    while (g_running.load(std::memory_order_relaxed)) {
        LanderState s;
        if (std::cin >> s.x >> s.y >> s.z >> s.vx >> s.vy >> s.vz
                     >> s.pitch >> s.yaw >> s.roll
                     >> s.pitch_rate >> s.yaw_rate >> s.roll_rate >> s.fuel) {
            s.valid = true;
            std::lock_guard<std::mutex> lk(g_mtx);
            g_state = s;
        } else {
            break;
        }
    }
    g_running = false;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::thread reader(reader_func);

    while (!g_ready.load() && g_running.load())
        std::this_thread::sleep_for(std::chrono::milliseconds(1));

    while (g_running.load(std::memory_order_relaxed)) {
        LanderState s;
        {
            std::lock_guard<std::mutex> lk(g_mtx);
            s = g_state;
        }

        if (!s.valid) {
            std::cout << "0 0 0 0\n";
            std::cout.flush();
            std::this_thread::sleep_for(std::chrono::milliseconds(50));
            continue;
        }

        double ex = g_target_x - s.x;
        double ey = g_target_y - s.y;
        double ez = g_target_z - s.z;
        double dist_h = std::sqrt(ex * ex + ey * ey);
        double abs_ez = std::abs(ez);
        double dist_3d = std::sqrt(ex * ex + ey * ey + ez * ez);

        double max_v_decel = std::max(g_max_thrust - g_gravity, 0.5);
        double max_h_decel = std::sqrt(std::max(g_max_thrust * g_max_thrust - g_gravity * g_gravity, 0.25));

        double safe_h = std::sqrt(2.0 * max_h_decel * 0.5 * dist_h);
        double prop_h = dist_h * 0.4;
        double desired_h_speed = std::min(safe_h, prop_h);
        if (dist_h < 3.0) desired_h_speed = dist_h * 0.3;
        if (dist_h < 0.5) desired_h_speed = dist_h * 0.2;

        double dvx = 0, dvy = 0;
        if (dist_h > 0.05) {
            dvx = (ex / dist_h) * desired_h_speed;
            dvy = (ey / dist_h) * desired_h_speed;
        }

        double safe_v = std::sqrt(2.0 * max_v_decel * 0.5 * abs_ez);
        double dvz;

        if (dist_h > std::max(30.0, abs_ez * 0.5)) {
            dvz = std::clamp(ez * 0.2, -safe_v * 0.5, safe_v * 0.5);
        } else if (abs_ez > 30.0) {
            double sign = (ez > 0) ? 1.0 : -1.0;
            dvz = sign * std::min(safe_v * 0.7, abs_ez * 0.3);
        } else if (abs_ez > 5.0) {
            dvz = std::clamp(ez * 0.35, -4.0, 4.0);
        } else if (abs_ez > 1.0) {
            dvz = std::clamp(ez * 0.4, -2.0, 2.0);
        } else {
            dvz = std::clamp(ez * 0.5, -1.0, 1.0);
        }

        bool near_target = (dist_3d < 10.0);

        if (g_terrain && !near_target) {
            double gnd = ground_height(s.x, s.y);
            double clearance = s.z - gnd;
            if (clearance < 15.0 && clearance > 0 && dvz < 0) {
                double min_dvz = -std::max(0.0, (clearance - 3.0) * 0.2);
                dvz = std::max(dvz, min_dvz);
            }

            for (double t = 1.0; t <= 5.0; t += 1.0) {
                double px = s.x + s.vx * t;
                double py = s.y + s.vy * t;
                double pz = s.z + s.vz * t - 0.5 * g_gravity * t * t;
                if (is_solid(px, py, pz) || is_solid(px, py, pz + 3)) {
                    double climb_rate = std::min(10.0, 5.0 / t);
                    dvz = std::max(dvz, climb_rate);
                    dvx *= 0.5;
                    dvy *= 0.5;
                    break;
                }
            }
        }

        double kp = 1.5;
        double ax = kp * (dvx - s.vx);
        double ay = kp * (dvy - s.vy);
        double az = kp * (dvz - s.vz) + g_gravity;
        az = std::max(az, 0.0);

        double ah = std::sqrt(ax * ax + ay * ay);
        double a_mag = std::sqrt(az * az + ah * ah);

        double target_pitch = 0.0;
        double target_yaw = s.yaw;
        double thrust_frac = g_gravity / g_max_thrust;

        if (a_mag > 0.01) {
            target_pitch = std::atan2(ah, az);

            double max_pitch = PI / 3.0;
            if (dist_3d < 50.0) max_pitch = PI / 4.0;
            if (dist_3d < 15.0) max_pitch = PI / 6.0;
            if (dist_3d < 5.0) max_pitch = PI / 10.0;
            target_pitch = std::clamp(target_pitch, 0.0, max_pitch);

            if (ah > 0.01) {
                target_yaw = std::atan2(-ay, -ax);
            }

            thrust_frac = std::clamp(a_mag / g_max_thrust, 0.0, 1.0);
        }

        double yaw_err = normalize_angle(target_yaw - s.yaw);
        double pitch_err = target_pitch - s.pitch;

        if (std::abs(yaw_err) > PI / 3.0) {
            double hover = g_gravity / g_max_thrust;
            thrust_frac = std::min(thrust_frac, hover * 1.3);
        }

        double kp_a = 4.0, kd_a = 2.0;
        double pitch_cmd = std::clamp(kp_a * pitch_err - kd_a * s.pitch_rate, -1.0, 1.0);
        double yaw_cmd = std::clamp(kp_a * yaw_err - kd_a * s.yaw_rate, -1.0, 1.0);
        double roll_cmd = std::clamp(-kp_a * s.roll - kd_a * s.roll_rate, -1.0, 1.0);

        std::cout << thrust_frac << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << "\n";
        std::cout.flush();

        std::this_thread::sleep_for(std::chrono::milliseconds(50));
    }

    if (reader.joinable()) reader.join();
    return 0;
}
