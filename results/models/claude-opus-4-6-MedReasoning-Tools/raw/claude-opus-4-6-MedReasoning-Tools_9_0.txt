using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static byte[] buf = new byte[1 << 16];
    static int bufPos, bufLen;
    static Stream stdin;

    static int ReadByte()
    {
        if (bufPos >= bufLen)
        {
            bufLen = stdin.Read(buf, 0, buf.Length);
            bufPos = 0;
            if (bufLen == 0) return -1;
        }
        return buf[bufPos++];
    }

    static int ReadInt()
    {
        int c;
        while ((c = ReadByte()) >= 0 && (c < '0' || c > '9')) ;
        if (c < 0) return -1;
        int v = c - '0';
        while ((c = ReadByte()) >= '0' && c <= '9')
            v = v * 10 + (c - '0');
        return v;
    }

    static int[] seg;
    static int SN;

    static void SegUp(int p, int d)
    {
        p += SN;
        seg[p] += d;
        for (p >>= 1; p >= 1; p >>= 1)
            seg[p] = seg[2 * p] + seg[2 * p + 1];
    }

    static int SegCeil(int lo)
    {
        return SegQ(1, 0, SN - 1, lo, SN - 1);
    }

    static int SegQ(int nd, int l, int r, int ql, int qr)
    {
        if (r < ql || l > qr || seg[nd] == 0) return -1;
        if (l == r) return l;
        int m = (l + r) >> 1;
        int res = SegQ(2 * nd, l, m, ql, qr);
        return res >= 0 ? res : SegQ(2 * nd + 1, m + 1, r, ql, qr);
    }

    static void Main()
    {
        stdin = Console.OpenStandardInput();
        int n = ReadInt();
        int stockLen = ReadInt();

        if (n == 0)
        {
            Console.Write("0\n");
            return;
        }

        var cuts = new int[n];
        for (int i = 0; i < n; i++)
            cuts[i] = ReadInt();

        var order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => cuts[b].CompareTo(cuts[a]));

        int[] binOf = new int[n];
        int numBins = 0;
        var capBins = new Dictionary<int, List<int>>();

        if (stockLen <= 16_000_000)
        {
            SN = 1;
            while (SN <= stockLen) SN <<= 1;
            seg = new int[2 * SN];

            for (int i = 0; i < n; i++)
            {
                int ci = order[i];
                int sz = cuts[ci];
                int cap = SegCeil(sz);

                if (cap < 0)
                {
                    int bi = numBins++;
                    binOf[ci] = bi;
                    int nc = stockLen - sz;
                    if (nc > 0)
                    {
                        SegUp(nc, 1);
                        if (!capBins.TryGetValue(nc, out var lst))
                            capBins[nc] = lst = new List<int>();
                        lst.Add(bi);
                    }
                }
                else
                {
                    var lst = capBins[cap];
                    int bi = lst[lst.Count - 1];
                    lst.RemoveAt(lst.Count - 1);
                    SegUp(cap, -1);
                    if (lst.Count == 0) capBins.Remove(cap);

                    binOf[ci] = bi;
                    int nc = cap - sz;
                    if (nc > 0)
                    {
                        SegUp(nc, 1);
                        if (!capBins.TryGetValue(nc, out var nlst))
                            capBins[nc] = nlst = new List<int>();
                        nlst.Add(bi);
                    }
                }
            }
        }
        else
        {
            var uniqueCaps = new SortedSet<int>();

            for (int i = 0; i < n; i++)
            {
                int ci = order[i];
                int sz = cuts[ci];
                int cap = -1;
                foreach (int c in uniqueCaps.GetViewBetween(sz, stockLen))
                {
                    cap = c;
                    break;
                }

                if (cap < 0)
                {
                    int bi = numBins++;
                    binOf[ci] = bi;
                    int nc = stockLen - sz;
                    if (nc > 0)
                    {
                        uniqueCaps.Add(nc);
                        if (!capBins.TryGetValue(nc, out var lst))
                            capBins[nc] = lst = new List<int>();
                        lst.Add(bi);
                    }
                }
                else
                {
                    var lst = capBins[cap];
                    int bi = lst[lst.Count - 1];
                    lst.RemoveAt(lst.Count - 1);
                    if (lst.Count == 0)
                    {
                        capBins.Remove(cap);
                        uniqueCaps.Remove(cap);
                    }

                    binOf[ci] = bi;
                    int nc = cap - sz;
                    if (nc > 0)
                    {
                        uniqueCaps.Add(nc);
                        if (!capBins.TryGetValue(nc, out var nlst))
                            capBins[nc] = nlst = new List<int>();
                        nlst.Add(bi);
                    }
                }
            }
        }

        seg = null;
        var stdout = Console.OpenStandardOutput();
        var writer = new StreamWriter(stdout, Encoding.ASCII, 1 << 16);

        var binHead = new int[numBins];
        var next = new int[n];
        Array.Fill(binHead, -1);
        for (int i = n - 1; i >= 0; i--)
        {
            int bi = binOf[i];
            next[i] = binHead[bi];
            binHead[bi] = i;
        }

        writer.Write(numBins);
        writer.Write('\n');
        for (int i = 0; i < numBins; i++)
        {
            bool first = true;
            int j = binHead[i];
            while (j >= 0)
            {
                if (!first) writer.Write(' ');
                writer.Write(j);
                first = false;
                j = next[j];
            }
            writer.Write('\n');
        }
        writer.Flush();
    }
}
