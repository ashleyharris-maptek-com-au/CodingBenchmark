using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int ContW, ContH, ContD;

    struct Box
    {
        public int Index, W, H, D;
        public long Volume => (long)W * H * D;
    }

    struct Placement
    {
        public int Index, X, Y, Z, W, H, D;
    }

    struct Point : IComparable<Point>
    {
        public int X, Y, Z;
        public int CompareTo(Point o)
        {
            int c = Y.CompareTo(o.Y);
            if (c != 0) return c;
            c = X.CompareTo(o.X);
            if (c != 0) return c;
            return Z.CompareTo(o.Z);
        }
    }

    static int cellSize;
    static int gridNX, gridNY, gridNZ;
    static List<int>[] grid;

    static void InitGrid()
    {
        long vol = (long)ContW * ContH * ContD;
        cellSize = Math.Max(1, (int)Math.Cbrt(vol / 5000.0));
        if (cellSize < 1) cellSize = 1;
        gridNX = (ContW + cellSize - 1) / cellSize;
        gridNY = (ContH + cellSize - 1) / cellSize;
        gridNZ = (ContD + cellSize - 1) / cellSize;
        long total = (long)gridNX * gridNY * gridNZ;
        if (total > 8_000_000)
        {
            cellSize = (int)Math.Cbrt((double)vol / 8_000_000) + 1;
            gridNX = (ContW + cellSize - 1) / cellSize;
            gridNY = (ContH + cellSize - 1) / cellSize;
            gridNZ = (ContD + cellSize - 1) / cellSize;
            total = (long)gridNX * gridNY * gridNZ;
        }
        grid = new List<int>[total];
    }

    static List<Placement> placements = new List<Placement>();

    static void AddToGrid(int idx)
    {
        var p = placements[idx];
        int x0 = p.X / cellSize, x1 = (p.X + p.W - 1) / cellSize;
        int y0 = p.Y / cellSize, y1 = (p.Y + p.H - 1) / cellSize;
        int z0 = p.Z / cellSize, z1 = (p.Z + p.D - 1) / cellSize;
        for (int gx = x0; gx <= x1; gx++)
            for (int gy = y0; gy <= y1; gy++)
                for (int gz = z0; gz <= z1; gz++)
                {
                    long gi = (long)gx * gridNY * gridNZ + (long)gy * gridNZ + gz;
                    if (grid[gi] == null) grid[gi] = new List<int>();
                    grid[gi].Add(idx);
                }
    }

    static bool Overlaps(int x, int y, int z, int w, int h, int d)
    {
        int x0 = x / cellSize, x1 = (x + w - 1) / cellSize;
        int y0 = y / cellSize, y1 = (y + h - 1) / cellSize;
        int z0 = z / cellSize, z1 = (z + d - 1) / cellSize;
        for (int gx = x0; gx <= x1; gx++)
            for (int gy = y0; gy <= y1; gy++)
                for (int gz = z0; gz <= z1; gz++)
                {
                    long gi = (long)gx * gridNY * gridNZ + (long)gy * gridNZ + gz;
                    var cell = grid[gi];
                    if (cell == null) continue;
                    for (int i = 0; i < cell.Count; i++)
                    {
                        var p = placements[cell[i]];
                        if (x < p.X + p.W && x + w > p.X &&
                            y < p.Y + p.H && y + h > p.Y &&
                            z < p.Z + p.D && z + d > p.Z)
                            return true;
                    }
                }
        return false;
    }

    static void Main()
    {
        var firstLine = Console.ReadLine().Trim().Split();
        int N = int.Parse(firstLine[0]);
        ContW = int.Parse(firstLine[1]);
        ContH = int.Parse(firstLine[2]);
        ContD = int.Parse(firstLine[3]);

        var boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split();
            int bw = int.Parse(parts[0]);
            int bh = int.Parse(parts[1]);
            int bd = int.Parse(parts[2]);
            boxes[i] = new Box { Index = i, W = bw, H = bh, D = bd };
        }

        InitGrid();

        var sortedIndices = Enumerable.Range(0, N)
            .OrderByDescending(i => boxes[i].Volume)
            .ThenByDescending(i => Math.Max(boxes[i].W, Math.Max(boxes[i].H, boxes[i].D)))
            .ToArray();

        var eps = new SortedSet<Point>();
        eps.Add(new Point { X = 0, Y = 0, Z = 0 });

        bool[] packed = new bool[N];
        int packedCount = 0;

        int maxEPs = N < 100 ? 2000 : (N < 1000 ? 800 : 400);

        foreach (int si in sortedIndices)
        {
            var box = boxes[si];
            if (box.W > ContW && box.W > ContH && box.W > ContD &&
                box.H > ContW && box.H > ContH && box.H > ContD &&
                box.D > ContW && box.D > ContH && box.D > ContD)
                continue;

            int[] dims = { box.W, box.H, box.D };
            Array.Sort(dims);
            var orientations = new HashSet<(int, int, int)>();
            orientations.Add((dims[0], dims[1], dims[2]));
            orientations.Add((dims[0], dims[2], dims[1]));
            orientations.Add((dims[1], dims[0], dims[2]));
            orientations.Add((dims[1], dims[2], dims[0]));
            orientations.Add((dims[2], dims[0], dims[1]));
            orientations.Add((dims[2], dims[1], dims[0]));

            var validOrients = new List<(int w, int h, int d)>();
            foreach (var (ow, oh, od) in orientations)
            {
                if (ow <= ContW && oh <= ContH && od <= ContD)
                    validOrients.Add((ow, oh, od));
            }
            if (validOrients.Count == 0) continue;

            validOrients.Sort((a, b) => a.h.CompareTo(b.h));

            bool placed = false;
            Point bestPt = default;
            int bestW = 0, bestH = 0, bestD = 0;
            long bestScore = long.MaxValue;

            foreach (var ep in eps)
            {
                if (placed && ep.Y > bestPt.Y) break;

                foreach (var (ow, oh, od) in validOrients)
                {
                    if (ep.X + ow > ContW || ep.Y + oh > ContH || ep.Z + od > ContD)
                        continue;

                    if (!Overlaps(ep.X, ep.Y, ep.Z, ow, oh, od))
                    {
                        long score = (long)ep.Y * 1000000L + (long)ep.X * 1000L + ep.Z;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestPt = ep;
                            bestW = ow; bestH = oh; bestD = od;
                            placed = true;
                        }
                        break;
                    }
                }
            }

            if (placed)
            {
                var pl = new Placement
                {
                    Index = si,
                    X = bestPt.X, Y = bestPt.Y, Z = bestPt.Z,
                    W = bestW, H = bestH, D = bestD
                };
                int pidx = placements.Count;
                placements.Add(pl);
                AddToGrid(pidx);
                packed[si] = true;
                packedCount++;

                eps.Remove(bestPt);

                var newPts = new Point[]
                {
                    new Point { X = bestPt.X + bestW, Y = bestPt.Y, Z = bestPt.Z },
                    new Point { X = bestPt.X, Y = bestPt.Y + bestH, Z = bestPt.Z },
                    new Point { X = bestPt.X, Y = bestPt.Y, Z = bestPt.Z + bestD }
                };

                foreach (var np in newPts)
                {
                    if (np.X < ContW && np.Y < ContH && np.Z < ContD)
                        eps.Add(np);
                }

                if (eps.Count > maxEPs)
                {
                    var keep = eps.Take(maxEPs).ToList();
                    eps.Clear();
                    foreach (var p in keep) eps.Add(p);
                }
            }
        }

        var output = new System.Text.StringBuilder();
        output.AppendLine(packedCount.ToString());
        foreach (var p in placements)
        {
            output.Append(p.Index); output.Append(' ');
            output.Append(p.X); output.Append(' ');
            output.Append(p.Y); output.Append(' ');
            output.AppendLine(p.Z.ToString());
        }
        Console.Write(output);
    }
}
