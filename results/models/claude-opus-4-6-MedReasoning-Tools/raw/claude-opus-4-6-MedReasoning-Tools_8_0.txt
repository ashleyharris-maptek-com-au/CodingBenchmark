use std::io::{self, Read, Write, BufWriter};
use std::collections::VecDeque;

fn write_usize(out: &mut impl Write, n: usize) {
    if n == 0 {
        out.write_all(b"0").unwrap();
        return;
    }
    let mut buf = [0u8; 20];
    let mut i = 20;
    let mut v = n;
    while v > 0 {
        i -= 1;
        buf[i] = b'0' + (v % 10) as u8;
        v /= 10;
    }
    out.write_all(&buf[i..]).unwrap();
}

fn main() {
    let mut input = Vec::new();
    io::stdin().lock().read_to_end(&mut input).unwrap();

    let stdout = io::stdout();
    let mut out = BufWriter::with_capacity(1 << 20, stdout.lock());

    let mut pos = 0usize;

    fn skip_ws(data: &[u8], pos: &mut usize) {
        while *pos < data.len()
            && (data[*pos] == b' ' || data[*pos] == b'\t'
                || data[*pos] == b'\r' || data[*pos] == b'\n')
        {
            *pos += 1;
        }
    }

    fn parse_num(data: &[u8], pos: &mut usize) -> usize {
        skip_ws(data, pos);
        let mut n = 0usize;
        while *pos < data.len() && data[*pos].is_ascii_digit() {
            n = n * 10 + (data[*pos] - b'0') as usize;
            *pos += 1;
        }
        n
    }

    let h = parse_num(&input, &mut pos);
    let w = parse_num(&input, &mut pos);

    while pos < input.len() && input[pos] != b'\n' {
        pos += 1;
    }
    if pos < input.len() {
        pos += 1;
    }

    let mut grid = vec![1u8; h * w];
    let mut start = 0usize;
    let mut end = 0usize;

    for r in 0..h {
        let row_start = pos;
        let mut line_end = pos;
        while line_end < input.len() && input[line_end] != b'\n' && input[line_end] != b'\r' {
            line_end += 1;
        }

        let cols = (line_end - row_start).min(w);
        for c in 0..cols {
            let idx = r * w + c;
            match input[row_start + c] {
                b'#' => {}
                b'A' => {
                    grid[idx] = 0;
                    start = idx;
                }
                b'B' => {
                    grid[idx] = 0;
                    end = idx;
                }
                _ => {
                    grid[idx] = 0;
                }
            }
        }

        pos = line_end;
        if pos < input.len() && input[pos] == b'\r' {
            pos += 1;
        }
        if pos < input.len() && input[pos] == b'\n' {
            pos += 1;
        }
    }

    drop(input);

    grid[start] = 6;
    let mut queue = VecDeque::with_capacity(4096);
    queue.push_back(start);

    let mut found = false;

    'bfs: while let Some(cur) = queue.pop_front() {
        if cur == end {
            found = true;
            break;
        }

        let r = cur / w;
        let c = cur % w;

        if r > 0 {
            let n = cur - w;
            if grid[n] == 0 {
                grid[n] = 3;
                if n == end { found = true; break 'bfs; }
                queue.push_back(n);
            }
        }
        if r + 1 < h {
            let n = cur + w;
            if grid[n] == 0 {
                grid[n] = 2;
                if n == end { found = true; break 'bfs; }
                queue.push_back(n);
            }
        }
        if c > 0 {
            let n = cur - 1;
            if grid[n] == 0 {
                grid[n] = 5;
                if n == end { found = true; break 'bfs; }
                queue.push_back(n);
            }
        }
        if c + 1 < w {
            let n = cur + 1;
            if grid[n] == 0 {
                grid[n] = 4;
                if n == end { found = true; break 'bfs; }
                queue.push_back(n);
            }
        }
    }

    if !found {
        out.write_all(b"0\n").unwrap();
        return;
    }

    let mut path = Vec::new();
    let mut cur = end;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = match grid[cur] {
            2 => cur - w,
            3 => cur + w,
            4 => cur - 1,
            5 => cur + 1,
            _ => break,
        };
    }
    path.reverse();

    write_usize(&mut out, path.len());
    out.write_all(b"\n").unwrap();
    for &p in &path {
        write_usize(&mut out, p / w);
        out.write_all(b" ").unwrap();
        write_usize(&mut out, p % w);
        out.write_all(b"\n").unwrap();
    }
}
