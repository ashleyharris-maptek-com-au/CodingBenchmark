use std::io::{self, Read, Write, BufWriter};
use std::collections::BinaryHeap;

fn add_to_cover(
    v: usize,
    adj: &[Vec<u32>],
    in_cover: &mut [bool],
    active: &mut [bool],
    deg: &mut [u32],
    queue: &mut Vec<u32>,
) {
    if !active[v] {
        return;
    }
    in_cover[v] = true;
    active[v] = false;
    for &u in &adj[v] {
        let u = u as usize;
        if active[u] {
            deg[u] -= 1;
            if deg[u] <= 1 {
                queue.push(u as u32);
            }
        }
    }
}

fn process_low_degree(
    adj: &[Vec<u32>],
    in_cover: &mut [bool],
    active: &mut [bool],
    deg: &mut [u32],
    queue: &mut Vec<u32>,
) {
    while let Some(vi) = queue.pop() {
        let v = vi as usize;
        if !active[v] {
            continue;
        }
        if deg[v] == 0 {
            active[v] = false;
            continue;
        }
        if deg[v] == 1 {
            if let Some(&u) = adj[v].iter().find(|&&u| active[u as usize]) {
                add_to_cover(u as usize, adj, in_cover, active, deg, queue);
            }
            if active[v] && deg[v] == 0 {
                active[v] = false;
            }
        }
    }
}

fn main() {
    let mut raw = Vec::new();
    io::stdin().lock().read_to_end(&mut raw).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut pos = 0;
    let input = &raw;

    let next_usize = |pos: &mut usize| -> usize {
        while *pos < input.len()
            && (input[*pos] == b' ' || input[*pos] == b'\n' || input[*pos] == b'\r')
        {
            *pos += 1;
        }
        let mut val = 0usize;
        while *pos < input.len() && input[*pos] >= b'0' && input[*pos] <= b'9' {
            val = val * 10 + (input[*pos] - b'0') as usize;
            *pos += 1;
        }
        val
    };

    let n = next_usize(&mut pos);
    let m = next_usize(&mut pos);

    let mut adj: Vec<Vec<u32>> = vec![vec![]; n];
    for _ in 0..m {
        let u = next_usize(&mut pos);
        let v = next_usize(&mut pos);
        adj[u].push(v as u32);
        adj[v].push(u as u32);
    }

    let mut in_cover = vec![false; n];
    let mut active = vec![true; n];
    let mut deg: Vec<u32> = (0..n).map(|i| adj[i].len() as u32).collect();
    let mut queue: Vec<u32> = Vec::new();

    for i in 0..n {
        if deg[i] <= 1 {
            queue.push(i as u32);
        }
    }
    process_low_degree(&adj, &mut in_cover, &mut active, &mut deg, &mut queue);

    let mut heap: BinaryHeap<(u32, u32)> = BinaryHeap::new();
    for i in 0..n {
        if active[i] && deg[i] > 0 {
            heap.push((deg[i], i as u32));
        }
    }

    while let Some((d, vi)) = heap.pop() {
        let v = vi as usize;
        if !active[v] || deg[v] != d {
            continue;
        }

        in_cover[v] = true;
        active[v] = false;
        for &u in &adj[v] {
            let u = u as usize;
            if active[u] {
                deg[u] -= 1;
                if deg[u] <= 1 {
                    queue.push(u as u32);
                } else {
                    heap.push((deg[u], u as u32));
                }
            }
        }

        process_low_degree(&adj, &mut in_cover, &mut active, &mut deg, &mut queue);
    }

    let mut cover_sorted: Vec<usize> = (0..n).filter(|&i| in_cover[i]).collect();
    cover_sorted.sort_by_key(|&v| adj[v].len());
    for v in cover_sorted {
        if in_cover[v] && adj[v].iter().all(|&u| in_cover[u as usize]) {
            in_cover[v] = false;
        }
    }

    let result: Vec<usize> = (0..n).filter(|&i| in_cover[i]).collect();
    writeln!(out, "{}", result.len()).unwrap();
    let mut buf = String::with_capacity(result.len() * 4);
    for (i, &v) in result.iter().enumerate() {
        if i > 0 {
            buf.push(' ');
        }
        buf.push_str(&v.to_string());
    }
    writeln!(out, "{}", buf).unwrap();
}
