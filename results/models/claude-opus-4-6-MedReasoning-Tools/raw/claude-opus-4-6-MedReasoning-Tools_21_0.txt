```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <queue>
#include <unordered_set>
#include <algorithm>
#include <climits>

using namespace std;
using Pos = vector<int>;

struct PosHash {
    size_t operator()(const Pos& p) const {
        size_t h = 0;
        for (int x : p)
            h ^= hash<int>()(x) + 0x9e3779b9 + (h << 6) + (h >> 2);
        return h;
    }
};

static int D;
static vector<int> bounds;
static deque<Pos> snake;
static unordered_set<Pos, PosHash> snake_set, obstacles, food_set;
static int target_len;

bool in_bounds(const Pos& p) {
    for (int i = 0; i < D; i++)
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    return true;
}

bool is_safe(const Pos& p) {
    return in_bounds(p) && !obstacles.count(p) && !snake_set.count(p);
}

pair<int,int> find_move() {
    const Pos& h = snake.front();

    int search_limit;
    if (D <= 3) search_limit = 500000;
    else if (D <= 6) search_limit = 100000;
    else if (D <= 10) search_limit = 20000;
    else search_limit = 0;

    if (search_limit > 0) {
        vector<const Pos*> targets;
        {
            vector<pair<int, const Pos*>> fd;
            for (const auto& f : food_set) {
                int d = 0;
                for (int i = 0; i < D; i++) d += abs(h[i] - f[i]);
                fd.push_back({d, &f});
            }
            sort(fd.begin(), fd.end());
            for (int i = 0; i < min((int)fd.size(), 5); i++)
                targets.push_back(fd[i].second);
        }

        auto heuristic = [&](const Pos& p) -> int {
            int best = INT_MAX;
            for (auto* t : targets) {
                int d = 0;
                for (int i = 0; i < D; i++) d += abs(p[i] - (*t)[i]);
                if (d < best) best = d;
            }
            return best;
        };

        struct Node {
            int f, g, first_axis, first_dir;
            Pos pos;
            bool operator>(const Node& o) const { return f > o.f; }
        };

        priority_queue<Node, vector<Node>, greater<Node>> pq;
        unordered_set<Pos, PosHash> visited;
        visited.insert(h);

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                Pos next = h;
                next[axis] += dir;
                if (is_safe(next) && !visited.count(next)) {
                    visited.insert(next);
                    if (food_set.count(next)) return {axis, dir};
                    pq.push({1 + heuristic(next), 1, axis, dir, next});
                }
            }
        }

        int explored = 0;
        while (!pq.empty() && explored < search_limit) {
            Node node = pq.top();
            pq.pop();
            explored++;

            for (int axis = 0; axis < D; axis++) {
                for (int dir : {-1, 1}) {
                    Pos next = node.pos;
                    next[axis] += dir;
                    if (is_safe(next) && !visited.count(next)) {
                        visited.insert(next);
                        if (food_set.count(next))
                            return {node.first_axis, node.first_dir};
                        int ng = node.g + 1;
                        pq.push({ng + heuristic(next), ng, node.first_axis, node.first_dir, next});
                    }
                }
            }
        }
    }

    int best_score = INT_MIN;
    int result_axis = -1, result_dir = 0;

    for (int axis = 0; axis < D; axis++) {
        for (int dir : {-1, 1}) {
            Pos next = h;
            next[axis] += dir;
            if (!is_safe(next)) continue;

            if (food_set.count(next)) return {axis, dir};

            int dist = INT_MAX;
            for (const auto& f : food_set) {
                int d = 0;
                for (int i = 0; i < D; i++) d += abs(next[i] - f[i]);
                dist = min(dist, d);
            }

            int neighbors = 0;
            Pos nn = next;
            for (int a = 0; a < D; a++) {
                nn[a]++;
                if (is_safe(nn)) neighbors++;
                nn[a] -= 2;
                if (is_safe(nn)) neighbors++;
                nn[a]++;
            }

            int score = -dist * 100 + neighbors;
            if (score > best_score) {
                best_score = score;
                result_axis = axis;
                result_dir = dir;
            }
        }
    }

    if (result_axis == -1) return {0, 1};
    return {result_axis, result_dir};
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int max_turns;
    cin >> D >> max_turns;

    bounds.resize(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    int snake_len;
    cin >> snake_len;
    Pos head(D);
    for (int i = 0; i < D; i++) cin >> head[i];

    snake.push_back(head);
    snake_set.insert(head);
    target_len = snake_len;

    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        Pos f(D);
        for (int j = 0; j < D; j++) cin >> f[j];
        food_set.insert(f);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        Pos ob(D);
        for (int j = 0; j < D; j++) cin >> ob[j];
        obstacles.insert(ob);
    }

    for (int turn = 0; turn < max_turns && !food_set.empty(); turn++) {
        auto [axis, dir] = find_move();

        cout << axis << " " << dir << "\n";
        cout.flush();

        Pos new_head = snake.front();
        new_head[axis] += dir;

        snake.push_front(new_head);
        snake_set.insert(new_head);

        if (food_set.count(new_head)) {
            food_set.erase(new_head);
            target_len++;
        }

        while ((int)snake.size() > target_len) {
            snake_set.erase(snake.back());
            snake.pop_back();
        }
    }

    return 0;
}
```
