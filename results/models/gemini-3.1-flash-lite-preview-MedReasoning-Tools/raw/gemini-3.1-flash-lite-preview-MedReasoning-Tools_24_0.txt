#include <iostream>
#include <string>
#include <vector>
#include <thread>
#include <mutex>
#include <atomic>
#include <fstream>
#include <cmath>
#include <iomanip>

struct State {
    double x, y, z;
    double vx, vy, vz;
    double pitch, yaw, roll;
    double pitch_rate, yaw_rate, roll_rate;
    double fuel;
};

struct TerrainCell {
    uint16_t ground_top;
    uint16_t floating_bottom;
    uint16_t floating_top;
};

class LanderController {
    std::atomic<bool> running{true};
    std::mutex state_mutex;
    State current_state;
    
    double width, depth, height, gravity, max_thrust, fuel;
    double target_x, target_y, target_z;
    std::string terrain_file_path;

public:
    void run() {
        if (!read_header()) return;
        
        std::thread terrain_thread(&LanderController::load_terrain, this);
        terrain_thread.detach();

        std::thread io_thread(&LanderController::input_loop, this);
        io_thread.detach();

        control_loop();
    }

private:
    bool read_header() {
        std::cin >> width >> depth >> height >> gravity >> max_thrust >> fuel;
        std::cin >> target_x >> target_y >> target_z;
        std::cin.ignore(); 
        std::getline(std::cin, terrain_file_path);
        return true;
    }

    void load_terrain() {
        std::ifstream file(terrain_file_path, std::ios::binary);
        if (!file.is_open()) return;

        size_t num_cells = static_cast<size_t>(width) * static_cast<size_t>(depth);
        std::vector<TerrainCell> terrain(num_cells);

        for (size_t i = 0; i < num_cells; ++i) {
            uint16_t buffer[3];
            file.read(reinterpret_cast<char*>(buffer), sizeof(buffer));
            terrain[i] = {buffer[0], buffer[1], buffer[2]};
        }
        // Could store terrain for collision detection if needed
    }

    void input_loop() {
        while (running) {
            State new_state;
            if (std::cin >> new_state.x >> new_state.y >> new_state.z >> 
                new_state.vx >> new_state.vy >> new_state.vz >> 
                new_state.pitch >> new_state.yaw >> new_state.roll >> 
                new_state.pitch_rate >> new_state.yaw_rate >> new_state.roll_rate >> 
                new_state.fuel) {
                
                std::lock_guard<std::mutex> lock(state_mutex);
                current_state = new_state;
            } else {
                running = false;
            }
        }
    }

    void control_loop() {
        while (running) {
            State s;
            {
                std::lock_guard<std::mutex> lock(state_mutex);
                s = current_state;
            }

            // Simple Proportional Controller to land at target
            double dx = target_x - s.x;
            double dy = target_y - s.y;
            double dz = target_z - s.z;

            // Simple vertical control
            double vertical_speed_target = -0.5; // slow landing
            double thrust = (s.vz < vertical_speed_target) ? 1.0 : (s.vz < 0.5 ? 0.3 : 0.0);
            
            // Simple horizontal control (pointing towards target)
            double target_pitch = (dx > 0) ? -0.1 : 0.1;
            double target_yaw = std::atan2(dy, dx);
            
            double pitch_cmd = (target_pitch - s.pitch) * 2.0 - s.pitch_rate;
            double yaw_cmd = (target_yaw - s.yaw) * 0.5 - s.yaw_rate;
            double roll_cmd = -s.roll * 0.1 - s.roll_rate * 0.1;

            // Clamp commands
            thrust = std::max(0.0, std::min(1.0, thrust));
            pitch_cmd = std::max(-1.0, std::min(1.0, pitch_cmd));
            yaw_cmd = std::max(-1.0, std::min(1.0, yaw_cmd));
            roll_cmd = std::max(-1.0, std::min(1.0, roll_cmd));

            std::cout << thrust << " " << pitch_cmd << " " << yaw_cmd << " " << roll_cmd << std::endl;
            
            std::this_thread::sleep_for(std::chrono::milliseconds(100));
        }
    }
};

int main() {
    LanderController controller;
    controller.run();
    return 0;
}
