using System;
using System.Collections.Generic;
using System.Globalization;

namespace AsteroidInterceptor
{
    public record Vector3(double X, double Y, double Z);

    public class Body
    {
        public string Name { get; set; }
        public double Mass { get; set; }
        public Vector3 Position { get; set; }
        public Vector3 Velocity { get; set; }
    }

    public class InputParser
    {
        public static void Parse(string[] lines, out int numBodies, out double warningDays, out double impactorMass, out double deltaVBudget, out double asteroidMass, out long t0, out List<Body> bodies, out string de421Path)
        {
            var firstLine = lines[0].Split(' ', StringSplitOptions.RemoveEmptyEntries);
            numBodies = int.Parse(firstLine[0]);
            warningDays = double.Parse(firstLine[1], CultureInfo.InvariantCulture);
            impactorMass = double.Parse(firstLine[2], CultureInfo.InvariantCulture);
            deltaVBudget = double.Parse(firstLine[3], CultureInfo.InvariantCulture);
            asteroidMass = double.Parse(firstLine[4], CultureInfo.InvariantCulture);
            t0 = long.Parse(firstLine[5]);

            bodies = new List<Body>();
            for (int i = 1; i <= numBodies; i++)
            {
                var bodyLine = lines[i].Split(' ', StringSplitOptions.RemoveEmptyEntries);
                bodies.Add(new Body
                {
                    Name = bodyLine[0],
                    Mass = double.Parse(bodyLine[1], CultureInfo.InvariantCulture),
                    Position = new Vector3(double.Parse(bodyLine[2], CultureInfo.InvariantCulture), double.Parse(bodyLine[3], CultureInfo.InvariantCulture), double.Parse(bodyLine[4], CultureInfo.InvariantCulture)),
                    Velocity = new Vector3(double.Parse(bodyLine[5], CultureInfo.InvariantCulture), double.Parse(bodyLine[6], CultureInfo.InvariantCulture), double.Parse(bodyLine[7], CultureInfo.InvariantCulture))
                });
            }
            de421Path = lines[numBodies + 1].Trim();
        }
    }
}
