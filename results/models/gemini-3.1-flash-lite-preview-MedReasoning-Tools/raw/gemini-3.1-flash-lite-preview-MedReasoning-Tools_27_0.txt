#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>

/**
 * Graph coloring using a greedy approach with Largest Degree First (LDF) heuristic.
 * This is generally efficient for large graphs.
 */

struct Edge {
    int u, v;
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_colors;
    if (!(std::cin >> num_vertices >> num_edges >> num_colors)) return 0;

    std::vector<std::vector<int>> adj(num_vertices);
    std::vector<int> degree(num_vertices, 0);

    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u >= 0 && u < num_vertices && v >= 0 && v < num_vertices) {
            adj[u].push_back(v);
            adj[v].push_back(u);
            degree[u]++;
            degree[v]++;
        }
    }

    // Heuristic: Process vertices with highest degree first
    std::vector<int> nodes(num_vertices);
    std::iota(nodes.begin(), nodes.end(), 0);
    std::sort(nodes.begin(), nodes.end(), [&](int a, int b) {
        return degree[a] > degree[b];
    });

    std::vector<int> result(num_vertices, -1);

    for (int u : nodes) {
        std::vector<bool> available(num_colors, true);
        for (int v : adj[u]) {
            if (result[v] != -1) {
                available[result[v]] = false;
            }
        }

        for (int c = 0; c < num_colors; ++c) {
            if (available[c]) {
                result[u] = c;
                break;
            }
        }
    }

    for (int i = 0; i < num_vertices; ++i) {
        std::cout << (result[i] == -1 ? 0 : result[i]) << (i == num_vertices - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
