using System;
using System.IO;
using System.Collections.Generic;

namespace MiningSolver
{
    // Basic structure for an operation
    struct Operation
    {
        public int DigX, DigY, DigZ;
        public List<(int, int)> Path;
        public int DestX, DestY;
    }

    class Program
    {
        static void Main(string[] args)
        {
            // Read input header
            string header = Console.ReadLine();
            if (string.IsNullOrEmpty(header)) return;
            string[] parts = header.Split(' ');
            int xv = int.Parse(parts[0]);
            int yv = int.Parse(parts[1]);
            int zv = int.Parse(parts[2]);
            double xb = double.Parse(parts[3]);
            double yb = double.Parse(parts[4]);
            double zb = double.Parse(parts[5]);

            List<Operation> operations = new List<Operation>();

            // Voxel reading and processing
            using (Stream stdin = Console.OpenStandardInput())
            {
                byte[] buffer = new byte[65536];
                int bytesRead;
                int currentX = 0, currentY = 0, currentZ = 0;

                while ((bytesRead = stdin.Read(buffer, 0, buffer.Length)) > 0)
                {
                    for (int i = 0; i < bytesRead; i++)
                    {
                        byte d = buffer[i];
                        int mass = d % 16;
                        int value = d / 16;

                        // Basic greedy: If ore is valuable, add it to operations
                        if (value > 0)
                        {
                            operations.Add(new Operation 
                            { 
                                DigX = currentX, DigY = currentY, DigZ = currentZ,
                                Path = new List<(int, int)>(), 
                                DestX = 0, DestY = 0 
                            });
                        }

                        currentZ++;
                        if (currentZ >= zv) { currentZ = 0; currentY++; }
                        if (currentY >= yv) { currentY = 0; currentX++; }
                        if (currentX >= xv) break;
                    }
                }
            }

            // Output format
            Console.WriteLine(operations.Count);
            foreach (var op in operations)
            {
                Console.WriteLine($"1 {op.DigX} {op.DigY} {op.DigZ} {op.Path.Count} {op.DestX} {op.DestY}");
            }
        }
    }
}
