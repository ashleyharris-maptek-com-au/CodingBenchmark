Writing a complete, competitive chess engine (like Stockfish) requires tens of thousands of lines of code to handle complex rules (en passant, castling, pins), advanced board representations (Bitboards), and highly optimized search heuristics.

However, we can build a **fully functional, minimal chess bot** in C++ that demonstrates the core AI concepts: **Move Generation**, **Position Evaluation**, and the **Minimax algorithm with Alpha-Beta Pruning**.

Here is a complete, single-file C++ chess bot. For brevity and readability, this example uses a simplified pseudo-legal move generator (it understands how pieces move and capture, but omits complex rules like castling, en passant, and check-validation). 

### The C++ Chess Bot

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <cctype>

using namespace std;

// --- Constants & Definitions ---
const int EMPTY = 0;
const int PAWN = 1, KNIGHT = 2, BISHOP = 3, ROOK = 4, QUEEN = 5, KING = 6;
const int WHITE = 1, BLACK = -1;

// Piece values for evaluation
const int PAWN_VAL = 100, KNIGHT_VAL = 320, BISHOP_VAL = 330;
const int ROOK_VAL = 500, QUEEN_VAL = 900, KING_VAL = 20000;

struct Move {
    int from;
    int to;
    
    // For printing the move in standard algebraic notation (e.g., e2e4)
    string toString() const {
        string s = "";
        s += (char)('a' + (from % 8));
        s += (char)('1' + (7 - (from / 8)));
        s += (char)('a' + (to % 8));
        s += (char)('1' + (7 - (to / 8)));
        return s;
    }
};

class ChessBot {
private:
    int board[64];
    int sideToMove;

    // Helper to check if a square is on the board
    bool isValidSquare(int sq) { return sq >= 0 && sq < 64; }

    // Helper to get piece color
    int getColor(int piece) { return (piece > 0) ? WHITE : (piece < 0) ? BLACK : EMPTY; }

public:
    ChessBot() {
        // Initialize empty board
        fill(begin(board), end(board), EMPTY);
        sideToMove = WHITE;
    }

    // Load a position using FEN (Forsyth-Edwards Notation)
    void loadFEN(const string& fen) {
        fill(begin(board), end(board), EMPTY);
        int sq = 0;
        for (char c : fen) {
            if (c == ' ') break; // Stop at the end of board layout
            if (c == '/') continue;
            if (isdigit(c)) {
                sq += (c - '0');
            } else {
                int color = isupper(c) ? WHITE : BLACK;
                int type = EMPTY;
                switch (tolower(c)) {
                    case 'p': type = PAWN; break;
                    case 'n': type = KNIGHT; break;
                    case 'b': type = BISHOP; break;
                    case 'r': type = ROOK; break;
                    case 'q': type = QUEEN; break;
                    case 'k': type = KING; break;
                }
                board[sq++] = type * color;
            }
        }
        // Simplified: assume White to move if not specified
        sideToMove = (fen.find(" b ") != string::npos) ? BLACK : WHITE;
    }

    // Basic Material Evaluation Function
    // Positive score means White is winning, negative means Black is winning
    int evaluate() {
        int score = 0;
        for (int i = 0; i < 64; i++) {
            int piece = board[i];
            if (piece == EMPTY) continue;
            
            int val = 0;
            switch (abs(piece)) {
                case PAWN: val = PAWN_VAL; break;
                case KNIGHT: val = KNIGHT_VAL; break;
                case BISHOP: val = BISHOP_VAL; break;
                case ROOK: val = ROOK_VAL; break;
                case QUEEN: val = QUEEN_VAL; break;
                case KING: val = KING_VAL; break;
            }
            score += (piece > 0) ? val : -val;
        }
        return score;
    }

    // Simplified Pseudo-Legal Move Generator
    vector<Move> generateMoves() {
        vector<Move> moves;
        int dirs[8] = {-9, -8, -7, -1, 1, 7, 8, 9}; // Sliding directions
        int knightDirs[8] = {-17, -15, -10, -6, 6, 10, 15, 17};

        for (int i = 0; i < 64; i++) {
            if (getColor(board[i]) != sideToMove) continue;

            int piece = abs(board[i]);
            int row = i / 8;
            int col = i % 8;

            if (piece == PAWN) {
                int dir = (sideToMove == WHITE) ? -8 : 8;
                int startRow = (sideToMove == WHITE) ? 6 : 1;
                
                // Move forward 1
                if (isValidSquare(i + dir) && board[i + dir] == EMPTY) {
                    moves.push_back({i, i + dir});
                    // Move forward 2
                    if (row == startRow && board[i + dir * 2] == EMPTY) {
                        moves.push_back({i, i + dir * 2});
                    }
                }
                // Captures
                for (int capDir : {dir - 1, dir + 1}) {
                    int target = i + capDir;
                    // Ensure we don't wrap around the board edges
                    if (isValidSquare(target) && abs((target % 8) - col) == 1) {
                        if (getColor(board[target]) == -sideToMove) {
                            moves.push_back({i, target});
                        }
                    }
                }
            } 
            else if (piece == KNIGHT) {
                for (int d : knightDirs) {
                    int target = i + d;
                    if (isValidSquare(target)) {
                        int tCol = target % 8;
                        // Prevent wrapping around the board
                        if (abs(col - tCol) <= 2 && getColor(board[target]) != sideToMove) {
                            moves.push_back({i, target});
                        }
                    }
                }
            } 
            else { // Sliding pieces (Bishop, Rook, Queen) and King
                int startDir = (piece == ROOK) ? 4 : 0;
                int endDir = (piece == BISHOP) ? 4 : 8;
                if (piece == ROOK) { startDir = 0; endDir = 4; dirs[0]=-8; dirs[1]=-1; dirs[2]=1; dirs[3]=8; }
                if (piece == BISHOP) { startDir = 0; endDir = 4; dirs[0]=-9; dirs[1]=-7; dirs[2]=7; dirs[3]=9; }
                if (piece == QUEEN || piece == KING) { startDir = 0; endDir = 8; dirs[0]=-9; dirs[1]=-8; dirs[2]=-7; dirs[3]=-1; dirs[4]=1; dirs[5]=7; dirs[6]=8; dirs[7]=9; }

                for (int d = startDir; d < endDir; d++) {
                    int target = i;
                    while (true) {
                        int prevCol = target % 8;
                        target += dirs[d];
                        if (!isValidSquare(target) || abs((target % 8) - prevCol) > 2) break;

                        if (getColor(board[target]) == sideToMove) break; // Blocked by own piece
                        
                        moves.push_back({i, target});
                        
                        if (getColor(board[target]) == -sideToMove) break; // Capture, then stop sliding
                        if (piece == KING) break; // King only moves 1 square
                    }
                }
            }
        }
        return moves;
    }

    // Make a move on the board
    int makeMove(Move move) {
        int capturedPiece = board[move.to];
        board[move.to] = board[move.from];
        board[move.from] = EMPTY;
        sideToMove = -sideToMove;
        return capturedPiece;
    }

    // Undo a move
    void undoMove(Move move, int capturedPiece) {
        board[move.from] = board[move.to];
        board[move.to] = capturedPiece;
        sideToMove = -sideToMove;
    }

    // Minimax Algorithm with Alpha-Beta Pruning
    int alphaBeta(int depth, int alpha, int beta, bool isMaximizing) {
        if (depth == 0) {
            // From White's perspective, positive is good. 
            // If it's Black's turn, we negate the evaluation so the search works symmetrically.
            return evaluate() * (sideToMove == WHITE ? 1 : -1);
        }

        vector<Move> moves = generateMoves();
        if (moves.empty()) return -99999; // No moves = checkmate or stalemate

        int bestScore = -999999;

        for (Move move : moves) {
            int captured = makeMove(move);
            
            // Recursive call (negamax style)
            int score = -alphaBeta(depth - 1, -beta, -alpha, !isMaximizing);
            
            undoMove(move, captured);

            if (score > bestScore) bestScore = score;
            if (bestScore > alpha) alpha = bestScore;
            if (alpha >= beta) break; // Alpha-Beta Pruning
        }
        return bestScore;
    }

    // Find the best move for the current position
    Move getBestMove(int depth) {
        vector<Move> moves = generateMoves();
        Move bestMove = moves[0];
        int bestScore = -999999;
        int alpha = -999999;
        int beta = 999999;

        for (Move move : moves) {
            int captured = makeMove(move);
            int score = -alphaBeta(depth - 1, -beta, -alpha, false);
            undoMove(move, captured);

            if (score > bestScore) {
                bestScore = score;
                bestMove = move;
            }
            if (score > alpha) alpha = score;
        }
        
        cout << "Evaluation Score: " << (sideToMove == WHITE ? bestScore : -bestScore) << "\n";
        return bestMove;
    }
};

int main() {
    ChessBot bot;
    
    // Standard starting position FEN
    string fen = "rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1";
    
    cout << "Loading position: " << fen << "\n";
    bot.loadFEN(fen);

    int searchDepth = 4; // Look 4 half-moves ahead
    cout << "Searching depth " << searchDepth << "...\n";
    
    Move bestMove = bot.getBestMove(searchDepth);
    
    cout << "Best move found: " << bestMove.toString() << "\n";

    return 0;
}
```

### How it Works

1. **Board Representation:** The board is a simple 1D array of 64 integers. Positive numbers represent White pieces, negative numbers represent Black pieces.
2. **Evaluation (`evaluate`):** It counts the material on the board. A Pawn is worth 100 points, a Knight 320, etc. If White has more material, the score is positive.
3. **Search (`alphaBeta`):** It uses the **Minimax algorithm** to look ahead into the future. It assumes the opponent will play their best possible move.
4. **Alpha-Beta Pruning:** This is an optimization inside the search. If the bot finds a move that is so good that the opponent would never allow it to happen in the first place, it stops calculating that branch, saving massive amounts of processing time.

### How to Compile and Run

1. Save the code to a file named `chess_bot.cpp`.
2. Open your terminal or command prompt.
3. Compile it using g++ or clang++:
   ```bash
   g++ -O3 chess_bot.cpp -o chess_bot
   ```
   *(Note: The `-O3` flag is highly recommended for chess engines as it turns on maximum compiler optimizations, making the search much faster).*
4. Run the executable:
   ```bash
   ./chess_bot
   ```

### Next Steps for a "Real" Engine
If you want to take this foundation and build a bot that can beat human grandmasters, you would need to implement:
* **Bitboards:** Representing the board using 64-bit integers (`uint64_t`) instead of arrays. This allows you to calculate moves using lightning-fast bitwise operations (`&`, `|`, `<<`).
* **Full Legal Move Generation:** Adding logic for En Passant, Castling, Pawn Promotions, and ensuring a move doesn't leave the King in check.
* **UCI Protocol:** Implementing the Universal Chess Interface so your bot can be plugged into GUIs like Arena, Lichess, or Chessbase.
* **Quiescence Search:** Extending the search depth dynamically if a capture is occurring, to prevent the bot from making a bad move just because the capture happened right at the edge of its search depth (the "Horizon Effect").
