#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>

#ifdef _WIN32
#include <io.h>
#include <fcntl.h>
#endif

using namespace std;

struct City {
    float x, y;
    int id;
};

const int BUF_SIZE = 1 << 20;
char in_buf[BUF_SIZE];
int in_pos = 0;
int in_len = 0;

inline char next_char() {
    if (in_pos >= in_len) {
        in_len = fread(in_buf, 1, BUF_SIZE, stdin);
        in_pos = 0;
        if (in_len == 0) return EOF;
    }
    return in_buf[in_pos++];
}

inline float read_float() {
    char c;
    while ((c = next_char()) != EOF && c <= ' ');
    if (c == EOF) return 0;
    int sign = 1;
    if (c == '-') {
        sign = -1;
        c = next_char();
    }
    double res = 0;
    while (c >= '0' && c <= '9') {
        res = res * 10 + (c - '0');
        c = next_char();
    }
    if (c == '.') {
        c = next_char();
        double frac = 1;
        while (c >= '0' && c <= '9') {
            frac *= 0.1;
            res += (c - '0') * frac;
            c = next_char();
        }
    }
    if (c == 'e' || c == 'E') {
        c = next_char();
        int exp_sign = 1;
        if (c == '-') { exp_sign = -1; c = next_char(); }
        else if (c == '+') { c = next_char(); }
        int exp = 0;
        while (c >= '0' && c <= '9') {
            exp = exp * 10 + (c - '0');
            c = next_char();
        }
        res *= std::pow(10.0, exp_sign * exp);
    }
    return (float)(res * sign);
}

inline int read_int() {
    char c;
    while ((c = next_char()) != EOF && c <= ' ');
    if (c == EOF) return 0;
    int sign = 1;
    if (c == '-') {
        sign = -1;
        c = next_char();
    }
    int res = 0;
    while (c >= '0' && c <= '9') {
        res = res * 10 + (c - '0');
        c = next_char();
    }
    return res * sign;
}

char out_buf[1 << 20];
int out_pos = 0;

inline void write_int(int x) {
    if (x == 0) {
        if (out_pos + 2 >= (1 << 20)) { fwrite(out_buf, 1, out_pos, stdout); out_pos = 0; }
        out_buf[out_pos++] = '0';
        out_buf[out_pos++] = ' ';
        return;
    }
    char temp[12];
    int len = 0;
    while (x) {
        temp[len++] = (x % 10) + '0';
        x /= 10;
    }
    if (out_pos + len + 1 >= (1 << 20)) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
    for (int i = len - 1; i >= 0; i--) {
        out_buf[out_pos++] = temp[i];
    }
    out_buf[out_pos++] = ' ';
}

inline void flush_out() {
    if (out_pos > 0) {
        fwrite(out_buf, 1, out_pos, stdout);
        out_pos = 0;
    }
}

inline uint64_t get_hilbert(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int i = 31; i >= 0; i--) {
        uint32_t s = 1U << i;
        uint32_t rx = (x & s) ? 1 : 0;
        uint32_t ry = (y & s) ? 1 : 0;
        d = (d << 2) | ((rx * 3) ^ ry);
        
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

inline float dist_sq(const City& a, const City& b) {
    float dx = a.x - b.x;
    float dy = a.y - b.y;
    return dx*dx + dy*dy;
}

inline float dist(const City& a, const City& b) {
    return std::sqrt(dist_sq(a, b));
}

int main() {
#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
    _setmode(_fileno(stdout), _O_BINARY);
#endif

    int N = read_int();
    if (N == 0) return 0;

    vector<City> cities(N);
    float min_x = 1e20f, max_x = -1e20f, min_y = 1e20f, max_y = -1e20f;
    for (int i = 0; i < N; ++i) {
        cities[i].x = read_float();
        cities[i].y = read_float();
        cities[i].id = i;
        if (cities[i].x < min_x) min_x = cities[i].x;
        if (cities[i].x > max_x) max_x = cities[i].x;
        if (cities[i].y < min_y) min_y = cities[i].y;
        if (cities[i].y > max_y) max_y = cities[i].y;
    }

    if (max_x == min_x) max_x += 1e-3f;
    if (max_y == min_y) max_y += 1e-3f;

    float scale_x = 4294967295.0f / (max_x - min_x);
    float scale_y = 4294967295.0f / (max_y - min_y);

    vector<uint64_t> z_values(N);
    for (int i = 0; i < N; ++i) {
        uint32_t ix = (uint32_t)((cities[i].x - min_x) * scale_x);
        uint32_t iy = (uint32_t)((cities[i].y - min_y) * scale_y);
        z_values[i] = get_hilbert(ix, iy);
    }

    vector<int> path(N);
    for (int i = 0; i < N; ++i) path[i] = i;

    // Use standard sort with a lambda to sort path by z_values
    // To handle huge scales fast, std::sort is O(N log N) which takes ~30s for 10^8 items
    std::sort(path.begin(), path.end(), [&](int a, int b) {
        return z_values[a] < z_values[b];
    });

    // Windowed 2-opt
    int window = 40; 
    if (N > 10000000) window = 0;
    else if (N > 1000000) window = 5;
    else if (N > 100000) window = 15;
    else if (N > 10000) window = 30;
    else if (N > 1000) window = 60;
    else window = N;

    int max_iters = (N > 10000) ? 5 : 20;

    if (window > 0) {
        bool improved = true;
        for (int iter = 0; iter < max_iters && improved; ++iter) {
            improved = false;
            for (int i = 0; i < N - 1; ++i) {
                int max_j = std::min(N - 1, i + window);
                for (int j = i + 2; j <= max_j; ++j) {
                    if (i == 0 && j == N - 1) continue;
                    
                    int p_i = path[i];
                    int p_i1 = path[i+1];
                    int p_j = path[j];
                    int p_j1 = path[(j+1)%N];
                    
                    float d1 = dist(cities[p_i], cities[p_i1]);
                    float d2 = dist(cities[p_j], cities[p_j1]);
                    float d3 = dist(cities[p_i], cities[p_j]);
                    float d4 = dist(cities[p_i1], cities[p_j1]);
                    
                    if (d3 + d4 + 1e-5f < d1 + d2) {
                        std::reverse(path.begin() + i + 1, path.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    auto it = std::find(path.begin(), path.end(), 0);
    if (it != path.end()) {
        std::rotate(path.begin(), it, path.end());
    }

    for (int i = 0; i < N; ++i) {
        write_int(path[i]);
    }
    flush_out();

    return 0;
}
