using System;
using System.IO;
using System.Collections.Generic;
using System.Text;

class Solution {
    class Edge {
        public int to;
        public int weight;
        public int revIndex;
        public bool used;
    }

    struct NodeState {
        public int src;
        public long dist;
        public int parent;
    }

    static void Main(string[] args) {
        var scanner = new Scanner(Console.OpenStandardInput());
        int N = scanner.NextInt();
        if (N == -1) return;
        int M = scanner.NextInt();

        List<Edge>[] adj = new List<Edge>[N];
        for (int i = 0; i < N; i++) {
            adj[i] = new List<Edge>();
        }

        int[] degree = new int[N];
        for (int i = 0; i < M; i++) {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            int w = scanner.NextInt();

            int uIdx = adj[u].Count;
            int vIdx = adj[v].Count;

            adj[u].Add(new Edge { to = v, weight = w, revIndex = vIdx, used = false });
            adj[v].Add(new Edge { to = u, weight = w, revIndex = uIdx, used = false });

            degree[u]++;
            degree[v]++;
        }

        List<int> unmatched = new List<int>();
        for (int i = 0; i < N; i++) {
            if (degree[i] % 2 != 0) {
                unmatched.Add(i);
            }
        }

        int K = unmatched.Count;
        if (K > 0) {
            if (K <= 20) {
                SolveExactDP(N, unmatched, adj);
            } else {
                SolveHeuristic(N, unmatched, adj);
            }
        }

        List<int> circuit = GetEulerianCircuit(N, adj);
        
        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 65536)) {
            for (int i = 0; i < circuit.Count; i++) {
                writer.Write(circuit[i]);
                if (i < circuit.Count - 1) writer.Write(' ');
            }
            writer.WriteLine();
        }
    }

    static void SolveExactDP(int N, List<int> unmatched, List<Edge>[] adj) {
        int K = unmatched.Count;
        long[,] dist = new long[K, K];
        int[][] parentMap = new int[K][];

        for (int i = 0; i < K; i++) {
            parentMap[i] = new int[N];
            long[] d = new long[N];
            Array.Fill(d, long.MaxValue);
            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            
            int startNode = unmatched[i];
            d[startNode] = 0;
            pq.Enqueue(startNode, 0);
            parentMap[i][startNode] = -1;

            while (pq.Count > 0) {
                pq.TryDequeue(out int u, out long d_u);
                if (d_u > d[u]) continue;

                foreach (var edge in adj[u]) {
                    int v = edge.to;
                    long nDist = d_u + edge.weight;
                    if (nDist < d[v]) {
                        d[v] = nDist;
                        parentMap[i][v] = u;
                        pq.Enqueue(v, nDist);
                    }
                }
            }
            for (int j = 0; j < K; j++) {
                dist[i, j] = d[unmatched[j]];
            }
        }

        long[] dp = new long[1 << K];
        Array.Fill(dp, long.MaxValue);
        dp[0] = 0;
        int[] match = new int[1 << K];

        for (int mask = 0; mask < (1 << K); mask++) {
            if (dp[mask] == long.MaxValue) continue;

            int i = -1;
            for (int b = 0; b < K; b++) {
                if ((mask & (1 << b)) == 0) {
                    i = b;
                    break;
                }
            }
            if (i == -1) continue;

            for (int j = i + 1; j < K; j++) {
                if ((mask & (1 << j)) == 0) {
                    if (dist[i, j] == long.MaxValue) continue;
                    long nCost = dp[mask] + dist[i, j];
                    int nMask = mask | (1 << i) | (1 << j);
                    if (nCost < dp[nMask]) {
                        dp[nMask] = nCost;
                        match[nMask] = j;
                    }
                }
            }
        }

        int currMask = (1 << K) - 1;
        while (currMask > 0) {
            int i = -1;
            for (int b = 0; b < K; b++) {
                if ((currMask & (1 << b)) != 0) {
                    i = b;
                    break;
                }
            }
            int j = match[currMask];

            int currNode = unmatched[j];
            int srcNode = unmatched[i];
            while (currNode != srcNode) {
                int p = parentMap[i][currNode];
                DuplicateEdge(currNode, p, adj);
                currNode = p;
            }

            currMask ^= (1 << i);
            currMask ^= (1 << j);
        }
    }

    static void SolveHeuristic(int N, List<int> unmatched, List<Edge>[] adj) {
        NodeState[] state = new NodeState[N];
        
        while (unmatched.Count > 0) {
            for (int i = 0; i < N; i++) {
                state[i].src = -1;
            }
            
            PriorityQueue<(int u, int src), long> pq = new PriorityQueue<(int, int), long>();

            foreach (int u in unmatched) {
                state[u].src = u;
                state[u].dist = 0;
                state[u].parent = -1;
                pq.Enqueue((u, u), 0);
            }

            bool[] matchedNow = new bool[N];
            int matchesInThisPhase = 0;

            while (pq.Count > 0) {
                pq.TryDequeue(out var item, out long d_u);
                int u = item.u;
                int src = item.src;

                if (matchedNow[src]) continue;
                if (state[u].dist != d_u || state[u].src != src) continue;

                foreach (var edge in adj[u]) {
                    int v = edge.to;
                    long nDist = state[u].dist + edge.weight;

                    if (state[v].src == -1) {
                        state[v].src = src;
                        state[v].dist = nDist;
                        state[v].parent = u;
                        pq.Enqueue((v, src), nDist);
                    } else if (state[v].src != src) {
                        int src2 = state[v].src;
                        if (!matchedNow[src] && !matchedNow[src2]) {
                            matchedNow[src] = true;
                            matchedNow[src2] = true;
                            matchesInThisPhase++;

                            AddPath(u, src, state, adj);
                            AddPath(v, src2, state, adj);
                            DuplicateEdge(u, v, adj);

                            break;
                        }
                    }
                }
            }

            if (matchesInThisPhase == 0) break;
            unmatched.RemoveAll(x => matchedNow[x]);
        }
    }

    static void AddPath(int endNode, int srcNode, NodeState[] state, List<Edge>[] adj) {
        int curr = endNode;
        while (curr != srcNode) {
            int p = state[curr].parent;
            if (p == -1) break;
            DuplicateEdge(curr, p, adj);
            curr = p;
        }
    }

    static void DuplicateEdge(int u, int v, List<Edge>[] adj) {
        int uIdx = adj[u].Count;
        int vIdx = adj[v].Count;
        adj[u].Add(new Edge { to = v, weight = 0, revIndex = vIdx, used = false });
        adj[v].Add(new Edge { to = u, weight = 0, revIndex = uIdx, used = false });
    }

    static List<int> GetEulerianCircuit(int N, List<Edge>[] adj) {
        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);

        int[] edgeIndex = new int[N];

        while (stack.Count > 0) {
            int u = stack.Peek();
            if (edgeIndex[u] < adj[u].Count) {
                Edge e = adj[u][edgeIndex[u]];
                edgeIndex[u]++;
                if (!e.used) {
                    e.used = true;
                    adj[e.to][e.revIndex].used = true;
                    stack.Push(e.to);
                }
            } else {
                circuit.Add(u);
                stack.Pop();
            }
        }
        circuit.Reverse();
        return circuit;
    }

    class Scanner {
        Stream stream;
        byte[] buffer = new byte[1024 * 1024];
        int head;
        int tail;

        public Scanner(Stream stream) {
            this.stream = stream;
        }

        public long NextLong() {
            int c = Read();
            while (c <= 32) {
                if (c == -1) return -1;
                c = Read();
            }
            long res = 0;
            bool neg = false;
            if (c == '-') {
                neg = true;
                c = Read();
            }
            while (c > 32) {
                res = res * 10 + c - '0';
                c = Read();
            }
            return neg ? -res : res;
        }

        public int NextInt() => (int)NextLong();

        int Read() {
            if (head >= tail) {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }
}
