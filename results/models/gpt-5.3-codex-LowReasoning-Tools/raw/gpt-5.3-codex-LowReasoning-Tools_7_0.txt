#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>

#include <utility>

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double norm2(const Vec3& v) { return dot(v, v); }

struct Tri { int a{}, b{}, c{}; };

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
};

struct AABB {
    Vec3 mn{ std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity() };
    Vec3 mx{ -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity() };
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }
};

static AABB bounds(const Mesh& m) {
    AABB b;
    for (const auto& p : m.v) b.expand(p);
    return b;
}

static bool overlap(const AABB& a, const AABB& b) {
    return !(a.mx.x < b.mn.x || b.mx.x < a.mn.x || a.mx.y < b.mn.y || b.mx.y < a.mn.y || a.mx.z < b.mn.z || b.mx.z < a.mn.z);
}

static Mesh concatenate(const Mesh& A, const Mesh& B) {
    Mesh out;
    out.v = A.v;
    out.v.insert(out.v.end(), B.v.begin(), B.v.end());
    out.f = A.f;
    int off = static_cast<int>(A.v.size());
    for (auto t : B.f) out.f.push_back({ t.a + off, t.b + off, t.c + off });
    return out;
}

static bool rayIntersectsTri(const Vec3& o, const Vec3& d, const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    const double eps = 1e-12;
    Vec3 e1 = v1 - v0;
    Vec3 e2 = v2 - v0;
    Vec3 p = cross(d, e2);
    double det = dot(e1, p);
    if (std::abs(det) < eps) return false;
    double invDet = 1.0 / det;
    Vec3 t = o - v0;
    double u = dot(t, p) * invDet;
    if (u < 0.0 || u > 1.0) return false;
    Vec3 q = cross(t, e1);
    double v = dot(d, q) * invDet;
    if (v < 0.0 || u + v > 1.0) return false;
    double tt = dot(e2, q) * invDet;
    return tt > eps;
}

static bool pointInsideMesh(const Mesh& m, const Vec3& p) {
    Vec3 d{1.0, 0.17364817766, 0.34202014332};
    int hits = 0;
    for (const auto& t : m.f) {
        if (rayIntersectsTri(p, d, m.v[t.a], m.v[t.b], m.v[t.c])) ++hits;
    }
    return (hits & 1) != 0;
}

static double pointTriDist2(const Vec3& p, const Vec3& a, const Vec3& b, const Vec3& c) {
    Vec3 ab = b - a, ac = c - a, ap = p - a;
    double d1 = dot(ab, ap), d2 = dot(ac, ap);
    if (d1 <= 0.0 && d2 <= 0.0) return norm2(ap);

    Vec3 bp = p - b;
    double d3 = dot(ab, bp), d4 = dot(ac, bp);
    if (d3 >= 0.0 && d4 <= d3) return norm2(bp);

    double vc = d1 * d4 - d3 * d2;
    if (vc <= 0.0 && d1 >= 0.0 && d3 <= 0.0) {
        double v = d1 / (d1 - d3);
        Vec3 proj = a + ab * v;
        return norm2(p - proj);
    }

    Vec3 cp = p - c;
    double d5 = dot(ab, cp), d6 = dot(ac, cp);
    if (d6 >= 0.0 && d5 <= d6) return norm2(cp);

    double vb = d5 * d2 - d1 * d6;
    if (vb <= 0.0 && d2 >= 0.0 && d6 <= 0.0) {
        double w = d2 / (d2 - d6);
        Vec3 proj = a + ac * w;
        return norm2(p - proj);
    }

    double va = d3 * d6 - d5 * d4;
    if (va <= 0.0 && (d4 - d3) >= 0.0 && (d5 - d6) >= 0.0) {
        Vec3 bc = c - b;
        double w = (d4 - d3) / ((d4 - d3) + (d5 - d6));
        Vec3 proj = b + bc * w;
        return norm2(p - proj);
    }

    double denom = 1.0 / (va + vb + vc);
    double v = vb * denom, w = vc * denom;
    Vec3 proj = a + ab * v + ac * w;
    return norm2(p - proj);
}

static double signedDistance(const Mesh& m, const Vec3& p) {
    double best = std::numeric_limits<double>::infinity();
    for (const auto& t : m.f) {
        best = std::min(best, pointTriDist2(p, m.v[t.a], m.v[t.b], m.v[t.c]));
    }
    double d = std::sqrt(best);
    return pointInsideMesh(m, p) ? -d : d;
}

struct IVec3Hash {
    std::size_t operator()(const std::array<int, 3>& k) const noexcept {
        std::uint64_t h = 1469598103934665603ull;
        auto mix = [&](int v) {
            h ^= static_cast<std::uint64_t>(static_cast<std::uint32_t>(v));
            h *= 1099511628211ull;
        };
        mix(k[0]); mix(k[1]); mix(k[2]);
        return static_cast<std::size_t>(h);
    }
};

static int addVertexDedup(std::vector<Vec3>& verts, std::unordered_map<std::array<int, 3>, int, IVec3Hash>& mp, const Vec3& p, double q) {
    std::array<int, 3> k{ static_cast<int>(std::llround(p.x / q)), static_cast<int>(std::llround(p.y / q)), static_cast<int>(std::llround(p.z / q)) };
    auto it = mp.find(k);
    if (it != mp.end()) return it->second;
    int id = static_cast<int>(verts.size());
    verts.push_back(p);
    mp.emplace(k, id);
    return id;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA); A.f.resize(FA);
    B.v.resize(VB); B.f.resize(FB);

    for (int i = 0; i < VA; ++i) std::cin >> A.v[i].x >> A.v[i].y >> A.v[i].z;
    for (int i = 0; i < FA; ++i) std::cin >> A.f[i].a >> A.f[i].b >> A.f[i].c;
    for (int i = 0; i < VB; ++i) std::cin >> B.v[i].x >> B.v[i].y >> B.v[i].z;
    for (int i = 0; i < FB; ++i) std::cin >> B.f[i].a >> B.f[i].b >> B.f[i].c;

    AABB a = bounds(A), b = bounds(B);
    if (!overlap(a, b)) {
        Mesh out = concatenate(A, B);
        std::cout << out.v.size() << ' ' << out.f.size() << '\n';
        for (auto& p : out.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (auto& t : out.f) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
        return 0;
    }

    Vec3 ca{0,0,0}, cb{0,0,0};
    for (auto& p : A.v) ca = ca + p;
    for (auto& p : B.v) cb = cb + p;
    ca = ca * (1.0 / std::max(1, VA));
    cb = cb * (1.0 / std::max(1, VB));

    if (pointInsideMesh(B, ca)) {
        std::cout << B.v.size() << ' ' << B.f.size() << '\n';
        for (auto& p : B.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (auto& t : B.f) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
        return 0;
    }
    if (pointInsideMesh(A, cb)) {
        std::cout << A.v.size() << ' ' << A.f.size() << '\n';
        for (auto& p : A.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
        for (auto& t : A.f) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
        return 0;
    }

    AABB u;
    for (auto& p : A.v) u.expand(p);
    for (auto& p : B.v) u.expand(p);

    std::size_t complexity = static_cast<std::size_t>(FA) + static_cast<std::size_t>(FB);
    int N = 64;
    if (complexity < 5000) N = 80;
    else if (complexity > 200000) N = 40;
    else if (complexity > 1000000) N = 28;

    Vec3 diag = u.mx - u.mn;
    double maxd = std::max({diag.x, diag.y, diag.z, 1e-9});
    double h = maxd / N;

    // Fallback manifold approximation: boundary voxel extraction of inside(A) || inside(B).
    std::vector<Vec3> outV;
    std::vector<Tri> outF;
    std::unordered_map<std::array<int,3>, int, IVec3Hash> vmap;
    const double q = h * 1e-3;

    auto sampleInside = [&](int i, int j, int k) {
        Vec3 p{u.mn.x + (i + 0.5) * h, u.mn.y + (j + 0.5) * h, u.mn.z + (k + 0.5) * h};
        return pointInsideMesh(A, p) || pointInsideMesh(B, p);
    };

    std::vector<std::uint8_t> grid(static_cast<std::size_t>(N) * N * N, 0);
    auto idx = [N](int i, int j, int k) -> std::size_t { return (static_cast<std::size_t>(i) * N + j) * N + k; };

    for (int i = 0; i < N; ++i) {
        for (int j = 0; j < N; ++j) {
            for (int k = 0; k < N; ++k) {
                grid[idx(i,j,k)] = sampleInside(i,j,k) ? 1 : 0;
            }
        }
    }

    const int di[6] = {1,-1,0,0,0,0};
    const int dj[6] = {0,0,1,-1,0,0};
    const int dk[6] = {0,0,0,0,1,-1};

    for (int i = 0; i < N; ++i) for (int j = 0; j < N; ++j) for (int k = 0; k < N; ++k) {
        if (!grid[idx(i,j,k)]) continue;
        for (int f = 0; f < 6; ++f) {
            int ni = i + di[f], nj = j + dj[f], nk = k + dk[f];
            bool outside = (ni < 0 || nj < 0 || nk < 0 || ni >= N || nj >= N || nk >= N) || !grid[idx(std::clamp(ni,0,N-1), std::clamp(nj,0,N-1), std::clamp(nk,0,N-1))];
            if (!outside) continue;

            double x0 = u.mn.x + i * h, x1 = x0 + h;
            double y0 = u.mn.y + j * h, y1 = y0 + h;
            double z0 = u.mn.z + k * h, z1 = z0 + h;

            Vec3 p[4];
            if (f == 0) { p[0]={x1,y0,z0}; p[1]={x1,y1,z0}; p[2]={x1,y1,z1}; p[3]={x1,y0,z1}; }
            if (f == 1) { p[0]={x0,y0,z0}; p[1]={x0,y0,z1}; p[2]={x0,y1,z1}; p[3]={x0,y1,z0}; }
            if (f == 2) { p[0]={x0,y1,z0}; p[1]={x0,y1,z1}; p[2]={x1,y1,z1}; p[3]={x1,y1,z0}; }
            if (f == 3) { p[0]={x0,y0,z0}; p[1]={x1,y0,z0}; p[2]={x1,y0,z1}; p[3]={x0,y0,z1}; }
            if (f == 4) { p[0]={x0,y0,z1}; p[1]={x1,y0,z1}; p[2]={x1,y1,z1}; p[3]={x0,y1,z1}; }
            if (f == 5) { p[0]={x0,y0,z0}; p[1]={x0,y1,z0}; p[2]={x1,y1,z0}; p[3]={x1,y0,z0}; }

            int i0 = addVertexDedup(outV, vmap, p[0], q);
            int i1 = addVertexDedup(outV, vmap, p[1], q);
            int i2 = addVertexDedup(outV, vmap, p[2], q);
            int i3 = addVertexDedup(outV, vmap, p[3], q);
            outF.push_back({i0,i1,i2});
            outF.push_back({i0,i2,i3});
        }
    }

    std::cout << outV.size() << ' ' << outF.size() << '\n';
    for (const auto& p : outV) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : outF) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
    return 0;
}
