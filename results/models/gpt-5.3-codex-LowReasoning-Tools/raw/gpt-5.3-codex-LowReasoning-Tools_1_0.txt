#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct City {
    float x;
    float y;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline uint32_t part1by1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}

static inline uint64_t morton2D_16bit(uint32_t x, uint32_t y) {
    return (static_cast<uint64_t>(part1by1(y)) << 1) | static_cast<uint64_t>(part1by1(x));
}

std::vector<int> mortonRoute(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<int> ord(n);
    std::iota(ord.begin(), ord.end(), 0);
    if (n <= 1) return ord;

    float minx = c[0].x, maxx = c[0].x, miny = c[0].y, maxy = c[0].y;
    for (const auto& p : c) {
        if (p.x < minx) minx = p.x;
        if (p.x > maxx) maxx = p.x;
        if (p.y < miny) miny = p.y;
        if (p.y > maxy) maxy = p.y;
    }

    const double rx = std::max(1e-30, static_cast<double>(maxx) - static_cast<double>(minx));
    const double ry = std::max(1e-30, static_cast<double>(maxy) - static_cast<double>(miny));

    std::vector<uint64_t> key(n);
    for (int i = 0; i < n; ++i) {
        const double nx = (static_cast<double>(c[i].x) - static_cast<double>(minx)) / rx;
        const double ny = (static_cast<double>(c[i].y) - static_cast<double>(miny)) / ry;
        const uint32_t qx = static_cast<uint32_t>(std::min(65535.0, std::max(0.0, nx * 65535.0)));
        const uint32_t qy = static_cast<uint32_t>(std::min(65535.0, std::max(0.0, ny * 65535.0)));
        key[i] = morton2D_16bit(qx, qy);
    }

    std::sort(ord.begin(), ord.end(), [&](int a, int b) {
        if (key[a] != key[b]) return key[a] < key[b];
        return a < b;
    });
    return ord;
}

std::vector<int> exactHeldKarp(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    if (n == 1) return {0};

    const int m = n - 1;
    const int S = 1 << m;
    const double INF = std::numeric_limits<double>::infinity();

    std::vector<double> dp(static_cast<size_t>(S) * m, INF);
    std::vector<int16_t> parent(static_cast<size_t>(S) * m, -1);

    auto at = [m](int mask, int j) -> size_t {
        return static_cast<size_t>(mask) * m + j;
    };

    for (int j = 0; j < m; ++j) {
        const int mask = 1 << j;
        dp[at(mask, j)] = std::sqrt(dist2(c[0], c[j + 1]));
    }

    for (int mask = 1; mask < S; ++mask) {
        for (int j = 0; j < m; ++j) {
            if (!(mask & (1 << j))) continue;
            const int pmask = mask ^ (1 << j);
            if (pmask == 0) continue;
            double best = INF;
            int bestk = -1;
            for (int k = 0; k < m; ++k) {
                if (!(pmask & (1 << k))) continue;
                const double cand = dp[at(pmask, k)] + std::sqrt(dist2(c[k + 1], c[j + 1]));
                if (cand < best) {
                    best = cand;
                    bestk = k;
                }
            }
            dp[at(mask, j)] = best;
            parent[at(mask, j)] = static_cast<int16_t>(bestk);
        }
    }

    int full = S - 1;
    double best = INF;
    int endj = -1;
    for (int j = 0; j < m; ++j) {
        const double cand = dp[at(full, j)] + std::sqrt(dist2(c[j + 1], c[0]));
        if (cand < best) {
            best = cand;
            endj = j;
        }
    }

    std::vector<int> route_rev;
    route_rev.reserve(m);
    int mask = full;
    int j = endj;
    while (j != -1) {
        route_rev.push_back(j + 1);
        const int pj = parent[at(mask, j)];
        mask ^= (1 << j);
        j = pj;
    }

    std::vector<int> route;
    route.reserve(n);
    route.push_back(0);
    for (int i = static_cast<int>(route_rev.size()) - 1; i >= 0; --i) route.push_back(route_rev[i]);
    return route;
}

std::vector<int> greedyFromOrder(const std::vector<City>& c, const std::vector<int>& base, int window) {
    const int n = static_cast<int>(c.size());
    std::vector<int> pos(n);
    for (int i = 0; i < n; ++i) pos[base[i]] = i;

    std::vector<char> used(n, 0);
    std::vector<int> route;
    route.reserve(n);
    int cur = 0;
    used[cur] = 1;
    route.push_back(cur);

    for (int step = 1; step < n; ++step) {
        const int p = pos[cur];
        int best = -1;
        double bestd = std::numeric_limits<double>::infinity();

        const int l = std::max(0, p - window);
        const int r = std::min(n - 1, p + window);

        for (int i = l; i <= r; ++i) {
            const int cand = base[i];
            if (used[cand]) continue;
            const double d = dist2(c[cur], c[cand]);
            if (d < bestd) {
                bestd = d;
                best = cand;
            }
        }

        if (best == -1) {
            int left = p - 1, right = p + 1;
            while (left >= 0 || right < n) {
                if (left >= 0) {
                    int cand = base[left--];
                    if (!used[cand]) {
                        best = cand;
                        break;
                    }
                }
                if (right < n) {
                    int cand = base[right++];
                    if (!used[cand]) {
                        best = cand;
                        break;
                    }
                }
            }
        }

        if (best == -1) {
            for (int i = 0; i < n; ++i) {
                if (!used[i]) {
                    best = i;
                    break;
                }
            }
        }

        used[best] = 1;
        route.push_back(best);
        cur = best;
    }
    return route;
}

void twoOptLimited(const std::vector<City>& c, std::vector<int>& route, int passes, int maxSpan) {
    const int n = static_cast<int>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 1; i + 2 < n; ++i) {
            const int a = route[i - 1], b = route[i];
            const int jEnd = std::min(n - 2, i + maxSpan);
            for (int j = i + 1; j <= jEnd; ++j) {
                const int c1 = route[j], d = route[j + 1];
                const double before = dist2(c[a], c[b]) + dist2(c[c1], c[d]);
                const double after = dist2(c[a], c[c1]) + dist2(c[b], c[d]);
                if (after + 1e-18 < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    if (!(std::cin >> n) || n <= 0) return 0;

    std::vector<City> cities(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> cities[i].x >> cities[i].y;
    }

    std::vector<int> route;

    if (n <= 12) {
        route = exactHeldKarp(cities);
    } else {
        auto base = mortonRoute(cities);

        if (n <= 2000) {
            route = greedyFromOrder(cities, base, 64);
            twoOptLimited(cities, route, 3, 128);
        } else if (n <= 20000) {
            route = greedyFromOrder(cities, base, 32);
            twoOptLimited(cities, route, 1, 48);
        } else {
            route = std::move(base);
        }

        if (!route.empty() && route[0] != 0) {
            auto it = std::find(route.begin(), route.end(), 0);
            if (it != route.end()) std::rotate(route.begin(), it, route.end());
        }
    }

    if (route.empty()) {
        route.resize(n);
        std::iota(route.begin(), route.end(), 0);
    }

    if (route[0] != 0) {
        auto it = std::find(route.begin(), route.end(), 0);
        if (it != route.end()) std::rotate(route.begin(), it, route.end());
    }

    for (int i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << route[i];
    }
    std::cout << '\n';
    return 0;
}
