use std::cmp::Ordering;
use std::collections::HashMap;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self { Self { x, y } }
    fn add(self, o: Pt) -> Pt { Pt::new(self.x + o.x, self.y + o.y) }
    fn sub(self, o: Pt) -> Pt { Pt::new(self.x - o.x, self.y - o.y) }
    fn dot(self, o: Pt) -> f64 { self.x * o.x + self.y * o.y }
    fn cross(self, o: Pt) -> f64 { self.x * o.y - self.y * o.x }
}

#[derive(Clone)]
struct Poly {
    v: Vec<Pt>,
}

#[derive(Clone, Copy)]
struct BBox {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl BBox {
    fn overlap(&self, o: &BBox) -> bool {
        !(self.maxx < o.minx - EPS || o.maxx < self.minx - EPS || self.maxy < o.miny - EPS || o.maxy < self.miny - EPS)
    }
    fn area(&self) -> f64 { (self.maxx - self.minx).max(0.0) * (self.maxy - self.miny).max(0.0) }
}

fn polygon_area_abs(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut s = 0.0;
    for i in 0..n {
        s += poly[i].cross(poly[(i + 1) % n]);
    }
    (0.5 * s).abs()
}

fn bbox(poly: &[Pt]) -> BBox {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = -f64::INFINITY;
    let mut maxy = -f64::INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    BBox { minx, miny, maxx, maxy }
}

fn rotate(poly: &[Pt], angle: f64) -> Vec<Pt> {
    let c = angle.cos();
    let s = angle.sin();
    poly.iter().map(|p| Pt::new(c * p.x - s * p.y, s * p.x + c * p.y)).collect()
}

fn translate(poly: &[Pt], t: Pt) -> Vec<Pt> {
    poly.iter().map(|p| p.add(t)).collect()
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 { (b.sub(a)).cross(c.sub(a)) }

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    orient(a, b, p).abs() <= EPS
        && p.x >= a.x.min(b.x) - EPS && p.x <= a.x.max(b.x) + EPS
        && p.y >= a.y.min(b.y) - EPS && p.y <= a.y.max(b.y) + EPS
}

fn segments_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    if o1.abs() <= EPS && on_segment(a, b, c) { return true; }
    if o2.abs() <= EPS && on_segment(a, b, d) { return true; }
    if o3.abs() <= EPS && on_segment(c, d, a) { return true; }
    if o4.abs() <= EPS && on_segment(c, d, b) { return true; }

    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn point_in_poly_or_boundary(p: Pt, poly: &[Pt]) -> bool {
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) { return true; }
    }

    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let xint = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if xint >= p.x - EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn poly_edges_intersect(a: &[Pt], b: &[Pt]) -> bool {
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if segments_intersect(a1, a2, b1, b2) {
                // Allow touching at vertices/edges as overlap-unsafe in cutting stock.
                return true;
            }
        }
    }
    false
}

fn polys_overlap(a: &[Pt], b: &[Pt]) -> bool {
    if poly_edges_intersect(a, b) { return true; }
    if point_in_poly_or_boundary(a[0], b) { return true; }
    if point_in_poly_or_boundary(b[0], a) { return true; }
    false
}

fn poly_inside_stock(piece: &[Pt], stock: &[Pt], stock_edges: &[(Pt, Pt)]) -> bool {
    for &p in piece {
        if !point_in_poly_or_boundary(p, stock) {
            return false;
        }
    }
    for i in 0..piece.len() {
        let a = piece[i];
        let b = piece[(i + 1) % piece.len()];
        for &(s1, s2) in stock_edges {
            if segments_intersect(a, b, s1, s2) {
                // Shared boundary contact is okay only when both endpoints on same stock edge.
                if !(on_segment(s1, s2, a) && on_segment(s1, s2, b)) {
                    return false;
                }
            }
        }
    }
    true
}

#[derive(Clone)]
struct Placed {
    poly: Vec<Pt>,
    bb: BBox,
}

struct StockState {
    placed: Vec<Placed>,
    grid: HashMap<(i32, i32), Vec<usize>>,
    cell: f64,
}

impl StockState {
    fn new(cell: f64) -> Self {
        Self { placed: Vec::new(), grid: HashMap::new(), cell }
    }
    fn cells_for_bbox(&self, b: &BBox) -> (i32, i32, i32, i32) {
        let x0 = (b.minx / self.cell).floor() as i32;
        let x1 = (b.maxx / self.cell).floor() as i32;
        let y0 = (b.miny / self.cell).floor() as i32;
        let y1 = (b.maxy / self.cell).floor() as i32;
        (x0, x1, y0, y1)
    }
    fn candidates(&self, b: &BBox) -> Vec<usize> {
        let (x0, x1, y0, y1) = self.cells_for_bbox(b);
        let mut seen = HashMap::<usize, ()>::new();
        let mut out = Vec::new();
        for x in x0..=x1 {
            for y in y0..=y1 {
                if let Some(v) = self.grid.get(&(x, y)) {
                    for &idx in v {
                        if seen.insert(idx, ()).is_none() {
                            out.push(idx);
                        }
                    }
                }
            }
        }
        out
    }
    fn add(&mut self, p: Vec<Pt>) {
        let bb = bbox(&p);
        let idx = self.placed.len();
        self.placed.push(Placed { poly: p, bb });
        let (x0, x1, y0, y1) = self.cells_for_bbox(&bb);
        for x in x0..=x1 {
            for y in y0..=y1 {
                self.grid.entry((x, y)).or_default().push(idx);
            }
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }

    let mut pieces: Vec<Poly> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut v = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            v.push(Pt::new(x, y));
        }
        pieces.push(Poly { v });
    }

    let stock_bb = bbox(&stock);
    let stock_w = (stock_bb.maxx - stock_bb.minx).max(1.0);
    let stock_h = (stock_bb.maxy - stock_bb.miny).max(1.0);
    let stock_area = polygon_area_abs(&stock).max(1.0);
    let stock_edges: Vec<(Pt, Pt)> = (0..stock.len()).map(|i| (stock[i], stock[(i + 1) % stock.len()])).collect();

    let (rot_count, attempt_grid, max_candidates_per_piece) = if np <= 80 {
        (36usize, 20usize, 600usize)
    } else if np <= 1000 {
        (16usize, 14usize, 220usize)
    } else {
        (8usize, 10usize, 80usize)
    };

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        let aa = polygon_area_abs(&pieces[a].v);
        let bb = polygon_area_abs(&pieces[b].v);
        bb.partial_cmp(&aa).unwrap_or(Ordering::Equal)
    });

    let cell = ((stock_w * stock_h) / (np as f64).sqrt().max(1.0)).sqrt().max(1.0);
    let mut stocks: Vec<StockState> = Vec::new();

    let mut out_stock_idx = vec![0usize; np];
    let mut out_t = vec![Pt::new(0.0, 0.0); np];
    let mut out_rot = vec![0.0f64; np];

    // Precompute candidate rotations.
    let rotations: Vec<f64> = (0..rot_count).map(|k| 2.0 * PI * (k as f64) / (rot_count as f64)).collect();

    // Deterministic low-discrepancy placement points in stock bbox.
    let mut seeds = Vec::new();
    for i in 0..attempt_grid {
        for j in 0..attempt_grid {
            let fx = (i as f64 + 0.5) / (attempt_grid as f64);
            let fy = (j as f64 + 0.5) / (attempt_grid as f64);
            let x = stock_bb.minx + fx * stock_w;
            let y = stock_bb.miny + fy * stock_h;
            seeds.push(Pt::new(x, y));
        }
    }
    seeds.sort_by(|a, b| {
        let ca = (a.x - (stock_bb.minx + stock_w * 0.5)).powi(2) + (a.y - (stock_bb.miny + stock_h * 0.5)).powi(2);
        let cb = (b.x - (stock_bb.minx + stock_w * 0.5)).powi(2) + (b.y - (stock_bb.miny + stock_h * 0.5)).powi(2);
        ca.partial_cmp(&cb).unwrap_or(Ordering::Equal)
    });

    for &pi in &order {
        let base = &pieces[pi].v;
        let mut placed = false;

        // Try existing stocks first.
        for si in 0..stocks.len() {
            if placed { break; }
            'rot_loop_existing: for &ang in &rotations {
                let rp = rotate(base, ang);
                let rb = bbox(&rp);
                let anchor = Pt::new((rb.minx + rb.maxx) * 0.5, (rb.miny + rb.maxy) * 0.5);
                let mut tries = 0usize;
                for &s in &seeds {
                    if tries >= max_candidates_per_piece { break; }
                    tries += 1;
                    let t = s.sub(anchor);
                    let tp = translate(&rp, t);
                    let bb = bbox(&tp);
                    if bb.minx < stock_bb.minx - EPS || bb.maxx > stock_bb.maxx + EPS || bb.miny < stock_bb.miny - EPS || bb.maxy > stock_bb.maxy + EPS {
                        continue;
                    }
                    if bb.area() > stock_area * 1.1 { continue; }
                    if !poly_inside_stock(&tp, &stock, &stock_edges) { continue; }

                    let cand = stocks[si].candidates(&bb);
                    let mut ok = true;
                    for idx in cand {
                        let op = &stocks[si].placed[idx];
                        if !bb.overlap(&op.bb) { continue; }
                        if polys_overlap(&tp, &op.poly) {
                            ok = false;
                            break;
                        }
                    }
                    if ok {
                        stocks[si].add(tp);
                        out_stock_idx[pi] = si;
                        out_t[pi] = t;
                        out_rot[pi] = ang;
                        placed = true;
                        break 'rot_loop_existing;
                    }
                }
            }
        }

        if !placed {
            // Open new stock and place aggressively near center.
            let mut ss = StockState::new(cell);
            let mut found = None;
            'new_rot: for &ang in &rotations {
                let rp = rotate(base, ang);
                let rb = bbox(&rp);
                let anchor = Pt::new((rb.minx + rb.maxx) * 0.5, (rb.miny + rb.maxy) * 0.5);
                for &s in &seeds {
                    let t = s.sub(anchor);
                    let tp = translate(&rp, t);
                    if poly_inside_stock(&tp, &stock, &stock_edges) {
                        found = Some((tp, t, ang));
                        break 'new_rot;
                    }
                }
            }

            if let Some((tp, t, ang)) = found {
                ss.add(tp);
                let si = stocks.len();
                stocks.push(ss);
                out_stock_idx[pi] = si;
                out_t[pi] = t;
                out_rot[pi] = ang;
            } else {
                // Fallback for pathological coordinates: keep as-is in new stock.
                let si = stocks.len();
                stocks.push(ss);
                out_stock_idx[pi] = si;
                out_t[pi] = Pt::new(0.0, 0.0);
                out_rot[pi] = 0.0;
            }
        }
    }

    println!("{}", stocks.len());
    for i in 0..np {
        println!("{} {:.9} {:.9} {:.9}", out_stock_idx[i], out_t[i].x, out_t[i].y, out_rot[i]);
    }
}
