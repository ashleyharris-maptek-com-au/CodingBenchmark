using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
            _len = 0;
            _ptr = 0;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private struct EdgeRef
    {
        public int To;
        public int Id;
        public EdgeRef(int to, int id)
        {
            To = to;
            Id = id;
        }
    }

    private struct Frame
    {
        public int Node;
        public int Parent;
        public int ParentEdge;
        public Frame(int node, int parent, int parentEdge)
        {
            Node = node;
            Parent = parent;
            ParentEdge = parentEdge;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n = fs.NextInt();
        int m = fs.NextInt();

        var adj = new List<EdgeRef>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<EdgeRef>();

        for (int id = 0; id < m; id++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt(); // weight not needed for feasibility route
            adj[u].Add(new EdgeRef(v, id));
            adj[v].Add(new EdgeRef(u, id));
        }

        var used = new bool[m];
        var it = new int[n];

        long estimatedLen = 2L * m + 1;
        int cap = estimatedLen > int.MaxValue ? int.MaxValue : (int)estimatedLen;
        var route = new List<int>(cap);
        route.Add(0);

        var stack = new Stack<Frame>();
        stack.Push(new Frame(0, -1, -1));

        while (stack.Count > 0)
        {
            var f = stack.Peek();
            int u = f.Node;

            bool advanced = false;
            var list = adj[u];
            int idx = it[u];
            while (idx < list.Count)
            {
                var e = list[idx++];
                if (used[e.Id]) continue;

                used[e.Id] = true;
                it[u] = idx;

                int v = e.To;
                route.Add(v);
                stack.Push(new Frame(v, u, e.Id));
                advanced = true;
                break;
            }
            if (advanced) continue;

            stack.Pop();
            if (f.Parent != -1)
            {
                route.Add(f.Parent);
            }
        }

        var sb = new StringBuilder(route.Count * 3);
        sb.Append(route[0]);
        for (int i = 1; i < route.Count; i++)
        {
            sb.Append(' ');
            sb.Append(route[i]);
        }

        Console.WriteLine(sb.ToString());
    }
}
