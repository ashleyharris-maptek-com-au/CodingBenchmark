using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private struct Cut
    {
        public int Length;
        public int Index;
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _ptr;
        private int _len;

        public FastReader(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());

        int n = reader.NextInt();
        if (n == int.MinValue)
        {
            return;
        }

        int stockLength = reader.NextInt();
        var cuts = new Cut[n];

        for (int i = 0; i < n; i++)
        {
            int len = reader.NextInt();
            if (len > stockLength)
            {
                // No valid packing exists; keep output deterministic.
                Console.WriteLine("0");
                return;
            }

            cuts[i] = new Cut { Length = len, Index = i };
        }

        Array.Sort(cuts, (a, b) => b.Length.CompareTo(a.Length));

        var bins = new List<List<int>>(Math.Max(4, n / 4));
        var pq = new PriorityQueue<int, int>();

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Length;
            int idx = cuts[i].Index;

            if (pq.Count == 0)
            {
                int binId = bins.Count;
                var list = new List<int>(4) { idx };
                bins.Add(list);

                int rem = stockLength - len;
                pq.Enqueue(binId, -rem);
                continue;
            }

            pq.TryPeek(out int bestBinId, out int bestPriority);
            int bestRem = -bestPriority;

            if (bestRem >= len)
            {
                pq.Dequeue();
                bins[bestBinId].Add(idx);

                int newRem = bestRem - len;
                pq.Enqueue(bestBinId, -newRem);
            }
            else
            {
                int binId = bins.Count;
                var list = new List<int>(4) { idx };
                bins.Add(list);

                int rem = stockLength - len;
                pq.Enqueue(binId, -rem);
            }
        }

        var sb = new StringBuilder(bins.Count * 16 + n * 4 + 32);
        sb.Append(bins.Count).Append('\n');

        for (int i = 0; i < bins.Count; i++)
        {
            List<int> list = bins[i];
            if (list.Count > 0)
            {
                sb.Append(list[0]);
                for (int j = 1; j < list.Count; j++)
                {
                    sb.Append(' ').Append(list[j]);
                }
            }
            sb.Append('\n');
        }

        Console.Write(sb.ToString());
    }
}
