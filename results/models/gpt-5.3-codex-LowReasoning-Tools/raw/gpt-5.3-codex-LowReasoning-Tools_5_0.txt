use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, PartialEq, Eq)]
struct Pt {
    x: i32,
    y: i32,
}

struct Input {
    w: usize,
    h: usize,
    require_cycle: bool,
    blocked: Vec<bool>,
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn inside(x: i32, y: i32, w: usize, h: usize) -> bool {
    x >= 0 && y >= 0 && (x as usize) < w && (y as usize) < h
}

fn parse_input() -> Option<Input> {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).ok()?;
    let mut it = s.split_whitespace();

    let w: usize = it.next()?.parse().ok()?;
    let h: usize = it.next()?.parse().ok()?;
    let obstacle_count: usize = it.next()?.parse().ok()?;
    let require_cycle: i32 = it.next()?.parse().ok()?;
    let mut blocked = vec![false; w * h];

    for _ in 0..obstacle_count {
        let x: usize = it.next()?.parse().ok()?;
        let y: usize = it.next()?.parse().ok()?;
        if x < w && y < h {
            blocked[idx(x, y, w)] = true;
        }
    }

    Some(Input {
        w,
        h,
        require_cycle: require_cycle != 0,
        blocked,
    })
}

fn neighbors_of(v: usize, w: usize, h: usize) -> [Option<usize>; 4] {
    let x = (v % w) as i32;
    let y = (v / w) as i32;
    let dirs = [(1, 0), (-1, 0), (0, 1), (0, -1)];
    let mut out = [None, None, None, None];
    for (i, (dx, dy)) in dirs.iter().enumerate() {
        let nx = x + dx;
        let ny = y + dy;
        if inside(nx, ny, w, h) {
            out[i] = Some(idx(nx as usize, ny as usize, w));
        }
    }
    out
}

fn bipartite_color(v: usize, w: usize) -> usize {
    let x = v % w;
    let y = v / w;
    (x + y) & 1
}

fn connectivity_count(start: usize, available: &[bool], blocked: &[bool], w: usize, h: usize) -> usize {
    if !available[start] || blocked[start] {
        return 0;
    }
    let mut q = VecDeque::new();
    let mut vis = vec![false; available.len()];
    vis[start] = true;
    q.push_back(start);
    let mut cnt = 0;

    while let Some(v) = q.pop_front() {
        cnt += 1;
        for n in neighbors_of(v, w, h).iter().flatten() {
            if !vis[*n] && available[*n] && !blocked[*n] {
                vis[*n] = true;
                q.push_back(*n);
            }
        }
    }
    cnt
}

fn all_open_connected(start: usize, blocked: &[bool], w: usize, h: usize) -> bool {
    if blocked[start] {
        return false;
    }
    let available: Vec<bool> = blocked.iter().map(|&b| !b).collect();
    let open = available.iter().filter(|&&x| x).count();
    connectivity_count(start, &available, blocked, w, h) == open
}

fn build_no_obstacle_serpentine(w: usize, h: usize) -> Option<Vec<usize>> {
    if w == 0 || h == 0 {
        return None;
    }
    if w == 1 && h == 1 {
        return Some(vec![0]);
    }

    // Start must be (0,0). Simple serpentine works when h is even.
    if h % 2 == 0 {
        let mut path = Vec::with_capacity(w * h);
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    path.push(idx(x, y, w));
                }
            } else {
                for x in (0..w).rev() {
                    path.push(idx(x, y, w));
                }
            }
        }
        return Some(path);
    }

    // If h is odd and w even, snake by columns.
    if w % 2 == 0 {
        let mut path = Vec::with_capacity(w * h);
        for x in 0..w {
            if x % 2 == 0 {
                for y in 0..h {
                    path.push(idx(x, y, w));
                }
            } else {
                for y in (0..h).rev() {
                    path.push(idx(x, y, w));
                }
            }
        }
        if path.first().copied() == Some(0) {
            return Some(path);
        }
    }

    // For odd x odd full grid >1, no Hamiltonian path from corner covering all cells.
    None
}

struct Solver {
    w: usize,
    h: usize,
    blocked: Vec<bool>,
    require_cycle: bool,
    n_open: usize,
    start: usize,

    used: Vec<bool>,
    path: Vec<usize>,
    remaining_deg: Vec<i16>,

    deadline: Instant,
}

impl Solver {
    fn new(input: Input, time_limit: Duration) -> Option<Self> {
        if input.w == 0 || input.h == 0 {
            return None;
        }
        let start = idx(0, 0, input.w);
        if input.blocked[start] {
            return None;
        }
        let n_open = input.blocked.iter().filter(|&&b| !b).count();
        if n_open == 0 {
            return None;
        }

        let mut remaining_deg = vec![0i16; input.w * input.h];
        for v in 0..input.w * input.h {
            if input.blocked[v] {
                continue;
            }
            let mut d = 0i16;
            for n in neighbors_of(v, input.w, input.h).iter().flatten() {
                if !input.blocked[*n] {
                    d += 1;
                }
            }
            remaining_deg[v] = d;
        }

        Some(Self {
            w: input.w,
            h: input.h,
            blocked: input.blocked,
            require_cycle: input.require_cycle,
            n_open,
            start,
            used: vec![false; input.w * input.h],
            path: Vec::with_capacity(n_open),
            remaining_deg,
            deadline: Instant::now() + time_limit,
        })
    }

    fn is_adjacent(&self, a: usize, b: usize) -> bool {
        let ax = a % self.w;
        let ay = a / self.w;
        let bx = b % self.w;
        let by = b / self.w;
        (ax as i32 - bx as i32).abs() + (ay as i32 - by as i32).abs() == 1
    }

    fn quick_feasible_checks(&self) -> bool {
        if !all_open_connected(self.start, &self.blocked, self.w, self.h) {
            return false;
        }

        // Bipartite parity constraints.
        let mut c = [0usize, 0usize];
        for v in 0..self.w * self.h {
            if !self.blocked[v] {
                c[bipartite_color(v, self.w)] += 1;
            }
        }
        let total = c[0] + c[1];
        let diff = c[0].abs_diff(c[1]);
        if diff > 1 {
            return false;
        }

        if self.require_cycle {
            if total > 1 && (total % 2 == 1) {
                return false;
            }
            if total > 1 {
                let s_col = bipartite_color(self.start, self.w);
                if c[s_col] != c[1 - s_col] {
                    return false;
                }
            }
        }

        true
    }

    fn place(&mut self, v: usize) {
        self.used[v] = true;
        self.path.push(v);
        for n in neighbors_of(v, self.w, self.h).iter().flatten() {
            if !self.blocked[*n] && !self.used[*n] {
                self.remaining_deg[*n] -= 1;
            }
        }
    }

    fn unplace(&mut self, v: usize) {
        for n in neighbors_of(v, self.w, self.h).iter().flatten() {
            if !self.blocked[*n] && !self.used[*n] {
                self.remaining_deg[*n] += 1;
            }
        }
        self.path.pop();
        self.used[v] = false;
    }

    fn remaining_connected_from(&self, from: usize) -> bool {
        let mut available = vec![false; self.w * self.h];
        let mut rem = 0usize;
        for v in 0..self.w * self.h {
            if !self.blocked[v] && !self.used[v] {
                available[v] = true;
                rem += 1;
            }
        }
        if rem == 0 {
            return true;
        }
        if !available[from] {
            return false;
        }
        connectivity_count(from, &available, &self.blocked, self.w, self.h) == rem
    }

    fn forced_dead_end_violation(&self, current: usize) -> bool {
        let mut degree1_count = 0usize;
        for v in 0..self.w * self.h {
            if self.blocked[v] || self.used[v] {
                continue;
            }
            let d = self.remaining_deg[v];
            if d <= 0 {
                return true;
            }
            if d == 1 {
                degree1_count += 1;
                if degree1_count > 2 {
                    return true;
                }
            }
        }

        // If we require cycle, no leaf can remain in unvisited graph except maybe when nothing remains.
        if self.require_cycle {
            let rem = self.n_open - self.path.len();
            if rem > 0 && degree1_count > 0 {
                return true;
            }
        }

        // Current should be able to move unless done.
        if self.path.len() < self.n_open {
            let mut options = 0;
            for n in neighbors_of(current, self.w, self.h).iter().flatten() {
                if !self.blocked[*n] && !self.used[*n] {
                    options += 1;
                }
            }
            if options == 0 {
                return true;
            }
        }

        false
    }

    fn choose_candidates(&self, current: usize) -> Vec<usize> {
        let mut cand = Vec::new();
        for n in neighbors_of(current, self.w, self.h).iter().flatten() {
            if !self.blocked[*n] && !self.used[*n] {
                cand.push(*n);
            }
        }

        // Warnsdorff: smallest onward degree first.
        cand.sort_unstable_by_key(|&v| {
            let mut onward = 0i16;
            for n2 in neighbors_of(v, self.w, self.h).iter().flatten() {
                if !self.blocked[*n2] && !self.used[*n2] && *n2 != current {
                    onward += 1;
                }
            }
            let dist_start = {
                let x = (v % self.w) as i32;
                let y = (v / self.w) as i32;
                x.abs() + y.abs()
            };
            (onward, dist_start)
        });

        cand
    }

    fn dfs(&mut self, current: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }

        if self.path.len() == self.n_open {
            return !self.require_cycle || self.is_adjacent(current, self.start);
        }

        if self.forced_dead_end_violation(current) {
            return false;
        }

        let candidates = self.choose_candidates(current);
        for nxt in candidates {
            self.place(nxt);

            let rem = self.n_open - self.path.len();
            let ok = if rem <= 1 {
                true
            } else {
                self.remaining_connected_from(nxt)
            };

            if ok && self.dfs(nxt) {
                return true;
            }

            self.unplace(nxt);
        }

        false
    }

    fn solve(mut self) -> Option<Vec<usize>> {
        if !self.quick_feasible_checks() {
            return None;
        }

        self.place(self.start);
        if self.n_open == 1 {
            if self.require_cycle {
                return None;
            }
            return Some(self.path);
        }

        if self.dfs(self.start) {
            Some(self.path)
        } else {
            None
        }
    }
}

fn print_solution(path: &[usize], w: usize) {
    println!("{}", path.len());
    for &v in path {
        let x = v % w;
        let y = v / w;
        println!("{} {}", x, y);
    }
}

fn main() {
    let input = match parse_input() {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let start = idx(0, 0, input.w);

    // Fast path: obstacle-free grid construction.
    let obstacle_count = input.blocked.iter().filter(|&&b| b).count();
    if obstacle_count == 0 {
        if let Some(path) = build_no_obstacle_serpentine(input.w, input.h) {
            if input.require_cycle {
                if path.len() <= 1 {
                    println!("0");
                    return;
                }
                let last = *path.last().unwrap();
                let ax = (last % input.w) as i32;
                let ay = (last / input.w) as i32;
                if ax.abs() + ay.abs() != 1 {
                    println!("0");
                    return;
                }
            }
            print_solution(&path, input.w);
            return;
        }
    }

    if input.blocked[start] {
        println!("0");
        return;
    }

    // Time budget tuned for benchmark constraints.
    let total_cells = input.w.saturating_mul(input.h);
    let time_limit = if total_cells <= 64 {
        Duration::from_secs(20)
    } else if total_cells <= 400 {
        Duration::from_secs(90)
    } else {
        Duration::from_secs(280)
    };

    let solver = match Solver::new(input, time_limit) {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };

    match solver.solve() {
        Some(path) => print_solution(&path, solver.w),
        None => println!("0"),
    }
}
