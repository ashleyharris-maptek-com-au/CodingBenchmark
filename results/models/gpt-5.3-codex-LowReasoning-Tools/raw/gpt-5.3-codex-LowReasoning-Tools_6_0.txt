using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

public static class Program
{
    private struct OrbitState
    {
        public double X, Y, Z;
        public double Vx, Vy, Vz;

        public OrbitState(double x, double y, double z, double vx, double vy, double vz)
        {
            X = x; Y = y; Z = z; Vx = vx; Vy = vy; Vz = vz;
        }
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;

        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first))
        {
            return;
        }

        var t0 = Split(first);
        int n = int.Parse(t0[0], ci);
        double mu = t0.Length > 1 ? double.Parse(t0[1], ci) : 398600.4418;

        OrbitState start = ReadState(ci);
        var stations = new OrbitState[n];
        for (int i = 0; i < n; i++)
            stations[i] = ReadState(ci);

        if (n <= 1)
        {
            Console.WriteLine(n == 1 ? "0" : "");
            return;
        }

        var sw = Stopwatch.StartNew();
        long budgetMs = 270000; // 4.5 minutes safety budget

        // Cost matrix over station indices only, plus start->station costs.
        var startCost = new double[n];
        var cost = new double[n, n];

        for (int i = 0; i < n; i++)
        {
            startCost[i] = TransferCost(start, stations[i], mu);
        }

        for (int i = 0; i < n; i++)
        {
            cost[i, i] = 0.0;
            for (int j = i + 1; j < n; j++)
            {
                double cij = TransferCost(stations[i], stations[j], mu);
                double cji = TransferCost(stations[j], stations[i], mu);
                cost[i, j] = cij;
                cost[j, i] = cji;
            }
        }

        int[] route;

        if (n <= 11)
        {
            route = SolveExactPathDp(n, startCost, cost);
        }
        else
        {
            route = BuildInitialRoute(n, startCost, cost);

            // Medium sizes: more local optimization.
            if (n <= 20)
            {
                MultiStartImprove(route, n, startCost, cost, sw, budgetMs, attempts: 24, aggressive: true);
            }
            else if (n <= 60)
            {
                MultiStartImprove(route, n, startCost, cost, sw, budgetMs, attempts: 14, aggressive: true);
            }
            else
            {
                MultiStartImprove(route, n, startCost, cost, sw, budgetMs, attempts: 8, aggressive: false);
            }
        }

        Console.WriteLine(string.Join(" ", route));
    }

    private static OrbitState ReadState(CultureInfo ci)
    {
        string? line = Console.ReadLine();
        while (line != null && line.Trim().Length == 0)
            line = Console.ReadLine();
        if (line == null) return new OrbitState();

        var t = Split(line);
        return new OrbitState(
            double.Parse(t[0], ci), double.Parse(t[1], ci), double.Parse(t[2], ci),
            double.Parse(t[3], ci), double.Parse(t[4], ci), double.Parse(t[5], ci)
        );
    }

    private static string[] Split(string s) =>
        s.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries);

    private static double TransferCost(OrbitState a, OrbitState b, double mu)
    {
        double rx = b.X - a.X, ry = b.Y - a.Y, rz = b.Z - a.Z;
        double vx = b.Vx - a.Vx, vy = b.Vy - a.Vy, vz = b.Vz - a.Vz;

        double dr = Norm(rx, ry, rz);                // km
        double dv = Norm(vx, vy, vz);                // km/s

        double ra = Norm(a.X, a.Y, a.Z);
        double rb = Norm(b.X, b.Y, b.Z);
        double rAvg = 0.5 * (ra + rb);

        // Angular momentum vectors -> plane change proxy.
        Cross(a.X, a.Y, a.Z, a.Vx, a.Vy, a.Vz, out double hax, out double hay, out double haz);
        Cross(b.X, b.Y, b.Z, b.Vx, b.Vy, b.Vz, out double hbx, out double hby, out double hbz);

        double ha = Norm(hax, hay, haz);
        double hb = Norm(hbx, hby, hbz);
        double cosAng = 1.0;
        if (ha > 1e-12 && hb > 1e-12)
        {
            cosAng = Clamp((hax * hbx + hay * hby + haz * hbz) / (ha * hb), -1.0, 1.0);
        }
        double angle = Math.Acos(cosAng);

        double vCirc = Math.Sqrt(mu / Math.Max(1.0, rAvg));
        double planeDv = 2.0 * vCirc * Math.Sin(0.5 * angle);

        // Specific orbital energy mismatch proxy.
        double ea = 0.5 * (a.Vx * a.Vx + a.Vy * a.Vy + a.Vz * a.Vz) - mu / Math.Max(1.0, ra);
        double eb = 0.5 * (b.Vx * b.Vx + b.Vy * b.Vy + b.Vz * b.Vz) - mu / Math.Max(1.0, rb);
        double energyTerm = Math.Abs(ea - eb) / 10.0;

        // Radial difference proxy (semi-major / altitude mismatch impact)
        double radialTerm = Math.Abs(ra - rb) / 1000.0;

        // Combined heuristic "effective delta-V" score.
        return 0.52 * dv + 0.20 * planeDv + 0.18 * (dr / 500.0) + 0.07 * radialTerm + 0.03 * energyTerm;
    }

    private static int[] SolveExactPathDp(int n, double[] startCost, double[,] cost)
    {
        int m = 1 << n;
        double[,] dp = new double[m, n];
        int[,] parent = new int[m, n];

        const double INF = 1e100;
        for (int mask = 0; mask < m; mask++)
        {
            for (int i = 0; i < n; i++)
            {
                dp[mask, i] = INF;
                parent[mask, i] = -1;
            }
        }

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            dp[mask, i] = startCost[i];
        }

        for (int mask = 1; mask < m; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (cur >= INF * 0.5) continue;

                int rem = (~mask) & (m - 1);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nx = BitIndex(bit);
                    int nm = mask | bit;
                    double nv = cur + cost[last, nx];
                    if (nv < dp[nm, nx])
                    {
                        dp[nm, nx] = nv;
                        parent[nm, nx] = last;
                    }
                    rem ^= bit;
                }
            }
        }

        int full = m - 1;
        double best = INF;
        int end = 0;
        for (int i = 0; i < n; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                end = i;
            }
        }

        int[] route = new int[n];
        int cm = full;
        int curNode = end;
        for (int p = n - 1; p >= 0; p--)
        {
            route[p] = curNode;
            int pr = parent[cm, curNode];
            cm ^= (1 << curNode);
            curNode = pr;
        }

        return route;
    }

    private static int[] BuildInitialRoute(int n, double[] startCost, double[,] cost)
    {
        var used = new bool[n];
        var route = new int[n];

        int first = 0;
        double best = startCost[0];
        for (int i = 1; i < n; i++)
        {
            if (startCost[i] < best)
            {
                best = startCost[i];
                first = i;
            }
        }

        route[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int prev = route[pos - 1];
            int cand = -1;
            double candVal = double.PositiveInfinity;

            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double v = cost[prev, j] + 0.20 * startCost[j];
                if (v < candVal)
                {
                    candVal = v;
                    cand = j;
                }
            }

            route[pos] = cand;
            used[cand] = true;
        }

        return route;
    }

    private static void MultiStartImprove(int[] bestRoute, int n, double[] startCost, double[,] cost, Stopwatch sw, long budgetMs, int attempts, bool aggressive)
    {
        var rng = new Random(1234567);
        double bestScore = RouteScore(bestRoute, startCost, cost);

        for (int a = 0; a < attempts; a++)
        {
            if (sw.ElapsedMilliseconds > budgetMs) break;

            int[] cand = (int[])bestRoute.Clone();

            // Randomized shake.
            int shakes = Math.Max(2, n / 12);
            for (int s = 0; s < shakes; s++)
            {
                int i = rng.Next(n);
                int j = rng.Next(n);
                (cand[i], cand[j]) = (cand[j], cand[i]);
            }

            ImproveRoute(cand, startCost, cost, sw, budgetMs, aggressive);

            double sc = RouteScore(cand, startCost, cost);
            if (sc < bestScore)
            {
                bestScore = sc;
                Array.Copy(cand, bestRoute, n);
            }
        }

        // Final polish on best.
        ImproveRoute(bestRoute, startCost, cost, sw, budgetMs, aggressive);
    }

    private static void ImproveRoute(int[] route, double[] startCost, double[,] cost, Stopwatch sw, long budgetMs, bool aggressive)
    {
        int n = route.Length;
        bool improved = true;
        int outer = 0;

        while (improved && sw.ElapsedMilliseconds <= budgetMs)
        {
            improved = false;
            outer++;

            // 2-opt for path (non-cyclic), with start fixed outside route.
            for (int i = 0; i < n - 1; i++)
            {
                if (sw.ElapsedMilliseconds > budgetMs) return;

                int aPrev = (i == 0) ? -1 : route[i - 1];
                int a = route[i];

                for (int k = i + 1; k < n; k++)
                {
                    int b = route[k];
                    int bNext = (k == n - 1) ? -1 : route[k + 1];

                    double before = EdgeCost(aPrev, a, startCost, cost) + EdgeCost(b, bNext, startCost, cost);
                    double after = EdgeCost(aPrev, b, startCost, cost) + EdgeCost(a, bNext, startCost, cost);

                    if (after + 1e-12 < before)
                    {
                        Reverse(route, i, k);
                        improved = true;
                    }
                }
            }

            if (!aggressive && outer >= 2) break;

            // 1-node relocate (Or-opt size 1).
            for (int i = 0; i < n; i++)
            {
                if (sw.ElapsedMilliseconds > budgetMs) return;

                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    if (Math.Abs(i - j) <= 1) continue;

                    if (TryRelocate(route, i, j, startCost, cost))
                        improved = true;
                }
            }

            if (!aggressive && outer >= 3) break;
            if (aggressive && outer >= 8) break;
        }
    }

    private static bool TryRelocate(int[] route, int from, int to, double[] startCost, double[,] cost)
    {
        int n = route.Length;
        int x = route[from];

        int prevFrom = (from == 0) ? -1 : route[from - 1];
        int nextFrom = (from == n - 1) ? -1 : route[from + 1];

        double removeDelta = -EdgeCost(prevFrom, x, startCost, cost) - EdgeCost(x, nextFrom, startCost, cost)
                             + EdgeCost(prevFrom, nextFrom, startCost, cost);

        int insertPos = to;
        if (to > from) insertPos = to - 1;

        int left = (insertPos < 0) ? -1 : route[insertPos];
        int right = (insertPos + 1 >= n) ? -1 : route[insertPos + 1];

        if (left == x || right == x) return false;

        double addDelta = -EdgeCost(left, right, startCost, cost)
                          + EdgeCost(left, x, startCost, cost)
                          + EdgeCost(x, right, startCost, cost);

        if (removeDelta + addDelta < -1e-12)
        {
            // Perform relocate.
            if (from < insertPos)
            {
                int temp = x;
                for (int i = from; i < insertPos; i++) route[i] = route[i + 1];
                route[insertPos] = temp;
            }
            else
            {
                int temp = x;
                for (int i = from; i > insertPos + 1; i--) route[i] = route[i - 1];
                route[insertPos + 1] = temp;
            }
            return true;
        }

        return false;
    }

    private static double RouteScore(int[] route, double[] startCost, double[,] cost)
    {
        if (route.Length == 0) return 0.0;
        double s = startCost[route[0]];
        for (int i = 1; i < route.Length; i++)
            s += cost[route[i - 1], route[i]];
        return s;
    }

    private static double EdgeCost(int from, int to, double[] startCost, double[,] cost)
    {
        if (to < 0) return 0.0;
        if (from < 0) return startCost[to];
        return cost[from, to];
    }

    private static void Reverse(int[] arr, int l, int r)
    {
        while (l < r)
        {
            (arr[l], arr[r]) = (arr[r], arr[l]);
            l++;
            r--;
        }
    }

    private static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    private static double Norm(double x, double y, double z) => Math.Sqrt(x * x + y * y + z * z);

    private static void Cross(double ax, double ay, double az, double bx, double by, double bz,
                              out double cx, out double cy, out double cz)
    {
        cx = ay * bz - az * by;
        cy = az * bx - ax * bz;
        cz = ax * by - ay * bx;
    }

    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }
}
