using namespace std;

namespace {

using Clock = chrono::steady_clock;

constexpr int INF = 1'000'000'000;
constexpr int MATE_SCORE = 500'000;

enum Piece : int {
    EMPTY = 0,
    WP = 1,
    WN,
    WB,
    WR,
    WQ,
    WK,
    BP,
    BN,
    BB,
    BR,
    BQ,
    BK
};

inline bool isWhite(Piece p) { return p >= WP && p <= WK; }
inline bool isBlack(Piece p) { return p >= BP && p <= BK; }
inline bool sameColor(Piece a, Piece b) {
    if (a == EMPTY || b == EMPTY) return false;
    return (isWhite(a) && isWhite(b)) || (isBlack(a) && isBlack(b));
}

struct Move {
    int from = -1;
    int to = -1;
    Piece promotion = EMPTY;
    bool isCapture = false;
    bool isEnPassant = false;
    bool isCastleKing = false;
    bool isCastleQueen = false;
};

struct Undo {
    Move m;
    Piece moved = EMPTY;
    Piece captured = EMPTY;
    int oldEnPassant = -1;
    uint8_t oldCastling = 0;
    int oldHalfmove = 0;
    int oldFullmove = 1;
};

struct Position {
    array<Piece, 64> board{};
    bool whiteToMove = true;
    uint8_t castling = 0; // 1 K, 2 Q, 4 k, 8 q
    int enPassant = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;

    [[nodiscard]] int kingSquare(bool white) const {
        Piece k = white ? WK : BK;
        for (int i = 0; i < 64; ++i) if (board[i] == k) return i;
        return -1;
    }
};

inline int fileOf(int sq) { return sq & 7; }
inline int rankOf(int sq) { return sq >> 3; }
inline bool onBoard(int sq) { return sq >= 0 && sq < 64; }

string squareToStr(int sq) {
    string s;
    s.push_back(static_cast<char>('a' + fileOf(sq)));
    s.push_back(static_cast<char>('1' + rankOf(sq)));
    return s;
}

int strToSquare(const string& s) {
    if (s.size() != 2) return -1;
    char f = s[0], r = s[1];
    if (f < 'a' || f > 'h' || r < '1' || r > '8') return -1;
    return (r - '1') * 8 + (f - 'a');
}

Piece charToPiece(char c) {
    switch (c) {
        case 'P': return WP; case 'N': return WN; case 'B': return WB;
        case 'R': return WR; case 'Q': return WQ; case 'K': return WK;
        case 'p': return BP; case 'n': return BN; case 'b': return BB;
        case 'r': return BR; case 'q': return BQ; case 'k': return BK;
        default: return EMPTY;
    }
}

char pieceToLetter(Piece p) {
    switch (p) {
        case WP: case BP: return 'p';
        case WN: case BN: return 'N';
        case WB: case BB: return 'B';
        case WR: case BR: return 'R';
        case WQ: case BQ: return 'Q';
        case WK: case BK: return 'K';
        default: return '?';
    }
}

bool parseFEN(const string& fen, Position& pos) {
    pos = Position{};
    istringstream iss(fen);
    string boardPart, activePart, castlingPart, epPart;
    if (!(iss >> boardPart >> activePart >> castlingPart >> epPart >> pos.halfmoveClock >> pos.fullmoveNumber)) return false;

    int sq = 56; // a8
    for (char c : boardPart) {
        if (c == '/') {
            sq -= 16;
            continue;
        }
        if (isdigit(static_cast<unsigned char>(c))) {
            sq += c - '0';
            continue;
        }
        Piece p = charToPiece(c);
        if (p == EMPTY) return false;
        if (!onBoard(sq)) return false;
        pos.board[sq++] = p;
    }

    pos.whiteToMove = (activePart == "w");

    pos.castling = 0;
    if (castlingPart != "-") {
        for (char c : castlingPart) {
            if (c == 'K') pos.castling |= 1;
            else if (c == 'Q') pos.castling |= 2;
            else if (c == 'k') pos.castling |= 4;
            else if (c == 'q') pos.castling |= 8;
        }
    }

    pos.enPassant = (epPart == "-") ? -1 : strToSquare(epPart);
    if (epPart != "-" && pos.enPassant == -1) return false;

    return true;
}

bool isSquareAttacked(const Position& pos, int sq, bool byWhite) {
    const auto& b = pos.board;
    int f = fileOf(sq), r = rankOf(sq);

    // Pawns
    if (byWhite) {
        if (r > 0) {
            if (f > 0 && b[sq - 9] == WP) return true;
            if (f < 7 && b[sq - 7] == WP) return true;
        }
    } else {
        if (r < 7) {
            if (f > 0 && b[sq + 7] == BP) return true;
            if (f < 7 && b[sq + 9] == BP) return true;
        }
    }

    // Knights
    static constexpr int kOff[8] = {15, 17, 10, 6, -15, -17, -10, -6};
    for (int d : kOff) {
        int t = sq + d;
        if (!onBoard(t)) continue;
        int df = abs(fileOf(t) - f), dr = abs(rankOf(t) - r);
        if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
        Piece p = b[t];
        if (byWhite && p == WN) return true;
        if (!byWhite && p == BN) return true;
    }

    // Bishops / Queens diagonals
    static constexpr int bDir[4] = {9, 7, -9, -7};
    for (int d : bDir) {
        int t = sq + d;
        while (onBoard(t) && abs(fileOf(t) - fileOf(t - d)) == 1) {
            Piece p = b[t];
            if (p != EMPTY) {
                if (byWhite && (p == WB || p == WQ)) return true;
                if (!byWhite && (p == BB || p == BQ)) return true;
                break;
            }
            t += d;
        }
    }

    // Rooks / Queens orthogonals
    static constexpr int rDir[4] = {8, -8, 1, -1};
    for (int d : rDir) {
        int t = sq + d;
        while (onBoard(t) && (d == 8 || d == -8 || fileOf(t) - fileOf(t - d) == d)) {
            Piece p = b[t];
            if (p != EMPTY) {
                if (byWhite && (p == WR || p == WQ)) return true;
                if (!byWhite && (p == BR || p == BQ)) return true;
                break;
            }
            t += d;
        }
    }

    // King
    for (int dr = -1; dr <= 1; ++dr) {
        for (int df = -1; df <= 1; ++df) {
            if (dr == 0 && df == 0) continue;
            int nf = f + df, nr = r + dr;
            if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
            Piece p = b[nr * 8 + nf];
            if (byWhite && p == WK) return true;
            if (!byWhite && p == BK) return true;
        }
    }

    return false;
}

void addMove(vector<Move>& moves, int from, int to, Piece promo = EMPTY, bool cap = false, bool ep = false) {
    Move m;
    m.from = from;
    m.to = to;
    m.promotion = promo;
    m.isCapture = cap;
    m.isEnPassant = ep;
    moves.push_back(m);
}

void generatePseudo(const Position& pos, vector<Move>& moves) {
    moves.clear();
    bool white = pos.whiteToMove;

    for (int sq = 0; sq < 64; ++sq) {
        Piece p = pos.board[sq];
        if (p == EMPTY) continue;
        if (white != isWhite(p)) continue;

        int f = fileOf(sq), r = rankOf(sq);

        if (p == WP || p == BP) {
            int dir = (p == WP) ? 8 : -8;
            int startRank = (p == WP) ? 1 : 6;
            int promoteRank = (p == WP) ? 6 : 1;
            int next = sq + dir;
            if (onBoard(next) && pos.board[next] == EMPTY) {
                if (r == promoteRank) {
                    addMove(moves, sq, next, (p == WP) ? WQ : BQ);
                    addMove(moves, sq, next, (p == WP) ? WR : BR);
                    addMove(moves, sq, next, (p == WP) ? WB : BB);
                    addMove(moves, sq, next, (p == WP) ? WN : BN);
                } else {
                    addMove(moves, sq, next);
                    if (r == startRank) {
                        int nn = sq + 2 * dir;
                        if (pos.board[nn] == EMPTY) addMove(moves, sq, nn);
                    }
                }
            }

            for (int df : {-1, 1}) {
                int nf = f + df;
                if (nf < 0 || nf > 7) continue;
                int to = next + df;
                if (!onBoard(to)) continue;
                Piece target = pos.board[to];
                if (target != EMPTY && !sameColor(p, target)) {
                    if (r == promoteRank) {
                        addMove(moves, sq, to, (p == WP) ? WQ : BQ, true);
                        addMove(moves, sq, to, (p == WP) ? WR : BR, true);
                        addMove(moves, sq, to, (p == WP) ? WB : BB, true);
                        addMove(moves, sq, to, (p == WP) ? WN : BN, true);
                    } else {
                        addMove(moves, sq, to, EMPTY, true);
                    }
                }
                if (to == pos.enPassant) {
                    addMove(moves, sq, to, EMPTY, true, true);
                }
            }
            continue;
        }

        if (p == WN || p == BN) {
            static constexpr int nOff[8] = {15, 17, 10, 6, -15, -17, -10, -6};
            for (int d : nOff) {
                int to = sq + d;
                if (!onBoard(to)) continue;
                int df = abs(fileOf(to) - f), dr = abs(rankOf(to) - r);
                if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
                Piece t = pos.board[to];
                if (t == EMPTY) addMove(moves, sq, to);
                else if (!sameColor(p, t)) addMove(moves, sq, to, EMPTY, true);
            }
            continue;
        }

        auto slide = [&](const int* dirs, int cnt) {
            for (int i = 0; i < cnt; ++i) {
                int d = dirs[i];
                int to = sq + d;
                while (onBoard(to) && (d == 8 || d == -8 || abs(fileOf(to) - fileOf(to - d)) == 1)) {
                    Piece t = pos.board[to];
                    if (t == EMPTY) addMove(moves, sq, to);
                    else {
                        if (!sameColor(p, t)) addMove(moves, sq, to, EMPTY, true);
                        break;
                    }
                    to += d;
                }
            }
        };

        if (p == WB || p == BB) {
            static constexpr int dirs[4] = {9, 7, -9, -7};
            slide(dirs, 4);
            continue;
        }
        if (p == WR || p == BR) {
            static constexpr int dirs[4] = {8, -8, 1, -1};
            slide(dirs, 4);
            continue;
        }
        if (p == WQ || p == BQ) {
            static constexpr int dirs[8] = {9, 7, -9, -7, 8, -8, 1, -1};
            slide(dirs, 8);
            continue;
        }

        // King
        for (int dr = -1; dr <= 1; ++dr) {
            for (int df = -1; df <= 1; ++df) {
                if (dr == 0 && df == 0) continue;
                int nf = f + df, nr = r + dr;
                if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
                int to = nr * 8 + nf;
                Piece t = pos.board[to];
                if (t == EMPTY) addMove(moves, sq, to);
                else if (!sameColor(p, t)) addMove(moves, sq, to, EMPTY, true);
            }
        }

        if (white) {
            if ((pos.castling & 1) && pos.board[5] == EMPTY && pos.board[6] == EMPTY) {
                Move m; m.from = 4; m.to = 6; m.isCastleKing = true; moves.push_back(m);
            }
            if ((pos.castling & 2) && pos.board[3] == EMPTY && pos.board[2] == EMPTY && pos.board[1] == EMPTY) {
                Move m; m.from = 4; m.to = 2; m.isCastleQueen = true; moves.push_back(m);
            }
        } else {
            if ((pos.castling & 4) && pos.board[61] == EMPTY && pos.board[62] == EMPTY) {
                Move m; m.from = 60; m.to = 62; m.isCastleKing = true; moves.push_back(m);
            }
            if ((pos.castling & 8) && pos.board[59] == EMPTY && pos.board[58] == EMPTY && pos.board[57] == EMPTY) {
                Move m; m.from = 60; m.to = 58; m.isCastleQueen = true; moves.push_back(m);
            }
        }
    }
}

bool makeMove(Position& pos, const Move& m, Undo& u) {
    u.m = m;
    u.oldEnPassant = pos.enPassant;
    u.oldCastling = pos.castling;
    u.oldHalfmove = pos.halfmoveClock;
    u.oldFullmove = pos.fullmoveNumber;

    Piece moving = pos.board[m.from];
    u.moved = moving;

    Piece captured = EMPTY;
    if (m.isEnPassant) {
        int capSq = m.to + (pos.whiteToMove ? -8 : 8);
        captured = pos.board[capSq];
        pos.board[capSq] = EMPTY;
    } else {
        captured = pos.board[m.to];
    }
    u.captured = captured;

    pos.board[m.from] = EMPTY;

    if (m.isCastleKing) {
        if (pos.whiteToMove) {
            pos.board[6] = WK; pos.board[5] = WR; pos.board[7] = EMPTY;
        } else {
            pos.board[62] = BK; pos.board[61] = BR; pos.board[63] = EMPTY;
        }
    } else if (m.isCastleQueen) {
        if (pos.whiteToMove) {
            pos.board[2] = WK; pos.board[3] = WR; pos.board[0] = EMPTY;
        } else {
            pos.board[58] = BK; pos.board[59] = BR; pos.board[56] = EMPTY;
        }
    } else {
        pos.board[m.to] = (m.promotion != EMPTY) ? m.promotion : moving;
    }

    // Update castling rights
    if (moving == WK) pos.castling &= ~uint8_t(1 | 2);
    if (moving == BK) pos.castling &= ~uint8_t(4 | 8);
    if (m.from == 0 || m.to == 0) pos.castling &= ~uint8_t(2);
    if (m.from == 7 || m.to == 7) pos.castling &= ~uint8_t(1);
    if (m.from == 56 || m.to == 56) pos.castling &= ~uint8_t(8);
    if (m.from == 63 || m.to == 63) pos.castling &= ~uint8_t(4);

    pos.enPassant = -1;
    if (moving == WP && m.to - m.from == 16) pos.enPassant = m.from + 8;
    if (moving == BP && m.from - m.to == 16) pos.enPassant = m.from - 8;

    if (moving == WP || moving == BP || captured != EMPTY) pos.halfmoveClock = 0;
    else pos.halfmoveClock++;

    if (!pos.whiteToMove) pos.fullmoveNumber++;

    pos.whiteToMove = !pos.whiteToMove;

    // legality: own king not in check
    int ks = pos.kingSquare(!pos.whiteToMove);
    if (ks == -1) return false;
    bool illegal = isSquareAttacked(pos, ks, pos.whiteToMove);
    if (illegal) return false;

    return true;
}

void unmakeMove(Position& pos, const Undo& u) {
    pos.whiteToMove = !pos.whiteToMove;
    pos.enPassant = u.oldEnPassant;
    pos.castling = u.oldCastling;
    pos.halfmoveClock = u.oldHalfmove;
    pos.fullmoveNumber = u.oldFullmove;

    const Move& m = u.m;

    if (m.isCastleKing) {
        if (pos.whiteToMove) {
            pos.board[4] = WK; pos.board[7] = WR; pos.board[5] = EMPTY; pos.board[6] = EMPTY;
        } else {
            pos.board[60] = BK; pos.board[63] = BR; pos.board[61] = EMPTY; pos.board[62] = EMPTY;
        }
    } else if (m.isCastleQueen) {
        if (pos.whiteToMove) {
            pos.board[4] = WK; pos.board[0] = WR; pos.board[2] = EMPTY; pos.board[3] = EMPTY;
        } else {
            pos.board[60] = BK; pos.board[56] = BR; pos.board[58] = EMPTY; pos.board[59] = EMPTY;
        }
    } else {
        pos.board[m.from] = u.moved;
        pos.board[m.to] = u.captured;
        if (m.isEnPassant) {
            pos.board[m.to] = EMPTY;
            int capSq = m.to + (pos.whiteToMove ? -8 : 8);
            pos.board[capSq] = u.captured;
        }
    }
}

vector<Move> generateLegal(const Position& pos) {
    vector<Move> pseudo;
    generatePseudo(pos, pseudo);
    vector<Move> legal;
    legal.reserve(pseudo.size());

    Position tmp = pos;
    for (const Move& m : pseudo) {
        if (m.isCastleKing || m.isCastleQueen) {
            int kFrom = pos.whiteToMove ? 4 : 60;
            if (isSquareAttacked(pos, kFrom, !pos.whiteToMove)) continue;
            int mid = m.isCastleKing ? (pos.whiteToMove ? 5 : 61) : (pos.whiteToMove ? 3 : 59);
            int dst = m.to;
            if (isSquareAttacked(pos, mid, !pos.whiteToMove) || isSquareAttacked(pos, dst, !pos.whiteToMove)) continue;
        }
        Undo u;
        tmp = pos;
        if (!makeMove(tmp, m, u)) continue;
        legal.push_back(m);
    }
    return legal;
}

int evaluate(const Position& pos) {
    static constexpr int val[13] = {0, 100, 320, 330, 500, 900, 0, -100, -320, -330, -500, -900, 0};
    int s = 0;
    for (Piece p : pos.board) s += val[p];
    return pos.whiteToMove ? s : -s;
}

struct Searcher {
    Clock::time_point deadline;
    bool timedOut = false;

    int quiesce(Position& pos, int alpha, int beta) {
        if (Clock::now() >= deadline) { timedOut = true; return alpha; }
        int stand = evaluate(pos);
        if (stand >= beta) return beta;
        if (stand > alpha) alpha = stand;

        vector<Move> moves = generateLegal(pos);
        for (const Move& m : moves) {
            if (!m.isCapture && !m.isEnPassant) continue;
            Undo u;
            if (!makeMove(pos, m, u)) continue;
            int score = -quiesce(pos, -beta, -alpha);
            unmakeMove(pos, u);
            if (timedOut) return alpha;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int alphabeta(Position& pos, int depth, int alpha, int beta) {
        if (Clock::now() >= deadline) { timedOut = true; return alpha; }

        vector<Move> moves = generateLegal(pos);
        if (depth == 0) return quiesce(pos, alpha, beta);

        if (moves.empty()) {
            int ksq = pos.kingSquare(pos.whiteToMove);
            bool inCheck = (ksq != -1) && isSquareAttacked(pos, ksq, !pos.whiteToMove);
            if (inCheck) return -MATE_SCORE + (6 - depth);
            return 0;
        }

        sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return int(a.isCapture) + int(a.promotion != EMPTY) > int(b.isCapture) + int(b.promotion != EMPTY);
        });

        for (const Move& m : moves) {
            Undo u;
            if (!makeMove(pos, m, u)) continue;
            int score = -alphabeta(pos, depth - 1, -beta, -alpha);
            unmakeMove(pos, u);
            if (timedOut) return alpha;
            if (score >= beta) return beta;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    optional<Move> findBest(Position pos, double seconds) {
        deadline = Clock::now() + chrono::milliseconds(static_cast<int>(seconds * 1000.0) - 50);
        if (seconds < 0.2) deadline = Clock::now() + chrono::milliseconds(100);

        optional<Move> best;
        int bestScore = -INF;

        vector<Move> legal = generateLegal(pos);
        if (legal.empty()) return nullopt;
        best = legal.front();

        for (int depth = 1; depth <= 64; ++depth) {
            if (Clock::now() >= deadline) break;
            int localBest = -INF;
            optional<Move> localMove;

            sort(legal.begin(), legal.end(), [&](const Move& a, const Move& b) {
                return int(a.isCapture) + int(a.promotion != EMPTY) > int(b.isCapture) + int(b.promotion != EMPTY);
            });

            for (const Move& m : legal) {
                Undo u;
                if (!makeMove(pos, m, u)) continue;
                int score = -alphabeta(pos, depth - 1, -INF, INF);
                unmakeMove(pos, u);
                if (timedOut) break;
                if (score > localBest) {
                    localBest = score;
                    localMove = m;
                }
            }

            if (!timedOut && localMove.has_value()) {
                best = localMove;
                bestScore = localBest;
                (void)bestScore;
            } else {
                timedOut = false;
                break;
            }
        }

        return best;
    }
};

string moveToNotation(const Position& pos, const Move& m) {
    if (m.isCastleKing) return "O-O";
    if (m.isCastleQueen) return "O-O-O";

    string out;
    Piece moving = pos.board[m.from];
    out.push_back(pieceToLetter(moving));
    out += squareToStr(m.from);
    if (m.isCapture || m.isEnPassant) out.push_back('x');
    out += squareToStr(m.to);
    if (m.isEnPassant) out += "ep";
    if (m.promotion != EMPTY) {
        out.push_back('/');
        out.push_back(pieceToLetter(m.promotion));
    }

    Position next = pos;
    Undo u;
    if (makeMove(next, m, u)) {
        vector<Move> reply = generateLegal(next);
        int enemyKing = next.kingSquare(next.whiteToMove);
        bool inCheck = enemyKing != -1 && isSquareAttacked(next, enemyKing, !next.whiteToMove);
        if (inCheck) {
            if (reply.empty()) out.push_back('#');
            else out.push_back('+');
        }
    }
    return out;
}

} // namespace

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSeconds;
    if (!(cin >> timeoutSeconds)) return 0;
    string fen;
    cin.ignore(numeric_limits<streamsize>::max(), '\n');
    getline(cin, fen);

    Position pos;
    if (!parseFEN(fen, pos)) {
        cout << "O-O\n";
        return 0;
    }

    Searcher s;
    optional<Move> best = s.findBest(pos, timeoutSeconds);
    if (!best.has_value()) {
        vector<Move> legal = generateLegal(pos);
        if (!legal.empty()) best = legal.front();
    }

    if (!best.has_value()) {
        cout << "O-O\n";
        return 0;
    }

    cout << moveToNotation(pos, *best) << '\n';
    return 0;
}
