use std::collections::VecDeque;
use std::io::{self, Read};

#[inline]
fn idx(r: usize, c: usize, w: usize) -> usize {
    r * w + c
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut lines = input.lines();

    let first = match lines.next() {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };

    let mut hw = first.split_whitespace();
    let h: usize = match hw.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let w: usize = match hw.next().and_then(|x| x.parse().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let total = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = vec![b'#'; total];
    let mut start = None::<usize>;
    let mut goal = None::<usize>;

    for r in 0..h {
        let line = match lines.next() {
            Some(s) => s,
            None => {
                println!("0");
                return;
            }
        };

        let bytes = line.as_bytes();
        if bytes.len() != w {
            println!("0");
            return;
        }

        for c in 0..w {
            let p = idx(r, c, w);
            let ch = bytes[c];
            grid[p] = ch;
            if ch == b'A' {
                start = Some(p);
            } else if ch == b'B' {
                goal = Some(p);
            }
        }
    }

    let s = match start {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let t = match goal {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    if s == t {
        let x = s / w;
        let y = s % w;
        println!("1");
        println!("{} {}", x, y);
        return;
    }

    // parent_dir encodes how we reached each cell from its parent:
    // 0 = unvisited, 1 = from up, 2 = from down, 3 = from left, 4 = from right
    let mut parent_dir = vec![0u8; total];
    let mut q = VecDeque::new();

    parent_dir[s] = 255; // start sentinel
    q.push_back(s);

    let mut found = false;

    while let Some(p) = q.pop_front() {
        let r = p / w;
        let c = p % w;

        // Up
        if r > 0 {
            let np = p - w;
            if parent_dir[np] == 0 && grid[np] != b'#' {
                parent_dir[np] = 1;
                if np == t {
                    found = true;
                    break;
                }
                q.push_back(np);
            }
        }

        // Down
        if r + 1 < h {
            let np = p + w;
            if parent_dir[np] == 0 && grid[np] != b'#' {
                parent_dir[np] = 2;
                if np == t {
                    found = true;
                    break;
                }
                q.push_back(np);
            }
        }

        // Left
        if c > 0 {
            let np = p - 1;
            if parent_dir[np] == 0 && grid[np] != b'#' {
                parent_dir[np] = 3;
                if np == t {
                    found = true;
                    break;
                }
                q.push_back(np);
            }
        }

        // Right
        if c + 1 < w {
            let np = p + 1;
            if parent_dir[np] == 0 && grid[np] != b'#' {
                parent_dir[np] = 4;
                if np == t {
                    found = true;
                    break;
                }
                q.push_back(np);
            }
        }
    }

    if !found && parent_dir[t] == 0 {
        println!("0");
        return;
    }

    // Reconstruct path from goal back to start.
    let mut rev_path = Vec::<usize>::new();
    let mut cur = t;
    rev_path.push(cur);

    while cur != s {
        match parent_dir[cur] {
            1 => cur += w, // came from up -> parent is below
            2 => cur -= w, // came from down -> parent is above
            3 => cur += 1, // came from left -> parent is right
            4 => cur -= 1, // came from right -> parent is left
            _ => {
                println!("0");
                return;
            }
        }
        rev_path.push(cur);
    }

    rev_path.reverse();

    println!("{}", rev_path.len());
    for p in rev_path {
        println!("{} {}", p / w, p % w);
    }
}
