#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct Rect {
    int w;
    int h;
    int idx;
};

struct Shelf {
    int y;
    int h;
    int used_w;
};

struct Board {
    int used_h = 0;
    long long used_area = 0;
    std::vector<Shelf> shelves;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Candidate {
    bool ok = false;
    double score = 0.0;
    int board = -1;
    int shelf = -1; // -1 means new shelf
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
};

static inline void consider_existing_shelves(
    const Board& b,
    int bidx,
    int W,
    int rw,
    int rh,
    int rotated,
    Candidate& best
) {
    for (int s = 0; s < (int)b.shelves.size(); ++s) {
        const Shelf& sh = b.shelves[s];
        if (rh > sh.h) continue;
        if (sh.used_w + rw > W) continue;

        int x = sh.used_w;
        int y = sh.y;
        int rem_w = W - (sh.used_w + rw);
        int h_gap = sh.h - rh;

        double score = (double)rem_w + 0.05 * (double)h_gap * (double)W;

        if (!best.ok || score < best.score) {
            best.ok = true;
            best.score = score;
            best.board = bidx;
            best.shelf = s;
            best.x = x;
            best.y = y;
            best.w = rw;
            best.h = rh;
            best.rotated = rotated;
        }
    }
}

static inline void consider_new_shelf(
    const Board& b,
    int bidx,
    int W,
    int H,
    int rw,
    int rh,
    int rotated,
    Candidate& best
) {
    if (b.used_h + rh > H) return;

    int rem_w = W - rw;
    int rem_h = H - (b.used_h + rh);

    double score = (double)rem_w + 0.01 * (double)rem_h;

    if (!best.ok || score < best.score) {
        best.ok = true;
        best.score = score;
        best.board = bidx;
        best.shelf = -1;
        best.x = 0;
        best.y = b.used_h;
        best.w = rw;
        best.h = rh;
        best.rotated = rotated;
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N, W, H;
    if (!(std::cin >> N >> W >> H)) return 0;

    std::vector<Rect> rects;
    rects.reserve(N);

    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;

        if ((w > W || h > H) && !(h <= W && w <= H)) {
            // Cannot fit even with rotation; keep as-is and let placement fail explicitly.
        }

        rects.push_back({w, h, i});
    }

    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        long long areaA = 1LL * a.w * a.h;
        long long areaB = 1LL * b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        int ma = std::max(a.w, a.h), mb = std::max(b.w, b.h);
        if (ma != mb) return ma > mb;
        int mi = std::min(a.w, a.h), mj = std::min(b.w, b.h);
        return mi > mj;
    });

    std::vector<Placement> ans(N);
    std::vector<Board> boards;
    boards.reserve(std::max(4, N / 8));

    const bool quality_mode = (N <= 2000);
    const int recent_k = quality_mode ? std::numeric_limits<int>::max() : 32;

    for (const Rect& r : rects) {
        Candidate best;

        int B = (int)boards.size();
        int start = 0;
        if (!quality_mode && B > recent_k) start = B - recent_k;

        // Try candidate boards.
        for (int b = start; b < B; ++b) {
            const Board& bd = boards[b];

            // Orientation 0
            if (r.w <= W && r.h <= H) {
                consider_existing_shelves(bd, b, W, r.w, r.h, 0, best);
                consider_new_shelf(bd, b, W, H, r.w, r.h, 0, best);
            }

            // Orientation 1 (rotated)
            if (r.h <= W && r.w <= H) {
                consider_existing_shelves(bd, b, W, r.h, r.w, 1, best);
                consider_new_shelf(bd, b, W, H, r.h, r.w, 1, best);
            }
        }

        if (!best.ok) {
            // Open new board and place at origin/new shelf.
            Board nb;
            bool placed = false;

            if (r.w <= W && r.h <= H) {
                nb.shelves.push_back({0, r.h, r.w});
                nb.used_h = r.h;
                nb.used_area = 1LL * r.w * r.h;
                int bidx = (int)boards.size();
                boards.push_back(std::move(nb));
                ans[r.idx] = {bidx, 0, 0, 0};
                placed = true;
            } else if (r.h <= W && r.w <= H) {
                nb.shelves.push_back({0, r.w, r.h});
                nb.used_h = r.w;
                nb.used_area = 1LL * r.w * r.h;
                int bidx = (int)boards.size();
                boards.push_back(std::move(nb));
                ans[r.idx] = {bidx, 0, 0, 1};
                placed = true;
            }

            if (!placed) {
                // Fallback invalid marker; keeps output size consistent.
                ans[r.idx] = {-1, 0, 0, 0};
            }
            continue;
        }

        Board& b = boards[best.board];
        if (best.shelf >= 0) {
            Shelf& sh = b.shelves[best.shelf];
            sh.used_w += best.w;
        } else {
            b.shelves.push_back({best.y, best.h, best.w});
            b.used_h += best.h;
        }
        b.used_area += 1LL * best.w * best.h;

        ans[r.idx] = {best.board, best.x, best.y, best.rotated};
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        const auto& p = ans[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
