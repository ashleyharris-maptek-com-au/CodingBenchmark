using namespace std;

static constexpr double EPS = 1e-9;

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return sqrt(norm2(v)); }
static inline Vec3 normalize(const Vec3& v) {
    double n = norm(v);
    if (n < EPS) return {0,0,0};
    return v / n;
}

struct Mat3 {
    double m[3][3]{};
    static Mat3 identity() {
        Mat3 I;
        I.m[0][0]=I.m[1][1]=I.m[2][2]=1.0;
        return I;
    }
};

static Vec3 mul(const Mat3& M, const Vec3& v) {
    return {
        M.m[0][0]*v.x + M.m[0][1]*v.y + M.m[0][2]*v.z,
        M.m[1][0]*v.x + M.m[1][1]*v.y + M.m[1][2]*v.z,
        M.m[2][0]*v.x + M.m[2][1]*v.y + M.m[2][2]*v.z
    };
}

static Mat3 matMul(const Mat3& A, const Mat3& B) {
    Mat3 C{};
    for (int i=0;i<3;i++) for (int j=0;j<3;j++) {
        double s = 0;
        for (int k=0;k<3;k++) s += A.m[i][k]*B.m[k][j];
        C.m[i][j] = s;
    }
    return C;
}

static Mat3 rotX(double a){
    Mat3 R = Mat3::identity();
    double c=cos(a), s=sin(a);
    R.m[1][1]=c; R.m[1][2]=-s; R.m[2][1]=s; R.m[2][2]=c;
    return R;
}
static Mat3 rotY(double a){
    Mat3 R = Mat3::identity();
    double c=cos(a), s=sin(a);
    R.m[0][0]=c; R.m[0][2]=s; R.m[2][0]=-s; R.m[2][2]=c;
    return R;
}
static Mat3 rotZ(double a){
    Mat3 R = Mat3::identity();
    double c=cos(a), s=sin(a);
    R.m[0][0]=c; R.m[0][1]=-s; R.m[1][0]=s; R.m[1][1]=c;
    return R;
}

static Mat3 fromEulerXYZ(double rx, double ry, double rz) {
    return matMul(matMul(rotZ(rz), rotY(ry)), rotX(rx));
}

struct Face {
    int a{}, b{}, c{};
    Vec3 n;
    double d{};
};

struct Hull {
    vector<Vec3> p;
    vector<Face> faces;
    Vec3 centroid;
    Vec3 bmin, bmax;
    double volumeApprox{};
};

static Hull buildConvexHullBruteforce(const vector<Vec3>& pts) {
    Hull H;
    H.p = pts;
    int n = (int)pts.size();
    H.centroid = {0,0,0};
    H.bmin = { numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity() };
    H.bmax = { -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity() };
    for (const auto& v: pts) {
        H.centroid = H.centroid + v;
        H.bmin.x = min(H.bmin.x, v.x); H.bmin.y = min(H.bmin.y, v.y); H.bmin.z = min(H.bmin.z, v.z);
        H.bmax.x = max(H.bmax.x, v.x); H.bmax.y = max(H.bmax.y, v.y); H.bmax.z = max(H.bmax.z, v.z);
    }
    H.centroid = H.centroid / max(1, n);

    vector<array<int,3>> rawFaces;
    rawFaces.reserve(n*n);
    for (int i=0;i<n;i++) for (int j=i+1;j<n;j++) for (int k=j+1;k<n;k++) {
        Vec3 a = pts[i], b = pts[j], c = pts[k];
        Vec3 nn = cross(b-a, c-a);
        double ln = norm(nn);
        if (ln < 1e-12) continue;
        int pos = 0, neg = 0;
        for (int t=0;t<n;t++) if (t!=i && t!=j && t!=k) {
            double s = dot(nn, pts[t]-a);
            if (s > 1e-8) pos++;
            else if (s < -1e-8) neg++;
            if (pos && neg) break;
        }
        if (pos && neg) continue;
        if (pos == 0) rawFaces.push_back({i,j,k});
        else rawFaces.push_back({i,k,j});
    }

    // Deduplicate coplanar triangle duplicates by normalized index triplets.
    sort(rawFaces.begin(), rawFaces.end());
    rawFaces.erase(unique(rawFaces.begin(), rawFaces.end()), rawFaces.end());

    H.faces.reserve(rawFaces.size());
    for (auto tri : rawFaces) {
        Face f;
        f.a = tri[0]; f.b = tri[1]; f.c = tri[2];
        Vec3 a = pts[f.a], b = pts[f.b], c = pts[f.c];
        f.n = normalize(cross(b-a, c-a));
        f.d = dot(f.n, a);
        if (dot(f.n, H.centroid) > f.d) {
            swap(f.b, f.c);
            f.n = f.n * -1.0;
            f.d = dot(f.n, a);
        }
        H.faces.push_back(f);
    }

    // Volume estimate via signed tetrahedra around origin.
    double V = 0;
    for (const auto& f: H.faces) {
        const Vec3 &a=pts[f.a], &b=pts[f.b], &c=pts[f.c];
        V += dot(a, cross(b,c)) / 6.0;
    }
    H.volumeApprox = fabs(V);
    return H;
}

static bool insideHull(const Hull& H, const Vec3& q) {
    for (const auto& f : H.faces) {
        if (dot(f.n, q) > f.d + 1e-7) return false;
    }
    return true;
}

struct TetPose { Vec3 c; double rx, ry, rz; array<Vec3,4> verts; };

static array<Vec3,4> canonicalTet(double e=1.0) {
    // Regular tetra, centered at origin.
    array<Vec3,4> v = {
        Vec3( 1, 1, 1),
        Vec3(-1,-1, 1),
        Vec3(-1, 1,-1),
        Vec3( 1,-1,-1)
    };
    // Edge length currently 2*sqrt(2). scale to e.
    double s = e / (2.0 * sqrt(2.0));
    for (auto& p : v) p = p * s;
    return v;
}

static array<Vec3,4> transformed(const array<Vec3,4>& base, const Mat3& R, const Vec3& c) {
    array<Vec3,4> out;
    for (int i=0;i<4;i++) out[i] = mul(R, base[i]) + c;
    return out;
}

static void project(const array<Vec3,4>& v, const Vec3& axis, double& mn, double& mx) {
    mn = mx = dot(v[0], axis);
    for (int i=1;i<4;i++) {
        double p = dot(v[i], axis);
        mn = min(mn, p);
        mx = max(mx, p);
    }
}

static bool overlap1D(double a0,double a1,double b0,double b1){ return !(a1 < b0 + 1e-10 || b1 < a0 + 1e-10); }

static bool tetraIntersectSAT(const array<Vec3,4>& A, const array<Vec3,4>& B) {
    static const int E[6][2] = {{0,1},{0,2},{0,3},{1,2},{1,3},{2,3}};
    vector<Vec3> axes;
    axes.reserve(44);

    // Face normals (4 each)
    auto addFaceNormals = [&](const array<Vec3,4>& T){
        int F[4][3] = {{0,1,2},{0,1,3},{0,2,3},{1,2,3}};
        for (auto& f : F) {
            Vec3 n = cross(T[f[1]]-T[f[0]], T[f[2]]-T[f[0]]);
            double l = norm(n);
            if (l > 1e-12) axes.push_back(n / l);
        }
    };
    addFaceNormals(A);
    addFaceNormals(B);

    // Edge cross products
    for (int i=0;i<6;i++) {
        Vec3 ea = A[E[i][1]] - A[E[i][0]];
        for (int j=0;j<6;j++) {
            Vec3 eb = B[E[j][1]] - B[E[j][0]];
            Vec3 ax = cross(ea, eb);
            double l = norm(ax);
            if (l > 1e-12) axes.push_back(ax / l);
        }
    }

    for (const Vec3& axis : axes) {
        double a0,a1,b0,b1;
        project(A, axis, a0, a1);
        project(B, axis, b0, b1);
        if (!overlap1D(a0,a1,b0,b1)) return false;
    }
    return true;
}

struct CellKey { int x,y,z; };
struct KeyHash {
    size_t operator()(const CellKey& k) const noexcept {
        uint64_t h = 1469598103934665603ull;
        h ^= (uint32_t)k.x; h *= 1099511628211ull;
        h ^= (uint32_t)k.y; h *= 1099511628211ull;
        h ^= (uint32_t)k.z; h *= 1099511628211ull;
        return (size_t)h;
    }
};
struct KeyEq { bool operator()(const CellKey& a,const CellKey& b) const noexcept { return a.x==b.x&&a.y==b.y&&a.z==b.z; }};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge) || V <= 0 || edge <= 0) {
        cout << 0 << "\n";
        return 0;
    }
    vector<Vec3> pts(V);
    for (int i=0;i<V;i++) cin >> pts[i].x >> pts[i].y >> pts[i].z;

    auto t0 = chrono::steady_clock::now();
    auto timeLeft = [&](double secLimit){
        double dt = chrono::duration<double>(chrono::steady_clock::now() - t0).count();
        return dt < secLimit;
    };

    Hull H = buildConvexHullBruteforce(pts);
    if (H.faces.empty()) {
        cout << 0 << "\n";
        return 0;
    }

    const double R = edge * sqrt(3.0/8.0);
    const double r = edge / sqrt(24.0);
    const auto base = canonicalTet(edge);

    vector<array<double,3>> eulers = {
        {0,0,0},
        {0,0,M_PI/2},
        {0,M_PI/2,0},
        {M_PI/2,0,0},
        {M_PI/3,M_PI/5,M_PI/7},
        {-M_PI/4,M_PI/6,-M_PI/3}
    };

    double boxVol = (H.bmax.x-H.bmin.x)*(H.bmax.y-H.bmin.y)*(H.bmax.z-H.bmin.z);
    double fillScale = (H.volumeApprox > EPS && boxVol > EPS) ? H.volumeApprox / boxVol : 0.2;

    double spacing = max(2.0*r*0.98, edge*0.58);
    if (boxVol > 5e6*edge*edge*edge) spacing *= 2.0;
    else if (boxVol > 5e4*edge*edge*edge) spacing *= 1.35;

    mt19937_64 rng(1234567);
    uniform_real_distribution<double> jitter(-0.18*spacing, 0.18*spacing);

    unordered_map<CellKey, vector<int>, KeyHash, KeyEq> grid;
    vector<TetPose> placed;
    placed.reserve(10000);
    double cell = 2.05 * R;

    auto keyOf = [&](const Vec3& c)->CellKey {
        return {(int)floor(c.x/cell), (int)floor(c.y/cell), (int)floor(c.z/cell)};
    };

    int nx = max(1, (int)floor((H.bmax.x-H.bmin.x)/spacing));
    int ny = max(1, (int)floor((H.bmax.y-H.bmin.y)/spacing));
    int nz = max(1, (int)floor((H.bmax.z-H.bmin.z)/spacing));

    vector<Vec3> candidates;
    candidates.reserve((size_t)nx*ny*nz);
    for (int ix=0; ix<=nx; ix++) {
        double x = H.bmin.x + ix*spacing;
        for (int iy=0; iy<=ny; iy++) {
            double y = H.bmin.y + iy*spacing;
            for (int iz=0; iz<=nz; iz++) {
                double z = H.bmin.z + iz*spacing;
                Vec3 c{x + jitter(rng), y + jitter(rng), z + jitter(rng)};
                if (insideHull(H, c)) candidates.push_back(c);
            }
        }
    }

    shuffle(candidates.begin(), candidates.end(), rng);

    size_t cap = candidates.size();
    if (cap > 250000) cap = 250000;
    if (fillScale < 0.05 && cap > 80000) cap = 80000;

    for (size_t ci=0; ci<cap && timeLeft(28.5); ci++) {
        const Vec3 c = candidates[ci];
        bool added = false;

        for (auto ang : eulers) {
            Mat3 Rm = fromEulerXYZ(ang[0], ang[1], ang[2]);
            auto verts = transformed(base, Rm, c);

            bool in = true;
            for (const auto& vtx : verts) if (!insideHull(H, vtx)) { in = false; break; }
            if (!in) continue;

            CellKey kc = keyOf(c);
            bool collide = false;

            for (int dx=-1; dx<=1 && !collide; dx++)
            for (int dy=-1; dy<=1 && !collide; dy++)
            for (int dz=-1; dz<=1 && !collide; dz++) {
                CellKey kn{kc.x+dx, kc.y+dy, kc.z+dz};
                auto it = grid.find(kn);
                if (it == grid.end()) continue;
                for (int idx : it->second) {
                    const auto& T = placed[idx];
                    // Fast culling by circumsphere.
                    if (norm2(T.c - c) > (2.0*R)*(2.0*R) + 1e-9) continue;
                    if (tetraIntersectSAT(T.verts, verts)) { collide = true; break; }
                }
            }
            if (collide) continue;

            TetPose tp;
            tp.c = c;
            tp.rx = ang[0]; tp.ry = ang[1]; tp.rz = ang[2];
            tp.verts = verts;
            int idx = (int)placed.size();
            placed.push_back(tp);
            grid[kc].push_back(idx);
            added = true;
            break;
        }

        if (!added && (ci % 1000 == 0) && !timeLeft(27.0)) break;
    }

    cout << placed.size() << "\n";
    for (const auto& t : placed) {
        cout << t.c.x << ' ' << t.c.y << ' ' << t.c.z << ' '
             << t.rx << ' ' << t.ry << ' ' << t.rz << "\n";
    }
    return 0;
}
