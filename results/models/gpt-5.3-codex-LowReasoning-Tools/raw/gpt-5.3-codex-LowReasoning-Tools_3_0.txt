use std::collections::{HashMap, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

#[derive(Clone, Copy, Default)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt { x: self.x + o.x, y: self.y + o.y }
    }
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt { x: self.x - o.x, y: self.y - o.y }
    }
    #[inline]
    fn scale(self, k: f64) -> Pt {
        Pt { x: self.x * k, y: self.y * k }
    }
    #[inline]
    fn norm2(self) -> f64 {
        self.x * self.x + self.y * self.y
    }
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { data: input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.data.len() {
            return None;
        }
        let mut val = 0usize;
        while self.idx < self.data.len() && !self.data[self.idx].is_ascii_whitespace() {
            let c = self.data[self.idx];
            if c.is_ascii_digit() {
                val = val.saturating_mul(10).saturating_add((c - b'0') as usize);
            }
            self.idx += 1;
        }
        Some(val)
    }
}

#[derive(Clone)]
struct XorShift64 {
    s: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        let s = if seed == 0 { 0x9e3779b97f4a7c15 } else { seed };
        Self { s }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.s;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.s = x;
        x
    }
    #[inline]
    fn next_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn connected_components(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut seen = vec![false; n];
    let mut comps = Vec::new();
    let mut q = VecDeque::new();
    for s in 0..n {
        if seen[s] {
            continue;
        }
        seen[s] = true;
        q.push_back(s);
        let mut comp = Vec::new();
        while let Some(u) = q.pop_front() {
            comp.push(u);
            for &v in &adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
        comps.push(comp);
    }
    comps
}

fn bfs_order_and_depth(comp: &[usize], adj: &[Vec<usize>]) -> (Vec<usize>, Vec<usize>) {
    if comp.is_empty() {
        return (Vec::new(), Vec::new());
    }
    let mut start = comp[0];
    for &u in comp {
        if adj[u].len() > adj[start].len() {
            start = u;
        }
    }

    let mut in_comp = HashMap::with_capacity(comp.len() * 2);
    for (i, &u) in comp.iter().enumerate() {
        in_comp.insert(u, i);
    }

    let mut depth = vec![usize::MAX; comp.len()];
    let mut order = Vec::with_capacity(comp.len());
    let mut q = VecDeque::new();
    depth[*in_comp.get(&start).unwrap()] = 0;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        order.push(u);
        let du = depth[*in_comp.get(&u).unwrap()];
        for &v in &adj[u] {
            if let Some(&iv) = in_comp.get(&v) {
                if depth[iv] == usize::MAX {
                    depth[iv] = du + 1;
                    q.push_back(v);
                }
            }
        }
    }

    if order.len() < comp.len() {
        let mut rem: Vec<usize> = comp
            .iter()
            .copied()
            .filter(|u| depth[*in_comp.get(u).unwrap()] == usize::MAX)
            .collect();
        rem.sort_by_key(|&u| std::cmp::Reverse(adj[u].len()));
        for u in rem {
            depth[*in_comp.get(&u).unwrap()] = 0;
            order.push(u);
        }
    }

    (order, depth)
}

fn layout_large_component(comp: &[usize], adj: &[Vec<usize>], pos: &mut [Pt]) {
    let (order, depth) = bfs_order_and_depth(comp, adj);
    if comp.len() == 1 {
        pos[comp[0]] = Pt { x: 0.0, y: 0.0 };
        return;
    }

    let mut in_comp = HashMap::with_capacity(comp.len() * 2);
    for (i, &u) in comp.iter().enumerate() {
        in_comp.insert(u, i);
    }

    let mut layers: HashMap<usize, Vec<usize>> = HashMap::new();
    for &u in &order {
        let d = depth[*in_comp.get(&u).unwrap()];
        layers.entry(d).or_default().push(u);
    }

    let mut keys: Vec<usize> = layers.keys().copied().collect();
    keys.sort_unstable();

    let base_r = 6.0f64;
    let layer_gap = 10.0f64;
    for d in keys {
        let arr = &mut layers.get_mut(&d).unwrap();
        arr.sort_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));
        let k = arr.len();
        let r = base_r + layer_gap * (d as f64).sqrt();
        for (i, &u) in arr.iter().enumerate() {
            let ang = 2.0 * std::f64::consts::PI * (i as f64) / (k.max(1) as f64);
            pos[u] = Pt { x: r * ang.cos(), y: r * ang.sin() };
        }
    }

    // One linear-time Laplacian smoothing pass (attractive-only) to reduce edge stretch.
    let alpha = 0.35;
    let mut newp = Vec::with_capacity(comp.len());
    for &u in comp {
        if adj[u].is_empty() {
            newp.push(pos[u]);
            continue;
        }
        let mut s = Pt { x: 0.0, y: 0.0 };
        let mut c = 0usize;
        for &v in &adj[u] {
            s = s.add(pos[v]);
            c += 1;
        }
        let avg = s.scale(1.0 / c as f64);
        newp.push(pos[u].scale(1.0 - alpha).add(avg.scale(alpha)));
    }
    for (i, &u) in comp.iter().enumerate() {
        pos[u] = newp[i];
    }
}

fn layout_force_component(comp: &[usize], edges_comp: &[Edge], adj: &[Vec<usize>], pos: &mut [Pt], rng: &mut XorShift64) {
    let k = comp.len();
    if k <= 1 {
        if k == 1 {
            pos[comp[0]] = Pt { x: 0.0, y: 0.0 };
        }
        return;
    }

    let area = (k as f64).max(10.0) * 40.0;
    let width = area.sqrt();
    let ideal = (area / k as f64).sqrt();

    for &u in comp {
        pos[u] = Pt {
            x: (rng.next_f64() - 0.5) * width,
            y: (rng.next_f64() - 0.5) * width,
        };
    }

    let iters = match k {
        0..=80 => 220,
        81..=300 => 160,
        301..=1000 => 110,
        _ => 70,
    };

    let mut t = width * 0.12;
    let t_decay = 0.95f64;
    let mut disp: HashMap<usize, Pt> = HashMap::with_capacity(k * 2);

    for _ in 0..iters {
        disp.clear();
        let cell = ideal * 2.2 + 1e-9;
        let mut grid: HashMap<(i32, i32), Vec<usize>> = HashMap::with_capacity(k * 2);

        for &u in comp {
            let cx = (pos[u].x / cell).floor() as i32;
            let cy = (pos[u].y / cell).floor() as i32;
            grid.entry((cx, cy)).or_default().push(u);
            disp.insert(u, Pt { x: 0.0, y: 0.0 });
        }

        for &u in comp {
            let pu = pos[u];
            let cx = (pu.x / cell).floor() as i32;
            let cy = (pu.y / cell).floor() as i32;
            let mut du = *disp.get(&u).unwrap_or(&Pt { x: 0.0, y: 0.0 });

            for dx in -1..=1 {
                for dy in -1..=1 {
                    if let Some(bucket) = grid.get(&(cx + dx, cy + dy)) {
                        for &v in bucket {
                            if u == v {
                                continue;
                            }
                            let delta = pu.sub(pos[v]);
                            let d2 = delta.norm2() + 1e-9;
                            let d = d2.sqrt();
                            let f = (ideal * ideal) / d;
                            du = du.add(delta.scale(f / d));
                        }
                    }
                }
            }
            disp.insert(u, du);
        }

        for &e in edges_comp {
            let u = e.u;
            let v = e.v;
            if u == v {
                continue;
            }
            let delta = pos[u].sub(pos[v]);
            let d2 = delta.norm2() + 1e-9;
            let d = d2.sqrt();
            let f = d2 / ideal;
            let pull = delta.scale(f / d);
            let du = disp.get(&u).copied().unwrap_or_default().sub(pull);
            let dv = disp.get(&v).copied().unwrap_or_default().add(pull);
            disp.insert(u, du);
            disp.insert(v, dv);
        }

        for &u in comp {
            let mut d = disp.get(&u).copied().unwrap_or_default();
            let mag2 = d.norm2();
            if mag2 > 0.0 {
                let mag = mag2.sqrt();
                let lim = if mag > t { t / mag } else { 1.0 };
                d = d.scale(lim);
                pos[u] = pos[u].add(d);
            }
        }

        t *= t_decay;
        if t < 0.01 {
            break;
        }
    }

    // Optional local smoothing for tiny/medium components.
    if k <= 1500 {
        let alpha = 0.1;
        let mut np = Vec::with_capacity(k);
        for &u in comp {
            if adj[u].is_empty() {
                np.push(pos[u]);
                continue;
            }
            let mut s = Pt { x: 0.0, y: 0.0 };
            for &v in &adj[u] {
                s = s.add(pos[v]);
            }
            let avg = s.scale(1.0 / adj[u].len() as f64);
            np.push(pos[u].scale(1.0 - alpha).add(avg.scale(alpha)));
        }
        for (i, &u) in comp.iter().enumerate() {
            pos[u] = np[i];
        }
    }
}

fn bbox(nodes: &[usize], pos: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &u in nodes {
        minx = minx.min(pos[u].x);
        maxx = maxx.max(pos[u].x);
        miny = miny.min(pos[u].y);
        maxy = maxy.max(pos[u].y);
    }
    (minx, maxx, miny, maxy)
}

fn main() {
    let mut input = FastInput::new();
    let n = input.next_usize().unwrap_or(0);
    let m = input.next_usize().unwrap_or(0);

    let mut edges = Vec::with_capacity(m);
    let mut adj = vec![Vec::<usize>::new(); n];

    for _ in 0..m {
        let u = input.next_usize().unwrap_or(0);
        let v = input.next_usize().unwrap_or(0);
        if u >= n || v >= n {
            continue;
        }
        edges.push(Edge { u, v });
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    if n == 0 {
        return;
    }

    let comps = connected_components(&adj);
    let mut pos = vec![Pt { x: 0.0, y: 0.0 }; n];
    let mut rng = XorShift64::new((n as u64).wrapping_mul(11995408973635179863u64) ^ (m as u64));

    // Map node -> component id for fast edge partition.
    let mut comp_id = vec![usize::MAX; n];
    for (cid, comp) in comps.iter().enumerate() {
        for &u in comp {
            comp_id[u] = cid;
        }
    }
    let mut edges_by_comp: Vec<Vec<Edge>> = vec![Vec::new(); comps.len()];
    for &e in &edges {
        if e.u < n && e.v < n {
            let cu = comp_id[e.u];
            let cv = comp_id[e.v];
            if cu == cv && cu != usize::MAX {
                edges_by_comp[cu].push(e);
            }
        }
    }

    for (cid, comp) in comps.iter().enumerate() {
        let sz = comp.len();
        let ecount = edges_by_comp[cid].len();
        let dense = if sz > 1 {
            (2.0 * ecount as f64) / ((sz * (sz - 1)) as f64)
        } else {
            0.0
        };

        if sz <= 4000 {
            layout_force_component(comp, &edges_by_comp[cid], &adj, &mut pos, &mut rng);
        } else if sz <= 20000 && dense < 0.02 {
            // Sparse large components still benefit from limited-force pass with fewer iterations.
            layout_force_component(comp, &edges_by_comp[cid], &adj, &mut pos, &mut rng);
        } else {
            layout_large_component(comp, &adj, &mut pos);
        }
    }

    // Pack disconnected components into a coarse grid to avoid overlap.
    let mut comp_order: Vec<usize> = (0..comps.len()).collect();
    comp_order.sort_by(|&a, &b| comps[b].len().cmp(&comps[a].len()));

    let mut placements = vec![Pt { x: 0.0, y: 0.0 }; comps.len()];
    let cols = ((comps.len() as f64).sqrt().ceil() as usize).max(1);
    let gap = 80.0;

    for (i, &cid) in comp_order.iter().enumerate() {
        let r = i / cols;
        let c = i % cols;
        let (minx, maxx, miny, maxy) = bbox(&comps[cid], &pos);
        let w = (maxx - minx).max(1.0);
        let h = (maxy - miny).max(1.0);
        placements[cid] = Pt {
            x: c as f64 * (gap + w),
            y: r as f64 * (gap + h),
        };
    }

    for cid in 0..comps.len() {
        let (minx, _, miny, _) = bbox(&comps[cid], &pos);
        let off = placements[cid];
        for &u in &comps[cid] {
            pos[u].x += off.x - minx;
            pos[u].y += off.y - miny;
        }
    }

    // Final normalization for nicer numeric scale.
    let (minx, maxx, miny, maxy) = {
        let nodes: Vec<usize> = (0..n).collect();
        bbox(&nodes, &pos)
    };
    let cx = 0.5 * (minx + maxx);
    let cy = 0.5 * (miny + maxy);
    let span = (maxx - minx).max(maxy - miny).max(1.0);
    let scale = 1000.0 / span;

    let mut out = String::with_capacity(n * 32);
    for i in 0..n {
        let x = (pos[i].x - cx) * scale;
        let y = (pos[i].y - cy) * scale;
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{}", out);
}
