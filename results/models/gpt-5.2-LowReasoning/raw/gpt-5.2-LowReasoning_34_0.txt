import sys
import math
from typing import List, Tuple


def dist(a: Tuple[float, float], b: Tuple[float, float]) -> float:
    dx = a[0] - b[0]
    dy = a[1] - b[1]
    return math.hypot(dx, dy)


def route_distance(route: List[int], pts: List[Tuple[float, float]], depot: Tuple[float, float]) -> float:
    if not route:
        return 0.0
    total = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        total += dist(pts[route[i]], pts[route[i + 1]])
    total += dist(pts[route[-1]], depot)
    return total


def two_opt(route: List[int], pts: List[Tuple[float, float]], depot: Tuple[float, float]) -> List[int]:
    # Standard 2-opt; O(n^2). Only used on moderate-sized routes.
    n = len(route)
    if n < 4:
        return route

    def seg_cost(i: int, j: int) -> float:
        # cost of edges (i-1,i) and (j,j+1)
        a = depot if i == 0 else pts[route[i - 1]]
        b = pts[route[i]]
        c = pts[route[j]]
        d = depot if j == n - 1 else pts[route[j + 1]]
        return dist(a, b) + dist(c, d)

    improved = True
    while improved:
        improved = False
        for i in range(0, n - 2):
            for j in range(i + 1, n - 1):
                before = seg_cost(i, j)
                a = depot if i == 0 else pts[route[i - 1]]
                b = pts[route[i]]
                c = pts[route[j]]
                d = depot if j == n - 1 else pts[route[j + 1]]
                after = dist(a, c) + dist(b, d)
                if after + 1e-12 < before:
                    route[i : j + 1] = reversed(route[i : j + 1])
                    improved = True
        # A single pass is usually enough; loop a few times at most.
        # If still improving, we'll keep going until a local optimum.
    return route


def clarke_wright(
    pts: List[Tuple[float, float]],
    demands: List[int],
    depot: Tuple[float, float],
    capacity: int,
    k_max: int,
) -> List[List[int]]:
    n = len(pts)

    # Start with one route per customer.
    routes: List[List[int]] = [[i] for i in range(n)]
    route_of = list(range(n))  # customer -> route index
    route_loads = demands[:]

    # Route endpoints for fast checks
    def endpoints(r: List[int]) -> Tuple[int, int]:
        return r[0], r[-1]

    # Precompute depot distances
    d0 = [dist(depot, pts[i]) for i in range(n)]

    # Build savings list: s(i,j) = d0[i] + d0[j] - dij
    savings = []
    for i in range(n):
        pi = pts[i]
        for j in range(i + 1, n):
            s = d0[i] + d0[j] - dist(pi, pts[j])
            savings.append((s, i, j))
    savings.sort(reverse=True, key=lambda x: x[0])

    def try_merge(i: int, j: int) -> bool:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            return False
        if route_loads[ri] + route_loads[rj] > capacity:
            return False
        r1 = routes[ri]
        r2 = routes[rj]
        if not r1 or not r2:
            return False

        a1, b1 = endpoints(r1)
        a2, b2 = endpoints(r2)

        # We can only merge if i is an end of r1 and j is an end of r2
        i_is_start = (i == a1)
        i_is_end = (i == b1)
        j_is_start = (j == a2)
        j_is_end = (j == b2)
        if not (i_is_start or i_is_end):
            return False
        if not (j_is_start or j_is_end):
            return False

        # Orient and concatenate so i ... j becomes a single path
        if i_is_end and j_is_start:
            merged = r1 + r2
        elif i_is_start and j_is_end:
            merged = r2 + r1
        elif i_is_start and j_is_start:
            merged = list(reversed(r1)) + r2
        else:  # i_is_end and j_is_end
            merged = r1 + list(reversed(r2))

        # Apply merge: keep ri, invalidate rj
        routes[ri] = merged
        routes[rj] = []
        route_loads[ri] += route_loads[rj]
        route_loads[rj] = 0
        for c in merged:
            route_of[c] = ri
        return True

    # Greedy savings merges (only if feasible and beneficial by savings ordering).
    for _s, i, j in savings:
        try_merge(i, j)

    # Compact routes
    compact = []
    compact_loads = []
    for r, load in zip(routes, route_loads):
        if r:
            compact.append(r)
            compact_loads.append(load)

    # If we still have too many routes, force merges with minimal distance increase.
    # This can happen if savings merges were blocked by end-point constraints ordering.
    # We allow "negative savings" merges by selecting best feasible endpoint merge each time.
    while len(compact) > k_max:
        best = None  # (delta, a_idx, b_idx, mode)
        m = len(compact)
        for ra in range(m):
            for rb in range(ra + 1, m):
                if compact_loads[ra] + compact_loads[rb] > capacity:
                    continue
                A = compact[ra]
                B = compact[rb]
                a0, a1 = A[0], A[-1]
                b0, b1 = B[0], B[-1]

                # Consider 4 endpoint orientations
                candidates = [
                    ("a_end_b_start", A, B, a1, b0, False, False),
                    ("a_start_b_end", B, A, b1, a0, False, False),
                    ("a_start_b_start", list(reversed(A)), B, a0, b0, True, False),
                    ("a_end_b_end", A, list(reversed(B)), a1, b1, False, True),
                ]
                for mode, left, right, left_end, right_start, _ra_rev, _rb_rev in candidates:
                    # delta = new - old (only at the join vs two depot legs)
                    join = dist(pts[left_end], pts[right_start])
                    delta = join - d0[left_end] - d0[right_start]
                    if best is None or delta < best[0]:
                        best = (delta, ra, rb, mode)
        if best is None:
            break  # no feasible forced merge (shouldn't happen if instance is feasible)
        _, ra, rb, mode = best
        A = compact[ra]
        B = compact[rb]
        if mode == "a_end_b_start":
            merged = A + B
        elif mode == "a_start_b_end":
            merged = B + A
        elif mode == "a_start_b_start":
            merged = list(reversed(A)) + B
        else:  # a_end_b_end
            merged = A + list(reversed(B))

        # Replace ra with merged, remove rb
        compact[ra] = merged
        compact_loads[ra] += compact_loads[rb]
        del compact[rb]
        del compact_loads[rb]

    return compact


def sweep_construct(
    pts: List[Tuple[float, float]],
    demands: List[int],
    depot: Tuple[float, float],
    capacity: int,
    k_max: int,
) -> List[List[int]]:
    n = len(pts)
    angles = []
    for i, (x, y) in enumerate(pts):
        angles.append((math.atan2(y - depot[1], x - depot[0]), i))
    angles.sort()

    routes: List[List[int]] = []
    loads: List[int] = []
    cur: List[int] = []
    cur_load = 0
    for _ang, i in angles:
        d = demands[i]
        if cur and cur_load + d > capacity:
            routes.append(cur)
            loads.append(cur_load)
            cur = []
            cur_load = 0
        cur.append(i)
        cur_load += d
    if cur:
        routes.append(cur)
        loads.append(cur_load)

    # If more than K, greedily merge adjacent routes when feasible
    while len(routes) > k_max:
        best = None  # (delta, idx)
        for idx in range(len(routes) - 1):
            if loads[idx] + loads[idx + 1] > capacity:
                continue
            A = routes[idx]
            B = routes[idx + 1]
            # Merge as A + B (keeps sweep order)
            join = dist(pts[A[-1]], pts[B[0]])
            delta = join - dist(depot, pts[A[-1]]) - dist(depot, pts[B[0]])
            if best is None or delta < best[0]:
                best = (delta, idx)
        if best is None:
            break
        _, idx = best
        routes[idx] = routes[idx] + routes[idx + 1]
        loads[idx] += loads[idx + 1]
        del routes[idx + 1]
        del loads[idx + 1]

    return routes


def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    k_max = int(next(it))
    capacity = int(next(it))
    depot = (float(next(it)), float(next(it)))

    pts: List[Tuple[float, float]] = []
    demands: List[int] = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        pts.append((x, y))
        demands.append(d)

    # Construct initial solution
    if n <= 600:
        routes = clarke_wright(pts, demands, depot, capacity, k_max)
    else:
        routes = sweep_construct(pts, demands, depot, capacity, k_max)

    # Ensure all customers covered exactly once
    seen = [False] * n
    fixed_routes: List[List[int]] = []
    for r in routes:
        nr = []
        load = 0
        for c in r:
            if 0 <= c < n and not seen[c]:
                if load + demands[c] <= capacity:
                    nr.append(c)
                    seen[c] = True
                    load += demands[c]
                else:
                    # Should not happen; split into a new route.
                    fixed_routes.append(nr)
                    nr = [c]
                    load = demands[c]
                    seen[c] = True
        if nr:
            fixed_routes.append(nr)

    # Add any missing customers (fallback)
    for c in range(n):
        if not seen[c]:
            fixed_routes.append([c])
            seen[c] = True

    routes = fixed_routes

    # If still > K, try to force merges similar to Clarke-Wright forced merge
    # (handles edge cases from splitting/fallback).
    loads = [sum(demands[c] for c in r) for r in routes]
    d0 = [dist(depot, pts[i]) for i in range(n)]
    while len(routes) > k_max:
        best = None  # (delta, ra, rb, mode)
        m = len(routes)
        for ra in range(m):
            for rb in range(ra + 1, m):
                if loads[ra] + loads[rb] > capacity:
                    continue
                A = routes[ra]
                B = routes[rb]
                a0, a1 = A[0], A[-1]
                b0, b1 = B[0], B[-1]
                options = [
                    ("a_end_b_start", A, B, a1, b0),
                    ("a_start_b_end", B, A, b1, a0),
                    ("a_start_b_start", list(reversed(A)), B, a0, b0),
                    ("a_end_b_end", A, list(reversed(B)), a1, b1),
                ]
                for mode, left, right, le, rs in options:
                    join = dist(pts[le], pts[rs])
                    delta = join - d0[le] - d0[rs]
                    if best is None or delta < best[0]:
                        best = (delta, ra, rb, mode)
        if best is None:
            break
        _, ra, rb, mode = best
        A = routes[ra]
        B = routes[rb]
        if mode == "a_end_b_start":
            merged = A + B
        elif mode == "a_start_b_end":
            merged = B + A
        elif mode == "a_start_b_start":
            merged = list(reversed(A)) + B
        else:
            merged = A + list(reversed(B))
        routes[ra] = merged
        loads[ra] += loads[rb]
        del routes[rb]
        del loads[rb]

    # Local improvement: 2-opt on each route (avoid huge routes)
    for idx, r in enumerate(routes):
        if 4 <= len(r) <= 250:
            routes[idx] = two_opt(r, pts, depot)

    total = sum(route_distance(r, pts, depot) for r in routes)

    out_lines = []
    out_lines.append(f"{total:.6f}")
    out_lines.append(str(len(routes)))
    for r in routes:
        out_lines.append(" ".join(str(c) for c in r))
    sys.stdout.write("\n".join(out_lines))


if __name__ == "__main__":
    main()
