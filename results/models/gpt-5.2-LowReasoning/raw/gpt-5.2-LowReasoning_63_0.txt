declare const require: any;
const fs = require("fs");

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

type Input = { grid: string[]; robots: Robot[]; jobs: Job[] };

class GridBFSCache {
  private readonly w: number;
  private readonly h: number;
  private readonly open: Uint8Array;
  private readonly n: number;

  private stamp = 1;
  private readonly seen: Uint32Array;
  private readonly dist: Int32Array;
  private readonly q: Int32Array;

  private readonly cache = new Map<number, Int32Array>();

  constructor(grid: string[]) {
    this.h = grid.length;
    this.w = grid.length ? grid[0].length : 0;
    this.n = this.w * this.h;
    this.open = new Uint8Array(this.n);
    for (let y = 0; y < this.h; y++) {
      const row = grid[y];
      for (let x = 0; x < this.w; x++) {
        this.open[y * this.w + x] = row.charCodeAt(x) === 46 /* '.' */ ? 1 : 0;
      }
    }
    this.seen = new Uint32Array(this.n);
    this.dist = new Int32Array(this.n);
    this.q = new Int32Array(this.n);
  }

  cellId(x: number, y: number): number {
    return y * this.w + x;
  }

  private bfsAllFrom(startCell: number): Int32Array {
    const cached = this.cache.get(startCell);
    if (cached) return cached;

    const w = this.w;
    const h = this.h;
    const n = this.n;
    const open = this.open;
    const seen = this.seen;
    const dist = this.dist;
    const q = this.q;

    let s = this.stamp + 1;
    if (s === 0xffffffff) {
      seen.fill(0);
      s = 1;
    }
    this.stamp = s;

    let head = 0;
    let tail = 0;
    seen[startCell] = s;
    dist[startCell] = 0;
    q[tail++] = startCell;

    while (head < tail) {
      const v = q[head++];
      const dv = dist[v] + 1;

      const x = v % w;
      const y = (v / w) | 0;

      if (x > 0) {
        const nb = v - 1;
        if (open[nb] && seen[nb] !== s) {
          seen[nb] = s;
          dist[nb] = dv;
          q[tail++] = nb;
        }
      }
      if (x + 1 < w) {
        const nb = v + 1;
        if (open[nb] && seen[nb] !== s) {
          seen[nb] = s;
          dist[nb] = dv;
          q[tail++] = nb;
        }
      }
      if (y > 0) {
        const nb = v - w;
        if (open[nb] && seen[nb] !== s) {
          seen[nb] = s;
          dist[nb] = dv;
          q[tail++] = nb;
        }
      }
      if (y + 1 < h) {
        const nb = v + w;
        if (open[nb] && seen[nb] !== s) {
          seen[nb] = s;
          dist[nb] = dv;
          q[tail++] = nb;
        }
      }
    }

    const out = new Int32Array(n);
    for (let i = 0; i < n; i++) out[i] = seen[i] === s ? dist[i] : -1;
    this.cache.set(startCell, out);
    return out;
  }

  distance(aCell: number, bCell: number): number {
    if (aCell === bCell) return 0;
    const da = this.bfsAllFrom(aCell);
    const d = da[bCell];
    return d >= 0 ? d : 1_000_000_000;
  }
}

function stableJobSort(a: Job, b: Job): number {
  if (a.priority !== b.priority) return b.priority - a.priority;
  return a.id - b.id;
}

function stableRobotSort(a: Robot, b: Robot): number {
  return a.id - b.id;
}

function main() {
  const input: string = fs.readFileSync(0, "utf8");
  const data: Input = JSON.parse(input);

  const bfs = new GridBFSCache(data.grid);

  const robots = data.robots.slice().sort(stableRobotSort);
  const jobs = data.jobs.slice().sort(stableJobSort);

  let maxRobotId = -1;
  for (const r of robots) if (r.id > maxRobotId) maxRobotId = r.id;
  const assignments: number[][] = Array.from({ length: maxRobotId + 1 }, () => []);

  const robotPos = new Map<number, number>();
  for (const r of robots) robotPos.set(r.id, bfs.cellId(r.x, r.y));

  for (const job of jobs) {
    const jobCell = bfs.cellId(job.x, job.y);

    let bestRobotId = robots[0]?.id ?? 0;
    let bestDist = 1_000_000_000;
    for (const r of robots) {
      const curCell = robotPos.get(r.id)!;
      const d = bfs.distance(curCell, jobCell);
      if (d < bestDist || (d === bestDist && r.id < bestRobotId)) {
        bestDist = d;
        bestRobotId = r.id;
      }
    }

    assignments[bestRobotId].push(job.id);
    robotPos.set(bestRobotId, jobCell);
  }

  process.stdout.write(JSON.stringify({ assignments }));
}

main();
