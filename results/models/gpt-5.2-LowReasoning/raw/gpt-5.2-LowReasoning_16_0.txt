// Job-Shop Scheduling (feasible heuristic solver)
// Input:  J M
// For each job j: T_j then T_j lines: machine duration
// Output: For each job j: "start machine" pairs for each task in order.
//
// This solver focuses on always producing a valid schedule and scaling to large instances.
// It uses a Serial Schedule Generation Scheme with a priority queue and lazy key updates.

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <iostream>
#include <limits>
#include <queue>
#include <string>
#include <tuple>
#include <utility>
#include <vector>

using i64 = long long;

struct Task {
  int machine = 0;
  int dur = 0;
};

struct Node {
  i64 key_time;        // cached earliest start time
  i64 key_tiebreak;    // cached secondary priority (lower is better)
  int job;
  int task_idx;
  i64 seen_job_ready;
  i64 seen_mach_ready;
};

struct NodeCmp {
  bool operator()(const Node& a, const Node& b) const {
    if (a.key_time != b.key_time) return a.key_time > b.key_time;          // min-heap behavior
    if (a.key_tiebreak != b.key_tiebreak) return a.key_tiebreak > b.key_tiebreak;
    if (a.job != b.job) return a.job > b.job;
    return a.task_idx > b.task_idx;
  }
};

static i64 safe_max(i64 a, i64 b) { return (a > b) ? a : b; }

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int J = 0, M = 0;
  if (!(std::cin >> J >> M)) return 0;
  if (J < 0 || M < 0) return 0;

  std::vector<std::vector<Task>> jobs;
  jobs.resize(J);

  i64 total_tasks = 0;
  for (int j = 0; j < J; ++j) {
    int T = 0;
    std::cin >> T;
    if (T < 0) T = 0;
    jobs[j].resize(T);
    total_tasks += T;
    for (int t = 0; t < T; ++t) {
      int m = 0, d = 0;
      std::cin >> m >> d;
      if (m < 0) m = 0;
      if (m >= M) m = (M > 0) ? (m % M) : 0;
      if (d < 0) d = 0;
      jobs[j][t] = Task{m, d};
    }
  }

  // Output data: start times for each task.
  std::vector<std::vector<i64>> starts(J);
  for (int j = 0; j < J; ++j) starts[j].assign(jobs[j].size(), 0);

  // Trivial edge cases.
  if (J == 0) return 0;
  if (M <= 0) {
    // No machines: schedule everything at time 0 on machine 0 (degenerate but prints required format).
    for (int j = 0; j < J; ++j) {
      for (size_t t = 0; t < jobs[j].size(); ++t) {
        if (t) std::cout << ' ';
        std::cout << 0 << ' ' << 0;
      }
      std::cout << "\n";
    }
    return 0;
  }

  // Adaptive heuristic knobs (cheap to compute; helps on varied scales).
  // For medium instances we bias tie-breaking towards jobs with larger remaining work.
  const bool use_remaining_bias = (total_tasks <= 200000); // keep it cheap on extreme sizes too

  // Precompute remaining work per job (suffix sums), stored as i64.
  std::vector<std::vector<i64>> suffix_work;
  if (use_remaining_bias) {
    suffix_work.resize(J);
    for (int j = 0; j < J; ++j) {
      const int T = (int)jobs[j].size();
      suffix_work[j].assign(T + 1, 0);
      for (int t = T - 1; t >= 0; --t) {
        suffix_work[j][t] = suffix_work[j][t + 1] + (i64)jobs[j][t].dur;
      }
    }
  }

  std::vector<i64> job_ready(J, 0);
  std::vector<int> job_next(J, 0);
  std::vector<i64> mach_ready(M, 0);

  // Serial schedule generation with lazy updates.
  std::priority_queue<Node, std::vector<Node>, NodeCmp> pq;
  pq = decltype(pq)(NodeCmp{}, {});

  auto push_op = [&](int j) {
    const int t = job_next[j];
    if (t >= (int)jobs[j].size()) return;
    const Task& op = jobs[j][t];
    const i64 jr = job_ready[j];
    const i64 mr = mach_ready[op.machine];
    const i64 est = safe_max(jr, mr);
    i64 tie = 0;
    if (use_remaining_bias) {
      // Lower is better, so use negative remaining work to prefer larger remaining work.
      tie = -suffix_work[j][t];
    } else {
      // Prefer shorter processing time for extreme scales to reduce heap churn.
      tie = (i64)op.dur;
    }
    pq.push(Node{est, tie, j, t, jr, mr});
  };

  for (int j = 0; j < J; ++j) push_op(j);

  i64 scheduled = 0;
  while (!pq.empty() && scheduled < total_tasks) {
    Node cur = pq.top();
    pq.pop();

    const int j = cur.job;
    const int t = cur.task_idx;
    if (t != job_next[j]) continue; // already scheduled/advanced

    const Task& op = jobs[j][t];
    const i64 jr = job_ready[j];
    const i64 mr = mach_ready[op.machine];
    const i64 est = safe_max(jr, mr);

    // Lazy key update if the cached view is stale.
    if (cur.seen_job_ready != jr || cur.seen_mach_ready != mr || cur.key_time != est) {
      i64 tie = cur.key_tiebreak;
      if (use_remaining_bias) {
        tie = -suffix_work[j][t];
      } else {
        tie = (i64)op.dur;
      }
      pq.push(Node{est, tie, j, t, jr, mr});
      continue;
    }

    // Schedule it.
    const i64 start = est;
    const i64 finish = start + (i64)op.dur;
    starts[j][t] = start;
    job_ready[j] = finish;
    mach_ready[op.machine] = finish;
    job_next[j] = t + 1;
    ++scheduled;

    // Push next operation of this job.
    push_op(j);
  }

  // If something went wrong (shouldn't), fall back to "as early as possible in job order".
  if (scheduled != total_tasks) {
    std::fill(job_ready.begin(), job_ready.end(), 0);
    std::fill(mach_ready.begin(), mach_ready.end(), 0);
    for (int j = 0; j < J; ++j) {
      for (int t = 0; t < (int)jobs[j].size(); ++t) {
        const Task& op = jobs[j][t];
        const i64 start = safe_max(job_ready[j], mach_ready[op.machine]);
        const i64 finish = start + (i64)op.dur;
        starts[j][t] = start;
        job_ready[j] = finish;
        mach_ready[op.machine] = finish;
      }
    }
  }

  // Emit schedule: start_time machine pairs per job, tasks in order.
  for (int j = 0; j < J; ++j) {
    for (int t = 0; t < (int)jobs[j].size(); ++t) {
      if (t) std::cout << ' ';
      std::cout << starts[j][t] << ' ' << jobs[j][t].machine;
    }
    std::cout << "\n";
  }

  return 0;
}
