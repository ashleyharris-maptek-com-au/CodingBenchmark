use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn add(self, o: Vec2) -> Vec2 {
        Vec2::new(self.x + o.x, self.y + o.y)
    }
    fn sub(self, o: Vec2) -> Vec2 {
        Vec2::new(self.x - o.x, self.y - o.y)
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min: Vec2,
    max: Vec2,
}

impl BBox {
    fn new(min: Vec2, max: Vec2) -> Self {
        Self { min, max }
    }
    fn empty() -> Self {
        Self {
            min: Vec2::new(f64::INFINITY, f64::INFINITY),
            max: Vec2::new(f64::NEG_INFINITY, f64::NEG_INFINITY),
        }
    }
    fn include(&mut self, p: Vec2) {
        self.min.x = self.min.x.min(p.x);
        self.min.y = self.min.y.min(p.y);
        self.max.x = self.max.x.max(p.x);
        self.max.y = self.max.y.max(p.y);
    }
    fn width(&self) -> f64 {
        self.max.x - self.min.x
    }
    fn height(&self) -> f64 {
        self.max.y - self.min.y
    }
    fn intersects(&self, o: &BBox) -> bool {
        !(self.max.x < o.min.x || self.min.x > o.max.x || self.max.y < o.min.y || self.min.y > o.max.y)
    }
}

fn rotate(p: Vec2, ang: f64) -> Vec2 {
    let (s, c) = ang.sin_cos();
    Vec2::new(p.x * c - p.y * s, p.x * s + p.y * c)
}

fn bbox(poly: &[Vec2]) -> BBox {
    let mut b = BBox::empty();
    for &p in poly {
        b.include(p);
    }
    b
}

fn transform(poly: &[Vec2], ang: f64, t: Vec2) -> Vec<Vec2> {
    poly.iter().map(|&p| rotate(p, ang).add(t)).collect()
}

fn cross(a: Vec2, b: Vec2, c: Vec2) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Vec2, b: Vec2, p: Vec2, eps: f64) -> bool {
    (cross(a, b, p)).abs() <= eps
        && p.x >= a.x.min(b.x) - eps
        && p.x <= a.x.max(b.x) + eps
        && p.y >= a.y.min(b.y) - eps
        && p.y <= a.y.max(b.y) + eps
}

fn segments_intersect(a1: Vec2, a2: Vec2, b1: Vec2, b2: Vec2, eps: f64) -> bool {
    let d1 = cross(a1, a2, b1);
    let d2 = cross(a1, a2, b2);
    let d3 = cross(b1, b2, a1);
    let d4 = cross(b1, b2, a2);

    if (d1 > eps && d2 < -eps || d1 < -eps && d2 > eps) && (d3 > eps && d4 < -eps || d3 < -eps && d4 > eps) {
        return true;
    }

    if d1.abs() <= eps && on_segment(a1, a2, b1, eps) {
        return true;
    }
    if d2.abs() <= eps && on_segment(a1, a2, b2, eps) {
        return true;
    }
    if d3.abs() <= eps && on_segment(b1, b2, a1, eps) {
        return true;
    }
    if d4.abs() <= eps && on_segment(b1, b2, a2, eps) {
        return true;
    }
    false
}

fn point_in_poly(p: Vec2, poly: &[Vec2]) -> bool {
    // Ray casting; treats boundary as inside.
    let eps = 1e-9;
    let n = poly.len();
    if n < 3 {
        return false;
    }
    // Boundary check
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p, 1e-9) {
            return true;
        }
    }

    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let (ay, by) = (a.y, b.y);
        // Check if edge straddles the horizontal ray.
        let cond = (ay > p.y + eps) != (by > p.y + eps);
        if cond {
            let t = (p.y - ay) / (by - ay);
            let x_int = a.x + t * (b.x - a.x);
            if x_int >= p.x - eps {
                inside = !inside;
            }
        }
    }
    inside
}

fn polys_intersect(a: &[Vec2], b: &[Vec2]) -> bool {
    let eps = 1e-9;
    let na = a.len();
    let nb = b.len();
    if na < 3 || nb < 3 {
        return false;
    }
    for i in 0..na {
        let a1 = a[i];
        let a2 = a[(i + 1) % na];
        for j in 0..nb {
            let b1 = b[j];
            let b2 = b[(j + 1) % nb];
            if segments_intersect(a1, a2, b1, b2, eps) {
                return true;
            }
        }
    }
    // Containment
    if point_in_poly(a[0], b) {
        return true;
    }
    if point_in_poly(b[0], a) {
        return true;
    }
    false
}

fn poly_within(stock: &[Vec2], piece: &[Vec2]) -> bool {
    // Sufficient for simple polygons: all vertices inside and no edge crosses boundary.
    if piece.len() < 3 || stock.len() < 3 {
        return false;
    }
    for &p in piece {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    // Ensure no crossing between piece edges and stock boundary.
    // (If all vertices are inside, crossings can still occur for concave stock.)
    let eps = 1e-9;
    let ns = stock.len();
    let np = piece.len();
    for i in 0..np {
        let p1 = piece[i];
        let p2 = piece[(i + 1) % np];
        for j in 0..ns {
            let s1 = stock[j];
            let s2 = stock[(j + 1) % ns];
            // Intersection at boundary is allowed only if it is fully on boundary; treat any intersection as invalid.
            if segments_intersect(p1, p2, s1, s2, eps) {
                // Allow shared endpoints exactly on boundary? For robustness, disallow.
                return false;
            }
        }
    }
    true
}

#[derive(Clone)]
struct Placed {
    poly: Vec<Vec2>,
    bb: BBox,
}

#[derive(Clone)]
struct StockState {
    placed: Vec<Placed>,
    candidates: Vec<Vec2>,
}

fn gen_candidates(stock: &[Vec2], target_count: usize) -> Vec<Vec2> {
    let bb = bbox(stock);
    let w = bb.width().abs().max(1.0);
    let h = bb.height().abs().max(1.0);
    let max_dim = w.max(h);

    // grid resolution: adapt to target_count.
    let grid_n = (target_count as f64).sqrt().ceil() as i32;
    let grid_n = grid_n.clamp(6, 64);
    let step_x = w / (grid_n as f64);
    let step_y = h / (grid_n as f64);

    let mut pts = Vec::new();
    for iy in 0..=grid_n {
        let y = bb.min.y + (iy as f64) * step_y;
        for ix in 0..=grid_n {
            let x = bb.min.x + (ix as f64) * step_x;
            let p = Vec2::new(x, y);
            if point_in_poly(p, stock) {
                pts.push(p);
            }
        }
    }
    // Always include bbox min (if inside) and centroid-ish average.
    let mut c = Vec2::new(0.0, 0.0);
    for &p in stock {
        c.x += p.x;
        c.y += p.y;
    }
    c.x /= stock.len() as f64;
    c.y /= stock.len() as f64;
    if point_in_poly(c, stock) {
        pts.push(c);
    }
    pts
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Vec2::new(x, y));
    }

    let mut pieces: Vec<Vec<Vec2>> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Vec2::new(x, y));
        }
        pieces.push(poly);
    }

    // Sort pieces by descending bbox area to help packing, but keep original indices.
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&i, &j| {
        let bi = bbox(&pieces[i]);
        let bj = bbox(&pieces[j]);
        let ai = bi.width().abs() * bi.height().abs();
        let aj = bj.width().abs() * bj.height().abs();
        aj.partial_cmp(&ai).unwrap_or(std::cmp::Ordering::Equal)
    });

    let rotations: Vec<f64> = if np <= 50 {
        vec![0.0, 0.5 * PI, PI, 1.5 * PI, 0.25 * PI, -0.25 * PI]
    } else if np <= 500 {
        vec![0.0, 0.5 * PI, PI, 1.5 * PI]
    } else {
        vec![0.0]
    };

    let cand_target = if np <= 200 { 900 } else if np <= 1000 { 225 } else { 81 };

    let base_candidates = gen_candidates(&stock, cand_target);

    let mut stocks: Vec<StockState> = Vec::new();
    let mut placements: Vec<(usize, Vec2, f64)> = vec![(0, Vec2::new(0.0, 0.0), 0.0); np];

    for &pi in &order {
        let piece = &pieces[pi];
        let mut placed_ok = false;

        // Try existing stocks first.
        for (si, st) in stocks.iter_mut().enumerate() {
            if try_place_in_stock(&stock, piece, &rotations, st, &mut placements[pi], si) {
                placed_ok = true;
                break;
            }
        }

        if !placed_ok {
            // New stock.
            let mut st = StockState {
                placed: Vec::new(),
                candidates: base_candidates.clone(),
            };
            let new_si = stocks.len();
            if !try_place_in_stock(&stock, piece, &rotations, &mut st, &mut placements[pi], new_si) {
                // Last-resort: place at origin with 0 rotation; may be invalid but ensures output.
                placements[pi] = (new_si, Vec2::new(0.0, 0.0), 0.0);
            }
            stocks.push(st);
        }
    }

    // Output in original piece order.
    println!("{}", stocks.len().max(1));
    for i in 0..np {
        let (si, t, ang) = placements[i];
        println!("{} {:.10} {:.10} {:.10}", si, t.x, t.y, ang);
    }
}

fn try_place_in_stock(
    stock: &[Vec2],
    piece: &[Vec2],
    rotations: &[f64],
    st: &mut StockState,
    out: &mut (usize, Vec2, f64),
    stock_index: usize,
) -> bool {
    let gap = 1e-6;

    for &ang in rotations {
        let rot_piece: Vec<Vec2> = piece.iter().map(|&p| rotate(p, ang)).collect();
        let bb = bbox(&rot_piece);

        // Iterate candidates in given order.
        for &cand in &st.candidates {
            let t = Vec2::new(cand.x - bb.min.x + gap, cand.y - bb.min.y + gap);
            let placed_poly = rot_piece.iter().map(|&p| p.add(t)).collect::<Vec<_>>();
            if !poly_within(stock, &placed_poly) {
                continue;
            }
            let placed_bb = bbox(&placed_poly);

            // Overlap check (bbox then polygon).
            let mut overlap = false;
            for other in &st.placed {
                if !placed_bb.intersects(&other.bb) {
                    continue;
                }
                if polys_intersect(&placed_poly, &other.poly) {
                    overlap = true;
                    break;
                }
            }
            if overlap {
                continue;
            }

            // Accept.
            st.placed.push(Placed {
                bb: placed_bb,
                poly: placed_poly,
            });

            // Add a couple new candidate anchors near this bbox (simple bottom-left heuristic).
            let a = Vec2::new(placed_bb.max.x + gap, placed_bb.min.y);
            let b = Vec2::new(placed_bb.min.x, placed_bb.max.y + gap);
            if point_in_poly(a, stock) {
                st.candidates.push(a);
            }
            if point_in_poly(b, stock) {
                st.candidates.push(b);
            }

            *out = (stock_index, t, ang);
            return true;
        }
    }
    false
}
