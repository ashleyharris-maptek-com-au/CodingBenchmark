; SPIR-V
; Version: 1.0
; Vulkan: 1.0
; Bound: 400
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%glsl = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_3 = OpConstant %float 0.3
%float_0_015 = OpConstant %float 0.015
%float_eps = OpConstant %float 0.00000001

%v2_00 = OpConstantComposite %v2float %float_0 %float_0
%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %fn_void
%entry = OpLabel

; ptr to gl_out.gl_Position
%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; === Load triangle positions (0,2,4) ===
%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0ptr
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2ptr
%p4ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4ptr

; === Load and dim colors (0,2,4) ===
%c0ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0ptr
%c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %c2ptr
%c4ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %c4ptr

%c0d = OpVectorTimesScalar %v3float %c0 %float_0_3
%c2d = OpVectorTimesScalar %v3float %c2 %float_0_3
%c4d = OpVectorTimesScalar %v3float %c4 %float_0_3

; === Emit filled triangle (3 verts) ===
OpStore %outPosPtr %p0
OpStore %outColor %c0d
OpEmitVertex

OpStore %outPosPtr %p2
OpStore %outColor %c2d
OpEmitVertex

OpStore %outPosPtr %p4
OpStore %outColor %c4d
OpEmitVertex
OpEndPrimitive

; === Emit thick edge quad for (pA,pB): repeated 3 times ===

; -------- Edge 0-2 --------
%wA0 = OpCompositeExtract %float %p0 3
%wB0 = OpCompositeExtract %float %p2 3
%wA0v2 = OpCompositeConstruct %v2float %wA0 %wA0
%wB0v2 = OpCompositeConstruct %v2float %wB0 %wB0
%xyA0 = OpVectorShuffle %v2float %p0 %p0 0 1
%xyB0 = OpVectorShuffle %v2float %p2 %p2 0 1
%ndcA0 = OpFDiv %v2float %xyA0 %wA0v2
%ndcB0 = OpFDiv %v2float %xyB0 %wB0v2
%dir0 = OpFSub %v2float %ndcB0 %ndcA0
%dx0 = OpCompositeExtract %float %dir0 0
%dy0 = OpCompositeExtract %float %dir0 1
%ndy0 = OpFNegate %float %dy0
%perp0 = OpCompositeConstruct %v2float %ndy0 %dx0
%dot0 = OpDot %float %perp0 %perp0
%len0 = OpExtInst %float %glsl Sqrt %dot0
%len0s = OpFAdd %float %len0 %float_eps
%invLen0 = OpFDiv %float %float_1 %len0s
%nperp0 = OpVectorTimesScalar %v2float %perp0 %invLen0
%off0 = OpVectorTimesScalar %v2float %nperp0 %float_0_015

%off0x = OpCompositeExtract %float %off0 0
%off0y = OpCompositeExtract %float %off0 1

%a0x = OpCompositeExtract %float %ndcA0 0
%a0y = OpCompositeExtract %float %ndcA0 1
%b0x = OpCompositeExtract %float %ndcB0 0
%b0y = OpCompositeExtract %float %ndcB0 1

%a0px = OpFAdd %float %a0x %off0x
%a0py = OpFAdd %float %a0y %off0y
%a0mx = OpFSub %float %a0x %off0x
%a0my = OpFSub %float %a0y %off0y
%b0px = OpFAdd %float %b0x %off0x
%b0py = OpFAdd %float %b0y %off0y
%b0mx = OpFSub %float %b0x %off0x
%b0my = OpFSub %float %b0y %off0y

%zA0 = OpCompositeExtract %float %p0 2
%zB0 = OpCompositeExtract %float %p2 2

%a0px_c = OpFMul %float %a0px %wA0
%a0py_c = OpFMul %float %a0py %wA0
%a0mx_c = OpFMul %float %a0mx %wA0
%a0my_c = OpFMul %float %a0my %wA0
%b0px_c = OpFMul %float %b0px %wB0
%b0py_c = OpFMul %float %b0py %wB0
%b0mx_c = OpFMul %float %b0mx %wB0
%b0my_c = OpFMul %float %b0my %wB0
%a0p_clip = OpCompositeConstruct %v4float %a0px_c %a0py_c %zA0 %wA0
%a0m_clip = OpCompositeConstruct %v4float %a0mx_c %a0my_c %zA0 %wA0
%b0p_clip = OpCompositeConstruct %v4float %b0px_c %b0py_c %zB0 %wB0
%b0m_clip = OpCompositeConstruct %v4float %b0mx_c %b0my_c %zB0 %wB0

OpStore %outPosPtr %a0p_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %a0m_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %b0p_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %b0m_clip
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

; -------- Edge 2-4 --------
%wA1 = OpCompositeExtract %float %p2 3
%wB1 = OpCompositeExtract %float %p4 3
%wA1v2 = OpCompositeConstruct %v2float %wA1 %wA1
%wB1v2 = OpCompositeConstruct %v2float %wB1 %wB1
%xyA1 = OpVectorShuffle %v2float %p2 %p2 0 1
%xyB1 = OpVectorShuffle %v2float %p4 %p4 0 1
%ndcA1 = OpFDiv %v2float %xyA1 %wA1v2
%ndcB1 = OpFDiv %v2float %xyB1 %wB1v2
%dir1 = OpFSub %v2float %ndcB1 %ndcA1
%dx1 = OpCompositeExtract %float %dir1 0
%dy1 = OpCompositeExtract %float %dir1 1
%ndy1 = OpFNegate %float %dy1
%perp1 = OpCompositeConstruct %v2float %ndy1 %dx1
%dot1 = OpDot %float %perp1 %perp1
%len1 = OpExtInst %float %glsl Sqrt %dot1
%len1s = OpFAdd %float %len1 %float_eps
%invLen1 = OpFDiv %float %float_1 %len1s
%nperp1 = OpVectorTimesScalar %v2float %perp1 %invLen1
%off1 = OpVectorTimesScalar %v2float %nperp1 %float_0_015

%off1x = OpCompositeExtract %float %off1 0
%off1y = OpCompositeExtract %float %off1 1

%a1x = OpCompositeExtract %float %ndcA1 0
%a1y = OpCompositeExtract %float %ndcA1 1
%b1x = OpCompositeExtract %float %ndcB1 0
%b1y = OpCompositeExtract %float %ndcB1 1

%a1px = OpFAdd %float %a1x %off1x
%a1py = OpFAdd %float %a1y %off1y
%a1mx = OpFSub %float %a1x %off1x
%a1my = OpFSub %float %a1y %off1y
%b1px = OpFAdd %float %b1x %off1x
%b1py = OpFAdd %float %b1y %off1y
%b1mx = OpFSub %float %b1x %off1x
%b1my = OpFSub %float %b1y %off1y

%zA1 = OpCompositeExtract %float %p2 2
%zB1 = OpCompositeExtract %float %p4 2

%a1px_c = OpFMul %float %a1px %wA1
%a1py_c = OpFMul %float %a1py %wA1
%a1mx_c = OpFMul %float %a1mx %wA1
%a1my_c = OpFMul %float %a1my %wA1
%b1px_c = OpFMul %float %b1px %wB1
%b1py_c = OpFMul %float %b1py %wB1
%b1mx_c = OpFMul %float %b1mx %wB1
%b1my_c = OpFMul %float %b1my %wB1
%a1p_clip = OpCompositeConstruct %v4float %a1px_c %a1py_c %zA1 %wA1
%a1m_clip = OpCompositeConstruct %v4float %a1mx_c %a1my_c %zA1 %wA1
%b1p_clip = OpCompositeConstruct %v4float %b1px_c %b1py_c %zB1 %wB1
%b1m_clip = OpCompositeConstruct %v4float %b1mx_c %b1my_c %zB1 %wB1

OpStore %outPosPtr %a1p_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %a1m_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %b1p_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %b1m_clip
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

; -------- Edge 4-0 --------
%wA2 = OpCompositeExtract %float %p4 3
%wB2 = OpCompositeExtract %float %p0 3
%wA2v2 = OpCompositeConstruct %v2float %wA2 %wA2
%wB2v2 = OpCompositeConstruct %v2float %wB2 %wB2
%xyA2 = OpVectorShuffle %v2float %p4 %p4 0 1
%xyB2 = OpVectorShuffle %v2float %p0 %p0 0 1
%ndcA2 = OpFDiv %v2float %xyA2 %wA2v2
%ndcB2 = OpFDiv %v2float %xyB2 %wB2v2
%dir2 = OpFSub %v2float %ndcB2 %ndcA2
%dx2 = OpCompositeExtract %float %dir2 0
%dy2 = OpCompositeExtract %float %dir2 1
%ndy2 = OpFNegate %float %dy2
%perp2 = OpCompositeConstruct %v2float %ndy2 %dx2
%dot2 = OpDot %float %perp2 %perp2
%len2 = OpExtInst %float %glsl Sqrt %dot2
%len2s = OpFAdd %float %len2 %float_eps
%invLen2 = OpFDiv %float %float_1 %len2s
%nperp2 = OpVectorTimesScalar %v2float %perp2 %invLen2
%off2 = OpVectorTimesScalar %v2float %nperp2 %float_0_015

%off2x = OpCompositeExtract %float %off2 0
%off2y = OpCompositeExtract %float %off2 1

%a2x = OpCompositeExtract %float %ndcA2 0
%a2y = OpCompositeExtract %float %ndcA2 1
%b2x = OpCompositeExtract %float %ndcB2 0
%b2y = OpCompositeExtract %float %ndcB2 1

%a2px = OpFAdd %float %a2x %off2x
%a2py = OpFAdd %float %a2y %off2y
%a2mx = OpFSub %float %a2x %off2x
%a2my = OpFSub %float %a2y %off2y
%b2px = OpFAdd %float %b2x %off2x
%b2py = OpFAdd %float %b2y %off2y
%b2mx = OpFSub %float %b2x %off2x
%b2my = OpFSub %float %b2y %off2y

%zA2 = OpCompositeExtract %float %p4 2
%zB2 = OpCompositeExtract %float %p0 2

%a2px_c = OpFMul %float %a2px %wA2
%a2py_c = OpFMul %float %a2py %wA2
%a2mx_c = OpFMul %float %a2mx %wA2
%a2my_c = OpFMul %float %a2my %wA2
%b2px_c = OpFMul %float %b2px %wB2
%b2py_c = OpFMul %float %b2py %wB2
%b2mx_c = OpFMul %float %b2mx %wB2
%b2my_c = OpFMul %float %b2my %wB2
%a2p_clip = OpCompositeConstruct %v4float %a2px_c %a2py_c %zA2 %wA2
%a2m_clip = OpCompositeConstruct %v4float %a2mx_c %a2my_c %zA2 %wA2
%b2p_clip = OpCompositeConstruct %v4float %b2px_c %b2py_c %zB2 %wB2
%b2m_clip = OpCompositeConstruct %v4float %b2mx_c %b2my_c %zB2 %wB2

OpStore %outPosPtr %a2p_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %a2m_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %b2p_clip
OpStore %outColor %v3_white
OpEmitVertex
OpStore %outPosPtr %b2m_clip
OpStore %outColor %v3_white
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
