use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Rng(u64);
impl Rng {
    fn new(seed: u64) -> Self {
        Self(seed ^ 0x9E3779B97F4A7C15)
    }
    fn next_u64(&mut self) -> u64 {
        // xorshift64*
        let mut x = self.0;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.0 = x;
        x.wrapping_mul(2685821657736338717)
    }
    fn next_f64(&mut self) -> f64 {
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1u64 << 53) as f64))
    }
    fn gen_usize(&mut self, n: usize) -> usize {
        if n <= 1 {
            0
        } else {
            (self.next_u64() as usize) % n
        }
    }
}

fn main() {
    // -------- Read input --------
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m_in: usize = it.next().unwrap().parse().unwrap();

    // First pass: read edges into arrays, compute degrees.
    let mut us: Vec<u32> = Vec::with_capacity(m_in);
    let mut vs: Vec<u32> = Vec::with_capacity(m_in);
    let mut deg = vec![0u32; n];
    for _ in 0..m_in {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // Self-loop: ignore for layout forces/crossings; still "graceful".
            continue;
        }
        us.push(u as u32);
        vs.push(v as u32);
        deg[u] = deg[u].saturating_add(1);
        deg[v] = deg[v].saturating_add(1);
    }
    let m = us.len();

    // Build CSR adjacency.
    let mut offsets = vec![0u32; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i];
    }
    let mut cur = offsets.clone();
    let mut adj = vec![0u32; offsets[n] as usize];
    for i in 0..m {
        let u = us[i] as usize;
        let v = vs[i] as usize;
        let pu = cur[u] as usize;
        adj[pu] = v as u32;
        cur[u] += 1;
        let pv = cur[v] as usize;
        adj[pv] = u as u32;
        cur[v] += 1;
    }

    // -------- Connected components --------
    let mut comp_id = vec![u32::MAX; n];
    let mut comps: Vec<Vec<u32>> = Vec::new();
    let mut stack: Vec<u32> = Vec::new();
    for s in 0..n {
        if comp_id[s] != u32::MAX {
            continue;
        }
        let cid = comps.len() as u32;
        let mut nodes: Vec<u32> = Vec::new();
        comp_id[s] = cid;
        stack.push(s as u32);
        while let Some(x) = stack.pop() {
            nodes.push(x);
            let xu = x as usize;
            let a = offsets[xu] as usize;
            let b = offsets[xu + 1] as usize;
            for &y in &adj[a..b] {
                let yu = y as usize;
                if comp_id[yu] == u32::MAX {
                    comp_id[yu] = cid;
                    stack.push(y);
                }
            }
        }
        comps.push(nodes);
    }

    // -------- Per-component layout --------
    let mut x = vec![0f64; n];
    let mut y = vec![0f64; n];

    // Component placement tiling.
    let mut cursor_x = 0.0f64;
    let mut cursor_y = 0.0f64;
    let mut row_h = 0.0f64;
    let target_row_w = 2000.0f64;

    // Deterministic-ish seed from graph summary.
    let mut rng = Rng::new((n as u64).wrapping_mul(1_000_003) ^ (m as u64).wrapping_mul(1_000_033));

    // Sort components by size (largest first) for nicer packing.
    let mut comp_order: Vec<usize> = (0..comps.len()).collect();
    comp_order.sort_by(|&a, &b| comps[b].len().cmp(&comps[a].len()));

    for &ci in &comp_order {
        let nodes = &comps[ci];
        let sz = nodes.len();

        // Choose layout mode.
        // - Tiny/Small: force-directed (exact repulsion for <=800, sampled for <=2000)
        // - Large: layered BFS tree-ish
        // - Huge: simple circle by degree ordering
        let (mut local_x, mut local_y, _w, _h) = if sz <= 2000 {
            layout_force_directed(nodes, n, &offsets, &adj, &deg, &mut rng)
        } else if sz <= 20000 {
            layout_bfs_layered(nodes, n, &offsets, &adj, &deg)
        } else {
            layout_circle(nodes, n, &deg)
        };

        // Normalize to positive bbox with margin.
        let (minx, miny, maxx, maxy) = bbox(&local_x, &local_y);
        let margin = 5.0;
        for i in 0..sz {
            local_x[i] = local_x[i] - minx + margin;
            local_y[i] = local_y[i] - miny + margin;
        }
        let ww = (maxx - minx) + 2.0 * margin;
        let hh = (maxy - miny) + 2.0 * margin;

        // Pack components on a loose grid.
        if cursor_x + ww > target_row_w && cursor_x > 0.0 {
            cursor_x = 0.0;
            cursor_y += row_h + 20.0;
            row_h = 0.0;
        }
        row_h = row_h.max(hh);

        for (i, &v) in nodes.iter().enumerate() {
            let vi = v as usize;
            x[vi] = local_x[i] + cursor_x;
            y[vi] = local_y[i] + cursor_y;
        }
        cursor_x += ww + 20.0;
    }

    // -------- Output --------
    // Optional N prefix not needed; output N lines.
    let mut out = String::with_capacity(n * 32);
    for i in 0..n {
        out.push_str(&format!("{:.6} {:.6}\n", x[i], y[i]));
    }
    print!("{out}");
}

fn bbox(xs: &[f64], ys: &[f64]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for i in 0..xs.len() {
        minx = minx.min(xs[i]);
        miny = miny.min(ys[i]);
        maxx = maxx.max(xs[i]);
        maxy = maxy.max(ys[i]);
    }
    (minx, miny, maxx, maxy)
}

fn layout_circle(nodes: &[u32], _n: usize, deg: &[u32]) -> (Vec<f64>, Vec<f64>, f64, f64) {
    let sz = nodes.len();
    let mut idx: Vec<usize> = (0..sz).collect();
    idx.sort_by(|&i, &j| {
        let a = deg[nodes[i] as usize];
        let b = deg[nodes[j] as usize];
        b.cmp(&a).then_with(|| (nodes[i]).cmp(&nodes[j]))
    });
    let r = (sz as f64).sqrt().max(10.0);
    let mut xs = vec![0.0; sz];
    let mut ys = vec![0.0; sz];
    for (k, &ii) in idx.iter().enumerate() {
        let t = 2.0 * PI * (k as f64) / (sz.max(1) as f64);
        xs[ii] = r * t.cos();
        ys[ii] = r * t.sin();
    }
    (xs, ys, 2.0 * r, 2.0 * r)
}

fn layout_bfs_layered(
    nodes: &[u32],
    n: usize,
    offsets: &[u32],
    adj: &[u32],
    deg: &[u32],
) -> (Vec<f64>, Vec<f64>, f64, f64) {
    let sz = nodes.len();
    let mut pos_in_comp = vec![u32::MAX; n];
    for (i, &v) in nodes.iter().enumerate() {
        pos_in_comp[v as usize] = i as u32;
    }
    // Choose root as max degree node for a wider tree.
    let mut root = nodes[0];
    let mut bestd = deg[root as usize];
    for &v in nodes {
        let d = deg[v as usize];
        if d > bestd {
            bestd = d;
            root = v;
        }
    }

    let mut level = vec![i32::MIN; sz];
    let mut q = VecDequeU32::with_capacity(sz);
    let rpos = pos_in_comp[root as usize] as usize;
    level[rpos] = 0;
    q.push(root);
    while let Some(x) = q.pop() {
        let xp = pos_in_comp[x as usize] as usize;
        let xl = level[xp];
        let a = offsets[x as usize] as usize;
        let b = offsets[x as usize + 1] as usize;
        for &y in &adj[a..b] {
            let yp0 = pos_in_comp[y as usize];
            if yp0 == u32::MAX {
                continue;
            }
            let yp = yp0 as usize;
            if level[yp] == i32::MIN {
                level[yp] = xl + 1;
                q.push(y);
            }
        }
    }

    // Any unreachable (shouldn't happen in a component) set to 0.
    for l in &mut level {
        if *l == i32::MIN {
            *l = 0;
        }
    }
    let max_level = *level.iter().max().unwrap_or(&0);
    let mut buckets: Vec<Vec<usize>> = vec![Vec::new(); (max_level + 1).max(1) as usize];
    for i in 0..sz {
        let li = level[i].max(0) as usize;
        buckets[li].push(i);
    }
    // Stable-ish order: by original node id to reduce edge criss-cross a bit.
    for b in &mut buckets {
        b.sort_by(|&i, &j| nodes[i].cmp(&nodes[j]));
    }

    let mut xs = vec![0.0; sz];
    let mut ys = vec![0.0; sz];
    let ygap = 3.0;
    let xgap = 2.0;
    let mut widest = 1usize;
    for (li, b) in buckets.iter().enumerate() {
        widest = widest.max(b.len());
        for (k, &i) in b.iter().enumerate() {
            xs[i] = (k as f64) * xgap;
            ys[i] = (li as f64) * ygap;
        }
    }
    (
        xs,
        ys,
        (widest as f64) * xgap,
        ((max_level + 1) as f64) * ygap,
    )
}

fn layout_force_directed(
    nodes: &[u32],
    n: usize,
    offsets: &[u32],
    adj: &[u32],
    _deg: &[u32],
    rng: &mut Rng,
) -> (Vec<f64>, Vec<f64>, f64, f64) {
    let sz = nodes.len();
    let mut pos_in_comp = vec![u32::MAX; n];
    for (i, &v) in nodes.iter().enumerate() {
        pos_in_comp[v as usize] = i as u32;
    }

    // Initialize on noisy circle.
    let r = (sz as f64).sqrt().max(2.0);
    let mut xs = vec![0.0; sz];
    let mut ys = vec![0.0; sz];
    for i in 0..sz {
        let t = 2.0 * PI * (i as f64) / (sz.max(1) as f64);
        let jitter = (rng.next_f64() - 0.5) * 0.1;
        xs[i] = (r + jitter) * t.cos();
        ys[i] = (r + jitter) * t.sin();
    }

    // Build local edge list for this component to avoid scanning all edges repeatedly.
    let mut edges: Vec<(u32, u32)> = Vec::new();
    edges.reserve(sz.saturating_mul(2));
    for (i, &v) in nodes.iter().enumerate() {
        let a = offsets[v as usize] as usize;
        let b = offsets[v as usize + 1] as usize;
        for &to in &adj[a..b] {
            let j0 = pos_in_comp[to as usize];
            if j0 == u32::MAX {
                continue;
            }
            let j = j0 as usize;
            if i < j {
                edges.push((i as u32, j as u32));
            }
        }
    }

    // FR params.
    let area = (sz as f64).max(1.0);
    let k = (area / (sz as f64)).sqrt().max(0.01);
    let iterations = if sz <= 100 { 80 } else if sz <= 500 { 60 } else { 40 };
    let mut t = (area.sqrt()).max(1.0);
    let cool = 0.92;
    let eps = 1e-9;

    let mut fx = vec![0.0f64; sz];
    let mut fy = vec![0.0f64; sz];
    let exact_repulsion = sz <= 800;
    let samples = if sz <= 1200 { 24 } else { 16 };

    for _ in 0..iterations {
        fx.fill(0.0);
        fy.fill(0.0);

        // Repulsive forces.
        if exact_repulsion {
            for i in 0..sz {
                for j in (i + 1)..sz {
                    let dx = xs[i] - xs[j];
                    let dy = ys[i] - ys[j];
                    let dist2 = dx * dx + dy * dy + eps;
                    let dist = dist2.sqrt();
                    let force = (k * k) / dist;
                    let ux = dx / dist;
                    let uy = dy / dist;
                    fx[i] += ux * force;
                    fy[i] += uy * force;
                    fx[j] -= ux * force;
                    fy[j] -= uy * force;
                }
            }
        } else {
            for i in 0..sz {
                for _ in 0..samples {
                    let j = rng.gen_usize(sz);
                    if j == i {
                        continue;
                    }
                    let dx = xs[i] - xs[j];
                    let dy = ys[i] - ys[j];
                    let dist2 = dx * dx + dy * dy + eps;
                    let dist = dist2.sqrt();
                    let force = (k * k) / dist;
                    fx[i] += (dx / dist) * force;
                    fy[i] += (dy / dist) * force;
                }
            }
        }

        // Attractive forces along edges.
        for &(a, b) in &edges {
            let i = a as usize;
            let j = b as usize;
            let dx = xs[i] - xs[j];
            let dy = ys[i] - ys[j];
            let dist2 = dx * dx + dy * dy + eps;
            let dist = dist2.sqrt();
            let force = (dist2) / k;
            let ux = dx / dist;
            let uy = dy / dist;
            fx[i] -= ux * force;
            fy[i] -= uy * force;
            fx[j] += ux * force;
            fy[j] += uy * force;
        }

        // Update positions with temperature cap.
        for i in 0..sz {
            let dx = fx[i];
            let dy = fy[i];
            let disp = (dx * dx + dy * dy).sqrt();
            if disp > eps {
                let step = disp.min(t);
                xs[i] += dx / disp * step;
                ys[i] += dy / disp * step;
            }
        }
        t *= cool;
    }

    let (minx, miny, maxx, maxy) = bbox(&xs, &ys);
    (xs, ys, maxx - minx, maxy - miny)
}

// Minimal deque for u32 to avoid std::collections overhead.
struct VecDequeU32 {
    buf: Vec<u32>,
    head: usize,
}
impl VecDequeU32 {
    fn with_capacity(cap: usize) -> Self {
        Self {
            buf: Vec::with_capacity(cap),
            head: 0,
        }
    }
    fn push(&mut self, v: u32) {
        self.buf.push(v);
    }
    fn pop(&mut self) -> Option<u32> {
        if self.head >= self.buf.len() {
            return None;
        }
        let v = self.buf[self.head];
        self.head += 1;
        // Occasionally compact.
        if self.head > 4096 && self.head * 2 > self.buf.len() {
            self.buf.drain(0..self.head);
            self.head = 0;
        }
        Some(v)
    }
}
