#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (idx >= N) return;

    uint base = idx * 2u;
    vec3 pos = inData[base + 0u].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 sepAcc = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == idx) continue;

        uint jbase = j * 2u;
        vec3 npos = inData[jbase + 0u].xyz;
        vec3 nvel = inData[jbase + 1u].xyz;

        vec3 d = npos - pos;
        float dist2 = dot(d, d);
        if (dist2 <= 0.0) continue;

        float dist = sqrt(dist2);

        if (dist < sepRadius) {
            // Push away; weight nearby neighbors more strongly.
            vec3 away = -d / (dist2 + 1e-6);
            sepAcc += away;
            sepCount++;
        }

        if (dist < alignRadius) {
            alignSum += nvel;
            alignCount++;
        }

        if (dist < cohRadius) {
            cohSum += npos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        vec3 sepAvg = sepAcc / float(sepCount);
        accel += sepAvg * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 alignAvgVel = alignSum / float(alignCount);
        vec3 alignSteer = alignAvgVel - vel;
        accel += alignSteer * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 cohCenter = cohSum / float(cohCount);
        vec3 cohSteer = cohCenter - pos;
        accel += cohSteer * cohWeight;
    }

    // Boundary force: if a component exceeds boundSize*0.4, push back toward center.
    float thresh = boundSize * 0.4;
    float scale = boundForce / max(boundSize * 0.5, 1e-6);
    if (abs(pos.x) > thresh) accel.x += (-pos.x) * scale;
    if (abs(pos.y) > thresh) accel.y += (-pos.y) * scale;
    if (abs(pos.z) > thresh) accel.z += (-pos.z) * scale;

    vel += accel * dt;
    float speed2 = dot(vel, vel);
    if (speed2 > maxSpeed * maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    pos += vel * dt;

    outData[base + 0u] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
