use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct HeapItem {
    gain: i32,
    v: usize,
}

impl Eq for HeapItem {}

impl PartialEq for HeapItem {
    fn eq(&self, other: &Self) -> bool {
        self.gain == other.gain && self.v == other.v
    }
}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        // Max-heap by gain, then by vertex id (stable-ish).
        self.gain
            .cmp(&other.gain)
            .then_with(|| other.v.cmp(&self.v))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn main() {
    // Read entire stdin.
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        if u == v || u >= n || v >= n {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    // Optional: remove parallel edges (helps quality a bit; keeps correctness).
    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }

    // undominated[v] == true means v is not yet dominated by chosen set.
    let mut undominated = vec![true; n];

    // gain[v] = number of currently undominated vertices in N[v] ∪ {v}.
    let mut gain: Vec<i32> = vec![0; n];
    for v in 0..n {
        gain[v] = (adj[v].len() as i32) + 1;
    }

    let mut heap = BinaryHeap::with_capacity(n * 2);
    for v in 0..n {
        heap.push(HeapItem { gain: gain[v], v });
    }

    let mut remaining_undominated = n;
    let mut chosen: Vec<usize> = Vec::new();

    // Helper to "remove" an undominated vertex x: mark it dominated and decrement
    // gain for every y that could dominate x (i.e., y in N[x] ∪ {x}).
    let remove_vertex = |x: usize,
                         undominated: &mut [bool],
                         gain: &mut [i32],
                         heap: &mut BinaryHeap<HeapItem>,
                         remaining: &mut usize| {
        if !undominated[x] {
            return;
        }
        undominated[x] = false;
        *remaining -= 1;

        // y = x itself
        if gain[x] > 0 {
            gain[x] -= 1;
            heap.push(HeapItem { gain: gain[x], v: x });
        }
        // y in neighbors
        for &y in &adj[x] {
            if gain[y] > 0 {
                gain[y] -= 1;
                heap.push(HeapItem { gain: gain[y], v: y });
            }
        }
    };

    while remaining_undominated > 0 {
        // Extract best vertex by current gain (lazy updates).
        let v = loop {
            match heap.pop() {
                Some(item) => {
                    let cur = gain[item.v];
                    if item.gain != cur {
                        continue;
                    }
                    if cur <= 0 {
                        // Should not happen if remaining_undominated > 0, but safe fallback:
                        // pick any still-undominated vertex and add it.
                        let mut fallback = None;
                        for i in 0..n {
                            if undominated[i] {
                                fallback = Some(i);
                                break;
                            }
                        }
                        break fallback.unwrap_or(0);
                    }
                    break item.v;
                }
                None => {
                    // Heap exhausted (shouldn't happen). Fallback to any undominated vertex.
                    let mut fallback = 0;
                    for i in 0..n {
                        if undominated[i] {
                            fallback = i;
                            break;
                        }
                    }
                    break fallback;
                }
            }
        };

        chosen.push(v);

        // Dominate v and its neighbors (remove undominated vertices in closed neighborhood).
        remove_vertex(v, &mut undominated, &mut gain, &mut heap, &mut remaining_undominated);
        for &x in &adj[v] {
            remove_vertex(
                x,
                &mut undominated,
                &mut gain,
                &mut heap,
                &mut remaining_undominated,
            );
        }
    }

    // Output.
    println!("{}", chosen.len());
    if !chosen.is_empty() {
        for (i, v) in chosen.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
    }
    println!();
}
