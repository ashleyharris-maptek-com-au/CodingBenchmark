use std::io::{self, Read, Write};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn skip_whitespace(&mut self) {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_u32(&mut self) -> Option<u32> {
        self.skip_whitespace();
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut x: u32 = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            x = x.saturating_mul(10).saturating_add((b - b'0') as u32);
            self.idx += 1;
        }
        Some(x)
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_u32() {
        Some(v) => v as usize,
        None => return,
    };
    let m = match input.next_u32() {
        Some(v) => v as usize,
        None => return,
    };

    let mut in_cover = vec![false; n];

    for _ in 0..m {
        let u = match input.next_u32() {
            Some(v) => v as usize,
            None => break,
        };
        let v = match input.next_u32() {
            Some(v) => v as usize,
            None => break,
        };

        if u >= n || v >= n {
            continue;
        }

        if u == v {
            if !in_cover[u] {
                in_cover[u] = true;
            }
            continue;
        }

        if !in_cover[u] && !in_cover[v] {
            in_cover[u] = true;
            in_cover[v] = true;
        }
    }

    let mut cover_vertices: Vec<u32> = Vec::new();
    cover_vertices.reserve(n / 8 + 1);
    for (i, &chosen) in in_cover.iter().enumerate() {
        if chosen {
            cover_vertices.push(i as u32);
        }
    }

    let mut out = Vec::with_capacity(32 + cover_vertices.len() * 6);
    out.extend_from_slice(cover_vertices.len().to_string().as_bytes());
    out.push(b'\n');
    for (i, v) in cover_vertices.iter().enumerate() {
        if i > 0 {
            out.push(b' ');
        }
        out.extend_from_slice(v.to_string().as_bytes());
    }
    out.push(b'\n');

    io::stdout().write_all(&out).unwrap();
}
