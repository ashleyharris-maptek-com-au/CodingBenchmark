using System;using System.Collections.Generic;using System.IO;using System.Text;public static class Program{    private struct Edge    {        public int To;        public int Id;        public Edge(int to, int id) { To = to; Id = id; }    }    // Fast scanner for big inputs    private sealed class FastScanner    {        private readonly Stream _s;        private readonly byte[] _buf = new byte[1 << 20];        private int _len, _ptr;        public FastScanner(Stream s) { _s = s; }        private int ReadByte()        {            if (_ptr >= _len)            {                _len = _s.Read(_buf, 0, _buf.Length);                _ptr = 0;                if (_len <= 0) return -1;            }            return _buf[_ptr++];        }        public int NextInt()        {            int c;            do { c = ReadByte(); } while (c <= 32 && c >= 0);            int sign = 1;            if (c == '-')            {                sign = -1;                c = ReadByte();            }            int x = 0;            while (c > 32 && c >= 0)            {                x = x * 10 + (c - '0');                c = ReadByte();            }            return x * sign;        }    }    public static void Main()    {        var fs = new FastScanner(Console.OpenStandardInput());        int n;        try        {            n = fs.NextInt();        }        catch        {            return;        }        int m = fs.NextInt();        var g = new List<Edge>[n];        for (int i = 0; i < n; i++) g[i] = new List<Edge>();        // Read edges (assume undirected). We ignore weights for feasibility route.        for (int id = 0; id < m; id++)        {            int u = fs.NextInt();            int v = fs.NextInt();            _ = fs.NextInt(); // weight            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;            g[u].Add(new Edge(v, id));            g[v].Add(new Edge(u, id));        }        // Build a spanning tree rooted at 0 (iterative DFS to avoid recursion limits).        var parent = new int[n];        var parentEdge = new int[n];        Array.Fill(parent, -1);        Array.Fill(parentEdge, -1);        var stack = new int[n];        int sp = 0;        parent[0] = 0;        stack[sp++] = 0;        while (sp > 0)        {            int u = stack[--sp];            var adj = g[u];            for (int i = 0; i < adj.Count; i++)            {                int v = adj[i].To;                if (parent[v] != -1) continue;                parent[v] = u;                parentEdge[v] = adj[i].Id;                stack[sp++] = v;            }        }        // Route construction:        // - Traverse the spanning tree with an explicit stack (Euler tour of tree).        // - At each vertex, "inspect" all non-tree incident edges by going to the neighbor and immediately back.        // This produces a closed walk starting/ending at 0 and traverses every edge at least once (indeed, twice).        var edgeDone = new bool[m];        var edgeIsTree = new bool[m];        for (int i = 1; i < n; i++)        {            int pe = parentEdge[i];            if (pe >= 0) edgeIsTree[pe] = true;        }        // Iteration pointers per vertex so we scan each adjacency list once overall.        var it = new int[n];        var route = new List<int>(Math.Max(4, 2 * m + 2 * n + 16));        route.Add(0);        // Build children lists for the spanning tree.        var children = new List<int>[n];        for (int i = 0; i < n; i++) children[i] = new List<int>();        for (int v = 1; v < n; v++)        {            int p = parent[v];            if (p >= 0 && p != v) children[p].Add(v);        }        // Iterative DFS over the spanning tree using explicit frames (u, nextChildIndex, enteredFlag).        var stU = new int[Math.Max(1, n)];        var stI = new int[stU.Length];        var stEntered = new bool[stU.Length];        int stp = 0;        stU[stp] = 0;        stI[stp] = 0;        stEntered[stp] = false;        stp++;        while (stp > 0)        {            int top = stp - 1;            int u = stU[top];            if (!stEntered[top])            {                stEntered[top] = true;                // On entering u: process all non-tree edges incident to u (each once, globally).                var adj = g[u];                int idx = it[u];                while (idx < adj.Count)                {                    var e = adj[idx++];                    int eid = e.Id;                    if ((uint)eid >= (uint)m) continue;                    if (edgeIsTree[eid]) continue;                    if (edgeDone[eid]) continue;                    edgeDone[eid] = true;                    int v = e.To;                    route.Add(v);                    route.Add(u);                }                it[u] = idx;            }            int ci = stI[top];            if (ci < children[u].Count)            {                int v = children[u][ci];                stI[top] = ci + 1;                // Move u -> v                route.Add(v);                // Push child frame                stU[stp] = v;                stI[stp] = 0;                stEntered[stp] = false;                stp++;            }            else            {                // Done with u; pop and backtrack to parent                stp--;                if (u != 0)                {                    int p = parent[u];                    route.Add(p);                    int pe = parentEdge[u];                    if ((uint)pe < (uint)m) edgeDone[pe] = true;                }            }        }        // Ensure we end at 0 (tree traversal should already return to 0).        if (route.Count == 0 || route[route.Count - 1] != 0) route.Add(0);        // Output just the node sequence (length prefix is optional; omit for speed/simplicity).        var sb = new StringBuilder(route.Count * 2);        for (int i = 0; i < route.Count; i++)        {            if (i > 0) sb.Append(' ');            sb.Append(route[i]);        }        Console.Out.Write(sb.ToString());    }}
