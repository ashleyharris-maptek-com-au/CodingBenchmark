use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn add(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x + o.x,
            y: self.y + o.y,
            z: self.z + o.z,
        }
    }
    fn mul(self, s: f64) -> Vec3 {
        Vec3 {
            x: self.x * s,
            y: self.y * s,
            z: self.z * s,
        }
    }
    fn norm(self) -> f64 {
        (self.x * self.x + self.y * self.y + self.z * self.z).sqrt()
    }
    fn unit_or(self, fallback: Vec3) -> Vec3 {
        let n = self.norm();
        if n > 0.0 {
            self.mul(1.0 / n)
        } else {
            fallback
        }
    }
    fn dist2(self, o: Vec3) -> f64 {
        let dx = self.x - o.x;
        let dy = self.y - o.y;
        let dz = self.z - o.z;
        dx * dx + dy * dy + dz * dz
    }
}

#[derive(Clone, Debug)]
struct Sample {
    depth: f64,
    // None means missing/invalid.
    props: Vec<Option<f64>>,
    pos: Vec3,
}

#[derive(Clone, Debug)]
struct Hole {
    id: String,
    samples: Vec<Sample>,
}

#[derive(Clone, Debug)]
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_name: String,
    confidence: f64,
}

fn parse_f64_opt(tok: &str) -> Option<f64> {
    let t = tok.trim();
    if t.is_empty() {
        return None;
    }
    let u = t.to_ascii_lowercase();
    if u == "na" || u == "nan" || u == "null" || u == "none" || u == "." {
        return None;
    }
    t.parse::<f64>().ok().filter(|v| v.is_finite())
}

fn median_in_place(values: &mut [f64]) -> f64 {
    let n = values.len();
    if n == 0 {
        return 0.0;
    }
    let mid = n / 2;
    values.select_nth_unstable_by(mid, |a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let m = values[mid];
    if n % 2 == 1 {
        m
    } else {
        // Need max of lower half.
        let mut max_lo = values[0];
        for &v in &values[..mid] {
            if v > max_lo {
                max_lo = v;
            }
        }
        0.5 * (max_lo + m)
    }
}

fn mad_scale(values: &mut [f64], med: f64) -> f64 {
    // Median absolute deviation scaled to stddev-ish.
    for v in values.iter_mut() {
        *v = (*v - med).abs();
    }
    let mad = median_in_place(values);
    // 1.4826 scales MAD to match stddev for normal distribution.
    mad * 1.4826
}

// Uniform grid hashing for 3D. Key packed into i128 to reduce HashMap overhead.
fn grid_key(ix: i32, iy: i32, iz: i32) -> i128 {
    // Pack 3 signed 32-bit into one 96-bit value.
    ((ix as i128) << 64) ^ ((iy as i128) << 32) ^ (iz as i128)
}

fn main() {
    // Read all input.
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let num_holes: usize = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => {
            return;
        }
    };
    let num_props: usize = it.next().and_then(|v| v.parse().ok()).unwrap_or(0);
    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(p) = it.next() {
            prop_names.push(p.to_string());
        } else {
            prop_names.push(format!("prop{}", prop_names.len() + 1));
        }
    }

    let mut holes: Vec<Hole> = Vec::with_capacity(num_holes);
    let mut total_samples: usize = 0;
    let mut total_values: usize = 0;
    let mut all_vals_by_prop: Vec<Vec<f64>> = vec![Vec::new(); num_props];

    for _ in 0..num_holes {
        let hole_id = match it.next() {
            Some(v) => v.to_string(),
            None => break,
        };
        let sx = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let sy = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let sz = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let dx = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let dy = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let dz = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(1.0);
        let _length = it.next().and_then(|v| v.parse::<f64>().ok()).unwrap_or(0.0);
        let num_samples = it.next().and_then(|v| v.parse::<usize>().ok()).unwrap_or(0);

        let start = Vec3 { x: sx, y: sy, z: sz };
        let dir = Vec3 { x: dx, y: dy, z: dz }.unit_or(Vec3 { x: 0.0, y: 0.0, z: 1.0 });

        let mut samples: Vec<Sample> = Vec::with_capacity(num_samples);
        for _ in 0..num_samples {
            let depth = match it.next() {
                Some(v) => v.parse::<f64>().unwrap_or(0.0),
                None => break,
            };
            let mut props: Vec<Option<f64>> = Vec::with_capacity(num_props);
            for pi in 0..num_props {
                let tok = it.next().unwrap_or("NA");
                let v = parse_f64_opt(tok);
                if let Some(x) = v {
                    all_vals_by_prop[pi].push(x);
                    total_values += 1;
                }
                props.push(v);
            }
            let pos = start.add(dir.mul(depth));
            samples.push(Sample { depth, props, pos });
        }
        total_samples += samples.len();
        // Sort by depth to handle duplicates/varying order robustly.
        samples.sort_by(|a, b| a.depth.partial_cmp(&b.depth).unwrap_or(Ordering::Equal));
        holes.push(Hole { id: hole_id, samples });
    }

    // Global robust stats per property (used for fallback + gating).
    let mut global_med: Vec<f64> = vec![0.0; num_props];
    let mut global_scale: Vec<f64> = vec![1.0; num_props];
    for pi in 0..num_props {
        let vals = &mut all_vals_by_prop[pi];
        if vals.len() >= 5 {
            let med = median_in_place(vals);
            // Reuse same vec for MAD (in-place absolute diffs).
            let scale = mad_scale(vals, med);
            global_med[pi] = med;
            global_scale[pi] = if scale.is_finite() && scale > 1e-12 { scale } else { 1.0 };
        } else {
            global_med[pi] = 0.0;
            global_scale[pi] = 1.0;
        }
    }

    // Adaptive strategy:
    // - Always run within-hole robust window check (cheap, catches many typos).
    // - Run cross-hole spatial neighbor check when total_samples is moderate.
    // - For huge datasets, build a representative spatial field using downsampled points.
    let run_spatial_full = total_samples <= 2_000_000 && total_values > 50_000;
    let run_spatial_downsampled = !run_spatial_full && total_samples <= 20_000_000 && total_values > 200_000;

    let mut suspects: Vec<Suspect> = Vec::new();

    // Thresholds (tuned to report "most obvious" outliers without over-reporting).
    let within_z_thresh = 6.0;
    let spatial_z_thresh = 7.0;
    let min_abs_rel = 3.0; // also require abs deviation > (min_abs_rel * global_scale) for tiny local scale

    // 1) Within-hole check.
    for hole in &holes {
        let n = hole.samples.len();
        if n == 0 {
            continue;
        }
        for si in 0..n {
            for pi in 0..num_props {
                let v = match hole.samples[si].props[pi] {
                    Some(x) => x,
                    None => continue,
                };
                // Collect neighbor values in a small window excluding self.
                let w: isize = 2;
                let mut neigh: [f64; 8] = [0.0; 8];
                let mut cnt = 0usize;
                for dj in -w..=w {
                    if dj == 0 {
                        continue;
                    }
                    let idx = si as isize + dj;
                    if idx < 0 || idx >= n as isize {
                        continue;
                    }
                    if let Some(x) = hole.samples[idx as usize].props[pi] {
                        neigh[cnt] = x;
                        cnt += 1;
                        if cnt == neigh.len() {
                            break;
                        }
                    }
                }
                if cnt < 3 {
                    continue;
                }
                let mut tmp = neigh[..cnt].to_vec();
                let med = median_in_place(&mut tmp);
                let scale = {
                    let mut diffs = neigh[..cnt].to_vec();
                    let s = mad_scale(&mut diffs, med);
                    if s.is_finite() && s > 1e-12 { s } else { global_scale[pi] }
                };
                let dev = (v - med).abs();
                if dev <= 1e-12 {
                    continue;
                }
                let z = dev / (scale + 1e-12);
                let abs_gate = min_abs_rel * global_scale[pi];
                if z >= within_z_thresh && dev >= abs_gate.min(dev) {
                    suspects.push(Suspect {
                        hole_id: hole.id.clone(),
                        sample_idx: si,
                        prop_name: prop_names.get(pi).cloned().unwrap_or_else(|| format!("p{}", pi)),
                        confidence: z,
                    });
                }
            }
        }
    }

    // Helper: build a grid from representative points and query neighborhood.
    // We store indices into `rep_points` vector.
    #[derive(Clone)]
    struct RepPoint {
        hole_idx: usize,
        sample_idx: usize,
        pos: Vec3,
        props: Vec<Option<f64>>,
    }

    let mut rep_points: Vec<RepPoint> = Vec::new();

    if run_spatial_full || run_spatial_downsampled {
        // Choose stride for downsampling; stride=1 means full.
        let stride = if run_spatial_full {
            1usize
        } else {
            // Aim for ~2M rep points max.
            let target = 2_000_000usize;
            let s = (total_samples / target).max(1);
            // Keep stride as power-of-two-ish for faster modulo.
            let mut p = 1usize;
            while p < s {
                p <<= 1;
            }
            p
        };

        rep_points.reserve((total_samples / stride).max(1));
        for (hi, hole) in holes.iter().enumerate() {
            for (si, s) in hole.samples.iter().enumerate() {
                if stride > 1 && (si % stride != 0) {
                    continue;
                }
                rep_points.push(RepPoint {
                    hole_idx: hi,
                    sample_idx: si,
                    pos: s.pos,
                    props: s.props.clone(),
                });
            }
        }

        // Estimate grid cell size from bounding box and rep point count.
        let mut minp = Vec3 { x: f64::INFINITY, y: f64::INFINITY, z: f64::INFINITY };
        let mut maxp = Vec3 { x: f64::NEG_INFINITY, y: f64::NEG_INFINITY, z: f64::NEG_INFINITY };
        for rp in &rep_points {
            minp.x = minp.x.min(rp.pos.x);
            minp.y = minp.y.min(rp.pos.y);
            minp.z = minp.z.min(rp.pos.z);
            maxp.x = maxp.x.max(rp.pos.x);
            maxp.y = maxp.y.max(rp.pos.y);
            maxp.z = maxp.z.max(rp.pos.z);
        }
        let dx = (maxp.x - minp.x).max(1e-6);
        let dy = (maxp.y - minp.y).max(1e-6);
        let dz = (maxp.z - minp.z).max(1e-6);
        let volume = dx * dy * dz;
        let density = rep_points.len().max(1) as f64 / volume;
        // Characteristic spacing for Poisson-ish points: spacing ~ density^(-1/3)
        let spacing = if density > 0.0 { density.powf(-1.0 / 3.0) } else { 10.0 };
        let cell = (spacing * 1.5).clamp(1.0, 500.0);
        let radius = (spacing * 3.0).clamp(cell * 1.5, 2000.0);
        let r2 = radius * radius;

        let mut grid: HashMap<i128, Vec<u32>> = HashMap::new();
        grid.reserve(rep_points.len() * 2);

        for (i, rp) in rep_points.iter().enumerate() {
            let ix = ((rp.pos.x - minp.x) / cell).floor() as i32;
            let iy = ((rp.pos.y - minp.y) / cell).floor() as i32;
            let iz = ((rp.pos.z - minp.z) / cell).floor() as i32;
            grid.entry(grid_key(ix, iy, iz)).or_default().push(i as u32);
        }

        // 2) Spatial check: compare each sample to neighborhood robust stats built from rep points.
        // For full: samples == rep points (stride 1), so we validate all points.
        // For downsampled: we validate all samples using rep neighborhood (approx).
        let max_neighbors = if run_spatial_full { 64usize } else { 48usize };

        for (hi, hole) in holes.iter().enumerate() {
            for (si, s) in hole.samples.iter().enumerate() {
                let pos = s.pos;
                let ix = ((pos.x - minp.x) / cell).floor() as i32;
                let iy = ((pos.y - minp.y) / cell).floor() as i32;
                let iz = ((pos.z - minp.z) / cell).floor() as i32;

                // Gather nearby rep point indices.
                let mut neigh_idx: [u32; 96] = [0; 96];
                let mut ncnt = 0usize;
                for oz in -1..=1 {
                    for oy in -1..=1 {
                        for ox in -1..=1 {
                            if let Some(list) = grid.get(&grid_key(ix + ox, iy + oy, iz + oz)) {
                                for &ri in list {
                                    // Basic distance filter (cheap).
                                    let rp = &rep_points[ri as usize];
                                    if rp.pos.dist2(pos) <= r2 {
                                        // Exclude exact same point in full mode (avoid comparing to itself).
                                        if run_spatial_full && rp.hole_idx == hi && rp.sample_idx == si {
                                            continue;
                                        }
                                        neigh_idx[ncnt] = ri;
                                        ncnt += 1;
                                        if ncnt == neigh_idx.len() {
                                            break;
                                        }
                                    }
                                }
                            }
                            if ncnt == neigh_idx.len() {
                                break;
                            }
                        }
                        if ncnt == neigh_idx.len() {
                            break;
                        }
                    }
                    if ncnt == neigh_idx.len() {
                        break;
                    }
                }

                if ncnt < 8 {
                    continue;
                }

                for pi in 0..num_props {
                    let v = match s.props[pi] {
                        Some(x) => x,
                        None => continue,
                    };

                    // Collect neighbor values for this property (cap for speed).
                    let mut vals: Vec<f64> = Vec::with_capacity(max_neighbors);
                    for &ri in &neigh_idx[..ncnt] {
                        let rp = &rep_points[ri as usize];
                        if let Some(x) = rp.props[pi] {
                            vals.push(x);
                            if vals.len() == max_neighbors {
                                break;
                            }
                        }
                    }
                    if vals.len() < 8 {
                        continue;
                    }
                    let med = median_in_place(&mut vals);
                    let scale = {
                        let mut diffs = vals;
                        let s2 = mad_scale(&mut diffs, med);
                        if s2.is_finite() && s2 > 1e-12 { s2 } else { global_scale[pi] }
                    };
                    let dev = (v - med).abs();
                    if dev <= 1e-12 {
                        continue;
                    }
                    let z = dev / (scale + 1e-12);
                    if z >= spatial_z_thresh && dev >= min_abs_rel * global_scale[pi] {
                        suspects.push(Suspect {
                            hole_id: hole.id.clone(),
                            sample_idx: si,
                            prop_name: prop_names.get(pi).cloned().unwrap_or_else(|| format!("p{}", pi)),
                            confidence: z * 1.1, // slightly favor spatial anomalies over within-hole
                        });
                    }
                }
            }
        }
    }

    // 3) Global sanity check for extreme values if dataset is tiny or has weak neighborhood support.
    if total_values > 0 && total_samples < 50_000 {
        for hole in &holes {
            for (si, s) in hole.samples.iter().enumerate() {
                for pi in 0..num_props {
                    let v = match s.props[pi] {
                        Some(x) => x,
                        None => continue,
                    };
                    let dev = (v - global_med[pi]).abs();
                    let z = dev / (global_scale[pi] + 1e-12);
                    if z >= 10.0 {
                        suspects.push(Suspect {
                            hole_id: hole.id.clone(),
                            sample_idx: si,
                            prop_name: prop_names.get(pi).cloned().unwrap_or_else(|| format!("p{}", pi)),
                            confidence: z * 0.8,
                        });
                    }
                }
            }
        }
    }

    // Deduplicate: keep max confidence per (hole_id, sample_idx, prop_name).
    let mut best: HashMap<(String, usize, String), f64> = HashMap::new();
    for s in suspects {
        let key = (s.hole_id, s.sample_idx, s.prop_name);
        let entry = best.entry(key).or_insert(f64::NEG_INFINITY);
        if s.confidence.is_finite() && s.confidence > *entry {
            *entry = s.confidence;
        }
    }

    let mut out: Vec<Suspect> = Vec::with_capacity(best.len());
    for ((hid, si, pn), conf) in best {
        out.push(Suspect {
            hole_id: hid,
            sample_idx: si,
            prop_name: pn,
            confidence: conf,
        });
    }

    // Sort by confidence descending, deterministic tiebreakers.
    out.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.prop_name.cmp(&b.prop_name))
    });

    // Limit output to a reasonable number for massive datasets.
    let max_report = if total_samples > 5_000_000 { 50_000 } else { 200_000 };
    if out.len() > max_report {
        out.truncate(max_report);
    }

    let mut stdout = String::new();
    stdout.push_str(&format!("{}\n", out.len()));
    for s in out {
        // confidence as float with 3 decimals.
        stdout.push_str(&format!(
            "{} {} {} {:.3}\n",
            s.hole_id, s.sample_idx, s.prop_name, s.confidence
        ));
    }
    print!("{stdout}");
}
