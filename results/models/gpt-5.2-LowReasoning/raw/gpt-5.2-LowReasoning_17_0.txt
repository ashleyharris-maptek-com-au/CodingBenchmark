using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private readonly struct Box
    {
        public readonly int Index;
        public readonly int W;
        public readonly int H;
        public readonly int D;

        public Box(int index, int w, int h, int d)
        {
            Index = index;
            W = w;
            H = h;
            D = d;
        }

        public long Volume => (long)W * H * D;
        public int MaxDim => W >= H ? (W >= D ? W : D) : (H >= D ? H : D);
    }

    private readonly struct Orientation
    {
        public readonly int W, H, D;
        public Orientation(int w, int h, int d) { W = w; H = h; D = d; }
    }

    private struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    private struct Space
    {
        public int X, Y, Z;
        public int W, H, D;
        public Space(int x, int y, int z, int w, int h, int d) { X = x; Y = y; Z = z; W = w; H = h; D = d; }
        public long Volume => (long)W * H * D;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        if (!fs.TryReadInt(out int n)) return;
        fs.ReadInt(out int containerW);
        fs.ReadInt(out int containerH);
        fs.ReadInt(out int containerD);

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            fs.ReadInt(out int w);
            fs.ReadInt(out int h);
            fs.ReadInt(out int d);
            boxes[i] = new Box(i, w, h, d);
        }

        Array.Sort(boxes, (a, b) =>
        {
            int vc = b.Volume.CompareTo(a.Volume);
            if (vc != 0) return vc;
            int mc = b.MaxDim.CompareTo(a.MaxDim);
            if (mc != 0) return mc;
            return a.Index.CompareTo(b.Index);
        });

        List<Placement> placements;
        // Heuristic switch: large N favors O(N) shelf packing; small N can use a stronger but still bounded free-space search.
        if (n <= 2500)
            placements = PackGuillotine(boxes, containerW, containerH, containerD);
        else
            placements = PackShelves(boxes, containerW, containerH, containerD);

        var sb = new StringBuilder(32 + placements.Count * 24);
        sb.Append(placements.Count).Append('\n');
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            sb.Append(p.Index).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).Append('\n');
        }
        Console.Out.Write(sb.ToString());
    }

    private static Orientation[] GetOrientations(in Box b)
    {
        int a = b.W, c = b.H, d = b.D;
        // Deduplicate orientations with equal dimensions.
        Span<Orientation> tmp = stackalloc Orientation[6]
        {
            new Orientation(a,c,d),
            new Orientation(a,d,c),
            new Orientation(c,a,d),
            new Orientation(c,d,a),
            new Orientation(d,a,c),
            new Orientation(d,c,a)
        };

        var list = new List<Orientation>(6);
        for (int i = 0; i < 6; i++)
        {
            var o = tmp[i];
            bool seen = false;
            for (int j = 0; j < list.Count; j++)
            {
                var e = list[j];
                if (e.W == o.W && e.H == o.H && e.D == o.D) { seen = true; break; }
            }
            if (!seen) list.Add(o);
        }
        return list.ToArray();
    }

    private static List<Placement> PackShelves(Box[] boxes, int W, int H, int D)
    {
        var result = new List<Placement>(Math.Min(boxes.Length, 10000));

        int x = 0, y = 0, z = 0;
        int rowHeight = 0;
        int layerDepth = 0;

        for (int i = 0; i < boxes.Length; i++)
        {
            var b = boxes[i];
            var orientations = GetOrientations(b);

            bool placed = TryPlaceInShelf(W, H, D, orientations, b.Index,
                ref x, ref y, ref z, ref rowHeight, ref layerDepth, result);

            if (!placed)
                continue;
        }

        return result;
    }

    private static bool TryPlaceInShelf(
        int W, int H, int D,
        Orientation[] orientations,
        int boxIndex,
        ref int x, ref int y, ref int z,
        ref int rowHeight, ref int layerDepth,
        List<Placement> result)
    {
        for (int attempt = 0; attempt < 3; attempt++)
        {
            if (TrySelectOrientationShelf(W, H, D, orientations, x, y, z, rowHeight, layerDepth,
                    out Orientation best, out int newRowHeight, out int newLayerDepth))
            {
                result.Add(new Placement
                {
                    Index = boxIndex,
                    X = x,
                    Y = y,
                    Z = z,
                    W = best.W,
                    H = best.H,
                    D = best.D
                });

                x += best.W;
                if (best.H > rowHeight) rowHeight = best.H;
                if (best.D > layerDepth) layerDepth = best.D;
                return true;
            }

            // Advance packing frontier: row -> layer -> fail
            if (attempt == 0)
            {
                // New row
                x = 0;
                y += rowHeight;
                rowHeight = 0;
                if (y >= H) attempt++; // fall through to layer on next loop
            }
            else if (attempt == 1)
            {
                // New layer
                x = 0;
                y = 0;
                z += layerDepth;
                layerDepth = 0;
                if (z >= D) return false;
            }
        }
        return false;
    }

    private static bool TrySelectOrientationShelf(
        int W, int H, int D,
        Orientation[] orientations,
        int x, int y, int z,
        int rowHeight, int layerDepth,
        out Orientation best,
        out int newRowHeight,
        out int newLayerDepth)
    {
        best = default;
        newRowHeight = rowHeight;
        newLayerDepth = layerDepth;

        long bestScore = long.MaxValue;
        bool any = false;

        int remX = W - x;
        int remY = H - y;
        int remZ = D - z;
        if (remX <= 0 || remY <= 0 || remZ <= 0) return false;

        for (int i = 0; i < orientations.Length; i++)
        {
            var o = orientations[i];
            if (o.W > remX || o.H > remY || o.D > remZ) continue;

            int rh = rowHeight;
            if (o.H > rh) rh = o.H;
            if (y + rh > H) continue;

            int ld = layerDepth;
            if (o.D > ld) ld = o.D;
            if (z + ld > D) continue;

            // Prefer tighter fits along X, then Y, then Z (cheap, stable for large N).
            long score = (long)(remX - o.W) * 1_000_000L + (long)(remY - o.H) * 1_000L + (remZ - o.D);
            if (score < bestScore)
            {
                bestScore = score;
                best = o;
                newRowHeight = rh;
                newLayerDepth = ld;
                any = true;
            }
        }

        return any;
    }

    private static List<Placement> PackGuillotine(Box[] boxes, int W, int H, int D)
    {
        var result = new List<Placement>(Math.Min(boxes.Length, 4000));
        var spaces = new List<Space>(4096) { new Space(0, 0, 0, W, H, D) };

        const int MaxSpaces = 20000; // hard cap to keep runtime bounded
        const int ScanLimit = 5000;  // scan only best candidates to avoid O(N*spaces) blowups

        for (int bi = 0; bi < boxes.Length; bi++)
        {
            var b = boxes[bi];
            var orientations = GetOrientations(b);

            int bestSpaceIdx = -1;
            Orientation bestOri = default;
            long bestWaste = long.MaxValue;

            // Scan spaces in descending volume (maintained approximately by occasional resort).
            int scanned = 0;
            for (int si = 0; si < spaces.Count && scanned < ScanLimit; si++, scanned++)
            {
                var s = spaces[si];
                if (s.W <= 0 || s.H <= 0 || s.D <= 0) continue;
                if (s.W < 1 || s.H < 1 || s.D < 1) continue;

                for (int oi = 0; oi < orientations.Length; oi++)
                {
                    var o = orientations[oi];
                    if (o.W <= s.W && o.H <= s.H && o.D <= s.D)
                    {
                        long waste = s.Volume - (long)o.W * o.H * o.D;
                        if (waste < bestWaste)
                        {
                            bestWaste = waste;
                            bestSpaceIdx = si;
                            bestOri = o;
                            if (waste == 0) break;
                        }
                    }
                }
                if (bestWaste == 0) break;
            }

            if (bestSpaceIdx < 0) continue;

            var chosen = spaces[bestSpaceIdx];
            int px = chosen.X, py = chosen.Y, pz = chosen.Z;
            result.Add(new Placement { Index = b.Index, X = px, Y = py, Z = pz, W = bestOri.W, H = bestOri.H, D = bestOri.D });

            // Remove chosen space and split into up to 3 orthogonal residuals.
            spaces[bestSpaceIdx] = spaces[spaces.Count - 1];
            spaces.RemoveAt(spaces.Count - 1);

            int rw = chosen.W - bestOri.W;
            int rh = chosen.H - bestOri.H;
            int rd = chosen.D - bestOri.D;

            // Split order heuristic: keep larger leftovers as independent spaces.
            if (rw > 0) spaces.Add(new Space(px + bestOri.W, py, pz, rw, chosen.H, chosen.D));
            if (rh > 0) spaces.Add(new Space(px, py + bestOri.H, pz, bestOri.W, rh, chosen.D));
            if (rd > 0) spaces.Add(new Space(px, py, pz + bestOri.D, bestOri.W, bestOri.H, rd));

            // Prune dominated/tiny spaces occasionally and keep list roughly volume-sorted.
            if (spaces.Count > MaxSpaces)
            {
                PruneSpaces(spaces);
                if (spaces.Count > MaxSpaces)
                    spaces.RemoveRange(MaxSpaces, spaces.Count - MaxSpaces);
            }
            if ((bi & 127) == 0)
            {
                spaces.Sort((a, c) => c.Volume.CompareTo(a.Volume));
            }
        }

        return result;
    }

    private static void PruneSpaces(List<Space> spaces)
    {
        // Remove spaces fully contained in another space (O(k^2) but k is capped).
        spaces.Sort((a, b) => b.Volume.CompareTo(a.Volume));
        for (int i = 0; i < spaces.Count; i++)
        {
            var a = spaces[i];
            if (a.W <= 0 || a.H <= 0 || a.D <= 0) { spaces[i] = new Space(0, 0, 0, 0, 0, 0); continue; }
            for (int j = i + 1; j < spaces.Count; j++)
            {
                var b = spaces[j];
                if (b.W <= 0 || b.H <= 0 || b.D <= 0) continue;
                if (IsContained(b, a))
                {
                    spaces[j] = new Space(0, 0, 0, 0, 0, 0);
                }
            }
        }
        int w = 0;
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            if (s.W > 0 && s.H > 0 && s.D > 0)
                spaces[w++] = s;
        }
        if (w < spaces.Count) spaces.RemoveRange(w, spaces.Count - w);
    }

    private static bool IsContained(in Space inner, in Space outer)
    {
        return inner.X >= outer.X && inner.Y >= outer.Y && inner.Z >= outer.Z
            && inner.X + inner.W <= outer.X + outer.W
            && inner.Y + inner.H <= outer.Y + outer.H
            && inner.Z + inner.D <= outer.Z + outer.D;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        public bool TryReadInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c < 0) return false;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int v = 0;
            while (c > 32 && c >= 0)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            value = v * sign;
            return true;
        }

        public void ReadInt(out int value)
        {
            if (!TryReadInt(out value)) value = 0;
        }

        private int Read()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }
    }
}
