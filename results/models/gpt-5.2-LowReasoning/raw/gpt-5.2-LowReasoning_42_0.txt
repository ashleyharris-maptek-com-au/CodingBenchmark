; SPIR-V
; Version: 1.0
; Generator: benchmark
; Bound: 200
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_in "gl_in"
OpName %in_color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_1 = OpConstant %uint 1
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%arr_v3_in = OpTypeArray %v3float %uint_1

%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Output_v3 = OpTypePointer Output %v3float

%int_0 = OpConstant %int 0
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_h = OpConstant %float 0.04

%gl_in = OpVariable %ptr_Input_arr_pv Input
%in_color = OpVariable %ptr_Input_arr_v3 Input

%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load input point position (gl_in[0].gl_Position)
%ptr_in_pos = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%in_pos = OpLoad %v4float %ptr_in_pos

; Load input color (color[0])
%ptr_in_col = OpAccessChain %ptr_Input_v3 %in_color %int_0
%col0 = OpLoad %v3float %ptr_in_col

; Extract x/y
%x = OpCompositeExtract %float %in_pos 0
%y = OpCompositeExtract %float %in_pos 1

; Precompute x +/- h, y +/- h
%x_minus = OpFSub %float %x %float_h
%x_plus  = OpFAdd %float %x %float_h
%y_minus = OpFSub %float %y %float_h
%y_plus  = OpFAdd %float %y %float_h

; Pointer to gl_out.gl_Position
%ptr_out_pos = OpAccessChain %ptr_Output_v4 %gl_out %int_0

; Emit vertex 0: (x-h, y-h)
%p0 = OpCompositeConstruct %v4float %x_minus %y_minus %float_0 %float_1
OpStore %ptr_out_pos %p0
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 1: (x+h, y-h)
%p1 = OpCompositeConstruct %v4float %x_plus %y_minus %float_0 %float_1
OpStore %ptr_out_pos %p1
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 2: (x-h, y+h)
%p2 = OpCompositeConstruct %v4float %x_minus %y_plus %float_0 %float_1
OpStore %ptr_out_pos %p2
OpStore %outColor %col0
OpEmitVertex

; Emit vertex 3: (x+h, y+h)
%p3 = OpCompositeConstruct %v4float %x_plus %y_plus %float_0 %float_1
OpStore %ptr_out_pos %p3
OpStore %outColor %col0
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
