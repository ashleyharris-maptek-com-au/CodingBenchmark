use std::io::{self, Read};

struct FastInput {
    buf: Vec<u8>,
    pos: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, pos: 0 })
    }

    fn skip_ws(&mut self) {
        while self.pos < self.buf.len() {
            match self.buf[self.pos] {
                b' ' | b'\n' | b'\r' | b'\t' => self.pos += 1,
                _ => break,
            }
        }
    }

    fn read_usize(&mut self) -> Option<usize> {
        self.skip_ws();
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut n: usize = 0;
        while self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            if !(b'0'..=b'9').contains(&b) {
                break;
            }
            n = n.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.pos += 1;
        }
        Some(n)
    }

    fn read_row_exact(&mut self, w: usize) -> Option<Vec<u8>> {
        let mut row = Vec::with_capacity(w);
        while row.len() < w && self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            self.pos += 1;
            if b == b'\r' {
                continue;
            }
            if b == b'\n' {
                continue;
            }
            row.push(b);
        }
        if row.len() != w {
            return None;
        }
        while self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            self.pos += 1;
            if b == b'\n' {
                break;
            }
        }
        Some(row)
    }
}

#[inline]
fn bit_get(bits: &[u64], idx: usize) -> bool {
    let word = idx >> 6;
    let bit = idx & 63;
    (bits[word] >> bit) & 1 != 0
}

#[inline]
fn bit_set(bits: &mut [u64], idx: usize) {
    let word = idx >> 6;
    let bit = idx & 63;
    bits[word] |= 1u64 << bit;
}

#[inline]
fn bits2_get(bits: &[u64], idx: usize) -> u8 {
    let bitpos = idx << 1;
    let word = bitpos >> 6;
    let off = bitpos & 63;
    if off <= 62 {
        ((bits[word] >> off) & 0b11) as u8
    } else {
        let lo = (bits[word] >> 63) & 1;
        let hi = bits[word + 1] & 1;
        ((lo | (hi << 1)) as u8) & 0b11
    }
}

#[inline]
fn bits2_set(bits: &mut [u64], idx: usize, val: u8) {
    let v = (val & 0b11) as u64;
    let bitpos = idx << 1;
    let word = bitpos >> 6;
    let off = bitpos & 63;
    if off <= 62 {
        let mask = !(0b11u64 << off);
        bits[word] = (bits[word] & mask) | (v << off);
    } else {
        bits[word] = (bits[word] & !(1u64 << 63)) | ((v & 1) << 63);
        bits[word + 1] = (bits[word + 1] & !1u64) | ((v >> 1) & 1);
    }
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let h = match input.read_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    let w = match input.read_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    if h == 0 || w == 0 {
        return Ok(());
    }
    let cells_u128 = (h as u128) * (w as u128);
    if cells_u128 > (usize::MAX as u128) {
        return Ok(());
    }
    let n = cells_u128 as usize;

    let mut walls = vec![0u64; (n + 63) >> 6];
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    for r in 0..h {
        let row = match input.read_row_exact(w) {
            Some(v) => v,
            None => return Ok(()),
        };
        let base = r * w;
        for c in 0..w {
            let ch = row[c];
            let idx = base + c;
            match ch {
                b'#' => bit_set(&mut walls, idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    let (s, g) = match (start, goal) {
        (Some(s), Some(g)) => (s, g),
        _ => {
            // No valid endpoints: output nothing.
            return Ok(());
        }
    };
    if s == g {
        println!("1");
        println!("{} {}", s / w, s % w);
        return Ok(());
    }

    let mut visited = vec![0u64; (n + 63) >> 6];
    let mut parent_dir = vec![0u64; (n * 2 + 63) >> 6];

    let mut queue: Vec<u32> = Vec::new();
    queue.push(s as u32);
    bit_set(&mut visited, s);

    let mut head: usize = 0;
    let mut found = false;

    while head < queue.len() {
        let cur = queue[head] as usize;
        head += 1;
        if cur == g {
            found = true;
            break;
        }

        let r = cur / w;
        let c = cur - r * w;

        // Up: neighbor = cur - w (if r > 0). Record parent_dir as 0 (came from down).
        if r > 0 {
            let nb = cur - w;
            if !bit_get(&walls, nb) && !bit_get(&visited, nb) {
                bit_set(&mut visited, nb);
                bits2_set(&mut parent_dir, nb, 0);
                queue.push(nb as u32);
            }
        }
        // Down: neighbor = cur + w (if r + 1 < h). Record as 1 (came from up).
        if r + 1 < h {
            let nb = cur + w;
            if !bit_get(&walls, nb) && !bit_get(&visited, nb) {
                bit_set(&mut visited, nb);
                bits2_set(&mut parent_dir, nb, 1);
                queue.push(nb as u32);
            }
        }
        // Left: neighbor = cur - 1 (if c > 0). Record as 2 (came from right).
        if c > 0 {
            let nb = cur - 1;
            if !bit_get(&walls, nb) && !bit_get(&visited, nb) {
                bit_set(&mut visited, nb);
                bits2_set(&mut parent_dir, nb, 2);
                queue.push(nb as u32);
            }
        }
        // Right: neighbor = cur + 1 (if c + 1 < w). Record as 3 (came from left).
        if c + 1 < w {
            let nb = cur + 1;
            if !bit_get(&walls, nb) && !bit_get(&visited, nb) {
                bit_set(&mut visited, nb);
                bits2_set(&mut parent_dir, nb, 3);
                queue.push(nb as u32);
            }
        }
    }

    if !found || !bit_get(&visited, g) {
        // Output nothing or 0; choose 0 for clarity.
        println!("0");
        return Ok(());
    }

    // Reconstruct path from goal back to start.
    let mut path: Vec<usize> = Vec::new();
    let mut cur = g;
    path.push(cur);
    while cur != s {
        let dir = bits2_get(&parent_dir, cur);
        cur = match dir {
            0 => cur + w, // came from down, so parent is below
            1 => cur - w, // came from up, so parent is above
            2 => cur + 1, // came from right, so parent is right
            3 => cur - 1, // came from left, so parent is left
            _ => break,
        };
        path.push(cur);
    }
    path.reverse();

    let mut out = String::new();
    out.push_str(&format!("{}\n", path.len()));
    for idx in path {
        out.push_str(&format!("{} {}\n", idx / w, idx % w));
    }
    print!("{out}");
    Ok(())
}
