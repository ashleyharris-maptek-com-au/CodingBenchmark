// Graph Coloring Heuristic Solver
// Reads: n m k followed by m edges (u v, 0-indexed)
// Writes: n integers (colors 0..k-1) forming a proper k-coloring.
//
// Strategy:
// - For very large graphs, use a fast greedy coloring in (static) descending-degree order,
//   with 64-bit neighbor-color masks when k<=64, plus a lightweight local recolor repair.
// - For smaller graphs, use a DSATUR-style greedy ordering (no backtracking) to reduce conflicts.
//
// Notes:
// - The benchmark instances are assumed to be k-colorable; this program focuses on quickly
//   finding a valid coloring within time limits on large graphs.

#include <algorithm>
#include <bit>
#include <chrono>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <limits>
#include <queue>
#include <random>
#include <vector>

using std::int32_t;
using std::int64_t;
using std::uint32_t;
using std::uint64_t;

static inline int fastReadInt() {
  int c = std::getc(stdin);
  while (c != EOF && c <= ' ') c = std::getc(stdin);
  int sign = 1;
  if (c == '-') {
    sign = -1;
    c = std::getc(stdin);
  }
  int x = 0;
  while (c > ' ') {
    x = x * 10 + (c - '0');
    c = std::getc(stdin);
  }
  return x * sign;
}

struct CSRGraph {
  int n = 0;
  int64_t m = 0; // undirected edges count
  std::vector<int> offset;  // size n+1
  std::vector<int> adj;     // size 2m
  std::vector<int> degree;  // size n
};

static CSRGraph readGraph() {
  CSRGraph g;
  g.n = fastReadInt();
  int64_t m = (int64_t)fastReadInt();
  g.m = m;
  (void)fastReadInt(); // k read by caller; keep stdin aligned if needed

  // Rewind: caller already consumed k? Actually caller will read all three; we can't here.
  // This function isn't used; keep for completeness.
  return g;
}

static void buildCSR(int n, int64_t m, const std::vector<std::pair<int, int>>& edges, CSRGraph& g) {
  g.n = n;
  g.m = m;
  g.degree.assign(n, 0);
  for (const auto& e : edges) {
    int u = e.first, v = e.second;
    if (u == v) continue;
    g.degree[u]++;
    g.degree[v]++;
  }
  g.offset.assign(n + 1, 0);
  for (int i = 0; i < n; i++) g.offset[i + 1] = g.offset[i] + g.degree[i];
  g.adj.assign((size_t)g.offset[n], 0);
  std::vector<int> cur = g.offset;
  for (const auto& e : edges) {
    int u = e.first, v = e.second;
    if (u == v) continue;
    g.adj[cur[u]++] = v;
    g.adj[cur[v]++] = u;
  }
}

static std::vector<int> orderByDescendingDegree(const CSRGraph& g) {
  int n = g.n;
  int maxDeg = 0;
  for (int d : g.degree) if (d > maxDeg) maxDeg = d;
  std::vector<int> bucket((size_t)maxDeg + 1, 0);
  for (int d : g.degree) bucket[d]++;
  for (int d = 1; d <= maxDeg; d++) bucket[d] += bucket[d - 1];
  std::vector<int> order(n);
  // Stable-ish: place vertices by degree ascending then reverse for descending
  for (int v = 0; v < n; v++) {
    int d = g.degree[v];
    int pos = --bucket[d];
    order[pos] = v;
  }
  std::reverse(order.begin(), order.end());
  return order;
}

static bool isValidColoring(const CSRGraph& g, const std::vector<int>& color) {
  for (int v = 0; v < g.n; v++) {
    int cv = color[v];
    if (cv < 0) return false;
    for (int i = g.offset[v]; i < g.offset[v + 1]; i++) {
      int u = g.adj[i];
      if (u > v && color[u] == cv) return false;
    }
  }
  return true;
}

static inline int pickFirstAvailable(uint64_t usedMask, int k) {
  uint64_t avail = (~usedMask);
  if (k < 64) avail &= ((1ULL << k) - 1ULL);
  if (!avail) return -1;
  // ctz is available via builtin on MSVC as _tzcnt_u64 but keep portable:
  for (int c = 0; c < k; c++) {
    if (avail & (1ULL << c)) return c;
  }
  return -1;
}

static bool greedyColoringMaskRepair(const CSRGraph& g, int k, const std::vector<int>& order, std::vector<int>& color,
                                    int maxRepairsPerVertex) {
  int n = g.n;
  color.assign(n, -1);
  for (int idx = 0; idx < n; idx++) {
    int v = order[idx];
    uint64_t used = 0;
    for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
      int u = g.adj[ei];
      int cu = color[u];
      if (cu >= 0 && cu < 64) used |= (1ULL << cu);
    }
    int c = pickFirstAvailable(used, k);
    if (c >= 0) {
      color[v] = c;
      continue;
    }

    // Lightweight repair: try to "steal" a color by recoloring one neighbor of that color.
    bool fixed = false;
    int repairsTried = 0;
    for (int targetColor = 0; targetColor < k && !fixed; targetColor++) {
      // Find one neighbor u with color targetColor
      int uPick = -1;
      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        if (color[u] == targetColor) {
          uPick = u;
          break;
        }
      }
      if (uPick < 0) continue;

      uint64_t usedU = 0;
      for (int ej = g.offset[uPick]; ej < g.offset[uPick + 1]; ej++) {
        int w = g.adj[ej];
        int cw = color[w];
        if (w == v) continue; // if we move v into targetColor, v will match uPick's old color, so ignore v
        if (cw >= 0 && cw < 64) usedU |= (1ULL << cw);
      }
      // Also exclude targetColor itself (we are trying to move uPick away from it)
      usedU |= (1ULL << targetColor);
      int newColorU = pickFirstAvailable(usedU, k);
      if (newColorU >= 0) {
        color[uPick] = newColorU;
        color[v] = targetColor;
        fixed = true;
        break;
      }

      if (++repairsTried >= maxRepairsPerVertex) break;
    }
    if (!fixed) return false;
  }
  return true;
}

// DSATUR-style greedy (no backtracking), suitable for smaller graphs where better ordering helps.
static bool dsaturGreedy(const CSRGraph& g, int k, std::vector<int>& color) {
  int n = g.n;
  color.assign(n, -1);
  std::vector<uint64_t> satMask(n, 0);
  std::vector<int> satDeg(n, 0);
  std::vector<int> deg = g.degree;

  struct NodeKey {
    int sat;
    int deg;
    int v;
  };
  struct Cmp {
    bool operator()(const NodeKey& a, const NodeKey& b) const {
      if (a.sat != b.sat) return a.sat < b.sat;   // max-heap by sat
      if (a.deg != b.deg) return a.deg < b.deg;   // then degree
      return a.v > b.v;                           // then smaller id first
    }
  };
  std::priority_queue<NodeKey, std::vector<NodeKey>, Cmp> pq;
  for (int v = 0; v < n; v++) pq.push({0, deg[v], v});
  std::vector<char> done(n, 0);

  int colored = 0;
  while (colored < n) {
    NodeKey top;
    do {
      if (pq.empty()) return false;
      top = pq.top();
      pq.pop();
    } while (done[top.v] || top.sat != satDeg[top.v] || top.deg != deg[top.v]);

    int v = top.v;
    uint64_t used = satMask[v];
    int c = pickFirstAvailable(used, k);
    if (c < 0) return false;
    color[v] = c;
    done[v] = 1;
    colored++;

    // Update neighbors' saturation
    uint64_t bit = (c < 64) ? (1ULL << c) : 0ULL;
    for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
      int u = g.adj[ei];
      if (done[u]) continue;
      uint64_t before = satMask[u];
      uint64_t after = before | bit;
      if (after != before) {
        satMask[u] = after;
        satDeg[u] = (int)std::popcount(after);
      }
      pq.push({satDeg[u], deg[u], u});
    }
  }
  return true;
}

static bool tryMultipleGreedy(const CSRGraph& g, int k, std::vector<int>& color) {
  auto baseOrder = orderByDescendingDegree(g);
  // Deterministic first try
  if (k <= 64) {
    if (greedyColoringMaskRepair(g, k, baseOrder, color, /*maxRepairsPerVertex=*/4)) return true;
  }

  // Randomized retries with slight perturbation for harder instances.
  std::mt19937_64 rng(0xC0FFEEULL);
  std::vector<int> order = baseOrder;
  int n = g.n;
  int tries = (n >= 200000) ? 6 : 16;
  for (int t = 0; t < tries; t++) {
    // Shuffle small blocks to preserve approximate degree ordering while changing tie breaks.
    int block = (n >= 200000) ? 2048 : 512;
    for (int i = 0; i < n; i += block) {
      int j = std::min(n, i + block);
      std::shuffle(order.begin() + i, order.begin() + j, rng);
    }
    if (k <= 64) {
      if (greedyColoringMaskRepair(g, k, order, color, /*maxRepairsPerVertex=*/6)) return true;
    }
  }
  return false;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n;
  int64_t m;
  int k;

  // Use fast stdio reader for performance on huge inputs.
  n = fastReadInt();
  m = (int64_t)fastReadInt();
  k = fastReadInt();

  std::vector<std::pair<int, int>> edges;
  edges.reserve((size_t)m);
  for (int64_t i = 0; i < m; i++) {
    int u = fastReadInt();
    int v = fastReadInt();
    edges.emplace_back(u, v);
  }

  CSRGraph g;
  buildCSR(n, m, edges, g);
  edges.clear();
  edges.shrink_to_fit();

  std::vector<int> color;

  bool ok = false;
  if (k <= 0) ok = false;
  else if (n <= 30000 && k <= 64) {
    ok = dsaturGreedy(g, k, color);
    if (!ok) ok = tryMultipleGreedy(g, k, color);
  } else {
    ok = tryMultipleGreedy(g, k, color);
  }

  // Final fallback: plain greedy in input order (should rarely be needed).
  if (!ok) {
    color.assign(n, -1);
    if (k <= 64) {
      std::vector<int> order(n);
      for (int i = 0; i < n; i++) order[i] = i;
      ok = greedyColoringMaskRepair(g, k, order, color, /*maxRepairsPerVertex=*/0);
    }
  }

  // If still not ok, emit a best-effort coloring (may fail validation). Most benchmarks
  // expect success on k-colorable instances; this is last resort.
  if (!ok) {
    for (int v = 0; v < n; v++) {
      if (color[v] < 0) color[v] = 0;
    }
  }

  // Output colors.
  // Use buffered stdout via std::cout (tied off) which is fine for large output.
  for (int i = 0; i < n; i++) {
    if (i) std::cout << ' ';
    int c = color[i];
    if (c < 0) c = 0;
    if (c >= k) c %= k;
    std::cout << c;
  }
  std::cout << "\n";

  return 0;
}
