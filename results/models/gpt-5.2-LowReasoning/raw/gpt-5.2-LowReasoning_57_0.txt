const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const width = data.width | 0;
const height = data.height | 0;
const grid = Array.isArray(data.grid) ? data.grid : [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
let budget = (data.cutBudget | 0) || 0;

const n = width * height;

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

// Precompute terrain.
const baseBlocked = new Uint8Array(n); // 1 if water/rock or out of forest
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    if (c !== 46 /* . */) baseBlocked[idx(x, y)] = 1; // treat anything not '.' as blocked
  }
}

// Town aggregation per cell.
const townValueAt = new Float64Array(n);
const isTownCell = new Uint8Array(n);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (!t) continue;
  const x = t.x | 0;
  const y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const v = Number(t.value);
  if (!Number.isFinite(v) || v <= 0) continue;
  const id = idx(x, y);
  townValueAt[id] += v;
  isTownCell[id] = 1;
}

// Ignition cells.
const isIgnitionCell = new Uint8Array(n);
for (let i = 0; i < ignitions.length; i++) {
  const p = ignitions[i];
  if (!Array.isArray(p) || p.length < 2) continue;
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (!inBounds(x, y)) continue;
  isIgnitionCell[idx(x, y)] = 1;
}

// Our chosen cuts.
const cutBlocked = new Uint8Array(n); // 1 if we cut here
const firebreaks = [];

function bfsWithParents() {
  const dist = new Int32Array(n);
  dist.fill(-1);
  const parent = new Int32Array(n);
  parent.fill(-1);
  const order = new Int32Array(n);
  let ordLen = 0;

  const q = new Int32Array(n);
  let qh = 0;
  let qt = 0;

  for (let i = 0; i < ignitions.length; i++) {
    const p = ignitions[i];
    if (!Array.isArray(p) || p.length < 2) continue;
    const x = p[0] | 0;
    const y = p[1] | 0;
    if (!inBounds(x, y)) continue;
    const s = idx(x, y);
    if (baseBlocked[s] || cutBlocked[s]) continue;
    if (dist[s] !== -1) continue;
    dist[s] = 0;
    q[qt++] = s;
    order[ordLen++] = s;
  }

  while (qh < qt) {
    const v = q[qh++];
    const x = v % width;
    const y = (v / width) | 0;
    // 4-neighbors
    // left
    if (x > 0) {
      const u = v - 1;
      if (dist[u] === -1 && !baseBlocked[u] && !cutBlocked[u]) {
        dist[u] = dist[v] + 1;
        parent[u] = v;
        q[qt++] = u;
        order[ordLen++] = u;
      }
    }
    // right
    if (x + 1 < width) {
      const u = v + 1;
      if (dist[u] === -1 && !baseBlocked[u] && !cutBlocked[u]) {
        dist[u] = dist[v] + 1;
        parent[u] = v;
        q[qt++] = u;
        order[ordLen++] = u;
      }
    }
    // up
    if (y > 0) {
      const u = v - width;
      if (dist[u] === -1 && !baseBlocked[u] && !cutBlocked[u]) {
        dist[u] = dist[v] + 1;
        parent[u] = v;
        q[qt++] = u;
        order[ordLen++] = u;
      }
    }
    // down
    if (y + 1 < height) {
      const u = v + width;
      if (dist[u] === -1 && !baseBlocked[u] && !cutBlocked[u]) {
        dist[u] = dist[v] + 1;
        parent[u] = v;
        q[qt++] = u;
        order[ordLen++] = u;
      }
    }
  }

  return { dist, parent, order, ordLen };
}

function pickCutsBatch(batchMax) {
  const { dist, parent, order, ordLen } = bfsWithParents();

  let threatenedValue = 0;
  for (let i = 0; i < n; i++) {
    if (townValueAt[i] > 0 && dist[i] !== -1) threatenedValue += townValueAt[i];
  }
  if (threatenedValue <= 0) return 0;

  // Build child lists via head/next (adjacency for BFS tree).
  const head = new Int32Array(n);
  head.fill(-1);
  const next = new Int32Array(ordLen);
  // We'll store children edges for nodes that are visited (in order array).
  // Use a separate edge index to keep next compact.
  let e = 0;
  for (let k = 0; k < ordLen; k++) {
    const v = order[k];
    const p = parent[v];
    if (p !== -1) {
      next[e] = head[p];
      head[p] = e;
      order[e] = v; // reuse front of order array as "to" for edge e
      e++;
    }
  }

  const subtree = new Float64Array(n);
  // Initialize with threatened town value at cell.
  for (let i = 0; i < n; i++) {
    if (townValueAt[i] > 0 && dist[i] !== -1) subtree[i] = townValueAt[i];
  }

  // Post-order accumulate using reverse BFS order.
  for (let k = ordLen - 1; k >= 0; k--) {
    const v = order[k];
    const p = parent[v];
    if (p !== -1) subtree[p] += subtree[v];
  }

  // Collect top candidates.
  // Benefit: subtree value saved if we cut this node in the BFS tree approximation.
  const candidates = [];
  for (let k = 0; k < ordLen; k++) {
    const v = order[k];
    if (subtree[v] <= 0) continue;
    if (baseBlocked[v] || cutBlocked[v]) continue;
    if (isIgnitionCell[v]) continue;
    if (isTownCell[v]) continue;
    candidates.push(v);
  }

  if (candidates.length === 0) return 0;

  candidates.sort((a, b) => subtree[b] - subtree[a]);

  const pickedThisBatch = new Uint8Array(n);
  let picked = 0;
  for (let ci = 0; ci < candidates.length && picked < batchMax; ci++) {
    const v = candidates[ci];
    if (cutBlocked[v] || pickedThisBatch[v]) continue;
    // Skip if any ancestor already picked in this batch.
    let u = parent[v];
    let ok = true;
    while (u !== -1) {
      if (pickedThisBatch[u]) {
        ok = false;
        break;
      }
      u = parent[u];
    }
    if (!ok) continue;
    pickedThisBatch[v] = 1;
    cutBlocked[v] = 1;
    firebreaks.push([v % width, (v / width) | 0]);
    picked++;
  }

  return picked;
}

// Greedy loop: recompute BFS/tree, pick a small batch each iteration.
while (budget > 0) {
  const batch = Math.min(5, budget);
  const picked = pickCutsBatch(batch);
  if (picked <= 0) break;
  budget -= picked;
}

process.stdout.write(JSON.stringify({ firebreaks }));
