// Deterministic, dependency-free VRP heuristic for benchmark grading.
// Strategy:
// 1) Sweep clustering by polar angle around depot into up to truckCount contiguous segments.
// 2) Within each segment, build an initial tour using a greedy insertion heuristic.
// 3) Improve with limited 2-opt (bounded window) for scalability.

declare const require: any;
const fs = require("fs");

type Pkg = { id: any; x: number; y: number };
type Input = { depot: [number, number]; truckCount: number; packages: Pkg[] };

const input: string = fs.readFileSync(0, "utf8").trim();
const data: Input = input.length ? JSON.parse(input) : ({} as any);

const depotX = data.depot?.[0] ?? 0;
const depotY = data.depot?.[1] ?? 0;
const pkgs: Pkg[] = Array.isArray(data.packages) ? data.packages : [];
const n = pkgs.length;

const kRaw = Math.max(1, (data.truckCount | 0) || 1);
const k = Math.min(kRaw, Math.max(1, n)); // never create more routes than packages

// Coordinate storage: index 0 is depot, 1..n are packages.
const xs = new Float64Array(n + 1);
const ys = new Float64Array(n + 1);
xs[0] = depotX;
ys[0] = depotY;
for (let i = 0; i < n; i++) {
  xs[i + 1] = pkgs[i].x;
  ys[i + 1] = pkgs[i].y;
}

function dist(a: number, b: number): number {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.hypot(dx, dy);
}

// Sweep order by angle, tie-break by radius (then index) for determinism.
type SweepItem = { idx: number; ang: number; r2: number };
const sweep: SweepItem[] = new Array(n);
for (let i = 0; i < n; i++) {
  const px = xs[i + 1] - depotX;
  const py = ys[i + 1] - depotY;
  const ang = Math.atan2(py, px);
  const r2 = px * px + py * py;
  sweep[i] = { idx: i + 1, ang, r2 };
}
sweep.sort((a, b) => (a.ang - b.ang) || (a.r2 - b.r2) || (a.idx - b.idx));

// Partition into k contiguous segments, as evenly as possible.
const clusters: number[][] = [];
clusters.length = k;
for (let i = 0; i < k; i++) clusters[i] = [];
for (let i = 0; i < n; i++) {
  const bucket = Math.floor((i * k) / n);
  clusters[bucket].push(sweep[i].idx);
}

// Greedy insertion heuristic to build a tour (sequence of node indices, excluding depot).
// Start with farthest-from-depot as seed to reduce early crisscrossing.
function buildTour(nodes: number[]): number[] {
  const m = nodes.length;
  if (m <= 2) return nodes.slice();

  // Find seed pair: farthest node and its nearest neighbor within cluster.
  let seed = nodes[0];
  let bestR2 = -1;
  for (let i = 0; i < m; i++) {
    const v = nodes[i];
    const dx = xs[v] - depotX;
    const dy = ys[v] - depotY;
    const r2 = dx * dx + dy * dy;
    if (r2 > bestR2 || (r2 === bestR2 && v < seed)) {
      bestR2 = r2;
      seed = v;
    }
  }

  let nn = nodes[0] === seed ? nodes[1] : nodes[0];
  let bestD = Infinity;
  for (let i = 0; i < m; i++) {
    const v = nodes[i];
    if (v === seed) continue;
    const d = dist(seed, v);
    if (d < bestD || (d === bestD && v < nn)) {
      bestD = d;
      nn = v;
    }
  }

  const inTour = new Uint8Array(n + 1);
  const tour: number[] = [seed, nn];
  inTour[seed] = 1;
  inTour[nn] = 1;

  // Insert remaining nodes where they cause minimal added distance in the cycle including depot.
  // We treat depot as connected to both ends for an open tour; insertion considers all edges
  // in depot->tour[0], tour[i]->tour[i+1], tour[last]->depot.
  for (let added = 2; added < m; added++) {
    let bestNode = -1;
    let bestPos = 0;
    let bestDelta = Infinity;

    for (let i = 0; i < m; i++) {
      const v = nodes[i];
      if (inTour[v]) continue;

      // Edge: depot -> first
      {
        const a = 0;
        const b = tour[0];
        const delta = dist(a, v) + dist(v, b) - dist(a, b);
        if (delta < bestDelta || (delta === bestDelta && (v < bestNode || (v === bestNode && 0 < bestPos)))) {
          bestDelta = delta;
          bestNode = v;
          bestPos = 0;
        }
      }

      // Edges within tour
      for (let pos = 0; pos < tour.length - 1; pos++) {
        const a = tour[pos];
        const b = tour[pos + 1];
        const delta = dist(a, v) + dist(v, b) - dist(a, b);
        if (delta < bestDelta || (delta === bestDelta && (v < bestNode || (v === bestNode && pos + 1 < bestPos)))) {
          bestDelta = delta;
          bestNode = v;
          bestPos = pos + 1;
        }
      }

      // Edge: last -> depot (insert at end)
      {
        const a = tour[tour.length - 1];
        const b = 0;
        const delta = dist(a, v) + dist(v, b) - dist(a, b);
        if (delta < bestDelta || (delta === bestDelta && (v < bestNode || (v === bestNode && tour.length < bestPos)))) {
          bestDelta = delta;
          bestNode = v;
          bestPos = tour.length;
        }
      }
    }

    tour.splice(bestPos, 0, bestNode);
    inTour[bestNode] = 1;
  }

  return tour;
}

// Limited 2-opt for an open tour with depot endpoints.
function twoOpt(tour: number[]): void {
  const m = tour.length;
  if (m < 4) return;

  // Window bounds control runtime: O(m * window * passes).
  const window = Math.min(60, m - 1);
  const passes = 2;

  for (let pass = 0; pass < passes; pass++) {
    let improved = false;
    for (let i = -1; i < m - 2; i++) {
      // edge (a,b) where a is depot when i=-1
      const a = i < 0 ? 0 : tour[i];
      const b = tour[i + 1];

      const jMax = Math.min(m - 1, i + 1 + window);
      for (let j = i + 2; j <= jMax; j++) {
        // edge (c,d) where d is depot when j==m-1
        const c = tour[j];
        const d = j === m - 1 ? 0 : tour[j + 1];

        const cur = dist(a, b) + dist(c, d);
        const alt = dist(a, c) + dist(b, d);
        if (alt + 1e-12 < cur) {
          // Reverse segment (i+1..j)
          let l = i + 1;
          let r = j;
          while (l < r) {
            const tmp = tour[l];
            tour[l] = tour[r];
            tour[r] = tmp;
            l++;
            r--;
          }
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
}

const routes: any[][] = [];

// Build and improve each cluster route.
for (let i = 0; i < clusters.length; i++) {
  const nodes = clusters[i];
  if (nodes.length === 0) continue;
  const tour = buildTour(nodes);
  twoOpt(tour);
  const ids = new Array(tour.length);
  for (let j = 0; j < tour.length; j++) ids[j] = pkgs[tour[j] - 1].id;
  routes.push(ids);
}

// Safety: ensure every package appears exactly once (repair deterministically if needed).
// This should not trigger, but protects against edge cases with malformed inputs.
const seen = new Map<any, number>();
for (let r = 0; r < routes.length; r++) {
  const route = routes[r];
  for (let j = 0; j < route.length; j++) {
    const id = route[j];
    seen.set(id, (seen.get(id) || 0) + 1);
  }
}
if (seen.size !== n || (() => { for (const c of seen.values()) if (c !== 1) return true; return false; })()) {
  const allIds = pkgs.map(p => p.id);
  allIds.sort((a, b) => {
    const sa = String(a), sb = String(b);
    return sa < sb ? -1 : sa > sb ? 1 : 0;
  });
  const fixed: any[][] = [];
  const kk = Math.min(k, Math.max(1, allIds.length));
  for (let i = 0; i < kk; i++) fixed.push([]);
  for (let i = 0; i < allIds.length; i++) fixed[Math.floor((i * kk) / allIds.length)].push(allIds[i]);
  process.stdout.write(JSON.stringify({ routes: fixed }));
} else {
  process.stdout.write(JSON.stringify({ routes }));
}
