const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const machineCount = Math.max(1, data.machineCount | 0);
const tasksIn = Array.isArray(data.tasks) ? data.tasks : [];

// Normalize tasks
const tasks = new Map();
for (const t of tasksIn) {
  if (!t || t.id === undefined || t.id === null) continue;
  const id = String(t.id);
  const duration = Math.max(0, (t.duration ?? 0) | 0);
  const depsArr = Array.isArray(t.deps) ? t.deps.map(String) : [];
  const deadlineRaw = t.deadline;
  const deadline =
    deadlineRaw === null || deadlineRaw === undefined ? null : (deadlineRaw | 0);
  const weight = Math.max(0, Number.isFinite(t.weight) ? +t.weight : 0);
  tasks.set(id, { id, duration, deps: depsArr, deadline, weight });
}

// Ensure all deps exist as nodes (zero-duration, zero-weight)
for (const t of tasks.values()) {
  for (const d of t.deps) {
    if (!tasks.has(d)) tasks.set(d, { id: d, duration: 0, deps: [], deadline: null, weight: 0 });
  }
}

const ids = Array.from(tasks.keys());
const n = ids.length;

// Build graph
const out = new Map();
const indeg = new Map();
for (const id of ids) {
  out.set(id, []);
  indeg.set(id, 0);
}
for (const t of tasks.values()) {
  for (const d of t.deps) {
    out.get(d).push(t.id);
    indeg.set(t.id, (indeg.get(t.id) || 0) + 1);
  }
}

// Topological order (Kahn)
const q = [];
for (const [id, deg] of indeg.entries()) if (deg === 0) q.push(id);
// Stable-ish ordering to reduce nondeterminism
q.sort();
const topo = [];
for (let qi = 0; qi < q.length; qi++) {
  const u = q[qi];
  topo.push(u);
  for (const v of out.get(u)) {
    const nd = (indeg.get(v) || 0) - 1;
    indeg.set(v, nd);
    if (nd === 0) q.push(v);
  }
}

// Cycle fallback: output any order (original ids order)
if (topo.length !== n) {
  process.stdout.write(JSON.stringify({ order: tasksIn.map(t => String(t.id)) }));
  process.exit(0);
}

// Compute bottom level (longest path to exit) in reverse topo.
const bottom = new Map();
for (let i = topo.length - 1; i >= 0; i--) {
  const u = topo[i];
  let bestChild = 0;
  for (const v of out.get(u)) {
    const bv = bottom.get(v) || 0;
    if (bv > bestChild) bestChild = bv;
  }
  bottom.set(u, (tasks.get(u).duration || 0) + bestChild);
}

// Dependency finish time tracker
const depFinish = new Map();
for (const id of ids) depFinish.set(id, 0);

// Remaining dependency count
const remDeps = new Map();
for (const id of ids) remDeps.set(id, (tasks.get(id).deps || []).length);

// Ready set
const ready = [];
for (const id of ids) if ((remDeps.get(id) || 0) === 0) ready.push(id);

// Machine availability times
const machAvail = new Array(machineCount).fill(0);

// Track finish times for tasks (needed by dependents)
const finishTime = new Map();

const schedule = [];
let scheduledCount = 0;

function argMin(arr) {
  let mi = 0;
  let mv = arr[0];
  for (let i = 1; i < arr.length; i++) {
    const v = arr[i];
    if (v < mv) {
      mv = v;
      mi = i;
    }
  }
  return mi;
}

function priorityScore(task, estStart) {
  const dur = task.duration;
  const end = estStart + dur;
  const dl = task.deadline;
  const w = task.weight;
  let urgency = 0;
  if (dl !== null) {
    const lateness = end - dl; // negative => early
    // Strongly prefer avoiding weighted tardiness; still differentiate among early tasks.
    const tard = lateness > 0 ? lateness : 0;
    urgency = tard * w * 1e6 + lateness * w * 10;
  }
  const bl = bottom.get(task.id) || 0;
  // Combine: primary urgency, secondary critical path.
  return urgency * 1000 + bl;
}

while (scheduledCount < n) {
  if (ready.length === 0) {
    // Shouldn't happen in DAG, but safety: emit topo order instead.
    process.stdout.write(JSON.stringify({ order: topo }));
    process.exit(0);
  }

  // Always schedule next task onto earliest-available machine.
  const m = argMin(machAvail);
  const mTime = machAvail[m];

  // Choose best ready task for this machine/time.
  let bestIdx = 0;
  let bestScore = -Infinity;
  let bestStart = 0;
  for (let i = 0; i < ready.length; i++) {
    const id = ready[i];
    const t = tasks.get(id);
    const rTime = depFinish.get(id) || 0;
    const estStart = rTime > mTime ? rTime : mTime;
    const sc = priorityScore(t, estStart);
    if (sc > bestScore) {
      bestScore = sc;
      bestIdx = i;
      bestStart = estStart;
    } else if (sc === bestScore) {
      // Tie-breaker: earlier start, then higher weight, then id.
      const id2 = ready[bestIdx];
      const t2 = tasks.get(id2);
      const r2 = depFinish.get(id2) || 0;
      const s2 = r2 > mTime ? r2 : mTime;
      if (estStart < s2 ||
          (estStart === s2 && (t.weight > t2.weight ||
           (t.weight === t2.weight && id < id2)))) {
        bestIdx = i;
        bestStart = estStart;
      }
    }
  }

  const id = ready.splice(bestIdx, 1)[0];
  const t = tasks.get(id);
  const start = bestStart | 0;
  const finish = start + (t.duration | 0);

  machAvail[m] = finish;
  finishTime.set(id, finish);
  scheduledCount++;

  // Only include tasks that were in the original input (ignore synthesized missing deps).
  if (tasksIn.find(x => x && String(x.id) === id)) {
    schedule.push({ id, start, machine: m });
  }

  // Update dependents
  for (const v of out.get(id)) {
    // Update depFinish[v] as max of its deps' finishes
    const cur = depFinish.get(v) || 0;
    if (finish > cur) depFinish.set(v, finish);
    const nd = (remDeps.get(v) || 0) - 1;
    remDeps.set(v, nd);
    if (nd === 0) ready.push(v);
  }
}

process.stdout.write(JSON.stringify({ schedule }));
