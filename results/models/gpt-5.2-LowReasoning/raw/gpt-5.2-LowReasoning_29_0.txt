using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;

public static class Program
{
    // Hybrid maximum clique solver:
    // - Exact (branch & bound with bitset + greedy coloring) for small graphs.
    // - Heuristic (multi-start greedy construction) for large graphs.
    //
    // Always outputs a valid clique; for small graphs attempts to prove optimality.

    public static void Main()
    {
        var scanner = new FastScanner();
        int n = scanner.NextInt();
        int m = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        var g = new Graph(n, m);
        for (int i = 0; i < m; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            g.AddEdge(u, v);
        }
        g.FinalizeBuild();

        var sw = Stopwatch.StartNew();
        // Heuristic pass first (good lower bound), then exact if feasible.
        int[] bestClique = HeuristicSolve(g, sw, softMillis: 60_000);

        // Exact for smaller graphs, using remaining time.
        // Threshold is conservative to avoid memory/time blowups.
        if (n <= 2000)
        {
            var exact = new ExactBitsetClique(g, sw, hardMillis: 295_000);
            if (bestClique.Length > 0) exact.Seed(bestClique);
            var optimal = exact.Solve();
            if (optimal.Length > bestClique.Length) bestClique = optimal;
        }

        Array.Sort(bestClique);
        Console.WriteLine(bestClique.Length);
        if (bestClique.Length == 0) Console.WriteLine();
        else Console.WriteLine(string.Join(" ", bestClique));
    }

    private static int[] HeuristicSolve(Graph g, Stopwatch sw, int softMillis)
    {
        // Multi-start greedy maximal clique construction.
        // Uses only adjacency lists (works up to 1e6 vertices).
        int n = g.N;
        int[] order = g.VerticesByDegreeDesc();
        int seeds = Math.Min(n, 20000); // cap work
        int timeLimit = softMillis;

        int[] globalBest = Array.Empty<int>();
        object gate = new object();

        int workerCount = Math.Max(1, Environment.ProcessorCount - 1);
        var cts = new CancellationTokenSource();
        var token = cts.Token;

        // Partition seeds across workers.
        int chunk = (seeds + workerCount - 1) / workerCount;
        var tasks = new Task[workerCount];
        for (int w = 0; w < workerCount; w++)
        {
            int start = w * chunk;
            int end = Math.Min(seeds, start + chunk);
            tasks[w] = Task.Run(() =>
            {
                var rand = new Random(unchecked(Environment.TickCount * 1103515245 + Thread.CurrentThread.ManagedThreadId));
                var localBest = new List<int>(64);

                // A reusable candidate list buffer.
                var candidates = new List<int>(256);

                for (int si = start; si < end && !token.IsCancellationRequested; si++)
                {
                    if (sw.ElapsedMilliseconds >= timeLimit) break;

                    int seed = order[si];
                    candidates.Clear();
                    // Start with neighbors of seed.
                    var neigh = g.Adj[seed];
                    for (int i = 0; i < neigh.Length; i++) candidates.Add(neigh[i]);

                    // Build clique greedily.
                    var clique = new List<int>(64) { seed };
                    while (candidates.Count > 0)
                    {
                        // Pick vertex with highest degree among candidates (ties randomized lightly).
                        int bestIdx = -1;
                        int bestScore = -1;
                        int sample = Math.Min(candidates.Count, 16);
                        // Sample a few to avoid O(k) scans on large candidate sets.
                        for (int t = 0; t < sample; t++)
                        {
                            int idx = (candidates.Count <= sample) ? t : rand.Next(candidates.Count);
                            int v = candidates[idx];
                            int score = g.Deg[v];
                            if (score > bestScore)
                            {
                                bestScore = score;
                                bestIdx = idx;
                            }
                        }
                        if (bestIdx < 0) break;
                        int vPick = candidates[bestIdx];
                        clique.Add(vPick);

                        // Filter candidates to those adjacent to vPick.
                        // Intersect current candidates with N(vPick) using two-pointer on sorted arrays.
                        var nv = g.Adj[vPick];
                        candidates = IntersectInPlace(candidates, nv);
                    }

                    if (clique.Count > localBest.Count)
                    {
                        localBest.Clear();
                        localBest.AddRange(clique);
                    }

                    // Occasional update to global best.
                    if (localBest.Count > Volatile.Read(ref _bestLen))
                    {
                        lock (gate)
                        {
                            if (localBest.Count > globalBest.Length)
                                globalBest = localBest.ToArray();
                        }
                        Volatile.Write(ref _bestLen, globalBest.Length);
                    }
                }
            }, token);
        }

        // Also run a random-walk on the ordering for a bit if time allows.
        var jitterTask = Task.Run(() =>
        {
            var rand = new Random(unchecked(Environment.TickCount * 2654435761));
            var candidates = new List<int>(256);
            var localBest = new List<int>(64);
            while (!token.IsCancellationRequested && sw.ElapsedMilliseconds < timeLimit)
            {
                int seed = order[rand.Next(seeds)];
                candidates.Clear();
                var neigh = g.Adj[seed];
                for (int i = 0; i < neigh.Length; i++) candidates.Add(neigh[i]);

                var clique = new List<int>(64) { seed };
                // Randomized greedy: sometimes pick random from top few.
                while (candidates.Count > 0)
                {
                    int pick = PickTopKByDegree(candidates, g.Deg, rand, topK: 5);
                    int vPick = candidates[pick];
                    clique.Add(vPick);
                    candidates = IntersectInPlace(candidates, g.Adj[vPick]);
                }

                if (clique.Count > localBest.Count)
                {
                    localBest.Clear();
                    localBest.AddRange(clique);
                    if (localBest.Count > Volatile.Read(ref _bestLen))
                    {
                        lock (gate)
                        {
                            if (localBest.Count > globalBest.Length)
                                globalBest = localBest.ToArray();
                        }
                        Volatile.Write(ref _bestLen, globalBest.Length);
                    }
                }
            }
        }, token);

        // Wait until soft time limit.
        while (sw.ElapsedMilliseconds < timeLimit) Thread.Sleep(5);
        cts.Cancel();
        try { Task.WaitAll(tasks.Concat(new[] { jitterTask }).ToArray(), 2000); } catch { /* ignore */ }

        if (globalBest.Length == 0)
        {
            // Always a clique: single vertex.
            return new[] { 0 };
        }
        return globalBest;
    }

    private static int _bestLen = 0;

    private static int PickTopKByDegree(List<int> candidates, int[] deg, Random rand, int topK)
    {
        // Return index of one of the topK by degree (approx via partial selection).
        if (candidates.Count <= 1) return 0;
        int k = Math.Min(topK, candidates.Count);
        // Sample up to 24 candidates to keep it cheap.
        int sample = Math.Min(candidates.Count, 24);
        var sampled = new (int score, int idx)[sample];
        for (int i = 0; i < sample; i++)
        {
            int idx = (candidates.Count <= sample) ? i : rand.Next(candidates.Count);
            sampled[i] = (deg[candidates[idx]], idx);
        }
        sampled.SortInPlaceDesc();
        // Pick randomly among top k of sampled.
        int pickRank = rand.Next(Math.Min(k, sample));
        return sampled[pickRank].idx;
    }

    private static List<int> IntersectInPlace(List<int> current, int[] sortedNeighbor)
    {
        // Both are sorted (Graph.FinalizeBuild sorts adjacency; current derived from sorted arrays and intersections).
        // Produce intersection into the same list and return it (possibly same instance).
        int i = 0, j = 0, outIdx = 0;
        // Ensure current is sorted. Most of the time it already is, but after random picks it remains sorted since we never reorder.
        // (We only remove elements, not rearrange.)
        while (i < current.Count && j < sortedNeighbor.Length)
        {
            int a = current[i];
            int b = sortedNeighbor[j];
            if (a == b)
            {
                current[outIdx++] = a;
                i++; j++;
            }
            else if (a < b) i++;
            else j++;
        }
        if (outIdx < current.Count) current.RemoveRange(outIdx, current.Count - outIdx);
        return current;
    }

    private sealed class Graph
    {
        public int N { get; }
        public int[] Deg { get; }
        public int[][] Adj { get; private set; }

        private readonly int[] _head;
        private readonly int[] _to;
        private readonly int[] _next;
        private int _edgePtr;

        public Graph(int n, int m)
        {
            N = n;
            Deg = new int[n];
            // store as forward star with capacity 2m
            _head = new int[n];
            Array.Fill(_head, -1);
            _to = new int[Math.Max(2, 2 * m)];
            _next = new int[Math.Max(2, 2 * m)];
            _edgePtr = 0;
            Adj = Array.Empty<int[]>();
        }

        public void AddEdge(int u, int v)
        {
            AddArc(u, v);
            AddArc(v, u);
            Deg[u]++; Deg[v]++;
        }

        private void AddArc(int u, int v)
        {
            if (_edgePtr >= _to.Length) return; // if input lies, ignore extras
            _to[_edgePtr] = v;
            _next[_edgePtr] = _head[u];
            _head[u] = _edgePtr++;
        }

        public void FinalizeBuild()
        {
            Adj = new int[N][];
            for (int u = 0; u < N; u++)
            {
                int d = Deg[u];
                var arr = new int[d];
                int idx = 0;
                for (int e = _head[u]; e != -1; e = _next[e])
                    arr[idx++] = _to[e];
                Array.Sort(arr);
                // de-dup (rarely needed, but safe)
                int w = 0;
                for (int i = 0; i < arr.Length; i++)
                {
                    if (i == 0 || arr[i] != arr[i - 1]) arr[w++] = arr[i];
                }
                if (w != arr.Length) Array.Resize(ref arr, w);
                Adj[u] = arr;
                Deg[u] = arr.Length;
            }
        }

        public int[] VerticesByDegreeDesc()
        {
            int[] v = new int[N];
            for (int i = 0; i < N; i++) v[i] = i;
            Array.Sort(v, (a, b) =>
            {
                int da = Deg[a], db = Deg[b];
                int c = db.CompareTo(da);
                return c != 0 ? c : a.CompareTo(b);
            });
            return v;
        }
    }

    private sealed class ExactBitsetClique
    {
        private readonly Graph _g;
        private readonly Stopwatch _sw;
        private readonly int _hardMillis;
        private readonly int _n;
        private readonly int _words;
        private readonly ulong[][] _adjBits; // [v][word]

        private int[] _best = Array.Empty<int>();

        public ExactBitsetClique(Graph g, Stopwatch sw, int hardMillis)
        {
            _g = g;
            _sw = sw;
            _hardMillis = hardMillis;
            _n = g.N;
            _words = (_n + 63) >> 6;
            _adjBits = BuildAdjBits(g);
        }

        public void Seed(int[] clique)
        {
            if (clique != null && clique.Length > _best.Length)
                _best = (int[])clique.Clone();
        }

        public int[] Solve()
        {
            // Initial candidate set: all vertices.
            var all = new ulong[_words];
            for (int w = 0; w < _words; w++) all[w] = ulong.MaxValue;
            int extra = (_words << 6) - _n;
            if (extra > 0) all[_words - 1] >>= extra;

            var r = new List<int>(64);
            Expand(all, r);
            return _best;
        }

        private void Expand(ulong[] p, List<int> r)
        {
            if (_sw.ElapsedMilliseconds >= _hardMillis) return;
            int pCount = PopCount(p);
            if (pCount == 0)
            {
                if (r.Count > _best.Length) _best = r.ToArray();
                return;
            }

            // Coloring bound on p to prune.
            var (order, colors) = ColorSort(p);
            for (int idx = order.Count - 1; idx >= 0; idx--)
            {
                if (_sw.ElapsedMilliseconds >= _hardMillis) return;

                int v = order[idx];
                int bound = r.Count + colors[idx];
                if (bound <= _best.Length) return; // all earlier have <= this color

                // r U {v}
                r.Add(v);
                var p2 = AndClone(p, _adjBits[v]);
                Expand(p2, r);
                r.RemoveAt(r.Count - 1);

                // p = p \ {v}
                ClearBit(p, v);
            }
        }

        private (List<int> order, int[] colors) ColorSort(ulong[] p)
        {
            // Greedy sequential coloring on subgraph induced by p.
            // Returns vertices in coloring order and an array with color numbers (1..k),
            // such that colors[i] is the color count up to i (non-decreasing).
            var vs = BitsToVertices(p);
            // Order by degree within p (approx via global degree) to improve coloring.
            vs.Sort((a, b) => _g.Deg[b].CompareTo(_g.Deg[a]));

            var colorOf = new int[vs.Count];
            var colorSets = new List<ulong[]>();

            for (int i = 0; i < vs.Count; i++)
            {
                int v = vs[i];
                int c = 0;
                for (; c < colorSets.Count; c++)
                {
                    // can we place v in color c (no edge with any vertex already in that color)?
                    // Equivalent: (colorSet & ~N(v)) == 0, but we have N(v) bits.
                    if (!AnyOverlap(colorSets[c], _adjBits[v]))
                    {
                        SetBit(colorSets[c], v);
                        break;
                    }
                }
                if (c == colorSets.Count)
                {
                    var set = new ulong[_words];
                    SetBit(set, v);
                    colorSets.Add(set);
                }
                colorOf[i] = c + 1;
            }

            // Convert to Tomita style: order is vs, colors[i] = max color among 0..i.
            int max = 0;
            var colors = new int[vs.Count];
            for (int i = 0; i < vs.Count; i++)
            {
                if (colorOf[i] > max) max = colorOf[i];
                colors[i] = max;
            }
            return (vs, colors);
        }

        private static ulong[][] BuildAdjBits(Graph g)
        {
            int n = g.N;
            int words = (n + 63) >> 6;
            var adj = new ulong[n][];
            for (int v = 0; v < n; v++)
            {
                var bits = new ulong[words];
                var neigh = g.Adj[v];
                for (int i = 0; i < neigh.Length; i++)
                {
                    int u = neigh[i];
                    bits[u >> 6] |= 1UL << (u & 63);
                }
                adj[v] = bits;
            }
            return adj;
        }

        private List<int> BitsToVertices(ulong[] bits)
        {
            var list = new List<int>(128);
            for (int w = 0; w < bits.Length; w++)
            {
                ulong x = bits[w];
                while (x != 0)
                {
                    int t = BitOperations.TrailingZeroCount(x);
                    int v = (w << 6) + t;
                    if (v < _n) list.Add(v);
                    x &= x - 1;
                }
            }
            return list;
        }

        private static int PopCount(ulong[] bits)
        {
            int c = 0;
            for (int i = 0; i < bits.Length; i++) c += BitOperations.PopCount(bits[i]);
            return c;
        }

        private static bool AnyOverlap(ulong[] a, ulong[] b)
        {
            for (int i = 0; i < a.Length; i++)
                if ((a[i] & b[i]) != 0) return true;
            return false;
        }

        private static ulong[] AndClone(ulong[] a, ulong[] b)
        {
            var r = new ulong[a.Length];
            for (int i = 0; i < a.Length; i++) r[i] = a[i] & b[i];
            return r;
        }

        private static void ClearBit(ulong[] bits, int v)
        {
            int w = v >> 6;
            bits[w] &= ~(1UL << (v & 63));
        }

        private static void SetBit(ulong[] bits, int v)
        {
            int w = v >> 6;
            bits[w] |= 1UL << (v & 63);
        }
    }

    private sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c != -1);
            int sgn = 1;
            if (c == '-')
            {
                sgn = -1;
                c = Read();
            }
            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }
            return x * sgn;
        }
    }
}

internal static class BitOperations
{
    // .NET's System.Numerics.BitOperations is available, but defining minimal wrappers
    // avoids version quirks in some csc environments.
    public static int PopCount(ulong x)
    {
        // Hamming weight (builtin via intrinsics when available).
        x -= (x >> 1) & 0x5555555555555555UL;
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        return (int)((((x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FUL) * 0x0101010101010101UL) >> 56);
    }

    public static int TrailingZeroCount(ulong x)
    {
        // de Bruijn
        if (x == 0) return 64;
        const ulong debruijn = 0x03f79d71b4cb0a89UL;
        ReadOnlySpan<byte> idx = new byte[64]
        {
            0, 1, 48, 2, 57, 49, 28, 3,
            61, 58, 50, 42, 38, 29, 17, 4,
            62, 55, 59, 36, 53, 51, 43, 22,
            45, 39, 33, 30, 24, 18, 12, 5,
            63, 47, 56, 27, 60, 41, 37, 16,
            54, 35, 52, 21, 44, 32, 23, 11,
            46, 26, 40, 15, 34, 20, 31, 10,
            25, 14, 19, 9, 13, 8, 7, 6
        };
        ulong lsb = x & (ulong)-(long)x;
        return idx[(int)((lsb * debruijn) >> 58)];
    }
}

internal static class Extensions
{
    public static void SortInPlaceDesc(this (int score, int idx)[] a)
    {
        Array.Sort(a, (x, y) =>
        {
            int c = y.score.CompareTo(x.score);
            return c != 0 ? c : x.idx.CompareTo(y.idx);
        });
    }
}
