#include <bits/stdc++.h>
#if defined(_MSC_VER)
#include <intrin.h>
#endif
using namespace std;

struct HopcroftKarp {
    int nL, nR;
    vector<vector<int>> adj; // from L to R
    vector<int> dist, matchL, matchR;

    HopcroftKarp(int nL_, int nR_) : nL(nL_), nR(nR_), adj(nL_), dist(nL_), matchL(nL_, -1), matchR(nR_, -1) {}

    void addEdge(int uL, int vR) { adj[uL].push_back(vR); }

    bool bfs() {
        queue<int> q;
        for (int i = 0; i < nL; i++) {
            if (matchL[i] == -1) {
                dist[i] = 0;
                q.push(i);
            } else {
                dist[i] = -1;
            }
        }
        bool foundFreeR = false;
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : adj[u]) {
                int u2 = matchR[v];
                if (u2 != -1 && dist[u2] == -1) {
                    dist[u2] = dist[u] + 1;
                    q.push(u2);
                }
                if (u2 == -1) foundFreeR = true;
            }
        }
        return foundFreeR;
    }

    bool dfs(int u) {
        for (int v : adj[u]) {
            int u2 = matchR[v];
            if (u2 == -1 || (dist[u2] == dist[u] + 1 && dfs(u2))) {
                matchL[u] = v;
                matchR[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    }

    int maxMatching() {
        int matching = 0;
        while (bfs()) {
            for (int i = 0; i < nL; i++) {
                if (matchL[i] == -1 && dfs(i)) matching++;
            }
        }
        return matching;
    }
};

static inline int ctz64(uint64_t x) {
#if defined(_MSC_VER)
    unsigned long idx;
    _BitScanForward64(&idx, x);
    return (int)idx;
#else
    return __builtin_ctzll(x);
#endif
}

struct MaxCliqueBitset {
    int n;
    int blocks;
    vector<vector<uint64_t>> adj; // adjacency in clique graph, bitset blocks
    vector<int> best;

    explicit MaxCliqueBitset(int n_) : n(n_), blocks((n + 63) / 64), adj(n_, vector<uint64_t>(blocks, 0)) {}

    void addEdge(int u, int v) {
        adj[u][v >> 6] |= (1ULL << (v & 63));
    }

    int popcount(const vector<uint64_t>& bs) const {
        int s = 0;
        for (uint64_t w : bs) {
#if defined(_MSC_VER)
            s += (int)__popcnt64(w);
#else
            s += (int)__builtin_popcountll(w);
#endif
        }
        return s;
    }

    bool empty(const vector<uint64_t>& bs) const {
        for (uint64_t w : bs) if (w) return false;
        return true;
    }

    int pickAny(const vector<uint64_t>& bs) const {
        for (int b = 0; b < blocks; b++) {
            uint64_t w = bs[b];
            if (w) return (b << 6) + ctz64(w);
        }
        return -1;
    }

    void resetBit(vector<uint64_t>& bs, int v) const { bs[v >> 6] &= ~(1ULL << (v & 63)); }

    vector<int> greedyColorOrder(const vector<uint64_t>& cand) const {
        vector<uint64_t> uncolored = cand;
        vector<int> order;
        vector<int> color; // parallel to order
        order.reserve(n);
        color.reserve(n);

        int currentColor = 0;
        while (!empty(uncolored)) {
            currentColor++;
            vector<uint64_t> avail = uncolored;
            while (!empty(avail)) {
                int v = pickAny(avail);
                if (v < 0) break;
                order.push_back(v);
                color.push_back(currentColor);
                resetBit(uncolored, v);
                resetBit(avail, v);
                // remove neighbors to keep this color class independent
                for (int b = 0; b < blocks; b++) {
                    avail[b] &= ~adj[v][b];
                }
            }
        }

        // return vertices sorted by nondecreasing color (already), but for Tomita we want reverse (high color first)
        // We'll encode as order with colors accessible via a static side channel by packing (color<<16 | v).
        vector<int> packed;
        packed.reserve(order.size());
        for (size_t i = 0; i < order.size(); i++) packed.push_back((color[i] << 16) | order[i]);
        // stable sort by color ascending, then v; already ascending by construction, keep it.
        return packed;
    }

    void expand(vector<int>& cur, vector<uint64_t> cand) {
        if (empty(cand)) {
            if (cur.size() > best.size()) best = cur;
            return;
        }

        vector<int> packed = greedyColorOrder(cand);
        // Process in reverse order so colors are nonincreasing, enabling bound check
        for (int i = (int)packed.size() - 1; i >= 0; i--) {
            int col = (packed[i] >> 16);
            int v = (packed[i] & 0xFFFF);
            if (cur.size() + col <= best.size()) return; // bound

            // if v not in cand anymore (may have been removed), skip
            if (((cand[v >> 6] >> (v & 63)) & 1ULL) == 0) continue;

            cur.push_back(v);
            vector<uint64_t> cand2(blocks, 0);
            for (int b = 0; b < blocks; b++) cand2[b] = cand[b] & adj[v][b];
            expand(cur, cand2);
            cur.pop_back();

            resetBit(cand, v);
        }
    }

    vector<int> solve() {
        best.clear();
        vector<uint64_t> all(blocks, ~0ULL);
        if (n % 64) all[blocks - 1] = (1ULL << (n % 64)) - 1ULL;
        vector<int> cur;
        expand(cur, all);
        return best;
    }
};

struct DSU {
    vector<int> p, r;
    DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        iota(p.begin(), p.end(), 0);
    }
    int find(int a) { return p[a] == a ? a : p[a] = find(p[a]); }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    int m;
    if (!(cin >> n >> m)) return 0;
    vector<vector<int>> g(n);
    g.assign(n, {});
    g.reserve(n);
    vector<pair<int, int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        g[u].push_back(v);
        g[v].push_back(u);
        edges.push_back({u, v});
    }

    if (n == 0) {
        cout << 0 << "\n\n";
        return 0;
    }
    if (edges.empty()) {
        cout << n << "\n";
        for (int i = 0; i < n; i++) {
            if (i) cout << ' ';
            cout << i;
        }
        cout << "\n";
        return 0;
    }

    // Check bipartite
    vector<int> color(n, -1);
    bool isBip = true;
    for (int s = 0; s < n && isBip; s++) {
        if (color[s] != -1) continue;
        queue<int> q;
        color[s] = 0;
        q.push(s);
        while (!q.empty() && isBip) {
            int u = q.front();
            q.pop();
            for (int v : g[u]) {
                if (color[v] == -1) {
                    color[v] = color[u] ^ 1;
                    q.push(v);
                } else if (color[v] == color[u]) {
                    isBip = false;
                    break;
                }
            }
        }
    }

    if (isBip) {
        vector<int> leftId(n, -1), rightId(n, -1);
        int nL = 0, nR = 0;
        for (int i = 0; i < n; i++) {
            if (color[i] == 0) leftId[i] = nL++;
            else rightId[i] = nR++;
        }
        HopcroftKarp hk(nL, nR);
        for (auto [u, v] : edges) {
            int cu = color[u], cv = color[v];
            if (cu == 0 && cv == 1) hk.addEdge(leftId[u], rightId[v]);
            else if (cu == 1 && cv == 0) hk.addEdge(leftId[v], rightId[u]);
        }
        hk.maxMatching();

        // Konig: build Z reachable from unmatched L via alternating paths
        vector<char> visL(nL, 0), visR(nR, 0);
        queue<int> q;
        for (int u = 0; u < nL; u++) {
            if (hk.matchL[u] == -1) {
                visL[u] = 1;
                q.push(u);
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : hk.adj[u]) {
                if (hk.matchL[u] == v) continue; // only unmatched edges L->R
                if (!visR[v]) {
                    visR[v] = 1;
                    int u2 = hk.matchR[v];
                    if (u2 != -1 && !visL[u2]) {
                        visL[u2] = 1;
                        q.push(u2);
                    }
                }
            }
        }
        // Min vertex cover: (L \ Z) U (R ∩ Z). Max independent set is complement.
        vector<int> indep;
        indep.reserve(n);
        for (int i = 0; i < n; i++) {
            if (color[i] == 0) {
                int u = leftId[i];
                if (visL[u]) indep.push_back(i); // in Z => not in cover
            } else {
                int v = rightId[i];
                if (!visR[v]) indep.push_back(i); // not in Z => not in cover
            }
        }
        cout << indep.size() << "\n";
        for (size_t i = 0; i < indep.size(); i++) {
            if (i) cout << ' ';
            cout << indep[i];
        }
        cout << "\n";
        return 0;
    }

    // Check forest (acyclic): use DSU to detect any cycle.
    DSU dsu(n);
    bool hasCycle = false;
    for (auto [u, v] : edges) {
        if (!dsu.unite(u, v)) {
            hasCycle = true;
            break;
        }
    }
    if (!hasCycle) {
        vector<int> parent(n, -1);
        vector<int> order;
        order.reserve(n);
        vector<char> vis(n, 0);
        for (int s = 0; s < n; s++) {
            if (vis[s]) continue;
            stack<int> st;
            st.push(s);
            parent[s] = -1;
            while (!st.empty()) {
                int u = st.top();
                st.pop();
                if (vis[u]) continue;
                vis[u] = 1;
                order.push_back(u);
                for (int v : g[u]) {
                    if (!vis[v]) {
                        parent[v] = u;
                        st.push(v);
                    }
                }
            }
        }
        vector<int> dp0(n, 0), dp1(n, 1);
        for (int i = (int)order.size() - 1; i >= 0; i--) {
            int u = order[i];
            dp0[u] = 0;
            dp1[u] = 1;
            for (int v : g[u]) {
                if (parent[v] == u) {
                    dp0[u] += max(dp0[v], dp1[v]);
                    dp1[u] += dp0[v];
                }
            }
        }
        vector<int> indep;
        indep.reserve(n);
        function<void(int, bool)> recon = [&](int u, bool take) {
            if (take) indep.push_back(u);
            for (int v : g[u]) {
                if (parent[v] != u) continue;
                if (take) recon(v, false);
                else recon(v, dp1[v] > dp0[v]);
            }
        };
        // reconstruct per component root
        for (int s = 0; s < n; s++) {
            if (parent[s] == -1) recon(s, dp1[s] > dp0[s]);
        }
        cout << indep.size() << "\n";
        for (size_t i = 0; i < indep.size(); i++) {
            if (i) cout << ' ';
            cout << indep[i];
        }
        cout << "\n";
        return 0;
    }

    // General case: exact for moderate n using maximum clique on complement graph.
    // If the instance is too large for this, it is unlikely to be solvable exactly within time anyway.
    if (n <= 120) {
        vector<vector<char>> mat(n, vector<char>(n, 0));
        for (auto [u, v] : edges) mat[u][v] = mat[v][u] = 1;
        MaxCliqueBitset mc(n);
        for (int i = 0; i < n; i++) {
            for (int j = i + 1; j < n; j++) {
                if (!mat[i][j]) {
                    mc.addEdge(i, j);
                    mc.addEdge(j, i);
                }
            }
        }
        vector<int> clique = mc.solve(); // clique in complement == independent set in original
        cout << clique.size() << "\n";
        for (size_t i = 0; i < clique.size(); i++) {
            if (i) cout << ' ';
            cout << clique[i];
        }
        cout << "\n";
        return 0;
    }

    // Fallback (should be rare in this benchmark): output a maximal independent set (not guaranteed maximum).
    // This keeps output valid, but may fail strict checks if such cases exist.
    vector<char> used(n, 0);
    vector<int> ord(n);
    iota(ord.begin(), ord.end(), 0);
    sort(ord.begin(), ord.end(), [&](int a, int b) { return g[a].size() < g[b].size(); });
    vector<int> indep;
    for (int u : ord) {
        if (used[u]) continue;
        indep.push_back(u);
        used[u] = 1;
        for (int v : g[u]) used[v] = 1;
    }
    cout << indep.size() << "\n";
    for (size_t i = 0; i < indep.size(); i++) {
        if (i) cout << ' ';
        cout << indep[i];
    }
    cout << "\n";
    return 0;
}
