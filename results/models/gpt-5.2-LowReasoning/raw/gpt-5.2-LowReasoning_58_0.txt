const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = H ? grid[0].length : 0;
const N = H * W;

function idx(x, y) {
  return y * W + x;
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}

// Precompute walkable cells
const walkable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    walkable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
  }
}

function bfsFrom(sx, sy) {
  const dist = new Int32Array(N);
  dist.fill(-1);
  if (!inBounds(sx, sy)) return dist;
  const s = idx(sx, sy);
  if (!walkable[s]) return dist;

  // Simple array queue (ring not necessary; at most N pushes)
  const qx = new Int32Array(N);
  const qy = new Int32Array(N);
  let head = 0;
  let tail = 0;

  qx[tail] = sx;
  qy[tail] = sy;
  tail++;
  dist[s] = 0;

  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;
    const base = dist[idx(x, y)];
    const nd = base + 1;

    // 4-neighborhood
    let nx = x + 1, ny = y;
    if (inBounds(nx, ny)) {
      const ni = idx(nx, ny);
      if (walkable[ni] && dist[ni] === -1) {
        dist[ni] = nd;
        qx[tail] = nx; qy[tail] = ny; tail++;
      }
    }
    nx = x - 1; ny = y;
    if (inBounds(nx, ny)) {
      const ni = idx(nx, ny);
      if (walkable[ni] && dist[ni] === -1) {
        dist[ni] = nd;
        qx[tail] = nx; qy[tail] = ny; tail++;
      }
    }
    nx = x; ny = y + 1;
    if (inBounds(nx, ny)) {
      const ni = idx(nx, ny);
      if (walkable[ni] && dist[ni] === -1) {
        dist[ni] = nd;
        qx[tail] = nx; qy[tail] = ny; tail++;
      }
    }
    nx = x; ny = y - 1;
    if (inBounds(nx, ny)) {
      const ni = idx(nx, ny);
      if (walkable[ni] && dist[ni] === -1) {
        dist[ni] = nd;
        qx[tail] = nx; qy[tail] = ny; tail++;
      }
    }
  }

  return dist;
}

// Determine assignment array length (array index is robot id).
let maxRobotId = -1;
for (const r of robots) {
  if (typeof r.id === 'number' && r.id > maxRobotId) maxRobotId = r.id;
}
const robotCount = Math.max(maxRobotId + 1, robots.length);
const assignments = Array.from({ length: robotCount }, () => []);

// Precompute distance map from each robot start position once.
const robotDist = new Array(robotCount).fill(null);
for (const r of robots) {
  if (typeof r.id !== 'number') continue;
  robotDist[r.id] = bfsFrom(r.x | 0, r.y | 0);
}
// For any missing robot ids, use an all -1 dist map.
for (let i = 0; i < robotCount; i++) {
  if (!robotDist[i]) {
    const d = new Int32Array(N);
    d.fill(-1);
    robotDist[i] = d;
  }
}

// Deterministic greedy:
// 1) Process jobs by priority desc, then by id asc (numeric if possible, else string).
// 2) Assign each job to robot that minimizes distance from robot start; tie by robot id.
function jobIdKey(j) {
  const id = j.id;
  if (typeof id === 'number') return { t: 0, n: id, s: '' };
  const asNum = Number(id);
  if (Number.isFinite(asNum) && String(asNum) === String(id).trim()) return { t: 0, n: asNum, s: '' };
  return { t: 1, n: 0, s: String(id) };
}

const jobsSorted = jobs.slice();
jobsSorted.sort((a, b) => {
  const pa = (a.priority ?? 0);
  const pb = (b.priority ?? 0);
  if (pa !== pb) return pb - pa;
  const ka = jobIdKey(a);
  const kb = jobIdKey(b);
  if (ka.t !== kb.t) return ka.t - kb.t;
  if (ka.t === 0) return ka.n - kb.n;
  return ka.s < kb.s ? -1 : ka.s > kb.s ? 1 : 0;
});

for (const j of jobsSorted) {
  const x = j.x | 0;
  const y = j.y | 0;
  const ji = inBounds(x, y) ? idx(x, y) : -1;

  let bestRobot = 0;
  let bestDist = 0x3fffffff;
  for (let rid = 0; rid < robotCount; rid++) {
    const d = ji >= 0 ? robotDist[rid][ji] : -1;
    const dd = d >= 0 ? d : 0x3fffffff;
    if (dd < bestDist) {
      bestDist = dd;
      bestRobot = rid;
    }
  }
  assignments[bestRobot].push(j.id);
}

process.stdout.write(JSON.stringify({ assignments }));
