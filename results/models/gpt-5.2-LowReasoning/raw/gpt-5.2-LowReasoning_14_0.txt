using System;
using System.Buffers;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private const byte Unknown = 0;
    private const byte Mine = 1;     // flagged or logically deduced mine
    private const byte Revealed = 2; // digit 0-8

    private static readonly int[] Dx = new int[] { -1, 0, 1, -1, 1, -1, 0, 1 };
    private static readonly int[] Dy = new int[] { -1, -1, -1, 0, 0, 1, 1, 1 };

    public static void Main()
    {
        // Read H W
        var first = Console.ReadLine();
        if (first == null)
            return;

        var parts = first.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2)
            return;

        int H = int.Parse(parts[0]);
        int W = int.Parse(parts[1]);
        long nCellsLong = (long)H * (long)W;
        if (nCellsLong <= 0 || nCellsLong > int.MaxValue)
        {
            // Boards larger than 2^31-1 cells are not realistic for this environment.
            Console.WriteLine("0");
            return;
        }
        int nCells = (int)nCellsLong;

        byte[] state = new byte[nCells];
        byte[] number = new byte[nCells]; // only valid when state==Revealed; else undefined
        var revealedList = new List<int>(Math.Min(nCells, 1 << 20));

        // Parse board
        for (int y = 0; y < H; y++)
        {
            string? line = Console.ReadLine();
            if (line == null) line = "";
            if (line.Length < W) line = line.PadRight(W, ' ');
            else if (line.Length > W) line = line.Substring(0, W);

            int rowBase = y * W;
            for (int x = 0; x < W; x++)
            {
                char c = line[x];
                int idx = rowBase + x;
                if (c == '*')
                {
                    state[idx] = Mine;
                }
                else if (c >= '0' && c <= '8')
                {
                    state[idx] = Revealed;
                    number[idx] = (byte)(c - '0');
                    revealedList.Add(idx);
                }
                else
                {
                    state[idx] = Unknown; // includes space and any other char treated as unknown
                }
            }
        }

        // Heuristic complexity classification
        int revealedCount = revealedList.Count;
        bool hugeBoard = nCells >= 10_000_000; // 10 million+ cells => keep reasoning simple/streaming
        bool allowExact = !hugeBoard && revealedCount <= 2_000_000;

        // Work queues and flags
        var q = new IntQueue(Math.Max(1024, revealedCount / 4));
        bool[] inQueue = new bool[revealedCount > 0 ? nCells : 1]; // if no revealed, keep tiny
        for (int i = 0; i < revealedList.Count; i++)
        {
            int idx = revealedList[i];
            q.Enqueue(idx);
            inQueue[idx] = true;
        }

        var safeOut = new List<int>(1024);

        // First pass: linear-time deterministic propagation (single-cell rules)
        PropagateSingles(H, W, state, number, q, inQueue, safeOut);

        // Second pass (adaptive): frontier component exact inference for small-ish problems
        if (allowExact)
        {
            // Build and solve constraint components near the frontier; this can deduce additional safe cells
            // beyond single-cell rules without guessing.
            DeduceByComponents(H, W, state, number, safeOut);

            // Re-run single propagation to cascade any new mine/safe info inferred.
            q.Clear();
            Array.Clear(inQueue, 0, inQueue.Length);
            for (int i = 0; i < revealedList.Count; i++)
            {
                int idx = revealedList[i];
                q.Enqueue(idx);
                inQueue[idx] = true;
            }
            PropagateSingles(H, W, state, number, q, inQueue, safeOut);
        }

        // Output all provably safe unknown cells discovered
        // De-duplicate: a cell may be added multiple times to safeOut.
        // Use a mark array for safety outputs only (avoid full HashSet overhead on huge boards).
        var outMark = new bool[nCells];
        int unique = 0;
        for (int i = 0; i < safeOut.Count; i++)
        {
            int idx = safeOut[i];
            if (!outMark[idx] && state[idx] == Unknown)
            {
                outMark[idx] = true;
                unique++;
            }
        }

        var sb = new StringBuilder(Math.Min(unique, 200000) * 8);
        sb.Append(unique).Append('\n');
        for (int i = 0; i < safeOut.Count; i++)
        {
            int idx = safeOut[i];
            if (!outMark[idx]) continue;
            outMark[idx] = false; // consume mark
            int y = idx / W;
            int x = idx - y * W;
            sb.Append(x).Append(' ').Append(y).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    private static void PropagateSingles(int H, int W, byte[] state, byte[] number, IntQueue q, bool[] inQueue, List<int> safeOut)
    {
        while (q.Count > 0)
        {
            int idx = q.Dequeue();
            inQueue[idx] = false;
            if (state[idx] != Revealed) continue;

            int y = idx / W;
            int x = idx - y * W;

            int mines = 0;
            int unknownCount = 0;

            // First scan neighbors to count
            for (int k = 0; k < 8; k++)
            {
                int nx = x + Dx[k];
                int ny = y + Dy[k];
                if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                int nidx = ny * W + nx;
                byte st = state[nidx];
                if (st == Mine) mines++;
                else if (st == Unknown) unknownCount++;
            }

            int need = number[idx] - mines;
            if (need < 0 || need > unknownCount) continue;
            if (unknownCount == 0) continue;

            if (need == 0)
            {
                // All unknown neighbors are safe
                for (int k = 0; k < 8; k++)
                {
                    int nx = x + Dx[k];
                    int ny = y + Dy[k];
                    if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                    int nidx = ny * W + nx;
                    if (state[nidx] == Unknown)
                    {
                        safeOut.Add(nidx);
                        // Mark as still Unknown; we only output, but "safe" knowledge helps no further unless we treat as revealed.
                        // We cannot reveal numbers ourselves, so do not change state.
                    }
                }
            }
            else if (need == unknownCount)
            {
                // All unknown neighbors must be mines
                for (int k = 0; k < 8; k++)
                {
                    int nx = x + Dx[k];
                    int ny = y + Dy[k];
                    if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                    int nidx = ny * W + nx;
                    if (state[nidx] == Unknown)
                    {
                        state[nidx] = Mine;
                        // Any revealed neighbors of this newly deduced mine might become solvable
                        EnqueueNeighborRevealed(H, W, state, q, inQueue, nidx);
                    }
                }
            }
        }
    }

    private static void EnqueueNeighborRevealed(int H, int W, byte[] state, IntQueue q, bool[] inQueue, int idx)
    {
        int y = idx / W;
        int x = idx - y * W;
        for (int k = 0; k < 8; k++)
        {
            int nx = x + Dx[k];
            int ny = y + Dy[k];
            if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
            int nidx = ny * W + nx;
            if (state[nidx] == Revealed && !inQueue[nidx])
            {
                inQueue[nidx] = true;
                q.Enqueue(nidx);
            }
        }
    }

    // Component-based exact reasoning on the unknown frontier.
    // Strategy:
    // - Build constraints from revealed cells adjacent to unknowns: sum(unknown vars in neighbor set) = need
    // - Partition into connected components (via shared variables)
    // - For each small component, enumerate all satisfying mine assignments (backtracking)
    // - If a variable is mine in all solutions => mark Mine; if safe in all solutions => output safe
    private static void DeduceByComponents(int H, int W, byte[] state, byte[] number, List<int> safeOut)
    {
        // Build frontier variable ids only for unknown cells adjacent to at least one revealed cell.
        // Using dictionaries only on frontier keeps memory reasonable.
        var varId = new Dictionary<int, int>(capacity: 65536);
        var vars = new List<int>(65536); // idx per var
        var constraints = new List<Constraint>(65536);

        // Collect constraints
        for (int y = 0; y < H; y++)
        {
            int rowBase = y * W;
            for (int x = 0; x < W; x++)
            {
                int idx = rowBase + x;
                if (state[idx] != Revealed) continue;

                int mines = 0;
                Span<int> tmp = stackalloc int[8];
                int tmpCount = 0;
                for (int k = 0; k < 8; k++)
                {
                    int nx = x + Dx[k];
                    int ny = y + Dy[k];
                    if ((uint)nx >= (uint)W || (uint)ny >= (uint)H) continue;
                    int nidx = ny * W + nx;
                    byte st = state[nidx];
                    if (st == Mine) mines++;
                    else if (st == Unknown)
                    {
                        tmp[tmpCount++] = nidx;
                    }
                }

                if (tmpCount == 0) continue;
                int need = number[idx] - mines;
                if (need < 0 || need > tmpCount) continue;

                var varList = new int[tmpCount];
                for (int i = 0; i < tmpCount; i++)
                {
                    int cellIdx = tmp[i];
                    if (!varId.TryGetValue(cellIdx, out int id))
                    {
                        id = vars.Count;
                        varId[cellIdx] = id;
                        vars.Add(cellIdx);
                    }
                    varList[i] = id;
                }
                Array.Sort(varList);
                constraints.Add(new Constraint(varList, need));
            }
        }

        if (vars.Count == 0 || constraints.Count == 0) return;

        // Build variable -> constraints adjacency
        var varToCons = new List<int>[vars.Count];
        for (int i = 0; i < varToCons.Length; i++) varToCons[i] = new List<int>(4);
        for (int ci = 0; ci < constraints.Count; ci++)
        {
            var vs = constraints[ci].Vars;
            for (int j = 0; j < vs.Length; j++) varToCons[vs[j]].Add(ci);
        }

        // Component discovery (BFS on bipartite graph vars<->constraints)
        var seenVar = new bool[vars.Count];
        var seenCon = new bool[constraints.Count];
        var compVars = new List<int>(256);
        var compCons = new List<int>(256);
        var queue = new Queue<int>(256);

        // Limits: keep exact enumeration bounded. Larger components are skipped (still safe).
        const int MaxVarsExact = 28;   // 2^28 worst-case ~268M, but pruning usually far less.
        const int MaxConsExact = 128;

        for (int v0 = 0; v0 < vars.Count; v0++)
        {
            if (seenVar[v0]) continue;

            compVars.Clear();
            compCons.Clear();
            queue.Clear();
            queue.Enqueue(v0);
            seenVar[v0] = true;

            while (queue.Count > 0)
            {
                int v = queue.Dequeue();
                compVars.Add(v);
                foreach (int ci in varToCons[v])
                {
                    if (seenCon[ci]) continue;
                    seenCon[ci] = true;
                    compCons.Add(ci);
                    var cvs = constraints[ci].Vars;
                    for (int j = 0; j < cvs.Length; j++)
                    {
                        int vv = cvs[j];
                        if (!seenVar[vv])
                        {
                            seenVar[vv] = true;
                            queue.Enqueue(vv);
                        }
                    }
                }
            }

            // Exact solve if small enough
            if (compVars.Count == 0) continue;
            if (compVars.Count > MaxVarsExact || compCons.Count > MaxConsExact)
                continue;

            SolveComponentExact(vars, constraints, compVars, compCons, state, safeOut);
        }
    }

    private static void SolveComponentExact(
        List<int> varsGlobal,
        List<Constraint> constraintsGlobal,
        List<int> compVarsGlobalIds,
        List<int> compConsGlobalIds,
        byte[] state,
        List<int> safeOut)
    {
        int nVars = compVarsGlobalIds.Count;
        int nCons = compConsGlobalIds.Count;

        // Map global var id -> local [0..nVars)
        var globalToLocal = new Dictionary<int, int>(nVars);
        for (int i = 0; i < nVars; i++) globalToLocal[compVarsGlobalIds[i]] = i;

        // Local constraints arrays
        var consVars = new int[nCons][];
        var consNeed = new int[nCons];
        for (int i = 0; i < nCons; i++)
        {
            var c = constraintsGlobal[compConsGlobalIds[i]];
            int[] local = new int[c.Vars.Length];
            for (int j = 0; j < local.Length; j++) local[j] = globalToLocal[c.Vars[j]];
            Array.Sort(local);
            consVars[i] = local;
            consNeed[i] = c.Need;
        }

        // Variable ordering: most constrained first (degree heuristic)
        int[] degree = new int[nVars];
        for (int i = 0; i < nCons; i++)
        {
            var vs = consVars[i];
            for (int j = 0; j < vs.Length; j++) degree[vs[j]]++;
        }
        int[] order = new int[nVars];
        for (int i = 0; i < nVars; i++) order[i] = i;
        Array.Sort(order, (a, b) => degree[b].CompareTo(degree[a]));

        // For pruning: for each constraint, track remaining unassigned vars and current mines count
        int[] assignedCount = new int[nCons];
        int[] mineCount = new int[nCons];
        int[] totalCount = new int[nCons];
        for (int i = 0; i < nCons; i++) totalCount[i] = consVars[i].Length;

        // For efficient updates: per variable, list of constraints it participates in
        var varCons = new List<int>[nVars];
        for (int i = 0; i < nVars; i++) varCons[i] = new List<int>(4);
        for (int ci = 0; ci < nCons; ci++)
        {
            foreach (int v in consVars[ci]) varCons[v].Add(ci);
        }

        // Track in how many solutions each variable is a mine
        long solutions = 0;
        long[] mineInSolutions = new long[nVars];

        // Assignment array by local var id: -1 unassigned, 0 safe, 1 mine
        sbyte[] assign = new sbyte[nVars];
        Array.Fill(assign, (sbyte)-1);

        void Dfs(int depth)
        {
            if (depth == nVars)
            {
                // All assigned; verify constraints satisfied
                for (int ci = 0; ci < nCons; ci++)
                {
                    if (mineCount[ci] != consNeed[ci]) return;
                }
                solutions++;
                for (int v = 0; v < nVars; v++)
                {
                    if (assign[v] == 1) mineInSolutions[v]++;
                }
                return;
            }

            int vLocal = order[depth];
            if (assign[vLocal] != -1)
            {
                Dfs(depth + 1);
                return;
            }

            // Try safe then mine (either order is fine; safe-first sometimes prunes faster)
            for (sbyte val = 0; val <= 1; val++)
            {
                // Apply assignment
                assign[vLocal] = val;
                bool ok = true;
                foreach (int ci in varCons[vLocal])
                {
                    assignedCount[ci]++;
                    if (val == 1) mineCount[ci]++;

                    int need = consNeed[ci];
                    int minesSoFar = mineCount[ci];
                    int assigned = assignedCount[ci];
                    int remaining = totalCount[ci] - assigned;

                    // Bounds: minesSoFar <= need <= minesSoFar + remaining
                    if (minesSoFar > need || minesSoFar + remaining < need)
                        ok = false;
                }

                if (ok) Dfs(depth + 1);

                // Revert
                foreach (int ci in varCons[vLocal])
                {
                    if (val == 1) mineCount[ci]--;
                    assignedCount[ci]--;
                }
                assign[vLocal] = -1;
            }
        }

        Dfs(0);
        if (solutions == 0) return; // inconsistent component; ignore safely

        // Infer variables fixed across all solutions
        for (int v = 0; v < nVars; v++)
        {
            int globalVarId = compVarsGlobalIds[v];
            int cellIdx = varsGlobal[globalVarId];
            long mineCountAll = mineInSolutions[v];
            if (mineCountAll == solutions)
            {
                // always mine
                if (state[cellIdx] == Unknown) state[cellIdx] = Mine;
            }
            else if (mineCountAll == 0)
            {
                // always safe
                if (state[cellIdx] == Unknown) safeOut.Add(cellIdx);
            }
        }
    }

    private readonly struct Constraint
    {
        public readonly int[] Vars;
        public readonly int Need;
        public Constraint(int[] vars, int need)
        {
            Vars = vars;
            Need = need;
        }
    }

    // Simple growable int queue (ring buffer)
    private sealed class IntQueue
    {
        private int[] _buf;
        private int _head;
        private int _tail;
        public int Count { get; private set; }

        public IntQueue(int capacity)
        {
            _buf = new int[Math.Max(4, capacity)];
        }

        public void Enqueue(int v)
        {
            if (Count == _buf.Length) Grow();
            _buf[_tail] = v;
            _tail++;
            if (_tail == _buf.Length) _tail = 0;
            Count++;
        }

        public int Dequeue()
        {
            int v = _buf[_head];
            _head++;
            if (_head == _buf.Length) _head = 0;
            Count--;
            return v;
        }

        public void Clear()
        {
            _head = 0;
            _tail = 0;
            Count = 0;
        }

        private void Grow()
        {
            int[] nb = new int[_buf.Length * 2];
            if (Count > 0)
            {
                if (_head < _tail)
                {
                    Array.Copy(_buf, _head, nb, 0, Count);
                }
                else
                {
                    int right = _buf.Length - _head;
                    Array.Copy(_buf, _head, nb, 0, right);
                    Array.Copy(_buf, 0, nb, right, _tail);
                }
            }
            _buf = nb;
            _head = 0;
            _tail = Count;
        }
    }
}
