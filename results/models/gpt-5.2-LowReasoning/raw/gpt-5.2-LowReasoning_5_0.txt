use std::collections::{HashMap, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
struct P {
    x: i32,
    y: i32,
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn in_bounds(x: i32, y: i32, w: i32, h: i32) -> bool {
    0 <= x && x < w && 0 <= y && y < h
}

fn neighbors(p: P) -> [P; 4] {
    [
        P { x: p.x + 1, y: p.y },
        P { x: p.x - 1, y: p.y },
        P { x: p.x, y: p.y + 1 },
        P { x: p.x, y: p.y - 1 },
    ]
}

fn is_adj(a: P, b: P) -> bool {
    (a.x - b.x).abs() + (a.y - b.y).abs() == 1
}

// ---------- Fast constructive solvers for empty rectangles ----------

fn rect_path_no_obstacles(w: usize, h: usize) -> Option<Vec<P>> {
    if w == 0 || h == 0 {
        return None;
    }
    let n = w * h;
    if n == 1 {
        return Some(vec![P { x: 0, y: 0 }]);
    }
    // Simple serpentine scan (row-wise). Starts at (0,0), visits all cells.
    // Always a Hamiltonian path for any rectangle.
    let mut path = Vec::with_capacity(n);
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push(P { x: x as i32, y: y as i32 });
            }
        } else {
            for x in (0..w).rev() {
                path.push(P { x: x as i32, y: y as i32 });
            }
        }
    }
    Some(path)
}

fn rect_cycle_no_obstacles(w: usize, h: usize) -> Option<Vec<P>> {
    if w == 0 || h == 0 {
        return None;
    }
    let n = w * h;
    if n < 4 {
        return None;
    }
    // A rectangular grid graph has a Hamiltonian cycle iff w,h > 1 and at least one dimension is even.
    if w <= 1 || h <= 1 || (w % 2 == 1 && h % 2 == 1) {
        return None;
    }
    // Construct a standard cycle:
    // If w is even: snake columns with a wrap on the last row.
    // If w is odd but h is even: do the symmetric construction by transposing.
    if w % 2 == 0 {
        let mut path = Vec::with_capacity(n);
        // Go down first column, then snake through remaining columns, leaving (0,0) as start.
        // A common cycle for even width:
        // - Traverse top row left->right
        // - Traverse interior rows as a "ladder" and return adjacent to start at (0,1)
        // Implementation: build an explicit cycle order then rotate to start at (0,0).
        //
        // Construct cycle:
        // Row 0: (0,0)->(w-1,0)
        for x in 0..w {
            path.push(P { x: x as i32, y: 0 });
        }
        // Rows 1..h-1: snake rightmost to leftmost, but keep first column for last to close.
        for y in 1..h {
            if y % 2 == 1 {
                for x in (1..w).rev() {
                    path.push(P { x: x as i32, y: y as i32 });
                }
            } else {
                for x in 1..w {
                    path.push(P { x: x as i32, y: y as i32 });
                }
            }
        }
        // Finally go up the first column from bottom to 1 (exclude (0,0)).
        for y in (1..h).rev() {
            path.push(P { x: 0, y: y as i32 });
        }
        // This is a Hamiltonian cycle order (successive adjacent) with last adjacent to first.
        // Verify adjacency quickly (debug-safety), but keep in release as cheap.
        for i in 0..path.len() {
            let a = path[i];
            let b = path[(i + 1) % path.len()];
            if !is_adj(a, b) {
                return None;
            }
        }
        Some(path)
    } else {
        // Transpose: build cycle for (h,w) where h is even, then swap coords.
        let cyc_t = rect_cycle_no_obstacles(h, w)?;
        Some(
            cyc_t
                .into_iter()
                .map(|p| P { x: p.y, y: p.x })
                .collect(),
        )
    }
}

// ---------- Shared checks ----------

fn bipartite_counts(w: usize, h: usize, blocked: &Vec<bool>) -> (usize, usize) {
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            if blocked[idx(x, y, w)] {
                continue;
            }
            if (x + y) % 2 == 0 {
                black += 1;
            } else {
                white += 1;
            }
        }
    }
    (black, white)
}

fn degree(p: P, w: usize, h: usize, blocked: &Vec<bool>) -> u8 {
    let mut d = 0u8;
    let w_i = w as i32;
    let h_i = h as i32;
    for q in neighbors(p) {
        if !in_bounds(q.x, q.y, w_i, h_i) {
            continue;
        }
        if blocked[idx(q.x as usize, q.y as usize, w)] {
            continue;
        }
        d += 1;
    }
    d
}

fn connected_all(w: usize, h: usize, blocked: &Vec<bool>) -> bool {
    let n = w * h;
    let mut start = None;
    for i in 0..n {
        if !blocked[i] {
            start = Some(i);
            break;
        }
    }
    let Some(s) = start else { return false };
    let mut vis = vec![false; n];
    let mut q = VecDeque::new();
    vis[s] = true;
    q.push_back(s);
    let w_i = w as i32;
    let h_i = h as i32;
    while let Some(v) = q.pop_front() {
        let x = (v % w) as i32;
        let y = (v / w) as i32;
        for nb in neighbors(P { x, y }) {
            if !in_bounds(nb.x, nb.y, w_i, h_i) {
                continue;
            }
            let ni = idx(nb.x as usize, nb.y as usize, w);
            if blocked[ni] || vis[ni] {
                continue;
            }
            vis[ni] = true;
            q.push_back(ni);
        }
    }
    for i in 0..n {
        if !blocked[i] && !vis[i] {
            return false;
        }
    }
    true
}

// ---------- Exact solver for small instances (<= 64 cells) ----------

fn solve_small_exact(w: usize, h: usize, blocked: &Vec<bool>, require_cycle: bool) -> Option<Vec<P>> {
    let mut free = Vec::new();
    let mut map = HashMap::new();
    for y in 0..h {
        for x in 0..w {
            let id = idx(x, y, w);
            if blocked[id] {
                continue;
            }
            let k = free.len();
            free.push(P { x: x as i32, y: y as i32 });
            map.insert(id, k);
        }
    }
    if free.is_empty() {
        return None;
    }
    let start_id = idx(0, 0, w);
    let start_k = *map.get(&start_id)?;
    let m = free.len();
    if m > 64 {
        return None;
    }
    let mut adj = vec![0u64; m];
    let w_i = w as i32;
    let h_i = h as i32;
    for (i, &p) in free.iter().enumerate() {
        let mut mask = 0u64;
        for q in neighbors(p) {
            if !in_bounds(q.x, q.y, w_i, h_i) {
                continue;
            }
            let id = idx(q.x as usize, q.y as usize, w);
            if blocked[id] {
                continue;
            }
            let j = map[&id];
            mask |= 1u64 << j;
        }
        adj[i] = mask;
    }

    // Quick necessary bipartite check for paths/cycles.
    let (black, white) = bipartite_counts(w, h, blocked);
    if require_cycle {
        if black != white {
            return None;
        }
    } else {
        let diff = if black > white { black - white } else { white - black };
        if diff > 1 {
            return None;
        }
    }

    let full: u64 = if m == 64 { u64::MAX } else { (1u64 << m) - 1 };
    let mut path = Vec::with_capacity(m);
    let mut used: u64 = 0;

    // Connectivity check on remaining graph (cheap BFS over bitmask).
    fn remaining_connected(adj: &Vec<u64>, rem_mask: u64, start: usize) -> bool {
        if rem_mask == 0 {
            return true;
        }
        if (rem_mask & (1u64 << start)) == 0 {
            // find any remaining start
            let s = rem_mask.trailing_zeros() as usize;
            return remaining_connected(adj, rem_mask, s);
        }
        let mut seen = 0u64;
        let mut stack = vec![start];
        seen |= 1u64 << start;
        while let Some(v) = stack.pop() {
            let mut nb = adj[v] & rem_mask & !seen;
            while nb != 0 {
                let lsb = nb & (!nb + 1);
                let u = lsb.trailing_zeros() as usize;
                seen |= lsb;
                stack.push(u);
                nb &= nb - 1;
            }
        }
        (seen & rem_mask) == rem_mask
    }

    // Choose next move using minimum remaining degree (Warnsdorff-style).
    fn next_candidates(adj: &Vec<u64>, used: u64, cur: usize) -> Vec<usize> {
        let mut cands = Vec::new();
        let mut nb = adj[cur] & !used;
        while nb != 0 {
            let lsb = nb & (!nb + 1);
            let u = lsb.trailing_zeros() as usize;
            cands.push(u);
            nb &= nb - 1;
        }
        cands.sort_by_key(|&u| {
            let deg = (adj[u] & !used).count_ones();
            deg
        });
        cands
    }

    fn dfs(
        adj: &Vec<u64>,
        full: u64,
        require_cycle: bool,
        start_k: usize,
        used: &mut u64,
        path: &mut Vec<usize>,
        cur: usize,
    ) -> bool {
        if *used == full {
            if !require_cycle {
                return true;
            }
            // Need last adjacent to start.
            return (adj[cur] & (1u64 << start_k)) != 0;
        }
        // Prune: remaining must be connected.
        let rem = full & !*used;
        if !remaining_connected(adj, rem | (1u64 << cur), cur) {
            return false;
        }
        // Prune: no isolated remaining vertex (degree 0 in remaining subgraph + current frontier).
        // (Rough but helpful.)
        let mut r = rem;
        while r != 0 {
            let lsb = r & (!r + 1);
            let v = lsb.trailing_zeros() as usize;
            let deg = (adj[v] & (rem | (1u64 << cur))).count_ones();
            if deg == 0 {
                return false;
            }
            r &= r - 1;
        }

        let cands = next_candidates(adj, *used, cur);
        for u in cands {
            if (*used & (1u64 << u)) != 0 {
                continue;
            }
            *used |= 1u64 << u;
            path.push(u);
            if dfs(adj, full, require_cycle, start_k, used, path, u) {
                return true;
            }
            path.pop();
            *used &= !(1u64 << u);
        }
        false
    }

    used |= 1u64 << start_k;
    path.push(start_k);
    if dfs(&adj, full, require_cycle, start_k, &mut used, &mut path, start_k) {
        let out = path.into_iter().map(|k| free[k]).collect();
        Some(out)
    } else {
        None
    }
}

// ---------- Heuristic solver for medium instances ----------

fn solve_medium_heuristic(
    w: usize,
    h: usize,
    blocked: &Vec<bool>,
    require_cycle: bool,
    time_budget_ms: u128,
) -> Option<Vec<P>> {
    let n = w * h;
    let start = idx(0, 0, w);
    if start >= n || blocked[start] {
        return None;
    }

    let (black, white) = bipartite_counts(w, h, blocked);
    if require_cycle {
        if black != white {
            return None;
        }
    } else {
        let diff = if black > white { black - white } else { white - black };
        if diff > 1 {
            return None;
        }
    }
    if !connected_all(w, h, blocked) {
        return None;
    }

    // Greedy + limited backtracking using a priority queue keyed by onward degree.
    let free_count = (0..n).filter(|&i| !blocked[i]).count();
    let w_i = w as i32;
    let h_i = h as i32;
    let start_time = std::time::Instant::now();

    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(free_count);
    let mut stack: Vec<(usize, Vec<usize>)> = Vec::new(); // (current idx, remaining candidates)

    let mut cur = start;
    used[cur] = true;
    path.push(cur);

    loop {
        if start_time.elapsed().as_millis() > time_budget_ms {
            return None;
        }
        if path.len() == free_count {
            if !require_cycle {
                break;
            } else {
                let last = path[path.len() - 1];
                let lx = (last % w) as i32;
                let ly = (last / w) as i32;
                if is_adj(P { x: lx, y: ly }, P { x: 0, y: 0 }) {
                    break;
                }
            }
        }

        // Build candidates for cur
        let cx = (cur % w) as i32;
        let cy = (cur / w) as i32;
        let mut cands = Vec::new();
        for nb in neighbors(P { x: cx, y: cy }) {
            if !in_bounds(nb.x, nb.y, w_i, h_i) {
                continue;
            }
            let ni = idx(nb.x as usize, nb.y as usize, w);
            if blocked[ni] || used[ni] {
                continue;
            }
            cands.push(ni);
        }
        if !cands.is_empty() {
            // Sort by onward degree (Warnsdorff).
            cands.sort_by_key(|&ni| {
                let nx = (ni % w) as i32;
                let ny = (ni / w) as i32;
                let mut deg = 0u32;
                for nb in neighbors(P { x: nx, y: ny }) {
                    if !in_bounds(nb.x, nb.y, w_i, h_i) {
                        continue;
                    }
                    let mi = idx(nb.x as usize, nb.y as usize, w);
                    if blocked[mi] || used[mi] {
                        continue;
                    }
                    deg += 1;
                }
                deg
            });
            let next = cands.remove(0);
            stack.push((cur, cands));
            cur = next;
            used[cur] = true;
            path.push(cur);
            continue;
        }

        // Dead end: backtrack
        let mut backtracked = false;
        while let Some((prev, mut rem)) = stack.pop() {
            let last = path.pop().unwrap();
            used[last] = false;
            cur = prev;
            if let Some(next) = rem.first().copied() {
                rem.remove(0);
                stack.push((prev, rem));
                cur = next;
                used[cur] = true;
                path.push(cur);
                backtracked = true;
                break;
            }
        }
        if !backtracked {
            return None;
        }
    }

    let out = path
        .into_iter()
        .map(|i| P { x: (i % w) as i32, y: (i / w) as i32 })
        .collect();
    Some(out)
}

fn validate_path(w: usize, h: usize, blocked: &Vec<bool>, require_cycle: bool, path: &Vec<P>) -> bool {
    let n = w * h;
    let mut used = vec![false; n];
    if path.is_empty() {
        return false;
    }
    if path[0] != (P { x: 0, y: 0 }) {
        return false;
    }
    for (i, &p) in path.iter().enumerate() {
        if !in_bounds(p.x, p.y, w as i32, h as i32) {
            return false;
        }
        let id = idx(p.x as usize, p.y as usize, w);
        if blocked[id] || used[id] {
            return false;
        }
        used[id] = true;
        if i + 1 < path.len() {
            if !is_adj(p, path[i + 1]) {
                return false;
            }
        }
    }
    // Must visit every free cell
    for i in 0..n {
        if !blocked[i] && !used[i] {
            return false;
        }
    }
    if require_cycle {
        let last = *path.last().unwrap();
        if !is_adj(last, P { x: 0, y: 0 }) {
            return false;
        }
    }
    true
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        println!("0");
        return;
    }
    let mut it = input.split_whitespace();
    let w: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("0");
            return;
        }
    };
    let h: usize = it.next().unwrap().parse().unwrap();
    let k: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<u8>().unwrap() != 0;

    let n = w.saturating_mul(h);
    if w == 0 || h == 0 || n == 0 {
        println!("0");
        return;
    }
    let mut blocked = vec![false; n];
    for _ in 0..k {
        if let (Some(xs), Some(ys)) = (it.next(), it.next()) {
            let x: i32 = xs.parse().unwrap();
            let y: i32 = ys.parse().unwrap();
            if in_bounds(x, y, w as i32, h as i32) {
                blocked[idx(x as usize, y as usize, w)] = true;
            }
        }
    }

    // Start must be free
    if blocked[idx(0, 0, w)] {
        println!("0");
        return;
    }

    let free_count = (0..n).filter(|&i| !blocked[i]).count();
    if free_count == 0 {
        println!("0");
        return;
    }

    // Necessary degree checks: any isolated free cell makes it impossible.
    // Also, in a Hamiltonian path on an undirected graph, number of vertices with degree 1
    // must be <= 2 (for a path) or 0 (for a cycle).
    let mut deg1 = 0usize;
    for y in 0..h {
        for x in 0..w {
            if blocked[idx(x, y, w)] {
                continue;
            }
            let d = degree(P { x: x as i32, y: y as i32 }, w, h, &blocked);
            if d == 0 {
                println!("0");
                return;
            }
            if d == 1 {
                deg1 += 1;
            }
        }
    }
    if require_cycle {
        if deg1 != 0 {
            println!("0");
            return;
        }
    } else {
        if deg1 > 2 {
            println!("0");
            return;
        }
    }

    let mut solution: Option<Vec<P>> = None;

    // Strategy 1: No obstacles -> constructive, scales to huge grids.
    if k == 0 {
        solution = if require_cycle {
            rect_cycle_no_obstacles(w, h)
        } else {
            rect_path_no_obstacles(w, h)
        };
    }

    // Strategy 2: Small exact (guaranteed correct).
    if solution.is_none() && free_count <= 64 {
        solution = solve_small_exact(w, h, &blocked, require_cycle);
    }

    // Strategy 3: Medium heuristic (best-effort within budget).
    if solution.is_none() && free_count <= 2500 {
        // Keep a conservative time budget to stay under 5 minutes in worst case.
        solution = solve_medium_heuristic(w, h, &blocked, require_cycle, 60_000);
    }

    if let Some(path) = solution {
        if validate_path(w, h, &blocked, require_cycle, &path) {
            println!("{}", path.len());
            for p in path {
                println!("{} {}", p.x, p.y);
            }
        } else {
            // Safety fallback
            println!("0");
        }
    } else {
        println!("0");
    }
}
