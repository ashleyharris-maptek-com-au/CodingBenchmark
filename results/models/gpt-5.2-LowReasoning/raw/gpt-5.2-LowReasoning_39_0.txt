use std::cmp::Ordering;
use std::io::{self, Read};

fn popcount_u64(x: u64) -> i32 {
    x.count_ones() as i32
}

fn cut_size(mask: u64, adj: &[u64]) -> i32 {
    // Sum edges from A to B by counting neighbors of vertices in A that lie outside A.
    let mut cut = 0i32;
    let outside = !mask;
    let n = adj.len();
    for u in 0..n {
        if ((mask >> u) & 1) != 0 {
            cut += popcount_u64(adj[u] & outside);
        }
    }
    cut
}

fn main() {
    // Read all input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();
    assert!(n % 2 == 0, "n must be even");
    if n == 0 {
        println!("0\n");
        return;
    }
    if n > 64 {
        // The benchmark instances are expected to be small; exact minimum bisection is NP-hard.
        // This solver supports up to 64 vertices via bitmasks.
        panic!("n > 64 is not supported by this exact solver");
    }

    let mut adj = vec![0u64; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u] |= 1u64 << v;
        adj[v] |= 1u64 << u;
    }

    let k = n / 2;
    let all_mask: u64 = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };

    // Symmetry breaking: force vertex 0 into partition A.
    // Then we only need choose k-1 vertices among {1..n-1}.
    let fixed = 1u64;
    let need = k - 1;

    let mut best_cut: i32 = i32::MAX;
    let mut best_mask: u64 = 0;

    // Recursive combination enumeration with pruning.
    fn dfs(
        idx: usize,
        n: usize,
        need: usize,
        chosen: usize,
        mask: u64,
        adj: &[u64],
        best_cut: &mut i32,
        best_mask: &mut u64,
    ) {
        // If impossible to reach need with remaining vertices, prune.
        let remaining = n - idx;
        if chosen > need || chosen + remaining < need {
            return;
        }
        if idx == n {
            if chosen == need {
                let cut = cut_size(mask, adj);
                match cut.cmp(best_cut) {
                    Ordering::Less => {
                        *best_cut = cut;
                        *best_mask = mask;
                    }
                    _ => {}
                }
            }
            return;
        }

        // Simple optimistic lower bound: 0 (can't do better than 0), so no prune here.
        // Keep enumeration deterministic.

        // Option 1: don't include idx in A
        dfs(idx + 1, n, need, chosen, mask, adj, best_cut, best_mask);
        // Option 2: include idx in A
        dfs(
            idx + 1,
            n,
            need,
            chosen + 1,
            mask | (1u64 << idx),
            adj,
            best_cut,
            best_mask,
        );
    }

    dfs(1, n, need, 0, fixed, &adj, &mut best_cut, &mut best_mask);

    // Ensure best_mask has exactly k vertices.
    debug_assert_eq!((best_mask.count_ones() as usize), k);

    // Output
    println!("{}", best_cut);
    let mut first = true;
    for v in 0..n {
        if ((best_mask >> v) & 1) != 0 {
            if !first {
                print!(" ");
            }
            first = false;
            print!("{}", v);
        }
    }
    println!();

    // Sanity: reported cut must match computed (in debug builds).
    debug_assert_eq!(best_cut, cut_size(best_mask, &adj));

    // Avoid unused warnings for all_mask in case of future adjustments.
    let _ = all_mask;
}
