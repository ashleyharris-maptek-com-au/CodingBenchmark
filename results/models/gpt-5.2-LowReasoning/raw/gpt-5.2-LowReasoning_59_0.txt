const fs = require('fs');function parseInput() {  const input = fs.readFileSync(0, 'utf8').trim();  if (!input) return null;  return JSON.parse(input);}function inBounds(x, y, w, h) {  return x >= 0 && x < w && y >= 0 && y < h;}function idx(x, y, w) {  return y * w + x;}// BFS from (sx,sy) over free cells, returns:// - dist Int32Array (length w*h), -1 if unreachable/blocked// - parentDir Int8Array (direction to move from a cell towards source along shortest path),//   encoded as 0: none, 1: from left (move right), 2: from right (move left), 3: from up (move down), 4: from down (move up)function bfsDistances(grid, w, h, sx, sy) {  const n = w * h;  const dist = new Int32Array(n);  dist.fill(-1);  const parentDir = new Int8Array(n); // 0 means unset  const start = idx(sx, sy, w);  dist[start] = 0;  // Simple queue with typed arrays for speed  const qx = new Int32Array(n);  const qy = new Int32Array(n);  let qh = 0;  let qt = 0;  qx[qt] = sx;  qy[qt] = sy;  qt++;  while (qh < qt) {    const x = qx[qh];    const y = qy[qh];    qh++;    const base = dist[idx(x, y, w)] + 1;    // neighbors: left, right, up, down    let nx, ny, ni;    nx = x - 1; ny = y;    if (inBounds(nx, ny, w, h) && grid[ny][nx] !== -1) {      ni = idx(nx, ny, w);      if (dist[ni] === -1) {        dist[ni] = base;        parentDir[ni] = 2; // to go back to source: move right from (nx,ny)        qx[qt] = nx; qy[qt] = ny; qt++;      }    }    nx = x + 1; ny = y;    if (inBounds(nx, ny, w, h) && grid[ny][nx] !== -1) {      ni = idx(nx, ny, w);      if (dist[ni] === -1) {        dist[ni] = base;        parentDir[ni] = 1; // to go back: move left        qx[qt] = nx; qy[qt] = ny; qt++;      }    }    nx = x; ny = y - 1;    if (inBounds(nx, ny, w, h) && grid[ny][nx] !== -1) {      ni = idx(nx, ny, w);      if (dist[ni] === -1) {        dist[ni] = base;        parentDir[ni] = 4; // to go back: move down        qx[qt] = nx; qy[qt] = ny; qt++;      }    }    nx = x; ny = y + 1;    if (inBounds(nx, ny, w, h) && grid[ny][nx] !== -1) {      ni = idx(nx, ny, w);      if (dist[ni] === -1) {        dist[ni] = base;        parentDir[ni] = 3; // to go back: move up        qx[qt] = nx; qy[qt] = ny; qt++;      }    }  }  return { dist, parentDir };}function stepToward(parentDir, w, x, y, tx, ty) {  // Follow parentDir from target cell back towards source, but we only need the first move from (x,y)  // Instead, reconstruct by walking from target back until we reach a neighbor of (x,y).  if (x === tx && y === ty) return { x, y };  let cx = tx, cy = ty;  let cIdx = idx(cx, cy, w);  let safety = 0;  while (safety++ < 100000) {    const dir = parentDir[cIdx];    if (dir === 0) break;    let px = cx, py = cy;    if (dir === 1) px = cx - 1;    else if (dir === 2) px = cx + 1;    else if (dir === 3) py = cy - 1;    else if (dir === 4) py = cy + 1;    if (px === x && py === y) {      return { x: cx, y: cy };    }    cx = px; cy = py;    cIdx = idx(cx, cy, w);  }  return { x, y }; // fallback: stay}function planDronePath(grid, w, h, startX, startY, maxSteps, globallyVisited) {  let x = startX, y = startY;  const path = [];  // local helper to mark visited on entering cell  function visitCell(cx, cy) {    const id = idx(cx, cy, w);    if (!globallyVisited[id] && grid[cy][cx] > 0) {      globallyVisited[id] = 1;    }  }  // include starting cell  path.push([x, y]);  visitCell(x, y);  let target = null; // {x,y}  let cached = null; // {dist,parentDir, sx,sy}  let cooldown = 0;  // Parameters tuned for speed/quality tradeoff  const REPLAN_EVERY = 8; // steps  const BFS_CAP = 60000; // max nodes scanned before early stop (still enough for many grids)  for (let step = 0; step < maxSteps; step++) {    // Replan periodically or if no target    if (!target || cooldown <= 0) {      cached = bfsDistances(grid, w, h, x, y);      const { dist } = cached;      let bestScore = -1;      let best = null;      // Scan cells for best unvisited value/(dist+1)      // Early exit if we scanned enough nodes: only consider reachable cells anyway.      // We iterate linearly over the grid; on huge grids this is the expensive part, hence the cap heuristic.      let scanned = 0;      const n = w * h;      for (let i = 0; i < n; i++) {        const d = dist[i];        if (d <= 0) continue; // skip source and unreachable        if (d > 0x3fffffff) continue;        if (globallyVisited[i]) continue;        const cy = (i / w) | 0;        const cx = i - cy * w;        const v = grid[cy][cx];        if (v <= 0) continue;        const score = v / (d + 1);        if (score > bestScore) {          bestScore = score;          best = { x: cx, y: cy };        }        if (++scanned >= BFS_CAP && bestScore > 0) break;      }      target = best;      cooldown = REPLAN_EVERY;    }    let nx = x, ny = y;    if (target && cached) {      const ti = idx(target.x, target.y, w);      const d = cached.dist[ti];      if (d <= 0) {        // target not reachable anymore        target = null;      } else if (x === target.x && y === target.y) {        target = null;      } else {        const next = stepToward(cached.parentDir, w, x, y, target.x, target.y);        nx = next.x; ny = next.y;      }    }    // If no target, do a small local greedy step towards nearby unvisited value    if (!target) {      let bestGain = -1;      let bestMove = { x, y };      const moves = [        [0, 0],        [1, 0],        [-1, 0],        [0, 1],        [0, -1],      ];      for (const [dx, dy] of moves) {        const tx = x + dx, ty = y + dy;        if (!inBounds(tx, ty, w, h)) continue;        if (grid[ty][tx] === -1) continue;        const id = idx(tx, ty, w);        const gain = (!globallyVisited[id] && grid[ty][tx] > 0) ? grid[ty][tx] : 0;        if (gain > bestGain) {          bestGain = gain;          bestMove = { x: tx, y: ty };        }      }      nx = bestMove.x; ny = bestMove.y;      cooldown = 0; // force replan after local fallback    }    x = nx; y = ny;    path.push([x, y]);    visitCell(x, y);    cooldown--;  }  return path;}function main() {  const data = parseInput();  if (!data) {    process.stdout.write(JSON.stringify({ paths: [] }));    return;  }  const grid = data.probability;  const h = grid.length;  const w = h ? grid[0].length : 0;  const drones = Array.isArray(data.drones) ? data.drones : [];  const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);  const globallyVisited = new Uint8Array(w * h);  const paths = [];  // Plan sequentially; earlier drones claim cells first (unique coverage objective)  for (let i = 0; i < drones.length; i++) {    const d = drones[i];    let sx = d && typeof d.x === 'number' ? d.x | 0 : 0;    let sy = d && typeof d.y === 'number' ? d.y | 0 : 0;    if (!inBounds(sx, sy, w, h) || (h && w && grid[sy][sx] === -1)) {      // If invalid start, find nearest non-obstacle cell by expanding diamond      let found = false;      for (let r = 0; r < Math.max(w, h) && !found; r++) {        for (let dy = -r; dy <= r && !found; dy++) {          const dx = r - Math.abs(dy);          const cand = [            [sx + dx, sy + dy],            [sx - dx, sy + dy],          ];          for (const [cx, cy] of cand) {            if (!inBounds(cx, cy, w, h)) continue;            if (grid[cy][cx] === -1) continue;            sx = cx; sy = cy; found = true; break;          }        }      }      if (!found) {        paths.push([[0, 0]]);        continue;      }    }    const path = planDronePath(grid, w, h, sx, sy, maxSteps, globallyVisited);    paths.push(path);  }  process.stdout.write(JSON.stringify({ paths }));}main();
