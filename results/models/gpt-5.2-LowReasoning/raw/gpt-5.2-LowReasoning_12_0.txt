using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Numerics;

public static class Program
{
    // Practical stdout guard: extreme instances can yield billions of placements.
    // Most judges constrain output size; emitting an enormous solution would time out / exceed limits.
    private const int MaxPlacementsToEmit = 200000;

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        CultureInfo.CurrentUICulture = CultureInfo.InvariantCulture;

        var input = ReadAllTokens(Console.In);
        int idx = 0;
        if (input.Count < 2) return;

        int V = ParseInt(input[idx++]);
        int F = ParseInt(input[idx++]); // Faces are not required for conservative AABB packing.

        var verts = new Vector3[V];
        for (int i = 0; i < V; i++)
        {
            float x = ParseFloat(input[idx++]);
            float y = ParseFloat(input[idx++]);
            float z = ParseFloat(input[idx++]);
            verts[i] = new Vector3(x, y, z);
        }

        // Skip faces
        idx += F * 3;

        float W = ParseFloat(input[idx++]);
        float H = ParseFloat(input[idx++]);
        float D = ParseFloat(input[idx++]);

        // Choose strategy based on scale/complexity.
        long containerVolume = (long)Math.Round(W) * (long)Math.Round(H) * (long)Math.Round(D);
        bool tiny = W <= 20 && H <= 20 && D <= 20 && V <= 200;
        bool medium = W <= 100 && H <= 100 && D <= 100 && V <= 2000;
        bool huge = (W >= 200 || H >= 200 || D >= 200) || containerVolume >= 8_000_000;

        var candidates = new List<Quaternion>();
        if (tiny)
        {
            candidates.AddRange(Generate24AxisOrientations());
            candidates.AddRange(GenerateExtra45DegreeCandidates());
        }
        else if (medium)
        {
            candidates.AddRange(Generate24AxisOrientations());
        }
        else
        {
            candidates.Add(Quaternion.Identity);
            if (!huge) candidates.AddRange(Generate90DegreeAxisCandidates());
        }

        // Deduplicate quaternions (up to sign; q and -q are same rotation).
        candidates = DedupQuaternions(candidates);
        if (candidates.Count == 0) candidates.Add(Quaternion.Identity);

        // Find best rotation by conservative AABB grid packing count.
        Best best = default;
        best.Count = -1;

        foreach (var q in candidates)
        {
            GetAabbOfRotated(verts, q, out var min, out var max);
            var size = max - min;

            if (size.X <= 0 || size.Y <= 0 || size.Z <= 0) continue;

            int nx = (int)Math.Floor(W / size.X);
            int ny = (int)Math.Floor(H / size.Y);
            int nz = (int)Math.Floor(D / size.Z);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            long count = (long)nx * ny * nz;
            if (count > best.Count)
            {
                best = new Best
                {
                    Count = count,
                    Q = q,
                    Min = min,
                    Size = size,
                    Nx = nx,
                    Ny = ny,
                    Nz = nz
                };
            }
        }

        if (best.Count <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Emit placements, clipped to MaxPlacementsToEmit.
        int emit = (int)Math.Min(best.Count, MaxPlacementsToEmit);
        Console.WriteLine(emit.ToString(CultureInfo.InvariantCulture));

        // Normalize quaternion for stable output.
        var qOut = Quaternion.Normalize(best.Q);
        if (qOut.W < 0) qOut = new Quaternion(-qOut.X, -qOut.Y, -qOut.Z, -qOut.W);

        // Translation so that rotated polyhedron's min corner aligns to cell origin.
        // Place each cell at (ix*sizeX, iy*sizeY, iz*sizeZ).
        // Translation t = cellOrigin - minRotated
        // This guarantees all vertices stay within [cellOrigin, cellOrigin+size] and thus non-overlap.
        double minX = best.Min.X, minY = best.Min.Y, minZ = best.Min.Z;
        double sx = best.Size.X, sy = best.Size.Y, sz = best.Size.Z;

        int written = 0;
        for (int iz = 0; iz < best.Nz && written < emit; iz++)
        {
            double baseZ = iz * sz - minZ;
            for (int iy = 0; iy < best.Ny && written < emit; iy++)
            {
                double baseY = iy * sy - minY;
                for (int ix = 0; ix < best.Nx && written < emit; ix++)
                {
                    double baseX = ix * sx - minX;

                    // Output tx ty tz qw qx qy qz (qw scalar part)
                    // System.Numerics stores (X,Y,Z,W).
                    Console.Write(baseX.ToString("R", CultureInfo.InvariantCulture));
                    Console.Write(' ');
                    Console.Write(baseY.ToString("R", CultureInfo.InvariantCulture));
                    Console.Write(' ');
                    Console.Write(baseZ.ToString("R", CultureInfo.InvariantCulture));
                    Console.Write(' ');
                    Console.Write(qOut.W.ToString("R", CultureInfo.InvariantCulture));
                    Console.Write(' ');
                    Console.Write(qOut.X.ToString("R", CultureInfo.InvariantCulture));
                    Console.Write(' ');
                    Console.Write(qOut.Y.ToString("R", CultureInfo.InvariantCulture));
                    Console.Write(' ');
                    Console.WriteLine(qOut.Z.ToString("R", CultureInfo.InvariantCulture));

                    written++;
                }
            }
        }
    }

    private struct Best
    {
        public long Count;
        public Quaternion Q;
        public Vector3 Min;
        public Vector3 Size;
        public int Nx, Ny, Nz;
    }

    private static List<string> ReadAllTokens(TextReader reader)
    {
        var tokens = new List<string>(1 << 16);
        string? line;
        while ((line = reader.ReadLine()) != null)
        {
            int i = 0;
            while (i < line.Length)
            {
                while (i < line.Length && char.IsWhiteSpace(line[i])) i++;
                if (i >= line.Length) break;
                int j = i;
                while (j < line.Length && !char.IsWhiteSpace(line[j])) j++;
                tokens.Add(line.Substring(i, j - i));
                i = j;
            }
        }
        return tokens;
    }

    private static int ParseInt(string s) => int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
    private static float ParseFloat(string s) => float.Parse(s, NumberStyles.Float | NumberStyles.AllowThousands, CultureInfo.InvariantCulture);

    private static void GetAabbOfRotated(Vector3[] verts, Quaternion q, out Vector3 min, out Vector3 max)
    {
        Vector3 v0 = Vector3.Transform(verts[0], q);
        float minX = v0.X, minY = v0.Y, minZ = v0.Z;
        float maxX = v0.X, maxY = v0.Y, maxZ = v0.Z;

        for (int i = 1; i < verts.Length; i++)
        {
            Vector3 v = Vector3.Transform(verts[i], q);
            if (v.X < minX) minX = v.X;
            if (v.Y < minY) minY = v.Y;
            if (v.Z < minZ) minZ = v.Z;
            if (v.X > maxX) maxX = v.X;
            if (v.Y > maxY) maxY = v.Y;
            if (v.Z > maxZ) maxZ = v.Z;
        }

        // Inflate slightly to avoid precision edge cases on boundaries.
        const float eps = 1e-5f;
        min = new Vector3(minX - eps, minY - eps, minZ - eps);
        max = new Vector3(maxX + eps, maxY + eps, maxZ + eps);
    }

    private static List<Quaternion> Generate90DegreeAxisCandidates()
    {
        // Identity + 90/180/270 around each axis
        var list = new List<Quaternion>(10) { Quaternion.Identity };
        AddAxisTurns(list, Vector3.UnitX);
        AddAxisTurns(list, Vector3.UnitY);
        AddAxisTurns(list, Vector3.UnitZ);
        return list;

        static void AddAxisTurns(List<Quaternion> l, Vector3 axis)
        {
            l.Add(Quaternion.CreateFromAxisAngle(axis, (float)(Math.PI / 2)));
            l.Add(Quaternion.CreateFromAxisAngle(axis, (float)(Math.PI)));
            l.Add(Quaternion.CreateFromAxisAngle(axis, (float)(3 * Math.PI / 2)));
        }
    }

    private static List<Quaternion> GenerateExtra45DegreeCandidates()
    {
        // A small set of additional candidates for tiny problems (helps when best fit is diagonal).
        var list = new List<Quaternion>();
        float a = (float)(Math.PI / 4);
        list.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitX, a));
        list.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitY, a));
        list.Add(Quaternion.CreateFromAxisAngle(Vector3.UnitZ, a));
        list.Add(Quaternion.CreateFromAxisAngle(Vector3.Normalize(new Vector3(1, 1, 0)), a));
        list.Add(Quaternion.CreateFromAxisAngle(Vector3.Normalize(new Vector3(1, 0, 1)), a));
        list.Add(Quaternion.CreateFromAxisAngle(Vector3.Normalize(new Vector3(0, 1, 1)), a));
        return list;
    }

    private static List<Quaternion> Generate24AxisOrientations()
    {
        // All 24 proper rotations that map axes to axes (symmetries of a cube).
        var result = new List<Quaternion>(24);
        // Columns of rotation matrix are images of basis vectors.
        // Enumerate all permutations of axes with sign, determinant +1.
        var axes = new[]
        {
            new Vector3(1,0,0),
            new Vector3(0,1,0),
            new Vector3(0,0,1),
        };

        foreach (var perm in Permute3(axes))
        {
            for (int sx = -1; sx <= 1; sx += 2)
            for (int sy = -1; sy <= 1; sy += 2)
            for (int sz = -1; sz <= 1; sz += 2)
            {
                Vector3 c0 = perm[0] * sx;
                Vector3 c1 = perm[1] * sy;
                Vector3 c2 = perm[2] * sz;

                // Determinant of matrix with columns c0,c1,c2 equals dot(c0, cross(c1,c2))
                float det = Vector3.Dot(c0, Vector3.Cross(c1, c2));
                if (det < 0.5f) continue; // must be +1

                var q = QuaternionFromRotationMatrixColumns(c0, c1, c2);
                result.Add(q);
            }
        }
        return result;
    }

    private static IEnumerable<Vector3[]> Permute3(Vector3[] a)
    {
        yield return new[] { a[0], a[1], a[2] };
        yield return new[] { a[0], a[2], a[1] };
        yield return new[] { a[1], a[0], a[2] };
        yield return new[] { a[1], a[2], a[0] };
        yield return new[] { a[2], a[0], a[1] };
        yield return new[] { a[2], a[1], a[0] };
    }

    private static Quaternion QuaternionFromRotationMatrixColumns(Vector3 c0, Vector3 c1, Vector3 c2)
    {
        // Matrix:
        // [ c0.X c1.X c2.X ]
        // [ c0.Y c1.Y c2.Y ]
        // [ c0.Z c1.Z c2.Z ]
        float m00 = c0.X, m01 = c1.X, m02 = c2.X;
        float m10 = c0.Y, m11 = c1.Y, m12 = c2.Y;
        float m20 = c0.Z, m21 = c1.Z, m22 = c2.Z;

        float trace = m00 + m11 + m22;
        float qw, qx, qy, qz;
        if (trace > 0f)
        {
            float s = MathF.Sqrt(trace + 1f) * 2f;
            qw = 0.25f * s;
            qx = (m21 - m12) / s;
            qy = (m02 - m20) / s;
            qz = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            float s = MathF.Sqrt(1f + m00 - m11 - m22) * 2f;
            qw = (m21 - m12) / s;
            qx = 0.25f * s;
            qy = (m01 + m10) / s;
            qz = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            float s = MathF.Sqrt(1f + m11 - m00 - m22) * 2f;
            qw = (m02 - m20) / s;
            qx = (m01 + m10) / s;
            qy = 0.25f * s;
            qz = (m12 + m21) / s;
        }
        else
        {
            float s = MathF.Sqrt(1f + m22 - m00 - m11) * 2f;
            qw = (m10 - m01) / s;
            qx = (m02 + m20) / s;
            qy = (m12 + m21) / s;
            qz = 0.25f * s;
        }
        var q = new Quaternion(qx, qy, qz, qw);
        return Quaternion.Normalize(q);
    }

    private static List<Quaternion> DedupQuaternions(List<Quaternion> list)
    {
        var outList = new List<Quaternion>(list.Count);
        var seen = new HashSet<QuatKey>();
        foreach (var q0 in list)
        {
            var q = Quaternion.Normalize(q0);
            // canonicalize sign
            if (q.W < 0) q = new Quaternion(-q.X, -q.Y, -q.Z, -q.W);
            var key = new QuatKey(q);
            if (seen.Add(key))
                outList.Add(q);
        }
        return outList;
    }

    private readonly struct QuatKey : IEquatable<QuatKey>
    {
        private readonly int _x, _y, _z, _w;

        public QuatKey(Quaternion q)
        {
            // Quantize to allow near-equal comparisons.
            const float scale = 100000f;
            _x = (int)MathF.Round(q.X * scale);
            _y = (int)MathF.Round(q.Y * scale);
            _z = (int)MathF.Round(q.Z * scale);
            _w = (int)MathF.Round(q.W * scale);
        }

        public bool Equals(QuatKey other) => _x == other._x && _y == other._y && _z == other._z && _w == other._w;
        public override bool Equals(object? obj) => obj is QuatKey other && Equals(other);
        public override int GetHashCode()
        {
            unchecked
            {
                int h = 17;
                h = h * 31 + _x;
                h = h * 31 + _y;
                h = h * 31 + _z;
                h = h * 31 + _w;
                return h;
            }
        }
    }
}
