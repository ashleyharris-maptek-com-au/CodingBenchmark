// Adaptive TSP heuristic solver (C++20, MSVC compatible)
// Outputs a permutation of 0..N-1 starting with 0.
//
// Strategy by N:
//  - N <= 12      : Held-Karp DP (exact)
//  - N <= 50000   : 2D Morton-order (rank-based) + light 2-opt on a small window
//  - N <= 5e6     : 2D Morton-order (rank-based)
//  - N  > 5e6     : input order (stream-safe, minimal memory)
//
// Notes:
//  - The prompt's extreme sizes are not realistically solvable as true TSP within time/IO limits.
//    This program always remains correct (valid permutation) and scales to very large N.

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <utility>
#include <vector>

struct FastInput {
    static constexpr size_t BUFSZ = 1 << 20;
    char buf[BUFSZ];
    size_t idx = 0, size = 0;

    inline char readChar() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSZ, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T &out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' '); // skip whitespace
        T sign = 1;
        if (c == '-') {
            sign = -1;
            c = readChar();
        }
        T val = 0;
        while (c > ' ') {
            if (c >= '0' && c <= '9') {
                val = val * 10 + (c - '0');
            } else {
                // unexpected char in int, treat as delimiter
                break;
            }
            c = readChar();
        }
        out = val * sign;
        return true;
    }

    bool readFloat(float &out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (c <= ' ');

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readChar();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + (c - '0');
            c = readChar();
        }

        if (c == '.') {
            c = readChar();
            double place = 0.1;
            while (c >= '0' && c <= '9') {
                val += (c - '0') * place;
                place *= 0.1;
                c = readChar();
            }
        }

        if (c == 'e' || c == 'E') {
            c = readChar();
            bool expNeg = false;
            if (c == '-') {
                expNeg = true;
                c = readChar();
            } else if (c == '+') {
                c = readChar();
            }
            int expVal = 0;
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = readChar();
            }
            double p10 = 1.0;
            double base = 10.0;
            int e = expVal;
            while (e > 0) {
                if (e & 1) p10 *= base;
                base *= base;
                e >>= 1;
            }
            if (expNeg) val /= p10;
            else val *= p10;
        }

        if (neg) val = -val;
        out = static_cast<float>(val);
        return true;
    }
};

static inline uint64_t part1by1_32(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2)) & 0x3333333333333333ULL;
    v = (v | (v << 1)) & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2D_32(uint32_t x, uint32_t y) {
    return (part1by1_32(x) << 1) | part1by1_32(y);
}

static inline float dist2(const std::vector<float> &xs, const std::vector<float> &ys, int a, int b) {
    float dx = xs[a] - xs[b];
    float dy = ys[a] - ys[b];
    return dx * dx + dy * dy;
}

// Exact DP for N<=12 (Held-Karp): returns best Hamiltonian path starting at 0 (no return edge)
static std::vector<int> solve_exact_small(const std::vector<float> &xs, const std::vector<float> &ys) {
    const int n = static_cast<int>(xs.size());
    const int FULL = 1 << n;
    const float INF = std::numeric_limits<float>::infinity();

    std::vector<float> dp(FULL * n, INF);
    std::vector<int16_t> parent(FULL * n, -1);

    dp[(1 << 0) * n + 0] = 0.0f;

    for (int mask = 1; mask < FULL; ++mask) {
        if (!(mask & 1)) continue; // must include start
        for (int last = 0; last < n; ++last) {
            if (!(mask & (1 << last))) continue;
            float cur = dp[mask * n + last];
            if (!(cur < INF)) continue;
            for (int nxt = 0; nxt < n; ++nxt) {
                if (mask & (1 << nxt)) continue;
                int nmask = mask | (1 << nxt);
                float cand = cur + std::sqrt(dist2(xs, ys, last, nxt));
                float &ref = dp[nmask * n + nxt];
                if (cand < ref) {
                    ref = cand;
                    parent[nmask * n + nxt] = static_cast<int16_t>(last);
                }
            }
        }
    }

    int fullMask = FULL - 1;
    int bestLast = 0;
    float best = INF;
    for (int last = 0; last < n; ++last) {
        float v = dp[fullMask * n + last];
        if (v < best) {
            best = v;
            bestLast = last;
        }
    }

    std::vector<int> tour(n);
    int mask = fullMask;
    int last = bestLast;
    for (int pos = n - 1; pos >= 0; --pos) {
        tour[pos] = last;
        int16_t p = parent[mask * n + last];
        mask ^= (1 << last);
        last = (p < 0 ? 0 : p);
        if (pos == 0) tour[pos] = 0;
    }

    // Ensure starts at 0 (it should)
    if (!tour.empty() && tour[0] != 0) {
        auto it = std::find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) std::rotate(tour.begin(), it, tour.end());
    }
    return tour;
}

// Rank-based Morton ordering (avoids needing coordinate normalization)
static std::vector<int> solve_morton_rank(const std::vector<float> &xs, const std::vector<float> &ys) {
    const int n = static_cast<int>(xs.size());
    std::vector<int> idx(n);
    std::iota(idx.begin(), idx.end(), 0);

    std::vector<int> xOrder = idx, yOrder = idx;
    std::sort(xOrder.begin(), xOrder.end(), [&](int a, int b) {
        if (xs[a] != xs[b]) return xs[a] < xs[b];
        return ys[a] < ys[b];
    });
    std::sort(yOrder.begin(), yOrder.end(), [&](int a, int b) {
        if (ys[a] != ys[b]) return ys[a] < ys[b];
        return xs[a] < xs[b];
    });

    std::vector<uint32_t> xRank(n), yRank(n);
    for (uint32_t r = 0; r < static_cast<uint32_t>(n); ++r) xRank[xOrder[r]] = r;
    for (uint32_t r = 0; r < static_cast<uint32_t>(n); ++r) yRank[yOrder[r]] = r;

    std::vector<uint64_t> key(n);
    for (int i = 0; i < n; ++i) {
        // Fold ranks down to 32 bits to keep interleave fast.
        // For n > 2^32 (not feasible to store anyway), this would alias.
        key[i] = morton2D_32(xRank[i], yRank[i]);
    }

    std::sort(idx.begin(), idx.end(), [&](int a, int b) {
        if (key[a] != key[b]) return key[a] < key[b];
        // stable-ish tie-break
        if (xs[a] != xs[b]) return xs[a] < xs[b];
        return ys[a] < ys[b];
    });

    // Rotate to start at 0
    auto it = std::find(idx.begin(), idx.end(), 0);
    if (it != idx.end()) std::rotate(idx.begin(), it, idx.end());
    return idx;
}

// Light 2-opt using a small lookahead window to avoid O(N^2).
static void two_opt_window(std::vector<int> &tour, const std::vector<float> &xs, const std::vector<float> &ys, int window, int passes) {
    const int n = static_cast<int>(tour.size());
    if (n < 4) return;
    if (window < 4) window = 4;
    if (passes < 1) passes = 1;

    for (int pass = 0; pass < passes; ++pass) {
        bool improved = false;
        for (int i = 0; i + 3 < n; ++i) {
            int a = tour[i];
            int b = tour[i + 1];
            float ab = dist2(xs, ys, a, b);
            int jMax = std::min(n - 2, i + window);
            for (int j = i + 2; j <= jMax; ++j) {
                int c = tour[j];
                int d = tour[j + 1];
                float cd = dist2(xs, ys, c, d);
                float ac = dist2(xs, ys, a, c);
                float bd = dist2(xs, ys, b, d);
                // If swapping reduces squared length, do it.
                if (ac + bd < ab + cd) {
                    std::reverse(tour.begin() + (i + 1), tour.begin() + (j + 1));
                    improved = true;
                    // update for next comparisons
                    b = tour[i + 1];
                    ab = dist2(xs, ys, a, b);
                }
            }
        }
        if (!improved) break;
    }
    // keep 0 at start
    if (!tour.empty() && tour[0] != 0) {
        auto it = std::find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) std::rotate(tour.begin(), it, tour.end());
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastInput in;
    int64_t N64 = 0;
    if (!in.readInt(N64)) return 0;
    if (N64 <= 0) return 0;

    // If N is astronomically large, we cannot store coordinates.
    // For N > 5e6, output input order (0..N-1) without reading all coordinates into memory.
    // Still need to consume input to avoid broken pipes in some judges? Usually not required; we can exit after output.
    // But to be safe, we read and discard coordinates.
    if (N64 > 5000000) {
        // Discard coordinates as they come in (best-effort).
        float x, y;
        for (int64_t i = 0; i < N64; ++i) {
            if (!in.readFloat(x)) break;
            if (!in.readFloat(y)) break;
        }
        // Output trivial tour: 0 1 2 ... N-1
        // Use buffered printing to handle huge output.
        std::string out;
        out.reserve(1 << 20);
        auto flush = [&]() {
            std::fwrite(out.data(), 1, out.size(), stdout);
            out.clear();
        };
        for (int64_t i = 0; i < N64; ++i) {
            char buf[32];
            int len = 0;
            if (i) out.push_back(' ');
            // itoa-like
            uint64_t v = static_cast<uint64_t>(i);
            char tmp[24];
            int t = 0;
            do {
                tmp[t++] = char('0' + (v % 10));
                v /= 10;
            } while (v);
            for (int k = t - 1; k >= 0; --k) buf[len++] = tmp[k];
            out.append(buf, buf + len);
            if (out.size() > (1 << 19)) flush();
        }
        out.push_back('\n');
        flush();
        return 0;
    }

    const int n = static_cast<int>(N64);
    std::vector<float> xs(n), ys(n);
    for (int i = 0; i < n; ++i) {
        float x = 0.0f, y = 0.0f;
        in.readFloat(x);
        in.readFloat(y);
        xs[i] = x;
        ys[i] = y;
    }

    std::vector<int> tour;
    if (n <= 12) {
        tour = solve_exact_small(xs, ys);
    } else {
        tour = solve_morton_rank(xs, ys);
        if (n <= 50000) {
            // Very light local improvement; keep bounded.
            two_opt_window(tour, xs, ys, /*window=*/64, /*passes=*/2);
        }
    }

    // Output tour as indices, one line.
    std::string out;
    out.reserve(std::min<size_t>(static_cast<size_t>(n) * 3, 1u << 20));
    auto flush = [&]() {
        std::fwrite(out.data(), 1, out.size(), stdout);
        out.clear();
    };
    for (int i = 0; i < n; ++i) {
        if (i) out.push_back(' ');
        int v = tour[i];
        char tmp[16];
        int t = 0;
        unsigned int uv = static_cast<unsigned int>(v);
        do {
            tmp[t++] = char('0' + (uv % 10));
            uv /= 10;
        } while (uv);
        for (int k = t - 1; k >= 0; --k) out.push_back(tmp[k]);
        if (out.size() > (1 << 19)) flush();
    }
    out.push_back('\n');
    flush();
    return 0;
}
