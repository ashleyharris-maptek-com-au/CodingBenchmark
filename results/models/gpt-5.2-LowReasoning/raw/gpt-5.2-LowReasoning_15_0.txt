use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }
    fn sub(self, o: Self) -> Self {
        Self::new(self.x - o.x, self.y - o.y)
    }
    fn dot(self, o: Self) -> f64 {
        self.x * o.x + self.y * o.y
    }
    fn cross(self, o: Self) -> f64 {
        self.x * o.y - self.y * o.x
    }
}

#[derive(Clone, Copy, Debug, Default)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }
    fn mul(self, s: f64) -> Self {
        Self::new(self.x * s, self.y * s, self.z * s)
    }
    fn dot(self, o: Self) -> f64 {
        self.x * o.x + self.y * o.y + self.z * o.z
    }
    fn len(self) -> f64 {
        self.dot(self).sqrt()
    }
    fn norm(self) -> Self {
        let l = self.len();
        if l == 0.0 {
            self
        } else {
            self.mul(1.0 / l)
        }
    }
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        a += poly[i].cross(poly[j]);
    }
    0.5 * a
}

fn point_in_poly(p: Vec2, poly: &[Vec2]) -> bool {
    // Even-odd rule; treats boundary as inside.
    let mut inside = false;
    let n = poly.len();
    if n < 3 {
        return false;
    }
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        // Boundary check (colinear + within segment)
        let ab = b.sub(a);
        let ap = p.sub(a);
        let cross = ab.cross(ap);
        if cross.abs() <= 1e-12 {
            let dot = ap.dot(ab);
            if dot >= -1e-12 && dot <= ab.dot(ab) + 1e-12 {
                return true;
            }
        }
        let ay = a.y;
        let by = b.y;
        let intersects = (ay > p.y) != (by > p.y);
        if intersects {
            let t = (p.y - ay) / (by - ay);
            let x_int = a.x + t * (b.x - a.x);
            if x_int >= p.x {
                inside = !inside;
            }
        }
    }
    inside
}

fn project_to_z0(p: Vec3, sun: Vec3) -> Vec2 {
    // Line: p + t*sun, solve z=0 => t = -p.z/sun.z
    let mut sz = sun.z;
    if sz.abs() < 1e-12 {
        sz = if sz >= 0.0 { 1e-12 } else { -1e-12 };
    }
    let t = -p.z / sz;
    Vec2::new(p.x + t * sun.x, p.y + t * sun.y)
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    // Monotone chain; returns CCW hull without duplicate last point.
    pts.sort_by(|a, b| {
        let ox = a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal);
        if ox != Ordering::Equal {
            return ox;
        }
        a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal)
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1e-12 && (a.y - b.y).abs() < 1e-12);
    if pts.len() <= 1 {
        return pts;
    }
    let mut lower: Vec<Vec2> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let l = lower.len();
            let a = lower[l - 2];
            let b = lower[l - 1];
            if b.sub(a).cross(p.sub(b)) <= 1e-12 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }
    let mut upper: Vec<Vec2> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let l = upper.len();
            let a = upper[l - 2];
            let b = upper[l - 1];
            if b.sub(a).cross(p.sub(b)) <= 1e-12 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn bbox(poly: &[Vec2]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for &p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn hull_centroid(poly: &[Vec2]) -> Vec2 {
    if poly.is_empty() {
        return Vec2::new(0.0, 0.0);
    }
    let mut cx = 0.0;
    let mut cy = 0.0;
    for p in poly {
        cx += p.x;
        cy += p.y;
    }
    let inv = 1.0 / (poly.len() as f64);
    Vec2::new(cx * inv, cy * inv)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let nv: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            return;
        }
    };
    let mut poly: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Vec2::new(x, y));
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    let sun = Vec3::new(sx, sy, sz).norm();

    // Ensure polygon orientation doesn't matter for point-in-poly
    let area = polygon_area(&poly).abs();
    let (minx, miny, maxx, maxy) = bbox(&poly);
    let bbox_area = (maxx - minx).abs() * (maxy - miny).abs();

    // Standard tetrahedron vertices (local space).
    let base = [
        Vec3::new(1.0, 0.0, -0.707),
        Vec3::new(-1.0, 0.0, -0.707),
        Vec3::new(0.0, 1.0, 0.707),
        Vec3::new(0.0, -1.0, 0.707),
    ];

    // Fixed rotation = identity quaternion.
    let qw = 1.0;
    let qx = 0.0;
    let qy = 0.0;
    let qz = 0.0;

    // Use max scale for fewer tetrahedrons.
    let scale = 10.0_f64;

    // Compute the shadow footprint for the base tetrahedron at origin.
    let mut proj: Vec<Vec2> = Vec::with_capacity(4);
    for &v in &base {
        let p = v.mul(scale);
        proj.push(project_to_z0(p, sun));
    }
    let hull = convex_hull(proj);
    let c0 = hull_centroid(&hull);
    let (hx0, hy0, hx1, hy1) = bbox(&hull);
    let hull_w = (hx1 - hx0).abs().max(1e-9);
    let hull_h = (hy1 - hy0).abs().max(1e-9);
    let min_dim = hull_w.min(hull_h);

    // Adaptive tiling step: dense for small/complex, capped for huge.
    let mut step = 0.25 * min_dim;
    if step.is_nan() || step <= 0.0 {
        step = 1.0;
    }

    // Cap output size to keep runtime/IO manageable for extreme inputs.
    let max_out: usize = if area < 1.0 { 2000 } else { 20000 };
    if bbox_area.is_finite() && bbox_area > 0.0 {
        let est = (bbox_area / (step * step)).ceil() as usize;
        if est > max_out {
            step = (bbox_area / (max_out as f64)).sqrt();
        }
    }

    // Small polygons: oversample to reduce risk of gaps.
    if area > 0.0 && area < 100.0 {
        step *= 0.5;
    }

    // Place tetrahedrons on a 2D grid; separate them in 3D along sun direction
    // so they never intersect, while keeping the shadows fixed.
    let sep = 500.0 * scale;
    let mut out: Vec<Vec3> = Vec::new();

    let mut y = miny - step;
    while y <= maxy + step {
        let mut x = minx - step;
        while x <= maxx + step {
            let p = Vec2::new(x, y);
            if point_in_poly(p, &poly) {
                // Compute translation to align tetra shadow centroid to p.
                // Shadow shift from dz: (-dz/sun.z) * (sun.x, sun.y).
                let dz = (out.len() as f64) * sep;
                let mut sz2 = sun.z;
                if sz2.abs() < 1e-12 {
                    sz2 = if sz2 >= 0.0 { 1e-12 } else { -1e-12 };
                }
                let shift = Vec2::new(-(dz / sz2) * sun.x, -(dz / sz2) * sun.y);
                let t_xy = p.sub(c0).sub(shift);
                let center = Vec3::new(t_xy.x, t_xy.y, dz);
                out.push(center);
                if out.len() >= max_out {
                    break;
                }
            }
            x += step;
        }
        if out.len() >= max_out {
            break;
        }
        y += step;
    }

    // Fallback: if polygon is degenerate or missed entirely, place one tetra at bbox center.
    if out.is_empty() {
        let cx = 0.5 * (minx + maxx);
        let cy = 0.5 * (miny + maxy);
        let dz = 0.0;
        let center = Vec3::new(cx - c0.x, cy - c0.y, dz);
        out.push(center);
    }

    // Output
    println!("{}", out.len());
    for center in out {
        // px py pz qw qx qy qz scale
        println!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            center.x, center.y, center.z, qw, qx, qy, qz, scale
        );
    }
}
