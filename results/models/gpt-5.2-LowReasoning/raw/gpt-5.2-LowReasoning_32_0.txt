// Steiner Tree heuristic solver (metric-closure MST + expansion + pruning MST)
// - Runs Dijkstra from each terminal
// - Builds MST over terminals using shortest-path distances
// - Expands MST edges back to original graph paths
// - Prunes to an acyclic connected subgraph via Kruskal MST on expanded edges

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <tuple>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::cin;
using std::cout;
using std::int64_t;

struct Edge {
  int to;
  int w;
};

static constexpr int64_t INF = (std::numeric_limits<int64_t>::max() / 4);

struct DSU {
  std::vector<int> p, r;
  explicit DSU(int n) : p(n), r(n, 0) {
    for (int i = 0; i < n; ++i) p[i] = i;
  }
  int find(int a) {
    while (p[a] != a) {
      p[a] = p[p[a]];
      a = p[a];
    }
    return a;
  }
  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (r[a] < r[b]) std::swap(a, b);
    p[b] = a;
    if (r[a] == r[b]) r[a]++;
    return true;
  }
};

static uint64_t pack_undirected(int u, int v) {
  if (u > v) std::swap(u, v);
  return (static_cast<uint64_t>(static_cast<uint32_t>(u)) << 32) |
         static_cast<uint32_t>(v);
}

struct DijkstraResult {
  std::vector<int64_t> dist;
  std::vector<int> parent;  // parent vertex on shortest path tree, -1 if none
};

static DijkstraResult dijkstra(int n, const std::vector<std::vector<Edge>>& g,
                               int src) {
  DijkstraResult res;
  res.dist.assign(n, INF);
  res.parent.assign(n, -1);
  res.dist[src] = 0;
  using Item = std::pair<int64_t, int>;
  std::priority_queue<Item, std::vector<Item>, std::greater<Item>> pq;
  pq.push({0, src});
  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != res.dist[v]) continue;
    for (const auto& e : g[v]) {
      int to = e.to;
      int64_t nd = d + static_cast<int64_t>(e.w);
      if (nd < res.dist[to]) {
        res.dist[to] = nd;
        res.parent[to] = v;
        pq.push({nd, to});
      }
    }
  }
  return res;
}

static void add_path_edges(int src, int dst, const std::vector<int>& parent,
                           std::unordered_map<uint64_t, int>& best_w,
                           const std::unordered_map<uint64_t, int>& input_best_w) {
  int cur = dst;
  while (cur != src && cur != -1) {
    int p = parent[cur];
    if (p == -1) break;
    uint64_t key = pack_undirected(cur, p);
    auto itw = input_best_w.find(key);
    if (itw != input_best_w.end()) {
      auto it = best_w.find(key);
      if (it == best_w.end() || itw->second < it->second) best_w[key] = itw->second;
    }
    cur = p;
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n = 0, m = 0, k = 0;
  if (!(cin >> n >> m >> k)) return 0;

  std::vector<std::vector<Edge>> g(n);
  std::unordered_map<uint64_t, int> input_best_w;
  input_best_w.reserve(static_cast<size_t>(m) * 2);

  for (int i = 0; i < m; ++i) {
    int u, v, w;
    cin >> u >> v >> w;
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    g[u].push_back({v, w});
    g[v].push_back({u, w});
    uint64_t key = pack_undirected(u, v);
    auto it = input_best_w.find(key);
    if (it == input_best_w.end() || w < it->second) input_best_w[key] = w;
  }

  std::vector<int> terminals(k);
  for (int i = 0; i < k; ++i) cin >> terminals[i];

  if (k <= 1) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  // Dijkstra from each terminal
  std::vector<DijkstraResult> dj;
  dj.reserve(k);
  for (int i = 0; i < k; ++i) {
    int t = terminals[i];
    if (t < 0 || t >= n) {
      cout << 0 << " " << 0 << "\n";
      return 0;
    }
    dj.push_back(dijkstra(n, g, t));
  }

  // Prim MST over terminal metric closure
  std::vector<int64_t> min_cost(k, INF);
  std::vector<int> parent_t(k, -1);
  std::vector<char> used(k, 0);
  min_cost[0] = 0;

  for (int it = 0; it < k; ++it) {
    int v = -1;
    for (int i = 0; i < k; ++i) {
      if (!used[i] && (v == -1 || min_cost[i] < min_cost[v])) v = i;
    }
    if (v == -1 || min_cost[v] >= INF / 2) {
      // Disconnected terminals in metric closure => infeasible
      cout << 0 << " " << 0 << "\n";
      return 0;
    }
    used[v] = 1;
    for (int u = 0; u < k; ++u) {
      if (used[u]) continue;
      int tu = terminals[u];
      int64_t d = dj[v].dist[tu];
      if (d < min_cost[u]) {
        min_cost[u] = d;
        parent_t[u] = v;
      }
    }
  }

  // Expand MST terminal edges into original edges (via shortest-path trees)
  std::unordered_map<uint64_t, int> expanded_best_w;
  expanded_best_w.reserve(static_cast<size_t>(k) * 32);

  for (int u = 1; u < k; ++u) {
    int v = parent_t[u];
    int src_idx = v;
    int dst_term = terminals[u];
    int src_term = terminals[src_idx];
    add_path_edges(src_term, dst_term, dj[src_idx].parent, expanded_best_w,
                   input_best_w);
  }

  // If expansion somehow produced nothing, output empty
  if (expanded_best_w.empty()) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  // Build edge list for Kruskal
  struct WEdge {
    int u, v;
    int w;
  };
  std::vector<WEdge> edges;
  edges.reserve(expanded_best_w.size());
  for (const auto& kv : expanded_best_w) {
    uint32_t u = static_cast<uint32_t>(kv.first >> 32);
    uint32_t v = static_cast<uint32_t>(kv.first & 0xFFFFFFFFu);
    edges.push_back({static_cast<int>(u), static_cast<int>(v), kv.second});
  }
  std::sort(edges.begin(), edges.end(),
            [](const WEdge& a, const WEdge& b) { return a.w < b.w; });

  DSU dsu(n);
  std::vector<WEdge> tree;
  tree.reserve(edges.size());
  int64_t total = 0;
  for (const auto& e : edges) {
    if (dsu.unite(e.u, e.v)) {
      tree.push_back(e);
      total += e.w;
    }
  }

  // Verify all terminals connected in resulting forest; if not, keep original expansion (as fallback)
  int root = dsu.find(terminals[0]);
  bool ok = true;
  for (int i = 1; i < k; ++i) {
    if (dsu.find(terminals[i]) != root) {
      ok = false;
      break;
    }
  }

  const auto& out_edges = ok ? tree : edges;
  int64_t out_total = 0;
  for (const auto& e : out_edges) out_total += e.w;

  cout << out_total << " " << static_cast<int>(out_edges.size()) << "\n";
  for (const auto& e : out_edges) {
    cout << e.u << " " << e.v << "\n";
  }
  return 0;
}
