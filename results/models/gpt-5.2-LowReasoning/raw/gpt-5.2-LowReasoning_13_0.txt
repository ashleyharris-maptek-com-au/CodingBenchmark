#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <vector>

using namespace std;

static string lcs_two_strings_dp(const string& a, const string& b) {
    // O(|a|*|b|) time, O(min(|a|,|b|)) memory
    const string* s1 = &a;
    const string* s2 = &b;
    if (s1->size() < s2->size()) {
        // keep s2 the shorter for memory? Actually dp uses s2 length.
    } else {
        swap(s1, s2);
    }
    const string& x = *s1;
    const string& y = *s2;

    vector<int> prev(y.size() + 1, 0), cur(y.size() + 1, 0);
    int bestLen = 0;
    size_t bestEnd = 0; // end position in x (1-based style for convenience)

    for (size_t i = 1; i <= x.size(); ++i) {
        cur[0] = 0;
        const unsigned char cx = static_cast<unsigned char>(x[i - 1]);
        for (size_t j = 1; j <= y.size(); ++j) {
            if (cx == static_cast<unsigned char>(y[j - 1])) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > bestLen) {
                    bestLen = cur[j];
                    bestEnd = i;
                }
            } else {
                cur[j] = 0;
            }
        }
        prev.swap(cur);
    }

    if (bestLen == 0) return string();
    return x.substr(bestEnd - bestLen, static_cast<size_t>(bestLen));
}

struct SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        int firstpos = -1; // end position in the base string
        int head = -1;     // head transition index
    };
    struct Edge {
        int to;
        int next;
        unsigned char ch;
    };

    vector<State> st;
    vector<Edge> edges;
    int last = 0;

    SuffixAutomaton() {
        st.reserve(1024);
        edges.reserve(1024);
        st.push_back(State{}); // root
        st[0].link = -1;
        st[0].len = 0;
        st[0].firstpos = -1;
        st[0].head = -1;
        last = 0;
    }

    int get_next(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    void set_next(int v, unsigned char c, int to) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
        edges.push_back(Edge{to, st[v].head, c});
        st[v].head = static_cast<int>(edges.size()) - 1;
    }

    int clone_state(int q, int newLen) {
        State clone = st[q];
        clone.len = newLen;
        clone.firstpos = st[q].firstpos;
        clone.head = -1;
        int cloneIdx = static_cast<int>(st.size());
        st.push_back(clone);

        // copy transitions adjacency list
        for (int e = st[q].head; e != -1; e = edges[e].next) {
            edges.push_back(Edge{edges[e].to, st[cloneIdx].head, edges[e].ch});
            st[cloneIdx].head = static_cast<int>(edges.size()) - 1;
        }
        return cloneIdx;
    }

    void extend(unsigned char c, int pos) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].firstpos = pos;
        st[cur].head = -1;

        int p = last;
        while (p != -1 && get_next(p, c) == -1) {
            set_next(p, c, cur);
            p = st[p].link;
        }
        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = get_next(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = clone_state(q, st[p].len + 1);
                while (p != -1 && get_next(p, c) == q) {
                    set_next(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = clone;
                st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static string lcs_multiple_strings_sam(const vector<string>& strings, int baseIdx) {
    const string& base = strings[baseIdx];
    if (base.empty()) return string();

    SuffixAutomaton sam;
    sam.st.reserve(base.size() * 2 + 1);
    sam.edges.reserve(base.size() * 3 + 4);

    for (size_t i = 0; i < base.size(); ++i) {
        sam.extend(static_cast<unsigned char>(base[i]), static_cast<int>(i));
    }

    const int sz = static_cast<int>(sam.st.size());

    vector<int> order(sz);
    for (int i = 0; i < sz; ++i) order[i] = i;
    sort(order.begin(), order.end(), [&](int a, int b) { return sam.st[a].len < sam.st[b].len; });

    vector<int> globalMin(sz);
    for (int v = 0; v < sz; ++v) globalMin[v] = sam.st[v].len;

    vector<int> best(sz, 0);

    for (int idx = 0; idx < static_cast<int>(strings.size()); ++idx) {
        if (idx == baseIdx) continue;
        fill(best.begin(), best.end(), 0);

        int v = 0;
        int l = 0;
        const string& s = strings[idx];
        for (size_t i = 0; i < s.size(); ++i) {
            unsigned char c = static_cast<unsigned char>(s[i]);
            int to = sam.get_next(v, c);
            while (v != 0 && to == -1) {
                v = sam.st[v].link;
                l = min(l, sam.st[v].len);
                to = sam.get_next(v, c);
            }
            if (to != -1) {
                v = to;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (l > best[v]) best[v] = l;
        }

        // propagate best values to suffix links in decreasing length order
        for (int oi = sz - 1; oi >= 1; --oi) {
            int state = order[oi];
            int parent = sam.st[state].link;
            if (parent >= 0) {
                int propagated = min(best[state], sam.st[parent].len);
                if (propagated > best[parent]) best[parent] = propagated;
            }
        }

        for (int vtx = 0; vtx < sz; ++vtx) {
            globalMin[vtx] = min(globalMin[vtx], best[vtx]);
        }
    }

    int bestState = 0;
    int bestLen = 0;
    for (int v = 0; v < sz; ++v) {
        if (globalMin[v] > bestLen) {
            bestLen = globalMin[v];
            bestState = v;
        }
    }
    if (bestLen == 0) return string();
    int endPos = sam.st[bestState].firstpos;
    return base.substr(static_cast<size_t>(endPos - bestLen + 1), static_cast<size_t>(bestLen));
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n = 0;
    if (!(cin >> n)) return 0;
    string dummy;
    getline(cin, dummy); // consume endline after n

    vector<string> s;
    s.reserve(max(0, n));
    for (int i = 0; i < n; ++i) {
        string line;
        getline(cin, line);
        s.push_back(std::move(line));
    }

    if (n <= 0) {
        cout << "\n";
        return 0;
    }
    if (n == 1) {
        cout << s[0] << "\n";
        return 0;
    }

    int baseIdx = 0;
    size_t minLen = s[0].size();
    uint64_t totalLen = s[0].size();
    for (int i = 1; i < n; ++i) {
        totalLen += s[i].size();
        if (s[i].size() < minLen) {
            minLen = s[i].size();
            baseIdx = i;
        }
    }

    // Adaptive strategy:
    // - If N==2 and sizes are small enough, use classic DP (fast, simple).
    // - Otherwise, use suffix automaton on the shortest string and stream-match all others.
    const uint64_t DP_LIMIT = 25'000'000ULL; // ~25M cell updates threshold
    if (n == 2) {
        uint64_t cells = static_cast<uint64_t>(s[0].size()) * static_cast<uint64_t>(s[1].size());
        if (cells <= DP_LIMIT) {
            cout << lcs_two_strings_dp(s[0], s[1]) << "\n";
            return 0;
        }
    }

    cout << lcs_multiple_strings_sam(s, baseIdx) << "\n";
    return 0;
}
