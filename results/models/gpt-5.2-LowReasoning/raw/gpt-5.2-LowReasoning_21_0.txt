// N-Dimensional Snake solver (single-process, offline output).
// Heuristic planner: greedy L1-to-nearest-food with collision avoidance + local mobility.
// C++20 / MSVC-friendly (no nonstandard headers).

#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <optional>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Coord = int32_t;

struct VecHash {
  size_t operator()(const std::vector<Coord>& v) const noexcept {
    // FNV-1a-ish mix; stable across runs.
    uint64_t h = 1469598103934665603ull;
    for (Coord x : v) {
      uint64_t y = static_cast<uint32_t>(x);
      h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
      h *= 1099511628211ull;
    }
    return static_cast<size_t>(h);
  }
};

static inline int64_t l1_dist(const std::vector<Coord>& a, const std::vector<Coord>& b) {
  int64_t s = 0;
  for (size_t i = 0; i < a.size(); ++i) s += std::llabs(static_cast<int64_t>(a[i]) - b[i]);
  return s;
}

static inline bool in_bounds(const std::vector<Coord>& p, const std::vector<Coord>& bounds) {
  for (size_t i = 0; i < p.size(); ++i) {
    if (p[i] < 0 || p[i] >= bounds[i]) return false;
  }
  return true;
}

struct Move {
  int axis = 0;
  int dir = 1;
};

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int D = 0;
  int max_turns = 0;
  if (!(std::cin >> D >> max_turns)) return 0;
  if (D <= 0 || D > 64 || max_turns < 0) return 0;

  std::vector<Coord> bounds(D);
  for (int i = 0; i < D; ++i) std::cin >> bounds[i];

  int snake_len = 0;
  std::vector<Coord> head(D);
  std::cin >> snake_len;
  for (int i = 0; i < D; ++i) std::cin >> head[i];

  // Input format only gives head; assume initial snake is length 1 at head if no further info.
  // If a harness provides full body in a future variant, it would appear as extra ints; we ignore.
  if (snake_len <= 0) snake_len = 1;

  int F = 0;
  std::cin >> F;
  std::vector<std::vector<Coord>> foods;
  foods.reserve(std::max(0, F));
  for (int i = 0; i < F; ++i) {
    std::vector<Coord> fp(D);
    for (int j = 0; j < D; ++j) std::cin >> fp[j];
    foods.push_back(std::move(fp));
  }

  int O = 0;
  std::cin >> O;
  std::unordered_set<std::vector<Coord>, VecHash> obstacles;
  obstacles.reserve(static_cast<size_t>(std::max(0, O)) * 2u + 8u);
  for (int i = 0; i < O; ++i) {
    std::vector<Coord> op(D);
    for (int j = 0; j < D; ++j) std::cin >> op[j];
    obstacles.insert(std::move(op));
  }

  // Snake body and occupancy.
  std::deque<std::vector<Coord>> snake;
  snake.push_front(head);
  while (static_cast<int>(snake.size()) < snake_len) snake.push_back(head);  // degenerate fill
  std::unordered_set<std::vector<Coord>, VecHash> occupied;
  occupied.reserve(static_cast<size_t>(snake_len) * 2u + 8u);
  for (const auto& p : snake) occupied.insert(p);

  auto is_food_at = [&](const std::vector<Coord>& p) -> int {
    for (int i = 0; i < static_cast<int>(foods.size()); ++i) {
      if (foods[i] == p) return i;
    }
    return -1;
  };

  auto remove_food_index = [&](int idx) {
    if (idx < 0 || idx >= static_cast<int>(foods.size())) return;
    foods[idx] = foods.back();
    foods.pop_back();
  };

  auto nearest_food_index = [&]() -> int {
    if (foods.empty()) return -1;
    int best = 0;
    int64_t bestd = l1_dist(snake.front(), foods[0]);
    for (int i = 1; i < static_cast<int>(foods.size()); ++i) {
      int64_t d = l1_dist(snake.front(), foods[i]);
      if (d < bestd) {
        bestd = d;
        best = i;
      }
    }
    return best;
  };

  auto count_free_neighbors = [&](const std::vector<Coord>& p,
                                  const std::vector<Coord>& tail_if_vacated) -> int {
    int c = 0;
    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, 1}) {
        std::vector<Coord> q = p;
        q[axis] += static_cast<Coord>(dir);
        if (!in_bounds(q, bounds)) continue;
        if (obstacles.find(q) != obstacles.end()) continue;
        if (occupied.find(q) != occupied.end() && q != tail_if_vacated) continue;
        ++c;
      }
    }
    return c;
  };

  auto choose_move = [&]() -> std::optional<Move> {
    const auto& cur = snake.front();
    const auto& tail = snake.back();
    int tgt = nearest_food_index();
    std::vector<Coord> target = (tgt >= 0) ? foods[tgt] : cur;

    // Evaluate all legal moves.
    struct Candidate {
      Move m;
      int64_t score;
      int64_t dist;
      int mobility;
    };
    std::vector<Candidate> candidates;
    candidates.reserve(static_cast<size_t>(D) * 2u);

    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, 1}) {
        std::vector<Coord> nxt = cur;
        nxt[axis] += static_cast<Coord>(dir);
        if (!in_bounds(nxt, bounds)) continue;
        if (obstacles.find(nxt) != obstacles.end()) continue;

        const int food_idx = is_food_at(nxt);
        const bool grows = (food_idx >= 0);
        const std::vector<Coord> vacated = grows ? std::vector<Coord>() : tail;

        if (occupied.find(nxt) != occupied.end()) {
          // Allow moving into the tail position only if tail will vacate.
          if (!( !grows && nxt == tail )) continue;
        }

        const int64_t d = l1_dist(nxt, target);
        const int mobility = count_free_neighbors(nxt, grows ? std::vector<Coord>() : tail);

        // Score: prioritize approaching target, avoid tight spots, and slightly prefer eating.
        int64_t score = 0;
        score -= d * 1000;
        score += static_cast<int64_t>(mobility) * 25;
        if (grows) score += 500;

        // Small tie-break: prefer lower axis then +dir for determinism.
        score -= axis;
        score -= (dir == -1 ? 1 : 0);

        candidates.push_back(Candidate{Move{axis, dir}, score, d, mobility});
      }
    }

    if (candidates.empty()) return std::nullopt;
    std::sort(candidates.begin(), candidates.end(),
              [](const Candidate& a, const Candidate& b) {
                if (a.score != b.score) return a.score > b.score;
                if (a.dist != b.dist) return a.dist < b.dist;
                if (a.mobility != b.mobility) return a.mobility > b.mobility;
                if (a.m.axis != b.m.axis) return a.m.axis < b.m.axis;
                return a.m.dir > b.m.dir;
              });
    return candidates.front().m;
  };

  auto apply_move = [&](const Move& m) -> bool {
    std::vector<Coord> nxt = snake.front();
    nxt[m.axis] += static_cast<Coord>(m.dir);
    if (!in_bounds(nxt, bounds)) return false;
    if (obstacles.find(nxt) != obstacles.end()) return false;

    const int food_idx = is_food_at(nxt);
    const bool grows = (food_idx >= 0);

    // Collision check with tail-vacate rule.
    const auto& tail = snake.back();
    if (occupied.find(nxt) != occupied.end() && !( !grows && nxt == tail )) return false;

    // Advance head.
    snake.push_front(nxt);
    occupied.insert(nxt);

    if (grows) {
      remove_food_index(food_idx);
    } else {
      // Remove tail.
      occupied.erase(tail);
      snake.pop_back();
    }
    return true;
  };

  // If starting cell contains food, eat it "for free".
  const int start_food = is_food_at(snake.front());
  if (start_food >= 0) remove_food_index(start_food);

  for (int t = 0; t < max_turns; ++t) {
    if (foods.empty()) break;
    auto mv = choose_move();
    if (!mv.has_value()) break;
    std::cout << mv->axis << " " << mv->dir << "\n";
    if (!apply_move(*mv)) break;
  }

  return 0;
}
