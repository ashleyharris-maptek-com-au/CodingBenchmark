const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const data = JSON.parse(input);
const depot = data.depot || [0, 0];
const depotX = +depot[0] || 0;
const depotY = +depot[1] || 0;
const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;
let truckCount = Math.max(0, (data.truckCount | 0) || 0);
if (truckCount <= 0) truckCount = 1;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const k = Math.min(truckCount, n);

// Precompute coordinates and ids for speed.
const xs = new Float64Array(n);
const ys = new Float64Array(n);
const ids = new Array(n);
for (let i = 0; i < n; i++) {
  const p = packages[i];
  xs[i] = +p.x || 0;
  ys[i] = +p.y || 0;
  ids[i] = p.id;
}

function distDepot(i) {
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  return Math.hypot(dx, dy);
}

function dist(i, j) {
  const dx = xs[i] - xs[j];
  const dy = ys[i] - ys[j];
  return Math.hypot(dx, dy);
}

function routeLength(route) {
  if (route.length === 0) return 0;
  let total = distDepot(route[0]);
  for (let i = 1; i < route.length; i++) total += dist(route[i - 1], route[i]);
  total += distDepot(route[route.length - 1]);
  return total;
}

// Sweep clustering by polar angle around the depot.
const order = new Array(n);
for (let i = 0; i < n; i++) order[i] = i;
order.sort((a, b) => {
  const aa = Math.atan2(ys[a] - depotY, xs[a] - depotX);
  const bb = Math.atan2(ys[b] - depotY, xs[b] - depotX);
  return aa - bb;
});

// Balanced partition sizes (contiguous in sweep order).
const baseSize = Math.floor(n / k);
let extra = n - baseSize * k; // first `extra` routes get +1

const groups = [];
let pos = 0;
for (let gi = 0; gi < k; gi++) {
  const size = baseSize + (extra > 0 ? 1 : 0);
  if (extra > 0) extra--;
  const g = order.slice(pos, pos + size);
  pos += size;
  groups.push(g);
}

// Build a route using cheapest insertion (good quality, still simple).
function buildCheapestInsertion(group) {
  const m = group.length;
  if (m <= 1) return group.slice();

  // Start with the farthest package from depot as an anchor, then add its nearest neighbor.
  let far = group[0];
  let farD = -1;
  for (let i = 0; i < m; i++) {
    const idx = group[i];
    const d = distDepot(idx);
    if (d > farD) {
      farD = d;
      far = idx;
    }
  }

  let near = null;
  let nearD = Infinity;
  for (let i = 0; i < m; i++) {
    const idx = group[i];
    if (idx === far) continue;
    const d = dist(far, idx);
    if (d < nearD) {
      nearD = d;
      near = idx;
    }
  }

  const used = new Uint8Array(n);
  used[far] = 1;
  const route = [far];
  if (near !== null) {
    used[near] = 1;
    route.push(near);
  }

  let remaining = m - route.length;
  while (remaining > 0) {
    let bestNode = -1;
    let bestPos = -1;
    let bestDelta = Infinity;

    // Consider inserting any unused node into any position with minimal length increase.
    for (let gi = 0; gi < m; gi++) {
      const node = group[gi];
      if (used[node]) continue;

      // Insert at beginning: depot -> node -> first
      {
        const delta = distDepot(node) + dist(node, route[0]) - distDepot(route[0]);
        if (delta < bestDelta) {
          bestDelta = delta;
          bestNode = node;
          bestPos = 0;
        }
      }

      // Insert between route[i-1] and route[i]
      for (let i = 1; i < route.length; i++) {
        const a = route[i - 1];
        const b = route[i];
        const delta = dist(a, node) + dist(node, b) - dist(a, b);
        if (delta < bestDelta) {
          bestDelta = delta;
          bestNode = node;
          bestPos = i;
        }
      }

      // Insert at end: last -> node -> depot
      {
        const last = route[route.length - 1];
        const delta = dist(last, node) + distDepot(node) - distDepot(last);
        if (delta < bestDelta) {
          bestDelta = delta;
          bestNode = node;
          bestPos = route.length;
        }
      }
    }

    used[bestNode] = 1;
    route.splice(bestPos, 0, bestNode);
    remaining--;
  }

  return route;
}

function twoOpt(route, maxChecks) {
  const m = route.length;
  if (m < 4) return route;
  let improved = true;
  let checks = 0;

  while (improved && checks < maxChecks) {
    improved = false;
    for (let i = 0; i < m - 1; i++) {
      const aPrev = i === 0 ? -1 : route[i - 1];
      const a = route[i];
      const b = route[i + 1];

      const distAPrevA = aPrev === -1 ? distDepot(a) : dist(aPrev, a);
      const distAB = dist(a, b);

      for (let j = i + 2; j < m; j++) {
        const c = route[j];
        const dNext = j === m - 1 ? -1 : route[j + 1];

        const distCDNext = dNext === -1 ? distDepot(c) : dist(c, dNext);
        const distBDNext = dNext === -1 ? distDepot(b) : dist(b, dNext);

        const distAC = dist(a, c);

        // Current edges: (aPrev->a) + (b->...?) are unaffected; we swap (a->b) and (c->dNext)
        // with (a->c) and (b->dNext) while reversing segment (b..c).
        const before = distAB + distCDNext;
        const after = distAC + distBDNext;

        checks++;
        if (after + 1e-9 < before) {
          // Reverse route[i+1..j]
          let l = i + 1;
          let r = j;
          while (l < r) {
            const tmp = route[l];
            route[l] = route[r];
            route[r] = tmp;
            l++;
            r--;
          }
          improved = true;
          break;
        }
        if (checks >= maxChecks) break;
      }
      if (improved || checks >= maxChecks) break;
      // Small micro-optimization: if depot->a is huge, 2-opt can help; otherwise continue.
      // (kept intentionally minimal and deterministic)
      void distAPrevA;
    }
  }
  return route;
}

// Build and locally improve each route.
const routesIdx = [];
for (let gi = 0; gi < groups.length; gi++) {
  const group = groups[gi];
  let route = buildCheapestInsertion(group);
  // Keep 2-opt checks roughly quadratic but capped, to scale.
  const m = route.length;
  const maxChecks = Math.min(250000, Math.max(2000, m * m * 4));
  route = twoOpt(route, maxChecks);
  routesIdx.push(route);
}

// Output package ids in route order.
const outRoutes = routesIdx.map((r) => r.map((idx) => ids[idx]));

process.stdout.write(JSON.stringify({ routes: outRoutes }));
