using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    // Tetris bot that maintains its own board state and chooses a placement per piece
    // using a fast heuristic evaluator. No next-piece preview assumed.

    private struct Move
    {
        public int Rot;
        public int Col;
        public int Score;
        public Move(int rot, int col, int score) { Rot = rot; Col = col; Score = score; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _pos;
        public FastScanner(Stream s) { _s = s; }
        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }
        public string? Next()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return null;
            } while (c <= 32);
            var sb = new System.Text.StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c == -1) break;
            }
            return sb.ToString();
        }
        public int NextInt()
        {
            var t = Next();
            if (t == null) throw new EndOfStreamException();
            return int.Parse(t);
        }
    }

    // Board stored bottom-up: row 0 is bottom.
    private static int _w, _h;
    private static ulong[] _rows = Array.Empty<ulong>();
    private static ulong _fullMask;
    private static int _maxHeight; // cached highest occupied cell (0..h)

    // Piece definitions: for each piece type and rotation, 4 blocks (x,y) with min x=0, min y=0.
    // Rotations follow standard shapes without wall kicks; placement is leftmost column of the rotation's bounding box.
    private static readonly Dictionary<char, (int w, int h, (int x, int y)[] blocks)[]> Pieces =
        new Dictionary<char, (int w, int h, (int x, int y)[] blocks)[]>
        {
            // I
            ['I'] = new[]
            {
                (4,1, new[]{(0,0),(1,0),(2,0),(3,0)}),
                (1,4, new[]{(0,0),(0,1),(0,2),(0,3)}),
                (4,1, new[]{(0,0),(1,0),(2,0),(3,0)}),
                (1,4, new[]{(0,0),(0,1),(0,2),(0,3)}),
            },
            // O
            ['O'] = new[]
            {
                (2,2, new[]{(0,0),(1,0),(0,1),(1,1)}),
                (2,2, new[]{(0,0),(1,0),(0,1),(1,1)}),
                (2,2, new[]{(0,0),(1,0),(0,1),(1,1)}),
                (2,2, new[]{(0,0),(1,0),(0,1),(1,1)}),
            },
            // T
            ['T'] = new[]
            {
                (3,2, new[]{(0,0),(1,0),(2,0),(1,1)}),
                (2,3, new[]{(0,0),(0,1),(0,2),(1,1)}),
                (3,2, new[]{(1,0),(0,1),(1,1),(2,1)}),
                (2,3, new[]{(1,0),(0,1),(1,1),(1,2)}),
            },
            // S (right-leaning)
            ['S'] = new[]
            {
                (3,2, new[]{(1,0),(2,0),(0,1),(1,1)}),
                (2,3, new[]{(0,0),(0,1),(1,1),(1,2)}),
                (3,2, new[]{(1,0),(2,0),(0,1),(1,1)}),
                (2,3, new[]{(0,0),(0,1),(1,1),(1,2)}),
            },
            // Z (left-leaning)
            ['Z'] = new[]
            {
                (3,2, new[]{(0,0),(1,0),(1,1),(2,1)}),
                (2,3, new[]{(1,0),(0,1),(1,1),(0,2)}),
                (3,2, new[]{(0,0),(1,0),(1,1),(2,1)}),
                (2,3, new[]{(1,0),(0,1),(1,1),(0,2)}),
            },
            // J (left-leaning L)
            ['J'] = new[]
            {
                (3,2, new[]{(0,0),(0,1),(1,0),(2,0)}),
                (2,3, new[]{(0,0),(1,0),(0,1),(0,2)}),
                (3,2, new[]{(0,1),(1,1),(2,1),(2,0)}),
                (2,3, new[]{(1,0),(1,1),(1,2),(0,2)}),
            },
            // L (right-leaning L)
            ['L'] = new[]
            {
                (3,2, new[]{(0,0),(1,0),(2,0),(2,1)}),
                (2,3, new[]{(0,0),(0,1),(0,2),(1,2)}),
                (3,2, new[]{(0,1),(0,0),(1,1),(2,1)}),
                (2,3, new[]{(0,0),(1,0),(1,1),(1,2)}),
            },
        };

    private static bool Collides((int w, int h, (int x, int y)[] blocks) shape, int col, int row)
    {
        foreach (var (bx, by) in shape.blocks)
        {
            int x = col + bx;
            int y = row + by;
            if ((uint)x >= (uint)_w) return true;
            if (y < 0) return true;
            if (y >= _h) return true;
            if (((_rows[y] >> x) & 1UL) != 0) return true;
        }
        return false;
    }

    private static int DropRow((int w, int h, (int x, int y)[] blocks) shape, int col)
    {
        // Start above top and drop until collision; final is last non-colliding.
        // We can start at _h - shape.h and move down, but stacks might be lower; fast scan downwards.
        int start = _h - shape.h;
        int r = start;
        while (r > 0 && !Collides(shape, col, r - 1)) r--;
        // r is lowest valid.
        return r;
    }

    private static int PlaceAndClear((int w, int h, (int x, int y)[] blocks) shape, int col, int row, ulong[] workRows)
    {
        // Copy current rows into workRows before calling.
        foreach (var (bx, by) in shape.blocks)
        {
            int x = col + bx;
            int y = row + by;
            workRows[y] |= 1UL << x;
        }

        int cleared = 0;
        int write = 0;
        for (int y = 0; y < _h; y++)
        {
            if ((workRows[y] & _fullMask) == _fullMask)
            {
                cleared++;
            }
            else
            {
                workRows[write++] = workRows[y];
            }
        }
        for (int y = write; y < _h; y++) workRows[y] = 0;
        return cleared;
    }

    private static int EvaluateBoard(ulong[] brd, int clearedLines, bool survivalMode)
    {
        // Heuristic: minimize holes and bumpiness and max height; reward line clears.
        int[] heights = new int[_w];
        int maxH = 0;
        int holes = 0;
        int blocks = 0;

        for (int x = 0; x < _w; x++)
        {
            int y = _h - 1;
            while (y >= 0 && ((brd[y] >> x) & 1UL) == 0) y--;
            int h = y + 1;
            heights[x] = h;
            if (h > maxH) maxH = h;

            bool seenBlock = false;
            for (int yy = _h - 1; yy >= 0; yy--)
            {
                bool filled = ((brd[yy] >> x) & 1UL) != 0;
                if (filled)
                {
                    seenBlock = true;
                    blocks++;
                }
                else if (seenBlock)
                {
                    holes++;
                }
            }
        }

        int bump = 0;
        for (int i = 0; i < _w - 1; i++) bump += Math.Abs(heights[i] - heights[i + 1]);

        int aggregateHeight = 0;
        for (int i = 0; i < _w; i++) aggregateHeight += heights[i];

        // Well depth (simple): count local minima relative to neighbors.
        int wells = 0;
        for (int x = 0; x < _w; x++)
        {
            int left = x == 0 ? maxH : heights[x - 1];
            int right = x == _w - 1 ? maxH : heights[x + 1];
            int minNeighbor = Math.Min(left, right);
            if (minNeighbor > heights[x]) wells += (minNeighbor - heights[x]);
        }

        // Adaptive weights.
        int wClears = survivalMode ? 250 : 400;
        int wMaxH = survivalMode ? 40 : 25;
        int wAggH = survivalMode ? 10 : 7;
        int wHoles = survivalMode ? 200 : 140;
        int wBump = survivalMode ? 20 : 12;
        int wWells = survivalMode ? 6 : 10; // in survival, avoid deep wells unless they help I pieces

        int score = 0;
        score += clearedLines * wClears;
        score -= maxH * wMaxH;
        score -= aggregateHeight * wAggH;
        score -= holes * wHoles;
        score -= bump * wBump;
        score -= wells * wWells;

        // Extra penalty as we near top.
        int danger = Math.Max(0, maxH - (_h * 3) / 4);
        score -= danger * danger * 15;

        // Slightly reward "flatness" on wide boards to reduce future holes.
        if (_w >= 14) score -= bump * 3;

        return score;
    }

    private static Move ChooseMove(char pieceType)
    {
        var shapes = Pieces[pieceType];
        bool survivalMode = _maxHeight >= (_h * 2) / 3;

        // Work buffer for simulation.
        ulong[] work = new ulong[_h];
        Move best = new Move(0, 0, int.MinValue);

        for (int rot = 0; rot < 4; rot++)
        {
            var shape = shapes[rot];
            int maxCol = _w - shape.w;
            if (maxCol < 0) continue;
            for (int col = 0; col <= maxCol; col++)
            {
                // Quick reject: collision at spawn top row region.
                int row = DropRow(shape, col);
                if (Collides(shape, col, row)) continue;

                Array.Copy(_rows, work, _h);
                int cleared = PlaceAndClear(shape, col, row, work);
                int score = EvaluateBoard(work, cleared, survivalMode);

                // Tie-breaker: prefer fewer rotations and leftmost placements to be deterministic.
                if (score > best.Score ||
                    (score == best.Score && rot < best.Rot) ||
                    (score == best.Score && rot == best.Rot && col < best.Col))
                {
                    best = new Move(rot, col, score);
                }
            }
        }

        // Fallback: if no legal move found, output something valid-looking.
        if (best.Score == int.MinValue) best = new Move(0, 0, int.MinValue);
        return best;
    }

    private static void ApplyOurMove(char pieceType, int rot, int col)
    {
        var shape = Pieces[pieceType][rot & 3];
        int row = DropRow(shape, col);
        if (Collides(shape, col, row))
        {
            // Our model says gameover; do nothing else.
            return;
        }
        foreach (var (bx, by) in shape.blocks)
        {
            int x = col + bx;
            int y = row + by;
            _rows[y] |= 1UL << x;
        }

        // Clear lines.
        int write = 0;
        for (int y = 0; y < _h; y++)
        {
            if ((_rows[y] & _fullMask) == _fullMask) { /* drop */ }
            else _rows[write++] = _rows[y];
        }
        for (int y = write; y < _h; y++) _rows[y] = 0;

        // Update cached max height.
        int mh = 0;
        for (int y = _h - 1; y >= 0; y--)
        {
            if (_rows[y] != 0) { mh = y + 1; break; }
        }
        _maxHeight = mh;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        string? tok = fs.Next();
        if (tok == null) return;
        _w = int.Parse(tok);
        _h = fs.NextInt();
        int numPieces = fs.NextInt();
        if (_w <= 0 || _h <= 0) return;
        if (_w > 63) _w = 63; // keep within ulong; spec says typically <= 20.

        _rows = new ulong[_h];
        _fullMask = _w == 64 ? ulong.MaxValue : ((1UL << _w) - 1UL);
        _maxHeight = 0;

        var output = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        int placed = 0;
        while (true)
        {
            if (numPieces >= 0 && placed >= numPieces) break;
            string? p = fs.Next();
            if (p == null) break;
            if (p.Length == 0) continue;
            char pieceType = p[0];
            if (!Pieces.ContainsKey(pieceType))
            {
                // Unknown piece; output a safe default.
                output.WriteLine("0 0");
                string? res0 = fs.Next();
                if (res0 == null) break;
                if (res0.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;
                placed++;
                continue;
            }

            Move mv = ChooseMove(pieceType);
            output.WriteLine(mv.Rot + " " + mv.Col);

            // Read result token (ok, cleared_N, gameover). We still update our own board deterministically.
            string? res = fs.Next();
            if (res == null) break;

            ApplyOurMove(pieceType, mv.Rot, mv.Col);

            if (res.StartsWith("gameover", StringComparison.OrdinalIgnoreCase)) break;
            placed++;
        }
    }
}
