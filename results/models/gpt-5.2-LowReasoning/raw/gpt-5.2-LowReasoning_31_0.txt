use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

fn parse_i64s(input: &[u8]) -> Vec<i64> {
    let mut out = Vec::new();
    let mut i = 0usize;
    while i < input.len() {
        while i < input.len() && input[i].is_ascii_whitespace() {
            i += 1;
        }
        if i >= input.len() {
            break;
        }
        let mut sign = 1i64;
        if input[i] == b'-' {
            sign = -1;
            i += 1;
        } else if input[i] == b'+' {
            i += 1;
        }
        let mut val: i64 = 0;
        while i < input.len() {
            let c = input[i];
            if !c.is_ascii_digit() {
                break;
            }
            val = val.saturating_mul(10).saturating_add((c - b'0') as i64);
            i += 1;
        }
        out.push(val * sign);
    }
    out
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    let mut left_sums: Vec<(i64, u64)> = Vec::with_capacity(1usize << left.len());
    for mask in 0u64..(1u64 << left.len()) {
        let mut s = 0i64;
        let mut m = mask;
        let mut idx = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                s = s.wrapping_add(left[idx]);
            }
            idx += 1;
            m >>= 1;
        }
        left_sums.push((s, mask));
    }
    left_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    let mut right_sums: Vec<(i64, u64)> = Vec::with_capacity(1usize << right.len());
    for mask in 0u64..(1u64 << right.len()) {
        let mut s = 0i64;
        let mut m = mask;
        let mut idx = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                s = s.wrapping_add(right[idx]);
            }
            idx += 1;
            m >>= 1;
        }
        right_sums.push((s, mask));
    }
    right_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    let mut i = 0usize;
    let mut j = right_sums.len().saturating_sub(1);
    while i < left_sums.len() && j < right_sums.len() {
        let s = left_sums[i].0.wrapping_add(right_sums[j].0);
        match s.cmp(&target) {
            Ordering::Equal => {
                let mut picks = Vec::new();
                let lm = left_sums[i].1;
                for k in 0..left.len() {
                    if ((lm >> k) & 1) != 0 {
                        picks.push(k);
                    }
                }
                let rm = right_sums[j].1;
                for k in 0..right.len() {
                    if ((rm >> k) & 1) != 0 {
                        picks.push(mid + k);
                    }
                }
                return Some(picks);
            }
            Ordering::Less => i += 1,
            Ordering::Greater => {
                if j == 0 {
                    break;
                }
                j -= 1;
            }
        }
    }
    None
}

fn dp_nonneg_small_target(nums: &[i64], target: usize) -> Option<Vec<usize>> {
    let mut reachable = vec![false; target + 1];
    let mut prev_sum: Vec<i32> = vec![-1; target + 1];
    let mut prev_idx: Vec<i32> = vec![-1; target + 1];
    reachable[0] = true;

    for (idx, &v) in nums.iter().enumerate() {
        if v <= 0 {
            continue;
        }
        let w = v as usize;
        if w > target {
            continue;
        }
        for s in (w..=target).rev() {
            if !reachable[s] && reachable[s - w] {
                reachable[s] = true;
                prev_sum[s] = (s - w) as i32;
                prev_idx[s] = idx as i32;
            }
        }
        if reachable[target] {
            break;
        }
    }
    if !reachable[target] {
        return None;
    }

    let mut picks = Vec::new();
    let mut cur = target as i32;
    while cur != 0 {
        let pi = prev_idx[cur as usize];
        if pi < 0 {
            break;
        }
        picks.push(pi as usize);
        cur = prev_sum[cur as usize];
        if cur < 0 {
            break;
        }
    }
    Some(picks)
}

fn hash_dp(nums: &[i64], target: i64, max_states: usize) -> Option<Vec<usize>> {
    let mut cur: HashMap<i64, (i64, i32)> = HashMap::new();
    cur.insert(0, (0, -1)); // sum -> (prev_sum, idx)

    for (idx, &v) in nums.iter().enumerate() {
        if cur.contains_key(&target) {
            break;
        }
        let mut next = cur.clone();
        for (&s, _) in cur.iter() {
            let ns = s.wrapping_add(v);
            if !next.contains_key(&ns) {
                next.insert(ns, (s, idx as i32));
            }
        }
        if next.len() > max_states {
            // prune: keep sums closest to target
            let mut vec: Vec<(i64, (i64, i32))> = next.into_iter().collect();
            vec.sort_unstable_by_key(|(s, _)| (target.wrapping_sub(*s)).abs());
            vec.truncate(max_states);
            next = vec.into_iter().collect();
        }
        cur = next;
    }

    if !cur.contains_key(&target) {
        return None;
    }
    let mut picks = Vec::new();
    let mut s = target;
    while s != 0 {
        let (ps, idx) = *cur.get(&s).unwrap();
        if idx < 0 {
            break;
        }
        picks.push(idx as usize);
        s = ps;
    }
    Some(picks)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let vals = parse_i64s(input.as_bytes());
    if vals.len() < 2 {
        print!("NO");
        return;
    }
    let n = vals[0].max(0) as usize;
    let target = vals[1];
    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for i in 0..n {
        if 2 + i < vals.len() {
            nums.push(vals[2 + i]);
        } else {
            break;
        }
    }

    if target == 0 {
        println!("YES");
        println!();
        return;
    }
    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            println!("YES");
            println!("{i}");
            return;
        }
    }

    let all_nonneg = nums.iter().all(|&x| x >= 0);
    let answer: Option<Vec<usize>> = if nums.len() <= 44 {
        meet_in_middle(&nums, target)
    } else if all_nonneg && target > 0 {
        // Exact DP when target is reasonably small.
        // 2_000_000 => ~2MB reachable + ~16MB parents; fine for 32GB machines.
        let t = target as u64;
        if t <= 2_000_000 {
            dp_nonneg_small_target(&nums, t as usize)
        } else if nums.len() <= 60 {
            // Larger target but smaller n: hash DP.
            hash_dp(&nums, target, 2_000_000)
        } else {
            // Huge n and huge target is intractable in worst case; try bounded hash DP.
            hash_dp(&nums, target, 400_000)
        }
    } else {
        // General integers (may include negatives): bounded hash DP.
        // This is exact if it finishes without pruning; pruning trades certainty for speed.
        let states = if nums.len() <= 2000 { 2_000_000 } else { 500_000 };
        hash_dp(&nums, target, states)
    };

    if let Some(mut picks) = answer {
        // validate and normalize
        picks.sort_unstable();
        picks.dedup();
        let mut sum = 0i64;
        for &i in picks.iter() {
            if i < nums.len() {
                sum = sum.wrapping_add(nums[i]);
            }
        }
        if sum != target {
            print!("NO");
            return;
        }
        println!("YES");
        if picks.is_empty() {
            println!();
        } else {
            for (k, idx) in picks.iter().enumerate() {
                if k != 0 {
                    print!(" ");
                }
                print!("{idx}");
            }
            println!();
        }
    } else {
        print!("NO");
    }
}
