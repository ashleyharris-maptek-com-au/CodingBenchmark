using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public double X, Y, Z;
        public double Vx, Vy, Vz;
    }

    private sealed class Scanner
    {
        private readonly string[] _tokens;
        private int _i;

        public Scanner(string all)
        {
            _tokens = all.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            _i = 0;
        }

        public bool HasNext => _i < _tokens.Length;

        public string Next()
        {
            if (_i >= _tokens.Length) throw new InvalidOperationException("Unexpected end of input.");
            return _tokens[_i++];
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), NumberStyles.Float, CultureInfo.InvariantCulture);
    }

    private static (double x, double y, double z) Add((double x, double y, double z) a, (double x, double y, double z) b)
        => (a.x + b.x, a.y + b.y, a.z + b.z);

    private static (double x, double y, double z) Sub((double x, double y, double z) a, (double x, double y, double z) b)
        => (a.x - b.x, a.y - b.y, a.z - b.z);

    private static (double x, double y, double z) Mul((double x, double y, double z) a, double s)
        => (a.x * s, a.y * s, a.z * s);

    private static double Dot((double x, double y, double z) a, (double x, double y, double z) b)
        => a.x * b.x + a.y * b.y + a.z * b.z;

    private static double Norm((double x, double y, double z) a)
        => Math.Sqrt(Dot(a, a));

    private static (double x, double y, double z) SafeUnit((double x, double y, double z) v)
    {
        var n = Norm(v);
        if (n <= 0) return (1, 0, 0);
        return (v.x / n, v.y / n, v.z / n);
    }

    private static (double x, double y, double z) PropPos(Body b, double tSeconds)
        => (b.X + b.Vx * tSeconds, b.Y + b.Vy * tSeconds, b.Z + b.Vz * tSeconds);

    private static (double x, double y, double z) PropVel(Body b, double tSeconds)
        => (b.Vx, b.Vy, b.Vz);

    public static void Main()
    {
        // Heuristic solver:
        // - Uses constant-velocity propagation of Earth/Asteroid over the warning horizon.
        // - Searches over launch time + time-of-flight to minimize miss distance at intercept.
        // - Outputs an impulsive launch burn; impact "burn" is 0 (kinetic impact event timing).
        //
        // This is intentionally lightweight to fit within strict time budgets and unknown scenarios.

        var input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            // Fallback: no burns.
            Console.WriteLine("0");
            return;
        }

        var sc = new Scanner(input);

        int numBodies = sc.NextInt();
        double warningDays = sc.NextDouble();
        _ = sc.NextDouble(); // impactor_mass (not used in this heuristic)
        double dvBudget = sc.NextDouble();
        _ = sc.NextDouble(); // asteroid_mass (not used directly)
        _ = sc.NextDouble(); // t0_unix_utc (not used)

        var bodies = new List<Body>(numBodies);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = sc.Next();
            b.Mass = sc.NextDouble();
            b.X = sc.NextDouble();
            b.Y = sc.NextDouble();
            b.Z = sc.NextDouble();
            b.Vx = sc.NextDouble();
            b.Vy = sc.NextDouble();
            b.Vz = sc.NextDouble();
            bodies.Add(b);
        }

        // Last token is de421.bsp file path; keep scanner aligned even if path contains spaces
        // (input spec says whitespace-separated, so path is a single token).
        string _de421Path = sc.HasNext ? sc.Next() : "";

        Body? earth = null;
        Body? asteroid = null;
        foreach (var b in bodies)
        {
            if (string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase)) earth = b;
            if (string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase)) asteroid = b;
        }

        if (earth == null || asteroid == null || warningDays <= 0 || dvBudget <= 0)
        {
            // Minimal valid output: one burn at t=0 with zero dv.
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        const double Day = 86400.0;
        double horizonSeconds = warningDays * Day;

        // Coarse-to-fine search.
        // Launch in [0, warningDays]. Intercept in (launch, warningDays].
        int coarseLaunchSteps = 40;
        int coarseTofSteps = 60;

        double bestLaunch = 0.0;
        double bestIntercept = Math.Min(warningDays, 1.0);
        (double x, double y, double z) bestDv = (0, 0, 0);
        double bestScore = double.PositiveInfinity;

        // Score combines predicted miss distance and dv usage (prefer within budget).
        double DvPenaltyWeight = 1e6; // meters per (m/s) to keep dv reasonable but prioritize intercept

        for (int i = 0; i <= coarseLaunchSteps; i++)
        {
            double launchDays = warningDays * i / coarseLaunchSteps;
            double launchT = launchDays * Day;

            var rE = PropPos(earth, launchT);
            var vE = PropVel(earth, launchT);

            // Avoid TOF too small; allow up to remaining horizon.
            double remainingDays = Math.Max(0.01, warningDays - launchDays);
            for (int j = 1; j <= coarseTofSteps; j++)
            {
                double tofDays = remainingDays * j / coarseTofSteps;
                if (tofDays < 0.02) continue;

                double interceptDays = launchDays + tofDays;
                double interceptT = interceptDays * Day;

                var rA = PropPos(asteroid, interceptT);
                var vA = PropVel(asteroid, interceptT);

                // Constant-velocity "Lambert": want spacecraft at rA at time intercept.
                // Spacecraft starts at rE with initial velocity vE + dv.
                // rE + (vE + dv)*tof = rA  => dv = (rA - rE)/tof - vE.
                var tof = tofDays * Day;
                var vReq = Mul(Sub(rA, rE), 1.0 / tof);
                var dv = Sub(vReq, vE);
                double dvMag = Norm(dv);

                // Predict miss distance at intercept if we apply this dv but clamp to budget.
                double scale = dvMag > 1e-9 ? Math.Min(1.0, dvBudget / dvMag) : 1.0;
                var dvClamped = Mul(dv, scale);
                var vSc = Add(vE, dvClamped);
                var rSc = Add(rE, Mul(vSc, tof));
                double miss = Norm(Sub(rSc, rA));

                // Prefer higher relative speed at impact for momentum transfer (weakly).
                var vScAtImpact = vSc; // constant velocity approx
                double vRel = Norm(Sub(vScAtImpact, vA));
                double relBonus = -1000.0 * vRel; // reduce score for higher vRel

                double dvOver = Math.Max(0.0, dvMag - dvBudget);
                double score = miss + DvPenaltyWeight * dvOver + relBonus;

                if (score < bestScore)
                {
                    bestScore = score;
                    bestLaunch = launchDays;
                    bestIntercept = interceptDays;
                    bestDv = dvClamped;
                }
            }
        }

        // Small refinement around best solution.
        void Refine(double launchCenter, double interceptCenter)
        {
            double launchSpan = Math.Max(0.5, warningDays / 20.0);
            double interceptSpan = Math.Max(0.5, warningDays / 15.0);

            double l0 = Math.Max(0.0, launchCenter - launchSpan);
            double l1 = Math.Min(warningDays, launchCenter + launchSpan);
            double i0 = Math.Max(0.0, interceptCenter - interceptSpan);
            double i1 = Math.Min(warningDays, interceptCenter + interceptSpan);

            int L = 50, K = 60;
            for (int i = 0; i <= L; i++)
            {
                double launchDays = l0 + (l1 - l0) * i / L;
                double launchT = launchDays * Day;
                var rE = PropPos(earth, launchT);
                var vE = PropVel(earth, launchT);

                for (int j = 0; j <= K; j++)
                {
                    double interceptDays = i0 + (i1 - i0) * j / K;
                    if (interceptDays <= launchDays + 0.02) continue;
                    double tofDays = interceptDays - launchDays;
                    double interceptT = interceptDays * Day;

                    var rA = PropPos(asteroid, interceptT);
                    var vA = PropVel(asteroid, interceptT);

                    var tof = tofDays * Day;
                    var vReq = Mul(Sub(rA, rE), 1.0 / tof);
                    var dv = Sub(vReq, vE);
                    double dvMag = Norm(dv);
                    double scale = dvMag > 1e-9 ? Math.Min(1.0, dvBudget / dvMag) : 1.0;
                    var dvClamped = Mul(dv, scale);

                    var vSc = Add(vE, dvClamped);
                    var rSc = Add(rE, Mul(vSc, tof));
                    double miss = Norm(Sub(rSc, rA));

                    double vRel = Norm(Sub(vSc, vA));
                    double relBonus = -1000.0 * vRel;
                    double dvOver = Math.Max(0.0, dvMag - dvBudget);
                    double score = miss + DvPenaltyWeight * dvOver + relBonus;

                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestLaunch = launchDays;
                        bestIntercept = interceptDays;
                        bestDv = dvClamped;
                    }
                }
            }
        }

        Refine(bestLaunch, bestIntercept);

        // Ensure within launch window and monotonic.
        if (bestLaunch < 0) bestLaunch = 0;
        if (bestLaunch > warningDays) bestLaunch = warningDays;
        if (bestIntercept <= bestLaunch) bestIntercept = Math.Min(warningDays, bestLaunch + 0.5);

        // If dv ends up nearly zero, still point toward asteroid for a minimal nudge.
        if (Norm(bestDv) < 1e-6)
        {
            var rE0 = PropPos(earth, bestLaunch * Day);
            var rA0 = PropPos(asteroid, bestIntercept * Day);
            var dir = SafeUnit(Sub(rA0, rE0));
            bestDv = Mul(dir, Math.Min(dvBudget, 1.0)); // 1 m/s or budget if smaller
        }

        // Output: launch burn and "impact event" line with zero dv.
        // Times are in days since t0.
        Console.WriteLine("2");
        Console.WriteLine(FormLine(bestLaunch, bestDv.x, bestDv.y, bestDv.z));
        Console.WriteLine(FormLine(bestIntercept, 0.0, 0.0, 0.0));
    }

    private static string FormLine(double tDays, double dvx, double dvy, double dvz)
    {
        return string.Create(CultureInfo.InvariantCulture, $"{tDays:R} {dvx:R} {dvy:R} {dvz:R}");
    }
}
