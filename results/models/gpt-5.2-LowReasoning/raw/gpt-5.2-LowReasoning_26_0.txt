use std::cmp::min;
use std::io::{self, Read, Write};

#[inline]
fn read_u64_ascii(buf: &[u8], idx: &mut usize) -> u64 {
    while *idx < buf.len() && buf[*idx].is_ascii_whitespace() {
        *idx += 1;
    }
    let mut v: u64 = 0;
    while *idx < buf.len() {
        let b = buf[*idx];
        if !b.is_ascii_digit() {
            break;
        }
        v = v * 10 + (b - b'0') as u64;
        *idx += 1;
    }
    v
}

#[inline]
fn read_f32_le(buf: &[u8], idx: &mut usize) -> f32 {
    let b0 = buf[*idx];
    let b1 = buf[*idx + 1];
    let b2 = buf[*idx + 2];
    let b3 = buf[*idx + 3];
    *idx += 4;
    f32::from_le_bytes([b0, b1, b2, b3])
}

#[inline]
fn dist2(a: [f32; 3], b: [f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn nearest_linear(p: [f32; 3], centroids: &[[f32; 3]]) -> (usize, f32) {
    let mut best_i = 0usize;
    let mut best_d = f32::INFINITY;
    for (i, &c) in centroids.iter().enumerate() {
        let d = dist2(p, c);
        if d < best_d {
            best_d = d;
            best_i = i;
        }
    }
    (best_i, best_d)
}

fn pick_coarse_count(k: usize) -> usize {
    if k <= 2048 {
        return 0;
    }
    // Keep this small to reduce per-point overhead; 256 is a good fixed target.
    let mut m = (k as f64).sqrt().round() as usize;
    m = m.clamp(64, 256);
    min(m, k)
}

fn build_partition(k: usize, coarse: usize) -> (Vec<usize>, Vec<(usize, usize)>) {
    // Assign fine centroids to coarse buckets by contiguous ranges:
    // fine_to_coarse[i] in [0, coarse)
    // coarse_ranges[j] = [start, end) in fine indices
    let mut fine_to_coarse = vec![0usize; k];
    let mut coarse_ranges = Vec::with_capacity(coarse);
    for j in 0..coarse {
        let start = (j * k) / coarse;
        let end = ((j + 1) * k) / coarse;
        coarse_ranges.push((start, end));
        for i in start..end {
            fine_to_coarse[i] = j;
        }
    }
    (fine_to_coarse, coarse_ranges)
}

fn online_streaming_cluster(
    mut input: &[u8],
    num_points: u64,
    k: usize,
    output: &mut dyn Write,
) -> io::Result<()> {
    // Single-pass online k-means:
    // - Initialize centroids from first k points (or repeat if n < k)
    // - For each point: assign nearest, update centroid with 1/count learning rate, emit label.
    //
    // For large k, use a coarse->fine partition to avoid scanning all k every time.
    let mut idx = 0usize;

    let mut centroids = vec![[0.0f32; 3]; k.max(1)];
    let mut counts = vec![0u64; k.max(1)];

    let mut points_read = 0u64;
    let init_n = min(num_points as usize, k.max(1));
    for i in 0..init_n {
        if idx + 12 > input.len() {
            break;
        }
        let x = read_f32_le(input, &mut idx);
        let y = read_f32_le(input, &mut idx);
        let z = read_f32_le(input, &mut idx);
        centroids[i] = [x, y, z];
        counts[i] = 1;
        // Emit initialization labels matching their own centroid index.
        output.write_all(&(i as u32).to_le_bytes())?;
        points_read += 1;
    }
    if num_points == 0 {
        return Ok(());
    }
    if k == 0 {
        // No clusters requested; output zeros.
        for _ in points_read..num_points {
            output.write_all(&0u32.to_le_bytes())?;
        }
        return Ok(());
    }
    if init_n == 0 {
        // No points available in buffer; nothing to do.
        return Ok(());
    }
    // If n < k, we still must output exactly one label per point; we already did.
    // Remaining centroids stay at [0,0,0] but will never be used.

    let coarse = pick_coarse_count(k);
    let (fine_to_coarse, coarse_ranges) = if coarse > 0 {
        build_partition(k, coarse)
    } else {
        (Vec::new(), Vec::new())
    };
    let mut coarse_centroids = if coarse > 0 {
        // Initialize coarse centroids as mean of the fine centroids in each range.
        let mut cc = vec![[0.0f32; 3]; coarse];
        for (j, (s, e)) in coarse_ranges.iter().copied().enumerate() {
            let mut sum = [0.0f32; 3];
            let mut cnt = 0.0f32;
            for i in s..e {
                sum[0] += centroids[i][0];
                sum[1] += centroids[i][1];
                sum[2] += centroids[i][2];
                cnt += 1.0;
            }
            if cnt > 0.0 {
                cc[j] = [sum[0] / cnt, sum[1] / cnt, sum[2] / cnt];
            } else {
                cc[j] = centroids[0];
            }
        }
        cc
    } else {
        Vec::new()
    };
    let mut coarse_counts = if coarse > 0 { vec![1u64; coarse] } else { Vec::new() };

    while points_read < num_points {
        if idx + 12 > input.len() {
            break;
        }
        let p = [
            read_f32_le(input, &mut idx),
            read_f32_le(input, &mut idx),
            read_f32_le(input, &mut idx),
        ];

        let chosen = if coarse > 0 {
            let (cj, _) = nearest_linear(p, &coarse_centroids);
            let (s, e) = coarse_ranges[cj];
            let (mut best_i, mut best_d) = (s, f32::INFINITY);
            for i in s..e {
                let d = dist2(p, centroids[i]);
                if d < best_d {
                    best_d = d;
                    best_i = i;
                }
            }
            // If the bucket is empty (shouldn't happen), fallback to global search.
            if s == e {
                nearest_linear(p, &centroids).0
            } else {
                best_i
            }
        } else {
            nearest_linear(p, &centroids).0
        };

        output.write_all(&(chosen as u32).to_le_bytes())?;

        // Update chosen centroid (online mean).
        let ccount = counts[chosen].saturating_add(1);
        counts[chosen] = ccount;
        let lr = 1.0f32 / (ccount as f32);
        let c = &mut centroids[chosen];
        c[0] += (p[0] - c[0]) * lr;
        c[1] += (p[1] - c[1]) * lr;
        c[2] += (p[2] - c[2]) * lr;

        if coarse > 0 {
            // Update the coarse centroid corresponding to this fine centroid.
            let cj = fine_to_coarse[chosen];
            let ccnt = coarse_counts[cj].saturating_add(1);
            coarse_counts[cj] = ccnt;
            let lr2 = 1.0f32 / (ccnt as f32);
            let cc = &mut coarse_centroids[cj];
            cc[0] += (p[0] - cc[0]) * lr2;
            cc[1] += (p[1] - cc[1]) * lr2;
            cc[2] += (p[2] - cc[2]) * lr2;
        }

        points_read += 1;
    }
    // If we ran out of input bytes early, do nothing further.
    Ok(())
}

fn kmeans_in_memory(
    points: &[[f32; 3]],
    k: usize,
    max_iters: usize,
) -> (Vec<[f32; 3]>, Vec<u32>) {
    let n = points.len();
    let k = k.max(1);
    let mut centroids = vec![[0.0f32; 3]; k];
    let mut labels = vec![0u32; n];

    // Deterministic-ish init: pick points evenly spaced through the array.
    for j in 0..k {
        let idx = ((j as u128) * (n as u128) / (k as u128)).min((n - 1) as u128) as usize;
        centroids[j] = points[idx];
    }

    let mut counts = vec![0u64; k];
    let mut sums = vec![[0.0f32; 3]; k];

    for _ in 0..max_iters {
        // Assign
        sums.fill([0.0, 0.0, 0.0]);
        counts.fill(0);
        let mut changed = 0u64;
        for (i, &p) in points.iter().enumerate() {
            let (best, _) = nearest_linear(p, &centroids);
            let prev = labels[i] as usize;
            if prev != best {
                changed += 1;
            }
            labels[i] = best as u32;
            sums[best][0] += p[0];
            sums[best][1] += p[1];
            sums[best][2] += p[2];
            counts[best] += 1;
        }
        // Update centroids, handle empty clusters by reusing a "far" point.
        for j in 0..k {
            if counts[j] > 0 {
                let inv = 1.0f32 / (counts[j] as f32);
                centroids[j] = [sums[j][0] * inv, sums[j][1] * inv, sums[j][2] * inv];
            }
        }
        if changed == 0 {
            break;
        }
    }
    (centroids, labels)
}

fn main() -> io::Result<()> {
    // Read all stdin into memory for simplicity and speed of parsing.
    // Adaptive behavior:
    // - For smaller problems where O(n*k) is feasible, do a few Lloyd iterations in-memory.
    // - Otherwise, fall back to single-pass online clustering and stream labels out.
    let mut stdin_bytes = Vec::new();
    io::stdin().read_to_end(&mut stdin_bytes)?;
    let mut idx = 0usize;

    let num_points = read_u64_ascii(&stdin_bytes, &mut idx);
    let num_clusters_u64 = read_u64_ascii(&stdin_bytes, &mut idx);
    let k = (num_clusters_u64 as usize).max(1);

    // Skip any trailing whitespace before binary block.
    while idx < stdin_bytes.len() && stdin_bytes[idx].is_ascii_whitespace() {
        idx += 1;
    }

    let bytes_left = stdin_bytes.len().saturating_sub(idx);
    let expected = (num_points as u128) * 12u128;
    let available_points = min(num_points as usize, (bytes_left / 12) as usize);

    let mut out = io::BufWriter::new(io::stdout().lock());

    // Heuristics: prefer in-memory Lloyd only when both memory and compute are reasonable.
    // - Points stored as 12 bytes; plus labels and overhead.
    // - Compute roughly n*k distance checks per iter.
    let can_store_all = expected <= (2u128 * 1024u128 * 1024u128 * 1024u128); // ~2 GiB
    let n = available_points as u64;
    let complexity = (n as f64) * (k as f64);
    let lloyd_ok = can_store_all && complexity <= 2.0e8 && n <= 15_000_000;

    if lloyd_ok {
        let mut points = Vec::with_capacity(available_points);
        for _ in 0..available_points {
            let x = read_f32_le(&stdin_bytes, &mut idx);
            let y = read_f32_le(&stdin_bytes, &mut idx);
            let z = read_f32_le(&stdin_bytes, &mut idx);
            points.push([x, y, z]);
        }
        let iters = if k <= 32 { 20 } else if k <= 256 { 15 } else { 10 };
        let (_centroids, labels) = kmeans_in_memory(&points, k, iters);
        for lab in labels {
            out.write_all(&lab.to_le_bytes())?;
        }
        // If header num_points > available_points (truncated input), don't emit more.
        out.flush()?;
        return Ok(());
    }

    // Streaming/online mode: emit labels as we parse points.
    online_streaming_cluster(&stdin_bytes[idx..], num_points, k, &mut out)?;
    out.flush()?;
    Ok(())
}
