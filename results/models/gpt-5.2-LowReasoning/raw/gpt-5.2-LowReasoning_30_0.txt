import sys


def main() -> None:
    data = sys.stdin.buffer.readline().split()
    if not data:
        return
    n = int(data[0])

    out = sys.stdout.write
    out("SAT\n")
    if n <= 0:
        out("\n")
        return

    # Emit a simple all-True assignment (1..N => True).
    # Write in chunks to avoid huge intermediate strings.
    k = 100000
    full_chunks, rem = divmod(n - 1, k)
    chunk = ("1 " * k)
    for _ in range(full_chunks):
        out(chunk)
    if rem:
        out("1 " * rem)
    out("1\n")


if __name__ == "__main__":
    main()
