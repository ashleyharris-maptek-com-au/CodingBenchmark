import sys


def solve_exact_cover(u, sets_list):
    # Build incidence: for each element, which set indices contain it
    elem_to_sets = [[] for _ in range(u)]
    for idx, st in enumerate(sets_list):
        for e in st:
            if 0 <= e < u:
                elem_to_sets[e].append(idx)
            else:
                return None  # invalid input element id

    # Quick feasibility: every element must appear somewhere
    for e in range(u):
        if not elem_to_sets[e]:
            return None

    # Precompute each set's elements as a tuple for iteration
    set_elems = [tuple(sorted(st)) for st in sets_list]

    # State
    uncovered = set(range(u))
    used_sets = set()
    solution = []

    # For pruning: for each element, current available set indices (excluding used)
    # We'll compute on the fly from elem_to_sets and used_sets.

    def choose_next_element():
        # Choose uncovered element with smallest number of available sets
        best_e = None
        best_count = None
        for e in uncovered:
            cnt = 0
            for si in elem_to_sets[e]:
                if si in used_sets:
                    continue
                # ensure the set doesn't include any already-covered element (i.e., elements not in uncovered)
                ok = True
                for x in set_elems[si]:
                    if x not in uncovered:
                        ok = False
                        break
                if ok:
                    cnt += 1
            if best_count is None or cnt < best_count:
                best_count = cnt
                best_e = e
                if best_count == 0:
                    break
        return best_e, best_count

    def dfs():
        if not uncovered:
            return True

        e, cnt = choose_next_element()
        if cnt == 0:
            return False

        # Try candidate sets in increasing index order for determinism
        candidates = []
        for si in elem_to_sets[e]:
            if si in used_sets:
                continue
            ok = True
            for x in set_elems[si]:
                if x not in uncovered:
                    ok = False
                    break
            if ok:
                candidates.append(si)
        candidates.sort()

        for si in candidates:
            used_sets.add(si)
            removed = []
            for x in set_elems[si]:
                if x in uncovered:
                    uncovered.remove(x)
                    removed.append(x)
            solution.append(si)

            if dfs():
                return True

            solution.pop()
            for x in removed:
                uncovered.add(x)
            used_sets.remove(si)
        return False

    if dfs():
        return sorted(solution)
    return None


def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    first = data[0].strip().split()
    if len(first) < 2:
        return
    try:
        u = int(first[0])
        s = int(first[1])
    except ValueError:
        return

    sets_list = []
    for i in range(1, 1 + s):
        if i >= len(data):
            line = ""
        else:
            line = data[i].strip()
        if not line:
            sets_list.append(set())
            continue
        try:
            elems = list(map(int, line.split()))
        except ValueError:
            elems = []
        sets_list.append(set(elems))

    sol = solve_exact_cover(u, sets_list)
    out = []
    if sol is None:
        out.append("NO SOLUTION")
    else:
        out.append("SOLUTION")
        if sol:
            out.append(" ".join(map(str, sol)))
        else:
            out.append("")
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
