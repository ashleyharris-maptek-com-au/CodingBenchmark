using System;
using System.Diagnostics;
using System.IO;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32 && c >= 0)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    private struct XorShift64
    {
        private ulong _x;
        public XorShift64(ulong seed) { _x = seed == 0 ? 88172645463325252UL : seed; }
        public ulong NextU64()
        {
            ulong x = _x;
            x ^= x << 7;
            x ^= x >> 9;
            _x = x;
            return x;
        }
        public int NextInt(int n)
        {
            return (int)(NextU64() % (ulong)n);
        }
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n;
        try
        {
            n = scanner.NextInt();
        }
        catch
        {
            return;
        }

        int[,] flow = new int[n, n];
        int[,] dist = new int[n, n];

        long[] flowSum = new long[n];
        long[] distSum = new long[n];

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                int v = scanner.NextInt();
                flow[i, j] = v;
                flowSum[i] += v;
            }
        }
        // Add column sums too (often helps); keep it symmetric-ish without assuming symmetry.
        for (int j = 0; j < n; j++)
            for (int i = 0; i < n; i++)
                flowSum[j] += flow[i, j];

        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                int v = scanner.NextInt();
                dist[i, j] = v;
                distSum[i] += v;
            }
        }
        for (int j = 0; j < n; j++)
            for (int i = 0; i < n; i++)
                distSum[j] += dist[i, j];

        int[] facilities = Enumerable.Range(0, n).ToArray();
        int[] locations = Enumerable.Range(0, n).ToArray();
        Array.Sort(facilities, (a, b) => flowSum[b].CompareTo(flowSum[a]));  // descending
        Array.Sort(locations, (a, b) => distSum[a].CompareTo(distSum[b]));   // ascending

        int[] p = new int[n];     // facility -> location
        int[] inv = new int[n];   // location -> facility
        for (int idx = 0; idx < n; idx++)
        {
            int f = facilities[idx];
            int loc = locations[idx];
            p[f] = loc;
            inv[loc] = f;
        }

        long cost = ComputeCost(flow, dist, p, n);

        // Local search with random sampled 2-swaps and O(n) delta computation.
        // Keep a strict time budget; for benchmarks usually enough.
        var sw = Stopwatch.StartNew();
        long timeLimitMs = 295_000; // ~4m55s safety margin for a 5 min limit

        ulong seed = (ulong)DateTime.UtcNow.Ticks ^ (ulong)n * 0x9E3779B97F4A7C15UL;
        var rng = new XorShift64(seed);

        int noImproveRounds = 0;
        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            bool improved = false;

            // First-improvement hillclimb by random swaps
            int tries = n < 100 ? 20000 : (n < 500 ? 30000 : 50000);
            for (int t = 0; t < tries && sw.ElapsedMilliseconds < timeLimitMs; t++)
            {
                int a = rng.NextInt(n);
                int b = rng.NextInt(n - 1);
                if (b >= a) b++;

                long delta = DeltaSwap(flow, dist, p, n, a, b);
                if (delta < 0)
                {
                    ApplySwap(p, inv, a, b);
                    cost += delta;
                    improved = true;
                }
            }

            // Occasional deeper sampling: pick a and scan a small neighborhood for best b
            int probes = n < 200 ? 200 : 400;
            for (int t = 0; t < probes && sw.ElapsedMilliseconds < timeLimitMs; t++)
            {
                int a = rng.NextInt(n);
                long bestDelta = 0;
                int bestB = -1;
                int samples = n < 200 ? Math.Min(n - 1, 40) : 25;
                for (int s = 0; s < samples; s++)
                {
                    int b = rng.NextInt(n - 1);
                    if (b >= a) b++;
                    long delta = DeltaSwap(flow, dist, p, n, a, b);
                    if (delta < bestDelta)
                    {
                        bestDelta = delta;
                        bestB = b;
                    }
                }
                if (bestB >= 0)
                {
                    ApplySwap(p, inv, a, bestB);
                    cost += bestDelta;
                    improved = true;
                }
            }

            if (!improved)
            {
                noImproveRounds++;
                // Mild perturbation if stuck: random shuffles of a few locations.
                if (noImproveRounds >= 3 && sw.ElapsedMilliseconds < timeLimitMs)
                {
                    int k = Math.Min(12, n);
                    int[] idxs = new int[k];
                    for (int i = 0; i < k; i++) idxs[i] = rng.NextInt(n);
                    for (int i = 0; i < k; i++)
                    {
                        int j = rng.NextInt(k);
                        int a = idxs[i];
                        int b = idxs[j];
                        if (a == b) continue;
                        ApplySwap(p, inv, a, b);
                    }
                    cost = ComputeCost(flow, dist, p, n);
                    noImproveRounds = 0;
                }
            }
            else
            {
                noImproveRounds = 0;
            }
        }

        // Final output
        var output = new StringWriter();
        output.WriteLine(cost);
        for (int i = 0; i < n; i++)
        {
            if (i > 0) output.Write(' ');
            output.Write(p[i]);
        }
        output.WriteLine();
        Console.Write(output.ToString());
    }

    private static void ApplySwap(int[] p, int[] inv, int a, int b)
    {
        int la = p[a];
        int lb = p[b];
        p[a] = lb;
        p[b] = la;
        inv[la] = b;
        inv[lb] = a;
    }

    private static long ComputeCost(int[,] flow, int[,] dist, int[] p, int n)
    {
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int li = p[i];
            for (int j = 0; j < n; j++)
            {
                total += (long)flow[i, j] * dist[li, p[j]];
            }
        }
        return total;
    }

    // Delta in total cost if we swap the locations assigned to facilities a and b.
    // O(n) implementation that works for non-symmetric flows/distances.
    private static long DeltaSwap(int[,] flow, int[,] dist, int[] p, int n, int a, int b)
    {
        if (a == b) return 0;
        int la = p[a];
        int lb = p[b];
        long delta = 0;

        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int lk = p[k];

            // Terms where (a,k) and (b,k)
            delta += (long)flow[a, k] * (dist[lb, lk] - dist[la, lk]);
            delta += (long)flow[b, k] * (dist[la, lk] - dist[lb, lk]);

            // Terms where (k,a) and (k,b)
            delta += (long)flow[k, a] * (dist[lk, lb] - dist[lk, la]);
            delta += (long)flow[k, b] * (dist[lk, la] - dist[lk, lb]);
        }

        // Terms involving (a,b) and (b,a) and self terms
        delta += (long)flow[a, a] * (dist[lb, lb] - dist[la, la]);
        delta += (long)flow[b, b] * (dist[la, la] - dist[lb, lb]);

        delta += (long)flow[a, b] * (dist[lb, la] - dist[la, lb]);
        delta += (long)flow[b, a] * (dist[la, lb] - dist[lb, la]);

        return delta;
    }
}
