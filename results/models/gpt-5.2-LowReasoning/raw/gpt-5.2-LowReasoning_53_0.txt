import math
from typing import Dict


# Module-level memory persists across calls within a scenario.
_mem = {
    "init": False,
    "pos": [0.0, 0.0, 0.0],
    "vel": [0.0, 0.0, 0.0],
    "have_pos": False,
    "have_vel": False,
    "last_range": None,
    "t": 0.0,
    "uncommanded_accel_bias": [0.0, 0.0, 0.0],
}


def _clamp(x: float, lo: float, hi: float) -> float:
    return lo if x < lo else hi if x > hi else x


def _norm(v) -> float:
    return math.sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2])


def _unit(v):
    n = _norm(v)
    if n <= 1e-9:
        return [0.0, 0.0, 0.0]
    return [v[0] / n, v[1] / n, v[2] / n]


def _median(vals):
    vals = sorted(vals)
    if not vals:
        return None
    m = len(vals) // 2
    if len(vals) % 2 == 1:
        return float(vals[m])
    return 0.5 * (float(vals[m - 1]) + float(vals[m]))


def _get_vec(state: Dict, prefix: str):
    # Returns (vec, vec_valid_mask[3], vel, vel_valid_mask[3]) or (None, ...)
    pos = [state.get(f"{prefix}_rel_pos_x_m", 0.0), state.get(f"{prefix}_rel_pos_y_m", 0.0), state.get(f"{prefix}_rel_pos_z_m", 0.0)]
    pos_valid = [
        bool(state.get(f"{prefix}_rel_x_valid", False)),
        bool(state.get(f"{prefix}_rel_y_valid", False)),
        bool(state.get(f"{prefix}_rel_z_valid", False)),
    ]
    vel = [state.get(f"{prefix}_rel_vel_x_ms", 0.0), state.get(f"{prefix}_rel_vel_y_ms", 0.0), state.get(f"{prefix}_rel_vel_z_ms", 0.0)]
    vel_valid = [
        bool(state.get(f"{prefix}_relv_x_valid", False)),
        bool(state.get(f"{prefix}_relv_y_valid", False)),
        bool(state.get(f"{prefix}_relv_z_valid", False)),
    ]
    if not any(pos_valid) and not any(vel_valid):
        return None, pos_valid, None, vel_valid
    return pos, pos_valid, vel, vel_valid


def _fuse_pos_vel(state: Dict, dt: float):
    # Prefer lidar for proximity, radar for longer range; fuse when both available.
    lidar_pos, lidar_pv, lidar_vel, lidar_vv = _get_vec(state, "lidar")
    radar_pos, radar_pv, radar_vel, radar_vv = _get_vec(state, "radar")

    # Determine rough range from any available vector.
    candidate_ranges = []
    if lidar_pos is not None and all(lidar_pv):
        candidate_ranges.append(_norm(lidar_pos))
    if radar_pos is not None and all(radar_pv):
        candidate_ranges.append(_norm(radar_pos))

    use_lidar = False
    use_radar = False
    if candidate_ranges:
        r_est = min(candidate_ranges)
        use_lidar = (lidar_pos is not None) and (r_est <= 350.0)
        use_radar = (radar_pos is not None) and not use_lidar
    else:
        use_lidar = lidar_pos is not None
        use_radar = radar_pos is not None and not use_lidar

    # Position fusion per-axis with availability.
    pos_meas = [None, None, None]
    vel_meas = [None, None, None]

    for i in range(3):
        vals = []
        if use_lidar and lidar_pos is not None and lidar_pv[i]:
            vals.append(float(lidar_pos[i]))
        if use_radar and radar_pos is not None and radar_pv[i]:
            vals.append(float(radar_pos[i]))
        if (lidar_pos is not None and radar_pos is not None and lidar_pv[i] and radar_pv[i]) and not vals:
            vals = [float(lidar_pos[i]), float(radar_pos[i])]
        if vals:
            pos_meas[i] = sum(vals) / len(vals)

        vvals = []
        if use_lidar and lidar_vel is not None and lidar_vv[i]:
            vvals.append(float(lidar_vel[i]))
        if use_radar and radar_vel is not None and radar_vv[i]:
            vvals.append(float(radar_vel[i]))
        if (lidar_vel is not None and radar_vel is not None and lidar_vv[i] and radar_vv[i]) and not vvals:
            vvals = [float(lidar_vel[i]), float(radar_vel[i])]
        if vvals:
            vel_meas[i] = sum(vvals) / len(vvals)

    # Range finders (scalar) for sanity + fallback.
    rf = []
    for k in (1, 2, 3):
        if state.get(f"range_{k}_valid", False):
            rf.append(float(state.get(f"range_{k}_m", 0.0)))
    range_meas = _median(rf)

    # Dock camera bearing + range: fallback for position direction.
    cam_ok = bool(state.get("dock_camera_active", False))
    if cam_ok:
        bx_ok = bool(state.get("dock_camera_bearing_x_valid", False))
        by_ok = bool(state.get("dock_camera_bearing_y_valid", False))
        bz_ok = bool(state.get("dock_camera_bearing_z_valid", False))
        if bx_ok and by_ok and bz_ok:
            bearing = [
                float(state.get("dock_camera_bearing_x", 0.0)),
                float(state.get("dock_camera_bearing_y", 0.0)),
                float(state.get("dock_camera_bearing_z", 0.0)),
            ]
            cam_range = state.get("dock_camera_range_m", None)
            if cam_range is not None:
                range_meas = float(cam_range)
            if range_meas is not None and range_meas > 0:
                u = _unit(bearing)
                for i in range(3):
                    if pos_meas[i] is None:
                        pos_meas[i] = u[i] * range_meas

    # If we only have scalar range, maintain direction from last estimate.
    if range_meas is not None and any(p is None for p in pos_meas):
        if _mem["have_pos"]:
            u = _unit(_mem["pos"])
            for i in range(3):
                if pos_meas[i] is None:
                    pos_meas[i] = u[i] * range_meas

    # Prediction step.
    if not _mem["init"]:
        _mem["init"] = True
        _mem["t"] = 0.0

    if _mem["have_pos"] and _mem["have_vel"]:
        _mem["pos"] = [_mem["pos"][i] + _mem["vel"][i] * dt for i in range(3)]

    # Alpha-beta-ish update (robust to missing axes).
    alpha_pos = 0.55
    alpha_vel = 0.45
    for i in range(3):
        if pos_meas[i] is not None:
            if not _mem["have_pos"]:
                _mem["pos"][i] = float(pos_meas[i])
            else:
                _mem["pos"][i] = (1.0 - alpha_pos) * _mem["pos"][i] + alpha_pos * float(pos_meas[i])
            _mem["have_pos"] = True
        if vel_meas[i] is not None:
            if not _mem["have_vel"]:
                _mem["vel"][i] = float(vel_meas[i])
            else:
                _mem["vel"][i] = (1.0 - alpha_vel) * _mem["vel"][i] + alpha_vel * float(vel_meas[i])
            _mem["have_vel"] = True

    # If velocity missing but position changes, infer (very gently).
    if _mem["have_pos"] and not _mem["have_vel"] and dt > 1e-3:
        _mem["vel"] = [0.0, 0.0, 0.0]
        _mem["have_vel"] = True

    # Keep a stable range estimate for phase logic.
    if _mem["have_pos"]:
        _mem["last_range"] = _norm(_mem["pos"])
    elif range_meas is not None:
        _mem["last_range"] = float(range_meas)


def _gyro_avg(state: Dict):
    # Average valid gyro rates (deg/s -> rad/s).
    axes = {"roll": [], "pitch": [], "yaw": []}
    for i in (1, 2, 3):
        if state.get(f"gyro_{i}_roll_valid", False):
            axes["roll"].append(float(state.get(f"gyro_{i}_roll_dps", 0.0)) * math.pi / 180.0)
        if state.get(f"gyro_{i}_pitch_valid", False):
            axes["pitch"].append(float(state.get(f"gyro_{i}_pitch_dps", 0.0)) * math.pi / 180.0)
        if state.get(f"gyro_{i}_yaw_valid", False):
            axes["yaw"].append(float(state.get(f"gyro_{i}_yaw_dps", 0.0)) * math.pi / 180.0)

    def avg(lst):
        return sum(lst) / len(lst) if lst else 0.0

    return avg(axes["roll"]), avg(axes["pitch"]), avg(axes["yaw"])


def autopilot_step(state: dict, dt: float) -> dict:
    dt = float(dt) if dt and dt > 0 else 0.05
    dt = _clamp(dt, 0.01, 0.5)

    _fuse_pos_vel(state, dt)

    # If navigation is completely unavailable, don't waste fuel.
    if not _mem["have_pos"]:
        gr, gp, gy = _gyro_avg(state)
        return {
            "thrust_x": 0.0,
            "thrust_y": 0.0,
            "thrust_z": 0.0,
            "att_rate_roll": _clamp(-0.8 * gr, -0.035, 0.035),
            "att_rate_pitch": _clamp(-0.8 * gp, -0.035, 0.035),
            "att_rate_yaw": _clamp(-0.8 * gy, -0.035, 0.035),
        }

    pos = _mem["pos"]
    vel = _mem["vel"] if _mem["have_vel"] else [0.0, 0.0, 0.0]
    rng = float(_mem["last_range"] if _mem["last_range"] is not None else _norm(pos))
    speed = _norm(vel)

    # Fuel-aware thrust scaling (keep some control authority even when low).
    fuel_kg = float(state.get("fuel_remaining_kg", 500.0) or 500.0)
    fuel_pct = float(state.get("fuel_remaining_pct", 100.0) or 100.0)
    fuel_score = _clamp(min(fuel_kg / 500.0, fuel_pct / 100.0), 0.05, 1.0)
    max_cmd = 1.0 if fuel_score > 0.2 else (0.35 + 0.65 * fuel_score / 0.2)
    max_cmd = _clamp(max_cmd, 0.2, 1.0)

    x, y, z = pos  # x radial, y along-track, z cross-track
    vx, vy, vz = vel

    # Phase targets.
    # - Far: correct along-track using counter-intuitive orbital mechanics rule.
    # - Medium: converge to a stable "V-bar hold" point behind the target.
    # - Close: linear regime PD; final: very gentle approach.
    thrust_x = thrust_y = thrust_z = 0.0

    if rng > 200.0:
        # Far rendezvous: primarily manage y via retro/prograde.
        # If behind (y<0): retrograde (negative y thrust) to lower orbit -> speed up -> catch up.
        # If ahead (y>0): prograde to raise orbit -> slow down -> let target catch up.
        y_scale = _clamp(abs(y) / 1500.0, 0.0, 1.0)
        # Reduce if already moving in the helpful direction.
        helpful = (-1.0 if y < 0 else 1.0)  # desired sign of thrust_y
        vy_help = helpful * vy  # positive means we're already doing the right closing trend
        vy_brake = _clamp(vy_help / 1.0, -1.0, 1.0)
        thrust_y = helpful * (0.6 * y_scale) - 0.25 * vy_brake

        # Keep x,z bounded to avoid large flyaround energy.
        thrust_x = _clamp(-0.08 * (x / 200.0) - 0.18 * (vx / 0.5), -0.25, 0.25)
        thrust_z = _clamp(-0.10 * (z / 200.0) - 0.20 * (vz / 0.5), -0.25, 0.25)

    elif rng > 50.0:
        # Medium: converge to a point slightly behind the station (V-bar approach).
        # Target a gentle in-track offset so we don't "ram" the docking port.
        y_target = -20.0  # meters behind
        x_target = 0.0
        z_target = 0.0

        ex, ey, ez = (x - x_target), (y - y_target), (z - z_target)

        thrust_y = _clamp(-0.06 * (ey / 50.0) - 0.22 * (vy / 0.5), -0.55, 0.55)
        thrust_x = _clamp(-0.08 * (ex / 50.0) - 0.28 * (vx / 0.5), -0.45, 0.45)
        thrust_z = _clamp(-0.10 * (ez / 50.0) - 0.30 * (vz / 0.5), -0.45, 0.45)

        # If closing too fast overall, bias to braking along the current approach direction.
        if speed > 1.0:
            u = _unit(vel)
            thrust_x += -0.15 * u[0]
            thrust_y += -0.15 * u[1]
            thrust_z += -0.15 * u[2]

    else:
        # Close: direct linear control.
        # Aim to null x,z and close y to 0 with a controlled approach speed.
        # Enforce a speed schedule vs range.
        if rng > 10.0:
            v_limit = 0.25
        elif rng > 2.0:
            v_limit = 0.10
        else:
            v_limit = 0.06

        # Desired velocity towards target with cap.
        to_target = [-x, -y, -z]
        u = _unit(to_target)
        v_des = [u[i] * v_limit for i in range(3)]

        # PD on velocity error plus mild position term for damping.
        ax_cmd = 0.55 * (v_des[0] - vx) + (-0.12 * x)
        ay_cmd = 0.55 * (v_des[1] - vy) + (-0.12 * y)
        az_cmd = 0.60 * (v_des[2] - vz) + (-0.12 * z)

        # Convert "desired accel" to normalized thrust command.
        # We don't know mass; treat this as heuristic scaling with saturation.
        thrust_x = _clamp(ax_cmd / 0.6, -0.6, 0.6)
        thrust_y = _clamp(ay_cmd / 0.6, -0.6, 0.6)
        thrust_z = _clamp(az_cmd / 0.6, -0.6, 0.6)

        # Final safety: if within 10 m and still fast, prioritize braking (anti-velocity).
        if rng < 10.0 and speed > v_limit * 1.4:
            uvel = _unit(vel)
            thrust_x += -0.35 * uvel[0]
            thrust_y += -0.35 * uvel[1]
            thrust_z += -0.35 * uvel[2]

    # Apply fuel scaling and final clamps.
    thrust_x = _clamp(thrust_x, -max_cmd, max_cmd)
    thrust_y = _clamp(thrust_y, -max_cmd, max_cmd)
    thrust_z = _clamp(thrust_z, -max_cmd, max_cmd)

    # Very small deadband to avoid chatter near equilibrium.
    def deadband(v, eps):
        return 0.0 if abs(v) < eps else v

    thrust_x = deadband(thrust_x, 0.02)
    thrust_y = deadband(thrust_y, 0.02)
    thrust_z = deadband(thrust_z, 0.02)

    # Attitude: damp body rates to ~0 (good enough unless a pointing law is required).
    gr, gp, gy = _gyro_avg(state)
    att_roll = _clamp(-0.9 * gr, -0.035, 0.035)
    att_pitch = _clamp(-0.9 * gp, -0.035, 0.035)
    att_yaw = _clamp(-0.9 * gy, -0.035, 0.035)

    return {
        "thrust_x": float(thrust_x),
        "thrust_y": float(thrust_y),
        "thrust_z": float(thrust_z),
        "att_rate_roll": float(att_roll),
        "att_rate_pitch": float(att_pitch),
        "att_rate_yaw": float(att_yaw),
    }
