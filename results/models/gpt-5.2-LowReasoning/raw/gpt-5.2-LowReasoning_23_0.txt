use std::f64::consts::PI;
use std::io::{self, Read, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct Params {
    gravity: f64,
    max_thrust: f64,
}

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
    has_state: bool,
}

#[derive(Clone, Copy, Debug, Default)]
struct MapInfo {
    target_x: f64,
    target_y: f64,
}

struct FastInput {
    buf: Vec<u8>,
    i: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        // Read everything available; in a streaming judge this will grow as more arrives.
        // If stdin is truly infinite, this will block forever; so we do not use it.
        // Instead, we will read incrementally with Read::read in next_token().
        buf.reserve(1 << 16);
        Self { buf, i: 0 }
    }

    fn refill(&mut self, stdin: &mut dyn Read) -> io::Result<usize> {
        if self.i > 0 {
            self.buf.drain(..self.i);
            self.i = 0;
        }
        let mut tmp = [0u8; 1 << 16];
        let n = stdin.read(&mut tmp)?;
        if n > 0 {
            self.buf.extend_from_slice(&tmp[..n]);
        }
        Ok(n)
    }

    fn next_token(&mut self, stdin: &mut dyn Read) -> io::Result<Option<Vec<u8>>> {
        loop {
            while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
                self.i += 1;
            }
            if self.i < self.buf.len() {
                let start = self.i;
                while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
                    self.i += 1;
                }
                return Ok(Some(self.buf[start..self.i].to_vec()));
            }
            let n = self.refill(stdin)?;
            if n == 0 {
                return Ok(None);
            }
        }
    }

    fn next_f64(&mut self, stdin: &mut dyn Read) -> io::Result<Option<f64>> {
        let tok = match self.next_token(stdin)? {
            Some(t) => t,
            None => return Ok(None),
        };
        let s = std::str::from_utf8(&tok).unwrap_or("0");
        Ok(Some(s.parse::<f64>().unwrap_or(0.0)))
    }

    fn next_usize(&mut self, stdin: &mut dyn Read) -> io::Result<Option<usize>> {
        let tok = match self.next_token(stdin)? {
            Some(t) => t,
            None => return Ok(None),
        };
        let s = std::str::from_utf8(&tok).unwrap_or("0");
        Ok(Some(s.parse::<usize>().unwrap_or(0)))
    }
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    v.max(lo).min(hi)
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

fn control(params: Params, map: MapInfo, s: State) -> (f64, f64) {
    if !s.has_state {
        return (0.0, 0.0);
    }

    // Guidance: PD towards target with a gentle approach. Keep tilt small near touchdown.
    let dx = map.target_x - s.x;

    let altitude = map.target_y - s.y; // positive when above target (since y increases downward)
    let near_ground = altitude < 15.0;

    // Horizontal desired acceleration (m/s^2)
    let kp_x = if near_ground { 0.10 } else { 0.20 };
    let kd_x = if near_ground { 0.35 } else { 0.45 };
    let mut ax_des = kp_x * dx + kd_x * (0.0 - s.vx);
    ax_des = clamp(ax_des, -4.0, 4.0);

    // Limit tilt more when close to ground.
    let max_tilt = if near_ground { 0.35 } else { 0.75 };
    // Small-angle mapping: sin(angle) ~ angle, so angle ~ ax/max_thrust.
    let angle_des = clamp(ax_des / params.max_thrust, -max_tilt, max_tilt);

    // Vertical speed profile (vy positive downward).
    // If far above, allow faster descent; near ground, slow down.
    let vy_target = if altitude > 80.0 {
        10.0
    } else if altitude > 30.0 {
        6.0
    } else if altitude > 10.0 {
        3.0
    } else {
        1.0
    };

    let vy_err = s.vy - vy_target;
    let kp_vy = if near_ground { 0.22 } else { 0.14 };

    // Required upward accel from engine to correct vertical motion, plus cancel gravity.
    // In y-down coordinates: ay_total = gravity - thrust*max_thrust*cos(angle)
    // We want ay_total ≈ -kp_vy * vy_err (negative if falling too fast).
    let ay_total_des = -kp_vy * vy_err;
    let up_accel_needed = params.gravity - ay_total_des; // thrust*max_thrust*cos(angle)

    let cos_a = angle_des.cos().max(0.15);
    let mut thrust = up_accel_needed / (params.max_thrust * cos_a);

    // Extra braking if very close and still moving down fast.
    if altitude < 3.0 && s.vy > 2.0 {
        thrust = 1.0;
    }

    // If out of fuel, nothing we can do.
    if s.fuel_remaining <= 0.0 {
        thrust = 0.0;
    }

    thrust = clamp(thrust, 0.0, 1.0);

    // Attitude control: PD on angle.
    let err = wrap_pi(angle_des - s.angle);
    let kp_ang = 3.2;
    let kd_ang = 0.9;
    let turn = clamp(kp_ang * err - kd_ang * s.angular_vel, -1.0, 1.0);

    (thrust, turn)
}

fn main() -> io::Result<()> {
    let params = Arc::new(Mutex::new(Params::default()));
    let map = Arc::new(Mutex::new(MapInfo::default()));
    let state = Arc::new(Mutex::new(State::default()));

    // Reader thread: parse initial map then streaming state updates.
    {
        let params = Arc::clone(&params);
        let map = Arc::clone(&map);
        let state = Arc::clone(&state);
        thread::spawn(move || {
            let mut stdin = io::stdin().lock();
            let mut inp = FastInput::new();

            // Initial header
            let width = match inp.next_usize(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let height = match inp.next_usize(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let gravity = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let max_thrust = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            // fuel (unused directly here; we use fuel_remaining from state updates)
            let _fuel = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };

            // start/target positions
            let _start_x = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let _start_y = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let target_x = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };
            let target_y = match inp.next_f64(&mut stdin) {
                Ok(Some(v)) => v,
                _ => return,
            };

            {
                let mut p = params.lock().unwrap();
                p.gravity = gravity;
                p.max_thrust = max_thrust.max(1e-6);
            }
            {
                let mut m = map.lock().unwrap();
                m.target_x = target_x;
                m.target_y = target_y;
            }

            // Consume map rows without storing (could be huge).
            // The format says "map rows" with characters; we treat each row as a token.
            for _ in 0..height {
                let _ = inp.next_token(&mut stdin);
            }

            // Streaming updates: x y vx vy angle angular_vel fuel_remaining
            loop {
                let x = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };
                let y = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };
                let vx = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };
                let vy = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };
                let angle = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };
                let angular_vel = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };
                let fuel_remaining = match inp.next_f64(&mut stdin) {
                    Ok(Some(v)) => v,
                    _ => return,
                };

                let mut s = state.lock().unwrap();
                s.x = x;
                s.y = y;
                s.vx = vx;
                s.vy = vy;
                s.angle = angle;
                s.angular_vel = angular_vel;
                s.fuel_remaining = fuel_remaining;
                s.has_state = true;
            }
        });
    }

    // Control loop: 20 Hz outputs.
    let mut stdout = io::BufWriter::new(io::stdout().lock());
    let dt = Duration::from_millis(50);
    let mut next = Instant::now();

    loop {
        let p = *params.lock().unwrap();
        let m = *map.lock().unwrap();
        let s = *state.lock().unwrap();
        let (thrust, turn) = control(p, m, s);
        let _ = writeln!(stdout, "{:.6} {:.6}", thrust, turn);
        let _ = stdout.flush();

        next += dt;
        let now = Instant::now();
        if next > now {
            thread::sleep(next - now);
        } else {
            next = now;
        }
    }
}
