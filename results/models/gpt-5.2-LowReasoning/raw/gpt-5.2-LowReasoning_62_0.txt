declare const require: any;
const fs = require('fs');

type Town = { id: any; x: number; y: number; value: number };

function main() {
  const input: string = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);

  const width: number = data.width | 0;
  const height: number = data.height | 0;
  const grid: string[] = data.grid;
  const ignitions: number[][] = data.ignitions || [];
  const towns: Town[] = data.towns || [];
  const cutBudget: number = Math.max(0, data.cutBudget | 0);

  const n = width * height;
  const burnable = new Uint8Array(n);
  for (let y = 0; y < height; y++) {
    const row = grid[y] || '';
    for (let x = 0; x < width; x++) {
      const ch = row.charCodeAt(x);
      // '.' (46) is burnable. Anything else treated as non-burnable blocker.
      burnable[y * width + x] = ch === 46 ? 1 : 0;
    }
  }

  const ignitionIdx = new Uint8Array(n);
  for (let i = 0; i < ignitions.length; i++) {
    const x = ignitions[i][0] | 0;
    const y = ignitions[i][1] | 0;
    if (x >= 0 && x < width && y >= 0 && y < height) ignitionIdx[y * width + x] = 1;
  }

  const townIdx = new Uint8Array(n);
  for (let i = 0; i < towns.length; i++) {
    const x = towns[i].x | 0;
    const y = towns[i].y | 0;
    if (x >= 0 && x < width && y >= 0 && y < height) townIdx[y * width + x] = 1;
  }

  // Baseline BFS (no cuts) from ignitions over burnable cells.
  const dist = new Int32Array(n);
  dist.fill(-1);
  const prev = new Int32Array(n);
  prev.fill(-1);

  const q = new Int32Array(n);
  let qh = 0,
    qt = 0;
  for (let i = 0; i < ignitions.length; i++) {
    const x = ignitions[i][0] | 0;
    const y = ignitions[i][1] | 0;
    if (x < 0 || x >= width || y < 0 || y >= height) continue;
    const idx = y * width + x;
    if (!burnable[idx]) continue;
    if (dist[idx] !== -1) continue;
    dist[idx] = 0;
    prev[idx] = -1;
    q[qt++] = idx;
  }

  const w = width;
  while (qh < qt) {
    const idx = q[qh++];
    const d = dist[idx] + 1;
    const x = idx % w;
    const y = (idx / w) | 0;

    // neighbor helper inline to reduce overhead
    if (x > 0) {
      const ni = idx - 1;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        prev[ni] = idx;
        q[qt++] = ni;
      }
    }
    if (x + 1 < w) {
      const ni = idx + 1;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        prev[ni] = idx;
        q[qt++] = ni;
      }
    }
    if (y > 0) {
      const ni = idx - w;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        prev[ni] = idx;
        q[qt++] = ni;
      }
    }
    if (y + 1 < height) {
      const ni = idx + w;
      if (burnable[ni] && dist[ni] === -1) {
        dist[ni] = d;
        prev[ni] = idx;
        q[qt++] = ni;
      }
    }
  }

  // Accumulate path-based "importance" weights for cells.
  const weight = new Float64Array(n);
  for (let i = 0; i < towns.length; i++) {
    const tx = towns[i].x | 0;
    const ty = towns[i].y | 0;
    if (tx < 0 || tx >= width || ty < 0 || ty >= height) continue;
    const tidx = ty * width + tx;
    if (!burnable[tidx]) continue; // if town is on blocker, it's safe already
    const dTown = dist[tidx];
    if (dTown < 0) continue; // not threatened

    // Weight higher-value towns; slightly prefer earlier cuts (closer to ignition).
    const base = Math.max(0, +towns[i].value);
    if (base <= 0) continue;
    const invPath = 1 / (1 + dTown);

    let cur = tidx;
    // Avoid cutting town cell itself; avoid ignition cells.
    while (cur !== -1) {
      const p = prev[cur];
      if (p === -1) break; // reached a source
      // cur is one step away from p toward the town; cutting cur blocks this tree-path.
      if (!townIdx[cur] && !ignitionIdx[cur]) {
        // Bias toward closer-to-source cuts: bigger when dist[cur] is smaller.
        const dc = dist[cur];
        const proximity = dc >= 0 ? 1 / (1 + dc) : 0;
        weight[cur] += base * invPath * (0.5 + proximity);
      }
      cur = p;
    }
  }

  // Select top-weight cells up to budget.
  const candidates: number[] = [];
  for (let idx = 0; idx < n; idx++) {
    if (weight[idx] > 0) candidates.push(idx);
  }
  candidates.sort((a, b) => weight[b] - weight[a]);

  const cut = new Uint8Array(n);
  const firebreaks: number[][] = [];
  for (let i = 0; i < candidates.length && firebreaks.length < cutBudget; i++) {
    const idx = candidates[i];
    if (!burnable[idx]) continue;
    if (cut[idx]) continue;
    if (ignitionIdx[idx]) continue;
    if (townIdx[idx]) continue;
    cut[idx] = 1;
    firebreaks.push([idx % width, (idx / width) | 0]);
  }

  process.stdout.write(JSON.stringify({ firebreaks }));
}

main();
