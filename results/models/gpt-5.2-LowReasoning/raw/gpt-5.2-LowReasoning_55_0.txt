// Chess bot for benchmark: reads timeout + FEN, outputs best move found within time.
// Move notation:
// - Pawn: 'p' + from + to, captures with 'x' (e.g. pe4xd5)
// - Other pieces: N/B/R/Q/K + from + to, captures with 'x'
// - Promotion: append "/Q", "/R", "/B", "/N"
// - Castling: "O-O" or "O-O-O"
// - En passant: append "ep"
// - Check: append '+', Checkmate: append '#'
//
// Build: MSVC /std:c++20

using namespace std;

namespace chess {

static constexpr int WHITE = 0;
static constexpr int BLACK = 1;

static inline int file_of(int sq) { return sq & 7; }
static inline int rank_of(int sq) { return sq >> 3; }
static inline int sq_of(int file, int rank) { return (rank << 3) | file; }

static inline string sq_to_str(int sq) {
  string s;
  s.push_back(char('a' + file_of(sq)));
  s.push_back(char('1' + rank_of(sq)));
  return s;
}

static inline int str_to_sq(string_view s) {
  if (s.size() != 2) return -1;
  char f = s[0], r = s[1];
  if (f < 'a' || f > 'h' || r < '1' || r > '8') return -1;
  return sq_of(f - 'a', r - '1');
}

enum Piece : int8_t {
  EMPTY = 0,
  WP = 1, WN = 2, WB = 3, WR = 4, WQ = 5, WK = 6,
  BP = -1, BN = -2, BB = -3, BR = -4, BQ = -5, BK = -6
};

static inline int color_of(Piece p) {
  if (p == EMPTY) return -1;
  return (p > 0) ? WHITE : BLACK;
}
static inline int abs_piece(Piece p) { return p > 0 ? int(p) : int(-p); }

struct Move {
  int from = -1;
  int to = -1;
  Piece promo = EMPTY;     // for promotions: WQ/WR/WB/WN or BQ/...
  bool is_capture = false;
  bool is_ep = false;
  bool is_castle = false;
  bool is_double_push = false;
};

struct Undo {
  Piece captured = EMPTY;
  int ep_square = -1;
  uint8_t castling = 0;
  int halfmove_clock = 0;
  int fullmove_number = 1;
};

// Castling bits: 1=K,2=Q,4=k,8=q
struct Position {
  array<Piece, 64> board{};
  int stm = WHITE;
  uint8_t castling = 0;
  int ep_square = -1; // square behind pawn that just double pushed
  int halfmove_clock = 0;
  int fullmove_number = 1;

  int king_sq(int color) const {
    Piece k = (color == WHITE) ? WK : BK;
    for (int i = 0; i < 64; ++i) if (board[i] == k) return i;
    return -1;
  }
};

static inline char piece_to_fen(Piece p) {
  switch (p) {
    case WP: return 'P'; case WN: return 'N'; case WB: return 'B'; case WR: return 'R'; case WQ: return 'Q'; case WK: return 'K';
    case BP: return 'p'; case BN: return 'n'; case BB: return 'b'; case BR: return 'r'; case BQ: return 'q'; case BK: return 'k';
    default: return '.';
  }
}

static inline Piece fen_to_piece(char c) {
  switch (c) {
    case 'P': return WP; case 'N': return WN; case 'B': return WB; case 'R': return WR; case 'Q': return WQ; case 'K': return WK;
    case 'p': return BP; case 'n': return BN; case 'b': return BB; case 'r': return BR; case 'q': return BQ; case 'k': return BK;
    default: return EMPTY;
  }
}

static bool parse_fen(const string& fen, Position& pos) {
  pos = Position{};
  istringstream iss(fen);
  string board_part, stm_part, castling_part, ep_part;
  if (!(iss >> board_part >> stm_part >> castling_part >> ep_part >> pos.halfmove_clock >> pos.fullmove_number)) {
    return false;
  }

  int rank = 7, file = 0;
  for (char c : board_part) {
    if (c == '/') {
      if (file != 8) return false;
      file = 0;
      --rank;
      continue;
    }
    if (isdigit((unsigned char)c)) {
      int n = c - '0';
      if (n < 1 || n > 8) return false;
      for (int i = 0; i < n; ++i) {
        if (file >= 8 || rank < 0) return false;
        pos.board[sq_of(file, rank)] = EMPTY;
        ++file;
      }
    } else {
      Piece p = fen_to_piece(c);
      if (p == EMPTY) return false;
      if (file >= 8 || rank < 0) return false;
      pos.board[sq_of(file, rank)] = p;
      ++file;
    }
  }
  if (rank != 0 || file != 8) return false;

  if (stm_part == "w") pos.stm = WHITE;
  else if (stm_part == "b") pos.stm = BLACK;
  else return false;

  pos.castling = 0;
  if (castling_part != "-") {
    for (char c : castling_part) {
      if (c == 'K') pos.castling |= 1;
      else if (c == 'Q') pos.castling |= 2;
      else if (c == 'k') pos.castling |= 4;
      else if (c == 'q') pos.castling |= 8;
      else return false;
    }
  }

  if (ep_part == "-") pos.ep_square = -1;
  else {
    int sq = str_to_sq(ep_part);
    if (sq < 0) return false;
    pos.ep_square = sq;
  }
  return true;
}

static inline bool on_board(int sq) { return sq >= 0 && sq < 64; }

static bool is_square_attacked(const Position& pos, int sq, int by_color) {
  // Pawns
  if (by_color == WHITE) {
    int r = rank_of(sq), f = file_of(sq);
    if (r > 0) {
      if (f > 0 && pos.board[sq - 9] == WP) return true;
      if (f < 7 && pos.board[sq - 7] == WP) return true;
    }
  } else {
    int r = rank_of(sq), f = file_of(sq);
    if (r < 7) {
      if (f > 0 && pos.board[sq + 7] == BP) return true;
      if (f < 7 && pos.board[sq + 9] == BP) return true;
    }
  }
  // Knights
  static constexpr int kD[8] = {17, 15, 10, 6, -6, -10, -15, -17};
  for (int d : kD) {
    int to = sq + d;
    if (!on_board(to)) continue;
    int df = abs(file_of(to) - file_of(sq));
    int dr = abs(rank_of(to) - rank_of(sq));
    if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
    Piece p = pos.board[to];
    if (by_color == WHITE && p == WN) return true;
    if (by_color == BLACK && p == BN) return true;
  }
  // King (adjacent)
  for (int dr = -1; dr <= 1; ++dr) for (int df = -1; df <= 1; ++df) {
    if (dr == 0 && df == 0) continue;
    int nf = file_of(sq) + df, nr = rank_of(sq) + dr;
    if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
    Piece p = pos.board[sq_of(nf, nr)];
    if (by_color == WHITE && p == WK) return true;
    if (by_color == BLACK && p == BK) return true;
  }
  // Sliding: bishops/queens (diagonals)
  static constexpr int diag[4] = {9, 7, -7, -9};
  for (int d : diag) {
    int cur = sq + d;
    while (on_board(cur) && abs(file_of(cur) - file_of(cur - d)) == 1) {
      Piece p = pos.board[cur];
      if (p != EMPTY) {
        if (by_color == WHITE && (p == WB || p == WQ)) return true;
        if (by_color == BLACK && (p == BB || p == BQ)) return true;
        break;
      }
      cur += d;
    }
  }
  // Sliding: rooks/queens (orthogonal)
  static constexpr int ortho[4] = {8, -8, 1, -1};
  for (int d : ortho) {
    int cur = sq + d;
    while (on_board(cur) && (d == 8 || d == -8 || abs(file_of(cur) - file_of(cur - d)) == 1)) {
      Piece p = pos.board[cur];
      if (p != EMPTY) {
        if (by_color == WHITE && (p == WR || p == WQ)) return true;
        if (by_color == BLACK && (p == BR || p == BQ)) return true;
        break;
      }
      cur += d;
    }
  }
  return false;
}

static bool in_check(const Position& pos, int color) {
  int ksq = pos.king_sq(color);
  if (ksq < 0) return false;
  return is_square_attacked(pos, ksq, color ^ 1);
}

static void gen_pseudo_moves(const Position& pos, vector<Move>& out) {
  out.clear();
  int us = pos.stm;
  int dir = (us == WHITE) ? 1 : -1;
  int pawn_start_rank = (us == WHITE) ? 1 : 6;
  int promo_rank = (us == WHITE) ? 6 : 1; // rank before promotion (to rank 7 for white, 0-indexed)
  int last_rank = (us == WHITE) ? 7 : 0;

  auto add_move = [&](int from, int to, bool cap) {
    Move m;
    m.from = from; m.to = to; m.is_capture = cap;
    out.push_back(m);
  };

  for (int sq = 0; sq < 64; ++sq) {
    Piece p = pos.board[sq];
    if (p == EMPTY || color_of(p) != us) continue;
    int af = file_of(sq), ar = rank_of(sq);
    int ap = abs_piece(p);

    if (ap == 1) {
      // Pawn
      int one = sq + dir * 8;
      if (on_board(one) && pos.board[one] == EMPTY) {
        if (ar == promo_rank) {
          for (Piece pr : {us == WHITE ? WQ : BQ, us == WHITE ? WR : BR, us == WHITE ? WB : BB, us == WHITE ? WN : BN}) {
            Move m; m.from = sq; m.to = one; m.promo = pr; out.push_back(m);
          }
        } else {
          Move m; m.from = sq; m.to = one; out.push_back(m);
          if (ar == pawn_start_rank) {
            int two = sq + dir * 16;
            if (pos.board[two] == EMPTY) {
              Move dm = m;
              dm.to = two;
              dm.is_double_push = true;
              out.push_back(dm);
            }
          }
        }
      }
      // Captures (including ep)
      for (int df : {-1, 1}) {
        int nf = af + df;
        if (nf < 0 || nf > 7) continue;
        int to = sq + dir * 8 + df;
        if (!on_board(to)) continue;
        if (pos.board[to] != EMPTY && color_of(pos.board[to]) != us) {
          if (ar == promo_rank) {
            for (Piece pr : {us == WHITE ? WQ : BQ, us == WHITE ? WR : BR, us == WHITE ? WB : BB, us == WHITE ? WN : BN}) {
              Move m; m.from = sq; m.to = to; m.is_capture = true; m.promo = pr; out.push_back(m);
            }
          } else {
            add_move(sq, to, true);
          }
        } else if (to == pos.ep_square) {
          Move m; m.from = sq; m.to = to; m.is_capture = true; m.is_ep = true;
          out.push_back(m);
        }
      }
      continue;
    }

    if (ap == 2) {
      // Knight
      static constexpr int kD[8] = {17, 15, 10, 6, -6, -10, -15, -17};
      for (int d : kD) {
        int to = sq + d;
        if (!on_board(to)) continue;
        int df = abs(file_of(to) - af);
        int dr = abs(rank_of(to) - ar);
        if (!((df == 1 && dr == 2) || (df == 2 && dr == 1))) continue;
        Piece t = pos.board[to];
        if (t == EMPTY) add_move(sq, to, false);
        else if (color_of(t) != us) add_move(sq, to, true);
      }
      continue;
    }

    if (ap == 6) {
      // King + castling
      for (int dr = -1; dr <= 1; ++dr) for (int df = -1; df <= 1; ++df) {
        if (dr == 0 && df == 0) continue;
        int nf = af + df, nr = ar + dr;
        if (nf < 0 || nf > 7 || nr < 0 || nr > 7) continue;
        int to = sq_of(nf, nr);
        Piece t = pos.board[to];
        if (t == EMPTY) add_move(sq, to, false);
        else if (color_of(t) != us) add_move(sq, to, true);
      }
      // Castling (require empty squares and not in/through check)
      if (us == WHITE) {
        if ((pos.castling & 1) && pos.board[str_to_sq("f1")] == EMPTY && pos.board[str_to_sq("g1")] == EMPTY) {
          Move m; m.is_castle = true; m.from = str_to_sq("e1"); m.to = str_to_sq("g1"); out.push_back(m);
        }
        if ((pos.castling & 2) && pos.board[str_to_sq("d1")] == EMPTY && pos.board[str_to_sq("c1")] == EMPTY && pos.board[str_to_sq("b1")] == EMPTY) {
          Move m; m.is_castle = true; m.from = str_to_sq("e1"); m.to = str_to_sq("c1"); out.push_back(m);
        }
      } else {
        if ((pos.castling & 4) && pos.board[str_to_sq("f8")] == EMPTY && pos.board[str_to_sq("g8")] == EMPTY) {
          Move m; m.is_castle = true; m.from = str_to_sq("e8"); m.to = str_to_sq("g8"); out.push_back(m);
        }
        if ((pos.castling & 8) && pos.board[str_to_sq("d8")] == EMPTY && pos.board[str_to_sq("c8")] == EMPTY && pos.board[str_to_sq("b8")] == EMPTY) {
          Move m; m.is_castle = true; m.from = str_to_sq("e8"); m.to = str_to_sq("c8"); out.push_back(m);
        }
      }
      continue;
    }

    // Sliding pieces
    auto gen_slide = [&](const int* dirs, int nd) {
      for (int i = 0; i < nd; ++i) {
        int d = dirs[i];
        int cur = sq + d;
        while (on_board(cur) && (d == 8 || d == -8 || abs(file_of(cur) - file_of(cur - d)) == 1)) {
          Piece t = pos.board[cur];
          if (t == EMPTY) {
            add_move(sq, cur, false);
          } else {
            if (color_of(t) != us) add_move(sq, cur, true);
            break;
          }
          cur += d;
        }
      }
    };

    static constexpr int bishop_dirs[4] = {9, 7, -7, -9};
    static constexpr int rook_dirs[4] = {8, -8, 1, -1};
    if (ap == 3) gen_slide(bishop_dirs, 4);
    else if (ap == 4) gen_slide(rook_dirs, 4);
    else if (ap == 5) {
      gen_slide(bishop_dirs, 4);
      gen_slide(rook_dirs, 4);
    }
  }
}

static bool make_move(Position& pos, const Move& m, Undo& u) {
  u.captured = EMPTY;
  u.ep_square = pos.ep_square;
  u.castling = pos.castling;
  u.halfmove_clock = pos.halfmove_clock;
  u.fullmove_number = pos.fullmove_number;

  Piece moving = pos.board[m.from];
  if (moving == EMPTY) return false;

  // Update clocks
  pos.halfmove_clock++;
  if (abs_piece(moving) == 1 || m.is_capture) pos.halfmove_clock = 0;

  // Clear ep by default
  pos.ep_square = -1;

  // Castling move
  if (m.is_castle) {
    // Must not castle out of/through check; validate here
    int us = pos.stm;
    if (in_check(pos, us)) return false;
    int e = (us == WHITE) ? str_to_sq("e1") : str_to_sq("e8");
    int g = (us == WHITE) ? str_to_sq("g1") : str_to_sq("g8");
    int c = (us == WHITE) ? str_to_sq("c1") : str_to_sq("c8");
    int f = (us == WHITE) ? str_to_sq("f1") : str_to_sq("f8");
    int d = (us == WHITE) ? str_to_sq("d1") : str_to_sq("d8");
    int h = (us == WHITE) ? str_to_sq("h1") : str_to_sq("h8");
    int a = (us == WHITE) ? str_to_sq("a1") : str_to_sq("a8");
    bool king_side = (m.to == g);
    // Squares king passes through must not be attacked
    int pass1 = king_side ? f : d;
    int dest = king_side ? g : c;
    if (is_square_attacked(pos, pass1, us ^ 1) || is_square_attacked(pos, dest, us ^ 1)) return false;

    // Move king
    pos.board[e] = EMPTY;
    pos.board[dest] = moving;
    // Move rook
    int rook_from = king_side ? h : a;
    int rook_to = king_side ? f : d;
    Piece rook = pos.board[rook_from];
    pos.board[rook_from] = EMPTY;
    pos.board[rook_to] = rook;

    // Update castling rights
    if (us == WHITE) pos.castling &= uint8_t(~(1 | 2));
    else pos.castling &= uint8_t(~(4 | 8));

    pos.stm ^= 1;
    if (pos.stm == WHITE) pos.fullmove_number++;
    return true;
  }

  // Capture
  if (m.is_ep) {
    // Captured pawn is behind target square
    int cap_sq = m.to + ((pos.stm == WHITE) ? -8 : 8);
    u.captured = pos.board[cap_sq];
    pos.board[cap_sq] = EMPTY;
  } else {
    u.captured = pos.board[m.to];
  }

  pos.board[m.from] = EMPTY;
  pos.board[m.to] = (m.promo != EMPTY) ? m.promo : moving;

  // Set ep target on double push
  if (m.is_double_push) {
    pos.ep_square = (pos.stm == WHITE) ? (m.from + 8) : (m.from - 8);
  }

  // Update castling rights if king/rook moves or rook captured
  auto clear_castle_if = [&](int sq, uint8_t mask) {
    if (m.from == sq || m.to == sq) pos.castling &= uint8_t(~mask);
  };
  clear_castle_if(str_to_sq("e1"), 1 | 2);
  clear_castle_if(str_to_sq("h1"), 1);
  clear_castle_if(str_to_sq("a1"), 2);
  clear_castle_if(str_to_sq("e8"), 4 | 8);
  clear_castle_if(str_to_sq("h8"), 4);
  clear_castle_if(str_to_sq("a8"), 8);

  pos.stm ^= 1;
  if (pos.stm == WHITE) pos.fullmove_number++;
  return true;
}

static void unmake_move(Position& pos, const Move& m, const Undo& u) {
  pos.stm ^= 1;
  pos.ep_square = u.ep_square;
  pos.castling = u.castling;
  pos.halfmove_clock = u.halfmove_clock;
  pos.fullmove_number = u.fullmove_number;

  if (m.is_castle) {
    int us = pos.stm;
    int e = (us == WHITE) ? str_to_sq("e1") : str_to_sq("e8");
    int g = (us == WHITE) ? str_to_sq("g1") : str_to_sq("g8");
    int c = (us == WHITE) ? str_to_sq("c1") : str_to_sq("c8");
    int f = (us == WHITE) ? str_to_sq("f1") : str_to_sq("f8");
    int d = (us == WHITE) ? str_to_sq("d1") : str_to_sq("d8");
    int h = (us == WHITE) ? str_to_sq("h1") : str_to_sq("h8");
    int a = (us == WHITE) ? str_to_sq("a1") : str_to_sq("a8");
    bool king_side = (m.to == g);
    int king_from = king_side ? g : c;
    int rook_from = king_side ? f : d;
    int rook_to = king_side ? h : a;
    Piece king = pos.board[king_from];
    Piece rook = pos.board[rook_from];
    pos.board[king_from] = EMPTY;
    pos.board[e] = king;
    pos.board[rook_from] = EMPTY;
    pos.board[rook_to] = rook;
    return;
  }

  Piece moved = pos.board[m.to];
  Piece original = moved;
  if (m.promo != EMPTY) {
    original = (pos.stm == WHITE) ? WP : BP;
  }
  pos.board[m.from] = original;
  pos.board[m.to] = u.captured;
  if (m.is_ep) {
    int cap_sq = m.to + ((pos.stm == WHITE) ? -8 : 8);
    pos.board[m.to] = EMPTY;
    pos.board[cap_sq] = u.captured;
  }
}

static void gen_legal_moves(const Position& pos, vector<Move>& out) {
  vector<Move> pseudo;
  gen_pseudo_moves(pos, pseudo);
  out.clear();
  out.reserve(pseudo.size());
  for (const Move& m : pseudo) {
    Position p2 = pos;
    Undo u;
    if (!make_move(p2, m, u)) continue;
    if (!in_check(p2, pos.stm)) out.push_back(m);
  }
}

// Basic evaluation (centipawns) from side-to-move perspective.
static int piece_value(int ap) {
  switch (ap) {
    case 1: return 100;
    case 2: return 320;
    case 3: return 330;
    case 4: return 500;
    case 5: return 900;
    case 6: return 0;
    default: return 0;
  }
}

  -20,-10,-10, -5, -5,-10,-10,-20,
  -10,  0,  0,  0,  0,  0,  0,-10,
  -10,  0,  5,  5,  5,  5,  0,-10,
   -5,  0,  5,  5,  5,  5,  0, -5,
    0,  0,  5,  5,  5,  5,  0, -5,
  -10,  5,  5,  5,  5,  5,  0,-10,
  -10,  0,  5,  0,  0,  0,  0,-10,
  -20,-10,-10, -5, -5,-10,-10,-20
};
static constexpr int PST_KING_MID[64] = {
  -30,-40,-40,-50,-50,-40,-40,-30,
  -30,-40,-40,-50,-50,-40,-40,-30,
  -30,-40,-40,-50,-50,-40,-40,-30,
  -30,-40,-40,-50,-50,-40,-40,-30,
  -20,-30,-30,-40,-40,-30,-30,-20,
  -10,-20,-20,-20,-20,-20,-20,-10,
   20, 20,  0,  0,  0,  0, 20, 20,
   20, 30, 10,  0,  0, 10, 30, 20
};

static inline int pst(Piece p, int sq) {
  int ap = abs_piece(p);
  int is_white = (p > 0);
  int idx = is_white ? sq : sq_of(file_of(sq), 7 - rank_of(sq));
  switch (ap) {
    case 1: return PST_PAWN[idx];
    case 2: return PST_KNIGHT[idx];
    case 3: return PST_BISHOP[idx];
    case 4: return PST_ROOK[idx];
    case 5: return PST_QUEEN[idx];
    case 6: return PST_KING_MID[idx];
    default: return 0;
  }
}

static int evaluate(const Position& pos) {
  int score = 0;
  for (int sq = 0; sq < 64; ++sq) {
    Piece p = pos.board[sq];
    if (p == EMPTY) continue;
    int ap = abs_piece(p);
    int val = piece_value(ap) + pst(p, sq);
    score += (p > 0) ? val : -val;
  }
  return (pos.stm == WHITE) ? score : -score;
}

struct TTEntry {
  uint64_t key = 0;
  int depth = -1;
  int score = 0;
  uint8_t flag = 0; // 0=exact,1=alpha,2=beta
  Move best{};
};

static uint64_t splitmix64(uint64_t x) {
  x += 0x9e3779b97f4a7c15ULL;
  x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
  x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
  return x ^ (x >> 31);
}

struct Zobrist {
  array<array<uint64_t, 64>, 13> piece_sq{}; // map piece index to squares
  uint64_t stm = 0;
  array<uint64_t, 16> castle{};
  array<uint64_t, 9> ep_file{}; // file 0..7 plus none=8

  static int idx(Piece p) {
    // map to 0..12 with 0 empty unused
    switch (p) {
      case WP: return 1; case WN: return 2; case WB: return 3; case WR: return 4; case WQ: return 5; case WK: return 6;
      case BP: return 7; case BN: return 8; case BB: return 9; case BR: return 10; case BQ: return 11; case BK: return 12;
      default: return 0;
    }
  }

  explicit Zobrist(uint64_t seed) {
    for (int i = 0; i < 13; ++i) {
      for (int sq = 0; sq < 64; ++sq) piece_sq[i][sq] = splitmix64(seed + uint64_t(i * 64 + sq));
    }
    stm = splitmix64(seed + 0xABCDEF01u);
    for (int i = 0; i < 16; ++i) castle[i] = splitmix64(seed + 0x1000u + i);
    for (int i = 0; i < 9; ++i) ep_file[i] = splitmix64(seed + 0x2000u + i);
  }

  uint64_t hash(const Position& pos) const {
    uint64_t h = 0;
    for (int sq = 0; sq < 64; ++sq) {
      Piece p = pos.board[sq];
      if (p == EMPTY) continue;
      int id = idx(p);
      h ^= piece_sq[id][sq];
    }
    if (pos.stm == BLACK) h ^= stm;
    h ^= castle[pos.castling & 15];
    int epf = (pos.ep_square < 0) ? 8 : file_of(pos.ep_square);
    h ^= ep_file[epf];
    return h;
  }
};

struct Search {
  chrono::steady_clock::time_point start;
  chrono::steady_clock::time_point deadline;
  atomic<bool> stop{false};
  Zobrist zob;
  vector<TTEntry> tt;
  uint64_t tt_mask = 0;

  explicit Search(double timeout_seconds)
      : zob(uint64_t(chrono::high_resolution_clock::now().time_since_epoch().count())) {
    start = chrono::steady_clock::now();
    auto ms = chrono::milliseconds((int64_t)(timeout_seconds * 1000.0));
    // Keep a small buffer to avoid timeouts (30ms or 2%).
    auto buffer = chrono::milliseconds(max<int64_t>(30, (int64_t)(timeout_seconds * 20.0)));
    deadline = start + ms - buffer;
    // TT ~ 8MB
    size_t entries = 1u << 18; // 262k
    tt.resize(entries);
    tt_mask = entries - 1;
  }

  bool time_up() {
    if (stop.load(memory_order_relaxed)) return true;
    if (chrono::steady_clock::now() >= deadline) {
      stop.store(true, memory_order_relaxed);
      return true;
    }
    return false;
  }

  TTEntry* tt_probe(uint64_t key) {
    TTEntry& e = tt[size_t(key) & tt_mask];
    if (e.key == key) return &e;
    return nullptr;
  }

  void tt_store(uint64_t key, int depth, int score, uint8_t flag, const Move& best) {
    TTEntry& e = tt[size_t(key) & tt_mask];
    if (e.key != key || depth >= e.depth) {
      e.key = key;
      e.depth = depth;
      e.score = score;
      e.flag = flag;
      e.best = best;
    }
  }

  static constexpr int INF = 30000;
  static constexpr int MATE = 29000;

  int quiesce(Position& pos, int alpha, int beta) {
    if (time_up()) return evaluate(pos);
    int stand = evaluate(pos);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;

    vector<Move> moves;
    gen_legal_moves(pos, moves);
    for (const Move& m : moves) {
      if (!(m.is_capture || m.is_ep || m.promo != EMPTY)) continue;
      Undo u;
      if (!make_move(pos, m, u)) continue;
      if (in_check(pos, pos.stm ^ 1)) { unmake_move(pos, m, u); continue; } // illegal safeguard
      int score = -quiesce(pos, -beta, -alpha);
      unmake_move(pos, m, u);
      if (score >= beta) return beta;
      if (score > alpha) alpha = score;
    }
    return alpha;
  }

  int negamax(Position& pos, int depth, int alpha, int beta, Move& pv_move) {
    if (time_up()) return evaluate(pos);

    uint64_t key = zob.hash(pos);
    int alpha0 = alpha;

    if (TTEntry* e = tt_probe(key)) {
      if (e->depth >= depth) {
        if (e->flag == 0) return e->score;
        if (e->flag == 1 && e->score <= alpha) return e->score;
        if (e->flag == 2 && e->score >= beta) return e->score;
      }
    }

    if (depth <= 0) return quiesce(pos, alpha, beta);

    vector<Move> moves;
    gen_legal_moves(pos, moves);
    if (moves.empty()) {
      if (in_check(pos, pos.stm)) return -MATE + (100 - depth); // checkmated
      return 0; // stalemate
    }

    // Move ordering: TT best, captures/promos first.
    Move ttbest{};
    bool has_ttbest = false;
    if (TTEntry* e = tt_probe(key)) { ttbest = e->best; has_ttbest = (e->best.from >= 0); }
    auto score_move = [&](const Move& m) {
      int s = 0;
      if (has_ttbest && m.from == ttbest.from && m.to == ttbest.to && m.promo == ttbest.promo && m.is_castle == ttbest.is_castle && m.is_ep == ttbest.is_ep) s += 1000000;
      if (m.is_castle) s += 50;
      if (m.promo != EMPTY) s += 900;
      if (m.is_capture || m.is_ep) s += 500;
      // small history-like: advance pawns/center
      int tf = file_of(m.to), tr = rank_of(m.to);
      if (tf >= 2 && tf <= 5 && tr >= 2 && tr <= 5) s += 10;
      return s;
    };
    stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) { return score_move(a) > score_move(b); });

    Move best{};
    int bestScore = -INF;

    for (size_t i = 0; i < moves.size(); ++i) {
      const Move& m = moves[i];
      Undo u;
      if (!make_move(pos, m, u)) continue;
      if (in_check(pos, pos.stm ^ 1)) { unmake_move(pos, m, u); continue; } // illegal

      Move childpv{};
      int score = -negamax(pos, depth - 1, -beta, -alpha, childpv);
      unmake_move(pos, m, u);

      if (stop.load(memory_order_relaxed)) break;
      if (score > bestScore) { bestScore = score; best = m; }
      if (score > alpha) { alpha = score; pv_move = m; }
      if (alpha >= beta) break;
    }

    uint8_t flag = 0;
    if (bestScore <= alpha0) flag = 1;
    else if (bestScore >= beta) flag = 2;
    else flag = 0;
    tt_store(key, depth, bestScore, flag, best.from >= 0 ? best : pv_move);
    return bestScore;
  }

  Move iterative_deepening(Position root) {
    vector<Move> legal;
    gen_legal_moves(root, legal);
    if (legal.empty()) return Move{};
    Move best = legal[0];
    int bestScore = -INF;

    for (int depth = 1; depth <= 64; ++depth) {
      if (time_up()) break;
      Position pos = root;
      Move pv{};
      int score = negamax(pos, depth, -INF, INF, pv);
      if (stop.load(memory_order_relaxed)) break;
      if (pv.from >= 0) { best = pv; bestScore = score; }
      // early exit on forced mate found
      if (abs(bestScore) > MATE - 200) break;
    }
    return best;
  }
};

static string move_to_notation(Position pos, const Move& m) {
  // For check/mate suffix, we need to apply the move and see opponent state.
  auto piece_letter = [&](Piece p) -> char {
    int ap = abs_piece(p);
    if (ap == 1) return 'p';
    if (ap == 2) return 'N';
    if (ap == 3) return 'B';
    if (ap == 4) return 'R';
    if (ap == 5) return 'Q';
    return 'K';
  };

  if (m.from < 0) return "0000";

  Piece moving = pos.board[m.from];
  if (m.is_castle) {
    string s = (file_of(m.to) == 6) ? "O-O" : "O-O-O";
    // Determine check/mate
    Undo u;
    if (make_move(pos, m, u)) {
      bool chk = in_check(pos, pos.stm);
      vector<Move> replies;
      gen_legal_moves(pos, replies);
      if (chk && replies.empty()) s.push_back('#');
      else if (chk) s.push_back('+');
    }
    return s;
  }

  string s;
  s.push_back(piece_letter(moving));
  s += sq_to_str(m.from);
  if (m.is_capture || m.is_ep) s.push_back('x');
  s += sq_to_str(m.to);
  if (m.promo != EMPTY) {
    s.push_back('/');
    s.push_back(piece_letter(m.promo)); // N/B/R/Q (promotion to K not possible)
    if (s.back() == 'p') s.back() = 'Q';
  }
  if (m.is_ep) s += "ep";

  Undo u;
  if (make_move(pos, m, u)) {
    bool chk = in_check(pos, pos.stm);
    vector<Move> replies;
    gen_legal_moves(pos, replies);
    if (chk && replies.empty()) s.push_back('#');
    else if (chk) s.push_back('+');
  }
  return s;
}

} // namespace chess

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  double timeout = 0;
  string fen;
  if (!(cin >> timeout)) return 0;
  cin.ignore(numeric_limits<streamsize>::max(), '\n');
  getline(cin, fen);
  if (fen.empty()) return 0;

  chess::Position pos;
  if (!chess::parse_fen(fen, pos)) {
    cout << "0000\n";
    return 0;
  }

  chess::Search search(timeout);
  chess::Move best = search.iterative_deepening(pos);
  string out = chess::move_to_notation(pos, best);
  cout << out << "\n";
  return 0;
}
