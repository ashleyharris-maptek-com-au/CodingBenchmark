// Mission-control workstation scheduler benchmark
// Reads JSON from stdin, outputs a JSON schedule.
//
// Heuristic: event-driven parallel list scheduling with a priority score that
// mixes weighted tardiness risk, deadlines, and critical-path (bottom level).

declare const require: any;
const fs = require("fs");

type TaskId = string | number;

type InputTask = {
  id: TaskId;
  duration: number;
  deps?: TaskId[];
  deadline?: number;
  weight?: number;
};

type InputData = {
  machineCount: number;
  tasks: InputTask[];
};

type ScheduleItem = { id: TaskId; start: number; machine: number };

class MinHeap<T> {
  private a: T[] = [];
  constructor(private less: (x: T, y: T) => boolean) {}
  size(): number {
    return this.a.length;
  }
  peek(): T | undefined {
    return this.a[0];
  }
  push(v: T): void {
    const a = this.a;
    a.push(v);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      while (true) {
        const l = i * 2 + 1;
        const r = l + 1;
        let m = i;
        if (l < a.length && this.less(a[l], a[m])) m = l;
        if (r < a.length && this.less(a[r], a[m])) m = r;
        if (m === i) break;
        const tmp = a[i];
        a[i] = a[m];
        a[m] = tmp;
        i = m;
      }
    }
    return top;
  }
}

function clampPos(x: number): number {
  return x > 0 ? x : 0;
}

const input: string = fs.readFileSync(0, "utf8");
const data: InputData = JSON.parse(input);

const m = Math.max(1, (data.machineCount | 0) || 1);
const tasksIn = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksIn.length;

// Map ids to indices
const idToIdx = new Map<string, number>();
const ids: TaskId[] = new Array(n);
for (let i = 0; i < n; i++) {
  const id = tasksIn[i].id;
  ids[i] = id;
  idToIdx.set(String(id), i);
}

const duration = new Int32Array(n);
const deadline = new Int32Array(n);
const weight = new Float64Array(n);
const indeg0 = new Int32Array(n);

const outEdges: number[][] = Array.from({ length: n }, () => []);
const inEdges: number[][] = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
  const t = tasksIn[i];
  const d = Math.max(0, (t.duration as any) | 0);
  duration[i] = d;
  const dl = t.deadline === undefined ? 0 : ((t.deadline as any) | 0);
  deadline[i] = dl;
  const w = t.weight === undefined ? 1 : Number(t.weight);
  weight[i] = Number.isFinite(w) && w > 0 ? w : 1;
}

for (let i = 0; i < n; i++) {
  const deps = tasksIn[i].deps || [];
  for (let j = 0; j < deps.length; j++) {
    const depId = deps[j];
    const u = idToIdx.get(String(depId));
    if (u === undefined) continue; // ignore unknown deps
    outEdges[u].push(i);
    inEdges[i].push(u);
    indeg0[i]++;
  }
}

// Kahn topo for bottom-level computation (detect cycles: fall back to input order).
const q: number[] = [];
for (let i = 0; i < n; i++) if (indeg0[i] === 0) q.push(i);
const topo: number[] = [];
for (let qi = 0; qi < q.length; qi++) {
  const u = q[qi];
  topo.push(u);
  const outs = outEdges[u];
  for (let k = 0; k < outs.length; k++) {
    const v = outs[k];
    indeg0[v]--;
    if (indeg0[v] === 0) q.push(v);
  }
}

const bottom = new Float64Array(n);
if (topo.length === n) {
  for (let ti = n - 1; ti >= 0; ti--) {
    const u = topo[ti];
    let best = 0;
    const outs = outEdges[u];
    for (let k = 0; k < outs.length; k++) {
      const v = outs[k];
      if (bottom[v] > best) best = bottom[v];
    }
    bottom[u] = duration[u] + best;
  }
} else {
  for (let i = 0; i < n; i++) bottom[i] = duration[i];
}

// Scheduling state
const remainingDeps = new Int32Array(n);
const maxDepFinish = new Int32Array(n);
for (let i = 0; i < n; i++) {
  remainingDeps[i] = inEdges[i].length | 0;
  maxDepFinish[i] = 0;
}

const startTime = new Int32Array(n);
const finishTime = new Int32Array(n);
startTime.fill(-1);
finishTime.fill(-1);

const ready: number[] = [];
for (let i = 0; i < n; i++) if (remainingDeps[i] === 0) ready.push(i);

// Machines heap by availability time then index
type MachineState = { t: number; i: number };
const machines = new MinHeap<MachineState>((a, b) => (a.t !== b.t ? a.t < b.t : a.i < b.i));
for (let i = 0; i < m; i++) machines.push({ t: 0, i });

// Helper: select best task among those with release<=t from ready[] (unsorted).
function pickTask(now: number): number {
  let bestIdx = -1;
  let bestScore = -Infinity;

  for (let pos = 0; pos < ready.length; pos++) {
    const u = ready[pos];
    const rel = maxDepFinish[u];
    if (rel > now) continue;

    const dur = duration[u];
    const dl = deadline[u];
    const w = weight[u];

    // Slack to deadline if started now
    const slack = dl - (now + dur);
    const tard = slack < 0 ? -slack : 0;

    // Heuristic score: prioritize tasks likely to be late (weighted),
    // then earlier deadlines, then critical path length.
    const invSlack = 1 / (1 + clampPos(slack));
    const urgency = (tard + 1) / (dur + 1);
    const score = 8 * w * urgency + 2.5 * w * invSlack + 0.15 * bottom[u] - 0.001 * dl;

    if (score > bestScore) {
      bestScore = score;
      bestIdx = pos;
    }
  }

  if (bestIdx < 0) return -1;
  const u = ready[bestIdx];
  // remove in O(1)
  ready[bestIdx] = ready[ready.length - 1];
  ready.pop();
  return u;
}

// Track next release time among ready tasks (release>now)
function nextRelease(after: number): number {
  let nr = Infinity;
  for (let i = 0; i < ready.length; i++) {
    const u = ready[i];
    const rel = maxDepFinish[u];
    if (rel > after && rel < nr) nr = rel;
  }
  return nr;
}

const schedule: ScheduleItem[] = [];
let scheduledCount = 0;

while (scheduledCount < n && machines.size() > 0) {
  const ms = machines.pop()!;
  let now = ms.t | 0;

  let u = pickTask(now);
  if (u < 0) {
    const nr = nextRelease(now);
    if (nr !== Infinity) {
      now = nr | 0;
      ms.t = now;
      u = pickTask(now);
    }
  }

  if (u < 0) {
    // Nothing is schedulable: likely a cycle or missing deps; fall back by forcing a task.
    // Pick an unscheduled task with smallest remainingDeps, then wait for its deps.
    let best = -1;
    let bestRem = 1e9;
    for (let i = 0; i < n; i++) {
      if (startTime[i] >= 0) continue;
      const r = remainingDeps[i];
      if (r < bestRem) {
        bestRem = r;
        best = i;
        if (r === 0) break;
      }
    }
    if (best < 0) break;
    // ensure it's in ready, then try again
    if (remainingDeps[best] === 0) ready.push(best);
    ms.t = now;
    machines.push(ms);
    continue;
  }

  const st = Math.max(now, maxDepFinish[u] | 0) | 0;
  const ft = (st + (duration[u] | 0)) | 0;

  startTime[u] = st;
  finishTime[u] = ft;
  schedule.push({ id: ids[u], start: st, machine: ms.i });
  scheduledCount++;

  // Update dependents
  const outs = outEdges[u];
  for (let k = 0; k < outs.length; k++) {
    const v = outs[k];
    if (startTime[v] >= 0) continue;
    if (ft > maxDepFinish[v]) maxDepFinish[v] = ft;
    const rem = (remainingDeps[v] - 1) | 0;
    remainingDeps[v] = rem;
    if (rem === 0) ready.push(v);
  }

  ms.t = ft;
  machines.push(ms);
}

// If any tasks remain unscheduled (cycles), output a simple order to let grader place greedily.
if (scheduledCount < n) {
  // Produce an order: try topo if available, else input order with deps-first heuristic.
  let order: number[] = [];
  if (topo.length === n) {
    order = topo.slice();
  } else {
    // Kahn with original indegrees (recompute)
    const indeg = new Int32Array(n);
    for (let i = 0; i < n; i++) indeg[i] = inEdges[i].length | 0;
    const qq: number[] = [];
    for (let i = 0; i < n; i++) if (indeg[i] === 0) qq.push(i);
    for (let qi = 0; qi < qq.length; qi++) {
      const u = qq[qi];
      order.push(u);
      const outs = outEdges[u];
      for (let k = 0; k < outs.length; k++) {
        const v = outs[k];
        indeg[v]--;
        if (indeg[v] === 0) qq.push(v);
      }
    }
    for (let i = 0; i < n; i++) if (order.indexOf(i) < 0) order.push(i);
  }
  const out = { order: order.map((i) => ids[i]) };
  process.stdout.write(JSON.stringify(out));
} else {
  process.stdout.write(JSON.stringify({ schedule }));
}
