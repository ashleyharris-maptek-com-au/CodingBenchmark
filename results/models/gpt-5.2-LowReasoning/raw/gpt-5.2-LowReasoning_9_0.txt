using System;
using System.IO;
using System.Text;
using System.Collections.Generic;

public static class Program
{
    // Fast input reader for large instances.
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len, _pos;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    // Treap keyed by remaining capacity; each key maps to a linked list of bin ids.
    private sealed class RemTreap
    {
        private sealed class Node
        {
            public int Key;
            public uint Prio;
            public Node L, R;
            public int HeadBin;     // head of singly-linked list of bin ids
            public int CountBins;   // number of bins at this key
        }

        private Node _root;
        private readonly Random _rng = new Random(1234567);
        private int[] _nextBin; // next pointers for bin linked lists (indexed by bin id)

        public RemTreap(int maxBinsHint)
        {
            _nextBin = new int[Math.Max(4, maxBinsHint)];
            Array.Fill(_nextBin, -1);
        }

        public void EnsureBinCapacity(int binId)
        {
            if (binId < _nextBin.Length) return;
            int oldLen = _nextBin.Length;
            int newLen = oldLen;
            while (newLen <= binId) newLen *= 2;
            Array.Resize(ref _nextBin, newLen);
            Array.Fill(_nextBin, -1, oldLen, newLen - oldLen);
        }

        private uint NextPrio()
        {
            // Fast-ish 32-bit priority from Random.
            return (uint)_rng.Next(int.MinValue, int.MaxValue);
        }

        private static Node RotateRight(Node y)
        {
            Node x = y.L;
            y.L = x.R;
            x.R = y;
            return x;
        }

        private static Node RotateLeft(Node x)
        {
            Node y = x.R;
            x.R = y.L;
            y.L = x;
            return y;
        }

        public void InsertBin(int key, int binId)
        {
            EnsureBinCapacity(binId);
            _root = Insert(_root, key, binId, NextPrio());
        }

        private Node Insert(Node node, int key, int binId, uint prio)
        {
            if (node == null)
            {
                var n = new Node { Key = key, Prio = prio, HeadBin = binId, CountBins = 1 };
                _nextBin[binId] = -1;
                return n;
            }

            if (key == node.Key)
            {
                _nextBin[binId] = node.HeadBin;
                node.HeadBin = binId;
                node.CountBins++;
                return node;
            }

            if (key < node.Key)
            {
                node.L = Insert(node.L, key, binId, prio);
                if (node.L != null && node.L.Prio > node.Prio) node = RotateRight(node);
            }
            else
            {
                node.R = Insert(node.R, key, binId, prio);
                if (node.R != null && node.R.Prio > node.Prio) node = RotateLeft(node);
            }
            return node;
        }

        public bool TryTakeLowerBound(int minKey, out int key, out int binId)
        {
            Node cur = _root;
            Node best = null;
            while (cur != null)
            {
                if (cur.Key >= minKey)
                {
                    best = cur;
                    cur = cur.L;
                }
                else
                {
                    cur = cur.R;
                }
            }

            if (best == null)
            {
                key = 0;
                binId = -1;
                return false;
            }

            key = best.Key;
            binId = best.HeadBin;
            best.HeadBin = _nextBin[binId];
            _nextBin[binId] = -1;
            best.CountBins--;
            if (best.CountBins == 0) _root = Erase(_root, best.Key);
            return true;
        }

        private Node Erase(Node node, int key)
        {
            if (node == null) return null;
            if (key < node.Key) { node.L = Erase(node.L, key); return node; }
            if (key > node.Key) { node.R = Erase(node.R, key); return node; }

            // Remove this node by rotating it down until a leaf.
            if (node.L == null) return node.R;
            if (node.R == null) return node.L;
            if (node.L.Prio > node.R.Prio)
            {
                node = RotateRight(node);
                node.R = Erase(node.R, key);
            }
            else
            {
                node = RotateLeft(node);
                node.L = Erase(node.L, key);
            }
            return node;
        }
    }

    private struct Bin
    {
        public int Remaining;
        public int[] Cuts;
        public int Count;

        public Bin(int remaining)
        {
            Remaining = remaining;
            Cuts = new int[4];
            Count = 0;
        }

        public void Add(int cutIndex)
        {
            if (Count >= Cuts.Length) Array.Resize(ref Cuts, Cuts.Length * 2);
            Cuts[Count++] = cutIndex;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int stockLength = fs.NextInt();
        if (n <= 0 || stockLength <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Pack (len, idx) into a long for faster sort + compact storage.
        long[] packed = new long[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            if (len == int.MinValue) len = 0;
            // Constraints say len <= stockLength; assume valid.
            packed[i] = ((long)len << 32) | (uint)i;
        }
        Array.Sort(packed); // ascending by len then idx

        var bins = new List<Bin>(Math.Min(n, 1 << 16));
        var treap = new RemTreap(Math.Min(n, 1 << 16));

        // Best-Fit Decreasing: for each cut (largest first), take smallest remaining >= len.
        for (int p = n - 1; p >= 0; p--)
        {
            int len = (int)(packed[p] >> 32);
            int idx = (int)(packed[p] & 0xFFFFFFFF);

            if (!treap.TryTakeLowerBound(len, out int remKey, out int binId))
            {
                // Create a new bin.
                int newId = bins.Count;
                var b = new Bin(stockLength - len);
                b.Add(idx);
                bins.Add(b);
                treap.InsertBin(b.Remaining, newId);
            }
            else
            {
                // Use existing bin.
                Bin b = bins[binId];
                b.Add(idx);
                int newRem = remKey - len;
                b.Remaining = newRem;
                bins[binId] = b;
                treap.InsertBin(newRem, binId);
            }
        }

        using var sw = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        sw.WriteLine(bins.Count);

        var sb = new StringBuilder(1 << 16);
        for (int i = 0; i < bins.Count; i++)
        {
            sb.Clear();
            var b = bins[i];
            for (int j = 0; j < b.Count; j++)
            {
                if (j != 0) sb.Append(' ');
                sb.Append(b.Cuts[j]);
            }
            sw.WriteLine(sb.ToString());
        }
    }
}
