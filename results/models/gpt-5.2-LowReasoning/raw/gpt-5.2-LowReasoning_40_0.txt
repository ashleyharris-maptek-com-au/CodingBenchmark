// Maximum Weight Independent Set (0/1 ILP with pairwise conflicts)
// C++17, exact branch-and-bound with greedy clique-cover bound.

#include <bits/stdc++.h>
using namespace std;

struct Bitset {
    int n = 0;
    int blocks = 0;
    vector<uint64_t> w;

    Bitset() = default;
    explicit Bitset(int n_) : n(n_), blocks((n_ + 63) / 64), w(blocks, 0) {}

    void set(int i) { w[i >> 6] |= (1ULL << (i & 63)); }
    void reset(int i) { w[i >> 6] &= ~(1ULL << (i & 63)); }
    bool test(int i) const { return (w[i >> 6] >> (i & 63)) & 1ULL; }

    bool empty() const {
        for (uint64_t x : w) if (x) return false;
        return true;
    }

    int pop_lsb() {
        for (int b = 0; b < blocks; b++) {
            uint64_t x = w[b];
            if (!x) continue;
            int bit = __builtin_ctzll(x);
            w[b] = x & (x - 1);
            return (b << 6) + bit;
        }
        return -1;
    }

    int lsb_index() const {
        for (int b = 0; b < blocks; b++) {
            uint64_t x = w[b];
            if (!x) continue;
            int bit = __builtin_ctzll(x);
            return (b << 6) + bit;
        }
        return -1;
    }

    int count() const {
        int c = 0;
        for (uint64_t x : w) c += __builtin_popcountll(x);
        return c;
    }

    void and_with(const Bitset& other) {
        for (int i = 0; i < blocks; i++) w[i] &= other.w[i];
    }
    void or_with(const Bitset& other) {
        for (int i = 0; i < blocks; i++) w[i] |= other.w[i];
    }
    void andnot_with(const Bitset& other) {
        for (int i = 0; i < blocks; i++) w[i] &= ~other.w[i];
    }

    static Bitset And(const Bitset& a, const Bitset& b) {
        Bitset r(a.n);
        for (int i = 0; i < a.blocks; i++) r.w[i] = a.w[i] & b.w[i];
        return r;
    }

    static Bitset AndNot(const Bitset& a, const Bitset& b) {
        Bitset r(a.n);
        for (int i = 0; i < a.blocks; i++) r.w[i] = a.w[i] & ~b.w[i];
        return r;
    }
};

static int N, M;
static vector<long long> weightv;
static vector<Bitset> adj; // adjacency in original graph (conflicts)

static long long bestWeight = 0;
static vector<char> bestSol;
static vector<char> curSol;

static long long greedy_clique_cover_bound(const Bitset& cand) {
    // Partition cand into cliques (in original graph) via greedy construction,
    // equivalent to greedy coloring in complement graph.
    // Upper bound on MWIS: sum over cliques of max weight in that clique.
    Bitset U = cand;
    long long ub = 0;
    while (!U.empty()) {
        Bitset C = U; // candidates for current clique
        long long bestInClique = 0;
        while (!C.empty()) {
            int v = C.pop_lsb();
            if (v < 0 || v >= N) break;
            if (!U.test(v)) continue; // might have been removed
            U.reset(v);
            bestInClique = max(bestInClique, weightv[v]);
            // To keep clique property: next vertices must be adjacent to v in original graph.
            C.and_with(adj[v]);
        }
        ub += bestInClique;
        if (ub + 0LL < bestWeight) {
            // early exit if already hopeless (minor speed-up)
        }
    }
    return ub;
}

static int choose_branch_vertex(const Bitset& cand) {
    // Heuristic: pick vertex with largest weight among candidates.
    int bestV = -1;
    long long bestW = LLONG_MIN;
    for (int b = 0; b < cand.blocks; b++) {
        uint64_t x = cand.w[b];
        while (x) {
            uint64_t lsb = x & -x;
            int bit = __builtin_ctzll(x);
            int v = (b << 6) + bit;
            if (v < N) {
                long long wv = weightv[v];
                if (wv > bestW) {
                    bestW = wv;
                    bestV = v;
                }
            }
            x ^= lsb;
        }
    }
    if (bestV != -1) return bestV;
    return cand.lsb_index();
}

static void dfs(Bitset cand, long long curW) {
    if (cand.empty()) {
        if (curW > bestWeight) {
            bestWeight = curW;
            bestSol = curSol;
        }
        return;
    }

    long long ub = curW + greedy_clique_cover_bound(cand);
    if (ub <= bestWeight) return;

    int v = choose_branch_vertex(cand);
    if (v < 0 || v >= N) {
        if (curW > bestWeight) {
            bestWeight = curW;
            bestSol = curSol;
        }
        return;
    }

    // Branch 1: include v (only if beneficial; negative weights can be skipped)
    if (weightv[v] > 0) {
        Bitset candInc = cand;
        candInc.reset(v);
        candInc.andnot_with(adj[v]); // remove neighbors of v
        curSol[v] = 1;
        dfs(std::move(candInc), curW + weightv[v]);
        curSol[v] = 0;
    }

    // Branch 2: exclude v
    Bitset candExc = cand;
    candExc.reset(v);
    curSol[v] = 0;
    dfs(std::move(candExc), curW);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> N >> M)) return 0;
    weightv.assign(N, 0);
    for (int i = 0; i < N; i++) cin >> weightv[i];

    adj.assign(N, Bitset(N));
    for (int i = 0; i < N; i++) adj[i] = Bitset(N);

    for (int i = 0; i < M; i++) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= N || v < 0 || v >= N || u == v) continue;
        adj[u].set(v);
        adj[v].set(u);
    }

    // Initial candidate set: all vertices with positive weight (never need negatives)
    Bitset cand(N);
    for (int i = 0; i < N; i++) {
        if (weightv[i] > 0) cand.set(i);
    }

    bestSol.assign(N, 0);
    curSol.assign(N, 0);

    dfs(std::move(cand), 0);

    // Output
    cout << bestWeight << "\n";
    for (int i = 0; i < N; i++) {
        if (i) cout << ' ';
        cout << int(bestSol[i] ? 1 : 0);
    }
    cout << "\n";
    return 0;
}
