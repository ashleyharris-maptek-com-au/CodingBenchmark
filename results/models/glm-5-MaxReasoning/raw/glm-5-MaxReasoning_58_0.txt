Looking at this problem, I need to:
1. Compute shortest grid walking distances (BFS) between robots and jobs
2. Assign jobs to robots greedily to minimize total travel distance
3. Handle the deterministic tie-breaking to match the reference

The key insight is that robots move after each job assignment, so I need to track their positions dynamically. I'll precompute BFS from all unique positions (robot starts + job locations) to avoid redundant computation.

For the greedy approach: pick the (robot, job) pair with minimum distance, breaking ties by priority (descending), then job ID (ascending), then robot ID (ascending).

const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const {grid, robots, jobs} = data;
const rows = grid.length;
const cols = rows > 0 ? grid[0].length : 0;
const INF = 1000000000;

const posKey = (x, y) => y * cols + x;

const positions = [];
const posToIndex = new Map();

function addPosition(x, y) {
    const key = posKey(x, y);
    if (!posToIndex.has(key)) {
        posToIndex.set(key, positions.length);
        positions.push({x, y});
    }
    return posToIndex.get(key);
}

    robotPosIdxArr[r.id] = addPosition(r.x, r.y);
    robotExists[r.id] = true;
}

for (const j of jobs) addPosition(j.x, j.y);
