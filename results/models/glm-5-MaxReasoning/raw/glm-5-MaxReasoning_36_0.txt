Looking at this problem, I need to implement an exact algorithm for Maximum Independent Set, which is NP-hard. For exact solutions, branch-and-bound with good pruning is the standard approach.

**Key optimizations I'll use:**
1. **Branch-and-bound**: For each vertex, either include it (remove it and neighbors) or exclude it
2. **Coloring-based upper bound**: Greedy graph coloring gives an upper bound on how many more vertices can be added
3. **Vertex selection heuristic**: Branch on highest-degree vertices first (they constrain the problem more)
4. **Simple count pruning**: If current_size + remaining_vertices ≤ best, prune

// Upper bound via greedy coloring
int color_ub(const vector<bool>& avail) {
    vector<int> order;
    for (int i = 0; i < n; i++) if (avail[i]) order.push_back(i);
    if (order.empty()) return 0;
    
    auto deg = [&](int v) -> int {
        int d = 0;
        for (int u : adj[v]) if (avail[u]) d++;
        return d;
    };
    sort(order.begin(), order.end(), [&](int a, int b) { return deg(a) > deg(b); });
    
    vector<int> col(n, -1);
    int colors = 0;
    for (int v : order) {
        set<int> used;
        for (int u : adj[v]) if (avail[u] && col[u] >= 0) used.insert(col[u]);
        int c = 0; while (used.count(c)) c++;
        col[v] = c;
        colors = max(colors, c + 1);
    }
    return colors;
}
