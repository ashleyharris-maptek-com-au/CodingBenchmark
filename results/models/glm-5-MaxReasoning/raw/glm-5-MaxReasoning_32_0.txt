Here is the C++ solution for the Steiner Tree problem. This implementation uses the **Dreyfus-Wagner algorithm** with Dynamic Programming and Dijkstra's algorithm. This is an exact algorithm suitable for graphs where the number of terminals ($K$) is relatively small (typically $K \le 20$), which is standard for this NP-Hard problem in competitive programming contexts.

The solution reconstructs the tree edges using a parent tracking table and outputs the minimum weight and the edge list.

```cpp
/**
 * Steiner Tree Solver (Dreyfus-Wagner Algorithm)
 * 
 * Complexity: O(3^K * V + 2^K * E log V)
 * Suitable for graphs with a small number of terminals (K <= 20).
 */

#include <iostream>
#include <vector>
#include <queue>
#include <set>
#include <algorithm>
#include <limits>

using namespace std;

// Use long long for weights to prevent overflow
using ll = long long;

const ll INF = 1e18;

struct Edge {
    int to;
    int weight;
};

int main() {
    // Optimize I/O operations
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        adj[u].push_back({v, w});
        adj[v].push_back({u, w});
    }

    vector<int> terminals(k);
    vector<int> terminal_map(n, -1); // Maps vertex index to terminal index [0..k-1]
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
        // Assuming valid indices in input
        terminal_map[terminals[i]] = i;
    }

    // DP Table: dp[mask][v] = minimum weight of a Steiner tree connecting 
    // the terminals in 'mask' (bitmask) ending at vertex 'v'.
    vector<vector<ll>> dp(1 << k, vector<ll>(n, INF));

    // Parent Table: for reconstruction.
    // parent[mask][v] stores info to backtrack the previous state.
    // Encoding:
    //  -1               : Start (terminal initialization)
    //  0..n-1           : Predecessor vertex index (implies edge relaxation via Dijkstra)
    //  n..n+2^K-1       : Submask used for merge (value - n = submask)
    vector<vector<int>> parent(1 << k, vector<int>(n, -1));

    // 1. Initialize DP for single terminals
    for (int i = 0; i < k; ++i) {
        int t = terminals[i];
        dp[1 << i][t] = 0;
    }

    // 2. Dreyfus-Wagner Algorithm
    for (int mask = 1; mask < (1 << k); ++mask) {
        
        // A. Merge subsets
        // Iterate over non-empty proper subsets of 'mask'
        // We iterate 'sub = (mask-1) & mask' down to 0.
        // This covers all splits. Note: processing both (sub, rest) and (rest, sub)
        // is redundant but safe. We update only if cost improves.
        for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
            int rest = mask ^ sub;
            // Ensure we process valid pairs efficiently?
            // Standard subset iteration covers all.
            
            for (int v = 0; v < n; ++v) {
                if (dp[sub][v] != INF && dp[rest][v] != INF) {
                    ll new_cost = dp[sub][v] + dp[rest][v];
                    if (new_cost < dp[mask][v]) {
                        dp[mask][v] = new_cost;
                        parent[mask][v] = n + sub; // Encode submask
                    }
                }
            }
        }

        // B. Extend paths (Dijkstra)
        // Find shortest paths extending the current mask's tree.
        // We treat vertices with finite dp[mask][v] as sources.
        priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
        
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] != INF) {
                pq.push({dp[mask][v], v});
            }
        }

        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();

            if (d > dp[mask][u]) continue;

            for (const auto& edge : adj[u]) {
                if (dp[mask][u] + edge.weight < dp[mask][edge.to]) {
                    dp[mask][edge.to] = dp[mask][u] + edge.weight;
                    parent[mask][edge.to] = u; // Store predecessor vertex
                    pq.push({dp[mask][edge.to], edge.to});
                }
            }
        }
    }

    // 3. Find the optimal solution
    int full_mask = (1 << k) - 1;
    ll min_weight = INF;
    int final_node = -1;

    for (int v = 0; v < n; ++v) {
        if (dp[full_mask][v] < min_weight) {
            min_weight = dp[full_mask][v];
            final_node = v;
        }
    }

    if (min_weight == INF) {
        // No solution exists (disconnected terminals)
        cout << "0 0\n";
        return 0;
    }

    // 4. Reconstruct the tree
    set<pair<int, int>> tree_edges;
    vector<pair<int, int>> stk; // Stack for DFS traversal of the state graph
    stk.push_back({full_mask, final_node});

    while (!stk.empty()) {
        auto [mask, u] = stk.back();
        stk.pop_back();

        int p = parent[mask][u];

        if (p == -1) {
            // Base case: reached an initial terminal state. 
            continue;
        }

        if (p < n) {
            // Case 1: Predecessor was a vertex (Edge relaxation)
            // The transition was from (mask, p) -> (mask, u)
            int v1 = u;
            int v2 = p;
            if (v1 > v2) swap(v1, v2);
            tree_edges.insert({v1, v2});
            stk.push_back({mask, p});
        } else {
            // Case 2: Predecessor was a submask (Merge)
            // The transition was from (sub, u) + (rest, u) -> (mask, u)
            int sub = p - n;
            int rest = mask ^ sub;
            stk.push_back({sub, u});
            stk.push_back({rest, u});
        }
    }

    // 5. Output
    cout << min_weight << " " << tree_edges.size() << "\n";
    for (const auto& edge : tree_edges) {
        cout << edge.first << " " << edge.second << "\n";
    }

    return 0;
}
```
