Looking at this problem, I need to implement an efficient heuristic for the Minimum Dominating Set problem. Since it's NP-hard, I'll use a combination of:

1. **Greedy construction** - Select vertices that cover the most uncovered vertices
2. **Redundancy removal** - Remove vertices that aren't essential
3. **Local search** - Try 2-for-1 swaps to improve the solution

Here's my complete solution:

```rust
use std::collections::HashSet;
use std::io::{self, BufRead, Write, BufWriter};
use std::time::Instant;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();
    let n = parts[0];
    let m = parts[1];

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let edge: Vec<usize> = line
            .split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();
        let u = edge[0];
        let v = edge[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    // Sort adjacency lists for faster lookups
    for list in &mut adj {
        list.sort();
        list.dedup();
    }

    let solution = solve_mds(&adj);

    writeln!(out, "{}", solution.len()).unwrap();
    if !solution.is_empty() {
        let vertices: Vec<String> = solution.iter().map(|v| v.to_string()).collect();
        writeln!(out, "{}", vertices.join(" ")).unwrap();
    }
}

fn solve_mds(adj: &[Vec<usize>]) -> Vec<usize> {
    let n = adj.len();
    if n == 0 {
        return vec![];
    }

    let start = Instant::now();
    let time_limit = std::time::Duration::from_secs(270);

    // Try multiple strategies and keep the best
    let mut best_solution: Vec<usize> = Vec::new();

    // Strategy 1: Coverage-based greedy
    let mut sol1 = greedy_by_coverage(adj);
    remove_redundant(&mut sol1, adj);
    best_solution = sol1.iter().cloned().collect();

    // Strategy 2: Degree-based greedy (good for certain graph types)
    if start.elapsed() < time_limit {
        let mut sol2 = greedy_by_degree(adj);
        remove_redundant(&mut sol2, adj);
        if sol2.len() < best_solution.len() {
            best_solution = sol2.iter().cloned().collect();
        }
    }

    // Strategy 3: Hybrid greedy with different tie-breaking
    if start.elapsed() < time_limit && n <= 50000 {
        let mut sol3 = greedy_hybrid(adj);
        remove_redundant(&mut sol3, adj);
        if sol3.len() < best_solution.len() {
            best_solution = sol3.iter().cloned().collect();
        }
    }

    // Local search on the best solution
    if start.elapsed() < time_limit {
        let mut best_set: HashSet<usize> = best_solution.iter().cloned().collect();
        local_search(&mut best_set, adj, start, time_limit);
        best_solution = best_set.iter().cloned().collect();
    }

    best_solution.sort();
    best_solution
}

fn greedy_by_coverage(adj: &[Vec<usize>]) -> HashSet<usize> {
    let n = adj.len();
    let mut solution: HashSet<usize> = HashSet::new();
    let mut dominated: Vec<bool> = vec![false; n];
    let mut undominated_count = n;

    while undominated_count > 0 {
        let mut best_v = 0;
        let mut best_gain = 0;
        let mut best_deg = 0;

        for v in 0..n {
            if solution.contains(&v) {
                continue;
            }

            let mut gain = 0;
            if !dominated[v] {
                gain += 1;
            }
            for &u in &adj[v] {
                if !dominated[u] {
                    gain += 1;
                }
            }

            // Prefer higher gain, use degree as tie-breaker
            if gain > best_gain
                || (gain == best_gain && gain > 0 && adj[v].len() > best_deg)
            {
                best_gain = gain;
                best_v = v;
                best_deg = adj[v].len();
            }
        }

        if best_gain == 0 {
            break;
        }

        solution.insert(best_v);
        if !dominated[best_v] {
            dominated[best_v] = true;
            undominated_count -= 1;
        }
        for &u in &adj[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                undominated_count -= 1;
            }
        }
    }

    solution
}

fn greedy_by_degree(adj: &[Vec<usize>]) -> HashSet<usize> {
    let n = adj.len();

    // Sort vertices by degree (descending)
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));

    let mut solution: HashSet<usize> = HashSet::new();
    let mut dominated: Vec<bool> = vec![false; n];

    for &v in &order {
        // Check if v or any neighbor is undominated
        let needs_adding = !dominated[v]
            || adj[v].iter().any(|&u| !dominated[u]);

        if needs_adding {
            solution.insert(v);
            dominated[v] = true;
            for &u in &adj[v] {
                dominated[u] = true;
            }
        }
    }

    solution
}

fn greedy_hybrid(adj: &[Vec<usize>]) -> HashSet<usize> {
    let n = adj.len();
    let mut solution: HashSet<usize> = HashSet::new();
    let mut dominated: Vec<bool> = vec![false; n];
    let mut undominated_count = n;

    while undominated_count > 0 {
        let mut best_v = 0;
        let mut best_score = f64::MIN;

        for v in 0..n {
            if solution.contains(&v) {
                continue;
            }

            let mut uncovered = 0;
            if !dominated[v] {
                uncovered += 1;
            }
            for &u in &adj[v] {
                if !dominated[u] {
                    uncovered += 1;
                }
            }

            if uncovered == 0 {
                continue;
            }

            // Score: coverage efficiency
            let score = uncovered as f64 / (1.0 + adj[v].len() as f64).sqrt();

            if score > best_score {
                best_score = score;
                best_v = v;
            }
        }

        if best_score == f64::MIN {
            break;
        }

        solution.insert(best_v);
        if !dominated[best_v] {
            dominated[best_v] = true;
            undominated_count -= 1;
        }
        for &u in &adj[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                undominated_count -= 1;
            }
        }
    }

    solution
}

fn remove_redundant(solution: &mut HashSet<usize>, adj: &[Vec<usize>]) {
    let mut changed = true;

    while changed {
        changed = false;
        let vertices: Vec<usize> = solution.iter().cloned().collect();

        for v in vertices {
            if !solution.contains(&v) {
                continue;
            }

            solution.remove(&v);

            // Check if v is still dominated
            let v_dominated = adj[v].iter().any(|&u| solution.contains(&u));

            if !v_dominated {
                solution.insert(v);
                continue;
            }

            // Check if all neighbors of v are still dominated
            let mut all_neighbors_ok = true;
            for &u in &adj[v] {
                if solution.contains(&u) {
                    continue;
                }
                let u_dominated = adj[u].iter().any(|&w| solution.contains(&w));
                if !u_dominated {
                    all_neighbors_ok = false;
                    break;
                }
            }

            if all_neighbors_ok {
                changed = true;
            } else {
                solution.insert(v);
            }
        }
    }
}

fn local_search(
    solution: &mut HashSet<usize>,
    adj: &[Vec<usize>],
    start: Instant,
    time_limit: std::time::Duration,
) {
    let n = adj.len();
    if solution.len() < 2 {
        return;
    }

    let mut improved = true;
    let mut iteration = 0;
    let max_iterations = 500;

    while improved && iteration < max_iterations && start.elapsed() < time_limit {
        improved = false;
        iteration += 1;

        let sol_vec: Vec<usize> = solution.iter().cloned().collect();
        let sol_size = sol_vec.len();

        // Try 2-for-1 swaps: remove 2 vertices, add 1
        'outer: for i in 0..sol_size {
            if start.elapsed() >= time_limit {
                break;
            }

            for j in (i + 1)..sol_size {
                let v1 = sol_vec[i];
                let v2 = sol_vec[j];

                solution.remove(&v1);
                solution.remove(&v2);

                // Find uncovered vertices
                let uncovered = find_uncovered(solution, adj, v1, v2);

                if uncovered.is_empty() {
                    // Pure improvement: removed 2, added 0
                    improved = true;
                    remove_redundant(solution, adj);
                    break 'outer;
                }

                // Try to cover all uncovered with a single vertex
                let mut found_swap = false;
                for u in 0..n {
                    if solution.contains(&u) {
                        continue;
                    }

                    let covers_all = uncovered.iter().all(|&x| {
                        x == u || adj[u].binary_search(&x).is_ok()
                    });

                    if covers_all {
                        solution.insert(u);
                        improved = true;
                        found_swap = true;
                        remove_redundant(solution, adj);
                        break 'outer;
                    }
                }

                if !found_swap {
                    solution.insert(v1);
                    solution.insert(v2);
                }
            }
        }

        // Try 1-for-0 (just removal) after improvements
        if improved {
            remove_redundant(solution, adj);
        }
    }
}

fn find_uncovered(
    solution: &HashSet<usize>,
    adj: &[Vec<usize>],
    v1: usize,
    v2: usize,
) -> Vec<usize> {
    let mut uncovered = Vec::new();

    // Check v1 and v2 themselves
    for &v in &[v1, v2] {
        if !solution.contains(&v) {
            let dominated = adj[v].iter().any(|&u| solution.contains(&u));
            if !dominated && !uncovered.contains(&v) {
                uncovered.push(v);
            }
        }
    }

    // Check neighbors of v1 and v2
    for &v in &[v1, v2] {
        for &u in &adj[v] {
            if solution.contains(&u) || uncovered.contains(&u) {
                continue;
            }
            let dominated = adj[u].iter().any(|&w| solution.contains(&w));
            if !dominated {
                uncovered.push(u);
            }
        }
    }

    uncovered
}
```

This solution implements:

1. **Multiple greedy strategies** with different selection criteria (coverage-based, degree-based, and hybrid)
2. **Efficient redundancy removal** that iteratively removes non-essential vertices
3. **Local search with 2-for-1 swaps** to improve solutions when time permits
4. **Time management** to stay within the 5-minute limit

The algorithm should produce good dominating sets for a wide variety of graph types and sizes.
