```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Light direction (surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // View direction (surface to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    
    // Ambient component
    float3 ambient = float3(0.1, 0.05, 0.05);
    
    // Diffuse component
    float NdotL = max(dot(N, L), 0.0);
    float3 diffuse = input.color * NdotL;
    
    // Specular component
    float3 R = reflect(-L, N);
    float RdotV = max(dot(R, V), 0.0);
    float3 specular = float3(1.0, 1.0, 1.0) * pow(RdotV, 32.0);
    
    // Combine and clamp
    float3 finalColor = clamp(ambient + diffuse + specular, 0.0, 1.0);
    
    return float4(finalColor, 1.0);
}
```
