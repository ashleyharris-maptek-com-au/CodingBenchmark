```spirv
; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler; 0
; Bound: 128
; Schema: 0
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpSource GLSL 450

; Decorations for Input Blocks
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2

; Decorations for Output Blocks
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; Constants
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_02 = OpConstant %float 0.02
%float_0_333 = OpConstant %float 0.333333
%float_0_666 = OpConstant %float 0.666666

; Input Types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4

; Output Types
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out

; Pointer Types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

; Helper function to emit a vertex
; Args: Pos (v4float), Color (v3float)
%func_emit = OpTypeFunction %void %v4float %v3float
%emit_vertex = OpFunction %void None %func_emit
%param_pos = OpFunctionParameter %v4float
%param_col = OpFunctionParameter %v3float
%lbl_emit = OpLabel
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %param_pos
OpStore %outColor %param_col
OpEmitVertex
OpReturn
OpFunctionEnd

%main = OpFunction %void None %3
%5 = OpLabel

; --- Load Inputs ---
; Positions
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %ptr_p3

; Colors
%ptr_c1 = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %ptr_c2

; --- Compute Tangents and Normals ---
; T1 = normalize(p2 - p0)
%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%t1_vec = OpFSub %v3float %p2_xyz %p0_xyz
%t1 = OpNormalize %v3float %t1_vec

; T2 = normalize(p3 - p1)
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2
%t2_vec = OpFSub %v3float %p3_xyz %p1_xyz
%t2 = OpNormalize %v3float %t2_vec

; N1 = vec3(-T1.y, T1.x, 0) * thickness
%t1_y = OpCompositeExtract %float %t1 1
%t1_x = OpCompositeExtract %float %t1 0
%n1_x = OpFNegate %float %t1_y
%n1_unscaled = OpCompositeConstruct %v3float %n1_x %t1_x %float_0
%n1 = OpVectorTimesScalar %v3float %n1_unscaled %float_0_02

; N2 = vec3(-T2.y, T2.x, 0) * thickness
%t2_y = OpCompositeExtract %float %t2 1
%t2_x = OpCompositeExtract %float %t2 0
%n2_x = OpFNegate %float %t2_y
%n2_unscaled = OpCompositeConstruct %v3float %n2_x %t2_x %float_0
%n2 = OpVectorTimesScalar %v3float %n2_unscaled %float_0_02

; --- Compute Interpolated Properties ---
; Interpolate Normals
%nA = OpFMix %v3float %n1 %n2 %float_0_333
%nB = OpFMix %v3float %n1 %n2 %float_0_666

; Interpolate Colors
%cA = OpFMix %v3float %c1 %c2 %float_0_333
%cB = OpFMix %v3float %c1 %c2 %float_0_666

; Interpolate Positions (for split points)
%pA = OpFMix %v4float %p1 %p2 %float_0_333
%pB = OpFMix %v4float %p1 %p2 %float_0_666

; --- Prepare Normal Offsets (vec4) ---
%zero_v4 = OpCompositeConstruct %v4float %float_0 %float_0 %float_0 %float_0
; Shuffle vec3 normal into vec4 (xyz, 0)
%n1_v4 = OpVectorShuffle %v4float %n1 %zero_v4 0 1 2 4
%n2_v4 = OpVectorShuffle %v4float %n2 %zero_v4 0 1 2 4
%nA_v4 = OpVectorShuffle %v4float %nA %zero_v4 0 1 2 4
%nB_v4 = OpVectorShuffle %v4float %nB %zero_v4 0 1 2 4

; --- Emit Quad 1 (Start -> A) ---
; V0: p1 + n1
%pos_v0 = OpFAdd %v4float %p1 %n1_v4
%call0 = OpFunctionCall %void %emit_vertex %pos_v0 %c1

; V1: p1 - n1
%pos_v1 = OpFSub %v4float %p1 %n1_v4
%call1 = OpFunctionCall %void %emit_vertex %pos_v1 %c1

; V2: pA + nA
%pos_v2 = OpFAdd %v4float %pA %nA_v4
%call2 = OpFunctionCall %void %emit_vertex %pos_v2 %cA

; V3: pA - nA
%pos_v3 = OpFSub %v4float %pA %nA_v4
%call3 = OpFunctionCall %void %emit_vertex %pos_v3 %cA

OpEndPrimitive

; --- Emit Quad 2 (B -> End) ---
; V0: pB + nB
%pos_v4 = OpFAdd %v4float %pB %nB_v4
%call4 = OpFunctionCall %void %emit_vertex %pos_v4 %cB

; V1: pB - nB
%pos_v5 = OpFSub %v4float %pB %nB_v4
%call5 = OpFunctionCall %void %emit_vertex %pos_v5 %cB

; V2: p2 + n2
%pos_v6 = OpFAdd %v4float %p2 %n2_v4
%call6 = OpFunctionCall %void %emit_vertex %pos_v6 %c2

; V3: p2 - n2
%pos_v7 = OpFSub %v4float %p2 %n2_v4
%call7 = OpFunctionCall %void %emit_vertex %pos_v7 %c2

OpEndPrimitive

OpReturn
OpFunctionEnd
```
