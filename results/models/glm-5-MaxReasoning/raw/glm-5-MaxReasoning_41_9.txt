spirv
; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 100

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; UBO Struct Definition
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; Pointer Types
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2
%ptr_Output_vec4 = OpTypePointer Output %vec4
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; Variables
%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal = OpVariable %ptr_Input_vec3 Input
%in_uv = OpVariable %ptr_Input_vec2 Input
%outColor = OpVariable %ptr_Output_vec4 Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

; Constants
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_8 = OpConstant %float 8
%int_3 = OpConstant %int 3

; Extended Instruction Set
%glsl = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %func
%entry = OpLabel

; --- Load Inputs ---
%uv = OpLoad %vec2 %in_uv
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal

; --- UV Checkerboard Logic ---
%uv_scaled = OpVectorTimesScalar %vec2 %uv %float_8
%uv_floor = OpExtInst %vec2 %glsl Floor %uv_scaled
%uv_x = OpCompositeExtract %float %uv_floor 0
%uv_y = OpCompositeExtract %float %uv_floor 1
%sum = OpFAdd %float %uv_x %uv_y
%mod_val = OpExtInst %float %glsl FMod %sum %float_2
%is_even = OpFOrdLessThan %bool %mod_val %float_1
%checker_col = OpSelect %float %is_even %float_0_9 %float_0_2

; --- Lambertian Shading ---
%lightPos_ptr = OpAccessChain %ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%lightPos_xyz = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2

%N = OpExtInst %vec3 %glsl Normalize %normal
%lightDir = OpFSub %vec3 %lightPos_xyz %worldPos
%L = OpExtInst %vec3 %glsl Normalize %lightDir

%dot_NL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %dot_NL %float_0

; Shading Factor = 0.5 + 0.5 * diffuse
%diff_scaled = OpFMul %float %diffuse %float_0_5
%intensity = OpFAdd %float %float_0_5 %diff_scaled

; --- Combine and Output ---
%final_val = OpFMul %float %checker_col %intensity
%out_rgb = OpCompositeConstruct %vec3 %final_val %final_val %final_val
%final_color = OpCompositeConstruct %vec4 %out_rgb %float_1

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
