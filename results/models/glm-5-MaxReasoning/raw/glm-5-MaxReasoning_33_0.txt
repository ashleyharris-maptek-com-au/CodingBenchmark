using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.Text;

namespace QAPSolver
{
    class Program
    {
        static void Main(string[] args)
        {
            // 1. Read Input
            var sr = new InputReader();
            int n = sr.ReadInt();
            if (n == 0) return;

            long[][] flow = new long[n][];
            long[][] dist = new long[n][];

            for (int i = 0; i < n; i++)
            {
                flow[i] = new long[n];
                for (int j = 0; j < n; j++)
                    flow[i][j] = sr.ReadLong();
            }

            for (int i = 0; i < n; i++)
            {
                dist[i] = new long[n];
                for (int j = 0; j < n; j++)
                    dist[i][j] = sr.ReadLong();
            }

            // 2. Solve
            var solver = new QAPSolverHeuristic(n, flow, dist);
            var (cost, p) = solver.Solve();

            // 3. Output
            Console.WriteLine(cost);
            Console.WriteLine(string.Join(" ", p));
        }
    }

    class QAPSolverHeuristic
    {
        private int N;
        private long[][] Flow;
        private long[][] Dist;
        private Random Rand = new Random(12345);
        private Stopwatch Sw = new Stopwatch();
        private const long TimeLimitMs = 290000; // 290s for 5min limit

        public QAPSolverHeuristic(int n, long[][] flow, long[][] dist)
        {
            N = n;
            Flow = flow;
            Dist = dist;
        }

        public (long cost, int[] p) Solve()
        {
            Sw.Start();
            
            // Initial solution using Greedy heuristic
            int[] p = GreedyInitialize();
            long cost = CalculateCost(p);
            
            int[] bestP = (int[])p.Clone();
            long bestCost = cost;

            // Simulated Annealing Parameters
            // We estimate initial temperature based on cost magnitude or delta
            double t_start = EstimateStartTemp(p, cost);
            double t_end = 1e-9;
            double alpha = 0.999; // Cooling rate

            // If N is small, we can iterate many times. If N is large, iterations take longer.
            // We adapt iterations per step based on elapsed time.
            long iterPerStep = N * 2; 
            
            // Used to check time periodically
            long totalIter = 0;
            long checkInterval = 1000;

            while (Sw.ElapsedMilliseconds < TimeLimitMs)
            {
                double t = t_start;
                
                // Run SA for a chunk of time or iterations
                // We integrate the SA loop with time checking
                for (long i = 0; i < iterPerStep; i++)
                {
                    totalIter++;
                    
                    // Pick two distinct facilities to swap
                    int u = Rand.Next(N);
                    int v = Rand.Next(N);
                    while (u == v) v = Rand.Next(N);

                    long delta = CalculateSwapDelta(p, u, v);

                    if (delta < 0)
                    {
                        // Accept improvement
                        Swap(p, u, v);
                        cost += delta;
                        if (cost < bestCost)
                        {
                            bestCost = cost;
                            bestP = (int[])p.Clone();
                        }
                    }
                    else
                    {
                        // Accept worsening move with probability
                        if (t > 1e-9)
                        {
                            double prob = Math.Exp(-delta / t);
                            if (Rand.NextDouble() < prob)
                            {
                                Swap(p, u, v);
                                cost += delta;
                            }
                        }
                    }
                }

                // Cool down
                t_start *= alpha;
                
                // Periodically check time and adapt
                if (totalIter % checkInterval == 0)
                {
                    if (Sw.ElapsedMilliseconds >= TimeLimitMs) break;
                    
                    // If we are cooling too fast or too slow?
                    // Just keep cooling geometrically.
                }
            }

            return (bestCost, bestP);
        }

        private int[] GreedyInitialize()
        {
            // Heuristic: Match high flow facilities to low distance locations.
            // Indices: 0..N-1
            var flowSums = new long[N];
            var distSums = new long[N];

            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    flowSums[i] += Flow[i][j];
                    distSums[i] += Dist[i][j];
                }
            }

            // Facilities sorted by total flow descending
            var sortedFacilities = Enumerable.Range(0, N).OrderByDescending(i => flowSums[i]).ToArray();
            // Locations sorted by total distance ascending
            var sortedLocations = Enumerable.Range(0, N).OrderBy(i => distSums[i]).ToArray();

            int[] p = new int[N];
            for (int i = 0; i < N; i++)
            {
                p[sortedFacilities[i]] = sortedLocations[i];
            }
            return p;
        }

        private long CalculateCost(int[] p)
        {
            long cost = 0;
            for (int i = 0; i < N; i++)
            {
                for (int j = 0; j < N; j++)
                {
                    cost += Flow[i][j] * Dist[p[i]][p[j]];
                }
            }
            return cost;
        }

        // Calculates the change in cost if we swap the locations of facility u and v
        private long CalculateSwapDelta(int[] p, int u, int v)
        {
            long delta = 0;
            int pu = p[u];
            int pv = p[v];

            // We use row/column access for speed.
            var flowU = Flow[u];
            var flowV = Flow[v];
            var flowColU = Flow; // Need column access
            var flowColV = Flow; 

            var distPu = Dist[pu];
            var distPv = Dist[pv];
            var distColPu = Dist; // Column access for Dist
            var distColPv = Dist;

            for (int k = 0; k < N; k++)
            {
                if (k == u || k == v) continue;

                int pk = p[k];
                
                // Row contributions for u
                // Old: flow[u][k] * dist[pu][pk]
                // New: flow[u][k] * dist[pv][pk]
                delta += flowU[k] * (distPv[pk] - distPu[pk]);

                // Row contributions for v
                // Old: flow[v][k] * dist[pv][pk]
                // New: flow[v][k] * dist[pu][pk]
                delta += flowV[k] * (distPu[pk] - distPv[pk]);

                // Column contributions for u (k acts as row index)
                // Old: flow[k][u] * dist[pk][pu]
                // New: flow[k][u] * dist[pk][pv]
                delta += flowColU[k][u] * (distColPu[pk][pv] - distColPu[pk][pu]);

                // Column contributions for v
                // Old: flow[k][v] * dist[pk][pv]
                // New: flow[k][v] * dist[pk][pu]
                delta += flowColV[k][v] * (distColPv[pk][pu] - distColPv[pk][pv]);
            }

            // Interaction between u and v themselves
            // Old: flow[u][v]*dist[pu][pv] + flow[v][u]*dist[pv][pu]
            // New: flow[u][v]*dist[pv][pu] + flow[v][u]*dist[pu][pv]
            delta += flowU[v] * (distPv[pu] - distPu[pv]);
            delta += flowV[u] * (distPu[pv] - distPv[pu]);

            return delta;
        }

        private void Swap(int[] p, int u, int v)
        {
            int temp = p[u];
            p[u] = p[v];
            p[v] = temp;
        }

        private double EstimateStartTemp(int[] p, long cost)
        {
            // Sample random deltas to set temperature
            long sumDelta = 0;
            int samples = Math.Min(N, 50);
            int count = 0;
            
            // Create a temporary copy to test swaps
            int[] tempP = (int[])p.Clone();

            for (int i = 0; i < samples; i++)
            {
                int u = Rand.Next(N);
                int v = Rand.Next(N);
                if (u == v) continue;
                
                long d = CalculateSwapDelta(tempP, u, v);
                if (d > 0)
                {
                    sumDelta += d;
                    count++;
                }
            }

            if (count == 0) return 1000.0; // Fallback
            double avgDelta = (double)sumDelta / count;
            
            // We want initial acceptance probability of bad moves to be roughly 0.5 to 0.8
            // P = exp(-delta/T) => T = -delta / ln(P)
            // Let's target P ~ 0.5, so T ~ delta / 0.69
            return avgDelta * 1.5; 
        }
    }

    // Fast Input Reader
    class InputReader
    {
        private System.IO.Stream stream;
        private byte[] buffer = new byte[1 << 20];
        private int curChar;
        private int numChars;

        public InputReader()
        {
            stream = Console.OpenStandardInput();
        }

        private int ReadByte()
        {
            if (curChar >= numChars)
            {
                curChar = 0;
                numChars = stream.Read(buffer, 0, buffer.Length);
                if (numChars <= 0) return -1;
            }
            return buffer[curChar++];
        }

        public int ReadInt()
        {
            int c = ReadByte();
            while (c != -1 && (c < '0' || c > '9') && c != '-') c = ReadByte();
            if (c == -1) return 0; // EOF

            bool neg = false;
            if (c == '-')
            {
                neg = true;
                c = ReadByte();
            }

            int val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return neg ? -val : val;
        }

        public long ReadLong()
        {
            int c = ReadByte();
            while (c != -1 && (c < '0' || c > '9') && c != '-') c = ReadByte();
            if (c == -1) return 0;

            bool neg = false;
            if (c == '-')
            {
                neg = true;
                c = ReadByte();
            }

            long val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return neg ? -val : val;
        }
    }
}
