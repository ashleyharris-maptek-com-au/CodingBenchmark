```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(location = 0) out vec4 outColor;

void main() {
    // Map UVs to the complex plane c
    // uv.x in [0, 1] -> [-2, 1]
    // uv.y in [0, 1] -> [-1, 1]
    vec2 c = vec2(uv.x * 3.0 - 2.0, uv.y * 2.0 - 1.0);
    
    vec2 z = vec2(0.0);
    int iterations = 0;
    
    // Iterate up to 20 steps
    for (int i = 0; i < 20; i++) {
        // z = z^2 + c
        z = vec2(z.x * z.x - z.y * z.y, 2.0 * z.x * z.y) + c;
        iterations++;
        
        // Escape condition: dot(z, z) > 4
        if (dot(z, z) > 4.0) {
            break;
        }
    }
    
    // Calculate color t = iterations / 20
    float t = float(iterations) / 20.0;
    
    // Output color (t, t*0.5, 1.0-t, 1.0)
    outColor = vec4(t, t * 0.5, 1.0 - t, 1.0);
}
```
