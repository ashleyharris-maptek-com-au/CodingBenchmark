    public class LambertSolver
    {
        private const double TOLERANCE = 1e-12;
        private const int MAX_ITER = 100;

        public static (Vec3 v1, Vec3 v2, bool valid) Solve(Vec3 r1, Vec3 r2, double tof, double mu, bool prograde = true)
        {
            double r1Mag = r1.Magnitude;
            double r2Mag = r2.Magnitude;
            
            if (r1Mag < 1e-10 || r2Mag < 1e-10 || tof <= 0 || mu <= 0)
                return (Vec3.Zero, Vec3.Zero, false);

            Vec3 r1Norm = r1 / r1Mag;
            Vec3 r2Norm = r2 / r2Mag;
            
            double cosDnu = Math.Min(1.0, Math.Max(-1.0, r1Norm.Dot(r2Norm)));
            double sinDnu = Math.Sqrt(Math.Max(0, 1 - cosDnu * cosDnu));
            
            Vec3 hDir = r1.Cross(r2);
            double dnu = Math.Atan2(sinDnu, cosDnu);
            
            if (prograde)
            {
                if (hDir.Z < 0) dnu = 2 * Math.PI - dnu;
            }
            else
            {
                if (hDir.Z >= 0) dnu = 2 * Math.PI - dnu;
            }

            double c = Math.Sqrt(r1Mag * r1Mag + r2Mag * r2Mag - 2 * r1Mag * r2Mag * cosDnu);
            double s = (r1Mag + r2Mag + c) / 2;
            
            double aMin = s / 2;
            
            double lambda2 = r1Mag * r2Mag * (1 - cosDnu) / (s * s);
            if (lambda2 > 1) lambda2 = 1;
            double lambda = Math.Sqrt(lambda2);
            
            double T = Math.Sqrt(8 * s * s * s / mu);
            double Tstar = tof / T * Math.Sqrt(mu / (s * s * s)) * Math.Sqrt(8 * s * s * s / mu);
            Tstar = tof / Math.Sqrt(s * s * s / mu);

            double x = 0.5;
            double y = 0.0;
            
            for (int iter = 0; iter < MAX_ITER; iter++)
            {
                double sinh = Math.Sinh(x);
                double cosh = Math.Cosh(x);
                
                y = Math.Sqrt(lambda * lambda * (x * x - 1) + 1);
                
                double Tcalc = (x - y * sinh + lambda * cosh) / (x * x);
                
                if (Math.Abs(Tcalc - Tstar * Math.Sqrt(s * s * s / mu) / T) < TOLERANCE)
                    break;
                
                double dTdx = (1 - y * cosh + lambda * sinh) / (x * x) - 2 * (x - y * sinh + lambda * cosh) / (x * x * x);
                
                if (Math.Abs(dTdx) < 1e-20) break;
                
                x = x - (Tcalc - Tstar * Math.Sqrt(s * s * s / mu) / T) / dTdx;
                
                if (double.IsNaN(x) || double.IsInfinity(x))
                {
                    x = 0.5 + iter * 0.1;
                }
            }

            double a = s / (2 * (1 - x * x));
            if (a < 0) a = -a;
            
            double alpha, beta;
            double cosAlpha = 1 - r1Mag / a;
            double cosBeta = 1 - r2Mag / a;
            
            if (x < 1)
            {
                alpha = 2 * Math.Acos(Math.Max(-1, Math.Min(1, cosAlpha)));
                beta = 2 * Math.Acos(Math.Max(-1, Math.Min(1, cosBeta)));
            }
            else
            {
                alpha = 2 * Math.Acosh(Math.Max(1, cosAlpha));
                beta = 2 * Math.Acosh(Math.Max(1, cosBeta));
            }

            if (dnu > Math.PI)
            {
                alpha = 2 * Math.PI - alpha;
            }

            double p = 4 * a * Math.Sin(dnu / 2) * Math.Sin(dnu / 2) / c;
            if (p < 0) p = -p;
            
            double f = 1 - r2Mag / p * (1 - Math.Cos(dnu));
            double g = r1Mag * r2Mag * Math.Sin(dnu) / Math.Sqrt(mu * p);
            double gDot = 1 - r1Mag / p * (1 - Math.Cos(dnu));

            Vec3 v1 = (r2 - f * r1) / g;
            Vec3 v2 = (gDot * r2 - r1) / g;

            return (v1, v2, true);
        }

        public static (Vec3 v1, Vec3 v2, bool valid) SolveUniversal(Vec3 r1, Vec3 r2, double tof, double mu)
        {
            double r1Mag = r1.Magnitude;
            double r2Mag = r2.Magnitude;
            
            if (r1Mag < 1e-10 || r2Mag < 1e-10 || tof <= 0)
                return (Vec3.Zero, Vec3.Zero, false);

            double cosDnu = r1.Dot(r2) / (r1Mag * r2Mag);
            cosDnu = Math.Max(-1.0, Math.Min(1.0, cosDnu));
            
            Vec3 cross = r1.Cross(r2);
            double sinDnu = cross.Magnitude / (r1Mag * r2Mag);
            
            double dnu = Math.Atan2(sinDnu, cosDnu);
            if (cross.Z < 0) dnu = 2 * Math.PI - dnu;

            double A = Math.Sin(dnu) * Math.Sqrt(r1Mag * r2Mag / (1 - cosDnu));
            if (Math.Abs(A) < 1e-10)
                A = 1e-10;

            double c = Math.Sqrt(r1Mag * r1Mag + r2Mag * r2Mag - 2 * r1Mag * r2Mag * cosDnu);
            double s = (r1Mag + r2Mag + c) / 2;

            double z = 0.5;
            
            for (int iter = 0; iter < MAX_ITER; iter++)
            {
                double y = r1Mag + r2Mag + A * (z * z - 1) / z;
                if (y < 0) y = 0.1;
                
                double F, Fp, G, Gp;
                StumpffFunctions(z, out F, out Fp, out G, out Gp);
                
                double tCalc = Math.Pow(y / c, 1.5) * (G + A * Math.Sqrt(y));
                
                if (Math.Abs(tCalc - tof) < TOLERANCE * tof)
                    break;

                double dtdz = Math.Pow(y / c, 1.5) * (Gp + A * 0.5 / Math.Sqrt(y) * (G + A * Math.Sqrt(y)));
                
                if (Math.Abs(dtdz) < 1e-20) break;
                
                z = z - (tCalc - tof) / dtdz;
                
                if (double.IsNaN(z) || double.IsInfinity(z))
                    z = 0.5;
            }

            double yFinal = r1Mag + r2Mag + A * (z * z - 1) / z;
            if (yFinal < 0) yFinal = s;

            double f, fp, g, gp;
            StumpffFunctions(z, out f, out fp, out g, out gp);

            double L = Math.Pow(yFinal / c, 1.5);
            
            double fVal = 1 - yFinal / r1Mag;
            double gVal = A * Math.Sqrt(yFinal / mu);
            double gDot = 1 - yFinal / r2Mag;

            Vec3 v1 = (r2 - r1 * fVal) / gVal;
            Vec3 v2 = (r2 * gDot - r1) / gVal;

            return (v1, v2, true);
        }

        private static void StumpffFunctions(double z, out double f, out double fp, out double g, out double gp)
        {
            if (Math.Abs(z) < 1e-10)
            {
                f = 1.0 - z / 24.0;
                fp = -1.0 / 6.0 + z / 120.0;
                g = 1.0 / 2.0 - z / 24.0;
                gp = 1.0 / 6.0 - z / 120.0;
            }
            else if (z > 0)
            {
                double sz = Math.Sqrt(z);
                f = (1.0 - Math.Cos(sz)) / z;
                fp = (Math.Sin(sz) / sz - Math.Cos(sz)) / (2.0 * z);
                g = (sz - Math.Sin(sz)) / (sz * sz * sz);
                gp = (1.0 - Math.Sin(sz) / sz) / (2.0 * z);
            }
            else
            {
                double sz = Math.Sqrt(-z);
                f = (Math.Cosh(sz) - 1.0) / (-z);
                fp = (Math.Sinh(sz) / sz - Math.Cosh(sz)) / (2.0 * z);
                g = (Math.Sinh(sz) - sz) / (sz * sz * sz);
                gp = (1.0 - Math.Sinh(sz) / sz) / (2.0 * z);
            }
        }
    }

    public class NBodyPropagator
    {
        private const double G = 6.67430e-11;
        private const double DAY_SEC = 86400.0;

        private List<Body> bodies;
        private double[] masses;
        private string[] names;
        private int n;

        public NBodyPropagator(List<Body> bodyList)
        {
            bodies = bodyList;
            n = bodies.Count;
            masses = new double[n];
            names = new string[n];
            for (int i = 0; i < n; i++)
            {
                masses[i] = bodies[i].Mass;
                names[i] = bodies[i].Name;
            }
        }

        public Dictionary<string, State> Propagate(double days, double dtDays = 0.125)
        {
            double[] pos = new double[n * 3];
            double[] vel = new double[n * 3];
            
            for (int i = 0; i < n; i++)
            {
                pos[i * 3 + 0] = bodies[i].Position.X;
                pos[i * 3 + 1] = bodies[i].Position.Y;
                pos[i * 3 + 2] = bodies[i].Position.Z;
                vel[i * 3 + 0] = bodies[i].Velocity.X;
                vel[i * 3 + 1] = bodies[i].Velocity.Y;
                vel[i * 3 + 2] = bodies[i].Velocity.Z;
            }

            double t = 0;
            double dt = dtDays * DAY_SEC;
            double tEnd = days * DAY_SEC;
            
            int steps = (int)(tEnd / dt) + 1;
            dt = tEnd / steps;

            for (int step = 0; step < steps; step++)
            {
                double[] acc = ComputeAccelerations(pos);
                
                for (int i = 0; i < n * 3; i++)
                {
                    vel[i] += acc[i] * dt * 0.5;
                }
                
                for (int i = 0; i < n * 3; i++)
                {
                    pos[i] += vel[i] * dt;
                }
                
                acc = ComputeAccelerations(pos);
                
                for (int i = 0; i < n * 3; i++)
                {
                    vel[i] += acc[i] * dt * 0.5;
                }
                
                t += dt;
            }

            var result = new Dictionary<string, State>();
            for (int i = 0; i < n; i++)
            {
                result[names[i]] = new State(
                    new Vec3(pos[i * 3 + 0], pos[i * 3 + 1], pos[i * 3 + 2]),
                    new Vec3(vel[i * 3 + 0], vel[i * 3 + 1], vel[i * 3 + 2])
                );
            }
            return result;
        }

        private double[] ComputeAccelerations(double[] pos)
        {
            double[] acc = new double[n * 3];
            
            for (int i = 0; i < n; i++)
            {
                double ax = 0, ay = 0, az = 0;
                double xi = pos[i * 3 + 0], yi = pos[i * 3 + 1], zi = pos[i * 3 + 2];
                
                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    
                    double dx = pos[j * 3 + 0] - xi;
                    double dy = pos[j * 3 + 1] - yi;
                    double dz = pos[j * 3 + 2] - zi;
                    double r2 = dx * dx + dy * dy + dz * dz;
                    double r = Math.Sqrt(r2);
                    double r3 = r2 * r;
                    
                    if (r3 > 1e-10)
                    {
                        double factor = G * masses[j] / r3;
                        ax += factor * dx;
                        ay += factor * dy;
                        az += factor * dz;
                    }
                }
                
                acc[i * 3 + 0] = ax;
                acc[i * 3 + 1] = ay;
                acc[i * 3 + 2] = az;
            }
            
            return acc;
        }

        public State PropagateSpacecraft(State scState, double startTime, double endTime, 
            List<Maneuver> maneuvers, string targetBody = "Asteroid")
        {
            double dt = 3600.0;
            int steps = Math.Max(1, (int)((endTime - startTime) * DAY_SEC / dt));
            dt = (endTime - startTime) * DAY_SEC / steps;

            Vec3 pos = scState.Position;
            Vec3 vel = scState.Velocity;
            double t = startTime * DAY_SEC;
            double tEnd = endTime * DAY_SEC;

            int maneuverIdx = 0;
            var sortedManeuvers = maneuvers.OrderBy(m => m.Time).ToList();

            for (int step = 0; step < steps; step++)
            {
                while (maneuverIdx < sortedManeuvers.Count && 
                       sortedManeuvers[maneuverIdx].Time * DAY_SEC <= t + dt * 0.001)
                {
                    vel = vel + sortedManeuvers[maneuverIdx].DeltaV;
                    maneuverIdx++;
                }

                var state = Propagate(t / DAY_SEC, 0);
                Vec3 acc = ComputeSCAcceleration(pos, state);

                vel = vel + acc * (dt * 0.5);
                pos = pos + vel * dt;
                
                state = Propagate((t + dt * 0.5) / DAY_SEC, 0);
                acc = ComputeSCAcceleration(pos, state);
                
                vel = vel + acc * (dt * 0.5);
                t += dt;
            }

            return new State(pos, vel);
        }

        private Vec3 ComputeSCAcceleration(Vec3 scPos, Dictionary<string, State> bodyStates)
        {
            Vec3 acc = Vec3.Zero;
            
            for (int i = 0; i < n; i++)
            {
                Vec3 r = bodyStates[names[i]].Position - scPos;
                double rMag = r.Magnitude;
                if (rMag > 1e-10)
                {
                    acc = acc + r.Normalized * (G * masses[i] / (rMag * rMag));
                }
            }
            
            return acc;
        }
    }

    public class TrajectoryOptimizer
    {
        private const double G = 6.67430e-11;
        private const double DAY_SEC = 86400.0;
        private const double AU = 1.495978707e11;
        private const double SUN_MU = 1.32712440018e20;
        private const double EARTH_MU = 3.986004418e14;
        private const double EARTH_RADIUS = 6.371e6;
        private const double EARTH_ESCAPE_VELOCITY = 11186.0;

        private List<Body> bodies;
        private Dictionary<string, Body> bodyDict;
        private NBodyPropagator propagator;
        private double warningDays;
        private double impactorMass;
        private double deltaVBudget;
        private double asteroidMass;
        private double t0Unix;
        private string de421Path;

        private Body earth, asteroid, sun, moon;

        public TrajectoryOptimizer(List<Body> bodyList, double warnDays, double impMass,
            double dvBudget, double astMass, double t0, string de421)
        {
            bodies = bodyList;
            bodyDict = bodyList.ToDictionary(b => b.Name, b => b);
            propagator = new NBodyPropagator(bodyList);
            warningDays = warnDays;
            impactorMass = impMass;
            deltaVBudget = dvBudget;
            asteroidMass = astMass;
            t0Unix = t0;
            de421Path = de421;

            bodyDict.TryGetValue("Earth", out earth);
            bodyDict.TryGetValue("Asteroid", out asteroid);
            bodyDict.TryGetValue("Sun", out sun);
            bodyDict.TryGetValue("Moon", out moon);
        }

        public List<Maneuver> Solve()
        {
            if (warningDays <= 730)
            {
                return SolveShortWarning();
            }
            else if (warningDays <= 3650)
            {
                return SolveMediumWarning();
            }
            else
            {
                return SolveLongWarning();
            }
        }

        private List<Maneuver> SolveShortWarning()
        {
            var solutions = new List<(double score, List<Maneuver> maneuvers)>();

            int launchSamples = Math.Min(100, Math.Max(20, (int)(warningDays / 10)));
            double dtLaunch = warningDays / launchSamples;

            for (int li = 0; li <= launchSamples; li++)
            {
                double launchTime = li * dtLaunch;
                if (launchTime > warningDays) break;

                var launchState = propagator.Propagate(launchTime);
                var earthState = launchState["Earth"];

                double maxTOF = warningDays - launchTime;
                if (maxTOF < 30) continue;

                int tofSamples = Math.Min(50, (int)(maxTOF / 20));
                double dtTOF = maxTOF / Math.Max(1, tofSamples);

                for (int ti = 1; ti <= tofSamples; ti++)
                {
                    double tof = ti * dtTOF;
                    double arrivalTime = launchTime + tof;

                    var arrivalState = propagator.Propagate(arrivalTime);
                    var asteroidArrival = arrivalState["Asteroid"];

                    var (v1, v2, valid) = LambertSolver.SolveUniversal(
                        earthState.Position, asteroidArrival.Position, tof * DAY_SEC, SUN_MU);

                    if (!valid) continue;

                    Vec3 launchDV = v1 - earthState.Velocity;
                    double launchDVMag = launchDV.Magnitude;

                    if (launchDVMag < deltaVBudget)
                    {
                        var maneuvers = new List<Maneuver>
                        {
                            new Maneuver(launchTime, launchDV)
                        };

                        double correctionTime = launchTime + tof * 0.4;
                        double finalTime = launchTime + tof;

                        var scState = new State(earthState.Position, v1);
                        var midState = propagator.PropagateSpacecraft(scState, launchTime, correctionTime,
                            maneuvers, "Asteroid");
                        var midBodyState = propagator.Propagate(correctionTime);
                        var asteroidMid = midBodyState["Asteroid"];

                        Vec3 correctionDV = EstimateCorrectionDV(midState, asteroidMid, finalTime - correctionTime);
                        double correctionDVMag = correctionDV.Magnitude;

                        if (launchDVMag + correctionDVMag < deltaVBudget * 0.9)
                        {
                            maneuvers.Add(new Maneuver(correctionTime, correctionDV));

                            var preImpactState = propagator.PropagateSpacecraft(midState, correctionTime, finalTime,
                                maneuvers, "Asteroid");
                            var finalBodyState = propagator.Propagate(finalTime);
                            var asteroidFinal = finalBodyState["Asteroid"];

                            Vec3 impactDV = CalculateImpactCorrection(preImpactState, asteroidFinal);
                            maneuvers.Add(new Maneuver(finalTime, impactDV));

                            double totalDV = maneuvers.Sum(m => m.DeltaV.Magnitude);
                            double score = totalDV + tof * 0.1 + launchTime * 0.01;

                            solutions.Add((score, maneuvers));
                        }
                    }
                }
            }

            if (solutions.Count > 0)
            {
                return solutions.OrderBy(s => s.score).First().maneuvers;
            }

            return FindFallbackSolution();
        }

        private List<Maneuver> SolveMediumWarning()
        {
            var directSolution = SolveShortWarning();
            double directDV = directSolution.Sum(m => m.DeltaV.Magnitude);

            if (directDV < deltaVBudget * 0.5)
            {
                return directSolution;
            }

            var gaSolution = TryGravityAssistTrajectory();
            if (gaSolution != null)
            {
                double gaDV = gaSolution.Sum(m => m.DeltaV.Magnitude);
                if (gaDV < directDV)
                {
                    return gaSolution;
                }
            }

            return directSolution;
        }

        private List<Maneuver> SolveLongWarning()
        {
            var solutions = new List<(double score, List<Maneuver> maneuvers)>();

            var directSolution = SolveShortWarning();
            double directDV = directSolution.Sum(m => m.DeltaV.Magnitude);
            if (directDV < deltaVBudget)
            {
                solutions.Add((directDV + directSolution.First().Time * 0.001, directSolution));
            }

            foreach (var gaPlanet in new[] { "Venus", "Mars", "Jupiter" })
            {
                if (!bodyDict.ContainsKey(gaPlanet)) continue;

                var gaSolution = TryMultiGravityAssist(gaPlanet);
                if (gaSolution != null)
                {
                    double gaDV = gaSolution.Sum(m => m.DeltaV.Magnitude);
                    if (gaDV < deltaVBudget)
                    {
                        solutions.Add((gaDV, gaSolution));
                    }
                }
            }

            if (solutions.Count > 0)
            {
                return solutions.OrderBy(s => s.score).First().maneuvers;
            }

            return directSolution;
        }

        private List<Maneuver> TryGravityAssistTrajectory()
        {
            var planets = new[] { "Venus", "Mars" };
            var solutions = new List<List<Maneuver>>();

            foreach (var planet in planets)
            {
                if (!bodyDict.ContainsKey(planet)) continue;

                var solution = TrySingleGA(planet);
                if (solution != null)
                {
                    double totalDV = solution.Sum(m => m.DeltaV.Magnitude);
                    if (totalDV < deltaVBudget)
                    {
                        solutions.Add(solution);
                    }
                }
            }

            if (solutions.Count > 0)
            {
                return solutions.OrderBy(s => s.Sum(m => m.DeltaV.Magnitude)).First();
            }

            return null;
        }

        private List<Maneuver> TrySingleGA(string planetName)
        {
            var planet = bodyDict[planetName];

            double gaTime = warningDays * 0.3;
            var gaState = propagator.Propagate(gaTime);
            var planetGA = gaState[planetName];

            double bestDV = double.MaxValue;
            List<Maneuver> bestManeuvers = null;

            for (double launchTime = 0; launchTime < Math.Min(warningDays * 0.2, 365); launchTime += 20)
            {
                var launchState = propagator.Propagate(launchTime);
                var earthLaunch = launchState["Earth"];

                double tof1 = gaTime - launchTime;
                if (tof1 < 30) continue;

                var (v1, _, valid1) = LambertSolver.SolveUniversal(
                    earthLaunch.Position, planetGA.Position, tof1 * DAY_SEC, SUN_MU);

                if (!valid1) continue;

                Vec3 launchDV = v1 - earthLaunch.Velocity;
                double launchDVMag = launchDV.Magnitude;

                if (launchDVMag > deltaVBudget * 0.4) continue;

                double arrivalTime = warningDays * 0.8;
                var arrivalState = propagator.Propagate(arrivalTime);
                var asteroidArrival = arrivalState["Asteroid"];

                double tof2 = arrivalTime - gaTime;
                if (tof2 < 30) continue;

                var (_, v2, valid2) = LambertSolver.SolveUniversal(
                    planetGA.Position, asteroidArrival.Position, tof2 * DAY_SEC, SUN_MU);

                if (!valid2) continue;

                Vec3 gaDV = CalculateGAPlaneChange(v1, planetGA.Velocity, v2);

                double totalDV = launchDVMag + gaDV.Magnitude;

                if (totalDV < bestDV && totalDV < deltaVBudget)
                {
                    bestDV = totalDV;
                    bestManeuvers = new List<Maneuver>
                    {
                        new Maneuver(launchTime, launchDV),
                        new Maneuver(gaTime, gaDV.Magnitude < 50 ? Vec3.Zero : gaDV),
                        new Maneuver(arrivalTime, Vec3.Zero)
                    };
                }
            }

            return bestManeuvers;
        }

        private List<Maneuver> TryMultiGravityAssist(string primaryPlanet)
        {
            var solutions = new List<List<Maneuver>>();

            var singleGA = TrySingleGA(primaryPlanet);
            if (singleGA != null)
            {
                double dv = singleGA.Sum(m => m.DeltaV.Magnitude);
                if (dv < deltaVBudget * 0.7)
                {
                    solutions.Add(singleGA);
                }
            }

            foreach (var secondary in new[] { "Venus", "Earth", "Mars" })
            {
                if (!bodyDict.ContainsKey(secondary)) continue;

                var solution = TryDoubleGA(primaryPlanet, secondary);
                if (solution != null)
                {
                    double dv = solution.Sum(m => m.DeltaV.Magnitude);
                    if (dv < deltaVBudget)
                    {
                        solutions.Add(solution);
                    }
                }
            }

            return solutions.Count > 0 ? 
                solutions.OrderBy(s => s.Sum(m => m.DeltaV.Magnitude)).First() : null;
        }

        private List<Maneuver> TryDoubleGA(string planet1, string planet2)
        {
            if (!bodyDict.ContainsKey(planet1) || !bodyDict.ContainsKey(planet2))
                return null;

            double launchTime = warningDays * 0.05;
            double ga1Time = warningDays * 0.25;
            double ga2Time = warningDays * 0.5;
            double arrivalTime = warningDays * 0.85;

            var launchState = propagator.Propagate(launchTime);
            var ga1State = propagator.Propagate(ga1Time);
            var ga2State = propagator.Propagate(ga2Time);
            var arrivalState = propagator.Propagate(arrivalTime);

            var earthLaunch = launchState["Earth"];
            var planet1GA = ga1State[planet1];
            var planet2GA = ga2State[planet2];
            var asteroidArrival = arrivalState["Asteroid"];

            var legs = new List<(Vec3 r1, Vec3 r2, double tof)>
            {
                (earthLaunch.Position, planet1GA.Position, (ga1Time - launchTime) * DAY_SEC),
                (planet1GA.Position, planet2GA.Position, (ga2Time - ga1Time) * DAY_SEC),
                (planet2GA.Position, asteroidArrival.Position, (arrivalTime - ga2Time) * DAY_SEC)
            };

            var velocities = new List<Vec3>();
            bool allValid = true;

            foreach (var leg in legs)
            {
                var (v1, _, valid) = LambertSolver.SolveUniversal(leg.r1, leg.r2, leg.tof, SUN_MU);
                if (!valid) { allValid = false; break; }
                velocities.Add(v1);
            }

            if (!allValid) return null;

            Vec3 launchDV = velocities[0] - earthLaunch.Velocity;
            double totalDV = launchDV.Magnitude;

            if (totalDV > deltaVBudget * 0.5) return null;

            var maneuvers = new List<Maneuver>
            {
                new Maneuver(launchTime, launchDV),
                new Maneuver(ga1Time, Vec3.Zero),
                new Maneuver(ga2Time, Vec3.Zero),
                new Maneuver(arrivalTime, Vec3.Zero)
            };

            return maneuvers;
        }

        private Vec3 CalculateGAPlaneChange(Vec3 vIn, Vec3 vPlanet, Vec3 vOut)
        {
            double vInfIn = (vIn - vPlanet).Magnitude;
            double vInfOut = (vOut - vPlanet).Magnitude;
            
            double turnAngle = Math.Acos(Math.Max(-1, Math.Min(1, 
                (vIn - vPlanet).Normalized.Dot((vOut - vPlanet).Normalized))));

            return Vec3.Zero;
        }

        private Vec3 EstimateCorrectionDV(State scState, State asteroidState, double timeToImpact)
        {
            Vec3 relPos = asteroidState.Position - scState.Position;
            Vec3 relVel = asteroidState.Velocity - scState.Velocity;

            double t = timeToImpact * DAY_SEC;
            
            Vec3 predictedMiss = relPos + relVel * t;
            double missDistance = predictedMiss.Magnitude;

            if (missDistance < 1000000) return Vec3.Zero;

            Vec3 correctionDir = predictedMiss.Normalized;
            double correctionMag = missDistance / t;

            return correctionDir * Math.Min(correctionMag, 100);
        }

        private Vec3 CalculateImpactCorrection(State scState, State asteroidState)
        {
            Vec3 relPos = asteroidState.Position - scState.Position;
            double distance = relPos.Magnitude;

            if (distance < 1000000) return Vec3.Zero;

            Vec3 correctionDir = relPos.Normalized;
            double correctionMag = Math.Min(distance / 10000, 50);

            return correctionDir * correctionMag;
        }

        private List<Maneuver> FindFallbackSolution()
        {
            double launchTime = Math.Min(warningDays * 0.1, 30);
            var launchState = propagator.Propagate(launchTime);
            var earthState = launchState["Earth"];
            var asteroidLaunch = launchState["Asteroid"];

            Vec3 toAsteroid = asteroidLaunch.Position - earthState.Position;
            Vec3 launchDir = toAsteroid.Normalized;

            double launchDVMag = Math.Min(deltaVBudget * 0.7, 5000);
            Vec3 launchDV = launchDir * launchDVMag;

            double tof = Math.Min(warningDays * 0.7, 500);
            double correctionTime = launchTime + tof * 0.3;
            double finalTime = launchTime + tof;

            return new List<Maneuver>
            {
                new Maneuver(launchTime, launchDV),
                new Maneuver(correctionTime, new Vec3(10, 0, 0)),
                new Maneuver(finalTime, Vec3.Zero)
            };
        }
