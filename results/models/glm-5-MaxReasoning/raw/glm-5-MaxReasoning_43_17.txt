```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Calculate Lambertian diffuse
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float NdotL = max(0.0, dot(N, L));

    // Procedural Brick Pattern
    // Scale UVs by 8x16
    float2 scaledUV = input.uv * float2(8.0, 16.0);

    // Stagger rows (offset x by 0.5 on every other row)
    // row index is floor(y)
    float row = floor(scaledUV.y);
    // If row is odd, offset x by 0.5
    scaledUV.x += (row % 2.0) * 0.5;

    // Get fractional coordinate within the brick cell
    float2 brickUV = frac(scaledUV);

    // Mortar thickness: 0.05 X, 0.1 Y
    // We use step to determine if we are inside the brick or in the mortar
    // step(a, b) is 1.0 if b >= a
    // We want 1 if brickUV > thick AND brickUV < 1-thick
    float2 mortarThick = float2(0.05, 0.1);
    
    // insideBrick is 1.0 if we are inside the brick bounds, 0.0 in mortar
    // Check lower bound: brickUV >= mortarThick
    // Check upper bound: brickUV <= 1.0 - mortarThick
    float2 insideBrick = step(mortarThick, brickUV) * step(brickUV, 1.0 - mortarThick);
    float brickMask = insideBrick.x * insideBrick.y;

    // Colors
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);

    // Blend between mortar and brick
    float3 surfaceColor = lerp(mortarColor, brickColor, brickMask);

    // Apply lighting factor: 0.4 + 0.6 * diffuse
    float3 finalColor = surfaceColor * (0.4 + 0.6 * NdotL);

    return float4(finalColor, 1.0);
}
```
