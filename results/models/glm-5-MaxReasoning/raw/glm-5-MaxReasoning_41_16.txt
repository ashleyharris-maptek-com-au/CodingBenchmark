; SPIR-V
; Version: 1.0
; Generator: Khronos SPIR-V Tools Assembler; 0
; Bound: 120
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

; Debug Names
OpName %main "main"
OpName %in_worldPos "in_worldPos"
OpName %in_normal "in_normal"
OpName %in_uv "in_uv"
OpName %in_tangent "in_tangent"
OpName %in_color "in_color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo_var "ubo"

; Decorations
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Function_float = OpTypePointer Function %float
%ptr_Function_int = OpTypePointer Function %int

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_3 = OpConstant %float 3
%float_4 = OpConstant %float 4
%float_20 = OpConstant %float 20
%float_0_5 = OpConstant %float 0.5
%float_m1 = OpConstant %float -1
%float_m2 = OpConstant %float -2

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20

; UBO Type
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_uv = OpVariable %ptr_Input_v2float Input
%in_tangent = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform

; Main Function
%main = OpFunction %void None %func
%main_entry = OpLabel

; Function local variables for loop state
%var_zx = OpVariable %ptr_Function_float Function
%var_zy = OpVariable %ptr_Function_float Function
%var_cx = OpVariable %ptr_Function_float Function
%var_cy = OpVariable %ptr_Function_float Function
%var_iter = OpVariable %ptr_Function_int Function

; Load UV
%uv_load = OpLoad %v2float %in_uv
%uv_x = OpCompositeExtract %float %uv_load 0
%uv_y = OpCompositeExtract %float %uv_load 1

; Calculate c = (uv.x*3-2, uv.y*2-1)
%mul_x = OpFMul %float %uv_x %float_3
%cx_val = OpFAdd %float %mul_x %float_m2
OpStore %var_cx %cx_val

%mul_y = OpFMul %float %uv_y %float_2
%cy_val = OpFAdd %float %mul_y %float_m1
OpStore %var_cy %cy_val

; Initialize loop variables
OpStore %var_zx %float_0
OpStore %var_zy %float_0
OpStore %var_iter %int_0

OpBranch %loop_header

%loop_header = OpLabel
OpLoopMerge %loop_merge %loop_continue None
OpBranch %loop_check

%loop_check = OpLabel
%iter_curr = OpLoad %int %var_iter
%cond_loop = OpSLessThan %bool %iter_curr %int_20
OpBranchConditional %cond_loop %loop_body %loop_merge

%loop_body = OpLabel
%zx_curr = OpLoad %float %var_zx
%zy_curr = OpLoad %float %var_zy

; Calculate |z|^2
%zx2 = OpFMul %float %zx_curr %zx_curr
%zy2 = OpFMul %float %zy_curr %zy_curr
%mag2 = OpFAdd %float %zx2 %zy2

; Check escape condition |z|^2 > 4
%cond_escape = OpFGreaterThan %bool %mag2 %float_4
OpSelectionMerge %loop_update None
OpBranchConditional %cond_escape %loop_merge %loop_update

%loop_update = OpLabel
; Calculate next z: z = z^2 + c
; new_zx = zx^2 - zy^2 + cx
%temp_zx = OpFSub %float %zx2 %zy2
%new_zx = OpFAdd %float %temp_zx %cx_val

; new_zy = 2 * zx * zy + cy
%zy_prod = OpFMul %float %zx_curr %zy_curr
%zy_prod2 = OpFMul %float %zy_prod %float_2
%new_zy = OpFAdd %float %zy_prod2 %cy_val

OpStore %var_zx %new_zx
OpStore %var_zy %new_zy

; Increment iteration
%iter_next = OpIAdd %int %iter_curr %int_1
OpStore %var_iter %iter_next

OpBranch %loop_continue

%loop_continue = OpLabel
OpBranch %loop_header

%loop_merge = OpLabel
; Calculate color
%iter_final = OpLoad %int %var_iter
%iter_f = OpConvertSToF %float %iter_final
%t = OpFDiv %float %iter_f %float_20

%g_val = OpFMul %float %t %float_0_5
%b_val = OpFSub %float %float_1 %t

%color_out = OpCompositeConstruct %v4float %t %g_val %b_val %float_1
OpStore %outColor %color_out

OpReturn
OpFunctionEnd
