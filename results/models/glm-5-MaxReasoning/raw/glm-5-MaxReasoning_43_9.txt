```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Lambertian diffuse factor
    float NdotL = dot(N, L);
    float diffuse = max(0.0, NdotL);
    
    // Apply shading factor: (0.5 + 0.5 * diffuse)
    float shading = 0.5 + 0.5 * diffuse;
    
    // UV Checkerboard 8x8
    // Scale UVs by 8 and take floor to get cell coordinates
    float2 cell = floor(input.uv * 8.0);
    
    // Determine parity: (x + y) % 2
    // fmod(x, 2.0) returns 0.0 for even, 1.0 for odd (given integers)
    float parity = fmod(cell.x + cell.y, 2.0);
    
    // Alternate colors 0.9 and 0.2 based on parity
    // If parity is 0, select 0.9. If parity is 1, select 0.2.
    float checkerColor = lerp(0.9, 0.2, parity);
    
    // Combine checkerboard color with lighting
    float3 finalColor = checkerColor * shading;
    
    return float4(finalColor, 1.0);
}
```
