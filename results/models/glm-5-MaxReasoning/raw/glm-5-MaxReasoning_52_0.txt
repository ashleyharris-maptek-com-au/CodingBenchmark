import math

# Module-level state for filtering and control persistence
class AutopilotState:
    def __init__(self):
        # Filtered sensor values
        self.speed_kmh = 0.0
        self.yaw_rate_dps = 0.0
        self.accel_lat = 0.0
        self.accel_lon = 0.0
        
        # PID integrals
        self.speed_i = 0.0
        self.speed_prev_err = 0.0
        self.steer_i = 0.0
        self.steer_prev_err = 0.0
        
        # Lane change state
        self.lane_change_timer = 0.0
        self.in_lane_change = False
        
        # Obstacle tracking
        self.obstacle_dist_history = []
        
        # Initialization
        self.initialized = False

_state = AutopilotState()


def median_of(values):
    """Return median of valid numeric values."""
    valid = [v for v in values if v is not None and not math.isnan(v) and not math.isinf(v)]
    if not valid:
        return None
    sorted_v = sorted(valid)
    n = len(sorted_v)
    return sorted_v[n // 2] if n % 2 else (sorted_v[n // 2 - 1] + sorted_v[n // 2]) / 2


def low_pass(current, new, alpha):
    """Simple exponential low-pass filter."""
    if current is None or not math.isfinite(current):
        return new if new is not None else 0.0
    if new is None or not math.isfinite(new):
        return current
    return alpha * new + (1 - alpha) * current


def clamp(value, min_val, max_val):
    """Clamp value to range."""
    if value is None or not math.isfinite(value):
        return min_val
    return max(min_val, min(max_val, value))


def fuse_wheel_speeds(state):
    """Fuse wheel speed sensors using median filter."""
    speeds = []
    for pos in ['fl', 'fr', 'rl', 'rr']:
        if state.get(f'wheel_speed_{pos}_valid', False):
            sp = state.get(f'wheel_speed_{pos}_kmh')
            if sp is not None and sp >= 0:
                speeds.append(sp)
    return median_of(speeds)


def fuse_gyros(state):
    """Fuse gyroscope sensors."""
    rates = []
    for i in [1, 2, 3]:
        if state.get(f'gyro_{i}_valid', False):
            r = state.get(f'gyro_{i}_yaw_rate_dps')
            if r is not None:
                rates.append(r)
    return median_of(rates)


def fuse_accelerometers(state):
    """Fuse accelerometer sensors."""
    lat_vals, lon_vals = [], []
    for i in [1, 2, 3]:
        if state.get(f'accel_{i}_valid', False):
            lat = state.get(f'accel_{i}_lateral_ms2')
            lon = state.get(f'accel_{i}_longitudinal_ms2')
            if lat is not None:
                lat_vals.append(lat)
            if lon is not None:
                lon_vals.append(lon)
    return median_of(lat_vals), median_of(lon_vals)


def fuse_gps_speed(state):
    """Fuse GPS speed readings."""
    speeds = []
    for i in [1, 2]:
        if state.get(f'gps_{i}_valid', False):
            sp = state.get(f'gps_{i}_speed_kmh')
            if sp is not None and sp >= 0:
                speeds.append(sp)
    return median_of(speeds)


def fuse_steering_angle(state):
    """Fuse steering angle sensors with disagreement detection."""
    angles = []
    for i in [1, 2]:
        if state.get(f'steering_angle_{i}_valid', False):
            a = state.get(f'steering_angle_{i}_deg')
            if a is not None:
                angles.append(a)
    
    if len(angles) == 2:
        # Check for disagreement (>15 degrees difference)
        if abs(angles[0] - angles[1]) > 15:
            return None  # Can't trust either
        return (angles[0] + angles[1]) / 2
    return median_of(angles)


def get_lidar_readings(state):
    """Extract valid lidar readings."""
    readings = {}
    angles = [5, 15, 30, 45, 60, 90, -5, -15, -30, -45, -60, -90]
    
    for angle in angles:
        sign = '+' if angle >= 0 else ''
        key = f'lidar_{sign}{abs(angle):03d}_deg'
        if state.get(f'{key}_valid', False):
            r = state.get(f'{key}_range_m')
            if r is not None and r > 0:
                readings[angle] = r
    return readings


def fuse_vision_data(state):
    """Fuse all three vision systems."""
    lane_offsets = []
    left_dists = []
    right_dists = []
    curvatures = []
    speed_limits = []
    obstacles = []
    
    for i in [1, 2, 3]:
        prefix = f'vision_{i}'
        
        # Lane offset
        offset = state.get(f'{prefix}_lane_offset_m')
        if offset is not None:
            lane_offsets.append(offset)
        
        # Lane lines
        if state.get(f'{prefix}_left_line_valid', False):
            ld = state.get(f'{prefix}_left_line_dist_m')
            if ld is not None:
                left_dists.append(ld)
        
        if state.get(f'{prefix}_right_line_valid', False):
            rd = state.get(f'{prefix}_right_line_dist_m')
            if rd is not None:
                right_dists.append(rd)
        
        # Curvature
        curv = state.get(f'{prefix}_road_curvature_1pm')
        if curv is not None and curv != 0:
            curvatures.append(curv)
        
        # Speed limit
        if state.get(f'{prefix}_speed_limit_valid', False):
            sl = state.get(f'{prefix}_speed_limit_kmh')
            if sl is not None and sl > 0:
                speed_limits.append(sl)
        
        # Obstacles
        if state.get(f'{prefix}_obstacle_detected', False) and state.get(f'{prefix}_obstacle_valid', False):
            od = state.get(f'{prefix}_obstacle_dist_m')
            os = state.get(f'{prefix}_obstacle_rel_speed_kmh', 0)
            if od is not None and od > 0:
                obstacles.append({'dist': od, 'rel_speed': os if os else 0})
    
    return {
        'lane_offset': median_of(lane_offsets),
        'left_dist': median_of(left_dists),
        'right_dist': median_of(right_dists),
        'curvature': median_of(curvatures) if curvatures else 0,
        'speed_limit': min(speed_limits) if speed_limits else None,
        'closest_obstacle': min(obstacles, key=lambda x: x['dist']) if obstacles else None
    }


def calculate_stopping_distance(speed_kmh, mu):
    """Calculate stopping distance in meters."""
    if speed_kmh <= 0:
        return 0
    v_ms = speed_kmh / 3.6
    mu = max(0.1, mu)
    g = 9.81
    return (v_ms ** 2) / (2 * mu * g)


def calculate_safe_speed(target_speed, mu):
    """Adjust target speed for friction conditions."""
    if mu >= 0.8:
        return target_speed
    # Reduce speed proportionally to sqrt(mu) for same stopping distance
    factor = max(0.3, math.sqrt(mu / 0.8))
    return target_speed * factor


def check_forward_obstacles(lidar, vision_obstacle, speed_kmh, mu):
    """Check for obstacles ahead and determine braking needed."""
    min_dist = float('inf')
    
    # Check vision obstacle
    if vision_obstacle:
        min_dist = min(min_dist, vision_obstacle['dist'])
    
    # Check forward lidar beams
    forward_angles = [-15, -5, 5, 15, 30, 45]
    for angle in forward_angles:
        if angle in lidar:
            # Adjust for angle (obstacle not directly ahead is less critical)
            effective_dist = lidar[angle] * math.cos(math.radians(abs(angle)))
            min_dist = min(min_dist, effective_dist)
    
    if min_dist == float('inf'):
        return False, 0.0, float('inf')
    
    # Calculate required stopping distance
    stop_dist = calculate_stopping_distance(speed_kmh, mu)
    safety_margin = max(5, speed_kmh * 0.1)  # 10% of speed or 5m minimum
    critical_dist = stop_dist + safety_margin
    
    # Check for emergency (very close)
    if min_dist < 8:
        return True, 1.0, min_dist
    
    if min_dist < critical_dist:
        # Proportional braking
        urgency = 1.0 - (min_dist / critical_dist)
        brake = clamp(urgency * 1.5, 0.3, 1.0)
        return True, brake, min_dist
    
    # Warning zone
    if min_dist < critical_dist * 1.5:
        return False, 0.1, min_dist
    
    return False, 0.0, min_dist


def autopilot_step(state: dict, dt: float) -> dict:
    global _state
    
    # Initialize on first call
    if not _state.initialized:
        _state.speed_kmh = 0.0
        _state.initialized = True
    
    # === SENSOR FUSION ===
    
    # Get friction coefficient
    mu = state.get('estimated_mu', 1.0)
    if mu is None or not math.isfinite(mu):
        mu = 1.0
    mu = clamp(mu, 0.1, 1.0)
    
    # Rain reduces friction
    rain = state.get('rain_sensor', 0)
    if rain:
        mu = min(mu, 0.6)
    
    # Temperature affects friction (ice below freezing)
    temp = state.get('temperature_c', 20)
    if temp is not None and temp < 0:
        mu = min(mu, 0.3)
    
    # Fuse wheel speeds
    wheel_speed = fuse_wheel_speeds(state)
    
    # Vehicle speed sensors (no validity flags)
    vs1 = state.get('vehicle_speed_1_kmh', 0)
    vs2 = state.get('vehicle_speed_2_kmh', 0)
    vehicle_speed = (vs1 + vs2) / 2 if (vs1 is not None and vs2 is not None) else None
    
    # GPS speed
    gps_speed = fuse_gps_speed(state)
    
    # Fuse all speed sources
    speeds = [v for v in [wheel_speed, vehicle_speed, gps_speed] if v is not None]
    fused_speed = median_of(speeds) if speeds else _state.speed_kmh
    
    # Low-pass filter speed
    _state.speed_kmh = low_pass(_state.speed_kmh, fused_speed, 0.4)
    
    # Fuse gyros
    yaw_rate = fuse_gyros(state)
    if yaw_rate is not None:
        _state.yaw_rate_dps = low_pass(_state.yaw_rate_dps, yaw_rate, 0.3)
    
    # Fuse accelerometers
    accel_lat, accel_lon = fuse_accelerometers(state)
    if accel_lat is not None:
        _state.accel_lat = low_pass(_state.accel_lat, accel_lat, 0.2)
    if accel_lon is not None:
        _state.accel_lon = low_pass(_state.accel_lon, accel_lon, 0.2)
    
    # Get lidar and vision data
    lidar = get_lidar_readings(state)
    vision = fuse_vision_data(state)
    
    # === TARGETS ===
    target_speed = state.get('target_speed_kmh', 100)
    target_lane = state.get('target_lane', 1)
    current_lane = state.get('current_lane', 1)
    
    # Adjust target speed for conditions
    safe_target_speed = calculate_safe_speed(target_speed, mu)
    
    # Respect speed limit
    if vision['speed_limit'] is not None:
        safe_target_speed = min(safe_target_speed, vision['speed_limit'])
    
    # === OBSTACLE DETECTION ===
    emergency, brake, obstacle_dist = check_forward_obstacles(
        lidar, vision['closest_obstacle'], _state.speed_kmh, mu
    )
    
    # Track obstacle history for filtering
    _state.obstacle_dist_history.append(obstacle_dist)
    if len(_state.obstacle_dist_history) > 10:
        _state.obstacle_dist_history.pop(0)
    
    # Use median of recent obstacle distances
    if _state.obstacle_dist_history:
        median_obstacle_dist = median_of(_state.obstacle_dist_history)
        if median_obstacle_dist is not None and median_obstacle_dist < obstacle_dist:
            obstacle_dist = median_obstacle_dist
    
    # === SPEED CONTROL ===
    # PID controller
    speed_target = 0.0 if emergency else safe_target_speed
    speed_error = speed_target - _state.speed_kmh
    
    # PID gains (conservative for stability)
    Kp = 0.025
    Ki = 0.0005
    Kd = 0.015
    
    # Anti-windup
    max_integral = 0.3
    _state.speed_i = clamp(_state.speed_i + speed_error * dt, -max_integral, max_integral)
    
    # Derivative with filtering
    speed_derivative = 0.0
    if dt > 0:
        speed_derivative = (speed_error - _state.speed_prev_err) / dt
    _state.speed_prev_err = speed_error
    
    # PID output
    pid_output = Kp * speed_error + Ki * _state.speed_i + Kd * speed_derivative
    
    # Convert to throttle/brake
    throttle = 0.0
    if not emergency:
        if pid_output > 0:
            throttle = clamp(pid_output * mu, 0.0, 1.0)
        else:
            # Coast or light braking
            brake = max(brake, clamp(-pid_output * 0.2, 0.0, 0.3))
    
    # === LANE KEEPING ===
    steering = 0.0
    
    # Lane offset correction (main control)
    if vision['lane_offset'] is not None:
        offset = vision['lane_offset']
        
        # PID for lane keeping
        Kp_s = 0.06
        Ki_s = 0.0005
        Kd_s = 0.02
        
        _state.steer_i = clamp(_state.steer_i + offset * dt, -0.2, 0.2)
        
        offset_derivative = 0.0
        if dt > 0:
            offset_derivative = (offset - _state.steer_prev_err) / dt
        _state.steer_prev_err = offset
        
        steering = -(Kp_s * offset + Ki_s * _state.steer_i + Kd_s * offset_derivative)
    
    # Additional correction from line distances
    if vision['left_dist'] is not None and vision['right_dist'] is not None:
        # Center error: positive means we're too far left
        center_error = (vision['right_dist'] - vision['left_dist']) / 2
        steering -= center_error * 0.015
    
    # Curvature feed-forward
    if vision['curvature'] is not None and vision['curvature'] != 0:
        steering += vision['curvature'] * 0.0003
    
    # Yaw rate damping (stability)
    steering -= _state.yaw_rate_dps * 0.002
    
    # === LANE CHANGE ===
    if target_lane != current_lane and not emergency:
        direction = 1 if target_lane > current_lane else -1
        
        # Check if lane change is safe using lidar
        if direction > 0:  # Moving right
            check_angles = [-30, -45, -60]
        else:  # Moving left
            check_angles = [30, 45, 60]
        
        # Check for vehicles in adjacent lane
        lane_clear = True
        for angle in check_angles:
            if angle in lidar and lidar[angle] < 25:
                lane_clear = False
                break
        
        # Check blind spot (90 degree beams)
        blind_angle = -90 if direction > 0 else 90
        if blind_angle in lidar and lidar[blind_angle] < 15:
            lane_clear = False
        
        if lane_clear:
            # Initiate lane change
            steering += direction * 0.10
            _state.in_lane_change = True
    else:
        _state.in_lane_change = False
    
    # === LIDAR-BASED LATERAL AVOIDANCE ===
    # Avoid obstacles on the sides
    if 60 in lidar and lidar[60] < 4:
        steering -= 0.08  # Steer right to avoid left obstacle
    if -60 in lidar and lidar[-60] < 4:
        steering += 0.08  # Steer left to avoid right obstacle
    
    # === TRACTION CONTROL ===
    # Reduce steering authority on low friction
    if mu < 0.5:
        # Gentler steering on ice/snow
        steering *= (mu * 2.0)
        steering = clamp(steering, -0.35, 0.35)
    
    # Detect potential skid using lateral acceleration
    if abs(_state.accel_lat) > 3.0 * mu:
        # Reduce steering if approaching lateral grip limit
        steering *= 0.5
    
    # === FINAL OUTPUT ===
    steering = clamp(steering, -1.0, 1.0)
    throttle = clamp(throttle, 0.0, 1.0)
    brake = clamp(brake, 0.0, 1.0)
    
    # Don't apply throttle while braking hard
    if brake > 0.2:
        throttle = 0.0
    
    return {
        'steering': steering,
        'throttle': throttle,
        'brake': brake
    }
