; Bound: 40

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %outColor
OpExecutionMode %main OriginUpperLeft

OpName %worldPos "worldPos"
OpName %outColor "outColor"

OpDecorate %worldPos Location 0
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4

%float_0_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1

%worldPos = OpVariable %ptr_Input_vec3 Input
%outColor = OpVariable %ptr_Output_vec4 Output

%main = OpFunction %void None %func
%entry = OpLabel

%pos = OpLoad %vec3 %worldPos
%y = OpCompositeExtract %float %pos 1

%y_scaled = OpFMul %float %y %float_0_5
%y_biased = OpFAdd %float %y_scaled %float_0_5

%t_max = OpFMax %float %y_biased %float_0_0
%t = OpFMin %float %t_max %float_1_0

%bottom = OpCompositeConstruct %vec3 %float_0_1 %float_0_4 %float_0_1
%top = OpCompositeConstruct %vec3 %float_1_0 %float_1_0 %float_1_0

%one_minus_t = OpFSub %float %float_1_0 %t
%term1 = OpVectorTimesScalar %vec3 %bottom %one_minus_t
%term2 = OpVectorTimesScalar %vec3 %top %t
%rgb = OpFAdd %vec3 %term1 %term2

%rgba = OpCompositeConstruct %vec4 %rgb %float_1_0
OpStore %outColor %rgba

OpReturn
OpFunctionEnd
