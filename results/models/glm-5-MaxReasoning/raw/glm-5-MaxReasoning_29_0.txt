Looking at this problem, I need to implement a maximum clique finder that handles graphs from 20 to 1 million vertices efficiently. Since this is NP-Hard, I'll use a hybrid approach:

1. **Fast greedy heuristics** for good initial solutions and large graphs
2. **Branch-and-bound with coloring** for exact solutions on smaller instances
3. **Core decomposition** for pruning and ordering
4. **Parallel processing** to utilize multiple cores

Key optimizations:
- Greedy coloring provides tight upper bounds for pruning
- Core numbers give vertex importance ordering and additional pruning
- Degeneracy ordering ensures efficient branch-and-bound
- Time-boxed execution to always produce a valid solution

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    static void Main()
    {
        var solver = new MaxCliqueSolver();
        solver.Run();
    }
}

class MaxCliqueSolver
{
    private int n;
    private HashSet<int>[] adj;
    private int[] bestClique;
    private int bestSize;
    private readonly object bestLock = new object();
    private Stopwatch stopwatch;
    private const long TimeLimitMs = 270000;

    public void Run()
    {
        ReadGraph();
        if (n == 0) { Console.WriteLine("0\n"); return; }

        stopwatch = Stopwatch.StartNew();

        // Phase 1: Fast greedy heuristics (always runs)
        RunGreedyPhase();

        // Phase 2: Exact search for moderate-sized graphs
        if (n <= 75000 && stopwatch.ElapsedMilliseconds < TimeLimitMs * 0.8)
            RunExactPhase();

        WriteResult();
    }

    private void ReadGraph()
    {
        var header = Console.ReadLine()?.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (header == null || header.Length < 2) { n = 0; return; }

        n = int.Parse(header[0]);
        int m = int.Parse(header[1]);

        adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            var line = Console.ReadLine()?.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (line == null || line.Length < 2) continue;
            int u = int.Parse(line[0]), v = int.Parse(line[1]);
            if (u >= 0 && u < n && v >= 0 && v < n && u != v)
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }
        bestClique = new int[n];
        bestSize = 0;
    }

    private void RunGreedyPhase()
    {
        Parallel.Invoke(
            () => GreedySearch(OrderByDegree(true)),
            () => GreedySearch(OrderByDegree(false)),
            () => GreedySearch(OrderByRandom(42)),
            () => GreedySearch(OrderByRandom(123)),
            () => GreedySearch(OrderByRandom(456)),
            () => GreedySearch(OrderByRandom(789)),
            () => GreedySearch(OrderByCore()),
            () => GreedySearch(OrderByRandom(2024))
        );
    }

    private int[] OrderByDegree(bool desc)
        => desc ? Enumerable.Range(0, n).OrderByDescending(v => adj[v].Count).ToArray()
                : Enumerable.Range(0, n).OrderBy(v => adj[v].Count).ToArray();

    private int[] OrderByRandom(int seed)
        => Enumerable.Range(0, n).OrderBy(_ => new Random(seed).Next()).ToArray();

    private int[] OrderByCore()
    {
        var cores = ComputeCoreNumbers();
        return Enumerable.Range(0, n).OrderByDescending(v => cores[v]).ToArray();
    }

    private List<int> GreedySearch(int[] order)
    {
        var clique = new List<int>();
        foreach (int v in order)
        {
            bool canAdd = true;
            foreach (int u in clique)
                if (!adj[v].Contains(u)) { canAdd = false; break; }
            if (canAdd) clique.Add(v);
        }
        UpdateBest(clique);
        return clique;
    }

    private void RunExactPhase()
    {
        var cores = ComputeCoreNumbers();
        var order = Enumerable.Range(0, n).OrderBy(v => cores[v]).ToArray();
        var pos = new int[n];
        for (int i = 0; i < n; i++) pos[order[i]] = i;

        Parallel.For(0, n, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, (i, state) =>
        {
            if (stopwatch.ElapsedMilliseconds > TimeLimitMs) { state.Break(); return; }
            int v = order[i];
            if (cores[v] + 1 <= Volatile.Read(ref bestSize)) return;

            var clique = new List<int> { v };
            var candidates = adj[v].Where(u => pos[u] > i && cores[u] + 1 > bestSize).ToList();
            BranchAndBound(clique, candidates);
        });
    }

    private void BranchAndBound(List<int> clique, List<int> candidates)
    {
        if (stopwatch.ElapsedMilliseconds > TimeLimitMs) return;
        int currentBest = Volatile.Read(ref bestSize);

        if (candidates.Count == 0) { UpdateBest(clique); return; }

        var colored = ColorVertices(candidates);
        if (clique.Count + colored[0].Color <= currentBest) return;

        for (int i = colored.Count - 1; i >= 0; i--)
        {
            if (stopwatch.ElapsedMilliseconds > TimeLimitMs) return;
            if (clique.Count + colored[i].Color <= Volatile.Read(ref bestSize)) return;

            int v = colored[i].Vertex;
            var newCand = new List<int>();
            for (int j = 0; j < i; j++)
                if (adj[v].Contains(colored[j].Vertex)) newCand.Add(colored[j].Vertex);

            clique.Add(v);
            BranchAndBound(clique, newCand);
            clique.RemoveAt(clique.Count - 1);
        }
    }

    private List<(int Vertex, int Color)> ColorVertices(List<int> vertices)
    {
        int count = vertices.Count;
        var colors = new int[count];
        var used = new bool[count + 1];

        for (int i = 0; i < count; i++)
        {
            Array.Clear(used, 0, count + 1);
            int v = vertices[i];
            for (int j = 0; j < i; j++)
                if (adj[v].Contains(vertices[j])) used[colors[j]] = true;
            int c = 1; while (used[c]) c++;
            colors[i] = c;
        }

        var result = new List<(int, int)>(count);
        for (int i = 0; i < count; i++) result.Add((vertices[i], colors[i]));
        result.Sort((a, b) => b.Item2.CompareTo(a.Item2));
        return result;
    }

    private int[] ComputeCoreNumbers()
    {
        var degree = adj.Select(a => a.Count).ToArray();
        var core = new int[n];
        var removed = new bool[n];
        int maxDeg = degree.DefaultIfEmpty(0).Max();

        var buckets = new LinkedList<int>[maxDeg + 1];
        var nodes = new LinkedListNode<int>[n];
        for (int d = 0; d <= maxDeg; d++) buckets[d] = new LinkedList<int>();
        for (int v = 0; v < n; v++) nodes[v] = buckets[degree[v]].AddLast(v);

        for (int d = 0; d <= maxDeg; d++)
        {
            while (buckets[d].Count > 0)
            {
                int v = buckets[d].First.Value;
                buckets[d].RemoveFirst();
                nodes[v] = null;
                if (removed[v]) continue;
                removed[v] = true;
                core[v] = d;

                foreach (int u in adj[v])
                {
                    if (!removed[u] && degree[u] > d)
                    {
                        int old = degree[u]--;
                        if (nodes[u] != null) { buckets[old].Remove(nodes[u]); nodes[u] = buckets[degree[u]].AddLast(u); }
                    }
                }
            }
        }
        return core;
    }

    private void UpdateBest(List<int> clique)
    {
        if (clique.Count <= Volatile.Read(ref bestSize)) return;
        lock (bestLock)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                for (int i = 0; i < clique.Count; i++) bestClique[i] = clique[i];
            }
        }
    }

    private void WriteResult()
    {
        var sb = new StringBuilder();
        sb.AppendLine(bestSize.ToString());
        if (bestSize > 0)
            sb.AppendLine(string.Join(" ", bestClique.Take(bestSize)));
        else
            sb.AppendLine();
        Console.Write(sb);
    }
}
```
