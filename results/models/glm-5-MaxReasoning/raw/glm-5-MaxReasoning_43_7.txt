```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate view direction
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    
    // Calculate light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Diffuse lighting
    float NdotL = max(dot(N, L), 0.0);
    
    // Rim (Fresnel) lighting: rim = pow(1 - max(dot(N,V),0), 3)
    float NdotV = max(dot(N, V), 0.0);
    float rim = pow(1.0 - NdotV, 3.0);
    
    // Base color modulated by (0.3 + 0.5*diffuse)
    float3 baseColor = float3(0.2, 0.3, 0.8);
    float3 diffuseColor = baseColor * (0.3 + 0.5 * NdotL);
    
    // Add rim tint (1.0, 0.8, 0.5) * rim
    float3 rimColor = float3(1.0, 0.8, 0.5) * rim;
    
    float3 finalColor = diffuseColor + rimColor;
    
    return float4(finalColor, 1.0);
}
```
