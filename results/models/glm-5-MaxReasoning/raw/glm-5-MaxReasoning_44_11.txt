```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate position within 6x6 grid cells
    vec2 cellUV = fract(uv * 6.0);
    
    // Get distance from cell center (0.5, 0.5)
    vec2 centered = cellUV - 0.5;
    float dist = length(centered);
    
    // Dot pattern: radius 0.3 in cell space
    float dotFactor = step(dist, 0.3);
    
    // Colors
    vec3 dotColor = vec3(1.0, 0.8, 0.0);   // Yellow
    vec3 bgColor = vec3(0.1, 0.1, 0.3);    // Dark blue
    
    // Select color based on pattern
    vec3 baseColor = mix(bgColor, dotColor, dotFactor);
    
    // Lambertian diffuse lighting
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    
    // Apply shading factor
    float shading = 0.3 + 0.7 * diffuse;
    vec3 finalColor = baseColor * shading;
    
    outColor = vec4(finalColor, 1.0);
}
```
