```glsl
#version 450

layout(local_size_x = 256) in;

layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;
    vec4 radii;
    vec4 weights;
    vec4 bounds;
};

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out of bounds invocation
    if (gid >= N) return;
    
    // Load current boid position and velocity
    vec3 pos = inData[gid * 2].xyz;
    vec3 vel = inData[gid * 2 + 1].xyz;
    
    // Extract parameters
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Accumulators for flocking forces
    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);
    
    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;
    
    // Iterate through all boids to find neighbors
    for (uint j = 0; j < N; j++) {
        // Skip self
        if (j == gid) continue;
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        vec3 diff = pos - otherPos;
        float dist = length(diff);
        
        // Skip zero-distance neighbors to avoid NaN
        if (dist < 1e-6) continue;
        
        // Separation - steer away from nearby boids (weighted by inverse distance)
        if (dist < sepRadius) {
            separation += diff / (dist * dist);
            sepCount++;
        }
        
        // Alignment - match velocity with nearby boids
        if (dist < alignRadius) {
            alignment += otherVel;
            alignCount++;
        }
        
        // Cohesion - steer toward center of mass of nearby boids
        if (dist < cohRadius) {
            cohesion += otherPos;
            cohCount++;
        }
    }
    
    // Compute acceleration from flocking rules
    vec3 acceleration = vec3(0.0);
    
    // Average and apply separation force
    if (sepCount > 0) {
        separation /= float(sepCount);
        acceleration += separation * sepWeight;
    }
    
    // Average and apply alignment force (steer toward average velocity)
    if (alignCount > 0) {
        alignment /= float(alignCount);
        alignment = alignment - vel;
        acceleration += alignment * alignWeight;
    }
    
    // Average and apply cohesion force (steer toward center of mass)
    if (cohCount > 0) {
        cohesion /= float(cohCount);
        cohesion = cohesion - pos;
        acceleration += cohesion * cohWeight;
    }
    
    // Boundary force - push toward center when exceeding threshold
    float threshold = boundSize * 0.4;
    float scale = boundSize * 0.5;
    
    if (abs(pos.x) > threshold) {
        acceleration.x -= sign(pos.x) * boundForce * (abs(pos.x) - threshold) / scale;
    }
    if (abs(pos.y) > threshold) {
        acceleration.y -= sign(pos.y) * boundForce * (abs(pos.y) - threshold) / scale;
    }
    if (abs(pos.z) > threshold) {
        acceleration.z -= sign(pos.z) * boundForce * (abs(pos.z) - threshold) / scale;
    }
    
    // Update velocity with acceleration
    vel += acceleration * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = vel * (maxSpeed / speed);
    }
    
    // Update position with velocity
    pos += vel * dt;
    
    // Write output
    outData[gid * 2] = vec4(pos, 0.0);
    outData[gid * 2 + 1] = vec4(vel, 0.0);
}
```
