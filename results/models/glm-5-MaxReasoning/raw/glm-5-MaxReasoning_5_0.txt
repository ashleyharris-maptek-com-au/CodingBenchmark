Looking at this Hamiltonian Path problem, I need to implement a scalable solver that handles grids from small to large (up to 2500 cells). Let me design a multi-tiered approach:

1. **Quick feasibility checks** (connectivity, parity)
2. **Pattern-based solving** for clean grids without obstacles
3. **Optimized backtracking** with Warnsdorff's heuristic and aggressive pruning

```rust
use std::io::{self, Read, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    let mut input = String::new();
    stdin.lock().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    
    let width: usize = iter.next().unwrap().parse().unwrap();
    let height: usize = iter.next().unwrap().parse().unwrap();
    let obstacle_count: usize = iter.next().unwrap().parse().unwrap();
    let require_cycle: bool = iter.next().unwrap() == "1";
    
    let n = width * height;
    let mut blocked = vec![false; n];
    
    for _ in 0..obstacle_count {
        let x: usize = iter.next().unwrap().parse().unwrap();
        let y: usize = iter.next().unwrap().parse().unwrap();
        blocked[y * width + x] = true;
    }
    
    match solve_hamiltonian(width, height, &blocked, require_cycle) {
        Some(path) => {
            writeln!(out, "{}", path.len()).unwrap();
            for (x, y) in path {
                writeln!(out, "{} {}", x, y).unwrap();
            }
        }
        None => { writeln!(out, "0").unwrap(); }
    }
}

fn solve_hamiltonian(width: usize, height: usize, blocked: &[bool], require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let n = width * height;
    let free_cells = blocked.iter().filter(|&&b| !b).count();
    
    if free_cells == 0 { return Some(vec![]); }
    if blocked[0] { return None; }
    
    let neighbors = build_neighbors(width, height, blocked);
    
    if !is_connected(&neighbors, free_cells) { return None; }
    if !check_parity(width, height, blocked, require_cycle) { return None; }
    
    // Pattern-based for clean grids
    if obstacle_count == 0 {
        if !require_cycle {
            return Some(snake_path(width, height));
        } else if width % 2 == 0 && height % 2 == 0 && width >= 2 && height >= 2 {
            if let Some(p) = cycle_pattern(width, height) { return Some(p); }
        }
    }
    
    backtrack_solve(width, height, blocked, &neighbors, free_cells, require_cycle)
}

fn build_neighbors(width: usize, height: usize, blocked: &[bool]) -> Vec<Vec<usize>> {
    let n = width * height;
    let mut neighbors = vec![Vec::with_capacity(4); n];
    let dirs: [(i32, i32); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];
    
    for y in 0..height {
        for x in 0..width {
            let idx = y * width + x;
            if blocked[idx] { continue; }
            for (dx, dy) in dirs {
                let (nx, ny) = (x as i32 + dx, y as i32 + dy);
                if nx >= 0 && ny >= 0 && nx < width as i32 && ny < height as i32 {
                    let nidx = (ny as usize) * width + (nx as usize);
                    if !blocked[nidx] { neighbors[idx].push(nidx); }
                }
            }
        }
    }
    neighbors
}

fn is_connected(neighbors: &[Vec<usize>], free_cells: usize) -> bool {
    let n = neighbors.len();
    let mut visited = vec![false; n];
    let mut queue = VecDeque::with_capacity(free_cells);
    queue.push_back(0);
    visited[0] = true;
    let mut count = 1;
    
    while let Some(idx) = queue.pop_front() {
        for &nidx in &neighbors[idx] {
            if !visited[nidx] {
                visited[nidx] = true;
                count += 1;
                queue.push_back(nidx);
            }
        }
    }
    count == free_cells
}

fn check_parity(width: usize, height: usize, blocked: &[bool], require_cycle: bool) -> bool {
    let (mut black, mut white) = (0usize, 0usize);
    for y in 0..height {
        for x in 0..width {
            if !blocked[y * width + x] {
                if (x + y) % 2 == 0 { black += 1; } else { white += 1; }
            }
        }
    }
    if require_cycle { black == white } else { black >= white && black <= white + 1 }
}

fn snake_path(width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut path = Vec::with_capacity(width * height);
    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width { path.push((x, y)); }
        } else {
            for x in (0..width).rev() { path.push((x, y)); }
        }
    }
    path
}

fn cycle_pattern(width: usize, height: usize) -> Option<Vec<(usize, usize)>> {
    // Construction for even x even: modified snake ending at (1,0)
    let mut path = Vec::with_capacity(width * height);
    
    // First column down (except (0,0))
    for y in (1..height).rev() { path.push((0, y)); }
    
    // Snake remaining columns
    for x in 1..width {
        if x % 2 == 1 {
            for y in 0..height { path.push((x, y)); }
        } else {
            for y in (0..height).rev() { path.push((x, y)); }
        }
    }
    
    // Check if valid cycle (ends adjacent to start)
    let last = *path.last()?;
    if (last.0 == 1 && last.1 == 0) || (last.0 == 0 && last.1 == 1) {
        // Prepend (0,0) and verify it's actually valid
        path.insert(0, (0, 0));
        // Need to reverse since we built it backwards
        path.reverse();
        // Rebuild properly
        let mut correct_path = vec![(0, 0)];
        // Column 0: go down
        for y in 1..height { correct_path.push((0, y)); }
        // Snake columns 1 to width-1
        for x in 1..width {
            if x % 2 == 1 {
                for y in (0..height).rev() { correct_path.push((x, y)); }
            } else {
                for y in 0..height { correct_path.push((x, y)); }
            }
        }
        let last = correct_path.last().unwrap();
        if (last.0 == 1 && last.1 == 0) { return Some(correct_path); }
    }
    None
}

struct Solver {
    width: usize,
    neighbors: Vec<Vec<usize>>,
    visited: Vec<u8>,
    degree: Vec<u8>,
    path: Vec<usize>,
    target: usize,
    require_cycle: bool,
}

impl Solver {
    fn update_degree(&mut self, idx: usize, delta: i8) {
        for &n in &self.neighbors[idx] {
            self.degree[n] = (self.degree[n] as i8 + delta) as u8;
        }
    }
    
    fn has_dead_end(&self) -> bool {
        for i in 0..self.visited.len() {
            if self.visited[i] == 0 && !self.neighbors[i].is_empty() && self.degree[i] == 0 {
                return true;
            }
        }
        false
    }
    
    fn forced_neighbor(&self, cur: usize) -> Option<usize> {
        for i in 0..self.visited.len() {
            if self.visited[i] == 0 && self.degree[i] == 1 && !self.neighbors[cur].contains(&i) {
                return Some(i);
            }
        }
        None
    }
    
    fn search(&mut self) -> bool {
        let len = self.path.len();
        if len == self.target {
            return if self.require_cycle {
                self.neighbors[0].contains(self.path.last().unwrap())
            } else { true };
        }
        
        if self.has_dead_end() { return false; }
        
        let cur = self.path[len - 1];
        if let Some(forced) = self.forced_neighbor(cur) { return false; }
        
        let mut nexts: Vec<usize> = self.neighbors[cur].iter()
            .copied().filter(|&n| self.visited[n] == 0).collect();
        nexts.sort_by_key(|&n| self.degree[n]);
        
        for next in nexts {
            self.path.push(next);
            self.visited[next] = 1;
            self.update_degree(next, -1);
            
            if self.search() { return true; }
            
            self.update_degree(next, 1);
            self.visited[next] = 0;
            self.path.pop();
        }
        false
    }
}

fn backtrack_solve(width: usize, height: usize, blocked: &[bool], neighbors: &[Vec<usize>], 
                   free_cells: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let n = width * height;
    let degree: Vec<u8> = neighbors.iter().map(|v| v.len() as u8).collect();
    
    let mut solver = Solver {
        width,
        neighbors: neighbors.to_vec(),
        visited: vec![0; n],
        degree,
        path: vec![0],
        target: free_cells,
        require_cycle,
    };
    
    solver.visited[0] = 1;
    solver.update_degree(0, -1);
    
    if solver.search() {
        Some(solver.path.into_iter().map(|i| (i % width, i / width)).collect())
    } else { None }
}
```

This solution implements:

1. **Connectivity check** via BFS to ensure all free cells are reachable
2. **Parity check** - Hamiltonian paths require the bipartite balance to be correct
3. **Snake pattern** for obstacle-free path problems
4. **Cycle pattern** attempt for even×even grids requiring cycles
5. **Backtracking** with:
   - Warnsdorff's rule (visit cells with fewer neighbors first)
   - Dead-end detection (cells with degree 0 become unreachable)
   - Forced move pruning (cells with degree 1 that aren't adjacent to current position)

The algorithm adapts to grid size by using pattern-based solutions when possible (O(n) time) and falling back to optimized backtracking for complex cases with obstacles.
