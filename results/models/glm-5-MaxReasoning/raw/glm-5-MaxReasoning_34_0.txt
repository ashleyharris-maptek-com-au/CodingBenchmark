import sys
import math

def solve_cvrp():
    # 1. Read Input
    input_data = sys.stdin.read().split()
    if not input_data:
        return

    iterator = iter(input_data)
    try:
        N = int(next(iterator))
        K = int(next(iterator))
        CAPACITY = int(next(iterator))
        
        depot_x = float(next(iterator))
        depot_y = float(next(iterator))
        
        # Store points: index 0 is depot, 1..N are customers
        points = [(depot_x, depot_y)]
        demands = [0]
        
        for _ in range(N):
            x = float(next(iterator))
            y = float(next(iterator))
            d = int(next(iterator))
            points.append((x, y))
            demands.append(d)
            
    except StopIteration:
        return

    # 2. Precompute Distance Matrix
    # dist[i][j] stores Euclidean distance between node i and node j
    dist = [[0.0] * (N + 1) for _ in range(N + 1)]
    for i in range(N + 1):
        for j in range(i + 1, N + 1):
            d_val = math.hypot(points[i][0] - points[j][0], points[i][1] - points[j][1])
            dist[i][j] = d_val
            dist[j][i] = d_val

    # 3. Clarke-Wright Savings Algorithm
    # Initialize: each customer is a separate route
    # routes stores the list of customers for each route
    # route_demands stores the total demand for each route
    routes = [[i] for i in range(1, N + 1)]
    route_demands = [demands[i] for i in range(1, N + 1)]
    
    # node_to_route maps a customer index (1..N) to its route index in `routes`
    node_to_route = [0] * (N + 1)
    for i in range(1, N + 1):
        node_to_route[i] = i - 1
        
    # Calculate savings: s_ij = d(0,i) + d(0,j) - d(i,j)
    savings = []
    for i in range(1, N + 1):
        for j in range(i + 1, N + 1):
            s = dist[0][i] + dist[0][j] - dist[i][j]
            savings.append((s, i, j))
            
    # Sort savings in descending order
    savings.sort(key=lambda x: x[0], reverse=True)
    
    # Merge routes based on savings
    for s, i, j in savings:
        r_i_idx = node_to_route[i]
        r_j_idx = node_to_route[j]
        
        # Skip if i and j are already in the same route
        if r_i_idx == r_j_idx:
            continue
            
        # Check if i is an endpoint of its route
        # i is endpoint if it's the first or last element
        route_i = routes[r_i_idx]
        i_is_endpoint = (route_i[0] == i or route_i[-1] == i)
        
        # Check if j is an endpoint of its route
        route_j = routes[r_j_idx]
        j_is_endpoint = (route_j[0] == j or route_j[-1] == j)
        
        if not i_is_endpoint or not j_is_endpoint:
            continue
            
        # Check capacity constraint
        if route_demands[r_i_idx] + route_demands[r_j_idx] > CAPACITY:
            continue
            
        # Perform merge
        # We need to link the endpoints i and j.
        # 4 cases based on positions of i and j in their respective routes
        
        new_route = []
        
        # Case 1: i is last, j is first -> route_i + route_j
        if route_i[-1] == i and route_j[0] == j:
            new_route = route_i + route_j
        # Case 2: i is last, j is last -> route_i + reverse(route_j)
        elif route_i[-1] == i and route_j[-1] == j:
            new_route = route_i + route_j[::-1]
        # Case 3: i is first, j is first -> reverse(route_i) + route_j
        elif route_i[0] == i and route_j[0] == j:
            new_route = route_i[::-1] + route_j
        # Case 4: i is first, j is last -> route_j + route_i
        elif route_i[0] == i and route_j[-1] == j:
            new_route = route_j + route_i
            
        # Update data structures
        # We keep r_i_idx and invalidate r_j_idx
        routes[r_i_idx] = new_route
        route_demands[r_i_idx] += route_demands[r_j_idx]
        
        # Invalidate old route
        routes[r_j_idx] = []
        route_demands[r_j_idx] = 0
        
        # Update node mappings for the merged route
        for node in new_route:
            node_to_route[node] = r_i_idx

    # Filter out empty routes
    final_routes = [r for r in routes if r]

    # 4. Ensure K constraint (Force merge if too many routes)
    # If current routes > K, we merge the closest pairs until we have K
    # This is a greedy post-processing step.
    while len(final_routes) > K:
        best_merge_cost = float('inf')
        best_pair = (-1, -1)
        best_new_route = None
        
        # Iterate all pairs of routes to find the cheapest link
        for i in range(len(final_routes)):
            for j in range(i + 1, len(final_routes)):
                r1 = final_routes[i]
                r2 = final_routes[j]
                
                if route_demands[node_to_route[r1[0]]] + route_demands[node_to_route[r2[0]]] > CAPACITY:
                    # Note: node_to_route might be stale for capacity checks if we modified routes in place heavily
                    # Let's recompute demand just to be safe or trust the logic.
                    # Re-computing demand is safer here.
                    d1 = sum(demands[n] for n in r1)
                    d2 = sum(demands[n] for n in r2)
                    if d1 + d2 > CAPACITY:
                        continue
                
                # Try merging endpoints
                # Endpoint pairs: (r1[-1], r2[0]), (r1[-1], r2[-1]), (r1[0], r2[0]), (r1[0], r2[-1])
                # We look for min added distance (link_cost - removed depot trips)
                
                # Helper to get cost delta
                def get_merge_cost(end1, start2, is_reversed):
                    # cost change = dist(end1, start2) - dist(end1, 0) - dist(0, start2)
                    # Since we are connecting two routes that currently return to depot,
                    # we remove the two trips to depot and add the link.
                    return dist[end1][start2] - dist[end1][0] - dist[0][start2]
                
                # Current best for this pair
                pairs_configs = [
                    (r1[-1], r2[0], r1 + r2),
                    (r1[-1], r2[-1], r1 + r2[::-1]),
                    (r1[0], r2[0], r1[::-1] + r2),
                    (r1[0], r2[-1], r2 + r1)
                ]
                
                for e1, s2, nr in pairs_configs:
                    cost = get_merge_cost(e1, s2, False)
                    if cost < best_merge_cost:
                        best_merge_cost = cost
                        best_pair = (i, j)
                        best_new_route = nr
        
        if best_pair[0] != -1:
            idx1, idx2 = best_pair
            # Merge route at idx2 into idx1
            final_routes[idx1] = best_new_route
            # Remove route at idx2
            final_routes.pop(idx2)
            # Update node_to_route for the moved nodes
            # Since we popped idx2, indices shifted.
            # For simplicity, we can just rebuild node_to_route mapping if needed,
            # but since we only use it for K-check loop, and we track final_routes directly,
            # we just need to be careful if we needed `route_demands`.
            # We recompute demands above.
        else:
            # Cannot merge anymore due to capacity constraints
            break

    # 5. Local Search (2-opt) for each route
    # Improves the path quality of the finalized routes
    for route in final_routes:
        improved = True
        while improved:
            improved = False
            n = len(route)
            for i in range(n - 1):
                for j in range(i + 1, n):
                    # Consider reversing the segment i...j
                    # Current edges: (i-1, i) and (j, j+1)
                    # New edges: (i-1, j) and (i, j+1)
                    # Indices in route list
                    
                    # Handle depot connections
                    # u is node before segment, v is start of segment
                    u = route[i-1] if i > 0 else 0
                    v = route[i]
                    # w is end of segment, x is node after segment
                    w = route[j]
                    x = route[j+1] if j < n - 1 else 0
                    
                    # Current distance
                    d_curr = dist[u][v] + dist[w][x]
                    # New distance
                    d_new = dist[u][w] + dist[v][x]
                    
                    if d_new < d_curr - 1e-9:
                        # Apply reversal
                        route[i:j+1] = route[i:j+1][::-1]
                        improved = True
                        break
                if improved:
                    break

    # 6. Output
    total_distance = 0.0
    for route in final_routes:
        if not route: continue
        # Depot to first
        total_distance += dist[0][route[0]]
        # Between customers
        for k in range(len(route) - 1):
            total_distance += dist[route[k]][route[k+1]]
        # Last to depot
        total_distance += dist[route[-1]][0]
        
    print(f"{total_distance:.2f}")
    print(len(final_routes))
    
    for route in final_routes:
        # Convert 1-based internal indices back to 0-based output
        print(" ".join(str(x - 1) for x in route))

if __name__ == "__main__":
    solve_cvrp()
