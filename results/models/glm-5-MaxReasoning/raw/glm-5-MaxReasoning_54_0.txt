import math

# === MODULE STATE (persists across calls within scenario) ===
_state = {}

def _init_state():
    global _state
    _state = {
        'phase': 'unknown',
        'flare_started': False,
        'flare_entry_alt': 0,
        'go_around': False,
        'last_radio_alt': None,
        'computed_vs': 0,
        'integrator_gs': 0,
        'integrator_loc': 0,
        'integrator_speed': 0,
        'smoothed_gs': 0,
        'smoothed_loc': 0,
        'total_time': 0,
        'eng1_failed': False,
        'eng2_failed': False,
        'last_ias': None,
        'last_throttle': 0.4,
        'approach_stable_time': 0,
        'aileron_integrator': 0,
    }

_init_state()

def _median_valid(sensors):
    """Get median of valid sensors. Returns (value, num_valid)"""
    valid = []
    for val, is_valid in sensors:
        if is_valid and val is not None:
            valid.append(val)
    if not valid:
        return None, 0
    valid.sort()
    n = len(valid)
    if n % 2 == 0:
        return (valid[n//2 - 1] + valid[n//2]) / 2.0, n
    return valid[n//2], n

def _clamp(v, lo, hi):
    return max(lo, min(hi, v))

def _rate_limit(current, target, max_rate, dt):
    """Rate limit a value change"""
    if dt <= 0:
        return target
    max_change = max_rate * dt
    if target > current:
        return min(target, current + max_change)
    else:
        return max(target, current - max_change)

def autopilot_step(state: dict, dt: float) -> dict:
    global _state
    
    # Time tracking
    _state['total_time'] += dt
    
    # Output defaults
    out = {
        'elevator': 0.0,
        'aileron': 0.0,
        'rudder': 0.0,
        'throttle_1': 0.5,
        'throttle_2': 0.5,
        'flaps': 30.0,
        'gear': 1.0,
        'speedbrake': 0.0,
        'spoiler': 0.0,
        'autobrake': 0.0,
        'reverse_1': 0.0,
        'reverse_2': 0.0,
    }
    
    # === SENSOR READING WITH VOTING ===
    
    # Glideslope (3 receivers)
    gs_val, gs_cnt = _median_valid([
        (state.get('glideslope_1_dots'), state.get('glideslope_1_valid', False)),
        (state.get('glideslope_2_dots'), state.get('glideslope_2_valid', False)),
        (state.get('glideslope_3_dots'), state.get('glideslope_3_valid', False)),
    ])
    
    # Localizer (3 receivers)
    loc_val, loc_cnt = _median_valid([
        (state.get('localizer_1_dots'), state.get('localizer_1_valid', False)),
        (state.get('localizer_2_dots'), state.get('localizer_2_valid', False)),
        (state.get('localizer_3_dots'), state.get('localizer_3_valid', False)),
    ])
    
    # Radio altitude
    radio_alt = state.get('radio_alt_ft', 0)
    
    # Barometric altitude
    baro_alt, _ = _median_valid([
        (state.get('baro_alt_1_ft'), state.get('baro_alt_1_valid', False)),
        (state.get('baro_alt_2_ft'), state.get('baro_alt_2_valid', False)),
        (state.get('baro_alt_3_ft'), state.get('baro_alt_3_valid', False)),
    ])
    
    # Airspeed (IAS)
    ias_val, ias_cnt = _median_valid([
        (state.get('ias_1_kt'), state.get('ias_1_valid', False)),
        (state.get('ias_2_kt'), state.get('ias_2_valid', False)),
        (state.get('ias_3_kt'), state.get('ias_3_valid', False)),
    ])
    
    # Angle of Attack
    aoa_val, _ = _median_valid([
        (state.get('aoa_1_deg'), state.get('aoa_1_valid', False)),
        (state.get('aoa_2_deg'), state.get('aoa_2_valid', False)),
        (state.get('aoa_3_deg'), state.get('aoa_3_valid', False)),
    ])
    
    # IRS - Attitude
    pitch_val, _ = _median_valid([
        (state.get('irs_1_pitch_deg'), state.get('irs_1_valid', False)),
        (state.get('irs_2_pitch_deg'), state.get('irs_2_valid', False)),
        (state.get('irs_3_pitch_deg'), state.get('irs_3_valid', False)),
    ])
    
    roll_val, _ = _median_valid([
        (state.get('irs_1_roll_deg'), state.get('irs_1_valid', False)),
        (state.get('irs_2_roll_deg'), state.get('irs_2_valid', False)),
        (state.get('irs_3_roll_deg'), state.get('irs_3_valid', False)),
    ])
    
    heading_val, _ = _median_valid([
        (state.get('irs_1_heading_deg'), state.get('irs_1_valid', False)),
        (state.get('irs_2_heading_deg'), state.get('irs_2_valid', False)),
        (state.get('irs_3_heading_deg'), state.get('irs_3_valid', False)),
    ])
    
    vs_val, _ = _median_valid([
        (state.get('irs_1_vs_fpm'), state.get('irs_1_valid', False)),
        (state.get('irs_2_vs_fpm'), state.get('irs_2_valid', False)),
        (state.get('irs_3_vs_fpm'), state.get('irs_3_valid', False)),
    ])
    
    # Ground tracking
    groundspeed = state.get('groundspeed_kt', 0) or 0
    on_ground = state.get('on_ground', False)
    wow = state.get('weight_on_wheels', False)
    
    # Engine status
    eng1_run = state.get('engine_1_running', True)
    eng2_run = state.get('engine_2_running', True)
    eng1_fire = state.get('engine_1_fire', False)
    eng2_fire = state.get('engine_2_fire', False)
    eng1_n1 = state.get('engine_1_n1_pct', 50) or 0
    eng2_n1 = state.get('engine_2_n1_pct', 50) or 0
    eng1_thrust = state.get('engine_1_thrust_pct', 0) or 0
    eng2_thrust = state.get('engine_2_thrust_pct', 0) or 0
    
    # Targets
    vref = state.get('target_vref_kt', 137) or 137
    target_ias = state.get('target_ias_kt', vref + 5) or (vref + 5)
    target_hdg = state.get('target_hdg_deg', 0)
    
    # DME
    dme = state.get('dme_1_nm', 10) or state.get('dme_2_nm', 10) or 10
    
    # === DEFAULTS FOR NONE VALUES ===
    gs = gs_val if gs_val is not None else 0
    loc = loc_val if loc_val is not None else 0
    ias = ias_val if ias_val is not None else target_ias
    pitch = pitch_val if pitch_val is not None else 0
    roll = roll_val if roll_val is not None else 0
    vs = vs_val if vs_val is not None else -700
    heading = heading_val if heading_val is not None else 0
    aoa = aoa_val if aoa_val is not None else 0
    
    # === COMPUTED VALUES ===
    
    # Compute vertical speed from radio altimeter
    if _state['last_radio_alt'] is not None and dt > 0.001:
        computed_vs_raw = (_state['last_radio_alt'] - radio_alt) / dt * 60.0
        # Low-pass filter
        _state['computed_vs'] = 0.6 * _state['computed_vs'] + 0.4 * computed_vs_raw
    elif _state['last_radio_alt'] is None:
        _state['computed_vs'] = vs
    _state['last_radio_alt'] = radio_alt
    
    computed_vs = _state['computed_vs'] if abs(_state['computed_vs']) > 10 else vs
    
    # Smooth ILS signals (handle flickering/scattering)
    smooth_alpha = 0.4 if radio_alt > 200 else 0.5  # Faster response near ground
    _state['smoothed_gs'] = smooth_alpha * gs + (1 - smooth_alpha) * _state['smoothed_gs']
    _state['smoothed_loc'] = smooth_alpha * loc + (1 - smooth_alpha) * _state['smoothed_loc']
    smooth_gs = _state['smoothed_gs']
    smooth_loc = _state['smoothed_loc']
    
    # Detect engine failures
    if not eng1_run or eng1_n1 < 5 or eng1_fire:
        _state['eng1_failed'] = True
    if not eng2_run or eng2_n1 < 5 or eng2_fire:
        _state['eng2_failed'] = True
    
    # Dual engine failure detection
    dual_eng_fail = _state['eng1_failed'] and _state['eng2_failed']
    single_eng_fail = _state['eng1_failed'] != _state['eng2_failed']
    
    # Speed trend
    if _state['last_ias'] is not None and dt > 0.001:
        speed_trend = (ias - _state['last_ias']) / dt
    else:
        speed_trend = 0
    _state['last_ias'] = ias
    
    # === PHASE DETERMINATION ===
    
    if on_ground or wow:
        phase = 'rollout'
    elif _state['go_around']:
        phase = 'go_around'
    elif radio_alt < 45 and radio_alt > 0:
        phase = 'flare'
    elif radio_alt < 200:
        phase = 'final'
    elif radio_alt < 1500:
        phase = 'approach'
    else:
        phase = 'intercept'
    
    _state['phase'] = phase
    
    # === GO-AROUND DECISION LOGIC ===
    
    go_around = False
    
    # Below decision height (200ft) - strict stability required
    if 50 < radio_alt < 200:
        # Glideslope deviation > 1 dot with valid signal
        if abs(smooth_gs) > 1.0 and gs_cnt >= 1:
            go_around = True
        # Localizer deviation > 1.5 dots
        if abs(smooth_loc) > 1.5 and loc_cnt >= 1:
            go_around = True
        # Speed outside limits (Vref-5 to Vref+20)
        if ias_cnt >= 1:
            if ias > vref + 20 or ias < vref - 5:
                go_around = True
        # Excessive descent rate
        if computed_vs < -1200:
            go_around = True
        # Climb during approach (not normal)
        if computed_vs > 200 and not dual_eng_fail:
            go_around = True
    
    # Engine fire
    if eng1_fire or eng2_fire:
        go_around = True
    
    # No valid ILS signals below 500ft
    if radio_alt < 500 and gs_cnt == 0 and loc_cnt == 0:
        # Check if we can continue visually (CAT I) or must go around (CAT III)
        if radio_alt < 200:
            go_around = True
    
    # Dual engine failure - cannot go around, must land
    if dual_eng_fail:
        go_around = False
    
    # Set go-around flag
    if go_around and not _state['go_around']:
        _state['go_around'] = True
    
    # === GO-AROUND EXECUTION ===
    
    if _state['go_around']:
        # TOGA thrust
        if not _state['eng1_failed']:
            out['throttle_1'] = 1.0
        if not _state['eng2_failed']:
            out['throttle_2'] = 1.0
        
        # Pitch up for climb
        target_pitch = 15.0 if radio_alt < 500 else 10.0
        pitch_error = target_pitch - pitch
        out['elevator'] = _clamp(pitch_error * 0.05, 0.1, 0.5)
        
        # Wings level
        out['aileron'] = _clamp(-roll / 20.0, -0.3, 0.3)
        
        # Clean up
        out['speedbrake'] = 0.0
        out['gear'] = 1.0  # Leave down initially
        out['flaps'] = 20.0
        
        # Rudder for engine failure
        if _state['eng1_failed'] and not _state['eng2_failed']:
            out['rudder'] = -0.35
        elif _state['eng2_failed'] and not _state['eng1_failed']:
            out['rudder'] = 0.35
        
        return out
    
    # === FLARE PHASE ===
    
    if phase == 'flare':
        if not _state['flare_started']:
            _state['flare_started'] = True
            _state['flare_entry_alt'] = radio_alt
            _state['flare_entry_vs'] = computed_vs
        
        # Progressive flare - reduce descent rate for soft touchdown
        # Target ~100 fpm at touchdown, linearly decreasing from entry
        
        entry_vs = _state.get('flare_entry_vs', -700)
        progress = 1.0 - (radio_alt / max(_state['flare_entry_alt'], 45))
        target_vs = entry_vs * (1.0 - progress) + (-100) * progress
        
        # More aggressive as we get closer
        if radio_alt < 20:
            target_vs = -50 - radio_alt * 2
        
        vs_error = target_vs - computed_vs
        
        # Pitch command
        elev = vs_error / 350.0
        
        # Add progressive pitch up (flare)
        flare_pitch = (45 - radio_alt) / 45.0 * 0.4
        elev += flare_pitch
        
        # Speed management - if too fast, more flare
        if ias > vref + 10:
            elev += 0.1
        
        # Protection against tailstrike
        if pitch > 12:
            elev -= 0.2
        
        out['elevator'] = _clamp(elev, -0.2, 0.6)
        
        # Idle thrust
        out['throttle_1'] = 0.0
        out['throttle_2'] = 0.0
        
        # Wings level in flare
        out['aileron'] = _clamp(-roll / 15.0, -0.2, 0.2)
    
    # === ROLLOUT PHASE ===
    
    elif phase == 'rollout':
        # Spoilers deploy on touchdown
        out['spoiler'] = 1.0
        
        # Autobrake
        out['autobrake'] = 0.85
        
        # Nose down to hold nosewheel on ground
        out['elevator'] = -0.2
        
        # Reverse thrust
        if groundspeed > 80:
            if not _state['eng1_failed']:
                out['reverse_1'] = 1.0
            if not _state['eng2_failed']:
                out['reverse_2'] = 1.0
        elif groundspeed > 40:
            if not _state['eng1_failed']:
                out['reverse_1'] = 0.6
            if not _state['eng2_failed']:
                out['reverse_2'] = 0.6
        
        # Maintain centerline
        out['aileron'] = _clamp(-roll / 15.0, -0.15, 0.15)
    
    # === APPROACH PHASE ===
    
    else:
        # === PITCH CONTROL (GLIDESLOPE TRACKING) ===
        
        # Glideslope error (+ = above, - = below)
        gs_error = -smooth_gs  # Invert: below GS needs pitch up
        
        # Target vertical speed based on glideslope error
        base_vs = -700  # Typical 3-degree GS
        target_vs = base_vs - gs_error * 200  # Correct for deviation
        
        # VS error
        vs_error = (target_vs - computed_vs) / 600.0
        
        # Pitch command from glideslope
        elev = gs_error * 0.12 + vs_error * 0.3
        
        # Speed correction (secondary)
        speed_error = ias - target_ias
        speed_corr = -speed_error / 120.0
        elev += speed_corr * 0.15
        
        # Integrator for steady-state error
