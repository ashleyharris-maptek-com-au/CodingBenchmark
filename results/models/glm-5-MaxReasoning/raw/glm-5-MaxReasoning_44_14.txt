```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate Lambertian diffuse factor
    float NdotL = max(dot(N, L), 0.0);
    float lambertianFactor = 0.3 + 0.7 * NdotL;
    
    // Base color with lighting applied
    vec3 baseColor = vec3(0.8, 0.2, 0.2) * lambertianFactor;
    
    // Calculate distance from camera for fog
    float dist = length(worldPos - ubo.cameraPos.xyz);
    
    // Calculate fog factor
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);
    
    // Define fog color
    vec3 fogColor = vec3(0.7, 0.7, 0.8);
    
    // Mix base color with fog
    vec3 finalColor = mix(baseColor, fogColor, fogFactor);
    
    outColor = vec4(finalColor, 1.0);
}
```
