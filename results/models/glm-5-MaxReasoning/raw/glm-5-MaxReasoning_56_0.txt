```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const { depot, truckCount, packages } = data;
const n = packages.length;

if (n === 0) {
    console.log(JSON.stringify({ routes: [] }));
    process.exit(0);
}

const k = Math.min(truckCount, n);

// Location map for O(1) lookups
const loc = new Map(packages.map(p => [p.id, { x: p.x, y: p.y }]));

// Distance function
const dist = (x1, y1, x2, y2) => Math.hypot(x1 - x2, y1 - y2);

// Precompute depot distances
const depotDist = new Map(packages.map(p => [p.id, dist(depot[0], depot[1], p.x, p.y)]));

// Package distance with optional caching
const useCache = n <= 5000;
const cache = useCache ? new Map() : null;

const pkgDist = (a, b) => {
    if (useCache) {
        const key = a < b ? `${a},${b}` : `${b},${a}`;
        let d = cache.get(key);
        if (d === undefined) {
            const la = loc.get(a), lb = loc.get(b);
            d = dist(la.x, la.y, lb.x, lb.y);
            cache.set(key, d);
        }
        return d;
    }
    const la = loc.get(a), lb = loc.get(b);
    return dist(la.x, la.y, lb.x, lb.y);
};

// Calculate total route distance
const routeDist = (r) => {
    if (!r.length) return 0;
    let d = depotDist.get(r[0]) + depotDist.get(r[r.length - 1]);
    for (let i = 1; i < r.length; i++) d += pkgDist(r[i - 1], r[i]);
    return d;
};

// 2-opt local search optimization
const twoOpt = (r) => {
    if (r.length < 2) return r;
    let best = [...r], bestD = routeDist(best);
    let improved = true;
    
    while (improved) {
        improved = false;
        for (let i = 0; i < best.length - 1 && !improved; i++) {
            for (let j = i + 1; j < best.length; j++) {
                const cand = [
                    ...best.slice(0, i),
                    ...best.slice(i, j + 1).reverse(),
                    ...best.slice(j + 1)
                ];
                const candD = routeDist(cand);
                if (candD < bestD - 1e-9) {
                    best = cand;
                    bestD = candD;
                    improved = true;
                    break;
                }
            }
        }
    }
    return best;
};

// Nearest neighbor TSP heuristic
const nearestNeighbor = (ids) => {
    if (ids.length <= 1) return ids.slice();
    
    const remaining = new Set(ids);
    const route = [];
    
    // Start from package furthest from depot
    let current = ids[0], maxD = depotDist.get(current);
    for (const id of ids) {
        const d = depotDist.get(id);
        if (d > maxD) { maxD = d; current = id; }
    }
    
    route.push(current);
    remaining.delete(current);
    
    while (remaining.size) {
        let best = null, bestD = Infinity;
        for (const id of remaining) {
            const d = pkgDist(current, id);
            if (d < bestD) { bestD = d; best = id; }
        }
        route.push(best);
        remaining.delete(best);
        current = best;
    }
    
    return route;
};

// Sweep algorithm for partitioning packages into routes
const sweep = () => {
    // Calculate angle from depot for each package
    const sorted = packages.map(p => ({
        id: p.id,
        angle: Math.atan2(p.y - depot[1], p.x - depot[0])
    }));
    
    // Sort by angle (deterministic ordering)
    sorted.sort((a, b) => {
        if (a.angle !== b.angle) return a.angle - b.angle;
        // Tie-breaker for consistent ordering
        return depotDist.get(a.id) - depotDist.get(b.id);
    });
    
    const perRoute = Math.ceil(n / k);
    const routes = [];
    
    for (let i = 0; i < k && i * perRoute < n; i++) {
        const start = i * perRoute;
        const end = Math.min(start + perRoute, n);
        routes.push(sorted.slice(start, end).map(p => p.id));
    }
    
    return routes;
};

// Build and optimize routes
let routes = sweep();
routes = routes.map(r => twoOpt(nearestNeighbor(r)));

// Output result
console.log(JSON.stringify({ routes }));
```
