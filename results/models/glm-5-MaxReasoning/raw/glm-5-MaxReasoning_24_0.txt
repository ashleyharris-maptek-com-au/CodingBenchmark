// Utility functions
inline double clamp(double val, double min_val, double max_val) {
    return val < min_val ? min_val : (val > max_val ? max_val : val);
}

inline double normalizeAngle(double angle) {
    while (angle > PI) angle -= 2.0 * PI;
    while (angle < -PI) angle += 2.0 * PI;
    return angle;
}

inline double sqr(double x) { return x * x; }
inline double mag(double x, double y) { return std::sqrt(x * x + y * y); }
inline double mag(double x, double y, double z) { return std::sqrt(x * x + y * y + z * z); }

// Terrain class with memory mapping for large files
class Terrain {
    int width_ = 0, depth_ = 0;
    
#ifdef _WIN32
    HANDLE file_handle_ = INVALID_HANDLE_VALUE;
    HANDLE mapping_handle_ = NULL;
    const uint8_t* mapped_data_ = nullptr;
#endif
    
    std::vector<uint16_t> ground_top_;
    std::vector<uint16_t> floating_bottom_;
    std::vector<uint16_t> floating_top_;
    bool use_vector_ = false;
    
public:
    std::atomic<bool> loaded{false};
    std::atomic<bool> loading{false};
    
    ~Terrain() {
#ifdef _WIN32
        if (mapped_data_) UnmapViewOfFile(mapped_data_);
        if (mapping_handle_) CloseHandle(mapping_handle_);
        if (file_handle_ != INVALID_HANDLE_VALUE) CloseHandle(file_handle_);
#endif
    }
    
    bool load(const std::string& path, int width, int depth) {
        loading = true;
        width_ = width;
        depth_ = depth;
        size_t expected_size = static_cast<size_t>(width) * depth * 6;
        
#ifdef _WIN32
        // Try memory mapping for large files
        file_handle_ = CreateFileA(
            path.c_str(), GENERIC_READ, FILE_SHARE_READ,
            NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL | FILE_FLAG_SEQUENTIAL_SCAN, NULL
        );
        
        if (file_handle_ != INVALID_HANDLE_VALUE) {
            LARGE_INTEGER file_size;
            GetFileSizeEx(file_handle_, &file_size);
            
            if (static_cast<size_t>(file_size.QuadPart) >= expected_size) {
                mapping_handle_ = CreateFileMappingA(file_handle_, NULL, PAGE_READONLY, 0, 0, NULL);
                if (mapping_handle_) {
                    mapped_data_ = static_cast<const uint8_t*>(
                        MapViewOfFile(mapping_handle_, FILE_MAP_READ, 0, 0, 0)
                    );
                    if (mapped_data_) {
                        use_vector_ = false;
                        loaded = true;
                        loading = false;
                        return true;
                    }
                    CloseHandle(mapping_handle_);
                    mapping_handle_ = NULL;
                }
            }
            CloseHandle(file_handle_);
            file_handle_ = INVALID_HANDLE_VALUE;
        }
#endif
        
        // Fallback: load into memory
        std::ifstream file(path, std::ios::binary);
        if (!file) {
            loaded = true;
            loading = false;
            return false;
        }
        
        size_t num_cells = static_cast<size_t>(width) * depth;
        ground_top_.resize(num_cells);
        floating_bottom_.resize(num_cells);
        floating_top_.resize(num_cells);
        
        std::vector<uint8_t> buf(6);
        for (int x = 0; x < width; x++) {
            for (int y = 0; y < depth; y++) {
                size_t idx = static_cast<size_t>(x) * depth + y;
                file.read(reinterpret_cast<char*>(buf.data()), 6);
                ground_top_[idx] = static_cast<uint16_t>(buf[0] | (buf[1] << 8));
                floating_bottom_[idx] = static_cast<uint16_t>(buf[2] | (buf[3] << 8));
                floating_top_[idx] = static_cast<uint16_t>(buf[4] | (buf[5] << 8));
            }
        }
        
        use_vector_ = true;
        loaded = true;
        loading = false;
        return true;
    }
    
    inline std::tuple<uint16_t, uint16_t, uint16_t> getTerrainAt(int x, int y) const {
        if (x < 0 || x >= width_ || y < 0 || y >= depth_) {
            return {0, 0, 0};
        }
        size_t idx = static_cast<size_t>(x) * depth_ + y;
        
        if (use_vector_) {
            return {ground_top_[idx], floating_bottom_[idx], floating_top_[idx]};
        }
        
#ifdef _WIN32
        if (mapped_data_) {
            const uint8_t* p = mapped_data_ + idx * 6;
            return {
                static_cast<uint16_t>(p[0] | (p[1] << 8)),
                static_cast<uint16_t>(p[2] | (p[3] << 8)),
                static_cast<uint16_t>(p[4] | (p[5] << 8))
            };
        }
#endif
        return {0, 0, 0};
    }
    
    bool isSolid(double x, double y, double z) const {
        int ix = static_cast<int>(std::floor(x));
        int iy = static_cast<int>(std::floor(y));
        int iz = static_cast<int>(std::floor(z));
        
        if (ix < 0 || ix >= width_ || iy < 0 || iy >= depth_ || iz < 0) return true;
        
        auto [a, b, c] = getTerrainAt(ix, iy);
        if (iz < static_cast<int>(a)) return true;
        if (iz >= static_cast<int>(b) && iz < static_cast<int>(c)) return true;
        return false;
    }
    
    double getGroundHeight(double x, double y) const {
        int ix = static_cast<int>(std::floor(x));
        int iy = static_cast<int>(std::floor(y));
        if (ix < 0 || ix >= width_ || iy < 0 || iy >= depth_) return 0.0;
        auto [a, b, c] = getTerrainAt(ix, iy);
        return static_cast<double>(a);
    }
    
    double getMaxClearance(double x, double y, double z) const {
        int ix = static_cast<int>(std::floor(x));
        int iy = static_cast<int>(std::floor(y));
        if (ix < 0 || ix >= width_ || iy < 0 || iy >= depth_) return z;
        
        auto [a, b, c] = getTerrainAt(ix, iy);
        int iz = static_cast<int>(std::floor(z));
        
        // Return distance to nearest obstacle above
        if (iz >= static_cast<int>(b) && iz < static_cast<int>(c)) {
            return static_cast<double>(c) - z; // Inside floating rock
        }
        if (z >= a && b > a && z < b) {
            return static_cast<double>(b) - z; // Below floating rock
        }
        return std::numeric_limits<double>::max();
    }
    
    // Check terrain in a small region for collision avoidance
    bool checkCollisionRegion(double cx, double cy, double cz, double radius) const {
        int min_x = std::max(0, static_cast<int>(cx - radius));
        int max_x = std::min(width_ - 1, static_cast<int>(cx + radius));
        int min_y = std::max(0, static_cast<int>(cy - radius));
        int max_y = std::min(depth_ - 1, static_cast<int>(cy + radius));
        
        for (int x = min_x; x <= max_x; x++) {
            for (int y = min_y; y <= max_y; y++) {
                auto [a, b, c] = getTerrainAt(x, y);
                int iz = static_cast<int>(cz);
                if (iz < static_cast<int>(a)) return true;
                if (iz >= static_cast<int>(b) && iz < static_cast<int>(c)) return true;
            }
        }
        return false;
    }
};

// Thread-safe command output
class CommandQueue {
    std::mutex mtx;
    std::atomic<double> thrust_{0}, pitch_{0}, yaw_{0}, roll_{0};
    
public:
    void set(double t, double p, double y, double r) {
        std::lock_guard<std::mutex> lock(mtx);
        thrust_ = t; pitch_ = p; yaw_ = y; roll_ = r;
    }
    
    void get(double& t, double& p, double& y, double& r) {
        std::lock_guard<std::mutex> lock(mtx);
        t = thrust_; p = pitch_; y = yaw_; r = roll_;
    }
};

// Advanced controller with adaptive gains
class LanderController {
    // Adaptive gains based on distance
    double base_kp_pos = 0.3;
    double base_kd_pos = 0.8;
    double base_kp_att = 4.0;
    double base_kd_att = 2.5;
    
    double approach_speed_factor = 0.3;
    double max_approach_speed = 25.0;
    double descent_speed = 2.0;
    
public:
    void compute(
        double x, double y, double z,
        double vx, double vy, double vz,
        double pitch, double yaw, double roll,
        double pitch_rate, double yaw_rate, double roll_rate,
        double fuel,
        double& out_thrust, double& out_pitch, double& out_yaw, double& out_roll
    ) {
        // Target
        double tx = world_params.target_x;
        double ty = world_params.target_y;
        double tz = world_params.target_z;
        
        // Errors
        double ex = tx - x;
        double ey = ty - y;
        double ez = tz - z;
        
        double dist_h = mag(ex, ey);
        double dist_v = std::abs(ez);
        double dist_total = mag(ex, ey, ez);
        
        // Current speed
        double speed_h = mag(vx, vy);
        double speed_v = std::abs(vz);
        double speed_total = mag(vx, vy, vz);
        
        // Scale-adjusted parameters
        double scale = world_params.scale;
        double scaled_approach_speed = std::min(max_approach_speed * std::sqrt(scale), 50.0);
        
        // Determine control phase
        double desired_vx, desired_vy, desired_vz;
        
        // Phase 1: Approach - move towards target horizontally
        if (dist_h > 50.0 * scale) {
            double approach_speed = std::min(dist_h * approach_speed_factor, scaled_approach_speed);
            
            // Slow down if approaching too fast
            if (speed_h > approach_speed * 1.2) {
                approach_speed = speed_h * 0.8;
            }
            
            desired_vx = approach_speed * ex / dist_h;
            desired_vy = approach_speed * ey / dist_h;
            
            // Maintain safe altitude
            double ground_h = terrain.loaded ? terrain.getGroundHeight(x, y) : 0.0;
            double target_alt = std::max(tz + 30.0, ground_h + 50.0);
            desired_vz = clamp((target_alt - z) * 0.3, -10.0, 10.0);
        }
        // Phase 2: Near target - hover above
        else if (dist_h > 10.0 || dist_v > 20.0) {
            double factor = dist_h / 50.0;
            double horiz_speed = std::min(speed_h + 2.0, 8.0) * factor;
            
            desired_vx = clamp(ex * 0.4, -horiz_speed, horiz_speed);
            desired_vy = clamp(ey * 0.4, -horiz_speed, horiz_speed);
            
            // Hover above landing zone
            double hover_alt = tz + 15.0;
            desired_vz = clamp((hover_alt - z) * 0.4, -5.0, 5.0);
        }
        // Phase 3: Final approach and landing
        else {
            // Reduce horizontal velocity
            desired_vx = clamp(ex * 0.5, -2.0, 2.0);
            desired_vy = clamp(ey * 0.5, -2.0, 2.0);
            
            // Controlled descent
            if (speed_h < 1.5 && dist_h < 5.0) {
                // Final descent
                if (ez > 0) {
                    desired_vz = clamp(-descent_speed, -2.0, 0.5);
                } else {
                    desired_vz = clamp(-vz * 0.5, -0.5, 0.5);
                }
            } else {
                // Hover
                desired_vz = clamp((tz + 5.0 - z) * 0.3, -3.0, 3.0);
            }
        }
        
        // Terrain avoidance
        if (terrain.loaded) {
            double ground_h = terrain.getGroundHeight(x, y);
            double clearance = z - ground_h;
            
            // Emergency climb if too low
            if (clearance < 20.0) {
                double urgency = (20.0 - clearance) / 20.0;
                desired_vz = std::max(desired_vz, 3.0 + urgency * 5.0);
            }
            
            // Check floating rocks
            double ceiling = terrain.getMaxClearance(x, y, z);
            if (ceiling < 15.0) {
                desired_vz = std::min(desired_vz, 1.0);
            }
            
            // Check if on collision course
            double look_ahead = 20.0;
            double future_x = x + vx * look_ahead / speed_h;
            double future_y = y + vy * look_ahead / speed_h;
            if (speed_h > 1.0) {
                double future_ground = terrain.getGroundHeight(future_x, future_y);
                if (future_ground > z - 20.0) {
                    // Terrain rising ahead, climb
                    desired_vz = std::max(desired_vz, 3.0);
                }
            }
        }
        
        // Velocity errors
        double evx = desired_vx - vx;
        double evy = desired_vy - vy;
        double evz = desired_vz - vz;
        
        // Desired accelerations
        double ax = evx * base_kd_pos;
        double ay = evy * base_kd_pos;
        double az = evz * base_kd_pos + world_params.gravity;
        
        // Compute thrust vector
        double thrust_mag = mag(ax, ay, az);
        
        // Thrust magnitude (as fraction of max)
        double thrust_frac = thrust_mag / world_params.max_thrust;
        out_thrust = clamp(thrust_frac, 0.0, 1.0);
        
        // Compute required attitude
        double required_pitch = 0.0;
        double required_yaw = yaw;
        
        if (thrust_mag > 0.01 * world_params.max_thrust) {
            // Normalized thrust direction
            double tnx = ax / thrust_mag;
            double tny = ay / thrust_mag;
            double tnz = az / thrust_mag;
            
            // Required pitch (tilt from vertical)
            double horiz = mag(tnx, tny);
            required_pitch = std::atan2(horiz, tnz);
            required_pitch = clamp(required_pitch, -MAX_PITCH, MAX_PITCH);
            
            // Required yaw
            if (horiz > 0.01) {
                // From thrust model: thrust_x = -sin(pitch)*cos(yaw), thrust_y = -sin(pitch)*sin(yaw)
                required_yaw = std::atan2(-tny, -tnx);
            }
        }
        
        // Attitude control with anti-windup
        double pitch_err = normalizeAngle(required_pitch - pitch);
        double yaw_err = normalizeAngle(required_yaw - yaw);
        double roll_err = normalizeAngle(-roll); // Keep roll at zero
        
        // PD control for attitude
