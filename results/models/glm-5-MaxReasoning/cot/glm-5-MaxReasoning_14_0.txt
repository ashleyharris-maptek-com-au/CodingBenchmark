This is a challenging problem - building a Minesweeper solver that handles boards from trivial (5x5) to massive (10000x10000) scales. Let me think through the approach:

## Key Challenges:
1. **Scale**: 10000x10000 = 100 million cells - need efficient data structures
2. **Constraint Satisfaction**: Complex boards require CSP solving
3. **Efficiency**: Must use appropriate algorithms for different scales
4. **Safety**: Must never return a cell that could be a mine

## Strategy Hierarchy:

### Level 1: Simple Rule-Based (O(cells) per pass)
- If a number cell has all mines accounted for, remaining unknown neighbors are safe
- If a number cell has exactly as many unknown neighbors as its count minus flagged neighbors, those are mines (and other cells become safe via the first rule)

### Level 2: Local Constraint Satisfaction
- For smaller boards or dense regions, solve local CSP
- Group cells into connected components based on shared constraints
- Use Gaussian elimination over GF(2) or subset enumeration

### Level 3: Global CSP
- For smaller boards, use full constraint satisfaction
- Could use SAT solver or constraint enumeration

### Level 4: Advanced Techniques
- Tank algorithm (enumerate all valid mine configurations)
- Gaussian elimination for mine probability
- Connected component analysis

## Implementation Plan:

1. **Input parsing**: Read board efficiently
2. **Data structures**: 
   - Use 1D arrays for cache efficiency
   - Sparse representation for large boards
3. **Strategy selection**: Based on board size and density
4. **Rule application**: Apply deterministic rules first
5. **CSP solving**: For appropriate complexity levels

```csharp
using System.Collections.Generic;
using System.Linq;

public class MinesweeperSolver
{
    private static int H, W;
    private static char[,] board;
    private static bool[,] safe, mine;
    private static List<(int r, int c)> frontier; // Unknown cells adjacent to numbers
    
    private static readonly int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    private static readonly int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };
    
    public static void Main()

        ReadInput();
        var result = Solve();
        OutputResult(result);
    }
    
    private static void ReadInput()
    {
        var parts = Console.ReadLine().Split();
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);
        
        board = new char[H, W];
        
        for (int r = 0; r < H; r++)
        {
            string line = Console.ReadLine();
            int len = Math.Min(line.Length, W);
            for (int c = 0; c < len; c++)
                board[r, c] = line[c];
            for (int c = len; c < W; c++)
                board[r, c] = ' ';
        }
    }
    
    private static HashSet<(int r, int c)> Solve()
    {
        safe = new bool[H, W];
        mine = new bool[H, W];
        
        // Initialize known mines
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == '*')
                    mine[r, c] = true;
        
        // Find frontier (unknowns adjacent to numbers)
        frontier = new List<(int r, int c)>();
        var frontierSet = new HashSet<(int r, int c)>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] == ' ')
                {
                    bool isFrontier = false;
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dr[d], nc = c + dc[d];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            if (board[nr, nc] >= '1' && board[nr, nc] <= '8')
                            {
                                isFrontier = true;
                                break;
                            }
                        }
                    }
                    if (isFrontier && frontierSet.Add((r, c)))
                        frontier.Add((r, c));
                }
            }
        }
        
        // Apply simple rules repeatedly
        while (ApplySimpleRules()) { }
        
        // For smaller frontiers, use constraint enumeration
        if (frontier.Count > 0 && frontier.Count <= 50)
        {
            EnumerateConfigurations();
        }
        
        // Collect safe cells
        var result = new HashSet<(int r, int c)>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == ' ' && safe[r, c])
                    result.Add((r, c));
        
        return result;
    }
    
    private static bool ApplySimpleRules()
    {
        bool changed = false;
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] >= '1' && board[r, c] <= '8')
                {
                    changed |= ProcessNumberCell(r, c);
                }
            }
        }
        
        return changed;
    }
    
    private static bool ProcessNumberCell(int r, int c)
    {
        int target = board[r, c] - '0';
        var unknowns = new List<(int r, int c)>();
        int mineCount = 0;
        
        for (int d = 0; d < 8; d++)
        {
            int nr = r + dr[d], nc = c + dc[d];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
            {
                if (board[nr, nc] == ' ')
                {
                    if (!safe[nr, nc] && !mine[nr, nc])
                        unknowns.Add((nr, nc));
                    else if (mine[nr, nc])
                        mineCount++;
                }
                else if (board[nr, nc] == '*')
                {
                    mineCount++;
                }
            }
        }
        
        bool changed = false;
        int remaining = target - mineCount;
        
        // All mines accounted for - remaining unknowns are safe
        if (remaining == 0)
        {
            foreach (var (ur, uc) in unknowns)
            {
                if (!safe[ur, uc])
                {
                    safe[ur, uc] = true;
                    changed = true;
                }
            }
        }
        // Exactly enough unknowns for remaining mines - all are mines
        else if (unknowns.Count == remaining)
        {
            foreach (var (ur, uc) in unknowns)
            {
                if (!mine[ur, uc])
                {
                    mine[ur, uc] = true;
                    changed = true;
                }
            }
        }
        
        return changed;
    }
    
    private static void EnumerateConfigurations()
    {
        // Build constraints from frontier cells
        var constraints = new List<(List<int> cells, int count)>();
        var cellToIndex = new Dictionary<(int r, int c), int>();
        
        // Map frontier cells to indices
        var activeFrontier = new List<(int r, int c)>();
        foreach (var (r, c) in frontier)
        {
            if (!safe[r, c] && !mine[r, c])
            {
                cellToIndex[(r, c)] = activeFrontier.Count;
                activeFrontier.Add((r, c));
            }
        }
        
        if (activeFrontier.Count == 0) return;
        
        // Build constraints from numbered cells
        var usedNumbers = new HashSet<(int r, int c)>();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] >= '1' && board[r, c] <= '8')
                {
                    int target = board[r, c] - '0';
                    var cells = new List<int>();
                    int mineCount = 0;
                    
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dr[d], nc = c + dc[d];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            if (board[nr, nc] == ' ')
                            {
                                if (mine[nr, nc])
                                    mineCount++;
                                else if (cellToIndex.TryGetValue((nr, nc), out int idx) && !safe[nr, nc])
                                    cells.Add(idx);
                            }
                            else if (board[nr, nc] == '*')
                                mineCount++;
                        }
                    }
                    
                    int remaining = target - mineCount;
                    if (cells.Count > 0 && remaining >= 0)
                    {
                        constraints.Add((cells, remaining));
                        usedNumbers.Add((r, c));
                    }
                }
            }
        }
        
        if (constraints.Count == 0) return;
        
        // Enumerate all valid configurations
        int n = activeFrontier.Count;
        var validConfigs = new List<bool[]>();
        
        EnumerateRecursive(0, new bool[n], constraints, validConfigs);
        
        if (validConfigs.Count == 0) return;
        
        // Check which cells are always safe or always mine
        for (int i = 0; i < n; i++)
        {
            bool alwaysMine = true, alwaysSafe = true;
            foreach (var config in validConfigs)
            {
                if (config[i]) alwaysSafe = false;
                else alwaysMine = false;
            }
            
            var (cr, cc) = activeFrontier[i];
            if (alwaysSafe)
                safe[cr, cc] = true;
            else if (alwaysMine)
                mine[cr, cc] = true;
        }
    }
    
    private static void EnumerateRecursive(int idx, bool[] config, 
        List<(List<int> cells, int count)> constraints, List<bool[]> validConfigs)
    {
        if (idx == config.Length)
        {
            validConfigs.Add((bool[])config.Clone());
            return;
        }
        
        // Try safe
        config[idx] = false;
        if (IsValidPartial(config, idx, constraints))
            EnumerateRecursive(idx + 1, config, constraints, validConfigs);
        
        // Try mine
        config[idx] = true;
        if (IsValidPartial(config, idx, constraints))
            EnumerateRecursive(idx + 1, config, constraints, validConfigs);
    }
    
    private static bool IsValidPartial(bool[] config, int idx,
        List<(List<int> cells, int count)> constraints)
    {
        foreach (var (cells, target) in constraints)
        {
            int mines = 0, unknowns = 0;
            foreach (int c in cells)
            {
                if (c > idx) unknowns++;
                else if (config[c]) mines++;
            }
            if (mines > target || mines + unknowns < target)
                return false;
        }
        return true;
    }
    
    private static void OutputResult(HashSet<(int r, int c)> cells)
    {
        Console.WriteLine(cells.Count);
        foreach (var (r, c) in cells)
            Console.WriteLine($"{c} {r}"); // x y = column row
    }
```

```csharp
using System.Collections.Generic;
using System.Linq;

class Program
{
    static int H, W;
    static char[,] B;
    static bool[,] Safe, Mine;
    static readonly int[] DR = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] DC = { -1, 0, 1, -1, 1, -1, 0, 1 };
    
    static void Main()
    {
        Read();
        var result = Solve();
        Console.WriteLine(result.Count);
        foreach (var (x, y) in result)
            Console.WriteLine($"{x} {y}");
    }
    
    static void Read()
    {
        var p = Console.ReadLine().Split();
        H = int.Parse(p[0]);
        W = int.Parse(p[1]);
        B = new char[H, W];
        
        for (int r = 0; r < H; r++)
        {
            var line = Console.ReadLine() ?? "";
            for (int c = 0; c < W; c++)
                B[r, c] = c < line.Length ? line[c] : ' ';
        }
    }
    
    static List<(int x, int y)> Solve()
    {
        Safe = new bool[H, W];
        Mine = new bool[H, W];
        
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (B[r, c] == '*')
                    Mine[r, c] = true;
        
        // Build constraints and solve
        while (ApplyRules()) { }
        
        // Advanced solving for remaining frontier
        SolveFrontier();
        
        var result = new List<(int, int)>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (B[r, c] == ' ' && Safe[r, c])
                    result.Add((c, r));
        
        return result;
    }
    
    static bool ApplyRules()
    {
        bool changed = false;
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (B[r, c] >= '1' && B[r, c] <= '8')
                    changed |= ProcessCell(r, c);
        return changed;
    }
    
    static bool ProcessCell(int r, int c)
    {
        int target = B[r, c] - '0';
        var unk = new List<(int, int)>();
        int mines = 0;
        
        for (int d = 0; d < 8; d++)
        {
            int nr = r + DR[d], nc = c + DC[d];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
            {
                if (B[nr, nc] == ' ')
                {
                    if (!Safe[nr, nc] && !Mine[nr, nc])
                        unk.Add((nr, nc));
                    else if (Mine[nr, nc])
                        mines++;
                }
                else if (B[nr, nc] == '*')
                    mines++;
            }
        }
        
        bool changed = false;
        int remaining = target - mines;
        
        if (remaining == 0)
        {
            foreach (var (ur, uc) in unk)
                if (!Safe[ur, uc]) { Safe[ur, uc] = true; changed = true; }
        }
        else if (unk.Count == remaining && remaining > 0)
        {
            foreach (var (ur, uc) in unk)
                if (!Mine[ur, uc]) { Mine[ur, uc] = true; changed = true; }
        }
        
        return changed;
    }
    
    static void SolveFrontier()
    {
        // Find frontier cells
        var frontier = new List<int>();
        var idx = new Dictionary<int, int>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (B[r, c] == ' ' && !Safe[r, c] && !Mine[r, c])
                {
                    bool isFrontier = false;
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + DR[d], nc = c + DC[d];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                            if (B[nr, nc] >= '1' && B[nr, nc] <= '8')
                                { isFrontier = true; break; }

                        idx[r * W + c] = frontier.Count;
                        frontier.Add(r * W + c);
                    }
                }
            }
        }
        
        if (frontier.Count == 0 || frontier.Count > 60) return;
        
        // Build constraints
        var constraints = new List<(List<int> cells, int count)>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (B[r, c] >= '1' && B[r, c] <= '8')

                            {

                                else if (idx.TryGetValue(nr * W + nc, out int i))

        int n = frontier.Count;

            if (allSafe) Safe[fr, fc] = true;
            if (allMine) Mine[fr, fc] = true;
        }

    static void Enumerate(int i, ulong cfg, int mines, 
        List<(List<int> cells, int count)> constraints, List<ulong> valid, int n)
    {
        if (i == n)
        {
            valid.Add(cfg);
            return;
        }
        
        // Check if placing safe is valid
        cfg &= ~(1UL << i);
        if (ValidPartial(cfg, i, constraints))
            Enumerate(i + 1, cfg, mines, constraints, valid, n);
        
        // Check if placing mine is valid
        cfg |= (1UL << i);
        if (ValidPartial(cfg, i, constraints))
            Enumerate(i + 1, cfg, mines + 1, constraints, valid, n);
    }
    
    static bool ValidPartial(ulong cfg, int idx, List<(List<int> cells, int count)> constraints)
    {
        foreach (var (cells, target) in constraints)
        {
            int mines = 0, unknowns = 0;
            foreach (int c in cells)
            {
                if (c > idx) unknowns++;
                else if ((cfg & (1UL << c)) != 0) mines++;
            }
            if (mines > target || mines + unknowns < target)
                return false;
        }
        return true;
```

```csharp
        W = int.Parse(parts[1]);
        B = new char[H, W];
        
        for (int r = 0; r < H; r++)
        {
            line = Console.ReadLine() ?? "";
            for (int c = 0; c < W; c++)
                B[r, c] = c < line.Length ? line[c] : ' ';
        }
    }
    
    static List<(int x, int y)> Solve()
    {
        Safe = new bool[H, W];
        Mine = new bool[H, W];
        
        // Initialize flagged mines
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (B[r, c] == '*')
                    Mine[r, c] = true;
        
        // Phase 1: Simple rule propagation
        while (ApplySimpleRules()) { }
        
        // Phase 2: Advanced analysis (for smaller frontiers)
        int frontierSize = CountFrontier();
        if (frontierSize <= 100)
            AdvancedSolve();
        
        // Phase 3: Collect results
        var result = new List<(int x, int y)>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (B[r, c] == ' ' && Safe[r, c])
                    result.Add((c, r));
        
        return result;
    }
    
    static int CountFrontier()
    {
        int count = 0;
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (B[r, c] == ' ' && !Safe[r, c] && !Mine[r, c] && IsAdjacentToNumber(r, c))
                    count++;
        return count;
    }
    
    static bool IsAdjacentToNumber(int r, int c)
    {
        for (int d = 0; d < 8; d++)
        {
            int nr = r + DR[d], nc = c + DC[d];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                if (B[nr, nc] >= '1' && B[nr, nc] <= '8')
                    return true;
        }
        return false;
    }
    
    static bool ApplySimpleRules()
    {
        bool changed = false;
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (B[r, c] >= '1' && B[r, c] <= '8')
                {
                    changed |= ProcessNumber(r, c);
                }
            }
        }
        
        return changed;
    }
    
    static bool ProcessNumber(int r, int c)
    {
        int target = B[r, c] - '0';
        var unknowns = new List<(int r, int c)>();
        int knownMines = 0;
        
        for (int d = 0; d < 8; d++)
        {
            int nr = r + DR[d], nc = c + DC[d];
            if (nr >= 0 && nr < H && nc >= 0 && nc < W)
            {
                if (B[nr, nc] == ' ')
                {
                    if (Mine[nr, nc]) knownMines++;
                    else if (!Safe[nr, nc]) unknowns.Add((nr, nc));
                }
                else if (B[nr, nc] == '*')
                {
                    knownMines++;
                }
            }
        }
        
        bool changed = false;
        int remaining = target - knownMines;
        
        if (remaining < 0) return false; // Invalid state
        
        // All mines accounted for
        if (remaining == 0 && unknowns.Count > 0)
        {
            foreach (var (ur, uc) in unknowns)
            {
                Safe[ur, uc] = true;
                changed = true;
            }
        }
        // Exactly enough unknowns for remaining mines
        else if (unknowns.Count == remaining && remaining > 0)
        {
            foreach (var (ur, uc) in unknowns)
            {
                Mine[ur, uc] = true;
                changed = true;
            }
        }
        
        return changed;
    }
    
    static void AdvancedSolve()
    {
        // Build frontier cells and constraints
        var frontier = new List<(int r, int c)>();
        var cellToIdx = new Dictionary<(int, int), int>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (B[r, c] == ' ' && !Safe[r, c] && !Mine[r, c] && IsAdjacentToNumber(r, c))
                {
                    cellToIdx[(r, c)] = frontier.Count;
                    frontier.Add((r, c));
                }
            }
        }
        
        if (frontier.Count == 0) return;
        
        // Build constraints from number cells
        var constraints = new List<Constraint>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (B[r, c] >= '1' && B[r, c] <= '8')
                {
                    var cells = new List<int>();
                    int knownMines = 0;
                    
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + DR[d], nc = c + DC[d];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            if (B[nr, nc] == ' ')
                            {
                                if (Mine[nr, nc]) knownMines++;
                                else if (cellToIdx.TryGetValue((nr, nc), out int idx))
                                    cells.Add(idx);
                            }
                            else if (B[nr, nc] == '*') knownMines++;
                        }
                    }
                    
                    int remaining = B[r, c] - '0' - knownMines;
                    if (cells.Count > 0 && remaining >= 0)
                        constraints.Add(new Constraint { Cells = cells, Count = remaining });
                }
            }
        }
        
        if (constraints.Count == 0) return;
        
        // Decompose into connected components
        var components = FindComponents(frontier, constraints);
        
        foreach (var comp in components)
        {
            SolveComponent(comp, frontier, constraints);
        }
    }
    
    static List<HashSet<int>> FindComponents(List<(int r, int c)> frontier, List<Constraint> constraints)
    {
        var graph = new Dictionary<int, HashSet<int>>();
        for (int i = 0; i < frontier.Count; i++)
            graph[i] = new HashSet<int>();
        
        // Connect cells that share a constraint
        foreach (var c in constraints)
        {
            for (int i = 0; i < c.Cells.Count; i++)
            {
                for (int j = i + 1; j < c.Cells.Count; j++)
                {
                    graph[c.Cells[i]].Add(c.Cells[j]);
                    graph[c.Cells[j]].Add(c.Cells[i]);
                }
            }
        }
        
        var visited = new bool[frontier.Count];
        var components = new List<HashSet<int>>();
        
        for (int i = 0; i < frontier.Count; i++)
        {
            if (!visited[i])
            {
                var comp = new HashSet<int>();
                var stack = new Stack<int>();
                stack.Push(i);
                
                while (stack.Count > 0)
                {
                    int u = stack.Pop();
                    if (visited[u]) continue;
                    visited[u] = true;
                    comp.Add(u);
                    
                    foreach (int v in graph[u])
                        if (!visited[v])
                            stack.Push(v);
                }
                
                components.Add(comp);
            }
        }
        
        return components;
    }
    
    static void SolveComponent(HashSet<int> component, List<(int r, int c)> frontier, List<Constraint> constraints)
    {
        // Filter constraints relevant to this component
        var relevantConstraints = constraints
            .Where(c => c.Cells.Any(cell => component.Contains(cell)))
            .ToList();
        
        // Map component cells to local indices
        var localIdx = new Dictionary<int, int>();
        var localCells = component.ToList();
        for (int i = 0; i < localCells.Count; i++)
            localIdx[localCells[i]] = i;
        
        // Convert constraints to local indices
        var localConstraints = relevantConstraints.Select(c => 
        {
            var localCellsList = c.Cells.Where(cell => component.Contains(cell)).Select(cell => localIdx[cell]).ToList();
            return (cells: localCellsList, count: c.Count);
        }).ToList();
        
        int n = localCells.Count;
        
        // Use enumeration for small components
        if (n <= 25)
        {
            var validConfigs = new List<int[]>();
            EnumerateConfigurations(0, new int[n], localConstraints, validConfigs);
            
            if (validConfigs.Count == 0) return;
            
            // Find always safe/mine
            for (int i = 0; i < n; i++)
            {
                bool allMine = validConfigs.All(cfg => cfg[i] == 1);
                bool allSafe = validConfigs.All(cfg => cfg[i] == 0);
                
                var (r, c) = frontier[localCells[i]];
                if (allSafe) Safe[r, c] = true;
                if (allMine) Mine[r, c] = true;
            }
        }
    }
    
    static void EnumerateConfigurations(int idx, int[] config, List<(List<int> cells, int count)> constraints, List<int[]> validConfigs)
    {
        if (idx == config.Length)
        {
            if (ValidConfig(config, constraints))
                validConfigs.Add((int[])config.Clone());
            return;
        }
        
        config[idx] = 0;
        if (ValidPartial(config, idx, constraints))
            EnumerateConfigurations(idx + 1, config, constraints, validConfigs);
        
        config[idx] = 1;
        if (ValidPartial(config, idx, constraints))
            EnumerateConfigurations(idx + 1, config, constraints, validConfigs);
    }
    
    static bool ValidPartial(int[] config, int idx, List<(List<int> cells, int count)> constraints)
    {
        foreach (var (cells, count) in constraints)
        {
            int mines = 0, unknowns = 0;
            foreach (int c in cells)
            {
                if (c > idx) unknowns++;
                else if (config[c] == 1) mines++;
            }
            if (mines > count || mines + unknowns < count)
                return false;
        }
        return true;
    }
    
    static bool ValidConfig(int[] config, List<(List<int> cells, int count)> constraints)
    {
        foreach (var (cells, count) in constraints)
        {
            int mines = cells.Sum(c => config[c]);
            if (mines != count) return false;
        }
        return true;
    }
    
    static void Output(List<(int x, int y)> cells)
    {
        Console.WriteLine(cells.Count);
        foreach (var (x, y) in cells)
            Console.WriteLine($"{x} {y}");
    }
    
    class Constraint
    {
        public List<int> Cells;
        public int Count;
    }
}
```

Actually, I realize there are more optimizations I should consider:

1. **Subset analysis**: Before enumeration, check for subset relationships between constraints
2. **Gaussian elimination**: For larger systems, use linear algebra over GF(2)
3. **Efficient data structures**: Use 1D arrays for better cache performance

Let me write a more polished version that handles all these cases:

Actually, let me also consider that for very large boards, even iterating through all cells might be slow. But with 128GB RAM and the requirement to handle 10000x10000 boards, iterating through 100M cells should still be reasonably fast (maybe a second or two).

The key insight is that the "frontier" - unknown cells adjacent to numbered cells - is usually much smaller than the total board, especially as the game progresses. For efficiency, we should focus our CSP solving on the frontier.

Here's my final approach:

1. **Phase 1: Initialize** - Parse board, mark known mines
2. **Phase 2: Simple Rules** - Repeatedly apply trivial deductions until no more progress
3. **Phase 3: Subset Analysis** - Check for constraint subsets (if one constraint's cells are a subset of another's)
4. **Phase 4: Component CSP** - Decompose frontier into components, solve small ones via enumeration
5. **Phase 5: Output** - Return all proven-safe cells

Let me also handle the edge case where there's no input or malformed input gracefully.

Here's my final implementation:

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class MinesweeperSolver
{
    static int H, W;
    static char[,] board;
    static bool[,] safe, mine;
    
    static readonly int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };
    
    static void Main()
    {
        try
        {
            ReadInput();
            var result = Solve();
            OutputResult(result);
        }
        catch (Exception)
        {
            Console.WriteLine(0);
        }
    }
    
    static void ReadInput()
    {
        var firstLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(firstLine)) return;
        
        var parts = firstLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);
        
        board = new char[H, W];
        
        for (int r = 0; r < H; r++)
        {
            var line = Console.ReadLine() ?? "";
            for (int c = 0; c < W; c++)
                board[r, c] = c < line.Length ? line[c] : ' ';
        }
    }
    
    static List<(int x, int y)> Solve()
    {
        safe = new bool[H, W];
        mine = new bool[H, W];
        
        // Initialize known mines from '*'
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == '*')
                    mine[r, c] = true;
        
        // Phase 1: Simple rule propagation
        bool changed;
        do { changed = ApplySimpleRules(); } while (changed);
        
        // Phase 2: Advanced solving (subset analysis + CSP)
        AdvancedSolve();
        
        // Collect all proven-safe cells
        var result = new List<(int x, int y)>();
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (board[r, c] == ' ' && safe[r, c])
                    result.Add((c, r));
        
        return result;
    }
    
    static bool ApplySimpleRules()
    {
        bool changed = false;
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] >= '1' && board[r, c] <= '8')
                {
                    changed |= ProcessNumberCell(r, c);
                }
            }
        }
        
        return changed;
    }
    
    static bool ProcessNumberCell(int r, int c)
    {
        int target = board[r, c] - '0';
        var unknowns = new List<(int r, int c)>();
        int knownMines = 0;
        
        for (int d = 0; d < 8; d++)
        {
            int nr = r + dr[d], nc = c + dc[d];
            if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
            
            char cell = board[nr, nc];
            if (cell == ' ')
            {
                if (mine[nr, nc]) knownMines++;
                else if (!safe[nr, nc]) unknowns.Add((nr, nc));
            }
            else if (cell == '*')
            {
                knownMines++;
            }
        }
        
        bool changed = false;
        int remaining = target - knownMines;
        
        if (remaining < 0) return false;
        
        // All mines found -> remaining unknowns are safe
        if (remaining == 0)
        {
            foreach (var (ur, uc) in unknowns)
            {
                if (!safe[ur, uc])
                {
                    safe[ur, uc] = true;
                    changed = true;
                }
            }
        }
        // Unknowns == remaining mines -> all unknowns are mines
        else if (unknowns.Count == remaining && remaining > 0)
        {
            foreach (var (ur, uc) in unknowns)
            {
                if (!mine[ur, uc])
                {
                    mine[ur, nc] = true;
                    changed = true;
                }
            }
        }
        
        return changed;
    }
    
    static void AdvancedSolve()
    {
        // Build frontier
        var frontier = new List<(int r, int c)>();
        var frontierIdx = new Dictionary<(int, int), int>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] == ' ' && !safe[r, c] && !mine[r, c])
                {
                    bool isFrontier = false;
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dr[d], nc = c + dc[d];
                        if (nr >= 0 && nr < H && nc >= 0 && nc < W)
                        {
                            if (board[nr, nc] >= '1' && board[nr, nc] <= '8')
                            {
                                isFrontier = true;
                                break;
                            }
                        }
                    }
                    
                    if (isFrontier)
                    {
                        frontierIdx[(r, c)] = frontier.Count;
                        frontier.Add((r, c));
                    }
                }
            }
        }
        
        if (frontier.Count == 0 || frontier.Count > 80) return;
        
        // Build constraints
        var constraints = new List<(List<int> cells, int count)>();
        
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] >= '1' && board[r, c] <= '8')
                {
                    int target = board[r, c] - '0';
                    var cells = new List<int>();
                    int knownMines = 0;
                    
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dr[d], nc = c + dc[d];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        
                        if (board[nr, nc] == ' ')
                        {
                            if (mine[nr, nc]) knownMines++;
                            else if (frontierIdx.TryGetValue((nr, nc), out int idx))
                                cells.Add(idx);
                        }
                        else if (board[nr, nc] == '*')
                        {
                            knownMines++;
                        }
                    }
                    
                    int remaining = target - knownMines;
                    if (cells.Count > 0 && remaining >= 0)
                        constraints.Add((cells, remaining));
                }
            }
        }
        
        if (constraints.Count == 0) return;
        
        // Subset analysis
        ApplySubsetAnalysis(frontier, constraints);
        
        // Re-propagate simple rules
        while (ApplySimpleRules()) { }
        
        // Update frontier after subset analysis
        frontier = frontier.Where(p => !safe[p.r, p.c] && !mine[p.r, p.c]).ToList();
        frontierIdx.Clear();
        for (int i = 0; i < frontier.Count; i++)
            frontierIdx[frontier[i]] = i;
        
        if (frontier.Count == 0 || frontier.Count > 60) return;
        
        // Rebuild constraints
        constraints.Clear();
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                if (board[r, c] >= '1' && board[r, c] <= '8')
                {
                    int target = board[r, c] - '0';
                    var cells = new List<int>();
                    int knownMines = 0;
                    
                    for (int d = 0; d < 8; d++)
                    {
                        int nr = r + dr[d], nc = c + dc[d];
                        if (nr < 0 || nr >= H || nc < 0 || nc >= W) continue;
                        
                        if (board[nr, nc] == ' ')
                        {
                            if (mine[nr, nc]) knownMines++;
                            else if (frontierIdx.TryGetValue((nr, nc), out int idx))
                                cells.Add(idx);
                        }
                        else if (board[nr, nc] == '*') knownMines++;
                    }
                    
                    int remaining = target - knownMines;
                    if (cells.Count > 0 && remaining >= 0)
                        constraints.Add((cells, remaining));
                }
            }
        }
        
        if (constraints.Count == 0) return;
        
        // Find connected components
        var components = FindConnectedComponents(frontier.Count, constraints);
        
        foreach (var comp in components)
        {
            if (comp.Count <= 25)
                SolveComponentEnumeration(comp, frontier, constraints);
        }
    }
    
    static void ApplySubsetAnalysis(List<(int r, int c)> frontier, List<(List<int> cells, int count)> constraints)
    {
        bool changed = true;
        while (changed)
        {
            changed = false;
            
            for (int i = 0; i < constraints.Count; i++)
            {
                for (int j = 0; j < constraints.Count; j++)
                {
                    if (i == j) continue;
                    
                    var (cellsA, countA) = constraints[i];
                    var (cellsB, countB) = constraints[j];
                    
                    // Check if A is subset of B
                    var setA = new HashSet<int>(cellsA);
                    var setB = new HashSet<int>(cellsB);
                    
                    if (setA.IsProperSubsetOf(setB))
                    {
                        // cells in B-A have exactly (countB - countA) mines
                        int diffCount = countB - countA;
                        var diffCells = cellsB.Where(c => !setA.Contains(c)).ToList();
                        
                        if (diffCells.Count == diffCount && diffCount > 0)
                        {
                            // All diff cells are mines
                            foreach (int idx in diffCells)
                            {
                                var (r, c) = frontier[idx];
                                if (!mine[r, c])
                                {
                                    mine[r, c] = true;
                                    changed = true;
                                }
                            }
                        }
                        else if (diffCount == 0)
                        {
                            // All diff cells are safe
                            foreach (int idx in diffCells)
                            {
                                var (r, c) = frontier[idx];
                                if (!safe[r, c])
                                {
                                    safe[r, c] = true;
                                    changed = true;
                                }
                            }
                        }
                    }
                }
            }
            
            if (changed)
                while (ApplySimpleRules()) { }
        }
    }
    
    static List<List<int>> FindConnectedComponents(int n, List<(List<int> cells, int count)> constraints)
    {
        var adj = new List<HashSet<int>>();
        for (int i = 0; i < n; i++)
            adj.Add(new HashSet<int>());
        
        foreach (var (cells, _) in constraints)
        {
            for (int i = 0; i < cells.Count; i++)
            {
                for (int j = i + 1; j < cells.Count; j++)
                {
                    adj[cells[i]].Add(cells[j]);
                    adj[cells[j]].Add(cells[i]);
                }
            }
        }
        
        var visited = new bool[n];
        var components = new List<List<int>>();
        
        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
            {
                var comp = new List<int>();
                var stack = new Stack<int>();
                stack.Push(i);
                
                while (stack.Count > 0)
                {
                    int u = stack.Pop();
                    if (visited[u]) continue;
                    visited[u] = true;
                    comp.Add(u);
                    
                    foreach (int v in adj[u])
                        if (!visited[v])
                            stack.Push(v);
                }
                
                components.Add(comp);
            }
        }
        
        return components;
    }
    
    static void SolveComponentEnumeration(List<int> component, List<(int r, int c)> frontier, List<(List<int> cells, int count)> constraints)
    {
        // Filter constraints relevant to this component
        var compSet = new HashSet<int>(component);
        var relevantConstraints = constraints
            .Where(c => c.cells.Any(cell => compSet.Contains(cell)))
            .Select(c => (
                cells: c.cells.Where(cell => compSet.Contains(cell)).ToList(),
                count: c.count
            ))
            .ToList();
        
        // Map to local indices
        var localIdx = new Dictionary<int, int>();
        for (int i = 0; i < component.Count; i++)
            localIdx[component[i]] = i;
        
        var localConstraints = relevantConstraints.Select(c => (
            cells: c.cells.Select(cell => localIdx[cell]).ToList(),
            count: c.count
        )).ToList();
        
        int n = component.Count;
        var validConfigs = new List<bool[]>();
        
        EnumerateConfigs(0, new bool[n], localConstraints, validConfigs);
        
        if (validConfigs.Count == 0) return;
        
        // Find always-safe or always-mine cells
        for (int i = 0; i < n; i++)
        {
            bool allMine = validConfigs.All(cfg => cfg[i]);
            bool allSafe = validConfigs.All(cfg => !cfg[i]);
            
            var (r, c) = frontier[component[i]];
            if (allSafe) safe[r, c] = true;
            if (allMine) mine[r, c] = true;
        }
    }
    
    static void EnumerateConfigs(int idx, bool[] config, List<(List<int> cells, int count)> constraints, List<bool[]> validConfigs)
    {
        if (idx == config.Length)
        {
            if (ValidFullConfig(config, constraints))
                validConfigs.Add((bool[])config.Clone());
            return;
        }
        
        // Try safe
        config[idx] = false;
        if (ValidPartialConfig(config, idx, constraints))
            EnumerateConfigs(idx + 1, config, constraints, validConfigs);
        
        // Try mine
        config[idx] = true;
        if (ValidPartialConfig(config, idx, constraints))
            EnumerateConfigs(idx + 1, config, constraints, validConfigs);
    }
    
    static bool ValidPartialConfig(bool[] config, int idx, List<(List<int> cells, int count)> constraints)
    {
        foreach (var (cells, count) in constraints)
        {
            int mines = 0, unknowns = 0;
            foreach (int c in cells)
            {
                if (c > idx) unknowns++;
                else if (config[c]) mines++;
            }
            if (mines > count || mines + unknowns < count)
                return false;
        }
        return true;
    }
    
    static bool ValidFullConfig(bool[] config, List<(List<int> cells, int count)> constraints)
    {
        foreach (var (cells, count) in constraints)
        {
            int mines = cells.Sum(c => config[c] ? 1 : 0);
            if (mines != count) return false;
        }
```

```csharp
        if (ValidPartial(cfg, idx, cons)) EnumRec(idx + 1, cfg, cons, valid);

            if (m > n || m + u < n) return false;

        return true;
```
