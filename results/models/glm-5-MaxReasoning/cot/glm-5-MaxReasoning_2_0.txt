## Chinese Postman Problem Overview

The CPP asks for the shortest closed walk that traverses every edge at least once.

### Key Steps:
1. **Check if Eulerian**: If all vertices have even degree, the graph is Eulerian - just find an Euler circuit

static int[] depth;

static void BuildSpanningTree()
{
    treeParent = new int[N];
    treeParentEdge = new int[N];
    depth = new int[N];
    
    for (int i = 0; i < N; i++)
    {
        treeParent[i] = -1;
        treeParentEdge[i] = -1;
        depth[i] = -1;
    }
    
    var queue = new Queue<int>();
    queue.Enqueue(0);
    treeParent[0] = 0;  // root's parent is itself
    depth[0] = 0;
    
    while (queue.Count > 0)
    {
        int u = queue.Dequeue();
        foreach (var ei in adj[u])
        {
            int v = (eu[ei] == u) ? ev[ei] : eu[ei];
            if (depth[v] < 0)
            {
                depth[v] = depth[u] + 1;
                treeParent[v] = u;
                treeParentEdge[v] = ei;
                queue.Enqueue(v);
            }
        }
    }
    
    // Build LCA table
    LOG = (int)Math.Ceiling(Math.Log(N, 2)) + 1;
    lca = new int[N, LOG];
    for (int i = 0; i < N; i++)
        lca[i, 0] = treeParent[i];
    for (int j = 1; j < LOG; j++)
        for (int i = 0; i < N; i++)
            lca[i, j] = lca[lca[i, j-1], j-1];
}

static int GetLCA(int u, int v)
{
    if (depth[u] < depth[v]) (u, v) = (v, u);
    
    // Bring u to same depth as v
    int diff = depth[u] - depth[v];
    for (int j = 0; j < LOG; j++)
        if ((diff & (1 << j)) != 0)
            u = lca[u, j];
    
    if (u == v) return u;
    
    for (int j = LOG - 1; j >= 0; j--)
        if (lca[u, j] != lca[v, j])
        {
            u = lca[u, j];
            v = lca[v, j];
        }
    
    return treeParent[u];
}

static void AddTreePath(int u, int v)
{
    int lca_uv = GetLCA(u, v);
    
    // Add path from u to LCA
    while (u != lca_uv)
    {
        edgeCount[treeParentEdge[u]]++;
        u = treeParent[u];
    }
    
    // Add path from v to LCA
    while (v != lca_uv)
    {
        edgeCount[treeParentEdge[v]]++;
        v = treeParent[v];
    }
}

{

    var remaining = new int[M];
    Array.Copy(edgeCount, remaining, M);

    var adjPos = new int[N];
    var adjList = new List<int>[N];
    for (int i = 0; i < N; i++)
    {
        adjList[i] = new List<int>(adj[i]);
        adjPos[i] = 0;
    }
    
    var stack = new Stack<int>();
    stack.Push(0);
    
    while (stack.Count > 0)
    {
        int u = stack.Peek();
        
        // Find next unused edge
        while (adjPos[u] < adjList[u].Count)
        {
            int ei = adjList[u][adjPos[u]];
            if (remaining[ei] > 0)
            {
                remaining[ei]--;
                int v = (eu[ei] == u) ? ev[ei] : eu[ei];
                stack.Push(v);
                break;
            }
            adjPos[u]++;
        }
        
        if (adjPos[u] >= adjList[u].Count)
        {
            result.Add(stack.Pop());
        }
    }
    
    return result;
}
