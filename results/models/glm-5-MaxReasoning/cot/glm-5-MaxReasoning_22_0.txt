1. For each incoming piece, try all valid placements (all rotations, all columns)
2. For each placement, simulate dropping the piece and evaluate the resulting board
3. Choose the placement with the best score

Evaluation heuristics:
1. Lines cleared (positive, high weight)
2. Holes (negative - a hole is an empty cell with at least one filled cell above it)
3. Bumpiness (sum of absolute differences between adjacent column heights)
4. Aggregate height (sum of all column heights)
5. Wells (deep single-column gaps - can be bad or good depending on situation)

```csharp
    static (int, int) FindBestPlacement(bool[,] board, char pieceType, int width, int height, 
                                         Dictionary<char, List<List<(int, int)>>> pieces)
    {
        int bestRot = 0;
        int bestCol = 0;
        double bestScore = double.NegativeInfinity;
        
        var rotations = pieces[pieceType];
        
        for (int rot = 0; rot < 4; rot++)
        {
            var cells = rotations[rot];
            int pieceWidth = cells.Max(c => c.Item2) + 1;
            
            for (int col = 0; col <= width - pieceWidth; col++)
            {
                // Simulate drop
                var (valid, resultBoard, linesCleared) = SimulateDrop(board, pieceType, rot, col, width, height, pieces);
                
                if (!valid) continue;
                
                double score = EvaluateBoard(resultBoard, width, height, linesCleared);
                
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }
        
        return (bestRot, bestCol);
    }
    
    static (bool, bool[,], int) SimulateDrop(bool[,] board, char pieceType, int rotation, int col,
                                              int width, int height, Dictionary<char, List<List<(int, int)>>> pieces)
    {
        var cells = pieces[pieceType][rotation];
        
        // Find the lowest row where piece can be placed
        int dropRow = 0;
        for (int row = 0; row < height; row++)
        {
            bool canPlace = true;
            foreach (var (dr, dc) in cells)
            {
                int r = row + dr;
                int c = col + dc;
                if (r >= height || (r >= 0 && board[r, c]))
                {
                    canPlace = false;
                    break;
                }
            }
            
            if (!canPlace)
            {
                dropRow = row - 1;
                break;
            }
            dropRow = row;
        }
        
        // Check if placement is valid
        if (dropRow < 0)
        {
            return (false, null, 0);
        }
        
        // Create new board with piece placed
        var newBoard = (bool[,])board.Clone();
        foreach (var (dr, dc) in cells)
        {
            int r = dropRow + dr;
            int c = col + dc;
            newBoard[r, c] = true;
        }
        
        // Clear lines
        int linesCleared = 0;
        for (int r = height - 1; r >= 0; r--)
        {
            bool fullLine = true;
            for (int c = 0; c < width; c++)
            {
                if (!newBoard[r, c])
                {
                    fullLine = false;
                    break;
                }
            }
            
            if (fullLine)
            {
                linesCleared++;
                // Shift rows down
                for (int rr = r; rr > 0; rr--)
                {
                    for (int c = 0; c < width; c++)
                    {
                        newBoard[rr, c] = newBoard[rr - 1, c];
                    }
                }
                for (int c = 0; c < width; c++)
                {
                    newBoard[0, c] = false;
                }
                r++; // Check this row again
            }
        }
        
        return (true, newBoard, linesCleared);
    }
    
    static double EvaluateBoard(bool[,] board, int width, int height, int linesCleared)
    {
        double score = 0;
        
        // Lines cleared (very positive)
        score += linesCleared * 1000;
        
        // Calculate heights
        int[] heights = new int[width];
        for (int c = 0; c < width; c++)
        {
            for (int r = 0; r < height; r++)
            {
                if (board[r, c])
                {
                    heights[c] = height - r;
                    break;
                }
            }
        }
        
        // Aggregate height (negative)
        int aggregateHeight = heights.Sum();
        score -= aggregateHeight * 0.5;
        
        // Holes (negative)
        int holes = 0;
        for (int c = 0; c < width; c++)
        {
            bool foundBlock = false;
            for (int r = 0; r < height; r++)
            {
                if (board[r, c])
                {
                    foundBlock = true;
                }
                else if (foundBlock)
                {
                    holes++;
                }
            }
        }
        score -= holes * 10;
        
        // Bumpiness (negative)
        int bumpiness = 0;
        for (int c = 0; c < width - 1; c++)
        {
            bumpiness += Math.Abs(heights[c] - heights[c + 1]);
        }
        score -= bumpiness * 0.5;
        
        return score;
    }
    
    static void UpdateBoard(bool[,] board, char pieceType, int rotation, int col, string result,
                            int width, int height, Dictionary<char, List<List<(int, int)>>> pieces)
    {
        if (result.StartsWith("gameover"))
        {
            return; // Game is over, don't update
        }
        
        var cells = pieces[pieceType][rotation];
        
        // Find the lowest row where piece was placed
        int dropRow = 0;
        for (int row = 0; row < height; row++)
        {
            bool canPlace = true;
            foreach (var (dr, dc) in cells)
            {
                int r = row + dr;
                int c = col + dc;
                if (r >= height || (r >= 0 && board[r, c]))
                {
                    canPlace = false;
                    break;
                }
            }
            
            if (!canPlace)
            {
                dropRow = row - 1;
                break;
            }
            dropRow = row;
        }
        
        // Place piece
        foreach (var (dr, dc) in cells)
        {
            int r = dropRow + dr;
            int c = col + dc;
            if (r >= 0 && r < height && c >= 0 && c < width)
            {
                board[r, c] = true;
            }
        }
        
        // Clear lines if indicated
        if (result.StartsWith("cleared_"))
        {
            int lines = int.Parse(result.Substring(8));
            ClearLines(board, width, height, lines);
        }
        else if (result == "ok")
        {
            // Check and clear any complete lines
            ClearAllLines(board, width, height);
        }
    }
    
    static void ClearLines(bool[,] board, int width, int height, int numLines)
    {
        // Clear from bottom, up to numLines
        int cleared = 0;
        for (int r = height - 1; r >= 0 && cleared < numLines; r--)
        {
            bool fullLine = true;
            for (int c = 0; c < width; c++)
            {
                if (!board[r, c])
                {
                    fullLine = false;
                    break;
                }
            }
            
            if (fullLine)
            {
                // Shift rows down
                for (int rr = r; rr > 0; rr--)
                {
                    for (int c = 0; c < width; c++)
                    {
                        board[rr, c] = board[rr - 1, c];
                    }
                }
                for (int c = 0; c < width; c++)
                {
                    board[0, c] = false;
                }
                cleared++;
                r++; // Check this row again
            }
        }
    }
    
    static void ClearAllLines(bool[,] board, int width, int height)
    {
        for (int r = height - 1; r >= 0; r--)
```

```csharp
        score -= holes * 30;
    }
    
    // Bumpiness (more important in wide/long games)
    if (isWide || isLongGame)
    {
        score -= bumpiness * 10;
    }
    else
    {
        score -= bumpiness * 3;
    }
    
    // Wells (can be strategic for I pieces)
    score -= wells * 5;
    
    // Aggregate height
    score -= aggregateHeight * 0.5;
    
    // Perfect clear
    if (aggregateHeight == 0)
    {
        score += 10000;
    }
    
    return score;
}
```

Actually, I think I'm overcomplicating this. Let me write a simpler, more robust solution first, and then optimize.

Also, I realized that I should handle the case where no valid placement exists. In that case, I should still output something (maybe 0 0).

Let me also consider the time constraint (5 seconds per piece). For a standard 10-column board, there are at most 10 columns * 4 rotations = 40 placements to evaluate. Each evaluation involves simulating a drop and calculating metrics, which should be very fast (< 1ms). So we have plenty of time.

But for a 20-column board, there are up to 80 placements, still very fast.

    static (int, int) FindBestPlacement(bool[,] board, char pieceType)
    {
        int bestRot = 0, bestCol = 0;
        double bestScore = double.NegativeInfinity;
        bool foundValid = false;
        
        var rotations = pieces[pieceType];
        
        for (int rot = 0; rot < 4; rot++)
        {
            var cells = rotations[rot];
            int pieceWidth = cells.Max(c => c.dc) - cells.Min(c => c.dc) + 1;
            
            for (int col = 0; col <= width - pieceWidth; col++)
            {
                var result = SimulateDrop(board, cells, col);
                
                if (!result.valid) continue;
                
                foundValid = true;
                double score = EvaluateBoard(result.board, result.linesCleared);
                
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }
        
        return (bestRot, bestCol);
    }
    
    static (bool valid, bool[,] board, int linesCleared) SimulateDrop(bool[,] board, List<(int dr, int dc)> cells, int col)
    {
        int dropRow = -4;
        
        for (int row = -4; row < height + 4; row++)
        {
            bool valid = true;
            foreach (var (dr, dc) in cells)
            {
                int r = row + dr;
                int c = col + dc;
                
                if (r >= height || (r >= 0 && board[r, c]))
                {
                    valid = false;
                    break;
                }
            }
            
            if (!valid)
            {
                dropRow = row - 1;
                break;
            }
            dropRow = row;
        }
        
        // Check if any cell is on the board
        bool anyOnBoard = cells.Any(c => dropRow + c.dr >= 0);
        if (!anyOnBoard)
        {
            return (false, null, 0);
        }
        
        var newBoard = (bool[,])board.Clone();
        foreach (var (dr, dc) in cells)
        {
            int r = dropRow + dr;
            int c = col + dc;
            if (r >= 0 && r < height)
            {
                newBoard[r, c] = true;
            }
        }
        
        int linesCleared = ClearLines(newBoard);
        
        return (true, newBoard, linesCleared);
    }
    
    static int ClearLines(bool[,] board)
    {
        int cleared = 0;
        
        for (int r = height - 1; r >= 0; r--)
        {
            bool full = true;
            for (int c = 0; c < width; c++)
            {
                if (!board[r, c])
                {
                    full = false;
                    break;
                }
            }
            
            if (full)
            {
                cleared++;
                for (int rr = r; rr > 0; rr--)
                {
                    for (int c = 0; c < width; c++)
                    {
                        board[rr, c] = board[rr - 1, c];
                    }
                }
                for (int c = 0; c < width; c++)
                {
                    board[0, c] = false;
                }
                r++;
            }
        }
        
        return cleared;
    }
    
    static double EvaluateBoard(bool[,] board, int linesCleared)
    {
        double score = 0;
        
        // Lines cleared
        score += linesCleared * linesCleared * 500;
        
        // Calculate heights
        int[] heights = new int[width];
        int totalHeight = 0;
        
        for (int c = 0; c < width; c++)
        {
            for (int r = 0; r < height; r++)
            {
                if (board[r, c])
                {
                    heights[c] = height - r;
                    totalHeight += heights[c];
                    break;
                }
            }
        }
        
        // Holes
        int holes = 0;
        for (int c = 0; c < width; c++)
        {
            bool hasBlock = heights[c] > 0;
            if (!hasBlock) continue;
            
            for (int r = 0; r < height; r++)
            {
                if (!board[r, c] && height - r < heights[c])
                {
                    holes++;
                }
            }
        }
