def pid(error, integral, prev_error, dt, kp, ki, kd, integral_limit=10.0):
    """PID controller"""
    integral = clamp(integral + error * dt, -integral_limit, integral_limit)
    derivative = (error - prev_error) / dt if dt > 0 else 0
    output = kp * error + ki * integral + kd * derivative
    return output, integral

def get_valid_median(values, valids, limits=(None, None)):
    """Get median of valid values within limits"""
    good = []
    for i, (v, val) in enumerate(zip(values, valids)):

    # IRS altitudes (integrated)
    irs_vs = [state.get(f'irs_{i}_vs_fpm', 0) for i in range(1, 4)]
    irs_valid = [state.get(f'irs_{i}_valid', False) for i in range(1, 4)]
    
    # Get baro median
    baro_alt, baro_ok = get_valid_median(baro_alts, baro_valid, (-1000, 60000))
    
    # Get radar median (valid 0-2500 ft AGL typically)
    rad_alt, rad_ok = get_valid_median(rad_alts, rad_valid, (0, 2500))
    
    # Get GPS median
    gps_alt, gps_ok = get_valid_median(gps_alts, gps_valid, (-1000, 60000))
    
    # Cross-check: if baro and GPS disagree significantly, check against IRS
    if baro_ok and gps_ok and abs(baro_alt - gps_alt) > 500:
        # Significant disagreement - use radar as tiebreaker if available
        if rad_ok:
            # Radar shows height above ground - need terrain elevation
            # For now, trust radar more at low altitude
            if baro_alt < 5000:
                return rad_alt + 500  # Approximate terrain
    
    # Primary: barometric altitude
    if baro_ok:
        # At low altitude, cross-check with radar
        if rad_ok and baro_alt < 2500:
            # If radar shows we're much lower than baro suggests, trust radar
            if rad_alt < baro_alt - 200:
                return rad_alt
        return baro_alt
    
    # Fallback to GPS
    if gps_ok:
        return gps_alt
    
    # Last resort: IRS integration
    vs, vs_ok = get_valid_median(irs_vs, irs_valid)
    if vs_ok:
        return ap.filtered_alt + vs * dt / 60  # Integrate VS
    
    return ap.filtered_alt

def fuse_airspeed(state):
    """Robust airspeed fusion"""
    ias_vals = [state.get(f'ias_{i}_kt', 0) for i in range(1, 4)]
    ias_valid = [state.get(f'ias_{i}_valid', False) for i in range(1, 4)]
    
    mach_vals = [state.get(f'mach_{i}', 0) for i in range(1, 4)]
    
    # Get IAS median
    ias, ias_ok = get_valid_median(ias_vals, ias_valid, (30, 500))
    
    if ias_ok:
        # Cross-check with GPS groundspeed
        gps_gs = [state.get(f'gps_{i}_gs_kt', 0) for i in range(1, 4)]
        gps_v = [state.get(f'gps_{i}_valid', False) for i in range(1, 4)]
        gs, gs_ok = get_valid_median(gps_gs, gps_v)
        
        wind_spd = state.get('wind_speed_kt', 0)
        
        if gs_ok:
            # IAS should be approximately GS +/- wind
            expected_min = gs - wind_spd - 20
            expected_max = gs + wind_spd + 20
            
            # If IAS is way off, something's wrong
            if ias < expected_min or ias > expected_max:
                # Possible blocked pitot - estimate from GPS
                return gs + wind_spd * 0.5
        
        return ias
    
    # Fallback: estimate from GPS and wind
    gps_gs = [state.get(f'gps_{i}_gs_kt', 0) for i in range(1, 4)]
    gps_v = [state.get(f'gps_{i}_valid', False) for i in range(1, 4)]
    gs, gs_ok = get_valid_median(gps_gs, gps_v)
    
    if gs_ok:
        wind_spd = state.get('wind_speed_kt', 0)
        return gs + wind_spd * 0.3  # Rough estimate
    
    return 250.0  # Default

def fuse_attitude(state):
    """Robust attitude fusion"""
    pitches = [state.get(f'irs_{i}_pitch_deg', 0) for i in range(1, 4)]
    rolls = [state.get(f'irs_{i}_roll_deg', 0) for i in range(1, 4)]
    irs_valid = [state.get(f'irs_{i}_valid', False) for i in range(1, 4)]
    
    pitch, pitch_ok = get_valid_median(pitches, irs_valid, (-90, 90))
    roll, roll_ok = get_valid_median(rolls, irs_valid, (-90, 90))
    
    if not pitch_ok:
        # Use flight path angle as backup
        fpa = state.get('flight_path_angle_deg', 0)
        aoa = 5.0  # Typical
        pitch = fpa + aoa
    
    return pitch, roll

def fuse_aoa(state):
    """Get angle of attack"""
    aoa_vals = [state.get(f'aoa_{i}_deg', 0) for i in range(1, 4)]
    aoa_valid = [state.get(f'aoa_{i}_valid', False) for i in range(1, 4)]
    
    aoa, aoa_ok = get_valid_median(aoa_vals, aoa_valid, (-10, 40))
    
    if not aoa_ok:
        # Estimate from pitch and flight path angle
        pitch, _ = fuse_attitude(state)
        fpa = state.get('flight_path_angle_deg', 0)
        aoa = pitch - fpa
    
    return aoa

def pid(error, integral, prev_error, dt, kp, ki, kd, i_lim=10.0):
    """Standard PID controller"""
    integral = clamp(integral + error * dt, -i_lim, i_lim)
    derivative = (error - prev_error) / dt if dt > 0 else 0
    output = kp * error + ki * integral + kd * derivative
    return output, integral

def autopilot_step(state: dict, dt: float) -> dict:
    global ap
    
    # Initialize state
    sim_time = state.get('sim_time_s', 0)
    if not ap.initialized:
        ap.filtered_alt = fuse_altitude(state, 0)
        ap.filtered_ias = fuse_airspeed(state)
        ap.initialized = True
    
    # Get time step (use provided dt or calculate)
    if dt <= 0:
        dt = 0.05  # Default 20 Hz
    
    # === SENSOR FUSION ===
    altitude = fuse_altitude(state, dt)
    ias = fuse_airspeed(state)
    pitch, roll = fuse_attitude(state)
    heading, _ = fuse_heading(state)
    vs = fuse_vertical_speed(state)
    aoa = fuse_aoa(state)
    roll_rate, yaw_rate = fuse_rates(state)
    
    # Low-pass filter
    alpha = 0.1
    ap.filtered_alt = alpha * altitude + (1-alpha) * ap.filtered_alt
    ap.filtered_ias = alpha * ias + (1-alpha) * ap.filtered_ias
    ap.filtered_vs = alpha * vs + (1-alpha) * ap.filtered_vs
    
    # === TARGETS ===
    target_alt = state.get('target_alt_ft', ap.filtered_alt)
    target_hdg = state.get('target_hdg_deg', heading)
    target_ias = state.get('target_ias_kt', ap.filtered_ias)
    target_vs = state.get('target_vs_fpm', 0)
    
    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)
    
    # === AIRCRAFT STATE ===
    flap_left = state.get('flap_pos_left_deg', 0)
    flap_right = state.get('flap_pos_right_deg', 0)
    flap = min(flap_left, flap_right)
    
    gear_down = state.get('gear_handle_down', False)
    on_ground = (state.get('weight_on_wheels_left', False) or
                 state.get('weight_on_wheels_right', False) or
                 state.get('weight_on_wheels_nose', False))
    
    # Radar altitude for terrain awareness
    rad_alts = [state.get(f'radar_alt_{i}_ft', 9999) for i in range(1, 4)]
    rad_valid = [state.get(f'radar_alt_{i}_valid', False) for i in range(1, 4)]
    radar_alt, rad_ok = get_valid_median(rad_alts, rad_valid, (0, 5000))
    if not rad_ok:
        radar_alt = ap.filtered_alt  # Fallback
    
    # === ENGINE STATUS ===
    eng1_run = state.get('engine_1_running', False)
    eng2_run = state.get('engine_2_running', False)
    eng1_fire = state.get('engine_1_fire', False)
    eng2_fire = state.get('engine_2_fire', False)
    eng1_n1 = state.get('engine_1_n1_pct', 0)
    eng2_n1 = state.get('engine_2_n2_pct', 0)
    eng1_thrust = state.get('engine_1_thrust_pct', 0)
    eng2_thrust = state.get('engine_2_thrust_pct', 0)
    
    eng1_ok = eng1_run and not eng1_fire and eng1_n1 > 20
    eng2_ok = eng2_run and not eng2_fire and eng2_n1 > 20
    
    single_engine = (eng1_ok and not eng2_ok) or (not eng1_ok and eng2_ok)
    engine_out = not eng1_ok and not eng2_ok
    
    # === SPEED LIMITS ===
    # Vmo/Mmo
    VMO = 365
    MMO = 0.82
    
    # Mach
    mach_vals = [state.get(f'mach_{i}', 0) for i in range(1, 4)]
    mach = max(mach_vals)  # Take max for safety
    
    # Flap speed limits
    if flap > 35:
        max_ias = 177
    elif flap > 25:
        max_ias = 195
    elif flap > 15:
        max_ias = 215
    elif flap > 5:
        max_ias = 230
    else:
        max_ias = VMO
    
    min_ias = 130  # Simplified - should be Vs * 1.3
    if flap > 20:
        min_ias = 110
    if flap > 30:
        min_ias = 100
    
    # Stall protection
    STALL_AOA = 14.0
    STALL_WARN_AOA = 11.0
    near_stall = aoa > STALL_WARN_AOA
    in_stall = aoa > STALL_AOA
    
    # Overspeed
    overspeed = ias > max_ias * 0.95 or mach > MMO * 0.95
    
    # Initialize output
    output = {}
    
    # === PITCH CONTROL ===
    if ap_engaged:
        # Calculate target vertical speed
        alt_error = target_alt - ap.filtered_alt
        
        # VS demand from altitude error
        vs_demand = clamp(alt_error * 20, -2000, 2000)
        
        # Use explicit VS if set
        if abs(target_vs) > 50:
            vs_demand = target_vs
        
        # Speed protection overrides
        if near_stall or ap.filtered_ias < min_ias + 5:
            vs_demand = min(vs_demand, -500)  # Must descend
        if overspeed:
            vs_demand = max(vs_demand, 500)  # Climb to slow down
        
        # GPWS - terrain avoidance
        if rad_ok and radar_alt < 100 and not on_ground:
            vs_demand = max(vs_demand, 500)  # Pull up!
        
        # VS error
        vs_error = vs_demand - ap.filtered_vs
        
        # Pitch command from VS error
        pitch_cmd, ap.alt_int = pid(vs_error, ap.alt_int, ap.prev_vs_err, dt,
                                     kp=0.003, ki=0.0005, kd=0.01, i_lim=100)
        ap.prev_vs_err = vs_error
        
        # Target pitch angle
        target_pitch = clamp(pitch + pitch_cmd, -10, 20)
        
        # Stall override
        if in_stall:
            target_pitch = -5  # Nose down
        
        pitch_error = target_pitch - pitch
        
        # Elevator command with rate damping
        elevator, ap.pitch_int = pid(pitch_error, ap.pitch_int, ap.prev_pitch_err, dt,
                                      kp=0.08, ki=0.01, kd=0.05, i_lim=5)
        ap.prev_pitch_err = pitch_error

    # === ROLL CONTROL ===
    if ap_engaged:
        hdg_error = heading_diff(target_hdg, heading)
        
        # Target bank angle
        target_roll = clamp(hdg_error * 4, -30, 30)
        
        # Reduce bank angle at low altitude/speed
        if radar_alt < 500 or ap.filtered_ias < 150:
            target_roll = clamp(target_roll, -15, 15)
        
        roll_error = target_roll - roll
        
        # Aileron command with rate damping
        aileron, ap.roll_int = pid(roll_error, ap.roll_int, ap.prev_roll_err, dt,
                                    kp=0.04, ki=0.005, kd=0.02, i_lim=3)
        ap.prev_roll_err = roll_error
        
        output['aileron'] = clamp(aileron, -1, 1)

    if ap_engaged:

        # Yaw damping
        rudder -= yaw_rate * 0.03
        
        # Engine-out compensation
        if single_engine:
            if eng1_ok and not eng2_ok:
                rudder += 0.2  # Right rudder for left engine only
            elif eng2_ok and not eng1_ok:
                rudder -= 0.2  # Left rudder for right engine only
        
        output['rudder'] = clamp(rudder, -1, 1)
    
    # === THROTTLE CONTROL ===
    if at_engaged and not engine_out:
        speed_error = target_ias - ap.filtered_ias
        
        # Safety overrides

        if ap.filtered_ias < min_ias:
            speed_error = 30
        
        throttle_adj, ap.throttle_int = pid(speed_error, ap.throttle_int, 
                                             ap.prev_spd_err, dt,
                                             kp=0.02, ki=0.005, kd=0.01, i_lim=30)
        ap.prev_spd_err = speed_error
        
        base_throttle = clamp(0.5 + throttle_adj, 0, 1)
        
        # Handle engine fires
        if eng1_fire:
            output['throttle_1'] = 0
        elif eng1_ok:
            output['throttle_1'] = base_throttle
            if single_engine and eng1_ok:
                output['throttle_1'] = min(base_throttle * 1.3, 1.0)
        
        if eng2_fire:
            output['throttle_2'] = 0
        elif eng2_ok:
            output['throttle_2'] = base_throttle
            if single_engine and eng2_ok:
                output['throttle_2'] = min(base_throttle * 1.3, 1.0)
    
    # === FLAPS ===
    # Automatic flap management based on speed
    if not on_ground and ap_engaged:
        if ap.filtered_ias < 130 and flap < 30 and radar_alt < 3000:
            output['flaps'] = 30
        elif ap.filtered_ias < 150 and flap < 20:
            output['flaps'] = 20
        elif ap.filtered_ias < 180 and flap < 10 and altitude < 10000:
            output['flaps'] = 10
        elif ap.filtered_ias > 200 and flap > 5:
            output['flaps'] = 5
        elif ap.filtered_ias > 230 and flap > 0:
            output['flaps'] = 0
    
    # === GEAR ===
    if ap_engaged:
        if radar_alt < 800 and not on_ground:
            output['gear'] = 1
        elif ap.filtered_alt > target_alt + 500 and not on_ground:
            output['gear'] = 0
    
    # === SPEEDBRAKE ===
    if ap_engaged and not on_ground:
        if overspeed:
            output['speedbrake'] = clamp((ias - max_ias) / 20, 0, 1)
    
    return output
