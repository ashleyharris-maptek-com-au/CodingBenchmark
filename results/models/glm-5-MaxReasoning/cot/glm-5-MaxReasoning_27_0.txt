        for (int u : adj[v]) {
            if (color[u] == -1) {
                // Check if this neighbor just got a new color in its neighborhood
                // by checking if color[v] is new for u
                bool new_color = true;
                for (int w : adj[u]) {
                    if (w != v && color[w] == color[v]) {
                        new_color = false;
                        break;
                    }
                }
                
                if (new_color) {
                    // Remove from old bucket
                    int old_sat = saturation[u];
                    auto& old_bucket = buckets[old_sat];
                    old_bucket.erase(remove(old_bucket.begin(), old_bucket.end(), u), old_bucket.end());
                    
                    // Add to new bucket
                    saturation[u]++;
                    if (saturation[u] <= max_degree + 1) {
                        buckets[saturation[u]].push_back(u);
                        max_sat = max(max_sat, saturation[u]);
                    }
                }
            }
        }
    }
    
    return colored_count == n;
}
