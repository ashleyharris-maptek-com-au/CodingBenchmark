## Problem Analysis

For the Longest Common Substring problem:
- **Trivial case**: Small strings - can use dynamic programming O(n1 * n2 * ... * nk) for k strings, but this becomes intractable quickly
- **Multiple strings**: This is NP-hard in general for k strings with total length n, with complexity O(n^k)
- **Large scale**: Need efficient algorithms

## Algorithm Options

1. **Suffix Automaton/Tree approach**: Build a suffix automaton for one string, then match others against it. For 2 strings: O(n+m). For k strings: O(k*n) where n is the length of the first string.

2. **Binary search with rolling hash**: Binary search on length, check if common substring of that length exists using rolling hash. O(k*n*log(min_len)) average case.

3. **Generalized Suffix Tree**: Build a generalized suffix tree for all strings, find deepest node with all string markers. O(total_length) to build, but high constant factor.

4. **Suffix Array + LCP**: Build generalized suffix array, find longest LCP that spans all strings. O(n log n) or O(n) with SA-IS.

class SuffixAutomaton {
public:
    vector<int> len, link;
    vector<unordered_map<char, int>> next;
    int last;
    
    SuffixAutomaton() {
        len.push_back(0);
        link.push_back(-1);
        next.emplace_back();
        last = 0;
    }
    
    void extend(char c) {
        int cur = len.size();
        len.push_back(len[last] + 1);
        link.push_back(0);
        next.emplace_back();
        
        int p = last;
        while (p != -1 && next[p].find(c) == next[p].end()) {
            next[p][c] = cur;
            p = link[p];
        }
        
        if (p == -1) {
            link[cur] = 0;
        } else {
            int q = next[p][c];
            if (len[p] + 1 == len[q]) {
                link[cur] = q;
            } else {
                int clone = len.size();
                len.push_back(len[p] + 1);
                link.push_back(link[q]);
                next.push_back(next[q]);

                    next[p][c] = clone;
                    p = link[p];
                }
                link[q] = link[cur] = clone;
            }
        }
        last = cur;
    }

        for (char c : s) extend(c);
    }

                l = len[v];

ull mod_mul(ull a, ull b, ull m) {
    // Modular multiplication avoiding overflow
    ull ret = a * b - m * (ull)(1.0L / m * a * b);
    return ret + m * (ret >> 63) - m;
}

ull mod_add(ull a, ull b, ull m) {
    a += b;
    if (a >= m) a -= m;
    return a;

// Rolling hash functions
vector<ull> compute_hashes(const string& s, int len_substr) {
    int n = s.size();
    if (len_substr > n) return {};
    
    vector<ull> hashes;
    hashes.reserve(n - len_substr + 1);
    
    ull h = 0;
    ull pow_base = 1;
    
    // Compute base^len_substr mod MOD
    for (int i = 0; i < len_substr; i++) {
        pow_base = mod_mul(pow_base, BASE, MOD);
    }
    
    // Initial hash
    for (int i = 0; i < len_substr; i++) {
        h = mod_add(mod_mul(h, BASE, MOD), s[i], MOD);
    }
    hashes.push_back(h);
    
    // Rolling
    for (int i = len_substr; i < n; i++) {
        h = mod_add(mod_mul(h, BASE, MOD), s[i], MOD);
        h = mod_add(h, MOD - mod_mul(s[i - len_substr], pow_base, MOD), MOD);
        hashes.push_back(h);
    }
    
    return hashes;
}

string extract_common_substring(const vector<string>& strings, int len_substr, int skip_idx) {
    const string& base = strings[skip_idx];
    int n = base.size();
    
    if (len_substr > n) return "";
    
    // Compute hashes for base string
    vector<ull> base_hashes = compute_hashes(base, len_substr);
    unordered_set<ull> common(base_hashes.begin(), base_hashes.end());
    
    // Intersect with each other string's hashes
    for (int i = 0; i < (int)strings.size(); i++) {
        if (i == skip_idx) continue;
        
        vector<ull> hashes = compute_hashes(strings[i], len_substr);
        unordered_set<ull> current(hashes.begin(), hashes.end());
        
        unordered_set<ull> new_common;
        for (ull h : common) {
            if (current.count(h)) {
                new_common.insert(h);
            }
        }
        common = move(new_common);
        
        if (common.empty()) return "";
    }
    
    // Find and return the actual substring
    ull target_hash = *common.begin();
    for (int i = 0; i + len_substr <= n; i++) {
        if (base_hashes[i] == target_hash) {
            return base.substr(i, len_substr);
        }
    }
    
    return "";
}

        }

    }

vector<HashPair> compute_rolling_hashes(const string& s, int len_substr, 
                                         const vector<ull>& pow1, const vector<ull>& pow2) {
    int n = (int)s.size();
    if (len_substr > n) return {};
    
    vector<HashPair> hashes;
    hashes.reserve(n - len_substr + 1);
    
    ull h1 = 0, h2 = 0;
    for (int i = 0; i < len_substr; i++) {
        h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;
        h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;
    }
    hashes.push_back({h1, h2});
    
    for (int i = len_substr; i < n; i++) {
        h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;
        h1 = (h1 + MOD1 - (pow1[len_substr] * (unsigned char)s[i - len_substr]) % MOD1) % MOD1;
        
        h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;
        h2 = (h2 + MOD2 - (pow2[len_substr] * (unsigned char)s[i - len_substr]) % MOD2) % MOD2;
        
        hashes.push_back({h1, h2});
    }
    
    return hashes;
}

bool has_common_substring(const vector<string>& strings, int len_substr,
                          const vector<ull>& pow1, const vector<ull>& pow2) {
    int n = (int)strings.size();
    if (len_substr <= 0) return true;
    
    // Check if any string is shorter than len_substr
    for (const auto& s : strings) {
        if ((int)s.size() < len_substr) return false;
    }
    
    // Compute hashes for first string
    vector<HashPair> hashes = compute_rolling_hashes(strings[0], len_substr, pow1, pow2);
    unordered_set<ull> common;
    for (const auto& hp : hashes) {
        common.insert(hp.first * MOD2 + hp.second);
    }
    
    // Intersect with each other string
    for (int i = 1; i < n && !common.empty(); i++) {
        vector<HashPair> current_hashes = compute_rolling_hashes(strings[i], len_substr, pow1, pow2);
        unordered_set<ull> current;
        for (const auto& hp : current_hashes) {
            current.insert(hp.first * MOD2 + hp.second);
        }
        
        unordered_set<ull> intersection;
        for (ull h : common) {
            if (current.count(h)) {
                intersection.insert(h);
            }
        }
        common = move(intersection);
    }
    
    return !common.empty();
}

string extract_substring(const vector<string>& strings, int len_substr,
                         const vector<ull>& pow1, const vector<ull>& pow2) {
    int n = (int)strings.size();
    
    // Compute hashes for first string
    vector<HashPair> hashes = compute_rolling_hashes(strings[0], len_substr, pow1, pow2);
    unordered_set<ull> common;
    for (const auto& hp : hashes) {
        common.insert(hp.first * MOD2 + hp.second);
    }
    
    // Intersect with each other string
    for (int i = 1; i < n && !common.empty(); i++) {
        vector<HashPair> current_hashes = compute_rolling_hashes(strings[i], len_substr, pow1, pow2);
        unordered_set<ull> current;
        for (const auto& hp : current_hashes) {
            current.insert(hp.first * MOD2 + hp.second);
        }
        
        unordered_set<ull> intersection;
        for (ull h : common) {
            if (current.count(h)) {
                intersection.insert(h);
            }
        }
        common = move(intersection);
    }
    
    if (common.empty()) return "";
    
    // Find the actual substring
    ull target = *common.begin();
    ull target_h1 = target / MOD2;
    ull target_h2 = target % MOD2;
    
    for (int i = 0; i + len_substr <= (int)strings[0].size(); i++) {
        if (hashes[i].first == target_h1 && hashes[i].second == target_h2) {
            string candidate = strings[0].substr(i, len_substr);
            // Verify
            bool valid = true;
            for (int j = 1; j < n; j++) {
                if (strings[j].find(candidate) == string::npos) {
                    valid = false;
                    break;
                }
            }
            if (valid) return candidate;
        }
    }
    
    return "";
}

pair<int, string> solve_with_binary_search(const vector<string>& strings,
                                            const vector<ull>& pow1, const vector<ull>& pow2) {
    int n = (int)strings.size();
    
    // Find min length
    int min_len = INT_MAX;
    for (const auto& s : strings) {
        min_len = min(min_len, (int)s.size());
    }
    
    // Binary search on length
    int lo = 0, hi = min_len;
    while (lo < hi) {
        int mid = (lo + hi + 1) / 2;
        if (has_common_substring(strings, mid, pow1, pow2)) {
            lo = mid;
        } else {
            hi = mid - 1;
        }
    }
    
    if (lo == 0) return {0, ""};
    
    string result = extract_substring(strings, lo, pow1, pow2);
    return {lo, result};
}

// ================== Suffix Automaton ==================

class SuffixAutomaton {
    vector<int> len, link;
    vector<unordered_map<unsigned char, int>> next;
    int last;
    
public:
    SuffixAutomaton() : last(0) {
        len.push_back(0);
        link.push_back(-1);
        next.emplace_back();
    }
    
    void extend(unsigned char c) {
        int cur = (int)len.size();
        len.push_back(len[last] + 1);
        link.push_back(0);
        next.emplace_back();
        
        int p = last;
        while (p != -1 && next[p].find(c) == next[p].end()) {
            next[p][c] = cur;
            p = link[p];
        }
        
        if (p == -1) {
            link[cur] = 0;
        } else {
            int q = next[p][c];
            if (len[p] + 1 == len[q]) {
                link[cur] = q;
            } else {
                int clone = (int)len.size();
                len.push_back(len[p] + 1);
                link.push_back(link[q]);
                next.push_back(next[q]);
                
                while (p != -1 && next[p][c] == q) {
                    next[p][c] = clone;
                    p = link[p];
                }
                link[q] = link[cur] = clone;
            }
        }
        last = cur;
    }
    
    void build(const string& s) {
        for (unsigned char c : s) extend(c);
    }
    
    vector<int> match_string(const string& s) {
        int n = (int)len.size();
        vector<int> match(n, 0);
        
        int v = 0, l = 0;
        for (unsigned char c : s) {
            auto it = next[v].find(c);
            while (v != 0 && it == next[v].end()) {
                v = link[v];
                l = len[v];
                it = next[v].find(c);
            }
            if (it != next[v].end()) {
                v = it->second;
                l++;
            }
            match[v] = max(match[v], l);
        }
        
        // Propagate via suffix links
        vector<int> order(n);
        for (int i = 0; i < n; i++) order[i] = i;
        sort(order.begin(), order.end(), [this](int a, int b) { return len[a] > len[b]; });
        
        for (int v : order) {
            if (link[v] > 0) {
                match[link[v]] = max(match[link[v]], min(match[v], len[link[v]]));
            }
        }
        
        return match;
    }
    
    int size() const { return (int)len.size(); }
};

class RollingHasher {
    const vector<ull>& pow1;
    const vector<ull>& pow2;
    
public:
    RollingHasher(const vector<ull>& p1, const vector<ull>& p2) : pow1(p1), pow2(p2) {}
    
    vector<HashPair> compute_hashes(const string& s, int len_substr) const {
        int n = (int)s.size();
        if (len_substr > n || len_substr <= 0) return {};
        
        vector<HashPair> hashes;
        hashes.reserve(n - len_substr + 1);
        
        ull h1 = 0, h2 = 0;
        for (int i = 0; i < len_substr; i++) {
            h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;
            h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;
        }
        hashes.push_back({h1, h2});
        
        for (int i = len_substr; i < n; i++) {
            h1 = (h1 * BASE + (unsigned char)s[i]) % MOD1;
            h1 = (h1 + MOD1 - (pow1[len_substr] * (unsigned char)s[i - len_substr]) % MOD1) % MOD1;
            
            h2 = (h2 * BASE + (unsigned char)s[i]) % MOD2;
            h2 = (h2 + MOD2 - (pow2[len_substr] * (unsigned char)s[i - len_substr]) % MOD2) % MOD2;
            
            hashes.push_back({h1, h2});
        }
        
        return hashes;
    }
    
    ull combine_hash(const HashPair& hp) const {
        return hp.first * MOD2 + hp.second;
    }
