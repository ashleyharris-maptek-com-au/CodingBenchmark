   ```cpp
   #include <iostream>
   #include <vector>
   #include <queue>
   #include <set>
   #include <algorithm>
   #include <limits>

   using namespace std;

   // Use long long for weights to prevent overflow
   using ll = long long;
   const ll INF = 1e18;

   struct Edge {
       int to;
       int weight;
   };

   int main() {
       // Fast I/O
       ios_base::sync_with_stdio(false);
       cin.tie(NULL);

       int n, m, k;
       if (!(cin >> n >> m >> k)) return 0;

       vector<vector<Edge>> adj(n);
       for (int i = 0; i < m; ++i) {
           int u, v, w;
           cin >> u >> v >> w;
           if (u < 0 || u >= n || v < 0 || v >= n) continue; // Basic validation
           adj[u].push_back({v, w});
           adj[v].push_back({u, w});
       }

       vector<int> terminals(k);
       // Map terminal vertex to its index 0..k-1
       vector<int> terminal_indices(n, -1);
       for (int i = 0; i < k; ++i) {
           cin >> terminals[i];
           if (terminals[i] >= 0 && terminals[i] < n) {
               terminal_indices[terminals[i]] = i;
           }
       }

       // dp[mask][v] = min weight to connect terminals in mask ending at v
       // Dimensions: 2^k by n
       vector<vector<ll>> dp(1 << k, vector<ll>(n, INF));
       
       // parent[mask][v] stores reconstruction info
       // -1: start
       // 0..n-1: previous vertex (edge relaxation)
       // n..n+2^k-1: submask (merge)
       vector<vector<int>> parent(1 << k, vector<int>(n, -1));

       // Initialize
       for (int i = 0; i < k; ++i) {
           int t = terminals[i];
           // If terminal index is valid
           if (terminal_indices[t] == i) {
               dp[1 << i][t] = 0;
               // parent is -1 (start)
           }
       }

       // Dreyfus-Wagner
       for (int mask = 1; mask < (1 << k); ++mask) {
           // 1. Merge subsets
           // Iterate over non-empty proper subsets
           for (int sub = (mask - 1) & mask; sub > 0; sub = (sub - 1) & mask) {
               int rest = mask ^ sub;
               // We only process if sub <= rest to avoid redundant calculations?
               // No, let's just iterate all.
               // Optimization: if sub < rest, we can process. But the standard loop works.
               // Since sub and rest are both smaller than mask, their DP states are computed.
               
               // The update:
               // dp[mask][v] = min(dp[sub][v] + dp[rest][v])
               for (int v = 0; v < n; ++v) {
                   if (dp[sub][v] != INF && dp[rest][v] != INF) {
                       ll new_cost = dp[sub][v] + dp[rest][v];
                       if (new_cost < dp[mask][v]) {
                           dp[mask][v] = new_cost;
                           parent[mask][v] = n + sub; // Encode submask
                       }
                   }
               }
           }

           // 2. Dijkstra (Steiner DP Transitions)
           // Priority queue: {cost, vertex}
           // We need to push all valid vertices for the current mask
           // Optimization: We can run Dijkstra starting from the "sources" defined by current dp[mask] values.
           priority_queue<pair<ll, int>, vector<pair<ll, int>>, greater<pair<ll, int>>> pq;
           
           for (int v = 0; v < n; ++v) {
               if (dp[mask][v] != INF) {
                   pq.push({dp[mask][v], v});
               }
           }

           while (!pq.empty()) {
               auto [d, u] = pq.top();
               pq.pop();

               if (d > dp[mask][u]) continue;

               for (const auto& edge : adj[u]) {
                   if (dp[mask][u] + edge.weight < dp[mask][edge.to]) {
                       dp[mask][edge.to] = dp[mask][u] + edge.weight;
                       parent[mask][edge.to] = u; // Store vertex predecessor
                       pq.push({dp[mask][edge.to], edge.to});
                   }
               }
           }
       }

       // Result Extraction
       int full_mask = (1 << k) - 1;
       ll min_weight = INF;
       int final_node = -1;

       for (int v = 0; v < n; ++v) {
           if (dp[full_mask][v] < min_weight) {
               min_weight = dp[full_mask][v];
               final_node = v;
           }
       }

       if (min_weight == INF) {
           cout << "0 0\n"; // No solution
           return 0;
       }

       // Reconstruct Edges
       set<pair<int, int>> edges;
       // We use a stack for DFS traversal of the DP state graph
       vector<pair<int, int>> stk; // {mask, node}
       stk.push_back({full_mask, final_node});

       // To avoid processing the same state multiple times if logic allows (though tree structure usually implies unique paths back)
       // Actually, merges create branches. We must traverse all branches.
       // Standard DFS on the structure defined by 'parent'.
       
       while (!stk.empty()) {
           auto [mask, u] = stk.back();
           stk.pop_back();

           int p = parent[mask][u];

           if (p == -1) {
               continue; // Base case
           }

           if (p < n) {
               // Edge case: predecessor is vertex p
               int a = u, b = p;
               if (a > b) swap(a, b);
               edges.insert({a, b});
               stk.push_back({mask, p});
           } else {
               // Merge case: predecessor is submask
               int sub = p - n;
               int rest = mask ^ sub;
               // Push both states to process
               stk.push_back({sub, u});
               stk.push_back({rest, u});
           }
       }

       cout << min_weight << " " << edges.size() << "\n";
       for (const auto& edge : edges) {
           cout << edge.first << " " << edge.second << "\n";
       }

       return 0;
   }
   ```
