- R = reflect(-L, N) - reflection of light direction around normal

#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);
    
    // Light direction (from surface to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // View direction (from surface to camera)
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    
    // Reflection vector
    vec3 R = reflect(-L, N);
    
    // Ambient component
    vec3 ambient = vec3(0.1, 0.05, 0.05);
    
    // Diffuse component
    float NdotL = max(dot(N, L), 0.0);
    vec3 diffuse = color * NdotL;
    
    // Specular component
    float RdotV = max(dot(R, V), 0.0);
    float shininess = 32.0;
    vec3 specular = vec3(1.0) * pow(RdotV, shininess);
    
    // Combine lighting
    vec3 finalColor = ambient + diffuse + specular;
    
    // Clamp to [0, 1]
    finalColor = clamp(finalColor, 0.0, 1.0);
    
    // Output with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
