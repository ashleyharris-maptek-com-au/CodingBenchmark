        result[axis] += dir;
        return result;

        return false;

        return d;

            minD = min(minD, manhattanDist(p, f));
        }
        return minD;

        if (foods.empty()) return make_pair(-1, 0);

        Pos tail = snake.back();
        
        queue<pair<Pos, pair<int, int>>> q;
        PosSet visited;

        for (int axis = 0; axis < D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Pos next = movePos(head, axis, dir);
                if (!isBlocked(next, tail)) {
                    q.push(make_pair(next, make_pair(axis, dir)));
                    visited.insert(next);

                for (int dir = -1; dir <= 1; dir += 2) {

                    if (!isBlocked(next, tail) && visited.find(next) == visited.end()) {
                        visited.insert(next);
                        q.push(make_pair(next, firstMove));
                    }
                }
            }
        }
        
        return make_pair(-1, 0);
    }

        Pos tail = snake.back();
        
        for (int axis = 0; axis < D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Pos next = movePos(head, axis, dir);
                if (!isBlocked(next, tail)) {

                }

                        q.push(next);
                    }
                }
            }
        }

            return make_pair(0, 1);

            if (bfsMove.first >= 0 && validMoves[i] == bfsMove) {

            }

            snakeSet.erase(newHead);
            if (!willEat && snake.size() > 1) {
                snakeSet.insert(oldTail);
            }
            
            if (score > bestScore) {
                bestScore = score;
                bestMove = validMoves[i];
            }
        }
        
        return bestMove;
    }

        Pos newHead = movePos(snake.front(), axis, dir);
        
        bool willEat = foods.count(newHead) > 0;
        if (willEat) {
            foods.erase(newHead);
        }
        
        snake.push_front(newHead);
        snakeSet.insert(newHead);
        
        if (!willEat) {
            Pos tail = snake.back();
            snake.pop_back();
            snakeSet.erase(tail);
        }
    }

                bool extended = false;
                for (int a = 1; a < D && !extended; a++) {
                    if (headPos[a] > 0) {
                        cur = headPos;
                        cur[a]--;
                        snake.push_back(cur);
                        snakeSet.insert(cur);
                        extended = true;
                    }
                }
                if (!extended) break;
            } else {
                snake.push_back(cur);
                snakeSet.insert(cur);
            }
        }

        for (int i = 0; i < foodCount; i++) {
            Pos f(D);
            for (int j = 0; j < D; j++) {

            }

/*
 * N-Dimensional Snake Solver
 * 
 * Strategy:
 * 1. BFS to find path to nearest food
 * 2. Evaluate each valid move by:
 *    - Is it on the BFS path? (high priority)
 *    - How accessible is the resulting position? (safety)
 *    - How close to nearest food? (greedy fallback)
 * 3. Handle all dimensions from 2D to 20D with configurable limits
 */

#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <queue>
#include <climits>
#include <algorithm>

using namespace std;

// Custom hash for vector<int> to use in unordered_set/map
struct VecHash {
    size_t operator()(const vector<int>& v) const {
        size_t h = 0;
        for (int x : v) {
            h = h * 31 + static_cast<size_t>(x);
        }
        return h;
    }
};

using Pos = vector<int>;
using PosSet = unordered_set<Pos, VecHash>;

class SnakeSolver {
private:
    int D;                    // Number of dimensions
    vector<int> bounds;       // Size of each dimension
    deque<Pos> snake;         // Snake body (front = head, back = tail)
    PosSet snakeSet;          // Set of snake positions for O(1) lookup
    PosSet obstacles;         // Set of obstacle positions
    PosSet foods;             // Set of food positions
    int maxTurns;             // Maximum number of turns
    int bfsLimit;             // Node limit for BFS

    // Check if position is within bounds
    bool inBounds(const Pos& p) const {
        for (int i = 0; i < D; i++) {
            if (p[i] < 0 || p[i] >= bounds[i]) return false;
        }
        return true;
    }

    // Move position along an axis
    Pos movePos(const Pos& p, int axis, int dir) const {
        Pos result = p;
        result[axis] += dir;
        return result;
    }

    // Check if position is blocked (out of bounds, obstacle, or snake body)
    // exceptTail allows passing through the tail (which will move)
    bool isBlocked(const Pos& p, const Pos& exceptTail) const {
        if (!inBounds(p)) return true;
        if (obstacles.count(p)) return true;
        if (snakeSet.count(p) && p != exceptTail) return true;
        return false;
    }

    // Manhattan distance between two positions
    int manhattanDist(const Pos& a, const Pos& b) const {
        int d = 0;
        for (int i = 0; i < D; i++) {
            d += abs(a[i] - b[i]);
        }
        return d;
    }

    // Minimum Manhattan distance to any food
    int minFoodDist(const Pos& p) const {
        if (foods.empty()) return INT_MAX;
        int minD = INT_MAX;
        for (const Pos& f : foods) {
            minD = min(minD, manhattanDist(p, f));
        }
        return minD;
    }

    // BFS to find nearest food and return first move toward it
    pair<int, int> bfsToFood() {
        if (foods.empty()) return make_pair(-1, 0);

        Pos head = snake.front();
        Pos tail = snake.back();

        queue<pair<Pos, pair<int, int>>> q;  // (current position, first move)
        PosSet visited;

        // Initialize with all valid moves from head
        for (int axis = 0; axis < D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Pos next = movePos(head, axis, dir);
                if (!isBlocked(next, tail)) {
                    q.push(make_pair(next, make_pair(axis, dir)));
                    visited.insert(next);
                }
            }
        }

        int nodes = 0;
        while (!q.empty() && nodes < bfsLimit) {
            auto front = q.front();
            q.pop();
            Pos cur = front.first;
            pair<int, int> firstMove = front.second;
            nodes++;

            if (foods.count(cur)) {
                return firstMove;
            }

            for (int axis = 0; axis < D; axis++) {
                for (int dir = -1; dir <= 1; dir += 2) {
                    Pos next = movePos(cur, axis, dir);
                    if (!isBlocked(next, tail) && visited.find(next) == visited.end()) {
                        visited.insert(next);
                        q.push(make_pair(next, firstMove));
                    }
                }
            }
        }

        return make_pair(-1, 0);  // No path found within limit
    }

        vector<pair<int, int>> moves;
        Pos head = snake.front();
        Pos tail = snake.back();

        for (int axis = 0; axis < D; axis++) {
            for (int dir = -1; dir <= 1; dir += 2) {
                Pos next = movePos(head, axis, dir);
                if (!isBlocked(next, tail)) {
                    moves.push_back(make_pair
