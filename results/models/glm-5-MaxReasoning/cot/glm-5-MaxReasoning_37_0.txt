This is the Directed Feedback Vertex Set (DFVS) problem, which is NP-hard. I need to find the minimum set of vertices to remove from a directed graph to make it acyclic.

Let me think about approaches:

1. **Brute Force**: Try all subsets of vertices - O(2^n * m) - only works for very small n.

2. **ILP (Integer Linear Programming)**: Can be used for moderate-sized graphs, but requires an external solver.

3. **Approximation Algorithms**: There are approximation algorithms, but we need the exact minimum.

4. **Branch and Bound**: A common approach for exact NP-hard problems.

5. **Reduction Rules**: There are several reduction rules that can simplify the problem:
   - Remove vertices with in-degree or out-degree 0 (they can't be part of any cycle)
   - Self-loops: vertices with self-loops must be in the FVS
   - Degree-2 reduction in certain cases
   -etc.

6. **Iterative Compression**: A well-known technique for DFVS that runs in O(5^k * k * n^2) where k is the solution size.

    }
    
    static void Reduce(int n, List<int>[] adj, List<int>[] radj, bool[] deleted, Stack<int> reduced)
    {
        var queue = new Queue<int>();
        
        for (int i = 0; i < n; i++)
        {
            if (!deleted[i] && (InDeg(radj, deleted, i) == 0 || OutDeg(adj, deleted, i) == 0))
                queue.Enqueue(i);
        }
        
        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (deleted[v]) continue;
            
            if (InDeg(radj, deleted, v) == 0 || OutDeg(adj, deleted, v) == 0)
            {
                deleted[v] = true;
                reduced.Push(v);
                
                foreach (int u in radj[v])
                {
                    if (!deleted[u] && OutDeg(adj, deleted, u) == 0)
                        queue.Enqueue(u);
                }
                foreach (int u in adj[v])
                {
                    if (!deleted[u] && InDeg(radj, deleted, u) == 0)
                        queue.Enqueue(u);
                }
            }
        }
    }
    
    static void Restore(bool[] deleted, Stack<int> reduced)
    {
        while (reduced.Count > 0)
            deleted[reduced.Pop()] = false;
    }
    
    static int InDeg(List<int>[] radj, bool[] deleted, int v)
    {
        int count = 0;
        foreach (int u in radj[v])
            if (!deleted[u]) count++;
        return count;
    }
    
    static int OutDeg(List<int>[] adj, bool[] deleted, int v)
    {
        int count = 0;
        foreach (int u in adj[v])
            if (!deleted[u]) count++;
        return count;
    }
    
    static List<int> FindCycle(int n, List<int>[] adj, bool[] deleted)
    {
        var color = new int[n];
        var parent = new int[n];
        Array.Fill(parent, -1);
        
        for (int start = 0; start < n; start++)
        {
            if (deleted[start] || color[start] != 0) continue;
            
            var stack = new Stack<int>();
            stack.Push(start);
            
            while (stack.Count > 0)
            {
                int v = stack.Peek();
                
                if (color[v] == 2)
                {
                    stack.Pop();
                    continue;
                }
                
                if (color[v] == 1)
                {
                    color[v] = 2;
                    stack.Pop();
                    continue;
                }
                
                color[v] = 1;
                
                foreach (int u in adj[v])
                {
                    if (deleted[u]) continue;
                    
                    if (color[u] == 1)
                    {
                        var cycle = new List<int>();
                        int cur = v;
                        while (cur != u && cur != -1)
                        {
                            cycle.Add(cur);
                            cur = parent[cur];
                        }
                        if (cur == u)
                        {
                            cycle.Add(u);
                            return cycle;
                        }
                    }
                    
                    if (color[u] == 0)
                    {
                        parent[u] = v;
                        stack.Push(u);
                    }
                }
            }
        }
        
        return null;
    }

static List<int> BruteForce(int n, List<int>[] adj)
{
    for (int k = 0; k <= n; k++)
    {
        // Generate all subsets of size k
        var fvs = new List<int>();
        if (TrySubset(n, adj, k, 0, fvs))
            return fvs;
    }
    return new List<int>();
}

static bool TrySubset(int n, List<int>[] adj, int k, int start, List<int> fvs)
{
    if (fvs.Count == k)
    {
        return IsDAG(n, adj, fvs);
    }
    
    if (start >= n) return false;
    
    // Include start
    fvs.Add(start);
    if (TrySubset(n, adj, k, start + 1, fvs))
        return true;
    fvs.RemoveAt(fvs.Count - 1);
    
    // Exclude start
    if (TrySubset(n, adj, k, start + 1, fvs))
        return true;
    
    return false;
}

            return new List<int>(fvs);
        return null;

    static bool Search(bool[] deleted, List<int> fvs, int maxSize)

        {
            Restore(deleted, stack);
            return false;
        }

            if (Search(deleted, fvs, maxSize))
                return true;
            
            fvs.RemoveAt(fvs.Count - 1);
            deleted[v] = false;
        }
        
        Restore(deleted, stack);
        return false;

    static void Reduce(bool[] deleted, Stack<int> stack)
    {
        var queue = new Queue<int>();
        
        for (int i = 0; i < n; i++)
        {
            if (!deleted[i])
            {
                int ind = InDegree(deleted, i);
                int outd = OutDegree(deleted, i);
                if (ind == 0 || outd == 0)
                    queue.Enqueue(i);
            }

        while (queue.Count > 0)
        {
            int v = queue.Dequeue();
            if (deleted[v]) continue;
            
            int ind = InDegree(deleted, v);
            int outd = OutDegree(deleted, v);

                {
                    if (!deleted[u] && OutDegree(deleted, u) == 0)
                        queue.Enqueue(u);
                }

                foreach (int u in adj[v])
                {
                    if (!deleted[u] && InDegree(deleted, u) == 0)
                        queue.Enqueue(u);
                }
            }
        }

            deleted[stack.Pop()] = false;

    static int InDegree(bool[] deleted, int v)

        int count = 0;
        foreach (int u in radj[v])
            if (!deleted[u]) count++;

    }
    
    static int OutDegree(bool[] deleted, int v)
    {
        int count = 0;
        foreach (int u in adj[v])
            if (!deleted[u]) count++;
        return count;
    }
    
    static int LowerBound(bool[] deleted)

        var temp = (bool[])deleted.Clone();
        int count = 0;
        
        while (true)
        {
            var cycle = FindCycle(temp);
            if (cycle == null) break;
            
            count++;
            foreach (int v in cycle)
                temp[v] = true;
        }
        
        return count;
    }
    
    static List<int> FindCycle(bool[] deleted)
    {

        var parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;
        
        for (int start = 0; start < n; start++)
        {
            if (deleted[start] || state[start] != 0) continue;
            
            var dfs = new Stack<int>();
            dfs.Push(start);
            
            while (dfs.Count > 0)
            {
                int v = dfs.Peek();
                
                if (state[v] == 2)
                {
                    dfs.Pop();
                    continue;
                }
                
                if (state[v] == 1)
                {
                    state[v] = 2;
                    dfs.Pop();
                    continue;
                }
                
                state[v] = 1;
                
                foreach (int u in adj[v])
                {
                    if (deleted[u]) continue;
                    
                    if (state[u] == 1)

                        // Found cycle

                        int cur = v;
                        while (cur != u)

                            cur = parent[cur];

                        }

                    if (state[u] == 0)
                    {
                        parent[u] = v;
                        dfs.Push(u);
                    }
                }
            }
        }
        
        return null;
    }
