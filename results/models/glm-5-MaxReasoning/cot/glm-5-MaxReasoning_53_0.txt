    _prev_pos = pos
    _prev_vel = vel
    
    return thrust

def guidance(pos, vel, range_m, phase):
    """Generate target velocity based on position and phase."""
    x, y, z = pos
    vx, vy, vz = vel if vel else [0, 0, 0]
    
    if phase == 'FAR':
        # Far range guidance
        # Use orbital mechanics: to catch up when behind, lower orbit
        
        # Along-track guidance
        if y < -200:
            # Far behind - aggressive approach
            target_vy = min(8.0, abs(y) * 0.008 + 2.0)
        elif y < -50:
            target_vy = min(4.0, abs(y) * 0.015)
        elif y < 0:
            target_vy = min(2.0, abs(y) * 0.02)
        elif y > 200:
            # Far ahead - need to slow down
            target_vy = -min(8.0, abs(y) * 0.008 + 2.0)
        elif y > 50:
            target_vy = -min(4.0, abs(y) * 0.015)
        else:
            target_vy = -y * 0.02
        
        # Radial guidance
        # For R-bar approach (x offset), use different strategy
        if abs(y) < 100:  # If y offset is small
            target_vx = -x * 0.015  # Correct radial offset
        else:
            target_vx = -x * 0.005  # Gentler correction if far in y
        
        # Cross-track guidance
        target_vz = -z * 0.01
        
    elif phase == 'MEDIUM':
        # Medium range - smoother approach
        target_vy = min(1.5, abs(y) * 0.02) if y < 0 else -min(0.8, abs(y) * 0.02)
        target_vx = -x * 0.02
        target_vz = -z * 0.02
        
    elif phase == 'CLOSE':
        # Close range - slow approach
        target_vy = min(0.25, abs(y) * 0.025) if y < 0 else 0
        target_vx = -x * 0.03
        target_vz = -z * 0.03
        
    else:  # DOCKING
        # Final approach - very slow
        target_vy = 0.04 if y < -0.5 else 0.02 if y < 0 else 0
        target_vx = -x * 0.08
        target_vz = -z * 0.08
    
    return [target_vx, target_vy, target_vz]

def control(pos, vel, target_v, range_m, phase, fuel_pct):
    """Compute thrust commands to achieve target velocity."""
    x, y, z = pos
    vx, vy, vz = vel if vel else [0, 0, 0]
    tvx, tvy, tvz = target_v
    
    # Velocity error
    evx = tvx - vx
    evy = tvy - vy
    evz = tvz - vz
    
    # Gains
    if phase == 'FAR':
        kv = 0.12
        kp = 0.002
    elif phase == 'MEDIUM':
        kv = 0.2
        kp = 0.008
    elif phase == 'CLOSE':
        kv = 0.3
        kp = 0.015
    else:
        kv = 0.4
        kp = 0.025
    
    # Calculate thrust
    thrust_x = kv * evx + kp * (-x)
    thrust_y = kv * evy + kp * (-y) * 0.2
    thrust_z = kv * evz + kp * (-z)
    
    # Clamp thrust
    thrust_x = clamp(thrust_x, -1.0, 1.0)
    thrust_y = clamp(thrust_y, -1.0, 1.0)
    thrust_z = clamp(thrust_z, -1.0, 1.0)
    
    # Low fuel mode
    if fuel_pct < 15:
        scale = max(0.3, fuel_pct / 15.0)
        thrust_x *= scale
        thrust_y *= scale
        thrust_z *= scale
    
    return {
        'thrust_x': thrust_x,
        'thrust_y': thrust_y,
        'thrust_z': thrust_z,
        'att_rate_roll': 0.0,
        'att_rate_pitch': 0.0,
        'att_rate_yaw': 0.0
    }

def get_fused_position(state):
    """Fuse position data from all available sensors."""
    pos = [0.0, 0.0, 0.0]
    count = [0, 0, 0]
    weight = [1.0, 1.0, 1.0]
    
    # Lidar (more accurate at close range)
    if state.get('lidar_rel_x_valid', False):
        pos[0] += state.get('lidar_rel_pos_x_m', 0) * weight[0]
        count[0] += weight[0]
    if state.get('lidar_rel_y_valid', False):

        count[1] += weight[1]
    if state.get('lidar_rel_z_valid', False):
        pos[2] += state.get('lidar_rel_pos_z_m', 0) * weight[2]
        count[2] += weight[2]
    
    # Radar
    if state.get('radar_rel_x_valid', False):
        pos[0] += state.get('radar_rel_pos_x_m', 0)
        count[0] += 1
    if state.get('radar_rel_y_valid', False):
        pos[1] += state.get('radar_rel_pos_y_m', 0)
        count[1] += 1
    if state.get('radar_rel_z_valid', False):
        pos[2] += state.get('radar_rel_pos_z_m', 0)
        count[2] += 1
    
    # Docking camera (only active within 500m)
    if state.get('dock_camera_active', False):
        if all(state.get(f'dock_camera_bearing_{a}_valid', False) for a in ['x', 'y', 'z']):
            bearing = [
                state.get('dock_camera_bearing_x', 0),
                state.get('dock_camera_bearing_y', 0),
                state.get('dock_camera_bearing_z', 0)
            ]
            cam_range = state.get('dock_camera_range_m', 0)

                    pos[i] += cam_pos[i] * w
                    count[i] += w

        return None
    
    return [pos[i] / count[i] for i in range(3)]
