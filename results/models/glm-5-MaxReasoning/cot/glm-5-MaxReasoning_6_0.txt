```csharp
    static int[] SolveHeldKarp(double[,] costs, int n)
    {
        // Held-Karp dynamic programming for TSP
        // Time: O(n² * 2^n), Space: O(n * 2^n)
        
        int maxStates = 1 << n;
        double[,] dp = new double[maxStates, n];
        int[,] parent = new int[maxStates, n];
        
        // Initialize
        for (int s = 0; s < maxStates; s++)
        {
            for (int i = 0; i < n; i++)
            {
                dp[s, i] = double.MaxValue;
                parent[s, i] = -1;
            }
        }
        
        // Base case: start from node 0
        dp[1, 0] = 0;
        
        // Fill DP table
        for (int s = 1; s < maxStates; s++)
        {
            if ((s & 1) == 0) continue; // Must include starting node
            
            for (int j = 0; j < n; j++)
            {
                if ((s & (1 << j)) == 0) continue;
                if (dp[s, j] == double.MaxValue) continue;
                
                int sWithoutJ = s ^ (1 << j);
                
                for (int k = 0; k < n; k++)
                {
                    if ((s & (1 << k)) != 0) continue;
                    
                    int sWithK = s | (1 << k);
                    double newCost = dp[s, j] + costs[j, k];
                    
                    if (newCost < dp[sWithK, k])
                    {
                        dp[sWithK, k] = newCost;
                        parent[sWithK, k] = j;
                    }
                }
            }
        }
        
        // Find minimum cost to complete tour
        int fullSet = maxStates - 1;
        double minCost = double.MaxValue;
        int lastNode = -1;
        
        for (int j = 1; j < n; j++)
        {
            double tourCost = dp[fullSet, j] + costs[j, 0];
            if (tourCost < minCost)
            {
                minCost = tourCost;
                lastNode = j;
            }
        }
        
        // Reconstruct path
        int[] result = new int[n];
        int state = fullSet;
        int current = lastNode;
        
        for (int i = n - 1; i >= 0; i--)
        {
            result[i] = current;
            int prev = parent[state, current];
            state ^= (1 << current);
            current = prev;
        }
        
        return result;
    }
```

2. This puts us on a transfer trajectory
3. At the destination, we perform another maneuver to match the target orbit

The cost is the sum of all delta-Vs required.

For simplicity, I'll compute:
1. The "distance" between two orbits based on their elements
2. Use this as the edge weight for TSP

Let me write a cleaner solution:

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    const double MU = 398600.4418;
    const double EARTH_RADIUS = 6371.0;
    
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public double Len => Math.Sqrt(X*X + Y*Y + Z*Z);
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X+b.X, a.Y+b.Y, a.Z+b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X-b.X, a.Y-b.Y, a.Z-b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X*k, a.Y*k, a.Z*k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X/k, a.Y/k, a.Z/k);
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y*b.Z - a.Z*b.Y, a.Z*b.X - a.X*b.Z, a.X*b.Y - a.Y*b.X);
        public static double Dot(Vec3 a, Vec3 b) => a.X*b.X + a.Y*b.Y + a.Z*b.Z;
    }
    
    struct OrbitalElements
    {
        public double a, e, i, Omega, omega, nu;
        public double rp, ra; // periapsis, apoapsis
        public Vec3 r, v;
    }
    
    static OrbitalElements ComputeElements(Vec3 r, Vec3 v)
    {
        var o = new OrbitalElements();
        o.r = r; o.v = v;
        
        double rm = r.Len, vm = v.Len;
        Vec3 h = Vec3.Cross(r, v);
        double hm = h.Len;
        
        // Eccentricity vector
        Vec3 evec = (r * (vm*vm - MU/rm) - v * Vec3.Dot(r, v)) / MU;
        o.e = evec.Len;
        
        // Semi-major axis
        o.a = 1.0 / (2.0/rm - vm*vm/MU);
        
        // Inclination
        o.i = Math.Acos(Math.Max(-1, Math.Min(1, h.Z / hm)));
        
        // Node vector
        Vec3 n = new Vec3(-h.Y, h.X, 0);
        double nm = n.Len;
        
        // RAAN
        if (nm > 1e-10)
        {
            o.Omega = Math.Acos(Math.Max(-1, Math.Min(1, n.X / nm)));
            if (n.Y < 0) o.Omega = 2*Math.PI - o.Omega;
        }
        else o.Omega = 0;
        
        // Argument of periapsis
        if (nm > 1e-10 && o.e > 1e-10)
        {
            o.omega = Math.Acos(Math.Max(-1, Math.Min(1, Vec3.Dot(n, evec) / (nm * o.e))));
            if (evec.Z < 0) o.omega = 2*Math.PI - o.omega;
        }
        else o.omega = 0;
        
        // True anomaly
        if (o.e > 1e-10)
        {
            o.nu = Math.Acos(Math.Max(-1, Math.Min(1, Vec3.Dot(evec, r) / (o.e * rm))));
            if (Vec3.Dot(r, v) < 0) o.nu = 2*Math.PI - o.nu;
        }
        else o.nu = 0;
        
        o.rp = o.a * (1 - o.e);
        o.ra = o.a * (1 + o.e);
        
        return o;
    }
    
    static double TransferCost(OrbitalElements o1, OrbitalElements o2)
    {
        // Simplified model based on orbital elements
        double r1 = o1.r.Len, r2 = o2.r.Len;
        
        // Characteristic velocity
        double vc = Math.Sqrt(MU / ((o1.a + o2.a) / 2));
        
        // Hohmann-like transfer cost
        double dv_a = Math.Abs(Math.Sqrt(MU/o1.a) - Math.Sqrt(MU/o2.a)) * 0.5;
        
        // Inclination change
        double di = Math.Abs(o1.i - o2.i);
        if (di > Math.PI) di = 2*Math.PI - di;
        double dv_i = vc * Math.Sin(di);
        
        // RAAN change
        double dO = Math.Abs(o1.Omega - o2.Omega);
        if (dO > Math.PI) dO = 2*Math.PI - dO;
        double dv_O = vc * Math.Sin(dO) * Math.Sin((o1.i + o2.i)/2) * 0.5;
        
        // Eccentricity penalty
        double eFactor = 1 + Math.Abs(o1.e - o2.e);
        
        // Position-based distance component
        Vec3 dr = o1.r - o2.r;
        double posDist = dr.Len / r1; // normalized
        
        return (dv_a + dv_i + dv_O) * eFactor + vc * posDist * 0.1;
    }
    
    static double[,] BuildCosts(OrbitalElements[] orbits)
    {
        int n = orbits.Length;
        double[,] c = new double[n, n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                c[i,j] = i == j ? 0 : TransferCost(orbits[i], orbits[j]);
        return c;
    }
    
    // TSP Solvers
    static int[] BruteForce(double[,] c, int n)
    {
        int[] best = Enumerable.Range(0, n).ToArray();
        double bestCost = TourCost(c, best);
        int[] perm = (int[])best.Clone();
        
        while (NextPerm(perm))
        {
            double cost = TourCost(c, perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                best = (int[])perm.Clone();
            }
        }
        return best;
    }
    
    static bool NextPerm(int[] a)
    {
        int n = a.Length, i = n - 2;
        while (i >= 0 && a[i] >= a[i+1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (a[j] <= a[i]) j--;
        (a[i], a[j]) = (a[j], a[i]);
        Array.Reverse(a, i+1, n-i-1);
        return true;
    }
    
    static double TourCost(double[,] c, int[] tour)
    {
        double sum = 0;
        for (int i = 0; i < tour.Length - 1; i++) sum += c[tour[i], tour[i+1]];
        return sum;
    }
    
    static int[] HeldKarp(double[,] c, int n)
    {
        int maxS = 1 << n;
        double[,] dp = new double[maxS, n];
        int[,] parent = new int[maxS, n];
        
        for (int s = 0; s < maxS; s++)
            for (int i = 0; i < n; i++)
            {
                dp[s, i] = double.MaxValue;
                parent[s, i] = -1;
            }
        
        dp[1, 0] = 0;
        
        for (int s = 1; s < maxS; s++)
        {
            if ((s & 1) == 0) continue;
            for (int j = 0; j < n; j++)
            {
                if ((s & (1 << j)) == 0 || dp[s, j] == double.MaxValue) continue;
                for (int k = 0; k < n; k++)
                {
                    if ((s & (1 << k)) != 0) continue;
                    int s2 = s | (1 << k);
                    double newCost = dp[s, j] + c[j, k];
                    if (newCost < dp[s2, k])
                    {
                        dp[s2, k] = newCost;
                        parent[s2, k] = j;
                    }
                }
            }
        }
        
        int full = maxS - 1;
        double minC = double.MaxValue;
        int last = -1;
        for (int j = 1; j < n; j++)
        {
            double cost = dp[full, j] + c[j, 0];
            if (cost < minC) { minC = cost; last = j; }
        }
        
        int[] result = new int[n];
        int state = full, cur = last;
        for (int i = n - 1; i >= 0; i--)
        {
            result[i] = cur;
            int prev = parent[state, cur];
            state ^= (1 << cur);
            cur = prev;
        }
        return result;
    }
    
    static int[] NearestNeighbor(double[,] c, int n)
    {
        int[] tour = new int[n];
        bool[] vis = new bool[n];
        tour[0] = 0; vis[0] = true;
        
        for (int i = 1; i < n; i++)
        {
            int last = tour[i-1];
            int best = -1;
            double bestD = double.MaxValue;
            for (int j = 0; j < n; j++)
                if (!vis[j] && c[last, j] < bestD) { bestD = c[last, j]; best = j; }
            tour[i] = best;
            vis[best] = true;
        }
        return tour;
    }
    
    static int[] Opt2(double[,] c, int[] tour)
    {
        int n = tour.Length;
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    int ni = (i + 1) % n, nj = (j + 1) % n;
                    double oldC = c[tour[i], tour[ni]] + c[tour[j], tour[nj]];
                    double newC = c[tour[i], tour[j]] + c[tour[ni], tour[nj]];
                    if (newC < oldC - 1e-10)
                    {
                        Array.Reverse(tour, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
        return tour;
    }
    
    static int[] SimulatedAnnealing(double[,] c, int n, Random rng)
    {
        int[] best = NearestNeighbor(c, n);
        best = Opt2(c, best);
        double bestCost = TourCost(c, best);
        
        int[] current = (int[])best.Clone();
        double currentCost = bestCost;
        
        double temp = 1000;
        int maxIter = Math.Min(5000000, n * n * 500);
        
        for (int iter = 0; iter < maxIter && temp > 1e-8; iter++)
        {
            // 2-opt style neighbor
            int i = rng.Next(n);
            int j = rng.Next(n);
            if (i == j) continue;
            if (i > j) (i, j) = (j, i);
            
            int[] neighbor = (int[])current.Clone();
            Array.Reverse(neighbor, i, j - i + 1);
            double neighborCost = TourCost(c, neighbor);
            
            if (neighborCost < currentCost || rng.NextDouble() < Math.Exp(-(neighborCost - currentCost) / temp))
            {
                current = neighbor;
                currentCost = neighborCost;
                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    best = (int[])current.Clone();
                }
            }
            temp *= 0.99997;
        }
        
        return Opt2(c, best);
    }
    
    static void Main()
    {
        var parts = Console.ReadLine().Split();
        int n = int.Parse(parts[0]);
        
        var startLine = Console.ReadLine().Split();
        // We read start orbit but primarily work with station orbits
        
        var orbits = new OrbitalElements[n];
        for (int i = 0; i < n; i++)
        {
            var p = Console.ReadLine().Split();
            Vec3 r = new Vec3(double.Parse(p[0]), double.Parse(p[1]), double.Parse(p[2]));
            Vec3 v = new Vec3(double.Parse(p[3]), double.Parse(p[4]), double.Parse(p[5]));
            orbits[i] = ComputeElements(r, v);
        }
        
        var costs = BuildCosts(orbits);
        
        int[] result;
        if (n <= 8) result = BruteForce(costs, n);
        else if (n <= 20) result = HeldKarp(costs, n);
        else result = SimulatedAnnealing(costs, n, new Random(42));
        
        Console.WriteLine(string.Join(" ", result));
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSPSolver
{
    const double MU = 398600.4418;
    const double PI = Math.PI;
    
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public double Len => Math.Sqrt(X*X + Y*Y + Z*Z);
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X+b.X, a.Y+b.Y, a.Z+b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X-b.X, a.Y-b.Y, a.Z-b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X*k, a.Y*k, a.Z*k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X/k, a.Y/k, a.Z/k);
        public static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(a.Y*b.Z - a.Z*b.Y, a.Z*b.X - a.X*b.Z, a.X*b.Y - a.Y*b.X);
        public static double Dot(Vec3 a, Vec3 b) => a.X*b.X + a.Y*b.Y + a.Z*b.Z;
    }
    
    struct OrbitalElements
    {
        public double a, e, i, Omega, omega, nu, rp, ra;
        public Vec3 r, v;
    }
    
    static OrbitalElements ComputeElements(Vec3 r, Vec3 v)
    {
        var o = new OrbitalElements { r = r, v = v };
        double rm = r.Len, vm = v.Len;
        Vec3 h = Vec3.Cross(r, v);
        double hm = h.Len;
        
        Vec3 evec = (r * (vm*vm - MU/rm) - v * Vec3.Dot(r, v)) / MU;
        o.e = evec.Len;
        o.a = 1.0 / (2.0/rm - vm*vm/MU);
        o.i = Math.Acos(Math.Clamp(h.Z / hm, -1, 1));
        
        Vec3 n = new Vec3(-h.Y, h.X, 0);
        double nm = n.Len;
        
        o.Omega = nm > 1e-10 ? Math.Acos(Math.Clamp(n.X / nm, -1, 1)) : 0;
        if (n.Y < 0 && o.Omega > 0) o.Omega = 2*PI - o.Omega;
        
        o.omega = (nm > 1e-10 && o.e > 1e-10) ? 
            Math.Acos(Math.Clamp(Vec3.Dot(n, evec) / (nm * o.e), -1, 1)) : 0;
        if (evec.Z < 0 && o.omega > 0) o.omega = 2*PI - o.omega;
        
        o.nu = o.e > 1e-10 ? Math.Acos(Math.Clamp(Vec3.Dot(evec, r) / (o.e * rm), -1, 1)) : 0;
        if (Vec3.Dot(r, v) < 0 && o.nu > 0) o.nu = 2*PI - o.nu;
        
        o.rp = o.a * (1 - o.e);
        o.ra = o.a * (1 + o.e);
        return o;
    }
    
    static double TransferCost(OrbitalElements o1, OrbitalElements o2)
    {
        double vc = Math.Sqrt(MU / ((o1.a + o2.a) / 2));
        
        // Semi-major axis transfer
        double dv_a = Math.Abs(Math.Sqrt(MU/o1.a) - Math.Sqrt(MU/o2.a));
        
        // Inclination change
        double di = Math.Abs(o1.i - o2.i);
        if (di > PI) di = 2*PI - di;
        double dv_i = vc * Math.Sin(di);
        
        // RAAN change
        double dO = Math.Abs(o1.Omega - o2.Omega);
        if (dO > PI) dO = 2*PI - dO;
        double dv_O = vc * Math.Sin(dO) * Math.Sin((o1.i + o2.i)/2);
        
        double eFactor = 1 + Math.Abs(o1.e - o2.e);
        
        return (dv_a + dv_i + dv_O) * eFactor;
    }
    
    static double TourCost(double[,] c, double[] sc, int[] tour)
    {
        double cost = sc[tour[0]];
        for (int i = 0; i < tour.Length - 1; i++)
            cost += c[tour[i], tour[i+1]];
        return cost;
    }
    
    static bool NextPerm(int[] a)
    {
        int n = a.Length, i = n - 2;
        while (i >= 0 && a[i] >= a[i+1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (a[j] <= a[i]) j--;
        (a[i], a[j]) = (a[j], a[i]);
        Array.Reverse(a, i+1, n-i-1);
        return true;
    }
    
    static int[] BruteForce(double[,] c, double[] sc, int n)
    {
        int[] best = Enumerable.Range(0, n).ToArray();
        double bestCost = TourCost(c, sc, best);
        int[] perm = (int[])best.Clone();
        
        while (NextPerm(perm))
        {
            double cost = TourCost(c, sc, perm);
            if (cost < bestCost)
            {
                bestCost = cost;
                best = (int[])perm.Clone();
            }
        }
        return best;
    }
    
    static int[] HeldKarp(double[,] c, double[] sc, int n)
    {
        int maxS = 1 << n;
        double[,] dp = new double[maxS, n];
        int[,] parent = new int[maxS, n];
        
        for (int s = 0; s < maxS; s++)
            for (int i = 0; i < n; i++)
            {
                dp[s, i] = double.MaxValue;
                parent[s, i] = -1;
            }
        
        // Initialize with costs from start
        for (int i = 0; i < n; i++)
            dp[1 << i, i] = sc[i];
        
        for (int s = 1; s < maxS; s++)
        {
            for (int j = 0; j < n; j++)
            {
                if ((s & (1 << j)) == 0 || dp[s, j] == double.MaxValue) continue;
                for (int k = 0; k < n; k++)
                {
                    if ((s & (1 << k)) != 0) continue;
                    int s2 = s | (1 << k);
                    double newCost = dp[s, j] + c[j, k];
                    if (newCost < dp[s2, k])
                    {
                        dp[s2, k] = newCost;
                        parent[s2, k] = j;
                    }
                }
            }
        }
        
        int full = maxS - 1;
        double minC = double.MaxValue;
        int last = 0;
        for (int j = 0; j < n; j++)
            if (dp[full, j] < minC) { minC = dp[full, j]; last = j; }
        
        int[] result = new int[n];
        int state = full, cur = last;
        for (int i = n - 1; i >= 0; i--)
        {
            result[i] = cur;
            int prev = parent[state, cur];
            state ^= (1 << cur);
            cur = prev;
        }
        return result;
    }
    
    static int[] NearestNeighbor(double[,] c, double[] sc, int n)
    {
        int[] tour = new int[n];
        bool[] vis = new bool[n];
        
        int first = 0;
        double minSc = sc[0];
        for (int i = 1; i < n; i++)
            if (sc[i] < minSc) { minSc = sc[i]; first = i; }
        
        tour[0] = first;
        vis[first] = true;
        
        for (int i = 1; i < n; i++)
        {
            int last = tour[i-1];
            int best = -1;
            double bestD = double.MaxValue;
            for (int j = 0; j < n; j++)
                if (!vis[j] && c[last, j] < bestD) { bestD = c[last, j]; best = j; }
            tour[i] = best;
            vis[best] = true;
        }
        return tour;
    }
    
    static int[] Opt2(double[,] c, double[] sc, int[] tour)
    {
        int n = tour.Length;
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1 && !improved; i++)
            {
                for (int j = i + 2; j < n && !improved; j++)
                {
                    double oldC = (i == 0 ? sc[tour[0]] : c[tour[i-1], tour[i]]) + c[tour[i], tour[i+1]];
                    double newC = (i == 0 ? sc[tour[j]] : c[tour[i-1], tour[j]]) + c[tour[j], tour[i+1]];
                    
                    // Actually need to recalculate the full affected segment
                    // Simplified 2-opt: reverse segment and check improvement
                    
                    double delta = c[tour[i], tour[i+1]] + c[tour[j], tour[(j+1)%n]] 
                                 - c[tour[i], tour[j]] - c[tour[i+1], tour[(j+1)%n]];
                    
                    if (delta > 1e-10)
                    {
                        Array.Reverse(tour, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
        return tour;
    }
    
    static int[] SimulatedAnnealing(double[,] c, double[] sc, int n, Random rng)
    {
        int[] best = NearestNeighbor(c, sc, n);
        double bestCost = TourCost(c, sc, best);
        
        int[] current = (int[])best.Clone();
        double currentCost = bestCost;
        
        double temp = 1000;
        int maxIter = Math.Min(5000000, n * n * 500);
        
        for (int iter = 0; iter < maxIter && temp > 1e-8; iter++)
        {
            int i = rng.Next(n);
            int j = rng.Next(n);
            if (i == j) continue;
            if (i > j) (i, j) = (j, i);
            if (j - i < 2) continue;
            
            int[] neighbor = (int[])current.Clone();
            Array.Reverse(neighbor, i, j - i + 1);
            double neighborCost = TourCost(c, sc, neighbor);
            
            if (neighborCost < currentCost || rng.NextDouble() < Math.Exp(-(neighborCost - currentCost) / temp))
            {
                current = neighbor;
                currentCost = neighborCost;
                if (currentCost < bestCost)
                {
                    bestCost = currentCost;
                    best = (int[])current.Clone();
                }
            }
            temp *= 0.99997;
        }
        
        return best;
    }
    
    static void Main()
    {
        var parts = Console.ReadLine().Split();
        int n = int.Parse(parts[0]);
```

Actually, let me think about what would be more accurate. In orbital mechanics, the transfer cost depends heavily on:
1. The relative positions of the two objects
2. The orbital planes (inclination, RAAN)
3. The semi-major axes

For a Hohmann transfer (circular orbits, same plane):
- ΔV = |√(μ/r₁) · (√(2r₂/(r₁+r₂)) - 1)| + |√(μ/r₂) · (1 - √(2r₁/(r₁+r₂)))|

For an inclination change:
- ΔV = 2v · sin(Δi/2)

For a combined maneuver (plane change + altitude change):
- ΔV = √(v₁² + v₂² - 2v₁v₂cos(Δi))

```csharp
static double TransferCost(OrbitalElements o1, OrbitalElements o2)
{
    // Get approximate orbital radii
    double r1 = o1.a, r2 = o2.a;
    
    // Approximate velocities at transfer points
    double v1 = Math.Sqrt(MU / r1);
    double v2 = Math.Sqrt(MU / r2);
    
    // Inclination difference
    double di = Math.Abs(o1.i - o2.i);
    if (di > Math.PI) di = 2 * Math.PI - di;
    
    // RAAN difference
    double dO = Math.Abs(o1.Omega - o2.Omega);
    if (dO > Math.PI) dO = 2 * Math.PI - dO;
    
    // Combined plane change angle
    // The angle between orbital planes
    double phi = Math.Acos(Math.Cos(di) * Math.Cos(dO) + Math.Sin(di) * Math.Sin(dO) * Math.Cos((o1.omega + o1.nu) - (o2.omega + o2.nu)));
    if (double.IsNaN(phi)) phi = di;
    
    // Approximate transfer cost
    // For different semi-major axes, use average velocity
    double vAvg = (v1 + v2) / 2;
    
    // Hohmann-like transfer cost
    double rTransfer = (r1 + r2) / 2;
    double vTransfer1 = Math.Sqrt(2 * MU * r2 / (r1 * (r1 + r2)));
    double vTransfer2 = Math.Sqrt(2 * MU * r1 / (r2 * (r1 + r2)));
    double dvHohmann = Math.Abs(v1 - vTransfer1) + Math.Abs(v2 - vTransfer2);
    
    // Plane change cost (can be combined with one of the burns)
    double dvPlane = vAvg * Math.Sin(phi);
    
    // Total cost (simplified - assumes plane change done at larger orbit)
    return dvHohmann + dvPlane;
}
```
