```csharp
List<(int x, int y)> FindPath(int startX, int startY, int endX, int endY, int mass, int[,] heightMap) {
    // A* with slope constraint
    var pq = new PriorityQueue<(int x, int y, List<(int,int)> path, int cost), int>();
    var visited = new HashSet<(int, int)>();
    
    pq.Enqueue((startX, startY, new List<(int,int)>(), 0), 0);
    
    while (pq.Count > 0) {
        var (x, y, path, cost) = pq.Dequeue();
        
        if (x == endX && y == endY) {
            return path;
        }
        
        if (visited.Contains((x, y))) continue;
        visited.Add((x, y));
        
        int[] dx = {0, 0, 1, -1};
        int[] dy = {1, -1, 0, 0};

            int nx = x + dx, ny = y + dy;
            if (nx < 0 || nx >= xCount || ny < 0 || ny >= yCount) continue;
            if (visited.Contains((nx, ny))) continue;
            
            int heightDiff = heightMap[nx, ny] - heightMap[x, y];
            if (Math.Abs(heightDiff) > 1) continue; // Slope too steep
            
            int moveCost;
            if (heightDiff > 0) moveCost = 5 * mass; // Uphill
            else if (heightDiff < 0) moveCost = 0;   // Downhill
            else moveCost = mass;                     // Flat
            
            var newPath = new List<(int,int)>(path) { (nx, ny) };
            int newCost = cost + moveCost;
            int heuristic = Math.Abs(nx - endX) + Math.Abs(ny - endY);
            pq.Enqueue((nx, ny, newPath, newCost), newCost + heuristic);
        }
    }
    
    return null; // No valid path
}
```

```csharp
        var pq = new PriorityQueue<(int x, int y, List<(int,int)> path, int cost), int>();
        var visited = new HashSet<(int, int)>();

                pq.Enqueue((nx, ny, newPath, newCost), newCost + heuristic * mass);
            }
        }
        
        return null;
```

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static int X, Y, Z;
    static byte[,,] world;
    static int[,] heightMap;
    static List<string> operations = new List<string>();
    
    static void Main()
    {
        ReadInput();
        Process();
        Output();
    }
    
    static void ReadInput()
    {
        var line = Console.ReadLine();
        while (string.IsNullOrWhiteSpace(line)) line = Console.ReadLine();
        var parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        
        X = int.Parse(parts[0]);
        Y = int.Parse(parts[1]);
        Z = int.Parse(parts[2]);
        
        // Use sparse representation for large worlds
        // For now, assume we can fit in memory
        world = new byte[X, Y, Z];
        
        using (var stream = Console.OpenStandardInput())
        {
            // Skip the newline after header
            // Actually, the header ends with newline, then binary data follows
            // The binary data starts immediately after the newline
            
            for (int x = 0; x < X; x++)
            {
                for (int y = 0; y < Y; y++)
                {
                    for (int z = 0; z < Z; z++)
                    {
                        int b = stream.ReadByte();
                        if (b < 0) b = 0; // EOF handling
                        world[x, y, z] = (byte)b;
                    }
                }
            }
        }
        
        BuildHeightMap();
    }
    
    static void BuildHeightMap()
    {
        heightMap = new int[X, Y];
        for (int x = 0; x < X; x++)
        {
            for (int y = 0; y < Y; y++)
            {
                heightMap[x, y] = -1;
                for (int z = Z - 1; z >= 0; z--)
                {
                    if (world[x, y, z] > 0)
                    {
                        heightMap[x, y] = z;
                        break;
                    }
                }
            }
        }
    }
    
    static void Process()
    {
        // Find all exposed voxels
        var exposed = new HashSet<(int, int, int)>();
        var exposedQueue = new Queue<(int, int, int)>();
        
        // Top surface
        for (int x = 0; x < X; x++)
        {
            for (int y = 0; y < Y; y++)
            {
                int h = heightMap[x, y];
                if (h >= 0 && world[x, y, h] > 0)
                {
                    exposed.Add((x, y, h));
                    exposedQueue.Enqueue((x, y, h));
                }
            }
        }
        
        // Side surfaces (cliffs)
        // For each position, check if adjacent positions have lower height
        for (int x = 0; x < X; x++)
        {
            for (int y = 0; y < Y; y++)
            {
                int h = heightMap[x, y];
                if (h < 0) continue;
                
                // Check each side
                int[][] dirs = { new[] { 1, 0 }, new[] { -1, 0 }, new[] { 0, 1 }, new[] { 0, -1 } };
                foreach (var dir in dirs)
                {
                    int nx = x + dir[0];
                    int ny = y + dir[1];
                    if (nx < 0 || nx >= X || ny < 0 || ny >= Y) continue;
                    
                    int nh = heightMap[nx, ny];
                    if (nh >= 0 && nh < h)
                    {
                        // Cliff: expose side voxels from nh+1 to h
                        for (int z = nh + 1; z <= h; z++)
                        {
                            if (world[x, y, z] > 0 && !exposed.Contains((x, y, z)))
                            {
                                exposed.Add((x, y, z));
                                exposedQueue.Enqueue((x, y, z));
                            }
                        }
                    }
                }
            }
        }
        
        // Process exposed voxels by priority
        var pq = new PriorityQueue<(int x, int y, int z), int>();
        foreach (var pos in exposed)
        {
            byte data = world[pos.Item1, pos.Item2, pos.Item3];
            int value = data / 16;
            pq.Enqueue(pos, -value); // Higher value = higher priority
        }
        
        while (pq.Count > 0)

            if (world[x, y, z] == 0) continue; // Already processed
            if (!IsExposed(x, y, z)) continue; // No longer exposed
            
            DigAndMove(x, y, z);
            
            // Check for newly exposed voxels
            AddNewlyExposed(x, y, z, pq);
        }
    }
    
    static bool IsExposed(int x, int y, int z)
    {
        // Check if exposed from above
        if (z >= Z - 1 || world[x, y, z + 1] == 0) return true;
        
        // Check if exposed from sides
        int[][] dirs = { new[] { 1, 0 }, new[] { -1, 0 }, new[] { 0, 1 }, new[] { 0, -1 } };
        foreach (var dir in dirs)
        {
            int nx = x + dir[0];
            int ny = y + dir[1];
            if (nx < 0 || nx >= X || ny < 0 || ny >= Y) return true; // Edge of world
            if (world[nx, ny, z] == 0) return true;
        }
        
        return false;

    static void AddNewlyExposed(int x, int y, int z, PriorityQueue<(int, int, int), int> pq)
    {

            int ny = y + dir[1];
            int nz = z + dir[2];

                int value = data / 16;
                pq.Enqueue((nx, ny, nz), -value);
            }
        }

    static void DigAndMove(int x, int y, int z)

        int mass = data % 16;
        int value = data / 16;
        
        int destX, destY;
        if (value > 0)

            destX = 0;
            destY = 0;

            destY = Y - 1;

            destY = Y / 2;

            if (path == null) return; // Skip
        }
        
        // Record operation
        operations.Add(FormatOperation(x, y, z, path, destX, destY));
        
        // Update world
        world[x, y, z] = 0;
        
        // Update height map
        if (z == heightMap[x, y])
        {
            for (int nz = z - 1; nz >= 0; nz--)
            {
                if (world[x, y, nz] > 0)
                {
                    heightMap[x, y] = nz;
                    break;
                }
                if (nz == 0) heightMap[x, y] = -1;
            }
        }
        
        // Update dump location height
        heightMap[destX, destY]++;
    }
    
    static List<(int, int)> FindPath(int startX, int startY, int endX, int endY, int mass)
    {
        if (startX == endX && startY == endY)
        {
            return new List<(int, int)>();
        }
        
        var prev = new Dictionary<(int, int), (int, int)>();
        var cost = new Dictionary<(int, int), int>();
        var pq = new PriorityQueue<(int, int, int), int>();
        
        cost[(startX, startY)] = 0;
        pq.Enqueue((startX, startY, GetHeight(startX, startY)), 0);
        
        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };
        
        while (pq.Count > 0)
        {
            var (x, y, h) = pq.Dequeue();
            
            if (x == endX && y == endY)
            {
                // Reconstruct path
                var path = new List<(int, int)>();
                var cur = (endX, endY);
                while (prev.ContainsKey(cur))
                {
                    path.Add(cur);
                    cur = prev[cur];
                }
                path.Reverse();
                return path;
            }
            
            if (cost.ContainsKey((x, y)) && cost[(x, y)] < h) continue;
            
            for (int i = 0; i < 4; i++)
            {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if (nx < 0 || nx >= X || ny < 0 || ny >= Y) continue;
                
                int hNext = GetHeight(nx, ny);
                int hDiff = hNext - h;
                
                if (Math.Abs(hDiff) > 1) continue;
                
                int moveCost;
                if (hDiff > 0) moveCost = 5 * mass;
                else if (hDiff < 0) moveCost = 0;
                else moveCost = mass;
                
                int newCost = cost[(x, y)] + moveCost;
                
                if (!cost.ContainsKey((nx, ny)) || newCost < cost[(nx, ny)])
                {
                    cost[(nx, ny)] = newCost;
                    prev[(nx, ny)] = (x, y);
                    int heuristic = Math.Abs(nx - endX) + Math.Abs(ny - endY);
                    pq.Enqueue((nx, ny, hNext), newCost + heuristic * mass);
                }
            }
        }
        
        return null;
    }
    
    static int GetHeight(int x, int y)
    {
        return Math.Max(0, heightMap[x, y] + 1); // Height is top of terrain
    }
    
    static string FormatOperation(int digX, int digY, int digZ, List<(int, int)> path, int destX, int destY)
    {
        var parts = new List<string>();
        parts.Add(path.Count.ToString());
        parts.Add(digX.ToString());
        parts.Add(digY.ToString());
        parts.Add(digZ.ToString());
        foreach (var (x, y) in path)
        {
            parts.Add(x.ToString());
            parts.Add(y.ToString());
        }
        parts.Add(destX.ToString());
        parts.Add(destY.ToString());
        return string.Join(" ", parts);
    }
    
    static void Output()
    {
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
```

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

class Program
{
    static int X, Y, Z;
    static Dictionary<(int, int, int), byte> worldSparse;
    static int[,] heightMap;
    static List<string> operations = new List<string>();
    
    static void Main()
    {
        Solve();
    }
    
    static void Solve()
    {
        ReadInput();
        
        // Choose strategy based on size
        long size = (long)X * Y * Z;
        if (size > 50_000_000)
        {
            SolveLarge();
        }
        else
        {
            SolveNormal();
        }
    }
    
    static void ReadInput()
    {
        var line = Console.ReadLine();
        while (string.IsNullOrWhiteSpace(line)) line = Console.ReadLine();
        var parts = line.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        
        X = int.Parse(parts[0]);
        Y = int.Parse(parts[1]);
        Z = int.Parse(parts[2]);
        
        heightMap = new int[X, Y];
        for (int i = 0; i < X; i++)
            for (int j = 0; j < Y; j++)
                heightMap[i, j] = -1;
        
        worldSparse = new Dictionary<(int, int, int), byte>();
        
        using (var stream = Console.OpenStandardInput())
        {
            for (int x = 0; x < X; x++)
            {
                for (int y = 0; y < Y; y++)
                {
                    int topZ = -1;
                    for (int z = 0; z < Z; z++)
                    {
                        int b = stream.ReadByte();
                        if (b < 0) b = 0;
                        if (b > 0)
                        {
                            worldSparse[(x, y, z)] = (byte)b;
                            topZ = z;
                        }
                    }
                    heightMap[x, y] = topZ;
                }
            }
        }
    }
    
    static void SolveNormal()
    {
        // Find valuable voxels
        var valuable = worldSparse.Where(kvp => kvp.Value / 16 > 0)
                                   .Select(kvp => (x: kvp.Key.Item1, y: kvp.Key.Item2, z: kvp.Key.Item3, 
                                                   mass: kvp.Value % 16, value: kvp.Value / 16))
                                   .OrderByDescending(v => v.value * v.mass) // High value first
                                   .ToList();
        
        // For each valuable, try to mine it
        foreach (var v in valuable)
        {
            TryMine(v.x, v.y, v.z);
        }
        
        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
    
    static void SolveLarge()
    {
        // For large grids, only process high-value ore near the plant
        var valuable = worldSparse.Where(kvp => kvp.Value / 16 > 0)
                                   .Select(kvp => (x: kvp.Key.Item1, y: kvp.Key.Item2, z: kvp.Key.Item3,
                                                   mass: kvp.Value % 16, value: kvp.Value / 16))
                                   .Where(v => v.x + v.y < X + Y / 4) // Within 1/4 of max distance
                                   .OrderByDescending(v => v.value * v.mass * 1000 - (v.x + v.y) * v.mass) // Profit estimate
                                   .Take(100000) // Limit operations
                                   .ToList();
        
        foreach (var v in valuable)
        {
            TryMine(v.x, v.y, v.z);
        }
        
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.WriteLine(op);
        }
    }
    
    static void TryMine(int x, int y, int z)
    {
        if (!worldSparse.ContainsKey((x, y, z))) return;
        if (!IsExposed(x, y, z)) return;
        
        byte data = worldSparse[(x, y, z)];
        int mass = data % 16;
        int value = data / 16;
        
        int destX = value > 0 ? 0 : X - 1;
        int destY = value > 0 ? 0 : Y - 1;
        
        var path = FindPath(x, y, destX, destY, mass);
        if (path == null || path.Count == 0 && (x != destX || y != destY))
        {
            // Try alternate dump
            if (value == 0)
            {
                destX = X / 2;
                destY = Y / 2;
                path = FindPath(x, y, destX, destY, mass);
            }
            if (path == null) return;
        }
        
        // Execute operation
        operations.Add(FormatOperation(x, y, z, path, destX, destY));
        
        // Update world
        worldSparse.Remove((x, y, z));
        
        // Update height map
        if (heightMap[x, y] == z)
        {
            int newH = -1;
            for (int nz = z - 1; nz >= 0; nz--)
            {
                if (worldSparse.ContainsKey((x, y, nz)))
                {
                    newH = nz;
                    break;
                }
            }
            heightMap[x, y] = newH;
        }
        
        // Raise dump location
        heightMap[destX, destY]++;
    }
    
    static bool IsExposed(int x, int y, int z)
    {
        // Exposed from above?
        if (!worldSparse.ContainsKey((x, y, z + 1))) return true;
        
        // Exposed from sides?
        int[][] dirs = { new[] { 1, 0, 0 }, new[] { -1, 0, 0 }, new[] { 0, 1, 0 }, new[] { 0, -1, 0 } };
        foreach (var dir in dirs)
        {
            int nx = x + dir[0];
            int ny = y + dir[1];
            if (nx < 0 || nx >= X || ny < 0 || ny >= Y) return true;
            if (!worldSparse.ContainsKey((nx, ny, z))) return true;
        }
        
        return false;
    }
    
    static List<(int, int)> FindPath(int startX, int startY, int endX, int endY, int mass)
    {
        if (startX == endX && startY == endY)
        {
            return new List<(int, int)> { (endX, endY) };
        }
        
        var prev = new Dictionary<(int, int), (int, int)>();
        var cost = new Dictionary<(int, int), int>();
        var pq = new PriorityQueue<(int, int), int>();
        
        int startH = GetSurfaceHeight(startX, startY);
        cost[(startX, startY)] = 0;
        pq.Enqueue((startX, startY), 0);
        
        int[] dx = { 0, 0, 1, -1 };
        int[] dy = { 1, -1, 0, 0 };
        
        while (pq.Count > 0)
        {
            var (x, y) = pq.Dequeue();
            int h = GetSurfaceHeight(x, y);
            
            if (x == endX && y == endY)
            {
                var path = new List<(int, int)>();
                var cur = (endX, endY);
                while (prev.ContainsKey(cur))
                {
                    path.Add(cur);
                    cur = prev[cur];
                }
                path.Reverse();
                return path;
            }
            
            int curCost = cost[(x, y)];
            
            for (int i = 0; i < 4; i++)
            {
                int nx = x + dx[i];
                int ny = y + dy[i];
                
                if (nx < 0 || nx >= X || ny < 0 || ny >= Y) continue;
                
                int hNext = GetSurfaceHeight(nx, ny);
                int hDiff = hNext - h;
                
                if (Math.Abs(hDiff) > 1) continue;
                
                int moveCost;
                if (hDiff > 0) moveCost = 5 * mass;
                else if (hDiff < 0) moveCost = 0;
                else moveCost = mass;
                
                int newCost = curCost + moveCost;
                
                if (!cost.ContainsKey((nx, ny)) || newCost < cost[(nx, ny)])
                {
                    cost[(nx, ny)] = newCost;
                    prev[(nx, ny)] = (x, y);
                    int heuristic = Math.Abs(nx - endX) + Math.Abs(ny - endY);
                    pq.Enqueue((nx, ny), newCost + heuristic);
                }
            }
        }
        
        return null;
    }
    
    static int GetSurfaceHeight(int x, int y)
    {
        return heightMap[x, y] + 1; // Surface is above the top voxel
    }
    
    static string FormatOperation(int digX, int digY, int digZ, List<(int, int)> path, int destX, int destY)
    {
        var parts = new List<string>();
        parts.Add(path.Count.ToString());
        parts.Add(digX.ToString());
        parts.Add(digY.ToString());
        parts.Add(digZ.ToString());
        foreach (var (x, y) in path)
        {
            parts.Add(x.ToString());
            parts.Add(y.ToString());
        }
        parts.Add(destX.ToString());
        parts.Add(destY.ToString());
        return string.Join(" ", parts);
    }
```
