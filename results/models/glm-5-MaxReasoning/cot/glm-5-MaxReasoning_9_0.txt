using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        // Fast input
        using var sr = new StreamReader(Console.OpenStandardInput());
        var line1 = sr.ReadLine();
        var sp = line1.IndexOf(' ');
        int n = int.Parse(line1.AsSpan(0, sp));
        int stockLen = int.Parse(line1.AsSpan(sp + 1));
        
        var cuts = new int[n];
        ParseInts(sr.ReadLine(), cuts);
        
        // Sort indices by cut length descending
        var indices = new int[n];
        for (int i = 0; i < n; i++) indices[i] = i;
        Array.Sort(indices, (a, b) => cuts[b].CompareTo(cuts[a]));
        
        // BFD with segment tree
        var stocks = new List<List<int>>();
        var remaining = new List<int>();
        var remainingMap = new Dictionary<int, Stack<int>>();
        
        // Segment tree to find first remaining >= threshold
        int treeSize = 1;
        while (treeSize <= stockLen) treeSize <<= 1;
        var segtree = new int[2 * treeSize];
        
        for (int i = 0; i < n; i++)
        {
            int cutIdx = indices[i];
            int cutLen = cuts[cutIdx];
            
            // Query segment tree for first remaining >= cutLen
            int bestRemaining = Query(segtree, treeSize, cutLen, stockLen);
            
            if (bestRemaining >= cutLen)
            {
                // Get a stock with this remaining
                var stack = remainingMap[bestRemaining];
                int stockIdx = stack.Pop();
                if (stack.Count == 0) remainingMap.Remove(bestRemaining);
                
                // Update segment tree
                Update(segtree, treeSize, bestRemaining, -1);
                
                // Place cut
                stocks[stockIdx].Add(cutIdx);
                int newRem = remaining[stockIdx] - cutLen;
                remaining[stockIdx] = newRem;
                
                // Add back to map and tree
                if (!remainingMap.ContainsKey(newRem))
                    remainingMap[newRem] = new Stack<int>();
                remainingMap[newRem].Push(stockIdx);
                Update(segtree, treeSize, newRem, 1);
            }
            else
            {
                // Create new stock
                int stockIdx = stocks.Count;
                stocks.Add(new List<int> { cutIdx });
                remaining.Add(stockLen - cutLen);
                
                int newRem = stockLen - cutLen;
                if (!remainingMap.ContainsKey(newRem))
                    remainingMap[newRem] = new Stack<int>();
                remainingMap[newRem].Push(stockIdx);
                Update(segtree, treeSize, newRem, 1);
            }
        }
        
        // Output
        Console.WriteLine(stocks.Count);
        var sb = new StringBuilder();
        foreach (var stock in stocks)
        {
            sb.Clear();
            for (int j = 0; j < stock.Count; j++)
            {
                if (j > 0) sb.Append(' ');
                sb.Append(stock[j]);
            }
            Console.WriteLine(sb);

    {

            lo = mid + 1;
        }
    }
