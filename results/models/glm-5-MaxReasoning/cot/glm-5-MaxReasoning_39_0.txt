fn bb(

    target: usize,
    in_a: &mut [bool],
    in_b: &mut [bool],
    a_count: &mut usize,
    b_count: &mut usize,
    current_cut: &mut usize,
    best_cut: &mut usize,
    best_part: &mut Vec<usize>,
    vertex: usize,
) {

        // All vertices assigned
        if *a_count == target {
            if *current_cut < *best_cut {
                *best_cut = *current_cut;
                *best_part.clear();
                for i in 0..n {
                    if in_a[i] {

fn kernighan_lin(adj: &[Vec<usize>], n: usize) -> (usize, Vec<usize>) {
    let half = n / 2;
    
    // Initial partition: first half in A
    let mut in_a: Vec<bool> = (0..n).map(|i| i < half).collect();
    
    // Compute initial D values (external - internal cost)
    // D[v] = E(v, other partition) - E(v, same partition)
    let mut d = vec![0i64; n];
    for v in 0..n {
        for &u in &adj[v] {
            if in_a[v] == in_a[u] {
                d[v] -= 1;
            } else {
                d[v] += 1;
            }
        }
    }
    
    let mut best_cut = compute_cut(adj, &in_a);
    let mut best_partition = in_a.clone();
    
    // Iterate KL passes
    for _ in 0..100 {
        let mut locked = vec![false; n];
        let mut gains = Vec::new();
        let mut local_d = d.clone();
        let mut local_in_a = in_a.clone();
        
        for _ in 0..half {
            // Find best swap pair (a in A, b in B)
            let mut best_gain = i64::MIN;
            let mut best_a = None;
            let mut best_b = None;
            
            for a in 0..n {
                if !locked[a] && local_in_a[a] {
                    for b in 0..n {
                        if !locked[b] && !local_in_a[b] {
                            // Gain of swapping a and b
                            let gain = local_d[a] + local_d[b] - 2 * edge_weight(adj, a, b);
                            if gain > best_gain {
                                best_gain = gain;
                                best_a = Some(a);
                                best_b = Some(b);
                            }
                        }
                    }
                }
            }
            
            if let (Some(a), Some(b)) = (best_a, best_b) {
                // Lock and swap
                locked[a] = true;
                locked[b] = true;
                gains.push((best_gain, a, b));
                
                // Update D values for unlocked neighbors
                for &u in &adj[a] {
                    if !locked[u] {
                        if local_in_a[u] {
                            local_d[u] += 2 * edge_weight(adj, a, u);
                        } else {
                            local_d[u] -= 2 * edge_weight(adj, a, u);
                        }
                    }
                }
                for &u in &adj[b] {
                    if !locked[u] {
                        if !local_in_a[u] {
                            local_d[u] += 2 * edge_weight(adj, b, u);
                        } else {
                            local_d[u] -= 2 * edge_weight(adj, b, u);
                        }
                    }
                }
                
                // Swap
                local_in_a[a] = false;
                local_in_a[b] = true;
            }
        }
        
        // Find best prefix of swaps
        let mut cum_gain = 0i64;
        let mut best_k = 0;
        let mut best_cum = 0i64;
        
        for (k, (g, _, _)) in gains.iter().enumerate() {
            cum_gain += *g;
            if cum_gain > best_cum {
                best_cum = cum_gain;
                best_k = k + 1;
            }
        }
        
        if best_cum > 0 {
            // Apply first best_k swaps
            for (_, a, b) in gains.iter().take(best_k) {
                in_a[*a] = false;
                in_a[*b] = true;
            }
            
            // Recompute D
            for v in 0..n {
                d[v] = 0;
                for &u in &adj[v] {
                    if in_a[v] == in_a[u] {
                        d[v] -= 1;
                    } else {
                        d[v] += 1;
                    }
                }
            }
            
            let cut = compute_cut(adj, &in_a);
            if cut < best_cut {
                best_cut = cut;
                best_partition = in_a.clone();
            }
        } else {
            break; // No improvement
        }
    }
    
    let partition: Vec<usize> = (0..n).filter(|&i| best_partition[i]).collect();
    (best_cut, partition)
}

fn solve(adj: &[Vec<usize>], n: usize) -> (usize, Vec<usize>) {
    let half = n / 2;
    
    // Sort vertices by degree (branch on high degree first)
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));
    
    // Get initial solution using Kernighan-Lin
    let (init_cut, init_part) = kernighan_lin(adj, n);
    
    let mut solver = Solver {
        n,
        half,
        adj,
        order,
        in_a: vec![false; n],
        in_b: vec![false; n],
        a_count: 0,
        b_count: 0,
        current_cut: 0,
        best_cut: init_cut,
        best_part: init_part,
    };
    
    solver.search(0);
    
    (solver.best_cut, solver.best_part)
}

struct Solver<'a> {
    n: usize,
    half: usize,
    adj: &'a [Vec<usize>],
    order: Vec<usize>,
    in_a: Vec<bool>,
    in_b: Vec<bool>,
    a_count: usize,
    b_count: usize,
    current_cut: usize,
    best_cut: usize,
    best_part: Vec<usize>,
}

impl<'a> Solver<'a> {
    fn search(&mut self, idx: usize) {
        if self.current_cut >= self.best_cut {
            return;
        }
        
        if idx == self.n {
            if self.a_count == self.half && self.current_cut < self.best_cut {
                self.best_cut = self.current_cut;
                self.best_part.clear();
                for i in 0..self.n {
                    if self.in_a[i] {
                        self.best_part.push(i);
                    }
                }
            }
            return;
        }
        
        let v = self.order[idx];
        
        // Compute lower bound
        let lb = self.lower_bound();
        if lb >= self.best_cut {
            return;
        }
        
        // Try assigning to A
        if self.a_count < self.half {
            let added = self.compute_added_cut(v, true);
            self.in_a[v] = true;
            self.a_count += 1;
            self.current_cut += added;
            
            self.search(idx + 1);
            
            self.current_cut -= added;
            self.a_count -= 1;
            self.in_a[v] = false;
        }
        
        // Try assigning to B
        if self.b_count < self.half {
            let added = self.compute_added_cut(v, false);
            self.in_b[v] = true;
            self.b_count += 1;
            self.current_cut += added;
            
            self.search(idx + 1);
            
            self.current_cut -= added;
            self.b_count -= 1;
            self.in_b[v] = false;
        }
    }
    
    fn compute_added_cut(&self, v: usize, to_a: bool) -> usize {
        let mut added = 0;
        for &u in &self.adj[v] {
            if to_a && self.in_b[u] {
                added += 1;
            } else if !to_a && self.in_a[u] {
                added += 1;
            }
        }
        added
    }
    
    fn lower_bound(&self) -> usize {
        let mut lb = self.current_cut;
        
        for v in 0..self.n {
            if !self.in_a[v] && !self.in_b[v] {
                let mut to_a = 0usize;
                let mut to_b = 0usize;
                for &u in &self.adj[v] {
                    if self.in_a[u] {
                        to_a += 1;
                    } else if self.in_b[u] {
                        to_b += 1;
                    }
                }
                lb += min(to_a, to_b);
            }
        }
        
        lb
    }
}

fn kernighan_lin(adj: &[Vec<usize>], n: usize) -> (usize, Vec<usize>) {
    let half = n / 2;
    
    // Multiple random starts
    let mut best_cut = usize::MAX;
    let mut best_partition = (0..half).collect::<Vec<_>>();
    
    for trial in 0..20 {
        let init_part = if trial == 0 {
            (0..half).collect::<Vec<_>>()
        } else {
            // Random partition
            let mut perm: Vec<usize> = (0..n).collect();
            // Simple shuffle using trial as seed
            for i in 0..n {
                let j = (i * 7 + trial * 13) % n;
                perm.swap(i, j);
            }
            perm.into_iter().take(half).collect()
        };
        
        let (cut, part) = kl_pass(adj, n, init_part);
        if cut < best_cut {
            best_cut = cut;
            best_partition = part;
        }
    }
    
    (best_cut, best_partition)
}

fn kl_pass(adj: &[Vec<usize>], n: usize, init_part: Vec<usize>) -> (usize, Vec<usize>) {
    let half = n / 2;
    
    let mut in_a = vec![false; n];
    for &v in &init_part {
        in_a[v] = true;
    }
    
    // Compute D values
    let mut d = vec![0i32; n];
    for v in 0..n {
        for &u in &adj[v] {
            if in_a[v] == in_a[u] {
                d[v] -= 1;
            } else {
                d[v] += 1;
            }
        }
    }
    
    let mut best_cut = compute_cut_from_in_a(adj, &in_a);
    let mut best_in_a = in_a.clone();
    
    for _ in 0..50 {
        let mut locked = vec![false; n];
        let mut local_d = d.clone();
        let mut local_in_a = in_a.clone();
        let mut swaps: Vec<(i32, usize, usize)> = Vec::new();
        
        for _ in 0..half {
            // Find best swap
            let mut best_gain = i32::MIN;
            let mut best_pair = (0, 0);
            
            // Get lists of unlocked vertices in A and B
            let a_verts: Vec<usize> = (0..n).filter(|&v| local_in_a[v] && !locked[v]).collect();
            let b_verts: Vec<usize> = (0..n).filter(|&v| !local_in_a[v] && !locked[v]).collect();
            
            for &a in &a_verts {
                for &b in &b_verts {
                    let edge = if adj[a].contains(&b) { 2 } else { 0 };
                    let gain = local_d[a] + local_d[b] - edge;
                    if gain > best_gain {
                        best_gain = gain;
                        best_pair = (a, b);
                    }
                }
            }
            
            let (a, b) = best_pair;
            
            // Lock and record
            locked[a] = true;
            locked[b] = true;
            swaps.push((best_gain, a, b));
            
            // Update D for neighbors
            for &u in &adj[a] {
                if !locked[u] {
                    let delta = if adj[a].contains(&u) { 2 } else { 0 };
                    if local_in_a[u] {
                        local_d[u] += delta;
                    } else {
                        local_d[u] -= delta;
                    }
                }
            }
            for &u in &adj[b] {
                if !locked[u] {
                    let delta = if adj[b].contains(&u) { 2 } else { 0 };
                    if local_in_a[u] {
                        local_d[u] -= delta;
                    } else {
                        local_d[u] += delta;
                    }
                }
            }
            
            // Perform swap locally
            local_in_a[a] = false;
            local_in_a[b] = true;
        }
        
        // Find best k for first k swaps
        let mut cum_gain = 0i32;
        let mut best_k = 0;
        let mut best_cum = 0i32;
        
        for (k, &(g, _, _)) in swaps.iter().enumerate() {
            cum_gain += g;
            if cum_gain > best_cum {
                best_cum = cum_gain;
                best_k = k + 1;
            }
        }
        
        if best_cum > 0 {
            // Apply swaps
            for &(_, a, b) in &swaps[..best_k] {
                in_a[a] = false;
                in_a[b] = true;
            }
            
            // Recompute D
            for v in 0..n {
                d[v] = 0;
                for &u in &adj[v] {
                    if in_a[v] == in_a[u] {
                        d[v] -= 1;
                    } else {
                        d[v] += 1;
                    }
                }
            }
            
            let cut = compute_cut_from_in_a(adj, &in_a);
            if cut < best_cut {
                best_cut = cut;
                best_in_a = in_a.clone();
            }
        } else {
            break;
        }
    }
    
    let partition: Vec<usize> = (0..n).filter(|&i| best_in_a[i]).collect();
    (best_cut, partition)
}

fn compute_cut_from_in_a(adj: &[Vec<usize>], in_a: &[bool]) -> usize {

        if in_a[u] {
            for &v in neighbors {
                if !in_a[v] {
                    cut += 1;
                }
            }
        }
    }
    cut
}

When we swap a and b:

struct BBState<'a> {
    n: usize,
    half: usize,
    adj: &'a [Vec<usize>],
    in_a: Vec<bool>,
    in_b: Vec<bool>,
    a_count: usize,
    b_count: usize,
    cut: usize,
    best_cut: usize,
    best_part: Vec<usize>,
}

impl<'a> BBState<'a> {
    fn search(&mut self, v: usize) {
        if self.cut >= self.best_cut {
            return;
        }
        
        if v == self.n {
            if self.a_count == self.half && self.cut < self.best_cut {
                self.best_cut = self.cut;
                self.best_part = (0..self.n).filter(|&i| self.in_a[i]).collect();
            }
            return;
        }
        
        // Prune with lower bound
        if self.lb() >= self.best_cut {
            return;
        }
        
        // Try A
        if self.a_count < self.half {
            let delta = self.delta(v, true);
            self.in_a[v] = true;
            self.a_count += 1;
            self.cut += delta;
            self.search(v + 1);
            self.cut -= delta;
            self.a_count -= 1;
            self.in_a[v] = false;
        }
        
        // Try B
        if self.b_count < self.half {
            let delta = self.delta(v, false);
            self.in_b[v] = true;
            self.b_count += 1;
            self.cut += delta;
            self.search(v + 1);
            self.cut -= delta;
            self.b_count -= 1;
            self.in_b[v] = false;
        }
    }
    
    fn delta(&self, v: usize, to_a: bool) -> usize {
        self.adj[v].iter().filter(|&&u| {
            if to_a { self.in_b[u] } else { self.in_a[u] }
        }).count()
    }
    
    fn lb(&self) -> usize {
        let mut lb = self.cut;
        for v in 0..self.n {
            if !self.in_a[v] && !self.in_b[v] {
                let to_a = self.adj[v].iter().filter(|&&u| self.in_a[u]).count();
                let to_b = self.adj[v].iter().filter(|&&u| self.in_b[u]).count();
                lb += to_a.min(to_b);
            }
        }
        lb
    }
}

        let mut gains: Vec<(i64, usize, usize)> = Vec::new();
        
        for _ in 0..half {
            // Find best pair
            let mut best = (i64::MIN, 0, 0);

            local_in_a[b] = true;

                                best = (gain, a, b);
                            }
                        }
                    }
                }
            }
            
            if best.0 == i64::MIN {
                break;
            }
            
            let (_, a, b) = best;
            locked[a] = true;
            locked[b] = true;
            swaps.push(best);
            
            // Update D values
            for &u in &adj[a] {
                if !locked[u] {
                    if local_in_a[u] {
                        d[u] += 2;
                    } else {
                        d[u] -= 2;
                    }
                }
            }
            for &u in &adj[b] {
                if !locked[u] {
                    if local_in_a[u] {
                        d[u] -= 2;
                    } else {
                        d[u] += 2;
                    }
                }
            }
            
            // Perform swap
            local_in_a[a] = false;
            local_in_a[b] = true;
            
            // Swap D values too
            std::mem::swap(&mut d[a], &mut d[b]);
            d[a] = -d[a];
            d[b] = -d[b];
        }
        
        // Find best k
        let mut cum = 0i64;
        let mut best_k = 0;
        let mut best_cum = 0i64;
        
        for (k, &(g, _, _)) in swaps.iter().enumerate() {
            cum += g;
            if cum > best_cum {
                best_cum = cum;
                best_k = k + 1;
            }
        }
        
        if best_cum > 0 {
            for &(_, a, b) in &swaps[..best_k] {
                in_a[a] = false;
                in_a[b] = true;
            }
            
            let cut = cut_size(adj, &in_a);
            if cut < best_cut {
                best_cut = cut;
                best_in_a = in_a.clone();
            }
        } else {
            break;
        }
    }
    
    let part: Vec<usize> = (0..n).filter(|&i| best_in_a[i]).collect();
    (best_cut, part)
}

fn cut_size(adj: &[Vec<usize>], in_a: &[bool]) -> usize {
    let mut cut = 0;
    for (u, neighbors) in adj.iter().enumerate() {
        if in_a[u] {
            for &v in neighbors {
                if !in_a[v] {
                    cut += 1;
                }
            }
        }
    }
    cut
}

struct BBState<'a> {
    n: usize,
    half: usize,
    adj: &'a [Vec<usize>],
    in_a: Vec<bool>,
    in_b: Vec<bool>,
    a_count: usize,
    b_count: usize,
    cut: usize,
    best_cut: usize,
    best_part: Vec<usize>,
}

impl<'a> BBState<'a> {
    fn new(n: usize, half: usize, adj: &'a [Vec<usize>], best_cut: usize, best_part: Vec<usize>) -> Self {
        Self {
            n,
            half,
            adj,
            in_a: vec![false; n],
            in_b: vec![false; n],
            a_count: 0,
            b_count: 0,
            cut: 0,
            best_cut,
            best_part,
        }
    }
    
    fn search(&mut self, v: usize) {
        if self.cut >= self.best_cut {
            return;
        }
        
        if v == self.n {
            if self.a_count == self.half && self.cut < self.best_cut {
                self.best_cut = self.cut;
                self.best_part = (0..self.n).filter(|&i| self.in_a[i]).collect();
            }
            return;
        }
        
        // Lower bound pruning
        if self.lower_bound() >= self.best_cut {
            return;
        }
        
        // Try A first (often better due to symmetry)
        if self.a_count < self.half {
            let delta = self.compute_delta(v, true);
            self.in_a[v] = true;
            self.a_count += 1;
            self.cut += delta;
            self.search(v + 1);
            self.cut -= delta;
            self.a_count -= 1;
            self.in_a[v] = false;
        }
        
        // Try B
        if self.b_count < self.half {
            let delta = self.compute_delta(v, false);
            self.in_b[v] = true;
            self.b_count += 1;
            self.cut += delta;
            self.search(v + 1);
            self.cut -= delta;
            self.b_count -= 1;
            self.in_b[v] = false;
        }
    }
    
    fn compute_delta(&self, v: usize, to_a: bool) -> usize {
        self.adj[v].iter().filter(|&&u| {
            if to_a { self.in_b[u] } else { self.in_a[u] }
        }).count()
    }
    
    fn lower_bound(&self) -> usize {
        let mut lb = self.cut;
        
        // Count unassigned vertices and their contribution
        let remaining = self.n - self.a_count - self.b_count;
        let a_slots = self.half - self.a_count;
        let b_slots = self.half - self.b_count;
        
        if remaining == 0 {
            return lb;
        }
        
        // Simple bound: each unassigned vertex contributes at least
        // min(edges to assigned A, edges to assigned B)
        for v in 0..self.n {
            if !self.in_a[v] && !self.in_b[v] {
                let to_a = self.adj[v].iter().filter(|&&u| self.in_a[u]).count();
                let to_b = self.adj[v].iter().filter(|&&u| self.in_b[u]).count();
                lb += to_a.min(to_b);
            }
        }
        
        lb
    }
}

fn solve(adj: &[Vec<usize>], n: usize) -> (usize, Vec<usize>) {
    let half = n / 2;
    
    if n == 0 {
        return (0, vec![]);
    }
    
    // Get upper bound from KL heuristic
    let (ub, init_part) = kl_heuristic(adj, n);
    
    // Branch and bound for optimal solution
    let mut state = BBState::new(n, half, adj, ub, init_part);
    state.search(0);
    
    (state.best_cut, state.best_part)
}

fn kl_heuristic(adj: &[Vec<usize>], n: usize) -> (usize, Vec<usize>) {
    let half = n / 2;
    let mut best_cut = usize::MAX;
    let mut best_part: Vec<usize> = (0..half).collect();
    
    for trial in 0..30 {
        let init = create_initial_partition(adj, n, trial);
        let (cut, part) = kl_run(adj, n, init);
        if cut < best_cut {
            best_cut = cut;
            best_part = part;
        }
    }
    
    (best_cut, best_part)
}

fn create_initial_partition(adj: &[Vec<usize>], n: usize, trial: usize) -> Vec<usize> {

    if trial == 0 {
        return (0..half).collect();
    }
    
    if trial == 1 {
        // By degree
        let mut indexed: Vec<(usize, usize)> = (0..n).map(|v| (v, adj[v].len())).collect();
        indexed.sort_unstable_by_key(|&(_, d)| std::cmp::Reverse(d));
        return indexed.into_iter().take(half).map(|(v, _)| v).collect();
    }
    
    // Pseudo-random
    let mut perm: Vec<usize> = (0..n).collect();
    let seed = trial as u64;
    for i in 0..n {
        let j = (i as u64).wrapping_mul(6364136223846793005)
            .wrapping_add(seed.wrapping_mul(1442695040888963407)) as usize % n;
        perm.swap(i, j);
