struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // 1. Normalize normal
    float3 N = normalize(input.normal);

    // 2. Calculate stripes
    // sin(uv.y * 3.14159 * 16.0)
    float sineVal = sin(input.uv.y * 3.14159 * 16.0);
    
    // Alternating Red (0.9, 0.1, 0.1) and Blue (0.1, 0.1, 0.9)
    // Let's assume sin > 0 is Red.
    float t = step(0.0, sineVal); // 1.0 if positive, 0.0 if negative
    
    float3 red = float3(0.9, 0.1, 0.1);
    float3 blue = float3(0.1, 0.1, 0.9);
    
    float3 baseColor = lerp(blue, red, t);

    // 3. Lambertian shading
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float NdotL = max(0.0, dot(N, L));
    
    // Factor: 0.4 + 0.6 * diffuse
    float intensity = 0.4 + 0.6 * NdotL;

    // 4. Combine
    float3 finalColor = baseColor * intensity;

    return float4(finalColor, 1.0);
}
