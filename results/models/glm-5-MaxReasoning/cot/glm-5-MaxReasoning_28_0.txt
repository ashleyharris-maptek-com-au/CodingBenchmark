use std::collections::VecDeque;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();
    
    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    
    // Remove duplicates
    for neighbors in &mut adj {
        neighbors.sort_unstable();
        neighbors.dedup();
    }
    
    let mut in_cover: Vec<bool> = vec![false; n];
    let mut removed: Vec<bool> = vec![false; n];
    let mut degree: Vec<usize> = adj.iter().map(|v| v.len()).collect();
    
    // Kernelization
    let mut queue: VecDeque<usize> = VecDeque::new();
    for i in 0..n {
        if degree[i] == 0 {
            removed[i] = true;
        } else if degree[i] == 1 {
            queue.push_back(i);
        }
    }
    
    while let Some(u) = queue.pop_front() {
        if removed[u] {
            continue;
        }
        if let Some(v) = adj[u].iter().cloned().find(|v| !removed[*v]) {
            in_cover[v] = true;
            removed[u] = true;
            removed[v] = true;
            for &w in &adj[v] {
                if !removed[w] {
                    degree[w] -= 1;
                    if degree[w] == 0 {
                        removed[w] = true;
                    } else if degree[w] == 1 {
                        queue.push_back(w);
                    }
                }
            }
        }
    }
    
    // Maximal matching
    let mut matched: Vec<bool> = vec![false; n];
    for u in 0..n {
        if removed[u] || matched[u] {
            continue;
        }
        for &v in &adj[u] {
            if !removed[v] && !matched[v] {
                matched[u] = true;
                matched[v] = true;
                in_cover[u] = true;
                in_cover[v] = true;
                break;
            }
        }
    }
    
    // Remove redundant vertices
    let mut to_check: VecDeque<usize> = (0..n).filter(|&i| in_cover[i]).collect();

            in_cover[u] = false;
            for &v in &adj[u] {
                if in_cover[v] {
                    to_check.push_back(v);
                }
            }
        }
    }
