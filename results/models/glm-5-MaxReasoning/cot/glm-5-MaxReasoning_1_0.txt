uint64_t hilbert_distance(double x, double y, double min_x, double min_y, double max_x, double max_y, int bits = 32) {

    uint32_t xi = (uint32_t)((x - min_x) / (max_x - min_x + 1e-12) * ((1ULL << bits) - 1));
    uint32_t yi = (uint32_t)((y - min_y) / (max_y - min_y + 1e-12) * ((1ULL << bits) - 1));

    uint64_t d = 0;

                yi = (~yi) & ((1 << bits) - 1);

#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <cstdint>

using namespace std;

typedef pair<double, double> Point;

static vector<Point> cities;
static int N;

inline double dist2(int i, int j) {
    double dx = cities[i].first - cities[j].first;
    double dy = cities[i].second - cities[j].second;
    return dx * dx + dy * dy;
}

inline double dist(int i, int j) {
    return sqrt(dist2(i, j));
}

vector<int> held_karp() {
    const double INF = 1e30;
    int size = 1 << N;
    vector<double> dp(size * N, INF);
    vector<int> parent(size * N, -1);
    
    dp[N] = 0;
    
    for (int mask = 1; mask < size; mask++) {
        for (int last = 0; last < N; last++) {
            if (!(mask & (1 << last))) continue;
            int idx = mask * N + last;
            double cur = dp[idx];
            if (cur >= INF) continue;
            
            for (int next = 0; next < N; next++) {
                if (mask & (1 << next)) continue;
                int new_mask = mask | (1 << next);
                double new_dist = cur + dist(last, next);
                int new_idx = new_mask * N + next;
                if (new_dist < dp[new_idx]) {
                    dp[new_idx] = new_dist;
                    parent[new_idx] = last;
                }
            }
        }
    }
    
    int full = size - 1;
    int best_last = 1;
    double best_dist = dp[full * N + 1];
    for (int i = 2; i < N; i++) {
        double d = dp[full * N + i];
        if (d < best_dist) {
            best_dist = d;
            best_last = i;
        }
    }
    
    vector<int> tour;
    int mask = full, last = best_last;
    while (last != -1) {
        tour.push_back(last);
        int prev = parent[mask * N + last];
        mask ^= (1 << last);
        last = prev;
    }
    reverse(tour.begin(), tour.end());
    
    return tour;
}

vector<int> nearest_neighbor() {
    vector<int> tour;
    vector<char> visited(N, 0);
    
    tour.reserve(N);
    tour.push_back(0);
    visited[0] = 1;
    
    for (int i = 1; i < N; i++) {
        int last = tour.back();
        double best_d = 1e30;
        int best = -1;
        
        for (int j = 0; j < N; j++) {
            if (visited[j]) continue;
            double d = dist2(last, j);
            if (d < best_d) {
                best_d = d;
                best = j;
            }
        }
        
        tour.push_back(best);
        visited[best] = 1;
    }
    
    return tour;
}

void two_opt(vector<int>& tour) {
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 2; i++) {
            for (int j = i + 2; j < N; j++) {
                int a = tour[i], b = tour[i + 1];
                int c = tour[j], d = tour[(j + 1) % N];
                
                double gain = dist2(a, b) + dist2(c, d) - dist2(a, c) - dist2(b, d);
                if (gain > 1e-9) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
        }
    }
}

void fast_two_opt(vector<int>& tour, int radius) {
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 2; i++) {
            int end_j = min(N, i + radius);
            for (int j = i + 2; j < end_j; j++) {
                int a = tour[i], b = tour[i + 1];
                int c = tour[j], d = tour[(j + 1) % N];

uint64_t hilbert_encode(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (uint32_t s = 1U << 31; s; s >>= 1) {
        uint32_t rx = (x & s) != 0;
        uint32_t ry = (y & s) != 0;
        d += (uint64_t)s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx) {
                x = ~x;
                y = ~y;
            }
            swap(x, y);
        }
    }
    return d;
}

vector<int> hilbert_tour() {
    double min_x = cities[0].first, max_x = cities[0].first;
    double min_y = cities[0].second, max_y = cities[0].second;
    
    for (int i = 1; i < N; i++) {
        if (cities[i].first < min_x) min_x = cities[i].first;
        if (cities[i].first > max_x) max_x = cities[i].first;
        if (cities[i].second < min_y) min_y = cities[i].second;
        if (cities[i].second > max_y) max_y = cities[i].second;
    }
    
    double range_x = max(max_x - min_x, 1e-12);
    double range_y = max(max_y - min_y, 1e-12);
    
    vector<pair<uint64_t, int>> order(N);
    for (int i = 0; i < N; i++) {
        uint32_t xi = (uint32_t)((cities[i].first - min_x) / range_x * 0xFFFFFFFF);
        uint32_t yi = (uint32_t)((cities[i].second - min_y) / range_y * 0xFFFFFFFF);
        order[i] = {hilbert_encode(xi, yi), i};
    }
    
    sort(order.begin(), order.end());
    
    vector<int> tour(N);
    int start = 0;
    for (int i = 0; i < N; i++) {
        tour[i] = order[i].second;
        if (tour[i] == 0) start = i;
    }
    
    rotate(tour.begin(), tour.begin() + start, tour.end());
    return tour;
}
