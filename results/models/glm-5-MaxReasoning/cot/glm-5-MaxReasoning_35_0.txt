Key heuristics I can use:
1. **Greedy approach**: Repeatedly select the vertex that covers the most uncovered vertices
2. **Greedy with weights**: Consider degrees and remaining uncovered neighbors
3. **Local search**: Start with a greedy solution and try to improve it by removing vertices or swapping
4. **Iterative improvement**: Try to remove redundant vertices

```rust
    let n = adj.len();
    if n == 0 {
        return vec![];
    }
    
    // Greedy construction
    let mut solution: HashSet<usize> = HashSet::new();
    let mut covered: Vec<bool> = vec![false; n];
    let mut uncovered_count = n;
    
    while uncovered_count > 0 {
        // Find vertex with best score (most uncovered neighbors + itself)
        let mut best_vertex = 0;
        let mut best_score = 0;
        
        for v in 0..n {
            let mut score = 0;
            if !covered[v] {
                score += 1;
            }
            for &u in &adj[v] {
                if !covered[u] {
                    score += 1;
                }
            }
            if score > best_score {
                best_score = score;
                best_vertex = v;
            }

        if best_score == 0 {
            break;

            uncovered_count -= 1;

                covered[u] = true;
                uncovered_count -= 1;
            }
        }

    let solution = remove_redundant(&solution, adj);
```

```rust
fn remove_redundant(solution: &HashSet<usize>, adj: &[Vec<usize>]) -> HashSet<usize> {
    let n = adj.len();
    let mut solution = solution.clone();
    let mut changed = true;
    
    while changed {
        changed = false;
        let vertices: Vec<usize> = solution.iter().cloned().collect();
        
        for v in vertices {
            if !solution.contains(&v) {
                continue;
            }
            // Check if v is essential
            // A vertex v is essential if there exists a vertex u that is only dominated by v
            let mut is_essential = false;
            
            // Check if v is the only dominator of itself
            if solution.iter().all(|&x| x != v) {
                // v dominates itself only if it's in the solution
                is_essential = true;
            } else {
                // Check neighbors
                for &u in &adj[v] {
                    if !solution.contains(&u) {
                        // u is not in solution, check if v is its only dominator
                        let mut dominator_count = 0;
                        for &w in &adj[u] {
                            if solution.contains(&w) && w != v {
                                dominator_count += 1;
                            }
                        }
                        if dominator_count == 0 {
                            is_essential = true;
                            break;
                        }
                    }
                }
            }
            
            if !is_essential {
                solution.remove(&v);
                changed = true;
            }
        }
    }
    
    solution
}
```

```rust
    order.sort_by(|&a, &b| adj[b].len().cmp(&adj[a].len()));
    
    let mut solution: HashSet<usize> = HashSet::new();
    let mut dominated: Vec<bool> = vec![false; n];
    
    for &v in &order {
        if dominated[v] {
            // Check if all neighbors are dominated
            let all_neighbors_dominated = adj[v].iter().all(|&u| dominated[u]);
            if all_neighbors_dominated {
                continue;
            }
        }
        
        // Add v to solution
        solution.insert(v);
        dominated[v] = true;
        for &u in &adj[v] {
            dominated[u] = true;
        }
    }
    
    solution
}

fn remove_redundant(solution: &mut HashSet<usize>, adj: &[Vec<usize>]) {
    let mut changed = true;
    
    while changed {
        changed = false;
        let vertices: Vec<usize> = solution.iter().cloned().collect();
        
        for v in vertices {
            if !solution.contains(&v) {
                continue;
            }
            
            solution.remove(&v);
            
            // Check if v is still dominated
            let mut v_dominated = false;
            for &u in &adj[v] {
                if solution.contains(&u) {
                    v_dominated = true;
                    break;
                }
            }
            
            if !v_dominated {
                solution.insert(v);
                continue;
            }
            
            // Check neighbors
            let mut all_covered = true;
            for &u in &adj[v] {
                if solution.contains(&u) {
                    continue;
                }
                let mut u_dominated = false;
                for &w in &adj[u] {
                    if solution.contains(&w) {
                        u_dominated = true;
                        break;
                    }
                }
                if !u_dominated {
                    all_covered = false;
                    break;
                }
            }
            
            if all_covered {
                changed = true;
            } else {
                solution.insert(v);
            }
        }
    }
}
```

The 2-1 swap is crucial for getting better solutions. Let me implement that:

```rust
fn local_search_2to1(solution: &mut HashSet<usize>, adj: &[Vec<usize>]) {
    let n = adj.len();
    let mut improved = true;
    
    while improved {
        improved = false;
        let sol_vec: Vec<usize> = solution.iter().cloned().collect();
        
        // Try removing pairs of vertices
        for i in 0..sol_vec.len() {
            for j in (i+1)..sol_vec.len() {
                let v1 = sol_vec[i];
                let v2 = sol_vec[j];
                
                solution.remove(&v1);
                solution.remove(&v2);
                
                // Find uncovered vertices
                let uncovered = find_uncovered(&solution, adj, &[v1, v2]);
                
                if uncovered.is_empty() {
                    // Found improvement! 2 removed, 0 added
                    improved = true;
                    break;
                }
                
                // Try to cover all uncovered with a single vertex
                for u in 0..n {
                    if solution.contains(&u) {
                        continue;
                    }
                    
                    // Check if u covers all uncovered
                    let covers_all = uncovered.iter().all(|&x| {
                        x == u || adj[u].contains(&x)
                    });
                    
                    if covers_all {
                        solution.insert(u);
                        improved = true;
                        break;
                    }
                }
                
                if improved {
                    break;
                }
                
                // Restore
                solution.insert(v1);
                solution.insert(v2);
            }
            if improved {
                break;
            }
        }
        
        if improved {
            remove_redundant(solution, adj);
        }
    }
}

fn find_uncovered(solution: &HashSet<usize>, adj: &[Vec<usize>], removed: &[usize]) -> Vec<usize> {
    let mut uncovered = Vec::new();
    
    for &v in removed {
        if !solution.contains(&v) {
            let mut v_dominated = false;
            for &u in &adj[v] {
                if solution.contains(&u) {
                    v_dominated = true;
                    break;
                }
            }
            if !v_dominated {
                uncovered.push(v);
            }
        }
    }
    
    // Check neighbors of removed vertices
    for &v in removed {
        for &u in &adj[v] {
            if solution.contains(&u) {
                continue;
            }
            let mut u_dominated = false;
            for &w in &adj[u] {
                if solution.contains(&w) {
                    u_dominated = true;
                    break;
                }
            }
            if !u_dominated && !uncovered.contains(&u) {
                uncovered.push(u);
            }
        }
    }
    
    uncovered
}
```

```rust
fn greedy_by_coverage(adj: &[Vec<usize>]) -> HashSet<usize> {
    let n = adj.len();
    let mut solution: HashSet<usize> = HashSet::new();
    let mut dominated: Vec<bool> = vec![false; n];
    let mut count = n;
    
    while count > 0 {
        let mut best_v = 0;
        let mut best_gain = 0;
        let mut best_deg = 0;
        
        for v in 0..n {
            let mut gain = 0;
            if !dominated[v] { gain += 1; }
            for &u in &adj[v] {
                if !dominated[u] { gain += 1; }
            }
            
            if gain > best_gain || (gain == best_gain && gain > 0 && adj[v].len() > best_deg) {
                best_gain = gain;
                best_v = v;
                best_deg = adj[v].len();
            }
        }
        
        if best_gain == 0 { break; }
        
        solution.insert(best_v);
        if !dominated[best_v] { dominated[best_v] = true; count -= 1; }
        for &u in &adj[best_v] {
            if !dominated[u] { dominated[u] = true; count -= 1; }
        }
    }
    
    solution
}
```
