        // KDK integrator step
        public void KDKStep(ref Vec3 pos, ref Vec3 vel, double dt, Func<Vec3, Vec3> acceleration)
        {
            // Kick
            vel = vel + acceleration(pos) * (dt * 0.5);
            // Drift
            pos = pos + vel * dt;
            // Kick
            vel = vel + acceleration(pos) * (dt * 0.5);

        public Vec3 GravitationalAcceleration(Vec3 pos, Dictionary<string, State> states)

                var body = kvp.Value;
                Vec3 r = states[kvp.Key].Position - pos;
                double rMag = r.Magnitude;
                if (rMag > 1e-10)

                    acc = acc + r.Normalized * (G * body.Mass / (rMag * rMag));
                }

        public (Vec3 v1, Vec3 v2, bool converged) SolveLambert(Vec3 r1, Vec3 r2, double tof, double mu)
        {
            // Simplified Lambert solver using Gooding's method
            double r1Mag = r1.Magnitude;
            double r2Mag = r2.Magnitude;
            
            Vec3 r1Norm = r1.Normalized;
            Vec3 r2Norm = r2.Normalized;
            
            double cosDeltaNu = r1Norm.Dot(r2Norm);
            double sinDeltaNu = Math.Max(0, Math.Sqrt(1 - cosDeltaNu * cosDeltaNu));
            
            // Check direction (prograde vs retrograde)
            Vec3 h = r1.Cross(r2);
            double deltaNu = Math.Atan2(sinDeltaNu, cosDeltaNu);
            if (h.Z < 0) deltaNu = 2 * Math.PI - deltaNu;
            
            double c = Math.Sqrt(r1Mag * r1Mag + r2Mag * r2Mag - 2 * r1Mag * r2Mag * cosDeltaNu);
            double s = (r1Mag + r2Mag + c) / 2;
            
            // Minimum energy semi-major axis
            double aMin = s / 2;
            
            // Iteration to find semi-major axis
            double a = aMin * 1.5; // Initial guess
            double tofActual = 0;
            
            for (int iter = 0; iter < 50; iter++)
            {
                double alpha = 1.0 / a;
                double beta = alpha * (s - c) / s;
                
                // Time of flight using universal variable formulation
                double p = Math.Sqrt(alpha);
                double q = Math.Sqrt(Math.Abs(beta));
                
                double lambda = Math.Sin(deltaNu / 2);
                double T = Math.Sqrt(a * a * a / mu) * (Math.Acos(1 - s/a) + Math.Sqrt(s*(s-c)/(a*a*a)));
                
                // Adjust for prograde/retrograde
                tofActual = T;
                
                if (Math.Abs(tofActual - tof) < 1e-6 * tof)
                    break;
                
                // Newton-Raphson update
                double dTda = -1.5 * tofActual / a;
                a = a - (tofActual - tof) / dTda;
                
                if (a < aMin) a = aMin * 1.01;
            }
            
            // Calculate velocities
            double f = 1 - r2Mag / a * (1 - Math.Cos(deltaNu));
            double g = r1Mag * r2Mag * Math.Sin(deltaNu) / Math.Sqrt(mu * a);
            double gDot = 1 - r1Mag / a * (1 - Math.Cos(deltaNu));
            
            Vec3 v1 = (r2 - r1 * f) / g;
            Vec3 v2 = (r2 * gDot - r1) / g;
            
            return (v1, v2, true);

            // Transfer 1: Earth to GA body
            var (v1, _, converged1) = SolveLambert(
                earthState.Position, gaBodyState.Position, 
                (gaTime - launchTime) * DAY_SEC, SUN_MU);
            
            if (!converged1) return null;
            
            Vec3 launchDV = v1 - earthState.Velocity;
            if (launchDV.Magnitude > deltaVBudget * 0.5) return null;
            
            maneuvers.Add(new Maneuver(launchTime, launchDV));
            
            // GA flyby (no burn, just direction change)
            maneuvers.Add(new Maneuver(gaTime, new Vec3(0, 0, 0)));
            
            // Transfer 2: GA body to Asteroid
            double arrivalTime = gaTime + 200;
            var stateAtArrival = PropagateBodies(arrivalTime);
            var asteroidState = stateAtArrival["Asteroid"];
            
            var (_, v2, converged2) = SolveLambert(
                gaBodyState.Position, asteroidState.Position,
                (arrivalTime - gaTime) * DAY_SEC, SUN_MU);
            
            if (!converged2) return null;
            
            maneuvers.Add(new Maneuver(arrivalTime, new Vec3(0, 0, 0)));
            
            return maneuvers;
        }
