    let mut lines = stdin.lock().lines();
    
    // Read dimensions
    let first_line = lines.next().unwrap().unwrap();
    let mut dims = first_line.split_whitespace();
    let h: usize = dims.next().unwrap().parse().unwrap();
    let w: usize = dims.next().unwrap().parse().unwrap();
    
    // Read maze into a flat vector
    let mut maze: Vec<u8> = Vec::with_capacity(h * w);
    let mut start = (0usize, 0usize);
    let mut end = (0usize, 0usize);
    
    for i in 0..h {
        let line = lines.next().unwrap().unwrap();
        let bytes = line.as_bytes();
        for j in 0..w.min(bytes.len()) {
            let c = bytes[j];
            if c == b'A' {
                start = (i, j);
            } else if c == b'B' {
                end = (i, j);
            }
            maze.push(c);
        }
        // Pad with spaces if line is shorter than w
        for _ in bytes.len()..w {
            maze.push(b' ');
        }
    }
    
    // BFS
    // Parent array: stores the direction we came from
    // 0 = from up, 1 = from down, 2 = from left, 3 = from right
    // 255 = not visited / no parent
    let mut parent: Vec<u8> = vec![255; h * w];
    
    let mut queue: VecDeque<(usize, usize)> = VecDeque::new();
    queue.push_back(start);
    parent[start.0 * w + start.1] = 254; // Mark start specially
    
    let directions: [(isize, isize); 4] = [(-1, 0), (1, 0), (0, -1), (0, 1)];
    // corresponding parent direction: from below, from above, from right, from left
    let parent_dirs: [u8; 4] = [1, 0, 3, 2];
    
    while let Some((i, j)) = queue.pop_front() {
        if (i, j) == end {
            break;
        }
        
        for (d, (di, dj)) in directions.iter().enumerate() {
            let ni = i as isize + di;
            let nj = j as isize + dj;
            
            if ni >= 0 && ni < h as isize && nj >= 0 && nj < w as isize {
                let ni = ni as usize;
                let nj = nj as usize;
                let idx = ni * w + nj;
                
                if parent[idx] == 255 && maze[idx] != b'#' {
                    parent[idx] = parent_dirs[d];
                    queue.push_back((ni, nj));
                }
            }
        }
    }
    
    // Reconstruct path
    let end_idx = end.0 * w + end.1;
    if parent[end_idx] == 255 && start != end {
        // No path found
        writeln!(out, "0").unwrap();
        return;
    }
    
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut cur = end;
    
    while cur != start {
        path.push(cur);
        let idx = cur.0 * w + cur.1;
        let dir = parent[idx];
        // Reverse the direction
        match dir {
            0 => cur = (cur.0 + 1, cur.1), // came from above, go down
            1 => cur = (cur.0.wrapping_sub(1), cur.1), // came from below, go up
            2 => cur = (cur.0, cur.1 + 1), // came from left, go right

use std::collections::VecDeque;

fn main() {
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    
    // Read all input at once for efficiency
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();

    // Parse height
    let mut h: usize = 0;
    while pos < len && input[pos] >= b'0' && input[pos] <= b'9' {
        h = h * 10 + (input[pos] - b'0') as usize;
        pos += 1;
    }
    
    // Skip space
    while pos < len && input[pos] == b' ' { pos += 1; }
    
    // Parse width
    let mut w: usize = 0;
    while pos < len && input[pos] >= b'0' && input[pos] <= b'9' {
        w = w * 10 + (input[pos] - b'0') as usize;
        pos += 1;
    }
    
    // Handle empty maze
    if h == 0 || w == 0 {
        writeln!(out, "0").unwrap();
        return;
    }
    
    // Skip to maze data
    while pos < len && (input[pos] == b' ' || input[pos] == b'\n' || input[pos] == b'\r') {
        pos += 1;
    }
    
    // Parse maze into flat array
    let total = h * w;
    let mut maze: Vec<u8> = vec![b' '; total];
    let mut start: Option<(usize, usize)> = None;
    let mut end: Option<(usize, usize)> = None;
    
    for i in 0..h {
        let row_start = i * w;
        for j in 0..w {
            if pos >= len { break; }
            let c = input[pos];
            if c == b'\n' || c == b'\r' { break; }
            maze[row_start + j] = c;
            if c == b'A' { start = Some((i, j)); }
            if c == b'B' { end = Some((i, j)); }
            pos += 1;
        }
        // Move to next line
        while pos < len && input[pos] != b'\n' { pos += 1; }
        if pos < len { pos += 1; }
    }
    
    // Validate start and end
    let start = match start {
        Some(s) => s,
        None => { writeln!(out, "0").unwrap(); return; }
    };
    let end = match end {
        Some(e) => e,
        None => { writeln!(out, "0").unwrap(); return; }
    };
    
    let start_idx = start.0 * w + start.1;
    let end_idx = end.0 * w + end.1;
    
    // Handle case where start == end
    if start_idx == end_idx {
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", start.1, start.0).unwrap();
        return;

    // BFS setup
    // parent[dir]: 0=parent above, 1=parent below, 2=parent left, 3=parent right, 255=unvisited
    let mut parent: Vec<u8> = vec![255; total];
    let mut queue: VecDeque<usize> = VecDeque::with_capacity(total.min(1 << 20));
    
    queue.push_back(start_idx);
    parent[start_idx] = 254; // Mark start as visited

        if i > 0 {

                if nidx == end_idx { found = true; break 'bfs; }
                queue.push_back(nidx);
            }

        if i + 1 < h {
            let nidx = idx + w;
            if parent[nidx] == 255 && maze[nidx] != b'#' {
                parent[nidx] = 0; // Parent is above
                if nidx == end_idx { found = true; break 'bfs; }
                queue.push_back(nidx);

                if nidx == end_idx { found = true; break 'bfs; }

            let nidx = idx + 1;
            if parent[nidx] == 255 && maze[nidx] != b'#' {
                parent[nidx] = 2; // Parent is left
                if nidx == end_idx { found = true; break 'bfs; }
                queue.push_back(nidx);
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
