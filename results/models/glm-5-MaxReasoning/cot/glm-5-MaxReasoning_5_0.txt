struct Grid {
    width: usize,
    height: usize,
    blocked: Vec<bool>,
}

impl Grid {
    fn new(width: usize, height: usize) -> Self {
        Grid {
            width,
            height,
            blocked: vec![false; width * height],
        }
    }
    
    fn idx(&self, x: usize, y: usize) -> usize {
        y * self.width + x
    }
    
    fn is_valid(&self, x: usize, y: usize) -> bool {
        x < self.width && y < self.height
    }
    
    fn is_blocked(&self, x: usize, y: usize) -> bool {
        if !self.is_valid(x, y) { return true; }
        self.blocked[self.idx(x, y)]
    }
    
    fn neighbors(&self, x: usize, y: usize) -> Vec<(usize, usize)> {
        let mut result = Vec::new();
        for (dx, dy) in [(-1i32, 0i32), (1, 0), (0, -1), (0, 1)] {
            let nx = x as i32 + dx;
            let ny = y as i32 + dy;
            if nx >= 0 && ny >= 0 && nx < self.width as i32 && ny < self.height as i32 {
                let (nx, ny) = (nx as usize, ny as usize);
                if !self.is_blocked(nx, ny) {
                    result.push((nx, ny));
                }
            }
        }
        result
    }
    
    fn count_free_cells(&self) -> usize {
        self.blocked.iter().filter(|&&b| !b).count()
    }
}

// Check connectivity using BFS
fn is_connected(grid: &Grid, start_x: usize, start_y: usize) -> bool {
    let total = grid.count_free_cells();
    if total == 0 { return true; }
    
    let mut visited = vec![false; grid.width * grid.height];
    let mut queue = VecDeque::new();
    queue.push_back((start_x, start_y));
    visited[grid.idx(start_x, start_y)] = true;
    let mut count = 1;
    
    while let Some((x, y)) = queue.pop_front() {
        for (nx, ny) in grid.neighbors(x, y) {
            let idx = grid.idx(nx, ny);
            if !visited[idx] {
                visited[idx] = true;
                count += 1;
                queue.push_back((nx, ny));
            }
        }
    }
    
    count == total
}

// Check bipartite parity condition
fn check_parity(grid: &Grid, require_cycle: bool) -> bool {
    let mut black = 0usize;
    let mut white = 0usize;
    
    for y in 0..grid.height {
        for x in 0..grid.width {
            if !grid.is_blocked(x, y) {
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }
    
    if require_cycle {
        // For a cycle, we need equal black and white
        black == white
    } else {
        // For a path, difference should be at most 1
        // And the starting cell (0,0) is black, so if black > white, we end on white
        // If white > black, we end on black (but start is black, so this is impossible)
        // Actually, for Hamiltonian path starting at (0,0):
        // - If we have n cells total, we visit black, white, black, white, ...
        // - So if n is odd, we end on black (same color as start), black = white + 1
        // - If n is even, we end on white, black = white
        let diff = if black > white { black - white } else { white - black };
        diff <= 1 && (black >= white) // Starting on black means we can't have more whites
    }
}

        if self.target_length == 0 {
            return Some(vec![]);
        }
        
        if self.grid.is_blocked(0, 0) {

            None
        }

        self.get_unvisited_neighbors(x, y).len()

        neighbors.sort_by_key(|&(nx, ny)| {
            self.count_unvisited_neighbors(nx, ny)
        });
        
        neighbors

        // Check if excluding this cell creates any dead end

            for x in 0..self.grid.width {
                if self.grid.is_blocked(x, y) || self.visited[self.grid.idx(x, y)] {

                }
                if self.count_unvisited_neighbors(x, y) == 0 {

                }

        }
        false

        let mut count = 0;
        let total_unvisited: usize = (0..self.grid.height)

            .filter(|&(x, y)| {
                !self.grid.is_blocked(x, y) && !self.visited[self.grid.idx(x, y)]
            })
            .count();
        
        if total_unvisited == 0 {
            return true;
        }

        let start = (0..self.grid.height)
            .flat_map(|y| (0..self.grid.width).map(move |x| (x, y)))
            .find(|&(x, y)| {
                !self.grid.is_blocked(x, y) && !self.visited[self.grid.idx(x, y)]
            });
        
        let Some((sx, sy)) = start else { return true; };
        
        let mut local_visited = vec![false; self.grid.width * self.grid.height];
        let mut queue = VecDeque::new();
        queue.push_back((sx, sy));
        local_visited[self.grid.idx(sx, sy)] = true;
        count = 1;
        
        while let Some((x, y)) = queue.pop_front() {
            for (nx, ny) in self.grid.neighbors(x, y) {
                let idx = self.grid.idx(nx, ny);
                if !self.visited[idx] && !local_visited[idx] {
                    local_visited[idx] = true;
                    count += 1;
                    queue.push_back((nx, ny));
                }
            }
        }
        
        count == total_unvisited
    }
    
    fn backtrack(&mut self) -> bool {
        let length = self.path.len();
        
        if length == self.target_length {
            // Check cycle condition
            if self.require_cycle {
                let (last_x, last_y) = self.path[length - 1];
                // Check if last cell is adjacent to (0, 0)
                let dx = (last_x as i32 - 0i32).abs();
                let dy = (last_y as i32 - 0i32).abs();
                return (dx == 1 && dy == 0) || (dx == 0 && dy == 1);
            }
            return true;
        }
        
        let (x, y) = self.path[length - 1];
        let neighbors = self.get_sorted_neighbors(x, y);
        
        for (nx, ny) in neighbors {
            // Make move
            self.visited[self.grid.idx(nx, ny)] = true;
            self.path.push((nx, ny));
            
            // Pruning checks
            let mut valid = true;
            
            // Check for dead ends
            if self.has_dead_end(nx, ny) {
                valid = false;
            }
            
            // Check connectivity periodically (expensive)
            if valid && length % 10 == 0 && length < self.target_length - 10 {
                if !self.unvisited_connected(nx, ny) {
                    valid = false;
                }
            }
            
            if valid && self.backtrack() {
                return true;
            }
            
            // Undo move
            self.path.pop();
            self.visited[self.grid.idx(nx, ny)] = false;
        }
        
        false
    }
}

struct Grid {
    width: usize,
    height: usize,
    blocked: Vec<bool>,
    neighbor_cache: Vec<Vec<(usize, usize)>>,
}

impl Grid {
    fn new(width: usize, height: usize) -> Self {
        let mut grid = Grid {
            width,
            height,
            blocked: vec![false; width * height],
            neighbor_cache: Vec::new(),

        grid

    fn build_neighbor_cache(&mut self) {
        self.neighbor_cache = vec![Vec::new(); self.width * self.height];
        for y in 0..self.height {
            for x in 0..self.width {
                let idx = self.idx(x, y);
                for (dx, dy) in [(-1i32, 0i32), (1, 0), (0, -1), (0, 1)] {
                    let nx = x as i32 + dx;
                    let ny = y as i32 + dy;

            diff <= 1 && black >= white
        }
    }
    
    fn backtrack(&mut self) -> bool {
        let length = self.path.len();
        
        if length == self.target_length {
            if self.require_cycle {
                let (last_x, last_y) = self.path[length - 1];
                let dx = (last_x as i32).abs();
                let dy = (last_y as i32).abs();
                return (dx == 1 && dy == 0) || (dx == 0 && dy == 1);
            }
            return true;
        }
        
        // Check for isolated cells (quick prune)
        if self.has_isolated_cell() {
            return false;
        }
        
        let (x, y) = self.path[length - 1];
        let neighbors = self.get_sorted_neighbors(x, y);
        
        if neighbors.is_empty() {
            return false;
        }
        
        // Try each neighbor
        for (nx, ny) in neighbors {
            self.path.push((nx, ny));
            self.mark_visited(nx, ny, true);
            
            if self.backtrack() {
                return true;
            }
            
            self.path.pop();
            self.mark_visited(nx, ny, false);
        }
        
        false
    }
}

            self.update_degrees(next, false);
            self.visited[next] = false;
            self.path.pop();
        }
        
        None
    }
}

fn backtrack_solve(
    width: usize, 
    height: usize, 
    blocked: &[bool], 
    neighbors: &[Vec<usize>], 
    free_cells: usize,
    require_cycle: bool
) -> Option<Vec<(usize, usize)>> {
    let n = width * height;
    let mut degree = vec![0u8; n];
    for idx in 0..n {
        if !blocked[idx] {
            degree[idx] = neighbors[idx].len() as u8;
        }
    }
    
    let mut state = BacktrackState {
        width,
        height,
        neighbors: neighbors.to_vec(),
        visited: vec![false; n],
        degree,
        path: Vec::new(),
        target_length: free_cells,
        require_cycle,
    };
    
    state.path.push(0);
    state.visited[0] = true;
    state.update_degrees(0, true);
    
    match state.solve() {
        Some(path) => {
            Some(path.into_iter().map(|idx| (idx % width, idx / width)).collect())
        }
        None => None,
    }
}
