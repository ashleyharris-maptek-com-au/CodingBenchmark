3. **Task:** Implement toon/cel shading with:
   - Calculate Lambertian diffuse: `d = max(dot(N, L), 0)`

   - Band values: 1.0, 0.7, 0.4, 0.2
   
   So the mapping should be:
   - d > 0.75 → value = 1.0
   - d > 0.5 → value = 0.7

layout(location = 1) in vec3 normal;

    vec4 cameraPos;
    vec4 params;

    // Normalize the interpolated normal
    vec3 N = normalize(normal);

    float d = max(dot(N, L), 0.0);
    
    // Quantize into bands
    float band;
    if (d > 0.75) {
        band = 1.0;
    } else if (d > 0.5) {
        band = 0.7;
    } else if (d > 0.25) {
        band = 0.4;
    } else {
        band = 0.2;
    }
    
    // Multiply vertex color by band
    outColor = vec4(color * band, 1.0);
}
