class Node:
    __slots__ = ('left','right','up','down','col','row_id')
    def __init__(self, row_id=None, col=None):
        self.left = self.right = self.up = self.down = self
        self.col = col
        self.row_id = row_id

class Column(Node):
    __slots__ = ('id','size')
    def __init__(self, id):
        super().__init__()
        self.id = id
        self.size = 0

def solve():
    import sys
    sys.setrecursionlimit(10000)
    data = sys.stdin.read().strip().splitlines()
    if not data:
        return
    first = data[0].split()
    if len(first) < 2:
        # maybe error
        print("NO SOLUTION")
        return
    U = int(first[0])
    S = int(first[1])
    if U == 0:
        # No elements, empty solution
        print("SOLUTION")
        print()
        return
    # Prepare columns
    columns = [Column(i) for i in range(U)]
    root = Node()
    # Link columns horizontally
    prev = root
    for col in columns:
        prev.right = col
        col.left = prev
        col.right = root
        root.left = col
        prev = col
    # Process sets
    for i in range(S):
        # line i+1 in data
        line = data[1+i] if i+1 < len(data) else ''
        parts = line.strip().split()
        if not parts:
            continue
        elements = set()
        for p in parts:
            elements.add(int(p))
        if not elements:
            continue
        # create row nodes
        row_nodes = []
        for e in sorted(elements):  # sorted for deterministic linking
            node = Node(row_id=i, col=columns[e])
            # insert node into column e's vertical list
            node.up = columns[e].up
            node.down = columns[e]
            columns[e].up.down = node
            columns[e].up = node
            columns[e].size += 1
            row_nodes.append(node)
        # link row nodes horizontally
        for idx, node in enumerate(row_nodes):
            node.left = row_nodes[idx-1]  # -1 works for first element (last node)
            node.right = row_nodes[(idx+1) % len(row_nodes)]
    # Now define DLX functions
    answer = []
    def cover(col):
        # col is a Column
        col.left.right = col.right
        col.right.left = col.left
        i = col.down
        while i != col:
            j = i.right
            while j != i:
                j.down.up = j.up
                j.up.down = j.down
                j.col.size -= 1
                j = j.right
            i = i.down
    def uncover(col):
        i = col.up
        while i != col:
            j = i.left
            while j != i:
                j.col.size += 1
                j.up.down = j
                j.down.up = j
                j = j.left
            i = i.up
        col.left.right = col
        col.right.left = col
    def search():
        nonlocal answer
        if root.right == root:
            return True
        # choose column with smallest size
        c = root.right
        min_col = c
        min_size = c.size
        col = c.right
        while col != root:
            if col.size < min_size:
                min_size = col.size
                min_col = col
            col = col.right
        if min_size == 0:
            return False
        c = min_col
        cover(c)
        r = c.down
        while r != c:
            answer.append(r.row_id)
            j = r.right
            while j != r:
                cover(j.col)
                j = j.right
            if search():
                return True
            answer.pop()
            j = r.left
            while j != r:
                uncover(j.col)
                j = j.left
            r = r.down
        uncover(c)
        return False
    if search():
        answer.sort()
        print("SOLUTION")
        print(" ".join(map(str, answer)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()

One potential bug: In cover, we use col.down to iterate. Initially, for a column with no nodes, col.down points to col itself (since column's down is set to self in Node.__init__). So the while loop will not execute. That's fine.

But when we cover a column, we need to ensure that we don't attempt to cover a column that is already covered (i.e., not in header list). In our algorithm, we only cover columns that are still in the header list because we choose them from the list, and we cover other columns from rows that are still present (since rows are only those still in the matrix).
