            const key = `${x},${y}`;
            if (!visited.has(key)) {
                totalValue += grid[x][y];

                seen.add(key);
                queue.push([nx, ny, dist + 1]);
            }
        }
    }
    
    return totalValue;

function computeDistances(startX: number, startY: number, grid: number[][], rows: number, cols: number): Map<string, number> {
    const distances = new Map<string, number>();
    const queue: [number, number, number][] = [[startX, startY, 0]];
    distances.set(`${startX},${startY}`, 0);
    
    const dirs = [[-1, 0], [1, 0], [0, -1], [0, 1]];
    
    while (queue.length > 0) {
        const [x, y, d] = queue.shift()!;

            const ny = y + dy;

    let y = startY;
    
    const dirs = [[-1, 0], [1, 0], [0, -1], [0, 1], [0, 0]];

            targets.push([vx, vy, value, dist]);
        }

    targets.sort((a, b) => {
        const ratioA = a[2] / (a[3] + 1);
        const ratioB = b[2] / (b[3] + 1);
        return ratioB - ratioA;

                    bestDist = newDist;
                    bestMove = [nx, ny];

                    // Prefer moves that visit valuable cells
                    const newVal = visited.has(`${nx},${ny}`) ? 0 : grid[nx][ny];
                    const bestVal = visited.has(`${bestMove[0]},${bestMove[1]}`) ? 0 : grid[bestMove[0]][bestMove[1]];
