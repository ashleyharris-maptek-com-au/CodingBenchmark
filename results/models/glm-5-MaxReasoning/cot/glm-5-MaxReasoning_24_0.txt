class Terrain {
    int width_ = 0, depth_ = 0;
    
#ifdef _WIN32
    HANDLE file_handle_ = INVALID_HANDLE_VALUE;
    HANDLE mapping_handle_ = NULL;
    const uint8_t* data_ = nullptr;
    size_t file_size_ = 0;
#else
    std::vector<uint8_t> data_;
#endif

    std::vector<uint16_t> floating_bottom_;
    std::vector<uint16_t> floating_top_;

public:
    std::atomic<bool> loaded{false};
    
    ~Terrain() {
#ifdef _WIN32
        if (data_) UnmapViewOfFile(data_);
        if (mapping_handle_) CloseHandle(mapping_handle_);
        if (file_handle_ != INVALID_HANDLE_VALUE) CloseHandle(file_handle_);
#endif
    }
    
    bool load(const std::string& path, int width, int depth) {
        width_ = width;
        depth_ = depth;
        size_t expected_size = static_cast<size_t>(width) * depth * 6;
        
#ifdef _WIN32
        // Try memory mapping first
        file_handle_ = CreateFileA(path.c_str(), GENERIC_READ, FILE_SHARE_READ,
                                   NULL, OPEN_EXISTING, FILE_ATTRIBUTE_NORMAL, NULL);
        if (file_handle_ == INVALID_HANDLE_VALUE) {
            // Fall back to regular loading
            return loadVector(path, width, depth);
        }
        
        LARGE_INTEGER file_size;
        GetFileSizeEx(file_handle_, &file_size);
        file_size_ = file_size.QuadPart;
        
        if (file_size_ < expected_size) {
            CloseHandle(file_handle_);
            file_handle_ = INVALID_HANDLE_VALUE;
            return false;
        }
        
        mapping_handle_ = CreateFileMappingA(file_handle_, NULL, PAGE_READONLY, 0, 0, NULL);
        if (!mapping_handle_) {
            CloseHandle(file_handle_);
            file_handle_ = INVALID_HANDLE_VALUE;
            return loadVector(path, width, depth);
        }
        
        data_ = static_cast<const uint8_t*>(MapViewOfFile(mapping_handle_, FILE_MAP_READ, 0, 0, 0));
        if (!data_) {
            CloseHandle(mapping_handle_);
            CloseHandle(file_handle_);
            mapping_handle_ = NULL;
            file_handle_ = INVALID_HANDLE_VALUE;
            return loadVector(path, width, depth);
        }
        
        use_vector_ = false;
#else
        return loadVector(path, width, depth);
#endif
        loaded = true;
        return true;

        std::ifstream file(path, std::ios::binary | std::ios::ate);
        if (!file) return false;

        file.seekg(0);
        
        size_t expected_size = static_cast<size_t>(width) * depth * 6;

                floating_bottom_[idx] = buf[2] | (buf[3] << 8);
                floating_top_[idx] = buf[4] | (buf[5] << 8);
            }

        use_vector_ = true;
        loaded = true;
        return true;

            return {0, 0, 0};

        double tx = world_params.target_x;
        double ty = world_params.target_y;

            }
        }
        
        // Velocity errors
        double evx = desired_vx - vx;
        double evy = desired_vy - vy;
        double evz = desired_vz - vz;
        
        // Desired accelerations (in world frame)
        double ax = evx * kd_xy;
        double ay = evy * kd_xy;
        double az = evz * kd_z + world_params.gravity; // Add gravity
        
        // Terrain avoidance - check if we need to go up
        if (terrain.loaded && z < 50) {
            double ground_h = terrain.getGroundHeight(x, y);
            if (z < ground_h + 10) {
                az = std::max(az, world_params.gravity + 3.0); // Push up
            }
        }
        
        // Compute thrust magnitude
        double thrust_mag = std::sqrt(ax * ax + ay * ay + az * az);
        
        // Compute thrust fraction
        out.thrust = clamp(thrust_mag / world_params.max_thrust, 0.0, 1.0);
        
        // Compute required attitude
        double required_pitch = 0;

        if (thrust_mag > 0.01) {

            double tx_d = ax / thrust_mag;
            double ty_d = ay / thrust_mag;

            // Compute yaw (rotation around Z)
            required_yaw = std::atan2(-tx_d, -ty_d);

    std::ifstream file(path, std::ios::binary);
    if (!file) {

        return false;

// Controller
class LanderController {
    // Control gains - tuned for various scales
    double kp_horiz = 0.4;
    double kd_horiz = 1.0;
    double kp_vert = 0.6;
    double kd_vert = 1.5;
    
    double kp_attitude = 4.0;
    double kd_attitude = 2.0;
    
public:
    void computeControl(
        double x, double y, double z,
        double vx, double vy, double vz,
        double pitch, double yaw, double roll,
        double pitch_rate, double yaw_rate, double roll_rate,
        double fuel,
        double& out_thrust, double& out_pitch, double& out_yaw, double& out_roll
    ) {
        // Target position
        double tx = world_params.target_x;
        double ty = world_params.target_y;
        double tz = world_params.target_z;
        
        // Position errors
        double ex = tx - x;
        double ey = ty - y;
        double ez = tz - z;
        
        double dist_h = std::sqrt(ex * ex + ey * ey);
        double dist_v = std::abs(ez);
        
        // Determine control phase
        double desired_vx, desired_vy, desired_vz;
        
        // Approach phase
        if (dist_h > 30.0) {
            // Move towards target horizontally
            double speed = std::min(dist_h * 0.3, 20.0);
            desired_vx = speed * ex / dist_h;
            desired_vy = speed * ey / dist_h;
            
            // Maintain safe altitude
            double ground_h = terrain.loaded ? terrain.getGroundHeight(x, y) : 0.0;
            double safe_alt = std::max(tz + 20.0, ground_h + 30.0);
            desired_vz = clamp((safe_alt - z) * 0.4, -8.0, 8.0);
        }
        // Near target - prepare for landing
        else if (dist_h > 10.0 || dist_v > 10.0) {
            double slow_factor = dist_h / 30.0;
            desired_vx = ex * 0.4 * slow_factor;
            desired_vy = ey * 0.4 * slow_factor;
            
            // Hover above target
            double hover_alt = tz + 15.0;
            desired_vz = clamp((hover_alt - z) * 0.4, -5.0, 5.0);
        }
        // Landing phase
        else {
            desired_vx = ex * 0.3;
            desired_vy = ey * 0.3;
            
            // Descend slowly
            double speed = std::sqrt(vx*vx + vy*vy + vz*vz);
            if (speed < 3.0) {
                desired_vz = clamp(ez * 0.3, -1.5, 1.5);
            } else {
                desired_vz = clamp(-vz * 0.5, -1.0, 1.0);
            }
        }
        
        // Terrain avoidance
        if (terrain.loaded) {
            double ground_h = terrain.getGroundHeight(x, y);
            double clearance = z - ground_h;
            
            if (clearance < 15.0) {
                // Push up
                desired_vz = std::max(desired_vz, 3.0);
            }
            
            // Check for floating rocks above
            double ceiling = terrain.getCeilingHeight(x, y, z);
            if (ceiling - z < 10.0 && ceiling < 1e10) {
                // Slow down vertical velocity
                desired_vz = std::min(desired_vz, 1.0);
            }
        }
        
        // Velocity errors
        double evx = desired_vx - vx;
        double evy = desired_vy - vy;
        double evz = desired_vz - vz;
        
        // Compute required acceleration (world frame)
        double ax = evx * kd_horiz;
        double ay = evy * kd_horiz;
        double az = evz * kd_vert + world_params.gravity;
        
        // Thrust magnitude
        double thrust_mag = std::sqrt(ax * ax + ay * ay + az * az);
        out_thrust = clamp(thrust_mag / world_params.max_thrust, 0.0, 1.0);
        
        // Compute required attitude
        double required_pitch = 0.0;
        double required_yaw = yaw;
        
        if (thrust_mag > 0.01 * world_params.max_thrust) {
            // Normalized thrust direction
            double tnx = ax / thrust_mag;
            double tny = ay / thrust_mag;
            double tnz = az / thrust_mag;
            
            // Compute required pitch
            double horiz = std::sqrt(tnx * tnx + tny * tny);
            required_pitch = std::atan2(horiz, tnz);
            required_pitch = clamp(required_pitch, -PI/2 + 0.05, PI/2 - 0.05);
            
            // Compute required yaw
            // From thrust equations: tx = -sin(pitch)*cos(yaw), ty = -sin(pitch)*sin(yaw)
            // So: cos(yaw) = -tx/sin(pitch), sin(yaw) = -ty/sin(pitch)
            if (horiz > 0.01) {
                required_yaw = std::atan2(-tny, -tnx);
            }
        }
        
        // Attitude control (PD)
        double pitch_error = normalizeAngle(required_pitch - pitch);
        double yaw_error = normalizeAngle(required_yaw - yaw);
        double roll_error = normalizeAngle(-roll); // Try to keep roll at 0
        
        out_pitch = clamp(pitch_error * kp_attitude - pitch_rate * kd_attitude, -1.0, 1.0);
        out_yaw = clamp(yaw_error * kp_attitude - yaw_rate * kd_attitude, -1.0, 1.0);
        out_roll = clamp(roll_error * kp_attitude - roll_rate * kd_attitude, -1.0, 1.0);
        
        // Fuel conservation - reduce thrust when not needed
        if (dist_h < 5.0 && dist_v < 5.0) {
            double speed = std::sqrt(vx*vx + vy*vy + vz*vz);
            if (speed < 2.0) {
                // Very close to landing, minimal thrust
                out_thrust = clamp(out_thrust, 0.0, 0.6);
            }
        }
    }
};
