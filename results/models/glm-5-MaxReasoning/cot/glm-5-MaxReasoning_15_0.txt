struct Tetrahedron {
    center: [f64; 3],
    quat: [f64; 4],
    scale: f64,
}

const SQRT2_2: f64 = 0.7071067811865476;

const TETRA_VERTICES: [[f64; 3]; 4] = [
    [1.0, 0.0, -SQRT2_2],
    [-1.0, 0.0, -SQRT2_2],
    [0.0, 1.0, SQRT2_2],
    [0.0, -1.0, SQRT2_2],
];

fn compute_shadow(tet: &Tetrahedron, sun: &[f64; 3]) -> Vec<[f64; 2]> {
    let mut shadow_points = Vec::with_capacity(4);
    
    for v in &TETRA_VERTICES {
        // Scale and translate
        let vx = tet.center[0] + tet.scale * v[0];
        let vy = tet.center[1] + tet.scale * v[1];
        let vz = tet.center[2] + tet.scale * v[2];
        
        // Project onto z=0 along sun direction
        // Ray: (vx + t*sx, vy + t*sy, vz + t*sz)
        // At z=0: vz + t*sz = 0 => t = -vz/sz
        // Assuming sz < 0 (sun pointing down) and vz > 0 (vertex above ground)
        
        if sun[2] < 0.0 && vz > 0.0 {
            let t = -vz / sun[2];
            let sx = vx + t * sun[0];
            let sy = vy + t * sun[1];
            shadow_points.push([sx, sy]);
        }
    }
    
    // Compute convex hull of shadow points
    convex_hull(&shadow_points)
}

fn convex_hull(points: &[[f64; 2]]) -> Vec<[f64; 2]> {
    if points.len() <= 3 {
        return points.to_vec();
    }
    
    // Simple convex hull for small number of points
    let n = points.len();
    let mut hull = Vec::new();
    
    // Find the leftmost point
    let mut leftmost = 0;
    for i in 1..n {
        if points[i][0] < points[leftmost][0] {
            leftmost = i;
        }
    }
    
    let mut p = leftmost;
    loop {
        hull.push(points[p]);
        let mut q = (p + 1) % n;
        
        for i in 0..n {
            if orientation(points[p], points[i], points[q]) == 2 {
                q = i;
            }
        }
        
        p = q;
        if p == leftmost {
            break;
        }
    }
    
    hull
}

fn orientation(p: [f64; 2], q: [f64; 2], r: [f64; 2]) -> i32 {
    let val = (q[1] - p[1]) * (r[0] - q[0]) - (q[0] - p[0]) * (r[1] - q[1]);
    if val > 0.0 { 1 } else if val < 0.0 { 2 } else { 0 }
}
