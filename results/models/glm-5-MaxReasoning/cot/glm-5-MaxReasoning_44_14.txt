**5. Writing the code:**

```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize normal because of interpolation artifacts
    vec3 N = normalize(normal);
    
    // Light direction (point light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Diffuse calculation
    float diffuse = max(dot(N, L), 0.0);
    
    // Lambertian factor as requested: 0.3 + 0.7 * diffuse
    float lambertian = 0.3 + 0.7 * diffuse;
    
    // Base color applied with lighting
    vec3 baseColor = vec3(0.8, 0.2, 0.2);
    vec3 litColor = baseColor * lambertian;
    
    // Fog calculation
    // Distance from camera to fragment
    float dist = length(worldPos - ubo.cameraPos.xyz);
    
    // Fog factor: clamp((dist - 1.5) / 3.0, 0.0, 1.0)
    // Assuming standard [0, 1] clamping for fog factor
    float fogFactor = clamp((dist - 1.5) / 3.0, 0.0, 1.0);
    
    // Fog color
    vec3 fogColor = vec3(0.7, 0.7, 0.8);
    
    // Mix lit color with fog color
    vec3 finalColor = mix(litColor, fogColor, fogFactor);
    
    // Output with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
```
