fn meet_in_the_middle(numbers: &[(usize, i64)], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mid = n / 2;
    
    let left = &numbers[..mid];
    let right = &numbers[mid..];
    
    // Enumerate all subset sums for left half
    let mut left_sums: HashMap<i64, Vec<usize>> = HashMap::new();
    let left_n = left.len();
    
    for mask in 0usize..(1usize << left_n) {
        let mut sum: i64 = 0;
        let mut indices = Vec::new();
        for i in 0..left_n {
            if mask & (1 << i) != 0 {
                sum += left[i].1;
                indices.push(left[i].0);
            }
        }
        left_sums.insert(sum, indices);
    }
    
    // Enumerate all subset sums for right half and check for target
    let right_n = right.len();
    for mask in 0usize..(1usize << right_n) {
        let mut sum: i64 = 0;
        let mut indices = Vec::new();
        for i in 0..right_n {
            if mask & (1 << i) != 0 {
                sum += right[i].1;
                indices.push(right[i].0);
            }
        }
        
        let needed = target - sum;
        if let Some(left_indices) = left_sums.get(&needed) {
            let mut result = left_indices.clone();
            result.extend(indices);
            return Some(result);
        }
    }
    
    None
}

fn dp_bitset(numbers: &[(usize, i64)], target: usize) -> Option<Vec<usize>> {
    let n = numbers.len();
    
    // dp[i] is a bitset indicating which sums are achievable
    // We'll track which element was used to achieve each sum
    let mut achievable = vec![false; target + 1];
    achievable[0] = true;
    
    let mut parent: Vec<Option<usize>> = vec![None; target + 1];
    
    for (idx, (_, &val)) in numbers.iter().enumerate() {
        if val as usize > target {
            continue;
        }
        let val = val as usize;
        
        // Iterate backwards to avoid using the same element twice
        for s in (val..=target).rev() {
            if achievable[s - val] && !achievable[s] {
                achievable[s] = true;
                parent[s] = Some(idx);
            }
        }
        
        if achievable[target] {
            // Reconstruct
            let mut indices = Vec::new();
            let mut s = target;
            while s > 0 {
                let idx = parent[s].unwrap();
                indices.push(numbers[idx].0);
                s -= numbers[idx].1 as usize;
            }
            return Some(indices);
        }
    }
    
    None
}
