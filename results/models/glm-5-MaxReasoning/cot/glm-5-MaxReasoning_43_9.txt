float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);

    // Calculate light direction
    // lightPos is float4, worldPos is float3. 
    // Assuming standard point light calculation.
    float3 L = normalize(input.worldPos - lightPos.xyz); 
    // Wait, standard diffuse usually does L = normalize(lightPos - Pos).
    // Vector from Point to Light is (Light - Point).
    // Vector from Light to Point is (Point - Light).
    // Dot(N, L) where L points TO light. So L = normalize(lightPos.xyz - input.worldPos).
    
    float3 L_dir = normalize(lightPos.xyz - input.worldPos);

    // Lambert term
    float NdotL = dot(N, L_dir);
    float diffuse = max(0.0, NdotL);

    // Specific factor requested: 0.5 + 0.5 * diffuse
    float intensity = 0.5 + 0.5 * diffuse;

    // Checkerboard 8x8
    // UV * 8 to get coordinates in [0, 8]
    float2 scaledUV = input.uv * 8.0;
    
    // Floor to get integer cell index
    // Cast to int for safety with modulo, or use fmod on floats
    float2 cell = floor(scaledUV);
    
    // Parity check: (x + y) % 2
    // fmod(x + y, 2.0) returns 0.0 or 1.0 (roughly)
    float parity = fmod(cell.x + cell.y, 2.0);
    
    // Alternating colors 0.9 and 0.2
    // If parity is 0 -> 0.9. If parity is 1 -> 0.2.
    float checkerColor = lerp(0.9, 0.2, parity);

    // Apply lighting
    float3 finalColor = checkerColor * intensity;

    return float4(finalColor, 1.0);
}
