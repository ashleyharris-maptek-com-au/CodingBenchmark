class MaxCliqueSolver
{
    private int n;
    private HashSet<int>[] adj;
    private int[] bestClique;
    private int bestSize;
    private object lockObj = new object();
    private Stopwatch sw;
    private const long TimeLimitMs = 290000;
    
    public MaxCliqueSolver(int n, List<(int, int)> edges)
    {
        this.n = n;
        adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new HashSet<int>();
        
        foreach (var (u, v) in edges)
        {
            adj[u].Add(v);
            adj[v].Add(u);
        }
        
        bestClique = new int[n];
        bestSize = 0;
        sw = Stopwatch.StartNew();
    }
    
    public List<int> Solve()
    {
        // Start with greedy solution
        GreedySolution();
        
        // Then try exact solution with time limit
        ExactSolution();
        
        return bestClique.Take(bestSize).ToList();
    }
    
    private void GreedySolution()
    {
        // Multiple greedy attempts with different orderings
        for (int attempt = 0; attempt < 10; attempt++)
        {
            var clique = GreedyClique(attempt);
            UpdateBest(clique);
        }
    }
    
    private List<int> GreedyClique(int seed)
    {
        var clique = new List<int>();
        var candidates = new HashSet<int>(Enumerable.Range(0, n));
        var rng = new Random(seed);
        
        // Order by degree (or random for different attempts)
        var ordered = candidates
            .OrderByDescending(v => seed == 0 ? adj[v].Count : adj[v].Count + rng.NextDouble())
            .ToList();
        
        foreach (int v in ordered)
        {
            if (clique.All(u => adj[v].Contains(u)))
            {
                clique.Add(v);
            }
        }
        
        return clique;
    }
    
    private void ExactSolution()
    {
        // Use degeneracy ordering for better pruning
        var ordering = DegeneracyOrdering();
        var orderingMap = new int[n];
        for (int i = 0; i < n; i++)
            orderingMap[ordering[i]] = i;
        
        // Process vertices in parallel
        var options = new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount };
        
        try
        {
            Parallel.For(0, n, options, (i, state) =>
            {
                if (sw.ElapsedMilliseconds > TimeLimitMs)
                {
                    state.Break();
                    return;
                }
                
                int v = ordering[i];
                var clique = new List<int> { v };
                var candidates = new List<int>();
                
                // Candidates are neighbors with higher order (smaller core number)
                foreach (int u in adj[v])
                {
                    if (orderingMap[u] > i)
                        candidates.Add(u);
                }
                
                // Sort by degree in induced subgraph
                candidates.Sort((a, b) => adj[b].Count - adj[a].Count);
                
                Expand(clique, candidates, 0);
            });
        }
        catch (OperationCanceledException) { }
    }
    
    private void Expand(List<int> clique, List<int> candidates, int depth)
    {
        if (sw.ElapsedMilliseconds > TimeLimitMs)
            return;
        
        if (candidates.Count == 0)
        {
            UpdateBest(clique);
            return;
        }
        
        // Color candidates for upper bound
        var colors = GreedyColor(candidates);
        
        // Prune if upper bound doesn't beat best
        int maxColor = colors.Max();
        if (clique.Count + maxColor <= bestSize)
            return;
        
        // Process candidates
        for (int i = candidates.Count - 1; i >= 0; i--)
        {
            if (clique.Count + colors[i] <= bestSize)
                return;
            
            int v = candidates[i];
            var newCandidates = new List<int>();
            
            for (int j = 0; j < i; j++)
            {
                int u = candidates[j];
                if (adj[v].Contains(u))
                    newCandidates.Add(u);
            }
            
            clique.Add(v);
            Expand(clique, newCandidates, depth + 1);
            clique.RemoveAt(clique.Count - 1);
        }
    }
    
    private int[] GreedyColor(List<int> vertices)
    {
        var colors = new int[vertices.Count];
        var usedColors = new HashSet<int>();
        var neighborColors = new HashSet<int>();
        
        for (int i = 0; i < vertices.Count; i++)
        {
            neighborColors.Clear();
            int v = vertices[i];
            
            for (int j = 0; j < i; j++)
            {
                int u = vertices[j];
                if (adj[v].Contains(u))
                    neighborColors.Add(colors[j]);
            }
            
            int color = 1;
            while (neighborColors.Contains(color))
                color++;
            
            colors[i] = color;
        }
        
        return colors;
    }
    
    private void UpdateBest(List<int> clique)
    {
        lock (lockObj)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                clique.CopyTo(bestClique);
            }
        }
    }
    
    private int[] DegeneracyOrdering()
    {
        // Compute core decomposition
        var degree = new int[n];
        for (int i = 0; i < n; i++)
            degree[i] = adj[i].Count;
        
        var maxDegree = degree.Max();
        var buckets = new List<int>[maxDegree + 1];
        for (int i = 0; i <= maxDegree; i++)
            buckets[i] = new List<int>();
        
        var inBucket = new bool[n];
        for (int i = 0; i < n; i++)
        {
            buckets[degree[i]].Add(i);
            inBucket[i] = true;
        }
        
        var ordering = new int[n];
        int idx = 0;
        
        for (int d = 0; d <= maxDegree; d++)
        {
            while (buckets[d].Count > 0)
            {
                int v = buckets[d][buckets[d].Count - 1];
                buckets[d].RemoveAt(buckets[d].Count - 1);
                
                if (!inBucket[v]) continue;
                inBucket[v] = false;
                ordering[idx++] = v;
                
                foreach (int u in adj[v])
                {
                    if (inBucket[u] && degree[u] > d)
                    {
                        int oldDeg = degree[u];
                        degree[u]--;
                        buckets[oldDeg].Remove(u);
                        buckets[degree[u]].Add(u);
                    }
                }
            }
        }
        
        return ordering;
    }
}

            BranchAndBoundColored(clique, newColored);

        var colorClass = new List<int>[vertices.Count + 1];
        for (int i = 0; i <= vertices.Count; i++)
            colorClass[i] = new List<int>();

        for (int i = 0; i < vertices.Count; i++)

            int v = vertices[i];

            bool[] usedColor = new bool[vertices.Count + 1];
            for (int j = 0; j < i; j++)
            {
                if (adj[v].Contains(vertices[j]))
                    usedColor[colors[j]] = true;
            }
            
            while (usedColor[color])
                color++;
            
            colors[i] = color;
            colorClass[color].Add(i);
        }
        
        // Build result - order by color for better pruning
        int maxColor = colors.Max();
        var result = new List<Tuple<int, int>>();
        
        // Process from highest color to lowest
        for (int c = maxColor; c >= 1; c--)
        {
            foreach (int idx in colorClass[c])
            {
                result.Add(Tuple.Create(vertices[idx], c));
            }
        }
        
        return (result, maxColor);

    {

        for (int i = 0; i < n; i++)
            degree[i] = adj[i].Count;

        var bucket = new List<int>[maxDeg + 1];
        for (int i = 0; i <= maxDeg; i++)
            bucket[i] = new List<int>();
        
        var inBucket = new bool[n];
        for (int i = 0; i < n; i++)
        {
            bucket[degree[i]].Add(i);
            inBucket[i] = true;
        }
        
        var order = new int[n];
        int idx = 0;
        
        for (int d = 0; d <= maxDeg; d++)
        {
            while (bucket[d].Count > 0)
            {
                int lastIdx = bucket[d].Count - 1;
                int v = bucket[d][lastIdx];
                bucket[d].RemoveAt(lastIdx);
                
                if (!inBucket[v])
                    continue;
                
                inBucket[v] = false;
                order[idx++] = v;
                
                foreach (int u in adj[v])
                {
                    if (inBucket[u] && degree[u] > d)
                    {
                        int oldDeg = degree[u];
                        degree[u]--;
                        
                        // Move to new bucket
                        bucket[oldDeg].Remove(u);
                        bucket[degree[u]].Add(u);
                    }
                }
            }
        }
        
        return order;
    }

                bucket[d].RemoveFirst();
                position[v] = null;
                
                removed[v] = true;

                        int oldDeg = degree[u];
                        degree[u]--;
                        
                        bucket[oldDeg].Remove(position[u]);
                        position[u] = bucket[degree[u]].AddLast(u);
                    }
                }
            }
        }
        
        return core;
    }

        }
    }

        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestClique[i]);
        }
        Console.WriteLine(sb.ToString());

    }

    private void ReadGraph()
    {
        using var reader = new System.IO.StreamReader(Console.OpenStandardInput());
        
        var header = reader.ReadLine()?.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        if (header == null || header.Length < 2)
        {
            n = 0;
            return;
        }

        n = int.Parse(header[0]);
        int m = int.Parse(header[1]);

        adj = new HashSet<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            var line = reader.ReadLine()?.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (line == null || line.Length < 2) continue;

            int u = int.Parse(line[0]);
            int v = int.Parse(line[1]);

            if (u >= 0 && u < n && v >= 0 && v < n && u != v)
            {
                adj[u].Add(v);
                adj[v].Add(u);
            }
        }

        bestClique = new int[n];
        bestSize = 0;
    }

    private void RunGreedyPhase()
    {
        var tasks = new List<Task>();

        // Various greedy strategies
        tasks.Add(Task.Run(() => GreedySearch(OrderByDegreeDescending())));
        tasks.Add(Task.Run(() => GreedySearch(OrderByDegreeAscending())));
        tasks.Add(Task.Run(() => GreedySearch(OrderByRandom(42))));
        tasks.Add(Task.Run(() => GreedySearch(OrderByRandom(123))));
        tasks.Add(Task.Run(() => GreedySearch(OrderByRandom(456))));
        tasks.Add(Task.Run(() => GreedySearch(OrderByRandom(789))));
        tasks.Add(Task.Run(() => GreedySearch(OrderByRandom(1234))));
        tasks.Add(Task.Run(() => GreedySearch(OrderByCoreNumber())));

        Task.WaitAll(tasks.ToArray());
    }

    private int[] OrderByDegreeDescending()
    {
        return Enumerable.Range(0, n).OrderByDescending(v => adj[v].Count).ToArray();
    }

    private int[] OrderByDegreeAscending()
    {
        return Enumerable.Range(0, n).OrderBy(v => adj[v].Count).ToArray();
    }

    private int[] OrderByRandom(int seed)
    {
        var random = new Random(seed);
        return Enumerable.Range(0, n).OrderBy(_ => random.Next()).ToArray();
    }

    private int[] OrderByCoreNumber()
    {
        var cores = ComputeCoreNumbers();
        return Enumerable.Range(0, n).OrderByDescending(v => cores[v]).ToArray();
    }

    private List<int> GreedySearch(int[] order)
    {
        var clique = new List<int>();

        foreach (int v in order)
        {
            bool canAdd = true;
            foreach (int u in clique)
            {
                if (!adj[v].Contains(u))
                {
                    canAdd = false;
                    break;
                }
            }
            if (canAdd)
                clique.Add(v);
        }

        UpdateBest(clique);
        return clique;
    }

    private void RunExactPhase()
    {
        var cores = ComputeCoreNumbers();
        var order = Enumerable.Range(0, n).OrderBy(v => cores[v]).ToArray();
        var position = new int[n];
        for (int i = 0; i < n; i++)
            position[order[i]] = i;

        Parallel.For(0, n, new ParallelOptions
        {
            MaxDegreeOfParallelism = Environment.ProcessorCount
        }, (i, state) =>
        {
            if (stopwatch.ElapsedMilliseconds > TimeLimitMs)
            {
                state.Break();
                return;
            }

            int v = order[i];
            if (cores[v] + 1 <= bestSize)
                return;

            var clique = new List<int> { v };
            var candidates = new List<int>();

            foreach (int u in adj[v])
            {
                if (position[u] > i && cores[u] + 1 > bestSize)
                    candidates.Add(u);
            }

            BranchAndBound(clique, candidates);
        });
    }

    private void BranchAndBound(List<int> clique, List<int> candidates)
    {
        if (stopwatch.ElapsedMilliseconds > TimeLimitMs)
            return;

        int currentBest = Volatile.Read(ref bestSize);

        if (candidates.Count == 0)
        {
            UpdateBest(clique);
            return;
        }

        // Coloring for upper bound
        var colored = ColorVertices(candidates);
        int maxColor = colored.Count > 0 ? colored[0].Color : 0;

        if (clique.Count + maxColor <= currentBest)
            return;

        for (int i = colored.Count - 1; i >= 0; i--)
        {
            if (stopwatch.ElapsedMilliseconds > TimeLimitMs)
                return;

            if (clique.Count + colored[i].Color <= Volatile.Read(ref bestSize))
                return;

            int v = colored[i].Vertex;
            var newCandidates = new List<int>();

            for (int j = 0; j < i; j++)
            {
                if (adj[v].Contains(colored[j].Vertex))
                    newCandidates.Add(colored[j].Vertex);
            }

            clique.Add(v);
            BranchAndBound(clique, newCandidates);
            clique.RemoveAt(clique.Count - 1);
        }
    }

    private List<ColoredVertex> ColorVertices(List<int> vertices)
    {
        int count = vertices.Count;
        var colors = new int[count];
        var result = new List<ColoredVertex>();

        for (int i = 0; i < count; i++)
        {
            var used = new bool[count + 1];
            int v = vertices[i];

            for (int j = 0; j < i; j++)
            {
                if (adj[v].Contains(vertices[j]))
                    used[colors[j]] = true;
            }

            int color = 1;
            while (used[color])
                color++;

            colors[i] = color;
        }

        for (int i = 0; i < count; i++)
            result.Add(new ColoredVertex { Vertex = vertices[i], Color = colors[i] });

        result.Sort((a, b) => b.Color.CompareTo(a.Color));
        return result;
    }

    private int[] ComputeCoreNumbers()
    {
        var degree = new int[n];
        for (int i = 0; i < n; i++)
            degree[i] = adj[i].Count;

        var core = new int[n];
        var removed = new bool[n];
        int maxDegree = degree.Max();

        if (maxDegree < 0) maxDegree = 0;

        var buckets = new LinkedList<int>[maxDegree + 1];
        var nodes = new LinkedListNode<int>[n];

        for (int d = 0; d <= maxDegree; d++)
            buckets[d] = new LinkedList<int>();

        for (int v = 0; v < n; v++)
        {
            if (degree[v] <= maxDegree)
                nodes[v] = buckets[degree[v]].AddLast(v);
        }

        for (int d = 0; d <= maxDegree; d++)
        {
            while (buckets[d] != null && buckets[d].Count > 0)
            {
                int v = buckets[d].First.Value;
                buckets[d].RemoveFirst();
                nodes[v] = null;

                if (removed[v])
                    continue;

                removed[v] = true;
                core[v] = d;

                foreach (int u in adj[v])
                {
                    if (!removed[u] && degree[u] > d)
                    {
                        int oldDeg = degree[u];
                        degree[u]--;

                        if (oldDeg <= maxDegree && nodes[u] != null)
                        {
                            buckets[oldDeg].Remove(nodes[u]);
                            nodes[u] = buckets[degree[u]].AddLast(u);
                        }
                    }
                }
            }
        }

        return core;
    }

    private void UpdateBest(List<int> clique)
    {
        if (clique.Count <= Volatile.Read(ref bestSize))
            return;

        lock (bestLock)
        {
            if (clique.Count > bestSize)
            {
                bestSize = clique.Count;
                for (int i = 0; i < clique.Count; i++)
                    bestClique[i] = clique[i];
            }
        }
    }

    private void WriteResult()
    {
        var sb = new StringBuilder();
        sb.AppendLine(bestSize.ToString());

        if (bestSize > 0)
        {
            var vertices = new string[bestSize];
            for (int i = 0; i < bestSize; i++)
                vertices[i] = bestClique[i].ToString();
            sb.AppendLine(string.Join(" ", vertices));
        }
        else
        {
            sb.AppendLine("");
        }

        Console.Write(sb.ToString());
    }

    {
        public int Vertex;
        public int Color;
    }
