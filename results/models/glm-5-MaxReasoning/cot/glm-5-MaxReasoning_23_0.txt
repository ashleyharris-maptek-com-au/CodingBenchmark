fn compute_control(state: &State, config: &GameConfig) -> (f64, f64) {
    // Distance to target
    let dx = config.target_x - state.x;
    let dy = config.target_y - state.y;
    let dist = (dx * dx + dy * dy).sqrt();
    
    // Phase determination
    let is_final_approach = dist < 50.0;
    let is_landing = dist < 10.0;
    
    // Calculate desired velocity
    let (desired_vx, desired_vy) = if is_landing {
        // Landing: zero velocity, hover
        (0.0, 0.0)
    } else if is_final_approach {
        // Final approach: slow down, aim for target
        (dx * 0.3, dy * 0.3)
    } else {
        // Transit: move toward target
        let speed = 10.0.min(dist * 0.5);
        (dx / dist * speed, dy / dist * speed)
    };
    
    // Velocity error
    let dvx = desired_vx - state.vx;
    let dvy = desired_vy - state.vy;
    
    // Required thrust to achieve desired velocity change + counteract gravity
    // F = ma, a = dv/dt, but we want to achieve dv over some time
    // For simplicity, thrust direction = (dvx, dvy + gravity_compensation)

    let thrust_world_y = dvy * 0.5 + config.gravity;

    // thrust_world = (sin(angle) * thrust, -cos(angle) * thrust)
    // We want thrust_world_y > 0 to counteract gravity
    // So we need -cos(angle) * thrust > 0, meaning cos(angle) < 0

    // Wait, let me reconsider. The thrust applies in the facing direction.

    // thrust_y = -cos(angle) * thrust = -g → with angle = 0, thrust = g

    // -cos(angle) * thrust = ty

    let angle_error = normalize_angle(desired_angle - state.angle);

                            vx: nums[2], vy: nums[3],
                            angle: nums[4], angular_vel: nums[5],
                            fuel: nums[6],
                        };
                        *state_reader.lock().unwrap() = Some(s);
                    }
                }
            }
        }
    });
    
    // Get config
    let config = config_rx.recv().expect("Failed to receive config");
    
    // Control loop
    let mut stdout = BufWriter::new(stdout.lock());
    
    loop {
        let current = state.lock().unwrap().clone();
        
        if let Some(s) = current {
            let (thrust, turn) = compute_control(&s, &config);
            writeln!(stdout, "{:.6} {:.6}", thrust, turn).unwrap();
            let _ = stdout.flush();
        } else {
            // No state yet, output idle
            writeln!(stdout, "0.0 0.0").unwrap();
            let _ = stdout.flush();
        }
        
        thread::sleep(Duration::from_millis(40));
    }
}

fn compute_control(state: &State, config: &Config) -> (f64, f64) {
    let px = config.target_x - state.x;
    let py = config.target_y - state.y;
    let dist = (px * px + py * py).sqrt();
    
    // Speed limit based on distance
    let max_speed = if dist < 10.0 { 2.0 } else if dist < 50.0 { 5.0 } else { 10.0 };
    
    // Desired velocity toward target
    let des_vx = if dist > 1.0 { (px / dist * max_speed * 0.3).clamp(-max_speed, max_speed) } else { 0.0 };
    let des_vy = if py > 1.0 { (py * 0.2).min(max_speed) } else if py > 0.0 { py * 0.1 } else { py * 0.05 };
    
    // Landing phase adjustment
    let des_vy = if dist < 5.0 { 0.5.min(des_vy) } else { des_vy };
    
    // Velocity error
    let dvx = des_vx - state.vx;
    let dvy = des_vy - state.vy;
    
    // Required acceleration
    let kp = 0.4;
    let ax = dvx * kp;
    let ay = dvy * kp + config.gravity;
    
    // Thrust direction
    let thrust_mag = (ax * ax + ay * ay).sqrt();
    let desired_angle = if thrust_mag > 0.001 { ax.atan2(-ay) } else { 0.0 };
    
    // Angle error
    let mut angle_err = desired_angle - state.angle;
    while angle_err > std::f64::consts::PI { angle_err -= std::f64::consts::TAU; }
    while angle_err < -std::f64::consts::PI { angle_err += std::f64::consts::TAU; }
    
    // Turn with damping
    let turn = (angle_err * 1.5 - state.angular_vel * 0.2).clamp(-1.0, 1.0);
    
    // Thrust
    let thrust = (thrust_mag / config.max_thrust).min(1.0);
    let thrust = if angle_err.abs() < 0.3 { thrust } else if angle_err.abs() < 0.6 { thrust * 0.3 } else { 0.0 };
    
    (thrust, turn)
}
