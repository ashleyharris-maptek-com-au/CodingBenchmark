        for (auto& shelf : shelves) {
            auto& [y, h, cx] = shelf;
            int rw = r.w, rh = r.h;
            
            // Try both orientations
            for (int rot = 0; rot <= 1; rot++) {
                if (rot) swap(rw, rh);
                
                if (rh <= h && cx + rw <= W) {
                    Placement p;
                    p.board = 0; // handled externally
                    p.x = cx;
                    p.y = y;
                    p.rotated = rot;
                    cx += rw;

        int rh = r.h, rw = r.w;

    int W, H;
    vector<pair<int, int>> skyline; // (x, y) - height at position x
    
public:
    Board(int w, int h) : W(w), H(h) {
        skyline = {{0, 0}}; // Initially flat
    }
    
    // Find position for rectangle
    bool place(int rw, int rh, int& out_x, int& out_y) {
        int best_x = -1, best_y = H + 1;
        
        // Try each skyline segment as starting point
        for (size_t i = 0; i < skyline.size(); i++) {
            int x = skyline[i].first;
            if (x + rw > W) continue;
            
            // Find max height in region [x, x+rw)
            int y = 0;
            for (size_t j = i; j < skyline.size() && skyline[j].first < x + rw; j++) {

                best_y = y;

            out_y = best_y;

            return true;
        }
        return false;

            return {true, px, y_start};

        return {false, 0, 0};

// Skyline-based packing (better quality for smaller problems)
class SkylineBoard {
    int W, H;
    vector<pair<int, int>> skyline; // (x, height)
    
public:
    SkylineBoard(int w, int h) : W(w), H(h) {
        skyline.push_back({0, 0});
    }
    
    bool tryPlace(int rw, int rh, int& out_x, int& out_y) {
        if (rw > W || rh > H) return false;
        
        int best_x = -1, best_y = H + 1;
        
        for (size_t i = 0; i < skyline.size(); i++) {
            int x = skyline[i].first;
            if (x + rw > W) continue;
            
            int y = 0;
            for (size_t j = i; j < skyline.size() && skyline[j].first < x + rw; j++) {
                y = max(y, skyline[j].second);
            }
            
            if (y + rh <= H && y < best_y) {
                best_x = x;
                best_y = y;
            }
        }
        
        if (best_x >= 0) {
            out_x = best_x;
            out_y = best_y;
            update(best_x, best_x + rw, best_y + rh);
            return true;
        }
        return false;
    }
    
    void update(int x1, int x2, int new_h) {
        vector<pair<int, int>> new_sky;
        size_t i = 0;
        
        while (i < skyline.size() && skyline[i].first < x1) {
            new_sky.push_back(skyline[i++]);
        }
        
        new_sky.push_back({x1, new_h});
        
        while (i < skyline.size() && skyline[i].first < x2) i++;
        
        int h_after = 0;
        if (i < skyline.size() && skyline[i].first == x2) {
            h_after = skyline[i].second;
        } else if (i > 0) {
            h_after = skyline[i-1].second;
        }
        
        if (h_after != new_h) {
            new_sky.push_back({x2, h_after});
        }
        
        while (i < skyline.size()) {
            new_sky.push_back(skyline[i++]);
        }
        
        skyline.clear();
        for (auto& s : new_sky) {
            if (skyline.empty() || skyline.back().second != s.second) {
                skyline.push_back(s);
            }
        }
    }
};

        int64_t area_b = (int64_t)rw[b] * rh[b];
        return area_a > area_b;
    });
    
    vector<SkylineBoard> boards;
    
    for (int idx : order) {

        for (int rot = 0; rot <= 1 && !placed; rot++) {
