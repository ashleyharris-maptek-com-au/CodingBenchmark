use std::collections::HashSet;

type Point = (f64, f64);

#[derive(Clone, Debug)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Self { vertices }
    }
    
    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        for &(x, y) in &self.vertices {
            min_x = min_x.min(x);
            max_x = max_x.max(x);
            min_y = min_y.min(y);
            max_y = max_y.max(y);
        }
        (min_x, max_x, min_y, max_y)
    }
    
    fn area(&self) -> f64 {
        let n = self.vertices.len();
        if n < 3 { return 0.0; }
        let mut area = 0.0;
        for i in 0..n {
            let (x1, y1) = self.vertices[i];
            let (x2, y2) = self.vertices[(i + 1) % n];
            area += x1 * y2 - x2 * y1;
        }
        area.abs() / 2.0
    }
    
    fn centroid(&self) -> Point {
        let n = self.vertices.len();
        if n == 0 { return (0.0, 0.0); }
        let mut cx = 0.0;
        let mut cy = 0.0;
        for &(x, y) in &self.vertices {
            cx += x;
            cy += y;
        }
        (cx / n as f64, cy / n as f64)
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Self {
        let vertices = self.vertices.iter()
            .map(|&(x, y)| (x + dx, y + dy))
            .collect();
        Self::new(vertices)
    }
    
    fn rotate(&self, angle: f64) -> Self {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices = self.vertices.iter()
            .map(|&(x, y)| (x * cos_a - y * sin_a, x * sin_a + y * cos_a))
            .collect();
        Self::new(vertices)
    }
    
    fn rotate_around(&self, angle: f64, cx: f64, cy: f64) -> Self {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices = self.vertices.iter()
            .map(|&(x, y)| {
                let dx = x - cx;
                let dy = y - cy;
                (cx + dx * cos_a - dy * sin_a, cy + dx * sin_a + dy * cos_a)
            })
            .collect();
        Self::new(vertices)
    }
}

// Check if a point is inside a polygon (ray casting algorithm)
fn point_in_polygon(point: Point, polygon: &[Point]) -> bool {
    let (px, py) = point;
    let n = polygon.len();
    if n < 3 { return false; }
    
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = polygon[i];
        let (xj, yj) = polygon[j];
        
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

// Check if all vertices of poly1 are inside poly2
fn polygon_inside_polygon(poly1: &[Point], poly2: &[Point]) -> bool {
    poly1.iter().all(|&p| point_in_polygon(p, poly2))
}

// Check if two line segments intersect
fn segments_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool {
    let (x1, y1) = p1;
    let (x2, y2) = p2;
    let (x3, y3) = p3;
    let (x4, y4) = p4;
    
    let d1 = direction(p3, p4, p1);
    let d2 = direction(p3, p4, p2);
    let d3 = direction(p1, p2, p3);
    let d4 = direction(p1, p2, p4);
    
    if ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
       ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0)) {
        return true;
    }
    
    if d1.abs() < 1e-10 && on_segment(p3, p1, p4) { return true; }
    if d2.abs() < 1e-10 && on_segment(p3, p2, p4) { return true; }
    if d3.abs() < 1e-10 && on_segment(p1, p3, p2) { return true; }
    if d4.abs() < 1e-10 && on_segment(p1, p4, p2) { return true; }
    
    false
}

fn direction(pi: Point, pj: Point, pk: Point) -> f64 {
    let (xi, yi) = pi;
    let (xj, yj) = pj;
    let (xk, yk) = pk;
    (xk - xi) * (yj - yi) - (xj - xi) * (yk - yi)
}

fn on_segment(pi: Point, pj: Point, pk: Point) -> bool {
    let (xi, yi) = pi;
    let (xj, yj) = pj;
    let (xk, yk) = pk;
    xk.min(xi) <= xj && xj <= xk.max(xi) && yk.min(yi) <= yj && yj <= yk.max(yi)
}

// Check if two polygons overlap
fn polygons_overlap(poly1: &[Point], poly2: &[Point]) -> bool {
    // Check if any edge of poly1 intersects any edge of poly2
    let n1 = poly1.len();
    let n2 = poly2.len();
    
    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(poly1[i], poly1[(i + 1) % n1], 
                                  poly2[j], poly2[(j + 1) % n2]) {
                return true;
            }
        }
    }
    
    // Check if any vertex of poly1 is inside poly2 or vice versa
    if poly1.iter().any(|&p| point_in_polygon(p, poly2)) { return true; }
    if poly2.iter().any(|&p| point_in_polygon(p, poly1)) { return true; }
    
    false
}

fn bounding_boxes_overlap(bb1: (f64, f64, f64, f64), bb2: (f64, f64, f64, f64)) -> bool {
    let (min_x1, max_x1, min_y1, max_y1) = bb1;
    let (min_x2, max_x2, min_y2, max_y2) = bb2;
    
    max_x1 >= min_x2 && max_x2 >= min_x1 && max_y1 >= min_y2 && max_y2 >= min_y1
}

struct PlacedPiece {
    piece_idx: usize,
    transformed_vertices: Vec<Point>,
    bb: (f64, f64, f64, f64),
}

struct Stock {
    placements: Vec<PlacedPiece>,
}

impl Stock {
    fn new() -> Self {
        Self { placements: Vec::new() }
    }
    
    fn can_place(&self, piece: &[Point], piece_bb: (f64, f64, f64, f64)) -> bool {
        for placed in &self.placements {
            if bounding_boxes_overlap(piece_bb, placed.bb) {
                if polygons_overlap(piece, &placed.transformed_vertices) {
                    return false;
                }
            }
        }
        true
    }
    
    fn add_piece(&mut self, piece_idx: usize, transformed_vertices: Vec<Point>, bb: (f64, f64, f64, f64)) {
        self.placements.push(PlacedPiece {
            piece_idx,
            transformed_vertices,
            bb,
        });
    }

use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;

type Point = (f64, f64);

// Cross product of vectors (p1 - p0) and (p2 - p0)
fn cross(p0: Point, p1: Point, p2: Point) -> f64 {
    let (x0, y0) = p0;
    let (x1, y1) = p1;
    let (x2, y2) = p2;
    (x1 - x0) * (y2 - y0) - (x2 - x0) * (y1 - y0)
}

// Check if point p is on segment (a, b)
fn on_segment(a: Point, b: Point, p: Point) -> bool {
    let (ax, ay) = a;
    let (bx, by) = b;
    let (px, py) = p;
    
    px >= ax.min(bx) - 1e-10 && px <= ax.max(bx) + 1e-10 &&
    py >= ay.min(by) - 1e-10 && py <= ay.max(by) + 1e-10
}

// Check if segments (p1, p2) and (p3, p4) intersect
fn segments_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool {
    let d1 = cross(p3, p4, p1);
    let d2 = cross(p3, p4, p2);
    let d3 = cross(p1, p2, p3);
    let d4 = cross(p1, p2, p4);
    
    if ((d1 > 1e-10 && d2 < -1e-10) || (d1 < -1e-10 && d2 > 1e-10)) &&
       ((d3 > 1e-10 && d4 < -1e-10) || (d3 < -1e-10 && d4 > 1e-10)) {
        return true;
    }
    
    if d1.abs() <= 1e-10 && on_segment(p3, p4, p1) { return true; }
    if d2.abs() <= 1e-10 && on_segment(p3, p4, p2) { return true; }
    if d3.abs() <= 1e-10 && on_segment(p1, p2, p3) { return true; }
    if d4.abs() <= 1e-10 && on_segment(p1, p2, p4) { return true; }
    
    false
}

// Check if point is inside polygon using winding number
fn point_in_polygon(p: Point, polygon: &[Point]) -> bool {
    let n = polygon.len();
    if n < 3 { return false; }
    
    let mut winding = 0i32;
    for i in 0..n {
        let p1 = polygon[i];
        let p2 = polygon[(i + 1) % n];
        
        if p1.1 <= p.1 {
            if p2.1 > p.1 && cross(p1, p2, p) > 0.0 {
                winding += 1;
            }
        } else {
            if p2.1 <= p.1 && cross(p1, p2, p) < 0.0 {
                winding -= 1;
            }
        }
    }
    
    winding != 0
}

// Check if all vertices of inner are inside outer and no edges cross
fn polygon_inside_polygon(inner: &[Point], outer: &[Point]) -> bool {
    // All vertices of inner must be inside outer
    for &p in inner {
        if !point_in_polygon(p, outer) {
            return false;
        }
    }
    
    // No edges should cross
    let n_inner = inner.len();
    let n_outer = outer.len();
    
    for i in 0..n_inner {
        for j in 0..n_outer {
            if segments_intersect(inner[i], inner[(i + 1) % n_inner],
                                  outer[j], outer[(j + 1) % n_outer]) {
                return false;
            }
        }
    }
    
    true
}

// Check if two polygons overlap
fn polygons_overlap(poly1: &[Point], poly2: &[Point]) -> bool {
    let n1 = poly1.len();
    let n2 = poly2.len();
    
    // Check edge intersections
    for i in 0..n1 {
        for j in 0..n2 {
            if segments_intersect(poly1[i], poly1[(i + 1) % n1],
                                  poly2[j], poly2[(j + 1) % n2]) {
                return true;
            }
        }
    }
    
    // Check if any vertex is inside the other polygon
    for &p in poly1 {
        if point_in_polygon(p, poly2) { return true; }
    }
    for &p in poly2 {
        if point_in_polygon(p, poly1) { return true; }
    }
    
    false
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<Point>,
    area: f64,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let area = Self::compute_area(&vertices);
        Self { vertices, area }
    }
    
    fn compute_area(vertices: &[Point]) -> f64 {
        let n = vertices.len();
        if n < 3 { return 0.0; }
        let mut area = 0.0;
        for i in 0..n {
            let (x1, y1) = vertices[i];
            let (x2, y2) = vertices[(i + 1) % n];
            area += x1 * y2 - x2 * y1;
        }
        area.abs() / 2.0
    }
    
    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        for &(x, y) in &self.vertices {
            min_x = min_x.min(x);
            max_x = max_x.max(x);
            min_y = min_y.min(y);
            max_y = max_y.max(y);
        }
        (min_x, max_x, min_y, max_y)
    }
    
    fn centroid(&self) -> Point {
        let n = self.vertices.len();
        if n == 0 { return (0.0, 0.0); }
        let mut cx = 0.0;
        let mut cy = 0.0;
        for &(x, y) in &self.vertices {
            cx += x;
            cy += y;
        }
        (cx / n as f64, cy / n as f64)
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let vertices: Vec<Point> = self.vertices.iter()
            .map(|&(x, y)| (x + dx, y + dy))
            .collect();
        Polygon::new(vertices)
    }
    
    fn rotate_around(&self, angle: f64, cx: f64, cy: f64) -> Polygon {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let vertices: Vec<Point> = self.vertices.iter()
            .map(|&(x, y)| {
                let dx = x - cx;
                let dy = y - cy;
                (cx + dx * cos_a - dy * sin_a, cy + dx * sin_a + dy * cos_a)
            })
            .collect();
        Polygon::new(vertices)
    }
    
    fn rotate(&self, angle: f64) -> Polygon {
        let (cx, cy) = self.centroid();
        self.rotate_around(angle, cx, cy)
    }
}

use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;

type Point = (f64, f64);

fn cross(o: Point, a: Point, b: Point) -> f64 {
    (a.0 - o.0) * (b.1 - o.1) - (a.1 - o.1) * (b.0 - o.0)
}

fn point_in_polygon(p: Point, poly: &[Point]) -> bool {
    let n = poly.len();
    if n < 3 { return false; }
    
    let mut inside = false;
    let mut j = n - 1;
    
    for i in 0..n {
        if ((poly[i].1 > p.1) != (poly[j].1 > p.1)) &&
           (p.0 < (poly[j].0 - poly[i].0) * (p.1 - poly[i].1) / (poly[j].1 - poly[i].1) + poly[i].0) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn seg_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool {
    let d1 = cross(p3, p4, p1);
    let d2 = cross(p3, p4, p2);
    let d3 = cross(p1, p2, p3);
    let d4 = cross(p1, p2, p4);
    
    if ((d1 > 1e-9 && d2 < -1e-9) || (d1 < -1e-9 && d2 > 1e-9)) &&
       ((d3 > 1e-9 && d4 < -1e-9) || (d3 < -1e-9 && d4 > 1e-9)) {
        return true;
    }
    
    let on_seg = |a: Point, b: Point, p: Point| -> bool {
        p.0 >= a.0.min(b.0) - 1e-9 && p.0 <= a.0.max(b.0) + 1e-9 &&
        p.1 >= a.1.min(b.1) - 1e-9 && p.1 <= a.1.max(b.1) + 1e-9
    };
    
    d1.abs() <= 1e-9 && on_seg(p3, p4, p1) ||
    d2.abs() <= 1e-9 && on_seg(p3, p4, p2) ||
    d3.abs() <= 1e-9 && on_seg(p1, p2, p3) ||
    d4.abs() <= 1e-9 && on_seg(p1, p2, p4)
}

fn polygons_overlap(p1: &[Point], p2: &[Point]) -> bool {
    let n1 = p1.len();
    let n2 = p2.len();
    
    for i in 0..n1 {
        for j in 0..n2 {
            if seg_intersect(p1[i], p1[(i+1)%n1], p2[j], p2[(j+1)%n2]) {
                return true;
            }
        }
    }
    
    p1.iter().any(|&p| point_in_polygon(p, p2)) ||
    p2.iter().any(|&p| point_in_polygon(p, p1))
}

#[derive(Clone)]
struct Polygon {
    v: Vec<Point>,
    area: f64,
}

impl Polygon {
    fn new(v: Vec<Point>) -> Self {
        let area = Self::calc_area(&v);
        Self { v, area }
    }
    
    fn calc_area(v: &[Point]) -> f64 {
        let n = v.len();
        if n < 3 { return 0.0; }
        let mut a = 0.0;
        for i in 0..n {
            a += v[i].0 * v[(i+1)%n].1 - v[(i+1)%n].0 * v[i].1;
        }
        a.abs() / 2.0
    }
    
    fn bb(&self) -> (f64, f64, f64, f64) {
        let (mut mix, mut mx, mut miy, mut my) = (f64::MAX, f64::MIN, f64::MAX, f64::MIN);
        for &(x, y) in &self.v {
            mix = mix.min(x); mx = mx.max(x);
            miy = miy.min(y); my = my.max(y);
        }
        (mix, mx, miy, my)
    }
    
    fn centroid(&self) -> Point {
        if self.v.is_empty() { return (0.0, 0.0); }
        let (mut x, mut y) = (0.0, 0.0);
        for &(px, py) in &self.v { x += px; y += py; }
        (x / self.v.len() as f64, y / self.v.len() as f64)
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Self {
        Self::new(self.v.iter().map(|&(x, y)| (x + dx, y + dy)).collect())
    }
    
    fn rotate(&self, angle: f64) -> Self {
        let (cx, cy) = self.centroid();
        let (c, s) = (angle.cos(), angle.sin());
        Self::new(self.v.iter().map(|&(x, y)| {
            let (dx, dy) = (x - cx, y - cy);
            (cx + dx * c - dy * s, cy + dx * s + dy * c)
        }).collect())
    }
}

struct Stock {
    placed: Vec<(Vec<Point>, (f64, f64, f64, f64))>,
}

impl Stock {
    fn new() -> Self { Self { placed: Vec::new() } }
    
    fn can_add(&self, poly: &[Point], bb: (f64, f64, f64, f64)) -> bool {
        for (p, b) in &self.placed {
            if bb.1 >= b.0 && bb.0 <= b.1 && bb.3 >= b.2 && bb.2 <= b.3 {
                if polygons_overlap(poly, p) { return false; }
            }
        }
        true
    }
    
    fn add(&mut self, poly: Vec<Point>, bb: (f64, f64, f64, f64)) {
        self.placed.push((poly, bb));
    }
}
