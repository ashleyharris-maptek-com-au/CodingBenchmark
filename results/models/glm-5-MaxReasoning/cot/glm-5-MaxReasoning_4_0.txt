// 3D Vector
struct Vec3 {
    double x, y, z;
    Vec3(double x = 0, double y = 0, double z = 0) : x(x), y(y), z(z) {}
    
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    double dot(const Vec3& o) const { return x * o.x + y * o.y + z * o.z; }
    Vec3 cross(const Vec3& o) const { 
        return {y * o.z - z * o.y, z * o.x - x * o.z, x * o.y - y * o.x}; 
    }
    double len() const { return sqrt(x * x + y * y + z * z); }
    double lenSq() const { return x * x + y * y + z * z; }
    Vec3 normalized() const { 
        double l = len(); 
        return l > 1e-12 ? *this * (1.0 / l) : Vec3(); 
    }
};

// 3x3 Matrix
struct Mat3 {
    double m[9];
    Mat3() : m{1,0,0,0,1,0,0,0,1} {}
    Mat3(std::initializer_list<double> l) { copy(l.begin(), l.end(), m); }
    
    static Mat3 rotX(double a) {
        double c = cos(a), s = sin(a);
        return {1,0,0, 0,c,-s, 0,s,c};
    }
    static Mat3 rotY(double a) {
        double c = cos(a), s = sin(a);
        return {c,0,s, 0,1,0, -s,0,c};
    }
    static Mat3 rotZ(double a) {
        double c = cos(a), s = sin(a);
        return {c,-s,0, s,c,0, 0,0,1};
    }
    static Mat3 euler(double rx, double ry, double rz) {
        return rotZ(rz) * rotY(ry) * rotX(rx);
    }
    
    Mat3 operator*(const Mat3& o) const {
        Mat3 r;
        for (int i = 0; i < 3; i++)
            for (int j = 0; j < 3; j++) {
                r.m[i*3+j] = 0;
                for (int k = 0; k < 3; k++)
                    r.m[i*3+j] += m[i*3+k] * o.m[k*3+j];
            }
        return r;
    }
    
    Vec3 operator*(const Vec3& v) const {
        return {m[0]*v.x + m[1]*v.y + m[2]*v.z,
                m[3]*v.x + m[4]*v.y + m[5]*v.z,
                m[6]*v.x + m[7]*v.y + m[8]*v.z};
    }
};

// Container (convex polyhedron)
struct Container {
    vector<Vec3> verts;
    vector<pair<Vec3, double>> faces; // (normal, d) for plane equation
    Vec3 bmin, bmax;
    
    void init(const vector<Vec3>& v, double scale) {
        verts = v;
        for (auto& p : verts) p = p * scale; // Scale if needed
        
        bmin = {1e100, 1e100, 1e100};
        bmax = {-1e100, -1e100, -1e100};
        for (const auto& p : verts) {
            bmin.x = min(bmin.x, p.x); bmax.x = max(bmax.x, p.x);
            bmin.y = min(bmin.y, p.y); bmax.y = max(bmax.y, p.y);
            bmin.z = min(bmin.z, p.z); bmax.z = max(bmax.z, p.z);
        }
        buildFaces();
    }
    
    void buildFaces() {
        int n = verts.size();
        for (int i = 0; i < n; i++) {
            for (int j = i+1; j < n; j++) {
                for (int k = j+1; k < n; k++) {
                    Vec3 e1 = verts[j] - verts[i];
                    Vec3 e2 = verts[k] - verts[i];
                    Vec3 normal = e1.cross(e2);
                    if (normal.lenSq() < 1e-18) continue;
                    
                    int sign = 0;
                    bool valid = true;
                    for (int l = 0; l < n; l++) {
                        if (l == i || l == j || l == k) continue;
                        double d = normal.dot(verts[l] - verts[i]);
                        if (abs(d) < 1e-12) continue;
                        int s = d > 0 ? 1 : -1;
                        if (sign == 0) sign = s;
                        else if (s != sign) { valid = false; break; }
                    }
                    if (valid && sign != 0) {
                        Vec3 n = normal.normalized();
                        if (sign > 0) n = n * -1;
                        faces.push_back({n, -n.dot(verts[i])});
                    }
                }
            }
        }
    }
    
    bool contains(const Vec3& p) const {
        for (const auto& [n, d] : faces) {
            if (n.dot(p) + d > 1e-10) return false;
        }
        return true;
    }
    
    bool containsAll(const array<Vec3, 4>& pts) const {
        for (const auto& p : pts) if (!contains(p)) return false;
        return true;
    }
    
    double volume() const {
        return (bmax.x - bmin.x) * (bmax.y - bmin.y) * (bmax.z - bmin.z);
    }
};

// Tetrahedron packer
class Packer {
    Container cont;
    double edgeLen;
    double tetR; // circumradius
    array<Vec3, 4> baseVerts;
    
    vector<tuple<Vec3, Vec3, array<Vec3, 4>>> placed; // center, rotation, vertices
    
    bool checkOverlap(const array<Vec3, 4>& a, const array<Vec3, 4>& b) const {
        // Bounding sphere check
        Vec3 ca{}, cb{};
        for (int i = 0; i < 4; i++) { ca = ca + a[i]; cb = cb + b[i]; }
        ca = ca * 0.25; cb = cb * 0.25;
        
        double dist = (ca - cb).len();
        if (dist > 2 * tetR * 1.1) return false; // Quick rejection
        
        // Check minimum vertex-vertex distance
        const double minDist = edgeLen * 0.02;
        const double minDistSq = minDist * minDist;
        
        for (int i = 0; i < 4; i++) {
            for (int j = 0; j < 4; j++) {
                if ((a[i] - b[j]).lenSq() < minDistSq) return true;
            }
        }
        
        // Center distance check
        if (dist < tetR * 0.9) return true;
        
        return false;
    }
    
    bool canPlace(const Vec3& c, const array<Vec3, 4>& v) const {
        if (!cont.containsAll(v)) return false;
        
        for (const auto& [pc, pr, pv] : placed) {
            if ((c - pc).len() < tetR * 3) {
                if (checkOverlap(v, pv)) return false;
            }
        }
        return true;
    }
    
public:
    void init(const Container& c, double e) {
        cont = c;
        edgeLen = e;
        tetR = e * sqrt(3.0 / 8.0);
        
        // Base tetrahedron vertices (edge length 1, centered at origin)
        double r = 1.0 / (2.0 * sqrt(2.0));
        baseVerts = {{
            {r, r, r}, {r, -r, -r}, {-r, r, -r}, {-r, -r, r}
        }};
        // Scale by edge length
        for (auto& v : baseVerts) v = v * e;
    }
    
    void pack() {
        placed.clear();
        
        double vol = cont.volume();
        
        // Adaptive parameters
        double step = edgeLen * 0.82;
        int rotPerAxis = vol > 1e8 ? 2 : vol > 1e5 ? 3 : 4;
        int attempts = vol > 1e8 ? 2 : 3;
        
        // Precompute rotations
        vector<Vec3> rots;
        for (int i = 0; i < rotPerAxis; i++)
            for (int j = 0; j < rotPerAxis; j++)
                for (int k = 0; k < rotPerAxis; k++)
                    rots.push_back({
                        i * PI / rotPerAxis,
                        j * PI / rotPerAxis,
                        k * 2.0 * PI / rotPerAxis
                    });
        
        mt19937 rng(12345);
        uniform_int_distribution<size_t> rotIdx(0, rots.size() - 1);
        
        double margin = tetR * 1.1;
        double x0 = cont.bmin.x + margin, x1 = cont.bmax.x - margin;
        double y0 = cont.bmin.y + margin, y1 = cont.bmax.y - margin;
        double z0 = cont.bmin.z + margin, z1 = cont.bmax.z - margin;
        
        for (double x = x0; x <= x1; x += step) {
            for (double y = y0; y <= y1; y += step) {
                for (double z = z0; z <= z1; z += step) {
                    Vec3 center(x, y, z);
                    
                    for (int t = 0; t < attempts; t++) {
                        Vec3 rot = rots[rotIdx(rng)];
                        Mat3 R = Mat3::euler(rot.x, rot.y, rot.z);
                        
                        array<Vec3, 4> verts;
                        for (int i = 0; i < 4; i++)
                            verts[i] = center + R * baseVerts[i];
                        
                        if (canPlace(center, verts)) {
                            placed.push_back({center, rot, verts});
                            break;
                        }
                    }
                }
            }
        }
    }
    
    void output() {
        cout << placed.size() << "\n";
        cout << fixed << setprecision(6);
        for (const auto& [c, r, v] : placed) {
            cout << c.x << " " << c.y << " " << c.z << " "
                 << r.x << " " << r.y << " " << r.z << "\n";
        }
    }
};
