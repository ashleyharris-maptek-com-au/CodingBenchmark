    private List<Vector3> vertices;

    private BoundingBox polyBBox;

        placements = Pack(scale);

        OutputResults();

    private void ReadInput()
    {
        var firstLine = Console.ReadLine().Split();
        int V = int.Parse(firstLine[0]);

        }
        
        faces = new List<int[]>(F);
        for (int i = 0; i < F; i++)
        {
            var parts = Console.ReadLine().Split();
            faces.Add(new int[] { int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]) });
        }
        
        var containerParts = Console.ReadLine().Split();
        containerW = double.Parse(containerParts[0]);
        containerH = double.Parse(containerParts[1]);
        containerD = double.Parse(containerParts[2]);
    }
    
    private void GenerateRotations()
    {
        rotations = new List<Quaternion>();
        
        // 24 rotations of a cube (rotations that preserve axis-alignment)
        // These are the rotations by 0, 90, 180, 270 degrees around each axis
        
        // Generate all unique rotations
        var rotationSet = new HashSet<string>();
        
        for (int i = 0; i < 4; i++) // X rotations
        {
            for (int j = 0; j < 4; j++) // Y rotations
            {
                for (int k = 0; k < 4; k++) // Z rotations
                {
                    var rx = Quaternion.FromAxisAngle(new Vector3(1, 0, 0), i * Math.PI / 2);
                    var ry = Quaternion.FromAxisAngle(new Vector3(0, 1, 0), j * Math.PI / 2);
                    var rz = Quaternion.FromAxisAngle(new Vector3(0, 0, 1), k * Math.PI / 2);
                    var q = rz * ry * rx;
                    
                    // Normalize
                    double len = Math.Sqrt(q.W * q.W + q.X * q.X + q.Y * q.Y + q.Z * q.Z);
                    q = new Quaternion(q.W / len, q.X / len, q.Y / len, q.Z / len);
                    
                    // Round for deduplication
                    string key = $"{Math.Round(q.W, 4)}_{Math.Round(q.X, 4)}_{Math.Round(q.Y, 4)}_{Math.Round(q.Z, 4)}";
                    if (!rotationSet.Contains(key))
                    {
                        rotationSet.Add(key);
                        rotations.Add(q);
                    }
                }
            }
        }
    }
    
    private void ComputePolyhedronProperties()
    {
        // Compute bounding box
        double minX = double.MaxValue, maxX = double.MinValue;
        double minY = double.MaxValue, maxY = double.MinValue;
        double minZ = double.MaxValue, maxZ = double.MinValue;
        
        foreach (var v in vertices)
        {
            minX = Math.Min(minX, v.X);
            maxX = Math.Max(maxX, v.X);
            minY = Math.Min(minY, v.Y);
            maxY = Math.Max(maxY, v.Y);
            minZ = Math.Min(minZ, v.Z);
            maxZ = Math.Max(maxZ, v.Z);
        }
        
        polyBBox = new BoundingBox(minX, maxX, minY, maxY, minZ, maxZ);
        polyCentroid = new Vector3((minX + maxX) / 2, (minY + maxY) / 2, (minZ + maxZ) / 2);
        
        // Estimate volume (using bounding box for now)
        polyVolume = (maxX - minX) * (maxY - minY) * (maxZ - minZ);
    }
    
    private ProblemScale DetermineScale()
    {
        double containerVolume = containerW * containerH * containerD;
        double ratio = containerVolume / Math.Max(polyVolume, 1e-10);
        
        if (containerW <= 10 && containerH <= 10 && containerD <= 10)
            return ProblemScale.Trivial;
        if (containerW <= 30 && containerH <= 30 && containerD <= 30)
            return ProblemScale.Small;
        if (containerW <= 100 && containerH <= 100 && containerD <= 100)
            return ProblemScale.Medium;
        if (containerW <= 300 && containerH <= 300 && containerD <= 300)
            return ProblemScale.Large;
        return ProblemScale.Extreme;
    }
    
    private List<Placement> Pack(ProblemScale scale)
    {
        var result = new List<Placement>();
        var occupied = new List<(BoundingBox bbox, List<Vector3> verts)>();
        
        // Get the best rotation (the one that minimizes bounding box)
        var bestRotation = GetBestRotation();
        var rotatedVerts = RotateVertices(vertices, bestRotation);
        var rotatedBBox = ComputeBoundingBox(rotatedVerts);
        
        // Grid-based placement
        double stepX = (rotatedBBox.MaxX - rotatedBBox.MinX) * 1.05; // Small gap
        double stepY = (rotatedBBox.MaxY - rotatedBBox.MinY) * 1.05;
        double stepZ = (rotatedBBox.MaxZ - rotatedBBox.MinZ) * 1.05;
        
        // Adjust step based on scale
        if (scale >= ProblemScale.Large)
        {
            stepX = (rotatedBBox.MaxX - rotatedBBox.MinX) * 1.02;
            stepY = (rotatedBBox.MaxY - rotatedBBox.MinY) * 1.02;
            stepZ = (rotatedBBox.MaxZ - rotatedBBox.MinZ) * 1.02;
        }
        
        // Place polyhedra in a grid pattern
        for (double z = 0; z + stepZ <= containerD; z += stepZ)
        {
            for (double y = 0; y + stepY <= containerH; y += stepY)
            {
                for (double x = 0; x + stepX <= containerW; x += stepX)
                {
                    // Translate rotated vertices
                    var translatedVerts = new List<Vector3>();
                    foreach (var v in rotatedVerts)
                    {
                        translatedVerts.Add(new Vector3(v.X + x, v.Y + y, v.Z + z));
                    }
                    
                    // Check if all vertices are inside container
                    if (!IsInsideContainer(translatedVerts))
                        continue;
                    
                    // Check for collisions with existing placements
                    bool collision = false;
                    foreach (var existing in occupied)
                    {
                        if (BoundingBoxesOverlap(existing.bbox, ComputeBoundingBox(translatedVerts)))
                        {
                            collision = true;
                            break;
                        }
                    }
                    
                    if (!collision)
                    {
                        var bbox = ComputeBoundingBox(translatedVerts);
                        occupied.Add((bbox, translatedVerts));
                        result.Add(new Placement
                        {
                            Translation = new Vector3(x - rotatedBBox.MinX, y - rotatedBBox.MinY, z - rotatedBBox.MinZ),
                            Rotation = bestRotation
                        });
                    }
                }
            }
        }
        
        return result;
    }
    
    private Quaternion GetBestRotation()
    {
        Quaternion bestRot = Quaternion.Identity;
        double minVolume = double.MaxValue;
        
        foreach (var rot in rotations)
        {
            var rotatedVerts = RotateVertices(vertices, rot);
            var bbox = ComputeBoundingBox(rotatedVerts);
            double volume = (bbox.MaxX - bbox.MinX) * (bbox.MaxY - bbox.MinY) * (bbox.MaxZ - bbox.MinZ);
            
            if (volume < minVolume)
            {
                minVolume = volume;
                bestRot = rot;
            }
        }
        
        return bestRot;
    }
    
    private List<Vector3> RotateVertices(List<Vector3> verts, Quaternion rot)
    {
        var result = new List<Vector3>(verts.Count);
        foreach (var v in verts)
        {
            result.Add(rot.Rotate(v));
        }
        return result;
    }
    
    private BoundingBox ComputeBoundingBox(List<Vector3> verts)
    {
        double minX = double.MaxValue, maxX = double.MinValue;
        double minY = double.MaxValue, maxY = double.MinValue;
        double minZ = double.MaxValue, maxZ = double.MinValue;
        
        foreach (var v in verts)
        {
            minX = Math.Min(minX, v.X);
            maxX = Math.Max(maxX, v.X);
            minY = Math.Min(minY, v.Y);
            maxY = Math.Max(maxY, v.Y);
            minZ = Math.Min(minZ, v.Z);
            maxZ = Math.Max(maxZ, v.Z);
        }
        
        return new BoundingBox(minX, maxX, minY, maxY, minZ, maxZ);
    }
    
    private bool IsInsideContainer(List<Vector3> verts)
    {
        foreach (var v in verts)
        {
            if (v.X < 0 || v.X > containerW ||
                v.Y < 0 || v.Y > containerH ||
                v.Z < 0 || v.Z > containerD)
            {
                return false;
            }
        }
        return true;
    }
    
    private bool BoundingBoxesOverlap(BoundingBox a, BoundingBox b)
    {
        return !(a.MaxX <= b.MinX || b.MaxX <= a.MinX ||
                 a.MaxY <= b.MinY || b.MaxY <= a.MinY ||
                 a.MaxZ <= b.MinZ || b.MaxZ <= a.MinZ);
    }
    
    private void OutputResults()
    {
        Console.WriteLine(placements.Count);
        foreach (var p in placements)
        {
            Console.WriteLine($"{p.Translation.X:F6} {p.Translation.Y:F6} {p.Translation.Z:F6} {p.Rotation.W:F6} {p.Rotation.X:F6} {p.Rotation.Y:F6} {p.Rotation.Z:F6}");
        }
    }
