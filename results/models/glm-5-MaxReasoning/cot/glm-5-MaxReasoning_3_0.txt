    if n == 0 {

    if n == 1 {

            Node {

                y: angle.sin(),
                dx: 0.0,
                dy: 0.0,

        (100, 0.3)
    } else if n <= 5000 {
        (50, 0.2)

        (20, 0.1)

    let area = (n as f64).sqrt();
    let k = area / (n as f64).sqrt(); // optimal distance

    for iter in 0..iterations {

        for node in &mut nodes {
            node.dx = 0.0;
            node.dy = 0.0;
        }

        if n <= 1000 {
