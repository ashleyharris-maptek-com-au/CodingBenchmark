```rust
use std::io::{self, BufReader, BufWriter, Read, Write, BufRead};

const BATCH_SIZE: usize = 100_000;
const MAX_IN_MEMORY_POINTS: usize = 50_000_000;
const MAX_STORED_ASSIGNMENTS: usize = 250_000_000;

#[derive(Clone, Copy, Default, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline]
    fn read_batch<R: Read>(reader: &mut R, buffer: &mut Vec<Point>, count: usize) -> io::Result<usize> {
        buffer.clear();
        buffer.reserve(count);
        
        let mut bytes = vec![0u8; count * 12];
        let bytes_read = reader.read(&mut bytes)?;
        let num_points = bytes_read / 12;
        
        for i in 0..num_points {
            let offset = i * 12;
            buffer.push(Point {
                x: f32::from_le_bytes([bytes[offset], bytes[offset+1], bytes[offset+2], bytes[offset+3]]),
                y: f32::from_le_bytes([bytes[offset+4], bytes[offset+5], bytes[offset+6], bytes[offset+7]]),
                z: f32::from_le_bytes([bytes[offset+8], bytes[offset+9], bytes[offset+10], bytes[offset+11]]),
            });
        }
        
        Ok(num_points)
    }
    
    #[inline]
    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
    
    #[inline]
    fn nearest_centroid(&self, centroids: &[Point]) -> usize {
        let mut min_dist = f32::MAX;
        let mut min_idx = 0;
        for (i, c) in centroids.iter().enumerate() {
            let d = self.distance_squared(c);
            if d < min_dist {
                min_dist = d;
                min_idx = i;
            }
        }
        min_idx
    }
}

struct CentroidState {
    centroid: Point,
    sum: Point,
    count: u64,
}

impl CentroidState {
    fn new(p: Point) -> Self {
        CentroidState {
            centroid: p,
            sum: Point { x: 0.0, y: 0.0, z: 0.0 },
            count: 0,
        }
    }
    
    fn add_point(&mut self, p: &Point) {
        self.sum.x += p.x;
        self.sum.y += p.y;
        self.sum.z += p.z;
        self.count += 1;
    }
    
    fn update(&mut self, learning_rate: f32) {
        if self.count > 0 {
            let avg_x = self.sum.x / self.count as f32;
            let avg_y = self.sum.y / self.count as f32;
            let avg_z = self.sum.z / self.count as f32;
            
            self.centroid.x = self.centroid.x * (1.0 - learning_rate) + avg_x * learning_rate;
            self.centroid.y = self.centroid.y * (1.0 - learning_rate) + avg_y * learning_rate;
            self.centroid.z = self.centroid.z * (1.0 - learning_rate) + avg_z * learning_rate;
        }
        self.sum = Point { x: 0.0, y: 0.0, z: 0.0 };
        self.count = 0;
    }
}

fn standard_kmeans<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    num_points: usize,
    num_clusters: usize,
) {
    // Load all points
    let mut points = Vec::with_capacity(num_points);
    let mut buffer = Vec::new();
    let mut remaining = num_points;
    
    while remaining > 0 {
        let to_read = remaining.min(BATCH_SIZE);
        let count = Point::read_batch(reader, &mut buffer, to_read).unwrap();
        points.extend_from_slice(&buffer[..count]);
        remaining -= count;
        if count == 0 { break; }
    }
    
    // Initialize centroids using k-means++
    let mut centroids = initialize_centroids_kmeanspp(&points, num_clusters);
    
    // Allocate assignments
    let mut assignments = vec![0u32; num_points];
    
    // K-means iterations
    let max_iterations = 100;
    for _ in 0..max_iterations {
        // Assign points
        let mut changed = false;
        for (i, p) in points.iter().enumerate() {
            let new_assignment = p.nearest_centroid(&centroids) as u32;
            if assignments[i] != new_assignment {
                assignments[i] = new_assignment;
                changed = true;
            }
        }
        
        if !changed {
            break;
        }
        
        // Update centroids
        let mut sums = vec![Point::default(); num_clusters];
        let mut counts = vec![0u64; num_clusters];
        
        for (i, p) in points.iter().enumerate() {
            let c = assignments[i] as usize;
            sums[c].x += p.x;
            sums[c].y += p.y;
            sums[c].z += p.z;
            counts[c] += 1;
        }
        
        for (i, c) in centroids.iter_mut().enumerate() {
            if counts[i] > 0 {
                c.x = sums[i].x / counts[i] as f32;
                c.y = sums[i].y / counts[i] as f32;
                c.z = sums[i].z / counts[i] as f32;
            }
        }
    }
    
    // Write output
    for &a in &assignments {
        writer.write_all(&a.to_le_bytes()).unwrap();
    }
}

fn streaming_minibatch_kmeans<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,

    num_clusters: usize,

    let mut assignments = Vec::new();

    let mut batch_count = 0u64;
    let mut points_in_batch = Vec::new();

            for state in &mut centroid_states {
                state.update(lr);
            }

        }

        remaining -= count as u64;
        
        if count == 0 { break; }
        
        for p in &buffer[..count] {
            let centroids: Vec<Point> = centroid_states.iter().map(|s| s.centroid).collect();
            let idx = p.nearest_centroid(&centroids);
            centroid_states[idx].add_point(p);
            points_in_batch.push(*p);
            assignments.push(idx as u32);
            
            if points_in_batch.len() >= minibatch_size {
                batch_count += 1;
                let lr = 1.0 / batch_count as f32;
                for state in &mut centroid_states {
                    state.update(lr);
                }
                points_in_batch.clear();
            }
        }

        let lr = 1.0 / batch_count as f32;

            state.update(lr);
        }

fn initialize_centroids_kmeanspp(points: &[Point], k: usize) -> Vec<Point> {
    let n = points.len();
    if n == 0 || k == 0 {
        return Vec::new();
    }
    
    let mut centroids = Vec::with_capacity(k);
    let mut rng = rand::thread_rng();
    
    // First centroid: random point
    let first_idx = rand::Rng::gen_range(&mut rng, 0..n);
    centroids.push(points[first_idx]);
    
    let mut distances = vec![f32::MAX; n];
    
    for _ in 1..k {
        // Update distances
        let last = centroids.last().unwrap();
        let mut total_dist = 0.0f64;
        
        for (i, p) in points.iter().enumerate() {
            let d = p.distance_squared(last);
            distances[i] = distances[i].min(d);
            total_dist += distances[i] as f64;
        }
        
        // Choose next centroid with probability proportional to distance
        let threshold = rand::Rng::gen::<f64>(&mut rng) * total_dist;
        let mut cumsum = 0.0;
        let mut next_idx = 0;
        
        for (i, &d) in distances.iter().enumerate() {
            cumsum += d as f64;
            if cumsum >= threshold {
                next_idx = i;
                break;
            }
        }
        
        centroids.push(points[next_idx]);
    }
    
    centroids
}

fn initialize_centroids_from_batch(batch: &[Point], k: usize) -> Vec<CentroidState> {

        // K-means++ initialization on the batch
        initialize_centroids_kmeanspp(batch, k)
    };
    
    centroids.into_iter().map(|p| CentroidState::new(p)).collect()
}
```

- Assignments: 10B * 4 bytes = 40GB (can't store)
- Centroids: 10000 * 12 bytes = 120KB (can store)

So we can only store centroids, not points or assignments.

```rust
fn streaming_minibatch_kmeans<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    num_points: u64,
    num_clusters: usize,
) {
    let mut buffer = Vec::new();
    let mut remaining = num_points;
    
    // Initialize centroids from first batch
    let init_batch_size = (num_clusters * 10).min(BATCH_SIZE);
    let _ = Point::read_batch(reader, &mut buffer, init_batch_size);
    remaining -= buffer.len() as u64;
    
    let mut centroids = initialize_centroids_from_batch_simple(&buffer, num_clusters);
    let mut counts = vec![0u64; num_clusters];
    
    // Process first batch
    for p in &buffer {
        let idx = p.nearest_centroid(&centroids);
        counts[idx] += 1;
        let lr = 1.0 / counts[idx] as f32;
        centroids[idx].x += lr * (p.x - centroids[idx].x);
        centroids[idx].y += lr * (p.y - centroids[idx].y);
        centroids[idx].z += lr * (p.z - centroids[idx].z);
        
        // Output assignment immediately
        writer.write_all(&(idx as u32).to_le_bytes()).unwrap();
    }
    
    // Continue streaming
    while remaining > 0 {
        let to_read = (remaining as usize).min(BATCH_SIZE);
        let count = Point::read_batch(reader, &mut buffer, to_read).unwrap();
        remaining -= count as u64;
        
        if count == 0 { break; }
        
        for p in &buffer[..count] {
            let idx = p.nearest_centroid(&centroids);
            counts[idx] += 1;
            let lr = 1.0 / counts[idx] as f32;
            centroids[idx].x += lr * (p.x - centroids[idx].x);
            centroids[idx].y += lr * (p.y - centroids[idx].y);
            centroids[idx].z += lr * (p.z - centroids[idx].z);
            
            writer.write_all(&(idx as u32).to_le_bytes()).unwrap();
        }
    }
}
```

```rust
use std::io::{self, BufReader, BufWriter, Read, Write, BufRead};

// Constants for adaptive behavior
const BATCH_SIZE: usize = 1_000_000;
const MAX_IN_MEMORY_POINTS: usize = 50_000_000;
const MAX_ITERATIONS: usize = 100;

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn read_batch<R: Read>(reader: &mut R, buffer: &mut Vec<Point>, count: usize) -> io::Result<usize> {
        buffer.clear();
        let byte_count = count * 12;
        let mut bytes = vec![0u8; byte_count];
        
        match reader.read_exact(&mut bytes) {
            Ok(()) => {
                for i in 0..count {
                    let offset = i * 12;
                    buffer.push(Point {
                        x: f32::from_le_bytes(bytes[offset..offset+4].try_into().unwrap()),
                        y: f32::from_le_bytes(bytes[offset+4..offset+8].try_into().unwrap()),
                        z: f32::from_le_bytes(bytes[offset+8..offset+12].try_into().unwrap()),
                    });
                }
                Ok(count)
            }
            Err(e) if e.kind() == io::ErrorKind::UnexpectedEof => {
                // Partial read
                let mut bytes = vec![0u8; byte_count];
                let n = reader.read(&mut bytes)?;
                let num_points = n / 12;
                for i in 0..num_points {
                    let offset = i * 12;
                    buffer.push(Point {
                        x: f32::from_le_bytes([bytes[offset], bytes[offset+1], bytes[offset+2], bytes[offset+3]]),
                        y: f32::from_le_bytes([bytes[offset+4], bytes[offset+5], bytes[offset+6], bytes[offset+7]]),
                        z: f32::from_le_bytes([bytes[offset+8], bytes[offset+9], bytes[offset+10], bytes[offset+11]]),
                    });
                }
                Ok(num_points)
            }
            Err(e) => Err(e),
        }
    }
    
    #[inline]
    fn distance_squared(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
    
    #[inline]
    fn nearest_centroid(&self, centroids: &[Point]) -> usize {
        let mut min_dist = f32::MAX;
        let mut min_idx = 0;
        for (i, c) in centroids.iter().enumerate() {
            let d = self.distance_squared(c);
            if d < min_dist {
                min_dist = d;
                min_idx = i;
            }
        }
        min_idx
    }
}

    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::with_capacity(16 * 1024 * 1024, stdin.lock());
    let mut writer = BufWriter::with_capacity(16 * 1024 * 1024, stdout.lock());
    
    // Read header line (text format)
    let mut header = String::new();
    reader.read_line(&mut header).expect("Failed to read header");
    let parts: Vec<&str> = header.trim().split_whitespace().collect();
    let num_points: u64 = parts[0].parse().expect("Invalid num_points");
    let num_clusters: usize = parts[1].parse().expect("Invalid num_clusters");
    
    if num_clusters == 0 || num_points == 0 {
        writer.flush().unwrap();
        return;
    }
    
    // Adaptive strategy selection
    if (num_points as usize) <= MAX_IN_MEMORY_POINTS {
        standard_kmeans(&mut reader, &mut writer, num_points as usize, num_clusters);
    } else {
        streaming_kmeans(&mut reader, &mut writer, num_points, num_clusters);
    }

fn standard_kmeans<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    num_points: usize,
    num_clusters: usize,
) {
    // Load all points into memory
    let mut points = Vec::with_capacity(num_points);
    let mut buffer = Vec::new();
    let mut remaining = num_points;
    
    while remaining > 0 {
        let to_read = remaining.min(BATCH_SIZE);
        match Point::read_batch(reader, &mut buffer, to_read) {
            Ok(count) => {
                points.extend(buffer.drain(..count));
                remaining -= count;
                if count == 0 { break; }
            }
            Err(_) => break,
        }
    }
    
    let actual_points = points.len();
    if actual_points == 0 {
        return;
    }
    
    // Initialize centroids using k-means++
    let mut centroids = kmeanspp_init(&points, num_clusters.min(actual_points));
    let k = centroids.len();
    
    if k == 0 {
        return;
    }
    
    // Allocate assignments
    let mut assignments = vec![0u32; actual_points];
    
    // K-means iterations
    for _ in 0..MAX_ITERATIONS {
        // Assignment step
        let mut changed = false;
        for (i, p) in points.iter().enumerate() {
            let new_assignment = p.nearest_centroid(&centroids) as u32;
            if assignments[i] != new_assignment {
                assignments[i] = new_assignment;
                changed = true;
            }
        }
        
        if !changed {
            break;
        }
        
        // Update step
        let mut sums = vec![[0.0f64; 3]; k];
        let mut counts = vec![0u64; k];
        
        for (i, p) in points.iter().enumerate() {
            let c = assignments[i] as usize;
            sums[c][0] += p.x as f64;
            sums[c][1] += p.y as f64;
            sums[c][2] += p.z as f64;
            counts[c] += 1;
        }
        
        for (i, c) in centroids.iter_mut().enumerate() {
            if counts[i] > 0 {
                c.x = (sums[i][0] / counts[i] as f64) as f32;
                c.y = (sums[i][1] / counts[i] as f64) as f32;
                c.z = (sums[i][2] / counts[i] as f64) as f32;
            }
        }
    }
    
    // Write output
    for &a in &assignments {
        writer.write_all(&a.to_le_bytes()).unwrap();
    }
}
```
