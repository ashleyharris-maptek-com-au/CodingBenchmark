def median_of_list(values):
    """Return median of a list, or None if empty."""
    if not values:
        return None
    sorted_vals = sorted(values)
    n = len(sorted_vals)
    if n % 2 == 1:
        return sorted_vals[n // 2]
    else:
        return (sorted_vals[n // 2 - 1] + sorted_vals[n // 2]) / 2

def low_pass_filter(current, new, alpha):
    """Simple low-pass filter."""
    if current is None:
        return new
    return alpha * new + (1 - alpha) * current

def calculate_safe_speed(mu, base_speed_kmh):
    """Calculate safe speed based on friction coefficient."""
    if mu is None or mu >= 0.8:
        return base_speed_kmh
    
    # Reduce speed for low friction
    # v² proportional to mu for same stopping distance
    factor = max(0.3, mu / 0.8)  # At least 30% of base speed
    return base_speed_kmh * math.sqrt(factor)

def calculate_stopping_distance(speed_kmh, mu=1.0):
    """Calculate stopping distance in meters."""
    speed_ms = speed_kmh / 3.6
    # d = v² / (2 * mu * g)
    g = 9.81
    if mu <= 0:
        mu = 0.1  # Minimum friction
    return (speed_ms ** 2) / (2 * mu * g)

    right_dists = []
    curvatures = []

    for i in [1, 2, 3]:
        p = f'vision_{i}'
        
        # Lane offset
        offset = state.get(f'{p}_lane_offset_m')

        if state.get(f'{p}_left_line_valid', False):

        'left_dist': median_filter(left_dists),

        'speed_limit': min(speed_limits) if speed_limits else None,
        'obstacle': min(obstacles, key=lambda x: x['dist']) if obstacles else None
    }

def stopping_distance(speed_kmh, mu=1.0):
    """Calculate stopping distance in meters."""
    v = speed_kmh / 3.6  # m/s
    mu = max(0.1, mu)
    return v * v / (2 * 9.81 * mu)

def autopilot_step(state: dict, dt: float) -> dict:
    global _state
    
    # Initialize
    if not _state.initialized:
        sensors = fuse_sensors(state)
        _state.speed_kmh = sensors['speed_kmh']
        _state.initialized = True
    
    # Get friction
    mu = state.get('estimated_mu', 1.0) or 1.0
    mu = max(0.1, min(1.0, mu))
    
    # Rain affects friction
    if state.get('rain_sensor', 0):
        mu = min(mu, 0.7)
    
    # Fuse sensors
    sensors = fuse_sensors(state)
    speed = sensors['speed_kmh']
    yaw = sensors['yaw_rate_dps']
    
    # Low-pass filter
    alpha = 0.3
    _state.speed_kmh = alpha * speed + (1 - alpha) * _state.speed_kmh
    _state.yaw_rate_dps = alpha * yaw + (1 - alpha) * _state.yaw_rate_dps
    
    # Get vision and lidar data
    vision = get_vision_data(state)
    lidar = get_lidar_readings(state)
    
    # Targets
    target_speed = state.get('target_speed_kmh', 100)
    target_lane = state.get('target_lane', 1)
    current_lane = state.get('current_lane', 1)
    
    # Apply friction-based speed limit
    safe_speed = target_speed * math.sqrt(max(0.3, mu))
    
    # Apply vision speed limit
    if vision['speed_limit']:
        safe_speed = min(safe_speed, vision['speed_limit'])
    
    # === OBSTACLE DETECTION ===
    emergency_brake = False
    brake = 0.0
    
    # Vision obstacle
    obstacle_dist = float('inf')
    if vision['obstacle']:
        obstacle_dist = min(obstacle_dist, vision['obstacle']['dist'])
    
    # Lidar obstacles (forward beams)
    for angle in [5, 15, 30, 45, -5, -15]:
        if angle in lidar:
            obstacle_dist = min(obstacle_dist, lidar[angle])
    
    # Check for emergency
    stop_dist = stopping_distance(_state.speed_kmh, mu)
    safety_margin = 10
    
    if obstacle_dist < stop_dist + safety_margin:
        emergency_brake = True
        brake = min(1.0, max(0.5, (stop_dist + safety_margin - obstacle_dist) / (stop_dist + safety_margin)))
    
    if obstacle_dist < 10:
        emergency_brake = True
        brake = 1.0
    
    # === SPEED CONTROL ===
    target = safe_speed if not emergency_brake else 0
    error = target - _state.speed_kmh
    
    # PID
    Kp, Ki, Kd = 0.03, 0.001, 0.01
    _state.speed_i = max(-0.5, min(0.5, _state.speed_i + error * dt))
    d_err = (error - _state.speed_prev_err) / dt if dt > 0 else 0
    _state.speed_prev_err = error
    
    control = Kp * error + Ki * _state.speed_i + Kd * d_err
    
    throttle = 0.0
    if not emergency_brake:
        if control > 0:
            throttle = min(1.0, control * mu)
        else:
            brake = max(brake, min(0.3, -control * 0.3))
    
    # === STEERING ===
    steering = 0.0
    
    # Lane offset correction
    if vision['lane_offset'] is not None:
        offset = vision['lane_offset']
        Kp_s, Ki_s, Kd_s = 0.08, 0.001, 0.03
        
        _state.steer_i = max(-0.3, min(0.3, _state.steer_i + offset * dt))
        d_off = (offset - _state.steer_prev_err) / dt if dt > 0 else 0
        _state.steer_prev_err = offset
        
        steering = -(Kp_s * offset + Ki_s * _state.steer_i + Kd_s * d_off)
    
    # Line distance correction
    if vision['left_dist'] is not None and vision['right_dist'] is not None:
        center_error = (vision['right_dist'] - vision['left_dist']) / 2
        steering -= center_error * 0.02
    
    # Curvature feed-forward
    steering += vision['curvature'] * 0.0005
    
    # Yaw damping
    steering -= _state.yaw_rate_dps * 0.003
    
    # Lane change
    if target_lane != current_lane and not emergency_brake:
        direction = 1 if target_lane > current_lane else -1
        
        # Check side clearance
        side_angles = [45, 60] if direction < 0 else [-45, -60]
        clear = all(lidar.get(a, 50) > 15 for a in side_angles)
        
        if clear:
            steering += direction * 0.12
    
    # Lidar side obstacle avoidance
    if 60 in lidar and lidar[60] < 5:
        steering -= 0.1  # Steer right
    if -60 in lidar and lidar[-60] < 5:
        steering += 0.1  # Steer left
    
    # Low friction steering limits
    if mu < 0.5:
        steering = max(-0.4, min(0.4, steering * mu * 2))
    
    # Clamp outputs
    steering = max(-1.0, min(1.0, steering))
    throttle = max(0.0, min(1.0, throttle))
    brake = max(0.0, min(1.0, brake))
    
    if brake > 0.3:
        throttle = 0.0
    
    return {'steering': steering, 'throttle': throttle, 'brake': brake}
