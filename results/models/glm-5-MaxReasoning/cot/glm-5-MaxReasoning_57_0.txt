```javascript
    // Convert grid to a more usable format
    const cells = [];
    for (let y = 0; y < height; y++) {
        cells[y] = [];
        for (let x = 0; x < width; x++) {
            cells[y][x] = grid[y][x];
        }

    // Function to simulate fire spread with given cuts
    function simulateFire(cuts) {
        const burned = new Set();
        const queue = [];
        
        // Initialize with ignition points
        for (const [ix, iy] of ignitions) {
            if (cells[iy]?.[ix] === '.' && !cuts.has(`${ix},${iy}`)) {
                burned.add(`${ix},${iy}`);
                queue.push([ix, iy]);

        while (queue.length > 0) {
            const [x, y] = queue.shift();
            const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
            
            for (const [dx, dy] of dirs) {
                const nx = x + dx, ny = y + dy;

                    cells[ny][nx] === '.' && !burned.has(key) && !cuts.has(key)) {

                    queue.push([nx, ny]);

            currentCuts.add(`${bestCell[0]},${bestCell[1]}`);
```

```javascript
function solve(data) {
    const {width, height, grid, ignitions, towns, cutBudget} = data;
    
    const dirs = [[0,1],[0,-1],[1,0],[-1,0]];
    const key = (x, y) => y * width + x; // Use numeric key for speed
    const isBurnable = (x, y) => x >= 0 && x < width && y >= 0 && y < height && grid[y][x] === '.';
    
    function simulateFire(cutSet) {
        const visited = new Uint8Array(width * height);
        const queue = [];
        let qPos = 0;
        
        for (const [sx, sy] of ignitions) {
            const k = key(sx, sy);
            if (isBurnable(sx, sy) && !cutSet[k]) {
                visited[k] = 1;
                queue.push(k);
            }
        }
        
        while (qPos < queue.length) {
            const k = queue[qPos++];
            const x = k % width, y = Math.floor(k / width);
            
            for (const [dx, dy] of dirs) {
                const nx = x + dx, ny = y + dy;
                if (isBurnable(nx, ny)) {
                    const nk = key(nx, ny);
                    if (!visited[nk] && !cutSet[nk]) {
                        visited[nk] = 1;
                        queue.push(nk);
                    }
```

Actually, finding articulation points in a grid graph is relatively efficient with Tarjan's algorithm.

**Tarjan's algorithm for articulation points:**

```javascript
function findArticulationPoints(grid, width, height) {
    const isBurnable = (x, y) => x >= 0 && x < width && y >= 0 && y < height && grid[y][x] === '.';
    const dirs = [[0,1],[0,-1],[1,0],[-1,0]];
    
    const visited = new Map();
    const disc = new Map();
    const low = new Map();
    const parent = new Map();
    const ap = new Set();
    let time = 0;
    
    function dfs(x, y) {
        const key = `${x},${y}`;
        visited.set(key, true);
        disc.set(key, time);
        low.set(key, time);
        time++;
        
        let children = 0;
        
        for (const [dx, dy] of dirs) {
            const nx = x + dx, ny = y + dy;
            const nkey = `${nx},${ny}`;
            
            if (isBurnable(nx, ny)) {
                if (!visited.get(nkey)) {
                    children++;
                    parent.set(nkey, key);
                    dfs(nx, ny);
                    
                    low.set(key, Math.min(low.get(key), low.get(nkey)));
                    
                    // Check if this is an articulation point
                    const p = parent.get(key);
                    if (p === undefined && children > 1) {
                        ap.add(key);
                    }
                    if (p !== undefined && low.get(nkey) >= disc.get(key)) {
                        ap.add(key);
                    }
                } else if (nkey !== parent.get(key)) {
                    low.set(key, Math.min(low.get(key), disc.get(nkey)));
                }
            }
        }
    }
    
    // Run DFS from each unvisited burnable cell
    for (let y = 0; y < height; y++) {
        for (let x = 0; x < width; x++) {
            if (isBurnable(x, y) && !visited.get(`${x},${y}`)) {
                dfs(x, y);
            }
        }
    }
    
    return ap;
}
```
