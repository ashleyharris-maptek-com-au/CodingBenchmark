1. Use a value-weighted potential field
2. Drones follow the gradient
3. Simple repulsion between drones

    return x >= 0 && x < cols && y >= 0 && y < rows && prob[y][x] !== -1;
}

const visited = new Set();

    if (prob[d.y][d.x] > 0) visited.add(`${d.x},${d.y}`);
    return [[d.x, d.y]];
});
const positions = drones.map(d => [d.x, d.y]);
const numDrones = drones.length;

// Compute attraction field from valuable cells
function computeAttraction() {
    const attract = Array.from({length: rows}, () => Array(cols).fill(0));
    
    for (let y = 0; y < rows; y++) {
        for (let x = 0; x < cols; x++) {
            if (prob[y][x] > 0 && !visited.has(`${x},${y}`)) {
                // Spread influence with distance decay
                const spread = 5;
                for (let dy = -spread; dy <= spread; dy++) {
                    for (let dx = -spread; dx <= spread; dx++) {
                        const nx = x + dx;
                        const ny = y + dy;
                        if (isValid(nx, ny)) {
                            const dist = Math.abs(dx) + Math.abs(dy);
                            attract[ny][nx] += prob[y][x] / (1 + dist * 0.5);
                        }
                    }
                }
            }
        }
    }
    
    return attract;
}

// Main loop
for (let step = 0; step < maxSteps; step++) {
    const attract = computeAttraction();
    
    // Add repulsion from other drones
    const repel = Array.from({length: rows}, () => Array(cols).fill(0));
    for (let di = 0; di < numDrones; di++) {
        const [px, py] = positions[di];
        for (let dy = -3; dy <= 3; dy++) {
            for (let dx = -3; dx <= 3; dx++) {
                const nx = px + dx;
                const ny = py + dy;
                if (nx >= 0 && nx < cols && ny >= 0 && ny < rows) {
                    const dist = Math.abs(dx) + Math.abs(dy);
                    if (dist > 0) {
                        repel[ny][nx] += 10 / dist;
                    }
                }
            }
        }
    }
    
    // Move each drone
    for (let di = 0; di < numDrones; di++) {
        const [cx, cy] = positions[di];
        
        let bestMove = [0, 0];
        let bestScore = -Infinity;
        
        for (const [dx, dy] of dirs) {
            const nx = cx + dx;
            const ny = cy + dy;
            
            if (!isValid(nx, ny)) continue;
            
            const key = `${nx},${ny}`;
            const cellValue = visited.has(key) ? 0 : prob[ny][nx];
            
            // Score = direct value + attraction - repulsion
            let score = attract[ny][nx] + cellValue * 5 - repel[ny][nx];
            
            if (score > bestScore) {
                bestScore = score;
                bestMove = [dx, dy];
            }
        }
        
        // Execute move
        const [dx, dy] = bestMove;
        positions[di] = [cx + dx, cy + dy];
        paths[di].push([...positions[di]]);
        
        if (prob[positions[di][1]][positions[di][0]] > 0) {
            visited.add(`${positions[di][0]},${positions[di][1]}`);
        }
    }
}
