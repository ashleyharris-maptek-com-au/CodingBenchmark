Let me write an HLSL pixel shader for Vulkan that implements Phong lighting with the specified requirements.

Requirements:
1. Ambient color: (0.1, 0.05, 0.05)
2. Diffuse: uses vertex color (input.color) with max(dot(N, L), 0)
3. Specular: (1, 1, 1) with shininess 32, R = reflect(-L, N)
4. Clamp the sum to [0, 1]
5. Output alpha 1.0

The Phong lighting model:
- Ambient: fixed color
- Diffuse: Kd * max(dot(N, L), 0) where L is light direction (from point to light)
- Specular: Ks * pow(max(dot(R, V), 0), shininess) where R = reflect(-L, N), V is view direction

Let me write the shader:

```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the interpolated normal
    float3 N = normalize(input.normal);
    
    // Light direction (from surface point to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // View direction (from surface point to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);
    
    // Ambient
    float3 ambient = float3(0.1, 0.05, 0.05);
    
    // Diffuse
    float NdotL = max(dot(N, L), 0.0);
    float3 diffuse = input.color * NdotL;
    
    // Specular
    float3 R = reflect(-L, N);
    float RdotV = max(dot(R, V), 0.0);
    float3 specular = float3(1.0, 1.0, 1.0) * pow(RdotV, 32.0);
    
    // Final color
    float3 finalColor = ambient + diffuse + specular;
    finalColor = clamp(finalColor, 0.0, 1.0);
    
    return float4(finalColor, 1.0);
}
```
