    struct PSInput {
        [[vk::location(0)]] float3 worldPos : TEXCOORD0;
        [[vk::location(1)]] float3 normal   : TEXCOORD1;
        [[vk::location(2)]] float2 uv       : TEXCOORD2;
        [[vk::location(3)]] float3 tangent  : TEXCOORD3; // unused
        [[vk::location(4)]] float3 color    : TEXCOORD4; // unused
    };

    [[vk::binding(0, 0)]]
    cbuffer UBO : register(b0) {
        float4x4 model;
        float4x4 view;
        float4x4 projection;
        float4 lightPos;
        float4 cameraPos;
        float4 params;
    };

    float4 main(PSInput input) : SV_Target0
    {
        // 1. Normalize inputs
        float3 N = normalize(input.normal);
        
        // 2. Lighting (Lambert)
        float3 L = normalize(UBO.lightPos.xyz - input.worldPos);
        float NdotL = max(0.0, dot(N, L));
        
        // 3. Brick Pattern
        // Scale UVs
        float2 scaledUV = input.uv * float2(8.0, 16.0);
        
        // Stagger rows
        // Determine row index
        float row = floor(scaledUV.y);
        // Apply offset for odd rows
        // Using fmod(row, 2.0) to check parity. If odd, result is ~1.0.
        if (fmod(row, 2.0) > 0.5) {
            scaledUV.x += 0.5;
        }
        
        // Get fractional part for brick coordinate
        float2 brickUV = frac(scaledUV);
        
        // Mortar mask
        // Mortar thickness: 0.05 x, 0.1 y
        float2 mortarThick = float2(0.05, 0.1);
        
        // Check if we are inside the brick (not in mortar)
        // step(thick, uv) is 1 if uv > thick
        // step(uv, 1-thick) is 1 if uv < 1-thick

        float isBrick = insideBrick.x * insideBrick.y;
        
        // Colors
        float3 mortarColor = float3(0.7, 0.7, 0.7);
        float3 brickColor = float3(0.7, 0.2, 0.1);
        
        float3 baseColor = lerp(mortarColor, brickColor, isBrick);
        
        // 4. Apply lighting factor
        // Factor: 0.4 + 0.6 * diffuse
        float3 finalColor = baseColor * (0.4 + 0.6 * NdotL);
        
        return float4(finalColor, 1.0);
    }
