fn median(values: &[f64]) -> f64 {
    if values.is_empty() {
        return f64::NAN;
    }
    let mut sorted: Vec<f64> = values.iter().cloned().collect();
    sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = sorted.len();
    if n % 2 == 0 {
        (sorted[n/2 - 1] + sorted[n/2]) / 2.0
    } else {
        sorted[n/2]
    }
}

fn mad(values: &[f64], med: f64) -> f64 {
    let deviations: Vec<f64> = values.iter()
        .map(|v| (v - med).abs())
        .collect();
    median(&deviations)
}

            }
            
            all_samples.push(Sample {
                hole_id,
                sample_idx,
                position,
                properties,
            });
        }
    }
    
    let total_samples = all_samples.len();
    
    // Adaptive parameters based on dataset size
    let (cell_size, search_radius, min_neighbors, z_threshold) = 
        if total_samples < 1_000 {
            (10.0, 25.0, 2, 2.5)
        } else if total_samples < 10_000 {
            (25.0, 50.0, 3, 3.0)
        } else if total_samples < 100_000 {
            (50.0, 100.0, 5, 3.0)
        } else if total_samples < 1_000_000 {
            (100.0, 200.0, 5, 3.5)
        } else {
            (200.0, 400.0, 5, 3.5)
        };
    
    // Build spatial grid
    let mut grid: HashMap<(i64, i64, i64), Vec<usize>> = HashMap::new();
    
    for (idx, sample) in all_samples.iter().enumerate() {
        let cell = (
            (sample.position[0] / cell_size).floor() as i64,
            (sample.position[1] / cell_size).floor() as i64,
            (sample.position[2] / cell_size).floor() as i64,
        );
        grid.entry(cell).or_default().push(idx);
    }
    
    // Compute global statistics for each property
    let mut global_stats: Vec<(f64, f64)> = Vec::with_capacity(num_properties);
    for prop_idx in 0..num_properties {
        let mut values: Vec<f64> = all_samples.iter()
            .map(|s| s.properties[prop_idx])
            .filter(|v| v.is_finite())
            .collect();
        global_stats.push(compute_stats(&mut values));
    }
    
    // Find outliers
    let mut results: Vec<AnalysisResult> = Vec::new();
    let search_radius_sq = search_radius * search_radius;
    
    for (sample_idx, sample) in all_samples.iter().enumerate() {
        let cell = (
            (sample.position[0] / cell_size).floor() as i64,
            (sample.position[1] / cell_size).floor() as i64,
            (sample.position[2] / cell_size).floor() as i64,
        );
        
        // Collect neighbors
        let mut neighbor_indices: Vec<usize> = Vec::new();
        for dx in -1i64..=1 {
            for dy in -1i64..=1 {
                for dz in -1i64..=1 {
                    let neighbor_cell = (cell.0 + dx, cell.1 + dy, cell.2 + dz);
                    if let Some(indices) = grid.get(&neighbor_cell) {
                        for &idx in indices {
                            if idx != sample_idx {
                                let other = &all_samples[idx];
                                let dist_sq = 
                                    (sample.position[0] - other.position[0]).powi(2) +
                                    (sample.position[1] - other.position[1]).powi(2) +
                                    (sample.position[2] - other.position[2]).powi(2);
                                if dist_sq <= search_radius_sq {
                                    neighbor_indices.push(idx);
                                }
                            }
                        }
                    }
                }
            }
        }
        
        // Analyze each property
        for prop_idx in 0..num_properties {
            let value = sample.properties[prop_idx];
            if !value.is_finite() {
                continue;
            }
            
            // Get neighbor values
            let mut neighbor_values: Vec<f64> = neighbor_indices.iter()
                .map(|&idx| all_samples[idx].properties[prop_idx])
                .filter(|v| v.is_finite())
                .collect();
            
            let (med, m) = if neighbor_values.len() >= min_neighbors {
                compute_stats(&mut neighbor_values)
            } else {
                global_stats[prop_idx]
            };
            
            if !med.is_finite() {
                continue;
            }
            
            let z_score = (value - med).abs() / m;
            
            if z_score > z_threshold {
                results.push(AnalysisResult {
                    hole_id: sample.hole_id,
                    sample_idx: sample.sample_idx,
                    property_name: property_names[prop_idx].clone(),
                    confidence: z_score,
                });
            }
        }
    }
    
    // Sort by confidence descending
    results.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal));
    
    // Output
    writeln!(stdout, "{}", results.len()).unwrap();
    for r in results {
        writeln!(stdout, "{} {} {} {:.6}", 
            r.hole_id, 
            r.sample_idx, 
            r.property_name, 
            r.confidence
        ).unwrap();
    }
}

use std::cmp::Ordering;

#[derive(Clone)]
struct Sample {
    hole_id: u64,
    sample_idx: usize,
    position: [f64; 3],
    properties: Vec<f64>,
}

fn median(values: &mut [f64]) -> f64 {
    if values.is_empty() {
        return f64::NAN;
    }
    values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = values.len();
    if n % 2 == 0 {
        (values[n/2 - 1] + values[n/2]) / 2.0
    } else {
        values[n/2]
    }
}

fn robust_stats(values: &mut [f64]) -> (f64, f64) {
    let med = median(values);
    let m: Vec<f64> = values.iter().map(|v| (v - med).abs()).collect();
    let mut m = m;
    let mad = median(&mut m);
    let scale = mad * 1.4826;
    (med, if scale > 0.0 { scale } else { 1.0 })
}
