class UnionFind {
    vector<int> parent, rank_;
public:
    UnionFind(int n) : parent(n), rank_(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }

        if (parent[x] != x) {
            parent[x] = find(parent[x]);
        }
        return parent[x];

        y = find(y);

#include <iostream>
#include <vector>

using namespace std;
using ll = long long;

class UnionFind {
    vector<int> parent, rank_;
public:
    UnionFind(int n) : parent(n), rank_(n, 0) {
        iota(parent.begin(), parent.end(), 0);
    }
    
    int find(int x) {
        return parent[x] == x ? x : parent[x] = find(parent[x]);
    }
    
    void unite(int x, int y) {
        x = find(x); y = find(y);
        if (x != y) {
            if (rank_[x] < rank_[y]) swap(x, y);
            parent[y] = x;
            if (rank_[x] == rank_[y]) rank_[x]++;
        }
    }
};

// Stoer-Wagner minimum cut
pair<ll, vector<int>> stoerWagner(int n, const vector<vector<int>>& adj) {
    // Build weighted adjacency matrix
    vector<vector<ll>> g(n, vector<ll>(n, 0));
    for (int i = 0; i < n; i++) {
        for (int j : adj[i]) {
            g[i][j]++;
        }
    }
    
    vector<vector<int>> merged(n);
    for (int i = 0; i < n; i++) merged[i] = {i};
    
    vector<int> v(n);
    iota(v.begin(), v.end(), 0);
    
    ll best_cut = LLONG_MAX;
    vector<int> best_partition;
    int num_v = n;
    
    while (num_v > 1) {
        vector<ll> w(n, 0);
        vector<bool> in_a(n, false);
        int prev = -1;
        
        for (int i = 0; i < num_v; i++) {
            int sel = -1;
            for (int j = 0; j < num_v; j++) {
                if (!in_a[v[j]] && (sel == -1 || w[v[j]] > w[v[sel]])) {
                    sel = j;
                }
            }
            
            in_a[v[sel]] = true;
            
            if (i == num_v - 1) {
                ll cut_val = w[v[sel]];
                if (cut_val < best_cut) {
                    best_cut = cut_val;
                    best_partition = merged[v[sel]];
                }
                
                // Merge
                merged[prev].insert(merged[prev].end(), merged[v[sel]].begin(), merged[v[sel]].end());
                for (int j = 0; j < n; j++) {
                    g[prev][j] += g[v[sel]][j];
                    g[j][prev] = g[prev][j];
                }
                g[prev][prev] = 0;
                
                swap(v[sel], v[num_v - 1]);
                num_v--;
                break;
            }
            
            prev = v[sel];
            for (int j = 0; j < num_v; j++) {
                if (!in_a[v[j]]) w[v[j]] += g[v[sel]][v[j]];
            }
        }
    }
    
    return {best_cut, best_partition};
}

// Karger-Stein
pair<ll, vector<int>> kargerStein(int n, const vector<pair<int, int>>& edges, int iter) {
    mt19937 rng(chrono::steady_clock::now().time_since_epoch().count());
    ll best_cut = LLONG_MAX;
    vector<int> best_part;
    
    for (int t = 0; t < iter; t++) {
        UnionFind uf(n);
        auto e = edges;
        shuffle(e.begin(), e.end(), rng);
        
        int comp = n;
        for (auto& [u, v] : e) {
            if (comp <= 2) break;
            if (uf.find(u) != uf.find(v)) {
                uf.unite(u, v);
                comp--;
            }
        }
        
        set<int> part;
        int r = uf.find(0);
        for (int i = 0; i < n; i++) if (uf.find(i) == r) part.insert(i);
        
        ll cut = 0;
        for (auto& [u, v] : edges) {
            if (part.count(u) != part.count(v)) cut++;
        }
        
        if (cut < best_cut) {
            best_cut = cut;
            best_part = vector<int>(part.begin(), part.end());
        }
    }
    
    return {best_cut, best_part};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m;
    cin >> n >> m;
    
    vector<vector<int>> adj(n);
    vector<pair<int, int>> edges;
    
    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        edges.emplace_back(u, v);
    }
    
    ll cut_val;
    vector<int> partition;
    
    if (n <= 8000) {
        tie(cut_val, partition) = stoerWagner(n, adj);
    } else {
        int iter = min(100000, (int)(n * log((double)n) + 1));
        tie(cut_val, partition) = kargerStein(n, edges, iter);
    }
    
    set<int> in_part(partition.begin(), partition.end());
    vector<pair<int, int>> cut_edges;
    
    for (auto& [u, v] : edges) {
        if (in_part.count(u) != in_part.count(v)) {
            cut_edges.emplace_back(min(u, v), max(u, v));
        }
    }
    
    sort(cut_edges.begin(), cut_edges.end());
    cut_edges.erase(unique(cut_edges.begin(), cut_edges.end()), cut_edges.end());
    
    cout << cut_edges.size() << '\n';
    for (auto& [u, v] : cut_edges) {
        cout << u << ' ' << v << '\n';
    }
    
    return 0;
}

1. Reading input in a streaming fashion
2. Using algorithms that work with limited memory

1. Read edges one at a time
2. Randomly decide whether to contract each edge
3. Maintain a Union-Find structure
