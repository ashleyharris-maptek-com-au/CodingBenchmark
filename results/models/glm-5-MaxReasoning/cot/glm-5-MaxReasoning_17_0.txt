    static List<Placement> Solve(Box[] boxes, int CW, int CH, int CD)
    {
        int n = boxes.Length;
        long containerVol = (long)CW * CH * CD;
        
        // Choose strategy based on problem size
        if (n <= 50)

        else

    static List<Placement> SolveGreedy(Box[] boxes, int CW, int CH, int CD)
    {
        var sortedBoxes = boxes.OrderByDescending(b => (long)b.W * b.H * b.D).ToArray();
        
        var placed = new List<PlacedBox>();
        var result = new List<Placement>();
        
        // Extreme points for placement
        var xs = new List<int> { 0 };
        var ys = new List<int> { 0 };
        var zs = new List<int> { 0 };
        
        foreach (var box in sortedBoxes)
        {
            var orients = GetOrientations(box);
            
            int bestX = -1, bestY = -1, bestZ = -1;
            int bestW = 0, bestH = 0, bestD = 0;
            
            foreach (var (w, h, d) in orients)
            {
                if (w > CW || h > CH || d > CD) continue;
                
                bool found = false;
                foreach (var z in zs)
                {
                    if (z + d > CD) break;
                    foreach (var y in ys)
                    {
                        if (y + h > CH) break;
                        foreach (var x in xs)
                        {
                            if (x + w > CW) break;
                            
                            if (!HasCollision(placed, x, y, z, w, h, d))
                            {
                                bestX = x; bestY = y; bestZ = z;
                                bestW = w; bestH = h; bestD = d;
                                found = true;
                                break;
                            }
                        }
                        if (found) break;
                    }
                    if (found) break;
                }
                if (found) break;
            }
            
            if (bestX >= 0)
            {
                placed.Add(new PlacedBox(bestX, bestY, bestZ, bestW, bestH, bestD));
                result.Add(new Placement { Index = box.Index, X = bestX, Y = bestY, Z = bestZ });
                
                AddIfNew(xs, bestX);
                AddIfNew(xs, bestX + bestW);
                AddIfNew(ys, bestY);
                AddIfNew(ys, bestY + bestH);
                AddIfNew(zs, bestZ);
                AddIfNew(zs, bestZ + bestD);
            }
        }
        
        return result;
    }
    
    static void AddIfNew(List<int> list, int val)
    {
        if (!list.Contains(val))
        {
            list.Add(val);
            list.Sort();
        }
    }
    
    static bool HasCollision(List<PlacedBox> placed, int x, int y, int z, int w, int h, int d)
    {
        foreach (var p in placed)
        {
            if (x < p.X + p.W && x + w > p.X &&
                y < p.Y + p.H && y + h > p.Y &&
                z < p.Z + p.D && z + d > p.Z)
                return true;
        }
        return false;
    }
    
    static List<(int, int, int)> GetOrientations(Box b)
    {
        return new List<(int, int, int)>
        {
            (b.W, b.H, b.D), (b.W, b.D, b.H), (b.H, b.W, b.D),
            (b.H, b.D, b.W), (b.D, b.W, b.H), (b.D, b.H, b.W)
        };
    }
    
    static List<Placement> SolveExact(Box[] boxes, int CW, int CH, int CD)
    {
        // For small problems, could try backtracking
        // For now, use same greedy approach
        return SolveGreedy(boxes, CW, CH, CD);
    }
