            if (burned.has(townKey(town))) {
                value += town.value;
            }
        }
        return value;
    }

    const initialBurned = simulateFire();
    const initialBurnedValue = getBurnedValue(initialBurned);

    const dy = [1, -1, 0, 0];
    
    function inBounds(x: number, y: number): boolean {

                dist.set(key, 0);
                queue.push([x, y, 0]);
            }
        }
        
        while (queue.length > 0) {
            const [x, y, d] = queue.shift()!;
            for (let i = 0; i < 4; i++) {
                const nx = x + dx[i];
                const ny = y + dy[i];
                const key = `${nx},${ny}`;
                if (isBurnable(nx, ny) && !dist.has(key) && !cutCells.has(key)) {
                    dist.set(key, d + 1);
                    queue.push([nx, ny, d + 1]);
                }

    id: number;
    x: number;
    y: number;
    value: number;
}

    height: number;
    grid: string[];

    cutBudget: number;

function solve(data: Input): [number, number][] {
    const { width, height, grid, ignitions, towns, cutBudget } = data;
    
    // Use arrays for better performance
    const burned = new Uint8Array(width * height);
    const blocked = new Uint8Array(width * height);
    
    const dx = [0, 0, 1, -1];
    const dy = [1, -1, 0, 0];
    
    function idx(x: number, y: number): number {
        return y * width + x;
    }
    
    function inBounds(x: number, y: number): boolean {
        return x >= 0 && x < width && y >= 0 && y < height;
    }
    
    function isBurnable(x: number, y: number): boolean {
        return inBounds(x, y) && grid[y][x] === '.';
    }
    
    // BFS simulation
    function simulateFire(): number {
        burned.fill(0);
        const queue: number[] = [];
        
        for (const [x, y] of ignitions) {
            const i = idx(x, y);
            if (isBurnable(x, y) && !blocked[i]) {
                burned[i] = 1;
                queue.push(i);
            }
        }
        
        let head = 0;
        while (head < queue.length) {
            const i = queue[head++];
            const x = i % width;
            const y = Math.floor(i / width);
            
            for (let d = 0; d < 4; d++) {
                const nx = x + dx[d];
                const ny = y + dy[d];
                if (isBurnable(nx, ny)) {
                    const ni = idx(nx, ny);
                    if (!burned[ni] && !blocked[ni]) {
                        burned[ni] = 1;
                        queue.push(ni);
                    }
                }
            }
        }
        
        // Calculate burned value
        let value = 0;
        for (const town of towns) {
            if (burned[idx(town.x, town.y)]) {
                value += town.value;
            }
        }
        return value;
    }
    
    const firebreaks: [number, number][] = [];
    
    for (let cut = 0; cut < cutBudget; cut++) {
        const currentBurned = simulateFire();
        if (currentBurned === 0) break;
        
        let bestCell = -1;
        let bestSavings = 0;
        
        // Find best cell to cut
        for (let i = 0; i < width * height; i++) {
            if (burned[i] && !blocked[i]) {
                blocked[i] = 1;
                const newBurned = simulateFire();
                const savings = currentBurned - newBurned;
                if (savings > bestSavings) {
                    bestSavings = savings;
                    bestCell = i;
                }
                blocked[i] = 0;

        if (bestCell === -1 || bestSavings === 0) break;
        
        blocked[bestCell] = 1;
        const x = bestCell % width;
        const y = Math.floor(bestCell / width);
        firebreaks.push([x, y]);
    }
    
    return firebreaks;

1. Compute fire spread (BFS from ignitions) - O(N)
2. For each burning cell, compute "downstream value" - the total value of towns that fire reaches through this cell - O(N * T) where T is number of towns
3. Select cells with highest downstream value

The key insight is: if we cut a cell, we save the downstream value ONLY IF that cell is the only path to those towns. If there are multiple paths, cutting one cell doesn't help.

For a more accurate heuristic, we can compute:
- For each town, compute all cells that are on ANY path from fire to town
- For each cell, track which towns have this cell as a "chokepoint" (all paths go through it)

This is related to computing "dominators" in graph theory.

**Dominator approach**:
1. Build the fire spread graph (directed from ignition outward, using BFS distances)
2. Add a super-source connected to all ignition points
3. Compute dominators for each town
4. A cell dominates a town if all paths from source to town go through that cell
5. Priority of cell = sum of values of towns it dominates

Computing dominators can be done in O(N * α(N)) where α is inverse Ackermann (very fast).

                if (y < height - 1) { const ni = i + width; if (dist[ni] === curDist - 1 && !visited[ni]) { visited[ni] = 1; tq[tqt++] = ni; } }
            }
        }
        
        return imp;
    }

        onFire.fill(0);
        let qh = 0, qt = 0;
        
        for (const [x, y] of ignitions) {

        }
        
        let loss = 0;
        for (let i = 0; i < n; i++) if (onFire[i]) loss += townVal[i];
        return loss;
    }
    
    const firebreaks: [number, number][] = [];
    let lastLoss = simulate();
    
    while (firebreaks.length < cutBudget && lastLoss > 0) {
        const imp = computeImportance();
        
        // Find highest importance cell not yet blocked
        let bestCell = -1, bestImp = 0;
        for (let i = 0; i < n; i++) {
            if (dist[i] >= 0 && !blocked[i] && imp[i] > bestImp) {
                bestImp = imp[i];
                bestCell = i;
            }
        }
        
        if (bestCell < 0 || bestImp === 0) break;
        
        // Try cutting this cell
        blocked[bestCell] = 1;
        const newLoss = simulate();
        
        if (newLoss < lastLoss) {
            firebreaks.push([bestCell % width, Math.floor(bestCell / width)]);
            lastLoss = newLoss;
        } else {
            blocked[bestCell] = 0;
            // Mark as low priority to avoid re-checking
            // Actually, we should continue with next best
        }
    }
    
    return firebreaks;
}
