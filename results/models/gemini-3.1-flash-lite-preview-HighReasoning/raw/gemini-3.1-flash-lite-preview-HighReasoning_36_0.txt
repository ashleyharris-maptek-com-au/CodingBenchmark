The provided C++ solution implements a Maximum Independent Set solver using a backtracking approach with bitset optimization and pruning. Given the NP-hard nature of the problem, this approach is suitable for exact solutions on reasonably sized graphs.

The code adheres to the requirements:
1.  **Standard C++**: Uses C++17/20 compatible syntax.
2.  **I/O**: Reads from `stdin` and writes to `stdout` using fast I/O (`ios_base::sync_with_stdio(false)`).
3.  **Completeness**: Includes `main()` and necessary headers.

```cpp
#include <iostream>
#include <vector>
#include <bitset>
#include <algorithm>

using namespace std;

// Maximum number of vertices, assumed 1024 for reasonable graph sizes in competitive programming.
const int MAXN = 1024;
bitset<MAXN> adj[MAXN];
int n, m;

vector<int> best_set;
vector<int> current_set;

void solve(bitset<MAXN> candidates) {
    if (candidates.none()) {
        if (current_set.size() > best_set.size()) {
            best_set = current_set;
        }
        return;
    }

    // Pruning: bound by current size + remaining candidates
    if (current_set.size() + candidates.count() <= best_set.size()) {
        return;
    }

    // Heuristic: pick vertex with max degree in candidates
    int v = -1;
    int max_deg = -1;
    for (int i = 0; i < n; ++i) {
        if (candidates.test(i)) {
            int deg = (adj[i] & candidates).count();
            if (deg > max_deg) {
                max_deg = deg;
                v = i;
            }
        }
    }

    if (v == -1) return;

    // Option 1: Include v in the independent set
    current_set.push_back(v);
    bitset<MAXN> next_candidates = candidates & ~adj[v];
    next_candidates.reset(v);
    solve(next_candidates);
    current_set.pop_back();

    // Option 2: Exclude v
    candidates.reset(v);
    solve(candidates);
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> n >> m)) return 0;

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].set(v);
            adj[v].set(u);
        }
    }

    bitset<MAXN> all;
    for (int i = 0; i < n; ++i) all.set(i);
    
    solve(all);

    // Output the result
    cout << best_set.size() << "\n";
    for (int i = 0; i < best_set.size(); ++i) {
        cout << best_set[i] << (i == best_set.size() - 1 ? "" : " ");
    }
    cout << "\n";

    return 0;
}
```
