The SPIR-V assembly shader implementation is complete:

```spvasm
; SPIR-V
; Version: 1.0
; Generator: manual

               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_color %outColor
               OpExecutionMode %main OriginUpperLeft
               OpSource GLSL 450

; --- Layout & Decorations ---
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_color Location 4
               OpDecorate %outColor Location 0
               OpDecorate %ubo_block Block
               OpMemberDecorate %ubo_block 0 Offset 0
               OpMemberDecorate %ubo_block 1 Offset 64
               OpMemberDecorate %ubo_block 2 Offset 128
               OpMemberDecorate %ubo_block 3 Offset 192
               OpMemberDecorate %ubo_block 4 Offset 208
               OpMemberDecorate %ubo_block 5 Offset 224
               OpDecorate %ubo_var DescriptorSet 0
               OpDecorate %ubo_var Binding 0

; --- Types ---
       %void = OpTypeVoid
       %func = OpTypeFunction %void
      %float = OpTypeFloat 32
        %int = OpTypeInt 32 1
      %vec3  = OpTypeVector %float 3
      %vec4  = OpTypeVector %float 4
   %mat4    = OpTypeMatrix %vec4 4
  %ubo_block = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
  %ptr_ubo = OpTypePointer Uniform %ubo_block
  %ptr_input = OpTypePointer Input %vec3
  %ptr_output = OpTypePointer Output %vec4
  %ptr_ubo_vec4 = OpTypePointer Uniform %vec4

; --- Globals ---
  %in_worldPos = OpVariable %ptr_input Input
  %in_normal = OpVariable %ptr_input Input
  %in_color = OpVariable %ptr_input Input
  %outColor = OpVariable %ptr_output Output
  %ubo_var = OpVariable %ptr_ubo Uniform

; --- Constants ---
    %int_3 = OpConstant %int 3
    %int_4 = OpConstant %int 4
    %float_0 = OpConstant %float 0
    %float_0_1 = OpConstant %float 0.1
    %float_0_6 = OpConstant %float 0.6
    %float_0_8 = OpConstant %float 0.8
    %float_64 = OpConstant %float 64
    %float_1 = OpConstant %float 1

; --- Main ---
       %main = OpFunction %void None %func
         %lb = OpLabel

; Load Inputs
         %p = OpLoad %vec3 %in_worldPos
         %n = OpLoad %vec3 %in_normal
         %norm = OpExtInst %vec3 %1 Normalize %n

; Load UBO
         %ptr_lp = OpAccessChain %ptr_ubo_vec4 %ubo_var %int_3
         %lp = OpLoad %vec4 %ptr_lp
         %ptr_cp = OpAccessChain %ptr_ubo_vec4 %ubo_var %int_4
         %cp = OpLoad %vec4 %ptr_cp

; Vectors
         %lp_vec3 = OpVectorShuffle %vec3 %lp %lp 0 1 2
         %cp_vec3 = OpVectorShuffle %vec3 %cp %cp 0 1 2

         %L_raw = OpFSub %vec3 %lp_vec3 %p
         %V_raw = OpFSub %vec3 %cp_vec3 %p
         %L = OpExtInst %vec3 %1 Normalize %L_raw
         %V = OpExtInst %vec3 %1 Normalize %V_raw
         %LV = OpFAdd %vec3 %L %V
         %H = OpExtInst %vec3 %1 Normalize %LV

; Diffuse
         %dotNL = OpDot %float %norm %L
         %diff = OpExtInst %float %1 FMax %dotNL %float_0

; Specular
         %dotNH = OpDot %float %norm %H
         %maxNH = OpExtInst %float %1 FMax %dotNH %float_0
         %spec = OpExtInst %float %1 Pow %maxNH %float_64

; Intensity: 0.1 + 0.6*diffuse + 0.8*specular
         %t1 = OpFMul %float %float_0_6 %diff
         %t2 = OpFMul %float %float_0_8 %spec
         %int1 = OpFAdd %float %float_0_1 %t1
         %intensity = OpFAdd %float %int1 %t2

; Output
         %out_vec = OpCompositeConstruct %vec4 %intensity %intensity %intensity %float_1
         OpStore %outColor %out_vec

               OpReturn
               OpFunctionEnd
```
