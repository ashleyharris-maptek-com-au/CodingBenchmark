#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 1. Procedural Brick Pattern
    vec2 scale = vec2(8.0, 16.0);
    vec2 scaledUV = uv * scale;

    // Staggered rows: shift every other row by 0.5
    if (int(floor(scaledUV.y)) % 2 == 0) {
        scaledUV.x += 0.5;
    }

    vec2 brick = floor(scaledUV);
    vec2 f = fract(scaledUV);

    // Mortar settings
    vec2 mortar = vec2(0.05, 0.1);
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);

    // Check if in mortar area
    bool isMortar = (f.x < mortar.x) || (f.y < mortar.y);
    vec3 baseColor = isMortar ? mortarColor : brickColor;

    // 2. Lambertian Shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diffuse;

    // Final color
    outColor = vec4(baseColor * lighting, 1.0);
}
