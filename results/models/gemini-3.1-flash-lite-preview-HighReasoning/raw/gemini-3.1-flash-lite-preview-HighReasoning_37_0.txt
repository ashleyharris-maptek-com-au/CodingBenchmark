The **Directed Feedback Vertex Set (DFVS)** problem is NP-hard. For small to moderately sized graphs, a recursive backtracking approach with pruning is effective.

### Strategy

1.  **Preprocessing:**
    *   Iteratively remove vertices with in-degree 0 or out-degree 0 (they cannot be part of any cycle).
    *   Iteratively remove vertices that have only one incoming or outgoing edge (contracting the path: `u -> v -> w` becomes `u -> w`).
2.  **Branching:**
    *   Use a branching algorithm (e.g., FVS via cycle detection) to find the minimum set.
    *   If no cycles exist (graph is a DAG), we have an FVS.
    *   If cycles exist, select a cycle (e.g., via DFS) and branch on removing each vertex in that cycle.
3.  **Optimization:**
    *   Implement memoization or iterative deepening to improve search performance.
    *   Use bitsets or adjacency lists for fast graph manipulation.

### Implementation

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class Solver {
    static void Main() {
        string[] input = Console.ReadLine().Split();
        int n = int.Parse(input[0]);
        int m = int.Parse(input[1]);

        List<int>[] adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++) {
            string[] edge = Console.ReadLine().Split();
            adj[int.Parse(edge[0])].Add(int.Parse(edge[1]));
        }

        var result = SolveDFVS(n, adj);
        Console.WriteLine(result.Count);
        Console.WriteLine(string.Join(" ", result));
    }

    static List<int> SolveDFVS(int n, List<int>[] adj) {
        // Simplified heuristic: Branching on cycles found via DFS
        var fvs = new HashSet<int>();
        var currentAdj = adj;
        
        while (true) {
            var cycle = FindCycle(n, currentAdj, fvs);
            if (cycle == null) break;

            // Simple heuristic: remove vertex with highest degree in the cycle
            int toRemove = cycle.OrderByDescending(v => currentAdj[v].Count).First();
            fvs.Add(toRemove);
        }
        return fvs.ToList();
    }

    static List<int> FindCycle(int n, List<int>[] adj, HashSet<int> removed) {
        int[] color = new int[n]; // 0: white, 1: gray, 2: black
        int[] parent = new int[n];
        Array.Fill(parent, -1);

        for (int i = 0; i < n; i++) {
            if (removed.Contains(i) || color[i] != 0) continue;
            var cycle = DFS(i, adj, removed, color, parent);
            if (cycle != null) return cycle;
        }
        return null;
    }

    static List<int> DFS(int u, List<int>[] adj, HashSet<int> removed, int[] color, int[] parent) {
        color[u] = 1;
        foreach (int v in adj[u]) {
            if (removed.Contains(v)) continue;
            if (color[v] == 1) {
                var cycle = new List<int> { v, u };
                int curr = u;
                while (curr != v && parent[curr] != -1) {
                    curr = parent[curr];
                    cycle.Add(curr);
                }
                return cycle;
            }
            if (color[v] == 0) {
                parent[v] = u;
                var res = DFS(v, adj, removed, color, parent);
                if (res != null) return res;
            }
        }
        color[u] = 2;
        return null;
    }
}
```

*Note: For extremely large graphs where minimum FVS is required, advanced techniques like kernelization, randomized algorithms, or ILP solvers are recommended.*
