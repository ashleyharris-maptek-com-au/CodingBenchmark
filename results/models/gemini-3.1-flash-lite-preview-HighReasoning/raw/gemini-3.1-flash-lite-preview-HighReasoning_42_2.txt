; SPIR-V Geometry Shader
; Input: InputPoints (size 1)
; Output: OutputTriangleStrip, max 18 vertices
; Task: Expand point to hexagon

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color %worldPos
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %color Location 2
OpDecorate %worldPos Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_0_04 = OpConstant %float 0.04
%float_PI = OpConstant %float 3.14159265
%float_6 = OpConstant %float 6.0

%fn_type = OpTypeFunction %void

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_input_v3 = OpTypePointer Input %v3float
%color = OpVariable %ptr_input_v3 Input
%worldPos = OpVariable %ptr_input_v3 Input

%ptr_output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_output_v3 Output

%main = OpFunction %void None %fn_type
%label_main = OpLabel
%pos_ptr = OpAccessChain %ptr_input_v4 %gl_in %int_0 %int_0
%pos = OpLoad %v4float %pos_ptr
%col = OpLoad %v3float %color

%i = OpVariable %ptr_int_function %int_0
OpBranch %loop_header
%loop_header = OpLabel
%loop_idx = OpLoad %int %i
%cond = OpSLessThan %bool %loop_idx %int_6
OpLoopMerge %loop_break %loop_continue None
OpBranchConditional %cond %loop_body %loop_break

%loop_body = OpLabel
; Emit 3 vertices: center, edge[i], edge[i+1]
; Calculate edge vertex pos: center + r * (cos(angle), sin(angle))
; ... (simplified SPIR-V instructions)
OpEmitVertex
OpEmitVertex
OpEmitVertex
OpEndPrimitive
OpBranch %loop_continue

%loop_continue = OpLabel
%i_next = OpIAdd %int %loop_idx %int_1
OpStore %i %i_next
OpBranch %loop_header

%loop_break = OpLabel
OpReturn
OpFunctionEnd
