; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %in_worldPos %outColor
OpExecutionMode %main OriginUpperLeft

; --- Layout Decorates ---
OpDecorate %outColor Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_worldPos Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo_var DescriptorSet 0
OpDecorate %ubo_var Binding 0

; --- Types ---
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%bool = OpTypeBool
%func_void = OpTypeFunction %void

; --- Constants ---
%const_0 = OpConstant %float 0.0
%const_0_5 = OpConstant %float 0.5
%const_1 = OpConstant %float 1.0
%const_2 = OpConstant %float 2.0
%const_int_0 = OpConstant %int 0
%const_int_4 = OpConstant %int 4

; --- Variables ---
%ubo_var = OpVariable %ptr_uniform_UBO Uniform
%in_normal = OpVariable %ptr_input_vec3 Input
%in_worldPos = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%main = OpFunction %void None %func_void
%label_main = OpLabel

%ptr_cameraPos = OpAccessChain %ptr_uniform_UBO %ubo_var %const_int_4
%cameraPos_vec4 = OpLoad %vec4 %ptr_cameraPos
%cameraPos_vec3 = OpVectorShuffle %vec3 %cameraPos_vec4 %cameraPos_vec4 0 1 2

%worldPos = OpLoad %vec3 %in_worldPos
%V_raw = OpFSub %vec3 %cameraPos_vec3 %worldPos
%V = OpExtInst %vec3 %1 Normalize %V_raw

%normal_raw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %1 Normalize %normal_raw

%dot_NV = OpDot %float %N %V
%NdotV = OpExtInst %float %1 FMax %dot_NV %const_0

%lt_05 = OpFLessThan %bool %NdotV %const_0_5

OpSelectionMerge %merge_label None
OpBranchConditional %lt_05 %if_lt %if_ge

%if_lt = OpLabel
%g_lt = OpFMul %float %const_2 %NdotV
%color_lt = OpCompositeConstruct %vec4 %const_1 %g_lt %const_0 %const_1
OpBranch %merge_label

%if_ge = OpLabel
%NdotV_minus_05 = OpFSub %float %NdotV %const_0_5
%val = OpFMul %float %const_2 %NdotV_minus_05
%rgb_val = OpFSub %float %const_1 %val
%color_ge = OpCompositeConstruct %vec4 %rgb_val %rgb_val %val %const_1
OpBranch %merge_label

%merge_label = OpLabel
%result_color = OpPhi %vec4 %color_lt %if_lt %color_ge %if_ge
OpStore %outColor %result_color
OpReturn
OpFunctionEnd
