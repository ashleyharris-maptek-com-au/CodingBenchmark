const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    if (!input.trim()) return;
    const data = JSON.parse(input);
    const { depot, truckCount, packages } = data;

    const routes = Array.from({ length: truckCount }, () => []);
    const remainingPackages = [...packages];

    // Simple greedy approach: 
    // Sort packages by distance from depot? 
    // Or just assign packages one by one to the current shortest route?
    
    // For now, let's distribute packages among trucks cyclically to ensure balance,
    // or use a simple nearest neighbor approach.
    
    // Nearest neighbor construction:
    const currentPositions = Array.from({ length: truckCount }, () => [...depot]);
    
    while (remainingPackages.length > 0) {
        let bestPackageIdx = -1;
        let bestTruckIdx = -1;
        let minDistance = Infinity;

        for (let t = 0; t < truckCount; t++) {
            for (let p = 0; p < remainingPackages.length; p++) {
                const pkg = remainingPackages[p];
                const dist = Math.sqrt(
                    Math.pow(pkg.x - currentPositions[t][0], 2) + 
                    Math.pow(pkg.y - currentPositions[t][1], 2)
                );
                
                if (dist < minDistance) {
                    minDistance = dist;
                    bestPackageIdx = p;
                    bestTruckIdx = t;
                }
            }
        }

        const pkg = remainingPackages.splice(bestPackageIdx, 1)[0];
        routes[bestTruckIdx].push(pkg.id);
        currentPositions[bestTruckIdx] = [pkg.x, pkg.y];
    }

    process.stdout.write(JSON.stringify({ routes }));
}

solve();
