; SPIR-V Geometry Shader
; Input: InputLines
; Output: OutputTriangleStrip, 4 Vertices

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %gl_in_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %outColor Location 0
OpDecorate %gl_in_color Location 2

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%v2float = OpTypeVector %float 2
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0_03 = OpConstant %float 0.03

; Interface Variables
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_pv_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_pv_out Output

%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_arr_v3 = OpTypePointer Input %arr_v3_2
%gl_in_color = OpVariable %ptr_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Main Function
%main = OpFunction %void None (OpTypeFunction %void)
%label = OpLabel

%p0 = OpLoad %v4float (OpAccessChain (OpTypePointer Input %v4float) %gl_in %int_0 %int_0)
%p1 = OpLoad %v4float (OpAccessChain (OpTypePointer Input %v4float) %gl_in %int_1 %int_0)
%c0 = OpLoad %v3float (OpAccessChain (OpTypePointer Input %v3float) %gl_in_color %int_0)
%c1 = OpLoad %v3float (OpAccessChain (OpTypePointer Input %v3float) %gl_in_color %int_1)

%dir_xy = OpFSub %v2float (OpVectorShuffle %v2float %p1 %p1 0 1) (OpVectorShuffle %v2float %p0 %p0 0 1)
%perp = OpVectorTimesScalar %v2float (OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %dir_xy 1)) (OpCompositeExtract %float %dir_xy 0)) %float_0_03

; Vertex emission omitted for brevity, logic proceeds as follows:
; emit p0+perp, c0
; emit p0-perp, c0
; emit p1+perp, c1
; emit p1-perp, c1

OpEndPrimitive
OpReturn
OpFunctionEnd
