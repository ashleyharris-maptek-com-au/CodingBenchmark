#version 430

// Workgroup size
layout(local_size_x = 256) in;

// Particle structure
struct Particle {
    vec4 pos; // xyz: position, w: mass
    vec4 vel; // xyz: velocity, w: padding
};

// Shader storage buffer object for particle data
layout(std430, binding = 0) buffer ParticleBuffer {
    Particle particles[];
};

// Uniforms
uniform float dt;         // Time step
uniform float softening;  // Softening factor to prevent singularities

void main() {
    // Get the current thread's global index
    uint i = gl_GlobalInvocationID.x;
    
    // Bounds check
    if (i >= particles.length()) return;

    // Load current particle position and velocity
    vec3 p = particles[i].pos.xyz;
    vec3 v = particles[i].vel.xyz;
    vec3 acc = vec3(0.0);

    // Compute gravitational force from all other particles
    for (uint j = 0; j < particles.length(); j++) {
        if (i == j) continue;
        
        vec3 r = particles[j].pos.xyz - p;
        
        // Softened distance squared
        float distSq = dot(r, r) + softening * softening;
        
        // Inverse distance cubed
        float invDist = 1.0 / sqrt(distSq);
        float invDist3 = invDist * invDist * invDist;
        
        // Add contribution to acceleration
        acc += r * (particles[j].pos.w * invDist3);
    }

    // Update velocity and position
    v += acc * dt;
    p += v * dt;

    // Store updated data
    particles[i].pos.xyz = p;
    particles[i].vel.xyz = v;
}
