#version 450

layout(local_size_x = 256) in;

layout(std430, binding = 0) buffer InputBuffer {
    uint data[];
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    // Naive inclusive prefix sum (for demonstration)
    // This is O(n^2) and not efficient, but simple.
    uint sum = 0;
    for (uint i = 0; i <= id; ++i) {
        sum += data[i];
    }
    data[id] = sum;
}
