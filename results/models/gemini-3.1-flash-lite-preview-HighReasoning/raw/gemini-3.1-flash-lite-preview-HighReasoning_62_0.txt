const fs = require('fs');

interface Town {
  id: string;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: [number, number][];
  towns: Town[];
  cutBudget: number;
}

function solve() {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);
  const { width, height, grid, ignitions, towns, cutBudget } = data;

  // BFS from all towns to find distance from fire source or town importance.
  // Actually, let's do BFS from ignitions to see which cells burn, 
  // and prioritize cutting cells that connect ignitions to towns.

  const fireDistance = Array.from({ length: height }, () => new Int32Array(width).fill(-1));
  const queue: [number, number][] = [];

  for (const [x, y] of ignitions) {
    if (x >= 0 && x < width && y >= 0 && y < height && grid[y][x] === '.') {
      fireDistance[y][x] = 0;
      queue.push([x, y]);
    }
  }

  let head = 0;
  while (head < queue.length) {
    const [x, y] = queue[head++];
    const d = fireDistance[y][x];

    const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
    for (const [dx, dy] of dirs) {
      const nx = x + dx;
      const ny = y + dy;
      if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.' && fireDistance[ny][nx] === -1) {
        fireDistance[ny][nx] = d + 1;
        queue.push([nx, ny]);
      }
    }
  }

  // Simple heuristic: pick cells that are between ignitions and towns.
  // For each town, find the path of fire that would reach it and cut a cell on it.
  const firebreaks: [number, number][] = [];
  let budgetUsed = 0;

  // Prioritize towns by value.
  const sortedTowns = [...towns].sort((a, b) => b.value - a.value);

  for (const town of sortedTowns) {
    if (budgetUsed >= cutBudget) break;

    // If the town is already reachable by fire (distance > -1), try to block it.
    if (fireDistance[town.y][town.x] !== -1) {
      // Find a cell on the path that is close to the town.
      // This is a very simple greedy approach.
      let currX = town.x;
      let currY = town.y;
      
      // Backtrack from town towards the ignition
      while (budgetUsed < cutBudget) {
        let bestNext: [number, number] | null = null;
        let minD = fireDistance[currY][currX];

        const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        for (const [dx, dy] of dirs) {
          const nx = currX + dx;
          const ny = currY + dy;
          if (nx >= 0 && nx < width && ny >= 0 && ny < height && fireDistance[ny][nx] !== -1 && fireDistance[ny][nx] < minD) {
            minD = fireDistance[ny][nx];
            bestNext = [nx, ny];
          }
        }

        if (!bestNext) break;
        
        // Block the path
        const [nx, ny] = bestNext;
        firebreaks.push([nx, ny]);
        budgetUsed++;
        
        // Stop blocking for this town if we reached near an ignition or path broken
        if (minD <= 1) break;
        
        currX = nx;
        currY = ny;
      }
    }
  }

  process.stdout.write(JSON.stringify({ firebreaks }));
}

solve();
