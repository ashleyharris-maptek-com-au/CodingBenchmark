const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const { width, height, grid, ignitions, towns, cutBudget } = data;

    // Use BFS to find the distance of each forest cell from the nearest ignition source
    const dist = Array.from({ length: height }, () => Array(width).fill(Infinity));
    const queue = [];

    for (const [x, y] of ignitions) {
        if (x >= 0 && x < width && y >= 0 && y < height && grid[y][x] === '.') {
            dist[y][x] = 0;
            queue.push([x, y]);
        }
    }

    let head = 0;
    while (head < queue.length) {
        const [x, y] = queue[head++];
        const d = dist[y][x];

        const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        for (const [dx, dy] of dirs) {
            const nx = x + dx;
            const ny = y + dy;

            if (nx >= 0 && nx < width && ny >= 0 && ny < height && grid[ny][nx] === '.' && dist[ny][nx] === Infinity) {
                dist[ny][nx] = d + 1;
                queue.push([nx, ny]);
            }
        }
    }

    // Identify critical cells: these are cells that are on paths to towns
    // We prioritize cutting cells that protect high-value towns that would burn soon.
    const candidates = [];
    for (const town of towns) {
        const { x, y, value } = town;
        if (dist[y][x] !== Infinity) {
            candidates.push({ x, y, value, d: dist[y][x] });
        }
    }

    // Sort candidates by value/distance ratio (heuristic)
    candidates.sort((a, b) => (b.value / (b.d + 1)) - (a.value / (a.d + 1)));

    const firebreaks = [];
    const used = new Set();
    let cutsLeft = cutBudget;

    for (const cand of candidates) {
        if (cutsLeft <= 0) break;
        
        // Simple heuristic: try to cut the cell just before the fire reaches the town
        // This is a basic approach and could be improved with more sophisticated pathing.
        const x = cand.x;
        const y = cand.y;
        
        // Find a neighbor that is closer to the ignition
        let bestPrev = null;
        let minDist = dist[y][x];
        
        const dirs = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        for (const [dx, dy] of dirs) {
            const nx = x + dx;
            const ny = y + dy;
            if (nx >= 0 && nx < width && ny >= 0 && ny < height && dist[ny][nx] < minDist) {
                minDist = dist[ny][nx];
                bestPrev = [nx, ny];
            }
        }
        
        if (bestPrev) {
            const [bx, by] = bestPrev;
            const key = `${bx},${by}`;
            if (!used.has(key)) {
                firebreaks.push([bx, by]);
                used.add(key);
                cutsLeft--;
            }
        }
    }

    console.log(JSON.stringify({ firebreaks }));
}

solve();
