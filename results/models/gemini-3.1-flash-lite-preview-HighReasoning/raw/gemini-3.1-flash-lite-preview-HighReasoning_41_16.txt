; SPIR-V
; Version: 1.0
; Bound: 100
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %uv
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %uv Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%bool = OpTypeBool
%int = OpTypeInt 32 1
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output_vec4 = OpTypePointer Output %vec4
%ptr_func_float = OpTypePointer Function %float
%ptr_func_int = OpTypePointer Function %int
%void_func = OpTypeFunction %void

; Inputs/Outputs
%uv = OpVariable %ptr_input_vec2 Input
%outColor = OpVariable %ptr_output_vec4 Output

; Constants
%c0 = OpConstant %float 0.0
%c0_5 = OpConstant %float 0.5
%c1 = OpConstant %float 1.0
%c2 = OpConstant %float 2.0
%c3 = OpConstant %float 3.0
%c4 = OpConstant %float 4.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%max_iter = OpConstant %int 20
%c_max_iter = OpConstant %float 20.0

%main = OpFunction %void None %void_func
%entry = OpLabel

; c = (uv.x*3-2, uv.y*2-1)
%uv_val = OpLoad %vec2 %uv
%uv_x = OpCompositeExtract %float %uv_val 0
%uv_y = OpCompositeExtract %float %uv_val 1
%cx_mul = OpFMul %float %uv_x %c3
%cx = OpFSub %float %cx_mul %c2
%cy_mul = OpFMul %float %uv_y %c2
%cy = OpFSub %float %cy_mul %c1

; z = (0, 0)
%zx = OpVariable %ptr_func_float Function
%zy = OpVariable %ptr_func_float Function
OpStore %zx %c0
OpStore %zy %c0
%i = OpVariable %ptr_func_int Function
OpStore %i %int_0

OpBranch %loop

%loop = OpLabel
%i_val = OpLoad %int %i
%cmp_i = OpSLessThan %bool %i_val %max_iter
OpLoopMerge %loop_end %loop_continue None
OpBranchConditional %cmp_i %loop_body %loop_end

%loop_body = OpLabel
%zx_val = OpLoad %float %zx
%zy_val = OpLoad %float %zy

; |z|^2 = zx*zx + zy*zy
%zx2 = OpFMul %float %zx_val %zx_val
%zy2 = OpFMul %float %zy_val %zy_val
%mag2 = OpFAdd %float %zx2 %zy2
%cmp_mag = OpFOrdGreaterThan %bool %mag2 %c4
OpSelectionMerge %if_merge None
OpBranchConditional %cmp_mag %loop_end %if_continue

%if_continue = OpLabel
; new_zx = zx*zx - zy*zy + cx
; new_zy = 2*zx*zy + cy
%new_zx = OpFSub %float %zx2 %zy2
%new_zx_c = OpFAdd %float %new_zx %cx
%zx_zy = OpFMul %float %zx_val %zy_val
%new_zy_2 = OpFMul %float %c2 %zx_zy
%new_zy = OpFAdd %float %new_zy_2 %cy

OpStore %zx %new_zx_c
OpStore %zy %new_zy
OpBranch %if_merge

%if_merge = OpLabel
OpBranch %loop_continue

%loop_continue = OpLabel
%next_i = OpIAdd %int %i_val %int_1
OpStore %i %next_i
OpBranch %loop

%loop_end = OpLabel
; t = iterations / 20
%final_i = OpLoad %int %i
%t = OpConvertSToF %float %final_i
%t_norm = OpFDiv %float %t %c_max_iter

; color = (t, t*0.5, 1.0-t)
%red = %t_norm
%green = OpFMul %float %t_norm %c0_5
%blue = OpFSub %float %c1 %t_norm
%color = OpCompositeConstruct %vec4 %red %green %blue %c1
OpStore %outColor %color
OpReturn
OpFunctionEnd
