use std::io::{self, Read};
use std::collections::HashSet;

fn simple_shuffle(nodes: &mut Vec<usize>, seed: &mut u64) {
    let n = nodes.len();
    for i in 0..n {
        *seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
        let j = (*seed as usize) % n;
        nodes.swap(i, j);
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut words = input.split_whitespace();
    
    let n: usize = words.next().expect("n").parse().expect("n parse");
    let m: usize = words.next().expect("m").parse().expect("m parse");

    let mut adj = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = words.next().expect("u").parse().expect("u parse");
        let v: usize = words.next().expect("v").parse().expect("v parse");
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut nodes: Vec<usize> = (0..n).collect();
    let mut seed = 42;

    let mut best_cut = isize::MAX;
    let mut best_partition = Vec::new();

    // Perform multiple random restarts
    for _ in 0..5 {
        simple_shuffle(&mut nodes, &mut seed);
        let mut partition_a: HashSet<usize> = nodes.iter().take(n / 2).cloned().collect();
        
        let mut current_cut: isize = 0;
        for u in 0..n {
            for &v in &adj[u] {
                if partition_a.contains(&u) != partition_a.contains(&v) {
                    current_cut += 1;
                }
            }
        }
        current_cut /= 2;

        let mut improved = true;
        let mut count = 0;
        while improved && count < 100 {
            improved = false;
            count += 1;
            for _ in 0..n * 10 {
                seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
                let u = (seed as usize) % n;
                seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
                let v = (seed as usize) % n;

                if partition_a.contains(&u) != partition_a.contains(&v) {
                    let u_is_a = partition_a.contains(&u);
                    
                    let mut diff: isize = 0;
                    for &neighbor in &adj[u] {
                        if partition_a.contains(&neighbor) == u_is_a {
                            diff += 1;
                        } else {
                            diff -= 1;
                        }
                    }
                    for &neighbor in &adj[v] {
                        if partition_a.contains(&neighbor) != u_is_a {
                            diff += 1;
                        } else {
                            diff -= 1;
                        }
                    }
                    
                    let mut connected = false;
                    for &neighbor in &adj[u] {
                        if neighbor == v {
                            connected = true;
                            break;
                        }
                    }
                    if connected {
                        diff -= 2;
                    }

                    if diff < 0 {
                        if partition_a.contains(&u) {
                            partition_a.remove(&u);
                            partition_a.insert(v);
                        } else {
                            partition_a.remove(&v);
                            partition_a.insert(u);
                        }
                        // Recalculate current_cut for safety
                        let mut new_cut = 0;
                        for i in 0..n {
                            for &neighbor in &adj[i] {
                                if partition_a.contains(&i) != partition_a.contains(&neighbor) {
                                    new_cut += 1;
                                }
                            }
                        }
                        current_cut = new_cut / 2;
                        improved = true;
                    }
                }
            }
        }

        if current_cut < best_cut {
            best_cut = current_cut;
            best_partition = partition_a.iter().cloned().collect();
        }
    }

    println!("{}", best_cut);
    let mut out_vec: Vec<usize> = best_partition.iter().cloned().collect();
    out_vec.sort();
    let out_str: Vec<String> = out_vec.iter().map(|n| n.to_string()).collect();
    println!("{}", out_str.join(" "));
}
