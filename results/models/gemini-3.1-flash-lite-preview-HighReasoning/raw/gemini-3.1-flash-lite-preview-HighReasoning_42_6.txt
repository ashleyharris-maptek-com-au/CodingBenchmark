; Passthrough Geometry Shader
OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %outColor %color
OpExecutionMode %main Triangles
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3

; Decorate
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %outColor Location 0
OpDecorate %color Location 2

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%void_func = OpTypeFunction %void

%main = OpFunction %void None %void_func
%label = OpLabel

; Loop indices
%idx = OpVariable %ptr_in_v4 None ; Reusing for loop counter
OpStore %idx (OpConstantNull %v4float) ; Placeholder for integer loop index logic
%i = OpVariable %ptr_in_v4 None ; Wait, simpler to unroll 3 vertices since it's just 3

; Vertex 0
%pos0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%load_pos0 = OpLoad %v4float %pos0
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %load_pos0
%col0 = OpAccessChain %ptr_in_v3 %color %int_0
%load_col0 = OpLoad %v3float %col0
OpStore %outColor %load_col0
OpEmitVertex

; Vertex 1
%pos1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%load_pos1 = OpLoad %v4float %pos1
OpStore %out_pos0 %load_pos1
%col1 = OpAccessChain %ptr_in_v3 %color %int_1
%load_col1 = OpLoad %v3float %col1
OpStore %outColor %load_col1
OpEmitVertex

; Vertex 2
%pos2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%load_pos2 = OpLoad %v4float %pos2
OpStore %out_pos0 %load_pos2
%col2 = OpAccessChain %ptr_in_v3 %color %int_2
%load_col2 = OpLoad %v3float %col2
OpStore %outColor %load_col2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
