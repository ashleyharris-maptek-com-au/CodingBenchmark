interface Job {
  id: string | number;
  x: number;
  y: number;
  priority: number;
}

interface Robot {
  id: number;
  x: number;
  y: number;
}

interface Input {
  grid: string[];
  robots: Robot[];
  jobs: Job[];
}

function solve() {
  const fs = require('fs');
  const input: string = fs.readFileSync(0, 'utf8');
  const data: Input = JSON.parse(input);

  const { grid, robots, jobs } = data;
  const rows = grid.length;
  const cols = grid[0].length;

  // BFS to calculate shortest path distances from a start point
  function getDistances(startX: number, startY: number): (number | null)[][] {
    const dists: (number | null)[][] = Array.from({ length: rows }, () => Array(cols).fill(null));
    const queue: [number, number, number][] = [[startY, startX, 0]];
    dists[startY][startX] = 0;

    while (queue.length > 0) {
      const [r, c, d] = queue.shift()!;

      const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
      for (const [dr, dc] of directions) {
        const nr = r + dr;
        const nc = c + dc;

        if (nr >= 0 && nr < rows && nc >= 0 && nc < cols && grid[nr][nc] === '.' && dists[nr][nc] === null) {
          dists[nr][nc] = d + 1;
          queue.push([nr, nc, d + 1]);
        }
      }
    }
    return dists;
  }

  // Precompute distances for each robot
  const robotDistMaps = robots.map(r => getDistances(r.x, r.y));

  const assignments: (string | number)[][] = Array.from({ length: robots.length }, () => []);
  const jobAssigned = new Array(jobs.length).fill(false);

  // Greedy assignment
  // Sort jobs by priority (descending)
  const sortedJobs = jobs.map((job, index) => ({ ...job, index }))
    .sort((a, b) => b.priority - a.priority);

  for (const job of sortedJobs) {
    let minTravel = Infinity;
    let bestRobotIdx = -1;

    for (let i = 0; i < robots.length; i++) {
      const dist = robotDistMaps[i][job.y][job.x];
      if (dist !== null && dist < minTravel) {
        minTravel = dist;
        bestRobotIdx = i;
      }
    }

    if (bestRobotIdx !== -1) {
      assignments[bestRobotIdx].push(job.id);
      jobAssigned[job.index] = true;
      // Update robot position for next job assignment
      robotDistMaps[bestRobotIdx] = getDistances(job.x, job.y);
    }
  }

  process.stdout.write(JSON.stringify({ assignments }));
}

solve();
