The following is a C# solution for the Quadratic Assignment Problem (QAP) that implements a Simulated Annealing heuristic. It is optimized to run efficiently for large N (up to 2000) within 5 minutes.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;

class QapSolver
{
    static void Main(string[] args)
    {
        // Reading all input from stdin
        string inputData = Console.In.ReadToEnd();
        string[] input = inputData.Split(new[] { ' ', '\r', '\n', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        
        if (input.Length == 0) return;

        int ptr = 0;
        int n = int.Parse(input[ptr++]);
        
        int[,] flow = new int[n, n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                flow[i, j] = int.Parse(input[ptr++]);
                
        int[,] dist = new int[n, n];
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                dist[i, j] = int.Parse(input[ptr++]);

        Solve(n, flow, dist);
    }

    static void Solve(int n, int[,] flow, int[,] dist)
    {
        Random rng = new Random(42);
        int[] p = Enumerable.Range(0, n).ToArray();
        
        // Initial random shuffle
        for (int i = n - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            int tmp = p[i];
            p[i] = p[j];
            p[j] = tmp;
        }

        long currentCost = CalculateCost(n, flow, dist, p);
        
        Stopwatch sw = Stopwatch.StartNew();
        int iterations = 0;
        
        // Simulated Annealing
        double temp = 1000.0;
        double cooling = 0.999999;

        while (sw.ElapsedMilliseconds < 4500)
        {
            int i = rng.Next(n);
            int j = rng.Next(n);
            if (i == j) continue;

            // Swap and evaluate delta cost
            long delta = CalculateSwapDelta(n, flow, dist, p, i, j);

            if (delta < 0 || (temp > 1e-9 && Math.Exp(-delta / temp) > rng.NextDouble()))
            {
                int tmp = p[i];
                p[i] = p[j];
                p[j] = tmp;
                currentCost += delta;
            }

            temp *= cooling;
            iterations++;
        }

        Console.WriteLine(currentCost);
        Console.WriteLine(string.Join(" ", p));
    }

    static long CalculateCost(int n, int[,] flow, int[,] dist, int[] p)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                cost += (long)flow[i, j] * dist[p[i], p[j]];
        return cost;
    }

    static long CalculateSwapDelta(int n, int[,] flow, int[,] dist, int[] p, int i, int j)
    {
        long delta = 0;
        
        // Only re-evaluate affected terms
        for (int k = 0; k < n; k++)
        {
            if (k == i || k == j) continue;
            
            delta += (long)flow[i, k] * dist[p[i], p[k]] + (long)flow[k, i] * dist[p[k], p[i]] +
                     (long)flow[j, k] * dist[p[j], p[k]] + (long)flow[k, j] * dist[p[k], p[j]];
                     
            delta -= (long)flow[i, k] * dist[p[j], p[k]] + (long)flow[k, i] * dist[p[k], p[j]] +
                     (long)flow[j, k] * dist[p[i], p[k]] + (long)flow[k, j] * dist[p[k], p[i]];
        }
        
        delta += (long)flow[i, i] * dist[p[i], p[i]] + (long)flow[j, j] * dist[p[j], p[j]] + 
                 (long)flow[i, j] * dist[p[i], p[j]] + (long)flow[j, i] * dist[p[j], p[i]];
        delta -= (long)flow[i, i] * dist[p[j], p[j]] + (long)flow[j, j] * dist[p[i], p[i]] + 
                 (long)flow[i, j] * dist[p[j], p[i]] + (long)flow[j, i] * dist[p[i], p[j]];
                 
        return delta;
    }
}
```
