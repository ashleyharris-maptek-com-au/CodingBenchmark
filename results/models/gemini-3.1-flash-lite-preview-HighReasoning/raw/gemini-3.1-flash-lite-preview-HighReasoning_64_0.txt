const fs = require('fs');

interface InputData {
    probability: number[][];
    drones: { id: number; x: number; y: number }[];
    maxStepsPerDrone: number;
}

function solve() {
    const input: string = fs.readFileSync(0, 'utf8');
    if (!input) return;
    const data: InputData = JSON.parse(input);

    const rows = data.probability.length;
    const cols = data.probability[0].length;
    const paths: [number, number][][] = [];

    // Simple greedy strategy:
    // Move to the highest available probability neighbor within range.
    for (const drone of data.drones) {
        let currX = drone.x;
        let currY = drone.y;
        const path: [number, number][] = [[currX, currY]];
        
        for (let step = 0; step < data.maxStepsPerDrone; step++) {
            let bestX = currX;
            let bestY = currY;
            let maxProb = -1;

            const neighbors = [
                [currX, currY],
                [currX + 1, currY],
                [currX - 1, currY],
                [currX, currY + 1],
                [currX, currY - 1]
            ];

            for (const [nx, ny] of neighbors) {
                if (nx >= 0 && nx < rows && ny >= 0 && ny < cols && data.probability[nx][ny] !== -1) {
                    if (data.probability[nx][ny] > maxProb) {
                        maxProb = data.probability[nx][ny];
                        bestX = nx;
                        bestY = ny;
                    }
                }
            }

            currX = bestX;
            currY = bestY;
            path.push([currX, currY]);
        }
        paths.push(path);
    }

    process.stdout.write(JSON.stringify({ paths }));
}

solve();
