const fs = require('fs');

function solve() {
  const input = fs.readFileSync(0, 'utf8');
  const { grid, robots, jobs } = JSON.parse(input);

  const rows = grid.length;
  const cols = grid[0].length;

  // BFS to find all distances from a starting point
  function getDistances(startX, startY) {
    const dists = Array.from({ length: rows }, () => Array(cols).fill(-1));
    const queue = [[startX, startY, 0]];
    dists[startY][startX] = 0;

    let head = 0;
    while (head < queue.length) {
      const [x, y, d] = queue[head++];

      const neighbors = [[x + 1, y], [x - 1, y], [x, y + 1], [x, y - 1]];
      for (const [nx, ny] of neighbors) {
        if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] === '.' && dists[ny][nx] === -1) {
          dists[ny][nx] = d + 1;
          queue.push([nx, ny, d + 1]);
        }
      }
    }
    return dists;
  }

  // Precompute robot distances to all jobs
  const robotJobDistances = robots.map(robot => {
    const dists = getDistances(robot.x, robot.y);
    return jobs.map(job => dists[job.y][job.x]);
  });

  const assignments = Array.from({ length: robots.length }, () => []);
  const jobTaken = new Array(jobs.length).fill(false);

  // Greedy assignment: assign closest available job to a robot
  // A simple heuristic: always pick the shortest distance among all unassigned jobs
  const remainingJobs = jobs.map((_, index) => index);
  
  while (remainingJobs.length > 0) {
    let minD = Infinity;
    let bestRobot = -1;
    let bestJobIdx = -1;

    for (let r = 0; r < robots.length; r++) {
      for (let i = 0; i < remainingJobs.length; i++) {
        const jIdx = remainingJobs[i];
        const dist = robotJobDistances[r][jIdx];
        if (dist !== -1 && dist < minD) {
          minD = dist;
          bestRobot = r;
          bestJobIdx = i;
        }
      }
    }

    if (bestRobot === -1) {
      // If a job is unreachable, assign it to the first robot (fallback)
      const jIdx = remainingJobs.shift();
      assignments[0].push(jobs[jIdx].id);
    } else {
      const jIdx = remainingJobs.splice(bestJobIdx, 1)[0];
      assignments[bestRobot].push(jobs[jIdx].id);
    }
  }

  process.stdout.write(JSON.stringify({ assignments }));
}

solve();
