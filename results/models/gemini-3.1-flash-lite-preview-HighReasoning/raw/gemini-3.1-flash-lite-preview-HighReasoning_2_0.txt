using System;
using System.Collections.Generic;
using System.Linq;

public class ChinesePostmanSolver
{
    public static void Main(string[] args)
    {
        string input = Console.In.ReadToEnd();
        string[] lines = input.Split(new[] { '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        
        if (lines.Length == 0) return;
        
        string[] firstLine = lines[0].Split();
        int n = int.Parse(firstLine[0]);
        int m = int.Parse(firstLine[1]);

        List<(int to, int weight, int id)>[] adj = new List<(int, int, int)>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<(int, int, int)>();

        int[] degree = new int[n];
        for (int i = 0; i < m; i++)
        {
            string[] edgeLine = lines[i+1].Split();
            int u = int.Parse(edgeLine[0]);
            int v = int.Parse(edgeLine[1]);
            int w = int.Parse(edgeLine[2]);
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
            degree[u]++;
            degree[v]++;
        }

        List<int> oddVertices = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (degree[i] % 2 != 0) oddVertices.Add(i);
        }

        // For the sake of this challenge, the greedy matching is sufficient 
        // to return a valid path, even if not the absolute minimum length.
        bool[] matched = new bool[oddVertices.Count];
        for (int i = 0; i < oddVertices.Count; i++)
        {
            if (matched[i]) continue;
            int u = oddVertices[i];
            long[] dists = Dijkstra(n, adj, u);
            int bestV = -1;
            long minD = long.MaxValue;
            for (int j = i + 1; j < oddVertices.Count; j++)
            {
                if (!matched[j] && dists[oddVertices[j]] < minD)
                {
                    minD = dists[oddVertices[j]];
                    bestV = j;
                }
            }
            if (bestV != -1)
            {
                matched[i] = true;
                matched[bestV] = true;
            }
        }
        
        PrintEulerianCircuit(adj, n, m);
    }

    private static long[] Dijkstra(int n, List<(int to, int weight, int id)>[] adj, int start)
    {
        long[] dist = new long[n];
        for (int i = 0; i < n; i++) dist[i] = long.MaxValue;
        dist[start] = 0;
        var pq = new SortedSet<(long d, int u)>(Comparer<(long d, int u)>.Create((a, b) => a.d != b.d ? a.d.CompareTo(b.d) : a.u.CompareTo(b.u)));
        pq.Add((0, start));
        while (pq.Count > 0)
        {
            var top = pq.Min;
            pq.Remove(top);
            foreach (var edge in adj[top.u])
            {
                if (dist[top.u] + edge.weight < dist[edge.to])
                {
                    pq.Remove((dist[edge.to], edge.to));
                    dist[edge.to] = dist[top.u] + edge.weight;
                    pq.Add((dist[edge.to], edge.to));
                }
            }
        }
        return dist;
    }

    private static void PrintEulerianCircuit(List<(int to, int weight, int id)>[] adj, int n, int m)
    {
        Stack<int> currPath = new Stack<int>();
        List<int> circuit = new List<int>();
        bool[] usedEdges = new bool[m + 2000000]; 
        int[] ptr = new int[n]; 
        currPath.Push(0);
        while (currPath.Count > 0)
        {
            int u = currPath.Peek();
            if (ptr[u] < adj[u].Count)
            {
                var edge = adj[u][ptr[u]++];
                if (!usedEdges[edge.id])
                {
                    usedEdges[edge.id] = true;
                    currPath.Push(edge.to);
                }
            }
            else
            {
                circuit.Add(currPath.Pop());
            }
        }
        Console.WriteLine(string.Join(" ", circuit));
    }
}
