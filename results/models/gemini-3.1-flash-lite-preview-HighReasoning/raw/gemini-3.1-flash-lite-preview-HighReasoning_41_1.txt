; Generator: manual
; Bound: 30
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %in_normal Location 1
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_input_vec3 = OpTypePointer Input %vec3
%ptr_output_vec4 = OpTypePointer Output %vec4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_5 = OpConstant %float 0.5
%vec3_0_5 = OpConstantComposite %vec3 %float_0_5 %float_0_5 %float_0_5

; Variables
%in_normal = OpVariable %ptr_input_vec3 Input
%outColor = OpVariable %ptr_output_vec4 Output

%func_main = OpTypeFunction %void

%main = OpFunction %void None %func_main
%entry = OpLabel
%normal_raw = OpLoad %vec3 %in_normal
%normal = OpExtInst %vec3 %1 Normalize %normal_raw
; Map [-1, 1] to [0, 1]
%scaled = OpVectorTimesScalar %vec3 %normal %float_0_5
%mapped = OpFAdd %vec3 %scaled %vec3_0_5
; Construct vec4 with alpha 1.0
%color = OpCompositeExtract %float %mapped 0
%color1 = OpCompositeExtract %float %mapped 1
%color2 = OpCompositeExtract %float %mapped 2
%final_color = OpCompositeConstruct %vec4 %color %color1 %color2 %float_1
OpStore %outColor %final_color
OpReturn
OpFunctionEnd

; External instruction set for math (required for Normalize)
%1 = OpExtInstImport "GLSL.std.450"
