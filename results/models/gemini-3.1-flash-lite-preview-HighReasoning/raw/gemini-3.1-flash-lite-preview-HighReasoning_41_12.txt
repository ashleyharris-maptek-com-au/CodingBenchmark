; SPIR-V
; Version: 1.0
OpCapability Shader
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv %tangent %color
OpExecutionMode %main OriginUpperLeft
OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_input = OpTypePointer Input %vec3
%ptr_input_vec2 = OpTypePointer Input %vec2
%ptr_output = OpTypePointer Output %vec4

%uboVar = OpVariable %ptr_ubo Uniform
%worldPos = OpVariable %ptr_input Input
%normal = OpVariable %ptr_input Input
%uv = OpVariable %ptr_input_vec2 Input
%tangent = OpVariable %ptr_input Input
%color = OpVariable %ptr_input Input
%outColor = OpVariable %ptr_output Output

%main = OpFunction %void None %func
%label = OpLabel
%n_raw = OpLoad %vec3 %normal
%n = OpExtInst %vec3 %GLSLstd450 Normalize %n_raw
%ny = OpCompositeExtract %float %n 1
%t = OpFMul %float %ny %float_0_5
%t_final = OpFAdd %float %t %float_0_5

%sky = OpConstantComposite %vec3 %float_0_4 %float_0_6 %float_1_0
%ground = OpConstantComposite %vec3 %float_0_3 %float_0_15 %float_0_05
%ambient = OpExtInst %vec3 %GLSLstd450 FMix %ground %sky %t_final

%int_3 = OpConstant %int 3
%l_pos_ptr = OpAccessChain %ptr_ubo %uboVar %int_3
%l_pos = OpLoad %vec4 %l_pos_ptr
%light_dir = OpVectorShuffle %vec3 %l_pos %l_pos 0 1 2
%w_pos = OpLoad %vec3 %worldPos
%v = OpFSub %vec3 %light_dir %w_pos
%v_norm = OpExtInst %vec3 %GLSLstd450 Normalize %v
%dot_nv = OpExtInst %float %GLSLstd450 Dot %n %v_norm
%d = OpExtInst %float %GLSLstd450 FMax %float_0 %dot_nv
%diffuse = OpFMul %float %float_0_3 %d
%diffuse_vec = OpCompositeConstruct %vec3 %diffuse %diffuse %diffuse
%rgb = OpFAdd %vec3 %ambient %diffuse_vec
%rgba = OpCompositeConstruct %vec4 %rgb %float_1_0
OpStore %outColor %rgba
OpReturn
OpFunctionEnd

%float_0 = OpConstant %float 0.0
%float_0_3 = OpConstant %float 0.3
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_1_0 = OpConstant %float 1.0
%float_0_05 = OpConstant %float 0.05
%float_0_15 = OpConstant %float 0.15
