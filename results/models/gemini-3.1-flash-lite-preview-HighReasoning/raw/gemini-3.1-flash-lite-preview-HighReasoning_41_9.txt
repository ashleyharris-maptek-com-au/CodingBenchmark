OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_color %outColor %ubo
OpExecutionMode %main OriginUpperLeft

; Decorations
OpDecorate %outColor Location 0
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_color Location 4
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec2 = OpTypeVector %float 2
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBO
%ptr_vec4 = OpTypePointer Uniform %vec4
%ptr_in_vec3 = OpTypePointer Input %vec3
%ptr_in_vec2 = OpTypePointer Input %vec2
%ptr_out_vec4 = OpTypePointer Output %vec4
%void_fn = OpTypeFunction %void
%int = OpTypeInt 32 1
%bool = OpTypeBool
%ptr_float = OpTypePointer Function %float

; Variables
%ubo = OpVariable %ptr_UBO Uniform
%in_worldPos = OpVariable %ptr_in_vec3 Input
%in_normal = OpVariable %ptr_in_vec3 Input
%in_uv = OpVariable %ptr_in_vec2 Input
%in_color = OpVariable %ptr_in_vec3 Input
%outColor = OpVariable %ptr_out_vec4 Output

%main = OpFunction %void None %void_fn
%label_main = OpLabel

; Lambertian Shading
%normal = OpLoad %vec3 %in_normal
%normal_n = OpExtInst %vec3 %1 Normalize %normal ; Assuming GLSL.std.450 imported as %1
%worldPos = OpLoad %vec3 %in_worldPos
%ptr_lightPos = OpAccessChain %ptr_vec4 %ubo %int 3
%lightPos = OpLoad %vec4 %ptr_lightPos
%lightPos_3 = OpVectorShuffle %vec3 %lightPos %lightPos 0 1 2
%lightDir = OpFSub %vec3 %lightPos_3 %worldPos
%lightDir_n = OpExtInst %vec3 %1 Normalize %lightDir
%dot = OpDot %float %normal_n %lightDir_n
%diffuse = OpExtInst %float %1 FMax %dot %float_0
%shade = OpFAdd %float %float_0_5 %OpFMul %float_0_5 %diffuse

; Checkerboard
%uv = OpLoad %vec2 %in_uv
%uv_scaled = OpVectorTimesScalar %vec2 %uv %float_8
%uv_floor = OpFloor %vec2 %uv_scaled
%parity_vec = OpFAdd %vec2 %uv_floor %uv_floor ; Parity check
; simplified checker logic: (floor(u*8) + floor(v*8)) % 2
%parity_sum = OpFAdd %float %uv_floor_x %uv_floor_y ; Need to extract components
%parity = OpFMod %float %parity_sum %float_2
%is_even = OpFOrdEqual %bool %parity %float_0
%checker = OpSelect %float %is_even %float_0_9 %float_0_2

%final_rgb = OpVectorTimesScalar %vec3 %vec3_ones %checker
%final_color = OpCompositeConstruct %vec4 %final_rgb %float_1

OpStore %outColor %final_color
OpReturn
OpFunctionEnd
