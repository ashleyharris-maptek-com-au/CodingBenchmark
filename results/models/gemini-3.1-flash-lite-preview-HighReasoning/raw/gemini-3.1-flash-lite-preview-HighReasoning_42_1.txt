; SPIR-V Geometry Shader: Expand Point to Diamond
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

               ; Types
               %void = OpTypeVoid
               %float = OpTypeFloat 32
               %v4float = OpTypeVector %float 4
               %v3float = OpTypeVector %float 3
               %uint = OpTypeInt 32 0
               %int = OpTypeInt 32 1
               %uint_1 = OpConstant %uint 1
               %uint_4 = OpConstant %uint 4
               %int_0 = OpConstant %int 0
               %float_0 = OpConstant %float 0.0
               %float_h = OpConstant %float 0.05
               %float_neg_h = OpConstant %float -0.05

               ; Input/Output structs
               %gl_PerVertex_in = OpTypeStruct %v4float
               %arr_in = OpTypeArray %gl_PerVertex_in %uint_1
               %ptr_arr_in = OpTypePointer Input %arr_in
               %gl_in = OpVariable %ptr_arr_in Input

               %color_arr = OpTypeArray %v3float %uint_1
               %ptr_color_arr = OpTypePointer Input %color_arr
               %color_in = OpVariable %ptr_color_arr Input

               %gl_PerVertex_out = OpTypeStruct %v4float
               %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
               %gl_out = OpVariable %ptr_out_pv Output
               
               %ptr_out_color = OpTypePointer Output %v3float
               %outColor = OpVariable %ptr_out_color Output

               %ptr_v4 = OpTypePointer Output %v4float
               %ptr_v3 = OpTypePointer Output %v3float

               %void_fn = OpTypeFunction %void

               %main = OpFunction %void None %void_fn
               %entry = OpLabel

               ; Load input point
               %p0_ptr = OpAccessChain %ptr_v4 %gl_in %int_0 %int_0
               %p0 = OpLoad %v4float %p0_ptr
               %x = OpCompositeExtract %float %p0 0
               %y = OpCompositeExtract %float %p0 1
               %z = OpCompositeExtract %float %p0 2
               %w = OpCompositeExtract %float %p0 3

               ; Load color
               %c0_ptr = OpAccessChain %ptr_v3 %color_in %int_0
               %c0 = OpLoad %v3float %c0_ptr

               ; Function to emit vertex
               ; Vertices: top(x,y+h), right(x+h,y), left(x-h,y), bottom(x,y-h)
               
               ; Helper macro concept (unrolled):

               ; Emit Top (x, y+h)
               %y_top = OpFAdd %float %y %float_h
               %pos_top = OpCompositeConstruct %v4float %x %y_top %z %w
               %pos_out_ptr = OpAccessChain %ptr_v4 %gl_out %int_0
               OpStore %pos_out_ptr %pos_top
               OpStore %outColor %c0
               OpEmitVertex

               ; Emit Right (x+h, y)
               %x_right = OpFAdd %float %x %float_h
               %pos_right = OpCompositeConstruct %v4float %x_right %y %z %w
               OpStore %pos_out_ptr %pos_right
               OpStore %outColor %c0
               OpEmitVertex

               ; Emit Left (x-h, y)
               %x_left = OpFAdd %float %x %float_neg_h
               %pos_left = OpCompositeConstruct %v4float %x_left %y %z %w
               OpStore %pos_out_ptr %pos_left
               OpStore %outColor %c0
               OpEmitVertex

               ; Emit Bottom (x, y-h)
               %y_bottom = OpFAdd %float %y %float_neg_h
               %pos_bottom = OpCompositeConstruct %v4float %x %y_bottom %z %w
               OpStore %pos_out_ptr %pos_bottom
               OpStore %outColor %c0
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
