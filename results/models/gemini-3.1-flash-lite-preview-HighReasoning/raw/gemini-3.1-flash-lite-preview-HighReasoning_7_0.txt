```cpp
#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <array>

struct Point {
    double x, y, z;

    Point operator-(const Point& other) const { return {x - other.x, y - other.y, z - other.z}; }
    Point operator+(const Point& other) const { return {x + other.x, y + other.y, z + other.z}; }
    Point operator*(double scalar) const { return {x * scalar, y * scalar, z * scalar}; }
};

struct Triangle {
    int a, b, c; // Indices into vertex list
};

struct Mesh {
    std::vector<Point> vertices;
    std::vector<Triangle> faces;
};

inline double dot(const Point& a, const Point& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int va, fa, vb, fb;
    if (!(std::cin >> va >> fa >> vb >> fb)) return 0;

    Mesh meshA, meshB;
    meshA.vertices.resize(va);
    for (int i = 0; i < va; ++i) std::cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    meshA.faces.resize(fa);
    for (int i = 0; i < fa; ++i) std::cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;

    meshB.vertices.resize(vb);
    for (int i = 0; i < vb; ++i) std::cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    meshB.faces.resize(fb);
    for (int i = 0; i < fb; ++i) std::cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;

    // Output combined mesh (Base functionality for IO handling)
    std::cout << va + vb << " " << fa + fb << "\n";
    for (const auto& v : meshA.vertices) std::cout << v.x << " " << v.y << " " << v.z << "\n";
    for (const auto& v : meshB.vertices) std::cout << v.x << " " << v.y << " " << v.z << "\n";
    for (const auto& f : meshA.faces) std::cout << f.a << " " << f.b << " " << f.c << "\n";
    for (const auto& f : meshB.faces) std::cout << f.a + va << " " << f.b + va << " " << f.c + va << "\n";
    
    return 0;
}
```
