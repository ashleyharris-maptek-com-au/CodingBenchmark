; SPIR-V Geometry Shader: Shrink Triangles
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

; Decorate Input
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_in Input
               OpDecorate %color_in Location 2

; Decorate Output
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_out Output
               OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_333 = OpConstant %float 0.33333334
%float_0_7 = OpConstant %float 0.7

; Input definitions
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%color_in = OpVariable %ptr_in_arr_v3 Input

; Output definitions
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float

; Function main
%void_fn = OpTypeFunction %void
%main = OpFunction %void None %void_fn
%label = OpLabel

; Access positions
%ptr_in_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%ptr_in_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%ptr_in_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %ptr_in_p0
%p1 = OpLoad %v4float %ptr_in_p1
%p2 = OpLoad %v4float %ptr_in_p2

; Compute Centroid C = (p0 + p1 + p2) * 0.33333334
%sum = OpFAdd %v4float %p0 %p1
%sum2 = OpFAdd %v4float %sum %p2
%centroid = OpVectorTimesScalar %v4float %sum2 %float_0_333

; Shrink: new_pos = C + 0.7 * (P - C)
; Vertex 0
%vec0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %vec0 %float_0_7
%new_p0 = OpFAdd %v4float %centroid %scaled0

; Vertex 1
%vec1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %vec1 %float_0_7
%new_p1 = OpFAdd %v4float %centroid %scaled1

; Vertex 2
%vec2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %vec2 %float_0_7
%new_p2 = OpFAdd %v4float %centroid %scaled2

; Emit V0
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %ptr_out_pos %new_p0
%ptr_c0 = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %ptr_c0
OpStore %outColor %c0
OpEmitVertex

; Emit V1
OpStore %ptr_out_pos %new_p1
%ptr_c1 = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %ptr_c1
OpStore %outColor %c1
OpEmitVertex

; Emit V2
OpStore %ptr_out_pos %new_p2
%ptr_c2 = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %ptr_c2
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
