declare const require: any;
const fs = require('fs');

interface Task {
  id: number;
  duration: number;
  deps: number[];
  deadline: number;
  weight: number;
}

interface InputData {
  machineCount: number;
  tasks: Task[];
}

interface ScheduleEntry {
  id: number;
  start: number;
  machine: number;
}

function solve() {
  const input: string = fs.readFileSync(0, 'utf8');
  const data: InputData = JSON.parse(input);
  const { machineCount, tasks } = data;

  const taskMap = new Map<number, Task>();
  const adj = new Map<number, number[]>();
  const inDegree = new Map<number, number>();

  for (const task of tasks) {
    taskMap.set(task.id, task);
    if (!adj.has(task.id)) adj.set(task.id, []);
    if (!inDegree.has(task.id)) inDegree.set(task.id, 0);
  }

  for (const task of tasks) {
    for (const dep of task.deps) {
      if (!adj.has(dep)) adj.set(dep, []);
      adj.get(dep)!.push(task.id);
      inDegree.set(task.id, (inDegree.get(task.id) || 0) + 1);
    }
  }

  // Greedy list scheduler based on priority
  // Priority: minimize (deadline - duration)
  const readyQueue: Task[] = [];
  for (const task of tasks) {
    if (inDegree.get(task.id) === 0) {
      readyQueue.push(task);
    }
  }

  readyQueue.sort((a, b) => (a.deadline - a.duration) - (b.deadline - b.duration));

  const machineAvailableTime = new Array(machineCount).fill(0);
  const taskFinishTime = new Map<number, number>();
  const schedule: ScheduleEntry[] = [];

  while (readyQueue.length > 0) {
    const task = readyQueue.shift()!;
    
    // Find earliest start time based on dependencies
    let startTime = 0;
    for (const depId of task.deps) {
      startTime = Math.max(startTime, taskFinishTime.get(depId) || 0);
    }

    // Find first available machine at or after startTime
    let bestMachine = 0;
    let bestStartTime = Infinity;

    for (let i = 0; i < machineCount; i++) {
      const actualStart = Math.max(startTime, machineAvailableTime[i]);
      if (actualStart < bestStartTime) {
        bestStartTime = actualStart;
        bestMachine = i;
      }
    }

    schedule.push({
      id: task.id,
      start: bestStartTime,
      machine: bestMachine
    });

    const finishTime = bestStartTime + task.duration;
    taskFinishTime.set(task.id, finishTime);
    machineAvailableTime[bestMachine] = finishTime;

    // Update ready queue
    for (const neighborId of adj.get(task.id) || []) {
      inDegree.set(neighborId, inDegree.get(neighborId)! - 1);
      if (inDegree.get(neighborId) === 0) {
        const neighborTask = taskMap.get(neighborId)!;
        readyQueue.push(neighborTask);
        // Keep queue sorted
        readyQueue.sort((a, b) => (a.deadline - a.duration) - (b.deadline - b.duration));
      }
    }
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

solve();
