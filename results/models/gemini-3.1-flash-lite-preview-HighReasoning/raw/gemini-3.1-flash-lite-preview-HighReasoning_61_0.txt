const fs = require('fs');

interface Point {
    x: number;
    y: number;
}

interface Package {
    id: number;
    x: number;
    y: number;
}

interface Input {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

function getDistance(p1: Point, p2: Point): number {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
}

function solve() {
    const inputStr = fs.readFileSync(0, 'utf8');
    if (!inputStr) return;
    const input: Input = JSON.parse(inputStr);

    const depot: Point = { x: input.depot[0], y: input.depot[1] };
    const packages = [...input.packages];
    const routes: number[][] = Array.from({ length: input.truckCount }, () => []);

    // Simple greedy approach: Nearest Neighbor to depot/current position
    // We'll distribute packages among trucks using a simple round-robin or load balancing if needed.
    // Given the task, just ensuring every package is assigned is the priority.
    
    let currentPositions: Point[] = Array(input.truckCount).fill(depot);
    
    while (packages.length > 0) {
        for (let i = 0; i < input.truckCount; i++) {
            if (packages.length === 0) break;

            let bestIdx = -1;
            let minDist = Infinity;

            for (let j = 0; j < packages.length; j++) {
                const dist = getDistance(currentPositions[i], packages[j]);
                if (dist < minDist) {
                    minDist = dist;
                    bestIdx = j;
                }
            }

            const pkg = packages.splice(bestIdx, 1)[0];
            routes[i].push(pkg.id);
            currentPositions[i] = pkg;
        }
    }

    console.log(JSON.stringify({ routes }));
}

solve();
