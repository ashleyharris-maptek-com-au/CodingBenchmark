; Geometry Shader: Flat shading with adjacency
OpCapability Geometry
OpExtension "SPV_KHR_multiview"
%std450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %normal_in %color_in %outColor_out %gl_PrimitiveIDIn
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorators
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_in Input
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Output
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor_out Location 0
OpDecorate %gl_PrimitiveIDIn BuiltIn PrimitiveId

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0 = OpConstant %float 0.0
%v3_light = OpConstantComposite %v3float %float 1.0 %float 1.0 %float 1.0

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%gl_PerVertex_out = OpTypeStruct %v4float
%func_void = OpTypeFunction %void

; Variables
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_in_arr_v3_6 = OpTypePointer Input %_arr_v3_6
%_arr_v3_6 = OpTypeArray %v3float %uint_6
%worldPos_in = OpVariable %ptr_in_arr_v3_6 Input
%normal_in = OpVariable %ptr_in_arr_v3_6 Input
%color_in = OpVariable %ptr_in_arr_v3_6 Input
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor_out = OpVariable %ptr_out_v3 Output
%ptr_in_int = OpTypePointer Input %int
%gl_PrimitiveIDIn = OpVariable %ptr_in_int Input

; Main Function
%main = OpFunction %void None %func_void
%label = OpLabel

; 1. Vertices: 0, 2, 4
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p4 = OpLoad %v4float %p4_ptr

; 2. N = cross(p2-p0, p4-p0)
%p2_v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p0_v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p4_v3 = OpVectorShuffle %v3float %p4 %p4 0 1 2
%v20 = OpFSub %v3float %p2_v3 %p0_v3
%v40 = OpFSub %v3float %p4_v3 %p0_v3
%cross = OpExtInst %v3float %std450 Cross %v20 %v40
%N = OpExtInst %v3float %std450 Normalize %cross

; 3-4. Diffuse
%L = OpExtInst %v3float %std450 Normalize %v3_light
%dot = OpDot %float %N %L
%diff = OpExtInst %float %std450 FMax %dot %float_0

; 5. Average color
%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color_in %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%c02 = OpFAdd %v3float %c0 %c2
%csum = OpFAdd %v3float %c02 %c4
%base_color = OpVectorTimesScalar %v3float %csum 0.333333

; 6. Final color = base * (0.2 + 0.8 * diff)
%diff_w = OpFMul %float %float_0_8 %diff
%factor = OpFAdd %float %float_0_2 %diff_w
%final_color = OpVectorTimesScalar %v3float %base_color %factor

; 7. Emit
%indices = OpConstantComposite %_arr_int_3 %int_0 %int_2 %int_4
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%_arr_int_3 = OpTypeArray %int %int_3
%int_3 = OpConstant %int 3

OpBranch %loop
%loop = OpLabel
%i = OpPhi %int %int_0 %label %next_i %loop_body
%idx = OpAccessChain %ptr_in_int %indices %i
%v_idx = OpLoad %int %idx

%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %v_idx %int_0
%pos = OpLoad %v4float %pos_ptr
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %pos
OpStore %outColor_out %final_color
OpEmitVertex
%next_i = OpIAdd %int %i %int_1
%cmp = OpSLessThan %bool %next_i %int_3
OpBranchConditional %cmp %loop %end
%bool = OpTypeBool
%ptr_out_v4 = OpTypePointer Output %v4float

%end = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
