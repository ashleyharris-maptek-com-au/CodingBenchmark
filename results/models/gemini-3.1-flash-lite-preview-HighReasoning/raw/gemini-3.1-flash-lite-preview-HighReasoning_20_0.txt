```rust
use std::io::{self, BufRead};

struct Sample {
    depth: f64,
    properties: Vec<f64>,
}

struct Drillhole {
    id: String,
    start: [f64; 3],
    dir: [f64; 3],
    length: f64,
    samples: Vec<Sample>,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    if let Some(Ok(first_line)) = lines.next() {
        let parts: Vec<&str> = first_line.split_whitespace().collect();
        if parts.len() < 2 { return; }
        let num_holes: usize = parts[0].parse().unwrap_or(0);
        let num_properties: usize = parts[1].parse().unwrap_or(0);

        if let Some(Ok(prop_line)) = lines.next() {
            let property_names: Vec<String> = prop_line.split_whitespace().map(|s| s.to_string()).collect();

            let mut drillholes = Vec::with_capacity(num_holes);

            for _ in 0..num_holes {
                if let Some(Ok(header_line)) = lines.next() {
                    let h: Vec<&str> = header_line.split_whitespace().collect();
                    let id = h[0].to_string();
                    let start = [h[1].parse().unwrap(), h[2].parse().unwrap(), h[3].parse().unwrap()];
                    let dir = [h[4].parse().unwrap(), h[5].parse().unwrap(), h[6].parse().unwrap()];
                    let length = h[7].parse().unwrap();
                    let num_samples: usize = h[8].parse().unwrap();

                    let mut samples = Vec::with_capacity(num_samples);
                    for _ in 0..num_samples {
                        if let Some(Ok(sample_line)) = lines.next() {
                            let s: Vec<f64> = sample_line.split_whitespace().map(|x| x.parse().unwrap()).collect();
                            samples.push(Sample { depth: s[0], properties: s[1..].to_vec() });
                        }
                    }
                    drillholes.push(Drillhole { id, start, dir, length, samples });
                }
            }

            // Simple outlier detection strategy: Compare to local moving average
            let mut results = Vec::new();
            for (h_idx, hole) in drillholes.iter().enumerate() {
                for (s_idx, sample) in hole.samples.iter().enumerate() {
                    for p_idx in 0..num_properties {
                        let mut local_values = Vec::new();
                        let start = if s_idx > 2 { s_idx - 2 } else { 0 };
                        let end = if s_idx + 3 < hole.samples.len() { s_idx + 3 } else { hole.samples.len() };
                        for i in start..end {
                            if i != s_idx {
                                local_values.push(hole.samples[i].properties[p_idx]);
                            }
                        }

                        if !local_values.is_empty() {
                            let avg: f64 = local_values.iter().sum::<f64>() / local_values.len() as f64;
                            let diff = (sample.properties[p_idx] - avg).abs();
                            if diff > avg * 0.5 { // Simple threshold check
                                results.push((&hole.id, s_idx, &property_names[p_idx], diff / (avg + 1.0)));
                            }
                        }
                    }
                }
            }

            results.sort_by(|a, b| b.3.partial_cmp(&a.3).unwrap());

            println!("{}", results.len());
            for r in results {
                println!("{} {} {} {:.4}", r.0, r.1, r.2, r.3);
            }
        }
    }
}
```
