#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

struct Body {
    vec4 position; // xyz for position, w for mass
    vec4 velocity; // xyz for velocity, w unused
};

layout(std430, binding = 0) buffer Bodies {
    Body bodies[];
};

uniform float deltaTime;
uniform int numBodies;
uniform float gravitationalConstant;
uniform float softeningSquared;

void main() {
    uint index = gl_GlobalInvocationID.x;

    if (index >= numBodies) {
        return;
    }

    vec3 p = bodies[index].position.xyz;
    vec3 v = bodies[index].velocity.xyz;
    float m = bodies[index].position.w;

    vec3 force = vec3(0.0);

    for (int i = 0; i < numBodies; ++i) {
        if (i == index) {
            continue;
        }

        vec3 other_p = bodies[i].position.xyz;
        float other_m = bodies[i].position.w;

        vec3 direction = other_p - p;
        float distanceSquared = dot(direction, direction) + softeningSquared;
        float distance = sqrt(distanceSquared);

        // F = G * (m1 * m2) / r^2
        // a = F / m1 = G * m2 / r^2
        float magnitude = gravitationalConstant * other_m / (distanceSquared * distance);
        force += direction * magnitude;
    }

    // Update velocity (a = F/m, dv = a * dt)
    v += force * deltaTime;

    // Update position (dp = v * dt)
    p += v * deltaTime;

    bodies[index].position = vec4(p, m);
    bodies[index].velocity = vec4(v, 0.0);
}
