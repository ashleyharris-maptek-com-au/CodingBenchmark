// Basic HLSL Pixel Shader for Vulkan
// Compiled to SPIR-V for use with Vulkan.

// Output structure for the pixel shader
struct PSOutput
{
    float4 Color : SV_TARGET0; // Render target 0
};

// Main function (pixel shader entry point)
PSOutput main()
{
    PSOutput output;
    output.Color = float4(1.0f, 0.0f, 0.0f, 1.0f); // Red color (RGBA)
    return output;
}
