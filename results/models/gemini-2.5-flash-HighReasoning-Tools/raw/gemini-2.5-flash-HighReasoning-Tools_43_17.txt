struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64
    float4x4 projection;     // offset 128
    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0 {
    // 1. Brick pattern calculation
    float2 scaledUV = input.uv * float2(8.0, 16.0); // 8x16 UV scale

    // Staggered rows: half-brick offset every other row
    // If the brick row is odd, add 0.5 to the x-coordinate
    if (fmod(floor(scaledUV.y), 2.0) >= 1.0) { // Check if row is odd
        scaledUV.x += 0.5;
    }

    float2 brickCoord = frac(scaledUV); // Coordinate within a single brick [0,1)

    // Mortar thickness: 0.05 in X and 0.1 in Y
    float2 mortarSize = float2(0.05, 0.1);

    // Determine if it's mortar or brick
    bool isMortarX = (brickCoord.x < mortarSize.x);
    bool isMortarY = (brickCoord.y < mortarSize.y);
    bool isMortar = isMortarX || isMortarY;

    float3 finalColor;
    if (isMortar) {
        finalColor = float3(0.7, 0.7, 0.7); // Mortar color
    } else {
        finalColor = float3(0.7, 0.2, 0.1); // Brick color
    }

    // 2. Lambertian Shading
    float3 N = normalize(input.normal); // Normalized world-space normal

    // Light direction (from fragment to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Diffuse component (max(0, dot(N, L)))
    float diffuse = max(0.0, dot(N, L));

    // Shading factor: (0.4 + 0.6*diffuse)
    float shadingFactor = (0.4 + 0.6 * diffuse);

    finalColor *= shadingFactor; // Apply shading

    return float4(finalColor, 1.0); // Output with alpha 1.0
}
