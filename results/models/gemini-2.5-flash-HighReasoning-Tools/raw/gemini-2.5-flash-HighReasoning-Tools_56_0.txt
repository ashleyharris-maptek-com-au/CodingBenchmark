const fs = require('fs');

// Euclidean distance function
function calculateDistance(p1, p2) {
    const dx = p1.x - p2.x;
    const dy = p1.y - p2.y;
    return Math.sqrt(dx * dx + dy * dy);
}

function solveVRP() {
    // Read input from stdin
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const depot_point = { x: data.depot[0], y: data.depot[1] };
    const packages = data.packages;
    const truckCount = data.truckCount;

    // Initialize routes
    // Each route will store an array of package objects
    const routes = Array.from({ length: truckCount }, () => []);
    const unassignedPackages = [...packages];

    // Main loop for cheapest insertion heuristic
    while (unassignedPackages.length > 0) {
        let bestPackageIndex = -1;
        let bestRouteIndex = -1;
        let bestInsertionPosition = -1; // Position in the route array
        let minCostIncrease = Infinity;

        // Iterate through all unassigned packages
        for (let pIdx = 0; pIdx < unassignedPackages.length; pIdx++) {
            const pkgToInsert = unassignedPackages[pIdx];

            // Iterate through all existing routes
            for (let rIdx = 0; rIdx < routes.length; rIdx++) {
                const currentRoute = routes[rIdx];

                // Consider all possible insertion points within the current route
                // An insertion point 'i' means inserting between route[i-1] and route[i]
                // i=0: insert between depot and route[0] (or as first if route empty)
                // i=currentRoute.length: insert between route[last] and depot
                for (let i = 0; i <= currentRoute.length; i++) {
                    let costIncrease;

                    if (currentRoute.length === 0) {
                        // Inserting into an empty route: depot -> pkgToInsert -> depot
                        costIncrease = calculateDistance(depot_point, pkgToInsert) * 2;
                    } else {
                        // Determine the points before and after the insertion
                        const prevPoint = (i === 0) ? depot_point : packages.find(p => p.id === currentRoute[i - 1]);
                        const nextPoint = (i === currentRoute.length) ? depot_point : packages.find(p => p.id === currentRoute[i]);

                        // Calculate cost: new segments - old segment
                        costIncrease = calculateDistance(prevPoint, pkgToInsert) +
                                       calculateDistance(pkgToInsert, nextPoint) -
                                       calculateDistance(prevPoint, nextPoint);
                    }

                    if (costIncrease < minCostIncrease) {
                        minCostIncrease = costIncrease;
                        bestPackageIndex = pIdx;
                        bestRouteIndex = rIdx;
                        bestInsertionPosition = i;
                    }
                }
            }
        }

        // Insert the chosen package into the chosen route at the best position
        const assignedPackage = unassignedPackages.splice(bestPackageIndex, 1)[0];
        routes[bestRouteIndex].splice(bestInsertionPosition, 0, assignedPackage.id);
    }

    // Format output as required: an array of routes, each containing package IDs
    const outputRoutes = routes.map(route => route.map(pkgId => pkgId));

    // Print the result to stdout
    console.log(JSON.stringify({ routes: outputRoutes }));
}

solveVRP();
