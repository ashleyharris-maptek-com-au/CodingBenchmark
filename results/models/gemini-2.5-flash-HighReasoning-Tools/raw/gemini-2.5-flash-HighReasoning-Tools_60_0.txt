const fs = require('fs');

function solve() {
    // Read input from stdin
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const machineCount = data.machineCount;
    const tasks = data.tasks;

    // Create an adjacency list and in-degree map for topological sort
    const adj = new Map(); // taskId -> [dependentTaskIds]
    const inDegree = new Map(); // taskId -> count of dependencies

    // Initialize in-degrees to 0 for all tasks and build adjacency list
    for (const task of tasks) {
        inDegree.set(task.id, 0);
        adj.set(task.id, []);
    }

    // Populate in-degrees and adjacency list based on dependencies
    for (const task of tasks) {
        for (const depId of task.deps) {
            // depId -> task.id means task.id depends on depId
            // So, depId is a prerequisite for task.id
            if (!adj.has(depId)) {
                // This handles cases where a dependency is not explicitly listed as a task
                // which shouldn't happen based on problem description but good for robustness
                adj.set(depId, []);
            }
            adj.get(depId).push(task.id);
            inDegree.set(task.id, inDegree.get(task.id) + 1);
        }
    }

    // Kahn's algorithm for topological sort
    const queue = [];
    for (const taskId of inDegree.keys()) {
        if (inDegree.get(taskId) === 0) {
            queue.push(taskId);
        }
    }

    const topologicalOrder = [];
    let head = 0; // Using a pointer for queue to avoid performance issues with shift()

    while (head < queue.length) {
        const currentTaskId = queue[head++];
        topologicalOrder.push(currentTaskId);

        for (const dependentTaskId of adj.get(currentTaskId)) {
            inDegree.set(dependentTaskId, inDegree.get(dependentTaskId) - 1);
            if (inDegree.get(dependentTaskId) === 0) {
                queue.push(dependentTaskId);
            }
        }
    }

    // Check for cycles (if topologicalOrder.length !== tasks.length, there's a cycle)
    if (topologicalOrder.length !== tasks.length) {
        // For this benchmark, we assume valid DAGs. If cycles are possible and need handling,
        // this part would need more sophisticated logic (e.g., error reporting or breaking cycles).
        // For now, we'll proceed with the partial order or assume no cycles in valid inputs.
        // The problem description doesn't explicitly mention how to handle cycles.
    }

    // Output the topological order
    console.log(JSON.stringify({ order: topologicalOrder }));
}

solve();
