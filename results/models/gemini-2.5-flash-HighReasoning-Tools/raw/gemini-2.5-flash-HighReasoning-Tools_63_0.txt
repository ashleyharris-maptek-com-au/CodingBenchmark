// interfaces.ts
interface InputData {
    grid: string[];
    robots: RobotInitial[];
    jobs: Job[];
}

interface RobotInitial {
    id: number;
    x: number;
    y: number;
}

interface Robot extends RobotInitial {
    currentX: number;
    currentY: number;
    totalTravel: number;
    assignedJobs: Job[];
}

interface Job {
    id: number;
    x: number;
    y: number;
    priority: number;
}

interface Output {
    assignments: number[][];
}

// GridPathfinder Class for BFS
class GridPathfinder {
    private grid: boolean[][]; // true for open ('.'), false for blocked ('#')
    private rows: number;
    private cols: number;

    constructor(gridStrings: string[]) {
        if (!gridStrings || gridStrings.length === 0 || gridStrings[0].length === 0) {
            throw new Error("Invalid grid: grid must not be empty.");
        }
        this.rows = gridStrings.length;
        this.cols = gridStrings[0].length;
        this.grid = gridStrings.map(row => row.split('').map(char => char === '.'));
    }

    private isValid(x: number, y: number): boolean {
        return x >= 0 && x < this.rows && y >= 0 && y < this.cols && this.grid[x][y];
    }

    getShortestDistance(startX: number, startY: number, endX: number, endY: number): number {
        // Check if start or end points are within bounds and on an open cell
        if (!this.isValid(startX, startY) || !this.isValid(endX, endY)) {
            // Return -1 to indicate an invalid or unreachable point
            return -1;
        }
        // If start and end are the same, distance is 0
        if (startX === endX && startY === endY) {
            return 0;
        }

        // BFS initialization
        const queue: [number, number, number][] = [[startX, startY, 0]]; // [x, y, distance]
        const visited: boolean[][] = Array(this.rows).fill(0).map(() => Array(this.cols).fill(false));
        visited[startX][startY] = true;

        // Directions for movement (up, down, left, right)
        const dx = [-1, 1, 0, 0];
        const dy = [0, 0, -1, 1];

        let head = 0; // Manual queue pointer for performance

        while (head < queue.length) {
            const [currentX, currentY, dist] = queue[head++];

            for (let i = 0; i < 4; i++) {
                const nextX = currentX + dx[i];
                const nextY = currentY + dy[i];

                if (this.isValid(nextX, nextY) && !visited[nextX][nextY]) {
                    if (nextX === endX && nextY === endY) {
                        return dist + 1; // Found the shortest path
                    }
                    visited[nextX][nextY] = true;
                    queue.push([nextX, nextY, dist + 1]);
                }
            }
        }

        return -1; // No path found
    }
}

// Main Logic for Job Assignment
declare const require: any;
const fs = require('fs');

try {
    const inputString: string = fs.readFileSync(0, 'utf8');
    const data: InputData = JSON.parse(inputString);

    function assignJobs(input: InputData): Output {
        const pathfinder = new GridPathfinder(input.grid);

        // Initialize robots with mutable state. Sort them by ID initially to ensure a consistent
        // order for output assignments and deterministic tie-breaking.
        const robots: Robot[] = input.robots
            .map(r => ({
                ...r,
                currentX: r.x,
                currentY: r.y,
                totalTravel: 0,
                assignedJobs: []
            }))
            .sort((a, b) => a.id - b.id);

        // Sort jobs for deterministic greedy assignment:
        // Primary sort key: priority (descending)
        // Secondary sort key: job ID (ascending) for consistent tie-breaking
        const sortedJobs = [...input.jobs].sort((a, b) => {
            if (a.priority !== b.priority) {
                return b.priority - a.priority;
            }
            return a.id - b.id;
        });

        for (const job of sortedJobs) {
            let minDistance = Infinity;
            let chosenRobot: Robot | null = null;

            for (const robot of robots) {
                const distance = pathfinder.getShortestDistance(robot.currentX, robot.currentY, job.x, job.y);

                // Only consider robots that can reach the job
                if (distance !== -1) {
                    // Tie-breaking logic for choosing a robot:
                    // 1. Prefer robot with the shortest distance to the current job.
                    // 2. If distances are equal, prefer robot with less accumulated total travel distance.
                    // 3. If distances and total travel are equal, prefer robot with a smaller ID.
                    if (distance < minDistance) {
                        minDistance = distance;
                        chosenRobot = robot;
                    } else if (distance === minDistance) {
                        if (chosenRobot === null) {
                            // This case should ideally not happen after the first iteration of the robot loop,
                            // as chosenRobot would be assigned. But as a safeguard:
                            chosenRobot = robot;
                        } else if (robot.totalTravel < chosenRobot.totalTravel) {
                            chosenRobot = robot;
                        } else if (robot.totalTravel === chosenRobot.totalTravel) {
                            if (robot.id < chosenRobot.id) {
                                chosenRobot = robot;
                            }
                        }
                    }
                }
            }

            if (chosenRobot) {
                chosenRobot.assignedJobs.push(job);
                chosenRobot.totalTravel += minDistance;
                // Update robot's current position to the location of the job it just picked up
                chosenRobot.currentX = job.x;
                chosenRobot.currentY = job.y;
            } else {
                // If a job is unreachable, this indicates a problem with the input grid or job placement.
                // The problem implies all jobs must be assigned.
                // For a benchmark, we assume valid inputs where all jobs are reachable.
                // If this scenario occurs, it would lead to a job not being assigned,
                // violating "Every job must appear exactly once."
                // console.error(`Error: Job ${job.id} at (${job.x}, ${job.y}) is unreachable.`);
                // In a production scenario, one might throw an error or handle accordingly.
                // For this benchmark, we must ensure all jobs are assigned.
                throw new Error(`Critical Error: Job ${job.id} at (${job.x}, ${job.y}) is unreachable by any robot.`);
            }
        }

        // Prepare output structure. The problem states "The array index is the robot id."
        // We interpreted this to mean the order of robots in the `assignments` array
        // should correspond to robots sorted by their ID, which is how `robots` array is already sorted.
        const outputAssignments: number[][] = robots.map(robot =>
            // The jobs within a robot's assignment are already in the order they were assigned
            // due to the greedy algorithm's sequential processing.
            robot.assignedJobs.map(job => job.id)
        );

        const output: Output = {
            assignments: outputAssignments
        };

        return output;
    }

    const result = assignJobs(data);
    console.log(JSON.stringify(result));

} catch (error: any) {
    // Standard error handling for Node.js programs that read from stdin/stdout
    console.error(`Program error: ${error.message}`);
    // Exit with a non-zero code to indicate failure
    process.exit(1);
}
