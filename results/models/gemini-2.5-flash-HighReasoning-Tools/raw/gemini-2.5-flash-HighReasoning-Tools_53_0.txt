import math

class Spacecraft:
    def __init__(self, x, y, vx, vy, mass):
        self.x = x
        self.y = y
        self.vx = vx
        self.vy = vy
        self.mass = mass

    def apply_thrust(self, fx, fy, dt):
        """Applies thrust and updates velocity based on force and time step."""
        ax = fx / self.mass
        ay = fy / self.mass
        self.vx += ax * dt
        self.vy += ay * dt

    def update_position(self, dt):
        """Updates position based on current velocity and time step."""
        self.x += self.vx * dt
        self.y += self.vy * dt

class PIDController:
    def __init__(self, Kp, Ki, Kd, setpoint=0):
        self.Kp = Kp
        self.Ki = Ki
        self.Kd = Kd
        self.setpoint = setpoint
        self.integral_error = 0
        self.previous_error = 0

    def update(self, current_value, dt):
        """Calculates control output based on PID algorithm."""
        error = self.setpoint - current_value
        self.integral_error += error * dt
        derivative_error = (error - self.previous_error) / dt
        output = (self.Kp * error) + (self.Ki * self.integral_error) + (self.Kd * derivative_error)
        self.previous_error = error
        return output

class Autopilot:
    def __init__(self, spacecraft, station_x, station_y):
        self.spacecraft = spacecraft
        self.station_x = station_x
        self.station_y = station_y # Assuming docking port is at station's origin for simplicity

        # PID controllers for position and velocity in X and Y axes
        # Tuned arbitrarily for now, will need adjustment based on simulation behavior
        self.pid_x_pos = PIDController(Kp=1.0, Ki=0.001, Kd=0.5) # Adjusted
        self.pid_y_pos = PIDController(Kp=1.0, Ki=0.001, Kd=0.5) # Adjusted
        self.pid_x_vel = PIDController(Kp=8.0, Ki=0.001, Kd=3.0) # Adjusted
        self.pid_y_vel = PIDController(Kp=8.0, Ki=0.001, Kd=3.0) # Adjusted

        # Target relative velocity for docking (should be very small)
        self.target_vx = 0.0
        self.target_vy = 0.0

        self.docked = False

    def control_step(self, dt):
        if self.docked:
            return

        # Relative position and velocity to the station's docking port
        rel_x = self.spacecraft.x - self.station_x
        rel_y = self.spacecraft.y - self.station_y
        rel_vx = self.spacecraft.vx - self.target_vx
        rel_vy = self.spacecraft.vy - self.target_vy

        # PID for position: calculate desired velocity based on position error
        desired_vx_from_pos = self.pid_x_pos.update(rel_x, dt)
        desired_vy_from_pos = self.pid_y_pos.update(rel_y, dt)

        # Update velocity PIDs with desired relative velocities
        self.pid_x_vel.setpoint = desired_vx_from_pos
        self.pid_y_vel.setpoint = desired_vy_from_pos

        # PID for velocity: calculate required force based on velocity error
        fx = self.pid_x_vel.update(rel_vx, dt)
        fy = self.pid_y_vel.update(rel_vy, dt)

        # Apply thrust to spacecraft
        self.spacecraft.apply_thrust(fx, fy, dt)
        self.spacecraft.update_position(dt)

        # Check for docking condition
        distance = math.sqrt(rel_x**2 + rel_y**2)
        relative_speed = math.sqrt(rel_vx**2 + rel_vy**2)

        docking_distance_threshold = 0.1 # meters
        docking_speed_threshold = 0.01 # m/s

        if distance < docking_distance_threshold and relative_speed < docking_speed_threshold:
            self.docked = True
            print(f"Docked at x={self.spacecraft.x:.2f}, y={self.spacecraft.y:.2f} with relative speed {relative_speed:.4f}")
            # Snap to station position if docked
            self.spacecraft.x = self.station_x
            self.spacecraft.y = self.station_y
            self.spacecraft.vx = self.target_vx
            self.spacecraft.vy = self.target_vy


# --- Simulation Setup ---
def run_simulation():
    # Initial conditions
    initial_x = 10.0 # meters away from station
    initial_y = 5.0
    initial_vx = -0.5 # approaching station
    initial_vy = -0.1
    craft_mass = 1000.0 # kg

    station_x = 0.0
    station_y = 0.0

    spacecraft = Spacecraft(initial_x, initial_y, initial_vx, initial_vy, craft_mass)
    autopilot = Autopilot(spacecraft, station_x, station_y)

    dt = 0.1 # seconds per step
    time = 0.0
    max_simulation_time = 300.0 # seconds

    print("Starting Autopilot Simulation...")
    print(f"Initial SC: x={spacecraft.x:.2f}, y={spacecraft.y:.2f}, vx={spacecraft.vx:.2f}, vy={spacecraft.vy:.2f}")

    log_interval = 10 # Log every X steps
    step_count = 0

    while not autopilot.docked and time < max_simulation_time:
        autopilot.control_step(dt)
        time += dt
        step_count += 1

        if step_count % log_interval == 0:
            rel_x = spacecraft.x - station_x
            rel_y = spacecraft.y - station_y
            rel_vx = spacecraft.vx - autopilot.target_vx
            rel_vy = spacecraft.vy - autopilot.target_vy
            distance = math.sqrt(rel_x**2 + rel_y**2)
            relative_speed = math.sqrt(rel_vx**2 + rel_vy**2)

            print(f"Time: {time:.1f}s | SC Pos: ({spacecraft.x:.2f}, {spacecraft.y:.2f}) | SC Vel: ({spacecraft.vx:.2f}, {spacecraft.vy:.2f}) | Dist: {distance:.2f}m | Speed: {relative_speed:.2f}m/s")

    if autopilot.docked:
        print(f"Simulation successful! Spacecraft docked in {time:.1f} seconds.")
    else:
        print(f"Simulation ended after {max_simulation_time:.1f} seconds. Docking failed.")
        print(f"Final SC: x={spacecraft.x:.2f}, y={spacecraft.y:.2f}, vx={spacecraft.vx:.2f}, vy={spacecraft.vy:.2f}")
        rel_x = spacecraft.x - station_x
        rel_y = spacecraft.y - station_y
        rel_vx = spacecraft.vx - autopilot.target_vx
        rel_vy = spacecraft.vy - autopilot.target_vy
        distance = math.sqrt(rel_x**2 + rel_y**2)
        relative_speed = math.sqrt(rel_vx**2 + rel_vy**2)
        print(f"Final Dist: {distance:.2f}m | Final Speed: {relative_speed:.2f}m/s")


if __name__ == "__main__":
    run_simulation()
