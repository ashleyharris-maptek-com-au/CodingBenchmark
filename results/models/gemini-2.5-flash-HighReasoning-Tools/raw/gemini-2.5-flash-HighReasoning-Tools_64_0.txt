// drone-path-planner.ts

interface DroneInput {
    id: number;
    x: number;
    y: number;
}

interface InputData {
    probability: number[][];
    drones: DroneInput[];
    maxStepsPerDrone: number;
}

interface OutputData {
    paths: [number, number][][];
}

interface Cell {
    x: number;
    y: number;
    value: number;
}

const fs = require('fs');
const inputString: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(inputString);

const { probability, drones, maxStepsPerDrone } = data;
const numRows = probability.length;
const numCols = probability[0].length;

// Keep track of all visited cells by any drone to ensure unique probability coverage
const visitedCells = new Set<string>(); // Stores "x,y" strings

// Helper to check if a cell is valid
function isValidCell(x: number, y: number): boolean {
    return x >= 0 && x < numCols && y >= 0 && y < numRows && probability[y][x] !== -1;
}

const paths: [number, number][][] = [];

// Define possible moves: stay, up, down, left, right
// Ordered to prioritize staying in place if it's the best option, then other directions.
const moves = [
    { dx: 0, dy: 0 }, // Stay
    { dx: 0, dy: -1 }, // Up
    { dx: 0, dy: 1 }, // Down
    { dx: -1, dy: 0 }, // Left
    { dx: 1, dy: 0 }, // Right
];

drones.forEach(drone => {
    const dronePath: [number, number][] = [];
    let currentX = drone.x;
    let currentY = drone.y;

    // Add starting position to path and mark as visited
    dronePath.push([currentX, currentY]);
    visitedCells.add(`${currentX},${currentY}`);

    for (let step = 1; step < maxStepsPerDrone; step++) {
        let bestNextX = -1;
        let bestNextY = -1;
        let highestProbability = -Infinity;

        // Find the best next move
        for (const move of moves) {
            const nextX = currentX + move.dx;
            const nextY = currentY + move.dy;
            const cellKey = `${nextX},${nextY}`;

            if (isValidCell(nextX, nextY) && !visitedCells.has(cellKey)) {
                const cellProbability = probability[nextY][nextX];
                if (cellProbability > highestProbability) {
                    highestProbability = cellProbability;
                    bestNextX = nextX;
                    bestNextY = nextY;
                }
            }
        }

        // If no valid and unvisited cell is found, stop moving
        if (bestNextX === -1) {
            // Fill remaining steps with the last known position
            for (let i = step; i < maxStepsPerDrone; i++) {
                dronePath.push([currentX, currentY]);
            }
            break;
        }

        // Move to the best cell
        currentX = bestNextX;
        currentY = bestNextY;
        dronePath.push([currentX, currentY]);
        visitedCells.add(`${currentX},${currentY}`);
    }
    paths.push(dronePath);
});

const output: OutputData = { paths };
console.log(JSON.stringify(output));
