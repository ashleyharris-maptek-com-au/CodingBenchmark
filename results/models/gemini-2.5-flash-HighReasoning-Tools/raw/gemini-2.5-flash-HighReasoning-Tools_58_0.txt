const fs = require('fs');

function bfs(grid, startX, startY, endX, endY) {
    const numRows = grid.length;
    const numCols = grid[0].length;

    if (startX === endX && startY === endY) {
        return 0;
    }

    const queue = [[startX, startY, 0]];
    const visited = new Set();
    visited.add(`${startX},${startY}`);

    const directions = [
        [0, 1],
        [0, -1],
        [1, 0],
        [-1, 0]
    ];

    while (queue.length > 0) {
        const [currX, currY, dist] = queue.shift();

        for (const [dx, dy] of directions) {
            const nextX = currX + dx;
            const nextY = currY + dy;

            if (nextX < 0 || nextX >= numCols || nextY < 0 || nextY >= numRows) {
                continue;
            }

            if (grid[nextY][nextX] === '#' || visited.has(`${nextX},${nextY}`)) {
                continue;
            }

            if (nextX === endX && nextY === endY) {
                return dist + 1;
            }

            visited.add(`${nextX},${nextY}`);
            queue.push([nextX, nextY, dist + 1]);
        }
    }

    return -1;
}

(function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);
    const grid = data.grid;
    const robots = data.robots;
    const jobs = data.jobs;

    const assignments = Array.from({ length: robots.length }, () => []);
    const robotCurrentPositions = robots.map(robot => ({ x: robot.x, y: robot.y }));
    const unassignedJobIds = new Set(jobs.map(job => job.id));
    const jobMap = new Map(jobs.map(job => [job.id, job]));

    while (unassignedJobIds.size > 0) {
        let bestDistance = Infinity;
        let bestRobotIndex = -1;
        let bestJobId = -1;

        for (let i = 0; i < robots.length; i++) {
            const robotPos = robotCurrentPositions[i];
            for (const jobId of unassignedJobIds) {
                const job = jobMap.get(jobId);
                const distance = bfs(grid, robotPos.x, robotPos.y, job.x, job.y);

                if (distance !== -1 && distance < bestDistance) {
                    bestDistance = distance;
                    bestRobotIndex = i;
                    bestJobId = jobId;
                }
            }
        }

        if (bestRobotIndex !== -1) {
            assignments[bestRobotIndex].push(bestJobId);
            const assignedJob = jobMap.get(bestJobId);
            robotCurrentPositions[bestRobotIndex] = { x: assignedJob.x, y: assignedJob.y };
            unassignedJobIds.delete(bestJobId);
        } else {
            break;
        }
    }

    console.log(JSON.stringify({ assignments: assignments }));
})();
