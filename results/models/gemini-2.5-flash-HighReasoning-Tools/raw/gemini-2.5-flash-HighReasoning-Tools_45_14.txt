import struct

def to_hex_le(value, num_bytes=4):
    """Converts an integer to a little-endian hex string of specified byte length."""
    return value.to_bytes(num_bytes, 'little').hex()

def float_to_hex_le(f_value):
    """Converts a float to its IEEE 754 single-precision little-endian hex string."""
    return struct.pack('<f', f_value).hex()

def string_to_spirv_hex(s):
    """Converts a string to a null-terminated, 32-bit word-padded, little-endian hex string."""
    # Encode string to bytes, null-terminate, and pad to a multiple of 4 bytes
    s_bytes = s.encode('utf-8') + b'\0'
    padding_needed = (4 - (len(s_bytes) % 4)) % 4
    s_bytes += b'\0' * padding_needed

    # Convert to little-endian 32-bit words
    hex_parts = []
    for i in range(0, len(s_bytes), 4):
        word = int.from_bytes(s_bytes[i:i+4], 'little')
        hex_parts.append(f'{word:08x}')
    return ''.join(hex_parts)

class SpirvBuilder:
    def __init__(self):
        self.id_counter = 1
        self.instructions = []
        self.bound_id = 0 # Will be updated at the end

    def next_id(self):
        current_id = self.id_counter
        self.id_counter += 1
        return current_id

    def add_instruction(self, opcode, *operands, result_id=None, type_id=None):
        # Opcode is the last 16 bits, WordCount is the upper 16 bits
        # WordCount includes the instruction word itself
        word_count = 1 + len(operands)
        if result_id is not None:
            word_count += 1
        if type_id is not None:
            word_count += 1

        instruction_word = (word_count << 16) | opcode
        
        # Convert instruction word to little-endian hex
        hex_instruction = to_hex_le(instruction_word)
        
        # Collect all parts of the instruction
        parts = [hex_instruction]
        if type_id is not None:
            parts.append(to_hex_le(type_id))
        if result_id is not None:
            parts.append(to_hex_le(result_id))
        
        for op in operands:
            if isinstance(op, int):
                parts.append(to_hex_le(op))
            elif isinstance(op, float):
                parts.append(float_to_hex_le(op))
            elif isinstance(op, str): # For string literals like "GLSL.std.450"
                parts.append(string_to_spirv_hex(op))
            else:
                raise ValueError(f"Unsupported operand type: {type(op)}")
        
        self.instructions.append("".join(parts))
        return result_id # Return the result_id if it was assigned

    def build(self):
        # Header
        magic_number = to_hex_le(0x07230203)
        version = to_hex_le(0x00010000)
        generator = to_hex_le(0)
        # Update bound_id with the final id_counter value
        self.bound_id = self.id_counter
        bound = to_hex_le(self.bound_id)
        reserved = to_hex_le(0)

        header = magic_number + version + generator + bound + reserved
        
        return header + "".join(self.instructions)
