```rust
use std::io::{self, BufRead, Write};
use std::f64;
use rand::Rng; // For random initial positions
use std::time::Instant; // For performance measurement

// Struct to represent a node with its position and displacement
struct Node {
    pos: [f64; 2],
    disp: [0.0; 2],
}

fn main() -> io::Result<()> {
    let start_time = Instant::now();
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read N and M (number of nodes and edges)
    let first_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "Empty input"))??;
    let nm: Vec<usize> = first_line
        .split_whitespace()
        .filter_map(|s| s.parse().ok())
        .collect();

    if nm.len() != 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidInput, "Expected two integers (N M) on the first line"));
    }
    let n = nm[0];
    let m = nm[1];

    if n == 0 { // Handle empty graph
        let stdout = io::stdout();
        let mut handle = stdout.lock();
        writeln!(handle, "0")?; // Output N = 0
        return Ok(());
    }

    // Adjacency list representation of the graph
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    // Read edges
    for i in 0..m {
        let line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, format!("Missing edge line {}", i + 2)))??;
        let edge: Vec<usize> = line
            .split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();

        if edge.len() != 2 {
            return Err(io::Error::new(io::ErrorKind::InvalidInput, format!("Expected two integers (u v) for edge on line {}", i + 2)));
        }
        let u = edge[0];
        let v = edge[1];

        // Validate node indices
        if u >= n || v >= n {
            return Err(io::Error::new(io::ErrorKind::InvalidInput, format!("Node index {} or {} out of bounds (0-{}) on line {}", u, v, n - 1, i + 2)));
        }

        adj[u].push(v);
        adj[v].push(u); // Graph is undirected
    }

    // Initialize nodes with random positions and zero displacement
    let mut nodes: Vec<Node> = Vec::with_capacity(n);
    let mut rng = rand::thread_rng();

    // Define the bounding box for initial placement and clamping
    let min_coord = 0.0;
    let max_coord = 1000.0; // Default canvas size

    for _i in 0..n {
        nodes.push(Node {
            pos: [rng.gen_range(min_coord..max_coord), rng.gen_range(min_coord..max_coord)],
            disp: [0.0, 0.0],
        });
    }

    // Graph layout parameters (constants for Fruchterman-Reingold)
    let area = (max_coord - min_coord) * (max_coord - min_coord);
    let k = (area / n as f64).sqrt(); // Optimal distance between nodes
    let c_rep = k * k; // Repulsive constant (often k^2)
    let c_attr = k;    // Attractive constant (often k)

    // Adaptive number of iterations based on graph size
    // For smaller graphs, more iterations yield better quality.
    // For larger graphs, fewer iterations are necessary for performance.
    let max_iterations = match n {
        0..=15 => 100,      // Trivial
        16..=100 => 70,     // Medium
        101..=500 => 50,    // Large-ish
        501..=5000 => 30,   // Larger graphs
        _ => 15,            // Extreme (5000+ nodes): very few iterations to meet time limits
    };
    let mut temperature = 100.0; // Initial "temperature" or step size
    let cooling_factor = 0.95; // Rate at which temperature decreases each iteration

    // Determine grid size for repulsive force optimization
    // Grid dimension scales with N's square root, capped to reasonable values.
    // This creates a spatial hash to quickly find nearby nodes.
    let grid_dim_f64 = (n as f64).sqrt().max(2.0).min(200.0);
    let grid_dim = grid_dim_f64 as usize;
    let cell_size = (max_coord - min_coord) / grid_dim_f64;

    // --- Force-Directed Layout Loop ---
    for _iter in 0..max_iterations {
        // Check if elapsed time is too long for extreme graphs
        if n > 5000 && start_time.elapsed().as_secs_f64() > 25.0 {
            break; // Stop early to meet performance requirements
        }

        // Reset displacements for all nodes at the beginning of each iteration
        for node in &mut nodes {
            node.disp = [0.0; 2]; // Use array literal for clarity and direct assignment
        }

        // --- Repulsive forces calculation (optimized using a grid) ---
        // Populate the grid: each cell contains a list of node indices
        // that fall within its spatial bounds.
        let mut grid: Vec<Vec<Vec<usize>>> = vec![vec![vec![]; grid_dim]; grid_dim];
        for i in 0..n {
            let p_i = nodes[i].pos;
            // Calculate grid coordinates for node i, relative to min_coord
            let grid_x = ((p_i[0] - min_coord) / cell_size).floor() as usize;
            let grid_y = ((p_i[1] - min_coord) / cell_size).floor() as usize;

            // Clamp grid coordinates to ensure they are always within the grid bounds
            let grid_x = grid_x.min(grid_dim - 1).max(0);
            let grid_y = grid_y.min(grid_dim - 1).max(0);
            grid[grid_x][grid_y].push(i);
        }

        // Calculate repulsive forces for each node by looking at its neighbors in the grid
        for i in 0..n {
            let p_i = nodes[i].pos;
            let grid_x = ((p_i[0] - min_coord) / cell_size).floor() as usize;
            let grid_y = ((p_i[1] - min_coord) / cell_size).floor() as usize;
            let grid_x = grid_x.min(grid_dim - 1).max(0);
            let grid_y = grid_y.min(grid_dim - 1).max(0);

            // Iterate over the current cell and its 8 surrounding neighbor cells (3x3 area)
            for dx in -1..=1 {
                for dy in -1..=1 {
                    let neighbor_grid_x_isize = grid_x as isize + dx;
                    let neighbor_grid_y_isize = grid_y as isize + dy;

                    // Check if the neighbor cell coordinates are valid (within grid bounds)
                    if neighbor_grid_x_isize >= 0 && neighbor_grid_x_isize < grid_dim as isize &&
                       neighbor_grid_y_isize >= 0 && neighbor_grid_y_isize < grid_dim as isize {

                        let neighbor_grid_x = neighbor_grid_x_isize as usize;
                        let neighbor_grid_y = neighbor_grid_y_isize as usize;

                        // Iterate through all nodes found in the valid neighbor cell
                        for &j in &grid[neighbor_grid_x][neighbor_grid_y] {
                            if i == j { continue; } // A node does not exert repulsive force on itself

                            let p_j = nodes[j].pos;
                            let delta = [p_j[0] - p_i[0], p_j[1] - p_i[1]];
                            let distance_sq = delta[0].powi(2) + delta[1].powi(2);
                            let distance = distance_sq.sqrt();

                            if distance == 0.0 {
                                // If two nodes are at the exact same position, apply a small random displacement
                                // to prevent division by zero and help them separate.
                                nodes[i].disp[0] -= rng.gen_range(-1.0..1.0);
                                nodes[i].disp[1] -= rng.gen_range(-1.0..1.0);
                            } else {
                                // Apply Fruchterman-Reingold repulsive force: F_rep = k^2 / d
                                let force = c_rep / distance;
                                nodes[i].disp[0] -= (delta[0] / distance) * force;
                                nodes[i].disp[1] -= (delta[1] / distance) * force;
                            }
                        }
                    }
                }
            }
        }

        // --- Attractive forces calculation (O(M)) ---
        // Iterate through all edges and apply attractive forces between connected nodes.
        for u in 0..n {
            for &v in &adj[u] {
                // To avoid processing each edge twice (e.g., u-v and v-u),
                // and to prevent self-loops from affecting attractive forces.
                if u < v {
                    let p_u = nodes[u].pos;
                    let p_v = nodes[v].pos;

                    let delta = [p_v[0] - p_u[0], p_v[1] - p_u[1]];
                    let distance_sq = delta[0].powi(2) + delta[1].powi(2);
                    let distance = distance_sq.sqrt();

                    if distance != 0.0 {
                        // Apply Fruchterman-Reingold attractive force: F_attr = d^2 / k
                        let force = distance_sq / c_attr;
                        let disp_x = (delta[0] / distance) * force;
                        let disp_y = (delta[1] / distance) * force;

                        nodes[u].disp[0] += disp_x;
                        nodes[u].disp[1] += disp_y;
                        nodes[v].disp[0] -= disp_x; // Equal and opposite force on the other node
                        nodes[v].disp[1] -= disp_y;
                    }
                }
            }
        }

        // --- Update node positions and apply temperature ---
        for i in 0..n {
            let d_i = nodes[i].disp;

            let displacement_magnitude_sq = d_i[0].powi(2) + d_i[1].powi(2);
            let displacement_magnitude = displacement_magnitude_sq.sqrt();

            if displacement_magnitude > temperature {
                // If the displacement is too large, scale it down by the current temperature
                // This prevents nodes from moving too much in one step, ensuring stability.
                let scale = temperature / displacement_magnitude;
                nodes[i].pos[0] += d_i[0] * scale;
                nodes[i].pos[1] += d_i[1] * scale;
            } else {
                // Otherwise, apply the full calculated displacement
                nodes[i].pos[0] += d_i[0];
                nodes[i].pos[1] += d_i[1];
            }

            // Clamp node positions to the defined bounding box.
            // This prevents nodes from moving outside the drawing area and stabilizes the layout.
            nodes[i].pos[0] = nodes[i].pos[0].max(min_coord).min(max_coord);
            nodes[i].pos[1] = nodes[i].pos[1].max(min_coord).min(max_coord);
        }

        // Decrease the temperature. The layout "cools down", and node movements become smaller,
        // allowing the graph to settle into a stable configuration.
        temperature *= cooling_factor;
    }

    // --- Output Results ---
    let stdout = io::stdout();
    let mut handle = stdout.lock();
    // Output N on the first line, as specified by the problem (optional prefix).
    writeln!(handle, "{}", n)?;

    // Output the final x, y coordinates for each node, formatted to 6 decimal places.
    for node in nodes {
        writeln!(handle, "{:.6} {:.6}", node.pos[0], node.pos[1])?;
    }

    Ok(())
}
```
