// Basic HLSL Pixel Shader for Vulkan

// Input structure for the pixel shader
// For a very basic shader, we might not need any specific input from the vertex shader
// other than the system-generated SV_POSITION.
struct PSInput
{
    float4 Position : SV_POSITION; // Pixel position in screen space
};

// Output structure for the pixel shader
struct PSOutput
{
    float4 Color : SV_TARGET0; // Output color to render target 0
};

// Main pixel shader function
PSOutput main(PSInput input)
{
    PSOutput output;
    // Output a solid red color for now
    output.Color = float4(1.0f, 0.0f, 0.0f, 1.0f); // Red, Green, Blue, Alpha
    return output;
}
