struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Polka dot pattern calculation
    float2 cell_uv = input.uv * 6.0;
    float2 cell_coord = frac(cell_uv); // UV within the current cell [0,1]
    float2 center = float2(0.5, 0.5);
    float dist = distance(cell_coord, center);

    float3 pattern_color;
    if (dist < 0.3) {
        pattern_color = float3(1.0, 0.8, 0.0); // Yellow dot
    } else {
        pattern_color = float3(0.1, 0.1, 0.3); // Dark blue background
    }

    // Lambertian shading calculation
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos); // light direction

    float diffuse = max(0.0, dot(N, L));
    float shading_factor = 0.3 + 0.7 * diffuse;

    // Final color with shading
    float3 final_color = pattern_color * shading_factor;

    return float4(final_color, 1.0); // Output with alpha 1.0
}
