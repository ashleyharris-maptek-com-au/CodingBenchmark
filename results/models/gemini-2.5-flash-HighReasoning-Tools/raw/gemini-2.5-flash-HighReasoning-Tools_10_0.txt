#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric> // For std::iota
#include <limits>  // For std::numeric_limits

// Forward declaration for Rectangle struct
struct Rectangle;

// Represents an available space on a board
struct FreeRectangle {
    int x, y, width, height;

    // Helper to check if two rectangles intersect
    bool Intersects(const FreeRectangle& other) const {
        return x < other.x + other.width && x + width > other.x &&
               y < other.y + other.height && y + height > other.y;
    }

    // Helper to check if this FreeRectangle contains another
    bool Contains(const FreeRectangle& other) const {
        return x <= other.x && y <= other.y &&
               x + width >= other.x + other.width &&
               y + height >= other.y + other.height;
    }
};

// Represents an input rectangle
struct Rectangle {
    int id;             // Original index for output
    int width;
    int height;
    int original_width; // Stores original dimensions for rotation logic
    int original_height;
    int board_idx = -1; // Index of the board it's placed on
    int x = -1;         // X-coordinate of placement
    int y = -1;         // Y-coordinate of placement
    bool rotated = false; // True if rotated 90 degrees

    // Constructor
    Rectangle(int id, int w, int h) : id(id), width(w), height(h), original_width(w), original_height(h) {}

    // Helper to get area for sorting
    long long area() const {
        return static_cast<long long>(width) * height;
    }
};

// Helper function to subtract a rectangle 'B' from rectangle 'A'.
// Returns a list of new rectangles that represent the area of 'A' not covered by 'B'.
std::vector<FreeRectangle> SubtractRect(const FreeRectangle& A, const FreeRectangle& B) {
    std::vector<FreeRectangle> result;

    // If A and B don't intersect, A is unchanged
    if (!A.Intersects(B)) {
        result.push_back(A);
        return result;
    }

    // A list of candidate rectangles that cover the original 'A'
    // This creates up to 4 rectangles around B's intersection with A.
    // We effectively carve B out of A.

    // Top slice of A, above B
    if (A.y < B.y) {
        result.push_back({A.x, A.y, A.width, B.y - A.y});
    }
    // Bottom slice of A, below B
    if (A.y + A.height > B.y + B.height) {
        result.push_back({A.x, B.y + B.height, A.width, (A.y + A.height) - (B.y + B.height)});
    }

    // Left slice of A, to the left of B, bounded by B's vertical extent
    if (A.x < B.x) {
        int sub_y = std::max(A.y, B.y);
        int sub_height = std::min(A.y + A.height, B.y + B.height) - sub_y;
        if (sub_height > 0) {
            result.push_back({A.x, sub_y, B.x - A.x, sub_height});
        }
    }
    // Right slice of A, to the right of B, bounded by B's vertical extent
    if (A.x + A.width > B.x + B.width) {
        int sub_y = std::max(A.y, B.y);
        int sub_height = std::min(A.y + A.height, B.y + B.height) - sub_y;
        if (sub_height > 0) {
            result.push_back({B.x + B.width, sub_y, (A.x + A.width) - (B.x + B.width), sub_height});
        }
    }

    // Filter out invalid rectangles (width or height <= 0)
    std::vector<FreeRectangle> filtered_result;
    for (const auto& fr : result) {
        if (fr.width > 0 && fr.height > 0) {
            filtered_result.push_back(fr);
        }
    }
    return filtered_result;
}


// Manages the packing for a single board
struct Board {
    int id;
    int board_width;
    int board_height;
    std::vector<FreeRectangle> free_rects; // List of available spaces on this board

    Board(int id, int width, int height) : id(id), board_width(width), board_height(height) {
        free_rects.push_back({0, 0, width, height});
    }

    // Method to place a rectangle on this board
    bool PlaceRectangle(Rectangle& rect);

private:
    // Helper methods for MaxRects
    void SplitFreeRectangle(const Rectangle& placed_rect);
    void MergeFreeRectangles();
    // BSSF heuristic: returns the index of the best free rectangle
    int FindBestFit(int rect_width, int rect_height, int& out_x, int& out_y, bool& out_rotated);
};

// Implementation of Board methods

// Finds the best free rectangle to place the given rect_width, rect_height
// using Best Short Side Fit (BSSF) heuristic.
// Returns the index of the best free rectangle, or -1 if no fit.
int Board::FindBestFit(int rect_width, int rect_height, int& out_x, int& out_y, bool& out_rotated) {
    int best_free_rect_idx = -1;
    int best_score = std::numeric_limits<int>::max(); // BSSF score: min(remaining_short_side)

    // Try original orientation
    for (size_t i = 0; i < free_rects.size(); ++i) {
        const auto& free_rect = free_rects[i];
        if (rect_width <= free_rect.width && rect_height <= free_rect.height) {
            // Calculate remaining space after placement
            int remaining_width = free_rect.width - rect_width;
            int remaining_height = free_rect.height - rect_height;
            int score = std::min(remaining_width, remaining_height);

            if (score < best_score) {
                best_score = score;
                best_free_rect_idx = i;
                out_x = free_rect.x;
                out_y = free_rect.y;
                out_rotated = false;
            }
        }
    }

    // Try rotated orientation
    int rotated_rect_width = rect_height;
    int rotated_rect_height = rect_width;
    for (size_t i = 0; i < free_rects.size(); ++i) {
        const auto& free_rect = free_rects[i];
        if (rotated_rect_width <= free_rect.width && rotated_rect_height <= free_rect.height) {
            // Calculate remaining space after placement
            int remaining_width = free_rect.width - rotated_rect_width;
            int remaining_height = free_rect.height - rotated_rect_height;
            int score = std::min(remaining_width, remaining_height);

            // If this rotated fit is better than the current best (or if no best yet)
            // and it's not the same score as an unrotated fit that already won
            if (score < best_score) {
                best_score = score;
                best_free_rect_idx = i;
                out_x = free_rect.x;
                out_y = free_rect.y;
                out_rotated = true;
            }
        }
    }

    return best_free_rect_idx;
}

// Splits free rectangles based on the newly placed rectangle.
void Board::SplitFreeRectangle(const Rectangle& placed_rect) {
    std::vector<FreeRectangle> new_free_rects_temp;
    FreeRectangle placed_area = {placed_rect.x, placed_rect.y, placed_rect.width, placed_rect.height};

    for (const auto& free_rect : free_rects) {
        // If the free_rect does not intersect the placed_area, keep it as is.
        if (!free_rect.Intersects(placed_area)) {
            new_free_rects_temp.push_back(free_rect);
        } else {
            // If it intersects, subtract the placed_area from the free_rect.
            // This will produce new free rectangles (0 to 4) representing the remaining space.
            std::vector<FreeRectangle> subtracted_parts = SubtractRect(free_rect, placed_area);
            new_free_rects_temp.insert(new_free_rects_temp.end(), subtracted_parts.begin(), subtracted_parts.end());
        }
    }
    free_rects = new_free_rects_temp;
}

// Merges free rectangles to reduce redundancy.
void Board::MergeFreeRectangles() {
    // Sort free rectangles to help with merging.
    // Sort by x, then y, then width, then height.
    std::sort(free_rects.begin(), free_rects.end(), [](const FreeRectangle& a, const FreeRectangle& b) {
        if (a.x != b.x) return a.x < b.x;
        if (a.y != b.y) return a.y < b.y;
        if (a.width != b.width) return a.width < b.width;
        return a.height < b.height;
    });

    std::vector<FreeRectangle> merged_list;
    for (const auto& current_fr : free_rects) {
        bool contained = false;
        for (const auto& existing_fr : merged_list) {
            if (existing_fr.Contains(current_fr)) {
                contained = true;
                break;
            }
        }
        if (!contained) {
            merged_list.push_back(current_fr);
        }
    }
    free_rects = merged_list;
}

// Attempts to place a rectangle on this board.
bool Board::PlaceRectangle(Rectangle& rect) {
    int best_x, best_y;
    bool best_rotated;

    int current_rect_width = rect.width;
    int current_rect_height = rect.height;

    int best_free_rect_idx = FindBestFit(current_rect_width, current_rect_height, best_x, best_y, best_rotated);

    if (best_free_rect_idx != -1) {
        rect.x = best_x;
        rect.y = best_y;
        rect.rotated = best_rotated;
        rect.board_idx = this->id;

        if (best_rotated) {
            std::swap(rect.width, rect.height);
        }

        // Update free rectangles
        SplitFreeRectangle(rect);
        MergeFreeRectangles(); // Merge after splitting to clean up

        return true;
    }
    return false;
}
