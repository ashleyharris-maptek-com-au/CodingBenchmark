import sys
import os
sys.path.insert(0, os.path.abspath(os.path.join(os.path.dirname(__file__), '..')))
import time
from autopilot_system.perception.perception_module import PerceptionModule
from autopilot_system.prediction.prediction_module import PredictionModule
from autopilot_system.planning.planning_module import PlanningModule
from autopilot_system.control.control_module import ControlModule
from autopilot_system.data_models import VehicleState, LaneInformation, EnvironmentState, ControlCommands, Trajectory

class Autopilot:
    """
    Orchestrates the perception, prediction, planning, and control modules
    to enable autonomous driving.
    """
    def __init__(self):
        self.perception_module = PerceptionModule()
        self.prediction_module = PredictionModule()
        self.planning_module = PlanningModule()
        self.control_module = ControlModule()
        self.ego_state = VehicleState(x=0.0, y=0.0, speed=0.0, heading=0.0, acceleration=0.0, lane_id=1)
        self.current_trajectory = Trajectory(states=[])

    def run_autopilot_step(self,
                           other_vehicle_states: list[VehicleState],
                           lane_data: LaneInformation,
                           timestamp: float,
                           target_speed: float = 25.0) -> ControlCommands:
        """
        Executes one step of the autopilot cycle.

        Args:
            other_vehicle_states: Current states of other vehicles in the environment.
            lane_data: Information about the lanes.
            timestamp: Current simulation timestamp.
            target_speed: Desired speed for the ego vehicle.

        Returns:
            ControlCommands to be applied to the ego vehicle.
        """
        # 1. Perception
        environment_state = self.perception_module.perceive(
            ego_vehicle_state=self.ego_state,
            other_vehicle_states=other_vehicle_states,
            lane_data=lane_data,
            timestamp=timestamp
        )

        # 2. Prediction
        # The prediction module currently modifies other_vehicles in environment_state
        # For a more robust system, it would return new predicted objects
        environment_state.other_vehicles = self.prediction_module.predict(
            environment_state.other_vehicles,
            environment_state.lane_info
        )

        # 3. Planning
        self.current_trajectory = self.planning_module.plan(
            environment_state=environment_state,
            target_speed=target_speed
        )

        # 4. Control
        control_commands = self.control_module.control(
            current_state=self.ego_state,
            trajectory=self.current_trajectory
        )

        # Update ego vehicle state based on control commands (simplified for simulation)
        # In a real simulation, a vehicle dynamics model would be used.
        self._update_ego_state_from_commands(control_commands, timestamp)

        return control_commands

    def _update_ego_state_from_commands(self, commands: ControlCommands, delta_time: float):
        """
        A very simplified model to update the ego vehicle's state based on control commands.
        This would be replaced by a proper physics simulation in a real system.
        """
        # For demonstration, let's just apply the acceleration/braking effect
        # and assume steering changes heading directly.
        
        # Simplified acceleration calculation
        effective_acceleration = (commands.acceleration_pedal - commands.brake_pedal) * 5.0 # Max accel/decel
        
        new_speed = self.ego_state.speed + effective_acceleration * delta_time
        new_speed = max(0.0, new_speed) # Speed cannot be negative

        # Update position based on new speed and current heading
        # This assumes delta_time is very small, otherwise proper integration is needed
        new_x = self.ego_state.x + new_speed * delta_time * (1 if self.ego_state.heading == 0 else 0)
        new_y = self.ego_state.y + new_speed * delta_time * (1 if self.ego_state.heading == 90 else 0)
        
        # Update heading (simplified)
        new_heading = self.ego_state.heading + commands.steering_angle * delta_time # Steering changes heading
        
        self.ego_state = VehicleState(
            x=new_x,
            y=new_y,
            speed=new_speed,
            heading=new_heading,
            acceleration=effective_acceleration,
            lane_id=self.ego_state.lane_id # Lane change logic would update this
        )

def main():
    autopilot = Autopilot()
    
    # --- Simulation Setup ---
    # Initial states
    initial_ego_state = VehicleState(x=0.0, y=0.0, speed=0.0, heading=0.0, acceleration=0.0, lane_id=1)
    autopilot.ego_state = initial_ego_state
    
    # Example other vehicles
    other_vehicles_states = [
        VehicleState(x=50.0, y=0.0, speed=15.0, heading=0.0, acceleration=0.0, lane_id=1), # Car ahead
        VehicleState(x=30.0, y=3.75, speed=20.0, heading=0.0, acceleration=0.0, lane_id=2) # Car in left lane
    ]
    
    # Example lane data (assuming 3.75m wide lanes)
    lane_data = LaneInformation(
        current_lane_id=1,
        left_lane_id=2,
        right_lane_id=None,
        lane_boundaries=[(0.0, 3.75), (3.75, 7.5)] # Lane 1 (0-3.75), Lane 2 (3.75-7.5)
    )

    simulation_time = 0.0
    time_step = 0.1 # seconds
    target_speed = 25.0 # m/s

    print("--- Starting Autopilot Simulation ---")
    print(f"Initial Ego State: {autopilot.ego_state}")
    print("-" * 30)

    for i in range(100): # Run for 10 seconds
        commands = autopilot.run_autopilot_step(
            other_vehicles_states,
            lane_data,
            simulation_time,
            target_speed
        )
        simulation_time += time_step

        print(f"Time: {simulation_time:.2f}s")
        print(f"  Ego State: X={autopilot.ego_state.x:.2f}, Y={autopilot.ego_state.y:.2f}, "
              f"Speed={autopilot.ego_state.speed:.2f}m/s, Heading={autopilot.ego_state.heading:.2f}rad")
        print(f"  Control: Accel={commands.acceleration_pedal:.2f}, Brake={commands.brake_pedal:.2f}, "
              f"Steering={commands.steering_angle:.2f}")
        print("-" * 30)
        
        # In a real simulation, you'd update other_vehicles_states based on their predictions
        # and the simulation physics. Here, they remain static for simplicity.
        # For a more dynamic simulation, you would have a simulation environment
        # that updates all vehicle states and provides them to the autopilot.

    print("--- Simulation Finished ---")

if __name__ == "__main__":
    main()
