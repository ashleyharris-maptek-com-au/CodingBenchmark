#include "steiner_tree_solver.h"
#include <set> // For cycle removal in reconstruct_steiner_tree
#include <queue> // For BFS in reconstruct_steiner_tree

SteinerTreeSolver::SteinerTreeSolver(const Graph& graph, const std::vector<int>& terminals)
    : original_graph(graph), terminals(terminals), num_original_nodes(graph.get_num_nodes()) {
}

std::pair<int, Graph> SteinerTreeSolver::solve() {
    std::map<std::pair<int, int>, std::vector<int>> shortest_paths_map;
    
    // Step 1 & 2: Build complete graph on terminals and compute APSP
    std::vector<Edge> terminal_edges = build_terminal_graph(shortest_paths_map);

    // Step 3: Find MST in the terminal graph
    std::vector<Edge> mst_edges = find_mst(terminal_edges, terminals.size());

    // Step 4: Reconstruct Steiner Tree in original graph
    Graph steiner_tree = reconstruct_steiner_tree(mst_edges, shortest_paths_map);
    
    int total_weight = 0;
    for (int u = 0; u < steiner_tree.get_num_nodes(); ++u) {
        for (const auto& edge : steiner_tree.get_adj()[u]) {
            // Since edges are added twice (u,v) and (v,u), divide by 2
            if (u < edge.first) { 
                total_weight += edge.second;
            }
        }
    }

    return {total_weight, steiner_tree};
}

std::vector<Edge> SteinerTreeSolver::build_terminal_graph(std::map<std::pair<int, int>, std::vector<int>>& shortest_paths_map) const {
    std::vector<Edge> terminal_edges;
    int num_terminals = terminals.size();

    // Compute all-pairs shortest paths for terminals
    for (int i = 0; i < num_terminals; ++i) {
        int start_terminal = terminals[i];
        std::vector<int> dist;
        std::vector<int> prev;
        original_graph.dijkstra(start_terminal, dist, prev);

        for (int j = i + 1; j < num_terminals; ++j) {
            int end_terminal = terminals[j];
            int weight = dist[end_terminal];
            
            if (weight != INF) {
                terminal_edges.push_back({start_terminal, end_terminal, weight});
                shortest_paths_map[{start_terminal, end_terminal}] = original_graph.get_shortest_path(start_terminal, end_terminal, prev);
                shortest_paths_map[{end_terminal, start_terminal}] = original_graph.get_shortest_path(end_terminal, start_terminal, prev);
            }
        }
    }
    return terminal_edges;
}

std::vector<Edge> SteinerTreeSolver::find_mst(const std::vector<Edge>& terminal_edges, int num_terminal_nodes) const {
    std::vector<Edge> mst_edges;
    UnionFind uf(num_original_nodes); // UnionFind on original graph nodes

    // Sort edges by weight
    std::vector<Edge> sorted_edges = terminal_edges;
    std::sort(sorted_edges.begin(), sorted_edges.end());

    for (const auto& edge : sorted_edges) {
        if (uf.find(edge.u) != uf.find(edge.v)) {
            uf.unite(edge.u, edge.v);
            mst_edges.push_back(edge);
            // Optimization: If num_terminal_nodes - 1 edges are added, MST is complete
            if (mst_edges.size() == num_terminal_nodes - 1) {
                break;
            }
        }
    }
    return mst_edges;
}

Graph SteinerTreeSolver::reconstruct_steiner_tree(const std::vector<Edge>& mst_edges,
                                                   const std::map<std::pair<int, int>, std::vector<int>>& shortest_paths_map) const {
    Graph steiner_tree(num_original_nodes);
    std::set<std::pair<int, int>> added_edges; // To avoid duplicate edges and cycles

    for (const auto& mst_edge : mst_edges) {
        int u = mst_edge.u;
        int v = mst_edge.v;

        std::vector<int> path = shortest_paths_map.at({u, v});

        for (size_t i = 0; i < path.size() - 1; ++i) {
            int node1 = path[i];
            int node2 = path[i+1];

            // Ensure consistent order for edge (node1, node2) to avoid duplicates
            if (node1 > node2) std::swap(node1, node2);

            if (added_edges.find({node1, node2}) == added_edges.end()) {
                // Find the weight of this edge in the original graph
                int weight = -1; // Should always be found
                for (const auto& original_edge : original_graph.get_adj()[node1]) {
                    if (original_edge.first == node2) {
                        weight = original_edge.second;
                        break;
                    }
                }
                
                if (weight != -1) {
                    steiner_tree.add_edge(node1, node2, weight);
                    added_edges.insert({node1, node2});
                }
            }
        }
    }
    return steiner_tree;
}
