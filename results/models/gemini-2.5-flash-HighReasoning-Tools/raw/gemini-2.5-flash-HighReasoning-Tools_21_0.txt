#include <iostream>
#include <vector>
#include "VectorN.h"

int main() {
    std::cout << "--- VectorN Test ---" << std::endl;

    // Test constructor with dimensions and initial value
    VectorN v1(3, 1); // 3 dimensions, all 1s
    std::cout << "v1 (3D, all 1s): " << v1 << std::endl; // Expected: (1, 1, 1)

    // Test constructor with a vector of coordinates
    std::vector<int> coords2D = {5, 10};
    VectorN v2(coords2D);
    std::cout << "v2 (2D, {5, 10}): " << v2 << std::endl; // Expected: (5, 10)

    std::vector<int> coords4D = {1, 2, 3, 4};
    VectorN v3(coords4D);
    std::cout << "v3 (4D, {1, 2, 3, 4}): " << v3 << std::endl; // Expected: (1, 2, 3, 4)

    // Test getDimensionCount
    std::cout << "v1 dimension count: " << v1.getDimensionCount() << std::endl; // Expected: 3
    std::cout << "v2 dimension count: " << v2.getDimensionCount() << std::endl; // Expected: 2

    // Test getCoordinate and setCoordinate
    std::cout << "v1 coord[0]: " << v1.getCoordinate(0) << std::endl; // Expected: 1
    v1.setCoordinate(0, 99);
    std::cout << "v1 after setCoord[0]=99: " << v1 << std::endl; // Expected: (99, 1, 1)

    // Test operator+
    VectorN v4(3, 5); // (5, 5, 5)
    VectorN v_sum = v1 + v4;
    std::cout << "v1 + v4: " << v_sum << std::endl; // Expected: (104, 6, 6)

    // Test operator-
    VectorN v_diff = v4 - v1;
    std::cout << "v4 - v1: " << v_diff << std::endl; // Expected: (-94, 4, 4)

    // Test operator== and operator!=
    VectorN v5(3, 1); // (1, 1, 1)
    VectorN v6(3, 2); // (2, 2, 2)
    std::cout << "v1 == v5: " << (v1 == v5 ? "true" : "false") << std::endl; // Expected: false (v1 is now (99,1,1))
    VectorN v1_reset(3,1);
    std::cout << "v1_reset == v5: " << (v1_reset == v5 ? "true" : "false") << std::endl; // Expected: true
    std::cout << "v5 != v6: " << (v5 != v6 ? "true" : "false") << std::endl; // Expected: true

    // Test invalid_argument for constructor
    try {
        VectorN invalid_v(0);
    } catch (const std::invalid_argument& e) {
        std::cout << "Caught expected error for 0 dimensions: " << e.what() << std::endl;
    }
    try {
        VectorN invalid_v({});
    } catch (const std::invalid_argument& e) {
        std::cout << "Caught expected error for empty coords: " << e.what() << std::endl;
    }

    // Test out_of_range for get/set coordinate
    try {
        v2.getCoordinate(2); // v2 is 2D, index 2 is out of range
    } catch (const std::out_of_range& e) {
        std::cout << "Caught expected error for out of range getCoordinate: " << e.what() << std::endl;
    }
    try {
        v2.setCoordinate(-1, 0); // index -1 is out of range
    } catch (const std::out_of_range& e) {
        std::cout << "Caught expected error for out of range setCoordinate: " << e.what() << std::endl;
    }

    // Test invalid_argument for operator+ with mismatched dimensions
    try {
        VectorN test_sum = v1 + v2;
    } catch (const std::invalid_argument& e) {
        std::cout << "Caught expected error for mismatched dimensions addition: " << e.what() << std::endl;
    }

    std::cout << "--- VectorN Test End ---" << std::endl;

    return 0;
}
