// A basic HLSL pixel shader for Vulkan
// This shader outputs a fixed red color.

struct PSInput
{
    float4 Position : SV_POSITION;
};

struct PSOutput
{
    float4 Color : SV_TARGET;
};

PSOutput main(PSInput input)
{
    PSOutput output;
    output.Color = float4(1.0f, 0.0f, 0.0f, 1.0f); // Red color
    return output;
}
