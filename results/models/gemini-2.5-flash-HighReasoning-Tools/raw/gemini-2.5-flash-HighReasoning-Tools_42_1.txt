    OpEntryPoint Geometry %main "main" %_ %gl_in %gl_Position %worldPos %color %outColor %gl_Position_0
    OpExecutionMode %main InputPoints
    OpExecutionMode %main OutputTriangleStrip
    OpExecutionMode %main OutputVertices 4
    OpExecutionMode %main Invocations 1

    OpSource GLSL 450
    OpName %main "main"
    OpName %gl_PerVertex "gl_PerVertex"
    OpMemberName %gl_PerVertex 0 "gl_Position"
    OpName %gl_in "gl_in"
    OpName %gl_Position "gl_Position"
    OpName %worldPos "worldPos"
    OpName %color "color"
    OpName %gl_PerVertex_0 "gl_PerVertex"
    OpMemberName %gl_PerVertex_0 0 "gl_Position"
    OpName %gl_out "gl_out"
    OpName %outColor "outColor"
    OpName %gl_Position_0 "gl_Position"

    OpDecorate %gl_PerVertex Block
    OpMemberDecorate %gl_PerVertex 0 BuiltIn Position
    OpDecorate %gl_in BuiltIn PerVertex
    OpDecorate %worldPos Location 0
    OpDecorate %color Location 2
    OpDecorate %gl_PerVertex_0 Block
    OpMemberDecorate %gl_PerVertex_0 0 BuiltIn Position
    OpDecorate %gl_out BuiltIn PerVertex
    OpDecorate %outColor Location 0
    OpDecorate %gl_Position_0 BuiltIn Position

    %void = OpTypeVoid
    %void_func = OpTypeFunction %void
    %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %gl_PerVertex = OpTypeStruct %v4float
    %uint = OpTypeInt 32 0
    %uint_1 = OpConstant %uint 1
    %arr_gl_PerVertex = OpTypeArray %gl_PerVertex %uint_1
    %ptr_Input_arr_gl_PerVertex = OpTypePointer Input %arr_gl_PerVertex
    %ptr_Input_v4float = OpTypePointer Input %v4float
    %gl_in = OpVariable %ptr_Input_arr_gl_PerVertex Input

    %v3float = OpTypeVector %float 3
    %arr_v3float_1 = OpTypeArray %v3float %uint_1
    %ptr_Input_arr_v3float_1 = OpTypePointer Input %arr_v3float_1
    %ptr_Input_v3float = OpTypePointer Input %v3float
    %worldPos = OpVariable %ptr_Input_arr_v3float_1 Input
    %color = OpVariable %ptr_Input_arr_v3float_1 Input

    %gl_PerVertex_0 = OpTypeStruct %v4float
    %ptr_Output_gl_PerVertex_0 = OpTypePointer Output %gl_PerVertex_0
    %ptr_Output_v4float = OpTypePointer Output %v4float
    %gl_out = OpVariable %ptr_Output_gl_PerVertex_0 Output

    %ptr_Output_v3float = OpTypePointer Output %v3float
    %outColor = OpVariable %ptr_Output_v3float Output

    %float_0_0 = OpConstant %float 0.0
    %float_0_05 = OpConstant %float 0.05
    %int_0 = OpConstant %uint 0
    %int_1 = OpConstant %uint 1
    %int_2 = OpConstant %uint 2
    %int_3 = OpConstant %uint 3
    %v4float_0_0_0_1_0 = OpConstantComposite %v4float %float_0_0 %float_0_0 %float_0_0 %float_1_0

    %main = OpFunction %void None %void_func
    %entry = OpLabel
    ; Read input position and color
    %idx_0 = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
    %in_pos = OpLoad %v4float %idx_0

    %idx_color = OpAccessChain %ptr_Input_v3float %color %int_0
    %in_color = OpLoad %v3float %idx_color

    %x_orig = OpCompositeExtract %float %in_pos %int_0
    %y_orig = OpCompositeExtract %float %in_pos %int_1
    %z_orig = OpCompositeExtract %float %in_pos %int_2
    %w_orig = OpCompositeExtract %float %in_pos %int_3

    ; Top vertex: (x, y+h)
    %y_top = OpFAdd %float %y_orig %float_0_05
    %pos_top = OpCompositeConstruct %v4float %x_orig %y_top %z_orig %w_orig
    OpStore %gl_Position_0 %pos_top
    OpStore %outColor %in_color
    OpEmitVertex

    ; Right vertex: (x+h, y)
    %x_right = OpFAdd %float %x_orig %float_0_05
    %pos_right = OpCompositeConstruct %v4float %x_right %y_orig %z_orig %w_orig
    OpStore %gl_Position_0 %pos_right
    OpStore %outColor %in_color
    OpEmitVertex

    ; Left vertex: (x-h, y)
    %x_left = OpFSub %float %x_orig %float_0_05
    %pos_left = OpCompositeConstruct %v4float %x_left %y_orig %z_orig %w_orig
    OpStore %gl_Position_0 %pos_left
    OpStore %outColor %in_color
    OpEmitVertex

    ; Bottom vertex: (x, y-h)
    %y_bottom = OpFSub %float %y_orig %float_0_05
    %pos_bottom = OpCompositeConstruct %v4float %x_orig %y_bottom %z_orig %w_orig
    OpStore %gl_Position_0 %pos_bottom
    OpStore %outColor %in_color
    OpEmitVertex

    OpEndPrimitive
    OpReturn
    OpFunctionEnd
