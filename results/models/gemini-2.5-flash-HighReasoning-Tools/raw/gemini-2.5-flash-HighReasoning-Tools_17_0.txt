using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    // Represents the container
    public class Container
    {
        public int Width { get; }
        public int Height { get; }
        public int Depth { get; }

        public Container(int width, int height, int depth)
        {
            Width = width;
            Height = height;
            Depth = depth;
        }
    }

    // Represents an input box before placement
    public class InputBox
    {
        public int Id { get; }
        public int OriginalWidth { get; }
        public int OriginalHeight { get; }
        public int OriginalDepth { get; }
        public long Volume { get; }

        public InputBox(int id, int w, int h, int d)
        {
            Id = id;
            OriginalWidth = w;
            OriginalHeight = h;
            OriginalDepth = d;
            Volume = (long)w * h * d;
        }
    }

    // Represents a box after it has been placed, including its orientation and position
    public class PackedBox
    {
        public int Id { get; }
        public int X { get; }
        public int Y { get; }
        public int Z { get; }
        public int Width { get; } // Actual width used after rotation
        public int Height { get; } // Actual height used after rotation
        public int Depth { get; } // Actual depth used after rotation

        public PackedBox(int id, int x, int y, int z, int w, int h, int d)
        {
            Id = id;
            X = x;
            Y = y;
            Z = z;
            Width = w;
            Height = h;
            Depth = d;
        }
    }

    // Represents an available rectangular 3D space in the container
    public class Rect3D
    {
        public int X { get; }
        public int Y { get; }
        public int Z { get; }
        public int Width { get; }
        public int Height { get; }
        public int Depth { get; }

        public Rect3D(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            Width = w;
            Height = h;
            Depth = d;
        }

        // Checks if a box of given dimensions fits into this space
        public bool Fits(int boxW, int boxH, int boxD)
        {
            return boxW <= Width && boxH <= Height && boxD <= Depth;
        }

        // Checks if this space overlaps with another PackedBox
        public bool Intersects(PackedBox other)
        {
            // The PackedBox's X, Y, Z are its bottom-left-front corner
            return X < other.X + other.Width && X + Width > other.X &&
                   Y < other.Y + other.Height && Y + Height > other.Y &&
                   Z < other.Z + other.Depth && Z + Depth > other.Z;
        }
        
        // Checks if this space is completely contained within another Rect3D
        public bool IsContainedIn(Rect3D other)
        {
            return other.X <= X && other.Y <= Y && other.Z <= Z &&
                   other.X + other.Width >= X + Width &&
                   other.Y + other.Height >= Y + Height &&
                   other.Z + other.Depth >= Z + Depth;
        }
    }

    // Subtracts a cutting box (PackedBox) from an original space (Rect3D)
    // Returns a list of new Rect3D representing the remaining space (disjoint).
    public static List<Rect3D> SubtractSpace(Rect3D originalSpace, PackedBox cuttingBox)
    {
        List<Rect3D> result = new List<Rect3D>();

        // If they don't intersect, the original space remains intact
        if (!originalSpace.Intersects(cuttingBox))
        {
            result.Add(originalSpace);
            return result;
        }

        // Split along X-axis
        // Left part
        if (cuttingBox.X > originalSpace.X)
            result.Add(new Rect3D(originalSpace.X, originalSpace.Y, originalSpace.Z, cuttingBox.X - originalSpace.X, originalSpace.Height, originalSpace.Depth));
        // Right part
        if (cuttingBox.X + cuttingBox.Width < originalSpace.X + originalSpace.Width)
            result.Add(new Rect3D(cuttingBox.X + cuttingBox.Width, originalSpace.Y, originalSpace.Z, (originalSpace.X + originalSpace.Width) - (cuttingBox.X + cuttingBox.Width), originalSpace.Height, originalSpace.Depth));

        // Calculate the overlapping X-region for Y and Z splits
        int overlapX1 = Math.Max(originalSpace.X, cuttingBox.X);
        int overlapX2 = Math.Min(originalSpace.X + originalSpace.Width, cuttingBox.X + cuttingBox.Width);
        int overlapWidth = overlapX2 - overlapX1;

        if (overlapWidth > 0) // If there's an X-overlap
        {
            // Split along Y-axis for the overlapping X-region
            // Bottom part
            if (cuttingBox.Y > originalSpace.Y)
                result.Add(new Rect3D(overlapX1, originalSpace.Y, originalSpace.Z, overlapWidth, cuttingBox.Y - originalSpace.Y, originalSpace.Depth));
            // Top part
            if (cuttingBox.Y + cuttingBox.Height < originalSpace.Y + originalSpace.Height)
                result.Add(new Rect3D(overlapX1, cuttingBox.Y + cuttingBox.Height, originalSpace.Z, overlapWidth, (originalSpace.Y + originalSpace.Height) - (cuttingBox.Y + cuttingBox.Height), originalSpace.Depth));
        }

        // Calculate the overlapping Y-region for Z splits
        int overlapY1 = Math.Max(originalSpace.Y, cuttingBox.Y);
        int overlapY2 = Math.Min(originalSpace.Y + originalSpace.Height, cuttingBox.Y + cuttingBox.Height);
        int overlapHeight = overlapY2 - overlapY1;

        // If there's an XY-overlap
        if (overlapWidth > 0 && overlapHeight > 0)
        {
            // Split along Z-axis for the overlapping XY-region
            // Back part
            if (cuttingBox.Z > originalSpace.Z)
                result.Add(new Rect3D(overlapX1, overlapY1, originalSpace.Z, overlapWidth, overlapHeight, cuttingBox.Z - originalSpace.Z));
            // Front part
            if (cuttingBox.Z + cuttingBox.Depth < originalSpace.Z + originalSpace.Depth)
                result.Add(new Rect3D(overlapX1, overlapY1, cuttingBox.Z + cuttingBox.Depth, overlapWidth, overlapHeight, (originalSpace.Z + originalSpace.Depth) - (cuttingBox.Z + cuttingBox.Depth)));
        }
        
        // Filter out zero-volume spaces
        return result.Where(s => s.Width > 0 && s.Height > 0 && s.Depth > 0).ToList();
    }


    public static void Main(string[] args)
    {
        // Read container and box count
        string[] line1 = Console.ReadLine().Split(' ');
        int N = int.Parse(line1[0]);
        Container container = new Container(int.Parse(line1[1]), int.Parse(line1[2]), int.Parse(line1[3]));

        List<InputBox> inputBoxes = new List<InputBox>(N);
        for (int i = 0; i < N; i++)
        {
            string[] boxDims = Console.ReadLine().Split(' ');
            inputBoxes.Add(new InputBox(i, int.Parse(boxDims[0]), int.Parse(boxDims[1]), int.Parse(boxDims[2])));
        }

        // Sort boxes by decreasing longest dimension, then by volume (common heuristic for better packing)
        inputBoxes = inputBoxes.OrderByDescending(b => Math.Max(b.OriginalWidth, Math.Max(b.OriginalHeight, b.OriginalDepth)))
                               .ThenByDescending(b => b.Volume)
                               .ToList();

        List<PackedBox> packedBoxes = new List<PackedBox>();
        // List of available empty spaces, initially the entire container
        List<Rect3D> availableSpaces = new List<Rect3D>();
        availableSpaces.Add(new Rect3D(0, 0, 0, container.Width, container.Height, container.Depth));

        foreach (var inputBox in inputBoxes)
        {
            // Try all 6 permutations for box dimensions
            int[][] rotations = new int[][]
            {
                new int[] { inputBox.OriginalWidth, inputBox.OriginalHeight, inputBox.OriginalDepth },
                new int[] { inputBox.OriginalWidth, inputBox.OriginalDepth, inputBox.OriginalHeight },
                new int[] { inputBox.OriginalHeight, inputBox.OriginalWidth, inputBox.OriginalDepth },
                new int[] { inputBox.OriginalHeight, inputBox.OriginalDepth, inputBox.OriginalWidth },
                new int[] { inputBox.OriginalDepth, inputBox.OriginalWidth, inputBox.OriginalHeight },
                new int[] { inputBox.OriginalDepth, inputBox.OriginalHeight, inputBox.OriginalWidth }
            };

            // Keep track of the best placement for the current box
            PackedBox currentBestPlacementCandidate = null;
            Rect3D currentBestSpaceCandidate = null;
            
            foreach (var rotation in rotations)
            {
                int currentBoxW = rotation[0];
                int currentBoxH = rotation[1];
                int currentBoxD = rotation[2];

                // Find the best available space for this rotated box
                // Prioritize lower Z, then lower Y, then lower X (bottom-left-front strategy)
                foreach (var space in availableSpaces)
                {
                    if (space.Fits(currentBoxW, currentBoxH, currentBoxD))
                    {
                        // Current potential placement is at the origin of the available space
                        PackedBox potentialPlacement = new PackedBox(inputBox.Id, space.X, space.Y, space.Z, currentBoxW, currentBoxH, currentBoxD);

                        // Check if this placement is better than the current best
                        if (currentBestPlacementCandidate == null ||
                            potentialPlacement.Z < currentBestPlacementCandidate.Z ||
                            (potentialPlacement.Z == currentBestPlacementCandidate.Z && potentialPlacement.Y < currentBestPlacementCandidate.Y) ||
                            (potentialPlacement.Z == currentBestPlacementCandidate.Z && potentialPlacement.Y == currentBestPlacementCandidate.Y && potentialPlacement.X < currentBestPlacementCandidate.X))
                        {
                            currentBestPlacementCandidate = potentialPlacement;
                            currentBestSpaceCandidate = space;
                        }
                    }
                }
            }

            if (currentBestPlacementCandidate != null)
            {
                packedBoxes.Add(currentBestPlacementCandidate);

                List<Rect3D> nextAvailableSpaces = new List<Rect3D>();
                foreach (var space in availableSpaces)
                {
                    // If the current 'space' is the one where 'currentBestPlacementCandidate' was made,
                    // we subtract the placed box from it and add the resulting new free spaces.
                    if (space.X == currentBestSpaceCandidate.X && space.Y == currentBestSpaceCandidate.Y && space.Z == currentBestSpaceCandidate.Z &&
                        space.Width == currentBestSpaceCandidate.Width && space.Height == currentBestSpaceCandidate.Height && space.Depth == currentBestSpaceCandidate.Depth)
                    {
                        nextAvailableSpaces.AddRange(SubtractSpace(space, currentBestPlacementCandidate));
                    }
                    else
                    {
                        // Other spaces are unaffected. Add them directly.
                        nextAvailableSpaces.Add(space);
                    }
                }

                // Filter out contained spaces and unique the list.
                // This step is important for performance and correctness by keeping the list of available spaces clean.
                availableSpaces = new List<Rect3D>();
                foreach (var s1 in nextAvailableSpaces)
                {
                    bool contained = false;
                    foreach (var s2 in nextAvailableSpaces)
                    {
                        if (!ReferenceEquals(s1, s2) && s1.IsContainedIn(s2))
                        {
                            contained = true;
                            break;
                        }
                    }
                    if (!contained)
                    {
                        availableSpaces.Add(s1);
                    }
                }
                // Sort available spaces to maintain some order, potentially helping with consistent placement or future pruning
                availableSpaces = availableSpaces.OrderBy(s => s.Z).ThenBy(s => s.Y).ThenBy(s => s.X).ToList();
            }
        }

        // Output results
        Console.WriteLine(packedBoxes.Count);
        foreach (var box in packedBoxes)
        {
            Console.WriteLine($"{box.Id} {box.X} {box.Y} {box.Z}");
        }
    }
}
