// wgsl_pow_shader.wgsl

// A simplified "hash" function for demonstration purposes.
// This is NOT cryptographically secure and should not be used in a real blockchain.
// It aims to mimic the behavior of a hash function by combining input and producing
// a seemingly random output.
fn simple_hash(nonce: u32, header: array<u32, 8>) -> u32 {
    var hash = nonce;
    for (var i = 0u; i < arrayLength(&header); i = i + 1u) {
        hash = hash ^ header[i]; // XOR with header parts for mixing
        hash = hash * 0x01000193; // FNV-1a prime for more mixing
        // The modulo operation ensures the hash stays within u32 range,
        // though WGSL u32 operations naturally wrap.
        // Explicit modulo might be needed for specific hash variants,
        // but for this simple example, it's illustrative.
        // hash = hash % 0xFFFFFFFF;
    }
    hash = hash ^ (hash >> 16u); // Another bitwise mix
    return hash;
}

// Struct to hold input data for the mining operation.
// This data would be provided by the CPU via a storage buffer.
struct BlockData {
    header: array<u32, 8>,      // Example: 8 u32 values representing a block header or data chunk
    target_difficulty: u32,     // Number of leading zeros required in the hash for a valid solution
    start_nonce: u32            // The starting nonce for this batch of workgroups
};

// Input buffer: Read-only storage for block data.
@group(0) @binding(0) var<storage, read> block_data: BlockData;

// Struct to hold the solution found by the compute shader.
// Using atomic types ensures that only one valid solution is recorded
// if multiple threads find one concurrently.
struct Solution {
    nonce: atomic<u32>,        // The nonce that resulted in a valid hash
    hash_result: atomic<u32>   // The hash result that met the difficulty target
};

// Output buffer: Read-write storage for the solution.
// Initial value of solution.nonce should be 0xFFFFFFFF (or a sentinel value) to indicate no solution yet.
@group(0) @binding(1) var<storage, read_write> solution: Solution;

// The main compute shader entry point.
// @workgroup_size defines the number of threads in each workgroup.
// (64, 1, 1) means 64 threads in the X dimension, 1 in Y, 1 in Z.
@compute @workgroup_size(64, 1, 1)
fn main(@builtin(global_invocation_id) global_id: vec3<u32>,
        @builtin(local_invocation_id) local_id: vec3<u32>,
        @builtin(workgroup_id) workgroup_id: vec3<u32>) {

    // Calculate the unique nonce candidate for this thread.
    // Each workgroup starts at `block_data.start_nonce + (workgroup_id.x * 64u)`.
    // Each thread within the workgroup then adds its `local_id.x`.
    let current_nonce_candidate = block_data.start_nonce +
                                 (workgroup_id.x * 64u) +
                                 local_id.x;

    // Perform the simplified hash calculation.
    let hash_result = simple_hash(current_nonce_candidate, block_data.header);

    // Check if the hash meets the difficulty target (number of leading zeros).
    var leading_zeros = 0u;
    var temp_hash = hash_result;

    // Count leading zeros. This is a simplified way to check difficulty.
    // In a real PoW, you might compare against a large target hash value.
    for (var i = 0u; i < 32u; i = i + 1u) {
        if ((temp_hash & 0x80000000u) == 0u) { // Check if the most significant bit is 0
            leading_zeros = leading_zeros + 1u;
            temp_hash = temp_hash << 1u; // Shift left to check the next bit
        } else {
            break; // Stop counting if a '1' is encountered
        }
    }

    if (leading_zeros >= block_data.target_difficulty) {
        // A solution is found!
        // Atomically try to store the nonce. The first thread to successfully
        // change `solution.nonce` from its initial sentinel value (0xFFFFFFFF)
        // is the one whose solution we keep.
        // We initialize solution.nonce to 0xFFFFFFFF on the CPU side before dispatch.
        let initial_solution_nonce = 0xFFFFFFFFu;
        let old_nonce = atomicCompareExchangeWeak(&solution.nonce, initial_solution_nonce, current_nonce_candidate).old_value;

        // If this thread was the first to write, also store the hash result.
        if (old_nonce == initial_solution_nonce) {
            atomicStore(&solution.hash_result, hash_result);
        }
    }
}
