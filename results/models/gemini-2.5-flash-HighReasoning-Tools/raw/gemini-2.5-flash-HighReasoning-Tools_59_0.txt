const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const probability = data.probability;
    const drones = data.drones;
    const maxStepsPerDrone = data.maxStepsPerDrone;

    const gridHeight = probability.length;
    const gridWidth = probability[0].length;

    // Initialize paths for each drone
    const paths = drones.map(drone => [[drone.x, drone.y]]);

    // Initialize current positions for each drone
    const currentPositions = drones.map(drone => ({ x: drone.x, y: drone.y }));

    // Keep track of all cells covered by any drone so far (as "x,y" strings)
    const coveredCells = new Set();
    drones.forEach(drone => coveredCells.add(`${drone.x},${drone.y}`));

    // Possible moves: stay, right, left, down, up
    const moveDirections = [[0, 0], [0, 1], [0, -1], [1, 0], [-1, 0]];

    for (let step = 0; step < maxStepsPerDrone; step++) {
        // Collect all possible move options for all drones at the start of this step
        // Each option: { drone_idx, next_pos: [x,y], gain }
        const allMoveOptions = [];
        for (let i = 0; i < drones.length; i++) {
            const { x: cx, y: cy } = currentPositions[i];

            for (const [dx, dy] of moveDirections) {
                const nx = cx + dx;
                const ny = cy + dy;

                // Check boundary and obstacle
                if (nx < 0 || nx >= gridHeight || ny < 0 || ny >= gridWidth || probability[nx][ny] === -1) {
                    continue; // Invalid move
                }

                // Calculate potential gain
                let gain = 0;
                if (!coveredCells.has(`${nx},${ny}`)) {
                    gain = probability[nx][ny];
                }
                allMoveOptions.push({ drone_idx: i, next_pos: [nx, ny], gain });
            }
        }

        // Determine the next position for each drone for this step
        // Using a greedy approach: at each sub-iteration, assign the move with the highest gain
        // among unassigned drones to an un-claimed cell for this step.
        const nextPositionsForDrones = new Array(drones.length).fill(null); // What each drone will move to
        const cellsClaimedInThisStep = new Set(); // Cells claimed by drones in this specific step

        // Sort all move options by gain in descending order
        allMoveOptions.sort((a, b) => b.gain - a.gain);

        // Iterate through sorted moves to assign positions
        for (const move of allMoveOptions) {
            const { drone_idx, next_pos, gain } = move;

            // If this drone already has a chosen move for this step, skip
            if (nextPositionsForDrones[drone_idx] !== null) {
                continue;
            }

            const targetCellStr = `${next_pos[0]},${next_pos[1]}`;

            // Check if the target cell is not globally covered AND not claimed by another drone in *this* step
            if (!coveredCells.has(targetCellStr) && !cellsClaimedInThisStep.has(targetCellStr)) {
                nextPositionsForDrones[drone_idx] = next_pos;
                cellsClaimedInThisStep.add(targetCellStr);
            }
            // If the target cell is already covered or claimed, this move is not the best for unique gain
            // so this drone will try to find another move later in the sorted list, or default to staying put.
        }

        // For any drone that hasn't been assigned a move yet (because its best options were taken or covered),
        // assign it to stay in its current position.
        for (let i = 0; i < drones.length; i++) {
            if (nextPositionsForDrones[i] === null) {
                nextPositionsForDrones[i] = [currentPositions[i].x, currentPositions[i].y];
            }
        }

        // Update global paths and covered cells for the next step
        for (let i = 0; i < drones.length; i++) {
            const nextX = nextPositionsForDrones[i][0];
            const nextY = nextPositionsForDrones[i][1];

            currentPositions[i].x = nextX;
            currentPositions[i].y = nextY;
            paths[i].push(nextPositionsForDrones[i]);
            coveredCells.add(`${nextX},${nextY}`); // Mark newly visited cells as covered
        }
    }

    // Output the result in the specified JSON format
    console.log(JSON.stringify({ paths: paths }));
}

solve();
