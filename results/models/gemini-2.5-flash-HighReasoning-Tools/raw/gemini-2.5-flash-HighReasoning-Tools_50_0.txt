#version 450
layout (local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

struct Boid {
    vec4 position; // xyz + w (maybe for something else or padding)
    vec4 velocity; // xyz + w (maybe for something else or padding)
};

layout(std430, binding = 0) readonly buffer BoidInput {
    Boid boids_in[];
};

layout(std430, binding = 1) writeonly buffer BoidOutput {
    Boid boids_out[];
};

uniform uint num_boids;
uniform float delta_time; // Time step for simulation

// Radii for boid rules
uniform float separation_radius;
uniform float alignment_radius;
uniform float cohesion_radius;

// Force magnitudes for boid rules
uniform float separation_weight;
uniform float alignment_weight;
uniform float cohesion_weight;

uniform float max_speed;     // Maximum velocity a boid can have
uniform float max_steer_force; // Maximum steering force applied per step

uniform vec3 world_min; // Minimum coordinates of the simulation bounding box
uniform vec3 world_max; // Maximum coordinates of the simulation bounding box

// Helper function to calculate steering force towards a target
vec3 steer_towards(vec3 current_vel, vec3 target_pos, vec3 current_pos, float max_speed, float max_force) {
    vec3 desired_velocity = normalize(target_pos - current_pos) * max_speed;
    vec3 steering_force = desired_velocity - current_vel;
    
    // Limit the steering force
    if (length(steering_force) > max_force) {
        steering_force = normalize(steering_force) * max_force;
    }
    return steering_force;
}

void main() {
    uint boid_idx = gl_GlobalInvocationID.x;

    if (boid_idx >= num_boids) {
        return;
    }

    Boid current_boid = boids_in[boid_idx];
    vec3  pos = current_boid.position.xyz;
    vec3  vel = current_boid.velocity.xyz;

    vec3 separation_force_vec = vec3(0.0);
    vec3 alignment_sum_vel = vec3(0.0);
    vec3 cohesion_sum_pos = vec3(0.0);
    int neighbor_count = 0;

    // Iterate over all other boids to find neighbors and apply rules
    for (uint i = 0; i < num_boids; ++i) {
        if (i == boid_idx) continue; // Don't compare with self

        Boid other_boid = boids_in[i];
        vec3 other_pos = other_boid.position.xyz;
        vec3 other_vel = other_boid.velocity.xyz;

        float dist = distance(pos, other_pos);

        // Separation: Steer to avoid crowding local flockmates
        if (dist < separation_radius && dist > 0.0) {
            vec3 diff = pos - other_pos;
            // Force is inversely proportional to distance, creating stronger avoidance when closer
            separation_force_vec += normalize(diff) / dist; 
        }

        // Only consider neighbors within alignment/cohesion radius for these rules
        if (dist < alignment_radius) { // alignment_radius typically covers cohesion too
            alignment_sum_vel += other_vel;
            cohesion_sum_pos += other_pos;
            neighbor_count++;
        }
    }

    vec3 final_steering_force = vec3(0.0);

    // Apply Separation
    if (length(separation_force_vec) > 0.0) {
        final_steering_force += steer_towards(vel, pos + separation_force_vec, pos, max_speed, max_steer_force) * separation_weight;
    }

    // Apply Alignment and Cohesion if neighbors are found
    if (neighbor_count > 0) {
        // Alignment: Steer towards the average heading of local flockmates
        alignment_sum_vel /= float(neighbor_count);
        if (length(alignment_sum_vel) > 0.0) {
            final_steering_force += steer_towards(vel, pos + alignment_sum_vel, pos, max_speed, max_steer_force) * alignment_weight;
        }

        // Cohesion: Steer to move towards the average position of local flockmates
        cohesion_sum_pos /= float(neighbor_count);
        final_steering_force += steer_towards(vel, cohesion_sum_pos, pos, max_speed, max_steer_force) * cohesion_weight;
    }

    // Update velocity
    vel += final_steering_force * delta_time;

    // Limit velocity to max_speed
    if (length(vel) > max_speed) {
        vel = normalize(vel) * max_speed;
    }

    // Update position
    pos += vel * delta_time;

    // Apply world boundary conditions (wrap around)
    // For each axis, if a boid goes beyond one boundary, it reappears on the opposite side.
    pos.x = mod(pos.x - world_min.x + (world_max.x - world_min.x), (world_max.x - world_min.x)) + world_min.x;
    pos.y = mod(pos.y - world_min.y + (world_max.y - world_min.y), (world_max.y - world_min.y)) + world_min.y;
    pos.z = mod(pos.z - world_min.z + (world_max.z - world_min.z), (world_max.z - world_min.z)) + world_min.z;
    
    // Store updated boid data in the output buffer
    boids_out[boid_idx].position.xyz = pos;
    boids_out[boid_idx].velocity.xyz = vel;
    boids_out[boid_idx].position.w = 1.0; // Example padding/extra data
    boids_out[boid_idx].velocity.w = 0.0; // Example padding/extra data
}
