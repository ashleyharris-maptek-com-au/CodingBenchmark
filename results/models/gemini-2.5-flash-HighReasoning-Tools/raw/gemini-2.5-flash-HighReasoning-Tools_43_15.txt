// Vulkan/HLSL Pixel Shader

// Input structure from Vertex Shader
// Note: In Vulkan/HLSL, SV_Position is usually the only system-value semantic
// automatically mapped. Other semantics like TEXCOORD need explicit mapping.
struct PSInput
{
    float4 Position : SV_Position;
    // float2 TexCoord : TEXCOORD0; // Example for texture coordinates
    // float3 Normal : NORMAL0;     // Example for normal
};

// Output structure for the Pixel Shader
struct PSOutput
{
    float4 Color : SV_Target0; // Render target output
};

PSOutput main(PSInput input)
{
    PSOutput output;
    // Output a fixed red color for now
    output.Color = float4(1.0f, 0.0f, 0.0f, 1.0f); // Red, opaque
    return output;
}
