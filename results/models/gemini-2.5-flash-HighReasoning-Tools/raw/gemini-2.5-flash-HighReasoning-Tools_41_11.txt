; Polka dot pattern
; cell = floor(uv * 6.0)
; in_cell_uv = fract(uv * 6.0)
; center_to_point = in_cell_uv - vec2(0.5, 0.5)
; distance_from_center = length(center_to_point)

%distance_from_center = OpLength %float %center_to_point

%c_f0_3_sq = OpFMul %float %c_f0_3 %c_f0_3 ; radius 0.3, compare squared distance to avoid sqrt

%is_in_circle = OpFOrdLessThanEqual %bool %distance_from_center %c_f0_3

%polka_color = OpSelect %v4float %is_in_circle %c_v4f_yellow %c_v4f_dark_blue

; Lambertian shading
; lightDir = normalize(lightPos.xyz - worldPos)
; norm_normal = normalize(normal)
; diffuseFactor = max(0.0, dot(norm_normal, lightDir))
; finalColor = (0.3 + 0.7 * diffuseFactor) * polka_color
