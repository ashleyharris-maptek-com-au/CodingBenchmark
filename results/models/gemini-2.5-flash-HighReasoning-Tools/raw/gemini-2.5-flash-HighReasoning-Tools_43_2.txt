// A basic HLSL pixel shader for Vulkan.

// Input structure from the vertex shader
// SV_POSITION is the pixel's screen-space position, typically interpolated from vertex shader.
// COLOR is an example of a custom semantic for color interpolation.
struct PSInput
{
    float4 Position : SV_POSITION;
    float4 Color    : COLOR; // Example: interpolated color from vertex shader
};

// Output structure to the render target
// SV_TARGET0 specifies the first render target (e.g., the framebuffer).
struct PSOutput
{
    float4 Target0 : SV_TARGET0;
};

// Main entry point for the pixel shader
PSOutput main(PSInput Input)
{
    PSOutput Output;

    // Output the interpolated color directly
    Output.Target0 = Input.Color;

    // Alternatively, output a fixed color:
    // Output.Target0 = float4(1.0f, 0.0f, 0.0f, 1.0f); // Red color

    return Output;
}
