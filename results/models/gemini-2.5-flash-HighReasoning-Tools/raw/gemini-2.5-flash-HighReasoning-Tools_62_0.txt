// solution.ts

interface Coord {
    x: number;
    y: number;
}

interface Town {
    id: string;
    x: number;
    y: number;
    value: number;
}

interface InputData {
    width: number;
    height: number;
    grid: string[];
    ignitions: Coord[];
    towns: Town[];
    cutBudget: number;
}

// Function to read stdin
function readInput(): InputData {
    const fs = require('fs');
    const input: string = fs.readFileSync(0, 'utf8');
    return JSON.parse(input);
}

// Function to print output
function printOutput(firebreaks: Coord[]): void {
    console.log(JSON.stringify({ firebreaks }));
}

// Helper to convert Coord to string key
function coordToString(coord: Coord): string {
    return `${coord.x},${coord.y}`;
}

// Helper to convert string key to Coord
function stringToCoord(s: string): Coord {
    const parts = s.split(',').map(Number);
    return { x: parts[0], y: parts[1] };
}

// BFS to simulate fire spread and collect distance/predecessors
function runBFS(
    width: number,
    height: number,
    initialGrid: string[][],
    ignitions: Coord[]
): {
    distance: Map<string, number>;
    predecessors: Map<string, Set<string>>;
    initialBurnedCells: Set<string>;
} {
    const distance = new Map<string, number>();
    const predecessors = new Map<string, Set<string>>();
    const q: Coord[] = [];
    const visited = new Set<string>();
    const initialBurnedCells = new Set<string>();

    for (const ignition of ignitions) {
        if (
            ignition.y >= 0 &&
            ignition.y < height &&
            ignition.x >= 0 &&
            ignition.x < width &&
            initialGrid[ignition.y][ignition.x] === '.' &&
            !visited.has(coordToString(ignition))
        ) {
            q.push(ignition);
            visited.add(coordToString(ignition));
            distance.set(coordToString(ignition), 0);
        }
    }

    let head = 0;
    while (head < q.length) {
        const { x, y } = q[head++];
        const currentDistance = distance.get(coordToString({ x, y }))!;
        initialBurnedCells.add(coordToString({ x, y }));

        const neighbors: Coord[] = [
            { x: x - 1, y: y },
            { x: x + 1, y: y },
            { x: x, y: y - 1 },
            { x: x, y: y + 1 },
        ];

        for (const neighbor of neighbors) {
            const { x: nx, y: ny } = neighbor;
            const neighborKey = coordToString(neighbor);

            if (
                nx >= 0 &&
                nx < width &&
                ny >= 0 &&
                ny < height &&
                initialGrid[ny][nx] === '.' // Can only spread to burnable forest
            ) {
                if (!distance.has(neighborKey) || currentDistance + 1 < distance.get(neighborKey)!) {
                    distance.set(neighborKey, currentDistance + 1);
                    predecessors.set(neighborKey, new Set([coordToString({ x, y })]));
                    if (!visited.has(neighborKey)) {
                        visited.add(neighborKey);
                        q.push(neighbor);
                    }
                } else if (currentDistance + 1 === distance.get(neighborKey)!) {
                    // Another shortest path found
                    predecessors.get(neighborKey)!.add(coordToString({ x, y }));
                }
            }
        }
    }

    return { distance, predecessors, initialBurnedCells };
}

// Simulate fire spread given cut cells
function simulateFireSpreadWithCuts(
    width: number,
    height: number,
    initialGrid: string[][],
    ignitions: Coord[],
    cutCells: Set<string>
): Set<string> {
    const q: Coord[] = [];
    const visited = new Set<string>();
    const burnedCells = new Set<string>();

    for (const ignition of ignitions) {
        if (
            ignition.y >= 0 &&
            ignition.y < height &&
            ignition.x >= 0 &&
            ignition.x < width &&
            initialGrid[ignition.y][ignition.x] === '.' &&
            !cutCells.has(coordToString(ignition)) && // Cannot ignite a cut cell
            !visited.has(coordToString(ignition))
        ) {
            q.push(ignition);
            visited.add(coordToString(ignition));
        }
    }

    let head = 0;
    while (head < q.length) {
        const { x, y } = q[head++];
        burnedCells.add(coordToString({ x, y }));

        const neighbors: Coord[] = [
            { x: x - 1, y: y },
            { x: x + 1, y: y },
            { x: x, y: y - 1 },
            { x: x, y: y + 1 },
        ];

        for (const neighbor of neighbors) {
            const { x: nx, y: ny } = neighbor;
            const neighborKey = coordToString(neighbor);

            if (
                nx >= 0 &&
                nx < width &&
                ny >= 0 &&
                ny < height &&
                initialGrid[ny][nx] === '.' && // Can only spread to burnable forest
                !cutCells.has(neighborKey) && // Fire blocked by cut cells
                !visited.has(neighborKey)
            ) {
                visited.add(neighborKey);
                q.push(neighbor);
            }
        }
    }

    return burnedCells;
}

// Calculate burned town value
function calculateBurnedTownValue(burnedCells: Set<string>, towns: Town[]): number {
    let totalValue = 0;
    for (const town of towns) {
        if (burnedCells.has(coordToString(town))) {
            totalValue += town.value;
        }
    }
    return totalValue;
}

function main() {
    const data = readInput();
    const { width, height, grid: gridStrings, ignitions, towns, cutBudget } = data;

    const initialGrid: string[][] = gridStrings.map(row => row.split(''));

    // Step 1: Simulate initial burn to get baseline
    const initialBurnedCells = simulateFireSpreadWithCuts(
        width,
        height,
        initialGrid,
        ignitions,
        new Set<string>() // No cuts
    );
    const initialBurnedTownValue = calculateBurnedTownValue(initialBurnedCells, towns);

    if (cutBudget === 0) {
        printOutput([]);
        return;
    }

    if (initialBurnedTownValue === 0) {
        printOutput([]);
        return;
    }

    // Step 2: Utilize the more scalable "value flow" approach for firebreak selection.
    // First, run BFS to get distances and predecessors from ignitions.
    const { distance, predecessors } = runBFS(width, height, initialGrid, ignitions);

    // Create a 2D array for cell value contributions, initialized to 0.
    // Maps are also suitable, but 2D array might be slightly faster for dense grids.
    const cellValueContribution: number[][] = Array(height).fill(0).map(() => Array(width).fill(0));

    // List all burnable cells to sort them by distance for the backward pass.
    const allBurnableCells: { coord: Coord; dist: number }[] = [];

    for (let y = 0; y < height; y++) {
        for (let x = 0; x < width; x++) {
            if (initialGrid[y][x] === '.') {
                const cellCoord: Coord = { x, y };
                const cellKey = coordToString(cellCoord);
                const dist = distance.has(cellKey) ? distance.get(cellKey)! : Infinity;
                
                if (dist !== Infinity) {
                    allBurnableCells.push({ coord: cellCoord, dist: dist });
                }
            }
        }
    }

    // Sort burnable cells by distance in descending order for correct value propagation.
    allBurnableCells.sort((a, b) => b.dist - a.dist);

    // Initialize town values into the contribution grid
    for (const town of towns) {
        if (distance.has(coordToString(town))) { // If town is reachable by fire
            cellValueContribution[town.y][town.x] += town.value;
        }
    }

    // Backward pass to propagate value contributions from towns up to their predecessors
    for (const { coord: { x, y } } of allBurnableCells) {
        const currentCellValue = cellValueContribution[y][x];
        const cellKey = coordToString({ x, y });
        const cellPredecessors = predecessors.get(cellKey);

        if (currentCellValue > 0 && cellPredecessors && cellPredecessors.size > 0) {
            // Distribute the current cell's value contribution among its predecessors
            const valuePerPredecessor = currentCellValue / cellPredecessors.size;
            for (const predKey of cellPredecessors) {
                const { x: px, y: py } = stringToCoord(predKey);
                cellValueContribution[py][px] += valuePerPredecessor;
            }
        }
    }

    // Collect all potential firebreaks and their calculated scores (value contributions)
    const firebreakCandidates: { coord: Coord; score: number }[] = [];
    for (let y = 0; y < height; y++) {
        for (let x = 0; x < width; x++) {
            if (initialGrid[y][x] === '.') { // Only consider burnable cells as candidates
                // A cell's score is its value contribution. Higher contribution means it protects more value.
                firebreakCandidates.push({ coord: { x, y }, score: cellValueContribution[y][x] });
            }
        }
    }

    // Sort candidates by score in descending order
    firebreakCandidates.sort((a, b) => b.score - a.score);

    // Select the top 'cutBudget' firebreaks, ensuring they are unique
    const selectedFirebreaks: Coord[] = [];
    const chosenFirebreakKeys = new Set<string>();

    for (let i = 0; i < firebreakCandidates.length && selectedFirebreaks.length < cutBudget; i++) {
        const candidateCoord = firebreakCandidates[i].coord;
        const candidateKey = coordToString(candidateCoord);

        // Ensure we don't pick the same cell twice (though sorting by score should naturally handle this,
        // it's good practice for robustness in case of identical scores or different greedy strategies)
        if (!chosenFirebreakKeys.has(candidateKey)) {
            selectedFirebreaks.push(candidateCoord);
            chosenFirebreakKeys.add(candidateKey);
        }
    }

    printOutput(selectedFirebreaks);
}

main();
