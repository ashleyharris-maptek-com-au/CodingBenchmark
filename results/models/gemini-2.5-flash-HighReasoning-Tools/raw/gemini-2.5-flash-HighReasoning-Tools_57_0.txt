const fs = require('fs');

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const { width, height, grid, ignitions, towns, cutBudget } = data;

    // Initialize the grid as a 2D array for easier manipulation
    // 0: burnable forest ('.')
    // 1: water/rock ('~')
    // 2: cut firebreak (will be updated)
    const processedGrid = grid.map(row => row.split('').map(cell => cell === '.' ? 0 : 1));

    // --- Fire Spread Simulation ---
    // This function will simulate fire spread without any firebreaks.
    // It returns a 2D array indicating which cells burn.
    function simulateFireSpread(currentGrid, initialIgnitions) {
        const burned = Array(height).fill(0).map(() => Array(width).fill(false));
        const queue = [];

        // Add initial ignitions to the queue
        for (const [x, y] of initialIgnitions) {
            if (currentGrid[y][x] === 0 && !burned[y][x]) { // Only burn burnable forest
                burned[y][x] = true;
                queue.push([x, y]);
            }
        }

        const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]]; // Down, Up, Right, Left

        let head = 0;
        while (head < queue.length) {
            const [x, y] = queue[head++];

            for (const [dx, dy] of directions) {
                const nx = x + dx;
                const ny = y + dy;

                // Check boundaries and if the cell is burnable and not already burned
                if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                    currentGrid[ny][nx] === 0 && !burned[ny][nx]) {
                    burned[ny][nx] = true;
                    queue.push([nx, ny]);
                }
            }
        }
        return burned;
    }

    // --- Greedy Firebreak Placement Strategy ---
    // This is a simple greedy approach. It's not optimal but serves the purpose of a benchmark program.
    // The strategy is to identify cells that, if cut, would protect the most town value.
    // A more sophisticated approach would involve shortest path algorithms, but that's beyond this scope.

    // First, simulate fire without any firebreaks to identify which towns are initially threatened.
    const initialBurnedCells = simulateFireSpread(processedGrid, ignitions);

    let totalTownValue = 0;
    let burnedTownValueWithoutFirebreaks = 0;

    for (const town of towns) {
        totalTownValue += town.value;
        if (initialBurnedCells[town.y][town.x]) {
            burnedTownValueWithoutFirebreaks += town.value;
        }
    }

    const firebreaks = [];
    const currentFirebreaksGrid = JSON.parse(JSON.stringify(processedGrid)); // Deep copy to add firebreaks

    // A very simple greedy strategy:
    // Iterate through towns that are burning without firebreaks.
    // For each burning town, find a path to an unburned area or a firebreak that would protect it.
    // This is a simplified BFS/DFS to find a "choke point" near the town.

    // This greedy strategy will attempt to place firebreaks on the direct path between an ignition and a town.
    // This is a simplified heuristic and far from optimal.
    
    // For each town, if it burns, try to cut a path to prevent it.
    // This needs a more sophisticated pathfinding or flow algorithm to be effective.
    // For a basic benchmark implementation, let's try to cut cells immediately adjacent to a burning town
    // if that cell is also burning and is forest. This is extremely simplistic.

    // Let's refine the greedy strategy:
    // Identify potential firebreak cells. These are cells that are forest ('.') and adjacent to a burning cell.
    // Prioritize cutting cells that protect the most value. This requires re-simulating fire spread after each cut,
    // or a more advanced flow network approach.

    // Given the constraints and the need for a complete runnable program,
    // a truly optimal firebreak placement is very complex.
    // I will implement a greedy approach that aims to block the fire's direct path to high-value towns.

    // A simpler greedy approach:
    // For each potential firebreak cell (a '.' cell), calculate its "protection value".
    // The protection value could be the sum of values of all towns that would be saved if this cell were cut.
    // This would require re-running fire simulation for each potential cut, which is too slow.

    // Alternative simple greedy:
    // Find all forest cells that are adjacent to an initial ignition point.
    // If we have budget, cut them. This won't be very effective for complex scenarios.

    // Let's use a "value density" heuristic.
    // For each burnable cell, determine how many towns it "protects" by being intact.
    // If it's cut, these towns might be exposed. But we want to protect towns.

    // The current fire spread simulation is a simple BFS.
    // To implement firebreaks, we need to choose 'cutBudget' cells to change from '.' to 'x' (cut).

    // Let's consider a simpler approach for firebreak placement:
    // Iterate `cutBudget` times. In each iteration, find the single forest cell ('.')
    // that, if converted to a firebreak, would prevent the most town value from burning.
    // This means re-simulating the fire for each potential cut, which is inefficient for large grids.

    // Given the constraints for a benchmark, a simple greedy strategy could be:
    // 1. Identify all cells that would burn without any firebreaks.
    // 2. Identify towns that would burn.
    // 3. For each burning town, find a path back to an ignition point. The cells on this path are candidates for firebreaks.
    // 4. Select `cutBudget` cells from these candidates.

    // Let's go with a more practical greedy for this benchmark:
    // Prioritize cutting cells that are on the "front" of the fire and close to high-value towns.
    // This is still hard to do efficiently.

    // The prompt says: "Use algorithms that scale".
    // Re-simulating fire for every potential cut is O(cutBudget * (width * height)). This might be too slow.

    // A simpler greedy approach for this benchmark:
    // Consider all burnable cells. For each burnable cell, calculate how many *unprotected* towns it currently links to.
    // If we cut this cell, we might disconnect some of those towns from the fire.

    // This problem can be modeled as a min-cut problem on a graph, but that's beyond a quick implementation.
    // Let's do a basic BFS from ignitions to determine the "time to burn" for each cell.
    // Then, choose firebreaks that have a high "protection value" based on nearby towns.

    // Simple BFS to get distance from nearest ignition
    const dist = Array(height).fill(0).map(() => Array(width).fill(-1));
    const q = [];

    for (const [x, y] of ignitions) {
        if (processedGrid[y][x] === 0 && dist[y][x] === -1) {
            dist[y][x] = 0;
            q.push([x, y]);
        }
    }

    const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]];
    let head = 0;
    while (head < q.length) {
        const [x, y] = q[head++];

        for (const [dx, dy] of directions) {
            const nx = x + dx;
            const ny = y + dy;

            if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                processedGrid[ny][nx] === 0 && dist[ny][nx] === -1) {
                dist[ny][nx] = dist[y][x] + 1;
                q.push([nx, ny]);
            }
        }
    }

    // Now, we have `dist` which tells us how "close" a cell is to an ignition (in terms of burn time).
    // We want to cut cells that are:
    // 1. Burnable forest (processedGrid[y][x] === 0)
    // 2. Are on a path to a town that would burn
    // 3. Are not already part of a firebreak.

    // A highly simplified greedy strategy:
    // Create a list of all burnable forest cells.
    // For each cell, calculate a "priority score".
    // Priority score: Sum of values of towns that are *currently* burning and whose shortest path to an ignition *goes through this cell*.
    // This is still too complex to calculate efficiently.

    // Let's simplify drastically for the benchmark's "complete runnable program" requirement:
    // For each town that would burn without firebreaks:
    // Find *a* path from the town to an ignition.
    // Add cells on this path to a candidate list, prioritizing those closer to the ignition.
    // This is still hard.

    // The most straightforward greedy approach for a benchmark:
    // Find all 'burnable' cells that are adjacent to any 'burned' cell (from initialBurnedCells).
    // These are the "fire front" cells.
    // From these fire front cells, pick `cutBudget` cells that are closest to towns that would burn.

    const candidateFirebreakCells = [];
    for (let y = 0; y < height; y++) {
        for (let x = 0; x < width; x++) {
            if (processedGrid[y][x] === 0 && initialBurnedCells[y][x]) { // It's a burnable cell that would burn
                // Check its neighbors. If any neighbor is a town, this cell is a good candidate to cut
                // This is still not robust.

                // A better heuristic for a simple greedy:
                // Prioritize cells that are:
                // 1. Burnable forest
                // 2. Have a short `dist` from an ignition (meaning fire will reach it quickly)
                // 3. Are "critical" for fire spread to towns. (hard to calculate quickly)

                // For a simple benchmark, let's just pick cells that are currently burning AND
                // are adjacent to an unburned town (if possible) or are on a likely path to a town.
                // This is still too abstract.

                // Let's implement a very basic greedy strategy:
                // Sort all burnable cells by their `dist` value (closest to ignition first).
                // Iterate through towns that are burning. For each town, find the cell with the smallest `dist`
                // that is on a path to the town and cut it if we have budget.

                // Simplified approach: Create a priority queue of potential firebreak cells.
                // Each cell's priority is based on how many "burning paths" to towns it blocks.
                // This is essentially a flow problem.

                // For this benchmark, a truly optimal solution is out of scope for a single turn.
                // Let's pick a simplistic greedy:
                // Find all cells that would burn, and are adjacent to a town that would burn.
                // If there are such cells, add them to firebreaks.
                // If not, pick cells that are closest to ignitions AND close to towns that would burn.

                // This is a common strategy for grid-based problems when a full solution is too complex:
                // 1. Identify "critical" cells.
                // 2. Sort them by some heuristic.
                // 3. Pick the top `cutBudget` cells.

                // Critical cells: Burnable cells that, if left as forest, allow fire to reach towns.
                // Heuristic: Sum of (town_value / distance_to_town_from_this_cell_if_cut) ?

                // Let's try this:
                // For every burnable cell, calculate a "protection potential" score.
                // The score could be:
                // - High if it's on a path from an ignition to a high-value town.
                // - High if it's close to an ignition and also close to a town.

                // Let's create a score for each burnable cell.
                // Score = sum of (town.value / (distance from cell to town)) for all burning towns.
                // Distance from cell to town requires another BFS for each town, or a multi-source BFS from all towns.

                // Multi-source BFS from all towns to find distance from each cell to nearest town.
                const distToTown = Array(height).fill(0).map(() => Array(width).fill(-1));
                const qTown = [];

                for (const town of towns) {
                    if (distToTown[town.y][town.x] === -1) {
                        distToTown[town.y][town.x] = 0;
                        qTown.push([town.x, town.y]);
                    }
                }

                let headTown = 0;
                while (headTown < qTown.length) {
                    const [x, y] = qTown[headTown++];

                    for (const [dx, dy] of directions) {
                        const nx = x + dx;
                        const ny = y + dy;

                        if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                            processedGrid[ny][nx] === 0 && distToTown[ny][nx] === -1) {
                            distToTown[ny][nx] = distToTown[y][x] + 1;
                            qTown.push([nx, ny]);
                        }
                    }
                }

                // Now for each burnable cell, we have `dist` (from ignition) and `distToTown` (to nearest town).
                // A very simple greedy score for a cell (x, y):
                // Score = (towns_protected_by_this_cut_value) / (dist[y][x] + 1)
                // towns_protected_by_this_cut_value is still hard.

                // Let's try this:
                // Create a list of all burnable forest cells that would eventually burn.
                // For each such cell (x, y), assign a score based on its proximity to burning towns
                // and its proximity to the fire source.
                // Score = (sum of values of towns that would burn AND this cell is on a path to them) / (dist[y][x] + 1)

                // This is becoming too complex for a quick benchmark implementation.
                // Let's fall back to a much simpler greedy for the benchmark.

                // Simple greedy:
                // 1. Identify all towns that would burn without firebreaks.
                // 2. For each such town, find all burnable cells that are on the shortest path (BFS) from an ignition to the town.
                // 3. These cells are candidates. Prioritize cells that are on paths to higher value towns, and are closer to ignitions.

                // For a quick benchmark, let's identify the cells on the "fire front" that are near towns.
                // "Fire front" cells are burnable cells that are adjacent to a cell that has already burned.

                const potentialFirebreakCandidates = [];

                // For each town that would burn without firebreaks
                for (const town of towns) {
                    if (initialBurnedCells[town.y][town.x]) {
                        // Find a path from an ignition to this town using BFS,
                        // and add cells on this path as candidates.
                        // We need to keep track of visited cells to avoid cycles and redundant work.

                        const parentMap = new Map(); // Stores [px, py] -> [cx, cy]
                        const pathQueue = [[town.x, town.y]];
                        const pathVisited = Array(height).fill(0).map(() => Array(width).fill(false));
                        pathVisited[town.y][town.x] = true;
                        let pathHead = 0;
                        let foundPathToIgnition = false;

                        while (pathHead < pathQueue.length) {
                            const [cx, cy] = pathQueue[pathHead++];

                            // If we reached an ignition point, we found a path
                            if (ignitions.some(([ix, iy]) => ix === cx && iy === cy)) {
                                foundPathToIgnition = true;
                                break;
                            }

                            for (const [dx, dy] of directions) {
                                const nx = cx + dx;
                                const ny = cy + dy;

                                if (nx >= 0 && nx < width && ny >= 0 && ny < height &&
                                    processedGrid[ny][nx] === 0 && // Must be burnable forest
                                    initialBurnedCells[ny][nx] && // Must be a cell that would burn
                                    !pathVisited[ny][nx]) {
                                    pathVisited[ny][nx] = true;
                                    parentMap.set(`${nx},${ny}`, [cx, cy]);
                                    pathQueue.push([nx, ny]);
                                }
                            }
                        }

                        if (foundPathToIgnition) {
                            // Reconstruct the path and add cells to candidates
                            let currentPathCell = [town.x, town.y];
                            while (currentPathCell && !ignitions.some(([ix, iy]) => ix === currentPathCell[0] && iy === currentPathCell[1])) {
                                const [cx, cy] = currentPathCell;
                                // Only add unique candidate cells; the Set 'chosenFirebreaks' will handle actual uniqueness later
                                potentialFirebreakCandidates.push({ x: cx, y: cy, value: town.value });
                                currentPathCell = parentMap.get(`${cx},${cy}`); // Move to parent
                            }
                        }
                    }
                }

                // Sort candidates by some heuristic. A simple one: cells closer to ignitions are higher priority,
                // and cells protecting higher value towns.
                // For simplicity, let's just sort by town value (desc) then distance from ignition (asc).
                potentialFirebreakCandidates.sort((a, b) => {
                    if (a.value !== b.value) {
                        return b.value - a.value; // Higher value town first
                    }
                    return dist[a.y][a.x] - dist[b.y][b.x]; // Closer to ignition first
                });

                const chosenFirebreaks = new Set(); // Use a Set to avoid duplicate firebreaks

                for (const candidate of potentialFirebreakCandidates) {
                    if (firebreaks.length < cutBudget) {
                        const cellKey = `${candidate.x},${candidate.y}`;
                        if (!chosenFirebreaks.has(cellKey)) {
                            firebreaks.push([candidate.x, candidate.y]);
                            chosenFirebreaks.add(cellKey);
                            // Mark in grid (or simulate this for scoring)
                            currentFirebreaksGrid[candidate.y][candidate.x] = 2; // Mark as cut
                        }
                    } else {
                        break;
                    }
                }
            }
        }
    }

    // --- Recalculate burned towns with firebreaks ---
    const burnedWithFirebreaks = simulateFireSpread(currentFirebreaksGrid, ignitions);

    let savedTownValue = 0;
    for (const town of towns) {
        // If a town was going to burn, but now it doesn't, its value is saved.
        if (initialBurnedCells[town.y][town.x] && !burnedWithFirebreaks[town.y][town.x]) {
            savedTownValue += town.value;
        }
    }

    const output = {
        firebreaks: firebreaks
    };

    console.log(JSON.stringify(output));
}

solve();
