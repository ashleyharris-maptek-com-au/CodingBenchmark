using System;
using System.Collections.Generic;
using System.Linq;

public enum CellState
{
    Unknown,
    Revealed,
    Flagged
}

public class Cell
{
    public int Row { get; }
    public int Col { get; }
    public char Value { get; set; }
    public CellState State { get; set; }
    public int MineCount { get; set; } // Only relevant if State is Revealed and Value is '1'-'8'
    public bool InQueue { get; set; } // To track if the cell is currently in CellsToProcess queue

    // Counts of neighbors in different states. These are dynamically updated.
    public int UnknownNeighborsCount { get; set; }
    public int FlaggedNeighborsCount { get; set; }
    // SafeNeighborsCount is not directly used in the deductions as much as Unknown and Flagged.

    public Cell(int row, int col, char value, CellState state)
    {
        Row = row;
        Col = col;
        Value = value;
        State = state;
        MineCount = 0;
        InQueue = false; // Initialize InQueue to false
        if (state == CellState.Revealed && char.IsDigit(value))
        {
            MineCount = value - '0';
        }
    }

    public override string ToString()
    {
        return $"({Col},{Row}) {Value} {State}";
    }
}

public class Board
{
    public Cell[,] Cells { get; }
    public int Height { get; }
    public int Width { get; }
    public List<(int, int)> FoundSafeCells { get; }
    public Queue<Cell> CellsToProcess { get; } = new Queue<Cell>();

    public Board(int height, int width)
    {
        Height = height;
        Width = width;
        Cells = new Cell[Height, Width];
        FoundSafeCells = new List<(int, int)>();
    }

    public void ParseInput(string[] inputLines)
    {
        for (int r = 0; r < Height; r++)
        {
            string rowStr = inputLines[r + 1]; // +1 to skip H W line
            for (int c = 0; c < Width; c++)
            {
                char cellChar = rowStr[c];
                CellState state;
                if (cellChar == ' ')
                {
                    state = CellState.Unknown;
                }
                else if (cellChar == '*')
                {
                    state = CellState.Flagged; // Initial input '*' means a known mine, so it's flagged.
                }
                else // '0'-'8'
                {
                    state = CellState.Revealed;
                }
                Cells[r, c] = new Cell(r, c, cellChar, state);

                if (state == CellState.Revealed)
                {
                    CellsToProcess.Enqueue(Cells[r, c]);
                    Cells[r, c].InQueue = true;
                }
            }
        }
        
        // Initial calculation of neighbor counts for all cells
        for (int r = 0; r < Height; r++)
        {
            for (int c = 0; c < Width; c++)
            {
                UpdateNeighborCounts(Cells[r,c], initialSetup: true);
            }
        }
    }

    public IEnumerable<Cell> GetNeighbors(int row, int col)
    {
        for (int dr = -1; dr <= 1; dr++)
        {
            for (int dc = -1; dc <= 1; dc++)
            {
                if (dr == 0 && dc == 0) continue; // Skip self

                int nr = row + dr;
                int nc = col + dc;

                if (nr >= 0 && nr < Height && nc >= 0 && nc < Width)
                {
                    yield return Cells[nr, nc];
                }
            }
        }
    }

    // Helper to calculate neighbor counts for a cell.
    // initialSetup is true when called for all cells during ParseInput.
    // Otherwise, it's called when a neighbor's state changes.
    private void UpdateNeighborCounts(Cell cell, bool initialSetup = false)
    {
        int unknown = 0;
        int flagged = 0;

        foreach (var neighbor in GetNeighbors(cell.Row, cell.Col))
        {
            if (neighbor.State == CellState.Unknown)
            {
                unknown++;
            }
            else if (neighbor.State == CellState.Flagged)
            {
                flagged++;
            }
        }

        if (cell.UnknownNeighborsCount != unknown || cell.FlaggedNeighborsCount != flagged || initialSetup)
        {
            cell.UnknownNeighborsCount = unknown;
            cell.FlaggedNeighborsCount = flagged;

            // If a cell's neighbor counts change, it might lead to new deductions, so re-enqueue it.
            // Avoid adding already processed cells to the queue unless there's a new reason.
            if (!initialSetup && cell.State == CellState.Revealed && !cell.InQueue) 
            {
                CellsToProcess.Enqueue(cell);
                cell.InQueue = true;
            }
        }
    }

    public void ApplyDeductions(Cell cell)
    {
        if (cell.State != CellState.Revealed) return; // Only apply deductions from revealed numbered cells

        // Rule 1: If number of flagged neighbors equals cell's mine count, all remaining unknown neighbors are safe.
        if (cell.MineCount == cell.FlaggedNeighborsCount && cell.UnknownNeighborsCount > 0)
        {
            foreach (var neighbor in GetNeighbors(cell.Row, cell.Col))
            {
                if (neighbor.State == CellState.Unknown)
                {
                    MarkCellAsSafe(neighbor);
                }
            }
        }
        // Rule 2: If sum of flagged and unknown neighbors equals cell's mine count, all unknown neighbors are mines.
        else if (cell.MineCount == (cell.FlaggedNeighborsCount + cell.UnknownNeighborsCount) && cell.UnknownNeighborsCount > 0)
        {
            foreach (var neighbor in GetNeighbors(cell.Row, cell.Col))
            {
                if (neighbor.State == CellState.Unknown)
                {
                    MarkCellAsMine(neighbor);
                }
            }
        }
    }

    private void MarkCellAsSafe(Cell cell)
    {
        if (cell.State == CellState.Unknown)
        {
            cell.State = CellState.Revealed;
            cell.Value = '0'; // Temporarily set to '0', actual number is not known yet.
            FoundSafeCells.Add((cell.Col, cell.Row)); // Note: Output format is x y (column row)
            CellsToProcess.Enqueue(cell); // This newly revealed safe cell might have new deductions.
            cell.InQueue = true;

            // Update neighbor counts for all its neighbors.
            foreach (var neighborOfSafe in GetNeighbors(cell.Row, cell.Col))
            {
                UpdateNeighborCounts(neighborOfSafe);
            }
        }
    }

    private void MarkCellAsMine(Cell cell)
    {
        if (cell.State == CellState.Unknown)
        {
            cell.State = CellState.Flagged;
            cell.Value = '*';
            CellsToProcess.Enqueue(cell); // This newly flagged mine might have new deductions.
            cell.InQueue = true;

            // Update neighbor counts for all its neighbors.
            foreach (var neighborOfMine in GetNeighbors(cell.Row, cell.Col))
            {
                UpdateNeighborCounts(neighborOfMine);
            }
        }
    }

    public void Solve()
    {
        while (CellsToProcess.Any())
        {
            Cell currentCell = CellsToProcess.Dequeue();
            currentCell.InQueue = false; // Mark as no longer in queue
            ApplyDeductions(currentCell);
        }
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        List<string> inputLines = new List<string>();
        string? line; // Declare as nullable string
        while ((line = Console.ReadLine()) != null && line != "")
        {
            inputLines.Add(line);
        }

        if (inputLines.Count == 0) return;

        string[] dimensions = inputLines[0].Split(' ');
        int height = int.Parse(dimensions[0]);
        int width = int.Parse(dimensions[1]);

        Board board = new Board(height, width);
        board.ParseInput(inputLines.ToArray());
        board.Solve();

        Console.WriteLine(board.FoundSafeCells.Count);
        foreach (var coord in board.FoundSafeCells)
        {
            Console.WriteLine($"{coord.Item1} {coord.Item2}");
        }
    }
}
