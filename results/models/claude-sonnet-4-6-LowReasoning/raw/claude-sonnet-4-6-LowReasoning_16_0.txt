#include <bits/stdc++.h>
using namespace std;

struct Task { int machine, duration; };
struct Job { vector<Task> tasks; };

static int NUM_MACHINES = 0;

// Greedy list scheduling with a given priority comparator
// priority_key(job_idx, task_idx, jobs, jobReady, machineReady) -> lower is better
int greedySchedule(const vector<Job>& jobs,
                   vector<vector<int>>& startTimes,
                   function<double(int,int,const vector<Job>&,const vector<int>&,const vector<int>&)> priorityKey)
{
    int J = (int)jobs.size();
    vector<int> nextTask(J, 0);
    vector<int> jobReady(J, 0);
    vector<int> machineReady(NUM_MACHINES, 0);

    startTimes.assign(J, {});
    for (int j = 0; j < J; j++)
        startTimes[j].resize(jobs[j].tasks.size(), 0);

    int totalTasks = 0;
    for (auto& job : jobs) totalTasks += (int)job.tasks.size();

    // Use event-driven: maintain a ready set
    // A task is "ready" when it's the next task for its job
    // We pick the one with lowest priority key among those with minimal earliest start

    for (int iter = 0; iter < totalTasks; iter++) {
        int bestJob = -1;
        int bestStart = INT_MAX;
        double bestKey = 1e18;

        for (int j = 0; j < J; j++) {
            int ti = nextTask[j];
            if (ti >= (int)jobs[j].tasks.size()) continue;
            int m = jobs[j].tasks[ti].machine;
            int start = max(jobReady[j], machineReady[m]);
            double key = priorityKey(j, ti, jobs, jobReady, machineReady);

            if (start < bestStart || (start == bestStart && key < bestKey)) {
                bestStart = start;
                bestKey = key;
                bestJob = j;
            }
        }

        int ti = nextTask[bestJob];
        int m = jobs[bestJob].tasks[ti].machine;
        int d = jobs[bestJob].tasks[ti].duration;
        int start = max(jobReady[bestJob], machineReady[m]);

        startTimes[bestJob][ti] = start;
        jobReady[bestJob] = start + d;
        machineReady[m] = start + d;
        nextTask[bestJob]++;
    }

    int makespan = 0;
    for (int j = 0; j < J; j++) {
        int last = (int)jobs[j].tasks.size() - 1;
        if (last >= 0)
            makespan = max(makespan, startTimes[j][last] + jobs[j].tasks[last].duration);
    }
    return makespan;
}

// Fast O(N log N) greedy using priority queue - for large instances
int fastGreedy(const vector<Job>& jobs, vector<vector<int>>& startTimes) {
    int J = (int)jobs.size();
    vector<int> nextTask(J, 0);
    vector<int> jobReady(J, 0);
    vector<int> machineReady(NUM_MACHINES, 0);

    // Precompute remaining work per job
    vector<int> remainWork(J, 0);
    for (int j = 0; j < J; j++)
        for (auto& t : jobs[j].tasks)
            remainWork[j] += t.duration;

    startTimes.assign(J, {});
    for (int j = 0; j < J; j++)
        startTimes[j].resize(jobs[j].tasks.size(), 0);

    int totalTasks = 0;
    for (auto& job : jobs) totalTasks += (int)job.tasks.size();

    // Priority queue: (earliest_start, -remaining_work, job_idx)
    using T3 = tuple<int,int,int>;
    priority_queue<T3, vector<T3>, greater<T3>> pq;

    for (int j = 0; j < J; j++) {
        if (!jobs[j].tasks.empty()) {
            int m = jobs[j].tasks[0].machine;
            int start = max(jobReady[j], machineReady[m]);
            pq.push({start, -remainWork[j], j});
        }
    }

    // Track which jobs are in the PQ
    vector<bool> inPQ(J, true);

    int scheduled = 0;
    while (scheduled < totalTasks && !pq.empty()) {
        auto [est, negRW, j] = pq.top();
        pq.pop();

        int ti = nextTask[j];
        if (ti >= (int)jobs[j].tasks.size()) continue;

        int m = jobs[j].tasks[ti].machine;
        int start = max(jobReady[j], machineReady[m]);
        int d = jobs[j].tasks[ti].duration;

        // Check if this entry is stale (start time changed)
        if (start != est) {
            pq.push({start, negRW, j});
            continue;
        }

        startTimes[j][ti] = start;
        remainWork[j] -= d;
        jobReady[j] = start + d;
        machineReady[m] = start + d;
        nextTask[j]++;
        scheduled++;

        if (nextTask[j] < (int)jobs[j].tasks.size()) {
            int nm = jobs[j].tasks[nextTask[j]].machine;
            int ns = max(jobReady[j], machineReady[nm]);
            pq.push({ns, -remainWork[j], j});
        }
    }

    int makespan = 0;
    for (int j = 0; j < J; j++) {
        int last = (int)jobs[j].tasks.size() - 1;
        if (last >= 0)
            makespan = max(makespan, startTimes[j][last] + jobs[j].tasks[last].duration);
    }
    return makespan;
}

// Compute makespan from a given schedule
int computeMakespan(const vector<Job>& jobs, const vector<vector<int>>& startTimes) {
    int makespan = 0;
    for (int j = 0; j < (int)jobs.size(); j++) {
        int last = (int)jobs[j].tasks.size() - 1;
        if (last >= 0)
            makespan = max(makespan, startTimes[j][last] + jobs[j].tasks[last].duration);
    }
    return makespan;
}

// Verify schedule validity
bool isValid(const vector<Job>& jobs, const vector<vector<int>>& startTimes) {
    int J = (int)jobs.size();
    // Check job ordering
    for (int j = 0; j < J; j++) {
        for (int t = 1; t < (int)jobs[j].tasks.size(); t++) {
            int prevEnd = startTimes[j][t-1] + jobs[j].tasks[t-1].duration;
            if (startTimes[j][t] < prevEnd) return false;
        }
    }
    // Check machine conflicts
    struct Interval { int start, end, job, task; };
    vector<vector<Interval>> mIntervals(NUM_MACHINES);
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)jobs[j].tasks.size(); t++) {
            int m = jobs[j].tasks[t].machine;
            mIntervals[m].push_back({startTimes[j][t], startTimes[j][t] + jobs[j].tasks[t].duration, j, t});
        }
    }
    for (int m = 0; m < NUM_MACHINES; m++) {
        auto& iv = mIntervals[m];
        sort(iv.begin(), iv.end(), [](auto& a, auto& b){ return a.start < b.start; });
        for (int i = 1; i < (int)iv.size(); i++) {
            if (iv[i].start < iv[i-1].end) return false;
        }
    }
    return true;
}

// Local search: try to improve by rescheduling critical path
// Simple neighborhood: for each machine, try reordering two consecutive tasks
int localSearchImprove(const vector<Job>& jobs, vector<vector<int>>& bestStart, int bestMakespan, double timeLimitSec) {
    auto startClock = chrono::steady_clock::now();
    auto elapsed = [&]() {
        return chrono::duration<double>(chrono::steady_clock::now() - startClock).count();
    };

    int J = (int)jobs.size();
    // Build machine schedule lists: for each machine, sorted list of (start, job, taskIdx)
    struct MTask { int start, end, job, taskIdx; };

    int improved = 0;
    bool anyImprovement = true;

    while (anyImprovement && elapsed() < timeLimitSec) {
        anyImprovement = false;

        // Rebuild machine task lists
        vector<vector<MTask>> machineTasks(NUM_MACHINES);
        for (int j = 0; j < J; j++) {
            for (int t = 0; t < (int)jobs[j].tasks.size(); t++) {
                int m = jobs[j].tasks[t].machine;
                machineTasks[m].push_back({bestStart[j][t], bestStart[j][t] + jobs[j].tasks[t].duration, j, t});
            }
        }
        for (int m = 0; m < NUM_MACHINES; m++) {
            sort(machineTasks[m].begin(), machineTasks[m].end(), [](auto& a, auto& b){ return a.start < b.start; });
        }

        // Try swapping adjacent tasks on each machine
        for (int m = 0; m < NUM_MACHINES && elapsed() < timeLimitSec; m++) {
            auto& mt = machineTasks[m];
            for (int i = 0; i + 1 < (int)mt.size() && elapsed() < timeLimitSec; i++) {
                int j1 = mt[i].job, t1 = mt[i].taskIdx;
                int j2 = mt[i+1].job, t2 = mt[i+1].taskIdx;
                if (j1 == j2) continue; // same job, can't swap

                // Try placing j2's task before j1's task on machine m
                // This is complex - just rebuild from scratch with swapped order
                // For simplicity, do a quick feasibility check
                int d1 = jobs[j1].tasks[t1].duration;
                int d2 = jobs[j2].tasks[t2].duration;

                // j1 task: needs jobReady[j1] (from prev task in j1)
                int j1PrevEnd = (t1 > 0) ? bestStart[j1][t1-1] + jobs[j1].tasks[t1-1].duration : 0;
                // j2 task: needs jobReady[j2]
                int j2PrevEnd = (t2 > 0) ? bestStart[j2][t2-1] + jobs[j2].tasks[t2-1].duration : 0;

                // Machine availability before these two tasks
                int mPrevEnd = (i > 0) ? mt[i-1].end : 0;

                // Current order: j1 then j2
                // s1 = max(mPrevEnd, j1PrevEnd)
                // s2 = max(s1+d1, j2PrevEnd)
                int cur_s1 = max(mPrevEnd, j1PrevEnd);
                int cur_s2 = max(cur_s1 + d1, j2PrevEnd);
                int cur_end = cur_s2 + d2;

                // j1 next task constraint
                int j1NextStart = (t1 + 1 < (int)jobs[j1].tasks.size()) ? bestStart[j1][t1+1] : INT_MAX;
                int j2NextStart = (t2 + 1 < (int)jobs[j2].tasks.size()) ? bestStart[j2][t2+1] : INT_MAX;

                // Swapped order: j2 then j1
                int sw_s2 = max(mPrevEnd, j2PrevEnd);
                int sw_s1 = max(sw_s2 + d2, j1PrevEnd);
                int sw_end = sw_s1 + d1;

                // Check if swap is feasible and better
                // j1's next task must start >= sw_s1 + d1
                // j2's next task must start >= sw_s2 + d2
                if (sw_s1 + d1 <= j1NextStart && sw_s2 + d2 <= j2NextStart) {
                    if (sw_end < cur_end) {
                        // Apply swap
                        bestStart[j1][t1] = sw_s1;
                        bestStart[j2][t2] = sw_s2;
                        anyImprovement = true;
                        improved++;
                        // Recompute makespan
                        bestMakespan = computeMakespan(jobs, bestStart);
                    }
                }
            }
        }
    }

    return bestMakespan;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;
    NUM_MACHINES = M;

    vector<Job> jobs(J);
    int totalTasks = 0;
    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].tasks.resize(T);
        for (int t = 0; t < T; t++) {
            cin >> jobs[j].tasks[t].machine >> jobs[j].tasks[t].duration;
        }
        totalTasks += T;
    }

    // Precompute remaining work per job
    vector<int> totalWork(J, 0);
    for (int j = 0; j < J; j++)
        for (auto& t : jobs[j].tasks)
            totalWork[j] += t.duration;

    vector<vector<int>> bestStart;
    int bestMakespan = INT_MAX;

    auto trySchedule = [&](auto priorityKey) {
        vector<vector<int>> st;
        int ms = greedySchedule(jobs, st, priorityKey);
        if (ms < bestMakespan) {
            bestMakespan = ms;
            bestStart = st;
        }
    };

    if (totalTasks <= 5000) {
        // Try multiple priority rules

        // Rule 1: Earliest start time, tiebreak by most remaining work (MWR)
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            int start = max(jr[j], mr[js[j].tasks[ti].machine]);
            return start * 1e9 - totalWork[j]; // lower start wins, tiebreak MWR
        });

        // Rule 2: Most work remaining first
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            return -(double)totalWork[j];
        });

        // Rule 3: Shortest processing time
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            return (double)js[j].tasks[ti].duration;
        });

        // Rule 4: Longest processing time
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            return -(double)js[j].tasks[ti].duration;
        });

        // Rule 5: Fewest remaining tasks
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            return (double)(js[j].tasks.size() - ti);
        });

        // Rule 6: Critical ratio - remaining work / remaining time budget
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            int start = max(jr[j], mr[js[j].tasks[ti].machine]);
            int remWork = 0;
            for (int k = ti; k < (int)js[j].tasks.size(); k++)
                remWork += js[j].tasks[k].duration;
            return start - remWork;
        });

        // Local search improvement with time limit
        double timeLimit = (totalTasks <= 500) ? 2.0 : (totalTasks <= 2000) ? 1.0 : 0.3;
        bestMakespan = localSearchImprove(jobs, bestStart, bestMakespan, timeLimit);

    } else {
        // Large instance: use fast greedy with priority queue
        vector<vector<int>> st;
        int ms = fastGreedy(jobs, st);
        bestMakespan = ms;
        bestStart = st;

        // Also try MWR greedy
        trySchedule([&](int j, int ti, const vector<Job>& js, const vector<int>& jr, const vector<int>& mr) -> double {
            return -(double)totalWork[j];
        });

        // Quick local search
        bestMakespan = localSearchImprove(jobs, bestStart, bestMakespan, 0.5);
    }

    // Output
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)jobs[j].tasks.size(); t++) {
            if (t > 0) cout << ' ';
            cout << bestStart[j][t] << ' ' << jobs[j].tasks[t].machine;
        }
        cout << '\n';
    }

    return 0;
}
