#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <queue>
#include <algorithm>
#include <climits>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const {
        size_t h = v.size();
        for (int x : v)
            h ^= (size_t)(x + 0x9e3779b9) + 0x9e3779b9 + (h << 6) + (h >> 2);
        return h;
    }
};

using PosSet = unordered_set<vector<int>, VecHash>;

int D, max_turns;
vector<int> bounds;
deque<vector<int>> snake;
int snake_length;
PosSet food_set, obstacles, body_set;

bool in_bounds(const vector<int>& p) {
    for (int i = 0; i < D; i++)
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    return true;
}

int manhattan(const vector<int>& a, const vector<int>& b) {
    int d = 0;
    for (int i = 0; i < D; i++) d += abs(a[i] - b[i]);
    return d;
}

// BFS to find a path to any food item, returns first move (axis, dir)
pair<int,int> bfs_move(const vector<int>& start, int max_nodes) {
    if (food_set.empty()) return {-1, 0};

    struct Node {
        vector<int> pos;
        int first_axis, first_dir;
    };

    PosSet visited;
    queue<Node> q;
    visited.insert(start);

    for (int axis = 0; axis < D; axis++) {
        for (int dir : {-1, 1}) {
            vector<int> next = start;
            next[axis] += dir;
            if (!in_bounds(next)) continue;
            if (obstacles.count(next)) continue;
            if (body_set.count(next)) continue;
            if (visited.count(next)) continue;
            visited.insert(next);
            if (food_set.count(next)) return {axis, dir};
            q.push({next, axis, dir});
        }
    }

    int nodes = 0;
    while (!q.empty() && nodes < max_nodes) {
        auto [pos, fa, fd] = q.front();
        q.pop();
        nodes++;

        for (int axis = 0; axis < D; axis++) {
            for (int dir : {-1, 1}) {
                vector<int> next = pos;
                next[axis] += dir;
                if (!in_bounds(next)) continue;
                if (obstacles.count(next)) continue;
                if (body_set.count(next)) continue;
                if (visited.count(next)) continue;
                visited.insert(next);
                if (food_set.count(next)) return {fa, fd};
                q.push({next, fa, fd});
            }
        }
    }

    return {-1, 0};
}

// Greedy: move along axis that reduces Manhattan distance to nearest food
pair<int,int> greedy_move(const vector<int>& head) {
    const vector<int>* nearest = nullptr;
    int best_dist = INT_MAX;
    for (auto& f : food_set) {
        int d = manhattan(head, f);
        if (d < best_dist) { best_dist = d; nearest = &f; }
    }
    if (!nearest) return {-1, 0};

    vector<pair<int,pair<int,int>>> moves;
    for (int axis = 0; axis < D; axis++) {
        for (int dir : {-1, 1}) {
            vector<int> next = head;
            next[axis] += dir;
            if (!in_bounds(next)) continue;
            if (obstacles.count(next)) continue;
            if (body_set.count(next)) continue;
            moves.push_back({manhattan(next, *nearest), {axis, dir}});
        }
    }

    if (moves.empty()) return {-1, 0};
    sort(moves.begin(), moves.end());
    return moves[0].second;
}

// Survival: pick any safe move
pair<int,int> safe_move(const vector<int>& head) {
    for (int axis = 0; axis < D; axis++) {
        for (int dir : {-1, 1}) {
            vector<int> next = head;
            next[axis] += dir;
            if (!in_bounds(next)) continue;
            if (obstacles.count(next)) continue;
            if (body_set.count(next)) continue;
            return {axis, dir};
        }
    }
    return {0, 1};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> D >> max_turns;
    bounds.resize(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    cin >> snake_length;
    vector<int> head(D);
    for (int i = 0; i < D; i++) cin >> head[i];
    snake.push_back(head);
    body_set.insert(head);

    int F;
    cin >> F;
    for (int i = 0; i < F; i++) {
        vector<int> f(D);
        for (int j = 0; j < D; j++) cin >> f[j];
        food_set.insert(f);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; i++) {
        vector<int> o(D);
        for (int j = 0; j < D; j++) cin >> o[j];
        obstacles.insert(o);
    }

    // Cap BFS nodes: generous for small grids, limited for huge ones
    long long total_cells = 1;
    for (int b : bounds) {
        if (total_cells > 2000000LL) { total_cells = 2000000LL; break; }
        total_cells *= b;
    }
    int max_nodes = (int)min(total_cells, (long long)200000);

    for (int turn = 0; turn < max_turns; turn++) {
        if (food_set.empty()) {
            // No food left - just output a safe move to keep alive
            auto [sa, sd] = safe_move(snake.front());
            cout << sa << " " << sd << "\n";
            cout.flush();
            // update state
            vector<int> nh = snake.front();
            nh[sa] += sd;
            snake.push_front(nh);
            body_set.insert(nh);
            while ((int)snake.size() > snake_length) {
                body_set.erase(snake.back());
                snake.pop_back();
            }
            continue;
        }

        auto& cur_head = snake.front();

        auto [axis, dir] = bfs_move(cur_head, max_nodes);

        if (axis == -1) {
            auto [ga, gd] = greedy_move(cur_head);
            if (ga != -1) { axis = ga; dir = gd; }
            else {
                auto [sa, sd] = safe_move(cur_head);
                axis = sa; dir = sd;
            }
        }

        cout << axis << " " << dir << "\n";
        cout.flush();

        vector<int> new_head = cur_head;
        new_head[axis] += dir;

        bool ate_food = food_set.count(new_head) > 0;
        if (ate_food) {
            food_set.erase(new_head);
            snake_length++;
        }

        snake.push_front(new_head);
        body_set.insert(new_head);

        while ((int)snake.size() > snake_length) {
            body_set.erase(snake.back());
            snake.pop_back();
        }
    }

    return 0;
}
