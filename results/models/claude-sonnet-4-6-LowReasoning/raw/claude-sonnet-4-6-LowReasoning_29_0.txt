using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Numerics;
using System.Threading;
using System.Threading.Tasks;
using System.Diagnostics;

class Program
{
    static int N, W;
    static ulong[][] G;   // bitset adjacency
    static int[][] GL;    // adjacency lists
    static bool useBits;

    static volatile int[] best = Array.Empty<int>();
    static volatile int bestN = 0;
    static readonly object lk = new object();

    static readonly Stopwatch sw = Stopwatch.StartNew();
    const long TL = 270_000L;
    static bool TO => sw.ElapsedMilliseconds >= TL;

    static void UpdBest(int[] c)
    {
        if (c.Length > bestN)
            lock (lk) { if (c.Length > bestN) { bestN = c.Length; best = c; } }
    }

    static void Main()
    {
        // Fast buffered input
        var sb = new FastReader();
        N = sb.NextInt(); int M = sb.NextInt();

        var adjS = new HashSet<int>[N];
        for (int i = 0; i < N; i++) adjS[i] = new HashSet<int>();

        for (int e = 0; e < M; e++)
        {
            int u = sb.NextInt(), v = sb.NextInt();
            if (u != v) { adjS[u].Add(v); adjS[v].Add(u); }
        }

        GL = new int[N][];
        for (int i = 0; i < N; i++) GL[i] = adjS[i].ToArray();
        adjS = null;

        // Use bitset adjacency if memory allows (~250 MB limit)
        long bitsMemory = (long)N * ((N + 63) / 64) * 8;
        useBits = bitsMemory <= 250_000_000L;

        if (useBits)
        {
            W = (N + 63) / 64;
            G = new ulong[N][];
            for (int i = 0; i < N; i++)
            {
                G[i] = new ulong[W];
                foreach (int j in GL[i]) G[i][j >> 6] |= 1UL << (j & 63);
            }
        }

        // Order vertices by degree descending
        int[] ord = Enumerable.Range(0, N).OrderByDescending(v => GL[v].Length).ToArray();

        // Greedy heuristic first
        Greedy(ord);

        // Exact search if time permits
        if (!TO)
        {
            if (useBits) ExactB(ord);
            else ExactL(ord);
        }

        Console.WriteLine(bestN);
        Console.WriteLine(string.Join(" ", best));
    }

    // Greedy clique: pick vertex with most connections to remaining candidates
    static void Greedy(int[] ord)
    {
        int tries = Math.Min(N, 500);
        Parallel.For(0, tries, new ParallelOptions { MaxDegreeOfParallelism = Environment.ProcessorCount }, i =>
        {
            if (TO) return;
            GreedyFrom(ord[i]);
        });
    }

    static void GreedyFrom(int start)
    {
        var clique = new List<int> { start };
        var cand = new HashSet<int>(GL[start]);

        while (cand.Count > 0)
        {
            int bv = -1, bd = -1;
            if (cand.Count <= 5000)
            {
                // exact: pick vertex with most neighbors in cand
                foreach (int v in cand)
                {
                    int d = 0;
                    foreach (int u in GL[v]) if (cand.Contains(u)) d++;
                    if (d > bd) { bd = d; bv = v; }
                }
            }
            else
            {
                // fast approximation: max degree in original graph within cand
                foreach (int v in cand)
                {
                    int d = GL[v].Length;
                    if (d > bd) { bd = d; bv = v; }
                }
            }
            clique.Add(bv);
            var nc = new HashSet<int>();
            foreach (int u in GL[bv]) if (cand.Contains(u)) nc.Add(u);
            cand = nc;
        }
        UpdBest(clique.ToArray());
    }

    // ---- Bitset helpers ----
    static ulong[] BAnd(ulong[] a, ulong[] b)
    {
        var r = new ulong[W];
        for (int i = 0; i < W; i++) r[i] = a[i] & b[i];
        return r;
    }
    static bool BEmp(ulong[] a) { for (int i = 0; i < W; i++) if (a[i] != 0) return false; return true; }
    static int BCnt(ulong[] a) { int c = 0; for (int i = 0; i < W; i++) c += BitOperations.PopCount(a[i]); return c; }
    static int BFst(ulong[] a) { for (int i = 0; i < W; i++) if (a[i] != 0) return i * 64 + BitOperations.TrailingZeroCount(a[i]); return -1; }
    static void BClr(ulong[] a, int v) => a[v >> 6] &= ~(1UL << (v & 63));
    static void BSet(ulong[] a, int v) => a[v >> 6] |= 1UL << (v & 63);

    // ---- Exact search with bitsets ----
    static void ExactB(int[] ord)
    {
        int[] pos = new int[N];
        for (int i = 0; i < N; i++) pos[ord[i]] = i;

        int threads = Environment.ProcessorCount;
        var sem = new SemaphoreSlim(threads);
        var tasks = new List<Task>();

        for (int i = 0; i < N && !TO; i++)
        {
            int v = ord[i];
            var P = new ulong[W]; var X = new ulong[W];
            foreach (int u in GL[v])
            {
                if (pos[u] > i) BSet(P, u);
                else BSet(X, u);
            }

            // Early prune at top level
            if (1 + BCnt(P) <= bestN)
            {
                // Still need to correctly track X - skip but X isn't used here
                continue;
            }

            sem.Wait();
            var lP = P; var lX = X;
            int vv = v;
            tasks.Add(Task.Run(() =>
            {
                try { BKB(new List<int> { vv }, lP, lX); }
                finally { sem.Release(); }
            }));
        }
        Task.WaitAll(tasks.ToArray());
    }

    static void BKB(List<int> R, ulong[] P, ulong[] X)
    {
        if (TO) return;
        if (BEmp(P))
        {
            if (BEmp(X)) UpdBest(R.ToArray());
            return;
        }
        if (R.Count + BCnt(P) <= bestN) return;

        // Choose pivot: vertex in P∪X maximizing |N(u) ∩ P|
        var PX = new ulong[W]; for (int i = 0; i < W; i++) PX[i] = P[i] | X[i];
        int piv = -1, pm = -1;
        var tmp = (ulong[])PX.Clone();
        while (true)
        {
            int u = BFst(tmp); if (u < 0) break;
            BClr(tmp, u);
            int c = BCnt(BAnd(G[u], P));
            if (c > pm) { pm = c; piv = u; }
        }

        // Candidates = P \ N(pivot)
        var cands = (ulong[])P.Clone();
        if (piv >= 0) for (int i = 0; i < W; i++) cands[i] &= ~G[piv][i];

        var lP = (ulong[])P.Clone();
        var lX = (ulong[])X.Clone();

        while (!TO)
        {
            int v = BFst(cands); if (v < 0) break;
            BClr(cands, v);
            if (R.Count + BCnt(lP) <= bestN) break;

            var nP = BAnd(lP, G[v]);
            var nX = BAnd(lX, G[v]);
            R.Add(v);
            BKB(R, nP, nX);
            R.RemoveAt(R.Count - 1);
            BClr(lP, v);
            BSet(lX, v);
        }
    }

    // ---- Exact search with HashSets (large graphs) ----
    static void ExactL(int[] ord)
    {
        int[] pos = new int[N];
        for (int i = 0; i < N; i++) pos[ord[i]] = i;

        for (int i = 0; i < N && !TO; i++)
        {
            int v = ord[i];
            var P = new HashSet<int>(); var X = new HashSet<int>();
            foreach (int u in GL[v])
            {
                if (pos[u] > i) P.Add(u);
                else X.Add(u);
            }
            if (1 + P.Count <= bestN) continue;
            BKL(new List<int> { v }, P, X);
        }
    }

    static void BKL(List<int> R, HashSet<int> P, HashSet<int> X)
    {
        if (TO) return;
        if (P.Count == 0) { if (X.Count == 0) UpdBest(R.ToArray()); return; }
        if (R.Count + P.Count <= bestN) return;

        // Pivot: max |N(u) ∩ P|
        int piv = -1, pm = -1;
        foreach (int u in P)
        {
            int c = 0; foreach (int w in GL[u]) if (P.Contains(w)) c++;
            if (c > pm) { pm = c; piv = u; }
        }
        foreach (int u in X)
        {
            int c = 0; foreach (int w in GL[u]) if (P.Contains(w)) c++;
            if (c > pm) { pm = c; piv = u; }
        }

        var pivN = piv >= 0 ? new HashSet<int>(GL[piv]) : null;
        var cands = P.Where(v => pivN == null || !pivN.Contains(v)).ToList();

        foreach (int v in cands)
        {
            if (TO) return;
            if (R.Count + P.Count <= bestN) break;

            var vN = new HashSet<int>(GL[v]);
            var nP = new HashSet<int>(); foreach (int w in P) if (vN.Contains(w)) nP.Add(w);
            var nX = new HashSet<int>(); foreach (int w in X) if (vN.Contains(w)) nX.Add(w);

            R.Add(v);
            BKL(R, nP, nX);
            R.RemoveAt(R.Count - 1);
            P.Remove(v); X.Add(v);
        }
    }

    // Fast input reader
    class FastReader
    {
        const int BUFSIZE = 1 << 20;
        readonly Stream _s;
        readonly byte[] _buf = new byte[1 << 20];
        int _pos, _len;

        public FastReader() { _s = Console.OpenStandardInput(); }

        int ReadByte()
        {
            if (_pos >= _len) { _len = _s.Read(_buf, 0, BUFSIZE); _pos = 0; }
            return _pos < _len ? _buf[_pos++] : -1;
        }

        public int NextInt()
        {
            int c; while ((c = ReadByte()) <= ' ') if (c < 0) return 0;
            bool neg = c == '-'; if (neg) c = ReadByte();
            int v = 0;
            while (c >= '0') { v = v * 10 + c - '0'; c = ReadByte(); }
            return neg ? -v : v;
        }
    }
}
