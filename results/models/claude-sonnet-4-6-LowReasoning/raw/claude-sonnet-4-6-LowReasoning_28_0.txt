use std::io::{self, Read, Write, BufWriter};
use std::collections::VecDeque;
use std::time::Instant;

fn main() {
    let start = Instant::now();
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut iter = input.split_ascii_whitespace();
    macro_rules! nxt {
        ($t:ty) => {
            iter.next().unwrap().parse::<$t>().unwrap()
        };
    }

    let n = nxt!(usize);
    let m = nxt!(usize);

    let mut adj: Vec<Vec<u32>> = vec![vec![]; n];
    for _ in 0..m {
        let u = nxt!(usize);
        let v = nxt!(usize);
        if u != v {
            adj[u].push(v as u32);
            adj[v].push(u as u32);
        }
    }

    let cover = solve(&adj, n, start);

    writeln!(out, "{}", cover.len()).unwrap();
    if cover.is_empty() {
        writeln!(out).unwrap();
    } else {
        let parts: Vec<_> = cover.iter().map(|x| x.to_string()).collect();
        writeln!(out, "{}", parts.join(" ")).unwrap();
    }
}

fn solve(adj: &[Vec<u32>], n: usize, start: Instant) -> Vec<usize> {
    let mut in_cover = vec![false; n];
    let mut active = vec![true; n];
    let mut deg: Vec<u32> = adj.iter().map(|a| a.len() as u32).collect();

    // Remove isolated vertices
    for v in 0..n {
        if deg[v] == 0 {
            active[v] = false;
        }
    }

    // Repeated degree-1 reduction
    let mut queue: VecDeque<usize> = (0..n).filter(|&v| active[v] && deg[v] == 1).collect();
    apply_degree1_rule(adj, &mut active, &mut deg, &mut in_cover, &mut queue);

    let remaining_count = (0..n).filter(|&v| active[v]).count();

    if remaining_count == 0 {
        return (0..n).filter(|&v| in_cover[v]).collect();
    }

    if remaining_count > 400_000 || start.elapsed().as_secs_f64() > 10.0 {
        two_approx_on_active(adj, &active, &mut in_cover, n);
        return (0..n).filter(|&v| in_cover[v]).collect();
    }

    // Find connected components and solve each
    let mut visited = vec![false; n];
    for start_v in 0..n {
        if !active[start_v] || visited[start_v] {
            continue;
        }

        let mut comp = Vec::new();
        let mut bfs = VecDeque::new();
        bfs.push_back(start_v);
        visited[start_v] = true;
        comp.push(start_v);
        while let Some(v) = bfs.pop_front() {
            for &u in &adj[v] {
                let u = u as usize;
                if active[u] && !visited[u] {
                    visited[u] = true;
                    comp.push(u);
                    bfs.push_back(u);
                }
            }
        }

        // Bipartite coloring
        let mut color = vec![-1i8; n];
        color[start_v] = 0;
        let mut bfs2 = VecDeque::new();
        bfs2.push_back(start_v);
        let mut is_bipartite = true;
        'color_loop: while let Some(v) = bfs2.pop_front() {
            for &u in &adj[v] {
                let u = u as usize;
                if !active[u] {
                    continue;
                }
                if color[u] == -1 {
                    color[u] = 1 - color[v];
                    bfs2.push_back(u);
                } else if color[u] == color[v] {
                    is_bipartite = false;
                    break 'color_loop;
                }
            }
        }

        if is_bipartite {
            let cover_verts = solve_bipartite(adj, &comp, &color, n);
            for v in cover_verts {
                in_cover[v] = true;
            }
        } else if comp.len() <= 80 {
            let cover_verts = branch_and_bound(adj, &active, &comp, n);
            for v in cover_verts {
                in_cover[v] = true;
            }
        } else {
            two_approx_component(adj, &active, &comp, &mut in_cover);
        }
    }

    (0..n).filter(|&v| in_cover[v]).collect()
}

fn apply_degree1_rule(
    adj: &[Vec<u32>],
    active: &mut Vec<bool>,
    deg: &mut Vec<u32>,
    in_cover: &mut Vec<bool>,
    queue: &mut VecDeque<usize>,
) {
    while let Some(v) = queue.pop_front() {
        if !active[v] || deg[v] != 1 {
            continue;
        }
        let mut neighbor = None;
        for &u in &adj[v] {
            if active[u as usize] {
                neighbor = Some(u as usize);
                break;
            }
        }
        let u = match neighbor {
            Some(x) => x,
            None => {
                active[v] = false;
                continue;
            }
        };
        in_cover[u] = true;
        active[u] = false;
        active[v] = false;
        for &w in &adj[u] {
            let w = w as usize;
            if active[w] {
                deg[w] = deg[w].saturating_sub(1);
                if deg[w] == 1 {
                    queue.push_back(w);
                } else if deg[w] == 0 {
                    active[w] = false;
                }
            }
        }
    }
}

fn two_approx_on_active(adj: &[Vec<u32>], active: &[bool], in_cover: &mut Vec<bool>, n: usize) {
    let mut matched = vec![false; n];
    for u in 0..n {
        if !active[u] || matched[u] {
            continue;
        }
        for &v in &adj[u] {
            let v = v as usize;
            if active[v] && !matched[v] {
                in_cover[u] = true;
                in_cover[v] = true;
                matched[u] = true;
                matched[v] = true;
                break;
            }
        }
    }
}

fn two_approx_component(adj: &[Vec<u32>], active: &[bool], comp: &[usize], in_cover: &mut Vec<bool>) {
    let mut matched = vec![false; adj.len()];
    for &u in comp {
        if !active[u] || matched[u] {
            continue;
        }
        for &v in &adj[u] {
            let v = v as usize;
            if active[v] && !matched[v] {
                in_cover[u] = true;
                in_cover[v] = true;
                matched[u] = true;
                matched[v] = true;
                break;
            }
        }
    }
}

fn solve_bipartite(adj: &[Vec<u32>], comp: &[usize], color: &[i8], n: usize) -> Vec<usize> {
    let left: Vec<usize> = comp.iter().filter(|&&v| color[v] == 0).copied().collect();
    let right: Vec<usize> = comp.iter().filter(|&&v| color[v] == 1).copied().collect();

    let nl = left.len();
    let nr = right.len();

    let mut left_idx = vec![u32::MAX; n];
    let mut right_idx = vec![u32::MAX; n];
    for (i, &v) in left.iter().enumerate() {
        left_idx[v] = i as u32;
    }
    for (i, &v) in right.iter().enumerate() {
        right_idx[v] = i as u32;
    }

    let mut adj_lr: Vec<Vec<u32>> = vec![vec![]; nl];
    for (li, &v) in left.iter().enumerate() {
        for &u in &adj[v] {
            let ri = right_idx[u as usize];
            if ri != u32::MAX {
                adj_lr[li].push(ri);
            }
        }
    }

    let mut match_l = vec![u32::MAX; nl];
    let mut match_r = vec![u32::MAX; nr];
    let mut dist = vec![u32::MAX; nl];

    loop {
        let mut q: VecDeque<usize> = VecDeque::new();
        for l in 0..nl {
            if match_l[l] == u32::MAX {
                dist[l] = 0;
                q.push_back(l);
            } else {
                dist[l] = u32::MAX;
            }
        }
        let mut found_free = false;
        while let Some(l) = q.pop_front() {
            for &r in &adj_lr[l] {
                let r = r as usize;
                let ml = match_r[r];
                if ml == u32::MAX {
                    found_free = true;
                } else {
                    let ml = ml as usize;
                    if dist[ml] == u32::MAX {
                        dist[ml] = dist[l] + 1;
                        q.push_back(ml);
                    }
                }
            }
        }
        if !found_free {
            break;
        }
        for l in 0..nl {
            if match_l[l] == u32::MAX {
                hk_dfs(l, &adj_lr, &mut match_l, &mut match_r, &mut dist);
            }
        }
    }

    // König's theorem
    let mut reach_l = vec![false; nl];
    let mut reach_r = vec![false; nr];
    let mut q: VecDeque<usize> = VecDeque::new();
    for l in 0..nl {
        if match_l[l] == u32::MAX {
            reach_l[l] = true;
            q.push_back(l);
        }
    }
    while let Some(l) = q.pop_front() {
        for &r in &adj_lr[l] {
            let r = r as usize;
            if !reach_r[r] {
                reach_r[r] = true;
                let ml = match_r[r];
                if ml != u32::MAX && !reach_l[ml as usize] {
                    reach_l[ml as usize] = true;
                    q.push_back(ml as usize);
                }
            }
        }
    }

    let mut cover = Vec::new();
    for (li, &v) in left.iter().enumerate() {
        if !reach_l[li] {
            cover.push(v);
        }
    }
    for (ri, &v) in right.iter().enumerate() {
        if reach_r[ri] {
            cover.push(v);
        }
    }
    cover
}

fn hk_dfs(
    l: usize,
    adj_lr: &[Vec<u32>],
    match_l: &mut Vec<u32>,
    match_r: &mut Vec<u32>,
    dist: &mut Vec<u32>,
) -> bool {
    for i in 0..adj_lr[l].len() {
        let r = adj_lr[l][i] as usize;
        let ml = match_r[r];
        if ml == u32::MAX
            || (dist[ml as usize] == dist[l] + 1
                && hk_dfs(ml as usize, adj_lr, match_l, match_r, dist))
        {
            match_l[l] = r as u32;
            match_r[r] = l as u32;
            return true;
        }
    }
    dist[l] = u32::MAX;
    false
}

fn branch_and_bound(adj: &[Vec<u32>], active: &[bool], comp: &[usize], n: usize) -> Vec<usize> {
    let cn = comp.len();
    let mut local_id = vec![u32::MAX; n];
    for (i, &v) in comp.iter().enumerate() {
        local_id[v] = i as u32;
    }

    let mut local_adj: Vec<Vec<u32>> = vec![vec![]; cn];
    for (i, &v) in comp.iter().enumerate() {
        for &u in &adj[v] {
            if active[u as usize] {
                let li = local_id[u as usize];
                if li != u32::MAX {
                    local_adj[i].push(li);
                }
            }
        }
    }

    // 2-approx for initial upper bound
    let mut best = vec![false; cn];
    let mut best_size: usize = 0;
    {
        let mut matched = vec![false; cn];
        for u in 0..cn {
            if matched[u] {
                continue;
            }
            for &v in &local_adj[u] {
                let v = v as usize;
                if !matched[v] {
                    best[u] = true;
                    best[v] = true;
                    matched[u] = true;
                    matched[v] = true;
                    break;
                }
            }
        }
        best_size = best.iter().filter(|&&x| x).count();
    }

    let mut chosen = vec![false; cn];
    bb_recurse(&local_adj, &mut chosen, 0, &mut best, &mut best_size, cn);

    best.iter()
        .enumerate()
        .filter(|&(_, &b)| b)
        .map(|(i, _)| comp[i])
        .collect()
}

fn bb_recurse(
    adj: &[Vec<u32>],
    chosen: &mut Vec<bool>,
    current_size: usize,
    best: &mut Vec<bool>,
    best_size: &mut usize,
    n: usize,
) {
    if current_size >= *best_size {
        return;
    }

    // Find an uncovered edge (neither endpoint in chosen)
    let mut branch_edge: Option<(usize, usize)> = None;
    'outer: for u in 0..n {
        if chosen[u] {
            continue;
        }
        for &v in &adj[u] {
            let v = v as usize;
            if !chosen[v] {
                branch_edge = Some((u, v));
                break 'outer;
            }
        }
    }

    let (u, v) = match branch_edge {
        None => {
            // Valid cover found
            *best_size = current_size;
            best.copy_from_slice(chosen);
            return;
        }
        Some(e) => e,
    };

    // Branch 1: take u
    chosen[u] = true;
    bb_recurse(adj, chosen, current_size + 1, best, best_size, n);
    chosen[u] = false;

    // Branch 2: take v
    chosen[v] = true;
    bb_recurse(adj, chosen, current_size + 1, best, best_size, n);
    chosen[v] = false;
}
