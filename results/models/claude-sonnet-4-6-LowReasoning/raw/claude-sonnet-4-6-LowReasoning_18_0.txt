#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <numeric>
#include <unordered_map>
#include <cstdint>

using namespace std;
typedef long long ll;
typedef pair<ll, int> pli;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    ll N, M;
    cin >> N >> M;

    // For graphs that don't fit in memory, fall back to streaming min-degree approach.
    // Rough budget: N*56 (empty maps) + M*48 (map entries) + M*8 (edge list) <= ~20 GB
    bool huge = (N > 80000000LL || M > 150000000LL);

    if (huge) {
        // Streaming pass: track degrees, store up to 100M edges
        vector<ll> deg(N, 0);
        const ll STORE_LIMIT = 100000000LL;
        vector<pair<int,int>> stored;
        stored.reserve((size_t)min(M, STORE_LIMIT));

        for (ll i = 0; i < M; i++) {
            int u, v; cin >> u >> v;
            deg[u]++;
            deg[v]++;
            if ((ll)stored.size() < STORE_LIMIT)
                stored.push_back({u, v});
        }

        // Min-degree vertex: cutting all its edges gives a valid (upper-bound) cut
        ll min_d = *min_element(deg.begin(), deg.end());
        ll min_v = (ll)(min_element(deg.begin(), deg.end()) - deg.begin());

        cout << min_d << "\n";
        ll emitted = 0;
        for (auto& [u, v] : stored) {
            if (emitted == min_d) break;
            if ((ll)u == min_v || (ll)v == min_v) {
                cout << u << " " << v << "\n";
                emitted++;
            }
        }
        return 0;
    }

    int n = (int)N;
    vector<pair<int,int>> edges;
    edges.reserve((size_t)M);

    // Sparse adjacency: multi-edges are summed (weighted)
    vector<unordered_map<int,ll>> adj(n);

    for (ll i = 0; i < M; i++) {
        int u, v; cin >> u >> v;
        edges.push_back({u, v});
        if (u != v) {
            adj[u][v]++;
            adj[v][u]++;
        }
    }

    // Stoer-Wagner global min-cut (sparse, O(V*(E + V log V)))
    // Each phase: max-adjacency ordering, last node gives cut-of-phase,
    // then merge last into second-to-last.

    vector<vector<int>> super_node(n);
    for (int i = 0; i < n; i++) super_node[i].push_back(i);

    vector<bool> dead(n, false);
    vector<ll>   key(n, 0LL);
    vector<bool> inA(n, false);
    int active = n;

    ll best_cut = LLONG_MAX;
    vector<int> best_side;

    while (active > 1) {
        fill(key.begin(), key.end(), 0LL);
        fill(inA.begin(), inA.end(), false);

        priority_queue<pli> pq;
        for (int i = 0; i < n; i++)
            if (!dead[i]) pq.push({0LL, i});

        int prev = -1, last = -1, cnt = 0;

        while (cnt < active) {
            // Drain stale entries (dead, already-in-A, or outdated priority)
            while (!pq.empty()) {
                auto [k, v] = pq.top();
                if (dead[v] || inA[v] || k < key[v]) { pq.pop(); continue; }
                break;
            }
            if (pq.empty()) break;

            auto [k, v] = pq.top(); pq.pop();
            inA[v] = true;
            prev = last;
            last = v;
            cnt++;

            for (auto& [u, w] : adj[v]) {
                if (!dead[u] && !inA[u]) {
                    key[u] += w;
                    pq.push({key[u], u});
                }
            }
        }

        if (last < 0 || prev < 0) break;

        // key[last] = sum of weights from last to all nodes added before it = cut-of-phase
        if (key[last] < best_cut) {
            best_cut = key[last];
            best_side = super_node[last];
        }

        // Merge 'last' into 'prev': redirect all last's edges to prev
        for (auto& [u, w] : adj[last]) {
            if (u == prev) continue;
            adj[prev][u] += w;
            adj[u][prev] += w;
            adj[u].erase(last);
        }
        adj[prev].erase(last);
        adj[last].clear();

        for (int v : super_node[last])
            super_node[prev].push_back(v);
        super_node[last].clear();

        dead[last] = true;
        active--;
    }

    // Determine which side of the cut each original node is on
    vector<int> comp(n, 1);
    for (int v : best_side) comp[v] = 0;

    cout << best_cut << "\n";
    for (auto& [u, v] : edges) {
        if (comp[u] != comp[v])
            cout << u << " " << v << "\n";
    }

    return 0;
}
