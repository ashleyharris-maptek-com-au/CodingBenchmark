using System;
using System.Collections.Generic;
using System.Text;

class MinesweeperSolver
{
    static int H, W;
    static char[][] board;
    static readonly HashSet<(int x, int y)> safeCells = new();

    static void Main()
    {
        var parts = Console.ReadLine()!.Trim().Split(' ');
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        board = new char[H][];
        for (int r = 0; r < H; r++)
        {
            var row = Console.ReadLine() ?? "";
            if (row.Length < W) row = row.PadRight(W, ' ');
            board[r] = row.ToCharArray();
        }

        // Phase 1: simple rules until convergence
        while (ApplySimpleRules()) { }

        // Phase 2: subset constraint solving (only for manageable boards)
        long size = (long)H * W;
        if (size <= 8_000_000)
        {
            while (ConstraintSubsetStep())
            {
                while (ApplySimpleRules()) { }
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(safeCells.Count.ToString());
        foreach (var (x, y) in safeCells)
            sb.AppendLine($"{x} {y}");
        Console.Write(sb.ToString());
    }

    // Returns true if any change was made
    static bool ApplySimpleRules()
    {
        bool changed = false;
        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                char cell = board[r][c];
                if (cell < '0' || cell > '8') continue;

                int num = cell - '0';
                int flagged = 0;
                int unknownCount = 0;
                // Collect unknowns separately only if needed
                for (int dr = -1; dr <= 1; dr++)
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int nr = r + dr, nc = c + dc;
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    char n = board[nr][nc];
                    if (n == '*') flagged++;
                    else if (n == ' ') unknownCount++;
                }

                int remaining = num - flagged;
                if (remaining < 0) continue;

                if (remaining == 0 && unknownCount > 0)
                {
                    // All unknowns around this cell are safe
                    for (int dr = -1; dr <= 1; dr++)
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        int nr = r + dr, nc = c + dc;
                        if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                        if (board[nr][nc] == ' ')
                        {
                            board[nr][nc] = 'S';
                            safeCells.Add((nc, nr));
                            changed = true;
                        }
                    }
                }
                else if (remaining > 0 && remaining == unknownCount)
                {
                    // All unknowns around this cell are mines
                    for (int dr = -1; dr <= 1; dr++)
                    for (int dc = -1; dc <= 1; dc++)
                    {
                        if (dr == 0 && dc == 0) continue;
                        int nr = r + dr, nc = c + dc;
                        if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                        if (board[nr][nc] == ' ')
                        {
                            board[nr][nc] = '*';
                            changed = true;
                        }
                    }
                }
            }
        }
        return changed;
    }

    // Subset constraint: if unknowns(A) ⊆ unknowns(B), then unknowns(B)\unknowns(A) has count(B)-count(A) mines
    // Only compare constraints from cells within 2 steps of each other (their unknowns can overlap)
    static bool ConstraintSubsetStep()
    {
        // Build constraint for each numbered cell with remaining unknowns
        // constraint[r,c] = (unknown cell set as bitmask positions, count)
        // We'll store as list per cell
        var constrCells = new List<(int r, int c, ulong mask, int[] unknowns, int count)>();

        // For efficiency, encode up to 8 unknowns per constraint cell using neighbor index
        // Actually just store arrays and use HashSet intersection

        // Build constraints indexed by cell position
        var constraintMap = new Dictionary<int, (HashSet<int> unknowns, int count)>();

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                char cell = board[r][c];
                if (cell < '1' || cell > '8') continue;

                int num = cell - '0';
                var unknowns = new HashSet<int>();
                int flagged = 0;

                for (int dr = -1; dr <= 1; dr++)
                for (int dc = -1; dc <= 1; dc++)
                {
                    if (dr == 0 && dc == 0) continue;
                    int nr = r + dr, nc = c + dc;
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    char n = board[nr][nc];
                    if (n == '*') flagged++;
                    else if (n == ' ') unknowns.Add(nr * W + nc);
                }

                int remaining = num - flagged;
                if (unknowns.Count > 0 && remaining >= 0)
                    constraintMap[r * W + c] = (unknowns, remaining);
            }
        }

        bool changed = false;

        foreach (var kvA in constraintMap)
        {
            int keyA = kvA.Key;
            int rA = keyA / W, cA = keyA % W;
            var (unknownsA, countA) = kvA.Value;

            // Only compare with constraints from cells within 4 steps (so unknowns can overlap)
            for (int dr = -4; dr <= 4; dr++)
            for (int dc = -4; dc <= 4; dc++)
            {
                if (dr == 0 && dc == 0) continue;
                int rB = rA + dr, cB = cA + dc;
                if ((uint)rB >= (uint)H || (uint)cB >= (uint)W) continue;
                int keyB = rB * W + cB;
                if (!constraintMap.TryGetValue(keyB, out var valB)) continue;

                var (unknownsB, countB) = valB;

                // Check if A ⊆ B
                if (unknownsA.Count >= unknownsB.Count) continue; // A must be strictly smaller
                if (!unknownsA.IsSubsetOf(unknownsB)) continue;

                // B \ A has countB - countA mines
                int diffCount = countB - countA;
                if (diffCount < 0) continue;

                int diffSize = unknownsB.Count - unknownsA.Count;
                if (diffSize == 0) continue;

                if (diffCount == 0)
                {
                    // All cells in B \ A are safe
                    foreach (int idx in unknownsB)
                    {
                        if (unknownsA.Contains(idx)) continue;
                        int nr = idx / W, nc = idx % W;
                        if (board[nr][nc] == ' ')
                        {
                            board[nr][nc] = 'S';
                            safeCells.Add((nc, nr));
                            changed = true;
                        }
                    }
                }
                else if (diffCount == diffSize)
                {
                    // All cells in B \ A are mines
                    foreach (int idx in unknownsB)
                    {
                        if (unknownsA.Contains(idx)) continue;
                        int nr = idx / W, nc = idx % W;
                        if (board[nr][nc] == ' ')
                        {
                            board[nr][nc] = '*';
                            changed = true;
                        }
                    }
                }
            }
        }

        return changed;
    }
}
