using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

class QAP
{
    static int N;
    static int[][] flow;
    static int[][] dist;
    static Random rng = new Random(42);

    static long ComputeCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            var fi = flow[i];
            int pi = p[i];
            for (int j = 0; j < N; j++)
                cost += (long)fi[j] * dist[pi][p[j]];
        }
        return cost;
    }

    // Delta cost for swapping p[r] and p[s]
    static long SwapDelta(int[] p, int r, int s)
    {
        long delta = 0;
        int pr = p[r], ps = p[s];
        for (int k = 0; k < N; k++)
        {
            if (k == r || k == s) continue;
            int pk = p[k];
            delta += (long)(flow[r][k] - flow[s][k]) * (dist[ps][pk] - dist[pr][pk]);
            delta += (long)(flow[k][r] - flow[k][s]) * (dist[pk][ps] - dist[pk][pr]);
        }
        delta += (long)(flow[r][s] - flow[s][r]) * (dist[ps][pr] - dist[pr][ps]);
        return delta;
    }

    static void LocalSearch2Opt(int[] p, Stopwatch sw, long timeLimitMs)
    {
        bool improved = true;
        while (improved && sw.ElapsedMilliseconds < timeLimitMs)
        {
            improved = false;
            for (int i = 0; i < N - 1 && sw.ElapsedMilliseconds < timeLimitMs; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    if (SwapDelta(p, i, j) < 0)
                    {
                        (p[i], p[j]) = (p[j], p[i]);
                        improved = true;
                    }
                }
            }
        }
    }

    // Greedy construction: assign facilities to locations greedily
    static int[] GreedyConstruct()
    {
        // Sort facilities by total outflow desc, locations by total distance asc
        long[] facilityFlow = new long[N];
        long[] locationDist = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                facilityFlow[i] += flow[i][j] + flow[j][i];
                locationDist[i] += dist[i][j] + dist[j][i];
            }
        }
        int[] facOrder = Enumerable.Range(0, N).OrderByDescending(i => facilityFlow[i]).ToArray();
        int[] locOrder = Enumerable.Range(0, N).OrderBy(i => locationDist[i]).ToArray();
        int[] p = new int[N];
        for (int i = 0; i < N; i++)
            p[facOrder[i]] = locOrder[i];
        return p;
    }

    static int[] RandomPermutation()
    {
        int[] p = Enumerable.Range(0, N).ToArray();
        for (int i = N - 1; i > 0; i--)
        {
            int j = rng.Next(i + 1);
            (p[i], p[j]) = (p[j], p[i]);
        }
        return p;
    }

    static void Main()
    {
        // Read input
        N = int.Parse(Console.ReadLine().Trim());
        flow = new int[N][];
        dist = new int[N][];

        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            flow[i] = new int[N];
            for (int j = 0; j < N; j++)
                flow[i][j] = int.Parse(parts[j]);
        }
        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split(' ', StringSplitOptions.RemoveEmptyEntries);
            dist[i] = new int[N];
            for (int j = 0; j < N; j++)
                dist[i][j] = int.Parse(parts[j]);
        }

        var sw = Stopwatch.StartNew();
        long timeLimitMs = 280000; // 4min 40sec

        int[] bestP = GreedyConstruct();
        LocalSearch2Opt(bestP, sw, timeLimitMs);
        long bestCost = ComputeCost(bestP);

        // For small N, do multiple random restarts
        if (N <= 200)
        {
            while (sw.ElapsedMilliseconds < timeLimitMs)
            {
                int[] p = RandomPermutation();
                LocalSearch2Opt(p, sw, timeLimitMs);
                long cost = ComputeCost(p);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestP = p;
                }
            }
        }
        else if (N <= 600)
        {
            // Perturbation-based ILS
            while (sw.ElapsedMilliseconds < timeLimitMs)
            {
                int[] p = (int[])bestP.Clone();
                // Random double-bridge or random swaps for perturbation
                int perturbCount = Math.Max(4, N / 20);
                for (int k = 0; k < perturbCount; k++)
                {
                    int a = rng.Next(N), b = rng.Next(N);
                    (p[a], p[b]) = (p[b], p[a]);
                }
                LocalSearch2Opt(p, sw, timeLimitMs);
                long cost = ComputeCost(p);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestP = p;
                }
            }
        }
        else
        {
            // Large N: just do one pass of 2-opt then perturbation if time allows
            while (sw.ElapsedMilliseconds < timeLimitMs)
            {
                int[] p = (int[])bestP.Clone();
                int perturbCount = Math.Max(10, N / 10);
                for (int k = 0; k < perturbCount; k++)
                {
                    int a = rng.Next(N), b = rng.Next(N);
                    (p[a], p[b]) = (p[b], p[a]);
                }
                LocalSearch2Opt(p, sw, timeLimitMs);
                long cost = ComputeCost(p);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestP = p;
                }
            }
        }

        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestP));
    }
}
