; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 80
; Schema: 0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Input/output decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 0 ColMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 1 ColMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 2 ColMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1

%UBOType = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBOType = OpTypePointer Uniform %UBOType
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%funcType = OpTypeFunction %void

; Variables
%inWorldPos = OpVariable %ptr_Input_v3float Input
%inNormal   = OpVariable %ptr_Input_v3float Input
%inUV       = OpVariable %ptr_Input_v2float Input
%inTangent  = OpVariable %ptr_Input_v3float Input
%inColor    = OpVariable %ptr_Input_v3float Input
%outColor   = OpVariable %ptr_Output_v4float Output
%ubo        = OpVariable %ptr_Uniform_UBOType Uniform

; Constants
%int_3       = OpConstant %int 3
%float_0     = OpConstant %float 0.0
%float_1     = OpConstant %float 1.0
%float_0_5   = OpConstant %float 0.5
%float_0_3   = OpConstant %float 0.3
%float_0_4   = OpConstant %float 0.4
%float_0_6   = OpConstant %float 0.6
%float_0_15  = OpConstant %float 0.15
%float_0_05  = OpConstant %float 0.05

%skyColor    = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05

; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel

; Normalize incoming normal
%normalRaw = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %glsl Normalize %normalRaw

; blend = N.y * 0.5 + 0.5
%Ny        = OpCompositeExtract %float %N 1
%halfNy    = OpFMul %float %Ny %float_0_5
%blend     = OpFAdd %float %halfNy %float_0_5
%blendClamped = OpExtInst %float %glsl FClamp %blend %float_0 %float_1

; ambient = mix(groundColor, skyColor, blend)
%ambient = OpExtInst %v3float %glsl FMix %groundColor %skyColor %blendClamped

; Load worldPos and lightPos for diffuse
%worldPosVal = OpLoad %v3float %inWorldPos
%lightPosPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%lightPos4   = OpLoad %v4float %lightPosPtr
%lightPos3   = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

; L = normalize(lightPos - worldPos)
%lightDir = OpFSub %v3float %lightPos3 %worldPosVal
%L = OpExtInst %v3float %glsl Normalize %lightDir

; d = max(dot(N, L), 0)
%dotNL = OpDot %float %N %L
%d     = OpExtInst %float %glsl FMax %dotNL %float_0

; diffuse = 0.3 * d as vec3
%diffuseScalar = OpFMul %float %float_0_3 %d
%diffuseVec    = OpCompositeConstruct %v3float %diffuseScalar %diffuseScalar %diffuseScalar

; total color = ambient + diffuse
%colorRGB = OpFAdd %v3float %ambient %diffuseVec

; Build output vec4 with alpha = 1.0
%r = OpCompositeExtract %float %colorRGB 0
%g = OpCompositeExtract %float %colorRGB 1
%b = OpCompositeExtract %float %colorRGB 2
%outRGBA = OpCompositeConstruct %v4float %r %g %b %float_1

OpStore %outColor %outRGBA
OpReturn
OpFunctionEnd
