use std::io::{self, BufRead, Write, BufWriter};

fn point_in_polygon(px: f64, py: f64, poly: &[(f64, f64)]) -> bool {
    let n = poly.len();
    if n < 3 {
        return false;
    }
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        if ((yi > py) != (yj > py)) && (px < (xj - xi) * (py - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn segment_aabb_intersect(
    p1: (f64, f64),
    p2: (f64, f64),
    x0: f64,
    y0: f64,
    x1: f64,
    y1: f64,
) -> bool {
    let dx = p2.0 - p1.0;
    let dy = p2.1 - p1.1;
    let p = [-dx, dx, -dy, dy];
    let q = [p1.0 - x0, x1 - p1.0, p1.1 - y0, y1 - p1.1];
    let mut t0 = 0.0f64;
    let mut t1 = 1.0f64;
    for i in 0..4 {
        if p[i].abs() < 1e-12 {
            if q[i] < 0.0 {
                return false;
            }
        } else if p[i] < 0.0 {
            t0 = t0.max(q[i] / p[i]);
        } else {
            t1 = t1.min(q[i] / p[i]);
        }
    }
    t0 <= t1
}

fn aabb_polygon_overlap(gx: f64, gy: f64, half: f64, poly: &[(f64, f64)]) -> bool {
    let x0 = gx - half;
    let x1 = gx + half;
    let y0 = gy - half;
    let y1 = gy + half;

    for &(cx, cy) in &[
        (gx, gy),
        (x0, y0),
        (x0, y1),
        (x1, y0),
        (x1, y1),
        (gx, y0),
        (gx, y1),
        (x0, gy),
        (x1, gy),
    ] {
        if point_in_polygon(cx, cy, poly) {
            return true;
        }
    }

    for &(px, py) in poly {
        if px >= x0 && px <= x1 && py >= y0 && py <= y1 {
            return true;
        }
    }

    let n = poly.len();
    for i in 0..n {
        if segment_aabb_intersect(poly[i], poly[(i + 1) % n], x0, y0, x1, y1) {
            return true;
        }
    }

    false
}

// Compute the half-side of the largest axis-aligned square centered at the origin
// that fits inside the convex polygon defined by verts (centroid must be at origin).
fn inscribed_square_half_side(verts: &[(f64, f64)]) -> f64 {
    let mut sv = verts.to_vec();
    sv.sort_by(|a, b| {
        let aa = a.1.atan2(a.0);
        let ab = b.1.atan2(b.0);
        aa.partial_cmp(&ab)
            .unwrap_or(std::cmp::Ordering::Equal)
    });
    let n = sv.len();
    let mut min_d = f64::INFINITY;
    for i in 0..n {
        let (ax, ay) = sv[i];
        let (bx, by) = sv[(i + 1) % n];
        // Outward normal for CCW polygon edge A->B: right-perpendicular of (B-A)
        let nx = by - ay;
        let ny = -(bx - ax);
        // Constraint: nx*(x-ax)+ny*(y-ay) <= 0 for interior
        // => nx*x + ny*y <= rhs = nx*ax + ny*ay
        let rhs = nx * ax + ny * ay;
        if rhs <= 0.0 {
            continue;
        }
        // For axis-aligned square [-d,d]^2: max of nx*x+ny*y = d*(|nx|+|ny|)
        let denom = nx.abs() + ny.abs();
        if denom < 1e-12 {
            continue;
        }
        let d = rhs / denom;
        if d < min_d {
            min_d = d;
        }
    }
    if min_d.is_finite() && min_d > 0.0 {
        min_d
    } else {
        0.5
    }
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let nv: usize = lines
        .next()
        .unwrap()
        .unwrap()
        .trim()
        .parse()
        .unwrap();

    let mut poly: Vec<(f64, f64)> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let mut parts = line.trim().split_whitespace();
        let x: f64 = parts.next().unwrap().parse().unwrap();
        let y: f64 = parts.next().unwrap().parse().unwrap();
        poly.push((x, y));
    }

    let sun_line = lines.next().unwrap().unwrap();
    let mut sp = sun_line.trim().split_whitespace();
    let sx: f64 = sp.next().unwrap().parse().unwrap();
    let sy: f64 = sp.next().unwrap().parse().unwrap();
    let sz: f64 = sp.next().unwrap().parse().unwrap();

    if poly.len() < 3 {
        writeln!(out, "0").unwrap();
        return;
    }

    let scale = 10.0f64;

    // kx = sx/sz, ky = sy/sz (slope of sun ray projected onto z=0)
    let sz_eff = if sz.abs() < 1e-9 {
        if sz >= 0.0 { 1e-9 } else { -1e-9 }
    } else {
        sz
    };
    let kx = sx / sz_eff;
    let ky = sy / sz_eff;

    // Shadow of standard tetrahedron vertex v=(vx,vy,vz) with center at (cx,cy,h):
    //   shadow = (cx - h*kx + scale*(vx - vz*kx),  cy - h*ky + scale*(vy - vz*ky))
    // Relative to shadow center (cx-h*kx, cy-h*ky), at scale=1:
    //   w = (vx - vz*kx,  vy - vz*ky)
    //
    // Standard verts: v1=(1,0,-0.707), v2=(-1,0,-0.707), v3=(0,1,0.707), v4=(0,-1,0.707)
    let shadow_unit: [(f64, f64); 4] = [
        (1.0 + 0.707 * kx,  0.707 * ky),        // v1
        (-1.0 + 0.707 * kx, 0.707 * ky),        // v2
        (-0.707 * kx,       1.0 - 0.707 * ky),  // v3
        (-0.707 * kx,      -1.0 - 0.707 * ky),  // v4
    ];

    // The inscribed square of the shadow (scale=1) has half-side = half_d.
    // With scale=10, grid_spacing = 2 * half_d * scale ensures any polygon point
    // within a grid cell is guaranteed to be inside that cell's shadow.
    let half_d = inscribed_square_half_side(&shadow_unit).max(0.01);
    let grid_spacing = 2.0 * half_d * scale;

    let min_x = poly.iter().map(|p| p.0).fold(f64::INFINITY, f64::min);
    let max_x = poly.iter().map(|p| p.0).fold(f64::NEG_INFINITY, f64::max);
    let min_y = poly.iter().map(|p| p.1).fold(f64::INFINITY, f64::min);
    let max_y = poly.iter().map(|p| p.1).fold(f64::NEG_INFINITY, f64::max);

    let ox = min_x - grid_spacing;
    let oy = min_y - grid_spacing;
    let nx = ((max_x - ox + grid_spacing) / grid_spacing).ceil() as i64 + 2;
    let ny = ((max_y - oy + grid_spacing) / grid_spacing).ceil() as i64 + 2;

    // Each tetrahedron gets a unique height so their z-ranges never overlap,
    // guaranteeing no 3D intersections.
    // z-range of tetrahedron at height h with scale s: [h - 0.707*s, h + 0.707*s]
    // Step > 2*0.707*s = 14.14 ensures disjoint ranges.
    let h_step = scale * 0.707 * 2.0 + 1.0;

    let mut results: Vec<[f64; 8]> = Vec::new();

    for iy in 0..ny {
        for ix in 0..nx {
            // Shadow center for this grid cell
            let gx = ox + (ix as f64 + 0.5) * grid_spacing;
            let gy = oy + (iy as f64 + 0.5) * grid_spacing;

            if !aabb_polygon_overlap(gx, gy, grid_spacing * 0.5, &poly) {
                continue;
            }

            // Height chosen so all z-ranges are disjoint
            let h = scale * 0.707 + results.len() as f64 * h_step;

            // 3D center: shadow center = (cx - h*kx, cy - h*ky), so:
            let cx = gx + h * kx;
            let cy = gy + h * ky;

            // Identity rotation quaternion (w=1, x=0, y=0, z=0)
            results.push([cx, cy, h, 1.0, 0.0, 0.0, 0.0, scale]);
        }
    }

    writeln!(out, "{}", results.len()).unwrap();
    for r in &results {
        writeln!(
            out,
            "{} {} {} {} {} {} {} {}",
            r[0], r[1], r[2], r[3], r[4], r[5], r[6], r[7]
        )
        .unwrap();
    }
}
