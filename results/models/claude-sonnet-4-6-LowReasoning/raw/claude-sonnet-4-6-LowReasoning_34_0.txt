import sys
import math
import random
from itertools import combinations

def dist(a, b):
    return math.sqrt((a[0]-b[0])**2 + (a[1]-b[1])**2)

def route_distance(route, depot, customers):
    if not route:
        return 0.0
    d = dist(depot, customers[route[0]])
    for i in range(len(route)-1):
        d += dist(customers[route[i]], customers[route[i+1]])
    d += dist(customers[route[-1]], depot)
    return d

def total_dist(routes, depot, customers):
    return sum(route_distance(r, depot, customers) for r in routes)

def savings_init(depot, customers, capacity, max_k):
    n = len(customers)
    # Clarke-Wright savings
    savings = []
    for i in range(n):
        for j in range(i+1, n):
            s = dist(depot, customers[i]) + dist(depot, customers[j]) - dist(customers[i], customers[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    # Start with individual routes
    routes = [[i] for i in range(n)]
    route_of = {i: i for i in range(n)}
    demands = [customers[i][2] for i in range(n)]
    route_demand = {i: demands[i] for i in range(n)}
    route_end = {i: (i, i) for i in range(n)}  # (first, last) customer in route

    for s, i, j in savings:
        ri = route_of[i]
        rj = route_of[j]
        if ri == rj:
            continue
        if len(routes) - 1 >= max_k and ri != rj:
            # merging reduces count, that's fine
            pass
        if route_demand[ri] + route_demand[rj] > capacity:
            continue
        # Check if i is at end of ri and j is at start of rj (or vice versa)
        fi, li = route_end[ri]
        fj, lj = route_end[rj]

        merged = None
        if li == i and fj == j:
            merged = routes[ri] + routes[rj]
        elif li == i and lj == j:
            merged = routes[ri] + list(reversed(routes[rj]))
        elif fi == i and fj == j:
            merged = list(reversed(routes[ri])) + routes[rj]
        elif fi == i and lj == j:
            merged = routes[rj] + routes[ri]

        if merged is None:
            continue

        new_id = ri
        routes[new_id] = merged
        route_demand[new_id] = route_demand[ri] + route_demand[rj]
        route_end[new_id] = (merged[0], merged[-1])
        routes[rj] = []
        for c in routes[new_id]:
            route_of[c] = new_id

    result = [r for r in routes if r]
    return result

def two_opt_route(route, depot, customers):
    improved = True
    while improved:
        improved = False
        n = len(route)
        for i in range(n-1):
            for j in range(i+2, n):
                a = customers[route[i]] if i >= 0 else depot
                b = customers[route[i+1]]
                c = customers[route[j]]
                d = customers[route[j+1]] if j+1 < n else depot
                # Use depot for boundary
                if i == -1:
                    a = depot
                d0 = dist(a if i >= 0 else depot, b) + dist(c, d)
                d1 = dist(a if i >= 0 else depot, c) + dist(b, d)
                if d1 < d0 - 1e-10:
                    route[i+1:j+1] = route[i+1:j+1][::-1]
                    improved = True
    return route

def or_opt_route(route, depot, customers):
    improved = True
    while improved:
        improved = False
        n = len(route)
        for seg_len in [1, 2, 3]:
            for i in range(n - seg_len + 1):
                seg = route[i:i+seg_len]
                rest = route[:i] + route[i+seg_len:]
                prev_i = customers[route[i-1]] if i > 0 else depot
                next_i = customers[route[i+seg_len]] if i+seg_len < n else depot
                seg_start = customers[seg[0]]
                seg_end = customers[seg[-1]]
                old = dist(prev_i, seg_start) + dist(seg_end, next_i)
                if seg_len == 1:
                    old += 0
                else:
                    old += sum(dist(customers[seg[k]], customers[seg[k+1]]) for k in range(len(seg)-1))
                # try inserting seg after each position in rest
                for j in range(len(rest)+1):
                    if j == i or j == i - seg_len:
                        continue
                    prev_j = customers[rest[j-1]] if j > 0 else depot
                    next_j = customers[rest[j]] if j < len(rest) else depot
                    gain = (dist(prev_i, next_i) + dist(prev_j, seg_start) + dist(seg_end, next_j)
                            - old - dist(prev_j, next_j))
                    if gain < -1e-10:
                        new_route = rest[:j] + seg + rest[j:]
                        route = new_route
                        improved = True
                        n = len(route)
                        break
                if improved:
                    break
            if improved:
                break
    return route

def relocate_between_routes(routes, depot, customers, capacity):
    demands = [c[2] for c in customers]
    route_demands = [sum(demands[c] for c in r) for r in routes]
    improved = True
    while improved:
        improved = False
        for i in range(len(routes)):
            for ci in range(len(routes[i])):
                c = routes[i][ci]
                d = demands[c]
                # Try inserting c into another route
                ri = routes[i]
                prev_c = customers[ri[ci-1]] if ci > 0 else depot
                next_c = customers[ri[ci+1]] if ci+1 < len(ri) else depot
                removal_gain = (dist(prev_c, customers[c]) + dist(customers[c], next_c)
                                - dist(prev_c, next_c))
                for j in range(len(routes)):
                    if i == j:
                        continue
                    if route_demands[j] + d > capacity:
                        continue
                    rj = routes[j]
                    for cj in range(len(rj)+1):
                        prev_j = customers[rj[cj-1]] if cj > 0 else depot
                        next_j = customers[rj[cj]] if cj < len(rj) else depot
                        insert_cost = (dist(prev_j, customers[c]) + dist(customers[c], next_j)
                                       - dist(prev_j, next_j))
                        if insert_cost - removal_gain < -1e-10:
                            routes[i] = ri[:ci] + ri[ci+1:]
                            routes[j] = rj[:cj] + [c] + rj[cj:]
                            route_demands[i] -= d
                            route_demands[j] += d
                            improved = True
                            break
                    if improved:
                        break
                if improved:
                    break
            if improved:
                break
        # Remove empty routes
        routes = [r for r in routes if r]
        route_demands = [sum(demands[c] for c in r) for r in routes]
    return routes

def solve():
    data = sys.stdin.read().split()
    idx = 0
    N, K, CAP = int(data[idx]), int(data[idx+1]), int(data[idx+2])
    idx += 3
    depot = (float(data[idx]), float(data[idx+1]))
    idx += 2
    customers = []
    for _ in range(N):
        x, y, dem = float(data[idx]), float(data[idx+1]), int(data[idx+2])
        customers.append((x, y, dem))
        idx += 3

    if N == 0:
        print(0.0)
        print(0)
        return

    # Clarke-Wright savings heuristic
    routes = savings_init(depot, customers, CAP, K)

    # If too many routes, merge greedily
    while len(routes) > K:
        best_gain = -1e18
        best_pair = None
        demands = [c[2] for c in customers]
        for i in range(len(routes)):
            for j in range(i+1, len(routes)):
                if sum(demands[c] for c in routes[i]) + sum(demands[c] for c in routes[j]) > CAP:
                    continue
                # Try merging
                d_before = route_distance(routes[i], depot, customers) + route_distance(routes[j], depot, customers)
                # Try all 4 orientations
                best_merge = None
                best_merge_d = 1e18
                for ri in [routes[i], list(reversed(routes[i]))]:
                    for rj in [routes[j], list(reversed(routes[j]))]:
                        merged = ri + rj
                        d = route_distance(merged, depot, customers)
                        if d < best_merge_d:
                            best_merge_d = d
                            best_merge = merged
                gain = d_before - best_merge_d
                if gain > best_gain:
                    best_gain = gain
                    best_pair = (i, j, best_merge)
        if best_pair is None:
            break
        i, j, merged = best_pair
        routes[i] = merged
        routes.pop(j)

    # Local search: 2-opt within each route
    for idx2 in range(len(routes)):
        routes[idx2] = two_opt_route(routes[idx2], depot, customers)

    # Relocate between routes
    routes = relocate_between_routes(routes, depot, customers, CAP)

    # 2-opt again after relocation
    for idx2 in range(len(routes)):
        routes[idx2] = two_opt_route(routes[idx2], depot, customers)

    td = total_dist(routes, depot, customers)
    print(f"{td:.6f}")
    print(len(routes))
    for r in routes:
        print(' '.join(map(str, r)))

solve()
