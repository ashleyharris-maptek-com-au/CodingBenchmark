; SPIR-V geometry shader: tapered ribbon from InputLines
OpCapability Geometry
%glsl450    = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for gl_PerVertex blocks
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %out_color Location 0

; Types
%void       = OpTypeVoid
%bool       = OpTypeBool
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 1
%uint       = OpTypeInt 32 0
%v3float    = OpTypeVector %float 3
%v4float    = OpTypeVector %float 4

; Constants
%uint_2              = OpConstant %uint 2
%int_0               = OpConstant %int 0
%int_1               = OpConstant %int 1
%float_0             = OpConstant %float 0
%float_1             = OpConstant %float 1
%float_thick_start   = OpConstant %float 0.04
%float_thick_end     = OpConstant %float 0.005

; Struct types
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Pointer and array types for input
%arr_pv_in      = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv  = OpTypePointer Input %arr_pv_in
%ptr_in_v4      = OpTypePointer Input %v4float

%arr_v3_2       = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3  = OpTypePointer Input %arr_v3_2
%ptr_in_v3      = OpTypePointer Input %v3float

; Pointer types for output
%ptr_out_pv     = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4     = OpTypePointer Output %v4float
%ptr_out_v3     = OpTypePointer Output %v3float

; Interface variables
%gl_in      = OpVariable %ptr_in_arr_pv Input
%gl_out     = OpVariable %ptr_out_pv Output
%in_color   = OpVariable %ptr_in_arr_v3 Input
%out_color  = OpVariable %ptr_out_v3 Output

; Function type
%fn_void    = OpTypeFunction %void

; Main function
%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0 position (vertex 0)
%ptr_p0_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0_v4      = OpLoad %v4float %ptr_p0_pos
%p0_x       = OpCompositeExtract %float %p0_v4 0
%p0_y       = OpCompositeExtract %float %p0_v4 1
%p0_z       = OpCompositeExtract %float %p0_v4 2
%p0_w       = OpCompositeExtract %float %p0_v4 3

; Load p1 position (vertex 1)
%ptr_p1_pos = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1_v4      = OpLoad %v4float %ptr_p1_pos
%p1_x       = OpCompositeExtract %float %p1_v4 0
%p1_y       = OpCompositeExtract %float %p1_v4 1
%p1_z       = OpCompositeExtract %float %p1_v4 2
%p1_w       = OpCompositeExtract %float %p1_v4 3

; Load colors
%ptr_c0     = OpAccessChain %ptr_in_v3 %in_color %int_0
%c0         = OpLoad %v3float %ptr_c0
%ptr_c1     = OpAccessChain %ptr_in_v3 %in_color %int_1
%c1         = OpLoad %v3float %ptr_c1

; Compute direction: (p1 - p0) in xy
%dir_x      = OpFSub %float %p1_x %p0_x
%dir_y      = OpFSub %float %p1_y %p0_y

; Perpendicular: (-dir_y, dir_x, 0) — normalized
%neg_dir_y  = OpFNegate %float %dir_y
%perp_x2    = OpFMul %float %neg_dir_y %neg_dir_y
%perp_y2    = OpFMul %float %dir_x %dir_x
%perp_len2  = OpFAdd %float %perp_x2 %perp_y2
%perp_len   = OpExtInst %float %glsl450 Sqrt %perp_len2
%eps        = OpConstant %float 0.000001
%safe_len   = OpExtInst %float %glsl450 FMax %perp_len %eps
%inv_len    = OpFDiv %float %float_1 %safe_len
%perp_nx    = OpFMul %float %neg_dir_y %inv_len
%perp_ny    = OpFMul %float %dir_x %inv_len

; Offsets at start (thick_start = 0.04)
%ots_x      = OpFMul %float %perp_nx %float_thick_start
%ots_y      = OpFMul %float %perp_ny %float_thick_start

; Offsets at end (thick_end = 0.005)
%ote_x      = OpFMul %float %perp_nx %float_thick_end
%ote_y      = OpFMul %float %perp_ny %float_thick_end

; Output position accessor
%ptr_gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Vertex 0: p0 + perp * thick_start, color = c0 ---
%v0_x       = OpFAdd %float %p0_x %ots_x
%v0_y       = OpFAdd %float %p0_y %ots_y
%v0_pos     = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
OpStore %ptr_gl_out_pos %v0_pos
OpStore %out_color %c0
OpEmitVertex

; --- Vertex 1: p0 - perp * thick_start, color = c0 ---
%v1_x       = OpFSub %float %p0_x %ots_x
%v1_y       = OpFSub %float %p0_y %ots_y
%v1_pos     = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
OpStore %ptr_gl_out_pos %v1_pos
OpStore %out_color %c0
OpEmitVertex

; --- Vertex 2: p1 + perp * thick_end, color = c1 ---
%v2_x       = OpFAdd %float %p1_x %ote_x
%v2_y       = OpFAdd %float %p1_y %ote_y
%v2_pos     = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
OpStore %ptr_gl_out_pos %v2_pos
OpStore %out_color %c1
OpEmitVertex

; --- Vertex 3: p1 - perp * thick_end, color = c1 ---
%v3_x       = OpFSub %float %p1_x %ote_x
%v3_y       = OpFSub %float %p1_y %ote_y
%v3_pos     = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
OpStore %ptr_gl_out_pos %v3_pos
OpStore %out_color %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
