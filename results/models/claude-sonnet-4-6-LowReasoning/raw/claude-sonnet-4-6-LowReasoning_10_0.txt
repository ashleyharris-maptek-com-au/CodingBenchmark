#include <iostream>
#include <vector>
#include <algorithm>
#include <numeric>
#include <deque>

using namespace std;

static int BW, BH;

struct Bin {
    vector<int> sky; // sky[x] = filled height at column x
    long long rem;   // remaining area (lower bound; for fast skip)

    Bin() : sky(BW, 0), rem((long long)BW * BH) {}

    // Find best (minimum height) position for a rw x rh rect using sliding window max.
    // Returns false if no valid position exists.
    bool fit(int rw, int rh, int& ox, int& oy) const {
        if (rw > BW || rh > BH) return false;
        if (rem < (long long)rw * rh) return false;

        deque<int> dq;
        int best_y = BH, best_x = -1;

        // Build initial window [0, rw-1]
        for (int i = 0; i < rw; i++) {
            while (!dq.empty() && sky[dq.back()] <= sky[i]) dq.pop_back();
            dq.push_back(i);
        }
        // Check position x=0
        {
            int h = sky[dq.front()];
            if (h + rh <= BH) { best_y = h; best_x = 0; }
        }
        // Slide window to the right
        for (int x = 1; x + rw <= BW; x++) {
            while (!dq.empty() && dq.front() < x) dq.pop_front();
            int ni = x + rw - 1;
            while (!dq.empty() && sky[dq.back()] <= sky[ni]) dq.pop_back();
            dq.push_back(ni);
            int h = sky[dq.front()];
            if (h + rh <= BH && h < best_y) {
                best_y = h; best_x = x;
            }
        }

        if (best_x < 0) return false;
        ox = best_x; oy = best_y;
        return true;
    }

    void place(int px, int py, int rw, int rh) {
        int nh = py + rh;
        rem -= (long long)rw * rh;
        fill(sky.begin() + px, sky.begin() + px + rw, nh);
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N >> BW >> BH;

    vector<int> RW(N), RH(N);
    for (int i = 0; i < N; i++) cin >> RW[i] >> RH[i];

    // Sort by decreasing max dimension, then decreasing min dimension
    vector<int> ord(N);
    iota(ord.begin(), ord.end(), 0);
    sort(ord.begin(), ord.end(), [&](int a, int b) {
        int ha = max(RW[a], RH[a]), hb = max(RW[b], RH[b]);
        if (ha != hb) return ha > hb;
        return min(RW[a], RH[a]) > min(RW[b], RH[b]);
    });

    vector<Bin> bins;
    vector<int> B(N, 0), X(N, 0), Y(N, 0), ROT(N, 0);

    for (int idx : ord) {
        int rw = RW[idx], rh = RH[idx];
        bool placed = false;
        int ox, oy;

        // Search bins from newest to oldest (newest most likely to have space)
        for (int bi = (int)bins.size() - 1; bi >= 0 && !placed; bi--) {
            if (bins[bi].fit(rw, rh, ox, oy)) {
                bins[bi].place(ox, oy, rw, rh);
                B[idx] = bi; X[idx] = ox; Y[idx] = oy; ROT[idx] = 0;
                placed = true;
            } else if (rw != rh && bins[bi].fit(rh, rw, ox, oy)) {
                bins[bi].place(ox, oy, rh, rw);
                B[idx] = bi; X[idx] = ox; Y[idx] = oy; ROT[idx] = 1;
                placed = true;
            }
        }

        if (!placed) {
            int bi = (int)bins.size();
            bins.emplace_back();
            bool rotated = false;
            if (!bins[bi].fit(rw, rh, ox, oy)) {
                // Try rotated in new bin
                if (rw != rh && bins[bi].fit(rh, rw, ox, oy)) rotated = true;
                else { ox = 0; oy = 0; } // fallback (shouldn't happen if rect fits board)
            }
            if (rotated) {
                bins[bi].place(ox, oy, rh, rw);
                ROT[idx] = 1;
            } else {
                bins[bi].place(ox, oy, rw, rh);
                ROT[idx] = 0;
            }
            B[idx] = bi; X[idx] = ox; Y[idx] = oy;
        }
    }

    cout << bins.size() << '\n';
    for (int i = 0; i < N; i++) {
        cout << B[i] << ' ' << X[i] << ' ' << Y[i] << ' ' << ROT[i] << '\n';
    }

    return 0;
}
