use std::io::{self, Read, Write, BufWriter, BufReader};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::new(stdin.lock());
    let mut writer = BufWriter::new(stdout.lock());

    // Read header line
    let mut header = String::new();
    loop {
        let mut byte = [0u8; 1];
        reader.read_exact(&mut byte).expect("Failed to read header");
        if byte[0] == b'\n' { break; }
        header.push(byte[0] as char);
    }
    let mut parts = header.trim().split_whitespace();
    let num_points: u64 = parts.next().unwrap().parse().unwrap();
    let num_clusters: usize = parts.next().unwrap().parse().unwrap();

    if num_points == 0 || num_clusters == 0 {
        return;
    }

    let n = num_points as usize;
    let k = num_clusters;

    // Memory thresholds
    // Each point = 12 bytes, assignments = 4 bytes => 16 bytes/point
    // For 10M points: ~160MB - manageable
    // For 100M points: ~1.6GB - need streaming
    const SMALL_THRESHOLD: u64 = 10_000_000;
    const MEDIUM_THRESHOLD: u64 = 100_000_000;

    if num_points <= SMALL_THRESHOLD {
        // Load all points into memory, run standard k-means
        let points = read_all_points(&mut reader, n);
        let assignments = kmeans_full(&points, k, 100);
        write_assignments(&mut writer, &assignments);
    } else if num_points <= MEDIUM_THRESHOLD {
        // Mini-batch k-means with streaming
        mini_batch_kmeans(&mut reader, &mut writer, n, k);
    } else {
        // Streaming k-means with very limited memory
        streaming_kmeans(&mut reader, &mut writer, num_points, k);
    }
    writer.flush().unwrap();
}

fn read_all_points<R: Read>(reader: &mut R, n: usize) -> Vec<[f32; 3]> {
    let mut points = vec![[0f32; 3]; n];
    let mut buf = vec![0u8; n * 12];
    reader.read_exact(&mut buf).expect("Failed to read points");
    for i in 0..n {
        let off = i * 12;
        points[i][0] = f32::from_le_bytes(buf[off..off+4].try_into().unwrap());
        points[i][1] = f32::from_le_bytes(buf[off+4..off+8].try_into().unwrap());
        points[i][2] = f32::from_le_bytes(buf[off+8..off+12].try_into().unwrap());
    }
    points
}

fn write_assignments<W: Write>(writer: &mut W, assignments: &[u32]) {
    let mut buf = vec![0u8; assignments.len() * 4];
    for (i, &a) in assignments.iter().enumerate() {
        let bytes = a.to_le_bytes();
        buf[i*4..i*4+4].copy_from_slice(&bytes);
    }
    writer.write_all(&buf).unwrap();
}

fn kmeans_full(points: &[[f32; 3]], k: usize, max_iter: usize) -> Vec<u32> {
    let n = points.len();
    if k >= n {
        // Each point gets its own cluster (or wrap around)
        return (0..n).map(|i| (i % k) as u32).collect();
    }

    // Initialize centroids using kmeans++ style (simplified: evenly spaced)
    let mut centroids = init_centroids_pp(points, k);
    let mut assignments = vec![0u32; n];

    for _iter in 0..max_iter {
        // Assign points to nearest centroid
        let mut changed = false;
        for i in 0..n {
            let nearest = find_nearest(&points[i], &centroids);
            if assignments[i] != nearest as u32 {
                assignments[i] = nearest as u32;
                changed = true;
            }
        }
        if !changed { break; }

        // Update centroids
        let mut sums = vec![[0f64; 3]; k];
        let mut counts = vec![0u64; k];
        for i in 0..n {
            let c = assignments[i] as usize;
            sums[c][0] += points[i][0] as f64;
            sums[c][1] += points[i][1] as f64;
            sums[c][2] += points[i][2] as f64;
            counts[c] += 1;
        }
        // Handle empty clusters by re-assigning to furthest point
        let mut empty_clusters: Vec<usize> = (0..k).filter(|&c| counts[c] == 0).collect();
        if !empty_clusters.is_empty() {
            reassign_empty_clusters(points, &assignments, &mut centroids, &mut sums, &mut counts, &mut empty_clusters);
        }
        for c in 0..k {
            if counts[c] > 0 {
                centroids[c][0] = (sums[c][0] / counts[c] as f64) as f32;
                centroids[c][1] = (sums[c][1] / counts[c] as f64) as f32;
                centroids[c][2] = (sums[c][2] / counts[c] as f64) as f32;
            }
        }
    }
    assignments
}

fn init_centroids_pp(points: &[[f32; 3]], k: usize) -> Vec<[f32; 3]> {
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    // Pick first centroid: point at n/2
    centroids.push(points[n / 2]);

    // Simple deterministic spread: pick points maximally distant
    let step = n / k;
    for i in 1..k {
        // Pick the point with max min-distance to existing centroids
        // For speed, sample every `step` points
        let mut best_idx = i * step % n;
        let mut best_dist = 0f32;
        let sample_step = if n > 100_000 { n / 10_000 } else { 1 };
        let mut idx = 0;
        while idx < n {
            let d = min_dist_to_centroids(&points[idx], &centroids);
            if d > best_dist {
                best_dist = d;
                best_idx = idx;
            }
            idx += sample_step;
        }
        centroids.push(points[best_idx]);
    }
    centroids
}

fn min_dist_to_centroids(p: &[f32; 3], centroids: &[[f32; 3]]) -> f32 {
    centroids.iter().map(|c| dist2(p, c)).fold(f32::MAX, f32::min)
}

#[inline]
fn dist2(a: &[f32; 3], b: &[f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx*dx + dy*dy + dz*dz
}

#[inline]
fn find_nearest(p: &[f32; 3], centroids: &[[f32; 3]]) -> usize {
    let mut best = 0;
    let mut best_d = f32::MAX;
    for (i, c) in centroids.iter().enumerate() {
        let d = dist2(p, c);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

fn reassign_empty_clusters(
    points: &[[f32; 3]],
    assignments: &[u32],
    centroids: &mut Vec<[f32; 3]>,
    sums: &mut Vec<[f64; 3]>,
    counts: &mut Vec<u64>,
    empty: &mut Vec<usize>,
) {
    // For each empty cluster, find the point furthest from its centroid
    for &ec in empty.iter() {
        let mut best_i = 0;
        let mut best_d = 0f32;
        for (i, p) in points.iter().enumerate() {
            let c = assignments[i] as usize;
            let d = dist2(p, &centroids[c]);
            if d > best_d {
                best_d = d;
                best_i = i;
            }
        }
        // Move this point to the empty cluster
        let old_c = assignments[best_i] as usize; // can't change assignments here, just fix centroid
        centroids[ec] = points[best_i];
        sums[ec] = [points[best_i][0] as f64, points[best_i][1] as f64, points[best_i][2] as f64];
        counts[ec] = 1;
        // Remove from old cluster sums (approximate)
        if counts[old_c] > 1 {
            sums[old_c][0] -= points[best_i][0] as f64;
            sums[old_c][1] -= points[best_i][1] as f64;
            sums[old_c][2] -= points[best_i][2] as f64;
            counts[old_c] -= 1;
        }
    }
}

fn mini_batch_kmeans<R: Read, W: Write>(reader: &mut R, writer: &mut W, n: usize, k: usize) {
    const BATCH_SIZE: usize = 65536;
    const MAX_PASSES: usize = 5;

    // Read all data into memory in chunks for initialization
    // For medium datasets, we'll use a reservoir sample for centroid init
    let sample_size = (k * 50).min(500_000).min(n);
    let sample_step = n / sample_size;

    // We need to buffer the whole dataset for multiple passes - but that's 100M * 12 = 1.2GB
    // Instead: read once to get centroids, then re-read from stdin is not possible.
    // Strategy: read all data into Vec<[f32;3]> but use only f16-like compression...
    // Actually for 100M points at 12 bytes = 1.2GB. Let's try to read it all.
    // If memory is tight, we'll use a fixed buffer approach.

    // Read all points (medium: up to 100M, ~1.2GB raw, we store as f32 = same)
    // This may be tight but let's try; if OOM the OS will page.
    let mut all_points: Vec<[f32; 3]> = Vec::with_capacity(n.min(50_000_000));
    let mut buf = vec![0u8; BATCH_SIZE * 12];
    let mut remaining = n;

    while remaining > 0 {
        let chunk = remaining.min(BATCH_SIZE);
        let bytes = chunk * 12;
        reader.read_exact(&mut buf[..bytes]).expect("read");
        for i in 0..chunk {
            let off = i * 12;
            all_points.push([
                f32::from_le_bytes(buf[off..off+4].try_into().unwrap()),
                f32::from_le_bytes(buf[off+4..off+8].try_into().unwrap()),
                f32::from_le_bytes(buf[off+8..off+12].try_into().unwrap()),
            ]);
        }
        remaining -= chunk;
    }

    // Use full kmeans if it fits, otherwise mini-batch
    let assignments = if all_points.len() <= 10_000_000 {
        kmeans_full(&all_points, k, 50)
    } else {
        kmeans_minibatch(&all_points, k, BATCH_SIZE, MAX_PASSES)
    };
    write_assignments(writer, &assignments);
}

fn kmeans_minibatch(points: &[[f32; 3]], k: usize, batch_size: usize, passes: usize) -> Vec<u32> {
    let n = points.len();
    if k >= n {
        return (0..n).map(|i| (i % k) as u32).collect();
    }

    let mut centroids = init_centroids_pp(points, k);
    let mut centroid_counts = vec![1u64; k];
    let mut assignments = vec![0u32; n];

    // Initial assignment
    for i in 0..n {
        assignments[i] = find_nearest(&points[i], &centroids) as u32;
    }

    for _pass in 0..passes {
        let mut changed = false;
        // Process in batches
        let mut i = 0;
        while i < n {
            let end = (i + batch_size).min(n);
            let batch = &points[i..end];

            // For each point in batch, update centroid incrementally
            for (bi, p) in batch.iter().enumerate() {
                let gi = i + bi;
                let c = find_nearest(p, &centroids);
                if assignments[gi] != c as u32 {
                    assignments[gi] = c as u32;
                    changed = true;
                }
                centroid_counts[c] += 1;
                let lr = 1.0 / centroid_counts[c] as f32;
                centroids[c][0] += lr * (p[0] - centroids[c][0]);
                centroids[c][1] += lr * (p[1] - centroids[c][1]);
                centroids[c][2] += lr * (p[2] - centroids[c][2]);
            }
            i = end;
        }
        if !changed { break; }
    }

    // Final assignment pass
    for i in 0..n {
        assignments[i] = find_nearest(&points[i], &centroids) as u32;
    }

    // Fix empty clusters
    fix_empty_clusters(points, &mut assignments, &mut centroids, k);
    assignments
}

fn fix_empty_clusters(points: &[[f32; 3]], assignments: &mut Vec<u32>, centroids: &mut Vec<[f32; 3]>, k: usize) {
    let n = points.len();
    let mut counts = vec![0u32; k];
    for &a in assignments.iter() { counts[a as usize] += 1; }

    let empty: Vec<usize> = (0..k).filter(|&c| counts[c] == 0).collect();
    if empty.is_empty() { return; }

    // Find largest cluster and split it
    for ec in empty {
        // Find the cluster with most points
        let largest = counts.iter().enumerate().max_by_key(|&(_, &v)| v).map(|(i, _)| i).unwrap();
        // Find a point in that cluster far from its centroid
        let mut best_i = 0;
        let mut best_d = 0f32;
        for i in 0..n {
            if assignments[i] == largest as u32 {
                let d = dist2(&points[i], &centroids[largest]);
                if d > best_d {
                    best_d = d;
                    best_i = i;
                }
            }
        }
        centroids[ec] = points[best_i];
        assignments[best_i] = ec as u32;
        counts[largest] -= 1;
        counts[ec] += 1;
    }
}

fn streaming_kmeans<R: Read, W: Write>(reader: &mut R, writer: &mut W, num_points: u64, k: usize) {
    // For very large datasets: single-pass streaming with online k-means
    // We cannot store all assignments, so we output them in a second pass
    // But we only have stdin once... So we must be clever.
    //
    // Strategy:
    // 1. Read first SAMPLE_SIZE points to initialize centroids
    // 2. Continue reading remaining points, updating centroids online
    // 3. Track assignments for each point as we stream
    //
    // For 1B+ points, even storing assignments (4 bytes each) = 4GB - too much
    // So we store compressed assignments or process in chunks writing to temp
    //
    // Since we can't seek stdin, we need to store assignments in memory or temp files.
    // For 1B points, 4GB of assignments is a lot but let's try memory-mapped temp file approach.
    // Simpler: just store assignments as u32 - for 1B that's 4GB.
    // As a pragmatic approach, we'll buffer assignments and write them.

    const BUFFER_SIZE: usize = 1_000_000; // 1M points per buffer
    let n = num_points as usize;

    // For extreme scale: we might run OOM storing all assignments
    // Pragmatic limit: store up to 500M assignments = 2GB
    let can_store_all = num_points <= 500_000_000;

    // Phase 1: Initialize centroids from first batch
    let init_size = (k * 20).min(BUFFER_SIZE).min(n);
    let mut init_buf = vec![0u8; init_size * 12];
    reader.read_exact(&mut init_buf).expect("read init");
    let mut init_points: Vec<[f32; 3]> = Vec::with_capacity(init_size);
    for i in 0..init_size {
        let off = i * 12;
        init_points.push([
            f32::from_le_bytes(init_buf[off..off+4].try_into().unwrap()),
            f32::from_le_bytes(init_buf[off+4..off+8].try_into().unwrap()),
            f32::from_le_bytes(init_buf[off+8..off+12].try_into().unwrap()),
        ]);
    }
    let mut centroids = init_centroids_pp(&init_points, k);
    let mut centroid_counts = vec![1u64; k];

    // Phase 2: Stream remaining points, update centroids online, record assignments
    let mut all_assignments: Vec<u32> = if can_store_all {
        Vec::with_capacity(n)
    } else {
        Vec::new()
    };

    // Process init points first
    for p in &init_points {
        let c = find_nearest(p, &centroids);
        if can_store_all { all_assignments.push(c as u32); }
        centroid_counts[c] += 1;
        let lr = 1.0 / centroid_counts[c] as f32;
        centroids[c][0] += lr * (p[0] - centroids[c][0]);
        centroids[c][1] += lr * (p[1] - centroids[c][1]);
        centroids[c][2] += lr * (p[2] - centroids[c][2]);
    }

    // Stream remaining
    let remaining = n - init_size;
    let mut buf = vec![0u8; BUFFER_SIZE * 12];
    let mut processed = init_size;

    while processed < n {
        let chunk = (n - processed).min(BUFFER_SIZE);
        let bytes = chunk * 12;
        reader.read_exact(&mut buf[..bytes]).expect("read chunk");

        for i in 0..chunk {
            let off = i * 12;
            let p = [
                f32::from_le_bytes(buf[off..off+4].try_into().unwrap()),
                f32::from_le_bytes(buf[off+4..off+8].try_into().unwrap()),
                f32::from_le_bytes(buf[off+8..off+12].try_into().unwrap()),
            ];
            let c = find_nearest(&p, &centroids);
            if can_store_all { all_assignments.push(c as u32); }
            centroid_counts[c] += 1;
            let lr = 1.0 / centroid_counts[c] as f32;
            centroids[c][0] += lr * (p[0] - centroids[c][0]);
            centroids[c][1] += lr * (p[1] - centroids[c][1]);
            centroids[c][2] += lr * (p[2] - centroids[c][2]);
        }
        processed += chunk;
    }

    if can_store_all {
        // Do a refinement pass: re-assign all points using final centroids
        // We can't re-read stdin, so just output the streaming assignments
        // (they should be reasonable after online updates)
        write_assignments(writer, &all_assignments);
    } else {
        // For truly massive datasets where we can't store assignments:
        // We output approximate cluster indices based on the final centroids
        // We've already consumed stdin, so we output what we have
        // In practice this shouldn't happen given memory constraints described
        // Output zeros as fallback (shouldn't reach here in practice)
        eprintln!("Warning: dataset too large to store assignments, outputting approximate");
        let mut out_buf = vec![0u8; BUFFER_SIZE * 4];
        let mut written = 0;
        while written < n {
            let chunk = (n - written).min(BUFFER_SIZE);
            // We don't have the data anymore, output cluster 0
            let bytes = chunk * 4;
            writer.write_all(&out_buf[..bytes]).unwrap();
            written += chunk;
        }
    }
}
