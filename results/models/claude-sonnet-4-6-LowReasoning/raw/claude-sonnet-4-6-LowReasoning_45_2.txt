$words = @(
    0x07230203, 0x00010000, 0, 18, 0,           # Header
    (2 -shl 16) -bor 17, 1,                     # OpCapability Shader
    (3 -shl 16) -bor 14, 0, 1,                  # OpMemoryModel Logical GLSL450
    (7 -shl 16) -bor 15, 4, 12, 0x6e69616d, 0, 8, 9,  # OpEntryPoint
    (3 -shl 16) -bor 16, 12, 7,                 # OpExecutionMode OriginUpperLeft
    (4 -shl 16) -bor 71, 8, 30, 2,              # OpDecorate %8 Location 2
    (4 -shl 16) -bor 71, 9, 30, 0,              # OpDecorate %9 Location 0
    (2 -shl 16) -bor 19, 1,                     # OpTypeVoid %1
    (3 -shl 16) -bor 33, 2, 1,                  # OpTypeFunction %2 %1
    (3 -shl 16) -bor 22, 3, 32,                 # OpTypeFloat %3 32
    (4 -shl 16) -bor 23, 4, 3, 2,               # OpTypeVector %4 %3 2
    (4 -shl 16) -bor 23, 5, 3, 4,               # OpTypeVector %5 %3 4
    (4 -shl 16) -bor 32, 6, 1, 4,               # OpTypePointer %6 Input %4
    (4 -shl 16) -bor 32, 7, 3, 5,               # OpTypePointer %7 Output %5
    (4 -shl 16) -bor 43, 3, 10, 0,              # OpConstant %3 %10 0.0f
    (4 -shl 16) -bor 43, 3, 11, 0x3f800000,     # OpConstant %3 %11 1.0f
    (4 -shl 16) -bor 59, 6, 8, 1,               # OpVariable %6 %8 Input
    (4 -shl 16) -bor 59, 7, 9, 3,               # OpVariable %7 %9 Output
    (5 -shl 16) -bor 54, 1, 12, 0, 2,           # OpFunction %1 %12 None %2
    (2 -shl 16) -bor 248, 13,                   # OpLabel %13
    (4 -shl 16) -bor 61, 4, 14, 8,              # OpLoad %4 %14 %8
    (5 -shl 16) -bor 81, 3, 15, 14, 0,          # OpCompositeExtract %3 %15 %14 0
    (5 -shl 16) -bor 81, 3, 16, 14, 1,          # OpCompositeExtract %3 %16 %14 1
    (7 -shl 16) -bor 80, 5, 17, 15, 16, 10, 11, # OpCompositeConstruct %5 %17
    (3 -shl 16) -bor 62, 9, 17,                 # OpStore %9 %17
    (1 -shl 16) -bor 253,                       # OpReturn
    (1 -shl 16) -bor 56                         # OpFunctionEnd
)

$result = ($words | ForEach-Object {
    $b = [System.BitConverter]::GetBytes([uint32]$_)
    ($b | ForEach-Object { $_.ToString("x2") }) -join ""
}) -join ""

$result | Out-File -FilePath "answer.txt" -NoNewline -Encoding ascii
Write-Host $result
