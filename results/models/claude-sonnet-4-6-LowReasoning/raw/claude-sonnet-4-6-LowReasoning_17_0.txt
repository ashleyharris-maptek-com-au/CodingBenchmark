using System;
using System.Collections.Generic;
using System.Linq;

class BinPacking3D
{
    struct Box
    {
        public int W, H, D;
        public Box(int w, int h, int d) { W = w; H = h; D = d; }
    }

    struct Placement
    {
        public int Index, X, Y, Z, W, H, D;
    }

    static int ContW, ContH, ContD;
    static List<Placement> Placed = new List<Placement>();

    // Voxel grid for fast overlap detection - used for small containers
    static bool[,,] Grid;
    static bool UseGrid;
    static long GridCells;

    static void Main()
    {
        var firstLine = Console.ReadLine().Trim().Split();
        int N = int.Parse(firstLine[0]);
        ContW = int.Parse(firstLine[1]);
        ContH = int.Parse(firstLine[2]);
        ContD = int.Parse(firstLine[3]);

        var boxes = new Box[N];
        for (int i = 0; i < N; i++)
        {
            var parts = Console.ReadLine().Trim().Split();
            boxes[i] = new Box(int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2]));
        }

        GridCells = (long)ContW * ContH * ContD;
        UseGrid = GridCells <= 4_000_000L;

        if (UseGrid)
            Grid = new bool[ContW, ContH, ContD];

        // Sort boxes by volume descending
        var order = Enumerable.Range(0, N)
            .OrderByDescending(i => (long)boxes[i].W * boxes[i].H * boxes[i].D)
            .ToArray();

        // Extreme points set
        var extremePoints = new HashSet<(int, int, int)>();
        extremePoints.Add((0, 0, 0));

        foreach (int idx in order)
        {
            var box = boxes[idx];
            var rotations = GetRotations(box);

            bool placed = false;
            Placement bestPlacement = default;
            int bestScore = int.MaxValue;

            // Try each extreme point with each rotation
            var epList = extremePoints.ToList();
            // Sort EPs: prefer lower z, then y, then x (bottom-back-left first)
            epList.Sort((a, b) => {
                int c = a.Item3.CompareTo(b.Item3);
                if (c != 0) return c;
                c = a.Item2.CompareTo(b.Item2);
                if (c != 0) return c;
                return a.Item1.CompareTo(b.Item1);
            });

            foreach (var ep in epList)
            {
                int ex = ep.Item1, ey = ep.Item2, ez = ep.Item3;
                foreach (var rot in rotations)
                {
                    int w = rot.W, h = rot.H, d = rot.D;
                    if (ex + w > ContW || ey + h > ContH || ez + d > ContD)
                        continue;

                    if (CanPlace(ex, ey, ez, w, h, d))
                    {
                        // Score: prefer placements that waste less space (lower z*y*x composite)
                        int score = ez * 10000 + ey * 100 + ex;
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestPlacement = new Placement { Index = idx, X = ex, Y = ey, Z = ez, W = w, H = h, D = d };
                            placed = true;
                        }
                        break; // Take first valid rotation for this EP
                    }
                }
                if (placed && bestScore == 0) break;
            }

            if (placed)
            {
                PlaceBox(bestPlacement);
                Placed.Add(bestPlacement);
                AddExtremePoints(extremePoints, bestPlacement);
            }
        }

        Console.WriteLine(Placed.Count);
        foreach (var p in Placed)
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
    }

    static IEnumerable<Box> GetRotations(Box b)
    {
        var seen = new HashSet<(int,int,int)>();
        int[] dims = { b.W, b.H, b.D };
        // All 6 permutations
        int[] idx = { 0, 1, 2 };
        var perms = new[] {
            (dims[0],dims[1],dims[2]),
            (dims[0],dims[2],dims[1]),
            (dims[1],dims[0],dims[2]),
            (dims[1],dims[2],dims[0]),
            (dims[2],dims[0],dims[1]),
            (dims[2],dims[1],dims[0]),
        };
        foreach (var (w,h,d) in perms)
        {
            if (seen.Add((w,h,d)))
                yield return new Box(w, h, d);
        }
    }

    static bool CanPlace(int x, int y, int z, int w, int h, int d)
    {
        if (UseGrid)
            return CanPlaceGrid(x, y, z, w, h, d);
        else
            return CanPlaceLinear(x, y, z, w, h, d);
    }

    static bool CanPlaceGrid(int x, int y, int z, int w, int h, int d)
    {
        // Check boundary
        if (x + w > ContW || y + h > ContH || z + d > ContD) return false;
        // Check any occupied cell
        for (int dz = z; dz < z + d; dz++)
        for (int dy = y; dy < y + h; dy++)
        for (int dx = x; dx < x + w; dx++)
        {
            if (Grid[dx, dy, dz]) return false;
        }
        return true;
    }

    static bool CanPlaceLinear(int x, int y, int z, int w, int h, int d)
    {
        foreach (var p in Placed)
        {
            if (Overlaps(x, y, z, w, h, d, p.X, p.Y, p.Z, p.W, p.H, p.D))
                return false;
        }
        return true;
    }

    static bool Overlaps(int ax, int ay, int az, int aw, int ah, int ad,
                          int bx, int by, int bz, int bw, int bh, int bd)
    {
        return ax < bx + bw && ax + aw > bx &&
               ay < by + bh && ay + ah > by &&
               az < bz + bd && az + ad > bz;
    }

    static void PlaceBox(Placement p)
    {
        if (!UseGrid) return;
        for (int dz = p.Z; dz < p.Z + p.D; dz++)
        for (int dy = p.Y; dy < p.Y + p.H; dy++)
        for (int dx = p.X; dx < p.X + p.W; dx++)
            Grid[dx, dy, dz] = true;
    }

    static void AddExtremePoints(HashSet<(int,int,int)> eps, Placement p)
    {
        int x1 = p.X + p.W;
        int y1 = p.Y + p.H;
        int z1 = p.Z + p.D;

        // Three new candidate extreme points
        if (x1 <= ContW) eps.Add((x1, p.Y, p.Z));
        if (y1 <= ContH) eps.Add((p.X, y1, p.Z));
        if (z1 <= ContD) eps.Add((p.X, p.Y, z1));

        // Project existing extreme points against new box faces
        // Project along X: if ep.x == p.X + p.W check y/z inside box
        // Simpler: add projections of box faces onto existing EPs
        var toAdd = new List<(int,int,int)>();
        foreach (var ep in eps)
        {
            int ex = ep.Item1, ey = ep.Item2, ez = ep.Item3;
            // Project EP onto face x=p.X+p.W if within box y,z range
            if (ey >= p.Y && ey < y1 && ez >= p.Z && ez < z1)
                toAdd.Add((x1, ey, ez));
            if (ex >= p.X && ex < x1 && ez >= p.Z && ez < z1)
                toAdd.Add((ex, y1, ez));
            if (ex >= p.X && ex < x1 && ey >= p.Y && ey < y1)
                toAdd.Add((ex, ey, z1));
        }
        foreach (var pt in toAdd)
        {
            if (pt.Item1 <= ContW && pt.Item2 <= ContH && pt.Item3 <= ContD)
                eps.Add(pt);
        }

        // Prune dominated extreme points (optional, keep set manageable)
        if (eps.Count > 5000)
        {
            // Remove points that are completely inside a placed box
            var pruned = new HashSet<(int,int,int)>();
            foreach (var ep in eps)
            {
                bool dominated = false;
                foreach (var pl in Placed)
                {
                    if (ep.Item1 >= pl.X && ep.Item1 < pl.X + pl.W &&
                        ep.Item2 >= pl.Y && ep.Item2 < pl.Y + pl.H &&
                        ep.Item3 >= pl.Z && ep.Item3 < pl.Z + pl.D)
                    {
                        dominated = true;
                        break;
                    }
                }
                if (!dominated) pruned.Add(ep);
            }
            eps.Clear();
            foreach (var ep in pruned) eps.Add(ep);
        }
    }
}
