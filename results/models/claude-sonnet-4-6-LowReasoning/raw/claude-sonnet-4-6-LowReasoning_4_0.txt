using namespace std;

static const double EDGE = 1.0;
static const double HEIGHT = EDGE * sqrt(2.0/3.0);
static const double CIRCUMRADIUS = EDGE * sqrt(3.0/8.0);
static const double INRADIUS = EDGE / sqrt(24.0);

struct Vec3 {
    double x, y, z;
    Vec3(double x=0, double y=0, double z=0): x(x), y(y), z(z) {}
    Vec3 operator+(const Vec3& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    Vec3 operator/(double s) const { return {x/s, y/s, z/s}; }
    double dot(const Vec3& o) const { return x*o.x + y*o.y + z*o.z; }
    Vec3 cross(const Vec3& o) const {
        return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x};
    }
    double norm() const { return sqrt(x*x + y*y + z*z); }
    Vec3 normalized() const { double n = norm(); return n > 1e-12 ? *this/n : Vec3(0,0,0); }
};

struct Plane {
    Vec3 n;
    double d;
};

vector<Plane> buildPlanes(const vector<Vec3>& verts) {
    vector<Plane> planes;
    int n = verts.size();

    Vec3 centroid(0,0,0);
    for (auto& v : verts) { centroid = centroid + v; }
    centroid = centroid / n;

    for (int i = 0; i < n; i++)
    for (int j = i+1; j < n; j++)
    for (int k = j+1; k < n; k++) {
        Vec3 a = verts[i], b = verts[j], c = verts[k];
        Vec3 normal = (b-a).cross(c-a);
        if (normal.norm() < 1e-10) continue;
        normal = normal.normalized();
        double d = normal.dot(a);

        bool allAbove = true, allBelow = true;
        for (int l = 0; l < n; l++) {
            double side = normal.dot(verts[l]) - d;
            if (side < -1e-9) allAbove = false;
            if (side > 1e-9) allBelow = false;
        }

        if (!allAbove && !allBelow) continue;

        double centerSide = normal.dot(centroid) - d;
        if (allAbove) {
            if (centerSide < 0) { normal = normal * -1; d = -d; }
        } else {
            if (centerSide > 0) { normal = normal * -1; d = -d; }
        }

        bool dup = false;
        for (auto& p : planes) {
            if (fabs(p.n.dot(normal) - 1.0) < 1e-6 && fabs(p.d - d) < 1e-6) {
                dup = true; break;
            }
        }
        if (!dup) {
            planes.push_back({normal, d});
        }
    }
    return planes;
}

double distToPlane(const Plane& pl, const Vec3& p) {
    return pl.n.dot(p) - pl.d;
}

array<Vec3,4> tetraVertices(const Vec3& center, double rx, double ry, double rz, double scale) {
    double ir = INRADIUS * scale;
    double h = HEIGHT * scale;
    double bR = scale / sqrt(3.0);

    array<Vec3,4> verts;
    for (int i = 0; i < 3; i++) {
        double angle = i * 2.0 * M_PI / 3.0;
        verts[i] = Vec3(bR * cos(angle), bR * sin(angle), -ir);
    }
    verts[3] = Vec3(0, 0, h - ir);

    double cx = cos(rx), sx = sin(rx);
    double cy = cos(ry), sy = sin(ry);
    double cz = cos(rz), sz = sin(rz);

    double R00 = cy*cz, R01 = cy*sz, R02 = -sy;
    double R10 = sx*sy*cz - cx*sz, R11 = sx*sy*sz + cx*cz, R12 = sx*cy;
    double R20 = cx*sy*cz + sx*sz, R21 = cx*sy*sz - sx*cz, R22 = cx*cy;

    for (auto& v : verts) {
        double nx = R00*v.x + R01*v.y + R02*v.z;
        double ny = R10*v.x + R11*v.y + R12*v.z;
        double nz = R20*v.x + R21*v.y + R22*v.z;
        v = center + Vec3(nx, ny, nz);
    }
    return verts;
}

bool tetraInsideConvex(const vector<Plane>& planes, const Vec3& center, double rx, double ry, double rz, double scale) {
    auto verts = tetraVertices(center, rx, ry, rz, scale);
    for (auto& v : verts) {
        for (auto& pl : planes) {
            if (distToPlane(pl, v) < -1e-9) return false;
        }
    }
    return true;
}

bool tetraOverlap(const array<Vec3,4>& A, const array<Vec3,4>& B) {
    auto project = [](const array<Vec3,4>& T, const Vec3& axis) -> pair<double,double> {
        double mn = 1e18, mx = -1e18;
        for (auto& v : T) {
            double p = axis.dot(v);
            mn = min(mn, p); mx = max(mx, p);
        }
        return {mn, mx};
    };

    auto separated = [&](const Vec3& axis) -> bool {
        if (axis.norm() < 1e-10) return false;
        Vec3 ax = axis.normalized();
        auto [a0, a1] = project(A, ax);
        auto [b0, b1] = project(B, ax);
        return a1 < b0 - 1e-9 || b1 < a0 - 1e-9;
    };

    int f[4][3] = {{0,1,2},{0,1,3},{0,2,3},{1,2,3}};
    for (auto& fi : f) {
        if (separated((A[fi[1]]-A[fi[0]]).cross(A[fi[2]]-A[fi[0]]))) return false;
    }
    for (auto& fi : f) {
        if (separated((B[fi[1]]-B[fi[0]]).cross(B[fi[2]]-B[fi[0]]))) return false;
    }
    int pairs[6][2] = {{0,1},{0,2},{0,3},{1,2},{1,3},{2,3}};
    Vec3 edgesA[6], edgesB[6];
    for (int i = 0; i < 6; i++) {
        edgesA[i] = A[pairs[i][1]] - A[pairs[i][0]];
        edgesB[i] = B[pairs[i][1]] - B[pairs[i][0]];
    }
    for (int i = 0; i < 6; i++)
    for (int j = 0; j < 6; j++) {
        if (separated(edgesA[i].cross(edgesB[j]))) return false;
    }
    return true;
}

struct Tetra {
    Vec3 center;
    double rx, ry, rz;
    array<Vec3,4> verts;
    Vec3 aabbMin, aabbMax;
};

Tetra makeTetra(const Vec3& c, double rx, double ry, double rz, double scale) {
    Tetra t;
    t.center = c; t.rx = rx; t.ry = ry; t.rz = rz;
    t.verts = tetraVertices(c, rx, ry, rz, scale);
    t.aabbMin = t.aabbMax = t.verts[0];
    for (int i = 1; i < 4; i++) {
        t.aabbMin.x = min(t.aabbMin.x, t.verts[i].x);
        t.aabbMin.y = min(t.aabbMin.y, t.verts[i].y);
        t.aabbMin.z = min(t.aabbMin.z, t.verts[i].z);
        t.aabbMax.x = max(t.aabbMax.x, t.verts[i].x);
        t.aabbMax.y = max(t.aabbMax.y, t.verts[i].y);
        t.aabbMax.z = max(t.aabbMax.z, t.verts[i].z);
    }
    return t;
}

bool aabbOverlap(const Tetra& a, const Tetra& b) {
    return !(a.aabbMax.x < b.aabbMin.x - 1e-9 || b.aabbMax.x < a.aabbMin.x - 1e-9 ||
             a.aabbMax.y < b.aabbMin.y - 1e-9 || b.aabbMax.y < a.aabbMin.y - 1e-9 ||
             a.aabbMax.z < b.aabbMin.z - 1e-9 || b.aabbMax.z < a.aabbMin.z - 1e-9);
}

bool overlapsAny(const vector<Tetra>& placed, const Tetra& t) {
    for (auto& p : placed) {
        if (!aabbOverlap(p, t)) continue;
        if (tetraOverlap(p.verts, t.verts)) return true;
    }
    return false;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    auto startTime = chrono::steady_clock::now();
    auto elapsed = [&]() {
        return chrono::duration<double>(chrono::steady_clock::now() - startTime).count();
    };

    int V;
    double edgeLen;
    cin >> V >> edgeLen;

    vector<Vec3> verts(V);
    for (int i = 0; i < V; i++) cin >> verts[i].x >> verts[i].y >> verts[i].z;

    double scale = edgeLen;

    auto planes = buildPlanes(verts);

    Vec3 boxMin = verts[0], boxMax = verts[0];
    for (auto& v : verts) {
        boxMin.x = min(boxMin.x, v.x); boxMin.y = min(boxMin.y, v.y); boxMin.z = min(boxMin.z, v.z);
        boxMax.x = max(boxMax.x, v.x); boxMax.y = max(boxMax.y, v.y); boxMax.z = max(boxMax.z, v.z);
    }

    vector<Tetra> placed;

    double dx = scale * 1.0;
    double dy = scale * sqrt(3.0) / 2.0;
    double dz = scale * sqrt(2.0/3.0);

    // Grid step slightly smaller than tetra spacing for denser packing
    double gx = dx * 0.88;
    double gy = dy * 0.88;
    double gz = dz * 0.88;

    double timeLimit = 25.0;

    vector<array<double,3>> orientations = {
        {0, 0, 0},
        {M_PI, 0, 0},
        {0, M_PI, 0},
        {M_PI/2, M_PI/4, 0},
        {M_PI/4, 0, M_PI/2},
        {0, M_PI/4, M_PI/4},
        {M_PI*2/3, M_PI/4, 0},
        {M_PI/3, M_PI/3, M_PI/3},
    };

    double margin = CIRCUMRADIUS * scale;

    int nx = (int)ceil((boxMax.x - boxMin.x + 2*margin) / gx) + 1;
    int ny = (int)ceil((boxMax.y - boxMin.y + 2*margin) / gy) + 1;
    int nz = (int)ceil((boxMax.z - boxMin.z + 2*margin) / gz) + 1;

    for (int iz = 0; iz < nz && elapsed() < timeLimit; iz++) {
        double z = boxMin.z - margin + iz * gz;
        for (int iy = 0; iy < ny && elapsed() < timeLimit; iy++) {
            double y = boxMin.y - margin + iy * gy + (iz % 2) * gy * 0.5;
            for (int ix = 0; ix < nx && elapsed() < timeLimit; ix++) {
                double x = boxMin.x - margin + ix * gx + ((iy+iz) % 2) * gx * 0.5;
                Vec3 center(x, y, z);

                for (auto& ori : orientations) {
                    if (!tetraInsideConvex(planes, center, ori[0], ori[1], ori[2], scale)) continue;
                    Tetra t = makeTetra(center, ori[0], ori[1], ori[2], scale);
                    if (!overlapsAny(placed, t)) {
                        placed.push_back(t);
                        break;
                    }
                }
            }
        }
    }

    // Random improvement
    mt19937 rng(42);
    uniform_real_distribution<double> xDist(boxMin.x, boxMax.x);
    uniform_real_distribution<double> yDist(boxMin.y, boxMax.y);
    uniform_real_distribution<double> zDist(boxMin.z, boxMax.z);
    uniform_real_distribution<double> angDist(0, 2*M_PI);

    while (elapsed() < timeLimit - 0.5) {
        Vec3 center(xDist(rng), yDist(rng), zDist(rng));
        double rx = angDist(rng), ry = angDist(rng), rz = angDist(rng);
        if (!tetraInsideConvex(planes, center, rx, ry, rz, scale)) continue;
        Tetra t = makeTetra(center, rx, ry, rz, scale);
        if (!overlapsAny(placed, t)) {
            placed.push_back(t);
        }
    }

    cout << placed.size() << "\n";
    cout.precision(10);
    for (auto& t : placed) {
        cout << fixed << t.center.x << " " << t.center.y << " " << t.center.z << " "
             << t.rx << " " << t.ry << " " << t.rz << "\n";
    }

    return 0;
}
