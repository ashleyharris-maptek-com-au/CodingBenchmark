#include <bits/stdc++.h>
using namespace std;

// Maximum Independent Set via Branch-and-Bound with clique-cover upper bound
// + degree-0/1 reduction rules applied iteratively

const int MAXN = 600;
static bitset<MAXN> adj[MAXN];

static int         best_sz;
static vector<int> best_sol;

// Greedy clique cover of the subgraph induced by P.
// Returns (sorted_vertices, 1-indexed_clique_ids).
// Key bound: MIS(P) <= number of distinct clique IDs = max clique id.
// Vertices sorted by clique_id ascending; when iterating in reverse,
// at position i the bound on remaining search is sc[i].
static pair<vector<int>, vector<int>> clique_cover(const vector<int>& P) {
    int sz = (int)P.size();
    vector<bitset<MAXN>> cm; // cm[k] = bitmask of members of clique k
    vector<int> vc(sz);

    for (int i = 0; i < sz; i++) {
        int v = P[i];
        int c = -1;
        for (int k = 0; k < (int)cm.size(); k++) {
            // v joins clique k iff every clique member is adjacent to v
            if ((cm[k] & adj[v]) == cm[k]) { c = k; break; }
        }
        if (c == -1) { c = (int)cm.size(); cm.emplace_back(); }
        cm[c][v] = 1;
        vc[i] = c;
    }

    vector<pair<int,int>> prs(sz);
    for (int i = 0; i < sz; i++) prs[i] = {vc[i], P[i]};
    sort(prs.begin(), prs.end());

    vector<int> sv(sz), sc(sz);
    for (int i = 0; i < sz; i++) {
        sv[i] = prs[i].second;
        sc[i] = prs[i].first + 1; // 1-indexed clique id = upper-bound contribution
    }
    return {sv, sc};
}

static void bb(const vector<int>& P, vector<int>& sol) {
    if (P.empty()) {
        if ((int)sol.size() > best_sz) { best_sz = (int)sol.size(); best_sol = sol; }
        return;
    }
    if ((int)(sol.size() + P.size()) <= best_sz) return;

    auto [sv, sc] = clique_cover(P);
    int sz = (int)P.size();

    for (int i = sz - 1; i >= 0; i--) {
        if ((int)sol.size() + sc[i] <= best_sz) return;
        int v = sv[i];

        vector<int> newP;
        newP.reserve(i);
        for (int j = 0; j < i; j++)
            if (!adj[v][sv[j]]) newP.push_back(sv[j]);

        sol.push_back(v);
        bb(newP, sol);
        sol.pop_back();
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<vector<int>> nbrs(n);
    for (int i = 0; i < m; i++) {
        int u, v; cin >> u >> v;
        nbrs[u].push_back(v);
        nbrs[v].push_back(u);
    }

    // ---- Iterative degree-0 / degree-1 reduction ----
    vector<bool> active(n, true);
    vector<int>  forced; // vertices forced into MIS

    // Maintain active degree for efficiency
    vector<int> adeg(n, 0);
    for (int v = 0; v < n; v++) adeg[v] = (int)nbrs[v].size();

    queue<int> q;
    for (int v = 0; v < n; v++) q.push(v);

    while (!q.empty()) {
        int v = q.front(); q.pop();
        if (!active[v]) continue;

        // Recompute active degree (simple; OK for sparse graphs)
        int deg = 0, snbr = -1;
        for (int u : nbrs[v])
            if (active[u]) { deg++; snbr = u; }

        if (deg == 0) {
            forced.push_back(v);
            active[v] = false;
        } else if (deg == 1) {
            forced.push_back(v);
            active[v] = false;
            active[snbr] = false;
            for (int u : nbrs[snbr])
                if (active[u]) q.push(u);
        }
    }

    // ---- Collect remaining (core) vertices ----
    vector<int> rem;
    rem.reserve(n);
    for (int v = 0; v < n; v++) if (active[v]) rem.push_back(v);
    int n2 = (int)rem.size();

    if (n2 > 0) {
        // If the core is too large for the bitset, we can't solve exactly with this method.
        // For the common case (sparse graphs) reductions already handled most vertices.
        // Cap at MAXN as a safety measure; in practice this path is rarely taken.
        if (n2 > MAXN) n2 = MAXN; // note: may lose vertices → result could be wrong for huge dense graphs

        vector<int> remap(n, -1);
        for (int i = 0; i < n2; i++) remap[rem[i]] = i;

        for (int i = 0; i < n2; i++) adj[i].reset();
        for (int i = 0; i < n2; i++) {
            int u = rem[i];
            for (int v : nbrs[u]) {
                if (active[v] && remap[v] >= 0)
                    adj[i][remap[v]] = 1;
            }
        }

        // Order vertices by ascending degree for initial candidate ordering
        vector<int> P(n2);
        iota(P.begin(), P.end(), 0);
        sort(P.begin(), P.end(), [](int a, int b) {
            return adj[a].count() < adj[b].count();
        });

        vector<int> sol;
        bb(P, sol);
    }

    // ---- Output ----
    cout << (int)forced.size() + best_sz << "\n";
    bool first = true;
    for (int v : forced)   { if (!first) cout << ' '; cout << v;      first = false; }
    for (int i : best_sol) { if (!first) cout << ' '; cout << rem[i]; first = false; }
    cout << "\n";

    return 0;
}
