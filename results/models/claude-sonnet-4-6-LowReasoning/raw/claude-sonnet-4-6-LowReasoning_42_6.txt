; SPIR-V passthrough geometry shader
; Input: Triangles, Output: TriangleStrip, 3 vertices

               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; User-defined input location 2 = color
               OpDecorate %in_color Location 2

; User-defined output location 0 = outColor
               OpDecorate %out_color Location 0

; Types
      %void = OpTypeVoid
      %bool = OpTypeBool
     %float = OpTypeFloat 32
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4
      %uint = OpTypeInt 32 0
       %int = OpTypeInt 32 1

; Constants
    %uint_3 = OpConstant %uint 3
     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %int_2 = OpConstant %int 2

; Input gl_PerVertex struct and array
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float
     %gl_in = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
    %gl_out = OpVariable %ptr_out_pv Output

; Input color array (location 2)
%arr_v3float = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float
%ptr_in_v3float = OpTypePointer Input %v3float
  %in_color = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3float = OpTypePointer Output %v3float
 %out_color = OpVariable %ptr_out_v3float Output

; Main function type
  %fn_void = OpTypeFunction %void

      %main = OpFunction %void None %fn_void
    %entry = OpLabel

; --- Vertex 0 ---
%ptr_pos_in_0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
     %pos_0 = OpLoad %v4float %ptr_pos_in_0
%ptr_pos_out = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %ptr_pos_out %pos_0

%ptr_col_in_0 = OpAccessChain %ptr_in_v3float %in_color %int_0
     %col_0 = OpLoad %v3float %ptr_col_in_0
               OpStore %out_color %col_0
               OpEmitVertex

; --- Vertex 1 ---
%ptr_pos_in_1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
     %pos_1 = OpLoad %v4float %ptr_pos_in_1
%ptr_pos_out1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %ptr_pos_out1 %pos_1

%ptr_col_in_1 = OpAccessChain %ptr_in_v3float %in_color %int_1
     %col_1 = OpLoad %v3float %ptr_col_in_1
               OpStore %out_color %col_1
               OpEmitVertex

; --- Vertex 2 ---
%ptr_pos_in_2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
     %pos_2 = OpLoad %v4float %ptr_pos_in_2
%ptr_pos_out2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %ptr_pos_out2 %pos_2

%ptr_col_in_2 = OpAccessChain %ptr_in_v3float %in_color %int_2
     %col_2 = OpLoad %v3float %ptr_col_in_2
               OpStore %out_color %col_2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
