import sys
import random
from array import array

def main():
    data = sys.stdin.buffer.read().split()
    pos = 0
    n = int(data[pos]); pos += 1
    m = int(data[pos]); pos += 1

    # Store all literals flat (int32)
    lits = array('i', (int(x) for x in data[pos:pos + m * 3]))

    # Initialize assignment: set each var to favor its more-common polarity
    pos_count = array('i', [0] * (n + 1))
    neg_count = array('i', [0] * (n + 1))
    for i in range(m * 3):
        lit = lits[i]
        if lit > 0:
            pos_count[lit] += 1
        else:
            neg_count[-lit] += 1

    assignment = bytearray(n + 1)
    for v in range(1, n + 1):
        assignment[v] = 1 if pos_count[v] >= neg_count[v] else 0

    # Compute initial sat_count per clause
    sat_count = array('b', [0] * m)
    for i in range(m):
        cnt = 0
        base = i * 3
        for j in range(3):
            lit = lits[base + j]
            v = abs(lit)
            if (lit > 0) == bool(assignment[v]):
                cnt += 1
        sat_count[i] = cnt

    # Build unsat list with O(1) random pick + O(1) add/remove
    unsat_list = []
    unsat_pos = {}
    for i in range(m):
        if sat_count[i] == 0:
            unsat_pos[i] = len(unsat_list)
            unsat_list.append(i)

    if not unsat_list:
        _output(n, assignment)
        return

    # Build CSR var->clauses index
    vc_count = array('i', [0] * (n + 1))
    for i in range(m * 3):
        vc_count[abs(lits[i])] += 1

    vc_offset = array('i', [0] * (n + 2))
    for v in range(1, n + 1):
        vc_offset[v + 1] = vc_offset[v] + vc_count[v]

    vc_data = array('i', [0] * vc_offset[n + 1])
    vc_fill = array('i', [0] * (n + 1))
    for i in range(m):
        base = i * 3
        for j in range(3):
            v = abs(lits[base + j])
            vc_data[vc_offset[v] + vc_fill[v]] = i
            vc_fill[v] += 1

    # WalkSAT
    p_noise = 0.57
    max_flips = min(5 * m + 10 * n + 1000000, 50000000)

    for _ in range(max_flips):
        if not unsat_list:
            break

        # Pick random unsatisfied clause
        clause_idx = unsat_list[random.randrange(len(unsat_list))]
        base = clause_idx * 3

        if random.random() < p_noise:
            # Noise: flip a random variable from the clause
            flip_var = abs(lits[base + random.randrange(3)])
        else:
            # Greedy: flip the variable with minimum "break count"
            best_var = -1
            best_break = 10**9
            for j in range(3):
                v = abs(lits[base + j])
                val = assignment[v]
                brk = 0
                start = vc_offset[v]
                end = vc_offset[v + 1]
                for k in range(start, end):
                    ci = vc_data[k]
                    if sat_count[ci] == 1:
                        # check if v is the sole satisfying literal
                        cb = ci * 3
                        for jj in range(3):
                            lit2 = lits[cb + jj]
                            if abs(lit2) == v:
                                if (lit2 > 0) == bool(val):
                                    brk += 1
                                break
                if brk < best_break:
                    best_break = brk
                    best_var = v
                    if brk == 0:
                        break
            flip_var = best_var

        # Flip flip_var and update sat_count / unsat structures
        old_val = assignment[flip_var]
        new_val = 1 - old_val
        assignment[flip_var] = new_val

        start = vc_offset[flip_var]
        end = vc_offset[flip_var + 1]
        for k in range(start, end):
            ci = vc_data[k]
            cb = ci * 3
            for jj in range(3):
                lit2 = lits[cb + jj]
                if abs(lit2) == flip_var:
                    was_sat = (lit2 > 0) == bool(old_val)
                    now_sat = (lit2 > 0) == bool(new_val)
                    if was_sat and not now_sat:
                        sat_count[ci] -= 1
                        if sat_count[ci] == 0:
                            unsat_pos[ci] = len(unsat_list)
                            unsat_list.append(ci)
                    elif not was_sat and now_sat:
                        sat_count[ci] += 1
                        if sat_count[ci] == 1:
                            p = unsat_pos.pop(ci)
                            last = unsat_list[-1]
                            unsat_list[p] = last
                            unsat_pos[last] = p
                            unsat_list.pop()
                    break

    _output(n, assignment)


def _output(n, assignment):
    out = sys.stdout
    out.write("SAT\n")
    out.write(" ".join(str(assignment[i]) for i in range(1, n + 1)))
    out.write("\n")


main()
