OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

               ; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

               ; Input color location
               OpDecorate %in_color Location 2

               ; Output color location
               OpDecorate %out_color Location 0

               ; Types
       %void = OpTypeVoid
      %voidfn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

      %uint_1 = OpConstant %uint 1
      %int_0  = OpConstant %int 0
      %int_1  = OpConstant %int 1

    %float_0    = OpConstant %float 0.0
    %float_1    = OpConstant %float 1.0
    %float_h    = OpConstant %float 0.04

; Input gl_PerVertex struct and array (size 1 for InputPoints)
%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%ptr_in_v4float   = OpTypePointer Input %v4float
%gl_in            = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float  = OpTypePointer Output %v4float
%gl_out           = OpVariable %ptr_out_pv Output

; Input color array (location 2, array size 1)
%arr_v3float_1    = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3float_1
%ptr_in_v3float   = OpTypePointer Input %v3float
%in_color         = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3float  = OpTypePointer Output %v3float
%out_color        = OpVariable %ptr_out_v3float Output

       %main = OpFunction %void None %voidfn
      %entry = OpLabel

               ; Load gl_in[0].gl_Position
  %pos_ptr   = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
  %pos       = OpLoad %v4float %pos_ptr

               ; Extract x, y
  %x         = OpCompositeExtract %float %pos 0
  %y         = OpCompositeExtract %float %pos 1

               ; Load color[0]
  %col_ptr   = OpAccessChain %ptr_in_v3float %in_color %int_0
  %color     = OpLoad %v3float %col_ptr

               ; Compute corner offsets
  %xmh       = OpFSub %float %x %float_h
  %xph       = OpFAdd %float %x %float_h
  %ymh       = OpFSub %float %y %float_h
  %yph       = OpFAdd %float %y %float_h

               ; Vertex 0: (x-h, y-h, 0, 1)
  %v0        = OpCompositeConstruct %v4float %xmh %ymh %float_0 %float_1
  %gp0       = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %gp0 %v0
               OpStore %out_color %color
               OpEmitVertex

               ; Vertex 1: (x+h, y-h, 0, 1)
  %v1        = OpCompositeConstruct %v4float %xph %ymh %float_0 %float_1
  %gp1       = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %gp1 %v1
               OpStore %out_color %color
               OpEmitVertex

               ; Vertex 2: (x-h, y+h, 0, 1)
  %v2        = OpCompositeConstruct %v4float %xmh %yph %float_0 %float_1
  %gp2       = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %gp2 %v2
               OpStore %out_color %color
               OpEmitVertex

               ; Vertex 3: (x+h, y+h, 0, 1)
  %v3        = OpCompositeConstruct %v4float %xph %yph %float_0 %float_1
  %gp3       = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %gp3 %v3
               OpStore %out_color %color
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
