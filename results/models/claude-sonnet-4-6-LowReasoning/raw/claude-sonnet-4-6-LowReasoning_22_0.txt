using System;
using System.Collections.Generic;
using System.Text;

class TetrisAI
{
    static int width, height;
    static bool[,] board;

    // Piece definitions: [pieceType][rotation] = list of (row, col) offsets
    // row 0 = top, col 0 = left
    static int[][][][] pieces = new int[][][][]
    {
        // I
        new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{0,3} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{3,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{0,3} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{3,0} },
        },
        // O
        new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{1,1} },
        },
        // T
        new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{2,0} },
            new int[][] { new[]{1,0}, new[]{0,1}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{2,0} },
        },
        // S
        new int[][][] {
            new int[][] { new[]{0,1}, new[]{0,2}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{2,1} },
            new int[][] { new[]{0,1}, new[]{0,2}, new[]{1,0}, new[]{1,1} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{2,1} },
        },
        // Z
        new int[][][] {
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,1}, new[]{1,0}, new[]{1,1}, new[]{2,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,1}, new[]{1,0}, new[]{1,1}, new[]{2,0} },
        },
        // J
        new int[][][] {
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,0}, new[]{2,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{1,2} },
            new int[][] { new[]{0,1}, new[]{1,1}, new[]{2,0}, new[]{2,1} },
        },
        // L
        new int[][][] {
            new int[][] { new[]{0,2}, new[]{1,0}, new[]{1,1}, new[]{1,2} },
            new int[][] { new[]{0,0}, new[]{1,0}, new[]{2,0}, new[]{2,1} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{0,2}, new[]{1,0} },
            new int[][] { new[]{0,0}, new[]{0,1}, new[]{1,1}, new[]{2,1} },
        },
    };

    static int PieceIndex(char c)
    {
        switch (c) {
            case 'I': return 0;
            case 'O': return 1;
            case 'T': return 2;
            case 'S': return 3;
            case 'Z': return 4;
            case 'J': return 5;
            case 'L': return 6;
        }
        return 0;
    }

    static int MaxCol(int[][] cells)
    {
        int m = 0;
        foreach (var c in cells) if (c[1] > m) m = c[1];
        return m;
    }

    // Returns the row offset (topmost row) where the piece lands when dropped at given col.
    // Returns -1 if placement is invalid (out of bounds).
    static int DropRow(int[][] cells, int col)
    {
        // For each piece cell column, find the highest occupied row in that board column
        // The piece can drop until it hits something.
        // We simulate by trying each possible drop row from bottom up.

        // First check horizontal bounds
        foreach (var c in cells)
        {
            int cc = col + c[1];
            if (cc < 0 || cc >= width) return -1;
        }

        // Find landing row: start from row=0, push down as far as possible
        // Binary search or linear scan from bottom
        int dropRow = 0;
        // Max possible dropRow: height - maxPieceRow
        int maxPieceRow = 0;
        foreach (var c in cells) if (c[0] > maxPieceRow) maxPieceRow = c[0];
        int maxDrop = height - 1 - maxPieceRow;

        for (int dr = 0; dr <= maxDrop; dr++)
        {
            // Check if piece can be at row dr
            bool fits = true;
            foreach (var c in cells)
            {
                int r = dr + c[0];
                int cc = col + c[1];
                if (board[r, cc]) { fits = false; break; }
            }
            if (fits) dropRow = dr;
            else break;
        }

        return dropRow;
    }

    static bool[,] SimulateBoard(bool[,] src, int[][] cells, int col, int dropRow)
    {
        bool[,] b = (bool[,])src.Clone();
        foreach (var c in cells)
            b[dropRow + c[0], col + c[1]] = true;

        // Clear lines
        for (int r = height - 1; r >= 0; r--)
        {
            bool full = true;
            for (int c = 0; c < width; c++) if (!b[r, c]) { full = false; break; }
            if (full)
            {
                for (int rr = r; rr > 0; rr--)
                    for (int c = 0; c < width; c++)
                        b[rr, c] = b[rr - 1, c];
                for (int c = 0; c < width; c++) b[0, c] = false;
                r++;
            }
        }
        return b;
    }

    static double EvaluateBoard(bool[,] b)
    {
        int[] h = new int[width];
        for (int c = 0; c < width; c++)
            for (int r = 0; r < height; r++)
                if (b[r, c]) { h[c] = height - r; break; }

        int aggHeight = 0;
        for (int c = 0; c < width; c++) aggHeight += h[c];

        int holes = 0;
        for (int c = 0; c < width; c++)
        {
            bool found = false;
            for (int r = 0; r < height; r++)
            {
                if (b[r, c]) found = true;
                else if (found) holes++;
            }
        }

        int bumpiness = 0;
        for (int c = 0; c < width - 1; c++)
            bumpiness += Math.Abs(h[c] - h[c + 1]);

        int lines = 0;
        for (int r = 0; r < height; r++)
        {
            bool full = true;
            for (int c = 0; c < width; c++) if (!b[r, c]) { full = false; break; }
            if (full) lines++;
        }

        return -0.510066 * aggHeight
             + 0.760666 * lines
             - 0.35663 * holes
             - 0.184483 * bumpiness;
    }

    static string RotKey(int[][] cells)
    {
        var sorted = new List<int[]>(cells);
        sorted.Sort((a, b2) => a[0] != b2[0] ? a[0].CompareTo(b2[0]) : a[1].CompareTo(b2[1]));
        var sb = new StringBuilder();
        foreach (var c in sorted) sb.Append(c[0]).Append(',').Append(c[1]).Append(';');
        return sb.ToString();
    }

    static (int rot, int col) FindBestMove(int pieceIdx)
    {
        double best = double.NegativeInfinity;
        int bestRot = 0, bestCol = 0;
        var seenRots = new HashSet<string>();

        for (int rot = 0; rot < 4; rot++)
        {
            var cells = pieces[pieceIdx][rot];
            string key = RotKey(cells);
            if (!seenRots.Add(key)) continue;

            int maxC = MaxCol(cells);
            int maxCols = width - maxC;

            for (int col = 0; col < maxCols; col++)
            {
                int dr = DropRow(cells, col);
                if (dr < 0) continue;

                bool[,] sim = SimulateBoard(board, cells, col, dr);
                double score = EvaluateBoard(sim);

                if (score > best)
                {
                    best = score;
                    bestRot = rot;
                    bestCol = col;
                }
            }
        }

        return (bestRot, bestCol);
    }

    static void ApplyToBoard(int[][] cells, int col, int dropRow)
    {
        foreach (var c in cells)
            board[dropRow + c[0], col + c[1]] = true;

        // Clear lines
        for (int r = height - 1; r >= 0; r--)
        {
            bool full = true;
            for (int c = 0; c < width; c++) if (!board[r, c]) { full = false; break; }
            if (full)
            {
                for (int rr = r; rr > 0; rr--)
                    for (int c = 0; c < width; c++)
                        board[rr, c] = board[rr - 1, c];
                for (int c = 0; c < width; c++) board[0, c] = false;
                r++;
            }
        }
    }

    static void Main()
    {
        var tokens = new Queue<string>();

        string ReadToken()
        {
            while (tokens.Count == 0)
            {
                string line = Console.ReadLine();
                if (line == null) return null;
                foreach (var t in line.Trim().Split(new char[]{' ','\t','\r','\n'}, StringSplitOptions.RemoveEmptyEntries))
                    tokens.Enqueue(t);
            }
            return tokens.Dequeue();
        }

        width = int.Parse(ReadToken());
        height = int.Parse(ReadToken());
        int numPieces = int.Parse(ReadToken());

        board = new bool[height, width];

        for (int i = 0; i < numPieces; i++)
        {
            string pieceStr = ReadToken();
            if (pieceStr == null) break;

            int pieceIdx = PieceIndex(pieceStr[0]);
            var (rot, col) = FindBestMove(pieceIdx);

            Console.WriteLine($"{rot} {col}");
            Console.Out.Flush();

            string result = ReadToken();
            if (result == null || result == "gameover") break;

            var cells = pieces[pieceIdx][rot];
            int dropRow = DropRow(cells, col);
            ApplyToBoard(cells, col, dropRow);
        }
    }
}
