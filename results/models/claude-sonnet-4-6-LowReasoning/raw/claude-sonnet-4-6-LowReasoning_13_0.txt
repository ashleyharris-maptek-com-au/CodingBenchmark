#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <algorithm>
#include <cstdint>

using namespace std;

static const uint64_t M1 = 1000000007ULL;
static const uint64_t M2 = 998244353ULL;
static const uint64_t B1 = 131ULL;
static const uint64_t B2 = 137ULL;

static uint64_t powmod(uint64_t b, long long n, uint64_t m) {
    uint64_t res = 1;
    b %= m;
    while (n > 0) {
        if (n & 1) res = res * b % m;
        b = b * b % m;
        n >>= 1;
    }
    return res;
}

// Returns true if a common substring of length L exists; sets result to one such substring.
// Uses rolling Rabin-Karp double hash + set intersection, starting from the shortest string.
bool checkLength(const vector<string>& strs, int L, string& result) {
    if (L == 0) { result = ""; return true; }

    int N = (int)strs.size();
    const string& s0 = strs[0];
    int n0 = (int)s0.size();
    if (n0 < L) return false;

    uint64_t bp1 = powmod(B1, L - 1, M1);
    uint64_t bp2 = powmod(B2, L - 1, M2);

    // Build hash set of all L-length windows in string[0] (shortest string)
    unordered_set<uint64_t> cands;
    cands.reserve((size_t)(n0 - L + 1) * 2 + 4);

    uint64_t h1 = 0, h2 = 0;
    for (int i = 0; i < L; i++) {
        h1 = (h1 * B1 + (uint8_t)s0[i]) % M1;
        h2 = (h2 * B2 + (uint8_t)s0[i]) % M2;
    }
    cands.insert(h1 * M2 + h2);
    for (int i = 1; i + L <= n0; i++) {
        h1 = (h1 + M1 - (uint64_t)(uint8_t)s0[i-1] * bp1 % M1) % M1;
        h1 = (h1 * B1 + (uint8_t)s0[i+L-1]) % M1;
        h2 = (h2 + M2 - (uint64_t)(uint8_t)s0[i-1] * bp2 % M2) % M2;
        h2 = (h2 * B2 + (uint8_t)s0[i+L-1]) % M2;
        cands.insert(h1 * M2 + h2);
    }

    // Intersect candidate set with each subsequent string
    for (int k = 1; k < N && !cands.empty(); k++) {
        const string& sk = strs[k];
        int nk = (int)sk.size();
        if (nk < L) return false;

        unordered_set<uint64_t> present;
        present.reserve(cands.size() * 2 + 4);

        h1 = h2 = 0;
        for (int i = 0; i < L; i++) {
            h1 = (h1 * B1 + (uint8_t)sk[i]) % M1;
            h2 = (h2 * B2 + (uint8_t)sk[i]) % M2;
        }
        uint64_t pk = h1 * M2 + h2;
        if (cands.count(pk)) present.insert(pk);

        for (int i = 1; i + L <= nk; i++) {
            h1 = (h1 + M1 - (uint64_t)(uint8_t)sk[i-1] * bp1 % M1) % M1;
            h1 = (h1 * B1 + (uint8_t)sk[i+L-1]) % M1;
            h2 = (h2 + M2 - (uint64_t)(uint8_t)sk[i-1] * bp2 % M2) % M2;
            h2 = (h2 * B2 + (uint8_t)sk[i+L-1]) % M2;
            pk = h1 * M2 + h2;
            if (cands.count(pk)) present.insert(pk);
        }

        cands = move(present);
    }

    if (cands.empty()) return false;

    // Recover the actual substring from string[0]
    uint64_t target = *cands.begin();
    h1 = h2 = 0;
    for (int i = 0; i < L; i++) {
        h1 = (h1 * B1 + (uint8_t)s0[i]) % M1;
        h2 = (h2 * B2 + (uint8_t)s0[i]) % M2;
    }
    if (h1 * M2 + h2 == target) {
        result = s0.substr(0, L);
        return true;
    }
    for (int i = 1; i + L <= n0; i++) {
        h1 = (h1 + M1 - (uint64_t)(uint8_t)s0[i-1] * bp1 % M1) % M1;
        h1 = (h1 * B1 + (uint8_t)s0[i+L-1]) % M1;
        h2 = (h2 + M2 - (uint64_t)(uint8_t)s0[i-1] * bp2 % M2) % M2;
        h2 = (h2 * B2 + (uint8_t)s0[i+L-1]) % M2;
        if (h1 * M2 + h2 == target) {
            result = s0.substr(i, L);
            return true;
        }
    }

    // Should never reach here with double hashing (collision probability ~1/10^18)
    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;
    cin.ignore();

    vector<string> strs(N);
    for (int i = 0; i < N; i++) {
        getline(cin, strs[i]);
    }

    if (N == 0) { cout << "\n"; return 0; }
    if (N == 1) { cout << strs[0] << "\n"; return 0; }

    // Process shortest string first to minimize candidate set size
    sort(strs.begin(), strs.end(), [](const string& a, const string& b) {
        return a.size() < b.size();
    });

    int minLen = (int)strs[0].size();

    // Binary search on the answer length
    int lo = 1, hi = minLen;
    string best = "";

    while (lo <= hi) {
        int mid = lo + (hi - lo) / 2;
        string result;
        if (checkLength(strs, mid, result)) {
            best = result;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    cout << best << "\n";
    return 0;
}
