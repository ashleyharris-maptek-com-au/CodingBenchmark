import struct

def w(v):
    return struct.pack('<I', v & 0xFFFFFFFF).hex()

def fw(f):
    return struct.pack('<f', f).hex()

def sw(s):
    b = s.encode('ascii') + b'\x00'
    while len(b) % 4: b += b'\x00'
    return b.hex()

def instr(opcode, *ops):
    wc = 1 + len(ops)
    return w((wc << 16) | opcode) + ''.join(w(o) for o in ops)

def const(tid, rid, fval):
    return w((4 << 16) | 43) + w(tid) + w(rid) + fw(fval)

def extInst(tid, rid, setid, extinst, *ops):
    wc = 1 + 1 + 1 + 1 + 1 + len(ops)
    return w((wc << 16) | 12) + w(tid) + w(rid) + w(setid) + w(extinst) + ''.join(w(o) for o in ops)

# IDs
TVOID=1; TFUNC=2; TFLOAT=3; TVEC3=4; TVEC4=5
PTRV3IN=6; PTRV4OUT=7
WPOS=8; OCOL=9; EXT=10
C05=11; C00=12; C10=13; C01=14; C04=15
MAIN=16; LBL=17
WPOSV=18; WY=19; HY=20; TU=21; T=22; MR=23; MG=24; FC=25

BOUND=26

h = ''
# Header
h += w(0x07230203) + w(0x00010000) + w(0) + w(BOUND) + w(0)

# Preamble
h += instr(17, 1)  # OpCapability Shader

# OpExtInstImport %EXT "GLSL.std.450"
glsl = sw("GLSL.std.450")  # 4 words
h += w((6 << 16) | 11) + w(EXT) + glsl

h += instr(14, 0, 1)  # OpMemoryModel Logical GLSL450

# OpEntryPoint Fragment %MAIN "main" %WPOS %OCOL
main_s = sw("main")  # 2 words -> wc=1+1+1+2+2=7
h += w((7 << 16) | 15) + w(4) + w(MAIN) + main_s + w(WPOS) + w(OCOL)

h += instr(16, MAIN, 7)  # OpExecutionMode OriginUpperLeft

# Decorations
h += instr(71, WPOS, 30, 0)   # Location 0
h += instr(71, OCOL, 30, 0)   # Location 0

# Types
h += instr(19, TVOID)
h += instr(33, TFUNC, TVOID)
h += instr(22, TFLOAT, 32)
h += instr(23, TVEC3, TFLOAT, 3)
h += instr(23, TVEC4, TFLOAT, 4)
h += instr(32, PTRV3IN,  1, TVEC3)   # Input
h += instr(32, PTRV4OUT, 3, TVEC4)   # Output

# Variables
h += instr(59, PTRV3IN,  WPOS, 1)
h += instr(59, PTRV4OUT, OCOL, 3)

# Constants
h += const(TFLOAT, C05, 0.5)
h += const(TFLOAT, C00, 0.0)
h += const(TFLOAT, C10, 1.0)
h += const(TFLOAT, C01, 0.1)
h += const(TFLOAT, C04, 0.4)

# Function body
h += instr(54, TVOID, MAIN, 0, TFUNC)  # OpFunction
h += instr(248, LBL)                    # OpLabel

h += instr(61, TVEC3, WPOSV, WPOS)       # load worldPos
h += instr(81, TFLOAT, WY, WPOSV, 1)    # extract y
h += instr(133, TFLOAT, HY, WY, C05)    # y*0.5
h += instr(129, TFLOAT, TU, HY, C05)    # y*0.5+0.5

h += extInst(TFLOAT, T,  EXT, 43, TU, C00, C10)  # FClamp(t,0,1)
h += extInst(TFLOAT, MR, EXT, 46, C01, C10, T)   # FMix(0.1,1.0,t)
h += extInst(TFLOAT, MG, EXT, 46, C04, C10, T)   # FMix(0.4,1.0,t)

h += instr(80, TVEC4, FC, MR, MG, MR, C10)  # vec4(r,g,b=r,1.0)
h += instr(62, OCOL, FC)                      # store
h += w((1 << 16) | 253)                       # OpReturn
h += w((1 << 16) | 56)                        # OpFunctionEnd

print(h)
