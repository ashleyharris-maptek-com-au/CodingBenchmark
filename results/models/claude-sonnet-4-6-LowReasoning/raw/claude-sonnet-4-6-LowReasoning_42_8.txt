; SPIR-V geometry shader: shrink triangles toward centroid
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input color array at location 2
OpDecorate %in_color Location 2

; Output color at location 0
OpDecorate %out_color Location 0

; Type declarations
%void       = OpTypeVoid
%bool       = OpTypeBool
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 1
%uint       = OpTypeInt 32 0
%v4float    = OpTypeVector %float 4
%v3float    = OpTypeVector %float 3

%uint_3     = OpConstant %uint 3
%int_0      = OpConstant %int 0
%int_1      = OpConstant %int 1
%int_2      = OpConstant %int 2
%float_0    = OpConstant %float 0.0
%float_1    = OpConstant %float 1.0
%float_inv3 = OpConstant %float 0.33333334
%float_s    = OpConstant %float 0.7

; Input gl_PerVertex struct and array
%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%ptr_in_v4float   = OpTypePointer Input %v4float
%gl_in            = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex struct
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float  = OpTypePointer Output %v4float
%gl_out           = OpVariable %ptr_out_pv Output

; Input color array (location 2)
%arr_v3float      = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3float
%ptr_in_v3float   = OpTypePointer Input %v3float
%in_color         = OpVariable %ptr_in_arr_v3 Input

; Output color (location 0)
%ptr_out_v3float  = OpTypePointer Output %v3float
%out_color        = OpVariable %ptr_out_v3float Output

; Function type
%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0
%ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0     = OpLoad %v4float %ptr_p0

; Load p1
%ptr_p1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1     = OpLoad %v4float %ptr_p1

; Load p2
%ptr_p2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%p2     = OpLoad %v4float %ptr_p2

; Compute centroid C = (p0 + p1 + p2) / 3
%sum01  = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%inv3   = OpCompositeConstruct %v4float %float_inv3 %float_inv3 %float_inv3 %float_inv3
%centroid = OpFMul %v4float %sum012 %inv3

; Shrink factor s = 0.7
%svec   = OpCompositeConstruct %v4float %float_s %float_s %float_s %float_s

; Emit vertex 0: new_pos = C + s * (p0 - C)
%d0     = OpFSub %v4float %p0 %centroid
%sd0    = OpFMul %v4float %svec %d0
%np0    = OpFAdd %v4float %centroid %sd0
%ptr_out_pos0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %ptr_out_pos0 %np0
%ptr_c0 = OpAccessChain %ptr_in_v3float %in_color %int_0
%c0     = OpLoad %v3float %ptr_c0
OpStore %out_color %c0
OpEmitVertex

; Emit vertex 1: new_pos = C + s * (p1 - C)
%d1     = OpFSub %v4float %p1 %centroid
%sd1    = OpFMul %v4float %svec %d1
%np1    = OpFAdd %v4float %centroid %sd1
%ptr_out_pos1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %ptr_out_pos1 %np1
%ptr_c1 = OpAccessChain %ptr_in_v3float %in_color %int_1
%c1     = OpLoad %v3float %ptr_c1
OpStore %out_color %c1
OpEmitVertex

; Emit vertex 2: new_pos = C + s * (p2 - C)
%d2     = OpFSub %v4float %p2 %centroid
%sd2    = OpFMul %v4float %svec %d2
%np2    = OpFAdd %v4float %centroid %sd2
%ptr_out_pos2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %ptr_out_pos2 %np2
%ptr_c2 = OpAccessChain %ptr_in_v3float %in_color %int_2
%c2     = OpLoad %v3float %ptr_c2
OpStore %out_color %c2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
