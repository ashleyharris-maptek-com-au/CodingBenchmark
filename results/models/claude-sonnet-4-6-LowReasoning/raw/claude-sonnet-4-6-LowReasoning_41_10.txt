               OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
               OpExecutionMode %main OriginUpperLeft

               OpDecorate %inWorldPos Location 0
               OpDecorate %inNormal Location 1
               OpDecorate %inUV Location 2
               OpDecorate %inTangent Location 3
               OpDecorate %inColor Location 4
               OpDecorate %outColor Location 0

; UBO decorations
               OpDecorate %uboVar DescriptorSet 0
               OpDecorate %uboVar Binding 0
               OpDecorate %UBOType Block
               OpMemberDecorate %UBOType 0 Offset 0
               OpMemberDecorate %UBOType 0 ColMajor
               OpMemberDecorate %UBOType 0 MatrixStride 16
               OpMemberDecorate %UBOType 1 Offset 64
               OpMemberDecorate %UBOType 1 ColMajor
               OpMemberDecorate %UBOType 1 MatrixStride 16
               OpMemberDecorate %UBOType 2 Offset 128
               OpMemberDecorate %UBOType 2 ColMajor
               OpMemberDecorate %UBOType 2 MatrixStride 16
               OpMemberDecorate %UBOType 3 Offset 192
               OpMemberDecorate %UBOType 4 Offset 208
               OpMemberDecorate %UBOType 5 Offset 224

; Types
       %void = OpTypeVoid
       %bool = OpTypeBool
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
    %mat4v4f = OpTypeMatrix %v4float 4
        %int = OpTypeInt 32 1
     %voidFn = OpTypeFunction %void

; UBO struct type
    %UBOType = OpTypeStruct %mat4v4f %mat4v4f %mat4v4f %v4float %v4float %v4float

; Pointer types
  %pInV3 = OpTypePointer Input %v3float
  %pInV2 = OpTypePointer Input %v2float
  %pOutV4 = OpTypePointer Output %v4float
  %pUBOType = OpTypePointer Uniform %UBOType
  %pUnifV4 = OpTypePointer Uniform %v4float

; Variables
 %inWorldPos = OpVariable %pInV3 Input
   %inNormal = OpVariable %pInV3 Input
       %inUV = OpVariable %pInV2 Input
  %inTangent = OpVariable %pInV3 Input
    %inColor = OpVariable %pInV3 Input
   %outColor = OpVariable %pOutV4 Output
     %uboVar = OpVariable %pUBOType Uniform

; Integer constants for access chains
      %int_3 = OpConstant %int 3

; Float constants
      %f_0_0 = OpConstant %float 0.0
      %f_1_0 = OpConstant %float 1.0
    %f_pi16  = OpConstant %float 50.26544  ; pi * 16
      %f_0_4 = OpConstant %float 0.4
      %f_0_6 = OpConstant %float 0.6
      %f_0_9 = OpConstant %float 0.9
      %f_0_1 = OpConstant %float 0.1

; Stripe colors
     %c_red  = OpConstantComposite %v3float %f_0_9 %f_0_1 %f_0_1
    %c_blue  = OpConstantComposite %v3float %f_0_1 %f_0_1 %f_0_9

; Main function
       %main = OpFunction %void None %voidFn
      %entry = OpLabel

; Load UV, extract y
         %uv = OpLoad %v2float %inUV
        %uvy = OpCompositeExtract %float %uv 1

; sin(uv.y * pi * 16)
     %sinArg = OpFMul %float %uvy %f_pi16
     %sinVal = OpExtInst %float %1 Sin %sinArg

; Select stripe color: sin >= 0 -> red, else -> blue
  %isPosSin  = OpFOrdGreaterThanEqual %bool %sinVal %f_0_0
%stripeColor = OpSelect %v3float %isPosSin %c_red %c_blue

; Load and normalize normal
      %normV = OpLoad %v3float %inNormal
     %normN  = OpExtInst %v3float %1 Normalize %normV

; Load worldPos
     %wpos   = OpLoad %v3float %inWorldPos

; Load lightPos from UBO member 3
  %lpPtr     = OpAccessChain %pUnifV4 %uboVar %int_3
  %lp4       = OpLoad %v4float %lpPtr
  %lp3       = OpVectorShuffle %v3float %lp4 %lp4 0 1 2

; Light direction (normalized)
  %ldir      = OpFSub %v3float %lp3 %wpos
  %ldirN     = OpExtInst %v3float %1 Normalize %ldir

; Lambertian diffuse = max(0, dot(N, L))
  %dotNL     = OpDot %float %normN %ldirN
  %diffuse   = OpExtInst %float %1 FMax %dotNL %f_0_0

; Shading factor = 0.4 + 0.6 * diffuse
  %scaled    = OpFMul %float %f_0_6 %diffuse
  %factor    = OpFAdd %float %f_0_4 %scaled

; Apply factor to stripe color components
  %sc_r      = OpCompositeExtract %float %stripeColor 0
  %sc_g      = OpCompositeExtract %float %stripeColor 1
  %sc_b      = OpCompositeExtract %float %stripeColor 2
  %out_r     = OpFMul %float %factor %sc_r
  %out_g     = OpFMul %float %factor %sc_g
  %out_b     = OpFMul %float %factor %sc_b

; Compose final color with alpha=1
  %finalCol  = OpCompositeConstruct %v4float %out_r %out_g %out_b %f_1_0
               OpStore %outColor %finalCol

               OpReturn
               OpFunctionEnd
