struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    float cx = input.uv.x * 3.0 - 2.0;
    float cy = input.uv.y * 2.0 - 1.0;

    float zx = 0.0;
    float zy = 0.0;
    int iter = 0;

    for (int i = 0; i < 20; i++) {
        float zx2 = zx * zx - zy * zy + cx;
        float zy2 = 2.0 * zx * zy + cy;
        zx = zx2;
        zy = zy2;
        if (zx * zx + zy * zy > 4.0) {
            iter = i + 1;
            break;
        }
        iter = i + 1;
    }

    float t = (float)iter / 20.0;
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
