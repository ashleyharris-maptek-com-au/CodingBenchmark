using System;
using System.Collections.Generic;
using System.Linq;

class AsteroidInterception
{
    const double G = 6.674e-11;
    const double DAY = 86400.0;
    const double AU = 1.496e11;

    static double[] V3(double x, double y, double z) => new[] { x, y, z };
    static double[] Add3(double[] a, double[] b) => new[] { a[0]+b[0], a[1]+b[1], a[2]+b[2] };
    static double[] Sub3(double[] a, double[] b) => new[] { a[0]-b[0], a[1]-b[1], a[2]-b[2] };
    static double[] Mul3(double[] a, double s) => new[] { a[0]*s, a[1]*s, a[2]*s };
    static double Dot3(double[] a, double[] b) => a[0]*b[0] + a[1]*b[1] + a[2]*b[2];
    static double Norm3(double[] a) => Math.Sqrt(Dot3(a, a));
    static double[] Unit3(double[] a) { double n = Norm3(a); return n < 1e-30 ? new double[3] : Mul3(a, 1.0/n); }
    static double[] Cross3(double[] a, double[] b) => new[] {
        a[1]*b[2]-a[2]*b[1], a[2]*b[0]-a[0]*b[2], a[0]*b[1]-a[1]*b[0]
    };
    static double[] CopyV(double[] v) => new[] { v[0], v[1], v[2] };

    struct Body
    {
        public string Name;
        public double Mass;
        public double[] P, V; // position, velocity
        public Body(string n, double m, double[] p, double[] v) { Name=n; Mass=m; P=p; V=v; }
        public Body Clone() => new Body(Name, Mass, CopyV(P), CopyV(V));
    }

    // Stumpff functions C(z) and S(z)
    static void StumpffCS(double z, out double C, out double S)
    {
        if (z > 1e-6)
        {
            double sq = Math.Sqrt(z);
            C = (1 - Math.Cos(sq)) / z;
            S = (sq - Math.Sin(sq)) / (sq * sq * sq);
        }
        else if (z < -1e-6)
        {
            double sq = Math.Sqrt(-z);
            C = (1 - Math.Cosh(sq)) / z;
            S = (Math.Sinh(sq) - sq) / (sq * sq * sq);
        }
        else
        {
            C = 0.5;
            S = 1.0 / 6.0;
        }
    }

    // Lambert solver using universal variable method (Battin/Gooding approach)
    // Returns (v1, v2) or null on failure
    static (double[] v1, double[] v2)? LambertSolve(double[] r1, double[] r2, double tof, double mu, bool prograde = true)
    {
        if (tof <= 0) return null;
        double r1m = Norm3(r1), r2m = Norm3(r2);
        if (r1m < 1e3 || r2m < 1e3) return null;

        double cosTheta = Dot3(r1, r2) / (r1m * r2m);
        cosTheta = Math.Max(-1, Math.Min(1, cosTheta));

        double[] cross = Cross3(r1, r2);
        double sinTheta = Norm3(cross) / (r1m * r2m);

        // Determine transfer direction
        bool longWay = false;
        if (prograde && cross[2] < 0) longWay = true;
        if (!prograde && cross[2] >= 0) longWay = true;

        double dm = longWay ? -1.0 : 1.0;

        double A = dm * Math.Sqrt(r1m * r2m * (1 + cosTheta));
        if (Math.Abs(A) < 1e-6) return null;

        // Newton iteration for z
        double z = 0.0;
        double C, S;

        for (int iter = 0; iter < 200; iter++)
        {
            StumpffCS(z, out C, out S);
            double sqrtC = Math.Sqrt(Math.Max(C, 1e-30));
            double y = r1m + r2m + A * (z * S - 1.0) / sqrtC;

            if (A > 0 && y < 0)
            {
                // Adjust z to make y positive
                z += 0.1;
                continue;
            }

            double sqrtY = Math.Sqrt(Math.Max(y, 1e-30));
            double x = sqrtY / sqrtC;
            double tCalc = (x * x * x * S + A * sqrtY) / Math.Sqrt(mu);

            // Derivative dt/dz
            double dtdz;
            if (Math.Abs(z) < 1e-6)
            {
                dtdz = (Math.Sqrt(2) / 40.0 * Math.Pow(y, 1.5) + A / 8.0 * (sqrtY + A * Math.Sqrt(0.5 / y))) / Math.Sqrt(mu);
            }
            else
            {
                StumpffCS(z, out C, out S);
                double dCdz = (1 - z * S - 2 * C) / (2 * z);
                double dSdz = (C - 3 * S) / (2 * z);
                double dydz = A / (2 * sqrtC) * (dSdz - S / (2 * C) * dCdz) +
                              A * dCdz / (2 * Math.Pow(C, 1.5)) * (z * S - 1);
                dtdz = (x * x * x * (dSdz - 3 * S * dCdz / (2 * C)) / (2 * z) +
                         A * (3 * S * sqrtY / C + A * Math.Sqrt(C / y)) / 8.0) / Math.Sqrt(mu);
                // Fallback simpler derivative
                if (!double.IsFinite(dtdz) || Math.Abs(dtdz) < 1e-50)
                    dtdz = (x * x * x * dSdz + A / 2.0 * (sqrtY / S / C + A * Math.Sqrt(C) / sqrtY)) / Math.Sqrt(mu);
            }

            if (!double.IsFinite(dtdz) || Math.Abs(dtdz) < 1e-60) dtdz = 1e-3;

            double dz = (tof - tCalc) / dtdz;
            // Clamp step
            dz = Math.Max(-2.0, Math.Min(2.0, dz));
            z += dz;

            if (Math.Abs(dz) < 1e-10 && Math.Abs(tCalc - tof) / tof < 1e-8)
                break;
        }

        StumpffCS(z, out C, out S);
        double yf = r1m + r2m + A * (z * S - 1.0) / Math.Sqrt(Math.Max(C, 1e-30));
        if (yf < 0) return null;

        double f = 1.0 - yf / r1m;
        double g = A * Math.Sqrt(yf / mu);
        if (Math.Abs(g) < 1e-30) return null;
        double gDot = 1.0 - yf / r2m;

        double[] v1 = new double[3], v2 = new double[3];
        for (int i = 0; i < 3; i++)
        {
            v1[i] = (r2[i] - f * r1[i]) / g;
            v2[i] = (gDot * r2[i] - r1[i]) / g;
        }

        if (!double.IsFinite(v1[0]) || !double.IsFinite(v2[0])) return null;
        return (v1, v2);
    }

    // Propagate N bodies with RK4
    static Body[] PropagateRK4(Body[] bodies, double dt)
    {
        int n = bodies.Length;
        int sz = n * 6;
        double[] s = new double[sz];
        double[] masses = new double[n];

        for (int i = 0; i < n; i++)
        {
            s[i*6]   = bodies[i].P[0]; s[i*6+1] = bodies[i].P[1]; s[i*6+2] = bodies[i].P[2];
            s[i*6+3] = bodies[i].V[0]; s[i*6+4] = bodies[i].V[1]; s[i*6+5] = bodies[i].V[2];
            masses[i] = bodies[i].Mass;
        }

        double[] k1 = Deriv(s, masses, n);
        double[] s2 = Axpy(1, s, dt/2, k1, sz);
        double[] k2 = Deriv(s2, masses, n);
        double[] s3 = Axpy(1, s, dt/2, k2, sz);
        double[] k3 = Deriv(s3, masses, n);
        double[] s4 = Axpy(1, s, dt, k3, sz);
        double[] k4 = Deriv(s4, masses, n);

        double[] ns = new double[sz];
        for (int i = 0; i < sz; i++)
            ns[i] = s[i] + dt/6.0 * (k1[i] + 2*k2[i] + 2*k3[i] + k4[i]);

        Body[] res = new Body[n];
        for (int i = 0; i < n; i++)
            res[i] = new Body(bodies[i].Name, bodies[i].Mass,
                new[] { ns[i*6], ns[i*6+1], ns[i*6+2] },
                new[] { ns[i*6+3], ns[i*6+4], ns[i*6+5] });
        return res;
    }

    static double[] Axpy(double a, double[] x, double b, double[] y, int sz)
    {
        double[] r = new double[sz];
        for (int i = 0; i < sz; i++) r[i] = a*x[i] + b*y[i];
        return r;
    }

    static double[] Deriv(double[] s, double[] masses, int n)
    {
        double[] d = new double[n * 6];
        for (int i = 0; i < n; i++)
        {
            d[i*6]   = s[i*6+3];
            d[i*6+1] = s[i*6+4];
            d[i*6+2] = s[i*6+5];
            double ax = 0, ay = 0, az = 0;
            double xi = s[i*6], yi = s[i*6+1], zi = s[i*6+2];
            for (int j = 0; j < n; j++)
            {
                if (i == j) continue;
                double dx = s[j*6] - xi;
                double dy = s[j*6+1] - yi;
                double dz = s[j*6+2] - zi;
                double r2 = dx*dx + dy*dy + dz*dz;
                if (r2 < 1e6) continue; // avoid singularity
                double gm_r3 = G * masses[j] / (r2 * Math.Sqrt(r2));
                ax += gm_r3 * dx;
                ay += gm_r3 * dy;
                az += gm_r3 * dz;
            }
            d[i*6+3] = ax; d[i*6+4] = ay; d[i*6+5] = az;
        }
        return d;
    }

    // Propagate a single spacecraft under influence of all bodies (no spacecraft mass)
    static (double[] pos, double[] vel) PropagateSC(double[] p0, double[] v0, Body[] bodies, double dtSec)
    {
        double[] sp = CopyV(p0), sv = CopyV(v0);

        double[] AccSC(double[] pos, double[] vel)
        {
            double ax = 0, ay = 0, az = 0;
            foreach (var b in bodies)
            {
                double dx = b.P[0] - pos[0];
                double dy = b.P[1] - pos[1];
                double dz = b.P[2] - pos[2];
                double r2 = dx*dx + dy*dy + dz*dz;
                if (r2 < 1e6) continue;
                double gm_r3 = G * b.Mass / (r2 * Math.Sqrt(r2));
                ax += gm_r3 * dx; ay += gm_r3 * dy; az += gm_r3 * dz;
            }
            return new[] { ax, ay, az };
        }

        // RK4
        double[] a1 = AccSC(sp, sv);
        double[] p2 = Add3(sp, Mul3(sv, dtSec/2)); double[] v2 = Add3(sv, Mul3(a1, dtSec/2));
        double[] a2 = AccSC(p2, v2);
        double[] p3 = Add3(sp, Mul3(v2, dtSec/2)); double[] v3 = Add3(sv, Mul3(a2, dtSec/2));
        double[] a3 = AccSC(p3, v3);
        double[] p4 = Add3(sp, Mul3(v3, dtSec)); double[] v4 = Add3(sv, Mul3(a3, dtSec));
        double[] a4 = AccSC(p4, v4);

        double[] np = new double[3], nv = new double[3];
        for (int i = 0; i < 3; i++)
        {
            np[i] = sp[i] + dtSec/6.0*(sv[i]+2*v2[i]+2*v3[i]+v4[i]);
            nv[i] = sv[i] + dtSec/6.0*(a1[i]+2*a2[i]+2*a3[i]+a4[i]);
        }
        return (np, nv);
    }

    // Precompute body positions at each day up to maxDays
    static (double[] pos, double[] vel)[] PrecomputeBodyTrajectory(Body[] initialBodies, int bodyIdx, int maxDays, double dtSec = DAY)
    {
        var traj = new (double[] pos, double[] vel)[maxDays + 1];
        Body[] state = initialBodies.Select(b => b.Clone()).ToArray();
        traj[0] = (CopyV(state[bodyIdx].P), CopyV(state[bodyIdx].V));
        for (int d = 1; d <= maxDays; d++)
        {
            state = PropagateRK4(state, dtSec);
            traj[d] = (CopyV(state[bodyIdx].P), CopyV(state[bodyIdx].V));
        }
        return traj;
    }

    // Precompute all body states at each day
    static Body[][] PrecomputeAllStates(Body[] initialBodies, int maxDays, double dtSec = DAY)
    {
        var states = new Body[maxDays + 1][];
        states[0] = initialBodies.Select(b => b.Clone()).ToArray();
        for (int d = 1; d <= maxDays; d++)
            states[d] = PropagateRK4(states[d-1], dtSec);
        return states;
    }

    static void Main(string[] args)
    {
        // --- Parse input ---
        var tokens = new List<string>();
        string ln;
        while ((ln = Console.ReadLine()) != null)
            foreach (var t in ln.Split(new char[]{' ','\t','\r','\n'}, StringSplitOptions.RemoveEmptyEntries))
                tokens.Add(t);

        int idx = 0;
        int numBodies    = int.Parse(tokens[idx++]);
        double warnDays  = double.Parse(tokens[idx++]);
        double scMass    = double.Parse(tokens[idx++]);
        double dvBudget  = double.Parse(tokens[idx++]);
        double astMass   = double.Parse(tokens[idx++]);
        double t0Unix    = double.Parse(tokens[idx++]);

        var bodies = new Body[numBodies];
        int earthIdx = -1, astIdx = -1, sunIdx = -1;

        for (int i = 0; i < numBodies; i++)
        {
            string nm = tokens[idx++];
            double mass = double.Parse(tokens[idx++]);
            double x  = double.Parse(tokens[idx++]);
            double y  = double.Parse(tokens[idx++]);
            double z  = double.Parse(tokens[idx++]);
            double vx = double.Parse(tokens[idx++]);
            double vy = double.Parse(tokens[idx++]);
            double vz = double.Parse(tokens[idx++]);
            bodies[i] = new Body(nm, mass, new[]{x,y,z}, new[]{vx,vy,vz});
            if (nm == "Earth")    earthIdx = i;
            if (nm == "Asteroid") astIdx   = i;
            if (nm == "Sun" || nm.ToLower()=="sun") sunIdx = i;
        }
        string spkPath = idx < tokens.Count ? tokens[idx] : "";

        double muSun = G * (sunIdx >= 0 ? bodies[sunIdx].Mass : 1.989e30);

        int maxDays = (int)Math.Ceiling(warnDays * 1.5);
        Console.Error.WriteLine($"Warning: {warnDays} days, dV budget: {dvBudget} m/s, maxDays: {maxDays}");
        Console.Error.WriteLine("Precomputing trajectories...");

        // --- Precompute all body states ---
        // For long missions use 1-day steps; for short missions use finer steps
        double propStep = DAY;
        var allStates = PrecomputeAllStates(bodies, maxDays, propStep);

        // --- Find asteroid-Earth closest approach (Earth impact) ---
        double impactTime   = warnDays;
        double impactDistMin = double.MaxValue;
        double earthRadii   = 6.371e6;

        for (int d = 0; d <= maxDays; d++)
        {
            double[] dp = Sub3(allStates[d][astIdx].P, allStates[d][earthIdx].P);
            double dist = Norm3(dp);
            if (dist < impactDistMin)
            {
                impactDistMin = dist;
                impactTime    = d;
            }
        }
        Console.Error.WriteLine($"Closest approach: {impactDistMin/1000:F0} km at day {impactTime}");

        // Use the time up to just before impact for intercept window
        double interceptWindow = Math.Min(impactTime - 10, warnDays * 0.95);

        // --- Scan for Lambert trajectory ---
        // Launch from Earth, intercept asteroid
        // Minimize launch delta-v; impact delta-v = relative velocity at intercept (free)

        double bestLaunchDV   = double.MaxValue;
        double bestLaunchDay  = -1;
        double bestArriveDay  = -1;
        double[] bestLaunchDVVec = null;
        double[] bestImpactVSC  = null; // spacecraft velocity at intercept
        double[] bestImpactVAst = null; // asteroid velocity at intercept

        // Adaptive scanning grid
        int launchSamples  = 80;
        int arriveSamples  = 80;

        double launchEnd   = Math.Min(warnDays * 0.4, interceptWindow - 30);
        double arriveStart = 30.0;
        double arriveEnd   = interceptWindow;

        // Two-pass: coarse then fine
        for (int pass = 0; pass < 2; pass++)
        {
            double lStep = Math.Max((launchEnd) / launchSamples, 0.5);
            double aStep = Math.Max((arriveEnd - arriveStart) / arriveSamples, 0.5);

            for (double ld = 0; ld <= launchEnd; ld += lStep)
            {
                int liD = (int)Math.Round(ld);
                liD = Math.Max(0, Math.Min(maxDays, liD));
                double[] ePos = allStates[liD][earthIdx].P;
                double[] eVel = allStates[liD][earthIdx].V;

                for (double ad = Math.Max(ld + 15, arriveStart); ad <= arriveEnd; ad += aStep)
                {
                    int aiD = (int)Math.Round(ad);
                    aiD = Math.Max(0, Math.Min(maxDays, aiD));
                    double tof = (ad - ld) * DAY;
                    if (tof < 10 * DAY) continue;

                    double[] aPos = allStates[aiD][astIdx].P;
                    double[] aVel = allStates[aiD][astIdx].V;

                    var result = LambertSolve(ePos, aPos, tof, muSun, true);
                    if (result == null) continue;
                    var (v1, v2) = result.Value;

                    double[] dvLaunch = Sub3(v1, eVel);
                    double ldv = Norm3(dvLaunch);

                    if (ldv < bestLaunchDV && ldv <= dvBudget * 0.95)
                    {
                        bestLaunchDV    = ldv;
                        bestLaunchDay   = ld;
                        bestArriveDay   = ad;
                        bestLaunchDVVec = dvLaunch;
                        bestImpactVSC   = v2;
                        bestImpactVAst  = aVel;
                    }
                }
            }

            // Second pass: zoom in around best solution
            if (pass == 0 && bestLaunchDay >= 0)
            {
                double margin = Math.Max(30, (arriveEnd - arriveStart) / 5.0);
                launchEnd   = Math.Min(bestLaunchDay + margin, warnDays * 0.4);
                double lStart2 = Math.Max(0, bestLaunchDay - margin);
                arriveStart = Math.Max(lStart2 + 15, bestArriveDay - margin);
                arriveEnd   = Math.Min(interceptWindow, bestArriveDay + margin);
                launchSamples = 100;
                arriveSamples = 100;
                // reset launchEnd for scan
                launchEnd = Math.Min(bestLaunchDay + margin, warnDays * 0.4);
            }
            else if (pass == 0 && bestLaunchDay < 0)
            {
                // Try other direction (retrograde)
                for (double ld = 0; ld <= launchEnd; ld += lStep)
                {
                    int liD = (int)Math.Round(ld);
                    liD = Math.Max(0, Math.Min(maxDays, liD));
                    double[] ePos = allStates[liD][earthIdx].P;
                    double[] eVel = allStates[liD][earthIdx].V;

                    for (double ad = Math.Max(ld + 15, arriveStart); ad <= arriveEnd; ad += aStep)
                    {
                        int aiD = (int)Math.Round(ad);
                        aiD = Math.Max(0, Math.Min(maxDays, aiD));
                        double tof = (ad - ld) * DAY;
                        if (tof < 10 * DAY) continue;

                        double[] aPos = allStates[aiD][astIdx].P;
                        double[] aVel = allStates[aiD][astIdx].V;

                        var result = LambertSolve(ePos, aPos, tof, muSun, false);
                        if (result == null) continue;
                        var (v1, v2) = result.Value;

                        double[] dvLaunch = Sub3(v1, eVel);
                        double ldv = Norm3(dvLaunch);

                        if (ldv < bestLaunchDV && ldv <= dvBudget * 0.95)
                        {
                            bestLaunchDV    = ldv;
                            bestLaunchDay   = ld;
                            bestArriveDay   = ad;
                            bestLaunchDVVec = dvLaunch;
                            bestImpactVSC   = v2;
                            bestImpactVAst  = aVel;
                        }
                    }
                }
            }
        }

        // Fallback if no solution found within budget
        if (bestLaunchDay < 0)
        {
            Console.Error.WriteLine("No solution within budget, using minimum dV solution...");
            double minDV = double.MaxValue;
            launchEnd  = warnDays * 0.4;
            arriveEnd  = interceptWindow;

            for (double ld = 0; ld <= launchEnd; ld += launchEnd/60.0)
            {
                int liD = Math.Max(0, Math.Min(maxDays, (int)ld));
                double[] ePos = allStates[liD][earthIdx].P;
                double[] eVel = allStates[liD][earthIdx].V;

                for (double ad = ld + 15; ad <= arriveEnd; ad += (arriveEnd - ld)/60.0)
                {
                    int aiD = Math.Max(0, Math.Min(maxDays, (int)ad));
                    double tof = (ad - ld) * DAY;
                    if (tof < 10 * DAY) continue;
                    double[] aPos = allStates[aiD][astIdx].P;
                    double[] aVel = allStates[aiD][astIdx].V;

                    foreach (bool prog in new[]{true,false})
                    {
                        var r = LambertSolve(ePos, aPos, tof, muSun, prog);
                        if (r == null) continue;
                        var (v1,v2) = r.Value;
                        double[] dvL = Sub3(v1, eVel);
                        double ldv = Norm3(dvL);
                        if (ldv < minDV)
                        {
                            minDV = ldv;
                            bestLaunchDay   = ld;
                            bestArriveDay   = ad;
                            bestLaunchDVVec = dvL;
                            bestImpactVSC   = v2;
                            bestImpactVAst  = aVel;
                            bestLaunchDV    = ldv;
                        }
                    }
                }
            }
        }

        // If still nothing, use trivial solution
        if (bestLaunchDay < 0)
        {
            bestLaunchDay = 1;
            bestArriveDay = Math.Max(2, interceptWindow * 0.5);
            bestLaunchDVVec = new[] { 1000.0, 0, 0 };
            bestImpactVAst  = CopyV(allStates[(int)bestArriveDay][astIdx].V);
            bestImpactVSC   = Add3(bestImpactVAst, new[]{500.0, 0, 0});
        }

        Console.Error.WriteLine($"Best: launch={bestLaunchDay:F1}d, arrive={bestArriveDay:F1}d, dV={bestLaunchDV:F1} m/s");

        // --- Mid-course correction ---
        // Compute actual spacecraft trajectory from launch to intercept and find correction
        // We do a simple split: propagate first half, compute Lambert for second half
        bool doMidCourse = (bestArriveDay - bestLaunchDay) > 60 || dvBudget < 3000;
        double correctionDay = -1;
        double[] corrDV = new double[3];

        if (doMidCourse)
        {
            correctionDay = (bestLaunchDay + bestArriveDay) / 2.0;
            int liD = Math.Max(0, Math.Min(maxDays, (int)bestLaunchDay));
            int ciD = Math.Max(0, Math.Min(maxDays, (int)correctionDay));
            int aiD = Math.Max(0, Math.Min(maxDays, (int)bestArriveDay));

            double[] scP = CopyV(allStates[liD][earthIdx].P);
            double[] scV = Add3(allStates[liD][earthIdx].V, bestLaunchDVVec);

            // Propagate spacecraft to correction point
            int corrSteps = ciD - liD;
            Body[] bodiesAtLaunch = allStates[liD];

            // Simple propagation: assume two-body (sun only) for efficiency
            // Propagate corrSteps days
            double[] scPcorr = scP, scVcorr = scV;
            for (int s = 0; s < corrSteps; s++)
            {
                // Use bodies at midpoint (interpolate crudely)
                int bIdx = Math.Min(liD + s, maxDays);
                (scPcorr, scVcorr) = PropagateSC(scPcorr, scVcorr, allStates[bIdx], DAY);
            }

            // Now compute Lambert from corrected position to asteroid at arrival
            double[] aPos2 = allStates[aiD][astIdx].P;
            double tof2 = (bestArriveDay - correctionDay) * DAY;

            var corr = LambertSolve(scPcorr, aPos2, tof2, muSun, true);
            if (corr == null) corr = LambertSolve(scPcorr, aPos2, tof2, muSun, false);

            if (corr != null)
            {
                var (vcorr1, vcorr2) = corr.Value;
                corrDV = Sub3(vcorr1, scVcorr);
                bestImpactVSC = vcorr2;
                bestImpactVAst = allStates[aiD][astIdx].V;
                Console.Error.WriteLine($"Mid-course dV: {Norm3(corrDV):F2} m/s");
            }
            else
            {
                doMidCourse = false;
                Console.Error.WriteLine("Mid-course correction Lambert failed, skipping.");
            }
        }

        // --- Compute impact velocity vector ---
        // Impact delta-v is the relative velocity of spacecraft w.r.t. asteroid
        double[] relVelAtImpact = Sub3(bestImpactVSC, bestImpactVAst);
        // The "impact burn" in the output represents the momentum transfer direction
        // Use the spacecraft's approach velocity vector (toward asteroid)
        double impactSpeed = Norm3(relVelAtImpact);

        Console.Error.WriteLine($"Impact speed: {impactSpeed:F0} m/s");
        Console.Error.WriteLine($"Launch dV: {Norm3(bestLaunchDVVec):F1} m/s");

        // Check deflection adequacy: momentum transfer p = 2 * m_sc * v_rel (elastic)
        // More realistically use beta factor ~1-10, here just verify impact
        double momentumTransfer = scMass * impactSpeed * 2.0; // approximate
        double astAccel = momentumTransfer / astMass;
        Console.Error.WriteLine($"Momentum transfer: {momentumTransfer:E3} kg*m/s, asteroid accel: {astAccel:E3} m/s");

        // Scale launch dv to use more budget if deflection might be marginal
        // and we have budget remaining
        double usedDV = Norm3(bestLaunchDVVec) + (doMidCourse ? Norm3(corrDV) : 0);
        Console.Error.WriteLine($"Total dV used: {usedDV:F1} / {dvBudget:F1} m/s");

        // --- Output ---
        int numBurns = doMidCourse ? 3 : 2;
        Console.WriteLine(numBurns);
        Console.WriteLine($"{bestLaunchDay:G6} {bestLaunchDVVec[0]:G10} {bestLaunchDVVec[1]:G10} {bestLaunchDVVec[2]:G10}");
        if (doMidCourse)
            Console.WriteLine($"{correctionDay:G6} {corrDV[0]:G10} {corrDV[1]:G10} {corrDV[2]:G10}");
        Console.WriteLine($"{bestArriveDay:G6} {relVelAtImpact[0]:G10} {relVelAtImpact[1]:G10} {relVelAtImpact[2]:G10}");
    }
