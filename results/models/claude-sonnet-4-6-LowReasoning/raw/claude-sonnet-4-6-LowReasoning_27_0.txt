using namespace std;
using Clock = chrono::steady_clock;

static Clock::time_point g_start;
static double g_time_limit = 280.0; // seconds

double elapsed() {
    return chrono::duration<double>(Clock::now() - g_start).count();
}

double time_remaining() {
    return g_time_limit - elapsed();
}

// ---- Graph structure ----
struct Graph {
    int n, m, k;
    vector<vector<int>> adj;
    vector<pair<int,int>> edges;

    void build(int n_, int m_, int k_) {
        n = n_; m = m_; k = k_;
        adj.assign(n, {});
        edges.reserve(m);
    }
    void add_edge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
        edges.push_back({u, v});
    }
};

// ---- DSatur coloring ----
vector<int> dsatur(const Graph& g) {
    int n = g.n, k = g.k;
    vector<int> color(n, -1);
    vector<int> saturation(n, 0);
    vector<int> degree(n, 0);
    vector<unordered_set<int>> neighbor_colors(n);

    for (int i = 0; i < n; i++) degree[i] = (int)g.adj[i].size();

    // Priority queue: max by (saturation, degree)
    auto cmp = [&](int a, int b) {
        if (saturation[a] != saturation[b]) return saturation[a] < saturation[b];
        return degree[a] < degree[b];
    };
    // Use simple selection for correctness
    vector<bool> colored(n, false);

    for (int step = 0; step < n; step++) {
        // Find uncolored vertex with max saturation (tie-break: degree)
        int v = -1;
        for (int i = 0; i < n; i++) {
            if (!colored[i]) {
                if (v == -1 || saturation[i] > saturation[v] ||
                    (saturation[i] == saturation[v] && degree[i] > degree[v]))
                    v = i;
            }
        }
        // Find smallest color not used by neighbors
        int c = 0;
        while (neighbor_colors[v].count(c)) c++;
        if (c >= k) {
            // Use mod fallback
            c = c % k;
        }
        color[v] = c;
        colored[v] = true;
        // Update saturation of uncolored neighbors
        for (int u : g.adj[v]) {
            if (!colored[u]) {
                if (!neighbor_colors[u].count(c)) {
                    neighbor_colors[u].insert(c);
                    saturation[u]++;
                }
            }
        }
    }
    return color;
}

// ---- Greedy coloring (fastest, for huge graphs) ----
vector<int> greedy_color(const Graph& g, const vector<int>& order) {
    int n = g.n, k = g.k;
    vector<int> color(n, -1);
    vector<bool> forbidden(k, false);

    for (int v : order) {
        fill(forbidden.begin(), forbidden.end(), false);
        for (int u : g.adj[v]) {
            if (color[u] >= 0 && color[u] < k)
                forbidden[color[u]] = true;
        }
        int c = 0;
        while (c < k && forbidden[c]) c++;
        color[v] = (c < k) ? c : (rand() % k);
    }
    return color;
}

// ---- Count conflicts ----
int count_conflicts(const Graph& g, const vector<int>& color) {
    int conflicts = 0;
    for (auto& [u, v] : g.edges)
        if (color[u] == color[v]) conflicts++;
    return conflicts;
}

// ---- Local search: Tabu Search ----
// For each conflicted vertex, try recoloring to minimize conflicts.
bool tabu_search(const Graph& g, vector<int>& color, double time_budget) {
    int n = g.n, k = g.k;
    int conflicts = count_conflicts(g, color);
    if (conflicts == 0) return true;

    mt19937 rng(42);

    // tabu tenure
    int tabu_tenure = max(5, min(n / 100, 20));
    // For large graphs limit tabu memory; use flat array with stride k
    // Max ~64M ints = 256MB, cap at reasonable size
    bool use_tabu = ((long long)n * k <= 64000000LL);
    vector<int> tabu;
    if (use_tabu) tabu.assign((long long)n * k, 0);
    int iter = 0;
    int best_conflicts = conflicts;
    vector<int> best_color = color;

    // Neighbor conflict count: for each vertex, how many neighbors have same color
    vector<int> vconflicts(n, 0);
    for (auto& [u, v] : g.edges) {
        if (color[u] == color[v]) {
            vconflicts[u]++;
            vconflicts[v]++;
        }
    }

    // Maintain a list of conflicted vertices
    vector<int> conf_list;
    vector<int> conf_pos(n, -1); // position in conf_list, -1 if not present
    conf_list.reserve(n);
    for (int i = 0; i < n; i++) {
        if (vconflicts[i] > 0) {
            conf_pos[i] = (int)conf_list.size();
            conf_list.push_back(i);
        }
    }

    auto add_to_conf = [&](int x) {
        if (conf_pos[x] == -1) {
            conf_pos[x] = (int)conf_list.size();
            conf_list.push_back(x);
        }
    };
    auto remove_from_conf = [&](int x) {
        if (conf_pos[x] != -1) {
            int pos = conf_pos[x];
            int last = conf_list.back();
            conf_list[pos] = last;
            conf_pos[last] = pos;
            conf_list.pop_back();
            conf_pos[x] = -1;
        }
    };

    double deadline = elapsed() + time_budget;
    while (elapsed() < deadline) {
        if (conflicts == 0) return true;
        if (conf_list.empty()) return true;

        // Pick random conflicted vertex
        int v = conf_list[rng() % conf_list.size()];

        // Find best recoloring for v
        // Count conflicts for each color
        vector<int> color_conflicts(k, 0);
        for (int u : g.adj[v]) {
            if (color[u] >= 0 && color[u] < k)
                color_conflicts[color[u]]++;
        }

        int best_c = -1, best_delta = INT_MAX;
        for (int c = 0; c < k; c++) {
            if (c == color[v]) continue;
            if (use_tabu && tabu[(long long)v * k + c] > iter) continue; // tabu
            int delta = color_conflicts[c] - vconflicts[v];
            if (delta < best_delta) {
                best_delta = delta;
                best_c = c;
            }
        }

        // If all moves tabu, pick aspiration or random
        if (best_c == -1) {
            // Aspiration: allow if would improve best
            for (int c = 0; c < k; c++) {
                if (c == color[v]) continue;
                int delta = color_conflicts[c] - vconflicts[v];
                int new_conf = conflicts + delta;
                if (new_conf < best_conflicts) {
                    best_c = c;
                    best_delta = delta;
                    break;
                }
            }
            if (best_c == -1)
                best_c = rng() % k;
        }

        // Apply move
        int old_c = color[v];
        if (use_tabu) tabu[(long long)v * k + old_c] = iter + tabu_tenure;

        // Update vconflicts and conf_list
        for (int u : g.adj[v]) {
            if (color[u] == old_c) {
                vconflicts[v]--;
                vconflicts[u]--;
                conflicts--;
                if (vconflicts[u] == 0) remove_from_conf(u);
            }
        }
        color[v] = best_c;
        for (int u : g.adj[v]) {
            if (color[u] == best_c) {
                vconflicts[v]++;
                vconflicts[u]++;
                conflicts++;
                add_to_conf(u);
            }
        }
        if (vconflicts[v] == 0) remove_from_conf(v);
        else add_to_conf(v);

        if (conflicts < best_conflicts) {
            best_conflicts = conflicts;
            best_color = color;
        }

        iter++;

        // Periodic restart from best
        if (iter % 50000 == 0) {
            if (best_conflicts == 0) { color = best_color; return true; }
            color = best_color;
            conflicts = best_conflicts;
            // Recompute vconflicts and conf_list
            fill(vconflicts.begin(), vconflicts.end(), 0);
            conf_list.clear();
            fill(conf_pos.begin(), conf_pos.end(), -1);
            for (auto& [u2, v2] : g.edges) {
                if (color[u2] == color[v2]) {
                    vconflicts[u2]++;
                    vconflicts[v2]++;
                }
            }
            for (int i = 0; i < n; i++) {
                if (vconflicts[i] > 0) {
                    conf_pos[i] = (int)conf_list.size();
                    conf_list.push_back(i);
                }
            }
        }
    }

    color = best_color;
    return best_conflicts == 0;
}

// ---- Simulated Annealing for medium graphs ----
bool simulated_annealing(const Graph& g, vector<int>& color, double time_budget) {
    int n = g.n, k = g.k;
    mt19937 rng(123);

    int conflicts = count_conflicts(g, color);
    if (conflicts == 0) return true;

    double T = (double)conflicts;
    double T_min = 0.01;
    double alpha = 0.9999;

    vector<int> best_color = color;
    int best_conflicts = conflicts;

    // vconflicts per vertex
    vector<int> vconf(n, 0);
    for (auto& [u, v] : g.edges)
        if (color[u] == color[v]) { vconf[u]++; vconf[v]++; }

    int iter = 0;
    double sa_deadline = elapsed() + time_budget;
    while (elapsed() < sa_deadline && T > T_min) {
        // Pick random vertex
        int v = rng() % n;
        int old_c = color[v];
        int new_c = rng() % k;
        if (new_c == old_c) { T *= alpha; continue; }

        // Compute delta
        int delta = 0;
        for (int u : g.adj[v]) {
            if (color[u] == new_c) delta++;
            if (color[u] == old_c) delta--;
        }

        if (delta < 0 || (T > 0 && (double)(rng() % 1000000) / 1000000.0 < exp(-delta / T))) {
            // Update vconf
            for (int u : g.adj[v]) {
                if (color[u] == old_c) { vconf[v]--; vconf[u]--; conflicts--; }
            }
            color[v] = new_c;
            for (int u : g.adj[v]) {
                if (color[u] == new_c) { vconf[v]++; vconf[u]++; conflicts++; }
            }
            if (conflicts < best_conflicts) {
                best_conflicts = conflicts;
                best_color = color;
                if (best_conflicts == 0) { return true; }
            }
        }
        T *= alpha;
        iter++;
    }

    color = best_color;
    return best_conflicts == 0;
}

// ---- Backtracking with forward checking for small graphs ----
struct BTSolver {
    const Graph& g;
    vector<int>& color;
    int n, k;
    vector<int> order; // vertex ordering

    BTSolver(const Graph& g_, vector<int>& c) : g(g_), color(c), n(g_.n), k(g_.k) {}

    bool solve() {
        // Order by degree descending
        order.resize(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b){
            return g.adj[a].size() > g.adj[b].size();
        });
        fill(color.begin(), color.end(), -1);
        return backtrack(0);
    }

    bool backtrack(int idx) {
        if (elapsed() > g_time_limit - 2.0) return false;
        if (idx == n) return true;
        int v = order[idx];

        // Forbidden colors
        vector<bool> forbidden(k, false);
        for (int u : g.adj[v])
            if (color[u] >= 0) forbidden[color[u]] = true;

        for (int c = 0; c < k; c++) {
            if (!forbidden[c]) {
                color[v] = c;
                if (backtrack(idx + 1)) return true;
                color[v] = -1;
            }
        }
        return false;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    g_start = Clock::now();

    int n, m, k;
    cin >> n >> m >> k;

    Graph g;
    g.build(n, m, k);

    for (int i = 0; i < m; i++) {
        int u, v;
        cin >> u >> v;
        if (u != v) g.add_edge(u, v);
    }

    vector<int> color(n, 0);

    if (n == 0) {
        // nothing
    } else if (k == 1) {
        // Only valid if no edges
        fill(color.begin(), color.end(), 0);
    } else if (n <= 100 && m <= 5000) {
        // Small: try backtracking first
        BTSolver bt(g, color);
        bool solved = bt.solve();
        if (!solved) {
            // Fall back to dsatur + tabu
            color = dsatur(g);
            if (count_conflicts(g, color) > 0)
                tabu_search(g, color, g_time_limit - elapsed());
        }
    } else if (n <= 50000) {
        // Medium: DSatur + tabu search
        color = dsatur(g);
        int conf = count_conflicts(g, color);
        if (conf > 0) {
            tabu_search(g, color, g_time_limit - elapsed());
        }
    } else {
        // Large: greedy (degree-ordered) + tabu search
        vector<int> order(n);
        iota(order.begin(), order.end(), 0);
        sort(order.begin(), order.end(), [&](int a, int b){
            return g.adj[a].size() > g.adj[b].size();
        });
        color = greedy_color(g, order);
        int conf = count_conflicts(g, color);
        if (conf > 0) {
            tabu_search(g, color, g_time_limit - elapsed());
        }
    }

    // Ensure all colors are in range [0, k-1]
    for (int i = 0; i < n; i++) {
        if (color[i] < 0 || color[i] >= k) color[i] = 0;
    }

    // Output
    for (int i = 0; i < n; i++) {
        cout << color[i];
        if (i + 1 < n) cout << ' ';
    }
    cout << '\n';

    return 0;
}
