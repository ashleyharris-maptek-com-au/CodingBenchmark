; SPIR-V Geometry Shader: Expand input points into hexagons (6 triangles, 18 vertices)
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %out_color
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

; Decorations for built-in blocks
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; User-defined input/output locations
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal   Location 1
               OpDecorate %in_color    Location 2
               OpDecorate %out_color   Location 0

; --- Types ---
       %void = OpTypeVoid
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
   %voidfunc = OpTypeFunction %void

; --- Constants ---
      %int_0 = OpConstant %int 0
     %uint_1 = OpConstant %uint 1

; Hex edge offsets for r = 0.04
;   e0: ( r,    0  )   e1: ( r/2, r*sin60 )   e2: (-r/2, r*sin60 )
;   e3: (-r,    0  )   e4: (-r/2,-r*sin60 )   e5: ( r/2,-r*sin60 )
   %float_r  = OpConstant %float  0.04
   %float_hr = OpConstant %float  0.02
   %float_sr = OpConstant %float  0.034641016
   %float_nr = OpConstant %float -0.04
  %float_nhr = OpConstant %float -0.02
  %float_nsr = OpConstant %float -0.034641016
%float_dark  = OpConstant %float  0.7

; --- Struct types ---
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; --- Array types (InputPoints => size 1) ---
 %_arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
  %_arr_v3_1 = OpTypeArray %v3float %uint_1

; --- Pointer types ---
%_ptr_in_arr_pv  = OpTypePointer Input  %_arr_pv_in
%_ptr_in_v4      = OpTypePointer Input  %v4float
%_ptr_in_arr_v3  = OpTypePointer Input  %_arr_v3_1
%_ptr_in_v3      = OpTypePointer Input  %v3float
%_ptr_out_pv     = OpTypePointer Output %gl_PerVertex_out
%_ptr_out_v4     = OpTypePointer Output %v4float
%_ptr_out_v3     = OpTypePointer Output %v3float

; --- Variables ---
      %gl_in = OpVariable %_ptr_in_arr_pv  Input
 %in_worldPos = OpVariable %_ptr_in_arr_v3 Input
   %in_normal = OpVariable %_ptr_in_arr_v3 Input
    %in_color = OpVariable %_ptr_in_arr_v3 Input
      %gl_out = OpVariable %_ptr_out_pv    Output
   %out_color = OpVariable %_ptr_out_v3    Output

; --- Main function ---
      %main = OpFunction %void None %voidfunc
     %entry = OpLabel

; Read center position: gl_in[0].gl_Position
   %ptr_pos = OpAccessChain %_ptr_in_v4 %gl_in %int_0 %int_0
       %pos = OpLoad %v4float %ptr_pos

; Read vertex color: in_color[0]
   %ptr_col = OpAccessChain %_ptr_in_v3 %in_color %int_0
       %col = OpLoad %v3float %ptr_col

; Edge color = center color * 0.7 (slightly darker)
  %edge_col = OpVectorTimesScalar %v3float %col %float_dark

; Extract center x, y, z, w
        %cx = OpCompositeExtract %float %pos 0
        %cy = OpCompositeExtract %float %pos 1
        %cz = OpCompositeExtract %float %pos 2
        %cw = OpCompositeExtract %float %pos 3

; Compute edge X coordinates
       %e0x = OpFAdd %float %cx %float_r
       %e1x = OpFAdd %float %cx %float_hr
       %e2x = OpFAdd %float %cx %float_nhr
       %e3x = OpFAdd %float %cx %float_nr
; e4x == e2x, e5x == e1x (reused below)

; Compute edge Y coordinates
       %e1y = OpFAdd %float %cy %float_sr
; e2y == e1y (same positive offset)
       %e4y = OpFAdd %float %cy %float_nsr
; e5y == e4y (same negative offset)

; Build edge positions (z and w same as center)
;   e0: (cx+r,   cy,  cz, cw)
;   e1: (cx+hr,  cy+sr, cz, cw)
;   e2: (cx-hr,  cy+sr, cz, cw)
;   e3: (cx-r,   cy,  cz, cw)
;   e4: (cx-hr,  cy-sr, cz, cw)
;   e5: (cx+hr,  cy-sr, cz, cw)
      %e0p = OpCompositeConstruct %v4float %e0x %cy  %cz %cw
      %e1p = OpCompositeConstruct %v4float %e1x %e1y %cz %cw
      %e2p = OpCompositeConstruct %v4float %e2x %e1y %cz %cw
      %e3p = OpCompositeConstruct %v4float %e3x %cy  %cz %cw
      %e4p = OpCompositeConstruct %v4float %e2x %e4y %cz %cw
      %e5p = OpCompositeConstruct %v4float %e1x %e4y %cz %cw

; Output gl_Position pointer
%out_pos_ptr = OpAccessChain %_ptr_out_v4 %gl_out %int_0

; === Triangle 0: center, e0, e1 ===
             OpStore %out_pos_ptr %pos
             OpStore %out_color   %col
             OpEmitVertex
             OpStore %out_pos_ptr %e0p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpStore %out_pos_ptr %e1p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpEndPrimitive

; === Triangle 1: center, e1, e2 ===
             OpStore %out_pos_ptr %pos
             OpStore %out_color   %col
             OpEmitVertex
             OpStore %out_pos_ptr %e1p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpStore %out_pos_ptr %e2p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpEndPrimitive

; === Triangle 2: center, e2, e3 ===
             OpStore %out_pos_ptr %pos
             OpStore %out_color   %col
             OpEmitVertex
             OpStore %out_pos_ptr %e2p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpStore %out_pos_ptr %e3p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpEndPrimitive

; === Triangle 3: center, e3, e4 ===
             OpStore %out_pos_ptr %pos
             OpStore %out_color   %col
             OpEmitVertex
             OpStore %out_pos_ptr %e3p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpStore %out_pos_ptr %e4p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpEndPrimitive

; === Triangle 4: center, e4, e5 ===
             OpStore %out_pos_ptr %pos
             OpStore %out_color   %col
             OpEmitVertex
             OpStore %out_pos_ptr %e4p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpStore %out_pos_ptr %e5p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpEndPrimitive

; === Triangle 5: center, e5, e0 ===
             OpStore %out_pos_ptr %pos
             OpStore %out_color   %col
             OpEmitVertex
             OpStore %out_pos_ptr %e5p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpStore %out_pos_ptr %e0p
             OpStore %out_color   %edge_col
             OpEmitVertex
             OpEndPrimitive

             OpReturn
             OpFunctionEnd
