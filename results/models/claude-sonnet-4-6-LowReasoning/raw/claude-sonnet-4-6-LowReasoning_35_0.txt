use std::io::{self, BufRead, Write};
use std::time::{Duration, Instant};
use std::collections::BinaryHeap;
use std::cmp::Reverse;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut iter = first_line.split_whitespace();
    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut it = line.split_whitespace();
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let deadline = Instant::now() + Duration::from_secs(270);

    // Greedy construction
    let mut best = greedy_dominating_set(n, &adj);

    // Local search improvement
    best = local_search(n, &adj, best, &deadline);

    // Try multiple random restarts
    let mut rng_state: u64 = 12345678901234567;
    while Instant::now() < deadline {
        let candidate = greedy_randomized(n, &adj, &mut rng_state);
        let candidate = local_search(n, &adj, candidate, &deadline);
        if candidate.len() < best.len() {
            best = candidate;
        }
    }

    let stdout = io::stdout();
    let mut out = stdout.lock();
    writeln!(out, "{}", best.len()).unwrap();
    let parts: Vec<String> = best.iter().map(|v| v.to_string()).collect();
    writeln!(out, "{}", parts.join(" ")).unwrap();
}

fn greedy_dominating_set(n: usize, adj: &[Vec<usize>]) -> Vec<usize> {
    // Score = number of undominated vertices this vertex would cover (itself + neighbors)
    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut undom_count = n; // number of undominated vertices

    // For each vertex, track how many undominated vertices it can cover
    let mut cover_count = vec![0usize; n];
    for v in 0..n {
        cover_count[v] = 1 + adj[v].len(); // itself + all neighbors (initially all undominated)
    }

    let mut result = Vec::new();

    while undom_count > 0 {
        // Pick vertex with highest cover count
        let best_v = (0..n)
            .filter(|&v| !in_set[v])
            .max_by_key(|&v| cover_count[v])
            .unwrap();

        in_set[best_v] = true;
        result.push(best_v);

        // Dominate best_v and its neighbors
        let to_dominate: Vec<usize> = std::iter::once(best_v)
            .chain(adj[best_v].iter().copied())
            .collect();

        for &u in &to_dominate {
            if !dominated[u] {
                dominated[u] = true;
                undom_count -= 1;
                // Update cover counts for u and u's neighbors
                // u is now dominated, so vertices that could cover u lose one count
                for &w in std::iter::once(&u).chain(adj[u].iter()) {
                    if cover_count[*w] > 0 {
                        cover_count[*w] -= 1;
                    }
                }
            }
        }
    }

    result
}

fn greedy_randomized(n: usize, adj: &[Vec<usize>], rng: &mut u64) -> Vec<usize> {
    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut undom_count = n;

    let mut cover_count = vec![0usize; n];
    for v in 0..n {
        cover_count[v] = 1 + adj[v].len();
    }

    let mut result = Vec::new();

    while undom_count > 0 {
        // Find max cover count
        let max_cover = (0..n)
            .filter(|&v| !in_set[v])
            .map(|v| cover_count[v])
            .max()
            .unwrap_or(0);

        // Collect candidates within some range of max (top candidates)
        let threshold = if max_cover > 2 { max_cover - max_cover / 4 } else { max_cover };
        let candidates: Vec<usize> = (0..n)
            .filter(|&v| !in_set[v] && cover_count[v] >= threshold)
            .collect();

        let idx = lcg_rand(rng) as usize % candidates.len();
        let best_v = candidates[idx];

        in_set[best_v] = true;
        result.push(best_v);

        let to_dominate: Vec<usize> = std::iter::once(best_v)
            .chain(adj[best_v].iter().copied())
            .collect();

        for &u in &to_dominate {
            if !dominated[u] {
                dominated[u] = true;
                undom_count -= 1;
                for &w in std::iter::once(&u).chain(adj[u].iter()) {
                    if cover_count[*w] > 0 {
                        cover_count[*w] -= 1;
                    }
                }
            }
        }
    }

    result
}

fn lcg_rand(state: &mut u64) -> u64 {
    *state = state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
    *state >> 33
}

fn local_search(n: usize, adj: &[Vec<usize>], mut set: Vec<usize>, deadline: &Instant) -> Vec<usize> {
    let mut in_set = vec![false; n];
    for &v in &set {
        in_set[v] = true;
    }

    // Count how many set-vertices cover each vertex (dominators)
    // dom_count[v] = number of vertices in set that dominate v (set members + set neighbors)
    let mut dom_count = vec![0usize; n];
    for &v in &set {
        dom_count[v] += 1;
        for &u in &adj[v] {
            dom_count[u] += 1;
        }
    }

    let mut improved = true;
    while improved && Instant::now() < *deadline {
        improved = false;

        // Try to remove a vertex from the set
        let set_snapshot = set.clone();
        for &v in &set_snapshot {
            if Instant::now() >= *deadline {
                break;
            }
            // v can be removed if every vertex it uniquely dominates is also dominated by another set member
            // v dominates itself and its neighbors
            // A vertex u is "uniquely dominated by v" if dom_count[u] == 1
            let can_remove = std::iter::once(v)
                .chain(adj[v].iter().copied())
                .all(|u| dom_count[u] > 1);

            if can_remove {
                // Remove v
                in_set[v] = false;
                dom_count[v] -= 1;
                for &u in &adj[v] {
                    dom_count[u] -= 1;
                }
                set.retain(|&x| x != v);
                improved = true;
            }
        }

        if Instant::now() >= *deadline {
            break;
        }

        // Try swap: remove one vertex, add a better replacement
        // Find vertices with dom_count == 1 (only dominated by one set member)
        // For each set vertex v, check if removing it leaves any vertex undominated
        // If so, try to find a non-set vertex that covers all those vertices
        let set_snapshot = set.clone();
        'outer: for &v in &set_snapshot {
            if Instant::now() >= *deadline {
                break;
            }

            // Find vertices uniquely dominated by v
            let uniquely_dominated: Vec<usize> = std::iter::once(v)
                .chain(adj[v].iter().copied())
                .filter(|&u| dom_count[u] == 1)
                .collect();

            if uniquely_dominated.is_empty() {
                continue;
            }

            // Find a non-set vertex that covers all uniquely dominated vertices
            // A vertex w covers u if w == u or w is adjacent to u
            for &w in &adj[v] {
                if in_set[w] {
                    continue;
                }
                // Check if w covers all uniquely_dominated
                let w_covers: std::collections::HashSet<usize> = std::iter::once(w)
                    .chain(adj[w].iter().copied())
                    .collect();

                if uniquely_dominated.iter().all(|&u| w_covers.contains(&u)) {
                    // Swap v for w
                    in_set[v] = false;
                    dom_count[v] -= 1;
                    for &u in &adj[v] {
                        dom_count[u] -= 1;
                    }

                    in_set[w] = true;
                    dom_count[w] += 1;
                    for &u in &adj[w] {
                        dom_count[u] += 1;
                    }

                    set.retain(|&x| x != v);
                    set.push(w);
                    improved = true;
                    continue 'outer;
                }
            }

            // Also check distance-2 vertices
            for &nb in &adj[v] {
                for &w in &adj[nb] {
                    if in_set[w] || w == v {
                        continue;
                    }
                    let w_covers: std::collections::HashSet<usize> = std::iter::once(w)
                        .chain(adj[w].iter().copied())
                        .collect();

                    if uniquely_dominated.iter().all(|&u| w_covers.contains(&u)) {
                        in_set[v] = false;
                        dom_count[v] -= 1;
                        for &u in &adj[v] {
                            dom_count[u] -= 1;
                        }

                        in_set[w] = true;
                        dom_count[w] += 1;
                        for &u in &adj[w] {
                            dom_count[u] += 1;
                        }

                        set.retain(|&x| x != v);
                        set.push(w);
                        improved = true;
                        continue 'outer;
                    }
                }
            }
        }
    }

    set
}
