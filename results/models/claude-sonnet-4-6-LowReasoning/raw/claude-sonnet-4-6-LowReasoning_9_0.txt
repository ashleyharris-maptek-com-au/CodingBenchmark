using System;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;
using System.Threading;

class Solution
{
    static void Main(string[] args)
    {
        var reader = new StreamReader(new BufferedStream(Console.OpenStandardInput(), 1 << 20));

        string line1 = reader.ReadLine();
        var parts1 = line1.Split(' ');
        int n = int.Parse(parts1[0]);
        int stockLength = int.Parse(parts1[1]);

        string line2 = reader.ReadLine();
        var parts2 = line2.Split(' ');

        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
            cuts[i] = int.Parse(parts2[i]);

        // Sort indices by cut length descending (First Fit Decreasing)
        int[] sortedIdx = new int[n];
        for (int i = 0; i < n; i++) sortedIdx[i] = i;
        Array.Sort(sortedIdx, (a, b) => cuts[b].CompareTo(cuts[a]));

        // Use a sorted multiset (SortedDictionary<remaining, list of bin ids>) to find best fit
        // Key: remaining capacity, Value: stack of bin indices with that remaining capacity
        var remainingToBins = new SortedDictionary<int, Stack<int>>();
        var binRemaining = new List<int>();
        var binContents = new List<List<int>>();

        foreach (int idx in sortedIdx)
        {
            int cut = cuts[idx];

            // Find smallest remaining capacity >= cut (Best Fit Decreasing variant)
            // Use SortedDictionary to find the view >= cut
            bool placed = false;

            // Try to find a bin with remaining capacity exactly >= cut
            // We want the smallest such remaining to minimize waste
            var view = remainingToBins.GetViewBetween(cut, stockLength);
            if (view.Count > 0)
            {
                var en = view.GetEnumerator();
                en.MoveNext();
                var kv = en.Current;
                int rem = kv.Key;
                var stack = kv.Value;
                int binId = stack.Pop();
                if (stack.Count == 0)
                    remainingToBins.Remove(rem);

                int newRem = rem - cut;
                binRemaining[binId] = newRem;
                binContents[binId].Add(idx);

                if (newRem > 0)
                {
                    if (!remainingToBins.TryGetValue(newRem, out var s))
                    {
                        s = new Stack<int>();
                        remainingToBins[newRem] = s;
                    }
                    s.Push(binId);
                }
                placed = true;
            }

            if (!placed)
            {
                int binId = binContents.Count;
                int rem = stockLength - cut;
                binRemaining.Add(rem);
                binContents.Add(new List<int> { idx });

                if (rem > 0)
                {
                    if (!remainingToBins.TryGetValue(rem, out var s))
                    {
                        s = new Stack<int>();
                        remainingToBins[rem] = s;
                    }
                    s.Push(binId);
                }
            }
        }

        int numBins = binContents.Count;

        // Output
        var writer = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1 << 20));
        writer.WriteLine(numBins);

        foreach (var bin in binContents)
        {
            writer.WriteLine(string.Join(" ", bin));
        }

        writer.Flush();
    }
}
