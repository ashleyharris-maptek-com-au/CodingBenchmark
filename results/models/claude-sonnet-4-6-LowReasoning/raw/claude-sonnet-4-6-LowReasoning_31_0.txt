use std::collections::HashMap;
use std::io::{Read, Write, BufWriter};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

fn main() {
    let start = Instant::now();
    let mut input = Vec::new();
    std::io::stdin().read_to_end(&mut input).unwrap();
    let stdout = std::io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut pos = 0usize;
    let n = parse_usize(&input, &mut pos);
    let target = parse_i64(&input, &mut pos);
    let mut nums: Vec<i64> = Vec::with_capacity(n);
    for _ in 0..n {
        nums.push(parse_i64(&input, &mut pos));
    }
    drop(input);
    match solve(&nums, target, start) {
        Some(indices) => {
            writeln!(out, "YES").unwrap();
            let s: Vec<String> = indices.iter().map(|x| x.to_string()).collect();
            writeln!(out, "[{}]", s.join(" ")).unwrap();
        }
        None => writeln!(out, "NO").unwrap(),
    }
}

fn parse_i64(buf: &[u8], pos: &mut usize) -> i64 {
    while *pos < buf.len() && buf[*pos] <= b' ' { *pos += 1; }
    let neg = *pos < buf.len() && buf[*pos] == b'-';
    if neg { *pos += 1; }
    let mut v = 0i64;
    while *pos < buf.len() && buf[*pos] >= b'0' && buf[*pos] <= b'9' {
        v = v * 10 + (buf[*pos] - b'0') as i64;
        *pos += 1;
    }
    if neg { -v } else { v }
}

fn parse_usize(buf: &[u8], pos: &mut usize) -> usize {
    while *pos < buf.len() && buf[*pos] <= b' ' { *pos += 1; }
    let mut v = 0usize;
    while *pos < buf.len() && buf[*pos] >= b'0' && buf[*pos] <= b'9' {
        v = v * 10 + (buf[*pos] - b'0') as usize;
        *pos += 1;
    }
    v
}

// dp |= dp shifted left by x bits (for positive x) or right by |x| bits (for negative x)
fn apply_elem(dp: &mut [u64], x: i64, n_words: usize) {
    if x == 0 || n_words == 0 { return; }
    if x > 0 {
        let shift = x as usize;
        if shift >= n_words * 64 { return; }
        let ws = shift / 64;
        let bs = shift % 64;
        if bs == 0 {
            for i in (ws..n_words).rev() {
                dp[i] |= dp[i - ws];
            }
        } else {
            let rs = 64 - bs;
            for i in (ws..n_words).rev() {
                let hi = dp[i - ws] << bs;
                let lo = if i > ws { dp[i - ws - 1] >> rs } else { 0 };
                dp[i] |= hi | lo;
            }
        }
    } else {
        let shift = (-x) as usize;
        if shift >= n_words * 64 { return; }
        let ws = shift / 64;
        let bs = shift % 64;
        if ws >= n_words { return; }
        let limit = n_words - ws;
        if bs == 0 {
            for i in 0..limit {
                dp[i] |= dp[i + ws];
            }
        } else {
            let rs = 64 - bs;
            for i in 0..limit {
                let lo = dp[i + ws] >> bs;
                let hi = if i + ws + 1 < n_words { dp[i + ws + 1] << rs } else { 0 };
                dp[i] |= lo | hi;
            }
        }
    }
}

#[inline]
fn bs_test(dp: &[u64], bit: usize) -> bool {
    let w = bit / 64;
    w < dp.len() && (dp[w] >> (bit % 64)) & 1 != 0
}

fn solve(nums: &[i64], target: i64, start: Instant) -> Option<Vec<usize>> {
    let n = nums.len();
    if target == 0 { return Some(vec![]); }
    if n == 0 { return None; }

    // Direct single-element match
    for (i, &x) in nums.iter().enumerate() {
        if x == target { return Some(vec![i]); }
    }

    let mut pos_sum = 0i64;
    let mut neg_sum = 0i64;
    for &x in nums {
        if x > 0 { pos_sum = pos_sum.saturating_add(x); }
        else if x < 0 { neg_sum = neg_sum.saturating_add(x); }
    }
    if target > pos_sum || target < neg_sum { return None; }

    // All elements sum to target
    let total: i64 = nums.iter().fold(0i64, |a, &x| a.saturating_add(x));
    if total == target { return Some((0..n).collect()); }

    if n <= 40 { return meet_in_middle(nums, target); }

    let range_i64 = pos_sum.saturating_sub(neg_sum);
    const MAX_RANGE: i64 = 200_000_000;

    if range_i64 > 0 && range_i64 <= MAX_RANGE {
        let range = range_i64 as usize;
        let n_words = range / 64 + 2;
        let k = ((n as f64).sqrt() as usize).max(2);
        let num_blocks = (n + k - 1) / k;
        let mem_bytes = (num_blocks + 1 + k) as u64 * n_words as u64 * 8;

        if mem_bytes <= 2_000_000_000 {
            return dp_with_backtrack(nums, target, neg_sum, n_words, k);
        }

        // DP confirm only, then random walk for indices
        let offset = (-neg_sum) as usize;
        let tbit_raw = target - neg_sum;
        if tbit_raw < 0 || tbit_raw as usize >= n_words * 64 { return None; }
        let tbit = tbit_raw as usize;

        let mut dp = vec![0u64; n_words];
        dp[offset / 64] |= 1u64 << (offset % 64);
        for &x in nums { apply_elem(&mut dp, x, n_words); }
        if !bs_test(&dp, tbit) { return None; }
        drop(dp);
        return random_walk_find(nums, target, start);
    }

    random_walk_find(nums, target, start)
}

fn dp_with_backtrack(
    nums: &[i64],
    target: i64,
    neg_sum: i64,
    n_words: usize,
    k: usize,
) -> Option<Vec<usize>> {
    let n = nums.len();
    let offset = (-neg_sum) as usize;
    let tbit_raw = target - neg_sum;
    if tbit_raw < 0 || tbit_raw as usize >= n_words * 64 { return None; }
    let tbit = tbit_raw as usize;

    let num_blocks = (n + k - 1) / k;
    let mut dp = vec![0u64; n_words];
    dp[offset / 64] |= 1u64 << (offset % 64);

    let mut checkpoints: Vec<Vec<u64>> = Vec::with_capacity(num_blocks + 1);
    checkpoints.push(dp.clone());

    for blk in 0..num_blocks {
        let l = blk * k;
        let r = ((blk + 1) * k).min(n);
        for i in l..r { apply_elem(&mut dp, nums[i], n_words); }
        checkpoints.push(dp.clone());
    }

    if !bs_test(&checkpoints[num_blocks], tbit) { return None; }

    let mut remaining = target;
    let mut result = Vec::new();

    'outer: for blk in (0..num_blocks).rev() {
        let l = blk * k;
        let r = ((blk + 1) * k).min(n);

        // Recompute within-block dp states
        let mut block_dp: Vec<Vec<u64>> = Vec::with_capacity(r - l + 1);
        let mut cur = checkpoints[blk].clone();
        block_dp.push(cur.clone());
        for i in l..r {
            apply_elem(&mut cur, nums[i], n_words);
            block_dp.push(cur.clone());
        }

        // Backtrack right to left within block
        for i in (l..r).rev() {
            let j = i - l;
            let rem_off = remaining - neg_sum;
            if rem_off < 0 || rem_off as usize >= n_words * 64 { return None; }
            let rbit = rem_off as usize;

            if bs_test(&block_dp[j], rbit) {
                // Can skip nums[i]
            } else {
                // Must include nums[i]
                result.push(i);
                remaining -= nums[i];
                if remaining == 0 { break 'outer; }
            }
        }
    }

    if remaining != 0 { return None; }
    result.sort_unstable();
    Some(result)
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let half = n / 2;
    let left = &nums[..half];
    let right = &nums[half..];
    let rn = right.len();
    let ln = left.len();

    let mut rmap: HashMap<i64, u64> = HashMap::with_capacity(1usize << rn);
    for mask in 0u64..(1u64 << rn) {
        let sum: i64 = (0..rn)
            .filter(|&b| (mask >> b) & 1 != 0)
            .fold(0i64, |s, b| s.saturating_add(right[b]));
        rmap.entry(sum).or_insert(mask);
    }

    for mask in 0u64..(1u64 << ln) {
        let sum: i64 = (0..ln)
            .filter(|&b| (mask >> b) & 1 != 0)
            .fold(0i64, |s, b| s.saturating_add(left[b]));
        if let Some(&rmask) = rmap.get(&(target - sum)) {
            let mut idx: Vec<usize> = (0..ln).filter(|&b| (mask >> b) & 1 != 0).collect();
            idx.extend((0..rn).filter(|&b| (rmask >> b) & 1 != 0).map(|b| half + b));
            return Some(idx);
        }
    }
    None
}

fn random_walk_find(nums: &[i64], target: i64, start: Instant) -> Option<Vec<usize>> {
    let timeout = Duration::from_secs(255);
    let result: Arc<Mutex<Option<Vec<usize>>>> = Arc::new(Mutex::new(None));
    let arc_nums = Arc::new(nums.to_vec());
    let mut handles = Vec::new();

    for t in 0..12usize {
        let nums_c = Arc::clone(&arc_nums);
        let res_c = Arc::clone(&result);
        let seed = (t as u64).wrapping_mul(6364136223846793005u64).wrapping_add(1442695040888963407u64);
        handles.push(thread::spawn(move || {
            rw_thread(&nums_c, target, seed, start, timeout, res_c);
        }));
    }

    for h in handles { let _ = h.join(); }

    let r = result.lock().unwrap();
    r.clone()
}

fn rw_thread(
    nums: &[i64],
    target: i64,
    seed: u64,
    start: Instant,
    timeout: Duration,
    result: Arc<Mutex<Option<Vec<usize>>>>,
) {
    let n = nums.len();
    if n == 0 { return; }
    let mut rng = if seed == 0 { 1 } else { seed };

    let mut incl = vec![false; n];
    let mut cur = 0i64;

    for i in 0..n {
        incl[i] = xs(&mut rng) % 2 == 0;
        if incl[i] { cur = cur.saturating_add(nums[i]); }
    }

    let mut iter = 0u64;
    loop {
        iter += 1;
        if iter % 100_000 == 0 {
            if start.elapsed() >= timeout { return; }
            if result.lock().unwrap().is_some() { return; }
        }

        if cur == target {
            let idx: Vec<usize> = (0..n).filter(|&i| incl[i]).collect();
            let mut r = result.lock().unwrap();
            if r.is_none() { *r = Some(idx); }
            return;
        }

        let idx = xs(&mut rng) as usize % n;
        let new_cur = if incl[idx] {
            cur.saturating_sub(nums[idx])
        } else {
            cur.saturating_add(nums[idx])
        };

        let d_old = (target - cur).abs();
        let d_new = (target - new_cur).abs();

        if d_new <= d_old || xs(&mut rng) % 200 == 0 {
            incl[idx] = !incl[idx];
            cur = new_cur;
        }

        if iter % 2_000_000 == 0 {
            cur = 0;
            for i in 0..n {
                incl[i] = xs(&mut rng) % 2 == 0;
                if incl[i] { cur = cur.saturating_add(nums[i]); }
            }
        }
    }
}

#[inline]
fn xs(s: &mut u64) -> u64 {
    *s ^= *s << 13;
    *s ^= *s >> 7;
    *s ^= *s << 17;
    *s
}
