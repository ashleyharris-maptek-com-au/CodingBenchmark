use std::collections::VecDeque;
use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let first = lines.next().unwrap().unwrap();
    let mut iter = first.split_whitespace();
    let h: usize = iter.next().unwrap().parse().unwrap();
    let w: usize = iter.next().unwrap().parse().unwrap();

    let mut grid: Vec<Vec<u8>> = Vec::with_capacity(h);
    let mut start = (0usize, 0usize);
    let mut end = (0usize, 0usize);

    for row in 0..h {
        let line = lines.next().unwrap().unwrap();
        let bytes = line.into_bytes();
        // Pad or truncate to width w
        let mut row_bytes = bytes;
        row_bytes.resize(w, b'#');
        for col in 0..w {
            match row_bytes[col] {
                b'A' => start = (row, col),
                b'B' => end = (row, col),
                _ => {}
            }
        }
        grid.push(row_bytes);
    }

    // BFS
    let total = h * w;
    let mut prev: Vec<u32> = vec![u32::MAX; total];
    let start_idx = (start.0 * w + start.1) as u32;
    prev[start_idx as usize] = start_idx; // mark start as visited with self-reference

    let mut queue = VecDeque::new();
    queue.push_back(start_idx);

    let end_idx = (end.0 * w + end.1) as u32;
    let mut found = false;

    'bfs: while let Some(idx) = queue.pop_front() {
        if idx == end_idx {
            found = true;
            break 'bfs;
        }
        let row = (idx as usize) / w;
        let col = (idx as usize) % w;

        // up, down, left, right
        let neighbors: [Option<u32>; 4] = [
            if row > 0 { Some(((row - 1) * w + col) as u32) } else { None },
            if row + 1 < h { Some(((row + 1) * w + col) as u32) } else { None },
            if col > 0 { Some((row * w + col - 1) as u32) } else { None },
            if col + 1 < w { Some((row * w + col + 1) as u32) } else { None },
        ];

        for nb in neighbors.into_iter().flatten() {
            if prev[nb as usize] == u32::MAX {
                let r = (nb as usize) / w;
                let c = (nb as usize) % w;
                let cell = grid[r][c];
                if cell != b'#' {
                    prev[nb as usize] = idx;
                    queue.push_back(nb);
                }
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    // Reconstruct path
    let mut path = Vec::new();
    let mut cur = end_idx;
    loop {
        path.push(cur);
        let p = prev[cur as usize];
        if p == cur {
            break;
        }
        cur = p;
    }
    path.reverse();

    writeln!(out, "{}", path.len()).unwrap();
    for idx in path {
        let row = (idx as usize) / w;
        let col = (idx as usize) % w;
        writeln!(out, "{} {}", col, row).unwrap();
    }
}
