import struct

def words(*vals):
    return b''.join(struct.pack('<I', v) for v in vals)

def string_words(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return b

ID_VOID = 1; ID_FUNC_TYPE = 2; ID_FLOAT = 3; ID_VEC4 = 4
ID_PTR_VEC4 = 5; ID_OUT_COLOR = 6; ID_ONE = 7; ID_ZERO = 8
ID_RED = 9; ID_MAIN = 10; ID_LABEL = 11; BOUND = 12

def instr(opcode, *operands):
    wc = 1 + len(operands)
    return words((wc << 16) | opcode, *operands)

buf = bytearray()
buf += words(0x07230203, 0x00010000, 0x00000000, BOUND, 0x00000000)
buf += instr(17, 1)
buf += instr(14, 0, 1)

name = string_words('main')
entry_words = [4, ID_MAIN] + list(struct.unpack('<' + 'I'*(len(name)//4), name)) + [ID_OUT_COLOR]
wc = 1 + len(entry_words)
buf += struct.pack('<I', (wc << 16) | 15)
for w in entry_words:
    buf += struct.pack('<I', w)

buf += instr(16, ID_MAIN, 7)
buf += instr(71, ID_OUT_COLOR, 30, 0)
buf += instr(19, ID_VOID)
buf += instr(33, ID_FUNC_TYPE, ID_VOID)
buf += instr(22, ID_FLOAT, 32)
buf += instr(23, ID_VEC4, ID_FLOAT, 4)
buf += instr(32, ID_PTR_VEC4, 3, ID_VEC4)
buf += instr(59, ID_PTR_VEC4, ID_OUT_COLOR, 3)
buf += instr(43, ID_FLOAT, ID_ONE, 0x3F800000)
buf += instr(43, ID_FLOAT, ID_ZERO, 0x00000000)
buf += instr(54, ID_VOID, ID_MAIN, 0, ID_FUNC_TYPE)
buf += instr(248, ID_LABEL)
buf += instr(80, ID_VEC4, ID_RED, ID_ONE, ID_ZERO, ID_ZERO, ID_ONE)
buf += instr(62, ID_OUT_COLOR, ID_RED)
buf += instr(253)
buf += instr(56)
print(buf.hex())
