#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <numeric>
#include <set>
#include <map>
#include <tuple>

using namespace std;
typedef long long ll;
typedef pair<ll,int> pli;
const ll INF = 1e18;

int N, M, K;
vector<vector<pair<int,ll>>> g;
vector<int> term;
set<int> term_set;
map<pair<int,int>,ll> ew;

pair<vector<ll>,vector<int>> dijkstra(int src) {
    vector<ll> d(N, INF);
    vector<int> p(N, -1);
    priority_queue<pli,vector<pli>,greater<>> pq;
    d[src]=0; pq.push({0,src});
    while (!pq.empty()) {
        auto [dd,u] = pq.top(); pq.pop();
        if (dd>d[u]) continue;
        for (auto [v,w] : g[u])
            if (d[u]+w < d[v]) { d[v]=d[u]+w; p[v]=u; pq.push({d[v],v}); }
    }
    return {d,p};
}

vector<vector<ll>> dt;
vector<vector<int>> pt;

// Exact Dreyfus-Wagner
vector<ll> dpf;
vector<int> prf;
int MASKS;

ll& DP(int m, int v) { return dpf[(ll)m*N+v]; }
int& PR(int m, int v) { return prf[(ll)m*N+v]; }

set<pair<int,int>> ans_edges;

void trace_to_terminal(int ti, int v) {
    while (pt[ti][v] != -1) {
        int u = pt[ti][v];
        ans_edges.insert({min(u,v),max(u,v)});
        v = u;
    }
}

void recon(int mask, int v) {
    if (__builtin_popcount(mask)==1) {
        trace_to_terminal(__builtin_ctz(mask), v);
        return;
    }
    int p = PR(mask,v);
    if (p >= 0) {
        recon(p, v);
        recon(mask^p, v);
    } else {
        int u = ~p;
        ans_edges.insert({min(u,v),max(u,v)});
        recon(mask, u);
    }
}

void solve_dw() {
    MASKS = 1<<K;
    dpf.assign((ll)MASKS*N, INF);
    prf.assign((ll)MASKS*N, -1);

    for (int i=0; i<K; i++) {
        int m = 1<<i;
        for (int v=0; v<N; v++) DP(m,v) = dt[i][v];
    }

    for (int mask=1; mask<MASKS; mask++) {
        if (__builtin_popcount(mask) < 2) continue;
        int low = mask & -mask;
        int rest = mask ^ low;
        for (int s2=rest; ; s2=(s2-1)&rest) {
            int sub = s2|low, comp = mask^sub;
            if (comp > 0) {
                for (int v=0; v<N; v++) {
                    if (DP(sub,v)==INF || DP(comp,v)==INF) continue;
                    ll c = DP(sub,v)+DP(comp,v);
                    if (c < DP(mask,v)) { DP(mask,v)=c; PR(mask,v)=sub; }
                }
            }
            if (s2==0) break;
        }
        // Dijkstra relaxation over edges
        priority_queue<pli,vector<pli>,greater<>> pq;
        for (int v=0; v<N; v++) if (DP(mask,v)<INF) pq.push({DP(mask,v),v});
        while (!pq.empty()) {
            auto [d,u] = pq.top(); pq.pop();
            if (d>DP(mask,u)) continue;
            for (auto [v,w] : g[u]) {
                ll nd = DP(mask,u)+w;
                if (nd < DP(mask,v)) { DP(mask,v)=nd; PR(mask,v)=~u; pq.push({nd,v}); }
            }
        }
    }

    int full = MASKS-1;
    ll best=INF; int bv=0;
    for (int v=0; v<N; v++) if (DP(full,v)<best) { best=DP(full,v); bv=v; }
    recon(full, bv);
}

struct UF {
    vector<int> p, sz;
    UF(int n): p(n), sz(n,1) { iota(p.begin(),p.end(),0); }
    int find(int x) { return p[x]==x?x:p[x]=find(p[x]); }
    bool unite(int a, int b) {
        a=find(a); b=find(b);
        if(a==b) return false;
        if(sz[a]<sz[b]) swap(a,b);
        p[b]=a; sz[a]+=sz[b]; return true;
    }
};

void solve_heur() {
    // Metric closure MST on terminals
    vector<tuple<ll,int,int>> me;
    for (int i=0; i<K; i++)
        for (int j=i+1; j<K; j++)
            me.push_back({dt[i][term[j]],i,j});
    sort(me.begin(), me.end());

    UF uf(K);
    set<pair<int,int>> sg;
    for (auto& [w,i,j] : me) {
        if (uf.unite(i,j)) {
            int v = term[j];
            while (v != term[i]) {
                int u = pt[i][v];
                sg.insert({min(u,v),max(u,v)});
                v = u;
            }
        }
    }

    // Build spanning tree of subgraph (handles any cycles from shared paths)
    vector<int> vlist;
    {
        set<int> vs;
        for (auto [u,v] : sg) { vs.insert(u); vs.insert(v); }
        vlist.assign(vs.begin(), vs.end());
    }
    int sv = (int)vlist.size();
    map<int,int> vidx;
    for (int i=0; i<sv; i++) vidx[vlist[i]]=i;

    vector<tuple<ll,int,int>> sge;
    for (auto [u,v] : sg) sge.push_back({ew[{u,v}],u,v});
    sort(sge.begin(), sge.end());

    UF uf2(sv);
    ans_edges.clear();
    for (auto& [w,u,v] : sge)
        if (uf2.unite(vidx[u],vidx[v]))
            ans_edges.insert({min(u,v),max(u,v)});

    // Prune non-terminal leaves
    bool changed = true;
    while (changed) {
        changed = false;
        map<int,int> deg;
        for (auto [u,v] : ans_edges) { deg[u]++; deg[v]++; }
        vector<pair<int,int>> rm;
        for (auto& e : ans_edges) {
            if (deg[e.first]==1 && !term_set.count(e.first)) rm.push_back(e);
            else if (deg[e.second]==1 && !term_set.count(e.second)) rm.push_back(e);
        }
        for (auto& e : rm) { ans_edges.erase(e); changed = true; }
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> N >> M >> K;
    g.resize(N);
    for (int i=0; i<M; i++) {
        int u,v; ll w; cin>>u>>v>>w;
        g[u].push_back({v,w}); g[v].push_back({u,w});
        auto e = make_pair(min(u,v),max(u,v));
        if (!ew.count(e)) ew[e]=w;
    }
    term.resize(K);
    for (int i=0; i<K; i++) { cin>>term[i]; term_set.insert(term[i]); }

    if (K==1) { cout<<0<<" "<<0<<"\n"; return 0; }

    dt.resize(K); pt.resize(K);
    for (int i=0; i<K; i++) { auto [d,p]=dijkstra(term[i]); dt[i]=d; pt[i]=p; }

    // Use exact DW when memory is manageable
    ll mem = (ll)(1<<min(K,30)) * N;
    if (K<=20 && mem<=15000000LL) {
        solve_dw();
    } else {
        solve_heur();
    }

    ll total=0;
    for (auto& e : ans_edges) total += ew[e];
    cout << total << " " << ans_edges.size() << "\n";
    for (auto [u,v] : ans_edges) cout << u << " " << v << "\n";
    return 0;
}
