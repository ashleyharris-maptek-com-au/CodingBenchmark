using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    const double MU_DEFAULT = 398600.4418;
    static double mu;

    struct State
    {
        public double x, y, z, vx, vy, vz;
    }

    static double Norm(double x, double y, double z) => Math.Sqrt(x*x + y*y + z*z);

    static double TransferCost(State a, State b)
    {
        double r1 = Norm(a.x, a.y, a.z);
        double r2 = Norm(b.x, b.y, b.z);

        double v1sq = a.vx*a.vx + a.vy*a.vy + a.vz*a.vz;
        double v2sq = b.vx*b.vx + b.vy*b.vy + b.vz*b.vz;

        double denom1 = 2*mu - r1*v1sq;
        double denom2 = 2*mu - r2*v2sq;

        double sma1 = (Math.Abs(denom1) > 1e-9) ? mu*r1/denom1 : r1;
        double sma2 = (Math.Abs(denom2) > 1e-9) ? mu*r2/denom2 : r2;

        // Hohmann transfer estimate
        double at = (r1 + r2) / 2.0;
        if (at <= 0) at = Math.Max(r1, r2);

        double vc1 = Math.Sqrt(mu / r1);
        double vc2 = Math.Sqrt(mu / r2);
        double vt1 = Math.Sqrt(Math.Max(0, mu * (2.0/r1 - 1.0/at)));
        double vt2 = Math.Sqrt(Math.Max(0, mu * (2.0/r2 - 1.0/at)));

        double dv1 = Math.Abs(vt1 - vc1);
        double dv2 = Math.Abs(vc2 - vt2);

        // Plane change penalty
        double dx = b.x - a.x, dy = b.y - a.y, dz = b.z - a.z;
        double sep = Norm(dx, dy, dz);
        double planeChangeCost = 0.1 * sep / Math.Max(r1, r2) * Math.Sqrt(mu / Math.Max(r1, r2));

        return dv1 + dv2 + planeChangeCost;
    }

    static double[,] BuildCostMatrix(State start, State[] stations)
    {
        int n = stations.Length;
        double[,] cost = new double[n + 1, n + 1];
        for (int i = 0; i < n; i++)
        {
            cost[0, i + 1] = TransferCost(start, stations[i]);
            cost[i + 1, 0] = TransferCost(stations[i], start);
            for (int j = 0; j < n; j++)
                cost[i + 1, j + 1] = (i == j) ? 0 : TransferCost(stations[i], stations[j]);
        }
        return cost;
    }

    static double TourCostMatrix(double[,] cost, int[] tour, int n)
    {
        double total = cost[0, tour[0] + 1];
        for (int i = 0; i < n - 1; i++)
            total += cost[tour[i] + 1, tour[i + 1] + 1];
        return total;
    }

    static int[] NearestNeighborMatrix(int n, double[,] cost, int startNode = 0)
    {
        bool[] visited = new bool[n];
        int[] tour = new int[n];
        int cur = startNode;

        if (startNode > 0)
        {
            visited[startNode - 1] = true;
            tour[0] = startNode - 1;
            for (int k = 1; k < n; k++)
            {
                double best = double.MaxValue;
                int bestIdx = -1;
                for (int i = 0; i < n; i++)
                    if (!visited[i] && cost[cur, i + 1] < best) { best = cost[cur, i + 1]; bestIdx = i; }
                tour[k] = bestIdx;
                visited[bestIdx] = true;
                cur = bestIdx + 1;
            }
            return tour;
        }

        for (int k = 0; k < n; k++)
        {
            double best = double.MaxValue;
            int bestIdx = -1;
            for (int i = 0; i < n; i++)
                if (!visited[i] && cost[cur, i + 1] < best) { best = cost[cur, i + 1]; bestIdx = i; }
            tour[k] = bestIdx;
            visited[bestIdx] = true;
            cur = bestIdx + 1;
        }
        return tour;
    }

    static int[] TwoOptMatrix(double[,] cost, int[] tour, int n, DateTime deadline)
    {
        bool improved = true;
        while (improved && DateTime.Now < deadline)
        {
            improved = false;
            double curCost = TourCostMatrix(cost, tour, n);
            for (int i = 0; i < n - 1 && DateTime.Now < deadline; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    int[] newTour = (int[])tour.Clone();
                    Array.Reverse(newTour, i, j - i + 1);
                    double nc = TourCostMatrix(cost, newTour, n);
                    if (nc < curCost - 1e-9)
                    {
                        tour = newTour;
                        curCost = nc;
                        improved = true;
                    }
                }
            }
        }
        return tour;
    }

    static int[] OrOptMatrix(double[,] cost, int[] tour, int n, DateTime deadline)
    {
        bool improved = true;
        while (improved && DateTime.Now < deadline)
        {
            improved = false;
            double bestCost = TourCostMatrix(cost, tour, n);
            for (int i = 0; i < n && DateTime.Now < deadline; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if (i == j) continue;
                    var newList = new List<int>(tour);
                    int val = newList[i];
                    newList.RemoveAt(i);
                    int ins = (j < i) ? j : j - 1;
                    if (ins < 0) ins = 0;
                    if (ins > newList.Count) ins = newList.Count;
                    newList.Insert(ins, val);
                    int[] nt = newList.ToArray();
                    double nc = TourCostMatrix(cost, nt, n);
                    if (nc < bestCost - 1e-9)
                    {
                        tour = nt;
                        bestCost = nc;
                        improved = true;
                        i = n; j = n;
                    }
                }
            }
        }
        return tour;
    }

    static int[] BruteForce(double[,] cost, int n)
    {
        int[] perm = Enumerable.Range(0, n).ToArray();
        int[] best = (int[])perm.Clone();
        double bestCost = TourCostMatrix(cost, best, n);
        foreach (var p in Permutations(perm, n))
        {
            double c = TourCostMatrix(cost, p, n);
            if (c < bestCost) { bestCost = c; best = (int[])p.Clone(); }
        }
        return best;
    }

    static IEnumerable<int[]> Permutations(int[] arr, int n)
    {
        if (n == 1) { yield return (int[])arr.Clone(); yield break; }
        for (int i = 0; i < n; i++)
        {
            (arr[i], arr[n-1]) = (arr[n-1], arr[i]);
            foreach (var p in Permutations(arr, n - 1)) yield return p;
            (arr[i], arr[n-1]) = (arr[n-1], arr[i]);
        }
    }

    static void Main()
    {
        var deadline = DateTime.Now.AddSeconds(270);

        string line1 = Console.ReadLine().Trim();
        var parts1 = line1.Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0]);
        mu = parts1.Length > 1 ? double.Parse(parts1[1]) : MU_DEFAULT;

        State start = ParseState(Console.ReadLine());
        State[] stations = new State[N];
        for (int i = 0; i < N; i++)
            stations[i] = ParseState(Console.ReadLine());

        int[] result;

        if (N == 1)
        {
            result = new[] { 0 };
        }
        else if (N <= 8)
        {
            double[,] cost = BuildCostMatrix(start, stations);
            result = BruteForce(cost, N);
        }
        else
        {
            double[,] cost = BuildCostMatrix(start, stations);

            result = NearestNeighborMatrix(N, cost);
            double bestCost = TourCostMatrix(cost, result, N);

            int maxStarts = Math.Min(N, N <= 30 ? N : 20);
            for (int s = 1; s <= maxStarts && DateTime.Now < deadline; s++)
            {
                var t = NearestNeighborMatrix(N, cost, s);
                double c = TourCostMatrix(cost, t, N);
                if (c < bestCost) { bestCost = c; result = t; }
            }

            result = TwoOptMatrix(cost, result, N, deadline);
            result = OrOptMatrix(cost, result, N, deadline);
            if (DateTime.Now < deadline)
                result = TwoOptMatrix(cost, result, N, deadline);
        }

        Console.WriteLine(string.Join(" ", result));
    }

    static State ParseState(string line)
    {
        var p = line.Trim().Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        return new State
        {
            x = double.Parse(p[0]), y = double.Parse(p[1]), z = double.Parse(p[2]),
            vx = double.Parse(p[3]), vy = double.Parse(p[4]), vz = double.Parse(p[5])
        };
    }
}
