```rust
use std::io::{self, BufRead, Write};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();

    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);
    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut p = line.split_whitespace();
        let u: usize = p.next().unwrap().parse().unwrap();
        let v: usize = p.next().unwrap().parse().unwrap();
        if u != v {
            edges.push((u, v));
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];

    if n == 0 {
        let stdout = io::stdout();
        let mut out = io::BufWriter::new(stdout.lock());
        writeln!(out, "{}", n).unwrap();
        return;
    }

    // BFS to find components and assign initial positions
    let mut visited = vec![false; n];
    let mut components: Vec<Vec<usize>> = Vec::new();

    for start in 0..n {
        if !visited[start] {
            let mut comp = Vec::new();
            let mut queue = VecDeque::new();
            queue.push_back(start);
            visited[start] = true;
            while let Some(node) = queue.pop_front() {
                comp.push(node);
                for &nb in &adj[node] {
                    if !visited[nb] {
                        visited[nb] = true;
                        queue.push_back(nb);
                    }
                }
            }
            components.push(comp);
        }
    }

    // Assign initial positions using BFS layers per component
    let mut comp_offset_x = 0.0f64;

    for comp in &components {
        let comp_size = comp.len();

        if comp_size == 1 {
            x[comp[0]] = comp_offset_x;
            y[comp[0]] = 0.0;
            comp_offset_x += 5.0;
            continue;
        }

        // BFS layering for initial layout
        let root = comp[0];
        let mut layer = vec![usize::MAX; n];
        let mut bfs_queue = VecDeque::new();
        bfs_queue.push_back(root);
        layer[root] = 0;
        let mut max_layer = 0;

        while let Some(node) = bfs_queue.pop_front() {
            for &nb in &adj[node] {
                if layer[nb] == usize::MAX {
                    layer[nb] = layer[node] + 1;
                    if layer[nb] > max_layer { max_layer = layer[nb]; }
                    bfs_queue.push_back(nb);
                }
            }
        }

        // Group nodes by layer
        let mut layers: Vec<Vec<usize>> = vec![vec![]; max_layer + 1];
        for &node in comp {
            let l = if layer[node] == usize::MAX { max_layer } else { layer[node] };
            layers[l].push(node);
        }

        // Assign initial positions
        for (ly, layer_nodes) in layers.iter().enumerate() {
            let count = layer_nodes.len();
            for (i, &node) in layer_nodes.iter().enumerate() {
                x[node] = comp_offset_x + i as f64 * 2.0 - (count as f64 - 1.0);
                y[node] = ly as f64 * 2.0;
            }
        }

        let max_width = layers.iter().map(|l| l.len()).max().unwrap_or(1);
        comp_offset_x += max_width as f64 * 2.0 + 5.0;
    }

    // Force-directed layout
    let iterations = if n <= 50 {
        500
    } else if n <= 200 {
        300
    } else if n <= 1000 {
        150
    } else if n <= 5000 {
        50
    } else if n <= 10000 {
        20
    } else {
        0
    };

    if iterations == 0 {
        let stdout = io::stdout();
        let mut out = io::BufWriter::new(stdout.lock());
        writeln!(out, "{}", n).unwrap();
        for i in 0..n {
            writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
        }
        return;
    }

    let k = (n as f64).sqrt() * 2.0;
    let mut temp = k * 5.0;
    let cooling = 0.95f64;

    let mut dx = vec![0.0f64; n];
    let mut dy = vec![0.0f64; n];

    // Simple LCG for deterministic pseudo-random sampling
    let mut rng_state: u64 = 12345;
    let lcg_next = |state: &mut u64| -> u64 {
        *state = state.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        *state
    };

    for _iter in 0..iterations {
        for i in 0..n {
            dx[i] = 0.0;
            dy[i] = 0.0;
        }

        // Repulsive forces
        if n <= 500 {
            for i in 0..n {
                for j in (i+1)..n {
                    let ddx = x[i] - x[j];
                    let ddy = y[i] - y[j];
                    let dist2 = ddx * ddx + ddy * ddy + 0.01;
                    let dist = dist2.sqrt();
                    let force = k * k / dist;
                    let fx = force * ddx / dist;
                    let fy = force * ddy / dist;
                    dx[i] += fx;
                    dy[i] += fy;
                    dx[j] -= fx;
                    dy[j] -= fy;
                }
            }
        } else {
            let sample = (n / 5).max(100).min(500);
            for i in 0..n {
                for _ in 0..sample {
                    let j = (lcg_next(&mut rng_state) as usize) % n;
                    if i == j { continue; }
                    let ddx = x[i] - x[j];
                    let ddy = y[i] - y[j];
                    let dist2 = ddx * ddx + ddy * ddy + 0.01;
                    let dist = dist2.sqrt();
                    let scale = n as f64 / sample as f64;
                    let force = k * k / dist * scale;
                    let fx = force * ddx / dist;
                    let fy = force * ddy / dist;
                    dx[i] += fx * 0.5;
                    dy[i] += fy * 0.5;
                }
            }
        }

        // Attractive forces along edges
        for &(u, v) in &edges {
            let ddx = x[u] - x[v];
            let ddy = y[u] - y[v];
            let dist = (ddx * ddx + ddy * ddy + 0.01).sqrt();
            let force = dist * dist / k;
            let fx = force * ddx / dist;
            let fy = force * ddy / dist;
            dx[u] -= fx;
            dy[u] -= fy;
            dx[v] += fx;
            dy[v] += fy;
        }

        // Apply displacements with temperature limiting
        for i in 0..n {
            let disp = (dx[i] * dx[i] + dy[i] * dy[i] + 0.01).sqrt();
            let scale = temp.min(disp) / disp;
            x[i] += dx[i] * scale;
            y[i] += dy[i] * scale;
        }

        temp *= cooling;
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}
```
