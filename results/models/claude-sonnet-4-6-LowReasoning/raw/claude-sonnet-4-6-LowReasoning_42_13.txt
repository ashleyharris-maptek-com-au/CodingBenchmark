; Wireframe geometry shader
; InputTrianglesAdjacency -> OutputTriangleStrip, 12 vertices
; Renders each triangle edge as a thin quad (thickness 0.008 in NDC)

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 12
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

; Basic types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
    %fn_type = OpTypeFunction %void

; Constants
     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4
    %float_0 = OpConstant %float 0.0
  %float_0_5 = OpConstant %float 0.5
   %float_th = OpConstant %float 0.008

; gl_PerVertex input block (array of 6 for InputTrianglesAdjacency)
%gl_PerVertex_in  = OpTypeStruct %v4float
      %arr_pv_in  = OpTypeArray %gl_PerVertex_in %uint_6
  %ptr_arr_pv_in  = OpTypePointer Input %arr_pv_in
; gl_PerVertex output block (single, per emitted vertex)
%gl_PerVertex_out = OpTypeStruct %v4float
     %ptr_pv_out  = OpTypePointer Output %gl_PerVertex_out

; Pointer types
    %ptr_v4_in = OpTypePointer Input %v4float
   %ptr_v4_out = OpTypePointer Output %v4float
    %arr_v3_6  = OpTypeArray %v3float %uint_6
 %ptr_arr_v3_in = OpTypePointer Input %arr_v3_6
    %ptr_v3_in = OpTypePointer Input %v3float
   %ptr_v3_out = OpTypePointer Output %v3float

; Interface variables
      %gl_in = OpVariable %ptr_arr_pv_in Input
     %gl_out = OpVariable %ptr_pv_out Output
 %in_worldPos = OpVariable %ptr_arr_v3_in Input
  %in_normal  = OpVariable %ptr_arr_v3_in Input
   %in_color  = OpVariable %ptr_arr_v3_in Input
  %out_color  = OpVariable %ptr_v3_out Output

; ============================================================
      %main = OpFunction %void None %fn_type
     %entry = OpLabel

; Load gl_Position for main triangle vertices (indices 0, 2, 4)
  %ac_pos0 = OpAccessChain %ptr_v4_in %gl_in %int_0 %int_0
     %pos0 = OpLoad %v4float %ac_pos0
  %ac_pos2 = OpAccessChain %ptr_v4_in %gl_in %int_2 %int_0
     %pos2 = OpLoad %v4float %ac_pos2
  %ac_pos4 = OpAccessChain %ptr_v4_in %gl_in %int_4 %int_0
     %pos4 = OpLoad %v4float %ac_pos4

; Load colors for main triangle vertices
  %ac_col0 = OpAccessChain %ptr_v3_in %in_color %int_0
     %col0 = OpLoad %v3float %ac_col0
  %ac_col2 = OpAccessChain %ptr_v3_in %in_color %int_2
     %col2 = OpLoad %v3float %ac_col2
  %ac_col4 = OpAccessChain %ptr_v3_in %in_color %int_4
     %col4 = OpLoad %v3float %ac_col4

; Average edge colors
   %sum_02 = OpFAdd %v3float %col0 %col2
   %avg_02 = OpVectorTimesScalar %v3float %sum_02 %float_0_5
   %sum_24 = OpFAdd %v3float %col2 %col4
   %avg_24 = OpVectorTimesScalar %v3float %sum_24 %float_0_5
   %sum_40 = OpFAdd %v3float %col4 %col0
   %avg_40 = OpVectorTimesScalar %v3float %sum_40 %float_0_5

; Output position pointer (reused for all emitted vertices)
%ac_out_pos = OpAccessChain %ptr_v4_out %gl_out %int_0

; ==================== EDGE 0 -> 2 ====================
; Extract clip-space components
      %p0x = OpCompositeExtract %float %pos0 0
      %p0y = OpCompositeExtract %float %pos0 1
      %p0z = OpCompositeExtract %float %pos0 2
      %p0w = OpCompositeExtract %float %pos0 3
      %p2x = OpCompositeExtract %float %pos2 0
      %p2y = OpCompositeExtract %float %pos2 1
      %p2z = OpCompositeExtract %float %pos2 2
      %p2w = OpCompositeExtract %float %pos2 3

; NDC xy: divide by w
     %n0x = OpFDiv %float %p0x %p0w
     %n0y = OpFDiv %float %p0y %p0w
     %n2x = OpFDiv %float %p2x %p2w
     %n2y = OpFDiv %float %p2y %p2w

; Edge direction in NDC, normalized
    %d02x = OpFSub %float %n2x %n0x
    %d02y = OpFSub %float %n2y %n0y
    %dir02 = OpCompositeConstruct %v2float %d02x %d02y
   %ndir02 = OpExtInst %v2float %1 Normalize %dir02
   %nd02x = OpCompositeExtract %float %ndir02 0
   %nd02y = OpCompositeExtract %float %ndir02 1

; Perpendicular: (-ny, nx) * thickness
  %neg02y = OpFNegate %float %nd02y
  %perp02 = OpCompositeConstruct %v2float %neg02y %nd02x
 %perp02t = OpVectorTimesScalar %v2float %perp02 %float_th
  %pt02x  = OpCompositeExtract %float %perp02t 0
  %pt02y  = OpCompositeExtract %float %perp02t 1

; Clip-space offset = perp_ndc * w (to achieve ndc offset = perp_ndc)
%of02_0x = OpFMul %float %pt02x %p0w
%of02_0y = OpFMul %float %pt02y %p0w
%of02_2x = OpFMul %float %pt02x %p2w
%of02_2y = OpFMul %float %pt02y %p2w

; 4 quad vertices: A-, A+, B-, B+
%e02v0x = OpFSub %float %p0x %of02_0x
%e02v0y = OpFSub %float %p0y %of02_0y
  %e02v0 = OpCompositeConstruct %v4float %e02v0x %e02v0y %p0z %p0w
%e02v1x = OpFAdd %float %p0x %of02_0x
%e02v1y = OpFAdd %float %p0y %of02_0y
  %e02v1 = OpCompositeConstruct %v4float %e02v1x %e02v1y %p0z %p0w
%e02v2x = OpFSub %float %p2x %of02_2x
%e02v2y = OpFSub %float %p2y %of02_2y
  %e02v2 = OpCompositeConstruct %v4float %e02v2x %e02v2y %p2z %p2w
%e02v3x = OpFAdd %float %p2x %of02_2x
%e02v3y = OpFAdd %float %p2y %of02_2y
  %e02v3 = OpCompositeConstruct %v4float %e02v3x %e02v3y %p2z %p2w

; Emit edge 0->2 quad
             OpStore %ac_out_pos %e02v0
             OpStore %out_color %avg_02
             OpEmitVertex
             OpStore %ac_out_pos %e02v1
             OpStore %out_color %avg_02
             OpEmitVertex
             OpStore %ac_out_pos %e02v2
             OpStore %out_color %avg_02
             OpEmitVertex
             OpStore %ac_out_pos %e02v3
             OpStore %out_color %avg_02
             OpEmitVertex
             OpEndPrimitive

; ==================== EDGE 2 -> 4 ====================
      %p4x = OpCompositeExtract %float %pos4 0
      %p4y = OpCompositeExtract %float %pos4 1
      %p4z = OpCompositeExtract %float %pos4 2
      %p4w = OpCompositeExtract %float %pos4 3

; NDC xy for pos4
     %n4x = OpFDiv %float %p4x %p4w
     %n4y = OpFDiv %float %p4y %p4w

; Edge direction 2->4 in NDC, normalized
    %d24x = OpFSub %float %n4x %n2x
    %d24y = OpFSub %float %n4y %n2y
    %dir24 = OpCompositeConstruct %v2float %d24x %d24y
   %ndir24 = OpExtInst %v2float %1 Normalize %dir24
   %nd24x = OpCompositeExtract %float %ndir24 0
   %nd24y = OpCompositeExtract %float %ndir24 1

; Perpendicular
  %neg24y = OpFNegate %float %nd24y
  %perp24 = OpCompositeConstruct %v2float %neg24y %nd24x
 %perp24t = OpVectorTimesScalar %v2float %perp24 %float_th
  %pt24x  = OpCompositeExtract %float %perp24t 0
  %pt24y  = OpCompositeExtract %float %perp24t 1

; Clip-space offsets
%of24_2x = OpFMul %float %pt24x %p2w
%of24_2y = OpFMul %float %pt24y %p2w
%of24_4x = OpFMul %float %pt24x %p4w
%of24_4y = OpFMul %float %pt24y %p4w

; 4 quad vertices
%e24v0x = OpFSub %float %p2x %of24_2x
%e24v0y = OpFSub %float %p2y %of24_2y
  %e24v0 = OpCompositeConstruct %v4float %e24v0x %e24v0y %p2z %p2w
%e24v1x = OpFAdd %float %p2x %of24_2x
%e24v1y = OpFAdd %float %p2y %of24_2y
  %e24v1 = OpCompositeConstruct %v4float %e24v1x %e24v1y %p2z %p2w
%e24v2x = OpFSub %float %p4x %of24_4x
%e24v2y = OpFSub %float %p4y %of24_4y
  %e24v2 = OpCompositeConstruct %v4float %e24v2x %e24v2y %p4z %p4w
%e24v3x = OpFAdd %float %p4x %of24_4x
%e24v3y = OpFAdd %float %p4y %of24_4y
  %e24v3 = OpCompositeConstruct %v4float %e24v3x %e24v3y %p4z %p4w

; Emit edge 2->4 quad
             OpStore %ac_out_pos %e24v0
             OpStore %out_color %avg_24
             OpEmitVertex
             OpStore %ac_out_pos %e24v1
             OpStore %out_color %avg_24
             OpEmitVertex
             OpStore %ac_out_pos %e24v2
             OpStore %out_color %avg_24
             OpEmitVertex
             OpStore %ac_out_pos %e24v3
             OpStore %out_color %avg_24
             OpEmitVertex
             OpEndPrimitive

; ==================== EDGE 4 -> 0 ====================
; Edge direction 4->0 in NDC, normalized
    %d40x = OpFSub %float %n0x %n4x
    %d40y = OpFSub %float %n0y %n4y
    %dir40 = OpCompositeConstruct %v2float %d40x %d40y
   %ndir40 = OpExtInst %v2float %1 Normalize %dir40
   %nd40x = OpCompositeExtract %float %ndir40 0
   %nd40y = OpCompositeExtract %float %ndir40 1

; Perpendicular
  %neg40y = OpFNegate %float %nd40y
  %perp40 = OpCompositeConstruct %v2float %neg40y %nd40x
 %perp40t = OpVectorTimesScalar %v2float %perp40 %float_th
  %pt40x  = OpCompositeExtract %float %perp40t 0
  %pt40y  = OpCompositeExtract %float %perp40t 1

; Clip-space offsets
%of40_4x = OpFMul %float %pt40x %p4w
%of40_4y = OpFMul %float %pt40y %p4w
%of40_0x = OpFMul %float %pt40x %p0w
%of40_0y = OpFMul %float %pt40y %p0w

; 4 quad vertices
%e40v0x = OpFSub %float %p4x %of40_4x
%e40v0y = OpFSub %float %p4y %of40_4y
  %e40v0 = OpCompositeConstruct %v4float %e40v0x %e40v0y %p4z %p4w
%e40v1x = OpFAdd %float %p4x %of40_4x
%e40v1y = OpFAdd %float %p4y %of40_4y
  %e40v1 = OpCompositeConstruct %v4float %e40v1x %e40v1y %p4z %p4w
%e40v2x = OpFSub %float %p0x %of40_0x
%e40v2y = OpFSub %float %p0y %of40_0y
  %e40v2 = OpCompositeConstruct %v4float %e40v2x %e40v2y %p0z %p0w
%e40v3x = OpFAdd %float %p0x %of40_0x
%e40v3y = OpFAdd %float %p0y %of40_0y
  %e40v3 = OpCompositeConstruct %v4float %e40v3x %e40v3y %p0z %p0w

; Emit edge 4->0 quad
             OpStore %ac_out_pos %e40v0
             OpStore %out_color %avg_40
             OpEmitVertex
             OpStore %ac_out_pos %e40v1
             OpStore %out_color %avg_40
             OpEmitVertex
             OpStore %ac_out_pos %e40v2
             OpStore %out_color %avg_40
             OpEmitVertex
             OpStore %ac_out_pos %e40v3
             OpStore %out_color %avg_40
             OpEmitVertex
             OpEndPrimitive

             OpReturn
             OpFunctionEnd
