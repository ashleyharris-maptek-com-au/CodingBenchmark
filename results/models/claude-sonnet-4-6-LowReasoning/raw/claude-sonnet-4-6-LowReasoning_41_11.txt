; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 100
; Schema: 0
OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Input/output decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 ColMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 ColMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 ColMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

; Types
%void       = OpTypeVoid
%bool       = OpTypeBool
%int        = OpTypeInt 32 1
%float      = OpTypeFloat 32
%vec2       = OpTypeVector %float 2
%vec3       = OpTypeVector %float 3
%vec4       = OpTypeVector %float 4
%mat4       = OpTypeMatrix %vec4 4

%funcType   = OpTypeFunction %void

%ptrInVec3      = OpTypePointer Input %vec3
%ptrInVec2      = OpTypePointer Input %vec2
%ptrOutVec4     = OpTypePointer Output %vec4
%UBOType        = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptrUniformUBO  = OpTypePointer Uniform %UBOType
%ptrUniformVec4 = OpTypePointer Uniform %vec4

; Variables
%inWorldPos = OpVariable %ptrInVec3  Input
%inNormal   = OpVariable %ptrInVec3  Input
%inUV       = OpVariable %ptrInVec2  Input
%inTangent  = OpVariable %ptrInVec3  Input
%inColor    = OpVariable %ptrInVec3  Input
%outColor   = OpVariable %ptrOutVec4 Output
%uboVar     = OpVariable %ptrUniformUBO Uniform

; Integer constants (for UBO member access)
%int_3 = OpConstant %int 3

; Float constants
%float_0   = OpConstant %float 0.0
%float_1   = OpConstant %float 1.0
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_6   = OpConstant %float 6.0

; Composite constants
%vec2_half  = OpConstantComposite %vec2 %float_0_5 %float_0_5
%vec2_6     = OpConstantComposite %vec2 %float_6 %float_6
%yellow     = OpConstantComposite %vec3 %float_1 %float_0_8 %float_0
%darkBlue   = OpConstantComposite %vec3 %float_0_1 %float_0_1 %float_0_3

; Entry point
%main = OpFunction %void None %funcType
%mainEntry = OpLabel

; Load UV, scale to 6x6 grid
%uvLoad    = OpLoad %vec2 %inUV
%cellCoord = OpFMul %vec2 %uvLoad %vec2_6

; fract(cellCoord) = cellCoord - floor(cellCoord)
%floorCoord = OpExtInst %vec2 %glsl450 Floor %cellCoord
%fracCoord  = OpFSub %vec2 %cellCoord %floorCoord

; Offset from cell center
%centered = OpFSub %vec2 %fracCoord %vec2_half

; Distance from center
%cx   = OpCompositeExtract %float %centered 0
%cy   = OpCompositeExtract %float %centered 1
%cx2  = OpFMul %float %cx %cx
%cy2  = OpFMul %float %cy %cy
%d2   = OpFAdd %float %cx2 %cy2
%dist = OpExtInst %float %glsl450 Sqrt %d2

; Inside dot if dist < 0.3
%inDot    = OpFOrdLessThan %bool %dist %float_0_3
%baseColor = OpSelect %vec3 %inDot %yellow %darkBlue

; Load geometry inputs
%worldPosLoad = OpLoad %vec3 %inWorldPos
%normalLoad   = OpLoad %vec3 %inNormal

; Normalize normal
%normN = OpExtInst %vec3 %glsl450 Normalize %normalLoad

; Load lightPos from UBO (member 3)
%lightPosPtr  = OpAccessChain %ptrUniformVec4 %uboVar %int_3
%lightPosVec4 = OpLoad %vec4 %lightPosPtr
%lpX          = OpCompositeExtract %float %lightPosVec4 0
%lpY          = OpCompositeExtract %float %lightPosVec4 1
%lpZ          = OpCompositeExtract %float %lightPosVec4 2
%lightPos3    = OpCompositeConstruct %vec3 %lpX %lpY %lpZ

; Light direction: normalize(lightPos - worldPos)
%lightDirRaw = OpFSub %vec3 %lightPos3 %worldPosLoad
%lightDir    = OpExtInst %vec3 %glsl450 Normalize %lightDirRaw

; Lambertian diffuse = max(0, dot(normal, lightDir))
%dotRaw  = OpDot %float %normN %lightDir
%diffuse = OpExtInst %float %glsl450 FMax %dotRaw %float_0

; Shading factor = 0.3 + 0.7 * diffuse
%diff07  = OpFMul %float %float_0_7 %diffuse
%shading = OpFAdd %float %float_0_3 %diff07

; Apply shading to base color
%r       = OpCompositeExtract %float %baseColor 0
%g       = OpCompositeExtract %float %baseColor 1
%b       = OpCompositeExtract %float %baseColor 2
%shadedR = OpFMul %float %r %shading
%shadedG = OpFMul %float %g %shading
%shadedB = OpFMul %float %b %shading

; Construct and store output color (alpha = 1.0)
%finalColor = OpCompositeConstruct %vec4 %shadedR %shadedG %shadedB %float_1
OpStore %outColor %finalColor

OpReturn
OpFunctionEnd
