; SPIR-V geometry shader: flat-shading with InputTrianglesAdjacency
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor %inWorldPos %inNormal
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_inv3 = OpConstant %float 0.33333334

%float_inv_sqrt3 = OpConstant %float 0.57735027
%L = OpConstantComposite %v3float %float_inv_sqrt3 %float_inv_sqrt3 %float_inv_sqrt3

; Struct types
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

%arr_v3float_6 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_6

; Variables
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3float Output

; Function type
%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load gl_Position for vertices 0, 2, 4
%ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%ptr_p2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
%ptr_p4 = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
%p0v4 = OpLoad %v4float %ptr_p0
%p2v4 = OpLoad %v4float %ptr_p2
%p4v4 = OpLoad %v4float %ptr_p4

; Extract xyz for cross product
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2
%p4 = OpVectorShuffle %v3float %p4v4 %p4v4 0 1 2

; a = p2 - p0, b = p4 - p0
%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

; Cross product manually: a x b
%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2
%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2

%cross_x_0 = OpFMul %float %a_y %b_z
%cross_x_1 = OpFMul %float %a_z %b_y
%cross_x = OpFSub %float %cross_x_0 %cross_x_1

%cross_y_0 = OpFMul %float %a_z %b_x
%cross_y_1 = OpFMul %float %a_x %b_z
%cross_y = OpFSub %float %cross_y_0 %cross_y_1

%cross_z_0 = OpFMul %float %a_x %b_y
%cross_z_1 = OpFMul %float %a_y %b_x
%cross_z = OpFSub %float %cross_z_0 %cross_z_1

%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z

; N = normalize(cross)
%N = OpExtInst %v3float %1 Normalize %cross

; dot(N, L)
%dot_NL = OpDot %float %N %L

; diffuse = max(dot(N,L), 0.0)
%diffuse = OpExtInst %float %1 FMax %dot_NL %float_0

; Load colors at indices 0, 2, 4
%ptr_c0 = OpAccessChain %ptr_in_v3float %inColor %int_0
%ptr_c2 = OpAccessChain %ptr_in_v3float %inColor %int_2
%ptr_c4 = OpAccessChain %ptr_in_v3float %inColor %int_4
%c0 = OpLoad %v3float %ptr_c0
%c2 = OpLoad %v3float %ptr_c2
%c4 = OpLoad %v3float %ptr_c4

; base = (c0 + c2 + c4) / 3
%c_sum0 = OpFAdd %v3float %c0 %c2
%c_sum1 = OpFAdd %v3float %c_sum0 %c4
%base = OpVectorTimesScalar %v3float %c_sum1 %float_inv3

; final = base * (0.2 + 0.8 * diffuse)
%d_scaled = OpFMul %float %float_0_8 %diffuse
%light_factor = OpFAdd %float %float_0_2 %d_scaled
%final_color = OpVectorTimesScalar %v3float %base %light_factor

; Emit vertex 0
%out_pos0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos0 %p0v4
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 2
%out_pos2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos2 %p2v4
OpStore %outColor %final_color
OpEmitVertex

; Emit vertex 4
%out_pos4 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos4 %p4v4
OpStore %outColor %final_color
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
