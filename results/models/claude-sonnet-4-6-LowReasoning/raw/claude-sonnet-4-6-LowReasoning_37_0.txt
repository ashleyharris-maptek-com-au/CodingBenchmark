using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

class Program
{
    static void Main()
    {
        var reader = new System.IO.StreamReader(Console.OpenStandardInput());
        var parts = reader.ReadLine().Trim().Split();
        int n = int.Parse(parts[0]), m = int.Parse(parts[1]);

        var adj = new HashSet<int>[n];
        var radj = new HashSet<int>[n];
        for (int i = 0; i < n; i++) { adj[i] = new HashSet<int>(); radj[i] = new HashSet<int>(); }

        var resultSet = new HashSet<int>();

        for (int i = 0; i < m; i++)
        {
            var p = reader.ReadLine().Trim().Split();
            int u = int.Parse(p[0]), v = int.Parse(p[1]);
            if (u == v) resultSet.Add(u);
            else
            {
                adj[u].Add(v);
                radj[v].Add(u);
            }
        }

        // Remove self-loop vertices
        foreach (var v in resultSet.ToList())
            RemoveV(v, adj, radj);

        // Reduce: remove vertices with in=0 or out=0
        Reduce(adj, radj, n);

        // Find FVS on remaining graph
        var fvs = FindMinFVS(adj, radj, n);
        foreach (var v in fvs) resultSet.Add(v);

        var sb = new StringBuilder();
        sb.AppendLine(resultSet.Count.ToString());
        sb.AppendLine(string.Join(" ", resultSet));
        Console.Write(sb);
    }

    static void RemoveV(int v, HashSet<int>[] adj, HashSet<int>[] radj)
    {
        foreach (var u in radj[v]) adj[u].Remove(v);
        foreach (var w in adj[v]) radj[w].Remove(v);
        adj[v].Clear(); radj[v].Clear();
    }

    static void Reduce(HashSet<int>[] adj, HashSet<int>[] radj, int n)
    {
        var q = new Queue<int>();
        var inQ = new bool[n];
        for (int i = 0; i < n; i++)
            if (adj[i].Count == 0 || radj[i].Count == 0) { q.Enqueue(i); inQ[i] = true; }

        while (q.Count > 0)
        {
            int v = q.Dequeue(); inQ[v] = false;
            if (adj[v].Count > 0 && radj[v].Count > 0) continue;
            var nb = adj[v].Union(radj[v]).ToList();
            foreach (var u in radj[v].ToList()) adj[u].Remove(v);
            foreach (var w in adj[v].ToList()) radj[w].Remove(v);
            adj[v].Clear(); radj[v].Clear();
            foreach (var u in nb)
                if (!inQ[u] && (adj[u].Count == 0 || radj[u].Count == 0)) { q.Enqueue(u); inQ[u] = true; }
        }
    }

    static List<int> FindMinFVS(HashSet<int>[] adj, HashSet<int>[] radj, int n)
    {
        var sccs = GetSCCs(adj, n);

        var sccGroups = new Dictionary<int, List<int>>();
        for (int v = 0; v < n; v++)
        {
            if (adj[v].Count == 0 && radj[v].Count == 0) continue;
            int c = sccs[v];
            if (!sccGroups.TryGetValue(c, out var g)) sccGroups[c] = g = new List<int>();
            g.Add(v);
        }

        var result = new List<int>();
        foreach (var (_, verts) in sccGroups)
        {
            if (verts.Count <= 1) continue;

            var vertSet = new HashSet<int>(verts);
            var sa = new HashSet<int>[n];
            var sra = new HashSet<int>[n];
            for (int i = 0; i < n; i++) { sa[i] = new HashSet<int>(); sra[i] = new HashSet<int>(); }
            foreach (var v in verts)
                foreach (var w in adj[v])
                    if (vertSet.Contains(w)) { sa[v].Add(w); sra[w].Add(v); }

            for (int k = 0; ; k++)
            {
                var sol = new List<int>();
                var aCopy = sa.Select(s => new HashSet<int>(s)).ToArray();
                var raCopy = sra.Select(s => new HashSet<int>(s)).ToArray();
                if (BranchFVS(aCopy, raCopy, n, k, sol))
                {
                    result.AddRange(sol);
                    break;
                }
            }
        }

        return result;
    }

    static bool BranchFVS(HashSet<int>[] adj, HashSet<int>[] radj, int n, int budget, List<int> sol)
    {
        // Kernelization: handle self-loops and degree-0 vertices
        var forced = new List<int>();
        bool changed = true;
        while (changed)
        {
            changed = false;
            for (int v = 0; v < n; v++)
            {
                if (adj[v].Count == 0 && radj[v].Count == 0) continue;
                if (adj[v].Contains(v))
                {
                    forced.Add(v);
                    RemoveV(v, adj, radj);
                    changed = true;
                    break;
                }
                if (adj[v].Count == 0 || radj[v].Count == 0)
                {
                    RemoveV(v, adj, radj);
                    changed = true;
                    break;
                }
            }
        }

        sol.AddRange(forced);
        budget -= forced.Count;

        if (budget < 0)
        {
            sol.RemoveRange(sol.Count - forced.Count, forced.Count);
            return false;
        }

        var cycle = FindShortCycle(adj, n);
        if (cycle == null) return true; // DAG

        if (budget == 0)
        {
            sol.RemoveRange(sol.Count - forced.Count, forced.Count);
            return false;
        }

        foreach (var v in cycle)
        {
            var aCopy = adj.Select(s => new HashSet<int>(s)).ToArray();
            var raCopy = radj.Select(s => new HashSet<int>(s)).ToArray();

            RemoveV(v, adj, radj);
            sol.Add(v);

            if (BranchFVS(adj, radj, n, budget - 1, sol))
                return true;

            sol.RemoveAt(sol.Count - 1);
            for (int i = 0; i < n; i++) { adj[i] = aCopy[i]; radj[i] = raCopy[i]; }
        }

        sol.RemoveRange(sol.Count - forced.Count, forced.Count);
        return false;
    }

    static List<int> FindShortCycle(HashSet<int>[] adj, int n)
    {
        // Check for length-2 cycles first (mutual edges)
        for (int v = 0; v < n; v++)
            foreach (var w in adj[v])
                if (w != v && adj[w].Contains(v))
                    return new List<int> { v, w };

        // DFS to find any cycle
        var color = new int[n]; // 0=white, 1=gray, 2=black
        var pathPos = new int[n];
        var pathStack = new List<int>();

        for (int start = 0; start < n; start++)
        {
            if (color[start] != 0 || (adj[start].Count == 0 && true)) { }
            if (color[start] != 0) continue;

            var dfsStack = new Stack<(int v, IEnumerator<int> it)>();
            color[start] = 1;
            pathPos[start] = pathStack.Count;
            pathStack.Add(start);
            dfsStack.Push((start, adj[start].GetEnumerator()));

            while (dfsStack.Count > 0)
            {
                var (v, it) = dfsStack.Peek();
                if (it.MoveNext())
                {
                    int w = it.Current;
                    if (color[w] == 1)
                    {
                        return pathStack.Skip(pathPos[w]).ToList();
                    }
                    if (color[w] == 0)
                    {
                        color[w] = 1;
                        pathPos[w] = pathStack.Count;
                        pathStack.Add(w);
                        dfsStack.Push((w, adj[w].GetEnumerator()));
                    }
                }
                else
                {
                    color[v] = 2;
                    pathStack.RemoveAt(pathStack.Count - 1);
                    dfsStack.Pop();
                }
            }
        }

        return null;
    }

    static int[] GetSCCs(HashSet<int>[] adj, int n)
    {
        var order = new List<int>();
        var visited = new bool[n];
        for (int i = 0; i < n; i++)
            if (!visited[i]) DFS1(i, adj, visited, order);

        var radjList = new List<int>[n];
        for (int i = 0; i < n; i++) radjList[i] = new List<int>();
        for (int u = 0; u < n; u++)
            foreach (var v in adj[u]) radjList[v].Add(u);

        var comp = new int[n];
        Array.Fill(comp, -1);
        int c = 0;
        visited = new bool[n];

        for (int i = order.Count - 1; i >= 0; i--)
        {
            int v = order[i];
            if (!visited[v]) DFS2(v, radjList, visited, comp, c++);
        }

        return comp;
    }

    static void DFS1(int start, HashSet<int>[] adj, bool[] visited, List<int> order)
    {
        var stack = new Stack<(int v, IEnumerator<int> it)>();
        visited[start] = true;
        stack.Push((start, adj[start].GetEnumerator()));
        while (stack.Count > 0)
        {
            var (v, it) = stack.Peek();
            if (it.MoveNext())
            {
                int w = it.Current;
                if (!visited[w]) { visited[w] = true; stack.Push((w, adj[w].GetEnumerator())); }
            }
            else { order.Add(v); stack.Pop(); }
        }
    }

    static void DFS2(int start, List<int>[] radj, bool[] visited, int[] comp, int c)
    {
        var stack = new Stack<int>();
        stack.Push(start); visited[start] = true; comp[start] = c;
        while (stack.Count > 0)
        {
            int v = stack.Pop();
            foreach (var w in radj[v])
                if (!visited[w]) { visited[w] = true; comp[w] = c; stack.Push(w); }
        }
    }
}
