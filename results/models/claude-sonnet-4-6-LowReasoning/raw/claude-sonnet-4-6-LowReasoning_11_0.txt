use std::io::{self, BufRead, Write, BufWriter};
use std::f64::consts::PI;
use std::collections::{HashMap, HashSet};

type Point = (f64, f64);
type Poly = Vec<Point>;

#[inline]
fn rotate_pt(p: Point, cos_a: f64, sin_a: f64) -> Point {
    (p.0 * cos_a - p.1 * sin_a, p.0 * sin_a + p.1 * cos_a)
}

fn bbox(poly: &[Point]) -> (f64, f64, f64, f64) {
    let mut mnx = f64::INFINITY;
    let mut mny = f64::INFINITY;
    let mut mxx = f64::NEG_INFINITY;
    let mut mxy = f64::NEG_INFINITY;
    for &(x, y) in poly {
        if x < mnx { mnx = x; }
        if y < mny { mny = y; }
        if x > mxx { mxx = x; }
        if y > mxy { mxy = y; }
    }
    (mnx, mny, mxx, mxy)
}

fn poly_area(poly: &[Point]) -> f64 {
    let n = poly.len();
    let mut a = 0.0f64;
    for i in 0..n {
        let j = (i + 1) % n;
        a += poly[i].0 * poly[j].1 - poly[j].0 * poly[i].1;
    }
    a.abs() * 0.5
}

fn point_in_poly(px: f64, py: f64, poly: &[Point]) -> bool {
    let n = poly.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        if (yi > py) != (yj > py) && px < (xj - xi) * (py - yi) / (yj - yi) + xi {
            inside = !inside;
        }
        j = i;
    }
    inside
}

// Returns true if segment p1-p2 and p3-p4 properly intersect (not at endpoints)
fn seg_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool {
    let d1x = p2.0 - p1.0;
    let d1y = p2.1 - p1.1;
    let d2x = p4.0 - p3.0;
    let d2y = p4.1 - p3.1;
    let denom = d1x * d2y - d1y * d2x;
    if denom.abs() < 1e-12 {
        return false;
    }
    let dx = p3.0 - p1.0;
    let dy = p3.1 - p1.1;
    let t = (dx * d2y - dy * d2x) / denom;
    let u = (dx * d1y - dy * d1x) / denom;
    t > 1e-9 && t < 1.0 - 1e-9 && u > 1e-9 && u < 1.0 - 1e-9
}

// Check if piece fits entirely within stock (all vertices inside + no edge crossings)
fn fits_in_stock(piece: &[Point], stock: &[Point], stock_is_convex: bool) -> bool {
    for &(px, py) in piece {
        if !point_in_poly(px, py, stock) {
            return false;
        }
    }
    if stock_is_convex {
        // For convex stock, all vertices inside => piece is inside
        return true;
    }
    // For non-convex stock, also check no piece edge crosses stock boundary
    let np = piece.len();
    let ns = stock.len();
    for i in 0..np {
        let j = (i + 1) % np;
        for k in 0..ns {
            let l = (k + 1) % ns;
            if seg_intersect(piece[i], piece[j], stock[k], stock[l]) {
                return false;
            }
        }
    }
    true
}

fn polys_overlap(a: &[Point], a_bb: (f64, f64, f64, f64), b: &[Point], b_bb: (f64, f64, f64, f64)) -> bool {
    // Fast bounding box rejection
    if a_bb.2 <= b_bb.0 + 1e-9
        || b_bb.2 <= a_bb.0 + 1e-9
        || a_bb.3 <= b_bb.1 + 1e-9
        || b_bb.3 <= a_bb.1 + 1e-9
    {
        return false;
    }
    // Edge intersection check
    let na = a.len();
    let nb = b.len();
    for i in 0..na {
        let j = (i + 1) % na;
        for k in 0..nb {
            let l = (k + 1) % nb;
            if seg_intersect(a[i], a[j], b[k], b[l]) {
                return true;
            }
        }
    }
    // One inside the other check
    if point_in_poly(a[0].0, a[0].1, b) || point_in_poly(b[0].0, b[0].1, a) {
        return true;
    }
    false
}

fn is_convex(poly: &[Point]) -> bool {
    let n = poly.len();
    if n < 3 {
        return true;
    }
    let mut sign = 0i32;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let c = poly[(i + 2) % n];
        let cross = (b.0 - a.0) * (c.1 - b.1) - (b.1 - a.1) * (c.0 - b.0);
        if cross.abs() > 1e-12 {
            let s = if cross > 0.0 { 1 } else { -1 };
            if sign == 0 {
                sign = s;
            } else if sign != s {
                return false;
            }
        }
    }
    true
}

struct Sheet {
    placed: Vec<(Poly, (f64, f64, f64, f64))>,
    grid: HashMap<(i32, i32), Vec<usize>>,
    cell_size: f64,
}

impl Sheet {
    fn new(cell_size: f64) -> Self {
        Sheet {
            placed: Vec::new(),
            grid: HashMap::new(),
            cell_size,
        }
    }

    fn cells_for_bb(&self, bb: (f64, f64, f64, f64)) -> impl Iterator<Item = (i32, i32)> {
        let cs = self.cell_size;
        let x0 = (bb.0 / cs).floor() as i32;
        let y0 = (bb.1 / cs).floor() as i32;
        let x1 = (bb.2 / cs).ceil() as i32;
        let y1 = (bb.3 / cs).ceil() as i32;
        let mut v = Vec::new();
        for x in x0..=x1 {
            for y in y0..=y1 {
                v.push((x, y));
            }
        }
        v.into_iter()
    }

    fn add(&mut self, poly: Poly, bb: (f64, f64, f64, f64)) {
        let idx = self.placed.len();
        let cells: Vec<_> = self.cells_for_bb(bb).collect();
        for cell in cells {
            self.grid.entry(cell).or_default().push(idx);
        }
        self.placed.push((poly, bb));
    }

    fn overlaps_any(&self, poly: &[Point], bb: (f64, f64, f64, f64)) -> bool {
        let cells: Vec<_> = self.cells_for_bb(bb).collect();
        let mut checked = HashSet::new();
        for cell in cells {
            if let Some(indices) = self.grid.get(&cell) {
                for &idx in indices {
                    if checked.insert(idx) {
                        let (ref p, p_bb) = self.placed[idx];
                        if polys_overlap(poly, bb, p, p_bb) {
                            return true;
                        }
                    }
                }
            }
        }
        false
    }
}

fn try_place_piece(
    norm: &[Point],
    pw: f64,
    ph: f64,
    stock: &[Point],
    stock_is_convex: bool,
    stock_w: f64,
    stock_h: f64,
    sheet: &Sheet,
    grid_step: f64,
) -> Option<(f64, f64)> {
    let max_tx = stock_w - pw;
    let max_ty = stock_h - ph;
    if max_tx < -1e-9 || max_ty < -1e-9 {
        return None;
    }

    let mut ty = 0.0f64;
    while ty <= max_ty + 1e-9 {
        let mut tx = 0.0f64;
        while tx <= max_tx + 1e-9 {
            let cand: Poly = norm.iter().map(|&(x, y)| (x + tx, y + ty)).collect();
            let cand_bb = (tx, ty, tx + pw, ty + ph);

            if fits_in_stock(&cand, stock, stock_is_convex) && !sheet.overlaps_any(&cand, cand_bb) {
                return Some((tx, ty));
            }

            tx += grid_step;
        }
        ty += grid_step;
    }
    None
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines().map(|l| l.unwrap());

    let first = lines.next().unwrap();
    let mut it = first.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_raw: Poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let mut p = line.split_whitespace();
        let x: f64 = p.next().unwrap().parse().unwrap();
        let y: f64 = p.next().unwrap().parse().unwrap();
        stock_raw.push((x, y));
    }

    let mut pieces: Vec<Poly> = Vec::with_capacity(np);
    for _ in 0..np {
        let pv: usize = lines.next().unwrap().trim().parse().unwrap();
        let mut poly: Poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let mut p = line.split_whitespace();
            let x: f64 = p.next().unwrap().parse().unwrap();
            let y: f64 = p.next().unwrap().parse().unwrap();
            poly.push((x, y));
        }
        pieces.push(poly);
    }

    // Normalize stock to start at origin
    let (sx0, sy0, sx1, sy1) = bbox(&stock_raw);
    let stock: Poly = stock_raw.iter().map(|&(x, y)| (x - sx0, y - sy0)).collect();
    let stock_w = sx1 - sx0;
    let stock_h = sy1 - sy0;
    let stock_is_convex = is_convex(&stock);

    // Determine rotations based on problem size
    let rotations: Vec<f64> = if np <= 20 {
        (0..16).map(|i| i as f64 * PI / 8.0).collect()
    } else if np <= 200 {
        (0..8).map(|i| i as f64 * PI / 4.0).collect()
    } else {
        vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0]
    };

    // Adaptive grid step based on scale
    let grid_step = if np <= 50 {
        0.5f64
    } else if np <= 500 {
        2.0
    } else {
        5.0
    };

    // Cell size for spatial grid
    let cell_size = (stock_w * stock_h / np.max(1) as f64).sqrt().max(grid_step * 4.0);

    // Sort pieces by area descending (largest first = better packing)
    let areas: Vec<f64> = pieces.iter().map(|p| poly_area(p)).collect();
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&a, &b| {
        areas[b]
            .partial_cmp(&areas[a])
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let mut results: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];
    let mut sheets: Vec<Sheet> = Vec::new();

    for &pi in &order {
        let piece = &pieces[pi];
        let mut placed = false;

        'outer: for si in 0..sheets.len() {
            for &rot in &rotations {
                let (cos_a, sin_a) = (rot.cos(), rot.sin());
                let rotated: Poly = piece.iter().map(|&p| rotate_pt(p, cos_a, sin_a)).collect();
                let (rx0, ry0, rx1, ry1) = bbox(&rotated);
                let pw = rx1 - rx0;
                let ph = ry1 - ry0;

                if pw > stock_w + 1e-9 || ph > stock_h + 1e-9 {
                    continue;
                }

                // Normalized: bounding box starts at (0,0)
                let norm: Poly = rotated.iter().map(|&(x, y)| (x - rx0, y - ry0)).collect();

                if let Some((tx, ty)) = try_place_piece(
                    &norm,
                    pw,
                    ph,
                    &stock,
                    stock_is_convex,
                    stock_w,
                    stock_h,
                    &sheets[si],
                    grid_step,
                ) {
                    let cand: Poly = norm.iter().map(|&(x, y)| (x + tx, y + ty)).collect();
                    let cand_bb = (tx, ty, tx + pw, ty + ph);
                    results[pi] = (si, tx - rx0, ty - ry0, rot);
                    sheets[si].add(cand, cand_bb);
                    placed = true;
                    break 'outer;
                }
            }
        }

        if !placed {
            let si = sheets.len();
            sheets.push(Sheet::new(cell_size));

            let mut placed_in_new = false;
            for &rot in &rotations {
                let (cos_a, sin_a) = (rot.cos(), rot.sin());
                let rotated: Poly = piece.iter().map(|&p| rotate_pt(p, cos_a, sin_a)).collect();
                let (rx0, ry0, rx1, ry1) = bbox(&rotated);
                let pw = rx1 - rx0;
                let ph = ry1 - ry0;

                if pw > stock_w + 1e-9 || ph > stock_h + 1e-9 {
                    continue;
                }

                let norm: Poly = rotated.iter().map(|&(x, y)| (x - rx0, y - ry0)).collect();

                // For new empty sheet, just check if it fits anywhere
                if let Some((tx, ty)) = try_place_piece(
                    &norm,
                    pw,
                    ph,
                    &stock,
                    stock_is_convex,
                    stock_w,
                    stock_h,
                    &sheets[si],
                    grid_step,
                ) {
                    let cand: Poly = norm.iter().map(|&(x, y)| (x + tx, y + ty)).collect();
                    let cand_bb = (tx, ty, tx + pw, ty + ph);
                    results[pi] = (si, tx - rx0, ty - ry0, rot);
                    sheets[si].add(cand, cand_bb);
                    placed_in_new = true;
                    break;
                }
            }

            if !placed_in_new {
                // Piece doesn't fit in stock at all; place anyway (constraint violation)
                results[pi] = (si, 0.0, 0.0, 0.0);
                let cand = piece.clone();
                let bb = bbox(&cand);
                sheets[si].add(cand, bb);
            }
        }
    }

    writeln!(out, "{}", sheets.len()).unwrap();
    for i in 0..np {
        let (si, tx, ty, rot) = results[i];
        writeln!(out, "{} {} {} {}", si, tx, ty, rot).unwrap();
    }
}
