; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 120
; Schema: 0
OpCapability Shader
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs/outputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; UBO block decoration
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 0 ColMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 1 ColMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 2 ColMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224

; Types
%void       = OpTypeVoid
%voidFn     = OpTypeFunction %void
%float      = OpTypeFloat 32
%v2float    = OpTypeVector %float 2
%v3float    = OpTypeVector %float 3
%v4float    = OpTypeVector %float 4
%mat4v4     = OpTypeMatrix %v4float 4
%int        = OpTypeInt 32 1

%ptrInV3    = OpTypePointer Input %v3float
%ptrInV2    = OpTypePointer Input %v2float
%ptrOutV4   = OpTypePointer Output %v4float

%UBOType    = OpTypeStruct %mat4v4 %mat4v4 %mat4v4 %v4float %v4float %v4float
%ptrUniUBO  = OpTypePointer Uniform %UBOType
%ptrUniV4   = OpTypePointer Uniform %v4float

; Variables
%inWorldPos = OpVariable %ptrInV3 Input
%inNormal   = OpVariable %ptrInV3 Input
%inUV       = OpVariable %ptrInV2 Input
%inTangent  = OpVariable %ptrInV3 Input
%inColor    = OpVariable %ptrInV3 Input
%outColor   = OpVariable %ptrOutV4 Output
%uboVar     = OpVariable %ptrUniUBO Uniform

; Constants - int indices
%idx0       = OpConstant %int 0
%idx1       = OpConstant %int 1
%idx2       = OpConstant %int 2
%idx3       = OpConstant %int 3
%idx4       = OpConstant %int 4
%idx5       = OpConstant %int 5

; Constants - float
%f0         = OpConstant %float 0.0
%f1         = OpConstant %float 1.0
%f0_2       = OpConstant %float 0.2
%f0_3       = OpConstant %float 0.3
%f0_5       = OpConstant %float 0.5
%f0_8       = OpConstant %float 0.8
%f3         = OpConstant %float 3.0

; Composite constants
%baseColor  = OpConstantComposite %v3float %f0_2 %f0_3 %f0_8
%rimTint    = OpConstantComposite %v3float %f1 %f0_8 %f0_5

; Main function
%main = OpFunction %void None %voidFn
%entryLabel = OpLabel

; Load inputs
%wp         = OpLoad %v3float %inWorldPos
%nrm        = OpLoad %v3float %inNormal

; Load lightPos (member 3) and cameraPos (member 4) from UBO
%lpPtr      = OpAccessChain %ptrUniV4 %uboVar %idx3
%lp4        = OpLoad %v4float %lpPtr
%lp3        = OpVectorShuffle %v3float %lp4 %lp4 0 1 2

%cpPtr      = OpAccessChain %ptrUniV4 %uboVar %idx4
%cp4        = OpLoad %v4float %cpPtr
%cp3        = OpVectorShuffle %v3float %cp4 %cp4 0 1 2

; N = normalize(normal)
%N          = OpExtInst %v3float %glsl450 Normalize %nrm

; V = normalize(cameraPos - worldPos)
%camDiff    = OpFSub %v3float %cp3 %wp
%V          = OpExtInst %v3float %glsl450 Normalize %camDiff

; L = normalize(lightPos - worldPos)
%lightDiff  = OpFSub %v3float %lp3 %wp
%L          = OpExtInst %v3float %glsl450 Normalize %lightDiff

; diffuse = max(dot(N, L), 0)
%NdotL      = OpDot %float %N %L
%diffuse    = OpExtInst %float %glsl450 FMax %NdotL %f0

; NdotV = max(dot(N, V), 0)
%NdotV      = OpDot %float %N %V
%NdotVc     = OpExtInst %float %glsl450 FMax %NdotV %f0

; rim = pow(1 - NdotVc, 3)
%oneMinusNV = OpFSub %float %f1 %NdotVc
%rim        = OpExtInst %float %glsl450 Pow %oneMinusNV %f3

; modulation = 0.3 + 0.5 * diffuse
%halfDiff   = OpFMul %float %f0_5 %diffuse
%mod        = OpFAdd %float %f0_3 %halfDiff

; modulatedBase = baseColor * modulation
%bc0        = OpCompositeExtract %float %baseColor 0
%bc1        = OpCompositeExtract %float %baseColor 1
%bc2        = OpCompositeExtract %float %baseColor 2
%mb0        = OpFMul %float %bc0 %mod
%mb1        = OpFMul %float %bc1 %mod
%mb2        = OpFMul %float %bc2 %mod
%modBase    = OpCompositeConstruct %v3float %mb0 %mb1 %mb2

; rimContrib = rimTint * rim
%rt0        = OpCompositeExtract %float %rimTint 0
%rt1        = OpCompositeExtract %float %rimTint 1
%rt2        = OpCompositeExtract %float %rimTint 2
%rc0        = OpFMul %float %rt0 %rim
%rc1        = OpFMul %float %rt1 %rim
%rc2        = OpFMul %float %rt2 %rim
%rimContrib = OpCompositeConstruct %v3float %rc0 %rc1 %rc2

; finalColor = modBase + rimContrib
%finalRGB   = OpFAdd %v3float %modBase %rimContrib

%fr0        = OpCompositeExtract %float %finalRGB 0
%fr1        = OpCompositeExtract %float %finalRGB 1
%fr2        = OpCompositeExtract %float %finalRGB 2

%outVec     = OpCompositeConstruct %v4float %fr0 %fr1 %fr2 %f1
OpStore %outColor %outVec

OpReturn
OpFunctionEnd
