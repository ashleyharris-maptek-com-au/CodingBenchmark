using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int N;
    static List<(int to, long w, int idx)>[] adj;

    static void Main()
    {
        var reader = new StreamReader(new BufferedStream(Console.OpenStandardInput(), 1 << 20));
        var inp = reader.ReadLine().Trim().Split();
        N = int.Parse(inp[0]);
        int M = int.Parse(inp[1]);

        adj = new List<(int, long, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, long, int)>();

        int[] eu = new int[M], ev = new int[M];
        long[] ew = new long[M];
        int[] deg = new int[N];

        for (int i = 0; i < M; i++)
        {
            var p = reader.ReadLine().Trim().Split();
            eu[i] = int.Parse(p[0]);
            ev[i] = int.Parse(p[1]);
            ew[i] = long.Parse(p[2]);
            adj[eu[i]].Add((ev[i], ew[i], i));
            adj[ev[i]].Add((eu[i], ew[i], i));
            deg[eu[i]]++;
            deg[ev[i]]++;
        }

        var oddV = new List<int>();
        for (int i = 0; i < N; i++)
            if ((deg[i] & 1) == 1) oddV.Add(i);

        int[] extra = new int[M];

        if (oddV.Count > 0)
            SolveMatching(oddV, eu, ev, extra);

        var route = FindEulerianCircuit(M, eu, ev, extra);

        var sb = new StringBuilder(route.Count * 5);
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb);
    }

    static void SolveMatching(List<int> oddV, int[] eu, int[] ev, int[] extra)
    {
        int k = oddV.Count;
        bool[] matched = new bool[k];

        if (k <= 20)
        {
            // Exact bitmask DP minimum weight matching
            var dists = new long[k][];
            var prevEdges = new int[k][];
            for (int i = 0; i < k; i++)
                (dists[i], prevEdges[i]) = Dijkstra(oddV[i]);

            long[,] d = new long[k, k];
            for (int i = 0; i < k; i++)
                for (int j = 0; j < k; j++)
                    d[i, j] = dists[i][oddV[j]];

            long[] dp = new long[1 << k];
            int[] dpFrom = new int[1 << k];
            Array.Fill(dp, long.MaxValue / 2);
            dp[0] = 0;

            for (int mask = 0; mask < (1 << k); mask++)
            {
                if (dp[mask] >= long.MaxValue / 2) continue;
                int first = -1;
                for (int i = 0; i < k; i++)
                    if ((mask & (1 << i)) == 0) { first = i; break; }
                if (first == -1) continue;
                for (int j = first + 1; j < k; j++)
                {
                    if ((mask & (1 << j)) != 0) continue;
                    int nm = mask | (1 << first) | (1 << j);
                    long nc = dp[mask] + d[first, j];
                    if (nc < dp[nm])
                    {
                        dp[nm] = nc;
                        dpFrom[nm] = first * k + j;
                    }
                }
            }

            int fullMask = (1 << k) - 1;
            while (fullMask != 0)
            {
                int pair = dpFrom[fullMask];
                int a = pair / k, b = pair % k;
                AddPath(oddV[a], oddV[b], prevEdges[a], eu, ev, extra);
                fullMask ^= (1 << a) | (1 << b);
            }
        }
        else if (k <= 600)
        {
            // All-pairs Dijkstra + greedy matching by ascending distance
            var dists = new long[k][];
            var prevEdges = new int[k][];
            for (int i = 0; i < k; i++)
                (dists[i], prevEdges[i]) = Dijkstra(oddV[i]);

            var pairs = new List<(long d, int a, int b)>(k * (k - 1) / 2);
            for (int a = 0; a < k; a++)
                for (int b = a + 1; b < k; b++)
                    pairs.Add((dists[a][oddV[b]], a, b));
            pairs.Sort((x, y) => x.d.CompareTo(y.d));

            int mCount = 0;
            foreach (var (_, a, b) in pairs)
            {
                if (matched[a] || matched[b]) continue;
                matched[a] = matched[b] = true;
                AddPath(oddV[a], oddV[b], prevEdges[a], eu, ev, extra);
                if (++mCount == k / 2) break;
            }
        }
        else
        {
            // Large k: greedy with local search window, Dijkstra per pair
            int budget = 1500;
            int done = 0;

            for (int i = 0; i < k && done < budget; i++)
            {
                if (matched[i]) continue;
                (var dist, var pe) = Dijkstra(oddV[i]);
                int best = -1;
                long bestD = long.MaxValue;
                int window = Math.Min(k, i + 200);
                for (int j = i + 1; j < window; j++)
                {
                    if (!matched[j] && dist[oddV[j]] < bestD)
                    {
                        bestD = dist[oddV[j]];
                        best = j;
                    }
                }
                if (best == -1)
                {
                    // search wider
                    for (int j = i + 1; j < k; j++)
                    {
                        if (!matched[j] && dist[oddV[j]] < bestD)
                        {
                            bestD = dist[oddV[j]];
                            best = j;
                        }
                    }
                }
                if (best == -1) continue;
                matched[i] = matched[best] = true;
                AddPath(oddV[i], oddV[best], pe, eu, ev, extra);
                done++;
            }

            // Handle any remaining unmatched with sequential pairing
            List<int> remaining = new List<int>();
            for (int i = 0; i < k; i++)
                if (!matched[i]) remaining.Add(i);
            for (int i = 0; i + 1 < remaining.Count; i += 2)
            {
                (var dist, var pe) = Dijkstra(oddV[remaining[i]]);
                AddPath(oddV[remaining[i]], oddV[remaining[i + 1]], pe, eu, ev, extra);
            }
        }
    }

    static void AddPath(int src, int dst, int[] prevEdge, int[] eu, int[] ev, int[] extra)
    {
        int cur = dst;
        int safety = N + 1;
        while (cur != src && safety-- > 0)
        {
            int eidx = prevEdge[cur];
            if (eidx < 0) break;
            extra[eidx]++;
            cur = (eu[eidx] == cur) ? ev[eidx] : eu[eidx];
        }
    }

    static (long[] dist, int[] prevEdge) Dijkstra(int src)
    {
        var dist = new long[N];
        var pe = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(pe, -1);
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            foreach (var (v, w, idx) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    pe[v] = idx;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, pe);
    }

    static List<int> FindEulerianCircuit(int M, int[] eu, int[] ev, int[] extra)
    {
        int[] rem = new int[M];
        for (int i = 0; i < M; i++) rem[i] = 1 + extra[i];

        var edgeList = new List<(int v, int idx)>[N];
        for (int i = 0; i < N; i++) edgeList[i] = new List<(int, int)>();
        for (int i = 0; i < M; i++)
        {
            edgeList[eu[i]].Add((ev[i], i));
            edgeList[ev[i]].Add((eu[i], i));
        }

        int[] ptr = new int[N];
        var stack = new Stack<int>(M + 1);
        var result = new List<int>(M + 1);
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;
            while (ptr[u] < edgeList[u].Count)
            {
                var (v, idx) = edgeList[u][ptr[u]];
                ptr[u]++;
                if (rem[idx] > 0)
                {
                    rem[idx]--;
                    stack.Push(v);
                    found = true;
                    break;
                }
            }
            if (!found)
                result.Add(stack.Pop());
        }

        result.Reverse();
        return result;
    }
}
