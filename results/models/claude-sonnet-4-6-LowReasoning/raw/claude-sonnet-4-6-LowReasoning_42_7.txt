; SPIR-V geometry shader: explode triangles along face normal
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
       %bool = OpTypeBool
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

; Constants
     %uint_3 = OpConstant %uint 3
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
    %float_0 = OpConstant %float 0
  %float_inv3 = OpConstant %float 0.33333334
  %float_008 = OpConstant %float 0.08

; Struct types
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array of 3 input gl_PerVertex
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3

; Pointer types
%ptr_in_arr_pv  = OpTypePointer Input  %arr_pv_in
%ptr_in_v4float = OpTypePointer Input  %v4float
%ptr_out_pv     = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4     = OpTypePointer Output %v4float

; Array for user color input
   %arr_v3_3 = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_3
%ptr_in_v3  = OpTypePointer Input  %v3float
%ptr_out_v3 = OpTypePointer Output %v3float

; Variables
      %gl_in = OpVariable %ptr_in_arr_pv  Input
     %gl_out = OpVariable %ptr_out_pv     Output
    %inColor = OpVariable %ptr_in_arr_v3  Input
   %outColor = OpVariable %ptr_out_v3     Output

; Function type
   %fn_void = OpTypeFunction %void

; Main function
       %main = OpFunction %void None %fn_void
      %entry = OpLabel

; Load p0 = gl_in[0].gl_Position
  %ptr_p0v4 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
       %p0v4 = OpLoad %v4float %ptr_p0v4

; Load p1 = gl_in[1].gl_Position
  %ptr_p1v4 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
       %p1v4 = OpLoad %v4float %ptr_p1v4

; Load p2 = gl_in[2].gl_Position
  %ptr_p2v4 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
       %p2v4 = OpLoad %v4float %ptr_p2v4

; Extract xyz from positions
        %p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2
        %p1 = OpVectorShuffle %v3float %p1v4 %p1v4 0 1 2
        %p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2

; e1 = p1 - p0
        %e1 = OpFSub %v3float %p1 %p0

; e2 = p2 - p0
        %e2 = OpFSub %v3float %p2 %p0

; cross(e1, e2) via GLSL.std.450
  %cross_raw = OpExtInst %v3float %1 Cross %e1 %e2

; normalize
      %nrm3 = OpExtInst %v3float %1 Normalize %cross_raw

; offset = nrm3 * 0.08
    %offset3 = OpVectorTimesScalar %v3float %nrm3 %float_008

; offset as vec4 (w=0)
  %off_x = OpCompositeExtract %float %offset3 0
  %off_y = OpCompositeExtract %float %offset3 1
  %off_z = OpCompositeExtract %float %offset3 2
  %offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

; Load colors
%ptr_c0 = OpAccessChain %ptr_in_v3 %inColor %int_0
    %c0 = OpLoad %v3float %ptr_c0
%ptr_c1 = OpAccessChain %ptr_in_v3 %inColor %int_1
    %c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3 %inColor %int_2
    %c2 = OpLoad %v3float %ptr_c2

; Emit vertex 0: gl_Position = p0v4 + offset4, color = c0
  %pos0out = OpFAdd %v4float %p0v4 %offset4
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ptr_out_pos %pos0out
               OpStore %outColor %c0
               OpEmitVertex

; Emit vertex 1
  %pos1out = OpFAdd %v4float %p1v4 %offset4
               OpStore %ptr_out_pos %pos1out
               OpStore %outColor %c1
               OpEmitVertex

; Emit vertex 2
  %pos2out = OpFAdd %v4float %p2v4 %offset4
               OpStore %ptr_out_pos %pos2out
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
