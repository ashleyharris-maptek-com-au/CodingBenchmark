%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft

; Decorations for inputs
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0

; UBO decorations
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0
OpDecorate %UBOType Block
OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 0 ColMajor
OpMemberDecorate %UBOType 0 MatrixStride 16
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 1 ColMajor
OpMemberDecorate %UBOType 1 MatrixStride 16
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 2 ColMajor
OpMemberDecorate %UBOType 2 MatrixStride 16
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float

%UBOType = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBOType = OpTypePointer Uniform %UBOType
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%functype = OpTypeFunction %void

; Variables
%inWorldPos = OpVariable %ptr_Input_v3float Input
%inNormal   = OpVariable %ptr_Input_v3float Input
%inUV       = OpVariable %ptr_Input_v2float Input
%inTangent  = OpVariable %ptr_Input_v3float Input
%inColor    = OpVariable %ptr_Input_v3float Input
%outColor   = OpVariable %ptr_Output_v4float Output
%uboVar     = OpVariable %ptr_Uniform_UBOType Uniform

; Constants
%c_int3 = OpConstant %int 3
%c_int4 = OpConstant %int 4
%c_float_0   = OpConstant %float 0.0
%c_float_1   = OpConstant %float 1.0
%c_float_0p2 = OpConstant %float 0.2
%c_float_0p5 = OpConstant %float 0.5
%c_float_0p8 = OpConstant %float 0.8
%c_float_32  = OpConstant %float 32.0

; Main function
%main = OpFunction %void None %functype
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %inWorldPos
%normal   = OpLoad %v3float %inNormal

; Load lightPos and cameraPos from UBO
%ptrLightPos  = OpAccessChain %ptr_Uniform_v4float %uboVar %c_int3
%ptrCamPos    = OpAccessChain %ptr_Uniform_v4float %uboVar %c_int4
%lightPos4    = OpLoad %v4float %ptrLightPos
%camPos4      = OpLoad %v4float %ptrCamPos

; Extract xyz from vec4
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%camPos   = OpVectorShuffle %v3float %camPos4   %camPos4   0 1 2

; N = normalize(normal)
%N = OpExtInst %v3float %glsl450 Normalize %normal

; L = normalize(lightPos - worldPos)
%Ldiff = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %glsl450 Normalize %Ldiff

; V = normalize(camPos - worldPos)
%Vdiff = OpFSub %v3float %camPos %worldPos
%V = OpExtInst %v3float %glsl450 Normalize %Vdiff

; negL = -L
%negL = OpFNegate %v3float %L

; R = reflect(-L, N)
%R = OpExtInst %v3float %glsl450 Reflect %negL %N

; diffuse = max(dot(N, L), 0)
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl450 FMax %dotNL %c_float_0

; specular = pow(max(dot(R, V), 0), 32)
%dotRV = OpDot %float %R %V
%dotRV_clamped = OpExtInst %float %glsl450 FMax %dotRV %c_float_0
%specular = OpExtInst %float %glsl450 Pow %dotRV_clamped %c_float_32

; intensity = 0.2 + 0.5*diffuse + 0.8*specular
%term1 = OpFMul %float %c_float_0p5 %diffuse
%term2 = OpFMul %float %c_float_0p8 %specular
%sum1  = OpFAdd %float %c_float_0p2 %term1
%intensity = OpFAdd %float %sum1 %term2

; outColor = vec4(intensity, intensity, intensity, 1.0)
%color = OpCompositeConstruct %v4float %intensity %intensity %intensity %c_float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
