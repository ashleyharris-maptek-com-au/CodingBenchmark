use std::io::{self, Read};

struct Solver {
    adj: Vec<Vec<i32>>,
    n: usize,
    half: usize,
    best_cut: usize,
    best_part: Vec<bool>,
}

impl Solver {
    fn compute_cut(&self, part: &[bool]) -> usize {
        let mut cut = 0usize;
        for i in 0..self.n {
            if part[i] {
                for j in 0..self.n {
                    if !part[j] {
                        cut += self.adj[i][j] as usize;
                    }
                }
            }
        }
        cut
    }

    fn kl_pass(&self, partition: &mut Vec<bool>) -> bool {
        let n = self.n;
        let half = self.half;

        let mut d = vec![0i32; n];
        for v in 0..n {
            for w in 0..n {
                if v != w {
                    if partition[v] != partition[w] {
                        d[v] += self.adj[v][w];
                    } else {
                        d[v] -= self.adj[v][w];
                    }
                }
            }
        }

        let mut locked = vec![false; n];
        let mut best_cum = 0i32;
        let mut best_k = 0usize;
        let mut cum_gain = 0i32;
        let mut swap_seq: Vec<(usize, usize)> = Vec::new();

        for _ in 0..half {
            let mut best_g = i32::MIN;
            let mut best_a = usize::MAX;
            let mut best_b = usize::MAX;

            for a in 0..n {
                if partition[a] && !locked[a] {
                    for b in 0..n {
                        if !partition[b] && !locked[b] {
                            let g = d[a] + d[b] - 2 * self.adj[a][b];
                            if g > best_g {
                                best_g = g;
                                best_a = a;
                                best_b = b;
                            }
                        }
                    }
                }
            }

            if best_a == usize::MAX {
                break;
            }

            cum_gain += best_g;
            swap_seq.push((best_a, best_b));

            if cum_gain > best_cum {
                best_cum = cum_gain;
                best_k = swap_seq.len();
            }

            locked[best_a] = true;
            locked[best_b] = true;

            for v in 0..n {
                if locked[v] {
                    continue;
                }
                if partition[v] {
                    d[v] += 2 * self.adj[v][best_a] - 2 * self.adj[v][best_b];
                } else {
                    d[v] += 2 * self.adj[v][best_b] - 2 * self.adj[v][best_a];
                }
            }

            partition[best_a] = false;
            partition[best_b] = true;
        }

        for k in (best_k..swap_seq.len()).rev() {
            let (a, b) = swap_seq[k];
            partition[a] = true;
            partition[b] = false;
        }

        best_cum > 0
    }

    fn run_kl(&mut self, mut partition: Vec<bool>) {
        loop {
            if !self.kl_pass(&mut partition) {
                break;
            }
        }
        let cut = self.compute_cut(&partition);
        if cut < self.best_cut {
            self.best_cut = cut;
            self.best_part = partition;
        }
    }

    fn branch_and_bound(
        &mut self,
        idx: usize,
        count_a: usize,
        current_cut: usize,
        part: &mut Vec<bool>,
    ) {
        let n = self.n;
        let half = self.half;

        if current_cut >= self.best_cut {
            return;
        }

        if idx == n {
            if current_cut < self.best_cut {
                self.best_cut = current_cut;
                self.best_part.copy_from_slice(part);
            }
            return;
        }

        let count_b = idx - count_a;
        let need_a = half - count_a;
        let need_b = half - count_b;

        if need_b == 0 {
            // All remaining go to A
            let mut cut = current_cut;
            for i in idx..n {
                for j in 0..idx {
                    if !part[j] {
                        cut += self.adj[i][j] as usize;
                    }
                }
                if cut >= self.best_cut {
                    return;
                }
            }
            if cut < self.best_cut {
                self.best_cut = cut;
                self.best_part[..idx].copy_from_slice(&part[..idx]);
                for i in idx..n {
                    self.best_part[i] = true;
                }
            }
            return;
        }

        if need_a == 0 {
            // All remaining go to B
            let mut cut = current_cut;
            for i in idx..n {
                for j in 0..idx {
                    if part[j] {
                        cut += self.adj[i][j] as usize;
                    }
                }
                if cut >= self.best_cut {
                    return;
                }
            }
            if cut < self.best_cut {
                self.best_cut = cut;
                self.best_part[..idx].copy_from_slice(&part[..idx]);
                for i in idx..n {
                    self.best_part[i] = false;
                }
            }
            return;
        }

        // Compute incremental cost for assigning idx to A or B
        let mut cost_a = 0usize;
        let mut cost_b = 0usize;
        for j in 0..idx {
            if part[j] {
                cost_b += self.adj[idx][j] as usize;
            } else {
                cost_a += self.adj[idx][j] as usize;
            }
        }

        // Try the cheaper option first for better pruning
        if cost_a <= cost_b {
            part[idx] = true;
            self.branch_and_bound(idx + 1, count_a + 1, current_cut + cost_a, part);
            part[idx] = false;
            self.branch_and_bound(idx + 1, count_a, current_cut + cost_b, part);
        } else {
            self.branch_and_bound(idx + 1, count_a, current_cut + cost_b, part);
            part[idx] = true;
            self.branch_and_bound(idx + 1, count_a + 1, current_cut + cost_a, part);
            part[idx] = false;
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_ascii_whitespace();

    let n: usize = iter.next().unwrap().parse().unwrap();
    let m: usize = iter.next().unwrap().parse().unwrap();

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let half = n / 2;
    let mut adj = vec![vec![0i32; n]; n];

    for _ in 0..m {
        let u: usize = iter.next().unwrap().parse().unwrap();
        let v: usize = iter.next().unwrap().parse().unwrap();
        if u != v {
            adj[u][v] += 1;
            adj[v][u] += 1;
        }
    }

    // Reorder vertices by degree descending for better B&B pruning
    let degree: Vec<i32> = (0..n).map(|v| (0..n).map(|w| adj[v][w]).sum()).collect();
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by(|&a, &b| degree[b].cmp(&degree[a]));
    let mut rank = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        rank[v] = i;
    }

    let mut adj2 = vec![vec![0i32; n]; n];
    for i in 0..n {
        for j in 0..n {
            adj2[i][j] = adj[order[i]][order[j]];
        }
    }

    let mut solver = Solver {
        adj: adj2,
        n,
        half,
        best_cut: usize::MAX,
        best_part: vec![false; n],
    };

    // Initial partition: first half (by rank) in A
    let mut init_part = vec![false; n];
    for i in 0..half {
        init_part[i] = true;
    }
    solver.run_kl(init_part);

    // Multiple KL starts with random permutations
    let mut rng = 0x1234567890abcdefu64;
    let num_starts = if n <= 20 { 20 } else if n <= 50 { 100 } else { 200 };

    let mut perm: Vec<usize> = (0..n).collect();
    for _ in 0..num_starts {
        // Fisher-Yates shuffle
        for i in (1..n).rev() {
            rng = rng
                .wrapping_mul(6364136223846793005)
                .wrapping_add(1442695040888963407);
            let j = ((rng >> 33) as usize) % (i + 1);
            perm.swap(i, j);
        }
        let mut part = vec![false; n];
        for i in 0..half {
            part[perm[i]] = true;
        }
        solver.run_kl(part);
    }

    // Exact branch and bound for small n
    if n <= 50 {
        let mut part = vec![false; n];
        solver.branch_and_bound(0, 0, 0, &mut part);
    }

    // Remap best_part back to original vertex indices
    let cut = solver.best_cut;
    let part_a: Vec<usize> = (0..n)
        .filter(|&i| solver.best_part[rank[i]])
        .collect();

    println!("{}", cut);
    let s: Vec<String> = part_a.iter().map(|v| v.to_string()).collect();
    println!("{}", s.join(" "));
}
