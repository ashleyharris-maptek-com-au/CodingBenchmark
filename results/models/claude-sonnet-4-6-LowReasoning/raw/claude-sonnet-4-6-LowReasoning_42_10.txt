; SPIR-V Geometry Shader
; Thick smooth curve via Catmull-Rom tangents, InputLinesAdjacency -> OutputTriangleStrip (10 vertices)

              OpCapability Geometry
         %1 = OpExtInstImport "GLSL.std.450"
              OpMemoryModel Logical GLSL450
              OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor
              OpExecutionMode %main InputLinesAdjacency
              OpExecutionMode %main OutputTriangleStrip
              OpExecutionMode %main OutputVertices 10
              OpExecutionMode %main Invocations 1

; Decorations
              OpDecorate %gl_PerVertex_in Block
              OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
              OpDecorate %gl_PerVertex_out Block
              OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
              OpDecorate %inColor Location 2
              OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
        %int = OpTypeInt 32 1
       %uint = OpTypeInt 32 0

%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

       %uint_4 = OpConstant %uint 4
    %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
    %arr_v3_4  = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %ptr_in_v4 = OpTypePointer Input %v4float
     %ptr_in_v3 = OpTypePointer Input %v3float
  %ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
    %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %ptr_out_v4 = OpTypePointer Output %v4float
    %ptr_out_v3 = OpTypePointer Output %v3float

; Variables
      %gl_in = OpVariable %ptr_in_arr_pv Input
     %gl_out = OpVariable %ptr_out_pv Output
    %inColor = OpVariable %ptr_in_arr_v3 Input
   %outColor = OpVariable %ptr_out_v3 Output

; Constants
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
   %float_0   = OpConstant %float 0.0
   %float_025 = OpConstant %float 0.25
   %float_05  = OpConstant %float 0.5
   %float_075 = OpConstant %float 0.75
   %float_1   = OpConstant %float 1.0
   %float_002 = OpConstant %float 0.02

  %func_type = OpTypeFunction %void

; Main function
      %main = OpFunction %void None %func_type
     %entry = OpLabel

; Load all 4 input positions
     %p0_ac = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ac
     %p1_ac = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ac
     %p2_ac = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ac
     %p3_ac = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %p3_ac

; Load colors c1 (index 1) and c2 (index 2)
     %c1_ac = OpAccessChain %ptr_in_v3 %inColor %int_1
         %c1 = OpLoad %v3float %c1_ac
     %c2_ac = OpAccessChain %ptr_in_v3 %inColor %int_2
         %c2 = OpLoad %v3float %c2_ac

; Position delta dp4 = p2 - p1
       %dp4 = OpFSub %v4float %p2 %p1

; Color delta dc3 = c2 - c1
       %dc3 = OpFSub %v3float %c2 %c1

; Extract xy for Catmull-Rom tangent computation
       %p0x = OpCompositeExtract %float %p0 0
       %p0y = OpCompositeExtract %float %p0 1
       %p1x = OpCompositeExtract %float %p1 0
       %p1y = OpCompositeExtract %float %p1 1
       %p2x = OpCompositeExtract %float %p2 0
       %p2y = OpCompositeExtract %float %p2 1
       %p3x = OpCompositeExtract %float %p3 0
       %p3y = OpCompositeExtract %float %p3 1

; T1 = P2 - P0 (tangent direction at start, t=0)
       %t1x = OpFSub %float %p2x %p0x
       %t1y = OpFSub %float %p2y %p0y
        %t1 = OpCompositeConstruct %v2float %t1x %t1y

; T2 = P3 - P1 (tangent direction at end, t=1)
       %t2x = OpFSub %float %p3x %p1x
       %t2y = OpFSub %float %p3y %p1y
        %t2 = OpCompositeConstruct %v2float %t2x %t2y

; Tangent delta dt2 = T2 - T1
       %dt2 = OpFSub %v2float %t2 %t1

; Output gl_Position accessor
   %opos_ac = OpAccessChain %ptr_out_v4 %gl_out %int_0

; ===== Point 0: t = 0.0, pos = p1, tan = t1, col = c1 =====
      %tn0 = OpExtInst %v2float %1 Normalize %t1
      %n0x = OpCompositeExtract %float %tn0 0
      %n0y = OpCompositeExtract %float %tn0 1
  %nn0y = OpFNegate %float %n0y
    %pd0x = OpFMul %float %nn0y %float_002
    %pd0y = OpFMul %float %n0x %float_002
    %p1_x = OpCompositeExtract %float %p1 0
    %p1_y = OpCompositeExtract %float %p1 1
    %p1_z = OpCompositeExtract %float %p1 2
    %p1_w = OpCompositeExtract %float %p1 3
   %v0p_x = OpFAdd %float %p1_x %pd0x
   %v0p_y = OpFAdd %float %p1_y %pd0y
  %v0pos_p = OpCompositeConstruct %v4float %v0p_x %v0p_y %p1_z %p1_w
             OpStore %opos_ac %v0pos_p
             OpStore %outColor %c1
             OpEmitVertex
   %v0m_x = OpFSub %float %p1_x %pd0x
   %v0m_y = OpFSub %float %p1_y %pd0y
  %v0pos_m = OpCompositeConstruct %v4float %v0m_x %v0m_y %p1_z %p1_w
             OpStore %opos_ac %v0pos_m
             OpStore %outColor %c1
             OpEmitVertex

; ===== Point 1: t = 0.25 =====
  %dp4_025 = OpVectorTimesScalar %v4float %dp4 %float_025
   %pos1_4 = OpFAdd %v4float %p1 %dp4_025
  %dt2_025 = OpVectorTimesScalar %v2float %dt2 %float_025
   %tan1_2 = OpFAdd %v2float %t1 %dt2_025
  %dc3_025 = OpVectorTimesScalar %v3float %dc3 %float_025
   %col1_3 = OpFAdd %v3float %c1 %dc3_025
      %tn1 = OpExtInst %v2float %1 Normalize %tan1_2
      %n1x = OpCompositeExtract %float %tn1 0
      %n1y = OpCompositeExtract %float %tn1 1
  %nn1y = OpFNegate %float %n1y
    %pd1x = OpFMul %float %nn1y %float_002
    %pd1y = OpFMul %float %n1x %float_002
  %pos1_x = OpCompositeExtract %float %pos1_4 0
  %pos1_y = OpCompositeExtract %float %pos1_4 1
  %pos1_z = OpCompositeExtract %float %pos1_4 2
  %pos1_w = OpCompositeExtract %float %pos1_4 3
   %v1p_x = OpFAdd %float %pos1_x %pd1x
   %v1p_y = OpFAdd %float %pos1_y %pd1y
  %v1pos_p = OpCompositeConstruct %v4float %v1p_x %v1p_y %pos1_z %pos1_w
             OpStore %opos_ac %v1pos_p
             OpStore %outColor %col1_3
             OpEmitVertex
   %v1m_x = OpFSub %float %pos1_x %pd1x
   %v1m_y = OpFSub %float %pos1_y %pd1y
  %v1pos_m = OpCompositeConstruct %v4float %v1m_x %v1m_y %pos1_z %pos1_w
             OpStore %opos_ac %v1pos_m
             OpStore %outColor %col1_3
             OpEmitVertex

; ===== Point 2: t = 0.5 =====
   %dp4_05 = OpVectorTimesScalar %v4float %dp4 %float_05
   %pos2_4 = OpFAdd %v4float %p1 %dp4_05
   %dt2_05 = OpVectorTimesScalar %v2float %dt2 %float_05
   %tan2_2 = OpFAdd %v2float %t1 %dt2_05
   %dc3_05 = OpVectorTimesScalar %v3float %dc3 %float_05
   %col2_3 = OpFAdd %v3float %c1 %dc3_05
      %tn2 = OpExtInst %v2float %1 Normalize %tan2_2
      %n2x = OpCompositeExtract %float %tn2 0
      %n2y = OpCompositeExtract %float %tn2 1
  %nn2y = OpFNegate %float %n2y
    %pd2x = OpFMul %float %nn2y %float_002
    %pd2y = OpFMul %float %n2x %float_002
  %pos2_x = OpCompositeExtract %float %pos2_4 0
  %pos2_y = OpCompositeExtract %float %pos2_4 1
  %pos2_z = OpCompositeExtract %float %pos2_4 2
  %pos2_w = OpCompositeExtract %float %pos2_4 3
   %v2p_x = OpFAdd %float %pos2_x %pd2x
   %v2p_y = OpFAdd %float %pos2_y %pd2y
  %v2pos_p = OpCompositeConstruct %v4float %v2p_x %v2p_y %pos2_z %pos2_w
             OpStore %opos_ac %v2pos_p
             OpStore %outColor %col2_3
             OpEmitVertex
   %v2m_x = OpFSub %float %pos2_x %pd2x
   %v2m_y = OpFSub %float %pos2_y %pd2y
  %v2pos_m = OpCompositeConstruct %v4float %v2m_x %v2m_y %pos2_z %pos2_w
             OpStore %opos_ac %v2pos_m
             OpStore %outColor %col2_3
             OpEmitVertex

; ===== Point 3: t = 0.75 =====
  %dp4_075 = OpVectorTimesScalar %v4float %dp4 %float_075
   %pos3_4 = OpFAdd %v4float %p1 %dp4_075
  %dt2_075 = OpVectorTimesScalar %v2float %dt2 %float_075
   %tan3_2 = OpFAdd %v2float %t1 %dt2_075
  %dc3_075 = OpVectorTimesScalar %v3float %dc3 %float_075
   %col3_3 = OpFAdd %v3float %c1 %dc3_075
      %tn3 = OpExtInst %v2float %1 Normalize %tan3_2
      %n3x = OpCompositeExtract %float %tn3 0
      %n3y = OpCompositeExtract %float %tn3 1
  %nn3y = OpFNegate %float %n3y
    %pd3x = OpFMul %float %nn3y %float_002
    %pd3y = OpFMul %float %n3x %float_002
  %pos3_x = OpCompositeExtract %float %pos3_4 0
  %pos3_y = OpCompositeExtract %float %pos3_4 1
  %pos3_z = OpCompositeExtract %float %pos3_4 2
  %pos3_w = OpCompositeExtract %float %pos3_4 3
   %v3p_x = OpFAdd %float %pos3_x %pd3x
   %v3p_y = OpFAdd %float %pos3_y %pd3y
  %v3pos_p = OpCompositeConstruct %v4float %v3p_x %v3p_y %pos3_z %pos3_w
             OpStore %opos_ac %v3pos_p
             OpStore %outColor %col3_3
             OpEmitVertex
   %v3m_x = OpFSub %float %pos3_x %pd3x
   %v3m_y = OpFSub %float %pos3_y %pd3y
  %v3pos_m = OpCompositeConstruct %v4float %v3m_x %v3m_y %pos3_z %pos3_w
             OpStore %opos_ac %v3pos_m
             OpStore %outColor %col3_3
             OpEmitVertex

; ===== Point 4: t = 1.0, pos = p2, tan = t2, col = c2 =====
      %tn4 = OpExtInst %v2float %1 Normalize %t2
      %n4x = OpCompositeExtract %float %tn4 0
      %n4y = OpCompositeExtract %float %tn4 1
  %nn4y = OpFNegate %float %n4y
    %pd4x = OpFMul %float %nn4y %float_002
    %pd4y = OpFMul %float %n4x %float_002
    %p2_x = OpCompositeExtract %float %p2 0
    %p2_y = OpCompositeExtract %float %p2 1
    %p2_z = OpCompositeExtract %float %p2 2
    %p2_w = OpCompositeExtract %float %p2 3
   %v4p_x = OpFAdd %float %p2_x %pd4x
   %v4p_y = OpFAdd %float %p2_y %pd4y
  %v4pos_p = OpCompositeConstruct %v4float %v4p_x %v4p_y %p2_z %p2_w
             OpStore %opos_ac %v4pos_p
             OpStore %outColor %c2
             OpEmitVertex
   %v4m_x = OpFSub %float %p2_x %pd4x
   %v4m_y = OpFSub %float %p2_y %pd4y
  %v4pos_m = OpCompositeConstruct %v4float %v4m_x %v4m_y %p2_z %p2_w
             OpStore %opos_ac %v4pos_m
             OpStore %outColor %c2
             OpEmitVertex

             OpEndPrimitive
             OpReturn
             OpFunctionEnd
