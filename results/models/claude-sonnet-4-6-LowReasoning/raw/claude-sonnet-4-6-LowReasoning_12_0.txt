using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Text;

class Program
{
    static double[] QMul(double[] a, double[] b) => new double[]{
        a[0]*b[0]-a[1]*b[1]-a[2]*b[2]-a[3]*b[3],
        a[0]*b[1]+a[1]*b[0]+a[2]*b[3]-a[3]*b[2],
        a[0]*b[2]-a[1]*b[3]+a[2]*b[0]+a[3]*b[1],
        a[0]*b[3]+a[1]*b[2]-a[2]*b[1]+a[3]*b[0]
    };

    static (double, double, double) QRot(double x, double y, double z, double[] q)
    {
        double w=q[0], qx=q[1], qy=q[2], qz=q[3];
        double tx=2*(qy*z-qz*y), ty=2*(qz*x-qx*z), tz=2*(qx*y-qy*x);
        return (x+w*tx+qy*tz-qz*ty, y+w*ty+qz*tx-qx*tz, z+w*tz+qx*ty-qy*tx);
    }

    static List<double[]> GetRotations()
    {
        double s = Math.Sqrt(0.5);
        double[][] gens = {
            new[]{s,s,0.0,0.0}, new[]{s,-s,0.0,0.0},
            new[]{s,0.0,s,0.0}, new[]{s,0.0,-s,0.0},
            new[]{s,0.0,0.0,s}, new[]{s,0.0,0.0,-s}
        };
        var seen = new HashSet<string>();
        var result = new List<double[]>();
        var queue = new Queue<double[]>();

        void Enq(double[] r)
        {
            double n = Math.Sqrt(r[0]*r[0]+r[1]*r[1]+r[2]*r[2]+r[3]*r[3]);
            for(int i=0;i<4;i++) r[i]/=n;
            for(int i=0;i<4;i++) {
                if(Math.Abs(r[i]) > 1e-8) {
                    if(r[i] < 0) for(int j=0;j<4;j++) r[j]=-r[j];
                    break;
                }
            }
            var key = $"{r[0]:F3},{r[1]:F3},{r[2]:F3},{r[3]:F3}";
            if(seen.Add(key)) { result.Add(r); queue.Enqueue(r); }
        }

        Enq(new[]{1.0,0.0,0.0,0.0});
        while(queue.Count > 0) {
            var q = queue.Dequeue();
            foreach(var g in gens) Enq(QMul(q, g));
        }
        return result;
    }

    struct RI { public double[] Q; public double Dx,Dy,Dz,Ox,Oy,Oz; }

    static void Main()
    {
        var sep = StringSplitOptions.RemoveEmptyEntries;
        char[] ws = {' ','\t','\r'};

        var p0 = (Console.ReadLine() ?? "").Split(ws, sep);
        int V = int.Parse(p0[0]), F = int.Parse(p0[1]);

        double[] vx = new double[V], vy = new double[V], vz = new double[V];
        for(int i = 0; i < V; i++) {
            var p = (Console.ReadLine() ?? "").Split(ws, sep);
            vx[i]=double.Parse(p[0]); vy[i]=double.Parse(p[1]); vz[i]=double.Parse(p[2]);
        }
        for(int i = 0; i < F; i++) Console.ReadLine();

        var pC = (Console.ReadLine() ?? "").Split(ws, sep);
        double W=double.Parse(pC[0]), H=double.Parse(pC[1]), D=double.Parse(pC[2]);

        var ris = new List<RI>();
        var dimSeen = new HashSet<string>();
        foreach(var q in GetRotations()) {
            double x0=1e18,y0=1e18,z0=1e18,x1=-1e18,y1=-1e18,z1=-1e18;
            for(int i=0;i<V;i++) {
                var (rx,ry,rz) = QRot(vx[i],vy[i],vz[i],q);
                if(rx<x0)x0=rx; if(ry<y0)y0=ry; if(rz<z0)z0=rz;
                if(rx>x1)x1=rx; if(ry>y1)y1=ry; if(rz>z1)z1=rz;
            }
            double dx=x1-x0, dy=y1-y0, dz=z1-z0;
            if(dx<1e-9 || dy<1e-9 || dz<1e-9) continue;
            var k = $"{dx:F4},{dy:F4},{dz:F4}";
            if(dimSeen.Add(k)) ris.Add(new RI{Q=q,Dx=dx,Dy=dy,Dz=dz,Ox=x0,Oy=y0,Oz=z0});
        }
        if(ris.Count == 0) { Console.WriteLine("0"); return; }

        ris.Sort((a,b) => (a.Dx*a.Dy*a.Dz).CompareTo(b.Dx*b.Dy*b.Dz));

        RI best = ris[0]; long bestN = 0;
        foreach(var r in ris) {
            long n = (long)(W/r.Dx) * (long)(H/r.Dy) * (long)(D/r.Dz);
            if(n > bestN) { bestN = n; best = r; }
        }
        if(bestN == 0) { Console.WriteLine("0"); return; }

        var sw = Stopwatch.StartNew();
        var result = new List<(double,double,double,double[])>();

        const long LARGE = 500000L;
        if(bestN >= LARGE) {
            int nx=(int)(W/best.Dx), ny=(int)(H/best.Dy), nz=(int)(D/best.Dz);
            result.Capacity = nx*ny*nz;
            for(int ix=0;ix<nx;ix++)
            for(int iy=0;iy<ny;iy++)
            for(int iz=0;iz<nz;iz++)
                result.Add((ix*best.Dx-best.Ox, iy*best.Dy-best.Oy, iz*best.Dz-best.Oz, best.Q));
        } else {
            EPPack(ris, W, H, D, result, sw, 260.0);
            if((long)result.Count < bestN) {
                result.Clear();
                int nx=(int)(W/best.Dx), ny=(int)(H/best.Dy), nz=(int)(D/best.Dz);
                result.Capacity = nx*ny*nz;
                for(int ix=0;ix<nx;ix++)
                for(int iy=0;iy<ny;iy++)
                for(int iz=0;iz<nz;iz++)
                    result.Add((ix*best.Dx-best.Ox, iy*best.Dy-best.Oy, iz*best.Dz-best.Oz, best.Q));
            }
        }

        var sb = new StringBuilder(result.Count * 70);
        sb.Append(result.Count); sb.Append('\n');
        foreach(var (tx,ty,tz,q) in result) {
            sb.Append(tx.ToString("G6")); sb.Append(' ');
            sb.Append(ty.ToString("G6")); sb.Append(' ');
            sb.Append(tz.ToString("G6")); sb.Append(' ');
            sb.Append(q[0].ToString("G6")); sb.Append(' ');
            sb.Append(q[1].ToString("G6")); sb.Append(' ');
            sb.Append(q[2].ToString("G6")); sb.Append(' ');
            sb.Append(q[3].ToString("G6")); sb.Append('\n');
        }
        Console.Write(sb);
    }

    static void EPPack(List<RI> ris, double W, double H, double D,
        List<(double,double,double,double[])> res, Stopwatch sw, double tLimit)
    {
        const double EPS = 1e-7;

        double cs = double.MaxValue;
        foreach(var r in ris)
            cs = Math.Min(cs, Math.Min(r.Dx, Math.Min(r.Dy, r.Dz)));
        cs = Math.Max(cs * 0.5, 1e-4);

        var spatialHash = new Dictionary<(int,int,int), List<int>>(4096);
        int cap = 4096;
        double[] bx = new double[cap*6];
        int[] stamps = new int[cap];
        int nBoxes = 0, gen = 0;

        void PlaceBox(double x1,double y1,double z1,double x2,double y2,double z2)
        {
            if(nBoxes >= cap) {
                int nc = cap * 2;
                var nb = new double[nc*6]; Array.Copy(bx, nb, bx.Length); bx = nb;
                var ns = new int[nc]; Array.Copy(stamps, ns, stamps.Length); stamps = ns;
                cap = nc;
            }
            int b = nBoxes * 6;
            bx[b]=x1; bx[b+1]=y1; bx[b+2]=z1; bx[b+3]=x2; bx[b+4]=y2; bx[b+5]=z2;
            for(int cx=(int)(x1/cs); cx<=(int)((x2+EPS)/cs); cx++)
            for(int cy=(int)(y1/cs); cy<=(int)((y2+EPS)/cs); cy++)
            for(int cz=(int)(z1/cs); cz<=(int)((z2+EPS)/cs); cz++) {
                var k = (cx,cy,cz);
                if(!spatialHash.TryGetValue(k, out var lst)) { lst=new List<int>(4); spatialHash[k]=lst; }
                lst.Add(nBoxes);
            }
            nBoxes++;
        }

        bool Overlaps(double x1,double y1,double z1,double x2,double y2,double z2)
        {
            if(++gen == 0) { gen = 1; Array.Clear(stamps, 0, stamps.Length); }
            for(int cx=(int)(x1/cs); cx<=(int)((x2-EPS)/cs); cx++)
            for(int cy=(int)(y1/cs); cy<=(int)((y2-EPS)/cs); cy++)
            for(int cz=(int)(z1/cs); cz<=(int)((z2-EPS)/cs); cz++) {
                if(!spatialHash.TryGetValue((cx,cy,cz), out var lst)) continue;
                foreach(int idx in lst) {
                    if(stamps[idx] == gen) continue;
                    stamps[idx] = gen;
                    int b = idx*6;
                    if(x1<bx[b+3]-EPS && x2>bx[b]+EPS &&
                       y1<bx[b+4]-EPS && y2>bx[b+1]+EPS &&
                       z1<bx[b+5]-EPS && z2>bx[b+2]+EPS) return true;
                }
            }
            return false;
        }

        var pq = new PriorityQueue<(double x,double y,double z), double>();
        var epSeen = new HashSet<(long,long,long)>();
        const double Q = 1e4;

        void TryEP(double x, double y, double z)
        {
            if(x > W+EPS || y > H+EPS || z > D+EPS) return;
            var k = ((long)Math.Round(x*Q), (long)Math.Round(y*Q), (long)Math.Round(z*Q));
            if(epSeen.Add(k)) pq.Enqueue((x,y,z), x+y+z);
        }

        TryEP(0, 0, 0);

        while(pq.Count > 0 && sw.Elapsed.TotalSeconds < tLimit) {
            var (ex,ey,ez) = pq.Dequeue();
            foreach(var ri in ris) {
                double x2=ex+ri.Dx, y2=ey+ri.Dy, z2=ez+ri.Dz;
                if(x2 > W+EPS || y2 > H+EPS || z2 > D+EPS) continue;
                if(!Overlaps(ex,ey,ez,x2,y2,z2)) {
                    res.Add((ex-ri.Ox, ey-ri.Oy, ez-ri.Oz, ri.Q));
                    PlaceBox(ex,ey,ez,x2,y2,z2);
                    TryEP(x2,ey,ez); TryEP(ex,y2,ez); TryEP(ex,ey,z2);
                    break;
                }
            }
        }
    }
}
