import struct

def w(v):
    return struct.pack('<I', v)

def instr(opcode, *operands):
    wc = 1 + len(operands)
    return w((wc << 16) | opcode) + b''.join(w(o) for o in operands)

def string_words(s):
    b = s.encode('utf-8') + b'\x00'
    pad = (4 - len(b) % 4) % 4
    b += b'\x00' * pad
    return b
