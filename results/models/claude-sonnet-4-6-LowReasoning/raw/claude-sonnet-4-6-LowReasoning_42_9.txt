; SPIR-V geometry shader: thick smooth curve with miter joins
; Input: InputLinesAdjacency (4 vertices), Output: OutputTriangleStrip (4 vertices)

               OpCapability Geometry
          %1  = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input color array location
               OpDecorate %in_color Location 2

; Output color location
               OpDecorate %out_color Location 0

; Types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

; Constants
     %uint_4 = OpConstant %uint 4
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
  %float_thickness = OpConstant %float 0.025

; Struct types
%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

; Array type for input (size 4 for InputLinesAdjacency)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4

; Array type for input color (size 4)
%arr_v3float_4 = OpTypeArray %v3float %uint_4

; Pointer types
%ptr_Input_arr_pv   = OpTypePointer Input  %arr_pv_in
%ptr_Input_v4float  = OpTypePointer Input  %v4float
%ptr_Input_arr_v3   = OpTypePointer Input  %arr_v3float_4
%ptr_Input_v3float  = OpTypePointer Input  %v3float
%ptr_Output_pv      = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

; Variables
      %gl_in = OpVariable %ptr_Input_arr_pv   Input
     %gl_out = OpVariable %ptr_Output_pv      Output
   %in_color = OpVariable %ptr_Input_arr_v3   Input
  %out_color = OpVariable %ptr_Output_v3float Output

; Function type
    %fn_void = OpTypeFunction %void

; Main function
      %main  = OpFunction %void None %fn_void
     %entry  = OpLabel

; --- Load positions ---
  %ptr_pos0  = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
      %pos0  = OpLoad %v4float %ptr_pos0

  %ptr_pos1  = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
      %pos1  = OpLoad %v4float %ptr_pos1

  %ptr_pos2  = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
      %pos2  = OpLoad %v4float %ptr_pos2

  %ptr_pos3  = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
      %pos3  = OpLoad %v4float %ptr_pos3

; --- Load colors ---
  %ptr_col1  = OpAccessChain %ptr_Input_v3float %in_color %int_1
      %col1  = OpLoad %v3float %ptr_col1
  %ptr_col2  = OpAccessChain %ptr_Input_v3float %in_color %int_2
      %col2  = OpLoad %v3float %ptr_col2

; --- Extract xy components ---
  %pos0_x    = OpCompositeExtract %float %pos0 0
  %pos0_y    = OpCompositeExtract %float %pos0 1
  %pos0_xy   = OpCompositeConstruct %v2float %pos0_x %pos0_y

  %pos1_x    = OpCompositeExtract %float %pos1 0
  %pos1_y    = OpCompositeExtract %float %pos1 1
  %pos1_xy   = OpCompositeConstruct %v2float %pos1_x %pos1_y
  %pos1_z    = OpCompositeExtract %float %pos1 2
  %pos1_w    = OpCompositeExtract %float %pos1 3

  %pos2_x    = OpCompositeExtract %float %pos2 0
  %pos2_y    = OpCompositeExtract %float %pos2 1
  %pos2_xy   = OpCompositeConstruct %v2float %pos2_x %pos2_y
  %pos2_z    = OpCompositeExtract %float %pos2 2
  %pos2_w    = OpCompositeExtract %float %pos2 3

  %pos3_x    = OpCompositeExtract %float %pos3 0
  %pos3_y    = OpCompositeExtract %float %pos3 1
  %pos3_xy   = OpCompositeConstruct %v2float %pos3_x %pos3_y

; --- Compute tangent at start: t0 = normalize(pos2.xy - pos0.xy) ---
  %diff0     = OpFSub %v2float %pos2_xy %pos0_xy
  %t0        = OpExtInst %v2float %1 Normalize %diff0

; --- Compute tangent at end: t1 = normalize(pos3.xy - pos1.xy) ---
  %diff1     = OpFSub %v2float %pos3_xy %pos1_xy
  %t1        = OpExtInst %v2float %1 Normalize %diff1

; --- Compute perpendiculars: perp = vec2(-t.y, t.x) ---
  %t0_x      = OpCompositeExtract %float %t0 0
  %t0_y      = OpCompositeExtract %float %t0 1
  %neg_t0_y  = OpFNegate %float %t0_y
  %perp0     = OpCompositeConstruct %v2float %neg_t0_y %t0_x

  %t1_x      = OpCompositeExtract %float %t1 0
  %t1_y      = OpCompositeExtract %float %t1 1
  %neg_t1_y  = OpFNegate %float %t1_y
  %perp1     = OpCompositeConstruct %v2float %neg_t1_y %t1_x

; --- Scale by thickness ---
  %thick2    = OpCompositeConstruct %v2float %float_thickness %float_thickness
  %perp0_t   = OpFMul %v2float %perp0 %thick2
  %perp1_t   = OpFMul %v2float %perp1 %thick2

; === Emit vertex 0: start + perp0*thickness ===
  %v0_xy     = OpFAdd %v2float %pos1_xy %perp0_t
  %v0_x      = OpCompositeExtract %float %v0_xy 0
  %v0_y      = OpCompositeExtract %float %v0_xy 1
  %v0_pos    = OpCompositeConstruct %v4float %v0_x %v0_y %pos1_z %pos1_w
  %ptr_out0  = OpAccessChain %ptr_Output_v4float %gl_out %int_0
               OpStore %ptr_out0 %v0_pos
               OpStore %out_color %col1
               OpEmitVertex

; === Emit vertex 1: start - perp0*thickness ===
  %v1_xy     = OpFSub %v2float %pos1_xy %perp0_t
  %v1_x      = OpCompositeExtract %float %v1_xy 0
  %v1_y      = OpCompositeExtract %float %v1_xy 1
  %v1_pos    = OpCompositeConstruct %v4float %v1_x %v1_y %pos1_z %pos1_w
  %ptr_out1  = OpAccessChain %ptr_Output_v4float %gl_out %int_0
               OpStore %ptr_out1 %v1_pos
               OpStore %out_color %col1
               OpEmitVertex

; === Emit vertex 2: end + perp1*thickness ===
  %v2_xy     = OpFAdd %v2float %pos2_xy %perp1_t
  %v2_x      = OpCompositeExtract %float %v2_xy 0
  %v2_y      = OpCompositeExtract %float %v2_xy 1
  %v2_pos    = OpCompositeConstruct %v4float %v2_x %v2_y %pos2_z %pos2_w
  %ptr_out2  = OpAccessChain %ptr_Output_v4float %gl_out %int_0
               OpStore %ptr_out2 %v2_pos
               OpStore %out_color %col2
               OpEmitVertex

; === Emit vertex 3: end - perp1*thickness ===
  %v3_xy     = OpFSub %v2float %pos2_xy %perp1_t
  %v3_x      = OpCompositeExtract %float %v3_xy 0
  %v3_y      = OpCompositeExtract %float %v3_xy 1
  %v3_pos    = OpCompositeConstruct %v4float %v3_x %v3_y %pos2_z %pos2_w
  %ptr_out3  = OpAccessChain %ptr_Output_v4float %gl_out %int_0
               OpStore %ptr_out3 %v3_pos
               OpStore %out_color %col2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
