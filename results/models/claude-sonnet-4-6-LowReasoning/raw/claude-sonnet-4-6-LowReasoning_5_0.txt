use std::io::{self, BufRead, Write};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<usize> = first_line.split_whitespace()
        .map(|x| x.parse().unwrap())
        .collect();

    let width = parts[0];
    let height = parts[1];
    let obstacle_count = parts[2];
    let require_cycle = parts[3] == 1;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|x| x.parse().unwrap())
            .collect();
        obstacles.insert((coords[0], coords[1]));
    }

    let total_cells = width * height;
    let target_cells = total_cells - obstacle_count;

    // Check if (0,0) is an obstacle
    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }

    // For large grids, use structured/heuristic approach
    // For smaller grids, use backtracking with pruning

    let result = if total_cells <= 400 {
        solve_backtrack(width, height, &obstacles, require_cycle, target_cells)
    } else {
        // Try structured approach first for large grids
        solve_large(width, height, &obstacles, require_cycle, target_cells)
    };

    let stdout = io::stdout();
    let mut out = stdout.lock();

    match result {
        Some(path) => {
            writeln!(out, "{}", path.len()).unwrap();
            for (x, y) in &path {
                writeln!(out, "{} {}", x, y).unwrap();
            }
        }
        None => {
            writeln!(out, "0").unwrap();
        }
    }
}

fn solve_backtrack(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target_cells: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![vec![false; width]; height];
    visited[0][0] = true;
    let mut path = vec![(0usize, 0usize)];

    if target_cells == 1 {
        return Some(path);
    }

    backtrack(
        &mut path,
        &mut visited,
        width,
        height,
        obstacles,
        require_cycle,
        target_cells,
    )
}

fn backtrack(
    path: &mut Vec<(usize, usize)>,
    visited: &mut Vec<Vec<bool>>,
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target_cells: usize,
) -> Option<Vec<(usize, usize)>> {
    if path.len() == target_cells {
        if require_cycle {
            let last = *path.last().unwrap();
            let start = path[0];
            if are_adjacent(last, start) {
                return Some(path.clone());
            } else {
                return None;
            }
        }
        return Some(path.clone());
    }

    let (cx, cy) = *path.last().unwrap();
    let remaining = target_cells - path.len();

    // Connectivity pruning: check if unvisited cells form connected region
    if remaining > 3 && path.len() % 5 == 0 {
        if !connectivity_check(cx, cy, visited, width, height, obstacles, remaining) {
            return None;
        }
    }

    let neighbors = get_neighbors(cx, cy, width, height);

    // Sort by Warnsdorff heuristic for better performance
    let mut valid_neighbors: Vec<(usize, usize, usize)> = neighbors.iter()
        .filter(|&&(nx, ny)| !visited[ny][nx] && !obstacles.contains(&(nx, ny)))
        .map(|&(nx, ny)| {
            let degree = get_neighbors(nx, ny, width, height).iter()
                .filter(|&&(nnx, nny)| !visited[nny][nnx] && !obstacles.contains(&(nnx, nny)))
                .count();
            (nx, ny, degree)
        })
        .collect();

    valid_neighbors.sort_by_key(|&(_, _, d)| d);

    for (nx, ny, _) in valid_neighbors {
        visited[ny][nx] = true;
        path.push((nx, ny));

        if let Some(result) = backtrack(path, visited, width, height, obstacles, require_cycle, target_cells) {
            return Some(result);
        }

        path.pop();
        visited[ny][nx] = false;
    }

    None
}

fn connectivity_check(
    cx: usize,
    cy: usize,
    visited: &Vec<Vec<bool>>,
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    remaining: usize,
) -> bool {
    // BFS from current position through unvisited cells
    let mut reachable = 0;
    let mut queue = std::collections::VecDeque::new();
    let mut seen = vec![vec![false; width]; height];

    for (nx, ny) in get_neighbors(cx, cy, width, height) {
        if !visited[ny][nx] && !obstacles.contains(&(nx, ny)) && !seen[ny][nx] {
            seen[ny][nx] = true;
            queue.push_back((nx, ny));
            reachable += 1;
        }
    }

    while let Some((x, y)) = queue.pop_front() {
        for (nx, ny) in get_neighbors(x, y, width, height) {
            if !visited[ny][nx] && !obstacles.contains(&(nx, ny)) && !seen[ny][nx] {
                seen[ny][nx] = true;
                queue.push_back((nx, ny));
                reachable += 1;
            }
        }
    }

    reachable == remaining
}

fn get_neighbors(x: usize, y: usize, width: usize, height: usize) -> Vec<(usize, usize)> {
    let mut neighbors = Vec::new();
    if x > 0 { neighbors.push((x - 1, y)); }
    if x + 1 < width { neighbors.push((x + 1, y)); }
    if y > 0 { neighbors.push((x, y - 1)); }
    if y + 1 < height { neighbors.push((x, y + 1)); }
    neighbors
}

fn are_adjacent(a: (usize, usize), b: (usize, usize)) -> bool {
    let dx = (a.0 as isize - b.0 as isize).abs();
    let dy = (a.1 as isize - b.1 as isize).abs();
    dx + dy == 1
}

fn solve_large(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target_cells: usize,
) -> Option<Vec<(usize, usize)>> {
    if obstacles.is_empty() {
        return solve_snake_no_obstacles(width, height, require_cycle);
    }

    // Try warnsdorff-like heuristic for large grids
    solve_warnsdorff(width, height, obstacles, require_cycle, target_cells)
}

fn solve_snake_no_obstacles(
    width: usize,
    height: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::new();

    for row in 0..height {
        if row % 2 == 0 {
            for col in 0..width {
                path.push((col, row));
            }
        } else {
            for col in (0..width).rev() {
                path.push((col, row));
            }
        }
    }

    if require_cycle {
        let last = *path.last().unwrap();
        if !are_adjacent(last, (0, 0)) {
            // Try column-based snake
            path.clear();
            for col in 0..width {
                if col % 2 == 0 {
                    for row in 0..height {
                        path.push((col, row));
                    }
                } else {
                    for row in (0..height).rev() {
                        path.push((col, row));
                    }
                }
            }
            let last = *path.last().unwrap();
            if !are_adjacent(last, (0, 0)) {
                return None;
            }
        }
    }

    Some(path)
}

fn solve_warnsdorff(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target_cells: usize,
) -> Option<Vec<(usize, usize)>> {
    let mut visited = vec![vec![false; width]; height];
    visited[0][0] = true;
    let mut path = vec![(0usize, 0usize)];

    loop {
        if path.len() == target_cells {
            if require_cycle {
                let last = *path.last().unwrap();
                if are_adjacent(last, (0, 0)) {
                    return Some(path);
                } else {
                    break;
                }
            }
            return Some(path);
        }

        let (cx, cy) = *path.last().unwrap();
        let neighbors = get_neighbors(cx, cy, width, height);

        let mut valid_neighbors: Vec<(usize, usize, usize)> = neighbors.iter()
            .filter(|&&(nx, ny)| !visited[ny][nx] && !obstacles.contains(&(nx, ny)))
            .map(|&(nx, ny)| {
                let degree = get_neighbors(nx, ny, width, height).iter()
                    .filter(|&&(nnx, nny)| !visited[nny][nnx] && !obstacles.contains(&(nnx, nny)))
                    .count();
                (nx, ny, degree)
            })
            .collect();

        if valid_neighbors.is_empty() {
            break;
        }

        valid_neighbors.sort_by_key(|&(_, _, d)| d);
        let (nx, ny, _) = valid_neighbors[0];

        visited[ny][nx] = true;
        path.push((nx, ny));
    }

    // Warnsdorff failed, fall back to backtracking if grid is small enough
    if target_cells <= 625 {
        solve_backtrack(width, height, obstacles, require_cycle, target_cells)
    } else {
        solve_backtrack_limited(width, height, obstacles, require_cycle, target_cells)
    }
}

fn solve_backtrack_limited(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target_cells: usize,
) -> Option<Vec<(usize, usize)>> {
    use std::time::Instant;
    let start_time = Instant::now();
    let time_limit = std::time::Duration::from_secs(270);

    let mut visited = vec![vec![false; width]; height];
    visited[0][0] = true;
    let mut path = vec![(0usize, 0usize)];

    backtrack_limited(
        &mut path,
        &mut visited,
        width,
        height,
        obstacles,
        require_cycle,
        target_cells,
        &start_time,
        &time_limit,
    )
}

fn backtrack_limited(
    path: &mut Vec<(usize, usize)>,
    visited: &mut Vec<Vec<bool>>,
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    require_cycle: bool,
    target_cells: usize,
    start_time: &std::time::Instant,
    time_limit: &std::time::Duration,
) -> Option<Vec<(usize, usize)>> {
    if path.len() == target_cells {
        if require_cycle {
            let last = *path.last().unwrap();
            let start = path[0];
            if are_adjacent(last, start) {
                return Some(path.clone());
            } else {
                return None;
            }
        }
        return Some(path.clone());
    }

    if path.len() % 1000 == 0 && start_time.elapsed() > *time_limit {
        return None;
    }

    let (cx, cy) = *path.last().unwrap();
    let remaining = target_cells - path.len();

    if remaining > 3 && path.len() % 10 == 0 {
        if !connectivity_check(cx, cy, visited, width, height, obstacles, remaining) {
            return None;
        }
    }

    let neighbors = get_neighbors(cx, cy, width, height);

    let mut valid_neighbors: Vec<(usize, usize, usize)> = neighbors.iter()
        .filter(|&&(nx, ny)| !visited[ny][nx] && !obstacles.contains(&(nx, ny)))
        .map(|&(nx, ny)| {
            let degree = get_neighbors(nx, ny, width, height).iter()
                .filter(|&&(nnx, nny)| !visited[nny][nnx] && !obstacles.contains(&(nnx, nny)))
                .count();
            (nx, ny, degree)
        })
        .collect();

    valid_neighbors.sort_by_key(|&(_, _, d)| d);

    for (nx, ny, _) in valid_neighbors {
        visited[ny][nx] = true;
        path.push((nx, ny));

        if let Some(result) = backtrack_limited(path, visited, width, height, obstacles, require_cycle, target_cells, start_time, time_limit) {
            return Some(result);
        }

        path.pop();
        visited[ny][nx] = false;
    }

    None
}
