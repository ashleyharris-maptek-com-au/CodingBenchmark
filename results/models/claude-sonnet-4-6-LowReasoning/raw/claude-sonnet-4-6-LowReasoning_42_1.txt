; SPIR-V geometry shader: expand points into diamonds
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

; Types
       %void = OpTypeVoid
       %bool = OpTypeBool
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

     %uint_1 = OpConstant %uint 1
      %int_0 = OpConstant %int 0

    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
  %float_h   = OpConstant %float 0.05

; Input gl_PerVertex block (array of 1 for InputPoints)
%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%ptr_in_v4float   = OpTypePointer Input %v4float
%gl_in            = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float  = OpTypePointer Output %v4float
%gl_out           = OpVariable %ptr_out_pv Output

; User-defined inputs: color at location 2 (array of 1)
%arr_v3float_1    = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3float_1
%ptr_in_v3float   = OpTypePointer Input %v3float
%in_color         = OpVariable %ptr_in_arr_v3 Input

; User-defined output: outColor at location 0
%ptr_out_v3float  = OpTypePointer Output %v3float
%out_color        = OpVariable %ptr_out_v3float Output

; Main function type
   %func_type = OpTypeFunction %void

       %main = OpFunction %void None %func_type
      %entry = OpLabel

; Load input position gl_in[0].gl_Position
  %ptr_pos_in = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
       %pos4  = OpLoad %v4float %ptr_pos_in

; Extract x, y, w
          %x = OpCompositeExtract %float %pos4 0
          %y = OpCompositeExtract %float %pos4 1
          %z = OpCompositeExtract %float %pos4 2
          %w = OpCompositeExtract %float %pos4 3

; Load color from in_color[0]
  %ptr_col_in = OpAccessChain %ptr_in_v3float %in_color %int_0
      %color  = OpLoad %v3float %ptr_col_in

; Compute offsets
     %y_plus  = OpFAdd %float %y %float_h
    %y_minus  = OpFSub %float %y %float_h
     %x_plus  = OpFAdd %float %x %float_h
    %x_minus  = OpFSub %float %x %float_h

; Emit vertex 1: Top (x, y+h, z, w)
  %pos_top = OpCompositeConstruct %v4float %x %y_plus %z %w
  %ptr_out_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %ptr_out_pos %pos_top
               OpStore %out_color %color
               OpEmitVertex

; Emit vertex 2: Right (x+h, y, z, w)
  %pos_right = OpCompositeConstruct %v4float %x_plus %y %z %w
               OpStore %ptr_out_pos %pos_right
               OpStore %out_color %color
               OpEmitVertex

; Emit vertex 3: Left (x-h, y, z, w)
  %pos_left = OpCompositeConstruct %v4float %x_minus %y %z %w
               OpStore %ptr_out_pos %pos_left
               OpStore %out_color %color
               OpEmitVertex

; Emit vertex 4: Bottom (x, y-h, z, w)
  %pos_bottom = OpCompositeConstruct %v4float %x %y_minus %z %w
               OpStore %ptr_out_pos %pos_bottom
               OpStore %out_color %color
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
