OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 8
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %in_worldPos Location 0
               OpDecorate %in_normal Location 1
               OpDecorate %in_color Location 2
               OpDecorate %out_color Location 0

; Types
      %void = OpTypeVoid
     %float = OpTypeFloat 32
       %int = OpTypeInt 32 1
      %uint = OpTypeInt 32 0
   %v2float = OpTypeVector %float 2
   %v3float = OpTypeVector %float 3
   %v4float = OpTypeVector %float 4

%gl_PerVertex_in  = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

    %uint_4 = OpConstant %uint 4
%arr_pv_in  = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_4   = OpTypeArray %v3float %uint_4

%ptr_in_arr_pv   = OpTypePointer Input %arr_pv_in
%ptr_in_v4float  = OpTypePointer Input %v4float
%ptr_in_arr_v3   = OpTypePointer Input %arr_v3_4
%ptr_in_v3float  = OpTypePointer Input %v3float
%ptr_out_pv      = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float

     %gl_in = OpVariable %ptr_in_arr_pv Input
    %gl_out = OpVariable %ptr_out_pv Output
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
  %in_normal = OpVariable %ptr_in_arr_v3 Input
   %in_color = OpVariable %ptr_in_arr_v3 Input
  %out_color = OpVariable %ptr_out_v3float Output

     %int_0 = OpConstant %int 0
     %int_1 = OpConstant %int 1
     %int_2 = OpConstant %int 2
     %int_3 = OpConstant %int 3
   %float_0 = OpConstant %float 0
   %float_1 = OpConstant %float 1
%float_third  = OpConstant %float 0.33333334
%float_2third = OpConstant %float 0.66666667
%float_thick  = OpConstant %float 0.02

   %fn_void = OpTypeFunction %void

      %main = OpFunction %void None %fn_void
%main_entry = OpLabel

; Load positions from gl_in
    %ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %ptr_p0
    %ptr_p1 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %ptr_p1
    %ptr_p2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %ptr_p2
    %ptr_p3 = OpAccessChain %ptr_in_v4float %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %ptr_p3

; Load colors at vertices 1 and 2
    %ptr_c1 = OpAccessChain %ptr_in_v3float %in_color %int_1
         %c1 = OpLoad %v3float %ptr_c1
    %ptr_c2 = OpAccessChain %ptr_in_v3float %in_color %int_2
         %c2 = OpLoad %v3float %ptr_c2

; Extract XY from positions for tangent computation
     %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
     %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
     %p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
     %p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1

; Adjacency tangents: central differences using adjacent vertices
; Tangent at start (vertex 1): normalize(p2.xy - p0.xy)
%tstart_raw = OpFSub %v2float %p2_xy %p0_xy
    %tstart = OpExtInst %v2float %1 69 %tstart_raw

; Tangent at end (vertex 2): normalize(p3.xy - p1.xy)
 %tend_raw = OpFSub %v2float %p3_xy %p1_xy
     %tend = OpExtInst %v2float %1 69 %tend_raw

; Perpendicular at start: (-tstart.y, tstart.x)
      %ts_x = OpCompositeExtract %float %tstart 0
      %ts_y = OpCompositeExtract %float %tstart 1
  %neg_ts_y = OpFNegate %float %ts_y
   %perp_s  = OpCompositeConstruct %v2float %neg_ts_y %ts_x

; Perpendicular at end: (-tend.y, tend.x)
      %te_x = OpCompositeExtract %float %tend 0
      %te_y = OpCompositeExtract %float %tend 1
  %neg_te_y = OpFNegate %float %te_y
   %perp_e  = OpCompositeConstruct %v2float %neg_te_y %te_x

; Segment vector p1->p2
       %seg = OpFSub %v4float %p2 %p1

; Points along segment divided into thirds
; B = p1 + seg*(1/3)
    %seg_t1 = OpVectorTimesScalar %v4float %seg %float_third
        %pB = OpFAdd %v4float %p1 %seg_t1
; C = p1 + seg*(2/3)
    %seg_t2 = OpVectorTimesScalar %v4float %seg %float_2third
        %pC = OpFAdd %v4float %p1 %seg_t2

; Interpolate perpendiculars along segment
%perp_diff  = OpFSub %v2float %perp_e %perp_s
%perp_dt1   = OpVectorTimesScalar %v2float %perp_diff %float_third
%perp_dt2   = OpVectorTimesScalar %v2float %perp_diff %float_2third
    %perp_B = OpFAdd %v2float %perp_s %perp_dt1
    %perp_C = OpFAdd %v2float %perp_s %perp_dt2

; Interpolate colors along segment
  %col_diff = OpFSub %v3float %c2 %c1
  %col_dt1  = OpVectorTimesScalar %v3float %col_diff %float_third
  %col_dt2  = OpVectorTimesScalar %v3float %col_diff %float_2third
    %col_B  = OpFAdd %v3float %c1 %col_dt1
    %col_C  = OpFAdd %v3float %c1 %col_dt2

; Build 4D offset vectors (scaled perpendiculars * thickness)
; Offset at A (start = p1): perp_s * thickness
    %oAx = OpFMul %float %neg_ts_y %float_thick
    %oAy = OpFMul %float %ts_x %float_thick
  %off_A = OpCompositeConstruct %v4float %oAx %oAy %float_0 %float_0

; Offset at B: perp_B * thickness
   %pBx = OpCompositeExtract %float %perp_B 0
   %pBy = OpCompositeExtract %float %perp_B 1
   %oBx = OpFMul %float %pBx %float_thick
   %oBy = OpFMul %float %pBy %float_thick
 %off_B = OpCompositeConstruct %v4float %oBx %oBy %float_0 %float_0

; Offset at C: perp_C * thickness
   %pCx = OpCompositeExtract %float %perp_C 0
   %pCy = OpCompositeExtract %float %perp_C 1
   %oCx = OpFMul %float %pCx %float_thick
   %oCy = OpFMul %float %pCy %float_thick
 %off_C = OpCompositeConstruct %v4float %oCx %oCy %float_0 %float_0

; Offset at D (end = p2): perp_e * thickness
    %oDx = OpFMul %float %neg_te_y %float_thick
    %oDy = OpFMul %float %te_x %float_thick
  %off_D = OpCompositeConstruct %v4float %oDx %oDy %float_0 %float_0

; Quad 1 corner positions: A to B (segment's first third)
 %vA_neg = OpFSub %v4float %p1 %off_A
 %vA_pos = OpFAdd %v4float %p1 %off_A
 %vB_neg = OpFSub %v4float %pB %off_B
 %vB_pos = OpFAdd %v4float %pB %off_B

; Quad 2 corner positions: C to D (segment's third third)
 %vC_neg = OpFSub %v4float %pC %off_C
 %vC_pos = OpFAdd %v4float %pC %off_C
 %vD_neg = OpFSub %v4float %p2 %off_D
 %vD_pos = OpFAdd %v4float %p2 %off_D

; Output position pointer (reused for all vertex stores)
 %pos_out = OpAccessChain %ptr_out_v4float %gl_out %int_0

; --- Emit quad 1: 4 vertices as triangle strip ---
; Vertex 1: A minus (p1 - perp_s*thickness)
               OpStore %pos_out %vA_neg
               OpStore %out_color %c1
               OpEmitVertex

; Vertex 2: A plus (p1 + perp_s*thickness)
               OpStore %pos_out %vA_pos
               OpStore %out_color %c1
               OpEmitVertex

; Vertex 3: B minus
               OpStore %pos_out %vB_neg
               OpStore %out_color %col_B
               OpEmitVertex

; Vertex 4: B plus
               OpStore %pos_out %vB_pos
               OpStore %out_color %col_B
               OpEmitVertex

               OpEndPrimitive

; --- Emit quad 2: 4 vertices as triangle strip (skip middle third = dash gap) ---
; Vertex 5: C minus
               OpStore %pos_out %vC_neg
               OpStore %out_color %col_C
               OpEmitVertex

; Vertex 6: C plus
               OpStore %pos_out %vC_pos
               OpStore %out_color %col_C
               OpEmitVertex

; Vertex 7: D minus (p2 - perp_e*thickness)
               OpStore %pos_out %vD_neg
               OpStore %out_color %c2
               OpEmitVertex

; Vertex 8: D plus (p2 + perp_e*thickness)
               OpStore %pos_out %vD_pos
               OpStore %out_color %c2
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
