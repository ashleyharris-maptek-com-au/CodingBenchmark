; SPIR-V geometry shader: expand lines into thick ribbons
OpCapability Geometry
%glsl450      = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_color %out_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input color array location decoration
OpDecorate %in_color Location 2

; Output color location decoration
OpDecorate %out_color Location 0

; Types
%void         = OpTypeVoid
%float        = OpTypeFloat 32
%int          = OpTypeInt 32 1
%uint         = OpTypeInt 32 0
%v2float      = OpTypeVector %float 2
%v3float      = OpTypeVector %float 3
%v4float      = OpTypeVector %float 4

; Constants
%int_0            = OpConstant %int 0
%int_1            = OpConstant %int 1
%uint_2           = OpConstant %uint 2
%float_0          = OpConstant %float 0.0
%float_1          = OpConstant %float 1.0
%float_thickness  = OpConstant %float 0.03

; Input gl_PerVertex struct and array
%gl_PerVertex_in   = OpTypeStruct %v4float
%arr_pv_in         = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv     = OpTypePointer Input %arr_pv_in
%ptr_in_v4float    = OpTypePointer Input %v4float
%gl_in             = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex struct
%gl_PerVertex_out  = OpTypeStruct %v4float
%ptr_out_pv        = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float   = OpTypePointer Output %v4float
%gl_out            = OpVariable %ptr_out_pv Output

; User-defined input color array (location 2)
%arr_v3float_2     = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3     = OpTypePointer Input %arr_v3float_2
%ptr_in_v3float    = OpTypePointer Input %v3float
%in_color          = OpVariable %ptr_in_arr_v3 Input

; User-defined output color (location 0)
%ptr_out_v3float   = OpTypePointer Output %v3float
%out_color         = OpVariable %ptr_out_v3float Output

; Main function
%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0 = gl_in[0].gl_Position
%ptr_p0  = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0      = OpLoad %v4float %ptr_p0

; Load p1 = gl_in[1].gl_Position
%ptr_p1  = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1      = OpLoad %v4float %ptr_p1

; Load c0 = color[0], c1 = color[1]
%ptr_c0  = OpAccessChain %ptr_in_v3float %in_color %int_0
%c0      = OpLoad %v3float %ptr_c0
%ptr_c1  = OpAccessChain %ptr_in_v3float %in_color %int_1
%c1      = OpLoad %v3float %ptr_c1

; Extract components
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3

; dir = p1 - p0 (xy)
%dirx = OpFSub %float %p1x %p0x
%diry = OpFSub %float %p1y %p0y

; perp2 = (-diry, dirx)
%neg_diry = OpFNegate %float %diry
%perp2    = OpCompositeConstruct %v2float %neg_diry %dirx

; normalize perp2
%perp2_n = OpExtInst %v2float %glsl450 Normalize %perp2

; perp * thickness
%pnx  = OpCompositeExtract %float %perp2_n 0
%pny  = OpCompositeExtract %float %perp2_n 1
%perpx = OpFMul %float %pnx %float_thickness
%perpy = OpFMul %float %pny %float_thickness

; Vertex 0: p0 + perp (color c0)
%v0x   = OpFAdd %float %p0x %perpx
%v0y   = OpFAdd %float %p0y %perpy
%v0pos = OpCompositeConstruct %v4float %v0x %v0y %p0z %p0w
%out_v0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
          OpStore %out_v0 %v0pos
          OpStore %out_color %c0
          OpEmitVertex

; Vertex 1: p0 - perp (color c0)
%v1x   = OpFSub %float %p0x %perpx
%v1y   = OpFSub %float %p0y %perpy
%v1pos = OpCompositeConstruct %v4float %v1x %v1y %p0z %p0w
%out_v1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
          OpStore %out_v1 %v1pos
          OpStore %out_color %c0
          OpEmitVertex

; Vertex 2: p1 + perp (color c1)
%v2x   = OpFAdd %float %p1x %perpx
%v2y   = OpFAdd %float %p1y %perpy
%v2pos = OpCompositeConstruct %v4float %v2x %v2y %p1z %p1w
%out_v2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
          OpStore %out_v2 %v2pos
          OpStore %out_color %c1
          OpEmitVertex

; Vertex 3: p1 - perp (color c1)
%v3x   = OpFSub %float %p1x %perpx
%v3y   = OpFSub %float %p1y %perpy
%v3pos = OpCompositeConstruct %v4float %v3x %v3y %p1z %p1w
%out_v3 = OpAccessChain %ptr_out_v4float %gl_out %int_0
          OpStore %out_v3 %v3pos
          OpStore %out_color %c1
          OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
