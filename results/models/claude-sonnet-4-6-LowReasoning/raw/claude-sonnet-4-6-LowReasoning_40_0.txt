#include <bits/stdc++.h>
using namespace std;

static int n, m;
static vector<long long> c;
static vector<vector<int>> adj;
static vector<bool> adjmat; // n*n flat adjacency matrix

static long long best_val;
static vector<int> best_x;
static vector<int> cur_x;

inline bool isAdj(int u, int v) { return adjmat[u * n + v]; }

// Greedy: pick highest-weight available vertex repeatedly
void greedy_init() {
    vector<bool> blocked(n, false);
    vector<int> x(n, 0);
    long long val = 0;
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [](int a, int b){ return c[a] > c[b]; });
    for (int v : order) {
        if (c[v] <= 0) break;
        if (!blocked[v]) {
            x[v] = 1;
            val += c[v];
            for (int u : adj[v]) blocked[u] = true;
        }
    }
    if (val > best_val) { best_val = val; best_x = x; }
}

void solve(vector<int>& cands, int start, long long cur) {
    // Upper bound: cur + sum of all positive weights in cands[start..]
    long long ub = cur;
    for (int i = start; i < (int)cands.size(); i++)
        if (c[cands[i]] > 0) ub += c[cands[i]];
    if (ub <= best_val) return;

    if (start == (int)cands.size()) {
        if (cur > best_val) { best_val = cur; best_x = cur_x; }
        return;
    }

    int v = cands[start];

    // Branch 1: include v
    // Find new candidate list: cands[start+1..] minus neighbors of v
    vector<int> new_cands;
    new_cands.reserve(cands.size() - start - 1);
    for (int i = start + 1; i < (int)cands.size(); i++) {
        if (!isAdj(v, cands[i])) new_cands.push_back(cands[i]);
    }
    cur_x[v] = 1;
    int save_start = 0;
    solve(new_cands, 0, cur + c[v]);
    cur_x[v] = 0;

    // Branch 2: exclude v
    solve(cands, start + 1, cur);
    (void)save_start;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> n >> m;
    c.resize(n);
    for (int i = 0; i < n; i++) cin >> c[i];

    adj.resize(n);
    adjmat.assign(n * n, false);
    for (int i = 0; i < m; i++) {
        int u, v; cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
        adjmat[u * n + v] = true;
        adjmat[v * n + u] = true;
    }

    best_val = 0;
    best_x.assign(n, 0);
    cur_x.assign(n, 0);

    // Greedy initial solution
    greedy_init();

    // Build candidate list: only positive-weight vertices, sorted by weight desc
    vector<int> cands;
    for (int i = 0; i < n; i++) if (c[i] > 0) cands.push_back(i);
    sort(cands.begin(), cands.end(), [](int a, int b){ return c[a] > c[b]; });

    solve(cands, 0, 0);

    cout << best_val << "\n";
    for (int i = 0; i < n; i++) {
        if (i) cout << " ";
        cout << best_x[i];
    }
    cout << "\n";
    return 0;
}
