using System;
using System.Collections.Generic;
using System.IO;
using System.Text;
using System.Diagnostics;

class Program
{
    static int NX, NY, NZ;
    static int[] surfZ;
    static byte[] surfByte;
    static byte[] fullGrid;
    static bool hasGrid;
    static float[] baseDist;
    static byte[] pathDir;
    static readonly int[] DX = { 1, -1, 0, 0 };
    static readonly int[] DY = { 0, 0, 1, -1 };

    static void Main()
    {
        var sw = Stopwatch.StartNew();
        var hdr = Console.ReadLine().Trim().Split(' ');
        NX = int.Parse(hdr[0]);
        NY = int.Parse(hdr[1]);
        NZ = int.Parse(hdr[2]);

        long total = (long)NX * NY * NZ;
        int cells = NX * NY;

        surfZ = new int[cells];
        surfByte = new byte[cells];
        for (int i = 0; i < cells; i++) surfZ[i] = -1;

        var stdinStream = Console.OpenStandardInput();
        // Load full grid if it fits comfortably in memory, otherwise stream
        hasGrid = total <= 400_000_000L;

        if (hasGrid)
        {
            fullGrid = new byte[(int)total];
            int totalInt = (int)total, read = 0;
            while (read < totalInt)
            {
                int r = stdinStream.Read(fullGrid, read, Math.Min(totalInt - read, 65536));
                if (r <= 0) break;
                read += r;
            }
            // Compute surface heights (top solid voxel per column)
            for (int x = 0; x < NX; x++)
                for (int y = 0; y < NY; y++)
                {
                    int ci = x * NY + y;
                    int bo = (int)((long)ci * NZ);
                    for (int z = NZ - 1; z >= 0; z--)
                        if ((fullGrid[bo + z] & 0xF) > 0)
                        { surfZ[ci] = z; surfByte[ci] = fullGrid[bo + z]; break; }
                }
        }
        else
        {
            // Stream column by column (x,y order in input)
            var col = new byte[NZ];
            for (int x = 0; x < NX; x++)
                for (int y = 0; y < NY; y++)
                {
                    int r = 0;
                    while (r < NZ) { int n = stdinStream.Read(col, r, NZ - r); if (n <= 0) break; r += n; }
                    int ci = x * NY + y;
                    for (int z = NZ - 1; z >= 0; z--)
                        if ((col[z] & 0xF) > 0) { surfZ[ci] = z; surfByte[ci] = col[z]; break; }
                }
        }

        // Ensure (0,0) has ground for the processing plant
        if (surfZ[0] < 0) surfZ[0] = 0;

        // Dijkstra from (0,0): compute minimum transport cost per unit mass
        // for each cell to deliver to the processing plant at (0,0).
        //
        // Edge semantics: when Dijkstra expands from node (x,y) to neighbor (nx,ny),
        // it represents a truck traveling from (nx,ny) toward (x,y) toward (0,0).
        // Slope for truck = surfZ[x,y] - surfZ[nx,ny] (positive = uphill).
        // Constraint: uphill slope max 1 Z per X/Y step.
        // Cost: 1 per step flat/downhill, 5 per step uphill.
        baseDist = new float[cells];
        pathDir = new byte[cells]; // direction from each cell toward (0,0), 1-indexed (0=unvisited)
        for (int i = 0; i < cells; i++) baseDist[i] = float.MaxValue;
        baseDist[0] = 0f;
        var pq = new PriorityQueue<int, float>();
        pq.Enqueue(0, 0f);

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int idx, out float d);
            if (d > baseDist[idx] + 1e-5f) continue;
            int x = idx / NY, y = idx % NY;
            int sz = surfZ[idx];
            if (sz < 0) continue;
            for (int dir = 0; dir < 4; dir++)
            {
                int nx = x + DX[dir], ny = y + DY[dir];
                if ((uint)nx >= (uint)NX || (uint)ny >= (uint)NY) continue;
                int nidx = nx * NY + ny;
                int nsz = surfZ[nidx];
                if (nsz < 0) continue;
                // Truck going from (nx,ny) to (x,y): height gain = sz - nsz
                int hd = sz - nsz;
                if (hd > 1) continue; // too steep uphill
                float cost = hd > 0 ? 5f : 1f;
                float nd = d + cost;
                if (nd < baseDist[nidx])
                {
                    baseDist[nidx] = nd;
                    // dir is from (x,y) to (nx,ny); reverse (dir^1) is from (nx,ny) toward (x,y)
                    pathDir[nidx] = (byte)((dir ^ 1) + 1);
                    pq.Enqueue(nidx, nd);
                }
            }
        }

        // Build candidate priority queue: process most profitable voxels first
        // profit = revenue - transport_cost = mass*value*1000 - baseDist*mass
        var cq = new PriorityQueue<int, double>(); // min-heap on (cost-rev) i.e. negative profit
        bool[] inCQ = new bool[cells];
        for (int i = 0; i < cells; i++)
        {
            if (baseDist[i] >= float.MaxValue / 2) continue;
            byte b = surfByte[i];
            int mass = b & 0xF, val = b >> 4;
            if (val <= 0 || mass <= 0) continue;
            double rev = (double)mass * val * 1000.0;
            double cst = (double)baseDist[i] * mass;
            if (rev > cst) { cq.Enqueue(i, cst - rev); inCQ[i] = true; }
        }

        var opList = new List<string>();
        var pb = new List<(int, int)>(Math.Max(NX, NY) * 4);
        int pathLimit = 4 * (NX + NY) + 32;

        while (cq.Count > 0)
        {
            if (sw.Elapsed.TotalSeconds > 250) break;

            cq.TryDequeue(out int cidx, out _);
            inCQ[cidx] = false;

            int cz = surfZ[cidx];
            if (cz < 0) continue;
            byte cb = surfByte[cidx];
            int mass = cb & 0xF, val = cb >> 4;
            if (val <= 0 || mass <= 0) continue;
            if (baseDist[cidx] >= float.MaxValue / 2) continue;
            double rev = (double)mass * val * 1000.0;
            double cst = (double)baseDist[cidx] * mass;
            if (rev <= cst) continue;

            // Reconstruct path from (cx,cy) to (0,0) by following pathDir pointers
            pb.Clear();
            bool ok = true;
            int px = cidx / NY, py = cidx % NY;
            int lim = pathLimit;
            while ((px != 0 || py != 0) && lim-- > 0)
            {
                int pidx = px * NY + py;
                int d = (pathDir[pidx] & 0xFF) - 1;
                if (d < 0) { ok = false; break; }
                pb.Add((DX[d], DY[d]));
                px += DX[d]; py += DY[d];
            }
            if (!ok || px != 0 || py != 0) continue;

            int cx = cidx / NY, cy = cidx % NY;
            var sb = new StringBuilder(pb.Count * 6 + 32);
            sb.Append(pb.Count).Append(' ').Append(cx).Append(' ')
              .Append(cy).Append(' ').Append(cz);
            foreach (var (sdx, sdy) in pb)
                sb.Append(' ').Append(sdx).Append(' ').Append(sdy);
            sb.Append(" 0 0");
            opList.Add(sb.ToString());

            // Update surface at dig site
            surfZ[cidx] = -1;
            surfByte[cidx] = 0;
            if (hasGrid)
            {
                int bo = (int)((long)cidx * NZ);
                for (int z = cz - 1; z >= 0; z--)
                    if ((fullGrid[bo + z] & 0xF) > 0)
                    { surfZ[cidx] = z; surfByte[cidx] = fullGrid[bo + z]; break; }
            }

            // If the newly exposed layer is profitable, queue it
            if (surfZ[cidx] >= 0 && !inCQ[cidx])
            {
                byte nb = surfByte[cidx];
                int nm = nb & 0xF, nv = nb >> 4;
                if (nv > 0 && nm > 0 && baseDist[cidx] < float.MaxValue / 2)
                {
                    double nr = (double)nm * nv * 1000.0;
                    double nc = (double)baseDist[cidx] * nm;
                    if (nr > nc) { cq.Enqueue(cidx, nc - nr); inCQ[cidx] = true; }
                }
            }
        }

        var output = new StringBuilder(opList.Count * 32 + 16);
        output.Append(opList.Count).Append('\n');
        foreach (var l in opList) output.Append(l).Append('\n');
        Console.Write(output);
    }
}
