import sys

def main():
    input_data = sys.stdin.read()
    lines = input_data.strip().split('\n')
    first = lines[0].split()
    U = int(first[0])
    S = int(first[1])

    sets = []
    for i in range(1, S + 1):
        if i < len(lines) and lines[i].strip():
            sets.append(list(map(int, lines[i].split())))
        else:
            sets.append([])

    covered = [False] * U
    chosen = []

    def backtrack():
        # Find first uncovered element
        target = -1
        for e in range(U):
            if not covered[e]:
                target = e
                break
        if target == -1:
            return True  # all covered

        for i in range(len(sets)):
            if target not in sets[i]:
                continue
            if any(covered[e] for e in sets[i]):
                continue
            for e in sets[i]:
                covered[e] = True
            chosen.append(i)
            if backtrack():
                return True
            chosen.pop()
            for e in sets[i]:
                covered[e] = False
        return False

    if backtrack():
        print("SOLUTION")
        print(*sorted(chosen))
    else:
        print("NO SOLUTION")

main()
