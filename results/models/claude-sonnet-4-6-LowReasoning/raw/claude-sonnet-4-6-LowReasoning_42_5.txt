; SPIR-V geometry shader: thick arrow lines
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %in_worldPos %in_normal %in_color %out_color
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; Decorations for input gl_PerVertex block
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

; Decorations for output gl_PerVertex block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

; Input user-defined arrays
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2

; Output
OpDecorate %out_color Location 0

; Types
%void       = OpTypeVoid
%float      = OpTypeFloat 32
%int        = OpTypeInt 32 1
%uint       = OpTypeInt 32 0
%v3float    = OpTypeVector %float 3
%v4float    = OpTypeVector %float 4
%fn_void    = OpTypeFunction %void

; Constants
%int_0      = OpConstant %int 0
%int_1      = OpConstant %int 1
%uint_2     = OpConstant %uint 2
%float_0    = OpConstant %float 0.0
%float_1    = OpConstant %float 1.0
%float_075  = OpConstant %float 0.75
%float_0015 = OpConstant %float 0.015
%float_005  = OpConstant %float 0.05

; Input gl_PerVertex block type (array of 2 for InputLines)
%gl_PerVertex_in  = OpTypeStruct %v4float
%arr_pv_in        = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv    = OpTypePointer Input %arr_pv_in
%ptr_in_v4float   = OpTypePointer Input %v4float
%gl_in            = OpVariable %ptr_in_arr_pv Input

; Output gl_PerVertex block type
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv       = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float  = OpTypePointer Output %v4float
%gl_out           = OpVariable %ptr_out_pv Output

; Input user arrays (array of 2)
%arr_v3_2         = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3    = OpTypePointer Input %arr_v3_2
%ptr_in_v3float   = OpTypePointer Input %v3float
%in_worldPos      = OpVariable %ptr_in_arr_v3 Input
%in_normal        = OpVariable %ptr_in_arr_v3 Input
%in_color         = OpVariable %ptr_in_arr_v3 Input

; Output color
%ptr_out_v3float  = OpTypePointer Output %v3float
%out_color        = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0 from gl_in[0].gl_Position
%p0_ptr_v4  = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0_v4      = OpLoad %v4float %p0_ptr_v4
%p0         = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2

; Load p1 from gl_in[1].gl_Position
%p1_ptr_v4  = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1_v4      = OpLoad %v4float %p1_ptr_v4
%p1         = OpVectorShuffle %v3float %p1_v4 %p1_v4 0 1 2

; Load c0, c1
%c0_ptr     = OpAccessChain %ptr_in_v3float %in_color %int_0
%c0         = OpLoad %v3float %c0_ptr
%c1_ptr     = OpAccessChain %ptr_in_v3float %in_color %int_1
%c1         = OpLoad %v3float %c1_ptr

; dir = p1 - p0
%dir        = OpFSub %v3float %p1 %p0

; perp = normalize(vec3(-dir.y, dir.x, 0))
%dir_x      = OpCompositeExtract %float %dir 0
%dir_y      = OpCompositeExtract %float %dir 1
%neg_dir_y  = OpFNegate %float %dir_y
%perp_raw   = OpCompositeConstruct %v3float %neg_dir_y %dir_x %float_0
%perp       = OpExtInst %v3float %glsl450 Normalize %perp_raw

; mid = p0 + 0.75 * dir
%dir_075    = OpVectorTimesScalar %v3float %dir %float_075
%mid        = OpFAdd %v3float %p0 %dir_075

; shaft perp offset = perp * 0.015
%perp_s     = OpVectorTimesScalar %v3float %perp %float_0015
%p0_plus    = OpFAdd %v3float %p0 %perp_s
%p0_minus   = OpFSub %v3float %p0 %perp_s
%mid_plus   = OpFAdd %v3float %mid %perp_s
%mid_minus  = OpFSub %v3float %mid %perp_s

; arrow perp offset = perp * 0.05
%perp_a     = OpVectorTimesScalar %v3float %perp %float_005
%arr_plus   = OpFAdd %v3float %mid %perp_a
%arr_minus  = OpFSub %v3float %mid %perp_a

; --- Shaft strip: p0+perp, p0-perp, mid+perp, mid-perp ---

; Vertex: p0 + perp_shaft
%sp0px = OpCompositeExtract %float %p0_plus 0
%sp0py = OpCompositeExtract %float %p0_plus 1
%sp0pz = OpCompositeExtract %float %p0_plus 2
%sp0pv = OpCompositeConstruct %v4float %sp0px %sp0py %sp0pz %float_1
%sp0p_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %sp0p_gptr %sp0pv
OpStore %out_color %c0
OpEmitVertex

; Vertex: p0 - perp_shaft
%sp0mx = OpCompositeExtract %float %p0_minus 0
%sp0my = OpCompositeExtract %float %p0_minus 1
%sp0mz = OpCompositeExtract %float %p0_minus 2
%sp0mv = OpCompositeConstruct %v4float %sp0mx %sp0my %sp0mz %float_1
%sp0m_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %sp0m_gptr %sp0mv
OpStore %out_color %c0
OpEmitVertex

; Vertex: mid + perp_shaft
%smpx = OpCompositeExtract %float %mid_plus 0
%smpy = OpCompositeExtract %float %mid_plus 1
%smpz = OpCompositeExtract %float %mid_plus 2
%smpv = OpCompositeConstruct %v4float %smpx %smpy %smpz %float_1
%smp_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %smp_gptr %smpv
OpStore %out_color %c0
OpEmitVertex

; Vertex: mid - perp_shaft
%smmx = OpCompositeExtract %float %mid_minus 0
%smmy = OpCompositeExtract %float %mid_minus 1
%smmz = OpCompositeExtract %float %mid_minus 2
%smmv = OpCompositeConstruct %v4float %smmx %smmy %smmz %float_1
%smm_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %smm_gptr %smmv
OpStore %out_color %c0
OpEmitVertex

OpEndPrimitive

; --- Arrowhead strip: mid+perp_arrow, mid-perp_arrow, p1 ---

; Vertex: mid + perp_arrow
%apx = OpCompositeExtract %float %arr_plus 0
%apy = OpCompositeExtract %float %arr_plus 1
%apz = OpCompositeExtract %float %arr_plus 2
%apv = OpCompositeConstruct %v4float %apx %apy %apz %float_1
%ap_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %ap_gptr %apv
OpStore %out_color %c1
OpEmitVertex

; Vertex: mid - perp_arrow
%amx = OpCompositeExtract %float %arr_minus 0
%amy = OpCompositeExtract %float %arr_minus 1
%amz = OpCompositeExtract %float %arr_minus 2
%amv = OpCompositeConstruct %v4float %amx %amy %amz %float_1
%am_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %am_gptr %amv
OpStore %out_color %c1
OpEmitVertex

; Vertex: p1 (tip)
%tipx = OpCompositeExtract %float %p1 0
%tipy = OpCompositeExtract %float %p1 1
%tipz = OpCompositeExtract %float %p1 2
%tipv = OpCompositeConstruct %v4float %tipx %tipy %tipz %float_1
%tip_gptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %tip_gptr %tipv
OpStore %out_color %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
