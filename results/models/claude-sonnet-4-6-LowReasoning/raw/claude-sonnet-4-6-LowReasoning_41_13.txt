%bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1

%main = OpFunction %void None %voidFn
%entry = OpLabel

; t = clamp(biased, 0.0, 1.0)

; mix(bottom, top, t) = bottom + t * (top - bottom)

; Extract RGB and construct output with alpha=1.0

%g = OpCompositeExtract %float %mixed 1
