#include <iostream>
#include <vector>
#include <numeric>

// Function to check if the current color assignment is safe for vertex v
bool isSafe(int v, int num_vertices, const std::vector<std::vector<int>>& graph, const std::vector<int>& color, int c) {
    for (int i = 0; i < num_vertices; ++i) {
        if (graph[v][i] && c == color[i]) {
            return false;
        }
    }
    return true;
}

// A recursive utility function to solve the graph coloring problem
bool graphColoringUtil(int num_vertices, const std::vector<std::vector<int>>& graph, int m, std::vector<int>& color, int v) {
    // Base case: If all vertices are assigned a color
    if (v == num_vertices) {
        return true;
    }

    // Try different colors for vertex v
    for (int c = 1; c <= m; ++c) {
        if (isSafe(v, num_vertices, graph, color, c)) {
            color[v] = c;

            // Recur for the next vertex
            if (graphColoringUtil(num_vertices, graph, m, color, v + 1)) {
                return true;
            }

            // If assigning color c doesn't lead to a solution, then backtrack
            color[v] = 0; // Unassign color
        }
    }

    // If no color can be assigned to this vertex
    return false;
}

// Main function to solve the graph coloring problem
bool graphColoring(int num_vertices, const std::vector<std::vector<int>>& graph, int m) {
    std::vector<int> color(num_vertices, 0); // Initialize all colors as 0 (no color)

    // Call the utility function for vertex 0
    if (!graphColoringUtil(num_vertices, graph, m, color, 0)) {
        std::cout << "Solution does not exist with " << m << " colors." << std::endl;
        return false;
    }

    // Print the solution
    std::cout << "Solution exists with " << m << " colors. Assigned colors are:" << std::endl;
    for (int i = 0; i < num_vertices; ++i) {
        std::cout << "Vertex " << i << ": Color " << color[i] << std::endl;
    }
    return true;
}

int main() {
    // Example Graph (4 vertices, 3 colors)
    // (0)--(1)
    // |   / |
    // |  /  |
    // | /   |
    // (3)--(2)

    int num_vertices = 4;
    std::vector<std::vector<int>> graph = {
        {0, 1, 0, 1},
        {1, 0, 1, 1},
        {0, 1, 0, 1},
        {1, 1, 1, 0},
    };
    int m = 3; // Number of colors

    graphColoring(num_vertices, graph, m);

    std::cout << "
Attempting with 2 colors for the same graph (should fail):" << std::endl;
    m = 2; // Try with 2 colors
    graphColoring(num_vertices, graph, m);


    // Another example graph (3 vertices, 2 colors)
    // (0)--(1)
    //  |  /
    //  | /
    //  (2)

    std::cout << "
Another graph example (3 vertices, 2 colors):" << std::endl;
    num_vertices = 3;
    std::vector<std::vector<int>> graph2 = {
        {0, 1, 1},
        {1, 0, 1},
        {1, 1, 0},
    };
    m = 2; // Number of colors
    graphColoring(num_vertices, graph2, m);

    std::cout << "
Another graph example (3 vertices, 3 colors):" << std::endl;
    num_vertices = 3;
    std::vector<std::vector<int>> graph3 = {
        {0, 1, 1},
        {1, 0, 1},
        {1, 1, 0},
    };
    m = 3; // Number of colors
    graphColoring(num_vertices, graph3, m);

    return 0;
}
