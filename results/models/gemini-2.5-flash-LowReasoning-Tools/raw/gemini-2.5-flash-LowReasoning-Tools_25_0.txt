using System;

namespace AsteroidIntercept
{
    class Program
    {
        // Astronomical constants (approximate values)
        // Masses in kilograms (kg)
        const double EARTH_MASS = 5.972e24; 
        const double ASTEROID_MASS = 1e12; // Example asteroid mass, significantly smaller than Earth
        const double PROBE_MASS = 1000;    // Example probe mass

        // Distances in meters (m)
        const double AU = 1.496e11; // Astronomical Unit (distance from Earth to Sun) - using as a general scale
        const double EARTH_ORBIT_RADIUS = 1.0 * AU; 

        // Time step for simulation in seconds (s)
        const double TIME_STEP = 3600; // 1 hour

        // Interception radius for the probe
        const double INTERCEPTION_RADIUS = 1000; // 1 km radius for interception

        static void Main(string[] args)
        {
            Console.WriteLine("Asteroid Interception Trajectory Calculation");
            Console.WriteLine("------------------------------------------");

            // Initial conditions
            // Earth: Position at (EARTH_ORBIT_RADIUS, 0, 0), moving in +Y direction (tangential velocity)
            // Orbital speed calculation for Earth (approximation, assuming circular orbit around a massive sun)
            // v = sqrt(G * M_sun / r)
            // For simplicity, let's assume Earth is at (EARTH_ORBIT_RADIUS, 0, 0) and moves in a perfect circle around origin (Sun).
            // We'll ignore the Sun's explicit gravity in this simplified 3-body system (Earth, Asteroid, Probe)
            // and instead give Earth an initial velocity that would keep it in orbit around the origin.
            // If the goal is just Asteroid-Earth-Probe, then Earth's velocity needs to reflect its motion relative to the Asteroid.
            // For this simulation, we'll set Earth's velocity to zero relative to the origin for simplicity of asteroid-probe relative motion.
            // This is a major simplification, but for a basic interception, it allows focus on asteroid and probe.
            // In a more complex simulation, the Sun would be another CelestialBody.

            // Let's place Earth at some distance, and give it an initial velocity
            Vector3D earthPosition = new Vector3D(0, EARTH_ORBIT_RADIUS, 0);
            Vector3D earthVelocity = new Vector3D(0, 0, 0); // Simplified: Earth is stationary for relative motion demo

            // Asteroid: Initially approaching Earth
            // Let's place the asteroid on a collision course, some distance away
            Vector3D asteroidPosition = new Vector3D(0, EARTH_ORBIT_RADIUS * 1.5, 0); // Further out than Earth
            Vector3D asteroidVelocity = new Vector3D(0, -1000, 0); // Moving towards Earth at 1 km/s

            // Probe: Launched from Earth, aiming for asteroid
            // Initial position same as Earth, initial velocity to intercept
            Vector3D probePosition = new Vector3D(0, EARTH_ORBIT_RADIUS, 0);
            // This is the tricky part: calculating the initial probe velocity for interception.
            // For a first pass, let's give it a velocity that *might* intercept.
            // A more advanced solution would involve Lambert's problem or optimization.
            Vector3D probeVelocity = new Vector3D(0, 5000, 0); // Example: 5 km/s towards asteroid

            CelestialBody earth = new CelestialBody(EARTH_MASS, earthPosition, earthVelocity);
            CelestialBody asteroid = new CelestialBody(ASTEROID_MASS, asteroidPosition, asteroidVelocity);
            CelestialBody probe = new CelestialBody(PROBE_MASS, probePosition, probeVelocity);

            Simulation simulation = new Simulation(asteroid, earth, probe);

            double currentTime = 0;
            long simulationSteps = 0;
            bool intercepted = false;

            // Run simulation for a maximum of 1 year (approx 365 days)
            double maxSimulationTime = 365 * 24 * 3600; // seconds

            Console.WriteLine($"Initial Earth Position: ({earth.Position.X:E}, {earth.Position.Y:E}, {earth.Position.Z:E}) m");
            Console.WriteLine($"Initial Asteroid Position: ({asteroid.Position.X:E}, {asteroid.Position.Y:E}, {asteroid.Position.Z:E}) m");
            Console.WriteLine($"Initial Probe Position: ({probe.Position.X:E}, {probe.Position.Y:E}, {probe.Position.Z:E}) m");
            Console.WriteLine($"Probe initial velocity: ({probe.Velocity.X:E}, {probe.Velocity.Y:E}, {probe.Velocity.Z:E}) m/s");
            Console.WriteLine($"Time Step: {TIME_STEP} seconds");
            Console.WriteLine($"Interception Radius: {INTERCEPTION_RADIUS} meters");


            while (currentTime < maxSimulationTime)
            {
                simulation.Step(TIME_STEP);
                simulationSteps++;
                currentTime += TIME_STEP;

                if (simulationSteps % (3600 / TIME_STEP * 24) == 0) // Log every simulated day
                {
                    Console.WriteLine($"Time: {currentTime / 3600 / 24:F2} days");
                    // Console.WriteLine($"  Asteroid: P=({asteroid.Position.X:E}, {asteroid.Position.Y:E}, {asteroid.Position.Z:E}), V=({asteroid.Velocity.X:E}, {asteroid.Velocity.Y:E}, {asteroid.Velocity.Z:E})");
                    // Console.WriteLine($"  Probe: P=({probe.Position.X:E}, {probe.Position.Y:E}, {probe.Position.Z:E}), V=({probe.Velocity.X:E}, {probe.Velocity.Y:E}, {probe.Velocity.Z:E})");
                }

                if (simulation.InterceptDetected(INTERCEPTION_RADIUS))
                {
                    intercepted = true;
                    Console.WriteLine("
--- INTERCEPTION DETECTED! ---");
                    Console.WriteLine($"Time of interception: {currentTime / 3600 / 24:F2} days ({currentTime:E} seconds)");
                    Console.WriteLine($"Asteroid Position at interception: ({asteroid.Position.X:E}, {asteroid.Position.Y:E}, {asteroid.Position.Z:E}) m");
                    Console.WriteLine($"Probe Position at interception: ({probe.Position.X:E}, {probe.Position.Y:E}, {probe.Position.Z:E}) m");
                    Console.WriteLine($"Distance between Probe and Asteroid: {(probe.Position - asteroid.Position).Magnitude():E} m");
                    break;
                }
            }

            if (!intercepted)
            {
                Console.WriteLine("
--- Simulation finished. No interception detected. ---");
                Console.WriteLine($"Final Earth Position: ({earth.Position.X:E}, {earth.Position.Y:E}, {earth.Position.Z:E}) m");
                Console.WriteLine($"Final Asteroid Position: ({asteroid.Position.X:E}, {asteroid.Position.Y:E}, {asteroid.Position.Z:E}) m");
                Console.WriteLine($"Final Probe Position: ({probe.Position.X:E}, {probe.Position.Y:E}, {probe.Position.Z:E}) m");
                Console.WriteLine($"Final distance between Probe and Asteroid: {(probe.Position - asteroid.Position).Magnitude():E} m");
            }
        }
    }
}
