using System;
using System.Collections.Generic;
using System.Linq;

public class Graph
{
    private bool[,] adjMatrix;
    public int NumVertices { get; private set; }

    public Graph(int numVertices)
    {
        NumVertices = numVertices;
        adjMatrix = new bool[numVertices, numVertices];
    }

    public void AddEdge(int u, int v)
    {
        if (u >= 0 && u < NumVertices && v >= 0 && v < NumVertices && u != v)
        {
            adjMatrix[u, v] = true;
            adjMatrix[v, u] = true;
        }
    }

    public bool HasEdge(int u, int v)
    {
        if (u >= 0 && u < NumVertices && v >= 0 && v < NumVertices)
        {
            return adjMatrix[u, v];
        }
        return false;
    }

    public List<int> GetNeighbors(int u)
    {
        List<int> neighbors = new List<int>();
        for (int v = 0; v < NumVertices; v++)
        {
            if (adjMatrix[u, v])
            {
                neighbors.Add(v);
            }
        }
        return neighbors;
    }
}

public class MaximumCliqueFinder
{
    private Graph graph;
    private List<int> maxClique;

    public MaximumCliqueFinder(Graph graph)
    {
        this.graph = graph;
        this.maxClique = new List<int>();
    }

    public List<int> FindMaximumClique()
    {
        List<int> R = new List<int>(); // Vertices in the current clique
        List<int> P = Enumerable.Range(0, graph.NumVertices).ToList(); // Candidate vertices
        List<int> X = new List<int>(); // Vertices already processed

        BronKerbosch(R, P, X);
        return maxClique;
    }

    private void BronKerbosch(List<int> R, List<int> P, List<int> X)
    {
        if (!P.Any() && !X.Any())
        {
            // R is a maximal clique
            if (R.Count > maxClique.Count)
            {
                maxClique = new List<int>(R);
            }
            return;
        }

        // Pivot selection: choose a pivot vertex u from P U X
        // The pivot selection strategy can significantly affect performance.
        // A common heuristic is to choose a vertex with the most neighbors in P.
        int u_pivot = -1;
        if (P.Any())
        {
            u_pivot = P.Union(X).OrderByDescending(v => graph.GetNeighbors(v).Intersect(P).Count()).First();
        }

        List<int> P_without_neighbors_of_u_pivot = P;
        if (u_pivot != -1)
        {
            P_without_neighbors_of_u_pivot = P.Except(graph.GetNeighbors(u_pivot)).ToList();
        }

        foreach (int v in P_without_neighbors_of_u_pivot.ToList()) // Use ToList() to avoid modification during iteration
        {
            List<int> N_v = graph.GetNeighbors(v);

            List<int> R_new = new List<int>(R) { v };
            List<int> P_new = P.Intersect(N_v).ToList();
            List<int> X_new = X.Intersect(N_v).ToList();

            BronKerbosch(R_new, P_new, X_new);

            P.Remove(v);
            X.Add(v);
        }
    }
}

public class Program
{
    public static void Main(string[] args)
    {
        // Example Usage:
        // Graph from CLRS Figure 34.12(b) example for MAX-CLIQUE
        // Vertices: 0, 1, 2, 3, 4
        // Edges: (0,1), (0,2), (1,2), (1,3), (2,3), (2,4), (3,4)
        Graph graph = new Graph(5);
        graph.AddEdge(0, 1);
        graph.AddEdge(0, 2);
        graph.AddEdge(1, 2);
        graph.AddEdge(1, 3);
        graph.AddEdge(2, 3);
        graph.AddEdge(2, 4);
        graph.AddEdge(3, 4);

        MaximumCliqueFinder finder = new MaximumCliqueFinder(graph);
        List<int> maxClique = finder.FindMaximumClique();

        Console.WriteLine("Maximum Clique: {" + string.Join(", ", maxClique) + "}");
        Console.WriteLine("Size of Maximum Clique: " + maxClique.Count); // Expected: 3 for {1,2,3} or {2,3,4}

        // Another example: complete graph K3
        Graph k3 = new Graph(3);
        k3.AddEdge(0, 1);
        k3.AddEdge(0, 2);
        k3.AddEdge(1, 2);
        MaximumCliqueFinder finderK3 = new MaximumCliqueFinder(k3);
        List<int> maxCliqueK3 = finderK3.FindMaximumClique();
        Console.WriteLine("
Maximum Clique for K3: {" + string.Join(", ", maxCliqueK3) + "}");
        Console.WriteLine("Size of Maximum Clique for K3: " + maxCliqueK3.Count); // Expected: 3

        // Another example: disconnected graph
        Graph disconnected = new Graph(4);
        disconnected.AddEdge(0, 1); // Component 1: {0,1}
        disconnected.AddEdge(2, 3); // Component 2: {2,3}
        MaximumCliqueFinder finderDisconnected = new MaximumCliqueFinder(disconnected);
        List<int> maxCliqueDisconnected = finderDisconnected.FindMaximumClique();
        Console.WriteLine("
Maximum Clique for disconnected graph: {" + string.Join(", ", maxCliqueDisconnected) + "}");
        Console.WriteLine("Size of Maximum Clique for disconnected graph: " + maxCliqueDisconnected.Count); // Expected: 2
    }
}
