#include <iostream>
#include <vector>
#include <cmath>
#include <limits>
#include <algorithm>
#include <numeric>

// Structure to represent a city
struct City {
    double x, y;
    int id;
};

// Function to calculate Euclidean distance between two cities
double calculateDistance(const City& c1, const City& c2) {
    return std::sqrt(std::pow(c1.x - c2.x, 2) + std::pow(c1.y - c2.y, 2));
}

// Nearest Neighbor algorithm to find an initial tour (with distMatrix)
std::vector<int> nearestNeighbor(int N, const std::vector<City>& cities, const std::vector<std::vector<double>>& distMatrix) {
    std::vector<int> tour;
    std::vector<bool> visited(N, false);

    // Start at city 0
    int currentCity = 0;
    tour.push_back(currentCity);
    visited[currentCity] = true;

    for (int i = 1; i < N; ++i) {
        double minDistance = std::numeric_limits<double>::max();
        int nextCity = -1;

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                if (distMatrix[currentCity][j] < minDistance) {
                    minDistance = distMatrix[currentCity][j];
                    nextCity = j;
                }
            }
        }
        currentCity = nextCity;
        tour.push_back(currentCity);
        visited[currentCity] = true;
    }
    return tour;
}

// Nearest Neighbor algorithm to find an initial tour (without distMatrix - calculates distances on the fly)
std::vector<int> nearestNeighbor(int N, const std::vector<City>& cities) {
    std::vector<int> tour;
    std::vector<bool> visited(N, false);

    // Start at city 0
    int currentCity = 0;
    tour.push_back(currentCity);
    visited[currentCity] = true;

    for (int i = 1; i < N; ++i) {
        double minDistance = std::numeric_limits<double>::max();
        int nextCity = -1;

        for (int j = 0; j < N; ++j) {
            if (!visited[j]) {
                double dist = calculateDistance(cities[currentCity], cities[j]);
                if (dist < minDistance) {
                    minDistance = dist;
                    nextCity = j;
                }
            }
        }
        currentCity = nextCity;
        tour.push_back(currentCity);
        visited[currentCity] = true;
    }
    return tour;
}

// Calculate total tour length (with distMatrix)
double calculateTourLength(const std::vector<int>& tour, const std::vector<std::vector<double>>& distMatrix) {
    double totalLength = 0;
    for (size_t i = 0; i < tour.size() - 1; ++i) {
        totalLength += distMatrix[tour[i]][tour[i+1]];
    }
    totalLength += distMatrix[tour.back()][tour[0]]; // Return to start
    return totalLength;
}

// Calculate total tour length (without distMatrix - calculates distances on the fly)
double calculateTourLength(const std::vector<int>& tour, const std::vector<City>& cities) {
    double totalLength = 0;
    for (size_t i = 0; i < tour.size() - 1; ++i) {
        totalLength += calculateDistance(cities[tour[i]], cities[tour[i+1]]);
    }
    totalLength += calculateDistance(cities[tour.back()], cities[tour[0]]); // Return to start
    return totalLength;
}

// 2-opt optimization
std::vector<int> twoOpt(int N, std::vector<int> tour, const std::vector<std::vector<double>>& distMatrix) {
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 1; i < N - 1; ++i) {
            for (int k = i + 1; k < N; ++k) {
                // Current edges: (tour[i-1], tour[i]) and (tour[k], tour[(k+1)%N])
                // New edges: (tour[i-1], tour[k]) and (tour[i], tour[(k+1)%N])

                int p1 = tour[i-1];
                int p2 = tour[i];
                int p3 = tour[k];
                int p4 = tour[(k + 1 == N) ? 0 : k + 1];

                double currentSegmentLength = distMatrix[p1][p2] + distMatrix[p3][p4];
                double newSegmentLength = distMatrix[p1][p3] + distMatrix[p2][p4];

                if (newSegmentLength < currentSegmentLength) {
                    std::reverse(tour.begin() + i, tour.begin() + k + 1);
                    improved = true;
                }
            }
        }
    }
    return tour;
}

// Main solving function
std::vector<int> solveTSP(int N, const std::vector<City>& cities) {
    std::vector<int> tour;

    if (N <= 15) { // Brute force for optimal solution up to N=15
        std::vector<std::vector<double>> distMatrix(N, std::vector<double>(N));
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                distMatrix[i][j] = calculateDistance(cities[i], cities[j]);
            }
        }

        std::vector<int> p(N);
        std::iota(p.begin(), p.end(), 0); // Fills p with 0, 1, ..., N-1

        double minTourLength = std::numeric_limits<double>::max();
        std::vector<int> bestTour;

        std::vector<int> currentPermutation(N);
        currentPermutation[0] = 0; // Fix city 0 as the starting point

        std::vector<int> citiesToPermute;
        for (int i = 1; i < N; ++i) {
            citiesToPermute.push_back(i);
        }

        std::sort(citiesToPermute.begin(), citiesToPermute.end()); // Ensure starting with smallest permutation

        do {
            for (int i = 0; i < citiesToPermute.size(); ++i) {
                currentPermutation[i + 1] = citiesToPermute[i];
            }

            double currentLength = calculateTourLength(currentPermutation, distMatrix);

            if (currentLength < minTourLength) {
                minTourLength = currentLength;
                bestTour = currentPermutation;
            }
        } while (std::next_permutation(citiesToPermute.begin(), citiesToPermute.end()));
        tour = bestTour;

    } else if (N <= 2000) { // Medium N: Nearest Neighbor + 2-opt with distMatrix
        std::vector<std::vector<double>> distMatrix(N, std::vector<double>(N));
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                distMatrix[i][j] = calculateDistance(cities[i], cities[j]);
            }
        }
        tour = nearestNeighbor(N, cities, distMatrix);
        tour = twoOpt(N, tour, distMatrix);
    } else { // N > 2000: Nearest Neighbor without distMatrix (distances on the fly)
        // For extremely large N (e.g., 10,000,000 cities), the O(N^2) complexity of this
        // Nearest Neighbor implementation will be too slow. A k-d tree or similar
        // data structure would be required to find nearest neighbors in O(log N) time,
        // reducing the overall complexity to O(N log N).
        tour = nearestNeighbor(N, cities);
        // For very large N, 2-opt is too slow, so we don't apply it here.
    }

    return tour;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int N;
    std::cin >> N;

    std::vector<City> cities(N);
    for (int i = 0; i < N; ++i) {
        cities[i].id = i;
        std::cin >> cities[i].x >> cities[i].y;
    }

    std::vector<int> resultTour = solveTSP(N, cities);

    for (int i = 0; i < N; ++i) {
        std::cout << resultTour[i] << (i == N - 1 ? "" : " ");
    }
    std::cout << std::endl;

    return 0;
}
