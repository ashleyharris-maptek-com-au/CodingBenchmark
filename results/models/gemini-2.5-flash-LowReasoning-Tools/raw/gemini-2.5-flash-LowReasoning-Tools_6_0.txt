namespace OrbitalTSP.Core
{
    public class OrbitalTspSolver
    {
        // Step 4.2: Implement Orbital Cost Function (Placeholder: Euclidean distance)
        public double CalculateOrbitalCost(OrbitalNode from, OrbitalNode to)
        {
            // In a real "orbital" scenario, this would involve complex astrodynamics
            // calculations (e.g., delta-v, travel time considering gravity, etc.).
            // For now, we use simple Euclidean distance as a placeholder.
            double dx = from.X - to.X;
            double dy = from.Y - to.Y;
            double dz = from.Z - to.Z;
            return Math.Sqrt(dx * dx + dy * dy + dz * dz);
        }

        // Step 4.3: Construct Cost Matrix
        public double[,] ConstructCostMatrix(List<OrbitalNode> nodes)
        {
            int n = nodes.Count;
            double[,] costMatrix = new double[n, n];

            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    if (i == j)
                    {
                        costMatrix[i, j] = 0; // Cost to travel to self is 0
                    }
                    else
                    {
                        costMatrix[i, j] = CalculateOrbitalCost(nodes[i], nodes[j]);
                    }
                }
            }
            return costMatrix;
        }

        // Step 4.4: Implement TSP Algorithm (Held-Karp Dynamic Programming)
        public (List<OrbitalNode> path, double totalCost) Solve(List<OrbitalNode> nodes, OrbitalNode startNode)
        {
            int n = nodes.Count;
            if (n == 0) return (new List<OrbitalNode>(), 0);
            if (n == 1) return (new List<OrbitalNode> { nodes[0] }, 0);

            int startIndex = nodes.IndexOf(startNode);
            if (startIndex == -1)
            {
                throw new ArgumentException("Start node not found in the provided list of nodes.");
            }

            double[,] costMatrix = ConstructCostMatrix(nodes);

            // dp[mask][i] will store the minimum cost to visit all nodes in 'mask', ending at node 'i'
            double[,] dp = new double[1 << n, n];
            int[,] parent = new int[1 << n, n]; // To reconstruct the path

            // Initialize dp table with infinity
            for (int i = 0; i < (1 << n); i++)
            {
                for (int j = 0; j < n; j++)
                {
                    dp[i, j] = double.PositiveInfinity;
                }
            }

            // Base case: Starting from the startNode, visiting only one node (itself)
            dp[1 << startIndex, startIndex] = 0;

            // Iterate over all possible masks (subsets of nodes)
            for (int mask = 1; mask < (1 << n); mask++)
            {
                for (int i = 0; i < n; i++) // Current end node 'i'
                {
                    // If node 'i' is part of the current mask
                    if ((mask & (1 << i)) != 0)
                    {
                        for (int j = 0; j < n; j++) // Previous node 'j'
                        {
                            if (i == j || (mask & (1 << j)) == 0) continue; // Skip if i==j or j is not in mask

                            // If we can reach mask without 'i', ending at 'j', and then go to 'i'
                            if (dp[mask ^ (1 << i), j] + costMatrix[j, i] < dp[mask, i])
                            {
                                dp[mask, i] = dp[mask ^ (1 << i), j] + costMatrix[j, i];
                                parent[mask, i] = j;
                            }
                        }
                    }
                }
            }

            // Find the minimum cost to return to the startNode from any other node
            double minCost = double.PositiveInfinity;
            int lastNodeIndex = -1;
            int finalMask = (1 << n) - 1; // All nodes visited

            for (int i = 0; i < n; i++)
            {
                if (i == startIndex) continue;
                if (dp[finalMask, i] + costMatrix[i, startIndex] < minCost)
                {
                    minCost = dp[finalMask, i] + costMatrix[i, startIndex];
                    lastNodeIndex = i;
                }
            }

            // Path Reconstruction
            List<OrbitalNode> path = new List<OrbitalNode>();
            if (lastNodeIndex != -1)
            {
                int currentMask = finalMask;
                int currentNodeIndex = lastNodeIndex;

                while (currentNodeIndex != startIndex || currentMask != (1 << startIndex))
                {
                    path.Insert(0, nodes[currentNodeIndex]);
                    int prevNodeIndex = parent[currentMask, currentNodeIndex];
                    currentMask ^= (1 << currentNodeIndex); // Remove current node from mask
                    currentNodeIndex = prevNodeIndex;
                }
                path.Insert(0, nodes[startIndex]); // Add the start node at the beginning
                path.Add(nodes[startIndex]); // Complete the cycle by returning to the start
            }

            return (path, minCost);
        }
    }
}
