#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <map>

// Represents an individual operation
struct Operation {
    int machine_id;
    int duration;
    int job_id;
    int operation_idx; // Index of this operation within its job's sequence
    int start_time = -1;
    int end_time = -1;
};

// Represents a job, which is a sequence of operations
struct Job {
    int id;
    std::vector<Operation> operations;
    int next_op_idx = 0; // Pointer to the next operation to be scheduled for this job
};

// Function to solve JSSP using a greedy SPT heuristic
void solveJSSP(int num_machines, std::vector<Job>& jobs) {
    int num_jobs = jobs.size();
    
    // machine_free_time[m] stores the time when machine 'm' becomes free
    std::vector<int> machine_free_time(num_machines, 0);
    // job_next_free_time[j] stores the time when job 'j' can start its next operation
    std::vector<int> job_next_free_time(num_jobs, 0);

    int completed_ops_count = 0;
    int total_ops = 0;
    for (const auto& job : jobs) {
        total_ops += job.operations.size();
    }

    std::cout << "Starting JSSP scheduling with " << num_jobs << " jobs and " << num_machines << " machines." << std::endl;

    // Loop until all operations are scheduled
    while (completed_ops_count < total_ops) {
        int best_job_to_schedule = -1;
        int earliest_possible_start_time = -1;
        int shortest_duration = -1;

        // Iterate through all jobs to find the next operation to schedule
        for (int i = 0; i < num_jobs; ++i) {
            // Check if job 'i' has unscheduled operations
            if (jobs[i].next_op_idx < jobs[i].operations.size()) {
                Operation& current_op = jobs[i].operations[jobs[i].next_op_idx];
                
                // Calculate the earliest time this operation can start
                // It must wait for its machine to be free AND for its job's previous operation to finish
                int current_op_start_time = std::max(machine_free_time[current_op.machine_id], job_next_free_time[i]);

                // Heuristic: Shortest Processing Time (SPT)
                // Select the operation that can start earliest, and among those, the one with the shortest duration.
                if (best_job_to_schedule == -1 || 
                    current_op_start_time < earliest_possible_start_time ||
                    (current_op_start_time == earliest_possible_start_time && current_op.duration < shortest_duration)) 
                {
                    earliest_possible_start_time = current_op_start_time;
                    shortest_duration = current_op.duration;
                    best_job_to_schedule = i;
                }
            }
        }

        // If an operation was selected, schedule it
        if (best_job_to_schedule != -1) {
            Job& job_to_update = jobs[best_job_to_schedule];
            Operation& op_to_schedule = job_to_update.operations[job_to_update.next_op_idx];

            op_to_schedule.start_time = earliest_possible_start_time;
            op_to_schedule.end_time = op_to_schedule.start_time + op_to_schedule.duration;

            // Update machine and job free times
            machine_free_time[op_to_schedule.machine_id] = op_to_schedule.end_time;
            job_next_free_time[best_job_to_schedule] = op_to_schedule.end_time;
            
            // Move to the next operation for this job
            job_to_update.next_op_idx++;
            completed_ops_count++;
            
            std::cout << "Scheduled Job " << op_to_schedule.job_id 
                      << " Op " << op_to_schedule.operation_idx 
                      << " on Machine " << op_to_schedule.machine_id 
                      << " from " << op_to_schedule.start_time 
                      << " to " << op_to_schedule.end_time 
                      << " (Duration: " << op_to_schedule.duration << ")" << std::endl;
        } else {
            // This case should ideally not happen if total_ops is correctly calculated and there are unscheduled ops
            std::cerr << "Error: No operation could be scheduled. This indicates a problem in the logic or input." << std::endl;
            break;
        }
    }

    // Calculate Makespan
    int makespan = 0;
    for (int time : machine_free_time) {
        if (time > makespan) {
            makespan = time;
        }
    }
    std::cout << "
--- Scheduling Complete ---" << std::endl;
    std::cout << "Final Makespan: " << makespan << std::endl;

    // Optional: Print detailed schedule
    std::cout << "
Detailed Schedule:" << std::endl;
    for (const auto& job : jobs) {
        std::cout << "Job " << job.id << ":" << std::endl;
        for (const auto& op : job.operations) {
            std::cout << "  Operation " << op.operation_idx 
                      << " (Machine " << op.machine_id 
                      << "): Start=" << op.start_time 
                      << ", End=" << op.end_time 
                      << ", Duration=" << op.duration << std::endl;
        }
    }
}

int main() {
    // Example JSSP Instance: (taken from the search result and slightly modified for clarity)
    // 2 Jobs, 2 Machines
    // Job 0: Operation 0 (Machine 0, Duration 3), Operation 1 (Machine 1, Duration 2)
    // Job 1: Operation 0 (Machine 1, Duration 2), Operation 1 (Machine 0, Duration 4)

    // Define operations for Job 0
    std::vector<Operation> job0_ops;
    job0_ops.push_back({0, 3, 0, 0}); // machine_id, duration, job_id, operation_idx
    job0_ops.push_back({1, 2, 0, 1});

    // Define operations for Job 1
    std::vector<Operation> job1_ops;
    job1_ops.push_back({1, 2, 1, 0});
    job1_ops.push_back({0, 4, 1, 1});

    // Create Job objects
    std::vector<Job> jobs;
    jobs.push_back({0, job0_ops});
    jobs.push_back({1, job1_ops});

    int num_machines = 2; // Total number of machines available

    // Solve the JSSP
    solveJSSP(num_machines, jobs);

    return 0;
}
