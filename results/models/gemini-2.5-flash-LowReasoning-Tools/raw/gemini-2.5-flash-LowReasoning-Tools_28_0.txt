use std::collections::{HashSet, HashMap};

#[derive(Debug, Clone, PartialEq, Eq, Hash)]
struct Vertex(usize);

#[derive(Debug)]
struct Graph {
    adj: HashMap<Vertex, HashSet<Vertex>>,
    num_vertices: usize,
}

impl Graph {
    fn new(num_vertices: usize) -> Self {
        let mut adj = HashMap::new();
        for i in 0..num_vertices {
            adj.insert(Vertex(i), HashSet::new());
        }
        Graph { adj, num_vertices }
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        let vert_u = Vertex(u);
        let vert_v = Vertex(v);
        self.adj.get_mut(&vert_u).unwrap().insert(vert_v.clone());
        self.adj.get_mut(&vert_v).unwrap().insert(vert_u.clone());
    }

    // A 2-approximation algorithm for Minimum Vertex Cover
    fn min_vertex_cover_approx(&self) -> HashSet<Vertex> {
        let mut cover = HashSet::new();
        let mut edges = self.get_all_edges();

        while let Some((u, v)) = edges.iter().next().cloned() {
            cover.insert(u.clone());
            cover.insert(v.clone());

            // Remove all edges incident to u or v
            edges.retain(|(e_u, e_v)| {
                !(e_u == &u || e_u == &v || e_v == &u || e_v == &v)
            });
        }
        cover
    }

    fn get_all_edges(&self) -> HashSet<(Vertex, Vertex)> {
        let mut edges = HashSet::new();
        for (u, neighbors) in &self.adj {
            for v in neighbors {
                // Ensure each edge is added only once (e.g., (u,v) but not (v,u))
                if u.0 < v.0 {
                    edges.insert((u.clone(), v.clone()));
                }
            }
        }
        edges
    }
}

fn main() {
    // Example Usage:
    // Create a graph with 7 vertices
    let mut graph = Graph::new(7);

    // Add edges
    graph.add_edge(0, 1);
    graph.add_edge(0, 2);
    graph.add_edge(1, 3);
    graph.add_edge(2, 3);
    graph.add_edge(3, 4);
    graph.add_edge(4, 5);
    graph.add_edge(5, 6);
    graph.add_edge(6, 0);

    println!("Graph: {:?}", graph);

    let cover = graph.min_vertex_cover_approx();
    println!("Approximate Minimum Vertex Cover: {:?}", cover);
    println!("Size of cover: {}", cover.len());

    // Another example
    let mut graph2 = Graph::new(5);
    graph2.add_edge(0, 1);
    graph2.add_edge(0, 2);
    graph2.add_edge(1, 3);
    graph2.add_edge(2, 4);
    graph2.add_edge(3, 4);

    println!("
Graph 2: {:?}", graph2);
    let cover2 = graph2.min_vertex_cover_approx();
    println!("Approximate Minimum Vertex Cover 2: {:?}", cover2);
    println!("Size of cover 2: {}", cover2.len());
}
