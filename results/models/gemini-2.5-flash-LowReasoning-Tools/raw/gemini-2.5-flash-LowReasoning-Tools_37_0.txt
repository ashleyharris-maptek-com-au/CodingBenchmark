using Xunit;
using System.Collections.Generic;
using System.Linq;

public class DFVSSolverTests
{
    private DFVSSolver _solver;

    public DFVSSolverTests()
    {
        _solver = new DFVSSolver();
    }

    [Fact]
    public void Solve_AcyclicGraph_ReturnsEmptyFVS()
    {
        // A -> B -> C (Acyclic)
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int> { 2 } },
            { 2, new List<int> { 3 } },
            { 3, new List<int>() }
        };

        var fvs = _solver.Solve(graph);
        Assert.Empty(fvs);
    }

    [Fact]
    public void Solve_SimpleCycle_ReturnsOneVertex()
    {
        // 1 -> 2 -> 3 -> 1 (Simple cycle)
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int> { 2 } },
            { 2, new List<int> { 3 } },
            { 3, new List<int> { 1 } }
        };

        var fvs = _solver.Solve(graph);
        Assert.Single(fvs);
        Assert.Contains(fvs.First(), new List<int> { 1, 2, 3 });
    }

    [Fact]
    public void Solve_MultipleDisjointCycles_ReturnsOneVertexPerCycle()
    {
        // 1 -> 2 -> 1 (Cycle 1)
        // 3 -> 4 -> 3 (Cycle 2)
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int> { 2 } },
            { 2, new List<int> { 1 } },
            { 3, new List<int> { 4 } },
            { 4, new List<int> { 3 } }
        };

        var fvs = _solver.Solve(graph);
        Assert.Equal(2, fvs.Count); // Expect one from {1,2} and one from {3,4}
        Assert.True((fvs.Contains(1) || fvs.Contains(2)) && (fvs.Contains(3) || fvs.Contains(4)));
    }

    [Fact]
    public void Solve_ComplexGraphWithIntersectingCycles_ReturnsValidFVS()
    {
        // Example: Figure 1 from "Feedback Vertex Set" on Wikipedia
        // Cycles: 1->2->3->1, 1->2->4->1, 3->4->1 (No, 3->4 then 4 has edge to 1, this makes cycle)
        // 1 -> 2
        // 2 -> 3, 2 -> 4
        // 3 -> 1
        // 4 -> 1
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int> { 2 } },
            { 2, new List<int> { 3, 4 } },
            { 3, new List<int> { 1 } },
            { 4, new List<int> { 1 } }
        };

        var fvs = _solver.Solve(graph);
        // Possible FVS: {1} or {2} or {3,4} (depends on greedy choice)
        // The greedy approach might yield {1} or {2} for this graph.
        Assert.Single(fvs); // In this specific graph, removing 1 or 2 makes it acyclic.
        Assert.True(fvs.Contains(1) || fvs.Contains(2));
    }

    [Fact]
    public void Solve_EmptyGraph_ReturnsEmptyFVS()
    {
        var graph = new Dictionary<int, List<int>>();
        var fvs = _solver.Solve(graph);
        Assert.Empty(fvs);
    }

    [Fact]
    public void Solve_SingleVertexGraph_ReturnsEmptyFVS()
    {
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int>() }
        };
        var fvs = _solver.Solve(graph);
        Assert.Empty(fvs);
    }

    [Fact]
    public void Solve_GraphWithSelfLoop_ReturnsVertexWithSelfLoop()
    {
        // 1 -> 1 (Self-loop)
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int> { 1 } }
        };
        var fvs = _solver.Solve(graph);
        Assert.Single(fvs);
        Assert.Contains(1, fvs);
    }

    [Fact]
    public void Solve_DisconnectedCycles_ReturnsCorrectFVS()
    {
        // 1 -> 2 -> 1
        // 3 -> 4 -> 5 -> 3
        var graph = new Dictionary<int, List<int>>
        {
            { 1, new List<int> { 2 } },
            { 2, new List<int> { 1 } },
            { 3, new List<int> { 4 } },
            { 4, new List<int> { 5 } },
            { 5, new List<int> { 3 } }
        };

        var fvs = _solver.Solve(graph);
        Assert.Equal(2, fvs.Count); // One from {1,2}, one from {3,4,5}
        Assert.True((fvs.Contains(1) || fvs.Contains(2)));
        Assert.True((fvs.Contains(3) || fvs.Contains(4) || fvs.Contains(5)));
    }
    
    [Fact]
    public void Solve_LargeGraph_ReturnsFVS()
    {
        var graph = new Dictionary<int, List<int>>();
        int numVertices = 100;
        for (int i = 0; i < numVertices; i++)
        {
            graph[i] = new List<int>();
        }

        // Create a large cycle
        for (int i = 0; i < numVertices; i++)
        {
            graph[i].Add((i + 1) % numVertices);
        }

        var fvs = _solver.Solve(graph);
        Assert.Single(fvs); // Greedy will likely pick one vertex to break the large cycle
        Assert.Contains(fvs.First(), Enumerable.Range(0, numVertices));
    }
}
