use geo_types::{polygon, Point, Polygon}; // Point and Polygon are needed for transformations
use log::{info, LevelFilter};
use env_logger::Builder;
use u_nesting_d2::{Boundary2D, Config, Geometry2D, Nester2D, Strategy, Solver};
use svg::{Document, node::element::Path}; // Removed SvgPolygon
use svg::node::element::path::Data;
use std::collections::HashMap; // Import HashMap
use std::fs;

// Helper function to convert geo_types::Polygon to svg::node::element::Path
fn geo_polygon_to_svg_path(poly: &Polygon, fill_color: &str, stroke_color: &str, stroke_width: f64) -> Path {
    let mut data = Data::new();

    // Exterior ring
    let mut exterior_coords = poly.exterior().coords();
    if let Some(first_coord) = exterior_coords.next() {
        data = data.move_to((first_coord.x, first_coord.y));
        for coord in exterior_coords {
            data = data.line_to((coord.x, coord.y));
        }
        data = data.close();
    }

    // Interior rings (holes)
    for interior_ring in poly.interiors() {
        let mut interior_coords = interior_ring.coords();
        if let Some(first_coord) = interior_coords.next() {
            data = data.move_to((first_coord.x, first_coord.y));
            for coord in interior_coords {
                data = data.line_to((coord.x, coord.y));
            }
            data = data.close();
        }
    }

    Path::new()
        .set("fill", fill_color)
        .set("stroke", stroke_color)
        .set("stroke-width", stroke_width)
        .set("d", data)
}

// Function to apply translation and rotation to a geo_types::Polygon
fn transform_geo_polygon(poly: &Polygon, translate_x: f64, translate_y: f64, rotate_deg: f64) -> Polygon {
    let mut transformed_exterior = poly.exterior().clone();
    let mut transformed_interiors = poly.interiors().to_vec();

    // Apply rotation around the polygon's centroid or a chosen origin (e.g., 0,0)
    // For simplicity, let's rotate around (0,0) first, then translate.
    // This is not strictly correct for polygon's own center, but sufficient for visual.
    let rotate_rad = rotate_deg.to_radians();
    let cos_theta = rotate_rad.cos();
    let sin_theta = rotate_rad.sin();

    let transform_point = |p: Point| -> Point {
        let rotated_x = p.x() * cos_theta - p.y() * sin_theta;
        let rotated_y = p.x() * sin_theta + p.y() * cos_theta;
        Point::new(rotated_x + translate_x, rotated_y + translate_y)
    };

    transformed_exterior = transformed_exterior.points().map(transform_point).collect();
    transformed_interiors = transformed_interiors.into_iter().map(|ring| {
        ring.points().map(transform_point).collect()
    }).collect();

    Polygon::new(transformed_exterior, transformed_interiors)
}


fn main() {
    // Initialize logging
    Builder::new()
        .filter_level(LevelFilter::Info)
        .init();

    info!("Starting polygon nesting application...");

    // Define the stock polygon (a simple rectangle)
    let stock_polygon_geo = polygon![
        (x: 0.0, y: 0.0),
        (x: 100.0, y: 0.0),
        (x: 100.0, y: 100.0),
        (x: 0.0, y: 100.0),
        (x: 0.0, y: 0.0),
    ];
    info!("Stock polygon defined (geo_types): {:?}", stock_polygon_geo);

    // Define an L-shaped item polygon
    let item_polygon_geo_1 = polygon![
        (x: 0.0, y: 0.0),
        (x: 30.0, y: 0.0),
        (x: 30.0, y: 10.0),
        (x: 10.0, y: 10.0),
        (x: 10.0, y: 30.0),
        (x: 0.0, y: 30.0),
        (x: 0.0, y: 0.0),
    ];
    info!("Item polygon 1 defined (geo_types): {:?}", item_polygon_geo_1);

    // Define another L-shaped item polygon (can be a copy for now)
    let item_polygon_geo_2 = polygon![
        (x: 0.0, y: 0.0),
        (x: 30.0, y: 0.0),
        (x: 30.0, y: 10.0),
        (x: 10.0, y: 10.0),
        (x: 10.0, y: 30.0),
        (x: 0.0, y: 30.0),
        (x: 0.0, y: 0.0),
    ];
    info!("Item polygon 2 defined (geo_types): {:?}", item_polygon_geo_2);

    let item_polygons_geo = vec![
        ("item_0".to_string(), item_polygon_geo_1.clone()),
        ("item_1".to_string(), item_polygon_geo_2.clone()),
    ];

    // Convert geo_types::Polygon to u_nesting_d2::Geometry2D and Boundary2D
    // For stock, use the bounding box or convert its exterior to Boundary2D
    // u-nesting-d2's Boundary2D::rectangle is convenient.
    let boundary = Boundary2D::rectangle(100.0, 100.0); // Assuming 100x100 for our stock
    info!("Boundary defined for u-nesting-d2: {:?}", boundary);

    let mut items_nesting: Vec<Geometry2D> = Vec::new();
    for (id, poly_geo) in item_polygons_geo.iter() {
        let exterior_coords: Vec<(f64, f64)> = poly_geo.exterior().coords().map(|c| (c.x, c.y)).collect();
        // Assuming no interior rings for simplicity for now
        let item_nesting = Geometry2D::new(id)
            .with_polygon(exterior_coords)
            .with_rotations_deg(vec![0.0, 90.0, 180.0, 270.0]); // Allow 90-degree rotations
        items_nesting.push(item_nesting);
    }
    info!("Converted geo_types items to u-nesting-d2 Geometry2D items.");

    // Configure the Nester
    let config = Config::new()
        .with_strategy(Strategy::NfpGuided) // Revert to NfpGuided strategy
        .with_spacing(0.1) // Small gap between parts
        .with_margin(0.1) // Small margin from boundary edges
        .with_time_limit(60000); // Set a timeout of 60 seconds (60000 ms)

    let nester = Nester2D::new(config);
    info!("Nester2D configured with NfpGuided strategy and a time limit.");

    // Solve the nesting problem
    match nester.solve(&items_nesting, &boundary) {
        Ok(result) => {
            info!("Nesting Complete!");
            println!("Nesting Complete!");
            println!("Items Placed: {}", result.placements.len());
            println!("Material Utilization: {:.2}%", result.utilization * 100.0);

            let mut document = Document::new().set("viewBox", (0, 0, 100, 100)); // Set viewBox to match stock size

            // Add stock polygon to SVG
            let stock_svg = geo_polygon_to_svg_path(&stock_polygon_geo, "lightgray", "black", 0.5);
            document = document.add(stock_svg);

            // Create a hashmap for easy lookup of original geo_types polygons by their ID
            let original_item_polygons: HashMap<String, Polygon> = item_polygons_geo.into_iter().collect();

            // Add placed items to SVG
            for placement in result.placements {
                println!(
                    "Item: {} | Position: ({:.1}, {:.1}) | Rotation: {:?}°",
                    placement.geometry_id,
                    placement.position[0],
                    placement.position[1],
                    placement.rotation
                );

                if let Some(original_poly) = original_item_polygons.get(&placement.geometry_id) {
                    // u-nesting-d2's `placement.rotation` is a Vec<f64>.
                    // Assuming the first element is the rotation in degrees for now based on the previous output.
                    // This is a simplification and might need adjustment if rotation is more complex.
                    let rotation_deg = placement.rotation.first().copied().unwrap_or(0.0);

                    let transformed_poly = transform_geo_polygon(
                        original_poly,
                        placement.position[0],
                        placement.position[1],
                        rotation_deg,
                    );
                    let item_svg = geo_polygon_to_svg_path(&transformed_poly, "skyblue", "blue", 0.2);
                    document = document.add(item_svg);
                } else {
                    log::warn!("Original polygon not found for geometry_id: {}", placement.geometry_id);
                }
            }

            // Save SVG to file
            match fs::write("output.svg", document.to_string()) {
                Ok(_) => info!("SVG output saved to output.svg"),
                Err(e) => log::error!("Failed to save SVG: {:?}", e),
            }
        }
        Err(e) => {
            eprintln!("Nesting failed: {:?}", e);
            log::error!("Nesting failed: {:?}", e);
        },
    }

    println!("Hello, world! Polygons defined and nesting attempted.");
}

#[cfg(test)]
mod tests {
    use super::*;
    use geo_types::{polygon, Point};

    // Helper for floating point comparisons
    const EPSILON: f64 = 1e-9;

    #[test]
    fn test_transform_geo_polygon_translation() {
        let original_square = polygon![
            (x: 0.0, y: 0.0),
            (x: 10.0, y: 0.0),
            (x: 10.0, y: 10.0),
            (x: 0.0, y: 10.0),
            (x: 0.0, y: 0.0),
        ];

        let translated_square = transform_geo_polygon(&original_square, 5.0, 5.0, 0.0);

        // Check a single point after translation
        let first_point = translated_square.exterior().points().next().unwrap();
        assert!((first_point.x() - 5.0).abs() < EPSILON);
        assert!((first_point.y() - 5.0).abs() < EPSILON);

        // Check another point
        let second_point = translated_square.exterior().points().nth(1).unwrap();
        assert!((second_point.x() - 15.0).abs() < EPSILON);
        assert!((second_point.y() - 5.0).abs() < EPSILON);
    }

    #[test]
    fn test_transform_geo_polygon_rotation_90_deg() {
        let original_point = Point::new(1.0, 0.0);
        // Rotate (1,0) by 90 degrees around (0,0) -> (0,1)

        let square_for_rotation = polygon![
            (x: 0.0, y: 0.0),
            (x: 1.0, y: 0.0),
            (x: 1.0, y: 1.0),
            (x: 0.0, y: 1.0),
            (x: 0.0, y: 0.0),
        ];

        let rotated_square = transform_geo_polygon(&square_for_rotation, 0.0, 0.0, 90.0);

        let first_rotated_point = rotated_square.exterior().points().nth(1).unwrap(); // Original (1,0)
        assert!((first_rotated_point.x() - 0.0).abs() < EPSILON);
        assert!((first_rotated_point.y() - 1.0).abs() < EPSILON);

        let second_rotated_point = rotated_square.exterior().points().nth(2).unwrap(); // Original (1,1)
        assert!((second_rotated_point.x() - -1.0).abs() < EPSILON);
        assert!((second_rotated_point.y() - 1.0).abs() < EPSILON);
    }

    #[test]
    fn test_transform_geo_polygon_rotation_and_translation() {
        let original_square = polygon![
            (x: 0.0, y: 0.0),
            (x: 10.0, y: 0.0),
            (x: 10.0, y: 10.0),
            (x: 0.0, y: 10.0),
            (x: 0.0, y: 0.0),
        ];

        // Rotate by 90 degrees and then translate by (5,5)
        let transformed_square = transform_geo_polygon(&original_square, 5.0, 5.0, 90.0);

        // Original point (10,0) after 90 deg rotation around (0,0) is (0,10).
        // Then translate by (5,5) -> (5, 15)
        let expected_x_point_2 = 5.0;
        let expected_y_point_2 = 15.0;

        let transformed_point_2 = transformed_square.exterior().points().nth(1).unwrap();
        assert!((transformed_point_2.x() - expected_x_point_2).abs() < EPSILON);
        assert!((transformed_point_2.y() - expected_y_point_2).abs() < EPSILON);
    }
}
