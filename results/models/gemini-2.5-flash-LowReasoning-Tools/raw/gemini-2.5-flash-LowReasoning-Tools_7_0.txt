// Define a small epsilon for floating-point comparisons
const double EPSILON = 1e-9;

// --- Data Structures ---

struct Vec3 {
    double x, y, z;

    // Default constructor
    Vec3() : x(0.0), y(0.0), z(0.0) {}

    // Parameterized constructor
    Vec3(double x_in, double y_in, double z_in) : x(x_in), y(y_in), z(z_in) {}

    // Vector operations
    Vec3 operator+(const Vec3& other) const { return Vec3(x + other.x, y + other.y, z + other.z); }
    Vec3 operator-(const Vec3& other) const { return Vec3(x - other.x, y - other.y, z - other.z); }
    Vec3 operator*(double scalar) const { return Vec3(x * scalar, y * scalar, z * scalar); }
    Vec3 operator/(double scalar) const { return Vec3(x / scalar, y / scalar, z / scalar); }
    double dot(const Vec3& other) const { return x * other.x + y * other.y + z * other.z; }
    Vec3 cross(const Vec3& other) const {
        return Vec3(y * other.z - z * other.y,
                    z * other.x - x * other.z,
                    x * other.y - y * other.x);
    }
    double length_sq() const { return x*x + y*y + z*z; }
    double length() const { return std::sqrt(length_sq()); }
    Vec3 normalize() const {
        double len = length();
        if (len > EPSILON) {
            return Vec3(x / len, y / len, z / len);
        }
        return Vec3(0.0, 0.0, 0.0); // Handle zero-length vector
    }

    // Comparison for sorting/hashing (consider EPSILON for floating point comparisons if needed)
    bool operator==(const Vec3& other) const {
        return std::abs(x - other.x) < EPSILON &&
               std::abs(y - other.y) < EPSILON &&
               std::abs(z - other.z) < EPSILON;
    }
    // This operator< is used for sorting in AABB tree and other places.
    // For std::map, a custom comparator `Vec3MapCompare` is used for precision control.
    bool operator<(const Vec3& other) const {
        if (x < other.x - EPSILON) return true;
        if (x > other.x + EPSILON) return false;
        if (y < other.y - EPSILON) return true;
        if (y > other.y + EPSILON) return false;
        return z < other.z - EPSILON;
    }
};

// Custom comparator for Vec3 in map to handle floating point precision
// This ensures a strict weak ordering required by std::map
struct Vec3MapCompare {
    bool operator()(const Vec3& a, const Vec3& b) const {
        if (std::abs(a.x - b.x) > EPSILON) return a.x < b.x;
        if (std::abs(a.y - b.y) > EPSILON) return a.y < b.y;
        return a.z < b.z - EPSILON; // Account for epsilon in the last dimension
    }
};


struct Triangle {
    int v[3]; // Indices into the vertex array

    Triangle(int v0, int v1, int v2) {
        v[0] = v0;
        v[1] = v1;
        v[2] = v2;
    }
};

// --- Bounding Box and AABB Tree Structures ---

struct BBox {
    Vec3 min_p = Vec3(std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max());
    Vec3 max_p = Vec3(std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest());

    BBox() = default;

    BBox(const Vec3& p) : min_p(p), max_p(p) {}

    BBox(const Vec3& p_min, const Vec3& p_max) : min_p(p_min), max_p(p_max) {}

    void extend(const Vec3& p) {
        min_p.x = std::min(min_p.x, p.x);
        min_p.y = std::min(min_p.y, p.y);
        min_p.z = std::min(min_p.z, p.z);
        max_p.x = std::max(max_p.x, p.x);
        max_p.y = std::max(max_p.y, p.y);
        max_p.z = std::max(max_p.z, p.z);
    }

    void extend(const BBox& other) {
        extend(other.min_p);
        extend(other.max_p);
    }
    
    // Check for ray-bbox intersection (optimized version)
    bool ray_intersects_bbox(const Vec3& ray_origin, const Vec3& ray_direction, double& t_min, double& t_max) const {
        t_min = std::numeric_limits<double>::lowest();
        t_max = std::numeric_limits<double>::max();

        for (int i = 0; i < 3; ++i) {
            double inv_d = 1.0 / (&ray_direction.x)[i];
            double t0 = (((&min_p.x)[i] - (&ray_origin.x)[i]) * inv_d);
            double t1 = (((&max_p.x)[i] - (&ray_origin.x)[i]) * inv_d);

            if (inv_d < 0.0) std::swap(t0, t1);

            t_min = std::max(t0, t_min);
            t_max = std::min(t1, t_max);

            if (t_max <= t_min) return false;
        }
        return true;
    }
};

struct AABBTreeNode {
    BBox bbox;
    std::vector<int> triangle_indices; // Indices to triangles in the Mesh's faces vector
    std::unique_ptr<AABBTreeNode> left;
    std::unique_ptr<AABBTreeNode> right;

    bool is_leaf() const {
        return !left && !right;
    }
};

struct Mesh {
    std::vector<Vec3> vertices;
    std::vector<Triangle> faces;
    std::unique_ptr<AABBTreeNode> aabb_tree_root; // Root of the AABB tree
};

// --- Helper Functions for Input/Output ---

void read_mesh(Mesh& mesh) {
    int v_count, f_count;
    std::cin >> v_count >> f_count;

    mesh.vertices.reserve(v_count);
    for (int i = 0; i < v_count; ++i) {
        double x, y, z;
        std::cin >> x >> y >> z;
        mesh.vertices.emplace_back(x, y, z);
    }

    mesh.faces.reserve(f_count);
    for (int i = 0; i < f_count; ++i) {
        int v0, v1, v2;
        std::cin >> v0 >> v1 >> v2;
        mesh.faces.emplace_back(v0, v1, v2);
    }
}

void write_mesh(const Mesh& mesh) {
    std::cout << mesh.vertices.size() << " " << mesh.faces.size() << std::endl;
    std::cout.precision(std::numeric_limits<double>::max_digits10); // Ensure full double precision for output
    for (const auto& v : mesh.vertices) {
        std::cout << v.x << " " << v.y << " " << v.z << std::endl;
    }
    for (const auto& f : mesh.faces) {
        std::cout << f.v[0] << " " << f.v[1] << " " << f.v[2] << std::endl;
    }
}

// --- Geometric Helper Functions ---

// Calculate the centroid of a triangle
Vec3 get_triangle_centroid(const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    return (v0 + v1 + v2) / 3.0;
}

// Calculate the normal of a triangle
Vec3 get_triangle_normal(const Vec3& v0, const Vec3& v1, const Vec3& v2) {
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    return edge1.cross(edge2).normalize();
}

// Moller-Trumbore algorithm for ray-triangle intersection
// Returns true if ray intersects triangle, and sets t_out to the distance along the ray
bool ray_intersects_triangle(const Vec3& ray_origin, const Vec3& ray_direction,
                             const Vec3& v0, const Vec3& v1, const Vec3& v2,
                             double& t_out) {
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 pvec = ray_direction.cross(edge2);
    double det = edge1.dot(pvec);

    // If det is close to 0, ray is parallel to triangle plane
    if (std::abs(det) < EPSILON) return false;

    double inv_det = 1.0 / det;

    Vec3 tvec = ray_origin - v0;
    double u = tvec.dot(pvec) * inv_det;
    if (u < 0.0 - EPSILON || u > 1.0 + EPSILON) return false;

    Vec3 qvec = tvec.cross(edge1);
    double v = ray_direction.dot(qvec) * inv_det;
    if (v < 0.0 - EPSILON || u + v > 1.0 + EPSILON) return false;

    t_out = edge2.dot(qvec) * inv_det;

    // Only consider intersections in the positive ray direction (t > 0)
    // And also discard intersections too close to the origin to avoid self-intersection issues
    return t_out > EPSILON;
}


// --- AABB Tree Construction and Traversal ---

// Recursive function to build the AABB tree
std::unique_ptr<AABBTreeNode> build_aabb_tree_recursive(const Mesh& mesh, std::vector<int>& triangle_indices, int depth = 0) {
    if (triangle_indices.empty()) {
        return nullptr;
    }

    std::unique_ptr<AABBTreeNode> node = std::make_unique<AABBTreeNode>();

    // Compute bounding box for current set of triangles
    for (int tri_idx : triangle_indices) {
        const Triangle& tri = mesh.faces[tri_idx];
        node->bbox.extend(mesh.vertices[tri.v[0]]);
        node->bbox.extend(mesh.vertices[tri.v[1]]);
        node->bbox.extend(mesh.vertices[tri.v[2]]);
    }

    // If too few triangles or max depth reached, make it a leaf node
    if (triangle_indices.size() <= 4 || depth > 20) { // Max 4 triangles per leaf, max depth 20
        node->triangle_indices = std::move(triangle_indices);
        return node;
    }

    // Determine the longest axis of the bounding box
    Vec3 diag = node->bbox.max_p - node->bbox.min_p;
    int axis = 0;
    if (diag.y > diag.x) axis = 1;
    if (diag.z > diag.y && diag.z > diag.x) axis = 2;

    // Sort triangles by centroid along the chosen axis
    std::sort(triangle_indices.begin(), triangle_indices.end(), [&](int a, int b) {
        Vec3 centroid_a = get_triangle_centroid(mesh.vertices[mesh.faces[a].v[0]],
                                                mesh.vertices[mesh.faces[a].v[1]],
                                                mesh.vertices[mesh.faces[a].v[2]]);
        Vec3 centroid_b = get_triangle_centroid(mesh.vertices[mesh.faces[b].v[0]],
                                                mesh.vertices[mesh.faces[b].v[1]],
                                                mesh.vertices[mesh.faces[b].v[2]]);
        return ((&centroid_a.x)[axis] < (&centroid_b.x)[axis]);
    });

    // Split into two halves
    int mid = triangle_indices.size() / 2;
    std::vector<int> left_indices(triangle_indices.begin(), triangle_indices.begin() + mid);
    std::vector<int> right_indices(triangle_indices.begin() + mid, triangle_indices.end());

    node->left = build_aabb_tree_recursive(mesh, left_indices, depth + 1);
    node->right = build_aabb_tree_recursive(mesh, right_indices, depth + 1);

    return node;
}

void build_aabb_tree(Mesh& mesh) {
    if (mesh.faces.empty()) {
        mesh.aabb_tree_root = nullptr;
        return;
    }

    std::vector<int> all_triangle_indices(mesh.faces.size());
    for (int i = 0; i < mesh.faces.size(); ++i) {
        all_triangle_indices[i] = i;
    }
    mesh.aabb_tree_root = build_aabb_tree_recursive(mesh, all_triangle_indices);
}


// Determine if a point is inside a mesh using ray casting and AABB tree
// Assumes a watertight, manifold mesh.
// Casts a ray in a random direction and counts intersections.
// Odd number of intersections means inside, even means outside.
bool is_point_inside_mesh(const Vec3& point, const Mesh& mesh) {
    if (mesh.faces.empty() || !mesh.aabb_tree_root) return false; // Cannot determine for an empty mesh or missing tree

    // Choose a random direction (fixed for simplicity, robustness requires truly random or multiple directions)
    // A direction that is unlikely to be parallel to any face normals of a typical mesh.
    Vec3 ray_direction(0.70710678, 0.70710678, 0.0); // Example: (1/sqrt(2), 1/sqrt(2), 0) normalized

    std::set<double> intersection_ts_set; // Use a set to automatically handle unique and sorted intersections

    std::vector<AABBTreeNode*> nodes_to_visit;
    nodes_to_visit.push_back(mesh.aabb_tree_root.get());

    while (!nodes_to_visit.empty()) {
        AABBTreeNode* current_node = nodes_to_visit.back();
        nodes_to_visit.pop_back();

        double t_min_bbox, t_max_bbox;
        // Optimization: if point is inside bbox, no need to check ray_intersects_bbox
        // But for a true ray, we need to check if the ray from point intersects the bbox
        if (!current_node->bbox.ray_intersects_bbox(point, ray_direction, t_min_bbox, t_max_bbox)) {
            continue;
        }

        if (current_node->is_leaf()) {
            for (int tri_idx : current_node->triangle_indices) {
                const Triangle& tri = mesh.faces[tri_idx];
                const Vec3& v0 = mesh.vertices[tri.v[0]];
                const Vec3& v1 = mesh.vertices[tri.v[1]];
                const Vec3& v2 = mesh.vertices[tri.v[2]];

                double t_hit;
                if (ray_intersects_triangle(point, ray_direction, v0, v1, v2, t_hit)) {
                    intersection_ts_set.insert(t_hit);
                }
            }
        } else {
            // Traverse children
            if (current_node->left) {
                nodes_to_visit.push_back(current_node->left.get());
            }
            if (current_node->right) {
                nodes_to_visit.push_back(current_node->right.get());
            }
        }
    }
    
    // The number of intersections is the size of the set.
    return (intersection_ts_set.size() % 2) == 1;
}

// --- Main CSG Union Logic ---
Mesh csg_union(const Mesh& mesh_a, const Mesh& mesh_b) {
    Mesh union_mesh;
    // For this simplified approach, we are not handling intersections explicitly.
    // This will produce an "approximate" union where:
    // - Triangles from A that are outside B are kept.
    // - Triangles from B that are outside A are kept.
    // Intersecting regions are not correctly handled, and new faces are not generated.

    // To manage unique vertices and mapping
    std::vector<Vec3> temp_vertices;
    std::vector<Triangle> temp_faces;
    
    // Using custom comparator for Vec3 to handle floating point precision in map keys
    std::map<Vec3, int, Vec3MapCompare> vertex_to_index_map;

    auto add_face_to_union = [&](const Vec3& v0, const Vec3& v1, const Vec3& v2) {
        auto get_or_add_vertex = [&](const Vec3& p) {
            auto it = vertex_to_index_map.find(p);
            if (it == vertex_to_index_map.end()) {
                int new_index = temp_vertices.size();
                temp_vertices.push_back(p);
                vertex_to_index_map[p] = new_index;
                return new_index;
            }
            return it->second;
        };

        int idx0 = get_or_add_vertex(v0);
        int idx1 = get_or_add_vertex(v1);
        int idx2 = get_or_add_vertex(v2);
        temp_faces.emplace_back(idx0, idx1, idx2);
    };


    // Stage 1: Classify and add non-intersecting triangles
    // Add faces from mesh_a that are entirely OUTSIDE mesh_b
    // This classification is based on the triangle's centroid.
    // LIMITATION: This approach is an approximation. Triangles that straddle the boundary
    // (partially inside and partially outside) will be incorrectly classified if their
    // centroid falls on the "wrong" side. This can lead to holes or incorrect geometry
    // where meshes overlap. A robust solution requires precise clipping
    // of intersecting triangles.
    for (const auto& face : mesh_a.faces) {
        const Vec3& v0 = mesh_a.vertices[face.v[0]];
        const Vec3& v1 = mesh_a.vertices[face.v[1]];
        const Vec3& v2 = mesh_a.vertices[face.v[2]];
        Vec3 centroid = get_triangle_centroid(v0, v1, v2);

        if (!is_point_inside_mesh(centroid, mesh_b)) {
            add_face_to_union(v0, v1, v2);
        }
    }

    // Add faces from mesh_b whose centroids are outside mesh_a
    // Same limitation as above applies.
    for (const auto& face : mesh_b.faces) {
        const Vec3& v0 = mesh_b.vertices[face.v[0]];
        const Vec3& v1 = mesh_b.vertices[face.v[1]];
        const Vec3& v2 = mesh_b.vertices[face.v[2]];
        Vec3 centroid = get_triangle_centroid(v0, v1, v2);

        if (!is_point_inside_mesh(centroid, mesh_a)) {
            add_face_to_union(v0, v1, v2);
        }
    }

    // Stage 2: Handling Intersecting Regions (Critical Missing Component for Robust CSG)
    //
    // A robust Constructive Solid Geometry (CSG) Union algorithm, especially for
    // "clean, manifold, watertight, and free of self-intersections" inputs
    // and demanding "correctness" with "ludicrous scale" meshes, would require:
    //
    // 1.  **Exact Triangle-Triangle Intersection Detection & Computation:**
    //     Efficiently find all pairs of triangles (one from mesh_a, one from mesh_b)
    //     that intersect. For each intersecting pair, compute the exact 3D line segment
    //     of intersection. AABB trees are crucial for pruning non-intersecting pairs.
    //
    // 2.  **Triangle Splitting/Clipping:**
    //     For every triangle that is intersected by another mesh, it must be split
    //     into multiple smaller (sub-)triangles or polygons. These smaller pieces
    //     then need to be individually classified. This often involves 2D polygon
    //     clipping algorithms applied in the plane of the triangle, projected from 3D.
    //
    // 3.  **Classification of Clipped Fragments:**
    //     Each new fragment resulting from the splitting must be accurately classified
    //     as either "inside" or "outside" the *other* mesh. For a union, "inside"
    //     fragments are typically discarded. This classification might involve checking
    //     a point on the fragment (e.g., centroid) using an accelerated point-in-mesh
    //     test, but careful handling is needed for points very close to the boundary.
    //
    // 4.  **Boundary Reconstruction (New Faces):**
    //     The intersection line segments computed in step 1 form a closed loop (or loops)
    //     on the surface where the two meshes cut each other. These loops must be
    //     triangulated to form new faces that connect the "outside" parts of the two
    //     original meshes, ensuring the resulting union mesh is watertight.
    //
    // 5.  **Topological Operations & Cleaning:**
    //     After collecting all fragments and new boundary faces, a topological merge
    //     is necessary. This involves:
    //     - Merging coincident vertices (as attempted by `vertex_to_index_map` above)
    //       to ensure a truly manifold output and prevent T-junctions or cracks.
    //     - Removing degenerate triangles (zero area).
    //     - Re-indexing all vertices and faces to form a coherent new mesh.
    //
    // This current implementation *does not perform* these steps for intersecting regions.
    // It relies solely on centroid-based classification, which is a heuristic and will
    // produce an incorrect or non-watertight result for meshes that actually intersect
    // each other beyond simple containment.
    
    union_mesh.vertices = std::move(temp_vertices);
    union_mesh.faces = std::move(temp_faces);

    return union_mesh;
}


// --- Main Function ---

int main() {
    // Optimize C++ streams for faster input/output
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    std::cout.tie(NULL);

    Mesh mesh_a, mesh_b;

    // Read Mesh A
    read_mesh(mesh_a);
    build_aabb_tree(mesh_a); // Build AABB tree for mesh A

    // Read Mesh B
    read_mesh(mesh_b);
    build_aabb_tree(mesh_b); // Build AABB tree for mesh B

    // Perform CSG Union operation
    Mesh union_mesh = csg_union(mesh_a, mesh_b);

    // Write the resulting union mesh
    write_mesh(union_mesh);

    return 0;
}
