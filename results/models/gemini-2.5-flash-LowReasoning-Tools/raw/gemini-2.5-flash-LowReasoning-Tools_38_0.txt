class ExactCover:
    def __init__(self, matrix):
        self.matrix = matrix
        self.num_rows = len(matrix)
        self.num_cols = len(matrix[0])
        self.solution = []
        self.min_solutions = []

    def solve(self):
        self._solve_recursive([], set(range(self.num_cols)))
        return self.min_solutions

    def _solve_recursive(self, current_solution, remaining_cols):
        if not remaining_cols:
            self.min_solutions.append(list(current_solution))
            return

        # Choose a column (deterministically choose the one with the fewest 1s)
        min_ones = float('inf')
        col_to_cover = -1
        for col in remaining_cols:
            ones_in_col = sum(self.matrix[row][col] for row in range(self.num_rows))
            if ones_in_col < min_ones:
                min_ones = ones_in_col
                col_to_cover = col

        if min_ones == 0:  # No row can cover this column
            return

        # Try each row that covers the chosen column
        for r in range(self.num_rows):
            if self.matrix[r][col_to_cover] == 1:
                # Add row r to the current solution
                current_solution.append(r)

                # Find all columns covered by row r
                covered_by_r = set()
                for c in range(self.num_cols):
                    if self.matrix[r][c] == 1:
                        covered_by_r.add(c)

                # Recursively solve for the remaining columns
                new_remaining_cols = remaining_cols - covered_by_r
                self._solve_recursive(current_solution, new_remaining_cols)

                # Backtrack: remove row r from the current solution
                current_solution.pop()

    # Example Usage:
    # Problem: Find an exact cover for the universe {0, 1, 2, 3, 4, 5, 6}
    # from the following sets:
    # A = {0, 3}
    # B = {0, 3, 6}
    # C = {1, 4}
    # D = {2, 5}
    # E = {3, 6}
    # F = {4, 5}

    # Represent the sets as a binary matrix where rows are sets and columns are elements.
    # A B C D E F  (Columns represent elements 0 to 6)
    matrix = [
        # 0  1  2  3  4  5  6
        [1, 0, 0, 1, 0, 0, 0],  # Set A
        [1, 0, 0, 1, 0, 0, 1],  # Set B
        [0, 1, 0, 0, 1, 0, 0],  # Set C
        [0, 0, 1, 0, 0, 1, 0],  # Set D
        [0, 0, 0, 1, 0, 0, 1],  # Set E
        [0, 0, 0, 0, 1, 1, 0]   # Set F
    ]

    solver = ExactCover(matrix)
    solutions = solver.solve()

    if solutions:
        print("Found exact cover solutions:")
        for sol in solutions:
            print(f"  Rows: {sol}")
    else:
        print("No exact cover found.")
