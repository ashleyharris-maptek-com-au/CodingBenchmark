; SPIR-V
; Version: 1.0
; Generator: Google-spirv-as
; Capabilities: Shader
; Memory Model: Logical GLSL450

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_color %out_color
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450
OpName %main "main"
OpName %in_color "in_color"
OpName %out_color "out_color"

; Annotations
OpDecorate %in_color Location 4
OpDecorate %out_color Location 0

; Type Declarations
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Output_vec4 = OpTypePointer Output %vec4
%float_1 = OpConstant %float 1.0

; Global Variables
%in_color = OpVariable %ptr_Input_vec3 Input
%out_color = OpVariable %ptr_Output_vec4 Output

; Main Function
%main = OpFunction %void None %void_func
%entry = OpLabel
%color_val = OpLoad %vec3 %in_color
%out_color_vec4 = OpCompositeConstruct %vec4 %color_val %float_1
OpStore %out_color %out_color_vec4
OpReturn
OpFunctionEnd
