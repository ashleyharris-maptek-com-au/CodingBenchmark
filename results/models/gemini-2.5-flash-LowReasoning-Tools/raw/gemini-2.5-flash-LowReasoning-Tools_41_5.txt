OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %ubo_var %outColor

; All types and constants
OpDecorate %worldPos Location 0

OpDecorate %tangent Location 3
OpDecorate %color Location 4

OpDecorate %UBO_block Block
OpMemberDecorate %UBO_block 0 ColMajor
OpMemberDecorate %UBO_block 0 Offset 0
OpMemberDecorate %UBO_block 1 ColMajor
OpMemberDecorate %UBO_block 1 Offset 64
OpMemberDecorate %UBO_block 2 ColMajor
OpMemberDecorate %UBO_block 2 Offset 128
OpMemberDecorate %UBO_block 3 Offset 192

%void = OpTypeVoid

%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_v4float = OpTypePointer Function %v4float
%_ptr_Function_mat4 = OpTypePointer Function %mat4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO_block = OpTypePointer Uniform %UBO_block
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%void_func_type = OpTypeFunction %void

%UBO_block = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO_block = OpTypePointer Uniform %UBO_block

%float_0_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_1_0 = OpConstant %float 1.0

; Global variables
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo_var = OpVariable %_ptr_Uniform_UBO_block Uniform

%main = OpFunction %void None %void_func_type
%entry = OpLabel
  ; Get input variables
  %_worldPos = OpLoad %v3float %worldPos
  %_normal = OpLoad %v3float %normal

  ; Get UBO values
  %lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo_var %int_3
  %lightPos = OpLoad %v4float %lightPos_ptr
  %cameraPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo_var %int_4
  %cameraPos = OpLoad %v4float %cameraPos_ptr

  ; Normalize N
  %N = OpNormalize %v3float %_normal

  ; Calculate L (light direction)
  %lightPos_v3 = OpVectorShuffle %v3float %lightPos %lightPos %int_0 %int_1 %int_2
  %L_unnorm = OpFSub %v3float %lightPos_v3 %_worldPos
  %L = OpNormalize %v3float %L_unnorm

  ; Calculate V (view direction)
  %cameraPos_v3 = OpVectorShuffle %v3float %cameraPos %cameraPos %int_0 %int_1 %int_2
  %V_unnorm = OpFSub %v3float %cameraPos_v3 %_worldPos
  %V = OpNormalize %v3float %V_unnorm

  ; Calculate diffuse
  %N_dot_L = OpDot %float %N %L
  %diffuse_pre_max = OpExtInst %float %GLSLstd450 FMax %N_dot_L %float_0_0
  %diffuse = OpCopyObject %float %diffuse_pre_max

  ; Calculate R (reflect(-L, N))
  %neg_L = OpFNegate %v3float %L
  %N_dot_neg_L = OpDot %float %N %neg_L
  %two = OpConstant %float 2.0 ; Use OpConstant for 2.0
  %two_times_N_dot_neg_L = OpFMul %float %two %N_dot_neg_L
  %two_N_dot_neg_L_times_N = OpVectorTimesScalar %v3float %N %two_times_N_dot_neg_L
  %R = OpFSub %v3float %neg_L %two_N_dot_neg_L_times_N
  %R_norm = OpNormalize %v3float %R

  ; Calculate specular
  %R_dot_V = OpDot %float %R_norm %V
  %specular_pre_max = OpExtInst %float %GLSLstd450 FMax %R_dot_V %float_0_0
  %specular = OpExtInst %float %GLSLstd450 Pow %specular_pre_max %uint_32

  ; Calculate final grayscale intensity
  %term1 = OpCopyObject %float %float_0_2
  %term2 = OpFMul %float %float_0_5 %diffuse
  %term3 = OpFMul %float %float_0_8 %specular
  %sum1 = OpFAdd %float %term1 %term2
  %intensity = OpFAdd %float %sum1 %term3

  ; Create output color
  %outColor_v4 = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1_0
  OpStore %outColor %outColor_v4
  OpReturn
OpFunctionEnd
