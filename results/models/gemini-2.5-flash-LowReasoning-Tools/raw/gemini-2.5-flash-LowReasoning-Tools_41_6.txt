; SPIR-V
; Version: 1.0
; Generator: Khronos; 0
; Bound: 100
; Schema: 0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %_uniform_UBO
               OpExecutionMode %main OriginUpperLeft

; --- Decorations ---
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0

; Uniform Buffer Decorations
               OpDecorate %UBO_Struct Block
               OpMemberDecorate %UBO_Struct 0 Offset 0    ; model
               OpMemberDecorate %UBO_Struct 1 Offset 64   ; view
               OpMemberDecorate %UBO_Struct 2 Offset 128  ; projection
               OpMemberDecorate %UBO_Struct 3 Offset 192  ; lightPos
               OpMemberDecorate %UBO_Struct 4 Offset 208  ; cameraPos
               OpMemberDecorate %UBO_Struct 5 Offset 224  ; params
               OpDecorate %_uniform_UBO DescriptorSet 0
               OpDecorate %_uniform_UBO Binding 0

; --- Types ---
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4 ; Column-major by default

; UBO Structure
 %UBO_Struct = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

; Pointers
%ptr_Input_v3 = OpTypePointer Input %v3float
%ptr_Input_v2 = OpTypePointer Input %v2float
%ptr_Input_v4 = OpTypePointer Input %v4float ; for color, though it's vec3, it's safer to read as v4 and shuffle.
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO_Struct
%ptr_Uniform_v4 = OpTypePointer Uniform %v4float

; --- Constants ---
    %float_0 = OpConstant %float 0.0
   %float_0_1 = OpConstant %float 0.1
   %float_0_6 = OpConstant %float 0.6
   %float_0_8 = OpConstant %float 0.8
    %float_1 = OpConstant %float 1.0
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
      %int_5 = OpConstant %int 5
      %int_64 = OpConstant %float 64.0 ; Shininess

; --- Variables ---
   %worldPos = OpVariable %ptr_Input_v3 Input
     %normal = OpVariable %ptr_Input_v3 Input
         %uv = OpVariable %ptr_Input_v2 Input
    %tangent = OpVariable %ptr_Input_v3 Input
       %color = OpVariable %ptr_Input_v3 Input
   %outColor = OpVariable %ptr_Output_v4 Output
%_uniform_UBO = OpVariable %ptr_Uniform_UBO Uniform

; --- Main Function ---
       %main = OpFunction %void None %void
 %main_label = OpLabel

; 1. Load Inputs
  %v3_worldPos = OpLoad %v3float %worldPos
    %v3_normal = OpLoad %v3float %normal
  %v3_in_color = OpLoad %v3float %color

; 2. Load UBO Data
; Access lightPos
 %ptr_lightPos = OpAccessChain %ptr_Uniform_v4 %_uniform_UBO %int_3
 %v4_lightPos = OpLoad %v4float %ptr_lightPos
  %v3_lightPos = OpVectorShuffle %v3float %v4_lightPos %v4_lightPos %int_0 %int_1 %int_2

; Access cameraPos
%ptr_cameraPos = OpAccessChain %ptr_Uniform_v4 %_uniform_UBO %int_4
%v4_cameraPos = OpLoad %v4float %ptr_cameraPos
 %v3_cameraPos = OpVectorShuffle %v3float %v4_cameraPos %v4_cameraPos %int_0 %int_1 %int_2

; 3. Lighting Vectors
; N = normalize(normal)
          %N = OpExtInst %v3float %1 Normalize %v3_normal

; L = normalize(lightPos - worldPos)
      %L_sub = OpFSub %v3float %v3_lightPos %v3_worldPos
          %L = OpExtInst %v3float %1 Normalize %L_sub

; V = normalize(cameraPos - worldPos)
      %V_sub = OpFSub %v3float %v3_cameraPos %v3_worldPos
          %V = OpExtInst %v3float %1 Normalize %V_sub

; H = normalize(L + V)
      %H_add = OpFAdd %v3float %L %V
          %H = OpExtInst %v3float %1 Normalize %H_add

; 4. Diffuse Component: max(dot(N,L),0)
    %N_dot_L = OpDot %float %N %L
    %diffuse = OpExtInst %float %1 FMax %N_dot_L %float_0

; 5. Specular Component: pow(max(dot(N,H),0),64)
    %N_dot_H = OpDot %float %N %H
  %spec_base = OpExtInst %float %1 FMax %N_dot_H %float_0
   %specular = OpExtInst %float %1 Pow %spec_base %int_64

; 6. Grayscale intensity = (0.1 + 0.6*diffuse + 0.8*specular) with alpha 1.0
%term_diffuse = OpFMul %float %float_0_6 %diffuse
%term_specular = OpFMul %float %float_0_8 %specular
    %sum_terms = OpFAdd %float %float_0_1 %term_diffuse
%grayscale_intensity = OpFAdd %float %sum_terms %term_specular

%final_color_v4 = OpCompositeConstruct %v4float %grayscale_intensity %grayscale_intensity %grayscale_intensity %float_1

               OpStore %outColor %final_color_v4
               OpReturn
               OpFunctionEnd
