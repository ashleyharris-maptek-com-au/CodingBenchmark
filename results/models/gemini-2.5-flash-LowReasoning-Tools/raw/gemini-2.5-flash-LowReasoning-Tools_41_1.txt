; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 1
; Capabilities: Shader
; Memory Model: Logical GLSL450
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %normal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %normal "normal"
OpName %outColor "outColor"

; Decorations
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float ; For local normal in main
%_ptr_Function_v4float = OpTypePointer Function %v4float ; For local color in main
%_ptr_Function_float = OpTypePointer Function %float ; For local floats if needed
%_ptr_Private_v3float = OpTypePointer Private %v3float ; For temporary normal storage

%func = OpTypeFunction %void

; Constants
%float_0_5 = OpConstant %float 0.5
%float_1_0 = OpConstant %float 1.0

; Global Variables
%normal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output

; Main function
%main = OpFunction %void None %func
%main_label = OpLabel

; Declare a local variable for the normal to work with
%_normal_local = OpVariable %_ptr_Function_v3float Function

; Load the input normal
%normal_val = OpLoad %v3float %normal

; Normalize the normal
%normalized_normal = OpVectorNormalize %v3float %normal_val

; Map from [-1, 1] to [0, 1]: (normalized_normal * 0.5) + 0.5
%scaled_normal = OpVectorTimesScalar %v3float %normalized_normal %float_0_5
%mapped_normal = OpFAdd %v3float %scaled_normal %float_0_5

; Construct the output color vec4 (r, g, b, 1.0)
%x = OpCompositeExtract %float %mapped_normal 0
%y = OpCompositeExtract %float %mapped_normal 1
%z = OpCompositeExtract %float %mapped_normal 2
%color_out = OpCompositeConstruct %v4float %x %y %z %float_1_0

; Store the result to outColor
OpStore %outColor %color_out

OpReturn
OpFunctionEnd
