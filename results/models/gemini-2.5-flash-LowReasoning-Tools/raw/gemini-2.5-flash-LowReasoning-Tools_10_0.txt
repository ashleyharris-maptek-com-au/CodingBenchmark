#include <iostream>
#include <vector>
#include <algorithm>
#include <tuple> // For std::tie
#include <limits> // For std::numeric_limits

// Represents a rectangle
struct Rect {
    int x, y, width, height;

    Rect(int x = 0, int y = 0, int width = 0, int height = 0)
        : x(x), y(y), width(width), height(height) {}

    int area() const { return width * height; }

    bool intersects(const Rect& other) const {
        return x < other.x + other.width &&
               x + width > other.x &&
               y < other.y + other.height &&
               y + height > other.y;
    }
     // Checks if this rectangle contains another rectangle
    bool contains(const Rect& other) const {
        return x <= other.x &&
               y <= other.y &&
               x + width >= other.x + other.width &&
               y + height >= other.y + other.height;
    }
};

// Represents an item to be packed
struct Item {
    int id;
    int originalWidth, originalHeight; // Store original dimensions
    int currentWidth, currentHeight; // Store current dimensions (might be rotated)
    int x, y;
    bool rotated;
    int binIndex; // New member to store the bin index

    Item(int id, int width, int height)
        : id(id), originalWidth(width), originalHeight(height),
          currentWidth(width), currentHeight(height),
          x(0), y(0), rotated(false), binIndex(-1) {} // Initialize binIndex to -1

    int area() const { return originalWidth * originalHeight; }

    // Helper to try rotating the item
    void rotate() {
        std::swap(currentWidth, currentHeight);
        rotated = !rotated;
    }

    // Reset to original orientation
    void resetRotation() {
        if (rotated) {
            std::swap(currentWidth, currentHeight);
            rotated = false;
        }
    }
};

// MaxRectsBinPack algorithm implementation for a single bin
class MaxRectsBinPack {
public:
    MaxRectsBinPack(int binWidth, int binHeight)
        : binWidth_(binWidth), binHeight_(binHeight) {
        freeRectangles_.push_back(Rect(0, 0, binWidth, binHeight));
    }

    // Tries to insert a rectangle into the bin using a best-fit heuristic
    // Returns the placed rectangle if successful, otherwise an empty Rect
    Rect Insert(int width, int height, bool canRotate, Item& item) {
        int bestRectIndex = -1;
        int bestScore1 = std::numeric_limits<int>::max(); // e.g., min short side fit
        int bestScore2 = std::numeric_limits<int>::max(); // e.g., min long side fit (tie-breaker)

        Rect bestRect;
        bool rotated = false;

        for (size_t i = 0; i < freeRectangles_.size(); ++i) {
            const Rect& freeRect = freeRectangles_[i];

            // Try non-rotated
            if (freeRect.width >= width && freeRect.height >= height) {
                int score1 = std::min(freeRect.width - width, freeRect.height - height); // Short side fit
                int score2 = std::max(freeRect.width - width, freeRect.height - height); // Long side fit
                
                if (score1 < bestScore1 || (score1 == bestScore1 && score2 < bestScore2)) {
                    bestScore1 = score1;
                    bestScore2 = score2;
                    bestRectIndex = i;
                    bestRect = Rect(freeRect.x, freeRect.y, width, height);
                    rotated = false;
                }
            }

            // Try rotated
            if (canRotate && width != height && freeRect.width >= height && freeRect.height >= width) {
                int score1 = std::min(freeRect.width - height, freeRect.height - width); // Short side fit
                int score2 = std::max(freeRect.width - height, freeRect.height - width); // Long side fit

                if (score1 < bestScore1 || (score1 == bestScore1 && score2 < bestScore2)) {
                    bestScore1 = score1;
                    bestScore2 = score2;
                    bestRectIndex = i;
                    bestRect = Rect(freeRect.x, freeRect.y, height, width); // Note: width and height swapped
                    rotated = true;
                }
            }
        }

        if (bestRectIndex != -1) {
            Rect freeRect = freeRectangles_[bestRectIndex];
            PlaceRect(bestRect, freeRect);
            item.x = bestRect.x;
            item.y = bestRect.y;
            item.rotated = rotated;
            if (rotated) {
                std::swap(item.currentWidth, item.currentHeight);
            }
            return bestRect;
        }

        return Rect(); // No fit found
    }

private:
    int binWidth_;
    int binHeight_;
    std::vector<Rect> freeRectangles_;

    void PlaceRect(const Rect& placedRect, Rect& freeRect) {
        // Remove the free rectangle that was used
        auto it = std::find_if(freeRectangles_.begin(), freeRectangles_.end(), 
                               [&](const Rect& r){ return r.x == freeRect.x && r.y == freeRect.y && r.width == freeRect.width && r.height == freeRect.height; });
        if (it != freeRectangles_.end()) {
            freeRectangles_.erase(it);
        }

        // Split the free rectangle into at most two new free rectangles (L-shaped split).
        // This is a more robust splitting strategy for MaxRectsBinPack.

        // Create a new free rectangle for the space above the placed rectangle
        if (freeRect.y + freeRect.height > placedRect.y + placedRect.height) {
            Rect newRect(freeRect.x, placedRect.y + placedRect.height,
                         freeRect.width, freeRect.y + freeRect.height - (placedRect.y + placedRect.height));
            freeRectangles_.push_back(newRect);
        }

        // Create a new free rectangle for the space to the right of the placed rectangle
        // This rectangle's height is limited to the placed rectangle's height.
        if (freeRect.x + freeRect.width > placedRect.x + placedRect.width) {
            Rect newRect(placedRect.x + placedRect.width, placedRect.y,
                         freeRect.x + freeRect.width - (placedRect.x + placedRect.width), placedRect.height);
            freeRectangles_.push_back(newRect);
        }
        
        PruneFreeRectangles();
    }

    void PruneFreeRectangles() {
        // Remove free rectangles that are contained within others.
        // This loop is O(N^2) but typically N is small.
        for (size_t i = 0; i < freeRectangles_.size(); ++i) {
            for (size_t j = i + 1; j < freeRectangles_.size(); ++j) {
                if (freeRectangles_[j].contains(freeRectangles_[i])) {
                    freeRectangles_.erase(freeRectangles_.begin() + i);
                    --i;
                    break; 
                }
                if (freeRectangles_[i].contains(freeRectangles_[j])) {
                    freeRectangles_.erase(freeRectangles_.begin() + j);
                    --j;
                }
            }
        }
        // After pruning, sort the free rectangles to keep them somewhat organized.
        // This might help with cache locality or certain heuristics.
        std::sort(freeRectangles_.begin(), freeRectangles_.end(), [](const Rect& a, const Rect& b) {
            if (a.y != b.y) return a.y < b.y;
            return a.x < b.x;
        });
    }
};

// Main solver class
class Solver {
public:
    Solver(int boardWidth, int boardHeight)
        : boardWidth_(boardWidth), boardHeight_(boardHeight) {}

    void solve(std::vector<Item>& items) {
        // Sort items by a heuristic (e.g., decreasing original area)
        std::sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
            return a.area() > b.area();
        });



        // Map to store the final placement for each original item ID
        // This is necessary because items vector is sorted, and we need to output by original ID.
        std::vector<Item> finalPlacedItems(items.size());

        for (auto& item : items) {
            bool placed = false;
            int currentBinIndex = -1;

            // Try to place in existing bins
            for (size_t i = 0; i < bins.size(); ++i) {
                MaxRectsBinPack& currentBin = bins[i];
                Item tempItem = item; // Create a temporary item for insertion attempt
                
                Rect placedRect = currentBin.Insert(tempItem.currentWidth, tempItem.currentHeight, true, tempItem);
                
                if (placedRect.area() > 0) {
                    // Placement successful, update original item and break
                    item.x = placedRect.x;
                    item.y = placedRect.y;
                    item.rotated = tempItem.rotated;
                    item.currentWidth = tempItem.currentWidth;
                    item.currentHeight = tempItem.currentHeight;

                    item.binIndex = i; // Store the bin index
                    finalPlacedItems[item.id] = item; // Store by original ID
                    placed = true;
                    // currentBinIndex = i; // This variable is no longer strictly needed
                    break;
                }
            }

            // If not placed, create a new bin
            if (!placed) {
                bins.emplace_back(boardWidth_, boardHeight_);
                MaxRectsBinPack& newBin = bins.back();
                
                Item tempItem = item; // Create a temporary item for insertion attempt
                Rect placedRect = newBin.Insert(tempItem.currentWidth, tempItem.currentHeight, true, tempItem);

                if (placedRect.area() > 0) {
                    item.x = placedRect.x;
                    item.y = placedRect.y;
                    item.rotated = tempItem.rotated;
                    item.currentWidth = tempItem.currentWidth;
                    item.currentHeight = tempItem.currentHeight;
                    item.binIndex = bins.size() - 1; // The index of the newly created bin

                    finalPlacedItems[item.id] = item; // Store by original ID
                } else {
                    // This should ideally not happen if bin dimensions are larger than item
                    // For robustness, handle error or skip item
                    std::cerr << "Error: Could not place item " << item.id << " even in a new bin." << std::endl;
                }
            }
        }

        // Output results
        std::cout << bins.size() << std::endl;
        for (const auto& placedItem : finalPlacedItems) {
            std::cout << placedItem.binIndex << " " << placedItem.x << " " << placedItem.y << " " << placedItem.rotated << std::endl;
        }
    }

private:
    int boardWidth_;
    int boardHeight_;
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int numRectangles, boardWidth, boardHeight;
    std::cin >> numRectangles >> boardWidth >> boardHeight;

    std::vector<Item> items;
    items.reserve(numRectangles);
    for (int i = 0; i < numRectangles; ++i) {
        int w, h;
        std::cin >> w >> h;
        items.emplace_back(i, w, h);
    }

    Solver solver(boardWidth, boardHeight);
    solver.solve(items);

    return 0;
}
