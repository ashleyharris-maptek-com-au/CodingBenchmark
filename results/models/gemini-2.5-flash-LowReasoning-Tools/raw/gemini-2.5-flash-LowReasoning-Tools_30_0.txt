def solve_dpll(clauses, assignment):
    """
    DPLL algorithm to solve the 3-SAT problem.

    Args:
        clauses: A list of clauses, where each clause is a list of literals.
                 Literals are represented as integers (positive for variable, negative for negation).
        assignment: A dictionary mapping variables (integers) to their boolean assignments (True/False).

    Returns:
        A tuple (bool, assignment) indicating satisfiability and the satisfying assignment,
        or (False, None) if unsatisfiable.
    """
    
    # Apply simplifications before checking base cases for current iteration
    # 2. Unit Propagation
    propagated_clauses, propagated_assignment = unit_propagate(clauses, assignment)
    if propagated_clauses is None:  # Conflict detected during unit propagation
        return False, None
    clauses, assignment = propagated_clauses, propagated_assignment

    # 3. Pure Literal Elimination
    pure_clauses, pure_assignment = pure_literal_eliminate(clauses, assignment)
    clauses, assignment = pure_clauses, pure_assignment

    # 1. Base Cases
    if not clauses:
        return True, assignment

    for clause in clauses:
        if not clause:
            return False, None

    # 4. Branching
    literal_to_branch = choose_literal(clauses, assignment)
    if literal_to_branch is None:
        # No more unassigned literals, but clauses are not empty. This means some clauses
        # are not satisfied by current assignments, but no conflicts either.
        # This case should ideally be covered by unit propagation and base cases,
        # but as a fallback, it means unsatisfiable if no empty clauses.
        return False, None

    # Try assigning the literal to True
    new_assignment_true = assignment.copy()
    new_assignment_true[abs(literal_to_branch)] = True
    # Create a new set of clauses that considers the new assignment
    clauses_if_true = simplify_clauses(clauses, literal_to_branch, True)
    
    satisfied, result_assignment = solve_dpll(clauses_if_true, new_assignment_true)
    if satisfied:
        return True, result_assignment

    # Try assigning the literal to False
    new_assignment_false = assignment.copy()
    new_assignment_false[abs(literal_to_branch)] = False
    # Create a new set of clauses that considers the new assignment
    clauses_if_false = simplify_clauses(clauses, literal_to_branch, False)

    return solve_dpll(clauses_if_false, new_assignment_false)


def simplify_clauses(clauses, literal, value):
    """
    Simplifies clauses based on a given literal assignment.

    Args:
        clauses: The current list of clauses.
        literal: The literal (variable) being assigned.
        value: The boolean value (True/False) assigned to the literal.

    Returns:
        A new list of clauses after simplification.
    """
    simplified = []
    assigned_literal = literal if value else -literal
    negated_assigned_literal = -assigned_literal

    for clause in clauses:
        if assigned_literal in clause:
            # Clause is satisfied, remove it
            continue
        if negated_assigned_literal in clause:
            # Remove the negated literal from the clause
            new_clause = [lit for lit in clause if lit != negated_assigned_literal]
            simplified.append(new_clause)
        else:
            simplified.append(clause)
    return simplified

def unit_propagate(clauses, assignment):
    """
    Performs unit propagation on the clauses.

    Args:
        clauses: Current list of clauses.
        assignment: Current variable assignments.

    Returns:
        A tuple (updated_clauses, updated_assignment) or (None, None) if a conflict is found.
    """
    new_clauses = [list(clause) for clause in clauses]  # Create a mutable copy
    new_assignment = assignment.copy()

    while True:
        unit_clause_found = False
        unit_literal = None

        # Find a unit clause
        for clause in new_clauses:
            unassigned_literals = []
            for literal in clause:
                var = abs(literal)
                if var not in new_assignment:
                    unassigned_literals.append(literal)
            
            if len(unassigned_literals) == 1:
                unit_literal = unassigned_literals[0]
                unit_clause_found = True
                break
        
        if not unit_clause_found:
            break  # No more unit clauses

        # Assign the unit literal
        var = abs(unit_literal)
        value = True if unit_literal > 0 else False
        new_assignment[var] = value

        # Update clauses
        updated_clauses_list = []
        conflict = False
        for clause in new_clauses:
            if unit_literal in clause:
                # Clause is satisfied, remove it
                continue
            
            # Remove the negation of the unit literal
            negated_unit_literal = -unit_literal
            if negated_unit_literal in clause:
                new_clause = [lit for lit in clause if lit != negated_unit_literal]
                if not new_clause:
                    # Conflict: clause becomes empty
                    conflict = True
                    break
                updated_clauses_list.append(new_clause)
            else:
                updated_clauses_list.append(clause)
        
        if conflict:
            return None, None # Return conflict
        
        new_clauses = updated_clauses_list
    
    return new_clauses, new_assignment

def pure_literal_eliminate(clauses, assignment):
    """
    Performs pure literal elimination on the clauses.

    Args:
        clauses: Current list of clauses.
        assignment: Current variable assignments.

    Returns:
        A tuple (updated_clauses, updated_assignment)
    """
    new_clauses = [list(clause) for clause in clauses] # Create a mutable copy
    new_assignment = assignment.copy()

    while True:
        pure_literal_found = False
        
        # Collect all literals and their polarities
        literals_in_clauses = {} # var -> {positive: bool, negative: bool}
        for clause in new_clauses:
            for literal in clause:
                var = abs(literal)
                if var not in new_assignment: # Only consider unassigned variables
                    if var not in literals_in_clauses:
                        literals_in_clauses[var] = {'positive': False, 'negative': False}
                    
                    if literal > 0:
                        literals_in_clauses[var]['positive'] = True
                    else:
                        literals_in_clauses[var]['negative'] = True
        
        pure_literal = None
        for var, polarities in literals_in_clauses.items():
            if polarities['positive'] and not polarities['negative']:
                pure_literal = var # Pure positive literal
                pure_literal_found = True
                break
            elif polarities['negative'] and not polarities['positive']:
                pure_literal = -var # Pure negative literal
                pure_literal_found = True
                break
        
        if not pure_literal_found:
            break # No more pure literals

        # Assign the pure literal
        var = abs(pure_literal)
        value = True if pure_literal > 0 else False
        new_assignment[var] = value

        # Remove clauses containing the pure literal
        updated_clauses_list = []
        for clause in new_clauses:
            if pure_literal not in clause:
                updated_clauses_list.append(clause)
        
        new_clauses = updated_clauses_list
    
    return new_clauses, new_assignment

def choose_literal(clauses, assignment):
    """
    Chooses an unassigned literal for branching.

    Args:
        clauses: Current list of clauses.
        assignment: Current variable assignments.

    Returns:
        An integer representing the chosen literal (positive for variable, negative for negation),
        or None if no unassigned literals are found.
    """
    for clause in clauses:
        for literal in clause:
            var = abs(literal)
            if var not in assignment:
                return var # Return the variable itself (positive literal for branching)
    return None # No unassigned literals found
