using NUnit.Framework;
using System.Collections.Generic;
using System.Linq;
using System;

[TestFixture]
public class BinPackerTests
{
    private BinPacker _packer;

    [SetUp]
    public void Setup()
    {
        _packer = new BinPacker();
    }

    [Test]
    public void Pack_SingleBoxPerfectFit_ReturnsAllPacked()
    {
        var bin = new Bin(10, 10, 10);
        var box = new Box("Box1", 10, 10, 10);
        var boxesToPack = new List<Box> { box };

        var result = _packer.Pack(bin, boxesToPack);

        Assert.That(result.AllBoxesPacked, Is.True);
        Assert.That(result.PackedBoxes.Count, Is.EqualTo(1));
        Assert.That(result.PackedBoxes[0].Id, Is.EqualTo("Box1"));
        Assert.That(result.PackedBoxes[0].X, Is.EqualTo(0));
        Assert.That(result.PackedBoxes[0].Y, Is.EqualTo(0));
        Assert.That(result.PackedBoxes[0].Z, Is.EqualTo(0));
        Assert.That(result.UnpackedBoxes, Is.Empty);
        Assert.That(result.UtilizedVolumePercentage, Is.EqualTo(100.0).Within(0.001));
    }

    [Test]
    public void Pack_MultipleBoxesPerfectFit_ReturnsAllPacked()
    {
        var bin = new Bin(20, 10, 10);
        var boxesToPack = new List<Box>
        {
            new Box("BoxA", 10, 10, 10),
            new Box("BoxB", 10, 10, 10)
        };

        var result = _packer.Pack(bin, boxesToPack);

        Assert.That(result.AllBoxesPacked, Is.True);
        Assert.That(result.PackedBoxes.Count, Is.EqualTo(2));
        Assert.That(result.UnpackedBoxes, Is.Empty);
        Assert.That(result.UtilizedVolumePercentage, Is.EqualTo(100.0).Within(0.001));

        // Verify positions (order might vary based on sorting and space splitting)
        var boxA = result.PackedBoxes.First(b => b.Id == "BoxA");
        var boxB = result.PackedBoxes.First(b => b.Id == "BoxB");

        // Assuming a simple packing along X-axis
        Assert.That(boxA.X == 0 && boxB.X == 10 || boxA.X == 10 && boxB.X == 0, "Boxes should be side-by-side along X");
        Assert.That(boxA.Y, Is.EqualTo(0));
        Assert.That(boxA.Z, Is.EqualTo(0));
        Assert.That(boxB.Y, Is.EqualTo(0));
        Assert.That(boxB.Z, Is.EqualTo(0));
    }

    [Test]
    public void Pack_BoxTooLargeForBin_ReturnsUnpackedBox()
    {
        var bin = new Bin(10, 10, 10);
        var box = new Box("LargeBox", 11, 10, 10); // Too long
        var boxesToPack = new List<Box> { box };

        var result = _packer.Pack(bin, boxesToPack);

        Assert.That(result.AllBoxesPacked, Is.False);
        Assert.That(result.PackedBoxes, Is.Empty);
        Assert.That(result.UnpackedBoxes.Count, Is.EqualTo(1));
        Assert.That(result.UnpackedBoxes[0].Id, Is.EqualTo("LargeBox"));
        Assert.That(result.UtilizedVolumePercentage, Is.EqualTo(0.0).Within(0.001));
    }

    [Test]
    public void Pack_SomeBoxesFitOthersDont_ReturnsPartialPacking()
    {
        var bin = new Bin(10, 10, 10);
        var boxesToPack = new List<Box>
        {
            new Box("SmallBox", 5, 5, 5),
            new Box("MediumBox", 6, 6, 6), // Should fit
            new Box("LargeBox", 11, 1, 1)  // Too large
        };

        var result = _packer.Pack(bin, boxesToPack);

        Assert.That(result.AllBoxesPacked, Is.False);
        Assert.That(result.PackedBoxes.Count, Is.EqualTo(2));
        Assert.That(result.UnpackedBoxes.Count, Is.EqualTo(1));
        Assert.That(result.UnpackedBoxes.Any(b => b.Id == "LargeBox"), Is.True);
        Assert.That(result.PackedBoxes.Any(b => b.Id == "SmallBox"), Is.True);
        Assert.That(result.PackedBoxes.Any(b => b.Id == "MediumBox"), Is.True);
    }
    
    // Test SplitSpace method directly using reflection to access private method
    // This is generally discouraged but useful for testing complex private logic.
    // For a production system, consider making this method internal and visible to the test assembly.
    private List<FreeSpace> CallSplitSpace(FreeSpace originalSpace, Box placedBox)
    {
        var method = typeof(BinPacker).GetMethod("SplitSpace", System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance);
        return (List<FreeSpace>)method.Invoke(_packer, new object[] { originalSpace, placedBox });
    }

    [Test]
    public void SplitSpace_BoxInCorner_GeneratesCorrectSpaces()
    {
        var originalSpace = new FreeSpace(0, 0, 0, 10, 10, 10);
        var placedBox = new Box("CornerBox", 5, 5, 5) { X = 0, Y = 0, Z = 0 };

        var newSpaces = CallSplitSpace(originalSpace, placedBox);

        Assert.That(newSpaces.Count, Is.EqualTo(3)); // Expect 3 spaces: right, above, in front

        // Space to the right (X-split)
        var spaceX = newSpaces.FirstOrDefault(s => s.X == 5 && s.Y == 0 && s.Z == 0);
        Assert.That(spaceX, Is.Not.Null);
        Assert.That(spaceX.Length, Is.EqualTo(5));
        Assert.That(spaceX.Width, Is.EqualTo(10));
        Assert.That(spaceX.Height, Is.EqualTo(10));

        // Space above (Y-split)
        var spaceY = newSpaces.FirstOrDefault(s => s.X == 0 && s.Y == 5 && s.Z == 0);
        Assert.That(spaceY, Is.Not.Null);
        Assert.That(spaceY.Length, Is.EqualTo(5));
        Assert.That(spaceY.Width, Is.EqualTo(5));
        Assert.That(spaceY.Height, Is.EqualTo(10));
        
        // Space in front (Z-split)
        var spaceZ = newSpaces.FirstOrDefault(s => s.X == 0 && s.Y == 0 && s.Z == 5);
        Assert.That(spaceZ, Is.Not.Null);
        Assert.That(spaceZ.Length, Is.EqualTo(5));
        Assert.That(spaceZ.Width, Is.EqualTo(5));
        Assert.That(spaceZ.Height, Is.EqualTo(5));
    }
    
    [Test]
    public void SplitSpace_BoxInMiddle_GeneratesCorrectSpaces()
    {
        var originalSpace = new FreeSpace(0, 0, 0, 10, 10, 10);
        var placedBox = new Box("MidBox", 4, 4, 4) { X = 3, Y = 3, Z = 3 };

        var newSpaces = CallSplitSpace(originalSpace, placedBox);

        // Expect 6 spaces: left, right, below, above, behind, in front
        // The implementation creates 6 spaces. Need to verify each.
        Assert.That(newSpaces.Count, Is.EqualTo(6)); 
        
        // Assertions for each of the 6 spaces
        // Space 1: "left" of the placed box (along X-axis)
        var space1 = newSpaces.FirstOrDefault(s => s.X == 0 && s.Y == 0 && s.Z == 0);
        Assert.That(space1.Length, Is.EqualTo(3));
        Assert.That(space1.Width, Is.EqualTo(10));
        Assert.That(space1.Height, Is.EqualTo(10));

        // Space 2: "right" of the placed box (along X-axis)
        var space2 = newSpaces.FirstOrDefault(s => s.X == 7 && s.Y == 0 && s.Z == 0);
        Assert.That(space2.Length, Is.EqualTo(3));
        Assert.That(space2.Width, Is.EqualTo(10));
        Assert.That(space2.Height, Is.EqualTo(10));

        // Space 3: "below" the placed box (along Y-axis, within the X-range of the box)
        var space3 = newSpaces.FirstOrDefault(s => s.X == 3 && s.Y == 0 && s.Z == 0);
        Assert.That(space3.Length, Is.EqualTo(4));
        Assert.That(space3.Width, Is.EqualTo(3));
        Assert.That(space3.Height, Is.EqualTo(10));

        // Space 4: "above" the placed box (along Y-axis, within the X-range of the box)
        var space4 = newSpaces.FirstOrDefault(s => s.X == 3 && s.Y == 7 && s.Z == 0);
        Assert.That(space4.Length, Is.EqualTo(4));
        Assert.That(space4.Width, Is.EqualTo(3));
        Assert.That(space4.Height, Is.EqualTo(10));

        // Space 5: "behind" the placed box (along Z-axis, within the X and Y ranges of the box)
        var space5 = newSpaces.FirstOrDefault(s => s.X == 3 && s.Y == 3 && s.Z == 0);
        Assert.That(space5.Length, Is.EqualTo(4));
        Assert.That(space5.Width, Is.EqualTo(4));
        Assert.That(space5.Height, Is.EqualTo(3));

        // Space 6: "in front" of the placed box (along Z-axis, within the X and Y ranges of the box)
        var space6 = newSpaces.FirstOrDefault(s => s.X == 3 && s.Y == 3 && s.Z == 7);
        Assert.That(space6.Length, Is.EqualTo(4));
        Assert.That(space6.Width, Is.EqualTo(4));
        Assert.That(space6.Height, Is.EqualTo(3));
    }
}
