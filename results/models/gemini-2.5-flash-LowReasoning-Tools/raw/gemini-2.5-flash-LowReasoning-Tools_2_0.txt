using System;
using System.Collections.Generic;
using System.Linq;

namespace ChinesePostman
{
    public class Edge
    {
        public int U { get; } // Start vertex
        public int V { get; } // End vertex
        public int Weight { get; }
        public int Id { get; } // Unique identifier for the edge

        public Edge(int u, int v, int weight, int id)
        {
            U = u;
            V = v;
            Weight = weight;
            Id = id;
        }

        public override string ToString()
        {
            return $"(U: {U}, V: {V}, W: {Weight})";
        }
    }

    public class Graph
    {
        private Dictionary<int, List<Edge>> _adj;
        public int NumVertices { get; private set; }
        public List<Edge> AllEdges { get; private set; }

        public Graph(IEnumerable<Edge> edges)
        {
            _adj = new Dictionary<int, List<Edge>>();
            AllEdges = new List<Edge>(edges);

            foreach (var edge in AllEdges)
            {
                if (!_adj.ContainsKey(edge.U)) _adj[edge.U] = new List<Edge>();
                if (!_adj.ContainsKey(edge.V)) _adj[edge.V] = new List<Edge>();

                _adj[edge.U].Add(edge);
                // For an undirected graph, add the edge to both vertices
                // Create a 'reverse' edge for representation, but conceptually it's the same edge
                _adj[edge.V].Add(new Edge(edge.V, edge.U, edge.Weight, edge.Id)); // Note: Same ID
            }
            NumVertices = _adj.Keys.Count;
        }

        public List<Edge> GetEdges(int vertex)
        {
            return _adj.TryGetValue(vertex, out var edges) ? edges : new List<Edge>();
        }

        public Dictionary<int, int> GetVertexDegrees()
        {
            var degrees = new Dictionary<int, int>();
            foreach (var vertex in _adj.Keys)
            {
                // Each edge in the adjacency list counts as one "connection"
                // For undirected graph, we count how many times a vertex appears as U or V in AllEdges
                // Or simply count the entries in its adjacency list for the full adjacency list.
                // Since we added (V,U) for each (U,V), the count in _adj[vertex] is already the degree.
                degrees[vertex] = _adj[vertex].Count;
            }
            return degrees;
        }

        public List<int> GetOddDegreeVertices()
        {
            var degrees = GetVertexDegrees();
            return degrees.Where(pair => pair.Value % 2 != 0).Select(pair => pair.Key).ToList();
        }

        // Method to add new edges (duplicates) to make degrees even
        public void AddEdges(IEnumerable<Edge> edgesToAdd)
        {
            foreach (var edge in edgesToAdd)
            {
                AllEdges.Add(edge);
                if (!_adj.ContainsKey(edge.U)) _adj[edge.U] = new List<Edge>();
                if (!_adj.ContainsKey(edge.V)) _adj[edge.V] = new List<Edge>();

                _adj[edge.U].Add(edge);
                _adj[edge.V].Add(new Edge(edge.V, edge.U, edge.Weight, edge.Id));
            }
            // Update NumVertices if new vertices are added, though unlikely in this scenario
            NumVertices = _adj.Keys.Count;
        }
    }
}
