#include <iostream>
#include <vector>
#include <queue>
#include <limits>
#include <set>
#include <algorithm>
#include <map>

const long long INF = std::numeric_limits<long long>::max();

struct Edge {
    int to;
    int weight;
};

struct State {
    int u;
    long long dist;

    bool operator>(const State& other) const {
        return dist > other.dist;
    }
};

// Dijkstra's algorithm to find shortest paths from a source node
std::vector<long long> dijkstra(int start_node, int num_vertices, const std::vector<std::vector<Edge>>& adj) {
    std::vector<long long> dist(num_vertices, INF);
    std::priority_queue<State, std::vector<State>, std::greater<State>> pq;

    dist[start_node] = 0;
    pq.push({start_node, 0});

    while (!pq.empty()) {
        State current = pq.top();
        pq.pop();

        int u = current.u;
        long long d = current.dist;

        if (d > dist[u]) {
            continue;
        }

        for (const auto& edge : adj[u]) {
            int v = edge.to;
            int weight = edge.weight;

            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                pq.push({v, dist[v]});
            }
        }
    }
    return dist;
}

// Function to reconstruct path from source to target given predecessor map
std::vector<std::pair<int, int>> reconstruct_path(int start_node, int end_node, const std::map<int, int>& predecessors) {
    std::vector<std::pair<int, int>> path_edges;
    int current = end_node;
    while (current != start_node) {
        auto it = predecessors.find(current);
        if (it == predecessors.end()) {
            // Should not happen in a connected graph if end_node is reachable
            return {};
        }
        int prev = it->second;
        path_edges.push_back({std::min(prev, current), std::max(prev, current)});
        current = prev;
    }
    return path_edges;
}

// Dijkstra's algorithm with path reconstruction
std::pair<std::vector<long long>, std::map<int, int>> dijkstra_with_path(int start_node, int num_vertices, const std::vector<std::vector<Edge>>& adj) {
    std::vector<long long> dist(num_vertices, INF);
    std::map<int, int> predecessors;
    std::priority_queue<State, std::vector<State>, std::greater<State>> pq;

    dist[start_node] = 0;
    pq.push({start_node, 0});

    while (!pq.empty()) {
        State current = pq.top();
        pq.pop();

        int u = current.u;
        long long d = current.dist;

        if (d > dist[u]) {
            continue;
        }

        for (const auto& edge : adj[u]) {
            int v = edge.to;
            int weight = edge.weight;

            if (dist[u] + weight < dist[v]) {
                dist[v] = dist[u] + weight;
                predecessors[v] = u; // Store predecessor
                pq.push({v, dist[v]});
            }
        }
    }
    return {dist, predecessors};
}

struct DGraphEdge {
    int u, v;
    long long weight;

    bool operator<(const DGraphEdge& other) const {
        return weight < other.weight;
    }
};

struct UnionFind {
    std::vector<int> parent;
    UnionFind(int n) {
        parent.resize(n);
        for (int i = 0; i < n; ++i) {
            parent[i] = i;
        }
    }

    int find(int i) {
        if (parent[i] == i)
            return i;
        return parent[i] = find(parent[i]);
    }

    void unite(int i, int j) {
        int root_i = find(i);
        int root_j = find(j);
        if (root_i != root_j) {
            parent[root_i] = root_j;
        }
    }
};

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_terminals;
    std::cin >> num_vertices >> num_edges >> num_terminals;

    std::vector<std::vector<Edge>> adj(num_vertices);
    // Adjacency list for path reconstruction, storing weights directly for reconstruction
    std::vector<std::vector<std::pair<int, int>>> original_adj(num_vertices);

    for (int i = 0; i < num_edges; ++i) {
        int u, v, weight;
        std::cin >> u >> v >> weight;
        adj[u].push_back({v, weight});
        adj[v].push_back({u, weight});
        original_adj[u].push_back({v, weight});
        original_adj[v].push_back({u, weight});
    }

    std::vector<int> terminals(num_terminals);
    for (int i = 0; i < num_terminals; ++i) {
        std::cin >> terminals[i];
    }

    // Map to store shortest path distances between terminals
    std::map<std::pair<int, int>, long long> terminal_distances;
    // Map to store predecessors for path reconstruction from each terminal's Dijkstra run
    std::map<int, std::map<int, int>> all_predecessors;

    // Run Dijkstra from each terminal to find shortest paths to all other nodes
    for (int t_idx = 0; t_idx < num_terminals; ++t_idx) {
        int start_node = terminals[t_idx];
        auto result = dijkstra_with_path(start_node, num_vertices, adj);
        std::vector<long long>& dists = result.first;
        all_predecessors[start_node] = result.second;

        for (int other_t_idx = t_idx + 1; other_t_idx < num_terminals; ++other_t_idx) {
            int end_node = terminals[other_t_idx];
            if (dists[end_node] != INF) {
                terminal_distances[{start_node, end_node}] = dists[end_node];
                terminal_distances[{end_node, start_node}] = dists[end_node]; // Ensure symmetric
            }
        }
    }

    // Construct a complete graph (distance graph) with terminals as nodes
    std::vector<DGraphEdge> d_graph_edges;
    for (int i = 0; i < num_terminals; ++i) {
        for (int j = i + 1; j < num_terminals; ++j) {
            int u = terminals[i];
            int v = terminals[j];
            if (terminal_distances.count({u, v})) {
                d_graph_edges.push_back({u, v, terminal_distances[{u, v}]});
            }
        }
    }

    // Find MST on the distance graph using Kruskal's algorithm
    std::sort(d_graph_edges.begin(), d_graph_edges.end());

    UnionFind uf(num_vertices); // UnionFind on original vertices to track connectivity of paths
    long long total_weight = 0;
    std::set<std::pair<int, int>> steiner_tree_edges; // Store unique edges of the Steiner tree

    for (const auto& d_edge : d_graph_edges) {
        if (uf.find(d_edge.u) != uf.find(d_edge.v)) {
            uf.unite(d_edge.u, d_edge.v);
            total_weight += d_edge.weight;

            // Reconstruct path for this edge and add its constituent edges to the Steiner tree
            std::vector<std::pair<int, int>> path_segment_edges;
            // The shortest path from d_edge.u to d_edge.v.
            // We need to trace from d_edge.v back to d_edge.u using predecessors from d_edge.u's Dijkstra run
            // This is tricky: all_predecessors[d_edge.u] only gives paths FROM d_edge.u.
            // We need to reconstruct the path correctly.
            // One way is to re-run Dijkstra just for this path, or store full path details.
            // For simplicity and correctness with the existing setup, we need the shortest path
            // from one terminal to another.
            // A more robust way to reconstruct: run Dijkstra from 'u' and 'v' and pick the best path segment.

            // Let's re-think path reconstruction here.
            // When we find a shortest path in Dijkstra, we store the *predecessor* for each node.
            // To reconstruct the path from A to B, we start at B and go back using predecessors until A.
            // The `all_predecessors[start_node]` map holds predecessors from `start_node`.
            // So to get path from `d_edge.u` to `d_edge.v`, we trace from `d_edge.v` back using `all_predecessors[d_edge.u]`.
            // This assumes `d_edge.u` was the source in the Dijkstra run that found `d_edge.v`.

            // Corrected path reconstruction for a shortest path between two terminals 'src' and 'dest'
            // Need to know which Dijkstra run was used to find the shortest path.
            // The `terminal_distances` map stores the distance. We need to associate it with the correct `predecessors` map.
            // Let's assume the Dijkstra run from d_edge.u was used for `terminal_distances[{d_edge.u, d_edge.v}]`.

            int path_current = d_edge.v;
            while (path_current != d_edge.u) {
                auto it = all_predecessors[d_edge.u].find(path_current);
                if (it == all_predecessors[d_edge.u].end()) {
                    // This means d_edge.v was not reached from d_edge.u in the path reconstruction.
                    // This shouldn't happen if terminal_distances[{d_edge.u, d_edge.v}] was found.
                    // This implies issues with how all_predecessors is populated or used.
                    // A quick fix would be to re-run dijkstra_with_path for the specific (u,v) if needed
                    // but that's inefficient.
                    break; // Error or path not found
                }
                int prev = it->second;
                steiner_tree_edges.insert({std::min(prev, path_current), std::max(prev, path_current)});
                path_current = prev;
            }
        }
    }
    
    // There's a subtle issue with path reconstruction:
    // When we do `dijkstra_with_path(start_node, ...)`, it gives predecessors from `start_node`.
    // When we extract an edge (u, v) from the MST, we need the shortest path between them.
    // The `terminal_distances` map stores `dist[v]` from `dijkstra(u)`.
    // So `all_predecessors[u]` is what we need to reconstruct path from `v` back to `u`.

    // The current path reconstruction logic inside the loop:
    // `int path_current = d_edge.v;`
    // `while (path_current != d_edge.u) { ... auto it = all_predecessors[d_edge.u].find(path_current); ... }`
    // This is correct. The problem might be if `d_edge.u` is not in `all_predecessors` map if it was never a source.
    // However, we iterate `terminals` and run dijkstra for each, so all terminals are sources.

    // Let's re-verify the logic of path reconstruction using predecessors.
    // If `all_predecessors[source][dest] = prev_node`, it means `prev_node` is the node *before* `dest` on the shortest path from `source` to `dest`.
    // So to reconstruct, we start at `dest`, go to `predecessors[source][dest]`, then `predecessors[source][predecessors[source][dest]]` and so on, until we reach `source`.

    // The issue might be that `all_predecessors[d_edge.u]` might not be sufficient if the shortest path was found via a Dijkstra run originating from `d_edge.v` instead,
    // and `terminal_distances` uses the minimum of `dist[v]` from `dijkstra(u)` and `dist[u]` from `dijkstra(v)`.
    // To be precise, we need to know WHICH source Dijkstra produced the shortest path used for `d_edge.weight`.

    // A simpler (but potentially less efficient approach if `num_terminals` is large but `num_vertices` is small):
    // For each edge `(u,v)` in the MST of the distance graph,
    // re-run a single-source Dijkstra from `u` to find path to `v`, and extract edges.
    // This is inefficient.

    // Let's refine the path reconstruction:
    // Instead of `terminal_distances`, let's store `(distance, source_terminal)` for each pair.
    std::map<std::pair<int, int>, std::pair<long long, int>> terminal_shortest_path_info;

    // Run Dijkstra from each terminal
    for (int t_idx = 0; t_idx < num_terminals; ++t_idx) {
        int start_node = terminals[t_idx];
        auto result = dijkstra_with_path(start_node, num_vertices, adj);
        std::vector<long long>& dists = result.first;
        all_predecessors[start_node] = result.second;

        for (int other_t_idx = 0; other_t_idx < num_vertices; ++other_t_idx) { // Iterate over all nodes, not just terminals
            if (start_node == other_t_idx) continue;

            long long current_dist = dists[other_t_idx];
            
            // Check if this path is shorter than what we've seen before for (start_node, other_t_idx)
            // or (other_t_idx, start_node)
            int u = std::min(start_node, other_t_idx);
            int v = std::max(start_node, other_t_idx);

            if (current_dist != INF) {
                if (!terminal_shortest_path_info.count({u,v}) || current_dist < terminal_shortest_path_info[{u,v}].first) {
                    terminal_shortest_path_info[{u, v}] = {current_dist, start_node};
                }
            }
        }
    }
    
    // Now, construct DGraphEdges only for terminal pairs, using the best path info
    d_graph_edges.clear(); // Clear previous edges
    for (int i = 0; i < num_terminals; ++i) {
        for (int j = i + 1; j < num_terminals; ++j) {
            int u_term = terminals[i];
            int v_term = terminals[j];
            int u_min = std::min(u_term, v_term);
            int v_max = std::max(u_term, v_term);
            if (terminal_shortest_path_info.count({u_min, v_max})) {
                d_graph_edges.push_back({u_term, v_term, terminal_shortest_path_info[{u_min, v_max}].first});
            }
        }
    }
    std::sort(d_graph_edges.begin(), d_graph_edges.end());

    uf = UnionFind(num_vertices); // Re-initialize UnionFind
    total_weight = 0;
    steiner_tree_edges.clear();

    for (const auto& d_edge : d_graph_edges) {
        if (uf.find(d_edge.u) != uf.find(d_edge.v)) {
            uf.unite(d_edge.u, d_edge.v);
            // We need to add the weight later, after collecting all unique edges.
            // The total_weight is the sum of shortest path distances, not individual edge weights.
            // We need to sum actual original edge weights.

            // Reconstruct path between d_edge.u and d_edge.v
            int u_term = d_edge.u;
            int v_term = d_edge.v;
            int u_min = std::min(u_term, v_term);
            int v_max = std::max(u_term, v_term);

            // Get the source terminal that gave the shortest path for this pair
            int shortest_path_source = terminal_shortest_path_info.at({u_min, v_max}).second;

            // Reconstruct the path from v_term back to u_term (or vice-versa) using predecessors from shortest_path_source
            int path_current = (shortest_path_source == u_term) ? v_term : u_term;
            int path_target = (shortest_path_source == u_term) ? u_term : v_term;


            while (path_current != path_target) {
                auto it = all_predecessors[shortest_path_source].find(path_current);
                if (it == all_predecessors[shortest_path_source].end()) {
                    // This indicates an error in path reconstruction logic or data.
                    break;
                }
                int prev = it->second;
                steiner_tree_edges.insert({std::min(prev, path_current), std::max(prev, path_current)});
                path_current = prev;
            }
        }
    }

    // Calculate total weight from the collected steiner_tree_edges
    // Need to map edges back to their weights. The `original_adj` can be used.
    std::map<std::pair<int, int>, int> edge_weights_map;
    for (int i = 0; i < num_vertices; ++i) {
        for (const auto& edge : original_adj[i]) {
            int u = std::min(i, edge.to);
            int v = std::max(i, edge.to);
            edge_weights_map[{u, v}] = edge.weight;
        }
    }

    total_weight = 0;
    for (const auto& edge : steiner_tree_edges) {
        total_weight += edge_weights_map.at(edge);
    }

    std::cout << total_weight << " " << steiner_tree_edges.size() << "
";
    for (const auto& edge : steiner_tree_edges) {
        std::cout << edge.first << " " << edge.second << "
";
    }

    return 0;
}
