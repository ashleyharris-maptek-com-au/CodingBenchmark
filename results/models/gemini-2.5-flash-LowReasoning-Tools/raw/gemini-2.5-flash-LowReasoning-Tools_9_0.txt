using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public static void Main(string[] args)
    {
        // Read N and stock_length
        string[] firstLine = Console.ReadLine().Split(' ');
        int N = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);

        // Read cut lengths
        List<int> cutLengths = Console.ReadLine().Split(' ').Select(int.Parse).ToList();

        // Create a list of Cut objects with original indices
        List<Cut> cuts = new List<Cut>();
        for (int i = 0; i < N; i++)
        {
            cuts.Add(new Cut { Length = cutLengths[i], OriginalIndex = i });
        }

        // Sort cuts in descending order of length for First Fit Decreasing (FFD)
        cuts = cuts.OrderByDescending(c => c.Length).ToList();

        List<Stock> stocks = new List<Stock>();

        foreach (Cut cut in cuts)
        {
            bool placed = false;
            foreach (Stock stock in stocks)
            {
                if (stock.RemainingCapacity >= cut.Length)
                {
                    stock.AddCut(cut);
                    placed = true;
                    break;
                }
            }

            if (!placed)
            {
                Stock newStock = new Stock(stockLength);
                newStock.AddCut(cut);
                stocks.Add(newStock);
            }
        }

        // Output results
        Console.WriteLine(stocks.Count);
        foreach (Stock stock in stocks)
        {
            Console.WriteLine(string.Join(" ", stock.CutIndices.OrderBy(idx => idx)));
        }
    }
}

public class Cut
{
    public int Length { get; set; }
    public int OriginalIndex { get; set; }
}

public class Stock
{
    public int Capacity { get; }
    public int RemainingCapacity { get; private set; }
    public List<int> CutIndices { get; }

    public Stock(int capacity)
    {
        Capacity = capacity;
        RemainingCapacity = capacity;
        CutIndices = new List<int>();
    }

    public void AddCut(Cut cut)
    {
        if (RemainingCapacity >= cut.Length)
        {
            RemainingCapacity -= cut.Length;
            CutIndices.Add(cut.OriginalIndex);
        }
        else
        {
            throw new InvalidOperationException("Cut does not fit in stock.");
        }
    }
}
