use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec2 {
    fn sub(self, o: Vec2) -> Vec2 {
        Vec2 { x: self.x - o.x, y: self.y - o.y }
    }
    fn add(self, o: Vec2) -> Vec2 {
        Vec2 { x: self.x + o.x, y: self.y + o.y }
    }
    fn mul(self, s: f64) -> Vec2 {
        Vec2 { x: self.x * s, y: self.y * s }
    }
    fn cross(self, o: Vec2) -> f64 {
        self.x * o.y - self.y * o.x
    }
    fn len(self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    // Monotonic chain; returns CCW hull without repeating first point.
    if pts.len() <= 1 {
        return pts;
    }
    pts.sort_by(|a, b| {
        a.x
            .partial_cmp(&b.x)
            .unwrap_or(std::cmp::Ordering::Equal)
            .then_with(|| a.y.partial_cmp(&b.y).unwrap_or(std::cmp::Ordering::Equal))
    });

    fn build_half(pts: &[Vec2]) -> Vec<Vec2> {
        let mut h: Vec<Vec2> = Vec::with_capacity(pts.len());
        for &p in pts {
            while h.len() >= 2 {
                let n = h.len();
                let a = h[n - 2];
                let b = h[n - 1];
                if b.sub(a).cross(p.sub(b)) <= 0.0 {
                    h.pop();
                } else {
                    break;
                }
            }
            h.push(p);
        }
        h
    }

    let lower = build_half(&pts);
    let mut rev = pts;
    rev.reverse();
    let upper = build_half(&rev);

    let mut hull = Vec::with_capacity(lower.len() + upper.len());
    for p in lower.into_iter().take_while(|_| true) {
        hull.push(p);
    }
    let upper_len = upper.len();
    for p in upper.into_iter().skip(1).take(upper_len.saturating_sub(2)) {
        hull.push(p);
    }

    // Handle all-collinear: fallback to endpoints.
    if hull.len() < 3 {
        let mut minx = hull[0];
        let mut maxx = hull[0];
        for &p in &hull {
            if p.x < minx.x || (p.x == minx.x && p.y < minx.y) {
                minx = p;
            }
            if p.x > maxx.x || (p.x == maxx.x && p.y > maxx.y) {
                maxx = p;
            }
        }
        if (minx.x - maxx.x).abs() < 1e-12 && (minx.y - maxx.y).abs() < 1e-12 {
            vec![minx]
        } else {
            vec![minx, maxx]
        }
    } else {
        hull
    }
}

fn polygon_area(poly: &[Vec2]) -> f64 {
    if poly.len() < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        a += poly[i].cross(poly[j]);
    }
    0.5 * a
}

fn point_in_polygon(p: Vec2, poly: &[Vec2]) -> bool {
    // Winding / ray casting (non-strict). Works for simple polygons.
    if poly.len() < 3 {
        return false;
    }
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[j];
        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0) + a.x);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn project_point(v: Vec3, s: Vec3) -> Vec2 {
    // Project along direction s onto z=0 plane.
    // If s.z is ~0, fallback to vertical projection.
    let eps = 1e-12;
    if s.z.abs() < eps {
        return Vec2 { x: v.x, y: v.y };
    }
    let t = -v.z / s.z;
    Vec2 { x: v.x + t * s.x, y: v.y + t * s.y }
}

fn convex_polygon_centroid(poly: &[Vec2]) -> Vec2 {
    // Area-weighted centroid; falls back to average.
    if poly.len() < 3 {
        if poly.is_empty() {
            return Vec2 { x: 0.0, y: 0.0 };
        }
        let mut c = Vec2 { x: 0.0, y: 0.0 };
        for &p in poly {
            c = c.add(p);
        }
        return c.mul(1.0 / (poly.len() as f64));
    }

    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut a = 0.0;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        let cross = poly[i].cross(poly[j]);
        a += cross;
        cx += (poly[i].x + poly[j].x) * cross;
        cy += (poly[i].y + poly[j].y) * cross;
    }
    if a.abs() < 1e-18 {
        let mut c = Vec2 { x: 0.0, y: 0.0 };
        for &p in poly {
            c = c.add(p);
        }
        return c.mul(1.0 / (poly.len() as f64));
    }
    a *= 0.5;
    Vec2 { x: cx / (6.0 * a), y: cy / (6.0 * a) }
}

fn convex_inradius_about_origin(poly: &[Vec2]) -> f64 {
    // Assumes poly is CCW and contains origin; returns min distance from origin to edges.
    if poly.len() < 3 {
        return 0.0;
    }
    let mut r = f64::INFINITY;
    for i in 0..poly.len() {
        let j = (i + 1) % poly.len();
        let a = poly[i];
        let b = poly[j];
        let e = b.sub(a);
        let len = e.len();
        if len < 1e-18 {
            continue;
        }
        // Distance from origin to line through a->b
        let dist = (a.cross(b)).abs() / len;
        if dist < r {
            r = dist;
        }
    }
    if r.is_finite() { r } else { 0.0 }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv_opt = it.next();
    if nv_opt.is_none() {
        println!("0");
        return;
    }
    let nv: usize = nv_opt.unwrap().parse().unwrap();
    let mut poly: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Vec2 { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    let s = Vec3 { x: sx, y: sy, z: sz };

    // Reference tetrahedron vertices.
    let base = [
        Vec3 { x: 1.0, y: 0.0, z: -0.707 },
        Vec3 { x: -1.0, y: 0.0, z: -0.707 },
        Vec3 { x: 0.0, y: 1.0, z: 0.707 },
        Vec3 { x: 0.0, y: -1.0, z: 0.707 },
    ];

    // Choose scale adaptively based on polygon size.
    let mut minx = poly[0].x;
    let mut maxx = poly[0].x;
    let mut miny = poly[0].y;
    let mut maxy = poly[0].y;
    for &p in &poly {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }
    let bbox_w = maxx - minx;
    let bbox_h = maxy - miny;
    let bbox_diag = (bbox_w * bbox_w + bbox_h * bbox_h).sqrt();

    // Start with max scale; for very small polygons, reduce scale to reduce overshoot.
    let mut scale = 10.0;
    if bbox_diag < 5.0 {
        scale = (bbox_diag / 0.5).clamp(0.2, 10.0);
    }

    // Compute the tetrahedron's shadow footprint (convex hull) for identity rotation.
    let mut shadow_pts: Vec<Vec2> = Vec::with_capacity(4);
    for &v in &base {
        let vv = Vec3 { x: v.x * scale, y: v.y * scale, z: v.z * scale };
        shadow_pts.push(project_point(vv, s));
    }
    let mut hull = convex_hull(shadow_pts);

    // Ensure CCW.
    if polygon_area(&hull) < 0.0 {
        hull.reverse();
    }

    let c_hull = convex_polygon_centroid(&hull);
    for p in &mut hull {
        p.x -= c_hull.x;
        p.y -= c_hull.y;
    }

    // Conservative inradius for coverage spacing.
    let mut r = convex_inradius_about_origin(&hull);
    if !r.is_finite() || r <= 1e-6 {
        r = 0.5 * scale;
    }

    // Adaptive strategy: for complex / large polygons, use a grid; otherwise just a few samples.
    let big = nv > 2000 || bbox_diag > 200.0;
    let step = if big { r * 1.6 } else { r * 1.3 };
    let step = step.max(1e-3);

    // Generate centers.
    let mut centers: Vec<Vec2> = Vec::new();

    // Boundary sampling to reduce misses near edges.
    let boundary_step = (r * 0.9).max(step * 0.5);
    for i in 0..nv {
        let a = poly[i];
        let b = poly[(i + 1) % nv];
        let ab = b.sub(a);
        let len = ab.len();
        if len < 1e-12 {
            continue;
        }
        let nseg = (len / boundary_step).ceil() as usize;
        let nseg = nseg.max(1);
        for k in 0..=nseg {
            let t = (k as f64) / (nseg as f64);
            let p = a.add(ab.mul(t));
            centers.push(p);
        }
    }

    // Interior grid.
    let pad = r * 2.0;
    let gx0 = minx - pad;
    let gx1 = maxx + pad;
    let gy0 = miny - pad;
    let gy1 = maxy + pad;

    let mut y = gy0;
    let mut row = 0usize;
    while y <= gy1 {
        let offset = if row % 2 == 0 { 0.0 } else { step * 0.5 };
        let mut x = gx0 + offset;
        while x <= gx1 {
            let p = Vec2 { x, y };
            if point_in_polygon(p, &poly) {
                centers.push(p);
            }
            x += step;
        }
        y += step * (3.0_f64).sqrt() * 0.5;
        row += 1;
        if centers.len() > 200_000 {
            break;
        }
    }

    // Deduplicate roughly using hashing grid.
    let cell = (r * 0.25).max(1e-3);
    use std::collections::HashSet;
    let mut seen: HashSet<(i64, i64)> = HashSet::new();
    let mut final_centers: Vec<Vec2> = Vec::new();
    for p in centers {
        let ix = (p.x / cell).round() as i64;
        let iy = (p.y / cell).round() as i64;
        if seen.insert((ix, iy)) {
            final_centers.push(p);
        }
        if final_centers.len() > 120_000 {
            break;
        }
    }

    // Output.
    // Use identity rotation quaternion; place all tetrahedra with the same z to keep shadows aligned.
    // Slightly lift above plane to avoid degeneracy.
    let z0 = 1.0;

    println!("{}", final_centers.len());
    for p in final_centers {
        let px = p.x;
        let py = p.y;
        let pz = z0;
        let qw = 1.0;
        let qx = 0.0;
        let qy = 0.0;
        let qz = 0.0;
        println!("{px:.9} {py:.9} {pz:.9} {qw:.9} {qx:.9} {qy:.9} {qz:.9} {scale:.9}");
    }
}
