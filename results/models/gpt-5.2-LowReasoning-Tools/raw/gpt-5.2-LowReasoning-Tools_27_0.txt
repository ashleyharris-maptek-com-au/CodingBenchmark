// Graph Coloring (k-coloring) solver
// - Reads an undirected graph from stdin: n m k then m edges (0-indexed)
// - Outputs n integers in [0, k-1] (one per vertex) such that adjacent vertices differ
//
// Heuristics:
// - For small graphs: DSATUR-style backtracking with pruning (exact if time allows)
// - For larger graphs: greedy coloring with multiple orderings (degree sort / degeneracy / random restarts)
//
// Notes:
// - Designed to be portable (MSVC /std:c++17+ and GCC/Clang).
// - Avoids non-standard headers.

#include <algorithm>
#include <chrono>
#include <cstdint>
#include <cstdlib>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <utility>
#include <vector>

using u32 = uint32_t;
using u64 = uint64_t;

struct CSRGraph {
    u32 n = 0;
    u64 m = 0; // undirected edges count in input (each counted once)
    std::vector<u32> off; // size n+1
    std::vector<u32> adj; // size 2m

    u32 degree(u32 v) const { return off[v + 1] - off[v]; }
};

static CSRGraph read_graph_stdin(u32& k_out) {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    u32 n;
    u64 m;
    u32 k;
    if (!(std::cin >> n >> m >> k)) {
        std::exit(0);
    }
    k_out = k;

    std::vector<std::pair<u32, u32>> edges;
    edges.reserve(static_cast<size_t>(m));
    std::vector<u32> deg(n, 0);

    for (u64 i = 0; i < m; ++i) {
        u32 u, v;
        std::cin >> u >> v;
        if (u == v) continue; // ignore self-loops (unsatisfiable otherwise)
        if (u >= n || v >= n) continue;
        edges.emplace_back(u, v);
        ++deg[u];
        ++deg[v];
    }

    CSRGraph g;
    g.n = n;
    g.m = static_cast<u64>(edges.size());
    g.off.assign(n + 1, 0);
    for (u32 i = 0; i < n; ++i) g.off[i + 1] = g.off[i] + deg[i];
    g.adj.assign(static_cast<size_t>(g.off[n]), 0);

    std::vector<u32> cur = g.off;
    for (auto [u, v] : edges) {
        g.adj[cur[u]++] = v;
        g.adj[cur[v]++] = u;
    }
    return g;
}

static inline u32 pick_smallest_available_color(u64 forbidden, u32 k) {
    if (k >= 64) {
        // Should not happen in the benchmark described (<=60), but keep a safe fallback.
        for (u32 c = 0; c < k; ++c) {
            if (((forbidden >> (c & 63u)) & 1ull) == 0ull) return c;
        }
        return std::numeric_limits<u32>::max();
    }
    u64 allowed = (~forbidden);
    if (k < 64) allowed &= ((k == 64) ? ~0ull : ((1ull << k) - 1ull));
    if (allowed == 0ull) return std::numeric_limits<u32>::max();
#if defined(_MSC_VER)
    unsigned long idx = 0;
    _BitScanForward64(&idx, allowed);
    return static_cast<u32>(idx);
#else
    return static_cast<u32>(__builtin_ctzll(allowed));
#endif
}

static bool greedy_color_with_order(
    const CSRGraph& g,
    u32 k,
    const std::vector<u32>& order,
    std::vector<int>& color_out
) {
    const u32 n = g.n;
    color_out.assign(n, -1);

    // For k <= 64 we can compute forbidden masks quickly.
    for (u32 idx = 0; idx < n; ++idx) {
        u32 v = order[idx];
        u64 forbidden = 0ull;
        for (u32 ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
            u32 u = g.adj[ei];
            int cu = color_out[u];
            if (cu >= 0) {
                u32 c = static_cast<u32>(cu);
                if (c < 64) forbidden |= (1ull << c);
            }
        }
        u32 c = pick_smallest_available_color(forbidden, k);
        if (c == std::numeric_limits<u32>::max()) return false;
        if (c >= k) return false;
        color_out[v] = static_cast<int>(c);
    }
    return true;
}

static std::vector<u32> order_by_desc_degree(const CSRGraph& g) {
    const u32 n = g.n;
    std::vector<u32> order(n);
    std::iota(order.begin(), order.end(), 0u);
    std::sort(order.begin(), order.end(), [&](u32 a, u32 b) {
        u32 da = g.degree(a), db = g.degree(b);
        if (da != db) return da > db;
        return a < b;
    });
    return order;
}

// Degeneracy ordering (Matula-Beck): returns order of removal from min-degree buckets.
// To color, process in reverse order.
static std::vector<u32> degeneracy_order(const CSRGraph& g) {
    const u32 n = g.n;
    u32 max_deg = 0;
    std::vector<u32> deg(n);
    for (u32 v = 0; v < n; ++v) {
        deg[v] = g.degree(v);
        max_deg = std::max(max_deg, deg[v]);
    }

    std::vector<int> head(static_cast<size_t>(max_deg) + 1, -1);
    std::vector<int> next(n, -1), prev(n, -1);
    auto bucket_insert = [&](u32 v, u32 d) {
        int& h = head[d];
        prev[v] = -1;
        next[v] = h;
        if (h != -1) prev[h] = static_cast<int>(v);
        h = static_cast<int>(v);
    };
    auto bucket_remove = [&](u32 v, u32 d) {
        int pv = prev[v], nv = next[v];
        if (pv != -1) next[pv] = nv;
        else head[d] = nv;
        if (nv != -1) prev[nv] = pv;
        prev[v] = next[v] = -1;
    };

    for (u32 v = 0; v < n; ++v) bucket_insert(v, deg[v]);

    std::vector<uint8_t> removed(n, 0);
    std::vector<u32> order;
    order.reserve(n);

    u32 mind = 0;
    for (u32 iter = 0; iter < n; ++iter) {
        while (mind <= max_deg && head[mind] == -1) ++mind;
        if (mind > max_deg) break; // should not happen
        u32 v = static_cast<u32>(head[mind]);
        bucket_remove(v, mind);
        removed[v] = 1;
        order.push_back(v);

        // Decrease degree of neighbors not removed
        for (u32 ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
            u32 u = g.adj[ei];
            if (removed[u]) continue;
            u32 du = deg[u];
            bucket_remove(u, du);
            deg[u] = du - 1;
            bucket_insert(u, du - 1);
            if (du - 1 < mind) mind = du - 1;
        }
    }
    return order;
}

static bool validate_coloring(const CSRGraph& g, u32 k, const std::vector<int>& col) {
    if (col.size() != static_cast<size_t>(g.n)) return false;
    for (u32 v = 0; v < g.n; ++v) {
        int cv = col[v];
        if (cv < 0 || static_cast<u32>(cv) >= k) return false;
        for (u32 ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
            u32 u = g.adj[ei];
            if (col[u] == cv) return false;
        }
    }
    return true;
}

// DSATUR-ish exact search for small graphs.
struct ExactDSATUR {
    const CSRGraph& g;
    u32 k;
    std::vector<int> color;         // current assignment
    std::vector<u64> neighMask;     // used colors among neighbors (bitmask)
    std::vector<u32> uncolored;     // list of uncolored vertices
    std::vector<u32> degree;
    std::chrono::steady_clock::time_point deadline;

    ExactDSATUR(const CSRGraph& graph, u32 kk, std::chrono::steady_clock::time_point dl)
        : g(graph), k(kk), color(graph.n, -1), neighMask(graph.n, 0ull),
          uncolored(graph.n), degree(graph.n, 0), deadline(dl) {
        std::iota(uncolored.begin(), uncolored.end(), 0u);
        for (u32 v = 0; v < g.n; ++v) degree[v] = g.degree(v);
    }

    bool timed_out() const { return std::chrono::steady_clock::now() >= deadline; }

    u32 pick_vertex() const {
        // Pick uncolored vertex with highest saturation (popcount(neighMask)),
        // tie-break on degree.
        u32 best = uncolored[0];
        int bestSat = -1;
        u32 bestDeg = 0;
        for (u32 v : uncolored) {
            int sat = 0;
            u64 m = neighMask[v];
            if (k < 64) m &= ((k == 64) ? ~0ull : ((1ull << k) - 1ull));
#if defined(_MSC_VER)
            sat = static_cast<int>(__popcnt64(m));
#else
            sat = static_cast<int>(__builtin_popcountll(m));
#endif
            u32 dv = degree[v];
            if (sat > bestSat || (sat == bestSat && dv > bestDeg) || (sat == bestSat && dv == bestDeg && v < best)) {
                best = v;
                bestSat = sat;
                bestDeg = dv;
            }
        }
        return best;
    }

    bool can_use_color(u32 v, u32 c) const {
        for (u32 ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
            u32 u = g.adj[ei];
            if (color[u] == static_cast<int>(c)) return false;
        }
        return true;
    }

    void apply(u32 v, u32 c, std::vector<u32>& changed) {
        color[v] = static_cast<int>(c);
        // Update neigh masks for uncolored neighbors
        for (u32 ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
            u32 u = g.adj[ei];
            if (color[u] == -1) {
                u64 before = neighMask[u];
                neighMask[u] |= (c < 64 ? (1ull << c) : 0ull);
                if (neighMask[u] != before) changed.push_back(u);
            }
        }
    }

    void undo(u32 v, u32 c, const std::vector<u32>& changed) {
        (void)c;
        color[v] = -1;
        // Recompute neighMask for changed vertices (cheap for small graphs).
        for (u32 u : changed) {
            u64 m = 0ull;
            for (u32 ei = g.off[u]; ei < g.off[u + 1]; ++ei) {
                u32 w = g.adj[ei];
                int cw = color[w];
                if (cw >= 0 && static_cast<u32>(cw) < 64) m |= (1ull << static_cast<u32>(cw));
            }
            neighMask[u] = m;
        }
    }

    bool dfs() {
        if (timed_out()) return false;
        if (uncolored.empty()) return true;

        u32 v = pick_vertex();
        // Remove v from uncolored
        auto it = std::find(uncolored.begin(), uncolored.end(), v);
        u32 idx = static_cast<u32>(it - uncolored.begin());
        std::swap(uncolored[idx], uncolored.back());
        uncolored.pop_back();

        u64 forbidden = neighMask[v];
        if (k < 64) forbidden &= ((k == 64) ? ~0ull : ((1ull << k) - 1ull));

        // Try colors in ascending order (often good for exact feasibility).
        for (u32 c = 0; c < k; ++c) {
            if (c < 64 && ((forbidden >> c) & 1ull)) continue;
            if (!can_use_color(v, c)) continue;
            std::vector<u32> changed;
            changed.reserve(16);
            apply(v, c, changed);
            if (dfs()) return true;
            undo(v, c, changed);
        }

        // Put v back
        uncolored.push_back(v);
        return false;
    }
};

static std::vector<u32> make_order_from_degeneracy_reverse(const CSRGraph& g) {
    std::vector<u32> rem = degeneracy_order(g);
    std::reverse(rem.begin(), rem.end());
    return rem;
}

static std::vector<u32> shuffled_order(u32 n, u64 seed) {
    std::vector<u32> ord(n);
    std::iota(ord.begin(), ord.end(), 0u);
    // Simple xorshift-based shuffle to avoid heavy RNG overhead.
    auto xorshift64 = [&seed]() -> u64 {
        seed ^= seed << 13;
        seed ^= seed >> 7;
        seed ^= seed << 17;
        return seed;
    };
    for (u32 i = n; i > 1; --i) {
        u64 r = xorshift64();
        u32 j = static_cast<u32>(r % i);
        std::swap(ord[i - 1], ord[j]);
    }
    return ord;
}

int main() {
    u32 k = 0;
    CSRGraph g = read_graph_stdin(k);
    const u32 n = g.n;
    if (n == 0) return 0;

    // If k==0, impossible; output zeros to keep format.
    if (k == 0) {
        for (u32 i = 0; i < n; ++i) {
            if (i) std::cout << ' ';
            std::cout << 0;
        }
        std::cout << '\n';
        return 0;
    }

    auto start = std::chrono::steady_clock::now();
    auto deadline = start + std::chrono::minutes(4) + std::chrono::seconds(40);

    std::vector<int> coloring;

    // Strategy 1: exact DSATUR for small graphs (fast to prove feasibility).
    if (n <= 5000 && k <= 64) {
        ExactDSATUR solver(g, k, deadline);
        if (solver.dfs()) {
            coloring = solver.color;
            // Output
            for (u32 i = 0; i < n; ++i) {
                if (i) std::cout << ' ';
                std::cout << coloring[i];
            }
            std::cout << '\n';
            return 0;
        }
        // Fall through to heuristics if timed out / not found quickly.
    }

    // Strategy 2: heuristic greedy with multiple orderings.
    std::vector<std::vector<u32>> candidate_orders;
    candidate_orders.reserve(8);

    if (n <= 200000) {
        candidate_orders.push_back(order_by_desc_degree(g));
    }
    // Degeneracy-based reverse order often good on sparse graphs.
    candidate_orders.push_back(make_order_from_degeneracy_reverse(g));
    // Natural order as a cheap baseline.
    {
        std::vector<u32> ord(n);
        std::iota(ord.begin(), ord.end(), 0u);
        candidate_orders.push_back(std::move(ord));
    }
    // Random restarts (use time-dependent seed)
    {
        u64 seed = static_cast<u64>(std::chrono::high_resolution_clock::now().time_since_epoch().count());
        int restarts = (n <= 200000 ? 6 : 3);
        for (int i = 0; i < restarts; ++i) {
            candidate_orders.push_back(shuffled_order(n, seed + 0x9e3779b97f4a7c15ull * static_cast<u64>(i + 1)));
        }
    }

    bool ok = false;
    for (auto& ord : candidate_orders) {
        if (std::chrono::steady_clock::now() >= deadline) break;
        if (greedy_color_with_order(g, k, ord, coloring) && validate_coloring(g, k, coloring)) {
            ok = true;
            break;
        }
    }

    // Last-chance: try coloring in degeneracy removal order (not reversed).
    if (!ok && std::chrono::steady_clock::now() < deadline) {
        std::vector<u32> ord = degeneracy_order(g);
        if (greedy_color_with_order(g, k, ord, coloring) && validate_coloring(g, k, coloring)) ok = true;
    }

    // If still not ok, output best-effort assignment (ensures all vertices colored).
    if (!ok) {
        // Force-color greedily with wrap-around (may violate constraints, but avoids incomplete output).
        coloring.assign(n, 0);
        std::vector<u32> ord = make_order_from_degeneracy_reverse(g);
        for (u32 v : ord) {
            u64 forbidden = 0ull;
            for (u32 ei = g.off[v]; ei < g.off[v + 1]; ++ei) {
                u32 u = g.adj[ei];
                int cu = coloring[u];
                if (cu >= 0 && static_cast<u32>(cu) < 64) forbidden |= (1ull << static_cast<u32>(cu));
            }
            u32 c = pick_smallest_available_color(forbidden, k);
            if (c == std::numeric_limits<u32>::max()) c = 0;
            if (c >= k) c = 0;
            coloring[v] = static_cast<int>(c);
        }
    }

    for (u32 i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << coloring[i];
    }
    std::cout << '\n';
    return 0;
}
