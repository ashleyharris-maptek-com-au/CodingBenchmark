
use std::cmp::min;
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Vec3 {
    x: f32,
    y: f32,
    z: f32,
}

impl Vec3 {
    #[inline]
    fn sub(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x - o.x,
            y: self.y - o.y,
            z: self.z - o.z,
        }
    }
    #[inline]
    fn add(self, o: Vec3) -> Vec3 {
        Vec3 {
            x: self.x + o.x,
            y: self.y + o.y,
            z: self.z + o.z,
        }
    }
    #[inline]
    fn dist2(self, o: Vec3) -> f32 {
        let d = self.sub(o);
        d.x * d.x + d.y * d.y + d.z * d.z
    }
}

struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    #[inline]
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 13;
        x ^= x >> 7;
        x ^= x << 17;
        self.state = x;
        x
    }
    #[inline]
    fn next_usize(&mut self, n: usize) -> usize {
        if n <= 1 {
            0
        } else {
            (self.next_u64() % (n as u64)) as usize
        }
    }
    #[inline]
    fn next_f32(&mut self) -> f32 {
        // [0,1)
        const SCALE: f32 = 1.0 / ((1u64 << 24) as f32);
        ((self.next_u64() >> 40) as u32 as f32) * SCALE
    }
}

struct StdinCursor {
    buf: Vec<u8>,
    pos: usize,
}
impl StdinCursor {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, pos: 0 })
    }
    #[inline]
    fn skip_ws(&mut self) {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
    }
    fn read_u64_ascii(&mut self) -> Option<u64> {
        self.skip_ws();
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut v: u64 = 0;
        let mut any = false;
        while self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            if b.is_ascii_digit() {
                any = true;
                v = v.saturating_mul(10).saturating_add((b - b'0') as u64);
                self.pos += 1;
            } else {
                break;
            }
        }
        if !any { None } else { Some(v) }
    }
    #[inline]
    fn remaining(&self) -> usize {
        self.buf.len().saturating_sub(self.pos)
    }
    #[inline]
    fn read_exact(&mut self, out: &mut [u8]) -> bool {
        if self.remaining() < out.len() {
            return false;
        }
        out.copy_from_slice(&self.buf[self.pos..self.pos + out.len()]);
        self.pos += out.len();
        true
    }
}

#[inline]
fn read_point(cur: &mut StdinCursor) -> Option<Vec3> {
    let mut bytes = [0u8; 12];
    if !cur.read_exact(&mut bytes) {
        return None;
    }
    let x = f32::from_le_bytes(bytes[0..4].try_into().unwrap());
    let y = f32::from_le_bytes(bytes[4..8].try_into().unwrap());
    let z = f32::from_le_bytes(bytes[8..12].try_into().unwrap());
    Some(Vec3 { x, y, z })
}

fn kmeanspp_init(sample: &[Vec3], k: usize, rng: &mut XorShift64) -> Vec<Vec3> {
    let n = sample.len();
    if k == 0 {
        return Vec::new();
    }
    if n == 0 {
        return vec![Vec3::default(); k];
    }

    let mut centers: Vec<Vec3> = Vec::with_capacity(k);
    centers.push(sample[rng.next_usize(n)]);

    let mut d2: Vec<f32> = vec![0.0; n];
    for i in 0..n {
        d2[i] = sample[i].dist2(centers[0]).max(0.0);
    }

    while centers.len() < k {
        let mut sum = 0.0f64;
        for &v in &d2 {
            sum += v as f64;
        }
        if !sum.is_finite() || sum <= 0.0 {
            centers.push(sample[rng.next_usize(n)]);
            continue;
        }
        let mut r = (rng.next_f32() as f64) * sum;
        let mut idx = 0usize;
        for i in 0..n {
            r -= d2[i] as f64;
            if r <= 0.0 {
                idx = i;
                break;
            }
        }
        let c = sample[idx];
        centers.push(c);
        for i in 0..n {
            let nd = sample[i].dist2(c).max(0.0);
            if nd < d2[i] {
                d2[i] = nd;
            }
        }
    }
    centers
}

#[inline]
fn nearest_bruteforce(p: Vec3, centers: &[Vec3]) -> usize {
    let mut best_i = 0usize;
    let mut best_d = p.dist2(centers[0]);
    for (i, &c) in centers.iter().enumerate().skip(1) {
        let d = p.dist2(c);
        if d < best_d {
            best_d = d;
            best_i = i;
        }
    }
    best_i
}

#[inline]
fn nearest_adaptive(p: Vec3, centers: &[Vec3], rng: &mut XorShift64, hint: usize) -> usize {
    let k = centers.len();
    if k <= 2048 {
        return nearest_bruteforce(p, centers);
    }

    let mut best_i = hint.min(k - 1);
    let mut best_d = p.dist2(centers[best_i]);

    let window = 32usize;
    let start = best_i.saturating_sub(window);
    let end = min(k, best_i + window + 1);
    for i in start..end {
        let d = p.dist2(centers[i]);
        if d < best_d {
            best_d = d;
            best_i = i;
        }
    }

    let probes = if k <= 10_000 { 512 } else { 768 };
    for _ in 0..probes {
        let i = rng.next_usize(k);
        let d = p.dist2(centers[i]);
        if d < best_d {
            best_d = d;
            best_i = i;
        }
    }
    best_i
}

fn main() -> io::Result<()> {
    // Practical benchmark note: stdin is typically bounded by runner resources.
    // This implementation reads stdin fully to robustly handle "text header + binary tail"
    // without seeking; the clustering itself is one-pass online k-means.
    let mut cur = StdinCursor::new()?;

    let num_points = cur
        .read_u64_ascii()
        .unwrap_or_else(|| panic!("missing num_points"));
    let num_clusters = cur
        .read_u64_ascii()
        .unwrap_or_else(|| panic!("missing num_clusters"));
    let n = num_points as usize;
    let k = num_clusters as usize;
    if k == 0 || n == 0 {
        return Ok(());
    }

    let init_cap_points: usize = 2_000_000; // 24MB for f32x3
    let init_points = min(n, init_cap_points);
    let mut prefix: Vec<Vec3> = Vec::with_capacity(init_points);

    let reservoir_cap: usize = min(1_000_000usize, init_points.max(k).min(n));
    let mut reservoir: Vec<Vec3> = Vec::with_capacity(reservoir_cap);

    let mut rng = XorShift64::new(
        0x9E3779B97F4A7C15u64 ^ (n as u64).wrapping_mul(0xBF58476D1CE4E5B9),
    );

    for i in 0..init_points {
        if let Some(p) = read_point(&mut cur) {
            prefix.push(p);
            if reservoir.len() < reservoir_cap {
                reservoir.push(p);
            } else {
                let j = rng.next_u64() % ((i as u64) + 1);
                if (j as usize) < reservoir_cap {
                    reservoir[j as usize] = p;
                }
            }
        } else {
            break;
        }
    }

    if reservoir.is_empty() {
        return Ok(());
    }

    let mut centers = kmeanspp_init(&reservoir, k.min(reservoir.len()), &mut rng);
    if centers.len() < k {
        let base = centers.clone();
        while centers.len() < k {
            let p = base[rng.next_usize(base.len())];
            let eps = 1e-3f32;
            let jitter = Vec3 {
                x: (rng.next_f32() - 0.5) * eps,
                y: (rng.next_f32() - 0.5) * eps,
                z: (rng.next_f32() - 0.5) * eps,
            };
            centers.push(p.add(jitter));
        }
    }

    let mut counts: Vec<u64> = vec![0; k];
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    let mut last_idx = 0usize;

    for &p in &prefix {
        let idx = nearest_adaptive(p, &centers, &mut rng, last_idx);
        last_idx = idx;
        out.write_all(&(idx as u32).to_le_bytes())?;
        let c = &mut centers[idx];
        let cnt = counts[idx].wrapping_add(1);
        counts[idx] = cnt;
        let inv = 1.0f32 / (cnt as f32);
        c.x += (p.x - c.x) * inv;
        c.y += (p.y - c.y) * inv;
        c.z += (p.z - c.z) * inv;
    }
    prefix.clear();

    for _ in init_points..n {
        let Some(p) = read_point(&mut cur) else { break };
        let idx = nearest_adaptive(p, &centers, &mut rng, last_idx);
        last_idx = idx;
        out.write_all(&(idx as u32).to_le_bytes())?;
        let c = &mut centers[idx];
        let cnt = counts[idx].wrapping_add(1);
        counts[idx] = cnt;
        let inv = 1.0f32 / (cnt as f32);
        c.x += (p.x - c.x) * inv;
        c.y += (p.y - c.y) * inv;
        c.z += (p.z - c.z) * inv;
    }

    out.flush()?;
    Ok(())
}
