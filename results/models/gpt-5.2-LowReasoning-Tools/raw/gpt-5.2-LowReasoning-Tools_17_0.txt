using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private readonly struct Box
    {
        public readonly int Index;
        public readonly int W;
        public readonly int H;
        public readonly int D;
        public Box(int index, int w, int h, int d) { Index = index; W = w; H = h; D = d; }
        public long Volume => (long)W * H * D;
        public int MinDim => Math.Min(W, Math.Min(H, D));
        public int MaxDim => Math.Max(W, Math.Max(H, D));
    }

    private readonly struct Oriented
    {
        public readonly int Index;
        public readonly int W;
        public readonly int H;
        public readonly int D;
        public Oriented(int index, int w, int h, int d) { Index = index; W = w; H = h; D = d; }
        public long BaseArea => (long)W * D;
        public long Volume => (long)W * H * D;
    }

    private struct Rect
    {
        public int X, Z, W, D;
        public Rect(int x, int z, int w, int d) { X = x; Z = z; W = w; D = d; }
    }

    private struct Placement
    {
        public int Index;
        public int X, Y, Z;
        public int W, H, D;
    }

    public static void Main()
    {
        var input = ReadAllTokens();
        if (input.Count < 4) return;
        int p = 0;
        int n = int.Parse(input[p++], CultureInfo.InvariantCulture);
        int W = int.Parse(input[p++], CultureInfo.InvariantCulture);
        int H = int.Parse(input[p++], CultureInfo.InvariantCulture);
        int D = int.Parse(input[p++], CultureInfo.InvariantCulture);

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = int.Parse(input[p++], CultureInfo.InvariantCulture);
            int h = int.Parse(input[p++], CultureInfo.InvariantCulture);
            int d = int.Parse(input[p++], CultureInfo.InvariantCulture);
            boxes[i] = new Box(i, w, h, d);
        }

        // Run a small portfolio of fast heuristics and keep the best.
        // The layer-based packer is robust from tiny to large instances.
        var best = new List<Placement>(n);
        TryUpdateBest(ref best, PackLayered(boxes, W, H, D, Strategy.MinHeightThenArea));
        TryUpdateBest(ref best, PackLayered(boxes, W, H, D, Strategy.MaxBaseArea));
        TryUpdateBest(ref best, PackLayered(boxes, W, H, D, Strategy.VolumeDesc));
        if (n <= 80) // small instances: add a tighter strategy set
        {
            TryUpdateBest(ref best, PackLayered(boxes, W, H, D, Strategy.MinHeightThenVolume));
            TryUpdateBest(ref best, PackLayered(boxes, W, H, D, Strategy.HeightDesc));
        }

        using var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };
        sw.WriteLine(best.Count.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < best.Count; i++)
        {
            var pl = best[i];
            sw.Write(pl.Index.ToString(CultureInfo.InvariantCulture));
            sw.Write(' ');
            sw.Write(pl.X.ToString(CultureInfo.InvariantCulture));
            sw.Write(' ');
            sw.Write(pl.Y.ToString(CultureInfo.InvariantCulture));
            sw.Write(' ');
            sw.WriteLine(pl.Z.ToString(CultureInfo.InvariantCulture));
        }
    }

    private enum Strategy
    {
        MinHeightThenArea,
        MinHeightThenVolume,
        MaxBaseArea,
        VolumeDesc,
        HeightDesc,
    }

    private static void TryUpdateBest(ref List<Placement> best, List<Placement> candidate)
    {
        if (candidate.Count > best.Count) best = candidate;
    }

    private static List<Placement> PackLayered(Box[] boxes, int containerW, int containerH, int containerD, Strategy strat)
    {
        // Build oriented choices: allow reordering w/h/d (6 orientations).
        // For speed, we choose a single orientation per box for the run, based on strategy.
        var oriented = new Oriented[boxes.Length];
        for (int i = 0; i < boxes.Length; i++)
            oriented[i] = ChooseOrientation(boxes[i], containerW, containerH, containerD, strat);

        Array.Sort(oriented, (a, b) =>
        {
            int cmp;
            switch (strat)
            {
                case Strategy.MaxBaseArea:
                    cmp = b.BaseArea.CompareTo(a.BaseArea);
                    if (cmp != 0) return cmp;
                    cmp = b.H.CompareTo(a.H);
                    if (cmp != 0) return cmp;
                    return b.Volume.CompareTo(a.Volume);
                case Strategy.VolumeDesc:
                    cmp = b.Volume.CompareTo(a.Volume);
                    if (cmp != 0) return cmp;
                    cmp = b.BaseArea.CompareTo(a.BaseArea);
                    if (cmp != 0) return cmp;
                    return b.H.CompareTo(a.H);
                case Strategy.HeightDesc:
                    cmp = b.H.CompareTo(a.H);
                    if (cmp != 0) return cmp;
                    cmp = b.BaseArea.CompareTo(a.BaseArea);
                    if (cmp != 0) return cmp;
                    return b.Volume.CompareTo(a.Volume);
                case Strategy.MinHeightThenVolume:
                    cmp = a.H.CompareTo(b.H);
                    if (cmp != 0) return cmp;
                    cmp = b.Volume.CompareTo(a.Volume);
                    if (cmp != 0) return cmp;
                    return b.BaseArea.CompareTo(a.BaseArea);
                default:
                    cmp = a.H.CompareTo(b.H);
                    if (cmp != 0) return cmp;
                    cmp = b.BaseArea.CompareTo(a.BaseArea);
                    if (cmp != 0) return cmp;
                    return b.Volume.CompareTo(a.Volume);
            }
        });

        var used = new bool[boxes.Length];
        var placements = new List<Placement>(boxes.Length);

        int y = 0;
        int remaining = boxes.Length;

        // Adaptive layer sizing:
        // - For huge N, we keep layers simple: layer height is the current box height.
        // - For small N, we allow some lookahead: choose layer height among a few candidates to improve fill.
        while (y < containerH && remaining > 0)
        {
            // Find next unused box that can fit at all.
            int seedIdx = -1;
            for (int i = 0; i < oriented.Length; i++)
            {
                var b = oriented[i];
                if (used[b.Index]) continue;
                if (b.W > containerW || b.D > containerD || b.H > (containerH - y)) continue;
                seedIdx = i;
                break;
            }
            if (seedIdx < 0) break;

            int layerH = oriented[seedIdx].H;
            if (boxes.Length <= 200)
            {
                // Try a couple of nearby layer heights (from next few unused boxes) and keep best for this layer.
                var candidates = new List<int>(6) { layerH };
                for (int j = seedIdx + 1; j < oriented.Length && candidates.Count < 6; j++)
                {
                    var b = oriented[j];
                    if (used[b.Index]) continue;
                    if (b.W > containerW || b.D > containerD || b.H > (containerH - y)) continue;
                    if (!candidates.Contains(b.H)) candidates.Add(b.H);
                }
                int bestCount = -1;
                int bestH = layerH;
                foreach (int h in candidates)
                {
                    int c = CountPackableInLayer(oriented, used, containerW, containerD, h);
                    if (c > bestCount)
                    {
                        bestCount = c;
                        bestH = h;
                    }
                }
                layerH = bestH;
            }

            var layerPlacements = PackOneLayer(oriented, used, containerW, containerD, layerH, y);
            if (layerPlacements.Count == 0)
            {
                // If we couldn't place anything, advance y minimally to avoid stalling.
                y += Math.Max(1, layerH);
                continue;
            }

            placements.AddRange(layerPlacements);
            foreach (var pl in layerPlacements)
            {
                if (!used[pl.Index])
                {
                    used[pl.Index] = true;
                    remaining--;
                }
            }

            y += layerH;
        }

        return placements;
    }

    private static int CountPackableInLayer(Oriented[] boxes, bool[] used, int layerW, int layerD, int layerH)
    {
        // Very fast estimate using greedy area fit without full split pruning.
        long area = (long)layerW * layerD;
        long usedArea = 0;
        int count = 0;
        for (int i = 0; i < boxes.Length; i++)
        {
            var b = boxes[i];
            if (used[b.Index]) continue;
            if (b.H > layerH) continue;
            if (b.W > layerW || b.D > layerD) continue;
            long a = (long)b.W * b.D;
            if (usedArea + a > area) continue;
            usedArea += a;
            count++;
        }
        return count;
    }

    private static List<Placement> PackOneLayer(Oriented[] boxes, bool[] used, int layerW, int layerD, int layerH, int yOffset)
    {
        // 2D guillotine packing in X-Z within a fixed layer height.
        var free = new List<Rect>(128) { new Rect(0, 0, layerW, layerD) };
        var placements = new List<Placement>(256);

        // Iterate in the given global order; this is deliberate for speed and stability.
        for (int i = 0; i < boxes.Length; i++)
        {
            var b = boxes[i];
            if (used[b.Index]) continue;
            if (b.H > layerH) continue;
            if (b.W > layerW || b.D > layerD) continue;

            int bestRectIdx = -1;
            long bestScore = long.MaxValue;
            for (int r = 0; r < free.Count; r++)
            {
                var fr = free[r];
                if (b.W <= fr.W && b.D <= fr.D)
                {
                    // Best area fit + tie-break by leftover short side.
                    long waste = (long)(fr.W * fr.D) - (long)(b.W * b.D);
                    int shortSide = Math.Min(fr.W - b.W, fr.D - b.D);
                    long score = (waste << 16) + (uint)shortSide;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestRectIdx = r;
                        if (score == 0) break;
                    }
                }
            }
            if (bestRectIdx < 0) continue;

            var target = free[bestRectIdx];
            placements.Add(new Placement
            {
                Index = b.Index,
                X = target.X,
                Y = yOffset,
                Z = target.Z,
                W = b.W,
                H = b.H,
                D = b.D
            });

            // Split guillotine: create right strip and bottom strip (in Z).
            // Heuristic: split along larger leftover dimension first to reduce fragmentation.
            free.RemoveAt(bestRectIdx);
            int remW = target.W - b.W;
            int remD = target.D - b.D;
            if (remW < 0 || remD < 0) continue; // safety

            if (remW > 0 && remD > 0)
            {
                if (remW >= remD)
                {
                    // Right then bottom
                    free.Add(new Rect(target.X + b.W, target.Z, remW, target.D));
                    free.Add(new Rect(target.X, target.Z + b.D, b.W, remD));
                }
                else
                {
                    // Bottom then right
                    free.Add(new Rect(target.X, target.Z + b.D, target.W, remD));
                    free.Add(new Rect(target.X + b.W, target.Z, remW, b.D));
                }
            }
            else if (remW > 0)
            {
                free.Add(new Rect(target.X + b.W, target.Z, remW, target.D));
            }
            else if (remD > 0)
            {
                free.Add(new Rect(target.X, target.Z + b.D, target.W, remD));
            }

            PruneFreeRects(free);
        }

        return placements;
    }

    private static void PruneFreeRects(List<Rect> free)
    {
        // Remove contained rectangles and very thin ones; keep list small.
        for (int i = 0; i < free.Count; i++)
        {
            var a = free[i];
            if (a.W <= 0 || a.D <= 0)
            {
                free.RemoveAt(i--);
                continue;
            }
        }

        for (int i = 0; i < free.Count; i++)
        {
            var a = free[i];
            for (int j = 0; j < free.Count; j++)
            {
                if (i == j) continue;
                var b = free[j];
                if (a.X >= b.X && a.Z >= b.Z && a.X + a.W <= b.X + b.W && a.Z + a.D <= b.Z + b.D)
                {
                    free.RemoveAt(i--);
                    break;
                }
            }
        }

        // Cap size defensively for performance on huge N.
        // Keep the largest rectangles by area.
        const int Cap = 4096;
        if (free.Count > Cap)
        {
            free.Sort((r1, r2) => ((long)r2.W * r2.D).CompareTo((long)r1.W * r1.D));
            free.RemoveRange(Cap, free.Count - Cap);
        }
    }

    private static Oriented ChooseOrientation(Box b, int W, int H, int D, Strategy strat)
    {
        // Generate 6 orientations and pick the one that best matches the strategy while fitting.
        Span<(int w, int h, int d)> o = stackalloc (int w, int h, int d)[]
        {
            (b.W, b.H, b.D),
            (b.W, b.D, b.H),
            (b.H, b.W, b.D),
            (b.H, b.D, b.W),
            (b.D, b.W, b.H),
            (b.D, b.H, b.W),
        };

        long bestScore = long.MinValue;
        int best = 0;

        for (int i = 0; i < o.Length; i++)
        {
            int w = o[i].w, h = o[i].h, d = o[i].d;
            if (w > W || h > H || d > D) continue;

            long baseArea = (long)w * d;
            long vol = (long)w * h * d;
            // Score: higher is better.
            long score;
            switch (strat)
            {
                case Strategy.MinHeightThenArea:
                    // Prefer smaller height; among those, bigger base.
                    score = -((long)h << 32) + baseArea;
                    break;
                case Strategy.MinHeightThenVolume:
                    score = -((long)h << 32) + (vol >> 8);
                    break;
                case Strategy.MaxBaseArea:
                    score = (baseArea << 16) - h;
                    break;
                case Strategy.HeightDesc:
                    score = ((long)h << 32) + baseArea;
                    break;
                default:
                    score = (vol << 8) + baseArea - h;
                    break;
            }

            if (score > bestScore)
            {
                bestScore = score;
                best = i;
            }
        }

        return new Oriented(b.Index, o[best].w, o[best].h, o[best].d);
    }

    private static List<string> ReadAllTokens()
    {
        using var sr = new StreamReader(Console.OpenStandardInput());
        string all = sr.ReadToEnd();
        var tokens = new List<string>(all.Length / 2);
        int i = 0;
        while (i < all.Length)
        {
            while (i < all.Length && char.IsWhiteSpace(all[i])) i++;
            if (i >= all.Length) break;
            int j = i + 1;
            while (j < all.Length && !char.IsWhiteSpace(all[j])) j++;
            tokens.Add(all.Substring(i, j - i));
            i = j;
        }
        return tokens;
    }
}
