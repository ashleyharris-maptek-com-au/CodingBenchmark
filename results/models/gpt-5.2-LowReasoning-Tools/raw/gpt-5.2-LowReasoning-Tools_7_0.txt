#include <algorithm>
#include <array>
#include <cfloat>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <vector>

using std::array;
using std::size_t;
using std::uint64_t;

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }

struct AABB {
    Vec3 mn{+DBL_MAX, +DBL_MAX, +DBL_MAX};
    Vec3 mx{-DBL_MAX, -DBL_MAX, -DBL_MAX};
    void expand(const Vec3& p) {
        mn.x = std::min(mn.x, p.x);
        mn.y = std::min(mn.y, p.y);
        mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x);
        mx.y = std::max(mx.y, p.y);
        mx.z = std::max(mx.z, p.z);
    }
    void expand(const AABB& b) { expand(b.mn), expand(b.mx); }
};

struct Tri {
    Vec3 a, b, c;
    AABB box;
    Vec3 centroid;
};

struct BVHNode {
    AABB box;
    int left = -1, right = -1;
    int start = 0, count = 0;
};

static inline bool rayTriX(const Vec3& orig, const Tri& t, double& outT) {
    // Ray direction fixed at (1,0,0). Moller-Trumbore.
    const Vec3 dir{1.0, 0.0, 0.0};
    const Vec3 e1 = t.b - t.a;
    const Vec3 e2 = t.c - t.a;
    const Vec3 pvec = cross(dir, e2);
    const double det = dot(e1, pvec);
    if (std::abs(det) < 1e-15) return false;
    const double invDet = 1.0 / det;
    const Vec3 tvec = orig - t.a;
    const double u = dot(tvec, pvec) * invDet;
    if (u < 0.0 || u > 1.0) return false;
    const Vec3 qvec = cross(tvec, e1);
    const double v = dot(dir, qvec) * invDet;
    if (v < 0.0 || u + v > 1.0) return false;
    const double tparam = dot(e2, qvec) * invDet;
    if (tparam <= 0.0) return false;
    outT = tparam;
    return true;
}

static inline bool rayAabbX(const Vec3& orig, const AABB& b) {
    if (orig.y < b.mn.y || orig.y > b.mx.y) return false;
    if (orig.z < b.mn.z || orig.z > b.mx.z) return false;
    return b.mx.x >= orig.x;
}

struct BVH {
    std::vector<BVHNode> nodes;
    std::vector<int> triIndices;
    const std::vector<Tri>* tris = nullptr;

    int buildNode(int start, int count) {
        BVHNode node;
        node.start = start;
        node.count = count;
        for (int i = 0; i < count; ++i) node.box.expand((*tris)[triIndices[start + i]].box);
        int idx = (int)nodes.size();
        nodes.push_back(node);
        if (count <= 8) return idx;

        AABB cb;
        for (int i = 0; i < count; ++i) cb.expand((*tris)[triIndices[start + i]].centroid);
        Vec3 ext = cb.mx - cb.mn;
        int axis = 0;
        if (ext.y > ext.x && ext.y >= ext.z) axis = 1;
        else if (ext.z > ext.x && ext.z > ext.y) axis = 2;

        int mid = start + count / 2;
        std::nth_element(triIndices.begin() + start, triIndices.begin() + mid, triIndices.begin() + start + count,
                         [&](int ia, int ib) {
                             const Vec3& ca = (*tris)[ia].centroid;
                             const Vec3& cb2 = (*tris)[ib].centroid;
                             if (axis == 0) return ca.x < cb2.x;
                             if (axis == 1) return ca.y < cb2.y;
                             return ca.z < cb2.z;
                         });

        nodes[idx].left = buildNode(start, mid - start);
        nodes[idx].right = buildNode(mid, start + count - mid);
        nodes[idx].count = 0;
        return idx;
    }

    void build(const std::vector<Tri>& inTris) {
        tris = &inTris;
        triIndices.resize((int)inTris.size());
        std::iota(triIndices.begin(), triIndices.end(), 0);
        nodes.clear();
        nodes.reserve(inTris.size() * 2);
        if (!inTris.empty()) buildNode(0, (int)inTris.size());
    }

    void raycastXCollect(const Vec3& orig, std::vector<double>& hits) const {
        hits.clear();
        if (nodes.empty()) return;
        std::vector<int> st;
        st.push_back(0);
        while (!st.empty()) {
            int ni = st.back();
            st.pop_back();
            const BVHNode& n = nodes[ni];
            if (!rayAabbX(orig, n.box)) continue;
            if (n.left == -1) {
                for (int i = 0; i < n.count; ++i) {
                    const Tri& tr = (*tris)[triIndices[n.start + i]];
                    double t;
                    if (rayTriX(orig, tr, t)) hits.push_back(t);
                }
            } else {
                st.push_back(n.left);
                st.push_back(n.right);
            }
        }
    }
};

struct Mesh {
    std::vector<Vec3> v;
    std::vector<array<int, 3>> f;
    std::vector<Tri> tris;
    AABB box;
    void buildTris() {
        box = AABB{};
        for (auto& p : v) box.expand(p);
        tris.clear();
        tris.reserve(f.size());
        for (auto& t : f) {
            Tri tr;
            tr.a = v[t[0]];
            tr.b = v[t[1]];
            tr.c = v[t[2]];
            tr.box = AABB{};
            tr.box.expand(tr.a);
            tr.box.expand(tr.b);
            tr.box.expand(tr.c);
            tr.centroid = (tr.a + tr.b + tr.c) * (1.0 / 3.0);
            tris.push_back(tr);
        }
    }
};

struct BitGrid {
    int nx = 0, ny = 0, nz = 0;
    std::vector<uint64_t> bits;
    void init(int X, int Y, int Z) {
        nx = X, ny = Y, nz = Z;
        uint64_t n = (uint64_t)nx * (uint64_t)ny * (uint64_t)nz;
        bits.assign((size_t)((n + 63ull) / 64ull), 0ull);
    }
    inline uint64_t idx(int x, int y, int z) const {
        return (uint64_t)x + (uint64_t)nx * ((uint64_t)y + (uint64_t)ny * (uint64_t)z);
    }
    inline bool get(int x, int y, int z) const {
        uint64_t i = idx(x, y, z);
        return (bits[(size_t)(i >> 6)] >> (i & 63ull)) & 1ull;
    }
    inline void set(int x, int y, int z, bool val) {
        uint64_t i = idx(x, y, z);
        uint64_t& w = bits[(size_t)(i >> 6)];
        uint64_t m = 1ull << (i & 63ull);
        if (val) w |= m;
        else w &= ~m;
    }
};

static inline int chooseBaseRes(const AABB& world, uint64_t triCountTotal) {
    Vec3 ext = world.mx - world.mn;
    double maxExt = std::max({ext.x, ext.y, ext.z, 1e-9});
    (void)maxExt;
    double t = std::log10((double)std::max<uint64_t>(triCountTotal, 1));
    int base = (int)std::round(64.0 + 24.0 * t);
    base = std::clamp(base, 64, 256);
    const uint64_t maxVox = 256ull * 256ull * 256ull;
    int nx = std::max(8, (int)std::round(base * (ext.x / std::max({ext.x, ext.y, ext.z, 1e-9}))));
    int ny = std::max(8, (int)std::round(base * (ext.y / std::max({ext.x, ext.y, ext.z, 1e-9}))));
    int nz = std::max(8, (int)std::round(base * (ext.z / std::max({ext.x, ext.y, ext.z, 1e-9}))));
    uint64_t vox = (uint64_t)nx * (uint64_t)ny * (uint64_t)nz;
    if (vox <= maxVox) return base;
    double scale = std::cbrt((double)maxVox / (double)vox);
    base = (int)std::floor(base * scale);
    return std::clamp(base, 32, 256);
}

static void voxelizeMesh(const Mesh& m, BitGrid& out, const AABB& world, int baseRes) {
    Vec3 ext = world.mx - world.mn;
    double maxExt = std::max({ext.x, ext.y, ext.z, 1e-9});
    int nx = std::max(8, (int)std::round(baseRes * (ext.x / maxExt)));
    int ny = std::max(8, (int)std::round(baseRes * (ext.y / maxExt)));
    int nz = std::max(8, (int)std::round(baseRes * (ext.z / maxExt)));
    out.init(nx, ny, nz);

    BVH bvh;
    bvh.build(m.tris);

    const double dx = ext.x / nx;
    const double dy = ext.y / ny;
    const double dz = ext.z / nz;
    const double eps = 1e-9 * maxExt;

    std::vector<double> hits;
    hits.reserve(1024);

    for (int kz = 0; kz < nz; ++kz) {
        double z = world.mn.z + (kz + 0.5) * dz;
        for (int jy = 0; jy < ny; ++jy) {
            double y = world.mn.y + (jy + 0.5) * dy;
            Vec3 orig(world.mn.x - 2.0 * dx - 1e-6, y, z);
            bvh.raycastXCollect(orig, hits);
            if (hits.empty()) continue;
            for (double& t : hits) t = orig.x + t;
            std::sort(hits.begin(), hits.end());
            int w = 0;
            for (size_t i = 0; i < hits.size(); ++i) {
                if (w == 0 || std::abs(hits[i] - hits[(size_t)w - 1]) > eps) hits[(size_t)w++] = hits[i];
            }
            hits.resize((size_t)w);
            if (hits.size() & 1u) hits.pop_back();
            for (size_t p = 0; p + 1 < hits.size(); p += 2) {
                double x0 = hits[p];
                double x1 = hits[p + 1];
                if (x1 <= world.mn.x || x0 >= world.mx.x) continue;
                x0 = std::max(x0, world.mn.x);
                x1 = std::min(x1, world.mx.x);
                int ix0 = (int)std::floor((x0 - world.mn.x) / dx);
                int ix1 = (int)std::floor((x1 - world.mn.x) / dx);
                ix0 = std::clamp(ix0, 0, nx - 1);
                ix1 = std::clamp(ix1, 0, nx - 1);
                for (int ix = ix0; ix <= ix1; ++ix) out.set(ix, jy, kz, true);
            }
        }
    }
}

struct OutMesh {
    std::vector<Vec3> v;
    std::vector<array<int, 3>> f;
};
static inline int addVertex(OutMesh& out, const Vec3& p) {
    out.v.push_back(p);
    return (int)out.v.size() - 1;
}

// Marching tetrahedra: split each cube into 6 tetrahedra.
static OutMesh marchingTetrahedra(const BitGrid& g, const AABB& world) {
    OutMesh out;
    Vec3 ext = world.mx - world.mn;
    double dx = ext.x / g.nx, dy = ext.y / g.ny, dz = ext.z / g.nz;

    auto gp = [&](int x, int y, int z) -> Vec3 {
        return {world.mn.x + x * dx, world.mn.y + y * dy, world.mn.z + z * dz};
    };

    // Cube corners indices:
    // 0:(0,0,0) 1:(1,0,0) 2:(1,1,0) 3:(0,1,0) 4:(0,0,1) 5:(1,0,1) 6:(1,1,1) 7:(0,1,1)
    // Tetrahedra (6) defined by 4 corner indices each
    static const int tets[6][4] = {
        {0, 5, 1, 6},
        {0, 1, 2, 6},
        {0, 2, 3, 6},
        {0, 3, 7, 6},
        {0, 7, 4, 6},
        {0, 4, 5, 6},
    };
    // Tetra edges (pairs of vertex indices within tet)
    static const int ePairs[6][2] = {{0,1},{0,2},{0,3},{1,2},{1,3},{2,3}};

    for (int z = 0; z < g.nz - 1; ++z) {
        for (int y = 0; y < g.ny - 1; ++y) {
            for (int x = 0; x < g.nx - 1; ++x) {
                Vec3 cpos[8] = {gp(x, y, z), gp(x + 1, y, z), gp(x + 1, y + 1, z), gp(x, y + 1, z),
                                gp(x, y, z + 1), gp(x + 1, y, z + 1), gp(x + 1, y + 1, z + 1),
                                gp(x, y + 1, z + 1)};
                bool cin[8] = {g.get(x, y, z), g.get(x + 1, y, z), g.get(x + 1, y + 1, z), g.get(x, y + 1, z),
                               g.get(x, y, z + 1), g.get(x + 1, y, z + 1), g.get(x + 1, y + 1, z + 1),
                               g.get(x, y + 1, z + 1)};

                for (int ti = 0; ti < 6; ++ti) {
                    int vidx[4] = {tets[ti][0], tets[ti][1], tets[ti][2], tets[ti][3]};
                    bool inside[4] = {cin[vidx[0]], cin[vidx[1]], cin[vidx[2]], cin[vidx[3]]};
                    int mask = (inside[0] ? 1 : 0) | (inside[1] ? 2 : 0) | (inside[2] ? 4 : 0) | (inside[3] ? 8 : 0);
                    if (mask == 0 || mask == 15) continue;

                    // Collect intersection points on tet edges where inside differs.
                    Vec3 ip[6];
                    int ipCount = 0;
                    for (int e = 0; e < 6; ++e) {
                        int a = ePairs[e][0], b = ePairs[e][1];
                        if (inside[a] == inside[b]) continue;
                        Vec3 p0 = cpos[vidx[a]];
                        Vec3 p1 = cpos[vidx[b]];
                        ip[ipCount++] = (p0 + p1) * 0.5; // binary field => midpoint
                    }
                    if (ipCount < 3) continue;
                    if (ipCount == 3) {
                        int ia = addVertex(out, ip[0]);
                        int ib = addVertex(out, ip[1]);
                        int ic = addVertex(out, ip[2]);
                        out.f.push_back({ia, ib, ic});
                    } else if (ipCount == 4) {
                        int ia = addVertex(out, ip[0]);
                        int ib = addVertex(out, ip[1]);
                        int ic = addVertex(out, ip[2]);
                        int id = addVertex(out, ip[3]);
                        out.f.push_back({ia, ib, ic});
                        out.f.push_back({ia, ic, id});
                    } else {
                        // Rare degeneracy: fan triangulate
                        int i0 = addVertex(out, ip[0]);
                        for (int k = 1; k + 1 < ipCount; ++k) {
                            int i1 = addVertex(out, ip[k]);
                            int i2 = addVertex(out, ip[k + 1]);
                            out.f.push_back({i0, i1, i2});
                        }
                    }
                }
            }
        }
    }
    return out;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.v.resize(VA);
    for (int i = 0; i < VA; ++i) std::cin >> A.v[i].x >> A.v[i].y >> A.v[i].z;
    A.f.resize(FA);
    for (int i = 0; i < FA; ++i) std::cin >> A.f[i][0] >> A.f[i][1] >> A.f[i][2];

    B.v.resize(VB);
    for (int i = 0; i < VB; ++i) std::cin >> B.v[i].x >> B.v[i].y >> B.v[i].z;
    B.f.resize(FB);
    for (int i = 0; i < FB; ++i) std::cin >> B.f[i][0] >> B.f[i][1] >> B.f[i][2];

    A.buildTris();
    B.buildTris();

    AABB world;
    world.expand(A.box);
    world.expand(B.box);
    Vec3 ext = world.mx - world.mn;
    double pad = 1e-6 * std::max({ext.x, ext.y, ext.z, 1.0});
    world.mn = world.mn - Vec3{pad, pad, pad};
    world.mx = world.mx + Vec3{pad, pad, pad};

    uint64_t triTotal = (uint64_t)A.f.size() + (uint64_t)B.f.size();
    int baseRes = chooseBaseRes(world, triTotal);

    BitGrid gA, gB;
    voxelizeMesh(A, gA, world, baseRes);
    voxelizeMesh(B, gB, world, baseRes);

    BitGrid gU;
    gU.init(gA.nx, gA.ny, gA.nz);
    for (size_t i = 0; i < gU.bits.size(); ++i) gU.bits[i] = gA.bits[i] | gB.bits[i];

    OutMesh out = marchingTetrahedra(gU, world);

    std::cout << out.v.size() << " " << out.f.size() << "\n";
    std::cout.setf(std::ios::fixed);
    std::cout.precision(9);
    for (auto& p : out.v) std::cout << p.x << " " << p.y << " " << p.z << "\n";
    for (auto& t : out.f) std::cout << t[0] << " " << t[1] << " " << t[2] << "\n";
    return 0;
}
