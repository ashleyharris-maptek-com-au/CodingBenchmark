use std::cmp::Ordering;
use std::io::{self, Read};

fn popcount(x: u64) -> i32 {
    x.count_ones() as i32
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    assert!(n % 2 == 0, "n must be even");
    assert!(n <= 64, "n up to 64 supported");

    let mut adj: Vec<u64> = vec![0u64; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u] |= 1u64 << v;
        adj[v] |= 1u64 << u;
    }

    let half = n / 2;
    let all_mask: u64 = if n == 64 { !0u64 } else { (1u64 << n) - 1 };

    // Symmetry breaking: force vertex 0 into partition A.
    let fixed_a_mask: u64 = 1u64 << 0;

    // Greedy initial solution for an upper bound
    let mut a_mask_greedy = fixed_a_mask;
    let mut in_a = vec![false; n];
    in_a[0] = true;
    let mut a_cnt = 1usize;
    // score: prefer putting a vertex with more edges to current A into A to reduce future cuts? Actually
    // to reduce cut we want neighbors to stay together; we pick side by comparing edges to current A.
    for v in 1..n {
        if a_cnt == half {
            break;
        }
        let edges_to_a = popcount(adj[v] & a_mask_greedy) as i64;
        let edges_to_b = popcount(adj[v] & (!a_mask_greedy & all_mask)) as i64;
        if edges_to_a >= edges_to_b {
            a_mask_greedy |= 1u64 << v;
            in_a[v] = true;
            a_cnt += 1;
        }
    }
    // fill if not enough
    if a_cnt < half {
        for v in 1..n {
            if a_cnt == half {
                break;
            }
            if (a_mask_greedy >> v) & 1 == 0 {
                a_mask_greedy |= 1u64 << v;
                a_cnt += 1;
            }
        }
    }
    let mut best_cut: i32 = {
        let b_mask = all_mask ^ a_mask_greedy;
        let mut cut = 0i32;
        let mut s = a_mask_greedy;
        while s != 0 {
            let v = s.trailing_zeros() as usize;
            s &= s - 1;
            cut += popcount(adj[v] & b_mask);
        }
        cut
    };
    let mut best_a_mask = a_mask_greedy;

    // Precompute degrees to speed up ordering / bounds.
    let mut degree: Vec<i32> = vec![0; n];
    for v in 0..n {
        degree[v] = popcount(adj[v] & all_mask);
    }

    // Vertex order: process high-degree first (excluding 0 which is fixed).
    let mut order: Vec<usize> = (1..n).collect();
    order.sort_by(|&a, &b| degree[b].cmp(&degree[a]));

    // Side assignment for ordered vertices: 0 unassigned, 1 in A, -1 in B.
    let mut side: Vec<i8> = vec![0; n];
    side[0] = 1;

    fn recurse(
        idx: usize,
        order: &[usize],
        n: usize,
        half: usize,
        all_mask: u64,
        adj: &[u64],
        side: &mut [i8],
        a_mask: u64,
        a_cnt: usize,
        current_cut: i32,
        best_cut: &mut i32,
        best_a_mask: &mut u64,
    ) {
        let processed = idx;
        let remaining = order.len() - processed;
        let b_cnt = processed + 1 - a_cnt; // +1 for vertex 0 fixed in A
        if a_cnt > half || b_cnt > half {
            return;
        }
        // If we must fill all remaining vertices into one side, do it quickly.
        if idx == order.len() {
            if a_cnt == half {
                if current_cut < *best_cut {
                    *best_cut = current_cut;
                    *best_a_mask = a_mask;
                }
            }
            return;
        }

        // Lower bound from edges between unassigned vertices and already assigned vertices, under size constraint.
        // For each unassigned vertex v, define:
        //   cost_if_A = edges to assigned B
        //   cost_if_B = edges to assigned A
        // Any completion must pay at least min(cost_if_A, cost_if_B) for each v,
        // plus extra to satisfy the exact remaining A slots.
        let need_a = (half as i32) - (a_cnt as i32);
        let need_b = (half as i32) - (b_cnt as i32);
        if need_a < 0 || need_b < 0 {
            return;
        }
        // Build masks of assigned A/B among all vertices.
        let assigned_mask = {
            let mut mask = 1u64; // vertex 0
            for j in 0..idx {
                mask |= 1u64 << order[j];
            }
            mask
        };
        let assigned_a_mask = a_mask & assigned_mask;
        let assigned_b_mask = assigned_mask ^ assigned_a_mask;

        let mut base_lb = current_cut;
        let mut deltas: Vec<i32> = Vec::with_capacity(remaining);
        for j in idx..order.len() {
            let v = order[j];
            let cost_a = popcount(adj[v] & assigned_b_mask);
            let cost_b = popcount(adj[v] & assigned_a_mask);
            if cost_a <= cost_b {
                base_lb += cost_a;
                deltas.push(cost_b - cost_a); // extra if we put v into B instead of min-choice(A)
            } else {
                base_lb += cost_b;
                deltas.push(cost_a - cost_b); // extra if we put v into A instead of min-choice(B)
            }
        }
        // Adjust for exact need_a slots.
        // Our base chose each vertex's cheaper side independently, which results in some count of A choices.
        // Compute how many of those "min choices" would put vertex in A.
        let mut min_a = 0i32;
        for j in idx..order.len() {
            let v = order[j];
            let cost_a = popcount(adj[v] & assigned_b_mask);
            let cost_b = popcount(adj[v] & assigned_a_mask);
            if cost_a <= cost_b {
                min_a += 1;
            }
        }
        // We need exactly need_a vertices in A among the remaining.
        // If min_a > need_a, we must flip (min_a - need_a) vertices from A->B, paying their delta.
        // If min_a < need_a, we must flip (need_a - min_a) vertices from B->A, paying their delta.
        let flip = (min_a - need_a).abs();
        if flip > remaining as i32 {
            return;
        }
        if flip > 0 {
            deltas.sort_unstable();
            let mut extra = 0i32;
            for d in deltas.iter().take(flip as usize) {
                extra += *d;
            }
            base_lb += extra;
        }
        if base_lb >= *best_cut {
            return;
        }

        // Choose next vertex
        let v = order[idx];

        // Compute incremental cut if assign v to A or B:
        // Only edges from v to already assigned vertices contribute now.
        // Assigned vertices: 0 plus order[0..idx)
        // If v in A: edges to assigned B add to cut
        // If v in B: edges to assigned A add to cut
        let inc_a = popcount(adj[v] & assigned_b_mask);
        let inc_b = popcount(adj[v] & assigned_a_mask);

        // Branch order: try the side that seems better first.
        // Respect remaining slot limits.
        let can_a = a_cnt < half;
        let can_b = b_cnt < half;
        if can_a && can_b {
            match inc_a.cmp(&inc_b) {
                Ordering::Less | Ordering::Equal => {
                    side[v] = 1;
                    recurse(
                        idx + 1,
                        order,
                        n,
                        half,
                        all_mask,
                        adj,
                        side,
                        a_mask | (1u64 << v),
                        a_cnt + 1,
                        current_cut + inc_a,
                        best_cut,
                        best_a_mask,
                    );
                    side[v] = -1;
                    recurse(
                        idx + 1,
                        order,
                        n,
                        half,
                        all_mask,
                        adj,
                        side,
                        a_mask,
                        a_cnt,
                        current_cut + inc_b,
                        best_cut,
                        best_a_mask,
                    );
                }
                Ordering::Greater => {
                    side[v] = -1;
                    recurse(
                        idx + 1,
                        order,
                        n,
                        half,
                        all_mask,
                        adj,
                        side,
                        a_mask,
                        a_cnt,
                        current_cut + inc_b,
                        best_cut,
                        best_a_mask,
                    );
                    side[v] = 1;
                    recurse(
                        idx + 1,
                        order,
                        n,
                        half,
                        all_mask,
                        adj,
                        side,
                        a_mask | (1u64 << v),
                        a_cnt + 1,
                        current_cut + inc_a,
                        best_cut,
                        best_a_mask,
                    );
                }
            }
            side[v] = 0;
        } else if can_a {
            side[v] = 1;
            recurse(
                idx + 1,
                order,
                n,
                half,
                all_mask,
                adj,
                side,
                a_mask | (1u64 << v),
                a_cnt + 1,
                current_cut + inc_a,
                best_cut,
                best_a_mask,
            );
            side[v] = 0;
        } else if can_b {
            side[v] = -1;
            recurse(
                idx + 1,
                order,
                n,
                half,
                all_mask,
                adj,
                side,
                a_mask,
                a_cnt,
                current_cut + inc_b,
                best_cut,
                best_a_mask,
            );
            side[v] = 0;
        }
    }

    // Start recursion
    recurse(
        0,
        &order,
        n,
        half,
        all_mask,
        &adj,
        &mut side,
        fixed_a_mask,
        1,
        0,
        &mut best_cut,
        &mut best_a_mask,
    );

    // Output best solution
    println!("{}", best_cut);
    let mut verts_a: Vec<usize> = Vec::with_capacity(half);
    let mut mask = best_a_mask;
    while mask != 0 {
        let v = mask.trailing_zeros() as usize;
        mask &= mask - 1;
        verts_a.push(v);
    }
    verts_a.sort_unstable();
    debug_assert_eq!(verts_a.len(), half);
    for (i, v) in verts_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
