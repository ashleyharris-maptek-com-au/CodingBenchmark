use std::cmp::Ordering;
use std::collections::{HashMap, HashSet};
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Aabb {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl Aabb {
    fn empty() -> Self {
        Self {
            minx: f64::INFINITY,
            miny: f64::INFINITY,
            maxx: f64::NEG_INFINITY,
            maxy: f64::NEG_INFINITY,
        }
    }
    fn from_pts(pts: &[Pt]) -> Self {
        let mut a = Self::empty();
        for &p in pts {
            a.minx = a.minx.min(p.x);
            a.miny = a.miny.min(p.y);
            a.maxx = a.maxx.max(p.x);
            a.maxy = a.maxy.max(p.y);
        }
        a
    }
    fn w(&self) -> f64 {
        self.maxx - self.minx
    }
    fn h(&self) -> f64 {
        self.maxy - self.miny
    }
    fn intersects(&self, o: &Aabb) -> bool {
        self.minx <= o.maxx
            && self.maxx >= o.minx
            && self.miny <= o.maxy
            && self.maxy >= o.miny
    }
    fn expand(&self, pad: f64) -> Aabb {
        Aabb {
            minx: self.minx - pad,
            miny: self.miny - pad,
            maxx: self.maxx + pad,
            maxy: self.maxy + pad,
        }
    }
}

fn area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    if n < 3 {
        return 0.0;
    }
    let mut s = 0.0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        s += a.x * b.y - b.x * a.y;
    }
    0.5 * s
}

fn rotate_translate(poly: &[Pt], angle: f64, dx: f64, dy: f64) -> Vec<Pt> {
    let (sa, ca) = angle.sin_cos();
    poly.iter()
        .map(|&p| Pt {
            x: p.x * ca - p.y * sa + dx,
            y: p.x * sa + p.y * ca + dy,
        })
        .collect()
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let minx = a.x.min(b.x) - 1e-12;
    let maxx = a.x.max(b.x) + 1e-12;
    let miny = a.y.min(b.y) - 1e-12;
    let maxy = a.y.max(b.y) + 1e-12;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn seg_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    // Proper + collinear intersection.
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);

    let eps = 1e-12;
    if (o1 > eps && o2 < -eps) || (o1 < -eps && o2 > eps) {
        if (o3 > eps && o4 < -eps) || (o3 < -eps && o4 > eps) {
            return true;
        }
    }
    if o1.abs() <= eps && on_segment(a, b, c) {
        return true;
    }
    if o2.abs() <= eps && on_segment(a, b, d) {
        return true;
    }
    if o3.abs() <= eps && on_segment(c, d, a) {
        return true;
    }
    if o4.abs() <= eps && on_segment(c, d, b) {
        return true;
    }
    false
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    // Ray casting with boundary treated as inside.
    let n = poly.len();
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let o = orient(a, b, p);
        if o.abs() <= 1e-12 && on_segment(a, b, p) {
            return true;
        }
        let cond = (a.y > p.y) != (b.y > p.y);
        if cond {
            let xint = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if xint >= p.x - 1e-12 {
                inside = !inside;
            }
        }
    }
    inside
}

fn poly_edges(poly: &[Pt]) -> impl Iterator<Item = (Pt, Pt)> + '_ {
    poly.iter()
        .enumerate()
        .map(move |(i, &a)| (a, poly[(i + 1) % poly.len()]))
}

fn polys_overlap(a: &[Pt], ab: &Aabb, b: &[Pt], bb: &Aabb) -> bool {
    if !ab.intersects(bb) {
        return false;
    }
    // Edge intersections
    for (a1, a2) in poly_edges(a) {
        for (b1, b2) in poly_edges(b) {
            if seg_intersect(a1, a2, b1, b2) {
                return true;
            }
        }
    }
    // Containment
    if !a.is_empty() && point_in_poly(a[0], b) {
        return true;
    }
    if !b.is_empty() && point_in_poly(b[0], a) {
        return true;
    }
    false
}

fn poly_within_stock(piece: &[Pt], stock: &[Pt]) -> bool {
    // All vertices inside stock and no edge crosses stock boundary.
    for &p in piece {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    // If any piece edge intersects stock edge at a non-boundary way, reject.
    for (p1, p2) in poly_edges(piece) {
        for (s1, s2) in poly_edges(stock) {
            if seg_intersect(p1, p2, s1, s2) {
                // Allow touching on boundary (collinear or endpoint) by checking midpoints.
                // If midpoint of a tiny segment around intersection is outside, it's crossing.
                // Practical robust check: sample midpoint of the piece edge and require inside.
                let mid = Pt {
                    x: (p1.x + p2.x) * 0.5,
                    y: (p1.y + p2.y) * 0.5,
                };
                if !point_in_poly(mid, stock) {
                    return false;
                }
            }
        }
    }
    true
}

#[derive(Clone)]
struct Piece {
    id: usize,
    poly: Vec<Pt>,
    area_abs: f64,
    aabb: Aabb,
}

#[derive(Clone)]
struct Placed {
    poly: Vec<Pt>,
    aabb: Aabb,
    _id: usize,
}

struct StockState {
    placed: Vec<Placed>,
    grid: HashMap<(i32, i32), Vec<usize>>,
    cursor_x: f64,
    cursor_y: f64,
    row_h: f64,
    stock_bbox: Aabb,
    cell: f64,
}

impl StockState {
    fn new(stock_bbox: Aabb) -> Self {
        let cell = ((stock_bbox.w().max(stock_bbox.h()) / 32.0).max(5.0)).min(200.0);
        Self {
            placed: Vec::new(),
            grid: HashMap::new(),
            cursor_x: stock_bbox.minx,
            cursor_y: stock_bbox.miny,
            row_h: 0.0,
            stock_bbox,
            cell,
        }
    }
    fn cells_for(&self, a: &Aabb) -> (i32, i32, i32, i32) {
        let c = self.cell;
        let x0 = ((a.minx - self.stock_bbox.minx) / c).floor() as i32;
        let x1 = ((a.maxx - self.stock_bbox.minx) / c).floor() as i32;
        let y0 = ((a.miny - self.stock_bbox.miny) / c).floor() as i32;
        let y1 = ((a.maxy - self.stock_bbox.miny) / c).floor() as i32;
        (x0, x1, y0, y1)
    }
    fn candidates(&self, a: &Aabb) -> Vec<usize> {
        let (x0, x1, y0, y1) = self.cells_for(a);
        let mut set = HashSet::new();
        for y in y0..=y1 {
            for x in x0..=x1 {
                if let Some(v) = self.grid.get(&(x, y)) {
                    for &idx in v {
                        set.insert(idx);
                    }
                }
            }
        }
        set.into_iter().collect()
    }
    fn insert_grid(&mut self, idx: usize, a: &Aabb) {
        let (x0, x1, y0, y1) = self.cells_for(a);
        for y in y0..=y1 {
            for x in x0..=x1 {
                self.grid.entry((x, y)).or_default().push(idx);
            }
        }
    }
}

fn build_angles(np: usize) -> Vec<f64> {
    if np >= 2000 {
        vec![0.0]
    } else if np >= 400 {
        vec![0.0, PI * 0.5, PI, PI * 1.5]
    } else if np >= 80 {
        let step = 15.0_f64.to_radians();
        (0..24).map(|i| i as f64 * step).collect()
    } else if np >= 10 {
        let step = 10.0_f64.to_radians();
        (0..36).map(|i| i as f64 * step).collect()
    } else {
        let step = 5.0_f64.to_radians();
        (0..72).map(|i| i as f64 * step).collect()
    }
}

fn try_place_in_stock(
    stock_poly: &[Pt],
    stock_state: &mut StockState,
    piece: &Piece,
    angles: &[f64],
) -> Option<(f64, f64, f64)> {
    let pad = 1e-9;
    let bbox = stock_state.stock_bbox;
    let mut attempts = 0usize;

    // Start from cursor (shelf heuristic).
    let mut base_positions = Vec::with_capacity(6);
    base_positions.push((stock_state.cursor_x, stock_state.cursor_y));
    base_positions.push((bbox.minx, stock_state.cursor_y));
    base_positions.push((stock_state.cursor_x, bbox.miny));
    base_positions.push((bbox.minx, bbox.miny));
    // A couple more positions to escape local jams.
    base_positions.push((bbox.minx, stock_state.cursor_y + stock_state.row_h));
    base_positions.push((stock_state.cursor_x, stock_state.cursor_y + stock_state.row_h));

    for &ang in angles {
        // Pre-rotate once for faster offsetting.
        let rot = rotate_translate(&piece.poly, ang, 0.0, 0.0);
        let ra = Aabb::from_pts(&rot);

        // Quick reject if rotated bbox can't fit stock bbox at all.
        if ra.w() > bbox.w() + 1e-9 || ra.h() > bbox.h() + 1e-9 {
            continue;
        }

        for &(bx, by) in &base_positions {
            // Align rotated bbox min to desired pos
            let dx0 = bx - ra.minx;
            let dy0 = by - ra.miny;

            // Generate a small neighborhood of candidate positions.
            let steps = [0.0, ra.w() * 0.05, ra.w() * 0.15];
            let ysteps = [0.0, ra.h() * 0.05, ra.h() * 0.15];
            for &sx in &steps {
                for &sy in &ysteps {
                    let dx = dx0 + sx;
                    let dy = dy0 + sy;
                    let placed = rot.iter().map(|&p| Pt { x: p.x + dx, y: p.y + dy }).collect::<Vec<_>>();
                    let pa = Aabb::from_pts(&placed);
                    if pa.minx < bbox.minx - 1e-9
                        || pa.miny < bbox.miny - 1e-9
                        || pa.maxx > bbox.maxx + 1e-9
                        || pa.maxy > bbox.maxy + 1e-9
                    {
                        continue;
                    }
                    attempts += 1;
                    if attempts > 2500 {
                        return None;
                    }
                    if !poly_within_stock(&placed, stock_poly) {
                        continue;
                    }
                    // Overlap check via grid.
                    let cand = stock_state.candidates(&pa.expand(pad));
                    let mut ok = true;
                    for idx in cand {
                        let other = &stock_state.placed[idx];
                        if polys_overlap(&placed, &pa, &other.poly, &other.aabb) {
                            ok = false;
                            break;
                        }
                    }
                    if ok {
                        return Some((dx, dy, ang));
                    }
                }
            }
        }
    }
    None
}

fn parse_all_numbers(input: &str) -> Vec<f64> {
    let mut out = Vec::new();
    let mut cur = String::new();
    for ch in input.chars() {
        if ch.is_ascii_digit() || ch == '-' || ch == '+' || ch == '.' || ch == 'e' || ch == 'E' {
            cur.push(ch);
        } else if !cur.is_empty() {
            if let Ok(v) = cur.parse::<f64>() {
                out.push(v);
            }
            cur.clear();
        }
    }
    if !cur.is_empty() {
        if let Ok(v) = cur.parse::<f64>() {
            out.push(v);
        }
    }
    out
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let nums = parse_all_numbers(&s);
    if nums.len() < 2 {
        return;
    }
    let mut it = nums.into_iter();
    let sv = it.next().unwrap().round() as usize;
    let np = it.next().unwrap().round() as usize;

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = it.next().unwrap_or(0.0);
        let y = it.next().unwrap_or(0.0);
        stock.push(Pt { x, y });
    }
    let stock_bbox = Aabb::from_pts(&stock);

    let mut pieces = Vec::with_capacity(np);
    for pid in 0..np {
        let pv = it.next().unwrap_or(0.0).round() as usize;
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = it.next().unwrap_or(0.0);
            let y = it.next().unwrap_or(0.0);
            poly.push(Pt { x, y });
        }
        let a = area(&poly).abs();
        let bb = Aabb::from_pts(&poly);
        pieces.push(Piece {
            id: pid,
            poly,
            area_abs: a,
            aabb: bb,
        });
    }

    // Sort by descending area, tie-break by bbox max dimension.
    pieces.sort_by(|a, b| {
        b.area_abs
            .partial_cmp(&a.area_abs)
            .unwrap_or(Ordering::Equal)
            .then_with(|| {
                let ad = a.aabb.w().max(a.aabb.h());
                let bd = b.aabb.w().max(b.aabb.h());
                bd.partial_cmp(&ad).unwrap_or(Ordering::Equal)
            })
    });

    let angles = build_angles(np);
    let mut stocks: Vec<StockState> = Vec::new();
    stocks.push(StockState::new(stock_bbox));

    let mut out_stock_idx = vec![0usize; np];
    let mut out_dx = vec![0.0f64; np];
    let mut out_dy = vec![0.0f64; np];
    let mut out_ang = vec![0.0f64; np];

    for piece in pieces {
        let mut placed = None;
        for (si, st) in stocks.iter_mut().enumerate() {
            if let Some((dx, dy, ang)) = try_place_in_stock(&stock, st, &piece, &angles) {
                placed = Some((si, dx, dy, ang));
                break;
            }
        }
        let (si, dx, dy, ang) = if let Some(v) = placed {
            v
        } else {
            // New stock.
            let mut st = StockState::new(stock_bbox);
            let v = try_place_in_stock(&stock, &mut st, &piece, &angles)
                .unwrap_or((stock_bbox.minx - piece.aabb.minx, stock_bbox.miny - piece.aabb.miny, 0.0));
            stocks.push(st);
            (stocks.len() - 1, v.0, v.1, v.2)
        };

        // Record and commit placement into chosen stock.
        let poly_t = rotate_translate(&piece.poly, ang, dx, dy);
        let aabb_t = Aabb::from_pts(&poly_t);
        let st = &mut stocks[si];
        let idx = st.placed.len();
        st.placed.push(Placed {
            poly: poly_t,
            aabb: aabb_t,
            _id: piece.id,
        });
        st.insert_grid(idx, &aabb_t);

        // Advance shelf cursor (simple packing on stock bbox).
        let pw = aabb_t.w();
        let ph = aabb_t.h();
        if st.cursor_x + pw <= st.stock_bbox.maxx + 1e-9 {
            st.cursor_x = st.cursor_x.max(aabb_t.maxx + 1e-6);
            st.row_h = st.row_h.max(ph + 1e-6);
        } else {
            st.cursor_x = st.stock_bbox.minx;
            st.cursor_y = (st.cursor_y + st.row_h).min(st.stock_bbox.maxy);
            st.row_h = ph + 1e-6;
        }

        out_stock_idx[piece.id] = si;
        out_dx[piece.id] = dx;
        out_dy[piece.id] = dy;
        out_ang[piece.id] = ang;
    }

    // Output
    let mut out = String::new();
    out.push_str(&format!("{}\n", stocks.len()));
    for i in 0..np {
        out.push_str(&format!(
            "{} {:.10} {:.10} {:.10}\n",
            out_stock_idx[i], out_dx[i], out_dy[i], out_ang[i]
        ));
    }
    print!("{out}");
}
