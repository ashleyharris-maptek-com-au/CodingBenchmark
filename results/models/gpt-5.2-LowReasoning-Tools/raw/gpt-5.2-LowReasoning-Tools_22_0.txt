using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    // Board is stored bottom-up: row 0 is the bottom.
    private sealed class Board
    {
        public readonly int W;
        public readonly int H;
        public readonly ulong[] Rows; // length H, bits within [0..W-1]
        public int[] ColHeights;      // cached column heights [0..W-1]
        public int MaxHeight;         // cached maximum height

        public Board(int w, int h)
        {
            W = w;
            H = h;
            Rows = new ulong[h];
            ColHeights = new int[w];
            MaxHeight = 0;
        }

        public Board Clone()
        {
            var b = new Board(W, H);
            Array.Copy(Rows, b.Rows, H);
            b.ColHeights = (int[])ColHeights.Clone();
            b.MaxHeight = MaxHeight;
            return b;
        }

        public bool IsGameOver => MaxHeight >= H;

        public bool CanPlace(in PieceRotation rot, int x, int y)
        {
            if (x < 0 || x + rot.Width > W) return false;
            if (y < 0) return false;
            if (y + rot.Height > H) return false;

            for (int i = 0; i < rot.Cells.Length; i++)
            {
                var c = rot.Cells[i];
                int cx = x + c.X;
                int cy = y + c.Y;
                ulong mask = 1UL << cx;
                if ((Rows[cy] & mask) != 0) return false;
            }
            return true;
        }

        public int FindDropY(in PieceRotation rot, int x)
        {
            // Start as high as possible within the visible board and drop down.
            int y = H - rot.Height;
            if (!CanPlace(rot, x, y))
            {
                // If it can't even fit at the top, it's game over for that placement.
                return -1;
            }
            while (y > 0 && CanPlace(rot, x, y - 1)) y--;
            return y;
        }

        public int PlaceAndClear(in PieceRotation rot, int x, int y)
        {
            for (int i = 0; i < rot.Cells.Length; i++)
            {
                var c = rot.Cells[i];
                int cx = x + c.X;
                int cy = y + c.Y;
                Rows[cy] |= (1UL << cx);
            }

            // Clear full lines.
            ulong fullMask = (W == 64) ? ulong.MaxValue : ((1UL << W) - 1UL);
            int write = 0;
            int cleared = 0;
            for (int read = 0; read < H; read++)
            {
                if ((Rows[read] & fullMask) == fullMask)
                {
                    cleared++;
                    continue;
                }
                if (write != read) Rows[write] = Rows[read];
                write++;
            }
            for (int r = write; r < H; r++) Rows[r] = 0;

            RecomputeHeights();
            return cleared;
        }

        public void RecomputeHeights()
        {
            Array.Clear(ColHeights, 0, ColHeights.Length);
            int max = 0;
            for (int x = 0; x < W; x++)
            {
                int h = 0;
                ulong bit = 1UL << x;
                for (int y = H - 1; y >= 0; y--)
                {
                    if ((Rows[y] & bit) != 0)
                    {
                        h = y + 1;
                        break;
                    }
                }
                ColHeights[x] = h;
                if (h > max) max = h;
            }
            MaxHeight = max;
        }

        public ulong Hash64()
        {
            // FNV-1a over rows (fits well enough for cache keys).
            unchecked
            {
                ulong h = 1469598103934665603UL;
                for (int i = 0; i < H; i++)
                {
                    h ^= Rows[i];
                    h *= 1099511628211UL;
                }
                return h ^ (ulong)W ^ ((ulong)H << 32);
            }
        }
    }

    private readonly struct Cell
    {
        public readonly int X;
        public readonly int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    private readonly struct PieceRotation
    {
        public readonly Cell[] Cells;
        public readonly int Width;
        public readonly int Height;
        public PieceRotation(Cell[] cells, int width, int height)
        {
            Cells = cells;
            Width = width;
            Height = height;
        }
    }

    private static readonly Dictionary<char, PieceRotation[]> Rotations = BuildRotations();
    private static readonly char[] AllPieceTypes = new[] { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };

    private sealed class FastScanner
    {
        private readonly TextReader _r;
        private readonly char[] _buf = new char[1 << 16];
        private int _len, _pos;

        public FastScanner(TextReader r) { _r = r; }

        private int ReadChar()
        {
            if (_pos >= _len)
            {
                _len = _r.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public string? Next()
        {
            int c;
            do
            {
                c = ReadChar();
                if (c < 0) return null;
            } while (c <= 32);

            var sb = new System.Text.StringBuilder(16);
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadChar();
                if (c < 0) break;
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            var s = Next();
            if (s == null) throw new EndOfStreamException();
            return int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
        }
    }

    private static void Main()
    {
        Console.OutputEncoding = System.Text.Encoding.ASCII;
        var scanner = new FastScanner(Console.In);

        string? wTok = scanner.Next();
        if (wTok == null) return;
        int width = int.Parse(wTok, CultureInfo.InvariantCulture);
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt(); // may be large; we stop on EOF/gameover

        var board = new Board(width, height);

        // Small cache used during 2-ply evaluation to avoid recomputing "best move score for (board,nextPiece)".
        var bestMoveCache = new Dictionary<(ulong hash, char piece), double>(capacity: 1 << 14);

        for (int i = 0; i < numPieces; i++)
        {
            string? pTok = scanner.Next();
            if (pTok == null) break;
            char piece = pTok[0];

            // Adaptive strategy:
            // - Most of the time: 1-ply heuristic (fast, stable for marathon).
            // - When the board is "dangerous": 2-ply expectimax over unknown next piece (still bounded).
            bool dangerous = IsDangerous(board);
            int rotationOut, columnOut;

            if (!dangerous)
            {
                (rotationOut, columnOut) = ChooseBestMove1(board, piece);
            }
            else
            {
                (rotationOut, columnOut) = ChooseBestMove2(board, piece, bestMoveCache);
            }

            Console.Write(rotationOut);
            Console.Write(' ');
            Console.WriteLine(columnOut);
            Console.Out.Flush();

            // Read the judge response and advance our internal board.
            string? res = scanner.Next();
            if (res == null) break;

            if (rotationOut < 0 || columnOut < 0)
            {
                // No legal move found; give up gracefully.
                break;
            }

            var rots = Rotations[piece];
            var rot = rots[rotationOut % rots.Length];
            int dropY = board.FindDropY(rot, columnOut);
            if (dropY < 0)
            {
                // Judge should say gameover; stop.
                break;
            }
            board.PlaceAndClear(rot, columnOut, dropY);

            if (res.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
                break;
        }
    }

    private static bool IsDangerous(Board b)
    {
        // Trigger deeper look when close to the top or when there are many holes.
        int max = b.MaxHeight;
        int dangerHeight = (int)(b.H * 0.70);
        if (max >= dangerHeight) return true;
        int holes = CountHoles(b);
        if (holes >= Math.Max(6, b.W / 2)) return true;
        return false;
    }

    private static (int rotation, int column) ChooseBestMove1(Board board, char piece)
    {
        double bestScore = double.NegativeInfinity;
        int bestRot = 0, bestCol = 0;

        var rots = Rotations[piece];
        for (int r = 0; r < rots.Length; r++)
        {
            var rot = rots[r];
            for (int x = 0; x <= board.W - rot.Width; x++)
            {
                int y = board.FindDropY(rot, x);
                if (y < 0) continue;

                var b2 = board.Clone();
                int cleared = b2.PlaceAndClear(rot, x, y);
                double score = Evaluate(b2, cleared);
                if (score > bestScore)
                {
                    bestScore = score;
                    bestRot = r;
                    bestCol = x;
                }
            }
        }

        return (bestRot, bestCol);
    }

    private static (int rotation, int column) ChooseBestMove2(Board board, char piece, Dictionary<(ulong hash, char piece), double> bestMoveCache)
    {
        // 2-ply expectimax: choose move that maximizes:
        //   immediateScore + discount * average_{nextPiece} bestScore(boardAfterMove, nextPiece)
        // To keep it bounded, we compute "bestScore for (state,nextPiece)" with memoization.
        const double discount = 0.95;

        double bestTotal = double.NegativeInfinity;
        int bestRot = 0, bestCol = 0;

        var rots = Rotations[piece];
        for (int r = 0; r < rots.Length; r++)
        {
            var rot = rots[r];
            for (int x = 0; x <= board.W - rot.Width; x++)
            {
                int y = board.FindDropY(rot, x);
                if (y < 0) continue;

                var b2 = board.Clone();
                int cleared = b2.PlaceAndClear(rot, x, y);
                double immediate = Evaluate(b2, cleared);

                // Expected best for unknown next piece (uniform over 7).
                double exp = 0.0;
                ulong h = b2.Hash64();
                for (int pi = 0; pi < AllPieceTypes.Length; pi++)
                {
                    char np = AllPieceTypes[pi];
                    if (!bestMoveCache.TryGetValue((h, np), out double bestForNp))
                    {
                        bestForNp = BestScoreForPiece(b2, np);
                        bestMoveCache[(h, np)] = bestForNp;
                    }
                    exp += bestForNp;
                }
                exp /= AllPieceTypes.Length;

                double total = immediate + discount * exp;
                if (total > bestTotal)
                {
                    bestTotal = total;
                    bestRot = r;
                    bestCol = x;
                }
            }
        }

        return (bestRot, bestCol);
    }

    private static double BestScoreForPiece(Board board, char piece)
    {
        double best = double.NegativeInfinity;
        var rots = Rotations[piece];
        for (int r = 0; r < rots.Length; r++)
        {
            var rot = rots[r];
            for (int x = 0; x <= board.W - rot.Width; x++)
            {
                int y = board.FindDropY(rot, x);
                if (y < 0) continue;
                var b2 = board.Clone();
                int cleared = b2.PlaceAndClear(rot, x, y);
                double s = Evaluate(b2, cleared);
                if (s > best) best = s;
            }
        }
        return best;
    }

    private static double Evaluate(Board b, int clearedLines)
    {
        if (b.IsGameOver) return -1e18;

        int aggregateHeight = 0;
        int bumpiness = 0;
        for (int x = 0; x < b.W; x++)
        {
            aggregateHeight += b.ColHeights[x];
            if (x > 0) bumpiness += Math.Abs(b.ColHeights[x] - b.ColHeights[x - 1]);
        }

        int holes = CountHoles(b);
        int wells = WellSums(b);
        int maxHeight = b.MaxHeight;

        // Weights tuned to be robust across widths/heights rather than razor-tuned to 10x20 only.
        // Scale-sensitive terms are normalized by width/height where appropriate.
        double w = b.W;
        double h = b.H;

        double score =
            3.50 * clearedLines
            - 0.35 * (aggregateHeight / w)
            - 0.80 * holes
            - 0.18 * (bumpiness / w)
            - 0.12 * wells
            - 1.00 * SoftDangerPenalty(maxHeight, (int)h);

        return score;
    }

    private static double SoftDangerPenalty(int maxHeight, int H)
    {
        // Encourage keeping the stack low; sharply penalize near-top stacks.
        int warn = (int)(H * 0.65);
        if (maxHeight <= warn) return 0.0;
        int over = maxHeight - warn;
        return over * over * 0.02;
    }

    private static int CountHoles(Board b)
    {
        int holes = 0;
        for (int x = 0; x < b.W; x++)
        {
            bool seenBlock = false;
            ulong bit = 1UL << x;
            for (int y = b.H - 1; y >= 0; y--)
            {
                bool filled = (b.Rows[y] & bit) != 0;
                if (filled) seenBlock = true;
                else if (seenBlock) holes++;
            }
        }
        return holes;
    }

    private static int WellSums(Board b)
    {
        // A "well" is a column lower than both neighbors (or edge neighbor).
        // Sum of well depths is a good proxy for hard-to-fill pits.
        int sum = 0;
        for (int x = 0; x < b.W; x++)
        {
            int left = (x == 0) ? int.MaxValue : b.ColHeights[x - 1];
            int right = (x == b.W - 1) ? int.MaxValue : b.ColHeights[x + 1];
            int minNbr = Math.Min(left, right);
            if (minNbr > b.ColHeights[x])
            {
                int depth = minNbr - b.ColHeights[x];
                // Triangular contribution makes deeper wells much worse.
                sum += depth * (depth + 1) / 2;
            }
        }
        return sum;
    }

    private static Dictionary<char, PieceRotation[]> BuildRotations()
    {
        // Rotations are defined as coordinates within a minimal bounding box (X right, Y up from bottom).
        // We'll canonicalize each rotation by normalizing to min X/Y = 0 and compute width/height.
        var dict = new Dictionary<char, PieceRotation[]>();

        dict['I'] = UniqueRots(new[]
        {
            new []{ new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(3,0) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(0,3) },
            new []{ new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(3,0) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(0,3) },
        });

        dict['O'] = UniqueRots(new[]
        {
            new []{ new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(1,1) },
            new []{ new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(1,1) },
            new []{ new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(1,1) },
            new []{ new Cell(0,0), new Cell(1,0), new Cell(0,1), new Cell(1,1) },
        });

        dict['T'] = UniqueRots(new[]
        {
            new []{ new Cell(0,0), new Cell(1,0), new Cell(2,0), new Cell(1,1) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(1,1) },
            new []{ new Cell(0,1), new Cell(1,1), new Cell(2,1), new Cell(1,0) },
            new []{ new Cell(1,0), new Cell(1,1), new Cell(1,2), new Cell(0,1) },
        });

        dict['S'] = UniqueRots(new[]
        {
            new []{ new Cell(1,0), new Cell(2,0), new Cell(0,1), new Cell(1,1) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(1,1), new Cell(1,2) },
            new []{ new Cell(1,0), new Cell(2,0), new Cell(0,1), new Cell(1,1) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(1,1), new Cell(1,2) },
        });

        dict['Z'] = UniqueRots(new[]
        {
            new []{ new Cell(0,0), new Cell(1,0), new Cell(1,1), new Cell(2,1) },
            new []{ new Cell(1,0), new Cell(0,1), new Cell(1,1), new Cell(0,2) },
            new []{ new Cell(0,0), new Cell(1,0), new Cell(1,1), new Cell(2,1) },
            new []{ new Cell(1,0), new Cell(0,1), new Cell(1,1), new Cell(0,2) },
        });

        dict['J'] = UniqueRots(new[]
        {
            new []{ new Cell(0,0), new Cell(0,1), new Cell(1,0), new Cell(2,0) },
            new []{ new Cell(0,0), new Cell(1,0), new Cell(1,1), new Cell(1,2) },
            new []{ new Cell(0,1), new Cell(1,1), new Cell(2,1), new Cell(2,0) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(1,2) },
        });

        dict['L'] = UniqueRots(new[]
        {
            new []{ new Cell(2,0), new Cell(0,0), new Cell(1,0), new Cell(2,1) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(0,2), new Cell(1,0) },
            new []{ new Cell(0,0), new Cell(0,1), new Cell(1,1), new Cell(2,1) },
            new []{ new Cell(0,2), new Cell(1,0), new Cell(1,1), new Cell(1,2) },
        });

        return dict;
    }

    private static PieceRotation[] UniqueRots(Cell[][] rots)
    {
        var list = new List<PieceRotation>(4);
        var seen = new HashSet<string>(StringComparer.Ordinal);
        for (int i = 0; i < rots.Length; i++)
        {
            var cells = rots[i];
            Normalize(cells, out Cell[] norm, out int w, out int h);
            // Serialize normalized cells to de-dup.
            Array.Sort(norm, (a, b) => (a.Y != b.Y) ? a.Y.CompareTo(b.Y) : a.X.CompareTo(b.X));
            var key = Key(norm);
            if (seen.Add(key))
                list.Add(new PieceRotation(norm, w, h));
        }
        return list.ToArray();
    }

    private static void Normalize(Cell[] cells, out Cell[] norm, out int width, out int height)
    {
        int minX = int.MaxValue, minY = int.MaxValue;
        int maxX = int.MinValue, maxY = int.MinValue;
        for (int i = 0; i < cells.Length; i++)
        {
            var c = cells[i];
            if (c.X < minX) minX = c.X;
            if (c.Y < minY) minY = c.Y;
            if (c.X > maxX) maxX = c.X;
            if (c.Y > maxY) maxY = c.Y;
        }
        width = (maxX - minX) + 1;
        height = (maxY - minY) + 1;
        norm = new Cell[cells.Length];
        for (int i = 0; i < cells.Length; i++)
        {
            var c = cells[i];
            norm[i] = new Cell(c.X - minX, c.Y - minY);
        }
    }

    private static string Key(Cell[] cells)
    {
        // cells must already be sorted
        var sb = new System.Text.StringBuilder(32);
        for (int i = 0; i < cells.Length; i++)
        {
            sb.Append(cells[i].X);
            sb.Append(',');
            sb.Append(cells[i].Y);
            sb.Append(';');
        }
        return sb.ToString();
    }
}
