// Node.js TypeScript solution: multi-drone coverage path planning
// Reads JSON from stdin, outputs {"paths": [...]}
declare const require: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8").trim();
const data = input.length ? JSON.parse(input) : {};

type Drone = { id: string | number; x: number; y: number };
type Cell = [number, number]; // [x,y]

const grid: number[][] = Array.isArray(data.probability) ? data.probability : [];
const drones: Drone[] = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone: number =
  typeof data.maxStepsPerDrone === "number" && isFinite(data.maxStepsPerDrone) ? (data.maxStepsPerDrone | 0) : 0;

const height = grid.length | 0;
const width = height > 0 ? (grid[0].length | 0) : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}
function isFree(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y][x] !== -1;
}
function idx(x: number, y: number): number {
  return y * width + x;
}

const totalCells = width * height;
const visited = new Uint8Array(totalCells); // counts unique covered probability cells

function markVisited(x: number, y: number): void {
  if (!inBounds(x, y)) return;
  const v = grid[y][x];
  if (v > 0) visited[idx(x, y)] = 1;
}

// Precompute neighbor list for each cell for speed
const neighbors: Int32Array = new Int32Array(totalCells * 4);
for (let i = 0; i < neighbors.length; i++) neighbors[i] = -1;
for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const base = idx(x, y) * 4;
    if (!isFree(x, y)) continue;
    // order: up, right, down, left (arbitrary but stable)
    if (isFree(x, y - 1)) neighbors[base + 0] = idx(x, y - 1);
    if (isFree(x + 1, y)) neighbors[base + 1] = idx(x + 1, y);
    if (isFree(x, y + 1)) neighbors[base + 2] = idx(x, y + 1);
    if (isFree(x - 1, y)) neighbors[base + 3] = idx(x - 1, y);
  }
}

function unpack(index: number): { x: number; y: number } {
  const y = (index / width) | 0;
  const x = index - y * width;
  return { x, y };
}

// Per-drone state
type DroneState = {
  x: number;
  y: number;
  stepsLeft: number;
  path: Cell[];
  targetIdx: number; // -1 if none
};

const states: DroneState[] = drones.map((d) => {
  const sx = (d.x | 0);
  const sy = (d.y | 0);
  const safeX = isFree(sx, sy) ? sx : Math.max(0, Math.min(width - 1, sx));
  const safeY = isFree(sx, sy) ? sy : Math.max(0, Math.min(height - 1, sy));
  const st: DroneState = {
    x: safeX,
    y: safeY,
    stepsLeft: Math.max(0, maxStepsPerDrone | 0),
    path: [[safeX, safeY]],
    targetIdx: -1,
  };
  markVisited(safeX, safeY);
  return st;
});

// Reusable BFS buffers (allocated per call size)
function chooseTargetAndNextStep(s: DroneState, lookahead: number): { nextX: number; nextY: number; newTarget: number } {
  const startIdx = idx(s.x, s.y);
  if (lookahead <= 0) return { nextX: s.x, nextY: s.y, newTarget: -1 };

  const dist = new Int16Array(totalCells);
  const parent = new Int32Array(totalCells);
  for (let i = 0; i < totalCells; i++) {
    dist[i] = -1;
    parent[i] = -1;
  }

  const q = new Int32Array(totalCells);
  let qh = 0;
  let qt = 0;
  q[qt++] = startIdx;
  dist[startIdx] = 0;

  let bestIdx = -1;
  let bestScore = -1e100;
  let bestDist = 0;

  while (qh < qt) {
    const cur = q[qh++];
    const dcur = dist[cur];
    if (dcur >= lookahead) continue;

    const base = cur * 4;
    for (let k = 0; k < 4; k++) {
      const nb = neighbors[base + k];
      if (nb < 0) continue;
      if (dist[nb] !== -1) continue;
      dist[nb] = (dcur + 1) as any;
      parent[nb] = cur;
      q[qt++] = nb;
    }
  }

  // Evaluate candidates in discovered set.
  // Score balances value, distance, and a slight preference to unvisited even if low.
  for (let i = 0; i < qt; i++) {
    const cell = q[i];
    const d = dist[cell];
    if (d < 0) continue;
    const { x, y } = unpack(cell);
    const v = grid[y][x];
    if (v <= 0) continue;
    if (visited[cell]) continue;
    // Prefer higher value and closer; mild bias for nearer cells to keep moving.
    const score = (v as number) / (1 + d) + 0.02 * (lookahead - d);
    if (score > bestScore || (score === bestScore && d < bestDist)) {
      bestScore = score;
      bestIdx = cell;
      bestDist = d;
    }
  }

  if (bestIdx === -1) {
    // No unvisited positive cell in radius: greedy local move to adjacent unvisited value, else drift.
    let bestMove = -1;
    let bestV = -1;
    const base = startIdx * 4;
    for (let k = 0; k < 4; k++) {
      const nb = neighbors[base + k];
      if (nb < 0) continue;
      const { x, y } = unpack(nb);
      const v = grid[y][x];
      const bonus = visited[nb] ? 0 : 1;
      const localScore = (v > 0 ? v : 0) + 0.1 * bonus;
      if (localScore > bestV) {
        bestV = localScore;
        bestMove = nb;
      }
    }
    if (bestMove >= 0) {
      const p = unpack(bestMove);
      return { nextX: p.x, nextY: p.y, newTarget: -1 };
    }
    return { nextX: s.x, nextY: s.y, newTarget: -1 };
  }

  // Walk back one step from bestIdx to start to get the immediate next move.
  let step = bestIdx;
  while (parent[step] !== -1 && parent[step] !== startIdx) step = parent[step];
  if (step === startIdx) {
    // Already at target (shouldn't happen often due to visited check, but safe).
    return { nextX: s.x, nextY: s.y, newTarget: bestIdx };
  }
  const nxt = unpack(step);
  return { nextX: nxt.x, nextY: nxt.y, newTarget: bestIdx };
}

function reachableWithin(s: DroneState, target: number, maxDepth: number): boolean {
  if (target < 0) return false;
  if (maxDepth <= 0) return false;
  const startIdx = idx(s.x, s.y);
  if (startIdx === target) return true;
  const dist = new Int16Array(totalCells);
  for (let i = 0; i < totalCells; i++) dist[i] = -1;
  const q = new Int32Array(totalCells);
  let qh = 0, qt = 0;
  q[qt++] = startIdx;
  dist[startIdx] = 0;
  while (qh < qt) {
    const cur = q[qh++];
    const dcur = dist[cur];
    if (dcur >= maxDepth) continue;
    const base = cur * 4;
    for (let k = 0; k < 4; k++) {
      const nb = neighbors[base + k];
      if (nb < 0) continue;
      if (dist[nb] !== -1) continue;
      dist[nb] = (dcur + 1) as any;
      if (nb === target) return true;
      q[qt++] = nb;
    }
  }
  return false;
}

// Round-robin stepping for coordination
const active = states.length;
for (let t = 0; t < maxStepsPerDrone; t++) {
  for (let i = 0; i < active; i++) {
    const s = states[i];
    if (s.stepsLeft <= 0) continue;

    // If current target is invalid, drop it.
    const curTarget = s.targetIdx;
    if (curTarget >= 0) {
      if (visited[curTarget]) s.targetIdx = -1;
      else if (!reachableWithin(s, curTarget, Math.min(s.stepsLeft, 30))) s.targetIdx = -1;
    }

    let nextX = s.x, nextY = s.y;
    if (s.targetIdx >= 0) {
      // Move one step toward current target with a short BFS (small horizon)
      const stepRes = chooseTargetAndNextStep(s, Math.min(s.stepsLeft, 20));
      nextX = stepRes.nextX;
      nextY = stepRes.nextY;
      // keep existing target unless recomputation finds better and sets it
    } else {
      const lookahead = Math.min(s.stepsLeft, 40);
      const stepRes = chooseTargetAndNextStep(s, lookahead);
      nextX = stepRes.nextX;
      nextY = stepRes.nextY;
      s.targetIdx = stepRes.newTarget;
    }

    // Validate move (including stay)
    if (!isFree(nextX, nextY)) {
      nextX = s.x; nextY = s.y;
      s.targetIdx = -1;
    }
    // Ensure move is 4-neighbor or stay
    const md = Math.abs(nextX - s.x) + Math.abs(nextY - s.y);
    if (md > 1) {
      nextX = s.x; nextY = s.y;
      s.targetIdx = -1;
    }

    s.x = nextX;
    s.y = nextY;
    s.stepsLeft--;
    s.path.push([s.x, s.y]);
    markVisited(s.x, s.y);
  }
}

const output = { paths: states.map((s) => s.path) };
process.stdout.write(JSON.stringify(output));
