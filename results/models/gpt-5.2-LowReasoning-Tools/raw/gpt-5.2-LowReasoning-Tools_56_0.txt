const fs = require('fs');

function dist2(ax, ay, bx, by) {
  const dx = ax - bx, dy = ay - by;
  return dx * dx + dy * dy;
}
function dist(ax, ay, bx, by) {
  return Math.hypot(ax - bx, ay - by);
}

function routeLength(order, depotX, depotY, packagesById) {
  let x = depotX, y = depotY;
  let total = 0;
  for (let i = 0; i < order.length; i++) {
    const p = packagesById.get(order[i]);
    total += dist(x, y, p.x, p.y);
    x = p.x; y = p.y;
  }
  total += dist(x, y, depotX, depotY);
  return total;
}

function twoOptImprove(order, depotX, depotY, packagesById, maxPasses, maxSwaps) {
  // Deterministic 2-opt with caps. Works on open path that starts at depot and ends at depot implicitly.
  const n = order.length;
  if (n < 4) return order;

  function getPoint(idx) {
    if (idx === -1) return { x: depotX, y: depotY };
    if (idx === n) return { x: depotX, y: depotY };
    const p = packagesById.get(order[idx]);
    return p;
  }

  let swaps = 0;
  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;
    for (let i = 0; i < n - 2; i++) {
      const a = getPoint(i - 1);
      const b = getPoint(i);
      for (let k = i + 1; k < n - 1; k++) {
        const c = getPoint(k);
        const d = getPoint(k + 1);
        const before = dist2(a.x, a.y, b.x, b.y) + dist2(c.x, c.y, d.x, d.y);
        const after = dist2(a.x, a.y, c.x, c.y) + dist2(b.x, b.y, d.x, d.y);
        if (after + 1e-12 < before) {
          // reverse segment [i..k]
          for (let l = i, r = k; l < r; l++, r--) {
            const tmp = order[l];
            order[l] = order[r];
            order[r] = tmp;
          }
          swaps++;
          improved = true;
          if (swaps >= maxSwaps) return order;
        }
      }
    }
    if (!improved) break;
  }
  return order;
}

function nearestNeighborOrder(cluster, depotX, depotY) {
  // cluster: array of {id,x,y}
  const remaining = cluster.slice();
  const order = [];
  let cx = depotX, cy = depotY;
  while (remaining.length) {
    let bestIdx = 0;
    let bestD = Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const p = remaining[i];
      const d = dist2(cx, cy, p.x, p.y);
      if (d < bestD) {
        bestD = d;
        bestIdx = i;
      }
    }
    const p = remaining[bestIdx];
    order.push(p.id);
    cx = p.x; cy = p.y;
    remaining.splice(bestIdx, 1);
  }
  return order;
}

function sweepClusters(packages, depotX, depotY, k) {
  // Deterministic sweep by angle, then contiguous chunking.
  const arr = packages.map(p => {
    const dx = p.x - depotX;
    const dy = p.y - depotY;
    let ang = Math.atan2(dy, dx);
    if (ang < 0) ang += Math.PI * 2;
    return { ...p, ang, r2: dx * dx + dy * dy };
  });
  arr.sort((a, b) => (a.ang - b.ang) || (a.r2 - b.r2) || (a.id < b.id ? -1 : a.id > b.id ? 1 : 0));

  const n = arr.length;
  const clusters = [];
  if (k <= 1) return [arr];
  const size = Math.ceil(n / k);
  for (let i = 0; i < n; i += size) clusters.push(arr.slice(i, i + size));

  // If chunking created fewer clusters than k due to size rounding, pad with empties.
  while (clusters.length < k) clusters.push([]);
  return clusters;
}

function rebalanceTinyClusters(clusters, minSize) {
  // Move items from largest clusters into tiny clusters to avoid empty routes when possible.
  // Deterministic: always take from the end of the currently largest cluster.
  const allCount = clusters.reduce((s, c) => s + c.length, 0);
  if (allCount === 0) return clusters;
  while (true) {
    let tiny = -1;
    for (let i = 0; i < clusters.length; i++) {
      if (clusters[i].length > 0 && clusters[i].length < minSize) { tiny = i; break; }
    }
    if (tiny === -1) break;
    let largest = -1, bestLen = 0;
    for (let i = 0; i < clusters.length; i++) {
      const len = clusters[i].length;
      if (len > bestLen) { bestLen = len; largest = i; }
    }
    if (largest === -1 || bestLen <= minSize) break;
    clusters[tiny].push(clusters[largest].pop());
  }
  return clusters;
}

function solve(data) {
  const depotX = data.depot[0], depotY = data.depot[1];
  const packages = data.packages || [];
  const n = packages.length;
  const k = Math.max(1, Math.min(data.truckCount | 0, n || 1));

  // Index packages
  const packagesById = new Map();
  for (const p of packages) packagesById.set(p.id, p);

  if (n === 0) return { routes: [] };
  if (k >= n) return { routes: packages.map(p => [p.id]) };

  let clusters = sweepClusters(packages, depotX, depotY, k);
  clusters = rebalanceTinyClusters(clusters, 2);

  const routes = [];
  for (const cluster of clusters) {
    if (!cluster.length) continue;
    const order = nearestNeighborOrder(cluster, depotX, depotY);

    // 2-opt caps scaled to cluster size (keeps runtime controlled).
    const maxPasses = 2;
    const maxSwaps = Math.max(50, Math.floor(order.length * order.length / 8));
    twoOptImprove(order, depotX, depotY, packagesById, maxPasses, maxSwaps);
    routes.push(order);
  }

  // If we ended up with fewer routes than k, it's okay; truckCount is a maximum.
  // Ensure every package appears exactly once (defensive).
  if (routes.length) {
    const seen = new Set();
    const fixed = [];
    for (const r of routes) {
      const nr = [];
      for (const id of r) {
        if (!seen.has(id)) { seen.add(id); nr.push(id); }
      }
      if (nr.length) fixed.push(nr);
    }
    if (seen.size !== n) {
      for (const p of packages) {
        if (!seen.has(p.id)) fixed.push([p.id]);
      }
    }
    return { routes: fixed };
  }

  // Fallback (shouldn't happen)
  return { routes: [packages.map(p => p.id)] };
}

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : { depot: [0, 0], truckCount: 1, packages: [] };
const out = solve(data);
process.stdout.write(JSON.stringify(out));
