// 3D Packing: regular tetrahedra in a (convex) polyhedron container.
// Strategy (fast & scalable, safe):
// - Treat each tetrahedron as its circumscribed sphere (radius R). If spheres do not overlap,
//   tetrahedra cannot overlap (regardless of orientation). We therefore pack spheres.
// - Interpret input vertices as a convex container (convex hull of vertices).
// - Shrink the container by R (offset each supporting plane inward by distance R),
//   so a sphere of radius R centered inside is fully contained.
// - Place sphere centers on an FCC lattice inside the shrunken container.
//
// Output: centers + fixed Euler angles (0,0,0).
//
// Notes:
// - This is conservative (quality not maximal) but robust, very fast, and scales to huge containers.
// - Assumes the intended container is the convex hull of the given vertices (common in such tasks).

#include <algorithm>
#include <array>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <random>
#include <unordered_set>
#include <utility>
#include <vector>

using std::array;
using std::vector;

static constexpr double EPS = 1e-9;

struct Vec3 {
  double x, y, z;
  Vec3() : x(0), y(0), z(0) {}
  Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
  Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
  Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
  Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
  Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
  return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double norm2(const Vec3& a) { return dot(a, a); }
static inline double norm(const Vec3& a) { return std::sqrt(norm2(a)); }

struct Plane {
  Vec3 n;     // outward normal (not necessarily unit)
  double d;   // plane offset: n·x <= d is inside (for outward n)
  double nn;  // |n|
};

struct Face {
  int a, b, c;     // vertex indices (CCW when viewed from outside)
  Plane p;
  vector<int> out; // point indices outside this face
  bool dead = false;
};

static Plane make_plane_outward(const Vec3& A, const Vec3& B, const Vec3& C, const Vec3& inside_point) {
  Vec3 n = cross(B - A, C - A);
  double d = dot(n, A);
  // Ensure inside_point satisfies n·x <= d; if not, flip.
  if (dot(n, inside_point) > d + 1e-10) {
    n = n * -1.0;
    d = -d;
  }
  Plane p;
  p.n = n;
  p.d = d;
  p.nn = norm(n);
  return p;
}

static inline double plane_dist_signed(const Plane& p, const Vec3& x) {
  // positive => outside (violates n·x <= d)
  return dot(p.n, x) - p.d;
}

// QuickHull 3D (incremental), returns triangular faces with consistent outward normals.
static vector<Face> quickhull3d(const vector<Vec3>& pts) {
  const int n = (int)pts.size();
  vector<Face> faces;
  if (n < 4) return faces;

  auto idx_minmax = [&](auto coord_fn) -> std::pair<int, int> {
    int imin = 0, imax = 0;
    double vmin = coord_fn(pts[0]), vmax = vmin;
    for (int i = 1; i < n; i++) {
      double v = coord_fn(pts[i]);
      if (v < vmin) { vmin = v; imin = i; }
      if (v > vmax) { vmax = v; imax = i; }
    }
    return {imin, imax};
  };

  auto [ix0, ix1] = idx_minmax([](const Vec3& v) { return v.x; });
  auto [iy0, iy1] = idx_minmax([](const Vec3& v) { return v.y; });
  auto [iz0, iz1] = idx_minmax([](const Vec3& v) { return v.z; });
  int a = ix0, b = ix1;
  // pick best extreme pair among x/y/z
  auto best_pair = [&](int i, int j) {
    return norm2(pts[j] - pts[i]);
  };
  double best = best_pair(a, b);
  if (best_pair(iy0, iy1) > best) { a = iy0; b = iy1; best = best_pair(a, b); }
  if (best_pair(iz0, iz1) > best) { a = iz0; b = iz1; best = best_pair(a, b); }

  // farthest from line ab
  int c = -1;
  {
    Vec3 ab = pts[b] - pts[a];
    double ab2 = norm2(ab);
    double bestd = -1;
    for (int i = 0; i < n; i++) if (i != a && i != b) {
      Vec3 ap = pts[i] - pts[a];
      Vec3 cr = cross(ab, ap);
      double d2 = norm2(cr) / (ab2 + EPS);
      if (d2 > bestd) { bestd = d2; c = i; }
    }
    if (c < 0 || bestd < 1e-18) return faces; // degenerate
  }

  // farthest from plane abc
  int d = -1;
  Plane basep;
  {
    Vec3 nrm = cross(pts[b] - pts[a], pts[c] - pts[a]);
    double nrmn = norm(nrm);
    if (nrmn < 1e-12) return faces;
    double dd = dot(nrm, pts[a]);
    double bestdist = -1;
    for (int i = 0; i < n; i++) if (i != a && i != b && i != c) {
      double dist = std::abs(dot(nrm, pts[i]) - dd) / (nrmn + EPS);
      if (dist > bestdist) { bestdist = dist; d = i; }
    }
    if (d < 0 || bestdist < 1e-12) return faces; // coplanar
    // We'll orient faces using the tetrahedron centroid as inside point.
    Vec3 inside = (pts[a] + pts[b] + pts[c] + pts[d]) / 4.0;
    basep = make_plane_outward(pts[a], pts[b], pts[c], inside);
  }

  // Create initial tetrahedron faces. Ensure all are outward oriented using centroid.
  Vec3 inside = (pts[a] + pts[b] + pts[c] + pts[d]) / 4.0;
  auto add_face = [&](int i, int j, int k) {
    Face f;
    f.a = i; f.b = j; f.c = k;
    f.p = make_plane_outward(pts[i], pts[j], pts[k], inside);
    faces.push_back(std::move(f));
  };
  add_face(a, b, c);
  add_face(a, d, b);
  add_face(b, d, c);
  add_face(c, d, a);

  // Assign outside points to initial faces
  vector<char> is_initial(n, 0);
  is_initial[a] = is_initial[b] = is_initial[c] = is_initial[d] = 1;
  for (int i = 0; i < n; i++) {
    if (is_initial[i]) continue;
    int bestf = -1;
    double bestout = 0;
    for (int fi = 0; fi < (int)faces.size(); fi++) {
      double sdist = plane_dist_signed(faces[fi].p, pts[i]);
      if (sdist > bestout + 1e-12) { bestout = sdist; bestf = fi; }
    }
    if (bestf >= 0) faces[bestf].out.push_back(i);
  }

  // Edge key for horizon building (directed).
  struct EdgeKey {
    int u, v;
    bool operator==(const EdgeKey& o) const { return u == o.u && v == o.v; }
  };
  struct EdgeHash {
    size_t operator()(const EdgeKey& e) const {
      return (size_t)((uint64_t)(uint32_t)e.u << 32) ^ (uint32_t)e.v;
    }
  };

  auto pick_face_with_out = [&]() -> int {
    for (int i = 0; i < (int)faces.size(); i++) if (!faces[i].dead && !faces[i].out.empty()) return i;
    return -1;
  };

  // Main loop
  while (true) {
    int seedf = pick_face_with_out();
    if (seedf < 0) break;

    // Choose farthest point from this face.
    int eye = -1;
    double bestdist = -1;
    {
      Face& f = faces[seedf];
      for (int pi : f.out) {
        double dist = plane_dist_signed(f.p, pts[pi]);
        if (dist > bestdist) { bestdist = dist; eye = pi; }
      }
    }
    if (eye < 0) { faces[seedf].out.clear(); continue; }

    // Find all faces visible from eye.
    vector<int> visible;
    visible.reserve(64);
    for (int fi = 0; fi < (int)faces.size(); fi++) {
      if (faces[fi].dead) continue;
      if (plane_dist_signed(faces[fi].p, pts[eye]) > 1e-10) visible.push_back(fi);
    }
    if (visible.empty()) {
      // numerically inside: remove from any outside lists
      for (auto& f : faces) {
        if (f.dead) continue;
        auto& o = f.out;
        o.erase(std::remove(o.begin(), o.end(), eye), o.end());
      }
      continue;
    }

    // Build horizon: boundary edges between visible and non-visible faces.
    // We'll collect directed edges (u->v) from visible faces; if reverse exists, they cancel.
    std::unordered_set<EdgeKey, EdgeHash> horizon_set;
    horizon_set.reserve(visible.size() * 6);

    auto add_or_cancel = [&](int u, int v) {
      EdgeKey rev{v, u};
      auto it = horizon_set.find(rev);
      if (it != horizon_set.end()) {
        horizon_set.erase(it);
      } else {
        horizon_set.insert(EdgeKey{u, v});
      }
    };

    vector<int> reassign_points;
    reassign_points.reserve(1024);
    for (int fi : visible) {
      Face& f = faces[fi];
      add_or_cancel(f.a, f.b);
      add_or_cancel(f.b, f.c);
      add_or_cancel(f.c, f.a);
      for (int pi : f.out) if (pi != eye) reassign_points.push_back(pi);
      f.out.clear();
      f.dead = true;
    }

    // Compute a stable inside point (centroid of all hull vertices so far approximated by pts[a]+...):
    // Use average of initial tetrahedron vertices + eye (good enough for orientation).
    Vec3 orient_inside = inside;
    orient_inside = (orient_inside * 0.8) + (pts[eye] * 0.2);

    // Create new faces from horizon edges to eye.
    vector<int> new_face_ids;
    new_face_ids.reserve(horizon_set.size());
    faces.reserve(faces.size() + horizon_set.size());
    for (const auto& e : horizon_set) {
      Face nf;
      nf.a = e.u;
      nf.b = e.v;
      nf.c = eye;
      nf.p = make_plane_outward(pts[nf.a], pts[nf.b], pts[nf.c], orient_inside);
      faces.push_back(std::move(nf));
      new_face_ids.push_back((int)faces.size() - 1);
    }

    // Reassign points to new faces (or existing non-dead, but new is enough for correctness).
    for (int pi : reassign_points) {
      int bestf = -1;
      double bestout = 0;
      for (int fi : new_face_ids) {
        double sdist = plane_dist_signed(faces[fi].p, pts[pi]);
        if (sdist > bestout + 1e-12) { bestout = sdist; bestf = fi; }
      }
      if (bestf >= 0) faces[bestf].out.push_back(pi);
    }
  }

  // Compact faces and ensure outward normals using hull centroid.
  vector<int> used;
  used.reserve(faces.size() * 3);
  for (auto& f : faces) if (!f.dead) { used.push_back(f.a); used.push_back(f.b); used.push_back(f.c); }
  Vec3 centroid(0, 0, 0);
  if (!used.empty()) {
    for (int idx : used) centroid = centroid + pts[idx];
    centroid = centroid / (double)used.size();
  }

  vector<Face> out_faces;
  out_faces.reserve(faces.size());
  for (auto& f : faces) {
    if (f.dead) continue;
    Face g = f;
    g.out.clear();
    g.dead = false;
    g.p = make_plane_outward(pts[g.a], pts[g.b], pts[g.c], centroid);
    out_faces.push_back(std::move(g));
  }
  return out_faces;
}

static bool inside_shrunk_convex(const vector<Plane>& planes, const Vec3& x) {
  for (const auto& p : planes) {
    if (dot(p.n, x) > p.d + 1e-10) return false;
  }
  return true;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int V;
  double e;
  if (!(std::cin >> V >> e)) return 0;
  vector<Vec3> pts;
  pts.reserve(std::max(0, V));
  for (int i = 0; i < V; i++) {
    double x, y, z;
    std::cin >> x >> y >> z;
    pts.emplace_back(x, y, z);
  }
  if (V < 4) {
    std::cout << 0 << "\n";
    return 0;
  }

  // Tetrahedron circumsphere radius.
  const double R = e * std::sqrt(3.0 / 8.0);

  // Convex hull faces -> planes.
  auto faces = quickhull3d(pts);
  if (faces.empty()) {
    // Degenerate: fall back to bounding box packing (still safe if we treat container as the bbox).
    double minx = pts[0].x, maxx = pts[0].x;
    double miny = pts[0].y, maxy = pts[0].y;
    double minz = pts[0].z, maxz = pts[0].z;
    for (auto& p : pts) {
      minx = std::min(minx, p.x); maxx = std::max(maxx, p.x);
      miny = std::min(miny, p.y); maxy = std::max(maxy, p.y);
      minz = std::min(minz, p.z); maxz = std::max(maxz, p.z);
    }
    minx += R; miny += R; minz += R;
    maxx -= R; maxy -= R; maxz -= R;
    if (minx > maxx || miny > maxy || minz > maxz) { std::cout << 0 << "\n"; return 0; }
    double s = 2.0 * R;
    int nx = (int)std::floor((maxx - minx) / s) + 1;
    int ny = (int)std::floor((maxy - miny) / s) + 1;
    int nz = (int)std::floor((maxz - minz) / s) + 1;
    int64_t K = (int64_t)nx * ny * nz;
    if (K < 0) K = 0;
    const int64_t CAP = 2000000;
    if (K > CAP) {
      double scale = std::cbrt((double)K / (double)CAP);
      s *= scale;
      nx = (int)std::floor((maxx - minx) / s) + 1;
      ny = (int)std::floor((maxy - miny) / s) + 1;
      nz = (int)std::floor((maxz - minz) / s) + 1;
    }
    vector<Vec3> centers;
    centers.reserve((size_t)std::max<int64_t>(0, std::min<int64_t>(CAP, (int64_t)nx * ny * nz)));
    for (int ix = 0; ix < nx; ix++) for (int iy = 0; iy < ny; iy++) for (int iz = 0; iz < nz; iz++) {
      centers.emplace_back(minx + ix * s, miny + iy * s, minz + iz * s);
      if ((int64_t)centers.size() >= CAP) break;
    }
    std::cout << centers.size() << "\n";
    std::cout << std::setprecision(17);
    for (auto& c : centers) {
      std::cout << c.x << " " << c.y << " " << c.z << " 0 0 0\n";
    }
    return 0;
  }

  // Build inward-offset (shrunk) planes by distance R along outward normals.
  vector<Plane> planes;
  planes.reserve(faces.size());
  for (const auto& f : faces) {
    Plane p = f.p;
    if (p.nn < 1e-12) continue;
    p.d -= R * p.nn; // n·x <= d - R|n|
    planes.push_back(p);
  }

  // Bounding box of shrunk region (approx): just bbox of original vertices, then shrink by R.
  double minx = pts[0].x, maxx = pts[0].x;
  double miny = pts[0].y, maxy = pts[0].y;
  double minz = pts[0].z, maxz = pts[0].z;
  for (auto& p : pts) {
    minx = std::min(minx, p.x); maxx = std::max(maxx, p.x);
    miny = std::min(miny, p.y); maxy = std::max(maxy, p.y);
    minz = std::min(minz, p.z); maxz = std::max(maxz, p.z);
  }
  minx += R; miny += R; minz += R;
  maxx -= R; maxy -= R; maxz -= R;
  if (minx > maxx || miny > maxy || minz > maxz) { std::cout << 0 << "\n"; return 0; }

  // FCC lattice parameters for sphere radius R:
  // Nearest-neighbor distance in FCC = a/sqrt(2). Set to 2R => a = 2R*sqrt(2).
  double aLat = 2.0 * R * std::sqrt(2.0);
  array<Vec3, 4> basis = {
      Vec3(0, 0, 0),
      Vec3(0, 0.5, 0.5),
      Vec3(0.5, 0, 0.5),
      Vec3(0.5, 0.5, 0),
  };

  // Estimate count; cap for huge containers.
  const int64_t CAP = 2500000; // keep runtime and output size bounded
  auto est_count = [&](double step) -> long double {
    long double lx = (maxx - minx);
    long double ly = (maxy - miny);
    long double lz = (maxz - minz);
    if (lx <= 0 || ly <= 0 || lz <= 0) return 0;
    long double gx = lx / step + 2;
    long double gy = ly / step + 2;
    long double gz = lz / step + 2;
    return gx * gy * gz * 4.0L;
  };

  double step = aLat;
  long double est = est_count(step);
  if (est > (long double)CAP) {
    // Increase step to reduce candidates ~ cubic.
    double scale = std::cbrt((double)(est / (long double)CAP));
    if (scale < 1.0) scale = 1.0;
    step *= scale;
  }

  // Generate centers.
  vector<Vec3> centers;
  centers.reserve((size_t)std::min<long double>((long double)CAP, est_count(step)));

  // Build a deterministic shuffle of basis order to avoid bias at boundaries.
  array<int, 4> order = {0, 1, 2, 3};
  std::mt19937_64 rng(0xC0FFEEULL);
  std::shuffle(order.begin(), order.end(), rng);

  // Determine index ranges; include one layer padding.
  auto floord = [](double v) -> int64_t { return (int64_t)std::floor(v); };
  int64_t ix0 = floord((minx) / step) - 2, ix1 = floord((maxx) / step) + 2;
  int64_t iy0 = floord((miny) / step) - 2, iy1 = floord((maxy) / step) + 2;
  int64_t iz0 = floord((minz) / step) - 2, iz1 = floord((maxz) / step) + 2;

  auto start = std::chrono::high_resolution_clock::now();
  const double TIME_LIMIT_SEC = 29.0;

  for (int64_t ix = ix0; ix <= ix1; ix++) {
    for (int64_t iy = iy0; iy <= iy1; iy++) {
      for (int64_t iz = iz0; iz <= iz1; iz++) {
        Vec3 cellOrigin(ix * step, iy * step, iz * step);
        for (int bi = 0; bi < 4; bi++) {
          Vec3 b0 = basis[order[bi]];
          Vec3 c = cellOrigin + Vec3(b0.x * step, b0.y * step, b0.z * step);
          if (c.x < minx - 1e-12 || c.x > maxx + 1e-12 ||
              c.y < miny - 1e-12 || c.y > maxy + 1e-12 ||
              c.z < minz - 1e-12 || c.z > maxz + 1e-12) {
            continue;
          }
          if (!inside_shrunk_convex(planes, c)) continue;
          centers.push_back(c);
          if ((int64_t)centers.size() >= CAP) goto done_gen;
        }
      }
    }
    auto now = std::chrono::high_resolution_clock::now();
    double sec = std::chrono::duration<double>(now - start).count();
    if (sec > TIME_LIMIT_SEC) break;
  }
done_gen:

  std::cout << centers.size() << "\n";
  std::cout << std::setprecision(17);
  for (auto& c : centers) {
    std::cout << c.x << " " << c.y << " " << c.z << " 0 0 0\n";
  }
  return 0;
}
