// Minimum Edge Cut (Global Min-Cut) solver for undirected unweighted graphs.
// Adaptive strategy:
//  1) Try to find a bridge (cut=1) quickly when feasible.
//  2) For small graphs, run deterministic Stoer-Wagner (exact).
//  3) Otherwise, run multiple randomized Karger contractions (high probability).
//
// MSVC /std:c++20 compatible. No non-standard headers.

#include <algorithm>
#include <array>
#include <cassert>

struct FastScanner {
  static constexpr size_t BUFSZ = 1u << 20;
  std::array<char, BUFSZ> buf{};
  size_t idx = 0, size = 0;

  inline char readChar() {
    if (idx >= size) {
      size = std::fread(buf.data(), 1, BUFSZ, stdin);
      idx = 0;
      if (size == 0) return 0;
    }
    return buf[idx++];
  }

  template <class T>
  bool readInt(T& out) {
    char c;
    do {
      c = readChar();
      if (!c) return false;
    } while (c <= ' '); // skip whitespace
    bool neg = false;
    if (c == '-') { neg = true; c = readChar(); }
    T val = 0;
    while (c > ' ') {
      val = val * 10 + (c - '0');
      c = readChar();
    }
    out = neg ? -val : val;
    return true;
  }
};

struct Edge {
  int u, v;
};

struct DSU {
  std::vector<int> parent, sz;
  int comps;
  DSU() : comps(0) {}
  explicit DSU(int n) : parent(n), sz(n, 1), comps(n) {
    std::iota(parent.begin(), parent.end(), 0);
  }
  int find(int a) {
    while (parent[a] != a) {
      parent[a] = parent[parent[a]];
      a = parent[a];
    }
    return a;
  }
  bool unite(int a, int b) {
    a = find(a); b = find(b);
    if (a == b) return false;
    if (sz[a] < sz[b]) std::swap(a, b);
    parent[b] = a;
    sz[a] += sz[b];
    --comps;
    return true;
  }
};

static std::vector<char> bridgeCutSide(
    int n,
    const std::vector<Edge>& edges,
    const std::vector<std::vector<std::pair<int,int>>>& adj) {
  // Iterative DFS for bridge-finding (Tarjan).
  std::vector<int> tin(n, -1), low(n, -1), parentV(n, -1), parentE(n, -1);
  std::vector<int> it(n, 0);
  int timer = 0;

  std::vector<int> stack;
  stack.reserve(n);
  stack.push_back(0);
  parentV[0] = 0;

  while (!stack.empty()) {
    int v = stack.back();
    if (tin[v] == -1) {
      tin[v] = low[v] = timer++;
    }
    if (it[v] < (int)adj[v].size()) {
      auto [to, ei] = adj[v][it[v]++];
      if (ei == parentE[v]) continue;
      if (tin[to] != -1) {
        low[v] = std::min(low[v], tin[to]);
      } else {
        parentV[to] = v;
        parentE[to] = ei;
        stack.push_back(to);
      }
    } else {
      stack.pop_back();
      if (v != 0) {
        int p = parentV[v];
        low[p] = std::min(low[p], low[v]);
        if (low[v] > tin[p]) {
          // parentE[v] is a bridge. Return a cut: one side = subtree at v.
          std::vector<char> side(n, 0);
          // Mark subtree using parent pointers: build children lists quickly.
          std::vector<std::vector<int>> children(n);
          for (int i = 1; i < n; ++i) if (parentV[i] != -1 && i != parentV[i]) children[parentV[i]].push_back(i);
          std::vector<int> st;
          st.push_back(v);
          side[v] = 1;
          while (!st.empty()) {
            int x = st.back(); st.pop_back();
            for (int ch : children[x]) {
              side[ch] = 1;
              st.push_back(ch);
            }
          }
          return side;
        }
      }
    }
  }
  return {}; // no bridge found
}

static std::pair<int, std::vector<char>> stoerWagnerMinCutSide(int n, const std::vector<Edge>& edges) {
  // Exact global min-cut for small n using dense weight matrix.
  // Complexity ~O(n^3) for dense, practical for n up to a couple thousand.
  std::vector<std::vector<int>> w(n, std::vector<int>(n, 0));
  for (const auto& e : edges) {
    if (e.u == e.v) continue;
    ++w[e.u][e.v];
    ++w[e.v][e.u];
  }

  std::vector<int> vtx(n);
  std::iota(vtx.begin(), vtx.end(), 0);
  std::vector<std::vector<int>> groups(n);
  for (int i = 0; i < n; ++i) groups[i] = {i};

  int best = std::numeric_limits<int>::max();
  std::vector<char> bestSide;

  std::vector<int> added(n, 0);
  std::vector<int> weight(n, 0);

  int curN = n;
  while (curN > 1) {
    std::fill(added.begin(), added.end(), 0);
    std::fill(weight.begin(), weight.end(), 0);
    int prev = -1;
    int last = -1;

    for (int i = 0; i < curN; ++i) {
      int sel = -1;
      for (int j = 0; j < curN; ++j) {
        int x = vtx[j];
        if (!added[x] && (sel == -1 || weight[x] > weight[sel])) sel = x;
      }
      if (sel == -1) break;
      added[sel] = 1;
      prev = last;
      last = sel;
      for (int j = 0; j < curN; ++j) {
        int x = vtx[j];
        if (!added[x]) weight[x] += w[sel][x];
      }
    }

    int cutVal = weight[last];
    if (cutVal < best) {
      best = cutVal;
      bestSide.assign(n, 0);
      for (int node : groups[last]) bestSide[node] = 1;
    }

    // Merge last into prev.
    if (prev == -1) break;
    for (int x : vtx) {
      if (x == prev || x == last) continue;
      w[prev][x] += w[last][x];
      w[x][prev] = w[prev][x];
    }
    // absorb group
    groups[prev].insert(groups[prev].end(), groups[last].begin(), groups[last].end());
    // remove last from active list
    int pos = -1;
    for (int i = 0; i < curN; ++i) if (vtx[i] == last) { pos = i; break; }
    if (pos != -1) {
      vtx.erase(vtx.begin() + pos);
      --curN;
    } else {
      break;
    }
  }

  return {best, bestSide};
}

static std::pair<int, std::vector<char>> kargerBestCutSide(int n, const std::vector<Edge>& edges) {
  // Repeated Karger contractions. Returns best cut found (high probability).
  // Trials chosen to balance speed vs confidence.
  uint64_t seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
  std::mt19937_64 rng(seed);

  auto runOnce = [&](std::vector<char>& sideOut) -> int {
    DSU dsu(n);
    std::uniform_int_distribution<size_t> dist(0, edges.size() - 1);
    while (dsu.comps > 2) {
      const Edge& e = edges[dist(rng)];
      dsu.unite(e.u, e.v);
    }
    // pick a representative of one component
    int repA = -1;
    for (int i = 0; i < n; ++i) { repA = dsu.find(i); break; }
    sideOut.assign(n, 0);
    for (int i = 0; i < n; ++i) if (dsu.find(i) == repA) sideOut[i] = 1;
    int cut = 0;
    for (const auto& e : edges) if (sideOut[e.u] != sideOut[e.v]) ++cut;
    return cut;
  };

  // Heuristic trial count: more for smaller graphs, capped.
  // For large graphs, a modest number of trials is often enough in practice.
  int trials = 1;
  if (n <= 200) trials = 2000;
  else if (n <= 1000) trials = 800;
  else if (n <= 5000) trials = 300;
  else if (n <= 20000) trials = 120;
  else trials = 40;
  // If very sparse, reduce a bit.
  if ((int64_t)edges.size() < (int64_t)n * 2) trials = std::max(10, trials / 2);

  int best = std::numeric_limits<int>::max();
  std::vector<char> bestSide, tmp;
  for (int t = 0; t < trials; ++t) {
    int cut = runOnce(tmp);
    if (cut < best) {
      best = cut;
      bestSide = tmp;
      if (best == 1) break;
    }
  }
  return {best, bestSide};
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  FastScanner fs;
  int n = 0;
  int64_t m64 = 0;
  if (!fs.readInt(n)) return 0;
  if (!fs.readInt(m64)) return 0;
  if (n <= 1) {
    std::cout << 0 << "\n";
    return 0;
  }
  const int64_t m = m64;

  std::vector<Edge> edges;
  edges.reserve((size_t)std::min<int64_t>(m, 10000000));

  // Build adjacency only when likely feasible (helps detect bridges early).
  bool buildAdj = (n <= 300000) && (m <= 6000000);
  std::vector<std::vector<std::pair<int,int>>> adj;
  if (buildAdj) adj.assign(n, {});

  for (int64_t i = 0; i < m; ++i) {
    int u, v;
    fs.readInt(u); fs.readInt(v);
    edges.push_back({u, v});
    if (buildAdj) {
      int ei = (int)edges.size() - 1;
      adj[u].push_back({v, ei});
      adj[v].push_back({u, ei});
    }
  }

  // Step 1: exact bridge cut if present.
  std::vector<char> side;
  int bestCut = std::numeric_limits<int>::max();
  if (buildAdj) {
    side = bridgeCutSide(n, edges, adj);
    if (!side.empty()) {
      bestCut = 1;
      // output the single bridge edge by scanning.
      // Find any edge crossing this partition; for bridge partition it should be exactly 1.
      int outCount = 0;
      for (const auto& e : edges) if (side[e.u] != side[e.v]) ++outCount;
      if (outCount == 1) {
        std::cout << 1 << "\n";
        for (const auto& e : edges) {
          if (side[e.u] != side[e.v]) { std::cout << e.u << " " << e.v << "\n"; break; }
        }
        return 0;
      }
      // If multi-edge across that partition, continue to exact/heuristic solvers.
    }
  }

  // Step 2: exact Stoer-Wagner for small n (dense matrix).
  // Threshold chosen to keep memory reasonable (n^2 ints).
  const int STOER_N_MAX = 2500; // 2500^2 ints ~ 25e5 -> 100MB
  if (n <= STOER_N_MAX) {
    auto [cutVal, cutSide] = stoerWagnerMinCutSide(n, edges);
    bestCut = cutVal;
    side = std::move(cutSide);
  } else {
    // Step 3: randomized Karger repetitions for larger graphs.
    auto [cutVal, cutSide] = kargerBestCutSide(n, edges);
    bestCut = cutVal;
    side = std::move(cutSide);
  }

  // Output cut edges.
  std::vector<Edge> cutEdges;
  cutEdges.reserve((size_t)bestCut);
  for (const auto& e : edges) {
    if (side[e.u] != side[e.v]) cutEdges.push_back(e);
  }

  // Ensure non-empty proper partition; if something went wrong, fall back to trivial cut.
  int cntSide = 0;
  for (char c : side) cntSide += (c != 0);
  if (cntSide == 0 || cntSide == n) {
    // Fall back: isolate vertex 0.
    side.assign(n, 0);
    side[0] = 1;
    cutEdges.clear();
    for (const auto& e : edges) if (side[e.u] != side[e.v]) cutEdges.push_back(e);
    bestCut = (int)cutEdges.size();
  } else {
    bestCut = (int)cutEdges.size(); // actual count
  }

  std::cout << bestCut << "\n";
  for (const auto& e : cutEdges) std::cout << e.u << " " << e.v << "\n";
  return 0;
}
