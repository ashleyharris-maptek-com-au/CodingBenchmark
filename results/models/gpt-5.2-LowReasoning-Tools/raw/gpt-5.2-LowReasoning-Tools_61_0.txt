declare const require: any;

// Deterministic, scalable heuristic VRP planner:
// 1) Sort packages by polar angle around depot (sweep clustering).
// 2) Split into up to truckCount contiguous clusters.
// 3) For each cluster: build a nearest-neighbor tour (starting from depot).
// 4) Apply a bounded 2-opt improvement pass (deterministic).
//
// Output: {"routes": [[packageId,...], ...]}

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Point = { x: number; y: number };
type Pkg = { id: any; x: number; y: number; _ang: number; _r2: number };

const depotArr: [number, number] = data.depot;
const depot: Point = { x: +depotArr[0], y: +depotArr[1] };
const truckCount: number = Math.max(0, Math.floor(+data.truckCount || 0));
const packagesIn: any[] = Array.isArray(data.packages) ? data.packages : [];

const pkgs: Pkg[] = packagesIn.map((p) => {
  const x = +p.x;
  const y = +p.y;
  const dx = x - depot.x;
  const dy = y - depot.y;
  return {
    id: p.id,
    x,
    y,
    _ang: Math.atan2(dy, dx),
    _r2: dx * dx + dy * dy,
  };
});

// Sort by angle then radius then id string to be deterministic across equal points.
pkgs.sort((a, b) => {
  if (a._ang !== b._ang) return a._ang - b._ang;
  if (a._r2 !== b._r2) return a._r2 - b._r2;
  const as = String(a.id);
  const bs = String(b.id);
  return as < bs ? -1 : as > bs ? 1 : 0;
});

const n = pkgs.length;
const k = Math.min(Math.max(1, truckCount || 1), n || 1);

function dist2(a: Point, b: Point): number {
  const dx = a.x - b.x;
  const dy = a.y - b.y;
  return dx * dx + dy * dy;
}

function dist(a: Point, b: Point): number {
  return Math.hypot(a.x - b.x, a.y - b.y);
}

function buildNearestNeighborTour(cluster: Pkg[]): Pkg[] {
  const m = cluster.length;
  if (m <= 1) return cluster.slice();
  const used = new Array<boolean>(m).fill(false);
  const tour: Pkg[] = [];
  let cur: Point = depot;
  for (let step = 0; step < m; step++) {
    let bestIdx = -1;
    let bestD2 = Infinity;
    for (let i = 0; i < m; i++) {
      if (used[i]) continue;
      const d2 = dist2(cur, cluster[i]);
      if (d2 < bestD2) {
        bestD2 = d2;
        bestIdx = i;
      } else if (d2 === bestD2 && bestIdx !== -1) {
        // deterministic tie-break by id
        const a = String(cluster[i].id);
        const b = String(cluster[bestIdx].id);
        if (a < b) bestIdx = i;
      }
    }
    used[bestIdx] = true;
    const p = cluster[bestIdx];
    tour.push(p);
    cur = p;
  }
  return tour;
}

function twoOptImprove(tour: Pkg[]): void {
  const m = tour.length;
  if (m < 4) return;

  // Bound work for large routes; keep deterministic ordering of checks.
  // For small routes, do a full pass (still O(m^2) but small m).
  const full = m <= 160;
  const maxChecks = full ? Number.MAX_SAFE_INTEGER : 45000;

  let improved = true;
  let checks = 0;
  // A few passes max; large instances get 1 pass.
  const maxPasses = full ? 4 : 1;
  for (let pass = 0; pass < maxPasses && improved; pass++) {
    improved = false;
    for (let i = 0; i < m - 1; i++) {
      const aPoint: Point = i === 0 ? depot : tour[i - 1];
      const bPoint: Point = tour[i];
      for (let j = i + 1; j < m; j++) {
        // edges: (aPoint -> bPoint) and (cPoint -> dPoint)
        const cPoint: Point = tour[j];
        const dPoint: Point = j === m - 1 ? depot : tour[j + 1];

        // Skip adjacent edges (no-op).
        if (j === i) continue;
        if (j === i - 1) continue;

        const before = dist(aPoint, bPoint) + dist(cPoint, dPoint);
        const after = dist(aPoint, cPoint) + dist(bPoint, dPoint);
        checks++;
        if (after + 1e-12 < before) {
          // Reverse segment [i..j]
          for (let l = i, r = j; l < r; l++, r--) {
            const tmp = tour[l];
            tour[l] = tour[r];
            tour[r] = tmp;
          }
          improved = true;
        }
        if (checks >= maxChecks) return;
      }
      if (checks >= maxChecks) return;
    }
  }
}

// Split into k contiguous clusters (angle sweep).
const clusters: Pkg[][] = [];
if (n === 0) {
  clusters.push([]);
} else {
  // Even split sizes: first (n % k) clusters get one extra.
  const base = Math.floor(n / k);
  const extra = n % k;
  let idx = 0;
  for (let c = 0; c < k; c++) {
    const size = base + (c < extra ? 1 : 0);
    clusters.push(pkgs.slice(idx, idx + size));
    idx += size;
  }
}

const routes: any[][] = [];
for (const cluster of clusters) {
  if (cluster.length === 0) continue;
  const tour = buildNearestNeighborTour(cluster);
  twoOptImprove(tour);
  routes.push(tour.map((p) => p.id));
}

process.stdout.write(JSON.stringify({ routes }));
