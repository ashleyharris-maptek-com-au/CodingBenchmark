use std::collections::VecDeque;
use std::io::{self, Read};

#[derive(Clone)]
struct BitSet {
    words: Vec<u64>,
}

impl BitSet {
    fn new(bits: usize) -> Self {
        let words = vec![0u64; (bits + 63) / 64];
        Self { words }
    }
    #[inline]
    fn get(&self, idx: usize) -> bool {
        let w = idx >> 6;
        let b = idx & 63;
        ((self.words[w] >> b) & 1) != 0
    }
    #[inline]
    fn set(&mut self, idx: usize) {
        let w = idx >> 6;
        let b = idx & 63;
        self.words[w] |= 1u64 << b;
    }
}

// Packed 4-bit state per cell (two cells per byte).
// 0 = unvisited
// 1..=4  : visited from A; value = 1 + dir_to_parent (0..3)
// 5..=8  : visited from B; value = 5 + dir_to_parent (0..3)
// 15     : reserved (unused)
#[derive(Clone)]
struct NibbleGrid {
    data: Vec<u8>,
}

impl NibbleGrid {
    fn new(cells: usize) -> Self {
        Self {
            data: vec![0u8; (cells + 1) / 2],
        }
    }
    #[inline]
    fn get(&self, idx: usize) -> u8 {
        let byte = self.data[idx >> 1];
        if (idx & 1) == 0 {
            byte & 0x0F
        } else {
            byte >> 4
        }
    }
    #[inline]
    fn set(&mut self, idx: usize, val: u8) {
        let slot = &mut self.data[idx >> 1];
        if (idx & 1) == 0 {
            *slot = (*slot & 0xF0) | (val & 0x0F);
        } else {
            *slot = (*slot & 0x0F) | ((val & 0x0F) << 4);
        }
    }
}

#[inline]
fn idx_of(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn main() {
    // Read all stdin.
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    if input.is_empty() {
        return;
    }

    // Fast integer parser.
    let mut p = 0usize;
    let next_usize = |buf: &[u8], p: &mut usize| -> Option<usize> {
        while *p < buf.len() && (buf[*p] == b' ' || buf[*p] == b'\n' || buf[*p] == b'\r' || buf[*p] == b'\t') {
            *p += 1;
        }
        if *p >= buf.len() {
            return None;
        }
        let mut n: usize = 0;
        while *p < buf.len() {
            let c = buf[*p];
            if c.is_ascii_digit() {
                n = n * 10 + (c - b'0') as usize;
                *p += 1;
            } else {
                break;
            }
        }
        Some(n)
    };

    let h = match next_usize(&input, &mut p) {
        Some(v) => v,
        None => return,
    };
    let w = match next_usize(&input, &mut p) {
        Some(v) => v,
        None => return,
    };
    if h == 0 || w == 0 {
        return;
    }

    // Consume to start of first row.
    while p < input.len() && input[p] != b'\n' {
        p += 1;
    }
    if p < input.len() && input[p] == b'\n' {
        p += 1;
    }

    let cells = h.saturating_mul(w);
    let mut walls = BitSet::new(cells);
    let mut state = NibbleGrid::new(cells);

    let mut start: Option<(usize, usize)> = None;
    let mut goal: Option<(usize, usize)> = None;

    // Read H rows of width W; tolerate CRLF and short lines by treating missing chars as walls.
    for y in 0..h {
        let row_start = p;
        let mut x = 0usize;
        while x < w && p < input.len() && input[p] != b'\n' {
            let c = input[p];
            if c != b'\r' {
                let idx = idx_of(x, y, w);
                match c {
                    b'#' => walls.set(idx),
                    b'A' => start = Some((x, y)),
                    b'B' => goal = Some((x, y)),
                    b' ' => {}
                    _ => {
                        // Unknown char => treat as wall.
                        walls.set(idx);
                    }
                }
                x += 1;
            }
            p += 1;
        }
        // If line ended early, fill remainder with walls.
        while x < w {
            walls.set(idx_of(x, y, w));
            x += 1;
        }
        // Consume newline.
        while p < input.len() && input[p] != b'\n' {
            p += 1;
        }
        if p < input.len() && input[p] == b'\n' {
            p += 1;
        }
        // Avoid unused warning for row_start in some edge compilers.
        let _ = row_start;
    }

    let Some((sx, sy)) = start else {
        return;
    };
    let Some((gx, gy)) = goal else {
        return;
    };

    if (sx, sy) == (gx, gy) {
        print!("1\n{} {}\n", sx, sy);
        return;
    }

    let s_idx = idx_of(sx, sy, w);
    let g_idx = idx_of(gx, gy, w);
    if walls.get(s_idx) || walls.get(g_idx) {
        // Start/goal inside wall.
        return;
    }

    // Directions: 0=Up,1=Down,2=Left,3=Right
    const DX: [isize; 4] = [0, 0, -1, 1];
    const DY: [isize; 4] = [-1, 1, 0, 0];
    const OPP: [u8; 4] = [1, 0, 3, 2];

    let mut qa: VecDeque<u32> = VecDeque::new();
    let mut qb: VecDeque<u32> = VecDeque::new();
    qa.push_back(s_idx as u32);
    qb.push_back(g_idx as u32);
    state.set(s_idx, 1); // visited A; parent dir arbitrary (0)
    state.set(g_idx, 5); // visited B; parent dir arbitrary (0)

    let mut meet: Option<u32> = None;

    while !qa.is_empty() && !qb.is_empty() {
        // Expand the smaller frontier to reduce work.
        let expand_a = qa.len() <= qb.len();
        let cur_idx_u32 = if expand_a {
            qa.pop_front().unwrap()
        } else {
            qb.pop_front().unwrap()
        };
        let cur = cur_idx_u32 as usize;
        let cx = cur % w;
        let cy = cur / w;

        let cur_state = state.get(cur);
        let from_a = cur_state >= 1 && cur_state <= 4;
        let base = if from_a { 1u8 } else { 5u8 };

        for dir in 0..4usize {
            let nx = cx as isize + DX[dir];
            let ny = cy as isize + DY[dir];
            if nx < 0 || ny < 0 || nx >= w as isize || ny >= h as isize {
                continue;
            }
            let nidx = idx_of(nx as usize, ny as usize, w);
            if walls.get(nidx) {
                continue;
            }
            let st = state.get(nidx);
            if st == 0 {
                // Mark visited from this side; store dir to parent (move back to current).
                state.set(nidx, base + (OPP[dir] as u8));
                if from_a {
                    qa.push_back(nidx as u32);
                } else {
                    qb.push_back(nidx as u32);
                }
            } else {
                let other_from_a = st >= 1 && st <= 4;
                if other_from_a != from_a {
                    meet = Some(nidx as u32);
                    break;
                }
            }
        }
        if meet.is_some() {
            break;
        }
    }

    let Some(midx_u32) = meet else {
        // No path.
        // Output 0 for compatibility with harnesses expecting a line.
        print!("0\n");
        return;
    };
    let midx = midx_u32 as usize;

    // Reconstruct path: A -> ... -> meet -> ... -> B
    let mut left: Vec<(usize, usize)> = Vec::new();
    let mut cur = midx;
    // Walk back to A following A-parent pointers; if meet was discovered from B first, step one to an A cell:
    if !(state.get(cur) >= 1 && state.get(cur) <= 4) {
        // Find an adjacent A-visited cell (the one that caused meeting).
        // This can happen if `meet` was found via an A expansion that hit a B cell.
        // Ensure `cur` is treated as meeting cell anyway by moving to an A neighbor for left half.
        'outer: for dir in 0..4usize {
            let nx = (cur % w) as isize + DX[dir];
            let ny = (cur / w) as isize + DY[dir];
            if nx < 0 || ny < 0 || nx >= w as isize || ny >= h as isize {
                continue;
            }
            let nidx = idx_of(nx as usize, ny as usize, w);
            let st = state.get(nidx);
            if st >= 1 && st <= 4 {
                cur = nidx;
                break 'outer;
            }
        }
    }

    loop {
        left.push((cur % w, cur / w));
        if cur == s_idx {
            break;
        }
        let st = state.get(cur);
        if !(st >= 1 && st <= 4) {
            // Corrupt/unexpected; abort.
            print!("0\n");
            return;
        }
        let dir_to_parent = (st - 1) as usize; // 0..3
        let px = cur % w;
        let py = cur / w;
        let nx = px as isize + DX[dir_to_parent];
        let ny = py as isize + DY[dir_to_parent];
        cur = idx_of(nx as usize, ny as usize, w);
    }
    left.reverse();

    // Right half: from meet to B following B-parent pointers.
    let mut right: Vec<(usize, usize)> = Vec::new();
    cur = midx;
    if !(state.get(cur) >= 5 && state.get(cur) <= 8) {
        // Move to an adjacent B-visited cell for right half.
        'outer2: for dir in 0..4usize {
            let nx = (cur % w) as isize + DX[dir];
            let ny = (cur / w) as isize + DY[dir];
            if nx < 0 || ny < 0 || nx >= w as isize || ny >= h as isize {
                continue;
            }
            let nidx = idx_of(nx as usize, ny as usize, w);
            let st = state.get(nidx);
            if st >= 5 && st <= 8 {
                cur = nidx;
                break 'outer2;
            }
        }
    }
    loop {
        right.push((cur % w, cur / w));
        if cur == g_idx {
            break;
        }
        let st = state.get(cur);
        if !(st >= 5 && st <= 8) {
            print!("0\n");
            return;
        }
        let dir_to_parent = (st - 5) as usize;
        let px = cur % w;
        let py = cur / w;
        let nx = px as isize + DX[dir_to_parent];
        let ny = py as isize + DY[dir_to_parent];
        cur = idx_of(nx as usize, ny as usize, w);
    }

    // Merge, ensuring we include meet if present in both halves only once.
    // left ends at A->...; right starts near meet toward B, but may not include exact meet if we shifted.
    // Ensure continuity by inserting the true meet cell coordinate if missing.
    let meet_coord = (midx % w, midx / w);
    let mut path: Vec<(usize, usize)> = Vec::with_capacity(left.len() + right.len() + 1);
    path.extend_from_slice(&left);
    if *path.last().unwrap() != meet_coord {
        path.push(meet_coord);
    }
    if right.first().copied() == Some(meet_coord) {
        path.extend_from_slice(&right[1..]);
    } else {
        path.extend_from_slice(&right);
    }

    // Output.
    let mut out = String::new();
    out.push_str(&format!("{}\n", path.len()));
    for (x, y) in path {
        out.push_str(&format!("{} {}\n", x, y));
    }
    print!("{out}");
}
