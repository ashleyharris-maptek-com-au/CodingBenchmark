use std::collections::{HashMap, VecDeque};
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let w: i32 = it.next().unwrap().parse().unwrap();
    let h: i32 = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().parse::<i32>().unwrap() != 0;

    if w <= 0 || h <= 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; (w as usize) * (h as usize)];
    for _ in 0..obstacle_count {
        let x: i32 = it.next().unwrap().parse().unwrap();
        let y: i32 = it.next().unwrap().parse().unwrap();
        if 0 <= x && x < w && 0 <= y && y < h {
            blocked[(y as usize) * (w as usize) + (x as usize)] = true;
        }
    }

    let start_idx = 0usize;
    if blocked[start_idx] {
        println!("0");
        return;
    }

    let free_cells = blocked.iter().filter(|&&b| !b).count();
    if free_cells == 0 {
        println!("0");
        return;
    }

    // Fast constructive solver for obstacle-free rectangles.
    if obstacle_count == 0 {
        if let Some(path) = construct_rect_path(w, h, require_cycle) {
            print_path(&path);
        } else {
            println!("0");
        }
        return;
    }

    // Exact solver for small instances with obstacles.
    if free_cells <= 64 {
        match solve_small_exact(w, h, &blocked, require_cycle) {
            Some(path) => print_path(&path),
            None => println!("0"),
        }
        return;
    }

    // General Hamiltonian path with obstacles is NP-complete; for very large instances we
    // only support the guaranteed-constructible obstacle-free case.
    println!("0");
}

fn print_path(path: &[(i32, i32)]) {
    println!("{}", path.len());
    for &(x, y) in path {
        println!("{} {}", x, y);
    }
}

fn construct_rect_path(w: i32, h: i32, require_cycle: bool) -> Option<Vec<(i32, i32)>> {
    let n = (w as i64) * (h as i64);
    if n == 1 {
        if require_cycle {
            return None;
        }
        return Some(vec![(0, 0)]);
    }
    if w == 1 {
        // Single column.
        let mut path = Vec::with_capacity(h as usize);
        for y in 0..h {
            path.push((0, y));
        }
        if require_cycle {
            // End is (0,h-1) adjacent to (0,0) only if h==2.
            if h == 2 {
                return Some(path);
            }
            return None;
        }
        return Some(path);
    }
    if h == 1 {
        // Single row.
        let mut path = Vec::with_capacity(w as usize);
        for x in 0..w {
            path.push((x, 0));
        }
        if require_cycle {
            if w == 2 {
                return Some(path);
            }
            return None;
        }
        return Some(path);
    }

    let mut path = Vec::with_capacity((w as usize) * (h as usize));
    // Standard serpentine row-scan from (0,0).
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push((x, y));
            }
        } else {
            for x in (0..w).rev() {
                path.push((x, y));
            }
        }
    }

    if require_cycle {
        // Hamiltonian cycle in an obstacle-free grid exists iff both dimensions are even.
        // Additionally, for (w,h) >= 2, this construction can be adjusted, but we keep it simple:
        // return a known cycle construction only for even-even.
        if w % 2 != 0 || h % 2 != 0 {
            return None;
        }
        // Build an explicit cycle starting at (0,0) by taking a simple even-even cycle:
        // Traverse first column down, then snake remaining columns up/down, finally return to start adjacency.
        // One known construction: go along a 2-wide perimeter weave.
        return Some(construct_even_even_cycle(w, h));
    }

    Some(path)
}

fn construct_even_even_cycle(w: i32, h: i32) -> Vec<(i32, i32)> {
    // Works for w,h even and >= 2. Produces a Hamiltonian cycle order (as a path list);
    // requirement is only that last is adjacent to (0,0).
    // Construction: traverse columns in a snake, but reserve the top row to stitch closure.
    // 1) Go down first column to bottom.
    // 2) For each pair of columns (x=1..w-1), snake upward/downward excluding (0,0) edge handled.
    // 3) Finish at (1,0), adjacent to (0,0).
    let mut path = Vec::with_capacity((w as usize) * (h as usize));
    path.push((0, 0));
    for y in 1..h {
        path.push((0, y));
    }
    let mut x = 1;
    while x < w {
        // Go right to (x, h-1)
        path.push((x, h - 1));
        // Up this column to y=1
        for y in (1..(h - 1)).rev() {
            path.push((x, y));
        }
        // If there's another column, step right at y=1, go down to bottom, etc.
        if x + 1 < w {
            path.push((x + 1, 1));
            for y in 2..h {
                path.push((x + 1, y));
            }
            x += 2;
        } else {
            x += 1;
        }
    }
    // Now we have visited all cells except the top row y=0 for x>=1.
    // Traverse the top row from right to 1.
    for xx in (1..w).rev() {
        path.push((xx, 0));
    }
    // Ends at (1,0), adjacent to (0,0).
    path
}

fn solve_small_exact(w: i32, h: i32, blocked: &[bool], require_cycle: bool) -> Option<Vec<(i32, i32)>> {
    let mut id_of: HashMap<usize, u8> = HashMap::new();
    let mut pos_of: Vec<(i32, i32)> = Vec::new();

    for y in 0..h {
        for x in 0..w {
            let idx = (y as usize) * (w as usize) + (x as usize);
            if !blocked[idx] {
                let id = pos_of.len() as u8;
                id_of.insert(idx, id);
                pos_of.push((x, y));
            }
        }
    }
    let n = pos_of.len();
    if n == 0 {
        return None;
    }

    // Start must be (0,0)
    let start_grid_idx = 0usize;
    let &start_id = id_of.get(&start_grid_idx)?;

    let mut adj: Vec<u64> = vec![0; n];
    for (i, &(x, y)) in pos_of.iter().enumerate() {
        let mut m = 0u64;
        for (nx, ny) in [(x - 1, y), (x + 1, y), (x, y - 1), (x, y + 1)] {
            if 0 <= nx && nx < w && 0 <= ny && ny < h {
                let nidx = (ny as usize) * (w as usize) + (nx as usize);
                if !blocked[nidx] {
                    let nid = *id_of.get(&nidx).unwrap() as usize;
                    m |= 1u64 << nid;
                }
            }
        }
        adj[i] = m;
    }

    // Quick necessary checks.
    if !connected_remaining(n as u8, &adj, 1u64 << (start_id as usize), (1u64 << n) - 1) {
        return None;
    }

    // Degree-1 forced endpoints for non-cycle.
    if !require_cycle {
        // At most two vertices can have degree 1 in a Hamiltonian path.
        let mut deg1 = 0;
        for i in 0..n {
            let d = (adj[i] & ((1u64 << n) - 1)).count_ones();
            if d == 0 {
                return None;
            }
            if d == 1 {
                deg1 += 1;
                if deg1 > 2 {
                    return None;
                }
            }
        }
    } else {
        // For a Hamiltonian cycle, min degree must be >=2.
        for i in 0..n {
            if (adj[i] & ((1u64 << n) - 1)).count_ones() < 2 {
                return None;
            }
        }
    }

    let all_mask = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };
    let mut path: Vec<u8> = Vec::with_capacity(n);
    path.push(start_id);
    let used0 = 1u64 << (start_id as usize);

    if dfs_exact(n as u8, &adj, &mut path, used0, all_mask, require_cycle) {
        let mut out = Vec::with_capacity(n);
        for &id in &path {
            out.push(pos_of[id as usize]);
        }
        return Some(out);
    }
    None
}

fn dfs_exact(n: u8, adj: &[u64], path: &mut Vec<u8>, used: u64, all: u64, require_cycle: bool) -> bool {
    if used == all {
        if !require_cycle {
            return true;
        }
        let start = path[0] as usize;
        let last = *path.last().unwrap() as usize;
        return (adj[last] & (1u64 << start)) != 0;
    }

    let cur = *path.last().unwrap() as usize;
    let mut candidates = adj[cur] & !used;
    if candidates == 0 {
        return false;
    }

    // Heuristic: choose next by minimum onward degree (Warnsdorff-like).
    let mut cand_list: Vec<(u32, u8)> = Vec::new();
    while candidates != 0 {
        let bit = candidates & (!candidates + 1);
        let v = bit.trailing_zeros() as usize;
        candidates ^= bit;
        let onward = (adj[v] & !used).count_ones();
        cand_list.push((onward, v as u8));
    }
    cand_list.sort_by_key(|&(d, _)| d);

    for &(_, v) in &cand_list {
        let v_usize = v as usize;
        let used2 = used | (1u64 << v_usize);
        path.push(v);

        // Prune: remaining graph must stay connected from current path end.
        if connected_remaining(n, adj, 1u64 << v_usize, all ^ used2) {
            // Prune: no isolated remaining vertex (degree 0 in remaining subgraph).
            if !has_isolated_remaining(n, adj, used2, all) {
                if dfs_exact(n, adj, path, used2, all, require_cycle) {
                    return true;
                }
            }
        }

        path.pop();
    }

    false
}

fn connected_remaining(n: u8, adj: &[u64], start_mask: u64, remaining_mask: u64) -> bool {
    let reachable_mask = start_mask | remaining_mask;
    // If nothing remains, connectedness is trivially true.
    if remaining_mask == 0 {
        return true;
    }
    let mut seen = 0u64;
    let start = start_mask.trailing_zeros() as usize;
    let mut q = VecDeque::new();
    q.push_back(start as u8);
    seen |= 1u64 << start;
    while let Some(u) = q.pop_front() {
        let u = u as usize;
        let mut nei = adj[u] & reachable_mask & !seen;
        while nei != 0 {
            let bit = nei & (!nei + 1);
            let v = bit.trailing_zeros() as usize;
            nei ^= bit;
            seen |= 1u64 << v;
            q.push_back(v as u8);
        }
    }
    (seen & remaining_mask) == remaining_mask
}

fn has_isolated_remaining(n: u8, adj: &[u64], used: u64, all: u64) -> bool {
    let rem = all ^ used;
    if rem == 0 {
        return false;
    }
    for i in 0..(n as usize) {
        if (rem & (1u64 << i)) != 0 {
            let deg = (adj[i] & rem).count_ones();
            if deg == 0 {
                return true;
            }
        }
    }
    false
}
