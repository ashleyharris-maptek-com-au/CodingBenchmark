
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};

const grid = data.grid || [];
const robotsIn = data.robots || [];
const jobsIn = data.jobs || [];

const h = grid.length;
const w = h ? grid[0].length : 0;
const n = w * h;

function idx(x, y) {
  return y * w + x;
}

// Passability map
const passable = new Uint8Array(n);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    passable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
  }
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

// BFS cache keyed by source index
const distCache = new Map();
const queue = new Int32Array(n);

function bfsFrom(sourceIndex) {
  let cached = distCache.get(sourceIndex);
  if (cached) return cached;

  const dist = new Int32Array(n);
  dist.fill(-1);

  if (sourceIndex < 0 || sourceIndex >= n || passable[sourceIndex] === 0) {
    distCache.set(sourceIndex, dist);
    return dist;
  }

  let head = 0;
  let tail = 0;
  dist[sourceIndex] = 0;
  queue[tail++] = sourceIndex;

  while (head < tail) {
    const v = queue[head++];
    const dv = dist[v] + 1;
    const x = v % w;
    const y = (v / w) | 0;

    // 4-neighbors
    // left
    if (x > 0) {
      const u = v - 1;
      if (passable[u] && dist[u] === -1) {
        dist[u] = dv;
        queue[tail++] = u;
      }
    }
    // right
    if (x + 1 < w) {
      const u = v + 1;
      if (passable[u] && dist[u] === -1) {
        dist[u] = dv;
        queue[tail++] = u;
      }
    }
    // up
    if (y > 0) {
      const u = v - w;
      if (passable[u] && dist[u] === -1) {
        dist[u] = dv;
        queue[tail++] = u;
      }
    }
    // down
    if (y + 1 < h) {
      const u = v + w;
      if (passable[u] && dist[u] === -1) {
        dist[u] = dv;
        queue[tail++] = u;
      }
    }
  }

  distCache.set(sourceIndex, dist);
  return dist;
}

function toIndex(pos) {
  const x = pos.x | 0;
  const y = pos.y | 0;
  if (!inBounds(x, y)) return -1;
  return idx(x, y);
}

// Build robot arrays by id (assumed 0..R-1 as per prompt)
let robotCount = robotsIn.length;
for (const r of robotsIn) {
  if ((r.id | 0) + 1 > robotCount) robotCount = (r.id | 0) + 1;
}

const robotPosIndex = new Int32Array(robotCount);
robotPosIndex.fill(-1);
for (const r of robotsIn) {
  const rid = r.id | 0;
  robotPosIndex[rid] = toIndex(r);
}

const assignments = Array.from({ length: robotCount }, () => []);

// Sort jobs: higher priority first, then id ascending (deterministic)
const jobs = jobsIn.slice();
jobs.sort((a, b) => {
  const pa = a.priority ?? 0;
  const pb = b.priority ?? 0;
  if (pa !== pb) return pb - pa;
  const ia = a.id;
  const ib = b.id;
  if (ia === ib) return 0;
  // Ensure deterministic ordering across numbers/strings
  return String(ia) < String(ib) ? -1 : 1;
});

for (const job of jobs) {
  const jx = job.x | 0;
  const jy = job.y | 0;
  const jIndex = inBounds(jx, jy) ? idx(jx, jy) : -1;

  let bestRobot = 0;
  let bestDist = Infinity;

  for (let rid = 0; rid < robotCount; rid++) {
    const src = robotPosIndex[rid];
    if (src === -1 || jIndex === -1) continue;
    const distArr = bfsFrom(src);
    const d = distArr[jIndex];
    if (d >= 0) {
      if (d < bestDist || (d === bestDist && rid < bestRobot)) {
        bestDist = d;
        bestRobot = rid;
      }
    }
  }

  // If unreachable from all, keep deterministic fallback to robot 0
  assignments[bestRobot].push(job.id);
  if (jIndex !== -1) robotPosIndex[bestRobot] = jIndex;
}

process.stdout.write(JSON.stringify({ assignments }));
