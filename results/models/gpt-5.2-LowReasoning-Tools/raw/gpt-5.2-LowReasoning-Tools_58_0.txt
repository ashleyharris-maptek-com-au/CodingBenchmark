const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8').trim();
  const data = input.length ? JSON.parse(input) : {};
  const gridRows = data.grid || [];
  const robots = data.robots || [];
  const jobs = data.jobs || [];

  const height = gridRows.length;
  const width = height ? gridRows[0].length : 0;
  const n = width * height;

  const passable = new Uint8Array(n);
  for (let y = 0; y < height; y++) {
    const row = gridRows[y];
    for (let x = 0; x < width; x++) {
      passable[y * width + x] = row.charCodeAt(x) === 46 ? 1 : 0; // '.'
    }
  }

  function inBounds(x, y) {
    return x >= 0 && x < width && y >= 0 && y < height;
  }

  // Robots may have non-contiguous ids; output array index is robot id.
  let maxRobotId = -1;
  for (const r of robots) {
    if (typeof r.id === 'number' && r.id > maxRobotId) maxRobotId = r.id;
  }
  const assignments = Array.from({ length: Math.max(0, maxRobotId + 1) }, () => []);

  // Track current positions per robot id.
  const robotIds = [];
  const robotPos = new Map(); // id -> idx
  for (const r of robots) {
    const id = r.id;
    robotIds.push(id);
    const x = r.x | 0;
    const y = r.y | 0;
    const idx = inBounds(x, y) ? (y * width + x) : -1;
    robotPos.set(id, idx);
    if (id >= 0 && id < assignments.length) assignments[id] = [];
  }
  robotIds.sort((a, b) => a - b);

  // Cache BFS distance arrays by start index.
  const distCache = new Map(); // idx -> Int32Array

  function bfsFrom(startIdx) {
    if (startIdx < 0 || startIdx >= n) return null;
    if (!passable[startIdx]) return null;
    const cached = distCache.get(startIdx);
    if (cached) return cached;

    const dist = new Int32Array(n);
    dist.fill(-1);
    const q = new Int32Array(n);
    let qh = 0, qt = 0;

    dist[startIdx] = 0;
    q[qt++] = startIdx;

    while (qh < qt) {
      const cur = q[qh++];
      const d = dist[cur] + 1;
      const x = cur % width;
      const y = (cur / width) | 0;

      // 4-neighborhood
      let ni;
      if (x > 0) {
        ni = cur - 1;
        if (passable[ni] && dist[ni] === -1) { dist[ni] = d; q[qt++] = ni; }
      }
      if (x + 1 < width) {
        ni = cur + 1;
        if (passable[ni] && dist[ni] === -1) { dist[ni] = d; q[qt++] = ni; }
      }
      if (y > 0) {
        ni = cur - width;
        if (passable[ni] && dist[ni] === -1) { dist[ni] = d; q[qt++] = ni; }
      }
      if (y + 1 < height) {
        ni = cur + width;
        if (passable[ni] && dist[ni] === -1) { dist[ni] = d; q[qt++] = ni; }
      }
    }

    distCache.set(startIdx, dist);
    return dist;
  }

  function jobIdx(job) {
    const x = job.x | 0;
    const y = job.y | 0;
    if (!inBounds(x, y)) return -1;
    const idx = y * width + x;
    return passable[idx] ? idx : -1;
  }

  // Deterministic greedy: highest priority first, then by id.
  const jobsSorted = jobs.slice();
  jobsSorted.sort((a, b) => {
    const pa = a.priority ?? 0;
    const pb = b.priority ?? 0;
    if (pa !== pb) return pb - pa;
    const sa = String(a.id);
    const sb = String(b.id);
    return sa < sb ? -1 : sa > sb ? 1 : 0;
  });

  for (const job of jobsSorted) {
    const jIdx = jobIdx(job);

    let bestRobotId = robotIds.length ? robotIds[0] : 0;
    let bestDist = 0x7fffffff;

    for (const rid of robotIds) {
      const rIdx = robotPos.get(rid);
      let d = 0x7fffffff;
      if (jIdx !== -1 && rIdx !== -1) {
        const distArr = bfsFrom(rIdx);
        if (distArr) {
          const v = distArr[jIdx];
          if (v !== -1) d = v;
        }
      }
      if (d < bestDist || (d === bestDist && rid < bestRobotId)) {
        bestDist = d;
        bestRobotId = rid;
      }
    }

    if (bestRobotId >= 0 && bestRobotId < assignments.length) {
      assignments[bestRobotId].push(job.id);
    }

    // Update robot position to the job location if it is on the grid; otherwise keep position.
    if (jIdx !== -1) robotPos.set(bestRobotId, jIdx);
  }

  process.stdout.write(JSON.stringify({ assignments }));
}

main();
