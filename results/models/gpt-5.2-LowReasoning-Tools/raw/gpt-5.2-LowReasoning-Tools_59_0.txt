const fs = require('fs');

function solve(data) {
  const prob = data.probability;
  const H = prob.length;
  const W = H ? prob[0].length : 0;
  const drones = Array.isArray(data.drones) ? data.drones : [];
  const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);

  const isBlocked = new Uint8Array(W * H);
  const value = new Float64Array(W * H);
  for (let y = 0; y < H; y++) {
    const row = prob[y];
    for (let x = 0; x < W; x++) {
      const v = row[x];
      const idx = y * W + x;
      if (v === -1) {
        isBlocked[idx] = 1;
        value[idx] = 0;
      } else {
        isBlocked[idx] = 0;
        value[idx] = v > 0 ? v : 0;
      }
    }
  }

  const covered = new Uint8Array(W * H);
  function markCovered(x, y) {
    if (x < 0 || y < 0 || x >= W || y >= H) return;
    const idx = y * W + x;
    if (isBlocked[idx]) return;
    if (value[idx] > 0) covered[idx] = 1;
  }

  const state = drones.map((d) => ({
    x: d.x | 0,
    y: d.y | 0,
    path: [[d.x | 0, d.y | 0]],
  }));

  for (const s of state) markCovered(s.x, s.y);

  // Limited BFS scoring:
  // From a candidate next position, explore up to radius R and score the best unvisited cell.
  // Score blends value and distance, with a gentle time/step pressure to prefer closer gains.
  const dirs = [
    [0, 0],
    [1, 0],
    [-1, 0],
    [0, 1],
    [0, -1],
  ];

  // Choose R based on grid size; keep bounded for scalability.
  const gridN = W * H;
  let R = 10;
  if (gridN >= 40000) R = 9;
  if (gridN >= 160000) R = 8;
  if (gridN >= 360000) R = 7;

  // Reusable BFS buffers sized for the whole grid to avoid reallocation.
  const seenStamp = new Int32Array(W * H);
  let stamp = 1;
  const qx = new Int32Array(W * H);
  const qy = new Int32Array(W * H);
  const qd = new Int16Array(W * H);

  function bfsBestScore(sx, sy, stepIndex) {
    if (sx < 0 || sy < 0 || sx >= W || sy >= H) return -1e18;
    const sidx = sy * W + sx;
    if (isBlocked[sidx]) return -1e18;

    // Immediate reward for landing on an unvisited positive cell.
    let best = 0;
    if (!covered[sidx] && value[sidx] > 0) best = value[sidx] * 1.25;

    const localStamp = stamp++;
    // If stamp overflows (very unlikely), reset.
    if (stamp === 0x7fffffff) {
      stamp = 1;
      for (let i = 0; i < seenStamp.length; i++) seenStamp[i] = 0;
    }

    let head = 0,
      tail = 0;
    seenStamp[sidx] = localStamp;
    qx[tail] = sx;
    qy[tail] = sy;
    qd[tail] = 0;
    tail++;

    // Soft factor to encourage earlier gains as remaining steps shrink.
    // This avoids drones wandering too far when near the horizon.
    const horizon = Math.max(1, maxSteps - stepIndex);
    const horizonFactor = 1 + 3 / horizon;

    while (head < tail) {
      const x = qx[head];
      const y = qy[head];
      const d = qd[head];
      head++;
      if (d >= R) continue;

      // Expand 4-neighbors (not stay).
      // Order is arbitrary; it can affect ties slightly.
      // Keep it tight for speed.
      const nx1 = x + 1,
        ny1 = y;
      const nx2 = x - 1,
        ny2 = y;
      const nx3 = x,
        ny3 = y + 1;
      const nx4 = x,
        ny4 = y - 1;

      // Inline addNeighbor for performance.
      // 1
      if (nx1 >= 0 && nx1 < W) {
        const idx = ny1 * W + nx1;
        if (!isBlocked[idx] && seenStamp[idx] !== localStamp) {
          seenStamp[idx] = localStamp;
          qx[tail] = nx1;
          qy[tail] = ny1;
          qd[tail] = d + 1;
          tail++;
          if (!covered[idx] && value[idx] > 0) {
            const sc = value[idx] / (1 + (d + 1) * horizonFactor);
            if (sc > best) best = sc;
          }
        }
      }
      // 2
      if (nx2 >= 0 && nx2 < W) {
        const idx = ny2 * W + nx2;
        if (!isBlocked[idx] && seenStamp[idx] !== localStamp) {
          seenStamp[idx] = localStamp;
          qx[tail] = nx2;
          qy[tail] = ny2;
          qd[tail] = d + 1;
          tail++;
          if (!covered[idx] && value[idx] > 0) {
            const sc = value[idx] / (1 + (d + 1) * horizonFactor);
            if (sc > best) best = sc;
          }
        }
      }
      // 3
      if (ny3 >= 0 && ny3 < H) {
        const idx = ny3 * W + nx3;
        if (!isBlocked[idx] && seenStamp[idx] !== localStamp) {
          seenStamp[idx] = localStamp;
          qx[tail] = nx3;
          qy[tail] = ny3;
          qd[tail] = d + 1;
          tail++;
          if (!covered[idx] && value[idx] > 0) {
            const sc = value[idx] / (1 + (d + 1) * horizonFactor);
            if (sc > best) best = sc;
          }
        }
      }
      // 4
      if (ny4 >= 0 && ny4 < H) {
        const idx = ny4 * W + nx4;
        if (!isBlocked[idx] && seenStamp[idx] !== localStamp) {
          seenStamp[idx] = localStamp;
          qx[tail] = nx4;
          qy[tail] = ny4;
          qd[tail] = d + 1;
          tail++;
          if (!covered[idx] && value[idx] > 0) {
            const sc = value[idx] / (1 + (d + 1) * horizonFactor);
            if (sc > best) best = sc;
          }
        }
      }
    }

    return best;
  }

  function validCell(x, y) {
    if (x < 0 || y < 0 || x >= W || y >= H) return false;
    const idx = y * W + x;
    return isBlocked[idx] === 0;
  }

  for (let step = 0; step < maxSteps; step++) {
    // Decide moves sequentially while updating covered so drones tend to diversify.
    for (let i = 0; i < state.length; i++) {
      const d = state[i];
      let bestMove = 0;
      let bestScore = -1e18;

      for (let m = 0; m < dirs.length; m++) {
        const nx = d.x + dirs[m][0];
        const ny = d.y + dirs[m][1];
        if (!validCell(nx, ny)) continue;

        // Slight penalty for staying still unless it collects value.
        const idx = ny * W + nx;
        const stayPenalty = m === 0 ? 0.015 : 0;
        const immediate = !covered[idx] && value[idx] > 0 ? value[idx] : 0;
        const future = bfsBestScore(nx, ny, step);
        const sc = immediate * 1.5 + future - stayPenalty;

        // Deterministic tie-break: prefer immediate gain, then move (not stay), then lexicographic.
        if (sc > bestScore) {
          bestScore = sc;
          bestMove = m;
        } else if (sc === bestScore) {
          const curBest = dirs[bestMove];
          const curNx = d.x + curBest[0];
          const curNy = d.y + curBest[1];
          const curIdx = curNy * W + curNx;
          const curImm = !covered[curIdx] && value[curIdx] > 0 ? value[curIdx] : 0;
          if (immediate > curImm) {
            bestMove = m;
          } else if (immediate === curImm) {
            if (bestMove === 0 && m !== 0) {
              bestMove = m;
            } else if (bestMove !== 0 && m !== 0) {
              if (ny < curNy || (ny === curNy && nx < curNx)) bestMove = m;
            } else if (bestMove === 0 && m === 0) {
              // equal
            }
          }
        }
      }

      const nx = d.x + dirs[bestMove][0];
      const ny = d.y + dirs[bestMove][1];
      d.x = nx;
      d.y = ny;
      d.path.push([nx, ny]);
      markCovered(nx, ny);
    }
  }

  return { paths: state.map((d) => d.path) };
}

const input = fs.readFileSync(0, 'utf8').trim();
const data = input.length ? JSON.parse(input) : {};
const out = solve(data);
process.stdout.write(JSON.stringify(out));
