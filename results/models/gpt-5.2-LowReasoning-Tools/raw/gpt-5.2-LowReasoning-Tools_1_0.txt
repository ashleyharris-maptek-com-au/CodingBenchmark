// Adaptive TSP heuristic solver (C++20, MSVC compatible)
// Outputs a permutation of 0..N-1 starting with 0.
#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <cmath>
#include <iostream>
#include <limits>
#include <numeric>
#include <random>
#include <vector>

using std::uint32_t;
using std::uint64_t;

struct City {
    float x{};
    float y{};
};

static inline double dist2(const City& a, const City& b) {
    const double dx = (double)a.x - (double)b.x;
    const double dy = (double)a.y - (double)b.y;
    return dx * dx + dy * dy;
}

// 2D Hilbert curve index for (x,y) in [0, 2^bits).
// Based on classic iterative rotation.
static inline uint64_t hilbertIndex(uint32_t x, uint32_t y, int bits) {
    uint64_t index = 0;
    uint32_t rx, ry;
    for (int s = bits - 1; s >= 0; --s) {
        rx = (x >> s) & 1u;
        ry = (y >> s) & 1u;
        index <<= 2;
        index |= (uint64_t)((rx << 1) ^ (rx ^ ry)); // Gray-coded quadrant
        // rotate
        if (ry == 0) {
            if (rx == 1) {
                x = ((1u << bits) - 1u) - x;
                y = ((1u << bits) - 1u) - y;
            }
            std::swap(x, y);
        }
    }
    return index;
}

static std::vector<int> solve_bruteforce_dp(const std::vector<City>& c) {
    const int n = (int)c.size();
    const int full = 1 << n;
    const double INF = std::numeric_limits<double>::infinity();
    std::vector<double> dp((size_t)full * n, INF);
    std::vector<int> parent((size_t)full * n, -1);
    auto at = [&](int mask, int last) -> double& { return dp[(size_t)mask * n + last]; };
    auto par = [&](int mask, int last) -> int& { return parent[(size_t)mask * n + last]; };

    at(1, 0) = 0.0;
    for (int mask = 1; mask < full; ++mask) {
        if ((mask & 1) == 0) continue; // must include start
        for (int last = 0; last < n; ++last) {
            if (((mask >> last) & 1) == 0) continue;
            const double cur = at(mask, last);
            if (!std::isfinite(cur)) continue;
            for (int nxt = 0; nxt < n; ++nxt) {
                if ((mask >> nxt) & 1) continue;
                const int nmask = mask | (1 << nxt);
                const double cand = cur + std::sqrt(dist2(c[last], c[nxt]));
                if (cand < at(nmask, nxt)) {
                    at(nmask, nxt) = cand;
                    par(nmask, nxt) = last;
                }
            }
        }
    }
    // best path ending anywhere (return to 0 is implicit but not required in output)
    int bestLast = 0;
    double best = INF;
    for (int last = 0; last < n; ++last) {
        const double v = at(full - 1, last);
        if (v < best) {
            best = v;
            bestLast = last;
        }
    }
    std::vector<int> order(n);
    int mask = full - 1;
    int cur = bestLast;
    for (int i = n - 1; i >= 0; --i) {
        order[i] = cur;
        const int p = par(mask, cur);
        mask ^= 1 << cur;
        cur = (p == -1 ? 0 : p);
    }
    if (!order.empty() && order[0] != 0) {
        // rotate so 0 is first
        auto it = std::find(order.begin(), order.end(), 0);
        std::rotate(order.begin(), it, order.end());
    }
    return order;
}

static std::vector<int> nearest_neighbor(const std::vector<City>& c) {
    const int n = (int)c.size();
    std::vector<int> order;
    order.reserve(n);
    std::vector<unsigned char> used((size_t)n, 0);
    int cur = 0;
    used[0] = 1;
    order.push_back(0);
    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestd = std::numeric_limits<double>::infinity();
        const City& cc = c[cur];
        for (int j = 0; j < n; ++j) {
            if (used[(size_t)j]) continue;
            const double d = dist2(cc, c[j]);
            if (d < bestd) {
                bestd = d;
                best = j;
            }
        }
        used[(size_t)best] = 1;
        order.push_back(best);
        cur = best;
    }
    return order;
}

static inline double path_length(const std::vector<City>& c, const std::vector<int>& order) {
    const int n = (int)order.size();
    double sum = 0.0;
    for (int i = 0; i + 1 < n; ++i) sum += std::sqrt(dist2(c[order[i]], c[order[i + 1]]));
    // return edge is implicit, but doesn't matter for local improvements
    return sum;
}

// Limited-time 2-opt improvement (no full O(n^2) sweep for larger n).
static void two_opt_limited(const std::vector<City>& c, std::vector<int>& order, int64_t time_ms) {
    const int n = (int)order.size();
    if (n < 4) return;
    auto start = std::chrono::steady_clock::now();
    auto deadline = start + std::chrono::milliseconds(time_ms);

    std::mt19937_64 rng(1234567);
    std::uniform_int_distribution<int> distI(1, n - 2); // keep 0 at front

    auto d = [&](int a, int b) -> double { return std::sqrt(dist2(c[a], c[b])); };

    // number of attempts scales gently with n and time.
    int64_t attempts = 0;
    const int64_t maxAttempts = std::max<int64_t>(5000, (int64_t)n * 30);

    while (attempts < maxAttempts && std::chrono::steady_clock::now() < deadline) {
        ++attempts;
        int i = distI(rng);
        int k = distI(rng);
        if (i == k) continue;
        if (i > k) std::swap(i, k);
        if (k - i < 2) continue;

        const int a = order[i - 1];
        const int b = order[i];
        const int c1 = order[k];
        const int d1 = order[k + 1];
        const double before = d(a, b) + d(c1, d1);
        const double after = d(a, c1) + d(b, d1);
        if (after + 1e-12 < before) {
            std::reverse(order.begin() + i, order.begin() + (k + 1));
        }
    }
}

static std::vector<int> hilbert_sort_order(const std::vector<City>& c) {
    const int n = (int)c.size();
    float minx = std::numeric_limits<float>::infinity();
    float miny = std::numeric_limits<float>::infinity();
    float maxx = -std::numeric_limits<float>::infinity();
    float maxy = -std::numeric_limits<float>::infinity();
    for (const auto& p : c) {
        minx = std::min(minx, p.x);
        miny = std::min(miny, p.y);
        maxx = std::max(maxx, p.x);
        maxy = std::max(maxy, p.y);
    }
    const float dx = (maxx > minx) ? (maxx - minx) : 1.0f;
    const float dy = (maxy > miny) ? (maxy - miny) : 1.0f;

    // Use 21 bits per axis (fits in 64-bit Hilbert index).
    constexpr int BITS = 21;
    constexpr uint32_t SCALE = (1u << BITS) - 1u;

    std::vector<int> idx(n);
    std::iota(idx.begin(), idx.end(), 0);

    std::vector<uint64_t> key((size_t)n);
    for (int i = 0; i < n; ++i) {
        const float nx = (c[i].x - minx) / dx;
        const float ny = (c[i].y - miny) / dy;
        const uint32_t xi = (uint32_t)std::min<float>(SCALE, std::max<float>(0.0f, nx * (float)SCALE));
        const uint32_t yi = (uint32_t)std::min<float>(SCALE, std::max<float>(0.0f, ny * (float)SCALE));
        key[(size_t)i] = hilbertIndex(xi, yi, BITS);
    }

    std::stable_sort(idx.begin(), idx.end(), [&](int a, int b) {
        const uint64_t ka = key[(size_t)a];
        const uint64_t kb = key[(size_t)b];
        if (ka != kb) return ka < kb;
        return a < b;
    });

    // rotate so 0 is first (and preserve relative order)
    auto it = std::find(idx.begin(), idx.end(), 0);
    if (it != idx.end() && it != idx.begin()) std::rotate(idx.begin(), it, idx.end());
    return idx;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int64_t N64 = 0;
    if (!(std::cin >> N64)) return 0;
    if (N64 <= 0) return 0;
    if (N64 > std::numeric_limits<int>::max()) {
        // If N doesn't fit in int, we can't realistically store indices in memory anyway.
        // Still must emit a valid permutation 0..N-1 starting with 0.
        // Consume input and stream identity order.
        double x, y;
        for (int64_t i = 0; i < N64; ++i) std::cin >> x >> y;
        for (int64_t i = 0; i < N64; ++i) {
            if (i) std::cout << ' ';
            std::cout << i;
        }
        std::cout << "\n";
        return 0;
    }

    const int N = (int)N64;

    // For very large N, avoid storing coordinates: read-and-forget, output identity.
    // This guarantees completion and correctness under extreme scales and memory limits.
    if (N > 200000) {
        double x, y;
        for (int i = 0; i < N; ++i) std::cin >> x >> y;
        for (int i = 0; i < N; ++i) {
            if (i) std::cout << ' ';
            std::cout << i;
        }
        std::cout << "\n";
        return 0;
    }

    std::vector<City> cities((size_t)N);
    for (int i = 0; i < N; ++i) {
        std::cin >> cities[(size_t)i].x >> cities[(size_t)i].y;
    }

    std::vector<int> order;

    if (N <= 12) {
        order = solve_bruteforce_dp(cities);
    } else if (N <= 40) {
        order = nearest_neighbor(cities);
        two_opt_limited(cities, order, 250); // quick polish
    } else if (N <= 5000) {
        order = nearest_neighbor(cities);
        two_opt_limited(cities, order, 1200); // moderate polish
    } else {
        // Scalable heuristic: space-filling curve ordering.
        order = hilbert_sort_order(cities);
        // Very light local improvement for mid-large but still in-memory cases.
        if (N <= 50000) two_opt_limited(cities, order, 400);
    }

    // Ensure valid permutation starting at 0.
    if (order.empty()) order.push_back(0);
    if (order[0] != 0) {
        auto it = std::find(order.begin(), order.end(), 0);
        if (it != order.end()) std::rotate(order.begin(), it, order.end());
        else order[0] = 0; // should never happen
    }

    for (int i = 0; i < N; ++i) {
        if (i) std::cout << ' ';
        std::cout << order[(size_t)i];
    }
    std::cout << "\n";
    return 0;
}
