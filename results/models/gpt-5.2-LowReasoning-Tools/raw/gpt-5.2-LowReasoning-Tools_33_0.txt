using System;
using System.Diagnostics;
using System.IO;

public class Program
{
    static int N;
    static int[] Flow;   // flat N*N
    static int[] Dist;   // flat N*N

    static int Idx(int i, int j) => i * N + j;

    static long ComputeCost(int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int li = perm[i];
            int rowBase = i * N;
            int distRowBase = li * N;
            for (int j = 0; j < N; j++)
            {
                cost += (long)Flow[rowBase + j] * Dist[distRowBase + perm[j]];
            }
        }
        return cost;
    }

    // Delta when swapping facilities a and b in current permutation.
    static long DeltaSwap(int[] perm, int a, int b)
    {
        if (a == b) return 0;
        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;

        for (int k = 0; k < N; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];

            // (a,k) and (b,k)
            delta += (long)(Flow[Idx(a, k)] - Flow[Idx(b, k)]) * (Dist[Idx(pb, pk)] - Dist[Idx(pa, pk)]);
            // (k,a) and (k,b)
            delta += (long)(Flow[Idx(k, a)] - Flow[Idx(k, b)]) * (Dist[Idx(pk, pb)] - Dist[Idx(pk, pa)]);
        }

        // Pair terms involving (a,b) and (b,a) and self terms
        delta += (long)Flow[Idx(a, a)] * (Dist[Idx(pb, pb)] - Dist[Idx(pa, pa)]);
        delta += (long)Flow[Idx(b, b)] * (Dist[Idx(pa, pa)] - Dist[Idx(pb, pb)]);
        delta += (long)Flow[Idx(a, b)] * (Dist[Idx(pb, pa)] - Dist[Idx(pa, pb)]);
        delta += (long)Flow[Idx(b, a)] * (Dist[Idx(pa, pb)] - Dist[Idx(pb, pa)]);

        return delta;
    }

    static void Swap(int[] perm, int a, int b)
    {
        int t = perm[a];
        perm[a] = perm[b];
        perm[b] = t;
    }

    static int[] GreedyInit(int seed)
    {
        // Simple greedy: sort facilities by outgoing+incoming flow sum descending,
        // locations by distance sum ascending; match ranks.
        long[] fScore = new long[N];
        long[] dScore = new long[N];

        for (int i = 0; i < N; i++)
        {
            long fs = 0, ds = 0;
            int ib = i * N;
            for (int j = 0; j < N; j++)
            {
                fs += Flow[ib + j] + Flow[j * N + i];
                ds += Dist[ib + j] + Dist[j * N + i];
            }
            fScore[i] = fs;
            dScore[i] = ds;
        }

        int[] facilities = new int[N];
        int[] locations = new int[N];
        for (int i = 0; i < N; i++) { facilities[i] = i; locations[i] = i; }

        Array.Sort(facilities, (a, b) => fScore[b].CompareTo(fScore[a]));
        Array.Sort(locations, (a, b) => dScore[a].CompareTo(dScore[b]));

        int[] perm = new int[N];
        for (int rank = 0; rank < N; rank++)
            perm[facilities[rank]] = locations[rank];

        // Small random shuffles to diversify.
        var rnd = new Random(seed);
        int sh = Math.Min(N * 2, 20000);
        for (int t = 0; t < sh; t++)
        {
            int a = rnd.Next(N);
            int b = rnd.Next(N);
            Swap(perm, a, b);
        }
        return perm;
    }

    static void Main()
    {
        // Fast-ish input
        string data = Console.In.ReadToEnd();
        int pos = 0;
        int len = data.Length;
        int NextInt()
        {
            while (pos < len && data[pos] <= ' ') pos++;
            int sign = 1;
            if (pos < len && data[pos] == '-') { sign = -1; pos++; }
            int val = 0;
            while (pos < len)
            {
                char c = data[pos];
                if (c < '0' || c > '9') break;
                val = val * 10 + (c - '0');
                pos++;
            }
            return val * sign;
        }

        N = NextInt();
        if (N <= 0)
        {
            Console.WriteLine("0\n");
            return;
        }

        Flow = new int[N * N];
        Dist = new int[N * N];

        for (int i = 0; i < N * N; i++) Flow[i] = NextInt();
        for (int i = 0; i < N * N; i++) Dist[i] = NextInt();

        var sw = Stopwatch.StartNew();
        double timeLimitSeconds =
            N <= 50 ? 1.0 :
            N <= 200 ? 5.0 :
            N <= 500 ? 30.0 :
            N <= 1000 ? 120.0 :
            295.0; // ~5 minutes, leave some margin

        int[] bestPerm = GreedyInit(12345);
        long bestCost = ComputeCost(bestPerm);

        // Multi-start if small; otherwise single start + local search.
        int starts = N <= 200 ? 8 : (N <= 500 ? 4 : 2);
        for (int s = 1; s < starts && sw.Elapsed.TotalSeconds < timeLimitSeconds * 0.20; s++)
        {
            int[] perm = GreedyInit(12345 + 10007 * s);
            long cost = ComputeCost(perm);
            if (cost < bestCost) { bestCost = cost; bestPerm = perm; }
        }

        // Local search (stochastic hillclimb with light annealing).
        int[] curPerm = (int[])bestPerm.Clone();
        long curCost = bestCost;
        var rnd2 = new Random(777);

        double t0 = 1.0;
        double tEnd = 0.001;

        long lastBest = bestCost;
        int stagnationBatches = 0;
        while (sw.Elapsed.TotalSeconds < timeLimitSeconds)
        {
            double frac = sw.Elapsed.TotalSeconds / timeLimitSeconds;
            double temp = t0 * Math.Pow(tEnd / t0, frac);

            // Batch of random proposals.
            int batch = N <= 300 ? 20000 : (N <= 1000 ? 30000 : 40000);
            for (int iter = 0; iter < batch; iter++)
            {
                int a = rnd2.Next(N);
                int b = rnd2.Next(N);
                if (a == b) continue;
                long delta = DeltaSwap(curPerm, a, b);
                if (delta < 0)
                {
                    Swap(curPerm, a, b);
                    curCost += delta;
                    if (curCost < bestCost)
                    {
                        bestCost = curCost;
                        bestPerm = (int[])curPerm.Clone();
                    }
                }
                else
                {
                    // Accept uphill move with small probability (annealing).
                    double p = Math.Exp(-delta / (temp * 1e6 + 1.0));
                    if (rnd2.NextDouble() < p)
                    {
                        Swap(curPerm, a, b);
                        curCost += delta;
                    }
                }
                if (sw.Elapsed.TotalSeconds >= timeLimitSeconds) break;
            }

            // Occasional restart from best to avoid drift.
            if (rnd2.Next(10) == 0)
            {
                curPerm = (int[])bestPerm.Clone();
                curCost = bestCost;
            }

            if (bestCost < lastBest)
            {
                lastBest = bestCost;
                stagnationBatches = 0;
            }
            else
            {
                stagnationBatches++;
                if (N <= 200 && stagnationBatches >= 5) break;
                if (N <= 500 && stagnationBatches >= 20) break;
            }
        }

        // Output
        using var w = new StreamWriter(Console.OpenStandardOutput());
        w.WriteLine(bestCost);
        for (int i = 0; i < N; i++)
        {
            if (i != 0) w.Write(' ');
            w.Write(bestPerm[i]);
        }
        w.WriteLine();
    }
}
