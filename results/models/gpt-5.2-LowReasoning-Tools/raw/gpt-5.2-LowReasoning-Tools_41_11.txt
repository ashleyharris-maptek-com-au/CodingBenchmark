OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

%glsl450 = OpExtInstImport "GLSL.std.450"

OpName %main "main"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_uv "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16

%void = OpTypeVoid
%fn = OpTypeFunction %void

%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%mat4 = OpTypeMatrix %v4float 4

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%in_worldPos = OpVariable %_ptr_Input_v3float Input
%in_normal = OpVariable %_ptr_Input_v3float Input
%in_uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_1 = OpConstant %float 1
%float_6 = OpConstant %float 6

%v2_6 = OpConstantComposite %v2float %float_6 %float_6
%v2_0_5 = OpConstantComposite %v2float %float_0_5 %float_0_5

%float_0_8 = OpConstant %float 0.8
%float_0_1 = OpConstant %float 0.1

%dot_rgb = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%bg_rgb = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3

%main = OpFunction %void None %fn
%entry = OpLabel

; Load varying inputs
%wp = OpLoad %v3float %in_worldPos
%n0 = OpLoad %v3float %in_normal
%uv0 = OpLoad %v2float %in_uv

; Normalize normal
%n = OpExtInst %v3float %glsl450 Normalize %n0

; Light direction: normalize(lightPos.xyz - worldPos)
%lp_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lp4 = OpLoad %v4float %lp_ptr
%lp3 = OpVectorShuffle %v3float %lp4 %lp4 0 1 2
%lvec = OpFSub %v3float %lp3 %wp
%l = OpExtInst %v3float %glsl450 Normalize %lvec

; Diffuse term
%ndotl = OpDot %float %n %l
%diffuse = OpExtInst %float %glsl450 FMax %ndotl %float_0
%scaled = OpFMul %float %diffuse %float_0_7
%shade = OpFAdd %float %float_0_3 %scaled

; 6x6 cell UV and dot test
%uv6 = OpFMul %v2float %uv0 %v2_6
%cell = OpExtInst %v2float %glsl450 Fract %uv6
%d = OpFSub %v2float %cell %v2_0_5
%len = OpExtInst %float %glsl450 Length %d
%isDot = OpFOrdLessThanEqual %bool %len %float_0_3

; Base color select
%base = OpSelect %v3float %isDot %dot_rgb %bg_rgb

; Apply shading
%rgb = OpVectorTimesScalar %v3float %base %shade
%out = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %out

OpReturn
OpFunctionEnd
