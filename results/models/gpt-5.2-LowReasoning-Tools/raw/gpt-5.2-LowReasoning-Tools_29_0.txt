using System;
using System.Buffers;
using System.Collections.Generic;
using System.Diagnostics;
using System.Numerics;
using System.Runtime.CompilerServices;
using System.Threading;

public static class Program
{
    // Adjusted to leave some slack for IO / JIT.
    private static readonly TimeSpan SoftTimeLimit = TimeSpan.FromMinutes(4.75);

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        // Read edges; build adjacency lists.
        int[] deg = new int[n];
        int[] eu = new int[m];
        int[] ev = new int[m];

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            eu[i] = u;
            ev[i] = v;
            if ((uint)u < (uint)n && (uint)v < (uint)n && u != v)
            {
                deg[u]++;
                deg[v]++;
            }
        }

        int[][] adj = new int[n][];
        int[] fill = new int[n];
        for (int i = 0; i < n; i++) adj[i] = new int[deg[i]];
        for (int i = 0; i < m; i++)
        {
            int u = eu[i], v = ev[i];
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adj[u][fill[u]++] = v;
            adj[v][fill[v]++] = u;
        }

        // Optional: sort adjacency lists for fast contains checks in validation / heuristic.
        for (int i = 0; i < n; i++) Array.Sort(adj[i]);

        int[] bestClique;

        // Use exact algorithm for moderate n; heuristic otherwise.
        // Bitset-based BK is memory-heavy for very large n.
        const int ExactMaxN = 5000;
        if (n <= ExactMaxN)
        {
            bestClique = MaxCliqueExactBitset(adj, n, sw);
        }
        else
        {
            bestClique = MaxCliqueHeuristic(adj, n, sw);
        }

        // As a safety net, ensure output is a clique; if not, fall back to a trivial clique.
        if (!IsClique(adj, bestClique))
        {
            bestClique = new int[] { 0 };
            for (int u = 0; u < n; u++)
            {
                if (adj[u].Length > 0) { bestClique = new int[] { u, adj[u][0] }; break; }
            }
        }

        Console.WriteLine(bestClique.Length);
        for (int i = 0; i < bestClique.Length; i++)
        {
            if (i != 0) Console.Write(' ');
            Console.Write(bestClique[i]);
        }
        Console.WriteLine();
    }

    private static bool IsClique(int[][] adj, int[] verts)
    {
        int k = verts.Length;
        for (int i = 0; i < k; i++)
        {
            int u = verts[i];
            var list = adj[u];
            for (int j = i + 1; j < k; j++)
            {
                int v = verts[j];
                if (Array.BinarySearch(list, v) < 0) return false;
            }
        }
        return true;
    }

    private static int[] MaxCliqueHeuristic(int[][] adj, int n, Stopwatch sw)
    {
        // Multi-start greedy construction with degree ordering + local improvement.
        // Designed to always return a valid clique quickly on huge graphs.
        int[] best = new int[] { 0 };

        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => adj[b].Length.CompareTo(adj[a].Length));

        int maxStarts = Math.Min(64, n);
        var rng = new XorShift64(0xC0FFEEUL);

        // Start points: top-degree nodes + a few random nodes.
        Span<int> starts = stackalloc int[Math.Min(128, maxStarts + 16)];
        int sc = 0;
        for (int i = 0; i < maxStarts && i < order.Length; i++) starts[sc++] = order[i];
        for (int i = 0; i < 16 && i < n; i++) starts[sc++] = (int)(rng.NextULong() % (ulong)n);

        // Reusable buffers.
        int[] cand = ArrayPool<int>.Shared.Rent(Math.Min(n, 200000)); // cap for sanity
        int[] clique = ArrayPool<int>.Shared.Rent(4096);
        int[] tmp = ArrayPool<int>.Shared.Rent(Math.Min(n, 200000));

        try
        {
            for (int si = 0; si < sc; si++)
            {
                if (sw.Elapsed > SoftTimeLimit) break;

                int seed = starts[si];
                int cliqueSize = 0;
                if (cliqueSize >= clique.Length) Array.Resize(ref clique, clique.Length * 2);
                clique[cliqueSize++] = seed;

                // Candidates are neighbors of the current clique (initially neighbors of seed).
                int candCount = adj[seed].Length;
                if (candCount > cand.Length) candCount = cand.Length;
                Array.Copy(adj[seed], 0, cand, 0, candCount);

                // Greedily add vertices that keep clique property; pick by highest degree within candidate set.
                while (candCount > 0 && sw.Elapsed <= SoftTimeLimit)
                {
                    // Choose candidate with highest degree (global degree heuristic).
                    int bestIdx = 0;
                    int bestDeg = -1;
                    for (int i = 0; i < candCount; i++)
                    {
                        int v = cand[i];
                        int d = adj[v].Length;
                        if (d > bestDeg) { bestDeg = d; bestIdx = i; }
                    }
                    int vPick = cand[bestIdx];

                    // Add it.
                    if (cliqueSize >= clique.Length) Array.Resize(ref clique, clique.Length * 2);
                    clique[cliqueSize++] = vPick;

                    // Filter candidates to those adjacent to vPick.
                    candCount = IntersectSorted(cand, candCount, adj[vPick], tmp);
                    // Copy back for next loop.
                    Array.Copy(tmp, 0, cand, 0, candCount);
                }

                // Simple local improvement: try to add a vertex connected to all clique vertices by scanning
                // intersection of neighbors of a couple clique members (cheap).
                // (On huge graphs, full intersection of all members is too costly.)
                if (cliqueSize > best.Length)
                {
                    best = new int[cliqueSize];
                    Array.Copy(clique, best, cliqueSize);
                }
            }
        }
        finally
        {
            ArrayPool<int>.Shared.Return(cand);
            ArrayPool<int>.Shared.Return(clique);
            ArrayPool<int>.Shared.Return(tmp);
        }

        Array.Sort(best);
        return best;
    }

    private static int IntersectSorted(int[] a, int aCount, int[] b, int[] outBuf)
    {
        int i = 0, j = 0, k = 0;
        while (i < aCount && j < b.Length)
        {
            int va = a[i], vb = b[j];
            if (va == vb) { outBuf[k++] = va; i++; j++; }
            else if (va < vb) i++;
            else j++;
        }
        return k;
    }

    private static int[] MaxCliqueExactBitset(int[][] adj, int n, Stopwatch sw)
    {
        int words = (n + 63) >> 6;
        ulong[][] adjBits = new ulong[n][];
        for (int i = 0; i < n; i++) adjBits[i] = new ulong[words];

        for (int u = 0; u < n; u++)
        {
            foreach (int v in adj[u])
            {
                adjBits[u][v >> 6] |= 1UL << (v & 63);
            }
        }

        // Initial candidate set: all vertices.
        ulong[] P = new ulong[words];
        for (int w = 0; w < words; w++) P[w] = ulong.MaxValue;
        int excess = (words << 6) - n;
        if (excess > 0) P[words - 1] >>= excess;

        ulong[] X = new ulong[words]; // empty
        int[] best = Array.Empty<int>();
        int[] current = new int[n];
        int currentSize = 0;

        // A cheap upper bound: size(current)+popcount(P).
        void BK(ulong[] p, ulong[] x)
        {
            if (sw.Elapsed > SoftTimeLimit) return;

            int pCount = PopCount(p);
            if (currentSize + pCount <= best.Length) return;

            if (pCount == 0)
            {
                if (PopCount(x) == 0 && currentSize > best.Length)
                {
                    best = new int[currentSize];
                    Array.Copy(current, best, currentSize);
                }
                return;
            }

            int pivot = ChoosePivot(p, x, adjBits, n);
            // Candidates = P \ N(pivot)
            ulong[] cand = (ulong[])p.Clone();
            if (pivot >= 0) AndNotInPlace(cand, adjBits[pivot]);

            // Iterate vertices in cand.
            int v;
            while ((v = ExtractFirstSetBit(cand)) >= 0)
            {
                if (sw.Elapsed > SoftTimeLimit) return;

                // Remove v from cand and from p.
                ClearBit(cand, v);
                ClearBit(p, v);

                // Add v to current clique.
                current[currentSize++] = v;

                // p2 = p ∩ N(v), x2 = x ∩ N(v)
                ulong[] p2 = Intersect(p, adjBits[v]);
                ulong[] x2 = Intersect(x, adjBits[v]);

                BK(p2, x2);

                // Backtrack.
                currentSize--;
                SetBit(x, v);
            }
        }

        BK(P, X);
        Array.Sort(best);
        return best.Length == 0 ? new int[] { 0 } : best;
    }

    private static ulong[] Intersect(ulong[] a, ulong[] b)
    {
        ulong[] r = new ulong[a.Length];
        for (int i = 0; i < a.Length; i++) r[i] = a[i] & b[i];
        return r;
    }

    private static void AndNotInPlace(ulong[] a, ulong[] b)
    {
        for (int i = 0; i < a.Length; i++) a[i] &= ~b[i];
    }

    private static int PopCount(ulong[] a)
    {
        int c = 0;
        for (int i = 0; i < a.Length; i++) c += BitOperations.PopCount(a[i]);
        return c;
    }

    private static int ChoosePivot(ulong[] p, ulong[] x, ulong[][] adjBits, int n)
    {
        // Choose u from P ∪ X maximizing |P ∩ N(u)|.
        int bestU = -1;
        int bestCount = -1;
        ulong[] union = (ulong[])p.Clone();
        for (int i = 0; i < union.Length; i++) union[i] |= x[i];

        int u;
        while ((u = ExtractFirstSetBit(union)) >= 0)
        {
            ClearBit(union, u);
            int c = CountIntersectionPop(p, adjBits[u]);
            if (c > bestCount) { bestCount = c; bestU = u; }
        }
        return bestU;
    }

    private static int CountIntersectionPop(ulong[] a, ulong[] b)
    {
        int c = 0;
        for (int i = 0; i < a.Length; i++) c += BitOperations.PopCount(a[i] & b[i]);
        return c;
    }

    private static int ExtractFirstSetBit(ulong[] a)
    {
        for (int i = 0; i < a.Length; i++)
        {
            ulong w = a[i];
            if (w == 0) continue;
            int bit = BitOperations.TrailingZeroCount(w);
            return (i << 6) + bit;
        }
        return -1;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void ClearBit(ulong[] a, int idx)
    {
        a[idx >> 6] &= ~(1UL << (idx & 63));
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void SetBit(ulong[] a, int idx)
    {
        a[idx >> 6] |= 1UL << (idx & 63);
    }

    private sealed class FastScanner
    {
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _pos;

        private int Read()
        {
            if (_pos >= _len)
            {
                _len = Console.OpenStandardInput().Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int x = 0;
            while (c > 32 && c >= 0)
            {
                x = (x * 10) + (c - '0');
                c = Read();
            }
            return x * sign;
        }
    }

    private struct XorShift64
    {
        private ulong _s;
        public XorShift64(ulong seed) { _s = seed == 0 ? 0x9E3779B97F4A7C15UL : seed; }
        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public ulong NextULong()
        {
            ulong x = _s;
            x ^= x << 7;
            x ^= x >> 9;
            _s = x;
            return x;
        }
    }
}
