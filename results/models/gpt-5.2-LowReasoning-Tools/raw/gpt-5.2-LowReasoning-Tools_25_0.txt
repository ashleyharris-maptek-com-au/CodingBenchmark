using System;using System.Collections.Generic;using System.Globalization;public static class Program{    // Two-body heliocentric approximation + Lambert targeting.    // This is intentionally self-contained and avoids external ephemeris parsing.    private const double Day = 86400.0;    private const double AU = 149597870700.0;    private static readonly CultureInfo Ci = CultureInfo.InvariantCulture;    private struct Vec3    {        public double X, Y, Z;        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);        public static Vec3 operator *(double s, Vec3 a) => a * s;        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;        public Vec3 Cross(Vec3 b) => new Vec3(            Y * b.Z - Z * b.Y,            Z * b.X - X * b.Z,            X * b.Y - Y * b.X        );        public double Norm() => Math.Sqrt(Dot(this));        public Vec3 Normalized()        {            double n = Norm();            return n > 0 ? this / n : new Vec3(0, 0, 0);        }        public override string ToString() => $"{X.ToString("R", Ci)} {Y.ToString("R", Ci)} {Z.ToString("R", Ci)}";    }    private sealed class Body    {        public string Name = "";        public double Mass;        public Vec3 R0;        public Vec3 V0;    }    public static void Main()    {        // Read all tokens from stdin (whitespace separated).        string input = Console.In.ReadToEnd();        var tokens = Tokenize(input);        int idx = 0;        if (tokens.Count < 6)        {            // Fallback: output a trivial plan.            Console.WriteLine("1");            Console.WriteLine("0 0 0 0");            return;        }        int numBodies = int.Parse(tokens[idx++], Ci);        double warningDays = ParseDouble(tokens[idx++]);        else        {            double total = 0.0;            foreach (var b in bodies) total += b.Mass;            mu = G * total;        }        if (!(mu > 0)) mu = 1.32712440018e20; // fallback to mu_sun        // Estimate "impact time" as the minimum Earth-Asteroid distance within [0, warningDays].        double impactTime = Math.Max(0.0, warningDays);            bestIntercept = Math.Max(0.1, 0.7 * impactTime);            bestDvLaunch = new Vec3(0, 0, 0);            bestDv = 0.0;        }        double usedDv = bestDv;        if (usedDv > dvBudget && usedDv > 0)        {            double s = dvBudget / usedDv;            bestDvLaunch = bestDvLaunch * s;            usedDv = dvBudget;        }        // Optional small midcourse correction: aim to reduce miss distance under two-body model.        // We compute a correction at halfway time based on linearized position error.        // This is lightweight and bounded.        double tCorr = (bestLaunch + bestIntercept) * 0.5;        Vec3 dvCorr = new Vec3(0, 0, 0);        double remainingBudget = Math.Max(0.0, dvBudget - usedDv);        if (remainingBudget > 1e-3 && (bestIntercept - bestLaunch) > 2.0)        {            // Propagate spacecraft with initial dv and see position at intercept time.            var s0 = PropagateKepler(earth.R0, earth.V0, mu, bestLaunch * Day);            var scEndNoCorr = PropagateKepler(rMid, vMid, mu, (bestIntercept - tCorr) * Day);            Vec3 rScEnd = scEndNoCorr.r;            Vec3 rAstEnd = PropagateKepler(asteroid.R0, asteroid.V0, mu, bestIntercept * Day).r;            Vec3 err = rAstEnd - rScEnd;            // Simple correction: adjust velocity to close position error over remaining time.            double dtRem = (bestIntercept - tCorr) * Day;            Vec3 dvNeed = err / Math.Max(1.0, dtRem);            double dvNeedMag = dvNeed.Norm();            // Clamp correction to a modest fraction of remaining budget to avoid destabilizing.            double dvCap = Math.Min(remainingBudget, Math.Max(0.0, 0.25 * dvBudget));            if (dvNeedMag > dvCap && dvNeedMag > 0) dvNeed = dvNeed * (dvCap / dvNeedMag);            dvCorr = dvNeed;            usedDv += dvCorr.Norm();        }        // Impact burn is zero (kinetic impact at intercept time).        // Output: num_burns then burns (time in days since t0).        int numBurnsOut = (dvCorr.Norm() > 1e-9) ? 3 : 2;        Console.WriteLine(numBurnsOut.ToString(Ci));        Console.WriteLine($"{bestLaunch.ToString("R", Ci)} {bestDvLaunch}");        if (numBurnsOut == 3)            Console.WriteLine($"{tCorr.ToString("R", Ci)} {dvCorr}");        Console.WriteLine($"{bestIntercept.ToString("R", Ci)} 0 0 0");    }    private static List<string> Tokenize(string s)    {        var list = new List<string>();        int n = s.Length;        int i = 0;        while (i < n)        {            while (i < n && char.IsWhiteSpace(s[i])) i++;            if (i >= n) break;            int j = i;            while (j < n && !char.IsWhiteSpace(s[j])) j++;            list.Add(s.Substring(i, j - i));            i = j;        }        return list;    }    private static double ParseDouble(string t)    {        if (double.TryParse(t, NumberStyles.Float, Ci, out double v)) return v;        return 0.0;    }    private static (Vec3 r, Vec3 v) PropagateKepler(Vec3 r0, Vec3 v0, double mu, double dt)    {        // Universal variable Kepler propagation (elliptic/hyperbolic).        double r0n = r0.Norm();        double v0n2 = v0.Dot(v0);        double vr0 = r0.Dot(v0) / Math.Max(1e-12, r0n);        double alpha = 2.0 / Math.Max(1e-12, r0n) - v0n2 / mu;        // Initial guess for chi.        double chi;        if (Math.Abs(alpha) > 1e-12)            chi = Math.Sqrt(mu) * Math.Abs(alpha) * dt;        else            chi = Math.Sqrt(mu) * dt / r0n;        // Newton iterations.        for (int it = 0; it < 60; it++)        {            double z = alpha * chi * chi;            double c2 = StumpffC2(z);            double c3 = StumpffC3(z);            double r = chi * chi * c2 + vr0 * chi * (1 - z * c3) + r0n * (1 - z * c2);            double dtEst = (chi * chi * chi * c3 + vr0 * chi * chi * c2 + r0n * chi * (1 - z * c3)) / Math.Sqrt(mu);            double f = dtEst - dt;            if (Math.Abs(f) < 1e-8) break;            double dtdchi = (chi * chi * c2 + vr0 * chi * (1 - z * c3) + r0n * (1 - z * c2)) / Math.Sqrt(mu);            if (Math.Abs(dtdchi) < 1e-20) break;            double step = f / dtdchi;            chi -= step;            if (Math.Abs(step) < 1e-10) break;        }        double zf = alpha * chi * chi;        double c2f = StumpffC2(zf);        double c3f = StumpffC3(zf);        double fcoef = 1 - (chi * chi / r0n) * c2f;        double gcoef = dt - (chi * chi * chi / Math.Sqrt(mu)) * c3f;        Vec3 r1 = fcoef * r0 + gcoef * v0;        double r1n = r1.Norm();        double fdot = (Math.Sqrt(mu) / (r0n * r1n)) * (zf * c3f - 1) * chi;        double gdot = 1 - (chi * chi / r1n) * c2f;        Vec3 v1 = fdot * r0 + gdot * v0;        return (r1, v1);    }    private static double StumpffC2(double z)    {        if (z > 1e-8)        {            double s = Math.Sqrt(z);            return (1 - Math.Cos(s)) / z;        }        if (z < -1e-8)        {            double s = Math.Sqrt(-z);            return (1 - Math.Cosh(s)) / z;        }        // series        return 0.5 - z / 24.0 + z * z / 720.0;    }    private static double StumpffC3(double z)    {        if (z > 1e-8)        {            double s = Math.Sqrt(z);            return (s - Math.Sin(s)) / (s * s * s);        }        if (z < -1e-8)        {            double s = Math.Sqrt(-z);            return (Math.Sinh(s) - s) / (s * s * s);        }        // series        return 1.0 / 6.0 - z / 120.0 + z * z / 5040.0;    }    private static bool TryLambertUV(Vec3 r1, Vec3 r2, double dt, double mu, bool prograde, out Vec3 v1, out Vec3 v2)    {        // Universal variable Lambert (single-rev), following Vallado-style formulation.        v1 = new Vec3();        v2 = new Vec3();        double r1n = r1.Norm();        double r2n = r2.Norm();        if (!(r1n > 0 && r2n > 0 && dt > 0 && mu > 0)) return false;        double cosDnu = Math.Clamp(r1.Dot(r2) / (r1n * r2n), -1.0, 1.0);        Vec3 cr = r1.Cross(r2);        double sinDnu = cr.Norm() / (r1n * r2n);        if (prograde)        {            if (cr.Z < 0) sinDnu = -sinDnu;        }        else        {            if (cr.Z >= 0) sinDnu = -sinDnu;        }        double dnu = Math.Atan2(sinDnu, cosDnu);        if (dnu < 0) dnu += 2 * Math.PI;        double A = sinDnu * Math.Sqrt(r1n * r2n / (1 - cosDnu));        if (double.IsNaN(A) || Math.Abs(A) < 1e-12) return false;        // Solve for z with Newton + bisection fallback        double z = 0.0;        double zLow = -4 * Math.PI * Math.PI;        double zHigh = 4 * Math.PI * Math.PI;        Func<double, double> timeOfFlight = (zz) =>        {            double c2 = StumpffC2(zz);            double c3 = StumpffC3(zz);            double y = r1n + r2n + (A * (zz * c3 - 1)) / Math.Sqrt(c2);            if (y < 0) return double.NaN;            double x = Math.Sqrt(y / c2);            return (x * x * x * c3 + A * Math.Sqrt(y)) / Math.Sqrt(mu);        };        // Ensure bracket contains a valid solution.        double t0 = timeOfFlight(z);        if (double.IsNaN(t0)) return false;        // Bisection to get close.        for (int it = 0; it < 60; it++)        {            double t = timeOfFlight(z);            if (double.IsNaN(t)) { z = (z + zHigh) * 0.5; continue; }            double f = t - dt;            if (Math.Abs(f) < 1e-6) break;            if (f > 0) zHigh = z;            else zLow = z;            z = 0.5 * (zLow + zHigh);        }        // Refine with Newton.        for (int it = 0; it < 30; it++)        {            double c2 = StumpffC2(z);            double c3 = StumpffC3(z);            double sqrtC2 = Math.Sqrt(Math.Max(1e-30, c2));            double y = r1n + r2n + (A * (z * c3 - 1)) / sqrtC2;            if (y < 0) return false;            double x = Math.Sqrt(y / c2);            double t = (x * x * x * c3 + A * Math.Sqrt(y)) / Math.Sqrt(mu);            double f = t - dt;            if (Math.Abs(f) < 1e-8) break;            // derivative (approx) via finite difference (robust).            double dz = Math.Max(1e-4, 1e-6 * (1 + Math.Abs(z)));            double tPlus = timeOfFlight(z + dz);            double tMinus = timeOfFlight(z - dz);            if (double.IsNaN(tPlus) || double.IsNaN(tMinus)) break;            double dtdz = (tPlus - tMinus) / (2 * dz);            if (Math.Abs(dtdz) < 1e-12) break;            z -= f / dtdz;        }        double c2f = StumpffC2(z);        double c3f = StumpffC3(z);        double yFinal = r1n + r2n + (A * (z * c3f - 1)) / Math.Sqrt(Math.Max(1e-30, c2f));        if (!(yFinal > 0)) return false;        double fLam = 1 - yFinal / r1n;        double gLam = A * Math.Sqrt(yFinal / mu);        double gDotLam = 1 - yFinal / r2n;        if (Math.Abs(gLam) < 1e-12) return false;        v1 = (r2 - fLam * r1) / gLam;        v2 = (gDotLam * r2 - r1) / gLam;        return true;    }}
