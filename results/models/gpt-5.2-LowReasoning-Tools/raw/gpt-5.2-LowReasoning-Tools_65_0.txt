declare const require: any;

// Mission-control scheduler for precedence-constrained tasks on identical parallel machines.
// Reads one JSON document from stdin, prints one JSON schedule to stdout.

type TaskId = string | number;

type InputTask = {
  id: TaskId;
  duration: number;
  deps?: TaskId[];
  deadline?: number;
  weight?: number;
};

type InputData = {
  machineCount: number;
  tasks: InputTask[];
};

type ScheduleEntry = { id: TaskId; start: number; machine: number };

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8").trim();
const data: InputData = input.length ? JSON.parse(input) : ({ machineCount: 1, tasks: [] } as any);

const machineCount = Math.max(1, (data.machineCount | 0) || 1);
const tasksArr: InputTask[] = Array.isArray(data.tasks) ? data.tasks : [];

// Normalize tasks, build id->index map
const n = tasksArr.length;
const idToIndex = new Map<TaskId, number>();
for (let i = 0; i < n; i++) idToIndex.set(tasksArr[i].id, i);

const duration = new Array<number>(n);
const deadline = new Array<number>(n);
const weight = new Array<number>(n);
const depsIdx: number[][] = new Array(n);
const children: number[][] = Array.from({ length: n }, () => []);

for (let i = 0; i < n; i++) {
  const t = tasksArr[i];
  duration[i] = Math.max(0, Number(t.duration) || 0);
  deadline[i] = Number.isFinite(t.deadline as any) ? Number(t.deadline) : Number.POSITIVE_INFINITY;
  weight[i] = Number.isFinite(t.weight as any) ? Math.max(0, Number(t.weight)) : 1;
  const deps = Array.isArray(t.deps) ? t.deps : [];
  const idxs: number[] = [];
  for (const d of deps) {
    const j = idToIndex.get(d);
    if (j !== undefined && j !== i) idxs.push(j);
  }
  depsIdx[i] = idxs;
}

for (let i = 0; i < n; i++) {
  for (const d of depsIdx[i]) children[d].push(i);
}

// Kahn topo sort (for cycle handling / heuristics); if cycles, fall back to input order.
const indeg0 = new Array<number>(n);
for (let i = 0; i < n; i++) indeg0[i] = depsIdx[i].length;
const q: number[] = [];
for (let i = 0; i < n; i++) if (indeg0[i] === 0) q.push(i);
const topo: number[] = [];
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  topo.push(u);
  for (const v of children[u]) {
    indeg0[v]--;
    if (indeg0[v] === 0) q.push(v);
  }
}
const hasCycle = topo.length !== n;

// Earliest-start lower bound from precedence only (ignores machines)
const est = new Array<number>(n).fill(0);
if (!hasCycle) {
  for (const u of topo) {
    const finish = est[u] + duration[u];
    for (const v of children[u]) {
      if (finish > est[v]) est[v] = finish;
    }
  }
}

// Critical-path length to sinks (reverse topo)
const cpl = new Array<number>(n).fill(0);
if (!hasCycle) {
  for (let k = topo.length - 1; k >= 0; k--) {
    const u = topo[k];
    let bestChild = 0;
    for (const v of children[u]) if (cpl[v] > bestChild) bestChild = cpl[v];
    cpl[u] = duration[u] + bestChild;
  }
} else {
  // If cyclic, at least have a monotone proxy.
  for (let i = 0; i < n; i++) cpl[i] = duration[i];
}

// Greedy list scheduler producing explicit (machine, start) assignments.
const scheduled = new Array<boolean>(n).fill(false);
const remainingDeps = new Array<number>(n);
for (let i = 0; i < n; i++) remainingDeps[i] = depsIdx[i].length;

const depFinishMax = new Array<number>(n).fill(0);
const finishTime = new Array<number>(n).fill(0);

// tasks become "schedulable" when all dependencies are scheduled (finish times known)
const schedulable: number[] = [];
for (let i = 0; i < n; i++) if (remainingDeps[i] === 0) schedulable.push(i);

const machineFree = new Array<number>(machineCount).fill(0);
const schedule: ScheduleEntry[] = [];

function scoreTask(i: number, startAt: number): number {
  const d = duration[i];
  const c = startAt + d;
  const dl = deadline[i];
  const w = weight[i];
  // tardiness pressure (higher if late), slack (higher if safe)
  const tard = Math.max(0, c - dl);
  const slack = Math.max(0, dl - c);
  // Priority: critical path & weight, plus urgency if at risk of missing deadline.
  // Tuned to be stable across scales (avoid huge constants).
  const cp = cpl[i];
  const urgency = tard > 0 ? (1 + Math.log1p(tard)) : 1 / (1 + Math.log1p(slack));
  const wTerm = Math.log1p(w);
  return 3.0 * Math.log1p(cp) + 2.0 * wTerm * urgency + 0.2 * Math.log1p(d);
}

function pickMachine(): number {
  let best = 0;
  let bestT = machineFree[0];
  for (let m = 1; m < machineCount; m++) {
    const t = machineFree[m];
    if (t < bestT) {
      bestT = t;
      best = m;
    }
  }
  return best;
}

let scheduledCount = 0;
while (scheduledCount < n) {
  // If we're stuck (cycle / missing deps), schedule something unscheduled in input order.
  if (schedulable.length === 0) {
    let forced = -1;
    for (let i = 0; i < n; i++) {
      if (!scheduled[i]) {
        forced = i;
        break;
      }
    }
    if (forced === -1) break;
    // Treat as schedulable by ignoring remaining deps to avoid infinite loop.
    schedulable.push(forced);
  }

  const m = pickMachine();
  const freeT = machineFree[m];

  // Choose best task for this machine by a look-ahead start time.
  let bestIdx = -1;
  let bestScore = -Infinity;
  let bestStart = 0;

  // Iterate all schedulable tasks; keep array compact by swap-delete.
  for (let k = 0; k < schedulable.length; k++) {
    const i = schedulable[k];
    if (scheduled[i]) continue;
    const startAt = Math.max(freeT, depFinishMax[i]);
    const s = scoreTask(i, startAt);
    if (s > bestScore) {
      bestScore = s;
      bestIdx = k;
      bestStart = startAt;
    }
  }

  if (bestIdx < 0) {
    // Nothing valid (shouldn't happen), advance earliest machine.
    machineFree[m] = freeT + 1;
    continue;
  }

  const taskIndex = schedulable[bestIdx];
  // Remove from schedulable
  schedulable[bestIdx] = schedulable[schedulable.length - 1];
  schedulable.pop();

  const st = bestStart;
  const ft = st + duration[taskIndex];
  scheduled[taskIndex] = true;
  finishTime[taskIndex] = ft;
  machineFree[m] = ft;
  schedule.push({ id: tasksArr[taskIndex].id, start: st, machine: m });
  scheduledCount++;

  // Unlock children (dependency "scheduled", and propagate dep-finish requirement)
  for (const v of children[taskIndex]) {
    if (ft > depFinishMax[v]) depFinishMax[v] = ft;
    remainingDeps[v]--;
    if (remainingDeps[v] === 0) schedulable.push(v);
  }
}

process.stdout.write(JSON.stringify({ schedule }));
