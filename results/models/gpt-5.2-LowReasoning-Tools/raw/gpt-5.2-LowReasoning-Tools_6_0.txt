using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

public static class Program
{
    private const double TwoPi = Math.PI * 2.0;

    private readonly struct Vec3
    {
        public readonly double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new(a.X * s, a.Y * s, a.Z * s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(Vec3 b) => new(Y * b.Z - Z * b.Y, Z * b.X - X * b.Z, X * b.Y - Y * b.X);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    private readonly struct OrbEl
    {
        public readonly double A;     // semi-major axis (km)
        public readonly double E;     // eccentricity
        public readonly double I;     // inclination (rad)
        public readonly double Raan;  // right ascension of ascending node (rad)
        public readonly double ArgP;  // argument of periapsis (rad)
        public readonly double Nu;    // true anomaly at epoch (rad)

        public OrbEl(double a, double e, double i, double raan, double argp, double nu)
        {
            A = a; E = e; I = i; Raan = raan; ArgP = argp; Nu = nu;
        }
    }

    private static double WrapAngle(double a)
    {
        a %= TwoPi;
        if (a < 0) a += TwoPi;
        return a;
    }

    private static double AngleDiff(double a, double b)
    {
        double d = Math.Abs(a - b) % TwoPi;
        return d > Math.PI ? TwoPi - d : d;
    }

    private static OrbEl ElementsFromState(Vec3 r, Vec3 v, double mu)
    {
        double rmag = r.Norm();
        double vmag = v.Norm();

        Vec3 h = r.Cross(v);
        double hmag = h.Norm();

        // Inclination
        double inc = hmag > 0 ? Math.Acos(Math.Clamp(h.Z / hmag, -1.0, 1.0)) : 0.0;

        // Node vector
        Vec3 k = new(0, 0, 1);
        Vec3 n = k.Cross(h);
        double nmag = n.Norm();

        // Eccentricity vector
        Vec3 evec = (v.Cross(h) * (1.0 / mu)) - (r * (1.0 / rmag));
        double ecc = evec.Norm();

        // Semi-major axis from specific orbital energy
        double energy = 0.5 * vmag * vmag - mu / rmag;
        double a;
        if (Math.Abs(energy) < 1e-12)
        {
            // Near-parabolic; fall back to large a to keep features finite
            a = 1e9;
        }
        else
        {
            a = -mu / (2.0 * energy);
        }

        // RAAN
        double raan = 0.0;
        if (nmag > 1e-12)
            raan = WrapAngle(Math.Atan2(n.Y, n.X));

        // Argument of periapsis
        double argp = 0.0;
        if (nmag > 1e-12 && ecc > 1e-10)
        {
            // argp = angle between n and evec, measured in orbital plane
            double cosArgp = Math.Clamp(n.Dot(evec) / (nmag * ecc), -1.0, 1.0);
            argp = Math.Acos(cosArgp);
            if (evec.Z < 0) argp = TwoPi - argp;
            argp = WrapAngle(argp);
        }
        else
        {
            argp = 0.0;
        }

        // True anomaly
        double nu = 0.0;
        if (ecc > 1e-10)
        {
            double cosNu = Math.Clamp(evec.Dot(r) / (ecc * rmag), -1.0, 1.0);
            nu = Math.Acos(cosNu);
            if (r.Dot(v) < 0) nu = TwoPi - nu;
            nu = WrapAngle(nu);
        }
        else
        {
            // For near-circular, use argument of latitude from node vector
            if (nmag > 1e-12)
            {
                double cosU = Math.Clamp(n.Dot(r) / (nmag * rmag), -1.0, 1.0);
                nu = Math.Acos(cosU);
                if (r.Z < 0) nu = TwoPi - nu;
                nu = WrapAngle(nu);
            }
            else
            {
                // Equatorial + circular: just use atan2 in inertial xy
                nu = WrapAngle(Math.Atan2(r.Y, r.X));
            }
        }

        return new OrbEl(a, ecc, inc, raan, argp, nu);
    }

    // Cheap surrogate for delta-V between orbits at epoch: distance in element-space.
    private static double Cost(in OrbEl a, in OrbEl b)
    {
        // Scale semi-major axis roughly around LEO/MEO.
        double da = Math.Abs(a.A - b.A) / 8000.0;
        double de = Math.Abs(a.E - b.E) * 4.0;
        double di = AngleDiff(a.I, b.I) * 2.0;
        double draan = AngleDiff(a.Raan, b.Raan) * 1.5;
        double dargp = AngleDiff(a.ArgP, b.ArgP) * 1.0;
        double dnu = AngleDiff(a.Nu, b.Nu) * 0.5;

        // Quadratic blend; keep it symmetric and smooth.
        return Math.Sqrt(da * da + de * de + di * di + draan * draan + dargp * dargp + dnu * dnu);
    }

    private static int[] SolveBruteforce(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        var used = new bool[n];
        var bestOrder = new int[n];
        var curOrder = new int[n];
        double best = double.PositiveInfinity;

        void Dfs(int depth, int last, double acc)
        {
            if (acc >= best) return;
            if (depth == n)
            {
                best = acc;
                Array.Copy(curOrder, bestOrder, n);
                return;
            }

            // Simple branching order: try cheapest next first.
            int remaining = n - depth;
            var opts = new (int idx, double c)[remaining];
            int k = 0;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double c = depth == 0 ? startCost[i] : cost[last, i];
                opts[k++] = (i, c);
            }
            Array.Sort(opts, 0, k, Comparer<(int idx, double c)>.Create((p, q) => p.c.CompareTo(q.c)));

            for (int t = 0; t < k; t++)
            {
                int i = opts[t].idx;
                double c = opts[t].c;
                used[i] = true;
                curOrder[depth] = i;
                Dfs(depth + 1, i, acc + c);
                used[i] = false;
            }
        }

        Dfs(0, -1, 0.0);
        return bestOrder;
    }

    private static int[] NearestNeighbor(double[] startCost, double[,] cost)
    {
        int n = startCost.Length;
        var order = new int[n];
        var used = new bool[n];

        // Start with closest-to-start
        int cur = 0;
        double best = double.PositiveInfinity;
        for (int i = 0; i < n; i++)
        {
            if (startCost[i] < best) { best = startCost[i]; cur = i; }
        }
        order[0] = cur;
        used[cur] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int next = -1;
            double bestc = double.PositiveInfinity;
            for (int j = 0; j < n; j++)
            {
                if (used[j]) continue;
                double c = cost[cur, j];
                if (c < bestc)
                {
                    bestc = c;
                    next = j;
                }
            }
            order[pos] = next;
            used[next] = true;
            cur = next;
        }
        return order;
    }

    private static double PathCost(int[] order, double[] startCost, double[,] cost)
    {
        if (order.Length == 0) return 0;
        double total = startCost[order[0]];
        for (int i = 1; i < order.Length; i++) total += cost[order[i - 1], order[i]];
        return total;
    }

    private static void TwoOptImprove(int[] order, double[] startCost, double[,] cost, Stopwatch sw, int maxMs)
    {
        int n = order.Length;
        if (n < 4) return;

        // Evaluate edges as directed path edges (no return-to-start).
        double CurrentEdgeCost(int i, int j)
        {
            // edge from node at position i-1 to i, with special start edge
            if (i == 0) return startCost[order[0]];
            return cost[order[i - 1], order[i]];
        }

        while (sw.ElapsedMilliseconds < maxMs)
        {
            bool improved = false;

            for (int i = 0; i < n - 2 && sw.ElapsedMilliseconds < maxMs; i++)
            {
                int a = order[i];
                int aPrev = (i == 0) ? -1 : order[i - 1];
                for (int k = i + 1; k < n - 1 && sw.ElapsedMilliseconds < maxMs; k++)
                {
                    int b = order[k];
                    int bNext = order[k + 1];

                    // current edges: (aPrev->a) + (b->bNext)
                    // new edges after reversing [i..k]: (aPrev->b) + (a->bNext)
                    double cur1 = (i == 0) ? startCost[a] : cost[aPrev, a];
                    double cur2 = cost[b, bNext];
                    double new1 = (i == 0) ? startCost[b] : cost[aPrev, b];
                    double new2 = cost[a, bNext];
                    double gain = (cur1 + cur2) - (new1 + new2);
                    if (gain > 1e-12)
                    {
                        Array.Reverse(order, i, k - i + 1);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }

            if (!improved) break;
        }
    }

    public static void Main()
    {
        var ci = CultureInfo.InvariantCulture;
        string? line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        var t1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int n = int.Parse(t1[0], ci);
        double mu = double.Parse(t1[1], ci);

        static (Vec3 r, Vec3 v) ReadState(string line)
        {
            var p = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            double x = double.Parse(p[0], CultureInfo.InvariantCulture);
            double y = double.Parse(p[1], CultureInfo.InvariantCulture);
            double z = double.Parse(p[2], CultureInfo.InvariantCulture);
            double vx = double.Parse(p[3], CultureInfo.InvariantCulture);
            double vy = double.Parse(p[4], CultureInfo.InvariantCulture);
            double vz = double.Parse(p[5], CultureInfo.InvariantCulture);
            return (new Vec3(x, y, z), new Vec3(vx, vy, vz));
        }

        string? startLine = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(startLine)) return;
        var startState = ReadState(startLine);
        OrbEl startEl = ElementsFromState(startState.r, startState.v, mu);

        var els = new OrbEl[n];
        for (int i = 0; i < n; i++)
        {
            string? s = Console.ReadLine();
            if (s == null) s = "";
            var st = ReadState(s);
            els[i] = ElementsFromState(st.r, st.v, mu);
        }

        // Precompute costs
        var startCost = new double[n];
        for (int i = 0; i < n; i++) startCost[i] = Cost(startEl, els[i]);

        var cost = new double[n, n];
        for (int i = 0; i < n; i++)
        {
            cost[i, i] = 0;
            for (int j = i + 1; j < n; j++)
            {
                double c = Cost(els[i], els[j]);
                cost[i, j] = c;
                cost[j, i] = c;
            }
        }

        int[] order;
        if (n <= 9)
        {
            order = SolveBruteforce(startCost, cost);
        }
        else
        {
            order = NearestNeighbor(startCost, cost);
            var sw = Stopwatch.StartNew();
            int maxMs;
            if (n <= 20) maxMs = 1200;
            else if (n <= 50) maxMs = 2000;
            else maxMs = 2600;
            TwoOptImprove(order, startCost, cost, sw, maxMs);
        }

        Console.WriteLine(string.Join(' ', order));
    }
}
