#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint idx = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (idx >= N) return;

    uint base = idx * 2u;
    vec3 pos = inData[base + 0u].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    float sepR2 = sepRadius * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2 = cohRadius * cohRadius;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < N; ++j) {
        if (j == idx) continue;

        uint jbase = j * 2u;
        vec3 op = inData[jbase + 0u].xyz;
        vec3 ov = inData[jbase + 1u].xyz;

        vec3 d = op - pos;
        float dist2 = dot(d, d);
        if (dist2 <= 0.0) continue;

        if (dist2 < sepR2) {
            float invDist = inversesqrt(dist2);
            // Push away from neighbor; stronger when closer.
            sepSum -= d * invDist;
            sepCount++;
        }

        if (dist2 < alignR2) {
            alignSum += ov;
            alignCount++;
        }

        if (dist2 < cohR2) {
            cohSum += op;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        vec3 sepDir = sepSum / float(sepCount);
        accel += sepDir * sepWeight;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignSum / float(alignCount);
        vec3 steerAlign = (avgVel - vel);
        accel += steerAlign * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        vec3 steerCoh = (center - pos);
        accel += steerCoh * cohWeight;
    }

    // Boundary force: when exceeding boundSize*0.4, push toward center.
    float limit = boundSize * 0.4;
    float half = boundSize * 0.5;
    if (pos.x > limit)  accel.x -= boundForce * ((pos.x - limit) / half);
    if (pos.x < -limit) accel.x += boundForce * ((-limit - pos.x) / half);
    if (pos.y > limit)  accel.y -= boundForce * ((pos.y - limit) / half);
    if (pos.y < -limit) accel.y += boundForce * ((-limit - pos.y) / half);
    if (pos.z > limit)  accel.z -= boundForce * ((pos.z - limit) / half);
    if (pos.z < -limit) accel.z += boundForce * ((-limit - pos.z) / half);

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = vel * (maxSpeed / speed);
    }

    pos += vel * dt;

    outData[base + 0u] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
