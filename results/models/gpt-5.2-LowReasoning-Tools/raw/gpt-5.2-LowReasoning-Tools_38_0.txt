import sys


def solve(U, sets):
    # Build incidence structures
    col_rows = {c: set() for c in range(U)}
    row_cols = []
    for r, elems in enumerate(sets):
        cols = []
        seen = set()
        for e in elems:
            if 0 <= e < U and e not in seen:
                seen.add(e)
                cols.append(e)
        row_cols.append(cols)
        for c in cols:
            col_rows[c].add(r)

    # Quick infeasibility: any column with no rows
    for c in range(U):
        if not col_rows[c]:
            return None

    solution = []

    def choose_column():
        # Choose uncovered column with smallest candidate rows
        best_c = None
        best_len = None
        for c, rows in col_rows.items():
            if rows is None:
                continue  # already covered
            l = len(rows)
            if best_len is None or l < best_len:
                best_len = l
                best_c = c
                if l <= 1:
                    break
        return best_c

    def cover(row):
        # Cover all columns in this row; remove conflicting rows from other columns.
        removed = []  # list of (col, prev_rows_set)
        for c in row_cols[row]:
            rows_c = col_rows.get(c)
            if rows_c is None:
                # Column already covered => selecting this row would cover element twice
                return None

        # First, gather all rows to remove per column
        to_remove_per_col = []
        for c in row_cols[row]:
            rows_c = col_rows[c]
            to_remove = list(rows_c)
            to_remove_per_col.append((c, to_remove))

        # Now perform removals
        for c, to_remove in to_remove_per_col:
            prev = col_rows[c]
            removed.append((c, prev))
            col_rows[c] = None  # mark covered
            for r2 in to_remove:
                if r2 == row:
                    continue
                for c2 in row_cols[r2]:
                    rows_set = col_rows.get(c2)
                    if rows_set is not None and r2 in rows_set:
                        rows_set.remove(r2)
        return removed

    def uncover(removed):
        # Restore in reverse order
        # First restore columns as uncovered
        # Note: removed stores original set objects, which were mutated; we need full restore.
        # We'll reconstruct from snapshots by storing copies instead.
        for c, prev in reversed(removed):
            col_rows[c] = prev

    # We need proper snapshotting since we mutate sets.
    # We'll instead implement cover/uncover with explicit tracking of deletions.

    def cover_row(row):
        covered_cols = []
        removed_from_cols = []  # list of (col, removed_rows_list)

        # reject if any column already covered
        for c in row_cols[row]:
            if col_rows[c] is None:
                return None

        # For each column c in row, cover it and remove rows that intersect it
        for c in row_cols[row]:
            rows_c = col_rows[c]
            covered_cols.append(c)
            col_rows[c] = None

            removed_rows = []
            for r2 in list(rows_c):
                if r2 == row:
                    continue
                # remove r2 from all other uncovered columns it belongs to
                for c2 in row_cols[r2]:
                    rows_set = col_rows[c2]
                    if rows_set is not None and r2 in rows_set:
                        rows_set.remove(r2)
                removed_rows.append(r2)
            removed_from_cols.append((c, rows_c, removed_rows))
        return covered_cols, removed_from_cols

    def uncover_row(state):
        covered_cols, removed_from_cols = state
        # restore columns in reverse
        for c, rows_c, removed_rows in reversed(removed_from_cols):
            # restore removed rows back into uncovered columns
            for r2 in removed_rows:
                for c2 in row_cols[r2]:
                    rows_set = col_rows[c2]
                    if rows_set is not None:
                        rows_set.add(r2)
            col_rows[c] = rows_c

    def backtrack():
        # If all columns covered
        if all(v is None for v in col_rows.values()):
            return True
        c = choose_column()
        if c is None:
            return True
        rows = col_rows[c]
        if rows is None or len(rows) == 0:
            return False

        for r in sorted(rows):
            state = cover_row(r)
            if state is None:
                continue
            solution.append(r)
            if backtrack():
                return True
            solution.pop()
            uncover_row(state)
        return False

    ok = backtrack()
    if not ok:
        return None
    return sorted(solution)


def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    first = data[0].strip().split()
    if len(first) < 2:
        return
    U = int(first[0])
    S = int(first[1])

    sets = []
    for i in range(S):
        line = data[i + 1] if i + 1 < len(data) else ''
        line = line.strip()
        if not line:
            sets.append([])
        else:
            sets.append([int(x) for x in line.split()])

    sol = solve(U, sets)
    out = []
    if sol is None:
        out.append('NO SOLUTION')
    else:
        out.append('SOLUTION')
        out.append(' '.join(str(i) for i in sol))
    sys.stdout.write('\n'.join(out))


if __name__ == '__main__':
    main()
