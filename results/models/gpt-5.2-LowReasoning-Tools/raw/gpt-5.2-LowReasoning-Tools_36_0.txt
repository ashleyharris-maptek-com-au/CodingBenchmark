#include <bits/stdc++.h>
using namespace std;

// Exact Maximum Independent Set via Maximum Clique on the complement graph
// using a Tomita-style branch-and-bound with greedy coloring.

struct DynBitset {
    int n = 0;
    int words = 0;
    vector<uint64_t> w;

    DynBitset() = default;
    explicit DynBitset(int n_) { reset_size(n_); }

    void reset_size(int n_) {
        n = n_;
        words = (n + 63) >> 6;
        w.assign(words, 0ULL);
    }

    inline void set_bit(int i) { w[i >> 6] |= (1ULL << (i & 63)); }
    inline void clear_bit(int i) { w[i >> 6] &= ~(1ULL << (i & 63)); }
    inline bool test_bit(int i) const { return (w[i >> 6] >> (i & 63)) & 1ULL; }

    inline bool any() const {
        for (uint64_t x : w) if (x) return true;
        return false;
    }

    inline void zero() { std::fill(w.begin(), w.end(), 0ULL); }

    inline int pop_lsb() {
        for (int wi = 0; wi < words; ++wi) {
            uint64_t x = w[wi];
            if (!x) continue;
            unsigned b = (unsigned)__builtin_ctzll(x);
            w[wi] = x & (x - 1);
            return (wi << 6) + (int)b;
        }
        return -1;
    }

    inline int count() const {
        int c = 0;
        for (uint64_t x : w) c += (int)__builtin_popcountll(x);
        return c;
    }

    inline void and_with(const DynBitset& other) {
        for (int i = 0; i < words; ++i) w[i] &= other.w[i];
    }

    inline DynBitset operator&(const DynBitset& other) const {
        DynBitset r;
        r.n = n;
        r.words = words;
        r.w.resize(words);
        for (int i = 0; i < words; ++i) r.w[i] = w[i] & other.w[i];
        return r;
    }

    inline DynBitset operator~() const {
        DynBitset r(*this);
        for (int i = 0; i < words; ++i) r.w[i] = ~r.w[i];
        // clear extra bits beyond n
        int extra = (words << 6) - n;
        if (extra) {
            uint64_t mask = (~0ULL) >> extra;
            r.w[words - 1] &= mask;
        }
        return r;
    }

    inline void or_with(const DynBitset& other) {
        for (int i = 0; i < words; ++i) w[i] |= other.w[i];
    }
};

struct MaxCliqueComplement {
    int n;
    vector<DynBitset> adj; // adjacency in complement graph
    vector<int> best_clique;

    explicit MaxCliqueComplement(int n_) : n(n_), adj(n_) {
        for (int i = 0; i < n; ++i) adj[i].reset_size(n);
    }

    // Greedy coloring of candidate set P in graph adj, producing an order with non-decreasing colors.
    // Returns vertices in "order" and their greedy colors in "colors" (same size).
    void greedy_color(const DynBitset& P_in, vector<int>& order, vector<int>& colors) const {
        DynBitset P = P_in;
        order.clear();
        colors.clear();
        order.reserve(P.count());
        colors.reserve(order.capacity());

        int color = 0;
        while (P.any()) {
            ++color;
            DynBitset Q = P; // vertices not yet colored
            DynBitset R = P; // vertices available for this color class
            // Build one color class as a maximal independent set in the graph:
            // repeatedly pick a vertex, remove it and its neighbors from R.
            while (R.any()) {
                int v = R.pop_lsb();
                if (v < 0) break;
                order.push_back(v);
                colors.push_back(color);
                // remove v from P and Q implicitly by clearing later
                // For this color class, forbid neighbors of v (and v itself).
                DynBitset forbid = adj[v]; // neighbors in complement graph
                forbid.set_bit(v);
                // R = R & ~forbid
                for (int i = 0; i < R.words; ++i) R.w[i] &= ~forbid.w[i];
            }
            // Remove all vertices colored with this color from P:
            // They are exactly those in (Q - R') but we didn't track; simplest: recompute by scanning order tail.
            // Instead: rebuild P by clearing bits for vertices we assigned this color (since count is moderate).
            // This is O(n) over colored vertices, acceptable within algorithm.
            int sz = (int)order.size();
            // Find vertices in this color class: they are at the end of order with colors==color.
            for (int i = sz - 1; i >= 0 && colors[i] == color; --i) {
                P.clear_bit(order[i]);
            }
        }
        // Tomita expects vertices ordered by non-decreasing color; we already append by color class, so OK.
    }

    // Choose a pivot from P to reduce branching: pick u in P with max |P ∩ N(u)|.
    int choose_pivot(const DynBitset& P) const {
        int pivot = -1;
        int best_deg = -1;
        // Iterate vertices in P (by popping from a copy).
        DynBitset tmp = P;
        while (tmp.any()) {
            int u = tmp.pop_lsb();
            if (u < 0 || u >= n) continue;
            int deg = (P & adj[u]).count();
            if (deg > best_deg) {
                best_deg = deg;
                pivot = u;
            }
        }
        return pivot;
    }

    void expand(vector<int>& C, DynBitset P) {
        if (!P.any()) {
            if ((int)C.size() > (int)best_clique.size()) best_clique = C;
            return;
        }

        // Bound via greedy coloring
        vector<int> order, colors;
        greedy_color(P, order, colors);

        // Explore vertices in reverse order (highest color first) for better pruning
        for (int idx = (int)order.size() - 1; idx >= 0; --idx) {
            int v = order[idx];
            int bound = (int)C.size() + colors[idx];
            if (bound <= (int)best_clique.size()) return; // cannot beat best

            if (!P.test_bit(v)) continue; // v might have been removed by earlier branching

            C.push_back(v);
            DynBitset P2 = P & adj[v];
            expand(C, P2);
            C.pop_back();
            P.clear_bit(v);
        }
    }

    vector<int> maximum_clique() {
        best_clique.clear();

        DynBitset P(n);
        for (int i = 0; i < n; ++i) P.set_bit(i);
        vector<int> C;
        C.reserve(n);

        // A quick greedy clique to seed a lower bound
        {
            DynBitset G = P;
            while (G.any()) {
                int v = G.pop_lsb();
                if (v < 0) break;
                best_clique.push_back(v);
                G.and_with(adj[v]);
            }
        }

        expand(C, P);
        return best_clique;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    long long mll;
    if (!(cin >> n >> mll)) return 0;
    long long m = mll;

    // Handle trivial cases quickly
    if (n <= 0) {
        cout << 0 << "\n\n";
        return 0;
    }

    // Read original graph adjacency as bitsets
    vector<DynBitset> g(n);
    for (int i = 0; i < n; ++i) g[i].reset_size(n);

    bool has_self_loop = false;
    for (long long i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= n || v >= n) continue; // ignore invalid edges
        if (u == v) {
            has_self_loop = true;
            g[u].set_bit(u);
        } else {
            g[u].set_bit(v);
            g[v].set_bit(u);
        }
    }

    // If there are self-loops, those vertices cannot be in an independent set.
    // We'll simply exclude them by adding edges to themselves in complement later.

    // Special fast paths
    if (m == 0) {
        cout << n << "\n";
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << i;
        }
        cout << "\n";
        return 0;
    }
    // Complete graph (no independent set larger than 1, unless n==0)
    if (!has_self_loop) {
        long long max_edges = 1LL * n * (n - 1) / 2;
        if (m >= max_edges) {
            cout << 1 << "\n0\n";
            return 0;
        }
    }

    // Build complement graph adjacency for maximum clique:
    // In complement, edge exists if not edge in original and u!=v, and u,v are not self-looped together.
    MaxCliqueComplement solver(n);
    DynBitset all(n);
    for (int i = 0; i < n; ++i) all.set_bit(i);

    for (int u = 0; u < n; ++u) {
        DynBitset comp = all;
        // remove neighbors in original graph
        for (int wi = 0; wi < comp.words; ++wi) comp.w[wi] &= ~g[u].w[wi];
        comp.clear_bit(u); // no self edge for clique adjacency
        // If u has a self-loop in original, disallow u entirely by clearing all adjacency
        if (g[u].test_bit(u)) {
            comp.zero();
        }
        solver.adj[u] = std::move(comp);
    }

    // Maximum clique in complement corresponds to maximum independent set in original
    vector<int> indep = solver.maximum_clique();

    // Output
    cout << indep.size() << "\n";
    for (size_t i = 0; i < indep.size(); ++i) {
        if (i) cout << ' ';
        cout << indep[i];
    }
    cout << "\n";
    return 0;
}
