// N-Dimensional Snake Game solver
// Reads full game description from stdin and outputs a move each turn (axis direction).
// Strategy: simulate the snake, greedily head toward nearest remaining food using a safe
// one-step lookahead; fall back to any safe move if blocked.

#include <algorithm>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::cin;
using std::cout;
using std::deque;
using std::int64_t;
using std::size_t;
using std::string;
using std::unordered_map;
using std::unordered_set;
using std::vector;

struct VecHash {
  size_t operator()(const vector<int>& v) const noexcept {
    // 64-bit mix (splitmix-like) over ints
    uint64_t x = 0x9e3779b97f4a7c15ULL;
    for (int a : v) {
      uint64_t z = static_cast<uint64_t>(static_cast<uint32_t>(a)) + 0x9e3779b97f4a7c15ULL;
      z ^= z >> 30;
      z *= 0xbf58476d1ce4e5b9ULL;
      z ^= z >> 27;
      z *= 0x94d049bb133111ebULL;
      z ^= z >> 31;
      x ^= z + 0x9e3779b97f4a7c15ULL + (x << 6) + (x >> 2);
    }
    return static_cast<size_t>(x);
  }
};

static inline int64_t manhattan(const vector<int>& a, const vector<int>& b) {
  int64_t s = 0;
  for (size_t i = 0; i < a.size(); ++i) s += std::llabs((int64_t)a[i] - (int64_t)b[i]);
  return s;
}

static inline bool in_bounds(const vector<int>& p, const vector<int>& bounds) {
  for (size_t i = 0; i < p.size(); ++i) {
    if (p[i] < 0 || p[i] >= bounds[i]) return false;
  }
  return true;
}

static inline vector<int> add_step(const vector<int>& p, int axis, int dir) {
  vector<int> q = p;
  q[axis] += dir;
  return q;
}

struct Move {
  int axis = 0;
  int dir = 1;
};

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int D = 0;
  int max_turns = 0;
  if (!(cin >> D >> max_turns)) return 0;

  vector<int> bounds(D);
  for (int i = 0; i < D; ++i) cin >> bounds[i];

  int snake_len = 0;
  vector<int> head(D);
  cin >> snake_len;
  for (int i = 0; i < D; ++i) cin >> head[i];

  int F = 0;
  cin >> F;
  vector<vector<int>> foods;
  foods.reserve(std::max(0, F));
  for (int i = 0; i < F; ++i) {
    vector<int> p(D);
    for (int j = 0; j < D; ++j) cin >> p[j];
    foods.push_back(std::move(p));
  }

  int O = 0;
  cin >> O;
  unordered_set<vector<int>, VecHash> obstacles;
  obstacles.reserve(static_cast<size_t>(std::max(0, O)) * 2u + 1u);
  for (int i = 0; i < O; ++i) {
    vector<int> p(D);
    for (int j = 0; j < D; ++j) cin >> p[j];
    obstacles.insert(std::move(p));
  }

  // Food lookup: position -> remaining count (usually 1).
  unordered_map<vector<int>, int, VecHash> food_count;
  food_count.reserve(static_cast<size_t>(foods.size()) * 2u + 1u);
  for (const auto& f : foods) food_count[f] += 1;

  // Snake body representation as deque of positions (head at front).
  // Since only length and head are provided, assume "stacked" start: all segments at head.
  deque<vector<int>> body;
  body.resize(std::max(1, snake_len), head);

  // Occupancy counts to handle the stacked start correctly.
  unordered_map<vector<int>, int, VecHash> occ;
  occ.reserve(static_cast<size_t>(body.size()) * 2u + 8u);
  occ[head] = (int)body.size();

  auto can_move_to = [&](const vector<int>& next, bool will_pop_tail, const vector<int>& tail_pos) -> bool {
    if (!in_bounds(next, bounds)) return false;
    if (obstacles.find(next) != obstacles.end()) return false;
    auto it = occ.find(next);
    if (it == occ.end()) return true;
    // If next is occupied, it can still be valid only if the tail is moving away from that same cell
    // and the cell's occupancy count would drop to 0 before the head arrives (standard snake rule).
    // With counts, allow collision-free if (next == tail_pos) and occ[next] == 1 and will_pop_tail.
    if (will_pop_tail && next == tail_pos && it->second == 1) return true;
    return false;
  };

  auto free_neighbor_count = [&](const vector<int>& p, bool will_pop_tail, const vector<int>& tail_pos) -> int {
    int c = 0;
    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, 1}) {
        vector<int> q = add_step(p, axis, dir);
        if (can_move_to(q, will_pop_tail, tail_pos)) ++c;
      }
    }
    return c;
  };

  vector<Move> planned;
  planned.reserve(std::max(0, max_turns));

  for (int turn = 0; turn < max_turns; ++turn) {
    if (food_count.empty()) break; // win

    // Choose target: nearest (Manhattan).
    vector<int> target;
    int64_t best_dist = std::numeric_limits<int64_t>::max();
    for (const auto& kv : food_count) {
      int64_t d = manhattan(head, kv.first);
      if (d < best_dist) {
        best_dist = d;
        target = kv.first;
      }
    }

    const vector<int>& tail_pos = body.back();
    // Determine if we will grow on this move for collision rules.
    // We don't know yet; collision check needs to know will_pop_tail; approximate by checking each candidate.

    // Evaluate all moves.
    Move chosen{0, 1};
    bool found = false;
    int64_t chosen_score = std::numeric_limits<int64_t>::max();
    int chosen_mobility = -1;

    for (int axis = 0; axis < D; ++axis) {
      for (int dir : {-1, 1}) {
        vector<int> next = add_step(head, axis, dir);
        bool will_grow = (food_count.find(next) != food_count.end());
        bool will_pop_tail = !will_grow;
        if (!can_move_to(next, will_pop_tail, tail_pos)) continue;

        int64_t d = manhattan(next, target);
        // Prefer moves that reduce distance; if equal, prefer higher local mobility.
        int mobility = free_neighbor_count(next, will_pop_tail, tail_pos);
        // Primary: distance to target, Secondary: -mobility, Tertiary: keep axis small (stable)
        // Compose as lexicographic.
        if (!found || d < chosen_score || (d == chosen_score && mobility > chosen_mobility)) {
          found = true;
          chosen = Move{axis, dir};
          chosen_score = d;
          chosen_mobility = mobility;
        }
      }
    }

    if (!found) {
      // No legal moves: output something (required format) and stop.
      cout << 0 << " " << 1 << "\n";
      break;
    }

    // Apply chosen move.
    vector<int> new_head = add_step(head, chosen.axis, chosen.dir);
    bool ate = false;
    auto fit = food_count.find(new_head);
    if (fit != food_count.end()) {
      ate = true;
      if (--fit->second <= 0) food_count.erase(fit);
    }

    // Update body and occupancy.
    body.push_front(new_head);
    occ[new_head] += 1;
    head = new_head;

    if (!ate) {
      vector<int> old_tail = body.back();
      body.pop_back();
      auto oit = occ.find(old_tail);
      if (oit != occ.end()) {
        oit->second -= 1;
        if (oit->second <= 0) occ.erase(oit);
      }
    }

    cout << chosen.axis << " " << chosen.dir << "\n";
  }

  return 0;
}
