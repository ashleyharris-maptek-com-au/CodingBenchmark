use std::collections::HashMap;
use std::io::{self, Read};

struct Scanner {
    input: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { input, idx: 0 }
    }

    fn skip_ws(&mut self) {
        while self.idx < self.input.len() {
            let b = self.input[self.idx];
            if b == b' ' || b == b'\n' || b == b'\r' || b == b'\t' {
                self.idx += 1;
            } else {
                break;
            }
        }
    }

    fn next_i64(&mut self) -> Option<i64> {
        self.skip_ws();
        if self.idx >= self.input.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.input[self.idx] == b'-' {
            sign = -1;
            self.idx += 1;
        }
        let mut val = 0i64;
        while self.idx < self.input.len() {
            let b = self.input[self.idx];
            if (b'0'..=b'9').contains(&b) {
                val = val * 10 + (b - b'0') as i64;
                self.idx += 1;
            } else {
                break;
            }
        }
        Some(sign * val)
    }
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let (a, b) = nums.split_at(mid);

    let mut sums_a: Vec<(i64, u64)> = Vec::with_capacity(1usize << a.len());
    sums_a.push((0, 0));
    for (i, &v) in a.iter().enumerate() {
        let bit = 1u64 << i;
        let current_len = sums_a.len();
        for j in 0..current_len {
            let (s, m) = sums_a[j];
            sums_a.push((s + v, m | bit));
        }
    }

    let mut sums_b: Vec<(i64, u64)> = Vec::with_capacity(1usize << b.len());
    sums_b.push((0, 0));
    for (i, &v) in b.iter().enumerate() {
        let bit = 1u64 << i;
        let current_len = sums_b.len();
        for j in 0..current_len {
            let (s, m) = sums_b[j];
            sums_b.push((s + v, m | bit));
        }
    }

    sums_b.sort_unstable_by(|x, y| x.0.cmp(&y.0));
    let sums_b_vals: Vec<i64> = sums_b.iter().map(|x| x.0).collect();

    for (sa, ma) in sums_a {
        let need = target - sa;
        match sums_b_vals.binary_search_by(|&x| x.cmp(&need)) {
            Ok(pos) => {
                let mb = sums_b[pos].1;
                let mut out = Vec::new();
                for i in 0..a.len() {
                    if ((ma >> i) & 1) == 1 {
                        out.push(i);
                    }
                }
                for i in 0..b.len() {
                    if ((mb >> i) & 1) == 1 {
                        out.push(mid + i);
                    }
                }
                return Some(out);
            }
            Err(_) => {}
        }
    }
    None
}

fn dp_offset_small_range(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut min_sum = 0i64;
    let mut max_sum = 0i64;
    for &v in nums {
        if v >= 0 {
            max_sum = max_sum.saturating_add(v);
        } else {
            min_sum = min_sum.saturating_add(v);
        }
    }

    let range = max_sum - min_sum;
    if range < 0 {
        return None;
    }

    let range_usize = match usize::try_from(range) {
        Ok(x) => x,
        Err(_) => return None,
    };
    if range_usize > 2_000_000 {
        return None;
    }

    let offset = -min_sum;
    let size = range_usize + 1;
    let mut reachable = vec![false; size];
    let mut parent_idx = vec![-1i32; size];
    let mut parent_sum = vec![0i32; size];

    let start = offset;
    let start_u = usize::try_from(start).ok()?;
    reachable[start_u] = true;

    for (i, &v) in nums.iter().enumerate() {
        let v_i32 = i32::try_from(i).ok()?;
        let delta = v;
        if delta == 0 {
            continue;
        }

        if delta > 0 {
            let d = usize::try_from(delta).ok()?;
            for s in (0..size - d).rev() {
                if reachable[s] && !reachable[s + d] {
                    reachable[s + d] = true;
                    parent_idx[s + d] = v_i32;
                    parent_sum[s + d] = s as i32;
                }
            }
        } else {
            let d = usize::try_from(-delta).ok()?;
            for s in d..size {
                if reachable[s] && !reachable[s - d] {
                    reachable[s - d] = true;
                    parent_idx[s - d] = v_i32;
                    parent_sum[s - d] = s as i32;
                }
            }
        }
    }

    let target_pos = target + offset;
    if target_pos < 0 {
        return None;
    }
    let t = usize::try_from(target_pos).ok()?;
    if t >= size || !reachable[t] {
        return None;
    }

    let mut out = Vec::new();
    let mut cur = t as i32;
    while cur != start_u as i32 {
        let pi = parent_idx[cur as usize];
        if pi < 0 {
            break;
        }
        out.push(pi as usize);
        cur = parent_sum[cur as usize];
    }
    out.reverse();
    Some(out)
}

fn dp_hashmap_bounded(nums: &[i64], target: i64, cap: usize) -> Option<Vec<usize>> {
    let mut map: HashMap<i64, (i64, i32)> = HashMap::new();
    map.insert(0, (0, -1));

    for (i, &v) in nums.iter().enumerate() {
        let i32i = i32::try_from(i).ok()?;
        let mut new_entries: Vec<(i64, (i64, i32))> = Vec::new();
        new_entries.reserve(map.len().min(4096));

        for (&s, &(_ps, _)) in map.iter() {
            let ns = s.saturating_add(v);
            if !map.contains_key(&ns) {
                new_entries.push((ns, (s, i32i)));
            }
        }

        for (ns, back) in new_entries {
            map.insert(ns, back);
        }

        if map.contains_key(&target) {
            break;
        }
        if map.len() > cap {
            // Keep exactness best-effort: shrink by keeping sums closest to target.
            let mut keys: Vec<i64> = map.keys().copied().collect();
            keys.sort_unstable_by(|a, b| {
                let da = (target - *a).abs();
                let db = (target - *b).abs();
                da.cmp(&db)
            });
            keys.truncate(cap);
            let mut new_map: HashMap<i64, (i64, i32)> = HashMap::with_capacity(keys.len() * 2);
            for k in keys {
                if let Some(vv) = map.get(&k) {
                    new_map.insert(k, *vv);
                }
            }
            map = new_map;
            map.entry(0).or_insert((0, -1));
        }
    }

    if !map.contains_key(&target) {
        return None;
    }

    let mut out = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let (prev, idx) = *map.get(&cur)?;
        if idx < 0 {
            break;
        }
        out.push(idx as usize);
        cur = prev;
    }
    out.reverse();
    Some(out)
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_i64() {
        Some(v) => v as usize,
        None => return,
    };
    let target = match sc.next_i64() {
        Some(v) => v,
        None => return,
    };

    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(v) = sc.next_i64() {
            nums.push(v);
        } else {
            break;
        }
    }

    // Quick exact hits.
    if target == 0 {
        print!("YES\n[]\n");
        return;
    }
    for (i, &v) in nums.iter().enumerate() {
        if v == target {
            print!("YES\n[{}]\n", i);
            return;
        }
    }

    let mut all_nonneg = true;
    for &v in &nums {
        if v < 0 {
            all_nonneg = false;
            break;
        }
    }

    let ans = if n <= 44 {
        meet_in_middle(&nums, target)
    } else {
        // Exact DP when total reachable range is small (handles negatives too).
        let exact_small = dp_offset_small_range(&nums, target);
        if exact_small.is_some() {
            exact_small
        } else if all_nonneg && target > 0 {
            // If target is small-ish, do classic parent DP (exact, reconstructable).
            if target <= 500_000 {
                let t = target as usize;
                let mut reachable = vec![false; t + 1];
                let mut parent_idx = vec![-1i32; t + 1];
                let mut parent_sum = vec![0i32; t + 1];
                reachable[0] = true;

                for (i, &v) in nums.iter().enumerate() {
                    if v <= 0 {
                        continue;
                    }
                    if v > target {
                        continue;
                    }
                    let vv = v as usize;
                    let i32i = i32::try_from(i).ok();
                    if i32i.is_none() {
                        break;
                    }
                    let i32i = i32i.unwrap();
                    for s in (0..=t - vv).rev() {
                        if reachable[s] && !reachable[s + vv] {
                            reachable[s + vv] = true;
                            parent_idx[s + vv] = i32i;
                            parent_sum[s + vv] = s as i32;
                        }
                    }
                    if reachable[t] {
                        break;
                    }
                }

                if !reachable[t] {
                    None
                } else {
                    let mut out = Vec::new();
                    let mut cur = t as i32;
                    while cur != 0 {
                        let pi = parent_idx[cur as usize];
                        if pi < 0 {
                            break;
                        }
                        out.push(pi as usize);
                        cur = parent_sum[cur as usize];
                    }
                    out.reverse();
                    Some(out)
                }
            } else {
                // Best-effort: bounded hashmap DP with reconstruction. May miss solutions for very hard cases.
                dp_hashmap_bounded(&nums, target, 2_000_000)
            }
        } else {
            // General case with large magnitudes: bounded hashmap DP.
            dp_hashmap_bounded(&nums, target, 2_000_000)
        }
    };

    if let Some(idxs) = ans {
        // Verify.
        let mut sum = 0i64;
        for &i in &idxs {
            if i >= nums.len() {
                println!("NO");
                return;
            }
            sum = sum.saturating_add(nums[i]);
        }
        if sum != target {
            println!("NO");
            return;
        }
        print!("YES\n[");
        for (k, i) in idxs.iter().enumerate() {
            if k != 0 {
                print!(" ");
            }
            print!("{}", i);
        }
        print!("]\n");
    } else {
        println!("NO");
    }
}
