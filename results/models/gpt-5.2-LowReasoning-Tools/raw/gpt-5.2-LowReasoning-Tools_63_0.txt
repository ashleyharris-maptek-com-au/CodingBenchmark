// Node.js TypeScript program: warehouse job assignment to robots.
// Reads one JSON document from stdin and prints {"assignments": [...]} to stdout.

declare const require: any;

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

const grid: string[] = data.grid;
const robots: Robot[] = data.robots;
const jobs: Job[] = data.jobs;

const H = grid.length | 0;
const W = (H > 0 ? grid[0].length : 0) | 0;

function idx(x: number, y: number): number {
  return (y * W + x) | 0;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y].charCodeAt(x) === 46; // '.'
}

// Determine output array size: index is robot id.
let maxRobotId = -1;
for (let i = 0; i < robots.length; i++) {
  const rid = robots[i].id | 0;
  if (rid > maxRobotId) maxRobotId = rid;
}
const assignments: number[][] = new Array(maxRobotId + 1);
for (let i = 0; i < assignments.length; i++) assignments[i] = [];

// Multi-source BFS from robot start positions to assign each open cell to its nearest robot.
const N = (W * H) | 0;
const dist = new Int32Array(N);
const owner = new Int32Array(N);
dist.fill(-1);
owner.fill(-1);

// Queue for BFS (ring buffer over arrays).
const qx = new Int32Array(N);
const qy = new Int32Array(N);
let qh = 0;
let qt = 0;

// Initialize sources in increasing robot id order for deterministic tie-breaking.
const robotOrder = robots.slice().sort((a, b) => (a.id - b.id) || (a.y - b.y) || (a.x - b.x));
for (let i = 0; i < robotOrder.length; i++) {
  const r = robotOrder[i];
  const x = r.x | 0;
  const y = r.y | 0;
  if (!isOpen(x, y)) continue;
  const p = idx(x, y);
  if (dist[p] !== -1) continue; // two robots on same cell; keep lowest id due to sort
  dist[p] = 0;
  owner[p] = r.id | 0;
  qx[qt] = x;
  qy[qt] = y;
  qt++;
}

const dx = [1, -1, 0, 0];
const dy = [0, 0, 1, -1];

while (qh < qt) {
  const x = qx[qh];
  const y = qy[qh];
  qh++;
  const p = idx(x, y);
  const d = dist[p] | 0;
  const o = owner[p] | 0;
  for (let k = 0; k < 4; k++) {
    const nx = (x + dx[k]) | 0;
    const ny = (y + dy[k]) | 0;
    if (!isOpen(nx, ny)) continue;
    const np = idx(nx, ny);
    if (dist[np] !== -1) continue;
    dist[np] = (d + 1) | 0;
    owner[np] = o;
    qx[qt] = nx;
    qy[qt] = ny;
    qt++;
  }
}

// Group jobs by owning robot.
const jobsByRobot: Map<number, Job[]> = new Map();
for (let i = 0; i < jobs.length; i++) {
  const j = jobs[i];
  let rid = 0;
  const x = j.x | 0;
  const y = j.y | 0;
  if (isOpen(x, y)) {
    const o = owner[idx(x, y)];
    if (o !== -1) rid = o | 0;
    else rid = (robotOrder.length ? robotOrder[0].id : 0) | 0;
  } else {
    rid = (robotOrder.length ? robotOrder[0].id : 0) | 0;
  }
  let arr = jobsByRobot.get(rid);
  if (arr === undefined) {
    arr = [];
    jobsByRobot.set(rid, arr);
  }
  arr.push(j);
}

// For stable ordering, compute each job's distance to its assigned robot start using dist[] from BFS.
// If unreachable/invalid, treat as very large.
const INF = 0x3fffffff;
const robotStart: Map<number, { x: number; y: number }> = new Map();
for (let i = 0; i < robots.length; i++) robotStart.set(robots[i].id | 0, { x: robots[i].x | 0, y: robots[i].y | 0 });

for (const [rid, list] of jobsByRobot.entries()) {
  // Sort by higher priority first, then closer to robot start, then job id.
  const rs = robotStart.get(rid);
  const baseX = rs ? rs.x : 0;
  const baseY = rs ? rs.y : 0;
  // Note: dist[] is distance to nearest robot, not necessarily this robot if unreachable in BFS init.
  // We'll approximate with dist[] at job cell when owner matches; otherwise fallback to manhattan.
  list.sort((a, b) => {
    const pa = (a.priority as number) | 0;
    const pb = (b.priority as number) | 0;
    if (pa !== pb) return pb - pa;
    let da = INF;
    let db = INF;
    const ax = a.x | 0, ay = a.y | 0;
    const bx = b.x | 0, by = b.y | 0;
    if (isOpen(ax, ay) && owner[idx(ax, ay)] === (rid | 0)) da = dist[idx(ax, ay)] | 0;
    else da = (Math.abs(ax - baseX) + Math.abs(ay - baseY)) | 0;
    if (isOpen(bx, by) && owner[idx(bx, by)] === (rid | 0)) db = dist[idx(bx, by)] | 0;
    else db = (Math.abs(bx - baseX) + Math.abs(by - baseY)) | 0;
    if (da !== db) return da - db;
    return ((a.id as number) | 0) - ((b.id as number) | 0);
  });
}

// Emit assignments: array index is robot id; ensure every job appears exactly once.
for (const [rid, list] of jobsByRobot.entries()) {
  if (rid < 0 || rid >= assignments.length) continue;
  const out = assignments[rid];
  for (let i = 0; i < list.length; i++) out.push(list[i].id);
}

process.stdout.write(JSON.stringify({ assignments }));
