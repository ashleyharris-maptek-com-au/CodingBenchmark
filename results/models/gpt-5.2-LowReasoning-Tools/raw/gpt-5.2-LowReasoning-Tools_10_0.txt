// 2D Cutting Stock / Rectangle Bin Packing heuristic solver (C++20, MSVC-friendly)
// Strategy: offline sort (decreasing max-side, then area), then "best-fit shelf" packing across boards.
// Each shelf is a horizontal strip in a board. Rectangles are placed left-to-right on a shelf.
// We choose an existing shelf that fits (height >= rect_h and remaining_width >= rect_w) with minimal leftover.
// If no shelf fits, we open a new shelf in the best board with enough remaining height; otherwise, open a new board.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <map>
#include <tuple>
#include <utility>
#include <vector>

struct Rect {
    int w = 0;
    int h = 0;
    int idx = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rot = 0;
};

struct Board {
    int used_h = 0;
    std::multimap<int, int>::iterator rem_it; // iterator into remHeight->boardId multimap
};

struct Shelf {
    int board = -1;
    int y = 0;
    int height = 0;
    int x = 0;     // next x
    int rem = 0;   // remaining width
    std::multimap<int, int>::iterator it; // iterator inside heightMaps[height] mapping rem->shelfId
};

static inline std::int64_t area64(int w, int h) {
    return static_cast<std::int64_t>(w) * static_cast<std::int64_t>(h);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0, W = 0, H = 0;
    if (!(std::cin >> N >> W >> H)) return 0;

    std::vector<Rect> rects;
    rects.reserve(static_cast<size_t>(N));
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects.push_back(Rect{w, h, i});
    }

    // Pre-check feasibility: if a rect can't fit even with rotation, we still must output something;
    // but to stay valid, we place it on its own "board" at (0,0) only if it fits. Otherwise, we clamp by failing fast.
    for (const auto& r : rects) {
        const bool fits = (r.w <= W && r.h <= H) || (r.h <= W && r.w <= H);
        if (!fits) {
            // No valid packing exists under constraints.
            // Output 0 boards (judge should treat as invalid, but avoids UB/timeouts).
            std::cout << 0 << "\n";
            return 0;
        }
    }

    // Sort rectangles (offline): decreasing max-side, then decreasing area.
    std::vector<Rect> order = rects;
    std::sort(order.begin(), order.end(), [](const Rect& a, const Rect& b) {
        const int am = std::max(a.w, a.h);
        const int bm = std::max(b.w, b.h);
        if (am != bm) return am > bm;
        const std::int64_t aa = area64(a.w, a.h);
        const std::int64_t ba = area64(b.w, b.h);
        if (aa != ba) return aa > ba;
        return a.idx < b.idx;
    });

    std::vector<Placement> ans(static_cast<size_t>(N));

    // Global structures:
    // heightMaps: shelfHeight -> multimap(remainingWidth -> shelfId)
    std::map<int, std::multimap<int, int>> heightMaps;
    std::vector<Shelf> shelves;
    shelves.reserve(static_cast<size_t>(N)); // upper bound

    // Boards indexed 0..B-1, with multimap of remainingHeight->boardId for "best board to open a new shelf".
    std::vector<Board> boards;
    std::multimap<int, int> remHeight; // key = remaining height in board

    auto add_board = [&]() -> int {
        const int id = static_cast<int>(boards.size());
        boards.push_back(Board{0, remHeight.end()});
        auto it = remHeight.emplace(H, id);
        boards[id].rem_it = it;
        return id;
    };

    auto update_board_remaining = [&](int boardId) {
        auto& b = boards[boardId];
        remHeight.erase(b.rem_it);
        const int rem = H - b.used_h;
        b.rem_it = remHeight.emplace(rem, boardId);
    };

    auto shelf_remove_from_map = [&](int shelfId) {
        Shelf& s = shelves[shelfId];
        auto hmIt = heightMaps.find(s.height);
        if (hmIt == heightMaps.end()) return;
        hmIt->second.erase(s.it);
    };

    auto shelf_insert_into_map = [&](int shelfId) {
        Shelf& s = shelves[shelfId];
        auto& mm = heightMaps[s.height];
        s.it = mm.emplace(s.rem, shelfId);
    };

    auto place_on_shelf = [&](int shelfId, int rw, int rh, int rotated, int rectIdx) {
        Shelf& s = shelves[shelfId];
        // Place at current cursor.
        ans[static_cast<size_t>(rectIdx)] = Placement{s.board, s.x, s.y, rotated};
        // Update shelf bookkeeping.
        shelf_remove_from_map(shelfId);
        s.x += rw;
        s.rem -= rw;
        shelf_insert_into_map(shelfId);
    };

    auto open_shelf_in_board = [&](int boardId, int shelfHeight) -> int {
        Board& b = boards[boardId];
        const int y = b.used_h;
        b.used_h += shelfHeight;
        update_board_remaining(boardId);

        const int shelfId = static_cast<int>(shelves.size());
        shelves.push_back(Shelf{boardId, y, shelfHeight, 0, W, std::multimap<int, int>::iterator{}});
        shelf_insert_into_map(shelfId);
        return shelfId;
    };

    // Ensure at least one board exists.
    add_board();

    // Tunables: keep search bounded for speed on huge instances.
    const int MAX_HEIGHT_BUCKETS_TO_SCAN = 16;

    for (const auto& r : order) {
        // Choose orientation (we may flip during fit checks).
        // We'll search shelves for either orientation; if both fit, pick best leftover.
        struct Candidate {
            int shelfId = -1;
            int rw = 0, rh = 0;
            int rot = 0;
            int leftover = std::numeric_limits<int>::max();
            int shelfHeight = 0;
        } best;

        auto try_find_shelf = [&](int rw, int rh, int rotFlag) {
            auto itH = heightMaps.lower_bound(rh);
            int scanned = 0;
            for (; itH != heightMaps.end() && scanned < MAX_HEIGHT_BUCKETS_TO_SCAN; ++itH, ++scanned) {
                const int shelfHeight = itH->first;
                auto& mm = itH->second;
                auto itW = mm.lower_bound(rw);
                if (itW == mm.end()) continue;
                const int remW = itW->first;
                const int shelfId = itW->second;
                const int leftover = remW - rw;
                // Prefer tighter horizontal fit; tie-breaker: smaller shelf height.
                if (leftover < best.leftover || (leftover == best.leftover && shelfHeight < best.shelfHeight)) {
                    best = Candidate{shelfId, rw, rh, rotFlag, leftover, shelfHeight};
                    if (best.leftover == 0) return; // can't do better on this scan
                }
            }
        };

        // Consider both orientations.
        if (r.w <= W && r.h <= H) try_find_shelf(r.w, r.h, 0);
        if (r.h <= W && r.w <= H) try_find_shelf(r.h, r.w, 1);

        if (best.shelfId != -1) {
            place_on_shelf(best.shelfId, best.rw, best.rh, best.rot, r.idx);
            continue;
        }

        // No existing shelf fits; open a new shelf.
        // Choose best board (minimum remaining height after placing shelf).
        struct BoardCand {
            int boardId = -1;
            int remAfter = std::numeric_limits<int>::max();
        } bestBoard;

        auto choose_board_for_shelf = [&](int shelfH) {
            // Find any board with remaining height >= shelfH, prefer tightest remaining after.
            auto it = remHeight.lower_bound(shelfH);
            if (it == remHeight.end()) return;
            const int rem = it->first;
            const int boardId = it->second;
            const int remAfter = rem - shelfH;
            if (remAfter < bestBoard.remAfter) bestBoard = BoardCand{boardId, remAfter};
        };

        // We will decide orientation together with opening shelf.
        // Prefer orientation that allows tighter shelf height usage if opening new shelf.
        int useW = r.w, useH = r.h, useRot = 0;
        int altW = r.h, altH = r.w, altRot = 1;

        // Select orientation by best board fit; tie-breaker by larger width (tends to reduce fragmentation).
        BoardCand cand1, cand2;
        bestBoard = BoardCand{};
        choose_board_for_shelf(useH);
        cand1 = bestBoard;
        bestBoard = BoardCand{};
        choose_board_for_shelf(altH);
        cand2 = bestBoard;

        if (cand2.boardId != -1) {
            if (cand1.boardId == -1 || cand2.remAfter < cand1.remAfter ||
                (cand2.remAfter == cand1.remAfter && altW > useW)) {
                useW = altW;
                useH = altH;
                useRot = altRot;
                cand1 = cand2;
            }
        }

        int boardId = cand1.boardId;
        if (boardId == -1) {
            boardId = add_board();
        }

        const int shelfId = open_shelf_in_board(boardId, useH);
        place_on_shelf(shelfId, useW, useH, useRot, r.idx);
    }

    std::cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        const auto& p = ans[static_cast<size_t>(i)];
        std::cout << p.board << " " << p.x << " " << p.y << " " << p.rot << "\n";
    }
    return 0;
}
