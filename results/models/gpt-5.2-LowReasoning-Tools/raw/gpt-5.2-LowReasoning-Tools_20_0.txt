use std::cmp::Ordering;
use std::collections::{BinaryHeap, HashMap};
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole: usize,
    idx: usize,
    x: f32,
    y: f32,
    z: f32,
    vals: Vec<Option<f32>>,
}

#[derive(Copy, Clone)]
struct BBox {
    min_x: f32,
    min_y: f32,
    min_z: f32,
    max_x: f32,
    max_y: f32,
    max_z: f32,
}

impl BBox {
    fn new() -> Self {
        Self {
            min_x: f32::INFINITY,
            min_y: f32::INFINITY,
            min_z: f32::INFINITY,
            max_x: f32::NEG_INFINITY,
            max_y: f32::NEG_INFINITY,
            max_z: f32::NEG_INFINITY,
        }
    }
    fn update(&mut self, x: f32, y: f32, z: f32) {
        self.min_x = self.min_x.min(x);
        self.min_y = self.min_y.min(y);
        self.min_z = self.min_z.min(z);
        self.max_x = self.max_x.max(x);
        self.max_y = self.max_y.max(y);
        self.max_z = self.max_z.max(z);
    }
    fn volume(&self) -> f64 {
        let dx = (self.max_x - self.min_x).max(1e-3) as f64;
        let dy = (self.max_y - self.min_y).max(1e-3) as f64;
        let dz = (self.max_z - self.min_z).max(1e-3) as f64;
        dx * dy * dz
    }
}

struct FastInput {
    buf: Vec<u8>,
    i: usize,
}

impl FastInput {
    fn new() -> io::Result<Self> {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf)?;
        Ok(Self { buf, i: 0 })
    }
    fn skip_ws(&mut self) {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
    }
    fn next_token(&mut self) -> Option<&[u8]> {
        self.skip_ws();
        if self.i >= self.buf.len() {
            return None;
        }
        let start = self.i;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        Some(&self.buf[start..self.i])
    }
    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token()?;
        std::str::from_utf8(t).ok()?.parse::<usize>().ok()
    }
    fn next_f32(&mut self) -> Option<f32> {
        let t = self.next_token()?;
        std::str::from_utf8(t).ok()?.parse::<f32>().ok()
    }
    fn next_string(&mut self) -> Option<String> {
        let t = self.next_token()?;
        Some(String::from_utf8_lossy(t).to_string())
    }
    fn next_opt_f32(&mut self) -> Option<Option<f32>> {
        let t = self.next_token()?;
        let s = std::str::from_utf8(t).ok()?;
        if s.eq_ignore_ascii_case("na") || s.eq_ignore_ascii_case("nan") || s == "." {
            return Some(None);
        }
        match s.parse::<f32>() {
            Ok(v) if v.is_finite() => Some(Some(v)),
            _ => Some(None),
        }
    }
}

fn norm3(x: f32, y: f32, z: f32) -> f32 {
    (x * x + y * y + z * z).sqrt()
}

fn clamp_f64(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn median_and_mad(values: &mut [f32]) -> (f32, f32) {
    values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = values.len();
    let med = if n == 0 {
        0.0
    } else if n % 2 == 1 {
        values[n / 2]
    } else {
        0.5 * (values[n / 2 - 1] + values[n / 2])
    };
    for v in values.iter_mut() {
        *v = (*v - med).abs();
    }
    values.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = if n == 0 {
        0.0
    } else if n % 2 == 1 {
        values[n / 2]
    } else {
        0.5 * (values[n / 2 - 1] + values[n / 2])
    };
    (med, mad)
}

#[derive(Clone)]
struct Suspect {
    conf: f32,
    hole: usize,
    idx: usize,
    prop: usize,
}

fn main() -> io::Result<()> {
    let mut input = FastInput::new()?;
    let num_holes = match input.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };
    let num_props = match input.next_usize() {
        Some(v) => v,
        None => return Ok(()),
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(input.next_string().unwrap_or_else(|| "prop".to_string()));
    }

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut hole_map: HashMap<String, usize> = HashMap::with_capacity(num_holes * 2 + 1);
    let mut samples: Vec<Sample> = Vec::new();
    let mut bbox = BBox::new();

    for _ in 0..num_holes {
        let hole_id = match input.next_string() {
            Some(s) => s,
            None => break,
        };
        let start_x = match input.next_f32() {
            Some(v) => v,
            None => break,
        };
        let start_y = input.next_f32().unwrap_or(0.0);
        let start_z = input.next_f32().unwrap_or(0.0);
        let dir_x = input.next_f32().unwrap_or(0.0);
        let dir_y = input.next_f32().unwrap_or(0.0);
        let dir_z = input.next_f32().unwrap_or(0.0);
        let _length = input.next_f32().unwrap_or(0.0);
        let num_samples = input.next_usize().unwrap_or(0);

        let hole_idx = if let Some(&idx) = hole_map.get(&hole_id) {
            idx
        } else {
            let idx = hole_ids.len();
            hole_ids.push(hole_id.clone());
            hole_map.insert(hole_id.clone(), idx);
            idx
        };

        let nrm = norm3(dir_x, dir_y, dir_z);
        let (ux, uy, uz) = if nrm > 0.0 {
            (dir_x / nrm, dir_y / nrm, dir_z / nrm)
        } else {
            (0.0, 0.0, 1.0)
        };

        for si in 0..num_samples {
            let depth = match input.next_f32() {
                Some(v) => v,
                None => break,
            };
            let mut vals = Vec::with_capacity(num_props);
            for _ in 0..num_props {
                vals.push(input.next_opt_f32().unwrap_or(None));
            }
            let x = start_x + ux * depth;
            let y = start_y + uy * depth;
            let z = start_z + uz * depth;
            bbox.update(x, y, z);
            samples.push(Sample {
                hole: hole_idx,
                idx: si,
                x,
                y,
                z,
                vals,
            });
        }
    }

    let n_points = samples.len();
    if n_points == 0 || num_props == 0 {
        println!("0");
        return Ok(());
    }

    // Adaptive neighborhood radius from data extent and density.
    let volume = bbox.volume();
    let density_len = (volume / (n_points as f64)).cbrt();
    let mut radius = density_len * 2.2;
    if n_points < 2_000 {
        radius *= 3.0;
    } else if n_points > 500_000 {
        radius *= 1.2;
    }
    radius = clamp_f64(radius, 0.5, 500.0);
    let base_radius = radius as f32;

    // Strategy switch: very small datasets brute-force, otherwise spatial hash grid.
    let use_bruteforce = n_points <= 3_000;

    // Build spatial index.
    let cell_size = base_radius.max(1e-3);
    let inv_cell = 1.0f32 / cell_size;
    let mut grid: HashMap<(i32, i32, i32), Vec<usize>> = HashMap::new();
    if !use_bruteforce {
        grid.reserve(n_points * 2);
        for (i, s) in samples.iter().enumerate() {
            let cx = (s.x * inv_cell).floor() as i32;
            let cy = (s.y * inv_cell).floor() as i32;
            let cz = (s.z * inv_cell).floor() as i32;
            grid.entry((cx, cy, cz)).or_default().push(i);
        }
    }

    // Suspect selection controls.
    let max_report = if n_points < 50_000 {
        5_000
    } else if n_points < 300_000 {
        2_500
    } else {
        1_500
    };
    let min_neighbors = if n_points < 10_000 { 8 } else { 12 };
    let max_neighbors = if n_points < 100_000 { 80 } else { 48 };

    // Thresholds: tuned to report "obvious" outliers.
    let z_threshold = if n_points < 5_000 { 4.0 } else { 5.0 };

    // Keep top suspects by confidence.
    // We maintain a min-heap by storing negative confidence in a max-heap.
    #[derive(Clone)]
    struct HeapItem {
        neg_conf: f32,
        s: Suspect,
    }
    impl PartialEq for HeapItem {
        fn eq(&self, other: &Self) -> bool {
            self.neg_conf == other.neg_conf
        }
    }
    impl Eq for HeapItem {}
    impl PartialOrd for HeapItem {
        fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
            self.neg_conf.partial_cmp(&other.neg_conf)
        }
    }
    impl Ord for HeapItem {
        fn cmp(&self, other: &Self) -> Ordering {
            self.partial_cmp(other).unwrap_or(Ordering::Equal)
        }
    }
    let mut heap: BinaryHeap<HeapItem> = BinaryHeap::new();

    let mut neighbor_buf: Vec<usize> = Vec::with_capacity(256);
    let mut dist_buf: Vec<(f32, usize)> = Vec::with_capacity(256);
    let mut val_buf: Vec<f32> = Vec::with_capacity(128);
    let mut tmp_vals: Vec<f32> = Vec::with_capacity(128);

    for i in 0..n_points {
        let sx = samples[i].x;
        let sy = samples[i].y;
        let sz = samples[i].z;

        neighbor_buf.clear();
        dist_buf.clear();

        let mut r = base_radius;
        let mut found = 0usize;
        for _pass in 0..3 {
            if use_bruteforce {
                for j in 0..n_points {
                    if j == i {
                        continue;
                    }
                    let dx = samples[j].x - sx;
                    let dy = samples[j].y - sy;
                    let dz = samples[j].z - sz;
                    let d2 = dx * dx + dy * dy + dz * dz;
                    if d2 <= r * r {
                        dist_buf.push((d2, j));
                    }
                }
            } else {
                let cx = (sx * inv_cell).floor() as i32;
                let cy = (sy * inv_cell).floor() as i32;
                let cz = (sz * inv_cell).floor() as i32;
                let cell_range = ((r / cell_size).ceil() as i32).max(1);
                for dx_cell in -cell_range..=cell_range {
                    for dy_cell in -cell_range..=cell_range {
                        for dz_cell in -cell_range..=cell_range {
                            if let Some(list) = grid.get(&(cx + dx_cell, cy + dy_cell, cz + dz_cell))
                            {
                                for &j in list {
                                    if j == i {
                                        continue;
                                    }
                                    let dx = samples[j].x - sx;
                                    let dy = samples[j].y - sy;
                                    let dz = samples[j].z - sz;
                                    let d2 = dx * dx + dy * dy + dz * dz;
                                    if d2 <= r * r {
                                        dist_buf.push((d2, j));
                                    }
                                }
                            }
                        }
                    }
                }
            }

            if dist_buf.len() >= min_neighbors {
                found = dist_buf.len();
                break;
            }
            dist_buf.clear();
            r *= 2.0;
        }
        if found < min_neighbors {
            continue;
        }

        dist_buf.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
        let take_n = dist_buf.len().min(max_neighbors);
        neighbor_buf.extend(dist_buf.iter().take(take_n).map(|&(_, j)| j));

        for p in 0..num_props {
            let Some(v0) = samples[i].vals[p] else { continue };

            val_buf.clear();
            for &j in neighbor_buf.iter() {
                if let Some(vn) = samples[j].vals[p] {
                    if vn.is_finite() {
                        val_buf.push(vn);
                    }
                }
            }
            if val_buf.len() < min_neighbors {
                continue;
            }

            tmp_vals.clear();
            tmp_vals.extend_from_slice(&val_buf);
            let (med, mad) = median_and_mad(&mut tmp_vals);
            let sigma = (1.4826 * mad).max(1e-6);
            let z = ((v0 - med).abs() / sigma) as f32;
            if z < z_threshold {
                continue;
            }

            let conf = (z - z_threshold + 1.0).min(1000.0);
            heap.push(HeapItem {
                neg_conf: -conf,
                s: Suspect {
                    conf,
                    hole: samples[i].hole,
                    idx: samples[i].idx,
                    prop: p,
                },
            });
            if heap.len() > max_report {
                heap.pop();
            }
        }
    }

    let mut suspects: Vec<Suspect> = heap.into_iter().map(|h| h.s).collect();
    suspects.sort_by(|a, b| b.conf.partial_cmp(&a.conf).unwrap_or(Ordering::Equal));

    println!("{}", suspects.len());
    for s in suspects {
        println!(
            "{} {} {} {:.6}",
            hole_ids[s.hole], s.idx, prop_names[s.prop], s.conf
        );
    }

    Ok(())
}
