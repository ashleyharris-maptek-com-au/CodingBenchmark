using System;
using System.Collections.Generic;

public static class Program
{
    private const byte Unknown = 255;
    private const byte FlaggedMine = 254;

    public static void Main()
    {
        var first = Console.ReadLine();
        if (string.IsNullOrEmpty(first))
        {
            Console.WriteLine(0);
            return;
        }

        var parts = first.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(parts[0]);
        int w = int.Parse(parts[1]);

        long cellCountLong = (long)h * w;
        if (cellCountLong > int.MaxValue)
        {
            // Input itself is already impractically large for this interface.
            Console.WriteLine(0);
            return;
        }
        int cellCount = (int)cellCountLong;

        var grid = new byte[cellCount]; // 0..8 numbers, Unknown, FlaggedMine
        var numberCells = new List<int>(Math.Min(cellCount, 1_000_000));

        for (int y = 0; y < h; y++)
        {
            string? line = Console.ReadLine();
            line ??= "";
            if (line.Length < w) line = line.PadRight(w, ' ');
            for (int x = 0; x < w; x++)
            {
                char c = line[x];
                int idx = y * w + x;
                if (c == ' ')
                {
                    grid[idx] = Unknown;
                }
                else if (c == '*')
                {
                    grid[idx] = FlaggedMine;
                }
                else if (c >= '0' && c <= '8')
                {
                    grid[idx] = (byte)(c - '0');
                    numberCells.Add(idx);
                }
                else
                {
                    // Treat any unexpected char as unknown.
                    grid[idx] = Unknown;
                }
            }
        }

        var safe = new bool[cellCount];
        var safeList = new List<int>(1024);

        // Phase 1: linear-time local constraint propagation (scales to huge boards).
        RunBasicPropagation(h, w, grid, numberCells, safe, safeList);

        // Phase 2: optional stronger reasoning on the "frontier" (unknowns adjacent to numbers).
        // Only run when the frontier is small enough to be worthwhile.
        bool allowFrontierCsp = cellCount <= 2_000_000 && numberCells.Count <= 400_000;
        if (allowFrontierCsp)
        {
            RunFrontierCsp(h, w, grid, numberCells, safe, safeList);
        }

        Console.WriteLine(safeList.Count);
        for (int i = 0; i < safeList.Count; i++)
        {
            int idx = safeList[i];
            int y = idx / w;
            int x = idx - y * w;
            Console.WriteLine($"{x} {y}");
        }
    }

    private static void RunBasicPropagation(
        int h, int w,
        byte[] grid,
        List<int> numberCells,
        bool[] safe,
        List<int> safeList)
    {
        var q = new Queue<int>(Math.Min(numberCells.Count, 1_000_000));
        var inQueue = new bool[grid.Length];
        for (int i = 0; i < numberCells.Count; i++)
        {
            int idx = numberCells[i];
            q.Enqueue(idx);
            inQueue[idx] = true;
        }

        while (q.Count > 0)
        {
            int idx = q.Dequeue();
            inQueue[idx] = false;
            byte cellVal = grid[idx];
            if (cellVal > 8) continue;

            int y = idx / w;
            int x = idx - y * w;

            int flagged = 0;
            int unknownCount = 0;
            Span<int> unknowns = stackalloc int[8];

            for (int dy = -1; dy <= 1; dy++)
            {
                int ny = y + dy;
                if ((uint)ny >= (uint)h) continue;
                int rowBase = ny * w;
                for (int dx = -1; dx <= 1; dx++)
                {
                    if (dx == 0 && dy == 0) continue;
                    int nx = x + dx;
                    if ((uint)nx >= (uint)w) continue;
                    int nidx = rowBase + nx;
                    byte v = grid[nidx];
                    if (v == FlaggedMine) flagged++;
                    else if (v == Unknown)
                    {
                        unknowns[unknownCount++] = nidx;
                    }
                }
            }

            int needed = cellVal - flagged;
            if (needed < 0) continue; // inconsistent input; ignore
            if (unknownCount == 0) continue;

            if (needed == 0)
            {
                for (int i = 0; i < unknownCount; i++)
                {
                    int u = unknowns[i];
                    if (!safe[u])
                    {
                        safe[u] = true;
                        safeList.Add(u);
                    }
                }
            }
            else if (needed == unknownCount)
            {
                // All unknown neighbors must be mines.
                for (int i = 0; i < unknownCount; i++)
                {
                    int u = unknowns[i];
                    if (grid[u] != Unknown) continue;
                    grid[u] = FlaggedMine;
                    EnqueueAdjacentNumbers(h, w, grid, u, w, q, inQueue);
                }
            }
        }
    }

    private static void EnqueueAdjacentNumbers(
        int h, int w,
        byte[] grid,
        int idx,
        int width,
        Queue<int> q,
        bool[] inQueue)
    {
        int y = idx / width;
        int x = idx - y * width;

        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)h) continue;
            int rowBase = ny * width;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)w) continue;
                int nidx = rowBase + nx;
                if (grid[nidx] <= 8 && !inQueue[nidx])
                {
                    inQueue[nidx] = true;
                    q.Enqueue(nidx);
                }
            }
        }
    }

    private readonly struct Constraint
    {
        public readonly int[] Vars;
        public readonly int MineCount;
        public Constraint(int[] vars, int mineCount)
        {
            Vars = vars;
            MineCount = mineCount;
        }
    }

    private static void RunFrontierCsp(
        int h, int w,
        byte[] grid,
        List<int> numberCells,
        bool[] safe,
        List<int> safeList)
    {
        // Build frontier variables: unknown cells adjacent to at least one numbered cell.
        var varIdByCell = new Dictionary<int, int>(capacity: Math.Min(200_000, grid.Length / 10));
        var cellByVarId = new List<int>(capacity: Math.Min(200_000, grid.Length / 10));
        var constraints = new List<Constraint>(capacity: Math.Min(200_000, numberCells.Count));

        for (int i = 0; i < numberCells.Count; i++)
        {
            int idx = numberCells[i];
            byte cellVal = grid[idx];
            if (cellVal > 8) continue;

            int y = idx / w;
            int x = idx - y * w;
            int flagged = 0;

            Span<int> localVars = stackalloc int[8];
            int localCount = 0;

            for (int dy = -1; dy <= 1; dy++)
            {
                int ny = y + dy;
                if ((uint)ny >= (uint)h) continue;
                int rowBase = ny * w;
                for (int dx = -1; dx <= 1; dx++)
                {
                    if (dx == 0 && dy == 0) continue;
                    int nx = x + dx;
                    if ((uint)nx >= (uint)w) continue;
                    int nidx = rowBase + nx;
                    byte v = grid[nidx];
                    if (v == FlaggedMine) { flagged++; continue; }
                    if (v != Unknown) continue;
                    if (safe[nidx]) continue; // already proven safe; doesn't participate

                    if (!varIdByCell.TryGetValue(nidx, out int vid))
                    {
                        vid = cellByVarId.Count;
                        varIdByCell[nidx] = vid;
                        cellByVarId.Add(nidx);
                    }
                    localVars[localCount++] = vid;
                }
            }

            int needed = cellVal - flagged;
            if (needed < 0) continue;
            if (localCount == 0) continue;

            var vars = new int[localCount];
            for (int k = 0; k < localCount; k++) vars[k] = localVars[k];
            Array.Sort(vars);
            constraints.Add(new Constraint(vars, needed));
        }

        int varCount = cellByVarId.Count;
        if (varCount == 0 || constraints.Count == 0) return;

        // If frontier is large, don't attempt CSP.
        if (varCount > 200_000 || constraints.Count > 400_000) return;

        // Build adjacency var<->constraints
        var consByVar = new List<int>[varCount];
        for (int v = 0; v < varCount; v++) consByVar[v] = new List<int>(4);
        for (int c = 0; c < constraints.Count; c++)
        {
            var vars = constraints[c].Vars;
            for (int i = 0; i < vars.Length; i++) consByVar[vars[i]].Add(c);
        }

        // Find connected components on bipartite graph via vars.
        var visitedVar = new bool[varCount];
        var visitedCons = new bool[constraints.Count];

        var varQueue = new Queue<int>();
        var consQueue = new Queue<int>();

        for (int startVar = 0; startVar < varCount; startVar++)
        {
            if (visitedVar[startVar]) continue;
            if (consByVar[startVar].Count == 0) { visitedVar[startVar] = true; continue; }

            var componentVars = new List<int>(32);
            var componentCons = new List<int>(32);

            visitedVar[startVar] = true;
            varQueue.Enqueue(startVar);

            while (varQueue.Count > 0)
            {
                int v = varQueue.Dequeue();
                componentVars.Add(v);
                foreach (int c in consByVar[v])
                {
                    if (visitedCons[c]) continue;
                    visitedCons[c] = true;
                    componentCons.Add(c);
                    consQueue.Enqueue(c);
                }

                while (consQueue.Count > 0)
                {
                    int c = consQueue.Dequeue();
                    var vars = constraints[c].Vars;
                    for (int i = 0; i < vars.Length; i++)
                    {
                        int nv = vars[i];
                        if (visitedVar[nv]) continue;
                        visitedVar[nv] = true;
                        varQueue.Enqueue(nv);
                    }
                }
            }

            // Only brute force small components.
            if (componentVars.Count > 24) continue;

            SolveComponentByEnumeration(componentVars, componentCons, constraints, cellByVarId, grid, safe, safeList);
        }
    }

    private static void SolveComponentByEnumeration(
        List<int> componentVars,
        List<int> componentCons,
        List<Constraint> constraints,
        List<int> cellByVarId,
        byte[] grid,
        bool[] safe,
        List<int> safeList)
    {
        int n = componentVars.Count;
        int m = componentCons.Count;

        // Map global varId -> local 0..n-1
        var localIndex = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) localIndex[componentVars[i]] = i;

        // Build local constraints as bitmasks + counts.
        var consMask = new uint[m];
        var consNeed = new int[m];
        for (int i = 0; i < m; i++)
        {
            var con = constraints[componentCons[i]];
            uint mask = 0;
            for (int k = 0; k < con.Vars.Length; k++)
            {
                mask |= 1u << localIndex[con.Vars[k]];
            }
            consMask[i] = mask;
            consNeed[i] = con.MineCount;
        }

        // Precompute constraint membership for faster incremental updates.
        var consByLocalVar = new List<int>[n];
        for (int i = 0; i < n; i++) consByLocalVar[i] = new List<int>(4);
        for (int ci = 0; ci < m; ci++)
        {
            uint mask = consMask[ci];
            for (int v = 0; v < n; v++)
            {
                if (((mask >> v) & 1u) != 0) consByLocalVar[v].Add(ci);
            }
        }

        // Track partial assignment stats per constraint.
        var assignedMines = new int[m];
        var assignedVars = new int[m];
        var consSize = new int[m];
        for (int i = 0; i < m; i++) consSize[i] = PopCount(consMask[i]);

        long totalSolutions = 0;
        long[] mineCountInSolutions = new long[n]; // how many solutions assign mine to var i

        // Variable ordering: most constrained first.
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => consByLocalVar[b].Count.CompareTo(consByLocalVar[a].Count));

        var currentAssignment = new bool[n];

        void Dfs(int depth)
        {
            if (depth == n)
            {
                // All constraints must be satisfied.
                for (int i = 0; i < m; i++)
                {
                    if (assignedMines[i] != consNeed[i]) return;
                }

                totalSolutions++;
                // Reconstruct mines from assignedMines deltas is hard; maintain current assignment separately.
                for (int i = 0; i < n; i++)
                {
                    if (currentAssignment[i]) mineCountInSolutions[i]++;
                }
                return;
            }

            int v = order[depth];

            // Try safe (0)
            if (Assign(v, isMine: false))
            {
                Dfs(depth + 1);
            }
            Unassign(v);

            // Try mine (1)
            if (Assign(v, isMine: true))
            {
                Dfs(depth + 1);
            }
            Unassign(v);
        }

        bool Assign(int localVar, bool isMine)
        {
            currentAssignment[localVar] = isMine;

            foreach (int ci in consByLocalVar[localVar])
            {
                assignedVars[ci]++;
                if (isMine) assignedMines[ci]++;
            }

            // Feasibility pruning (component is small, so check all constraints)
            for (int ci = 0; ci < m; ci++)
            {
                int mines = assignedMines[ci];
                int assigned = assignedVars[ci];
                int need = consNeed[ci];
                int remaining = consSize[ci] - assigned;
                if (mines > need) return false;
                if (mines + remaining < need) return false;
            }

            return true;
        }

        void Unassign(int localVar)
        {
            bool wasMine = currentAssignment[localVar];
            currentAssignment[localVar] = false;
            foreach (int ci in consByLocalVar[localVar])
            {
                assignedVars[ci]--;
                if (wasMine) assignedMines[ci]--;
            }
        }

        // Run enumeration.
        Dfs(0);

        if (totalSolutions == 0) return; // inconsistent component; ignore

        // Any variable that is always 0 across all solutions is provably safe; always 1 is provably mine.
        for (int i = 0; i < n; i++)
        {
            int globalVar = componentVars[i];
            int cellIdx = cellByVarId[globalVar];
            if (grid[cellIdx] != Unknown) continue;
            if (safe[cellIdx]) continue;

            long minesIn = mineCountInSolutions[i];
            if (minesIn == 0)
            {
                safe[cellIdx] = true;
                safeList.Add(cellIdx);
            }
            else if (minesIn == totalSolutions)
            {
                grid[cellIdx] = FlaggedMine;
            }
        }
    }

    private static int PopCount(uint x)
    {
        // builtin popcount for net10 via System.Numerics.BitOperations is available, but keep simple.
        int c = 0;
        while (x != 0)
        {
            x &= x - 1;
            c++;
        }
        return c;
    }
}
