const fs = require('fs');

class MinHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp || ((x, y) => x - y);
  }
  size() { return this.a.length; }
  peek() { return this.a[0]; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) >= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      this._down(0);
    }
    return top;
  }
  _down(i) {
    const a = this.a, n = a.length;
    for (;;) {
      let l = i * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let m = (r < n && this.cmp(a[r], a[l]) < 0) ? r : l;
      if (this.cmp(a[m], a[i]) >= 0) break;
      [a[i], a[m]] = [a[m], a[i]];
      i = m;
    }
  }
}

class MaxHeap {
  constructor(cmp) {
    this.a = [];
    this.cmp = cmp || ((x, y) => x - y);
  }
  size() { return this.a.length; }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.cmp(a[i], a[p]) <= 0) break;
      [a[i], a[p]] = [a[p], a[i]];
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (a.length) {
      a[0] = last;
      this._down(0);
    }
    return top;
  }
  _down(i) {
    const a = this.a, n = a.length;
    for (;;) {
      let l = i * 2 + 1;
      if (l >= n) break;
      let r = l + 1;
      let m = (r < n && this.cmp(a[r], a[l]) > 0) ? r : l;
      if (this.cmp(a[m], a[i]) <= 0) break;
      [a[i], a[m]] = [a[m], a[i]];
      i = m;
    }
  }
}

function parseInput() {
  const input = fs.readFileSync(0, 'utf8').trim();
  if (!input) return null;
  return JSON.parse(input);
}

function main() {
  const data = parseInput();
  if (!data) {
    process.stdout.write(JSON.stringify({ schedule: [] }));
    return;
  }
  const machineCount = Math.max(1, data.machineCount | 0);
  const tasksArr = Array.isArray(data.tasks) ? data.tasks : [];
  const n = tasksArr.length;
  if (n === 0) {
    process.stdout.write(JSON.stringify({ schedule: [] }));
    return;
  }

  // Map task ids to indices
  const idToIdx = new Map();
  const ids = new Array(n);
  for (let i = 0; i < n; i++) {
    const id = tasksArr[i].id;
    ids[i] = id;
    idToIdx.set(id, i);
  }

  const duration = new Array(n);
  const deadline = new Array(n);
  const weight = new Array(n);
  const indeg = new Array(n).fill(0);
  const succ = Array.from({ length: n }, () => []);
  const depsList = Array.from({ length: n }, () => []);

  for (let i = 0; i < n; i++) {
    const t = tasksArr[i] || {};
    duration[i] = Math.max(0, t.duration | 0);
    deadline[i] = Number.isFinite(t.deadline) ? t.deadline : (t.deadline | 0);
    weight[i] = Number.isFinite(t.weight) ? t.weight : (t.weight | 0);
    const deps = Array.isArray(t.deps) ? t.deps : [];
    for (let j = 0; j < deps.length; j++) {
      const dId = deps[j];
      const di = idToIdx.get(dId);
      if (di === undefined) continue; // ignore unknown deps
      succ[di].push(i);
      depsList[i].push(di);
      indeg[i]++;
    }
  }

  // Compute a topological order (also detects cycles).
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  const topo = [];
  for (let qi = 0; qi < q.length; qi++) {
    const u = q[qi];
    topo.push(u);
    const s = succ[u];
    for (let k = 0; k < s.length; k++) {
      const v = s[k];
      indeg[v]--;
      if (indeg[v] === 0) q.push(v);
    }
  }

  if (topo.length !== n) {
    // Cycle or unresolved deps: fall back to stable order of ids.
    process.stdout.write(JSON.stringify({ order: ids }));
    return;
  }

  // Rebuild indegrees for scheduling.
  const remIn = new Array(n).fill(0);
  for (let i = 0; i < n; i++) remIn[i] = depsList[i].length;

  // Critical path length to end (duration + max successor CP).
  const cp = new Array(n).fill(0);
  for (let ti = n - 1; ti >= 0; ti--) {
    const u = topo[ti];
    let best = 0;
    const s = succ[u];
    for (let k = 0; k < s.length; k++) {
      const v = s[k];
      if (cp[v] > best) best = cp[v];
    }
    cp[u] = duration[u] + best;
  }

  const latestDepFinish = new Array(n).fill(0);
  const startTime = new Array(n).fill(0);
  const finishTime = new Array(n).fill(0);
  const assignedMachine = new Array(n).fill(-1);
  const scheduled = new Array(n).fill(false);

  // Machines available times.
  const machineHeap = new MinHeap((a, b) => (a.time - b.time) || (a.machine - b.machine));
  for (let m = 0; m < machineCount; m++) machineHeap.push({ time: 0, machine: m });

  // Ready tasks heap (by dynamic priority).
  // We'll store {idx, stamp, score} where stamp is time used to compute score.
  const readyHeap = new MaxHeap((a, b) => (a.score - b.score) || (cp[a.idx] - cp[b.idx]) || (weight[a.idx] - weight[b.idx]));

  function computeScore(i, now) {
    const d = duration[i] || 0;
    const w = weight[i] || 0;
    const dl = deadline[i];
    const slack = (Number.isFinite(dl) ? (dl - now - d) : 0);
    const posSlack = slack > 0 ? slack : 0;
    const urgency = (w + 1) / (1 + posSlack); // higher when slack is small/negative
    const dens = (w + 1) / (1 + d);
    // Tuned weights: emphasize deadlines/weights, then critical path.
    return 1000 * urgency + 10 * cp[i] + dens;
  }

  // Initial ready tasks.
  for (let i = 0; i < n; i++) {
    if (remIn[i] === 0) {
      const est = latestDepFinish[i];
      // For dynamic scoring, stamp with est (best guess time when it could start).
      readyHeap.push({ idx: i, stamp: est, score: computeScore(i, est) });
    }
  }

  // We schedule tasks as machines become available; enforce earliestStart based on deps.
  let doneCount = 0;
  while (doneCount < n) {
    const mach = machineHeap.pop();
    let now = mach.time;
    const m = mach.machine;

    // If no ready tasks (shouldn't happen in DAG, but can due to time constraints handling),
    // advance time to the earliest dependency completion that unlocks something.
    if (readyHeap.size() === 0) {
      // Find the next time any unscheduled task becomes ready by deps finishing.
      let nextT = Infinity;
      for (let i = 0; i < n; i++) {
        if (!scheduled[i] && remIn[i] === 0) {
          if (latestDepFinish[i] < nextT) nextT = latestDepFinish[i];
        }
      }
      if (!Number.isFinite(nextT)) nextT = now;
      now = Math.max(now, nextT);
    }

    // Select best task that is actually ready by time (deps done, and est <= now).
    let chosen = null;
    while (readyHeap.size() > 0) {
      const cand = readyHeap.pop();
      const i = cand.idx;
      if (scheduled[i]) continue;
      const est = latestDepFinish[i];
      const stamp = Math.max(now, est);
      if (cand.stamp !== stamp) {
        cand.stamp = stamp;
        cand.score = computeScore(i, stamp);
        readyHeap.push(cand);
        continue;
      }
      chosen = i;
      break;
    }

    if (chosen === null) {
      // No task could be chosen yet; push machine back with advanced time.
      machineHeap.push({ time: now, machine: m });
      continue;
    }

    const i = chosen;
    const est = latestDepFinish[i];
    const st = Math.max(now, est);
    const ft = st + duration[i];
    scheduled[i] = true;
    startTime[i] = st;
    finishTime[i] = ft;
    assignedMachine[i] = m;
    doneCount++;

    // Update successors
    const s = succ[i];
    for (let k = 0; k < s.length; k++) {
      const v = s[k];
      if (finishTime[i] > latestDepFinish[v]) latestDepFinish[v] = finishTime[i];
      remIn[v]--;
      if (remIn[v] === 0 && !scheduled[v]) {
        const estV = latestDepFinish[v];
        readyHeap.push({ idx: v, stamp: estV, score: computeScore(v, estV) });
      }
    }

    // Machine becomes available at ft
    machineHeap.push({ time: ft, machine: m });
  }

  // Emit schedule sorted by start time (tie by machine, then id for determinism).
  const sched = [];
  for (let i = 0; i < n; i++) {
    sched.push({ id: ids[i], start: startTime[i] | 0, machine: assignedMachine[i] | 0 });
  }
  sched.sort((a, b) => (a.start - b.start) || (a.machine - b.machine) || (String(a.id).localeCompare(String(b.id))));
  process.stdout.write(JSON.stringify({ schedule: sched }));
}

main();
