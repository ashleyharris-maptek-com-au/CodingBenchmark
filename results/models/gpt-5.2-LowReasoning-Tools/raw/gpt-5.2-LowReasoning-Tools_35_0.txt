use std::cmp::Ordering;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}
impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.state;
        x ^= x << 7;
        x ^= x >> 9;
        x ^= x << 8;
        self.state = x;
        x
    }
    fn gen_range(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            (self.next_u64() as usize) % upper
        }
    }
    fn shuffle<T>(&mut self, a: &mut [T]) {
        for i in (1..a.len()).rev() {
            let j = self.gen_range(i + 1);
            a.swap(i, j);
        }
    }
}

fn elapsed_since(start: Instant) -> Duration {
    match Instant::now().checked_duration_since(start) {
        Some(d) => d,
        None => Duration::from_secs(1_000_000),
    }
}

fn main() {
    // Read input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        if let (Some(us), Some(vs)) = (it.next(), it.next()) {
            let u: usize = us.parse().unwrap();
            let v: usize = vs.parse().unwrap();
            if u == v || u >= n || v >= n {
                continue;
            }
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    for a in &mut adj {
        a.sort_unstable();
        a.dedup();
    }

    // Precompute closed neighborhoods N[v] ∪ {v}
    let mut closed: Vec<Vec<usize>> = Vec::with_capacity(n);
    closed.resize_with(n, || Vec::new());
    for v in 0..n {
        let mut nb = Vec::with_capacity(adj[v].len() + 1);
        nb.push(v);
        nb.extend_from_slice(&adj[v]);
        nb.sort_unstable();
        nb.dedup();
        closed[v] = nb;
    }

    // Seed RNG from a stable-ish hash of input size.
    let seed = (n as u64).wrapping_mul(0x9E3779B97F4A7C15).wrapping_add(m as u64);
    let mut rng = XorShift64::new(seed ^ 0xD1B54A32D192ED03);

    // Fast constructive heuristic + redundancy pruning.
    let mut best = greedy_construct(n, &closed, &mut rng, true);
    prune_redundant(n, &closed, &mut best, &mut rng);
    if !is_dominating(n, &closed, &best) {
        make_dominating(n, &closed, &mut best, &mut rng);
        prune_redundant(n, &closed, &mut best, &mut rng);
    }

    // Output
    println!("{}", best.len());
    if !best.is_empty() {
        for (i, v) in best.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
    }
    println!();
}

fn greedy_construct(n: usize, closed: &[Vec<usize>], rng: &mut XorShift64, randomized: bool) -> Vec<usize> {
    let mut dominated = vec![false; n];
    let mut undom = n;
    let mut score: Vec<u32> = vec![0; n];
    for v in 0..n {
        score[v] = closed[v].len() as u32;
    }

    let mut set: Vec<usize> = Vec::new();

    // Handle isolated vertices quickly.
    for v in 0..n {
        if closed[v].len() == 1 {
            set.push(v);
            if !dominated[v] {
                dominated[v] = true;
                undom -= 1;
            }
        }
    }

    // Recompute scores for uncovered coverage only.
    while undom > 0 {
        let mut best_v = 0usize;
        let mut best_gain = 0u32;
        let mut best_deg = 0u32;

        // Randomized greedy: sample a subset to reduce time on huge graphs.
        let trials = if randomized {
            (64usize).min(n.max(1))
        } else {
            n.max(1)
        };

        for t in 0..trials {
            let v = if randomized { rng.gen_range(n) } else { t };
            // Gain = number of undominated vertices in closed[v]
            let mut gain = 0u32;
            for &u in &closed[v] {
                if !dominated[u] {
                    gain += 1;
                }
            }
            if gain == 0 {
                continue;
            }
            let deg = score[v];
            match gain.cmp(&best_gain) {
                Ordering::Greater => {
                    best_gain = gain;
                    best_v = v;
                    best_deg = deg;
                }
                Ordering::Equal => {
                    if deg > best_deg || (deg == best_deg && v < best_v) {
                        best_v = v;
                        best_deg = deg;
                    }
                }
                Ordering::Less => {}
            }
        }

        // Fallback to full scan if randomized sampling didn't find a gain.
        if best_gain == 0 {
            for v in 0..n {
                let mut gain = 0u32;
                for &u in &closed[v] {
                    if !dominated[u] {
                        gain += 1;
                    }
                }
                if gain > best_gain {
                    best_gain = gain;
                    best_v = v;
                    best_deg = score[v];
                } else if gain == best_gain && gain > 0 {
                    let deg = score[v];
                    if deg > best_deg || (deg == best_deg && v < best_v) {
                        best_v = v;
                        best_deg = deg;
                    }
                }
            }
        }

        set.push(best_v);
        for &u in &closed[best_v] {
            if !dominated[u] {
                dominated[u] = true;
                undom -= 1;
            }
        }
    }

    set.sort_unstable();
    set.dedup();
    set
}

fn is_dominating(n: usize, closed: &[Vec<usize>], set: &[usize]) -> bool {
    let mut dominated = vec![false; n];
    for &v in set {
        for &u in &closed[v] {
            dominated[u] = true;
        }
    }
    dominated.into_iter().all(|x| x)
}

fn make_dominating(n: usize, closed: &[Vec<usize>], set: &mut Vec<usize>, rng: &mut XorShift64) {
    let mut in_set = vec![false; n];
    for &v in set.iter() {
        if v < n {
            in_set[v] = true;
        }
    }

    let mut dominated = vec![false; n];
    for &v in set.iter() {
        if v >= n {
            continue;
        }
        for &u in &closed[v] {
            dominated[u] = true;
        }
    }

    let mut undominated: Vec<usize> = (0..n).filter(|&v| !dominated[v]).collect();
    while let Some(v) = undominated.pop() {
        // Add a vertex from v's closed neighborhood that covers most uncovered nearby.
        let mut best = v;
        let mut best_gain = 0u32;
        for &cand in &closed[v] {
            let mut gain = 0u32;
            for &u in &closed[cand] {
                if !dominated[u] {
                    gain += 1;
                }
            }
            if gain > best_gain || (gain == best_gain && cand < best) {
                best_gain = gain;
                best = cand;
            }
        }
        if !in_set[best] {
            set.push(best);
            in_set[best] = true;
            for &u in &closed[best] {
                dominated[u] = true;
            }
        }
        // Refresh undominated list lazily with some randomness to avoid O(n^2).
        if undominated.is_empty() {
            for v2 in 0..n {
                if !dominated[v2] && rng.gen_range(4) != 0 {
                    undominated.push(v2);
                }
            }
            if undominated.is_empty() {
                undominated = (0..n).filter(|&x| !dominated[x]).collect();
            }
        }
    }
}

fn prune_redundant(n: usize, closed: &[Vec<usize>], set: &mut Vec<usize>, rng: &mut XorShift64) {
    // Remove vertices that are not needed (reverse-delete heuristic).
    set.sort_unstable();
    set.dedup();

    let mut in_set = vec![false; n];
    for &v in set.iter() {
        if v < n {
            in_set[v] = true;
        }
    }

    let mut cover_count = vec![0u32; n];
    for &v in set.iter() {
        if v >= n {
            continue;
        }
        for &u in &closed[v] {
            cover_count[u] += 1;
        }
    }

    let mut order: Vec<usize> = set.clone();
    rng.shuffle(&mut order);
    // Prefer trying to remove low-impact vertices first
    order.sort_unstable_by(|&a, &b| closed[a].len().cmp(&closed[b].len()).then_with(|| a.cmp(&b)));

    for v in order {
        if !in_set[v] {
            continue;
        }
        // Check if removing v leaves all covered
        let mut ok = true;
        for &u in &closed[v] {
            if cover_count[u] <= 1 {
                ok = false;
                break;
            }
        }
        if ok {
            in_set[v] = false;
            for &u in &closed[v] {
                cover_count[u] -= 1;
            }
        }
    }

    set.retain(|&v| v < n && in_set[v]);
}

fn local_improve(
    n: usize,
    closed: &[Vec<usize>],
    set: &mut Vec<usize>,
    rng: &mut XorShift64,
    start: Instant,
    time_limit: Duration,
) {
    // Simple hill-climbing:
    // 1) Ensure dominating
    // 2) Repeatedly attempt removals and small (1-for-2) replacements.
    set.sort_unstable();
    set.dedup();
    if !is_dominating(n, closed, set) {
        make_dominating(n, closed, set, rng);
        prune_redundant(n, closed, set, rng);
    }

    let mut in_set = vec![false; n];
    for &v in set.iter() {
        in_set[v] = true;
    }

    let mut cover_count = vec![0u32; n];
    for &v in set.iter() {
        for &u in &closed[v] {
            cover_count[u] += 1;
        }
    }

    // Candidate list for add operations: endpoints from uncovered vertices when trying a swap.
    let mut tmp_uncovered: Vec<usize> = Vec::new();

    // Limit the number of local iterations to stay fast.
    let mut rounds = 0u32;
    while elapsed_since(start) < time_limit {
        rounds += 1;
        if rounds > 2000 {
            break;
        }

        // Try removals in random order.
        let mut order = set.clone();
        rng.shuffle(&mut order);
        let mut changed = false;
        for v in order {
            if elapsed_since(start) >= time_limit {
                break;
            }
            if !in_set[v] {
                continue;
            }
            let mut ok = true;
            for &u in &closed[v] {
                if cover_count[u] <= 1 {
                    ok = false;
                    break;
                }
            }
            if ok {
                in_set[v] = false;
                for &u in &closed[v] {
                    cover_count[u] -= 1;
                }
                changed = true;
            }
        }

        // Materialize set after removals.
        if changed {
            set.retain(|&v| in_set[v]);
        }

        // Try a small replacement: remove one vertex v, add up to two vertices to cover what v uniquely covered.
        // This can reduce size when one added vertex allows removal of multiple later.
        if elapsed_since(start) >= time_limit {
            break;
        }
        let mut attempts = 0u32;
        while attempts < 200 && elapsed_since(start) < time_limit {
            attempts += 1;
            if set.is_empty() {
                break;
            }
            let idx = rng.gen_range(set.len());
            let v = set[idx];
            // Identify vertices that would become uncovered if we remove v.
            tmp_uncovered.clear();
            for &u in &closed[v] {
                if cover_count[u] == 1 {
                    tmp_uncovered.push(u);
                }
            }
            if tmp_uncovered.is_empty() {
                // v is redundant; remove it.
                in_set[v] = false;
                for &u in &closed[v] {
                    cover_count[u] -= 1;
                }
                set.swap_remove(idx);
                continue;
            }

            // Remove v temporarily
            in_set[v] = false;
            for &u in &closed[v] {
                cover_count[u] -= 1;
            }

            // Greedily add one or two vertices that cover tmp_uncovered.
            let mut added: Vec<usize> = Vec::new();
            for _ in 0..2 {
                // If everything is already covered, stop.
                if tmp_uncovered.iter().all(|&u| cover_count[u] > 0) {
                    break;
                }
                let mut best_cand = None;
                let mut best_gain = 0u32;
                // Sample candidates from neighborhoods of uncovered vertices.
                let samples = (16usize).min(tmp_uncovered.len().max(1));
                for _ in 0..samples {
                    let u = tmp_uncovered[rng.gen_range(tmp_uncovered.len())];
                    for &cand in &closed[u] {
                        if in_set[cand] {
                            continue;
                        }
                        let mut gain = 0u32;
                        for &x in &closed[cand] {
                            if cover_count[x] == 0 {
                                gain += 1;
                            }
                        }
                        if gain > best_gain {
                            best_gain = gain;
                            best_cand = Some(cand);
                        }
                    }
                }
                if let Some(cand) = best_cand {
                    in_set[cand] = true;
                    added.push(cand);
                    for &x in &closed[cand] {
                        cover_count[x] += 1;
                    }
                } else {
                    break;
                }
            }

            let success = tmp_uncovered.iter().all(|&u| cover_count[u] > 0);
            if success {
                // Keep changes: v removed, added kept. Update set vector.
                set.swap_remove(idx);
                for a in added {
                    set.push(a);
                }
                prune_redundant(n, closed, set, rng);
                // Rebuild cover_count/in_set after pruning (simple and safe).
                in_set.fill(false);
                cover_count.fill(0);
                for &vv in set.iter() {
                    in_set[vv] = true;
                    for &u in &closed[vv] {
                        cover_count[u] += 1;
                    }
                }
            } else {
                // Roll back: remove added, restore v
                for a in added {
                    if in_set[a] {
                        in_set[a] = false;
                        for &x in &closed[a] {
                            cover_count[x] -= 1;
                        }
                    }
                }
                in_set[v] = true;
                for &u in &closed[v] {
                    cover_count[u] += 1;
                }
            }
        }

        // If no progress possible quickly, stop.
        if !changed {
            break;
        }
    }

    // Final cleanup
    prune_redundant(n, closed, set, rng);
    if !is_dominating(n, closed, set) {
        make_dominating(n, closed, set, rng);
        prune_redundant(n, closed, set, rng);
    }
}
