#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <tuple>
#include <utility>
#include <vector>

using i64 = long long;

struct Task {
  int machine = 0;
  int duration = 0;
};

struct Problem {
  int jobs = 0;
  int machines = 0;
  std::vector<std::vector<Task>> job_tasks;
  int total_tasks = 0;
};

enum class Rule {
  EarliestStart_SPT,   // earliest feasible start, then shortest processing time
  EarliestStart_LPT,   // earliest feasible start, then longest processing time
  EarliestStart_MWKR,  // earliest start, then most work remaining in job
  EarliestStart_LWKR,  // earliest start, then least work remaining in job
};

struct Schedule {
  std::vector<std::vector<i64>> start_time;
  std::vector<std::vector<int>> machine;
  i64 makespan = 0;
};

static Problem read_problem() {
  Problem p;
  if (!(std::cin >> p.jobs >> p.machines)) return p;
  p.job_tasks.resize(p.jobs);
  p.total_tasks = 0;
  for (int j = 0; j < p.jobs; ++j) {
    int t = 0;
    std::cin >> t;
    p.job_tasks[j].resize(t);
    p.total_tasks += t;
    for (int k = 0; k < t; ++k) {
      int m = 0, d = 0;
      std::cin >> m >> d;
      p.job_tasks[j][k] = Task{m, d};
    }
  }
  return p;
}

static std::vector<i64> compute_work_remaining(const Problem& p) {
  // Total remaining work per job; updated incrementally during scheduling.
  std::vector<i64> total(p.jobs, 0);
  for (int j = 0; j < p.jobs; ++j) {
    i64 sum = 0;
    for (const auto& task : p.job_tasks[j]) sum += task.duration;
    total[j] = sum;
  }
  return total;
}

struct Candidate {
  int job = 0;
  int idx = 0;
  int machine = 0;
  int duration = 0;
  i64 cached_est = 0;
  i64 cached_tie = 0;
};

struct CandCmp {
  bool operator()(const Candidate& a, const Candidate& b) const {
    if (a.cached_est != b.cached_est) return a.cached_est > b.cached_est; // min-heap by est
    if (a.cached_tie != b.cached_tie) return a.cached_tie > b.cached_tie;
    if (a.duration != b.duration) return a.duration > b.duration;
    if (a.machine != b.machine) return a.machine > b.machine;
    return a.job > b.job;
  }
};

static i64 tie_value(Rule rule, int duration, i64 work_remaining) {
  switch (rule) {
    case Rule::EarliestStart_SPT: return duration;                    // smaller duration first
    case Rule::EarliestStart_LPT: return -static_cast<i64>(duration); // larger duration first
    case Rule::EarliestStart_MWKR: return -work_remaining;            // more work remaining first
    case Rule::EarliestStart_LWKR: return work_remaining;             // less work remaining first
  }
  return duration;
}

static Schedule build_schedule(const Problem& p, Rule rule) {
  Schedule s;
  s.start_time.resize(p.jobs);
  s.machine.resize(p.jobs);
  for (int j = 0; j < p.jobs; ++j) {
    s.start_time[j].assign(p.job_tasks[j].size(), 0);
    s.machine[j].assign(p.job_tasks[j].size(), 0);
  }

  std::vector<i64> job_ready(p.jobs, 0);
  std::vector<i64> machine_ready(p.machines, 0);
  std::vector<int> job_next(p.jobs, 0);
  std::vector<i64> work_rem = compute_work_remaining(p);

  std::priority_queue<Candidate, std::vector<Candidate>, CandCmp> pq;

  for (int j = 0; j < p.jobs; ++j) {
    if (p.job_tasks[j].empty()) continue;
    const auto& t = p.job_tasks[j][0];
    Candidate c;
    c.job = j;
    c.idx = 0;
    c.machine = t.machine;
    c.duration = t.duration;
    c.cached_est = std::max<i64>(job_ready[j], machine_ready[c.machine]);
    c.cached_tie = tie_value(rule, c.duration, work_rem[j]);
    pq.push(c);
  }

  while (!pq.empty()) {
    Candidate c = pq.top();
    pq.pop();

    // Stale candidate if the job has advanced.
    if (c.idx != job_next[c.job]) continue;

    // Recompute dynamic priority (lazy update).
    i64 est = std::max<i64>(job_ready[c.job], machine_ready[c.machine]);
    i64 tie = tie_value(rule, c.duration, work_rem[c.job]);
    if (est != c.cached_est || tie != c.cached_tie) {
      c.cached_est = est;
      c.cached_tie = tie;
      pq.push(c);
      continue;
    }

    const i64 start = est;
    const i64 end = start + c.duration;
    s.start_time[c.job][c.idx] = start;
    s.machine[c.job][c.idx] = c.machine;

    job_ready[c.job] = end;
    machine_ready[c.machine] = end;
    ++job_next[c.job];
    work_rem[c.job] -= c.duration;
    s.makespan = std::max(s.makespan, end);

    const int next_idx = job_next[c.job];
    if (next_idx < static_cast<int>(p.job_tasks[c.job].size())) {
      const auto& nt = p.job_tasks[c.job][next_idx];
      Candidate n;
      n.job = c.job;
      n.idx = next_idx;
      n.machine = nt.machine;
      n.duration = nt.duration;
      n.cached_est = std::max<i64>(job_ready[n.job], machine_ready[n.machine]);
      n.cached_tie = tie_value(rule, n.duration, work_rem[n.job]);
      pq.push(n);
    }
  }

  return s;
}

static Schedule solve(const Problem& p) {
  const int n = p.total_tasks;
  if (n == 0 || p.jobs == 0) return Schedule{};

  // Adaptive logic: more passes for smaller instances, one fast pass for huge ones.
  const bool very_large = (n >= 50000) || (p.jobs >= 1500) || (p.machines >= 80);
  const bool small = (n <= 2000);

  std::vector<Rule> rules;
  if (very_large) {
    rules = {Rule::EarliestStart_MWKR};
  } else if (small) {
    rules = {Rule::EarliestStart_MWKR, Rule::EarliestStart_SPT, Rule::EarliestStart_LPT,
             Rule::EarliestStart_LWKR};
  } else {
    rules = {Rule::EarliestStart_MWKR, Rule::EarliestStart_SPT};
  }

  Schedule best;
  best.makespan = std::numeric_limits<i64>::max();
  for (Rule r : rules) {
    Schedule cand = build_schedule(p, r);
    if (cand.makespan < best.makespan) best = std::move(cand);
  }
  return best;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  Problem p = read_problem();
  Schedule s = solve(p);

  for (int j = 0; j < p.jobs; ++j) {
    const int t = static_cast<int>(p.job_tasks[j].size());
    for (int k = 0; k < t; ++k) {
      if (k) std::cout << ' ';
      std::cout << s.start_time[j][k] << ' ' << s.machine[j][k];
    }
    std::cout << "\n";
  }
  return 0;
}
