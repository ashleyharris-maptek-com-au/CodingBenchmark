// MiningSolver.cs
// Benchmark solution: fast, safe heuristic with adaptive scaling.
// Output is a sequence of "move voxel" operations. If no safe profitable moves are found,
// outputs 0 operations (always valid).

using System;
using System.Buffers;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    private const long SmallVoxelLimit = 1_000_000; // adaptive cutoff

    public static void Main()
    {
        // Parse header (space-delimited)
        var stdin = Console.OpenStandardInput();
        var header = ReadHeaderLine(stdin);
        if (header == null || header.Length < 6)
        {
            Console.WriteLine("0");
            return;
        }

        int xCount = ParseInt(header[0]);
        int yCount = ParseInt(header[1]);
        int zCount = ParseInt(header[2]);

        // Block sizes are not required for this heuristic, but parse to validate input.
        _ = ParseDouble(header[3]);
        _ = ParseDouble(header[4]);
        _ = ParseDouble(header[5]);

        long total = (long)xCount * yCount * zCount;
        if (total <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        // Adaptive strategy:
        // - For huge worlds, do nothing (valid, completes fast).
        // - For small worlds, attempt to move only the highest-profit surface ore
        //   that is reachable with a simple slope-safe Manhattan path.
        if (total > SmallVoxelLimit || xCount > 2000 || yCount > 2000 || zCount > 2000)
        {
            // Stream and discard voxel bytes to fully consume stdin if required by harness.
            DrainBytes(stdin, total);
            Console.WriteLine("0");
            return;
        }

        // Load entire grid for small case.
        // Layout matches input order: x major, then y, then z.
        int xy = checked(xCount * yCount);
        byte[] voxels = new byte[checked((int)total)];
        ReadExactly(stdin, voxels);

        // Compute top z for each (x,y) where mass>0.
        short[] topZ = new short[xy];
        for (int i = 0; i < xy; i++) topZ[i] = -1;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int baseIdx = ((x * yCount) + y) * zCount;
                for (int z = zCount - 1; z >= 0; z--)
                {
                    byte d = voxels[baseIdx + z];
                    int mass = d & 0x0F;
                    if (mass == 0) continue;
                    topZ[x * yCount + y] = (short)z;
                    break;
                }
            }
        }

        // Find best candidate surface voxel (highest value*mass, tie-break closer to plant).
        int bestX = -1, bestY = -1, bestZ = -1;
        int bestMass = 0, bestValue = 0;
        long bestScore = long.MinValue;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int tz = topZ[x * yCount + y];
                if (tz < 0) continue;

                int idx = ((x * yCount) + y) * zCount + tz;
                byte d = voxels[idx];
                int mass = d & 0x0F;
                int value = (d >> 4) & 0x0F;
                if (mass <= 0) continue;

                // Skip junk to avoid plant damage (value==0).
                if (value <= 0) continue;

                long gross = 1000L * mass * value;
                // Very rough fuel estimate (flat $1 per step per ton); uphill ignored in score.
                int manhattan = x + y;
                long fuel = 1L * mass * manhattan;
                long score = gross - fuel;

                // Prefer closer if similar; also avoid far negative.
                if (score > bestScore || (score == bestScore && manhattan < (bestX + bestY)))
                {
                    bestScore = score;
                    bestX = x; bestY = y; bestZ = tz;
                    bestMass = mass; bestValue = value;
                }
            }
        }

        if (bestX < 0)
        {
            Console.WriteLine("0");
            return;
        }

        // Build a slope-safe Manhattan path on current terrain (topZ).
        // Model travel height at a cell as (topZ+1). Require abs(delta) <= 1 per move.
        if (!TryBuildGreedySlopePath(bestX, bestY, topZ, xCount, yCount, out var steps))
        {
            Console.WriteLine("0");
            return;
        }

        // Output single operation:
        // N digX digY digZ stepX1 stepY1 ... stepXN stepYN destX destY
        // We include each intermediate position (excluding the starting cell), ending at (0,0).
        // N equals number of XY steps.
        Console.WriteLine("1");
        Console.Write(steps.Count);
        Console.Write(' ');
        Console.Write(bestX);
        Console.Write(' ');
        Console.Write(bestY);
        Console.Write(' ');
        Console.Write(bestZ);

        for (int i = 0; i < steps.Count; i++)
        {
            Console.Write(' ');
            Console.Write(steps[i].x);
            Console.Write(' ');
            Console.Write(steps[i].y);
        }

        // Destination
        Console.Write(' ');
        Console.Write(0);
        Console.Write(' ');
        Console.Write(0);
        Console.WriteLine();
    }

    private static bool TryBuildGreedySlopePath(int startX, int startY, short[] topZ, int xCount, int yCount, out List<(int x, int y)> steps)
    {
        steps = new List<(int x, int y)>(startX + startY);
        int x = startX, y = startY;

        int currentH = GetTravelHeight(x, y, topZ, yCount);
        int safety = (xCount + yCount) * 4 + 10;

        // Greedy: always try to step towards (0,0), prefer x then y, but respect slope.
        while ((x != 0 || y != 0) && safety-- > 0)
        {
            bool moved = false;
            // Candidate moves toward plant
            var candidates = new (int nx, int ny)[]
            {
                (x > 0 ? x - 1 : x, y),
                (x, y > 0 ? y - 1 : y),
                // Allow detour by one step if needed (still monotone-ish)
                (x < xCount - 1 ? x + 1 : x, y),
                (x, y < yCount - 1 ? y + 1 : y),
            };

            for (int i = 0; i < candidates.Length; i++)
            {
                int nx = candidates[i].nx;
                int ny = candidates[i].ny;
                if (nx == x && ny == y) continue;

                // Keep within bounds
                if ((uint)nx >= (uint)xCount || (uint)ny >= (uint)yCount) continue;

                // Avoid moving away unless necessary
                int distNow = x + y;
                int distNext = nx + ny;
                if (i < 2)
                {
                    // good, closer
                }
                else if (distNext > distNow + 1)
                {
                    continue;
                }

                int nextH = GetTravelHeight(nx, ny, topZ, yCount);
                if (Math.Abs(nextH - currentH) > 1) continue;

                x = nx; y = ny; currentH = nextH;
                steps.Add((x, y));
                moved = true;
                break;
            }

            if (!moved) return false;
        }

        return x == 0 && y == 0;
    }

    private static int GetTravelHeight(int x, int y, short[] topZ, int yCount)
    {
        int tz = topZ[x * yCount + y];
        // If empty column, treat as ground at -1 => travel height 0
        return (tz < 0) ? 0 : (tz + 1);
    }

    private static string[]? ReadHeaderLine(Stream s)
    {
        using var ms = new MemoryStream();
        int b;
        while ((b = s.ReadByte()) != -1)
        {
            if (b == '\n') break;
            if (b != '\r') ms.WriteByte((byte)b);
            if (ms.Length > 1024 * 1024) break; // sanity
        }
        if (ms.Length == 0 && b == -1) return null;
        string line = System.Text.Encoding.ASCII.GetString(ms.ToArray());
        return line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
    }

    private static int ParseInt(string s) => int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);
    private static double ParseDouble(string s) => double.Parse(s, NumberStyles.Float, CultureInfo.InvariantCulture);

    private static void ReadExactly(Stream s, byte[] buffer)
    {
        int offset = 0;
        while (offset < buffer.Length)
        {
            int n = s.Read(buffer, offset, buffer.Length - offset);
            if (n <= 0) throw new EndOfStreamException();
            offset += n;
        }
    }

    private static void DrainBytes(Stream s, long count)
    {
        byte[] buf = ArrayPool<byte>.Shared.Rent(1 << 20);
        try
        {
            long remaining = count;
            while (remaining > 0)
            {
                int want = (int)Math.Min(buf.Length, remaining);
                int n = s.Read(buf, 0, want);
                if (n <= 0) break;
                remaining -= n;
            }
        }
        finally
        {
            ArrayPool<byte>.Shared.Return(buf);
        }
    }
}
