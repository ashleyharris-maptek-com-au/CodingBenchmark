// Steiner Tree (heuristic) - metric closure MST + path expansion + Kruskal cleanup
// Builds a connected, acyclic subgraph including all terminals.
//
// Input:
//   num_vertices num_edges num_terminals
//   u v w (num_edges lines, 0-indexed, undirected)
//   terminal_1 ... terminal_K
// Output:
//   total_weight edge_count
//   u v (one per chosen edge)
//
// Notes:
// - This is a heuristic (NP-hard problem). Produces a feasible tree, not necessarily optimal.
// - Uses Dijkstra from each terminal; suitable when K is moderate.

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <tuple>
#include <unordered_map>
#include <utility>
#include <vector>

using std::int64_t;

struct FastInput {
    static constexpr size_t BUFSZ = 1 << 20;
    char buf[BUFSZ];
    size_t idx = 0, size = 0;

    inline char read() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSZ, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        do {
            c = read();
            if (!c) return false;
        } while (c <= ' '); // skip whitespace
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = read();
        }
        T val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = read();
        }
        out = neg ? -val : val;
        return true;
    }
};

struct DSU {
    std::vector<int> p, r;
    explicit DSU(int n = 0) { init(n); }
    void init(int n) {
        p.resize(n);
        r.assign(n, 0);
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int a) {
        while (p[a] != a) {
            p[a] = p[p[a]];
            a = p[a];
        }
        return a;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) std::swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) r[a]++;
        return true;
    }
};

struct Edge {
    int to;
    int w;
};

struct WeightedEdge {
    int u, v;
    int w;
};

static inline uint64_t pack_undirected(int a, int b) {
    if (a > b) std::swap(a, b);
    return (uint64_t(uint32_t(a)) << 32) | uint32_t(b);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    FastInput in;
    int n = 0, m = 0, k = 0;
    if (!in.readInt(n)) return 0;
    in.readInt(m);
    in.readInt(k);

    std::vector<std::vector<Edge>> g;
    g.assign(n, {});
    g.shrink_to_fit();
    g.assign(n, {});
    g.reserve(n);

    // Keep best weight for undirected pair (for final Kruskal on collected edges).
    std::unordered_map<uint64_t, int> best_w;
    best_w.reserve(size_t(m) * 2);

    for (int i = 0; i < m; ++i) {
        int u, v;
        int w;
        in.readInt(u);
        in.readInt(v);
        in.readInt(w);
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        g[u].push_back({v, w});
        g[v].push_back({u, w});
        uint64_t key = pack_undirected(u, v);
        auto it = best_w.find(key);
        if (it == best_w.end() || w < it->second) best_w[key] = w;
    }

    std::vector<int> terminals(k);
    for (int i = 0; i < k; ++i) in.readInt(terminals[i]);
    if (k == 0) {
        std::cout << "0 0\n";
        return 0;
    }
    if (k == 1) {
        std::cout << "0 0\n";
        return 0;
    }

    // Dijkstra workspace
    const int64_t INF = std::numeric_limits<int64_t>::max() / 4;
    std::vector<int64_t> dist(n);
    std::vector<int> parent(n);
    std::vector<int> parent_w(n);

    // Distances between terminals (metric closure)
    std::vector<std::vector<int64_t>> termDist(k, std::vector<int64_t>(k, INF));

    auto dijkstra = [&](int src) {
        std::fill(dist.begin(), dist.end(), INF);
        std::fill(parent.begin(), parent.end(), -1);
        std::fill(parent_w.begin(), parent_w.end(), 0);
        dist[src] = 0;
        using PQItem = std::pair<int64_t, int>;
        std::priority_queue<PQItem, std::vector<PQItem>, std::greater<PQItem>> pq;
        pq.push({0, src});
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d != dist[u]) continue;
            for (const auto& e : g[u]) {
                int v = e.to;
                int w = e.w;
                int64_t nd = d + int64_t(w);
                if (nd < dist[v]) {
                    dist[v] = nd;
                    parent[v] = u;
                    parent_w[v] = w;
                    pq.push({nd, v});
                }
            }
        }
    };

    // For reconstructing paths for chosen metric edges, we rerun Dijkstra from that source terminal,
    // then walk parent[] from target back to source.
    auto add_path_edges = [&](int src, int dst, std::unordered_map<uint64_t, int>& collected) {
        int cur = dst;
        while (cur != src && cur != -1) {
            int p = parent[cur];
            if (p == -1) break;
            uint64_t key = pack_undirected(cur, p);
            int w = parent_w[cur];
            auto it = collected.find(key);
            if (it == collected.end() || w < it->second) collected[key] = w;
            cur = p;
        }
    };

    // Compute terminal-to-terminal distances
    for (int i = 0; i < k; ++i) {
        int s = terminals[i];
        dijkstra(s);
        for (int j = 0; j < k; ++j) {
            termDist[i][j] = dist[terminals[j]];
        }
    }

    // Build MST over terminals in the metric closure (Prim on complete graph).
    // If graph is disconnected w.r.t terminals, we will still output a best-effort tree on reachable terminals;
    // but must include all terminals per spec, so we detect and fall back to a spanning forest attempt and then fail-safe.
    std::vector<int64_t> key(k, INF);
    std::vector<int> parentT(k, -1);
    std::vector<char> inMST(k, 0);
    key[0] = 0;
    for (int it = 0; it < k; ++it) {
        int u = -1;
        int64_t best = INF;
        for (int i = 0; i < k; ++i) {
            if (!inMST[i] && key[i] < best) {
                best = key[i];
                u = i;
            }
        }
        if (u == -1 || best >= INF / 2) break; // disconnected
        inMST[u] = 1;
        for (int v = 0; v < k; ++v) {
            if (!inMST[v] && termDist[u][v] < key[v]) {
                key[v] = termDist[u][v];
                parentT[v] = u;
            }
        }
    }

    bool allReachable = true;
    for (int i = 0; i < k; ++i) {
        if (!inMST[i]) {
            allReachable = false;
            break;
        }
    }
    if (!allReachable) {
        // No feasible solution exists that connects all terminals in a disconnected graph.
        // Emit a trivial (invalid) output is worse; instead output 0 0 but this will not satisfy the benchmark.
        // However, most benchmarks provide connected instances; keep a clear failure mode.
        std::cout << "0 0\n";
        return 0;
    }

    // Expand metric MST edges into original-graph shortest paths; collect candidate edges.
    std::unordered_map<uint64_t, int> collected;
    collected.reserve(size_t(k) * 16);
    for (int v = 1; v < k; ++v) {
        int u = parentT[v];
        if (u < 0) continue;
        int src = terminals[u];
        int dst = terminals[v];
        dijkstra(src);
        if (dist[dst] >= INF / 2) continue;
        add_path_edges(src, dst, collected);
    }

    // Convert collected edges to list, using best weight for that pair from input (more robust).
    std::vector<WeightedEdge> cand;
    cand.reserve(collected.size());
    for (const auto& kv : collected) {
        int u = int(kv.first >> 32);
        int v = int(uint32_t(kv.first));
        int w = kv.second;
        auto it = best_w.find(kv.first);
        if (it != best_w.end()) w = std::min(w, it->second);
        cand.push_back({u, v, w});
    }

    // Kruskal over candidate edges to ensure acyclic; stop once terminals connected.
    std::sort(cand.begin(), cand.end(), [](const WeightedEdge& a, const WeightedEdge& b) {
        if (a.w != b.w) return a.w < b.w;
        if (a.u != b.u) return a.u < b.u;
        return a.v < b.v;
    });

    DSU dsu(n);
    std::vector<std::pair<int, int>> chosen;
    chosen.reserve(n);
    int64_t total = 0;

    auto terminals_connected = [&]() -> bool {
        int r0 = dsu.find(terminals[0]);
        for (int i = 1; i < k; ++i) {
            if (dsu.find(terminals[i]) != r0) return false;
        }
        return true;
    };

    for (const auto& e : cand) {
        if (dsu.unite(e.u, e.v)) {
            chosen.push_back({e.u, e.v});
            total += e.w;
            if (terminals_connected()) break;
        }
    }

    if (!terminals_connected()) {
        // As a fallback, run Kruskal on the whole graph's unique edges in best_w map.
        // This guarantees a spanning tree for the connected component if it exists.
        cand.clear();
        cand.reserve(best_w.size());
        for (const auto& kv : best_w) {
            int u = int(kv.first >> 32);
            int v = int(uint32_t(kv.first));
            cand.push_back({u, v, kv.second});
        }
        std::sort(cand.begin(), cand.end(), [](const WeightedEdge& a, const WeightedEdge& b) {
            if (a.w != b.w) return a.w < b.w;
            if (a.u != b.u) return a.u < b.u;
            return a.v < b.v;
        });
        dsu.init(n);
        chosen.clear();
        total = 0;
        for (const auto& e : cand) {
            if (dsu.unite(e.u, e.v)) {
                chosen.push_back({e.u, e.v});
                total += e.w;
                if (terminals_connected()) break;
            }
        }
        if (!terminals_connected()) {
            std::cout << "0 0\n";
            return 0;
        }
    }

    std::cout << total << " " << chosen.size() << "\n";
    for (auto [u, v] : chosen) {
        std::cout << u << " " << v << "\n";
    }
    return 0;
}
