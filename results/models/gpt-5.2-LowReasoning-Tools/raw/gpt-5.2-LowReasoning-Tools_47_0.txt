@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, shift: u32) -> u32 {
  return (x << shift) | (x >> (32u - shift));
}

fn quarter_round_4(state: ptr<function, array<u32, 16>>, base: u32) {
  var a: u32 = (*state)[base + 0u];
  var b: u32 = (*state)[base + 1u];
  var c: u32 = (*state)[base + 2u];
  var d: u32 = (*state)[base + 3u];

  a = a + b; d = d ^ a; d = rotl32(d, 16u);
  c = c + d; b = b ^ c; b = rotl32(b, 12u);
  a = a + b; d = d ^ a; d = rotl32(d,  8u);
  c = c + d; b = b ^ c; b = rotl32(b,  7u);

  (*state)[base + 0u] = a;
  (*state)[base + 1u] = b;
  (*state)[base + 2u] = c;
  (*state)[base + 3u] = d;
}

fn custom_hash(nonce: u32) -> vec2<u32> {
  var state: array<u32, 16>;

  // state[0..15] = data[0..15]; where data = (baseData XOR nonce)
  // then: for i in 0..15: state[i] ^= (nonce + i)
  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = (baseData[i] ^ nonce) ^ (nonce + i);
  }

  // 8 rounds of quarter-round mixing (ChaCha-like), groups unrolled
  for (var r: u32 = 0u; r < 8u; r = r + 1u) {
    quarter_round_4(&state, 0u);
    quarter_round_4(&state, 4u);
    quarter_round_4(&state, 8u);
    quarter_round_4(&state, 12u);
  }

  // Fold 16 words to 2:
  let h0: u32 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
  let h1: u32 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];
  return vec2<u32>(h0, h1);
}

fn trailing_zeros_64(h0: u32, h1: u32) -> u32 {
  if (h0 != 0u) {
    return countTrailingZeros(h0);
  }
  return 32u + countTrailingZeros(h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  if (gid.x >= params.z) {
    return;
  }

  let difficulty: u32 = params.x;
  let nonce: u32 = params.y + gid.x;

  let h: vec2<u32> = custom_hash(nonce);
  let h0: u32 = h.x;
  let h1: u32 = h.y;

  let tz: u32 = trailing_zeros_64(h0, h1);
  if (tz >= difficulty) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h0;
    result[3] = h1;
  }
}
