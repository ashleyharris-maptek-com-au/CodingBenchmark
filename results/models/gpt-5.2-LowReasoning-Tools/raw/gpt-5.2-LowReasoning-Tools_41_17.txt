%main = OpFunction %void None %func
%label = OpLabel

; load inputs
%wp = OpLoad %v3float %in_worldPos
%n_in = OpLoad %v3float %in_normal
%uv = OpLoad %v2float %in_uv

; normalize normal
%n = OpExtInst %v3float %ext Normalize %n_in

; load light position (vec4) from UBO member 3
%lp_ptr = OpAccessChain %ptr_Uniform_v4float %uboVar %int_3
%lp4 = OpLoad %v4float %lp_ptr
%lp3 = OpVectorShuffle %v3float %lp4 %lp4 0 1 2

; compute light direction
%lvec = OpFSub %v3float %lp3 %wp
%ldir = OpExtInst %v3float %ext Normalize %lvec

; diffuse = max(dot(n, ldir), 0)
%dot = OpExtInst %float %ext Dot %n %ldir
%diffuse = OpExtInst %float %ext FMax %dot %float_0

; shade = 0.4 + 0.6 * diffuse
%diff_scaled = OpFMul %float %float_0_6 %diffuse
%shade = OpFAdd %float %float_0_4 %diff_scaled

; procedural brick pattern
%uvs = OpFMul %v2float %uv %v2_scale

; row = floor(uvs.y)
%uvs_y = OpCompositeExtract %float %uvs 1
%row_f = OpExtInst %float %ext Floor %uvs_y
%row_i = OpConvertFToS %int %row_f

; rowOdd = (row_i & 1)
%row_and = OpBitwiseAnd %int %row_i %int_1
%row_and_u = OpBitcast %uint %row_and
%row_and_f = OpConvertUToF %float %row_and_u
%x_offset = OpFMul %float %row_and_f %float_0_5

; x' = uvs.x + x_offset
%uvs_x = OpCompositeExtract %float %uvs 0
%x_prime = OpFAdd %float %uvs_x %x_offset

; cell = fract(vec2(x_prime, uvs_y))
%cell_in = OpCompositeConstruct %v2float %x_prime %uvs_y
%cell = OpExtInst %v2float %ext Fract %cell_in
%cell_x = OpCompositeExtract %float %cell 0
%cell_y = OpCompositeExtract %float %cell 1

; mortar checks
%one_minus_mx = OpFSub %float %float_1 %float_0_05
%one_minus_my = OpFSub %float %float_1 %float_0_1

%mx_lo = OpFOrdLessThan %bool %cell_x %float_0_05
%mx_hi = OpFOrdGreaterThan %bool %cell_x %one_minus_mx
%my_lo = OpFOrdLessThan %bool %cell_y %float_0_1
%my_hi = OpFOrdGreaterThan %bool %cell_y %one_minus_my

%m_x = OpLogicalOr %bool %mx_lo %mx_hi
%m_y = OpLogicalOr %bool %my_lo %my_hi
%is_mortar = OpLogicalOr %bool %m_x %m_y

; select base color
%base = OpSelect %v3float %is_mortar %v3_mortar %v3_brick

; apply shading
%shade3 = OpCompositeConstruct %v3float %shade %shade %shade
%rgb = OpFMul %v3float %base %shade3
%out = OpCompositeConstruct %v4float %rgb %float_1

OpStore %outColor %out
