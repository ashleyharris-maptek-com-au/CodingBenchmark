const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
const data = input ? JSON.parse(input) : {};

const width = data.width | 0;
const height = data.height | 0;
const gridRows = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget | 0);

if (!width || !height || width * height === 0 || cutBudget === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const n = width * height;
const passable = new Uint8Array(n); // 1 for burnable forest
for (let y = 0; y < height; y++) {
  const row = gridRows[y] || '';
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    if (c === 46) passable[y * width + x] = 1; // '.'
  }
}

const isTown = new Uint8Array(n);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0, y = t.y | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) isTown[y * width + x] = 1;
}

const isIgnition = new Uint8Array(n);
for (let i = 0; i < ignitions.length; i++) {
  const ig = ignitions[i] || [];
  const x = ig[0] | 0, y = ig[1] | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) isIgnition[y * width + x] = 1;
}

// Multi-source BFS from ignitions over passable cells.
const dist = new Int32Array(n);
dist.fill(-1);
const queue = new Int32Array(n);
let qh = 0, qt = 0;

for (let i = 0; i < ignitions.length; i++) {
  const ig = ignitions[i] || [];
  const x = ig[0] | 0, y = ig[1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const idx = y * width + x;
  if (!passable[idx]) continue;
  if (dist[idx] !== -1) continue;
  dist[idx] = 0;
  queue[qt++] = idx;
}

if (qt === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const bfsOrder = new Int32Array(n);
let ordLen = 0;
while (qh < qt) {
  const v = queue[qh++];
  bfsOrder[ordLen++] = v;
  const dv = dist[v] + 1;
  const x = v % width;
  const y = (v / width) | 0;

  // left
  if (x > 0) {
    const u = v - 1;
    if (passable[u] && dist[u] === -1) {
      dist[u] = dv;
      queue[qt++] = u;
    }
  }
  // right
  if (x + 1 < width) {
    const u = v + 1;
    if (passable[u] && dist[u] === -1) {
      dist[u] = dv;
      queue[qt++] = u;
    }
  }
  // up
  if (y > 0) {
    const u = v - width;
    if (passable[u] && dist[u] === -1) {
      dist[u] = dv;
      queue[qt++] = u;
    }
  }
  // down
  if (y + 1 < height) {
    const u = v + width;
    if (passable[u] && dist[u] === -1) {
      dist[u] = dv;
      queue[qt++] = u;
    }
  }
}

// Count number of parents on the shortest-path DAG (neighbors with dist = d-1).
const parentCount = new Uint8Array(n);
for (let oi = 0; oi < ordLen; oi++) {
  const v = bfsOrder[oi];
  const d = dist[v];
  if (d === 0) {
    parentCount[v] = 1;
    continue;
  }
  let pc = 0;
  const x = v % width;
  const y = (v / width) | 0;
  if (x > 0 && dist[v - 1] === d - 1) pc++;
  if (x + 1 < width && dist[v + 1] === d - 1) pc++;
  if (y > 0 && dist[v - width] === d - 1) pc++;
  if (y + 1 < height && dist[v + width] === d - 1) pc++;
  parentCount[v] = pc;
}

// Back-propagate town "value demand" along the shortest-path DAG to estimate influential cut cells.
const demand = new Float64Array(n);
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  const x = t.x | 0, y = t.y | 0;
  const val = +t.value || 0;
  if (val <= 0) continue;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const idx = y * width + x;
  if (!passable[idx]) continue;
  if (dist[idx] < 0) continue; // already safe (unreachable from fire)
  demand[idx] += val;
}

for (let oi = ordLen - 1; oi >= 0; oi--) {
  const v = bfsOrder[oi];
  const d = dist[v];
  if (d <= 0) continue;
  const dv = demand[v];
  if (dv === 0) continue;
  const pc = parentCount[v] | 0;
  if (pc <= 0) continue;
  const share = dv / pc;
  const x = v % width;
  const y = (v / width) | 0;
  if (x > 0) {
    const u = v - 1;
    if (dist[u] === d - 1) demand[u] += share;
  }
  if (x + 1 < width) {
    const u = v + 1;
    if (dist[u] === d - 1) demand[u] += share;
  }
  if (y > 0) {
    const u = v - width;
    if (dist[u] === d - 1) demand[u] += share;
  }
  if (y + 1 < height) {
    const u = v + width;
    if (dist[u] === d - 1) demand[u] += share;
  }
}

// Pick top-scoring candidate cells using a small min-heap.
function heapPush(heap, item) {
  heap.push(item);
  let i = heap.length - 1;
  while (i > 0) {
    const p = (i - 1) >> 1;
    if (heap[p][0] <= heap[i][0]) break;
    const tmp = heap[p];
    heap[p] = heap[i];
    heap[i] = tmp;
    i = p;
  }
}

function heapReplaceTop(heap, item) {
  heap[0] = item;
  let i = 0;
  while (true) {
    let l = i * 2 + 1;
    if (l >= heap.length) break;
    let r = l + 1;
    let m = r < heap.length && heap[r][0] < heap[l][0] ? r : l;
    if (heap[i][0] <= heap[m][0]) break;
    const tmp = heap[i];
    heap[i] = heap[m];
    heap[m] = tmp;
    i = m;
  }
}

const heap = [];
const heapMax = Math.min(n, Math.max(2000, cutBudget * 30));

for (let oi = 0; oi < ordLen; oi++) {
  const idx = bfsOrder[oi];
  if (!passable[idx]) continue;
  if (dist[idx] <= 0) continue; // avoid cutting ignitions and dist<0
  if (isTown[idx] || isIgnition[idx]) continue;
  const dv = demand[idx];
  if (dv <= 0) continue;
  const score = dv / (1 + dist[idx]);
  if (heap.length < heapMax) heapPush(heap, [score, idx]);
  else if (score > heap[0][0]) heapReplaceTop(heap, [score, idx]);
}

heap.sort((a, b) => b[0] - a[0]);

const firebreaks = [];
const picked = new Uint8Array(n);
for (let i = 0; i < heap.length && firebreaks.length < cutBudget; i++) {
  const idx = heap[i][1];
  if (picked[idx]) continue;
  picked[idx] = 1;
  const x = idx % width;
  const y = (idx / width) | 0;
  firebreaks.push([x, y]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
