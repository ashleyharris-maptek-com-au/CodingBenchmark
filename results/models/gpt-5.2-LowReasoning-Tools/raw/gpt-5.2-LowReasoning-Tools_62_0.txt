// Wildfire Firebreak Planner - Node.js TypeScript (no deps)

declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Town = { id: string | number; x: number; y: number; value: number };

type Result = { firebreaks: [number, number][] };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: [number, number][] = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const n = width * height;

function idx(x: number, y: number): number {
  return y * width + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < width && y >= 0 && y < height;
}

// 1 = forest (burnable/cuttable), 0 = water/rock
const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    const c = row.charCodeAt(x);
    // '.' => 46
    burnable[idx(x, y)] = c === 46 ? 1 : 0;
  }
}

const isIgnition = new Uint8Array(n);
for (const [x, y] of ignitions) {
  if (!inBounds(x, y)) continue;
  isIgnition[idx(x, y)] = 1;
}

const isTown = new Uint8Array(n);
const townIdxs: number[] = [];
const townValues: number[] = [];
for (const t of towns) {
  const x = t.x | 0;
  const y = t.y | 0;
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  isTown[id] = 1;
  townIdxs.push(id);
  townValues.push(+t.value || 0);
}

const blocked = new Uint8Array(n); // our cuts

function bfs(): { dist: Int32Array; parent: Int32Array } {
  const dist = new Int32Array(n);
  const parent = new Int32Array(n);
  dist.fill(-1);
  parent.fill(-1);

  const q = new Int32Array(n);
  let head = 0;
  let tail = 0;

  for (const [x, y] of ignitions) {
    if (!inBounds(x, y)) continue;
    const s = idx(x, y);
    if (burnable[s] === 0) continue;
    if (blocked[s]) continue;
    if (dist[s] !== -1) continue;
    dist[s] = 0;
    parent[s] = -1;
    q[tail++] = s;
  }

  while (head < tail) {
    const cur = q[head++];
    const cd = dist[cur];
    const x = cur % width;
    const y = (cur / width) | 0;

    // 4-neighbors
    if (x > 0) {
      const ni = cur - 1;
      if (dist[ni] === -1 && burnable[ni] && !blocked[ni]) {
        dist[ni] = cd + 1;
        parent[ni] = cur;
        q[tail++] = ni;
      }
    }
    if (x + 1 < width) {
      const ni = cur + 1;
      if (dist[ni] === -1 && burnable[ni] && !blocked[ni]) {
        dist[ni] = cd + 1;
        parent[ni] = cur;
        q[tail++] = ni;
      }
    }
    if (y > 0) {
      const ni = cur - width;
      if (dist[ni] === -1 && burnable[ni] && !blocked[ni]) {
        dist[ni] = cd + 1;
        parent[ni] = cur;
        q[tail++] = ni;
      }
    }
    if (y + 1 < height) {
      const ni = cur + width;
      if (dist[ni] === -1 && burnable[ni] && !blocked[ni]) {
        dist[ni] = cd + 1;
        parent[ni] = cur;
        q[tail++] = ni;
      }
    }
  }

  return { dist, parent };
}

function pickBestCut(dist: Int32Array, parent: Int32Array): number {
  const weights = new Map<number, number>();

  for (let i = 0; i < townIdxs.length; i++) {
    const tIdx = townIdxs[i];
    const val = townValues[i];
    if (val <= 0) continue;
    if (!burnable[tIdx] || blocked[tIdx]) continue;

    const td = dist[tIdx];
    if (td < 0) continue;

    let cur = tIdx;
    let steps = 0;
    while (cur !== -1 && steps <= td + 2) {
      if (burnable[cur] && !blocked[cur] && !isIgnition[cur] && !isTown[cur]) {
        const cd = dist[cur];
        if (cd >= 0) {
          const w = val / (cd + 1);
          weights.set(cur, (weights.get(cur) || 0) + w);
        }
      }
      cur = parent[cur];
      steps++;
    }
  }

  let best = -1;
  let bestW = -1;
  for (const [cell, w] of weights) {
    if (w > bestW) {
      bestW = w;
      best = cell;
    }
  }
  return best;
}

function cellToXY(cell: number): [number, number] {
  const y = (cell / width) | 0;
  const x = cell - y * width;
  return [x, y];
}

const firebreaks: [number, number][] = [];

if (cutBudget > 0 && ignitions.length > 0 && townIdxs.length > 0) {
  const iterativeLimit = Math.min(cutBudget, 200);

  for (let step = 0; step < iterativeLimit; step++) {
    const { dist, parent } = bfs();
    const best = pickBestCut(dist, parent);
    if (best === -1) break;
    blocked[best] = 1;
    firebreaks.push(cellToXY(best));
  }

  const remaining = cutBudget - firebreaks.length;
  if (remaining > 0) {
    const { dist, parent } = bfs();

    const weights = new Map<number, number>();
    for (let i = 0; i < townIdxs.length; i++) {
      const tIdx = townIdxs[i];
      const val = townValues[i];
      if (val <= 0) continue;
      if (!burnable[tIdx] || blocked[tIdx]) continue;
      const td = dist[tIdx];
      if (td < 0) continue;

      let cur = tIdx;
      let steps = 0;
      while (cur !== -1 && steps <= td + 2) {
        if (burnable[cur] && !blocked[cur] && !isIgnition[cur] && !isTown[cur]) {
          const cd = dist[cur];
          if (cd >= 0) {
            const w = val / (cd + 1);
            weights.set(cur, (weights.get(cur) || 0) + w);
          }
        }
        cur = parent[cur];
        steps++;
      }
    }

    const entries = Array.from(weights.entries())
      .filter(([cell]) => !blocked[cell] && burnable[cell])
      .sort((a, b) => b[1] - a[1]);

    for (let i = 0; i < entries.length && firebreaks.length < cutBudget; i++) {
      const cell = entries[i][0];
      if (blocked[cell]) continue;
      blocked[cell] = 1;
      firebreaks.push(cellToXY(cell));
    }
  }
}

const out: Result = { firebreaks };
process.stdout.write(JSON.stringify(out));
