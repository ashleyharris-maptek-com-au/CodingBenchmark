#!/usr/bin/env python3
import sys
import math
from typing import List, Tuple, Optional


def dist(a: Tuple[float, float], b: Tuple[float, float]) -> float:
    dx = a[0] - b[0]
    dy = a[1] - b[1]
    return math.hypot(dx, dy)


def route_distance(route: List[int], coords: List[Tuple[float, float]], depot: Tuple[float, float]) -> float:
    if not route:
        return 0.0
    total = dist(depot, coords[route[0]])
    for i in range(len(route) - 1):
        total += dist(coords[route[i]], coords[route[i + 1]])
    total += dist(coords[route[-1]], depot)
    return total


def two_opt(route: List[int], coords: List[Tuple[float, float]], depot: Tuple[float, float]) -> List[int]:
    # Standard 2-opt on a single route.
    n = len(route)
    if n < 4:
        return route

    # Precompute pair distances on demand via function; keep simple.
    improved = True
    best = route
    best_len = route_distance(best, coords, depot)

    while improved:
        improved = False
        for i in range(0, n - 2):
            for k in range(i + 1, n - 1):
                # edges: (prev_i -> i) and (k -> next_k)
                a = depot if i == 0 else coords[best[i - 1]]
                b = coords[best[i]]
                c = coords[best[k]]
                d = depot if k == n - 1 else coords[best[k + 1]]

                before = dist(a, b) + dist(c, d)
                after = dist(a, c) + dist(b, d)
                if after + 1e-12 < before:
                    cand = best[:i] + list(reversed(best[i:k + 1])) + best[k + 1:]
                    best = cand
                    best_len = best_len - before + after
                    improved = True
                    break
            if improved:
                break
    return best


def parse_input() -> Tuple[int, int, int, Tuple[float, float], List[Tuple[float, float]], List[int]]:
    data = sys.stdin.read().strip().split()
    if not data:
        raise SystemExit(0)
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = int(next(it))
    depot = (float(next(it)), float(next(it)))
    coords = []
    demands = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = int(next(it))
        coords.append((x, y))
        demands.append(d)
    return n, k, cap, depot, coords, demands


def clarke_wright(n: int, k: int, cap: int, depot: Tuple[float, float], coords: List[Tuple[float, float]], demands: List[int]) -> List[List[int]]:
    # Start with one route per customer.
    routes: List[List[int]] = [[i] for i in range(n)]
    route_demand = [demands[i] for i in range(n)]

    # Track endpoints for quick merges.
    # route_idx_of_customer is updated when routes merge.
    route_idx_of_customer = list(range(n))

    # Savings list: (saving, i, j)
    savings = []
    for i in range(n):
        di = dist(depot, coords[i])
        for j in range(i + 1, n):
            dj = dist(depot, coords[j])
            sij = di + dj - dist(coords[i], coords[j])
            savings.append((sij, i, j))
    savings.sort(reverse=True)

    def endpoints(r: List[int]) -> Tuple[int, int]:
        return r[0], r[-1]

    # Helper to attempt merging routes that contain i and j at endpoints.
    def try_merge(i: int, j: int) -> bool:
        ri = route_idx_of_customer[i]
        rj = route_idx_of_customer[j]
        if ri == rj:
            return False
        if routes[ri] is None or routes[rj] is None:
            return False
        r1 = routes[ri]
        r2 = routes[rj]
        if r1 is None or r2 is None:
            return False
        if route_demand[ri] + route_demand[rj] > cap:
            return False

        a1, b1 = endpoints(r1)
        a2, b2 = endpoints(r2)

        # Need i and j to be at ends of their routes, connect appropriately.
        merged: Optional[List[int]] = None
        if b1 == i and a2 == j:
            merged = r1 + r2
        elif b1 == i and b2 == j:
            merged = r1 + list(reversed(r2))
        elif a1 == i and a2 == j:
            merged = list(reversed(r1)) + r2
        elif a1 == i and b2 == j:
            merged = list(reversed(r1)) + list(reversed(r2))
        else:
            return False

        # Apply merge: keep ri, delete rj.
        routes[ri] = merged
        route_demand[ri] = route_demand[ri] + route_demand[rj]
        routes[rj] = None
        route_demand[rj] = 0
        for c in merged:
            route_idx_of_customer[c] = ri
        return True

    # First pass merges by savings.
    for _, i, j in savings:
        # Stop early if already <= K routes.
        if sum(1 for r in routes if r is not None) <= k:
            break
        try_merge(i, j)

    # If still too many routes, force merges with minimal added distance (still respecting capacity).
    def best_forced_merge() -> Optional[Tuple[int, int, List[int]]]:
        active = [idx for idx, r in enumerate(routes) if r is not None]
        best_delta = float('inf')
        best_pair = None
        for a in range(len(active)):
            ra = active[a]
            r1 = routes[ra]
            if r1 is None:
                continue
            for b in range(a + 1, len(active)):
                rb = active[b]
                r2 = routes[rb]
                if r2 is None:
                    continue
                if route_demand[ra] + route_demand[rb] > cap:
                    continue

                # Evaluate 4 ways to connect endpoints.
                a1, b1 = r1[0], r1[-1]
                a2, b2 = r2[0], r2[-1]

                # Current end costs (without depot connections simplified by delta evaluation).
                # We approximate delta by change in inter-route connection vs two depot legs:
                # connect end of r1 to start of r2, etc.
                options = []
                options.append((dist(coords[b1], coords[a2]) - dist(depot, coords[b1]) - dist(depot, coords[a2]), r1 + r2))
                options.append((dist(coords[b1], coords[b2]) - dist(depot, coords[b1]) - dist(depot, coords[b2]), r1 + list(reversed(r2))))
                options.append((dist(coords[a1], coords[a2]) - dist(depot, coords[a1]) - dist(depot, coords[a2]), list(reversed(r1)) + r2))
                options.append((dist(coords[a1], coords[b2]) - dist(depot, coords[a1]) - dist(depot, coords[b2]), list(reversed(r1)) + list(reversed(r2))))

                for delta, merged in options:
                    if delta < best_delta:
                        best_delta = delta
                        best_pair = (ra, rb, merged)
        return best_pair

    while sum(1 for r in routes if r is not None) > k:
        cand = best_forced_merge()
        if cand is None:
            break
        ra, rb, merged = cand
        routes[ra] = merged
        route_demand[ra] = route_demand[ra] + route_demand[rb]
        routes[rb] = None
        route_demand[rb] = 0
        for c in merged:
            route_idx_of_customer[c] = ra

    final_routes = [r for r in routes if r is not None]
    return final_routes


def sweep_fallback(n: int, k: int, cap: int, depot: Tuple[float, float], coords: List[Tuple[float, float]], demands: List[int]) -> List[List[int]]:
    # Polar-angle sweep, then pack into routes by capacity.
    angles = []
    for i in range(n):
        dx = coords[i][0] - depot[0]
        dy = coords[i][1] - depot[1]
        angles.append((math.atan2(dy, dx), i))
    angles.sort()

    routes: List[List[int]] = []
    current: List[int] = []
    load = 0
    for _, i in angles:
        d = demands[i]
        if current and load + d > cap:
            routes.append(current)
            current = []
            load = 0
        current.append(i)
        load += d
    if current:
        routes.append(current)

    # If we have more than K routes, try to merge adjacent routes if possible.
    changed = True
    while changed and len(routes) > k:
        changed = False
        best = None
        best_increase = float('inf')
        for a in range(len(routes)):
            for b in range(a + 1, len(routes)):
                la = sum(demands[i] for i in routes[a])
                lb = sum(demands[i] for i in routes[b])
                if la + lb > cap:
                    continue
                inc = route_distance(routes[a] + routes[b], coords, depot) - route_distance(routes[a], coords, depot) - route_distance(routes[b], coords, depot)
                if inc < best_increase:
                    best_increase = inc
                    best = (a, b)
        if best is not None:
            a, b = best
            routes[a] = routes[a] + routes[b]
            del routes[b]
            changed = True

    return routes


def main() -> None:
    n, k, cap, depot, coords, demands = parse_input()

    routes = clarke_wright(n, k, cap, depot, coords, demands)
    # Ensure feasibility and <= K; if not, fallback.
    if len(routes) > k:
        routes = sweep_fallback(n, k, cap, depot, coords, demands)

    # If still too many routes (should not happen on feasible instances), just truncate by rebuilding greedily.
    if len(routes) > k:
        # Greedy nearest insertion into K routes.
        unassigned = set(range(n))
        routes = []
        for _ in range(k):
            if not unassigned:
                break
            route = []
            load = 0
            cur = depot
            while True:
                best = None
                best_d = float('inf')
                for c in list(unassigned):
                    if load + demands[c] > cap:
                        continue
                    dcur = dist(cur, coords[c])
                    if dcur < best_d:
                        best_d = dcur
                        best = c
                if best is None:
                    break
                route.append(best)
                unassigned.remove(best)
                load += demands[best]
                cur = coords[best]
            routes.append(route)
        if unassigned:
            # As last resort (should not happen), place remaining customers one-per-route beyond K.
            for c in sorted(unassigned):
                routes.append([c])

    # Improve each route with 2-opt.
    improved_routes = []
    for r in routes:
        improved_routes.append(two_opt(r, coords, depot))
    routes = improved_routes

    # Validate coverage and capacity; if anything is off, rebuild with a simple sweep heuristic.
    seen = [0] * n
    for r in routes:
        for i in r:
            seen[i] += 1

    if any(v != 1 for v in seen) or any(sum(demands[i] for i in r) > cap for r in routes) or len(routes) > k:
        routes = sweep_fallback(n, k, cap, depot, coords, demands)
        routes = [two_opt(r, coords, depot) for r in routes]

    # Final compute total distance.
    total = sum(route_distance(r, coords, depot) for r in routes)

    out = []
    out.append(f"{total:.6f}")
    out.append(str(len(routes)))
    for r in routes:
        out.append(" ".join(str(i) for i in r))
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
