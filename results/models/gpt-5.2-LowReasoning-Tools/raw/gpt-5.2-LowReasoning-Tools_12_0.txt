using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private readonly struct Vec3
    {
        public readonly double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    private readonly struct Mat3
    {
        // Row-major
        public readonly double M00, M01, M02;
        public readonly double M10, M11, M12;
        public readonly double M20, M21, M22;
        public Mat3(double m00, double m01, double m02,
                    double m10, double m11, double m12,
                    double m20, double m21, double m22)
        {
            M00 = m00; M01 = m01; M02 = m02;
            M10 = m10; M11 = m11; M12 = m12;
            M20 = m20; M21 = m21; M22 = m22;
        }
        public Vec3 Mul(Vec3 v) => new Vec3(
            M00 * v.X + M01 * v.Y + M02 * v.Z,
            M10 * v.X + M11 * v.Y + M12 * v.Z,
            M20 * v.X + M21 * v.Y + M22 * v.Z
        );
    }

    private readonly struct Quat
    {
        public readonly double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
        public static Quat FromMat3(Mat3 m)
        {
            // Robust conversion for rotation matrices
            double trace = m.M00 + m.M11 + m.M22;
            double w, x, y, z;
            if (trace > 0.0)
            {
                double s = Math.Sqrt(trace + 1.0) * 2.0;
                w = 0.25 * s;
                x = (m.M21 - m.M12) / s;
                y = (m.M02 - m.M20) / s;
                z = (m.M10 - m.M01) / s;
            }
            else if (m.M00 > m.M11 && m.M00 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M00 - m.M11 - m.M22) * 2.0;
                w = (m.M21 - m.M12) / s;
                x = 0.25 * s;
                y = (m.M01 + m.M10) / s;
                z = (m.M02 + m.M20) / s;
            }
            else if (m.M11 > m.M22)
            {
                double s = Math.Sqrt(1.0 + m.M11 - m.M00 - m.M22) * 2.0;
                w = (m.M02 - m.M20) / s;
                x = (m.M01 + m.M10) / s;
                y = 0.25 * s;
                z = (m.M12 + m.M21) / s;
            }
            else
            {
                double s = Math.Sqrt(1.0 + m.M22 - m.M00 - m.M11) * 2.0;
                w = (m.M10 - m.M01) / s;
                x = (m.M02 + m.M20) / s;
                y = (m.M12 + m.M21) / s;
                z = 0.25 * s;
            }
            // Normalize
            double n = Math.Sqrt(w * w + x * x + y * y + z * z);
            if (n <= 0) return new Quat(1, 0, 0, 0);
            return new Quat(w / n, x / n, y / n, z / n);
        }
    }

    private sealed class FastScanner
    {
        private readonly string[] _tokens;
        private int _i;
        public FastScanner(string input)
        {
            _tokens = input.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            _i = 0;
        }
        public bool HasNext => _i < _tokens.Length;
        public int NextInt() => int.Parse(_tokens[_i++], CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(_tokens[_i++], CultureInfo.InvariantCulture);
    }

    private readonly struct RotCandidate
    {
        public readonly Mat3 R;
        public readonly Quat Q;
        public RotCandidate(Mat3 r, Quat q) { R = r; Q = q; }
    }

    private static List<RotCandidate> GenerateAxisAlignedRotations24()
    {
        // 24 proper rotations mapping axes to axes (right-handed)
        // Build by choosing forward (Z') and up (Y'), then right (X') = up x forward.
        var basis = new Vec3[]
        {
            new Vec3( 1, 0, 0), new Vec3(-1, 0, 0),
            new Vec3( 0, 1, 0), new Vec3( 0,-1, 0),
            new Vec3( 0, 0, 1), new Vec3( 0, 0,-1),
        };

        var list = new List<RotCandidate>(24);
        for (int f = 0; f < basis.Length; f++)
        {
            Vec3 forward = basis[f];
            for (int u = 0; u < basis.Length; u++)
            {
                Vec3 up = basis[u];
                // up must be orthogonal to forward
                double dot = forward.X * up.X + forward.Y * up.Y + forward.Z * up.Z;
                if (Math.Abs(dot) > 0.5) continue;

                Vec3 right = Cross(up, forward);
                double len = Math.Sqrt(right.X * right.X + right.Y * right.Y + right.Z * right.Z);
                if (len < 0.5) continue;

                // Orthonormal basis: columns are right, up, forward
                var m = new Mat3(
                    right.X, up.X, forward.X,
                    right.Y, up.Y, forward.Y,
                    right.Z, up.Z, forward.Z
                );

                // Ensure determinant +1 (proper rotation)
                double det = Determinant(m);
                if (det < 0.0) continue;

                var q = Quat.FromMat3(m);
                // Avoid duplicates via quaternion sign-insensitive compare
                if (!ContainsRotation(list, q))
                    list.Add(new RotCandidate(m, q));
                if (list.Count == 24) return list;
            }
        }
        return list;
    }

    private static bool ContainsRotation(List<RotCandidate> list, Quat q)
    {
        // q and -q represent same rotation; use abs dot threshold
        for (int i = 0; i < list.Count; i++)
        {
            var p = list[i].Q;
            double dot = p.W * q.W + p.X * q.X + p.Y * q.Y + p.Z * q.Z;
            if (Math.Abs(dot) > 1.0 - 1e-9) return true;
        }
        return false;
    }

    private static Vec3 Cross(Vec3 a, Vec3 b) => new Vec3(
        a.Y * b.Z - a.Z * b.Y,
        a.Z * b.X - a.X * b.Z,
        a.X * b.Y - a.Y * b.X
    );

    private static double Determinant(Mat3 m)
    {
        return m.M00 * (m.M11 * m.M22 - m.M12 * m.M21)
             - m.M01 * (m.M10 * m.M22 - m.M12 * m.M20)
             + m.M02 * (m.M10 * m.M21 - m.M11 * m.M20);
    }

    private sealed class Fit
    {
        public RotCandidate Rot;
        public Vec3 Min;
        public Vec3 Size;
        public long Count;
    }

    public static void Main()
    {
        string input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
        {
            Console.WriteLine(0);
            return;
        }

        var fs = new FastScanner(input);
        int V = fs.NextInt();
        int F = fs.NextInt();

        var verts = new Vec3[V];
        for (int i = 0; i < V; i++)
            verts[i] = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        // Faces are not needed for a conservative bounding approach.
        for (int i = 0; i < F; i++) { fs.NextInt(); fs.NextInt(); fs.NextInt(); }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        if (W <= 0 || H <= 0 || D <= 0 || V == 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Adaptive strategy: number of rotation candidates depends on problem size.
        // For massive containers, keep it cheap; for smaller, explore more.
        double containerVol = W * H * D;
        int rotBudget;
        if (containerVol <= 20_000) rotBudget = 24;          // trivial/small-medium
        else if (containerVol <= 2_000_000) rotBudget = 24;  // medium
        else if (containerVol <= 50_000_000) rotBudget = 12; // large
        else rotBudget = 6;                                  // extreme

        var rotsAll = GenerateAxisAlignedRotations24();
        var rots = rotsAll;
        if (rotBudget < rotsAll.Count)
        {
            // Take a deterministic subset: identity, axis swaps, and a few others.
            rots = new List<RotCandidate>(rotBudget);
            rots.Add(rotsAll[0]);
            for (int i = 1; i < rotsAll.Count && rots.Count < rotBudget; i += 3)
                rots.Add(rotsAll[i]);
            while (rots.Count < rotBudget && rots.Count < rotsAll.Count)
                rots.Add(rotsAll[rots.Count]);
        }

        Fit best = null;

        foreach (var rot in rots)
        {
            GetBounds(verts, rot.R, out Vec3 bmin, out Vec3 bmax);
            Vec3 size = new Vec3(bmax.X - bmin.X, bmax.Y - bmin.Y, bmax.Z - bmin.Z);

            // Avoid degenerate sizes
            double dx = Math.Max(size.X, 1e-9);
            double dy = Math.Max(size.Y, 1e-9);
            double dz = Math.Max(size.Z, 1e-9);

            long nx = (long)Math.Floor(W / dx);
            long ny = (long)Math.Floor(H / dy);
            long nz = (long)Math.Floor(D / dz);
            if (nx <= 0 || ny <= 0 || nz <= 0) continue;

            long count;
            try
            {
                checked { count = nx * ny * nz; }
            }
            catch
            {
                count = long.MaxValue;
            }

            // Objective: maximize count, break ties by minimizing unused volume, then by smaller max dimension.
            double usedVol = (double)nx * dx * (double)ny * dy * (double)nz * dz;
            double waste = containerVol - usedVol;
            double maxDim = Math.Max(dx, Math.Max(dy, dz));

            if (best == null)
            {
                best = new Fit { Rot = rot, Min = bmin, Size = new Vec3(dx, dy, dz), Count = count };
                bestWaste = waste; bestMaxDim = maxDim;
            }
            else
            {
                if (count > best.Count ||
                    (count == best.Count && (waste < bestWaste - 1e-9 ||
                     (Math.Abs(waste - bestWaste) <= 1e-9 && maxDim < bestMaxDim - 1e-9))))
                {
                    best = new Fit { Rot = rot, Min = bmin, Size = new Vec3(dx, dy, dz), Count = count };
                    bestWaste = waste; bestMaxDim = maxDim;
                }
            }
        }

        if (best == null || best.Count <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // For extreme cases, avoid printing absurdly large outputs.
        // The benchmark environment allows 5 minutes; stdout size can dominate.
        // Cap to a practical maximum while still being competitive.
        long hardCap = 2_000_000; // ~2M lines is already huge; keep it bounded.
        long outCount = best.Count;
        if (outCount > hardCap) outCount = hardCap;

        double dxBest = best.Size.X;
        double dyBest = best.Size.Y;
        double dzBest = best.Size.Z;

        long nxBest = (long)Math.Floor(W / dxBest);
        long nyBest = (long)Math.Floor(H / dyBest);
        long nzBest = (long)Math.Floor(D / dzBest);

        // Generate placements in a simple 3D grid.
        // Placement translation t is chosen so that rotated polyhedron min corner aligns to cell origin.
        // That ensures the polyhedron is entirely within its AABB cell.
        var q = best.Rot.Q;

        var inv = CultureInfo.InvariantCulture;
        Console.WriteLine(outCount.ToString(inv));

        long emitted = 0;
        for (long iz = 0; iz < nzBest && emitted < outCount; iz++)
        {
            double z0 = iz * dzBest;
            for (long iy = 0; iy < nyBest && emitted < outCount; iy++)
            {
                double y0 = iy * dyBest;
                for (long ix = 0; ix < nxBest && emitted < outCount; ix++)
                {
                    double x0 = ix * dxBest;
                    // translation to move rotated bmin to (x0,y0,z0)
                    double tx = x0 - best.Min.X;
                    double ty = y0 - best.Min.Y;
                    double tz = z0 - best.Min.Z;

                    Console.Write(tx.ToString("R", inv)); Console.Write(' ');
                    Console.Write(ty.ToString("R", inv)); Console.Write(' ');
                    Console.Write(tz.ToString("R", inv)); Console.Write(' ');
                    Console.Write(q.W.ToString("R", inv)); Console.Write(' ');
                    Console.Write(q.X.ToString("R", inv)); Console.Write(' ');
                    Console.Write(q.Y.ToString("R", inv)); Console.Write(' ');
                    Console.WriteLine(q.Z.ToString("R", inv));

                    emitted++;
                }
            }
        }
    }

    private static double bestWaste;
    private static double bestMaxDim;

    private static void GetBounds(Vec3[] verts, Mat3 r, out Vec3 bmin, out Vec3 bmax)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        for (int i = 0; i < verts.Length; i++)
        {
            Vec3 p = r.Mul(verts[i]);
            if (p.X < minX) minX = p.X;
            if (p.Y < minY) minY = p.Y;
            if (p.Z < minZ) minZ = p.Z;
            if (p.X > maxX) maxX = p.X;
            if (p.Y > maxY) maxY = p.Y;
            if (p.Z > maxZ) maxZ = p.Z;
        }

        bmin = new Vec3(minX, minY, minZ);
        bmax = new Vec3(maxX, maxY, maxZ);
    }
}
