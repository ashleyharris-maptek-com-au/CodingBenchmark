using System;
using System.Buffers;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len, _pos;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private struct Cut
    {
        public int Len;
        public int Idx;
    }

    private sealed class Fenwick
    {
        private readonly int[] _tree;
        public int Size { get; }

        public Fenwick(int n)
        {
            Size = n;
            _tree = new int[n + 1];
        }

        public void Add(int i, int delta)
        {
            for (int x = i; x <= Size; x += x & -x) _tree[x] += delta;
        }

        public int Sum(int i)
        {
            int s = 0;
            for (int x = i; x > 0; x -= x & -x) s += _tree[x];
            return s;
        }

        // Returns smallest idx in [1..Size] with prefix sum >= k (assumes 1 <= k <= total sum).
        public int LowerBound(int k)
        {
            int idx = 0;
            int bit = 1;
            while ((bit << 1) <= Size) bit <<= 1;
            for (; bit != 0; bit >>= 1)
            {
                int next = idx + bit;
                if (next <= Size && _tree[next] < k)
                {
                    idx = next;
                    k -= _tree[next];
                }
            }
            return idx + 1;
        }
    }

    private sealed class TreapMap
    {
        private sealed class Node
        {
            public int Key;
            public int Pri;
            public Node? L;
            public Node? R;
            public Stack<int> Stocks = new Stack<int>();
        }

        private Node? _root;
        private uint _rng = 2463534242u;

        private int NextPri()
        {
            // xorshift32
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return (int)(x & 0x7fffffff);
        }

        private static void RotateRight(ref Node p)
        {
            Node q = p.L!;
            p.L = q.R;
            q.R = p;
            p = q;
        }

        private static void RotateLeft(ref Node p)
        {
            Node q = p.R!;
            p.R = q.L;
            q.L = p;
            p = q;
        }

        public void AddStock(int key, int stockId)
        {
            _root = Insert(_root, key, stockId);
        }

        private Node Insert(Node? n, int key, int stockId)
        {
            if (n == null)
            {
                var node = new Node { Key = key, Pri = NextPri() };
                node.Stocks.Push(stockId);
                return node;
            }
            if (key == n.Key)
            {
                n.Stocks.Push(stockId);
                return n;
            }
            if (key < n.Key)
            {
                n.L = Insert(n.L, key, stockId);
                if (n.L!.Pri > n.Pri) RotateRight(ref n);
            }
            else
            {
                n.R = Insert(n.R, key, stockId);
                if (n.R!.Pri > n.Pri) RotateLeft(ref n);
            }
            return n;
        }

        public bool TryPopLowerBound(int minKey, out int key, out int stockId)
        {
            Node? cur = _root;
            Node? best = null;
            while (cur != null)
            {
                if (cur.Key >= minKey)
                {
                    best = cur;
                    cur = cur.L;
                }
                else cur = cur.R;
            }
            if (best == null)
            {
                key = 0;
                stockId = -1;
                return false;
            }
            key = best.Key;
            stockId = best.Stocks.Pop();
            if (best.Stocks.Count == 0) _root = Delete(_root, best.Key);
            return true;
        }

        private Node? Delete(Node? n, int key)
        {
            if (n == null) return null;
            if (key < n.Key) { n.L = Delete(n.L, key); return n; }
            if (key > n.Key) { n.R = Delete(n.R, key); return n; }

            // remove this node by rotating down
            if (n.L == null) return n.R;
            if (n.R == null) return n.L;
            if (n.L.Pri > n.R.Pri)
            {
                RotateRight(ref n);
                n.R = Delete(n.R, key);
            }
            else
            {
                RotateLeft(ref n);
                n.L = Delete(n.L, key);
            }
            return n;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int stockLen = fs.NextInt();
        if (n < 0) return;

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            cuts[i] = new Cut { Len = len, Idx = i };
            if (len > stockLen)
            {
                // Invalid input per constraints; still produce something predictable.
                // Put each cut alone (will "fit" only if equal); otherwise ignore.
            }
        }

        Array.Sort(cuts, (a, b) => b.Len.CompareTo(a.Len));

        // Store assignments
        var stockCuts = new List<List<int>>(1024);
        var stockRemaining = new List<int>(1024);

        // Heuristic: Best-Fit Decreasing using either Fenwick+stacks (when stockLen small)
        // or treap map (when stockLen large).
        const int FenwickMax = 2_000_000;
        bool useFenwick = stockLen <= FenwickMax;

        Stack<int>?[]? buckets = null;
        Fenwick? bit = null;
        TreapMap? treap = null;

        if (useFenwick)
        {
            buckets = new Stack<int>?[stockLen + 1];
            bit = new Fenwick(stockLen + 1); // index rem+1
        }
        else
        {
            treap = new TreapMap();
        }

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Len;
            int origIdx = cuts[i].Idx;

            int chosenStock = -1;
            int rem = 0;

            if (len <= stockLen)
            {
                if (useFenwick)
                {
                    int need = len;
                    // Query any rem >= need => rem+1 in [need+1 .. stockLen+1]
                    int prefix = bit!.Sum(need); // sums indices 1..need
                    int total = bit.Sum(stockLen + 1);
                    if (prefix < total)
                    {
                        int idx = bit.LowerBound(prefix + 1); // first index with cumulative > prefix
                        rem = idx - 1;
                        var st = buckets![rem]!;
                        chosenStock = st.Pop();
                        if (st.Count == 0) buckets[rem] = null;
                        bit.Add(idx, -1);
                    }
                }
                else
                {
                    if (treap!.TryPopLowerBound(len, out rem, out chosenStock))
                    {
                        // got it
                    }
                }
            }

            if (chosenStock == -1)
            {
                chosenStock = stockCuts.Count;
                stockCuts.Add(new List<int>(4) { origIdx });
                stockRemaining.Add(stockLen - len);
                int newRem = stockLen - len;
                if (len <= stockLen && newRem > 0)
                {
                    if (useFenwick)
                    {
                        var st = buckets![newRem];
                        if (st == null) buckets[newRem] = st = new Stack<int>();
                        st.Push(chosenStock);
                        bit!.Add(newRem + 1, 1);
                    }
                    else
                    {
                        treap!.AddStock(newRem, chosenStock);
                    }
                }
            }
            else
            {
                // add to existing stock
                stockCuts[chosenStock].Add(origIdx);
                int newRem = rem - len;
                stockRemaining[chosenStock] = newRem;
                if (newRem > 0)
                {
                    if (useFenwick)
                    {
                        var st = buckets![newRem];
                        if (st == null) buckets[newRem] = st = new Stack<int>();
                        st.Push(chosenStock);
                        bit!.Add(newRem + 1, 1);
                    }
                    else
                    {
                        treap!.AddStock(newRem, chosenStock);
                    }
                }
            }
        }

        var sb = new StringBuilder(Math.Min(1 << 26, 32 + n * 4));
        sb.Append(stockCuts.Count).Append('\n');
        for (int s = 0; s < stockCuts.Count; s++)
        {
            var list = stockCuts[s];
            for (int j = 0; j < list.Count; j++)
            {
                if (j != 0) sb.Append(' ');
                sb.Append(list[j]);
            }
            sb.Append('\n');
        }

        Console.Out.Write(sb.ToString());
    }
}
