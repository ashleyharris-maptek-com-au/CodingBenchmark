// Longest Common Substring (multiple strings) - adaptive solver
// MSVC /std:c++20 compatible, standard library only.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <string_view>
#include <vector>

using std::string;
using std::vector;

static string lcs_two_dp(const string& a, const string& b) {
    if (a.empty() || b.empty()) return "";
    // O(|a||b|) DP with rolling row; suitable for small inputs.
    const int n = static_cast<int>(a.size());
    const int m = static_cast<int>(b.size());
    vector<int> prev(m + 1, 0), cur(m + 1, 0);
    int bestLen = 0;
    int bestEnd = 0; // end position in a (1-based index)
    for (int i = 1; i <= n; ++i) {
        std::fill(cur.begin(), cur.end(), 0);
        for (int j = 1; j <= m; ++j) {
            if (a[i - 1] == b[j - 1]) {
                cur[j] = prev[j - 1] + 1;
                if (cur[j] > bestLen) {
                    bestLen = cur[j];
                    bestEnd = i;
                }
            }
        }
        prev.swap(cur);
    }
    if (bestLen == 0) return "";
    return a.substr(static_cast<size_t>(bestEnd - bestLen), static_cast<size_t>(bestLen));
}

struct SAM {
    struct Edge {
        unsigned char ch;
        int to;
        int next;
    };
    struct State {
        int link = -1;
        int len = 0;
        int head = -1;     // head of linked list in edges[]
        int first_pos = 0; // end position in base string
    };

    vector<State> st;
    vector<Edge> edges;
    int last = 0;

    SAM(size_t reserve_len = 0) {
        st.reserve(reserve_len ? (2 * reserve_len) : 0);
        edges.reserve(reserve_len ? (3 * reserve_len) : 0); // typical upper bound
        st.push_back(State{}); // root
        last = 0;
    }

    int get_next(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return edges[e].to;
        }
        return -1;
    }

    void set_next(int v, unsigned char c, int to) {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) {
                edges[e].to = to;
                return;
            }
        }
        Edge ne{c, to, st[v].head};
        st[v].head = static_cast<int>(edges.size());
        edges.push_back(ne);
    }

    void extend(unsigned char c, int pos_1based) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;
        st[cur].first_pos = pos_1based;

        int p = last;
        while (p != -1 && get_next(p, c) == -1) {
            set_next(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = get_next(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = static_cast<int>(st.size());
                st.push_back(st[q]); // copies head/link/len/first_pos
                st[clone].len = st[p].len + 1;
                // first_pos of clone can remain as in q

                while (p != -1 && get_next(p, c) == q) {
                    set_next(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }
};

static vector<int> sam_order_by_len(const SAM& sam) {
    int sz = static_cast<int>(sam.st.size());
    int maxlen = 0;
    for (const auto& s : sam.st) maxlen = std::max(maxlen, s.len);

    vector<int> cnt(static_cast<size_t>(maxlen) + 1, 0);
    for (const auto& s : sam.st) ++cnt[static_cast<size_t>(s.len)];
    for (int i = 1; i <= maxlen; ++i) cnt[static_cast<size_t>(i)] += cnt[static_cast<size_t>(i - 1)];

    vector<int> order(static_cast<size_t>(sz));
    for (int i = sz - 1; i >= 0; --i) {
        int l = sam.st[i].len;
        order[static_cast<size_t>(--cnt[static_cast<size_t>(l)])] = i;
    }
    return order; // increasing by len
}

static string lcs_multiple_sam(const vector<string>& strings) {
    if (strings.empty()) return "";
    for (const auto& s : strings) {
        if (s.empty()) return ""; // if any string empty -> no non-empty common substring
    }

    // Build SAM from shortest string to minimize states/edges.
    int baseIdx = 0;
    for (int i = 1; i < static_cast<int>(strings.size()); ++i) {
        if (strings[i].size() < strings[baseIdx].size()) baseIdx = i;
    }

    const string& base = strings[baseIdx];
    SAM sam(base.size());
    for (size_t i = 0; i < base.size(); ++i) {
        sam.extend(static_cast<unsigned char>(base[i]), static_cast<int>(i) + 1);
    }

    const int sz = static_cast<int>(sam.st.size());
    vector<int> order_inc = sam_order_by_len(sam);
    vector<int> order_dec = order_inc;
    std::reverse(order_dec.begin(), order_dec.end());

    vector<int> mn(static_cast<size_t>(sz));
    for (int i = 0; i < sz; ++i) mn[static_cast<size_t>(i)] = sam.st[i].len;

    vector<int> best(static_cast<size_t>(sz));

    for (int idx = 0; idx < static_cast<int>(strings.size()); ++idx) {
        if (idx == baseIdx) continue;
        const string& t = strings[idx];

        std::fill(best.begin(), best.end(), 0);
        int v = 0;
        int l = 0;
        for (unsigned char c : t) {
            int to = sam.get_next(v, c);
            while (v != 0 && to == -1) {
                v = sam.st[v].link;
                l = std::min(l, sam.st[v].len);
                to = sam.get_next(v, c);
            }
            if (to != -1) {
                v = to;
                ++l;
            } else {
                v = 0;
                l = 0;
            }
            if (l > best[static_cast<size_t>(v)]) best[static_cast<size_t>(v)] = l;
        }

        // Propagate matches to suffix links.
        for (int state : order_dec) {
            int p = sam.st[state].link;
            if (p >= 0) {
                int propagated = std::min(best[static_cast<size_t>(state)], sam.st[p].len);
                if (propagated > best[static_cast<size_t>(p)]) best[static_cast<size_t>(p)] = propagated;
            }
        }

        for (int i = 0; i < sz; ++i) {
            mn[static_cast<size_t>(i)] = std::min(mn[static_cast<size_t>(i)], best[static_cast<size_t>(i)]);
        }
    }

    int ansLen = 0;
    int ansState = 0;
    for (int i = 0; i < sz; ++i) {
        if (mn[static_cast<size_t>(i)] > ansLen) {
            ansLen = mn[static_cast<size_t>(i)];
            ansState = i;
        }
    }
    if (ansLen == 0) return "";

    int endPos1 = sam.st[ansState].first_pos; // 1-based end position in base
    int start0 = endPos1 - ansLen;            // 0-based start
    if (start0 < 0) start0 = 0;
    return base.substr(static_cast<size_t>(start0), static_cast<size_t>(ansLen));
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    if (!(std::cin >> N)) return 0;
    string dummy;
    std::getline(std::cin, dummy); // consume endline

    vector<string> s;
    s.reserve(N > 0 ? static_cast<size_t>(N) : 0);
    for (int i = 0; i < N; ++i) {
        string line;
        std::getline(std::cin, line);
        // Keep line exactly as-is (excluding trailing '\r' if present).
        if (!line.empty() && line.back() == '\r') line.pop_back();
        s.push_back(std::move(line));
    }

    if (N <= 0) {
        std::cout << "\n";
        return 0;
    }

    // Adaptive selection:
    // - Tiny two-string cases: DP is simplest and very fast.
    // - Everything else: suffix automaton on shortest string (linear / exact).
    uint64_t total = 0;
    size_t minLen = std::numeric_limits<size_t>::max();
    size_t maxLen = 0;
    for (const auto& str : s) {
        total += static_cast<uint64_t>(str.size());
        minLen = std::min(minLen, str.size());
        maxLen = std::max(maxLen, str.size());
    }

    string ans;
    if (N == 2 && (total <= 2'000'000ULL) && (static_cast<uint64_t>(s[0].size()) * static_cast<uint64_t>(s[1].size()) <= 30'000'000ULL)) {
        ans = lcs_two_dp(s[0], s[1]);
    } else {
        // For very large N/strings, prefer SAM built on shortest string to control memory.
        // Complexity: O(sum lengths) with exact matching.
        ans = lcs_multiple_sam(s);
    }

    std::cout << ans << "\n";
    return 0;
}
