%v2float = OpTypeVector %float 2

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%_ptr_Input_v3float = OpTypePointer Input %v3float
%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input

%_ptr_Output_v4float = OpTypePointer Output %v4float
%outColor = OpVariable %_ptr_Output_v4float Output

%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3.0

%float_0_8b = OpConstant %float 0.8

%c3_0_8_0_2_0_2 = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%c3_0_7_0_7_0_8 = OpConstantComposite %v3float %float_0_7b %float_0_7b %float_0_8b

%main = OpFunction %void None %3
%5 = OpLabel

; load inputs
%wp = OpLoad %v3float %inWorldPos
%n_in = OpLoad %v3float %inNormal

; normalize normal
%n = OpExtInst %v3float %1 Normalize %n_in

; read lightPos.xyz and cameraPos.xyz from UBO
%lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPos_ptr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2

%camPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%camPos4 = OpLoad %v4float %camPos_ptr
%camPos = OpVectorShuffle %v3float %camPos4 %camPos4 0 1 2

; light direction
%lightVec = OpFSub %v3float %lightPos %wp
%l = OpExtInst %v3float %1 Normalize %lightVec

; diffuse = max(dot(n,l),0)
%ndotl = OpDot %float %n %l
%diffuse = OpFMax %float %ndotl %float_0

; lambert factor = 0.3 + 0.7*diffuse
%scaled = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %scaled

; baseColor = (0.8,0.2,0.2) * lambert
%lambert3 = OpCompositeConstruct %v3float %lambert %lambert %lambert
%baseColor = OpFMul %v3float %c3_0_8_0_2_0_2 %lambert3

; dist = length(camPos - wp)
%camVec = OpFSub %v3float %camPos %wp
%dist = OpExtInst %float %1 Length %camVec

; fog = clamp((dist - 1.5)/3.0, 0, 1)
%dsub = OpFSub %float %dist %float_1_5
%ddiv = OpFDiv %float %dsub %float_3
%fog = OpExtInst %float %1 FClamp %ddiv %float_0 %float_1

; final = baseColor*(1-fog) + fogColor*fog
%oneMinusFog = OpFSub %float %float_1 %fog
%oneMinusFog3 = OpCompositeConstruct %v3float %oneMinusFog %oneMinusFog %oneMinusFog
%fog3 = OpCompositeConstruct %v3float %fog %fog %fog
%partA = OpFMul %v3float %baseColor %oneMinusFog3
%partB = OpFMul %v3float %c3_0_7_0_7_0_8 %fog3
%final3 = OpFAdd %v3float %partA %partB

; outColor = vec4(final3, 1.0)
%fx = OpCompositeExtract %float %final3 0
%fy = OpCompositeExtract %float %final3 1
%fz = OpCompositeExtract %float %final3 2
%out = OpCompositeConstruct %v4float %fx %fy %fz %float_1
OpStore %outColor %out
OpReturn
