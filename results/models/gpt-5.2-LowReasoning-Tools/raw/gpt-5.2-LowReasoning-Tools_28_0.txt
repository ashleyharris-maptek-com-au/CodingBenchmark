use std::io::{self, Read, Write};

struct FastInput {
    buf: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut val: usize = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if b.is_ascii_whitespace() {
                break;
            }
            val = val * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        Some(val)
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = match input.next_usize() {
        Some(v) => v,
        None => return,
    };

    // Build a maximal matching in a single pass over edges.
    // The set of endpoints of a maximal matching is always a vertex cover.
    let mut matched = vec![false; n];
    let mut cover: Vec<usize> = Vec::new();
    cover.reserve(n.min(2 * 1024 * 1024));

    for _ in 0..m {
        let u = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        let v = match input.next_usize() {
            Some(v) => v,
            None => break,
        };
        if u < n && v < n && !matched[u] && !matched[v] && u != v {
            matched[u] = true;
            matched[v] = true;
            cover.push(u);
            cover.push(v);
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", cover.len()));
    if !cover.is_empty() {
        for (i, x) in cover.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&x.to_string());
        }
        out.push('\n');
    } else {
        out.push('\n');
    }

    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(out.as_bytes()).unwrap();
}
