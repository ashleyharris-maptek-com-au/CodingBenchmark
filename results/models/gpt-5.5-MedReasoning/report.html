struct City {
    float x = 0.0f;
    float y = 0.0f;
    uint32_t id = 0;
    uint64_t key = 0;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline double dist_id(const std::vector<City>& c, uint32_t a, uint32_t b) {
    return std::sqrt(dist2(c[a], c[b]));
}

static uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static uint64_t morton2(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

static void rotate_to_zero(std::vector<uint32_t>& route) {
    auto it = std::find(route.begin(), route.end(), 0u);
    if (it != route.end()) {
        std::rotate(route.begin(), it, route.end());
    }
}

static std::vector<uint32_t> brute_force_route(const std::vector<City>& c) {
    const uint32_t n = static_cast<uint32_t>(c.size());
    std::vector<uint32_t> perm;
    for (uint32_t i = 1; i < n; ++i) perm.push_back(i);

    std::vector<uint32_t> best = perm;
    double best_len = std::numeric_limits<double>::infinity();

    do {
        double len = dist_id(c, 0, perm[0]);
        for (uint32_t i = 1; i < perm.size(); ++i) {
            len += dist_id(c, perm[i - 1], perm[i]);
        }
        len += dist_id(c, perm.back(), 0);
        if (len < best_len) {
            best_len = len;
            best = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<uint32_t> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), best.begin(), best.end());
    return route;
}

static std::vector<uint32_t> nearest_neighbor_route(const std::vector<City>& c) {
    const uint32_t n = static_cast<uint32_t>(c.size());
    std::vector<uint32_t> route;
    std::vector<uint32_t> remaining;
    route.reserve(n);
    remaining.reserve(n > 0 ? n - 1 : 0);

    route.push_back(0);
    for (uint32_t i = 1; i < n; ++i) remaining.push_back(i);

    uint32_t cur = 0;
    while (!remaining.empty()) {
        size_t best_pos = 0;
        double best_d = dist2(c[cur], c[remaining[0]]);
        for (size_t i = 1; i < remaining.size(); ++i) {
            const double d = dist2(c[cur], c[remaining[i]]);
            if (d < best_d) {
                best_d = d;
                best_pos = i;
            }
        }
        cur = remaining[best_pos];
        route.push_back(cur);
        remaining[best_pos] = remaining.back();
        remaining.pop_back();
    }
    return route;
}

static void two_opt_limited(std::vector<uint32_t>& route,
                            const std::vector<City>& c,
                            int passes,
                            uint32_t window) {
    const uint32_t n = static_cast<uint32_t>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool changed = false;
        for (uint32_t i = 1; i + 2 < n; ++i) {
            const uint32_t max_j = std::min<uint32_t>(n - 1, i + window);
            for (uint32_t j = i + 1; j < max_j; ++j) {
                const uint32_t a = route[i - 1];
                const uint32_t b = route[i];
                const uint32_t d = route[j + 1];
                const uint32_t e = route[j];
                const double before = dist_id(c, a, b) + dist_id(c, e, d);
                const double after = dist_id(c, a, e) + dist_id(c, b, d);
                if (after + 1e-12 < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    changed = true;
                }
            }
        }
        if (!changed) break;
    }
}

static std::vector<uint32_t> morton_route(std::vector<City>& c,
                                          double min_x,
                                          double max_x,
                                          double min_y,
                                          double max_y) {
    const double sx = (max_x > min_x) ? (4294967295.0 / (max_x - min_x)) : 0.0;
    const double sy = (max_y > min_y) ? (4294967295.0 / (max_y - min_y)) : 0.0;

    for (City& city : c) {
        double nx = (static_cast<double>(city.x) - min_x) * sx;
        double ny = (static_cast<double>(city.y) - min_y) * sy;
        if (nx < 0.0) nx = 0.0;
        if (ny < 0.0) ny = 0.0;
        if (nx > 4294967295.0) nx = 4294967295.0;
        if (ny > 4294967295.0) ny = 4294967295.0;
        city.key = morton2(static_cast<uint32_t>(nx), static_cast<uint32_t>(ny));
    }

    std::vector<uint32_t> route(c.size());
    std::iota(route.begin(), route.end(), 0u);
    std::sort(route.begin(), route.end(), [&c](uint32_t a, uint32_t b) {
        if (c[a].key != c[b].key) return c[a].key < c[b].key;
        return a < b;
    });

    rotate_to_zero(route);
    return route;
}

static vector<Plane> build_convex_hull_planes(const vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    vector<Plane> planes;
    if (n < 4) return planes;

    Vec3 centroid;
    for (const Vec3& p : pts) centroid = centroid + p;
    centroid = centroid / static_cast<double>(n);

    Vec3 mn = pts[0], mx = pts[0];
    for (const Vec3& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    const double diag = max(1.0, norm(mx - mn));
    const double eps = 1e-9 * diag;

    unordered_set<string> seen;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 normal = crossp(pts[j] - pts[i], pts[k] - pts[i]);
                double len = norm(normal);
                if (len <= eps) continue;
                normal = normal / len;
                double d = -dotp(normal, pts[i]);

                int pos = 0, neg = 0;
                for (const Vec3& p : pts) {
                    double s = dotp(normal, p) + d;
                    if (s > eps) ++pos;
                    else if (s < -eps) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                if (pos && !neg) {
                    normal = normal * -1.0;
                    d = -d;
                }

                // Make sure the centroid is on the inside side. This also handles
                // numerical ambiguity when all points are nearly coplanar.
                if (dotp(normal, centroid) + d > 0.0) {
                    normal = normal * -1.0;
                    d = -d;
                }

                Plane pl{normal, d};
                string key = plane_key(pl);
                if (seen.insert(key).second) planes.push_back(pl);
            }
        }
    }
    return planes;
}

static vector<Vec3> reduced_hull_points(const vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    const int LIMIT = 180;
    if (n <= LIMIT) return pts;

    vector<Vec3> dirs;
    for (int dx = -2; dx <= 2; ++dx) {
        for (int dy = -2; dy <= 2; ++dy) {
            for (int dz = -2; dz <= 2; ++dz) {
                if (dx == 0 && dy == 0 && dz == 0) continue;
                Vec3 d{static_cast<double>(dx), static_cast<double>(dy), static_cast<double>(dz)};
                double len = norm(d);
                dirs.push_back(d / len);
            }
        }
    }

    vector<int> chosen;
    vector<char> used(n, 0);
    auto add_index = [&](int idx) {
        if (idx >= 0 && !used[idx]) {
            used[idx] = 1;
            chosen.push_back(idx);
        }
    };

    for (const Vec3& d : dirs) {
        int imin = 0, imax = 0;
        double vmin = dotp(d, pts[0]), vmax = vmin;
        for (int i = 1; i < n; ++i) {
            double v = dotp(d, pts[i]);
            if (v < vmin) { vmin = v; imin = i; }
            if (v > vmax) { vmax = v; imax = i; }
        }
        add_index(imin);
        add_index(imax);
    }

    vector<Vec3> reduced;
    reduced.reserve(chosen.size());
    for (int idx : chosen) reduced.push_back(pts[idx]);
    if (static_cast<int>(reduced.size()) > LIMIT) reduced.resize(LIMIT);
    return reduced;
}

#[derive(Clone)]
struct Solver {
    n: usize,
    start: usize,
    require_cycle: bool,
    adj: Vec<Vec<usize>>,
    coord: Vec<(usize, usize)>,
    used: Vec<bool>,
    path: Vec<usize>,
    deadline: Instant,
}

impl Solver {
    fn adjacent_to_start(&self, v: usize) -> bool {
        self.adj[v].iter().any(|&u| u == self.start)
    }

    fn connected_from_start(&self) -> bool {
        if self.n == 0 {
            return false;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[self.start] = true;
        q.push_back(self.start);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            for &u in &self.adj[v] {
                if !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == self.n
    }

    fn residual_connected(&self, cur: usize, remaining: usize) -> bool {
        if remaining == 0 {
            return true;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[cur] = true;
        q.push_back(cur);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            for &u in &self.adj[v] {
                if !seen[u] && (!self.used[u] || u == cur) {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == remaining + 1
    }

    fn residual_degree_prune(&self, cur: usize, remaining: usize) -> bool {
        if remaining == 0 {
            return true;
        }

        let mut forced_endpoints = 0usize;
        for v in 0..self.n {
            if self.used[v] {
                continue;
            }
            let mut deg = 0usize;
            for &u in &self.adj[v] {
                if !self.used[u] || u == cur {
                    deg += 1;
                }
            }
            if deg == 0 {
                return false;
            }
            if deg == 1 {
                forced_endpoints += 1;
                if forced_endpoints > 1 {
                    return false;
                }
            }
        }
        true
    }

    fn onward_degree(&self, v: usize) -> usize {
        self.adj[v].iter().filter(|&&u| !self.used[u]).count()
    }

    fn dfs(&mut self, cur: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }

        let remaining = self.n - self.path.len();
        if remaining == 0 {
            return !self.require_cycle || self.adjacent_to_start(cur);
        }

        if !self.residual_connected(cur, remaining) {
            return false;
        }
        if !self.residual_degree_prune(cur, remaining) {
            return false;
        }

        let mut cand: Vec<usize> = self.adj[cur]
            .iter()
            .copied()
            .filter(|&u| !self.used[u])
            .collect();

        cand.sort_by_key(|&u| {
            let close_penalty = if self.require_cycle && remaining > 1 && self.adjacent_to_start(u) {
                10000usize
            } else {
                0usize
            };
            close_penalty + self.onward_degree(u)
        });

        for u in cand {
            self.used[u] = true;
            self.path.push(u);
            if self.dfs(u) {
                return true;
            }
            self.path.pop();
            self.used[u] = false;
        }
        false
    }
}

fn rectangular_path(w: usize, h: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {

                }
            }
        }

        n = normalize(cross(b - a, c - a));
        w = dot(n, a);
    }

    void flip() {
        n = n * -1.0;
        w = -w;
    }
};

struct Polygon {
    vector<Vec3> v;
    Plane plane;

    Polygon() = default;
    explicit Polygon(vector<Vec3> verts) : v(std::move(verts)) {
        if (v.size() >= 3) plane = Plane(v[0], v[1], v[2]);
    }

    void flip() {
        std::reverse(v.begin(), v.end());
        plane.flip();
    }
};

static void splitPolygon(const Plane& pl,
                         const Polygon& poly,
                         vector<Polygon>& coplanarFront,
                         vector<Polygon>& coplanarBack,
                         vector<Polygon>& front,
                         vector<Polygon>& back) {
    enum { COPLANAR = 0, FRONT = 1, BACK = 2, SPANNING = 3 };
    int polygonType = 0;
    vector<int> types(poly.v.size());

    for (size_t i = 0; i < poly.v.size(); ++i) {
        double t = dot(pl.n, poly.v[i]) - pl.w;
        int type = (t < -EPS) ? BACK : (t > EPS ? FRONT : COPLANAR);
        polygonType |= type;
        types[i] = type;
    }

    if (polygonType == COPLANAR) {
        if (dot(pl.n, poly.plane.n) > 0.0) coplanarFront.push_back(poly);
        else coplanarBack.push_back(poly);
    } else if (polygonType == FRONT) {
        front.push_back(poly);
    } else if (polygonType == BACK) {
        back.push_back(poly);
    } else {
        vector<Vec3> f, b;
        for (size_t i = 0; i < poly.v.size(); ++i) {
            size_t j = (i + 1) % poly.v.size();
            int ti = types[i], tj = types[j];
            const Vec3& vi = poly.v[i];
            const Vec3& vj = poly.v[j];

            if (ti != BACK) f.push_back(vi);
            if (ti != FRONT) b.push_back(vi);
            if ((ti | tj) == SPANNING) {
                double denom = dot(pl.n, vj - vi);
                if (std::abs(denom) > EPS) {
                    double t = (pl.w - dot(pl.n, vi)) / denom;
                    Vec3 p = vi + (vj - vi) * t;
                    f.push_back(p);
                    b.push_back(p);
                }
            }
        }
        if (f.size() >= 3) front.emplace_back(std::move(f));
        if (b.size() >= 3) back.emplace_back(std::move(b));
    }
}

class BSPNode {
public:
    Plane plane;
    bool hasPlane = false;
    vector<Polygon> polygons;
    std::unique_ptr<BSPNode> front;
    std::unique_ptr<BSPNode> back;

    BSPNode() = default;
    explicit BSPNode(const vector<Polygon>& list) { build(list); }

    vector<Polygon> allPolygons() const {
        vector<Polygon> result = polygons;
        if (front) {
            vector<Polygon> f = front->allPolygons();
            result.insert(result.end(), f.begin(), f.end());
        }
        if (back) {
            vector<Polygon> b = back->allPolygons();
            result.insert(result.end(), b.begin(), b.end());
        }
        return result;
    }

    void invert() {
        for (auto& p : polygons) p.flip();
        plane.flip();
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    vector<Polygon> clipPolygons(const vector<Polygon>& list) const {
        if (!hasPlane) return list;
        vector<Polygon> f, b;
        for (const auto& p : list) {
            vector<Polygon> cf, cb;
            splitPolygon(plane, p, cf, cb, f, b);
            f.insert(f.end(), cf.begin(), cf.end());
            b.insert(b.end(), cb.begin(), cb.end());
        }
        if (front) f = front->clipPolygons(f);
        if (back) b = back->clipPolygons(b);
        else b.clear();
        f.insert(f.end(), b.begin(), b.end());
        return f;
    }

    void clipTo(const BSPNode& other) {
        polygons = other.clipPolygons(polygons);
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }

    void build(const vector<Polygon>& list) {
        if (list.empty()) return;
        if (!hasPlane) {
            plane = list[0].plane;
            hasPlane = true;
        }
        vector<Polygon> f, b;
        for (const auto& p : list) {
            splitPolygon(plane, p, polygons, polygons, f, b);
        }
        if (!f.empty()) {
            if (!front) front = std::make_unique<BSPNode>();
            front->build(f);
        }
        if (!b.empty()) {
            if (!back) back = std::make_unique<BSPNode>();
            back->build(b);
        }
    }
};

static AABB bounds(const Mesh& m) {
    AABB b;
    for (const auto& p : m.vertices) {
        b.mn.x = std::min(b.mn.x, p.x); b.mn.y = std::min(b.mn.y, p.y); b.mn.z = std::min(b.mn.z, p.z);
        b.mx.x = std::max(b.mx.x, p.x); b.mx.y = std::max(b.mx.y, p.y); b.mx.z = std::max(b.mx.z, p.z);
    }
    return b;
}

static bool disjoint(const AABB& a, const AABB& b) {
    return a.mx.x < b.mn.x - EPS || b.mx.x < a.mn.x - EPS ||
           a.mx.y < b.mn.y - EPS || b.mx.y < a.mn.y - EPS ||
           a.mx.z < b.mn.z - EPS || b.mx.z < a.mn.z - EPS;
}

static bool rayIntersectsTriangle(const Vec3& orig, const Vec3& dir,
                                  const Vec3& a, const Vec3& b, const Vec3& c,
                                  double& t) {
    Vec3 e1 = b - a, e2 = c - a;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (std::abs(det) < EPS) return false;
    double inv = 1.0 / det;
    Vec3 s = orig - a;
    double u = dot(s, p) * inv;
    if (u < -EPS || u > 1.0 + EPS) return false;
    Vec3 q = cross(s, e1);
    double v = dot(dir, q) * inv;
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    t = dot(e2, q) * inv;
    return t > EPS;
}

static bool pointInsideMesh(const Vec3& p, const Mesh& m) {
    Vec3 dir{0.87654321, 0.23456789, 0.41975311};
    int hits = 0;
    for (const auto& f : m.faces) {
        double t = 0.0;
        if (rayIntersectsTriangle(p, dir, m.vertices[f[0]], m.vertices[f[1]], m.vertices[f[2]], t)) {
            ++hits;
        }
    }
    return (hits & 1) != 0;
}

static vector<Polygon> toPolygons(const Mesh& m) {
    vector<Polygon> out;
    out.reserve(m.faces.size());

    }
    return out;
}

static Mesh concatenate(const Mesh& a, const Mesh& b) {
    Mesh r = a;
    int off = static_cast<int>(r.vertices.size());
    r.vertices.insert(r.vertices.end(), b.vertices.begin(), b.vertices.end());

struct KeyHash {

        uint64_t h = 1469598103934665603ull;
        auto mix = [&](long long v) {
            uint64_t x = static_cast<uint64_t>(v);

        };
        mix(k.x); mix(k.y); mix(k.z);
        return static_cast<size_t>(h);
    }
};

static Key makeKey(const Vec3& p) {
    const double s = 1e9;
    return {static_cast<long long>(std::llround(p.x * s)),
            static_cast<long long>(std::llround(p.y * s)),
            static_cast<long long>(std::llround(p.z * s))};
}

static Mesh fromPolygons(const vector<Polygon>& polys) {
    Mesh m;
    std::unordered_map<Key, int, KeyHash> ids;
    auto add = [&](const Vec3& p) {
        Key k = makeKey(p);
        auto it = ids.find(k);
        if (it != ids.end()) return it->second;
        int id = static_cast<int>(m.vertices.size());
        ids.emplace(k, id);
        m.vertices.push_back(p);
        return id;
    };

    for (const auto& p : polys) {
        if (p.v.size() < 3) continue;
        int i0 = add(p.v[0]);
        for (size_t i = 1; i + 1 < p.v.size(); ++i) {
            int i1 = add(p.v[i]);
            int i2 = add(p.v[i + 1]);
            if (i0 != i1 && i1 != i2 && i2 != i0) m.faces.push_back({i0, i1, i2});
        }
    }
    return m;
}

static Mesh csgUnion(const Mesh& a, const Mesh& b) {
    AABB ab = bounds(a), bb = bounds(b);
    if (disjoint(ab, bb)) return concatenate(a, b);

    if (!a.vertices.empty() && !b.vertices.empty()) {
        bool a0inb = pointInsideMesh(a.vertices[0], b);
        bool b0ina = pointInsideMesh(b.vertices[0], a);
        if (a0inb && !b0ina &&

            ab.mx.x <= bb.mx.x + EPS && ab.mx.y <= bb.mx.y + EPS && ab.mx.z <= bb.mx.z + EPS) {

            bb.mx.x <= ab.mx.x + EPS && bb.mx.y <= ab.mx.y + EPS && bb.mx.z <= ab.mx.z + EPS) {
            return a;

    nb.invert();

    na.build(bp);
    return fromPolygons(na.allPolygons());
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn dot(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.x - a.x) + (b.y - a.y) * (c.y - a.y)
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= EPS &&
    p.x >= a.x.min(b.x) - EPS && p.x <= a.x.max(b.x) + EPS &&
    p.y >= a.y.min(b.y) - EPS && p.y <= a.y.max(b.y) + EPS
}

fn proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let c1 = cross(a, b, c);
    let c2 = cross(a, b, d);
    let c3 = cross(c, d, a);
    let c4 = cross(c, d, b);
    c1 * c2 < -EPS && c3 * c4 < -EPS
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.x * b.y - a.y * b.x;
    }
    0.5 * s.abs()
}

fn bbox(poly: &[Pt]) -> BBox {
    let mut b = BBox { minx: f64::INFINITY, miny: f64::INFINITY, maxx: f64::NEG_INFINITY, maxy: f64::NEG_INFINITY };
    for &p in poly {
        b.minx = b.minx.min(p.x);
        b.miny = b.miny.min(p.y);
        b.maxx = b.maxx.max(p.x);
        b.maxy = b.maxy.max(p.y);
    }
    b
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
        let crosses = (a.y > p.y) != (b.y > p.y);
        if crosses {
            let x_at_y = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if x_at_y > p.x + EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_strict_in_poly(p: Pt, poly: &[Pt]) -> bool {
    for i in 0..poly.len() {
        if on_segment(poly[i], poly[(i + 1) % poly.len()], p) {
            return false;
        }
    }
    point_in_poly(p, poly)
}

fn transform(poly: &[Pt], rot: f64, tr: Pt) -> Vec<Pt> {
    let (s, c) = rot.sin_cos();
    poly.iter().map(|p| Pt {
        x: p.x * c - p.y * s + tr.x,
        y: p.x * s + p.y * c + tr.y,
    }).collect()
}

fn rotated(poly: &[Pt], rot: f64) -> Vec<Pt> {
    transform(poly, rot, Pt { x: 0.0, y: 0.0 })
}

fn fits_stock(poly: &[Pt], stock: &[Pt], stock_bbox: &BBox) -> bool {
    let b = bbox(poly);
    if b.minx < stock_bbox.minx - EPS || b.maxx > stock_bbox.maxx + EPS ||
       b.miny < stock_bbox.miny - EPS || b.maxy > stock_bbox.maxy + EPS {
        return false;

    for &p in poly {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    for i in 0..poly.len() {

        let b = poly[(i + 1) % poly.len()];

            let c = stock[j];
            let d = stock[(j + 1) % stock.len()];

fn overlaps(a: &[Pt], ab: &BBox, b: &[Pt], bb: &BBox) -> bool {
    if !ab.intersects(bb) {
        return false;
    }
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if proper_intersect(a1, a2, b1, b2) {
                return true;
            }
            if on_segment(a1, a2, b1) && dot(a1, a2, b1) > EPS && dot(a2, a1, b1) > EPS {
                return true;
            }
            if on_segment(b1, b2, a1) && dot(b1, b2, a1) > EPS && dot(b2, b1, a1) > EPS {
                return true;
            }
        }
    }
    point_strict_in_poly(a[0], b) || point_strict_in_poly(b[0], a)
}

    fn w(&self) -> f64 { self.maxx - self.minx }
    fn h(&self) -> f64 { self.maxy - self.miny }
    fn intersects(&self, o: &BBox) -> bool {
        self.minx <= o.maxx + EPS && self.maxx + EPS >= o.minx &&
        self.miny <= o.maxy + EPS && self.maxy + EPS >= o.miny
    }
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn dot(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.x - a.x) + (b.y - a.y) * (c.y - a.y)
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= EPS &&
    p.x >= a.x.min(b.x) - EPS && p.x <= a.x.max(b.x) + EPS &&
    p.y >= a.y.min(b.y) - EPS && p.y <= a.y.max(b.y) + EPS
}

fn proper_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let c1 = cross(a, b, c);
    let c2 = cross(a, b, d);
    let c3 = cross(c, d, a);
    let c4 = cross(c, d, b);
    c1 * c2 < -EPS && c3 * c4 < -EPS
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.x * b.y - a.y * b.x;
    }
    0.5 * s.abs()
}

fn bbox(poly: &[Pt]) -> BBox {
    let mut b = BBox { minx: f64::INFINITY, miny: f64::INFINITY, maxx: f64::NEG_INFINITY, maxy: f64::NEG_INFINITY };
    for &p in poly {
        b.minx = b.minx.min(p.x);
        b.miny = b.miny.min(p.y);
        b.maxx = b.maxx.max(p.x);
        b.maxy = b.maxy.max(p.y);
    }
    b
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p) {
            return true;
        }
        let crosses = (a.y > p.y) != (b.y > p.y);
        if crosses {
            let x_at_y = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if x_at_y > p.x + EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn point_strict_in_poly(p: Pt, poly: &[Pt]) -> bool {
    for i in 0..poly.len() {
        if on_segment(poly[i], poly[(i + 1) % poly.len()], p) {
            return false;
        }
    }
    point_in_poly(p, poly)
}

fn transform(poly: &[Pt], rot: f64, tr: Pt) -> Vec<Pt> {
    let (s, c) = rot.sin_cos();
    poly.iter().map(|p| Pt {
        x: p.x * c - p.y * s + tr.x,
        y: p.x * s + p.y * c + tr.y,
    }).collect()
}

fn rotated(poly: &[Pt], rot: f64) -> Vec<Pt> {
    transform(poly, rot, Pt { x: 0.0, y: 0.0 })
}

fn fits_stock(poly: &[Pt], stock: &[Pt], stock_bbox: &BBox) -> bool {
    let b = bbox(poly);
    if b.minx < stock_bbox.minx - EPS || b.maxx > stock_bbox.maxx + EPS ||
       b.miny < stock_bbox.miny - EPS || b.maxy > stock_bbox.maxy + EPS {
        return false;
    }
    for &p in poly {
        if !point_in_poly(p, stock) {
            return false;
        }
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        for j in 0..stock.len() {
            let c = stock[j];
            let d = stock[(j + 1) % stock.len()];
            if proper_intersect(a, b, c, d) {
                return false;
            }
        }
    }
    true
}

fn overlaps(a: &[Pt], ab: &BBox, b: &[Pt], bb: &BBox) -> bool {
    if !ab.intersects(bb) {
        return false;
    }
    for i in 0..a.len() {
        let a1 = a[i];
        let a2 = a[(i + 1) % a.len()];
        for j in 0..b.len() {
            let b1 = b[j];
            let b2 = b[(j + 1) % b.len()];
            if proper_intersect(a1, a2, b1, b2) {
                return true;
            }
            if on_segment(a1, a2, b1) && dot(a1, a2, b1) > EPS && dot(a2, a1, b1) > EPS {
                return true;
            }
            if on_segment(b1, b2, a1) && dot(b1, b2, a1) > EPS && dot(b2, b1, a1) > EPS {
                return true;
            }
        }
    }
    point_strict_in_poly(a[0], b) || point_strict_in_poly(b[0], a)
}

fn can_place(poly: &[Pt], b: &BBox, state: &StockState, stock: &[Pt], sb: &BBox) -> bool {
    if !fits_stock(poly, stock, sb) {
        return false;
    }
    for q in &state.placed {
        if overlaps(poly, b, &q.poly, &q.bbox) {
            return false;
        }
    }
    true
}

fn make_angles(np: usize) -> Vec<f64> {
    let n = if np <= 20 { 32 } else if np <= 200 { 16 } else if np <= 1000 { 8 } else { 4 };
    (0..n).map(|i| 2.0 * PI * (i as f64) / (n as f64)).collect()
}

fn try_at(piece: &Piece, rot: f64, left: f64, bottom: f64) -> (Vec<Pt>, BBox, f64, f64) {
    let rp = rotated(&piece.pts, rot);
    let rb = bbox(&rp);
    let tr = Pt { x: left - rb.minx, y: bottom - rb.miny };
    let poly = transform(&piece.pts, rot, tr);
    let b = bbox(&poly);
    (poly, b, tr.x, tr.y)
}

fn place_in_state(
    piece: &Piece,
    state: &mut StockState,
    stock: &[Pt],
    sb: &BBox,
    angles: &[f64],
    grid_div: usize,
) -> Option<Placed> {
    let stock_w = sb.w();
    let stock_h = sb.h();

        let rp = rotated(&piece.pts, rot);
        let rb = bbox(&rp);
        let rw = rb.w();

            continue;

            if rh <= state.shelves[si].height + EPS && state.shelves[si].next_x + rw <= sb.maxx + EPS {
                let left = state.shelves[si].next_x;
                let bottom = state.shelves[si].y;

                if can_place(&poly, &b, state, stock, sb) {
                    state.shelves[si].next_x += rw;

                    state.placed.push(placed.clone());
                    return Some(placed);
                }
            }
        }

        let new_y = state.shelves.iter().map(|s| s.y + s.height).fold(sb.miny, f64::max);
        if new_y + rh <= sb.maxy + EPS {
            let (poly, b, x, y) = try_at(piece, rot, sb.minx, new_y);
            if can_place(&poly, &b, state, stock, sb) {
                state.shelves.push(Shelf { y: new_y, height: rh, next_x: sb.minx + rw });
                let placed = Placed { piece_id: piece.id, poly, bbox: b, x, y, rot };
                state.placed.push(placed.clone());
                return Some(placed);
            }
        }
    }

    let mut xs = vec![sb.minx];
    let mut ys = vec![sb.miny];
    for q in &state.placed {
        xs.push(q.bbox.maxx);
        ys.push(q.bbox.maxy);
    }
    xs.sort_by(|a, b| a.partial_cmp(b).unwrap());
    ys.sort_by(|a, b| a.partial_cmp(b).unwrap());
    xs.dedup_by(|a, b| (*a - *b).abs() < EPS);
    ys.dedup_by(|a, b| (*a - *b).abs() < EPS);

        let rp = rotated(&piece.pts, rot);
        let rb = bbox(&rp);
        let rw = rb.w();
        let rh = rb.h();
        for &left in &xs {
            if left + rw > sb.maxx + EPS { continue; }
            for &bottom in &ys {
                if bottom + rh > sb.maxy + EPS { continue; }
                let (poly, b, x, y) = try_at(piece, rot, left, bottom);
                if can_place(&poly, &b, state, stock, sb) {
                    let placed = Placed { piece_id: piece.id, poly, bbox: b, x, y, rot };
                    state.placed.push(placed.clone());
                    return Some(placed);
                }
            }
        }
    }

    let max_candidates = grid_div + 1;
    for &rot in angles {
        let rp = rotated(&piece.pts, rot);
        let rb = bbox(&rp);
        let rw = rb.w();
        let rh = rb.h();
        if rw > stock_w + EPS || rh > stock_h + EPS { continue; }
        for ix in 0..max_candidates {
            let left = if max_candidates <= 1 { sb.minx } else { sb.minx + (stock_w - rw) * (ix as f64) / ((max_candidates - 1) as f64) };
            for iy in 0..max_candidates {
                let bottom = if max_candidates <= 1 { sb.miny } else { sb.miny + (stock_h - rh) * (iy as f64) / ((max_candidates - 1) as f64) };

                if can_place(&poly, &b, state, stock, sb) {
                    let placed = Placed { piece_id: piece.id, poly, bbox: b, x, y, rot };
                    state.placed.push(placed.clone());
                    return Some(placed);
                }
            }
        }
    }
    None
}

fn single_piece_fallback(piece: &Piece, stock: &[Pt], sb: &BBox, angles: &[f64], grid_div: usize) -> Placed {
    let mut state = StockState { placed: Vec::new(), shelves: Vec::new() };
    if let Some(p) = place_in_state(piece, &mut state, stock, sb, angles, grid_div.max(20)) {
        return p;
    }

    let stock_c = Pt { x: (sb.minx + sb.maxx) * 0.5, y: (sb.miny + sb.maxy) * 0.5 };
    for &rot in angles {
        let rp = rotated(&piece.pts, rot);
        let rb = bbox(&rp);
        let tr = Pt {
            x: stock_c.x - (rb.minx + rb.maxx) * 0.5,
            y: stock_c.y - (rb.miny + rb.maxy) * 0.5,

        let b = bbox(&poly);

            return Placed { piece_id: piece.id, poly, bbox: b, x: tr.x, y: tr.y, rot };

    let b = bbox(&poly);

}
