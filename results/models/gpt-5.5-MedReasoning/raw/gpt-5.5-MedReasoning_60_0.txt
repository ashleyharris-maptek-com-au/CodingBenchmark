const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

class Heap {
  constructor(less) {
    this.a = [];
    this.less = less;
  }
  get size() {
    return this.a.length;
  }
  peek() {
    return this.a[0];
  }
  push(x) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.less(a[i], a[p])) break;
      const t = a[i]; a[i] = a[p]; a[p] = t;
      i = p;
    }
  }
  pop() {
    const a = this.a;
    if (a.length === 0) return undefined;
    const root = a[0];
    const last = a.pop();
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1, r = l + 1, b = i;
        if (l < a.length && this.less(a[l], a[b])) b = l;
        if (r < a.length && this.less(a[r], a[b])) b = r;
        if (b === i) break;
        const t = a[i]; a[i] = a[b]; a[b] = t;
        i = b;
      }
    }
    return root;
  }
}

const tasks = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;
const m = Math.max(1, Math.floor(Number(data.machineCount) || 1));

const ids = new Array(n);
const dur = new Array(n);
const due = new Array(n);
const wt = new Array(n);
const idToIndex = new Map();

for (let i = 0; i < n; i++) {
  const t = tasks[i] || {};
  ids[i] = t.id;
  idToIndex.set(t.id, i);
  dur[i] = Math.max(0, Math.floor(Number(t.duration) || 0));
  const d = Number(t.deadline);
  due[i] = Number.isFinite(d) ? d : 1e18;
  const w = Number(t.weight);
  wt[i] = Number.isFinite(w) && w > 0 ? w : 1;
}

const succ = Array.from({ length: n }, () => []);
const indeg0 = new Array(n).fill(0);
for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i] && tasks[i].deps) ? tasks[i].deps : [];
  const seen = new Set();
  for (const depId of deps) {
    const j = idToIndex.get(depId);
    if (j === undefined || j === i || seen.has(j)) continue;
    seen.add(j);
    succ[j].push(i);
    indeg0[i]++;
  }
}

// Topological pass for downstream critical-path estimates used by the priority rule.
const topo = [];
const q = [];
const tmpIndeg = indeg0.slice();
for (let i = 0; i < n; i++) if (tmpIndeg[i] === 0) q.push(i);
for (let qi = 0; qi < q.length; qi++) {
  const v = q[qi];
  topo.push(v);
  for (const u of succ[v]) {
    tmpIndeg[u]--;
    if (tmpIndeg[u] === 0) q.push(u);
  }
}
if (topo.length < n) {
  for (let i = 0; i < n; i++) if (tmpIndeg[i] > 0) topo.push(i);
}

const tail = new Array(n).fill(0);
const downWeight = new Array(n).fill(0);
for (let ti = topo.length - 1; ti >= 0; ti--) {
  const v = topo[ti];
  let bestTail = 0;
  let sw = 0;
  for (const u of succ[v]) {
    const cp = dur[u] + tail[u];
    if (cp > bestTail) bestTail = cp;
    sw += wt[u] + 0.15 * downWeight[u];
  }
  tail[v] = bestTail;
  downWeight[v] = sw;
}

let totalDur = 0;
for (const p of dur) totalDur += p;
const avgDur = Math.max(1, totalDur / Math.max(1, n));
const idKey = ids.map(x => String(x));

function staticLess(a, b) {
  const ad = due[a] - tail[a];
  const bd = due[b] - tail[b];
  if (ad !== bd) return ad < bd;
  const ar = wt[a] / Math.max(1, dur[a]);
  const br = wt[b] / Math.max(1, dur[b]);
  if (ar !== br) return ar > br;
  if (tail[a] !== tail[b]) return tail[a] > tail[b];
  if (dur[a] !== dur[b]) return dur[a] < dur[b];
  return idKey[a] < idKey[b];
}

function dynamicScore(i, now) {
  const p = Math.max(1, dur[i]);
  const finish = now + dur[i];
  const slack = due[i] - (finish + tail[i]);
  const density = wt[i] / p;
  const urgency = Math.exp(-Math.max(0, slack) / (2.5 * avgDur));
  const late = Math.max(0, -slack);
  return density * (3 + 7 * urgency) +
         (late * wt[i]) / p +
         0.025 * tail[i] +
         0.01 * downWeight[i] / p;
}

const ready = new Heap(staticLess);
const indeg = indeg0.slice();
for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push(i);

const machines = new Heap((a, b) => a.time !== b.time ? a.time < b.time : a.machine < b.machine);
for (let i = 0; i < m; i++) machines.push({ time: 0, machine: i });

const running = new Heap((a, b) => a.finish !== b.finish ? a.finish < b.finish : a.task < b.task);
const scheduled = new Array(n).fill(false);
const result = [];
let scheduledCount = 0;

function releaseFinished(upTo) {
  while (running.size && running.peek().finish <= upTo) {
    const ev = running.pop();
    for (const u of succ[ev.task]) {
      indeg[u]--;
      if (indeg[u] === 0 && !scheduled[u]) ready.push(u);
    }
  }
}

function pickTask(now) {
  const take = ready.size <= 128 ? ready.size : 64;
  let best = -1;
  let bestScore = -Infinity;
  const buf = [];
  for (let k = 0; k < take; k++) {
    const x = ready.pop();
    const s = dynamicScore(x, now);
    if (s > bestScore || (s === bestScore && staticLess(x, best))) {
      if (best !== -1) buf.push(best);
      best = x;
      bestScore = s;
    } else {
      buf.push(x);
    }
  }
  for (const x of buf) ready.push(x);
  return best;
}

while (scheduledCount < n) {
  let mach = machines.pop();
  let now = mach.time;
  releaseFinished(now);

  if (ready.size === 0) {
    if (running.size) {
      now = Math.max(now, running.peek().finish);
      releaseFinished(now);
      mach.time = now;
    } else {
      // Invalid cyclic input: emit a deterministic best-effort order rather than hanging.
      for (let i = 0; i < n; i++) if (!scheduled[i]) ready.push(i);
    }
  }

  if (ready.size === 0) {
    machines.push(mach);
    continue;
  }

  const task = pickTask(now);
  if (task === -1 || scheduled[task]) {
    machines.push(mach);
    continue;
  }

  scheduled[task] = true;
  scheduledCount++;
  result.push({ id: ids[task], start: now, machine: mach.machine });

  const finish = now + dur[task];
  running.push({ finish, task, machine: mach.machine });
  mach.time = finish;
  machines.push(mach);
}

process.stdout.write(JSON.stringify({ schedule: result }));
