; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %outColor "outColor"

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fnVoid = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformV4 = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 1
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%float0 = OpConstant %float 0.0
%float1 = OpConstant %float 1.0
%float02 = OpConstant %float 0.2
%float05 = OpConstant %float 0.5
%float08 = OpConstant %float 0.8
%float32 = OpConstant %float 32.0

%inWorldPos = OpVariable %ptrInputV3 Input
%inNormal = OpVariable %ptrInputV3 Input
%outColor = OpVariable %ptrOutputV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%main = OpFunction %void None %fnVoid
%entry = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normalIn = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %glsl Normalize %normalIn
%lightPtr = OpAccessChain %ptrUniformV4 %ubo %int3
%light4 = OpLoad %v4float %lightPtr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%lightDelta = OpFSub %v3float %light3 %worldPos
%L = OpExtInst %v3float %glsl Normalize %lightDelta
%cameraPtr = OpAccessChain %ptrUniformV4 %ubo %int4
%camera4 = OpLoad %v4float %cameraPtr
%camera3 = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%viewDelta = OpFSub %v3float %camera3 %worldPos
%V = OpExtInst %v3float %glsl Normalize %viewDelta
%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %glsl Reflect %negL %N
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %ndotl %float0
%rdotv = OpDot %float %R %V
%specBase = OpExtInst %float %glsl FMax %rdotv %float0
%specular = OpExtInst %float %glsl Pow %specBase %float32
%diffuseTerm = OpFMul %float %float05 %diffuse
%specularTerm = OpFMul %float %float08 %specular
%basePlusDiffuse = OpFAdd %float %float02 %diffuseTerm
%intensity = OpFAdd %float %basePlusDiffuse %specularTerm
%color = OpCompositeConstruct %v4float %intensity %intensity %intensity %float1
OpStore %outColor %color
OpReturn
OpFunctionEnd
