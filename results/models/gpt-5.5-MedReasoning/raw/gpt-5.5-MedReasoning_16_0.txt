#include <algorithm>
#include <chrono>
#include <iostream>
#include <limits>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    long long duration = 0;
};

struct Schedule {
    vector<vector<long long>> start;
    long long makespan = numeric_limits<long long>::max();
};

struct Node {
    long long a, b, c, d;
    int job;

    bool operator<(const Node& other) const {
        return tie(a, b, c, d) < tie(other.a, other.b, other.c, other.d);
    }
};

static Schedule buildByRule(const vector<vector<Task>>& jobs,
                            const vector<vector<long long>>& remainingWork,
                            const vector<long long>& machineLoad,
                            int machineCount,
                            int rule) {
    const int jobCount = static_cast<int>(jobs.size());
    vector<int> nextOp(jobCount, 0);
    vector<long long> jobReady(jobCount, 0), machineReady(machineCount, 0);
    Schedule result;
    result.start.resize(jobCount);
    for (int j = 0; j < jobCount; ++j) {
        result.start[j].assign(jobs[j].size(), 0);
    }

    auto makeNode = [&](int j) {
        int i = nextOp[j];
        const Task& t = jobs[j][i];
        long long earliestStart = max(jobReady[j], machineReady[t.machine]);
        long long remainingOps = static_cast<long long>(jobs[j].size() - i);

        long long a = 0, b = 0, c = 0;
        switch (rule) {
            case 0: // Most work remaining.
                a = remainingWork[j][i];
                b = t.duration;
                c = -jobReady[j];
                break;
            case 1: // Longest processing time.
                a = t.duration;
                b = remainingWork[j][i];
                c = -jobReady[j];
                break;
            case 2: // Shortest processing time.
                a = -t.duration;
                b = remainingWork[j][i];
                c = -jobReady[j];
                break;
            case 3: // Most operations remaining.
                a = remainingOps;
                b = remainingWork[j][i];
                c = t.duration;
                break;
            case 4: // Earliest completion using the state when the operation becomes available.
                a = -(earliestStart + t.duration);
                b = remainingWork[j][i];
                c = -earliestStart;
                break;
            case 5: // Prefer globally loaded machines.
                a = machineLoad[t.machine];
                b = remainingWork[j][i];
                c = t.duration;
                break;
            case 6: // FIFO by job ready time.
                a = -jobReady[j];
                b = remainingWork[j][i];
                c = -t.duration;
                break;
            default: // Balanced bottleneck/work rule.
                a = remainingWork[j][i] + machineLoad[t.machine];
                b = t.duration;
                c = -earliestStart;
                break;
        }

        return Node{a, b, c, -static_cast<long long>(j), j};
    };

    priority_queue<Node> pq;
    for (int j = 0; j < jobCount; ++j) {
        if (!jobs[j].empty()) pq.push(makeNode(j));
    }

    while (!pq.empty()) {
        int j = pq.top().job;
        pq.pop();
        int i = nextOp[j];
        const Task& t = jobs[j][i];

        long long s = max(jobReady[j], machineReady[t.machine]);
        result.start[j][i] = s;
        long long finish = s + t.duration;
        jobReady[j] = finish;
        machineReady[t.machine] = finish;
        result.makespan = max(result.makespan == numeric_limits<long long>::max() ? 0 : result.makespan, finish);

        ++nextOp[j];
        if (nextOp[j] < static_cast<int>(jobs[j].size())) {
            pq.push(makeNode(j));
        }
    }

    if (result.makespan == numeric_limits<long long>::max()) result.makespan = 0;
    return result;
}

static void improveTinyExactly(const vector<vector<Task>>& jobs,
                               int machineCount,
                               Schedule& best) {
    const int jobCount = static_cast<int>(jobs.size());
    int totalOps = 0;
    for (const auto& job : jobs) totalOps += static_cast<int>(job.size());
    if (totalOps > 12) return;

    vector<int> nextOp(jobCount, 0);
    vector<long long> jobReady(jobCount, 0), machineReady(machineCount, 0);
    vector<vector<long long>> starts(jobCount);
    for (int j = 0; j < jobCount; ++j) starts[j].assign(jobs[j].size(), 0);

    long long nodes = 0;
    const long long nodeLimit = 500000;
    auto deadline = chrono::steady_clock::now() + chrono::milliseconds(250);

    auto dfs = [&](auto&& self, int placed, long long currentMax) -> void {
        if (++nodes > nodeLimit || chrono::steady_clock::now() > deadline) return;
        if (currentMax >= best.makespan) return;

        if (placed == totalOps) {
            best.makespan = currentMax;
            best.start = starts;
            return;
        }

        vector<int> candidates;
        candidates.reserve(jobCount);
        for (int j = 0; j < jobCount; ++j) {
            if (nextOp[j] < static_cast<int>(jobs[j].size())) candidates.push_back(j);
        }
        sort(candidates.begin(), candidates.end(), [&](int x, int y) {
            const Task& tx = jobs[x][nextOp[x]];
            const Task& ty = jobs[y][nextOp[y]];
            long long fx = max(jobReady[x], machineReady[tx.machine]) + tx.duration;
            long long fy = max(jobReady[y], machineReady[ty.machine]) + ty.duration;
            if (fx != fy) return fx < fy;
            return tx.duration > ty.duration;
        });

        for (int j : candidates) {
            int i = nextOp[j];
            const Task& t = jobs[j][i];
            long long oldJobReady = jobReady[j];
            long long oldMachineReady = machineReady[t.machine];
            long long s = max(oldJobReady, oldMachineReady);
            long long f = s + t.duration;

            starts[j][i] = s;
            jobReady[j] = f;
            machineReady[t.machine] = f;
            ++nextOp[j];

            self(self, placed + 1, max(currentMax, f));

            --nextOp[j];
            jobReady[j] = oldJobReady;
            machineReady[t.machine] = oldMachineReady;
        }
    };

    dfs(dfs, 0, 0);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    long long totalOps = 0;
    vector<long long> machineLoad(max(0, M), 0);

    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        totalOps += T;
        for (int i = 0; i < T; ++i) {
            cin >> jobs[j][i].machine >> jobs[j][i].duration;
            if (jobs[j][i].machine >= 0 && jobs[j][i].machine < M) {
                machineLoad[jobs[j][i].machine] += jobs[j][i].duration;
            }
        }
    }

    vector<vector<long long>> remainingWork(J);
    for (int j = 0; j < J; ++j) {
        int T = static_cast<int>(jobs[j].size());
        remainingWork[j].assign(T + 1, 0);
        for (int i = T - 1; i >= 0; --i) {
            remainingWork[j][i] = remainingWork[j][i + 1] + jobs[j][i].duration;
        }
    }

    vector<int> rules;
    if (totalOps <= 2000) {
        rules = {0, 1, 2, 3, 4, 5, 6, 7};
    } else if (totalOps <= 100000) {
        rules = {0, 1, 3, 4, 5};
    } else {
        rules = {0, 3, 5};
    }

    Schedule best;
    for (int rule : rules) {
        Schedule candidate = buildByRule(jobs, remainingWork, machineLoad, M, rule);
        if (candidate.makespan < best.makespan) best = move(candidate);
    }

    improveTinyExactly(jobs, M, best);

    for (int j = 0; j < J; ++j) {
        for (int i = 0; i < static_cast<int>(jobs[j].size()); ++i) {
            if (i) cout << ' ';
            cout << best.start[j][i] << ' ' << jobs[j][i].machine;
        }
        cout << '\n';
    }

    return 0;
}
