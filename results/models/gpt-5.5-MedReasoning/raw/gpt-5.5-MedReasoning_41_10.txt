; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fun = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%int_3 = OpConstant %int 3
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_0_1 = OpConstant %float 0.1
%float_0_4 = OpConstant %float 0.4
%float_0_6 = OpConstant %float 0.6
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1
%float_pi = OpConstant %float 3.14159
%float_16 = OpConstant %float 16
%red = OpConstantComposite %v3float %float_0_9 %float_0_1 %float_0_1
%blue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_9

%main = OpFunction %void None %fun
%entry = OpLabel
%uvVal = OpLoad %v2float %uv
%uvY = OpCompositeExtract %float %uvVal 1
%stripeA = OpFMul %float %uvY %float_pi
%stripePhase = OpFMul %float %stripeA %float_16
%stripeSin = OpExtInst %float %1 Sin %stripePhase
%isRed = OpFOrdGreaterThan %bool %stripeSin %float_0
%baseColor = OpSelect %v3float %isRed %red %blue

%nIn = OpLoad %v3float %normal
%n = OpExtInst %v3float %1 Normalize %nIn
%p = OpLoad %v3float %worldPos
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%light = OpVectorShuffle %v3float %light4 %light4 0 1 2
%toLightRaw = OpFSub %v3float %light %p
%toLight = OpExtInst %v3float %1 Normalize %toLightRaw
%ndotl = OpDot %float %n %toLight
%diffuse = OpExtInst %float %1 FMax %ndotl %float_0
%diffScale = OpFMul %float %float_0_6 %diffuse
%shade = OpFAdd %float %float_0_4 %diffScale
%litColor = OpVectorTimesScalar %v3float %baseColor %shade
%r = OpCompositeExtract %float %litColor 0
%g = OpCompositeExtract %float %litColor 1
%b = OpCompositeExtract %float %litColor 2
%finalColor = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
