import sys


def read_input():
    lines = sys.stdin.read().splitlines()
    if not lines:
        return 0, []

    first = lines[0].split()
    if len(first) < 2:
        return 0, []

    universe_size = int(first[0])
    set_count = int(first[1])
    sets = []

    for i in range(set_count):
        line = lines[i + 1] if i + 1 < len(lines) else ""
        elems = []
        seen = set()
        for token in line.split():
            e = int(token)
            if e not in seen:
                seen.add(e)
                elems.append(e)
        sets.append(elems)

    return universe_size, sets


def solve(universe_size, sets):
    full_mask = (1 << universe_size) - 1
    set_masks = []
    candidates = [[] for _ in range(universe_size)]

    for idx, elems in enumerate(sets):
        mask = 0
        valid = True
        for e in elems:
            if e < 0 or e >= universe_size:
                valid = False
                break
            mask |= 1 << e
        set_masks.append(mask if valid else 0)
        if valid and mask:
            for e in elems:
                candidates[e].append(idx)

    for e in range(universe_size):
        if not candidates[e]:
            return None

    suffix_union = [0] * (len(sets) + 1)
    for i in range(len(sets) - 1, -1, -1):
        suffix_union[i] = suffix_union[i + 1] | set_masks[i]

    answer = []

    def search(covered):
        if covered == full_mask:
            return True

        remaining = full_mask ^ covered
        if remaining & ~suffix_union[0]:
            return False

        best_elem = -1
        best_options = None
        m = remaining
        while m:
            bit = m & -m
            e = bit.bit_length() - 1
            opts = [i for i in candidates[e] if set_masks[i] and (set_masks[i] & covered) == 0]
            if not opts:
                return False
            if best_options is None or len(opts) < len(best_options):
                best_elem = e
                best_options = opts
                if len(opts) == 1:
                    break
            m -= bit

        best_options.sort(key=lambda i: (len(sets[i]), i))

        for idx in best_options:
            mask = set_masks[idx]
            if mask & covered:
                continue
            answer.append(idx)
            if search(covered | mask):
                return True
            answer.pop()

        return False

    if search(0):
        return answer
    return None


def main():
    universe_size, sets = read_input()
    solution = solve(universe_size, sets)
    if solution is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(map(str, solution)))


if __name__ == "__main__":
    main()
