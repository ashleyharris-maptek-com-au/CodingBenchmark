; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft
OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%bool = OpTypeBool
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 1
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%inWorldPos = OpVariable %ptr_Input_v3float Input
%inNormal = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None %func
%entry = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normalIn = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %1 Normalize %normalIn
%cameraPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_4
%camera4 = OpLoad %v4float %cameraPtr
%cameraPos = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%viewDelta = OpFSub %v3float %cameraPos %worldPos
%V = OpExtInst %v3float %1 Normalize %viewDelta
%dotNV = OpDot %float %N %V
%ndotv = OpExtInst %float %1 FMax %dotNV %float_0
%twiceNdotV = OpFMul %float %ndotv %float_2
%aboveHalf = OpFSub %float %ndotv %float_0_5
%t = OpFMul %float %aboveHalf %float_2
%hiRG = OpFSub %float %float_1 %t
%isLow = OpFOrdLessThan %bool %ndotv %float_0_5
%r = OpSelect %float %isLow %float_1 %hiRG
%g = OpSelect %float %isLow %twiceNdotV %hiRG
%b = OpSelect %float %isLow %float_0 %t
%color = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
