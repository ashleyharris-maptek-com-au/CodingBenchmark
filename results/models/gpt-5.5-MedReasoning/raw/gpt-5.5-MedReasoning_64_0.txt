declare const require: any;
declare const process: any;

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Point = [number, number];
type Drone = { id?: string; x: number; y: number };

const probability: number[][] = Array.isArray(data.probability) ? data.probability : [];
const drones: Drone[] = Array.isArray(data.drones) ? data.drones : [];
const maxSteps: number = Math.max(0, Math.floor(Number(data.maxStepsPerDrone) || 0));

const height = probability.length;
const width = height > 0 && Array.isArray(probability[0]) ? probability[0].length : 0;
const n = width * height;

function idx(x: number, y: number): number {
  return y * width + x;
}

function xOf(i: number): number {
  return i % width;
}

function yOf(i: number): number {
  return Math.floor(i / width);
}

function inside(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < width && y < height;
}

function valueAtIndex(i: number): number {
  if (i < 0 || i >= n) return -1;
  return probability[yOf(i)][xOf(i)];
}

function validIndex(i: number): boolean {
  return i >= 0 && i < n && valueAtIndex(i) >= 0;
}

function inputIndex(x: number, y: number): number {
  return inside(x, y) ? idx(x, y) : -1;
}

const values = new Float64Array(n);
const valid = new Uint8Array(n);
const positiveCells: number[] = [];

for (let y = 0; y < height; y++) {
  for (let x = 0; x < width; x++) {
    const i = idx(x, y);
    const v = Number(probability[y][x]);
    values[i] = Number.isFinite(v) ? v : -1;
    if (values[i] >= 0) valid[i] = 1;
    if (values[i] > 0) positiveCells.push(i);
  }
}

const density = new Float64Array(n);
for (const i of positiveCells) {
  const x = xOf(i);
  const y = yOf(i);
  let s = 0;
  for (let dy = -2; dy <= 2; dy++) {
    const yy = y + dy;
    if (yy < 0 || yy >= height) continue;
    for (let dx = -2; dx <= 2; dx++) {
      const xx = x + dx;
      if (xx < 0 || xx >= width) continue;
      const j = idx(xx, yy);
      if (values[j] > 0) s += values[j] / (1 + Math.abs(dx) + Math.abs(dy));
    }
  }
  density[i] = s;
}

positiveCells.sort((a, b) => {
  const sa = values[a] + 0.18 * density[a];
  const sb = values[b] + 0.18 * density[b];
  return sb - sa;
});

const dist = new Int32Array(n);
const prevNode = new Int32Array(n);
const queue = new Int32Array(n);

function nearestValid(startX: number, startY: number): number {
  const s = inputIndex(startX, startY);
  if (s >= 0 && valid[s]) return s;
  if (n === 0) return -1;
  dist.fill(-1);
  let head = 0;
  let tail = 0;
  if (s >= 0) {
    dist[s] = 0;
    queue[tail++] = s;
  } else {
    for (let y = 0; y < height; y++) {
      for (let x = 0; x < width; x++) {
        const i = idx(x, y);
        dist[i] = Math.abs(x - startX) + Math.abs(y - startY);
        queue[tail++] = i;
      }
    }
  }
  while (head < tail) {
    const cur = queue[head++];
    if (valid[cur]) return cur;
    const x = xOf(cur);
    const y = yOf(cur);
    const nd = dist[cur] + 1;
    if (x > 0) {
      const j = cur - 1;
      if (dist[j] < 0) { dist[j] = nd; queue[tail++] = j; }
    }
    if (x + 1 < width) {
      const j = cur + 1;
      if (dist[j] < 0) { dist[j] = nd; queue[tail++] = j; }
    }
    if (y > 0) {
      const j = cur - width;
      if (dist[j] < 0) { dist[j] = nd; queue[tail++] = j; }
    }
    if (y + 1 < height) {
      const j = cur + width;
      if (dist[j] < 0) { dist[j] = nd; queue[tail++] = j; }
    }
  }
  return -1;
}

function bfs(start: number, limit: number): number {
  dist.fill(-1);
  prevNode.fill(-1);
  let head = 0;
  let tail = 0;
  dist[start] = 0;
  queue[tail++] = start;
  while (head < tail) {
    const cur = queue[head++];
    const d = dist[cur];
    if (d >= limit) continue;
    const x = xOf(cur);
    const y = yOf(cur);
    const nd = d + 1;
    if (x > 0) {
      const j = cur - 1;
      if (valid[j] && dist[j] < 0) { dist[j] = nd; prevNode[j] = cur; queue[tail++] = j; }
    }
    if (x + 1 < width) {
      const j = cur + 1;
      if (valid[j] && dist[j] < 0) { dist[j] = nd; prevNode[j] = cur; queue[tail++] = j; }
    }
    if (y > 0) {
      const j = cur - width;
      if (valid[j] && dist[j] < 0) { dist[j] = nd; prevNode[j] = cur; queue[tail++] = j; }
    }
    if (y + 1 < height) {
      const j = cur + width;
      if (valid[j] && dist[j] < 0) { dist[j] = nd; prevNode[j] = cur; queue[tail++] = j; }
    }
  }
  return tail;
}

function reconstruct(target: number, start: number): number[] {
  const rev: number[] = [];
  let cur = target;
  while (cur !== start && cur >= 0) {
    rev.push(cur);
    cur = prevNode[cur];
  }
  rev.reverse();
  return rev;
}

function bestNeighbor(current: number, covered: Uint8Array): number {
  let best = current;
  let bestScore = covered[current] ? 0 : Math.max(0, values[current]);
  const choices = [current];
  const x = xOf(current);
  const y = yOf(current);
  if (x > 0 && valid[current - 1]) choices.push(current - 1);
  if (x + 1 < width && valid[current + 1]) choices.push(current + 1);
  if (y > 0 && valid[current - width]) choices.push(current - width);
  if (y + 1 < height && valid[current + width]) choices.push(current + width);
  for (const j of choices) {
    const gain = covered[j] ? 0 : Math.max(0, values[j]);
    const score = gain + 0.03 * density[j];
    if (score > bestScore) {
      bestScore = score;
      best = j;
    }
  }
  return best;
}

function planOne(start: number, baseCovered: Uint8Array): { path: number[]; cells: number[]; gain: number } {
  const covered = new Uint8Array(baseCovered);
  const path: number[] = [start];
  const unique = new Uint8Array(n);
  const cells: number[] = [];
  let gain = 0;

  function visit(i: number): void {
    if (values[i] > 0 && !unique[i]) {
      unique[i] = 1;
      cells.push(i);
    }
    if (values[i] > 0 && !covered[i]) {
      covered[i] = 1;
      gain += values[i];
    }
  }

  visit(start);
  let current = start;
  let stepsUsed = 0;

  while (stepsUsed < maxSteps) {
    const remaining = maxSteps - stepsUsed;
    const reached = bfs(current, remaining);
    let best = -1;
    let bestScore = -1;

    const scanTop = Math.min(positiveCells.length, Math.max(1200, Math.min(8000, 60 * remaining + 600)));
    for (let k = 0; k < scanTop; k++) {
      const c = positiveCells[k];
      const d = dist[c];
      if (d < 0 || d > remaining || covered[c]) continue;
      const urgency = remaining > 0 ? 1 + 0.08 * (remaining - d) / remaining : 1;
      const score = (values[c] + 0.22 * density[c]) * urgency / Math.pow(d + 1, 0.72);
      if (score > bestScore) {
        bestScore = score;
        best = c;
      }
    }

    if (best < 0) {
      for (let q = 0; q < reached; q++) {
        const c = queue[q];
        if (values[c] <= 0 || covered[c]) continue;
        const d = dist[c];
        const score = (values[c] + 0.15 * density[c]) / Math.pow(d + 1, 0.8);
        if (score > bestScore) {
          bestScore = score;
          best = c;
        }
      }
    }

    if (best >= 0 && best !== current) {
      const route = reconstruct(best, current);
      for (const step of route) {
        if (stepsUsed >= maxSteps) break;
        path.push(step);
        stepsUsed++;
        visit(step);
      }
      current = path[path.length - 1];
    } else {
      const next = bestNeighbor(current, covered);
      path.push(next);
      stepsUsed++;
      current = next;
      visit(current);
    }
  }

  return { path, cells, gain };
}

function cellsToCovered(counts: Uint16Array, excluded?: number[]): Uint8Array {
  const out = new Uint8Array(n);
  if (!excluded) {
    for (let i = 0; i < n; i++) if (counts[i] > 0) out[i] = 1;
    return out;
  }
  const removed = new Uint16Array(n);
  for (const c of excluded) removed[c]++;
  for (let i = 0; i < n; i++) if (counts[i] > removed[i]) out[i] = 1;
  return out;
}

function uniqueGainAgainst(counts: Uint16Array, cells: number[]): number {
  let s = 0;
  for (const c of cells) if (counts[c] === 1) s += values[c];
  return s;
}

const starts = drones.map((d) => nearestValid(Math.floor(Number(d.x) || 0), Math.floor(Number(d.y) || 0)));
const resultPaths: number[][] = new Array(drones.length);
const resultCells: number[][] = new Array(drones.length);
const counts = new Uint16Array(n);

const order = drones.map((_, i) => i).sort((a, b) => {
  const sa = starts[a] >= 0 ? density[starts[a]] + Math.max(0, values[starts[a]]) : 0;
  const sb = starts[b] >= 0 ? density[starts[b]] + Math.max(0, values[starts[b]]) : 0;
  return sb - sa;
});

for (const di of order) {
  const s = starts[di];
  if (s < 0) {
    resultPaths[di] = [];
    resultCells[di] = [];
    continue;
  }
  const plan = planOne(s, cellsToCovered(counts));
  resultPaths[di] = plan.path;
  resultCells[di] = plan.cells;
  for (const c of plan.cells) counts[c]++;
}

for (const di of order) {
  const s = starts[di];
  if (s < 0) continue;
  const oldCells = resultCells[di] || [];
  const oldUniqueGain = uniqueGainAgainst(counts, oldCells);
  const plan = planOne(s, cellsToCovered(counts, oldCells));
  if (plan.gain >= oldUniqueGain) {
    for (const c of oldCells) if (counts[c] > 0) counts[c]--;
    resultPaths[di] = plan.path;
    resultCells[di] = plan.cells;
    for (const c of plan.cells) counts[c]++;
  }
}

const outputPaths: Point[][] = resultPaths.map((p, di) => {
  if (!p || p.length === 0) {
    const s = starts[di];
    if (s >= 0) return [[xOf(s), yOf(s)]];
    return [];
  }
  return p.map((i) => [xOf(i), yOf(i)] as Point);
});

process.stdout.write(JSON.stringify({ paths: outputPaths }));
