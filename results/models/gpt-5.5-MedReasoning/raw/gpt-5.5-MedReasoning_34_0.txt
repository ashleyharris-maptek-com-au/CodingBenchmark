import sys
import math
import random


def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return 0, 0, 0, (0.0, 0.0), []
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = float(next(it))
    depot = (float(next(it)), float(next(it)))
    customers = []
    for i in range(n):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append((x, y, d))
    return n, k, cap, depot, customers


def solve(n, k, cap, depot, customers):
    if n == 0:
        return []

    coords = [depot] + [(x, y) for x, y, _ in customers]
    demands = [d for _, _, d in customers]

    def dist_node(a, b):
        ax, ay = coords[a]
        bx, by = coords[b]
        return math.hypot(ax - bx, ay - by)

    def route_distance(route):
        if not route:
            return 0.0
        total = dist_node(0, route[0] + 1)
        for a, b in zip(route, route[1:]):
            total += dist_node(a + 1, b + 1)
        total += dist_node(route[-1] + 1, 0)
        return total

    def total_distance(routes):
        return sum(route_distance(r) for r in routes)

    def two_opt(route):
        m = len(route)
        if m < 4:
            return route[:]
        r = route[:]
        improved = True
        passes = 0
        while improved and passes < 25:
            improved = False
            passes += 1
            for i in range(m - 1):
                a = 0 if i == 0 else r[i - 1] + 1
                b = r[i] + 1
                for j in range(i + 2, m + 1):
                    if j == m:
                        c = r[j - 1] + 1
                        d = 0
                    else:
                        c = r[j - 1] + 1
                        d = r[j] + 1
                    old = dist_node(a, b) + dist_node(c, d)
                    new = dist_node(a, c) + dist_node(b, d)
                    if new + 1e-9 < old:
                        r[i:j] = reversed(r[i:j])
                        improved = True
                        break
                if improved:
                    break
        return r

    def canonical(routes):
        return [r for r in routes if r]

    def improve_routes(routes):
        routes = [two_opt(r) for r in canonical(routes)]

        # A small relocate pass: move a customer to another feasible route when it
        # directly reduces distance. This is cheap and often fixes greedy choices.
        loads = [sum(demands[c] for c in r) for r in routes]
        for _ in range(3):
            changed = False
            for ri in range(len(routes)):
                pos = 0
                while pos < len(routes[ri]):
                    c = routes[ri][pos]
                    before = route_distance(routes[ri])
                    src = routes[ri][:pos] + routes[ri][pos + 1 :]
                    remove_gain = before - route_distance(src)
                    best = None
                    for rj in range(len(routes)):
                        if ri == rj or loads[rj] + demands[c] > cap + 1e-9:
                            continue
                        base = route_distance(routes[rj])
                        for ins in range(len(routes[rj]) + 1):
                            cand = routes[rj][:ins] + [c] + routes[rj][ins:]
                            delta = route_distance(cand) - base - remove_gain
                            if best is None or delta < best[0]:
                                best = (delta, rj, ins, cand)
                    if best is not None and best[0] < -1e-8:
                        _, rj, ins, cand = best
                        routes[ri] = src
                        routes[rj] = cand
                        loads[ri] -= demands[c]
                        loads[rj] += demands[c]
                        changed = True
                    else:
                        pos += 1
            routes = canonical(routes)
            loads = [sum(demands[c] for c in r) for r in routes]
            if not changed:
                break
        return [two_opt(r) for r in routes]

    def greedy_insertion(order):
        routes = [[] for _ in range(k)]
        loads = [0.0] * k
        for c in order:
            if demands[c] > cap + 1e-9:
                return None
            best = None
            for ri in range(k):
                if loads[ri] + demands[c] > cap + 1e-9:
                    continue
                base = route_distance(routes[ri])
                for pos in range(len(routes[ri]) + 1):
                    cand = routes[ri][:pos] + [c] + routes[ri][pos:]
                    delta = route_distance(cand) - base
                    # Prefer compact load usage very slightly when insertion cost ties.
                    score = delta - 1e-7 * loads[ri]
                    if best is None or score < best[0]:
                        best = (score, ri, pos)
            if best is None:
                return None
            _, ri, pos = best
            routes[ri].insert(pos, c)
            loads[ri] += demands[c]
        return improve_routes(routes)

    def sweep(order):
        routes = []
        cur = []
        load = 0.0
        for c in order:
            if demands[c] > cap + 1e-9:
                return None
            if cur and load + demands[c] > cap + 1e-9:
                routes.append(cur)
                cur = []
                load = 0.0
            cur.append(c)
            load += demands[c]
        if cur:
            routes.append(cur)
        if len(routes) > k:
            return None
        return improve_routes(routes)

    def clarke_wright():
        routes = [[i] for i in range(n)]
        loads = [demands[i] for i in range(n)]
        owner = list(range(n))
        savings = []
        for i in range(n):
            for j in range(i + 1, n):
                s = dist_node(0, i + 1) + dist_node(0, j + 1) - dist_node(i + 1, j + 1)
                savings.append((s, i, j))
        savings.sort(reverse=True)

        active = [True] * n
        count = n
        for _, i, j in savings:
            ri = owner[i]
            rj = owner[j]
            if ri == rj or not active[ri] or not active[rj]:
                continue
            a = routes[ri]
            b = routes[rj]
            if loads[ri] + loads[rj] > cap + 1e-9:
                continue

            merged = None
            if a[-1] == i and b[0] == j:
                merged = a + b
            elif a[0] == i and b[-1] == j:
                merged = b + a
            elif a[0] == i and b[0] == j:
                merged = list(reversed(a)) + b
            elif a[-1] == i and b[-1] == j:
                merged = a + list(reversed(b))
            if merged is None:
                continue

            routes[ri] = merged
            loads[ri] += loads[rj]
            active[rj] = False
            for c in merged:
                owner[c] = ri
            count -= 1

        result = [routes[i] for i in range(n) if active[i]]
        if count <= k:
            return improve_routes(result)
        return None

    def nearest_neighbor_order(start):
        unused = set(range(n))
        order = []
        current = start
        unused.remove(start)
        order.append(start)
        while unused:
            nxt = min(unused, key=lambda c: dist_node(current + 1, c + 1))
            unused.remove(nxt)
            order.append(nxt)
            current = nxt
        return order

    angles = [
        math.atan2(customers[i][1] - depot[1], customers[i][0] - depot[0])
        for i in range(n)
    ]
    radii = [dist_node(0, i + 1) for i in range(n)]

    orders = []
    base_angle = sorted(range(n), key=lambda c: angles[c])
    for step in range(min(n, 24)):
        off = (step * n) // min(n, 24)
        rotated = base_angle[off:] + base_angle[:off]
        orders.append(rotated)
        orders.append(list(reversed(rotated)))
    orders.append(sorted(range(n), key=lambda c: (-demands[c], -radii[c])))
    orders.append(sorted(range(n), key=lambda c: -radii[c]))
    orders.append(sorted(range(n), key=lambda c: (angles[c], -demands[c])))
    if n:
        for s in sorted(range(n), key=lambda c: -radii[c])[: min(8, n)]:
            orders.append(nearest_neighbor_order(s))

    rng = random.Random(1234567)
    heavy = sorted(range(n), key=lambda c: (-demands[c], -radii[c]))
    for _ in range(20):
        arr = heavy[:]
        # Shuffle in small windows so large demands are still placed early.
        for i in range(0, n, 8):
            block = arr[i : i + 8]
            rng.shuffle(block)
            arr[i : i + 8] = block
        orders.append(arr)

    candidates = []
    cw = clarke_wright()
    if cw is not None:
        candidates.append(cw)

    seen_orders = set()
    for order in orders:
        key = tuple(order)
        if key in seen_orders:
            continue
        seen_orders.add(key)
        gi = greedy_insertion(order)
        if gi is not None:
            candidates.append(gi)
        sw = sweep(order)
        if sw is not None:
            candidates.append(sw)

    if not candidates:
        # Last-resort constructive attempt. Valid feasible benchmark instances
        # should normally be solved above; this keeps output well-formed.
        routes = greedy_insertion(sorted(range(n), key=lambda c: -demands[c]))
        if routes is not None:
            candidates.append(routes)
        else:
            routes = [[i] for i in range(n)]
            candidates.append(routes[:k])

    return min(candidates, key=total_distance)


def main():
    n, k, cap, depot, customers = read_input()
    routes = solve(n, k, cap, depot, customers)

    coords = [depot] + [(x, y) for x, y, _ in customers]

    def d(a, b):
        ax, ay = coords[a]
        bx, by = coords[b]
        return math.hypot(ax - bx, ay - by)

    total = 0.0
    for r in routes:
        if not r:
            continue
        total += d(0, r[0] + 1)
        for a, b in zip(r, r[1:]):
            total += d(a + 1, b + 1)
        total += d(r[-1] + 1, 0)

    out = [f"{total:.6f}", str(len([r for r in routes if r]))]
    out.extend(" ".join(map(str, r)) for r in routes if r)
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    main()
