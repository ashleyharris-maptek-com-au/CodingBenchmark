OpCapability Shader
OpCapability Geometry
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%bool = OpTypeBool
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_02 = OpConstant %float 0.02
%float_0_333333343 = OpConstant %float 0.333333343
%float_0_666666687 = OpConstant %float 0.666666687
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%fun = OpTypeFunction %void
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %fun
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr
%p3 = OpLoad %v4float %p3_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%dir_start = OpFSub %v2float %p2_xy %p0_xy
%dir_end = OpFSub %v2float %p3_xy %p1_xy
%tan_start = OpExtInst %v2float %GLSLstd450 Normalize %dir_start
%tan_end = OpExtInst %v2float %GLSLstd450 Normalize %dir_end
%tan_start_x = OpCompositeExtract %float %tan_start 0
%tan_start_y = OpCompositeExtract %float %tan_start 1
%neg_tan_start_y = OpFNegate %float %tan_start_y
%perp_start = OpCompositeConstruct %v2float %neg_tan_start_y %tan_start_x
%tan_end_x = OpCompositeExtract %float %tan_end 0
%tan_end_y = OpCompositeExtract %float %tan_end 1
%neg_tan_end_y = OpFNegate %float %tan_end_y
%perp_end = OpCompositeConstruct %v2float %neg_tan_end_y %tan_end_x
%off_start = OpVectorTimesScalar %v2float %perp_start %float_0_02
%off_end = OpVectorTimesScalar %v2float %perp_end %float_0_02
%delta_p = OpFSub %v4float %p2 %p1
%delta_p_one = OpVectorTimesScalar %v4float %delta_p %float_0_333333343
%delta_p_two = OpVectorTimesScalar %v4float %delta_p %float_0_666666687
%q0 = OpCopyObject %v4float %p1
%q1 = OpFAdd %v4float %p1 %delta_p_one
%q2 = OpFAdd %v4float %p1 %delta_p_two
%q3 = OpCopyObject %v4float %p2
%delta_off = OpFSub %v2float %off_end %off_start
%delta_off_one = OpVectorTimesScalar %v2float %delta_off %float_0_333333343
%delta_off_two = OpVectorTimesScalar %v2float %delta_off %float_0_666666687
%off0 = OpCopyObject %v2float %off_start
%off1 = OpFAdd %v2float %off_start %delta_off_one
%off2 = OpFAdd %v2float %off_start %delta_off_two
%off3 = OpCopyObject %v2float %off_end
%delta_c = OpFSub %v3float %c2 %c1
%delta_c_one = OpVectorTimesScalar %v3float %delta_c %float_0_333333343
%delta_c_two = OpVectorTimesScalar %v3float %delta_c %float_0_666666687
%col0 = OpCopyObject %v3float %c1
%col1 = OpFAdd %v3float %c1 %delta_c_one
%col2 = OpFAdd %v3float %c1 %delta_c_two
%col3 = OpCopyObject %v3float %c2
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%off0_x = OpCompositeExtract %float %off0 0
%off0_y = OpCompositeExtract %float %off0 1
%off0_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0
%v0_pos = OpFAdd %v4float %q0 %off0_v4
OpStore %pos_out_ptr %v0_pos
OpStore %outColor %col0
OpEmitVertex
%v1_pos = OpFSub %v4float %q0 %off0_v4
OpStore %pos_out_ptr %v1_pos
OpStore %outColor %col0
OpEmitVertex
%off1_x = OpCompositeExtract %float %off1 0
%off1_y = OpCompositeExtract %float %off1 1
%off1_v4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0
%v2_pos = OpFAdd %v4float %q1 %off1_v4
OpStore %pos_out_ptr %v2_pos
OpStore %outColor %col1
OpEmitVertex
%v3_pos = OpFSub %v4float %q1 %off1_v4
OpStore %pos_out_ptr %v3_pos
OpStore %outColor %col1
OpEmitVertex
OpEndPrimitive
%off2_x = OpCompositeExtract %float %off2 0
%off2_y = OpCompositeExtract %float %off2 1
%off2_v4 = OpCompositeConstruct %v4float %off2_x %off2_y %float_0 %float_0
%v4_pos = OpFAdd %v4float %q2 %off2_v4
OpStore %pos_out_ptr %v4_pos
OpStore %outColor %col2
OpEmitVertex
%v5_pos = OpFSub %v4float %q2 %off2_v4
OpStore %pos_out_ptr %v5_pos
OpStore %outColor %col2
OpEmitVertex
%off3_x = OpCompositeExtract %float %off3 0
%off3_y = OpCompositeExtract %float %off3 1
%off3_v4 = OpCompositeConstruct %v4float %off3_x %off3_y %float_0 %float_0
%v6_pos = OpFAdd %v4float %q3 %off3_v4
OpStore %pos_out_ptr %v6_pos
OpStore %outColor %col3
OpEmitVertex
%v7_pos = OpFSub %v4float %q3 %off3_v4
OpStore %pos_out_ptr %v7_pos
OpStore %outColor %col3
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
