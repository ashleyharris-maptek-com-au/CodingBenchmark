using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    struct Box
    {
        public int Id;
        public long W, H, D, V;
        public long MaxSide => Math.Max(W, Math.Max(H, D));
        public long Footprint => W * D;
    }

    struct Place
    {
        public int Id;
        public long X, Y, Z;
        public long W, H, D;
        public long V => W * H * D;
    }

    sealed class FastScanner
    {
        readonly System.IO.Stream input = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 16];
        int len, ptr;

        int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }

        public long NextLong()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            long value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = (int)fs.NextLong();
        long W = fs.NextLong(), H = fs.NextLong(), D = fs.NextLong();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            long w = fs.NextLong(), h = fs.NextLong(), d = fs.NextLong();
            boxes[i] = new Box { Id = i, W = w, H = h, D = d, V = w * h * d };
        }

        var best = new List<Place>();
        long containerVolume = SafeMul(W, SafeMul(H, D));

        if (W <= 80 && H <= 80 && D <= 80 && containerVolume <= 200000)
            best = Better(best, VoxelPack(boxes, W, H, D));

        if (n <= 1500)
            best = Better(best, ExtremePointPack(boxes, W, H, D, 6000));

        for (int order = 0; order < 6; order++)
            best = Better(best, ShelfPack(boxes, W, H, D, order));

        var sb = new StringBuilder();
        sb.AppendLine(best.Count.ToString());
        foreach (var p in best)
            sb.Append(p.Id).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).AppendLine();
        Console.Write(sb.ToString());
    }

    static long SafeMul(long a, long b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    static List<Place> Better(List<Place> a, List<Place> b)
    {
        if (b.Count != a.Count) return b.Count > a.Count ? b : a;
        long va = 0, vb = 0;
        foreach (var p in a) va += p.V;
        foreach (var p in b) vb += p.V;
        return vb > va ? b : a;
    }

    static Box[] Ordered(Box[] boxes, int mode)
    {
        var a = (Box[])boxes.Clone();
        Array.Sort(a, (x, y) =>
        {
            int c = mode switch
            {
                0 => x.V.CompareTo(y.V),
                1 => x.MaxSide.CompareTo(y.MaxSide),
                2 => x.H.CompareTo(y.H),
                3 => x.Footprint.CompareTo(y.Footprint),
                4 => (x.W + x.H + x.D).CompareTo(y.W + y.H + y.D),
                _ => y.V.CompareTo(x.V)
            };
            if (c != 0) return c;
            c = x.MaxSide.CompareTo(y.MaxSide);
            return c != 0 ? c : x.Id.CompareTo(y.Id);
        });
        return a;
    }

    static List<Place> ShelfPack(Box[] input, long W, long H, long D, int orderMode)
    {
        var boxes = Ordered(input, orderMode);
        var placed = new List<Place>();

        long x = 0, y = 0, z = 0;
        long rowDepth = 0, layerHeight = 0;

        foreach (var b in boxes)
        {
            if (b.W > W || b.H > H || b.D > D) continue;

            if (x + b.W <= W && z + b.D <= D && y + b.H <= H)
            {
                Add();
                continue;
            }

            long nx = 0, nz = z + rowDepth;
            if (nz + b.D <= D && y + b.H <= H)
            {
                x = nx;
                z = nz;
                rowDepth = 0;
                Add();
                continue;
            }

            long ny = y + layerHeight;
            if (ny + b.H <= H)
            {
                x = 0;
                z = 0;
                y = ny;
                rowDepth = 0;
                layerHeight = 0;
                if (b.W <= W && b.D <= D)
                    Add();
            }

            void Add()
            {
                placed.Add(new Place { Id = b.Id, X = x, Y = y, Z = z, W = b.W, H = b.H, D = b.D });
                x += b.W;
                if (b.D > rowDepth) rowDepth = b.D;
                if (b.H > layerHeight) layerHeight = b.H;
            }
        }

        return placed;
    }

    static List<Place> ExtremePointPack(Box[] input, long W, long H, long D, int pointLimit)
    {
        var boxes = Ordered(input, 0);
        var placed = new List<Place>();
        var points = new List<(long X, long Y, long Z)> { (0, 0, 0) };
        var seen = new HashSet<string> { "0,0,0" };

        foreach (var b in boxes)
        {
            if (b.W > W || b.H > H || b.D > D) continue;

            int bestIndex = -1;
            (long X, long Y, long Z) bestPoint = default;
            for (int i = 0; i < points.Count; i++)
            {
                var p = points[i];
                if (p.X + b.W > W || p.Y + b.H > H || p.Z + b.D > D) continue;
                if (!OverlapsAny(placed, p.X, p.Y, p.Z, b.W, b.H, b.D))
                {
                    if (bestIndex < 0 || BetterPoint(p, bestPoint))
                    {
                        bestIndex = i;
                        bestPoint = p;
                    }
                }
            }

            if (bestIndex < 0) continue;

            points.RemoveAt(bestIndex);
            placed.Add(new Place { Id = b.Id, X = bestPoint.X, Y = bestPoint.Y, Z = bestPoint.Z, W = b.W, H = b.H, D = b.D });
            TryAdd(bestPoint.X + b.W, bestPoint.Y, bestPoint.Z);
            TryAdd(bestPoint.X, bestPoint.Y + b.H, bestPoint.Z);
            TryAdd(bestPoint.X, bestPoint.Y, bestPoint.Z + b.D);

            if (points.Count > pointLimit)
            {
                points.Sort((a, c) => a.Y != c.Y ? a.Y.CompareTo(c.Y) : a.Z != c.Z ? a.Z.CompareTo(c.Z) : a.X.CompareTo(c.X));
                points.RemoveRange(pointLimit, points.Count - pointLimit);
            }

            void TryAdd(long px, long py, long pz)
            {
                if (px >= W || py >= H || pz >= D) return;
                if (InsideAny(placed, px, py, pz)) return;
                string key = px + "," + py + "," + pz;
                if (seen.Add(key)) points.Add((px, py, pz));
            }
        }

        return placed;
    }

    static bool BetterPoint((long X, long Y, long Z) a, (long X, long Y, long Z) b)
    {
        if (a.Y != b.Y) return a.Y < b.Y;
        if (a.Z != b.Z) return a.Z < b.Z;
        return a.X < b.X;
    }

    static bool InsideAny(List<Place> placed, long x, long y, long z)
    {
        foreach (var p in placed)
            if (x >= p.X && x < p.X + p.W && y >= p.Y && y < p.Y + p.H && z >= p.Z && z < p.Z + p.D)
                return true;
        return false;
    }

    static bool OverlapsAny(List<Place> placed, long x, long y, long z, long w, long h, long d)
    {
        foreach (var p in placed)
            if (x < p.X + p.W && x + w > p.X &&
                y < p.Y + p.H && y + h > p.Y &&
                z < p.Z + p.D && z + d > p.Z)
                return true;
        return false;
    }

    static List<Place> VoxelPack(Box[] input, long Wl, long Hl, long Dl)
    {
        int W = (int)Wl, H = (int)Hl, D = (int)Dl;
        var occ = new bool[W * H * D];
        var placed = new List<Place>();
        var boxes = Ordered(input, 0);

        foreach (var b in boxes)
        {
            if (b.W > W || b.H > H || b.D > D) continue;
            int bw = (int)b.W, bh = (int)b.H, bd = (int)b.D;
            bool done = false;
            for (int y = 0; y <= H - bh && !done; y++)
            for (int z = 0; z <= D - bd && !done; z++)
            for (int x = 0; x <= W - bw && !done; x++)
            {
                if (!CanPlaceVoxel(occ, W, D, x, y, z, bw, bh, bd)) continue;
                FillVoxel(occ, W, D, x, y, z, bw, bh, bd);
                placed.Add(new Place { Id = b.Id, X = x, Y = y, Z = z, W = bw, H = bh, D = bd });
                done = true;
            }
        }

        return placed;
    }

    static bool CanPlaceVoxel(bool[] occ, int W, int D, int x, int y, int z, int bw, int bh, int bd)
    {
        for (int yy = y; yy < y + bh; yy++)
        for (int zz = z; zz < z + bd; zz++)
        {
            int baseIndex = (yy * D + zz) * W + x;
            for (int xx = 0; xx < bw; xx++)
                if (occ[baseIndex + xx]) return false;
        }
        return true;
    }

    static void FillVoxel(bool[] occ, int W, int D, int x, int y, int z, int bw, int bh, int bd)
    {
        for (int yy = y; yy < y + bh; yy++)
        for (int zz = z; zz < z + bd; zz++)
        {
            int baseIndex = (yy * D + zz) * W + x;
            for (int xx = 0; xx < bw; xx++)
                occ[baseIndex + xx] = true;
        }
    }
}
