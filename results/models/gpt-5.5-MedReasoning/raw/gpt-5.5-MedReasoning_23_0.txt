use std::io::{self, Read, Write, BufWriter};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone, Copy, Debug)]
struct World {
    gravity: f64,
    max_thrust: f64,
    fuel0: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    pad_left: f64,
    pad_right: f64,
}

struct FastInput<R: Read> {
    r: R,
    buf: [u8; 1 << 16],
    pos: usize,
    len: usize,
}

impl<R: Read> FastInput<R> {
    fn new(r: R) -> Self {
        Self { r, buf: [0; 1 << 16], pos: 0, len: 0 }
    }

    fn byte(&mut self) -> Option<u8> {
        if self.pos >= self.len {
            self.len = self.r.read(&mut self.buf).ok()?;
            self.pos = 0;
            if self.len == 0 {
                return None;
            }
        }
        let b = self.buf[self.pos];
        self.pos += 1;
        Some(b)
    }

    fn token(&mut self) -> Option<String> {
        let mut b = self.byte()?;
        while b.is_ascii_whitespace() {
            b = self.byte()?;
        }
        let mut v = Vec::with_capacity(32);
        while !b.is_ascii_whitespace() {
            v.push(b);
            match self.byte() {
                Some(n) => b = n,
                None => break,
            }
        }
        String::from_utf8(v).ok()
    }

    fn f64(&mut self) -> Option<f64> {
        self.token()?.parse().ok()
    }

    fn usize(&mut self) -> Option<usize> {
        self.token()?.parse().ok()
    }

    fn scan_map_rows(&mut self, w: usize, h: usize, tx: f64, ty: f64) -> (f64, f64, f64) {
        let mut best_left = tx.floor();
        let mut best_right = tx.floor() + 1.0;
        let mut best_y = ty.floor();
        let mut best_score = f64::INFINITY;

        for row in 0..h {
            let mut col = 0usize;
            while col < w {
                let mut b = match self.byte() {
                    Some(x) => x,
                    None => return (best_left, best_right, best_y),
                };
                while b.is_ascii_whitespace() {
                    b = match self.byte() {
                        Some(x) => x,
                        None => return (best_left, best_right, best_y),
                    };
                }

                let mut run_start: Option<usize> = None;
                let mut last_l: Option<usize> = None;
                loop {
                    if b == b'L' || b == b'T' {
                        if run_start.is_none() {
                            run_start = Some(col);
                        }
                        last_l = Some(col);
                    } else if let (Some(a), Some(z)) = (run_start, last_l) {
                        let cy = row as f64 + 0.5;
                        let cx = (a as f64 + z as f64 + 1.0) * 0.5;
                        let contains_target = (tx >= a as f64 && tx <= z as f64 + 1.0
                            && (ty - cy).abs() < 3.0) as i32;
                        let score = if contains_target != 0 {
                            -1.0
                        } else {
                            (cx - tx).abs() + 4.0 * (cy - ty).abs()
                        };
                        if score < best_score {
                            best_score = score;
                            best_left = a as f64;
                            best_right = z as f64 + 1.0;
                            best_y = cy;
                        }
                        run_start = None;
                        last_l = None;
                    }

                    col += 1;
                    if col >= w {
                        break;
                    }
                    b = match self.byte() {
                        Some(x) => x,
                        None => return (best_left, best_right, best_y),
                    };
                }

                if let (Some(a), Some(z)) = (run_start, last_l) {
                    let cy = row as f64 + 0.5;
                    let cx = (a as f64 + z as f64 + 1.0) * 0.5;
                    let contains_target = (tx >= a as f64 && tx <= z as f64 + 1.0
                        && (ty - cy).abs() < 3.0) as i32;
                    let score = if contains_target != 0 {
                        -1.0
                    } else {
                        (cx - tx).abs() + 4.0 * (cy - ty).abs()
                    };
                    if score < best_score {
                        best_left = a as f64;
                        best_right = z as f64 + 1.0;
                        best_y = cy;
                    }
                }
            }
        }
        (best_left, best_right, best_y)
    }
}

fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo { lo } else if x > hi { hi } else { x }
}

fn parse_initial<R: Read>(inp: &mut FastInput<R>) -> Option<World> {
    let width = inp.usize()?;
    let height = inp.usize()?;
    let gravity = inp.f64()?;
    let max_thrust = inp.f64()?;
    let fuel0 = inp.f64()?;
    let start_x = inp.f64()?;
    let start_y = inp.f64()?;
    let target_x = inp.f64()?;
    let target_y = inp.f64()?;
    let (pad_left, pad_right, pad_y) = inp.scan_map_rows(width, height, target_x, target_y);
    Some(World {
        gravity, max_thrust, fuel0, start_x, start_y,
        target_x: (pad_left + pad_right) * 0.5,
        target_y: pad_y,
        pad_left,
        pad_right,
    })
}

fn control(w: &World, s: State) -> (f64, f64) {
    let pad_margin = 1.2;
    let pad_center = w.target_x;
    let usable_left = w.pad_left + pad_margin;
    let usable_right = w.pad_right - pad_margin;
    let near_ground = (w.target_y - s.y).max(0.0);
    let mut aim_x = pad_center;

    if s.x < usable_left {
        aim_x = usable_left.min(pad_center);
    } else if s.x > usable_right {
        aim_x = usable_right.max(pad_center);
    }

    let dx = aim_x - s.x;
    let max_side_speed = if near_ground < 20.0 { 1.0 } else if near_ground < 70.0 { 4.0 } else { 12.0 };
    let desired_vx = clamp(dx * 0.10, -max_side_speed, max_side_speed);
    let ax_cmd = clamp((desired_vx - s.vx) * 0.75, -w.max_thrust * 0.55, w.max_thrust * 0.55);

    let desired_vy = if near_ground > 180.0 {
        24.0
    } else if near_ground > 80.0 {
        14.0
    } else if near_ground > 25.0 {
        6.0
    } else {
        clamp(near_ground * 0.16 + 0.35, 0.35, 3.5)
    };

    let mut ay_cmd = clamp((desired_vy - s.vy) * 0.95, -w.max_thrust * 0.95, w.gravity * 0.8);

    let speed = (s.vx * s.vx + s.vy * s.vy).sqrt();
    if near_ground < 12.0 && speed > 1.8 {
        ay_cmd = ay_cmd.min(-w.max_thrust * 0.65);
    }
    if near_ground < 4.0 {
        ay_cmd = ay_cmd.min(-w.max_thrust * 0.35);
    }

    let thrust_x = ax_cmd;
    let thrust_y = ay_cmd - w.gravity;
    let mut desired_angle = thrust_x.atan2(-thrust_y);
    let angle_limit = if near_ground < 18.0 { 0.28 } else if near_ground < 55.0 { 0.55 } else { 0.95 };
    desired_angle = clamp(desired_angle, -angle_limit, angle_limit);

    let angle_err = desired_angle - s.angle;
    let turn = clamp(angle_err * 5.0 - s.angular_vel * 1.8, -1.0, 1.0);

    let vertical_authority = (w.max_thrust * s.angle.cos().max(0.20)).max(0.1);
    let mut thrust = (w.gravity - ay_cmd) / vertical_authority;
    thrust += (s.angle - desired_angle).abs() * 0.10;

    if s.fuel <= 0.05 {
        thrust = 0.0;
    } else if near_ground < 40.0 || s.vy > desired_vy + 1.0 || speed > 10.0 {
        thrust = thrust.max(0.15);
    }

    (clamp(thrust, 0.0, 1.0), turn)
}

fn main() {
    let mut inp = FastInput::new(io::stdin());
    let world = match parse_initial(&mut inp) {
        Some(w) => w,
        None => return,
    };

    let latest = Arc::new(Mutex::new(State {
        x: world.start_x,
        y: world.start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: world.fuel0,
    }));
    let reader_latest = latest.clone();

    thread::spawn(move || {
        while let (Some(x), Some(y), Some(vx), Some(vy), Some(angle), Some(angular_vel), Some(fuel)) =
            (inp.f64(), inp.f64(), inp.f64(), inp.f64(), inp.f64(), inp.f64(), inp.f64())
        {
            if let Ok(mut st) = reader_latest.lock() {
                *st = State { x, y, vx, vy, angle, angular_vel, fuel };
            }
        }
    });

    let stdout = io::stdout();
    let mut out = BufWriter::with_capacity(1 << 16, stdout.lock());
    let frame = Duration::from_millis(20);
    let mut next = Instant::now();

    loop {
        let st = match latest.lock() {
            Ok(g) => *g,
            Err(_) => break,
        };
        let (thrust, turn) = control(&world, st);
        if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
            break;
        }
        let _ = out.flush();

        next += frame;
        let now = Instant::now();
        if next > now {
            thread::sleep(next - now);
        } else {
            next = now;
        }
    }
}
