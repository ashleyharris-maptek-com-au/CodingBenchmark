#include <algorithm>
#include <atomic>
#include <cmath>
#include <cstdint>
#include <fstream>
#include <iostream>
#include <limits>
#include <string>
#include <thread>
#include <vector>

using std::atomic;
using std::cerr;
using std::cin;
using std::cout;
using std::ifstream;
using std::ios;
using std::string;
using std::thread;
using std::uint64_t;

static constexpr double PI = 3.141592653589793238462643383279502884;

static double clampd(double v, double lo, double hi) {
    return std::max(lo, std::min(hi, v));
}

static double wrap_pi(double a) {
    while (a > PI) a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

static double safe_hypot3(double x, double y, double z) {
    return std::sqrt(x * x + y * y + z * z);
}

static uint16_t read_le16(const unsigned char* p) {
    return static_cast<uint16_t>(p[0] | (static_cast<uint16_t>(p[1]) << 8));
}

struct TerrainInfo {
    atomic<int> max_solid_top{0};
    atomic<bool> done{false};
};

static void scan_terrain_async(string path, uint64_t width, uint64_t depth, TerrainInfo* info) {
    ifstream f(path, ios::binary);
    if (!f) {
        info->done.store(true, std::memory_order_release);
        return;
    }

    const size_t BUF_SIZE = 1u << 20; // multiple of 6 is not required; remainder is carried.
    std::vector<unsigned char> buf(BUF_SIZE + 6);
    size_t carry = 0;
    int local_max = 0;
    uint64_t cells_seen = 0;
    const uint64_t total_cells = width * depth;

    while (f && cells_seen < total_cells) {
        f.read(reinterpret_cast<char*>(buf.data() + carry), static_cast<std::streamsize>(BUF_SIZE));
        size_t got = static_cast<size_t>(f.gcount()) + carry;
        size_t usable = got - (got % 6);

        for (size_t i = 0; i + 5 < usable && cells_seen < total_cells; i += 6, ++cells_seen) {
            uint16_t a = read_le16(buf.data() + i);
            uint16_t b = read_le16(buf.data() + i + 2);
            uint16_t c = read_le16(buf.data() + i + 4);
            int top = (b == c) ? static_cast<int>(a) : static_cast<int>(std::max(a, c));
            if (top > local_max) local_max = top;
        }

        info->max_solid_top.store(local_max, std::memory_order_release);
        carry = got - usable;
        if (carry) {
            std::copy(buf.data() + usable, buf.data() + got, buf.data());
        }
    }

    info->max_solid_top.store(local_max, std::memory_order_release);
    info->done.store(true, std::memory_order_release);
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);
    cout.setf(std::ios::fixed);
    cout.precision(6);

    double width_d, depth_d, height, gravity, max_thrust, fuel0;
    double start_x, start_y, start_z, target_x, target_y, target_z;

    if (!(cin >> width_d >> depth_d >> height >> gravity >> max_thrust >> fuel0)) return 0;
    if (!(cin >> start_x >> start_y >> start_z >> target_x >> target_y >> target_z)) return 0;

    string terrain_path;
    std::getline(cin, terrain_path); // end of previous numeric line
    std::getline(cin, terrain_path);

    string state_marker;
    cin >> state_marker;

    uint64_t width = static_cast<uint64_t>(std::max(0.0, std::llround(width_d)));
    uint64_t depth = static_cast<uint64_t>(std::max(0.0, std::llround(depth_d)));

    TerrainInfo* terrain = new TerrainInfo();
    thread scanner(scan_terrain_async, terrain_path, width, depth, terrain);
    scanner.detach();

    const double world_min_xy = std::max(1.0, std::min(width_d, depth_d));
    const double landing_radius = std::max(5.0, world_min_xy * 0.02);
    const double world_scale = std::max({100.0, width_d, depth_d, height});
    const double effective_max_thrust = std::max(max_thrust, gravity + 0.1);

    double x, y, z, vx, vy, vz, pitch, yaw, roll, pitch_rate, yaw_rate, roll_rate, fuel_remaining;

    while (cin >> x >> y >> z >> vx >> vy >> vz
               >> pitch >> yaw >> roll >> pitch_rate >> yaw_rate >> roll_rate >> fuel_remaining) {
        const double rx = target_x - x;
        const double ry = target_y - y;
        const double dist_xy = std::hypot(rx, ry);

        const double max_solid = static_cast<double>(terrain->max_solid_top.load(std::memory_order_acquire));
        const double ceiling_margin = std::max(8.0, std::min(250.0, height * 0.04));
        const double soft_ceiling = std::max(target_z + 5.0, height - ceiling_margin);

        double route_clearance = clampd(dist_xy * 0.20 + landing_radius * 2.0, 20.0, world_scale * 0.12);
        double cruise_z = std::max(target_z + route_clearance, max_solid + std::max(12.0, landing_radius));
        cruise_z = clampd(cruise_z, std::min(target_z + 8.0, soft_ceiling), soft_ceiling);

        double desired_z;
        if (dist_xy > landing_radius * 8.0) {
            desired_z = cruise_z;
        } else {
            double glide = clampd(dist_xy * 0.45 + 4.0, 2.0, std::max(6.0, cruise_z - target_z));
            desired_z = target_z + glide;
        }
        desired_z = std::min(desired_z, soft_ceiling);

        const double near_pad = clampd(1.0 - dist_xy / std::max(landing_radius * 3.0, 1.0), 0.0, 1.0);
        const double max_tilt = (near_pad > 0.0) ? (0.72 - 0.42 * near_pad) : 0.72;
        const double usable_acc = std::max(0.5, effective_max_thrust - gravity);
        const double horiz_acc_limit = std::max(0.3, std::min(usable_acc * 0.75, gravity * 1.8));

        double dir_x = 0.0, dir_y = 0.0;
        if (dist_xy > 1e-6) {
            dir_x = rx / dist_xy;
            dir_y = ry / dist_xy;
        }

        double vmax_far = clampd(std::sqrt(std::max(0.0, 2.0 * horiz_acc_limit * dist_xy)) * 0.65,
                                 2.0, std::max(12.0, world_scale * 0.035));
        double vmax_near = std::max(1.2, landing_radius * 0.25);
        double blend_near = clampd(dist_xy / std::max(landing_radius * 6.0, 1.0), 0.0, 1.0);
        double vmax = vmax_near * (1.0 - blend_near) + vmax_far * blend_near;

        double desired_vx = dir_x * vmax;
        double desired_vy = dir_y * vmax;
        if (dist_xy < landing_radius * 1.2) {
            desired_vx = rx * 0.18;
            desired_vy = ry * 0.18;
        }

        double ax = clampd((desired_vx - vx) * 0.55, -horiz_acc_limit, horiz_acc_limit);
        double ay = clampd((desired_vy - vy) * 0.55, -horiz_acc_limit, horiz_acc_limit);
        double ah = std::hypot(ax, ay);
        if (ah > horiz_acc_limit) {
            ax *= horiz_acc_limit / ah;
            ay *= horiz_acc_limit / ah;
            ah = horiz_acc_limit;
        }

        double max_desc = (dist_xy < landing_radius * 2.5) ? 1.2 : clampd(std::sqrt(std::max(0.0, 2.0 * usable_acc * std::max(0.0, z - target_z))) * 0.45,
                                                                           2.0, std::max(8.0, world_scale * 0.01));
        double max_asc = std::max(4.0, std::min(25.0, usable_acc * 3.0));
        double desired_vz = clampd((desired_z - z) * 0.35, -max_desc, max_asc);
        if (dist_xy < landing_radius && z - target_z < 8.0) {
            desired_vz = clampd(-0.45 - 0.08 * (z - target_z), -1.1, -0.25);
        }

        double az_cmd = clampd((desired_vz - vz) * 0.75, -gravity * 0.9, usable_acc * 0.9);
        double up_acc = clampd(gravity + az_cmd, gravity * 0.15, effective_max_thrust * 0.98);

        double tilt_limited_h = std::tan(max_tilt) * std::max(0.1, up_acc);
        ah = std::hypot(ax, ay);
        if (ah > tilt_limited_h) {
            ax *= tilt_limited_h / ah;
            ay *= tilt_limited_h / ah;
            ah = tilt_limited_h;
        }

        double thrust_acc = safe_hypot3(ax, ay, up_acc);
        double thrust = clampd(thrust_acc / effective_max_thrust, 0.0, 1.0);

        double desired_pitch = 0.0;
        double desired_yaw = yaw;
        if (ah > 1e-5 && thrust_acc > 1e-5) {
            desired_pitch = clampd(std::asin(clampd(ah / thrust_acc, -1.0, 1.0)), -max_tilt, max_tilt);
            desired_yaw = std::atan2(-ay, -ax);
        }

        // When nearly centered, prioritize killing residual tilt over chasing tiny lateral errors.
        if (dist_xy < landing_radius * 0.5 && std::hypot(vx, vy) < 0.8) {
            desired_pitch = 0.0;
            desired_yaw = yaw;
        }

        double pitch_err = wrap_pi(desired_pitch - pitch);
        double yaw_err = wrap_pi(desired_yaw - yaw);
        double roll_err = wrap_pi(0.0 - roll);

        double pitch_cmd = clampd(2.8 * pitch_err - 1.35 * pitch_rate, -1.0, 1.0);
        double yaw_cmd = clampd(2.4 * yaw_err - 1.20 * yaw_rate, -1.0, 1.0);
        double roll_cmd = clampd(2.0 * roll_err - 1.10 * roll_rate, -1.0, 1.0);

        // If descending too fast very close to the pad, temporarily give vertical braking priority.
        if (dist_xy < landing_radius * 1.5 && z > target_z && vz < -2.2) {
            thrust = clampd((gravity + std::min(usable_acc, (-vz) * 0.9)) / effective_max_thrust, thrust, 1.0);
            pitch_cmd = clampd(pitch_cmd - 0.6 * pitch, -1.0, 1.0);
        }

        cout << thrust << ' ' << pitch_cmd << ' ' << yaw_cmd << ' ' << roll_cmd << '\n' << std::flush;
    }

    return 0;
}
