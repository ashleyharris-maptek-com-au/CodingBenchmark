use std::io::{self, Read};

const WORD_BITS: usize = 64;
const UNSET_PREV: u64 = u64::MAX;
const DP_TARGET_LIMIT: usize = 50_000_000;
const OFFSET_RANGE_LIMIT: usize = 8_000_000;
const SPARSE_LIMIT: usize = 2_000_000;
const MITM_LIMIT: usize = 44;

struct Scanner {
    data: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.data.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.data[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        } else if self.data[self.pos] == b'+' {
            self.pos += 1;
        }
        let mut value = 0i64;
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_digit() {
            value = value
                .saturating_mul(10)
                .saturating_add((self.data[self.pos] - b'0') as i64);
            self.pos += 1;
        }
        Some(value.saturating_mul(sign))
    }
}

fn print_yes(mut indices: Vec<u32>) {
    indices.sort_unstable();
    println!("YES");
    if indices.is_empty() {
        println!();
        return;
    }
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{idx}");
    }
    println!();
}

fn print_no() {
    println!("NO");
}

fn gcd_i128(mut a: i128, mut b: i128) -> i128 {
    if a < 0 {
        a = -a;
    }
    if b < 0 {
        b = -b;
    }
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn check_single(values: &[i64], target: i64) -> Option<Vec<u32>> {
    for (i, &value) in values.iter().enumerate() {
        if value == target {
            return Some(vec![i as u32]);
        }
    }
    None
}

fn meet_in_middle(values: &[i64], target: i64) -> Option<Vec<u32>> {
    let n = values.len();
    if n > MITM_LIMIT {
        return None;
    }
    let left_len = n / 2;
    let right_len = n - left_len;
    let left_count = 1usize << left_len;
    let right_count = 1usize << right_len;
    let mut left = Vec::with_capacity(left_count);
    for mask in 0..left_count {
        let mut sum = 0i128;
        for (j, &value) in values[..left_len].iter().enumerate() {
            if ((mask >> j) & 1) != 0 {
                sum += value as i128;
            }
        }
        left.push((sum, mask as u64));
    }
    left.sort_unstable_by_key(|entry| entry.0);
    for mask in 0..right_count {
        let mut sum = 0i128;
        for (j, &value) in values[left_len..].iter().enumerate() {
            if ((mask >> j) & 1) != 0 {
                sum += value as i128;
            }
        }
        let need = target as i128 - sum;
        if let Ok(pos) = left.binary_search_by_key(&need, |entry| entry.0) {
            let mut result = Vec::new();
            let left_mask = left[pos].1;
            for j in 0..left_len {
                if ((left_mask >> j) & 1) != 0 {
                    result.push(j as u32);
                }
            }
            for j in 0..right_len {
                if ((mask >> j) & 1) != 0 {
                    result.push((left_len + j) as u32);
                }
            }
            return Some(result);
        }
    }
    None
}

fn clear_extra_bits(bits: &mut [u64], size: usize) {
    let extra = bits.len() * WORD_BITS - size;
    if extra > 0 {
        let keep = WORD_BITS - extra;
        let mask = if keep == WORD_BITS {
            u64::MAX
        } else {
            (1u64 << keep) - 1
        };
        if let Some(last) = bits.last_mut() {
            *last &= mask;
        }
    }
}

fn nonnegative_bitset_dp(values: &[i64], target: usize) -> Option<Vec<u32>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if target > DP_TARGET_LIMIT {
        return None;
    }
    let words = target / WORD_BITS + 1;
    let mut bits = vec![0u64; words];
    let mut shifted = vec![0u64; words];
    let mut parent_prev = vec![UNSET_PREV; target + 1];
    let mut parent_idx = vec![u32::MAX; target + 1];
    bits[0] = 1;
    parent_prev[0] = 0;

    for (idx, &raw_value) in values.iter().enumerate() {
        if raw_value <= 0 {
            continue;
        }
        let value = raw_value as usize;
        if value > target {
            continue;
        }
        shifted.fill(0);
        let word_shift = value / WORD_BITS;
        let bit_shift = value % WORD_BITS;
        for src in 0..words {
            let word = bits[src];
            if word == 0 {
                continue;
            }
            let dst = src + word_shift;
            if dst < words {
                shifted[dst] |= word << bit_shift;
            }
            if bit_shift != 0 && dst + 1 < words {
                shifted[dst + 1] |= word >> (WORD_BITS - bit_shift);
            }
        }
        clear_extra_bits(&mut shifted, target + 1);
        for w in 0..words {
            let mut newly = shifted[w] & !bits[w];
            while newly != 0 {
                let bit = newly.trailing_zeros() as usize;
                let sum = w * WORD_BITS + bit;
                if sum <= target {
                    parent_prev[sum] = (sum - value) as u64;
                    parent_idx[sum] = idx as u32;
                }
                newly &= newly - 1;
            }
            bits[w] |= shifted[w];
        }
        if ((bits[target / WORD_BITS] >> (target % WORD_BITS)) & 1) != 0 {
            let mut result = Vec::new();
            let mut cur = target;
            while cur != 0 {
                let idx_at = parent_idx[cur];
                if idx_at == u32::MAX {
                    return None;
                }
                result.push(idx_at);
                cur = parent_prev[cur] as usize;
            }
            return Some(result);
        }
    }
    None
}

fn offset_bitset_dp(values: &[i64], target: i64) -> Option<Vec<u32>> {
    let mut min_sum = 0i128;
    let mut max_sum = 0i128;
    for &value in values {
        if value < 0 {
            min_sum += value as i128;
        } else {
            max_sum += value as i128;
        }
    }
    let target_i = target as i128;
    if target_i < min_sum || target_i > max_sum {
        return None;
    }
    let range_i = max_sum - min_sum + 1;
    if range_i <= 0 || range_i as usize > OFFSET_RANGE_LIMIT {
        return None;
    }
    let range = range_i as usize;
    let offset = (-min_sum) as usize;
    let target_pos = (target_i - min_sum) as usize;
    let words = range / WORD_BITS + 1;
    let mut bits = vec![0u64; words];
    let mut shifted = vec![0u64; words];
    let mut parent_prev = vec![i64::MIN; range];
    let mut parent_idx = vec![u32::MAX; range];
    bits[offset / WORD_BITS] |= 1u64 << (offset % WORD_BITS);
    parent_prev[offset] = offset as i64;

    for (idx, &value) in values.iter().enumerate() {
        if value == 0 {
            continue;
        }
        shifted.fill(0);
        let dist = value.unsigned_abs() as usize;
        let word_shift = dist / WORD_BITS;
        let bit_shift = dist % WORD_BITS;
        if value > 0 {
            for src in 0..words {
                let word = bits[src];
                if word == 0 {
                    continue;
                }
                let dst = src + word_shift;
                if dst < words {
                    shifted[dst] |= word << bit_shift;
                }
                if bit_shift != 0 && dst + 1 < words {
                    shifted[dst + 1] |= word >> (WORD_BITS - bit_shift);
                }
            }
        } else {
            for src in word_shift..words {
                let word = bits[src];
                if word == 0 {
                    continue;
                }
                let dst = src - word_shift;
                shifted[dst] |= word >> bit_shift;
                if bit_shift != 0 && dst > 0 {
                    shifted[dst - 1] |= word << (WORD_BITS - bit_shift);
                }
            }
        }
        clear_extra_bits(&mut shifted, range);
        for w in 0..words {
            let mut newly = shifted[w] & !bits[w];
            while newly != 0 {
                let bit = newly.trailing_zeros() as usize;
                let pos = w * WORD_BITS + bit;
                if pos < range {
                    let prev = pos as i128 - value as i128;
                    if prev >= 0 && prev < range as i128 {
                        parent_prev[pos] = prev as i64;
                        parent_idx[pos] = idx as u32;
                    }
                }
                newly &= newly - 1;
            }
            bits[w] |= shifted[w];
        }
        if ((bits[target_pos / WORD_BITS] >> (target_pos % WORD_BITS)) & 1) != 0 {
            let mut result = Vec::new();
            let mut cur = target_pos;
            while cur != offset {
                let idx_at = parent_idx[cur];
                if idx_at == u32::MAX {
                    return None;
                }
                result.push(idx_at);
                cur = parent_prev[cur] as usize;
            }
            return Some(result);
        }
    }
    None
}

fn sparse_dp(values: &[i64], target: i64) -> Option<Vec<u32>> {
    use std::collections::HashMap;

    let mut states: HashMap<i64, (i64, u32)> = HashMap::new();
    states.insert(0, (0, u32::MAX));
    for (idx, &value) in values.iter().enumerate() {
        let keys: Vec<i64> = states.keys().copied().collect();
        let mut additions = Vec::new();
        for sum in keys {
            let next = sum.saturating_add(value);
            if !states.contains_key(&next) {
                additions.push((next, (sum, idx as u32)));
                if next == target {
                    for (new_sum, parent) in additions {
                        states.insert(new_sum, parent);
                    }
                    let mut result = Vec::new();
                    let mut cur = target;
                    while cur != 0 {
                        let &(prev, used_idx) = states.get(&cur)?;
                        if used_idx == u32::MAX {
                            return None;
                        }
                        result.push(used_idx);
                        cur = prev;
                    }
                    return Some(result);
                }
                if states.len() + additions.len() > SPARSE_LIMIT {
                    return None;
                }
            }
        }
        for (new_sum, parent) in additions {
            states.insert(new_sum, parent);
        }
    }
    None
}

fn greedy_positive(values: &[i64], target: i64) -> Option<Vec<u32>> {
    if target < 0 {
        return None;
    }
    let mut pairs = Vec::new();
    for (idx, &value) in values.iter().enumerate() {
        if value > 0 && value <= target {
            pairs.push((value, idx as u32));
        }
    }
    pairs.sort_unstable_by(|a, b| b.0.cmp(&a.0));
    let mut remaining = target;
    let mut result = Vec::new();
    for (value, idx) in pairs {
        if value <= remaining {
            remaining -= value;
            result.push(idx);
            if remaining == 0 {
                return Some(result);
            }
        }
    }
    None
}

fn main() {
    let mut scanner = Scanner::new();
    let Some(num_numbers_raw) = scanner.next_i64() else {
        print_no();
        return;
    };
    let Some(target) = scanner.next_i64() else {
        print_no();
        return;
    };
    if num_numbers_raw < 0 {
        print_no();
        return;
    }
    let n = num_numbers_raw as usize;
    let mut values = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(value) = scanner.next_i64() {
            values.push(value);
        } else {
            break;
        }
    }

    if target == 0 {
        print_yes(Vec::new());
        return;
    }
    if let Some(answer) = check_single(&values, target) {
        print_yes(answer);
        return;
    }

    let mut gcd = 0i128;
    let mut positive_sum = 0i128;
    let mut negative_sum = 0i128;
    let mut all_nonnegative = true;
    for &value in &values {
        gcd = gcd_i128(gcd, value as i128);
        if value < 0 {
            all_nonnegative = false;
            negative_sum += value as i128;
        } else {
            positive_sum += value as i128;
        }
    }
    if gcd != 0 && (target as i128) % gcd != 0 {
        print_no();
        return;
    }
    if all_nonnegative && target < 0 {
        print_no();
        return;
    }
    if all_nonnegative && positive_sum < target as i128 {
        print_no();
        return;
    }
    if !all_nonnegative && ((target as i128) < negative_sum || (target as i128) > positive_sum) {
        print_no();
        return;
    }

    if all_nonnegative {
        if let Some(answer) = nonnegative_bitset_dp(&values, target as usize) {
            print_yes(answer);
            return;
        }
    } else if let Some(answer) = offset_bitset_dp(&values, target) {
        print_yes(answer);
        return;
    }

    if let Some(answer) = meet_in_middle(&values, target) {
        print_yes(answer);
        return;
    }
    if let Some(answer) = sparse_dp(&values, target) {
        print_yes(answer);
        return;
    }
    if all_nonnegative {
        if let Some(answer) = greedy_positive(&values, target) {
            print_yes(answer);
            return;
        }
    }

    print_no();
}
