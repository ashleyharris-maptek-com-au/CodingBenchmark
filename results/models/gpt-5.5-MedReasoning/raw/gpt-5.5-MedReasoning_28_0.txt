use std::collections::VecDeque;
use std::io::{self, Read};

const NIL: u32 = u32::MAX;

struct Scanner {
    data: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            value = value * 10 + usize::from(self.data[self.pos] - b'0');
            self.pos += 1;
        }
        value
    }
}

struct Graph {
    n: usize,
    edges: Vec<(u32, u32)>,
    offsets: Vec<usize>,
    adj: Vec<u32>,
    forced: Vec<bool>,
    degree: Vec<usize>,
}

fn build_graph() -> Graph {
    let mut scanner = Scanner::new();
    let n = scanner.next_usize();
    let m = scanner.next_usize();
    let mut edges = Vec::with_capacity(m);
    let mut forced = vec![false; n];
    let mut degree = vec![0usize; n];

    for _ in 0..m {
        let u = scanner.next_usize();
        let v = scanner.next_usize();
        if u == v {
            forced[u] = true;
        } else {
            edges.push((u as u32, v as u32));
            degree[u] += 1;
            degree[v] += 1;
        }
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degree[i];
    }

    let mut adj = vec![0u32; offsets[n]];
    let mut cursor = offsets[..n].to_vec();
    for &(u, v) in &edges {
        let ui = u as usize;
        let vi = v as usize;
        adj[cursor[ui]] = v;
        cursor[ui] += 1;
        adj[cursor[vi]] = u;
        cursor[vi] += 1;
    }

    Graph {
        n,
        edges,
        offsets,
        adj,
        forced,
        degree,
    }
}

fn bipartition(graph: &Graph) -> Option<Vec<i8>> {
    let mut color = vec![-1i8; graph.n];
    let mut queue = VecDeque::new();

    for start in 0..graph.n {
        if color[start] != -1 {
            continue;
        }
        color[start] = 0;
        if graph.forced[start] {
            continue;
        }
        queue.push_back(start);

        while let Some(u) = queue.pop_front() {
            let next_color = 1 - color[u];
            for &v_u32 in &graph.adj[graph.offsets[u]..graph.offsets[u + 1]] {
                let v = v_u32 as usize;
                if graph.forced[v] {
                    continue;
                }
                if color[v] == -1 {
                    color[v] = next_color;
                    queue.push_back(v);
                } else if color[v] == color[u] {
                    return None;
                }
            }
        }
    }

    Some(color)
}

fn hk_bfs(graph: &Graph, color: &[i8], pair: &[u32], dist: &mut [i32]) -> bool {
    let mut queue = VecDeque::new();
    let mut found = false;

    for u in 0..graph.n {
        if color[u] == 0 && !graph.forced[u] && pair[u] == NIL {
            dist[u] = 0;
            queue.push_back(u);
        } else {
            dist[u] = -1;
        }
    }

    while let Some(u) = queue.pop_front() {
        for &v_u32 in &graph.adj[graph.offsets[u]..graph.offsets[u + 1]] {
            let v = v_u32 as usize;
            if graph.forced[v] {
                continue;
            }
            let mate = pair[v];
            if mate == NIL {
                found = true;
            } else {
                let next = mate as usize;
                if dist[next] < 0 {
                    dist[next] = dist[u] + 1;
                    queue.push_back(next);
                }
            }
        }
    }

    found
}

fn hk_dfs(u: usize, graph: &Graph, pair: &mut [u32], dist: &mut [i32]) -> bool {
    for &v_u32 in &graph.adj[graph.offsets[u]..graph.offsets[u + 1]] {
        let v = v_u32 as usize;
        if graph.forced[v] {
            continue;
        }
        let mate = pair[v];
        if mate == NIL || (dist[mate as usize] == dist[u] + 1 && hk_dfs(mate as usize, graph, pair, dist)) {
            pair[u] = v_u32;
            pair[v] = u as u32;
            return true;
        }
    }
    dist[u] = -1;
    false
}

fn solve_bipartite(graph: &Graph, color: &[i8]) -> Vec<bool> {
    let mut pair = vec![NIL; graph.n];
    let mut dist = vec![-1i32; graph.n];

    while hk_bfs(graph, color, &pair, &mut dist) {
        for u in 0..graph.n {
            if color[u] == 0 && !graph.forced[u] && pair[u] == NIL {
                hk_dfs(u, graph, &mut pair, &mut dist);
            }
        }
    }

    let mut seen = vec![false; graph.n];
    let mut queue = VecDeque::new();
    for u in 0..graph.n {
        if color[u] == 0 && !graph.forced[u] && pair[u] == NIL {
            seen[u] = true;
            queue.push_back(u);
        }
    }

    while let Some(u) = queue.pop_front() {
        for &v_u32 in &graph.adj[graph.offsets[u]..graph.offsets[u + 1]] {
            let v = v_u32 as usize;
            if graph.forced[v] {
                continue;
            }
            if pair[u] == v_u32 {
                continue;
            }
            if !seen[v] {
                seen[v] = true;
                let mate = pair[v];
                if mate != NIL {
                    let next = mate as usize;
                    if !seen[next] {
                        seen[next] = true;
                        queue.push_back(next);
                    }
                }
            }
        }
    }

    let mut cover = vec![false; graph.n];
    for i in 0..graph.n {
        cover[i] = (color[i] == 0 && !seen[i]) || (color[i] == 1 && seen[i]) || graph.forced[i];
    }
    cover
}

fn prune_cover(graph: &Graph, cover: &mut [bool], order: &[usize]) {
    for &v in order {
        if !cover[v] || graph.forced[v] {
            continue;
        }
        let mut removable = true;
        for &to_u32 in &graph.adj[graph.offsets[v]..graph.offsets[v + 1]] {
            if !cover[to_u32 as usize] {
                removable = false;
                break;
            }
        }
        if removable {
            cover[v] = false;
        }
    }
}

fn matching_cover(graph: &Graph, order: &[usize]) -> Vec<bool> {
    let mut cover = graph.forced.clone();
    let mut matched = vec![false; graph.n];

    for &(u_u32, v_u32) in &graph.edges {
        let u = u_u32 as usize;
        let v = v_u32 as usize;
        if !matched[u] && !matched[v] {
            matched[u] = true;
            matched[v] = true;
            cover[u] = true;
            cover[v] = true;
        }
    }

    prune_cover(graph, &mut cover, order);
    cover
}

fn independent_cover(graph: &Graph, order: &[usize]) -> Vec<bool> {
    let mut independent = vec![false; graph.n];
    let mut blocked = vec![false; graph.n];

    for &v in order {
        if graph.forced[v] || blocked[v] {
            continue;
        }
        independent[v] = true;
        for &to_u32 in &graph.adj[graph.offsets[v]..graph.offsets[v + 1]] {
            blocked[to_u32 as usize] = true;
        }
    }

    let mut cover = vec![false; graph.n];
    for i in 0..graph.n {
        cover[i] = graph.forced[i] || !independent[i];
    }
    prune_cover(graph, &mut cover, order);
    cover
}

fn cover_size(cover: &[bool]) -> usize {
    cover.iter().filter(|&&chosen| chosen).count()
}

fn solve_general(graph: &Graph) -> Vec<bool> {
    let mut low_degree_order: Vec<usize> = (0..graph.n).collect();
    low_degree_order.sort_unstable_by_key(|&v| (graph.degree[v], v));

    let mut high_degree_order = low_degree_order.clone();
    high_degree_order.reverse();

    let cover_a = matching_cover(graph, &low_degree_order);
    let cover_b = independent_cover(graph, &low_degree_order);
    let cover_c = independent_cover(graph, &high_degree_order);

    let mut best = cover_a;
    if cover_size(&cover_b) < cover_size(&best) {
        best = cover_b;
    }
    if cover_size(&cover_c) < cover_size(&best) {
        best = cover_c;
    }
    best
}

fn print_cover(cover: &[bool]) {
    let chosen: Vec<usize> = cover
        .iter()
        .enumerate()
        .filter_map(|(i, &is_chosen)| if is_chosen { Some(i) } else { None })
        .collect();

    let mut output = String::new();
    output.push_str(&chosen.len().to_string());
    output.push('\n');
    for (i, vertex) in chosen.iter().enumerate() {
        if i > 0 {
            output.push(' ');
        }
        output.push_str(&vertex.to_string());
    }
    output.push('\n');
    print!("{output}");
}

fn main() {
    let graph = build_graph();
    let cover = if let Some(color) = bipartition(&graph) {
        solve_bipartite(&graph, &color)
    } else {
        solve_general(&graph)
    };
    print_cover(&cover);
}
