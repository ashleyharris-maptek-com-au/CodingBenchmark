using System;
using System.Diagnostics;
using System.IO;
using System.Text;

public class Program
{
    const int TimeLimitMilliseconds = 285000;

    static int N;
    static int[][] Flow = new int[0][];
    static int[][] Dist = new int[0][];
    static readonly Random Rng = new Random(1234567);

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = fs.NextInt();
        Flow = new int[N][];
        Dist = new int[N][];

        long[] flowWeight = new long[N];
        long[] distWeight = new long[N];

        for (int i = 0; i < N; i++)
        {
            Flow[i] = new int[N];
            for (int j = 0; j < N; j++)
            {
                int v = fs.NextInt();
                Flow[i][j] = v;
                flowWeight[i] += v;
            }
        }

        for (int i = 0; i < N; i++)
        {
            Dist[i] = new int[N];
            for (int j = 0; j < N; j++)
            {
                int v = fs.NextInt();
                Dist[i][j] = v;
                distWeight[i] += v;
            }
        }

        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowWeight[i] += Flow[j][i];
                distWeight[i] += Dist[j][i];
            }
        }

        int[] p;
        long bestCost;

        if (N <= 9)
        {
            p = SolveExactlySmall(out bestCost);
        }
        else
        {
            var sw = Stopwatch.StartNew();
            p = BuildInitial(flowWeight, distWeight);
            bestCost = TotalCost(p);

            TryAlternativeInitials(ref p, ref bestCost, flowWeight, distWeight);
            bestCost = ImproveBySwaps(p, bestCost, sw);
        }

        var sb = new StringBuilder();
        sb.AppendLine(bestCost.ToString());
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    static int[] BuildInitial(long[] flowWeight, long[] distWeight)
    {
        int[] facilities = Range(N);
        int[] locations = Range(N);

        Array.Sort(facilities, (a, b) =>
        {
            int c = flowWeight[b].CompareTo(flowWeight[a]);
            return c != 0 ? c : a.CompareTo(b);
        });

        Array.Sort(locations, (a, b) =>
        {
            int c = distWeight[a].CompareTo(distWeight[b]);
            return c != 0 ? c : a.CompareTo(b);
        });

        int[] p = new int[N];
        for (int k = 0; k < N; k++) p[facilities[k]] = locations[k];
        return p;
    }

    static void TryAlternativeInitials(ref int[] bestP, ref long bestCost, long[] flowWeight, long[] distWeight)
    {
        int[] candidate = Range(N);
        Consider(candidate, ref bestP, ref bestCost);

        Array.Reverse(candidate);
        Consider(candidate, ref bestP, ref bestCost);

        if (N <= 600)
        {
            for (int t = 0; t < 4; t++)
            {
                candidate = BuildInitial(flowWeight, distWeight);
                for (int i = N - 1; i > 0; i--)
                {
                    int j = Rng.Next(i + 1);
                    if (Rng.NextDouble() < 0.20)
                    {
                        int tmp = candidate[i];
                        candidate[i] = candidate[j];
                        candidate[j] = tmp;
                    }
                }
                Consider(candidate, ref bestP, ref bestCost);
            }
        }
    }

    static void Consider(int[] candidate, ref int[] bestP, ref long bestCost)
    {
        long c = TotalCost(candidate);
        if (c < bestCost)
        {
            bestCost = c;
            bestP = (int[])candidate.Clone();
        }
    }

    static long ImproveBySwaps(int[] p, long cost, Stopwatch sw)
    {
        if (N <= 1) return cost;

        if (N <= 180)
        {
            bool improved = true;
            while (improved && sw.ElapsedMilliseconds < TimeLimitMilliseconds)
            {
                improved = false;
                for (int a = 0; a < N - 1 && !improved; a++)
                {
                    for (int b = a + 1; b < N; b++)
                    {
                        long d = SwapDelta(p, a, b);
                        if (d < 0)
                        {
                            Swap(p, a, b);
                            cost += d;
                            improved = true;
                            break;
                        }
                    }
                }
            }
            return cost;
        }

        int[] heavy = Range(N);
        Array.Sort(heavy, (a, b) =>
        {
            long wa = 0, wb = 0;
            for (int k = 0; k < N; k++)
            {
                wa += Flow[a][k] + Flow[k][a];
                wb += Flow[b][k] + Flow[k][b];
            }
            int c = wb.CompareTo(wa);
            return c != 0 ? c : a.CompareTo(b);
        });

        int top = Math.Min(N, Math.Max(20, (int)Math.Sqrt(N) * 4));
        long checks = 0;
        long noImprove = 0;
        long staleLimit = N <= 500 ? 200000L : (N <= 1000 ? 150000L : 90000L);
        int seqA = 0;
        int seqB = 1;

        while (sw.ElapsedMilliseconds < TimeLimitMilliseconds && noImprove < staleLimit)
        {
            int a, b;
            int mode = Rng.Next(100);
            if (mode < 45)
            {
                a = Rng.Next(N);
                b = Rng.Next(N - 1);
                if (b >= a) b++;
            }
            else if (mode < 75)
            {
                a = heavy[Rng.Next(top)];
                b = Rng.Next(N - 1);
                if (b >= a) b++;
            }
            else
            {
                a = seqA;
                b = seqB;
                seqB++;
                if (seqB >= N)
                {
                    seqA++;
                    if (seqA >= N - 1) seqA = 0;
                    seqB = seqA + 1;
                }
            }

            long d = SwapDelta(p, a, b);
            if (d < 0)
            {
                Swap(p, a, b);
                cost += d;
                noImprove = 0;
            }
            else
            {
                noImprove++;
            }

            checks++;
            if ((checks & 1023) == 0 && sw.ElapsedMilliseconds >= TimeLimitMilliseconds) break;
        }

        return cost;
    }

    static int[] SolveExactlySmall(out long bestCost)
    {
        int[] p = Range(N);
        int[] best = (int[])p.Clone();
        bestCost = TotalCost(p);

        while (NextPermutation(p))
        {
            long c = TotalCost(p);
            if (c < bestCost)
            {
                bestCost = c;
                Array.Copy(p, best, N);
            }
        }

        return best;
    }

    static long TotalCost(int[] p)
    {
        long total = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            int[] fi = Flow[i];
            int[] dpi = Dist[pi];
            for (int j = 0; j < N; j++)
            {
                total += (long)fi[j] * dpi[p[j]];
            }
        }
        return total;
    }

    static long SwapDelta(int[] p, int a, int b)
    {
        if (a == b) return 0;

        int la = p[a], lb = p[b];
        long delta = 0;

        for (int k = 0; k < N; k++)
        {
            if (k == a || k == b) continue;
            int lk = p[k];

            delta += (long)Flow[a][k] * (Dist[lb][lk] - Dist[la][lk]);
            delta += (long)Flow[b][k] * (Dist[la][lk] - Dist[lb][lk]);
            delta += (long)Flow[k][a] * (Dist[lk][lb] - Dist[lk][la]);
            delta += (long)Flow[k][b] * (Dist[lk][la] - Dist[lk][lb]);
        }

        delta += (long)Flow[a][a] * (Dist[lb][lb] - Dist[la][la]);
        delta += (long)Flow[b][b] * (Dist[la][la] - Dist[lb][lb]);
        delta += (long)Flow[a][b] * (Dist[lb][la] - Dist[la][lb]);
        delta += (long)Flow[b][a] * (Dist[la][lb] - Dist[lb][la]);

        return delta;
    }

    static void Swap(int[] p, int a, int b)
    {
        int t = p[a];
        p[a] = p[b];
        p[b] = t;
    }

    static int[] Range(int n)
    {
        int[] a = new int[n];
        for (int i = 0; i < n; i++) a[i] = i;
        return a;
    }

    static bool NextPermutation(int[] a)
    {
        int i = a.Length - 2;
        while (i >= 0 && a[i] >= a[i + 1]) i--;
        if (i < 0) return false;

        int j = a.Length - 1;
        while (a[j] <= a[i]) j--;
        SwapArray(a, i, j);
        Array.Reverse(a, i + 1, a.Length - i - 1);
        return true;
    }

    static void SwapArray(int[] a, int i, int j)
    {
        int t = a[i];
        a[i] = a[j];
        a[j] = t;
    }

    sealed class FastScanner
    {
        readonly Stream _stream;
        readonly byte[] _buffer = new byte[1 << 16];
        int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }
}
