; SPIR-V
; Version: 1.0
OpCapability Shader
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_15 = OpConstant %float 0.15
%float_0_3 = OpConstant %float 0.3
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_1 = OpConstant %float 1.0
%skyColor = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
%main = OpFunction %void None %fn_void
%entry = OpLabel
%normalRaw = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %GLSLstd450 Normalize %normalRaw
%ny = OpCompositeExtract %float %N 1
%nyHalf = OpFMul %float %ny %float_0_5
%hemiT = OpFAdd %float %nyHalf %float_0_5
%groundT = OpFSub %float %float_1 %hemiT
%skyPart = OpVectorTimesScalar %v3float %skyColor %hemiT
%groundPart = OpVectorTimesScalar %v3float %groundColor %groundT
%ambient = OpFAdd %v3float %groundPart %skyPart
%worldPos = OpLoad %v3float %inWorldPos
%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%lightVec = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %GLSLstd450 Normalize %lightVec
%ndotl = OpDot %float %N %L
%d = OpExtInst %float %GLSLstd450 FMax %ndotl %float_0
%diffuse = OpFMul %float %float_0_3 %d
%diffuse3 = OpCompositeConstruct %v3float %diffuse %diffuse %diffuse
%rgb = OpFAdd %v3float %ambient %diffuse3
%out = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %out
OpReturn
OpFunctionEnd
