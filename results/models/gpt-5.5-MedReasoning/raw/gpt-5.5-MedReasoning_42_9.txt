OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_025 = OpConstant %float 0.025
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input
%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%fn_void = OpTypeFunction %void
%main = OpFunction %void None %fn_void
%entry = OpLabel
%prev_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%start_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%end_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%next_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%prev = OpLoad %v4float %prev_ptr
%start = OpLoad %v4float %start_ptr
%end = OpLoad %v4float %end_ptr
%next = OpLoad %v4float %next_ptr
%prev_xy = OpVectorShuffle %v2float %prev %prev 0 1
%start_xy = OpVectorShuffle %v2float %start %start 0 1
%end_xy = OpVectorShuffle %v2float %end %end 0 1
%next_xy = OpVectorShuffle %v2float %next %next 0 1
%d0 = OpFSub %v2float %end_xy %prev_xy
%d1 = OpFSub %v2float %next_xy %start_xy
%t0 = OpExtInst %v2float %glsl Normalize %d0
%t1 = OpExtInst %v2float %glsl Normalize %d1
%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t0_y = OpFNegate %float %t0_y
%neg_t1_y = OpFNegate %float %t1_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x
%off0 = OpVectorTimesScalar %v2float %perp0 %float_0_025
%off1 = OpVectorTimesScalar %v2float %perp1 %float_0_025
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr
%start_z = OpCompositeExtract %float %start 2
%start_w = OpCompositeExtract %float %start 3
%end_z = OpCompositeExtract %float %end 2
%end_w = OpCompositeExtract %float %end 3
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%v0_xy = OpFAdd %v2float %start_xy %off0
%v0_x = OpCompositeExtract %float %v0_xy 0
%v0_y = OpCompositeExtract %float %v0_xy 1
%v0 = OpCompositeConstruct %v4float %v0_x %v0_y %start_z %start_w
OpStore %pos_out_ptr %v0
OpStore %outColor %c1
OpEmitVertex
%v1_xy = OpFSub %v2float %start_xy %off0
%v1_x = OpCompositeExtract %float %v1_xy 0
%v1_y = OpCompositeExtract %float %v1_xy 1
%v1 = OpCompositeConstruct %v4float %v1_x %v1_y %start_z %start_w
OpStore %pos_out_ptr %v1
OpStore %outColor %c1
OpEmitVertex
%v2_xy = OpFAdd %v2float %end_xy %off1
%v2_x = OpCompositeExtract %float %v2_xy 0
%v2_y = OpCompositeExtract %float %v2_xy 1
%v2 = OpCompositeConstruct %v4float %v2_x %v2_y %end_z %end_w
OpStore %pos_out_ptr %v2
OpStore %outColor %c2
OpEmitVertex
%v3_xy = OpFSub %v2float %end_xy %off1
%v3_x = OpCompositeExtract %float %v3_xy 0
%v3_y = OpCompositeExtract %float %v3_xy 1
%v3 = OpCompositeConstruct %v4float %v3_x %v3_y %end_z %end_w
OpStore %pos_out_ptr %v3
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
