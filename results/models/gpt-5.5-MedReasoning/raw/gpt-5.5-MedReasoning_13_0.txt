#include <algorithm>
#include <cstddef>
#include <iostream>
#include <numeric>
#include <string>
#include <unordered_set>
#include <vector>

using namespace std;

static string bruteForceSmall(const vector<string>& s) {
    int n = static_cast<int>(s.size());
    int base = 0;
    for (int i = 1; i < n; ++i) {
        if (s[i].size() < s[base].size()) base = i;
    }

    const string& b = s[base];
    for (int len = static_cast<int>(b.size()); len > 0; --len) {
        unordered_set<string> tried;
        for (int start = 0; start + len <= static_cast<int>(b.size()); ++start) {
            string cand = b.substr(static_cast<size_t>(start), static_cast<size_t>(len));
            if (!tried.insert(cand).second) continue;

            bool ok = true;
            for (int i = 0; i < n; ++i) {
                if (i != base && s[i].find(cand) == string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return cand;
        }
    }
    return "";
}

class SuffixAutomaton {
public:
    explicit SuffixAutomaton(const string& source) : text(source) {
        const size_t reserveStates = max<size_t>(2, source.size() * 2 + 1);
        link.reserve(reserveStates);
        maxLen.reserve(reserveStates);
        head.reserve(reserveStates);
        firstPos.reserve(reserveStates);

        to.reserve(source.size() * 3 + 8);
        next.reserve(source.size() * 3 + 8);
        label.reserve(source.size() * 3 + 8);

        newState(0);
        link[0] = -1;
        last = 0;

        for (unsigned char c : source) extend(c);
        buildLengthOrder();
        common.assign(size(), 0);
        for (int i = 0; i < size(); ++i) common[i] = maxLen[i];
    }

    string longestCommonSubstring(const vector<string>& strings, int baseIndex) {
        vector<int> match(size(), 0);

        for (int i = 0; i < static_cast<int>(strings.size()); ++i) {
            if (i == baseIndex) continue;

            fill(match.begin(), match.end(), 0);
            int state = 0;
            int length = 0;

            for (unsigned char c : strings[i]) {
                int e = findEdge(state, c);
                if (e != -1) {
                    state = to[e];
                    ++length;
                } else {
                    while (state != -1 && (e = findEdge(state, c)) == -1) {
                        state = link[state];
                    }
                    if (state == -1) {
                        state = 0;
                        length = 0;
                    } else {
                        length = maxLen[state] + 1;
                        state = to[e];
                    }
                }
                if (length > match[state]) match[state] = length;
            }

            for (int k = static_cast<int>(order.size()) - 1; k > 0; --k) {
                int v = order[k];
                int p = link[v];
                if (p >= 0) {
                    int inherited = min(match[v], maxLen[p]);
                    if (inherited > match[p]) match[p] = inherited;
                }
            }

            int bestAfterThisString = 0;
            for (int v = 0; v < size(); ++v) {
                if (match[v] < common[v]) common[v] = match[v];
                if (common[v] > bestAfterThisString) bestAfterThisString = common[v];
            }
            if (bestAfterThisString == 0) return "";
        }

        int bestState = 0;
        int bestLength = 0;
        for (int v = 1; v < size(); ++v) {
            if (common[v] > bestLength) {
                bestLength = common[v];
                bestState = v;
            }
        }

        if (bestLength == 0) return "";
        int start = firstPos[bestState] - bestLength + 1;
        return text.substr(static_cast<size_t>(start), static_cast<size_t>(bestLength));
    }

private:
    const string& text;
    vector<int> link;
    vector<int> maxLen;
    vector<int> head;
    vector<int> firstPos;
    vector<int> common;
    vector<int> order;

    vector<int> to;
    vector<int> next;
    vector<unsigned char> label;

    int last = 0;

    int size() const {
        return static_cast<int>(link.size());
    }

    int newState(int len) {
        int id = size();
        link.push_back(0);
        maxLen.push_back(len);
        head.push_back(-1);
        firstPos.push_back(-1);
        return id;
    }

    int findEdge(int state, unsigned char c) const {
        for (int e = head[state]; e != -1; e = next[e]) {
            if (label[e] == c) return e;
        }
        return -1;
    }

    void addEdge(int from, unsigned char c, int dest) {
        label.push_back(c);
        to.push_back(dest);
        next.push_back(head[from]);
        head[from] = static_cast<int>(to.size()) - 1;
    }

    void copyEdges(int from, int into) {
        for (int e = head[from]; e != -1; e = next[e]) {
            addEdge(into, label[e], to[e]);
        }
    }

    void extend(unsigned char c) {
        int cur = newState(maxLen[last] + 1);
        firstPos[cur] = maxLen[cur] - 1;

        int p = last;
        while (p != -1 && findEdge(p, c) == -1) {
            addEdge(p, c, cur);
            p = link[p];
        }

        if (p == -1) {
            link[cur] = 0;
        } else {
            int edge = findEdge(p, c);
            int q = to[edge];

            if (maxLen[p] + 1 == maxLen[q]) {
                link[cur] = q;
            } else {
                int clone = newState(maxLen[p] + 1);
                link[clone] = link[q];
                firstPos[clone] = firstPos[q];
                copyEdges(q, clone);

                while (p != -1) {
                    int pe = findEdge(p, c);
                    if (pe == -1 || to[pe] != q) break;
                    to[pe] = clone;
                    p = link[p];
                }

                link[q] = clone;
                link[cur] = clone;
            }
        }

        last = cur;
    }

    void buildLengthOrder() {
        int maxTextLen = static_cast<int>(text.size());
        vector<int> count(static_cast<size_t>(maxTextLen) + 1, 0);
        for (int len : maxLen) ++count[len];
        for (int i = 1; i <= maxTextLen; ++i) count[i] += count[i - 1];

        order.assign(size(), 0);
        for (int i = size() - 1; i >= 0; --i) {
            order[--count[maxLen[i]]] = i;
        }
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;
    string discard;
    getline(cin, discard);

    vector<string> s(static_cast<size_t>(n));
    for (int i = 0; i < n; ++i) getline(cin, s[i]);

    if (n <= 0) {
        cout << '\n';
        return 0;
    }
    if (n == 1) {
        cout << s[0] << '\n';
        return 0;
    }

    int base = 0;
    size_t total = 0;
    for (int i = 0; i < n; ++i) {
        total += s[i].size();
        if (s[i].empty()) {
            cout << '\n';
            return 0;
        }
        if (s[i].size() < s[base].size()) base = i;
    }

    // Adaptive path: tiny cases are faster with direct substring checks; all
    // realistic large cases use the exact linear suffix-automaton algorithm.
    if (n <= 3 && s[base].size() <= 128 && total <= 10000) {
        cout << bruteForceSmall(s) << '\n';
        return 0;
    }

    SuffixAutomaton sam(s[base]);
    cout << sam.longestCommonSubstring(s, base) << '\n';
    return 0;
}
