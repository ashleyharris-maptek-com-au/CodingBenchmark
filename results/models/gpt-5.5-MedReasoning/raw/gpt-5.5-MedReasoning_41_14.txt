; SPIR-V
; Version: 1.0

OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft

OpSource GLSL 450

OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fnVoid = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformV4 = OpTypePointer Uniform %v4float

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3.0
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4

%baseColor = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%fogColor = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_8

%inWorldPos = OpVariable %ptrInputV3 Input
%inNormal = OpVariable %ptrInputV3 Input
%outColor = OpVariable %ptrOutputV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform

%main = OpFunction %void None %fnVoid
%entry = OpLabel

%worldPos = OpLoad %v3float %inWorldPos
%normalIn = OpLoad %v3float %inNormal
%normal = OpExtInst %v3float %glsl Normalize %normalIn

%lightPtr = OpAccessChain %ptrUniformV4 %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%lightVec = OpFSub %v3float %lightPos %worldPos
%lightDir = OpExtInst %v3float %glsl Normalize %lightVec
%diffuseRaw = OpDot %float %normal %lightDir
%diffuse = OpExtInst %float %glsl FMax %diffuseRaw %float_0
%diffuseScaled = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %diffuseScaled
%litColor = OpVectorTimesScalar %v3float %baseColor %lambert

%cameraPtr = OpAccessChain %ptrUniformV4 %ubo %int_4
%cameraPos4 = OpLoad %v4float %cameraPtr
%cameraPos = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2
%viewVec = OpFSub %v3float %cameraPos %worldPos
%dist = OpExtInst %float %glsl Length %viewVec
%fogNumer = OpFSub %float %dist %float_1_5
%fogUnclamped = OpFDiv %float %fogNumer %float_3
%fog = OpExtInst %float %glsl FClamp %fogUnclamped %float_0 %float_1
%oneMinusFog = OpFSub %float %float_1 %fog
%litPart = OpVectorTimesScalar %v3float %litColor %oneMinusFog
%fogPart = OpVectorTimesScalar %v3float %fogColor %fog
%rgb = OpFAdd %v3float %litPart %fogPart
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%color = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %color

OpReturn
OpFunctionEnd
