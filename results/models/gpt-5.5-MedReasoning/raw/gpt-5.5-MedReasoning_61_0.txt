declare const require: any;
declare const process: any;

const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type Package = { id: any; x: number; y: number };
type RouteRec = { seq: number[]; active: boolean };
type Saving = { i: number; j: number; s: number };

const packages: Package[] = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;
const depotX = Number(data.depot?.[0] ?? 0);
const depotY = Number(data.depot?.[1] ?? 0);
const maxRoutes = Math.max(1, Math.min(n || 1, Math.floor(Number(data.truckCount) || 1)));

const xs = new Array<number>(n);
const ys = new Array<number>(n);
const depotDist = new Array<number>(n);
for (let i = 0; i < n; i++) {
  xs[i] = Number(packages[i].x);
  ys[i] = Number(packages[i].y);
  depotDist[i] = Math.hypot(xs[i] - depotX, ys[i] - depotY);
}

function dist(a: number, b: number): number {
  return Math.hypot(xs[a] - xs[b], ys[a] - ys[b]);
}

function edgeCost(a: number, b: number, route: number[]): number {
  if (a < 0) return depotDist[route[b]];
  if (b >= route.length) return depotDist[route[a]];
  return dist(route[a], route[b]);
}

function routeLength(route: number[]): number {
  if (route.length === 0) return 0;
  let total = depotDist[route[0]] + depotDist[route[route.length - 1]];
  for (let i = 1; i < route.length; i++) total += dist(route[i - 1], route[i]);
  return total;
}

function reverseSegment(route: number[], lo: number, hi: number): void {
  while (lo < hi) {
    const t = route[lo];
    route[lo++] = route[hi];
    route[hi--] = t;
  }
}

function twoOpt(route: number[]): void {
  const m = route.length;
  if (m < 4) return;
  const fullSearch = m <= 550;
  const window = fullSearch ? m : 80;
  const maxPasses = m <= 550 ? 18 : m <= 1800 ? 6 : 3;

  for (let pass = 0; pass < maxPasses; pass++) {
    let improved = false;
    for (let i = 0; i < m - 1; i++) {
      const maxK = Math.min(m - 1, i + window);
      for (let k = i + 1; k <= maxK; k++) {
        if (i === 0 && k === m - 1) continue;
        const before = edgeCost(i - 1, i, route) + edgeCost(k, k + 1, route);
        const after = edgeCost(i - 1, k, route) + edgeCost(i, k + 1, route);
        if (after + 1e-9 < before) {
          reverseSegment(route, i, k);
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
}

function orientMerge(a: number[], i: number, b: number[], j: number): number[] | null {
  const aStart = a[0] === i;
  const aEnd = a[a.length - 1] === i;
  const bStart = b[0] === j;
  const bEnd = b[b.length - 1] === j;
  if (!aStart && !aEnd) return null;
  if (!bStart && !bEnd) return null;

  if (aEnd && bStart) return a.concat(b);
  if (aStart && bEnd) return b.concat(a);
  if (aStart && bStart) return a.slice().reverse().concat(b);
  return a.concat(b.slice().reverse());
}

function savingsRoutes(): number[][] {
  const routes: RouteRec[] = [];
  const routeOf = new Array<number>(n);
  for (let i = 0; i < n; i++) {
    routes.push({ seq: [i], active: true });
    routeOf[i] = i;
  }

  const savings: Saving[] = [];
  for (let i = 0; i < n; i++) {
    for (let j = i + 1; j < n; j++) {
      savings.push({ i, j, s: depotDist[i] + depotDist[j] - dist(i, j) });
    }
  }
  savings.sort((a, b) => b.s - a.s || a.i - b.i || a.j - b.j);

  let activeCount = n;
  for (const sv of savings) {
    if (activeCount <= maxRoutes && sv.s <= 1e-9) break;
    const ri = routeOf[sv.i];
    const rj = routeOf[sv.j];
    if (ri === rj || !routes[ri].active || !routes[rj].active) continue;

    const merged = orientMerge(routes[ri].seq, sv.i, routes[rj].seq, sv.j);
    if (merged === null) continue;

    routes[ri].seq = merged;
    routes[rj].active = false;
    for (const p of routes[rj].seq) routeOf[p] = ri;
    activeCount--;
  }

  return routes.filter(r => r.active).map(r => r.seq);
}

function sweepRoutes(): number[][] {
  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => {
    const aa = Math.atan2(ys[a] - depotY, xs[a] - depotX);
    const bb = Math.atan2(ys[b] - depotY, xs[b] - depotX);
    return aa - bb || depotDist[b] - depotDist[a] || a - b;
  });

  if (n > 2) {
    let bestGap = -1;
    let split = 0;
    for (let i = 0; i < n; i++) {
      const a = Math.atan2(ys[order[i]] - depotY, xs[order[i]] - depotX);
      const b0 = Math.atan2(ys[order[(i + 1) % n]] - depotY, xs[order[(i + 1) % n]] - depotX);
      const b = i + 1 === n ? b0 + Math.PI * 2 : b0;
      const gap = b - a;
      if (gap > bestGap) {
        bestGap = gap;
        split = i + 1;
      }
    }
    if (split > 0 && split < n) {
      const rotated = order.slice(split).concat(order.slice(0, split));
      for (let i = 0; i < n; i++) order[i] = rotated[i];
    }
  }

  const k = maxRoutes;
  const routes: number[][] = [];
  let start = 0;
  for (let r = 0; r < k; r++) {
    const remainingItems = n - start;
    const remainingRoutes = k - r;
    const size = Math.ceil(remainingItems / remainingRoutes);
    routes.push(order.slice(start, start + size));
    start += size;
  }
  return routes.filter(r => r.length > 0);
}

function nearestNeighbor(route: number[]): number[] {
  const m = route.length;
  if (m <= 2 || m > 1600) return route;
  const unused = route.slice();
  let bestPos = 0;
  for (let i = 1; i < m; i++) {
    if (depotDist[unused[i]] > depotDist[unused[bestPos]]) bestPos = i;
  }
  const result: number[] = [];
  let current = unused[bestPos];
  unused.splice(bestPos, 1);
  result.push(current);

  while (unused.length) {
    let pos = 0;
    let best = dist(current, unused[0]);
    for (let i = 1; i < unused.length; i++) {
      const d = dist(current, unused[i]);
      if (d < best - 1e-12 || (Math.abs(d - best) <= 1e-12 && unused[i] < unused[pos])) {
        best = d;
        pos = i;
      }
    }
    current = unused[pos];
    unused.splice(pos, 1);
    result.push(current);
  }
  return result;
}

function improveRoute(route: number[]): number[] {
  if (route.length <= 1) return route;
  const nn = nearestNeighbor(route);
  const keep = route.slice();
  twoOpt(nn);
  twoOpt(keep);
  return routeLength(nn) < routeLength(keep) ? nn : keep;
}

function rebalanceSingletons(routes: number[][]): number[][] {
  if (routes.length <= 1 || routes.length >= maxRoutes) return routes;
  return routes;
}

let routes: number[][];
if (n === 0) {
  routes = [];
} else if (n <= 2200) {
  routes = savingsRoutes();
} else {
  routes = sweepRoutes();
}

routes = rebalanceSingletons(routes).map(improveRoute);
routes.sort((a, b) => {
  const da = a.length ? depotDist[a[0]] : 0;
  const db = b.length ? depotDist[b[0]] : 0;
  return da - db || a.length - b.length;
});

const output = {
  routes: routes.map(route => route.map(i => packages[i].id))
};

process.stdout.write(JSON.stringify(output));
