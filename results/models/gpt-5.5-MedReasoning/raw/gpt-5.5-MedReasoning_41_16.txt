; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %in_uv %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %in_uv Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Function_int = OpTypePointer Function %int
%ptr_Function_float = OpTypePointer Function %float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%in_uv = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None %fn_void
%entry = OpLabel
%iter = OpVariable %ptr_Function_int Function
%zx = OpVariable %ptr_Function_float Function
%zy = OpVariable %ptr_Function_float Function
OpStore %iter %int_0
OpStore %zx %float_0
OpStore %zy %float_0
OpBranch %loop
%loop = OpLabel
OpLoopMerge %merge %continue None
%i = OpLoad %int %iter
%zxv = OpLoad %float %zx
%zyv = OpLoad %float %zy
%lt20 = OpSLessThan %bool %i %int_20
%zx2 = OpFMul %float %zxv %zxv
%zy2 = OpFMul %float %zyv %zyv
%mag2 = OpFAdd %float %zx2 %zy2
%inside = OpFOrdLessThanEqual %bool %mag2 %float_4
%keepGoing = OpLogicalAnd %bool %lt20 %inside
OpBranchConditional %keepGoing %body %merge
%body = OpLabel
%uv = OpLoad %v2float %in_uv
%u = OpCompositeExtract %float %uv 0
%v = OpCompositeExtract %float %uv 1
%u3 = OpFMul %float %u %float_3
%cx = OpFSub %float %u3 %float_2
%v2 = OpFMul %float %v %float_2
%cy = OpFSub %float %v2 %float_1
%zx2b = OpFMul %float %zxv %zxv
%zy2b = OpFMul %float %zyv %zyv
%realPart = OpFSub %float %zx2b %zy2b
%newZx = OpFAdd %float %realPart %cx
%twiceZx = OpFMul %float %float_2 %zxv
%imagPart = OpFMul %float %twiceZx %zyv
%newZy = OpFAdd %float %imagPart %cy
OpStore %zx %newZx
OpStore %zy %newZy
OpBranch %continue
%continue = OpLabel
%i_continue = OpLoad %int %iter
%i_next = OpIAdd %int %i_continue %int_1
OpStore %iter %i_next
OpBranch %loop
%merge = OpLabel
%finalIter = OpLoad %int %iter
%iterFloat = OpConvertSToF %float %finalIter
%t = OpFMul %float %iterFloat %float_0_05
%g = OpFMul %float %t %float_0_5
%b = OpFSub %float %float_1 %t
%color = OpCompositeConstruct %v4float %t %g %b %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
