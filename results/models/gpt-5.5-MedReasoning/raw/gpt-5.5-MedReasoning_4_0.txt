#include <algorithm>
#include <array>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <string>
#include <tuple>
#include <unordered_set>
#include <vector>

using namespace std;

struct Vec3 {
    double x = 0.0, y = 0.0, z = 0.0;
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static Vec3 operator/(const Vec3& a, double s) { return {a.x / s, a.y / s, a.z / s}; }

static double dotp(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static Vec3 crossp(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

static double norm(const Vec3& a) {
    return sqrt(dotp(a, a));
}

struct Plane {
    Vec3 n;
    double d = 0.0; // inside is n.x*x+n.y*y+n.z*z+d <= 0
};

struct Placement {
    Vec3 c;
};

static long long qround(double v, double scale) {
    return llround(v * scale);
}

static string plane_key(const Plane& p) {
    const double scale = 100000000.0;
    return to_string(qround(p.n.x, scale)) + "," + to_string(qround(p.n.y, scale)) + "," +
           to_string(qround(p.n.z, scale)) + "," + to_string(qround(p.d, scale));
}

static bool inside_poly(const Vec3& p, const vector<Plane>& planes, double eps) {
    for (const Plane& pl : planes) {
        if (dotp(pl.n, p) + pl.d > eps) return false;
    }
    return true;
}

static bool tetra_inside(const Vec3& c, const array<Vec3, 4>& tv, const vector<Plane>& planes, double eps) {
    for (const Vec3& v : tv) {
        if (!inside_poly(c + v, planes, eps)) return false;
    }
    return true;
}

static vector<Plane> build_convex_hull_planes(const vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    vector<Plane> planes;
    if (n < 4) return planes;

    Vec3 centroid;
    for (const Vec3& p : pts) centroid = centroid + p;
    centroid = centroid / static_cast<double>(n);

    Vec3 mn = pts[0], mx = pts[0];
    for (const Vec3& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }
    const double diag = max(1.0, norm(mx - mn));
    const double eps = 1e-9 * diag;

    unordered_set<string> seen;
    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 normal = crossp(pts[j] - pts[i], pts[k] - pts[i]);
                double len = norm(normal);
                if (len <= eps) continue;
                normal = normal / len;
                double d = -dotp(normal, pts[i]);

                int pos = 0, neg = 0;
                for (const Vec3& p : pts) {
                    double s = dotp(normal, p) + d;
                    if (s > eps) ++pos;
                    else if (s < -eps) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                if (pos && !neg) {
                    normal = normal * -1.0;
                    d = -d;
                }

                // Make sure the centroid is on the inside side. This also handles
                // numerical ambiguity when all points are nearly coplanar.
                if (dotp(normal, centroid) + d > 0.0) {
                    normal = normal * -1.0;
                    d = -d;
                }

                Plane pl{normal, d};
                string key = plane_key(pl);
                if (seen.insert(key).second) planes.push_back(pl);
            }
        }
    }
    return planes;
}

static vector<Vec3> reduced_hull_points(const vector<Vec3>& pts) {
    const int n = static_cast<int>(pts.size());
    const int LIMIT = 180;
    if (n <= LIMIT) return pts;

    vector<Vec3> dirs;
    for (int dx = -2; dx <= 2; ++dx) {
        for (int dy = -2; dy <= 2; ++dy) {
            for (int dz = -2; dz <= 2; ++dz) {
                if (dx == 0 && dy == 0 && dz == 0) continue;
                Vec3 d{static_cast<double>(dx), static_cast<double>(dy), static_cast<double>(dz)};
                double len = norm(d);
                dirs.push_back(d / len);
            }
        }
    }

    vector<int> chosen;
    vector<char> used(n, 0);
    auto add_index = [&](int idx) {
        if (idx >= 0 && !used[idx]) {
            used[idx] = 1;
            chosen.push_back(idx);
        }
    };

    for (const Vec3& d : dirs) {
        int imin = 0, imax = 0;
        double vmin = dotp(d, pts[0]), vmax = vmin;
        for (int i = 1; i < n; ++i) {
            double v = dotp(d, pts[i]);
            if (v < vmin) { vmin = v; imin = i; }
            if (v > vmax) { vmax = v; imax = i; }
        }
        add_index(imin);
        add_index(imax);
    }

    vector<Vec3> reduced;
    reduced.reserve(chosen.size());
    for (int idx : chosen) reduced.push_back(pts[idx]);
    if (static_cast<int>(reduced.size()) > LIMIT) reduced.resize(LIMIT);
    return reduced;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge) || V <= 0 || edge <= 0.0) {
        cout << 0 << '\n';
        return 0;
    }

    vector<Vec3> pts(V);
    for (int i = 0; i < V; ++i) cin >> pts[i].x >> pts[i].y >> pts[i].z;

    Vec3 mn = pts[0], mx = pts[0], centroid;
    for (const Vec3& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
        centroid = centroid + p;
    }
    centroid = centroid / static_cast<double>(V);

    vector<Vec3> hullPts = reduced_hull_points(pts);
    vector<Plane> planes = build_convex_hull_planes(hullPts);
    if (planes.empty()) {
        cout << 0 << '\n';
        return 0;
    }

    const double scale = edge / (2.0 * sqrt(2.0));
    const array<Vec3, 4> tetra = {
        Vec3{ scale,  scale,  scale},
        Vec3{ scale, -scale, -scale},
        Vec3{-scale,  scale, -scale},
        Vec3{-scale, -scale,  scale}
    };

    const double cell = 2.0 * scale * (1.0 + 1e-8);
    const double eps = max(1e-8 * edge, 1e-10 * max(1.0, norm(mx - mn)));
    const int MAX_OUTPUT = 250000;
    const int64_t MAX_VISITS = 3000000;

    auto span_count = [&](double lo, double hi) -> int64_t {
        if (hi < lo) return 0;
        long double v = floor((hi - lo) / cell) + 1.0L;
        if (v > static_cast<long double>(numeric_limits<int64_t>::max() / 4)) {
            return numeric_limits<int64_t>::max() / 4;
        }
        return max<int64_t>(0, static_cast<int64_t>(v));
    };

    const double loX = mn.x + scale, hiX = mx.x - scale;
    const double loY = mn.y + scale, hiY = mx.y - scale;
    const double loZ = mn.z + scale, hiZ = mx.z - scale;
    int64_t nx = span_count(loX, hiX), ny = span_count(loY, hiY), nz = span_count(loZ, hiZ);
    if (nx == 0 || ny == 0 || nz == 0) {
        cout << 0 << '\n';
        return 0;
    }

    long double total = static_cast<long double>(nx) * static_cast<long double>(ny) * static_cast<long double>(nz);
    int stride = 1;
    if (total > MAX_VISITS) {
        stride = max(1, static_cast<int>(ceil(cbrt(total / static_cast<long double>(MAX_VISITS)))));
    }

    vector<Placement> out;
    out.reserve( min(MAX_OUTPUT, 100000) );
    auto started = chrono::steady_clock::now();

    for (int64_t iz = 0; iz < nz && static_cast<int>(out.size()) < MAX_OUTPUT; iz += stride) {
        double z = loZ + static_cast<double>(iz) * cell;
        for (int64_t iy = 0; iy < ny && static_cast<int>(out.size()) < MAX_OUTPUT; iy += stride) {
            double y = loY + static_cast<double>(iy) * cell;
            for (int64_t ix = 0; ix < nx && static_cast<int>(out.size()) < MAX_OUTPUT; ix += stride) {
                double x = loX + static_cast<double>(ix) * cell;
                Vec3 c{x, y, z};
                if (tetra_inside(c, tetra, planes, eps)) out.push_back({c});
            }
        }
        if ((iz / max(1, stride)) % 64 == 0) {
            double elapsed = chrono::duration<double>(chrono::steady_clock::now() - started).count();
            if (elapsed > 26.0) break;
        }
    }

    // If the regular sweep missed a valid position in a small or skewed body,
    // try the vertex centroid before giving up.
    if (out.empty() && tetra_inside(centroid, tetra, planes, eps)) {
        out.push_back({centroid});
    }

    cout << out.size() << '\n';
    cout << fixed << setprecision(10);
    for (const Placement& p : out) {
        cout << p.c.x << ' ' << p.c.y << ' ' << p.c.z << " 0 0 0\n";
    }
    return 0;
}
