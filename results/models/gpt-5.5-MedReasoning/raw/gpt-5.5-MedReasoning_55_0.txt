#include <algorithm>
#include <array>
#include <atomic>
#include <chrono>
#include <cctype>
#include <cstdlib>
#include <cstdint>
#include <future>
#include <iostream>
#include <limits>
#include <sstream>
#include <string>
#include <utility>
#include <vector>

using namespace std;

struct Move {
    int from = -1, to = -1;
    char promo = 0;
    bool capture = false;
    bool enPassant = false;
    bool castle = false;
};

struct Board {
    array<char, 64> sq{};
    bool white = true;
    bool wk = false, wq = false, bk = false, bq = false;
    int ep = -1;
    int halfmove = 0, fullmove = 1;
};

static chrono::steady_clock::time_point deadline;
static atomic_bool stopped = false;
static Move rootBest;

static int fileOf(int s) { return s & 7; }
static int rankOf(int s) { return s >> 3; }
static bool onBoard(int f, int r) { return f >= 0 && f < 8 && r >= 0 && r < 8; }
static bool isWhitePiece(char p) { return p >= 'A' && p <= 'Z'; }
static bool isBlackPiece(char p) { return p >= 'a' && p <= 'z'; }
static bool sameColor(char p, bool white) { return p && (white ? isWhitePiece(p) : isBlackPiece(p)); }
static bool enemyColor(char p, bool white) { return p && (white ? isBlackPiece(p) : isWhitePiece(p)); }
static char lowerPiece(char p) { return static_cast<char>(tolower(static_cast<unsigned char>(p))); }

static string squareName(int s) {
    string r;
    r.push_back(static_cast<char>('a' + fileOf(s)));
    r.push_back(static_cast<char>('1' + rankOf(s)));
    return r;
}

static int squareFromName(const string& s) {
    if (s.size() != 2 || s[0] < 'a' || s[0] > 'h' || s[1] < '1' || s[1] > '8') return -1;
    return (s[1] - '1') * 8 + (s[0] - 'a');
}

static Board parseFen(const string& fen) {
    Board b;
    b.sq.fill(0);
    istringstream in(fen);
    string placement, side, castling, ep;
    in >> placement >> side >> castling >> ep >> b.halfmove >> b.fullmove;
    int r = 7, f = 0;
    for (char c : placement) {
        if (c == '/') { --r; f = 0; }
        else if (isdigit(static_cast<unsigned char>(c))) f += c - '0';
        else if (onBoard(f, r)) b.sq[r * 8 + f++] = c;
    }
    b.white = side != "b";
    b.wk = castling.find('K') != string::npos;
    b.wq = castling.find('Q') != string::npos;
    b.bk = castling.find('k') != string::npos;
    b.bq = castling.find('q') != string::npos;
    b.ep = ep == "-" ? -1 : squareFromName(ep);
    return b;
}

static int kingSquare(const Board& b, bool white) {
    char k = white ? 'K' : 'k';
    for (int i = 0; i < 64; ++i) if (b.sq[i] == k) return i;
    return -1;
}

static bool attackedBy(const Board& b, int s, bool byWhite) {
    int f = fileOf(s), r = rankOf(s);
    int pr = r + (byWhite ? -1 : 1);
    if (pr >= 0 && pr < 8) {
        for (int df : {-1, 1}) {
            int pf = f + df;
            if (pf >= 0 && pf < 8 && b.sq[pr * 8 + pf] == (byWhite ? 'P' : 'p')) return true;
        }
    }
    static const int nf[8] = {1,2,2,1,-1,-2,-2,-1};
    static const int nr[8] = {2,1,-1,-2,-2,-1,1,2};
    for (int i = 0; i < 8; ++i) {
        int ff = f + nf[i], rr = r + nr[i];
        if (onBoard(ff, rr) && b.sq[rr * 8 + ff] == (byWhite ? 'N' : 'n')) return true;
    }
    static const int bf[4] = {1,1,-1,-1};
    static const int br[4] = {1,-1,1,-1};
    for (int d = 0; d < 4; ++d) {
        int ff = f + bf[d], rr = r + br[d];
        while (onBoard(ff, rr)) {
            char p = b.sq[rr * 8 + ff];
            if (p) {
                if ((byWhite ? isWhitePiece(p) : isBlackPiece(p)) && (lowerPiece(p) == 'b' || lowerPiece(p) == 'q')) return true;
                break;
            }
            ff += bf[d]; rr += br[d];
        }
    }
    static const int rf[4] = {1,-1,0,0};
    static const int rrD[4] = {0,0,1,-1};
    for (int d = 0; d < 4; ++d) {
        int ff = f + rf[d], rr = r + rrD[d];
        while (onBoard(ff, rr)) {
            char p = b.sq[rr * 8 + ff];
            if (p) {
                if ((byWhite ? isWhitePiece(p) : isBlackPiece(p)) && (lowerPiece(p) == 'r' || lowerPiece(p) == 'q')) return true;
                break;
            }
            ff += rf[d]; rr += rrD[d];
        }
    }
    for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) {
        if (!df && !dr) continue;
        int ff = f + df, rr = r + dr;
        if (onBoard(ff, rr) && b.sq[rr * 8 + ff] == (byWhite ? 'K' : 'k')) return true;
    }
    return false;
}

static bool inCheck(const Board& b, bool white) {
    int k = kingSquare(b, white);
    return k >= 0 && attackedBy(b, k, !white);
}

static void addMove(vector<Move>& out, int from, int to, const Board& b, char promo = 0, bool ep = false, bool castle = false) {
    Move m;
    m.from = from; m.to = to; m.promo = promo; m.enPassant = ep; m.castle = castle;
    m.capture = ep || b.sq[to] != 0;
    out.push_back(m);
}

static Board makeMove(const Board& b, const Move& m) {
    Board n = b;
    char p = n.sq[m.from];
    char target = n.sq[m.to];
    n.sq[m.from] = 0;
    if (m.enPassant) n.sq[m.to + (b.white ? -8 : 8)] = 0;
    if (m.castle) {
        if (m.to == 6) { n.sq[5] = 'R'; n.sq[7] = 0; }
        else if (m.to == 2) { n.sq[3] = 'R'; n.sq[0] = 0; }
        else if (m.to == 62) { n.sq[61] = 'r'; n.sq[63] = 0; }
        else if (m.to == 58) { n.sq[59] = 'r'; n.sq[56] = 0; }
    }
    if (m.promo) n.sq[m.to] = b.white ? m.promo : static_cast<char>(tolower(static_cast<unsigned char>(m.promo)));
    else n.sq[m.to] = p;

    if (p == 'K') n.wk = n.wq = false;
    if (p == 'k') n.bk = n.bq = false;
    if (p == 'R' && m.from == 0) n.wq = false;
    if (p == 'R' && m.from == 7) n.wk = false;
    if (p == 'r' && m.from == 56) n.bq = false;
    if (p == 'r' && m.from == 63) n.bk = false;
    if (target == 'R' && m.to == 0) n.wq = false;
    if (target == 'R' && m.to == 7) n.wk = false;
    if (target == 'r' && m.to == 56) n.bq = false;
    if (target == 'r' && m.to == 63) n.bk = false;

    n.ep = -1;
    if (lowerPiece(p) == 'p' && abs(m.to - m.from) == 16) n.ep = (m.from + m.to) / 2;
    n.white = !b.white;
    if (n.white) ++n.fullmove;
    n.halfmove = (lowerPiece(p) == 'p' || m.capture) ? 0 : b.halfmove + 1;
    return n;
}

static vector<Move> pseudoMoves(const Board& b, bool capturesOnly = false) {
    vector<Move> mv;
    bool w = b.white;
    for (int s = 0; s < 64; ++s) {
        char p = b.sq[s];
        if (!sameColor(p, w)) continue;
        int f = fileOf(s), r = rankOf(s);
        char lp = lowerPiece(p);
        if (lp == 'p') {
            int dir = w ? 1 : -1;
            int one = s + dir * 8;
            int promoteRank = w ? 6 : 1;
            if (!capturesOnly && r != promoteRank && one >= 0 && one < 64 && !b.sq[one]) {
                addMove(mv, s, one, b);
                int startRank = w ? 1 : 6;
                int two = s + dir * 16;
                if (r == startRank && !b.sq[two]) addMove(mv, s, two, b);
            } else if (!capturesOnly && r == promoteRank && one >= 0 && one < 64 && !b.sq[one]) {
                for (char pr : {'Q','R','B','N'}) addMove(mv, s, one, b, pr);
            }
            for (int df : {-1, 1}) {
                int ff = f + df, rr = r + dir;
                if (!onBoard(ff, rr)) continue;
                int t = rr * 8 + ff;
                if (enemyColor(b.sq[t], w)) {
                    if (r == promoteRank) for (char pr : {'Q','R','B','N'}) addMove(mv, s, t, b, pr);
                    else addMove(mv, s, t, b);
                }
                if (t == b.ep) addMove(mv, s, t, b, 0, true);
            }
        } else if (lp == 'n') {
            static const int df[8] = {1,2,2,1,-1,-2,-2,-1};
            static const int dr[8] = {2,1,-1,-2,-2,-1,1,2};
            for (int i = 0; i < 8; ++i) {
                int ff = f + df[i], rr = r + dr[i];
                if (!onBoard(ff, rr)) continue;
                int t = rr * 8 + ff;
                if (!sameColor(b.sq[t], w) && (!capturesOnly || b.sq[t])) addMove(mv, s, t, b);
            }
        } else if (lp == 'b' || lp == 'r' || lp == 'q') {
            static const int df[8] = {1,1,-1,-1,1,-1,0,0};
            static const int dr[8] = {1,-1,1,-1,0,0,1,-1};
            int begin = lp == 'b' ? 0 : (lp == 'r' ? 4 : 0);
            int end = lp == 'b' ? 4 : 8;
            for (int d = begin; d < end; ++d) {
                int ff = f + df[d], rr = r + dr[d];
                while (onBoard(ff, rr)) {
                    int t = rr * 8 + ff;
                    if (sameColor(b.sq[t], w)) break;
                    if (!capturesOnly || b.sq[t]) addMove(mv, s, t, b);
                    if (b.sq[t]) break;
                    ff += df[d]; rr += dr[d];
                }
            }
        } else if (lp == 'k') {
            for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) {
                if (!df && !dr) continue;
                int ff = f + df, rr = r + dr;
                if (!onBoard(ff, rr)) continue;
                int t = rr * 8 + ff;
                if (!sameColor(b.sq[t], w) && (!capturesOnly || b.sq[t])) addMove(mv, s, t, b);
            }
            if (!capturesOnly && !inCheck(b, w)) {
                if (w && s == 4 && b.wk && b.sq[7] == 'R' && !b.sq[5] && !b.sq[6] && !attackedBy(b, 5, false) && !attackedBy(b, 6, false)) addMove(mv, 4, 6, b, 0, false, true);
                if (w && s == 4 && b.wq && b.sq[0] == 'R' && !b.sq[3] && !b.sq[2] && !b.sq[1] && !attackedBy(b, 3, false) && !attackedBy(b, 2, false)) addMove(mv, 4, 2, b, 0, false, true);
                if (!w && s == 60 && b.bk && b.sq[63] == 'r' && !b.sq[61] && !b.sq[62] && !attackedBy(b, 61, true) && !attackedBy(b, 62, true)) addMove(mv, 60, 62, b, 0, false, true);
                if (!w && s == 60 && b.bq && b.sq[56] == 'r' && !b.sq[59] && !b.sq[58] && !b.sq[57] && !attackedBy(b, 59, true) && !attackedBy(b, 58, true)) addMove(mv, 60, 58, b, 0, false, true);
            }
        }
    }
    return mv;
}

static vector<Move> legalMoves(const Board& b, bool capturesOnly = false) {
    vector<Move> out;
    for (const Move& m : pseudoMoves(b, capturesOnly)) {
        Board n = makeMove(b, m);
        if (!inCheck(n, !n.white)) out.push_back(m);
    }
    return out;
}

static int pieceValue(char p) {
    switch (lowerPiece(p)) {
        case 'p': return 100;
        case 'n': return 320;
        case 'b': return 330;
        case 'r': return 500;
        case 'q': return 900;
        case 'k': return 0;
    }
    return 0;
}

static int pstBonus(char p, int s) {
    int f = fileOf(s), r = rankOf(s);
    if (isBlackPiece(p)) r = 7 - r;
    int center = 14 - (abs(f - 3) + abs(f - 4) + abs(r - 3) + abs(r - 4)) * 2;
    switch (lowerPiece(p)) {
        case 'p': return r * 8 - abs(f - 3) * 2;
        case 'n': return center * 3;
        case 'b': return center * 2;
        case 'r': return r * 2;
        case 'q': return center;
        case 'k': return r < 2 ? -center : center / 2;
    }
    return 0;
}

static int evaluate(const Board& b) {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        char p = b.sq[i];
        if (!p) continue;
        int v = pieceValue(p) + pstBonus(p, i);
        score += isWhitePiece(p) ? v : -v;
    }
    return b.white ? score : -score;
}

static bool timeUp() {
    if (stopped.load()) return true;
    if (chrono::steady_clock::now() >= deadline) {
        stopped.store(true);
        return true;
    }
    return false;
}

static int moveScore(const Board& b, const Move& m) {
    char moving = b.sq[m.from], captured = m.enPassant ? (b.white ? 'p' : 'P') : b.sq[m.to];
    int s = 0;
    if (captured) s += 10 * pieceValue(captured) - pieceValue(moving);
    if (m.promo) s += pieceValue(m.promo);
    if (m.castle) s += 50;
    return s;
}

static int quiesce(const Board& b, int alpha, int beta) {
    if (timeUp()) return evaluate(b);
    int stand = evaluate(b);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;
    vector<Move> mv = legalMoves(b, true);
    sort(mv.begin(), mv.end(), [&](const Move& a, const Move& c) { return moveScore(b, a) > moveScore(b, c); });
    for (const Move& m : mv) {
        Board n = makeMove(b, m);
        int score = -quiesce(n, -beta, -alpha);
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

static int search(const Board& b, int depth, int alpha, int beta, int ply) {
    if (timeUp()) return evaluate(b);
    if (depth <= 0) return quiesce(b, alpha, beta);
    vector<Move> mv = legalMoves(b);
    if (mv.empty()) return inCheck(b, b.white) ? -100000 + ply : 0;
    sort(mv.begin(), mv.end(), [&](const Move& a, const Move& c) { return moveScore(b, a) > moveScore(b, c); });
    Move bestLocal = mv[0];
    for (const Move& m : mv) {
        Board n = makeMove(b, m);
        int score = -search(n, depth - 1, -beta, -alpha, ply + 1);
        if (timeUp()) return alpha;
        if (score > alpha) {
            alpha = score;
            bestLocal = m;
            if (ply == 0) rootBest = m;
        }
        if (alpha >= beta) break;
    }
    if (ply == 0) rootBest = bestLocal;
    return alpha;
}

static Move chooseMove(const Board& b, int timeoutSeconds) {
    vector<Move> mv = legalMoves(b);
    if (mv.empty()) return Move();
    sort(mv.begin(), mv.end(), [&](const Move& a, const Move& c) { return moveScore(b, a) > moveScore(b, c); });
    rootBest = mv[0];
    auto now = chrono::steady_clock::now();
    int reserveMs = max(100, min(750, timeoutSeconds * 100));
    deadline = now + chrono::milliseconds(timeoutSeconds * 1000 - reserveMs);
    stopped.store(false);
    for (int depth = 1; depth <= 64; ++depth) {
        Move before = rootBest;
        vector<future<pair<int, Move>>> jobs;
        jobs.reserve(mv.size());
        for (const Move& m : mv) {
            jobs.push_back(async(launch::async, [b, m, depth]() {
                Board n = makeMove(b, m);
                int score = -search(n, depth - 1, -1000000, 1000000, 1);
                return pair<int, Move>(score, m);
            }));
        }
        int bestScore = -1000001;
        Move completedBest = before;
        for (auto& job : jobs) {
            auto result = job.get();
            if (result.first > bestScore) {
                bestScore = result.first;
                completedBest = result.second;
            }
        }
        if (stopped.load()) {
            rootBest = before;
            break;
        }
        rootBest = completedBest;
    }
    return rootBest;
}

static string formatMove(const Board& b, const Move& m) {
    if (m.from < 0) return "resign";
    Board n = makeMove(b, m);
    bool check = inCheck(n, n.white);
    bool mate = check && legalMoves(n).empty();
    string out;
    if (m.castle) out = (m.to > m.from) ? "O-O" : "O-O-O";
    else {
        char p = b.sq[m.from];
        char lp = lowerPiece(p);
        out.push_back(lp == 'p' ? 'p' : static_cast<char>(toupper(static_cast<unsigned char>(lp))));
        out += squareName(m.from);
        if (m.capture) out.push_back('x');
        out += squareName(m.to);
        if (m.promo) {
            out.push_back('/');
            out.push_back(m.promo);
        }
        if (m.enPassant) out += "ep";
    }
    if (mate) out.push_back('#');
    else if (check) out.push_back('+');
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int timeoutSeconds = 5;
    string fen;
    if (!(cin >> timeoutSeconds)) return 0;
    string rest;
    getline(cin, rest);
    getline(cin, fen);
    Board b = parseFen(fen);
    Move best = chooseMove(b, timeoutSeconds);
    cout << formatMove(b, best) << '\n';
    return 0;
}
