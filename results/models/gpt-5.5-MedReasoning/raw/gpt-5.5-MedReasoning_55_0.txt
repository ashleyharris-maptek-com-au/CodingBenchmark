using namespace std;

struct Move {
    int from = -1, to = -1;
    char promo = 0;
    bool ep = false;
    bool castle = false;
};

struct Undo {
    array<char, 64> b;
    bool white;
    int castling;
    int ep;
    int halfmove;
};

struct TTEntry {
    int depth = -1;
    int score = 0;
    int flag = 0;
    Move best;
};

static const int INF = 1000000000;
static chrono::steady_clock::time_point deadline;
static bool stopped = false;
static uint64_t nodes = 0;
static unordered_map<uint64_t, TTEntry> tt;
static uint64_t zob[64][12], zobSide, zobCastle[16], zobEp[64];

int fileOf(int s) { return s & 7; }
int rankOf(int s) { return s >> 3; }
bool on(int f, int r) { return f >= 0 && f < 8 && r >= 0 && r < 8; }
bool isW(char p) { return p >= 'A' && p <= 'Z'; }
bool isB(char p) { return p >= 'a' && p <= 'z'; }
bool empty(char p) { return p == '.'; }
int sq(int f, int r) { return r * 8 + f; }
string sqName(int s) { string x; x += char('a' + fileOf(s)); x += char('1' + rankOf(s)); return x; }

int pieceIndex(char p) {
    switch (p) {
        case 'P': return 0; case 'N': return 1; case 'B': return 2; case 'R': return 3; case 'Q': return 4; case 'K': return 5;
        case 'p': return 6; case 'n': return 7; case 'b': return 8; case 'r': return 9; case 'q': return 10; case 'k': return 11;
    }
    return -1;
}

int val(char p) {
    switch (tolower((unsigned char)p)) {
        case 'p': return 100;
        case 'n': return 320;
        case 'b': return 330;
        case 'r': return 500;
        case 'q': return 900;
        case 'k': return 0;
    }
    return 0;
}

struct Pos {
    array<char, 64> b{};
    bool white = true;
    int castling = 0; // 1 K, 2 Q, 4 k, 8 q
    int ep = -1;
    int halfmove = 0, fullmove = 1;

    uint64_t hash() const {
        uint64_t h = 0;
        for (int i = 0; i < 64; ++i) {
            int pi = pieceIndex(b[i]);
            if (pi >= 0) h ^= zob[i][pi];
        }
        if (!white) h ^= zobSide;
        h ^= zobCastle[castling];
        if (ep >= 0) h ^= zobEp[ep];
        return h;
    }
};

void initZobrist() {
    mt19937_64 rng(123456789);
    for (auto &a : zob) for (auto &x : a) x = rng();
    zobSide = rng();
    for (auto &x : zobCastle) x = rng();
    for (auto &x : zobEp) x = rng();
}

Pos parseFen(const string& fen) {
    Pos p;
    p.b.fill('.');
    string board, stm, castle, ep, hm, fm;
    stringstream ss(fen);
    ss >> board >> stm >> castle >> ep >> hm >> fm;
    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') { --r; f = 0; }
        else if (isdigit((unsigned char)c)) f += c - '0';
        else p.b[sq(f++, r)] = c;
    }
    p.white = (stm != "b");
    p.castling = 0;
    if (castle.find('K') != string::npos) p.castling |= 1;
    if (castle.find('Q') != string::npos) p.castling |= 2;
    if (castle.find('k') != string::npos) p.castling |= 4;
    if (castle.find('q') != string::npos) p.castling |= 8;
    if (ep != "-" && ep.size() == 2) p.ep = sq(ep[0] - 'a', ep[1] - '1');
    if (!hm.empty()) p.halfmove = stoi(hm);
    if (!fm.empty()) p.fullmove = stoi(fm);
    return p;
}

bool attacked(const Pos& p, int s, bool byWhite) {
    int f = fileOf(s), r = rankOf(s);
    int pr = byWhite ? r - 1 : r + 1;
    for (int df : {-1, 1}) {
        int ff = f + df;
        if (on(ff, pr) && p.b[sq(ff, pr)] == (byWhite ? 'P' : 'p')) return true;
    }
    static const int nf[8] = {1,2,2,1,-1,-2,-2,-1};
    static const int nr[8] = {2,1,-1,-2,-2,-1,1,2};
    for (int i = 0; i < 8; ++i) {
        int ff = f + nf[i], rr = r + nr[i];
        if (on(ff, rr) && p.b[sq(ff, rr)] == (byWhite ? 'N' : 'n')) return true;
    }
    static const int bd[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (auto &d : bd) {
        int ff = f + d[0], rr = r + d[1];
        while (on(ff, rr)) {
            char pc = p.b[sq(ff, rr)];
            if (!empty(pc)) {
                if ((byWhite ? isW(pc) : isB(pc)) && (tolower((unsigned char)pc) == 'b' || tolower((unsigned char)pc) == 'q')) return true;
                break;
            }
            ff += d[0]; rr += d[1];
        }
    }
    static const int rd[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (auto &d : rd) {
        int ff = f + d[0], rr = r + d[1];
        while (on(ff, rr)) {
            char pc = p.b[sq(ff, rr)];
            if (!empty(pc)) {
                if ((byWhite ? isW(pc) : isB(pc)) && (tolower((unsigned char)pc) == 'r' || tolower((unsigned char)pc) == 'q')) return true;
                break;
            }
            ff += d[0]; rr += d[1];
        }
    }
    for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) if (df || dr) {
        int ff = f + df, rr = r + dr;
        if (on(ff, rr) && p.b[sq(ff, rr)] == (byWhite ? 'K' : 'k')) return true;
    }
    return false;
}

int kingSq(const Pos& p, bool white) {
    char k = white ? 'K' : 'k';
    for (int i = 0; i < 64; ++i) if (p.b[i] == k) return i;
    return -1;
}

bool inCheck(const Pos& p, bool white) {
    int k = kingSq(p, white);
    return k >= 0 && attacked(p, k, !white);
}

void addPawn(vector<Move>& m, const Pos& p, int s, int to, bool cap, bool ep = false) {
    int r = rankOf(to);
    if (r == 0 || r == 7) {
        for (char q : {'Q','R','B','N'}) m.push_back({s, to, q, ep, false});
    } else {
        m.push_back({s, to, 0, ep, false});
    }
}

vector<Move> pseudo(const Pos& p) {
    vector<Move> m;
    bool usW = p.white;
    for (int s = 0; s < 64; ++s) {
        char pc = p.b[s];
        if (empty(pc) || (usW ? !isW(pc) : !isB(pc))) continue;
        int f = fileOf(s), r = rankOf(s);
        char tpc = (char)tolower((unsigned char)pc);
        if (tpc == 'p') {
            int dir = usW ? 1 : -1, start = usW ? 1 : 6;
            int one = sq(f, r + dir);
            if (on(f, r + dir) && empty(p.b[one])) {
                addPawn(m, p, s, one, false);
                int two = sq(f, r + 2 * dir);
                if (r == start && empty(p.b[two])) m.push_back({s, two, 0, false, false});
            }
            for (int df : {-1, 1}) {
                int ff = f + df, rr = r + dir;
                if (!on(ff, rr)) continue;
                int to = sq(ff, rr);
                if (!empty(p.b[to]) && (usW ? isB(p.b[to]) : isW(p.b[to]))) addPawn(m, p, s, to, true);
                if (to == p.ep) addPawn(m, p, s, to, true, true);
            }
        } else if (tpc == 'n') {
            static const int nf[8] = {1,2,2,1,-1,-2,-2,-1};
            static const int nr[8] = {2,1,-1,-2,-2,-1,1,2};
            for (int i = 0; i < 8; ++i) {
                int ff = f + nf[i], rr = r + nr[i];
                if (on(ff, rr)) {
                    int to = sq(ff, rr);
                    if (empty(p.b[to]) || (usW ? isB(p.b[to]) : isW(p.b[to]))) m.push_back({s, to, 0, false, false});
                }
            }
        } else if (tpc == 'b' || tpc == 'r' || tpc == 'q') {
            vector<pair<int,int>> ds;
            if (tpc == 'b' || tpc == 'q') for (auto d : {pair{1,1}, pair{1,-1}, pair{-1,1}, pair{-1,-1}}) ds.push_back(d);
            if (tpc == 'r' || tpc == 'q') for (auto d : {pair{1,0}, pair{-1,0}, pair{0,1}, pair{0,-1}}) ds.push_back(d);
            for (auto [df, dr] : ds) {
                int ff = f + df, rr = r + dr;
                while (on(ff, rr)) {
                    int to = sq(ff, rr);
                    if (empty(p.b[to])) m.push_back({s, to, 0, false, false});
                    else {
                        if (usW ? isB(p.b[to]) : isW(p.b[to])) m.push_back({s, to, 0, false, false});
                        break;
                    }
                    ff += df; rr += dr;
                }
            }
        } else if (tpc == 'k') {
            for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) if (df || dr) {
                int ff = f + df, rr = r + dr;
                if (on(ff, rr)) {
                    int to = sq(ff, rr);
                    if (empty(p.b[to]) || (usW ? isB(p.b[to]) : isW(p.b[to]))) m.push_back({s, to, 0, false, false});
                }
            }
            if (usW && s == sq(4,0) && !inCheck(p, true)) {
                if ((p.castling & 1) && p.b[sq(5,0)] == '.' && p.b[sq(6,0)] == '.' && !attacked(p, sq(5,0), false) && !attacked(p, sq(6,0), false)) m.push_back({s, sq(6,0), 0, false, true});
                if ((p.castling & 2) && p.b[sq(3,0)] == '.' && p.b[sq(2,0)] == '.' && p.b[sq(1,0)] == '.' && !attacked(p, sq(3,0), false) && !attacked(p, sq(2,0), false)) m.push_back({s, sq(2,0), 0, false, true});
            }
            if (!usW && s == sq(4,7) && !inCheck(p, false)) {
                if ((p.castling & 4) && p.b[sq(5,7)] == '.' && p.b[sq(6,7)] == '.' && !attacked(p, sq(5,7), true) && !attacked(p, sq(6,7), true)) m.push_back({s, sq(6,7), 0, false, true});
                if ((p.castling & 8) && p.b[sq(3,7)] == '.' && p.b[sq(2,7)] == '.' && p.b[sq(1,7)] == '.' && !attacked(p, sq(3,7), true) && !attacked(p, sq(2,7), true)) m.push_back({s, sq(2,7), 0, false, true});
            }
        }
    }
    return m;
}

Undo makeMove(Pos& p, const Move& m) {
    Undo u{p.b, p.white, p.castling, p.ep, p.halfmove};
    char pc = p.b[m.from], cap = p.b[m.to];
    p.ep = -1;
    p.b[m.from] = '.';
    if (m.ep) p.b[sq(fileOf(m.to), rankOf(m.from))] = '.';
    if (m.castle) {
        p.b[m.to] = pc;
        if (m.to == sq(6,0)) { p.b[sq(5,0)] = 'R'; p.b[sq(7,0)] = '.'; }
        if (m.to == sq(2,0)) { p.b[sq(3,0)] = 'R'; p.b[sq(0,0)] = '.'; }
        if (m.to == sq(6,7)) { p.b[sq(5,7)] = 'r'; p.b[sq(7,7)] = '.'; }
        if (m.to == sq(2,7)) { p.b[sq(3,7)] = 'r'; p.b[sq(0,7)] = '.'; }
    } else {
        char np = pc;
        if (m.promo) np = isW(pc) ? m.promo : (char)tolower((unsigned char)m.promo);
        p.b[m.to] = np;
    }
    if (tolower((unsigned char)pc) == 'p' && abs(m.to - m.from) == 16) p.ep = (m.from + m.to) / 2;
    if (pc == 'K') p.castling &= ~3;
    if (pc == 'k') p.castling &= ~12;
    if (m.from == sq(0,0) || m.to == sq(0,0)) p.castling &= ~2;
    if (m.from == sq(7,0) || m.to == sq(7,0)) p.castling &= ~1;
    if (m.from == sq(0,7) || m.to == sq(0,7)) p.castling &= ~8;
    if (m.from == sq(7,7) || m.to == sq(7,7)) p.castling &= ~4;
    p.halfmove = (tolower((unsigned char)pc) == 'p' || !empty(cap) || m.ep) ? 0 : p.halfmove + 1;
    p.white = !p.white;
    return u;
}

void undoMove(Pos& p, const Undo& u) {
    p.b = u.b; p.white = u.white; p.castling = u.castling; p.ep = u.ep; p.halfmove = u.halfmove;
}

vector<Move> legal(Pos& p) {
    vector<Move> res;
    bool side = p.white;
    for (const Move& m : pseudo(p)) {
        Undo u = makeMove(p, m);
        bool ok = !inCheck(p, side);
        undoMove(p, u);
        if (ok) res.push_back(m);
    }
    return res;
}

int pst(char pc, int s) {
    int f = fileOf(s), r = rankOf(s);
    if (isB(pc)) r = 7 - r;
    int center = 14 - (abs(f - 3) + abs(f - 4) + abs(r - 3) + abs(r - 4)) * 2;
    switch (tolower((unsigned char)pc)) {
        case 'p': return r * 8 - abs(f - 3) * 2;
        case 'n': return center * 3;
        case 'b': return center * 2;
        case 'r': return r * 2;
        case 'q': return center;
        case 'k': return -center;
    }
    return 0;
}

int evaluate(const Pos& p) {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        char pc = p.b[i];
        if (empty(pc)) continue;
        int v = val(pc) + pst(pc, i);
        score += isW(pc) ? v : -v;
    }
    return p.white ? score : -score;
}

int moveScore(const Pos& p, const Move& m, const Move& ttBest) {
    if (m.from == ttBest.from && m.to == ttBest.to && m.promo == ttBest.promo) return 10000000;
    char pc = p.b[m.from], cap = m.ep ? (p.white ? 'p' : 'P') : p.b[m.to];
    int s = 0;
    if (!empty(cap)) s += 10000 + val(cap) * 10 - val(pc);
    if (m.promo) s += val(m.promo);
    if (m.castle) s += 50;
    return s;
}

bool timeUp() {
    if ((++nodes & 2047) == 0 && chrono::steady_clock::now() >= deadline) stopped = true;
    return stopped;
}

int qsearch(Pos& p, int alpha, int beta) {
    if (timeUp()) return evaluate(p);
    bool checked = inCheck(p, p.white);
    if (!checked) {
        int stand = evaluate(p);
        if (stand >= beta) return beta;
        alpha = max(alpha, stand);
    }
    auto ms = legal(p);
    if (checked && ms.empty()) return -100000;
    if (!checked) ms.erase(remove_if(ms.begin(), ms.end(), [&](const Move& m) { return p.b[m.to] == '.' && !m.ep && !m.promo; }), ms.end());
    sort(ms.begin(), ms.end(), [&](const Move& a, const Move& b) { return moveScore(p, a, Move{}) > moveScore(p, b, Move{}); });
    for (const Move& m : ms) {
        Undo u = makeMove(p, m);
        int sc = -qsearch(p, -beta, -alpha);
        undoMove(p, u);
        if (stopped) return alpha;
        if (sc >= beta) return beta;
        alpha = max(alpha, sc);
    }
    return alpha;
}

int search(Pos& p, int depth, int alpha, int beta, Move& best) {
    if (timeUp()) return evaluate(p);
    if (depth <= 0) return qsearch(p, alpha, beta);
    uint64_t h = p.hash();
    int origAlpha = alpha;
    Move ttBest;
    auto it = tt.find(h);
    if (it != tt.end() && it->second.depth >= depth) {
        const TTEntry& e = it->second;
        ttBest = e.best;
        if (e.flag == 0) return e.score;
        if (e.flag == 1) alpha = max(alpha, e.score);
        if (e.flag == 2) beta = min(beta, e.score);
        if (alpha >= beta) return e.score;
    } else if (it != tt.end()) {
        ttBest = it->second.best;
    }
    auto ms = legal(p);
    if (ms.empty()) return inCheck(p, p.white) ? -100000 + (20 - depth) : 0;
    sort(ms.begin(), ms.end(), [&](const Move& a, const Move& b) { return moveScore(p, a, ttBest) > moveScore(p, b, ttBest); });
    int bestScore = -INF;
    Move localBest = ms[0];
    for (const Move& m : ms) {
        Undo u = makeMove(p, m);
        Move child;
        int sc = -search(p, depth - 1, -beta, -alpha, child);
        undoMove(p, u);
        if (stopped) return alpha;
        if (sc > bestScore) { bestScore = sc; localBest = m; }
        if (sc > alpha) { alpha = sc; best = m; }
        if (alpha >= beta) break;
    }
    TTEntry e;
    e.depth = depth; e.score = bestScore; e.best = localBest;
    e.flag = (bestScore <= origAlpha) ? 2 : (bestScore >= beta ? 1 : 0);
    tt[h] = e;
    return bestScore;
}

string notation(Pos p, const Move& m) {
    char pc = p.b[m.from];
    if (m.castle) {
        Undo u = makeMove(p, m);
        bool chk = inCheck(p, p.white);
        auto ms = legal(p);
        string s = (fileOf(m.to) == 6) ? "O-O" : "O-O-O";
        if (chk && ms.empty()) s += "#";
        else if (chk) s += "+";
        return s;
    }
    string s;
    s += (tolower((unsigned char)pc) == 'p') ? 'p' : (char)toupper((unsigned char)pc);
    s += sqName(m.from);
    bool cap = m.ep || p.b[m.to] != '.';
    s += cap ? "x" : "";
    s += sqName(m.to);
    if (m.promo) { s += "/"; s += m.promo; }
    if (m.ep) s += "ep";
    Undo u = makeMove(p, m);
    bool chk = inCheck(p, p.white);
    auto ms = legal(p);
    if (chk && ms.empty()) s += "#";
    else if (chk) s += "+";
    return s;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    initZobrist();
    double timeout;
    string fen;
    if (!(cin >> timeout)) return 0;
    string dummy;
    getline(cin, dummy);
    getline(cin, fen);
    Pos p = parseFen(fen);
    auto start = chrono::steady_clock::now();
    int millis = max(100, (int)(timeout * 1000.0) - 150);
    deadline = start + chrono::milliseconds(millis);
    Move best;
    auto ms = legal(p);
    if (ms.empty()) return 0;
    best = ms[0];
    for (int depth = 1; depth <= 64; ++depth) {
        Move iterBest = best;
        int score = search(p, depth, -INF, INF, iterBest);
        (void)score;
        if (stopped) break;
        best = iterBest;
    }
    cout << notation(p, best) << "\n";
    return 0;
}
