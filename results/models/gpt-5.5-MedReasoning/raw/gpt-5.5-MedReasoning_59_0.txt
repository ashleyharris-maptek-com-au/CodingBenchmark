const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);
const height = grid.length;
const width = height ? grid[0].length : 0;
const n = width * height;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < width && y < height;
}

function idOf(x, y) {
  return y * width + x;
}

function xyOf(id) {
  return [id % width, Math.floor(id / width)];
}

const value = new Float64Array(n);
const blocked = new Uint8Array(n);
let totalPositive = 0;
const moveDx = [0, 1, -1, 0, 0];
const moveDy = [0, 0, 0, 1, -1];

for (let y = 0; y < height; y++) {
  const row = grid[y] || [];
  for (let x = 0; x < width; x++) {
    const id = idOf(x, y);
    const v = Number(row[x] || 0);
    if (v < 0) {
      blocked[id] = 1;
    } else {
      value[id] = v;
      if (v > 0) totalPositive++;
    }
  }
}

const starts = drones.map((d) => {
  const x = d.x | 0;
  const y = d.y | 0;
  if (inBounds(x, y) && !blocked[idOf(x, y)]) return idOf(x, y);
  for (let i = 0; i < n; i++) {
    if (!blocked[i]) return i;
  }
  return 0;
});

if (!width || !height || n === 0 || starts.length === 0) {
  console.log(JSON.stringify({ paths: drones.map(() => []) }));
  process.exit(0);
}

const component = new Int32Array(n);
component.fill(-1);
const componentRemaining = [];
const componentQueue = new Int32Array(n);
let componentCount = 0;

for (let i = 0; i < n; i++) {
  if (blocked[i] || component[i] !== -1) continue;
  let head = 0;
  let tail = 0;
  let positives = 0;
  component[i] = componentCount;
  componentQueue[tail++] = i;
  while (head < tail) {
    const cur = componentQueue[head++];
    if (value[cur] > 0) positives++;
    for (let m = 1; m <= 4; m++) {
      const x = cur % width;
      const y = (cur / width) | 0;
      const nx = x + moveDx[m];
      const ny = y + moveDy[m];
      if (nx < 0 || ny < 0 || nx >= width || ny >= height) continue;
      const nid = ny * width + nx;
      if (blocked[nid] || component[nid] !== -1) continue;
      component[nid] = componentCount;
      componentQueue[tail++] = nid;
    }
  }
  componentRemaining[componentCount++] = positives;
}

const reachableComponent = new Uint8Array(componentCount);
for (const s of starts) {
  if (component[s] >= 0) reachableComponent[component[s]] = 1;
}
let reachablePositive = 0;
for (let c = 0; c < componentCount; c++) {
  if (reachableComponent[c]) reachablePositive += componentRemaining[c];
}

const paths = starts.map((s) => [xyOf(s)]);
const pos = starts.slice();
const covered = new Uint8Array(n);
let coveredCount = 0;

for (const s of starts) {
  if (!covered[s] && value[s] > 0) {
    covered[s] = 1;
    coveredCount++;
    componentRemaining[component[s]]--;
  }
}

const q = new Int32Array(n);
const seen = new Int32Array(n);
const dist = new Int32Array(n);
const first = new Int8Array(n);
const reserved = new Int32Array(n);
let seenStamp = 1;
let reserveStamp = 1;

function neighborId(id, move) {
  const x = id % width;
  const y = (id / width) | 0;
  const nx = x + moveDx[move];
  const ny = y + moveDy[move];
  if (nx < 0 || ny < 0 || nx >= width || ny >= height) return -1;
  const nid = ny * width + nx;
  return blocked[nid] ? -1 : nid;
}

function localRadius(remaining) {
  if (remaining <= 0) return 0;
  if (n <= 2500) return Math.min(remaining, 22);
  if (n <= 10000) return Math.min(remaining, 16);
  if (n <= 50000) return Math.min(remaining, 11);
  return Math.min(remaining, 8);
}

function scoreReachable(start, remaining, radius, useFullFallback) {
  seenStamp++;
  if (seenStamp === 2147483000) {
    seen.fill(0);
    seenStamp = 1;
  }

  const scores = [0, 0, 0, 0, 0];
  let head = 0;
  let tail = 0;
  q[tail++] = start;
  seen[start] = seenStamp;
  dist[start] = 0;
  first[start] = 0;

  const maxVisits = useFullFallback ? n : n;
  let visits = 0;
  let bestFallbackMove = 0;
  let bestFallbackScore = -1;

  while (head < tail && visits < maxVisits) {
    const cur = q[head++];
    visits++;
    const d = dist[cur];

    if (d > 0 && d <= remaining && value[cur] > 0 && !covered[cur] && reserved[cur] !== reserveStamp) {
      const f = first[cur];
      if (useFullFallback) {
        const s = value[cur] / (d + 0.75);
        if (s > bestFallbackScore) {
          bestFallbackScore = s;
          bestFallbackMove = f;
        }
      } else {
        const closeness = 1 / (d + 0.55);
        const urgency = d <= 2 ? 1.35 : 1;
        scores[f] += value[cur] * closeness * urgency;
      }
    }

    if (d >= radius) continue;

    for (let m = 1; m <= 4; m++) {
      const nid = neighborId(cur, m);
      if (nid < 0 || seen[nid] === seenStamp) continue;
      seen[nid] = seenStamp;
      dist[nid] = d + 1;
      first[nid] = d === 0 ? m : first[cur];
      q[tail++] = nid;
    }
  }

  if (useFullFallback) return { move: bestFallbackMove, score: bestFallbackScore };
  return { scores };
}

function chooseMove(start, remaining) {
  const comp = component[start];
  if (comp < 0 || componentRemaining[comp] <= 0) return 0;

  const radius = localRadius(remaining);
  const local = scoreReachable(start, remaining, radius, false).scores;

  for (let m = 0; m <= 4; m++) {
    const nid = neighborId(start, m);
    if (nid < 0) {
      local[m] = -Infinity;
      continue;
    }
    if (value[nid] > 0 && !covered[nid] && reserved[nid] !== reserveStamp) {
      local[m] += value[nid] * 2.25;
    }
    if (m !== 0) local[m] += 1e-9;
  }

  let bestMove = 0;
  let bestScore = local[0];
  for (let m = 1; m <= 4; m++) {
    if (local[m] > bestScore) {
      bestScore = local[m];
      bestMove = m;
    }
  }

  if (bestScore > 0) return bestMove;

  const fallbackRadius = Math.min(remaining, n);
  const fallback = scoreReachable(start, remaining, fallbackRadius, true);
  if (fallback.score >= 0) return fallback.move;

  for (let m = 1; m <= 4; m++) {
    if (neighborId(start, m) >= 0) return m;
  }
  return 0;
}

for (let step = 0; step < maxSteps; step++) {
  if (coveredCount >= reachablePositive) {
    for (let i = 0; i < pos.length; i++) paths[i].push(xyOf(pos[i]));
    continue;
  }

  reserveStamp++;
  if (reserveStamp === 2147483000) {
    reserved.fill(0);
    reserveStamp = 1;
  }

  const next = pos.slice();
  const order = pos.map((_, i) => i);
  order.sort((a, b) => {
    const va = value[pos[a]] > 0 && !covered[pos[a]] ? value[pos[a]] : 0;
    const vb = value[pos[b]] > 0 && !covered[pos[b]] ? value[pos[b]] : 0;
    return vb - va;
  });

  for (const i of order) {
    const move = chooseMove(pos[i], maxSteps - step);
    const nid = neighborId(pos[i], move);
    next[i] = nid >= 0 ? nid : pos[i];
    if (value[next[i]] > 0 && !covered[next[i]]) reserved[next[i]] = reserveStamp;
  }

  for (let i = 0; i < pos.length; i++) {
    pos[i] = next[i];
    if (!covered[pos[i]] && value[pos[i]] > 0) {
      covered[pos[i]] = 1;
      coveredCount++;
      componentRemaining[component[pos[i]]]--;
    }
    paths[i].push(xyOf(pos[i]));
  }
}
