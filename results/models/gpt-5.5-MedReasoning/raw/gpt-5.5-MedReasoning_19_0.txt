using System;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const long SmallWorldCellLimit = 80_000_000L;
    private const long StreamedCellLimit = 400_000_000L;
    private const long ColumnLimit = 35_000_000L;
    private const int StripLookAhead = 18;

    private struct PathPlan
    {
        public int[] Coords;
        public int Count;
        public long FuelCost;
    }

    public static void Main()
    {
        Stream input = Console.OpenStandardInput();
        var reader = new FastInput(input);

        int xCount = reader.NextInt();
        int yCount = reader.NextInt();
        int zCount = reader.NextInt();
        reader.NextToken(); // XBlockSize
        reader.NextToken(); // YBlockSize
        reader.NextToken(); // ZBlockSize

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long columnsLong = (long)xCount * yCount;
        long cellsLong = columnsLong * zCount;

        if (columnsLong > ColumnLimit || cellsLong > StreamedCellLimit)
        {
            // Any useful plan would be too large to keep safely. Produce a valid empty plan.
            Console.WriteLine(0);
            return;
        }

        int columns = (int)columnsLong;
        int[] heights = new int[columns];
        byte[] top = new byte[columns];
        bool keepWorld = cellsLong <= SmallWorldCellLimit && cellsLong <= int.MaxValue;
        byte[] world = keepWorld ? new byte[(int)cellsLong] : Array.Empty<byte>();

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int col = Col(x, y, yCount);
                for (int z = 0; z < zCount; z++)
                {
                    int raw = reader.ReadDataByte();
                    if (raw < 0) raw = 0;
                    byte b = (byte)raw;
                    if (keepWorld) world[Index(x, y, z, yCount, zCount)] = b;
                    if ((b & 15) != 0)
                    {
                        heights[col] = z + 1;
                        top[col] = b;
                    }
                }
            }
        }

        string tempPath = Path.Combine(Path.GetTempPath(), "mine_plan_" + Guid.NewGuid().ToString("N") + ".txt");
        long operations = 0;

        using (var writer = new StreamWriter(tempPath, false, Encoding.ASCII, 1 << 20))
        {
            if (keepWorld)
                operations = PlanSmall(xCount, yCount, zCount, heights, world, writer);
            else
                operations = PlanLargeTopOnly(xCount, yCount, heights, top, writer);
        }

        using (var output = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 1 << 20))
        {
            output.WriteLine(operations.ToString(CultureInfo.InvariantCulture));
            using (var readerOps = new StreamReader(tempPath, Encoding.ASCII, false, 1 << 20))
            {
                char[] buffer = new char[1 << 16];
                int n;
                while ((n = readerOps.Read(buffer, 0, buffer.Length)) > 0)
                    output.Write(buffer, 0, n);
            }
        }

        try { File.Delete(tempPath); } catch { }
    }

    private static long PlanSmall(int xCount, int yCount, int zCount, int[] heights, byte[] world, StreamWriter writer)
    {
        long operations = 0;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int col = Col(x, y, yCount);

                while (heights[col] > 0)
                {
                    int z = heights[col] - 1;
                    byte b = world[Index(x, y, z, yCount, zCount)];
                    int mass = b & 15;
                    int value = b >> 4;
                    if (mass == 0)
                    {
                        heights[col]--;
                        continue;
                    }

                    bool sendToPlant = value > 0;
                    if (!sendToPlant)
                    {
                        int bestBelow = BestReachableValueBelow(x, y, z, yCount, zCount, world);
                        if (bestBelow <= 0) break;

                        long damage = 1000L * mass;
                        long likelyGross = 1000L * bestBelow;
                        if (likelyGross <= damage) break;
                        sendToPlant = true;
                    }

                    if (!TryBuildPath(x, y, mass, heights, yCount, true, out PathPlan path))
                        break;

                    long revenue = value > 0 ? 1000L * mass * value : -1000L * mass;
                    if (value > 0 && revenue <= path.FuelCost)
                        break;

                    WriteOperation(writer, path, x, y, z);
                    operations++;
                    world[Index(x, y, z, yCount, zCount)] = 0;
                    heights[col]--;
                    while (heights[col] > 0)
                    {
                        byte nb = world[Index(x, y, heights[col] - 1, yCount, zCount)];
                        if ((nb & 15) != 0) break;
                        heights[col]--;
                    }

                    if (!sendToPlant) break;
                }
            }
        }

        return operations;
    }

    private static long PlanLargeTopOnly(int xCount, int yCount, int[] heights, byte[] top, StreamWriter writer)
    {
        long operations = 0;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                int col = Col(x, y, yCount);
                byte b = top[col];
                int mass = b & 15;
                int value = b >> 4;
                if (heights[col] <= 0 || mass == 0 || value == 0) continue;

                int z = heights[col] - 1;
                if (!TryBuildPath(x, y, mass, heights, yCount, true, out PathPlan path))
                    continue;

                long revenue = 1000L * mass * value;
                if (revenue <= path.FuelCost) continue;

                WriteOperation(writer, path, x, y, z);
                operations++;
                heights[col]--;
            }
        }

        return operations;
    }

    private static int BestReachableValueBelow(int x, int y, int z, int yCount, int zCount, byte[] world)
    {
        int best = 0;
        int minZ = Math.Max(0, z - StripLookAhead);
        for (int zz = z - 1; zz >= minZ; zz--)
        {
            byte b = world[Index(x, y, zz, yCount, zCount)];
            int mass = b & 15;
            int value = b >> 4;
            if (mass == 0) continue;
            if (value > best) best = value * mass;
        }
        return best;
    }

    private static bool TryBuildPath(int sx, int sy, int mass, int[] heights, int yCount, bool removeSourceFirst, out PathPlan plan)
    {
        plan = default;
        if (sx == 0 && sy == 0)
        {
            plan = new PathPlan { Coords = Array.Empty<int>(), Count = 0, FuelCost = 0 };
            return true;
        }

        if (TryBuildOrderedPath(sx, sy, mass, heights, yCount, removeSourceFirst, true, out plan))
            return true;
        return TryBuildOrderedPath(sx, sy, mass, heights, yCount, removeSourceFirst, false, out plan);
    }

    private static bool TryBuildOrderedPath(int sx, int sy, int mass, int[] heights, int yCount, bool removeSourceFirst, bool xFirst, out PathPlan plan)
    {
        int maxPairs = Math.Max(0, sx + sy - 1);
        int[] coords = maxPairs == 0 ? Array.Empty<int>() : new int[maxPairs * 2];
        int count = 0;
        long fuel = 0;
        int x = sx;
        int y = sy;
        int prevH = EffectiveHeight(x, y, sx, sy, heights, yCount, removeSourceFirst);

        if (xFirst)
        {
            if (!WalkAxis(ref x, ref y, sx, sy, mass, heights, yCount, removeSourceFirst, true, coords, ref count, ref fuel, ref prevH)) { plan = default; return false; }
            if (!WalkAxis(ref x, ref y, sx, sy, mass, heights, yCount, removeSourceFirst, false, coords, ref count, ref fuel, ref prevH)) { plan = default; return false; }
        }
        else
        {
            if (!WalkAxis(ref x, ref y, sx, sy, mass, heights, yCount, removeSourceFirst, false, coords, ref count, ref fuel, ref prevH)) { plan = default; return false; }
            if (!WalkAxis(ref x, ref y, sx, sy, mass, heights, yCount, removeSourceFirst, true, coords, ref count, ref fuel, ref prevH)) { plan = default; return false; }
        }

        plan = new PathPlan { Coords = coords, Count = count, FuelCost = fuel };
        return x == 0 && y == 0;
    }

    private static bool WalkAxis(ref int x, ref int y, int sx, int sy, int mass, int[] heights, int yCount, bool removeSourceFirst, bool moveX, int[] coords, ref int count, ref long fuel, ref int prevH)
    {
        while ((moveX ? x : y) > 0)
        {
            int nx = x - (moveX ? 1 : 0);
            int ny = y - (moveX ? 0 : 1);
            int nextH = EffectiveHeight(nx, ny, sx, sy, heights, yCount, removeSourceFirst);
            int dz = nextH - prevH;
            if (dz > 1 || dz < -1) return false;

            if (dz > 0) fuel += 5L * mass;
            else if (dz == 0) fuel += mass;

            x = nx;
            y = ny;
            prevH = nextH;

            if (!(x == 0 && y == 0))
            {
                int p = count * 2;
                coords[p] = x;
                coords[p + 1] = y;
                count++;
            }
        }

        return true;
    }

    private static int EffectiveHeight(int x, int y, int sx, int sy, int[] heights, int yCount, bool removeSourceFirst)
    {
        int h = heights[Col(x, y, yCount)];
        if (removeSourceFirst && x == sx && y == sy) h--;
        return h;
    }

    private static void WriteOperation(StreamWriter writer, PathPlan path, int x, int y, int z)
    {
        writer.Write(path.Count.ToString(CultureInfo.InvariantCulture));
        writer.Write(' ');
        writer.Write(x.ToString(CultureInfo.InvariantCulture));
        writer.Write(' ');
        writer.Write(y.ToString(CultureInfo.InvariantCulture));
        writer.Write(' ');
        writer.Write(z.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < path.Count; i++)
        {
            writer.Write(' ');
            writer.Write(path.Coords[i * 2].ToString(CultureInfo.InvariantCulture));
            writer.Write(' ');
            writer.Write(path.Coords[i * 2 + 1].ToString(CultureInfo.InvariantCulture));
        }
        writer.Write(" 0 0");
        writer.WriteLine();
    }

    private static int Col(int x, int y, int yCount) => x * yCount + y;

    private static int Index(int x, int y, int z, int yCount, int zCount)
    {
        return ((x * yCount) + y) * zCount + z;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _pos;
        private int _len;

        public FastInput(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            string s = NextToken();
            if (s.Length == 0) return 0;
            return int.Parse(s, CultureInfo.InvariantCulture);
        }

        public string NextToken()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return string.Empty;
            } while (c <= 32);

            var sb = new StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c < 0) break;
            }
            return sb.ToString();
        }

        public int ReadDataByte()
        {
            return ReadByte();
        }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }
    }
}
