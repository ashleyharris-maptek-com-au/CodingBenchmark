declare const require: any;
declare const process: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid || [];
const robots: Robot[] = data.robots || [];
const jobs: Job[] = data.jobs || [];
const h = grid.length;
const w = h ? grid[0].length : 0;
const nCells = w * h;

function cellOf(x: number, y: number): number {
  return y * w + x;
}

function isOpenCell(c: number): boolean {
  if (c < 0 || c >= nCells || w <= 0) return false;
  const y = Math.floor(c / w);
  const x = c - y * w;
  return y >= 0 && y < h && x >= 0 && x < grid[y].length && grid[y].charCodeAt(x) !== 35;
}

function pointCell(p: { x: number; y: number }): number {
  if (!Number.isFinite(p.x) || !Number.isFinite(p.y)) return -1;
  const x = p.x | 0;
  const y = p.y | 0;
  if (x < 0 || y < 0 || y >= h || x >= w) return -1;
  return cellOf(x, y);
}

function bfsDistancesFrom(start: number, targets?: Int32Array): Int32Array {
  const dist = new Int32Array(nCells);
  dist.fill(-1);
  if (start < 0 || start >= nCells || !isOpenCell(start)) return dist;

  const q = new Int32Array(nCells);
  let head = 0;
  let tail = 0;
  q[tail++] = start;
  dist[start] = 0;

  let remaining = targets ? targets.length : -1;
  while (head < tail) {
    const c = q[head++];
    if (targets && remaining > 0) {
      for (let i = 0; i < targets.length; i++) {
        if (targets[i] === c) {
          remaining--;
          break;
        }
      }
      if (remaining === 0) break;
    }

    const d = dist[c] + 1;
    const x = c % w;
    let nb = c - w;
    if (c >= w && dist[nb] < 0 && isOpenCell(nb)) { dist[nb] = d; q[tail++] = nb; }
    nb = c + w;
    if (nb < nCells && dist[nb] < 0 && isOpenCell(nb)) { dist[nb] = d; q[tail++] = nb; }
    nb = c - 1;
    if (x > 0 && dist[nb] < 0 && isOpenCell(nb)) { dist[nb] = d; q[tail++] = nb; }
    nb = c + 1;
    if (x + 1 < w && dist[nb] < 0 && isOpenCell(nb)) { dist[nb] = d; q[tail++] = nb; }
  }
  return dist;
}

const robotCount = robots.length;
const jobCount = jobs.length;
const maxRobotId = robots.reduce((m, r) => Math.max(m, r.id | 0), -1);
const assignmentsByRobotIndex: number[][] = Array.from({ length: Math.max(maxRobotId + 1, robotCount) }, () => []);

if (robotCount === 0) {
  console.log(JSON.stringify({ assignments: assignmentsByRobotIndex }));
  process.exit(0);
}

const robotCells = Int32Array.from(robots.map(pointCell));
const jobCells = Int32Array.from(jobs.map(pointCell));

function manhattanCell(a: number, b: number): number {
  if (a < 0 || b < 0) return 1000000000;
  const ax = a % w, ay = Math.floor(a / w);
  const bx = b % w, by = Math.floor(b / w);
  return Math.abs(ax - bx) + Math.abs(ay - by);
}

function buildDistanceMatrix(cells: Int32Array): Int32Array {
  const p = cells.length;
  const matrix = new Int32Array(p * p);
  matrix.fill(-1);
  for (let i = 0; i < p; i++) {
    const dist = bfsDistancesFrom(cells[i]);
    const row = i * p;
    for (let j = 0; j < p; j++) matrix[row + j] = dist[cells[j]];
  }
  return matrix;
}

function solveWithMatrix(cells: Int32Array, matrix: Int32Array): number[][] {
  const p = cells.length;
  const routes: number[][] = Array.from({ length: robotCount }, () => []);
  const order = Array.from({ length: jobCount }, (_, i) => i);
  order.sort((a, b) => {
    const pr = (jobs[b].priority || 0) - (jobs[a].priority || 0);
    return pr || jobs[a].id - jobs[b].id;
  });

  function dPoint(a: number, b: number): number {
    if (a < 0 || b < 0) return 1000000000;
    const v = matrix[a * p + b];
    return v >= 0 ? v : manhattanCell(cells[a], cells[b]) + 1000000;
  }
  function rToJ(r: number, j: number): number { return dPoint(r, robotCount + j); }
  function jToJ(a: number, b: number): number { return dPoint(robotCount + a, robotCount + b); }

  function insertionDelta(route: number[], r: number, pos: number, job: number): number {
    if (route.length === 0) return rToJ(r, job);
    if (pos === 0) return rToJ(r, job) + jToJ(job, route[0]) - rToJ(r, route[0]);
    if (pos === route.length) return jToJ(route[pos - 1], job);
    return jToJ(route[pos - 1], job) + jToJ(job, route[pos]) - jToJ(route[pos - 1], route[pos]);
  }

  for (const job of order) {
    let bestR = 0, bestPos = 0, bestDelta = 2147483647;
    for (let r = 0; r < robotCount; r++) {
      const route = routes[r];
      for (let pos = 0; pos <= route.length; pos++) {
        const delta = insertionDelta(route, r, pos, job);
        if (delta < bestDelta || (delta === bestDelta && (r < bestR || (r === bestR && pos < bestPos)))) {
          bestDelta = delta; bestR = r; bestPos = pos;
        }
      }
    }
    routes[bestR].splice(bestPos, 0, job);
  }

  const maxImproveJobs = jobCount <= 1200 ? jobCount : 0;
  for (let pass = 0; pass < 2 && maxImproveJobs > 0; pass++) {
    let changed = false;
    for (let r0 = 0; r0 < robotCount; r0++) {
      for (let i = 0; i < routes[r0].length; i++) {
        const job = routes[r0][i];
        const oldRoute = routes[r0];
        let bestGain = 0, bestR = r0, bestPos = i;
        oldRoute.splice(i, 1);
        const before = insertionDelta(oldRoute, r0, Math.min(i, oldRoute.length), job);
        for (let r1 = 0; r1 < robotCount; r1++) {
          const route = routes[r1];
          for (let pos = 0; pos <= route.length; pos++) {
            if (r1 === r0 && (pos === i || pos === i + 1)) continue;
            const delta = insertionDelta(route, r1, pos, job);
            const gain = before - delta;
            if (gain > bestGain) { bestGain = gain; bestR = r1; bestPos = pos; }
          }
        }
        routes[bestR].splice(bestPos, 0, job);
        if (bestR !== r0 || bestPos !== i) changed = true;
        if (bestR === r0 && bestPos <= i) i++;
      }
    }
    if (!changed) break;
  }

  if (jobCount <= 1600) {
    for (let r = 0; r < robotCount; r++) {
      const route = routes[r];
      let improved = true;
      let guard = 0;
      while (improved && guard++ < 3) {
        improved = false;
        for (let i = 0; i + 2 < route.length; i++) {
          for (let k = i + 2; k < route.length; k++) {
            const aToB = i === 0 ? rToJ(r, route[i]) : jToJ(route[i - 1], route[i]);
            const cToD = k + 1 < route.length ? jToJ(route[k], route[k + 1]) : 0;
            const aToC = i === 0 ? rToJ(r, route[k]) : jToJ(route[i - 1], route[k]);
            const bToD = k + 1 < route.length ? jToJ(route[i], route[k + 1]) : 0;
            if (aToC + bToD < aToB + cToD) {
              for (let l = i, rr = k; l < rr; l++, rr--) {
                const tmp = route[l]; route[l] = route[rr]; route[rr] = tmp;
              }
              improved = true;
            }
          }
        }
      }
    }
  }
  return routes;
}

function multiSourceRobotOwners(): { owner: Int32Array; dist: Int32Array } {
  const owner = new Int32Array(nCells);
  const dist = new Int32Array(nCells);
  owner.fill(-1);
  dist.fill(-1);
  const q = new Int32Array(nCells);
  let head = 0, tail = 0;

  const starts = robots.map((r, i) => ({ i, c: robotCells[i] }))
    .filter(s => s.c >= 0 && isOpenCell(s.c))
    .sort((a, b) => a.i - b.i);
  for (const s of starts) {
    if (dist[s.c] < 0) {
      dist[s.c] = 0;
      owner[s.c] = s.i;
      q[tail++] = s.c;
    }
  }
  while (head < tail) {
    const c = q[head++];
    const d = dist[c] + 1;
    const own = owner[c];
    const x = c % w;
    const tryPush = (nb: number) => {
      if (dist[nb] < 0 && isOpenCell(nb)) {
        dist[nb] = d; owner[nb] = own; q[tail++] = nb;
      }
    };
    if (c >= w) tryPush(c - w);
    if (c + w < nCells) tryPush(c + w);
    if (x > 0) tryPush(c - 1);
    if (x + 1 < w) tryPush(c + 1);
  }
  return { owner, dist };
}

function solveLarge(): number[][] {
  const routes: number[][] = Array.from({ length: robotCount }, () => []);
  const { owner, dist } = multiSourceRobotOwners();
  for (let j = 0; j < jobCount; j++) {
    let r = jobCells[j] >= 0 ? owner[jobCells[j]] : -1;
    if (r < 0) {
      let best = 0, bd = 1000000000;
      for (let i = 0; i < robotCount; i++) {
        const d = manhattanCell(robotCells[i], jobCells[j]);
        if (d < bd) { bd = d; best = i; }
      }
      r = best;
    }
    routes[r].push(j);
  }

  for (let r = 0; r < robotCount; r++) {
    routes[r].sort((a, b) => {
      const da = jobCells[a] >= 0 ? dist[jobCells[a]] : 1000000000;
      const db = jobCells[b] >= 0 ? dist[jobCells[b]] : 1000000000;
      return da - db || (jobs[b].priority || 0) - (jobs[a].priority || 0) || jobs[a].id - jobs[b].id;
    });
    if (routes[r].length <= 450 && (routes[r].length + 1) * nCells <= 70000000) {
      const ids = routes[r].slice();
      const cells = Int32Array.from([robotCells[r], ...ids.map(j => jobCells[j])]);
      const mat = buildDistanceMatrix(cells);
      const unused: boolean[] = ids.map(() => true);
      let unusedCount = ids.length;
      const ordered: number[] = [];
      let cur = 0;
      while (unusedCount > 0) {
        let bi = -1, bd = 2147483647;
        for (let local = 0; local < ids.length; local++) {
          if (!unused[local]) continue;
          const d = mat[cur * cells.length + local + 1];
          const score = d >= 0 ? d : manhattanCell(cells[cur], cells[local + 1]) + 1000000;
          if (score < bd || (score === bd && jobs[ids[local]].id < jobs[ids[bi < 0 ? local : bi]].id)) {
            bd = score; bi = local;
          }
        }
        unused[bi] = false;
        unusedCount--;
        ordered.push(ids[bi]);
        cur = bi + 1;
      }
      routes[r] = ordered;
    }
  }
  return routes;
}

let routes: number[][];
const totalPoints = robotCount + jobCount;
if (totalPoints > 0 && totalPoints <= 1800 && totalPoints * Math.max(1, nCells) <= 120000000) {
  const cells = Int32Array.from([...Array.from(robotCells), ...Array.from(jobCells)]);
  routes = solveWithMatrix(cells, buildDistanceMatrix(cells));
} else {
  routes = solveLarge();
}

for (let i = 0; i < robotCount; i++) {
  const rid = robots[i].id | 0;
  while (assignmentsByRobotIndex.length <= rid) assignmentsByRobotIndex.push([]);
  assignmentsByRobotIndex[rid] = routes[i].map(j => jobs[j].id);
}

console.log(JSON.stringify({ assignments: assignmentsByRobotIndex }));
