#include <algorithm>

using namespace std;

struct BitSet {
    int n = 0;
    vector<uint64_t> w;

    BitSet() = default;
    explicit BitSet(int n_) : n(n_), w((n_ + 63) >> 6, 0) {}

    bool any() const {
        for (uint64_t x : w) if (x) return true;
        return false;
    }
    static int popcount64(uint64_t x) {
        int c = 0;
        while (x) {
            x &= x - 1;
            ++c;
        }
        return c;
    }
    static int ctz64(uint64_t x) {
        int c = 0;
        while ((x & 1ULL) == 0) {
            x >>= 1;
            ++c;
        }
        return c;
    }
    int count() const {
        int s = 0;
        for (uint64_t x : w) s += popcount64(x);
        return s;
    }
    void set(int i) { w[i >> 6] |= 1ULL << (i & 63); }
    void reset(int i) { w[i >> 6] &= ~(1ULL << (i & 63)); }
    bool test(int i) const { return (w[i >> 6] >> (i & 63)) & 1ULL; }

    void intersect_with(const BitSet& b) {
        for (size_t i = 0; i < w.size(); ++i) w[i] &= b.w[i];
    }
    BitSet intersected(const BitSet& b) const {
        BitSet r(*this);
        r.intersect_with(b);
        return r;
    }
    int first() const {
        for (int i = 0; i < (int)w.size(); ++i) {
            if (w[i]) return (i << 6) + ctz64(w[i]);
        }
        return -1;
    }
};

struct ComplementCliqueSolver {
    int n;
    const vector<BitSet>& original_adj;
    vector<BitSet> comp_adj;
    vector<int> cur, best;

    ComplementCliqueSolver(const vector<BitSet>& original_adj_) : n((int)original_adj_.size()), original_adj(original_adj_), comp_adj(n, BitSet(n)) {
        for (int i = 0; i < n; ++i) {
            for (int b = 0; b < (int)comp_adj[i].w.size(); ++b) comp_adj[i].w[b] = ~original_adj[i].w[b];
            comp_adj[i].reset(i);
            int extra = ((int)comp_adj[i].w.size() << 6) - n;
            if (extra) comp_adj[i].w.back() &= (~0ULL >> extra);
        }
    }

    void color_sort(const BitSet& P, vector<int>& order, vector<int>& color) {
        order.clear();
        color.clear();
        BitSet U = P;
        int c = 0;
        while (U.any()) {
            ++c;
            BitSet Q = U;
            while (Q.any()) {
                int v = Q.first();
                order.push_back(v);
                color.push_back(c);
                U.reset(v);
                Q.reset(v);
                Q.intersect_with(original_adj[v]);
                Q.intersect_with(U);
            }
        }
    }

    void expand(BitSet P) {
        if (!P.any()) {
            if (cur.size() > best.size()) best = cur;
            return;
        }

        vector<int> order, color;
        color_sort(P, order, color);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            if ((int)cur.size() + color[i] <= (int)best.size()) return;

            int v = order[i];
            if (!P.test(v)) continue;

            cur.push_back(v);
            BitSet next = P.intersected(comp_adj[v]);
            if (!next.any()) {
                if (cur.size() > best.size()) best = cur;
            } else if ((int)cur.size() + next.count() > (int)best.size()) {
                expand(next);
            }
            cur.pop_back();
            P.reset(v);
        }
    }

    vector<int> solve() {
        BitSet P(n);
        for (int i = 0; i < n; ++i) P.set(i);
        expand(P);
        return best;
    }
};

static bool is_bipartite_component(const vector<int>& verts,
                                   const vector<vector<int>>& adj,
                                   vector<int>& side) {
    side.assign(adj.size(), -1);
    queue<int> q;
    for (int s : verts) {
        if (side[s] != -1) continue;
        side[s] = 0;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (int u : adj[v]) {
                if (side[u] == -1) {
                    side[u] = side[v] ^ 1;
                    q.push(u);
                } else if (side[u] == side[v]) {
                    return false;
                }
            }
        }
    }
    return true;
}

static vector<int> bipartite_mis(const vector<int>& verts,
                                 const vector<vector<int>>& adj,
                                 const vector<int>& side) {
    vector<int> left, right;
    vector<int> lid(adj.size(), -1), rid(adj.size(), -1);
    for (int v : verts) {
        if (side[v] == 0) {
            lid[v] = (int)left.size();
            left.push_back(v);
        } else {
            rid[v] = (int)right.size();
            right.push_back(v);
        }
    }

    vector<vector<int>> g(left.size());
    for (int i = 0; i < (int)left.size(); ++i) {
        int v = left[i];
        for (int u : adj[v]) if (rid[u] != -1) g[i].push_back(rid[u]);
    }

    vector<int> pairU(left.size(), -1), pairV(right.size(), -1), dist(left.size());
    auto bfs = [&]() {
        queue<int> q;
        bool found = false;
        for (int i = 0; i < (int)left.size(); ++i) {
            if (pairU[i] == -1) {
                dist[i] = 0;
                q.push(i);
            } else {
                dist[i] = -1;
            }
        }
        while (!q.empty()) {
            int u = q.front();
            q.pop();
            for (int v : g[u]) {
                int pu = pairV[v];
                if (pu == -1) {
                    found = true;
                } else if (dist[pu] == -1) {
                    dist[pu] = dist[u] + 1;
                    q.push(pu);
                }
            }
        }
        return found;
    };
    auto dfs = [&](auto&& self, int u) -> bool {
        for (int v : g[u]) {
            int pu = pairV[v];
            if (pu == -1 || (dist[pu] == dist[u] + 1 && self(self, pu))) {
                pairU[u] = v;
                pairV[v] = u;
                return true;
            }
        }
        dist[u] = -1;
        return false;
    };
    while (bfs()) {
        for (int i = 0; i < (int)left.size(); ++i) {
            if (pairU[i] == -1) dfs(dfs, i);
        }
    }

    vector<char> visL(left.size(), 0), visR(right.size(), 0);
    queue<int> q;
    for (int i = 0; i < (int)left.size(); ++i) {
        if (pairU[i] == -1) {
            visL[i] = 1;
            q.push(i);
        }
    }
    while (!q.empty()) {
        int u = q.front();
        q.pop();
        for (int v : g[u]) {
            if (pairU[u] != v && !visR[v]) {
                visR[v] = 1;
                int pu = pairV[v];
                if (pu != -1 && !visL[pu]) {
                    visL[pu] = 1;
                    q.push(pu);
                }
            }
        }
    }

    vector<int> ans;
    for (int i = 0; i < (int)left.size(); ++i) if (visL[i]) ans.push_back(left[i]);
    for (int i = 0; i < (int)right.size(); ++i) if (!visR[i]) ans.push_back(right[i]);
    return ans;
}

static vector<int> path_mis(const vector<int>& order, int l, int r) {
    if (l > r) return {};
    int len = r - l + 1;
    vector<int> dp(len + 1, 0);
    for (int i = 1; i <= len; ++i) {
        int take = 1 + (i >= 2 ? dp[i - 2] : 0);
        dp[i] = max(dp[i - 1], take);
    }
    vector<int> ans;
    for (int i = len; i >= 1;) {
        int take = 1 + (i >= 2 ? dp[i - 2] : 0);
        if (take > dp[i - 1]) {
            ans.push_back(order[l + i - 1]);
            i -= 2;
        } else {
            --i;
        }
    }
    return ans;
}

static vector<int> degree_two_mis(const vector<int>& verts, const vector<vector<int>>& adj) {
    int start = verts[0];
    bool cycle = true;
    for (int v : verts) {
        if ((int)adj[v].size() <= 1) {
            start = v;
            cycle = false;
            break;
        }
    }

    vector<int> order;
    int prev = -1, cur = start;
    while (cur != -1) {
        order.push_back(cur);
        int nxt = -1;
        for (int u : adj[cur]) {
            if (u != prev) {
                if (!cycle || u != start || (int)order.size() == (int)verts.size()) {
                    nxt = u;
                    break;
                }
            }
        }
        prev = cur;
        cur = nxt;
        if (cycle && cur == start) break;
    }

    if (!cycle) return path_mis(order, 0, (int)order.size() - 1);

    vector<int> a = path_mis(order, 1, (int)order.size() - 1);
    vector<int> b;
    b.push_back(order[0]);
    vector<int> tail = path_mis(order, 2, (int)order.size() - 2);
    b.insert(b.end(), tail.begin(), tail.end());
    return b.size() > a.size() ? b : a;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<vector<int>> adj(n);
    vector<pair<int, int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v || u < 0 || v < 0 || u >= n || v >= n) continue;
        adj[u].push_back(v);
        adj[v].push_back(u);
        edges.push_back({u, v});
    }
    for (auto& a : adj) {
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
    }

    vector<char> seen(n, 0);
    vector<int> answer;
    vector<int> side;

    for (int s = 0; s < n; ++s) {
        if (seen[s]) continue;
        vector<int> verts;
        queue<int> q;
        seen[s] = 1;
        q.push(s);
        while (!q.empty()) {
            int v = q.front();
            q.pop();
            verts.push_back(v);
            for (int u : adj[v]) if (!seen[u]) {
                seen[u] = 1;
                q.push(u);
            }
        }

        if (verts.size() == 1) {
            answer.push_back(verts[0]);
            continue;
        }

        int max_deg = 0;
        long long degree_sum = 0;
        for (int v : verts) {
            max_deg = max(max_deg, (int)adj[v].size());
            degree_sum += adj[v].size();
        }
        long long internal_edges = degree_sum / 2;
        long long szll = (long long)verts.size();

        if (max_deg <= 2) {
            vector<int> part = degree_two_mis(verts, adj);
            answer.insert(answer.end(), part.begin(), part.end());
            continue;
        }

        if (is_bipartite_component(verts, adj, side)) {
            vector<int> part = bipartite_mis(verts, adj, side);
            answer.insert(answer.end(), part.begin(), part.end());
            continue;
        }

        int sz = (int)verts.size();
        if (internal_edges == szll * (szll - 1) / 2) {
            answer.push_back(verts[0]);
            continue;
        }

        vector<int> id(n, -1);
        for (int i = 0; i < sz; ++i) id[verts[i]] = i;
        vector<BitSet> local_adj(sz, BitSet(sz));
        for (int i = 0; i < sz; ++i) {
            int v = verts[i];
            for (int u : adj[v]) if (id[u] != -1) local_adj[i].set(id[u]);
        }

        ComplementCliqueSolver solver(local_adj);
        vector<int> local = solver.solve();
        for (int x : local) answer.push_back(verts[x]);
    }

    sort(answer.begin(), answer.end());
    cout << answer.size() << '\n';
    for (int i = 0; i < (int)answer.size(); ++i) {
        if (i) cout << ' ';
        cout << answer[i];
    }
    cout << '\n';
    return 0;
}
