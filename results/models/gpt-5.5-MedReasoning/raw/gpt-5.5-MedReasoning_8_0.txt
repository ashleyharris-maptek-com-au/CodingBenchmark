use std::collections::VecDeque;
use std::io::{self, BufRead, BufReader, BufWriter, Write};

#[inline]
fn get_bit(bits: &[u64], idx: usize) -> bool {
    ((bits[idx >> 6] >> (idx & 63)) & 1) != 0
}

#[inline]
fn set_bit(bits: &mut [u64], idx: usize) {
    bits[idx >> 6] |= 1u64 << (idx & 63);
}

#[inline]
fn get_parent(parent: &[u64], idx: usize) -> u8 {
    let shift = (idx & 31) * 2;
    ((parent[idx >> 5] >> shift) & 3) as u8
}

#[inline]
fn set_parent(parent: &mut [u64], idx: usize, dir_to_parent: u8) {
    let word = idx >> 5;
    let shift = (idx & 31) * 2;
    let mask = 3u64 << shift;
    parent[word] = (parent[word] & !mask) | ((dir_to_parent as u64) << shift);
}

#[inline]
fn try_push(
    next: usize,
    parent_dir: u8,
    walls: &[u64],
    visited: &mut [u64],
    parent: &mut [u64],
    queue: &mut VecDeque<u32>,
) {
    if !get_bit(walls, next) && !get_bit(visited, next) {
        set_bit(visited, next);
        set_parent(parent, next, parent_dir);
        queue.push_back(next as u32);
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = BufReader::new(stdin.lock());

    let mut first = String::new();
    if input.read_line(&mut first)? == 0 {
        return Ok(());
    }
    let mut dims = first.split_whitespace();
    let h: usize = match dims.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };
    let w: usize = match dims.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return Ok(()),
    };

    let cells = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return Ok(());
        }
    };
    if cells > u32::MAX as usize {
        println!("0");
        return Ok(());
    }

    let bit_words = (cells + 63) / 64;
    let parent_words = (cells + 31) / 32;
    let mut walls = vec![0u64; bit_words];
    let mut visited = vec![0u64; bit_words];
    let mut parent = vec![0u64; parent_words];
    let mut start: Option<usize> = None;
    let mut goal: Option<usize> = None;

    let mut row = Vec::with_capacity(w + 2);
    for y in 0..h {
        row.clear();
        input.read_until(b'\n', &mut row)?;
        if row.last() == Some(&b'\n') {
            row.pop();
        }
        if row.last() == Some(&b'\r') {
            row.pop();
        }

        for x in 0..w {
            let ch = row.get(x).copied().unwrap_or(b' ');
            let idx = y * w + x;
            match ch {
                b'#' => set_bit(&mut walls, idx),
                b'A' => start = Some(idx),
                b'B' => goal = Some(idx),
                _ => {}
            }
        }
    }

    let (start, goal) = match (start, goal) {
        (Some(a), Some(b)) => (a, b),
        _ => {
            println!("0");
            return Ok(());
        }
    };

    let mut queue = VecDeque::new();
    set_bit(&mut visited, start);
    queue.push_back(start as u32);

    while let Some(cur_u32) = queue.pop_front() {
        let cur = cur_u32 as usize;
        if cur == goal {
            break;
        }

        let x = cur % w;

        if cur >= w {
            try_push(cur - w, 1, &walls, &mut visited, &mut parent, &mut queue);
        }
        if cur + w < cells {
            try_push(cur + w, 0, &walls, &mut visited, &mut parent, &mut queue);
        }
        if x > 0 {
            try_push(cur - 1, 3, &walls, &mut visited, &mut parent, &mut queue);
        }
        if x + 1 < w {
            try_push(cur + 1, 2, &walls, &mut visited, &mut parent, &mut queue);
        }
    }

    if !get_bit(&visited, goal) {
        println!("0");
        return Ok(());
    }

    drop(queue);
    drop(walls);
    drop(visited);

    let mut path: Vec<u32> = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur as u32);
        if cur == start {
            break;
        }
        match get_parent(&parent, cur) {
            0 => cur -= w,
            1 => cur += w,
            2 => cur -= 1,
            3 => cur += 1,
            _ => unreachable!(),
        }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", path.len())?;
    for &idx_u32 in path.iter().rev() {
        let idx = idx_u32 as usize;
        let y = idx / w;
        let x = idx % w;
        writeln!(out, "{} {}", x, y)?;
    }

    Ok(())
}
