OpCapability Geometry
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0
%float_0_08 = OpConstant %float 0.08
%float_0_333333343 = OpConstant %float 0.333333343
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%gl_in = OpVariable %ptr_in_arr_pv Input
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %fn_void
%entry = OpLabel
%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1 = OpLoad %v4float %p1ptr
%p2 = OpLoad %v4float %p2ptr
%p01 = OpFAdd %v4float %p0 %p1
%psum = OpFAdd %v4float %p01 %p2
%center = OpVectorTimesScalar %v4float %psum %float_0_333333343
%p0xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%e1 = OpFSub %v3float %p1xyz %p0xyz
%e2 = OpFSub %v3float %p2xyz %p0xyz
%faceNormal = OpExtInst %v3float %GLSLstd450 Cross %e1 %e2
%normal = OpExtInst %v3float %GLSLstd450 Normalize %faceNormal
%offset3 = OpVectorTimesScalar %v3float %normal %float_0_08
%offx = OpCompositeExtract %float %offset3 0
%offy = OpCompositeExtract %float %offset3 1
%offz = OpCompositeExtract %float %offset3 2
%offset4 = OpCompositeConstruct %v4float %offx %offy %offz %float_0
%outPosPtr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%shifted0 = OpFAdd %v4float %p0 %offset4
%c0ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c0 = OpLoad %v3float %c0ptr
OpStore %outPosPtr0 %shifted0
OpStore %outColor %c0
OpEmitVertex
%outPosPtr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%shifted1 = OpFAdd %v4float %p1 %offset4
%c1ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c1 = OpLoad %v3float %c1ptr
OpStore %outPosPtr1 %shifted1
OpStore %outColor %c1
OpEmitVertex
%outPosPtr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
%shifted2 = OpFAdd %v4float %p2 %offset4
%c2ptr = OpAccessChain %ptr_in_v3 %inColor %int_2
%c2 = OpLoad %v3float %c2ptr
OpStore %outPosPtr2 %shifted2
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
