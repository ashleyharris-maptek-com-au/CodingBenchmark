#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main()
{
    uint i = gl_GlobalInvocationID.x;
    uint n = counts.x;

    if (i >= n) {
        return;
    }

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float sepRadius2 = sepRadius * sepRadius;
    float alignRadius2 = alignRadius * alignRadius;
    float cohRadius2 = cohRadius * cohRadius;

    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    for (uint j = 0u; j < n; ++j) {
        if (j == i) {
            continue;
        }

        uint otherBase = j * 2u;
        vec3 otherPos = inData[otherBase].xyz;
        vec3 otherVel = inData[otherBase + 1u].xyz;

        vec3 offset = pos - otherPos;
        float dist2 = dot(offset, offset);

        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            separation += offset / dist2;
            sepCount++;
        }

        if (dist2 < alignRadius2) {
            alignment += otherVel;
            alignCount++;
        }

        if (dist2 < cohRadius2) {
            cohesion += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        separation /= float(sepCount);
        accel += separation * sepWeight;
    }

    if (alignCount > 0u) {
        alignment = alignment / float(alignCount) - vel;
        accel += alignment * alignWeight;
    }

    if (cohCount > 0u) {
        vec3 center = cohesion / float(cohCount);
        cohesion = center - pos;
        accel += cohesion * cohWeight;
    }

    float boundSize = bounds.x;
    float boundForce = bounds.y;
    float threshold = boundSize * 0.4;
    float scale = boundSize * 0.5;

    vec3 boundary = vec3(0.0);

    if (pos.x > threshold) {
        boundary.x -= (pos.x - threshold) / scale;
    } else if (pos.x < -threshold) {
        boundary.x += (-threshold - pos.x) / scale;
    }

    if (pos.y > threshold) {
        boundary.y -= (pos.y - threshold) / scale;
    } else if (pos.y < -threshold) {
        boundary.y += (-threshold - pos.y) / scale;
    }

    if (pos.z > threshold) {
        boundary.z -= (pos.z - threshold) / scale;
    } else if (pos.z < -threshold) {
        boundary.z += (-threshold - pos.z) / scale;
    }

    accel += boundary * boundForce;

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = vel * (maxSpeed / speed);
    }

    pos += vel * dt;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
