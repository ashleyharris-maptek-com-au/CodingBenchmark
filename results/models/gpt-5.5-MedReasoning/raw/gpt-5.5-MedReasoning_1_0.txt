// Adaptive Travelling Salesman heuristic solver.
// MSVC C++20 compatible: uses only the C++ standard library.

#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct City {
    float x = 0.0f;
    float y = 0.0f;
    uint32_t id = 0;
    uint64_t key = 0;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline double dist_id(const std::vector<City>& c, uint32_t a, uint32_t b) {
    return std::sqrt(dist2(c[a], c[b]));
}

static uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static uint64_t morton2(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

static void rotate_to_zero(std::vector<uint32_t>& route) {
    auto it = std::find(route.begin(), route.end(), 0u);
    if (it != route.end()) {
        std::rotate(route.begin(), it, route.end());
    }
}

static std::vector<uint32_t> brute_force_route(const std::vector<City>& c) {
    const uint32_t n = static_cast<uint32_t>(c.size());
    std::vector<uint32_t> perm;
    for (uint32_t i = 1; i < n; ++i) perm.push_back(i);

    std::vector<uint32_t> best = perm;
    double best_len = std::numeric_limits<double>::infinity();

    do {
        double len = dist_id(c, 0, perm[0]);
        for (uint32_t i = 1; i < perm.size(); ++i) {
            len += dist_id(c, perm[i - 1], perm[i]);
        }
        len += dist_id(c, perm.back(), 0);
        if (len < best_len) {
            best_len = len;
            best = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<uint32_t> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), best.begin(), best.end());
    return route;
}

static std::vector<uint32_t> nearest_neighbor_route(const std::vector<City>& c) {
    const uint32_t n = static_cast<uint32_t>(c.size());
    std::vector<uint32_t> route;
    std::vector<uint32_t> remaining;
    route.reserve(n);
    remaining.reserve(n > 0 ? n - 1 : 0);

    route.push_back(0);
    for (uint32_t i = 1; i < n; ++i) remaining.push_back(i);

    uint32_t cur = 0;
    while (!remaining.empty()) {
        size_t best_pos = 0;
        double best_d = dist2(c[cur], c[remaining[0]]);
        for (size_t i = 1; i < remaining.size(); ++i) {
            const double d = dist2(c[cur], c[remaining[i]]);
            if (d < best_d) {
                best_d = d;
                best_pos = i;
            }
        }
        cur = remaining[best_pos];
        route.push_back(cur);
        remaining[best_pos] = remaining.back();
        remaining.pop_back();
    }
    return route;
}

static void two_opt_limited(std::vector<uint32_t>& route,
                            const std::vector<City>& c,
                            int passes,
                            uint32_t window) {
    const uint32_t n = static_cast<uint32_t>(route.size());
    if (n < 4) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool changed = false;
        for (uint32_t i = 1; i + 2 < n; ++i) {
            const uint32_t max_j = std::min<uint32_t>(n - 1, i + window);
            for (uint32_t j = i + 1; j < max_j; ++j) {
                const uint32_t a = route[i - 1];
                const uint32_t b = route[i];
                const uint32_t d = route[j + 1];
                const uint32_t e = route[j];
                const double before = dist_id(c, a, b) + dist_id(c, e, d);
                const double after = dist_id(c, a, e) + dist_id(c, b, d);
                if (after + 1e-12 < before) {
                    std::reverse(route.begin() + i, route.begin() + j + 1);
                    changed = true;
                }
            }
        }
        if (!changed) break;
    }
}

static std::vector<uint32_t> morton_route(std::vector<City>& c,
                                          double min_x,
                                          double max_x,
                                          double min_y,
                                          double max_y) {
    const double sx = (max_x > min_x) ? (4294967295.0 / (max_x - min_x)) : 0.0;
    const double sy = (max_y > min_y) ? (4294967295.0 / (max_y - min_y)) : 0.0;

    for (City& city : c) {
        double nx = (static_cast<double>(city.x) - min_x) * sx;
        double ny = (static_cast<double>(city.y) - min_y) * sy;
        if (nx < 0.0) nx = 0.0;
        if (ny < 0.0) ny = 0.0;
        if (nx > 4294967295.0) nx = 4294967295.0;
        if (ny > 4294967295.0) ny = 4294967295.0;
        city.key = morton2(static_cast<uint32_t>(nx), static_cast<uint32_t>(ny));
    }

    std::vector<uint32_t> route(c.size());
    std::iota(route.begin(), route.end(), 0u);
    std::sort(route.begin(), route.end(), [&c](uint32_t a, uint32_t b) {
        if (c[a].key != c[b].key) return c[a].key < c[b].key;
        return a < b;
    });

    rotate_to_zero(route);
    return route;
}

static void print_route(const std::vector<uint32_t>& route) {
    for (size_t i = 0; i < route.size(); ++i) {
        if (i) std::cout << ' ';
        std::cout << route[i];
    }
    std::cout << '\n';
}

static void print_identity_stream(uint64_t n) {
    for (uint64_t i = 0; i < n; ++i) {
        if (i) std::cout << ' ';
        std::cout << i;
    }
    std::cout << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint64_t n64 = 0;
    if (!(std::cin >> n64)) return 0;
    if (n64 == 0) {
        std::cout << '\n';
        return 0;
    }

    // Beyond this size, producing the output itself dominates the run time and
    // memory pressure is the real risk. Stream a valid route while consuming
    // input instead of storing coordinates.
    constexpr uint64_t STORE_LIMIT = 20000000ULL;
    if (n64 > STORE_LIMIT || n64 > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max())) {
        double x, y;
        for (uint64_t i = 0; i < n64; ++i) std::cin >> x >> y;
        print_identity_stream(n64);
        return 0;
    }

    const uint32_t n = static_cast<uint32_t>(n64);
    std::vector<City> cities;
    cities.reserve(n);

    double min_x = std::numeric_limits<double>::infinity();
    double min_y = std::numeric_limits<double>::infinity();
    double max_x = -std::numeric_limits<double>::infinity();
    double max_y = -std::numeric_limits<double>::infinity();

    for (uint32_t i = 0; i < n; ++i) {
        double x, y;
        std::cin >> x >> y;
        City city;
        city.x = static_cast<float>(x);
        city.y = static_cast<float>(y);
        city.id = i;
        cities.push_back(city);
        min_x = std::min(min_x, x);
        max_x = std::max(max_x, x);
        min_y = std::min(min_y, y);
        max_y = std::max(max_y, y);
    }

    std::vector<uint32_t> route;
    if (n <= 1) {
        route.push_back(0);
    } else if (n <= 10) {
        route = brute_force_route(cities);
    } else if (n <= 5000) {
        route = nearest_neighbor_route(cities);
        const int passes = (n <= 1000) ? 4 : 2;
        const uint32_t window = (n <= 1000) ? n - 2 : 96;
        two_opt_limited(route, cities, passes, window);
    } else {
        route = morton_route(cities, min_x, max_x, min_y, max_y);
        if (n <= 200000) {
            two_opt_limited(route, cities, 2, 64);
        }
    }

    rotate_to_zero(route);
    print_route(route);
    return 0;
}
