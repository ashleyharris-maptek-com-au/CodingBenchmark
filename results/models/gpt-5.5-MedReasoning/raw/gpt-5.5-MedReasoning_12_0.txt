using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;

public static class Program
{
    const double Eps = 1e-9;

    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Rot
    {
        public double W, X, Y, Z;
        public double M00, M01, M02, M10, M11, M12, M20, M21, M22;
    }

    struct Candidate
    {
        public Rot R;
        public double MinX, MinY, MinZ, SizeX, SizeY, SizeZ;
        public long Nx, Ny, Nz, Count;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int v = fs.NextInt();
        int f = fs.NextInt();
        if (v <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var vertices = new Vec[v];
        for (int i = 0; i < v; i++)
            vertices[i] = new Vec(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

        for (int i = 0; i < f; i++)
        {
            fs.NextInt();
            fs.NextInt();
            fs.NextInt();
        }

        double containerW = fs.NextDouble();
        double containerH = fs.NextDouble();
        double containerD = fs.NextDouble();

        if (containerW <= Eps || containerH <= Eps || containerD <= Eps)
        {
            Console.WriteLine(0);
            return;
        }

        var watch = Stopwatch.StartNew();
        List<Rot> rotations = BuildRotationSet(vertices, containerW, containerH, containerD);
        Candidate best = default;
        bool haveBest = false;

        foreach (var r in rotations)
        {
            Candidate c = Evaluate(vertices, r, containerW, containerH, containerD);
            if (c.Count <= 0) continue;

            if (!haveBest || c.Count > best.Count ||
                (c.Count == best.Count && c.SizeX * c.SizeY * c.SizeZ < best.SizeX * best.SizeY * best.SizeZ))
            {
                best = c;
                haveBest = true;
            }

            if (watch.ElapsedMilliseconds > 2500 && rotations.Count > 32)
                break;
        }

        if (!haveBest)
        {
            Console.WriteLine(0);
            return;
        }

        long maxLines = OutputBudget(best.Count);
        long emitCount = Math.Min(best.Count, maxLines);

        using var output = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 1 << 20);
        output.WriteLine(emitCount.ToString(CultureInfo.InvariantCulture));

        long emitted = 0;
        for (long iz = 0; iz < best.Nz && emitted < emitCount; iz++)
        {
            double tzBase = iz * best.SizeZ - best.MinZ;
            for (long iy = 0; iy < best.Ny && emitted < emitCount; iy++)
            {
                double tyBase = iy * best.SizeY - best.MinY;
                for (long ix = 0; ix < best.Nx && emitted < emitCount; ix++)
                {
                    double tx = ix * best.SizeX - best.MinX;
                    WriteDouble(output, tx); output.Write(' ');
                    WriteDouble(output, tyBase); output.Write(' ');
                    WriteDouble(output, tzBase); output.Write(' ');
                    WriteDouble(output, best.R.W); output.Write(' ');
                    WriteDouble(output, best.R.X); output.Write(' ');
                    WriteDouble(output, best.R.Y); output.Write(' ');
                    WriteDouble(output, best.R.Z);
                    output.WriteLine();
                    emitted++;
                }
            }
        }
    }

    static long OutputBudget(long possible)
    {
        if (possible <= 2_000_000) return possible;
        return 2_000_000;
    }

    static Candidate Evaluate(Vec[] vertices, Rot r, double cw, double ch, double cd)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

        for (int i = 0; i < vertices.Length; i++)
        {
            Vec p = vertices[i];
            double x = r.M00 * p.X + r.M01 * p.Y + r.M02 * p.Z;
            double y = r.M10 * p.X + r.M11 * p.Y + r.M12 * p.Z;
            double z = r.M20 * p.X + r.M21 * p.Y + r.M22 * p.Z;

            if (x < minX) minX = x; if (x > maxX) maxX = x;
            if (y < minY) minY = y; if (y > maxY) maxY = y;
            if (z < minZ) minZ = z; if (z > maxZ) maxZ = z;
        }

        double sx = maxX - minX, sy = maxY - minY, sz = maxZ - minZ;
        if (sx <= Eps || sy <= Eps || sz <= Eps)
            return default;

        long nx = FloorCount(cw, sx);
        long ny = FloorCount(ch, sy);
        long nz = FloorCount(cd, sz);
        long count = SafeProduct(nx, ny, nz);

        return new Candidate
        {
            R = r,
            MinX = minX,
            MinY = minY,
            MinZ = minZ,
            SizeX = sx,
            SizeY = sy,
            SizeZ = sz,
            Nx = nx,
            Ny = ny,
            Nz = nz,
            Count = count
        };
    }

    static long FloorCount(double container, double item)
    {
        if (item <= Eps) return 0;
        double raw = Math.Floor((container + 1e-7) / item);
        if (raw <= 0) return 0;
        if (raw > long.MaxValue / 4) return long.MaxValue / 4;
        return (long)raw;
    }

    static long SafeProduct(long a, long b, long c)
    {
        if (a <= 0 || b <= 0 || c <= 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        long ab = a * b;
        if (ab > long.MaxValue / c) return long.MaxValue;
        return ab * c;
    }

    static List<Rot> BuildRotationSet(Vec[] vertices, double cw, double ch, double cd)
    {
        var list = new List<Rot>(512);
        AddCubeOrientations(list);

        double scale = Math.Max(cw, Math.Max(ch, cd));
        int extra;
        if (vertices.Length <= 200 && scale <= 25) extra = 320;
        else if (vertices.Length <= 1000 && scale <= 100) extra = 96;
        else if (vertices.Length <= 5000 && scale <= 500) extra = 24;
        else extra = 0;

        var rng = new Random(1234567);
        for (int i = 0; i < extra; i++)
            list.Add(RandomRotation(rng));

        return list;
    }

    static void AddCubeOrientations(List<Rot> list)
    {
        int[][] perms =
        {
            new[] {0,1,2}, new[] {0,2,1}, new[] {1,0,2},
            new[] {1,2,0}, new[] {2,0,1}, new[] {2,1,0}
        };

        foreach (var p in perms)
        {
            int parity = PermutationParity(p);
            for (int sx = -1; sx <= 1; sx += 2)
            for (int sy = -1; sy <= 1; sy += 2)
            for (int sz = -1; sz <= 1; sz += 2)
            {
                if (parity * sx * sy * sz != 1) continue;

                double[,] m = new double[3, 3];
                m[0, p[0]] = sx;
                m[1, p[1]] = sy;
                m[2, p[2]] = sz;
                list.Add(MatrixRotation(
                    m[0,0], m[0,1], m[0,2],
                    m[1,0], m[1,1], m[1,2],
                    m[2,0], m[2,1], m[2,2]));
            }
        }
    }

    static int PermutationParity(int[] p)
    {
        int inv = 0;
        for (int i = 0; i < 3; i++)
            for (int j = i + 1; j < 3; j++)
                if (p[i] > p[j]) inv++;
        return (inv & 1) == 0 ? 1 : -1;
    }

    static Rot RandomRotation(Random rng)
    {
        double u1 = rng.NextDouble();
        double u2 = rng.NextDouble();
        double u3 = rng.NextDouble();
        double s1 = Math.Sqrt(1.0 - u1);
        double s2 = Math.Sqrt(u1);
        double a = 2.0 * Math.PI * u2;
        double b = 2.0 * Math.PI * u3;

        double qx = s1 * Math.Sin(a);
        double qy = s1 * Math.Cos(a);
        double qz = s2 * Math.Sin(b);
        double qw = s2 * Math.Cos(b);
        return QuaternionRotation(qw, qx, qy, qz);
    }

    static Rot QuaternionRotation(double qw, double qx, double qy, double qz)
    {
        double n = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        qw /= n; qx /= n; qy /= n; qz /= n;

        double xx = qx * qx, yy = qy * qy, zz = qz * qz;
        double xy = qx * qy, xz = qx * qz, yz = qy * qz;
        double wx = qw * qx, wy = qw * qy, wz = qw * qz;

        return new Rot
        {
            W = qw, X = qx, Y = qy, Z = qz,
            M00 = 1 - 2 * (yy + zz),
            M01 = 2 * (xy - wz),
            M02 = 2 * (xz + wy),
            M10 = 2 * (xy + wz),
            M11 = 1 - 2 * (xx + zz),
            M12 = 2 * (yz - wx),
            M20 = 2 * (xz - wy),
            M21 = 2 * (yz + wx),
            M22 = 1 - 2 * (xx + yy)
        };
    }

    static Rot MatrixRotation(
        double m00, double m01, double m02,
        double m10, double m11, double m12,
        double m20, double m21, double m22)
    {
        double qw, qx, qy, qz;
        double trace = m00 + m11 + m22;
        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            qw = 0.25 * s;
            qx = (m21 - m12) / s;
            qy = (m02 - m20) / s;
            qz = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0;
            qw = (m21 - m12) / s;
            qx = 0.25 * s;
            qy = (m01 + m10) / s;
            qz = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0;
            qw = (m02 - m20) / s;
            qx = (m01 + m10) / s;
            qy = 0.25 * s;
            qz = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0;
            qw = (m10 - m01) / s;
            qx = (m02 + m20) / s;
            qy = (m12 + m21) / s;
            qz = 0.25 * s;
        }

        Rot r = QuaternionRotation(qw, qx, qy, qz);
        r.M00 = m00; r.M01 = m01; r.M02 = m02;
        r.M10 = m10; r.M11 = m11; r.M12 = m12;
        r.M20 = m20; r.M21 = m21; r.M22 = m22;
        return r;
    }

    static void WriteDouble(StreamWriter output, double value)
    {
        output.Write(value.ToString("G17", CultureInfo.InvariantCulture));
    }

    sealed class FastScanner
    {
        readonly Stream _stream;
        readonly byte[] _buffer = new byte[1 << 16];
        int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            return (int)NextDouble();
        }

        public double NextDouble()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            double value = 0;
            while (c >= '0' && c <= '9')
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double place = 0.1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    value += (c - '0') * place;
                    place *= 0.1;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int expSign = 1;
                int exp = 0;
                c = Read();
                if (c == '-')
                {
                    expSign = -1;
                    c = Read();
                }
                else if (c == '+')
                {
                    c = Read();
                }
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                value *= Math.Pow(10, expSign * exp);
            }

            return sign * value;
        }
    }
}
