const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];
const H = grid.length;
const W = H ? grid[0].length : 0;
const N = H * W;
const INF = 1 << 28;

function cellOf(x, y) {
  return y * W + x;
}

const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) open[cellOf(x, y)] = row[x] !== '#' ? 1 : 0;
}

const robotCells = robots.map(r => cellOf(r.x, r.y));
const jobCells = jobs.map(j => cellOf(j.x, j.y));
const maxRobotId = robots.reduce((m, r) => Math.max(m, Number(r.id) || 0), -1);
const outLen = Math.max(maxRobotId + 1, robots.length);

const q = new Int32Array(Math.max(1, N));

function bfs(start) {
  const dist = new Int32Array(N);
  dist.fill(-1);
  if (start < 0 || start >= N || !open[start]) return dist;
  let head = 0, tail = 0;
  q[tail++] = start;
  dist[start] = 0;
  while (head < tail) {
    const c = q[head++];
    const d = dist[c] + 1;
    const x = c % W;
    let n = c - W;
    if (c >= W && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
    n = c + W;
    if (n < N && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
    n = c - 1;
    if (x > 0 && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
    n = c + 1;
    if (x + 1 < W && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
  }
  return dist;
}

function multiSourceOwners() {
  const dist = new Int32Array(N);
  const owner = new Int32Array(N);
  dist.fill(-1);
  owner.fill(-1);
  let head = 0, tail = 0;
  for (let r = 0; r < robotCells.length; r++) {
    const c = robotCells[r];
    if (c >= 0 && c < N && open[c] && dist[c] < 0) {
      dist[c] = 0;
      owner[c] = r;
      q[tail++] = c;
    }
  }
  while (head < tail) {
    const c = q[head++];
    const d = dist[c] + 1;
    const own = owner[c];
    const x = c % W;
    let n = c - W;
    if (c >= W && open[n] && dist[n] < 0) { dist[n] = d; owner[n] = own; q[tail++] = n; }
    n = c + W;
    if (n < N && open[n] && dist[n] < 0) { dist[n] = d; owner[n] = own; q[tail++] = n; }
    n = c - 1;
    if (x > 0 && open[n] && dist[n] < 0) { dist[n] = d; owner[n] = own; q[tail++] = n; }
    n = c + 1;
    if (x + 1 < W && open[n] && dist[n] < 0) { dist[n] = d; owner[n] = own; q[tail++] = n; }
  }
  return owner;
}

function makeNearestCandidate() {
  const clusters = Array.from({ length: robots.length }, () => []);
  if (!robots.length) return clusters;
  const owner = multiSourceOwners();
  for (let j = 0; j < jobs.length; j++) {
    let r = owner[jobCells[j]];
    if (r < 0) r = j % robots.length;
    clusters[r].push(j);
  }
  return clusters;
}

function makeBalancedCandidate() {
  const R = robots.length, J = jobs.length;
  if (!R) return [];
  if (R * Math.max(1, N) > 25000000) return null;
  const dists = Array.from({ length: R }, () => new Int32Array(J));
  let sum = 0, cnt = 0;
  for (let r = 0; r < R; r++) {
    const d = bfs(robotCells[r]);
    for (let j = 0; j < J; j++) {
      const v = d[jobCells[j]];
      dists[r][j] = v;
      if (v >= 0) { sum += v; cnt++; }
    }
  }
  const avg = cnt ? sum / cnt : 1;
  const target = Math.max(1, J / R);
  const counts = new Int32Array(R);
  const clusters = Array.from({ length: R }, () => []);
  const order = Array.from({ length: J }, (_, i) => i);
  order.sort((a, b) => {
    const pa = jobs[a].priority ?? 0, pb = jobs[b].priority ?? 0;
    if (pb !== pa) return pb - pa;
    return String(jobs[a].id).localeCompare(String(jobs[b].id), undefined, { numeric: true });
  });
  for (const j of order) {
    let bestR = 0, best = Infinity;
    for (let r = 0; r < R; r++) {
      const d = dists[r][j];
      if (d < 0) continue;
      const load = counts[r] / target;
      const score = d + avg * 0.55 * load * load;
      if (score < best - 1e-9 || (score === best && r < bestR)) {
        best = score;
        bestR = r;
      }
    }
    clusters[bestR].push(j);
    counts[bestR]++;
  }
  return clusters;
}

function routeCluster(cluster, startCell) {
  if (cluster.length <= 1) {
    const len = cluster.length ? Math.max(0, bfs(startCell)[jobCells[cluster[0]]]) : 0;
    return { route: cluster.slice(), length: len };
  }

  const byCell = new Map();
  for (const j of cluster) {
    const c = jobCells[j];
    let a = byCell.get(c);
    if (!a) byCell.set(c, a = []);
    a.push(j);
  }
  for (const a of byCell.values()) {
    a.sort((u, v) => {
      const pu = jobs[u].priority ?? 0, pv = jobs[v].priority ?? 0;
      if (pv !== pu) return pv - pu;
      return String(jobs[u].id).localeCompare(String(jobs[v].id), undefined, { numeric: true });
    });
  }

  const target = new Uint8Array(N);
  for (const c of byCell.keys()) target[c] = 1;
  let remaining = byCell.size;
  let cur = startCell;
  let length = 0;
  const route = [];

  while (remaining > 0) {
    const dist = new Int32Array(N);
    dist.fill(-1);
    let head = 0, tail = 0, found = -1;
    if (cur >= 0 && cur < N && open[cur]) {
      q[tail++] = cur;
      dist[cur] = 0;
    }
    while (head < tail) {
      const c = q[head++];
      if (target[c]) { found = c; break; }
      const d = dist[c] + 1;
      const x = c % W;
      let n = c - W;
      if (c >= W && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
      n = c + W;
      if (n < N && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
      n = c - 1;
      if (x > 0 && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
      n = c + 1;
      if (x + 1 < W && open[n] && dist[n] < 0) { dist[n] = d; q[tail++] = n; }
    }
    if (found < 0) {
      const rest = [];
      for (const [c, a] of byCell) if (target[c]) rest.push(...a);
      rest.sort((u, v) => String(jobs[u].id).localeCompare(String(jobs[v].id), undefined, { numeric: true }));
      route.push(...rest);
      break;
    }
    length += dist[found];
    route.push(...byCell.get(found));
    target[found] = 0;
    remaining--;
    cur = found;
  }

  if (route.length <= 180) {
    const improved = twoOpt(route, startCell);
    if (improved.length <= length) return improved;
  }
  return { route, length };
}

function twoOpt(route, startCell) {
  const m = route.length;
  const nodes = [startCell, ...route.map(j => jobCells[j])];
  const mat = Array.from({ length: m + 1 }, () => new Int32Array(m + 1));
  for (let i = 0; i <= m; i++) {
    const d = bfs(nodes[i]);
    for (let k = 0; k <= m; k++) mat[i][k] = d[nodes[k]] < 0 ? INF : d[nodes[k]];
  }
  const pos = new Map();
  for (let i = 0; i < m; i++) pos.set(route[i], i + 1);
  const edgeDist = (a, b) => mat[pos.get(route[a])][pos.get(route[b])];
  const startDist = a => mat[0][pos.get(route[a])];
  let length = startDist(0);
  for (let i = 1; i < m; i++) length += edgeDist(i - 1, i);

  let changed = true, passes = 0;
  while (changed && passes++ < 6) {
    changed = false;
    for (let i = 0; i < m - 1; i++) {
      for (let k = i + 1; k < m; k++) {
        const beforeLeft = i === 0 ? startDist(i) : edgeDist(i - 1, i);
        const afterLeft = i === 0 ? startDist(k) : edgeDist(i - 1, k);
        let delta = afterLeft - beforeLeft;
        if (k + 1 < m) delta += edgeDist(i, k + 1) - edgeDist(k, k + 1);
        if (delta < 0) {
          let a = i, b = k;
          while (a < b) {
            const t = route[a]; route[a] = route[b]; route[b] = t;
            a++; b--;
          }
          length += delta;
          changed = true;
        }
      }
    }
  }
  return { route, length };
}

function solveCandidate(clusters) {
  const routes = Array.from({ length: robots.length }, () => []);
  let total = 0;
  for (let r = 0; r < robots.length; r++) {
    const rr = routeCluster(clusters[r] || [], robotCells[r]);
    routes[r] = rr.route;
    total += rr.length;
  }
  return { routes, total };
}

let best = solveCandidate(makeNearestCandidate());
const balanced = makeBalancedCandidate();
if (balanced) {
  const b = solveCandidate(balanced);
  if (b.total < best.total) best = b;
}

const assignments = Array.from({ length: outLen }, () => []);
for (let r = 0; r < robots.length; r++) {
  const rid = Number(robots[r].id);
  const outIndex = Number.isInteger(rid) && rid >= 0 ? rid : r;
  if (outIndex >= assignments.length) {
    while (assignments.length <= outIndex) assignments.push([]);
  }
  assignments[outIndex] = best.routes[r].map(j => jobs[j].id);
}
