; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %outColor
OpExecutionMode %main OriginUpperLeft
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%voidfn = OpTypeFunction %void
%bool = OpTypeBool
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int = OpTypeInt 32 1
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_0_7 = OpConstant %float 0.7
%float_1 = OpConstant %float 1
%float_2 = OpConstant %float 2
%float_8 = OpConstant %float 8
%float_16 = OpConstant %float 16
%uvScale = OpConstantComposite %v2float %float_8 %float_16
%mortarColor = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_7
%brickColor = OpConstantComposite %v3float %float_0_7 %float_0_2 %float_0_1
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None %voidfn
%entry = OpLabel
%uvVal = OpLoad %v2float %uv
%scaledUV = OpFMul %v2float %uvVal %uvScale
%scaledX = OpCompositeExtract %float %scaledUV 0
%scaledY = OpCompositeExtract %float %scaledUV 1
%row = OpExtInst %float %1 Floor %scaledY
%rowMod = OpFMod %float %row %float_2
%rowOffset = OpFMul %float %rowMod %float_0_5
%staggeredX = OpFAdd %float %scaledX %rowOffset
%brickX = OpExtInst %float %1 Fract %staggeredX
%brickY = OpExtInst %float %1 Fract %scaledY
%mortarX = OpFOrdLessThan %bool %brickX %float_0_05
%mortarY = OpFOrdLessThan %bool %brickY %float_0_1
%isMortar = OpLogicalOr %bool %mortarX %mortarY
%baseColor = OpSelect %v3float %isMortar %mortarColor %brickColor
%worldVal = OpLoad %v3float %worldPos
%normalVal = OpLoad %v3float %normal
%unitNormal = OpExtInst %v3float %1 Normalize %normalVal
%lightPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%lightVal4 = OpLoad %v4float %lightPtr
%lightVal = OpVectorShuffle %v3float %lightVal4 %lightVal4 0 1 2
%lightVec = OpFSub %v3float %lightVal %worldVal
%lightDir = OpExtInst %v3float %1 Normalize %lightVec
%ndotl = OpDot %float %unitNormal %lightDir
%diffuse = OpExtInst %float %1 FMax %ndotl %float_0
%litPart = OpFMul %float %diffuse %float_0_6
%shade = OpFAdd %float %float_0_4 %litPart
%rgb = OpVectorTimesScalar %v3float %baseColor %shade
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%finalColor = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
