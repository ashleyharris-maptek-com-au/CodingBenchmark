use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Eq, PartialEq, Hash)]
struct Cell {
    x: i64,
    y: i64,
    z: i64,
}

#[derive(Clone, Debug)]
struct Sample {
    hole: usize,
    idx: usize,
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Clone, Debug)]
struct RobustStats {
    median: f64,
    mad_sigma: f64,
    iqr_sigma: f64,
    mean: f64,
    std: f64,
}

#[derive(Clone, Debug)]
struct Finding {
    confidence: f64,
    hole_id: String,
    sample_idx: usize,
    property: String,
}

fn parse_f64(s: Option<&str>) -> f64 {
    match s.and_then(|v| v.parse::<f64>().ok()) {
        Some(v) if v.is_finite() => v,
        _ => f64::NAN,
    }
}

fn finite_cmp(a: &f64, b: &f64) -> Ordering {
    a.partial_cmp(b).unwrap_or(Ordering::Equal)
}

fn median_sorted(v: &[f64]) -> f64 {
    if v.is_empty() {
        return f64::NAN;
    }
    let n = v.len();
    if n & 1 == 1 {
        v[n / 2]
    } else {
        (v[n / 2 - 1] + v[n / 2]) * 0.5
    }
}

fn percentile_sorted(v: &[f64], p: f64) -> f64 {
    if v.is_empty() {
        return f64::NAN;
    }
    if v.len() == 1 {
        return v[0];
    }
    let pos = p.clamp(0.0, 1.0) * ((v.len() - 1) as f64);
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        v[lo]
    } else {
        v[lo] * (hi as f64 - pos) + v[hi] * (pos - lo as f64)
    }
}

fn robust_stats(values: &[f64], prop: usize, props: usize, max_take: usize) -> RobustStats {
    let mut vals = Vec::new();
    let total = values.len() / props;
    if total == 0 {
        return RobustStats {
            median: 0.0,
            mad_sigma: 1.0,
            iqr_sigma: 1.0,
            mean: 0.0,
            std: 1.0,
        };
    }
    let stride = (total / max_take.max(1)).max(1);
    let mut mean = 0.0;
    let mut m2 = 0.0;
    let mut cnt = 0usize;

    for i in 0..total {
        let x = values[i * props + prop];
        if !x.is_finite() {
            continue;
        }
        cnt += 1;
        let d = x - mean;
        mean += d / cnt as f64;
        m2 += d * (x - mean);
        if i % stride == 0 {
            vals.push(x);
        }
    }

    if vals.is_empty() {
        return RobustStats {
            median: 0.0,
            mad_sigma: 1.0,
            iqr_sigma: 1.0,
            mean: 0.0,
            std: 1.0,
        };
    }

    vals.sort_by(finite_cmp);
    let med = median_sorted(&vals);
    let q1 = percentile_sorted(&vals, 0.25);
    let q3 = percentile_sorted(&vals, 0.75);

    let mut dev: Vec<f64> = vals.iter().map(|x| (x - med).abs()).collect();
    dev.sort_by(finite_cmp);
    let mad = median_sorted(&dev) * 1.4826;
    let iqr = (q3 - q1).abs() / 1.349;
    let std = if cnt > 1 {
        (m2 / (cnt as f64 - 1.0)).sqrt()
    } else {
        1.0
    };

    RobustStats {
        median: med,
        mad_sigma: mad.max(1e-12),
        iqr_sigma: iqr.max(1e-12),
        mean,
        std: std.max(1e-12),
    }
}

fn cell_of(s: &Sample, minx: f64, miny: f64, minz: f64, cell: f64) -> Cell {
    Cell {
        x: ((s.x - minx) / cell).floor() as i64,
        y: ((s.y - miny) / cell).floor() as i64,
        z: ((s.z - minz) / cell).floor() as i64,
    }
}

fn dist2(a: &Sample, b: &Sample) -> f64 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

fn local_score(vals: &mut Vec<f64>, x: f64) -> Option<f64> {
    vals.retain(|v| v.is_finite());
    if vals.len() < 4 || !x.is_finite() {
        return None;
    }
    vals.sort_by(finite_cmp);
    let med = median_sorted(vals);
    let q1 = percentile_sorted(vals, 0.25);
    let q3 = percentile_sorted(vals, 0.75);

    let mut dev: Vec<f64> = vals.iter().map(|v| (v - med).abs()).collect();
    dev.sort_by(finite_cmp);
    let mad_sigma = (median_sorted(&dev) * 1.4826).max(1e-12);
    let iqr_sigma = ((q3 - q1).abs() / 1.349).max(1e-12);
    let scale = mad_sigma.max(iqr_sigma * 0.5);
    Some((x - med).abs() / scale)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let num_holes = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    let num_props = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);
    if num_holes == 0 || num_props == 0 {
        println!("0");
        return;
    }

    let mut prop_names = Vec::with_capacity(num_props);
    for p in 0..num_props {
        prop_names.push(it.next().map(str::to_string).unwrap_or_else(|| format!("prop{}", p + 1)));
    }

    let mut hole_ids = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    let mut values: Vec<f64> = Vec::new();

    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut minz = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    let mut maxz = f64::NEG_INFINITY;

    for h in 0..num_holes {
        let hole_id = match it.next() {
            Some(s) => s.to_string(),
            None => break,
        };
        let sx = parse_f64(it.next());
        let sy = parse_f64(it.next());
        let sz = parse_f64(it.next());
        let mut dx = parse_f64(it.next());
        let mut dy = parse_f64(it.next());
        let mut dz = parse_f64(it.next());
        let _length = parse_f64(it.next());
        let n_samples = it.next().and_then(|s| s.parse::<usize>().ok()).unwrap_or(0);

        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        if !norm.is_finite() || norm <= 0.0 {
            dx = 0.0;
            dy = 0.0;
            dz = 1.0;
        } else {
            dx /= norm;
            dy /= norm;
            dz /= norm;
        }

        let hole_index = hole_ids.len();
        hole_ids.push(hole_id);
        for si in 0..n_samples {
            let depth = parse_f64(it.next());
            let x = sx + dx * depth;
            let y = sy + dy * depth;
            let z = sz + dz * depth;
            if x.is_finite() && y.is_finite() && z.is_finite() {
                minx = minx.min(x);
                miny = miny.min(y);
                minz = minz.min(z);
                maxx = maxx.max(x);
                maxy = maxy.max(y);
                maxz = maxz.max(z);
            }
            samples.push(Sample {
                hole: hole_index,
                idx: si,
                x,
                y,
                z,
            });
            for _ in 0..num_props {
                values.push(parse_f64(it.next()));
            }
        }

        if h + 1 >= num_holes {
            break;
        }
    }

    let n = samples.len();
    if n == 0 {
        println!("0");
        return;
    }

    if !minx.is_finite() {
        minx = 0.0;
        miny = 0.0;
        minz = 0.0;
        maxx = 1.0;
        maxy = 1.0;
        maxz = 1.0;
    }

    let stat_sample = if n <= 20_000 {
        n
    } else if n <= 1_000_000 {
        120_000
    } else {
        250_000
    };
    let stats: Vec<RobustStats> = (0..num_props)
        .map(|p| robust_stats(&values, p, num_props, stat_sample))
        .collect();

    let ex = (maxx - minx).abs();
    let ey = (maxy - miny).abs();
    let ez = (maxz - minz).abs();
    let max_extent = ex.max(ey).max(ez).max(1.0);
    let volume = (ex.max(1e-9)) * (ey.max(1e-9)) * (ez.max(1e-9));
    let target_per_cell = if n < 50_000 { 16.0 } else if n < 1_000_000 { 32.0 } else { 64.0 };
    let mut cell_size = (volume * target_per_cell / n as f64).cbrt();
    if !cell_size.is_finite() || cell_size <= 0.0 || cell_size < max_extent * 1e-9 {
        cell_size = max_extent / ((n as f64 / target_per_cell).cbrt().max(1.0));
    }
    cell_size = cell_size.max(1e-9);

    let mut grid: HashMap<Cell, Vec<usize>> = HashMap::with_capacity(n / 4 + 1);
    for (i, s) in samples.iter().enumerate() {
        if s.x.is_finite() && s.y.is_finite() && s.z.is_finite() {
            grid.entry(cell_of(s, minx, miny, minz, cell_size)).or_default().push(i);
        }
    }

    let max_neighbors = if n < 50_000 { 96 } else { 64 };
    let threshold = if n < 200 { 4.5 } else if n < 20_000 { 5.0 } else { 6.0 };
    let max_report = if n < 10_000 { 2_000 } else { 20_000 };
    let mut findings: Vec<Finding> = Vec::new();
    let mut neighbor_pairs: Vec<(f64, usize)> = Vec::new();
    let mut local_vals: Vec<f64> = Vec::new();

    for i in 0..n {
        let s = &samples[i];
        if !s.x.is_finite() || !s.y.is_finite() || !s.z.is_finite() {
            continue;
        }
        neighbor_pairs.clear();
        let c = cell_of(s, minx, miny, minz, cell_size);

        let mut radius = 1i64;
        while radius <= 3 && neighbor_pairs.len() < 8 {
            neighbor_pairs.clear();
            for ix in c.x - radius..=c.x + radius {
                for iy in c.y - radius..=c.y + radius {
                    for iz in c.z - radius..=c.z + radius {
                        if let Some(bucket) = grid.get(&Cell { x: ix, y: iy, z: iz }) {
                            for &j in bucket {
                                if j == i {
                                    continue;
                                }
                                let d2 = dist2(s, &samples[j]);
                                if d2.is_finite() {
                                    neighbor_pairs.push((d2, j));
                                }
                            }
                        }
                    }
                }
            }
            radius += 1;
        }

        if neighbor_pairs.len() > max_neighbors {
            neighbor_pairs.select_nth_unstable_by(max_neighbors, |a, b| {
                a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal)
            });
            neighbor_pairs.truncate(max_neighbors);
        }

        for p in 0..num_props {
            let x = values[i * num_props + p];
            if !x.is_finite() {
                continue;
            }
            let st = &stats[p];
            let global_robust = ((x - st.median).abs() / st.mad_sigma.max(st.iqr_sigma * 0.5)).max(0.0);
            let global_std = ((x - st.mean).abs() / st.std).max(0.0);

            local_vals.clear();
            for &(_, j) in &neighbor_pairs {
                let v = values[j * num_props + p];
                if v.is_finite() {
                    local_vals.push(v);
                }
            }

            let local = local_score(&mut local_vals, x).unwrap_or(global_robust * 0.7);
            let confidence = local.max(global_robust * 0.85).max(global_std * 0.5);

            let is_spatial_bad = local >= threshold && global_robust >= 2.5;
            let is_extreme_global = global_robust >= threshold + 3.0 && global_std >= 4.0;
            if is_spatial_bad || is_extreme_global {
                findings.push(Finding {
                    confidence,
                    hole_id: hole_ids.get(s.hole).cloned().unwrap_or_else(|| s.hole.to_string()),
                    sample_idx: s.idx,
                    property: prop_names[p].clone(),
                });
            }
        }
    }

    findings.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole_id.cmp(&b.hole_id))
            .then_with(|| a.sample_idx.cmp(&b.sample_idx))
            .then_with(|| a.property.cmp(&b.property))
    });
    findings.dedup_by(|a, b| a.hole_id == b.hole_id && a.sample_idx == b.sample_idx && a.property == b.property);
    if findings.len() > max_report {
        findings.truncate(max_report);
    }

    println!("{}", findings.len());
    for f in findings {
        println!("{} {} {} {:.6}", f.hole_id, f.sample_idx, f.property, f.confidence);
    }
}
