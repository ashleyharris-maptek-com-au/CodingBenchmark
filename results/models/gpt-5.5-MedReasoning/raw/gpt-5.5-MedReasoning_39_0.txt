use std::io::{self, Read};

struct Solver {
    n: usize,
    k: usize,
    adj: Vec<Vec<usize>>,
    orig: Vec<usize>,
    side: Vec<i8>,
    best_side: Vec<i8>,
    best_cut: i32,
}

impl Solver {
    fn cut_of(&self, side: &[i8]) -> i32 {
        let mut cut = 0;
        for v in 0..self.n {
            for &u in &self.adj[v] {
                if u > v && side[u] != side[v] {
                    cut += 1;
                }
            }
        }
        cut
    }

    fn improve_partition(&self, side: &mut [i8]) {
        loop {
            let mut best_delta = 0;
            let mut best_pair = None;

            for a in 0..self.n {
                if side[a] != 0 {
                    continue;
                }
                let mut same_a = 0;
                let mut opp_a = 0;
                for &x in &self.adj[a] {
                    if side[x] == side[a] {
                        same_a += 1;
                    } else {
                        opp_a += 1;
                    }
                }
                let flip_a_delta = same_a - opp_a;

                for b in 0..self.n {
                    if side[b] != 1 {
                        continue;
                    }
                    let mut same_b = 0;
                    let mut opp_b = 0;
                    for &x in &self.adj[b] {
                        if side[x] == side[b] {
                            same_b += 1;
                        } else {
                            opp_b += 1;
                        }
                    }

                    let mut delta = flip_a_delta + same_b - opp_b;
                    delta += 2 * self.adj[a].iter().filter(|&&x| x == b).count() as i32;

                    if delta < best_delta {
                        best_delta = delta;
                        best_pair = Some((a, b));
                    }
                }
            }

            if let Some((a, b)) = best_pair {
                side[a] = 1;
                side[b] = 0;
            } else {
                break;
            }
        }
    }

    fn seed_upper_bound(&mut self) {
        let mut candidates: Vec<Vec<i8>> = Vec::new();

        let mut by_order = vec![1; self.n];
        for i in 0..self.k {
            by_order[i] = 0;
        }
        candidates.push(by_order);

        let mut alternating = vec![1; self.n];
        for i in 0..self.k {
            alternating[i * 2 % self.n] = 0;
        }
        candidates.push(alternating);

        let mut rng: u64 = 0x9e3779b97f4a7c15u64 ^ (self.n as u64);
        let trials = if self.n <= 20 { 16 } else { 96 };
        for _ in 0..trials {
            let mut keys: Vec<(u64, usize)> = (0..self.n)
                .map(|i| {
                    rng ^= rng << 7;
                    rng ^= rng >> 9;
                    rng = rng.wrapping_mul(0xbf58476d1ce4e5b9);
                    (rng, i)
                })
                .collect();
            keys.sort_unstable();
            let mut s = vec![1; self.n];
            for i in 0..self.k {
                s[keys[i].1] = 0;
            }
            candidates.push(s);
        }

        for mut cand in candidates {
            self.improve_partition(&mut cand);
            let cut = self.cut_of(&cand);
            if cut < self.best_cut {
                self.best_cut = cut;
                self.best_side = cand;
            }
        }
    }

    fn lower_bound(&self, idx: usize, a_count: usize, b_count: usize, cur_cut: i32) -> i32 {
        let rem = self.n - idx;
        if a_count > self.k || b_count > self.k {
            return i32::MAX / 4;
        }
        let need_a = self.k - a_count;
        let need_b = self.k - b_count;
        if need_a + need_b != rem {
            return i32::MAX / 4;
        }

        let mut base = 0;
        let mut diffs = Vec::with_capacity(rem);
        for v in idx..self.n {
            let mut as_a = 0;
            let mut as_b = 0;
            for &u in &self.adj[v] {
                if u < idx {
                    if self.side[u] == 0 {
                        as_b += 1;
                    } else {
                        as_a += 1;
                    }
                }
            }
            base += as_b;
            diffs.push(as_a - as_b);
        }
        diffs.sort_unstable();
        cur_cut + base + diffs.iter().take(need_a).sum::<i32>()
    }

    fn assign_cost(&self, v: usize, s: i8, idx: usize) -> i32 {
        let mut add = 0;
        for &u in &self.adj[v] {
            if u < idx && self.side[u] != s {
                add += 1;
            }
        }
        add
    }

    fn search(&mut self, idx: usize, a_count: usize, b_count: usize, cur_cut: i32) {
        if cur_cut >= self.best_cut {
            return;
        }

        let lb = self.lower_bound(idx, a_count, b_count, cur_cut);
        if lb >= self.best_cut {
            return;
        }

        if idx == self.n {
            self.best_cut = cur_cut;
            self.best_side.clone_from(&self.side);
            return;
        }

        let mut branches = Vec::with_capacity(2);
        if a_count < self.k {
            branches.push((self.assign_cost(idx, 0, idx), 0i8));
        }
        if b_count < self.k {
            branches.push((self.assign_cost(idx, 1, idx), 1i8));
        }
        branches.sort_unstable_by_key(|&(add, _)| add);

        for (add, s) in branches {
            self.side[idx] = s;
            if s == 0 {
                self.search(idx + 1, a_count + 1, b_count, cur_cut + add);
            } else {
                self.search(idx + 1, a_count, b_count + 1, cur_cut + add);
            }
            self.side[idx] = -1;
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    if n == 0 {
        println!("0\n");
        return;
    }

    let mut edges = Vec::with_capacity(m);
    let mut degree = vec![0usize; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        edges.push((u, v));
        degree[u] += 1;
        degree[v] += 1;
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by(|&a, &b| degree[b].cmp(&degree[a]).then_with(|| a.cmp(&b)));

    let mut pos = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        pos[v] = i;
    }

    let mut adj = vec![Vec::new(); n];
    for (u0, v0) in edges {
        let u = pos[u0];
        let v = pos[v0];
        adj[u].push(v);
        adj[v].push(u);
    }

    for xs in &mut adj {
        xs.sort_unstable();
    }

    let mut solver = Solver {
        n,
        k: n / 2,
        adj,
        orig: order,
        side: vec![-1; n],
        best_side: vec![1; n],
        best_cut: i32::MAX / 4,
    };

    solver.seed_upper_bound();

    solver.side[0] = 0;
    solver.search(1, 1, 0, 0);

    let mut part_a = Vec::with_capacity(solver.k);
    for i in 0..n {
        if solver.best_side[i] == 0 {
            part_a.push(solver.orig[i]);
        }
    }
    part_a.sort_unstable();

    println!("{}", solver.best_cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
