use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Solver {
    n: usize,
    start: usize,
    require_cycle: bool,
    adj: Vec<Vec<usize>>,
    coord: Vec<(usize, usize)>,
    used: Vec<bool>,
    path: Vec<usize>,
    deadline: Instant,
}

impl Solver {
    fn adjacent_to_start(&self, v: usize) -> bool {
        self.adj[v].iter().any(|&u| u == self.start)
    }

    fn connected_from_start(&self) -> bool {
        if self.n == 0 {
            return false;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[self.start] = true;
        q.push_back(self.start);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            for &u in &self.adj[v] {
                if !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == self.n
    }

    fn residual_connected(&self, cur: usize, remaining: usize) -> bool {
        if remaining == 0 {
            return true;
        }
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        seen[cur] = true;
        q.push_back(cur);
        let mut count = 0usize;
        while let Some(v) = q.pop_front() {
            count += 1;
            for &u in &self.adj[v] {
                if !seen[u] && (!self.used[u] || u == cur) {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        count == remaining + 1
    }

    fn residual_degree_prune(&self, cur: usize, remaining: usize) -> bool {
        if remaining == 0 {
            return true;
        }

        let mut forced_endpoints = 0usize;
        for v in 0..self.n {
            if self.used[v] {
                continue;
            }
            let mut deg = 0usize;
            for &u in &self.adj[v] {
                if !self.used[u] || u == cur {
                    deg += 1;
                }
            }
            if deg == 0 {
                return false;
            }
            if deg == 1 {
                forced_endpoints += 1;
                if forced_endpoints > 1 {
                    return false;
                }
            }
        }
        true
    }

    fn onward_degree(&self, v: usize) -> usize {
        self.adj[v].iter().filter(|&&u| !self.used[u]).count()
    }

    fn dfs(&mut self, cur: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }

        let remaining = self.n - self.path.len();
        if remaining == 0 {
            return !self.require_cycle || self.adjacent_to_start(cur);
        }

        if !self.residual_connected(cur, remaining) {
            return false;
        }
        if !self.residual_degree_prune(cur, remaining) {
            return false;
        }

        let mut cand: Vec<usize> = self.adj[cur]
            .iter()
            .copied()
            .filter(|&u| !self.used[u])
            .collect();

        cand.sort_by_key(|&u| {
            let close_penalty = if self.require_cycle && remaining > 1 && self.adjacent_to_start(u) {
                10000usize
            } else {
                0usize
            };
            close_penalty + self.onward_degree(u)
        });

        for u in cand {
            self.used[u] = true;
            self.path.push(u);
            if self.dfs(u) {
                return true;
            }
            self.path.pop();
            self.used[u] = false;
        }
        false
    }
}

fn rectangular_path(w: usize, h: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    if w == 0 || h == 0 {
        return None;
    }
    if w == 1 && h == 1 {
        return if require_cycle { None } else { Some(vec![(0, 0)]) };
    }

    let mut p = Vec::with_capacity(w * h);
    if !require_cycle {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    p.push((x, y));
                }
            } else {
                for x in (0..w).rev() {
                    p.push((x, y));
                }
            }
        }
        return Some(p);
    }

    if w == 1 || h == 1 || (w * h) % 2 == 1 {
        return None;
    }

    if w % 2 == 0 {
        for x in 0..w {
            p.push((x, 0));
        }
        for x in (0..w).rev() {
            if (w - 1 - x) % 2 == 0 {
                for y in 1..h {
                    p.push((x, y));
                }
            } else {
                for y in (1..h).rev() {
                    p.push((x, y));
                }
            }
        }
    } else {
        for y in 0..h {
            p.push((0, y));
        }
        for y in (0..h).rev() {
            if (h - 1 - y) % 2 == 0 {
                for x in 1..w {
                    p.push((x, y));
                }
            } else {
                for x in (1..w).rev() {
                    p.push((x, y));
                }
            }
        }
    }

    if p.last().is_some_and(|&(x, y)| x + y == 1) {
        Some(p)
    } else {
        None
    }
}

fn print_none() {
    println!("0");
}

fn print_path(path: &[(usize, usize)]) {
    println!("{}", path.len());
    for &(x, y) in path {
        println!("{} {}", x, y);
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            print_none();
            return;
        }
    };
    let h: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            print_none();
            return;
        }
    };
    let obstacle_count: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => {
            print_none();
            return;
        }
    };
    let require_cycle = match it.next().and_then(|s| s.parse::<u32>().ok()) {
        Some(v) => v != 0,
        None => {
            print_none();
            return;
        }
    };

    if w == 0 || h == 0 {
        print_none();
        return;
    }

    let total = match w.checked_mul(h) {
        Some(v) => v,
        None => {
            print_none();
            return;
        }
    };

    let mut blocked = vec![false; total];
    for _ in 0..obstacle_count {
        let x: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => {
                print_none();
                return;
            }
        };
        let y: usize = match it.next().and_then(|s| s.parse().ok()) {
            Some(v) => v,
            None => {
                print_none();
                return;
            }
        };
        if x >= w || y >= h {
            print_none();
            return;
        }
        blocked[y * w + x] = true;
    }

    if blocked[0] {
        print_none();
        return;
    }

    let free_count = blocked.iter().filter(|&&b| !b).count();
    if free_count == 0 || (require_cycle && free_count < 4) {
        print_none();
        return;
    }

    if obstacle_count == 0 {
        if let Some(path) = rectangular_path(w, h, require_cycle) {
            print_path(&path);
        } else {
            print_none();
        }
        return;
    }

    let mut id = vec![usize::MAX; total];
    let mut coord = Vec::with_capacity(free_count);
    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            let flat = y * w + x;
            if !blocked[flat] {
                let idx = coord.len();
                id[flat] = idx;
                coord.push((x, y));
                if (x + y) % 2 == 0 {
                    black += 1;
                } else {
                    white += 1;
                }
            }
        }
    }

    if require_cycle {
        if black != white {
            print_none();
            return;
        }
    } else {
        let start_black = true;
        if black.abs_diff(white) > 1 || (black != white && !start_black && white < black) {
            print_none();
            return;
        }
        if black != white && black < white {
            print_none();
            return;
        }
    }

    let n = coord.len();
    let mut adj = vec![Vec::new(); n];
    let dirs = [(1isize, 0isize), (-1, 0), (0, 1), (0, -1)];
    for i in 0..n {
        let (x, y) = coord[i];
        for (dx, dy) in dirs {
            let nx = x as isize + dx;
            let ny = y as isize + dy;
            if nx >= 0 && ny >= 0 && (nx as usize) < w && (ny as usize) < h {
                let j = id[ny as usize * w + nx as usize];
                if j != usize::MAX {
                    adj[i].push(j);
                }
            }
        }
    }

    let start = id[0];
    let degree_one = adj.iter().filter(|a| a.len() == 1).count();
    if require_cycle {
        if adj.iter().any(|a| a.len() < 2) {
            print_none();
            return;
        }
    } else if n > 1 {
        if adj[start].is_empty() || degree_one > 2 {
            print_none();
            return;
        }
        if degree_one == 2 && adj[start].len() != 1 {
            print_none();
            return;
        }
    }

    let mut solver = Solver {
        n,
        start,
        require_cycle,
        adj,
        coord,
        used: vec![false; n],
        path: Vec::with_capacity(n),
        deadline: Instant::now() + Duration::from_secs(285),
    };

    if !solver.connected_from_start() {
        print_none();
        return;
    }

    solver.used[start] = true;
    solver.path.push(start);

    if solver.dfs(start) {
        let out: Vec<(usize, usize)> = solver.path.iter().map(|&i| solver.coord[i]).collect();
        print_path(&out);
    } else {
        print_none();
    }
}
