using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec operator +(Vec a, Vec b) => new Vec(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec operator -(Vec a, Vec b) => new Vec(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec operator *(Vec a, double s) => new Vec(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Norm2() => X * X + Y * Y + Z * Z;
        public static double Dot(Vec a, Vec b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec Cross(Vec a, Vec b) => new Vec(
            a.Y * b.Z - a.Z * b.Y,
            a.Z * b.X - a.X * b.Z,
            a.X * b.Y - a.Y * b.X);
    }

    struct Orbit
    {
        public Vec R, V, H;
        public double Rn, Vn, A, Ecc, Period;

        public Orbit(Vec r, Vec v, double mu)
        {
            R = r;
            V = v;
            Rn = Math.Max(1e-9, r.Norm());
            Vn = v.Norm();
            H = Vec.Cross(r, v);

            double eps = Vn * Vn * 0.5 - mu / Rn;
            A = eps < -1e-12 ? -mu / (2.0 * eps) : Rn;
            if (!double.IsFinite(A) || A <= 0) A = Rn;

            Vec eVec = (Vec.Cross(v, H) * (1.0 / mu)) - (r * (1.0 / Rn));
            Ecc = Math.Min(2.0, Math.Max(0.0, eVec.Norm()));
            Period = 2.0 * Math.PI * Math.Sqrt(Math.Max(1.0, A * A * A / mu));
        }
    }

    static double Mu;
    static Orbit Start;
    static Orbit[] Stations = Array.Empty<Orbit>();
    static double[,] Cost = new double[0, 0]; // node 0 is start, nodes 1..N are stations

    public static void Main()
    {
        var tokens = ReadAllTokens();
        if (tokens.Count < 8) return;

        int p = 0;
        int n = int.Parse(tokens[p++], CultureInfo.InvariantCulture);
        Mu = double.Parse(tokens[p++], CultureInfo.InvariantCulture);
        Start = ReadOrbit(tokens, ref p);

        Stations = new Orbit[n];
        for (int i = 0; i < n; i++) Stations[i] = ReadOrbit(tokens, ref p);

        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        BuildCosts(n);
        int[] route = n <= 10 ? ExactPath(n) : HeuristicPath(n);
        Console.WriteLine(string.Join(" ", route));
    }

    static List<string> ReadAllTokens()
    {
        string input = Console.In.ReadToEnd();
        return input.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries).ToList();
    }

    static Orbit ReadOrbit(List<string> t, ref int p)
    {
        double x = double.Parse(t[p++], CultureInfo.InvariantCulture);
        double y = double.Parse(t[p++], CultureInfo.InvariantCulture);
        double z = double.Parse(t[p++], CultureInfo.InvariantCulture);
        double vx = double.Parse(t[p++], CultureInfo.InvariantCulture);
        double vy = double.Parse(t[p++], CultureInfo.InvariantCulture);
        double vz = double.Parse(t[p++], CultureInfo.InvariantCulture);
        return new Orbit(new Vec(x, y, z), new Vec(vx, vy, vz), Mu);
    }

    static void BuildCosts(int n)
    {
        Cost = new double[n + 1, n + 1];
        Orbit[] nodes = new Orbit[n + 1];
        nodes[0] = Start;
        for (int i = 0; i < n; i++) nodes[i + 1] = Stations[i];

        for (int i = 0; i <= n; i++)
        {
            for (int j = 0; j <= n; j++)
            {
                Cost[i, j] = i == j ? 0.0 : TransferCost(nodes[i], nodes[j]);
            }
        }
    }

    static double TransferCost(Orbit a, Orbit b)
    {
        double r1 = Math.Max(1.0, a.A);
        double r2 = Math.Max(1.0, b.A);
        double sum = Math.Max(1.0, r1 + r2);

        double vc1 = Math.Sqrt(Mu / r1);
        double vc2 = Math.Sqrt(Mu / r2);
        double dv1 = Math.Abs(vc1 * (Math.Sqrt(2.0 * r2 / sum) - 1.0));
        double dv2 = Math.Abs(vc2 * (1.0 - Math.Sqrt(2.0 * r1 / sum)));
        double hohmann = dv1 + dv2;

        double hn1 = Math.Max(1e-12, a.H.Norm());
        double hn2 = Math.Max(1e-12, b.H.Norm());
        double cosI = Clamp(Vec.Dot(a.H, b.H) / (hn1 * hn2), -1.0, 1.0);
        double dInc = Math.Acos(cosI);
        double vAvg = Math.Sqrt(Mu / Math.Max(1.0, 0.5 * (a.Rn + b.Rn)));
        double plane = 2.0 * vAvg * Math.Sin(0.5 * dInc);

        double cosPos = Clamp(Vec.Dot(a.R, b.R) / (a.Rn * b.Rn), -1.0, 1.0);
        double phaseAngle = Math.Acos(cosPos);
        double transferTime = Math.PI * Math.Sqrt(Math.Pow(0.5 * (r1 + r2), 3.0) / Mu);
        double timing = Math.Abs(a.Period - b.Period) / Math.Max(a.Period, b.Period);

        double eccentricity = Math.Abs(a.Ecc - b.Ecc) * vAvg;
        double speed = Math.Abs(a.Vn - b.Vn);
        double altitude = Math.Abs(a.Rn - b.Rn) / Math.Max(a.Rn, b.Rn);

        return hohmann
             + 0.92 * plane
             + 0.18 * eccentricity
             + 0.055 * speed
             + 0.025 * vAvg * phaseAngle
             + 0.015 * vAvg * timing
             + 0.02 * vAvg * altitude
             + 1e-6 * transferTime;
    }

    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    static int[] ExactPath(int n)
    {
        int states = 1 << n;
        double[,] dp = new double[states, n];
        int[,] parent = new int[states, n];
        const double Inf = 1e100;

        for (int s = 0; s < states; s++)
            for (int i = 0; i < n; i++)
            {
                dp[s, i] = Inf;
                parent[s, i] = -1;
            }

        for (int i = 0; i < n; i++) dp[1 << i, i] = Cost[0, i + 1];

        for (int mask = 1; mask < states; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (cur >= Inf * 0.5) continue;
                int remaining = (states - 1) ^ mask;
                for (int bits = remaining; bits != 0; bits &= bits - 1)
                {
                    int next = BitIndex(bits & -bits);
                    int nm = mask | (1 << next);
                    double cand = cur + Cost[last + 1, next + 1];
                    if (cand < dp[nm, next])
                    {
                        dp[nm, next] = cand;
                        parent[nm, next] = last;
                    }
                }
            }
        }

        int full = states - 1;
        int bestLast = 0;
        double best = dp[full, 0];
        for (int i = 1; i < n; i++)
        {
            if (dp[full, i] < best)
            {
                best = dp[full, i];
                bestLast = i;
            }
        }

        int[] route = new int[n];
        int m = full, at = bestLast;
        for (int k = n - 1; k >= 0; k--)
        {
            route[k] = at;
            int prev = parent[m, at];
            m ^= 1 << at;
            at = prev;
        }
        return route;
    }

    static int BitIndex(int bit)
    {
        int i = 0;
        while ((bit >>= 1) != 0) i++;
        return i;
    }

    static int[] HeuristicPath(int n)
    {
        var candidates = new List<int[]>();

        candidates.Add(NearestNeighbor(n, 0));
        candidates.Add(NearestNeighbor(n, FarthestFromStart(n)));

        int starts = Math.Min(n, n <= 25 ? 8 : 16);
        foreach (int s in Enumerable.Range(0, n).OrderBy(i => Cost[0, i + 1]).Take(starts))
            candidates.Add(NearestNeighbor(n, s));

        candidates.Add(CheapestInsertion(n));

        int[] bestRoute = candidates[0];
        double best = RouteCost(bestRoute);
        foreach (var r in candidates)
        {
            Improve(r, n);
            double c = RouteCost(r);
            if (c < best)
            {
                best = c;
                bestRoute = r;
            }
        }

        Improve(bestRoute, n);
        return bestRoute;
    }

    static int FarthestFromStart(int n)
    {
        int best = 0;
        for (int i = 1; i < n; i++)
            if (Cost[0, i + 1] > Cost[0, best + 1]) best = i;
        return best;
    }

    static int[] NearestNeighbor(int n, int first)
    {
        bool[] used = new bool[n];
        int[] route = new int[n];
        route[0] = first;
        used[first] = true;

        for (int pos = 1; pos < n; pos++)
        {
            int prevNode = route[pos - 1] + 1;
            int best = -1;
            double bestCost = double.PositiveInfinity;
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                double c = Cost[prevNode, i + 1];
                if (c < bestCost)
                {
                    bestCost = c;
                    best = i;
                }
            }
            route[pos] = best;
            used[best] = true;
        }

        if (Cost[0, route[0] + 1] > Cost[0, route[n - 1] + 1])
            Array.Reverse(route);
        return route;
    }

    static int[] CheapestInsertion(int n)
    {
        bool[] used = new bool[n];
        var route = new List<int>();
        int first = 0;
        int second = -1;
        double bestPair = double.PositiveInfinity;

        for (int i = 0; i < n; i++)
            for (int j = 0; j < n; j++)
                if (i != j)
                {
                    double c = Cost[0, i + 1] + Cost[i + 1, j + 1];
                    if (c < bestPair) { bestPair = c; first = i; second = j; }
                }

        route.Add(first);
        used[first] = true;
        if (second >= 0)
        {
            route.Add(second);
            used[second] = true;
        }

        while (route.Count < n)
        {
            int bestStation = -1, bestPos = 0;
            double bestIncrease = double.PositiveInfinity;

            for (int s = 0; s < n; s++)
            {
                if (used[s]) continue;
                for (int pos = 0; pos <= route.Count; pos++)
                {
                    int prevNode = pos == 0 ? 0 : route[pos - 1] + 1;
                    double remove = pos == 0 ? 0.0 : Cost[prevNode, pos < route.Count ? route[pos] + 1 : prevNode];
                    double add = Cost[prevNode, s + 1];
                    if (pos < route.Count) add += Cost[s + 1, route[pos] + 1];
                    double inc = add - remove;
                    if (inc < bestIncrease)
                    {
                        bestIncrease = inc;
                        bestStation = s;
                        bestPos = pos;
                    }
                }
            }

            route.Insert(bestPos, bestStation);
            used[bestStation] = true;
        }

        return route.ToArray();
    }

    static void Improve(int[] route, int n)
    {
        TwoOpt(route, n);
        Relocate(route, n);
        TwoOpt(route, n);
    }

    static void TwoOpt(int[] route, int n)
    {
        bool improved = true;
        int passes = 0;
        while (improved && passes++ < 80)
        {
            improved = false;
            for (int i = 0; i < n - 2; i++)
            {
                int a = i == 0 ? 0 : route[i - 1] + 1;
                int b = route[i] + 1;
                for (int k = i + 1; k < n; k++)
                {
                    int c = route[k] + 1;
                    int d = k + 1 < n ? route[k + 1] + 1 : -1;

                    double oldCost = Cost[a, b] + (d >= 0 ? Cost[c, d] : 0.0);
                    double newCost = Cost[a, c] + (d >= 0 ? Cost[b, d] : 0.0);
                    if (newCost + 1e-10 < oldCost)
                    {
                        Array.Reverse(route, i, k - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static void Relocate(int[] route, int n)
    {
        bool improved = true;
        int passes = 0;
        while (improved && passes++ < 40)
        {
            improved = false;
            for (int from = 0; from < n; from++)
            {
                int moved = route[from];
                int before = from == 0 ? 0 : route[from - 1] + 1;
                int after = from + 1 < n ? route[from + 1] + 1 : -1;
                double removal = Cost[before, moved + 1] + (after >= 0 ? Cost[moved + 1, after] - Cost[before, after] : 0.0);

                for (int to = 0; to < n; to++)
                {
                    if (to == from || to == from + 1) continue;
                    int insertBefore = to == 0 ? 0 : route[to - 1] + 1;
                    int insertAfter = to < n ? route[to] + 1 : -1;
                    if (insertBefore == moved + 1 || insertAfter == moved + 1) continue;

                    double insertion = Cost[insertBefore, moved + 1] + (insertAfter >= 0 ? Cost[moved + 1, insertAfter] - Cost[insertBefore, insertAfter] : 0.0);
                    if (insertion + 1e-10 < removal)
                    {
                        Move(route, from, to);
                        improved = true;
                        goto NextPass;
                    }
                }
            }
        NextPass:;
        }
    }

    static void Move(int[] route, int from, int to)
    {
        int x = route[from];
        if (from < to)
        {
            Array.Copy(route, from + 1, route, from, to - from - 1);
            route[to - 1] = x;
        }
        else
        {
            Array.Copy(route, to, route, to + 1, from - to);
            route[to] = x;
        }
    }

    static double RouteCost(int[] route)
    {
        if (route.Length == 0) return 0.0;
        double total = Cost[0, route[0] + 1];
        for (int i = 1; i < route.Length; i++)
            total += Cost[route[i - 1] + 1, route[i] + 1];
        return total;
    }
}
