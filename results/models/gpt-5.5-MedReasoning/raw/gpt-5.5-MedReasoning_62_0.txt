declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Town = { id: string; x: number; y: number; value: number; idx: number; baselineBurned: boolean };
type Candidate = { cells: number[]; est: number; kind: string };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid || [];
const n = width * height;
const budget: number = Math.max(0, data.cutBudget | 0);

function idx(x: number, y: number): number { return y * width + x; }
function xOf(i: number): number { return i % width; }
function yOf(i: number): number { return (i / width) | 0; }

const burnable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || "";
  for (let x = 0; x < width; x++) {
    if (row.charCodeAt(x) === 46) burnable[idx(x, y)] = 1; // '.'
  }
}

const ignitionMark = new Uint8Array(n);
const ignitions: number[] = [];
for (const p of (data.ignitions || [])) {
  const x = p[0] | 0, y = p[1] | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) {
    const id = idx(x, y);
    if (burnable[id] && !ignitionMark[id]) {
      ignitionMark[id] = 1;
      ignitions.push(id);
    }
  }
}

const townMark = new Uint8Array(n);
const townValueAt = new Float64Array(n);
const towns: Town[] = [];
for (const t of (data.towns || [])) {
  const x = t.x | 0, y = t.y | 0;
  if (x >= 0 && x < width && y >= 0 && y < height) {
    const id = idx(x, y);
    townMark[id] = 1;
    townValueAt[id] += Number(t.value) || 0;
    towns.push({ id: String(t.id), x, y, value: Number(t.value) || 0, idx: id, baselineBurned: false });
  }
}

const cutMark = new Uint8Array(n);
let visitStamp = 1;
const visited = new Int32Array(n);
const queue = new Int32Array(Math.max(1, n));

function simulateBurnedValue(): number {
  if (ignitions.length === 0) return 0;
  const stamp = visitStamp++;
  if (visitStamp > 2000000000) {
    visited.fill(0);
    visitStamp = 1;
  }
  let head = 0, tail = 0;
  for (const s of ignitions) {
    if (burnable[s] && !cutMark[s] && visited[s] !== stamp) {
      visited[s] = stamp;
      queue[tail++] = s;
    }
  }
  while (head < tail) {
    const v = queue[head++];
    const x = v % width;
    let u = v - 1;
    if (x > 0 && burnable[u] && !cutMark[u] && visited[u] !== stamp) { visited[u] = stamp; queue[tail++] = u; }
    u = v + 1;
    if (x + 1 < width && burnable[u] && !cutMark[u] && visited[u] !== stamp) { visited[u] = stamp; queue[tail++] = u; }
    u = v - width;
    if (v >= width && burnable[u] && !cutMark[u] && visited[u] !== stamp) { visited[u] = stamp; queue[tail++] = u; }
    u = v + width;
    if (v + width < n && burnable[u] && !cutMark[u] && visited[u] !== stamp) { visited[u] = stamp; queue[tail++] = u; }
  }
  let burned = 0;
  for (const t of towns) {
    if (t.baselineBurned && visited[t.idx] === stamp) burned += t.value;
  }
  return burned;
}

function baselineBfs(): { reach: Uint8Array; dist: Int32Array; parent: Int32Array } {
  const reach = new Uint8Array(n);
  const dist = new Int32Array(n);
  const parent = new Int32Array(n);
  dist.fill(-1);
  parent.fill(-1);
  let head = 0, tail = 0;
  for (const s of ignitions) {
    if (burnable[s] && !reach[s]) {
      reach[s] = 1;
      dist[s] = 0;
      parent[s] = s;
      queue[tail++] = s;
    }
  }
  while (head < tail) {
    const v = queue[head++];
    const nd = dist[v] + 1;
    const x = v % width;
    let u = v - 1;
    if (x > 0 && burnable[u] && !reach[u]) { reach[u] = 1; dist[u] = nd; parent[u] = v; queue[tail++] = u; }
    u = v + 1;
    if (x + 1 < width && burnable[u] && !reach[u]) { reach[u] = 1; dist[u] = nd; parent[u] = v; queue[tail++] = u; }
    u = v - width;
    if (v >= width && burnable[u] && !reach[u]) { reach[u] = 1; dist[u] = nd; parent[u] = v; queue[tail++] = u; }
    u = v + width;
    if (v + width < n && burnable[u] && !reach[u]) { reach[u] = 1; dist[u] = nd; parent[u] = v; queue[tail++] = u; }
  }
  return { reach, dist, parent };
}

const base = baselineBfs();
let baselineBurnedValue = 0;
for (const t of towns) {
  t.baselineBurned = burnable[t.idx] === 1 && base.reach[t.idx] === 1;
  if (t.baselineBurned) baselineBurnedValue += t.value;
}

if (budget <= 0 || baselineBurnedValue <= 0 || ignitions.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const candidates: Candidate[] = [];
const seenCandidates = new Set<string>();

function addCandidate(rawCells: number[], est: number, kind: string) {
  if (est <= 0 || rawCells.length === 0) return;
  const cells: number[] = [];
  const local = new Set<number>();
  for (const c of rawCells) {
    if (c < 0 || c >= n || local.has(c)) continue;
    if (!burnable[c] || !base.reach[c] || ignitionMark[c] || townMark[c]) return;
    local.add(c);
    cells.push(c);
  }
  if (cells.length === 0 || cells.length > budget) return;
  cells.sort((a, b) => a - b);
  const key = cells.length + ":" + cells.join(",");
  if (seenCandidates.has(key)) return;
  seenCandidates.add(key);
  candidates.push({ cells, est, kind });
}

function generateTownRings() {
  const burnedTowns = towns
    .filter(t => t.baselineBurned && t.value > 0)
    .sort((a, b) => b.value - a.value);
  const townLimit = Math.min(burnedTowns.length, n > 300000 ? 35 : 70);
  const dist = new Int32Array(n);
  const touched: number[] = [];
  const maxVisitedPerTown = Math.min(n, n > 300000 ? 60000 : 160000);
  for (let ti = 0; ti < townLimit; ti++) {
    dist.fill(-1);
    touched.length = 0;
    const start = burnedTowns[ti].idx;
    let head = 0, tail = 0;
    dist[start] = 0;
    touched.push(start);
    queue[tail++] = start;
    let currentDepth = 0;
    let layer: number[] = [];
    let insideValue = burnedTowns[ti].value;
    while (head < tail && touched.length < maxVisitedPerTown) {
      const v = queue[head++];
      const d = dist[v];
      if (d !== currentDepth) {
        if (currentDepth > 0 && layer.length <= budget) addCandidate(layer, insideValue / Math.pow(Math.max(1, layer.length), 0.85), "ring");
        layer = [];
        currentDepth = d;
        if (currentDepth > Math.max(width, height)) break;
      }
      if (d > 0) layer.push(v);
      const x = v % width;
      const tryPush = (u: number) => {
        if (burnable[u] && base.reach[u] && dist[u] < 0) {
          dist[u] = d + 1;
          touched.push(u);
          queue[tail++] = u;
          if (townMark[u]) insideValue += townValueAt[u];
        }
      };
      if (x > 0) tryPush(v - 1);
      if (x + 1 < width) tryPush(v + 1);
      if (v >= width) tryPush(v - width);
      if (v + width < n) tryPush(v + width);
    }
    if (currentDepth > 0 && layer.length <= budget) addCandidate(layer, insideValue / Math.pow(Math.max(1, layer.length), 0.85), "ring");
  }
}

function generateFireFronts() {
  let maxD = 0;
  for (const t of towns) if (t.baselineBurned && base.dist[t.idx] > maxD) maxD = base.dist[t.idx];
  if (maxD <= 1) return;
  const levelCounts = new Int32Array(maxD + 1);
  for (let i = 0; i < n; i++) {
    const d = base.dist[i];
    if (d > 0 && d <= maxD && !townMark[i] && !ignitionMark[i]) levelCounts[d]++;
  }
  const valueBeyond = new Float64Array(maxD + 2);
  for (const t of towns) if (t.baselineBurned) valueBeyond[base.dist[t.idx]] += t.value;
  for (let d = maxD - 1; d >= 0; d--) valueBeyond[d] += valueBeyond[d + 1];
  const step = Math.max(1, Math.floor(maxD / 120));
  for (let d = 1; d <= maxD; d += step) {
    const count = levelCounts[d];
    if (count > 0 && count <= budget && valueBeyond[d + 1] > 0) {
      const cells: number[] = [];
      for (let i = 0; i < n; i++) if (base.dist[i] === d && !townMark[i] && !ignitionMark[i]) cells.push(i);
      addCandidate(cells, valueBeyond[d + 1] / Math.pow(count, 0.85), "front");
    }
  }
}

function generatePathCells() {
  const weights = new Float64Array(n);
  const burnedTowns = towns
    .filter(t => t.baselineBurned && t.value > 0)
    .sort((a, b) => b.value - a.value)
    .slice(0, n > 300000 ? 120 : 260);
  for (const t of burnedTowns) {
    let v = t.idx;
    let guard = 0;
    while (base.parent[v] >= 0 && base.parent[v] !== v && guard++ < n) {
      if (!townMark[v] && !ignitionMark[v]) weights[v] += t.value;
      v = base.parent[v];
    }
  }
  const best: number[] = [];
  for (let i = 0; i < n; i++) if (weights[i] > 0 && burnable[i] && base.reach[i] && !townMark[i] && !ignitionMark[i]) best.push(i);
  best.sort((a, b) => weights[b] - weights[a]);
  const limit = Math.min(best.length, n > 300000 ? 350 : 800);
  for (let i = 0; i < limit; i++) addCandidate([best[i]], weights[best[i]], "path");
}

generateTownRings();
generateFireFronts();
generatePathCells();

candidates.sort((a, b) => b.est - a.est);
const maxCandidates = n > 300000 ? 700 : 1300;
if (candidates.length > maxCandidates) candidates.length = maxCandidates;

const selected: number[] = [];
let remaining = budget;
let currentBurned = simulateBurnedValue();
const startTime = Date.now();
const timeLimitMs = n > 300000 ? 8500 : 12000;

function candidateNewCells(c: Candidate): number[] {
  const out: number[] = [];
  for (const cell of c.cells) if (!cutMark[cell]) out.push(cell);
  return out;
}

while (remaining > 0 && currentBurned > 0 && Date.now() - startTime < timeLimitMs) {
  let best: Candidate | null = null;
  let bestCells: number[] = [];
  let bestBurned = currentBurned;
  let bestScore = 0;
  let evaluated = 0;
  const evalLimit = n > 300000 ? 70 : 150;
  for (const c of candidates) {
    if (Date.now() - startTime >= timeLimitMs) break;
    if (evaluated >= evalLimit && bestScore > 0) break;
    const newCells = candidateNewCells(c);
    if (newCells.length === 0 || newCells.length > remaining) continue;
    for (const cell of newCells) cutMark[cell] = 1;
    const burned = simulateBurnedValue();
    for (const cell of newCells) cutMark[cell] = 0;
    evaluated++;
    const gain = currentBurned - burned;
    if (gain <= 0) continue;
    const score = gain / Math.pow(newCells.length, 0.9);
    if (score > bestScore || (score === bestScore && gain > currentBurned - bestBurned)) {
      bestScore = score;
      best = c;
      bestCells = newCells;
      bestBurned = burned;
    }
  }
  if (!best || bestCells.length === 0) break;
  for (const cell of bestCells) {
    if (!cutMark[cell] && remaining > 0) {
      cutMark[cell] = 1;
      selected.push(cell);
      remaining--;
    }
  }
  currentBurned = bestBurned;
  best.est = -1;
  candidates.sort((a, b) => b.est - a.est);
}

const firebreaks = selected.slice(0, budget).map(i => [xOf(i), yOf(i)]);
process.stdout.write(JSON.stringify({ firebreaks }));
