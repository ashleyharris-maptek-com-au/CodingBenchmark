using namespace std;

static int popcount64(uint64_t x) {
    int count = 0;
    while (x != 0) {
        x &= x - 1;
        ++count;
    }
    return count;
}

struct Timer {
    chrono::steady_clock::time_point start = chrono::steady_clock::now();
    double seconds() const {
        return chrono::duration<double>(chrono::steady_clock::now() - start).count();
    }
};

struct Graph {
    int n = 0;
    int m = 0;
    int k = 0;
    vector<size_t> off;
    vector<int> adj;
    vector<pair<int, int>> edges;
    vector<int> deg;
};

static Graph read_graph() {
    Graph g;
    if (!(cin >> g.n >> g.m >> g.k)) return g;

    g.edges.reserve(g.m);
    g.deg.assign(g.n, 0);
    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= g.n || v < 0 || v >= g.n || u == v) continue;
        g.edges.push_back({u, v});
        ++g.deg[u];
        ++g.deg[v];
    }
    g.m = static_cast<int>(g.edges.size());

    g.off.assign(static_cast<size_t>(g.n) + 1, 0);
    for (int i = 0; i < g.n; ++i) g.off[i + 1] = g.off[i] + static_cast<size_t>(g.deg[i]);
    g.adj.assign(g.off[g.n], 0);
    vector<size_t> cur = g.off;
    for (const auto &e : g.edges) {
        g.adj[cur[e.first]++] = e.second;
        g.adj[cur[e.second]++] = e.first;
    }
    return g;
}

static bool exact_dsatur_rec(const Graph &g, vector<int> &color, vector<uint64_t> &sat,
                             vector<int> &class_size, int left, const Timer &timer) {
    if (left == 0) return true;
    if (timer.seconds() > 8.0) return false;

    int v = -1, best_sat = -1, best_deg = -1;
    for (int i = 0; i < g.n; ++i) {
        if (color[i] != -1) continue;
        int s = popcount64(sat[i]);
        if (s > best_sat || (s == best_sat && g.deg[i] > best_deg)) {
            v = i;
            best_sat = s;
            best_deg = g.deg[i];
        }
    }

    vector<int> choices;
    choices.reserve(g.k);
    for (int c = 0; c < g.k; ++c) {
        if ((sat[v] & (1ULL << c)) == 0) choices.push_back(c);
    }
    sort(choices.begin(), choices.end(), [&](int a, int b) {
        return class_size[a] < class_size[b];
    });

    for (int c : choices) {
        color[v] = c;
        ++class_size[c];
        vector<pair<int, uint64_t>> changed;
        uint64_t bit = 1ULL << c;
        for (size_t p = g.off[v]; p < g.off[v + 1]; ++p) {
            int u = g.adj[p];
            if (color[u] == -1 && (sat[u] & bit) == 0) {
                changed.push_back({u, sat[u]});
                sat[u] |= bit;
            }
        }

        if (exact_dsatur_rec(g, color, sat, class_size, left - 1, timer)) return true;

        for (const auto &x : changed) sat[x.first] = x.second;
        --class_size[c];
        color[v] = -1;
    }
    return false;
}

static bool try_exact_small(const Graph &g, vector<int> &color) {
    if (g.n > 90 || g.k > 63) return false;
    Timer timer;
    color.assign(g.n, -1);
    vector<uint64_t> sat(g.n, 0);
    vector<int> class_size(g.k, 0);
    return exact_dsatur_rec(g, color, sat, class_size, g.n, timer);
}

static void greedy_dsatur(const Graph &g, vector<int> &color) {
    color.assign(g.n, -1);
    vector<uint64_t> sat(g.n, 0);
    vector<int> class_size(g.k, 0);

    for (int step = 0; step < g.n; ++step) {
        int v = -1, best_sat = -1, best_deg = -1;
        for (int i = 0; i < g.n; ++i) {
            if (color[i] != -1) continue;
            int s = popcount64(sat[i]);
            if (s > best_sat || (s == best_sat && g.deg[i] > best_deg)) {
                v = i;
                best_sat = s;
                best_deg = g.deg[i];
            }
        }

        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if ((sat[v] & (1ULL << c)) != 0) continue;
            if (chosen < 0 || class_size[c] < class_size[chosen]) chosen = c;
        }
        if (chosen < 0) chosen = 0;

        color[v] = chosen;
        ++class_size[chosen];
        uint64_t bit = 1ULL << chosen;
        for (size_t p = g.off[v]; p < g.off[v + 1]; ++p) {
            int u = g.adj[p];
            if (color[u] == -1) sat[u] |= bit;
        }
    }
}

static void greedy_degree_order(const Graph &g, vector<int> &color) {
    color.assign(g.n, -1);
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
        return a < b;
    });

    vector<int> seen(g.k, -1), cnt(g.k, 0), class_size(g.k, 0);
    int stamp = 0;
    for (int v : order) {
        ++stamp;
        fill(cnt.begin(), cnt.end(), 0);
        for (size_t p = g.off[v]; p < g.off[v + 1]; ++p) {
            int c = color[g.adj[p]];
            if (c >= 0) {
                seen[c] = stamp;
                ++cnt[c];
            }
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (seen[c] == stamp) continue;
            if (chosen < 0 || class_size[c] < class_size[chosen]) chosen = c;
        }
        if (chosen < 0) {
            chosen = static_cast<int>(min_element(cnt.begin(), cnt.end()) - cnt.begin());
        }
        color[v] = chosen;
        ++class_size[chosen];
    }
}

static void greedy_smallest_last(const Graph &g, vector<int> &color) {
    vector<int> remaining_deg = g.deg;
    vector<char> removed(g.n, 0);
    priority_queue<pair<int, int>, vector<pair<int, int>>, greater<pair<int, int>>> pq;
    for (int i = 0; i < g.n; ++i) pq.push({remaining_deg[i], i});

    vector<int> order;
    order.reserve(g.n);
    while (!pq.empty()) {
        int d = pq.top().first;
        int v = pq.top().second;
        pq.pop();
        if (removed[v] || d != remaining_deg[v]) continue;
        removed[v] = 1;
        order.push_back(v);
        for (size_t p = g.off[v]; p < g.off[v + 1]; ++p) {
            int u = g.adj[p];
            if (!removed[u]) {
                --remaining_deg[u];
                pq.push({remaining_deg[u], u});
            }
        }
    }

    color.assign(g.n, -1);
    vector<int> seen(g.k, -1), cnt(g.k, 0), class_size(g.k, 0);
    int stamp = 0;
    for (auto it = order.rbegin(); it != order.rend(); ++it) {
        int v = *it;
        ++stamp;
        fill(cnt.begin(), cnt.end(), 0);
        for (size_t p = g.off[v]; p < g.off[v + 1]; ++p) {
            int c = color[g.adj[p]];
            if (c >= 0) {
                seen[c] = stamp;
                ++cnt[c];
            }
        }

        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (seen[c] == stamp) continue;
            if (chosen < 0 || class_size[c] < class_size[chosen]) chosen = c;
        }
        if (chosen < 0) {
            chosen = static_cast<int>(min_element(cnt.begin(), cnt.end()) - cnt.begin());
        }
        color[v] = chosen;
        ++class_size[chosen];
    }
}

static long long conflict_count(const Graph &g, const vector<int> &color, vector<int> *bad_vertices = nullptr) {
    long long conflicts = 0;
    if (bad_vertices) bad_vertices->clear();
    vector<char> mark;
    if (bad_vertices) mark.assign(g.n, 0);

    for (const auto &e : g.edges) {
        if (color[e.first] == color[e.second]) {
            ++conflicts;
            if (bad_vertices) {
                if (!mark[e.first]) {
                    mark[e.first] = 1;
                    bad_vertices->push_back(e.first);
                }
                if (!mark[e.second]) {
                    mark[e.second] = 1;
                    bad_vertices->push_back(e.second);
                }
            }
        }
    }
    return conflicts;
}

static void repair_coloring(const Graph &g, vector<int> &color, double time_limit_seconds) {
    if (g.k <= 1) return;

    Timer timer;
    mt19937 rng(1234567);
    vector<int> bad, cnt(g.k, 0), class_size(g.k, 0);
    for (int c : color) {
        if (c >= 0 && c < g.k) ++class_size[c];
    }

    long long best_conf = conflict_count(g, color, &bad);
    vector<int> best = color;
    int stall = 0;

    while (best_conf > 0 && timer.seconds() < time_limit_seconds) {
        shuffle(bad.begin(), bad.end(), rng);
        bool changed_any = false;

        for (int v : bad) {
            fill(cnt.begin(), cnt.end(), 0);
            for (size_t p = g.off[v]; p < g.off[v + 1]; ++p) {
                int c = color[g.adj[p]];
                if (c >= 0) ++cnt[c];
            }

            int old = color[v];
            int chosen = old;
            for (int c = 0; c < g.k; ++c) {
                if (cnt[c] < cnt[chosen] ||
                    (cnt[c] == cnt[chosen] && class_size[c] < class_size[chosen])) {
                    chosen = c;
                }
            }
            if (chosen != old && cnt[chosen] <= cnt[old]) {
                --class_size[old];
                color[v] = chosen;
                ++class_size[chosen];
                changed_any = true;
            }
        }

        long long now = conflict_count(g, color, &bad);
        if (now < best_conf) {
            best_conf = now;
            best = color;
            stall = 0;
        } else {
            ++stall;
        }

        if (bad.empty()) {
            best_conf = 0;
            best = color;
            break;
        }

        if (!changed_any || stall >= 8) {
            int kicks = max(1, min<int>(static_cast<int>(bad.size()), 1 + static_cast<int>(bad.size() / 50)));
            uniform_int_distribution<int> pick_vertex(0, static_cast<int>(bad.size()) - 1);
            uniform_int_distribution<int> pick_color(0, g.k - 1);
            for (int i = 0; i < kicks; ++i) {
                int v = bad[pick_vertex(rng)];
                int old = color[v];
                int nc = pick_color(rng);
                if (old != nc) {
                    --class_size[old];
                    color[v] = nc;
                    ++class_size[nc];
                }
            }
            stall = 0;
        }
    }

    color = best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g = read_graph();
    if (g.n <= 0) return 0;

    if (g.k <= 0) return 0;
    if (g.k == 1) {
        for (int i = 0; i < g.n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << '\n';
        return 0;
    }

    vector<int> color;

    if (!try_exact_small(g, color)) {
        if (g.k <= 63 && g.n <= 5000) {
            greedy_dsatur(g, color);
        } else if (g.n > 0 && (2LL * g.m) / max(1, g.n) <= 200) {
            greedy_smallest_last(g, color);
        } else {
            greedy_degree_order(g, color);
        }
    }

    long long conflicts = conflict_count(g, color, nullptr);
    if (conflicts > 0) {
        repair_coloring(g, color, 280.0);
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        int c = color[i];
        if (c < 0 || c >= g.k) c = ((c % g.k) + g.k) % g.k;
        cout << c;
    }
    cout << '\n';
    return 0;
}
