; SPIR-V
; Version: 1.0
OpCapability Shader
%GLSL450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inUV "uv"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_input_v2float = OpTypePointer Input %v2float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float
%inWorldPos = OpVariable %ptr_input_v3float Input
%inNormal = OpVariable %ptr_input_v3float Input
%inUV = OpVariable %ptr_input_v2float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo = OpVariable %ptr_uniform_UBO Uniform
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_5 = OpConstant %float 0.5
%float_0_9 = OpConstant %float 0.9
%float_1 = OpConstant %float 1
%float_8 = OpConstant %float 8
%v2float_8 = OpConstantComposite %v2float %float_8 %float_8
%main = OpFunction %void None %fn_void
%entry = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%uv = OpLoad %v2float %inUV
%uvScaled = OpFMul %v2float %uv %v2float_8
%uvFloor = OpExtInst %v2float %GLSL450 Floor %uvScaled
%floorX = OpCompositeExtract %float %uvFloor 0
%floorY = OpCompositeExtract %float %uvFloor 1
%ix = OpConvertFToS %int %floorX
%iy = OpConvertFToS %int %floorY
%sum = OpIAdd %int %ix %iy
%parity = OpBitwiseAnd %int %sum %int_1
%isEven = OpIEqual %bool %parity %int_0
%checker = OpSelect %float %isEven %float_0_9 %float_0_2
%lightPtr = OpAccessChain %ptr_uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%lightVec = OpFSub %v3float %lightPos %worldPos
%lightDir = OpExtInst %v3float %GLSL450 Normalize %lightVec
%normDir = OpExtInst %v3float %GLSL450 Normalize %normal
%dotNL = OpDot %float %normDir %lightDir
%diffuse = OpExtInst %float %GLSL450 FMax %dotNL %float_0
%halfDiffuse = OpFMul %float %float_0_5 %diffuse
%shade = OpFAdd %float %float_0_5 %halfDiffuse
%rgb = OpFMul %float %checker %shade
%color = OpCompositeConstruct %v4float %rgb %rgb %rgb %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
