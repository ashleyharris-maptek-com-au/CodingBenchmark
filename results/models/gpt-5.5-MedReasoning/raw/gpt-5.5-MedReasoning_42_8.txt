%v4float = OpTypeVector %float 4

%uint_3 = OpConstant %uint 3

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input
%arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %fn_void
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0

%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum012 %float_third
%delta0 = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %delta0 %float_shrink
%new0 = OpFAdd %v4float %centroid %scaled0
%color0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%color0 = OpLoad %v3float %color0_ptr
%out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos0 %new0
OpStore %outColor %color0
OpEmitVertex
%delta1 = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %delta1 %float_shrink
%new1 = OpFAdd %v4float %centroid %scaled1
%color1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%color1 = OpLoad %v3float %color1_ptr
%out_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos1 %new1
OpStore %outColor %color1
OpEmitVertex
%delta2 = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %delta2 %float_shrink
%new2 = OpFAdd %v4float %centroid %scaled2
%color2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%color2 = OpLoad %v3float %color2_ptr

OpStore %out_pos2 %new2
OpStore %outColor %color2
