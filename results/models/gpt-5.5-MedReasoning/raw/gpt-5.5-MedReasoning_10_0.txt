#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <vector>

using std::cin;
using std::cout;
using std::int64_t;
using std::vector;

struct Rect {
    int id;
    int w;
    int h;
    int64_t area;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct SkylineNode {
    int x;
    int y;
    int width;
};

struct Candidate {
    bool ok = false;
    int node = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    int waste = 0;
    int top = 0;
};

class Board {
public:
    Board(int width, int height) : W(width), H(height), remainingArea(int64_t(width) * height) {
        nodes.push_back({0, 0, width});
    }

    bool canPossiblyFit(const Rect& r) const {
        if (remainingArea < r.area) return false;
        return (r.w <= W && r.h <= H) || (r.h <= W && r.w <= H);
    }

    Candidate bestFor(const Rect& r) const {
        Candidate best;
        consider(r.w, r.h, 0, best);
        if (r.w != r.h) consider(r.h, r.w, 1, best);
        return best;
    }

    void place(const Candidate& c, const Rect& r) {
        addSkylineLevel(c.node, c.x, c.y, c.w, c.h);
        remainingArea -= r.area;
    }

    int64_t remaining() const {
        return remainingArea;
    }

private:
    int W;
    int H;
    int64_t remainingArea;
    vector<SkylineNode> nodes;

    bool fitAtNode(int index, int rw, int rh, int& outY, int& outWaste) const {
        const int x = nodes[index].x;
        if (x + rw > W) return false;

        int widthLeft = rw;
        int y = nodes[index].y;
        int i = index;
        while (widthLeft > 0) {
            if (i >= static_cast<int>(nodes.size())) return false;
            y = std::max(y, nodes[i].y);
            if (y + rh > H) return false;
            widthLeft -= nodes[i].width;
            ++i;
        }

        widthLeft = rw;
        i = index;
        int waste = 0;
        while (widthLeft > 0) {
            const int span = std::min(widthLeft, nodes[i].width);
            waste += span * (y - nodes[i].y);
            widthLeft -= span;
            ++i;
        }

        outY = y;
        outWaste = waste;
        return true;
    }

    void consider(int rw, int rh, int rotated, Candidate& best) const {
        if (rw > W || rh > H) return;

        for (int i = 0; i < static_cast<int>(nodes.size()); ++i) {
            int y = 0;
            int waste = 0;
            if (!fitAtNode(i, rw, rh, y, waste)) continue;

            Candidate c;
            c.ok = true;
            c.node = i;
            c.x = nodes[i].x;
            c.y = y;
            c.w = rw;
            c.h = rh;
            c.rotated = rotated;
            c.waste = waste;
            c.top = y + rh;

            if (!best.ok ||
                c.top < best.top ||
                (c.top == best.top && c.waste < best.waste) ||
                (c.top == best.top && c.waste == best.waste && c.y < best.y) ||
                (c.top == best.top && c.waste == best.waste && c.y == best.y && c.x < best.x)) {
                best = c;
            }
        }
    }

    void addSkylineLevel(int index, int x, int y, int rw, int rh) {
        nodes.insert(nodes.begin() + index, {x, y + rh, rw});

        for (int i = index + 1; i < static_cast<int>(nodes.size()); ++i) {
            SkylineNode& prev = nodes[i - 1];
            SkylineNode& cur = nodes[i];
            const int shrink = (prev.x + prev.width) - cur.x;
            if (shrink <= 0) break;

            cur.x += shrink;
            cur.width -= shrink;
            if (cur.width > 0) break;

            nodes.erase(nodes.begin() + i);
            --i;
        }

        for (int i = 0; i + 1 < static_cast<int>(nodes.size()); ++i) {
            if (nodes[i].y == nodes[i + 1].y) {
                nodes[i].width += nodes[i + 1].width;
                nodes.erase(nodes.begin() + i + 1);
                --i;
            }
        }
    }
};

static int searchLimitFor(int n) {
    if (n <= 2000) return std::numeric_limits<int>::max();
    if (n <= 10000) return 512;
    if (n <= 30000) return 192;
    return 96;
}

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N = 0;
    int boardW = 0;
    int boardH = 0;
    if (!(cin >> N >> boardW >> boardH)) return 0;

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w = 0;
        int h = 0;
        cin >> w >> h;
        rects.push_back({i, w, h, int64_t(w) * h});
    }

    std::stable_sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        const int aLong = std::max(a.w, a.h);
        const int bLong = std::max(b.w, b.h);
        const int aShort = std::min(a.w, a.h);
        const int bShort = std::min(b.w, b.h);
        if (a.area != b.area) return a.area > b.area;
        if (aLong != bLong) return aLong > bLong;
        return aShort > bShort;
    });

    vector<Board> boards;
    vector<Placement> answer(N);
    const int boardSearchLimit = searchLimitFor(N);

    for (const Rect& r : rects) {
        int bestBoard = -1;
        Candidate bestCandidate;

        const int boardCount = static_cast<int>(boards.size());
        int firstBoard = 0;
        if (boardSearchLimit != std::numeric_limits<int>::max() && boardCount > boardSearchLimit) {
            firstBoard = boardCount - boardSearchLimit;
        }

        for (int b = firstBoard; b < boardCount; ++b) {
            if (!boards[b].canPossiblyFit(r)) continue;

            Candidate c = boards[b].bestFor(r);
            if (!c.ok) continue;

            if (bestBoard == -1 ||
                c.top < bestCandidate.top ||
                (c.top == bestCandidate.top && c.waste < bestCandidate.waste) ||
                (c.top == bestCandidate.top && c.waste == bestCandidate.waste && boards[b].remaining() < boards[bestBoard].remaining())) {
                bestBoard = b;
                bestCandidate = c;
            }
        }

        if (bestBoard == -1) {
            boards.emplace_back(boardW, boardH);
            bestBoard = static_cast<int>(boards.size()) - 1;
            bestCandidate = boards.back().bestFor(r);
        }

        boards[bestBoard].place(bestCandidate, r);
        answer[r.id] = {bestBoard, bestCandidate.x, bestCandidate.y, bestCandidate.rotated};
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << answer[i].board << ' '
             << answer[i].x << ' '
             << answer[i].y << ' '
             << answer[i].rotated << '\n';
    }

    return 0;
}
