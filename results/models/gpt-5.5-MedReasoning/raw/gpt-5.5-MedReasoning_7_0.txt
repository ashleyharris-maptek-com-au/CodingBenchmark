using std::vector;

static constexpr double EPS = 1e-9;

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec3 operator/(double s) const { return {x / s, y / s, z / s}; }
};

static double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

static double norm(const Vec3& a) {
    return std::sqrt(dot(a, a));
}

static Vec3 normalize(const Vec3& a) {
    double n = norm(a);
    return n <= EPS ? Vec3{} : a / n;
}

struct Plane {
    Vec3 n;
    double w{};

    Plane() = default;
    Plane(const Vec3& a, const Vec3& b, const Vec3& c) {
        n = normalize(cross(b - a, c - a));
        w = dot(n, a);
    }

    void flip() {
        n = n * -1.0;
        w = -w;
    }
};

struct Polygon {
    vector<Vec3> v;
    Plane plane;

    Polygon() = default;
    explicit Polygon(vector<Vec3> verts) : v(std::move(verts)) {
        if (v.size() >= 3) plane = Plane(v[0], v[1], v[2]);
    }

    void flip() {
        std::reverse(v.begin(), v.end());
        plane.flip();
    }
};

static void splitPolygon(const Plane& pl,
                         const Polygon& poly,
                         vector<Polygon>& coplanarFront,
                         vector<Polygon>& coplanarBack,
                         vector<Polygon>& front,
                         vector<Polygon>& back) {
    enum { COPLANAR = 0, FRONT = 1, BACK = 2, SPANNING = 3 };
    int polygonType = 0;
    vector<int> types(poly.v.size());

    for (size_t i = 0; i < poly.v.size(); ++i) {
        double t = dot(pl.n, poly.v[i]) - pl.w;
        int type = (t < -EPS) ? BACK : (t > EPS ? FRONT : COPLANAR);
        polygonType |= type;
        types[i] = type;
    }

    if (polygonType == COPLANAR) {
        if (dot(pl.n, poly.plane.n) > 0.0) coplanarFront.push_back(poly);
        else coplanarBack.push_back(poly);
    } else if (polygonType == FRONT) {
        front.push_back(poly);
    } else if (polygonType == BACK) {
        back.push_back(poly);
    } else {
        vector<Vec3> f, b;
        for (size_t i = 0; i < poly.v.size(); ++i) {
            size_t j = (i + 1) % poly.v.size();
            int ti = types[i], tj = types[j];
            const Vec3& vi = poly.v[i];
            const Vec3& vj = poly.v[j];

            if (ti != BACK) f.push_back(vi);
            if (ti != FRONT) b.push_back(vi);
            if ((ti | tj) == SPANNING) {
                double denom = dot(pl.n, vj - vi);
                if (std::abs(denom) > EPS) {
                    double t = (pl.w - dot(pl.n, vi)) / denom;
                    Vec3 p = vi + (vj - vi) * t;
                    f.push_back(p);
                    b.push_back(p);
                }
            }
        }
        if (f.size() >= 3) front.emplace_back(std::move(f));
        if (b.size() >= 3) back.emplace_back(std::move(b));
    }
}

class BSPNode {
public:
    Plane plane;
    bool hasPlane = false;
    vector<Polygon> polygons;
    std::unique_ptr<BSPNode> front;
    std::unique_ptr<BSPNode> back;

    BSPNode() = default;
    explicit BSPNode(const vector<Polygon>& list) { build(list); }

    vector<Polygon> allPolygons() const {
        vector<Polygon> result = polygons;
        if (front) {
            vector<Polygon> f = front->allPolygons();
            result.insert(result.end(), f.begin(), f.end());
        }
        if (back) {
            vector<Polygon> b = back->allPolygons();
            result.insert(result.end(), b.begin(), b.end());
        }
        return result;
    }

    void invert() {
        for (auto& p : polygons) p.flip();
        plane.flip();
        if (front) front->invert();
        if (back) back->invert();
        std::swap(front, back);
    }

    vector<Polygon> clipPolygons(const vector<Polygon>& list) const {
        if (!hasPlane) return list;
        vector<Polygon> f, b;
        for (const auto& p : list) {
            vector<Polygon> cf, cb;
            splitPolygon(plane, p, cf, cb, f, b);
            f.insert(f.end(), cf.begin(), cf.end());
            b.insert(b.end(), cb.begin(), cb.end());
        }
        if (front) f = front->clipPolygons(f);
        if (back) b = back->clipPolygons(b);
        else b.clear();
        f.insert(f.end(), b.begin(), b.end());
        return f;
    }

    void clipTo(const BSPNode& other) {
        polygons = other.clipPolygons(polygons);
        if (front) front->clipTo(other);
        if (back) back->clipTo(other);
    }

    void build(const vector<Polygon>& list) {
        if (list.empty()) return;
        if (!hasPlane) {
            plane = list[0].plane;
            hasPlane = true;
        }
        vector<Polygon> f, b;
        for (const auto& p : list) {
            splitPolygon(plane, p, polygons, polygons, f, b);
        }
        if (!f.empty()) {
            if (!front) front = std::make_unique<BSPNode>();
            front->build(f);
        }
        if (!b.empty()) {
            if (!back) back = std::make_unique<BSPNode>();
            back->build(b);
        }
    }
};

struct Mesh {
    vector<Vec3> vertices;
    vector<std::array<int, 3>> faces;
};

struct AABB {
    Vec3 mn{std::numeric_limits<double>::infinity(),
            std::numeric_limits<double>::infinity(),
            std::numeric_limits<double>::infinity()};
    Vec3 mx{-std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity()};
};

static AABB bounds(const Mesh& m) {
    AABB b;
    for (const auto& p : m.vertices) {
        b.mn.x = std::min(b.mn.x, p.x); b.mn.y = std::min(b.mn.y, p.y); b.mn.z = std::min(b.mn.z, p.z);
        b.mx.x = std::max(b.mx.x, p.x); b.mx.y = std::max(b.mx.y, p.y); b.mx.z = std::max(b.mx.z, p.z);
    }
    return b;
}

static bool disjoint(const AABB& a, const AABB& b) {
    return a.mx.x < b.mn.x - EPS || b.mx.x < a.mn.x - EPS ||
           a.mx.y < b.mn.y - EPS || b.mx.y < a.mn.y - EPS ||
           a.mx.z < b.mn.z - EPS || b.mx.z < a.mn.z - EPS;
}

static bool rayIntersectsTriangle(const Vec3& orig, const Vec3& dir,
                                  const Vec3& a, const Vec3& b, const Vec3& c,
                                  double& t) {
    Vec3 e1 = b - a, e2 = c - a;
    Vec3 p = cross(dir, e2);
    double det = dot(e1, p);
    if (std::abs(det) < EPS) return false;
    double inv = 1.0 / det;
    Vec3 s = orig - a;
    double u = dot(s, p) * inv;
    if (u < -EPS || u > 1.0 + EPS) return false;
    Vec3 q = cross(s, e1);
    double v = dot(dir, q) * inv;
    if (v < -EPS || u + v > 1.0 + EPS) return false;
    t = dot(e2, q) * inv;
    return t > EPS;
}

static bool pointInsideMesh(const Vec3& p, const Mesh& m) {
    Vec3 dir{0.87654321, 0.23456789, 0.41975311};
    int hits = 0;
    for (const auto& f : m.faces) {
        double t = 0.0;
        if (rayIntersectsTriangle(p, dir, m.vertices[f[0]], m.vertices[f[1]], m.vertices[f[2]], t)) {
            ++hits;
        }
    }
    return (hits & 1) != 0;
}

static vector<Polygon> toPolygons(const Mesh& m) {
    vector<Polygon> out;
    out.reserve(m.faces.size());
    for (const auto& f : m.faces) {
        out.emplace_back(vector<Vec3>{m.vertices[f[0]], m.vertices[f[1]], m.vertices[f[2]]});
    }
    return out;
}

static Mesh concatenate(const Mesh& a, const Mesh& b) {
    Mesh r = a;
    int off = static_cast<int>(r.vertices.size());
    r.vertices.insert(r.vertices.end(), b.vertices.begin(), b.vertices.end());
    r.faces.reserve(a.faces.size() + b.faces.size());
    for (auto f : b.faces) r.faces.push_back({f[0] + off, f[1] + off, f[2] + off});
    return r;
}

struct Key {
    long long x, y, z;
    bool operator==(const Key& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct KeyHash {
    size_t operator()(const Key& k) const {
        uint64_t h = 1469598103934665603ull;
        auto mix = [&](long long v) {
            uint64_t x = static_cast<uint64_t>(v);
            h ^= x; h *= 1099511628211ull;
        };
        mix(k.x); mix(k.y); mix(k.z);
        return static_cast<size_t>(h);
    }
};

static Key makeKey(const Vec3& p) {
    const double s = 1e9;
    return {static_cast<long long>(std::llround(p.x * s)),
            static_cast<long long>(std::llround(p.y * s)),
            static_cast<long long>(std::llround(p.z * s))};
}

static Mesh fromPolygons(const vector<Polygon>& polys) {
    Mesh m;
    std::unordered_map<Key, int, KeyHash> ids;
    auto add = [&](const Vec3& p) {
        Key k = makeKey(p);
        auto it = ids.find(k);
        if (it != ids.end()) return it->second;
        int id = static_cast<int>(m.vertices.size());
        ids.emplace(k, id);
        m.vertices.push_back(p);
        return id;
    };

    for (const auto& p : polys) {
        if (p.v.size() < 3) continue;
        int i0 = add(p.v[0]);
        for (size_t i = 1; i + 1 < p.v.size(); ++i) {
            int i1 = add(p.v[i]);
            int i2 = add(p.v[i + 1]);
            if (i0 != i1 && i1 != i2 && i2 != i0) m.faces.push_back({i0, i1, i2});
        }
    }
    return m;
}

static Mesh csgUnion(const Mesh& a, const Mesh& b) {
    AABB ab = bounds(a), bb = bounds(b);
    if (disjoint(ab, bb)) return concatenate(a, b);

    if (!a.vertices.empty() && !b.vertices.empty()) {
        bool a0inb = pointInsideMesh(a.vertices[0], b);
        bool b0ina = pointInsideMesh(b.vertices[0], a);
        if (a0inb && !b0ina &&
            ab.mn.x >= bb.mn.x - EPS && ab.mn.y >= bb.mn.y - EPS && ab.mn.z >= bb.mn.z - EPS &&
            ab.mx.x <= bb.mx.x + EPS && ab.mx.y <= bb.mx.y + EPS && ab.mx.z <= bb.mx.z + EPS) {
            return b;
        }
        if (b0ina && !a0inb &&
            bb.mn.x >= ab.mn.x - EPS && bb.mn.y >= ab.mn.y - EPS && bb.mn.z >= ab.mn.z - EPS &&
            bb.mx.x <= ab.mx.x + EPS && bb.mx.y <= ab.mx.y + EPS && bb.mx.z <= ab.mx.z + EPS) {
            return a;
        }
    }

    BSPNode na(toPolygons(a));
    BSPNode nb(toPolygons(b));
    na.clipTo(nb);
    nb.clipTo(na);
    nb.invert();
    nb.clipTo(na);
    nb.invert();
    vector<Polygon> bp = nb.allPolygons();
    na.build(bp);
    return fromPolygons(na.allPolygons());
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.vertices.resize(VA);
    A.faces.resize(FA);
    B.vertices.resize(VB);
    B.faces.resize(FB);

    for (auto& p : A.vertices) std::cin >> p.x >> p.y >> p.z;
    for (auto& f : A.faces) std::cin >> f[0] >> f[1] >> f[2];
    for (auto& p : B.vertices) std::cin >> p.x >> p.y >> p.z;
    for (auto& f : B.faces) std::cin >> f[0] >> f[1] >> f[2];

    Mesh U = csgUnion(A, B);

    std::cout << U.vertices.size() << ' ' << U.faces.size() << '\n';
    std::cout << std::setprecision(17);
    for (const auto& p : U.vertices) {
        std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    }
    for (const auto& f : U.faces) {
        std::cout << f[0] << ' ' << f[1] << ' ' << f[2] << '\n';
    }
    return 0;
}
