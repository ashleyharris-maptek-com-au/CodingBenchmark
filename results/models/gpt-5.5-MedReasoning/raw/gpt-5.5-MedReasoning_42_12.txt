OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%fn_main = OpTypeFunction %void
%fn_emit_edge = OpTypeFunction %void %v4float %v4float
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1
%white = OpConstantComposite %v3float %float_1 %float_1 %float_1
%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%emitEdge = OpFunction %void None %fn_emit_edge
%edge_a = OpFunctionParameter %v4float
%edge_b = OpFunctionParameter %v4float
%edge_entry = OpLabel
%a_xy = OpVectorShuffle %v2float %edge_a %edge_a 0 1
%b_xy = OpVectorShuffle %v2float %edge_b %edge_b 0 1
%a_w = OpCompositeExtract %float %edge_a 3
%b_w = OpCompositeExtract %float %edge_b 3
%a_w2 = OpCompositeConstruct %v2float %a_w %a_w
%b_w2 = OpCompositeConstruct %v2float %b_w %b_w
%a_ndc = OpFDiv %v2float %a_xy %a_w2
%b_ndc = OpFDiv %v2float %b_xy %b_w2
%edge_delta = OpFSub %v2float %b_ndc %a_ndc
%edge_dir = OpExtInst %v2float %glsl Normalize %edge_delta
%dir_x = OpCompositeExtract %float %edge_dir 0
%dir_y = OpCompositeExtract %float %edge_dir 1
%neg_dir_y = OpFNegate %float %dir_y
%edge_perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%a_scale = OpFMul %float %a_w %float_0_015
%b_scale = OpFMul %float %b_w %float_0_015
%a_off2 = OpVectorTimesScalar %v2float %edge_perp %a_scale
%b_off2 = OpVectorTimesScalar %v2float %edge_perp %b_scale
%a_off_x = OpCompositeExtract %float %a_off2 0
%a_off_y = OpCompositeExtract %float %a_off2 1
%b_off_x = OpCompositeExtract %float %b_off2 0
%b_off_y = OpCompositeExtract %float %b_off2 1
%a_off4 = OpCompositeConstruct %v4float %a_off_x %a_off_y %float_0 %float_0
%b_off4 = OpCompositeConstruct %v4float %b_off_x %b_off_y %float_0 %float_0
%a_plus = OpFAdd %v4float %edge_a %a_off4
%a_minus = OpFSub %v4float %edge_a %a_off4
%b_plus = OpFAdd %v4float %edge_b %b_off4
%b_minus = OpFSub %v4float %edge_b %b_off4
%edge_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %edge_pos0 %a_plus
OpStore %outColor %white
OpEmitVertex
%edge_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %edge_pos1 %a_minus
OpStore %outColor %white
OpEmitVertex
%edge_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %edge_pos2 %b_plus
OpStore %outColor %white
OpEmitVertex
%edge_pos3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %edge_pos3 %b_minus
OpStore %outColor %white
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd

%main = OpFunction %void None %fn_main
%main_entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2 = OpLoad %v4float %p2_ptr
%p4 = OpLoad %v4float %p4_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%c0_dim = OpVectorTimesScalar %v3float %c0 %float_0_3
%c2_dim = OpVectorTimesScalar %v3float %c2 %float_0_3
%c4_dim = OpVectorTimesScalar %v3float %c4 %float_0_3
%tri_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %tri_pos0 %p0
OpStore %outColor %c0_dim
OpEmitVertex
%tri_pos1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %tri_pos1 %p2
OpStore %outColor %c2_dim
OpEmitVertex
%tri_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %tri_pos2 %p4
OpStore %outColor %c4_dim
OpEmitVertex
OpEndPrimitive
%call_edge_02 = OpFunctionCall %void %emitEdge %p0 %p2
%call_edge_24 = OpFunctionCall %void %emitEdge %p2 %p4
%call_edge_40 = OpFunctionCall %void %emitEdge %p4 %p0
OpReturn
OpFunctionEnd
