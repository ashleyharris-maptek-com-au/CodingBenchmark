#include <algorithm>
#include <cstdint>
#include <cstdlib>
#include <deque>
#include <initializer_list>
#include <iostream>
#include <limits>
#include <queue>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using Coord = std::vector<int>;

struct CoordHash {
    std::size_t operator()(const Coord& v) const noexcept {
        std::uint64_t h = 1469598103934665603ull;
        for (int x : v) {
            std::uint64_t y = static_cast<std::uint64_t>(static_cast<std::int64_t>(x) + 0x9e3779b97f4a7c15ll);
            h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
        }
        return static_cast<std::size_t>(h);
    }
};

struct Move {
    int axis = 0;
    int dir = 1;
};

static long long manhattan(const Coord& a, const Coord& b) {
    long long s = 0;
    for (std::size_t i = 0; i < a.size(); ++i) {
        s += std::llabs(static_cast<long long>(a[i]) - b[i]);
    }
    return s;
}

static bool in_bounds(const Coord& p, const std::vector<int>& bounds) {
    for (std::size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static Coord stepped(Coord p, int axis, int dir) {
    p[axis] += dir;
    return p;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D = 0;
    long long max_turns = 0;
    if (!(std::cin >> D >> max_turns) || D <= 0 || max_turns <= 0) return 0;

    std::vector<int> bounds(D);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snake_length = 1;
    Coord head(D);
    std::cin >> snake_length;
    for (int i = 0; i < D; ++i) std::cin >> head[i];
    if (snake_length < 1) snake_length = 1;

    int F = 0;
    std::cin >> F;
    std::vector<Coord> foods;
    foods.reserve(std::max(0, F));
    std::unordered_set<Coord, CoordHash> food_set;
    for (int i = 0; i < F; ++i) {
        Coord f(D);
        for (int j = 0; j < D; ++j) std::cin >> f[j];
        if (in_bounds(f, bounds) && food_set.insert(f).second) foods.push_back(f);
    }

    int O = 0;
    std::cin >> O;
    std::unordered_set<Coord, CoordHash> obstacles;
    obstacles.reserve(static_cast<std::size_t>(std::max(0, O)) * 2 + 1);
    for (int i = 0; i < O; ++i) {
        Coord o(D);
        for (int j = 0; j < D; ++j) std::cin >> o[j];
        if (in_bounds(o, bounds)) obstacles.insert(o);
    }

    if (!in_bounds(head, bounds) || obstacles.count(head)) return 0;

    std::deque<Coord> body;
    std::unordered_map<Coord, int, CoordHash> occ;
    for (int i = 0; i < snake_length; ++i) {
        body.push_back(head);
        ++occ[head];
    }

    if (food_set.erase(head) != 0) {
        foods.erase(std::remove(foods.begin(), foods.end(), head), foods.end());
    }

    auto occupied_after_tail_moves = [&](const Coord& p, bool grows) -> bool {
        auto it = occ.find(p);
        if (it == occ.end() || it->second <= 0) return false;
        if (!grows && !body.empty() && p == body.back() && it->second == 1) return false;
        return true;
    };

    auto legal_move = [&](const Move& m) -> bool {
        Coord n = stepped(head, m.axis, m.dir);
        if (!in_bounds(n, bounds)) return false;
        if (obstacles.count(n)) return false;
        bool grows = food_set.count(n) != 0;
        return !occupied_after_tail_moves(n, grows);
    };

    auto apply_move = [&](const Move& m) {
        Coord n = stepped(head, m.axis, m.dir);
        bool grows = food_set.count(n) != 0;
        body.push_front(n);
        ++occ[n];
        head = n;
        if (grows) {
            food_set.erase(n);
            foods.erase(std::remove(foods.begin(), foods.end(), n), foods.end());
        } else {
            Coord tail = body.back();
            body.pop_back();
            auto it = occ.find(tail);
            if (it != occ.end()) {
                --it->second;
                if (it->second == 0) occ.erase(it);
            }
        }
    };

    auto board_volume_small = [&]() -> bool {
        const long long limit = 750000;
        long long v = 1;
        for (int b : bounds) {
            if (b <= 0) return false;
            if (v > limit / b) return false;
            v *= b;
        }
        return v <= limit;
    };

    auto bfs_move = [&]() -> std::pair<bool, Move> {
        if (food_set.empty() || !board_volume_small()) return {false, Move{}};

        struct Parent {
            Coord prev;
            Move move;
            bool has_prev = false;
        };

        std::queue<Coord> q;
        std::unordered_map<Coord, Parent, CoordHash> parent;
        q.push(head);
        parent.emplace(head, Parent{});

        while (!q.empty()) {
            Coord cur = q.front();
            q.pop();

            if (cur != head && food_set.count(cur)) {
                Coord at = cur;
                Move first;
                while (true) {
                    const Parent& p = parent[at];
                    if (p.prev == head) {
                        first = p.move;
                        break;
                    }
                    at = p.prev;
                }
                return {true, first};
            }

            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    Coord nxt = stepped(cur, axis, dir);
                    if (!in_bounds(nxt, bounds) || obstacles.count(nxt)) continue;
                    if (parent.find(nxt) != parent.end()) continue;
                    bool grows = food_set.count(nxt) != 0;
                    if (nxt != head && occupied_after_tail_moves(nxt, grows)) continue;
                    Parent p;
                    p.prev = cur;
                    p.move = Move{axis, dir};
                    p.has_prev = true;
                    parent.emplace(nxt, std::move(p));
                    q.push(nxt);
                }
            }
        }
        return {false, Move{}};
    };

    auto heuristic_move = [&]() -> std::pair<bool, Move> {
        std::vector<Move> moves;
        moves.reserve(static_cast<std::size_t>(2 * D));
        for (int axis = 0; axis < D; ++axis) {
            moves.push_back(Move{axis, -1});
            moves.push_back(Move{axis, 1});
        }

        Coord target;
        bool have_target = false;
        long long best_dist = std::numeric_limits<long long>::max();
        for (const Coord& f : foods) {
            long long d = manhattan(head, f);
            if (d < best_dist) {
                best_dist = d;
                target = f;
                have_target = true;
            }
        }

        std::stable_sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            Coord na = stepped(head, a.axis, a.dir);
            Coord nb = stepped(head, b.axis, b.dir);
            long long sa = have_target ? manhattan(na, target) : 0;
            long long sb = have_target ? manhattan(nb, target) : 0;
            if (sa != sb) return sa < sb;

            int free_a = 0, free_b = 0;
            for (int axis = 0; axis < D; ++axis) {
                for (int dir : {-1, 1}) {
                    Coord aa = stepped(na, axis, dir);
                    Coord bb = stepped(nb, axis, dir);
                    bool ga = food_set.count(aa) != 0;
                    bool gb = food_set.count(bb) != 0;
                    if (in_bounds(aa, bounds) && !obstacles.count(aa) && !occupied_after_tail_moves(aa, ga)) ++free_a;
                    if (in_bounds(bb, bounds) && !obstacles.count(bb) && !occupied_after_tail_moves(bb, gb)) ++free_b;
                }
            }
            if (free_a != free_b) return free_a > free_b;
            return std::make_pair(a.axis, a.dir) < std::make_pair(b.axis, b.dir);
        });

        for (const Move& m : moves) {
            if (legal_move(m)) return {true, m};
        }
        return {false, Move{}};
    };

    long long turns = 0;
    while (turns < max_turns && !food_set.empty()) {
        std::pair<bool, Move> chosen = bfs_move();
        if (!chosen.first) chosen = heuristic_move();
        if (!chosen.first) break;

        std::cout << chosen.second.axis << ' ' << chosen.second.dir << '\n';
        apply_move(chosen.second);
        ++turns;
    }

    return 0;
}
