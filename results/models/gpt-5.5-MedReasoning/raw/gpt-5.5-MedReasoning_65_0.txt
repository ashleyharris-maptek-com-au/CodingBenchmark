declare const require: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data = JSON.parse(input);

type RawTask = {
  id: any;
  duration: number;
  deps?: any[];
  deadline?: number;
  weight?: number;
};

const tasks: RawTask[] = data.tasks || [];
const n = tasks.length;
const m = Math.max(1, Math.floor(Number(data.machineCount || 1)));

const idToIndex = new Map<any, number>();
for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

const duration = new Array<number>(n);
const deadline = new Array<number>(n);
const weight = new Array<number>(n);
const succ: number[][] = Array.from({ length: n }, () => []);
const pred: number[][] = Array.from({ length: n }, () => []);
const indeg0 = new Array<number>(n).fill(0);

for (let i = 0; i < n; i++) {
  duration[i] = Math.max(0, Math.floor(Number(tasks[i].duration || 0)));
  const d = Number(tasks[i].deadline);
  deadline[i] = Number.isFinite(d) ? d : 0;
  const w = Number(tasks[i].weight);
  weight[i] = Number.isFinite(w) && w > 0 ? w : 1;
}

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps! : [];
  for (const depId of deps) {
    const j = idToIndex.get(depId);
    if (j === undefined) continue;
    succ[j].push(i);
    pred[i].push(j);
    indeg0[i]++;
  }
}

function topologicalOrder(): number[] {
  const indeg = indeg0.slice();
  const q: number[] = [];
  let head = 0;
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  while (head < q.length) {
    const u = q[head++];
    for (const v of succ[u]) if (--indeg[v] === 0) q.push(v);
  }
  return q;
}

const topo = topologicalOrder();
const tail = duration.slice();
const downstream = weight.slice();
for (let p = topo.length - 1; p >= 0; p--) {
  const u = topo[p];
  let bestTail = 0;
  let down = weight[u];
  for (const v of succ[u]) {
    if (tail[v] > bestTail) bestTail = tail[v];
    down += 0.35 * downstream[v];
  }
  tail[u] = duration[u] + bestTail;
  downstream[u] = Math.min(1e15, down);
}

const latestStart = new Array<number>(n);
const dueMinusProc = new Array<number>(n);
const ratio = new Array<number>(n);
for (let i = 0; i < n; i++) {
  latestStart[i] = deadline[i] - tail[i];
  dueMinusProc[i] = deadline[i] - duration[i];
  ratio[i] = duration[i] / weight[i];
}

class ReadyHeap {
  a: number[] = [];
  mode: number;
  constructor(mode: number) { this.mode = mode; }
  better(x: number, y: number): boolean {
    const mx = this.metric(x), my = this.metric(y);
    if (mx !== my) return mx < my;
    if (deadline[x] !== deadline[y]) return deadline[x] < deadline[y];
    if (tail[x] !== tail[y]) return tail[x] > tail[y];
    if (weight[x] !== weight[y]) return weight[x] > weight[y];
    return x < y;
  }
  metric(i: number): number {
    switch (this.mode) {
      case 0: return deadline[i];
      case 1: return latestStart[i];
      case 2: return -tail[i];
      case 3: return ratio[i];
      case 4: return dueMinusProc[i] - 2 * tail[i] - 4 * Math.log1p(downstream[i]);
      case 5: return deadline[i] - 1.5 * tail[i] - 6 * Math.log1p(weight[i]);
      default: return latestStart[i];
    }
  }
  push(x: number) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.better(a[i], a[p])) break;
      const t = a[i]; a[i] = a[p]; a[p] = t; i = p;
    }
  }
  pop(): number {
    const a = this.a;
    const r = a[0];
    const x = a.pop()!;
    if (a.length) {
      a[0] = x;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1, rr = l + 1, b = i;
        if (l < a.length && this.better(a[l], a[b])) b = l;
        if (rr < a.length && this.better(a[rr], a[b])) b = rr;
        if (b === i) break;
        const t = a[i]; a[i] = a[b]; a[b] = t; i = b;
      }
    }
    return r;
  }
  get length(): number { return this.a.length; }
}

type RunItem = { finish: number; task: number; machine: number };
class RunHeap {
  a: RunItem[] = [];
  better(x: RunItem, y: RunItem): boolean {
    return x.finish !== y.finish ? x.finish < y.finish : x.machine < y.machine;
  }
  push(x: RunItem) {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!this.better(a[i], a[p])) break;
      const t = a[i]; a[i] = a[p]; a[p] = t; i = p;
    }
  }
  pop(): RunItem {
    const a = this.a;
    const r = a[0];
    const x = a.pop()!;
    if (a.length) {
      a[0] = x;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1, rr = l + 1, b = i;
        if (l < a.length && this.better(a[l], a[b])) b = l;
        if (rr < a.length && this.better(a[rr], a[b])) b = rr;
        if (b === i) break;
        const t = a[i]; a[i] = a[b]; a[b] = t; i = b;
      }
    }
    return r;
  }
  peek(): RunItem { return this.a[0]; }
  get length(): number { return this.a.length; }
}

function buildSchedule(mode: number) {
  const indeg = indeg0.slice();
  const ready = new ReadyHeap(mode);
  const running = new RunHeap();
  const idle: number[] = [];
  const starts = new Array<number>(n).fill(0);
  const mach = new Array<number>(n).fill(0);
  const finish = new Array<number>(n).fill(0);
  const scheduled: number[] = [];

  for (let i = 0; i < Math.min(m, Math.max(1, n)); i++) idle.push(i);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push(i);

  let time = 0;
  let done = 0;
  while (scheduled.length < n) {
    while (idle.length && ready.length) {
      const t = ready.pop();
      const machine = idle.pop()!;
      starts[t] = time;
      mach[t] = machine;
      finish[t] = time + duration[t];
      scheduled.push(t);
      running.push({ finish: finish[t], task: t, machine });
    }
    if (scheduled.length === n) break;
    if (!running.length) break;
    const nextTime = running.peek().finish;
    time = nextTime;
    while (running.length && running.peek().finish === nextTime) {
      const r = running.pop();
      idle.push(r.machine);
      done++;
      for (const v of succ[r.task]) if (--indeg[v] === 0) ready.push(v);
    }
  }

  let makespan = 0;
  let weightedLateness = 0;
  for (let i = 0; i < n; i++) {
    if (finish[i] > makespan) makespan = finish[i];
    const late = finish[i] - deadline[i];
    if (late > 0) weightedLateness += late * weight[i];
  }
  const score = weightedLateness + makespan;
  return { starts, mach, finish, score, scheduledCount: scheduled.length, done };
}

let best = buildSchedule(0);
for (let mode = 1; mode <= 5; mode++) {
  const s = buildSchedule(mode);
  if (s.scheduledCount > best.scheduledCount || (s.scheduledCount === best.scheduledCount && s.score < best.score)) best = s;
}

if (best.scheduledCount < n && topo.length === n) {
  console.log(JSON.stringify({ order: topo.map(i => tasks[i].id) }));
} else {
  const schedule = [];
  for (let i = 0; i < n; i++) schedule.push({ id: tasks[i].id, start: best.starts[i], machine: best.mach[i] });
  schedule.sort((a, b) => a.start - b.start || a.machine - b.machine || String(a.id).localeCompare(String(b.id)));
  console.log(JSON.stringify({ schedule }));
}
