OpCapability Geometry
%std450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_0_333333343 = OpConstant %float 0.333333343
%float_1 = OpConstant %float 1
%vec_one = OpConstantComposite %v3float %float_1 %float_1 %float_1
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %func
%entry = OpLabel
%wp0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%wp2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%wp4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p0 = OpLoad %v3float %wp0_ptr
%p2 = OpLoad %v3float %wp2_ptr
%p4 = OpLoad %v3float %wp4_ptr
%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0
%ax = OpCompositeExtract %float %a 0
%ay = OpCompositeExtract %float %a 1
%az = OpCompositeExtract %float %a 2
%bx = OpCompositeExtract %float %b 0
%by = OpCompositeExtract %float %b 1
%bz = OpCompositeExtract %float %b 2
%ay_bz = OpFMul %float %ay %bz
%az_by = OpFMul %float %az %by
%cross_x = OpFSub %float %ay_bz %az_by
%az_bx = OpFMul %float %az %bx
%ax_bz = OpFMul %float %ax %bz
%cross_y = OpFSub %float %az_bx %ax_bz
%ax_by = OpFMul %float %ax %by
%ay_bx = OpFMul %float %ay %bx
%cross_z = OpFSub %float %ax_by %ay_bx
%cross = OpCompositeConstruct %v3float %cross_x %cross_y %cross_z
%N = OpExtInst %v3float %std450 Normalize %cross
%L = OpExtInst %v3float %std450 Normalize %vec_one
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %std450 FMax %ndotl %float_0
%diffuse_scaled = OpFMul %float %float_0_8 %diffuse
%light_factor = OpFAdd %float %float_0_2 %diffuse_scaled
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%color_sum_02 = OpFAdd %v3float %c0 %c2
%color_sum = OpFAdd %v3float %color_sum_02 %c4
%base_color = OpVectorTimesScalar %v3float %color_sum %float_0_333333343
%final_color = OpVectorTimesScalar %v3float %base_color %light_factor
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%in_pos0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %in_pos0_ptr
OpStore %out_pos_ptr %pos0
OpStore %outColor %final_color
OpEmitVertex
%in_pos2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %in_pos2_ptr
OpStore %out_pos_ptr %pos2
OpStore %outColor %final_color
OpEmitVertex
%in_pos4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %in_pos4_ptr
OpStore %out_pos_ptr %pos4
OpStore %outColor %final_color
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
