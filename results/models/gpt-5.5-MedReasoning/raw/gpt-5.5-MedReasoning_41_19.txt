; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main OriginUpperLeft

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_0_05 = OpConstant %float 0.05
%float_0_1 = OpConstant %float 0.1
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_32 = OpConstant %float 32.0
%v3_zero = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v3_one = OpConstantComposite %v3float %float_1 %float_1 %float_1
%ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05

%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %fn_void
%entry = OpLabel
%pos = OpLoad %v3float %worldPos
%normal_in = OpLoad %v3float %normal
%mat_color = OpLoad %v3float %color
%light_ptr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %light_ptr
%camera_ptr = OpAccessChain %ptr_Uniform_v4float %ubo %int_4
%camera4 = OpLoad %v4float %camera_ptr
%light_xyz = OpVectorShuffle %v3float %light4 %light4 0 1 2
%camera_xyz = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%N = OpExtInst %v3float %glsl Normalize %normal_in
%light_delta = OpFSub %v3float %light_xyz %pos
%L = OpExtInst %v3float %glsl Normalize %light_delta
%camera_delta = OpFSub %v3float %camera_xyz %pos
%V = OpExtInst %v3float %glsl Normalize %camera_delta
%dot_nl_raw = OpDot %float %N %L
%dot_nl = OpExtInst %float %glsl FMax %dot_nl_raw %float_0
%diffuse = OpVectorTimesScalar %v3float %mat_color %dot_nl
%neg_l = OpFNegate %v3float %L
%two_dot_nl = OpFMul %float %float_2 %dot_nl_raw
%reflect_term = OpVectorTimesScalar %v3float %N %two_dot_nl
%reflect_raw = OpFAdd %v3float %neg_l %reflect_term
%R = OpExtInst %v3float %glsl Normalize %reflect_raw
%dot_rv_raw = OpDot %float %R %V
%dot_rv = OpExtInst %float %glsl FMax %dot_rv_raw %float_0
%spec = OpExtInst %float %glsl Pow %dot_rv %float_32
%specular = OpVectorTimesScalar %v3float %v3_one %spec
%amb_diff = OpFAdd %v3float %ambient %diffuse
%sum = OpFAdd %v3float %amb_diff %specular
%clamped = OpExtInst %v3float %glsl FClamp %sum %v3_zero %v3_one
%out_r = OpCompositeExtract %float %clamped 0
%out_g = OpCompositeExtract %float %clamped 1
%out_b = OpCompositeExtract %float %clamped 2
%final = OpCompositeConstruct %v4float %out_r %out_g %out_b %float_1
OpStore %outColor %final
OpReturn
OpFunctionEnd
