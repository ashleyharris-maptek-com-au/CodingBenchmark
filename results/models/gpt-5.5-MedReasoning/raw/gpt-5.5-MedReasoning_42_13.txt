OpCapability Geometry
%ext = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%fn_void = OpTypeFunction %void
%fn_emit_edge = OpTypeFunction %void %v4float %v4float %v3float %v3float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_1 = OpConstant %float 1
%float_half = OpConstant %float 0.5
%float_half_width = OpConstant %float 0.004
%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%emitEdge = OpFunction %void None %fn_emit_edge
%pa = OpFunctionParameter %v4float
%pb = OpFunctionParameter %v4float
%ca = OpFunctionParameter %v3float
%cb = OpFunctionParameter %v3float
%emit_entry = OpLabel
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%ax_clip = OpCompositeExtract %float %pa 0
%ay_clip = OpCompositeExtract %float %pa 1
%az_clip = OpCompositeExtract %float %pa 2
%aw_clip = OpCompositeExtract %float %pa 3
%bx_clip = OpCompositeExtract %float %pb 0
%by_clip = OpCompositeExtract %float %pb 1
%bz_clip = OpCompositeExtract %float %pb 2
%bw_clip = OpCompositeExtract %float %pb 3
%inv_aw = OpFDiv %float %float_1 %aw_clip
%inv_bw = OpFDiv %float %float_1 %bw_clip
%ax_ndc = OpFMul %float %ax_clip %inv_aw
%ay_ndc = OpFMul %float %ay_clip %inv_aw
%bx_ndc = OpFMul %float %bx_clip %inv_bw
%by_ndc = OpFMul %float %by_clip %inv_bw
%dx = OpFSub %float %bx_ndc %ax_ndc
%dy = OpFSub %float %by_ndc %ay_ndc
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len2 = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %ext Sqrt %len2
%neg_dy = OpFNegate %float %dy
%perp_x_unit = OpFDiv %float %neg_dy %len
%perp_y_unit = OpFDiv %float %dx %len
%perp_x = OpFMul %float %perp_x_unit %float_half_width
%perp_y = OpFMul %float %perp_y_unit %float_half_width
%a_off_x = OpFMul %float %perp_x %aw_clip
%a_off_y = OpFMul %float %perp_y %aw_clip
%b_off_x = OpFMul %float %perp_x %bw_clip
%b_off_y = OpFMul %float %perp_y %bw_clip
%a_minus_x = OpFSub %float %ax_clip %a_off_x
%a_minus_y = OpFSub %float %ay_clip %a_off_y
%a_plus_x = OpFAdd %float %ax_clip %a_off_x
%a_plus_y = OpFAdd %float %ay_clip %a_off_y
%b_minus_x = OpFSub %float %bx_clip %b_off_x
%b_minus_y = OpFSub %float %by_clip %b_off_y
%b_plus_x = OpFAdd %float %bx_clip %b_off_x
%b_plus_y = OpFAdd %float %by_clip %b_off_y
%avg_sum = OpFAdd %v3float %ca %cb
%edge_color = OpVectorTimesScalar %v3float %avg_sum %float_half
%pos_a_minus = OpCompositeConstruct %v4float %a_minus_x %a_minus_y %az_clip %aw_clip
OpStore %out_pos_ptr %pos_a_minus
OpStore %outColor %edge_color
OpEmitVertex
%pos_a_plus = OpCompositeConstruct %v4float %a_plus_x %a_plus_y %az_clip %aw_clip
OpStore %out_pos_ptr %pos_a_plus
OpStore %outColor %edge_color
OpEmitVertex
%pos_b_minus = OpCompositeConstruct %v4float %b_minus_x %b_minus_y %bz_clip %bw_clip
OpStore %out_pos_ptr %pos_b_minus
OpStore %outColor %edge_color
OpEmitVertex
%pos_b_plus = OpCompositeConstruct %v4float %b_plus_x %b_plus_y %bz_clip %bw_clip
OpStore %out_pos_ptr %pos_b_plus
OpStore %outColor %edge_color
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd

%main = OpFunction %void None %fn_void
%main_entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p2 = OpLoad %v4float %p2_ptr
%p4 = OpLoad %v4float %p4_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%unused_wp_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%unused_n_ptr = OpAccessChain %ptr_in_v3 %normal %int_0
%unused_wp = OpLoad %v3float %unused_wp_ptr
%unused_n = OpLoad %v3float %unused_n_ptr
OpFunctionCall %void %emitEdge %p0 %p2 %c0 %c2
OpFunctionCall %void %emitEdge %p2 %p4 %c2 %c4
OpFunctionCall %void %emitEdge %p4 %p0 %c4 %c0
OpReturn
OpFunctionEnd
