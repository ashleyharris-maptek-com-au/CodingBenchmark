use std::collections::{HashMap, VecDeque};
use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Pt {
    x: f64,
    y: f64,
}

struct Scanner {
    buf: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { buf: input, pos: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut v = 0usize;
        while self.pos < self.buf.len() && !self.buf[self.pos].is_ascii_whitespace() {
            let c = self.buf[self.pos];
            if c.is_ascii_digit() {
                v = v.saturating_mul(10).saturating_add((c - b'0') as usize);
            }
            self.pos += 1;
        }
        Some(v)
    }
}

struct Rng {
    s: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Self { s: seed | 1 }
    }
    fn next_f64(&mut self) -> f64 {
        self.s = self.s.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
        let x = self.s >> 11;
        (x as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn add_sample_edge(adj: &mut [Vec<usize>], u: usize, v: usize, cap: usize) {
    if adj[u].len() < cap {
        adj[u].push(v);
    }
    if adj[v].len() < cap {
        adj[v].push(u);
    }
}

fn components(adj: &[Vec<usize>]) -> Vec<Vec<usize>> {
    let n = adj.len();
    let mut seen = vec![false; n];
    let mut comps = Vec::new();
    let mut q = VecDeque::new();
    for s in 0..n {
        if seen[s] {
            continue;
        }
        seen[s] = true;
        q.push_back(s);
        let mut c = Vec::new();
        while let Some(u) = q.pop_front() {
            c.push(u);
            for &v in &adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
        comps.push(c);
    }
    comps.sort_by_key(|c| std::cmp::Reverse(c.len()));
    comps
}

fn initial_layout(n: usize, adj: &[Vec<usize>], comps: &[Vec<usize>], deg: &[usize]) -> Vec<Pt> {
    let mut p = vec![Pt::default(); n];
    let mut rng = Rng::new((n as u64).wrapping_mul(11995408973635179863));
    let mut cursor_x = 0.0f64;
    let mut cursor_y = 0.0f64;
    let mut row_h = 0.0f64;
    let total_side = ((n as f64).sqrt() * 180.0).max(600.0);

    for comp in comps {
        let sz = comp.len();
        let radius = (sz as f64).sqrt().max(1.0) * 85.0;
        if cursor_x + 2.0 * radius > total_side {
            cursor_x = 0.0;
            cursor_y += row_h + 220.0;
            row_h = 0.0;
        }
        let cx = cursor_x + radius;
        let cy = cursor_y + radius;
        cursor_x += 2.0 * radius + 220.0;
        row_h = row_h.max(2.0 * radius);

        if sz == 1 {
            p[comp[0]] = Pt { x: cx, y: cy };
            continue;
        }

        let root = *comp.iter().max_by_key(|&&u| deg[u]).unwrap();
        let mut local_dist: HashMap<usize, usize> = HashMap::with_capacity(sz * 2);
        let mut q = VecDeque::new();
        local_dist.insert(root, 0);
        q.push_back(root);
        while let Some(u) = q.pop_front() {
            let du = local_dist[&u];
            for &v in &adj[u] {
                if !local_dist.contains_key(&v) {
                    local_dist.insert(v, du + 1);
                    q.push_back(v);
                }
            }
        }
        let mut levels: HashMap<usize, Vec<usize>> = HashMap::new();
        let mut max_level = 0usize;
        for &u in comp {
            let d = *local_dist.get(&u).unwrap_or(&0);
            max_level = max_level.max(d);
            levels.entry(d).or_default().push(u);
        }

        if max_level <= 1 && sz > 2 {
            for (i, &u) in comp.iter().enumerate() {
                let a = 2.0 * std::f64::consts::PI * (i as f64) / (sz as f64);
                let r = radius * (0.55 + 0.35 * rng.next_f64());
                p[u] = Pt { x: cx + r * a.cos(), y: cy + r * a.sin() };
            }
        } else {
            for (lvl, nodes) in levels.iter_mut() {
                nodes.sort_by_key(|&u| std::cmp::Reverse(deg[u]));
                let count = nodes.len().max(1);
                let r = if *lvl == 0 { 0.0 } else { 45.0 * (*lvl as f64).powf(0.9) };
                let offset = rng.next_f64() * 2.0 * std::f64::consts::PI;
                for (i, &u) in nodes.iter().enumerate() {
                    let a = offset + 2.0 * std::f64::consts::PI * (i as f64) / (count as f64);
                    p[u] = Pt {
                        x: cx + r * a.cos() + (rng.next_f64() - 0.5) * 8.0,
                        y: cy + r * a.sin() + (rng.next_f64() - 0.5) * 8.0,
                    };
                }
            }
        }
    }
    p
}

fn relax_pairwise(pos: &mut [Pt], edges: &[(usize, usize)], iterations: usize) {
    let n = pos.len();
    if n <= 1 {
        return;
    }
    let k = 95.0;
    let mut disp = vec![Pt::default(); n];
    let mut temp = 180.0;
    for _ in 0..iterations {
        disp.fill(Pt::default());
        for i in 0..n {
            for j in (i + 1)..n {
                let dx = pos[i].x - pos[j].x;
                let dy = pos[i].y - pos[j].y;
                let d2 = (dx * dx + dy * dy).max(0.01);
                let d = d2.sqrt();
                let f = (k * k / d).min(500.0);
                let fx = dx / d * f;
                let fy = dy / d * f;
                disp[i].x += fx;
                disp[i].y += fy;
                disp[j].x -= fx;
                disp[j].y -= fy;
            }
        }
        for &(u, v) in edges {
            let dx = pos[u].x - pos[v].x;
            let dy = pos[u].y - pos[v].y;
            let d = (dx * dx + dy * dy).sqrt().max(0.01);
            let f = (d * d / k).min(700.0);
            let fx = dx / d * f;
            let fy = dy / d * f;
            disp[u].x -= fx;
            disp[u].y -= fy;
            disp[v].x += fx;
            disp[v].y += fy;
        }
        for i in 0..n {
            let d = (disp[i].x * disp[i].x + disp[i].y * disp[i].y).sqrt().max(0.01);
            let step = d.min(temp);
            pos[i].x += disp[i].x / d * step;
            pos[i].y += disp[i].y / d * step;
        }
        temp *= 0.94;
    }
}

fn relax_grid(pos: &mut [Pt], edges: &[(usize, usize)], iterations: usize) {
    let n = pos.len();
    if n <= 1 {
        return;
    }
    let k = 80.0;
    let cell = 170.0;
    let mut disp = vec![Pt::default(); n];
    let mut temp = 130.0;
    for _ in 0..iterations {
        disp.fill(Pt::default());
        let mut grid: HashMap<(i64, i64), Vec<usize>> = HashMap::with_capacity(n * 2);
        for (i, p) in pos.iter().enumerate() {
            grid.entry(((p.x / cell).floor() as i64, (p.y / cell).floor() as i64))
                .or_default()
                .push(i);
        }
        for i in 0..n {
            let cx = (pos[i].x / cell).floor() as i64;
            let cy = (pos[i].y / cell).floor() as i64;
            for gx in (cx - 1)..=(cx + 1) {
                for gy in (cy - 1)..=(cy + 1) {
                    if let Some(bucket) = grid.get(&(gx, gy)) {
                        for &j in bucket {
                            if j <= i {
                                continue;
                            }
                            let dx = pos[i].x - pos[j].x;
                            let dy = pos[i].y - pos[j].y;
                            let d2 = (dx * dx + dy * dy).max(0.01);
                            if d2 > cell * cell * 2.5 {
                                continue;
                            }
                            let d = d2.sqrt();
                            let f = (k * k / d).min(350.0);
                            let fx = dx / d * f;
                            let fy = dy / d * f;
                            disp[i].x += fx;
                            disp[i].y += fy;
                            disp[j].x -= fx;
                            disp[j].y -= fy;
                        }
                    }
                }
            }
        }
        for &(u, v) in edges {
            let dx = pos[u].x - pos[v].x;
            let dy = pos[u].y - pos[v].y;
            let d = (dx * dx + dy * dy).sqrt().max(0.01);
            let f = (d * d / k).min(450.0);
            let fx = dx / d * f;
            let fy = dy / d * f;
            disp[u].x -= fx;
            disp[u].y -= fy;
            disp[v].x += fx;
            disp[v].y += fy;
        }
        for i in 0..n {
            let d = (disp[i].x * disp[i].x + disp[i].y * disp[i].y).sqrt().max(0.01);
            let step = d.min(temp);
            pos[i].x += disp[i].x / d * step;
            pos[i].y += disp[i].y / d * step;
        }
        temp *= 0.92;
    }
}

fn smooth_large(pos: &mut [Pt], adj: &[Vec<usize>], deg: &[usize], iterations: usize) {
    let n = pos.len();
    let mut next = pos.to_vec();
    for it in 0..iterations {
        let alpha = 0.28 / (1.0 + it as f64 * 0.08);
        for u in 0..n {
            if adj[u].is_empty() {
                continue;
            }
            let mut sx = 0.0;
            let mut sy = 0.0;
            let mut wsum = 0.0;
            for &v in &adj[u] {
                let w = 1.0 / (1.0 + deg[v] as f64).sqrt();
                sx += pos[v].x * w;
                sy += pos[v].y * w;
                wsum += w;
            }
            if wsum > 0.0 {
                next[u].x = pos[u].x * (1.0 - alpha) + (sx / wsum) * alpha;
                next[u].y = pos[u].y * (1.0 - alpha) + (sy / wsum) * alpha;
            }
        }
        pos.copy_from_slice(&next);
    }
}

fn normalize(pos: &mut [Pt]) {
    if pos.is_empty() {
        return;
    }
    let mut min_x = pos[0].x;
    let mut max_x = pos[0].x;
    let mut min_y = pos[0].y;
    let mut max_y = pos[0].y;
    for p in pos.iter() {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }
    let scale = 100000.0 / (max_x - min_x).max(max_y - min_y).max(1.0);
    for p in pos.iter_mut() {
        p.x = (p.x - min_x) * scale;
        p.y = (p.y - min_y) * scale;
        if !p.x.is_finite() {
            p.x = 0.0;
        }
        if !p.y.is_finite() {
            p.y = 0.0;
        }
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = sc.next_usize().unwrap_or(0);

    let adj_cap = if n <= 10_000 { usize::MAX } else if n <= 80_000 { 64 } else { 24 };
    let edge_cap = if n <= 2_000 {
        usize::MAX
    } else if n <= 20_000 {
        400_000
    } else {
        900_000
    };

    let mut adj = vec![Vec::<usize>::new(); n];
    let mut deg = vec![0usize; n];
    let mut edges = Vec::<(usize, usize)>::new();

    for _ in 0..m {
        let Some(u) = sc.next_usize() else { break };
        let Some(v) = sc.next_usize() else { break };
        if u >= n || v >= n || u == v {
            continue;
        }
        deg[u] = deg[u].saturating_add(1);
        deg[v] = deg[v].saturating_add(1);
        if edges.len() < edge_cap {
            edges.push((u, v));
        }
        add_sample_edge(&mut adj, u, v, adj_cap);
    }

    let comps = components(&adj);
    let mut pos = initial_layout(n, &adj, &comps, &deg);

    if n <= 1_600 {
        let iters = if n <= 80 { 220 } else if n <= 400 { 120 } else { 70 };
        relax_pairwise(&mut pos, &edges, iters);
    } else if n <= 12_000 {
        let iters = if n <= 5_000 { 75 } else { 35 };
        relax_grid(&mut pos, &edges, iters);
    } else {
        smooth_large(&mut pos, &adj, &deg, 18);
    }

    normalize(&mut pos);
    let mut out = String::with_capacity(n.saturating_mul(28));
    for p in pos {
        out.push_str(&format!("{:.6} {:.6}\n", p.x, p.y));
    }
    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(out.as_bytes()).unwrap();
}
