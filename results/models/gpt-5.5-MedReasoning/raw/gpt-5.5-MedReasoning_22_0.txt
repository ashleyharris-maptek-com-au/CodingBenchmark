using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private struct Cell
    {
        public int X;
        public int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    private struct Move
    {
        public int Rotation;
        public int Column;
        public double Score;
        public Cell[] Shape;
        public int DropY;
    }

    private static int width, height, totalPieces;
    private static bool[,] board = new bool[1, 1]; // [y, x], y=0 is the bottom row
    private static readonly Dictionary<char, Cell[][]> Shapes = BuildShapes();

    public static void Main()
    {
        string? tok = NextToken();
        if (tok == null) return;
        width = int.Parse(tok, CultureInfo.InvariantCulture);
        height = int.Parse(NextToken() ?? "20", CultureInfo.InvariantCulture);
        totalPieces = int.Parse(NextToken() ?? "0", CultureInfo.InvariantCulture);

        width = Math.Max(4, width);
        height = Math.Max(4, height);
        board = new bool[height, width];

        for (int i = 0; i < totalPieces; i++)
        {
            string? pieceTok = NextToken();
            if (pieceTok == null) break;
            char piece = char.ToUpperInvariant(pieceTok[0]);

            Move best = ChooseMove(piece, i);
            Console.WriteLine(best.Rotation.ToString(CultureInfo.InvariantCulture) + " " +
                              best.Column.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            Apply(best);

            string? result = NextToken();
            if (result == null || result.Equals("gameover", StringComparison.OrdinalIgnoreCase))
                break;
        }
    }

    private static Move ChooseMove(char piece, int pieceIndex)
    {
        if (!Shapes.TryGetValue(piece, out Cell[][]? rotations))
            rotations = Shapes['O'];

        bool marathon = totalPieces <= 0 || totalPieces > 750;
        bool cramped = width <= 8 || height <= 16;
        int remaining = totalPieces <= 0 ? int.MaxValue : totalPieces - pieceIndex;

        // Adaptive strategy: short games can spend more risk to clear lines; marathon and
        // cramped boards value low, smooth, hole-free stacks much more heavily.
        double lineWeight = remaining < 80 ? 1.25 : 0.78;
        double holeWeight = marathon || cramped ? 9.2 : 7.1;
        double heightWeight = marathon || cramped ? 1.18 : 0.83;
        double bumpWeight = cramped ? 0.86 : 0.54;
        double wellWeight = width >= 10 && marathon ? 0.23 : 0.08;
        double transitionWeight = marathon ? 0.28 : 0.16;

        Move best = new Move { Rotation = 0, Column = 0, Score = double.NegativeInfinity, Shape = rotations[0], DropY = height };

        for (int r = 0; r < rotations.Length; r++)
        {
            Cell[] shape = rotations[r];
            int pieceWidth = ShapeWidth(shape);
            for (int col = 0; col <= width - pieceWidth; col++)
            {
                int dropY = FindDropY(shape, col);
                if (dropY == int.MinValue) continue;

                bool[,] next = CloneBoard(board);
                int overflow = PlaceOn(next, shape, col, dropY);
                int cleared = ClearLines(next);

                int[] heights = ColumnHeights(next);
                int maxHeight = Max(heights);
                int aggregateHeight = Sum(heights);
                int holes = CountHoles(next, heights);
                int bumpiness = Bumpiness(heights);
                int wells = WellDepth(heights);
                int rowTransitions = RowTransitions(next);
                int colTransitions = ColumnTransitions(next);
                int danger = Math.Max(0, maxHeight - (height * 7 / 10));

                double score =
                    cleared * cleared * lineWeight * width
                    - holes * holeWeight
                    - aggregateHeight * heightWeight / Math.Max(1, width)
                    - bumpiness * bumpWeight
                    - (rowTransitions + colTransitions) * transitionWeight
                    - danger * danger * 1.7
                    - overflow * 10000.0;

                // Preserve a right-side well for I pieces on wide marathon boards, without
                // making it a hard rule that would fail on narrow boards.
                if (width >= 10)
                {
                    int rightWell = Math.Max(0, Math.Min(heights[width - 2], height) - heights[width - 1]);
                    score += rightWell * wellWeight;
                    if (piece == 'I' && r % 2 == 1 && col == width - 1 && rightWell >= 3)
                        score += 5.0;
                }
                score -= wells * 0.03; // very deep accidental wells are still bad

                // Prefer central placements for similar scores; it reduces jagged edges early.
                double center = col + (pieceWidth - 1) / 2.0;
                score -= Math.Abs(center - (width - 1) / 2.0) * 0.015;

                if (score > best.Score)
                    best = new Move { Rotation = r, Column = col, Score = score, Shape = shape, DropY = dropY };
            }
        }

        return best;
    }

    private static int FindDropY(Cell[] shape, int col)
    {
        int y = height + 4;
        while (CanPlace(shape, col, y - 1)) y--;
        return CanPlace(shape, col, y) ? y : int.MinValue;
    }

    private static bool CanPlace(Cell[] shape, int col, int y)
    {
        foreach (Cell c in shape)
        {
            int x = col + c.X;
            int by = y + c.Y;
            if (x < 0 || x >= width || by < 0) return false;
            if (by < height && board[by, x]) return false;
        }
        return true;
    }

    private static void Apply(Move m)
    {
        PlaceOn(board, m.Shape, m.Column, m.DropY);
        ClearLines(board);
    }

    private static int PlaceOn(bool[,] b, Cell[] shape, int col, int y)
    {
        int overflow = 0;
        foreach (Cell c in shape)
        {
            int x = col + c.X;
            int by = y + c.Y;
            if (x < 0 || x >= width || by < 0) { overflow += 100; continue; }
            if (by >= height) overflow++;
            else b[by, x] = true;
        }
        return overflow;
    }

    private static int ClearLines(bool[,] b)
    {
        int write = 0, cleared = 0;
        for (int y = 0; y < height; y++)
        {
            bool full = true;
            for (int x = 0; x < width; x++)
                if (!b[y, x]) { full = false; break; }
            if (full) { cleared++; continue; }
            if (write != y)
                for (int x = 0; x < width; x++) b[write, x] = b[y, x];
            write++;
        }
        for (int y = write; y < height; y++)
            for (int x = 0; x < width; x++) b[y, x] = false;
        return cleared;
    }

    private static int[] ColumnHeights(bool[,] b)
    {
        int[] h = new int[width];
        for (int x = 0; x < width; x++)
            for (int y = height - 1; y >= 0; y--)
                if (b[y, x]) { h[x] = y + 1; break; }
        return h;
    }

    private static int CountHoles(bool[,] b, int[] heights)
    {
        int holes = 0;
        for (int x = 0; x < width; x++)
            for (int y = 0; y < heights[x]; y++)
                if (!b[y, x]) holes++;
        return holes;
    }

    private static int Bumpiness(int[] h)
    {
        int v = 0;
        for (int i = 0; i + 1 < h.Length; i++) v += Math.Abs(h[i] - h[i + 1]);
        return v;
    }

    private static int WellDepth(int[] h)
    {
        int total = 0;
        for (int i = 0; i < h.Length; i++)
        {
            int left = i == 0 ? height : h[i - 1];
            int right = i == h.Length - 1 ? height : h[i + 1];
            int d = Math.Min(left, right) - h[i];
            if (d > 0) total += d * (d + 1) / 2;
        }
        return total;
    }

    private static int RowTransitions(bool[,] b)
    {
        int t = 0;
        for (int y = 0; y < height; y++)
        {
            bool prev = true; // wall
            for (int x = 0; x < width; x++)
            {
                bool cur = b[y, x];
                if (cur != prev) t++;
                prev = cur;
            }
            if (!prev) t++; // right wall
        }
        return t;
    }

    private static int ColumnTransitions(bool[,] b)
    {
        int t = 0;
        for (int x = 0; x < width; x++)
        {
            bool prev = true; // floor
            for (int y = 0; y < height; y++)
            {
                bool cur = b[y, x];
                if (cur != prev) t++;
                prev = cur;
            }
            if (!prev) t++; // ceiling
        }
        return t;
    }

    private static int ShapeWidth(Cell[] s)
    {
        int max = 0;
        foreach (Cell c in s) if (c.X > max) max = c.X;
        return max + 1;
    }

    private static int Sum(int[] a) { int s = 0; foreach (int v in a) s += v; return s; }
    private static int Max(int[] a) { int m = 0; foreach (int v in a) if (v > m) m = v; return m; }

    private static bool[,] CloneBoard(bool[,] src)
    {
        bool[,] dst = new bool[height, width];
        Array.Copy(src, dst, src.Length);
        return dst;
    }

    private static string? NextToken()
    {
        int ch;
        do
        {
            ch = Console.In.Read();
            if (ch < 0) return null;
        } while (char.IsWhiteSpace((char)ch));

        var chars = new List<char>(16);
        do
        {
            chars.Add((char)ch);
            ch = Console.In.Read();
        } while (ch >= 0 && !char.IsWhiteSpace((char)ch));
        return new string(chars.ToArray());
    }

    private static Dictionary<char, Cell[][]> BuildShapes()
    {
        var d = new Dictionary<char, Cell[][]>();
        d['I'] = Rotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(3, 0) });
        d['O'] = Rotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(0, 1), new Cell(1, 1) });
        d['T'] = Rotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(1, 1) });
        d['S'] = Rotations(new[] { new Cell(1, 0), new Cell(2, 0), new Cell(0, 1), new Cell(1, 1) });
        d['Z'] = Rotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(1, 1), new Cell(2, 1) });
        d['J'] = Rotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(0, 1) });
        d['L'] = Rotations(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(2, 1) });
        return d;
    }

    private static Cell[][] Rotations(Cell[] baseShape)
    {
        Cell[][] result = new Cell[4][];
        result[0] = Normalize(baseShape);
        for (int i = 1; i < 4; i++)
        {
            Cell[] prev = result[i - 1];
            Cell[] next = new Cell[prev.Length];
            for (int j = 0; j < prev.Length; j++)
                next[j] = new Cell(prev[j].Y, -prev[j].X); // 90 degrees clockwise, then normalize
            result[i] = Normalize(next);
        }
        return result;
    }

    private static Cell[] Normalize(Cell[] s)
    {
        int minX = int.MaxValue, minY = int.MaxValue;
        foreach (Cell c in s)
        {
            if (c.X < minX) minX = c.X;
            if (c.Y < minY) minY = c.Y;
        }
        Cell[] n = new Cell[s.Length];
        for (int i = 0; i < s.Length; i++) n[i] = new Cell(s[i].X - minX, s[i].Y - minY);
        Array.Sort(n, (a, b) => a.X == b.X ? a.Y.CompareTo(b.Y) : a.X.CompareTo(b.X));
        return n;
    }
}
