#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

using namespace std;

class Solver {
public:
    int n = 0;
    vector<long long> w;
    vector<vector<unsigned long long>> adj; // complement graph adjacency
    long long best = 0;
    vector<int> cur, bestSet;

    bool adjacent(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void colorSort(const vector<int>& p, vector<int>& order, vector<long long>& bound) const {
        vector<int> uncolored = p;
        order.clear();
        bound.clear();

        long long cumulative = 0;
        while (!uncolored.empty()) {
            vector<int> available = uncolored;
            vector<int> colorClass;
            long long colorWeight = 0;

            while (!available.empty()) {
                int bestPos = 0;
                for (int i = 1; i < (int)available.size(); ++i) {
                    if (w[available[i]] > w[available[bestPos]]) bestPos = i;
                }

                int v = available[bestPos];
                colorClass.push_back(v);
                colorWeight = max(colorWeight, w[v]);

                uncolored.erase(remove(uncolored.begin(), uncolored.end(), v), uncolored.end());

                vector<int> nextAvailable;
                nextAvailable.reserve(available.size());
                for (int u : available) {
                    if (u != v && !adjacent(u, v)) nextAvailable.push_back(u);
                }
                available.swap(nextAvailable);
            }

            cumulative += colorWeight;
            for (int v : colorClass) {
                order.push_back(v);
                bound.push_back(cumulative);
            }
        }
    }

    void expand(vector<int> p, long long value) {
        if (p.empty()) {
            if (value > best) {
                best = value;
                bestSet = cur;
            }
            return;
        }

        vector<int> order;
        vector<long long> bound;
        colorSort(p, order, bound);

        for (int i = (int)order.size() - 1; i >= 0; --i) {
            if (value + bound[i] <= best) return;

            int v = order[i];
            cur.push_back(v);

            vector<int> next;
            next.reserve(i);
            for (int j = 0; j < i; ++j) {
                int u = order[j];
                if (adjacent(v, u)) next.push_back(u);
            }

            long long nextValue = value + w[v];
            if (nextValue > best) {
                best = nextValue;
                bestSet = cur;
            }
            if (!next.empty()) expand(next, nextValue);

            cur.pop_back();
        }
    }

    vector<int> solve() {
        vector<int> p;
        p.reserve(n);
        for (int i = 0; i < n; ++i) {
            if (w[i] > 0) p.push_back(i);
        }
        sort(p.begin(), p.end(), [&](int a, int b) {
            if (w[a] != w[b]) return w[a] > w[b];
            return a < b;
        });

        expand(p, 0);

        vector<int> x(n, 0);
        for (int v : bestSet) x[v] = 1;
        return x;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    int words = (n + 63) >> 6;
    vector<vector<unsigned long long>> forbidden(n, vector<unsigned long long>(words, 0));
    for (int i = 0; i < n; ++i) forbidden[i][i >> 6] |= 1ULL << (i & 63);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (0 <= u && u < n && 0 <= v && v < n) {
            forbidden[u][v >> 6] |= 1ULL << (v & 63);
            forbidden[v][u >> 6] |= 1ULL << (u & 63);
        }
    }

    Solver solver;
    solver.n = n;
    solver.w = c;
    solver.adj.assign(n, vector<unsigned long long>(words, 0));

    for (int i = 0; i < n; ++i) {
        for (int b = 0; b < words; ++b) solver.adj[i][b] = ~forbidden[i][b];
        if (n & 63) solver.adj[i][words - 1] &= (1ULL << (n & 63)) - 1ULL;
    }

    vector<int> x = solver.solve();
    long long objective = 0;
    for (int i = 0; i < n; ++i) {
        if (x[i]) objective += c[i];
    }

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << '\n';

    return 0;
}
