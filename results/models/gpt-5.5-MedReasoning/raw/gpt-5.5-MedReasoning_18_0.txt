#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <utility>
#include <vector>

struct Edge {
    int u;
    int v;
};

static int find_bridge(int n, const std::vector<Edge>& edges,
                       const std::vector<std::vector<std::pair<int, int>>>& g) {
    std::vector<int> tin(n, -1), low(n, 0), parent(n, -1), parentEdge(n, -1), it(n, 0);
    int timer = 0;

    for (int root = 0; root < n; ++root) {
        if (tin[root] != -1) continue;

        std::vector<int> st;
        st.push_back(root);
        parent[root] = -1;
        parentEdge[root] = -1;

        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }

            if (it[v] < static_cast<int>(g[v].size())) {
                auto [to, id] = g[v][it[v]++];
                if (id == parentEdge[v]) continue;

                if (tin[to] == -1) {
                    parent[to] = v;
                    parentEdge[to] = id;
                    st.push_back(to);
                } else {
                    low[v] = std::min(low[v], tin[to]);
                }
            } else {
                st.pop_back();
                int p = parent[v];
                if (p != -1) {
                    low[p] = std::min(low[p], low[v]);
                    if (low[v] > tin[p]) {
                        return parentEdge[v];
                    }
                }
            }
        }
    }

    (void)edges;
    return -1;
}

static std::vector<int> stoer_wagner_sparse(int n, const std::vector<Edge>& edges) {
    std::vector<std::unordered_map<int, int>> adj(n);
    for (const Edge& e : edges) {
        if (e.u == e.v) continue;
        ++adj[e.u][e.v];
        ++adj[e.v][e.u];
    }

    std::vector<char> alive(n, 1), inA(n, 0);
    std::vector<long long> weight(n, 0);
    std::vector<std::vector<int>> part(n);
    for (int i = 0; i < n; ++i) part[i].push_back(i);

    int aliveCount = n;
    long long best = std::numeric_limits<long long>::max();
    std::vector<int> bestSide;

    auto first_alive = [&]() -> int {
        for (int i = 0; i < n; ++i) {
            if (alive[i]) return i;
        }
        return -1;
    };

    while (aliveCount > 1) {
        std::fill(inA.begin(), inA.end(), 0);
        std::fill(weight.begin(), weight.end(), 0);

        using Item = std::pair<long long, int>;
        std::priority_queue<Item> pq;

        int prev = -1;
        int last = -1;
        int start = first_alive();

        for (int step = 0; step < aliveCount; ++step) {
            int v = -1;
            if (step == 0) {
                v = start;
            } else {
                while (!pq.empty()) {
                    auto [w, x] = pq.top();
                    pq.pop();
                    if (alive[x] && !inA[x] && weight[x] == w) {
                        v = x;
                        break;
                    }
                }

                if (v == -1) {
                    for (int x = 0; x < n; ++x) {
                        if (alive[x] && !inA[x]) {
                            v = x;
                            break;
                        }
                    }
                }
            }

            prev = last;
            last = v;
            inA[v] = 1;

            if (step == aliveCount - 1) {
                if (weight[v] < best) {
                    best = weight[v];
                    bestSide = part[v];
                }

                int s = prev;
                int t = v;
                if (s == -1) break;

                std::vector<std::pair<int, int>> incident;
                incident.reserve(adj[t].size());
                for (const auto& kv : adj[t]) incident.push_back(kv);

                for (auto [x, w] : incident) {
                    if (!alive[x] || x == s) continue;
                    adj[s][x] += w;
                    adj[x][s] += w;
                    adj[x].erase(t);
                }

                adj[s].erase(t);
                adj[t].clear();
                alive[t] = 0;
                --aliveCount;

                part[s].insert(part[s].end(), part[t].begin(), part[t].end());
                std::vector<int>().swap(part[t]);
                break;
            }

            for (const auto& kv : adj[v]) {
                int to = kv.first;
                int w = kv.second;
                if (alive[to] && !inA[to]) {
                    weight[to] += w;
                    pq.push({weight[to], to});
                }
            }
        }
    }

    return bestSide;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n;
    long long mInput;
    if (!(std::cin >> n >> mInput)) return 0;

    if (n <= 1) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Edge> edges;
    edges.reserve(static_cast<size_t>(std::max<long long>(0, mInput)));
    std::vector<std::vector<std::pair<int, int>>> g(n);
    std::vector<int> degree(n, 0);

    for (long long i = 0; i < mInput; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;

        int id = static_cast<int>(edges.size());
        edges.push_back({u, v});
        if (u != v) {
            g[u].push_back({v, id});
            g[v].push_back({u, id});
            ++degree[u];
            ++degree[v];
        }
    }

    int bridgeId = find_bridge(n, edges, g);
    if (bridgeId != -1) {
        std::cout << 1 << '\n';
        std::cout << edges[bridgeId].u << ' ' << edges[bridgeId].v << '\n';
        return 0;
    }

    int minDegreeVertex = 0;
    for (int i = 1; i < n; ++i) {
        if (degree[i] < degree[minDegreeVertex]) minDegreeVertex = i;
    }

    if (degree[minDegreeVertex] == 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<int> side = stoer_wagner_sparse(n, edges);
    if (side.empty() || static_cast<int>(side.size()) == n) {
        side.clear();
        side.push_back(minDegreeVertex);
    }

    std::vector<char> mark(n, 0);
    for (int v : side) mark[v] = 1;

    std::vector<Edge> cut;
    cut.reserve(static_cast<size_t>(degree[minDegreeVertex]));
    for (const Edge& e : edges) {
        if (e.u == e.v) continue;
        if (mark[e.u] != mark[e.v]) cut.push_back(e);
    }

    std::cout << cut.size() << '\n';
    for (const Edge& e : cut) {
        std::cout << e.u << ' ' << e.v << '\n';
    }

    return 0;
}
