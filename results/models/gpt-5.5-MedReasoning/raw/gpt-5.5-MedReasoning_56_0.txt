const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;
const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const dx = Number(depot[0]) || 0;
const dy = Number(depot[1]) || 0;
const maxRoutes = Math.max(1, Math.min(n || 1, Math.floor(Number(data.truckCount) || 1)));

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const ids = new Array(n);
const xs = new Float64Array(n);
const ys = new Float64Array(n);
const depotDist = new Float64Array(n);
const angles = new Float64Array(n);
const radius2 = new Float64Array(n);

for (let i = 0; i < n; i++) {
  const p = packages[i];
  ids[i] = p.id;
  const x = Number(p.x) || 0;
  const y = Number(p.y) || 0;
  xs[i] = x;
  ys[i] = y;
  const ax = x - dx;
  const ay = y - dy;
  depotDist[i] = Math.hypot(ax, ay);
  angles[i] = Math.atan2(ay, ax);
  radius2[i] = ax * ax + ay * ay;
}

function distIdx(a, b) {
  return Math.hypot(xs[a] - xs[b], ys[a] - ys[b]);
}

function distNode(a, b) {
  if (a < 0 && b < 0) return 0;
  if (a < 0) return depotDist[b];
  if (b < 0) return depotDist[a];
  return distIdx(a, b);
}

function reversePart(arr, lo, hi) {
  while (lo < hi) {
    const t = arr[lo];
    arr[lo++] = arr[hi];
    arr[hi--] = t;
  }
}

function routeCost(route) {
  const m = route.length;
  if (m === 0) return 0;
  let total = depotDist[route[0]] + depotDist[route[m - 1]];
  for (let i = 1; i < m; i++) total += distIdx(route[i - 1], route[i]);
  return total;
}

function solutionCost(routes) {
  let total = 0;
  for (const r of routes) total += routeCost(r);
  return total;
}

function twoOptRoute(route, maxPasses, windowSize) {
  const m = route.length;
  if (m < 3) return;
  const window = Math.max(3, Math.min(m, windowSize || m));
  const eps = 1e-10;

  for (let pass = 0; pass < maxPasses; pass++) {
    let changed = false;
    for (let i = -1; i < m - 1; i++) {
      const upper = Math.min(m - 1, i + window);
      for (let j = i + 2; j <= upper; j++) {
        if (i < 0 && j === m - 1) continue;
        const a = i < 0 ? -1 : route[i];
        const b = route[i + 1];
        const c = route[j];
        const d = j === m - 1 ? -1 : route[j + 1];
        const before = distNode(a, b) + distNode(c, d);
        const after = distNode(a, c) + distNode(b, d);
        if (after + eps < before) {
          reversePart(route, i + 1, j);
          changed = true;
        }
      }
    }
    if (!changed) break;
  }
}

function splitOrder(order) {
  const m = order.length;
  const prefix = new Float64Array(m);
  for (let i = 1; i < m; i++) {
    prefix[i] = prefix[i - 1] + distIdx(order[i - 1], order[i]);
  }

  const routeLimit = Math.min(maxRoutes, m);
  const parents = new Array(routeLimit + 1);
  parents[1] = new Int32Array(m);
  let prev = new Float64Array(m);
  for (let j = 0; j < m; j++) {
    prev[j] = depotDist[order[0]] + prefix[j] + depotDist[order[j]];
    parents[1][j] = 0;
  }

  let bestT = 1;
  let bestCost = prev[m - 1];
  let bestDp = prev;

  for (let t = 2; t <= routeLimit; t++) {
    const cur = new Float64Array(m);
    cur.fill(Infinity);
    const par = new Int32Array(m);
    par.fill(-1);
    parents[t] = par;

    let bestVal = Infinity;
    let bestStart = -1;
    for (let j = t - 1; j < m; j++) {
      const candidateStart = j;
      const prior = prev[candidateStart - 1];
      const val = prior + depotDist[order[candidateStart]] - prefix[candidateStart];
      if (val < bestVal) {
        bestVal = val;
        bestStart = candidateStart;
      }
      cur[j] = prefix[j] + depotDist[order[j]] + bestVal;
      par[j] = bestStart;
    }

    if (cur[m - 1] < bestCost) {
      bestCost = cur[m - 1];
      bestT = t;
      bestDp = cur;
    }
    prev = cur;
  }

  const routes = [];
  let end = m - 1;
  for (let t = bestT; t >= 1; t--) {
    const start = t === 1 ? 0 : parents[t][end];
    routes.push(order.slice(start, end + 1));
    end = start - 1;
  }
  routes.reverse();
  return { routes, cost: bestDp[m - 1] };
}

function rotate(order, start) {
  if (start <= 0) return order.slice();
  return order.slice(start).concat(order.slice(0, start));
}

function angularOrder(radialDescending) {
  const order = Array.from({ length: n }, (_, i) => i);
  order.sort((a, b) => {
    const da = angles[a] - angles[b];
    if (da !== 0) return da;
    return radialDescending ? radius2[b] - radius2[a] : radius2[a] - radius2[b];
  });
  return order;
}

function largestAngleGapStart(order) {
  if (order.length < 2) return 0;
  let bestGap = -1;
  let best = 0;
  for (let i = 0; i < order.length; i++) {
    const a = angles[order[i]];
    const b = angles[order[(i + 1) % order.length]] + (i + 1 === order.length ? Math.PI * 2 : 0);
    const gap = b - a;
    if (gap > bestGap) {
      bestGap = gap;
      best = (i + 1) % order.length;
    }
  }
  return best;
}

function nearestNeighborOrder() {
  const remaining = Array.from({ length: n }, (_, i) => i);
  const order = [];
  let cur = -1;
  while (remaining.length) {
    let bestPos = 0;
    let bestD = Infinity;
    for (let i = 0; i < remaining.length; i++) {
      const p = remaining[i];
      const d = cur < 0 ? depotDist[p] : distIdx(cur, p);
      if (d < bestD) {
        bestD = d;
        bestPos = i;
      }
    }
    cur = remaining[bestPos];
    order.push(cur);
    remaining[bestPos] = remaining[remaining.length - 1];
    remaining.pop();
  }
  return order;
}

function exactSmallOrder() {
  if (n > 16) return null;
  const states = 1 << n;
  const size = states * n;
  const dp = new Float64Array(size);
  dp.fill(Infinity);
  const parent = new Int16Array(size);
  parent.fill(-1);

  for (let i = 0; i < n; i++) dp[((1 << i) * n) + i] = depotDist[i];

  for (let mask = 1; mask < states; mask++) {
    const base = mask * n;
    for (let last = 0; last < n; last++) {
      const cur = dp[base + last];
      if (!Number.isFinite(cur)) continue;
      let rest = (states - 1) ^ mask;
      while (rest) {
        const bit = rest & -rest;
        const next = Math.trunc(Math.log2(bit));
        const nextMask = mask | bit;
        const idx = nextMask * n + next;
        const val = cur + distIdx(last, next);
        if (val < dp[idx]) {
          dp[idx] = val;
          parent[idx] = last;
        }
        rest ^= bit;
      }
    }
  }

  const full = states - 1;
  let bestEnd = 0;
  let bestCost = Infinity;
  for (let i = 0; i < n; i++) {
    const val = dp[full * n + i] + depotDist[i];
    if (val < bestCost) {
      bestCost = val;
      bestEnd = i;
    }
  }

  const order = new Array(n);
  let mask = full;
  let cur = bestEnd;
  for (let pos = n - 1; pos >= 0; pos--) {
    order[pos] = cur;
    const prev = parent[mask * n + cur];
    mask ^= 1 << cur;
    cur = prev;
  }
  return order;
}

function consider(order, best) {
  const trial = order.slice();
  if (trial.length <= 450) {
    twoOptRoute(trial, 4, trial.length);
  } else if (trial.length <= 2500) {
    twoOptRoute(trial, 2, 80);
  } else {
    twoOptRoute(trial, 1, 40);
  }

  const split = splitOrder(trial);
  for (const r of split.routes) {
    if (r.length <= 350) {
      twoOptRoute(r, 5, r.length);
    } else if (r.length <= 2500) {
      twoOptRoute(r, 3, 90);
    } else {
      twoOptRoute(r, 2, 45);
    }
  }
  const cost = solutionCost(split.routes);
  if (!best || cost < best.cost) return { cost, routes: split.routes };
  return best;
}

const exactOrder = exactSmallOrder();
if (exactOrder) {
  process.stdout.write(JSON.stringify({ routes: [exactOrder.map(i => ids[i])] }));
  process.exit(0);
}

let best = null;
const baseA = angularOrder(false);
const baseB = angularOrder(true);
const starts = new Set([
  0,
  largestAngleGapStart(baseA),
  Math.floor(n / 4),
  Math.floor(n / 2),
  Math.floor((3 * n) / 4)
]);

for (const s of starts) {
  best = consider(rotate(baseA, s), best);
  best = consider(rotate(baseB, s), best);
}

if (n <= 4000) {
  best = consider(nearestNeighborOrder(), best);
}

const outputRoutes = best.routes.map(route => route.map(i => ids[i]));
process.stdout.write(JSON.stringify({ routes: outputRoutes }));
