#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int u, v;
    long long w;
};

struct Arc {
    int to;
    int id;
    long long w;
};

struct DSU {
    vector<int> p, r;
    explicit DSU(int n = 0) { reset(n); }
    void reset(int n) {
        p.resize(n);
        r.assign(n, 0);
        iota(p.begin(), p.end(), 0);
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static const long long INF = numeric_limits<long long>::max() / 4;

static vector<int> normalize_tree_edges(int n, const vector<Edge>& edges,
                                        const vector<int>& terminals,
                                        vector<int> ids) {
    if (terminals.size() <= 1) return {};

    sort(ids.begin(), ids.end());
    ids.erase(unique(ids.begin(), ids.end()), ids.end());
    sort(ids.begin(), ids.end(), [&](int a, int b) {
        if (edges[a].w != edges[b].w) return edges[a].w < edges[b].w;
        return a < b;
    });

    DSU dsu(n);
    vector<int> tree;
    for (int id : ids) {
        if (dsu.unite(edges[id].u, edges[id].v)) tree.push_back(id);
    }

    vector<char> is_terminal(n, 0);
    for (int t : terminals) is_terminal[t] = 1;

    vector<vector<pair<int, int>>> adj(n);
    vector<int> degree(n, 0);
    for (int id : tree) {
        int u = edges[id].u, v = edges[id].v;
        adj[u].push_back({v, id});
        adj[v].push_back({u, id});
        ++degree[u];
        ++degree[v];
    }

    vector<char> removed_edge(edges.size(), 0), removed_vertex(n, 0);
    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!is_terminal[v] && degree[v] <= 1) q.push(v);
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (removed_vertex[v] || is_terminal[v] || degree[v] > 1) continue;
        removed_vertex[v] = 1;
        for (auto [to, id] : adj[v]) {
            if (removed_edge[id]) continue;
            removed_edge[id] = 1;
            --degree[v];
            --degree[to];
            if (!is_terminal[to] && degree[to] <= 1) q.push(to);
        }
    }

    vector<int> result;
    for (int id : tree) {
        if (!removed_edge[id]) result.push_back(id);
    }
    return result;
}

static vector<long long> dijkstra_dist(int n, const vector<vector<Arc>>& graph,
                                       int source, vector<int>* parent_v,
                                       vector<int>* parent_e) {
    vector<long long> dist(n, INF);
    if (parent_v) parent_v->assign(n, -1);
    if (parent_e) parent_e->assign(n, -1);

    priority_queue<pair<long long, int>, vector<pair<long long, int>>,
                   greater<pair<long long, int>>> pq;
    dist[source] = 0;
    pq.push({0, source});
    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (d != dist[v]) continue;
        for (const Arc& a : graph[v]) {
            long long nd = d + a.w;
            if (nd < dist[a.to]) {
                dist[a.to] = nd;
                if (parent_v) (*parent_v)[a.to] = v;
                if (parent_e) (*parent_e)[a.to] = a.id;
                pq.push({nd, a.to});
            }
        }
    }
    return dist;
}

static vector<int> exact_steiner(int n, const vector<Edge>& edges,
                                 const vector<vector<Arc>>& graph,
                                 const vector<int>& terminals) {
    int k = static_cast<int>(terminals.size());
    if (k <= 1) return {};
    int states = 1 << k;
    size_t cells = static_cast<size_t>(states) * static_cast<size_t>(n);

    vector<long long> dp(cells, INF);
    vector<int> parent_v(cells, -1), parent_e(cells, -1), split(cells, 0);

    auto idx = [n](int mask, int v) -> size_t {
        return static_cast<size_t>(mask) * static_cast<size_t>(n) + v;
    };

    for (int i = 0; i < k; ++i) dp[idx(1 << i, terminals[i])] = 0;

    for (int mask = 1; mask < states; ++mask) {
        if ((mask & (mask - 1)) != 0) {
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (sub > other) continue;
                size_t off = idx(mask, 0), aoff = idx(sub, 0), boff = idx(other, 0);
                for (int v = 0; v < n; ++v) {
                    long long da = dp[aoff + v], db = dp[boff + v];
                    if (da == INF || db == INF) continue;
                    long long cand = da + db;
                    if (cand < dp[off + v]) {
                        dp[off + v] = cand;
                        split[off + v] = sub;
                        parent_v[off + v] = -1;
                        parent_e[off + v] = -1;
                    }
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>,
                       greater<pair<long long, int>>> pq;
        size_t off = idx(mask, 0);
        for (int v = 0; v < n; ++v) {
            if (dp[off + v] < INF) pq.push({dp[off + v], v});
        }
        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dp[off + v]) continue;
            for (const Arc& a : graph[v]) {
                long long nd = d + a.w;
                if (nd < dp[off + a.to]) {
                    dp[off + a.to] = nd;
                    split[off + a.to] = 0;
                    parent_v[off + a.to] = v;
                    parent_e[off + a.to] = a.id;
                    pq.push({nd, a.to});
                }
            }
        }
    }

    int full = states - 1;
    int best_v = -1;
    long long best = INF;
    for (int v = 0; v < n; ++v) {
        if (dp[idx(full, v)] < best) {
            best = dp[idx(full, v)];
            best_v = v;
        }
    }
    if (best_v < 0) return {};

    vector<int> collected;
    auto collect = [&](auto&& self, int mask, int v) -> void {
        size_t at = idx(mask, v);
        if (split[at] != 0) {
            int s = split[at];
            self(self, s, v);
            self(self, mask ^ s, v);
        } else if (parent_v[at] != -1) {
            collected.push_back(parent_e[at]);
            self(self, mask, parent_v[at]);
        }
    };
    collect(collect, full, best_v);
    return normalize_tree_edges(n, edges, terminals, collected);
}

static vector<int> metric_mst_heuristic(int n, const vector<Edge>& edges,
                                        const vector<vector<Arc>>& graph,
                                        const vector<int>& terminals) {
    int k = static_cast<int>(terminals.size());
    vector<vector<long long>> td(k, vector<long long>(k, INF));
    vector<vector<int>> parent_v(k, vector<int>(n, -1));
    vector<vector<int>> parent_e(k, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) {
        vector<int> pv, pe;
        vector<long long> dist = dijkstra_dist(n, graph, terminals[i], &pv, &pe);
        parent_v[i].swap(pv);
        parent_e[i].swap(pe);
        for (int j = 0; j < k; ++j) td[i][j] = dist[terminals[j]];
    }

    vector<tuple<long long, int, int>> complete;
    complete.reserve(static_cast<size_t>(k) * static_cast<size_t>(k - 1) / 2);
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            complete.push_back({td[i][j], i, j});
        }
    }
    sort(complete.begin(), complete.end());

    DSU dsu(k);
    vector<int> selected;
    for (auto [d, i, j] : complete) {
        if (d == INF) continue;
        if (!dsu.unite(i, j)) continue;
        int cur = terminals[j];
        while (cur != terminals[i]) {
            int id = parent_e[i][cur];
            int pv = parent_v[i][cur];
            if (id < 0 || pv < 0) break;
            selected.push_back(id);
            cur = pv;
        }
    }
    return normalize_tree_edges(n, edges, terminals, selected);
}

static vector<int> greedy_heuristic(int n, const vector<Edge>& edges,
                                    const vector<vector<Arc>>& graph,
                                    const vector<int>& terminals) {
    int k = static_cast<int>(terminals.size());
    if (k <= 1) return {};

    vector<char> is_terminal(n, 0), in_tree(n, 0), done_terminal(k, 0);
    for (int t : terminals) is_terminal[t] = 1;
    (void)is_terminal;

    in_tree[terminals[0]] = 1;
    done_terminal[0] = 1;
    int done = 1;
    vector<int> selected;

    while (done < k) {
        vector<long long> dist(n, INF);
        vector<int> pv(n, -1), pe(n, -1);
        priority_queue<pair<long long, int>, vector<pair<long long, int>>,
                       greater<pair<long long, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (in_tree[v]) {
                dist[v] = 0;
                pq.push({0, v});
            }
        }
        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dist[v]) continue;
            for (const Arc& a : graph[v]) {
                long long nd = d + a.w;
                if (nd < dist[a.to]) {
                    dist[a.to] = nd;
                    pv[a.to] = v;
                    pe[a.to] = a.id;
                    pq.push({nd, a.to});
                }
            }
        }

        int best_i = -1;
        long long best = INF;
        for (int i = 0; i < k; ++i) {
            if (!done_terminal[i] && dist[terminals[i]] < best) {
                best = dist[terminals[i]];
                best_i = i;
            }
        }
        if (best_i < 0) break;

        int cur = terminals[best_i];
        vector<int> path_vertices;
        while (!in_tree[cur]) {
            if (pe[cur] < 0 || pv[cur] < 0) break;
            selected.push_back(pe[cur]);
            path_vertices.push_back(cur);
            cur = pv[cur];
        }
        for (int v : path_vertices) in_tree[v] = 1;
        in_tree[terminals[best_i]] = 1;
        done_terminal[best_i] = 1;
        ++done;
    }

    return normalize_tree_edges(n, edges, terminals, selected);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<Edge> edges;
    edges.reserve(m);
    vector<vector<Arc>> graph(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        graph[u].push_back({v, i, w});
        graph[v].push_back({u, i, w});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];
    sort(terminals.begin(), terminals.end());
    terminals.erase(unique(terminals.begin(), terminals.end()), terminals.end());
    k = static_cast<int>(terminals.size());

    vector<int> answer;
    size_t exact_cells = (k < 31) ? (static_cast<size_t>(1) << k) * static_cast<size_t>(max(0, n))
                                  : numeric_limits<size_t>::max();

    if (k <= 13 && exact_cells <= 30000000ULL) {
        answer = exact_steiner(n, edges, graph, terminals);
    } else if (static_cast<size_t>(k) * static_cast<size_t>(max(0, n)) <= 20000000ULL) {
        answer = metric_mst_heuristic(n, edges, graph, terminals);
    } else {
        answer = greedy_heuristic(n, edges, graph, terminals);
    }

    long long total = 0;
    for (int id : answer) total += edges[id].w;

    cout << total << ' ' << answer.size() << '\n';
    for (int id : answer) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
    return 0;
}
