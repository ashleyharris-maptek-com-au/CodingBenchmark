using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    private sealed class Constraint
    {
        public int[] Vars = Array.Empty<int>();
        public int Need;
    }

    public static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first)) return;

        string[] hw = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int h = int.Parse(hw[0]);
        int w = int.Parse(hw[1]);

        string[] board = new string[h];
        for (int r = 0; r < h; r++)
        {
            string line = Console.ReadLine() ?? "";
            if (line.Length < w) line = line.PadRight(w);
            board[r] = line;
        }

        long totalLong = (long)h * w;
        if (totalLong > int.MaxValue)
        {
            Console.WriteLine(0);
            return;
        }

        int total = (int)totalLong;
        byte[] state = new byte[total]; // 0 = undecided, 1 = inferred mine, 2 = inferred safe
        HashSet<int> safe = new HashSet<int>();

        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        int Index(int r, int c) => r * w + c;
        bool IsDigit(char ch) => ch >= '0' && ch <= '8';

        bool MarkSafe(int idx)
        {
            int r = idx / w, c = idx - r * w;
            if (board[r][c] != ' ' || state[idx] == 1 || state[idx] == 2) return false;
            state[idx] = 2;
            safe.Add(idx);
            return true;
        }

        bool MarkMine(int idx)
        {
            int r = idx / w, c = idx - r * w;
            if (board[r][c] != ' ' || state[idx] == 1 || state[idx] == 2) return false;
            state[idx] = 1;
            return true;
        }

        bool Propagate()
        {
            bool any = false;
            bool changed;
            List<int> unknowns = new List<int>(8);

            do
            {
                changed = false;
                for (int r = 0; r < h; r++)
                {
                    string row = board[r];
                    for (int c = 0; c < w; c++)
                    {
                        char ch = row[c];
                        if (!IsDigit(ch)) continue;

                        int need = ch - '0';
                        int knownMines = 0;
                        unknowns.Clear();

                        for (int k = 0; k < 8; k++)
                        {
                            int nr = r + dr[k], nc = c + dc[k];
                            if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                            int ni = Index(nr, nc);
                            char nch = board[nr][nc];

                            if (nch == '*' || state[ni] == 1) knownMines++;
                            else if (nch == ' ' && state[ni] == 0) unknowns.Add(ni);
                        }

                        int remaining = need - knownMines;
                        if (remaining < 0 || remaining > unknowns.Count) continue;

                        if (remaining == 0)
                        {
                            foreach (int idx in unknowns)
                                if (MarkSafe(idx)) changed = any = true;
                        }
                        else if (remaining == unknowns.Count)
                        {
                            foreach (int idx in unknowns)
                                if (MarkMine(idx)) changed = any = true;
                        }
                    }
                }
            } while (changed);

            return any;
        }

        List<Constraint> BuildConstraints(out Dictionary<int, int> varId, out List<int> varCell)
        {
            Dictionary<int, int> localVarId = new Dictionary<int, int>();
            List<int> localVarCell = new List<int>();
            List<Constraint> constraints = new List<Constraint>();
            List<int> vars = new List<int>(8);

            int GetVar(int cell)
            {
                if (localVarId.TryGetValue(cell, out int id)) return id;
                id = localVarCell.Count;
                localVarId[cell] = id;
                localVarCell.Add(cell);
                return id;
            }

            for (int r = 0; r < h; r++)
            {
                string row = board[r];
                for (int c = 0; c < w; c++)
                {
                    char ch = row[c];
                    if (!IsDigit(ch)) continue;

                    int need = ch - '0';
                    int knownMines = 0;
                    vars.Clear();

                    for (int k = 0; k < 8; k++)
                    {
                        int nr = r + dr[k], nc = c + dc[k];
                        if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                        int ni = Index(nr, nc);
                        char nch = board[nr][nc];

                        if (nch == '*' || state[ni] == 1) knownMines++;
                        else if (nch == ' ' && state[ni] == 0) vars.Add(GetVar(ni));
                    }

                    int remaining = need - knownMines;
                    if (remaining < 0 || remaining > vars.Count || vars.Count == 0) continue;

                    if (remaining == 0)
                    {
                        foreach (int v in vars) MarkSafe(localVarCell[v]);
                    }
                    else if (remaining == vars.Count)
                    {
                        foreach (int v in vars) MarkMine(localVarCell[v]);
                    }
                    else
                    {
                        constraints.Add(new Constraint { Vars = vars.ToArray(), Need = remaining });
                    }
                }
            }

            varId = localVarId;
            varCell = localVarCell;
            return constraints;
        }

        bool ExactReasoning()
        {
            List<Constraint> constraints = BuildConstraints(out Dictionary<int, int> varId, out List<int> varCell);
            if (constraints.Count == 0 || varCell.Count == 0) return false;

            List<int>[] varToCons = new List<int>[varCell.Count];
            for (int i = 0; i < varToCons.Length; i++) varToCons[i] = new List<int>();
            for (int ci = 0; ci < constraints.Count; ci++)
                foreach (int v in constraints[ci].Vars)
                    varToCons[v].Add(ci);

            bool[] seenVar = new bool[varCell.Count];
            bool[] seenCon = new bool[constraints.Count];
            Queue<int> q = new Queue<int>();
            bool changed = false;

            for (int start = 0; start < varCell.Count; start++)
            {
                if (seenVar[start]) continue;

                List<int> compVars = new List<int>();
                List<int> compCons = new List<int>();
                seenVar[start] = true;
                q.Enqueue(start);

                while (q.Count > 0)
                {
                    int v = q.Dequeue();
                    compVars.Add(v);

                    foreach (int ci in varToCons[v])
                    {
                        if (!seenCon[ci])
                        {
                            seenCon[ci] = true;
                            compCons.Add(ci);
                        }

                        foreach (int nv in constraints[ci].Vars)
                        {
                            if (!seenVar[nv])
                            {
                                seenVar[nv] = true;
                                q.Enqueue(nv);
                            }
                        }
                    }
                }

                if (compVars.Count == 0) continue;

                int maxVars = compVars.Count <= 24 ? 24 : 30;
                long nodeLimit = compVars.Count <= 24 ? 5_000_000L : 750_000L;
                if (compVars.Count > maxVars || compCons.Count > 120) continue;

                changed |= SolveComponent(compVars, compCons, constraints, varToCons, varCell, nodeLimit);
            }

            return changed;
        }

        bool SolveComponent(
            List<int> compVars,
            List<int> compCons,
            List<Constraint> constraints,
            List<int>[] varToCons,
            List<int> varCell,
            long nodeLimit)
        {
            Dictionary<int, int> conLocal = new Dictionary<int, int>();
            for (int i = 0; i < compCons.Count; i++) conLocal[compCons[i]] = i;

            int m = compCons.Count;
            int[] need = new int[m];
            int[] rem = new int[m];
            for (int i = 0; i < m; i++)
            {
                Constraint con = constraints[compCons[i]];
                need[i] = con.Need;
                rem[i] = con.Vars.Length;
            }

            compVars.Sort((a, b) => varToCons[b].Count.CompareTo(varToCons[a].Count));
            int n = compVars.Count;
            bool[] mineEver = new bool[n];
            bool[] safeEver = new bool[n];
            bool[] assign = new bool[n];
            long models = 0, nodes = 0;
            bool aborted = false;

            bool Apply(int var, bool mine, List<int> touched)
            {
                foreach (int ci in varToCons[var])
                {
                    if (!conLocal.TryGetValue(ci, out int li)) continue;
                    touched.Add(li);
                    rem[li]--;
                    if (mine) need[li]--;
                    if (need[li] < 0 || need[li] > rem[li]) return false;
                }
                return true;
            }

            void Undo(bool mine, List<int> touched)
            {
                for (int i = touched.Count - 1; i >= 0; i--)
                {
                    int li = touched[i];
                    if (mine) need[li]++;
                    rem[li]++;
                }
                touched.Clear();
            }

            void Dfs(int pos)
            {
                if (aborted) return;
                if (++nodes > nodeLimit)
                {
                    aborted = true;
                    return;
                }

                if (pos == n)
                {
                    for (int i = 0; i < m; i++)
                        if (need[i] != 0) return;

                    models++;
                    for (int i = 0; i < n; i++)
                    {
                        if (assign[i]) mineEver[i] = true;
                        else safeEver[i] = true;
                    }
                    return;
                }

                int var = compVars[pos];
                List<int> touched = new List<int>(varToCons[var].Count);

                assign[pos] = false;
                if (Apply(var, false, touched)) Dfs(pos + 1);
                Undo(false, touched);

                assign[pos] = true;
                if (Apply(var, true, touched)) Dfs(pos + 1);
                Undo(true, touched);
            }

            Dfs(0);
            if (aborted || models == 0) return false;

            bool changed = false;
            for (int i = 0; i < n; i++)
            {
                int cell = varCell[compVars[i]];
                if (safeEver[i] && !mineEver[i]) changed |= MarkSafe(cell);
                else if (mineEver[i] && !safeEver[i]) changed |= MarkMine(cell);
            }

            return changed;
        }

        bool changedOuter;
        int rounds = 0;
        do
        {
            changedOuter = Propagate();
            changedOuter |= ExactReasoning();
            rounds++;
        } while (changedOuter && rounds < 20);

        StringBuilder output = new StringBuilder();
        output.AppendLine(safe.Count.ToString());
        foreach (int idx in safe)
        {
            int y = idx / w;
            int x = idx - y * w;
            output.Append(x).Append(' ').Append(y).AppendLine();
        }

        Console.Write(output.ToString());
    }
}
