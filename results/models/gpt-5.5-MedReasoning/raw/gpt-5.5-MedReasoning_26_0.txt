use std::cmp::Ordering;
use std::fs::{remove_file, File};
use std::io::{self, BufReader, BufWriter, Read, Write};
use std::path::PathBuf;

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    fn coord(self, axis: u8) -> f32 {
        match axis {
            0 => self.x,
            1 => self.y,
            _ => self.z,
        }
    }

    fn dist2(self, other: Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

#[derive(Clone, Copy)]
struct KdNode {
    centroid: usize,
    left: Option<usize>,
    right: Option<usize>,
    axis: u8,
}

struct KdTree {
    nodes: Vec<KdNode>,
    root: Option<usize>,
}

impl KdTree {
    fn new(centroids: &[Point]) -> Self {
        let mut order: Vec<usize> = (0..centroids.len()).collect();
        let mut nodes = Vec::with_capacity(centroids.len());
        let root = Self::build(&mut order, centroids, &mut nodes);
        Self { nodes, root }
    }

    fn build(order: &mut [usize], centroids: &[Point], nodes: &mut Vec<KdNode>) -> Option<usize> {
        if order.is_empty() {
            return None;
        }

        let axis = widest_axis(order, centroids);
        order.sort_unstable_by(|&a, &b| {
            centroids[a]
                .coord(axis)
                .partial_cmp(&centroids[b].coord(axis))
                .unwrap_or(Ordering::Equal)
        });

        let mid = order.len() / 2;
        let (left_order, rest) = order.split_at_mut(mid);
        let centroid = rest[0];
        let right_order = &mut rest[1..];

        let node_index = nodes.len();
        nodes.push(KdNode {
            centroid,
            left: None,
            right: None,
            axis,
        });

        let left = Self::build(left_order, centroids, nodes);
        let right = Self::build(right_order, centroids, nodes);
        nodes[node_index].left = left;
        nodes[node_index].right = right;
        Some(node_index)
    }

    fn nearest(&self, point: Point, centroids: &[Point]) -> usize {
        let Some(root) = self.root else {
            return 0;
        };
        let mut best = (self.nodes[root].centroid, f32::INFINITY);
        self.nearest_rec(root, point, centroids, &mut best);
        best.0
    }

    fn nearest_rec(
        &self,
        node_index: usize,
        point: Point,
        centroids: &[Point],
        best: &mut (usize, f32),
    ) {
        let node = self.nodes[node_index];
        let center = centroids[node.centroid];
        let d = point.dist2(center);
        if d < best.1 {
            *best = (node.centroid, d);
        }

        let delta = point.coord(node.axis) - center.coord(node.axis);
        let (near, far) = if delta <= 0.0 {
            (node.left, node.right)
        } else {
            (node.right, node.left)
        };

        if let Some(child) = near {
            self.nearest_rec(child, point, centroids, best);
        }
        if delta * delta < best.1 {
            if let Some(child) = far {
                self.nearest_rec(child, point, centroids, best);
            }
        }
    }
}

fn widest_axis(order: &[usize], centroids: &[Point]) -> u8 {
    let mut min = Point {
        x: f32::INFINITY,
        y: f32::INFINITY,
        z: f32::INFINITY,
    };
    let mut max = Point {
        x: f32::NEG_INFINITY,
        y: f32::NEG_INFINITY,
        z: f32::NEG_INFINITY,
    };

    for &i in order {
        let p = centroids[i];
        min.x = min.x.min(p.x);
        min.y = min.y.min(p.y);
        min.z = min.z.min(p.z);
        max.x = max.x.max(p.x);
        max.y = max.y.max(p.y);
        max.z = max.z.max(p.z);
    }

    let sx = max.x - min.x;
    let sy = max.y - min.y;
    let sz = max.z - min.z;
    if sx >= sy && sx >= sz {
        0
    } else if sy >= sz {
        1
    } else {
        2
    }
}

struct Rng(u64);

impl Rng {
    fn new(seed: u64) -> Self {
        Self(seed.max(1))
    }

    fn next_u64(&mut self) -> u64 {
        let mut x = self.0;
        x ^= x << 7;
        x ^= x >> 9;
        x = x.wrapping_mul(0x9E37_79B9_7F4A_7C15);
        self.0 = x;
        x
    }

    fn usize(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            0
        } else {
            (self.next_u64() as usize) % upper
        }
    }

    fn u64_below(&mut self, upper: u64) -> u64 {
        if upper <= 1 {
            0
        } else {
            self.next_u64() % upper
        }
    }
}

fn read_header<R: Read>(reader: &mut R) -> io::Result<(u64, usize)> {
    let mut header = Vec::with_capacity(64);
    let mut byte = [0u8; 1];
    loop {
        reader.read_exact(&mut byte)?;
        header.push(byte[0]);
        if byte[0] == b'\n' {
            break;
        }
    }

    let text = std::str::from_utf8(&header)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "header is not UTF-8"))?;
    let mut fields = text.split_whitespace();
    let n = fields
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing point count"))?
        .parse::<u64>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "bad point count"))?;
    let k = fields
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing cluster count"))?
        .parse::<usize>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "bad cluster count"))?;
    Ok((n, k))
}

fn point_from_bytes(bytes: &[u8; 12]) -> Point {
    Point {
        x: f32::from_le_bytes(bytes[0..4].try_into().unwrap()),
        y: f32::from_le_bytes(bytes[4..8].try_into().unwrap()),
        z: f32::from_le_bytes(bytes[8..12].try_into().unwrap()),
    }
}

fn choose_sample_target(n: u64, k: usize) -> usize {
    if n == 0 || k == 0 {
        return 0;
    }
    let per_cluster = if n <= 10_000_000 {
        80
    } else if n <= 100_000_000 {
        50
    } else {
        25
    };
    let wanted = k.saturating_mul(per_cluster).max(20_000);
    let cap = if n <= 10_000_000 {
        1_000_000
    } else {
        600_000
    };
    wanted.min(cap).min(n as usize)
}

fn initialize_centroids(sample: &[Point], k: usize, rng: &mut Rng) -> Vec<Point> {
    let mut centroids = Vec::with_capacity(k);
    if sample.is_empty() || k == 0 {
        return centroids;
    }

    let stride = (sample.len() / k.max(1)).max(1);
    let offset = rng.usize(sample.len());
    for i in 0..k {
        let idx = (offset + i.saturating_mul(stride)) % sample.len();
        centroids.push(sample[idx]);
    }
    centroids
}

fn train_centroids(sample: &[Point], k: usize, n: u64) -> Vec<Point> {
    let mut rng = Rng::new(n ^ ((k as u64) << 32) ^ 0xD1B5_4A32_D192_ED03);
    let mut centroids = initialize_centroids(sample, k, &mut rng);
    if centroids.is_empty() {
        return centroids;
    }

    let iterations = if n <= 10_000_000 {
        24
    } else if n <= 100_000_000 {
        16
    } else if n <= 1_000_000_000 {
        10
    } else {
        6
    };

    for _ in 0..iterations {
        let tree = KdTree::new(&centroids);
        let mut sums = vec![Point::default(); k];
        let mut counts = vec![0u32; k];

        for &p in sample {
            let c = tree.nearest(p, &centroids);
            sums[c].x += p.x;
            sums[c].y += p.y;
            sums[c].z += p.z;
            counts[c] += 1;
        }

        let mut max_shift = 0.0f32;
        for i in 0..k {
            let next = if counts[i] > 0 {
                let inv = 1.0 / counts[i] as f32;
                Point {
                    x: sums[i].x * inv,
                    y: sums[i].y * inv,
                    z: sums[i].z * inv,
                }
            } else {
                sample[rng.usize(sample.len())]
            };
            max_shift = max_shift.max(centroids[i].dist2(next));
            centroids[i] = next;
        }

        if max_shift < 1.0e-10 {
            break;
        }
    }

    centroids
}

fn temp_path() -> PathBuf {
    let mut path = std::env::temp_dir();
    path.push(format!(
        "rust_streaming_kmeans_{}_{}.bin",
        std::process::id(),
        std::time::SystemTime::now()
            .duration_since(std::time::UNIX_EPOCH)
            .map(|d| d.as_nanos())
            .unwrap_or(0)
    ));
    path
}

fn emit_assignment<W: Write>(
    writer: &mut W,
    point: Point,
    tree: &KdTree,
    centroids: &[Point],
) -> io::Result<()> {
    let c = tree.nearest(point, centroids) as u32;
    writer.write_all(&c.to_le_bytes())
}

fn run() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = BufReader::with_capacity(1 << 20, stdin.lock());
    let (n, k) = read_header(&mut input)?;

    if k == 0 {
        return Ok(());
    }

    let keep_in_memory = n <= 12_000_000;
    let sample_target = choose_sample_target(n, k);
    let mut sample = Vec::with_capacity(sample_target);
    let mut points = if keep_in_memory {
        Some(Vec::with_capacity(n as usize))
    } else {
        None
    };

    let path = if keep_in_memory { None } else { Some(temp_path()) };
    let mut spool = match &path {
        Some(p) => Some(BufWriter::with_capacity(1 << 20, File::create(p)?)),
        None => None,
    };

    let mut rng = Rng::new(n ^ 0xA076_1D64_78BD_642F);
    let mut raw = [0u8; 12];
    for seen in 0..n {
        input.read_exact(&mut raw)?;
        let p = point_from_bytes(&raw);

        if let Some(v) = points.as_mut() {
            v.push(p);
        }
        if let Some(w) = spool.as_mut() {
            w.write_all(&raw)?;
        }

        if sample.len() < sample_target {
            sample.push(p);
        } else if sample_target > 0 {
            let j = rng.u64_below(seen + 1);
            if j < sample_target as u64 {
                sample[j as usize] = p;
            }
        }
    }
    drop(input);
    if let Some(w) = spool.as_mut() {
        w.flush()?;
    }

    let centroids = train_centroids(&sample, k, n);
    let tree = KdTree::new(&centroids);
    let stdout = io::stdout();
    let mut output = BufWriter::with_capacity(1 << 20, stdout.lock());

    if let Some(v) = points {
        for p in v {
            emit_assignment(&mut output, p, &tree, &centroids)?;
        }
    } else if let Some(p) = &path {
        let mut file = BufReader::with_capacity(1 << 20, File::open(p)?);
        for _ in 0..n {
            file.read_exact(&mut raw)?;
            emit_assignment(&mut output, point_from_bytes(&raw), &tree, &centroids)?;
        }
        let _ = remove_file(p);
    }

    output.flush()
}

fn main() {
    if let Err(_) = run() {
        std::process::exit(1);
    }
}
