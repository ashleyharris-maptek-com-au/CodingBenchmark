import math


_ap_mem = {
    "last_time": None,
    "last_alt": None,
    "last_ias": None,
    "last_hdg": None,
    "alt_i": 0.0,
    "ias_i": 0.0,
    "vs_i": 0.0,
    "trim_i": 0.0,
    "last_elev": 0.0,
    "last_ail": 0.0,
    "last_thr": 0.55,
}


def _clamp(x, lo, hi):
    try:
        if x != x:
            return (lo + hi) * 0.5
        return max(lo, min(hi, float(x)))
    except Exception:
        return (lo + hi) * 0.5


def _num(state, key, default=0.0):
    try:
        v = state.get(key, default)
        if v is None or v != v:
            return default
        return float(v)
    except Exception:
        return default


def _bool(state, key, default=False):
    return bool(state.get(key, default))


def _wrap180(deg):
    return (float(deg) + 180.0) % 360.0 - 180.0


def _median(xs):
    xs = sorted(float(x) for x in xs if x is not None and x == x)
    n = len(xs)
    if not n:
        return None
    m = n // 2
    if n & 1:
        return xs[m]
    return 0.5 * (xs[m - 1] + xs[m])


def _fuse_triplet(state, stem, suffix, lo, hi, valid_suffix="_valid", prior=None, max_rate=None, dt=0.05):
    vals = []
    for i in (1, 2, 3):
        key = f"{stem}_{i}_{suffix}"
        if key not in state:
            continue
        v = _num(state, key, None)
        if v is None or v < lo or v > hi:
            continue
        score = 1.0
        if valid_suffix and f"{stem}_{i}{valid_suffix}" in state and not _bool(state, f"{stem}_{i}{valid_suffix}", True):
            score -= 0.35
        if prior is not None and max_rate is not None:
            if abs(v - prior) > max_rate * max(dt, 0.02) + 2.0:
                score -= 0.45
        vals.append((v, score))
    if not vals:
        return prior if prior is not None else (lo + hi) * 0.5
    med = _median([v for v, _ in vals])
    survivors = []
    for v, score in vals:
        if abs(v - med) <= max(3.0, 0.08 * max(abs(med), 1.0)) or len(vals) <= 2:
            survivors.append((v, max(score, 0.05)))
    if not survivors:
        survivors = vals
    sw = sum(w for _, w in survivors)
    return sum(v * w for v, w in survivors) / sw


def _fuse_angle_triplet(state, stem, suffix, valid_suffix="_valid", prior=None):
    vals = []
    ref = prior
    for i in (1, 2, 3):
        key = f"{stem}_{i}_{suffix}"
        if key not in state:
            continue
        v = _num(state, key, None)
        if v is None:
            continue
        if ref is None:
            ref = v
        v = ref + _wrap180(v - ref)
        score = 1.0
        if valid_suffix and f"{stem}_{i}{valid_suffix}" in state and not _bool(state, f"{stem}_{i}{valid_suffix}", True):
            score -= 0.35
        vals.append((v, max(score, 0.05)))
    if not vals:
        return (prior or 0.0) % 360.0
    med = _median([v for v, _ in vals])
    survivors = [(v, w) for v, w in vals if abs(v - med) <= 20.0 or len(vals) <= 2]
    if not survivors:
        survivors = vals
    sw = sum(w for _, w in survivors)
    return (sum(v * w for v, w in survivors) / sw) % 360.0


def _rate_limit(new, old, max_delta):
    return old + _clamp(new - old, -max_delta, max_delta)


def autopilot_step(state: dict, dt: float) -> dict:
    global _ap_mem

    dt = _clamp(dt or state.get("dt", 0.05), 0.01, 0.25)
    t = _num(state, "sim_time_s", 0.0)
    if _ap_mem["last_time"] is not None and t + 0.5 < _ap_mem["last_time"]:
        _ap_mem = {
            "last_time": None, "last_alt": None, "last_ias": None, "last_hdg": None,
            "alt_i": 0.0, "ias_i": 0.0, "vs_i": 0.0, "trim_i": 0.0,
            "last_elev": 0.0, "last_ail": 0.0, "last_thr": 0.55,
        }
    _ap_mem["last_time"] = t

    ias = _fuse_triplet(state, "ias", "kt", 40.0, 450.0, prior=_ap_mem["last_ias"], max_rate=80.0, dt=dt)
    mach = _median([_num(state, f"mach_{i}", None) for i in (1, 2, 3)])
    if mach is None:
        mach = 0.0
    mach = _clamp(mach, 0.0, 0.95)
    alt_baro = _fuse_triplet(state, "baro_alt", "ft", -1500.0, 50000.0, prior=_ap_mem["last_alt"], max_rate=9000.0 / 60.0, dt=dt)
    alt_gps = _fuse_triplet(state, "gps", "alt_ft", -1500.0, 50000.0, prior=_ap_mem["last_alt"], max_rate=12000.0 / 60.0, dt=dt)
    alt = alt_baro
    if abs(alt_gps - alt_baro) < 250.0:
        alt = 0.70 * alt_baro + 0.30 * alt_gps
    elif _ap_mem["last_alt"] is not None:
        # Prefer the source whose short-term motion is most physically continuous.
        rb = abs((alt_baro - _ap_mem["last_alt"]) / dt)
        rg = abs((alt_gps - _ap_mem["last_alt"]) / dt)
        alt = alt_baro if rb <= rg else alt_gps
    radar_alt = _fuse_triplet(state, "radar_alt", "ft", -20.0, 8000.0, prior=None, max_rate=None, dt=dt)

    pitch = _fuse_triplet(state, "irs", "pitch_deg", -35.0, 35.0, prior=None, max_rate=None, dt=dt)
    roll = _fuse_triplet(state, "irs", "roll_deg", -80.0, 80.0, prior=None, max_rate=None, dt=dt)
    hdg = _fuse_angle_triplet(state, "irs", "heading_deg", prior=_ap_mem["last_hdg"])
    if "magnetic_heading_deg" in state:
        nav_hdg = _num(state, "magnetic_heading_deg", hdg)
        if abs(_wrap180(nav_hdg - hdg)) < 25.0:
            hdg = 0.6 * hdg + 0.4 * (hdg + _wrap180(nav_hdg - hdg))

    vs_irs = _fuse_triplet(state, "irs", "vs_fpm", -12000.0, 12000.0, prior=None, max_rate=None, dt=dt)
    vs_nav = _num(state, "vertical_speed_fpm", vs_irs)
    vs = vs_irs if abs(vs_irs - vs_nav) > 1500.0 else 0.55 * vs_irs + 0.45 * vs_nav
    if _ap_mem["last_alt"] is not None:
        derived_vs = (alt - _ap_mem["last_alt"]) / dt * 60.0
        if abs(derived_vs - vs) < 2500.0:
            vs = 0.65 * vs + 0.35 * derived_vs

    aoa = _fuse_triplet(state, "aoa", "deg", -10.0, 35.0, prior=None, max_rate=None, dt=dt)
    roll_rate = _fuse_triplet(state, "gyro", "roll_rate_dps", -120.0, 120.0, prior=None, max_rate=None, dt=dt)
    pitch_rate = _fuse_triplet(state, "gyro", "pitch_rate_dps", -80.0, 80.0, prior=None, max_rate=None, dt=dt)
    yaw_rate = _fuse_triplet(state, "gyro", "yaw_rate_dps", -80.0, 80.0, prior=None, max_rate=None, dt=dt)
    nz = _fuse_triplet(state, "accel", "z_g", -2.0, 4.0, prior=None, max_rate=None, dt=dt)

    target_alt = _clamp(_num(state, "target_alt_ft", alt), -1000.0, 43000.0)
    target_hdg = _num(state, "target_hdg_deg", hdg) % 360.0
    target_ias = _clamp(_num(state, "target_ias_kt", 250.0), 140.0, 340.0)
    target_vs = _clamp(_num(state, "target_vs_fpm", 0.0), -2500.0, 2500.0)

    gw = _clamp(_num(state, "gross_weight_kg", 65000.0), 35000.0, 90000.0)
    flaps_now = max(_num(state, "flap_pos_left_deg", 0.0), _num(state, "flap_pos_right_deg", 0.0))
    stall_speed = 118.0 + 0.00075 * max(0.0, gw - 55000.0) - min(flaps_now, 30.0) * 0.75
    min_safe = max(135.0, stall_speed + 28.0)
    low_alt = radar_alt < 2500.0

    eng1_ok = _bool(state, "engine_1_running", True) and not _bool(state, "engine_1_fire", False)
    eng2_ok = _bool(state, "engine_2_running", True) and not _bool(state, "engine_2_fire", False)
    e1_thrust = _num(state, "engine_1_thrust_pct", 100.0 if eng1_ok else 0.0)
    e2_thrust = _num(state, "engine_2_thrust_pct", 100.0 if eng2_ok else 0.0)

    out = {}

    # Envelope protections override navigation.  They are intentionally simple,
    # because preserving energy and attitude matters more than tracking targets.
    stall = (aoa > 11.5) or (ias < min_safe - 8.0 and pitch > 4.0) or (nz > 1.8 and ias < min_safe + 15.0)
    overspeed = ias > 365.0 or mach > 0.82
    ground_danger = radar_alt < 900.0 and (vs < -900.0 or pitch < -6.0) and not (_bool(state, "gear_left_locked") and _bool(state, "gear_right_locked"))

    if stall:
        desired_roll = 0.0
        aileron = _clamp((-roll * 0.045) - roll_rate * 0.018, -0.65, 0.65)
        elevator = _clamp(-0.55 - max(0.0, aoa - 12.0) * 0.035 - pitch_rate * 0.025, -1.0, 0.05)
        throttle = 1.0
        out.update({
            "elevator": elevator,
            "aileron": aileron,
            "rudder": _clamp(-yaw_rate * 0.045 + (e1_thrust - e2_thrust) * 0.004, -0.55, 0.55),
            "throttle_1": 1.0 if eng1_ok else 0.0,
            "throttle_2": 1.0 if eng2_ok else 0.0,
            "elevator_trim": -0.25,
            "flaps": 0 if ias > 210.0 else 15,
            "speedbrake": 0.0,
            "gear": 1 if radar_alt < 1200.0 else 0,
        })
        _ap_mem["last_alt"], _ap_mem["last_ias"], _ap_mem["last_hdg"] = alt, ias, hdg
        _ap_mem["last_elev"], _ap_mem["last_ail"], _ap_mem["last_thr"] = elevator, aileron, throttle
        return out

    alt_err = _clamp(target_alt - alt, -6000.0, 6000.0)
    capture_vs = _clamp(alt_err * 0.45, -1800.0, 1800.0)
    if abs(alt_err) > 700.0 and abs(target_vs) > 100.0:
        desired_vs = target_vs if (target_vs * alt_err) > 0.0 else capture_vs
    else:
        desired_vs = capture_vs
    desired_vs = _clamp(desired_vs, -2200.0, 2200.0)
    if low_alt and desired_vs < -900.0:
        desired_vs = -900.0
    if ground_danger:
        desired_vs = 1500.0

    _ap_mem["vs_i"] = _clamp(_ap_mem["vs_i"] + (desired_vs - vs) * dt / 600.0, -0.35, 0.35)
    pitch_cmd = _clamp((desired_vs - vs) * 0.0030 + _ap_mem["vs_i"], -8.0, 8.0)
    if alt_err > 150.0 and ias < min_safe + 10.0:
        pitch_cmd = min(pitch_cmd, 4.0)
    if overspeed:
        pitch_cmd = max(pitch_cmd, 2.5)
    if ground_danger:
        pitch_cmd = max(pitch_cmd, 8.0)

    elevator = (pitch_cmd - pitch) * 0.085 - pitch_rate * 0.035
    elevator = _clamp(elevator, -0.85, 0.85)

    hdg_err = _wrap180(target_hdg - hdg)
    bank_limit = 28.0
    if ias < min_safe + 15.0 or radar_alt < 700.0:
        bank_limit = 15.0
    desired_roll = _clamp(hdg_err * 0.55, -bank_limit, bank_limit)
    aileron = _clamp((desired_roll - roll) * 0.045 - roll_rate * 0.020, -0.85, 0.85)

    rudder = _clamp(-yaw_rate * 0.050 - _wrap180(desired_roll - roll) * 0.004 + (e1_thrust - e2_thrust) * 0.0045, -0.6, 0.6)

    speed_target = max(target_ias, min_safe)
    if target_alt > 10000.0:
        speed_target = min(speed_target, 320.0)
    if low_alt:
        speed_target = min(speed_target, 250.0)
    if overspeed:
        speed_target = min(speed_target, 300.0)

    speed_err = _clamp(speed_target - ias, -120.0, 120.0)
    _ap_mem["ias_i"] = _clamp(_ap_mem["ias_i"] + speed_err * dt * 0.003, -0.25, 0.25)
    base_thr = 0.50 + max(0.0, desired_vs) / 7000.0 - max(0.0, -desired_vs) / 12000.0
    throttle = _clamp(base_thr + speed_err * 0.010 + _ap_mem["ias_i"], 0.05, 1.0)
    if overspeed:
        throttle = 0.02
    if ground_danger:
        throttle = 1.0
    if ias < min_safe:
        throttle = 1.0

    throttle = _rate_limit(throttle, _ap_mem["last_thr"], 0.35 * dt)
    thr1 = throttle if eng1_ok else 0.0
    thr2 = throttle if eng2_ok else 0.0
    if eng1_ok and not eng2_ok:
        thr1 = min(thr1, 0.82 if ias > min_safe + 10.0 else 1.0)
    if eng2_ok and not eng1_ok:
        thr2 = min(thr2, 0.82 if ias > min_safe + 10.0 else 1.0)

    if overspeed:
        speedbrake = 0.55 if roll < 20.0 and radar_alt > 1000.0 else 0.25
    elif ias > speed_target + 25.0 and desired_vs <= 0.0 and radar_alt > 800.0:
        speedbrake = 0.25
    else:
        speedbrake = 0.0

    if radar_alt < 1800.0 and target_alt < alt + 700.0 and ias < 230.0:
        gear = 1
    else:
        gear = 0 if alt > 2500.0 and radar_alt > 1800.0 else int(_bool(state, "gear_handle_down", False))

    if ias > 235.0:
        flaps = 0
    elif gear and radar_alt < 1800.0:
        flaps = 25 if ias < 185.0 else 15
    elif ias < 180.0 and radar_alt < 3000.0:
        flaps = 10
    else:
        flaps = 0

    _ap_mem["trim_i"] = _clamp(_ap_mem["trim_i"] + elevator * dt * 0.08, -0.6, 0.6)
    trim_cmd = _clamp(_ap_mem["trim_i"], -1.0, 1.0)

    elevator = _rate_limit(elevator, _ap_mem["last_elev"], 4.0 * dt)
    aileron = _rate_limit(aileron, _ap_mem["last_ail"], 5.0 * dt)

    out.update({
        "elevator": _clamp(elevator, -1.0, 1.0),
        "aileron": _clamp(aileron, -1.0, 1.0),
        "rudder": _clamp(rudder, -1.0, 1.0),
        "throttle_1": _clamp(thr1, 0.0, 1.0),
        "throttle_2": _clamp(thr2, 0.0, 1.0),
        "elevator_trim": trim_cmd,
        "flaps": int(_clamp(flaps, 0, 40)),
        "speedbrake": _clamp(speedbrake, 0.0, 1.0),
        "gear": int(gear),
    })

    _ap_mem["last_alt"] = alt
    _ap_mem["last_ias"] = ias
    _ap_mem["last_hdg"] = hdg
    _ap_mem["last_elev"] = out["elevator"]
    _ap_mem["last_ail"] = out["aileron"]
    _ap_mem["last_thr"] = throttle
    return out
