use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Eq, PartialEq)]
struct HeapItem {
    gain: usize,
    degree: usize,
    vertex: usize,
}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        self.gain
            .cmp(&other.gain)
            .then_with(|| self.degree.cmp(&other.degree))
            .then_with(|| other.vertex.cmp(&self.vertex))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

struct Scanner {
    data: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { data: input, pos: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            value = value * 10 + (self.data[self.pos] - b'0') as usize;
            self.pos += 1;
        }
        value
    }
}

fn gain(v: usize, adj: &[Vec<usize>], dominated: &[bool]) -> usize {
    let mut g = usize::from(!dominated[v]);
    for &u in &adj[v] {
        if !dominated[u] {
            g += 1;
        }
    }
    g
}

fn add_vertex(v: usize, adj: &[Vec<usize>], selected: &mut [bool], dom_count: &mut [u16]) {
    if selected[v] {
        return;
    }
    selected[v] = true;
    dom_count[v] = dom_count[v].saturating_add(1);
    for &u in &adj[v] {
        dom_count[u] = dom_count[u].saturating_add(1);
    }
}

fn remove_vertex(v: usize, adj: &[Vec<usize>], selected: &mut [bool], dom_count: &mut [u16]) {
    if !selected[v] {
        return;
    }
    selected[v] = false;
    dom_count[v] -= 1;
    for &u in &adj[v] {
        dom_count[u] -= 1;
    }
}

fn removable(v: usize, adj: &[Vec<usize>], dom_count: &[u16]) -> bool {
    if dom_count[v] <= 1 {
        return false;
    }
    for &u in &adj[v] {
        if dom_count[u] <= 1 {
            return false;
        }
    }
    true
}

fn prune(adj: &[Vec<usize>], selected: &mut [bool], dom_count: &mut [u16]) -> bool {
    let mut changed = false;
    loop {
        let mut order: Vec<usize> = (0..selected.len()).filter(|&v| selected[v]).collect();
        order.sort_unstable_by_key(|&v| adj[v].len());

        let mut removed_this_round = false;
        for v in order {
            if selected[v] && removable(v, adj, dom_count) {
                remove_vertex(v, adj, selected, dom_count);
                changed = true;
                removed_this_round = true;
            }
        }
        if !removed_this_round {
            break;
        }
    }
    changed
}

fn greedy_solution(adj: &[Vec<usize>]) -> (Vec<bool>, Vec<u16>) {
    let n = adj.len();
    let mut dominated = vec![false; n];
    let mut selected = vec![false; n];
    let mut dom_count = vec![0u16; n];
    let mut heap = BinaryHeap::new();

    for v in 0..n {
        heap.push(HeapItem {
            gain: adj[v].len() + 1,
            degree: adj[v].len(),
            vertex: v,
        });
    }

    let mut undominated = n;
    while undominated > 0 {
        let Some(item) = heap.pop() else {
            break;
        };
        if selected[item.vertex] {
            continue;
        }
        let current_gain = gain(item.vertex, adj, &dominated);
        if current_gain == 0 {
            continue;
        }
        if current_gain != item.gain {
            heap.push(HeapItem {
                gain: current_gain,
                degree: adj[item.vertex].len(),
                vertex: item.vertex,
            });
            continue;
        }

        let v = item.vertex;
        selected[v] = true;
        if !dominated[v] {
            dominated[v] = true;
            undominated -= 1;
        }
        dom_count[v] = dom_count[v].saturating_add(1);
        for &u in &adj[v] {
            if !dominated[u] {
                dominated[u] = true;
                undominated -= 1;
            }
            dom_count[u] = dom_count[u].saturating_add(1);
        }
    }

    (selected, dom_count)
}

fn improve_by_swaps(
    adj: &[Vec<usize>],
    selected: &mut [bool],
    dom_count: &mut [u16],
    deadline: Instant,
) {
    let n = adj.len();
    if n == 0 {
        return;
    }

    let mut vertices: Vec<usize> = (0..n).collect();
    vertices.sort_unstable_by_key(|&v| std::cmp::Reverse(adj[v].len()));

    let candidate_limit = if n <= 4_000 {
        n
    } else if n <= 25_000 {
        6_000
    } else {
        2_000
    };

    let mut covered_stamp = vec![0u32; n];
    let mut loss = vec![0u16; n];
    let mut touched = Vec::new();
    let mut stamp = 1u32;

    loop {
        if Instant::now() >= deadline {
            return;
        }

        let mut changed = false;
        for &t in vertices.iter().take(candidate_limit) {
            if Instant::now() >= deadline {
                return;
            }
            if selected[t] {
                continue;
            }

            stamp = stamp.wrapping_add(1);
            if stamp == 0 {
                covered_stamp.fill(0);
                stamp = 1;
            }
            covered_stamp[t] = stamp;
            for &u in &adj[t] {
                covered_stamp[u] = stamp;
            }

            touched.clear();
            let mut removed = Vec::new();
            let mut candidates = Vec::new();
            if selected[t] {
                candidates.push(t);
            }
            for &s in &adj[t] {
                if selected[s] {
                    candidates.push(s);
                }
            }
            candidates.sort_unstable_by_key(|&s| adj[s].len());
            candidates.dedup();

            for s in candidates {
                if !selected[s] {
                    continue;
                }
                let mut ok = true;

                let after_t = dom_count[s] + u16::from(covered_stamp[s] == stamp);
                if after_t <= loss[s] + 1 {
                    ok = false;
                }
                if ok {
                    for &w in &adj[s] {
                        let after_t = dom_count[w] + u16::from(covered_stamp[w] == stamp);
                        if after_t <= loss[w] + 1 {
                            ok = false;
                            break;
                        }
                    }
                }

                if ok {
                    removed.push(s);
                    if loss[s] == 0 {
                        touched.push(s);
                    }
                    loss[s] += 1;
                    for &w in &adj[s] {
                        if loss[w] == 0 {
                            touched.push(w);
                        }
                        loss[w] += 1;
                    }
                }
            }

            if removed.len() >= 2 {
                add_vertex(t, adj, selected, dom_count);
                for s in removed {
                    remove_vertex(s, adj, selected, dom_count);
                }
                for &w in &touched {
                    loss[w] = 0;
                }
                prune(adj, selected, dom_count);
                changed = true;
                break;
            }

            for &w in &touched {
                loss[w] = 0;
            }
        }

        if !changed {
            return;
        }
    }
}

fn validate(adj: &[Vec<usize>], selected: &[bool]) -> bool {
    for v in 0..adj.len() {
        if selected[v] {
            continue;
        }
        if !adj[v].iter().any(|&u| selected[u]) {
            return false;
        }
    }
    true
}

fn main() {
    let start = Instant::now();
    let deadline = start + Duration::from_secs(285);

    let mut sc = Scanner::new();
    let n = sc.next_usize();
    let m = sc.next_usize();
    let mut adj = vec![Vec::<usize>::new(); n];

    for _ in 0..m {
        let u = sc.next_usize();
        let v = sc.next_usize();
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    let (mut selected, mut dom_count) = greedy_solution(&adj);
    prune(&adj, &mut selected, &mut dom_count);
    improve_by_swaps(&adj, &mut selected, &mut dom_count, deadline);
    prune(&adj, &mut selected, &mut dom_count);

    if !validate(&adj, &selected) {
        selected.fill(false);
        dom_count.fill(0);
        for v in 0..n {
            add_vertex(v, &adj, &mut selected, &mut dom_count);
        }
    }

    let solution: Vec<usize> = (0..n).filter(|&v| selected[v]).collect();
    println!("{}", solution.len());
    for (i, v) in solution.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{v}");
    }
    println!();
}
