using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        int[] head = new int[n];
        Array.Fill(head, -1);

        int arcs = Math.Max(0, m * 2);
        int[] to = new int[arcs];
        int[] next = new int[arcs];
        int[] edgeId = new int[arcs];
        int arc = 0;

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            fs.NextInt(); // Weight is irrelevant for a valid inspection route.

            to[arc] = v;
            edgeId[arc] = i;
            next[arc] = head[u];
            head[u] = arc++;

            to[arc] = u;
            edgeId[arc] = i;
            next[arc] = head[v];
            head[v] = arc++;
        }

        bool[] used = new bool[m];
        int[] iter = new int[n];
        Array.Copy(head, iter, n);

        int[] stack = new int[m + 1];
        int top = 0;
        stack[0] = 0;

        var output = new StringBuilder(Math.Min(1 << 24, Math.Max(16, (2 * m + 1) * 4)));
        output.Append(2 * m + 1).Append(' ');
        output.Append(0);

        int traversed = 0;
        while (top >= 0)
        {
            int u = stack[top];
            int a = iter[u];

            while (a != -1 && used[edgeId[a]])
                a = next[a];

            iter[u] = a;

            if (a == -1)
            {
                top--;
                if (top >= 0)
                    output.Append(' ').Append(stack[top]);
                continue;
            }

            used[edgeId[a]] = true;
            iter[u] = next[a];

            int v = to[a];
            stack[++top] = v;
            traversed++;
            output.Append(' ').Append(v);
        }

        // A closed walk covering every edge exists only when every edge is reachable from 0.
        // Challenge inputs are expected to satisfy that. If not, emit the best valid walk
        // from node 0 rather than inventing non-edge jumps.
        if (traversed != m)
        {
            // The prefixed length would be wrong for a disconnected graph, so rebuild without it.
            Array.Fill(used, false);
            Array.Copy(head, iter, n);
            top = 0;
            stack[0] = 0;
            output.Clear();
            output.Append(0);

            while (top >= 0)
            {
                int u = stack[top];
                int a = iter[u];
                while (a != -1 && used[edgeId[a]])
                    a = next[a];
                iter[u] = a;

                if (a == -1)
                {
                    top--;
                    if (top >= 0)
                        output.Append(' ').Append(stack[top]);
                    continue;
                }

                used[edgeId[a]] = true;
                iter[u] = next[a];
                int v = to[a];
                stack[++top] = v;
                output.Append(' ').Append(v);
            }
        }

        Console.Out.Write(output.ToString());
    }
}
