using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = Console.OpenStandardInput();
        var input = new FastScanner(fs);

        int n = input.NextInt();
        int stockLength = input.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        int[] lengths = new int[n];
        int[] ids = new int[n];

        for (int i = 0; i < n; i++)
        {
            int len = input.NextInt();
            lengths[i] = len;
            ids[i] = i;
        }

        Array.Sort(lengths, ids);

        int[] cutNext = new int[n];
        Array.Fill(cutNext, -1);

        int capacity = Math.Min(1024, Math.Max(1, n));
        int[] stockHead = new int[capacity];
        int[] stockCount = new int[capacity];
        int[] sameRemainingNext = new int[capacity];
        Array.Fill(stockHead, -1);
        Array.Fill(sameRemainingNext, -1);

        var spaces = new ResidualTreap(Math.Min(n, 1 << 20));
        int stockCountUsed = 0;

        for (int pos = n - 1; pos >= 0; pos--)
        {
            int len = lengths[pos];
            int cutId = ids[pos];

            int stockId;
            int remaining = spaces.PopSmallestAtLeast(len, sameRemainingNext);
            if (remaining >= 0)
            {
                stockId = spaces.LastPoppedStock;
            }
            else
            {
                stockId = stockCountUsed++;
                if (stockId == stockHead.Length)
                {
                    int old = stockHead.Length;
                    int nextCap = old < 1_000_000 ? old * 2 : old + (old >> 1);
                    Array.Resize(ref stockHead, nextCap);
                    Array.Resize(ref stockCount, nextCap);
                    Array.Resize(ref sameRemainingNext, nextCap);
                    Array.Fill(stockHead, -1, old, nextCap - old);
                    Array.Fill(sameRemainingNext, -1, old, nextCap - old);
                }
                remaining = stockLength;
            }

            cutNext[cutId] = stockHead[stockId];
            stockHead[stockId] = cutId;
            stockCount[stockId]++;

            int newRemaining = remaining - len;
            if (newRemaining > 0)
            {
                spaces.Push(newRemaining, stockId, sameRemainingNext);
            }
        }

        var output = new FastOutput(Console.OpenStandardOutput());
        output.WriteInt(stockCountUsed);
        output.WriteByte((byte)'\n');

        for (int stock = 0; stock < stockCountUsed; stock++)
        {
            bool first = true;
            for (int cut = stockHead[stock]; cut >= 0; cut = cutNext[cut])
            {
                if (!first) output.WriteByte((byte)' ');
                first = false;
                output.WriteInt(cut);
            }
            output.WriteByte((byte)'\n');
        }

        output.Flush();
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _length;
        private int _position;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            }
            while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }

            return value * sign;
        }

        private int Read()
        {
            if (_position >= _length)
            {
                _length = _stream.Read(_buffer, 0, _buffer.Length);
                _position = 0;
                if (_length == 0) return -1;
            }

            return _buffer[_position++];
        }
    }

    private sealed class FastOutput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _position;

        public FastOutput(Stream stream)
        {
            _stream = stream;
        }

        public void WriteByte(byte value)
        {
            if (_position == _buffer.Length) Flush();
            _buffer[_position++] = value;
        }

        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (value < 0)
            {
                WriteByte((byte)'-');
                value = -value;
            }

            Span<byte> digits = stackalloc byte[11];
            int count = 0;
            while (value > 0)
            {
                digits[count++] = (byte)('0' + value % 10);
                value /= 10;
            }

            for (int i = count - 1; i >= 0; i--) WriteByte(digits[i]);
        }

        public void Flush()
        {
            if (_position == 0) return;
            _stream.Write(_buffer, 0, _position);
            _position = 0;
        }
    }

    private sealed class ResidualTreap
    {
        private int _root = -1;
        private int _count;
        private int[] _key;
        private int[] _priority;
        private int[] _left;
        private int[] _right;
        private int[] _topStock;
        private uint _rng = 2463534242u;

        public int LastPoppedStock { get; private set; }

        public ResidualTreap(int initialCapacity)
        {
            initialCapacity = Math.Max(16, initialCapacity);
            _key = new int[initialCapacity];
            _priority = new int[initialCapacity];
            _left = new int[initialCapacity];
            _right = new int[initialCapacity];
            _topStock = new int[initialCapacity];
            Array.Fill(_left, -1);
            Array.Fill(_right, -1);
        }

        public void Push(int remaining, int stockId, int[] nextSameRemaining)
        {
            int node = FindNode(remaining);
            if (node >= 0)
            {
                nextSameRemaining[stockId] = _topStock[node];
                _topStock[node] = stockId;
                return;
            }

            EnsureCapacity();
            node = _count++;
            _key[node] = remaining;
            _priority[node] = NextPriority();
            _left[node] = -1;
            _right[node] = -1;
            _topStock[node] = stockId;
            nextSameRemaining[stockId] = -1;
            Insert(ref _root, node);
        }

        public int PopSmallestAtLeast(int required, int[] nextSameRemaining)
        {
            int node = LowerBound(required);
            if (node < 0)
            {
                LastPoppedStock = -1;
                return -1;
            }

            int remaining = _key[node];
            int stock = _topStock[node];
            LastPoppedStock = stock;
            _topStock[node] = nextSameRemaining[stock];
            nextSameRemaining[stock] = -1;

            if (_topStock[node] < 0)
            {
                Delete(ref _root, remaining);
            }

            return remaining;
        }

        private int LowerBound(int required)
        {
            int current = _root;
            int answer = -1;
            while (current >= 0)
            {
                if (_key[current] >= required)
                {
                    answer = current;
                    current = _left[current];
                }
                else
                {
                    current = _right[current];
                }
            }

            return answer;
        }

        private int FindNode(int key)
        {
            int current = _root;
            while (current >= 0)
            {
                if (key == _key[current]) return current;
                current = key < _key[current] ? _left[current] : _right[current];
            }

            return -1;
        }

        private void Insert(ref int root, int node)
        {
            if (root < 0)
            {
                root = node;
                return;
            }

            if (_key[node] < _key[root])
            {
                Insert(ref _left[root], node);
                if (_priority[_left[root]] > _priority[root]) RotateRight(ref root);
            }
            else
            {
                Insert(ref _right[root], node);
                if (_priority[_right[root]] > _priority[root]) RotateLeft(ref root);
            }
        }

        private void Delete(ref int root, int key)
        {
            if (root < 0) return;

            if (key < _key[root])
            {
                Delete(ref _left[root], key);
            }
            else if (key > _key[root])
            {
                Delete(ref _right[root], key);
            }
            else
            {
                if (_left[root] < 0)
                {
                    root = _right[root];
                }
                else if (_right[root] < 0)
                {
                    root = _left[root];
                }
                else if (_priority[_left[root]] > _priority[_right[root]])
                {
                    RotateRight(ref root);
                    Delete(ref _right[root], key);
                }
                else
                {
                    RotateLeft(ref root);
                    Delete(ref _left[root], key);
                }
            }
        }

        private void RotateLeft(ref int root)
        {
            int x = _right[root];
            _right[root] = _left[x];
            _left[x] = root;
            root = x;
        }

        private void RotateRight(ref int root)
        {
            int x = _left[root];
            _left[root] = _right[x];
            _right[x] = root;
            root = x;
        }

        private int NextPriority()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return unchecked((int)x);
        }

        private void EnsureCapacity()
        {
            if (_count < _key.Length) return;

            int old = _key.Length;
            int next = old < 1_000_000 ? old * 2 : old + (old >> 1);
            Array.Resize(ref _key, next);
            Array.Resize(ref _priority, next);
            Array.Resize(ref _left, next);
            Array.Resize(ref _right, next);
            Array.Resize(ref _topStock, next);
            Array.Fill(_left, -1, old, next - old);
            Array.Fill(_right, -1, old, next - old);
        }
    }
}
