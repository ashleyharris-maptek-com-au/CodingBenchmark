using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();

        var g = new List<int>[n];
        var rg = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            g[i] = new List<int>();
            rg[i] = new List<int>();
        }

        var forced = new bool[n];
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u == v) forced[u] = true;
            g[u].Add(v);
            rg[v].Add(u);
        }

        var answer = new List<int>();
        for (int i = 0; i < n; i++)
            if (forced[i]) answer.Add(i);

        foreach (var comp in StrongComponents(g, rg, forced))
        {
            if (comp.Count <= 1) continue;

            var id = new Dictionary<int, int>(comp.Count);
            for (int i = 0; i < comp.Count; i++) id[comp[i]] = i;

            var local = new List<int>[comp.Count];
            for (int i = 0; i < comp.Count; i++) local[i] = new List<int>();

            for (int i = 0; i < comp.Count; i++)
            {
                int old = comp[i];
                foreach (int to in g[old])
                {
                    if (!forced[to] && id.TryGetValue(to, out int j) && i != j)
                        local[i].Add(j);
                }
            }

            var solver = new ExactDfvs(local);
            foreach (int v in solver.Solve())
                answer.Add(comp[v]);
        }

        answer.Sort();
        Console.WriteLine(answer.Count);
        Console.WriteLine(string.Join(" ", answer));
    }

    private static List<List<int>> StrongComponents(List<int>[] g, List<int>[] rg, bool[] removed)
    {
        int n = g.Length;
        var seen = new bool[n];
        var order = new List<int>(n);

        for (int s = 0; s < n; s++)
        {
            if (removed[s] || seen[s]) continue;
            var stV = new Stack<int>();
            var stI = new Stack<int>();
            stV.Push(s);
            stI.Push(0);
            seen[s] = true;

            while (stV.Count > 0)
            {
                int v = stV.Peek();
                int idx = stI.Pop();
                if (idx < g[v].Count)
                {
                    stI.Push(idx + 1);
                    int to = g[v][idx];
                    if (!removed[to] && !seen[to])
                    {
                        seen[to] = true;
                        stV.Push(to);
                        stI.Push(0);
                    }
                }
                else
                {
                    stV.Pop();
                    order.Add(v);
                }
            }
        }

        var comps = new List<List<int>>();
        Array.Fill(seen, false);
        for (int oi = order.Count - 1; oi >= 0; oi--)
        {
            int s = order[oi];
            if (seen[s]) continue;
            var comp = new List<int>();
            var stack = new Stack<int>();
            stack.Push(s);
            seen[s] = true;
            while (stack.Count > 0)
            {
                int v = stack.Pop();
                comp.Add(v);
                foreach (int to in rg[v])
                {
                    if (!removed[to] && !seen[to])
                    {
                        seen[to] = true;
                        stack.Push(to);
                    }
                }
            }
            comps.Add(comp);
        }
        return comps;
    }

    private sealed class ExactDfvs
    {
        private readonly int n;
        private readonly int[][] adj;
        private readonly int[][] radj;
        private bool[] best = Array.Empty<bool>();
        private int bestCount;

        public ExactDfvs(List<int>[] graph)
        {
            n = graph.Length;
            adj = new int[n][];
            var rev = new List<int>[n];
            for (int i = 0; i < n; i++) rev[i] = new List<int>();

            for (int i = 0; i < n; i++)
            {
                adj[i] = graph[i].Distinct().ToArray();
                foreach (int to in adj[i]) rev[to].Add(i);
            }

            radj = new int[n][];
            for (int i = 0; i < n; i++) radj[i] = rev[i].Distinct().ToArray();
        }

        public List<int> Solve()
        {
            var active = new bool[n];
            Array.Fill(active, true);
            Reduce(active);

            best = GreedyUpperBound(active);
            bestCount = Count(best);

            var current = new bool[n];
            Search(active, current, 0);

            var res = new List<int>();
            for (int i = 0; i < n; i++)
                if (best[i]) res.Add(i);
            return res;
        }

        private void Search(bool[] active, bool[] chosen, int chosenCount)
        {
            if (chosenCount >= bestCount) return;

            active = (bool[])active.Clone();
            Reduce(active);

            int lb = GreedyDisjointCycleLowerBound(active);
            if (chosenCount + lb >= bestCount) return;

            int[]? cycle = FindCycle(active);
            if (cycle == null)
            {
                bestCount = chosenCount;
                best = (bool[])chosen.Clone();
                return;
            }

            Array.Sort(cycle, (a, b) => Score(b, active).CompareTo(Score(a, active)));

            foreach (int v in cycle)
            {
                if (!active[v]) continue;
                var nextActive = (bool[])active.Clone();
                nextActive[v] = false;
                chosen[v] = true;
                Search(nextActive, chosen, chosenCount + 1);
                chosen[v] = false;
            }
        }

        private bool[] GreedyUpperBound(bool[] start)
        {
            var active = (bool[])start.Clone();
            var deleted = new bool[n];
            Reduce(active);

            while (true)
            {
                int[]? cycle = FindCycle(active);
                if (cycle == null) return deleted;

                int bestV = cycle[0];
                int bestScore = -1;
                foreach (int v in cycle)
                {
                    int sc = Score(v, active);
                    if (sc > bestScore)
                    {
                        bestScore = sc;
                        bestV = v;
                    }
                }

                deleted[bestV] = true;
                active[bestV] = false;
                Reduce(active);
            }
        }

        private int GreedyDisjointCycleLowerBound(bool[] start)
        {
            var active = (bool[])start.Clone();
            int count = 0;

            while (count < bestCount)
            {
                Reduce(active);
                int[]? cycle = FindCycle(active);
                if (cycle == null) break;
                count++;
                foreach (int v in cycle) active[v] = false;
            }

            return count;
        }

        private void Reduce(bool[] active)
        {
            var indeg = new int[n];
            var outdeg = new int[n];
            var q = new Queue<int>();
            var queued = new bool[n];

            for (int v = 0; v < n; v++)
            {
                if (!active[v]) continue;
                foreach (int to in adj[v]) if (active[to]) outdeg[v]++;
                foreach (int from in radj[v]) if (active[from]) indeg[v]++;
            }

            for (int v = 0; v < n; v++)
            {
                if (active[v] && (indeg[v] == 0 || outdeg[v] == 0))
                {
                    q.Enqueue(v);
                    queued[v] = true;
                }
            }

            while (q.Count > 0)
            {
                int v = q.Dequeue();
                if (!active[v]) continue;
                active[v] = false;

                foreach (int to in adj[v])
                {
                    if (!active[to]) continue;
                    indeg[to]--;
                    if (!queued[to] && (indeg[to] == 0 || outdeg[to] == 0))
                    {
                        queued[to] = true;
                        q.Enqueue(to);
                    }
                }

                foreach (int from in radj[v])
                {
                    if (!active[from]) continue;
                    outdeg[from]--;
                    if (!queued[from] && (indeg[from] == 0 || outdeg[from] == 0))
                    {
                        queued[from] = true;
                        q.Enqueue(from);
                    }
                }
            }
        }

        private int[]? FindCycle(bool[] active)
        {
            var color = new byte[n];
            var parent = new int[n];
            Array.Fill(parent, -1);

            for (int s = 0; s < n; s++)
            {
                if (!active[s] || color[s] != 0) continue;

                var stV = new Stack<int>();
                var stI = new Stack<int>();
                stV.Push(s);
                stI.Push(0);
                color[s] = 1;

                while (stV.Count > 0)
                {
                    int v = stV.Peek();
                    int idx = stI.Pop();
                    if (idx < adj[v].Length)
                    {
                        stI.Push(idx + 1);
                        int to = adj[v][idx];
                        if (!active[to]) continue;

                        if (color[to] == 0)
                        {
                            parent[to] = v;
                            color[to] = 1;
                            stV.Push(to);
                            stI.Push(0);
                        }
                        else if (color[to] == 1)
                        {
                            var cycle = new List<int> { to };
                            for (int x = v; x != to && x != -1; x = parent[x])
                                cycle.Add(x);
                            return cycle.ToArray();
                        }
                    }
                    else
                    {
                        color[v] = 2;
                        stV.Pop();
                    }
                }
            }

            return null;
        }

        private int Score(int v, bool[] active)
        {
            int inn = 0, outt = 0;
            foreach (int to in adj[v]) if (active[to]) outt++;
            foreach (int from in radj[v]) if (active[from]) inn++;
            return inn * outt + inn + outt;
        }

        private static int Count(bool[] a)
        {
            int c = 0;
            foreach (bool b in a) if (b) c++;
            return c;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream input;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream input)
        {
            this.input = input;
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + c - '0';
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }
    }
}
