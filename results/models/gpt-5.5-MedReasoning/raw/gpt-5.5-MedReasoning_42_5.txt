OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%fn_void = OpTypeFunction %void
%fn_v4_v3 = OpTypeFunction %v4float %v3float
%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%make_pos = OpFunction %v4float None %fn_v4_v3
%pos_in = OpFunctionParameter %v3float
%make_pos_entry = OpLabel
%pos_x = OpCompositeExtract %float %pos_in 0
%pos_y = OpCompositeExtract %float %pos_in 1
%pos_z = OpCompositeExtract %float %pos_in 2
%pos_out = OpCompositeConstruct %v4float %pos_x %pos_y %pos_z %float_1
OpReturnValue %pos_out
OpFunctionEnd
%main = OpFunction %void None %fn_void
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v3 %in_worldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3 %in_worldPos %int_1
%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %in_color %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%dir = OpFSub %v3float %p1 %p0
%dir_scaled = OpVectorTimesScalar %v3float %dir %float_0_75
%mid = OpFAdd %v3float %p0 %dir_scaled
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp = OpExtInst %v2float %glsl Normalize %perp_raw
%shaft_off2 = OpVectorTimesScalar %v2float %perp %float_0_015
%head_off2 = OpVectorTimesScalar %v2float %perp %float_0_05
%shaft_off_x = OpCompositeExtract %float %shaft_off2 0
%shaft_off_y = OpCompositeExtract %float %shaft_off2 1
%head_off_x = OpCompositeExtract %float %head_off2 0
%head_off_y = OpCompositeExtract %float %head_off2 1
%shaft_off3 = OpCompositeConstruct %v3float %shaft_off_x %shaft_off_y %float_0
%head_off3 = OpCompositeConstruct %v3float %head_off_x %head_off_y %float_0
%neg_shaft_off3 = OpFNegate %v3float %shaft_off3
%neg_head_off3 = OpFNegate %v3float %head_off3
%shaft0 = OpFAdd %v3float %p0 %shaft_off3
%shaft1 = OpFAdd %v3float %p0 %neg_shaft_off3
%shaft2 = OpFAdd %v3float %mid %shaft_off3
%shaft3 = OpFAdd %v3float %mid %neg_shaft_off3
%head0 = OpFAdd %v3float %mid %head_off3
%head1 = OpFAdd %v3float %mid %neg_head_off3
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
%shaft0_pos = OpFunctionCall %v4float %make_pos %shaft0
OpStore %out_pos_ptr %shaft0_pos
OpStore %outColor %c0
OpEmitVertex
%shaft1_pos = OpFunctionCall %v4float %make_pos %shaft1
OpStore %out_pos_ptr %shaft1_pos
OpStore %outColor %c0
OpEmitVertex
%shaft2_pos = OpFunctionCall %v4float %make_pos %shaft2
OpStore %out_pos_ptr %shaft2_pos
OpStore %outColor %c0
OpEmitVertex
%shaft3_pos = OpFunctionCall %v4float %make_pos %shaft3
OpStore %out_pos_ptr %shaft3_pos
OpStore %outColor %c0
OpEmitVertex
OpEndPrimitive
%head0_pos = OpFunctionCall %v4float %make_pos %head0
OpStore %out_pos_ptr %head0_pos
OpStore %outColor %c1
OpEmitVertex
%head1_pos = OpFunctionCall %v4float %make_pos %head1
OpStore %out_pos_ptr %head1_pos
OpStore %outColor %c1
OpEmitVertex
%tip_pos = OpFunctionCall %v4float %make_pos %p1
OpStore %out_pos_ptr %tip_pos
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
