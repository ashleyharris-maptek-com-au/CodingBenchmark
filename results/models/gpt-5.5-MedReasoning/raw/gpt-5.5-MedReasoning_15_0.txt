use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    let eps = 1.0e-9;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if cross(a, b, p).abs() <= eps
            && p.x >= a.x.min(b.x) - eps
            && p.x <= a.x.max(b.x) + eps
            && p.y >= a.y.min(b.y) - eps
            && p.y <= a.y.max(b.y) + eps
        {
            return true;
        }
        let intersects = (a.y > p.y) != (b.y > p.y);
        if intersects {
            let x_at_y = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if p.x < x_at_y {
                inside = !inside;
            }
        }
    }
    inside
}

fn orient(a: Pt, b: Pt, c: Pt) -> i32 {
    let v = cross(a, b, c);
    if v > 1.0e-9 {
        1
    } else if v < -1.0e-9 {
        -1
    } else {
        0
    }
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= 1.0e-9
        && p.x >= a.x.min(b.x) - 1.0e-9
        && p.x <= a.x.max(b.x) + 1.0e-9
        && p.y >= a.y.min(b.y) - 1.0e-9
        && p.y <= a.y.max(b.y) + 1.0e-9
}

fn segments_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1 != o2 && o3 != o4 {
        return true;
    }
    (o1 == 0 && on_segment(a, b, c))
        || (o2 == 0 && on_segment(a, b, d))
        || (o3 == 0 && on_segment(c, d, a))
        || (o4 == 0 && on_segment(c, d, b))
}

fn rect_intersects_poly(cx: f64, cy: f64, half: f64, poly: &[Pt]) -> bool {
    let min_x = cx - half;
    let max_x = cx + half;
    let min_y = cy - half;
    let max_y = cy + half;
    for p in poly {
        if p.x >= min_x && p.x <= max_x && p.y >= min_y && p.y <= max_y {
            return true;
        }
    }
    let corners = [
        Pt { x: min_x, y: min_y },
        Pt { x: max_x, y: min_y },
        Pt { x: max_x, y: max_y },
        Pt { x: min_x, y: max_y },
    ];
    for c in corners {
        if point_in_poly(c, poly) {
            return true;
        }
    }
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        for j in 0..4 {
            if segments_intersect(a, b, corners[j], corners[(j + 1) % 4]) {
                return true;
            }
        }
    }
    point_in_poly(Pt { x: cx, y: cy }, poly)
}

fn convex_hull(mut pts: Vec<Pt>) -> Vec<Pt> {
    pts.sort_by(|a, b| a.x.total_cmp(&b.x).then(a.y.total_cmp(&b.y)));
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1.0e-12 && (a.y - b.y).abs() < 1.0e-12);
    if pts.len() <= 1 {
        return pts;
    }
    let mut lower: Vec<Pt> = Vec::new();
    for p in &pts {
        while lower.len() >= 2
            && cross(lower[lower.len() - 2], lower[lower.len() - 1], *p) <= 1.0e-12
        {
            lower.pop();
        }
        lower.push(*p);
    }
    let mut upper: Vec<Pt> = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2
            && cross(upper[upper.len() - 2], upper[upper.len() - 1], *p) <= 1.0e-12
        {
            upper.pop();
        }
        upper.push(*p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn origin_inradius(poly: &[Pt]) -> f64 {
    let mut best = f64::INFINITY;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let num = ((b.x - a.x) * (-a.y) - (b.y - a.y) * (-a.x)).abs();
        let den = ((b.x - a.x).powi(2) + (b.y - a.y).powi(2)).sqrt();
        if den > 0.0 {
            best = best.min(num / den);
        }
    }
    best
}

fn print_tetra(
    index: usize,
    target: Pt,
    sx: f64,
    sy: f64,
    sz: f64,
    scale: f64,
    separation: f64,
    base_z: f64,
    uz: f64,
) {
    let lift = index as f64 * separation;
    let z = base_z - lift * uz;
    let x = target.x + z * sx / sz;
    let y = target.y + z * sy / sz;
    println!("{x:.9} {y:.9} {z:.9} 1.0 0.0 0.0 0.0 {scale:.9}");
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let nv = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
        let y = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
        poly.push(Pt { x, y });
    }
    let sx = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
    let sy = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
    let mut sz = it.next().unwrap_or("-1").parse::<f64>().unwrap_or(-1.0);
    if sz.abs() < 1.0e-6 {
        sz = if sz.is_sign_negative() { -1.0e-6 } else { 1.0e-6 };
    }
    if poly.is_empty() {
        println!("0");
        return;
    }

    let scale = 10.0_f64;
    let verts = [
        (1.0, 0.0, -0.707),
        (-1.0, 0.0, -0.707),
        (0.0, 1.0, 0.707),
        (0.0, -1.0, 0.707),
    ];
    let projected: Vec<Pt> = verts
        .iter()
        .map(|(x, y, z)| Pt {
            x: scale * (x - z * sx / sz),
            y: scale * (y - z * sy / sz),
        })
        .collect();
    let hull = convex_hull(projected);
    let mut radius = origin_inradius(&hull);
    if !radius.is_finite() || radius < 0.05 {
        radius = 0.05;
    }
    let half = radius / 2.0_f64.sqrt() * 0.985;
    let step = 2.0 * half;

    let mut min_x = poly[0].x;
    let mut max_x = poly[0].x;
    let mut min_y = poly[0].y;
    let mut max_y = poly[0].y;
    for p in &poly {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }

    let nx = (((max_x - min_x) / step).ceil() as usize).saturating_add(1);
    let ny = (((max_y - min_y) / step).ceil() as usize).saturating_add(1);
    let dense_scan = nx.saturating_mul(ny) <= 2_000_000;
    let mut targets: Vec<Pt> = Vec::new();

    if dense_scan {
        for ix in 0..nx {
            let cx = min_x + half + ix as f64 * step;
            for iy in 0..ny {
                let cy = min_y + half + iy as f64 * step;
                if rect_intersects_poly(cx, cy, half, &poly) {
                    targets.push(Pt { x: cx, y: cy });
                }
            }
        }
    } else {
        for ix in 0..nx {
            let cx = min_x + half + ix as f64 * step;
            for iy in 0..ny {
                targets.push(Pt {
                    x: cx,
                    y: min_y + half + iy as f64 * step,
                });
            }
        }
    }

    if targets.is_empty() {
        targets.push(Pt {
            x: (min_x + max_x) * 0.5,
            y: (min_y + max_y) * 0.5,
        });
    }

    let sun_len = (sx * sx + sy * sy + sz * sz).sqrt().max(1.0e-9);
    let uz = sz / sun_len;
    let object_radius = scale * (1.0_f64 + 0.707_f64 * 0.707_f64).sqrt();
    let separation = object_radius * 2.25;
    let base_z = scale * 0.8 + 1.0;

    if dense_scan {
        println!("{}", targets.len());
        for (i, t) in targets.iter().enumerate() {
            print_tetra(i, *t, sx, sy, sz, scale, separation, base_z, uz);
        }
    } else {
        let count = nx.saturating_mul(ny);
        println!("{count}");
        let mut index = 0usize;
        for ix in 0..nx {
            let cx = min_x + half + ix as f64 * step;
            for iy in 0..ny {
                let target = Pt {
                    x: cx,
                    y: min_y + half + iy as f64 * step,
                };
                print_tetra(index, target, sx, sy, sz, scale, separation, base_z, uz);
                index = index.saturating_add(1);
            }
        }
    }
}
