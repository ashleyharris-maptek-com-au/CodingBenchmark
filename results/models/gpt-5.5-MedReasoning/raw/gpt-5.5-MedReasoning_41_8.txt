; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inColor %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_25 = OpConstant %float 0.25
%float_0_5 = OpConstant %float 0.5
%float_0_75 = OpConstant %float 0.75
%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_7 = OpConstant %float 0.7
%float_1 = OpConstant %float 1
%inWorldPos = OpVariable %ptr_Input_v3float Input
%inNormal = OpVariable %ptr_Input_v3float Input
%inColor = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%main = OpFunction %void None %func
%entry = OpLabel
%world = OpLoad %v3float %inWorldPos
%normalValue = OpLoad %v3float %inNormal
%n = OpExtInst %v3float %glsl Normalize %normalValue
%lightPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%toLight = OpFSub %v3float %light3 %world
%l = OpExtInst %v3float %glsl Normalize %toLight
%dotNL = OpDot %float %n %l
%d = OpExtInst %float %glsl FMax %dotNL %float_0
%ge25 = OpFOrdGreaterThanEqual %bool %d %float_0_25
%band25 = OpSelect %float %ge25 %float_0_4 %float_0_2
%ge50 = OpFOrdGreaterThanEqual %bool %d %float_0_5
%band50 = OpSelect %float %ge50 %float_0_7 %band25
%ge75 = OpFOrdGreaterThanEqual %bool %d %float_0_75
%band = OpSelect %float %ge75 %float_1 %band50
%vertexColor = OpLoad %v3float %inColor
%bandVec = OpCompositeConstruct %v3float %band %band %band
%rgb = OpFMul %v3float %vertexColor %bandVec
%r = OpCompositeExtract %float %rgb 0
%g = OpCompositeExtract %float %rgb 1
%b = OpCompositeExtract %float %rgb 2
%finalColor = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
