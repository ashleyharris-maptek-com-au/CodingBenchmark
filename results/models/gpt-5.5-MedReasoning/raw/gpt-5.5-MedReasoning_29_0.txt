using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    const long TimeLimitTicks = TimeSpan.TicksPerSecond * 285;
    static readonly long StartTicks = DateTime.UtcNow.Ticks;

    static int N;
    static int[][] Adj = Array.Empty<int[]>();
    static int[] Degree = Array.Empty<int>();
    static int[] Best = Array.Empty<int>();
    static readonly List<int> WorkClique = new List<int>();
    static bool TimedOut;

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        N = fs.NextInt();
        int m = fs.NextInt();
        if (N <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        var eu = new int[m];
        var ev = new int[m];
        Degree = new int[N];
        int ecount = 0;

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if (u < 0 || v < 0 || u >= N || v >= N || u == v) continue;
            eu[ecount] = u;
            ev[ecount] = v;
            Degree[u]++;
            Degree[v]++;
            ecount++;
        }

        var temp = new int[N][];
        for (int i = 0; i < N; i++) temp[i] = new int[Degree[i]];
        Array.Clear(Degree, 0, N);
        for (int i = 0; i < ecount; i++)
        {
            int u = eu[i], v = ev[i];
            temp[u][Degree[u]++] = v;
            temp[v][Degree[v]++] = u;
        }

        Adj = new int[N][];
        long uniqueM2 = 0;
        for (int i = 0; i < N; i++)
        {
            Array.Sort(temp[i]);
            int w = 0;
            int last = -1;
            for (int j = 0; j < temp[i].Length; j++)
            {
                int x = temp[i][j];
                if (x != last)
                {
                    temp[i][w++] = x;
                    last = x;
                }
            }
            if (w == temp[i].Length) Adj[i] = temp[i];
            else
            {
                Adj[i] = new int[w];
                Array.Copy(temp[i], Adj[i], w);
            }
            Degree[i] = w;
            uniqueM2 += w;
        }

        long completeEdges2 = (long)N * (N - 1);
        if (uniqueM2 == completeEdges2)
        {
            PrintClique(Enumerable.Range(0, N).ToArray());
            return;
        }
        if (uniqueM2 == 0)
        {
            PrintClique(new[] { 0 });
            return;
        }

        var fixedUniversal = new List<int>();
        var remaining = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (Degree[i] == N - 1) fixedUniversal.Add(i);
            else remaining.Add(i);
        }

        GreedyInitialClique();
        if (fixedUniversal.Count > Best.Length)
            Best = fixedUniversal.ToArray();

        if (remaining.Count == 0)
        {
            PrintClique(fixedUniversal.ToArray());
            return;
        }

        int[] vertices = remaining.ToArray();
        Array.Sort(vertices, (a, b) =>
        {
            int c = Degree[a].CompareTo(Degree[b]);
            return c != 0 ? c : a.CompareTo(b);
        });

        int[] rank = new int[N];
        for (int i = 0; i < vertices.Length; i++) rank[vertices[i]] = i;

        for (int idx = 0; idx < vertices.Length && !TimedOut; idx++)
        {
            if (OutOfTime()) break;
            int v = vertices[idx];
            if (fixedUniversal.Count + 1 + Degree[v] <= Best.Length) continue;

            var p = new List<int>();
            foreach (int u in Adj[v])
                if (Degree[u] != N - 1 && rank[u] > idx)
                    p.Add(u);

            WorkClique.Clear();
            WorkClique.AddRange(fixedUniversal);
            WorkClique.Add(v);
            if (WorkClique.Count > Best.Length) Best = WorkClique.ToArray();
            Expand(p);
        }

        PrintClique(Best.Length == 0 ? new[] { 0 } : Best);
    }

    static void GreedyInitialClique()
    {
        int[] order = Enumerable.Range(0, N).ToArray();
        Array.Sort(order, (a, b) =>
        {
            int c = Degree[b].CompareTo(Degree[a]);
            return c != 0 ? c : a.CompareTo(b);
        });

        var clique = new List<int>();
        foreach (int v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                if (!IsEdge(v, clique[i]))
                {
                    ok = false;
                    break;
                }
            }
            if (ok) clique.Add(v);
        }
        Best = clique.ToArray();
    }

    static void Expand(List<int> candidates)
    {
        if (candidates.Count == 0)
        {
            if (WorkClique.Count > Best.Length) Best = WorkClique.ToArray();
            return;
        }
        if (WorkClique.Count + candidates.Count <= Best.Length) return;
        if (OutOfTime()) return;

        int count = candidates.Count;
        int[] order = new int[count];
        int[] colors = new int[count];
        ColorSort(candidates, order, colors);

        for (int i = count - 1; i >= 0; i--)
        {
            if (WorkClique.Count + colors[i] <= Best.Length) return;
            if (OutOfTime()) return;

            int v = order[i];
            var next = new List<int>();
            for (int j = 0; j < i; j++)
            {
                int u = order[j];
                if (IsEdge(v, u)) next.Add(u);
            }

            WorkClique.Add(v);
            if (WorkClique.Count > Best.Length) Best = WorkClique.ToArray();
            Expand(next);
            WorkClique.RemoveAt(WorkClique.Count - 1);
        }
    }

    static void ColorSort(List<int> input, int[] order, int[] colors)
    {
        int n = input.Count;
        int[] verts = input.ToArray();
        Array.Sort(verts, (a, b) =>
        {
            int c = Degree[b].CompareTo(Degree[a]);
            return c != 0 ? c : a.CompareTo(b);
        });

        bool[] used = new bool[n];
        int placed = 0;
        int color = 0;
        while (placed < n)
        {
            color++;
            var classVerts = new List<int>();
            for (int i = 0; i < n; i++)
            {
                if (used[i]) continue;
                int v = verts[i];
                bool conflicts = false;
                for (int j = 0; j < classVerts.Count; j++)
                {
                    if (IsEdge(v, classVerts[j]))
                    {
                        conflicts = true;
                        break;
                    }
                }
                if (!conflicts)
                {
                    used[i] = true;
                    classVerts.Add(v);
                    order[placed] = v;
                    colors[placed] = color;
                    placed++;
                }
            }
        }
    }

    static bool IsEdge(int u, int v)
    {
        int[] a = Adj[u];
        int lo = 0, hi = a.Length - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = a[mid];
            if (x == v) return true;
            if (x < v) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    static bool OutOfTime()
    {
        if (DateTime.UtcNow.Ticks - StartTicks > TimeLimitTicks)
        {
            TimedOut = true;
            return true;
        }
        return false;
    }

    static void PrintClique(int[] clique)
    {
        Array.Sort(clique);
        Console.WriteLine(clique.Length);
        Console.WriteLine(string.Join(" ", clique));
    }

    sealed class FastScanner
    {
        readonly Stream _stream;
        readonly byte[] _buffer = new byte[1 << 16];
        int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
