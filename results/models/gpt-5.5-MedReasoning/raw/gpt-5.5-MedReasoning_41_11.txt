; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inUV "uv"
OpName %outColor "outColor"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%bool = OpTypeBool
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%ubo = OpVariable %ptr_Uniform_UBO Uniform
%inWorldPos = OpVariable %ptr_Input_v3float Input
%inNormal = OpVariable %ptr_Input_v3float Input
%inUV = OpVariable %ptr_Input_v2float Input
%outColor = OpVariable %ptr_Output_v4float Output
%int_3 = OpConstant %int 3
%float_0 = OpConstant %float 0
%float_0_09 = OpConstant %float 0.09
%float_0_1 = OpConstant %float 0.1
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1
%float_6 = OpConstant %float 6
%v2float_6 = OpConstantComposite %v2float %float_6 %float_6
%v2float_0_5 = OpConstantComposite %v2float %float_0_5 %float_0_5
%yellow = OpConstantComposite %v3float %float_1 %float_0_8 %float_0
%darkBlue = OpConstantComposite %v3float %float_0_1 %float_0_1 %float_0_3
%main = OpFunction %void None %func
%entry = OpLabel
%uv = OpLoad %v2float %inUV
%scaledUV = OpFMul %v2float %uv %v2float_6
%cellBase = OpExtInst %v2float %1 Floor %scaledUV
%cellUV = OpFSub %v2float %scaledUV %cellBase
%centered = OpFSub %v2float %cellUV %v2float_0_5
%dist2 = OpDot %float %centered %centered
%isDot = OpFOrdLessThan %bool %dist2 %float_0_09
%baseColor = OpSelect %v3float %isDot %yellow %darkBlue
%normalValue = OpLoad %v3float %inNormal
%normalUnit = OpExtInst %v3float %1 Normalize %normalValue
%worldPos = OpLoad %v3float %inWorldPos
%lightPtr = OpAccessChain %ptr_Uniform_v4float %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %light4 %light4 0 1 2
%lightVec = OpFSub %v3float %lightPos %worldPos
%lightDir = OpExtInst %v3float %1 Normalize %lightVec
%ndotlRaw = OpDot %float %normalUnit %lightDir
%diffuse = OpExtInst %float %1 FMax %ndotlRaw %float_0
%diffuseScaled = OpFMul %float %float_0_7 %diffuse
%shade = OpFAdd %float %float_0_3 %diffuseScaled
%litColor = OpVectorTimesScalar %v3float %baseColor %shade
%r = OpCompositeExtract %float %litColor 0
%g = OpCompositeExtract %float %litColor 1
%b = OpCompositeExtract %float %litColor 2
%finalColor = OpCompositeConstruct %v4float %r %g %b %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
