using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;

public static class Program
{
    const double Day = 86400.0;
    const double G = 6.67430e-11;

    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec operator +(Vec a, Vec b) => new Vec(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec operator -(Vec a, Vec b) => new Vec(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec operator *(Vec a, double k) => new Vec(a.X * k, a.Y * k, a.Z * k);
        public static Vec operator /(Vec a, double k) => new Vec(a.X / k, a.Y / k, a.Z / k);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vec Unit()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec(1, 0, 0);
        }
        public static double Dot(Vec a, Vec b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec Cross(Vec a, Vec b) => new Vec(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    class Body
    {
        public string Name = "";
        public double Mass;
        public Vec R, V;
    }

    struct State
    {
        public Vec R, V;
        public State(Vec r, Vec v) { R = r; V = v; }
    }

    class Candidate
    {
        public double LaunchDay, ImpactDay, LaunchDv, Score;
        public Vec LaunchBurn, ArrivalVelocity, AstVelocity, AstPosition, EarthPosition;
    }

    public static void Main()
    {
        var toks = new List<string>();
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            foreach (var t in line.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries))
                toks.Add(t);
        }
        if (toks.Count < 7) return;

        int p = 0;
        int n = int.Parse(toks[p++], CultureInfo.InvariantCulture);
        double warningDays = D(toks[p++]);
        double impactorMass = D(toks[p++]);
        double dvBudget = Math.Max(0.0, D(toks[p++]));
        double asteroidMass = Math.Max(1.0, D(toks[p++]));
        p++; // t0_unix_utc: not needed for this self-contained solver.

        var bodies = new List<Body>();
        for (int i = 0; i < n && p + 7 < toks.Count; i++)
        {
            var b = new Body();
            b.Name = toks[p++];
            b.Mass = D(toks[p++]);
            b.R = new Vec(D(toks[p++]), D(toks[p++]), D(toks[p++]));
            b.V = new Vec(D(toks[p++]), D(toks[p++]), D(toks[p++]));
            bodies.Add(b);
        }

        Body? earth = null, ast = null, sun = null;
        foreach (var b in bodies)
        {
            if (string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase)) earth = b;
            if (string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase)) ast = b;
            if (string.Equals(b.Name, "Sun", StringComparison.OrdinalIgnoreCase)) sun = b;
            if (sun == null || b.Mass > sun.Mass) sun = b;
        }

        if (earth == null || ast == null || sun == null || warningDays <= 0 || dvBudget <= 0)
        {
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        double mu = G * Math.Max(sun.Mass, 1.0);
        Candidate? best = Search(earth, ast, sun, mu, warningDays, dvBudget, impactorMass, asteroidMass);

        if (best == null)
        {
            Vec fallback = (ast.V - earth.V).Unit() * Math.Min(dvBudget, 1000.0);
            Console.WriteLine("1");
            Console.WriteLine(FormattableString.Invariant($"0 {fallback.X:R} {fallback.Y:R} {fallback.Z:R}"));
            return;
        }

        double used = best.LaunchDv;
        double remain = Math.Max(0.0, dvBudget - used);
        Vec rel = best.ArrivalVelocity - best.AstVelocity;
        Vec awayEarth = (best.AstPosition - best.EarthPosition).Unit();
        Vec transverse = Vec.Cross(best.AstVelocity.Unit(), Vec.Cross(awayEarth, best.AstVelocity.Unit())).Unit();
        Vec desired = (awayEarth * 0.75 + transverse * 0.25).Unit();
        if (Vec.Dot(desired, rel.Unit()) < -0.2) desired = desired * -1.0;
        Vec terminal = desired * remain;

        double midDay = (best.LaunchDay + best.ImpactDay) * 0.5;
        if (best.ImpactDay - best.LaunchDay < 2.0) midDay = best.LaunchDay;

        Console.WriteLine("3");
        Console.WriteLine(FormattableString.Invariant($"{best.LaunchDay:R} {best.LaunchBurn.X:R} {best.LaunchBurn.Y:R} {best.LaunchBurn.Z:R}"));
        Console.WriteLine(FormattableString.Invariant($"{midDay:R} 0 0 0"));
        Console.WriteLine(FormattableString.Invariant($"{best.ImpactDay:R} {terminal.X:R} {terminal.Y:R} {terminal.Z:R}"));
    }

    static Candidate? Search(Body earth, Body ast, Body sun, double mu, double warningDays, double dvBudget, double impactorMass, double asteroidMass)
    {
        Candidate? best = null;
        var launches = Grid(0.0, Math.Min(warningDays * 0.35, Math.Max(30.0, warningDays * 0.15)), 18);
        launches.Insert(0, 0.0);
        double minTof = Math.Max(1.0, Math.Min(20.0, warningDays * 0.03));

        foreach (double ld in launches)
        {
            if (ld > warningDays - minTof) continue;
            int arrivals = warningDays < 900 ? 80 : 120;
            for (int j = 0; j < arrivals; j++)
            {
                double f = (j + 1.0) / arrivals;
                double ad = ld + minTof + (warningDays * 0.985 - ld - minTof) * f;
                if (ad <= ld || ad >= warningDays) continue;

                State e = BodyState(earth, sun, mu, ld * Day);
                State a = BodyState(ast, sun, mu, ad * Day);
                foreach (int sign in new int[] { 1, -1 })
                {
                    if (!Lambert(e.R, a.R, (ad - ld) * Day, mu, sign, out Vec v1, out Vec v2)) continue;
                    Vec burn = v1 - e.V;
                    double launchDv = burn.Norm();
                    if (double.IsNaN(launchDv) || double.IsInfinity(launchDv) || launchDv > dvBudget * 0.995) continue;

                    State ea = BodyState(earth, sun, mu, ad * Day);
                    double remain = Math.Max(0.0, dvBudget - launchDv);
                    double lead = Math.Max(0.0, warningDays - ad);
                    double impactSpeed = (v2 - a.V).Norm() + remain;
                    double momentum = impactSpeed * impactorMass / asteroidMass;
                    double score = momentum * (lead + 1.0) / (1.0 + launchDv / Math.Max(1.0, dvBudget));

                    if (best == null || score > best.Score)
                    {
                        best = new Candidate
                        {
                            LaunchDay = ld,
                            ImpactDay = ad,
                            LaunchDv = launchDv,
                            LaunchBurn = burn,
                            ArrivalVelocity = v2,
                            AstVelocity = a.V,
                            AstPosition = a.R,
                            EarthPosition = ea.R,
                            Score = score
                        };
                    }
                }
            }
        }
        return best;
    }

    static List<double> Grid(double a, double b, int count)
    {
        var r = new List<double>();
        if (count <= 1 || b <= a) { r.Add(a); return r; }
        for (int i = 0; i < count; i++) r.Add(a + (b - a) * i / (count - 1));
        return r;
    }

    static State BodyState(Body b, Body sun, double mu, double t)
    {
        Vec sr = sun.R + sun.V * t;
        Vec sv = sun.V;
        State rel = PropagateKepler(b.R - sun.R, b.V - sun.V, t, mu);
        return new State(sr + rel.R, sv + rel.V);
    }

    static State PropagateKepler(Vec r0, Vec v0, double dt, double mu)
    {
        double r0n = r0.Norm();
        double vr0 = Vec.Dot(r0, v0) / Math.Max(r0n, 1.0);
        double alpha = 2.0 / Math.Max(r0n, 1.0) - Vec.Dot(v0, v0) / mu;
        double sqrtMu = Math.Sqrt(mu);
        double x = Math.Abs(alpha) > 1e-12 ? sqrtMu * Math.Abs(alpha) * dt : sqrtMu * dt / Math.Max(r0n, 1.0);
        if (x == 0) return new State(r0, v0);

        for (int i = 0; i < 35; i++)
        {
            double z = alpha * x * x;
            Stumpff(z, out double c, out double s);
            double f = r0n * vr0 / sqrtMu * x * x * c + (1.0 - alpha * r0n) * x * x * x * s + r0n * x - sqrtMu * dt;
            double fp = r0n * vr0 / sqrtMu * x * (1.0 - z * s) + (1.0 - alpha * r0n) * x * x * c + r0n;
            if (Math.Abs(fp) < 1e-9) break;
            double dx = f / fp;
            x -= dx;
            if (Math.Abs(dx) < 1e-6) break;
        }

        double zz = alpha * x * x;
        Stumpff(zz, out double C, out double S);
        double fcoef = 1.0 - x * x / r0n * C;
        double gcoef = dt - x * x * x / sqrtMu * S;
        Vec r = r0 * fcoef + v0 * gcoef;
        double rn = Math.Max(r.Norm(), 1.0);
        double fdot = sqrtMu / (rn * r0n) * (zz * S - 1.0) * x;
        double gdot = 1.0 - x * x / rn * C;
        Vec v = r0 * fdot + v0 * gdot;
        return new State(r, v);
    }

    static bool Lambert(Vec r1, Vec r2, double dt, double mu, int sinSign, out Vec v1, out Vec v2)
    {
        v1 = new Vec(); v2 = new Vec();
        double r1n = r1.Norm(), r2n = r2.Norm();
        if (r1n <= 0 || r2n <= 0 || dt <= 0) return false;
        double cosD = Math.Max(-1.0, Math.Min(1.0, Vec.Dot(r1, r2) / (r1n * r2n)));
        double sinD = sinSign * Math.Sqrt(Math.Max(0.0, 1.0 - cosD * cosD));
        double denom = 1.0 - cosD;
        if (Math.Abs(sinD) < 1e-10 || denom <= 1e-12) return false;
        double A = sinD * Math.Sqrt(r1n * r2n / denom);
        if (Math.Abs(A) < 1e-9) return false;

        double zLow = -4.0 * Math.PI * Math.PI, zHigh = 4.0 * Math.PI * Math.PI;
        double z = 0.0, y = 0.0, tof = 0.0;
        for (int i = 0; i < 80; i++)
        {
            z = 0.5 * (zLow + zHigh);
            if (!LambertTime(z, A, r1n, r2n, mu, out tof, out y) || y < 0)
            {
                zLow = z;
                continue;
            }
            if (tof <= dt) zLow = z; else zHigh = z;
        }
        z = 0.5 * (zLow + zHigh);
        if (!LambertTime(z, A, r1n, r2n, mu, out tof, out y) || y <= 0) return false;
        Stumpff(z, out double C, out double S);
        if (C <= 0) return false;
        double f = 1.0 - y / r1n;
        double g = A * Math.Sqrt(y / mu);
        double gd = 1.0 - y / r2n;
        if (Math.Abs(g) < 1e-9) return false;
        v1 = (r2 - r1 * f) / g;
        v2 = (r2 * gd - r1) / g;
        return !(double.IsNaN(v1.X) || double.IsNaN(v2.X));
    }

    static bool LambertTime(double z, double A, double r1, double r2, double mu, out double tof, out double y)
    {
        tof = 0; y = 0;
        Stumpff(z, out double C, out double S);
        if (C <= 0) return false;
        y = r1 + r2 + A * (z * S - 1.0) / Math.Sqrt(C);
        if (y < 0) return false;
        double x = Math.Sqrt(y / C);
        tof = (x * x * x * S + A * Math.Sqrt(y)) / Math.Sqrt(mu);
        return !(double.IsNaN(tof) || double.IsInfinity(tof));
    }

    static void Stumpff(double z, out double C, out double S)
    {
        if (z > 1e-8)
        {
            double sz = Math.Sqrt(z);
            C = (1.0 - Math.Cos(sz)) / z;
            S = (sz - Math.Sin(sz)) / (sz * sz * sz);
        }
        else if (z < -1e-8)
        {
            double sz = Math.Sqrt(-z);
            C = (Math.Cosh(sz) - 1.0) / (-z);
            S = (Math.Sinh(sz) - sz) / (sz * sz * sz);
        }
        else
        {
            C = 0.5;
            S = 1.0 / 6.0;
        }
    }

    static double D(string s) => double.Parse(s, NumberStyles.Float, CultureInfo.InvariantCulture);
}
