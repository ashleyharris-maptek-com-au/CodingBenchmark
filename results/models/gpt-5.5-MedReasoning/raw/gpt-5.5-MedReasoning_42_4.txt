OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %outColor "outColor"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0
%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_thick_start = OpConstant %float 0.04
%float_thick_end = OpConstant %float 0.005
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%arr_v3float_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_2
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output
%main = OpFunction %void None %fn_void
%entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v3 %inWorldPos %int_0
%p1_ptr = OpAccessChain %ptr_in_v3 %inWorldPos %int_1
%p0 = OpLoad %v3float %p0_ptr
%p1 = OpLoad %v3float %p1_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %inColor %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p0_z = OpCompositeExtract %float %p0 2
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%p1_z = OpCompositeExtract %float %p1 2
%dx = OpFSub %float %p1_x %p0_x
%dy = OpFSub %float %p1_y %p0_y
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len2 = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSLstd450 Sqrt %len2
%inv_len = OpFDiv %float %float_1 %len
%neg_dy = OpFNegate %float %dy
%perp_x = OpFMul %float %neg_dy %inv_len
%perp_y = OpFMul %float %dx %inv_len
%off0_x = OpFMul %float %perp_x %float_thick_start
%off0_y = OpFMul %float %perp_y %float_thick_start
%off1_x = OpFMul %float %perp_x %float_thick_end
%off1_y = OpFMul %float %perp_y %float_thick_end
%p0_plus_x = OpFAdd %float %p0_x %off0_x
%p0_plus_y = OpFAdd %float %p0_y %off0_y
%p0_minus_x = OpFSub %float %p0_x %off0_x
%p0_minus_y = OpFSub %float %p0_y %off0_y
%p1_plus_x = OpFAdd %float %p1_x %off1_x
%p1_plus_y = OpFAdd %float %p1_y %off1_y
%p1_minus_x = OpFSub %float %p1_x %off1_x
%p1_minus_y = OpFSub %float %p1_y %off1_y
%v0 = OpCompositeConstruct %v4float %p0_plus_x %p0_plus_y %p0_z %float_1
%v1 = OpCompositeConstruct %v4float %p0_minus_x %p0_minus_y %p0_z %float_1
%v2 = OpCompositeConstruct %v4float %p1_plus_x %p1_plus_y %p1_z %float_1
%v3 = OpCompositeConstruct %v4float %p1_minus_x %p1_minus_y %p1_z %float_1
%pos_out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr0 %v0
OpStore %outColor %c0
OpEmitVertex
%pos_out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr1 %v1
OpStore %outColor %c0
OpEmitVertex
%pos_out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr2 %v2
OpStore %outColor %c1
OpEmitVertex
%pos_out_ptr3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_out_ptr3 %v3
OpStore %outColor %c1
OpEmitVertex
OpEndPrimitive
OpReturn
