; SPIR-V
; Version: 1.0
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_5 = OpConstant %float 0.5
%float_0_8 = OpConstant %float 0.8
%float_3 = OpConstant %float 3
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%baseColor = OpConstantComposite %v3float %float_0_2 %float_0_3 %float_0_8
%rimTint = OpConstantComposite %v3float %float_1 %float_0_8 %float_0_5
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptrInputV3 = OpTypePointer Input %v3float
%ptrOutputV4 = OpTypePointer Output %v4float
%ptrUniformUBO = OpTypePointer Uniform %UBO
%ptrUniformV4 = OpTypePointer Uniform %v4float
%inWorldPos = OpVariable %ptrInputV3 Input
%inNormal = OpVariable %ptrInputV3 Input
%outColor = OpVariable %ptrOutputV4 Output
%ubo = OpVariable %ptrUniformUBO Uniform
%main = OpFunction %void None %func
%entry = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %1 Normalize %normal
%lightPtr = OpAccessChain %ptrUniformV4 %ubo %int_3
%light4 = OpLoad %v4float %lightPtr
%lightPos = OpVectorShuffle %v3float %light4 %light4 0 1 2
%lightDelta = OpFSub %v3float %lightPos %worldPos
%L = OpExtInst %v3float %1 Normalize %lightDelta
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %dotNL %float_0
%cameraPtr = OpAccessChain %ptrUniformV4 %ubo %int_4
%camera4 = OpLoad %v4float %cameraPtr
%cameraPos = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%viewDelta = OpFSub %v3float %cameraPos %worldPos
%V = OpExtInst %v3float %1 Normalize %viewDelta
%dotNV = OpDot %float %N %V
%facing = OpExtInst %float %1 FMax %dotNV %float_0
%oneMinusFacing = OpFSub %float %float_1 %facing
%rim = OpExtInst %float %1 Pow %oneMinusFacing %float_3
%halfDiffuse = OpFMul %float %float_0_5 %diffuse
%diffuseScale = OpFAdd %float %float_0_3 %halfDiffuse
%litBase = OpVectorTimesScalar %v3float %baseColor %diffuseScale
%rimColor = OpVectorTimesScalar %v3float %rimTint %rim
%rgb = OpFAdd %v3float %litBase %rimColor
%rgba = OpCompositeExtract %float %rgb 0
%rgbb = OpCompositeExtract %float %rgb 1
%rgbc = OpCompositeExtract %float %rgb 2
%finalColor = OpCompositeConstruct %v4float %rgba %rgbb %rgbc %float_1
OpStore %outColor %finalColor
OpReturn
OpFunctionEnd
