#include <algorithm>
#include <array>
#include <cstdint>
#include <cstring>
#include <iostream>

#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using u32 = std::uint32_t;
using u64 = std::uint64_t;

static inline void trim_trailing_cr(std::string& s) {
    if (!s.empty() && s.back() == '\r') {
        s.pop_back();
    }
}

static inline u64 splitmix64(u64 x) noexcept {
    x += 0x9e3779b97f4a7c15ULL;
    x = (x ^ (x >> 30)) * 0xbf58476d1ce4e5b9ULL;
    x = (x ^ (x >> 27)) * 0x94d049bb133111ebULL;
    return x ^ (x >> 31);
}

struct HashKey {
    u64 h1;
    u64 h2;

    bool operator==(const HashKey& other) const noexcept {
        return h1 == other.h1 && h2 == other.h2;
    }
};

struct HashKeyHasher {
    std::size_t operator()(const HashKey& key) const noexcept {
        return static_cast<std::size_t>(splitmix64(key.h1) ^ (splitmix64(key.h2) >> 1));
    }
};

static bool contains_substring_exact(const std::string& text, std::string_view pattern) {
    if (pattern.empty()) {
        return true;
    }
    return text.find(pattern.data(), 0, pattern.size()) != std::string::npos;
}

/*
 * Strategy 1: tiny 2-string cases
 * O(|a| * |b|) time, O(min(|a|, |b|)) memory
 */
static std::string solve_dp_two_strings(const std::string& a, const std::string& b) {
    const std::string* outer = &a;
    const std::string* inner = &b;
    if (inner->size() > outer->size()) {
        std::swap(outer, inner);
    }

    std::vector<u32> dp(inner->size() + 1, 0u);
    u32 best_len = 0;
    std::size_t end_pos = 0;

    for (std::size_t i = 1; i <= outer->size(); ++i) {
        for (std::size_t j = inner->size(); j > 0; --j) {
            if ((*outer)[i - 1] == (*inner)[j - 1]) {
                dp[j] = dp[j - 1] + 1u;
                if (dp[j] > best_len) {
                    best_len = dp[j];
                    end_pos = i;
                }
            } else {
                dp[j] = 0;
            }
        }
    }

    if (best_len == 0) {
        return {};
    }

    const std::size_t start = end_pos - best_len;
    return std::string(outer->data() + start, best_len);
}

/*
 * Strategy 2: moderate cases
 * Binary search on answer length + double rolling hash as filter.
 * Exactness is preserved because every surviving candidate is validated
 * by literal substring search before acceptance.
 */
class RollingHash2 {
private:
    static constexpr u64 BASE1 = 0x9E3779B185EBCA87ULL;
    static constexpr u64 BASE2 = 0xC2B2AE3D27D4EB4FULL;

public:
    std::vector<u64> pref1;
    std::vector<u64> pref2;

    void build(const std::string& s) {
        pref1.assign(s.size() + 1, 0);
        pref2.assign(s.size() + 1, 0);

        const unsigned char* data =
            reinterpret_cast<const unsigned char*>(s.data());

        for (std::size_t i = 0; i < s.size(); ++i) {
            const u64 x = static_cast<u64>(data[i]) + 1u;
            pref1[i + 1] = pref1[i] * BASE1 + x;
            pref2[i + 1] = pref2[i] * BASE2 + x;
        }
    }

    static void build_powers(std::size_t max_len,
                             std::vector<u64>& pow1,
                             std::vector<u64>& pow2) {
        pow1.assign(max_len + 1, 0);
        pow2.assign(max_len + 1, 0);
        pow1[0] = 1;
        pow2[0] = 1;

        for (std::size_t i = 1; i <= max_len; ++i) {
            pow1[i] = pow1[i - 1] * BASE1;
            pow2[i] = pow2[i - 1] * BASE2;
        }
    }

    HashKey get(std::size_t pos,
                std::size_t len,
                const std::vector<u64>& pow1,
                const std::vector<u64>& pow2) const noexcept {
        return {
            pref1[pos + len] - pref1[pos] * pow1[len],
            pref2[pos + len] - pref2[pos] * pow2[len]
        };
    }
};

static std::string solve_hash_binary_search(const std::vector<std::string>& strings) {
    const std::size_t n = strings.size();
    const std::string& base = strings[0];

    std::vector<u64> pow1;
    std::vector<u64> pow2;
    RollingHash2::build_powers(base.size(), pow1, pow2);

    std::vector<RollingHash2> hashes(n);
    for (std::size_t i = 0; i < n; ++i) {
        hashes[i].build(strings[i]);
    }

    auto check = [&](std::size_t L, std::string& witness) -> bool {
        if (L == 0) {
            witness.clear();
            return true;
        }
        if (base.size() < L) {
            return false;
        }

        using PosList = std::vector<u32>;
        std::unordered_map<HashKey, PosList, HashKeyHasher> candidates;
        candidates.max_load_factor(0.7f);

        const std::size_t base_count = base.size() - L + 1;
        candidates.reserve(base_count);

        for (std::size_t pos = 0; pos < base_count; ++pos) {
            const HashKey h = hashes[0].get(pos, L, pow1, pow2);
            auto [it, inserted] = candidates.try_emplace(h);
            it->second.push_back(static_cast<u32>(pos));
        }

        for (std::size_t i = 1; i < n; ++i) {
            if (strings[i].size() < L) {
                return false;
            }

            const std::size_t cnt = strings[i].size() - L + 1;
            std::unordered_set<HashKey, HashKeyHasher> present;
            present.max_load_factor(0.7f);
            present.reserve(cnt);

            for (std::size_t pos = 0; pos < cnt; ++pos) {
                present.insert(hashes[i].get(pos, L, pow1, pow2));
            }

            for (auto it = candidates.begin(); it != candidates.end();) {
                if (present.find(it->first) == present.end()) {
                    it = candidates.erase(it);
                } else {
                    ++it;
                }
            }

            if (candidates.empty()) {
                return false;
            }
        }

        for (const auto& entry : candidates) {
            for (u32 pos : entry.second) {
                const std::string_view sub(base.data() + pos, L);
                bool ok = true;
                for (std::size_t i = 1; i < n; ++i) {
                    if (!contains_substring_exact(strings[i], sub)) {
                        ok = false;
                        break;
                    }
                }
                if (ok) {
                    witness.assign(sub.data(), sub.size());
                    return true;
                }
            }
        }

        return false;
    };

    std::string best;
    std::size_t lo = 0;
    std::size_t hi = base.size();

    while (lo < hi) {
        const std::size_t mid = lo + (hi - lo + 1) / 2;
        std::string witness;
        if (check(mid, witness)) {
            lo = mid;
            best = std::move(witness);
        } else {
            hi = mid - 1;
        }
    }

    return best;
}

/*
 * Strategy 3: large / extreme cases
 * Generalized suffix automaton built on the shortest string.
 *
 * Deterministic and exact.
 * Memory is O(shortest_string_length).
 * Time is roughly O(total_input + states * number_of_other_strings),
 * which is the intended workhorse for huge inputs.
 */
class GeneralizedSuffixAutomaton {
private:
    static constexpr u32 NONE = std::numeric_limits<u32>::max();

    struct Edge {
        u32 to;
        u32 next;
        unsigned char ch;
    };

    std::vector<u32> len_;
    std::vector<u32> link_;
    std::vector<u32> head_;
    std::vector<u32> first_pos_;
    std::vector<Edge> edges_;
    std::array<u32, 256> root_next_{};
    u32 last_ = 0;

    u32 new_state(u32 length) {
        const u32 id = static_cast<u32>(len_.size());
        len_.push_back(length);
        link_.push_back(NONE);
        head_.push_back(NONE);
        first_pos_.push_back(0);
        return id;
    }

    inline u32 next_state(u32 state, unsigned char c) const noexcept {
        if (state == 0) {
            return root_next_[c];
        }
        for (u32 e = head_[state]; e != NONE; e = edges_[e].next) {
            if (edges_[e].ch == c) {
                return edges_[e].to;
            }
        }
        return NONE;
    }

    inline u32* find_existing_transition_ptr(u32 state, unsigned char c) noexcept {
        if (state == 0) {
            return (root_next_[c] == NONE) ? nullptr : &root_next_[c];
        }
        for (u32 e = head_[state]; e != NONE; e = edges_[e].next) {
            if (edges_[e].ch == c) {
                return &edges_[e].to;
            }
        }
        return nullptr;
    }

    inline void add_transition(u32 state, unsigned char c, u32 to) {
        if (state == 0) {
            root_next_[c] = to;
            return;
        }
        edges_.push_back(Edge{to, head_[state], c});
        head_[state] = static_cast<u32>(edges_.size() - 1);
    }

    void copy_transitions(u32 from, u32 to) {
        u32 new_head = NONE;

        if (from == 0) {
            for (u32 c = 0; c < 256; ++c) {
                if (root_next_[c] != NONE) {
                    edges_.push_back(Edge{
                        root_next_[c],
                        new_head,
                        static_cast<unsigned char>(c)
                    });
                    new_head = static_cast<u32>(edges_.size() - 1);
                }
            }
        } else {
            for (u32 e = head_[from]; e != NONE; e = edges_[e].next) {
                edges_.push_back(Edge{edges_[e].to, new_head, edges_[e].ch});
                new_head = static_cast<u32>(edges_.size() - 1);
            }
        }

        head_[to] = new_head;
    }

    void extend(unsigned char c, u32 pos) {
        const u32 cur = new_state(len_[last_] + 1u);
        first_pos_[cur] = pos;

        u32 p = last_;
        while (p != NONE) {
            u32* trans = find_existing_transition_ptr(p, c);
            if (trans == nullptr) {
                add_transition(p, c, cur);
                p = link_[p];
                continue;
            }

            const u32 q = *trans;
            if (len_[p] + 1u == len_[q]) {
                link_[cur] = q;
            } else {
                const u32 clone = new_state(len_[p] + 1u);
                copy_transitions(q, clone);
                link_[clone] = link_[q];
                first_pos_[clone] = first_pos_[q];

                while (p != NONE) {
                    u32* t = find_existing_transition_ptr(p, c);
                    if (t == nullptr || *t != q) {
                        break;
                    }
                    *t = clone;
                    p = link_[p];
                }

                link_[q] = clone;
                link_[cur] = clone;
            }

            last_ = cur;
            return;
        }

        link_[cur] = 0;
        last_ = cur;
    }

    static std::vector<u32> radix_order_by_len(const std::vector<u32>& key) {
        const std::size_t n = key.size();
        std::vector<u32> order(n);
        std::vector<u32> temp(n);

        for (std::size_t i = 0; i < n; ++i) {
            order[i] = static_cast<u32>(i);
        }

        std::vector<std::size_t> count(1u << 16, 0);

        auto pass = [&](unsigned shift,
                        const std::vector<u32>& src,
                        std::vector<u32>& dst) {
            std::fill(count.begin(), count.end(), 0);

            for (std::size_t i = 0; i < n; ++i) {
                ++count[(key[src[i]] >> shift) & 0xFFFFu];
            }

            std::size_t sum = 0;
            for (std::size_t i = 0; i < count.size(); ++i) {
                const std::size_t c = count[i];
                count[i] = sum;
                sum += c;
            }

            for (std::size_t i = 0; i < n; ++i) {
                const u32 state = src[i];
                dst[count[(key[state] >> shift) & 0xFFFFu]++] = state;
            }
        };

        pass(0, order, temp);
        pass(16, temp, order);
        return order;
    }

    void process_string(const std::string& s,
                        const std::vector<u32>& order,
                        std::vector<u32>& min_match,
                        std::vector<u32>& cur_match,
                        u32& round_best) const {
        const std::size_t state_count = len_.size();
        std::memset(cur_match.data(), 0, state_count * sizeof(u32));

        const u32* len = len_.data();
        const u32* link = link_.data();
        const u32* ord = order.data();
        u32* mn = min_match.data();
        u32* cur = cur_match.data();

        u32 v = 0;
        u32 l = 0;

        const unsigned char* data =
            reinterpret_cast<const unsigned char*>(s.data());

        for (std::size_t i = 0; i < s.size(); ++i) {
            const unsigned char c = data[i];
            u32 nxt = next_state(v, c);

            while (v != 0 && nxt == NONE) {
                v = link[v];
                l = len[v];
                nxt = next_state(v, c);
            }

            if (nxt != NONE) {
                v = nxt;
                ++l;
            } else {
                v = 0;
                l = 0;
            }

            if (l > cur[v]) {
                cur[v] = l;
            }
        }

        round_best = 0;

        for (std::size_t idx = state_count; idx-- > 1;) {
            const u32 state = ord[idx];
            const u32 matched = cur[state];

            if (matched < mn[state]) {
                mn[state] = matched;
            }
            if (mn[state] > round_best) {
                round_best = mn[state];
            }

            const u32 parent = link[state];
            if (parent != NONE && matched != 0) {
                u32 propagated = matched;
                if (propagated > len[parent]) {
                    propagated = len[parent];
                }
                if (propagated > cur[parent]) {
                    cur[parent] = propagated;
                }
            }
        }

        mn[0] = 0;
    }

public:
    explicit GeneralizedSuffixAutomaton(std::size_t base_len) {
        const std::size_t state_reserve = base_len * 2ull + 1ull;
        const std::size_t edge_reserve = base_len * 3ull + 4ull;

        len_.reserve(state_reserve);
        link_.reserve(state_reserve);
        head_.reserve(state_reserve);
        first_pos_.reserve(state_reserve);
        edges_.reserve(edge_reserve);

        root_next_.fill(NONE);

        new_state(0);
        link_[0] = NONE;
        last_ = 0;
    }

    void build(const std::string& s) {
        const unsigned char* data =
            reinterpret_cast<const unsigned char*>(s.data());

        for (std::size_t i = 0; i < s.size(); ++i) {
            extend(data[i], static_cast<u32>(i));
        }
    }

    std::string longest_common_across(std::vector<std::string>& strings) const {
        const std::string& base = strings[0];

        std::vector<u32> order = radix_order_by_len(len_);
        std::vector<u32> min_match = len_;
        std::vector<u32> cur_match(len_.size(), 0u);

        for (std::size_t i = 1; i < strings.size(); ++i) {
            u32 round_best = 0;
            process_string(strings[i], order, min_match, cur_match, round_best);

            if (round_best == 0) {
                return {};
            }

            // Free processed string memory early on huge inputs.
            std::string().swap(strings[i]);
        }

        u32 best_state = 0;
        u32 best_len = 0;

        for (std::size_t state = 1; state < min_match.size(); ++state) {
            if (min_match[state] > best_len) {
                best_len = min_match[state];
                best_state = static_cast<u32>(state);
            }
        }

        if (best_len == 0) {
            return {};
        }

        const std::size_t start =
            static_cast<std::size_t>(first_pos_[best_state]) + 1u -
            static_cast<std::size_t>(best_len);

        return std::string(base.data() + start, best_len);
    }
};

static std::string solve_generalized_sam(std::vector<std::string>& strings) {
    GeneralizedSuffixAutomaton sam(strings[0].size());
    sam.build(strings[0]);
    return sam.longest_common_across(strings);
}

enum class Strategy {
    DynamicProgramming,
    RollingHash,
    SuffixAutomaton
};

static Strategy choose_strategy(const std::vector<std::string>& strings, u64 total_len) {
    static constexpr u64 DP_CELL_LIMIT = 40'000'000ULL;
    static constexpr std::size_t HASH_SHORTEST_LIMIT = 300'000;
    static constexpr u64 HASH_TOTAL_LIMIT = 5'000'000ULL;
    static constexpr std::size_t HASH_N_LIMIT = 8;

    if (strings.size() == 2) {
        const u64 cells =
            static_cast<u64>(strings[0].size()) *
            static_cast<u64>(strings[1].size());

        if (cells <= DP_CELL_LIMIT) {
            return Strategy::DynamicProgramming;
        }
    }

    if (strings.size() <= HASH_N_LIMIT &&
        strings[0].size() <= HASH_SHORTEST_LIMIT &&
        total_len <= HASH_TOTAL_LIMIT) {
        return Strategy::RollingHash;
    }

    return Strategy::SuffixAutomaton;
}

static std::string solve_adaptive(std::vector<std::string>& strings) {
    if (strings.empty()) {
        return {};
    }
    if (strings.size() == 1) {
        return strings[0];
    }

    u64 total_len = 0;
    std::size_t min_index = 0;

    for (std::size_t i = 0; i < strings.size(); ++i) {
        total_len += static_cast<u64>(strings[i].size());
        if (strings[i].empty()) {
            return {};
        }
        if (strings[i].size() < strings[min_index].size()) {
            min_index = i;
        }
    }

    if (min_index != 0) {
        std::swap(strings[0], strings[min_index]);
    }

    if (strings.size() > 2) {
        std::sort(strings.begin() + 1, strings.end(),
                  [](const std::string& lhs, const std::string& rhs) {
                      return lhs.size() < rhs.size();
                  });
    }

    switch (choose_strategy(strings, total_len)) {
    case Strategy::DynamicProgramming:
        return solve_dp_two_strings(strings[0], strings[1]);
    case Strategy::RollingHash:
        return solve_hash_binary_search(strings);
    case Strategy::SuffixAutomaton:
    default:
        return solve_generalized_sam(strings);
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    std::size_t n = 0;
    if (!(std::cin >> n)) {
        return 0;
    }

    std::cin.ignore(std::numeric_limits<std::streamsize>::max(), '\n');

    std::vector<std::string> strings;
    strings.reserve(n);

    for (std::size_t i = 0; i < n; ++i) {
        std::string line;
        if (!std::getline(std::cin, line)) {
            line.clear();
        }
        trim_trailing_cr(line);
        strings.push_back(std::move(line));
    }

    const std::string answer = solve_adaptive(strings);
    std::cout << answer << '\n';
    return 0;
}
