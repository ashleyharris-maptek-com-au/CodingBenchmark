use std::cmp::Reverse;

const INVALID: usize = usize::MAX;
const FR_MAX_N: usize = 1500;
const FR_BUDGET_INIT: u64 = 120_000_000;

#[derive(Clone, Copy, Debug)]
struct BBox {
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
}

impl BBox {
    fn empty() -> Self {
        Self {
            minx: f64::INFINITY,
            maxx: f64::NEG_INFINITY,
            miny: f64::INFINITY,
            maxy: f64::NEG_INFINITY,
        }
    }

    fn update(&mut self, x: f64, y: f64) {
        if !x.is_finite() || !y.is_finite() {
            return;
        }
        if x < self.minx {
            self.minx = x;
        }
        if x > self.maxx {
            self.maxx = x;
        }
        if y < self.miny {
            self.miny = y;
        }
        if y > self.maxy {
            self.maxy = y;
        }
    }

    fn width(&self) -> f64 {
        if self.minx.is_infinite() {
            0.0
        } else {
            self.maxx - self.minx
        }
    }

    fn height(&self) -> f64 {
        if self.miny.is_infinite() {
            0.0
        } else {
            self.maxy - self.miny
        }
    }
}

struct FastInput {
    data: Vec<u8>,
    idx: usize,
}

impl FastInput {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, idx: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.idx < self.data.len() && self.data[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        let mut num = 0usize;
        while self.idx < self.data.len() {
            let b = self.data[self.idx];
            if b.is_ascii_digit() {
                num = num * 10 + (b - b'0') as usize;
                self.idx += 1;
            } else {
                break;
            }
        }
        num
    }
}

struct BfsWork {
    queue: Vec<usize>,
    seen: Vec<u32>,
    dist: Vec<i32>,
    parent: Vec<usize>,
    stamp: u32,
}

impl BfsWork {
    fn new(n: usize) -> Self {
        Self {
            queue: Vec::new(),
            seen: vec![0; n],
            dist: vec![0; n],
            parent: vec![0; n],
            stamp: 0,
        }
    }

    fn next_stamp(&mut self) -> u32 {
        self.stamp = self.stamp.wrapping_add(1);
        if self.stamp == 0 {
            self.seen.fill(0);
            self.stamp = 1;
        }
        self.stamp
    }
}

fn bfs(graph: &Graph, work: &mut BfsWork, start: usize, record_parent: bool) -> (usize, i32) {
    let stamp = work.next_stamp();
    work.queue.clear();
    work.queue.push(start);
    work.seen[start] = stamp;
    work.dist[start] = 0;
    if record_parent {
        work.parent[start] = start;
    }

    let mut head = 0usize;
    let mut farthest = start;
    let mut far_dist = 0i32;

    while head < work.queue.len() {
        let v = work.queue[head];
        head += 1;
        let dv = work.dist[v];
        if dv > far_dist {
            far_dist = dv;
            farthest = v;
        }

        for ei in graph.offsets[v]..graph.offsets[v + 1] {
            let to = graph.adj[ei] as usize;
            if work.seen[to] != stamp {
                work.seen[to] = stamp;
                work.dist[to] = dv + 1;
                if record_parent {
                    work.parent[to] = v;
                }
                work.queue.push(to);
            }
        }
    }

    (farthest, far_dist)
}

fn splitmix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E3779B97F4A7C15);
    let mut z = x;
    z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
    z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
    z ^ (z >> 31)
}

fn pseudo01(x: u64) -> f64 {
    let v = splitmix64(x) >> 11;
    (v as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn pseudo_angle(x: u64) -> f64 {
    2.0 * PI * pseudo01(x)
}

fn bbox_of_vertices(coords: &[(f64, f64)], verts: &[usize]) -> BBox {
    let mut bb = BBox::empty();
    for &v in verts {
        bb.update(coords[v].0, coords[v].1);
    }
    bb
}

fn normalize_component(verts: &[usize], coords: &mut [(f64, f64)]) -> BBox {
    if verts.is_empty() {
        return BBox::empty();
    }

    for &v in verts {
        if !coords[v].0.is_finite() || !coords[v].1.is_finite() {
            coords[v] = (0.0, 0.0);
        }
    }

    let mut bb = bbox_of_vertices(coords, verts);

    if verts.len() <= 1 {
        return bb;
    }

    let mut base = bb.width().max(bb.height());
    if base < 1e-9 {
        let n = verts.len();
        for (i, &v) in verts.iter().enumerate() {
            let ang = 2.0 * PI * (i as f64) / (n as f64);
            coords[v] = (ang.cos(), ang.sin());
        }
        bb = bbox_of_vertices(coords, verts);
        base = bb.width().max(bb.height()).max(1e-9);
    }

    let cx = 0.5 * (bb.minx + bb.maxx);
    let cy = 0.5 * (bb.miny + bb.maxy);
    let target_extent = (verts.len() as f64).sqrt() * 2.5 + 1.0;
    let scale = target_extent / base.max(1e-9);

    for &v in verts {
        coords[v].0 = (coords[v].0 - cx) * scale;
        coords[v].1 = (coords[v].1 - cy) * scale;
    }

    bbox_of_vertices(coords, verts)
}

fn midpoint_on_parent_path(start: usize, end: usize, parent: &[usize]) -> usize {
    let mut path = Vec::new();
    let mut cur = end;
    path.push(cur);
    while cur != start {
        cur = parent[cur];
        path.push(cur);
    }
    path[path.len() / 2]
}

fn radial_layout_from_rooted_tree(
    verts: &[usize],
    root: usize,
    parent: &[usize],
    dist: &[i32],
    order_globals: &[usize],
    local_index: &mut [usize],
    coords: &mut [(f64, f64)],
) {
    let n = verts.len();
    for (i, &v) in verts.iter().enumerate() {
        local_index[v] = i;
    }

    let root_li = local_index[root];

    let mut depth = vec![0usize; n];
    let mut child_count = vec![0usize; n];

    for &v in order_globals {
        let li = local_index[v];
        depth[li] = dist[v].max(0) as usize;
        if v != root {
            let p = parent[v];
            let pli = local_index[p];
            child_count[pli] += 1;
        }
    }

    let mut child_offsets = vec![0usize; n + 1];
    for i in 0..n {
        child_offsets[i + 1] = child_offsets[i] + child_count[i];
    }

    let mut fill = child_offsets[..n].to_vec();
    let mut children = vec![0usize; n.saturating_sub(1)];

    for &v in order_globals {
        if v == root {
            continue;
        }
        let li = local_index[v];
        let p = parent[v];
        let pli = local_index[p];
        let pos = fill[pli];
        children[pos] = li;
        fill[pli] += 1;
    }

    let mut subtree = vec![1usize; n];
    for &v in order_globals.iter().rev() {
        if v == root {
            continue;
        }
        let li = local_index[v];
        let pli = local_index[parent[v]];
        subtree[pli] += subtree[li];
    }

    coords[root] = (0.0, 0.0);

    let mut stack: Vec<(usize, f64, f64)> = Vec::with_capacity(n);
    stack.push((root_li, 0.0, 2.0 * PI));

    while let Some((li, l, r)) = stack.pop() {
        let v = verts[li];

        if li != root_li {
            let ang = 0.5 * (l + r);
            let rad = depth[li] as f64;
            coords[v] = (rad * ang.cos(), rad * ang.sin());
        }

        let start = child_offsets[li];
        let end = child_offsets[li + 1];
        if start == end {
            continue;
        }

        let total = subtree[li] - 1;
        let span = r - l;
        let mut cur = l;
        let mut segments: Vec<(usize, f64, f64)> = Vec::with_capacity(end - start);

        for idx in start..end {
            let child = children[idx];
            let frac = subtree[child] as f64 / total as f64;
            let cl = cur;
            let cr = if idx + 1 == end { r } else { cur + span * frac };
            segments.push((child, cl, cr));
            cur = cr;
        }

        for seg in segments.into_iter().rev() {
            stack.push(seg);
        }
    }

    for &v in verts {
        local_index[v] = INVALID;
    }
}

fn tree_layout(
    graph: &Graph,
    verts: &[usize],
    bfs_work: &mut BfsWork,
    local_index: &mut [usize],
    coords: &mut [(f64, f64)],
) {
    let seed = verts[0];
    let (a, _) = bfs(graph, bfs_work, seed, false);
    let (b, _) = bfs(graph, bfs_work, a, true);
    let root = midpoint_on_parent_path(a, b, &bfs_work.parent);

    let _ = bfs(graph, bfs_work, root, true);
    radial_layout_from_rooted_tree(
        verts,
        root,
        &bfs_work.parent,
        &bfs_work.dist,
        &bfs_work.queue,
        local_index,
        coords,
    );
}

fn cycle_layout(graph: &Graph, verts: &[usize], coords: &mut [(f64, f64)]) -> bool {
    let n = verts.len();
    if n < 3 {
        return false;
    }

    let start = verts[0];
    let mut order = Vec::with_capacity(n);
    let mut prev = INVALID;
    let mut cur = start;

    for _ in 0..n {
        order.push(cur);
        let mut next = INVALID;
        for ei in graph.offsets[cur]..graph.offsets[cur + 1] {
            let to = graph.adj[ei] as usize;
            if to != prev {
                next = to;
                break;
            }
        }
        if next == INVALID {
            return false;
        }
        prev = cur;
        cur = next;
        if cur == start {
            break;
        }
    }

    if order.len() != n {
        return false;
    }

    for (i, &v) in order.iter().enumerate() {
        let ang = 2.0 * PI * (i as f64) / (n as f64);
        coords[v] = (ang.cos(), ang.sin());
    }
    true
}

fn shell_layout(
    graph: &Graph,
    verts: &[usize],
    root: usize,
    local_index: &mut [usize],
    bfs_work: &mut BfsWork,
    coords: &mut [(f64, f64)],
) {
    for (i, &v) in verts.iter().enumerate() {
        local_index[v] = i;
    }

    let _ = bfs(graph, bfs_work, root, false);

    let mut max_layer = 0usize;
    for &v in verts {
        let d = bfs_work.dist[v].max(0) as usize;
        if d > max_layer {
            max_layer = d;
        }
    }

    let mut layers: Vec<Vec<usize>> = (0..=max_layer).map(|_| Vec::new()).collect();
    for &v in verts {
        let li = local_index[v];
        let d = bfs_work.dist[v].max(0) as usize;
        layers[d].push(li);
    }

    let mut angles = vec![0.0f64; verts.len()];
    let root_li = local_index[root];
    angles[root_li] = 0.0;
    coords[root] = (0.0, 0.0);

    let layer_gap = 1.8;

    for layer in 1..=max_layer {
        let mut items: Vec<(f64, usize)> = Vec::with_capacity(layers[layer].len());

        for &li in &layers[layer] {
            let v = verts[li];
            let mut sx = 0.0;
            let mut sy = 0.0;

            for ei in graph.offsets[v]..graph.offsets[v + 1] {
                let to = graph.adj[ei] as usize;
                if bfs_work.dist[to] == bfs_work.dist[v] - 1 {
                    let tli = local_index[to];
                    let a = angles[tli];
                    sx += a.cos();
                    sy += a.sin();
                }
            }

            let pref = if sx.abs() + sy.abs() > 1e-12 {
                sy.atan2(sx)
            } else {
                pseudo_angle(v as u64)
            };
            items.push((pref, li));
        }

        items.sort_by(|a, b| {
            a.0.total_cmp(&b.0)
                .then_with(|| a.1.cmp(&b.1))
        });

        let k = items.len();
        if k == 0 {
            continue;
        }

        let r = layer_gap * layer as f64;

        if k == 1 {
            let ang = items[0].0;
            let li = items[0].1;
            angles[li] = ang;
            let v = verts[li];
            coords[v] = (r * ang.cos(), r * ang.sin());
            continue;
        }

        let step = 2.0 * PI / (k as f64);
        let mut sumc = 0.0;
        let mut sums = 0.0;
        for (rank, (pref, _)) in items.iter().enumerate() {
            let base = step * (rank as f64);
            let diff = *pref - base;
            sumc += diff.cos();
            sums += diff.sin();
        }
        let offset = if sumc.abs() + sums.abs() > 1e-12 {
            sums.atan2(sumc)
        } else {
            0.0
        };

        for (rank, (_, li)) in items.into_iter().enumerate() {
            let ang = offset + step * (rank as f64);
            angles[li] = ang;
            let v = verts[li];
            coords[v] = (r * ang.cos(), r * ang.sin());
        }
    }

    for &v in verts {
        local_index[v] = INVALID;
    }
}

fn choose_pivot_count(n: usize) -> usize {
    if n <= 128 {
        4
    } else if n <= 2_000 {
        6
    } else if n <= 20_000 {
        8
    } else {
        6
    }
}

fn choose_next_pivot(verts: &[usize], min_d: &[i32], pivots: &[usize]) -> Option<(usize, i32)> {
    let mut best_v = None;
    let mut best_d = -1i32;

    'outer: for (i, &v) in verts.iter().enumerate() {
        for &p in pivots {
            if p == v {
                continue 'outer;
            }
        }
        let d = min_d[i];
        if d > best_d {
            best_d = d;
            best_v = Some(v);
        }
    }

    best_v.map(|v| (v, best_d))
}

fn normalize_vec(x: &mut [f64]) -> bool {
    let norm = x.iter().map(|v| v * v).sum::<f64>().sqrt();
    if norm <= 1e-12 {
        return false;
    }
    for v in x {
        *v /= norm;
    }
    true
}

fn dominant_eigen(cov: &[f64], k: usize, orth: &[&[f64]]) -> Vec<f64> {
    let mut x = vec![0.0f64; k];
    for i in 0..k {
        x[i] = (i as f64 + 1.37).sin() + 0.5 * (i as f64 * 1.91 + 0.7).cos();
    }
    for q in orth {
        let mut dot = 0.0;
        for i in 0..k {
            dot += x[i] * q[i];
        }
        for i in 0..k {
            x[i] -= dot * q[i];
        }
    }
    if !normalize_vec(&mut x) && k > 0 {
        x.fill(0.0);
        x[0] = 1.0;
    }

    let mut y = vec![0.0f64; k];

    for _ in 0..48 {
        for i in 0..k {
            let row = &cov[i * k..(i + 1) * k];
            let mut s = 0.0;
            for j in 0..k {
                s += row[j] * x[j];
            }
            y[i] = s;
        }

        for q in orth {
            let mut dot = 0.0;
            for i in 0..k {
                dot += y[i] * q[i];
            }
            for i in 0..k {
                y[i] -= dot * q[i];
            }
        }

        let norm = y.iter().map(|v| v * v).sum::<f64>().sqrt();
        if norm <= 1e-12 {
            break;
        }
        for i in 0..k {
            x[i] = y[i] / norm;
        }
    }

    if !normalize_vec(&mut x) && k > 0 {
        x.fill(0.0);
        x[0] = 1.0;
    }
    x
}

fn rayleigh(cov: &[f64], k: usize, x: &[f64]) -> f64 {
    let mut y = vec![0.0f64; k];
    for i in 0..k {
        let row = &cov[i * k..(i + 1) * k];
        let mut s = 0.0;
        for j in 0..k {
            s += row[j] * x[j];
        }
        y[i] = s;
    }
    let mut res = 0.0;
    for i in 0..k {
        res += x[i] * y[i];
    }
    res
}

fn hde_layout_from_current(
    graph: &Graph,
    verts: &[usize],
    first_pivot: usize,
    first_far: usize,
    edge_count: usize,
    coords: &mut [(f64, f64)],
    bfs_work: &mut BfsWork,
) -> bool {
    let n = verts.len();
    if n == 0 {
        return false;
    }

    let maxk = choose_pivot_count(n).min(n);
    let mut features = vec![0.0f32; maxk * n];
    let mut min_d = vec![i32::MAX; n];
    let mut pivots: Vec<usize> = Vec::with_capacity(maxk);

    // column 0 already available in bfs_work.dist (distances from first_pivot)
    for (i, &v) in verts.iter().enumerate() {
        let d = bfs_work.dist[v];
        features[i] = d as f32;
        min_d[i] = d;
    }
    pivots.push(first_pivot);

    let mut k_actual = 1usize;
    let mut next_pivot = first_far;

    while k_actual < maxk {
        if pivots.iter().any(|&p| p == next_pivot) {
            if let Some((v, _)) = choose_next_pivot(verts, &min_d, &pivots) {
                next_pivot = v;
            } else {
                break;
            }
        }

        let _ = bfs(graph, bfs_work, next_pivot, false);

        let col_off = k_actual * n;
        for (i, &v) in verts.iter().enumerate() {
            let d = bfs_work.dist[v];
            features[col_off + i] = d as f32;
            if d < min_d[i] {
                min_d[i] = d;
            }
        }

        pivots.push(next_pivot);
        k_actual += 1;

        if let Some((v, best_d)) = choose_next_pivot(verts, &min_d, &pivots) {
            if best_d <= 1 && k_actual >= 2 {
                break;
            }
            next_pivot = v;
        } else {
            break;
        }
    }

    let k = k_actual;
    let mut effective_dims = 0usize;

    for j in 0..k {
        let col = &mut features[j * n..(j + 1) * n];
        let mean = col.iter().map(|&x| x as f64).sum::<f64>() / n as f64;
        let mut var = 0.0;
        for &x in col.iter() {
            let d = x as f64 - mean;
            var += d * d;
        }
        let std = (var / n as f64).sqrt();
        if std > 1e-9 {
            effective_dims += 1;
            for x in col.iter_mut() {
                *x = (((*x as f64) - mean) / std) as f32;
            }
        } else {
            for x in col.iter_mut() {
                *x = 0.0;
            }
        }
    }

    if effective_dims == 0 {
        return false;
    }

    let mut cov = vec![0.0f64; k * k];
    for a in 0..k {
        let ca = &features[a * n..(a + 1) * n];
        for b in 0..=a {
            let cb = &features[b * n..(b + 1) * n];
            let mut s = 0.0;
            for i in 0..n {
                s += (ca[i] as f64) * (cb[i] as f64);
            }
            s /= n as f64;
            cov[a * k + b] = s;
            cov[b * k + a] = s;
        }
    }

    let orth_empty: [&[f64]; 0] = [];
    let v1 = dominant_eigen(&cov, k, &orth_empty);
    let lambda1 = rayleigh(&cov, k, &v1).abs();

    let (v2, lambda2) = if k >= 2 {
        let orth1 = [&v1[..]];
        let v2 = dominant_eigen(&cov, k, &orth1);
        let lambda2 = rayleigh(&cov, k, &v2).abs();
        (v2, lambda2)
    } else {
        (vec![0.0f64; k], 0.0)
    };

    for (i, &v) in verts.iter().enumerate() {
        let mut x = 0.0;
        let mut y = 0.0;
        for j in 0..k {
            let val = features[j * n + i] as f64;
            x += val * v1[j];
            y += val * v2[j];
        }
        coords[v] = (x, y);
    }

    if lambda1 <= 1e-12 {
        return false;
    }

    // If essentially 1D, add tiny deterministic jitter.
    if lambda2 <= 1e-9 {
        let amp = 0.05;
        for (i, &v) in verts.iter().enumerate() {
            let t = pseudo01((v as u64) ^ ((i as u64) << 32)) - 0.5;
            coords[v].1 += amp * t;
        }
    }

    let bb = bbox_of_vertices(coords, verts);
    let max_dim = bb.width().max(bb.height());
    let min_dim = bb.width().min(bb.height());
    if max_dim < 1e-9 {
        return false;
    }

    // Dense graph but HDE collapsed to almost a line -> shell fallback.
    if edge_count >= 3 * n && min_dim < 0.03 * max_dim {
        return false;
    }

    true
}

fn collect_local_edges(
    graph: &Graph,
    verts: &[usize],
    local_index: &mut [usize],
    edge_hint: usize,
) -> Vec<(usize, usize)> {
    for (i, &v) in verts.iter().enumerate() {
        local_index[v] = i;
    }

    let mut edges = Vec::with_capacity(edge_hint.min(1_000_000));
    for (i, &v) in verts.iter().enumerate() {
        for ei in graph.offsets[v]..graph.offsets[v + 1] {
            let to = graph.adj[ei] as usize;
            let j = local_index[to];
            if i < j {
                edges.push((i, j));
            }
        }
    }

    if verts.len() <= 256 || edges.len() <= 300_000 {
        edges.sort_unstable();
        edges.dedup();
    }

    for &v in verts {
        local_index[v] = INVALID;
    }

    edges
}

fn desired_fr_iters(n: usize, m: usize) -> usize {
    let base = if n <= 20 {
        260
    } else if n <= 80 {
        180
    } else if n <= 250 {
        120
    } else if n <= 800 {
        80
    } else {
        50
    };

    let avg_deg = 2.0 * (m as f64) / (n as f64);
    let factor = if avg_deg > 50.0 {
        0.4
    } else if avg_deg > 20.0 {
        0.6
    } else if avg_deg > 8.0 {
        0.8
    } else {
        1.0
    };

    (((base as f64) * factor).round()).max(20.0) as usize
}

fn force_directed_refine(
    verts: &[usize],
    edges: &[(usize, usize)],
    coords: &mut [(f64, f64)],
    iterations: usize,
) {
    let n = verts.len();
    if n <= 1 || iterations == 0 {
        return;
    }

    let mut xs = vec![0.0f64; n];
    let mut ys = vec![0.0f64; n];
    for (i, &v) in verts.iter().enumerate() {
        xs[i] = coords[v].0;
        ys[i] = coords[v].1;
    }

    let mut bb = BBox::empty();
    for i in 0..n {
        bb.update(xs[i], ys[i]);
    }
    let base = bb.width().max(bb.height());

    if base < 1e-9 {
        for i in 0..n {
            let ang = 2.0 * PI * (i as f64) / (n as f64);
            xs[i] = ang.cos();
            ys[i] = ang.sin();
        }
    } else {
        let cx = 0.5 * (bb.minx + bb.maxx);
        let cy = 0.5 * (bb.miny + bb.maxy);
        let target_extent = (n as f64).sqrt() * 2.5 + 1.0;
        let scale = target_extent / base;
        for i in 0..n {
            xs[i] = (xs[i] - cx) * scale;
            ys[i] = (ys[i] - cy) * scale;
        }
    }

    let mut disp_x = vec![0.0f64; n];
    let mut disp_y = vec![0.0f64; n];

    let avg_deg = (2.0 * edges.len() as f64 / n as f64).max(1.0);
    let attract_scale = (8.0 / avg_deg).clamp(0.05, 1.0);
    let gravity = 0.01;
    let k = 2.0;

    let mut temp = (n as f64).sqrt().max(1.0);
    let min_temp = 0.02;
    let cooling = (min_temp / temp).powf(1.0 / iterations as f64);

    for it in 0..iterations {
        disp_x.fill(0.0);
        disp_y.fill(0.0);

        // Repulsion
        for i in 0..n {
            for j in (i + 1)..n {
                let mut dx = xs[i] - xs[j];
                let mut dy = ys[i] - ys[j];
                let mut d2 = dx * dx + dy * dy;
                if d2 < 1e-12 {
                    let ang = pseudo_angle(((i as u64) << 32) ^ (j as u64));
                    dx = 1e-6 * ang.cos();
                    dy = 1e-6 * ang.sin();
                    d2 = dx * dx + dy * dy;
                }
                let d = d2.sqrt();
                let f = (k * k) / d;
                let fx = dx / d * f;
                let fy = dy / d * f;
                disp_x[i] += fx;
                disp_y[i] += fy;
                disp_x[j] -= fx;
                disp_y[j] -= fy;
            }
        }

        // Attraction
        for &(u, v) in edges {
            let mut dx = xs[u] - xs[v];
            let mut dy = ys[u] - ys[v];
            let mut d2 = dx * dx + dy * dy;
            if d2 < 1e-12 {
                let ang = pseudo_angle(((u as u64) << 32) ^ (v as u64));
                dx = 1e-6 * ang.cos();
                dy = 1e-6 * ang.sin();
                d2 = dx * dx + dy * dy;
            }
            let d = d2.sqrt();
            let f = attract_scale * (d * d / k);
            let fx = dx / d * f;
            let fy = dy / d * f;
            disp_x[u] -= fx;
            disp_y[u] -= fy;
            disp_x[v] += fx;
            disp_y[v] += fy;
        }

        // Mild gravity
        for i in 0..n {
            disp_x[i] -= gravity * xs[i];
            disp_y[i] -= gravity * ys[i];
        }

        let mut max_move = 0.0;
        for i in 0..n {
            let d = (disp_x[i] * disp_x[i] + disp_y[i] * disp_y[i]).sqrt();
            if d > 1e-12 {
                let step = temp.min(d);
                xs[i] += disp_x[i] / d * step;
                ys[i] += disp_y[i] / d * step;
                if step > max_move {
                    max_move = step;
                }
            }
        }

        if it % 10 == 9 {
            let cx = xs.iter().sum::<f64>() / n as f64;
            let cy = ys.iter().sum::<f64>() / n as f64;
            for i in 0..n {
                xs[i] -= cx;
                ys[i] -= cy;
            }
        }

        temp *= cooling;
        if max_move < 1e-3 {
            break;
        }
    }

    for (i, &v) in verts.iter().enumerate() {
        coords[v] = (xs[i], ys[i]);
    }
}

fn layout_component(
    graph: &Graph,
    verts: &[usize],
    edge_count: usize,
    max_degree_vertex: usize,
    coords: &mut [(f64, f64)],
    local_index: &mut [usize],
    bfs_work: &mut BfsWork,
    fr_budget: &mut u64,
) -> BBox {
    let n = verts.len();

    if n == 0 {
        return BBox::empty();
    }
    if n == 1 {
        coords[verts[0]] = (0.0, 0.0);
        return bbox_of_vertices(coords, verts);
    }
    if n == 2 {
        coords[verts[0]] = (-1.0, 0.0);
        coords[verts[1]] = (1.0, 0.0);
        return normalize_component(verts, coords);
    }

    // Exact tree
    if edge_count + 1 == n {
        tree_layout(graph, verts, bfs_work, local_index, coords);
        return normalize_component(verts, coords);
    }

    // Exact simple cycle
    if edge_count == n {
        let mut all_deg2 = true;
        for &v in verts {
            if graph.degree(v) != 2 {
                all_deg2 = false;
                break;
            }
        }
        if all_deg2 && cycle_layout(graph, verts, coords) {
            return normalize_component(verts, coords);
        }
    }

    let extra = edge_count + 1 - n;
    let near_tree = edge_count <= 2 * n && (extra <= 12 || extra <= n / 10);

    if near_tree {
        spanning_tree_radial_layout(
            graph,
            verts,
            max_degree_vertex,
            bfs_work,
            local_index,
            coords,
        );
    } else if edge_count >= 6 * n {
        shell_layout(
            graph,
            verts,
            max_degree_vertex,
            local_index,
            bfs_work,
            coords,
        );
    } else {
        let seed = max_degree_vertex;
        let (p0, _) = bfs(graph, bfs_work, seed, false);
        let (p1, diam_est) = bfs(graph, bfs_work, p0, false);

        if diam_est <= 3 && edge_count >= 2 * n {
            shell_layout(graph, verts, seed, local_index, bfs_work, coords);
        } else {
            let ok = hde_layout_from_current(graph, verts, p0, p1, edge_count, coords, bfs_work);
            if !ok {
                shell_layout(graph, verts, seed, local_index, bfs_work, coords);
            }
        }
    }

    // Small/medium exact FR polish, under a global quadratic-work budget.
    if n <= FR_MAX_N {
        let desired = desired_fr_iters(n, edge_count);
        let pairs = (n as u64) * ((n as u64) - 1) / 2;
        if pairs > 0 {
            let possible = (*fr_budget / pairs) as usize;
            let iters = desired.min(possible);
            if iters >= 20 {
                let edges = collect_local_edges(graph, verts, local_index, edge_count);
                if !edges.is_empty() {
                    force_directed_refine(verts, &edges, coords, iters);
                    *fr_budget -= pairs * (iters as u64);
                }
            }
        }
    }

    normalize_component(verts, coords)
}

fn find_components(graph: &Graph) -> (Vec<ComponentInfo>, Vec<usize>) {
    let n = graph.n;
    let mut seen = vec![0u8; n];
    let mut queue = Vec::<usize>::new();
    let mut comps = Vec::<ComponentInfo>::new();
    let mut all_vertices = Vec::<usize>::with_capacity(n);

    for s in 0..n {
        if seen[s] != 0 {
            continue;
        }

        let start = all_vertices.len();
        queue.clear();
        queue.push(s);
        seen[s] = 1;

        let mut head = 0usize;
        let mut edge_sum = 0usize;
        let mut max_v = s;
        let mut max_deg = graph.degree(s);

        while head < queue.len() {
            let v = queue[head];
            head += 1;
            all_vertices.push(v);

            let deg = graph.degree(v);
            edge_sum += deg;
            if deg > max_deg {
                max_deg = deg;
                max_v = v;
            }

            for ei in graph.offsets[v]..graph.offsets[v + 1] {
                let to = graph.adj[ei] as usize;
                if seen[to] == 0 {
                    seen[to] = 1;
                    queue.push(to);
                }
            }
        }

        let end = all_vertices.len();
        comps.push(ComponentInfo {
            start,
            end,
            edge_count: edge_sum / 2,
            max_degree_vertex: max_v,
        });
    }

    (comps, all_vertices)
}

fn pack_components(
    components: &[ComponentInfo],
    all_vertices: &[usize],
    coords: &mut [(f64, f64)],
    bboxes: &[BBox],
) {
    let c = components.len();
    if c == 0 {
        return;
    }

    let mut sizes = vec![(0.0f64, 0.0f64); c];
    let mut total_area = 0.0;
    for i in 0..c {
        let w = bboxes[i].width().max(1.0);
        let h = bboxes[i].height().max(1.0);
        sizes[i] = (w, h);
        total_area += (w + PACK_GAP) * (h + PACK_GAP);
    }

    let mut order: Vec<usize> = (0..c).collect();
    order.sort_by(|&a, &b| {
        let aa = sizes[a].0 * sizes[a].1;
        let bb = sizes[b].0 * sizes[b].1;
        bb.total_cmp(&aa)
    });

    let row_limit = total_area.sqrt() * 1.2 + PACK_GAP;
    let mut shifts = vec![(0.0f64, 0.0f64); c];

    let mut x = 0.0;
    let mut y = 0.0;
    let mut row_h = 0.0;

    for idx in order {
        let (w, h) = sizes[idx];
        if x > 0.0 && x + w > row_limit {
            x = 0.0;
            y += row_h + PACK_GAP;
            row_h = 0.0;
        }

        let actual_w = bboxes[idx].width();
        let actual_h = bboxes[idx].height();

        let shift_x = x - bboxes[idx].minx + 0.5 * (w - actual_w);
        let shift_y = y - bboxes[idx].miny + 0.5 * (h - actual_h);
        shifts[idx] = (shift_x, shift_y);

        x += w + PACK_GAP;
        if h > row_h {
            row_h = h;
        }
    }

    for (i, comp) in components.iter().enumerate() {
        let (sx, sy) = shifts[i];
        for &v in &all_vertices[comp.start..comp.end] {
            coords[v].0 += sx;
            coords[v].1 += sy;
        }
    }

    let mut global_bb = BBox::empty();
    for &(x, y) in coords.iter() {
        global_bb.update(x, y);
    }
    if !global_bb.minx.is_infinite() {
        let cx = 0.5 * (global_bb.minx + global_bb.maxx);
        let cy = 0.5 * (global_bb.miny + global_bb.maxy);
        for p in coords.iter_mut() {
            p.0 -= cx;
            p.1 -= cy;
        }
    }
}

fn main() {
    let mut input = FastInput::new();
    let n = input.next_usize();
    let m = input.next_usize();

    if n == 0 {
        return;
    }

    let mut degrees = vec![0usize; n];
    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u = input.next_usize();
        let v = input.next_usize();
        if u >= n || v >= n {
            continue;
        }
        if u == v {
            // Ignore self-loops for layout; node still exists/component still handled.
            continue;
        }
        edges.push((u as u32, v as u32));
        degrees[u] += 1;
        degrees[v] += 1;
    }

    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + degrees[i];
    }

    let mut cur = offsets.clone();
    let mut adj = vec![0u32; offsets[n]];
    for &(u32u, u32v) in &edges {
        let u = u32u as usize;
        let v = u32v as usize;
        adj[cur[u]] = u32v;
        cur[u] += 1;
        adj[cur[v]] = u32u;
        cur[v] += 1;
    }
    drop(edges);

    let graph = Graph { n, offsets, adj };

    let (components, all_vertices) = find_components(&graph);

    let mut coords = vec![(0.0f64, 0.0f64); n];
    let mut bboxes = vec![BBox::empty(); components.len()];
    let mut bfs_work = BfsWork::new(n);
    let mut local_index = vec![INVALID; n];

    let mut process_order: Vec<usize> = (0..components.len()).collect();
    process_order.sort_by_key(|&i| Reverse(components[i].len()));

    let mut fr_budget = FR_BUDGET_INIT;

    for idx in process_order {
        let comp = &components[idx];
        let verts = &all_vertices[comp.start..comp.end];
        let bb = layout_component(
            &graph,
            verts,
            comp.edge_count,
            comp.max_degree_vertex,
            &mut coords,
            &mut local_index,
            &mut bfs_work,
            &mut fr_budget,
        );
        bboxes[idx] = bb;
    }

    pack_components(&components, &all_vertices, &mut coords, &bboxes);

    let mut out = io::BufWriter::with_capacity(1 << 20, io::stdout().lock());
    for &(x, y) in &coords {
        writeln!(out, "{:.6} {:.6}", x, y).unwrap();
    }
}
