#include <algorithm>
#include <array>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using i64 = long long;
using Clock = std::chrono::steady_clock;

static constexpr i64 INF = std::numeric_limits<i64>::max() / 4;

struct Task {
    int job = -1;
    int index = -1;
    int machine = -1;
    int global = -1;

    i64 duration = 0;

    int jobPrev = -1;
    int jobNext = -1;

    i64 remWork = 0;               // remaining work in job from this task onward
    int remOps = 0;                // remaining ops in job from this task onward
    i64 jobTotalWork = 0;          // total job work
    i64 nextMachineLoad = 0;       // load of next machine in this job
    i64 futureBottleneckLoad = 0;  // max machine load over remaining tasks in job
};

struct Instance {
    int J = 0;
    int M = 0;
    int N = 0;

    std::vector<Task> tasks;
    std::vector<std::vector<int>> jobTasks;  // global task ids by job order
    std::vector<i64> machineLoad;
    std::vector<int> machineTaskCount;
};

struct Schedule {
    std::vector<i64> start;                    // start time by global task id
    std::vector<std::vector<int>> machineOrder; // task order on each machine
    i64 makespan = INF;
    bool feasible = false;
};

struct SequenceEval {
    std::vector<i64> start;
    std::vector<int> machPrev;
    std::vector<int> machNext;
    i64 makespan = INF;
    bool feasible = false;
};

enum class RuleType {
    FIFO,
    MWKR,
    MOR,
    SPT,
    LPT,
    RW_SPT,
    NEXT_MACHINE,
    BOTTLENECK,
    TOTAL_JOB
};

struct Rule {
    RuleType type = RuleType::MWKR;
    bool randomized = false;
    std::uint64_t seed = 0;
};

static std::uint64_t splitmix64(std::uint64_t x) noexcept {
    x += 0x9E3779B97F4A7C15ULL;
    x = (x ^ (x >> 30)) * 0xBF58476D1CE4E5B9ULL;
    x = (x ^ (x >> 27)) * 0x94D049BB133111EBULL;
    return x ^ (x >> 31);
}

static void preprocess_instance(Instance& inst, const std::vector<i64>& totalJobWork) {
    inst.N = static_cast<int>(inst.tasks.size());

    for (int j = 0; j < inst.J; ++j) {
        const int T = static_cast<int>(inst.jobTasks[j].size());
        i64 rem = 0;
        i64 bottleneck = 0;

        for (int k = T - 1; k >= 0; --k) {
            const int tid = inst.jobTasks[j][k];
            Task& t = inst.tasks[tid];

            rem += t.duration;
            bottleneck = std::max(bottleneck, inst.machineLoad[t.machine]);

            t.remWork = rem;
            t.remOps = T - k;
            t.jobTotalWork = totalJobWork[j];
            t.futureBottleneckLoad = bottleneck;
            t.jobPrev = (k > 0 ? inst.jobTasks[j][k - 1] : -1);
            t.jobNext = (k + 1 < T ? inst.jobTasks[j][k + 1] : -1);
        }

        for (int k = 0; k < T; ++k) {
            const int tid = inst.jobTasks[j][k];
            const int nxt = inst.tasks[tid].jobNext;
            inst.tasks[tid].nextMachineLoad = (nxt != -1 ? inst.machineLoad[inst.tasks[nxt].machine] : 0);
        }
    }
}

static std::vector<Rule> choose_rules(const Instance& inst) {
    std::vector<Rule> rules;
    auto add = [&](RuleType t, bool rnd = false, std::uint64_t seed = 0) {
        rules.push_back(Rule{t, rnd, seed});
    };

    constexpr std::uint64_t S1 = 0x123456789ABCDEF0ULL;
    constexpr std::uint64_t S2 = 0x9E3779B97F4A7C15ULL;
    constexpr std::uint64_t S3 = 0xD1B54A32D192ED03ULL;

    if (inst.N <= 120) {
        add(RuleType::MWKR);
        add(RuleType::MOR);
        add(RuleType::SPT);
        add(RuleType::LPT);
        add(RuleType::RW_SPT);
        add(RuleType::NEXT_MACHINE);
        add(RuleType::BOTTLENECK);
        add(RuleType::TOTAL_JOB);
        add(RuleType::FIFO);
        add(RuleType::MWKR, true, S1);
        add(RuleType::SPT, true, S2);
        add(RuleType::BOTTLENECK, true, S3);
    } else if (inst.N <= 2000) {
        add(RuleType::MWKR);
        add(RuleType::MOR);
        add(RuleType::SPT);
        add(RuleType::RW_SPT);
        add(RuleType::NEXT_MACHINE);
        add(RuleType::BOTTLENECK);
        add(RuleType::TOTAL_JOB);
        add(RuleType::FIFO);
        add(RuleType::MWKR, true, S1);
    } else if (inst.N <= 15000) {
        add(RuleType::MWKR);
        add(RuleType::SPT);
        add(RuleType::RW_SPT);
        add(RuleType::BOTTLENECK);
        add(RuleType::TOTAL_JOB);
    } else {
        add(RuleType::MWKR);
        add(RuleType::RW_SPT);
        add(RuleType::BOTTLENECK);
    }

    return rules;
}

struct ReadyItem {
    std::array<i64, 5> key{};
    std::uint64_t tie = 0;
    int taskId = -1;

    bool operator<(const ReadyItem& other) const noexcept {
        if (key != other.key) return key < other.key;
        if (tie != other.tie) return tie < other.tie;
        return taskId < other.taskId;
    }
};

struct RunEvent {
    i64 time = 0;
    int machine = -1;
    int taskId = -1;
};

struct EventGreater {
    bool operator()(const RunEvent& a, const RunEvent& b) const noexcept {
        if (a.time != b.time) return a.time > b.time;
        if (a.machine != b.machine) return a.machine > b.machine;
        return a.taskId > b.taskId;
    }
};

static ReadyItem make_ready_item(const Instance& inst, int taskId, const Rule& rule, std::uint64_t arrivalSerial) {
    const Task& t = inst.tasks[taskId];
    const i64 zeroFirst = (t.duration == 0 ? 1LL : 0LL);

    ReadyItem item;
    switch (rule.type) {
    case RuleType::FIFO:
        item.key = {zeroFirst, 0, 0, 0, 0};
        break;
    case RuleType::MWKR:
        item.key = {zeroFirst, t.remWork, static_cast<i64>(t.remOps), -t.duration, 0};
        break;
    case RuleType::MOR:
        item.key = {zeroFirst, static_cast<i64>(t.remOps), t.remWork, -t.duration, 0};
        break;
    case RuleType::SPT:
        item.key = {zeroFirst, -t.duration, t.remWork, static_cast<i64>(t.remOps), 0};
        break;
    case RuleType::LPT:
        item.key = {zeroFirst, t.duration, t.remWork, static_cast<i64>(t.remOps), 0};
        break;
    case RuleType::RW_SPT:
        item.key = {zeroFirst, t.remWork, -t.duration, static_cast<i64>(t.remOps), 0};
        break;
    case RuleType::NEXT_MACHINE:
        item.key = {zeroFirst, t.nextMachineLoad, t.remWork, -t.duration, 0};
        break;
    case RuleType::BOTTLENECK:
        item.key = {zeroFirst, t.futureBottleneckLoad, t.remWork, -t.duration, 0};
        break;
    case RuleType::TOTAL_JOB:
        item.key = {zeroFirst, t.jobTotalWork, t.remWork, -t.duration, 0};
        break;
    }

    if (rule.randomized) {
        item.tie = splitmix64(rule.seed ^ (static_cast<std::uint64_t>(taskId) * 0x9E3779B97F4A7C15ULL) ^ arrivalSerial);
    } else {
        item.tie = std::numeric_limits<std::uint64_t>::max() - arrivalSerial;
    }

    item.taskId = taskId;
    return item;
}

static Schedule dispatch_schedule(const Instance& inst, const Rule& rule) {
    Schedule out;
    out.start.assign(inst.N, -1);
    out.machineOrder.assign(inst.M, {});
    for (int m = 0; m < inst.M; ++m) {
        out.machineOrder[m].reserve(inst.machineTaskCount[m]);
    }

    std::vector<std::priority_queue<ReadyItem>> ready(inst.M);
    std::vector<unsigned char> busy(inst.M, 0);
    std::priority_queue<RunEvent, std::vector<RunEvent>, EventGreater> events;

    std::uint64_t arrivalSerial = 0;
    auto enqueue_task = [&](int tid) {
        const int m = inst.tasks[tid].machine;
        ready[m].push(make_ready_item(inst, tid, rule, arrivalSerial++));
    };

    for (int j = 0; j < inst.J; ++j) {
        if (!inst.jobTasks[j].empty()) {
            enqueue_task(inst.jobTasks[j][0]);
        }
    }

    i64 current = 0;
    int done = 0;

    while (done < inst.N) {
        // Exhaust zero-duration closure at current time.
        while (true) {
            bool progressed = false;
            for (int m = 0; m < inst.M; ++m) {
                if (busy[m] || ready[m].empty()) continue;

                const int tid = ready[m].top().taskId;
                if (inst.tasks[tid].duration != 0) continue;

                ready[m].pop();
                out.start[tid] = current;
                out.machineOrder[m].push_back(tid);
                ++done;
                progressed = true;

                const int nxt = inst.tasks[tid].jobNext;
                if (nxt != -1) enqueue_task(nxt);
            }
            if (!progressed) break;
        }

        if (done >= inst.N) break;

        // Start all positive-duration tasks that can start now.
        for (int m = 0; m < inst.M; ++m) {
            if (busy[m] || ready[m].empty()) continue;

            const int tid = ready[m].top().taskId;
            ready[m].pop();

            out.start[tid] = current;
            out.machineOrder[m].push_back(tid);
            busy[m] = 1;
            events.push(RunEvent{current + inst.tasks[tid].duration, m, tid});
        }

        if (events.empty()) {
            // Should not happen for a valid instance unless all tasks are already done.
            break;
        }

        current = events.top().time;
        while (!events.empty() && events.top().time == current) {
            RunEvent ev = events.top();
            events.pop();

            busy[ev.machine] = 0;
            ++done;

            const int nxt = inst.tasks[ev.taskId].jobNext;
            if (nxt != -1) enqueue_task(nxt);
        }
    }

    if (done == inst.N) {
        i64 ms = 0;
        for (int tid = 0; tid < inst.N; ++tid) {
            if (out.start[tid] < 0) {
                out.feasible = false;
                out.makespan = INF;
                return out;
            }
            ms = std::max(ms, out.start[tid] + inst.tasks[tid].duration);
        }
        out.feasible = true;
        out.makespan = ms;
    }

    return out;
}

static Schedule serial_fallback_schedule(const Instance& inst) {
    Schedule out;
    out.start.assign(inst.N, -1);
    out.machineOrder.assign(inst.M, {});
    for (int m = 0; m < inst.M; ++m) {
        out.machineOrder[m].reserve(inst.machineTaskCount[m]);
    }

    std::vector<i64> jobEnd(inst.J, 0);
    std::vector<i64> machEnd(inst.M, 0);
    i64 ms = 0;

    for (int j = 0; j < inst.J; ++j) {
        for (int tid : inst.jobTasks[j]) {
            const Task& t = inst.tasks[tid];
            const i64 s = std::max(jobEnd[j], machEnd[t.machine]);
            const i64 e = s + t.duration;
            out.start[tid] = s;
            jobEnd[j] = e;
            machEnd[t.machine] = e;
            out.machineOrder[t.machine].push_back(tid);
            ms = std::max(ms, e);
        }
    }

    out.feasible = true;
    out.makespan = ms;
    return out;
}

static SequenceEval evaluate_sequences(const Instance& inst, const std::vector<std::vector<int>>& machineOrder) {
    SequenceEval ev;
    ev.start.assign(inst.N, 0);
    ev.machPrev.assign(inst.N, -1);
    ev.machNext.assign(inst.N, -1);

    for (int m = 0; m < inst.M; ++m) {
        const auto& seq = machineOrder[m];
        for (int i = 0; i < static_cast<int>(seq.size()); ++i) {
            if (i > 0) ev.machPrev[seq[i]] = seq[i - 1];
            if (i + 1 < static_cast<int>(seq.size())) ev.machNext[seq[i]] = seq[i + 1];
        }
    }

    std::vector<int> indeg(inst.N, 0);
    std::vector<i64> est(inst.N, 0);
    std::vector<int> q;
    q.reserve(inst.N);

    for (int tid = 0; tid < inst.N; ++tid) {
        indeg[tid] = (inst.tasks[tid].jobPrev != -1 ? 1 : 0) + (ev.machPrev[tid] != -1 ? 1 : 0);
        if (indeg[tid] == 0) q.push_back(tid);
    }

    int processed = 0;
    i64 ms = 0;

    for (std::size_t qi = 0; qi < q.size(); ++qi) {
        const int u = q[qi];
        ev.start[u] = est[u];
        const i64 finish = ev.start[u] + inst.tasks[u].duration;
        ms = std::max(ms, finish);
        ++processed;

        const int jn = inst.tasks[u].jobNext;
        if (jn != -1) {
            est[jn] = std::max(est[jn], finish);
            if (--indeg[jn] == 0) q.push_back(jn);
        }

        const int mn = ev.machNext[u];
        if (mn != -1) {
            est[mn] = std::max(est[mn], finish);
            if (--indeg[mn] == 0) q.push_back(mn);
        }
    }

    ev.feasible = (processed == inst.N);
    ev.makespan = ev.feasible ? ms : INF;
    return ev;
}

static std::vector<int> critical_path(const Instance& inst, const SequenceEval& ev) {
    std::vector<int> path;
    if (!ev.feasible || inst.N == 0) return path;

    std::vector<i64> finish(inst.N, 0);
    for (int tid = 0; tid < inst.N; ++tid) {
        finish[tid] = ev.start[tid] + inst.tasks[tid].duration;
    }

    auto machine_critical = [&](int tid) -> bool {
        if (tid < 0) return false;
        const int mp = ev.machPrev[tid];
        return (mp != -1 && finish[mp] == ev.start[tid]);
    };

    int endTask = -1;
    i64 best = -1;
    for (int tid = 0; tid < inst.N; ++tid) {
        if (finish[tid] > best) {
            best = finish[tid];
            endTask = tid;
        } else if (finish[tid] == best) {
            if (machine_critical(tid) && !machine_critical(endTask)) {
                endTask = tid;
            }
        }
    }

    int cur = endTask;
    while (cur != -1) {
        path.push_back(cur);

        const int mp = ev.machPrev[cur];
        const int jp = inst.tasks[cur].jobPrev;
        int pred = -1;

        if (mp != -1 && finish[mp] == ev.start[cur]) {
            pred = mp; // prefer machine predecessor to expose swap candidates
        } else if (jp != -1 && finish[jp] == ev.start[cur]) {
            pred = jp;
        }

        cur = pred;
    }

    std::reverse(path.begin(), path.end());
    return path;
}

static Schedule local_search_critical_swaps(
    const Instance& inst,
    const Schedule& initial,
    int maxIter,
    Clock::time_point deadline
) {
    if (!initial.feasible || initial.machineOrder.empty()) return initial;

    Schedule current = initial;
    SequenceEval curEval = evaluate_sequences(inst, current.machineOrder);
    if (!curEval.feasible) return initial;

    current.start = curEval.start;
    current.makespan = curEval.makespan;
    current.feasible = true;

    for (int iter = 0; iter < maxIter && Clock::now() < deadline; ++iter) {
        const std::vector<int> path = critical_path(inst, curEval);
        if (path.size() < 2) break;

        std::vector<int> pos(inst.N, -1);
        for (int m = 0; m < inst.M; ++m) {
            const auto& seq = current.machineOrder[m];
            for (int i = 0; i < static_cast<int>(seq.size()); ++i) {
                pos[seq[i]] = i;
            }
        }

        std::vector<std::pair<int, int>> candidates; // (machine, leftPos)
        for (std::size_t i = 0; i < path.size();) {
            const int firstTask = path[i];
            const int machine = inst.tasks[firstTask].machine;
            std::size_t j = i;

            while (j + 1 < path.size()) {
                const int a = path[j];
                const int b = path[j + 1];
                if (inst.tasks[b].machine != machine) break;
                if (pos[a] == -1 || pos[b] != pos[a] + 1) break;
                ++j;
            }

            if (j > i) {
                const int p1 = pos[path[i]];
                if (p1 >= 0) candidates.emplace_back(machine, p1);

                const int p2 = pos[path[j - 1]];
                if (p2 >= 0) candidates.emplace_back(machine, p2);
            }

            i = j + 1;
        }

        if (candidates.empty()) break;

        std::sort(candidates.begin(), candidates.end());
        candidates.erase(std::unique(candidates.begin(), candidates.end()), candidates.end());

        bool improved = false;
        i64 bestMs = current.makespan;
        std::pair<int, int> bestSwap{-1, -1};

        for (const auto& cand : candidates) {
            if (Clock::now() >= deadline) break;

            const int m = cand.first;
            const int p = cand.second;

            if (p < 0 || p + 1 >= static_cast<int>(current.machineOrder[m].size())) continue;

            std::swap(current.machineOrder[m][p], current.machineOrder[m][p + 1]);
            SequenceEval trial = evaluate_sequences(inst, current.machineOrder);
            std::swap(current.machineOrder[m][p], current.machineOrder[m][p + 1]);

            if (trial.feasible && trial.makespan < bestMs) {
                bestMs = trial.makespan;
                bestSwap = cand;
                improved = true;
            }
        }

        if (!improved) break;

        std::swap(current.machineOrder[bestSwap.first][bestSwap.second],
                  current.machineOrder[bestSwap.first][bestSwap.second + 1]);

        curEval = evaluate_sequences(inst, current.machineOrder);
        if (!curEval.feasible) break;

        current.start = curEval.start;
        current.makespan = curEval.makespan;
        current.feasible = true;
    }

    return current;
}

class ExactSolver {
public:
    explicit ExactSolver(const Instance& instance) : inst(instance) {}

    Schedule solve(const Schedule& incumbent, std::chrono::milliseconds budget) {
        nextPos.assign(inst.J, 0);
        jobTime.assign(inst.J, 0);
        machineTime.assign(inst.M, 0);
        remMachineWork = inst.machineLoad;

        curStart.assign(inst.N, -1);
        curMachineOrder.assign(inst.M, {});
        for (int m = 0; m < inst.M; ++m) {
            curMachineOrder[m].reserve(inst.machineTaskCount[m]);
        }

        bestMakespan = incumbent.feasible ? incumbent.makespan : INF;
        bestStart = incumbent.start;
        bestMachineOrder = incumbent.machineOrder;
        if (bestMachineOrder.empty()) bestMachineOrder.assign(inst.M, {});

        deadline = Clock::now() + budget;
        stop = false;
        nodes = 0;

        dfs(0, 0);

        Schedule out;
        out.start = bestStart;
        out.machineOrder = bestMachineOrder;
        out.makespan = bestMakespan;
        out.feasible = (bestMakespan < INF / 2);
        return out;
    }

private:
    struct Cand {
        int job = -1;
        int tid = -1;
        int machine = -1;
        i64 start = 0;
        i64 finish = 0;
        i64 remWork = 0;
        i64 bottleneck = 0;
    };

    const Instance& inst;

    std::vector<int> nextPos;
    std::vector<i64> jobTime;
    std::vector<i64> machineTime;
    std::vector<i64> remMachineWork;

    std::vector<i64> curStart;
    std::vector<i64> bestStart;

    std::vector<std::vector<int>> curMachineOrder;
    std::vector<std::vector<int>> bestMachineOrder;

    i64 bestMakespan = INF;

    Clock::time_point deadline{};
    bool stop = false;
    std::uint64_t nodes = 0;

    void dfs(int scheduledCount, i64 currentMax) {
        if (stop) return;
        if ((++nodes & 2047ULL) == 0ULL && Clock::now() >= deadline) {
            stop = true;
            return;
        }

        if (currentMax >= bestMakespan) return;

        if (scheduledCount == inst.N) {
            if (currentMax < bestMakespan) {
                bestMakespan = currentMax;
                bestStart = curStart;
                bestMachineOrder = curMachineOrder;
            }
            return;
        }

        i64 lb = currentMax;

        for (int j = 0; j < inst.J; ++j) {
            if (nextPos[j] < static_cast<int>(inst.jobTasks[j].size())) {
                const int tid = inst.jobTasks[j][nextPos[j]];
                lb = std::max(lb, jobTime[j] + inst.tasks[tid].remWork);
            }
        }

        for (int m = 0; m < inst.M; ++m) {
            lb = std::max(lb, machineTime[m] + remMachineWork[m]);
        }

        if (lb >= bestMakespan) return;

        std::vector<Cand> cands;
        cands.reserve(inst.J);

        for (int j = 0; j < inst.J; ++j) {
            if (nextPos[j] >= static_cast<int>(inst.jobTasks[j].size())) continue;

            const int tid = inst.jobTasks[j][nextPos[j]];
            const Task& t = inst.tasks[tid];
            const i64 s = std::max(jobTime[j], machineTime[t.machine]);
            const i64 f = s + t.duration;

            cands.push_back(Cand{
                j, tid, t.machine, s, f, t.remWork, t.futureBottleneckLoad
            });
        }

        std::sort(cands.begin(), cands.end(), [&](const Cand& a, const Cand& b) {
            if (a.finish != b.finish) return a.finish < b.finish;
            if (a.start != b.start) return a.start < b.start;
            if (a.remWork != b.remWork) return a.remWork > b.remWork;
            if (a.bottleneck != b.bottleneck) return a.bottleneck > b.bottleneck;
            return a.tid < b.tid;
        });

        for (const Cand& c : cands) {
            const int j = c.job;
            const int tid = c.tid;
            const int m = c.machine;

            const int oldNext = nextPos[j];
            const i64 oldJobTime = jobTime[j];
            const i64 oldMachTime = machineTime[m];
            const i64 oldStart = curStart[tid];

            nextPos[j] = oldNext + 1;
            jobTime[j] = c.finish;
            machineTime[m] = c.finish;
            remMachineWork[m] -= inst.tasks[tid].duration;
            curStart[tid] = c.start;
            curMachineOrder[m].push_back(tid);

            dfs(scheduledCount + 1, std::max(currentMax, c.finish));

            curMachineOrder[m].pop_back();
            curStart[tid] = oldStart;
            remMachineWork[m] += inst.tasks[tid].duration;
            machineTime[m] = oldMachTime;
            jobTime[j] = oldJobTime;
            nextPos[j] = oldNext;

            if (stop) return;
        }
    }
};

static bool verify_schedule(const Instance& inst, const Schedule& sched, i64* outMakespan = nullptr) {
    if (!sched.feasible) return false;
    if (static_cast<int>(sched.start.size()) != inst.N) return false;

    i64 ms = 0;
    for (int tid = 0; tid < inst.N; ++tid) {
        if (sched.start[tid] < 0) return false;
        ms = std::max(ms, sched.start[tid] + inst.tasks[tid].duration);
    }

    for (int j = 0; j < inst.J; ++j) {
        const auto& jt = inst.jobTasks[j];
        for (int k = 1; k < static_cast<int>(jt.size()); ++k) {
            const int prev = jt[k - 1];
            const int cur = jt[k];
            if (sched.start[cur] < sched.start[prev] + inst.tasks[prev].duration) {
                return false;
            }
        }
    }

    std::vector<std::vector<int>> byMachine(inst.M);
    for (int m = 0; m < inst.M; ++m) {
        byMachine[m].reserve(inst.machineTaskCount[m]);
    }
    for (int tid = 0; tid < inst.N; ++tid) {
        byMachine[inst.tasks[tid].machine].push_back(tid);
    }

    for (int m = 0; m < inst.M; ++m) {
        auto& vec = byMachine[m];
        std::sort(vec.begin(), vec.end(), [&](int a, int b) {
            if (sched.start[a] != sched.start[b]) return sched.start[a] < sched.start[b];
            const i64 ea = sched.start[a] + inst.tasks[a].duration;
            const i64 eb = sched.start[b] + inst.tasks[b].duration;
            if (ea != eb) return ea < eb; // zero-duration tasks first when same start
            return a < b;
        });

        bool first = true;
        i64 prevEnd = 0;
        for (int tid : vec) {
            const i64 s = sched.start[tid];
            const i64 e = s + inst.tasks[tid].duration;
            if (!first && s < prevEnd) return false;
            prevEnd = std::max(prevEnd, e);
            first = false;
        }
    }

    if (outMakespan) *outMakespan = ms;
    return true;
}

static bool is_better(const Schedule& cand, const Schedule& best) {
    return cand.feasible && (!best.feasible || cand.makespan < best.makespan);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Instance inst;
    if (!(std::cin >> inst.J >> inst.M)) {
        return 0;
    }

    inst.jobTasks.resize(inst.J);
    inst.machineLoad.assign(inst.M, 0);
    inst.machineTaskCount.assign(inst.M, 0);

    std::vector<i64> totalJobWork(inst.J, 0);

    for (int j = 0; j < inst.J; ++j) {
        int T = 0;
        std::cin >> T;
        inst.jobTasks[j].resize(T);

        for (int k = 0; k < T; ++k) {
            int machine = 0;
            i64 duration = 0;
            std::cin >> machine >> duration;

            const int gid = static_cast<int>(inst.tasks.size());
            inst.jobTasks[j][k] = gid;

            Task t;
            t.job = j;
            t.index = k;
            t.machine = machine;
            t.global = gid;
            t.duration = duration;
            inst.tasks.push_back(t);

            if (machine >= 0 && machine < inst.M) {
                inst.machineLoad[machine] += duration;
                inst.machineTaskCount[machine] += 1;
            }
            totalJobWork[j] += duration;
        }
    }

    preprocess_instance(inst, totalJobWork);

    if (inst.N == 0) {
        for (int j = 0; j < inst.J; ++j) {
            std::cout << '\n';
        }
        return 0;
    }

    Schedule best = serial_fallback_schedule(inst);

    const bool exactEligible =
        (inst.N <= 14 && inst.J <= 8 && inst.M <= 8) ||
        (inst.N <= 16 && inst.J <= 4 && inst.M <= 4);

    const bool localSearchEligible = (!exactEligible && inst.N <= 2500);

    std::vector<Schedule> pool;
    if (localSearchEligible) {
        pool.reserve(16);
    }

    const std::vector<Rule> rules = choose_rules(inst);
    for (const Rule& rule : rules) {
        Schedule cand = dispatch_schedule(inst, rule);
        if (is_better(cand, best)) best = cand;
        if (localSearchEligible && cand.feasible) {
            pool.push_back(std::move(cand));
        }
    }

    if (exactEligible) {
        ExactSolver solver(inst);
        Schedule exact = solver.solve(best, std::chrono::milliseconds(1500));
        if (is_better(exact, best)) best = std::move(exact);
        else best = std::move(exact); // exact solver returns incumbent or better
    } else if (localSearchEligible && !pool.empty()) {
        std::sort(pool.begin(), pool.end(), [](const Schedule& a, const Schedule& b) {
            return a.makespan < b.makespan;
        });

        const int topK = std::min<int>(static_cast<int>(pool.size()), (inst.N <= 600 ? 3 : 2));
        const int maxIter = (inst.N <= 200 ? 40 : (inst.N <= 1000 ? 20 : 10));
        const auto deadline = Clock::now() + std::chrono::milliseconds(inst.N <= 500 ? 1200 : 800);

        for (int i = 0; i < topK && Clock::now() < deadline; ++i) {
            Schedule improved = local_search_critical_swaps(inst, pool[i], maxIter, deadline);
            if (is_better(improved, best)) best = std::move(improved);
        }
    }

    i64 verifiedMakespan = INF;
    if (!verify_schedule(inst, best, &verifiedMakespan)) {
        best = serial_fallback_schedule(inst);
        verify_schedule(inst, best, &verifiedMakespan);
    }
    best.makespan = verifiedMakespan;

    for (int j = 0; j < inst.J; ++j) {
        bool first = true;
        for (int tid : inst.jobTasks[j]) {
            if (!first) std::cout << ' ';
            std::cout << best.start[tid] << ' ' << inst.tasks[tid].machine;
            first = false;
        }
        std::cout << '\n';
    }

    return 0;
}
