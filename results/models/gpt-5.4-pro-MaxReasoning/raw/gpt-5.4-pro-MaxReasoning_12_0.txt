Below is a complete C# solver.  
It is conservative but always valid: it packs rotated copies of the polyhedron inside non-overlapping axis-aligned cells defined by the polyhedron’s rotated AABB. That guarantees no overlap and containment.

It adapts by:
- generating orientations differently based on scale/complexity,
- doing deeper guillotine/beam search on small/medium instances,
- falling back to faster greedy search on large/extreme instances.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;
using System.Text;

public class Program
{
    private static readonly CultureInfo Invariant = CultureInfo.InvariantCulture;
    private const double Tol = 1e-9;

    public static void Main()
    {
        FastScanner fs = new FastScanner(Console.OpenStandardInput());

        int vCount;
        int fCount;
        try
        {
            vCount = fs.NextInt();
            fCount = fs.NextInt();
        }
        catch
        {
            return;
        }

        Vec3[] vertices = new Vec3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            double x = fs.NextDouble();
            double y = fs.NextDouble();
            double z = fs.NextDouble();
            vertices[i] = new Vec3(x, y, z);
        }

        RawFace[] faces = new RawFace[fCount];
        for (int i = 0; i < fCount; i++)
        {
            int a = fs.NextInt();
            int b = fs.NextInt();
            int c = fs.NextInt();
            faces[i] = new RawFace(a, b, c);
        }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        Solver solver = new Solver(vertices, faces, W, H, D);
        PlanNode plan = solver.Solve();

        using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20))
        {
            sw.NewLine = "\n";
            sw.WriteLine(plan.Count.ToString(Invariant));
            if (plan.Count > 0)
            {
                plan.Write(sw);
            }
        }
    }

    private static string FormatDouble(double x)
    {
        if (double.IsNaN(x) || double.IsInfinity(x)) x = 0.0;
        if (Math.Abs(x) < 1e-12) x = 0.0;
        return x.ToString("G17", Invariant);
    }

    private static long SafeAdd(long a, long b)
    {
        if (a >= long.MaxValue - b) return long.MaxValue;
        return a + b;
    }

    private static long SafeMul(long a, long b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    private static long SafeMul3(long a, long b, long c)
    {
        return SafeMul(SafeMul(a, b), c);
    }

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        }

        public static Vec3 operator -(Vec3 a, Vec3 b)
        {
            return new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        }

        public static Vec3 operator *(Vec3 a, double s)
        {
            return new Vec3(a.X * s, a.Y * s, a.Z * s);
        }

        public static Vec3 operator /(Vec3 a, double s)
        {
            return new Vec3(a.X / s, a.Y / s, a.Z / s);
        }

        public double Dot(Vec3 b)
        {
            return X * b.X + Y * b.Y + Z * b.Z;
        }

        public Vec3 Cross(Vec3 b)
        {
            return new Vec3(
                Y * b.Z - Z * b.Y,
                Z * b.X - X * b.Z,
                X * b.Y - Y * b.X
            );
        }

        public double LengthSquared()
        {
            return X * X + Y * Y + Z * Z;
        }

        public double Length()
        {
            return Math.Sqrt(LengthSquared());
        }

        public Vec3 Normalized()
        {
            double len = Length();
            if (len < 1e-18) return new Vec3(1.0, 0.0, 0.0);
            return this / len;
        }
    }

    private struct Mat3
    {
        public double M11, M12, M13;
        public double M21, M22, M23;
        public double M31, M32, M33;

        public Mat3(
            double m11, double m12, double m13,
            double m21, double m22, double m23,
            double m31, double m32, double m33)
        {
            M11 = m11; M12 = m12; M13 = m13;
            M21 = m21; M22 = m22; M23 = m23;
            M31 = m31; M32 = m32; M33 = m33;
        }

        public static Mat3 Identity
        {
            get { return new Mat3(1, 0, 0, 0, 1, 0, 0, 0, 1); }
        }

        public Vec3 Row1
        {
            get { return new Vec3(M11, M12, M13); }
        }

        public Vec3 Row2
        {
            get { return new Vec3(M21, M22, M23); }
        }

        public Vec3 Row3
        {
            get { return new Vec3(M31, M32, M33); }
        }

        public static Mat3 FromRows(Vec3 r1, Vec3 r2, Vec3 r3)
        {
            return new Mat3(
                r1.X, r1.Y, r1.Z,
                r2.X, r2.Y, r2.Z,
                r3.X, r3.Y, r3.Z
            );
        }

        public double Determinant()
        {
            return
                M11 * (M22 * M33 - M23 * M32) -
                M12 * (M21 * M33 - M23 * M31) +
                M13 * (M21 * M32 - M22 * M31);
        }

        public static Mat3 operator *(Mat3 a, Mat3 b)
        {
            return new Mat3(
                a.M11 * b.M11 + a.M12 * b.M21 + a.M13 * b.M31,
                a.M11 * b.M12 + a.M12 * b.M22 + a.M13 * b.M32,
                a.M11 * b.M13 + a.M12 * b.M23 + a.M13 * b.M33,

                a.M21 * b.M11 + a.M22 * b.M21 + a.M23 * b.M31,
                a.M21 * b.M12 + a.M22 * b.M22 + a.M23 * b.M32,
                a.M21 * b.M13 + a.M22 * b.M23 + a.M23 * b.M33,

                a.M31 * b.M11 + a.M32 * b.M21 + a.M33 * b.M31,
                a.M31 * b.M12 + a.M32 * b.M22 + a.M33 * b.M32,
                a.M31 * b.M13 + a.M32 * b.M23 + a.M33 * b.M33
            );
        }
    }

    private struct Quat
    {
        public double W;
        public double X;
        public double Y;
        public double Z;

        public Quat(double w, double x, double y, double z)
        {
            W = w;
            X = x;
            Y = y;
            Z = z;
        }

        public Quat Normalized()
        {
            double n = Math.Sqrt(W * W + X * X + Y * Y + Z * Z);
            if (n < 1e-18) return new Quat(1, 0, 0, 0);
            return new Quat(W / n, X / n, Y / n, Z / n);
        }

        public Quat Canonicalized()
        {
            Quat q = Normalized();
            if (q.W < 0.0 ||
                (Math.Abs(q.W) < 1e-15 && q.X < 0.0) ||
                (Math.Abs(q.W) < 1e-15 && Math.Abs(q.X) < 1e-15 && q.Y < 0.0) ||
                (Math.Abs(q.W) < 1e-15 && Math.Abs(q.X) < 1e-15 && Math.Abs(q.Y) < 1e-15 && q.Z < 0.0))
            {
                q.W = -q.W;
                q.X = -q.X;
                q.Y = -q.Y;
                q.Z = -q.Z;
            }
            return q;
        }

        public Mat3 ToMatrix()
        {
            Quat q = Normalized();
            double xx = q.X * q.X;
            double yy = q.Y * q.Y;
            double zz = q.Z * q.Z;
            double xy = q.X * q.Y;
            double xz = q.X * q.Z;
            double yz = q.Y * q.Z;
            double wx = q.W * q.X;
            double wy = q.W * q.Y;
            double wz = q.W * q.Z;

            return new Mat3(
                1.0 - 2.0 * (yy + zz),
                2.0 * (xy - wz),
                2.0 * (xz + wy),

                2.0 * (xy + wz),
                1.0 - 2.0 * (xx + zz),
                2.0 * (yz - wx),

                2.0 * (xz - wy),
                2.0 * (yz + wx),
                1.0 - 2.0 * (xx + yy)
            );
        }

        public static Quat FromMatrix(Mat3 m)
        {
            double trace = m.M11 + m.M22 + m.M33;
            double w, x, y, z;

            if (trace > 0.0)
            {
                double s = Math.Sqrt(trace + 1.0) * 2.0;
                w = 0.25 * s;
                x = (m.M32 - m.M23) / s;
                y = (m.M13 - m.M31) / s;
                z = (m.M21 - m.M12) / s;
            }
            else if (m.M11 > m.M22 && m.M11 > m.M33)
            {
                double s = Math.Sqrt(1.0 + m.M11 - m.M22 - m.M33) * 2.0;
                w = (m.M32 - m.M23) / s;
                x = 0.25 * s;
                y = (m.M12 + m.M21) / s;
                z = (m.M13 + m.M31) / s;
            }
            else if (m.M22 > m.M33)
            {
                double s = Math.Sqrt(1.0 + m.M22 - m.M11 - m.M33) * 2.0;
                w = (m.M13 - m.M31) / s;
                x = (m.M12 + m.M21) / s;
                y = 0.25 * s;
                z = (m.M23 + m.M32) / s;
            }
            else
            {
                double s = Math.Sqrt(1.0 + m.M33 - m.M11 - m.M22) * 2.0;
                w = (m.M21 - m.M12) / s;
                x = (m.M13 + m.M31) / s;
                y = (m.M23 + m.M32) / s;
                z = 0.25 * s;
            }

            return new Quat(w, x, y, z).Canonicalized();
        }

        public static Quat FromAxisAngle(Vec3 axis, double angle)
        {
            Vec3 a = axis.Normalized();
            double half = 0.5 * angle;
            double s = Math.Sin(half);
            return new Quat(Math.Cos(half), a.X * s, a.Y * s, a.Z * s).Canonicalized();
        }

        public static Quat operator *(Quat a, Quat b)
        {
            return new Quat(
                a.W * b.W - a.X * b.X - a.Y * b.Y - a.Z * b.Z,
                a.W * b.X + a.X * b.W + a.Y * b.Z - a.Z * b.Y,
                a.W * b.Y - a.X * b.Z + a.Y * b.W + a.Z * b.X,
                a.W * b.Z + a.X * b.Y - a.Y * b.X + a.Z * b.W
            );
        }
    }

    private sealed class FillPlan : PlanNode
    {
        public OrientationType Type;
        public long Nx;
        public long Ny;
        public long Nz;
        public double OriginX;
        public double OriginY;
        public double OriginZ;

        public FillPlan(OrientationType type, long nx, long ny, long nz, double ox, double oy, double oz)
        {
            Type = type;
            Nx = nx;
            Ny = ny;
            Nz = nz;
            OriginX = ox;
            OriginY = oy;
            OriginZ = oz;
            Count = SafeMul3(nx, ny, nz);
            UsedBoxVolume = type.BoxVolume * (double)Count;
        }

        public override void Write(StreamWriter sw)
        {
            if (Count <= 0) return;

            StringBuilder sb = new StringBuilder(1 << 20);

            double baseTx = OriginX - Type.MinX;
            double baseTy = OriginY - Type.MinY;
            double baseTz = OriginZ - Type.MinZ;
            double stepX = Type.SizeX;
            double stepY = Type.SizeY;
            double stepZ = Type.SizeZ;
            string tail = Type.QTail;

            for (long k = 0; k < Nz; k++)
            {
                double tz = baseTz + k * stepZ;
                string tzs = FormatDouble(tz);

                for (long j = 0; j < Ny; j++)
                {
                    double ty = baseTy + j * stepY;
                    string tys = FormatDouble(ty);

                    for (long i = 0; i < Nx; i++)
                    {
                        double tx = baseTx + i * stepX;
                        sb.Append(FormatDouble(tx)).Append(' ')
                          .Append(tys).Append(' ')
                          .Append(tzs).Append(tail);

                        if (sb.Length >= (1 << 20))
                        {
                            sw.Write(sb.ToString());
                            sb.Clear();
                        }
                    }
                }
            }

            if (sb.Length > 0)
            {
                sw.Write(sb.ToString());
            }
        }
    }

    private sealed class CompositePlan : PlanNode
    {
        public readonly List<PlanNode> Children;

        public CompositePlan(List<PlanNode> children)
        {
            Children = children;
            long total = 0;
            double vol = 0.0;
            for (int i = 0; i < children.Count; i++)
            {
                total = SafeAdd(total, children[i].Count);
                vol += children[i].UsedBoxVolume;
            }
            Count = total;
            UsedBoxVolume = vol;
        }

        public override void Write(StreamWriter sw)
        {
            for (int i = 0; i < Children.Count; i++)
            {
                Children[i].Write(sw);
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        private int Skip()
        {
            int c;
            do
            {
                c = Read();
            } while (c != -1 && c <= 32);
            return c;
        }

        public int NextInt()
        {
            int c = Skip();
            if (c == -1) throw new EndOfStreamException();

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            else if (c == '+')
            {
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }

            return sign * val;
        }

        public double NextDouble()
        {
            int c = Skip();
            if (c == -1) throw new EndOfStreamException();

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            else if (c == '+')
            {
                c = Read();
            }

            double val = 0.0;
            while (c > 32 && c != '.' && c != 'e' && c != 'E')
            {
                val = val * 10.0 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                double frac = 0.1;
                c = Read();
                while (c > 32 && c != 'e' && c != 'E')
                {
                    val += (c - '0') * frac;
                    frac *= 0.1;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                int expSign = 1;
                int exp = 0;
                c = Read();
                if (c == '-')
                {
                    expSign = -1;
                    c = Read();
                }
                else if (c == '+')
                {
                    c = Read();
                }

                while (c > 32)
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }

                val *= Math.Pow(10.0, expSign * exp);
            }

            return sign * val;
        }
    }

    private sealed class Solver
    {
        private readonly Vec3[] vertices;      // centered vertices
        private readonly RawFace[] rawFaces;
        private readonly double containerW;
        private readonly double containerH;
        private readonly double containerD;
        private readonly Vec3 meshCenter;      // original center shift
        private readonly Stopwatch watch;

        private SearchConfig config;
        private List<OrientationType> types;
        private double minBoxVolume;
        private int[][] orders;

        private readonly List<NearFitSeed> nearFitSeeds = new List<NearFitSeed>(8);

        private const long HardTimeLimitMs = 295000;
        private const double KeyScale = 1e7;

        private static readonly int[][] AllOrders = new int[][]
        {
            new int[] { 0, 1, 2 },
            new int[] { 0, 2, 1 },
            new int[] { 1, 0, 2 },
            new int[] { 1, 2, 0 },
            new int[] { 2, 0, 1 },
            new int[] { 2, 1, 0 }
        };

        private static readonly int[][] MainOrders = new int[][]
        {
            new int[] { 0, 1, 2 },
            new int[] { 1, 2, 0 },
            new int[] { 2, 0, 1 }
        };

        private static readonly List<Mat3> CubeRotations = BuildCubeRotations();

        private struct NearFitSeed
        {
            public Quat Q;
            public double Score;

            public NearFitSeed(Quat q, double score)
            {
                Q = q;
                Score = score;
            }
        }

        private sealed class ActionChoice
        {
            public OrientationType Type;
            public int[] Order;
            public long Nx;
            public long Ny;
            public long Nz;
            public Region[] Children;
            public long BaseCount;
            public long LowerBound;
            public long UpperBound;
            public double BaseUsedVolume;
        }

        public Solver(Vec3[] originalVertices, RawFace[] faces, double w, double h, double d)
        {
            containerW = w;
            containerH = h;
            containerD = d;
            rawFaces = faces;
            watch = Stopwatch.StartNew();

            if (originalVertices.Length == 0)
            {
                vertices = originalVertices;
                meshCenter = new Vec3(0, 0, 0);
                return;
            }

            double sx = 0.0, sy = 0.0, sz = 0.0;
            for (int i = 0; i < originalVertices.Length; i++)
            {
                sx += originalVertices[i].X;
                sy += originalVertices[i].Y;
                sz += originalVertices[i].Z;
            }

            meshCenter = new Vec3(sx / originalVertices.Length, sy / originalVertices.Length, sz / originalVertices.Length);

            vertices = new Vec3[originalVertices.Length];
            for (int i = 0; i < originalVertices.Length; i++)
            {
                vertices[i] = originalVertices[i] - meshCenter;
            }
        }

        public PlanNode Solve()
        {
            if (containerW <= 0 || containerH <= 0 || containerD <= 0 || vertices.Length == 0)
                return EmptyPlan.Instance;

            config = BuildConfig();
            BuildCandidates();

            if (types == null || types.Count == 0)
                return EmptyPlan.Instance;

            for (int i = 0; i < types.Count; i++)
            {
                types[i].PrepareOutputText();
            }

            minBoxVolume = double.PositiveInfinity;
            for (int i = 0; i < types.Count; i++)
            {
                if (types[i].BoxVolume < minBoxVolume) minBoxVolume = types[i].BoxVolume;
            }
            if (!(minBoxVolume > 0.0) || double.IsInfinity(minBoxVolume))
                return EmptyPlan.Instance;

            orders = config.UseAllOrders ? AllOrders : MainOrders;

            long bestSingle = types[0].RootSingleCount;

            if (bestSingle > 1000000)
            {
                config.SearchDepth = 0;
                config.BeamWidth = Math.Min(config.BeamWidth, 4);
            }
            else if (bestSingle > 200000)
            {
                config.SearchDepth = Math.Min(config.SearchDepth, 1);
                config.BeamWidth = Math.Min(config.BeamWidth, 6);
            }
            else if (bestSingle > 50000)
            {
                config.SearchDepth = Math.Min(config.SearchDepth, 2);
            }
            else
            {
                if (bestSingle <= 5000 && types.Count <= 12 && config.SearchDepth < 3) config.SearchDepth++;
                if (bestSingle <= 200 && types.Count <= 10 && config.SearchDepth < 4) config.SearchDepth++;
                if (bestSingle <= 50 && types.Count <= 8 && config.SearchDepth < 4) config.SearchDepth++;
            }

            if (TimeAlmostUp())
            {
                config.SearchDepth = 0;
            }

            Region root = new Region(0.0, 0.0, 0.0, containerW, containerH, containerD);

            if (config.SearchDepth <= 0)
                return GetShallowGreedyPlan(root);

            return SearchRegion(root, config.SearchDepth);
        }

        private SearchConfig BuildConfig()
        {
            SearchConfig cfg = new SearchConfig();

            double maxDim = Math.Max(containerW, Math.Max(containerH, containerD));
            long complexity = (long)vertices.Length + (long)rawFaces.Length;

            int level;
            if (maxDim <= 8.0) level = 0;
            else if (maxDim <= 25.0) level = 1;
            else if (maxDim <= 120.0) level = 2;
            else level = 3;

            if (complexity > 2000) level = Math.Max(level, 1);
            if (complexity > 20000) level = Math.Max(level, 2);
            if (complexity > 100000) level = 3;

            switch (level)
            {
                case 0:
                    cfg.Scale = ScaleClass.Trivial;
                    cfg.CandidateLimit = 40;
                    cfg.SearchDepth = 4;
                    cfg.BeamWidth = 28;
                    cfg.RandomSamples = 192;
                    cfg.FaceOrientationCap = 12;
                    cfg.UseAllOrders = true;
                    cfg.AxisExhaustiveLimit = 18;
                    cfg.LocalPerturbAngleDeg = 20.0;
                    break;
                case 1:
                    cfg.Scale = ScaleClass.Medium;
                    cfg.CandidateLimit = 28;
                    cfg.SearchDepth = 3;
                    cfg.BeamWidth = 18;
                    cfg.RandomSamples = 96;
                    cfg.FaceOrientationCap = 8;
                    cfg.UseAllOrders = true;
                    cfg.AxisExhaustiveLimit = 10;
                    cfg.LocalPerturbAngleDeg = 14.0;
                    break;
                case 2:
                    cfg.Scale = ScaleClass.Large;
                    cfg.CandidateLimit = 16;
                    cfg.SearchDepth = 2;
                    cfg.BeamWidth = 10;
                    cfg.RandomSamples = 32;
                    cfg.FaceOrientationCap = 4;
                    cfg.UseAllOrders = false;
                    cfg.AxisExhaustiveLimit = 6;
                    cfg.LocalPerturbAngleDeg = 10.0;
                    break;
                default:
                    cfg.Scale = ScaleClass.Extreme;
                    cfg.CandidateLimit = 10;
                    cfg.SearchDepth = 2;
                    cfg.BeamWidth = 6;
                    cfg.RandomSamples = 12;
                    cfg.FaceOrientationCap = 2;
                    cfg.UseAllOrders = false;
                    cfg.AxisExhaustiveLimit = 4;
                    cfg.LocalPerturbAngleDeg = 7.0;
                    break;
            }

            int geomScale = Math.Max(1, Math.Max(vertices.Length / 20000, rawFaces.Length / 50000));
            cfg.RandomSamples = Math.Max(8, cfg.RandomSamples / geomScale);
            if (cfg.FaceOrientationCap > 0)
                cfg.FaceOrientationCap = Math.Max(1, cfg.FaceOrientationCap / geomScale);

            if (vertices.Length > 100000 || rawFaces.Length > 200000)
            {
                cfg.CandidateLimit = Math.Min(cfg.CandidateLimit, 16);
                cfg.BeamWidth = Math.Min(cfg.BeamWidth, 10);
                cfg.SearchDepth = Math.Min(cfg.SearchDepth, 2);
            }

            if (vertices.Length > 300000 || rawFaces.Length > 600000)
            {
                cfg.CandidateLimit = Math.Min(cfg.CandidateLimit, 10);
                cfg.BeamWidth = Math.Min(cfg.BeamWidth, 6);
                cfg.SearchDepth = Math.Min(cfg.SearchDepth, 1);
                cfg.RandomSamples = Math.Min(cfg.RandomSamples, 8);
                cfg.FaceOrientationCap = 0;
            }

            return cfg;
        }

        private void BuildCandidates()
        {
            Dictionary<DimKey, OrientationType> registry = new Dictionary<DimKey, OrientationType>();

            List<Mat3> bases = new List<Mat3>();
            bases.Add(Mat3.Identity);
            bases.Add(BuildPcaBase());
            AddFaceBases(bases);

            for (int bi = 0; bi < bases.Count; bi++)
            {
                if (TimeAlmostUp()) break;
                Mat3 baseMat = OrthonormalizeRows(bases[bi]);

                for (int ri = 0; ri < CubeRotations.Count; ri++)
                {
                    AddCandidateFromMatrix(registry, CubeRotations[ri] * baseMat);
                    if (TimeAlmostUp()) break;
                }
            }

            Random rng = new Random(123456789);
            int globalRandom = Math.Max(4, config.RandomSamples / 2);
            for (int i = 0; i < globalRandom && !TimeAlmostUp(); i++)
            {
                AddCandidateFromQuaternion(registry, RandomQuaternion(rng));
            }

            List<OrientationType> provisional = new List<OrientationType>(registry.Values);
            provisional.Sort(CompareOrientation);

            List<Quat> seeds = new List<Quat>();
            for (int i = 0; i < provisional.Count && i < 4; i++)
                seeds.Add(provisional[i].Q);

            if (seeds.Count == 0)
            {
                for (int i = 0; i < nearFitSeeds.Count && i < 4; i++)
                    seeds.Add(nearFitSeeds[i].Q);
            }

            int localTotal = Math.Max(0, config.RandomSamples - globalRandom);
            if (seeds.Count > 0)
            {
                int each = Math.Max(4, localTotal / seeds.Count);
                double angle = config.LocalPerturbAngleDeg * Math.PI / 180.0;

                for (int s = 0; s < seeds.Count && !TimeAlmostUp(); s++)
                {
                    for (int j = 0; j < each && !TimeAlmostUp(); j++)
                    {
                        AddCandidateFromQuaternion(registry, PerturbQuaternion(seeds[s], rng, angle));
                        if ((j & 1) == 0)
                            AddCandidateFromQuaternion(registry, PerturbQuaternion(seeds[s], rng, angle * 0.4));
                    }
                }
            }

            if (registry.Count == 0 && !TimeAlmostUp())
            {
                int rescue = Math.Max(64, config.RandomSamples * 6);

                if (nearFitSeeds.Count > 0)
                {
                    double angle = Math.Max(config.LocalPerturbAngleDeg * 2.0, 25.0) * Math.PI / 180.0;
                    int each = Math.Max(16, rescue / nearFitSeeds.Count);

                    for (int s = 0; s < nearFitSeeds.Count && !TimeAlmostUp(); s++)
                    {
                        for (int j = 0; j < each && !TimeAlmostUp(); j++)
                        {
                            AddCandidateFromQuaternion(registry, PerturbQuaternion(nearFitSeeds[s].Q, rng, angle));
                        }
                    }
                }

                for (int i = 0; i < rescue && registry.Count == 0 && !TimeAlmostUp(); i++)
                {
                    AddCandidateFromQuaternion(registry, RandomQuaternion(rng));
                }
            }

            types = SelectCandidates(registry.Values);
        }

        private void AddFaceBases(List<Mat3> bases)
        {
            if (config.FaceOrientationCap <= 0 || rawFaces.Length == 0) return;

            List<int> offsets = new List<int>(2);
            if (FacesValidWithOffset(0)) offsets.Add(0);
            if (FacesValidWithOffset(1)) offsets.Add(1);

            if (offsets.Count == 0) return;

            int capPerOffset = Math.Max(1, config.FaceOrientationCap / offsets.Count);

            for (int oi = 0; oi < offsets.Count; oi++)
            {
                if (TimeAlmostUp()) break;

                int keep = Math.Max(capPerOffset * 4, capPerOffset + 4);
                List<FaceSeed> seeds = CollectTopFaceSeeds(offsets[oi], keep);
                seeds.Sort(delegate (FaceSeed a, FaceSeed b)
                {
                    if (a.Area > b.Area) return -1;
                    if (a.Area < b.Area) return 1;
                    return 0;
                });

                int added = 0;
                for (int i = 0; i < seeds.Count && added < capPerOffset; i++)
                {
                    Mat3 basis;
                    if (TryMakeFaceBasis(seeds[i], out basis))
                    {
                        bases.Add(basis);
                        added++;
                    }
                }
            }
        }

        private bool FacesValidWithOffset(int offset)
        {
            int n = vertices.Length;
            for (int i = 0; i < rawFaces.Length; i++)
            {
                int a = rawFaces[i].A - offset;
                int b = rawFaces[i].B - offset;
                int c = rawFaces[i].C - offset;
                if ((uint)a >= (uint)n || (uint)b >= (uint)n || (uint)c >= (uint)n)
                    return false;
            }
            return true;
        }

        private List<FaceSeed> CollectTopFaceSeeds(int offset, int keep)
        {
            List<FaceSeed> best = new List<FaceSeed>(keep);
            if (keep <= 0) return best;

            for (int i = 0; i < rawFaces.Length; i++)
            {
                int a = rawFaces[i].A - offset;
                int b = rawFaces[i].B - offset;
                int c = rawFaces[i].C - offset;

                if ((uint)a >= (uint)vertices.Length || (uint)b >= (uint)vertices.Length || (uint)c >= (uint)vertices.Length)
                    continue;

                Vec3 va = vertices[a];
                Vec3 vb = vertices[b];
                Vec3 vc = vertices[c];

                Vec3 e1 = vb - va;
                Vec3 e2 = vc - va;
                Vec3 n = e1.Cross(e2);
                double area = n.Length();

                if (area < 1e-15) continue;

                FaceSeed seed = new FaceSeed(area, n, e1, e2);

                if (best.Count < keep)
                {
                    best.Add(seed);
                }
                else
                {
                    int minIdx = 0;
                    double minArea = best[0].Area;
                    for (int j = 1; j < best.Count; j++)
                    {
                        if (best[j].Area < minArea)
                        {
                            minArea = best[j].Area;
                            minIdx = j;
                        }
                    }

                    if (area > minArea)
                    {
                        best[minIdx] = seed;
                    }
                }
            }

            return best;
        }

        private bool TryMakeFaceBasis(FaceSeed seed, out Mat3 basis)
        {
            Vec3 z = seed.Normal.Normalized();
            Vec3 x = seed.Edge1 - z * z.Dot(seed.Edge1);
            if (x.LengthSquared() < 1e-20)
                x = seed.Edge2 - z * z.Dot(seed.Edge2);

            if (x.LengthSquared() < 1e-20)
            {
                basis = Mat3.Identity;
                return false;
            }

            x = x.Normalized();
            Vec3 y = z.Cross(x);
            if (y.LengthSquared() < 1e-20)
            {
                basis = Mat3.Identity;
                return false;
            }

            y = y.Normalized();
            x = y.Cross(z).Normalized();

            basis = Mat3.FromRows(x, y, z);
            return true;
        }

        private Mat3 BuildPcaBase()
        {
            if (vertices.Length == 0) return Mat3.Identity;

            double a00 = 0.0, a01 = 0.0, a02 = 0.0;
            double a11 = 0.0, a12 = 0.0, a22 = 0.0;

            for (int i = 0; i < vertices.Length; i++)
            {
                double x = vertices[i].X;
                double y = vertices[i].Y;
                double z = vertices[i].Z;
                a00 += x * x;
                a01 += x * y;
                a02 += x * z;
                a11 += y * y;
                a12 += y * z;
                a22 += z * z;
            }

            double[,] a = new double[3, 3];
            a[0, 0] = a00; a[0, 1] = a01; a[0, 2] = a02;
            a[1, 0] = a01; a[1, 1] = a11; a[1, 2] = a12;
            a[2, 0] = a02; a[2, 1] = a12; a[2, 2] = a22;

            double[,] v = new double[3, 3];
            v[0, 0] = 1.0; v[0, 1] = 0.0; v[0, 2] = 0.0;
            v[1, 0] = 0.0; v[1, 1] = 1.0; v[1, 2] = 0.0;
            v[2, 0] = 0.0; v[2, 1] = 0.0; v[2, 2] = 1.0;

            for (int iter = 0; iter < 32; iter++)
            {
                int p = 0, q = 1;
                double max = Math.Abs(a[0, 1]);

                if (Math.Abs(a[0, 2]) > max) { max = Math.Abs(a[0, 2]); p = 0; q = 2; }
                if (Math.Abs(a[1, 2]) > max) { max = Math.Abs(a[1, 2]); p = 1; q = 2; }

                if (max < 1e-15) break;

                double app = a[p, p];
                double aqq = a[q, q];
                double apq = a[p, q];

                double phi = 0.5 * Math.Atan2(2.0 * apq, aqq - app);
                double c = Math.Cos(phi);
                double s = Math.Sin(phi);

                for (int i = 0; i < 3; i++)
                {
                    if (i == p || i == q) continue;
                    double aip = a[i, p];
                    double aiq = a[i, q];
                    a[i, p] = a[p, i] = c * aip - s * aiq;
                    a[i, q] = a[q, i] = s * aip + c * aiq;
                }

                double newApp = c * c * app - 2.0 * s * c * apq + s * s * aqq;
                double newAqq = s * s * app + 2.0 * s * c * apq + c * c * aqq;
                a[p, p] = newApp;
                a[q, q] = newAqq;
                a[p, q] = a[q, p] = 0.0;

                for (int i = 0; i < 3; i++)
                {
                    double vip = v[i, p];
                    double viq = v[i, q];
                    v[i, p] = c * vip - s * viq;
                    v[i, q] = s * vip + c * viq;
                }
            }

            double[] eval = new double[] { a[0, 0], a[1, 1], a[2, 2] };
            int[] idx = new int[] { 0, 1, 2 };
            for (int i = 0; i < 2; i++)
            {
                for (int j = i + 1; j < 3; j++)
                {
                    if (eval[idx[j]] > eval[idx[i]])
                    {
                        int t = idx[i];
                        idx[i] = idx[j];
                        idx[j] = t;
                    }
                }
            }

            Vec3 e0 = new Vec3(v[0, idx[0]], v[1, idx[0]], v[2, idx[0]]).Normalized();
            Vec3 e1 = new Vec3(v[0, idx[1]], v[1, idx[1]], v[2, idx[1]]).Normalized();
            Vec3 e2 = new Vec3(v[0, idx[2]], v[1, idx[2]], v[2, idx[2]]).Normalized();

            Mat3 basis = Mat3.FromRows(e0, e1, e2);
            if (basis.Determinant() < 0.0)
            {
                e2 = e2 * -1.0;
                basis = Mat3.FromRows(e0, e1, e2);
            }

            return OrthonormalizeRows(basis);
        }

        private static Mat3 OrthonormalizeRows(Mat3 m)
        {
            Vec3 x = m.Row1;
            if (x.LengthSquared() < 1e-20) x = new Vec3(1, 0, 0);
            x = x.Normalized();

            Vec3 y = m.Row2 - x * x.Dot(m.Row2);
            if (y.LengthSquared() < 1e-20)
                y = m.Row3 - x * x.Dot(m.Row3);

            if (y.LengthSquared() < 1e-20)
            {
                Vec3 tmp = Math.Abs(x.X) < 0.9 ? new Vec3(1, 0, 0) : new Vec3(0, 1, 0);
                y = tmp - x * x.Dot(tmp);
            }

            y = y.Normalized();
            Vec3 z = x.Cross(y);
            if (z.LengthSquared() < 1e-20)
                z = new Vec3(0, 0, 1);

            z = z.Normalized();
            y = z.Cross(x).Normalized();

            return Mat3.FromRows(x, y, z);
        }

        private void AddCandidateFromMatrix(Dictionary<DimKey, OrientationType> registry, Mat3 matrix)
        {
            Mat3 ortho = OrthonormalizeRows(matrix);
            Quat q = Quat.FromMatrix(ortho).Canonicalized();
            Mat3 r = q.ToMatrix();
            AddCandidateCore(registry, q, r);
        }

        private void AddCandidateFromQuaternion(Dictionary<DimKey, OrientationType> registry, Quat q)
        {
            q = q.Canonicalized();
            Mat3 r = q.ToMatrix();
            AddCandidateCore(registry, q, r);
        }

        private void AddCandidateCore(Dictionary<DimKey, OrientationType> registry, Quat q, Mat3 r)
        {
            double minX = double.PositiveInfinity;
            double minY = double.PositiveInfinity;
            double minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity;
            double maxY = double.NegativeInfinity;
            double maxZ = double.NegativeInfinity;

            double m11 = r.M11, m12 = r.M12, m13 = r.M13;
            double m21 = r.M21, m22 = r.M22, m23 = r.M23;
            double m31 = r.M31, m32 = r.M32, m33 = r.M33;

            for (int i = 0; i < vertices.Length; i++)
            {
                double vx = vertices[i].X;
                double vy = vertices[i].Y;
                double vz = vertices[i].Z;

                double x = m11 * vx + m12 * vy + m13 * vz;
                double y = m21 * vx + m22 * vy + m23 * vz;
                double z = m31 * vx + m32 * vy + m33 * vz;

                if (x < minX) minX = x;
                if (y < minY) minY = y;
                if (z < minZ) minZ = z;
                if (x > maxX) maxX = x;
                if (y > maxY) maxY = y;
                if (z > maxZ) maxZ = z;
            }

            double sx = maxX - minX;
            double sy = maxY - minY;
            double sz = maxZ - minZ;

            if (!(sx > 1e-15) || !(sy > 1e-15) || !(sz > 1e-15)) return;
            if (double.IsNaN(sx) || double.IsNaN(sy) || double.IsNaN(sz)) return;

            double cx = m11 * meshCenter.X + m12 * meshCenter.Y + m13 * meshCenter.Z;
            double cy = m21 * meshCenter.X + m22 * meshCenter.Y + m23 * meshCenter.Z;
            double cz = m31 * meshCenter.X + m32 * meshCenter.Y + m33 * meshCenter.Z;

            long nx = FitCount(containerW, sx);
            long ny = FitCount(containerH, sy);
            long nz = FitCount(containerD, sz);
            long count = SafeMul3(nx, ny, nz);

            if (count <= 0)
            {
                double rx = sx / containerW;
                double ry = sy / containerH;
                double rz = sz / containerD;
                double score = Math.Max(rx, Math.Max(ry, rz)) + 0.001 * (rx + ry + rz);
                ConsiderNearFit(q, score);
                return;
            }

            OrientationType cand = new OrientationType();
            cand.Q = q;
            cand.MinX = minX + cx;
            cand.MinY = minY + cy;
            cand.MinZ = minZ + cz;
            cand.SizeX = sx;
            cand.SizeY = sy;
            cand.SizeZ = sz;
            cand.BoxVolume = sx * sy * sz;
            cand.RootSingleCount = count;
            cand.Key = MakeKey(sx, sy, sz);

            OrientationType existing;
            if (registry.TryGetValue(cand.Key, out existing))
            {
                bool replace = false;
                if (cand.RootSingleCount > existing.RootSingleCount)
                {
                    replace = true;
                }
                else if (cand.RootSingleCount == existing.RootSingleCount)
                {
                    if (cand.BoxVolume < existing.BoxVolume - 1e-12)
                        replace = true;
                    else
                    {
                        double candMax = Math.Max(cand.SizeX, Math.Max(cand.SizeY, cand.SizeZ));
                        double exMax = Math.Max(existing.SizeX, Math.Max(existing.SizeY, existing.SizeZ));
                        if (candMax < exMax - 1e-12)
                            replace = true;
                    }
                }

                if (replace) registry[cand.Key] = cand;
            }
            else
            {
                registry.Add(cand.Key, cand);
            }
        }

        private DimKey MakeKey(double sx, double sy, double sz)
        {
            long kx = (long)Math.Round(sx * KeyScale);
            long ky = (long)Math.Round(sy * KeyScale);
            long kz = (long)Math.Round(sz * KeyScale);
            return new DimKey(kx, ky, kz);
        }

        private void ConsiderNearFit(Quat q, double score)
        {
            if (double.IsNaN(score) || double.IsInfinity(score)) return;

            if (nearFitSeeds.Count < 8)
            {
                nearFitSeeds.Add(new NearFitSeed(q, score));
                nearFitSeeds.Sort(delegate (NearFitSeed a, NearFitSeed b)
                {
                    if (a.Score < b.Score) return -1;
                    if (a.Score > b.Score) return 1;
                    return 0;
                });
                return;
            }

            int worst = 0;
            double worstScore = nearFitSeeds[0].Score;
            for (int i = 1; i < nearFitSeeds.Count; i++)
            {
                if (nearFitSeeds[i].Score > worstScore)
                {
                    worstScore = nearFitSeeds[i].Score;
                    worst = i;
                }
            }

            if (score < worstScore)
            {
                nearFitSeeds[worst] = new NearFitSeed(q, score);
                nearFitSeeds.Sort(delegate (NearFitSeed a, NearFitSeed b)
                {
                    if (a.Score < b.Score) return -1;
                    if (a.Score > b.Score) return 1;
                    return 0;
                });
            }
        }

        private List<OrientationType> SelectCandidates(ICollection<OrientationType> all)
        {
            List<OrientationType> list = new List<OrientationType>(all);
            list.Sort(CompareOrientation);

            if (list.Count == 0)
                return list;

            if (list.Count <= config.CandidateLimit)
                return list;

            List<OrientationType> selected = new List<OrientationType>(config.CandidateLimit);
            HashSet<DimKey> used = new HashSet<DimKey>();

            int topOverall = Math.Min(3, Math.Max(1, config.CandidateLimit / 4));
            for (int i = 0; i < list.Count && i < topOverall; i++)
                AddUniqueCandidate(selected, used, list[i]);

            int perAxis = config.CandidateLimit >= 20 ? 2 : 1;

            AddSortedByMetric(selected, used, list, 0, perAxis);
            AddSortedByMetric(selected, used, list, 1, perAxis);
            AddSortedByMetric(selected, used, list, 2, perAxis);
            AddSortedByBoxVolume(selected, used, list, perAxis);

            OrientationType bestLayerX = FindBestLayerCandidate(list, 0);
            OrientationType bestLayerY = FindBestLayerCandidate(list, 1);
            OrientationType bestLayerZ = FindBestLayerCandidate(list, 2);
            AddUniqueCandidate(selected, used, bestLayerX);
            AddUniqueCandidate(selected, used, bestLayerY);
            AddUniqueCandidate(selected, used, bestLayerZ);

            for (int i = 0; i < list.Count && selected.Count < config.CandidateLimit; i++)
                AddUniqueCandidate(selected, used, list[i]);

            selected.Sort(CompareOrientation);
            if (selected.Count > config.CandidateLimit)
                selected.RemoveRange(config.CandidateLimit, selected.Count - config.CandidateLimit);

            return selected;
        }

        private static void AddUniqueCandidate(List<OrientationType> selected, HashSet<DimKey> used, OrientationType c)
        {
            if (c == null) return;
            if (used.Add(c.Key))
                selected.Add(c);
        }

        private static void AddSortedByMetric(List<OrientationType> selected, HashSet<DimKey> used, List<OrientationType> list, int axis, int take)
        {
            List<OrientationType> tmp = new List<OrientationType>(list);
            tmp.Sort(delegate (OrientationType a, OrientationType b)
            {
                double da = a.GetSize(axis);
                double db = b.GetSize(axis);
                if (da < db - 1e-12) return -1;
                if (da > db + 1e-12) return 1;
                return CompareOrientation(a, b);
            });

            for (int i = 0; i < tmp.Count && i < take; i++)
                AddUniqueCandidate(selected, used, tmp[i]);
        }

        private static void AddSortedByBoxVolume(List<OrientationType> selected, HashSet<DimKey> used, List<OrientationType> list, int take)
        {
            List<OrientationType> tmp = new List<OrientationType>(list);
            tmp.Sort(delegate (OrientationType a, OrientationType b)
            {
                if (a.BoxVolume < b.BoxVolume - 1e-12) return -1;
                if (a.BoxVolume > b.BoxVolume + 1e-12) return 1;
                return CompareOrientation(a, b);
            });

            for (int i = 0; i < tmp.Count && i < take; i++)
                AddUniqueCandidate(selected, used, tmp[i]);
        }

        private OrientationType FindBestLayerCandidate(List<OrientationType> list, int axis)
        {
            OrientationType best = null;
            long bestLayerCount = -1;
            double bestThickness = double.PositiveInfinity;

            for (int i = 0; i < list.Count; i++)
            {
                long layerCount;
                double thickness;

                if (axis == 0)
                {
                    layerCount = SafeMul(FitCount(containerH, list[i].SizeY), FitCount(containerD, list[i].SizeZ));
                    thickness = list[i].SizeX;
                }
                else if (axis == 1)
                {
                    layerCount = SafeMul(FitCount(containerW, list[i].SizeX), FitCount(containerD, list[i].SizeZ));
                    thickness = list[i].SizeY;
                }
                else
                {
                    layerCount = SafeMul(FitCount(containerW, list[i].SizeX), FitCount(containerH, list[i].SizeY));
                    thickness = list[i].SizeZ;
                }

                if (best == null ||
                    layerCount > bestLayerCount ||
                    (layerCount == bestLayerCount && thickness < bestThickness - 1e-12))
                {
                    best = list[i];
                    bestLayerCount = layerCount;
                    bestThickness = thickness;
                }
            }

            return best;
        }

        private PlanNode SearchRegion(Region r, int depth)
        {
            if (!CanAnyTypeFit(r))
                return EmptyPlan.Instance;

            PlanNode best = GetShallowGreedyPlan(r);
            long regionUpper = VolumeBoundCount(r.Volume);

            if (best.Count >= regionUpper)
                return best;

            if (depth <= 0 || TimeUp())
                return best;

            List<ActionChoice> actions = new List<ActionChoice>();

            for (int ti = 0; ti < types.Count; ti++)
            {
                OrientationType t = types[ti];

                long mx = FitCount(r.W, t.SizeX);
                long my = FitCount(r.H, t.SizeY);
                long mz = FitCount(r.D, t.SizeZ);

                if (mx <= 0 || my <= 0 || mz <= 0) continue;

                for (int oi = 0; oi < orders.Length; oi++)
                {
                    int[] order = orders[oi];
                    int firstAxis = order[0];
                    long maxFirst = firstAxis == 0 ? mx : (firstAxis == 1 ? my : mz);

                    List<long> options = GenerateAxisOptions(maxFirst, depth);

                    for (int opi = 0; opi < options.Count; opi++)
                    {
                        long nx = mx;
                        long ny = my;
                        long nz = mz;

                        long n0 = options[opi];
                        if (firstAxis == 0) nx = n0;
                        else if (firstAxis == 1) ny = n0;
                        else nz = n0;

                        if (nx <= 0 || ny <= 0 || nz <= 0) continue;

                        double ux = nx * t.SizeX;
                        double uy = ny * t.SizeY;
                        double uz = nz * t.SizeZ;

                        if (ux > r.W + 1e-8 || uy > r.H + 1e-8 || uz > r.D + 1e-8)
                            continue;

                        ActionChoice ac = new ActionChoice();
                        ac.Type = t;
                        ac.Order = order;
                        ac.Nx = nx;
                        ac.Ny = ny;
                        ac.Nz = nz;
                        ac.BaseCount = SafeMul3(nx, ny, nz);
                        ac.BaseUsedVolume = ac.BaseCount * t.BoxVolume;
                        ac.Children = BuildChildren(r, ux, uy, uz, order);

                        long lb = ac.BaseCount;
                        long ub = ac.BaseCount;

                        for (int ci = 0; ci < ac.Children.Length; ci++)
                        {
                            Region child = ac.Children[ci];
                            if (!CanAnyTypeFit(child)) continue;
                            lb = SafeAdd(lb, GetShallowGreedyCount(child));
                            ub = SafeAdd(ub, VolumeBoundCount(child.Volume));
                        }

                        ac.LowerBound = lb;
                        ac.UpperBound = ub;

                        if (ac.UpperBound < best.Count)
                            continue;

                        actions.Add(ac);
                    }
                }

                if (TimeAlmostUp())
                    break;
            }

            if (actions.Count == 0)
                return best;

            actions.Sort(delegate (ActionChoice a, ActionChoice b)
            {
                if (a.LowerBound != b.LowerBound)
                    return b.LowerBound.CompareTo(a.LowerBound);
                if (a.BaseCount != b.BaseCount)
                    return b.BaseCount.CompareTo(a.BaseCount);
                if (a.UpperBound != b.UpperBound)
                    return b.UpperBound.CompareTo(a.UpperBound);
                if (a.BaseUsedVolume > b.BaseUsedVolume + 1e-12) return -1;
                if (a.BaseUsedVolume + 1e-12 < b.BaseUsedVolume) return 1;
                return 0;
            });

            int beam = Math.Min(actions.Count, GetBeamWidth(depth));

            for (int ai = 0; ai < beam; ai++)
            {
                if (TimeUp()) break;

                ActionChoice ac = actions[ai];

                List<PlanNode> parts = new List<PlanNode>(1 + ac.Children.Length);
                FillPlan fill = new FillPlan(ac.Type, ac.Nx, ac.Ny, ac.Nz, r.Ox, r.Oy, r.Oz);
                parts.Add(fill);

                Region[] childList = new Region[ac.Children.Length];
                Array.Copy(ac.Children, childList, ac.Children.Length);
                Array.Sort(childList, delegate (Region a, Region b)
                {
                    double va = a.Volume;
                    double vb = b.Volume;
                    if (va > vb) return -1;
                    if (va < vb) return 1;
                    return 0;
                });

                for (int ci = 0; ci < childList.Length; ci++)
                {
                    Region child = childList[ci];
                    if (!CanAnyTypeFit(child)) continue;
                    PlanNode childPlan = SearchRegion(child, depth - 1);
                    if (childPlan.Count > 0)
                        parts.Add(childPlan);
                }

                PlanNode candidate = CombineNodes(parts);
                if (Better(candidate.Count, candidate.UsedBoxVolume, best.Count, best.UsedBoxVolume))
                {
                    best = candidate;
                    if (best.Count >= regionUpper)
                        return best;
                }
            }

            return best;
        }

        private int GetBeamWidth(int depth)
        {
            int beam = config.BeamWidth;
            int reduction = config.SearchDepth - depth;
            for (int i = 0; i < reduction; i++)
            {
                beam = Math.Max(4, beam / 2);
            }
            return beam;
        }

        private List<long> GenerateAxisOptions(long maxN, int depth)
        {
            List<long> result = new List<long>();
            if (maxN <= 0) return result;

            if (maxN <= config.AxisExhaustiveLimit && depth >= Math.Max(1, config.SearchDepth - 1))
            {
                for (long n = maxN; n >= 1; n--)
                    result.Add(n);
                return result;
            }

            HashSet<long> seen = new HashSet<long>();

            Action<long> add = delegate (long x)
            {
                if (x >= 1 && x <= maxN && seen.Add(x))
                    result.Add(x);
            };

            add(maxN);
            add(maxN - 1);
            add(maxN - 2);
            if (depth >= 2) add(maxN - 3);
            add((3 * maxN + 3) / 4);
            add((maxN + 1) / 2);
            add((maxN + 2) / 3);
            add(1);

            result.Sort(delegate (long a, long b) { return b.CompareTo(a); });
            return result;
        }

        private PlanNode GetBestSingleFillPlan(Region r)
        {
            if (!CanAnyTypeFit(r))
                return EmptyPlan.Instance;

            OrientationType bestType = null;
            long bestCount = 0;
            double bestUsedVol = -1.0;
            long bestNx = 0, bestNy = 0, bestNz = 0;

            for (int i = 0; i < types.Count; i++)
            {
                long nx = FitCount(r.W, types[i].SizeX);
                long ny = FitCount(r.H, types[i].SizeY);
                long nz = FitCount(r.D, types[i].SizeZ);
                if (nx <= 0 || ny <= 0 || nz <= 0) continue;

                long count = SafeMul3(nx, ny, nz);
                double usedVol = count * types[i].BoxVolume;

                if (count > bestCount ||
                    (count == bestCount && usedVol > bestUsedVol + 1e-12))
                {
                    bestCount = count;
                    bestUsedVol = usedVol;
                    bestType = types[i];
                    bestNx = nx;
                    bestNy = ny;
                    bestNz = nz;
                }
            }

            if (bestType == null || bestCount <= 0)
                return EmptyPlan.Instance;

            return new FillPlan(bestType, bestNx, bestNy, bestNz, r.Ox, r.Oy, r.Oz);
        }

        private long GetBestSingleCount(double w, double h, double d)
        {
            if (w <= Tol || h <= Tol || d <= Tol) return 0;

            long best = 0;
            for (int i = 0; i < types.Count; i++)
            {
                long nx = FitCount(w, types[i].SizeX);
                long ny = FitCount(h, types[i].SizeY);
                long nz = FitCount(d, types[i].SizeZ);
                if (nx <= 0 || ny <= 0 || nz <= 0) continue;
                long count = SafeMul3(nx, ny, nz);
                if (count > best) best = count;
            }
            return best;
        }

        private long GetShallowGreedyCount(Region r)
        {
            if (!CanAnyTypeFit(r))
                return 0;

            long best = GetBestSingleCount(r.W, r.H, r.D);
            if (best >= VolumeBoundCount(r.Volume))
                return best;

            for (int ti = 0; ti < types.Count; ti++)
            {
                OrientationType t = types[ti];
                long mx = FitCount(r.W, t.SizeX);
                long my = FitCount(r.H, t.SizeY);
                long mz = FitCount(r.D, t.SizeZ);
                if (mx <= 0 || my <= 0 || mz <= 0) continue;

                long baseCount = SafeMul3(mx, my, mz);
                double ux = mx * t.SizeX;
                double uy = my * t.SizeY;
                double uz = mz * t.SizeZ;

                for (int oi = 0; oi < orders.Length; oi++)
                {
                    Region[] children = BuildChildren(r, ux, uy, uz, orders[oi]);
                    long total = baseCount;

                    for (int ci = 0; ci < children.Length; ci++)
                    {
                        if (!CanAnyTypeFit(children[ci])) continue;
                        total = SafeAdd(total, GetBestSingleCount(children[ci].W, children[ci].H, children[ci].D));
                    }

                    if (total > best)
                        best = total;
                }
            }

            return best;
        }

        private PlanNode GetShallowGreedyPlan(Region r)
        {
            if (!CanAnyTypeFit(r))
                return EmptyPlan.Instance;

            PlanNode best = GetBestSingleFillPlan(r);
            long regionUpper = VolumeBoundCount(r.Volume);
            if (best.Count >= regionUpper)
                return best;

            for (int ti = 0; ti < types.Count; ti++)
            {
                OrientationType t = types[ti];
                long mx = FitCount(r.W, t.SizeX);
                long my = FitCount(r.H, t.SizeY);
                long mz = FitCount(r.D, t.SizeZ);
                if (mx <= 0 || my <= 0 || mz <= 0) continue;

                double ux = mx * t.SizeX;
                double uy = my * t.SizeY;
                double uz = mz * t.SizeZ;
                FillPlan fill = new FillPlan(t, mx, my, mz, r.Ox, r.Oy, r.Oz);

                for (int oi = 0; oi < orders.Length; oi++)
                {
                    List<PlanNode> parts = new List<PlanNode>(4);
                    parts.Add(fill);

                    Region[] children = BuildChildren(r, ux, uy, uz, orders[oi]);
                    for (int ci = 0; ci < children.Length; ci++)
                    {
                        if (!CanAnyTypeFit(children[ci])) continue;
                        PlanNode child = GetBestSingleFillPlan(children[ci]);
                        if (child.Count > 0)
                            parts.Add(child);
                    }

                    PlanNode candidate = CombineNodes(parts);
                    if (Better(candidate.Count, candidate.UsedBoxVolume, best.Count, best.UsedBoxVolume))
                    {
                        best = candidate;
                        if (best.Count >= regionUpper)
                            return best;
                    }
                }
            }

            return best;
        }

        private Region[] BuildChildren(Region r, double ux, double uy, double uz, int[] order)
        {
            double[] rem = new double[] { r.W, r.H, r.D };
            double[] used = new double[] { ux, uy, uz };
            Region[] tmp = new Region[3];
            int cnt = 0;

            for (int i = 0; i < 3; i++)
            {
                int ax = order[i];
                double u = used[ax];
                double leftover = rem[ax] - u;
                if (leftover < 0.0 && leftover > -1e-9) leftover = 0.0;

                if (leftover > 1e-12)
                {
                    double ox = r.Ox;
                    double oy = r.Oy;
                    double oz = r.Oz;
                    double w = rem[0];
                    double h = rem[1];
                    double d = rem[2];

                    if (ax == 0)
                    {
                        ox += u;
                        w = leftover;
                    }
                    else if (ax == 1)
                    {
                        oy += u;
                        h = leftover;
                    }
                    else
                    {
                        oz += u;
                        d = leftover;
                    }

                    tmp[cnt++] = new Region(ox, oy, oz, w, h, d);
                }

                rem[ax] = u;
            }

            Region[] result = new Region[cnt];
            Array.Copy(tmp, result, cnt);
            return result;
        }

        private bool CanAnyTypeFit(Region r)
        {
            if (r.W <= Tol || r.H <= Tol || r.D <= Tol) return false;
            if (r.Volume + 1e-12 < minBoxVolume) return false;

            for (int i = 0; i < types.Count; i++)
            {
                if (types[i].SizeX <= r.W + 1e-9 &&
                    types[i].SizeY <= r.H + 1e-9 &&
                    types[i].SizeZ <= r.D + 1e-9)
                    return true;
            }
            return false;
        }

        private PlanNode CombineNodes(List<PlanNode> parts)
        {
            List<PlanNode> flat = new List<PlanNode>(parts.Count);
            for (int i = 0; i < parts.Count; i++)
            {
                PlanNode p = parts[i];
                if (p == null || p.Count <= 0) continue;

                CompositePlan cp = p as CompositePlan;
                if (cp != null)
                {
                    for (int j = 0; j < cp.Children.Count; j++)
                        flat.Add(cp.Children[j]);
                }
                else
                {
                    flat.Add(p);
                }
            }

            if (flat.Count == 0) return EmptyPlan.Instance;
            if (flat.Count == 1) return flat[0];
            return new CompositePlan(flat);
        }

        private long VolumeBoundCount(double volume)
        {
            if (!(minBoxVolume > 0.0) || volume <= 0.0) return 0;
            double q = volume / minBoxVolume + 1e-9;
            if (q >= long.MaxValue) return long.MaxValue;
            return (long)Math.Ceiling(q);
        }

        private static bool Better(long countA, double volA, long countB, double volB)
        {
            if (countA != countB) return countA > countB;
            return volA > volB + 1e-12;
        }

        private bool TimeUp()
        {
            return watch.ElapsedMilliseconds >= HardTimeLimitMs;
        }

        private bool TimeAlmostUp()
        {
            return watch.ElapsedMilliseconds >= HardTimeLimitMs - 5000;
        }

        private static long FitCount(double dim, double step)
        {
            if (!(dim > 0.0) || !(step > 0.0)) return 0;

            double ratio = dim / step;
            long n = (long)Math.Floor(ratio + 1e-12);
            if (n < 0) n = 0;

            double tol = 1e-9 * Math.Max(1.0, Math.Max(Math.Abs(dim), Math.Abs(step)));

            while (n < long.MaxValue / 2 && (n + 1) * step <= dim + tol) n++;
            while (n > 0 && n * step > dim + tol) n--;

            return n;
        }

        private static int CompareOrientation(OrientationType a, OrientationType b)
        {
            if (a.RootSingleCount != b.RootSingleCount)
                return b.RootSingleCount.CompareTo(a.RootSingleCount);

            if (a.BoxVolume < b.BoxVolume - 1e-12) return -1;
            if (a.BoxVolume > b.BoxVolume + 1e-12) return 1;

            double aMax = Math.Max(a.SizeX, Math.Max(a.SizeY, a.SizeZ));
            double bMax = Math.Max(b.SizeX, Math.Max(b.SizeY, b.SizeZ));
            if (aMax < bMax - 1e-12) return -1;
            if (aMax > bMax + 1e-12) return 1;

            return 0;
        }

        private static Quat RandomQuaternion(Random rng)
        {
            double u1 = rng.NextDouble();
            double u2 = rng.NextDouble();
            double u3 = rng.NextDouble();

            double sqrt1MinusU1 = Math.Sqrt(1.0 - u1);
            double sqrtU1 = Math.Sqrt(u1);
            double theta1 = 2.0 * Math.PI * u2;
            double theta2 = 2.0 * Math.PI * u3;

            double x = sqrt1MinusU1 * Math.Sin(theta1);
            double y = sqrt1MinusU1 * Math.Cos(theta1);
            double z = sqrtU1 * Math.Sin(theta2);
            double w = sqrtU1 * Math.Cos(theta2);

            return new Quat(w, x, y, z).Canonicalized();
        }

        private static Vec3 RandomUnitVector(Random rng)
        {
            double z = 2.0 * rng.NextDouble() - 1.0;
            double t = 2.0 * Math.PI * rng.NextDouble();
            double r = Math.Sqrt(Math.Max(0.0, 1.0 - z * z));
            return new Vec3(r * Math.Cos(t), r * Math.Sin(t), z);
        }

        private static Quat PerturbQuaternion(Quat seed, Random rng, double maxAngle)
        {
            double angle = (rng.NextDouble() * 2.0 - 1.0) * maxAngle;
            Quat p = Quat.FromAxisAngle(RandomUnitVector(rng), angle);
            return (p * seed).Canonicalized();
        }

        private static List<Mat3> BuildCubeRotations()
        {
            int[][] perms = new int[][]
            {
                new int[] { 0, 1, 2 },
                new int[] { 0, 2, 1 },
                new int[] { 1, 0, 2 },
                new int[] { 1, 2, 0 },
                new int[] { 2, 0, 1 },
                new int[] { 2, 1, 0 }
            };

            List<Mat3> list = new List<Mat3>(24);
            int[] signs = new int[] { -1, 1 };

            for (int pi = 0; pi < perms.Length; pi++)
            {
                int parity = PermParity(perms[pi]);

                for (int si = 0; si < signs.Length; si++)
                {
                    for (int sj = 0; sj < signs.Length; sj++)
                    {
                        for (int sk = 0; sk < signs.Length; sk++)
                        {
                            int sx = signs[si];
                            int sy = signs[sj];
                            int sz = signs[sk];

                            if (sx * sy * sz * parity != 1)
                                continue;

                            double m11 = 0, m12 = 0, m13 = 0;
                            double m21 = 0, m22 = 0, m23 = 0;
                            double m31 = 0, m32 = 0, m33 = 0;

                            if (perms[pi][0] == 0) m11 = sx;
                            else if (perms[pi][0] == 1) m12 = sx;
                            else m13 = sx;

                            if (perms[pi][1] == 0) m21 = sy;
                            else if (perms[pi][1] == 1) m22 = sy;
                            else m23 = sy;

                            if (perms[pi][2] == 0) m31 = sz;
                            else if (perms[pi][2] == 1) m32 = sz;
                            else m33 = sz;

                            list.Add(new Mat3(
                                m11, m12, m13,
                                m21, m22, m23,
                                m31, m32, m33
                            ));
                        }
                    }
                }
            }

            return list;
        }

        private static int PermParity(int[] p)
        {
            int inv = 0;
            for (int i = 0; i < p.Length; i++)
                for (int j = i + 1; j < p.Length; j++)
                    if (p[i] > p[j]) inv++;
            return (inv & 1) == 0 ? 1 : -1;
        }
    }
```
