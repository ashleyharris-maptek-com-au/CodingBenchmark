using System;
using System.IO;

public static class Program
{
    const long INF = 1L << 60;
    const long EXACT_BUDGET = 5_000_000L;
    const long HEURISTIC_BUDGET = 20_000_000L;
    const int EXACT_K = 20;
    const int EXACT_K_SOFT = 22;
    const int HEURISTIC_K = 400;

    static int N, M;
    static int[] edgeU = Array.Empty<int>();
    static int[] edgeV = Array.Empty<int>();
    static int[] edgeW = Array.Empty<int>();

    static int[] head = Array.Empty<int>();
    static int[] to = Array.Empty<int>();
    static int[] next = Array.Empty<int>();
    static int[] arcEdge = Array.Empty<int>();
    static int[] degree = Array.Empty<int>();

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n0 = fs.NextInt();
        if (n0 == int.MinValue) return;

        N = n0;
        M = fs.NextInt();
        if (N <= 0)
            return;

        edgeU = new int[M];
        edgeV = new int[M];
        edgeW = new int[M];
        degree = new int[N];

        head = new int[N];
        FillInt(head, -1);

        to = new int[2 * M];
        next = new int[2 * M];
        arcEdge = new int[2 * M];

        bool hasNegative = false;
        int arc = 0;

        for (int i = 0; i < M; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();

            edgeU[i] = u;
            edgeV[i] = v;
            edgeW[i] = w;
            if (w < 0) hasNegative = true;

            degree[u]++;
            degree[v]++;

            to[arc] = v;
            arcEdge[arc] = i;
            next[arc] = head[u];
            head[u] = arc++;
            to[arc] = u;
            arcEdge[arc] = i;
            next[arc] = head[v];
            head[v] = arc++;
        }

        var fw = new FastWriter(Console.OpenStandardOutput());

        if (M == 0)
        {
            fw.WriteInt(0);
            fw.WriteByte((byte)'\n');
            fw.Flush();
            return;
        }

        if (!IsConnectedFromZero())
        {
            // No valid closed walk from 0 exists if edge-containing vertices are disconnected from 0.
            fw.WriteInt(0);
            fw.WriteByte((byte)'\n');
            fw.Flush();
            return;
        }

        int activeVertices = 0;
        for (int i = 0; i < N; i++)
            if (degree[i] > 0)
                activeVertices++;

        int[] dup = new int[M];
        int[] odd = GetOddVertices();

        var heap = new MinHeap(Math.Max(1024, Math.Min(1_048_576, Math.Max(1024, N * 2))));
        long[] workDist = new long[N];
        int[] workParent = new int[N];
        int[] workParentEdge = new int[N];

        bool ok = true;
        if (odd.Length > 0)
        {
            ok = TryAugmentAdaptive(
                odd,
                dup,
                hasNegative,
                activeVertices,
                heap,
                workDist,
                workParent,
                workParentEdge);

            if (!ok || !IsEulerianAfterDup(dup))
            {
                for (int i = 0; i < M; i++) dup[i] = 1; // safe fallback: duplicate every edge once
            }
        }

        int[]? circuit;
        int count;
        bool built = TryBuildEulerCircuit(0, dup, out circuit, out count);

        if (!built || circuit == null || count <= 0 || circuit[count - 1] != 0 || circuit[0] != 0)
        {
            for (int i = 0; i < M; i++) dup[i] = 1;
            built = TryBuildEulerCircuit(0, dup, out circuit, out count);
        }

        if (!built || circuit == null || count <= 0 || circuit[count - 1] != 0 || circuit[0] != 0)
        {
            fw.WriteInt(0);
            fw.WriteByte((byte)'\n');
            fw.Flush();
            return;
        }

        WriteCircuit(fw, circuit, count);
    }

    static bool TryAugmentAdaptive(
        int[] odd,
        int[] dup,
        bool hasNegative,
        int activeVertices,
        MinHeap heap,
        long[] workDist,
        int[] workParent,
        int[] workParentEdge)
    {
        if ((odd.Length & 1) != 0) return false;

        // Exact and fast on trees.
        if (activeVertices > 0 && M == activeVertices - 1)
        {
            Array.Clear(dup, 0, M);
            if (TryTreeParityAugmentation(dup, heap, workDist, workParent, workParentEdge) &&
                IsEulerianAfterDup(dup))
                return true;
        }

        if (!hasNegative)
        {
            long metricBudget = (long)odd.Length * Math.Max(1, M);

            if (odd.Length <= EXACT_K || (odd.Length <= EXACT_K_SOFT && metricBudget <= EXACT_BUDGET))
            {
                Array.Clear(dup, 0, M);
                if (TryExactMatchingAugmentation(odd, dup, heap, workDist, workParent, workParentEdge) &&
                    IsEulerianAfterDup(dup))
                    return true;
            }

            if (odd.Length <= HEURISTIC_K && metricBudget <= HEURISTIC_BUDGET)
            {
                Array.Clear(dup, 0, M);
                if (TryGreedyMetricAugmentation(odd, dup, heap, workDist, workParent, workParentEdge) &&
                    IsEulerianAfterDup(dup))
                    return true;
            }
        }

        Array.Clear(dup, 0, M);
        if (TryTreeParityAugmentation(dup, heap, workDist, workParent, workParentEdge) &&
            IsEulerianAfterDup(dup))
            return true;

        return false;
    }

    static bool TryExactMatchingAugmentation(
        int[] odd,
        int[] dup,
        MinHeap heap,
        long[] workDist,
        int[] workParent,
        int[] workParentEdge)
    {
        int k = odd.Length;
        if ((k & 1) != 0) return false;

        long[] matrix = new long[k * k];
        if (!ComputeOddDistanceMatrix(odd, matrix, heap, workDist))
            return false;

        int[] mate = ExactMetricMatching(matrix, k);
        return AddMatchedPaths(odd, mate, dup, heap, workDist, workParent, workParentEdge);
    }

    static bool TryGreedyMetricAugmentation(
        int[] odd,
        int[] dup,
        MinHeap heap,
        long[] workDist,
        int[] workParent,
        int[] workParentEdge)
    {
        int k = odd.Length;
        if ((k & 1) != 0) return false;

        long[] matrix = new long[k * k];
        if (!ComputeOddDistanceMatrix(odd, matrix, heap, workDist))
            return false;

        int[] mate = GreedyMetricMatching(matrix, k);
        ImproveMatching2Opt(mate, matrix, k);
        return AddMatchedPaths(odd, mate, dup, heap, workDist, workParent, workParentEdge);
    }

    static bool ComputeOddDistanceMatrix(int[] odd, long[] matrix, MinHeap heap, long[] dist)
    {
        int k = odd.Length;
        int[] oddMap = new int[N];
        FillInt(oddMap, -1);
        for (int i = 0; i < k; i++) oddMap[odd[i]] = i;

        for (int si = 0; si < k; si++)
        {
            int rowBase = si * k;
            for (int j = 0; j < k; j++) matrix[rowBase + j] = INF;

            FillLong(dist, INF);
            heap.Clear();

            int src = odd[si];
            dist[src] = 0;
            heap.Push(0, src);

            int found = 0;
            while (heap.Count > 0 && found < k)
            {
                heap.Pop(out long d, out int u);
                if (d != dist[u]) continue;

                int oi = oddMap[u];
                if (oi >= 0 && matrix[rowBase + oi] == INF)
                {
                    matrix[rowBase + oi] = d;
                    found++;
                }

                for (int a = head[u]; a != -1; a = next[a])
                {
                    int e = arcEdge[a];
                    int v = to[a];
                    long nd = d + edgeW[e];
                    if (nd < dist[v])
                    {
                        dist[v] = nd;
                        heap.Push(nd, v);
                    }
                }
            }

            if (found < k) return false;
        }

        return true;
    }

    static int[] ExactMetricMatching(long[] matrix, int k)
    {
        int size = 1 << k;
        long[] memo = new long[size];
        int[] choice = new int[size];
        for (int i = 0; i < size; i++)
        {
            memo[i] = -1;
            choice[i] = -1;
        }

        long Solve(int mask)
        {
            if (mask == 0) return 0;
            long cached = memo[mask];
            if (cached >= 0) return cached;

            int i = 0;
            while (((mask >> i) & 1) == 0) i++;

            int maskWithoutI = mask ^ (1 << i);
            long best = INF;
            int bestJ = -1;

            for (int j = i + 1; j < k; j++)
            {
                if (((maskWithoutI >> j) & 1) == 0) continue;
                long cand = matrix[i * k + j] + Solve(maskWithoutI ^ (1 << j));
                if (cand < best)
                {
                    best = cand;
                    bestJ = j;
                }
            }

            choice[mask] = bestJ;
            memo[mask] = best;
            return best;
        }

        int full = size - 1;
        Solve(full);

        int[] mate = new int[k];
        FillInt(mate, -1);

        int cur = full;
        while (cur != 0)
        {
            int i = 0;
            while (((cur >> i) & 1) == 0) i++;
            int j = choice[cur];
            mate[i] = j;
            mate[j] = i;
            cur ^= (1 << i);
            cur ^= (1 << j);
        }

        return mate;
    }

    struct PairCand : IComparable<PairCand>
    {
        public int A;
        public int B;
        public long D;

        public int CompareTo(PairCand other)
        {
            if (D < other.D) return -1;
            if (D > other.D) return 1;
            if (A < other.A) return -1;
            if (A > other.A) return 1;
            if (B < other.B) return -1;
            if (B > other.B) return 1;
            return 0;
        }
    }

    static int[] GreedyMetricMatching(long[] matrix, int k)
    {
        int pairCount = k * (k - 1) / 2;
        PairCand[] pairs = new PairCand[pairCount];
        int p = 0;
        for (int i = 0; i < k; i++)
        {
            int ib = i * k;
            for (int j = i + 1; j < k; j++)
            {
                pairs[p++] = new PairCand { A = i, B = j, D = matrix[ib + j] };
            }
        }

        Array.Sort(pairs);

        int[] mate = new int[k];
        FillInt(mate, -1);

        for (int i = 0; i < pairs.Length; i++)
        {
            int a = pairs[i].A;
            int b = pairs[i].B;
            if (mate[a] < 0 && mate[b] < 0)
            {
                mate[a] = b;
                mate[b] = a;
            }
        }

        return mate;
    }

    static void ImproveMatching2Opt(int[] mate, long[] matrix, int k)
    {
        if (k < 4) return;

        int pairCount = k / 2;
        int[] pa = new int[pairCount];
        int[] pb = new int[pairCount];

        for (int pass = 0; pass < 8; pass++)
        {
            int pc = 0;
            for (int i = 0; i < k; i++)
            {
                if (i < mate[i])
                {
                    pa[pc] = i;
                    pb[pc] = mate[i];
                    pc++;
                }
            }

            bool improved = false;

            for (int i = 0; i < pc; i++)
            {
                int a = pa[i];
                int b = pb[i];
                long curAB = matrix[a * k + b];

                for (int j = i + 1; j < pc; j++)
                {
                    int c = pa[j];
                    int d = pb[j];
                    long cur = curAB + matrix[c * k + d];

                    long alt1 = matrix[a * k + c] + matrix[b * k + d];
                    if (alt1 < cur)
                    {
                        mate[a] = c; mate[c] = a;
                        mate[b] = d; mate[d] = b;
                        improved = true;
                        goto NextPass;
                    }

                    long alt2 = matrix[a * k + d] + matrix[b * k + c];
                    if (alt2 < cur)
                    {
                        mate[a] = d; mate[d] = a;
                        mate[b] = c; mate[c] = b;
                        improved = true;
                        goto NextPass;
                    }
                }
            }

        NextPass:
            if (!improved) break;
        }
    }

    static bool AddMatchedPaths(
        int[] odd,
        int[] mate,
        int[] dup,
        MinHeap heap,
        long[] dist,
        int[] parent,
        int[] parentEdge)
    {
        int k = odd.Length;
        for (int i = 0; i < k; i++)
        {
            int j = mate[i];
            if (j < 0) return false;
            if (i < j)
            {
                if (!AddShortestPathDuplicates(odd[i], odd[j], dup, heap, dist, parent, parentEdge))
                    return false;
            }
        }
        return true;
    }

    static bool AddShortestPathDuplicates(
        int src,
        int dst,
        int[] dup,
        MinHeap heap,
        long[] dist,
        int[] parent,
        int[] parentEdge)
    {
        if (src == dst) return true;

        FillLong(dist, INF);
        heap.Clear();

        dist[src] = 0;
        parent[src] = -1;
        parentEdge[src] = -1;
        heap.Push(0, src);

        while (heap.Count > 0)
        {
            heap.Pop(out long d, out int u);
            if (d != dist[u]) continue;
            if (u == dst) break;

            for (int a = head[u]; a != -1; a = next[a])
            {
                int e = arcEdge[a];
                int v = to[a];
                long nd = d + edgeW[e];
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parent[v] = u;
                    parentEdge[v] = e;
                    heap.Push(nd, v);
                }
            }
        }

        if (dist[dst] >= INF) return false;

        int cur = dst;
        while (cur != src)
        {
            int e = parentEdge[cur];
            if (e < 0) return false;
            dup[e]++;
            cur = parent[cur];
        }

        return true;
    }

    static bool TryTreeParityAugmentation(
        int[] dup,
        MinHeap heap,
        long[] best,
        int[] parent,
        int[] parentEdge)
    {
        FillLong(best, INF);
        FillInt(parent, -1);
        FillInt(parentEdge, -1);

        bool[] inTree = new bool[N];
        int[] order = new int[N];
        int count = 0;

        heap.Clear();
        best[0] = 0;
        heap.Push(0, 0);

        while (heap.Count > 0)
        {
            heap.Pop(out long key, out int u);
            if (inTree[u] || key != best[u]) continue;

            inTree[u] = true;
            order[count++] = u;

            for (int a = head[u]; a != -1; a = next[a])
            {
                int e = arcEdge[a];
                int v = to[a];
                long w = edgeW[e];
                if (!inTree[v] && w < best[v])
                {
                    best[v] = w;
                    parent[v] = u;
                    parentEdge[v] = e;
                    heap.Push(w, v);
                }
            }
        }

        for (int i = 0; i < N; i++)
            if (degree[i] > 0 && !inTree[i])
                return false;

        int[] parity = new int[N];
        for (int i = 0; i < N; i++)
            parity[i] = degree[i] & 1;

        for (int idx = count - 1; idx >= 0; idx--)
        {
            int v = order[idx];
            if (v == 0) continue;

            if ((parity[v] & 1) != 0)
            {
                int e = parentEdge[v];
                int p = parent[v];
                if (e < 0 || p < 0) return false;

                dup[e]++;
                parity[v] ^= 1;
                parity[p] ^= 1;
            }
        }

        return (parity[0] & 1) == 0;
    }

    static bool IsEulerianAfterDup(int[] dup)
    {
        int[] parity = new int[N];
        for (int i = 0; i < N; i++)
            parity[i] = degree[i] & 1;

        for (int e = 0; e < M; e++)
        {
            if ((dup[e] & 1) != 0)
            {
                parity[edgeU[e]] ^= 1;
                parity[edgeV[e]] ^= 1;
            }
        }

        for (int i = 0; i < N; i++)
            if ((parity[i] & 1) != 0)
                return false;

        return true;
    }

    static bool TryBuildEulerCircuit(int start, int[] dup, out int[]? circuit, out int count)
    {
        long totalEdges = 0;
        for (int e = 0; e < M; e++)
            totalEdges += 1L + dup[e];

        long totalNodesL = totalEdges + 1;
        if (totalNodesL <= 0 || totalNodesL > int.MaxValue)
        {
            circuit = null;
            count = 0;
            return false;
        }

        int totalNodes = (int)totalNodesL;
        int[] rem = new int[M];
        for (int e = 0; e < M; e++)
            rem[e] = dup[e] + 1;

        int[] ptr = new int[N];
        Array.Copy(head, ptr, N);

        int[] stack = new int[totalNodes];
        circuit = new int[totalNodes];

        int sp = 0;
        int cp = 0;
        stack[sp++] = start;

        while (sp > 0)
        {
            int u = stack[sp - 1];
            int a = ptr[u];
            while (a != -1 && rem[arcEdge[a]] == 0)
                a = next[a];
            ptr[u] = a;

            if (a == -1)
            {
                circuit[cp++] = u;
                sp--;
            }
            else
            {
                int e = arcEdge[a];
                rem[e]--;
                stack[sp++] = to[a];
            }
        }

        count = cp;
        if (cp != totalNodes)
        {
            circuit = null;
            return false;
        }

        return true;
    }

    static bool IsConnectedFromZero()
    {
        bool[] vis = new bool[N];
        int[] stack = new int[Math.Max(1, N)];
        int sp = 0;

        vis[0] = true;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int u = stack[--sp];
            for (int a = head[u]; a != -1; a = next[a])
            {
                int v = to[a];
                if (!vis[v])
                {
                    vis[v] = true;
                    stack[sp++] = v;
                }
            }
        }

        for (int i = 0; i < N; i++)
            if (degree[i] > 0 && !vis[i])
                return false;

        return true;
    }

    static int[] GetOddVertices()
    {
        int cnt = 0;
        for (int i = 0; i < N; i++)
            if ((degree[i] & 1) != 0)
                cnt++;

        int[] odd = new int[cnt];
        int p = 0;
        for (int i = 0; i < N; i++)
            if ((degree[i] & 1) != 0)
                odd[p++] = i;

        return odd;
    }

    static void WriteCircuit(FastWriter fw, int[] reversedCircuit, int count)
    {
        fw.WriteInt(reversedCircuit[count - 1]);
        for (int i = count - 2; i >= 0; i--)
        {
            fw.WriteByte((byte)' ');
            fw.WriteInt(reversedCircuit[i]);
        }
        fw.WriteByte((byte)'\n');
        fw.Flush();
    }

    static void FillInt(int[] arr, int value)
    {
        for (int i = 0; i < arr.Length; i++) arr[i] = value;
    }

    static void FillLong(long[] arr, long value)
    {
        for (int i = 0; i < arr.Length; i++) arr[i] = value;
    }

    sealed class MinHeap
    {
        long[] keys;
        int[] vals;
        int size;

        public int Count => size;

        public MinHeap(int capacity)
        {
            if (capacity < 4) capacity = 4;
            keys = new long[capacity];
            vals = new int[capacity];
            size = 0;
        }

        public void Clear()
        {
            size = 0;
        }

        public void Push(long key, int val)
        {
            if (size == keys.Length) Grow();
            int i = size++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (keys[p] <= key) break;
                keys[i] = keys[p];
                vals[i] = vals[p];
                i = p;
            }
            keys[i] = key;
            vals[i] = val;
        }

        public void Pop(out long key, out int val)
        {
            key = keys[0];
            val = vals[0];

            int n = --size;
            if (n == 0) return;

            long lastKey = keys[n];
            int lastVal = vals[n];

            int i = 0;
            while (true)
            {
                int l = (i << 1) + 1;
                if (l >= n) break;
                int r = l + 1;
                int c = (r < n && keys[r] < keys[l]) ? r : l;
                if (keys[c] >= lastKey) break;
                keys[i] = keys[c];
                vals[i] = vals[c];
                i = c;
            }

            keys[i] = lastKey;
            vals[i] = lastVal;
        }

        void Grow()
        {
            int n = keys.Length << 1;
            Array.Resize(ref keys, n);
            Array.Resize(ref vals, n);
        }
    }

    sealed class FastScanner
    {
        readonly Stream stream;
        readonly byte[] buffer;
        int len, ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            this.stream = stream;
            buffer = new byte[bufferSize];
            len = ptr = 0;
        }

        int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = Read();
            }

            return x * sign;
        }
    }

    sealed class FastWriter
    {
        readonly Stream stream;
        readonly byte[] buffer;
        readonly byte[] numBuf;
        int ptr;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            this.stream = stream;
            buffer = new byte[bufferSize];
            numBuf = new byte[32];
            ptr = 0;
        }

        public void WriteByte(byte b)
        {
            if (ptr == buffer.Length) FlushBuffer();
            buffer[ptr++] = b;
        }

        public void WriteInt(long x)
        {
            if (x == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            if (x < 0)
            {
                WriteByte((byte)'-');
                x = -x;
            }

            int n = 0;
            while (x > 0)
            {
                numBuf[n++] = (byte)('0' + (x % 10));
                x /= 10;
            }

            for (int i = n - 1; i >= 0; i--)
                WriteByte(numBuf[i]);
        }

        void FlushBuffer()
        {
            if (ptr > 0)
            {
                stream.Write(buffer, 0, ptr);
                ptr = 0;
            }
        }

        public void Flush()
        {
            FlushBuffer();
            stream.Flush();
        }
    }
