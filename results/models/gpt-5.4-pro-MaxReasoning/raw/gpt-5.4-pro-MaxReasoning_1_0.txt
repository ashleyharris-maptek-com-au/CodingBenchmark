#include <algorithm>
#include <cmath>
#include <cstdint>

namespace {

using u32 = std::uint32_t;
using u64 = std::uint64_t;

static constexpr double INF_D = 1e100;

// ---------- Fast I/O ----------

class FastScanner {
public:
    FastScanner() : buf(BUF_SIZE), idx(0), size(0) {}

    bool readUInt64(u64& out) {
        char c;
        if (!skipBlanks(c)) return false;
        if (c == '+') c = getChar();
        u64 val = 0;
        while (c >= '0' && c <= '9') {
            val = val * 10 + static_cast<u64>(c - '0');
            c = getChar();
        }
        out = val;
        return true;
    }

    bool readDouble(double& out) {
        char c;
        if (!skipBlanks(c)) return false;

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        double val = 0.0;
        while (c >= '0' && c <= '9') {
            val = val * 10.0 + static_cast<double>(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double place = 0.1;
            c = getChar();
            while (c >= '0' && c <= '9') {
                val += static_cast<double>(c - '0') * place;
                place *= 0.1;
                c = getChar();
            }
        }

        if (c == 'e' || c == 'E') {
            int expSign = 1;
            int expVal = 0;
            c = getChar();
            if (c == '-') {
                expSign = -1;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            while (c >= '0' && c <= '9') {
                expVal = expVal * 10 + (c - '0');
                c = getChar();
            }
            val *= std::pow(10.0, expSign * expVal);
        }

        out = static_cast<double>(sign) * val;
        return true;
    }

private:
    static constexpr std::size_t BUF_SIZE = 1u << 20;
    std::vector<char> buf;
    std::size_t idx;
    std::size_t size;

    inline char getChar() {
        if (idx >= size) {
            size = std::fread(buf.data(), 1, buf.size(), stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    inline bool skipBlanks(char& c) {
        do {
            c = getChar();
            if (c == 0) return false;
        } while (c <= ' ');
        return true;
    }
};

class FastOutput {
public:
    FastOutput() : buf(BUF_SIZE), idx(0) {}
    ~FastOutput() { flush(); }

    void pushChar(char c) {
        if (idx == buf.size()) flush();
        buf[idx++] = c;
    }

    void writeUInt(u64 x) {
        char tmp[32];
        int n = 0;
        do {
            tmp[n++] = static_cast<char>('0' + (x % 10));
            x /= 10;
        } while (x);
        while (n--) pushChar(tmp[n]);
    }

    void flush() {
        if (idx) {
            std::fwrite(buf.data(), 1, idx, stdout);
            idx = 0;
        }
    }

private:
    static constexpr std::size_t BUF_SIZE = 1u << 20;
    std::vector<char> buf;
    std::size_t idx;
};

// ---------- Geometry helpers ----------

inline double euclid(const float* xs, const float* ys, u32 a, u32 b) {
    const double dx = static_cast<double>(xs[a]) - static_cast<double>(xs[b]);
    const double dy = static_cast<double>(ys[a]) - static_cast<double>(ys[b]);
    return std::sqrt(dx * dx + dy * dy);
}

inline double dist_sq(const float* xs, const float* ys, u32 a, u32 b) {
    const double dx = static_cast<double>(xs[a]) - static_cast<double>(xs[b]);
    const double dy = static_cast<double>(ys[a]) - static_cast<double>(ys[b]);
    return dx * dx + dy * dy;
}

// ---------- Exact Held-Karp for tiny instances ----------

std::vector<u32> solve_exact_held_karp(const std::vector<float>& xs, const std::vector<float>& ys) {
    const std::size_t n = xs.size();
    if (n == 0) return {};
    if (n == 1) return {0u};

    std::vector<double> dist(n * n, 0.0);
    const float* px = xs.data();
    const float* py = ys.data();
    for (std::size_t i = 0; i < n; ++i) {
        for (std::size_t j = i + 1; j < n; ++j) {
            const double d = euclid(px, py, static_cast<u32>(i), static_cast<u32>(j));
            dist[i * n + j] = d;
            dist[j * n + i] = d;
        }
    }

    const int m = static_cast<int>(n) - 1; // cities 1..n-1
    const u32 totalMasks = 1u << m;
    const std::size_t dpSize = static_cast<std::size_t>(totalMasks) * static_cast<std::size_t>(m);

    std::vector<double> dp(dpSize, INF_D);
    std::vector<std::int16_t> parent(dpSize, static_cast<std::int16_t>(-1));

    auto idx = [m](u32 mask, int last) -> std::size_t {
        return static_cast<std::size_t>(mask) * static_cast<std::size_t>(m) + static_cast<std::size_t>(last);
    };

    for (int j = 0; j < m; ++j) {
        dp[idx(1u << j, j)] = dist[j + 1];
    }

    for (u32 mask = 1; mask < totalMasks; ++mask) {
        for (int j = 0; j < m; ++j) {
            const u32 bit = 1u << j;
            if ((mask & bit) == 0) continue;

            const u32 prevMask = mask ^ bit;
            if (prevMask == 0) continue;

            double best = INF_D;
            std::int16_t bestPrev = -1;
            for (int k = 0; k < m; ++k) {
                if ((prevMask & (1u << k)) == 0) continue;
                const double cand = dp[idx(prevMask, k)] + dist[static_cast<std::size_t>(k + 1) * n + static_cast<std::size_t>(j + 1)];
                if (cand < best) {
                    best = cand;
                    bestPrev = static_cast<std::int16_t>(k);
                }
            }
            dp[idx(mask, j)] = best;
            parent[idx(mask, j)] = bestPrev;
        }
    }

    const u32 fullMask = totalMasks - 1;
    double bestTour = INF_D;
    int last = -1;
    for (int j = 0; j < m; ++j) {
        const double cand = dp[idx(fullMask, j)] + dist[static_cast<std::size_t>(j + 1) * n];
        if (cand < bestTour) {
            bestTour = cand;
            last = j;
        }
    }

    std::vector<u32> rev;
    rev.reserve(n - 1);
    u32 mask = fullMask;
    while (last != -1) {
        rev.push_back(static_cast<u32>(last + 1));
        const int p = parent[idx(mask, last)];
        mask ^= (1u << last);
        last = p;
    }

    std::reverse(rev.begin(), rev.end());
    std::vector<u32> tour;
    tour.reserve(n);
    tour.push_back(0u);
    tour.insert(tour.end(), rev.begin(), rev.end());
    return tour;
}

// ---------- Small/medium: multi-start NN + full 2-opt ----------

std::vector<float> build_distance_matrix(const std::vector<float>& xs, const std::vector<float>& ys) {
    const std::size_t n = xs.size();
    std::vector<float> d(n * n, 0.0f);
    const float* px = xs.data();
    const float* py = ys.data();

    for (std::size_t i = 0; i < n; ++i) {
        for (std::size_t j = i + 1; j < n; ++j) {
            const float val = static_cast<float>(euclid(px, py, static_cast<u32>(i), static_cast<u32>(j)));
            d[i * n + j] = val;
            d[j * n + i] = val;
        }
    }
    return d;
}

inline float dmat(const std::vector<float>& d, std::size_t n, u32 a, u32 b) {
    return d[static_cast<std::size_t>(a) * n + static_cast<std::size_t>(b)];
}

std::vector<u32> build_nearest_neighbor_from_start(u32 start, const std::vector<float>& d, std::size_t n) {
    std::vector<u32> tour;
    tour.reserve(n);
    std::vector<std::uint8_t> used(n, 0);

    u32 cur = start;
    used[cur] = 1;
    tour.push_back(cur);

    for (std::size_t step = 1; step < n; ++step) {
        float best = std::numeric_limits<float>::infinity();
        u32 bestCity = 0;

        for (u32 j = 0; j < static_cast<u32>(n); ++j) {
            if (used[j]) continue;
            const float cand = dmat(d, n, cur, j);
            if (cand < best) {
                best = cand;
                bestCity = j;
            }
        }

        used[bestCity] = 1;
        tour.push_back(bestCity);
        cur = bestCity;
    }

    return tour;
}

double cycle_length_matrix(const std::vector<u32>& tour, const std::vector<float>& d) {
    const std::size_t n = tour.size();
    if (n < 2) return 0.0;

    double sum = 0.0;
    for (std::size_t i = 0; i < n; ++i) {
        sum += static_cast<double>(d[static_cast<std::size_t>(tour[i]) * n + static_cast<std::size_t>(tour[(i + 1) % n])]);
    }
    return sum;
}

void full_two_opt_cycle(std::vector<u32>& tour, const std::vector<float>& d) {
    const std::size_t n = tour.size();
    if (n < 4) return;

    constexpr double EPS = 1e-6;

    for (int pass = 0; pass < 30; ++pass) {
        bool any = false;

        for (std::size_t i = 0; i + 1 < n; ++i) {
            bool changed = false;
            const u32 a = tour[i];
            const u32 b = tour[i + 1];

            for (std::size_t j = i + 2; j < n; ++j) {
                if (i == 0 && j == n - 1) continue; // adjacent in cycle

                const u32 c = tour[j];
                const u32 dd = tour[(j + 1 == n) ? 0 : (j + 1)];

                const double delta =
                    static_cast<double>(d[static_cast<std::size_t>(a) * n + static_cast<std::size_t>(c)]) +
                    static_cast<double>(d[static_cast<std::size_t>(b) * n + static_cast<std::size_t>(dd)]) -
                    static_cast<double>(d[static_cast<std::size_t>(a) * n + static_cast<std::size_t>(b)]) -
                    static_cast<double>(d[static_cast<std::size_t>(c) * n + static_cast<std::size_t>(dd)]);

                if (delta < -EPS) {
                    std::reverse(tour.begin() + static_cast<std::ptrdiff_t>(i + 1),
                                 tour.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    any = true;
                    changed = true;
                    break;
                }
            }

            if (changed && i > 0) --i;
        }

        if (!any) break;
    }
}

std::vector<u32> solve_small_multistart_nn_2opt(const std::vector<float>& xs, const std::vector<float>& ys) {
    const std::size_t n = xs.size();
    if (n == 0) return {};
    if (n == 1) return {0u};

    const std::vector<float> d = build_distance_matrix(xs, ys);

    u32 minX = 0, maxX = 0, minY = 0, maxY = 0, far0 = 0, nearCenter = 0;
    double sumX = 0.0, sumY = 0.0;
    const float* px = xs.data();
    const float* py = ys.data();

    for (u32 i = 0; i < static_cast<u32>(n); ++i) {
        if (px[i] < px[minX]) minX = i;
        if (px[i] > px[maxX]) maxX = i;
        if (py[i] < py[minY]) minY = i;
        if (py[i] > py[maxY]) maxY = i;
        sumX += static_cast<double>(px[i]);
        sumY += static_cast<double>(py[i]);
    }

    double bestFar = -1.0;
    for (u32 i = 0; i < static_cast<u32>(n); ++i) {
        const double ds = dist_sq(px, py, 0u, i);
        if (ds > bestFar) {
            bestFar = ds;
            far0 = i;
        }
    }

    const double cx = sumX / static_cast<double>(n);
    const double cy = sumY / static_cast<double>(n);
    double bestNearCenter = INF_D;
    for (u32 i = 0; i < static_cast<u32>(n); ++i) {
        const double dx = static_cast<double>(px[i]) - cx;
        const double dy = static_cast<double>(py[i]) - cy;
        const double ds = dx * dx + dy * dy;
        if (ds < bestNearCenter) {
            bestNearCenter = ds;
            nearCenter = i;
        }
    }

    std::vector<u32> starts;
    auto addStart = [&starts](u32 s) {
        for (u32 v : starts) if (v == s) return;
        starts.push_back(s);
    };

    addStart(0u);
    addStart(minX);
    addStart(maxX);
    addStart(minY);
    addStart(maxY);
    addStart(far0);
    addStart(nearCenter);

    double bestLen = INF_D;
    std::vector<u32> bestTour;

    for (u32 s : starts) {
        std::vector<u32> tour = build_nearest_neighbor_from_start(s, d, n);
        const double len = cycle_length_matrix(tour, d);
        if (len < bestLen) {
            bestLen = len;
            bestTour = std::move(tour);
        }
    }

    full_two_opt_cycle(bestTour, d);
    return bestTour;
}

// ---------- Morton order via radix sort ----------

inline u64 part1by1(u32 x) {
    u64 v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFULL;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFULL;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0FULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

inline u64 morton2D(u32 x, u32 y) {
    return part1by1(x) | (part1by1(y) << 1);
}

inline u32 quantize32(float v, double minV, double scale) {
    if (!(scale > 0.0)) return 0u;
    double t = (static_cast<double>(v) - minV) * scale;
    if (t <= 0.0) return 0u;
    if (t >= 4294967295.0) return 0xFFFFFFFFu;
    return static_cast<u32>(t + 0.5);
}

std::vector<u32> morton_radix_order(
    const std::vector<float>& xs,
    const std::vector<float>& ys,
    double minX, double maxX,
    double minY, double maxY
) {
    const std::size_t n = xs.size();
    std::vector<u64> keys(n), tmpKeys(n);
    std::vector<u32> order(n), tmpOrder(n);

    const double scaleX = (maxX > minX) ? (4294967295.0 / (maxX - minX)) : 0.0;
    const double scaleY = (maxY > minY) ? (4294967295.0 / (maxY - minY)) : 0.0;

    for (u32 i = 0; i < static_cast<u32>(n); ++i) {
        const u32 qx = quantize32(xs[i], minX, scaleX);
        const u32 qy = quantize32(ys[i], minY, scaleY);
        keys[i] = morton2D(qx, qy);
        order[i] = i;
    }

    std::vector<std::size_t> count(1u << 16);

    for (int pass = 0; pass < 4; ++pass) {
        std::fill(count.begin(), count.end(), 0);
        const int shift = pass * 16;

        for (std::size_t i = 0; i < n; ++i) {
            ++count[static_cast<std::size_t>((keys[i] >> shift) & 0xFFFFULL)];
        }

        std::size_t sum = 0;
        for (std::size_t i = 0; i < count.size(); ++i) {
            const std::size_t c = count[i];
            count[i] = sum;
            sum += c;
        }

        for (std::size_t i = 0; i < n; ++i) {
            const std::size_t digit = static_cast<std::size_t>((keys[i] >> shift) & 0xFFFFULL);
            const std::size_t pos = count[digit]++;
            tmpKeys[pos] = keys[i];
            tmpOrder[pos] = order[i];
        }

        keys.swap(tmpKeys);
        order.swap(tmpOrder);
    }

    return order;
}

// ---------- Restricted 2-opt / closure fix for larger instances ----------

bool improve_closure_edge_once(std::vector<u32>& tour, const std::vector<float>& xs, const std::vector<float>& ys) {
    const std::size_t n = tour.size();
    if (n < 4) return false;

    const float* px = xs.data();
    const float* py = ys.data();

    const u32 first = tour.front();
    const u32 last = tour.back();
    const double baseClosure = euclid(px, py, last, first);

    double bestDelta = 0.0;
    std::size_t bestK = n;

    for (std::size_t k = 0; k + 1 < n; ++k) {
        const u32 a = tour[k];
        const u32 b = tour[k + 1];

        const double delta =
            euclid(px, py, a, last) +
            euclid(px, py, b, first) -
            euclid(px, py, a, b) -
            baseClosure;

        if (delta < bestDelta) {
            bestDelta = delta;
            bestK = k;
        }
    }

    if (bestK < n) {
        std::reverse(tour.begin() + static_cast<std::ptrdiff_t>(bestK + 1), tour.end());
        return true;
    }
    return false;
}

void window_two_opt_linear(std::vector<u32>& tour,
                           const std::vector<float>& xs,
                           const std::vector<float>& ys,
                           std::size_t window,
                           int passes) {
    const std::size_t n = tour.size();
    if (n < 4) return;

    const float* px = xs.data();
    const float* py = ys.data();
    constexpr double EPS = 1e-12;

    improve_closure_edge_once(tour, xs, ys);

    for (int pass = 0; pass < passes; ++pass) {
        bool any = false;

        for (std::size_t i = 0; i + 3 < n; ++i) {
            bool changed = false;
            std::size_t maxJ = i + window;
            if (maxJ > n - 2) maxJ = n - 2;

            const u32 a = tour[i];
            const u32 b = tour[i + 1];

            for (std::size_t j = i + 2; j <= maxJ; ++j) {
                const u32 c = tour[j];
                const u32 d = tour[j + 1];

                const double delta =
                    euclid(px, py, a, c) +
                    euclid(px, py, b, d) -
                    euclid(px, py, a, b) -
                    euclid(px, py, c, d);

                if (delta < -EPS) {
                    std::reverse(tour.begin() + static_cast<std::ptrdiff_t>(i + 1),
                                 tour.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    any = true;
                    changed = true;
                    break;
                }
            }

            if (changed && i > 0) --i;
        }

        const bool closeImproved = improve_closure_edge_once(tour, xs, ys);
        if (!any && !closeImproved) break;
    }
}

// ---------- Coarse grid snake fallback for very large / low-memory cases ----------

inline u32 quantizeGrid(float v, double minV, double scale, u32 maxVal) {
    if (maxVal == 0 || !(scale > 0.0)) return 0u;
    double t = (static_cast<double>(v) - minV) * scale;
    if (t <= 0.0) return 0u;
    if (t >= static_cast<double>(maxVal)) return maxVal;
    return static_cast<u32>(t);
}

std::vector<u32> grid_snake_order(
    const std::vector<float>& xs,
    const std::vector<float>& ys,
    double minX, double maxX,
    double minY, double maxY
) {
    const std::size_t n = xs.size();
    if (n == 0) return {};

    const std::size_t targetBucket = 32;
    std::size_t targetCells = n / targetBucket;
    if (targetCells == 0) targetCells = 1;

    double aspect = 1.0;
    if (maxX > minX && maxY > minY) {
        aspect = (maxX - minX) / (maxY - minY);
        if (!(aspect > 0.0)) aspect = 1.0;
    }

    const u32 maxSide = 2048;
    u32 gx = static_cast<u32>(std::sqrt(static_cast<double>(targetCells) * aspect));
    if (gx < 1) gx = 1;
    if (gx > maxSide) gx = maxSide;

    u32 gy = static_cast<u32>((targetCells + gx - 1) / gx);
    if (gy < 1) gy = 1;
    if (gy > maxSide) gy = maxSide;

    const std::size_t cells = static_cast<std::size_t>(gx) * static_cast<std::size_t>(gy);
    std::vector<std::size_t> start(cells + 1, 0);

    const double scaleGX = (gx > 1 && maxX > minX) ? (static_cast<double>(gx - 1) / (maxX - minX)) : 0.0;
    const double scaleGY = (gy > 1 && maxY > minY) ? (static_cast<double>(gy - 1) / (maxY - minY)) : 0.0;

    auto cellId = [&](u32 idx) -> std::size_t {
        const u32 cx = quantizeGrid(xs[idx], minX, scaleGX, gx > 0 ? gx - 1 : 0);
        const u32 cy = quantizeGrid(ys[idx], minY, scaleGY, gy > 0 ? gy - 1 : 0);
        return static_cast<std::size_t>(cy) * static_cast<std::size_t>(gx) + static_cast<std::size_t>(cx);
    };

    for (u32 i = 0; i < static_cast<u32>(n); ++i) {
        ++start[cellId(i) + 1];
    }
    for (std::size_t i = 1; i <= cells; ++i) {
        start[i] += start[i - 1];
    }

    std::vector<std::size_t> next = start;
    std::vector<u32> grouped(n);
    for (u32 i = 0; i < static_cast<u32>(n); ++i) {
        const std::size_t id = cellId(i);
        grouped[next[id]++] = i;
    }

    const bool snakeRows = (gx >= gy);

    for (u32 cy = 0; cy < gy; ++cy) {
        for (u32 cx = 0; cx < gx; ++cx) {
            const std::size_t id = static_cast<std::size_t>(cy) * static_cast<std::size_t>(gx) + static_cast<std::size_t>(cx);
            const std::size_t l = start[id];
            const std::size_t r = start[id + 1];
            if (r - l <= 1) continue;

            const bool asc = (((cx + cy) & 1u) == 0u);

            if (snakeRows) {
                if (asc) {
                    std::sort(grouped.begin() + static_cast<std::ptrdiff_t>(l),
                              grouped.begin() + static_cast<std::ptrdiff_t>(r),
                              [&](u32 a, u32 b) {
                                  if (ys[a] != ys[b]) return ys[a] < ys[b];
                                  if (xs[a] != xs[b]) return xs[a] < xs[b];
                                  return a < b;
                              });
                } else {
                    std::sort(grouped.begin() + static_cast<std::ptrdiff_t>(l),
                              grouped.begin() + static_cast<std::ptrdiff_t>(r),
                              [&](u32 a, u32 b) {
                                  if (ys[a] != ys[b]) return ys[a] > ys[b];
                                  if (xs[a] != xs[b]) return xs[a] > xs[b];
                                  return a < b;
                              });
                }
            } else {
                if (asc) {
                    std::sort(grouped.begin() + static_cast<std::ptrdiff_t>(l),
                              grouped.begin() + static_cast<std::ptrdiff_t>(r),
                              [&](u32 a, u32 b) {
                                  if (xs[a] != xs[b]) return xs[a] < xs[b];
                                  if (ys[a] != ys[b]) return ys[a] < ys[b];
                                  return a < b;
                              });
                } else {
                    std::sort(grouped.begin() + static_cast<std::ptrdiff_t>(l),
                              grouped.begin() + static_cast<std::ptrdiff_t>(r),
                              [&](u32 a, u32 b) {
                                  if (xs[a] != xs[b]) return xs[a] > xs[b];
                                  if (ys[a] != ys[b]) return ys[a] > ys[b];
                                  return a < b;
                              });
                }
            }
        }
    }

    std::vector<u32> route;
    route.reserve(n);

    if (snakeRows) {
        for (u32 cy = 0; cy < gy; ++cy) {
            if ((cy & 1u) == 0u) {
                for (u32 cx = 0; cx < gx; ++cx) {
                    const std::size_t id = static_cast<std::size_t>(cy) * static_cast<std::size_t>(gx) + static_cast<std::size_t>(cx);
                    for (std::size_t p = start[id]; p < start[id + 1]; ++p) route.push_back(grouped[p]);
                }
            } else {
                for (u32 cx = gx; cx-- > 0;) {
                    const std::size_t id = static_cast<std::size_t>(cy) * static_cast<std::size_t>(gx) + static_cast<std::size_t>(cx);
                    for (std::size_t p = start[id]; p < start[id + 1]; ++p) route.push_back(grouped[p]);
                }
            }
        }
    } else {
        for (u32 cx = 0; cx < gx; ++cx) {
            if ((cx & 1u) == 0u) {
                for (u32 cy = 0; cy < gy; ++cy) {
                    const std::size_t id = static_cast<std::size_t>(cy) * static_cast<std::size_t>(gx) + static_cast<std::size_t>(cx);
                    for (std::size_t p = start[id]; p < start[id + 1]; ++p) route.push_back(grouped[p]);
                }
            } else {
                for (u32 cy = gy; cy-- > 0;) {
                    const std::size_t id = static_cast<std::size_t>(cy) * static_cast<std::size_t>(gx) + static_cast<std::size_t>(cx);
                    for (std::size_t p = start[id]; p < start[id + 1]; ++p) route.push_back(grouped[p]);
                }
            }
        }
    }

    return route;
}

// ---------- Output helpers ----------

void output_identity_u64(u64 n, FastOutput& out) {
    if (n == 0) {
        out.pushChar('\n');
        return;
    }
    out.writeUInt(0);
    for (u64 i = 1; i < n; ++i) {
        out.pushChar(' ');
        out.writeUInt(i);
    }
    out.pushChar('\n');
}

void output_tour_starting_at_zero(const std::vector<u32>& tour, FastOutput& out) {
    const std::size_t n = tour.size();
    if (n == 0) {
        out.pushChar('\n');
        return;
    }

    std::size_t pos0 = n;
    for (std::size_t i = 0; i < n; ++i) {
        if (tour[i] == 0u) {
            pos0 = i;
            break;
        }
    }

    if (pos0 == n) {
        output_identity_u64(static_cast<u64>(n), out);
        return;
    }

    bool first = true;
    for (std::size_t i = pos0; i < n; ++i) {
        if (!first) out.pushChar(' ');
        first = false;
        out.writeUInt(static_cast<u64>(tour[i]));
    }
    for (std::size_t i = 0; i < pos0; ++i) {
        out.pushChar(' ');
        out.writeUInt(static_cast<u64>(tour[i]));
    }
    out.pushChar('\n');
}

} // namespace

int main() {
    FastScanner in;
    FastOutput out;

    u64 n64 = 0;
    if (!in.readUInt64(n64)) {
        return 0;
    }

    if (n64 == 0) {
        out.pushChar('\n');
        out.flush();
        return 0;
    }

    // If the instance is beyond our 32-bit city-index storage, fall back to
    // a streaming identity permutation after consuming the input.
    if (n64 > static_cast<u64>(std::numeric_limits<u32>::max())) {
        double tmpx, tmpy;
        for (u64 i = 0; i < n64; ++i) {
            in.readDouble(tmpx);
            in.readDouble(tmpy);
        }
        output_identity_u64(n64, out);
        out.flush();
        return 0;
    }

    const u32 n = static_cast<u32>(n64);

    std::vector<float> xs;
    std::vector<float> ys;
    xs.resize(n);
    ys.resize(n);

    double minX = std::numeric_limits<double>::infinity();
    double maxX = -std::numeric_limits<double>::infinity();
    double minY = std::numeric_limits<double>::infinity();
    double maxY = -std::numeric_limits<double>::infinity();

    for (u32 i = 0; i < n; ++i) {
        double xd = 0.0, yd = 0.0;
        in.readDouble(xd);
        in.readDouble(yd);
        xs[i] = static_cast<float>(xd);
        ys[i] = static_cast<float>(yd);

        if (xd < minX) minX = xd;
        if (xd > maxX) maxX = xd;
        if (yd < minY) minY = yd;
        if (yd > maxY) maxY = yd;
    }

    std::vector<u32> tour;
    bool solved = false;

    try {
        if (n <= 22u) {
            tour = solve_exact_held_karp(xs, ys);
            solved = true;
        } else if (n <= 3000u) {
            tour = solve_small_multistart_nn_2opt(xs, ys);
            solved = true;
        } else {
            static constexpr u32 MORTON_LIMIT = 20000000u;

            try {
                if (n <= MORTON_LIMIT) {
                    tour = morton_radix_order(xs, ys, minX, maxX, minY, maxY);

                    if (n <= 50000u) {
                        window_two_opt_linear(tour, xs, ys, 64, 3);
                    } else if (n <= 300000u) {
                        window_two_opt_linear(tour, xs, ys, 24, 2);
                    } else if (n <= 2000000u) {
                        window_two_opt_linear(tour, xs, ys, 8, 1);
                    } else if (n <= 10000000u) {
                        improve_closure_edge_once(tour, xs, ys);
                    }
                } else {
                    tour = grid_snake_order(xs, ys, minX, maxX, minY, maxY);
                }
                solved = true;
            } catch (const std::bad_alloc&) {
                // Lower-memory fallback.
                tour = grid_snake_order(xs, ys, minX, maxX, minY, maxY);
                solved = true;
            }
        }
    } catch (const std::bad_alloc&) {
        solved = false;
    }

    if (!solved || tour.size() != static_cast<std::size_t>(n)) {
        output_identity_u64(static_cast<u64>(n), out);
        out.flush();
        return 0;
    }

    output_tour_starting_at_zero(tour, out);
    out.flush();
    return 0;
}
