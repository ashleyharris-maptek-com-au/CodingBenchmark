Below is a complete C++20 solver using an adaptive portfolio of fast guillotine free-rectangle heuristics.

- Valid placements only
- Rotation supported
- Scales to large `N`
- Uses 0-based `board_index`

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <set>
#include <tuple>
#include <numeric>
#include <limits>
#include <cstdint>
#include <climits>

struct RectInput {
    int id = -1;
    int w = 0;
    int h = 0;
    int maxSide = 0;
    int minSide = 0;
    int perimeter = 0;
    int flexibility = 0; // 1 if only one orientation is usable, 2 if both are usable
    long long area = 0;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct FreeRect {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int board = 0;
    bool active = false;
};

enum class OrderMode {
    MaxSideArea,
    AreaMaxSide,
    MinSideMaxSide
};

enum class ScoreMode {
    BestShortSide,
    BestAreaFit
};

enum class SplitMode {
    ShorterResidualFirst,
    LongerResidualFirst
};

enum class OrientationBias {
    PreferShortHeight,
    PreferLongHeight
};

struct VariantConfig {
    OrderMode order = OrderMode::MaxSideArea;
    ScoreMode score = ScoreMode::BestShortSide;
    SplitMode split = SplitMode::ShorterResidualFirst;
    OrientationBias bias = OrientationBias::PreferShortHeight;
};

struct Candidate {
    bool valid = false;
    bool opensNewBoard = true;
    int freeId = -1;
    int freeW = 0;
    int freeH = 0;
    int placedW = 0;
    int placedH = 0;
    int rotated = 0;
    int boardHint = 0;
};

struct PackResult {
    bool success = false;
    int boards = INT_MAX;
    long long qualityWaste = std::numeric_limits<long long>::max();
    int freeCount = INT_MAX;
    std::vector<Placement> placements;
};

class SegTreeMax {
public:
    SegTreeMax() : n_(1) {}

    void init(int size) {
        n_ = 1;
        while (n_ < size) n_ <<= 1;
        tree_.assign(2 * n_, 0);
    }

    void set_value(int pos, int value) {
        pos += n_;
        tree_[pos] = value;
        for (pos >>= 1; pos > 0; pos >>= 1) {
            tree_[pos] = std::max(tree_[pos << 1], tree_[(pos << 1) | 1]);
        }
    }

    int find_first(int left, int need) const {
        if (left >= n_) return -1;
        return find_first_impl(1, 0, n_ - 1, left, need);
    }

private:
    int find_first_impl(int node, int nl, int nr, int left, int need) const {
        if (nr < left || tree_[node] < need) return -1;
        if (nl == nr) return nl;
        const int mid = (nl + nr) >> 1;
        const int resLeft = find_first_impl(node << 1, nl, mid, left, need);
        if (resLeft != -1) return resLeft;
        return find_first_impl((node << 1) | 1, mid + 1, nr, left, need);
    }

    int n_;
    std::vector<int> tree_;
};

class FreeIndex {
public:
    FreeIndex(int maxW, int maxH)
        : maxW_(maxW),
          maxH_(maxH),
          byH_(static_cast<std::size_t>(maxH + 1)),
          byW_(static_cast<std::size_t>(maxW + 1)),
          count_(0) {
        segH_.init(maxH + 1);
        segW_.init(maxW + 1);
    }

    void insert(int id, int w, int h) {
        byH_[h].insert({w, id});
        byW_[w].insert({h, id});
        segH_.set_value(h, byH_[h].empty() ? 0 : byH_[h].rbegin()->first);
        segW_.set_value(w, byW_[w].empty() ? 0 : byW_[w].rbegin()->first);
        ++count_;
    }

    void erase(int id, int w, int h) {
        auto itH = byH_[h].find({w, id});
        if (itH != byH_[h].end()) byH_[h].erase(itH);

        auto itW = byW_[w].find({h, id});
        if (itW != byW_[w].end()) byW_[w].erase(itW);

        segH_.set_value(h, byH_[h].empty() ? 0 : byH_[h].rbegin()->first);
        segW_.set_value(w, byW_[w].empty() ? 0 : byW_[w].rbegin()->first);
        --count_;
    }

    int query_by_height(int needH, int needW) const {
        if (needH > maxH_ || needW > maxW_) return -1;
        const int h = segH_.find_first(needH, needW);
        if (h < 0 || h > maxH_) return -1;
        auto it = byH_[h].lower_bound({needW, -1});
        if (it == byH_[h].end()) return -1;
        return it->second;
    }

    int query_by_width(int needW, int needH) const {
        if (needH > maxH_ || needW > maxW_) return -1;
        const int w = segW_.find_first(needW, needH);
        if (w < 0 || w > maxW_) return -1;
        auto it = byW_[w].lower_bound({needH, -1});
        if (it == byW_[w].end()) return -1;
        return it->second;
    }

    int count() const {
        return count_;
    }

private:
    int maxW_;
    int maxH_;
    std::vector<std::set<std::pair<int, int>>> byH_;
    std::vector<std::set<std::pair<int, int>>> byW_;
    SegTreeMax segH_;
    SegTreeMax segW_;
    int count_;
};

static bool better_candidate(const Candidate& a, const Candidate& b, const VariantConfig& cfg) {
    if (!a.valid) return false;
    if (!b.valid) return true;

    if (a.opensNewBoard != b.opensNewBoard) {
        return !a.opensNewBoard; // always prefer not opening a new board
    }

    const int adw = a.freeW - a.placedW;
    const int adh = a.freeH - a.placedH;
    const int bdw = b.freeW - b.placedW;
    const int bdh = b.freeH - b.placedH;

    const int amin = std::min(adw, adh);
    const int amax = std::max(adw, adh);
    const int bmin = std::min(bdw, bdh);
    const int bmax = std::max(bdw, bdh);

    const long long aAreaWaste = static_cast<long long>(a.freeW) * a.freeH
                               - static_cast<long long>(a.placedW) * a.placedH;
    const long long bAreaWaste = static_cast<long long>(b.freeW) * b.freeH
                               - static_cast<long long>(b.placedW) * b.placedH;

    if (cfg.score == ScoreMode::BestShortSide) {
        if (amin != bmin) return amin < bmin;
        if (amax != bmax) return amax < bmax;
        if (aAreaWaste != bAreaWaste) return aAreaWaste < bAreaWaste;
    } else {
        if (aAreaWaste != bAreaWaste) return aAreaWaste < bAreaWaste;
        if (amin != bmin) return amin < bmin;
        if (amax != bmax) return amax < bmax;
    }

    if (cfg.bias == OrientationBias::PreferShortHeight) {
        if (a.placedH != b.placedH) return a.placedH < b.placedH;
        if (a.placedW != b.placedW) return a.placedW > b.placedW;
    } else {
        if (a.placedH != b.placedH) return a.placedH > b.placedH;
        if (a.placedW != b.placedW) return a.placedW < b.placedW;
    }

    if (a.boardHint != b.boardHint) return a.boardHint < b.boardHint;
    if (a.rotated != b.rotated) return a.rotated < b.rotated;
    return a.freeId < b.freeId;
}

static bool better_result(const PackResult& a, const PackResult& b) {
    if (!a.success) return false;
    if (!b.success) return true;
    if (a.boards != b.boards) return a.boards < b.boards;
    if (a.qualityWaste != b.qualityWaste) return a.qualityWaste < b.qualityWaste;
    if (a.freeCount != b.freeCount) return a.freeCount < b.freeCount;
    return false;
}

static std::vector<int> make_order(const std::vector<RectInput>& rects, OrderMode mode) {
    std::vector<int> ord(rects.size());
    std::iota(ord.begin(), ord.end(), 0);

    auto cmp = [&](int ia, int ib) {
        const RectInput& a = rects[ia];
        const RectInput& b = rects[ib];

        switch (mode) {
        case OrderMode::MaxSideArea:
            if (a.maxSide != b.maxSide) return a.maxSide > b.maxSide;
            if (a.flexibility != b.flexibility) return a.flexibility < b.flexibility;
            if (a.area != b.area) return a.area > b.area;
            if (a.minSide != b.minSide) return a.minSide > b.minSide;
            return a.id < b.id;

        case OrderMode::AreaMaxSide:
            if (a.area != b.area) return a.area > b.area;
            if (a.flexibility != b.flexibility) return a.flexibility < b.flexibility;
            if (a.maxSide != b.maxSide) return a.maxSide > b.maxSide;
            if (a.minSide != b.minSide) return a.minSide > b.minSide;
            return a.id < b.id;

        case OrderMode::MinSideMaxSide:
            if (a.minSide != b.minSide) return a.minSide > b.minSide;
            if (a.flexibility != b.flexibility) return a.flexibility < b.flexibility;
            if (a.maxSide != b.maxSide) return a.maxSide > b.maxSide;
            if (a.area != b.area) return a.area > b.area;
            return a.id < b.id;
        }

        return a.id < b.id;
    };

    std::sort(ord.begin(), ord.end(), cmp);
    return ord;
}

static PackResult pack_variant(const std::vector<RectInput>& rects,
                               const std::vector<int>& order,
                               int boardW,
                               int boardH,
                               const VariantConfig& cfg) {
    PackResult result;
    result.success = true;
    result.placements.assign(rects.size(), Placement{});

    std::vector<FreeRect> freeRects;
    freeRects.reserve(static_cast<std::size_t>(rects.size()) * 2ULL + 8ULL);

    FreeIndex index(boardW, boardH);
    int boardsOpened = 0;
    long long qualityWaste = 0;

    auto add_free_rect = [&](int x, int y, int w, int h, int board) {
        if (w <= 0 || h <= 0) return;
        const int id = static_cast<int>(freeRects.size());
        freeRects.push_back(FreeRect{x, y, w, h, board, true});
        index.insert(id, w, h);
    };

    auto remove_free_rect = [&](int id) {
        if (id < 0) return;
        FreeRect& fr = freeRects[id];
        if (!fr.active) return;
        index.erase(id, fr.w, fr.h);
        fr.active = false;
    };

    auto split_source = [&](const FreeRect& src, int usedW, int usedH) {
        const int dw = src.w - usedW;
        const int dh = src.h - usedH;
        if (dw <= 0 && dh <= 0) return;

        bool horizontalFirst;
        if (dw <= 0) {
            horizontalFirst = true;
        } else if (dh <= 0) {
            horizontalFirst = false;
        } else if (cfg.split == SplitMode::ShorterResidualFirst) {
            horizontalFirst = (dw <= dh);
        } else {
            horizontalFirst = (dw > dh);
        }

        if (horizontalFirst) {
            // right strip with height = usedH, plus bottom strip with full width
            if (dw > 0) add_free_rect(src.x + usedW, src.y, dw, usedH, src.board);
            if (dh > 0) add_free_rect(src.x, src.y + usedH, src.w, dh, src.board);
        } else {
            // right strip with full height, plus bottom-left block
            if (dw > 0) add_free_rect(src.x + usedW, src.y, dw, src.h, src.board);
            if (dh > 0) add_free_rect(src.x, src.y + usedH, usedW, dh, src.board);
        }
    };

    for (int ridx : order) {
        const RectInput& rect = rects[ridx];
        Candidate best;

        auto consider_existing = [&](int pw, int ph, int rotated) {
            if (pw > boardW || ph > boardH) return;

            const int idByH = index.query_by_height(ph, pw);
            if (idByH >= 0) {
                const FreeRect& fr = freeRects[idByH];
                Candidate c;
                c.valid = true;
                c.opensNewBoard = false;
                c.freeId = idByH;
                c.freeW = fr.w;
                c.freeH = fr.h;
                c.placedW = pw;
                c.placedH = ph;
                c.rotated = rotated;
                c.boardHint = fr.board;
                if (better_candidate(c, best, cfg)) best = c;
            }

            const int idByW = index.query_by_width(pw, ph);
            if (idByW >= 0) {
                const FreeRect& fr = freeRects[idByW];
                Candidate c;
                c.valid = true;
                c.opensNewBoard = false;
                c.freeId = idByW;
                c.freeW = fr.w;
                c.freeH = fr.h;
                c.placedW = pw;
                c.placedH = ph;
                c.rotated = rotated;
                c.boardHint = fr.board;
                if (better_candidate(c, best, cfg)) best = c;
            }
        };

        consider_existing(rect.w, rect.h, 0);
        if (rect.w != rect.h) {
            consider_existing(rect.h, rect.w, 1);
        }

        if (!best.valid) {
            auto consider_new = [&](int pw, int ph, int rotated) {
                if (pw > boardW || ph > boardH) return;
                Candidate c;
                c.valid = true;
                c.opensNewBoard = true;
                c.freeId = -1;
                c.freeW = boardW;
                c.freeH = boardH;
                c.placedW = pw;
                c.placedH = ph;
                c.rotated = rotated;
                c.boardHint = boardsOpened;
                if (better_candidate(c, best, cfg)) best = c;
            };

            consider_new(rect.w, rect.h, 0);
            if (rect.w != rect.h) {
                consider_new(rect.h, rect.w, 1);
            }
        }

        if (!best.valid) {
            result.success = false;
            result.boards = INT_MAX;
            result.placements.clear();
            return result;
        }

        FreeRect source;
        int boardId = 0;

        if (best.opensNewBoard) {
            boardId = boardsOpened++;
            source = FreeRect{0, 0, boardW, boardH, boardId, true};
        } else {
            source = freeRects[best.freeId];
            boardId = source.board;
            remove_free_rect(best.freeId);
        }

        result.placements[rect.id] = Placement{boardId, source.x, source.y, best.rotated};
        qualityWaste += static_cast<long long>(source.w) * source.h - rect.area;
        split_source(source, best.placedW, best.placedH);
    }

    result.boards = boardsOpened;
    result.qualityWaste = qualityWaste;
    result.freeCount = index.count();
    return result;
}

static std::vector<VariantConfig> build_variants(int n) {
    std::vector<VariantConfig> vars;

    // Strong fast core portfolio
    vars.push_back({OrderMode::MaxSideArea, ScoreMode::BestShortSide, SplitMode::ShorterResidualFirst, OrientationBias::PreferShortHeight});
    vars.push_back({OrderMode::MaxSideArea, ScoreMode::BestShortSide, SplitMode::ShorterResidualFirst, OrientationBias::PreferLongHeight});
    vars.push_back({OrderMode::MaxSideArea, ScoreMode::BestShortSide, SplitMode::LongerResidualFirst, OrientationBias::PreferShortHeight});
    vars.push_back({OrderMode::MaxSideArea, ScoreMode::BestShortSide, SplitMode::LongerResidualFirst, OrientationBias::PreferLongHeight});
    vars.push_back({OrderMode::AreaMaxSide, ScoreMode::BestAreaFit, SplitMode::ShorterResidualFirst, OrientationBias::PreferShortHeight});
    vars.push_back({OrderMode::AreaMaxSide, ScoreMode::BestAreaFit, SplitMode::LongerResidualFirst, OrientationBias::PreferLongHeight});

    // Medium-size extra diversity
    if (n <= 20000) {
        vars.push_back({OrderMode::MaxSideArea, ScoreMode::BestAreaFit, SplitMode::ShorterResidualFirst, OrientationBias::PreferShortHeight});
        vars.push_back({OrderMode::AreaMaxSide, ScoreMode::BestShortSide, SplitMode::LongerResidualFirst, OrientationBias::PreferLongHeight});
    }

    // Small-instance extra diversity
    if (n <= 4000) {
        vars.push_back({OrderMode::MinSideMaxSide, ScoreMode::BestShortSide, SplitMode::ShorterResidualFirst, OrientationBias::PreferShortHeight});
        vars.push_back({OrderMode::MinSideMaxSide, ScoreMode::BestShortSide, SplitMode::LongerResidualFirst, OrientationBias::PreferLongHeight});
        vars.push_back({OrderMode::MinSideMaxSide, ScoreMode::BestAreaFit, SplitMode::ShorterResidualFirst, OrientationBias::PreferLongHeight});
        vars.push_back({OrderMode::MinSideMaxSide, ScoreMode::BestAreaFit, SplitMode::LongerResidualFirst, OrientationBias::PreferShortHeight});
    }

    return vars;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    int boardW = 0;
    int boardH = 0;

    if (!(std::cin >> N >> boardW >> boardH)) {
        return 0;
    }

    if (N == 0) {
        std::cout << 0 << '\n';
        return 0;
    }

    if (boardW <= 0 || boardH <= 0) {
        std::cout << -1 << '\n';
        return 0;
    }

    std::vector<RectInput> rects(static_cast<std::size_t>(N));
    long long sumArea = 0;

    for (int i = 0; i < N; ++i) {
        int w = 0;
        int h = 0;
        std::cin >> w >> h;

        const bool fit0 = (w <= boardW && h <= boardH);
        const bool fit1 = (w != h && h <= boardW && w <= boardH);

        if (!fit0 && !fit1) {
            std::cout << -1 << '\n';
            return 0;
        }

        RectInput r;
        r.id = i;
        r.w = w;
        r.h = h;
        r.maxSide = std::max(w, h);
        r.minSide = std::min(w, h);
        r.perimeter = w + h;
        r.flexibility = static_cast<int>(fit0) + static_cast<int>(fit1);
        r.area = static_cast<long long>(w) * h;
        rects[i] = r;
        sumArea += r.area;
    }

    const long long boardArea = static_cast<long long>(boardW) * boardH;
    const long long areaLowerBound = (sumArea + boardArea - 1) / boardArea;

    const std::vector<int> orderMaxSide = make_order(rects, OrderMode::MaxSideArea);
    const std::vector<int> orderArea = make_order(rects, OrderMode::AreaMaxSide);
    const std::vector<int> orderMinSide = make_order(rects, OrderMode::MinSideMaxSide);

    auto get_order = [&](OrderMode mode) -> const std::vector<int>& {
        switch (mode) {
        case OrderMode::MaxSideArea: return orderMaxSide;
        case OrderMode::AreaMaxSide: return orderArea;
        case OrderMode::MinSideMaxSide: return orderMinSide;
        }
        return orderMaxSide;
    };

    const std::vector<VariantConfig> variants = build_variants(N);

    PackResult best;

    for (const VariantConfig& cfg : variants) {
        PackResult current = pack_variant(rects, get_order(cfg.order), boardW, boardH, cfg);
        if (better_result(current, best)) {
            best = std::move(current);
            if (static_cast<long long>(best.boards) == areaLowerBound) {
                break; // cannot beat the area lower bound
            }
        }
    }

    if (!best.success) {
        std::cout << -1 << '\n';
        return 0;
    }

    std::cout << best.boards << '\n';
    for (int i = 0; i < N; ++i) {
        const Placement& p = best.placements[i];
        std::cout << p.board << ' ' << p.x << ' ' << p.y << ' ' << p.rotated << '\n';
    }

    return 0;
}
```
