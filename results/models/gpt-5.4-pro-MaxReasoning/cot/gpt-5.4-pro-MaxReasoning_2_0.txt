        _buffer[_pos++] = b; 
    }

    public void WriteInt(long x) {

            return; 
        }
        if (x < 0) { 
            WriteByte((byte)'-'); 
            x = -x; 
        }
        int len = 0;
        while (x > 0) {

        }

    }

            _stream.Write(_buffer, 0, _pos); 
            _pos = 0; 
        } 
    }

        _stream.Flush(); 
    }
}
