```csharp
static int[][] BuildNeighborOrder(double[][] dist, int n, int totalNodes)
{
    int[][] order = new int[totalNodes][];
    for (int i = 0; i < totalNodes; i++)
    {
        int[] idx = new int[n];
        double[] keys = new double[n];
        for (int j = 0; j < n; j++)
        {
            idx[j] = j;
            keys[j] = dist[i][j];
        }
        Array.Sort(keys, idx);
        order[i] = idx;
    }
    return order;
}
```
