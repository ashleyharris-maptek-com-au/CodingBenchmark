#include <vector>
#include <cmath>
#include <algorithm>
#include <limits>
#include <numeric> // For std::iota
#include <iomanip> // For std::fixed and std::setprecision

// Define a structure for a city
struct City {
    double x, y;
    int id;
};

// Function to calculate Euclidean distance between two cities
inline double calculateDistance(const City& c1, const City& c2) {
    return std::sqrt(std::pow(c1.x - c2.x, 2) + std::pow(c1.y - c2.y, 2));
}

// Function to calculate the total tour length
double calculateTourLength(const std::vector<int>& tour, const std::vector<City>& cities) {
    double totalLength = 0.0;
    for (size_t i = 0; i < tour.size() - 1; ++i) {
        totalLength += calculateDistance(cities[tour[i]], cities[tour[i+1]]);
    }
    // Add distance from last city back to the starting city (city 0)
    totalLength += calculateDistance(cities[tour.back()], cities[tour[0]]);
    return totalLength;
}

// Nearest Neighbor heuristic for initial tour
std::vector<int> nearestNeighbor(const std::vector<City>& cities) {
    int n = cities.size();
    std::vector<int> tour;
    std::vector<bool> visited(n, false);

    // Start at city 0
    int currentCityIndex = 0;
    tour.push_back(currentCityIndex);
    visited[currentCityIndex] = true;

    for (int i = 1; i < n; ++i) {
        double minDistance = std::numeric_limits<double>::max();
        int nextCityIndex = -1;

        for (int j = 0; j < n; ++j) {
            if (!visited[j]) {
                double dist = calculateDistance(cities[currentCityIndex], cities[j]);
                if (dist < minDistance) {
                    minDistance = dist;
                    nextCityIndex = j;
                }
            }
        }
        tour.push_back(nextCityIndex);
        visited[nextCityIndex] = true;
        currentCityIndex = nextCityIndex;
    }
    return tour;
}

// Optimized 2-opt local search improvement
std::vector<int> twoOpt(std::vector<int> currentTour, const std::vector<City>& cities) {
    int n = currentTour.size();
    bool improved = true;
    double currentDistance = calculateTourLength(currentTour, cities);

    while (improved) {
        improved = false;
        for (int i = 0; i < n - 1; ++i) { // i and j are indices in the tour
            for (int j = i + 2; j < n; ++j) { // j must be at least i+2 to form distinct edges
                // Edges are (currentTour[i], currentTour[i+1]) and (currentTour[j], currentTour[j+1 % n])
                // We are swapping by reversing segment between i+1 and j.
                // The new edges are (currentTour[i], currentTour[j]) and (currentTour[i+1], currentTour[j+1 % n])

                // Handle the wrap-around for the last segment
                int c1_idx = currentTour[i];
                int c2_idx = currentTour[(i + 1) % n];
                int c3_idx = currentTour[j];
                int c4_idx = currentTour[(j + 1) % n];

                double old_dist = calculateDistance(cities[c1_idx], cities[c2_idx]) +
                                  calculateDistance(cities[c3_idx], cities[c4_idx]);
                double new_dist = calculateDistance(cities[c1_idx], cities[c3_idx]) +
                                  calculateDistance(cities[c2_idx], cities[c4_idx]);

                if (new_dist < old_dist) {
                    std::reverse(currentTour.begin() + (i + 1), currentTour.begin() + j + 1);
                    currentDistance = currentDistance - old_dist + new_dist;
                    improved = true;
                }
            }
        }
    }
    return currentTour;
}


// --- Hilbert Curve for large N ---

// Rotate/flip a quadrant appropriately
void rot(int n, int* x, int* y, int rx, int ry) {
    if (ry == 0) {
        if (rx == 1) {
            *x = n - 1 - *x;
            *y = n - 1 - *y;
        }
        // Swap x and y
        int t = *x;
        *x = *y;
        *y = t;
    }
}

// Convert (x,y) to d
// n: grid size (power of 2)
// x, y: coordinates within [0, n-1]
long long xy_to_d(int n, int x, int y) {
    long long d = 0;
    for (int s = n / 2; s > 0; s /= 2) {
        int rx = (x & s) > 0;
        int ry = (y & s) > 0;
        d += (long long)s * s * ((3 * rx) ^ ry);
        rot(s, &x, &y, rx, ry);
    }
    return d;
}

struct HilbertCity {
    City city;
    long long hilbert_index;
};

// Comparison for sorting Hilbert cities
bool compareHilbertCities(const HilbertCity& a, const HilbertCity& b) {
    return a.hilbert_index < b.hilbert_index;
}

// Hilbert curve approximation for TSP
std::vector<int> hilbertApproximation(const std::vector<City>& cities) {
    int n = cities.size();
    if (n == 0) return {};

    // 1. Normalize coordinates to a [0, 1] range
    double min_x = cities[0].x, max_x = cities[0].x;
    double min_y = cities[0].y, max_y = cities[0].y;
    for (int i = 1; i < n; ++i) {
        if (cities[i].x < min_x) min_x = cities[i].x;
        if (cities[i].x > max_x) max_x = cities[i].x;
        if (cities[i].y < min_y) min_y = cities[i].y;
        if (cities[i].y > max_y) max_y = cities[i].y;
    }

    double range_x = max_x - min_x;
    double range_y = max_y - min_y;

    // To avoid division by zero if all points are on a line or single point
    if (range_x == 0) range_x = 1.0;
    if (range_y == 0) range_y = 1.0;

    // Determine the grid size (power of 2)
    // We need enough resolution for large N. Max coord value is usually 2^m - 1.
    // Let's target a reasonable max coordinate value, e.g., 2^16 (65536)
    // The number of levels for Hilbert curve. 2^M. Max M can be around 30 for long long d.
    // For x,y to be int, s could be up to 2^15. So 2^16 for n.
    int M = 16; // Number of bits for each coordinate, resulting in grid size N = 2^M
    int grid_n = 1 << M; // 2^16 = 65536

    std::vector<HilbertCity> hilbertCities(n);
    for (int i = 0; i < n; ++i) {
        hilbertCities[i].city = cities[i];
        
        // Scale and shift coordinates to be within [0, grid_n - 1] integers
        int scaled_x = static_cast<int>(std::floor((cities[i].x - min_x) / range_x * (grid_n - 1)));
        int scaled_y = static_cast<int>(std::floor((cities[i].y - min_y) / range_y * (grid_n - 1)));
        
        hilbertCities[i].hilbert_index = xy_to_d(grid_n, scaled_x, scaled_y);
    }

    // 2. Sort cities by their Hilbert curve index
    std::sort(hilbertCities.begin(), hilbertCities.end(), compareHilbertCities);

    // 3. Construct the tour from the sorted order
    std::vector<int> tour;
    tour.reserve(n);
    for (const auto& hc : hilbertCities) {
        tour.push_back(hc.city.id);
    }

    // Ensure tour starts with city 0 if it exists
    if (n > 0 && tour[0] != 0) {
        // Find city 0 in the tour and rotate the tour so it starts with 0
        auto it = std::find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) {
            std::rotate(tour.begin(), it, tour.end());
        }
    }
    
    return tour;
}
