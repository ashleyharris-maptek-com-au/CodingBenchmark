use std::io::{self, BufRead};

struct Solver {
    W: usize,
    H: usize,
    adj: Vec<Vec<usize>>,
    visited: Vec<bool>,
    deg: Vec<usize>,
    total_deg0: usize,
    total_deg1: usize,
    deg1_xor: usize,
    start: usize,
    require_cycle: bool,
    path: Vec<usize>,
    bfs_vis: Vec<u32>,
    bfs_gen: u32,
    bfs_q: Vec<usize>,
}

impl Solver {
    fn update_deg_decrement(&mut self, v: usize) {
        let d = self.deg[v];
        if d == 1 {
            self.total_deg1 -= 1;
            self.deg1_xor ^= v;
        } else if d == 0 {
            self.total_deg0 -= 1;
        }
        
        self.deg[v] -= 1;
        let nd = self.deg[v];
        
        if nd == 1 {
            self.total_deg1 += 1;
            self.deg1_xor ^= v;
        } else if nd == 0 {
            self.total_deg0 += 1;
        }
    }

    fn update_deg_increment(&mut self, v: usize) {
        let d = self.deg[v];
        if d == 1 {
            self.total_deg1 -= 1;
            self.deg1_xor ^= v;
        } else if d == 0 {
            self.total_deg0 -= 1;
        }
        
        self.deg[v] += 1;
        let nd = self.deg[v];
        
        if nd == 1 {
            self.total_deg1 += 1;
            self.deg1_xor ^= v;
        } else if nd == 0 {
            self.total_deg0 += 1;
        }
    }
    
    fn solve(&mut self, curr: usize, unvisited_count: usize) -> bool {
        if unvisited_count == 0 {
            if self.require_cycle {
                return self.adj[curr].contains(&self.start);
            }
            return true;
        }
        
        if self.total_deg0 > 0 {
            if unvisited_count > 1 { return false; }
        }
        
        let mut adj_deg1 = 0;
        let mut adj_deg1_xor = 0;
        let mut start_node = None;
        for &v in &self.adj[curr] {
            if !self.visited[v] {
                if start_node.is_none() {
                    start_node = Some(v);
                }
                if self.deg[v] == 1 {
                    adj_deg1 += 1;
                    adj_deg1_xor ^= v;
                }
            }
        }
        
        if start_node.is_none() {
            return false;
        }
        
        let ends_count = self.total_deg1 - adj_deg1;
        if ends_count > 1 { return false; }
        if self.require_cycle && ends_count == 1 {
            let end_node = self.deg1_xor ^ adj_deg1_xor;
            if !self.adj[end_node].contains(&self.start) {
                return false;
            }
        }
        
        self.bfs_gen += 1;
        let start_v = start_node.unwrap();
        self.bfs_q[0] = start_v;
        self.bfs_vis[start_v] = self.bfs_gen;
        let mut reach_count = 1;
        let mut head = 0;
        let mut tail = 1;
        
        while head < tail {
            let u = self.bfs_q[head];
            head += 1;
            for &v in &self.adj[u] {
                if !self.visited[v] && self.bfs_vis[v] != self.bfs_gen {
                    self.bfs_vis[v] = self.bfs_gen;
                    self.bfs_q[tail] = v;
                    tail += 1;
                    reach_count += 1;
                }
            }
        }
        
        if reach_count < unvisited_count {
            return false;
        }
        
        let mut neighbors = [0; 4];
        let mut n_len = 0;
        for &v in &self.adj[curr] {
            if !self.visited[v] {
                neighbors[n_len] = v;
                n_len += 1;
            }
        }
        neighbors[..n_len].sort_unstable_by_key(|&v| self.deg[v]);
        
        for i in 0..n_len {
            let nxt = neighbors[i];
            self.visited[nxt] = true;
            for &w in &self.adj[nxt] {
                if !self.visited[w] {
                    self.update_deg_decrement(w);
                }
            }
            self.path.push(nxt);
            
            if self.solve(nxt, unvisited_count - 1) {
                return true;
            }
            
            self.path.pop();
            for &w in &self.adj[nxt] {
                if !self.visited[w] {
                    self.update_deg_increment(w);
                }
            }
            self.visited[nxt] = false;
        }
        
        false
    }
}

fn solve_main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().map(|l| l.unwrap());
    
    let first_line = match lines.next() {
        Some(l) => l,
        None => return,
    };
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.is_empty() { return; }
    let w: usize = parts[0].parse().unwrap();
    let h: usize = parts[1].parse().unwrap();
    let obstacles_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<u8>().unwrap() != 0;
    
    let mut obstacles = vec![vec![false; w]; h];
    for _ in 0..obstacles_count {
        if let Some(line) = lines.next() {
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() >= 2 {
                let x: usize = parts[0].parse().unwrap();
                let y: usize = parts[1].parse().unwrap();
                obstacles[y][x] = true;
            }
        }
    }
    
    if obstacles[0][0] {
        println!("0");
        return;
    }
    
    let mut total_w = 0;
    let mut total_b = 0;
    for y in 0..h {
        for x in 0..w {
            if !obstacles[y][x] {
                if (x + y) % 2 == 0 { total_w += 1; }
                else { total_b += 1; }
            }
        }
    }
    
    if require_cycle {
        if total_w != total_b {
            println!("0");
            return;
        }
    } else {
        if total_w != total_b && total_w != total_b + 1 {
            println!("0");
            return;
        }
    }
    
    let mut adj = vec![Vec::new(); w * h];
    for y in 0..h {
        for x in 0..w {
            if obstacles[y][x] { continue; }
            let u = y * w + x;
            
            let dx = [0, 1, 0, -1];
            let dy = [-1, 0, 1, 0];
            for i in 0..4 {
                let nx = x as i32 + dx[i];
                let ny = y as i32 + dy[i];
                if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                    if !obstacles[ny as usize][nx as usize] {
                        adj[u].push((ny * w as i32 + nx) as usize);
                    }
                }
            }
        }
    }
    
    let mut deg = vec![0; w * h];
    let mut total_deg0 = 0;
    let mut total_deg1 = 0;
    let mut deg1_xor = 0;
    
    for u in 0..(w * h) {
        if obstacles[u / w][u % w] { continue; }
        deg[u] = adj[u].len();
        if deg[u] == 0 { total_deg0 += 1; }
        if deg[u] == 1 { total_deg1 += 1; deg1_xor ^= u; }
    }
    
    let mut solver = Solver {
        W: w,
        H: h,
        adj,
        visited: vec![false; w * h],
        deg,
        total_deg0,
        total_deg1,
        deg1_xor,
        start: 0,
        require_cycle,
        path: Vec::with_capacity(w * h),
        bfs_vis: vec![0; w * h],
        bfs_gen: 0,
        bfs_q: vec![0; w * h],
    };
    
    let unvisited_count = total_w + total_b - 1;
    
    solver.visited[0] = true;
    for &v in &solver.adj[0] {
        if !solver.visited[v] {
            solver.update_deg_decrement(v);
        }
    }
    solver.path.push(0);
    
    if solver.solve(0, unvisited_count) {
        println!("{}", solver.path.len());
        for &u in &solver.path {
            println!("{} {}", u % w, u / w);
        }
    } else {
        println!("0");
    }
}

fn main() {
    std::thread::Builder::new()
        .stack_size(64 * 1024 * 1024)
        .spawn(solve_main)
        .unwrap()
        .join()
        .unwrap();
}
