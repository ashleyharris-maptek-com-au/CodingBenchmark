using System;
using System.Collections.Generic;
using System.IO;

class MinesweeperSolver
{
    static int H, W;
    static byte[] grid;
    static byte[] unknownCount;
    static byte[] mineCount;
    static bool[] inQueue;
    static Queue<int> queue;
    static List<int> safeCells;

    static void Main()
    {
        Stream inputStream = Console.OpenStandardInput();
        List<byte> line1Bytes = new List<byte>();
        int b;
        while ((b = inputStream.ReadByte()) != -1)
        {
            if (b == '\n') break;
            if (b != '\r') line1Bytes.Add((byte)b);
        }
        if (line1Bytes.Count == 0) return;
        string line1 = System.Text.Encoding.ASCII.GetString(line1Bytes.ToArray());
        string[] parts = line1.Split(' ', StringSplitOptions.RemoveEmptyEntries);
        if (parts.Length < 2) return;
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        grid = new byte[H * W];
        int pos = 0;
        int expected = H * W;
        int bufferSize = 65536;
        byte[] buffer = new byte[bufferSize];
        while (pos < expected)
        {
            int read = inputStream.Read(buffer, 0, bufferSize);
            if (read == 0) break;
            for (int i = 0; i < read; i++)
            {
                byte c = buffer[i];
                if (c == ' ' || (c >= '0' && c <= '8') || c == '*')
                {
                    grid[pos++] = c;
                    if (pos == expected) break;
                }
            }
        }

        unknownCount = new byte[H * W];
        mineCount = new byte[H * W];
        inQueue = new bool[H * W];
        queue = new Queue<int>();
        safeCells = new List<int>();

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                int idx = r * W + c;
                byte val = grid[idx];
                if (val >= '0' && val <= '8')
                {
                    byte u = 0;
                    byte m = 0;
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        int nr = r + dr;
                        if (nr < 0 || nr >= H) continue;
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            int nc = c + dc;
                            if (nc < 0 || nc >= W) continue;
                            if (dr == 0 && dc == 0) continue;

                            byte nval = grid[nr * W + nc];
                            if (nval == ' ') u++;
                            else if (nval == '*') m++;
                        }
                    }
                    unknownCount[idx] = u;
                    mineCount[idx] = m;
                    if (u > 0)
                    {
                        queue.Enqueue(idx);
                        inQueue[idx] = true;
                    }
                }
            }
        }

        bool changed = true;
        while (changed)
        {
            changed = false;

            while (queue.Count > 0)
            {
                int idx = queue.Dequeue();
                inQueue[idx] = false;

                byte u = unknownCount[idx];
                if (u == 0) continue;

                byte m = mineCount[idx];
                byte target = (byte)(grid[idx] - '0');

                if (m == target)
                {
                    MarkNeighbors(idx, true);
                    changed = true;
                }
                else if (m + u == target)
                {
                    MarkNeighbors(idx, false);
                    changed = true;
                }
            }

            if (changed) continue;

            List<int> activeNumbers = new List<int>();
            for(int i = 0; i < H * W; i++) {
                if (grid[i] >= '0' && grid[i] <= '8' && unknownCount[i] > 0) {
                    activeNumbers.Add(i);
                }
            }

            if (activeNumbers.Count == 0) break;

            bool subsetChanged = false;
            foreach (int a in activeNumbers)
            {
                if (unknownCount[a] == 0) continue;

                int rA = a / W;
                int cA = a % W;

                Span<int> unkA = stackalloc int[8];
                int unkACount = 0;
                for(int dr = -1; dr <= 1; dr++) {
                    int nr = rA + dr;
                    if (nr < 0 || nr >= H) continue;
                    for(int dc = -1; dc <= 1; dc++) {
                        int nc = cA + dc;
                        if (nc < 0 || nc >= W) continue;
                        if (dr == 0 && dc == 0) continue;
                        int nidx = nr * W + nc;
                        if (grid[nidx] == ' ') {
                            unkA[unkACount++] = nidx;
                        }
                    }
                }

                for (int dr = -2; dr <= 2; dr++)
                {
                    int nrB = rA + dr;
                    if (nrB < 0 || nrB >= H) continue;
                    for (int dc = -2; dc <= 2; dc++)
                    {
                        int ncB = cA + dc;
                        if (ncB < 0 || ncB >= W) continue;
                        if (dr == 0 && dc == 0) continue;

                        int b = nrB * W + ncB;
                        if (grid[b] >= '0' && grid[b] <= '8' && unknownCount[b] > 0)
                        {
                            if (unknownCount[a] >= unknownCount[b]) continue;

                            Span<int> unkB = stackalloc int[8];
                            int unkBCount = 0;
                            for (int drB = -1; drB <= 1; drB++)
                            {
                                int nnr = nrB + drB;
                                if (nnr < 0 || nnr >= H) continue;
                                for (int dcB = -1; dcB <= 1; dcB++)
                                {
                                    int nnc = ncB + dcB;
                                    if (nnc < 0 || nnc >= W) continue;
                                    if (drB == 0 && dcB == 0) continue;
                                    int nnidx = nnr * W + nnc;
                                    if (grid[nnidx] == ' ')
                                    {
                                        unkB[unkBCount++] = nnidx;
                                    }
                                }
                            }

                            bool isSubset = true;
                            for (int i = 0; i < unkACount; i++)
                            {
                                bool found = false;
                                for (int j = 0; j < unkBCount; j++)
                                {
                                    if (unkA[i] == unkB[j])
                                    {
                                        found = true;
                                        break;
                                    }
                                }
                                if (!found)
                                {
                                    isSubset = false;
                                    break;
                                }
                            }

                            if (isSubset)
                            {
                                int mA = grid[a] - '0' - mineCount[a];
                                int mB = grid[b] - '0' - mineCount[b];
                                int diff = mB - mA;
                                int remainingCells = unkBCount - unkACount;

                                if (diff == 0)
                                {
                                    for (int j = 0; j < unkBCount; j++)
                                    {
                                        bool inA = false;
                                        for (int i = 0; i < unkACount; i++)
                                        {
                                            if (unkB[j] == unkA[i]) { inA = true; break; }
                                        }
                                        if (!inA && grid[unkB[j]] == ' ')
                                        {
                                            grid[unkB[j]] = (byte)'S';
                                            safeCells.Add(unkB[j]);
                                            UpdateNeighborsOf(unkB[j], true);
                                            subsetChanged = true;
                                        }
                                    }
                                }
                                else if (diff == remainingCells)
                                {
                                    for (int j = 0; j < unkBCount; j++)
                                    {
                                        bool inA = false;
                                        for (int i = 0; i < unkACount; i++)
                                        {
                                            if (unkB[j] == unkA[i]) { inA = true; break; }
                                        }
                                        if (!inA && grid[unkB[j]] == ' ')
                                        {
                                            grid[unkB[j]] = (byte)'M';
                                            UpdateNeighborsOf(unkB[j], false);
                                            subsetChanged = true;
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
            }

            if (subsetChanged)
            {
                changed = true;
                continue;
            }

            bool cspChanged = false;
            
            List<List<int>> componentsUnk = new List<List<int>>();
            List<List<int>> componentsNum = new List<List<int>>();

            HashSet<int> visitedNum = new HashSet<int>();
            HashSet<int> visitedUnk = new HashSet<int>();

            foreach (int num in activeNumbers)
            {
                if (unknownCount[num] == 0) continue;
                if (visitedNum.Contains(num)) continue;

                List<int> compNum = new List<int>();
                List<int> compUnk = new List<int>();
                Queue<int> qNum = new Queue<int>();

                qNum.Enqueue(num);
                visitedNum.Add(num);

                while (qNum.Count > 0)
                {
                    int currNum = qNum.Dequeue();
                    compNum.Add(currNum);

                    int r = currNum / W;
                    int c = currNum % W;
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        int nr = r + dr;
                        if (nr < 0 || nr >= H) continue;
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            int nc = c + dc;
                            if (nc < 0 || nc >= W) continue;
                            if (dr == 0 && dc == 0) continue;
                            int nidx = nr * W + nc;
                            if (grid[nidx] == ' ')
                            {
                                if (visitedUnk.Add(nidx))
                                {
                                    compUnk.Add(nidx);
                                    int ur = nidx / W;
                                    int uc = nidx % W;
                                    for (int urr = -1; urr <= 1; urr++)
                                    {
                                        int nur = ur + urr;
                                        if (nur < 0 || nur >= H) continue;
                                        for (int ucc = -1; ucc <= 1; ucc++)
                                        {
                                            int nuc = uc + ucc;
                                            if (nuc < 0 || nuc >= W) continue;
                                            if (urr == 0 && ucc == 0) continue;
                                            int nnidx = nur * W + nuc;
                                            if (grid[nnidx] >= '0' && grid[nnidx] <= '8' && unknownCount[nnidx] > 0)
                                            {
                                                if (visitedNum.Add(nnidx))
                                                {
                                                    qNum.Enqueue(nnidx);
                                                }
                                            }
                                        }
                                    }
                                }
                            }
                        }
                    }
                }
                componentsNum.Add(compNum);
                componentsUnk.Add(compUnk);
            }

            for (int cIdx = 0; cIdx < componentsUnk.Count; cIdx++)
            {
                List<int> compUnk = componentsUnk[cIdx];
                List<int> compNum = componentsNum[cIdx];

                int N = compUnk.Count;
                if (N == 0 || N > 5000) continue;

                Dictionary<int, int> unkToIdx = new Dictionary<int, int>();
                for (int i = 0; i < N; i++) unkToIdx[compUnk[i]] = i;

                int M = compNum.Count;
                int[] targets = new int[M];
                int[] unknownsLeft = new int[M];
                List<int>[] numToVars = new List<int>[M];
                List<int>[] varToNums = new List<int>[N];

                for (int i = 0; i < N; i++) varToNums[i] = new List<int>();

                for (int i = 0; i < M; i++)
                {
                    int num = compNum[i];
                    targets[i] = grid[num] - '0' - mineCount[num];
                    numToVars[i] = new List<int>();
                    
                    int r = num / W;
                    int c = num % W;
                    for (int dr = -1; dr <= 1; dr++)
                    {
                        int nr = r + dr;
                        if (nr < 0 || nr >= H) continue;
                        for (int dc = -1; dc <= 1; dc++)
                        {
                            int nc = c + dc;
                            if (nc < 0 || nc >= W) continue;
                            if (dr == 0 && dc == 0) continue;
                            int nidx = nr * W + nc;
                            if (unkToIdx.TryGetValue(nidx, out int vIdx))
                            {
                                numToVars[i].Add(vIdx);
                                varToNums[vIdx].Add(i);
                            }
                        }
                    }
                    unknownsLeft[i] = numToVars[i].Count;
                }

                bool initialValid = true;
                for (int i = 0; i < M; i++) {
                    if (targets[i] < 0 || targets[i] > unknownsLeft[i]) {
                        initialValid = false; break;
                    }
                }
                if (!initialValid) continue;

                int validAssignments = 0;
                int[] mineInValid = new int[N];
                bool[] currentAssignment = new bool[N];
                int nodesVisited = 0;
                bool aborted = false;

                void SolveCSP(int varIdx)
                {
                    if (aborted) return;
                    if (nodesVisited++ > 150000) { aborted = true; return; }

                    if (varIdx == N)
                    {
                        validAssignments++;
                        for (int i = 0; i < N; i++) {
                            if (currentAssignment[i]) mineInValid[i]++;
                        }
                        return;
                    }

                    foreach (int numIdx in varToNums[varIdx]) unknownsLeft[numIdx]--;

                    bool canBe0 = true;
                    foreach (int numIdx in varToNums[varIdx])
                    {
                        if (targets[numIdx] > unknownsLeft[numIdx]) { canBe0 = false; break; }
                    }
                    if (canBe0)
                    {
                        currentAssignment[varIdx] = false;
                        SolveCSP(varIdx + 1);
                    }

                    bool canBe1 = true;
                    foreach (int numIdx in varToNums[varIdx]) {
                        if (targets[numIdx] - 1 < 0) { canBe1 = false; break; }
                    }
                    if (canBe1)
                    {
                        foreach (int numIdx in varToNums[varIdx]) targets[numIdx]--;
                        
                        currentAssignment[varIdx] = true;
                        SolveCSP(varIdx + 1);
                        
                        foreach (int numIdx in varToNums[varIdx]) targets[numIdx]++;
                    }

                    foreach (int numIdx in varToNums[varIdx]) unknownsLeft[numIdx]++;
                }

                SolveCSP(0);

                if (!aborted && validAssignments > 0)
                {
                    for (int i = 0; i < N; i++)
                    {
                        if (mineInValid[i] == 0)
                        {
                            int nidx = compUnk[i];
                            if (grid[nidx] == ' ')
                            {
                                grid[nidx] = (byte)'S';
                                safeCells.Add(nidx);
                                UpdateNeighborsOf(nidx, true);
                                cspChanged = true;
                            }
                        }
                        else if (mineInValid[i] == validAssignments)
                        {
                            int nidx = compUnk[i];
                            if (grid[nidx] == ' ')
                            {
                                grid[nidx] = (byte)'M';
                                UpdateNeighborsOf(nidx, false);
                                cspChanged = true;
                            }
                        }
                    }
                }
            }

            if (cspChanged)
            {
                changed = true;
            }
        }

        using (StreamWriter writer = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 65536))
        {
            writer.WriteLine(safeCells.Count);
            foreach (int cell in safeCells)
            {
                int r = cell / W;
                int c = cell % W;
                writer.WriteLine($"{c} {r}");
            }
        }
    }

    static void MarkNeighbors(int idx, bool isSafe)
    {
        int r = idx / W;
        int c = idx % W;
        for (int dr = -1; dr <= 1; dr++)
        {
            int nr = r + dr;
            if (nr < 0 || nr >= H) continue;
            for (int dc = -1; dc <= 1; dc++)
            {
                int nc = c + dc;
                if (nc < 0 || nc >= W) continue;
                if (dr == 0 && dc == 0) continue;
                
                int nidx = nr * W + nc;
                if (grid[nidx] == ' ')
                {
                    if (isSafe)
                    {
                        grid[nidx] = (byte)'S';
                        safeCells.Add(nidx);
                        UpdateNeighborsOf(nidx, true);
                    }
                    else
                    {
                        grid[nidx] = (byte)'M';
                        UpdateNeighborsOf(nidx, false);
                    }
                }
            }
        }
    }

    static void UpdateNeighborsOf(int idx, bool isSafe)
    {
        int r = idx / W;
        int c = idx % W;
        for (int dr = -1; dr <= 1; dr++)
        {
            int nr = r + dr;
            if (nr < 0 || nr >= H) continue;
            for (int dc = -1; dc <= 1; dc++)
            {
                int nc = c + dc;
                if (nc < 0 || nc >= W) continue;
                if (dr == 0 && dc == 0) continue;
                
                int nidx = nr * W + nc;
                byte val = grid[nidx];
                if (val >= '0' && val <= '8')
                {
                    unknownCount[nidx]--;
                    if (!isSafe)
                    {
                        mineCount[nidx]++;
                    }
                    if (unknownCount[nidx] > 0 && !inQueue[nidx])
                    {
                        queue.Enqueue(nidx);
                        inQueue[nidx] = true;
                    }
                }
            }
        }
    }
}
