#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

struct Item {
    int id;
    int w, h;
    long long area;
};

struct FreeRect {
    int x, y, w, h;
};

struct Board {
    vector<FreeRect> free_rects;
    int max_w = 0;
    int max_h = 0;
};

struct Result {
    int board_idx;
    int x, y;
    int rotated;
};

void place_rect(Board& B, int px, int py, int pw, int ph, int min_side) {
    vector<FreeRect> new_rects;
    for (int i = 0; i < (int)B.free_rects.size(); /* manual */) {
        const FreeRect& F = B.free_rects[i];
        
        // Check for intersection
        if (!(px >= F.x + F.w || px + pw <= F.x || py >= F.y + F.h || py + ph <= F.y)) {
            // Top
            if (py > F.y) {
                FreeRect N = {F.x, F.y, F.w, py - F.y};
                if (N.w >= min_side && N.h >= min_side) new_rects.push_back(N);
            }
            // Bottom
            if (py + ph < F.y + F.h) {
                FreeRect N = {F.x, py + ph, F.w, F.y + F.h - (py + ph)};
                if (N.w >= min_side && N.h >= min_side) new_rects.push_back(N);
            }
            // Left
            if (px > F.x) {
                FreeRect N = {F.x, F.y, px - F.x, F.h};
                if (N.w >= min_side && N.h >= min_side) new_rects.push_back(N);
            }
            // Right
            if (px + pw < F.x + F.w) {
                FreeRect N = {px + pw, F.y, F.x + F.w - (px + pw), F.h};
                if (N.w >= min_side && N.h >= min_side) new_rects.push_back(N);
            }
            
            // Remove the split rectangle
            B.free_rects[i] = B.free_rects.back();
            B.free_rects.pop_back();
        } else {
            ++i;
        }
    }
    
    // Add new rectangles and remove redundant ones
    for (const auto& N : new_rects) {
        bool contained = false;
        for (const auto& F : B.free_rects) {
            if (F.x <= N.x && F.y <= N.y && F.x + F.w >= N.x + N.w && F.y + F.h >= N.y + N.h) {
                contained = true;
                break;
            }
        }
        if (!contained) {
            for (int i = 0; i < (int)B.free_rects.size(); /* manual */) {
                const FreeRect& F = B.free_rects[i];
                if (N.x <= F.x && N.y <= F.y && N.x + N.w >= F.x + F.w && N.y + N.h >= F.y + F.h) {
                    B.free_rects[i] = B.free_rects.back();
                    B.free_rects.pop_back();
                } else {
                    ++i;
                }
            }
            B.free_rects.push_back(N);
        }
    }
    
    // Bound the number of free rectangles to prevent exponential growth
    if (B.free_rects.size() > 500) {
        sort(B.free_rects.begin(), B.free_rects.end(), [](const FreeRect& a, const FreeRect& b) {
            return (long long)a.w * a.h > (long long)b.w * b.h;
        });
        B.free_rects.resize(500);
    }
    
    // Update board limits for fast pruning
    B.max_w = B.max_h = 0;
    for (const auto& F : B.free_rects) {
        B.max_w = max(B.max_w, F.w);
        B.max_h = max(B.max_h, F.h);
    }
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int N;
    int board_width, board_height;
    if (!(cin >> N >> board_width >> board_height)) return 0;

    vector<Item> items(N);
    int min_side = 1e9;
    for (int i = 0; i < N; ++i) {
        items[i].id = i;
        cin >> items[i].w >> items[i].h;
        items[i].area = (long long)items[i].w * items[i].h;
        min_side = min({min_side, items[i].w, items[i].h});
    }

    // Sort items by Area descending, then by Max Dimension descending
    sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        if (a.area != b.area) return a.area > b.area;
        return max(a.w, a.h) > max(b.w, b.h);
    });

    vector<Board> boards;
    vector<Result> results(N);

    for (int i = 0; i < N; ++i) {
        const auto& item = items[i];
        
        int best_board = -1;
        int best_rect_idx = -1;
        int best_short_side = 2e9;
        long long best_area = 2e18;
        bool best_rotated = false;
        
        for (int b = 0; b < (int)boards.size(); ++b) {
            // Fast bounding box pruning
            if (!((boards[b].max_w >= item.w && boards[b].max_h >= item.h) || 
                  (boards[b].max_w >= item.h && boards[b].max_h >= item.w))) {
                continue;
            }
            
            for (int j = 0; j < (int)boards[b].free_rects.size(); ++j) {
                const auto& F = boards[b].free_rects[j];
                
                // Unrotated check
                if (F.w >= item.w && F.h >= item.h) {
                    int short_side = min(F.w - item.w, F.h - item.h);
                    long long area_fit = (long long)F.w * F.h - item.area;
                    if (short_side < best_short_side || (short_side == best_short_side && area_fit < best_area)) {
                        best_board = b; 
                        best_rect_idx = j;
                        best_short_side = short_side; 
                        best_area = area_fit;
                        best_rotated = false;
                    }
                }
                
                // Rotated check
                if (F.w >= item.h && F.h >= item.w) {
                    int short_side = min(F.w - item.h, F.h - item.w);
                    long long area_fit = (long long)F.w * F.h - item.area;
                    if (short_side < best_short_side || (short_side == best_short_side && area_fit < best_area)) {
                        best_board = b; 
                        best_rect_idx = j;
                        best_short_side = short_side; 
                        best_area = area_fit;
                        best_rotated = true;
                    }
                }
            }
        }
        
        // Create a new board if no fit is found
        if (best_board == -1) {
            boards.push_back(Board());
            best_board = (int)boards.size() - 1;
            boards.back().free_rects.push_back({0, 0, board_width, board_height});
            boards.back().max_w = board_width;
            boards.back().max_h = board_height;
            
            best_rect_idx = 0;
            // Determine best rotation for a fresh board
            if (board_width >= item.w && board_height >= item.h && board_width >= item.h && board_height >= item.w) {
                if ((board_width >= board_height) == (item.w >= item.h)) {
                    best_rotated = false;
                } else {
                    best_rotated = true;
                }
            } else if (board_width >= item.h && board_height >= item.w) {
                best_rotated = true;
            } else {
                best_rotated = false;
            }
        }
        
        int pw = best_rotated ? item.h : item.w;
        int ph = best_rotated ? item.w : item.h;
        int px = boards[best_board].free_rects[best_rect_idx].x;
        int py = boards[best_board].free_rects[best_rect_idx].y;
        
        results[item.id] = {best_board, px, py, best_rotated ? 1 : 0};
        
        place_rect(boards[best_board], px, py, pw, ph, min_side);
    }
    
    // Output the results
    cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        cout << results[i].board_idx << " " 
             << results[i].x << " " 
             << results[i].y << " " 
             << results[i].rotated << "\n";
    }

    return 0;
}
