use std::io::{self, BufRead};

#[derive(Clone)]
struct Rect {
    id: usize,
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

#[derive(Clone)]
struct PieceVariant {
    angle: f64,
    w: f64,
    h: f64,
    min_x: f64,
    min_y: f64,
}

struct PieceData {
    id: usize,
    variants: Vec<PieceVariant>,
    area: f64,
}

struct Stock {
    poly: Vec<(f64, f64)>,
    placed_pieces: Vec<Rect>,
    candidates: Vec<(f64, f64)>,
    s_min_x: f64,
    s_max_x: f64,
    s_min_y: f64,
    s_max_y: f64,
}

#[derive(Clone)]
struct Output {
    stock_index: usize,
    tx: f64,
    ty: f64,
    rotation: f64,
}

fn poly_area(poly: &[(f64, f64)]) -> f64 {
    let mut area = 0.0;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        area += (poly[j].0 + poly[i].0) * (poly[j].1 - poly[i].1);
        j = i;
    }
    (area / 2.0).abs()
}

fn stock_aabb(poly: &[(f64, f64)]) -> (f64, f64, f64, f64) {
    let mut min_x = f64::MAX;
    let mut max_x = f64::MIN;
    let mut min_y = f64::MAX;
    let mut max_y = f64::MIN;
    for &(x, y) in poly {
        if x < min_x { min_x = x; }
        if x > max_x { max_x = x; }
        if y < min_y { min_y = y; }
        if y > max_y { max_y = y; }
    }
    (min_x, max_x, min_y, max_y)
}

fn aabb_overlaps(min_x1: f64, min_y1: f64, max_x1: f64, max_y1: f64,
                 min_x2: f64, min_y2: f64, max_x2: f64, max_y2: f64) -> bool {
    let eps = 1e-6;
    if max_x1 <= min_x2 + eps || max_x2 <= min_x1 + eps { return false; }
    if max_y1 <= min_y2 + eps || max_y2 <= min_y1 + eps { return false; }
    true
}

fn ccw(a: (f64, f64), b: (f64, f64), c: (f64, f64)) -> f64 {
    (c.1 - a.1) * (b.0 - a.0) - (b.1 - a.1) * (c.0 - a.0)
}

fn segment_intersects_segment(a: (f64, f64), b: (f64, f64), c: (f64, f64), d: (f64, f64)) -> bool {
    let ccw1 = ccw(a, c, d);
    let ccw2 = ccw(b, c, d);
    let ccw3 = ccw(a, b, c);
    let ccw4 = ccw(a, b, d);
    
    if ccw1 * ccw2 < 0.0 && ccw3 * ccw4 < 0.0 {
        return true;
    }
    false
}

fn point_strictly_in_aabb(p: (f64, f64), min_x: f64, min_y: f64, max_x: f64, max_y: f64) -> bool {
    let eps = 1e-7;
    p.0 > min_x + eps && p.0 < max_x - eps && p.1 > min_y + eps && p.1 < max_y - eps
}

fn segment_intersects_aabb(p1: (f64, f64), p2: (f64, f64), min_x: f64, min_y: f64, max_x: f64, max_y: f64) -> bool {
    if point_strictly_in_aabb(p1, min_x, min_y, max_x, max_y) || point_strictly_in_aabb(p2, min_x, min_y, max_x, max_y) {
        return true;
    }
    
    let eps = 1e-7;
    let ix_min = min_x + eps;
    let ix_max = max_x - eps;
    let iy_min = min_y + eps;
    let iy_max = max_y - eps;
    
    if ix_min >= ix_max || iy_min >= iy_max { return false; }
    
    if segment_intersects_segment(p1, p2, (ix_min, iy_min), (ix_min, iy_max)) { return true; }
    if segment_intersects_segment(p1, p2, (ix_max, iy_min), (ix_max, iy_max)) { return true; }
    if segment_intersects_segment(p1, p2, (ix_min, iy_min), (ix_max, iy_min)) { return true; }
    if segment_intersects_segment(p1, p2, (ix_min, iy_max), (ix_max, iy_max)) { return true; }
    
    false
}

fn point_in_polygon(x: f64, y: f64, poly: &[(f64, f64)]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        
        let intersect = ((yi > y) != (yj > y)) && (x < (xj - xi) * (y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn aabb_in_poly(min_x: f64, min_y: f64, max_x: f64, max_y: f64, poly: &[(f64, f64)]) -> bool {
    let eps = 1e-7;
    let cx_min = min_x + eps;
    let cx_max = if max_x - eps > cx_min { max_x - eps } else { cx_min };
    let cy_min = min_y + eps;
    let cy_max = if max_y - eps > cy_min { max_y - eps } else { cy_min };
    
    let corners = [
        (cx_min, cy_min),
        (cx_max, cy_min),
        (cx_max, cy_max),
        (cx_min, cy_max),
        ((min_x + max_x) / 2.0, (min_y + max_y) / 2.0),
    ];
    for &c in &corners {
        if !point_in_polygon(c.0, c.1, poly) {
            return false;
        }
    }
    
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        if segment_intersects_aabb(poly[j], poly[i], min_x, min_y, max_x, max_y) {
            return false;
        }
        j = i;
    }
    true
}

fn try_place_piece_in_stock(piece: &PieceData, stock: &Stock) -> Option<(f64, f64, usize)> {
    let mut candidates = Vec::new();
    candidates.extend(&stock.candidates);
    
    candidates.push((stock.s_min_x, stock.s_min_y));
    for var in &piece.variants {
        candidates.push((stock.s_max_x - var.w, stock.s_min_y));
        candidates.push((stock.s_min_x, stock.s_max_y - var.h));
        candidates.push((stock.s_max_x - var.w, stock.s_max_y - var.h));
    }
    
    let step = if stock.poly.len() > 50 { stock.poly.len() / 10 } else { 1 };
    for (i, &(vx, vy)) in stock.poly.iter().enumerate() {
        if i % step != 0 { continue; }
        candidates.push((vx, vy));
        for var in &piece.variants {
            candidates.push((vx - var.w, vy));
            candidates.push((vx, vy - var.h));
            candidates.push((vx - var.w, vy - var.h));
        }
    }
    
    let step_x = (stock.s_max_x - stock.s_min_x) / 10.0;
    let step_y = (stock.s_max_y - stock.s_min_y) / 10.0;
    if step_x > 1e-3 && step_y > 1e-3 {
        for i in 0..=10 {
            for j in 0..=10 {
                candidates.push((stock.s_min_x + i as f64 * step_x, stock.s_min_y + j as f64 * step_y));
            }
        }
    }
    
    candidates.sort_by(|a, b| {
        a.1.partial_cmp(&b.1).unwrap_or(std::cmp::Ordering::Equal)
         .then(a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal))
    });
    
    let mut unique_cands = Vec::with_capacity(candidates.len());
    for c in candidates {
        if unique_cands.is_empty() {
            unique_cands.push(c);
        } else {
            let last = unique_cands.last().unwrap();
            if (last.0 - c.0).abs() > 1e-5 || (last.1 - c.1).abs() > 1e-5 {
                unique_cands.push(c);
            }
        }
    }
    
    for &(cx, cy) in &unique_cands {
        for (var_idx, var) in piece.variants.iter().enumerate() {
            if cx < stock.s_min_x || cy < stock.s_min_y || 
               cx + var.w > stock.s_max_x || cy + var.h > stock.s_max_y {
                continue;
            }
            
            let mut overlap = false;
            for placed_rect in stock.placed_pieces.iter().rev() {
                if aabb_overlaps(cx, cy, cx + var.w, cy + var.h,
                                 placed_rect.min_x, placed_rect.min_y,
                                 placed_rect.max_x, placed_rect.max_y) {
                    overlap = true;
                    break;
                }
            }
            if overlap { continue; }
            
            if aabb_in_poly(cx, cy, cx + var.w, cy + var.h, &stock.poly) {
                return Some((cx, cy, var_idx));
            }
        }
    }
    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines().filter_map(Result::ok).filter(|l| !l.trim().is_empty());
    
    let first_line = lines.next().expect("Missing first line");
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();
    
    let mut stock_poly = Vec::with_capacity(sv);
    for _ in 0..sv {
        let line = lines.next().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        stock_poly.push((parts[0].parse().unwrap(), parts[1].parse().unwrap()));
    }
    
    let mut pieces = Vec::with_capacity(np);
    for i in 0..np {
        let line = lines.next().unwrap();
        let pv: usize = line.trim().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let line = lines.next().unwrap();
            let parts: Vec<&str> = line.split_whitespace().collect();
            poly.push((parts[0].parse().unwrap(), parts[1].parse().unwrap()));
        }
        pieces.push((i, poly));
    }
    
    let mut piece_data = Vec::with_capacity(np);
    for (id, poly) in pieces {
        let area = poly_area(&poly);
        let mut variants = Vec::new();
        
        let num_rotations = 8;
        for i in 0..num_rotations {
            let angle = (i as f64) * 2.0 * std::f64::consts::PI / (num_rotations as f64);
            let cos_a = angle.cos();
            let sin_a = angle.sin();
            
            let mut min_x = f64::MAX;
            let mut max_x = f64::MIN;
            let mut min_y = f64::MAX;
            let mut max_y = f64::MIN;
            
            for &(x, y) in &poly {
                let rx = x * cos_a - y * sin_a;
                let ry = x * sin_a + y * cos_a;
                if rx < min_x { min_x = rx; }
                if rx > max_x { max_x = rx; }
                if ry < min_y { min_y = ry; }
                if ry > max_y { max_y = ry; }
            }
            
            variants.push(PieceVariant {
                angle,
                w: max_x - min_x,
                h: max_y - min_y,
                min_x,
                min_y,
            });
        }
        
        piece_data.push(PieceData {
            id,
            variants,
            area,
        });
    }
    
    piece_data.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap_or(std::cmp::Ordering::Equal));
    
    let (g_s_min_x, g_s_max_x, g_s_min_y, g_s_max_y) = stock_aabb(&stock_poly);
    
    let mut stocks: Vec<Stock> = Vec::new();
    let mut outputs = vec![None; np];
    
    for piece in &piece_data {
        let mut placed = false;
        
        for (stock_idx, stock) in stocks.iter_mut().enumerate() {
            let mut can_fit = false;
            for var in &piece.variants {
                if var.w <= stock.s_max_x - stock.s_min_x + 1e-5 && var.h <= stock.s_max_y - stock.s_min_y + 1e-5 {
                    can_fit = true;
                    break;
                }
            }
            if !can_fit { continue; }
            
            if let Some((cx, cy, var_idx)) = try_place_piece_in_stock(piece, stock) {
                let var = &piece.variants[var_idx];
                stock.placed_pieces.push(Rect {
                    id: piece.id,
                    min_x: cx,
                    min_y: cy,
                    max_x: cx + var.w,
                    max_y: cy + var.h,
                });
                stock.candidates.push((cx + var.w, cy));
                stock.candidates.push((cx, cy + var.h));
                stock.candidates.push((cx + var.w, cy + var.h));
                
                outputs[piece.id] = Some(Output {
                    stock_index: stock_idx,
                    tx: cx - var.min_x,
                    ty: cy - var.min_y,
                    rotation: var.angle,
                });
                placed = true;
                break;
            }
        }
        
        if !placed {
            let mut new_stock = Stock {
                poly: stock_poly.clone(),
                placed_pieces: Vec::new(),
                candidates: vec![],
                s_min_x: g_s_min_x,
                s_max_x: g_s_max_x,
                s_min_y: g_s_min_y,
                s_max_y: g_s_max_y,
            };
            
            if let Some((cx, cy, var_idx)) = try_place_piece_in_stock(piece, &new_stock) {
                let var = &piece.variants[var_idx];
                new_stock.placed_pieces.push(Rect {
                    id: piece.id,
                    min_x: cx,
                    min_y: cy,
                    max_x: cx + var.w,
                    max_y: cy + var.h,
                });
                new_stock.candidates.push((cx + var.w, cy));
                new_stock.candidates.push((cx, cy + var.h));
                new_stock.candidates.push((cx + var.w, cy + var.h));
                
                outputs[piece.id] = Some(Output {
                    stock_index: stocks.len(),
                    tx: cx - var.min_x,
                    ty: cy - var.min_y,
                    rotation: var.angle,
                });
                stocks.push(new_stock);
            } else {
                // Fallback
                let var = &piece.variants[0];
                let cx = g_s_min_x + (g_s_max_x - g_s_min_x - var.w) / 2.0;
                let cy = g_s_min_y + (g_s_max_y - g_s_min_y - var.h) / 2.0;
                new_stock.placed_pieces.push(Rect {
                    id: piece.id,
                    min_x: cx,
                    min_y: cy,
                    max_x: cx + var.w,
                    max_y: cy + var.h,
                });
                outputs[piece.id] = Some(Output {
                    stock_index: stocks.len(),
                    tx: cx - var.min_x,
                    ty: cy - var.min_y,
                    rotation: var.angle,
                });
                stocks.push(new_stock);
            }
        }
    }
    
    println!("{}", stocks.len());
    for out in outputs {
        if let Some(o) = out {
            println!("{} {} {} {}", o.stock_index, o.tx, o.ty, o.rotation);
        } else {
            println!("0 0.0 0.0 0.0");
        }
    }
}
