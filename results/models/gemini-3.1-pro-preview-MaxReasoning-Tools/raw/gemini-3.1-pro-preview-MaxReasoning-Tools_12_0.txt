using System;
using System.IO;
using System.Globalization;
using System.Collections.Generic;
using System.Numerics;
using System.Runtime.CompilerServices;
using System.Linq;
using System.Diagnostics;

namespace BinPacking
{
    public struct Quat
    {
        public float W, X, Y, Z;
        public Quat(float w, float x, float y, float z) { W = w; X = x; Y = y; Z = z; }
    }

    class Candidate
    {
        public Quat q;
        public float w, h, d, minX, minY, minZ;
    }

    class Box
    {
        public float x, y, z, w, h, d;
        public Box(float x, float y, float z, float w, float h, float d)
        {
            this.x = x; this.y = y; this.z = z; this.w = w; this.h = h; this.d = d;
        }
    }

    class Program
    {
        static void Main()
        {
            var sw = Stopwatch.StartNew();
            using var reader = new StreamReader(Console.OpenStandardInput(), bufferSize: 65536);
            string first = ReadLine(reader);
            if (first == null) return;
            string[] firstLine = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            int V = int.Parse(firstLine[0]);
            int F = int.Parse(firstLine[1]);

            float[] X = new float[V];
            float[] Y = new float[V];
            float[] Z = new float[V];

            for (int i = 0; i < V; i++)
            {
                string[] parts = ReadLine(reader).Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
                X[i] = float.Parse(parts[0], CultureInfo.InvariantCulture);
                Y[i] = float.Parse(parts[1], CultureInfo.InvariantCulture);
                Z[i] = float.Parse(parts[2], CultureInfo.InvariantCulture);
            }

            for (int i = 0; i < F; i++)
            {
                ReadLine(reader); // Skip face indices as AABB doesn't need them
            }

            string[] lastLine = ReadLine(reader).Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            float W = float.Parse(lastLine[0], CultureInfo.InvariantCulture);
            float H = float.Parse(lastLine[1], CultureInfo.InvariantCulture);
            float D = float.Parse(lastLine[2], CultureInfo.InvariantCulture);

            GetAABB(X, Y, Z, 1, 0, 0, 0, out float minX, out float maxX, out float minY, out float maxY, out float minZ, out float maxZ);
            float volS = (maxX - minX) * (maxY - minY) * (maxZ - minZ);
            float volC = W * H * D;
            long estN = (long)(volC / volS);

            // Adaptive Strategy based on problem scale
            if (estN <= 5000)
            {
                // Trivial / Medium scale: Maximal Empty Spaces (MES) heuristic allows mixed-orientations to fit in gaps
                SolveMES(W, H, D, X, Y, Z, sw);
            }
            else
            {
                // Large / Extreme scale: Pure optimal grid packing guarantees scaling up to millions of items
                SolveGrid(W, H, D, X, Y, Z, 50000);
            }
        }

        static string ReadLine(StreamReader reader)
        {
            string line;
            while ((line = reader.ReadLine()) != null)
            {
                if (!string.IsNullOrWhiteSpace(line)) return line;
            }
            return null;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        static void GetAABB(float[] X, float[] Y, float[] Z, float qw, float qx, float qy, float qz, out float minX, out float maxX, out float minY, out float maxY, out float minZ, out float maxZ)
        {
            int count = X.Length;
            int vCount = Vector<float>.Count;
            int i = 0;

            var minXV = new Vector<float>(float.MaxValue);
            var maxXV = new Vector<float>(float.MinValue);
            var minYV = new Vector<float>(float.MaxValue);
            var maxYV = new Vector<float>(float.MinValue);
            var minZV = new Vector<float>(float.MaxValue);
            var maxZV = new Vector<float>(float.MinValue);

            var vqw = new Vector<float>(qw);
            var vqx = new Vector<float>(qx);
            var vqy = new Vector<float>(qy);
            var vqz = new Vector<float>(qz);

            // High Performance: SIMD vectorized bounds calculation
            for (; i <= count - vCount; i += vCount)
            {
                var px = new Vector<float>(X, i);
                var py = new Vector<float>(Y, i);
                var pz = new Vector<float>(Z, i);

                var tx = 2f * (vqy * pz - vqz * py);
                var ty = 2f * (vqz * px - vqx * pz);
                var tz = 2f * (vqx * py - vqy * px);

                var pnx = px + vqw * tx + vqy * tz - vqz * ty;
                var pny = py + vqw * ty + vqz * tx - vqx * tz;
                var pnz = pz + vqw * tz + vqx * ty - vqy * tx;

                minXV = Vector.Min(minXV, pnx);
                maxXV = Vector.Max(maxXV, pnx);
                minYV = Vector.Min(minYV, pny);
                maxYV = Vector.Max(maxYV, pny);
                minZV = Vector.Min(minZV, pnz);
                maxZV = Vector.Max(maxZV, pnz);
            }

            minX = float.MaxValue; maxX = float.MinValue;
            minY = float.MaxValue; maxY = float.MinValue;
            minZ = float.MaxValue; maxZ = float.MinValue;

            for (int j = 0; j < vCount; j++)
            {
                if (minXV[j] < minX) minX = minXV[j];
                if (maxXV[j] > maxX) maxX = maxXV[j];
                if (minYV[j] < minY) minY = minYV[j];
                if (maxYV[j] > maxY) maxY = maxYV[j];
                if (minZV[j] < minZ) minZ = minZV[j];
                if (maxZV[j] > maxZ) maxZ = maxZV[j];
            }

            // Fallback for remaining elements
            for (; i < count; i++)
            {
                float px = X[i], py = Y[i], pz = Z[i];
                float tx = 2f * (qy * pz - qz * py);
                float ty = 2f * (qz * px - qx * pz);
                float tz = 2f * (qx * py - qy * px);

                float pnx = px + qw * tx + qy * tz - qz * ty;
                float pny = py + qw * ty + qz * tx - qx * tz;
                float pnz = pz + qw * tz + qx * ty - qy * tx;

                if (pnx < minX) minX = pnx;
                if (pnx > maxX) maxX = pnx;
                if (pny < minY) minY = pny;
                if (pny > maxY) maxY = pny;
                if (pnz < minZ) minZ = pnz;
                if (pnz > maxZ) maxZ = pnz;
            }
        }

        static Quat EulerToQuat(float pitch, float yaw, float roll)
        {
            float cy = (float)Math.Cos(yaw * 0.5f);
            float sy = (float)Math.Sin(yaw * 0.5f);
            float cp = (float)Math.Cos(pitch * 0.5f);
            float sp = (float)Math.Sin(pitch * 0.5f);
            float cr = (float)Math.Cos(roll * 0.5f);
            float sr = (float)Math.Sin(roll * 0.5f);

            float qw = cr * cp * cy + sr * sp * sy;
            float qx = sr * cp * cy - cr * sp * sy;
            float qy = cr * sp * cy + sr * cp * sy;
            float qz = cr * cp * sy - sr * sp * cy;

            return new Quat(qw, qx, qy, qz);
        }

        static List<Quat> GenerateRotations(int count)
        {
            var list = new List<Quat>();
            
            // First 24 are exact standard cubic orientations
            for (int x = 0; x < 4; x++)
            {
                for (int y = 0; y < 4; y++)
                {
                    for (int z = 0; z < 4; z++)
                    {
                        var q = EulerToQuat(x * (float)Math.PI / 2, y * (float)Math.PI / 2, z * (float)Math.PI / 2);
                        bool exists = false;
                        foreach (var existing in list)
                        {
                            float dot = Math.Abs(q.W * existing.W + q.X * existing.X + q.Y * existing.Y + q.Z * existing.Z);
                            if (dot > 0.999f) { exists = true; break; }
                        }
                        if (!exists) list.Add(q);
                    }
                }
            }

            Random rnd = new Random(42);
            for (int i = list.Count; i < count; i++)
            {
                double u1 = rnd.NextDouble();
                double u2 = rnd.NextDouble();
                double u3 = rnd.NextDouble();

                double sq1u1 = Math.Sqrt(1 - u1);
                double squ1 = Math.Sqrt(u1);

                float qw = (float)(sq1u1 * Math.Sin(2 * Math.PI * u2));
                float qx = (float)(sq1u1 * Math.Cos(2 * Math.PI * u2));
                float qy = (float)(squ1 * Math.Sin(2 * Math.PI * u3));
                float qz = (float)(squ1 * Math.Cos(2 * Math.PI * u3));

                list.Add(new Quat(qw, qx, qy, qz));
            }
            return list;
        }

        static void SolveGrid(float W, float H, float D, float[] X, float[] Y, float[] Z, int numRotations)
        {
            var rots = GenerateRotations(numRotations);
            long bestCount = 0;
            Candidate bestCand = null;

            foreach (var q in rots)
            {
                GetAABB(X, Y, Z, q.W, q.X, q.Y, q.Z, out float minX, out float maxX, out float minY, out float maxY, out float minZ, out float maxZ);
                float w = maxX - minX;
                float h = maxY - minY;
                float d = maxZ - minZ;

                long nx = (long)(W / w);
                long ny = (long)(H / h);
                long nz = (long)(D / d);
                long count = nx * ny * nz;

                if (count > bestCount)
                {
                    bestCount = count;
                    bestCand = new Candidate { q = q, w = w, h = h, d = d, minX = minX, minY = minY, minZ = minZ };
                }
            }

            if (bestCount == 0 || bestCand == null)
            {
                Console.WriteLine(0);
                return;
            }

            using var stdout = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
            stdout.WriteLine(bestCount);
            float bestW = bestCand.w;
            float bestH = bestCand.h;
            float bestD = bestCand.d;
            long bnx = (long)(W / bestW);
            long bny = (long)(H / bestH);
            long bnz = (long)(D / bestD);

            for (long i = 0; i < bnx; i++)
            {
                for (long j = 0; j < bny; j++)
                {
                    for (long k = 0; k < bnz; k++)
                    {
                        float bx = i * bestW;
                        float by = j * bestH;
                        float bz = k * bestD;
                        float tx = bx - bestCand.minX;
                        float ty = by - bestCand.minY;
                        float tz = bz - bestCand.minZ;
                        stdout.WriteLine(string.Format(CultureInfo.InvariantCulture, "{0:F6} {1:F6} {2:F6} {3:F6} {4:F6} {5:F6} {6:F6}", tx, ty, tz, bestCand.q.W, bestCand.q.X, bestCand.q.Y, bestCand.q.Z));
                    }
                }
            }
        }

        static void SolveMES(float W, float H, float D, float[] X, float[] Y, float[] Z, Stopwatch sw)
        {
            var rots = GenerateRotations(1000);
            var candidates = new List<Candidate>(rots.Count);
            foreach (var q in rots)
            {
                GetAABB(X, Y, Z, q.W, q.X, q.Y, q.Z, out float minX, out float maxX, out float minY, out float maxY, out float minZ, out float maxZ);
                candidates.Add(new Candidate { q = q, w = maxX - minX, h = maxY - minY, d = maxZ - minZ, minX = minX, minY = minY, minZ = minZ });
            }

            var spaces = new List<Box> { new Box(0, 0, 0, W, H, D) };
            var placements = new List<string>();

            float minCandW = candidates.Min(c => c.w);
            float minCandH = candidates.Min(c => c.h);
            float minCandD = candidates.Min(c => c.d);

            while (spaces.Count > 0)
            {
                if (sw.ElapsedMilliseconds > 280000) break; // 4.6 mins timeout threshold

                Box bestSpace = null;
                Candidate bestCand = null;
                float bestScore = float.MaxValue;

                // Pick the placement closest to origin to remain packed densely
                foreach (var s in spaces)
                {
                    foreach (var c in candidates)
                    {
                        if (c.w <= s.w + 1e-5f && c.h <= s.h + 1e-5f && c.d <= s.d + 1e-5f)
                        {
                            float score = s.x + s.y + s.z;
                            if (score < bestScore)
                            {
                                bestScore = score;
                                bestSpace = s;
                                bestCand = c;
                            }
                        }
                    }
                }

                if (bestSpace == null)
                    break;

                float px = bestSpace.x;
                float py = bestSpace.y;
                float pz = bestSpace.z;
                float tx = px - bestCand.minX;
                float ty = py - bestCand.minY;
                float tz = pz - bestCand.minZ;
                
                placements.Add(string.Format(CultureInfo.InvariantCulture, "{0:F6} {1:F6} {2:F6} {3:F6} {4:F6} {5:F6} {6:F6}", tx, ty, tz, bestCand.q.W, bestCand.q.X, bestCand.q.Y, bestCand.q.Z));

                var newSpaces = new List<Box>();
                foreach (var space in spaces)
                {
                    if (Intersects(space, px, py, pz, bestCand.w, bestCand.h, bestCand.d))
                    {
                        if (px > space.x)
                            newSpaces.Add(new Box(space.x, space.y, space.z, px - space.x, space.h, space.d));
                        if (px + bestCand.w < space.x + space.w)
                            newSpaces.Add(new Box(px + bestCand.w, space.y, space.z, (space.x + space.w) - (px + bestCand.w), space.h, space.d));
                        if (py > space.y)
                            newSpaces.Add(new Box(space.x, space.y, space.z, space.w, py - space.y, space.d));
                        if (py + bestCand.h < space.y + space.h)
                            newSpaces.Add(new Box(space.x, py + bestCand.h, space.z, space.w, (space.y + space.h) - (py + bestCand.h), space.d));
                        if (pz > space.z)
                            newSpaces.Add(new Box(space.x, space.y, space.z, space.w, space.h, pz - space.z));
                        if (pz + bestCand.d < space.z + space.d)
                            newSpaces.Add(new Box(space.x, space.y, pz + bestCand.d, space.w, space.h, (space.z + space.d) - (pz + bestCand.d)));
                    }
                    else
                    {
                        newSpaces.Add(space);
                    }
                }

                // Cull unviable or redundant spaces to keep MES scalable
                var pruned = new List<Box>();
                for (int i = 0; i < newSpaces.Count; i++)
                {
                    var s1 = newSpaces[i];
                    bool canFitAny = false;
                    foreach (var c in candidates)
                    {
                        if (c.w <= s1.w + 1e-5f && c.h <= s1.h + 1e-5f && c.d <= s1.d + 1e-5f)
                        {
                            canFitAny = true; break;
                        }
                    }
                    if (!canFitAny) continue;

                    bool contained = false;
                    for (int j = 0; j < newSpaces.Count; j++)
                    {
                        if (i != j && Contains(newSpaces[j], s1))
                        {
                            contained = true;
                            break;
                        }
                    }
                    if (!contained) pruned.Add(s1);
                }

                if (pruned.Count > 1000)
                {
                    pruned = pruned.OrderBy(s => s.x + s.y + s.z).Take(1000).ToList();
                }

                spaces = pruned;
            }

            using var stdout = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536);
            stdout.WriteLine(placements.Count);
            foreach (var p in placements)
                stdout.WriteLine(p);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        static bool Intersects(Box b, float x, float y, float z, float w, float h, float d)
        {
            return x < b.x + b.w && x + w > b.x &&
                   y < b.y + b.h && y + h > b.y &&
                   z < b.z + b.d && z + d > b.z;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        static bool Contains(Box outer, Box inner)
        {
            return inner.x >= outer.x - 1e-5f &&
                   inner.y >= outer.y - 1e-5f &&
                   inner.z >= outer.z - 1e-5f &&
                   inner.x + inner.w <= outer.x + outer.w + 1e-5f &&
                   inner.y + inner.h <= outer.y + outer.h + 1e-5f &&
                   inner.z + inner.d <= outer.z + outer.d + 1e-5f;
        }
    }
}
