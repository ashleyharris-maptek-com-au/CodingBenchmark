#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>

#include <random>

#include <iomanip>

using namespace std;

struct Point {
    double x, y, z;
    Point operator-(const Point& o) const { return {x-o.x, y-o.y, z-o.z}; }
    Point operator+(const Point& o) const { return {x+o.x, y+o.y, z+o.z}; }
    Point cross(const Point& o) const { return {y*o.z - z*o.y, z*o.x - x*o.z, x*o.y - y*o.x}; }
    double dot(const Point& o) const { return x*o.x + y*o.y + z*o.z; }
    double length() const { return sqrt(x*x + y*y + z*z); }
    Point normalize() const {
        double len = length();
        return len > 1e-9 ? Point{x/len, y/len, z/len} : Point{0,0,0};
    }
};

struct Face {
    int v[3];
    Point n;
    double d;
    bool dead = false;
    void calc_normal(const vector<Point>& pts) {
        n = (pts[v[1]] - pts[v[0]]).cross(pts[v[2]] - pts[v[0]]).normalize();
        d = n.dot(pts[v[0]]);
    }
};

struct Interval { double min_v, max_v; };

double edge_length;
double R_circum;
Point verts[4];
vector<Point> sat_axes;
vector<Interval> base_intervals;
vector<Face> final_faces;

struct BoundingBox {
    double min_x=1e18, max_x=-1e18;
    double min_y=1e18, max_y=-1e18;
    double min_z=1e18, max_z=-1e18;
} bbox;

void build_hull(vector<Point>& pts) {
    int n = pts.size();
    if (n < 4) return;
    mt19937 rng(42);
    shuffle(pts.begin(), pts.end(), rng);

    int p1=0, p2=1, p3=-1, p4=-1;
    for (int i=2; i<n; ++i) {
        if ((pts[1]-pts[0]).cross(pts[i]-pts[0]).length() > 1e-7) {
            p3 = i; swap(pts[2], pts[i]); break;
        }
    }
    if (p3 != -1) {
        for (int i=3; i<n; ++i) {
            Point n_temp = (pts[1]-pts[0]).cross(pts[2]-pts[0]);
            if (abs(n_temp.dot(pts[i]-pts[0])) > 1e-7) {
                p4 = i; swap(pts[3], pts[i]); break;
            }
        }
    }
    
    vector<Face> faces;
    if (p4 != -1) {
        faces.push_back({0, 1, 2, {0,0,0}, 0, false}); faces.back().calc_normal(pts);
        faces.push_back({0, 2, 3, {0,0,0}, 0, false}); faces.back().calc_normal(pts);
        faces.push_back({0, 3, 1, {0,0,0}, 0, false}); faces.back().calc_normal(pts);
        faces.push_back({1, 3, 2, {0,0,0}, 0, false}); faces.back().calc_normal(pts);
        
        Point center = {(pts[0].x+pts[1].x+pts[2].x+pts[3].x)/4, 
                        (pts[0].y+pts[1].y+pts[2].y+pts[3].y)/4, 
                        (pts[0].z+pts[1].z+pts[2].z+pts[3].z)/4};
        for (auto& f : faces) {
            if (f.n.dot(center) > f.d) {
                swap(f.v[1], f.v[2]);
                f.calc_normal(pts);
            }
        }
        
        for (int i = 4; i < n; ++i) {
            vector<int> visible;
            for (size_t j = 0; j < faces.size(); ++j) {
                if (!faces[j].dead && faces[j].n.dot(pts[i]) > faces[j].d + 1e-7) {
                    visible.push_back((int)j);
                }
            }
            if (visible.empty()) continue;
            
            map<pair<int,int>, int> edge_counts;
            for (int j : visible) {
                faces[j].dead = true;
                edge_counts[{faces[j].v[0], faces[j].v[1]}]++;
                edge_counts[{faces[j].v[1], faces[j].v[2]}]++;
                edge_counts[{faces[j].v[2], faces[j].v[0]}]++;
            }
            
            for (int j : visible) {
                for (int k = 0; k < 3; ++k) {
                    int a = faces[j].v[k];
                    int b = faces[j].v[(k+1)%3];
                    if (edge_counts.find({b, a}) == edge_counts.end()) {
                        Face new_f = {b, a, i, {0,0,0}, 0, false};
                        new_f.calc_normal(pts);
                        faces.push_back(new_f);
                    }
                }
            }
        }
    }
    for (auto& f : faces) {
        if (!f.dead) final_faces.push_back(f);
    }
}

bool is_inside(const Point& p) {
    if (p.x < bbox.min_x || p.x > bbox.max_x ||
        p.y < bbox.min_y || p.y > bbox.max_y ||
        p.z < bbox.min_z || p.z > bbox.max_z) return false;
    if (final_faces.empty()) return true; // Fallback
    for (const auto& f : final_faces) {
        if (f.n.dot(p) > f.d + 1e-7) return false;
    }
    return true;
}

bool intersect(const Point& c1, const Point& c2) {
    Point d = c2 - c1;
    for (size_t i = 0; i < sat_axes.size(); ++i) {
        double proj = d.dot(sat_axes[i]);
        double min1 = base_intervals[i].min_v;
        double max1 = base_intervals[i].max_v;
        double min2 = base_intervals[i].min_v + proj;
        double max2 = base_intervals[i].max_v + proj;
        if (max1 <= min2 + 1e-7 || max2 <= min1 + 1e-7) {
            return false;
        }
    }
    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    
    auto start_time = chrono::high_resolution_clock::now();
    auto time_left = [&]() {
        auto now = chrono::high_resolution_clock::now();
        return 28.0 - chrono::duration<double>(now - start_time).count();
    };

    int V;
    if (!(cin >> V >> edge_length)) return 0;
    
    vector<Point> input_pts(V);
    for (int i=0; i<V; ++i) {
        cin >> input_pts[i].x >> input_pts[i].y >> input_pts[i].z;
        bbox.min_x = min(bbox.min_x, input_pts[i].x);
        bbox.max_x = max(bbox.max_x, input_pts[i].x);
        bbox.min_y = min(bbox.min_y, input_pts[i].y);
        bbox.max_y = max(bbox.max_y, input_pts[i].y);
        bbox.min_z = min(bbox.min_z, input_pts[i].z);
        bbox.max_z = max(bbox.max_z, input_pts[i].z);
    }
    
    build_hull(input_pts);
    
    double e = edge_length;
    R_circum = e * sqrt(3.0 / 8.0);
    double r_in = e / sqrt(24.0);

    verts[0] = {0, 0, R_circum};
    verts[1] = {e / sqrt(3.0), 0, -r_in};
    verts[2] = {-e / sqrt(12.0), e / 2.0, -r_in};
    verts[3] = {-e / sqrt(12.0), -e / 2.0, -r_in};
    
    Point faces_n[4];
    faces_n[0] = (verts[1] - verts[0]).cross(verts[2] - verts[0]).normalize();
    faces_n[1] = (verts[2] - verts[0]).cross(verts[3] - verts[0]).normalize();
    faces_n[2] = (verts[3] - verts[0]).cross(verts[1] - verts[0]).normalize();
    faces_n[3] = (verts[1] - verts[3]).cross(verts[2] - verts[3]).normalize();

    Point edges_n[6];
    edges_n[0] = (verts[1] - verts[0]).normalize();
    edges_n[1] = (verts[2] - verts[0]).normalize();
    edges_n[2] = (verts[3] - verts[0]).normalize();
    edges_n[3] = (verts[2] - verts[1]).normalize();
    edges_n[4] = (verts[3] - verts[2]).normalize();
    edges_n[5] = (verts[1] - verts[3]).normalize();

    for (int i=0; i<4; ++i) sat_axes.push_back(faces_n[i]);
    for (int i=0; i<6; ++i) {
        for (int j=i+1; j<6; ++j) {
            Point cross = edges_n[i].cross(edges_n[j]);
            if (cross.length() > 1e-5) {
                sat_axes.push_back(cross.normalize());
            }
        }
    }

    for (const Point& axis : sat_axes) {
        double min_v = 1e18, max_v = -1e18;
        for (int i=0; i<4; ++i) {
            double proj = verts[i].dot(axis);
            min_v = min(min_v, proj);
            max_v = max(max_v, proj);
        }
        base_intervals.push_back({min_v, max_v});
    }

    double v_box = (bbox.max_x - bbox.min_x) * (bbox.max_y - bbox.min_y) * (bbox.max_z - bbox.min_z);
    double target_tests = 1.5e7;
    double S = cbrt(v_box / target_tests);
    S = max(0.2 * e, min(S, 1.2 * e));
    
    double cell_size = 2.0 * R_circum;
    int nx = ceil((bbox.max_x - bbox.min_x) / cell_size) + 1;
    int ny = ceil((bbox.max_y - bbox.min_y) / cell_size) + 1;
    int nz = ceil((bbox.max_z - bbox.min_z) / cell_size) + 1;

    while ((long long)nx * ny * nz > 20000000) {
        cell_size *= 1.5;
        nx = ceil((bbox.max_x - bbox.min_x) / cell_size) + 1;
        ny = ceil((bbox.max_y - bbox.min_y) / cell_size) + 1;
        nz = ceil((bbox.max_z - bbox.min_z) / cell_size) + 1;
    }

    vector<int> head(nx * ny * nz, -1);
    struct Node { Point c; int next; };
    vector<Node> nodes;
    nodes.reserve(2000000);

    auto get_cell_idx = [&](int cx, int cy, int cz) {
        if (cx < 0 || cx >= nx || cy < 0 || cy >= ny || cz < 0 || cz >= nz) return -1;
        return cx + cy * nx + cz * nx * ny;
    };

    vector<Point> placed;
    
    for (double z = bbox.min_z; z <= bbox.max_z; z += S) {
        for (double y = bbox.min_y; y <= bbox.max_y; y += S) {
            for (double x = bbox.min_x; x <= bbox.max_x; x += S) {
                if (time_left() < 0) goto end_search;
                
                Point p = {x, y, z};
                bool ok = true;
                for (int i=0; i<4; ++i) {
                    if (!is_inside(p + verts[i])) { ok = false; break; }
                }
                if (!ok) continue;

                int min_cx = floor((p.x - R_circum - bbox.min_x) / cell_size);
                int max_cx = floor((p.x + R_circum - bbox.min_x) / cell_size);
                int min_cy = floor((p.y - R_circum - bbox.min_y) / cell_size);
                int max_cy = floor((p.y + R_circum - bbox.min_y) / cell_size);
                int min_cz = floor((p.z - R_circum - bbox.min_z) / cell_size);
                int max_cz = floor((p.z + R_circum - bbox.min_z) / cell_size);
                
                bool intersects = false;
                for (int cx = min_cx; cx <= max_cx && !intersects; ++cx) {
                    for (int cy = min_cy; cy <= max_cy && !intersects; ++cy) {
                        for (int cz = min_cz; cz <= max_cz && !intersects; ++cz) {
                            int idx = get_cell_idx(cx, cy, cz);
                            if (idx != -1) {
                                int curr = head[idx];
                                while (curr != -1) {
                                    if (intersect(p, nodes[curr].c)) {
                                        intersects = true; break;
                                    }
                                    curr = nodes[curr].next;
                                }
                            }
                        }
                    }
                }
                
                if (!intersects) {
                    placed.push_back(p);
                    int cx = floor((p.x - bbox.min_x) / cell_size);
                    int cy = floor((p.y - bbox.min_y) / cell_size);
                    int cz = floor((p.z - bbox.min_z) / cell_size);
                    int idx = get_cell_idx(cx, cy, cz);
                    if (idx != -1) {
                        nodes.push_back({p, head[idx]});
                        head[idx] = nodes.size() - 1;
                    }
                }
            }
        }
    }
end_search:
    
    cout << placed.size() << "\n";
    cout << fixed << setprecision(6);
    for (const Point& p : placed) {
        cout << p.x << " " << p.y << " " << p.z << " 0.0 0.0 0.0\n";
    }

    return 0;
}
