using System;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

class Program
{
    struct Vector3
    {
        public double x, y, z;
        public double Magnitude => Math.Sqrt(x * x + y * y + z * z);
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3 { x = a.y * b.z - a.z * b.y, y = a.z * b.x - a.x * b.z, z = a.x * b.y - a.y * b.x };
        public static double Dot(Vector3 a, Vector3 b) => a.x * b.x + a.y * b.y + a.z * b.z;
        public static Vector3 Add(Vector3 a, Vector3 b) => new Vector3 { x = a.x + b.x, y = a.y + b.y, z = a.z + b.z };
        public static Vector3 Sub(Vector3 a, Vector3 b) => new Vector3 { x = a.x - b.x, y = a.y - b.y, z = a.z - b.z };
        public static Vector3 Mul(Vector3 a, double d) => new Vector3 { x = a.x * d, y = a.y * d, z = a.z * d };
    }

    class Orbit
    {
        public Vector3 r;
        public Vector3 v;
        public Vector3 h;
        public double a;
        public Vector3 e_vec;
        public double Vc;

        public Orbit(Vector3 r, Vector3 v, double mu)
        {
            this.r = r;
            this.v = v;
            this.h = Vector3.Cross(r, v);
            double r_mag = r.Magnitude;
            double v_mag = v.Magnitude;
            double E = v_mag * v_mag / 2 - mu / r_mag;
            if (Math.Abs(E) < 1e-12) E = E < 0 ? -1e-12 : 1e-12;
            this.a = -mu / (2 * E);
            Vector3 v_cross_h = Vector3.Cross(v, h);
            this.e_vec = Vector3.Sub(Vector3.Mul(v_cross_h, 1 / mu), Vector3.Mul(r, 1 / r_mag));
            this.Vc = Math.Sqrt(mu / Math.Abs(this.a));
        }
    }

    static double Distance(Orbit o1, Orbit o2)
    {
        double h_mag = o1.h.Magnitude * o2.h.Magnitude;
        double h_dot = h_mag > 1e-12 ? Vector3.Dot(o1.h, o2.h) / h_mag : 1.0;
        h_dot = Math.Max(-1.0, Math.Min(1.0, h_dot));
        double theta = Math.Acos(h_dot);
        double avg_Vc = (o1.Vc + o2.Vc) / 2.0;
        double dv_plane = 2.0 * avg_Vc * Math.Sin(theta / 2.0);

        double dv_alt = Math.Abs(o1.Vc - o2.Vc);

        double e_diff = Vector3.Sub(o1.e_vec, o2.e_vec).Magnitude;
        double dv_ecc = avg_Vc * e_diff;

        return dv_plane + dv_alt + dv_ecc;
    }

    static void Main(string[] args)
    {
        string line1 = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(line1)) return;
        string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int N = int.Parse(parts1[0]);
        double mu = double.Parse(parts1[1], CultureInfo.InvariantCulture);

        string line2 = Console.ReadLine();
        string[] parts2 = line2.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        Orbit startOrbit = new Orbit(
            new Vector3 { x = double.Parse(parts2[0], CultureInfo.InvariantCulture), y = double.Parse(parts2[1], CultureInfo.InvariantCulture), z = double.Parse(parts2[2], CultureInfo.InvariantCulture) },
            new Vector3 { x = double.Parse(parts2[3], CultureInfo.InvariantCulture), y = double.Parse(parts2[4], CultureInfo.InvariantCulture), z = double.Parse(parts2[5], CultureInfo.InvariantCulture) },
            mu
        );

        Orbit[] stations = new Orbit[N];
        for (int i = 0; i < N; i++)
        {
            string line = Console.ReadLine();
            string[] parts = line.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            stations[i] = new Orbit(
                new Vector3 { x = double.Parse(parts[0], CultureInfo.InvariantCulture), y = double.Parse(parts[1], CultureInfo.InvariantCulture), z = double.Parse(parts[2], CultureInfo.InvariantCulture) },
                new Vector3 { x = double.Parse(parts[3], CultureInfo.InvariantCulture), y = double.Parse(parts[4], CultureInfo.InvariantCulture), z = double.Parse(parts[5], CultureInfo.InvariantCulture) },
                mu
            );
        }

        double[,] D = new double[N + 1, N + 1];
        for (int i = 0; i < N; i++)
        {
            D[0, i + 1] = Distance(startOrbit, stations[i]);
            D[i + 1, 0] = D[0, i + 1];
        }
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                if (i == j) D[i + 1, j + 1] = 0;
                else D[i + 1, j + 1] = Distance(stations[i], stations[j]);
            }
        }

        int[] bestPath = null;
        double bestCost = double.MaxValue;

        for (int startStation = 1; startStation <= N; startStation++)
        {
            int[] path = new int[N];
            bool[] visited = new bool[N + 1];
            path[0] = startStation;
            visited[startStation] = true;
            int curr = startStation;
            double cost = D[0, startStation];

            for (int step = 1; step < N; step++)
            {
                double minD = double.MaxValue;
                int next = -1;
                for (int i = 1; i <= N; i++)
                {
                    if (!visited[i] && D[curr, i] < minD)
                    {
                        minD = D[curr, i];
                        next = i;
                    }
                }
                path[step] = next;
                visited[next] = true;
                cost += minD;
                curr = next;
            }

            if (cost < bestCost)
            {
                bestCost = cost;
                bestPath = (int[])path.Clone();
            }
        }

        if (bestPath != null)
        {
            bestPath = Optimize2Opt(bestPath, D, N);
            bestCost = CalculateCost(bestPath, D, N);
        }

        if (N <= 10)
        {
            int[] arr = new int[N];
            for (int i = 0; i < N; i++) arr[i] = i + 1;
            Permute(arr, 0, N, D, ref bestCost, ref bestPath);
        }
        else
        {
            Random rnd = new Random(42);
            int maxIterations = 200000;
            for (int iter = 0; iter < maxIterations; iter++)
            {
                int[] currentPath = new int[N];
                for (int i = 0; i < N; i++) currentPath[i] = i + 1;

                for (int i = N - 1; i > 0; i--)
                {
                    int j = rnd.Next(i + 1);
                    int temp = currentPath[i];
                    currentPath[i] = currentPath[j];
                    currentPath[j] = temp;
                }

                currentPath = Optimize2Opt(currentPath, D, N);
                double cost = CalculateCost(currentPath, D, N);

                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestPath = (int[])currentPath.Clone();
                }
            }
        }

        Console.WriteLine(string.Join(" ", bestPath.Select(x => x - 1)));
    }

    static void Permute(int[] arr, int k, int N, double[,] D, ref double bestCost, ref int[] bestPath)
    {
        if (k == N)
        {
            double cost = CalculateCost(arr, D, N);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestPath = (int[])arr.Clone();
            }
        }
        else
        {
            for (int i = k; i < N; i++)
            {
                int temp = arr[i];
                arr[i] = arr[k];
                arr[k] = temp;

                Permute(arr, k + 1, N, D, ref bestCost, ref bestPath);

                temp = arr[i];
                arr[i] = arr[k];
                arr[k] = temp;
            }
        }
    }

    static double CalculateCost(int[] path, double[,] D, int N)
    {
        double cost = D[0, path[0]];
        for (int i = 0; i < N - 1; i++)
        {
            cost += D[path[i], path[i + 1]];
        }
        return cost;
    }

    static int[] Optimize2Opt(int[] path, double[,] D, int N)
    {
        int[] currentPath = (int[])path.Clone();
        bool improvement = true;
        while (improvement)
        {
            improvement = false;
            for (int i = 0; i < N - 1; i++)
            {
                for (int k = i + 1; k < N; k++)
                {
                    double currentDelta = 0;
                    double newDelta = 0;

                    if (i == 0)
                    {
                        currentDelta += D[0, currentPath[0]];
                        newDelta += D[0, currentPath[k]];
                    }
                    else
                    {
                        currentDelta += D[currentPath[i - 1], currentPath[i]];
                        newDelta += D[currentPath[i - 1], currentPath[k]];
                    }

                    if (k == N - 1)
                    {
                        // No loop back to start
                    }
                    else
                    {
                        currentDelta += D[currentPath[k], currentPath[k + 1]];
                        newDelta += D[currentPath[i], currentPath[k + 1]];
                    }

                    if (newDelta < currentDelta - 1e-9)
                    {
                        Array.Reverse(currentPath, i, k - i + 1);
                        improvement = true;
                    }
                }
            }
        }
        return currentPath;
    }
}
