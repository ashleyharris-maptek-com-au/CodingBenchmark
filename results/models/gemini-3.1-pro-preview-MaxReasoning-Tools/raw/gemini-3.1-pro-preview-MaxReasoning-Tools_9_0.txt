using System;
using System.IO;
using System.Collections.Generic;
using System.Diagnostics;
using System.Threading;
using System.Threading.Tasks;

class Program
{
    struct Cut : IComparable<Cut>
    {
        public int length;
        public int index;
        public int CompareTo(Cut other)
        {
            return other.length.CompareTo(this.length);
        }
    }

    static int N;
    static int stock_length;
    static Cut[] cuts;
    static int M;

    static void Main()
    {
        var sw = Stopwatch.StartNew();

        using (Stream stdin = Console.OpenStandardInput())
        {
            FastScanner scanner = new FastScanner(stdin);
            N = scanner.ReadInt();
            if (N == -1) return;
            if (N <= 0)
            {
                Console.WriteLine(0);
                return;
            }
            stock_length = scanner.ReadInt();
            cuts = new Cut[N];
            long sum = 0;
            for (int i = 0; i < N; i++)
            {
                cuts[i] = new Cut { length = scanner.ReadInt(), index = i };
                sum += cuts[i].length;
            }

            int L1 = (int)((sum + stock_length - 1) / stock_length);
            Array.Sort(cuts);

            M = 1;
            while (M < N) M *= 2;

            int[] head = new int[N];
            int[] next = new int[N];
            int[] val = new int[N];
            int[] tree = new int[2 * M];

            int activeBins = RunFFD_Local(cuts, head, next, val, tree);

            if (activeBins == L1)
            {
                PrintOutput(activeBins, head, next, val);
                return;
            }

            if (N <= 120)
            {
                SolveBB(sw, L1, activeBins, head, next, val);
            }
            else
            {
                int bestBins = activeBins;
                int[] bestHead = (int[])head.Clone();
                int[] bestNext = (int[])next.Clone();
                int[] bestVal = (int[])val.Clone();

                int numThreads = Environment.ProcessorCount;
                if (numThreads <= 0) numThreads = 1;
                if (numThreads > 12) numThreads = 12; // cap based on specs
                
                long memoryPerThread = (long)N * 20 + (long)M * 4 + 1024 * 1024; 
                long maxMemory = 10L * 1024 * 1024 * 1024; // 10 GB
                while (numThreads > 1 && numThreads * memoryPerThread > maxMemory)
                {
                    numThreads--;
                }

                object lockObj = new object();

                Parallel.For(0, numThreads, (threadId) =>
                {
                    Random localRand = new Random(42 + threadId);
                    Cut[] localShuffled = new Cut[N];
                    Array.Copy(cuts, localShuffled, N);

                    int[] localHead = new int[N];
                    int[] localNext = new int[N];
                    int[] localVal = new int[N];
                    int[] localTree = new int[2 * M];

                    while (sw.ElapsedMilliseconds < 280000 && bestBins > L1)
                    {
                        Array.Copy(cuts, localShuffled, N);
                        int numSwaps = Math.Max(1, N / 100);
                        for (int k = 0; k < numSwaps; k++)
                        {
                            int i = localRand.Next(N);
                            int offset = localRand.Next(-5, 6);
                            int j = i + offset;
                            if (j >= 0 && j < N)
                            {
                                var temp = localShuffled[i];
                                localShuffled[i] = localShuffled[j];
                                localShuffled[j] = temp;
                            }
                        }

                        int bins = RunFFD_Local(localShuffled, localHead, localNext, localVal, localTree);

                        if (bins < bestBins)
                        {
                            lock (lockObj)
                            {
                                if (bins < bestBins)
                                {
                                    bestBins = bins;
                                    Array.Copy(localHead, bestHead, N);
                                    Array.Copy(localNext, bestNext, N);
                                    Array.Copy(localVal, bestVal, N);
                                }
                            }
                        }
                    }
                });

                PrintOutput(bestBins, bestHead, bestNext, bestVal);
            }
        }
    }

    static int RunFFD_Local(Cut[] currentCuts, int[] head, int[] next, int[] val, int[] localTree)
    {
        Array.Fill(head, -1);
        Array.Fill(localTree, stock_length, 1, 2 * M - 1);
        int activeBins = 0;
        for (int i = 0; i < N; i++)
        {
            int len = currentCuts[i].length;
            int origIdx = currentCuts[i].index;

            int node = 1;
            while (node < M)
            {
                if (localTree[2 * node] >= len) node = 2 * node;
                else node = 2 * node + 1;
            }

            int binIndex = node - M;
            if (binIndex >= activeBins) activeBins = binIndex + 1;

            next[i] = head[binIndex];
            head[binIndex] = i;
            val[i] = origIdx;

            localTree[node] -= len;
            node /= 2;
            while (node > 0)
            {
                localTree[node] = Math.Max(localTree[2 * node], localTree[2 * node + 1]);
                node /= 2;
            }
        }
        return activeBins;
    }

    static void SolveBB(Stopwatch sw, int L1, int ffdBins, int[] ffdHead, int[] ffdNext, int[] ffdVal)
    {
        int bestBins = ffdBins;
        int[] currentAssignment = new int[N];
        int[] bestAssignment = new int[N];

        bool foundBetter = false;
        int[] binCap = new int[N];
        bool timeout = false;

        void DFS(int itemIdx, int currentBins, long emptyArea)
        {
            if (timeout || sw.ElapsedMilliseconds > 280000)
            {
                timeout = true;
                return;
            }

            if (currentBins >= bestBins) return;

            long remainingSum = 0;
            for (int i = itemIdx; i < N; i++) remainingSum += cuts[i].length;

            long needed = remainingSum - emptyArea;
            int minAdditional = needed > 0 ? (int)((needed + stock_length - 1) / stock_length) : 0;

            if (currentBins + minAdditional >= bestBins) return;

            if (itemIdx == N)
            {
                bestBins = currentBins;
                Array.Copy(currentAssignment, bestAssignment, N);
                foundBetter = true;
                return;
            }

            int len = cuts[itemIdx].length;

            List<int> validBins = new List<int>();
            for (int i = 0; i < currentBins; i++)
            {
                if (binCap[i] >= len) validBins.Add(i);
            }
            validBins.Sort((a, b) => binCap[a].CompareTo(binCap[b]));

            HashSet<int> tried = new HashSet<int>();
            foreach (int i in validBins)
            {
                if (!tried.Add(binCap[i])) continue;

                bool perfect = (binCap[i] == len);
                binCap[i] -= len;
                currentAssignment[itemIdx] = i;

                DFS(itemIdx + 1, currentBins, emptyArea - len);

                binCap[i] += len;

                if (perfect) return;
                if (timeout) return;
            }

            if (currentBins < bestBins - 1)
            {
                binCap[currentBins] = stock_length - len;
                currentAssignment[itemIdx] = currentBins;
                DFS(itemIdx + 1, currentBins + 1, emptyArea + stock_length - len);
                binCap[currentBins] = 0;
            }
        }

        DFS(0, 0, 0);

        if (foundBetter)
        {
            int[] head = new int[N];
            int[] next = new int[N];
            int[] val = new int[N];
            Array.Fill(head, -1);

            for (int i = 0; i < N; i++)
            {
                int b = bestAssignment[i];
                next[i] = head[b];
                head[b] = i;
                val[i] = cuts[i].index;
            }
            PrintOutput(bestBins, head, next, val);
        }
        else
        {
            PrintOutput(ffdBins, ffdHead, ffdNext, ffdVal);
        }
    }

    static void PrintOutput(int activeBins, int[] head, int[] next, int[] val)
    {
        using (var outStream = new BufferedStream(Console.OpenStandardOutput(), 1024 * 1024))
        using (var writer = new StreamWriter(outStream, System.Text.Encoding.ASCII))
        {
            writer.WriteLine(activeBins);
            for (int i = 0; i < activeBins; i++)
            {
                int curr = head[i];
                bool first = true;
                while (curr != -1)
                {
                    if (!first) writer.Write(" ");
                    writer.Write(val[curr]);
                    first = false;
                    curr = next[curr];
                }
                writer.WriteLine();
            }
        }
    }

    class FastScanner
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int head;
        private int tail;

        public FastScanner(Stream stream)
        {
            this.stream = stream;
        }

        public int ReadInt()
        {
            int c = Read();
            while (c <= 32)
            {
                if (c == -1) return -1;
                c = Read();
            }
            int res = 0;
            while (c > 32)
            {
                if (c < '0' || c > '9')
                    throw new Exception("Invalid digit");
                res = res * 10 + c - '0';
                c = Read();
            }
            return res;
        }

        private int Read()
        {
            if (head >= tail)
            {
                head = 0;
                tail = stream.Read(buffer, 0, buffer.Length);
                if (tail == 0) return -1;
            }
            return buffer[head++];
        }
    }
}
