#![allow(warnings)]
use std::io::{self, Read};
use std::collections::HashSet;

fn segments_intersect(a: (f64, f64), b: (f64, f64), c: (f64, f64), d: (f64, f64)) -> bool {
    fn ccw(a: (f64, f64), b: (f64, f64), c: (f64, f64)) -> f64 {
        (b.0 - a.0) * (c.1 - a.1) - (b.1 - a.1) * (c.0 - a.0)
    }
    fn on_segment(p: (f64, f64), a: (f64, f64), b: (f64, f64)) -> bool {
        p.0 >= a.0.min(b.0) - 1e-7 && p.0 <= a.0.max(b.0) + 1e-7 &&
        p.1 >= a.1.min(b.1) - 1e-7 && p.1 <= a.1.max(b.1) + 1e-7
    }
    let cp1 = ccw(a, b, c);
    let cp2 = ccw(a, b, d);
    let cp3 = ccw(c, d, a);
    let cp4 = ccw(c, d, b);

    if cp1 * cp2 < 0.0 && cp3 * cp4 < 0.0 { return true; }
    if cp1.abs() < 1e-7 && on_segment(c, a, b) { return true; }
    if cp2.abs() < 1e-7 && on_segment(d, a, b) { return true; }
    if cp3.abs() < 1e-7 && on_segment(a, c, d) { return true; }
    if cp4.abs() < 1e-7 && on_segment(b, c, d) { return true; }
    false
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut tokens = input.split_whitespace();
    
    let nv_str = match tokens.next() {
        Some(s) => s,
        None => return,
    };
    let nv: usize = nv_str.parse().unwrap();
    let mut polygon = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = tokens.next().unwrap().parse().unwrap();
        let y: f64 = tokens.next().unwrap().parse().unwrap();
        polygon.push((x, y));
    }
    let sx: f64 = tokens.next().unwrap().parse().unwrap();
    let sy: f64 = tokens.next().unwrap().parse().unwrap();
    let sz: f64 = tokens.next().unwrap().parse().unwrap();
    
    let len_s = (sx * sx + sy * sy + sz * sz).sqrt();
    let ux = sx / len_s;
    let uy = sy / len_s;
    let uz = sz / len_s;
    
    let vx = -ux;
    let vy = -uy;
    let vz = -uz;
    
    let qw = ((1.0 + vz) / 2.0).max(0.0).sqrt();
    let (qx, qy, qz) = if qw > 1e-6 {
        (-vy / (2.0 * qw), vx / (2.0 * qw), 0.0)
    } else {
        (1.0, 0.0, 0.0)
    };
    
    let qx2 = qx * qx; let qy2 = qy * qy;
    let r00 = 1.0 - 2.0 * qy2;
    let r10 = 2.0 * qx * qy;
    let r20 = -2.0 * qw * qy;
    
    let r01 = 2.0 * qx * qy;
    let r11 = 1.0 - 2.0 * qx2;
    let r21 = 2.0 * qw * qx;
    
    let m_proj = |vx: f64, vy: f64, vz: f64| -> (f64, f64) {
        (vx - vz * sx / sz, vy - vz * sy / sz)
    };
    
    let d1 = m_proj(r00, r10, r20);
    let d2 = m_proj(r01, r11, r21);
    
    let e1x = (d1.0 - d2.0) * 10.0;
    let e1y = (d1.1 - d2.1) * 10.0;
    let e2x = (d1.0 + d2.0) * 10.0;
    let e2y = (d1.1 + d2.1) * 10.0;
    
    let det = e1x * e2y - e1y * e2x;
    
    let mut poly_ab = Vec::with_capacity(nv);
    for &(x, y) in &polygon {
        let alpha = (x * e2y - y * e2x) / det;
        let beta = (y * e1x - x * e1y) / det;
        poly_ab.push((alpha, beta));
    }
    
    let mut min_a = poly_ab[0].0;
    let mut max_a = poly_ab[0].0;
    let mut min_b = poly_ab[0].1;
    let mut max_b = poly_ab[0].1;
    for &p in &poly_ab {
        if p.0 < min_a { min_a = p.0; }
        if p.0 > max_a { max_a = p.0; }
        if p.1 < min_b { min_b = p.1; }
        if p.1 > max_b { max_b = p.1; }
    }
    
    let offset_a = (min_a + max_a) / 2.0;
    let offset_b = (min_b + max_b) / 2.0;
    
    for p in &mut poly_ab {
        p.0 -= offset_a;
        p.1 -= offset_b;
    }
    
    min_a -= offset_a;
    max_a -= offset_a;
    min_b -= offset_b;
    max_b -= offset_b;
    
    let mut tiles = HashSet::new();
    
    for i in 0..nv {
        let p1 = poly_ab[i];
        let p2 = poly_ab[(i + 1) % nv];
        
        let min_edge_a = p1.0.min(p2.0);
        let max_edge_a = p1.0.max(p2.0);
        let min_edge_b = p1.1.min(p2.1);
        let max_edge_b = p1.1.max(p2.1);
        
        let m_start = (min_edge_a - 0.5).floor() as i32;
        let m_end = (max_edge_a + 0.5).ceil() as i32;
        let n_start = (min_edge_b - 0.5).floor() as i32;
        let n_end = (max_edge_b + 0.5).ceil() as i32;
        
        for m in m_start..=m_end {
            for n in n_start..=n_end {
                let s_min_a = m as f64 - 0.5;
                let s_max_a = m as f64 + 0.5;
                let s_min_b = n as f64 - 0.5;
                let s_max_b = n as f64 + 0.5;
                
                let inside = |p: (f64, f64)| {
                    p.0 >= s_min_a - 1e-7 && p.0 <= s_max_a + 1e-7 && 
                    p.1 >= s_min_b - 1e-7 && p.1 <= s_max_b + 1e-7
                };
                if inside(p1) || inside(p2) {
                    tiles.insert((m, n));
                    continue;
                }
                
                let c1 = (s_min_a, s_min_b);
                let c2 = (s_max_a, s_min_b);
                let c3 = (s_max_a, s_max_b);
                let c4 = (s_min_a, s_max_b);
                
                if segments_intersect(p1, p2, c1, c2) ||
                   segments_intersect(p1, p2, c2, c3) ||
                   segments_intersect(p1, p2, c3, c4) ||
                   segments_intersect(p1, p2, c4, c1) {
                    tiles.insert((m, n));
                }
            }
        }
    }
    
    let n_min = min_b.floor() as i32 - 1;
    let n_max = max_b.ceil() as i32 + 1;
    
    for n in n_min..=n_max {
        let b_line = n as f64;
        let mut intersections = Vec::new();
        for i in 0..nv {
            let p1 = poly_ab[i];
            let p2 = poly_ab[(i + 1) % nv];
            
            if (p1.1 <= b_line && b_line < p2.1) || (p2.1 <= b_line && b_line < p1.1) {
                let t = (b_line - p1.1) / (p2.1 - p1.1);
                let a_int = p1.0 + t * (p2.0 - p1.0);
                intersections.push(a_int);
            }
        }
        intersections.sort_by(|a, b| a.partial_cmp(b).unwrap());
        
        for chunk in intersections.chunks(2) {
            if chunk.len() == 2 {
                let start_a = chunk[0];
                let end_a = chunk[1];
                let m_start = (start_a - 1e-7).ceil() as i32;
                let m_end = (end_a + 1e-7).floor() as i32;
                for m in m_start..=m_end {
                    if m as f64 >= start_a - 1e-7 && m as f64 <= end_a + 1e-7 {
                        tiles.insert((m, n));
                    }
                }
            }
        }
    }
    
    println!("{}", tiles.len());
    let mut i = 0;
    for &(m, n) in &tiles {
        let real_m = m as f64 + offset_a;
        let real_n = n as f64 + offset_b;
        let cx = real_m * e1x + real_n * e2x;
        let cy = real_m * e1y + real_n * e2y;
        
        let shift = i as f64 * 40.0;
        let px = cx - shift * ux;
        let py = cy - shift * uy;
        let pz = 0.0 - shift * uz;
        
        println!("{:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6} {:.6}", px, py, pz, qw, qx, qy, qz, 10.0);
        i += 1;
    }
}
