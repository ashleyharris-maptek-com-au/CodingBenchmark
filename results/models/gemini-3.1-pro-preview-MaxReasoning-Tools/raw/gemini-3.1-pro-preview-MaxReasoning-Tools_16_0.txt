#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <numeric>
#include <random>
#include <cmath>

struct Task {
    int job_id;
    int task_index;
    int machine;
    int duration;
};

struct Job {
    std::vector<Task> tasks;
};

struct TaskResult {
    int start_time;
    int machine;
};

struct TimeInterval {
    int start;
    int end;
};

int J, M;
std::vector<Job> jobs;
int total_tasks = 0;
std::vector<std::vector<TaskResult>> global_best_schedule;

double compute_perms() {
    if (total_tasks > 15) return 1e18; 
    double num_perms = 1.0;
    int n = total_tasks;
    std::vector<int> counts(J);
    for (int j = 0; j < J; ++j) counts[j] = jobs[j].tasks.size();
    
    for (int i = 1; i <= n; ++i) num_perms *= i;
    for (int j = 0; j < J; ++j) {
        for (int i = 1; i <= counts[j]; ++i) {
            num_perms /= i;
        }
    }
    return num_perms;
}

int get_makespan_semiactive(const std::vector<int>& sequence) {
    std::vector<int> job_next_task(J, 0);
    std::vector<int> job_avail(J, 0);
    std::vector<int> machine_avail(M, 0);
    
    int makespan = 0;
    for (int job_id : sequence) {
        int task_idx = job_next_task[job_id];
        const Task& task = jobs[job_id].tasks[task_idx];
        
        int start_time = std::max(job_avail[job_id], machine_avail[task.machine]);
        int end_time = start_time + task.duration;
        
        job_avail[job_id] = end_time;
        machine_avail[task.machine] = end_time;
        job_next_task[job_id]++;
        
        if (end_time > makespan) makespan = end_time;
    }
    return makespan;
}

int get_makespan_active(const std::vector<int>& sequence) {
    std::vector<int> job_next_task(J, 0);
    std::vector<int> job_avail(J, 0);
    std::vector<std::vector<TimeInterval>> machine_intervals(M);
    
    int makespan = 0;
    for (int job_id : sequence) {
        int task_idx = job_next_task[job_id];
        const Task& task = jobs[job_id].tasks[task_idx];
        
        int T_avail = job_avail[job_id];
        int m = task.machine;
        int duration = task.duration;
        
        auto& intervals = machine_intervals[m];
        int earliest_start = T_avail;
        
        if (intervals.empty()) {
            earliest_start = T_avail;
            intervals.push_back({earliest_start, earliest_start + duration});
        } else {
            if (T_avail + duration <= intervals[0].start) {
                earliest_start = T_avail;
                intervals.insert(intervals.begin(), {earliest_start, earliest_start + duration});
            } else {
                bool found = false;
                for (size_t i = 0; i < intervals.size() - 1; ++i) {
                    int gap_start = std::max(T_avail, intervals[i].end);
                    int gap_end = intervals[i+1].start;
                    if (gap_start + duration <= gap_end) {
                        earliest_start = gap_start;
                        intervals.insert(intervals.begin() + i + 1, {earliest_start, earliest_start + duration});
                        found = true;
                        break;
                    }
                }
                if (!found) {
                    earliest_start = std::max(T_avail, intervals.back().end);
                    intervals.push_back({earliest_start, earliest_start + duration});
                }
            }
        }
        
        int end_time = earliest_start + duration;
        job_avail[job_id] = end_time;
        job_next_task[job_id]++;
        
        if (end_time > makespan) makespan = end_time;
    }
    return makespan;
}

void decode_sequence_active(const std::vector<int>& sequence, std::vector<std::vector<TaskResult>>& schedule) {
    std::vector<int> job_next_task(J, 0);
    std::vector<int> job_avail(J, 0);
    std::vector<std::vector<TimeInterval>> machine_intervals(M);
    
    schedule.assign(J, std::vector<TaskResult>());
    for (int j = 0; j < J; ++j) {
        schedule[j].resize(jobs[j].tasks.size());
    }
    
    for (int job_id : sequence) {
        int task_idx = job_next_task[job_id];
        const Task& task = jobs[job_id].tasks[task_idx];
        
        int T_avail = job_avail[job_id];
        int m = task.machine;
        int duration = task.duration;
        
        auto& intervals = machine_intervals[m];
        int earliest_start = T_avail;
        
        if (intervals.empty()) {
            earliest_start = T_avail;
            intervals.push_back({earliest_start, earliest_start + duration});
        } else {
            if (T_avail + duration <= intervals[0].start) {
                earliest_start = T_avail;
                intervals.insert(intervals.begin(), {earliest_start, earliest_start + duration});
            } else {
                bool found = false;
                for (size_t i = 0; i < intervals.size() - 1; ++i) {
                    int gap_start = std::max(T_avail, intervals[i].end);
                    int gap_end = intervals[i+1].start;
                    if (gap_start + duration <= gap_end) {
                        earliest_start = gap_start;
                        intervals.insert(intervals.begin() + i + 1, {earliest_start, earliest_start + duration});
                        found = true;
                        break;
                    }
                }
                if (!found) {
                    earliest_start = std::max(T_avail, intervals.back().end);
                    intervals.push_back({earliest_start, earliest_start + duration});
                }
            }
        }
        
        int end_time = earliest_start + duration;
        schedule[job_id][task_idx] = {earliest_start, m};
        job_avail[job_id] = end_time;
        job_next_task[job_id]++;
    }
}

void strategy_exhaustive() {
    std::vector<int> seq;
    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < jobs[j].tasks.size(); ++t) {
            seq.push_back(j);
        }
    }
    std::sort(seq.begin(), seq.end());
    
    int best_makespan = 2e9;
    std::vector<int> best_seq = seq;
    
    do {
        int m = get_makespan_active(seq);
        if (m < best_makespan) {
            best_makespan = m;
            best_seq = seq;
        }
    } while (std::next_permutation(seq.begin(), seq.end()));
    
    decode_sequence_active(best_seq, global_best_schedule);
}

void strategy_simulated_annealing(double time_limit_sec, bool use_active) {
    auto start_time = std::chrono::steady_clock::now();
    
    std::vector<int> current_seq;
    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < jobs[j].tasks.size(); ++t) {
            current_seq.push_back(j);
        }
    }
    
    if (current_seq.empty()) return;
    
    bool all_same = true;
    for (size_t i = 1; i < current_seq.size(); ++i) {
        if (current_seq[i] != current_seq[0]) {
            all_same = false;
            break;
        }
    }
    if (all_same) {
        decode_sequence_active(current_seq, global_best_schedule);
        return;
    }
    
    std::mt19937 rng(42);
    
    std::vector<int> job_work(J, 0);
    for (int j = 0; j < J; ++j) {
        for (const auto& task : jobs[j].tasks) {
            job_work[j] += task.duration;
        }
    }
    
    std::vector<int> job_order(J);
    std::iota(job_order.begin(), job_order.end(), 0);
    std::sort(job_order.begin(), job_order.end(), [&](int a, int b) {
        return job_work[a] > job_work[b];
    });
    
    current_seq.clear();
    std::vector<int> job_rem(J);
    for(int j = 0; j < J; ++j) job_rem[j] = jobs[j].tasks.size();
    
    bool added = true;
    while(added) {
        added = false;
        for (int j : job_order) {
            if (job_rem[j] > 0) {
                current_seq.push_back(j);
                job_rem[j]--;
                added = true;
            }
        }
    }
    
    int current_makespan = use_active ? get_makespan_active(current_seq) : get_makespan_semiactive(current_seq);
    int best_makespan = current_makespan;
    std::vector<int> best_seq = current_seq;
    std::vector<int> next_seq = current_seq;
    
    double T = 1000.0;
    double T_min = 0.001;
    double alpha = 0.9999; 
    
    int iter = 0;
    int check_interval = use_active ? 127 : 1023;
    
    while (true) {
        if ((iter & check_interval) == 0) {
            auto now = std::chrono::steady_clock::now();
            std::chrono::duration<double> elapsed = now - start_time;
            if (elapsed.count() > time_limit_sec) break;
        }
        
        int idx1 = rng() % current_seq.size();
        int idx2 = rng() % current_seq.size();
        int attempts = 0;
        while (current_seq[idx1] == current_seq[idx2] && attempts < 100) {
            idx2 = rng() % current_seq.size();
            attempts++;
        }
        
        if (current_seq[idx1] == current_seq[idx2]) {
            iter++;
            continue;
        }
        
        std::swap(next_seq[idx1], next_seq[idx2]);
        
        int next_makespan = use_active ? get_makespan_active(next_seq) : get_makespan_semiactive(next_seq);
        
        if (next_makespan < current_makespan) {
            current_makespan = next_makespan;
            current_seq = next_seq;
            if (next_makespan < best_makespan) {
                best_makespan = next_makespan;
                best_seq = current_seq;
            }
        } else {
            double delta = next_makespan - current_makespan;
            double prob = std::exp(-delta / T);
            std::uniform_real_distribution<double> dist(0.0, 1.0);
            if (dist(rng) < prob) {
                current_makespan = next_makespan;
                current_seq = next_seq;
            } else {
                std::swap(next_seq[idx1], next_seq[idx2]); 
            }
        }
        
        T *= alpha;
        if (T < T_min) T = 1000.0;
        iter++;
    }
    
    decode_sequence_active(best_seq, global_best_schedule);
}

void solve() {
    if (total_tasks == 0) return;
    
    double num_perms = compute_perms();
    if (num_perms <= 5e6) {
        strategy_exhaustive();
    } else if (total_tasks <= 5000) {
        strategy_simulated_annealing(1.9, true); 
    } else {
        strategy_simulated_annealing(1.9, false); 
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);
    
    if (!(std::cin >> J >> M)) return 0;
    
    jobs.resize(J);
    for (int i = 0; i < J; ++i) {
        int T_i;
        std::cin >> T_i;
        for (int t = 0; t < T_i; ++t) {
            int machine, duration;
            std::cin >> machine >> duration;
            jobs[i].tasks.push_back({i, t, machine, duration});
            total_tasks++;
        }
    }
    
    solve();
    
    for (int j = 0; j < J; ++j) {
        for (size_t t = 0; t < global_best_schedule[j].size(); ++t) {
            if (t > 0) std::cout << " ";
            std::cout << global_best_schedule[j][t].start_time << " " << global_best_schedule[j][t].machine;
        }
        std::cout << "\n";
    }
    
    return 0;
}
