using namespace std;

auto start_time = std::chrono::steady_clock::now();
inline double elapsed() {
    auto now = std::chrono::steady_clock::now();
    return std::chrono::duration<double>(now - start_time).count();
}

const int BUFFER_SIZE = 1 << 24; // 16MB
char buffer[BUFFER_SIZE];
int buffer_pos = 0;
int buffer_len = 0;

inline bool peek_char(char &c) {
    if (buffer_pos >= buffer_len) {
        buffer_len = fread(buffer, 1, BUFFER_SIZE, stdin);
        buffer_pos = 0;
        if (buffer_len == 0) return false;
    }
    c = buffer[buffer_pos];
    return true;
}

inline void skip_whitespace() {
    char c;
    while (peek_char(c) && c <= 32) {
        buffer_pos++;
    }
}

inline bool read_int(int &val) {
    skip_whitespace();
    char c;
    if (!peek_char(c)) return false;
    val = 0;
    while (peek_char(c) && c >= '0' && c <= '9') {
        val = val * 10 + (c - '0');
        buffer_pos++;
    }
    return true;
}

double exp_table[20];
void init_exp_table() {
    exp_table[0] = 1.0;
    for (int i = 1; i < 20; i++) exp_table[i] = exp_table[i - 1] * 10.0;
}

inline bool read_double(double &val) {
    skip_whitespace();
    char c;
    if (!peek_char(c)) return false;
    bool neg = false;
    if (c == '-') {
        neg = true;
        buffer_pos++;
    } else if (c == '+') {
        buffer_pos++;
    }
    val = 0.0;
    while (peek_char(c) && c >= '0' && c <= '9') {
        val = val * 10.0 + (c - '0');
        buffer_pos++;
    }
    if (peek_char(c) && c == '.') {
        buffer_pos++;
        double frac = 0.0;
        int frac_len = 0;
        while (peek_char(c) && c >= '0' && c <= '9') {
            frac = frac * 10.0 + (c - '0');
            frac_len++;
            buffer_pos++;
        }
        if (frac_len > 0) {
            if (frac_len < 20) {
                val += frac / exp_table[frac_len];
            } else {
                val += frac / pow(10.0, frac_len);
            }
        }
    }
    if (peek_char(c) && (c == 'e' || c == 'E')) {
        buffer_pos++;
        bool eneg = false;
        if (peek_char(c)) {
            if (c == '-') {
                eneg = true;
                buffer_pos++;
            } else if (c == '+') {
                buffer_pos++;
            }
        }
        int exp = 0;
        while (peek_char(c) && c >= '0' && c <= '9') {
            exp = exp * 10 + (c - '0');
            buffer_pos++;
        }
        if (exp < 20) {
            if (eneg) val /= exp_table[exp];
            else val *= exp_table[exp];
        } else {
            if (eneg) val *= pow(10.0, -exp);
            else val *= pow(10.0, exp);
        }
    }
    if (neg) val = -val;
    return true;
}

char out_buffer[BUFFER_SIZE];
int out_pos = 0;
inline void flush_out() {
    if (out_pos > 0) {
        fwrite(out_buffer, 1, out_pos, stdout);
        out_pos = 0;
    }
}
inline void write_char(char c) {
    if (out_pos == BUFFER_SIZE) flush_out();
    out_buffer[out_pos++] = c;
}
inline void write_int(int val) {
    if (val == 0) {
        write_char('0');
        return;
    }
    char temp[16];
    int t = 0;
    while (val > 0) {
        temp[t++] = '0' + (val % 10);
        val /= 10;
    }
    while (t > 0) {
        write_char(temp[--t]);
    }
}

struct Point {
    double x, y;
};

int N;
vector<Point> pts;

inline double dist_sq(const Point &a, const Point &b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}
inline double dist(const Point &a, const Point &b) {
    return sqrt(dist_sq(a, b));
}

uint64_t hilbert(uint32_t x, uint32_t y) {
    uint64_t d = 0;
    for (int s = 31; s >= 0; s--) {
        uint32_t rx = (x >> s) & 1;
        uint32_t ry = (y >> s) & 1;
        d |= ((uint64_t)((3 * rx) ^ ry) << (2 * s));
        if (ry == 0) {
            if (rx == 1) {
                x = ~x;
                y = ~y;
            }
            uint32_t t = x; x = y; y = t;
        }
    }
    return d;
}

void two_opt(vector<int>& tour, const vector<Point>& pts, int W) {
    bool improvement = true;
    int passes = 0;
    
    vector<double> tour_dist(N);
    for(int i = 0; i < N; i++) {
        tour_dist[i] = dist(pts[tour[i]], pts[tour[(i+1)%N]]);
    }

    while (passes < 30) {
        improvement = false;
        passes++;
        for (int i = 0; i < N; i++) {
            int limit = i + min(N - 2, W);
            int n1 = tour[i];
            int n2 = tour[(i + 1) % N];
            double d1 = tour_dist[i];
            
            for (int j = i + 2; j <= limit; j++) {
                int j_mod = j % N;
                int n3 = tour[j_mod];
                int n4 = tour[(j + 1) % N];

                double d2 = tour_dist[j_mod];
                double d3 = dist(pts[n1], pts[n3]);
                double d4 = dist(pts[n2], pts[n4]);

                if (d3 + d4 + 1e-9 < d1 + d2) {
                    int left = (i + 1) % N;
                    int right = j_mod;
                    while (left != right) {
                        swap(tour[left], tour[right]);
                        left = (left + 1) % N;
                        if (left == right) break;
                        right = (right - 1 + N) % N;
                    }
                    improvement = true;
                    n2 = tour[(i + 1) % N];
                    
                    int k = i;
                    while (true) {
                        tour_dist[k] = dist(pts[tour[k]], pts[tour[(k + 1) % N]]);
                        if (k == j_mod) break;
                        k = (k + 1) % N;
                    }
                    d1 = tour_dist[i];
                }
            }
            if ((i & 65535) == 0 && elapsed() > 290.0) return;
        }
        if (!improvement) break;
        if (elapsed() > 290.0) break;
    }
}

int main() {
#ifdef _WIN32
    _setmode(_fileno(stdin), _O_BINARY);
    _setmode(_fileno(stdout), _O_BINARY);
#endif
    init_exp_table();

    if (!read_int(N)) return 0;
    pts.resize(N);
    for (int i = 0; i < N; i++) {
        read_double(pts[i].x);
        read_double(pts[i].y);
    }

    vector<int> tour;
    if (N <= 20000) {
        int starts = (N <= 2000) ? min(N, 20) : 1;
        double best_dist = 1e18;
        for (int s = 0; s < starts; s++) {
            int start_node = (starts == N) ? s : rand() % N;
            vector<int> current_tour;
            current_tour.reserve(N);
            vector<bool> visited(N, false);
            
            current_tour.push_back(start_node);
            visited[start_node] = true;
            int curr = start_node;
            
            for (int step = 1; step < N; step++) {
                int best_next = -1;
                double min_d = 1e18;
                for (int i = 0; i < N; i++) {
                    if (!visited[i]) {
                        double d = dist_sq(pts[curr], pts[i]);
                        if (d < min_d) {
                            min_d = d;
                            best_next = i;
                        }
                    }
                }
                current_tour.push_back(best_next);
                visited[best_next] = true;
                curr = best_next;
            }
            
            long long target_ops = 200000000LL;
            int W = (int)min((long long)N, target_ops / N);
            W = max(10, W);
            
            two_opt(current_tour, pts, W);
            
            double current_dist = 0;
            for (int i = 0; i < N; i++) {
                current_dist += dist(pts[current_tour[i]], pts[current_tour[(i + 1) % N]]);
            }
            
            if (current_dist < best_dist) {
                best_dist = current_dist;
                tour = current_tour;
            }
            if (elapsed() > 280.0) break;
        }
    } else {
        double min_x = 1e18, max_x = -1e18;
        double min_y = 1e18, max_y = -1e18;
        for (int i = 0; i < N; i++) {
            if (pts[i].x < min_x) min_x = pts[i].x;
            if (pts[i].x > max_x) max_x = pts[i].x;
            if (pts[i].y < min_y) min_y = pts[i].y;
            if (pts[i].y > max_y) max_y = pts[i].y;
        }
        double range_x = max_x - min_x;
        double range_y = max_y - min_y;
        double max_range = max(range_x, range_y);
        if (max_range == 0) max_range = 1.0;

        vector<pair<uint64_t, int>> hcodes(N);
        for (int i = 0; i < N; i++) {
            uint32_t hx = (uint32_t)min(4294967295.0, max(0.0, ((pts[i].x - min_x) / max_range) * 4294967295.0));
            uint32_t hy = (uint32_t)min(4294967295.0, max(0.0, ((pts[i].y - min_y) / max_range) * 4294967295.0));
            hcodes[i] = {hilbert(hx, hy), i};
        }
        sort(hcodes.begin(), hcodes.end(), [](const pair<uint64_t, int>& a, const pair<uint64_t, int>& b) {
            return a.first < b.first;
        });

        tour.resize(N);
        for (int i = 0; i < N; i++) {
            tour[i] = hcodes[i].second;
        }

        long long target_ops = 500000000LL;
        int W = (int)min((long long)N, target_ops / N);
        W = max(20, W);
        
        two_opt(tour, pts, W);
    }

    int start_idx = 0;
    for (int i = 0; i < N; i++) {
        if (tour[i] == 0) {
            start_idx = i;
            break;
        }
    }
    
    for (int i = 0; i < N; i++) {
        write_int(tour[(start_idx + i) % N]);
        if (i < N - 1) write_char(' ');
    }
    write_char('\n');
    flush_out();

    return 0;
}
