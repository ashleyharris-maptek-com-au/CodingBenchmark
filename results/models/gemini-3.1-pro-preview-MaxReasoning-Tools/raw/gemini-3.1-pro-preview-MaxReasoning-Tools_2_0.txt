using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

struct Edge
{
    public int u;
    public int v;
    public long w;
    public Edge(int u, int v, long w) { this.u = u; this.v = v; this.w = w; }
}

struct EdgeComparer : IComparer<int>
{
    public int Compare(int x, int y)
    {
        return Program.edges[x].w.CompareTo(Program.edges[y].w);
    }
}

class Program
{
    static int N;
    static int M;
    public static List<Edge> edges = new List<Edge>();
    static List<int>[] adj;
    static int[] originalDegree;

    static void Main()
    {
        byte[] buffer = new byte[65536];
        Stream inputStream = Console.OpenStandardInput();
        int bytesRead = 0;
        int bufferPos = 0;

        long ReadLong()
        {
            long res = 0;
            bool started = false;
            bool isNeg = false;
            while (true)
            {
                if (bufferPos >= bytesRead)
                {
                    bytesRead = inputStream.Read(buffer, 0, buffer.Length);
                    bufferPos = 0;
                    if (bytesRead == 0) break;
                }
                byte b = buffer[bufferPos++];
                if (b == '-')
                {
                    isNeg = true;
                    started = true;
                }
                else if (b >= '0' && b <= '9')
                {
                    res = res * 10 + (b - '0');
                    started = true;
                }
                else if (started)
                {
                    break;
                }
            }
            return isNeg ? -res : res;
        }

        N = (int)ReadLong();
        M = (int)ReadLong();

        if (N == 0)
        {
            Console.WriteLine("0");
            return;
        }

        adj = new List<int>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<int>();
        originalDegree = new int[N];

        for (int i = 0; i < M; i++)
        {
            int u = (int)ReadLong();
            int v = (int)ReadLong();
            long w = ReadLong();
            AddEdge(u, v, w);
        }

        List<int> oddNodes = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (originalDegree[i] % 2 != 0)
            {
                oddNodes.Add(i);
            }
        }

        if (oddNodes.Count > 0)
        {
            if (oddNodes.Count <= 18)
            {
                StrategyA(oddNodes);
            }
            else if (oddNodes.Count <= 4000)
            {
                StrategyB2(oddNodes);
            }
            else
            {
                StrategyC(oddNodes);
            }
        }

        List<int> circuit = new List<int>();
        Stack<int> stack = new Stack<int>();
        stack.Push(0);

        int[] currentHead = new int[N];
        bool[] used = new bool[edges.Count];

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;

            while (currentHead[u] < adj[u].Count)
            {
                int e = adj[u][currentHead[u]];
                currentHead[u]++;

                if (!used[e])
                {
                    used[e] = true;
                    int v = edges[e].u == u ? edges[e].v : edges[e].u;
                    stack.Push(v);
                    found = true;
                    break;
                }
            }

            if (!found)
            {
                circuit.Add(stack.Pop());
            }
        }

        circuit.Reverse();

        using (StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), bufferSize: 65536))
        {
            sw.Write(circuit.Count);
            sw.Write(" ");
            for (int i = 0; i < circuit.Count; i++)
            {
                sw.Write(circuit[i]);
                if (i < circuit.Count - 1) sw.Write(" ");
            }
            sw.WriteLine();
        }
    }

    static void AddEdge(int u, int v, long w)
    {
        int eIdx = edges.Count;
        edges.Add(new Edge(u, v, w));
        adj[u].Add(eIdx);
        adj[v].Add(eIdx);
        originalDegree[u]++;
        originalDegree[v]++;
    }

    static void DuplicateEdge(int u, int v, long w)
    {
        int eIdx = edges.Count;
        edges.Add(new Edge(u, v, w));
        adj[u].Add(eIdx);
        adj[v].Add(eIdx);
    }

    static void StrategyA(List<int> oddNodes)
    {
        int K = oddNodes.Count;
        long[,] distMatrix = new long[K, K];
        int[][] parentEdge = new int[K][];
        for (int i = 0; i < K; i++) parentEdge[i] = new int[N];

        for (int i = 0; i < K; i++)
        {
            int start = oddNodes[i];
            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            long[] d = new long[N];
            for (int j = 0; j < N; j++) d[j] = long.MaxValue;

            pq.Enqueue(start, 0);
            d[start] = 0;

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long currD);
                if (currD > d[u]) continue;

                foreach (int e in adj[u])
                {
                    int v = edges[e].u == u ? edges[e].v : edges[e].u;
                    long w = edges[e].w;
                    long nd = currD + w;

                    if (nd < d[v])
                    {
                        d[v] = nd;
                        parentEdge[i][v] = e;
                        pq.Enqueue(v, nd);
                    }
                }
            }

            for (int j = 0; j < K; j++)
            {
                distMatrix[i, j] = d[oddNodes[j]];
            }
        }

        long[] dp = new long[1 << K];
        for (int i = 0; i < (1 << K); i++) dp[i] = long.MaxValue;
        dp[0] = 0;

        for (int mask = 0; mask < (1 << K); mask++)
        {
            if (dp[mask] == long.MaxValue) continue;

            int i = -1;
            for (int b = 0; b < K; b++)
            {
                if ((mask & (1 << b)) == 0)
                {
                    i = b;
                    break;
                }
            }

            if (i == -1) continue;

            for (int j = i + 1; j < K; j++)
            {
                if ((mask & (1 << j)) == 0)
                {
                    long cost = distMatrix[i, j];
                    if (cost != long.MaxValue)
                    {
                        int nmask = mask | (1 << i) | (1 << j);
                        if (dp[mask] + cost < dp[nmask])
                        {
                            dp[nmask] = dp[mask] + cost;
                        }
                    }
                }
            }
        }

        int currMask = (1 << K) - 1;
        while (currMask > 0)
        {
            bool found = false;
            for (int i = 0; i < K; i++)
            {
                if ((currMask & (1 << i)) != 0)
                {
                    for (int j = i + 1; j < K; j++)
                    {
                        if ((currMask & (1 << j)) != 0)
                        {
                            int prevMask = currMask ^ (1 << i) ^ (1 << j);
                            int lowestUnset = -1;
                            for (int b = 0; b < K; b++)
                            {
                                if ((prevMask & (1 << b)) == 0)
                                {
                                    lowestUnset = b;
                                    break;
                                }
                            }

                            if (lowestUnset == i && dp[currMask] == dp[prevMask] + distMatrix[i, j])
                            {
                                int curr = oddNodes[j];
                                int start = oddNodes[i];
                                while (curr != start)
                                {
                                    int e = parentEdge[i][curr];
                                    DuplicateEdge(edges[e].u, edges[e].v, edges[e].w);
                                    curr = edges[e].u == curr ? edges[e].v : edges[e].u;
                                }
                                currMask = prevMask;
                                found = true;
                                break;
                            }
                        }
                    }
                }
                if (found) break;
            }
        }
    }

    static void StrategyB2(List<int> oddNodes)
    {
        HashSet<int> unmatched = new HashSet<int>(oddNodes);

        long[] dist = new long[N];
        int[] parentEdge = new int[N];
        for (int i = 0; i < N; i++) dist[i] = long.MaxValue;

        List<int> visitedNodes = new List<int>();

        while (unmatched.Count > 0)
        {
            int start = unmatched.First();

            PriorityQueue<int, long> pq = new PriorityQueue<int, long>();
            pq.Enqueue(start, 0);

            dist[start] = 0;
            visitedNodes.Add(start);
            int found = -1;

            while (pq.Count > 0)
            {
                pq.TryDequeue(out int u, out long d);
                if (d > dist[u]) continue;

                if (u != start && unmatched.Contains(u))
                {
                    found = u;
                    break;
                }

                foreach (int e in adj[u])
                {
                    int v = edges[e].u == u ? edges[e].v : edges[e].u;
                    long w = edges[e].w;
                    long nd = d + w;

                    if (nd < dist[v])
                    {
                        if (dist[v] == long.MaxValue) visitedNodes.Add(v);
                        dist[v] = nd;
                        parentEdge[v] = e;
                        pq.Enqueue(v, nd);
                    }
                }
            }

            if (found != -1)
            {
                unmatched.Remove(start);
                unmatched.Remove(found);

                int curr = found;
                while (curr != start)
                {
                    int e = parentEdge[curr];
                    DuplicateEdge(edges[e].u, edges[e].v, edges[e].w);
                    curr = edges[e].u == curr ? edges[e].v : edges[e].u;
                }
            }
            else
            {
                unmatched.Remove(start);
            }

            foreach (int v in visitedNodes)
            {
                dist[v] = long.MaxValue;
            }
            visitedNodes.Clear();
        }
    }

    static void StrategyC(List<int> oddNodes)
    {
        int originalEdgeCount = edges.Count;
        int[] edgeIndices = new int[originalEdgeCount];
        for (int i = 0; i < originalEdgeCount; i++) edgeIndices[i] = i;

        Array.Sort(edgeIndices, new EdgeComparer());

        int[] parent = new int[N];
        for (int i = 0; i < N; i++) parent[i] = i;

        int Find(int i)
        {
            int root = i;
            while (root != parent[root]) root = parent[root];
            int curr = i;
            while (curr != root)
            {
                int nxt = parent[curr];
                parent[curr] = root;
                curr = nxt;
            }
            return root;
        }

        void Union(int i, int j)
        {
            int rootI = Find(i);
            int rootJ = Find(j);
            if (rootI != rootJ)
            {
                parent[rootI] = rootJ;
            }
        }

        List<int>[] treeAdj = new List<int>[N];
        for (int i = 0; i < N; i++) treeAdj[i] = new List<int>();

        foreach (int e in edgeIndices)
        {
            int u = edges[e].u;
            int v = edges[e].v;
            if (Find(u) != Find(v))
            {
                Union(u, v);
                treeAdj[u].Add(e);
                treeAdj[v].Add(e);
            }
        }

        int[] treeParentEdge = new int[N];
        for (int i = 0; i < N; i++) treeParentEdge[i] = -1;
        int[] treeParentNode = new int[N];
        for (int i = 0; i < N; i++) treeParentNode[i] = -1;

        List<int> order = new List<int>(N);
        bool[] visited = new bool[N];

        for (int startNode = 0; startNode < N; startNode++)
        {
            if (visited[startNode]) continue;

            Queue<int> q = new Queue<int>();
            q.Enqueue(startNode);
            visited[startNode] = true;

            int componentStartIdx = order.Count;

            while (q.Count > 0)
            {
                int u = q.Dequeue();
                order.Add(u);

                foreach (int e in treeAdj[u])
                {
                    int v = edges[e].u == u ? edges[e].v : edges[e].u;
                    if (!visited[v])
                    {
                        visited[v] = true;
                        treeParentEdge[v] = e;
                        treeParentNode[v] = u;
                        q.Enqueue(v);
                    }
                }
            }

            int count = order.Count - componentStartIdx;
            for (int i = 0; i < count / 2; i++)
            {
                int temp = order[componentStartIdx + i];
                order[componentStartIdx + i] = order[componentStartIdx + count - 1 - i];
                order[componentStartIdx + count - 1 - i] = temp;
            }
        }

        int[] currentDegree = new int[N];
        for (int i = 0; i < N; i++) currentDegree[i] = originalDegree[i];

        foreach (int u in order)
        {
            int p = treeParentNode[u];
            if (p != -1)
            {
                if (currentDegree[u] % 2 != 0)
                {
                    int e = treeParentEdge[u];
                    DuplicateEdge(edges[e].u, edges[e].v, edges[e].w);
                    currentDegree[u]++;
                    currentDegree[p]++;
                }
            }
        }
    }
}
