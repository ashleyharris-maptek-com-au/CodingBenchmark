const fs = require('fs');

function solve() {
    const startTime = Date.now();
    const TIME_LIMIT = 4000; 

    const input = fs.readFileSync(0, 'utf8');
    if (!input || input.trim() === '') return;
    const data = JSON.parse(input);
    const depot = { id: 'depot', x: data.depot[0], y: data.depot[1] };
    const maxTrucks = data.truckCount;
    const packages = data.packages;

    if (!packages || packages.length === 0) {
        console.log(JSON.stringify({ routes: [] }));
        return;
    }

    const dist = (a, b) => {
        const dx = a.x - b.x;
        const dy = a.y - b.y;
        return Math.sqrt(dx * dx + dy * dy);
    };

    packages.forEach(p => {
        p.angle = Math.atan2(p.y - depot.y, p.x - depot.x);
    });
    
    packages.sort((a, b) => a.angle - b.angle);

    const routes = [];
    const numRoutes = Math.min(maxTrucks, packages.length);
    const baseSize = Math.floor(packages.length / numRoutes);
    let remainder = packages.length % numRoutes;

    let pIdx = 0;
    for (let i = 0; i < numRoutes; i++) {
        const routeSize = baseSize + (remainder > 0 ? 1 : 0);
        remainder--;
        const route = [];
        for (let j = 0; j < routeSize; j++) {
            route.push(packages[pIdx++]);
        }
        routes.push(route);
    }

    for (let r = 0; r < routes.length; r++) {
        if (routes[r].length === 0) continue;
        
        if (routes[r].length <= 15000) {
            const ordered = [];
            const unvisited = new Set(routes[r]);
            let curr = depot;
            while (unvisited.size > 0) {
                let bestDist = Infinity;
                let bestNode = null;
                for (const node of unvisited) {
                    const d = dist(curr, node);
                    if (d < bestDist) {
                        bestDist = d;
                        bestNode = node;
                    }
                }
                ordered.push(bestNode);
                unvisited.delete(bestNode);
                curr = bestNode;
            }
            routes[r] = ordered;
        }
    }

    let iterCount = 0;
    const isTimeUp = () => {
        if ((++iterCount & 63) === 0) {
            return Date.now() - startTime > TIME_LIMIT;
        }
        return false;
    };

    const optimizeRoute = (route) => {
        if (route.length < 2) return;
        let improved = true;
        while (improved) {
            improved = false;
            for (let i = 0; i < route.length - 1; i++) {
                if (isTimeUp()) return;
                for (let j = i + 1; j < route.length; j++) {
                    const nodeI = route[i];
                    const prevI = i === 0 ? depot : route[i-1];
                    const nodeJ = route[j];
                    const nextJ = j === route.length - 1 ? depot : route[j+1];

                    const costBefore = dist(prevI, nodeI) + dist(nodeJ, nextJ);
                    const costAfter = dist(prevI, nodeJ) + dist(nodeI, nextJ);

                    if (costBefore - costAfter > 1e-7) {
                        let left = i;
                        let right = j;
                        while (left < right) {
                            const temp = route[left];
                            route[left] = route[right];
                            route[right] = temp;
                            left++;
                            right--;
                        }
                        improved = true;
                    }
                }
            }
        }
    };

    for (const r of routes) optimizeRoute(r);

    let improved = true;
    searchLoop: while (improved) {
        if (isTimeUp()) break;
        improved = false;

        // 1. Relocate
        for (let r1 = 0; r1 < routes.length; r1++) {
            for (let i = 0; i < routes[r1].length; i++) {
                if (isTimeUp()) break searchLoop;
                const node = routes[r1][i];
                const prev1 = i === 0 ? depot : routes[r1][i-1];
                const next1 = i === routes[r1].length - 1 ? depot : routes[r1][i+1];
                const costRemoved = dist(prev1, node) + dist(node, next1) - dist(prev1, next1);

                let bestDelta = 1e-7;
                let bestMove = null;

                for (let r2 = 0; r2 < routes.length; r2++) {
                    if (r1 === r2) continue;
                    for (let j = 0; j <= routes[r2].length; j++) {
                        const prev2 = j === 0 ? depot : routes[r2][j-1];
                        const next2 = j === routes[r2].length ? depot : routes[r2][j];
                        
                        const costAdded = dist(prev2, node) + dist(node, next2) - dist(prev2, next2);
                        const delta = costRemoved - costAdded;
                        if (delta > bestDelta) {
                            bestDelta = delta;
                            bestMove = { r2, j };
                        }
                    }
                }

                if (bestMove) {
                    routes[r1].splice(i, 1);
                    routes[bestMove.r2].splice(bestMove.j, 0, node);
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
        if (improved) continue;

        // 2. Swap
        for (let r1 = 0; r1 < routes.length; r1++) {
            for (let i = 0; i < routes[r1].length; i++) {
                if (isTimeUp()) break searchLoop;
                const node1 = routes[r1][i];
                const prev1 = i === 0 ? depot : routes[r1][i-1];
                const next1 = i === routes[r1].length - 1 ? depot : routes[r1][i+1];

                let bestDelta = 1e-7;
                let bestMove = null;

                for (let r2 = r1 + 1; r2 < routes.length; r2++) {
                    for (let j = 0; j < routes[r2].length; j++) {
                        const node2 = routes[r2][j];
                        const prev2 = j === 0 ? depot : routes[r2][j-1];
                        const next2 = j === routes[r2].length - 1 ? depot : routes[r2][j+1];

                        const costBefore = dist(prev1, node1) + dist(node1, next1) +
                                           dist(prev2, node2) + dist(node2, next2);
                        
                        const costAfter = dist(prev1, node2) + dist(node2, next1) +
                                          dist(prev2, node1) + dist(node1, next2);

                        const delta = costBefore - costAfter;
                        if (delta > bestDelta) {
                            bestDelta = delta;
                            bestMove = { r2, j };
                        }
                    }
                }

                if (bestMove) {
                    const temp = routes[r1][i];
                    routes[r1][i] = routes[bestMove.r2][bestMove.j];
                    routes[bestMove.r2][bestMove.j] = temp;
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
        if (improved) continue;

        // 3. Cross-exchange
        for (let r1 = 0; r1 < routes.length; r1++) {
            for (let r2 = r1 + 1; r2 < routes.length; r2++) {
                for (let i = 0; i <= routes[r1].length; i++) {
                    if (isTimeUp()) break searchLoop;
                    for (let j = 0; j <= routes[r2].length; j++) {
                        const prev1 = i === 0 ? depot : routes[r1][i-1];
                        const next1 = i === routes[r1].length ? depot : routes[r1][i];
                        
                        const prev2 = j === 0 ? depot : routes[r2][j-1];
                        const next2 = j === routes[r2].length ? depot : routes[r2][j];
                        
                        const costBefore = dist(prev1, next1) + dist(prev2, next2);
                        const costAfter = dist(prev1, next2) + dist(prev2, next1);
                        
                        if (costBefore - costAfter > 1e-7) {
                            const tail1 = routes[r1].splice(i);
                            const tail2 = routes[r2].splice(j);
                            
                            routes[r1].push(...tail2);
                            routes[r2].push(...tail1);
                            improved = true;
                            break;
                        }
                    }
                    if (improved) break;
                }
                if (improved) break;
            }
            if (improved) break;
        }

        if (improved) {
            for (const r of routes) optimizeRoute(r);
        }
    }

    const output = {
        routes: routes.filter(r => r.length > 0).map(r => r.map(p => p.id))
    };

    console.log(JSON.stringify(output));
}

solve();
