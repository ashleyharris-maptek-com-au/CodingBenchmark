```rust
use std::collections::VecDeque;
use std::io::{self, BufRead, Read, Write};

const PARENT_UP: u8 = 1;
const PARENT_DOWN: u8 = 2;
const PARENT_LEFT: u8 = 3;
const PARENT_RIGHT: u8 = 4;

fn main() {
    let stdin = io::stdin();
    let mut handle = stdin.lock();

    // Read the first line to get H and W
    let mut header = String::new();
    loop {
        let n = handle.read_line(&mut header).expect("Failed to read");
        if n == 0 {
            return;
        }
        if header.trim().is_empty() {
            header.clear();
            continue;
        }
        break;
    }

    let mut parts = header.trim().split_whitespace();
    let h: u32 = parts.next().unwrap().parse().unwrap();
    let w: u32 = parts.next().unwrap().parse().unwrap();

    let total = (h as usize) * (w as usize);
    let mut maze = Vec::with_capacity(total);
    let mut start = 0;
    let mut end = 0;
    let maze_ptr = maze.as_mut_ptr();
    let mut ptr = 0;

    // Use a large buffer to minimize system calls
    let mut buf = vec![0u8; 1024 * 1024];
    
    // Fast path parsing, ignoring newlines and extracting start/end
    'read_loop: while ptr < total {
        let n = handle.read(&mut buf).expect("Failed to read");
        if n == 0 {
            break;
        }
        for &b in &buf[..n] {
            if b >= 32 { // Ignore control characters like \n and \r
                unsafe {
                    std::ptr::write(maze_ptr.add(ptr), b);
                }
                if b == b'A' {
                    start = ptr as u32;
                } else if b == b'B' {
                    end = ptr as u32;
                }
                ptr += 1;
                if ptr == total {
                    break 'read_loop;
                }
            }
        }
    }
    
    if ptr < total {
        panic!("Incomplete maze data");
    }
    
    unsafe { maze.set_len(total); }

    let w_u32 = w as u32;
    let total_u32 = total as u32;

    // Use VecDeque for BFS to minimize memory footprint
    let mut q = VecDeque::with_capacity(1024 * 1024);
    // Queue stores (1D index, current column) to avoid expensive modulo operations
    q.push_back((start, start % w_u32));

    let mut found = false;

    while let Some((curr, c)) = q.pop_front() {
        if curr == end {
            found = true;
            break;
        }

        // UP
        if curr >= w_u32 {
            let next = curr - w_u32;
            let cell = maze[next as usize];
            if cell == b' ' || cell == b'B' {
                maze[next as usize] = PARENT_DOWN;
                q.push_back((next, c));
                if cell == b'B' {
                    found = true;
                    break;
                }
            }
        }
        // DOWN
        if curr + w_u32 < total_u32 {
            let next = curr + w_u32;
            let cell = maze[next as usize];
            if cell == b' ' || cell == b'B' {
                maze[next as usize] = PARENT_UP;
                q.push_back((next, c));
                if cell == b'B' {
                    found = true;
                    break;
                }
            }
        }
        // LEFT
        if c > 0 {
            let next = curr - 1;
            let cell = maze[next as usize];
            if cell == b' ' || cell == b'B' {
                maze[next as usize] = PARENT_RIGHT;
                q.push_back((next, c - 1));
                if cell == b'B' {
                    found = true;
                    break;
                }
            }
        }
        // RIGHT
        if c + 1 < w_u32 {
            let next = curr + 1;
            let cell = maze[next as usize];
            if cell == b' ' || cell == b'B' {
                maze[next as usize] = PARENT_LEFT;
                q.push_back((next, c + 1));
                if cell == b'B' {
                    found = true;
                    break;
                }
            }
        }
    }

    if found {
        let mut path = Vec::new();
        let mut curr = end;
        // Trace back the shortest path
        while curr != start {
            path.push(curr);
            let dir = maze[curr as usize];
            if dir == PARENT_UP {
                curr -= w_u32;
            } else if dir == PARENT_DOWN {
                curr += w_u32;
            } else if dir == PARENT_LEFT {
                curr -= 1;
            } else if dir == PARENT_RIGHT {
                curr += 1;
            } else {
                panic!("Invalid path trace at {}", curr);
            }
        }
        path.push(start);
        path.reverse();

        // High-performance stdout buffering
        let stdout = io::stdout();
        let mut handle = io::BufWriter::new(stdout.lock());
        writeln!(handle, "{}", path.len()).unwrap();
        for p in path {
            let r = p / w_u32;
            let col = p % w_u32;
            // Output format: x y (0-indexed coordinates from top-left)
            writeln!(handle, "{} {}", col, r).unwrap();
        }
    } else {
        println!("0");
    }
}
```
