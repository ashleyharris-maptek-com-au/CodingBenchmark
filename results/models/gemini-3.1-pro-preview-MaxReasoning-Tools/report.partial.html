declare const require: any;
const fs = require('fs');

interface Point {
    x: number;
    y: number;
}

interface Package extends Point {
    id: any;
}

interface InputData {
    depot: [number, number];
    truckCount: number;
    packages: Package[];
}

function solve() {
    const input: string = fs.readFileSync(0, 'utf8');
    if (!input.trim()) return;
    
    const data: InputData = JSON.parse(input);
    const depot: Point = { x: data.depot[0], y: data.depot[1] };
    const truckCount: number = data.truckCount;
    let packages: Package[] = data.packages;

    if (!packages || packages.length === 0) {
        console.log(JSON.stringify({ routes: [] }));
        return;
    }

    const dist = (p1: Point, p2: Point) => Math.sqrt((p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y));

    // Sort by angle from depot
    packages.sort((a, b) => {
        let angleA = Math.atan2(a.y - depot.y, a.x - depot.x);
        let angleB = Math.atan2(b.y - depot.y, b.x - depot.x);
        return angleA - angleB;
    });

    let maxGap = 0;
    let maxGapIdx = 0;
    const angles = packages.map(p => {
        let a = Math.atan2(p.y - depot.y, p.x - depot.x);
        return a < 0 ? a + 2 * Math.PI : a;
    });
    
    for (let i = 0; i < packages.length; i++) {
        let next = (i + 1) % packages.length;
        let gap = angles[next] - angles[i];
        if (gap < 0) gap += 2 * Math.PI;
        if (gap > maxGap) {
            maxGap = gap;
            maxGapIdx = i;
        }
    }
    
    const shift = (maxGapIdx + 1) % packages.length;
    const shiftedPackages = packages.slice(shift).concat(packages.slice(0, shift));

    const CHUNK_SIZE = 1500;
    const M = Math.max(truckCount, Math.ceil(shiftedPackages.length / CHUNK_SIZE));
    const actualChunks = Math.min(M, shiftedPackages.length);

    const chunks: Package[][] = Array.from({ length: actualChunks }, () => []);
    for (let i = 0; i < shiftedPackages.length; i++) {
        const chunkIdx = Math.floor((i * actualChunks) / shiftedPackages.length);
        chunks[chunkIdx].push(shiftedPackages[i]);
    }

    let routesArr: Package[][] = [];
    for (let c = 0; c < actualChunks; c++) {
        const cluster = chunks[c];
        if (cluster.length === 0) continue;
        
        let unvisited = [...cluster];
        let current: Point = depot;
        const route: Package[] = [];
        
        while (unvisited.length > 0) {
            let bestDistSq = Infinity;
            let bestIdx = -1;
            for (let i = 0; i < unvisited.length; i++) {
                const node = unvisited[i];
                const dx = current.x - node.x;
                const dy = current.y - node.y;
                const dSq = dx * dx + dy * dy; 
                if (dSq < bestDistSq) {
                    bestDistSq = dSq;
                    bestIdx = i;
                }
            }
            route.push(unvisited[bestIdx]);
            current = unvisited[bestIdx];
            unvisited[bestIdx] = unvisited[unvisited.length - 1];
            unvisited.pop();
        }
        routesArr.push(route);
    }

    const startTime = Date.now();
    const timeLimit = 1500; 

    let improvedGlobal = true;
    while (improvedGlobal) {
        if (Date.now() - startTime > timeLimit) break;
        improvedGlobal = false;
        
        for (let r = 0; r < routesArr.length; r++) {
            if (Date.now() - startTime > timeLimit) break;
            const route = routesArr[r];
            if (route.length < 2) continue;
            
            let improvedThisPass = false;
            for (let i = 0; i < route.length - 1; i++) {
                if (Date.now() - startTime > timeLimit) break;
                for (let k = i + 1; k < route.length; k++) {
                    const p1 = i === 0 ? depot : route[i - 1];
                    const p2 = route[i];
                    const p3 = route[k];
                    const p4 = k === route.length - 1 ? depot : route[k + 1];

                    const dBefore = dist(p1, p2) + dist(p3, p4);
                    const dAfter = dist(p1, p3) + dist(p2, p4);

                    if (dAfter < dBefore - 1e-6) {
                        let left = i;
                        let right = k;
                        while (left < right) {
                            const temp = route[left];
                            route[left] = route[right];
                            route[right] = temp;
                            left++;
                            right--;
                        }
                        improvedThisPass = true;
                        improvedGlobal = true;
                    }
                }
            }
        }
    }

    while (routesArr.length > truckCount && routesArr.length > 1) {
        let nextRoutes: Package[][] = [];
        for (let i = 0; i < routesArr.length; i += 2) {
            if (i + 1 < routesArr.length) {
                let R1 = routesArr[i];
                let R2 = routesArr[i+1];
                let d1 = dist(R1[R1.length - 1], R2[0]);
                let d2 = dist(R1[R1.length - 1], R2[R2.length - 1]);
                let d3 = dist(R1[0], R2[0]);
                let d4 = dist(R1[0], R2[R2.length - 1]);
                
                let minD = Math.min(d1, d2, d3, d4);
                if (minD === d2) {
                    R2.reverse();
                } else if (minD === d3) {
                    R1.reverse();
                } else if (minD === d4) {
                    R1.reverse();
                    R2.reverse();
                }
                nextRoutes.push(R1.concat(R2));
            } else {
                nextRoutes.push(routesArr[i]);
            }
        }
        routesArr = nextRoutes;
    }

    const finalRoutes = routesArr.map(route => route.map(p => p.id));
    console.log(JSON.stringify({ routes: finalRoutes }));
}

solve();
