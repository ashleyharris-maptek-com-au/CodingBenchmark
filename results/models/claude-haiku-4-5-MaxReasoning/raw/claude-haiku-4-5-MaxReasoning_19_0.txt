using System;
using System.Collections.Generic;
using System.Linq;

class MiningOptimizer
{
    struct Voxel
    {
        public int x, y, z;
        public int mass;
        public int value;
    }

    struct Operation
    {
        public int digX, digY, digZ;
        public List<(int, int)> steps;
        public int destX, destY;
    }

    static void Main()
    {
        var header = Console.ReadLine().Split(' ');
        int xCount = int.Parse(header[0]);
        int yCount = int.Parse(header[1]);
        int zCount = int.Parse(header[2]);
        double xBlockSize = double.Parse(header[3]);
        double yBlockSize = double.Parse(header[4]);
        double zBlockSize = double.Parse(header[5]);

        // Read voxels
        var voxels = new Dictionary<(int, int, int), (int mass, int value)>();
        var heightMap = new int[xCount, yCount];

        for (int i = 0; i < xCount; i++)
            for (int j = 0; j < yCount; j++)
                heightMap[i, j] = 0;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int b = Console.In.Read();
                    if (b == -1) break;

                    int mass = b % 16;
                    int value = b / 16;

                    if (mass > 0)
                    {
                        voxels[(x, y, z)] = (mass, value);
                        if (z >= heightMap[x, y])
                            heightMap[x, y] = z + 1;
                    }
                }
            }
        }

        var operations = new List<Operation>();
        var processed = new HashSet<(int, int, int)>();
        var currentHeights = new int[xCount, yCount];
        Array.Copy(heightMap, currentHeights, heightMap.Length);

        // Sort voxels by profitability (value * mass) in descending order
        var sortedVoxels = voxels
            .Where(kv => kv.Value.value > 0) // Only valuable ore
            .OrderByDescending(kv => kv.Value.value * kv.Value.mass)
            .ToList();

        foreach (var voxel in sortedVoxels)
        {
            int x = voxel.Key.x;
            int y = voxel.Key.y;
            int z = voxel.Key.z;

            if (processed.Contains((x, y, z)))
                continue;

            // Check if accessible (exposed)
            if (!IsExposed(voxels, processed, x, y, z, currentHeights))
                continue;

            // Find path to processing plant (0, 0)
            var path = FindPath(x, y, currentHeights, xCount, yCount);
            if (path == null || path.Count == 0)
                continue;

            var op = new Operation
            {
                digX = x,
                digY = y,
                digZ = z,
                steps = path,
                destX = 0,
                destY = 0
            };

            operations.Add(op);
            processed.Add((x, y, z));

            // Update height map after digging
            bool hasMore = false;
            for (int zz = z; zz >= 0; zz--)
            {
                if (voxels.ContainsKey((x, y, zz)) && !processed.Contains((x, y, zz)))
                {
                    currentHeights[x, y] = zz + 1;
                    hasMore = true;
                    break;
                }
            }
            if (!hasMore)
                currentHeights[x, y] = 0;
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.steps.Count} {op.digX} {op.digY} {op.digZ}");
            foreach (var step in op.steps)
                Console.Write($" {step.Item1} {step.Item2}");
            Console.WriteLine($" {op.destX} {op.destY}");
        }
    }

    static bool IsExposed(
        Dictionary<(int, int, int), (int mass, int value)> voxels,
        HashSet<(int, int, int)> processed,
        int x, int y, int z,
        int[,] heights)
    {
        // Check if exposed from above
        if (z >= heights[x, y] - 1)
            return true;

        // Check if exposed from sides
        int[] dx = { 1, -1, 0, 0 };
        int[] dy = { 0, 0, 1, -1 };

        foreach (int d = 0; d < 4; d++)
        {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx >= 0 && ny >= 0 && nx < heights.GetLength(0) && ny < heights.GetLength(1))
            {
                if (!voxels.ContainsKey((nx, ny, z)))
                    return true;
            }
        }

        return false;
    }

    static List<(int, int)> FindPath(int sx, int sy, int[,] heights, int xCount, int yCount)
    {
        // Simple pathfinding: try to move towards (0, 0)
        var path = new List<(int, int)>();
        int x = sx, y = sy;
        int steps = 0;
        int maxSteps = xCount + yCount + 100;

        while ((x != 0 || y != 0) && steps < maxSteps)
        {
            int bestDx = 0, bestDy = 0;
            int bestCost = int.MaxValue;

            // Try moving in 4 directions
            int[] dx = { 1, -1, 0, 0 };
            int[] dy = { 0, 0, 1, -1 };

            for (int d = 0; d < 4; d++)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx < 0 || ny < 0 || nx >= xCount || ny >= yCount)
                    continue;

                int heightDiff = Math.Abs(heights[nx, ny] - heights[x, y]);
                if (heightDiff > 1)
                    continue; // Slope constraint

                int cost = Math.Abs(nx) + Math.Abs(ny);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestDx = dx[d];
                    bestDy = dy[d];
                }
            }

            if (bestCost == int.MaxValue)
                break;

            x += bestDx;
            y += bestDy;
            path.Add((bestDx, bestDy));
            steps++;
        }

        return (x == 0 && y == 0) ? path : null;
    }
}
