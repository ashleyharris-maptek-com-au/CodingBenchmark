using System;
using System.Collections.Generic;

class MinesweeperSolver
{
    static void Main()
    {
        string[] parts = Console.ReadLine().Split();
        int height = int.Parse(parts[0]);
        int width = int.Parse(parts[1]);

        var board = new char[height][];
        for (int i = 0; i < height; i++)
        {
            board[i] = Console.ReadLine().ToCharArray();
        }

        var safeCells = FindSafeCells(board, height, width);

        Console.WriteLine(safeCells.Count);
        foreach (var (x, y) in safeCells)
        {
            Console.WriteLine($"{x} {y}");
        }
    }

    static List<(int, int)> FindSafeCells(char[][] board, int height, int width)
    {
        var safeCells = new HashSet<(int, int)>();
        var mineLocations = new HashSet<(int, int)>();
        var queue = new Queue<(int, int)>();

        // Identify initial mines and queue numbered cells
        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                if (board[y][x] == '*')
                    mineLocations.Add((x, y));
                else if (board[y][x] >= '0' && board[y][x] <= '8')
                    queue.Enqueue((x, y));
            }
        }

        var processed = new HashSet<(int, int)>();

        // Process numbered cells using constraint propagation
        while (queue.Count > 0)
        {
            var (x, y) = queue.Dequeue();

            if (processed.Contains((x, y)))
                continue;
            processed.Add((x, y));

            if (board[y][x] < '0' || board[y][x] > '8')
                continue;

            int mineCount = board[y][x] - '0';
            var neighbors = GetNeighbors(x, y, width, height);

            int flaggedCount = 0;
            var unknownCells = new List<(int, int)>();

            // Analyze neighbors
            foreach (var (nx, ny) in neighbors)
            {
                if (mineLocations.Contains((nx, ny)))
                    flaggedCount++;
                else if (board[ny][nx] == ' ')
                    unknownCells.Add((nx, ny));
            }

            // Deduction 1: If all mines are flagged, remaining unknowns are safe
            if (flaggedCount == mineCount)
            {
                foreach (var (ux, uy) in unknownCells)
                {
                    if (safeCells.Add((ux, uy)))
                    {
                        board[uy][ux] = '0';

                        // Queue neighboring numbered cells for reprocessing
                        var neighbors2 = GetNeighbors(ux, uy, width, height);
                        foreach (var (nnx, nny) in neighbors2)
                        {
                            if (board[nny][nnx] >= '0' && board[nny][nnx] <= '8')
                                queue.Enqueue((nnx, nny));
                        }
                    }
                }
            }
            // Deduction 2: If unknowns + flagged = mineCount, all unknowns are mines
            else if (unknownCells.Count + flaggedCount == mineCount && unknownCells.Count > 0)
            {
                foreach (var (ux, uy) in unknownCells)
                {
                    if (mineLocations.Add((ux, uy)))
                    {
                        board[uy][ux] = '*';

                        // Queue neighboring numbered cells for reprocessing
                        var neighbors2 = GetNeighbors(ux, uy, width, height);
                        foreach (var (nnx, nny) in neighbors2)
                        {
                            if (board[nny][nnx] >= '0' && board[nny][nnx] <= '8')
                                queue.Enqueue((nnx, nny));
                        }
                    }
                }
            }
        }

        // Sort results by x, then by y
        var result = new List<(int, int)>(safeCells);
        result.Sort((a, b) =>
        {
            if (a.Item1 != b.Item1) return a.Item1.CompareTo(b.Item1);
            return a.Item2.CompareTo(b.Item2);
        });

        return result;
    }

    static List<(int, int)> GetNeighbors(int x, int y, int width, int height)
    {
        var neighbors = new List<(int, int)>();
        for (int dy = -1; dy <= 1; dy++)
        {
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                int ny = y + dy;
                if (nx >= 0 && nx < width && ny >= 0 && ny < height)
                    neighbors.Add((nx, ny));
            }
        }
        return neighbors;
    }
}
