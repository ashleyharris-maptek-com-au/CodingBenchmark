#include <iostream>
#include <string>
#include <vector>
#include <algorithm>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;
    cin.ignore();

    vector<string> strings(n);
    for (int i = 0; i < n; i++) {
        getline(cin, strings[i]);
    }

    if (n == 0) {
        cout << "\n";
        return 0;
    }

    // Find the shortest string - LCS cannot exceed its length
    int min_idx = 0;
    size_t min_len = strings[0].length();
    for (int i = 1; i < n; i++) {
        if (strings[i].length() < min_len) {
            min_len = strings[i].length();
            min_idx = i;
        }
    }

    // Try substrings of the shortest string from longest to shortest
    // This allows early termination when we find a match
    for (int len = min_len; len >= 1; len--) {
        for (size_t start = 0; start <= min_len - len; start++) {
            string substring = strings[min_idx].substr(start, len);

            // Check if this substring is in all other strings
            bool in_all = true;
            for (int i = 0; i < n; i++) {
                if (strings[i].find(substring) == string::npos) {
                    in_all = false;
                    break;
                }
            }

            if (in_all) {
                cout << substring << "\n";
                return 0;
            }
        }
    }

    // No common substring found
    cout << "\n";
    return 0;
}
