#include <iostream>
#include <vector>
#include <climits>

using namespace std;

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    cin >> J >> M;

    // Store job specifications: jobs[j][t] = {machine, duration}
    vector<vector<pair<int, int>>> jobs(J);
    int total_tasks = 0;

    for (int j = 0; j < J; j++) {
        int T;
        cin >> T;
        jobs[j].resize(T);
        total_tasks += T;
        for (int t = 0; t < T; t++) {
            cin >> jobs[j][t].first >> jobs[j][t].second;
        }
    }

    // Adaptive greedy scheduling algorithm:
    // Works efficiently for both small and large problem instances
    // by interleaving job scheduling and always choosing the next ready task

    vector<long long> machine_free_time(M, 0);
    vector<int> next_task(J, 0);
    vector<long long> job_free_time(J, 0);
    vector<vector<pair<long long, int>>> schedule(J);

    for (int j = 0; j < J; j++) {
        schedule[j].resize(jobs[j].size());
    }

    // Schedule all tasks greedily
    for (int scheduled = 0; scheduled < total_tasks; scheduled++) {
        // Find the job with the earliest available time for its next task
        int best_job = -1;
        long long best_time = LLONG_MAX;

        for (int j = 0; j < J; j++) {
            if (next_task[j] < (int)jobs[j].size() && job_free_time[j] < best_time) {
                best_time = job_free_time[j];
                best_job = j;
            }
        }

        if (best_job == -1) break;

        int j = best_job;
        int t = next_task[j];
        int machine = jobs[j][t].first;
        int duration = jobs[j][t].second;

        // Schedule this task as early as possible on its machine
        // considering both job sequence constraint and machine availability
        long long start_time = max(job_free_time[j], machine_free_time[machine]);
        schedule[j][t] = {start_time, machine};

        long long end_time = start_time + duration;
        machine_free_time[machine] = end_time;
        job_free_time[j] = end_time;
        next_task[j]++;
    }

    // Output schedule for each job
    for (int j = 0; j < J; j++) {
        for (int t = 0; t < (int)schedule[j].size(); t++) {
            if (t > 0) cout << " ";
            cout << schedule[j][t].first << " " << schedule[j][t].second;
        }
        cout << "\n";
    }

    return 0;
}
