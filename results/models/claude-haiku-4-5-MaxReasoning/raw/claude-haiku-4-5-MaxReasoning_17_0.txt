using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    struct Box
    {
        public int W, H, D, Index;
        public long Volume => (long)W * H * D;
    }

    struct Placement
    {
        public int Index, X, Y, Z;
    }

    struct PackedBox
    {
        public int X, Y, Z, W, H, D;
    }

    static void Main()
    {
        var line = Console.ReadLine().Split();
        int N = int.Parse(line[0]);
        int containerW = int.Parse(line[1]);
        int containerH = int.Parse(line[2]);
        int containerD = int.Parse(line[3]);

        var boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            var dims = Console.ReadLine().Split();
            boxes.Add(new Box
            {
                W = int.Parse(dims[0]),
                H = int.Parse(dims[1]),
                D = int.Parse(dims[2]),
                Index = i
            });
        }

        var packed = PackBoxes(boxes, containerW, containerH, containerD);

        Console.WriteLine(packed.Count);
        foreach (var p in packed)
            Console.WriteLine($"{p.Index} {p.X} {p.Y} {p.Z}");
    }

    static List<Placement> PackBoxes(List<Box> boxes, int cW, int cH, int cD)
    {
        var sorted = boxes.OrderByDescending(b => b.Volume).ToList();
        var placements = new List<Placement>();
        var packed = new List<PackedBox>();

        foreach (var box in sorted)
        {
            var orientations = GetOrientations(box);
            bool placed = false;

            foreach (var (w, h, d) in orientations)
            {
                var pos = FindPosition(w, h, d, cW, cH, cD, packed);
                if (pos.HasValue)
                {
                    var (x, y, z) = pos.Value;
                    packed.Add(new PackedBox { X = x, Y = y, Z = z, W = w, H = h, D = d });
                    placements.Add(new Placement { Index = box.Index, X = x, Y = y, Z = z });
                    placed = true;
                    break;
                }
            }
        }

        return placements;
    }

    static (int, int, int)? FindPosition(int w, int h, int d, int cW, int cH, int cD, List<PackedBox> packed)
    {
        var candidates = new List<(int, int, int)>();

        candidates.Add((0, 0, 0));

        foreach (var p in packed)
        {
            if (p.X + p.W < cW) candidates.Add((p.X + p.W, p.Y, p.Z));
            if (p.Y + p.H < cH) candidates.Add((p.X, p.Y + p.H, p.Z));
            if (p.Z + p.D < cD) candidates.Add((p.X, p.Y, p.Z + p.D));
        }

        foreach (var (x, y, z) in candidates.Distinct())
        {
            if (x + w <= cW && y + h <= cH && z + d <= cD)
            {
                bool canPlace = true;
                foreach (var p in packed)
                {
                    if (Overlaps(x, y, z, w, h, d, p.X, p.Y, p.Z, p.W, p.H, p.D))
                    {
                        canPlace = false;
                        break;
                    }
                }
                if (canPlace) return (x, y, z);
            }
        }

        return null;
    }

    static bool Overlaps(int x1, int y1, int z1, int w1, int h1, int d1,
                         int x2, int y2, int z2, int w2, int h2, int d2)
    {
        return !(x1 + w1 <= x2 || x2 + w2 <= x1 ||
                 y1 + h1 <= y2 || y2 + h2 <= y1 ||
                 z1 + d1 <= z2 || z2 + d2 <= z1);
    }

    static List<(int, int, int)> GetOrientations(Box b)
    {
        var set = new HashSet<(int, int, int)>
        {
            (b.W, b.H, b.D),
            (b.W, b.D, b.H),
            (b.H, b.W, b.D),
            (b.H, b.D, b.W),
            (b.D, b.W, b.H),
            (b.D, b.H, b.W)
        };
        return set.ToList();
    }
}
