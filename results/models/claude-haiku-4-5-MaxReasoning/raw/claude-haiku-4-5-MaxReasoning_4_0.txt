#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <random>
#include <array>
#include <numeric>
#include <chrono>

using namespace std;

struct Vec3 {
    double x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(double x, double y, double z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
    double dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x); }
    double length() const { return sqrt(x * x + y * y + z * z); }
    double dist(const Vec3& v) const { return (*this - v).length(); }
};

struct Tetrahedron {
    Vec3 center;
    double rx, ry, rz;
};

const double TETRA_HEIGHT = 0.8164965809;
const double TETRA_CIRCUMRADIUS = 0.6123724357;
const double TETRA_INRADIUS = 0.2041241884;
const double TETRA_EDGE_LENGTH = 1.0;

double tetrahedronVolume(double edgeLength) {
    return edgeLength * edgeLength * edgeLength / (6.0 * sqrt(2.0));
}

vector<Vec3> getTetrahedronVertices(const Vec3& center, double rx, double ry, double rz) {
    double h = TETRA_CIRCUMRADIUS;
    vector<Vec3> verts = {
        Vec3(0, 0, h),
        Vec3(2*h/sqrt(3), 0, -h/3),
        Vec3(-h/sqrt(3), h, -h/3),
        Vec3(-h/sqrt(3), -h, -h/3)
    };

    for (auto& v : verts) {
        // Rotation around x-axis
        double y = v.y * cos(rx) - v.z * sin(rx);
        double z = v.y * sin(rx) + v.z * cos(rx);
        v.y = y;
        v.z = z;

        // Rotation around y-axis
        double x = v.x * cos(ry) + v.z * sin(ry);
        z = -v.x * sin(ry) + v.z * cos(ry);
        v.x = x;
        v.z = z;

        // Rotation around z-axis
        x = v.x * cos(rz) - v.y * sin(rz);
        y = v.x * sin(rz) + v.y * cos(rz);
        v.x = x;
        v.y = y;

        v = v + center;
    }

    return verts;
}

bool pointInConvexPolyhedron(const Vec3& p, const vector<Vec3>& vertices) {
    if (vertices.size() < 4) return false;

    Vec3 center;
    for (const auto& v : vertices) center = center + v;
    center = center * (1.0 / vertices.size());

    int n = vertices.size();
    for (int i = 0; i < n; i++) {
        Vec3 v0 = vertices[i];
        Vec3 v1 = vertices[(i+1) % n];
        Vec3 v2 = center;

        Vec3 normal = (v1 - v0).cross(v2 - v0);
        double side = (p - v0).dot(normal);
        double centerSide = (center - v0).dot(normal);

        if ((side > 0) != (centerSide > 0)) return false;
    }

    return true;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edgeLength;
    cin >> V >> edgeLength;

    vector<Vec3> containerVertices(V);
    for (int i = 0; i < V; i++) {
        cin >> containerVertices[i].x >> containerVertices[i].y >> containerVertices[i].z;
    }

    Vec3 containerCenter;
    for (const auto& v : containerVertices) containerCenter = containerCenter + v;
    containerCenter = containerCenter * (1.0 / V);

    double maxDist = 0;
    for (const auto& v : containerVertices) {
        maxDist = max(maxDist, containerCenter.dist(v));
    }

    double tetraVolume = tetrahedronVolume(edgeLength);
    double estimatedCount = (4.0 / 3.0 * M_PI * maxDist * maxDist * maxDist) / (tetraVolume * 2.5);

    vector<Tetrahedron> placed;

    auto start = chrono::high_resolution_clock::now();
    mt19937 rng(42);

    double gridSpacing = edgeLength * 1.2;
    double radius = maxDist * 1.5;

    // Grid-based placement with jitter
    for (double x = containerCenter.x - radius; x <= containerCenter.x + radius; x += gridSpacing) {
        for (double y = containerCenter.y - radius; y <= containerCenter.y + radius; y += gridSpacing) {
            for (double z = containerCenter.z - radius; z <= containerCenter.z + radius; z += gridSpacing) {
                auto now = chrono::high_resolution_clock::now();
                if (chrono::duration<double>(now - start).count() > 27.0) break;

                Vec3 center(x, y, z);

                uniform_real_distribution<> jitter(-gridSpacing * 0.2, gridSpacing * 0.2);
                center.x += jitter(rng);
                center.y += jitter(rng);
                center.z += jitter(rng);

                if (!pointInConvexPolyhedron(center, containerVertices)) continue;

                uniform_real_distribution<> angleDist(0, 2 * M_PI);
                double rx = angleDist(rng);
                double ry = angleDist(rng);
                double rz = angleDist(rng);

                auto tetraVerts = getTetrahedronVertices(center, rx, ry, rz);

                bool valid = true;
                for (const auto& v : tetraVerts) {
                    if (!pointInConvexPolyhedron(v, containerVertices)) {
                        valid = false;
                        break;
                    }
                }

                if (!valid) continue;

                bool collision = false;
                for (const auto& existing : placed) {
                    double dist = center.dist(existing.center);
                    if (dist < TETRA_CIRCUMRADIUS * 1.8) {
                        collision = true;
                        break;
                    }
                }

                if (!collision) {
                    placed.push_back({center, rx, ry, rz});
                }
            }
            auto now = chrono::high_resolution_clock::now();
            if (chrono::duration<double>(now - start).count() > 27.0) break;
        }
        auto now = chrono::high_resolution_clock::now();
        if (chrono::duration<double>(now - start).count() > 27.0) break;
    }

    // Random placement for remaining space
    uniform_real_distribution<> coordDist(-1.0, 1.0);
    for (int iter = 0; iter < (int)(estimatedCount * 20) && placed.size() < (int)estimatedCount * 3; iter++) {
        auto now = chrono::high_resolution_clock::now();
        if (chrono::duration<double>(now - start).count() > 28.5) break;

        Vec3 center(
            containerCenter.x + coordDist(rng) * maxDist,
            containerCenter.y + coordDist(rng) * maxDist,
            containerCenter.z + coordDist(rng) * maxDist
        );

        if (!pointInConvexPolyhedron(center, containerVertices)) continue;

        uniform_real_distribution<> angleDist(0, 2 * M_PI);
        double rx = angleDist(rng);
        double ry = angleDist(rng);
        double rz = angleDist(rng);

        auto tetraVerts = getTetrahedronVertices(center, rx, ry, rz);

        bool valid = true;
        for (const auto& v : tetraVerts) {
            if (!pointInConvexPolyhedron(v, containerVertices)) {
                valid = false;
                break;
            }
        }

        if (!valid) continue;

        bool collision = false;
        for (const auto& existing : placed) {
            double dist = center.dist(existing.center);
            if (dist < TETRA_CIRCUMRADIUS * 1.8) {
                collision = true;
                break;
            }
        }

        if (!collision) {
            placed.push_back({center, rx, ry, rz});
        }
    }

    cout << placed.size() << "\n";
    cout << fixed;
    cout.precision(6);
    for (const auto& t : placed) {
        cout << t.center.x << " " << t.center.y << " " << t.center.z << " "
             << t.rx << " " << t.ry << " " << t.rz << "\n";
    }

    return 0;
}
