#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>

using namespace std;

const int INF = 1e9;

struct MaxFlow {
    struct Edge {
        int to, cap, rev;
    };

    vector<vector<Edge>> graph;
    vector<int> level, iter;
    int n;

    MaxFlow(int n) : n(n), graph(n), level(n), iter(n) {}

    void addUndirectedEdge(int u, int v, int cap) {
        graph[u].push_back({v, cap, (int)graph[v].size()});
        graph[v].push_back({u, cap, (int)graph[u].size() - 1});
    }

    bool bfs(int s, int t) {
        fill(level.begin(), level.end(), -1);
        queue<int> q;
        level[s] = 0;
        q.push(s);

        while (!q.empty()) {
            int v = q.front();
            q.pop();
            for (auto& e : graph[v]) {
                if (level[e.to] < 0 && e.cap > 0) {
                    level[e.to] = level[v] + 1;
                    q.push(e.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(int v, int t, int f) {
        if (v == t) return f;
        for (int& i = iter[v]; i < (int)graph[v].size(); i++) {
            Edge& e = graph[v][i];
            if (level[v] < level[e.to] && e.cap > 0) {
                int d = dfs(e.to, t, min(f, e.cap));
                if (d > 0) {
                    e.cap -= d;
                    graph[e.to][e.rev].cap += d;
                    return d;
                }
            }
        }
        return 0;
    }

    int maxFlow(int s, int t) {
        int flow = 0;
        while (bfs(s, t)) {
            fill(iter.begin(), iter.end(), 0);
            int f;
            while ((f = dfs(s, t, INF)) > 0) {
                flow += f;
            }
        }
        return flow;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    cin >> n >> m;

    vector<pair<int, int>> edges(m);

    for (int i = 0; i < m; i++) {
        cin >> edges[i].first >> edges[i].second;
    }

    int minCut = INF;
    vector<pair<int, int>> resultEdges;

    for (int t = 1; t < n; t++) {
        MaxFlow mf(n);

        for (const auto& e : edges) {
            mf.addUndirectedEdge(e.first, e.second, 1);
        }

        int flow = mf.maxFlow(0, t);

        if (flow < minCut) {
            minCut = flow;
            resultEdges.clear();

            for (int u = 0; u < n; u++) {
                for (const auto& edge : mf.graph[u]) {
                    if (edge.cap == 0) {
                        int v = edge.to;
                        for (const auto& orig : edges) {
                            if ((orig.first == u && orig.second == v) ||
                                (orig.first == v && orig.second == u)) {
                                bool found = false;
                                for (const auto& res : resultEdges) {
                                    if ((res.first == orig.first && res.second == orig.second) ||
                                        (res.first == orig.second && res.second == orig.first)) {
                                        found = true;
                                        break;
                                    }
                                }
                                if (!found) {
                                    resultEdges.push_back(orig);
                                }
                                break;
                            }
                        }
                    }
                }
            }
        }
    }

    cout << minCut << "\n";
    for (const auto& e : resultEdges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
