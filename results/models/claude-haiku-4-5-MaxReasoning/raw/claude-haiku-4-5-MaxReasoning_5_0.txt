use std::io::{self, BufRead};
use std::collections::{HashSet, VecDeque};

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let width: usize = parts[0].parse().unwrap();
    let height: usize = parts[1].parse().unwrap();
    let obstacle_count: usize = parts[2].parse().unwrap();
    let require_cycle: bool = parts[3].parse::<u8>().unwrap() == 1;

    let mut obstacles = HashSet::new();
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<usize> = line.split_whitespace()
            .map(|s| s.parse().unwrap())
            .collect();
        obstacles.insert((coords[0], coords[1]));
    }

    if obstacles.contains(&(0, 0)) {
        println!("0");
        return;
    }

    let mut cells = Vec::new();
    for y in 0..height {
        for x in 0..width {
            if !obstacles.contains(&(x, y)) {
                cells.push((x, y));
            }
        }
    }

    let total_cells = cells.len();

    if total_cells == 0 {
        println!("0");
        return;
    }

    if total_cells == 1 {
        println!("1");
        println!("0 0");
        return;
    }

    let mut black_count = 0;
    let mut white_count = 0;
    for (x, y) in &cells {
        if (x + y) % 2 == 0 {
            black_count += 1;
        } else {
            white_count += 1;
        }
    }

    if (black_count as i32 - white_count as i32).abs() > 1 {
        println!("0");
        return;
    }

    if obstacle_count < 20 {
        if let Some(path) = try_snake_pattern(width, height, &obstacles, total_cells, require_cycle) {
            println!("{}", path.len());
            for (x, y) in path {
                println!("{} {}", x, y);
            }
            return;
        }
    }

    let mut visited = HashSet::new();
    let mut path = Vec::new();

    visited.insert((0, 0));
    path.push((0, 0));

    if dfs(width, height, &obstacles, &mut visited, &mut path, total_cells, require_cycle) {
        println!("{}", path.len());
        for (x, y) in path {
            println!("{} {}", x, y);
        }
    } else {
        println!("0");
    }
}

fn try_snake_pattern(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    total_cells: usize,
    require_cycle: bool,
) -> Option<Vec<(usize, usize)>> {
    let mut path = Vec::new();
    let mut visited = HashSet::new();

    for y in 0..height {
        if y % 2 == 0 {
            for x in 0..width {
                if !obstacles.contains(&(x, y)) {
                    if path.is_empty() || (path.last().unwrap().0.abs_diff(x) +
                        path.last().unwrap().1.abs_diff(y)) == 1 {
                        path.push((x, y));
                        visited.insert((x, y));
                    } else {
                        return None;
                    }
                }
            }
        } else {
            for x in (0..width).rev() {
                if !obstacles.contains(&(x, y)) {
                    if path.is_empty() || (path.last().unwrap().0.abs_diff(x) +
                        path.last().unwrap().1.abs_diff(y)) == 1 {
                        path.push((x, y));
                        visited.insert((x, y));
                    } else {
                        return None;
                    }
                }
            }
        }
    }

    if visited.len() != total_cells {
        return None;
    }

    if require_cycle {
        let (last_x, last_y) = path[path.len() - 1];
        if (last_x.abs_diff(0) + last_y.abs_diff(0)) != 1 {
            return None;
        }
    }

    Some(path)
}

fn dfs(
    width: usize,
    height: usize,
    obstacles: &HashSet<(usize, usize)>,
    visited: &mut HashSet<(usize, usize)>,
    path: &mut Vec<(usize, usize)>,
    total_cells: usize,
    require_cycle: bool,
) -> bool {
    if visited.len() == total_cells {
        if require_cycle {
            let (last_x, last_y) = path[path.len() - 1];
            if (last_x.abs_diff(0) + last_y.abs_diff(0)) == 1 {
                return true;
            }
            return false;
        }
        return true;
    }

    let (x, y) = path[path.len() - 1];
    let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];

    let mut candidates = Vec::new();
    for (dx, dy) in &directions {
        let nx = x as i32 + dx;
        let ny = y as i32 + dy;

        if nx >= 0 && nx < width as i32 && ny >= 0 && ny < height as i32 {
            let nx = nx as usize;
            let ny = ny as usize;

            if !visited.contains(&(nx, ny)) && !obstacles.contains(&(nx, ny)) {
                candidates.push((nx, ny));
            }
        }
    }

    for (nx, ny) in candidates {
        visited.insert((nx, ny));
        path.push((nx, ny));

        if dfs(width, height, obstacles, visited, path, total_cells, require_cycle) {
            return true;
        }

        path.pop();
        visited.remove(&(nx, ny));
    }

    false
}
