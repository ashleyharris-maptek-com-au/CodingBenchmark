use std::io::{self, BufRead};

#[derive(Clone, Copy, Debug)]
struct Node {
    x: f64,
    y: f64,
}

struct Graph {
    nodes: Vec<Node>,
    edges: Vec<(usize, usize)>,
    n: usize,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read N and M
    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let n: usize = parts[0].parse().unwrap();
    let m: usize = parts[1].parse().unwrap();

    // Initialize nodes
    let nodes = vec![Node { x: 0.0, y: 0.0 }; n];
    let mut edges = Vec::new();

    // Read edges
    let mut edge_set = std::collections::HashSet::new();
    for _ in 0..m {
        if let Some(Ok(line)) = lines.next() {
            let parts: Vec<&str> = line.split_whitespace().collect();
            if parts.len() >= 2 {
                let u: usize = parts[0].parse().unwrap_or(0);
                let v: usize = parts[1].parse().unwrap_or(0);
                if u < n && v < n && u != v {
                    // Avoid duplicate edges
                    let edge = if u < v { (u, v) } else { (v, u) };
                    edge_set.insert(edge);
                }
            }
        }
    }

    for edge in edge_set {
        edges.push(edge);
    }

    let mut graph = Graph { nodes, edges, n };

    // Run layout algorithm
    layout(&mut graph);

    // Output
    println!("{}", n);
    for node in &graph.nodes {
        println!("{} {}", node.x, node.y);
    }
}

fn layout(graph: &mut Graph) {
    let n = graph.n;

    if n == 0 {
        return;
    }

    // Initialize positions in a circle
    let radius = (n as f64).sqrt() * 10.0;
    for i in 0..n {
        let angle = 2.0 * std::f64::consts::PI * (i as f64) / (n as f64);
        graph.nodes[i].x = angle.cos() * radius;
        graph.nodes[i].y = angle.sin() * radius;
    }

    // Adaptive iteration count based on graph size
    let iterations = match n {
        0..=50 => 200,
        51..=200 => 150,
        201..=1000 => 100,
        1001..=5000 => 50,
        5001..=20000 => 25,
        _ => 10,
    };

    // Run force-directed layout
    for iteration in 0..iterations {
        let progress = iteration as f64 / iterations as f64;
        let temp = 100.0 * (1.0 - progress).powi(2); // Quadratic cooling

        if n <= 3000 {
            // For smaller graphs, use all-pairs repulsion (O(n²))
            apply_forces_allpairs(graph, temp, iteration, iterations);
        } else {
            // For larger graphs, use grid-based spatial indexing
            apply_forces_grid(graph, temp, iteration, iterations);
        }
    }
}

fn apply_forces_allpairs(graph: &mut Graph, temp: f64, iteration: usize, total_iterations: usize) {
    let n = graph.n;
    let mut forces = vec![(0.0, 0.0); n];

    let k = (100.0 / (n as f64).sqrt()).max(1.0);

    // Repulsive forces (all pairs)
    for i in 0..n {
        for j in (i + 1)..n {
            let dx = graph.nodes[j].x - graph.nodes[i].x;
            let dy = graph.nodes[j].y - graph.nodes[i].y;
            let dist_sq = dx * dx + dy * dy;
            let dist = dist_sq.sqrt();

            if dist > 0.01 {
                let repulsive = k * k / (dist + 0.1);
                let fx = (dx / dist) * repulsive;
                let fy = (dy / dist) * repulsive;

                forces[i].0 -= fx;
                forces[i].1 -= fy;
                forces[j].0 += fx;
                forces[j].1 += fy;
            }
        }
    }

    // Attractive forces along edges
    for &(u, v) in &graph.edges {
        let dx = graph.nodes[v].x - graph.nodes[u].x;
        let dy = graph.nodes[v].y - graph.nodes[u].y;
        let dist = (dx * dx + dy * dy).sqrt();

        if dist > 0.01 {
            let attractive = dist * dist / (k * k);
            let fx = (dx / dist) * attractive;
            let fy = (dy / dist) * attractive;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }
    }

    // Update positions with temperature control
    for i in 0..n {
        let (fx, fy) = forces[i];
        let dist = (fx * fx + fy * fy).sqrt();

        if dist > 0.0 {
            let displacement = temp.min(dist);
            graph.nodes[i].x += (fx / dist) * displacement;
            graph.nodes[i].y += (fy / dist) * displacement;
        }
    }
}

fn apply_forces_grid(graph: &mut Graph, temp: f64, iteration: usize, total_iterations: usize) {
    let n = graph.n;
    let mut forces = vec![(0.0, 0.0); n];

    // Find bounds
    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for node in &graph.nodes {
        min_x = min_x.min(node.x);
        max_x = max_x.max(node.x);
        min_y = min_y.min(node.y);
        max_y = max_y.max(node.y);
    }

    let width = (max_x - min_x).max(1.0);
    let height = (max_y - min_y).max(1.0);

    // Create spatial grid
    let grid_size = ((n as f64).sqrt() / 2.0).ceil() as usize;
    let grid_size = grid_size.max(2).min(100);
    let cell_width = width / grid_size as f64;
    let cell_height = height / grid_size as f64;

    let mut grid: Vec<Vec<Vec<usize>>> = vec![vec![vec![]; grid_size]; grid_size];

    // Place nodes in grid
    for (i, node) in graph.nodes.iter().enumerate() {
        let gx = ((node.x - min_x) / cell_width) as usize;
        let gy = ((node.y - min_y) / cell_height) as usize;
        let gx = gx.min(grid_size - 1);
        let gy = gy.min(grid_size - 1);
        grid[gx][gy].push(i);
    }

    let k = (100.0 / (n as f64).sqrt()).max(1.0);

    // Calculate repulsive forces using grid
    for i in 0..n {
        let gx = ((graph.nodes[i].x - min_x) / cell_width) as usize;
        let gy = ((graph.nodes[i].y - min_y) / cell_height) as usize;
        let gx = gx.min(grid_size - 1);
        let gy = gy.min(grid_size - 1);

        // Check nearby cells (3x3 neighborhood)
        for dx in -1..=1 {
            for dy in -1..=1 {
                let nx = (gx as i32 + dx) as usize;
                let ny = (gy as i32 + dy) as usize;

                if nx < grid_size && ny < grid_size {
                    for &j in &grid[nx][ny] {
                        if i < j {
                            let dx = graph.nodes[j].x - graph.nodes[i].x;
                            let dy = graph.nodes[j].y - graph.nodes[i].y;
                            let dist = (dx * dx + dy * dy).sqrt();

                            if dist > 0.01 {
                                let repulsive = k * k / (dist + 0.1);
                                let fx = (dx / dist) * repulsive;
                                let fy = (dy / dist) * repulsive;

                                forces[i].0 -= fx;
                                forces[i].1 -= fy;
                                forces[j].0 += fx;
                                forces[j].1 += fy;
                            }
                        }
                    }
                }
            }
        }
    }

    // Attractive forces along edges
    for &(u, v) in &graph.edges {
        let dx = graph.nodes[v].x - graph.nodes[u].x;
        let dy = graph.nodes[v].y - graph.nodes[u].y;
        let dist = (dx * dx + dy * dy).sqrt();

        if dist > 0.01 {
            let attractive = dist * dist / (k * k);
            let fx = (dx / dist) * attractive;
            let fy = (dy / dist) * attractive;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }
    }

    // Update positions
    for i in 0..n {
        let (fx, fy) = forces[i];
        let dist = (fx * fx + fy * fy).sqrt();

        if dist > 0.0 {
            let displacement = temp.min(dist);
            graph.nodes[i].x += (fx / dist) * displacement;
            graph.nodes[i].y += (fy / dist) * displacement;
        }
    }
}
