use std::io::{self, BufRead};

#[derive(Debug, Clone)]
struct Sample {
    depth: f64,
    properties: Vec<f64>,
}

#[derive(Debug)]
struct Hole {
    id: u32,
    samples: Vec<Sample>,
}

#[derive(Debug, Clone)]
struct Outlier {
    hole_id: u32,
    sample_idx: usize,
    property_name: String,
    confidence: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Parse header
    let first_line = lines.next().unwrap().unwrap();
    let mut parts = first_line.split_whitespace();
    let num_holes: usize = parts.next().unwrap().parse().unwrap();
    let num_properties: usize = parts.next().unwrap().parse().unwrap();

    // Parse property names
    let props_line = lines.next().unwrap().unwrap();
    let property_names: Vec<String> = props_line.split_whitespace()
        .map(|s| s.to_string())
        .collect();

    // Parse holes
    let mut holes = Vec::new();

    for _ in 0..num_holes {
        let hole_line = lines.next().unwrap().unwrap();
        let mut parts = hole_line.split_whitespace();
        let hole_id: u32 = parts.next().unwrap().parse().unwrap();
        let _start_x: f64 = parts.next().unwrap().parse().unwrap();
        let _start_y: f64 = parts.next().unwrap().parse().unwrap();
        let _start_z: f64 = parts.next().unwrap().parse().unwrap();
        let _dir_x: f64 = parts.next().unwrap().parse().unwrap();
        let _dir_y: f64 = parts.next().unwrap().parse().unwrap();
        let _dir_z: f64 = parts.next().unwrap().parse().unwrap();
        let _length: f64 = parts.next().unwrap().parse().unwrap();
        let num_samples: usize = parts.next().unwrap().parse().unwrap();

        let mut samples = Vec::new();
        for _ in 0..num_samples {
            let sample_line = lines.next().unwrap().unwrap();
            let mut sample_parts = sample_line.split_whitespace();
            let depth: f64 = sample_parts.next().unwrap().parse().unwrap();
            let mut properties = Vec::new();
            for _ in 0..num_properties {
                properties.push(sample_parts.next().unwrap().parse().unwrap());
            }
            samples.push(Sample { depth, properties });
        }

        holes.push(Hole { id: hole_id, samples });
    }

    // Find outliers using local neighbor analysis
    let mut outliers = Vec::new();

    for hole in &holes {
        for prop_idx in 0..num_properties {
            for sample_idx in 0..hole.samples.len() {
                let value = hole.samples[sample_idx].properties[prop_idx];

                // Collect neighboring samples within ±2 positions
                let window = 2;
                let start = sample_idx.saturating_sub(window);
                let end = std::cmp::min(sample_idx + window + 1, hole.samples.len());

                let mut neighbors = Vec::new();
                for i in start..end {
                    if i != sample_idx {
                        neighbors.push(hole.samples[i].properties[prop_idx]);
                    }
                }

                if neighbors.is_empty() {
                    continue;
                }

                // Calculate Z-score relative to neighboring samples
                let mean = neighbors.iter().sum::<f64>() / neighbors.len() as f64;
                let variance = neighbors.iter()
                    .map(|&x| (x - mean).powi(2))
                    .sum::<f64>() / neighbors.len() as f64;
                let std_dev = variance.sqrt();

                // Identify outliers where Z-score > 2 (95% confidence level)
                if std_dev > 1e-10 {
                    let z_score = (value - mean).abs() / std_dev;
                    if z_score > 2.0 {
                        outliers.push(Outlier {
                            hole_id: hole.id,
                            sample_idx,
                            property_name: property_names[prop_idx].clone(),
                            confidence: z_score,
                        });
                    }
                }
            }
        }
    }

    // Sort by confidence (Z-score) in descending order
    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap());

    // Output results
    println!("{}", outliers.len());
    for outlier in outliers {
        println!("{} {} {} {:.4}",
                 outlier.hole_id,
                 outlier.sample_idx,
                 outlier.property_name,
                 outlier.confidence);
    }
}
