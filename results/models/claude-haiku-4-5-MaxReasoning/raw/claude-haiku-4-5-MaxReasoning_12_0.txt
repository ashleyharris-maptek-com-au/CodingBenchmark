using System;
using System.Collections.Generic;
using System.Linq;

class Vector3
{
    public double X, Y, Z;
    public Vector3(double x = 0, double y = 0, double z = 0) => (X, Y, Z) = (x, y, z);
    public Vector3 Add(Vector3 v) => new(X + v.X, Y + v.Y, Z + v.Z);
    public Vector3 Sub(Vector3 v) => new(X - v.X, Y - v.Y, Z - v.Z);
    public Vector3 Mul(double s) => new(X * s, Y * s, Z * s);
    public double Dot(Vector3 v) => X * v.X + Y * v.Y + Z * v.Z;
    public double Length() => Math.Sqrt(X * X + Y * Y + Z * Z);
    public Vector3 Normalize() { var len = Length(); return len > 1e-9 ? Mul(1.0 / len) : this; }
    public Vector3 Cross(Vector3 v) => new(Y * v.Z - Z * v.Y, Z * v.X - X * v.Z, X * v.Y - Y * v.X);
}

class Quaternion
{
    public double W, X, Y, Z;
    public Quaternion(double w = 1, double x = 0, double y = 0, double z = 0) => (W, X, Y, Z) = (w, x, y, z);

    public Vector3 Rotate(Vector3 p)
    {
        var qv = new Vector3(X, Y, Z);
        var cross1 = qv.Cross(p);
        var cross2 = qv.Cross(cross1);
        return p.Add(cross1.Mul(2 * W)).Add(cross2.Mul(2));
    }

    public static Quaternion RotationZ(double angle)
    {
        var half = angle / 2;
        return new(Math.Cos(half), 0, 0, Math.Sin(half));
    }
}

struct Placement
{
    public Vector3 Position;
    public Quaternion Rotation;
}

class Program
{
    static List<Vector3> vertices;
    static List<(int, int, int)> faces;
    static double containerW, containerH, containerD;
    static List<Placement> placements;

    static Vector3[] GetRotatedVertices(Quaternion q)
    {
        return vertices.Select(v => q.Rotate(v)).ToArray();
    }

    static (Vector3, Vector3) GetBounds(Vector3[] verts)
    {
        var min = new Vector3(verts[0].X, verts[0].Y, verts[0].Z);
        var max = new Vector3(verts[0].X, verts[0].Y, verts[0].Z);
        foreach(var v in verts)
        {
            min.X = Math.Min(min.X, v.X);
            min.Y = Math.Min(min.Y, v.Y);
            min.Z = Math.Min(min.Z, v.Z);
            max.X = Math.Max(max.X, v.X);
            max.Y = Math.Max(max.Y, v.Y);
            max.Z = Math.Max(max.Z, v.Z);
        }
        return (min, max);
    }

    static bool CheckCollision(Vector3[] verts1, Vector3 pos1, Vector3[] verts2, Vector3 pos2)
    {
        var (min1, max1) = GetBounds(verts1);
        var (min2, max2) = GetBounds(verts2);

        min1 = min1.Add(pos1);
        max1 = max1.Add(pos1);
        min2 = min2.Add(pos2);
        max2 = max2.Add(pos2);

        return !(max1.X < min2.X || min1.X > max2.X ||
                 max1.Y < min2.Y || min1.Y > max2.Y ||
                 max1.Z < min2.Z || min1.Z > max2.Z);
    }

    static bool CanPlace(Placement p, Vector3[] rotVerts)
    {
        var (min, max) = GetBounds(rotVerts);
        var pos = p.Position;

        if (pos.X + min.X < 0 || pos.X + max.X > containerW ||
            pos.Y + min.Y < 0 || pos.Y + max.Y > containerH ||
            pos.Z + min.Z < 0 || pos.Z + max.Z > containerD)
            return false;

        var basVerts = GetRotatedVertices(new Quaternion(1, 0, 0, 0));
        foreach(var placed in placements)
        {
            var pVerts = GetRotatedVertices(placed.Rotation);
            if (CheckCollision(rotVerts, pos, pVerts, placed.Position))
                return false;
        }
        return true;
    }

    static Vector3 FindGoodPosition(Vector3[] rotVerts)
    {
        var (min, max) = GetBounds(rotVerts);
        var w = max.X - min.X;
        var h = max.Y - min.Y;
        var d = max.Z - min.Z;

        var gridSize = Math.Max(5, (int)Math.Sqrt(containerW * containerH * containerD) / 20);
        var stepX = Math.Max(1, containerW / gridSize);
        var stepY = Math.Max(1, containerH / gridSize);
        var stepZ = Math.Max(1, containerD / gridSize);

        for(double z = 0; z <= containerD - d; z += stepZ)
            for(double y = 0; y <= containerH - h; y += stepY)
                for(double x = 0; x <= containerW - w; x += stepX)
                {
                    var p = new Placement { Position = new(x - min.X, y - min.Y, z - min.Z), Rotation = new(1, 0, 0, 0) };
                    if (CanPlace(p, rotVerts))
                        return p.Position;
                }
        return null;
    }

    static void Main()
    {
        var line1 = Console.ReadLine().Split();
        int vCount = int.Parse(line1[0]);
        int fCount = int.Parse(line1[1]);

        vertices = new();
        for(int i = 0; i < vCount; i++)
        {
            var parts = Console.ReadLine().Split();
            vertices.Add(new(double.Parse(parts[0]), double.Parse(parts[1]), double.Parse(parts[2])));
        }

        faces = new();
        for(int i = 0; i < fCount; i++)
        {
            var parts = Console.ReadLine().Split();
            faces.Add((int.Parse(parts[0]), int.Parse(parts[1]), int.Parse(parts[2])));
        }

        var dimParts = Console.ReadLine().Split();
        containerW = double.Parse(dimParts[0]);
        containerH = double.Parse(dimParts[1]);
        containerD = double.Parse(dimParts[2]);

        placements = new();
        var rotations = new[] {
            new Quaternion(1, 0, 0, 0),
            new Quaternion(0.707, 0, 0, 0.707),
            new Quaternion(0.707, 0, 0.707, 0),
            new Quaternion(0.707, 0.707, 0, 0)
        };

        bool placed = true;
        while(placed)
        {
            placed = false;
            foreach(var rot in rotations)
            {
                var rotVerts = GetRotatedVertices(rot);
                var pos = FindGoodPosition(rotVerts);
                if (pos != null)
                {
                    placements.Add(new Placement { Position = pos, Rotation = rot });
                    placed = true;
                    break;
                }
            }
        }

        Console.WriteLine(placements.Count);
        foreach(var p in placements)
            Console.WriteLine($"{p.Position.X} {p.Position.Y} {p.Position.Z} {p.Rotation.W} {p.Rotation.X} {p.Rotation.Y} {p.Rotation.Z}");
    }
}
