using System;
using System.Collections.Generic;
using System.Linq;

class ChinesePostman
{
    static void Main()
    {
        var line = Console.ReadLine().Split();
        int n = int.Parse(line[0]);
        int m = int.Parse(line[1]);

        // Store edges: (u, v, weight)
        var edges = new List<(int, int, int)>();
        var adj = new List<int>[n];
        for (int i = 0; i < n; i++)
            adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);

            edges.Add((u, v, w));
            adj[u].Add(i);
            if (u != v)
                adj[v].Add(i);
        }

        // Find vertices with odd degree
        var oddVerts = new List<int>();
        for (int i = 0; i < n; i++)
        {
            int degree = 0;
            foreach (int edgeIdx in adj[i])
            {
                var (u, v, _) = edges[edgeIdx];
                degree += (u == v) ? 2 : 1;
            }
            if (degree % 2 == 1)
                oddVerts.Add(i);
        }

        // If graph is Eulerian, just find circuit
        if (oddVerts.Count == 0)
        {
            var route = FindEulerianCircuit(n, edges);
            PrintRoute(route);
            return;
        }

        // Compute shortest paths between odd vertices
        var dist = ComputeShortestPaths(n, edges, adj, oddVerts);

        // Find minimum weight perfect matching
        var matching = FindMinimumMatching(dist, oddVerts);

        // Add shortest paths for each matched pair
        for (int i = 0; i < matching.Count; i++)
        {
            var (u, v) = matching[i];
            var path = FindShortestPath(n, edges, adj, u, v);
            for (int j = 0; j < path.Count - 1; j++)
            {
                int a = path[j];
                int b = path[j + 1];
                int w = ComputePathWeight(n, edges, adj, a, b);
                edges.Add((a, b, w));
                int newIdx = edges.Count - 1;
                adj[a].Add(newIdx);
                if (a != b)
                    adj[b].Add(newIdx);
            }
        }

        // Find Eulerian circuit
        var result = FindEulerianCircuit(n, edges);
        PrintRoute(result);
    }

    static int[][] ComputeShortestPaths(int n, List<(int, int, int)> edges, List<int>[] adj, List<int> oddVerts)
    {
        int k = oddVerts.Count;
        var result = new int[k][];

        for (int i = 0; i < k; i++)
        {
            var dist = Dijkstra(n, edges, adj, oddVerts[i]);
            result[i] = new int[k];
            for (int j = 0; j < k; j++)
                result[i][j] = dist[oddVerts[j]];
        }

        return result;
    }

    static int[] Dijkstra(int n, List<(int, int, int)> edges, List<int>[] adj, int start)
    {
        var dist = new int[n];
        for (int i = 0; i < n; i++)
            dist[i] = int.MaxValue / 2;
        dist[start] = 0;

        var visited = new bool[n];

        for (int iter = 0; iter < n; iter++)
        {
            int u = -1;
            for (int i = 0; i < n; i++)
                if (!visited[i] && (u == -1 || dist[i] < dist[u]))
                    u = i;

            if (dist[u] >= int.MaxValue / 2) break;
            visited[u] = true;

            foreach (int edgeIdx in adj[u])
            {
                var (eu, ev, w) = edges[edgeIdx];
                int v = (eu == u) ? ev : eu;
                if (dist[u] + w < dist[v])
                    dist[v] = dist[u] + w;
            }
        }

        return dist;
    }

    static List<(int, int)> FindMinimumMatching(int[][] dist, List<int> oddVerts)
    {
        int k = oddVerts.Count;
        var used = new bool[k];
        var matching = new List<(int, int)>();

        // Greedy matching
        for (int i = 0; i < k; i++)
        {
            if (used[i]) continue;
            int best = -1;
            for (int j = i + 1; j < k; j++)
            {
                if (!used[j] && (best == -1 || dist[i][j] < dist[i][best]))
                    best = j;
            }
            if (best != -1)
            {
                used[i] = true;
                used[best] = true;
                matching.Add((oddVerts[i], oddVerts[best]));
            }
        }

        return matching;
    }

    static List<int> FindShortestPath(int n, List<(int, int, int)> edges, List<int>[] adj, int start, int end)
    {
        var dist = new int[n];
        var parent = new int[n];
        for (int i = 0; i < n; i++)
        {
            dist[i] = int.MaxValue / 2;
            parent[i] = -1;
        }
        dist[start] = 0;

        var visited = new bool[n];

        for (int iter = 0; iter < n; iter++)
        {
            int u = -1;
            for (int i = 0; i < n; i++)
                if (!visited[i] && (u == -1 || dist[i] < dist[u]))
                    u = i;

            if (dist[u] >= int.MaxValue / 2) break;
            visited[u] = true;

            foreach (int edgeIdx in adj[u])
            {
                var (eu, ev, w) = edges[edgeIdx];
                int v = (eu == u) ? ev : eu;
                if (dist[u] + w < dist[v])
                {
                    dist[v] = dist[u] + w;
                    parent[v] = u;
                }
            }
        }

        var path = new List<int>();
        int curr = end;
        while (curr != -1)
        {
            path.Add(curr);
            curr = parent[curr];
        }
        path.Reverse();
        return path;
    }

    static int ComputePathWeight(int n, List<(int, int, int)> edges, List<int>[] adj, int start, int end)
    {
        var dist = new int[n];
        for (int i = 0; i < n; i++)
            dist[i] = int.MaxValue / 2;
        dist[start] = 0;

        var visited = new bool[n];

        for (int iter = 0; iter < n; iter++)
        {
            int u = -1;
            for (int i = 0; i < n; i++)
                if (!visited[i] && (u == -1 || dist[i] < dist[u]))
                    u = i;

            if (dist[u] >= int.MaxValue / 2) break;
            visited[u] = true;

            foreach (int edgeIdx in adj[u])
            {
                var (eu, ev, w) = edges[edgeIdx];
                int v = (eu == u) ? ev : eu;
                if (dist[u] + w < dist[v])
                    dist[v] = dist[u] + w;
            }
        }

        return dist[end];
    }

    static List<int> FindEulerianCircuit(int n, List<(int, int, int)> edges)
    {
        var edgeList = new List<int>[n];
        for (int i = 0; i < n; i++)
            edgeList[i] = new List<int>();

        for (int i = 0; i < edges.Count; i++)
        {
            var (u, v, _) = edges[i];
            edgeList[u].Add(i);
            if (u != v)
                edgeList[v].Add(i);
        }

        var used = new bool[edges.Count];
        var result = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            bool found = false;

            for (int i = edgeList[u].Count - 1; i >= 0; i--)
            {
                int edgeIdx = edgeList[u][i];
                if (!used[edgeIdx])
                {
                    used[edgeIdx] = true;
                    var (eu, ev, _) = edges[edgeIdx];
                    int v = (eu == u) ? ev : eu;
                    stack.Push(v);
                    found = true;
                    break;
                }
            }

            if (!found)
                result.Add(stack.Pop());
        }

        result.Reverse();
        return result;
    }

    static void PrintRoute(List<int> route)
    {
        Console.WriteLine(route.Count);
        Console.WriteLine(string.Join(" ", route));
    }
}
