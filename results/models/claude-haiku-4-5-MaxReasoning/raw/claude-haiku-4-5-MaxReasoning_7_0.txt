#include <iostream>

using namespace std;

const float EPS = 1e-6f;

struct Vec3 {
    float x, y, z;

    Vec3() : x(0), y(0), z(0) {}
    Vec3(float x, float y, float z) : x(x), y(y), z(z) {}

    Vec3 operator+(const Vec3& v) const { return Vec3(x + v.x, y + v.y, z + v.z); }
    Vec3 operator-(const Vec3& v) const { return Vec3(x - v.x, y - v.y, z - v.z); }
    Vec3 operator*(float t) const { return Vec3(x * t, y * t, z * t); }
    float dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x); }
    float length() const { return sqrt(x * x + y * y + z * z); }
    Vec3 normalized() const { float len = length(); return len > EPS ? Vec3(x/len, y/len, z/len) : Vec3(0, 0, 0); }
};

struct Face {
    int a, b, c;
    Face() {}
    Face(int a, int b, int c) : a(a), b(b), c(c) {}
};

struct Mesh {
    vector<Vec3> vertices;
    vector<Face> faces;
};

float signedDistanceToPlane(Vec3 point, Vec3 planePoint, Vec3 normal) {
    return (point - planePoint).dot(normal);
}

bool isPointInMesh(Vec3 p, const Mesh& mesh) {
    int count = 0;
    for (const auto& face : mesh.faces) {
        Vec3 v0 = mesh.vertices[face.a];
        Vec3 v1 = mesh.vertices[face.b];
        Vec3 v2 = mesh.vertices[face.c];

        Vec3 e1 = v1 - v0;
        Vec3 e2 = v2 - v0;
        Vec3 normal = e1.cross(e2).normalized();

        float d = signedDistanceToPlane(p, v0, normal);
        if (d < -EPS) count++;
    }
    return (count % 2) == 1;
}

bool rayIntersectTriangle(Vec3 rayOrigin, Vec3 rayDir, Vec3 v0, Vec3 v1, Vec3 v2, float& t) {
    const float EPSILON = 1e-6f;
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 h = rayDir.cross(edge2);
    float a = edge1.dot(h);

    if (abs(a) < EPSILON) return false;

    float f = 1.0f / a;
    Vec3 s = rayOrigin - v0;
    float u = f * s.dot(h);

    if (u < 0.0f || u > 1.0f) return false;

    Vec3 q = s.cross(edge1);
    float v = f * rayDir.dot(q);

    if (v < 0.0f || u + v > 1.0f) return false;

    t = f * edge2.dot(q);
    return t > EPSILON;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    cin >> VA >> FA >> VB >> FB;

    Mesh meshA, meshB;
    meshA.vertices.resize(VA);
    meshA.faces.resize(FA);
    meshB.vertices.resize(VB);
    meshB.faces.resize(FB);

    for (int i = 0; i < VA; i++) {
        cin >> meshA.vertices[i].x >> meshA.vertices[i].y >> meshA.vertices[i].z;
    }

    for (int i = 0; i < FA; i++) {
        cin >> meshA.faces[i].a >> meshA.faces[i].b >> meshA.faces[i].c;
    }

    for (int i = 0; i < VB; i++) {
        cin >> meshB.vertices[i].x >> meshB.vertices[i].y >> meshB.vertices[i].z;
    }

    for (int i = 0; i < FB; i++) {
        cin >> meshB.faces[i].a >> meshB.faces[i].b >> meshB.faces[i].c;
    }

    Mesh result;

    // Add vertices and classify faces from mesh A
    vector<int> vertexMapA;
    vector<int> faceIncludeA;

    for (int i = 0; i < VA; i++) {
        result.vertices.push_back(meshA.vertices[i]);
        vertexMapA.push_back(result.vertices.size() - 1);
    }

    for (int i = 0; i < FA; i++) {
        Vec3 v0 = meshA.vertices[meshA.faces[i].a];
        Vec3 v1 = meshA.vertices[meshA.faces[i].b];
        Vec3 v2 = meshA.vertices[meshA.faces[i].c];
        Vec3 centroid = (v0 + v1 + v2) * (1.0f/3.0f);

        bool inside = isPointInMesh(centroid, meshB);
        faceIncludeA.push_back(!inside);

        if (!inside) {
            result.faces.push_back(Face(
                vertexMapA[meshA.faces[i].a],
                vertexMapA[meshA.faces[i].b],
                vertexMapA[meshA.faces[i].c]
            ));
        }
    }

    // Add vertices and classify faces from mesh B
    vector<int> vertexMapB;
    vector<int> faceIncludeB;

    for (int i = 0; i < VB; i++) {
        result.vertices.push_back(meshB.vertices[i]);
        vertexMapB.push_back(result.vertices.size() - 1);
    }

    for (int i = 0; i < FB; i++) {
        Vec3 v0 = meshB.vertices[meshB.faces[i].a];
        Vec3 v1 = meshB.vertices[meshB.faces[i].b];
        Vec3 v2 = meshB.vertices[meshB.faces[i].c];
        Vec3 centroid = (v0 + v1 + v2) * (1.0f/3.0f);

        bool inside = isPointInMesh(centroid, meshA);
        faceIncludeB.push_back(!inside);

        if (!inside) {
            result.faces.push_back(Face(
                vertexMapB[meshB.faces[i].a],
                vertexMapB[meshB.faces[i].b],
                vertexMapB[meshB.faces[i].c]
            ));
        }
    }

    cout << result.vertices.size() << " " << result.faces.size() << "\n";

    for (const auto& v : result.vertices) {
        cout << fixed;
        cout.precision(6);
        cout << v.x << " " << v.y << " " << v.z << "\n";
    }

    for (const auto& f : result.faces) {
        cout << f.a << " " << f.b << " " << f.c << "\n";
    }

    return 0;
}
