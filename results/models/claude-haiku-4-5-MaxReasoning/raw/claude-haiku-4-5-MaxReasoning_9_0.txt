using System;
using System.Collections.Generic;
using System.Linq;

class Program
{
    static void Main()
    {
        string[] line1 = Console.ReadLine().Split();
        int n = int.Parse(line1[0]);
        int stockLength = int.Parse(line1[1]);

        int[] cuts = Console.ReadLine().Split().Select(int.Parse).ToArray();

        // Create list of (cut_length, original_index)
        var indexedCuts = cuts.Select((len, idx) => (length: len, index: idx)).ToList();

        // Sort by length descending (First Fit Decreasing heuristic)
        indexedCuts.Sort((a, b) => b.length.CompareTo(a.length));

        // List of stocks, each stock stores the indices of cuts assigned to it
        var stocks = new List<List<int>>();
        var stockRemaining = new List<int>();

        // Assign each cut to a stock using First Fit
        foreach (var (length, index) in indexedCuts)
        {
            // Find first stock with enough space
            int stockIdx = -1;
            for (int i = 0; i < stocks.Count; i++)
            {
                if (stockRemaining[i] >= length)
                {
                    stockIdx = i;
                    break;
                }
            }

            // If no stock found, create new one
            if (stockIdx == -1)
            {
                stocks.Add(new List<int>());
                stockRemaining.Add(stockLength);
                stockIdx = stocks.Count - 1;
            }

            // Assign cut to stock
            stocks[stockIdx].Add(index);
            stockRemaining[stockIdx] -= length;
        }

        // Output
        Console.WriteLine(stocks.Count);
        foreach (var stock in stocks)
        {
            stock.Sort();
            Console.WriteLine(string.Join(" ", stock));
        }
    }
}
