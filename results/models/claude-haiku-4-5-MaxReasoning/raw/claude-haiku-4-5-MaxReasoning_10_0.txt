#include <iostream>
#include <vector>
#include <algorithm>

struct Rect {
    int w, h, id;
};

int main() {
    int n, board_w, board_h;
    std::cin >> n >> board_w >> board_h;

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; i++) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    // Sort by area descending (better packing heuristic)
    std::sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        return (long long)a.w * a.h > (long long)b.w * b.h;
    });

    std::vector<int> placement_board(n);
    std::vector<int> placement_x(n);
    std::vector<int> placement_y(n);
    std::vector<int> placement_rotated(n);

    // Each board has a height map tracking the height at each x coordinate
    std::vector<std::vector<int>> boards;

    for (const auto& rect : rects) {
        int best_board = -1;
        int best_x = 0;
        int best_y = board_h + 1;
        int best_rotated = 0;

        // Try to place in existing boards
        for (int b = 0; b < (int)boards.size(); b++) {
            // Try both orientations (normal and rotated 90 degrees)
            for (int rot = 0; rot < 2; rot++) {
                int w = (rot == 0) ? rect.w : rect.h;
                int h = (rot == 0) ? rect.h : rect.w;

                if (w > board_w || h > board_h) continue;

                // Find the position with the lowest y value that fits
                for (int x = 0; x <= board_w - w; x++) {
                    int y = 0;
                    // Get the maximum height in the range [x, x+w)
                    for (int dx = 0; dx < w; dx++) {
                        y = std::max(y, boards[b][x + dx]);
                    }

                    // Check if rectangle fits and is better than current best
                    if (y + h <= board_h && y < best_y) {
                        best_board = b;
                        best_x = x;
                        best_y = y;
                        best_rotated = rot;
                    }
                }
            }
        }

        // Create new board if rectangle couldn't be placed in existing ones
        if (best_board == -1) {
            best_board = boards.size();
            boards.push_back(std::vector<int>(board_w, 0));
            best_x = 0;
            best_y = 0;
            best_rotated = 0;
        }

        // Record the placement
        placement_board[rect.id] = best_board;
        placement_x[rect.id] = best_x;
        placement_y[rect.id] = best_y;
        placement_rotated[rect.id] = best_rotated;

        // Update the height map for the selected board
        int w = (best_rotated == 0) ? rect.w : rect.h;
        int h = (best_rotated == 0) ? rect.h : rect.w;

        for (int x = best_x; x < best_x + w; x++) {
            boards[best_board][x] = std::max(boards[best_board][x], best_y + h);
        }
    }

    // Output results
    std::cout << boards.size() << "\n";
    for (int i = 0; i < n; i++) {
        std::cout << placement_board[i] << " " << placement_x[i] << " "
                  << placement_y[i] << " " << placement_rotated[i] << "\n";
    }

    return 0;
}
