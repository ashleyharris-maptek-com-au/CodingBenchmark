using System;
using System.Collections.Generic;
using System.Linq;

class OrbitalTSP
{
    struct Orbit
    {
        public double X, Y, Z, Vx, Vy, Vz;
        public double R => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double V => Math.Sqrt(Vx * Vx + Vy * Vy + Vz * Vz);
    }

    static void Main()
    {
        string[] lines = Console.In.ReadToEnd().Split(new[] { '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
        var firstLine = lines[0].Trim().Split();
        int n = int.Parse(firstLine[0]);
        double mu = double.Parse(firstLine[1]);
        var orbits = new Orbit[n + 1];
        var startLine = lines[1].Trim().Split();
        orbits[0] = ParseOrbit(startLine);
        for (int i = 0; i < n; i++) orbits[i + 1] = ParseOrbit(lines[2 + i].Trim().Split());
        double[][] costs = BuildCostMatrix(orbits, n, mu);
        int[] tour = SolveTSP(costs, n);
        Console.WriteLine(string.Join(" ", tour.Skip(1).Take(n).Select(x => x - 1)));
    }

    static Orbit ParseOrbit(string[] parts)
    {
        return new Orbit { X = double.Parse(parts[0]), Y = double.Parse(parts[1]), Z = double.Parse(parts[2]), Vx = double.Parse(parts[3]), Vy = double.Parse(parts[4]), Vz = double.Parse(parts[5]) };
    }

    static double[][] BuildCostMatrix(Orbit[] orbits, int n, double mu)
    {
        double[][] costs = new double[n + 1][];
        for (int i = 0; i <= n; i++) costs[i] = new double[n + 1];
        for (int j = 1; j <= n; j++) costs[0][j] = CalculateDeltaV(orbits[0], orbits[j], mu);
        for (int i = 1; i <= n; i++)
            for (int j = 1; j <= n; j++)
                costs[i][j] = (i == j) ? 0 : CalculateDeltaV(orbits[i], orbits[j], mu);
        return costs;
    }

    static double CalculateDeltaV(Orbit from, Orbit to, double mu)
    {
        double r1 = from.R, r2 = to.R;
        if (r1 < 100 || r2 < 100) return double.MaxValue;
        double v1 = Math.Sqrt(mu / r1), v2 = Math.Sqrt(mu / r2);
        double a_trans = (r1 + r2) / 2;
        double v_trans1 = Math.Sqrt(mu * (2.0 / r1 - 1.0 / a_trans));
        double v_trans2 = Math.Sqrt(mu * (2.0 / r2 - 1.0 / a_trans));
        double deltaV1 = Math.Abs(v_trans1 - v1), deltaV2 = Math.Abs(v2 - v_trans2);
        double dx = to.X - from.X, dy = to.Y - from.Y, dz = to.Z - from.Z;
        double posCost = Math.Sqrt(dx * dx + dy * dy + dz * dz) * 0.001;
        return deltaV1 + deltaV2 + posCost;
    }

    static int[] SolveTSP(double[][] costs, int n)
    {
        if (n <= 10) return SolveSmall(costs, n);
        else if (n <= 25) return SolveMedium(costs, n);
        else return SolveLarge(costs, n);
    }

    static int[] SolveSmall(double[][] costs, int n)
    {
        int[] tour = NearestNeighbor(costs, n);
        TwoOptImprove(tour, costs, 5000);
        return tour;
    }

    static int[] SolveMedium(double[][] costs, int n)
    {
        int[] tour = NearestNeighbor(costs, n);
        TwoOptImprove(tour, costs, 1000);
        return tour;
    }

    static int[] SolveLarge(double[][] costs, int n)
    {
        int[] bestTour = NearestNeighbor(costs, n);
        double bestCost = CalculateTourCost(bestTour, costs);
        int starts = Math.Min(10, n);
        for (int start = 1; start < starts; start++)
        {
            int[] tour = NearestNeighborFrom(costs, n, start);
            double cost = CalculateTourCost(tour, costs);
            if (cost < bestCost) { bestTour = tour; bestCost = cost; }
        }
        TwoOptImprove(bestTour, costs, 100);
        return bestTour;
    }

    static int[] NearestNeighbor(double[][] costs, int n) => NearestNeighborFrom(costs, n, 0);

    static int[] NearestNeighborFrom(double[][] costs, int n, int start)
    {
        var tour = new int[n + 1];
        var visited = new bool[n + 1];
        int current = start;
        tour[0] = current;
        visited[current] = true;
        for (int i = 1; i <= n; i++)
        {
            int best = -1;
            double bestCost = double.MaxValue;
            for (int j = 0; j <= n; j++)
                if (!visited[j] && costs[current][j] < bestCost) { best = j; bestCost = costs[current][j]; }
            current = best;
            tour[i] = current;
            visited[current] = true;
        }
        return tour;
    }

    static void TwoOptImprove(int[] tour, double[][] costs, int iterations)
    {
        int n = tour.Length - 1;
        bool improved = true;
        int iter = 0;
        while (improved && iter < iterations)
        {
            improved = false;
            iter++;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 2; j < n; j++)
                {
                    int a = tour[i], b = tour[i + 1], c = tour[j], d = tour[(j + 1) % (n + 1)];
                    double oldCost = costs[a][b] + costs[c][d];
                    double newCost = costs[a][c] + costs[b][d];
                    if (newCost < oldCost - 1e-9)
                    {
                        Array.Reverse(tour, i + 1, j - i);
                        improved = true;
                    }
                }
            }
        }
    }

    static double CalculateTourCost(int[] tour, double[][] costs)
    {
        double cost = 0;
        for (int i = 0; i < tour.Length - 1; i++)
            cost += costs[tour[i]][tour[i + 1]];
        return cost;
    }
}
