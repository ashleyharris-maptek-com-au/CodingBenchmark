use std::io::{self, BufRead};
use std::f64::consts::PI;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

impl Point {
    fn new(x: f64, y: f64) -> Self {
        Point { x, y }
    }

    fn rotate(self, angle: f64) -> Point {
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        Point {
            x: self.x * cos_a - self.y * sin_a,
            y: self.x * sin_a + self.y * cos_a,
        }
    }

    fn translate(self, dx: f64, dy: f64) -> Point {
        Point {
            x: self.x + dx,
            y: self.y + dy,
        }
    }
}

#[derive(Clone)]
struct Polygon {
    vertices: Vec<Point>,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        Polygon { vertices }
    }

    fn rotate(&self, angle: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.rotate(angle)).collect(),
        }
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        Polygon {
            vertices: self.vertices.iter().map(|p| p.translate(dx, dy)).collect(),
        }
    }

    fn bounding_box(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for p in &self.vertices {
            min_x = min_x.min(p.x);
            max_x = max_x.max(p.x);
            min_y = min_y.min(p.y);
            max_y = max_y.max(p.y);
        }

        (min_x, min_y, max_x, max_y)
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
}

fn point_in_polygon(point: &Point, polygon: &Polygon) -> bool {
    let mut inside = false;
    let n = polygon.vertices.len();

    for i in 0..n {
        let j = (i + 1) % n;
        let xi = polygon.vertices[i].x;
        let yi = polygon.vertices[i].y;
        let xj = polygon.vertices[j].x;
        let yj = polygon.vertices[j].y;

        if ((yi > point.y) != (yj > point.y)) &&
           (point.x < (xj - xi) * (point.y - yi) / (yj - yi) + xi) {
            inside = !inside;
        }
    }

    inside
}

fn segments_intersect(p1: &Point, p2: &Point, p3: &Point, p4: &Point) -> bool {
    fn ccw(a: &Point, b: &Point, c: &Point) -> bool {
        (c.y - a.y) * (b.x - a.x) > (b.y - a.y) * (c.x - a.x)
    }

    ccw(p1, p3, p4) != ccw(p2, p3, p4) && ccw(p1, p2, p3) != ccw(p1, p2, p4)
}

fn polygons_overlap(p1: &Polygon, p2: &Polygon) -> bool {
    let (min1_x, min1_y, max1_x, max1_y) = p1.bounding_box();
    let (min2_x, min2_y, max2_x, max2_y) = p2.bounding_box();

    if max1_x < min2_x || max2_x < min1_x || max1_y < min2_y || max2_y < min1_y {
        return false;
    }

    for v in &p1.vertices {
        if point_in_polygon(v, p2) {
            return true;
        }
    }

    for v in &p2.vertices {
        if point_in_polygon(v, p1) {
            return true;
        }
    }

    for i in 0..p1.vertices.len() {
        for j in 0..p2.vertices.len() {
            let i_next = (i + 1) % p1.vertices.len();
            let j_next = (j + 1) % p2.vertices.len();
            if segments_intersect(&p1.vertices[i], &p1.vertices[i_next],
                                 &p2.vertices[j], &p2.vertices[j_next]) {
                return true;
            }
        }
    }

    false
}

fn piece_fits_in_stock(piece: &Polygon, stock: &Polygon) -> bool {
    let eps = 1e-6;

    for v in &piece.vertices {
        let mut inside = false;
        let n = stock.vertices.len();

        for i in 0..n {
            let j = (i + 1) % n;
            let xi = stock.vertices[i].x;
            let yi = stock.vertices[i].y;
            let xj = stock.vertices[j].x;
            let yj = stock.vertices[j].y;

            if ((yi > v.y) != (yj > v.y)) &&
               (v.x < (xj - xi) * (v.y - yi) / (yj - yi) + xi) {
                inside = !inside;
            }
        }

        if !inside {
            let mut on_edge = false;
            for i in 0..n {
                let j = (i + 1) % n;
                let sx1 = stock.vertices[i].x;
                let sy1 = stock.vertices[i].y;
                let sx2 = stock.vertices[j].x;
                let sy2 = stock.vertices[j].y;

                let dx = sx2 - sx1;
                let dy = sy2 - sy1;
                let len_sq = dx * dx + dy * dy;

                if len_sq > eps * eps {
                    let t = ((v.x - sx1) * dx + (v.y - sy1) * dy) / len_sq;
                    if t >= -eps && t <= 1.0 + eps {
                        let closest_x = sx1 + t * dx;
                        let closest_y = sy1 + t * dy;
                        let dist_sq = (v.x - closest_x) * (v.x - closest_x) +
                                     (v.y - closest_y) * (v.y - closest_y);
                        if dist_sq < eps * eps {
                            on_edge = true;
                            break;
                        }
                    }
                }
            }

            if !on_edge {
                return false;
            }
        }
    }
    true
}

struct Placement {
    stock_index: usize,
    x: f64,
    y: f64,
    rotation: f64,
}

fn find_placement(piece: &Polygon, stocks: &[Vec<Polygon>], stock_polygon: &Polygon) -> Option<Placement> {
    let num_rotations = match stocks.len() {
        0..=10 => 12,
        11..=100 => 8,
        _ => 4,
    };

    for (stock_idx, placed_pieces) in stocks.iter().enumerate() {
        let mut candidates = vec![(0.0, 0.0)];

        for placed in placed_pieces {
            let (_, _, max_x, max_y) = placed.bounding_box();
            candidates.push((max_x + 0.05, 0.0));
            candidates.push((0.0, max_y + 0.05));
        }

        for rot_idx in 0..num_rotations {
            let rotation = (rot_idx as f64) * 2.0 * PI / (num_rotations as f64);
            let rotated_piece = piece.rotate(rotation);
            let (min_x, min_y, _, _) = rotated_piece.bounding_box();

            for &(base_x, base_y) in &candidates {
                let tx = base_x - min_x;
                let ty = base_y - min_y;

                let test_piece = rotated_piece.translate(tx, ty);

                if !piece_fits_in_stock(&test_piece, stock_polygon) {
                    continue;
                }

                let mut overlaps = false;
                for existing in placed_pieces {
                    if polygons_overlap(&test_piece, existing) {
                        overlaps = true;
                        break;
                    }
                }

                if !overlaps {
                    return Some(Placement {
                        stock_index: stock_idx,
                        x: tx,
                        y: ty,
                        rotation,
                    });
                }
            }
        }
    }

    for rot_idx in 0..num_rotations {
        let rotation = (rot_idx as f64) * 2.0 * PI / (num_rotations as f64);
        let rotated_piece = piece.rotate(rotation);

        let (min_x, min_y, _, _) = rotated_piece.bounding_box();
        let tx = -min_x;
        let ty = -min_y;

        let test_piece = rotated_piece.translate(tx, ty);

        if piece_fits_in_stock(&test_piece, stock_polygon) {
            return Some(Placement {
                stock_index: stocks.len(),
                x: tx,
                y: ty,
                rotation,
            });
        }
    }

    None
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let first_line = lines.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let sv: usize = parts[0].parse().unwrap();
    let np: usize = parts[1].parse().unwrap();

    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        stock_vertices.push(Point::new(x, y));
    }
    let stock_polygon = Polygon::new(stock_vertices);

    let mut pieces: Vec<(Polygon, usize)> = Vec::new();
    for i in 0..np {
        let line = lines.next().unwrap().unwrap();
        let pv: usize = line.parse().unwrap();

        let mut piece_vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let parts: Vec<&str> = line.split_whitespace().collect();
            let x: f64 = parts[0].parse().unwrap();
            let y: f64 = parts[1].parse().unwrap();
            piece_vertices.push(Point::new(x, y));
        }
        pieces.push((Polygon::new(piece_vertices), i));
    }

    pieces.sort_by(|a, b| {
        b.0.area().partial_cmp(&a.0.area()).unwrap()
    });

    let mut stocks: Vec<Vec<Polygon>> = Vec::new();
    let mut placements = vec![(0, 0.0, 0.0, 0.0); np];

    for (piece, original_idx) in pieces {
        if let Some(placement) = find_placement(&piece, &stocks, &stock_polygon) {
            let rotated = piece.rotate(placement.rotation);
            let placed = rotated.translate(placement.x, placement.y);

            if placement.stock_index >= stocks.len() {
                stocks.push(Vec::new());
            }
            stocks[placement.stock_index].push(placed);
            placements[original_idx] = (placement.stock_index, placement.x, placement.y, placement.rotation);
        }
    }

    println!("{}", stocks.len());
    for (stock_idx, x, y, rotation) in placements {
        println!("{} {} {} {}", stock_idx, x, y, rotation);
    }
}
