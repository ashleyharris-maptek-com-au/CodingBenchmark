use std::io::{self, BufRead};

#[derive(Debug, Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}

#[derive(Debug, Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Debug, Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

#[derive(Debug, Clone)]
struct Tetrahedron {
    pos: Vec3,
    quat: Quat,
    scale: f64,
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    let nv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();

    let mut polygon: Vec<Vec2> = Vec::new();
    for _ in 0..nv {
        let line = lines.next().unwrap().unwrap();
        let parts: Vec<&str> = line.split_whitespace().collect();
        let x: f64 = parts[0].parse().unwrap();
        let y: f64 = parts[1].parse().unwrap();
        polygon.push(Vec2 { x, y });
    }

    let sun_line = lines.next().unwrap().unwrap();
    let sun_parts: Vec<&str> = sun_line.split_whitespace().collect();
    let mut sun = Vec3 {
        x: sun_parts[0].parse().unwrap(),
        y: sun_parts[1].parse().unwrap(),
        z: sun_parts[2].parse().unwrap(),
    };

    let sun_len = (sun.x * sun.x + sun.y * sun.y + sun.z * sun.z).sqrt();
    sun.x /= sun_len;
    sun.y /= sun_len;
    sun.z /= sun_len;

    let tetrahedrons = solve_shadow_covering(&polygon, &sun);

    println!("{}", tetrahedrons.len());
    for tet in tetrahedrons {
        println!("{} {} {} {} {} {} {} {}",
            tet.pos.x, tet.pos.y, tet.pos.z,
            tet.quat.w, tet.quat.x, tet.quat.y, tet.quat.z,
            tet.scale);
    }
}

fn solve_shadow_covering(polygon: &[Vec2], sun: &Vec3) -> Vec<Tetrahedron> {
    let mut tetrahedrons = Vec::new();

    let (min_x, min_y, max_x, max_y) = compute_bounds(polygon);
    let center_x = (min_x + max_x) / 2.0;
    let center_y = (min_y + max_y) / 2.0;
    let width = max_x - min_x;
    let height = max_y - min_y;
    let max_dim = width.max(height);
    let area = compute_polygon_area(polygon);
    let vertex_count = polygon.len();

    if vertex_count <= 4 {
        let scale = (max_dim * 0.6).min(10.0);
        let tet = Tetrahedron {
            pos: Vec3 {
                x: center_x,
                y: center_y,
                z: scale * 0.5,
            },
            quat: Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 },
            scale,
        };
        tetrahedrons.push(tet);
    } else if area < 50.0 || vertex_count <= 8 {
        let scale = (max_dim * 0.5).min(10.0);
        let tet = Tetrahedron {
            pos: Vec3 {
                x: center_x,
                y: center_y,
                z: scale * 0.5,
            },
            quat: Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 },
            scale,
        };
        tetrahedrons.push(tet);
    } else {
        let grid_density = ((area / 30.0).sqrt() as usize).max(2).min(10);
        let step_x = width / grid_density as f64;
        let step_y = height / grid_density as f64;
        let tetra_scale = ((step_x + step_y) / 4.0).min(10.0);

        for i in 0..grid_density {
            for j in 0..grid_density {
                let x = min_x + (i as f64 + 0.5) * step_x;
                let y = min_y + (j as f64 + 0.5) * step_y;

                if point_in_polygon(&Vec2 { x, y }, polygon) {
                    let tet = Tetrahedron {
                        pos: Vec3 {
                            x,
                            y,
                            z: tetra_scale * 0.5,
                        },
                        quat: Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 },
                        scale: tetra_scale,
                    };
                    tetrahedrons.push(tet);
                }
            }
        }

        if tetrahedrons.is_empty() {
            let scale = (max_dim * 0.5).min(10.0);
            let tet = Tetrahedron {
                pos: Vec3 {
                    x: center_x,
                    y: center_y,
                    z: scale * 0.5,
                },
                quat: Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 },
                scale,
            };
            tetrahedrons.push(tet);
        }
    }

    if tetrahedrons.is_empty() {
        let scale = (max_dim * 0.5).min(10.0);
        let tet = Tetrahedron {
            pos: Vec3 {
                x: center_x,
                y: center_y,
                z: scale * 0.5,
            },
            quat: Quat { w: 1.0, x: 0.0, y: 0.0, z: 0.0 },
            scale,
        };
        tetrahedrons.push(tet);
    }

    tetrahedrons
}

fn compute_bounds(polygon: &[Vec2]) -> (f64, f64, f64, f64) {
    let min_x = polygon.iter().map(|p| p.x).fold(f64::INFINITY, f64::min);
    let max_x = polygon.iter().map(|p| p.x).fold(f64::NEG_INFINITY, f64::max);
    let min_y = polygon.iter().map(|p| p.y).fold(f64::INFINITY, f64::min);
    let max_y = polygon.iter().map(|p| p.y).fold(f64::NEG_INFINITY, f64::max);
    (min_x, min_y, max_x, max_y)
}

fn compute_polygon_area(polygon: &[Vec2]) -> f64 {
    let mut area = 0.0;
    for i in 0..polygon.len() {
        let j = (i + 1) % polygon.len();
        area += polygon[i].x * polygon[j].y;
        area -= polygon[j].x * polygon[i].y;
    }
    (area / 2.0).abs()
}

fn point_in_polygon(p: &Vec2, polygon: &[Vec2]) -> bool {
    let mut inside = false;
    let mut j = polygon.len() - 1;

    for i in 0..polygon.len() {
        let xi = polygon[i].x;
        let yi = polygon[i].y;
        let xj = polygon[j].x;
        let yj = polygon[j].y;

        let intersect = ((yi > p.y) != (yj > p.y)) &&
                       (p.x < (xj - xi) * (p.y - yi) / (yj - yi) + xi);
        if intersect {
            inside = !inside;
        }
        j = i;
    }

    inside
}
