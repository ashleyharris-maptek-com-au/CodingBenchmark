#include <iostream>
#include <vector>
#include <cmath>
#include <algorithm>
#include <numeric>

using namespace std;

struct Point {
    double x, y;
};

inline double distance(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return sqrt(dx*dx + dy*dy);
}

// Nearest neighbor heuristic - O(n²)
vector<int> nearestNeighbor(const vector<Point>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);
    tour.reserve(n);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour.back();
        int nearest = -1;
        double minDist = 1e18;

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                double d = distance(cities[last], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// 2-opt local search improvement
void twoOpt(vector<int>& tour, const vector<Point>& cities, int maxIterations) {
    int n = tour.size();
    bool improved = true;
    int iterations = 0;

    while (improved && iterations < maxIterations) {
        improved = false;
        iterations++;

        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 2; j < n; j++) {
                double d1 = distance(cities[tour[i]], cities[tour[i+1]]);
                double d2 = distance(cities[tour[j]], cities[tour[(j+1)%n]]);
                double d3 = distance(cities[tour[i]], cities[tour[j]]);
                double d4 = distance(cities[tour[i+1]], cities[tour[(j+1)%n]]);

                if (d3 + d4 < d1 + d2 - 1e-9) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
    }
}

// Space-filling curve (Z-order Morton code) - O(n log n)
vector<int> spaceFilling(const vector<Point>& cities) {
    int n = cities.size();

    // Find bounding box
    double minX = cities[0].x, maxX = cities[0].x;
    double minY = cities[0].y, maxY = cities[0].y;

    for (const auto& p : cities) {
        if (p.x < minX) minX = p.x;
        if (p.x > maxX) maxX = p.x;
        if (p.y < minY) minY = p.y;
        if (p.y > maxY) maxY = p.y;
    }

    double rangeX = maxX - minX;
    double rangeY = maxY - minY;
    if (rangeX < 1e-9) rangeX = 1e-9;
    if (rangeY < 1e-9) rangeY = 1e-9;

    // Compute Morton codes
    vector<pair<long long, int>> coded;
    coded.reserve(n);

    for (int i = 0; i < n; i++) {
        double nx = (cities[i].x - minX) / rangeX;
        double ny = (cities[i].y - minY) / rangeY;

        int ix = (int)(nx * 65535.0);
        int iy = (int)(ny * 65535.0);

        // Interleave bits to create Morton code
        long long code = 0;
        for (int bit = 0; bit < 16; bit++) {
            code |= ((long long)((ix >> bit) & 1)) << (2*bit);
            code |= ((long long)((iy >> bit) & 1)) << (2*bit + 1);
        }

        coded.push_back({code, i});
    }

    // Sort by Morton code
    sort(coded.begin(), coded.end());

    // Build tour
    vector<int> tour(n);
    for (int i = 0; i < n; i++) {
        tour[i] = coded[i].second;
    }

    return tour;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    cin >> n;

    vector<Point> cities(n);
    cities.reserve(n);

    for (int i = 0; i < n; i++) {
        cin >> cities[i].x >> cities[i].y;
    }

    vector<int> tour;

    // Adaptive algorithm selection based on problem size
    if (n <= 40) {
        // Trivial: Nearest neighbor + aggressive 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(tour, cities, 100);
    } else if (n <= 1000) {
        // Medium: Nearest neighbor + moderate 2-opt
        tour = nearestNeighbor(cities);
        twoOpt(tour, cities, 10);
    } else if (n <= 100000) {
        // Large: Nearest neighbor only
        tour = nearestNeighbor(cities);
    } else {
        // Extreme/Epic: Space-filling curve (O(n log n))
        tour = spaceFilling(cities);
    }

    // Output tour
    for (int i = 0; i < n; i++) {
        if (i > 0) cout << " ";
        cout << tour[i];
    }
    cout << "\n";

    return 0;
}
