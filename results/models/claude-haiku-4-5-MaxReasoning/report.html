struct Point {
    double x, y;
};

inline double distance(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return sqrt(dx*dx + dy*dy);
}

// Nearest neighbor heuristic - O(n²)
vector<int> nearestNeighbor(const vector<Point>& cities) {
    int n = cities.size();
    vector<int> tour;
    vector<bool> visited(n, false);
    tour.reserve(n);

    tour.push_back(0);
    visited[0] = true;

    for (int i = 1; i < n; i++) {
        int last = tour.back();
        int nearest = -1;
        double minDist = 1e18;

        for (int j = 0; j < n; j++) {
            if (!visited[j]) {
                double d = distance(cities[last], cities[j]);
                if (d < minDist) {
                    minDist = d;
                    nearest = j;
                }
            }
        }

        tour.push_back(nearest);
        visited[nearest] = true;
    }

    return tour;
}

// 2-opt local search improvement
void twoOpt(vector<int>& tour, const vector<Point>& cities, int maxIterations) {
    int n = tour.size();
    bool improved = true;
    int iterations = 0;

    while (improved && iterations < maxIterations) {
        improved = false;
        iterations++;

        for (int i = 0; i < n - 1; i++) {
            for (int j = i + 2; j < n; j++) {
                double d1 = distance(cities[tour[i]], cities[tour[i+1]]);
                double d2 = distance(cities[tour[j]], cities[tour[(j+1)%n]]);
                double d3 = distance(cities[tour[i]], cities[tour[j]]);
                double d4 = distance(cities[tour[i+1]], cities[tour[(j+1)%n]]);

                if (d3 + d4 < d1 + d2 - 1e-9) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                    break;
                }
            }
            if (improved) break;
        }
    }
}

// Space-filling curve (Z-order Morton code) - O(n log n)
vector<int> spaceFilling(const vector<Point>& cities) {
    int n = cities.size();

    // Find bounding box
    double minX = cities[0].x, maxX = cities[0].x;
    double minY = cities[0].y, maxY = cities[0].y;

    for (const auto& p : cities) {
        if (p.x < minX) minX = p.x;
        if (p.x > maxX) maxX = p.x;
        if (p.y < minY) minY = p.y;
        if (p.y > maxY) maxY = p.y;
    }

    double rangeX = maxX - minX;
    double rangeY = maxY - minY;
    if (rangeX < 1e-9) rangeX = 1e-9;
    if (rangeY < 1e-9) rangeY = 1e-9;

    // Compute Morton codes
    vector<pair<long long, int>> coded;
    coded.reserve(n);

    for (int i = 0; i < n; i++) {
        double nx = (cities[i].x - minX) / rangeX;
        double ny = (cities[i].y - minY) / rangeY;

        int ix = (int)(nx * 65535.0);
        int iy = (int)(ny * 65535.0);

        // Interleave bits to create Morton code
        long long code = 0;
        for (int bit = 0; bit < 16; bit++) {
            code |= ((long long)((ix >> bit) & 1)) << (2*bit);
            code |= ((long long)((iy >> bit) & 1)) << (2*bit + 1);
        }

        coded.push_back({code, i});
    }

    // Sort by Morton code
    sort(coded.begin(), coded.end());

    // Build tour
    vector<int> tour(n);
    for (int i = 0; i < n; i++) {
        tour[i] = coded[i].second;
    }

    return tour;
}

    for i in 0..n {
        for j in (i + 1)..n {
            let dx = graph.nodes[j].x - graph.nodes[i].x;
            let dy = graph.nodes[j].y - graph.nodes[i].y;
            let dist_sq = dx * dx + dy * dy;
            let dist = dist_sq.sqrt();

            if dist > 0.01 {
                let repulsive = k * k / (dist + 0.1);
                let fx = (dx / dist) * repulsive;
                let fy = (dy / dist) * repulsive;

                forces[i].0 -= fx;
                forces[i].1 -= fy;
                forces[j].0 += fx;
                forces[j].1 += fy;
            }
        }
    }

    // Attractive forces along edges
    for &(u, v) in &graph.edges {
        let dx = graph.nodes[v].x - graph.nodes[u].x;
        let dy = graph.nodes[v].y - graph.nodes[u].y;

            let fx = (dx / dist) * attractive;
            let fy = (dy / dist) * attractive;

            forces[u].0 += fx;
            forces[u].1 += fy;
            forces[v].0 -= fx;
            forces[v].1 -= fy;
        }

    // Update positions with temperature control
    for i in 0..n {
        let (fx, fy) = forces[i];

            let displacement = temp.min(dist);

vector<Vec3> getTetrahedronVertices(const Vec3& center, double rx, double ry, double rz) {
    double h = TETRA_CIRCUMRADIUS;
    vector<Vec3> verts = {
        Vec3(0, 0, h),
        Vec3(2*h/sqrt(3), 0, -h/3),
        Vec3(-h/sqrt(3), h, -h/3),
        Vec3(-h/sqrt(3), -h, -h/3)
    };

    for (auto& v : verts) {
        // Rotation around x-axis
        double y = v.y * cos(rx) - v.z * sin(rx);
        double z = v.y * sin(rx) + v.z * cos(rx);
        v.y = y;
        v.z = z;

bool pointInConvexPolyhedron(const Vec3& p, const vector<Vec3>& vertices) {
    if (vertices.size() < 4) return false;

    Vec3 center;
    for (const auto& v : vertices) center = center + v;
    center = center * (1.0 / vertices.size());

    int n = vertices.size();
    for (int i = 0; i < n; i++) {
        Vec3 v0 = vertices[i];
        Vec3 v1 = vertices[(i+1) % n];
        Vec3 v2 = center;

        Vec3 normal = (v1 - v0).cross(v2 - v0);
        double side = (p - v0).dot(normal);
        double centerSide = (center - v0).dot(normal);

        if ((side > 0) != (centerSide > 0)) return false;
    }

    return true;
}

    {
        public double X, Y, Z, Vx, Vy, Vz;
        public double R => Math.Sqrt(X * X + Y * Y + Z * Z);

    static Orbit ParseOrbit(string[] parts)
    {

    static double[][] BuildCostMatrix(Orbit[] orbits, int n, double mu)
    {
        double[][] costs = new double[n + 1][];
        for (int i = 0; i <= n; i++) costs[i] = new double[n + 1];
        for (int j = 1; j <= n; j++) costs[0][j] = CalculateDeltaV(orbits[0], orbits[j], mu);

                costs[i][j] = (i == j) ? 0 : CalculateDeltaV(orbits[i], orbits[j], mu);
        return costs;
    }

    static double CalculateDeltaV(Orbit from, Orbit to, double mu)
    {
        double r1 = from.R, r2 = to.R;
        if (r1 < 100 || r2 < 100) return double.MaxValue;
        double v1 = Math.Sqrt(mu / r1), v2 = Math.Sqrt(mu / r2);
        double a_trans = (r1 + r2) / 2;
        double v_trans1 = Math.Sqrt(mu * (2.0 / r1 - 1.0 / a_trans));
        double v_trans2 = Math.Sqrt(mu * (2.0 / r2 - 1.0 / a_trans));
        double deltaV1 = Math.Abs(v_trans1 - v1), deltaV2 = Math.Abs(v2 - v_trans2);
        double dx = to.X - from.X, dy = to.Y - from.Y, dz = to.Z - from.Z;
        double posCost = Math.Sqrt(dx * dx + dy * dy + dz * dz) * 0.001;
        return deltaV1 + deltaV2 + posCost;
    }

    static int[] SolveTSP(double[][] costs, int n)
    {

    float dot(const Vec3& v) const { return x * v.x + y * v.y + z * v.z; }
    Vec3 cross(const Vec3& v) const { return Vec3(y * v.z - z * v.y, z * v.x - x * v.z, x * v.y - y * v.x); }

float signedDistanceToPlane(Vec3 point, Vec3 planePoint, Vec3 normal) {
    return (point - planePoint).dot(normal);
}

bool isPointInMesh(Vec3 p, const Mesh& mesh) {
    int count = 0;
    for (const auto& face : mesh.faces) {
        Vec3 v0 = mesh.vertices[face.a];
        Vec3 v1 = mesh.vertices[face.b];
        Vec3 v2 = mesh.vertices[face.c];

        Vec3 e1 = v1 - v0;
        Vec3 e2 = v2 - v0;
        Vec3 normal = e1.cross(e2).normalized();

        float d = signedDistanceToPlane(p, v0, normal);
        if (d < -EPS) count++;
    }
    return (count % 2) == 1;
}

bool rayIntersectTriangle(Vec3 rayOrigin, Vec3 rayDir, Vec3 v0, Vec3 v1, Vec3 v2, float& t) {
    const float EPSILON = 1e-6f;
    Vec3 edge1 = v1 - v0;
    Vec3 edge2 = v2 - v0;
    Vec3 h = rayDir.cross(edge2);
    float a = edge1.dot(h);

    if (abs(a) < EPSILON) return false;

    float f = 1.0f / a;
    Vec3 s = rayOrigin - v0;
    float u = f * s.dot(h);

    if (u < 0.0f || u > 1.0f) return false;

    Vec3 q = s.cross(edge1);
    float v = f * rayDir.dot(q);

    if (v < 0.0f || u + v > 1.0f) return false;

    t = f * edge2.dot(q);
    return t > EPSILON;
}

    let mut reader = stdin.lock().lines();

    // Read H and W
    let first_line = reader.next().unwrap().unwrap();
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    let h: usize = parts[0].parse().unwrap();
    let w: usize = parts[1].parse().unwrap();

    // Read maze
    let mut maze = Vec::new();
    let mut start = (0, 0);
    let mut goal = (0, 0);

    for i in 0..h {
        let line = reader.next().unwrap().unwrap();
        let row: Vec<char> = line.chars().collect();

        for j in 0..row.len() {
            match row[j] {
                'A' => start = (i, j),
                'B' => goal = (i, j),
                _ => {}
            }
        }
        maze.push(row);
    }

    // BFS
    let mut queue = VecDeque::new();
    let mut visited = vec![false; h * w];
    let mut parent: HashMap<(usize, usize), (usize, usize)> = HashMap::new();

    queue.push_back(start);
    visited[start.0 * w + start.1] = true;

    let directions = [(0, 1), (0, -1), (1, 0), (-1, 0)];
    let mut found = false;

    while let Some((y, x)) = queue.pop_front() {
        if (y, x) == goal {
            found = true;
            break;
        }

        for (dy, dx) in &directions {
            let ny = (y as i32 + dy) as usize;
            let nx = (x as i32 + dx) as usize;

            if ny < h && nx < w && !visited[ny * w + nx] {
                if let Some(cell) = maze.get(ny).and_then(|row| row.get(nx)) {
                    if *cell != '#' {
                        visited[ny * w + nx] = true;
                        parent.insert((ny, nx), (y, x));
                        queue.push_back((ny, nx));
                    }
                }
            }
        }
    }

    if found {
        // Reconstruct path
        let mut path = Vec::new();
        let mut current = goal;

        loop {
            path.push((current.1, current.0)); // Convert to x, y format
            if current == start {
                break;

            current = parent[&current];
        }

        path.reverse();
        println!("{}", path.len());
        for (x, y) in path {

        }

        println!("0");

}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {

    fn rotate(&self, angle: f64) -> Polygon {

        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;

        for p in &self.vertices {
            min_x = min_x.min(p.x);
            max_x = max_x.max(p.x);
            min_y = min_y.min(p.y);
            max_y = max_y.max(p.y);

        (min_x, min_y, max_x, max_y)
    }

        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {

            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;

fn polygons_overlap(p1: &Polygon, p2: &Polygon) -> bool {

            let yi = stock.vertices[i].y;
            let xj = stock.vertices[j].x;
            let yj = stock.vertices[j].y;

               (v.x < (xj - xi) * (v.y - yi) / (yj - yi) + xi) {
                inside = !inside;
            }

                let sy1 = stock.vertices[i].y;

                    if t >= -eps && t <= 1.0 + eps {
                        let closest_x = sx1 + t * dx;

                                     (v.y - closest_y) * (v.y - closest_y);

                            on_edge = true;
                            break;
                        }
                    }
                }
            }

            if !on_edge {
                return false;
            }
        }
    }
    true
}

fn compute_bounds(polygon: &[Vec2]) -> (f64, f64, f64, f64) {
    let min_x = polygon.iter().map(|p| p.x).fold(f64::INFINITY, f64::min);

fn compute_polygon_area(polygon: &[Vec2]) -> f64 {
    let mut area = 0.0;
    for i in 0..polygon.len() {
        let j = (i + 1) % polygon.len();
        area += polygon[i].x * polygon[j].y;
        area -= polygon[j].x * polygon[i].y;
    }
    (area / 2.0).abs()
}

fn point_in_polygon(p: &Vec2, polygon: &[Vec2]) -> bool {
    let mut inside = false;

        if intersect {
            inside = !inside;
        }
        j = i;
    }

    inside

            for (int j = 0; j < yCount; j++)
                heightMap[i, j] = 0;

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                for (int z = 0; z < zCount; z++)
                {
                    int b = Console.In.Read();
                    if (b == -1) break;

                    int mass = b % 16;
                    int value = b / 16;

                    if (mass > 0)
                    {
                        voxels[(x, y, z)] = (mass, value);
                        if (z >= heightMap[x, y])
                            heightMap[x, y] = z + 1;
                    }
                }
            }
        }

        var operations = new List<Operation>();
        var processed = new HashSet<(int, int, int)>();
        var currentHeights = new int[xCount, yCount];
        Array.Copy(heightMap, currentHeights, heightMap.Length);

        // Sort voxels by profitability (value * mass) in descending order
        var sortedVoxels = voxels
            .Where(kv => kv.Value.value > 0) // Only valuable ore
            .OrderByDescending(kv => kv.Value.value * kv.Value.mass)
            .ToList();

        foreach (var voxel in sortedVoxels)
        {
            int x = voxel.Key.x;
            int y = voxel.Key.y;
            int z = voxel.Key.z;

            if (processed.Contains((x, y, z)))
                continue;

            // Check if accessible (exposed)
            if (!IsExposed(voxels, processed, x, y, z, currentHeights))
                continue;

            // Find path to processing plant (0, 0)
            var path = FindPath(x, y, currentHeights, xCount, yCount);
            if (path == null || path.Count == 0)
                continue;

            var op = new Operation
            {
                digX = x,
                digY = y,
                digZ = z,
                steps = path,
                destX = 0,
                destY = 0
            };

            operations.Add(op);
            processed.Add((x, y, z));

            // Update height map after digging
            bool hasMore = false;
            for (int zz = z; zz >= 0; zz--)
            {
                if (voxels.ContainsKey((x, y, zz)) && !processed.Contains((x, y, zz)))
                {
                    currentHeights[x, y] = zz + 1;
                    hasMore = true;
                    break;
                }
            }
            if (!hasMore)
                currentHeights[x, y] = 0;
        }

        // Output
        Console.WriteLine(operations.Count);
        foreach (var op in operations)
        {
            Console.Write($"{op.steps.Count} {op.digX} {op.digY} {op.digZ}");
            foreach (var step in op.steps)
                Console.Write($" {step.Item1} {step.Item2}");
            Console.WriteLine($" {op.destX} {op.destY}");
        }
    }

    static bool IsExposed(
        Dictionary<(int, int, int), (int mass, int value)> voxels,
        HashSet<(int, int, int)> processed,
        int x, int y, int z,
        int[,] heights)
    {
        // Check if exposed from above
        if (z >= heights[x, y] - 1)
            return true;

        // Check if exposed from sides
        int[] dx = { 1, -1, 0, 0 };
        int[] dy = { 0, 0, 1, -1 };

        foreach (int d = 0; d < 4; d++)
        {
            int nx = x + dx[d];
            int ny = y + dy[d];

            if (nx >= 0 && ny >= 0 && nx < heights.GetLength(0) && ny < heights.GetLength(1))
            {
                if (!voxels.ContainsKey((nx, ny, z)))
                    return true;
            }
        }

        return false;
    }

    static List<(int, int)> FindPath(int sx, int sy, int[,] heights, int xCount, int yCount)
    {
        // Simple pathfinding: try to move towards (0, 0)
        var path = new List<(int, int)>();
        int x = sx, y = sy;
        int steps = 0;
        int maxSteps = xCount + yCount + 100;

        while ((x != 0 || y != 0) && steps < maxSteps)
        {
            int bestDx = 0, bestDy = 0;
            int bestCost = int.MaxValue;

            // Try moving in 4 directions
            int[] dx = { 1, -1, 0, 0 };
            int[] dy = { 0, 0, 1, -1 };

            for (int d = 0; d < 4; d++)
            {
                int nx = x + dx[d];
                int ny = y + dy[d];

                if (nx < 0 || ny < 0 || nx >= xCount || ny >= yCount)
                    continue;

                int heightDiff = Math.Abs(heights[nx, ny] - heights[x, y]);
                if (heightDiff > 1)
                    continue; // Slope constraint

                int cost = Math.Abs(nx) + Math.Abs(ny);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestDx = dx[d];
                    bestDy = dy[d];
                }
            }

            if (bestCost == int.MaxValue)
                break;

            x += bestDx;
            y += bestDy;
            path.Add((bestDx, bestDy));
            steps++;
        }

        return (x == 0 && y == 0) ? path : null;
    }
