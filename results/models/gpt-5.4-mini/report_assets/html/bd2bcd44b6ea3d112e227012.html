use std::f64::consts::PI;
use std::io::{self, Read, Write};
use std::str::SplitWhitespace;
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy, Default)]
struct Pad {
    x1: f64,
    x2: f64,
    y: f64,
}

fn parse_next<T: std::str::FromStr>(it: &mut SplitWhitespace<'_>) -> Option<T> {
    it.next()?.parse().ok()
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo { lo } else if v > hi { hi } else { v }
}

fn wrap_angle(mut a: f64) -> f64 {
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }
    a
}

fn angle_from_dxdy(dx: f64, dy: f64) -> f64 {
    dx.atan2(-dy)
}

fn main() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut it = input.split_whitespace();

    let _width: usize = match parse_next(&mut it) { Some(v) => v, None => return };
    let _height: usize = match parse_next(&mut it) { Some(v) => v, None => return };
    let gravity: f64 = match parse_next(&mut it) { Some(v) => v, None => return };
    let max_thrust: f64 = match parse_next(&mut it) { Some(v) => v, None => return };
    let _fuel: f64 = match parse_next(&mut it) { Some(v) => v, None => return };

    let start_x: f64 = match parse_next(&mut it) { Some(v) => v, None => return };
    let start_y: f64 = match parse_next(&mut it) { Some(v) => v, None => return };
    let target_x: f64 = match parse_next(&mut it) { Some(v) => v, None => return };
    let target_y: f64 = match parse_next(&mut it) { Some(v) => v, None => return };

    let mut rows: Vec<String> = Vec::new();
    for _ in 0.._height {
        if let Some(row) = it.next() {
            rows.push(row.to_string());
        } else {
            break;
        }
    }

    let mut pad = Pad { x1: target_x, x2: target_x + 1.0, y: target_y };
    for row in &rows {
        let bytes = row.as_bytes();
        let mut x = 0usize;
        while x < bytes.len() {
            if bytes[x] == b'L' || bytes[x] == b'T' {
                let start = x;
                while x < bytes.len() && (bytes[x] == b'L' || bytes[x] == b'T') { x += 1; }
                pad = Pad { x1: start as f64, x2: x as f64, y: rows.len().saturating_sub(1) as f64 };
            } else {
                x += 1;
            }
        }
    }

    let mut out = io::BufWriter::new(io::stdout());
    let mut prev_y = start_y;
    let mut prev_x = start_x;

    loop {
        let x: f64 = match parse_next(&mut it) { Some(v) => v, None => break };
        let y: f64 = match parse_next(&mut it) { Some(v) => v, None => break };
        let vx: f64 = match parse_next(&mut it) { Some(v) => v, None => break };
        let vy: f64 = match parse_next(&mut it) { Some(v) => v, None => break };
        let angle: f64 = match parse_next(&mut it) { Some(v) => v, None => break };
        let angular_vel: f64 = match parse_next(&mut it) { Some(v) => v, None => break };
        let fuel_remaining: f64 = match parse_next(&mut it) { Some(v) => v, None => break };

        let pad_cx = 0.5 * (pad.x1 + pad.x2);
        let dx = pad_cx - x;
        let dy = pad.y - y;
        let dist = (dx * dx + dy * dy).sqrt();
        let alt = (pad.y - y).max(0.0);

        let desired_vy = if alt > 80.0 { 18.0 } else if alt > 35.0 { 10.0 } else if alt > 15.0 { 5.0 } else { 1.5 };
        let desired_vx = clamp(dx * 0.08, -6.0, 6.0);

        let mut thrust = 1.0;
        if fuel_remaining < 5.0 { thrust = 0.85; }
        if alt < 12.0 { thrust = 1.0; }
        if vy > desired_vy + 1.0 { thrust = 1.0; }
        else if vy < desired_vy - 3.0 { thrust = 0.15; }

        let mut target_angle = if alt > 12.0 {
            let ax = clamp((desired_vx - vx) * 0.18, -1.0, 1.0);
            let ay = clamp((desired_vy - vy + gravity * 0.25) * 0.12, -1.0, 1.0);
            angle_from_dxdy(ax, 1.0 + ay)
        } else {
            let horizontal = clamp(-vx * 0.16 + dx * 0.05, -0.6, 0.6);
            if vy > 2.0 { angle_from_dxdy(horizontal, 1.0) } else { 0.0 }
        };

        if dist < 15.0 && alt < 20.0 {
            target_angle *= 0.5;
        }

        let angle_error = wrap_angle(target_angle - angle);
        let turn = clamp(angle_error * 1.8 - angular_vel * 0.35, -1.0, 1.0);

        if alt < 8.0 && vy > 2.5 {
            thrust = 1.0;
        }

        if fuel_remaining < 1.0 {
            thrust = 0.0;
        }

        let _ = writeln!(out, "{:.3} {:.3}", clamp(thrust, 0.0, 1.0), turn);
        let _ = out.flush();
        prev_x = x;
        prev_y = y;

        thread::sleep(Duration::from_millis(20));
    }
}
