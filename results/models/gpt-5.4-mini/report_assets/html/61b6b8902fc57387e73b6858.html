use std::f64::consts::PI;
use std::io::{self, Read, Write};
use std::str::SplitWhitespace;
use std::thread;
use std::time::Duration;

#[derive(Clone, Copy, Default)]
struct Pad {
    x1: f64,
    x2: f64,
    y: f64,
}

fn parse_next<T: std::str::FromStr>(it: &mut SplitWhitespace<'_>) -> Option<T> {
    it.next()?.parse().ok()
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo { lo } else if v > hi { hi } else { v }
}

fn wrap_angle(mut a: f64) -> f64 {
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }
    a
}

fn angle_from_dxdy(dx: f64, dy: f64) -> f64 {
    dx.atan2(-dy)
}

fn main() {

        return;

    }
