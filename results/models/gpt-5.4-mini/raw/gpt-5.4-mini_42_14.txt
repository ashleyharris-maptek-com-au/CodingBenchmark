OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_2 = OpConstant %float 0.2
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_8 = OpConstant %float 0.8

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%_ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%_ptr_Input_v4float = OpTypePointer Input %v4float

%arr_v3_6 = OpTypeArray %v3float %uint_6
%_ptr_Input_arr_v3_6 = OpTypePointer Input %arr_v3_6
%_ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%gl_in = OpVariable %_ptr_Input_arr_pv_in Input
%worldPos = OpVariable %_ptr_Input_arr_v3_6 Input
%normal = OpVariable %_ptr_Input_arr_v3_6 Input
%color = OpVariable %_ptr_Input_arr_v3_6 Input
%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output

%main = OpFunction %void None %fn
%label = OpLabel

%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_4 %int_0
%p0_v4 = OpLoad %v4float %p0_ptr
%p2_v4 = OpLoad %v4float %p2_ptr
%p4_v4 = OpLoad %v4float %p4_ptr
%p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2
%p4 = OpVectorShuffle %v3float %p4_v4 %p4_v4 0 1 2

%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0

%e1x = OpCompositeExtract %float %e1 0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2
%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2

%t0 = OpFMul %float %e1y %e2z
%t1 = OpFMul %float %e1z %e2y
%cx = OpFSub %float %t0 %t1
%t2 = OpFMul %float %e1z %e2x
%t3 = OpFMul %float %e1x %e2z
%cy = OpFSub %float %t2 %t3
%t4 = OpFMul %float %e1x %e2y
%t5 = OpFMul %float %e1y %e2x
%cz = OpFSub %float %t4 %t5
%cross = OpCompositeConstruct %v3float %cx %cy %cz

%1 = OpExtInstImport "GLSL.std.450"
%N = OpExtInst %v3float %1 Normalize %cross
%Lraw = OpCompositeConstruct %v3float %float_1 %float_1 %float_1
%L = OpExtInst %v3float %1 Normalize %Lraw
%dotNL = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %dotNL %float_0

%c0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%c2_ptr = OpAccessChain %_ptr_Input_v3float %color %int_2
%c4_ptr = OpAccessChain %_ptr_Input_v3float %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%sum01 = OpFAdd %v3float %c0 %c2
%sum012 = OpFAdd %v3float %sum01 %c4
%base = OpVectorTimesScalar %v3float %sum012 %float_0_33333334

%shade0 = OpFMul %float %float_0_8 %diffuse
%shade = OpFAdd %float %float_0_2 %shade0
%finalColor = OpVectorTimesScalar %v3float %base %shade

%pos_out_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0

OpStore %pos_out_ptr %p0_v4
OpStore %outColor %finalColor
OpEmitVertex

OpStore %pos_out_ptr %p2_v4
OpStore %outColor %finalColor
OpEmitVertex

OpStore %pos_out_ptr %p4_v4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
