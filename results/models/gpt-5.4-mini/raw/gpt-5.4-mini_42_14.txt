; SPIR-V
OpCapability Geometry
OpCapability Shader
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_in "gl_PerVertex"
OpName %gl_out "gl_out"
OpName %gl_PerVertex_out "gl_PerVertex"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "proj"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_6 = OpConstant %uint 6
%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_02 = OpConstant %float 0.2
%float_08 = OpConstant %float 0.8
%float_third = OpConstant %float 0.33333334
%float_inv_sqrt3 = OpConstant %float 0.57735026
%v3float_light = OpConstantComposite %v3float %float_inv_sqrt3 %float_inv_sqrt3 %float_inv_sqrt3

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%arr_v3 = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ubo_struct = OpTypeStruct %v4float %v4float %v4float %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_struct

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %funct
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%p0v4 = OpLoad %v4float %p0_ptr
%p2v4 = OpLoad %v4float %p2_ptr
%p4v4 = OpLoad %v4float %p4_ptr
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2
%p4 = OpVectorShuffle %v3float %p4v4 %p4v4 0 1 2

%e1 = OpFSub %v3float %p2 %p0
%e2 = OpFSub %v3float %p4 %p0
%e1y = OpCompositeExtract %float %e1 1
%e1z = OpCompositeExtract %float %e1 2
%e1x = OpCompositeExtract %float %e1 0
%e2x = OpCompositeExtract %float %e2 0
%e2y = OpCompositeExtract %float %e2 1
%e2z = OpCompositeExtract %float %e2 2

%t0 = OpFMul %float %e1y %e2z
%t1 = OpFMul %float %e1z %e2y
%nx = OpFSub %float %t0 %t1
%t2 = OpFMul %float %e1z %e2x
%t3 = OpFMul %float %e1x %e2z
%ny = OpFSub %float %t2 %t3
%t4 = OpFMul %float %e1x %e2y
%t5 = OpFMul %float %e1y %e2x
%nz = OpFSub %float %t4 %t5
%cross = OpCompositeConstruct %v3float %nx %ny %nz
%N = OpExtInst %v3float %1 Normalize %cross
%L = OpExtInst %v3float %1 Normalize %v3float_light
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %1 FMax %ndotl %float_0

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c0 = OpLoad %v3float %c0_ptr
%c2 = OpLoad %v3float %c2_ptr
%c4 = OpLoad %v3float %c4_ptr
%sum01 = OpFAdd %v3float %c0 %c2
%sum012 = OpFAdd %v3float %sum01 %c4
%base = OpVectorTimesScalar %v3float %sum012 %float_third
%lit1 = OpFMul %float %float_08 %diffuse
%lit = OpFAdd %float %float_02 %lit1
%final = OpVectorTimesScalar %v3float %base %lit

OpStore %outColor %final
%pos0_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos0_out %p0v4
OpEmitVertex

OpStore %outColor %final
%pos2_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos2_out %p2v4
OpEmitVertex

OpStore %outColor %final
%pos4_out = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos4_out %p4v4
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
