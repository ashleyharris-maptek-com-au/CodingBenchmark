use std::collections::{HashMap, HashSet, VecDeque};
use std::io::{self, Read};

#[derive(Clone, Copy, PartialEq, Eq, Hash)]
struct Cell {
    x: i32,
    y: i32,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w: i32 = match it.next() { Some(v) => v.parse().unwrap(), None => return };
    let h: i32 = match it.next() { Some(v) => v.parse().unwrap(), None => return };
    let m: usize = match it.next() { Some(v) => v.parse().unwrap(), None => return };
    let require_cycle: i32 = match it.next() { Some(v) => v.parse().unwrap(), None => return };

    let mut blocked = HashSet::new();
    for _ in 0..m {
        if let (Some(xs), Some(ys)) = (it.next(), it.next()) {
            let x: i32 = xs.parse().unwrap();
            let y: i32 = ys.parse().unwrap();
            blocked.insert((x, y));
        }
    }

    let start = (0i32, 0i32);
    if start.0 < 0 || start.1 < 0 || start.0 >= w || start.1 >= h || blocked.contains(&start) {
        println!("0");
        return;
    }

    let mut cells = Vec::new();
    let mut idx = HashMap::new();
    for y in 0..h {
        for x in 0..w {
            if !blocked.contains(&(x, y)) {
                idx.insert((x, y), cells.len());
                cells.push((x, y));
            }
        }
    }

    let n = cells.len();
    if n == 0 {
        println!("0");
        return;
    }

    if require_cycle == 1 && n < 4 {
        println!("0");
        return;
    }

    let s = match idx.get(&start) {
        Some(&v) => v,
        None => { println!("0"); return; }
    };

    let mut adj = vec![Vec::<usize>::new(); n];
    for (i, &(x, y)) in cells.iter().enumerate() {
        let neigh = [(x + 1, y), (x - 1, y), (x, y + 1), (x, y - 1)];
        for nb in neigh.iter() {
            if let Some(&j) = idx.get(nb) {
                adj[i].push(j);
            }
        }
    }

    let mut deg = vec![0usize; n];
    for i in 0..n { deg[i] = adj[i].len(); }

    // Quick impossibility checks.
    for i in 0..n {
        if deg[i] == 0 {
            println!("0");
            return;
        }
    }
    if require_cycle == 1 {
        for i in 0..n {
            if deg[i] < 2 {
                println!("0");
                return;
            }
        }
    } else {
        let odd = deg.iter().filter(|&&d| d % 2 == 1).count();
        if odd > 2 {
            println!("0");
            return;
        }
    }

    // For large open-ish rectangles, use a deterministic serpentine construction.
    if blocked.is_empty() {
        if let Some(path) = open_grid_path(w, h, require_cycle == 1) {
            output_path(&path);
            return;
        }
    }

    // General case: DFS with strong pruning and heuristic ordering.
    let mut used = vec![false; n];
    let mut path = Vec::with_capacity(n);
    used[s] = true;
    path.push(s);

    // Connectivity / parity check helper.
    let mut best_solution: Option<Vec<usize>> = None;

    fn reachable_count(start: usize, used: &Vec<bool>, adj: &Vec<Vec<usize>>) -> usize {
        let mut q = VecDeque::new();
        let mut vis = vec![false; used.len()];
        q.push_back(start);
        vis[start] = true;
        let mut cnt = 0usize;
        while let Some(u) = q.pop_front() {
            cnt += 1;
            for &v in &adj[u] {
                if !used[v] && !vis[v] {
                    vis[v] = true;
                    q.push_back(v);
                }
            }
        }
        cnt
    }

    fn can_finish(cur: usize, used: &Vec<bool>, adj: &Vec<Vec<usize>>, total: usize) -> bool {
        let remain = total - used.iter().filter(|&&b| b).count();
        if remain == 0 { return true; }
        let mut first = None;
        for &v in &adj[cur] {
            if !used[v] { first = Some(v); break; }
        }
        let Some(st) = first else { return false; };
        let reach = reachable_count(st, used, adj);
        reach >= remain
    }

    fn dfs(cur: usize, start: usize, require_cycle: bool, used: &mut Vec<bool>, path: &mut Vec<usize>, adj: &Vec<Vec<usize>>, total: usize, best_solution: &mut Option<Vec<usize>>) -> bool {
        if path.len() == total {
            if !require_cycle || adj[cur].iter().any(|&v| v == start) {
                *best_solution = Some(path.clone());
                return true;
            }
            return false;
        }

        if !can_finish(cur, used, adj, total) {
            return false;
        }

        let mut cand: Vec<usize> = adj[cur].iter().copied().filter(|&v| !used[v]).collect();
        cand.sort_by_key(|&v| adj[v].iter().filter(|&&u| !used[u]).count());

        for v in cand {
            used[v] = true;
            path.push(v);

            let mut fail = false;
            // Local degree pruning: every unused vertex must retain enough unused neighbors.
            for u in 0..total {
                if !used[u] {
                    let d = adj[u].iter().filter(|&&z| !used[z]).count();
                    if d == 0 {
                        fail = true;
                        break;
                    }
                    if require_cycle && d == 1 && u != start {
                        fail = true;
                        break;
                    }
                }
            }

            if !fail && dfs(v, start, require_cycle, used, path, adj, total, best_solution) {
                return true;
            }

            path.pop();
            used[v] = false;
        }
        false
    }

    if dfs(s, s, require_cycle == 1, &mut used, &mut path, &adj, n, &mut best_solution) {
        if let Some(sol) = best_solution {
            let out: Vec<(i32, i32)> = sol.into_iter().map(|i| cells[i]).collect();
            output_path(&out);
            return;
        }
    }

    println!("0");
}

fn open_grid_path(w: i32, h: i32, cycle: bool) -> Option<Vec<(i32, i32)>> {
    let mut path = Vec::new();
    if w == 1 && h == 1 {
        if cycle { return None; }
        path.push((0, 0));
        return Some(path);
    }

    if !cycle {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w { path.push((x, y)); }
            } else {
                for x in (0..w).rev() { path.push((x, y)); }
            }
        }
        return Some(path);
    }

    if w == 1 || h == 1 {
        return None;
    }

    if w % 2 == 0 {
        for x in 0..w {
            if x % 2 == 0 {
                for y in 0..h { path.push((x, y)); }
            } else {
                for y in (0..h).rev() { path.push((x, y)); }
            }
        }
        if adjacent(path[0], *path.last().unwrap()) { Some(path) } else { None }
    } else if h % 2 == 0 {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w { path.push((x, y)); }
            } else {
                for x in (0..w).rev() { path.push((x, y)); }
            }
        }
        if adjacent(path[0], *path.last().unwrap()) { Some(path) } else { None }
    } else {
        None
    }
}

fn adjacent(a: (i32, i32), b: (i32, i32)) -> bool {
    (a.0 - b.0).abs() + (a.1 - b.1).abs() == 1
}

fn output_path(path: &[(i32, i32)]) {
    println!("{}", path.len());
    for &(x, y) in path {
        println!("{} {}", x, y);
    }
}
