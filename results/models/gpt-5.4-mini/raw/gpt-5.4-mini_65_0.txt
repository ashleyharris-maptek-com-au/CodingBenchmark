declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Task = {
  id: string | number;
  duration: number;
  deps: Array<string | number>;
  deadline?: number;
  weight?: number;
};

type ParsedTask = {
  id: string;
  idx: number;
  duration: number;
  deps: number[];
  deadline: number;
  weight: number;
  out: number[];
  indeg: number;
  rank: number;
};

const machineCount = Math.max(1, Number(data.machineCount) | 0);
const tasks: Task[] = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasks.length;

const idToIndex = new Map<string, number>();
for (let i = 0; i < n; i++) idToIndex.set(String(tasks[i].id), i);

const parsed: ParsedTask[] = tasks.map((t, idx) => ({
  id: String(t.id),
  idx,
  duration: Math.max(0, Number(t.duration) || 0),
  deps: [],
  deadline: Number.isFinite(Number(t.deadline)) ? Number(t.deadline) : 0,
  weight: Number.isFinite(Number(t.weight)) ? Number(t.weight) : 1,
  out: [],
  indeg: 0,
  rank: 0,
}));

for (let i = 0; i < n; i++) {
  const deps = Array.isArray(tasks[i].deps) ? tasks[i].deps : [];
  for (const dep of deps) {
    const j = idToIndex.get(String(dep));
    if (j !== undefined && j !== i) parsed[i].deps.push(j);
  }
}

for (let i = 0; i < n; i++) {
  for (const d of parsed[i].deps) {
    parsed[d].out.push(i);
    parsed[i].indeg++;
  }
}

const indeg0 = parsed.map(t => t.indeg);
const q: number[] = [];
for (let i = 0; i < n; i++) if (indeg0[i] === 0) q.push(i);
const topo: number[] = [];
for (let head = 0; head < q.length; head++) {
  const u = q[head];
  topo.push(u);
  for (const v of parsed[u].out) {
    indeg0[v]--;
    if (indeg0[v] === 0) q.push(v);
  }
}

if (topo.length !== n) {
  const order = parsed.map(t => t.id);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

for (let i = n - 1; i >= 0; i--) {
  const u = topo[i];
  let best = 0;
  for (const v of parsed[u].out) {
    const cand = parsed[v].rank + parsed[v].duration;
    if (cand > best) best = cand;
  }
  const slack = Math.max(0, parsed[u].deadline);
  const urgency = parsed[u].weight / (1 + Math.max(0, parsed[u].deadline - best - parsed[u].duration));
  parsed[u].rank = best + parsed[u].duration + urgency * 0.001 + parsed[u].weight * 1e-9 - slack * 1e-12;
}

const ready: number[] = [];
const remainingDeps = parsed.map(t => t.indeg);
for (let i = 0; i < n; i++) if (remainingDeps[i] === 0) ready.push(i);

const machineFree: number[] = new Array(machineCount).fill(0);
const finishTime: number[] = new Array(n).fill(0);
const startTime: number[] = new Array(n).fill(0);
const assignedMachine: number[] = new Array(n).fill(0);
let completed = 0;

function pickReady(): number {
  let bestIdx = 0;
  for (let i = 1; i < ready.length; i++) {
    const a = parsed[ready[i]];
    const b = parsed[ready[bestIdx]];
    if (
      a.rank > b.rank ||
      (a.rank === b.rank && a.weight > b.weight) ||
      (a.rank === b.rank && a.weight === b.weight && a.deadline < b.deadline) ||
      (a.rank === b.rank && a.weight === b.weight && a.deadline === b.deadline && a.duration > b.duration) ||
      (a.rank === b.rank && a.weight === b.weight && a.deadline === b.deadline && a.duration === b.duration && a.idx < b.idx)
    ) bestIdx = i;
  }
  const task = ready[bestIdx];
  ready[bestIdx] = ready[ready.length - 1];
  ready.pop();
  return task;
}

while (completed < n) {
  if (ready.length === 0) {
    const order = topo.map(i => parsed[i].id);
    process.stdout.write(JSON.stringify({ order }));
    process.exit(0);
  }

  const taskIdx = pickReady();
  const task = parsed[taskIdx];
  let bestMachine = 0;
  let bestStart = Math.max(machineFree[0], ...task.deps.map(d => finishTime[d]));
  let bestFinish = bestStart + task.duration;

  for (let m = 1; m < machineCount; m++) {
    const s = Math.max(machineFree[m], ...task.deps.map(d => finishTime[d]));
    const f = s + task.duration;
    if (
      f < bestFinish ||
      (f === bestFinish && s < bestStart) ||
      (f === bestFinish && s === bestStart && machineFree[m] < machineFree[bestMachine])
    ) {
      bestMachine = m;
      bestStart = s;
      bestFinish = f;
    }
  }

  startTime[taskIdx] = bestStart;
  finishTime[taskIdx] = bestFinish;
  assignedMachine[taskIdx] = bestMachine;
  machineFree[bestMachine] = bestFinish;
  completed++;

  for (const v of task.out) {
    remainingDeps[v]--;
    if (remainingDeps[v] === 0) ready.push(v);
  }
}

const schedule = topo.map(i => ({
  id: parsed[i].id,
  start: startTime[i] | 0,
  machine: assignedMachine[i] | 0,
}));

process.stdout.write(JSON.stringify({ schedule }));
