declare const require: any;
const fs = require('fs');

interface Task {
  id: string | number;
  duration: number;
  deps?: Array<string | number>;
  deadline?: number;
  weight?: number;
}

interface InputData {
  machineCount: number;
  tasks: Task[];
}

interface NodeInfo {
  id: string | number;
  duration: number;
  deadline: number;
  weight: number;
  deps: Array<string | number>;
  succ: number[];
  indeg: number;
  remainingLen: number;
}

class MaxHeap<T> {
  private data: T[] = [];
  constructor(private readonly score: (x: T) => number) {}
  get size(): number { return this.data.length; }
  push(x: T): void {
    const a = this.data;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.score(a[p]) >= this.score(a[i])) break;
      [a[p], a[i]] = [a[i], a[p]];
      i = p;
    }
  }
  pop(): T | undefined {
    const a = this.data;
    if (a.length === 0) return undefined;
    const top = a[0];
    const last = a.pop()!;
    if (a.length > 0) {
      a[0] = last;
      let i = 0;
      for (;;) {
        let l = i * 2 + 1;
        let r = l + 1;
        let best = i;
        if (l < a.length && this.score(a[l]) > this.score(a[best])) best = l;
        if (r < a.length && this.score(a[r]) > this.score(a[best])) best = r;
        if (best === i) break;
        [a[i], a[best]] = [a[best], a[i]];
        i = best;
      }
    }
    return top;
  }
}

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input) as InputData;
const tasks = Array.isArray(data.tasks) ? data.tasks : [];

const n = tasks.length;
const idToIndex = new Map<string | number, number>();
for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

const nodes: NodeInfo[] = tasks.map((t) => ({
  id: t.id,
  duration: Number.isFinite(t.duration) ? Math.max(0, Math.floor(t.duration)) : 0,
  deadline: Number.isFinite(t.deadline as number) ? Math.floor(t.deadline as number) : Number.MAX_SAFE_INTEGER,
  weight: Number.isFinite(t.weight as number) ? Number(t.weight) : 1,
  deps: Array.isArray(t.deps) ? t.deps.slice() : [],
  succ: [],
  indeg: 0,
  remainingLen: 0,
}));

for (let i = 0; i < n; i++) {
  const deps = nodes[i].deps;
  for (const depId of deps) {
    const j = idToIndex.get(depId);
    if (j !== undefined) {
      nodes[i].indeg++;
      nodes[j].succ.push(i);
    }
  }
}

// Compute a critical-path style priority: duration plus longest successor chain.
const memo = new Array<number>(n).fill(-1);
const visiting = new Array<boolean>(n).fill(false);
const dfs = (i: number): number => {
  if (memo[i] >= 0) return memo[i];
  if (visiting[i]) return nodes[i].duration;
  visiting[i] = true;
  let best = 0;
  for (const s of nodes[i].succ) {
    const cand = dfs(s);
    if (cand > best) best = cand;
  }
  visiting[i] = false;
  memo[i] = nodes[i].duration + best;
  return memo[i];
};
for (let i = 0; i < n; i++) nodes[i].remainingLen = dfs(i);

const indeg = nodes.map((x) => x.indeg);
const ready = new MaxHeap<number>((i) => {
  const t = nodes[i];
  const slack = t.deadline === Number.MAX_SAFE_INTEGER ? 0 : -(t.deadline - t.remainingLen);
  const weightDensity = t.duration > 0 ? t.weight / t.duration : t.weight * 10;
  return t.weight * 1e9 + t.remainingLen * 1e6 + weightDensity * 1e3 + slack;
});

for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push(i);

const order: Array<string | number> = [];
while (ready.size > 0) {
  const i = ready.pop()!;
  order.push(nodes[i].id);
  for (const s of nodes[i].succ) {
    indeg[s]--;
    if (indeg[s] === 0) ready.push(s);
  }
}

if (order.length !== n) {
  // Fallback for malformed cyclic input: output original order to remain valid JSON.
  const fallback = tasks.map((t) => t.id);
  process.stdout.write(JSON.stringify({ order: fallback }));
} else {
  process.stdout.write(JSON.stringify({ order }));
}
