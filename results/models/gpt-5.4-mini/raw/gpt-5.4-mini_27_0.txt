#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <numeric>
#include <limits>
#include <cstdint>

using namespace std;

struct Graph {
    int n = 0, m = 0, k = 0;
    vector<vector<int>> adj;
    vector<int> deg;
};

static bool validate_coloring(const Graph& g, const vector<int>& color) {
    if ((int)color.size() != g.n) return false;
    for (int u = 0; u < g.n; ++u) {
        if (color[u] < 0 || color[u] >= g.k) return false;
        for (int v : g.adj[u]) {
            if (u < v && color[u] == color[v]) return false;
        }
    }
    return true;
}

static vector<int> greedy_order_coloring(const Graph& g) {
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        if (g.deg[a] != g.deg[b]) return g.deg[a] > g.deg[b];
        return a < b;
    });

    vector<int> color(g.n, -1);
    vector<int> used(g.k, -1);
    int stamp = 0;
    for (int u : order) {
        ++stamp;
        for (int v : g.adj[u]) {
            int c = color[v];
            if (c >= 0) used[c] = stamp;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (used[c] != stamp) { chosen = c; break; }
        }
        if (chosen == -1) {
            // Allow temporary conflict-free fallback only if k is zero impossible.
            chosen = 0;
        }
        color[u] = chosen;
    }
    return color;
}

static vector<int> dsatur_coloring(const Graph& g) {
    vector<int> color(g.n, -1);
    vector<int> sat(g.n, 0);
    vector<int> neighbor_mask_count(g.n, 0);
    vector<vector<char>> seen(g.n, vector<char>(g.k, 0));
    vector<int> degree = g.deg;

    for (int iter = 0; iter < g.n; ++iter) {
        int best = -1;
        for (int i = 0; i < g.n; ++i) {
            if (color[i] != -1) continue;
            if (best == -1 || sat[i] > sat[best] || (sat[i] == sat[best] && degree[i] > degree[best]) || (sat[i] == sat[best] && degree[i] == degree[best] && i < best)) {
                best = i;
            }
        }
        if (best == -1) break;

        vector<char> forbidden(g.k, 0);
        for (int v : g.adj[best]) {
            if (color[v] != -1) forbidden[color[v]] = 1;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (!forbidden[c]) { chosen = c; break; }
        }
        if (chosen == -1) chosen = 0;
        color[best] = chosen;
        for (int v : g.adj[best]) {
            if (color[v] == -1 && !seen[v][chosen]) {
                seen[v][chosen] = 1;
                ++sat[v];
            }
        }
    }
    return color;
}

static bool repair_coloring(const Graph& g, vector<int>& color, int max_rounds) {
    if (g.k <= 0) return false;
    vector<int> conflict(g.n, 0);
    auto rebuild_conflicts = [&]() {
        fill(conflict.begin(), conflict.end(), 0);
        for (int u = 0; u < g.n; ++u) {
            for (int v : g.adj[u]) {
                if (u < v && color[u] == color[v]) {
                    ++conflict[u];
                    ++conflict[v];
                }
            }
        }
    };

    rebuild_conflicts();
    vector<int> count(g.k, 0);
    for (int round = 0; round < max_rounds; ++round) {
        int worst = -1;
        for (int i = 0; i < g.n; ++i) {
            if (conflict[i] > 0 && (worst == -1 || conflict[i] > conflict[worst] || (conflict[i] == conflict[worst] && g.deg[i] > g.deg[worst]))) {
                worst = i;
            }
        }
        if (worst == -1) return true;

        fill(count.begin(), count.end(), 0);
        for (int v : g.adj[worst]) {
            if (color[v] >= 0) ++count[color[v]];
        }
        int bestc = -1, bestscore = numeric_limits<int>::max();
        for (int c = 0; c < g.k; ++c) {
            if (count[c] == 0) {
                bestc = c;
                break;
            }
            if (count[c] < bestscore) {
                bestscore = count[c];
                bestc = c;
            }
        }
        if (bestc < 0) bestc = 0;
        color[worst] = bestc;
        rebuild_conflicts();
    }
    return validate_coloring(g, color);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Graph g;
    cin >> g.n >> g.m >> g.k;
    g.adj.assign(g.n, {});
    g.deg.assign(g.n, 0);

    for (int i = 0; i < g.m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= g.n || v < 0 || v >= g.n || u == v) continue;
        g.adj[u].push_back(v);
        g.adj[v].push_back(u);
    }
    for (int i = 0; i < g.n; ++i) {
        auto& a = g.adj[i];
        sort(a.begin(), a.end());
        a.erase(unique(a.begin(), a.end()), a.end());
        g.deg[i] = (int)a.size();
    }

    vector<int> color;
    if (g.n <= 20000 || (long long)g.m <= 10LL * g.n) {
        color = dsatur_coloring(g);
    } else {
        color = greedy_order_coloring(g);
    }

    if (!validate_coloring(g, color)) {
        if (!repair_coloring(g, color, max(1000, g.n * 2))) {
            color = dsatur_coloring(g);
            repair_coloring(g, color, max(1000, g.n * 3));
        }
    }

    if ((int)color.size() != g.n) color.assign(g.n, 0);
    for (int i = 0; i < g.n; ++i) {
        if (color[i] < 0 || color[i] >= g.k) color[i] = 0;
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        cout << color[i];
    }
    cout << '\n';
    return 0;
}
