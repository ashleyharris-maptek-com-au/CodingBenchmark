OpCapability Shader
OpCapability Geometry
OpExtension "SPV_KHR_storage_buffer_storage_class"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main TriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%uint_3 = OpConstant %uint 3
%float_0_08 = OpConstant %float 0.08
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_3 = OpConstant %float 3.0
%v4_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_0

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%color = OpVariable %ptr_in_v3 Input
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

%outColor = OpVariable %ptr_out_v3 Output
OpDecorate %outColor Location 0

%main = OpFunction %void None %func
%label = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%p2 = OpLoad %v4float %p2_ptr

%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2

e1x = OpFSub %float %p1x %p0x
e1y = OpFSub %float %p1y %p0y
e1z = OpFSub %float %p1z %p0z
e2x = OpFSub %float %p2x %p0x
e2y = OpFSub %float %p2y %p0y
e2z = OpFSub %float %p2z %p0z

%nx = OpFSub %float (OpFMul %float %e1y %e2z) (OpFMul %float %e1z %e2y)
%ny = OpFSub %float (OpFMul %float %e1z %e2x) (OpFMul %float %e1x %e2z)
%nz = OpFSub %float (OpFMul %float %e1x %e2y) (OpFMul %float %e1y %e2x)
%len2 = OpFAdd %float (OpFAdd %float (OpFMul %float %nx %nx) (OpFMul %float %ny %ny)) (OpFMul %float %nz %nz)
%len = OpExtInstImport "GLSL.std.450"
%invLen = OpExtInst %float %len Normalize %len2

%main
OpFunctionEnd
