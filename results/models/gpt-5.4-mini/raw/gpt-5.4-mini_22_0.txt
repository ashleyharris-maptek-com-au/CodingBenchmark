using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public class Program
{
    enum PieceType { I, O, T, S, Z, J, L }

    struct Move
    {
        public int Rotation;
        public int Column;
        public int Score;
        public Move(int r, int c, int s) { Rotation = r; Column = c; Score = s; }
    }

    static readonly string[] PieceNames = { "I", "O", "T", "S", "Z", "J", "L" };

    static readonly int[][][] Shapes = new int[][][]
    {
        new int[][] { new[]{0,0,0,0}, new[]{1,1,1,1}, new[]{0,0,0,0}, new[]{0,0,0,0} }, // I
        new int[][] { new[]{1,1}, new[]{1,1} }, // O
        new int[][] { new[]{0,1,0}, new[]{1,1,1}, new[]{0,0,0} }, // T
        new int[][] { new[]{0,1,1}, new[]{1,1,0}, new[]{0,0,0} }, // S
        new int[][] { new[]{1,1,0}, new[]{0,1,1}, new[]{0,0,0} }, // Z
        new int[][] { new[]{1,0,0}, new[]{1,1,1}, new[]{0,0,0} }, // J
        new int[][] { new[]{0,0,1}, new[]{1,1,1}, new[]{0,0,0} }, // L
    };

    static void Main()
    {
        var input = new FastScanner(Console.OpenStandardInput());
        int width = input.NextInt();
        int height = input.NextInt();
        int numPieces = input.NextInt();

        var board = new Board(width, height);
        var outWriter = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = true };

        int pieceIndex = 0;
        while (pieceIndex < numPieces)
        {
            string token = input.Next();
            if (token == null) break;
            PieceType piece = ParsePiece(token);

            var move = ChooseMove(board, piece, pieceIndex, numPieces);
            outWriter.WriteLine($"{move.Rotation} {move.Column}");
            outWriter.Flush();

            string result = input.Next();
            if (result == null) break;
            if (result == "gameover") break;
            if (result.StartsWith("cleared_", StringComparison.Ordinal))
            {
                int cleared = int.Parse(result.Substring(8));
                board.ApplyMove(piece, move.Rotation, move.Column, cleared);
            }
            else
            {
                board.ApplyMove(piece, move.Rotation, move.Column, 0);
            }

            pieceIndex++;
        }
    }

    static PieceType ParsePiece(string s)
    {
        return s[0] switch
        {
            'I' => PieceType.I,
            'O' => PieceType.O,
            'T' => PieceType.T,
            'S' => PieceType.S,
            'Z' => PieceType.Z,
            'J' => PieceType.J,
            'L' => PieceType.L,
            _ => PieceType.T
        };
    }

    static Move ChooseMove(Board board, PieceType piece, int pieceIndex, int numPieces)
    {
        var candidates = board.GetCandidates(piece);
        if (candidates.Count == 0) return new Move(0, 0, int.MinValue / 2);

        int empties = board.EmptyCells();
        int filled = board.Width * board.Height - empties;
        bool complex = board.Width >= 12 || board.Height >= 20 || pieceIndex > 30 || filled > board.Width * board.Height / 4;
        int depth = complex ? 2 : 1;
        if (numPieces - pieceIndex < 10) depth = 1;

        Move best = new Move(0, candidates[0].Column, int.MinValue / 2);
        foreach (var cand in candidates)
        {
            var sim = board.Clone();
            if (!sim.TryPlace(piece, cand.Rotation, cand.Column, out int cleared)) continue;
            double score = ScoreBoard(sim, cleared, piece, cand.Rotation, cand.Column);

            if (depth > 1)
            {
                var nextPieceBias = 0.0;
                foreach (var nextPiece in AllPieces())
                {
                    var nextMoves = sim.GetCandidates(nextPiece);
                    if (nextMoves.Count == 0) continue;
                    double localBest = double.NegativeInfinity;
                    int limit = Math.Min(nextMoves.Count, complex ? 8 : 4);
                    for (int i = 0; i < limit; i++)
                    {
                        var nm = nextMoves[i];
                        var sim2 = sim.Clone();
                        if (!sim2.TryPlace(nextPiece, nm.Rotation, nm.Column, out int cleared2)) continue;
                        double s2 = ScoreBoard(sim2, cleared2, nextPiece, nm.Rotation, nm.Column);
                        if (s2 > localBest) localBest = s2;
                    }
                    if (localBest > double.NegativeInfinity / 2) nextPieceBias += localBest;
                }
                score += 0.08 * nextPieceBias / 7.0;
            }

            int intScore = (int)Math.Round(score * 1000.0);
            if (intScore > best.Score || (intScore == best.Score && TieBreak(cand.Rotation, cand.Column, best.Rotation, best.Column)))
                best = new Move(cand.Rotation, cand.Column, intScore);
        }
        return best;
    }

    static bool TieBreak(int rotA, int colA, int rotB, int colB)
    {
        if (rotA != rotB) return rotA < rotB;
        return colA < colB;
    }

    static IEnumerable<PieceType> AllPieces()
    {
        yield return PieceType.I;
        yield return PieceType.O;
        yield return PieceType.T;
        yield return PieceType.S;
        yield return PieceType.Z;
        yield return PieceType.J;
        yield return PieceType.L;
    }

    static double ScoreBoard(Board b, int cleared, PieceType piece, int rot, int col)
    {
        int[] heights = b.GetColumnHeights();
        int aggregateHeight = 0;
        int holes = 0;
        int bumpiness = 0;
        int maxHeight = 0;
        for (int i = 0; i < heights.Length; i++)
        {
            aggregateHeight += heights[i];
            if (heights[i] > maxHeight) maxHeight = heights[i];
            if (i + 1 < heights.Length) bumpiness += Math.Abs(heights[i] - heights[i + 1]);
            holes += b.CountHolesInColumn(i, heights[i]);
        }

        double score = 0;
        score += cleared * 8.0;
        score -= aggregateHeight * 0.52;
        score -= holes * 1.25;
        score -= bumpiness * 0.22;
        score -= maxHeight * 0.35;

        int center = b.Width / 2;
        int pieceWidth = GetRotatedShape(piece, rot).GetLength(1);
        int pieceCenter = col + pieceWidth / 2;
        score -= Math.Abs(pieceCenter - center) * 0.08;

        if (maxHeight >= b.Height - 2) score -= 1000;
        if (holes > b.Width * 2) score -= 25;
        return score;
    }

    static int[][] GetRotatedShape(PieceType piece, int rotation)
    {
        int idx = (int)piece;
        int[][] shape = Shapes[idx];
        rotation &= 3;
        for (int r = 0; r < rotation; r++) shape = Rotate(shape);
        return Trim(shape);
    }

    static int[][] Rotate(int[][] src)
    {
        int n = src.Length;
        int m = src[0].Length;
        var dst = new int[m][];
        for (int i = 0; i < m; i++)
        {
            dst[i] = new int[n];
            for (int j = 0; j < n; j++) dst[i][j] = src[n - 1 - j][i];
        }
        return dst;
    }

    static int[][] Trim(int[][] src)
    {
        int top = src.Length, bottom = -1, left = src[0].Length, right = -1;
        for (int i = 0; i < src.Length; i++)
        for (int j = 0; j < src[i].Length; j++)
            if (src[i][j] != 0)
            {
                if (i < top) top = i;
                if (i > bottom) bottom = i;
                if (j < left) left = j;
                if (j > right) right = j;
            }
        int h = bottom - top + 1;
        int w = right - left + 1;
        var t = new int[h][];
        for (int i = 0; i < h; i++)
        {
            t[i] = new int[w];
            for (int j = 0; j < w; j++) t[i][j] = src[top + i][left + j];
        }
        return t;
    }

    class Board
    {
        public int Width { get; }
        public int Height { get; }
        private readonly bool[,] cells;

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            cells = new bool[height, width];
        }

        private Board(int width, int height, bool[,] clone)
        {
            Width = width;
            Height = height;
            cells = clone;
        }

        public Board Clone()
        {
            var clone = new bool[Height, Width];
            Array.Copy(cells, clone, cells.Length);
            return new Board(Width, Height, clone);
        }

        public int EmptyCells()
        {
            int empty = 0;
            for (int y = 0; y < Height; y++)
            for (int x = 0; x < Width; x++)
                if (!cells[y, x]) empty++;
            return empty;
        }

        public List<(int Rotation, int Column)> GetCandidates(PieceType piece)
        {
            var list = new List<(int, int)>();
            for (int rot = 0; rot < 4; rot++)
            {
                var shape = GetRotatedShape(piece, rot);
                int w = shape[0].Length;
                int h = shape.Length;
                if (w > Width || h > Height) continue;
                for (int col = 0; col <= Width - w; col++) list.Add((rot, col));
            }
            return list;
        }

        public bool TryPlace(PieceType piece, int rotation, int column, out int cleared)
        {
            cleared = 0;
            var shape = GetRotatedShape(piece, rotation);
            int w = shape[0].Length;
            int h = shape.Length;
            int row = FindDropRow(shape, column);
            if (row < 0 || row + h > Height) return false;
            for (int y = 0; y < h; y++)
            for (int x = 0; x < w; x++)
                if (shape[y][x] != 0)
                {
                    int by = row + y;
                    int bx = column + x;
                    if (by < 0 || by >= Height || bx < 0 || bx >= Width || cells[by, bx]) return false;
                }
            for (int y = 0; y < h; y++)
            for (int x = 0; x < w; x++)
                if (shape[y][x] != 0) cells[row + y, column + x] = true;
            cleared = ClearLines();
            return true;
        }

        public void ApplyMove(PieceType piece, int rotation, int column, int cleared)
        {
            TryPlace(piece, rotation, column, out _);
        }

        private int FindDropRow(int[][] shape, int column)
        {
            int w = shape[0].Length;
            int h = shape.Length;
            int row = 0;
            while (row + h <= Height)
            {
                if (!Collides(shape, column, row + 1)) row++;
                else break;
            }
            if (Collides(shape, column, row)) return -1;
            return row;
        }

        private bool Collides(int[][] shape, int col, int row)
        {
            int h = shape.Length;
            int w = shape[0].Length;
            for (int y = 0; y < h; y++)
            for (int x = 0; x < w; x++)
                if (shape[y][x] != 0)
                {
                    int by = row + y;
                    int bx = col + x;
                    if (by >= Height || bx < 0 || bx >= Width) return true;
                    if (by >= 0 && cells[by, bx]) return true;
                }
            return false;
        }

        private int ClearLines()
        {
            int cleared = 0;
            for (int y = Height - 1; y >= 0; y--)
            {
                bool full = true;
                for (int x = 0; x < Width; x++)
                {
                    if (!cells[y, x]) { full = false; break; }
                }
                if (full)
                {
                    cleared++;
                    for (int yy = y; yy > 0; yy--)
                        for (int x = 0; x < Width; x++)
                            cells[yy, x] = cells[yy - 1, x];
                    for (int x = 0; x < Width; x++) cells[0, x] = false;
                    y++;
                }
            }
            return cleared;
        }

        public int[] GetColumnHeights()
        {
            int[] heights = new int[Width];
            for (int x = 0; x < Width; x++)
            {
                int h = 0;
                for (int y = 0; y < Height; y++)
                {
                    if (cells[y, x]) { h = Height - y; break; }
                }
                heights[x] = h;
            }
            return heights;
        }

        public int CountHolesInColumn(int x, int colHeight)
        {
            int holes = 0;
            bool blockSeen = false;
            for (int y = 0; y < Height; y++)
            {
                if (cells[y, x]) blockSeen = true;
                else if (blockSeen) holes++;
            }
            return holes;
        }
    }

    class FastScanner
    {
        private readonly Stream stream;
        private readonly byte[] buffer = new byte[1 << 16];
        private int len, ptr;

        public FastScanner(Stream s) { stream = s; }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public string Next()
        {
            int c;
            do { c = Read(); if (c == -1) return null; } while (c <= 32);
            var sb = new System.Text.StringBuilder();
            while (c > 32 && c != -1)
            {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }

        public int NextInt() => int.Parse(Next());
    }
}
