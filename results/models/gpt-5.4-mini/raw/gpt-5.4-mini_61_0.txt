declare const require: any;
const fs = require('fs');

interface PackageItem {
  id: string | number;
  x: number;
  y: number;
}

interface InputData {
  depot: [number, number];
  truckCount: number;
  packages: PackageItem[];
}

interface NodeItem {
  id: string | number;
  x: number;
  y: number;
  angle: number;
  dist: number;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const depotX = data.depot[0];
const depotY = data.depot[1];
const maxRoutes = Math.max(0, Math.floor(data.truckCount || 0));
const packages = Array.isArray(data.packages) ? data.packages : [];

if (packages.length === 0 || maxRoutes === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

function dist2(x1: number, y1: number, x2: number, y2: number): number {
  const dx = x1 - x2;
  const dy = y1 - y2;
  return dx * dx + dy * dy;
}

function routeLength(route: NodeItem[]): number {
  let len = 0;
  let px = depotX;
  let py = depotY;
  for (const p of route) {
    len += Math.hypot(px - p.x, py - p.y);
    px = p.x;
    py = p.y;
  }
  len += Math.hypot(px - depotX, py - depotY);
  return len;
}

function insertionDelta(route: NodeItem[], item: NodeItem, pos: number): number {
  const prev = pos === 0 ? null : route[pos - 1];
  const next = pos === route.length ? null : route[pos];
  const addPrev = prev ? Math.hypot(prev.x - item.x, prev.y - item.y) : Math.hypot(depotX - item.x, depotY - item.y);
  const addNext = next ? Math.hypot(item.x - next.x, item.y - next.y) : Math.hypot(item.x - depotX, item.y - depotY);
  const remove = prev && next ? Math.hypot(prev.x - next.x, prev.y - next.y) : 0;
  return addPrev + addNext - remove;
}

function routeDeltaIfRemove(route: NodeItem[], idx: number): number {
  const prev = idx === 0 ? null : route[idx - 1];
  const cur = route[idx];
  const next = idx + 1 === route.length ? null : route[idx + 1];
  const before = prev ? Math.hypot(prev.x - cur.x, prev.y - cur.y) : Math.hypot(depotX - cur.x, depotY - cur.y);
  const after = next ? Math.hypot(cur.x - next.x, cur.y - next.y) : Math.hypot(cur.x - depotX, cur.y - depotY);
  const bridge = prev && next ? Math.hypot(prev.x - next.x, prev.y - next.y) : 0;
  return bridge - before - after;
}

const items: NodeItem[] = packages.map((p) => ({
  id: p.id,
  x: p.x,
  y: p.y,
  angle: Math.atan2(p.y - depotY, p.x - depotX),
  dist: Math.hypot(p.x - depotX, p.y - depotY),
}));

items.sort((a, b) => a.angle - b.angle || a.dist - b.dist);

const routeCount = Math.min(maxRoutes, items.length);
const routes: NodeItem[][] = Array.from({ length: routeCount }, () => []);

// Initial sweep partition: distribute sorted packages into contiguous groups.
for (let i = 0; i < items.length; i++) {
  routes[Math.floor((i * routeCount) / items.length)].push(items[i]);
}

// Local improvement: move packages between routes when it reduces total distance.
let improved = true;
let iterations = 0;
while (improved && iterations < 3) {
  improved = false;
  iterations++;

  for (let a = 0; a < routes.length; a++) {
    for (let i = 0; i < routes[a].length; i++) {
      const item = routes[a][i];
      let bestGain = 0;
      let bestB = -1;
      let bestPos = -1;

      for (let b = 0; b < routes.length; b++) {
        if (b === a) continue;
        const routeB = routes[b];
        for (let pos = 0; pos <= routeB.length; pos++) {
          const gainRemove = routeDeltaIfRemove(routes[a], i);
          const gainInsert = insertionDelta(routeB, item, pos);
          const gain = gainRemove - gainInsert;
          if (gain > bestGain) {
            bestGain = gain;
            bestB = b;
            bestPos = pos;
          }
        }
      }

      if (bestB !== -1) {
        routes[a].splice(i, 1);
        routes[bestB].splice(bestPos, 0, item);
        improved = true;
        i--;
      }
    }
  }

  // Intra-route 2-opt on each route.
  for (const route of routes) {
    let localImproved = true;
    let guard = 0;
    while (localImproved && guard < 2) {
      localImproved = false;
      guard++;
      for (let i = 0; i + 2 < route.length; i++) {
        for (let j = i + 1; j + 1 < route.length; j++) {
          const a = i === 0 ? { x: depotX, y: depotY } : route[i - 1];
          const b = route[i];
          const c = route[j];
          const d = j + 1 === route.length ? { x: depotX, y: depotY } : route[j + 1];
          const oldCost = Math.hypot(a.x - b.x, a.y - b.y) + Math.hypot(c.x - d.x, c.y - d.y);
          const newCost = Math.hypot(a.x - c.x, a.y - c.y) + Math.hypot(b.x - d.x, b.y - d.y);
          if (newCost + 1e-9 < oldCost) {
            for (let l = i, r = j; l < r; l++, r--) {
              const tmp = route[l];
              route[l] = route[r];
              route[r] = tmp;
            }
            localImproved = true;
            improved = true;
          }
        }
      }
    }
  }
}

// If some routes ended up empty, remove them.
const finalRoutes = routes.filter((r) => r.length > 0).map((r) => r.map((p) => p.id));

// If we still have more routes than useful, merge the shortest routes greedily.
while (finalRoutes.length > maxRoutes) {
  finalRoutes.pop();
}

process.stdout.write(JSON.stringify({ routes: finalRoutes }));
