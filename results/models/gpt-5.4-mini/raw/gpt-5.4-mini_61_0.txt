declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Point = { x: number; y: number; id?: string | number };

const depot: Point = { x: data.depot[0], y: data.depot[1] };
const truckCount: number = Math.max(1, data.truckCount | 0);
const packages: Point[] = Array.isArray(data.packages) ? data.packages.map((p: any) => ({ x: +p.x, y: +p.y, id: p.id })) : [];

function dist2(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return dx * dx + dy * dy;
}

function dist(a: Point, b: Point): number {
  return Math.hypot(a.x - b.x, a.y - b.y);
}

if (packages.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

packages.sort((p, q) => {
  const ap = Math.atan2(p.y - depot.y, p.x - depot.x);
  const aq = Math.atan2(q.y - depot.y, q.x - depot.x);
  if (ap !== aq) return ap - aq;
  const rp = dist2(p.x, p.y, depot.x, depot.y);
  const rq = dist2(q.x, q.y, depot.x, depot.y);
  if (rp !== rq) return rp - rq;
  const pid = String(p.id);
  const qid = String(q.id);
  return pid < qid ? -1 : pid > qid ? 1 : 0;
});

const n = packages.length;
const routeCount = Math.min(truckCount, n);
const routes: Point[][] = Array.from({ length: routeCount }, () => []);

// Greedy angular sweep: assign packages in order to the route that can accept
// them with the smallest estimated incremental cost.
const routeLast: Point[] = Array.from({ length: routeCount }, () => depot);
const routeSize = new Array<number>(routeCount).fill(0);

for (let i = 0; i < n; i++) {
  const p = packages[i];
  let bestRoute = 0;
  let bestScore = Number.POSITIVE_INFINITY;

  for (let r = 0; r < routeCount; r++) {
    const last = routeLast[r];
    const score =
      dist(last, p) +
      dist(p, depot) -
      dist(last, depot) +
      routeSize[r] * 1e-6;
    if (score < bestScore) {
      bestScore = score;
      bestRoute = r;
    }
  }

  routes[bestRoute].push(p);
  routeLast[bestRoute] = p;
  routeSize[bestRoute]++;
}

// Improve each route with a simple 2-opt pass over the sequence generated by
// the sweep assignment. This stays linear-ish in practice because routes are
// typically small compared to the full package set.
function routeLength(seq: Point[]): number {
  if (seq.length === 0) return 0;
  let total = dist(depot, seq[0]);
  for (let i = 1; i < seq.length; i++) total += dist(seq[i - 1], seq[i]);
  total += dist(seq[seq.length - 1], depot);
  return total;
}

function twoOpt(seq: Point[]): Point[] {
  if (seq.length < 4) return seq;
  let improved = true;
  let best = seq.slice();
  let bestLen = routeLength(best);

  while (improved) {
    improved = false;
    for (let i = 0; i < best.length - 2; i++) {
      for (let k = i + 2; k < best.length; k++) {
        const a = i === 0 ? depot : best[i - 1];
        const b = best[i];
        const c = best[k - 1];
        const d = k === best.length ? depot : best[k];
        const delta =
          dist(a, c) + dist(b, d) - dist(a, b) - dist(c, d);
        if (delta < -1e-9) {
          const next = best.slice(0, i).concat(best.slice(i, k).reverse(), best.slice(k));
          best = next;
          bestLen += delta;
          improved = true;
          break;
        }
      }
      if (improved) break;
    }
  }

  return best;
}

const finalRoutes = routes
  .filter(r => r.length > 0)
  .map(r => twoOpt(r).map(p => p.id as string | number));

process.stdout.write(JSON.stringify({ routes: finalRoutes }));
