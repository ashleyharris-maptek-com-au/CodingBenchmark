; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo
OpExecutionMode %main OriginUpperLeft

OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %ubo "ubo"

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3.0
%v3float_base = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%v3float_fog = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_8

%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_uniform_v4 = OpTypePointer Uniform %v4float

%ubo = OpVariable %ptr_ubo Uniform
%worldPos = OpVariable %ptr_in_v3 Input
%normal = OpVariable %ptr_in_v3 Input
%outColor = OpVariable %ptr_out_v4 Output

%1 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %funct
%label = OpLabel
%wp = OpLoad %v3float %worldPos
%n = OpLoad %v3float %normal
%nrm = OpExtInst %v3float %1 Normalize %n
%lightptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light = OpLoad %v4float %lightptr
%cameraptr = OpAccessChain %ptr_uniform_v4 %ubo %int_4
%camera = OpLoad %v4float %cameraptr
%lightxyz = OpVectorShuffle %v3float %light %light 0 1 2
%cameraxyz = OpVectorShuffle %v3float %camera %camera 0 1 2
%toLight = OpFSub %v3float %lightxyz %wp
%toCamera = OpFSub %v3float %cameraxyz %wp
%lightDir = OpExtInst %v3float %1 Normalize %toLight
%diffuse = OpExtInst %float %1 FMax %float_0 %OpDot
%lambert = OpFMul %float %float_0_7 %diffuse
%lambert = OpFAdd %float %float_0_3 %lambert
%lambert3 = OpCompositeConstruct %v3float %lambert %lambert %lambert
%baseLit = OpFMul %v3float %v3float_base %lambert3
%dist = OpExtInst %float %1 Length %toCamera
%fognum = OpFSub %float %dist %float_1_5
%fogfactor = OpFDiv %float %fognum %float_3
%fogclamp = OpExtInst %float %1 FClamp %fogfactor %float_0 %float_1
%fog3 = OpCompositeConstruct %v3float %fogclamp %fogclamp %fogclamp
%oneMinusFog = OpFSub %float %float_1 %fogclamp
%oneMinusFog3 = OpCompositeConstruct %v3float %oneMinusFog %oneMinusFog %oneMinusFog
%scene = OpFMul %v3float %baseLit %oneMinusFog3
%fogpart = OpFMul %v3float %v3float_fog %fog3
%finalrgb = OpFAdd %v3float %scene %fogpart
%final = OpCompositeConstruct %v4float %finalrgb %float_1
OpStore %outColor %final
OpReturn
OpFunctionEnd
