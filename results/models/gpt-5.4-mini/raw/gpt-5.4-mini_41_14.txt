OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%ptr_in_v3 = OpTypePointer Function %v3float
%glsl450 = OpExtInstImport "GLSL.std.450"

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_1_5 = OpConstant %float 1.5
%float_3 = OpConstant %float 3.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_8 = OpConstant %float 0.8
%float_0_2 = OpConstant %float 0.2
%base_rgb = OpConstantComposite %v3float %float_0_8 %float_0_2 %float_0_2
%fog_rgb = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_8
%int_3 = OpConstant %int 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %func
%entry = OpLabel
%wp = OpLoad %v3float %worldPos
%nrm = OpLoad %v3float %normal
%light_ptr = OpAccessChain %ptr_uniform_v4 %ubo %int_3
%light4 = OpLoad %v4float %light_ptr
%light = OpVectorShuffle %v3float %light4 %light4 0 1 2
%to_light = OpFSub %v3float %light %wp
%dist = OpExtInst %float %glsl450 Length %to_light
%n = OpExtInst %v3float %glsl450 Normalize %nrm
%to_light_n = OpExtInst %v3float %glsl450 Normalize %to_light
%diffuse = OpExtInst %float %glsl450 FMax %float_0 %glsl450 Dot %n %to_light_n
%lambert = OpExtInst %float %glsl450 Fma %float_0_7 %diffuse %float_0_3
%lit_rgb = OpVectorTimesScalar %v3float %base_rgb %lambert
%fog_raw = OpFDiv %float %OpFSub %float %dist %float_1_5 %float_3
%fog = OpExtInst %float %glsl450 FClamp %fog_raw %float_0 %float_1
%one_minus_fog = OpFSub %float %float_1 %fog
%lit_scaled = OpVectorTimesScalar %v3float %lit_rgb %one_minus_fog
%fog_scaled = OpVectorTimesScalar %v3float %fog_rgb %fog
%color_rgb = OpFAdd %v3float %lit_scaled %fog_scaled
%color_rgba = OpCompositeConstruct %v4float %color_rgb 1 2 3
OpStore %outColor %color_rgba
OpReturn
OpFunctionEnd
