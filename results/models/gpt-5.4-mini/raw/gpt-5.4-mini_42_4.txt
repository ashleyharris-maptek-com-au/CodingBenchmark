; SPIR-V
; Geometry shader: tapered ribbon from line segment
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %_ ""

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %outColor Location 0

OpDecorate %ubo Block
OpMemberDecorate %ubo 0 ColMajor
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 0 MatrixStride 16
OpMemberDecorate %ubo 1 ColMajor
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 1 MatrixStride 16
OpMemberDecorate %ubo 2 ColMajor
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 2 MatrixStride 16
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%f_0 = OpConstant %float 0
%f_1 = OpConstant %float 1
%f_004 = OpConstant %float 0.04
%f_0005 = OpConstant %float 0.005
%v4_0 = OpConstantComposite %v4float %f_0 %f_0 %f_0 %f_0
%v3_0 = OpConstantComposite %v3float %f_0 %f_0 %f_0
%v3_1 = OpConstantComposite %v3float %f_1 %f_1 %f_1

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v3 = OpTypePointer Input %v3float
%arr_v3_2 = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_2
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v4float
%ptr_ubo = OpTypePointer Uniform %ubo_struct
%ubo = OpVariable %ptr_ubo Uniform
%ptr_ubo_mat4 = OpTypePointer Uniform %mat4
%ptr_ubo_v3 = OpTypePointer Uniform %v3float
%ptr_ubo_v4 = OpTypePointer Uniform %v4float

%const0 = OpConstant %uint 0
%const1 = OpConstant %uint 1

%main = OpFunction %void None %funct
%label = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%dir = OpFSub %v2float %p1xy %p0xy
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%negdy = OpFNegate %float %dy
%perp = OpCompositeConstruct %v2float %negdy %dx

%perp_start = OpVectorTimesScalar %v2float %perp %f_004
%perp_end = OpVectorTimesScalar %v2float %perp %f_0005

%p0_plus = OpFAdd %v2float %p0xy %perp_start
%p0_minus = OpFSub %v2float %p0xy %perp_start
%p1_plus = OpFAdd %v2float %p1xy %perp_end
%p1_minus = OpFSub %v2float %p1xy %perp_end

%z0 = OpCompositeExtract %float %p0 2
%w0 = OpCompositeExtract %float %p0 3
%z1 = OpCompositeExtract %float %p1 2
%w1 = OpCompositeExtract %float %p1 3

%v0 = OpCompositeConstruct %v4float %p0_plus  %z0 %w0
%v1 = OpCompositeConstruct %v4float %p0_minus %z0 %w0
%v2 = OpCompositeConstruct %v4float %p1_plus  %z1 %w1
%v3 = OpCompositeConstruct %v4float %p1_minus %z1 %w1

%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %v0
OpStore %outColor %c0
OpEmitVertex

%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %v1
OpStore %outColor %c0
OpEmitVertex

%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %v2
OpStore %outColor %c1
OpEmitVertex

%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %v3
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
