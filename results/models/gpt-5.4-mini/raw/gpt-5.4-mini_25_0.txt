using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class Program
{
    const double Day = 86400.0;

    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm2() => Dot(this);
        public double Norm() => Math.Sqrt(Norm2());
        public Vec3 Normalized()
        {
            double n = Norm();
            return n > 0 ? this / n : new Vec3(0, 0, 0);
        }
    }

    class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R;
        public Vec3 V;
    }

    static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    static Vec3 LinInterp(Vec3 a, Vec3 b, double t) => a + (b - a) * t;

    static double EarthMoonLikeLaunchCost(double distance, double timeDays)
    {
        double baseCost = 2500.0;
        double rangeCost = Math.Min(6000.0, distance / 1e7 * 10.0);
        double timeCost = Math.Max(0.0, (timeDays - 30.0) * 8.0);
        return baseCost + rangeCost + timeCost;
    }

    static void Main()
    {
        var inv = CultureInfo.InvariantCulture;
        var tokens = new List<string>();
        string? line;
        while ((line = Console.ReadLine()) != null)
        {
            foreach (var t in line.Split((char[])null!, StringSplitOptions.RemoveEmptyEntries))
                tokens.Add(t);
        }

        int p = 0;
        int numBodies = int.Parse(tokens[p++], inv);
        double warningDays = double.Parse(tokens[p++], inv);
        double impactorMass = double.Parse(tokens[p++], inv);
        double dvBudget = double.Parse(tokens[p++], inv);
        double asteroidMass = double.Parse(tokens[p++], inv);
        double t0 = double.Parse(tokens[p++], inv);

        var bodies = new List<Body>();
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tokens[p++];
            b.Mass = double.Parse(tokens[p++], inv);
            b.R = new Vec3(double.Parse(tokens[p++], inv), double.Parse(tokens[p++], inv), double.Parse(tokens[p++], inv));
            b.V = new Vec3(double.Parse(tokens[p++], inv), double.Parse(tokens[p++], inv), double.Parse(tokens[p++], inv));
            bodies.Add(b);
        }

        _ = tokens[p++]; // DE421 path, unused in this baseline solver

        var earth = bodies.FirstOrDefault(b => b.Name == "Earth");
        var asteroid = bodies.FirstOrDefault(b => b.Name == "Asteroid");
        if (earth == null || asteroid == null)
        {
            Console.WriteLine("0");
            return;
        }

        double bestT = 0.0;
        double bestDist2 = double.PositiveInfinity;
        int samples = Math.Max(80, (int)(warningDays * 8));
        for (int i = 0; i <= samples; i++)
        {
            double t = warningDays * i / samples;
            Vec3 er = earth.R + earth.V * (t * Day);
            Vec3 ar = asteroid.R + asteroid.V * (t * Day);
            double d2 = (ar - er).Norm2();
            if (d2 < bestDist2)
            {
                bestDist2 = d2;
                bestT = t;
            }
        }

        double interceptTime = Clamp(bestT - Math.Max(1.0, warningDays * 0.08), 0.0, warningDays);
        Vec3 earthAtImpact = earth.R + earth.V * (interceptTime * Day);
        Vec3 astAtImpact = asteroid.R + asteroid.V * (interceptTime * Day);
        Vec3 rel = astAtImpact - earthAtImpact;
        Vec3 relV = asteroid.V - earth.V;

        Vec3 aim = rel.Norm() > 0 ? rel.Normalized() : new Vec3(1, 0, 0);
        Vec3 along = relV.Norm() > 0 ? relV.Normalized() : aim;
        Vec3 lateral = new Vec3(-along.Y, along.X, 0);
        if (lateral.Norm() < 1e-9) lateral = new Vec3(0, -along.Z, along.Y);
        lateral = lateral.Normalized();

        double desiredMiss = 1_000_000.0 + 2_000_000.0 * Math.Min(1.0, warningDays / 3650.0);
        double timeToGrow = Math.Max(1.0, (warningDays - interceptTime) * Day);
        double requiredDvAtImpact = desiredMiss / timeToGrow;
        double availableImpactDv = dvBudget * 0.7;
        double impactDvMag = Math.Min(availableImpactDv, Math.Max(0.0, requiredDvAtImpact));
        if (impactDvMag < 1.0) impactDvMag = Math.Min(availableImpactDv, Math.Max(50.0, dvBudget * 0.2));

        double ratio = asteroidMass > 0 ? Math.Min(1.0, impactorMass / asteroidMass) : 0.0;
        double asteroidDeltaV = impactDvMag * ratio;
        double neededOffset = desiredMiss / Math.Max(1.0, (warningDays - interceptTime) * Day);
        if (asteroidDeltaV < neededOffset)
            impactDvMag = Math.Min(availableImpactDv, neededOffset / Math.Max(1e-9, ratio));

        double launchTime = Math.Max(0.0, Math.Min(warningDays, Math.Max(0.0, interceptTime - Math.Min(365.0, warningDays * 0.4))));
        double launchCost = EarthMoonLikeLaunchCost(rel.Norm(), launchTime);
        double remaining = Math.Max(0.0, dvBudget - impactDvMag);
        double correction1 = Math.Min(remaining * 0.35, Math.Max(0.0, remaining - launchCost) * 0.5);
        double correction2 = Math.Max(0.0, remaining - correction1 - impactDvMag - launchCost);
        if (launchCost + correction1 + correction2 + impactDvMag > dvBudget)
        {
            double scale = dvBudget / Math.Max(1e-9, launchCost + correction1 + correction2 + impactDvMag);
            launchCost *= scale;
            correction1 *= scale;
            correction2 *= scale;
            impactDvMag *= scale;
        }

        Vec3 launchDir = rel.Normalized();
        Vec3 corrDir = lateral;
        Vec3 impactDir = lateral;

        Console.WriteLine(4);
        Console.WriteLine(string.Format(inv, "{0:F6} {1:E12} {2:E12} {3:E12}", launchTime, launchDir.X * launchCost, launchDir.Y * launchCost, launchDir.Z * launchCost));
        Console.WriteLine(string.Format(inv, "{0:F6} {1:E12} {2:E12} {3:E12}", Math.Max(0.0, (launchTime + interceptTime) * 0.5), corrDir.X * correction1, corrDir.Y * correction1, corrDir.Z * correction1));
        Console.WriteLine(string.Format(inv, "{0:F6} {1:E12} {2:E12} {3:E12}", Math.Max(0.0, interceptTime - 1.0), corrDir.X * correction2, corrDir.Y * correction2, corrDir.Z * correction2));
        Console.WriteLine(string.Format(inv, "{0:F6} {1:E12} {2:E12} {3:E12}", interceptTime, impactDir.X * impactDvMag, impactDir.Y * impactDvMag, impactDir.Z * impactDvMag));
    }
}
