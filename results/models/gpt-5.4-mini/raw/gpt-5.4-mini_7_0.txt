using namespace std;

struct Vec3 {
    double x{}, y{}, z{};
    Vec3() = default;
    Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
    Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline Vec3 cross(const Vec3& a, const Vec3& b) {
    return {
        a.y * b.z - a.z * b.y,
        a.z * b.x - a.x * b.z,
        a.x * b.y - a.y * b.x
    };
}

static inline double dot(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static inline double norm2(const Vec3& a) { return dot(a, a); }

struct Plane {
    Vec3 n;
    double d{};
    double eval(const Vec3& p) const { return dot(n, p) + d; }
};

struct Tri {
    array<int, 3> v{};
};

struct Mesh {
    vector<Vec3> verts;
    vector<Tri> tris;
};

static constexpr double EPS = 1e-9;

static inline int sgn(double x) {
    if (x > EPS) return 1;
    if (x < -EPS) return -1;
    return 0;
}

static inline Vec3 lerp(const Vec3& a, const Vec3& b, double t) {
    return a + (b - a) * t;
}

static Plane plane_from_tri(const Vec3& a, const Vec3& b, const Vec3& c) {
    Plane p;
    p.n = cross(b - a, c - a);
    double nn = sqrt(max(1e-30, norm2(p.n)));
    p.n = p.n * (1.0 / nn);
    p.d = -dot(p.n, a);
    return p;
}

static bool point_inside_mesh(const Mesh& m, const Vec3& p) {
    Vec3 dir{1.0, 0.3141592653589793, 0.2718281828459045};
    int hits = 0;
    for (const auto& t : m.tris) {
        const Vec3& a = m.verts[t.v[0]];
        const Vec3& b = m.verts[t.v[1]];
        const Vec3& c = m.verts[t.v[2]];
        Vec3 ab = b - a, ac = c - a;
        Vec3 n = cross(ab, ac);
        double denom = dot(n, dir);
        if (fabs(denom) < EPS) continue;
        double tval = dot(n, a - p) / denom;
        if (tval <= EPS) continue;
        Vec3 q = p + dir * tval;
        Vec3 ap = q - a, bp = q - b, cp = q - c;
        Vec3 c0 = cross(ab, ap);
        Vec3 c1 = cross(c - b, bp);
        Vec3 c2 = cross(a - c, cp);
        if (dot(n, c0) >= -EPS && dot(n, c1) >= -EPS && dot(n, c2) >= -EPS) {
            ++hits;
        }
    }
    return (hits & 1) != 0;
}

struct QuantKey {
    long long x, y, z;
    bool operator==(const QuantKey& o) const { return x == o.x && y == o.y && z == o.z; }
    bool operator<(const QuantKey& o) const {
        if (x != o.x) return x < o.x;
        if (y != o.y) return y < o.y;
        return z < o.z;
    }
};

struct KeyHash {
    size_t operator()(const QuantKey& k) const noexcept {
        size_t h = std::hash<long long>{}(k.x);
        h ^= std::hash<long long>{}(k.y) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        h ^= std::hash<long long>{}(k.z) + 0x9e3779b97f4a7c15ULL + (h << 6) + (h >> 2);
        return h;
    }
};

static QuantKey quant(const Vec3& p) {
    const double S = 1e8;
    return {
        (long long)llround(p.x * S),
        (long long)llround(p.y * S),
        (long long)llround(p.z * S)
    };
}

struct EdgeKey {
    int a, b;
    bool operator<(const EdgeKey& o) const {
        if (a != o.a) return a < o.a;
        return b < o.b;
    }
};

static vector<Tri> triangulate_polygon(const vector<int>& poly) {
    vector<Tri> out;
    if (poly.size() < 3) return out;
    for (size_t i = 1; i + 1 < poly.size(); ++i) {
        out.push_back({{poly[0], poly[i], poly[i + 1]}});
    }
    return out;
}

static vector<Tri> split_triangles_by_plane(const Mesh& src, const vector<int>& tris, const Plane& pl, vector<Vec3>& verts, unordered_map<QuantKey, int, KeyHash>& vmap) {
    vector<Tri> out;
    for (int ti : tris) {
        const Tri& t = src.tris[ti];
        array<Vec3, 3> p = {src.verts[t.v[0]], src.verts[t.v[1]], src.verts[t.v[2]]};
        array<double, 3> d = {pl.eval(p[0]), pl.eval(p[1]), pl.eval(p[2])};
        array<int, 3> side = {sgn(d[0]), sgn(d[1]), sgn(d[2])};
        vector<Vec3> poly_front, poly_back;
        auto add_point = [&](vector<Vec3>& poly, const Vec3& q) {
            poly.push_back(q);
        };
        for (int i = 0; i < 3; ++i) {
            int j = (i + 1) % 3;
            const Vec3& a = p[i];
            const Vec3& b = p[j];
            double da = d[i], db = d[j];
            int sa = side[i], sb = side[j];
            if (sa >= 0) add_point(poly_front, a);
            if (sa <= 0) add_point(poly_back, a);
            if ((sa > 0 && sb < 0) || (sa < 0 && sb > 0)) {
                double t = da / (da - db);
                Vec3 q = lerp(a, b, t);
                add_point(poly_front, q);
                add_point(poly_back, q);
            }
        }
        auto emit = [&](const vector<Vec3>& poly) {
            vector<int> idx;
            idx.reserve(poly.size());
            for (const auto& q : poly) {
                QuantKey k = quant(q);
                auto it = vmap.find(k);
                if (it != vmap.end()) {
                    idx.push_back(it->second);
                } else {
                    int id = (int)verts.size();
                    verts.push_back(q);
                    vmap.emplace(k, id);
                    idx.push_back(id);
                }
            }
            auto tris2 = triangulate_polygon(idx);
            out.insert(out.end(), tris2.begin(), tris2.end());
        };
        if (poly_front.size() >= 3) emit(poly_front);
        if (poly_back.size() >= 3) emit(poly_back);
    }
    return out;
}

static Mesh union_mesh(const Mesh& A, const Mesh& B) {
    vector<Vec3> verts;
    vector<Tri> tris;
    verts.reserve(A.verts.size() + B.verts.size());
    unordered_map<QuantKey, int, KeyHash> vmap;
    vmap.reserve((A.verts.size() + B.verts.size()) * 2 + 1);

    auto add_mesh_direct = [&](const Mesh& m) {
        for (const auto& v : m.verts) {
            QuantKey k = quant(v);
            if (!vmap.count(k)) {
                vmap.emplace(k, (int)verts.size());
                verts.push_back(v);
            }
        }
    };
    add_mesh_direct(A);
    add_mesh_direct(B);

    vector<int> aidx(A.tris.size()), bidx(B.tris.size());
    for (int i = 0; i < (int)A.tris.size(); ++i) aidx[i] = i;
    for (int i = 0; i < (int)B.tris.size(); ++i) bidx[i] = i;

    vector<Plane> planes;
    planes.reserve(A.tris.size() + B.tris.size());
    for (const auto& t : A.tris) planes.push_back(plane_from_tri(A.verts[t.v[0]], A.verts[t.v[1]], A.verts[t.v[2]]));
    for (const auto& t : B.tris) planes.push_back(plane_from_tri(B.verts[t.v[0]], B.verts[t.v[1]], B.verts[t.v[2]]));

    vector<Tri> aout, bout;
    aout.reserve(A.tris.size() * 2);
    bout.reserve(B.tris.size() * 2);

    vector<Vec3> newVerts;
    unordered_map<QuantKey, int, KeyHash> newMap;
    newMap.reserve((A.verts.size() + B.verts.size()) * 4 + 1);

    auto classify_tri = [&](const Mesh& m, const Tri& t, const Mesh& other) -> bool {
        Vec3 c = (m.verts[t.v[0]] + m.verts[t.v[1]] + m.verts[t.v[2]]) * (1.0 / 3.0);
        return point_inside_mesh(other, c);
    };

    for (const auto& t : A.tris) {
        if (!classify_tri(A, t, B)) aout.push_back(t);
    }
    for (const auto& t : B.tris) {
        if (!classify_tri(B, t, A)) bout.push_back(t);
    }

    vector<Tri> all = aout;
    all.insert(all.end(), bout.begin(), bout.end());

    Mesh result;
    result.verts = verts;
    result.tris = all;
    return result;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(cin >> VA >> FA >> VB >> FB)) return 0;

    Mesh A, B;
    A.verts.resize(VA);
    A.tris.resize(FA);
    B.verts.resize(VB);
    B.tris.resize(FB);

    for (int i = 0; i < VA; ++i) cin >> A.verts[i].x >> A.verts[i].y >> A.verts[i].z;
    for (int i = 0; i < FA; ++i) cin >> A.tris[i].v[0] >> A.tris[i].v[1] >> A.tris[i].v[2];
    for (int i = 0; i < VB; ++i) cin >> B.verts[i].x >> B.verts[i].y >> B.verts[i].z;
    for (int i = 0; i < FB; ++i) cin >> B.tris[i].v[0] >> B.tris[i].v[1] >> B.tris[i].v[2];

    Mesh U = union_mesh(A, B);

    vector<Vec3> outVerts;
    vector<Tri> outTris;
    unordered_map<QuantKey, int, KeyHash> idmap;
    idmap.reserve(U.verts.size() * 2 + 1);

    auto addv = [&](const Vec3& p) {
        QuantKey k = quant(p);
        auto it = idmap.find(k);
        if (it != idmap.end()) return it->second;
        int id = (int)outVerts.size();
        outVerts.push_back(p);
        idmap.emplace(k, id);
        return id;
    };

    for (const auto& t : U.tris) {
        Tri q;
        q.v[0] = addv(U.verts[t.v[0]]);
        q.v[1] = addv(U.verts[t.v[1]]);
        q.v[2] = addv(U.verts[t.v[2]]);
        if (q.v[0] != q.v[1] && q.v[1] != q.v[2] && q.v[2] != q.v[0]) outTris.push_back(q);
    }

    cout << outVerts.size() << ' ' << outTris.size() << '\n';
    cout.setf(std::ios::fixed);
    cout.precision(10);
    for (const auto& p : outVerts) cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : outTris) cout << t.v[0] << ' ' << t.v[1] << ' ' << t.v[2] << '\n';
    return 0;
}
