use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole: usize,
    idx: usize,
    depth: f64,
    values: Vec<Option<f64>>,
    order: usize,
}

#[derive(Clone)]
struct Outlier {
    hole: String,
    idx: usize,
    property: String,
    confidence: f64,
}

fn parse_f64(tok: Option<&str>) -> Option<f64> {
    tok.and_then(|t| t.parse::<f64>().ok())
}

fn percentile(sorted: &[f64], p: f64) -> f64 {
    if sorted.is_empty() {
        return 0.0;
    }
    if sorted.len() == 1 {
        return sorted[0];
    }
    let pos = p.clamp(0.0, 1.0) * (sorted.len() - 1) as f64;
    let lo = pos.floor() as usize;
    let hi = pos.ceil() as usize;
    if lo == hi {
        sorted[lo]
    } else {
        let w = pos - lo as f64;
        sorted[lo] * (1.0 - w) + sorted[hi] * w
    }
}

fn median(sorted: &[f64]) -> f64 {
    percentile(sorted, 0.5)
}

fn robust_scale(vals: &[f64]) -> (f64, f64) {
    if vals.is_empty() {
        return (0.0, 1.0);
    }
    let mut s = vals.to_vec();
    s.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let med = median(&s);
    let q1 = percentile(&s, 0.25);
    let q3 = percentile(&s, 0.75);
    let iqr = (q3 - q1).abs();
    let mad_vals: Vec<f64> = s.iter().map(|v| (v - med).abs()).collect();
    let mut mad_sorted = mad_vals;
    mad_sorted.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let mad = median(&mad_sorted) * 1.4826;
    let scale = if mad.is_finite() && mad > 1e-12 { mad } else if iqr > 1e-12 { iqr / 1.349 } else { 1.0 };
    (med, scale.max(1e-12))
}

fn trimmed_mean_std(vals: &[f64]) -> (f64, f64) {
    if vals.is_empty() {
        return (0.0, 1.0);
    }
    let mut s = vals.to_vec();
    s.sort_by(|a, b| a.partial_cmp(b).unwrap_or(Ordering::Equal));
    let n = s.len();
    let trim = (n / 20).min(n / 2);
    let slice = &s[trim..n - trim];
    let mean = slice.iter().sum::<f64>() / slice.len() as f64;
    let var = slice.iter().map(|v| (v - mean) * (v - mean)).sum::<f64>() / slice.len().max(1) as f64;
    let std = var.sqrt().max(1e-12);
    (mean, std)
}

fn score_anomaly(v: f64, center: f64, scale: f64, count: usize) -> f64 {
    let z = (v - center).abs() / scale.max(1e-12);
    let base = if z <= 2.0 { 0.0 } else { (z - 2.0).min(20.0) / 20.0 };
    let support = ((count as f64).ln_1p() / 10.0).clamp(0.0, 0.25);
    ((base + support).min(1.0) * 1000.0).round() / 1000.0
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let num_holes: usize = match it.next().and_then(|t| t.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let num_props: usize = match it.next().and_then(|t| t.parse().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut prop_names = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        prop_names.push(it.next().unwrap_or("prop").to_string());
    }

    let mut holes: Vec<String> = Vec::with_capacity(num_holes);
    let mut samples: Vec<Sample> = Vec::new();
    let mut prop_values: Vec<Vec<f64>> = vec![Vec::new(); num_props];
    let mut hole_samples: Vec<Vec<usize>> = Vec::new();

    for hole_idx in 0..num_holes {
        let hole_id = match it.next() {
            Some(v) => v.to_string(),
            None => break,
        };
        holes.push(hole_id);
        let _start_x = parse_f64(it.next());
        let _start_y = parse_f64(it.next());
        let _start_z = parse_f64(it.next());
        let _dir_x = parse_f64(it.next());
        let _dir_y = parse_f64(it.next());
        let _dir_z = parse_f64(it.next());
        let _length = parse_f64(it.next());
        let num_samples: usize = it.next().and_then(|t| t.parse().ok()).unwrap_or(0);
        hole_samples.push(Vec::new());

        for sidx in 0..num_samples {
            let depth = parse_f64(it.next()).unwrap_or(f64::NAN);
            let mut values = Vec::with_capacity(num_props);
            for p in 0..num_props {
                let v = parse_f64(it.next());
                if let Some(x) = v {
                    if x.is_finite() {
                        prop_values[p].push(x);
                    }
                }
                values.push(v);
            }
            let idx = samples.len();
            samples.push(Sample { hole: hole_idx, idx: sidx, depth, values, order: idx });
            hole_samples[hole_idx].push(idx);
        }
    }

    let total_samples = samples.len();
    let large = total_samples > 80_000 || num_holes > 5_000;

    let mut global_center = vec![0.0; num_props];
    let mut global_scale = vec![1.0; num_props];
    let mut global_mean = vec![0.0; num_props];
    let mut global_std = vec![1.0; num_props];

    for p in 0..num_props {
        if prop_values[p].is_empty() {
            continue;
        }
        if large {
            let (m, s) = trimmed_mean_std(&prop_values[p]);
            global_mean[p] = m;
            global_std[p] = s;
        } else {
            let (c, s) = robust_scale(&prop_values[p]);
            global_center[p] = c;
            global_scale[p] = s;
        }
    }

    let mut outliers: Vec<Outlier> = Vec::new();

    if large {
        for p in 0..num_props {
            let vals = &prop_values[p];
            if vals.len() < 4 { continue; }
            let mean = global_mean[p];
            let std = global_std[p];
            for smp in &samples {
                if let Some(v) = smp.values[p] {
                    if !v.is_finite() { continue; }
                    let z = (v - mean).abs() / std;
                    if z >= 4.0 {
                        let conf = ((z - 4.0) / 8.0 + 0.5).min(1.0);
                        outliers.push(Outlier {
                            hole: holes[smp.hole].clone(),
                            idx: smp.idx + 1,
                            property: prop_names[p].clone(),
                            confidence: (conf * 1000.0).round() / 1000.0,
                        });
                    }
                }
            }
        }
    } else {
        for p in 0..num_props {
            let vals = &prop_values[p];
            if vals.len() < 4 { continue; }
            let center = global_center[p];
            let scale = global_scale[p];
            for smp in &samples {
                if let Some(v) = smp.values[p] {
                    if !v.is_finite() { continue; }
                    let z = (v - center).abs() / scale;
                    if z >= 3.5 {
                        let conf = score_anomaly(v, center, scale, vals.len());
                        outliers.push(Outlier {
                            hole: holes[smp.hole].clone(),
                            idx: smp.idx + 1,
                            property: prop_names[p].clone(),
                            confidence: conf,
                        });
                    }
                }
            }
        }
    }

    if outliers.is_empty() {
        println!("0");
        return;
    }

    outliers.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(Ordering::Equal)
        .then_with(|| a.hole.cmp(&b.hole))
        .then_with(|| a.idx.cmp(&b.idx))
        .then_with(|| a.property.cmp(&b.property)));

    println!("{}", outliers.len());
    for o in outliers {
        println!("{} {} {} {:.3}", o.hole, o.idx, o.property, o.confidence);
    }
}
