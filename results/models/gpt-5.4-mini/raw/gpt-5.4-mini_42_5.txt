OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%uint_2 = OpConstant %uint 2
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%_ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %_ptr_Input_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%_ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %_ptr_Input_arr_v3 Input
%normal = OpVariable %_ptr_Input_arr_v3 Input
%color = OpVariable %_ptr_Input_arr_v3 Input
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%_ptr_Output_gl_PerVertex = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %_ptr_Output_gl_PerVertex Output
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %_ptr_Output_v3float Output
OpDecorate %outColor Location 0

%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_float = OpTypePointer Function %float

%glsl450 = OpExtInstImport "GLSL.std.450"

%main = OpFunction %void None %fn
%label = OpLabel
%p0_ptr = OpVariable %_ptr_Function_v3float Function
%p1_ptr = OpVariable %_ptr_Function_v3float Function
%c0_ptr = OpVariable %_ptr_Function_v3float Function
%c1_ptr = OpVariable %_ptr_Function_v3float Function
%dir_ptr = OpVariable %_ptr_Function_v3float Function
%perp_ptr = OpVariable %_ptr_Function_v3float Function
%mid_ptr = OpVariable %_ptr_Function_v3float Function
%tmpf_ptr = OpVariable %_ptr_Function_float Function

%p0_acc = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0v4 = OpLoad %v4float %p0_acc
%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2
OpStore %p0_ptr %p0
%p1_acc = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1v4 = OpLoad %v4float %p1_acc
%p1 = OpVectorShuffle %v3float %p1v4 %p1v4 0 1 2
OpStore %p1_ptr %p1

%c0_acc = OpAccessChain %_ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_acc
OpStore %c0_ptr %c0
%c1_acc = OpAccessChain %_ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_acc
OpStore %c1_ptr %c1

%p1x = OpCompositeExtract %float %p1 0
%p0x = OpCompositeExtract %float %p0 0
%dx = OpFSub %float %p1x %p0x
%p1y = OpCompositeExtract %float %p1 1
%p0y = OpCompositeExtract %float %p0 1
%dy = OpFSub %float %p1y %p0y
%dir = OpCompositeConstruct %v3float %dx %dy %float_0
OpStore %dir_ptr %dir

%negdy = OpFNegate %float %dy
%perp_len2 = OpFMul %float %negdy %negdy
%dx2 = OpFMul %float %dx %dx
%perp_len2b = OpFAdd %float %perp_len2 %dx2
%perp_len = OpExtInst %float %glsl450 Sqrt %perp_len2b
%perp_norm_x = OpFDiv %float %negdy %perp_len
%perp_norm_y = OpFDiv %float %dx %perp_len
%perp = OpCompositeConstruct %v3float %perp_norm_x %perp_norm_y %float_0
OpStore %perp_ptr %perp

%dx75 = OpFMul %float %dx %float_0_75
%dy75 = OpFMul %float %dy %float_0_75
%midx = OpFAdd %float %p0x %dx75
%midy = OpFAdd %float %p0y %dy75
%mid = OpCompositeConstruct %v3float %midx %midy %float_0
OpStore %mid_ptr %mid

%sx = OpFMul %float %perp_norm_x %float_0_015
%sy = OpFMul %float %perp_norm_y %float_0_015
%a0x = OpFSub %float %p0x %sx
%a0y = OpFSub %float %p0y %sy
%a1x = OpFAdd %float %p0x %sx
%a1y = OpFAdd %float %p0y %sy
%a2x = OpFSub %float %midx %sx
%a2y = OpFSub %float %midy %sy
%a3x = OpFAdd %float %midx %sx
%a3y = OpFAdd %float %midy %sy

%emit0 = OpCompositeConstruct %v4float %a0x %a0y %float_0 %float_1
%emit1 = OpCompositeConstruct %v4float %a1x %a1y %float_0 %float_1
%emit2 = OpCompositeConstruct %v4float %a2x %a2y %float_0 %float_1
%emit3 = OpCompositeConstruct %v4float %a3x %a3y %float_0 %float_1

%gpos = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
OpStore %outColor %c0
OpStore %gpos %emit0
OpEmitVertex
OpStore %outColor %c0
OpStore %gpos %emit1
OpEmitVertex
OpStore %outColor %c0
OpStore %gpos %emit2
OpEmitVertex
OpStore %outColor %c0
OpStore %gpos %emit3
OpEmitVertex
OpEndPrimitive

%hx = OpFMul %float %perp_norm_x %float_0_05
%hy = OpFMul %float %perp_norm_y %float_0_05
%h0x = OpFAdd %float %midx %hx
%h0y = OpFAdd %float %midy %hy
%h1x = OpFSub %float %midx %hx
%h1y = OpFSub %float %midy %hy

%head0 = OpCompositeConstruct %v4float %h0x %h0y %float_0 %float_1
%head1 = OpCompositeConstruct %v4float %h1x %h1y %float_0 %float_1
%tip = OpCompositeConstruct %v4float %p1x %p1y %float_0 %float_1

OpStore %outColor %c1
OpStore %gpos %head0
OpEmitVertex
OpStore %outColor %c1
OpStore %gpos %head1
OpEmitVertex
OpStore %outColor %c1
OpStore %gpos %tip
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
