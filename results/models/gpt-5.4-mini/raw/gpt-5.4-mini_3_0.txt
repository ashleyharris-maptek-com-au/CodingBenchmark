use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone)]
struct Edge {
    u: usize,
    v: usize,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut deg = vec![0usize; n];
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        edges.push(Edge { u, v });
        if u < n {
            adj[u].push(v);
            deg[u] += 1;
        }
        if v < n && v != u {
            adj[v].push(u);
            deg[v] += 1;
        }
    }

    let mut comp_id = vec![usize::MAX; n];
    let mut comps: Vec<Vec<usize>> = Vec::new();
    let mut q = VecDeque::new();
    for s in 0..n {
        if comp_id[s] != usize::MAX {
            continue;
        }
        let cid = comps.len();
        comp_id[s] = cid;
        q.push_back(s);
        let mut nodes = Vec::new();
        while let Some(u) = q.pop_front() {
            nodes.push(u);
            for &v in &adj[u] {
                if v < n && comp_id[v] == usize::MAX {
                    comp_id[v] = cid;
                    q.push_back(v);
                }
            }
        }
        comps.push(nodes);
    }

    let mut comp_edges: Vec<usize> = vec![0; comps.len()];
    for e in &edges {
        if e.u < n && e.v < n {
            let cu = comp_id[e.u];
            let cv = comp_id[e.v];
            if cu == cv {
                comp_edges[cu] += 1;
            }
        }
    }

    let mut positions = vec![(0.0f64, 0.0f64); n];
    let mut comp_order: Vec<usize> = (0..comps.len()).collect();
    comp_order.sort_by_key(|&cid| std::cmp::Reverse(comps[cid].len()));

    let max_per_row = ((comps.len() as f64).sqrt().ceil() as usize).max(1);
    let mut grid_idx = 0usize;
    let mut row = 0usize;
    let mut col = 0usize;

    for cid in comp_order {
        let nodes = &comps[cid];
        let k = nodes.len();
        let is_tree_like = comp_edges[cid] <= k.saturating_sub(1) + k / 10;
        let is_small = k <= 24;
        let radius = if is_small {
            6.0 + (k as f64).sqrt() * 2.0
        } else if is_tree_like {
            8.0 + (k as f64).sqrt() * 3.0
        } else {
            10.0 + (k as f64).sqrt() * 4.0
        };

        let cx = (col as f64) * 120.0;
        let cy = (row as f64) * 120.0;

        if k == 1 {
            positions[nodes[0]] = (cx, cy);
        } else if is_small {
            let mut order = nodes.clone();
            order.sort_by_key(|&u| std::cmp::Reverse(deg[u]));
            for (i, &u) in order.iter().enumerate() {
                let a = 2.0 * PI * (i as f64) / (k as f64);
                positions[u] = (cx + radius * a.cos(), cy + radius * a.sin());
            }
        } else if is_tree_like {
            let root = nodes.iter().copied().max_by_key(|&u| deg[u]).unwrap();
            let mut parent = vec![usize::MAX; n];
            let mut order = Vec::with_capacity(k);
            let mut qq = VecDeque::new();
            parent[root] = root;
            qq.push_back(root);
            while let Some(u) = qq.pop_front() {
                order.push(u);
                for &v in &adj[u] {
                    if v < n && comp_id[v] == cid && parent[v] == usize::MAX {
                        parent[v] = u;
                        qq.push_back(v);
                    }
                }
            }
            let mut children: Vec<Vec<usize>> = vec![Vec::new(); n];
            for &u in &order {
                if parent[u] != u && parent[u] != usize::MAX {
                    children[parent[u]].push(u);
                }
            }

            let mut subtree = vec![1usize; n];
            for &u in order.iter().rev() {
                let mut sum = 1usize;
                for &ch in &children[u] {
                    sum += subtree[ch];
                }
                subtree[u] = sum;
            }

            fn place_tree(
                u: usize,
                d: usize,
                a0: f64,
                a1: f64,
                cx: f64,
                cy: f64,
                positions: &mut [(f64, f64)],
                children: &Vec<Vec<usize>>,
                subtree: &Vec<usize>,
                radius_step: f64,
            ) {
                let ang = (a0 + a1) * 0.5;
                let r = radius_step * (d as f64);
                positions[u] = (cx + r * ang.cos(), cy + r * ang.sin());
                let mut cur = a0;
                for &ch in &children[u] {
                    let span = (a1 - a0) * (subtree[ch] as f64) / (subtree[u].saturating_sub(1).max(1) as f64);
                    place_tree(
                        ch,
                        d + 1,
                        cur,
                        cur + span,
                        cx,
                        cy,
                        positions,
                        children,
                        subtree,
                        radius_step,
                    );
                    cur += span;
                }
            }

            place_tree(
                root,
                0,
                0.0,
                2.0 * PI,
                cx,
                cy,
                &mut positions,
                &children,
                &subtree,
                radius / 4.0 + 1.0,
            );
        } else {
            let root = nodes.iter().copied().max_by_key(|&u| deg[u]).unwrap();
            let mut dist = vec![usize::MAX; n];
            let mut bfs_order = Vec::with_capacity(k);
            let mut qq = VecDeque::new();
            dist[root] = 0;
            qq.push_back(root);
            while let Some(u) = qq.pop_front() {
                bfs_order.push(u);
                for &v in &adj[u] {
                    if v < n && comp_id[v] == cid && dist[v] == usize::MAX {
                        dist[v] = dist[u] + 1;
                        qq.push_back(v);
                    }
                }
            }
            bfs_order.sort_by_key(|&u| (dist[u], std::cmp::Reverse(deg[u]), u));
            let layers = dist.iter().filter(|&&d| d != usize::MAX).copied().max().unwrap_or(0) + 1;
            let mut per_layer: Vec<Vec<usize>> = vec![Vec::new(); layers];
            for &u in &bfs_order {
                per_layer[dist[u]].push(u);
            }
            for (layer, nodes_l) in per_layer.iter().enumerate() {
                let r = radius * (0.4 + layer as f64 / layers.max(1) as f64);
                let kk = nodes_l.len().max(1);
                for (i, &u) in nodes_l.iter().enumerate() {
                    let a = 2.0 * PI * (i as f64) / (kk as f64) + (layer as f64) * 0.21;
                    positions[u] = (cx + r * a.cos(), cy + r * a.sin());
                }
            }
        }

        col += 1;
        grid_idx += 1;
        if grid_idx >= max_per_row {
            grid_idx = 0;
            row += 1;
            col = 0;
        }
    }

    if comps.len() == 1 && n <= 2000 {
        let iterations = if n <= 100 { 60 } else { 20 };
        let area = (n as f64).sqrt().max(1.0) * 30.0;
        for _ in 0..iterations {
            let mut fx = vec![0.0f64; n];
            let mut fy = vec![0.0f64; n];

            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = positions[i].0 - positions[j].0;
                    let dy = positions[i].1 - positions[j].1;
                    let mut d2 = dx * dx + dy * dy + 1e-6;
                    if d2 > area * area * 16.0 {
                        continue;
                    }
                    let f = 40.0 / d2;
                    fx[i] += dx * f;
                    fy[i] += dy * f;
                    fx[j] -= dx * f;
                    fy[j] -= dy * f;
                }
            }

            for e in &edges {
                if e.u >= n || e.v >= n || e.u == e.v {
                    continue;
                }
                let dx = positions[e.v].0 - positions[e.u].0;
                let dy = positions[e.v].1 - positions[e.u].1;
                fx[e.u] += dx * 0.01;
                fy[e.u] += dy * 0.01;
                fx[e.v] -= dx * 0.01;
                fy[e.v] -= dy * 0.01;
            }

            let step = 0.02;
            for i in 0..n {
                positions[i].0 += fx[i] * step;
                positions[i].1 += fy[i] * step;
            }
        }
    }

    let mut out = String::new();
    for i in 0..n {
        out.push_str(&format!("{:.6} {:.6}\n", positions[i].0, positions[i].1));
    }
    print!("{}", out);
}
