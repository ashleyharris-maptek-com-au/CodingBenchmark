use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn new(x: f64, y: f64) -> Self { Self { x, y } }
    fn add(self, o: Pt) -> Pt { Pt::new(self.x + o.x, self.y + o.y) }
    fn sub(self, o: Pt) -> Pt { Pt::new(self.x - o.x, self.y - o.y) }
    fn dot(self, o: Pt) -> f64 { self.x * o.x + self.y * o.y }
    fn cross(self, o: Pt) -> f64 { self.x * o.y - self.y * o.x }
}

fn area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.cross(b);
    }
    s * 0.5
}

fn centroid(poly: &[Pt]) -> Pt {
    let mut a = 0.0;
    let mut cx = 0.0;
    let mut cy = 0.0;
    for i in 0..poly.len() {
        let p = poly[i];
        let q = poly[(i + 1) % poly.len()];
        let c = p.cross(q);
        a += c;
        cx += (p.x + q.x) * c;
        cy += (p.y + q.y) * c;
    }
    if a.abs() < 1e-12 {
        return Pt::new(0.0, 0.0);
    }
    let k = 1.0 / (3.0 * a);
    Pt::new(cx * k, cy * k)
}

fn rotate(poly: &[Pt], ang: f64) -> Vec<Pt> {
    let c = ang.cos();
    let s = ang.sin();
    poly.iter().map(|p| Pt::new(p.x * c - p.y * s, p.x * s + p.y * c)).collect()
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        miny = miny.min(p.y);
        maxx = maxx.max(p.x);
        maxy = maxy.max(p.y);
    }
    (minx, miny, maxx, maxy)
}

fn point_in_poly(poly: &[Pt], p: Pt) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        let intersect = ((a.y > p.y) != (b.y > p.y)) &&
            (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 1e-18) + a.x);
        if intersect { inside = !inside; }
    }
    inside
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 { (b.sub(a)).cross(c.sub(a)) }

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    let eps = 1e-9;
    p.x >= a.x.min(b.x) - eps && p.x <= a.x.max(b.x) + eps &&
    p.y >= a.y.min(b.y) - eps && p.y <= a.y.max(b.y) + eps &&
    orient(a, b, p).abs() <= 1e-9
}

fn segments_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() < 1e-9 && on_segment(a, b, c) { return true; }
    if o2.abs() < 1e-9 && on_segment(a, b, d) { return true; }
    if o3.abs() < 1e-9 && on_segment(c, d, a) { return true; }
    if o4.abs() < 1e-9 && on_segment(c, d, b) { return true; }
    (o1 > 0.0) != (o2 > 0.0) && (o3 > 0.0) != (o4 > 0.0)
}

fn poly_edges(poly: &[Pt]) -> impl Iterator<Item = (Pt, Pt)> + '_ {
    (0..poly.len()).map(move |i| (poly[i], poly[(i + 1) % poly.len()]))
}

fn polys_overlap(a: &[Pt], b: &[Pt]) -> bool {
    for (a1, a2) in poly_edges(a) {
        for (b1, b2) in poly_edges(b) {
            if segments_intersect(a1, a2, b1, b2) { return true; }
        }
    }
    if point_in_poly(a, b[0]) || point_in_poly(b, a[0]) { return true; }
    false
}

fn fits(piece: &[Pt], stock: &[Pt], placed: &[Vec<Pt>]) -> bool {
    for &v in piece {
        if !point_in_poly(stock, v) { return false; }
    }
    for i in 0..piece.len() {
        let a = piece[i];
        let b = piece[(i + 1) % piece.len()];
        for (s1, s2) in poly_edges(stock) {
            if segments_intersect(a, b, s1, s2) && !on_segment(a, b, s1) && !on_segment(a, b, s2) {
                return false;
            }
        }
    }
    for p in placed {
        if polys_overlap(piece, p) { return false; }
    }
    true
}

fn translate(poly: &[Pt], dx: f64, dy: f64) -> Vec<Pt> {
    poly.iter().map(|p| Pt::new(p.x + dx, p.y + dy)).collect()
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock.push(Pt::new(x, y));
    }
    if area(&stock) < 0.0 { stock.reverse(); }

    let mut pieces: Vec<(usize, Vec<Pt>, f64, Pt)> = Vec::with_capacity(np);
    for idx in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut poly = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            poly.push(Pt::new(x, y));
        }
        if area(&poly) < 0.0 { poly.reverse(); }
        let a = area(&poly).abs();
        let c = centroid(&poly);
        pieces.push((idx, poly, a, c));
    }

    pieces.sort_by(|a, b| b.2.partial_cmp(&a.2).unwrap_or(Ordering::Equal));

    let stock_center = centroid(&stock);
    let (sminx, sminy, smaxx, smaxy) = bbox(&stock);
    let stock_w = smaxx - sminx;
    let stock_h = smaxy - sminy;

    let mut placements: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];
    let mut used_stocks: Vec<Vec<Vec<Pt>>> = vec![Vec::new()];

    let angles: Vec<f64> = (0..24).map(|k| (k as f64) * PI / 12.0).collect();

    for (orig_idx, poly, _a, c) in pieces {
        let mut placed = false;
        let mut best: Option<(usize, f64, f64, f64, Vec<Pt>)> = None;
        for &ang in &angles {
            let r = rotate(&poly, ang);
            let rc = rotate(&[c], ang)[0];
            let rr = r.iter().map(|p| p.sub(rc)).collect::<Vec<_>>();
            let (pminx, pminy, pmaxx, pmaxy) = bbox(&rr);
            let candidate_offsets = [
                Pt::new(stock_center.x - pminx - (pmaxx - pminx) * 0.5, stock_center.y - pminy - (pmaxy - pminy) * 0.5),
                Pt::new(sminx - pminx, sminy - pminy),
                Pt::new(smaxx - pmaxx, sminy - pminy),
                Pt::new(sminx - pminx, smaxy - pmaxy),
                Pt::new(smaxx - pmaxx, smaxy - pmaxy),
            ];
            for off in candidate_offsets {
                let t = translate(&rr, off.x, off.y);
                if !fits(&t, &stock, &used_stocks[0]) { continue; }
                best = Some((0, off.x, off.y, ang, t));
                placed = true;
                break;
            }
            if placed { break; }
        }

        if let Some((si, x, y, ang, tpoly)) = best {
            used_stocks[si].push(tpoly);
            placements[orig_idx] = (si, x, y, ang);
        } else {
            let mut found = None;
            'outer: for (si, placed_polys) in used_stocks.iter().enumerate() {
                for &ang in &angles {
                    let r = rotate(&poly, ang);
                    let rc = rotate(&[c], ang)[0];
                    let rr = r.iter().map(|p| p.sub(rc)).collect::<Vec<_>>();
                    let (pminx, pminy, pmaxx, pmaxy) = bbox(&rr);
                    let candidate_offsets = [
                        Pt::new(stock_center.x - pminx - (pmaxx - pminx) * 0.5, stock_center.y - pminy - (pmaxy - pminy) * 0.5),
                        Pt::new(sminx - pminx, sminy - pminy),
                        Pt::new(smaxx - pmaxx, sminy - pminy),
                        Pt::new(sminx - pminx, smaxy - pmaxy),
                        Pt::new(smaxx - pmaxx, smaxy - pmaxy),
                    ];
                    for off in candidate_offsets {
                        let t = translate(&rr, off.x, off.y);
                        if fits(&t, &stock, placed_polys) {
                            found = Some((si, off.x, off.y, ang, t));
                            break 'outer;
                        }
                    }
                }
            }
            if let Some((si, x, y, ang, tpoly)) = found {
                used_stocks[si].push(tpoly);
                placements[orig_idx] = (si, x, y, ang);
            } else {
                let si = used_stocks.len();
                used_stocks.push(Vec::new());
                let ang = 0.0;
                let r = rotate(&poly, ang);
                let rc = rotate(&[c], ang)[0];
                let rr = r.iter().map(|p| p.sub(rc)).collect::<Vec<_>>();
                let (pminx, pminy, _, _) = bbox(&rr);
                let off = Pt::new(sminx - pminx, sminy - pminy);
                let t = translate(&rr, off.x, off.y);
                used_stocks[si].push(t);
                placements[orig_idx] = (si, off.x, off.y, ang);
            }
        }
    }

    println!("{}", used_stocks.len());
    for (si, x, y, ang) in placements {
        println!("{} {} {} {}", si, x, y, ang);
    }
}
