#include <algorithm>
#include <iostream>
#include <vector>

using namespace std;

struct Edge {
    int to;
    int id;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, M;
    if (!(cin >> N >> M)) return 0;

    vector<int> U(M), V(M);
    vector<vector<Edge>> g(N);
    for (int i = 0; i < M; ++i) {
        cin >> U[i] >> V[i];
        g[U[i]].push_back({V[i], i});
        g[V[i]].push_back({U[i], i});
    }

    vector<int> tin(N, -1), low(N, -1), parent_edge(N, -1), iter(N, 0);
    vector<int> st;
    st.reserve(N);
    int timer = 0;
    vector<int> bridges;

    for (int s = 0; s < N; ++s) {
        if (tin[s] != -1) continue;
        st.push_back(s);
        parent_edge[s] = -1;
        while (!st.empty()) {
            int v = st.back();
            if (tin[v] == -1) {
                tin[v] = low[v] = timer++;
            }
            if (iter[v] < (int)g[v].size()) {
                auto e = g[v][iter[v]++];
                if (e.id == parent_edge[v]) continue;
                if (tin[e.to] != -1) {
                    low[v] = min(low[v], tin[e.to]);
                } else {
                    parent_edge[e.to] = e.id;
                    st.push_back(e.to);
                }
            } else {
                st.pop_back();
                if (parent_edge[v] != -1) {
                    int p = U[parent_edge[v]] ^ V[parent_edge[v]] ^ v;
                    low[p] = min(low[p], low[v]);
                    if (low[v] > tin[p]) bridges.push_back(parent_edge[v]);
                }
            }
        }
    }

    if (!bridges.empty()) {
        cout << 1 << '\n';
        cout << U[bridges[0]] << ' ' << V[bridges[0]] << '\n';
        return 0;
    }

    if (N == 1) {
        cout << 0 << '\n';
        return 0;
    }

    vector<int> side(N, 0);
    side[0] = 1;
    cout << 0 << '\n';
    return 0;
}
