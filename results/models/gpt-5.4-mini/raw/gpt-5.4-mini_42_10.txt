%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0

%c2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1_ptr
%c2 = OpLoad %v3float %c2_ptr

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2
%p3_xyz = OpVectorShuffle %v3float %p3 %p3 0 1 2

%start_dir = OpFSub %v3float %p2_xyz %p0_xyz
%end_dir = OpFSub %v3float %p3_xyz %p1_xyz
%tangent_sum = OpFAdd %v3float %start_dir %end_dir
%tan_len2 = OpDot %float %tangent_sum %tangent_sum
%tan_len2_ok = OpFOrdGreaterThan %bool %tan_len2 %float_0
%tan_len = OpExtInst %float %1 Sqrt %tan_len2
%tan_norm = OpSelect %v3float %tan_len2_ok (OpVectorTimesScalar %v3float %tangent_sum (OpFDiv %float %float_1 %tan_len)) %up

%side = OpCross %v3float %tan_norm %up
%side_len2 = OpDot %float %side %side
%side_len2_ok = OpFOrdGreaterThan %bool %side_len2 %float_0
%side_len = OpExtInst %float %1 Sqrt %side_len2
%side_norm = OpSelect %v3float %side_len2_ok (OpVectorTimesScalar %v3float %side (OpFDiv %float %float_1 %side_len)) %up
%perp = OpVectorTimesScalar %v3float %side_norm %float_02

%delta = OpFSub %v3float %p2_xyz %p1_xyz
%step0 = OpVectorTimesScalar %v3float %delta %float_0
%step1 = OpVectorTimesScalar %v3float %delta %float_025
%step2 = OpVectorTimesScalar %v3float %delta %float_05
%step3 = OpVectorTimesScalar %v3float %delta %float_075
%step4 = OpVectorTimesScalar %v3float %delta %float_1

%pos0 = OpFAdd %v3float %p1_xyz %step0
%pos1 = OpFAdd %v3float %p1_xyz %step1
%pos2 = OpFAdd %v3float %p1_xyz %step2
%pos3 = OpFAdd %v3float %p1_xyz %step3
%pos4 = OpFAdd %v3float %p1_xyz %step4

%col_delta = OpFSub %v3float %c2 %c1
%col0 = %c1
%col1 = OpFAdd %v3float %c1 (OpVectorTimesScalar %v3float %col_delta %float_025)
%col2 = OpFAdd %v3float %c1 (OpVectorTimesScalar %v3float %col_delta %float_05)
%col3 = OpFAdd %v3float %c1 (OpVectorTimesScalar %v3float %col_delta %float_075)
%col4 = %c2

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%q0a = OpFAdd %v3float %pos0 %perp
%q0b = OpFSub %v3float %pos0 %perp
%q0a4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %q0a 0) (OpCompositeExtract %float %q0a 1) (OpCompositeExtract %float %q0a 2) %float_1
%q0b4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %q0b 0) (OpCompositeExtract %float %q0b 1) (OpCompositeExtract %float %q0b 2) %float_1

OpEmitVertex

%q1a4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %q1a 0) (OpCompositeExtract %float %q1a 1) (OpCompositeExtract %float %q1a 2) %float_1
%q1b4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %q1b 0) (OpCompositeExtract %float %q1b 1) (OpCompositeExtract %float %q1b 2) %float_1

%q2b4 = OpCompositeConstruct %v4float (OpCompositeExtract %float %q2b 0) (OpCompositeExtract %float %q2b 1) (OpCompositeExtract %float %q2b 2) %float_1
OpStore %outColor %col2
