const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid;
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const towns = Array.isArray(data.towns) ? data.towns : [];
const budget = Math.max(0, data.cutBudget | 0);

const n = width * height;
const passable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    passable[y * width + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const idx = (x, y) => y * width + x;
const inBounds = (x, y) => x >= 0 && y >= 0 && x < width && y < height;

const fire = new Uint8Array(n);
const q = new Int32Array(n);
let qh = 0, qt = 0;
for (const [sx, sy] of ignitions) {
  if (!inBounds(sx, sy)) continue;
  const s = idx(sx, sy);
  if (!passable[s] || fire[s]) continue;
  fire[s] = 1;
  q[qt++] = s;
}

while (qh < qt) {
  const v = q[qh++];
  const x = v % width;
  const y = (v / width) | 0;
  if (x > 0) {
    const u = v - 1;
    if (passable[u] && !fire[u]) { fire[u] = 1; q[qt++] = u; }
  }
  if (x + 1 < width) {
    const u = v + 1;
    if (passable[u] && !fire[u]) { fire[u] = 1; q[qt++] = u; }
  }
  if (y > 0) {
    const u = v - width;
    if (passable[u] && !fire[u]) { fire[u] = 1; q[qt++] = u; }
  }
  if (y + 1 < height) {
    const u = v + width;
    if (passable[u] && !fire[u]) { fire[u] = 1; q[qt++] = u; }
  }
}

const townValue = new Float64Array(n);
const townMask = new Uint32Array(n);
let totalAtRisk = 0;
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (!t || !inBounds(t.x, t.y)) continue;
  const p = idx(t.x, t.y);
  const v = Number(t.value) || 0;
  townValue[p] += v;
  townMask[p] = 1;
  if (fire[p]) totalAtRisk += v;
}

if (budget === 0 || totalAtRisk <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const visited = new Uint8Array(n);
const dist = new Int32Array(n);
dist.fill(-1);
const dq = new Int32Array(n);
let dh = 0, dt = 0;

for (let i = 0; i < n; i++) {
  if (fire[i]) {
    visited[i] = 1;
    dist[i] = 0;
    dq[dt++] = i;
  }
}

const components = [];
while (dh < dt) {
  const start = dq[dh++];
  const comp = [start];
  const sx = start % width;
  const sy = (start / width) | 0;
  const base = dist[start];

  const push = (u) => {
    if (!passable[u] || visited[u]) return;
    visited[u] = 1;
    dist[u] = base + 1;
    dq[dt++] = u;
    comp.push(u);
  };

  if (sx > 0) push(start - 1);
  if (sx + 1 < width) push(start + 1);
  if (sy > 0) push(start - width);
  if (sy + 1 < height) push(start + width);

  components.push(comp);
}

const scores = [];
const seenCells = new Uint8Array(n);

for (const comp of components) {
  let boundaryValue = 0;
  const frontier = [];
  for (const cell of comp) {
    const x = cell % width;
    const y = (cell / width) | 0;
    let isBoundary = false;
    if (x > 0 && !fire[cell - 1] && passable[cell - 1]) isBoundary = true;
    if (x + 1 < width && !fire[cell + 1] && passable[cell + 1]) isBoundary = true;
    if (y > 0 && !fire[cell - width] && passable[cell - width]) isBoundary = true;
    if (y + 1 < height && !fire[cell + width] && passable[cell + width]) isBoundary = true;
    if (isBoundary) frontier.push(cell);
    if (townValue[cell] > 0) boundaryValue += townValue[cell];
  }
  if (boundaryValue <= 0 || frontier.length === 0) continue;
  const compSize = comp.length;
  for (const cell of frontier) {
    if (seenCells[cell]) continue;
    seenCells[cell] = 1;
    const x = cell % width;
    const y = (cell / width) | 0;
    let adjFire = 0;
    let adjSafe = 0;
    if (x > 0) { if (fire[cell - 1]) adjFire++; else if (passable[cell - 1]) adjSafe++; }
    if (x + 1 < width) { if (fire[cell + 1]) adjFire++; else if (passable[cell + 1]) adjSafe++; }
    if (y > 0) { if (fire[cell - width]) adjFire++; else if (passable[cell - width]) adjSafe++; }
    if (y + 1 < height) { if (fire[cell + width]) adjFire++; else if (passable[cell + width]) adjSafe++; }
    const localTown = townValue[cell] || 0;
    const score = (boundaryValue + localTown * 3) * (1 + adjFire) / (1 + adjSafe) / Math.sqrt(compSize + 1);
    scores.push([score, cell]);
  }
}

scores.sort((a, b) => b[0] - a[0]);

const chosen = [];
const cut = new Uint8Array(n);

for (let i = 0; i < scores.length && chosen.length < budget; i++) {
  const cell = scores[i][1];
  if (cut[cell] || fire[cell] || !passable[cell]) continue;
  cut[cell] = 1;
  chosen.push([cell % width, (cell / width) | 0]);
}

if (chosen.length < budget) {
  const extras = [];
  for (let y = 0; y < height; y++) {
    for (let x = 0; x < width; x++) {
      const p = idx(x, y);
      if (!passable[p] || fire[p] || cut[p]) continue;
      let d = 0;
      if (x > 0 && fire[p - 1]) d++;
      if (x + 1 < width && fire[p + 1]) d++;
      if (y > 0 && fire[p - width]) d++;
      if (y + 1 < height && fire[p + width]) d++;
      if (d > 0) extras.push([d, p]);
    }
  }
  extras.sort((a, b) => b[0] - a[0]);
  for (let i = 0; i < extras.length && chosen.length < budget; i++) {
    const p = extras[i][1];
    cut[p] = 1;
    chosen.push([p % width, (p / width) | 0]);
  }
}

process.stdout.write(JSON.stringify({ firebreaks: chosen }));
