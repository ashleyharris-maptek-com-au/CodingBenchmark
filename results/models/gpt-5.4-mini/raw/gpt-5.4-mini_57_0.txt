const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const width = data.width | 0;
const height = data.height | 0;
const grid = data.grid || [];
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const cutBudget = Math.max(0, data.cutBudget | 0);

const n = width * height;
const passable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < width; x++) {
    passable[y * width + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

function idx(x, y) {
  return y * width + x;
}

function inBounds(x, y) {
  return x >= 0 && x < width && y >= 0 && y < height;
}

const townByCell = new Map();
for (const t of towns) {
  if (inBounds(t.x, t.y)) townByCell.set(idx(t.x, t.y), t);
}

const fire = new Uint8Array(n);
const qx = new Int32Array(n);
const qy = new Int32Array(n);
let qh = 0, qt = 0;
for (const [x, y] of ignitions) {
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  if (!passable[id] || fire[id]) continue;
  fire[id] = 1;
  qx[qt] = x;
  qy[qt] = y;
  qt++;
}

const reachable = new Uint8Array(n);
for (let i = 0; i < qt; i++) reachable[idx(qx[i], qy[i])] = 1;
while (qh < qt) {
  const x = qx[qh];
  const y = qy[qh];
  qh++;
  const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
  for (const [dx, dy] of dirs) {
    const nx = x + dx, ny = y + dy;
    if (!inBounds(nx, ny)) continue;
    const id = idx(nx, ny);
    if (!passable[id] || reachable[id]) continue;
    reachable[id] = 1;
    qx[qt] = nx;
    qy[qt] = ny;
    qt++;
  }
}

const townCount = towns.length;
const townReachable = new Array(townCount).fill(false);
const townValue = new Array(townCount).fill(0);
for (let i = 0; i < townCount; i++) townValue[i] = towns[i].value || 0;
let totalThreatenedValue = 0;
for (let i = 0; i < townCount; i++) {
  const t = towns[i];
  if (inBounds(t.x, t.y) && reachable[idx(t.x, t.y)]) {
    townReachable[i] = true;
    totalThreatenedValue += townValue[i];
  }
}

const dist = new Int32Array(n);
dist.fill(-1);
const prev = new Int32Array(n);
prev.fill(-1);
const bfsx = new Int32Array(n);
const bfsy = new Int32Array(n);
qh = 0; qt = 0;
for (const [x, y] of ignitions) {
  if (!inBounds(x, y)) continue;
  const id = idx(x, y);
  if (!passable[id] || dist[id] !== -1) continue;
  dist[id] = 0;
  bfsx[qt] = x;
  bfsy[qt] = y;
  qt++;
}

const targets = [];
const targetSet = new Set();
while (qh < qt) {
  const x = bfsx[qh];
  const y = bfsy[qh];
  const id = idx(x, y);
  qh++;
  const t = townByCell.get(id);
  if (t && !targetSet.has(id)) {
    targetSet.add(id);
    targets.push(id);
  }
  const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
  for (const [dx, dy] of dirs) {
    const nx = x + dx, ny = y + dy;
    if (!inBounds(nx, ny)) continue;
    const nid = idx(nx, ny);
    if (!passable[nid] || dist[nid] !== -1) continue;
    dist[nid] = dist[id] + 1;
    prev[nid] = id;
    bfsx[qt] = nx;
    bfsy[qt] = ny;
    qt++;
  }
}

const candidateScore = new Map();
for (const tid of targets) {
  let cur = prev[tid];
  let steps = 0;
  while (cur !== -1 && steps < 10000) {
    steps++;
    const x = cur % width;
    const y = (cur / width) | 0;
    const cellId = cur;
    let contribution = 0;
    const neighbors = [
      cellId,
      cellId - 1,
      cellId + 1,
      cellId - width,
      cellId + width,
    ];
    for (const nid of neighbors) {
      if (nid < 0 || nid >= n) continue;
      if (!reachable[nid]) continue;
      const t = townByCell.get(nid);
      if (t) contribution += t.value || 0;
    }
    if (contribution > 0) {
      candidateScore.set(cellId, (candidateScore.get(cellId) || 0) + contribution);
    }
    cur = prev[cur];
  }
}

const candidates = [];
for (const [cellId, score] of candidateScore.entries()) {
  if (!passable[cellId]) continue;
  if (fire[cellId]) continue;
  candidates.push({ cellId, score });
}

candidates.sort((a, b) => b.score - a.score || a.cellId - b.cellId);

const chosen = [];
const cut = new Uint8Array(n);
for (const c of candidates) {
  if (chosen.length >= cutBudget) break;
  if (cut[c.cellId]) continue;
  const x = c.cellId % width;
  const y = (c.cellId / width) | 0;
  if (!passable[c.cellId]) continue;
  if (townByCell.has(c.cellId)) continue;
  cut[c.cellId] = 1;
  chosen.push([x, y]);
}

if (chosen.length === 0 && cutBudget > 0 && ignitions.length > 0) {
  const seen = new Set();
  for (const [sx, sy] of ignitions) {
    const around = [[sx+1,sy],[sx-1,sy],[sx,sy+1],[sx,sy-1]];
    for (const [x, y] of around) {
      if (chosen.length >= cutBudget) break;
      if (!inBounds(x, y)) continue;
      const id = idx(x, y);
      if (!passable[id] || fire[id] || townByCell.has(id) || seen.has(id)) continue;
      seen.add(id);
      chosen.push([x, y]);
    }
    if (chosen.length >= cutBudget) break;
  }
}

process.stdout.write(JSON.stringify({ firebreaks: chosen }));
