float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv * float2(8.0, 16.0);
    float row = floor(uv.y);
    float2 cellUv = frac(uv);

    if (fmod(row, 2.0) >= 1.0)
    {
        cellUv.x = frac(cellUv.x + 0.5);
    }

    float mortarX = step(cellUv.x, 0.05) + step(1.0 - 0.05, cellUv.x);
    float mortarY = step(cellUv.y, 0.10) + step(1.0 - 0.10, cellUv.y);
    float mortar = saturate(mortarX + mortarY);

    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = lerp(brickColor, mortarColor, mortar);

    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    return float4(baseColor * shade, 1.0);
}
