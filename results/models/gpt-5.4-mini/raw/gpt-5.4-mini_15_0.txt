use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn dot(self, other: Vec3) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }
    fn cross(self, other: Vec3) -> Vec3 {
        Vec3 {
            x: self.y * other.z - self.z * other.y,
            y: self.z * other.x - self.x * other.z,
            z: self.x * other.y - self.y * other.x,
        }
    }
    fn norm(self) -> f64 {
        self.dot(self).sqrt()
    }
    fn normalize(self) -> Vec3 {
        let n = self.norm();
        if n <= 1e-15 {
            Vec3 { x: 0.0, y: 0.0, z: 1.0 }
        } else {
            Vec3 { x: self.x / n, y: self.y / n, z: self.z / n }
        }
    }
    fn scale(self, s: f64) -> Vec3 {
        Vec3 { x: self.x * s, y: self.y * s, z: self.z * s }
    }
    fn add(self, other: Vec3) -> Vec3 {
        Vec3 { x: self.x + other.x, y: self.y + other.y, z: self.z + other.z }
    }
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    let mut a = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        a += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    0.5 * a.abs()
}

fn centroid(poly: &[Pt]) -> Pt {
    let mut cx = 0.0;
    let mut cy = 0.0;
    let mut a = 0.0;
    let n = poly.len();
    for i in 0..n {
        let j = (i + 1) % n;
        let cross = poly[i].x * poly[j].y - poly[j].x * poly[i].y;
        cx += (poly[i].x + poly[j].x) * cross;
        cy += (poly[i].y + poly[j].y) * cross;
        a += cross;
    }
    if a.abs() < 1e-12 {
        return Pt { x: poly.iter().map(|p| p.x).sum::<f64>() / n as f64, y: poly.iter().map(|p| p.y).sum::<f64>() / n as f64 };
    }
    let k = 1.0 / (3.0 * a);
    Pt { x: cx * k, y: cy * k }
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let j = (i + 1) % n;
        let a = poly[i];
        let b = poly[j];
        let intersect = ((a.y > p.y) != (b.y > p.y))
            && (p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 1e-30) + a.x);
        if intersect {
            inside = !inside;
        }
    }
    inside
}

fn bbox(poly: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for p in poly {
        minx = minx.min(p.x);
        maxx = maxx.max(p.x);
        miny = miny.min(p.y);
        maxy = maxy.max(p.y);
    }
    (minx, maxx, miny, maxy)
}

fn quat_from_to(a: Vec3, b: Vec3) -> (f64, f64, f64, f64) {
    let v1 = a.normalize();
    let v2 = b.normalize();
    let d = v1.dot(v2);
    if d < -0.999999 {
        let axis = if v1.x.abs() < 0.1 { Vec3 { x: 1.0, y: 0.0, z: 0.0 } } else { Vec3 { x: 0.0, y: 1.0, z: 0.0 } }.cross(v1).normalize();
        return (0.0, axis.x, axis.y, axis.z);
    }
    let c = v1.cross(v2);
    let s = (1.0 + d).sqrt() * 2.0;
    let invs = 1.0 / s;
    let q = (s * 0.5, c.x * invs, c.y * invs, c.z * invs);
    let n = (q.0 * q.0 + q.1 * q.1 + q.2 * q.2 + q.3 * q.3).sqrt();
    (q.0 / n, q.1 / n, q.2 / n, q.3 / n)
}

fn choose_basis(sun: Vec3) -> (Vec3, Vec3, Vec3) {
    let w = sun.normalize().scale(-1.0);
    let mut a = if w.x.abs() < 0.9 { Vec3 { x: 1.0, y: 0.0, z: 0.0 } } else { Vec3 { x: 0.0, y: 1.0, z: 0.0 } };
    let u = a.cross(w).normalize();
    a = if u.norm() < 1e-12 { Vec3 { x: 0.0, y: 0.0, z: 1.0 } } else { a };
    let v = w.cross(u).normalize();
    (u, v, w)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let mut poly = Vec::with_capacity(n);
    for _ in 0..n {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Pt { x, y });
    }
    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let sz: f64 = it.next().unwrap().parse().unwrap();
    let sun = Vec3 { x: sx, y: sy, z: sz }.normalize();

    let area = polygon_area(&poly);
    let c = centroid(&poly);
    let (minx, maxx, miny, maxy) = bbox(&poly);
    let diag = ((maxx - minx).hypot(maxy - miny)).max(1.0);
    let strategy_small = n <= 12 || area < diag * diag * 0.5;

    let mut out: Vec<(Vec3, (f64, f64, f64, f64), f64)> = Vec::new();

    if strategy_small {
        let (u, v, w) = choose_basis(sun);
        let layers = if n <= 6 { 2 } else { 3 };
        let mut centers = Vec::new();
        centers.push(c);
        for i in 0..layers {
            let ang = 2.0 * PI * (i as f64) / (layers as f64);
            let r = diag * 0.18 * (1.0 + i as f64 * 0.15);
            centers.push(Pt { x: c.x + r * ang.cos(), y: c.y + r * ang.sin() });
        }
        let base_scale = (diag / (3.0 + n as f64 * 0.15)).clamp(0.25, 10.0);
        for (idx, p2) in centers.into_iter().enumerate() {
            let h = 1.5 + idx as f64 * 0.9;
            let pos = Vec3 { x: p2.x, y: p2.y, z: h };
            let q = quat_from_to(Vec3 { x: 0.0, y: 0.0, z: 1.0 }, w);
            let scale = (base_scale * (1.0 - 0.12 * idx as f64)).clamp(0.2, 10.0);
            let shifted = pos.add(u.scale(0.0)).add(v.scale(0.0));
            out.push((shifted, q, scale));
        }
    } else {
        let (_, _, w) = choose_basis(sun);
        let q = quat_from_to(Vec3 { x: 0.0, y: 0.0, z: 1.0 }, w);
        let max_dim = (maxx - minx).max(maxy - miny);
        let step = (max_dim / 3.0).max(1.0);
        let mut y = miny + step * 0.5;
        while y <= maxy + 1e-9 {
            let mut x = minx + step * 0.5;
            while x <= maxx + 1e-9 {
                let p = Pt { x, y };
                if point_in_poly(p, &poly) {
                    let pos = Vec3 { x, y, z: 1.0 + out.len() as f64 * 0.2 };
                    out.push((pos, q, (step * 0.35).clamp(0.2, 10.0)));
                    if out.len() >= 64 {
                        break;
                    }
                }
                x += step;
            }
            if out.len() >= 64 {
                break;
            }
            y += step;
        }
        if out.is_empty() {
            out.push((Vec3 { x: c.x, y: c.y, z: 1.0 }, q, (diag * 0.25).clamp(0.2, 10.0)));
        }
    }

    println!("{}", out.len());
    for (p, q, s) in out {
        println!("{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}", p.x, p.y, p.z, q.0, q.1, q.2, q.3, s);
    }
}
