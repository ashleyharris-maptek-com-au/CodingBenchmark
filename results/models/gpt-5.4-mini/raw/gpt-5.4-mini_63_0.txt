declare const require: any;
const fs = require('fs');

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };
type InputData = { grid: string[]; robots: Robot[]; jobs: Job[] };

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const grid = data.grid;
const h = grid.length;
const w = h > 0 ? grid[0].length : 0;

const robots = [...data.robots].sort((a, b) => a.id - b.id);
const jobs = [...data.jobs].sort((a, b) => b.priority - a.priority || a.id - b.id);

const cellCount = h * w;
const isOpen = new Uint8Array(cellCount);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    isOpen[y * w + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

function idx(x: number, y: number): number {
  return y * w + x;
}

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < w && y >= 0 && y < h;
}

function buildDistances(sources: { x: number; y: number }[]): Int32Array[] {
  const dists: Int32Array[] = [];
  const qx = new Int32Array(cellCount);
  const qy = new Int32Array(cellCount);
  const dirs = [1, 0, -1, 0, 1];

  for (const s of sources) {
    const dist = new Int32Array(cellCount);
    dist.fill(-1);
    if (!inBounds(s.x, s.y) || !isOpen[idx(s.x, s.y)]) {
      dists.push(dist);
      continue;
    }
    let head = 0, tail = 0;
    const start = idx(s.x, s.y);
    dist[start] = 0;
    qx[tail] = s.x;
    qy[tail] = s.y;
    tail++;

    while (head < tail) {
      const x = qx[head];
      const y = qy[head];
      head++;
      const base = idx(x, y);
      const nd = dist[base] + 1;
      for (let k = 0; k < 4; k++) {
        const nx = x + dirs[k];
        const ny = y + dirs[k + 1];
        if (!inBounds(nx, ny)) continue;
        const ni = idx(nx, ny);
        if (!isOpen[ni] || dist[ni] !== -1) continue;
        dist[ni] = nd;
        qx[tail] = nx;
        qy[tail] = ny;
        tail++;
      }
    }
    dists.push(dist);
  }
  return dists;
}

const robotDists = buildDistances(robots);
const jobDists = buildDistances(jobs);

const assignments: number[][] = Array.from({ length: robots.length }, () => []);
const robotState = robots.map((r) => ({ x: r.x, y: r.y, lastPriority: Number.POSITIVE_INFINITY, load: 0 }));

function tieBetter(aRobot: number, aCost: number, aJob: Job, bRobot: number, bCost: number, bJob: Job): boolean {
  if (aCost !== bCost) return aCost < bCost;
  if (aJob.priority !== bJob.priority) return aJob.priority > bJob.priority;
  if (robotState[aRobot].load !== robotState[bRobot].load) return robotState[aRobot].load < robotState[bRobot].load;
  return aJob.id < bJob.id;
}

for (const job of jobs) {
  let bestRobot = 0;
  let bestCost = Number.POSITIVE_INFINITY;
  for (let r = 0; r < robots.length; r++) {
    const dist = robotDists[r][idx(job.x, job.y)];
    if (dist < 0) continue;
    const cost = dist + robotState[r].load * 0.001;
    if (bestCost === Number.POSITIVE_INFINITY || cost < bestCost || (cost === bestCost && tieBetter(r, cost, job, bestRobot, bestCost, job))) {
      bestRobot = r;
      bestCost = cost;
    }
  }
  assignments[bestRobot].push(job.id);
  robotState[bestRobot].load += 1;
  robotState[bestRobot].x = job.x;
  robotState[bestRobot].y = job.y;
  robotState[bestRobot].lastPriority = job.priority;
}

// Local refinement: attempt to reduce travel by swapping neighboring job orders per robot
for (let r = 0; r < assignments.length; r++) {
  const list = assignments[r];
  if (list.length <= 1) continue;
  const jobMap = new Map<number, Job>();
  for (const j of data.jobs) jobMap.set(j.id, j);
  list.sort((a, b) => {
    const ja = jobMap.get(a)!;
    const jb = jobMap.get(b)!;
    return jb.priority - ja.priority || a - b;
  });
}

process.stdout.write(JSON.stringify({ assignments }));
