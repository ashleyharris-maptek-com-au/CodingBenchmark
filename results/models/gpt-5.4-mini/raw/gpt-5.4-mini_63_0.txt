declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Robot = { id: number; x: number; y: number };
type Job = { id: number; x: number; y: number; priority: number };

const grid: string[] = data.grid ?? [];
const robotsInput: Robot[] = data.robots ?? [];
const jobsInput: Job[] = data.jobs ?? [];

const height = grid.length;
const width = height > 0 ? grid[0].length : 0;
const passable = (x: number, y: number): boolean =>
  y >= 0 && y < height && x >= 0 && x < width && grid[y][x] === '.';

const robotCount = robotsInput.length;
const assignments: number[][] = Array.from({ length: robotCount }, () => []);

if (jobsInput.length === 0 || robotCount === 0) {
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const robots = robotsInput.map((r, index) => ({ ...r, index }));
const jobs = jobsInput.map((j, index) => ({ ...j, index }));

function bfsFrom(startX: number, startY: number): Int32Array {
  const dist = new Int32Array(width * height);
  dist.fill(-1);
  if (!passable(startX, startY)) return dist;
  const queueX = new Int32Array(width * height);
  const queueY = new Int32Array(width * height);
  let head = 0;
  let tail = 0;
  const startIdx = startY * width + startX;
  dist[startIdx] = 0;
  queueX[tail] = startX;
  queueY[tail] = startY;
  tail++;
  while (head < tail) {
    const x = queueX[head];
    const y = queueY[head];
    head++;
    const base = dist[y * width + x] + 1;
    if (x > 0) {
      const nx = x - 1;
      const ny = y;
      const idx = ny * width + nx;
      if (grid[ny][nx] === '.' && dist[idx] === -1) {
        dist[idx] = base;
        queueX[tail] = nx;
        queueY[tail] = ny;
        tail++;
      }
    }
    if (x + 1 < width) {
      const nx = x + 1;
      const ny = y;
      const idx = ny * width + nx;
      if (grid[ny][nx] === '.' && dist[idx] === -1) {
        dist[idx] = base;
        queueX[tail] = nx;
        queueY[tail] = ny;
        tail++;
      }
    }
    if (y > 0) {
      const nx = x;
      const ny = y - 1;
      const idx = ny * width + nx;
      if (grid[ny][nx] === '.' && dist[idx] === -1) {
        dist[idx] = base;
        queueX[tail] = nx;
        queueY[tail] = ny;
        tail++;
      }
    }
    if (y + 1 < height) {
      const nx = x;
      const ny = y + 1;
      const idx = ny * width + nx;
      if (grid[ny][nx] === '.' && dist[idx] === -1) {
        dist[idx] = base;
        queueX[tail] = nx;
        queueY[tail] = ny;
        tail++;
      }
    }
  }
  return dist;
}

const robotDistances: Int32Array[] = robots.map(r => bfsFrom(r.x, r.y));

type Candidate = {
  robotIndex: number;
  jobIndex: number;
  dist: number;
  score: number;
};

const candidates: Candidate[] = [];
for (let ri = 0; ri < robots.length; ri++) {
  const dist = robotDistances[ri];
  for (let ji = 0; ji < jobs.length; ji++) {
    const job = jobs[ji];
    if (!passable(job.x, job.y)) continue;
    const d = dist[job.y * width + job.x];
    if (d >= 0) {
      candidates.push({
        robotIndex: ri,
        jobIndex: ji,
        dist: d,
        score: d - job.priority * 1000000,
      });
    }
  }
}

candidates.sort((a, b) =>
  a.score !== b.score ? a.score - b.score :
  a.dist !== b.dist ? a.dist - b.dist :
  jobs[a.jobIndex].priority !== jobs[b.jobIndex].priority ? jobs[b.jobIndex].priority - jobs[a.jobIndex].priority :
  robots[a.robotIndex].id - robots[b.robotIndex].id ||
  jobs[a.jobIndex].id - jobs[b.jobIndex].id
);

const assignedJob = new Array<boolean>(jobs.length).fill(false);
const robotLoad = new Int32Array(robots.length);

for (const cand of candidates) {
  if (!assignedJob[cand.jobIndex]) {
    assignedJob[cand.jobIndex] = true;
    assignments[cand.robotIndex].push(jobs[cand.jobIndex].id);
    robotLoad[cand.robotIndex]++;
  }
}

const unassigned: number[] = [];
for (let ji = 0; ji < jobs.length; ji++) {
  if (!assignedJob[ji]) unassigned.push(ji);
}

if (unassigned.length > 0) {
  const robotsByLoad = robots.map((_, i) => i).sort((a, b) => robotLoad[a] - robotLoad[b] || robots[a].id - robots[b].id);
  for (const ji of unassigned) {
    let bestRobot = robotsByLoad[0];
    let bestDist = Number.POSITIVE_INFINITY;
    const job = jobs[ji];
    for (const ri of robotsByLoad) {
      const d = robotDistances[ri][job.y * width + job.x];
      if (d >= 0 && d < bestDist) {
        bestDist = d;
        bestRobot = ri;
      }
    }
    assignments[bestRobot].push(job.id);
    robotLoad[bestRobot]++;
  }
}

for (const arr of assignments) arr.sort((a, b) => a - b);

process.stdout.write(JSON.stringify({ assignments }));
