#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) {
        return;
    }

    uint base = i * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    float sepRadius2 = radii.y * radii.y;
    float alignRadius2 = radii.z * radii.z;
    float cohRadius2 = radii.w * radii.w;

    for (uint j = 0u; j < N; ++j) {
        if (j == i) {
            continue;
        }

        uint jBase = j * 2u;
        vec3 otherPos = inData[jBase].xyz;
        vec3 offset = otherPos - pos;
        float dist2 = dot(offset, offset);

        if (dist2 <= 0.0) {
            continue;
        }

        if (dist2 < sepRadius2) {
            float dist = sqrt(dist2);
            sepForce -= offset / dist;
            sepCount++;
        }

        if (dist2 < alignRadius2) {
            alignForce += inData[jBase + 1u].xyz;
            alignCount++;
        }

        if (dist2 < cohRadius2) {
            cohForce += otherPos;
            cohCount++;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        sepForce /= float(sepCount);
        accel += normalize(sepForce) * weights.x;
    }

    if (alignCount > 0u) {
        vec3 avgVel = alignForce / float(alignCount);
        accel += (avgVel - vel) * weights.y;
    }

    if (cohCount > 0u) {
        vec3 center = cohForce / float(cohCount);
        accel += (center - pos) * weights.z;
    }

    float halfBound = bounds.x * 0.5;
    float limit = bounds.x * 0.4;
    vec3 boundaryAccel = vec3(0.0);

    if (pos.x > limit) boundaryAccel.x -= bounds.y * ((pos.x - limit) / halfBound);
    if (pos.x < -limit) boundaryAccel.x += bounds.y * ((-limit - pos.x) / halfBound);
    if (pos.y > limit) boundaryAccel.y -= bounds.y * ((pos.y - limit) / halfBound);
    if (pos.y < -limit) boundaryAccel.y += bounds.y * ((-limit - pos.y) / halfBound);
    if (pos.z > limit) boundaryAccel.z -= bounds.y * ((pos.z - limit) / halfBound);
    if (pos.z < -limit) boundaryAccel.z += bounds.y * ((-limit - pos.z) / halfBound);

    accel += boundaryAccel;

    vel += accel * radii.x;

    float speed2 = dot(vel, vel);
    float maxSpeed = weights.w;
    if (speed2 > maxSpeed * maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    pos += vel * radii.x;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
