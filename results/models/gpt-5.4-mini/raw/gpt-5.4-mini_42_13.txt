OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpSource GLSL 450

%void = OpTypeVoid
%functype = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%uint_6 = OpConstant %uint 6

%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_half = OpConstant %float 0.5
%float_thickness = OpConstant %float 0.008

%v3_zero = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v3_half = OpConstantComposite %v3float %float_half %float_half %float_half
%v4_w1 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_arr_pv_in = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_arr_pv_in Input

%arr_v3_6 = OpTypeArray %v3float %uint_6
%ptr_arr_v3_in = OpTypePointer Input %arr_v3_6
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
%worldPos = OpVariable %ptr_arr_v3_in Input
%normal = OpVariable %ptr_arr_v3_in Input
%color = OpVariable %ptr_arr_v3_in Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
OpDecorate %outColor Location 0
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %functype
%entry = OpLabel

%pos0_ptr = OpAccessChain %ptr_v4 %gl_in %int_0 %int_0
%pos2_ptr = OpAccessChain %ptr_v4 %gl_in %int_2 %int_0
%pos4_ptr = OpAccessChain %ptr_v4 %gl_in %int_4 %int_0
%col0_ptr = OpAccessChain %ptr_v3 %color %int_0
%col2_ptr = OpAccessChain %ptr_v3 %color %int_2
%col4_ptr = OpAccessChain %ptr_v3 %color %int_4

%pos0 = OpLoad %v4float %pos0_ptr
%pos2 = OpLoad %v4float %pos2_ptr
%pos4 = OpLoad %v4float %pos4_ptr
%col0 = OpLoad %v3float %col0_ptr
%col2 = OpLoad %v3float %col2_ptr
%col4 = OpLoad %v3float %col4_ptr

%col02_sum = OpFAdd %v3float %col0 %col2
%col24_sum = OpFAdd %v3float %col2 %col4
%col40_sum = OpFAdd %v3float %col4 %col0
%col02 = OpFMul %v3float %col02_sum %v3_half
%col24 = OpFMul %v3float %col24_sum %v3_half
%col40 = OpFMul %v3float %col40_sum %v3_half

%edge02 = OpFSub %v4float %pos2 %pos0
%edge24 = OpFSub %v4float %pos4 %pos2
%edge40 = OpFSub %v4float %pos0 %pos4

%edge02_xy = OpVectorShuffle %v2float %edge02 %edge02 0 1
%edge24_xy = OpVectorShuffle %v2float %edge24 %edge24 0 1
%edge40_xy = OpVectorShuffle %v2float %edge40 %edge40 0 1

%len02 = OpExtInst %float %1 GLM.length %edge02_xy
%len24 = OpExtInst %float %1 GLM.length %edge24_xy
%len40 = OpExtInst %float %1 GLM.length %edge40_xy

%invlen02 = OpFDiv %float %float_1 %len02
%invlen24 = OpFDiv %float %float_1 %len24
%invlen40 = OpFDiv %float %float_1 %len40

%dir02 = OpVectorTimesScalar %v2float %edge02_xy %invlen02
%dir24 = OpVectorTimesScalar %v2float %edge24_xy %invlen24
%dir40 = OpVectorTimesScalar %v2float %edge40_xy %invlen40

%perp02 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %dir02 1)) (OpCompositeExtract %float %dir02 0)
%perp24 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %dir24 1)) (OpCompositeExtract %float %dir24 0)
%perp40 = OpCompositeConstruct %v2float (OpFNegate %float (OpCompositeExtract %float %dir40 1)) (OpCompositeExtract %float %dir40 0)

%offset02 = OpVectorTimesScalar %v2float %perp02 %float_thickness
%offset24 = OpVectorTimesScalar %v2float %perp24 %float_thickness
%offset40 = OpVectorTimesScalar %v2float %perp40 %float_thickness

%emit_edge = OpFunctionCall %void %emit_quad %pos0 %pos2 %offset02 %col02
%emit_edge2 = OpFunctionCall %void %emit_quad %pos2 %pos4 %offset24 %col24
%emit_edge3 = OpFunctionCall %void %emit_quad %pos4 %pos0 %offset40 %col40

OpReturn
OpFunctionEnd

%emit_quad = OpFunction %void None %functype_quad
%a = OpFunctionParameter %v4float
%b = OpFunctionParameter %v4float
%o = OpFunctionParameter %v2float
%c = OpFunctionParameter %v3float
%qentry = OpLabel
%o4 = OpCompositeConstruct %v4float %o_x %o_y %float_0 %float_0
%n_o4 = OpCompositeConstruct %v4float %o_x_neg %o_y_neg %float_0 %float_0
%p0 = OpFAdd %v4float %a %o4
%p1 = OpFSub %v4float %a %o4
%p2 = OpFAdd %v4float %b %o4
%p3 = OpFSub %v4float %b %o4
OpStore %outColor %c
%gp0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gp0 %p0
OpEmitVertex
OpStore %outColor %c
%gp1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gp1 %p1
OpEmitVertex
OpStore %outColor %c
%gp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gp2 %p2
OpEmitVertex
OpStore %outColor %c
%gp3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gp3 %p3
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd

%ptr_v4 = OpTypePointer Input %v4float
%ptr_v3 = OpTypePointer Input %v3float
%functype_quad = OpTypeFunction %void %v4float %v4float %v2float %v3float
