#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main()
{
    vec2 brickUV = uv * vec2(8.0, 16.0);
    float row = floor(brickUV.y);
    vec2 cell = fract(brickUV);

    if (mod(row, 2.0) >= 1.0) {
        cell.x = fract(cell.x + 0.5);
    }

    float mortarX = 0.05;
    float mortarY = 0.10;
    float isMortar = step(cell.x, mortarX) + step(1.0 - mortarX, cell.x) + step(cell.y, mortarY) + step(1.0 - mortarY, cell.y);
    isMortar = clamp(isMortar, 0.0, 1.0);

    vec3 baseColor = mix(vec3(0.7, 0.2, 0.1), vec3(0.7), isMortar);

    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lighting = 0.4 + 0.6 * diffuse;

    outColor = vec4(baseColor * lighting, 1.0);
}
