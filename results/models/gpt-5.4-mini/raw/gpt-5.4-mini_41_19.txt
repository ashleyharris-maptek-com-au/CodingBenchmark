%void = OpTypeVoid
%func = OpTypeFunction %void

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_uniform_ubo = OpTypePointer Uniform %UBO
%ptr_uniform_v4 = OpTypePointer Uniform %v4float
%ptr_uniform_float = OpTypePointer Uniform %float
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%color = OpVariable %ptr_input_v3 Input
%outColor = OpVariable %ptr_output_v4 Output
%ubo = OpVariable %ptr_uniform_ubo Uniform
%c0 = OpConstant %float 0
%c1 = OpConstant %float 1
%c2 = OpConstant %float 2
%c32 = OpConstant %float 32
%c01 = OpConstantComposite %v3float %c1 %c0 %c0
%cambient = OpConstantComposite %v3float %c0.1 %c0.05 %c0.05
%cwhite = OpConstantComposite %v3float %c1 %c1 %c1
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%main = OpFunction %void None %func
%entry = OpLabel
%n_raw = OpLoad %v3float %normal
%n = OpExtInst %v3float %1 Normalize %n_raw
%light4ptr = OpAccessChain %ptr_uniform_v4 %ubo %int3
%light4 = OpLoad %v4float %light4ptr
%light = OpVectorShuffle %v3float %light4 %light4 0 1 2
%light_dir_raw = OpFSub %v3float %light %worldPos
%l = OpExtInst %v3float %1 Normalize %light_dir_raw
%cam4ptr = OpAccessChain %ptr_uniform_v4 %ubo %int4
%cam4 = OpLoad %v4float %cam4ptr
%cam = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
%view_raw = OpFSub %v3float %cam %worldPos
%v = OpExtInst %v3float %1 Normalize %view_raw
%ndotl = OpDot %float %n %l
%diff = OpExtInst %float %1 FMax %ndotl %c0
%diffuse = OpVectorTimesScalar %v3float %color %diff
%neg_l = OpFNegate %v3float %l
%r = OpExtInst %v3float %1 Reflect %neg_l %n
%rdotv = OpDot %float %r %v
%spec_base = OpExtInst %float %1 FMax %rdotv %c0
%spec_pow = OpExtInst %float %1 Pow %spec_base %c32
%specular = OpVectorTimesScalar %v3float %cwhite %spec_pow
%ambient = OpVectorTimesScalar %v3float %cambient %c1
%sum1 = OpFAdd %v3float %ambient %diffuse
%sum2 = OpFAdd %v3float %sum1 %specular
%clamp0 = OpExtInst %v3float %1 FMax %sum2 %c0
%clamp1 = OpExtInst %v3float %1 FMin %clamp0 %c1
%out = OpCompositeConstruct %v4float %clamp1 %c1
OpStore %outColor %out
OpReturn
