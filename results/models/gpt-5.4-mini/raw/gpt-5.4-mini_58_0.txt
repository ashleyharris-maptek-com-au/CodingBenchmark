const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = H > 0 ? grid[0].length : 0;
const N = H * W;

function idx(x, y) {
  return y * W + x;
}

function inBounds(x, y) {
  return x >= 0 && x < W && y >= 0 && y < H;
}

function isOpen(x, y) {
  return inBounds(x, y) && grid[y].charCodeAt(x) !== 35; // '#'
}

const jobByCell = new Map();
for (const job of jobs) {
  const key = idx(job.x, job.y);
  if (!jobByCell.has(key)) jobByCell.set(key, []);
  jobByCell.get(key).push(job);
}

function bfsFromSources(sources) {
  const dist = new Int32Array(N);
  dist.fill(-1);
  const q = new Int32Array(N);
  let head = 0, tail = 0;
  for (const s of sources) {
    const id = idx(s.x, s.y);
    if (id < 0 || id >= N) continue;
    if (!isOpen(s.x, s.y)) continue;
    if (dist[id] !== -1) continue;
    dist[id] = 0;
    q[tail++] = id;
  }
  while (head < tail) {
    const cur = q[head++];
    const cy = (cur / W) | 0;
    const cx = cur - cy * W;
    const nd = dist[cur] + 1;

    let nx = cx + 1, ny = cy;
    if (nx < W && grid[ny].charCodeAt(nx) !== 35) {
      const ni = cur + 1;
      if (dist[ni] === -1) { dist[ni] = nd; q[tail++] = ni; }
    }
    nx = cx - 1; ny = cy;
    if (nx >= 0 && grid[ny].charCodeAt(nx) !== 35) {
      const ni = cur - 1;
      if (dist[ni] === -1) { dist[ni] = nd; q[tail++] = ni; }
    }
    nx = cx; ny = cy + 1;
    if (ny < H && grid[ny].charCodeAt(nx) !== 35) {
      const ni = cur + W;
      if (dist[ni] === -1) { dist[ni] = nd; q[tail++] = ni; }
    }
    ny = cy - 1;
    if (ny >= 0 && grid[ny].charCodeAt(nx) !== 35) {
      const ni = cur - W;
      if (dist[ni] === -1) { dist[ni] = nd; q[tail++] = ni; }
    }
  }
  return dist;
}

const jobCellList = [];
for (const job of jobs) jobCellList.push(job);

const robotDists = robots.map(r => bfsFromSources([r]));
const jobDists = jobs.length <= 0 ? [] : jobs.map(j => bfsFromSources([j]));

function distFromGrid(distArr, x, y) {
  if (!inBounds(x, y)) return -1;
  return distArr[idx(x, y)];
}

// Precompute each job's best robot distance and a rough difficulty score.
const jobInfos = jobs.map((job, jIndex) => {
  let best = Number.POSITIVE_INFINITY;
  for (let r = 0; r < robots.length; r++) {
    const d = distFromGrid(robotDists[r], job.x, job.y);
    if (d >= 0 && d < best) best = d;
  }
  if (!isFinite(best)) best = 1e9;
  return { job, jIndex, best };
});

jobInfos.sort((a, b) => {
  if (b.job.priority !== a.job.priority) return b.job.priority - a.job.priority;
  if (a.best !== b.best) return a.best - b.best;
  return a.jIndex - b.jIndex;
});

const assignments = Array.from({ length: robots.length }, () => []);
const robotState = robots.map((r, i) => ({
  id: r.id,
  x: r.x,
  y: r.y,
  load: 0,
  total: 0,
  index: i,
}));

function currentDistToJob(robot, job) {
  const d = distFromGrid(robotDists[robot.index], job.x, job.y);
  return d >= 0 ? d : 1e9;
}

for (const info of jobInfos) {
  let bestRobot = 0;
  let bestScore = Infinity;
  for (let r = 0; r < robotState.length; r++) {
    const robot = robotState[r];
    const d = currentDistToJob(robot, info.job);
    const score = d + robot.total * 0.15 + robot.load * 0.5 - info.job.priority * 0.01;
    if (score < bestScore || (score === bestScore && robot.index < robotState[bestRobot].index)) {
      bestScore = score;
      bestRobot = r;
    }
  }
  assignments[bestRobot].push(info.job.id);
  const robot = robotState[bestRobot];
  const d = currentDistToJob(robot, info.job);
  if (d < 1e9) robot.total += d;
  robot.load += 1;
  robot.x = info.job.x;
  robot.y = info.job.y;
}

process.stdout.write(JSON.stringify({ assignments }));
