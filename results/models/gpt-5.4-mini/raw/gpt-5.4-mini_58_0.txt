const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid || [];
const robots = data.robots || [];
const jobs = data.jobs || [];

const H = grid.length;
const W = H ? grid[0].length : 0;
const N = H * W;

function idx(x, y) {
  return y * W + x;
}

function inBounds(x, y) {
  return x >= 0 && x < W && y >= 0 && y < H;
}

const passable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y];
  for (let x = 0; x < W; x++) {
    passable[idx(x, y)] = row[x] !== '#' ? 1 : 0;
  }
}

function bfsFrom(startX, startY) {
  const dist = new Int32Array(N);
  dist.fill(-1);
  if (!inBounds(startX, startY)) return dist;
  const s = idx(startX, startY);
  if (!passable[s]) return dist;

  const q = new Int32Array(N);
  let head = 0, tail = 0;
  q[tail++] = s;
  dist[s] = 0;

  while (head < tail) {
    const v = q[head++];
    const d = dist[v] + 1;
    const x = v % W;
    const y = (v / W) | 0;

    let u;
    if (x > 0) {
      u = v - 1;
      if (passable[u] && dist[u] === -1) { dist[u] = d; q[tail++] = u; }
    }
    if (x + 1 < W) {
      u = v + 1;
      if (passable[u] && dist[u] === -1) { dist[u] = d; q[tail++] = u; }
    }
    if (y > 0) {
      u = v - W;
      if (passable[u] && dist[u] === -1) { dist[u] = d; q[tail++] = u; }
    }
    if (y + 1 < H) {
      u = v + W;
      if (passable[u] && dist[u] === -1) { dist[u] = d; q[tail++] = u; }
    }
  }
  return dist;
}

const robotCount = robots.length;
const jobCount = jobs.length;

// Precompute distances from robots and jobs only when needed.
const robotDists = new Array(robotCount);
for (let i = 0; i < robotCount; i++) {
  robotDists[i] = bfsFrom(robots[i].x, robots[i].y);
}

const jobDists = new Array(jobCount);
for (let j = 0; j < jobCount; j++) {
  jobDists[j] = bfsFrom(jobs[j].x, jobs[j].y);
}

// Sort jobs by priority descending, then by closeness to nearest robot.
const nearestRobotDist = new Int32Array(jobCount);
for (let j = 0; j < jobCount; j++) {
  let best = 1e9;
  const jd = jobDists[j];
  for (let i = 0; i < robotCount; i++) {
    const d = jd[idx(robots[i].x, robots[i].y)];
    if (d >= 0 && d < best) best = d;
  }
  nearestRobotDist[j] = best === 1e9 ? 1e9 : best;
}

const order = jobs.map((job, j) => ({
  j,
  priority: job.priority || 0,
  near: nearestRobotDist[j]
})).sort((a, b) => {
  if (b.priority !== a.priority) return b.priority - a.priority;
  if (a.near !== b.near) return a.near - b.near;
  return a.j - b.j;
});

const assignments = Array.from({ length: robotCount }, () => []);
const robotCurrent = robots.map(r => ({ x: r.x, y: r.y }));
const used = new Uint8Array(jobCount);

function distanceFromRobotToJob(robotIndex, jobIndex) {
  const r = robotCurrent[robotIndex];
  const d = jobDists[jobIndex][idx(r.x, r.y)];
  return d;
}

// Greedy insertion: each job goes to the robot for which it is cheapest now,
// with priority breaking ties by favoring higher-priority jobs.
for (const item of order) {
  let bestRobot = -1;
  let bestDist = 1e18;
  let bestLoad = 1e18;
  for (let i = 0; i < robotCount; i++) {
    const d = distanceFromRobotToJob(i, item.j);
    if (d < 0) continue;
    const load = assignments[i].length;
    if (d < bestDist || (d === bestDist && load < bestLoad)) {
      bestDist = d;
      bestLoad = load;
      bestRobot = i;
    }
  }

  if (bestRobot === -1) {
    // Fall back to the least-loaded robot if the job is unreachable from all
    // current positions. This preserves a valid partition of all jobs.
    let minLoad = 1e18;
    for (let i = 0; i < robotCount; i++) {
      if (assignments[i].length < minLoad) {
        minLoad = assignments[i].length;
        bestRobot = i;
      }
    }
  }

  assignments[bestRobot].push(jobs[item.j].id);
  used[item.j] = 1;
  const job = jobs[item.j];
  robotCurrent[bestRobot] = { x: job.x, y: job.y };
}

// Guarantee every job appears exactly once.
for (let j = 0; j < jobCount; j++) {
  if (used[j]) continue;
  assignments[0].push(jobs[j].id);
}

process.stdout.write(JSON.stringify({ assignments }));
