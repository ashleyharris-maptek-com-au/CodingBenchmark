float4 main(PSInput input) : SV_Target0
{
    float3 n = normalize(input.normal);
    float3 l = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(n, l), 0.0);
    float shade = 0.3 + 0.7 * diffuse;

    float2 cellUV = frac(input.uv * 6.0);
    float2 d = cellUV - 0.5;
    float dist = length(d);

    float3 bgColor = float3(0.1, 0.1, 0.3);
    float3 dotColor = float3(1.0, 0.8, 0.0);
    float3 baseColor = (dist <= 0.3) ? dotColor : bgColor;

    return float4(baseColor * shade, 1.0);
}
