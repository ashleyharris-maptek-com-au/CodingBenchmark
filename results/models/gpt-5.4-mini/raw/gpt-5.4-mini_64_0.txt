declare const require: any;
const fs = require('fs');

type Cell = { x: number; y: number };
type Drone = { id: string | number; x: number; y: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid: number[][] = data.probability || [];
const drones: Drone[] = data.drones || [];
const maxSteps: number = Math.max(0, data.maxStepsPerDrone | 0);

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;

function inBounds(x: number, y: number): boolean {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function isOpen(x: number, y: number): boolean {
  return inBounds(x, y) && grid[y][x] !== -1;
}

const dirs = [
  [0, 0],
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

const targets: Cell[] = [];
for (let y = 0; y < h; y++) {
  for (let x = 0; x < w; x++) {
    if (grid[y][x] > 0) targets.push({ x, y });
  }
}

targets.sort((a, b) => {
  const va = grid[a.y][a.x];
  const vb = grid[b.y][b.x];
  if (vb !== va) return vb - va;
  const da = a.x + a.y;
  const db = b.x + b.y;
  if (da !== db) return da - db;
  if (a.y !== b.y) return a.y - b.y;
  return a.x - b.x;
});

function bfsFrom(startX: number, startY: number): number[][] {
  const dist: number[][] = Array.from({ length: h }, () => Array(w).fill(-1));
  if (!isOpen(startX, startY)) return dist;
  const qx = new Int32Array(w * h);
  const qy = new Int32Array(w * h);
  let head = 0;
  let tail = 0;
  dist[startY][startX] = 0;
  qx[tail] = startX;
  qy[tail] = startY;
  tail++;
  while (head < tail) {
    const x = qx[head];
    const y = qy[head];
    head++;
    const nd = dist[y][x] + 1;
    for (let k = 1; k < 5; k++) {
      const nx = x + dirs[k][0];
      const ny = y + dirs[k][1];
      if (!isOpen(nx, ny) || dist[ny][nx] !== -1) continue;
      dist[ny][nx] = nd;
      qx[tail] = nx;
      qy[tail] = ny;
      tail++;
    }
  }
  return dist;
}

const droneCount = drones.length;
const droneDist: number[][][] = [];
for (let i = 0; i < droneCount; i++) {
  droneDist.push(bfsFrom(drones[i].x, drones[i].y));
}

const owner: number[][] = Array.from({ length: h }, () => Array(w).fill(-1));
const bestScore: number[][] = Array.from({ length: h }, () => Array(w).fill(Number.POSITIVE_INFINITY));

for (let ti = 0; ti < targets.length; ti++) {
  const t = targets[ti];
  let bestDrone = -1;
  let bestMetric = Number.POSITIVE_INFINITY;
  for (let i = 0; i < droneCount; i++) {
    const d = droneDist[i][t.y][t.x];
    if (d < 0) continue;
    const metric = d * 1000 - grid[t.y][t.x] * 10 + i;
    if (metric < bestMetric) {
      bestMetric = metric;
      bestDrone = i;
    }
  }
  if (bestDrone >= 0) {
    owner[t.y][t.x] = bestDrone;
    bestScore[t.y][t.x] = bestMetric;
  }
}

function reconstructPath(dist: number[][], startX: number, startY: number, goalX: number, goalY: number): Cell[] | null {
  if (!isOpen(goalX, goalY) || dist[goalY][goalX] < 0) return null;
  const path: Cell[] = [];
  let x = goalX;
  let y = goalY;
  while (!(x === startX && y === startY)) {
    path.push({ x, y });
    const d = dist[y][x];
    let moved = false;
    for (let k = 1; k < 5; k++) {
      const px = x + dirs[k][0];
      const py = y + dirs[k][1];
      if (inBounds(px, py) && dist[py][px] === d - 1) {
        x = px;
        y = py;
        moved = true;
        break;
      }
    }
    if (!moved) return null;
  }
  path.push({ x: startX, y: startY });
  path.reverse();
  return path;
}

function scoreCell(x: number, y: number, visited: Set<number>, droneIndex: number, curX: number, curY: number): number {
  if (!isOpen(x, y) || visited.has(y * w + x)) return -1e18;
  const value = grid[y][x];
  const dist = Math.abs(curX - x) + Math.abs(curY - y);
  const ownership = owner[y][x] === droneIndex ? 25 : 0;
  return value * 1000 + ownership - dist * 8 - (bestScore[y][x] === Number.POSITIVE_INFINITY ? 0 : bestScore[y][x] * 0.01);
}

const paths: number[][][] = [];
const cellIndices: number[] = [];
for (let y = 0; y < h; y++) {
  for (let x = 0; x < w; x++) {
    if (grid[y][x] > 0) cellIndices.push(y * w + x);
  }
}

for (let i = 0; i < droneCount; i++) {
  const start = drones[i];
  const visited = new Set<number>();
  const path: number[][] = [[start.x, start.y]];
  visited.add(start.y * w + start.x);
  let curX = start.x;
  let curY = start.y;
  let steps = 0;

  while (steps < maxSteps) {
    let bestTarget: Cell | null = null;
    let bestTargetScore = -1e18;

    const localCandidates: Cell[] = [];
    for (let c = 0; c < cellIndices.length; c++) {
      const idx = cellIndices[c];
      if (visited.has(idx)) continue;
      const x = idx % w;
      const y = (idx / w) | 0;
      if (owner[y][x] !== -1 && owner[y][x] !== i) continue;
      const d = Math.abs(curX - x) + Math.abs(curY - y);
      if (d > maxSteps - steps + 2) continue;
      localCandidates.push({ x, y });
    }

    if (localCandidates.length === 0) break;

    for (let c = 0; c < localCandidates.length; c++) {
      const t = localCandidates[c];
      const s = scoreCell(t.x, t.y, visited, i, curX, curY);
      if (s > bestTargetScore) {
        bestTargetScore = s;
        bestTarget = t;
      }
    }

    if (!bestTarget) break;

    const dist = bfsFrom(curX, curY);
    const route = reconstructPath(dist, curX, curY, bestTarget.x, bestTarget.y);
    if (!route || route.length <= 1) {
      visited.add(bestTarget.y * w + bestTarget.x);
      continue;
    }

    for (let r = 1; r < route.length && steps < maxSteps; r++) {
      const p = route[r];
      if (!isOpen(p.x, p.y)) break;
      path.push([p.x, p.y]);
      steps++;
      curX = p.x;
      curY = p.y;
      visited.add(curY * w + curX);
    }

    if (steps >= maxSteps) break;
    visited.add(bestTarget.y * w + bestTarget.x);
  }

  while (path.length < maxSteps + 1) {
    path.push([curX, curY]);
  }
  paths.push(path);
}

process.stdout.write(JSON.stringify({ paths }));
