declare const require: any;
const fs = require('fs');

type Cell = { x: number; y: number };

const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const prob: number[][] = data.probability;
const drones: { id: number; x: number; y: number }[] = data.drones;
const maxStepsPerDrone: number = data.maxStepsPerDrone;

const H = prob.length;
const W = H > 0 ? prob[0].length : 0;

const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
  [0, 0],
];

function inBounds(x: number, y: number): boolean {
  return x >= 0 && x < W && y >= 0 && y < H;
}

function idx(x: number, y: number): number {
  return y * W + x;
}

function bfsDistances(startX: number, startY: number): Int32Array {
  const dist = new Int32Array(W * H);
  dist.fill(-1);
  if (!inBounds(startX, startY) || prob[startY][startX] === -1) return dist;
  const qx = new Int32Array(W * H);
  const qy = new Int32Array(W * H);
  let qh = 0;
  let qt = 0;
  dist[idx(startX, startY)] = 0;
  qx[qt] = startX;
  qy[qt] = startY;
  qt++;
  while (qh < qt) {
    const x = qx[qh];
    const y = qy[qh];
    qh++;
    const nd = dist[idx(x, y)] + 1;
    for (let k = 0; k < 4; k++) {
      const nx = x + dirs[k][0];
      const ny = y + dirs[k][1];
      if (!inBounds(nx, ny) || prob[ny][nx] === -1) continue;
      const ni = idx(nx, ny);
      if (dist[ni] !== -1) continue;
      dist[ni] = nd;
      qx[qt] = nx;
      qy[qt] = ny;
      qt++;
    }
  }
  return dist;
}

const droneCount = drones.length;
const reach: Int32Array[] = [];
for (let i = 0; i < droneCount; i++) {
  reach.push(bfsDistances(drones[i].x, drones[i].y));
}

const assigned = new Int32Array(W * H);
assigned.fill(-1);
const visitedByDrone: boolean[][] = Array.from({ length: droneCount }, () => new Array(W * H).fill(false));

const paths: number[][][] = [];

for (let di = 0; di < droneCount; di++) {
  const d = drones[di];
  let x = d.x;
  let y = d.y;
  const path: number[][] = [[x, y]];
  const localVisited = visitedByDrone[di];
  if (inBounds(x, y) && prob[y][x] > 0) {
    localVisited[idx(x, y)] = true;
    assigned[idx(x, y)] = di;
  }

  for (let step = 0; step < maxStepsPerDrone; step++) {
    let bestX = x;
    let bestY = y;
    let bestScore = -1;
    let bestDist = 1e9;

    for (let k = 0; k < dirs.length; k++) {
      const nx = x + dirs[k][0];
      const ny = y + dirs[k][1];
      if (!inBounds(nx, ny) || prob[ny][nx] === -1) continue;
      const ni = idx(nx, ny);
      const distHere = reach[di][ni];
      if (distHere === -1 || distHere > maxStepsPerDrone) continue;

      let score = 0;
      if (prob[ny][nx] > 0 && !localVisited[ni] && assigned[ni] === -1) {
        score += prob[ny][nx] * 1000;
      }

      for (let kk = 0; kk < 4; kk++) {
        const ax = nx + dirs[kk][0];
        const ay = ny + dirs[kk][1];
        if (!inBounds(ax, ay) || prob[ay][ax] <= 0 || prob[ay][ax] === -1) continue;
        const ai = idx(ax, ay);
        if (!localVisited[ai] && assigned[ai] === -1) score += prob[ay][ax] * 50;
      }

      const manhattan = Math.abs(nx - d.x) + Math.abs(ny - d.y);
      if (score > bestScore || (score === bestScore && manhattan < bestDist)) {
        bestScore = score;
        bestDist = manhattan;
        bestX = nx;
        bestY = ny;
      }
    }

    if (bestX === x && bestY === y && bestScore <= 0) {
      path.push([x, y]);
      continue;
    }

    x = bestX;
    y = bestY;
    path.push([x, y]);

    const ci = idx(x, y);
    if (prob[y][x] > 0 && !localVisited[ci] && assigned[ci] === -1) {
      localVisited[ci] = true;
      assigned[ci] = di;
    }
  }

  paths.push(path);
}

process.stdout.write(JSON.stringify({ paths }));
