from math import atan2, copysign, exp, radians, sin, cos

_S = {
    "go_arounds": 0,
    "phase": "intercept",
    "trim": 0.0,
    "flare_bias": 0.0,
    "touchdown": False,
}


def _valid_values(state, prefix, suffix):
    vals = []
    for i in (1, 2, 3):
        vkey = f"{prefix}_{i}_{suffix}"
        valid = state.get(f"{prefix}_{i}_valid", True)
        val = state.get(vkey)
        if valid and val is not None:
            try:
                vals.append(float(val))
            except Exception:
                pass
    return vals


def _median(vals, default=0.0):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return default
    if n == 1:
        return vals[0]
    if n == 2:
        return 0.5 * (vals[0] + vals[1])
    return vals[n // 2] if n % 2 else 0.5 * (vals[n // 2 - 1] + vals[n // 2])


def _mean(vals, default=0.0):
    return sum(vals) / len(vals) if vals else default


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _blend(a, b, w):
    return a * (1.0 - w) + b * w


def _rate_limit(prev, target, max_step):
    if target > prev + max_step:
        return prev + max_step
    if target < prev - max_step:
        return prev - max_step
    return target


def _any_fire(state):
    return bool(state.get("engine_1_fire")) or bool(state.get("engine_2_fire"))


def _engine_out(state):
    e1 = bool(state.get("engine_1_running", True))
    e2 = bool(state.get("engine_2_running", True))
    return int(e1) + int(e2) < 2


def _approach_energy(state):
    alt = state.get("radio_alt_ft")
    if alt is None:
        alt = _median(_valid_values(state, "baro_alt", "ft"), 1000.0)
    ias = _median(_valid_values(state, "ias", "kt"), state.get("target_ias_kt", 140.0))
    vref = float(state.get("target_vref_kt", 137.0) or 137.0)
    return alt + max(0.0, ias - vref) * 8.0


def autopilot_step(state: dict, dt: float) -> dict:
    global _S

    gs_vals = _valid_values(state, "glideslope", "dots")
    loc_vals = _valid_values(state, "localizer", "dots")
    baro_vals = _valid_values(state, "baro_alt", "ft")
    ias_vals = _valid_values(state, "ias", "kt")
    aoa_vals = _valid_values(state, "aoa", "deg")
    gps_gs_vals = _valid_values(state, "gps", "gs_kt")
    irs_gs_vals = _valid_values(state, "irs", "gs_kt")
    irs_vs_vals = _valid_values(state, "irs", "vs_fpm")
    irs_pitch_vals = _valid_values(state, "irs", "pitch_deg")
    irs_roll_vals = _valid_values(state, "irs", "roll_deg")
    track_vals = []
    for p in ("gps", "irs"):
        for i in (1, 2, 3):
            if state.get(f"{p}_{i}_valid", True):
                t = state.get(f"{p}_{i}_track_deg")
                if t is not None:
                    track_vals.append(float(t))

    gs = _median(gs_vals, 0.0)
    loc = _median(loc_vals, 0.0)
    alt = state.get("radio_alt_ft")
    if alt is None:
        alt = _median(baro_vals, 3000.0)
    alt = float(alt)
    ias = _median(ias_vals, state.get("target_ias_kt", 140.0))
    vref = float(state.get("target_vref_kt", 137.0) or 137.0)
    target_ias = float(state.get("target_ias_kt", vref + 10.0) or (vref + 10.0))
    target_vs = float(state.get("target_vs_fpm", -700.0) or -700.0)
    hdg = float(state.get("target_hdg_deg", 0.0) or 0.0)
    track = _median(track_vals, hdg)
    pitch = _median(irs_pitch_vals, 0.0)
    roll = _median(irs_roll_vals, 0.0)
    vs = _median(irs_vs_vals, target_vs)
    gspd = _median(gps_gs_vals, _median(irs_gs_vals, ias))
    aoa = _median(aoa_vals, 5.0)

    wow = bool(state.get("weight_on_wheels"))
    on_ground = bool(state.get("on_ground"))
    dh = bool(state.get("below_decision_height"))
    dme = _mean([state.get("dme_1_nm"), state.get("dme_2_nm")], 999.0)
    engine_out = _engine_out(state)
    fire = _any_fire(state)
    unstable = False
    if abs(loc) > 1.5 and alt < 1500:
        unstable = True
    if abs(gs) > 1.2 and alt < 2000:
        unstable = True
    if abs(ias - target_ias) > max(18.0, 0.12 * target_ias) and alt < 2000:
        unstable = True
    if alt < 1000 and abs(vs) > 1400:
        unstable = True
    if fire:
        unstable = True

    if wow or on_ground or state.get("radio_alt_ft", 9999) < 1.5 and gspd < 50:
        _S["touchdown"] = True
        _S["phase"] = "rollout"
    elif alt > 1000:
        _S["phase"] = "intercept"
    elif alt > 200:
        _S["phase"] = "approach"
    else:
        _S["phase"] = "flare"

    if (_S["go_arounds"] < 10 and (fire or (alt < 500 and (abs(loc) > 2.0 or abs(gs) > 1.8)) or (alt < 250 and ias < vref - 15) or (alt < 300 and ias > target_ias + 30) or (alt < 350 and abs(roll) > 20) or (alt < 250 and abs(vs) > 1600 and not wow) or (dme < 0.8 and alt > 800))):
        _S["go_arounds"] += 1
        _S["phase"] = "goaround"
        _S["touchdown"] = False
        _S["flare_bias"] = 0.0
        return {
            "throttle_1": 1.0,
            "throttle_2": 1.0,
            "elevator": 0.6,
            "aileron": _clamp(-0.15 * loc - 0.02 * roll, -0.4, 0.4),
            "rudder": _clamp(-0.08 * loc + 0.04 * engine_out * copysign(1.0, -loc if loc != 0 else 1.0), -0.3, 0.3),
            "speedbrake": 0.0,
            "flaps": 20 if alt < 500 else 10,
            "gear": 1.0,
        }

    out = {}

    target_flaps = 0
    if alt > 3000:
        target_flaps = 10
    elif alt > 1500:
        target_flaps = 20
    elif alt > 800:
        target_flaps = 30
    else:
        target_flaps = 40
    if engine_out:
        target_flaps = max(target_flaps, 30)
    if dme < 2.0:
        target_flaps = max(target_flaps, 30)
    out["flaps"] = target_flaps
    out["gear"] = 1.0

    if alt > 50:
        out["speedbrake"] = 0.0
    else:
        out["speedbrake"] = 0.0

    if wow or on_ground:
        out["spoiler"] = 1.0
        out["autobrake"] = 1.0 if state.get("groundspeed_kt", 999) > 100 else 0.7
        out["reverse_1"] = 1.0 if state.get("engine_1_running", True) else 0.0
        out["reverse_2"] = 1.0 if state.get("engine_2_running", True) else 0.0
        out["throttle_1"] = 0.0
        out["throttle_2"] = 0.0
        out["elevator"] = -0.2
        out["aileron"] = _clamp(-0.05 * loc - 0.02 * roll, -0.3, 0.3)
        out["rudder"] = _clamp(-0.12 * loc, -0.3, 0.3)
        return out

    if _S["phase"] == "flare":
        flare = _clamp((50.0 - alt) / 50.0, 0.0, 1.0)
        _S["flare_bias"] = _blend(_S["flare_bias"], flare, 0.2)
        pitch_cmd = 0.04 * (target_ias - ias) + 0.08 * (-vs / 1000.0) + 0.15 * _S["flare_bias"]
        out["elevator"] = _clamp(pitch_cmd, -0.35, 0.5)
        out["throttle_1"] = 0.08 if engine_out else 0.12
        out["throttle_2"] = 0.08 if engine_out else 0.12
    else:
        alt_factor = _clamp((alt - 50.0) / 1200.0, 0.0, 1.0)
        gs_error = -gs if abs(gs) < 9 else 0.0
        loc_error = -loc if abs(loc) < 9 else 0.0
        speed_error = target_ias - ias
        pitch_cmd = 0.06 * gs_error + 0.02 * speed_error / 10.0 + 0.002 * (-vs)
        if _S["phase"] == "intercept":
            pitch_cmd += 0.05
        if _S["phase"] == "approach":
            pitch_cmd += 0.02
        if engine_out:
            pitch_cmd += 0.03
        if alt < 400:
            pitch_cmd += 0.03
        out["elevator"] = _clamp(pitch_cmd, -0.4, 0.35)
        bank_cmd = _clamp(0.7 * loc_error - 0.03 * roll, -1.0, 1.0)
        if abs(state.get("target_loc_dots", 0.0) or 0.0) > 0.5:
            bank_cmd += _clamp(-0.15 * float(state.get("target_loc_dots", 0.0)), -0.3, 0.3)
        out["aileron"] = _clamp(bank_cmd, -0.6, 0.6)
        rudder_cmd = _clamp(-0.15 * loc + 0.02 * (-roll) + 0.1 * (0.0 if len(track_vals) < 2 else (track_vals[-1] - track_vals[0]) / 30.0), -0.35, 0.35)
        if engine_out:
            rudder_cmd += _clamp(0.18 * (1 if state.get("engine_1_running", True) else -1), -0.18, 0.18)
        if abs(aoa) > 9.0:
            pitch_cmd -= 0.05
        out["rudder"] = _clamp(rudder_cmd, -0.4, 0.4)

        thrust_base = 0.55 if alt > 1000 else 0.45
        thrust_adj = _clamp(0.015 * speed_error + 0.0004 * (-vs), -0.25, 0.25)
        if alt < 250:
            thrust_base = 0.25
            thrust_adj = _clamp(0.02 * speed_error + 0.0002 * (-vs), -0.2, 0.15)
        if engine_out:
            if not state.get("engine_1_running", True):
                out["throttle_1"] = 0.0
                out["throttle_2"] = _clamp(thrust_base + 0.08 + thrust_adj, 0.2, 1.0)
            elif not state.get("engine_2_running", True):
                out["throttle_2"] = 0.0
                out["throttle_1"] = _clamp(thrust_base + 0.08 + thrust_adj, 0.2, 1.0)
            else:
                out["throttle_1"] = out["throttle_2"] = _clamp(thrust_base + thrust_adj, 0.2, 1.0)
        else:
            asym = 0.02 * _clamp(loc, -2.0, 2.0)
            out["throttle_1"] = _clamp(thrust_base + thrust_adj - asym, 0.0, 1.0)
            out["throttle_2"] = _clamp(thrust_base + thrust_adj + asym, 0.0, 1.0)

    trim_target = _clamp(0.0025 * (target_ias - ias) + 0.0008 * (-vs) + 0.05 * _clamp(gs, -2.0, 2.0), -0.5, 0.5)
    if alt < 60:
        trim_target += 0.08
    if engine_out:
        trim_target += 0.05
    _S["trim"] = _rate_limit(_S["trim"], trim_target, max(0.02, 0.08 * dt))
    out["elevator_trim"] = _clamp(_S["trim"], -1.0, 1.0)

    if alt > 1000:
        out["gear"] = 1.0
    elif alt < 80:
        out["gear"] = 1.0
    else:
        out["gear"] = 1.0

    if alt < 30 and vs > -200:
        out["elevator"] = _clamp(out.get("elevator", 0.0) + 0.1, -1.0, 1.0)
        out["throttle_1"] = min(out.get("throttle_1", 0.0), 0.15)
        out["throttle_2"] = min(out.get("throttle_2", 0.0), 0.15)

    if alt < 10 or wow:
        out["spoiler"] = 1.0
        out["autobrake"] = 1.0
        out["reverse_1"] = 1.0 if state.get("engine_1_running", True) else 0.0
        out["reverse_2"] = 1.0 if state.get("engine_2_running", True) else 0.0
    else:
        out["spoiler"] = 0.0
        out["autobrake"] = 0.0
        out["reverse_1"] = 0.0
        out["reverse_2"] = 0.0

    if abs(loc) > 2.5 or abs(gs) > 2.5:
        out["rudder"] = _clamp(out.get("rudder", 0.0) + 0.1 * copysign(1.0, -loc if abs(loc) > abs(gs) else -gs), -1.0, 1.0)

    if abs(ias - target_ias) > 35 and alt > 1000:
        out["speedbrake"] = 0.0

    return {k: v for k, v in out.items() if v is not None}
