import math

_state = {
    'goaround_count': 0,
    'phase': 'approach',
}


def _valid_get(state, key, default=None):
    v = state.get(key, default)
    return default if v is None else v


def _vals(state, base, suffix):
    out = []
    for i in (1, 2, 3):
        val = state.get(f'{base}_{i}_{suffix}')
        valid = state.get(f'{base}_{i}_valid', True)
        if valid and val is not None and not (isinstance(val, float) and math.isnan(val)):
            out.append(val)
    return out


def _median(vals, default=0.0):
    if not vals:
        return default
    vals = sorted(vals)
    n = len(vals)
    mid = n // 2
    return vals[mid] if n % 2 else 0.5 * (vals[mid - 1] + vals[mid])


def _avg(vals, default=0.0):
    return sum(vals) / len(vals) if vals else default


def _bool_any(state, key):
    return bool(state.get(key, False))


def _clamp(x, lo, hi):
    return max(lo, min(hi, x))


def _norm_deg_err(target, current):
    if target is None or current is None:
        return 0.0
    d = (target - current + 180.0) % 360.0 - 180.0
    return d


def autopilot_step(state: dict, dt: float) -> dict:
    global _state

    # Reset heuristic between scenarios if the aircraft is back on the runway.
    if _bool_any(state, 'on_ground') and _bool_any(state, 'weight_on_wheels') and state.get('groundspeed_kt', 999.0) < 5.0:
        _state['phase'] = 'rollout'

    # Sensor fusion.
    loc = _median(_vals(state, 'localizer', 'dots'), 0.0)
    gs = _median(_vals(state, 'glideslope', 'dots'), 0.0)
    ias = _median(_vals(state, 'ias', 'kt'), state.get('target_ias_kt', 140.0))
    tas = _median(_vals(state, 'tas', 'kt'), ias)
    aoa = _median(_vals(state, 'aoa', 'deg'), 0.0)
    baro_alt = _median(_vals(state, 'baro_alt', 'ft'), state.get('target_alt_ft', 0.0))
    radio_alt = state.get('radio_alt_ft')
    if radio_alt is None:
        radio_alt = state.get('ground_radio_alt_ft', baro_alt)
    groundspeed = state.get('groundspeed_kt', tas)
    vref = state.get('target_vref_kt', state.get('target_ias_kt', 137.0))
    target_ias = state.get('target_ias_kt', vref + 10.0)
    hdg = _median(_vals(state, 'irs', 'heading_deg'), state.get('target_hdg_deg', 0.0))
    track = _median(_vals(state, 'irs', 'track_deg'), hdg)
    target_hdg = state.get('target_hdg_deg', hdg)
    target_vs = state.get('target_vs_fpm', -700.0)

    # Derived conditions.
    engine1 = state.get('engine_1_running', True) and not state.get('engine_1_fire', False)
    engine2 = state.get('engine_2_running', True) and not state.get('engine_2_fire', False)
    engines = int(bool(engine1)) + int(bool(engine2))
    wet_or_icy = bool(state.get('runway_wet', False)) or bool(state.get('runway_icy', False))
    strong_crosswind = abs(_norm_deg_err(track, target_hdg)) > 8.0 and abs(loc) > 0.3

    unstable = False
    if radio_alt is not None and radio_alt > 500.0:
        unstable |= abs(loc) > 2.0 or abs(gs) > 2.0
        unstable |= ias > target_ias + 20.0
        unstable |= ias < max(110.0, vref - 15.0)
    if radio_alt is not None and radio_alt <= 500.0:
        unstable |= abs(loc) > 1.5 or abs(gs) > 1.5
        unstable |= abs(target_vs) > 1200.0
        unstable |= ias > target_ias + 25.0
        unstable |= ias < max(100.0, vref - 20.0)

    # Go-around for unsafe configurations or severe failures.
    if (state.get('below_decision_height') is False and radio_alt is not None and radio_alt < 200.0 and (abs(loc) > 2.0 or abs(gs) > 2.0)) or \
       (radio_alt is not None and radio_alt < 50.0 and ias < max(90.0, vref - 25.0)) or \
       (engines == 0 and radio_alt is not None and radio_alt > 200.0) or \
       (state.get('missed_approach_required', False)) or \
       unstable and radio_alt is not None and radio_alt > 100.0:
        _state['goaround_count'] += 1
        _state['phase'] = 'goaround'
        return {
            'throttle_1': 1.0,
            'throttle_2': 1.0,
            'elevator': 0.35,
            'elevator_trim': 0.15,
            'aileron': _clamp(-loc * 0.15, -0.3, 0.3),
            'rudder': _clamp((target_hdg - track) / 45.0, -0.3, 0.3),
            'flaps': 15.0,
            'gear': 1.0,
            'speedbrake': 0.0,
            'spoiler': 0.0,
            'reverse_1': 0.0,
            'reverse_2': 0.0,
            'autobrake': 0.0,
        }

    # Rollout / touchdown.
    wow = _bool_any(state, 'weight_on_wheels') or _bool_any(state, 'on_ground')
    if wow or (radio_alt is not None and radio_alt < 5.0 and groundspeed < 80.0):
        _state['phase'] = 'rollout'
        thrust_cut = 0.0
        braking = 1.0 if groundspeed > 60.0 or wet_or_icy else 0.6
        return {
            'throttle_1': thrust_cut,
            'throttle_2': thrust_cut,
            'elevator': 0.0,
            'elevator_trim': 0.0,
            'aileron': _clamp(-loc * 0.1, -0.2, 0.2),
            'rudder': _clamp((target_hdg - track) / 60.0, -0.2, 0.2),
            'flaps': 30.0,
            'gear': 1.0,
            'speedbrake': 0.0,
            'spoiler': 1.0,
            'reverse_1': 1.0 if engines >= 1 else 0.0,
            'reverse_2': 1.0 if engines >= 2 else 0.0,
            'autobrake': braking,
        }

    # Landing config.
    if radio_alt is None:
        radio_alt = max(0.0, baro_alt)

    if radio_alt > 1000.0:
        flaps = 5.0
        gear = 0.0
        trim = _clamp((vref - ias) / 100.0, -0.3, 0.3)
        throttle = _clamp(0.55 + (target_ias - ias) * 0.015, 0.25, 0.85)
    elif radio_alt > 200.0:
        flaps = 30.0
        gear = 1.0
        trim = _clamp((vref - ias) / 120.0, -0.4, 0.4)
        throttle = _clamp(0.45 + (target_ias - ias) * 0.02, 0.2, 0.75)
    else:
        flaps = 30.0
        gear = 1.0
        trim = _clamp((vref - ias) / 140.0, -0.5, 0.5)
        throttle = _clamp(0.25 + (target_ias - ias) * 0.02, 0.05, 0.45)

    # Flare management.
    if radio_alt < 50.0:
        _state['phase'] = 'flare'
        flare_gain = _clamp((50.0 - radio_alt) / 50.0, 0.0, 1.0)
        elevator = _clamp(0.05 + flare_gain * 0.22 - target_vs / 5000.0, -0.1, 0.4)
        throttle = 0.08 if radio_alt < 20.0 else min(throttle, 0.2)
    else:
        elevator = _clamp((-gs * 0.08) + (vref - ias) / 180.0, -0.25, 0.25)
        if radio_alt < 200.0:
            elevator += _clamp(-target_vs / 4000.0, -0.08, 0.15)

    # Lateral / yaw control.
    bank_cmd = _clamp(-loc * 0.12, -0.25, 0.25)
    aileron = bank_cmd
    rudder = _clamp((target_hdg - track) / 50.0 + (-loc * 0.03), -0.35, 0.35)

    # Engine-out and crosswind compensation.
    if engines == 1:
        if not engine1:
            rudder += 0.12
            aileron += 0.04
        else:
            rudder -= 0.12
            aileron -= 0.04
    if strong_crosswind:
        aileron += _clamp(-loc * 0.05, -0.08, 0.08)
        rudder += _clamp((target_hdg - track) / 35.0, -0.12, 0.12)

    # Speed management and defaults.
    speedbrake = 0.0 if radio_alt > 20.0 else 0.0
    spoiler = 0.0
    reverse_1 = 0.0
    reverse_2 = 0.0
    autobrake = 0.0

    if radio_alt < 10.0:
        throttle = min(throttle, 0.05)
    if radio_alt < 5.0:
        elevator = max(elevator, 0.12)

    # Fail-safe for severe tailwind / overspeed.
    if ias > target_ias + 35.0 and radio_alt > 100.0:
        throttle = 0.0
        speedbrake = 1.0
        flaps = 15.0

    return {
        'elevator': _clamp(elevator, -1.0, 1.0),
        'aileron': _clamp(aileron, -1.0, 1.0),
        'rudder': _clamp(rudder, -1.0, 1.0),
        'throttle_1': _clamp(throttle if engine1 or engines == 0 else 0.0, 0.0, 1.0),
        'throttle_2': _clamp(throttle if engine2 or engines == 0 else 0.0, 0.0, 1.0),
        'elevator_trim': _clamp(trim, -1.0, 1.0),
        'flaps': _clamp(flaps, 0.0, 40.0),
        'gear': _clamp(gear, 0.0, 1.0),
        'speedbrake': _clamp(speedbrake, 0.0, 1.0),
        'spoiler': _clamp(spoiler, 0.0, 1.0),
        'autobrake': _clamp(autobrake, 0.0, 1.0),
        'reverse_1': _clamp(reverse_1, 0.0, 1.0),
        'reverse_2': _clamp(reverse_2, 0.0, 1.0),
    }
