; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpName %_ ""
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_64 = OpConstant %float 64.0
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Uniform_v3float = OpTypePointer Uniform %v3float
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%outColor = OpVariable %_ptr_Output_v4float Output
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %func
%label = OpLabel
%wp = OpLoad %v3float %worldPos
%n0 = OpLoad %v3float %normal
%ubolightptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%ubocameraptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%light4 = OpLoad %v4float %ubolightptr
%cam4 = OpLoad %v4float %ubocameraptr
%light3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%cam3 = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
%n = OpExtInst %v3float %1 Normalize %n0
%lsub = OpFSub %v3float %light3 %wp
%l = OpExtInst %v3float %1 Normalize %lsub
%vsub = OpFSub %v3float %cam3 %wp
%v = OpExtInst %v3float %1 Normalize %vsub
%lsum = OpFAdd %v3float %l %v
%h = OpExtInst %v3float %1 Normalize %lsum
%ndotl = OpDot %float %n %l
%ndotl_clamped = OpExtInst %float %1 FMax %ndotl %float_0
%ndoth = OpDot %float %n %h
%ndoth_clamped = OpExtInst %float %1 FMax %ndoth %float_0
%spec = OpExtInst %float %1 Pow %ndoth_clamped %float_64
%diff_term = OpFMul %float %float_0_6 %ndotl_clamped
%spec_term = OpFMul %float %float_0_8 %spec
%sum1 = OpFAdd %float %float_0_1 %diff_term
%intensity = OpFAdd %float %sum1 %spec_term
%rgb = OpCompositeConstruct %v3float %intensity %intensity %intensity
%color = OpCompositeConstruct %v4float %rgb %float_1
OpStore %outColor %color
OpReturn
OpFunctionEnd
