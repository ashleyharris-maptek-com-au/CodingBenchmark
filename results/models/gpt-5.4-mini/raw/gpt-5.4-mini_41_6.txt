OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %uv %tangent %color %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %tangent "tangent"
OpName %color "color"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_float = OpTypePointer Function %float

%zero = OpConstant %float 0.0
%one = OpConstant %float 1.0
%two = OpConstant %float 2.0
%half = OpConstant %float 0.5
%point1 = OpConstant %float 0.1
%point6 = OpConstant %float 0.6
%point8 = OpConstant %float 0.8
%sixtyfour = OpConstant %float 64.0
%int0 = OpConstant %int 0
%int1 = OpConstant %int 1
%int3 = OpConstant %int 3
%int4 = OpConstant %int 4
%int5 = OpConstant %int 5

%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%main = OpFunction %void None %func
%entry = OpLabel
%wp = OpLoad %v3float %worldPos
%n_in = OpLoad %v3float %normal
%n_len = OpExtInst %float %GLSL.std.450 Normalize %n_in
%light = OpAccessChain %_ptr_Uniform_v4float %ubo %int3
%light4 = OpLoad %v4float %light
%cam = OpAccessChain %_ptr_Uniform_v4float %ubo %int4
%cam4 = OpLoad %v4float %cam
%lightPos3 = OpVectorShuffle %v3float %light4 %light4 0 1 2
%camPos3 = OpVectorShuffle %v3float %cam4 %cam4 0 1 2
%lvec = OpFSub %v3float %lightPos3 %wp
%l = OpExtInst %v3float %GLSL.std.450 Normalize %lvec
%vvec = OpFSub %v3float %camPos3 %wp
%v = OpExtInst %v3float %GLSL.std.450 Normalize %vvec
%lh = OpFAdd %v3float %l %v
%h = OpExtInst %v3float %GLSL.std.450 Normalize %lh
%ndl = OpDot %float %n_len %l
%zero_diff = OpExtInst %float %GLSL.std.450 FMax %ndl %zero
%ndh = OpDot %float %n_len %h
%zero_spec = OpExtInst %float %GLSL.std.450 FMax %ndh %zero
%spec = OpExtInst %float %GLSL.std.450 Pow %zero_spec %sixtyfour
%diff_term = OpFMul %float %point6 %zero_diff
%spe_term = OpFMul %float %point8 %spec
%sum1 = OpFAdd %float %point1 %diff_term
%intensity = OpFAdd %float %sum1 %spe_term
%rgb = OpCompositeConstruct %v3float %intensity %intensity %intensity
%rgba = OpCompositeConstruct %v4float %rgb %one
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
