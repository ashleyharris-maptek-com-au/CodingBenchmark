OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor

OpExecutionMode %main OutputTriangleStrip

OpExecutionMode %main Invocations 1

OpName %main "main"

OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"

OpName %normal "normal"
OpName %color "color"

OpDecorate %gl_PerVertex_out Block

OpDecorate %gl_in BuiltIn Position
OpDecorate %gl_out BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0

%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

%v3_1 = OpConstantComposite %v3float %float_1 %float_1 %float_1
%v3_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0

%main = OpFunction %void None %functype
%label = OpLabel

%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0

%c4 = OpLoad %v3float %c4_ptr

%bc0 = OpFMul %v3float %c0 %float_03
%bc2 = OpFMul %v3float %c2 %float_03
%bc4 = OpFMul %v3float %c4 %float_03

; Emit filled triangle
%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %p0
OpStore %outColor %bc0
OpEmitVertex
%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %p2

OpEmitVertex
%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpEndPrimitive
