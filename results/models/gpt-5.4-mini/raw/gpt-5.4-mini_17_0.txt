using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    struct Box
    {
        public int W, H, D, Index;
        public long Volume;
    }

    struct Placement
    {
        public int Index, X, Y, Z;
        public Placement(int index, int x, int y, int z)
        {
            Index = index; X = x; Y = y; Z = z;
        }
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    static bool BetterOrientation(int w1, int h1, int d1, int w2, int h2, int d2)
    {
        long v1 = (long)w1 * h1 * d1;
        long v2 = (long)w2 * h2 * d2;
        if (v1 != v2) return v1 < v2;
        long m1 = Math.Max(w1, Math.Max(h1, d1));
        long m2 = Math.Max(w2, Math.Max(h2, d2));
        if (m1 != m2) return m1 < m2;
        long s1 = (long)w1 + h1 + d1;
        long s2 = (long)w2 + h2 + d2;
        return s1 < s2;
    }

    static void SortDims(ref int a, ref int b, ref int c)
    {
        if (a > b) { int t = a; a = b; b = t; }
        if (b > c) { int t = b; b = c; c = t; }
        if (a > b) { int t = a; a = b; b = t; }
    }

    static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int W = fs.NextInt();
        int H = fs.NextInt();
        int D = fs.NextInt();

        var boxes = new Box[n];
        for (int i = 0; i < n; i++)
        {
            int w = fs.NextInt();
            int h = fs.NextInt();
            int d = fs.NextInt();
            SortDims(ref w, ref h, ref d);
            boxes[i] = new Box { W = w, H = h, D = d, Index = i, Volume = (long)w * h * d };
        }

        Array.Sort(boxes, (a, b) =>
        {
            int cmp = b.Volume.CompareTo(a.Volume);
            if (cmp != 0) return cmp;
            cmp = Math.Max(b.W, Math.Max(b.H, b.D)).CompareTo(Math.Max(a.W, Math.Max(a.H, a.D)));
            if (cmp != 0) return cmp;
            cmp = (b.W + b.H + b.D).CompareTo(a.W + a.H + a.D);
            if (cmp != 0) return cmp;
            return a.Index.CompareTo(b.Index);
        });

        var placements = new List<Placement>(Math.Min(n, 65536));
        int layersY = 0;
        int y = 0;
        while (y < H && layersY < 3)
        {
            int remainingH = H - y;
            int layerH = 0;
            int cursorX = 0;
            int cursorZ = 0;
            int rowD = 0;
            bool anyPlacedInLayer = false;

            for (int i = 0; i < n; i++)
            {
                if (boxes[i].Index < 0) continue;
                int bw = boxes[i].W, bh = boxes[i].H, bd = boxes[i].D;
                if (bh > remainingH) continue;

                bool placed = false;
                for (int perm = 0; perm < 6; perm++)
                {
                    int pw = bw, ph = bh, pd = bd;
                    switch (perm)
                    {
                        case 0: break;
                        case 1: (pw, ph) = (ph, pw); break;
                        case 2: (pw, pd) = (pd, pw); break;
                        case 3: (ph, pd) = (pd, ph); break;
                        case 4: (pw, ph, pd) = (ph, pd, pw); break;
                        case 5: (pw, ph, pd) = (pd, pw, ph); break;
                    }
                    if (pw > W || ph > remainingH || pd > D) continue;

                    if (cursorX + pw <= W && cursorZ + pd <= D)
                    {
                        placements.Add(new Placement(boxes[i].Index, cursorX, y, cursorZ));
                        cursorX += pw;
                        rowD = Math.Max(rowD, pd);
                        layerH = Math.Max(layerH, ph);
                        boxes[i].Index = -1;
                        placed = true;
                        anyPlacedInLayer = true;
                        break;
                    }
                    if (cursorX + pw > W && cursorZ + rowD + pd <= D && pw <= W)
                    {
                        cursorX = 0;
                        cursorZ += rowD;
                        rowD = 0;
                        if (cursorZ + pd <= D)
                        {
                            placements.Add(new Placement(boxes[i].Index, cursorX, y, cursorZ));
                            cursorX += pw;
                            rowD = Math.Max(rowD, pd);
                            layerH = Math.Max(layerH, ph);
                            boxes[i].Index = -1;
                            placed = true;
                            anyPlacedInLayer = true;
                            break;
                        }
                    }
                }
            }

            if (!anyPlacedInLayer)
            {
                int bestH = int.MaxValue;
                for (int i = 0; i < n; i++)
                {
                    if (boxes[i].Index < 0) continue;
                    if (boxes[i].H >= 1 && boxes[i].H < bestH && boxes[i].H <= remainingH)
                        bestH = boxes[i].H;
                }
                if (bestH == int.MaxValue) break;
                y += bestH;
                layersY++;
            }
            else
            {
                if (layerH <= 0)
                {
                    int minH = int.MaxValue;
                    for (int i = 0; i < n; i++)
                    {
                        if (boxes[i].Index < 0) continue;
                        minH = Math.Min(minH, boxes[i].H);
                    }
                    if (minH == int.MaxValue) break;
                    layerH = minH;
                }
                y += layerH;
                layersY++;
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(placements.Count.ToString());
        for (int i = 0; i < placements.Count; i++)
        {
            var p = placements[i];
            sb.Append(p.Index).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }
}
