from math import atan2, degrees, radians, sin, cos, tan, exp


_S = {
    "v_kmh": 0.0,
    "yaw_dps": 0.0,
    "lane_offset_m": 0.0,
    "heading_deg": 0.0,
    "speed_err_i": 0.0,
    "steer_i": 0.0,
    "last_throttle": 0.0,
    "last_brake": 0.0,
    "last_steering": 0.0,
    "prev_obs_close": 999.0,
    "prev_lane_error": 0.0,
}


def _finite(x):
    return isinstance(x, (int, float)) and x == x and x != float("inf") and x != float("-inf")


def _get(state, key, default=None):
    v = state.get(key, default)
    return v if _finite(v) else default


def _valid_flag(state, key, default=False):
    return bool(state.get(key, default))


def _median(vals, default=None):
    vals = [v for v in vals if _finite(v)]
    if not vals:
        return default
    vals.sort()
    n = len(vals)
    mid = n // 2
    return vals[mid] if n % 2 else 0.5 * (vals[mid - 1] + vals[mid])


def _clip(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _wrap_deg(a):
    while a > 180:
        a -= 360
    while a < -180:
        a += 360
    return a


def _smooth(name, value, alpha=0.25):
    prev = _S.get(name, value)
    if not _finite(prev):
        prev = value
    out = prev * (1.0 - alpha) + value * alpha
    _S[name] = out
    return out


def _read_speed_kmh(state):
    vals = []
    for k in ("wheel_speed_fl_kmh", "wheel_speed_fr_kmh", "wheel_speed_rl_kmh", "wheel_speed_rr_kmh"):
        vals.append(_get(state, k))
    for k in ("vehicle_speed_1_kmh", "vehicle_speed_2_kmh", "gps_1_speed_kmh", "gps_2_speed_kmh"):
        vals.append(_get(state, k))
    v = _median(vals, _S["v_kmh"])
    if v is None:
        v = _S["v_kmh"]
    return _smooth("v_kmh", _clip(v, 0.0, 320.0), 0.18)


def _read_yaw_rate_dps(state):
    vals = []
    for k in ("gyro_1_yaw_rate_dps", "gyro_2_yaw_rate_dps", "gyro_3_yaw_rate_dps"):
        vals.append(_get(state, k))
    # Fallback from lateral accel / speed when gyros fail.
    v = _median(vals, None)
    if v is None:
        v_ms = _S["v_kmh"] / 3.6
        accs = [_get(state, "accel_1_lateral_ms2"), _get(state, "accel_2_lateral_ms2"), _get(state, "accel_3_lateral_ms2")]
        a_lat = _median(accs, 0.0) or 0.0
        if abs(v_ms) > 0.5:
            v = degrees(a_lat / max(v_ms, 0.5))
        else:
            v = 0.0
    return _smooth("yaw_dps", _clip(v, -120.0, 120.0), 0.22)


def _read_lane_offset(state):
    vals = []
    for i in (1, 2, 3):
        v = _get(state, f"vision_{i}_lane_offset_m")
        if v is not None:
            vals.append(v)
    v = _median(vals, _S["lane_offset_m"])
    if v is None:
        # Infer offset from lidar asymmetry when vision is unreliable.
        lefts = []
        rights = []
        for a in ("+005", "+015", "+030", "+045", "+060", "+090"):
            lefts.append(_get(state, f"lidar_{a}_deg_range_m"))
        for a in ("-005", "-015", "-030", "-045", "-060", "-090"):
            rights.append(_get(state, f"lidar_{a}_deg_range_m"))
        l = _median(lefts, 50.0) or 50.0
        r = _median(rights, 50.0) or 50.0
        v = 0.25 * (r - l)
    return _smooth("lane_offset_m", _clip(v, -3.5, 3.5), 0.2)


def _lane_boundaries(state):
    left, right = [], []
    for i in (1, 2, 3):
        if _valid_flag(state, f"vision_{i}_left_line_valid", False):
            d = _get(state, f"vision_{i}_left_line_dist_m")
            if d is not None:
                left.append(d)
        if _valid_flag(state, f"vision_{i}_right_line_valid", False):
            d = _get(state, f"vision_{i}_right_line_dist_m")
            if d is not None:
                right.append(d)
    left_d = _median(left, None)
    right_d = _median(right, None)
    if left_d is None or right_d is None:
        # Lidar fallback: approximate lane width from side beams.
        ll = _median([_get(state, "lidar_+060_deg_range_m"), _get(state, "lidar_+090_deg_range_m")], 8.0) or 8.0
        rr = _median([_get(state, "lidar_-060_deg_range_m"), _get(state, "lidar_-090_deg_range_m")], 8.0) or 8.0
        # Use current offset to infer a center corridor if line data are missing.
        width = _clip((ll + rr) * 0.35, 2.9, 4.2)
        left_d = width * 0.5 - _S["lane_offset_m"]
        right_d = width * 0.5 + _S["lane_offset_m"]
    return left_d, right_d


def _road_curvature(state):
    vals = []
    for i in (1, 2, 3):
        v = _get(state, f"vision_{i}_road_curvature_1pm")
        if v is not None:
            vals.append(v)
    return _median(vals, 0.0) or 0.0


def _speed_limit(state):
    vals = []
    for i in (1, 2, 3):
        if _valid_flag(state, f"vision_{i}_speed_limit_valid", False):
            v = _get(state, f"vision_{i}_speed_limit_kmh")
            if v is not None:
                vals.append(v)
    v = _median(vals, None)
    if v is None:
        v = _S["v_kmh"] + 20.0
    return _clip(v, 30.0, 140.0)


def _obstacle_info(state):
    dists = []
    rels = []
    detected = False
    for i in (1, 2, 3):
        if _valid_flag(state, f"vision_{i}_obstacle_valid", False):
            if state.get(f"vision_{i}_obstacle_detected"):
                detected = True
                d = _get(state, f"vision_{i}_obstacle_dist_m")
                r = _get(state, f"vision_{i}_obstacle_rel_speed_kmh")
                if d is not None:
                    dists.append(d)
                if r is not None:
                    rels.append(r)
    front = [_get(state, k) for k in ("lidar_+005_deg_range_m", "lidar_+015_deg_range_m", "lidar_+030_deg_range_m", "lidar_-005_deg_range_m", "lidar_-015_deg_range_m", "lidar_-030_deg_range_m")]
    front = [v for v in front if v is not None]
    lidar_close = _median(front, 999.0) or 999.0
    obs_d = _median(dists, None)
    rel = _median(rels, 0.0) or 0.0
    if obs_d is None or obs_d > lidar_close + 10.0:
        obs_d = lidar_close
    if obs_d < 999.0:
        detected = detected or obs_d < 80.0
    return detected, obs_d, rel


def _pick_lane(state):
    current = int(_get(state, "current_lane", 1) or 1)
    target = int(_get(state, "target_lane", current) or current)
    return _clip(target, 0, 2), _clip(current, 0, 2)


def autopilot_step(state: dict, dt: float) -> dict:
    if not _finite(dt) or dt <= 0:
        dt = _get(state, "dt", 0.1) or 0.1

    v_kmh = _read_speed_kmh(state)
    yaw_dps = _read_yaw_rate_dps(state)
    lane_offset = _read_lane_offset(state)
    left_d, right_d = _lane_boundaries(state)
    curvature = _road_curvature(state)
    speed_limit = _speed_limit(state)
    obstacle_detected, obstacle_dist, obstacle_rel = _obstacle_info(state)
    target_speed = _clip(_get(state, "target_speed_kmh", speed_limit) or speed_limit, 0.0, 150.0)
    mu = _clip(_get(state, "estimated_mu", 0.8) or 0.8, 0.05, 1.2)
    target_lane, current_lane = _pick_lane(state)

    # Lane target centered in the requested lane; use vision offset and line distances.
    lane_center_error = lane_offset
    if _finite(left_d) and _finite(right_d):
        lane_center_error += 0.5 * (left_d - right_d)
    lane_center_error += 0.35 * curvature * (v_kmh / 3.6) ** 2
    lane_center_error = _clip(lane_center_error, -3.0, 3.0)

    # Obstacle handling: prefer braking first, then lane change if adjacent space looks better.
    front_gap = obstacle_dist if obstacle_detected else 999.0
    closing = max(0.0, -(obstacle_rel or 0.0))
    safe_gap = max(10.0, 8.0 + 0.4 * v_kmh / 3.6)
    if mu < 0.4:
        safe_gap *= 1.7
    elif mu < 0.25:
        safe_gap *= 2.4
    desired_speed = min(target_speed, speed_limit)
    desired_speed *= (0.55 + 0.45 * mu)
    if mu < 0.3:
        desired_speed = min(desired_speed, 55.0)
    if obstacle_detected:
        if front_gap < 8.0 or closing > 15.0:
            desired_speed = 0.0
        else:
            desired_speed = min(desired_speed, max(0.0, (front_gap - safe_gap) * 2.0))
    desired_speed = max(0.0, desired_speed)

    # Steering control with damping and fail-safe conservative gains.
    speed_ms = max(v_kmh / 3.6, 0.1)
    steer_error = lane_center_error
    yaw_term = yaw_dps * 0.012
    steer_cmd = -0.42 * steer_error - 0.08 * yaw_term - 0.12 * curvature * speed_ms
    if obstacle_detected and front_gap < 40.0:
        # Nudge away from the obstacle if a lane change is requested or the center is blocked.
        if target_lane < current_lane:
            steer_cmd -= 0.15
        elif target_lane > current_lane:
            steer_cmd += 0.15
        else:
            steer_cmd += 0.08 if lane_center_error < 0 else -0.08
    if mu < 0.3:
        steer_cmd *= 0.55
    if abs(yaw_dps) > 15.0:
        steer_cmd -= 0.03 * yaw_dps
    steer_cmd = _clip(steer_cmd, -1.0, 1.0)
    steer_cmd = _smooth("last_steering", steer_cmd, 0.35)

    # Speed controller with anti-windup.
    speed_err = desired_speed - v_kmh
    _S["speed_err_i"] = _clip(_S["speed_err_i"] + speed_err * dt, -80.0, 80.0)
    kp = 0.020
    ki = 0.004
    kd = 0.010
    if mu < 0.5:
        kp *= 0.7
        ki *= 0.4
        kd *= 0.5
    if mu < 0.3:
        kp *= 0.55
        ki *= 0.2
        kd *= 0.4
    accel_term = 0.0
    for k in ("accel_1_longitudinal_ms2", "accel_2_longitudinal_ms2", "accel_3_longitudinal_ms2"):
        av = _get(state, k)
        if av is not None:
            accel_term += av
    accel_term /= 3.0
    accel_term = _clip(accel_term, -6.0, 6.0)
    throttle = kp * speed_err + ki * _S["speed_err_i"] - kd * accel_term
    brake = 0.0

    if desired_speed <= 0.5 and v_kmh > 1.0:
        brake = 0.65
        throttle = 0.0
    else:
        if speed_err < -1.0:
            brake = _clip((-speed_err) * 0.03, 0.0, 1.0)
            throttle = 0.0
        elif obstacle_detected and front_gap < safe_gap * 1.4:
            brake = _clip((safe_gap * 1.4 - front_gap) / max(safe_gap, 1.0), 0.0, 1.0)
            throttle = min(throttle, 0.1)
        elif speed_err > 0.5:
            throttle = _clip(throttle, 0.0, 1.0)
        else:
            throttle = _clip(throttle, 0.0, 0.35)

    # Traction management and stuck throttle/brake plausibility.
    if mu < 0.3:
        throttle = min(throttle, 0.28)
        brake = max(brake, 0.05 if speed_err < -2.0 else 0.0)
    if mu < 0.18:
        throttle = min(throttle, 0.15)
        brake = max(brake, 0.12 if obstacle_detected or speed_err < -5.0 else 0.0)

    # Crosswind / curvature stability: keep speeds lower when lateral demand is high.
    lateral_demand = abs(steer_cmd) + abs(curvature) * speed_ms * speed_ms * 0.04
    if lateral_demand > 0.8:
        throttle = min(throttle, 0.25)
    if lateral_demand > 1.2:
        brake = max(brake, 0.08)

    # If the car is already fast relative to target, prefer braking over coasting.
    if v_kmh > desired_speed + 8.0:
        brake = max(brake, _clip((v_kmh - desired_speed - 8.0) / 40.0, 0.0, 0.9))
        throttle = 0.0

    throttle = _clip(throttle, 0.0, 1.0)
    brake = _clip(brake, 0.0, 1.0)
    if brake > 0.15:
        throttle = 0.0

    # Smooth outputs but keep emergency response quick.
    if brake > 0.5 or obstacle_detected and front_gap < 10.0:
        steer_cmd = _clip(steer_cmd, -0.8, 0.8)
    throttle = _smooth("last_throttle", throttle, 0.28)
    brake = _smooth("last_brake", brake, 0.35)

    # Emergency override for stopped obstacle and low gap.
    if obstacle_detected and front_gap < 6.0:
        throttle = 0.0
        brake = 1.0
        steer_cmd = _clip(steer_cmd, -0.5, 0.5)

    # Prevent overly aggressive input on low friction or uncertain sensing.
    sensor_confidence = 0.0
    for prefix in ("vision_1", "vision_2", "vision_3"):
        if _valid_flag(state, f"{prefix}_left_line_valid", False) or _valid_flag(state, f"{prefix}_right_line_valid", False):
            sensor_confidence += 1.0
    if _finite(_get(state, "gps_1_x_m")) or _finite(_get(state, "gps_2_x_m")):
        sensor_confidence += 0.5
    if sensor_confidence < 1.0:
        throttle = min(throttle, 0.25)
        brake = max(brake, 0.0)
        steer_cmd *= 0.8

    return {
        "steering": _clip(steer_cmd, -1.0, 1.0),
        "throttle": _clip(throttle, 0.0, 1.0),
        "brake": _clip(brake, 0.0, 1.0),
    }
