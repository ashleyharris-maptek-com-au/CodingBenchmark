@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

fn rotl32(x : u32, s : u32) -> u32 {
    return (x << s) | (x >> (32u - s));
}

fn quarter_round(a_in : u32, b_in : u32, c_in : u32, d_in : u32) -> vec4<u32> {
    var a : u32 = a_in;
    var b : u32 = b_in;
    var c : u32 = c_in;
    var d : u32 = d_in;

    a = a + b;
    d = d ^ a;
    d = rotl32(d, 16u);

    c = c + d;
    b = b ^ c;
    b = rotl32(b, 12u);

    a = a + b;
    d = d ^ a;
    d = rotl32(d, 8u);

    c = c + d;
    b = b ^ c;
    b = rotl32(b, 7u);

    return vec4<u32>(a, b, c, d);
}

fn custom_hash(data : array<u32, 16>, nonce : u32) -> vec2<u32> {
    var state : array<u32, 16>;

    state[0u] = data[0u];
    state[1u] = data[1u];
    state[2u] = data[2u];
    state[3u] = data[3u];
    state[4u] = data[4u];
    state[5u] = data[5u];
    state[6u] = data[6u];
    state[7u] = data[7u];
    state[8u] = data[8u];
    state[9u] = data[9u];
    state[10u] = data[10u];
    state[11u] = data[11u];
    state[12u] = data[12u];
    state[13u] = data[13u];
    state[14u] = data[14u];
    state[15u] = data[15u];

    state[0u] = state[0u] ^ (nonce + 0u);
    state[1u] = state[1u] ^ (nonce + 1u);
    state[2u] = state[2u] ^ (nonce + 2u);
    state[3u] = state[3u] ^ (nonce + 3u);
    state[4u] = state[4u] ^ (nonce + 4u);
    state[5u] = state[5u] ^ (nonce + 5u);
    state[6u] = state[6u] ^ (nonce + 6u);
    state[7u] = state[7u] ^ (nonce + 7u);
    state[8u] = state[8u] ^ (nonce + 8u);
    state[9u] = state[9u] ^ (nonce + 9u);
    state[10u] = state[10u] ^ (nonce + 10u);
    state[11u] = state[11u] ^ (nonce + 11u);
    state[12u] = state[12u] ^ (nonce + 12u);
    state[13u] = state[13u] ^ (nonce + 13u);
    state[14u] = state[14u] ^ (nonce + 14u);
    state[15u] = state[15u] ^ (nonce + 15u);

    var qr : vec4<u32>;

    for (var round : u32 = 0u; round < 8u; round = round + 1u) {
        qr = quarter_round(state[0u], state[1u], state[2u], state[3u]);
        state[0u] = qr.x; state[1u] = qr.y; state[2u] = qr.z; state[3u] = qr.w;

        qr = quarter_round(state[4u], state[5u], state[6u], state[7u]);
        state[4u] = qr.x; state[5u] = qr.y; state[6u] = qr.z; state[7u] = qr.w;

        qr = quarter_round(state[8u], state[9u], state[10u], state[11u]);
        state[8u] = qr.x; state[9u] = qr.y; state[10u] = qr.z; state[11u] = qr.w;

        qr = quarter_round(state[12u], state[13u], state[14u], state[15u]);
        state[12u] = qr.x; state[13u] = qr.y; state[14u] = qr.z; state[15u] = qr.w;
    }

    let h0 : u32 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^ state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
    let h1 : u32 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^ state[9u] ^ state[11u] ^ state[13u] ^ state[15u];
    return vec2<u32>(h0, h1);
}

fn trailing_zeros_64(lo : u32, hi : u32) -> u32 {
    if (lo != 0u) {
        return countTrailingZeros(lo);
    }
    return 32u + countTrailingZeros(hi);
}

@compute @workgroup_size(256u)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    if (gid.x >= params.z) {
        return;
    }

    let nonce : u32 = params.y + gid.x;
    let hash : vec2<u32> = custom_hash(baseData, nonce);
    let tz : u32 = trailing_zeros_64(hash.x, hash.y);

    if (tz >= params.x) {
        result[0u] = 1u;
        result[1u] = nonce;
        result[2u] = hash.x;
        result[3u] = hash.y;
    }
}
