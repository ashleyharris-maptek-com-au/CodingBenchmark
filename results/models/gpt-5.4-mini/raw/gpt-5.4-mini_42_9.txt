OpCapability Shader
OpCapability Geometry
OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "proj"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_025 = OpConstant %float 0.025
%float_05 = OpConstant %float 0.5
%uint_0 = OpConstant %uint 0
%glsl450 = OpExtInstImport "GLSL.std.450"

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v3 = OpTypePointer Output %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%outColor = OpVariable %ptr_out_v3 Output

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_ubo Uniform

%main = OpFunction %void None %fn
%label = OpLabel

%glin0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%glin1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%glin2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%glin3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p0 = OpLoad %v4float %glin0
%p1 = OpLoad %v4float %glin1
%p2 = OpLoad %v4float %glin2
%p3 = OpLoad %v4float %glin3

%c1ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c1 = OpLoad %v3float %c1ptr
%c2 = OpLoad %v3float %c2ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p2xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p3xy = OpVectorShuffle %v2float %p3 %p3 0 1
%d0 = OpFSub %v2float %p2xy %p0xy
%d1 = OpFSub %v2float %p3xy %p1xy
%len0sq = OpDot %float %d0 %d0
%len1sq = OpDot %float %d1 %d1
%len0 = OpExtInst %float %glsl450 Sqrt %len0sq
%len1 = OpExtInst %float %glsl450 Sqrt %len1sq
%inv0 = OpFDiv %float %float_1 %len0
%inv1 = OpFDiv %float %float_1 %len1
%t0 = OpVectorTimesScalar %v2float %d0 %inv0
%t1 = OpVectorTimesScalar %v2float %d1 %inv1
%t0x = OpCompositeExtract %float %t0 0
%t0y = OpCompositeExtract %float %t0 1
%t1x = OpCompositeExtract %float %t1 0
%t1y = OpCompositeExtract %float %t1 1
%perp0 = OpCompositeConstruct %v2float %t0y %t0x
%perp1 = OpCompositeConstruct %v2float %t1y %t1x
%neg1 = OpConstant %float -1
%perp0n = OpVectorTimesScalar %v2float %perp0 %neg1
%perp1n = OpVectorTimesScalar %v2float %perp1 %neg1
%off0 = OpVectorTimesScalar %v2float %perp0 %float_025
%off0n = OpVectorTimesScalar %v2float %perp0n %float_025
%off1 = OpVectorTimesScalar %v2float %perp1 %float_025
%off1n = OpVectorTimesScalar %v2float %perp1n %float_025
%splus = OpFAdd %v2float %p1xy %off0
%sminus = OpFAdd %v2float %p1xy %off0n
%eplus = OpFAdd %v2float %p2xy %off1
%eminus = OpFAdd %v2float %p2xy %off1n
%splusx = OpCompositeExtract %float %splus 0
%splusy = OpCompositeExtract %float %splus 1
%sminusx = OpCompositeExtract %float %sminus 0
%sminusy = OpCompositeExtract %float %sminus 1
%eplusx = OpCompositeExtract %float %eplus 0
%eplusy = OpCompositeExtract %float %eplus 1
%eminusx = OpCompositeExtract %float %eminus 0
%eminusy = OpCompositeExtract %float %eminus 1

%pos0 = OpCompositeConstruct %v4float %splusx %splusy %float_0 %float_1
%pos1 = OpCompositeConstruct %v4float %sminusx %sminusy %float_0 %float_1
%pos2 = OpCompositeConstruct %v4float %eplusx %eplusy %float_0 %float_1
%pos3 = OpCompositeConstruct %v4float %eminusx %eminusy %float_0 %float_1

%out0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out0 %pos0
OpStore %outColor %c1
OpEmitVertex

%out1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out1 %pos1
OpStore %outColor %c1
OpEmitVertex

%out2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out2 %pos2
OpStore %outColor %c2
OpEmitVertex

%out3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out3 %pos3
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
