; SPIR-V
; Version: 1.0
; Generator: Codex
; Bound: 27
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %color %outColor
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450
OpName %main "main"
OpName %color "color"
OpName %outColor "outColor"
OpDecorate %color Location 4
OpDecorate %outColor Location 0
%void = OpTypeVoid
%funct = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%color = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%float_1 = OpConstant %float 1
%main = OpFunction %void None %funct
%label = OpLabel
%color_val = OpLoad %v3float %color
%x = OpCompositeExtract %float %color_val 0
%y = OpCompositeExtract %float %color_val 1
%z = OpCompositeExtract %float %color_val 2
%rgba = OpCompositeConstruct %v4float %x %y %z %float_1
OpStore %outColor %rgba
OpReturn
OpFunctionEnd
