use std::cmp::Reverse;
use std::collections::{BinaryHeap, HashSet};
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj = vec![Vec::<usize>::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    for neigh in &mut adj {
        neigh.sort_unstable();
        neigh.dedup();
    }

    let mut dominated = vec![false; n];
    let mut chosen = vec![false; n];
    let mut solution: Vec<usize> = Vec::new();

    let mut heap: BinaryHeap<(Reverse<usize>, usize)> = BinaryHeap::new();
    for v in 0..n {
        heap.push((Reverse(adj[v].len() + 1), v));
    }

    let mut uncovered = n;
    while uncovered > 0 {
        let mut best_v = None;
        let mut best_gain = 0usize;

        while let Some((Reverse(_prio), v)) = heap.pop() {
            let mut gain = if !dominated[v] { 1 } else { 0 };
            for &to in &adj[v] {
                if !dominated[to] {
                    gain += 1;
                }
            }
            if gain == 0 {
                continue;
            }
            best_v = Some(v);
            best_gain = gain;
            break;
        }

        if let Some(v) = best_v {
            if chosen[v] {
                continue;
            }
            chosen[v] = true;
            solution.push(v);
            if !dominated[v] {
                dominated[v] = true;
                uncovered -= 1;
            }
            for &to in &adj[v] {
                if !dominated[to] {
                    dominated[to] = true;
                    uncovered -= 1;
                }
            }
        } else {
            break;
        }
    }

    let mut dom_count = vec![0usize; n];
    let mut witness = vec![usize::MAX; n];
    for &v in &solution {
        if witness[v] == usize::MAX {
            witness[v] = v;
            dom_count[v] += 1;
        }
        for &to in &adj[v] {
            if witness[to] == usize::MAX {
                witness[to] = v;
                dom_count[v] += 1;
            } else {
                dom_count[witness[to]] += 0;
            }
        }
    }

    let mut in_solution = HashSet::with_capacity(solution.len() * 2 + 1);
    for &v in &solution {
        in_solution.insert(v);
    }

    let mut order = solution.clone();
    order.sort_by_key(|&v| adj[v].len());
    for &v in &order {
        if !in_solution.contains(&v) {
            continue;
        }
        let mut can_remove = true;
        let mut affected: Vec<usize> = Vec::new();
        if witness[v] == v {
            affected.push(v);
        }
        for &to in &adj[v] {
            if witness[to] == v {
                affected.push(to);
            }
        }
        affected.sort_unstable();
        affected.dedup();

        for &x in &affected {
            let mut covered_elsewhere = false;
            if x == v {
                for &y in &adj[x] {
                    if y != v && in_solution.contains(&y) {
                        covered_elsewhere = true;
                        break;
                    }
                }
                if !covered_elsewhere {
                    can_remove = false;
                    break;
                }
            } else {
                let mut ok = false;
                for &y in &adj[x] {
                    if y != v && in_solution.contains(&y) {
                        ok = true;
                        break;
                    }
                }
                if !ok {
                    can_remove = false;
                    break;
                }
            }
        }

        if can_remove {
            in_solution.remove(&v);
        }
    }

    let mut sol: Vec<usize> = in_solution.into_iter().collect();
    sol.sort_unstable();

    let mut final_dom = vec![false; n];
    for &v in &sol {
        final_dom[v] = true;
        for &to in &adj[v] {
            final_dom[to] = true;
        }
    }
    if final_dom.iter().any(|&b| !b) {
        for v in 0..n {
            if !final_dom[v] {
                sol.push(v);
                final_dom[v] = true;
                for &to in &adj[v] {
                    final_dom[to] = true;
                }
            }
        }
        sol.sort_unstable();
        sol.dedup();
    }

    println!("{}", sol.len());
    if sol.is_empty() {
        println!();
    } else {
        for (i, v) in sol.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
