use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by_key(|&v| std::cmp::Reverse(adj[v].len()));

    let mut dominated = vec![false; n];
    let mut in_set = vec![false; n];
    let mut set = Vec::new();

    // Greedy phase: repeatedly pick the vertex that dominates the most currently undominated vertices.
    while dominated.iter().any(|&x| !x) {
        let mut best_v = 0usize;
        let mut best_gain = 0usize;
        for &v in &order {
            if in_set[v] {
                continue;
            }
            let mut gain = if dominated[v] { 0 } else { 1 };
            for &to in &adj[v] {
                if !dominated[to] {
                    gain += 1;
                }
            }
            if gain > best_gain {
                best_gain = gain;
                best_v = v;
            }
        }

        in_set[best_v] = true;
        set.push(best_v);
        if !dominated[best_v] {
            dominated[best_v] = true;
        }
        for &to in &adj[best_v] {
            dominated[to] = true;
        }
    }

    // Local cleanup: try removing redundant vertices in reverse order.
    let mut cover_count = vec![0usize; n];
    for &v in &set {
        cover_count[v] += 1;
        for &to in &adj[v] {
            cover_count[to] += 1;
        }
    }

    let mut keep = vec![true; set.len()];
    for idx in (0..set.len()).rev() {
        let v = set[idx];
        let mut removable = true;
        if cover_count[v] == 1 {
            removable = false;
        } else {
            for &to in &adj[v] {
                if cover_count[to] == 1 {
                    removable = false;
                    break;
                }
            }
        }
        if removable {
            keep[idx] = false;
            cover_count[v] -= 1;
            for &to in &adj[v] {
                cover_count[to] -= 1;
            }
        }
    }

    let mut ans: Vec<usize> = set
        .into_iter()
        .enumerate()
        .filter_map(|(i, v)| if keep[i] { Some(v) } else { None })
        .collect();

    // Safety pass: if cleanup ever broke coverage due to an edge case, restore greedily.
    let mut dominated = vec![false; n];
    for &v in &ans {
        dominated[v] = true;
        for &to in &adj[v] {
            dominated[to] = true;
        }
    }
    if dominated.iter().any(|&x| !x) {
        let mut undom: Vec<usize> = (0..n).filter(|&v| !dominated[v]).collect();
        undom.sort_unstable_by_key(|&v| std::cmp::Reverse(adj[v].len()));
        for v in undom {
            if !dominated[v] {
                ans.push(v);
                dominated[v] = true;
                for &to in &adj[v] {
                    dominated[to] = true;
                }
            }
        }
    }

    ans.sort_unstable();
    ans.dedup();

    println!("{}", ans.len());
    if ans.is_empty() {
        println!();
    } else {
        for (i, v) in ans.iter().enumerate() {
            if i > 0 {
                print!(" ");
            }
            print!("{}", v);
        }
        println!();
    }
}
