#include <bits/stdc++.h>
using namespace std;

static inline int sq(int f, int r) { return r * 8 + f; }
static inline int fileOf(int s) { return s & 7; }
static inline int rankOf(int s) { return s >> 3; }
static inline bool onBoard(int f, int r) { return (unsigned)f < 8u && (unsigned)r < 8u; }
static inline string sqStr(int s) {
    string t;
    t += char('a' + fileOf(s));
    t += char('1' + rankOf(s));
    return t;
}
static inline int pieceVal(char p) {
    switch (tolower((unsigned char)p)) {
        case 'p': return 100;
        case 'n': return 320;
        case 'b': return 330;
        case 'r': return 500;
        case 'q': return 900;
        case 'k': return 20000;
    }
    return 0;
}

struct Move {
    int from = -1, to = -1;
    char promo = 0;
    bool capture = false, ep = false, castle = false;
    bool kingSide = false;
};

struct State {
    array<char, 64> b{};
    bool white = true;
    bool wkc = false, wqc = false, bkc = false, bqc = false;
    int ep = -1;
    int halfmove = 0, fullmove = 1;
};

struct Undo {
    char captured = 0;
    bool wkc, wqc, bkc, bqc;
    int ep, halfmove, fullmove;
    char movedFrom, movedTo;
};

static atomic<bool> g_timeout(false);
static chrono::steady_clock::time_point g_deadline;
static State g_bestRoot;
static Move g_bestMove;
static int g_bestScore = -1e9;

static inline bool timeUp() {
    return chrono::steady_clock::now() >= g_deadline;
}

static State parseFen(const string& fen) {
    State s;
    stringstream ss(fen);
    string board, turn, castling, ep;
    ss >> board >> turn >> castling >> ep >> s.halfmove >> s.fullmove;
    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') { --r; f = 0; continue; }
        if (isdigit((unsigned char)c)) { f += c - '0'; continue; }
        s.b[sq(f, r)] = c;
        ++f;
    }
    s.white = (turn == "w");
    s.wkc = castling.find('K') != string::npos;
    s.wqc = castling.find('Q') != string::npos;
    s.bkc = castling.find('k') != string::npos;
    s.bqc = castling.find('q') != string::npos;
    if (ep != "-") {
        int f0 = ep[0] - 'a', r0 = ep[1] - '1';
        s.ep = sq(f0, r0);
    }
    return s;
}

static inline bool whitePiece(char c) { return c >= 'A' && c <= 'Z'; }
static inline bool blackPiece(char c) { return c >= 'a' && c <= 'z'; }
static inline bool sameSide(char p, bool white) { return p && (white ? whitePiece(p) : blackPiece(p)); }
static inline bool enemySide(char p, bool white) { return p && (white ? blackPiece(p) : whitePiece(p)); }

static bool attacked(const State& s, int sqr, bool byWhite) {
    int f = fileOf(sqr), r = rankOf(sqr);
    int dir = byWhite ? 1 : -1;
    for (int df : {-1, 1}) {
        int nf = f + df, nr = r - dir;
        if (onBoard(nf, nr)) {
            char p = s.b[sq(nf, nr)];
            if (p == (byWhite ? 'P' : 'p')) return true;
        }
    }
    const int kn[8][2] = {{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
    for (auto& d : kn) {
        int nf = f + d[0], nr = r + d[1];
        if (onBoard(nf, nr)) {
            char p = s.b[sq(nf, nr)];
            if (p == (byWhite ? 'N' : 'n')) return true;
        }
    }
    const int bd[4][2] = {{1,1},{1,-1},{-1,1},{-1,-1}};
    for (auto& d : bd) {
        int nf = f + d[0], nr = r + d[1];
        while (onBoard(nf, nr)) {
            char p = s.b[sq(nf, nr)];
            if (p) {
                if (p == (byWhite ? 'B' : 'b') || p == (byWhite ? 'Q' : 'q')) return true;
                break;
            }
            nf += d[0]; nr += d[1];
        }
    }
    const int rd[4][2] = {{1,0},{-1,0},{0,1},{0,-1}};
    for (auto& d : rd) {
        int nf = f + d[0], nr = r + d[1];
        while (onBoard(nf, nr)) {
            char p = s.b[sq(nf, nr)];
            if (p) {
                if (p == (byWhite ? 'R' : 'r') || p == (byWhite ? 'Q' : 'q')) return true;
                break;
            }
            nf += d[0]; nr += d[1];
        }
    }
    for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) {
        if (!df && !dr) continue;
        int nf = f + df, nr = r + dr;
        if (onBoard(nf, nr)) {
            char p = s.b[sq(nf, nr)];
            if (p == (byWhite ? 'K' : 'k')) return true;
        }
    }
    return false;
}

static int kingSq(const State& s, bool white) {
    for (int i = 0; i < 64; ++i) if (s.b[i] == (white ? 'K' : 'k')) return i;
    return -1;
}

static bool inCheck(const State& s, bool white) {
    int k = kingSq(s, white);
    return k >= 0 && attacked(s, k, !white);
}

static void addMove(vector<Move>& mv, int from, int to, bool cap=false, bool ep=false, bool castle=false, bool ks=false, char promo=0) {
    Move m;
    m.from = from; m.to = to; m.capture = cap; m.ep = ep; m.castle = castle; m.kingSide = ks; m.promo = promo;
    mv.push_back(m);
}

static vector<Move> genPseudo(const State& s) {
    vector<Move> mv;
    bool w = s.white;
    for (int i = 0; i < 64; ++i) {
        char p = s.b[i];
        if (!sameSide(p, w)) continue;
        int f = fileOf(i), r = rankOf(i);
        if (tolower((unsigned char)p) == 'p') {
            int dir = w ? 1 : -1;
            int nr = r + dir;
            if (onBoard(f, nr) && !s.b[sq(f, nr)]) {
                int to = sq(f, nr);
                if (nr == (w ? 7 : 0)) {
                    for (char pr : {'Q','R','B','N'}) addMove(mv, i, to, false, false, false, false, pr);
                } else addMove(mv, i, to);
                if ((w && r == 1) || (!w && r == 6)) {
                    int nr2 = r + 2 * dir;
                    if (!s.b[sq(f, nr2)]) addMove(mv, i, sq(f, nr2));
                }
            }
            for (int df : {-1, 1}) {
                int nf = f + df;
                if (!onBoard(nf, nr)) continue;
                int to = sq(nf, nr);
                char tp = s.b[to];
                if (enemySide(tp, w)) {
                    if (nr == (w ? 7 : 0)) {
                        for (char pr : {'Q','R','B','N'}) addMove(mv, i, to, true, false, false, false, pr);
                    } else addMove(mv, i, to, true);
                }
                if (to == s.ep) addMove(mv, i, to, true, true);
            }
        } else if (tolower((unsigned char)p) == 'n') {
            const int d[8][2] = {{1,2},{2,1},{2,-1},{1,-2},{-1,-2},{-2,-1},{-2,1},{-1,2}};
            for (auto& x : d) {
                int nf = f + x[0], nr = r + x[1];
                if (!onBoard(nf, nr)) continue;
                int to = sq(nf, nr);
                char tp = s.b[to];
                if (!sameSide(tp, w)) addMove(mv, i, to, tp != 0);
            }
        } else if (tolower((unsigned char)p) == 'b' || tolower((unsigned char)p) == 'r' || tolower((unsigned char)p) == 'q') {
            vector<pair<int,int>> ds;
            if (tolower((unsigned char)p) != 'r') ds.insert(ds.end(), {{1,1},{1,-1},{-1,1},{-1,-1}});
            if (tolower((unsigned char)p) != 'b') ds.insert(ds.end(), {{1,0},{-1,0},{0,1},{0,-1}});
            for (auto [df, dr] : ds) {
                int nf = f + df, nr = r + dr;
                while (onBoard(nf, nr)) {
                    int to = sq(nf, nr);
                    char tp = s.b[to];
                    if (!tp) addMove(mv, i, to);
                    else {
                        if (enemySide(tp, w)) addMove(mv, i, to, true);
                        break;
                    }
                    nf += df; nr += dr;
                }
            }
        } else if (tolower((unsigned char)p) == 'k') {
            for (int df = -1; df <= 1; ++df) for (int dr = -1; dr <= 1; ++dr) {
                if (!df && !dr) continue;
                int nf = f + df, nr = r + dr;
                if (!onBoard(nf, nr)) continue;
                int to = sq(nf, nr);
                char tp = s.b[to];
                if (!sameSide(tp, w)) addMove(mv, i, to, tp != 0);
            }
            if (w) {
                if (s.wkc && !s.b[sq(5,0)] && !s.b[sq(6,0)] && !inCheck(s, true) &&
                    !attacked(s, sq(5,0), false) && !attacked(s, sq(6,0), false)) addMove(mv, i, sq(6,0), false, false, true, true);
                if (s.wqc && !s.b[sq(1,0)] && !s.b[sq(2,0)] && !s.b[sq(3,0)] && !inCheck(s, true) &&
                    !attacked(s, sq(3,0), false) && !attacked(s, sq(2,0), false)) addMove(mv, i, sq(2,0), false, false, true, false);
            } else {
                if (s.bkc && !s.b[sq(5,7)] && !s.b[sq(6,7)] && !inCheck(s, false) &&
                    !attacked(s, sq(5,7), true) && !attacked(s, sq(6,7), true)) addMove(mv, i, sq(6,7), false, false, true, true);
                if (s.bqc && !s.b[sq(1,7)] && !s.b[sq(2,7)] && !s.b[sq(3,7)] && !inCheck(s, false) &&
                    !attacked(s, sq(3,7), true) && !attacked(s, sq(2,7), true)) addMove(mv, i, sq(2,7), false, false, true, false);
            }
        }
    }
    return mv;
}

static void makeMove(State& s, const Move& m, Undo& u) {
    u.captured = m.ep ? (s.white ? 'p' : 'P') : s.b[m.to];
    u.wkc = s.wkc; u.wqc = s.wqc; u.bkc = s.bkc; u.bqc = s.bqc;
    u.ep = s.ep; u.halfmove = s.halfmove; u.fullmove = s.fullmove;
    u.movedFrom = s.b[m.from]; u.movedTo = s.b[m.to];
    char p = s.b[m.from];
    s.b[m.from] = 0;
    s.ep = -1;
    if (tolower((unsigned char)p) == 'p' || u.captured) s.halfmove = 0; else ++s.halfmove;
    if (m.ep) {
        int capSq = s.white ? m.to - 8 : m.to + 8;
        s.b[capSq] = 0;
    }
    if (m.castle) {
        if (s.white) {
            if (m.kingSide) { s.b[sq(6,0)] = 'K'; s.b[sq(5,0)] = 'R'; s.b[sq(7,0)] = 0; }
            else { s.b[sq(2,0)] = 'K'; s.b[sq(3,0)] = 'R'; s.b[sq(0,0)] = 0; }
            s.wkc = s.wqc = false;
        } else {
            if (m.kingSide) { s.b[sq(6,7)] = 'k'; s.b[sq(5,7)] = 'r'; s.b[sq(7,7)] = 0; }
            else { s.b[sq(2,7)] = 'k'; s.b[sq(3,7)] = 'r'; s.b[sq(0,7)] = 0; }
            s.bkc = s.bqc = false;
        }
    } else {
        if (m.promo) p = s.white ? m.promo : char(tolower((unsigned char)m.promo));
        s.b[m.to] = p;
        if (p == 'K') s.wkc = s.wqc = false;
        if (p == 'k') s.bkc = s.bqc = false;
        if (m.from == sq(0,0) || m.to == sq(0,0)) s.wqc = false;
        if (m.from == sq(7,0) || m.to == sq(7,0)) s.wkc = false;
        if (m.from == sq(0,7) || m.to == sq(0,7)) s.bqc = false;
        if (m.from == sq(7,7) || m.to == sq(7,7)) s.bkc = false;
    }
    if (tolower((unsigned char)u.movedFrom) == 'k') {
        if (s.white) s.wkc = s.wqc = false; else s.bkc = s.bqc = false;
    }
    if (m.from == sq(4,0) || m.to == sq(4,0)) { s.wkc = s.wqc = false; }
    if (m.from == sq(4,7) || m.to == sq(4,7)) { s.bkc = s.bqc = false; }
    if (tolower((unsigned char)u.movedFrom) == 'p' && abs(rankOf(m.to) - rankOf(m.from)) == 2) {
        s.ep = s.white ? m.from + 8 : m.from - 8;
    }
    s.white = !s.white;
    if (s.white) ++s.fullmove;
}

static void undoMove(State& s, const Move& m, const Undo& u) {
    s.white = !s.white;
    s.wkc = u.wkc; s.wqc = u.wqc; s.bkc = u.bkc; s.bqc = u.bqc;
    s.ep = u.ep; s.halfmove = u.halfmove; s.fullmove = u.fullmove;
    if (m.castle) {
        if (s.white) {
            if (m.kingSide) { s.b[sq(4,0)] = 'K'; s.b[sq(7,0)] = 'R'; s.b[sq(6,0)] = 0; s.b[sq(5,0)] = 0; }
            else { s.b[sq(4,0)] = 'K'; s.b[sq(0,0)] = 'R'; s.b[sq(2,0)] = 0; s.b[sq(3,0)] = 0; }
        } else {
            if (m.kingSide) { s.b[sq(4,7)] = 'k'; s.b[sq(7,7)] = 'r'; s.b[sq(6,7)] = 0; s.b[sq(5,7)] = 0; }
            else { s.b[sq(4,7)] = 'k'; s.b[sq(0,7)] = 'r'; s.b[sq(2,7)] = 0; s.b[sq(3,7)] = 0; }
        }
    } else {
        s.b[m.from] = u.movedFrom;
        s.b[m.to] = u.captured;
        if (m.ep) {
            int capSq = s.white ? m.to - 8 : m.to + 8;
            s.b[capSq] = s.white ? 'p' : 'P';
            s.b[m.to] = 0;
        }
    }
}

static int evaluate(const State& s) {
    int score = 0;
    for (char p : s.b) {
        if (!p) continue;
        int v = pieceVal(p);
        score += whitePiece(p) ? v : -v;
    }
    return s.white ? score : -score;
}

static int moveScore(const State& s, const Move& m) {
    int sc = 0;
    if (m.castle) sc += 50;
    if (m.capture) {
        char cap = m.ep ? (s.white ? 'p' : 'P') : s.b[m.to];
        char p = s.b[m.from];
        sc += 10 * pieceVal(cap) - pieceVal(p);
    }
    if (m.promo) sc += pieceVal(m.promo) + 800;
    return sc;
}

static int qsearch(State& s, int alpha, int beta) {
    if (timeUp()) throw runtime_error("timeout");
    int stand = evaluate(s);
    if (stand >= beta) return beta;
    alpha = max(alpha, stand);
    auto moves = genPseudo(s);
    sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b){ return moveScore(s,a) > moveScore(s,b); });
    for (auto& m : moves) {
        if (!m.capture && !m.promo && !m.ep) continue;
        Undo u;
        makeMove(s, m, u);
        if (inCheck(s, !s.white)) { undoMove(s, m, u); continue; }
        int sc = -qsearch(s, -beta, -alpha);
        undoMove(s, m, u);
        if (sc >= beta) return beta;
        alpha = max(alpha, sc);
    }
    return alpha;
}

static int negamax(State& s, int depth, int alpha, int beta, Move* pvMove) {
    if (timeUp()) throw runtime_error("timeout");
    bool side = s.white;
    auto moves = genPseudo(s);
    vector<Move> legal;
    legal.reserve(moves.size());
    for (auto& m : moves) {
        Undo u;
        makeMove(s, m, u);
        if (!inCheck(s, !s.white)) legal.push_back(m);
        undoMove(s, m, u);
    }
    if (legal.empty()) return inCheck(s, side) ? -100000 + (5 - depth) : 0;
    sort(legal.begin(), legal.end(), [&](const Move& a, const Move& b){ return moveScore(s,a) > moveScore(s,b); });
    int best = -10000000;
    Move bestm;
    for (auto& m : legal) {
        Undo u;
        makeMove(s, m, u);
        int sc;
        if (depth <= 1) sc = -qsearch(s, -beta, -alpha);
        else sc = -negamax(s, depth - 1, -beta, -alpha, nullptr);
        undoMove(s, m, u);
        if (sc > best) { best = sc; bestm = m; }
        alpha = max(alpha, sc);
        if (alpha >= beta) break;
    }
    if (pvMove) *pvMove = bestm;
    return best;
}

static string formatMove(const State& s, const Move& m) {
    if (m.castle) return m.kingSide ? "O-O" : "O-O-O";
    char p = s.b[m.from];
    string out;
    out += (char)tolower((unsigned char)p);
    out += sqStr(m.from);
    out += m.capture ? 'x' : '';
    out += sqStr(m.to);
    if (m.ep) out += "ep";
    if (m.promo) { out += '/'; out += m.promo; }
    State t = s;
    Undo u;
    makeMove(t, m, u);
    bool check = inCheck(t, t.white);
    auto replies = genPseudo(t);
    bool mate = check;
    if (check) {
        for (auto& rm : replies) {
            Undo uu;
            makeMove(t, rm, uu);
            if (!inCheck(t, !t.white)) { mate = false; undoMove(t, rm, uu); break; }
            undoMove(t, rm, uu);
        }
    }
    if (mate) out += '#'; else if (check) out += '+';
    return out;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeoutSec;
    string fen;
    if (!(cin >> timeoutSec)) return 0;
    getline(cin, fen);
    if (fen.empty()) getline(cin, fen);
    if (!fen.empty() && fen[0] == ' ') fen.erase(fen.begin());

    State root = parseFen(fen);
    g_deadline = chrono::steady_clock::now() + chrono::milliseconds(max(100LL, (long long)(timeoutSec * 1000.0) - 50));

    auto rootMoves = genPseudo(root);
    vector<Move> legal;
    for (auto& m : rootMoves) {
        Undo u;
        makeMove(root, m, u);
        if (!inCheck(root, !root.white)) legal.push_back(m);
        undoMove(root, m, u);
    }
    if (legal.empty()) {
        cout << "O-O\n";
        return 0;
    }

    sort(legal.begin(), legal.end(), [&](const Move& a, const Move& b){ return moveScore(root,a) > moveScore(root,b); });
    g_bestMove = legal.front();
    g_bestScore = -10000000;

    try {
        for (int depth = 1; ; ++depth) {
            int alpha = -10000000, beta = 10000000;
            Move best;
            int bestScore = -10000000;
            for (auto& m : legal) {
                if (timeUp()) throw runtime_error("timeout");
                Undo u;
                makeMove(root, m, u);
                int sc = (depth <= 1) ? -qsearch(root, -beta, -alpha) : -negamax(root, depth - 1, -beta, -alpha, nullptr);
                undoMove(root, m, u);
                if (sc > bestScore) { bestScore = sc; best = m; }
                alpha = max(alpha, sc);
            }
            g_bestMove = best;
            g_bestScore = bestScore;
            if (bestScore > 900000) break;
        }
    } catch (...) {
    }

    cout << formatMove(root, g_bestMove) << "\n";
    return 0;
}
