use std::collections::{HashMap, HashSet};
use std::io::{self, Read};

fn parse_input() -> Option<(usize, i64, Vec<i64>)> {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok()?;
    let mut it = input.split_whitespace();
    let n: usize = it.next()?.parse().ok()?;
    let target: i64 = it.next()?.parse().ok()?;
    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        nums.push(it.next()?.parse().ok()?);
    }
    Some((n, target, nums))
}

fn output_yes(indices: &[usize]) {
    let mut out = String::new();
    out.push_str("YES\n");
    out.push('[');
    for (i, idx) in indices.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&idx.to_string());
    }
    out.push_str("]\n");
    print!("{out}");
}

fn output_no() {
    print!("NO\n");
}

fn exact_mitm(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    let mut left_map: HashMap<i64, Vec<usize>> = HashMap::new();
    let lcount = 1usize << left.len();
    for mask in 0..lcount {
        let mut sum = 0i64;
        let mut idxs = Vec::new();
        for i in 0..left.len() {
            if (mask >> i) & 1 == 1 {
                sum += left[i];
                idxs.push(i);
            }
        }
        left_map.entry(sum).or_insert(idxs);
    }

    let rcount = 1usize << right.len();
    for mask in 0..rcount {
        let mut sum = 0i64;
        let mut idxs = Vec::new();
        for i in 0..right.len() {
            if (mask >> i) & 1 == 1 {
                sum += right[i];
                idxs.push(mid + i);
            }
        }
        let need = target - sum;
        if let Some(left_idxs) = left_map.get(&need) {
            let mut ans = left_idxs.clone();
            ans.extend(idxs);
            return Some(ans);
        }
    }
    None
}

fn exact_hash_dp(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut states: HashMap<i64, Vec<usize>> = HashMap::new();
    states.insert(0, Vec::new());
    for (i, &v) in nums.iter().enumerate() {
        let mut next = states.clone();
        for (&sum, idxs) in &states {
            let ns = sum + v;
            if !next.contains_key(&ns) {
                let mut nidxs = idxs.clone();
                nidxs.push(i);
                next.insert(ns, nidxs);
            }
        }
        states = next;
        if states.contains_key(&target) {
            return states.get(&target).cloned();
        }
        if states.len() > 300_000 {
            break;
        }
    }
    states.get(&target).cloned()
}

fn greedy_fallback(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut pairs: Vec<(i64, usize)> = nums.iter().copied().enumerate().map(|(i, v)| (v, i)).collect();
    pairs.sort_by_key(|&(v, _)| -v.abs());
    let mut sum = 0i64;
    let mut chosen = Vec::new();
    let mut seen = HashSet::new();
    seen.insert(0i64);
    for (v, i) in pairs {
        if sum == target {
            break;
        }
        let ns = sum + v;
        if !seen.contains(&ns) {
            chosen.push(i);
            sum = ns;
            seen.insert(sum);
        }
    }
    if sum == target { Some(chosen) } else { None }
}

fn main() {
    let Some((n, target, nums)) = parse_input() else {
        return;
    };

    if target == 0 {
        output_yes(&[]);
        return;
    }
    if n == 0 {
        output_no();
        return;
    }
    if let Some(ans) = if n <= 42 {
        exact_mitm(&nums, target)
    } else {
        exact_hash_dp(&nums, target).or_else(|| if n <= 2000 { exact_mitm(&nums, target) } else { None })
    } {
        output_yes(&ans);
        return;
    }
    if let Some(ans) = greedy_fallback(&nums, target) {
        if ans.iter().map(|&i| nums[i]).sum::<i64>() == target {
            output_yes(&ans);
            return;
        }
    }
    output_no();
}
