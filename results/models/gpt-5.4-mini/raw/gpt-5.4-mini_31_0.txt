use std::collections::HashMap;
use std::io::{self, Read};

fn parse_input() -> Option<(usize, i64, Vec<i64>)> {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).ok()?;
    let mut it = input.split_whitespace();

    let n: usize = it.next()?.parse().ok()?;
    let target: i64 = it.next()?.parse().ok()?;

    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(tok) = it.next() {
            if let Ok(v) = tok.parse::<i64>() {
                nums.push(v);
            } else {
                return None;
            }
        } else {
            break;
        }
    }

    Some((n, target, nums))
}

fn reconstruct(parent: &[Option<(u64, usize)>], end_mask: u64) -> Vec<usize> {
    let mut indices = Vec::new();
    let mut mask = end_mask;
    while mask != 0 {
        let bit = mask.trailing_zeros() as usize;
        indices.push(bit);
        if let Some((prev, _idx)) = parent[bit] {
            mask = prev;
        } else {
            break;
        }
    }
    indices.reverse();
    indices
}

fn subset_sum_small(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    if n == 0 {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let total_masks = if n >= 60 { None } else { Some(1u64 << n) };
    if let Some(limit) = total_masks {
        if limit <= 1_000_000 {
            let mut s = 0u64;
            while s < limit {
                let mut sum = 0i64;
                let mut subset = Vec::new();
                let mut m = s;
                let mut i = 0usize;
                while m != 0 {
                    if (m & 1) != 0 {
                        sum += nums[i];
                        subset.push(i);
                    }
                    i += 1;
                    m >>= 1;
                }
                if sum == target {
                    return Some(subset);
                }
                s += 1;
            }
        }
    }

    None
}

fn subset_sum_dp_nonnegative(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = target as usize;
    let mut parent_sum: Vec<Option<usize>> = vec![None; t + 1];
    let mut parent_idx: Vec<Option<usize>> = vec![None; t + 1];
    parent_sum[0] = Some(0);

    for (i, &x) in nums.iter().enumerate() {
        if x < 0 {
            return None;
        }
        let v = x as usize;
        if v > t {
            continue;
        }
        for s in (v..=t).rev() {
            if parent_sum[s].is_none() && parent_sum[s - v].is_some() {
                parent_sum[s] = Some(s - v);
                parent_idx[s] = Some(i);
            }
        }
    }

    if parent_sum[t].is_none() {
        return None;
    }

    let mut result = Vec::new();
    let mut cur = t;
    while cur != 0 {
        let idx = parent_idx[cur]?;
        result.push(idx);
        cur = parent_sum[cur]?;
    }
    result.reverse();
    Some(result)
}

fn meet_in_middle(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left = &nums[..mid];
    let right = &nums[mid..];

    let mut left_map: HashMap<i64, u64> = HashMap::new();
    let left_limit = 1u64.checked_shl(left.len() as u32).unwrap_or(0);
    if left_limit == 0 {
        return None;
    }

    let mut mask = 0u64;
    while mask < left_limit {
        let mut sum = 0i64;
        let mut m = mask;
        let mut i = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                sum += left[i];
            }
            i += 1;
            m >>= 1;
        }
        left_map.entry(sum).or_insert(mask);
        mask += 1;
    }

    let right_limit = 1u64.checked_shl(right.len() as u32).unwrap_or(0);
    if right_limit == 0 {
        return None;
    }

    let mut rmask = 0u64;
    while rmask < right_limit {
        let mut sum = 0i64;
        let mut m = rmask;
        let mut i = 0usize;
        while m != 0 {
            if (m & 1) != 0 {
                sum += right[i];
            }
            i += 1;
            m >>= 1;
        }

        let need = target - sum;
        if let Some(&lmask) = left_map.get(&need) {
            let mut ans = Vec::new();
            let mut lm = lmask;
            let mut i = 0usize;
            while lm != 0 {
                if (lm & 1) != 0 {
                    ans.push(i);
                }
                i += 1;
                lm >>= 1;
            }
            let mut rm = rmask;
            let mut j = 0usize;
            while rm != 0 {
                if (rm & 1) != 0 {
                    ans.push(mid + j);
                }
                j += 1;
                rm >>= 1;
            }
            ans.sort_unstable();
            return Some(ans);
        }

        rmask += 1;
    }

    None
}

fn subset_sum_signed(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if nums.is_empty() {
        return if target == 0 { Some(Vec::new()) } else { None };
    }

    let all_nonnegative = nums.iter().all(|&x| x >= 0);
    if all_nonnegative && target >= 0 {
        let sum: i128 = nums.iter().map(|&x| x as i128).sum();
        if sum <= 2_000_000_i128 && target <= 2_000_000_i64 {
            if let Some(ans) = subset_sum_dp_nonnegative(nums, target) {
                return Some(ans);
            }
        }
    }

    if nums.len() <= 44 {
        if let Some(ans) = meet_in_middle(nums, target) {
            return Some(ans);
        }
    }

    if nums.len() <= 24 {
        if let Some(ans) = subset_sum_small(nums, target) {
            return Some(ans);
        }
    }

    None
}

fn main() {
    let (n, target, mut nums) = match parse_input() {
        Some(v) => v,
        None => return,
    };

    if nums.len() > n {
        nums.truncate(n);
    }

    if target == 0 {
        println!("YES");
        println!();
        return;
    }

    if nums.is_empty() {
        println!("NO");
        return;
    }

    if let Some(ans) = subset_sum_signed(&nums, target) {
        println!("YES");
        if ans.is_empty() {
            println!();
        } else {
            let out = ans
                .iter()
                .map(|v| v.to_string())
                .collect::<Vec<_>>()
                .join(" ");
            println!("{out}");
        }
    } else {
        println!("NO");
    }
}
