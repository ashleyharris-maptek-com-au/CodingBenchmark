#include <bits/stdc++.h>
using namespace std;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<vector<int>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            c[u] = LLONG_MIN / 4;
            continue;
        }
        g[u].push_back(v);
        g[v].push_back(u);
    }

    // Compress away variables that are forced to 0 by self-constraints.
    vector<int> active_id(n, -1);
    vector<int> idx_to_var;
    idx_to_var.reserve(n);
    for (int i = 0; i < n; ++i) {
        if (c[i] > LLONG_MIN / 8) {
            active_id[i] = (int)idx_to_var.size();
            idx_to_var.push_back(i);
        }
    }

    int N = (int)idx_to_var.size();
    vector<long long> w(N);
    vector<vector<int>> adj(N);
    for (int ii = 0; ii < N; ++ii) {
        int i = idx_to_var[ii];
        w[ii] = c[i];
        for (int j : g[i]) {
            int aj = active_id[j];
            if (aj != -1 && aj != ii) adj[ii].push_back(aj);
        }
    }
    for (int i = 0; i < N; ++i) {
        sort(adj[i].begin(), adj[i].end());
        adj[i].erase(unique(adj[i].begin(), adj[i].end()), adj[i].end());
    }

    // Branch and bound for maximum-weight independent set on the conflict graph.
    vector<int> order(N);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        long long sa = llabs(w[a]) + (long long)adj[a].size();
        long long sb = llabs(w[b]) + (long long)adj[b].size();
        if (sa != sb) return sa > sb;
        return a < b;
    });

    vector<int> pos(N);
    for (int i = 0; i < N; ++i) pos[order[i]] = i;
    vector<vector<int>> ord_adj(N);
    for (int i = 0; i < N; ++i) {
        ord_adj[i] = adj[i];
        sort(ord_adj[i].begin(), ord_adj[i].end(), [&](int a, int b) { return pos[a] < pos[b]; });
    }

    vector<int> best_take(N, 0), cur_take(N, 0);
    long long best_val = LLONG_MIN;

    function<void(vector<int>&, long long)> dfs = [&](vector<int>& cand, long long cur_val) {
        if (cand.empty()) {
            if (cur_val > best_val) {
                best_val = cur_val;
                best_take = cur_take;
            }
            return;
        }

        long long ub = cur_val;
        for (int v : cand) if (w[v] > 0) ub += w[v];
        if (ub <= best_val) return;

        int v = cand.back();
        cand.pop_back();

        // Branch 1: exclude v.
        dfs(cand, cur_val);

        // Branch 2: include v.
        vector<int> next;
        next.reserve(cand.size());
        for (int u : cand) {
            if (!binary_search(adj[v].begin(), adj[v].end(), u)) next.push_back(u);
        }
        cur_take[v] = 1;
        dfs(next, cur_val + w[v]);
        cur_take[v] = 0;

        cand.push_back(v);
    };

    vector<int> cand = order;
    if (N == 0) {
        cout << 0 << '\n';
        for (int i = 0; i < n; ++i) {
            if (i) cout << ' ';
            cout << 0;
        }
        cout << '\n';
        return 0;
    }

    best_val = 0;
    dfs(cand, 0);

    vector<int> ans(n, 0);
    for (int i = 0; i < N; ++i) ans[idx_to_var[i]] = best_take[i];

    long long objective = 0;
    for (int i = 0; i < n; ++i) if (ans[i]) objective += c[i];

    cout << objective << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
