#include <bits/stdc++.h>
using namespace std;

struct Solver {
    int n;
    vector<long long> c;
    vector<vector<int>> adj;
    vector<char> forced0, forced1;

    long long bestValue = LLONG_MIN;
    vector<char> bestAssign;

    long long greedyBound(const vector<int>& verts) const {
        long long s = 0;
        for (int v : verts) if (!forced0[v]) s += max(0LL, c[v]);
        return s;
    }

    int pickVertex(const vector<int>& verts) const {
        int best = -1;
        int bestDeg = -1;
        for (int v : verts) {
            if (forced0[v] || forced1[v]) continue;
            int deg = 0;
            for (int to : adj[v]) if (!forced0[to] && !forced1[to]) ++deg;
            if (deg > bestDeg || (deg == bestDeg && c[v] > c[best])) {
                bestDeg = deg;
                best = v;
            }
        }
        return best;
    }

    bool propagate(vector<int>& stack) {
        while (!stack.empty()) {
            int v = stack.back();
            stack.pop_back();
            if (forced1[v]) {
                for (int to : adj[v]) {
                    if (forced1[to]) return false;
                    if (!forced0[to]) {
                        forced0[to] = 1;
                        stack.push_back(to);
                    }
                }
            } else if (forced0[v]) {
                continue;
            } else {
                forced1[v] = 1;
            }
        }
        return true;
    }

    void dfs(long long currentValue, vector<char>& assign) {
        vector<int> active;
        active.reserve(n);
        for (int i = 0; i < n; ++i) {
            if (!forced0[i] && !forced1[i]) active.push_back(i);
        }

        long long ub = currentValue + greedyBound(active);
        if (ub <= bestValue) return;

        int v = pickVertex(active);
        if (v == -1) {
            if (currentValue > bestValue) {
                bestValue = currentValue;
                bestAssign = assign;
            }
            return;
        }

        {
            auto f0 = forced0, f1 = forced1, a = assign;
            forced1[v] = 1;
            a[v] = 1;
            vector<int> stack = {v};
            if (propagate(stack)) {
                long long nv = currentValue + c[v];
                for (int to : adj[v]) if (!f0[to] && !f1[to]) {
                    if (a[to] == 1) nv -= c[to];
                }
                dfs(nv, a);
            }
            forced0 = move(f0);
            forced1 = move(f1);
        }

        {
            auto f0 = forced0, f1 = forced1, a = assign;
            forced0[v] = 1;
            a[v] = 0;
            vector<int> stack = {v};
            if (propagate(stack)) dfs(currentValue, a);
            forced0 = move(f0);
            forced1 = move(f1);
        }
    }

    vector<int> solve() {
        forced0.assign(n, 0);
        forced1.assign(n, 0);
        bestAssign.assign(n, 0);
        vector<char> assign(n, 0);

        // Isolated positive vertices can be taken greedily.
        for (int i = 0; i < n; ++i) {
            if (adj[i].empty() && c[i] > 0) {
                forced1[i] = 1;
                assign[i] = 1;
            }
        }
        long long start = 0;
        for (int i = 0; i < n; ++i) if (forced1[i]) start += c[i];

        dfs(start, assign);

        vector<int> result(n, 0);
        for (int i = 0; i < n; ++i) result[i] = bestAssign[i] ? 1 : 0;
        return result;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    Solver s;
    s.n = n;
    s.c.resize(n);
    for (int i = 0; i < n; ++i) cin >> s.c[i];
    s.adj.assign(n, {});
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u == v) continue;
        s.adj[u].push_back(v);
        s.adj[v].push_back(u);
    }

    vector<int> x = s.solve();
    long long obj = 0;
    for (int i = 0; i < n; ++i) if (x[i]) obj += s.c[i];

    cout << obj << '\n';
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << x[i];
    }
    cout << '\n';
    return 0;
}
