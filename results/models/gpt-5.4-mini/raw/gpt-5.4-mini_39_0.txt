use std::cmp::Reverse;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();
    let words = (n + 63) / 64;
    let mut adj = vec![vec![0u64; words]; n];
    let mut deg = vec![0usize; n];

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        let (wu, bu) = (v / 64, v % 64);
        let (wv, bv) = (u / 64, u % 64);
        if (adj[u][wu] >> bu) & 1 == 0 {
            adj[u][wu] |= 1u64 << bu;
            adj[v][wv] |= 1u64 << bv;
            deg[u] += 1;
            deg[v] += 1;
        }
    }

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_by_key(|&v| Reverse(deg[v]));

    let mut pos = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        pos[v] = i;
    }

    let mut adj_ord = vec![vec![0u64; words]; n];
    for i in 0..n {
        let v = order[i];
        for j in 0..n {
            if ((adj[v][j / 64] >> (j % 64)) & 1) == 1 {
                let pj = pos[j];
                adj_ord[i][pj / 64] |= 1u64 << (pj % 64);
            }
        }
    }

    let mut best_cut = usize::MAX;
    let mut best_mask = vec![false; n];
    let mut chosen = vec![false; n];

    fn search(
        idx: usize,
        chosen_cnt: usize,
        cut_so_far: usize,
        n: usize,
        adj: &Vec<Vec<u64>>,
        chosen: &mut Vec<bool>,
        best_cut: &mut usize,
        best_mask: &mut Vec<bool>,
    ) {
        if cut_so_far >= *best_cut {
            return;
        }
        if chosen_cnt > n / 2 {
            return;
        }
        let remaining = n - idx;
        if chosen_cnt + remaining < n / 2 {
            return;
        }
        if idx == n {
            if chosen_cnt == n / 2 && cut_so_far < *best_cut {
                *best_cut = cut_so_far;
                *best_mask = chosen.clone();
            }
            return;
        }

        let need = n / 2 - chosen_cnt;
        if need == remaining {
            let mut add = 0usize;
            for j in 0..idx {
                if chosen[j] {
                    continue;
                }
                if ((adj[idx][j / 64] >> (j % 64)) & 1) == 1 {
                    add += 1;
                }
            }
            chosen[idx] = true;
            search(idx + 1, chosen_cnt + 1, cut_so_far + add, n, adj, chosen, best_cut, best_mask);
            chosen[idx] = false;
            return;
        }
        if chosen_cnt == n / 2 {
            let mut add = 0usize;
            for j in 0..idx {
                if chosen[j] {
                    if ((adj[idx][j / 64] >> (j % 64)) & 1) == 1 {
                        add += 1;
                    }
                }
            }
            chosen[idx] = false;
            search(idx + 1, chosen_cnt, cut_so_far + add, n, adj, chosen, best_cut, best_mask);
            return;
        }

        let mut add_in = 0usize;
        let mut add_out = 0usize;
        for j in 0..idx {
            let edge = ((adj[idx][j / 64] >> (j % 64)) & 1) == 1;
            if edge {
                if chosen[j] {
                    add_out += 1;
                } else {
                    add_in += 1;
                }
            }
        }

        chosen[idx] = true;
        search(idx + 1, chosen_cnt + 1, cut_so_far + add_out, n, adj, chosen, best_cut, best_mask);
        chosen[idx] = false;

        search(idx + 1, chosen_cnt, cut_so_far + add_in, n, adj, chosen, best_cut, best_mask);
    }

    search(0, 0, 0, n, &adj_ord, &mut chosen, &mut best_cut, &mut best_mask);

    let mut part_a = Vec::new();
    for i in 0..n {
        if best_mask[i] {
            part_a.push(order[i]);
        }
    }
    part_a.sort_unstable();

    println!("{}", best_cut);
    for (i, v) in part_a.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
