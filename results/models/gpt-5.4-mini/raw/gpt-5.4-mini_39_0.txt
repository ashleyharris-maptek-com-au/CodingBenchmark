use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let words = (n + 63) / 64;
    let mut adj = vec![vec![0u64; words]; n];
    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            continue;
        }
        adj[u][v / 64] |= 1u64 << (v % 64);
        adj[v][u / 64] |= 1u64 << (u % 64);
    }

    if n == 0 {
        println!("0");
        println!();
        return;
    }

    let half = n / 2;
    let mut degree: Vec<(usize, usize)> = (0..n)
        .map(|v| {
            let d = adj[v].iter().map(|w| w.count_ones() as usize).sum();
            (d, v)
        })
        .collect();
    degree.sort_by(|a, b| b.cmp(a));
    let order: Vec<usize> = degree.into_iter().map(|(_, v)| v).collect();

    let mut pos = vec![0usize; n];
    for (i, &v) in order.iter().enumerate() {
        pos[v] = i;
    }

    let mut suffix_edges = vec![0usize; n + 1];
    for i in (0..n).rev() {
        let v = order[i];
        let mut cnt = 0usize;
        for j in (i + 1)..n {
            let u = order[j];
            let edge = ((adj[v][u / 64] >> (u % 64)) & 1) != 0;
            if edge {
                cnt += 1;
            }
        }
        suffix_edges[i] = suffix_edges[i + 1] + cnt;
    }

    let mut best_cut = usize::MAX;
    let mut best_mask = vec![false; n];
    let mut cur_mask = vec![false; n];
    cur_mask[pos[0]] = true;

    fn count_cut(mask: &Vec<bool>, adj: &Vec<Vec<u64>>, order: &Vec<usize>) -> usize {
        let n = order.len();
        let mut cut = 0usize;
        for i in 0..n {
            let a = mask[i];
            let v = order[i];
            for j in (i + 1)..n {
                let edge = ((adj[v][order[j] / 64] >> (order[j] % 64)) & 1) != 0;
                if edge && a != mask[j] {
                    cut += 1;
                }
            }
        }
        cut
    }

    let mut mandatory_a = 1usize;
    let mut mandatory_b = 0usize;

    fn lower_bound(
        idx: usize,
        order: &Vec<usize>,
        mask: &Vec<bool>,
        adj: &Vec<Vec<u64>>,
        cur_cut: usize,
    ) -> usize {
        let n = order.len();
        let mut lb = cur_cut;
        for i in idx..n {
            let v = order[i];
            let mut to_a = 0usize;
            let mut to_b = 0usize;
            for j in 0..idx {
                let u = order[j];
                let edge = ((adj[v][u / 64] >> (u % 64)) & 1) != 0;
                if !edge {
                    continue;
                }
                if mask[j] {
                    to_a += 1;
                } else {
                    to_b += 1;
                }
            }
            lb += to_a.min(to_b);
        }
        lb
    }

    fn dfs(
        idx: usize,
        order: &Vec<usize>,
        pos: &Vec<usize>,
        adj: &Vec<Vec<u64>>,
        mask: &mut Vec<bool>,
        a_count: usize,
        b_count: usize,
        cur_cut: usize,
        best_cut: &mut usize,
        best_mask: &mut Vec<bool>,
        half: usize,
    ) {
        let n = order.len();
        if cur_cut >= *best_cut {
            return;
        }
        if a_count > half || b_count > half {
            return;
        }
        if idx == n {
            if a_count == half && b_count == half && cur_cut < *best_cut {
                *best_cut = cur_cut;
                *best_mask = mask.clone();
            }
            return;
        }

        let lb = lower_bound(idx, order, mask, adj, cur_cut);
        if lb >= *best_cut {
            return;
        }

        let v = order[idx];
        if v == 0 {
            dfs(
                idx + 1,
                order,
                pos,
                adj,
                mask,
                a_count + 1,
                b_count,
                cur_cut,
                best_cut,
                best_mask,
                half,
            );
            return;
        }

        let remaining = n - idx;
        if a_count + remaining < half || b_count + remaining < half {
            return;
        }

        let mut add_a = 0usize;
        let mut add_b = 0usize;
        for j in 0..idx {
            let u = order[j];
            let edge = ((adj[v][u / 64] >> (u % 64)) & 1) != 0;
            if !edge {
                continue;
            }
            if mask[j] {
                add_b += 1;
            } else {
                add_a += 1;
            }
        }

        // Try the side that looks cheaper first.
        if add_a <= add_b {
            mask[idx] = true;
            dfs(
                idx + 1,
                order,
                pos,
                adj,
                mask,
                a_count + 1,
                b_count,
                cur_cut + add_a,
                best_cut,
                best_mask,
                half,
            );
            mask[idx] = false;

            mask[idx] = false;
            dfs(
                idx + 1,
                order,
                pos,
                adj,
                mask,
                a_count,
                b_count + 1,
                cur_cut + add_b,
                best_cut,
                best_mask,
                half,
            );
        } else {
            mask[idx] = false;
            dfs(
                idx + 1,
                order,
                pos,
                adj,
                mask,
                a_count,
                b_count + 1,
                cur_cut + add_b,
                best_cut,
                best_mask,
                half,
            );

            mask[idx] = true;
            dfs(
                idx + 1,
                order,
                pos,
                adj,
                mask,
                a_count + 1,
                b_count,
                cur_cut + add_a,
                best_cut,
                best_mask,
                half,
            );
            mask[idx] = false;
        }
    }

    // Fast initial solution from greedy swapping to tighten the bound.
    let mut greedy_mask = vec![false; n];
    greedy_mask[pos[0]] = true;
    let mut a_cnt = 1usize;
    for &v in order.iter().skip(1) {
        if a_cnt == half {
            break;
        }
        greedy_mask[pos[v]] = true;
        a_cnt += 1;
    }
    let mut greedy_cut = 0usize;
    for i in 0..n {
        let vi = order[i];
        for j in (i + 1)..n {
            let vj = order[j];
            let edge = ((adj[vi][vj / 64] >> (vj % 64)) & 1) != 0;
            if edge && greedy_mask[i] != greedy_mask[j] {
                greedy_cut += 1;
            }
        }
    }
    best_cut = greedy_cut;
    best_mask = greedy_mask.clone();

    dfs(
        0,
        &order,
        &pos,
        &adj,
        &mut cur_mask,
        0,
        0,
        0,
        &mut best_cut,
        &mut best_mask,
        half,
    );

    let mut part = Vec::with_capacity(half);
    for i in 0..n {
        if best_mask[pos[i]] {
            part.push(i);
        }
    }
    part.sort_unstable();

    let mut actual_cut = 0usize;
    for i in 0..n {
        for j in (i + 1)..n {
            let edge = ((adj[i][j / 64] >> (j % 64)) & 1) != 0;
            if edge && best_mask[pos[i]] != best_mask[pos[j]] {
                actual_cut += 1;
            }
        }
    }

    println!("{}", actual_cut);
    for (i, v) in part.iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{}", v);
    }
    println!();
}
