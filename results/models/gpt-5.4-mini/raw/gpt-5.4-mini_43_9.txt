float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv * 8.0;
    float checker = fmod(floor(uv.x) + floor(uv.y), 2.0);
    float baseColor = lerp(0.9, 0.2, checker);

    float3 n = normalize(input.normal);
    float3 l = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(n, l), 0.0);
    float lighting = 0.5 + 0.5 * diffuse;

    float rgb = baseColor * lighting;
    return float4(rgb, rgb, rgb, 1.0);
}
