```python
import math

_S = {
    "alt_i": 0.0,
    "spd_i": 0.0,
    "hdg_i": 0.0,
    "vs_i": 0.0,
    "trim": 0.0,
    "last_alt": None,
    "last_hdg": None,
    "last_time": None,
}


def _wrap180(x):
    if x is None:
        return 0.0
    return (x + 180.0) % 360.0 - 180.0


def _wrap360(x):
    if x is None:
        return 0.0
    return x % 360.0


def _clamp(x, lo, hi):
    return lo if x < lo else hi if x > hi else x


def _median(vals):
    vals = [v for v in vals if v is not None and not isinstance(v, bool)]
    if not vals:
        return None
    vals = sorted(vals)
    n = len(vals)
    mid = n // 2
    return vals[mid] if n % 2 else 0.5 * (vals[mid - 1] + vals[mid])


def _vote_triplet(state, names, valid_names=None, default=None):
    vals = []
    for i, n in enumerate(names):
        v = state.get(n)
        ok = True
        if valid_names is not None:
            ok = bool(state.get(valid_names[i], True))
        if ok and isinstance(v, (int, float)) and math.isfinite(v):
            vals.append(v)
    med = _median(vals)
    if med is not None:
        return med
    if default is not None:
        return default
    for n in names:
        v = state.get(n)
        if isinstance(v, (int, float)) and math.isfinite(v):
            return v
    return 0.0


def _best_heading(state):
    cands = []
    for k in ("target_hdg_deg", "magnetic_heading_deg", "true_heading_deg", "track_deg"):
        v = state.get(k)
        if isinstance(v, (int, float)) and math.isfinite(v):
            cands.append(v)
    return _median(cands) if cands else 0.0


def _estimate_altitude(state):
    baro = _vote_triplet(
        state,
        ["baro_alt_1_ft", "baro_alt_2_ft", "baro_alt_3_ft"],
        ["baro_alt_1_valid", "baro_alt_2_valid", "baro_alt_3_valid"],
    )
    radar = _vote_triplet(
        state,
        ["radar_alt_1_ft", "radar_alt_2_ft", "radar_alt_3_ft"],
        ["radar_alt_1_valid", "radar_alt_2_valid", "radar_alt_3_valid"],
    )
    gps = _vote_triplet(
        state,
        ["gps_1_alt_ft", "gps_2_alt_ft", "gps_3_alt_ft"],
        ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
    )
    adc = _vote_triplet(
        state,
        ["adc_1_pressure_alt_ft", "adc_2_pressure_alt_ft", "adc_3_pressure_alt_ft"],
        ["adc_1_valid", "adc_2_valid", "adc_3_valid"],
    )
    candidates = [v for v in [baro, radar, gps, adc] if v is not None]
    if not candidates:
        return 0.0
    if radar is not None and radar < 2500.0:
        near = [v for v in [radar, gps, baro] if v is not None]
        return _median(near)
    spread = max(candidates) - min(candidates)
    if spread > 1000.0 and radar is not None:
        if gps is not None and abs(gps - radar) < abs(baro - radar):
            return 0.7 * gps + 0.3 * radar
        return 0.7 * baro + 0.3 * radar
    return _median(candidates)


def _estimate_speed(state):
    ias = _vote_triplet(
        state,
        ["ias_1_kt", "ias_2_kt", "ias_3_kt"],
        ["ias_1_valid", "ias_2_valid", "ias_3_valid"],
    )
    tas = _vote_triplet(state, ["tas_1_kt", "tas_2_kt", "tas_3_kt"])
    gs = _vote_triplet(
        state,
        ["gps_1_gs_kt", "gps_2_gs_kt", "gps_3_gs_kt"],
        ["gps_1_valid", "gps_2_valid", "gps_3_valid"],
    )
    cands = [v for v in [ias, tas, gs] if v is not None]
    return _median(cands) if cands else 0.0


def _estimate_vertical(state, alt_est, dt):
    vs = state.get("vertical_speed_fpm")
    irs_vs = _vote_triplet(
        state,
        ["irs_1_vs_fpm", "irs_2_vs_fpm", "irs_3_vs_fpm"],
        ["irs_1_valid", "irs_2_valid", "irs_3_valid"],
    )
    if isinstance(vs, (int, float)) and math.isfinite(vs):
        a = 0.6 * vs + 0.4 * irs_vs
    else:
        a = irs_vs
    if _S["last_alt"] is not None and dt > 1e-3:
        b = (alt_est - _S["last_alt"]) / dt * 60.0
        if abs(b - a) > 1500.0:
            a = 0.7 * b + 0.3 * a
        else:
            a = 0.5 * a + 0.5 * b
    _S["last_alt"] = alt_est
    return a


def autopilot_step(state: dict, dt: float) -> dict:
    dt = float(dt if isinstance(dt, (int, float)) and dt > 1e-3 else state.get("dt", 0.05) or 0.05)
    ap = bool(state.get("ap_engaged", 1))
    at = bool(state.get("at_engaged", 1))

    alt = _estimate_altitude(state)
    spd = _estimate_speed(state)
    hdg = _best_heading(state)
    vs = _estimate_vertical(state, alt, dt)
    aoa = _vote_triplet(state, ["aoa_1_deg", "aoa_2_deg", "aoa_3_deg"], ["aoa_1_valid", "aoa_2_valid", "aoa_3_valid"], 0.0)
    pitch = _vote_triplet(state, ["irs_1_pitch_deg", "irs_2_pitch_deg", "irs_3_pitch_deg"], ["irs_1_valid", "irs_2_valid", "irs_3_valid"], 0.0)
    roll = _vote_triplet(state, ["irs_1_roll_deg", "irs_2_roll_deg", "irs_3_roll_deg"], ["irs_1_valid", "irs_2_valid", "irs_3_valid"], 0.0)
    yaw_rate = _vote_triplet(state, ["gyro_1_yaw_rate_dps", "gyro_2_yaw_rate_dps", "gyro_3_yaw_rate_dps"], ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"], 0.0)
    pitch_rate = _vote_triplet(state, ["gyro_1_pitch_rate_dps", "gyro_2_pitch_rate_dps", "gyro_3_pitch_rate_dps"], ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"], 0.0)
    roll_rate = _vote_triplet(state, ["gyro_1_roll_rate_dps", "gyro_2_roll_rate_dps", "gyro_3_roll_rate_dps"], ["gyro_1_valid", "gyro_2_valid", "gyro_3_valid"], 0.0)
    wind = state.get("wind_speed_kt", 0.0) or 0.0
    icing = state.get("icing_rate", 0.0) or 0.0
    turb = state.get("turbulence_intensity", 0.0) or 0.0
    gw = state.get("gross_weight_kg", 60000.0) or 60000.0
    gear_down = bool(state.get("gear_handle_down", 0))
    wow = any(bool(state.get(k, 0)) for k in ("weight_on_wheels_left", "weight_on_wheels_nose", "weight_on_wheels_right"))

    target_alt = float(state.get("target_alt_ft", alt))
    target_hdg = float(state.get("target_hdg_deg", hdg))
    target_spd = float(state.get("target_ias_kt", spd))
    target_vs = float(state.get("target_vs_fpm", 0.0))

    alt_err = target_alt - alt
    hdg_err = _wrap180(target_hdg - hdg)
    spd_err = target_spd - spd
    vs_err = target_vs - vs

    stall_margin = max(0.0, 15.0 - aoa) if aoa is not None else 0.0
    high_speed = spd > 320.0 or state.get("mach_1", 0.0) > 0.78 or state.get("mach_2", 0.0) > 0.78 or state.get("mach_3", 0.0) > 0.78

    if wow or alt < 1200.0:
        flaps = 20 if spd < 160.0 else 0
        gear = 1
    else:
        flaps = 0
        gear = 0

    if spd < 140.0 or stall_margin < 2.0 or aoa > 12.0:
        flaps = max(flaps, 20)
    if spd < 120.0 and alt < 5000.0:
        flaps = max(flaps, 30)
    if spd > 200.0 and alt > 500.0:
        flaps = 0

    speedbrake = 0.0
    if spd > 365.0 or state.get("mach_1", 0.0) > 0.82 or state.get("mach_2", 0.0) > 0.82 or state.get("mach_3", 0.0) > 0.82:
        speedbrake = 1.0
    elif spd > target_spd + 20.0 and alt_err < 0:
        speedbrake = _clamp((spd - (target_spd + 20.0)) / 80.0, 0.0, 0.8)

    # Longitudinal control
    alt_k = 0.00035 if alt > 1000.0 else 0.00025
    vs_k = 0.00012
    spd_to_pitch = -0.003 if spd > target_spd else 0.002
    desired_pitch = _clamp(alt_k * alt_err + vs_k * vs_err + spd_to_pitch * spd_err, -10.0, 15.0)
    pitch_err = desired_pitch - pitch
    p_term = 0.07 * pitch_err - 0.015 * pitch_rate
    _S["alt_i"] = _clamp(_S["alt_i"] + alt_err * dt * 0.00001, -0.4, 0.4)
    elevator = _clamp(p_term + _S["alt_i"] - 0.02 * aoa + 0.001 * vs_err, -1.0, 1.0)

    # Speed control with thrust management
    thrust_base = 0.55 + 0.0022 * spd_err + 0.00015 * alt_err / 100.0 - 0.001 * wind / 10.0
    if at or True:
        if spd < 160.0 and alt < 10000.0:
            thrust_base += 0.08
        if icing > 0.1:
            thrust_base += 0.05
        if gw > 70000.0:
            thrust_base += 0.03
    thrust_base = _clamp(thrust_base, 0.0, 1.0)

    if high_speed:
        thrust_base = min(thrust_base, 0.6)
    if speedbrake > 0.2:
        thrust_base = min(thrust_base, 0.45)
    if spd < 130.0 or stall_margin < 4.0 or aoa > 14.0:
        thrust_base = max(thrust_base, 0.85)

    asym = _clamp(0.004 * yaw_rate + 0.0025 * roll_rate + 0.0008 * state.get("drift_angle_deg", 0.0), -0.12, 0.12)
    throttle_1 = _clamp(thrust_base - asym, 0.0, 1.0)
    throttle_2 = _clamp(thrust_base + asym, 0.0, 1.0)

    # Lateral control
    turn_bank = _clamp(0.3 * hdg_err + 0.02 * state.get("track_deg", hdg) - 0.015 * state.get("drift_angle_deg", 0.0), -25.0, 25.0)
    roll_err = turn_bank - roll
    aileron = _clamp(0.05 * roll_err - 0.015 * roll_rate + 0.0007 * hdg_err, -1.0, 1.0)
    rudder = _clamp(0.03 * (-yaw_rate) + 0.015 * (-state.get("drift_angle_deg", 0.0)) + 0.25 * (throttle_2 - throttle_1), -1.0, 1.0)

    # Trim adaptation: keep elevator close to neutral over time
    _S["trim"] = _clamp(_S["trim"] + 0.002 * elevator * dt + 0.00002 * alt_err * dt, -1.0, 1.0)
    elevator_trim = _S["trim"]

    # Safety overrides
    if any(bool(state.get(k, 0)) for k in ("engine_1_fire", "engine_2_fire")):
        throttle_1 = 0.0 if state.get("engine_1_fire") else throttle_1
        throttle_2 = 0.0 if state.get("engine_2_fire") else throttle_2
        speedbrake = max(speedbrake, 0.2)
    if state.get("hyd_a_press_psi", 3000.0) < 1000.0 or state.get("hyd_b_press_psi", 3000.0) < 1000.0:
        aileron = _clamp(aileron, -0.7, 0.7)
        rudder = _clamp(rudder, -0.7, 0.7)
    if state.get("bus_ess_voltage", 28.0) < 20.0:
        _S["alt_i"] *= 0.98
        _S["hdg_i"] *= 0.98

    if abs(hdg_err) > 90.0 and spd < 180.0:
        flaps = max(flaps, 15)

    if alt < 1500.0 and spd < 170.0 and not wow:
        gear = 1
    elif alt > 2500.0:
        gear = 0

    return {
        "elevator": float(_clamp(elevator, -1.0, 1.0)),
        "aileron": float(_clamp(aileron, -1.0, 1.0)),
        "rudder": float(_clamp(rudder, -1.0, 1.0)),
        "throttle_1": float(_clamp(throttle_1, 0.0, 1.0)),
        "throttle_2": float(_clamp(throttle_2, 0.0, 1.0)),
        "elevator_trim": float(_clamp(elevator_trim, -1.0, 1.0)),
        "flaps": int(_clamp(flaps, 0, 40)),
        "speedbrake": float(_clamp(speedbrake, 0.0, 1.0)),
        "gear": int(gear),
    }
```
