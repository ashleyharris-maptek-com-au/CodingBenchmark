using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n;
        try
        {
            n = fs.NextInt();
        }
        catch
        {
            return;
        }

        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        long[][] flow = new long[n][];
        long[][] dist = new long[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new long[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextInt();
        }
        for (int i = 0; i < n; i++)
        {
            dist[i] = new long[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextInt();
        }

        int[] perm = BuildInitialPermutation(flow, dist, n);
        long cost = ComputeCost(flow, dist, perm);
        Improve(flow, dist, perm, ref cost);

        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private static int[] BuildInitialPermutation(long[][] flow, long[][] dist, int n)
    {
        int[] perm = new int[n];
        bool[] used = new bool[n];

        long[] flowScore = new long[n];
        long[] distScore = new long[n];
        for (int i = 0; i < n; i++)
        {
            long fs = 0, ds = 0;
            for (int j = 0; j < n; j++)
            {
                fs += flow[i][j] + flow[j][i];
                ds += dist[i][j] + dist[j][i];
            }
            flowScore[i] = fs;
            distScore[i] = ds;
        }

        int[] facilityOrder = new int[n];
        int[] locationOrder = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilityOrder[i] = i;
            locationOrder[i] = i;
        }
        Array.Sort(facilityOrder, (a, b) => flowScore[b].CompareTo(flowScore[a]));
        Array.Sort(locationOrder, (a, b) => distScore[a].CompareTo(distScore[b]));

        for (int k = 0; k < n; k++)
        {
            int f = facilityOrder[k];
            int bestLoc = -1;
            long best = long.MaxValue;
            int limit = n <= 80 ? n : Math.Min(n, 100);
            for (int t = 0; t < limit; t++)
            {
                int loc = locationOrder[t];
                if (used[loc]) continue;
                long score = 0;
                for (int i = 0; i < k; i++)
                {
                    int fi = facilityOrder[i];
                    int li = perm[fi];
                    score += flow[f][fi] * dist[loc][li] + flow[fi][f] * dist[li][loc];
                }
                if (score < best)
                {
                    best = score;
                    bestLoc = loc;
                }
            }

            if (bestLoc < 0)
            {
                for (int loc = 0; loc < n; loc++)
                {
                    if (!used[loc])
                    {
                        bestLoc = loc;
                        break;
                    }
                }
            }

            perm[f] = bestLoc;
            used[bestLoc] = true;
        }

        return perm;
    }

    private static long ComputeCost(long[][] flow, long[][] dist, int[] perm)
    {
        int n = perm.Length;
        long total = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            for (int j = 0; j < n; j++)
                total += flow[i][j] * dist[pi][perm[j]];
        }
        return total;
    }

    private static void Improve(long[][] flow, long[][] dist, int[] perm, ref long cost)
    {
        int n = perm.Length;
        if (n <= 1) return;

        int[] posOfLoc = new int[n];
        for (int i = 0; i < n; i++) posOfLoc[perm[i]] = i;

        long[] rowSum = new long[n];
        long[] colSum = new long[n];
        long[] distRow = new long[n];
        long[] distCol = new long[n];

        for (int i = 0; i < n; i++)
        {
            long rs = 0, cs = 0, dr = 0, dc = 0;
            int pi = perm[i];
            for (int k = 0; k < n; k++)
            {
                rs += flow[i][k];
                cs += flow[k][i];
                dr += dist[pi][k];
                dc += dist[k][pi];
            }
            rowSum[i] = rs;
            colSum[i] = cs;
            distRow[i] = dr;
            distCol[i] = dc;
        }

        bool improved = true;
        int passes = 0;
        while (improved && passes < 4)
        {
            improved = false;
            passes++;
            for (int a = 0; a < n; a++)
            {
                int pa = perm[a];
                for (int b = a + 1; b < n; b++)
                {
                    int pb = perm[b];
                    long delta = 0;
                    for (int k = 0; k < n; k++)
                    {
                        if (k != a && k != b)
                        {
                            int pk = perm[k];
                            delta += flow[a][k] * (dist[pb][pk] - dist[pa][pk]);
                            delta += flow[b][k] * (dist[pa][pk] - dist[pb][pk]);
                            delta += flow[k][a] * (dist[pk][pb] - dist[pk][pa]);
                            delta += flow[k][b] * (dist[pk][pa] - dist[pk][pb]);
                        }
                    }
                    delta += flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
                    delta += flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
                    delta += flow[a][b] * (dist[pb][pa] - dist[pa][pb]);
                    delta += flow[b][a] * (dist[pa][pb] - dist[pb][pa]);

                    if (delta < 0)
                    {
                        perm[a] = pb;
                        perm[b] = pa;
                        cost += delta;
                        improved = true;
                        pa = perm[a];
                        pb = perm[b];
                    }
                }
            }
        }
    }
}
