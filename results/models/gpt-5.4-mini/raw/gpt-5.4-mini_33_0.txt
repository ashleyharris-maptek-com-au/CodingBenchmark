using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        private byte ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = ReadByte(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    private static long CalcCost(int[] perm, int[][] flow, int[][] dist)
    {
        int n = perm.Length;
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = perm[i];
            int[] fi = flow[i];
            for (int j = 0; j < n; j++) cost += (long)fi[j] * dist[pi][perm[j]];
        }
        return cost;
    }

    private static long DeltaSwap(int a, int b, int[] perm, int[][] flow, int[][] dist)
    {
        if (a == b) return 0;
        int n = perm.Length;
        int pa = perm[a];
        int pb = perm[b];
        long delta = 0;
        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = perm[k];
            long fik = flow[a][k] + flow[k][a];
            long fjk = flow[b][k] + flow[k][b];
            delta += fik * (dist[pb][pk] - dist[pa][pk]);
            delta += fjk * (dist[pa][pk] - dist[pb][pk]);
        }
        long fab = flow[a][b] + flow[b][a];
        delta += fab * (dist[pb][pa] - dist[pa][pb]);
        delta += (long)flow[a][a] * (dist[pb][pb] - dist[pa][pa]);
        delta += (long)flow[b][b] * (dist[pa][pa] - dist[pb][pb]);
        return delta;
    }

    private static void ImproveLocal(int[] perm, int[][] flow, int[][] dist, int maxPasses)
    {
        int n = perm.Length;
        for (int pass = 0; pass < maxPasses; pass++)
        {
            bool improved = false;
            for (int i = 0; i < n; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    long d = DeltaSwap(i, j, perm, flow, dist);
                    if (d < 0)
                    {
                        int t = perm[i]; perm[i] = perm[j]; perm[j] = t;
                        improved = true;
                    }
                }
            }
            if (!improved) break;
        }
    }

    private static int[] GreedyConstruct(int[][] flow, int[][] dist)
    {
        int n = flow.Length;
        int[] orderF = new int[n];
        int[] orderL = new int[n];
        long[] scoreF = new long[n];
        long[] scoreL = new long[n];
        for (int i = 0; i < n; i++)
        {
            long sf = 0, sl = 0;
            for (int j = 0; j < n; j++)
            {
                sf += Math.Abs((long)flow[i][j]) + Math.Abs((long)flow[j][i]);
                sl += Math.Abs((long)dist[i][j]) + Math.Abs((long)dist[j][i]);
            }
            scoreF[i] = sf;
            scoreL[i] = sl;
            orderF[i] = i;
            orderL[i] = i;
        }
        Array.Sort(scoreF, orderF);
        Array.Sort(scoreL, orderL);
        Array.Reverse(orderF);
        Array.Reverse(orderL);

        int[] perm = new int[n];
        for (int i = 0; i < n; i++) perm[orderF[i]] = orderL[i];
        return perm;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int[][] flow = new int[n][];
        int[][] dist = new int[n][];
        for (int i = 0; i < n; i++)
        {
            flow[i] = new int[n];
            for (int j = 0; j < n; j++) flow[i][j] = fs.NextInt();
        }
        for (int i = 0; i < n; i++)
        {
            dist[i] = new int[n];
            for (int j = 0; j < n; j++) dist[i][j] = fs.NextInt();
        }

        int[] perm = GreedyConstruct(flow, dist);

        if (n <= 250)
        {
            ImproveLocal(perm, flow, dist, 4);
        }
        else if (n <= 800)
        {
            ImproveLocal(perm, flow, dist, 1);
        }

        long cost = CalcCost(perm, flow, dist);
        var sb = new StringBuilder();
        sb.AppendLine(cost.ToString());
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(perm[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }
}
