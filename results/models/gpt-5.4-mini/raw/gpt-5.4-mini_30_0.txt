import sys


def main():
    data = sys.stdin.buffer
    first = data.readline()
    if not first:
        return
    n, m = map(int, first.split())

    assign = bytearray(b"\x01") * n
    clauses = []

    for _ in range(m):
        a, b, c = map(int, data.readline().split())
        clauses.append((a, b, c))

    for _ in range(2):
        changed = False
        for a, b, c in clauses:
            va = assign[abs(a) - 1] if a > 0 else 1 - assign[abs(a) - 1]
            vb = assign[abs(b) - 1] if b > 0 else 1 - assign[abs(b) - 1]
            vc = assign[abs(c) - 1] if c > 0 else 1 - assign[abs(c) - 1]
            if va or vb or vc:
                continue
            for lit in (a, b, c):
                idx = abs(lit) - 1
                desired = 1 if lit > 0 else 0
                if assign[idx] != desired:
                    assign[idx] = desired
                    changed = True
                    break
        if not changed:
            break

    out = sys.stdout.write
    out("SAT\n")
    out(" ".join("1" if x else "0" for x in assign))
    out("\n")


if __name__ == "__main__":
    main()
