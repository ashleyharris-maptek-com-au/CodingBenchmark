OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0_33333334 = OpConstant %float 0.333333343
%float_0_7 = OpConstant %float 0.699999988
%uint_3 = OpConstant %uint 3

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%ptr_in_v4 = OpTypePointer Input %v4float
%worldPos_arr = OpTypeArray %v3float %uint_3
%ptr_worldPos_arr = OpTypePointer Input %worldPos_arr
OpDecorate %worldPos Location 0
%worldPos = OpVariable %ptr_worldPos_arr Input

%normal_arr = OpTypeArray %v3float %uint_3
%ptr_normal_arr = OpTypePointer Input %normal_arr
OpDecorate %normal Location 1
%normal = OpVariable %ptr_normal_arr Input

%color_arr = OpTypeArray %v3float %uint_3
%ptr_color_arr = OpTypePointer Input %color_arr
OpDecorate %color Location 2
%color = OpVariable %ptr_color_arr Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float

%ptr_out_v3 = OpTypePointer Output %v3float
OpDecorate %outColor Location 0
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %func
%label = OpLabel

%p0p = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1p = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2p = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0v4 = OpLoad %v4float %p0p
%p1v4 = OpLoad %v4float %p1p
%p2v4 = OpLoad %v4float %p2p

%p0 = OpVectorShuffle %v3float %p0v4 %p0v4 0 1 2
%p1 = OpVectorShuffle %v3float %p1v4 %p1v4 0 1 2
%p2 = OpVectorShuffle %v3float %p2v4 %p2v4 0 1 2

%sum01 = OpFAdd %v3float %p0 %p1
%sum = OpFAdd %v3float %sum01 %p2
%centroid = OpVectorTimesScalar %v3float %sum %float_0_33333334

%c0p = OpAccessChain %ptr_color_arr %color %int_0
%c1p = OpAccessChain %ptr_color_arr %color %int_1
%c2p = OpAccessChain %ptr_color_arr %color %int_2
%c0 = OpLoad %v3float %c0p
%c1 = OpLoad %v3float %c1p
%c2 = OpLoad %v3float %c2p

%d0 = OpFSub %v3float %p0 %centroid
%d1 = OpFSub %v3float %p1 %centroid
%d2 = OpFSub %v3float %p2 %centroid

%sd0 = OpVectorTimesScalar %v3float %d0 %float_0_7
%sd1 = OpVectorTimesScalar %v3float %d1 %float_0_7
%sd2 = OpVectorTimesScalar %v3float %d2 %float_0_7

%n0 = OpFAdd %v3float %centroid %sd0
%n1 = OpFAdd %v3float %centroid %sd1
%n2 = OpFAdd %v3float %centroid %sd2

%n0w = OpCompositeExtract %float %p0v4 3
%n1w = OpCompositeExtract %float %p1v4 3
%n2w = OpCompositeExtract %float %p2v4 3

%n0v4 = OpCompositeConstruct %v4float %n0 %n0w
%n1v4 = OpCompositeConstruct %v4float %n1 %n1w
%n2v4 = OpCompositeConstruct %v4float %n2 %n2w

%outp0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp0 %n0v4
OpStore %outColor %c0
OpEmitVertex

%outp1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp1 %n1v4
OpStore %outColor %c1
OpEmitVertex

%outp2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %outp2 %n2v4
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpFunctionEnd
