OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %outColor %ubo
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpSource GLSL 450
OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_out "gl_out"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "proj"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_in BuiltIn Position
OpDecorate %outColor Location 0
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_ubo = OpTypePointer Uniform %UBO
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo = OpVariable %ptr_ubo Uniform

%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%float_0 = OpConstant %float 0
%float_0_33333334 = OpConstant %float 0.333333343
%float_0_7 = OpConstant %float 0.699999988
%float_1 = OpConstant %float 1

%main = OpFunction %void None %func
%label = OpLabel
%p0ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p2ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p0 = OpLoad %v4float %p0ptr
%p1 = OpLoad %v4float %p1ptr
%p2 = OpLoad %v4float %p2ptr
%sum01 = OpFAdd %v4float %p0 %p1
%sum = OpFAdd %v4float %sum01 %p2
%C = OpVectorTimesScalar %v4float %sum %float_0_33333334

%diff0 = OpFSub %v4float %p0 %C
%scaled0 = OpVectorTimesScalar %v4float %diff0 %float_0_7
%n0 = OpFAdd %v4float %C %scaled0
%diff1 = OpFSub %v4float %p1 %C
%scaled1 = OpVectorTimesScalar %v4float %diff1 %float_0_7
%n1 = OpFAdd %v4float %C %scaled1
%diff2 = OpFSub %v4float %p2 %C
%scaled2 = OpVectorTimesScalar %v4float %diff2 %float_0_7
%n2 = OpFAdd %v4float %C %scaled2

%c0ptr = OpAccessChain %ptr_in_v3 %gl_in %int_0 %int_0
%c1ptr = OpAccessChain %ptr_in_v3 %gl_in %int_1 %int_0
%c2ptr = OpAccessChain %ptr_in_v3 %gl_in %int_2 %int_0
%c0 = OpLoad %v3float %c0ptr
%c1 = OpLoad %v3float %c1ptr
%c2 = OpLoad %v3float %c2ptr

OpStore %gl_out %n0
OpStore %outColor %c0
OpEmitVertex
OpStore %gl_out %n1
OpStore %outColor %c1
OpEmitVertex
OpStore %gl_out %n2
OpStore %outColor %c2
OpEmitVertex
OpEndPrimitive
OpFunctionEnd
