import sys
import math


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_length(route, pts):
    if not route:
        return 0.0
    total = dist(pts[0], pts[route[0]])
    for i in range(len(route) - 1):
        total += dist(pts[route[i]], pts[route[i + 1]])
    total += dist(pts[route[-1]], pts[0])
    return total


def insertion_cost(route, pos, customer, pts):
    prev_pt = pts[0] if pos == 0 else pts[route[pos - 1]]
    next_pt = pts[0] if pos == len(route) else pts[route[pos]]
    return dist(prev_pt, pts[customer]) + dist(pts[customer], next_pt) - dist(prev_pt, next_pt)


def best_insertion_pos(route, customer, pts):
    best_pos = 0
    best_cost = float("inf")
    for pos in range(len(route) + 1):
        c = insertion_cost(route, pos, customer, pts)
        if c < best_cost:
            best_cost = c
            best_pos = pos
    return best_pos, best_cost


def two_opt(route, pts):
    if len(route) < 4:
        return route
    improved = True
    while improved:
        improved = False
        for i in range(len(route) - 2):
            a = pts[0] if i == 0 else pts[route[i - 1]]
            b = pts[route[i]]
            for j in range(i + 2, len(route)):
                c = pts[route[j - 1]]
                d = pts[0] if j == len(route) else pts[route[j]]
                old = dist(a, b) + dist(c, d)
                new = dist(a, c) + dist(b, d)
                if new + 1e-12 < old:
                    route[i:j] = reversed(route[i:j])
                    improved = True
                    break
            if improved:
                break
    return route


def solve():
    data = sys.stdin.read().strip().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = float(next(it))

    depot = (float(next(it)), float(next(it)))
    pts = [depot]
    demand = [0.0]
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        pts.append((x, y))
        demand.append(d)

    customers = list(range(1, n + 1))
    customers.sort(key=lambda i: (-demand[i], dist(depot, pts[i])))

    routes = []
    loads = []

    for cust in customers:
        best_route = -1
        best_pos = 0
        best_score = float("inf")

        for r_idx, route in enumerate(routes):
            if loads[r_idx] + demand[cust] > cap + 1e-12:
                continue
            pos, cost = best_insertion_pos(route, cust, pts)
            score = cost + 0.05 * len(route)
            if score < best_score:
                best_score = score
                best_route = r_idx
                best_pos = pos

        if best_route == -1:
            routes.append([cust])
            loads.append(demand[cust])
        else:
            routes[best_route].insert(best_pos, cust)
            loads[best_route] += demand[cust]

    for idx in range(len(routes)):
        routes[idx] = two_opt(routes[idx], pts)

    route_infos = []
    for route in routes:
        route_infos.append((route_length(route, pts), route))

    route_infos.sort(key=lambda x: (len(x[1]), x[0]))
    routes = [r for _, r in route_infos]

    total = sum(route_length(route, pts) for route in routes)

    out = [f"{total:.6f}", str(len(routes))]
    for route in routes:
        out.append(" ".join(str(i - 1) for i in route))
    sys.stdout.write("\n".join(out))


if __name__ == "__main__":
    solve()
