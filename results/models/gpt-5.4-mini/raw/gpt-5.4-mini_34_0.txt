import sys
import math
import random


def read_input():
    data = sys.stdin.buffer.read().split()
    if not data:
        return None
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = float(next(it))
    depot = (float(next(it)), float(next(it)))
    customers = []
    for i in range(n):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        customers.append((x, y, d, i))
    return n, k, capacity, depot, customers


def dist(a, b):
    return math.hypot(a[0] - b[0], a[1] - b[1])


def route_distance(route, coords, depot):
    if not route:
        return 0.0
    total = dist(depot, coords[route[0]])
    for i in range(len(route) - 1):
        total += dist(coords[route[i]], coords[route[i + 1]])
    total += dist(coords[route[-1]], depot)
    return total


def savings_routes(customers, capacity, depot):
    coords = [(x, y) for x, y, _, _ in customers]
    demands = [d for _, _, d, _ in customers]
    n = len(customers)
    if n == 0:
        return []

    if sum(demands) <= capacity:
        return [list(range(n))]

    order = sorted(range(n), key=lambda i: math.atan2(coords[i][1] - depot[1], coords[i][0] - depot[0]))

    routes = [[i] for i in order]
    route_load = {i: demands[i] for i in order}
    route_of = {i: i for i in order}

    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            s = dist(depot, coords[i]) + dist(depot, coords[j]) - dist(coords[i], coords[j])
            savings.append((s, i, j))
    savings.sort(reverse=True)

    def endpoint(route, customer, side):
        return route[0] if side == "head" else route[-1]

    for _, i, j in savings:
        ri = route_of.get(i)
        rj = route_of.get(j)
        if ri is None or rj is None or ri == rj:
            continue
        r1 = routes[ri]
        r2 = routes[rj]
        if not r1 or not r2:
            continue
        if route_load[ri] + route_load[rj] > capacity:
            continue
        merged = None
        if r1[-1] == i and r2[0] == j:
            merged = r1 + r2
        elif r1[0] == i and r2[-1] == j:
            merged = r2 + r1
        elif r1[0] == i and r2[0] == j:
            merged = list(reversed(r1)) + r2
        elif r1[-1] == i and r2[-1] == j:
            merged = r1 + list(reversed(r2))
        if merged is None:
            continue
        routes[ri] = merged
        route_load[ri] += route_load[rj]
        routes[rj] = []
        route_load[rj] = 0.0
        for c in merged:
            route_of[c] = ri

    out = [r for r in routes if r]
    if len(out) <= k:
        return out

    out.sort(key=lambda r: route_distance(r, coords, depot), reverse=True)
    while len(out) > k:
        a = out.pop(0)
        b = out.pop(0)
        merged = best_merge(a, b, coords, depot, demands, capacity)
        out.append(merged)
        out.sort(key=lambda r: route_distance(r, coords, depot), reverse=True)
    return out


def best_merge(a, b, coords, depot, demands, capacity):
    candidates = [
        a + b,
        a + list(reversed(b)),
        list(reversed(a)) + b,
        list(reversed(a)) + list(reversed(b)),
        b + a,
        b + list(reversed(a)),
        list(reversed(b)) + a,
        list(reversed(b)) + list(reversed(a)),
    ]
    best = None
    bestd = float("inf")
    load = sum(demands[i] for i in a) + sum(demands[i] for i in b)
    if load > capacity:
        return a if route_distance(a, coords, depot) <= route_distance(b, coords, depot) else b
    for r in candidates:
        d = route_distance(r, coords, depot)
        if d < bestd:
            bestd = d
            best = r
    return best


def two_opt(route, coords, depot):
    if len(route) < 4:
        return route
    improved = True
    best = route[:]
    best_dist = route_distance(best, coords, depot)
    while improved:
        improved = False
        n = len(best)
        for i in range(n - 2):
            for j in range(i + 2, n):
                if i == 0 and j == n - 1:
                    continue
                cand = best[:i + 1] + best[i + 1:j + 1][::-1] + best[j + 1:]
                d = route_distance(cand, coords, depot)
                if d + 1e-9 < best_dist:
                    best = cand
                    best_dist = d
                    improved = True
                    break
            if improved:
                break
    return best


def relocate_between(routes, coords, depot, demands, capacity):
    loads = [sum(demands[i] for i in r) for r in routes]
    changed = True
    while changed:
        changed = False
        route_order = sorted(range(len(routes)), key=lambda idx: route_distance(routes[idx], coords, depot), reverse=True)
        for a_idx in route_order:
            a = routes[a_idx]
            if not a:
                continue
            a_dist = route_distance(a, coords, depot)
            for pos, cust in enumerate(a):
                d = demands[cust]
                best_gain = 0.0
                best_move = None
                for b_idx, b in enumerate(routes):
                    if a_idx == b_idx:
                        continue
                    if loads[b_idx] + d > capacity:
                        continue
                    for ins in range(len(b) + 1):
                        cand_b = b[:ins] + [cust] + b[ins:]
                        cand_a = a[:pos] + a[pos + 1:]
                        if not cand_a:
                            continue
                        gain = (a_dist + route_distance(b, coords, depot)) - (route_distance(cand_a, coords, depot) + route_distance(cand_b, coords, depot))
                        if gain > best_gain + 1e-9:
                            best_gain = gain
                            best_move = (b_idx, ins, cand_a, cand_b)
                if best_move:
                    b_idx, ins, cand_a, cand_b = best_move
                    routes[a_idx] = cand_a
                    routes[b_idx] = cand_b
                    loads[a_idx] -= d
                    loads[b_idx] += d
                    changed = True
                    break
            if changed:
                break
    return [r for r in routes if r]


def solve():
    parsed = read_input()
    if parsed is None:
        return
    n, k, capacity, depot, customers = parsed
    coords = [(x, y) for x, y, _, _ in customers]
    demands = [d for _, _, d, _ in customers]

    routes = savings_routes(customers, capacity, depot)
    routes = [two_opt(r, coords, depot) for r in routes]
    routes = relocate_between(routes, coords, depot, demands, capacity)
    routes = [two_opt(r, coords, depot) for r in routes]

    all_customers = [c for r in routes for c in r]
    if len(all_customers) != n or len(set(all_customers)) != n:
        seen = set(all_customers)
        missing = [i for i in range(n) if i not in seen]
        routes = [r for r in routes if r]
        while missing and routes:
            routes.sort(key=len)
            r = routes[0]
            room = capacity - sum(demands[i] for i in r)
            moved = []
            for c in missing[:]:
                if demands[c] <= room:
                    r.append(c)
                    room -= demands[c]
                    moved.append(c)
            for c in moved:
                missing.remove(c)
            routes[0] = r
            if not moved:
                break
        for c in missing:
            routes.append([c])

    routes = [r for r in routes if r]
    while len(routes) > k:
        routes.sort(key=lambda r: route_distance(r, coords, depot), reverse=True)
        a = routes.pop(0)
        b = routes.pop(0)
        routes.append(best_merge(a, b, coords, depot, demands, capacity))
        routes = [two_opt(r, coords, depot) for r in routes]

    routes = [r for r in routes if r]
    total = sum(route_distance(r, coords, depot) for r in routes)
    sys.stdout.write(f"{total:.10f}\n")
    sys.stdout.write(f"{len(routes)}\n")
    for r in routes:
        sys.stdout.write(" ".join(map(str, r)) + "\n")


if __name__ == "__main__":
    random.seed(0)
    solve()
