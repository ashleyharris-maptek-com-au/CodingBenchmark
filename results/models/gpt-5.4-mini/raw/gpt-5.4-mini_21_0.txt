#include <algorithm>
#include <array>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        size_t h = 1469598103934665603ULL;
        for (int x : v) {
            size_t z = static_cast<size_t>(x) + 0x9e3779b97f4a7c15ULL;
            h ^= z + (h << 6) + (h >> 2);
        }
        return h;
    }
};

static bool inside(const vector<int>& p, const vector<int>& bounds) {
    for (size_t i = 0; i < p.size(); ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

static vector<vector<int>> neighbors(const vector<int>& p) {
    vector<vector<int>> out;
    out.reserve(p.size() * 2);
    for (size_t a = 0; a < p.size(); ++a) {
        for (int d : {-1, 1}) {
            vector<int> q = p;
            q[a] += d;
            out.push_back(move(q));
        }
    }
    return out;
}

struct State {
    vector<int> head;
    vector<vector<int>> body;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int D, max_turns;
    if (!(cin >> D >> max_turns)) return 0;

    vector<int> bounds(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];

    int snake_length;
    vector<int> head(D);
    cin >> snake_length;
    for (int i = 0; i < D; ++i) cin >> head[i];

    int F;
    cin >> F;
    vector<vector<int>> foods(F, vector<int>(D));
    for (int i = 0; i < F; ++i) {
        for (int j = 0; j < D; ++j) cin >> foods[i][j];
    }

    int O;
    cin >> O;
    unordered_set<vector<int>, VecHash> blocked;
    blocked.reserve(static_cast<size_t>(O) * 2 + 16);
    for (int i = 0; i < O; ++i) {
        vector<int> p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        blocked.insert(move(p));
    }

    vector<vector<int>> body;
    body.push_back(head);
    for (int i = 1; i < snake_length; ++i) {
        vector<int> seg = head;
        if (D > 0) seg[0] -= i;
        body.push_back(seg);
    }

    auto food_set = unordered_set<vector<int>, VecHash>();
    food_set.reserve(foods.size() * 2 + 16);
    for (auto& f : foods) food_set.insert(f);

    auto occupied = [&](const vector<int>& p) {
        for (const auto& s : body) if (s == p) return true;
        return false;
    };

    auto passable = [&](const vector<int>& p, const vector<vector<int>>& cur_body, bool tail_free) {
        if (!inside(p, bounds)) return false;
        if (blocked.find(p) != blocked.end()) return false;
        for (size_t i = 0; i < cur_body.size(); ++i) {
            if (cur_body[i] == p) {
                if (tail_free && i + 1 == cur_body.size()) return true;
                return false;
            }
        }
        return true;
    };

    auto bfs_to_goal = [&](const vector<vector<int>>& cur_body, const unordered_set<vector<int>, VecHash>& goals) {
        struct Node { vector<int> p; int prev; int axis; int dir; };
        unordered_map<vector<int>, int, VecHash> seen;
        queue<int> q;
        vector<Node> nodes;
        nodes.push_back({cur_body.front(), -1, -1, 0});
        seen[nodes[0].p] = 0;
        q.push(0);
        bool tail_free = true;
        while (!q.empty()) {
            int idx = q.front(); q.pop();
            const auto& cur = nodes[idx].p;
            if (goals.find(cur) != goals.end()) {
                int t = idx;
                while (nodes[t].prev != 0 && nodes[t].prev != -1) t = nodes[t].prev;
                if (nodes[t].prev == 0 || nodes[t].prev == -1) return make_pair(nodes[idx].p, nodes[idx].prev);
            }
            for (int a = 0; a < D; ++a) {
                for (int d : {-1, 1}) {
                    vector<int> nxt = cur;
                    nxt[a] += d;
                    if (!passable(nxt, cur_body, tail_free)) continue;
                    if (seen.find(nxt) != seen.end()) continue;
                    int ni = (int)nodes.size();
                    seen[nxt] = ni;
                    nodes.push_back({move(nxt), idx, a, d});
                    q.push(ni);
                }
            }
        }
        return make_pair(vector<int>(), -2);
    };

    auto choose_move = [&](const vector<vector<int>>& cur_body, const vector<vector<int>>& rem_foods) {
        vector<int> h = cur_body.front();
        int bestDist = numeric_limits<int>::max();
        vector<int> bestMove(D + 1, 0);
        auto legal = [&](const vector<int>& p) {
            if (!inside(p, bounds)) return false;
            if (blocked.find(p) != blocked.end()) return false;
            for (size_t i = 0; i + 1 < cur_body.size(); ++i) if (cur_body[i] == p) return false;
            return true;
        };
        for (int a = 0; a < D; ++a) {
            for (int d : {-1, 1}) {
                vector<int> nxt = h;
                nxt[a] += d;
                if (!legal(nxt)) continue;
                int md = numeric_limits<int>::max();
                for (auto& f : rem_foods) {
                    int dist = 0;
                    for (int i = 0; i < D; ++i) dist += abs(f[i] - nxt[i]);
                    md = min(md, dist);
                }
                if (rem_foods.empty()) md = 0;
                if (md < bestDist) {
                    bestDist = md;
                    bestMove = {a, d};
                }
            }
        }
        if (bestDist != numeric_limits<int>::max()) return bestMove;
        for (int a = 0; a < D; ++a) {
            for (int d : {-1, 1}) {
                vector<int> nxt = h;
                nxt[a] += d;
                if (legal(nxt)) return vector<int>{a, d};
            }
        }
        return vector<int>{0, 1};
    };

    for (int turn = 0; turn < max_turns; ++turn) {
        if (food_set.empty()) break;
        vector<vector<int>> rem_foods(food_set.begin(), food_set.end());
        sort(rem_foods.begin(), rem_foods.end());

        int best_idx = -1;
        int best_score = numeric_limits<int>::max();
        vector<int> chosen_move;

        for (int fi = 0; fi < (int)rem_foods.size(); ++fi) {
            unordered_set<vector<int>, VecHash> goals;
            goals.insert(rem_foods[fi]);
            auto [pos, prev] = bfs_to_goal(body, goals);
            if (prev == -2) continue;
            vector<int> mv = choose_move(body, rem_foods);
            int score = 0;
            for (int i = 0; i < D; ++i) score += abs(rem_foods[fi][i] - body.front()[i]);
            if (score < best_score) {
                best_score = score;
                best_idx = fi;
                chosen_move = mv;
            }
        }

        if (chosen_move.empty()) chosen_move = choose_move(body, rem_foods);
        cout << chosen_move[0] << ' ' << chosen_move[1] << '\n';
        cout.flush();

        vector<int> nxt = body.front();
        nxt[chosen_move[0]] += chosen_move[1];
        bool ate = food_set.find(nxt) != food_set.end();
        body.insert(body.begin(), nxt);
        if (ate) {
            food_set.erase(nxt);
        } else {
            body.pop_back();
        }
    }

    return 0;
}
