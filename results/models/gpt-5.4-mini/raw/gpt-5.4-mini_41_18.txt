; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %outColor %worldPos %normal %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %outColor "outColor"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %outColor Location 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%functy = OpTypeFunction %void
%float = OpTypeFloat 32
%bool = OpTypeBool
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
%UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float
%ptr_function_float = OpTypePointer Function %float
%ptr_function_v3float = OpTypePointer Function %v3float
%ptr_function_v4float = OpTypePointer Function %v4float
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0

%worldPos = OpVariable %ptr_input_v3float Input
%normal = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output
%ubo = OpVariable %ptr_uniform_UBO Uniform

%main = OpFunction %void None %functy
%entry = OpLabel
%norm = OpVariable %ptr_function_v3float Function
%viewDir = OpVariable %ptr_function_v3float Function
%ndotv = OpVariable %ptr_function_float Function
%r = OpVariable %ptr_function_float Function
%g = OpVariable %ptr_function_float Function
%b = OpVariable %ptr_function_float Function
%color = OpVariable %ptr_function_v4float Function

%normal_val = OpLoad %v3float %normal
%normal_norm = OpExtInst %v3float %1 Normalize %normal_val
OpStore %norm %normal_norm

%camera_ptr = OpAccessChain %ptr_uniform_v4float %ubo %int_0 %int_4
%camera4 = OpLoad %v4float %camera_ptr
%camera3 = OpVectorShuffle %v3float %camera4 %camera4 0 1 2
%world = OpLoad %v3float %worldPos
%diff = OpFSub %v3float %camera3 %world
%diff_norm = OpExtInst %v3float %1 Normalize %diff
OpStore %viewDir %diff_norm

%n = OpLoad %v3float %norm
%v = OpLoad %v3float %viewDir
%dotnv = OpDot %float %n %v
%clamped = OpExtInst %float %1 FMax %dotnv %float_0
OpStore %ndotv %clamped

%nd = OpLoad %float %ndotv
%lt = OpFOrdLessThan %bool %nd %float_0_5
OpSelectionMerge %merge None
OpBranchConditional %lt %then %else

%then = OpLabel
OpStore %r %float_1
%gval = OpFMul %float %nd %float_2
OpStore %g %gval
OpStore %b %float_0
OpBranch %merge

%else = OpLabel
%shift = OpFSub %float %nd %float_0_5
%blue = OpFMul %float %shift %float_2
%one_minus = OpFSub %float_1 %blue
OpStore %r %one_minus
OpStore %g %one_minus
OpStore %b %blue
OpBranch %merge

%merge = OpLabel
%rv = OpLoad %float %r
%gv = OpLoad %float %g
%bv = OpLoad %float %b
%final = OpCompositeConstruct %v4float %rv %gv %bv %float_1
OpStore %color %final
%out = OpLoad %v4float %color
OpStore %outColor %out
OpReturn
OpFunctionEnd
