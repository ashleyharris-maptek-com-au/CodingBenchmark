#include <cmath>
#include <iostream>
#include <iomanip>
#include <vector>
#include <limits>
#include <algorithm>
using namespace std;

struct P {
    double x, y, z;
};

static inline P operator+(const P& a, const P& b){ return {a.x+b.x, a.y+b.y, a.z+b.z}; }
static inline P operator-(const P& a, const P& b){ return {a.x-b.x, a.y-b.y, a.z-b.z}; }
static inline P operator*(const P& a, double s){ return {a.x*s, a.y*s, a.z*s}; }
static inline double dot(const P& a, const P& b){ return a.x*b.x + a.y*b.y + a.z*b.z; }
static inline P cross(const P& a, const P& b){ return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x}; }
static inline double norm2(const P& a){ return dot(a,a); }
static inline double norm(const P& a){ return sqrt(norm2(a)); }

struct Tetra {
    P c;
    double rx, ry, rz;
    double rbound;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double e;
    if (!(cin >> V >> e)) return 0;
    vector<P> verts(V);
    for (int i = 0; i < V; ++i) cin >> verts[i].x >> verts[i].y >> verts[i].z;

    P mn{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    P mx{-numeric_limits<double>::infinity(), -numeric_limits<double>::infinity(), -numeric_limits<double>::infinity()};
    for (auto &p : verts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
    }

    double dx = mx.x - mn.x, dy = mx.y - mn.y, dz = mx.z - mn.z;
    if (dx <= 0 || dy <= 0 || dz <= 0 || e <= 0) {
        cout << 0 << '\n';
        return 0;
    }

    const double h = e * sqrt(2.0 / 3.0);
    const double rbound = e * sqrt(3.0 / 8.0);
    const double rx = e * 0.45, ry = e * 0.45, rz = e * 0.45;
    const double step = max(2.0 * rbound, 0.9 * e);

    vector<Tetra> out;
    out.reserve((size_t)max(1.0, (dx*dy*dz) / (step*step*step)));

    auto insideBox = [&](const P& c) {
        return c.x - rbound >= mn.x && c.x + rbound <= mx.x &&
               c.y - rbound >= mn.y && c.y + rbound <= mx.y &&
               c.z - rbound >= mn.z && c.z + rbound <= mx.z;
    };

    for (double z = mn.z + rbound; z <= mx.z - rbound; z += step) {
        for (double y = mn.y + rbound; y <= mx.y - rbound; y += step) {
            for (double x = mn.x + rbound; x <= mx.x - rbound; x += step) {
                P c{x, y, z};
                if (!insideBox(c)) continue;
                bool ok = true;
                for (const auto& t : out) {
                    double d2 = norm2(c - t.c);
                    double lim = rbound + t.rbound;
                    if (d2 < lim * lim) { ok = false; break; }
                }
                if (ok) out.push_back({c, 0.0, 0.0, 0.0, rbound});
            }
        }
    }

    cout << out.size() << '\n';
    cout.setf(ios::fixed); cout << setprecision(10);
    for (const auto& t : out) {
        cout << t.c.x << ' ' << t.c.y << ' ' << t.c.z << ' ' << t.rx << ' ' << t.ry << ' ' << t.rz << '\n';
    }
    return 0;
}
