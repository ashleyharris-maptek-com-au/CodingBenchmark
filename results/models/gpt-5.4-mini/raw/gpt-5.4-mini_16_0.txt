#include <bits/stdc++.h>
using namespace std;

struct Task {
    int machine;
    long long duration;
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<vector<Task>> jobs(J);
    long long totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        totalTasks += T;
        jobs[j].resize(T);
        for (int i = 0; i < T; ++i) {
            int machine;
            long long duration;
            cin >> machine >> duration;
            jobs[j][i] = {machine, duration};
        }
    }

    vector<vector<long long>> startTimes(J);
    for (int j = 0; j < J; ++j) startTimes[j].assign(jobs[j].size(), 0);

    vector<long long> machineFree(M, 0);
    vector<long long> jobReady(J, 0);
    vector<size_t> nextTask(J, 0);

    struct Item {
        long long ready;
        int job;
        int machine;
        long long duration;
    };

    auto cmp = [](const Item& a, const Item& b) {
        if (a.ready != b.ready) return a.ready > b.ready;
        if (a.duration != b.duration) return a.duration > b.duration;
        if (a.machine != b.machine) return a.machine > b.machine;
        return a.job > b.job;
    };

    priority_queue<Item, vector<Item>, decltype(cmp)> pq(cmp);

    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            auto &t = jobs[j][0];
            pq.push({0, j, t.machine, t.duration});
        }
    }

    // Adaptive strategy: for small inputs, the same greedy policy is fine; for large inputs,
    // this remains O(totalTasks log J) and avoids any time-stepped simulation.
    while (!pq.empty()) {
        Item cur = pq.top();
        pq.pop();

        int j = cur.job;
        size_t idx = nextTask[j];
        if (idx >= jobs[j].size()) continue;
        const Task &t = jobs[j][idx];
        if (t.machine != cur.machine || t.duration != cur.duration) {
            continue;
        }

        long long s = max(jobReady[j], machineFree[t.machine]);
        startTimes[j][idx] = s;
        long long finish = s + t.duration;
        jobReady[j] = finish;
        machineFree[t.machine] = finish;
        ++nextTask[j];

        if (nextTask[j] < jobs[j].size()) {
            const Task &nt = jobs[j][nextTask[j]];
            pq.push({jobReady[j], j, nt.machine, nt.duration});
        }
    }

    // Output one line per job: start_time machine ...
    for (int j = 0; j < J; ++j) {
        for (size_t i = 0; i < jobs[j].size(); ++i) {
            if (i) cout << ' ';
            cout << startTimes[j][i] << ' ' << jobs[j][i].machine;
        }
        if (j + 1 < J) cout << '\n';
    }

    return 0;
}
