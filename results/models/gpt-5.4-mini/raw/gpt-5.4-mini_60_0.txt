const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const machineCount = Math.max(1, data.machineCount | 0);
const tasksInput = Array.isArray(data.tasks) ? data.tasks : [];
const n = tasksInput.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ schedule: [] }));
  process.exit(0);
}

const idToIndex = new Map();
for (let i = 0; i < n; i++) idToIndex.set(tasksInput[i].id, i);

const duration = new Array(n);
const deadline = new Array(n);
const weight = new Array(n);
const deps = new Array(n);
const dependents = Array.from({ length: n }, () => []);
const indeg = new Array(n).fill(0);

for (let i = 0; i < n; i++) {
  const t = tasksInput[i];
  duration[i] = Math.max(0, t.duration | 0);
  deadline[i] = Number.isFinite(t.deadline) ? t.deadline : 0;
  weight[i] = Number.isFinite(t.weight) ? t.weight : 0;
  const arr = Array.isArray(t.deps) ? t.deps : [];
  const depIdx = new Array(arr.length);
  for (let j = 0; j < arr.length; j++) {
    const d = idToIndex.get(arr[j]);
    if (d !== undefined) {
      depIdx[j] = d;
      dependents[d].push(i);
      indeg[i]++;
    }
  }
  deps[i] = depIdx;
}

function makeHeap(compare) {
  const a = [];
  return {
    size: () => a.length,
    peek: () => a[0],
    push(v) {
      let i = a.length;
      a.push(v);
      while (i > 0) {
        const p = (i - 1) >> 1;
        if (compare(a[p], v) <= 0) break;
        a[i] = a[p];
        i = p;
      }
      a[i] = v;
    },
    pop() {
      if (a.length === 0) return undefined;
      const top = a[0];
      const v = a.pop();
      if (a.length > 0) {
        let i = 0;
        while (true) {
          let l = i * 2 + 1;
          if (l >= a.length) break;
          let r = l + 1;
          let c = l;
          if (r < a.length && compare(a[r], a[l]) < 0) c = r;
          if (compare(a[c], v) >= 0) break;
          a[i] = a[c];
          i = c;
        }
        a[i] = v;
      }
      return top;
    },
    _raw: a,
  };
}

// Priority heuristic:
// 1) critical-path length remaining
// 2) higher weight
// 3) earlier deadline
// 4) shorter duration
const critical = new Array(n).fill(0);
const topo = [];
const q = [];
for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
for (let qi = 0; qi < q.length; qi++) {
  const u = q[qi];
  topo.push(u);
  for (const v of dependents[u]) {
    if (--indeg[v] === 0) q.push(v);
  }
}

// If there are cycles, fall back to a deterministic order over all tasks.
if (topo.length !== n) {
  const order = tasksInput.map(t => t.id);
  process.stdout.write(JSON.stringify({ order }));
  process.exit(0);
}

for (let i = n - 1; i >= 0; i--) {
  const u = topo[i];
  let best = duration[u];
  for (const v of dependents[u]) {
    const cand = duration[u] + critical[v];
    if (cand > best) best = cand;
  }
  critical[u] = best;
}

const indeg2 = new Array(n);
for (let i = 0; i < n; i++) indeg2[i] = deps[i].length;

const readyCompare = (a, b) => {
  if (critical[b] !== critical[a]) return critical[b] - critical[a];
  if (weight[b] !== weight[a]) return weight[b] - weight[a];
  if (deadline[a] !== deadline[b]) return deadline[a] - deadline[b];
  if (duration[a] !== duration[b]) return duration[a] - duration[b];
  return a - b;
};

const runningCompare = (a, b) => a.end - b.end || a.machine - b.machine || a.task - b.task;

const ready = makeHeap(readyCompare);
const running = makeHeap(runningCompare);

for (let i = 0; i < n; i++) {
  if (indeg2[i] === 0) ready.push(i);
}

const machineFreeAt = new Array(machineCount).fill(0);
const schedule = new Array(n);
let scheduled = 0;
let now = 0;

function unlockFinished(upToTime) {
  while (running.size() > 0 && running.peek().end <= upToTime) {
    const item = running.pop();
    const u = item.task;
    for (const v of dependents[u]) {
      if (--indeg2[v] === 0) ready.push(v);
    }
  }
}

while (scheduled < n) {
  unlockFinished(now);

  let chosenMachine = -1;
  let earliest = Infinity;
  for (let m = 0; m < machineCount; m++) {
    if (machineFreeAt[m] <= now && machineFreeAt[m] < earliest) {
      earliest = machineFreeAt[m];
      chosenMachine = m;
    }
  }

  if (chosenMachine !== -1 && ready.size() > 0) {
    const task = ready.pop();
    const depFinish = deps[task].reduce((mx, d) => Math.max(mx, schedule[d].start + duration[d]), 0);
    const start = Math.max(now, machineFreeAt[chosenMachine], depFinish);
    const end = start + duration[task];
    schedule[task] = { id: tasksInput[task].id, start, machine: chosenMachine };
    machineFreeAt[chosenMachine] = end;
    running.push({ end, machine: chosenMachine, task });
    scheduled++;
    continue;
  }

  if (ready.size() > 0) {
    let bestMachine = 0;
    let bestStart = Infinity;
    const task = ready.peek();
    for (let m = 0; m < machineCount; m++) {
      const depFinish = deps[task].reduce((mx, d) => Math.max(mx, schedule[d].start + duration[d]), 0);
      const start = Math.max(now, machineFreeAt[m], depFinish);
      if (start < bestStart || (start === bestStart && m < bestMachine)) {
        bestStart = start;
        bestMachine = m;
      }
    }
    const taskId = ready.pop();
    const depFinish = deps[taskId].reduce((mx, d) => Math.max(mx, schedule[d].start + duration[d]), 0);
    const start = Math.max(now, machineFreeAt[bestMachine], depFinish);
    const end = start + duration[taskId];
    schedule[taskId] = { id: tasksInput[taskId].id, start, machine: bestMachine };
    machineFreeAt[bestMachine] = end;
    running.push({ end, machine: bestMachine, task: taskId });
    scheduled++;
    continue;
  }

  const nextRun = running.size() > 0 ? running.peek().end : Infinity;
  const nextFree = Math.min(...machineFreeAt);
  now = Math.min(nextRun, nextFree);
  if (!Number.isFinite(now)) break;
}

process.stdout.write(JSON.stringify({ schedule }));
