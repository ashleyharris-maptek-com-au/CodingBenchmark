const fs = require('fs');

function main() {
  const input = fs.readFileSync(0, 'utf8');
  const data = JSON.parse(input);
  const machineCount = Math.max(1, data.machineCount | 0);
  const tasks = Array.isArray(data.tasks) ? data.tasks : [];
  const n = tasks.length;

  if (n === 0) {
    process.stdout.write(JSON.stringify({ order: [] }));
    return;
  }

  const idToIndex = new Map();
  for (let i = 0; i < n; i++) idToIndex.set(tasks[i].id, i);

  const duration = new Array(n);
  const deadline = new Array(n);
  const weight = new Array(n);
  const deps = new Array(n);
  const indeg = new Array(n).fill(0);
  const succ = Array.from({ length: n }, () => []);

  for (let i = 0; i < n; i++) {
    const t = tasks[i];
    duration[i] = Math.max(0, t.duration | 0);
    deadline[i] = Number.isFinite(t.deadline) ? t.deadline : Number.POSITIVE_INFINITY;
    weight[i] = Number.isFinite(t.weight) ? t.weight : 1;
    const ds = Array.isArray(t.deps) ? t.deps : [];
    deps[i] = ds;
    for (let k = 0; k < ds.length; k++) {
      const j = idToIndex.get(ds[k]);
      if (j !== undefined) {
        indeg[i]++;
        succ[j].push(i);
      }
    }
  }

  const remainingDesc = new Array(n).fill(0);
  const topo = [];
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  for (let qi = 0; qi < q.length; qi++) {
    const u = q[qi];
    topo.push(u);
    for (const v of succ[u]) {
      if (--indeg[v] === 0) q.push(v);
    }
  }

  if (topo.length !== n) {
    const order = tasks.map(t => t.id);
    process.stdout.write(JSON.stringify({ order }));
    return;
  }

  for (let i = n - 1; i >= 0; i--) {
    const u = topo[i];
    let sum = 0;
    for (const v of succ[u]) sum += 1 + remainingDesc[v];
    remainingDesc[u] = sum;
  }

  const predsLeft = new Array(n);
  const readyAt = new Array(n).fill(0);
  for (let i = 0; i < n; i++) predsLeft[i] = deps[i].length;

  const ready = [];
  for (let i = 0; i < n; i++) if (predsLeft[i] === 0) ready.push(i);

  function taskScore(i, t0) {
    const slack = deadline[i] === Number.POSITIVE_INFINITY ? 1e18 : deadline[i] - (t0 + duration[i]);
    return (
      weight[i] * 1e9 +
      Math.max(0, -slack) * 1e6 +
      Math.max(0, remainingDesc[i]) * 1e3 -
      duration[i] * 10 -
      deadline[i] * 0.000001
    );
  }

  function betterTask(a, b, t0) {
    if (b === -1) return true;
    const sa = taskScore(a, t0);
    const sb = taskScore(b, t0);
    if (sa !== sb) return sa > sb;
    if (weight[a] !== weight[b]) return weight[a] > weight[b];
    if (deadline[a] !== deadline[b]) return deadline[a] < deadline[b];
    if (duration[a] !== duration[b]) return duration[a] < duration[b];
    return tasks[a].id < tasks[b].id;
  }

  const machineFree = new Array(machineCount).fill(0);
  const machineTask = new Array(machineCount).fill(-1);
  const machineFinish = new Array(machineCount).fill(0);
  const starts = new Array(n);
  const assigned = new Array(n).fill(false);
  const finished = new Array(n).fill(false);
  const schedule = [];

  let done = 0;
  let time = 0;

  while (done < n) {
    let progressed = false;
    for (let m = 0; m < machineCount; m++) {
      if (machineTask[m] !== -1 && machineFinish[m] <= time) {
        const u = machineTask[m];
        machineTask[m] = -1;
        finished[u] = true;
        done++;
        for (const v of succ[u]) {
          if (--predsLeft[v] === 0) {
            readyAt[v] = time;
            ready.push(v);
          }
        }
        progressed = true;
      }
    }

    let nextTime = Number.POSITIVE_INFINITY;
    for (let m = 0; m < machineCount; m++) {
      if (machineTask[m] !== -1 && machineFinish[m] < nextTime) nextTime = machineFinish[m];
      if (machineTask[m] === -1 && machineFree[m] > time && machineFree[m] < nextTime) nextTime = machineFree[m];
    }

    let availableMachines = [];
    for (let m = 0; m < machineCount; m++) {
      if (machineTask[m] === -1 && machineFree[m] <= time) availableMachines.push(m);
    }

    if (ready.length > 0 && availableMachines.length > 0) {
      ready.sort((a, b) => {
        const sa = taskScore(a, Math.max(time, readyAt[a]));
        const sb = taskScore(b, Math.max(time, readyAt[b]));
        if (sa !== sb) return sb - sa;
        if (weight[a] !== weight[b]) return weight[b] - weight[a];
        if (deadline[a] !== deadline[b]) return deadline[a] - deadline[b];
        if (duration[a] !== duration[b]) return duration[a] - duration[b];
        return String(tasks[a].id).localeCompare(String(tasks[b].id));
      });
      availableMachines.sort((a, b) => a - b);
      for (let mi = 0; mi < availableMachines.length && ready.length > 0; mi++) {
        const m = availableMachines[mi];
        let bestIdx = 0;
        let bestTask = ready[0];
        for (let i = 1; i < ready.length; i++) {
          const cand = ready[i];
          if (betterTask(cand, bestTask, time)) {
            bestTask = cand;
            bestIdx = i;
          }
        }
        ready.splice(bestIdx, 1);
        assigned[bestTask] = true;
        starts[bestTask] = time;
        machineTask[m] = bestTask;
        machineFinish[m] = time + duration[bestTask];
        schedule.push({ id: tasks[bestTask].id, start: time, machine: m });
        progressed = true;
      }
    }

    if (done >= n) break;
    if (!progressed) {
      if (nextTime === Number.POSITIVE_INFINITY) {
        break;
      }
      time = Math.max(time + 1, nextTime);
      continue;
    }

    let hasWork = false;
    for (let m = 0; m < machineCount; m++) {
      if (machineTask[m] !== -1) { hasWork = true; break; }
    }
    if (!hasWork && ready.length === 0) {
      let minReady = Number.POSITIVE_INFINITY;
      for (let i = 0; i < n; i++) {
        if (!assigned[i] && predsLeft[i] === 0 && readyAt[i] < minReady) minReady = readyAt[i];
      }
      if (minReady < Number.POSITIVE_INFINITY) time = Math.max(time + 1, minReady);
      else if (nextTime < Number.POSITIVE_INFINITY) time = Math.max(time + 1, nextTime);
      else break;
    }
  }

  process.stdout.write(JSON.stringify({ schedule }));
}

main();
