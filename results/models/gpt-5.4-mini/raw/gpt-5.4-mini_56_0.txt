const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];
const truckCount = Math.max(1, data.truckCount | 0);

function dist2(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return dx * dx + dy * dy;
}

function dist(a, b) {
  return Math.hypot(a[0] - b[0], a[1] - b[1]);
}

if (packages.length === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const cx = depot[0], cy = depot[1];
for (const p of packages) {
  p._ang = Math.atan2(p.y - cy, p.x - cx);
  p._r2 = dist2(p.x, p.y, cx, cy);
}

packages.sort((a, b) => a._ang - b._ang || a._r2 - b._r2 || (a.id < b.id ? -1 : a.id > b.id ? 1 : 0));

const n = packages.length;
const routeCount = Math.min(truckCount, n);

const routes = Array.from({ length: routeCount }, () => []);
const routePoints = Array.from({ length: routeCount }, () => []);
const routeLen = new Array(routeCount).fill(0);

const base = Math.floor(n / routeCount);
let extra = n % routeCount;
let idx = 0;
for (let r = 0; r < routeCount; r++) {
  const take = base + (extra > 0 ? 1 : 0);
  if (extra > 0) extra--;
  for (let k = 0; k < take && idx < n; k++, idx++) {
    const p = packages[idx];
    routes[r].push(p.id);
    routePoints[r].push([p.x, p.y]);
  }
}

function computeRouteLength(points) {
  let prev = depot;
  let total = 0;
  for (const pt of points) {
    total += dist(prev, pt);
    prev = pt;
  }
  total += dist(prev, depot);
  return total;
}

for (let r = 0; r < routeCount; r++) {
  routeLen[r] = computeRouteLength(routePoints[r]);
}

function savingsMove(routeA, routeB, i, j) {
  const A = routePoints[routeA];
  const B = routePoints[routeB];
  const pa = A[i];
  const pb = B[j];

  const prevA = i === 0 ? depot : A[i - 1];
  const nextA = i === A.length - 1 ? depot : A[i + 1];
  const prevB = j === 0 ? depot : B[j - 1];
  const nextB = j === B.length - 1 ? depot : B[j + 1];

  const before = dist(prevA, pa) + dist(pa, nextA) + dist(prevB, pb) + dist(pb, nextB);
  const after = dist(prevA, pb) + dist(pb, nextA) + dist(prevB, pa) + dist(pa, nextB);
  return before - after;
}

function applySwap(routeA, routeB, i, j) {
  const A = routePoints[routeA];
  const B = routePoints[routeB];
  const tmpP = A[i];
  A[i] = B[j];
  B[j] = tmpP;
  const tmpId = routes[routeA][i];
  routes[routeA][i] = routes[routeB][j];
  routes[routeB][j] = tmpId;
}

for (let iter = 0; iter < 3; iter++) {
  let improved = false;
  for (let a = 0; a < routeCount; a++) {
    for (let b = a + 1; b < routeCount; b++) {
      for (let i = 0; i < routePoints[a].length; i++) {
        for (let j = 0; j < routePoints[b].length; j++) {
          const gain = savingsMove(a, b, i, j);
          if (gain > 1e-9) {
            applySwap(a, b, i, j);
            routeLen[a] -= gain * 0.5;
            routeLen[b] -= gain * 0.5;
            improved = true;
          }
        }
      }
    }
  }
  if (!improved) break;
}

function twoOpt(points) {
  let best = computeRouteLength(points);
  if (points.length < 4) return { points, best };
  let changed = true;
  while (changed) {
    changed = false;
    for (let i = 0; i < points.length - 2; i++) {
      for (let k = i + 2; k < points.length; k++) {
        const a = i === -1 ? depot : points[i];
        const b = points[i + 1];
        const c = points[k];
        const d = k === points.length - 1 ? depot : points[k + 1];
        const before = dist(a, b) + dist(c, d);
        const after = dist(a, c) + dist(b, d);
        if (after + 1e-9 < before) {
          let l = i + 1, r = k;
          while (l < r) {
            const t = points[l];
            points[l] = points[r];
            points[r] = t;
            l++; r--;
          }
          best += after - before;
          changed = true;
        }
      }
    }
  }
  return { points, best };
}

for (let r = 0; r < routeCount; r++) {
  const res = twoOpt(routePoints[r]);
  routePoints[r] = res.points;
  routeLen[r] = res.best;
  routes[r] = routePoints[r].map((_, i) => routes[r][i]);
}

const out = { routes: routes.filter(route => route.length > 0) };
process.stdout.write(JSON.stringify(out));
