const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot || [0, 0];
const packages = Array.isArray(data.packages) ? data.packages.slice() : [];
const truckCount = Math.max(1, data.truckCount | 0);

const dx0 = depot[0];
const dy0 = depot[1];

for (const p of packages) {
  p._dx = p.x - dx0;
  p._dy = p.y - dy0;
  p._dist = Math.hypot(p._dx, p._dy);
  p._angle = Math.atan2(p._dy, p._dx);
}

packages.sort((a, b) => {
  if (a._angle !== b._angle) return a._angle - b._angle;
  if (a._dist !== b._dist) return a._dist - b._dist;
  return String(a.id).localeCompare(String(b.id));
});

function routeDistance(route) {
  let prevX = dx0;
  let prevY = dy0;
  let dist = 0;
  for (const p of route) {
    dist += Math.hypot(p.x - prevX, p.y - prevY);
    prevX = p.x;
    prevY = p.y;
  }
  dist += Math.hypot(prevX - dx0, prevY - dy0);
  return dist;
}

function improve2Opt(route) {
  if (route.length < 4) return route;
  let improved = true;
  let passes = 0;
  while (improved && passes < 2) {
    improved = false;
    passes++;
    for (let i = 0; i < route.length - 2; i++) {
      const a = i === 0 ? null : route[i - 1];
      const b = route[i];
      const bNext = route[i + 1];
      const c = route[i + 2];
      const d = i + 3 < route.length ? route[i + 3] : null;

      const x1 = a ? a.x : dx0;
      const y1 = a ? a.y : dy0;
      const x2 = b.x, y2 = b.y;
      const x3 = bNext.x, y3 = bNext.y;
      const x4 = c.x, y4 = c.y;
      const x5 = d ? d.x : dx0;
      const y5 = d ? d.y : dy0;

      const current =
        Math.hypot(x2 - x1, y2 - y1) +
        Math.hypot(x3 - x2, y3 - y2) +
        Math.hypot(x4 - x3, y4 - y3) +
        Math.hypot(x5 - x4, y5 - y4);
      const swapped =
        Math.hypot(x3 - x1, y3 - y1) +
        Math.hypot(x4 - x3, y4 - y3) +
        Math.hypot(x2 - x4, y2 - y4) +
        Math.hypot(x5 - x2, y5 - y2);

      if (swapped + 1e-9 < current) {
        route[i + 1] = c;
        route[i + 2] = bNext;
        improved = true;
      }
    }
  }
  return route;
}

function buildRoute(slice) {
  const route = slice.slice();
  route.sort((a, b) => {
    if (a._dist !== b._dist) return a._dist - b._dist;
    return String(a.id).localeCompare(String(b.id));
  });
  route.sort((a, b) => a._angle - b._angle);
  return improve2Opt(route);
}

const n = packages.length;
const k = Math.min(truckCount, n === 0 ? 1 : n);
const routes = [];

if (n > 0) {
  const base = Math.floor(n / k);
  let extra = n % k;
  let idx = 0;
  for (let i = 0; i < k; i++) {
    const size = base + (extra > 0 ? 1 : 0);
    if (extra > 0) extra--;
    if (size <= 0) continue;
    const slice = packages.slice(idx, idx + size);
    idx += size;
    const route = buildRoute(slice);
    routes.push(route.map(p => p.id));
  }
}

if (routes.length === 0) routes.push([]);

process.stdout.write(JSON.stringify({ routes }));
