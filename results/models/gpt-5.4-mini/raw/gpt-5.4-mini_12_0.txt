using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class Program
{
    struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    }

    struct Quat
    {
        public double W, X, Y, Z;
        public Quat(double w, double x, double y, double z) { W = w; X = x; Y = y; Z = z; }
    }

    struct Placement
    {
        public double Tx, Ty, Tz;
        public Quat Q;
        public Placement(double tx, double ty, double tz, Quat q) { Tx = tx; Ty = ty; Tz = tz; Q = q; }
    }

    static double[] Input = Array.Empty<double>();
    static int Ptr;

    static double NextDouble() => Input[Ptr++];
    static int NextInt() => (int)NextDouble();

    static Vec3 Rotate(Vec3 p, Quat q)
    {
        double w = q.W, x = q.X, y = q.Y, z = q.Z;
        double xx = x + x, yy = y + y, zz = z + z;
        double wx = w * xx, wy = w * yy, wz = w * zz;
        double xx2 = x * xx, xy = x * yy, xz = x * zz;
        double yy2 = y * yy, yz = y * zz, zz2 = z * zz;
        return new Vec3(
            (1 - (yy2 + zz2)) * p.X + (xy - wz) * p.Y + (xz + wy) * p.Z,
            (xy + wz) * p.X + (1 - (xx2 + zz2)) * p.Y + (yz - wx) * p.Z,
            (xz - wy) * p.X + (yz + wx) * p.Y + (1 - (xx2 + yy2)) * p.Z
        );
    }

    static (Vec3 min, Vec3 max, Vec3 size) Bounds(Vec3[] verts, Quat q)
    {
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        foreach (var v in verts)
        {
            var r = Rotate(v, q);
            if (r.X < minX) minX = r.X; if (r.Y < minY) minY = r.Y; if (r.Z < minZ) minZ = r.Z;
            if (r.X > maxX) maxX = r.X; if (r.Y > maxY) maxY = r.Y; if (r.Z > maxZ) maxZ = r.Z;
        }
        return (new Vec3(minX, minY, minZ), new Vec3(maxX, maxY, maxZ), new Vec3(maxX - minX, maxY - minY, maxZ - minZ));
    }

    static void Main()
    {
        var all = new List<double>();
        string? s;
        while ((s = Console.ReadLine()) != null)
        {
            var parts = s.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
            foreach (var p in parts)
                all.Add(double.Parse(p, CultureInfo.InvariantCulture));
        }
        Input = all.ToArray();
        if (Input.Length < 2) return;
        Ptr = 0;

        int V = NextInt();
        int F = NextInt();
        var verts = new Vec3[V];
        for (int i = 0; i < V; i++) verts[i] = new Vec3(NextDouble(), NextDouble(), NextDouble());
        for (int i = 0; i < F; i++) { NextInt(); NextInt(); NextInt(); }
        double W = NextDouble(), H = NextDouble(), D = NextDouble();

        var candidates = new List<Placement>
        {
            new Placement(0,0,0,new Quat(1,0,0,0)),
            new Placement(0,0,0,new Quat(0.7071067811865476,0,0,0.7071067811865476)),
            new Placement(0,0,0,new Quat(0.7071067811865476,0,0.7071067811865476,0)),
            new Placement(0,0,0,new Quat(0.7071067811865476,0.7071067811865476,0,0)),
            new Placement(0,0,0,new Quat(0.5,0.5,0.5,0.5)),
            new Placement(0,0,0,new Quat(0.5,-0.5,0.5,0.5)),
            new Placement(0,0,0,new Quat(0.5,0.5,-0.5,0.5)),
            new Placement(0,0,0,new Quat(0.5,0.5,0.5,-0.5))
        };

        var scored = new List<(double vol, Placement place, Vec3 size, Vec3 min)>
        ;
        foreach (var c in candidates)
        {
            var b = Bounds(verts, c.Q);
            scored.Add((b.size.X * b.size.Y * b.size.Z, c, b.size, b.min));
        }
        scored.Sort((a, b) => a.vol.CompareTo(b.vol));

        var best = scored[0];
        double bx = best.size.X, by = best.size.Y, bz = best.size.Z;
        if (bx <= 0 || by <= 0 || bz <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long nx = (long)Math.Floor(W / bx);
        long ny = (long)Math.Floor(H / by);
        long nz = (long)Math.Floor(D / bz);
        long count = nx * ny * nz;
        Console.WriteLine(count);

        for (long ix = 0; ix < nx; ix++)
        for (long iy = 0; iy < ny; iy++)
        for (long iz = 0; iz < nz; iz++)
        {
            double tx = ix * bx - best.min.X;
            double ty = iy * by - best.min.Y;
            double tz = iz * bz - best.min.Z;
            Console.WriteLine(string.Format(CultureInfo.InvariantCulture, "{0:F9} {1:F9} {2:F9} {3:F9} {4:F9} {5:F9} {6:F9}", tx, ty, tz, best.place.Q.W, best.place.Q.X, best.place.Q.Y, best.place.Q.Z));
        }
    }
}
