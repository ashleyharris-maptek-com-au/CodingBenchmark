#include <bits/stdc++.h>
using namespace std;

struct Solver {
    int n;
    vector<vector<int>> g;
    vector<int> order;
    vector<int> best, cur;
    vector<char> alive;

    int remaining_vertices() const {
        int c = 0;
        for (char x : alive) c += x;
        return c;
    }

    int degree_alive(int v) const {
        int d = 0;
        for (int to : g[v]) if (alive[to]) ++d;
        return d;
    }

    int greedy_coloring_bound() const {
        vector<int> verts;
        verts.reserve(n);
        for (int i = 0; i < n; ++i) if (alive[i]) verts.push_back(i);
        if (verts.empty()) return 0;

        vector<int> color(verts.size(), 0);
        int colors = 0;
        vector<int> idx(n, -1);
        for (int i = 0; i < (int)verts.size(); ++i) idx[verts[i]] = i;

        vector<vector<int>> buckets;
        for (int v : verts) {
            int c = 0;
            for (;;) {
                bool ok = true;
                for (int to : g[v]) {
                    int p = idx[to];
                    if (p != -1 && color[p] == c) {
                        ok = false;
                        break;
                    }
                }
                if (ok) break;
                ++c;
            }
            color[idx[v]] = c;
            colors = max(colors, c + 1);
        }
        return colors;
    }

    int choose_vertex() const {
        int bestv = -1, bestd = -1;
        for (int i = 0; i < n; ++i) if (alive[i]) {
            int d = degree_alive(i);
            if (d > bestd) bestd = d, bestv = i;
        }
        return bestv;
    }

    void include_vertex(int v, vector<int>& removed) {
        if (alive[v]) {
            alive[v] = 0;
            removed.push_back(v);
        }
        for (int to : g[v]) if (alive[to]) {
            alive[to] = 0;
            removed.push_back(to);
        }
    }

    void restore(const vector<int>& removed) {
        for (int v : removed) alive[v] = 1;
    }

    void propagate_forced(vector<int>& picked, vector<int>& removed) {
        bool changed = true;
        while (changed) {
            changed = false;
            for (int v = 0; v < n; ++v) if (alive[v]) {
                int d = 0, last = -1;
                for (int to : g[v]) if (alive[to]) {
                    ++d;
                    last = to;
                    if (d > 1) break;
                }
                if (d == 0) {
                    alive[v] = 0;
                    removed.push_back(v);
                    picked.push_back(v);
                    changed = true;
                    break;
                }
                if (d == 1) {
                    include_vertex(last, removed);
                    picked.push_back(last);
                    changed = true;
                    break;
                }
            }
        }
    }

    void dfs() {
        vector<int> local_picked;
        vector<int> local_removed;
        propagate_forced(local_picked, local_removed);
        for (int v : local_picked) cur.push_back(v);

        int rem = remaining_vertices();
        if ((int)cur.size() + rem <= (int)best.size()) {
            for (int v : local_picked) cur.pop_back();
            restore(local_removed);
            return;
        }

        if (rem == 0) {
            if (cur.size() > best.size()) best = cur;
            for (int v : local_picked) cur.pop_back();
            restore(local_removed);
            return;
        }

        int bound = greedy_coloring_bound();
        if ((int)cur.size() + bound <= (int)best.size()) {
            for (int v : local_picked) cur.pop_back();
            restore(local_removed);
            return;
        }

        int v = choose_vertex();
        if (v == -1) {
            if (cur.size() > best.size()) best = cur;
            for (int x : local_picked) cur.pop_back();
            restore(local_removed);
            return;
        }

        {
            vector<int> removed;
            include_vertex(v, removed);
            cur.push_back(v);
            dfs();
            cur.pop_back();
            restore(removed);
        }

        {
            vector<int> removed;
            alive[v] = 0;
            removed.push_back(v);
            dfs();
            restore(removed);
        }

        for (int v : local_picked) cur.pop_back();
        restore(local_removed);
    }

    vector<int> solve() {
        alive.assign(n, 1);
        dfs();
        sort(best.begin(), best.end());
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    Solver s;
    s.n = n;
    s.g.assign(n, {});
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
        s.g[u].push_back(v);
        s.g[v].push_back(u);
    }
    for (int i = 0; i < n; ++i) {
        sort(s.g[i].begin(), s.g[i].end());
        s.g[i].erase(unique(s.g[i].begin(), s.g[i].end()), s.g[i].end());
    }

    vector<int> ans = s.solve();
    cout << ans.size() << '\n';
    for (size_t i = 0; i < ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
