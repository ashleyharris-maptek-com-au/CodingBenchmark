#include <bits/stdc++.h>
using namespace std;

using ull = unsigned long long;

static int popcount64(ull x) { return (int)__popcnt64(x); }
static int ctz64(ull x) { return (int)__tzcnt_u64(x); }

struct Solver {
    int n;
    int W;
    vector<vector<ull>> adj;
    vector<int> order;
    vector<int> best, cur;
    int bestSize = 0;

    bool isAdjacent(int u, int v) const {
        return (adj[u][v >> 6] >> (v & 63)) & 1ULL;
    }

    void removeVertex(vector<int>& verts, int v) {
        verts.erase(remove(verts.begin(), verts.end(), v), verts.end());
    }

    vector<int> neighborhoodPlusSelf(const vector<int>& verts, int v) const {
        vector<int> out;
        out.reserve(verts.size());
        for (int x : verts) if (x == v || isAdjacent(v, x)) out.push_back(x);
        return out;
    }

    int greedyColorBound(const vector<int>& verts) const {
        vector<int> rem = verts;
        int colors = 0;
        vector<int> colorClass;
        while (!rem.empty()) {
            ++colors;
            colorClass.clear();
            vector<int> next;
            for (int v : rem) {
                bool ok = true;
                for (int u : colorClass) {
                    if (isAdjacent(u, v)) { ok = false; break; }
                }
                if (ok) colorClass.push_back(v);
                else next.push_back(v);
            }
            rem.swap(next);
        }
        return colors;
    }

    int choosePivot(const vector<int>& cand) const {
        int bestv = cand[0], bestd = -1;
        for (int v : cand) {
            int d = 0;
            for (int u : cand) if (u != v && isAdjacent(u, v)) ++d;
            if (d > bestd) bestd = d, bestv = v;
        }
        return bestv;
    }

    void dfs(vector<int> cand) {
        if ((int)cur.size() + (int)cand.size() <= bestSize) return;
        int ub = (int)cur.size() + greedyColorBound(cand);
        if (ub <= bestSize) return;
        if (cand.empty()) {
            if ((int)cur.size() > bestSize) {
                bestSize = (int)cur.size();
                best = cur;
            }
            return;
        }
        int v = choosePivot(cand);
        vector<int> withV = cur;
        withV.push_back(v);
        vector<int> nextCand;
        for (int x : cand) if (x != v && !isAdjacent(v, x)) nextCand.push_back(x);
        cur.push_back(v);
        dfs(nextCand);
        cur.pop_back();

        vector<int> withoutV;
        withoutV.reserve(cand.size() - 1);
        for (int x : cand) if (x != v) withoutV.push_back(x);
        dfs(withoutV);
    }

    vector<int> solve() {
        vector<int> cand(n);
        iota(cand.begin(), cand.end(), 0);
        dfs(cand);
        sort(best.begin(), best.end());
        return best;
    }
};

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    Solver s;
    s.n = n;
    s.W = (n + 63) / 64;
    s.adj.assign(n, vector<ull>(s.W, 0));
    for (int i = 0; i < m; ++i) {
        int u, v; cin >> u >> v;
        if (u == v) continue;
        s.adj[u][v >> 6] |= 1ULL << (v & 63);
        s.adj[v][u >> 6] |= 1ULL << (u & 63);
    }

    vector<int> ans = s.solve();
    cout << ans.size() << '\n';
    for (size_t i = 0; i < ans.size(); ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << '\n';
    return 0;
}
