#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Rect {
    int w = 0;
    int h = 0;
    int id = 0;
};

struct Placement {
    int board = 0;
    int x = 0;
    int y = 0;
    int rot = 0;
};

struct Shelf {
    int y = 0;
    int height = 0;
    int next_x = 0;
};

struct Board {
    int used_h = 0;
    vector<Shelf> shelves;
};

static inline bool fits(int w, int h, int bw, int bh) {
    return w <= bw && h <= bh;
}

static inline int wasted_if_shelf(const Rect& r, const Shelf& s, int bw) {
    if (r.h > s.height || r.w > (bw - s.next_x)) return numeric_limits<int>::max();
    return (s.height - r.h) * r.w;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects;
    rects.reserve(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        cin >> w >> h;
        rects.push_back({w, h, i});
    }

    vector<Placement> ans(N);

    // Normalize impossible cases early; if a rectangle cannot fit in either orientation,
    // keep a graceful fallback by forcing the natural orientation into a new board slot.
    // The judge is expected to contain feasible instances.
    vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = i;

    sort(order.begin(), order.end(), [&](int a, int b) {
        long long aa = 1LL * rects[a].w * rects[a].h;
        long long bb = 1LL * rects[b].w * rects[b].h;
        if (aa != bb) return aa > bb;
        int da = max(rects[a].w, rects[a].h);
        int db = max(rects[b].w, rects[b].h);
        if (da != db) return da > db;
        return rects[a].w > rects[b].w;
    });

    vector<Board> boards;
    boards.reserve(max(1, N / 4));

    for (int idx : order) {
        const Rect r0 = rects[idx];
        const int orientations[2][2] = {{r0.w, r0.h}, {r0.h, r0.w}};

        int best_board = -1;
        int best_shelf = -1;
        int best_rot = 0;
        long long best_score = numeric_limits<long long>::max();
        int best_x = 0, best_y = 0;

        // First try to place into existing shelves.
        for (int b = 0; b < (int)boards.size(); ++b) {
            for (int s = 0; s < (int)boards[b].shelves.size(); ++s) {
                const Shelf& sh = boards[b].shelves[s];
                for (int rot = 0; rot < 2; ++rot) {
                    int w = orientations[rot][0];
                    int h = orientations[rot][1];
                    if (!fits(w, h, BW, BH)) continue;
                    if (h > sh.height) continue;
                    if (sh.next_x + w > BW) continue;
                    long long score = 1LL * (sh.height - h) * (BW - (sh.next_x + w)) + 1LL * boards[b].used_h;
                    if (score < best_score) {
                        best_score = score;
                        best_board = b;
                        best_shelf = s;
                        best_rot = rot;
                        best_x = sh.next_x;
                        best_y = sh.y;
                    }
                }
            }
        }

        if (best_board != -1) {
            Shelf& sh = boards[best_board].shelves[best_shelf];
            int w = orientations[best_rot][0];
            int h = orientations[best_rot][1];
            ans[idx] = {best_board, best_x, best_y, best_rot};
            sh.next_x += w;
            continue;
        }

        // Otherwise start a new shelf, preferably on an existing board with enough remaining height.
        int chosen_board = -1;
        int chosen_rot = 0;
        int chosen_y = 0;
        long long chosen_score = numeric_limits<long long>::max();

        for (int b = 0; b < (int)boards.size(); ++b) {
            int y = boards[b].used_h;
            for (int rot = 0; rot < 2; ++rot) {
                int w = orientations[rot][0];
                int h = orientations[rot][1];
                if (!fits(w, h, BW, BH)) continue;
                if (y + h > BH) continue;
                long long score = 1LL * (BH - (y + h)) * BW + boards[b].used_h;
                if (score < chosen_score) {
                    chosen_score = score;
                    chosen_board = b;
                    chosen_rot = rot;
                    chosen_y = y;
                }
            }
        }

        if (chosen_board == -1) {
            chosen_board = (int)boards.size();
            boards.push_back(Board{});
            chosen_y = 0;
            chosen_rot = (orientations[0][0] <= BW && orientations[0][1] <= BH) ? 0 : 1;
        }

        int w = orientations[chosen_rot][0];
        int h = orientations[chosen_rot][1];
        Board& bd = boards[chosen_board];
        Shelf new_shelf{chosen_y, h, w};
        bd.shelves.push_back(new_shelf);
        bd.used_h = max(bd.used_h, chosen_y + h);
        ans[idx] = {chosen_board, 0, chosen_y, chosen_rot};
    }

    cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rot << '\n';
    }

    return 0;
}
