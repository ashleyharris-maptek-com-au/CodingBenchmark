OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %outColor %ubo

OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"

OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %outColor Location 0

OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
%void = OpTypeVoid
%functy = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float
%UBO = OpTypeStruct %v4float %v4float %v4float %v4float %v4float %v4float
%ptr_uniform_UBO = OpTypePointer Uniform %UBO
%ptr_uniform_v4float = OpTypePointer Uniform %v4float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%int_5 = OpConstant %int 5
%float_0 = OpConstant %float 0.0
%float_02 = OpConstant %float 0.2
%float_05 = OpConstant %float 0.5
%float_08 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_32 = OpConstant %float 32.0
%worldPos = OpVariable %ptr_in_v3float Input
%normal = OpVariable %ptr_in_v3float Input
%outColor = OpVariable %ptr_out_v4float Output
%ubo = OpVariable %ptr_uniform_UBO Uniform
%1 = OpExtInstImport "GLSL.std.450"
%main = OpFunction %void None %functy
%label = OpLabel
%N = OpLoad %v3float %normal
%Nn = OpExtInst %v3float %1 Normalize %N
%P = OpLoad %v3float %worldPos
%light = OpAccessChain %ptr_uniform_v4float %ubo %int_3
%lightv4 = OpLoad %v4float %light
%cam = OpAccessChain %ptr_uniform_v4float %ubo %int_4
%camv4 = OpLoad %v4float %cam
%lightxyz = OpVectorShuffle %v3float %lightv4 %lightv4 0 1 2
%camxyz = OpVectorShuffle %v3float %camv4 %camv4 0 1 2
%L0 = OpFSub %v3float %lightxyz %P
%L = OpExtInst %v3float %1 Normalize %L0
%V0 = OpFSub %v3float %camxyz %P
%V = OpExtInst %v3float %1 Normalize %V0
%diffraw = OpDot %float %Nn %L
%diffuse = OpExtInst %float %1 FMax %diffraw %float_0
%negL = OpFNegate %v3float %L
%R = OpExtInst %v3float %1 Reflect %negL %Nn
%specdot = OpDot %float %R %V
%specclamp = OpExtInst %float %1 FMax %specdot %float_0
%specular = OpExtInst %float %1 Pow %specclamp %float_32
%term1 = OpFMul %float %float_05 %diffuse
%term2 = OpFMul %float %float_08 %specular
%sum1 = OpFAdd %float %float_02 %term1
%intensity = OpFAdd %float %sum1 %term2
%gray = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
OpStore %outColor %gray
OpReturn
OpFunctionEnd
