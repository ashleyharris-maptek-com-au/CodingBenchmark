using System;
using System.Collections.Generic;
using System.IO;

public class Program
{
    private readonly struct Cell
    {
        public readonly int R;
        public readonly int C;
        public Cell(int r, int c) { R = r; C = c; }
    }

    private sealed class Constraint
    {
        public readonly int[] Cells;
        public int Remaining;
        public Constraint(int[] cells, int remaining) { Cells = cells; Remaining = remaining; }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int h = fs.NextInt();
        int w = fs.NextInt();
        if (h <= 0 || w <= 0) return;

        var board = new char[h][];
        for (int r = 0; r < h; r++)
            board[r] = fs.NextBoardLine(w).ToCharArray();

        var safe = Solve(board, h, w);

        using var sw = new StreamWriter(Console.OpenStandardOutput());
        sw.WriteLine(safe.Count);
        for (int i = 0; i < safe.Count; i++)
            sw.WriteLine($"{safe[i].C} {safe[i].R}");
    }

    private static List<Cell> Solve(char[][] board, int h, int w)
    {
        var result = new List<Cell>();
        var emitted = new HashSet<int>();

        var unknown = new bool[h, w];
        var flagged = new bool[h, w];
        var isNumber = new bool[h, w];

        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                char ch = board[r][c];
                if (ch >= '0' && ch <= '8') isNumber[r, c] = true;
                else if (ch == '*') flagged[r, c] = true;
                else unknown[r, c] = true;
            }
        }

        int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
        int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

        var constraints = new List<Constraint>();
        var cellToConstraints = new Dictionary<int, List<int>>();

        for (int r = 0; r < h; r++)
        {
            for (int c = 0; c < w; c++)
            {
                if (!isNumber[r, c]) continue;

                int count = board[r][c] - '0';
                int flaggedCount = 0;
                var unk = new List<int>(8);

                for (int k = 0; k < 8; k++)
                {
                    int nr = r + dr[k], nc = c + dc[k];
                    if ((uint)nr >= (uint)h || (uint)nc >= (uint)w) continue;
                    if (flagged[nr, nc]) flaggedCount++;
                    else if (unknown[nr, nc]) unk.Add(nr * w + nc);
                }

                int remaining = count - flaggedCount;
                if (remaining < 0) continue;
                if (unk.Count == 0) continue;

                if (remaining == 0)
                {
                    for (int i = 0; i < unk.Count; i++) AddSafe(unk[i], w, emitted, result);
                    continue;
                }

                if (remaining == unk.Count)
                {
                    for (int i = 0; i < unk.Count; i++)
                    {
                        int id = unk[i];
                        int rr = id / w, cc = id % w;
                        unknown[rr, cc] = false;
                        flagged[rr, cc] = true;
                    }
                    continue;
                }

                int cid = constraints.Count;
                constraints.Add(new Constraint(unk.ToArray(), remaining));
                for (int i = 0; i < unk.Count; i++)
                {
                    int id = unk[i];
                    if (!cellToConstraints.TryGetValue(id, out var list))
                    {
                        list = new List<int>();
                        cellToConstraints[id] = list;
                    }
                    list.Add(cid);
                }
            }
        }

        if (constraints.Count == 0) return result;

        var vars = new List<int>(cellToConstraints.Keys);
        if (vars.Count == 0) return result;

        var varToIndex = new Dictionary<int, int>(vars.Count);
        for (int i = 0; i < vars.Count; i++) varToIndex[vars[i]] = i;

        var components = BuildComponents(vars, cellToConstraints, constraints, varToIndex);
        foreach (var comp in components)
        {
            if (comp.Count == 0) continue;
            if (comp.Count > 24)
            {
                ApplyLocalRules(comp, constraints, emitted, result, w);
            }
            else
            {
                ExhaustiveDeductions(comp, constraints, emitted, result, w, varToIndex);
            }
        }

        return result;
    }

    private static List<List<int>> BuildComponents(
        List<int> vars,
        Dictionary<int, List<int>> cellToConstraints,
        List<Constraint> constraints,
        Dictionary<int, int> varToIndex)
    {
        var adj = new List<int>[vars.Count];
        for (int i = 0; i < vars.Count; i++) adj[i] = new List<int>();

        foreach (var kv in cellToConstraints)
        {
            int a = varToIndex[kv.Key];
            foreach (int cid in kv.Value)
            {
                var cells = constraints[cid].Cells;
                for (int i = 0; i < cells.Length; i++)
                {
                    int bId = cells[i];
                    if (bId == kv.Key) continue;
                    if (varToIndex.TryGetValue(bId, out int b) && a != b)
                    {
                        adj[a].Add(b);
                    }
                }
            }
        }

        var seen = new bool[vars.Count];
        var comps = new List<List<int>>();
        for (int i = 0; i < vars.Count; i++)
        {
            if (seen[i]) continue;
            var comp = new List<int>();
            var q = new Queue<int>();
            q.Enqueue(i);
            seen[i] = true;
            while (q.Count > 0)
            {
                int u = q.Dequeue();
                comp.Add(vars[u]);
                for (int j = 0; j < adj[u].Count; j++)
                {
                    int v = adj[u][j];
                    if (!seen[v])
                    {
                        seen[v] = true;
                        q.Enqueue(v);
                    }
                }
            }
            comps.Add(comp);
        }
        return comps;
    }

    private static void ExhaustiveDeductions(
        List<int> comp,
        List<Constraint> constraints,
        HashSet<int> emitted,
        List<Cell> result,
        int w,
        Dictionary<int, int> varToIndex)
    {
        var localIndex = new Dictionary<int, int>(comp.Count);
        for (int i = 0; i < comp.Count; i++) localIndex[comp[i]] = i;

        var relevant = new List<int>();
        for (int cid = 0; cid < constraints.Count; cid++)
        {
            var cells = constraints[cid].Cells;
            bool used = false;
            for (int i = 0; i < cells.Length; i++)
            {
                if (localIndex.ContainsKey(cells[i]))
                {
                    used = true;
                    break;
                }
            }
            if (used) relevant.Add(cid);
        }

        int n = comp.Count;
        var consCells = new int[relevant.Count][];
        var consNeed = new int[relevant.Count];
        for (int i = 0; i < relevant.Count; i++)
        {
            var con = constraints[relevant[i]];
            var list = new List<int>();
            for (int j = 0; j < con.Cells.Length; j++)
            {
                if (localIndex.TryGetValue(con.Cells[j], out int idx))
                    list.Add(idx);
            }
            consCells[i] = list.ToArray();
            consNeed[i] = con.Remaining;
        }

        var assignment = new int[n];
        Array.Fill(assignment, -1);
        var canMine = new bool[n];
        var canSafe = new bool[n];

        Dfs(0);

        void Dfs(int pos)
        {
            if (pos == n)
            {
                for (int i = 0; i < n; i++)
                {
                    if (assignment[i] == 1) canMine[i] = true;
                    else canSafe[i] = true;
                }
                return;
            }

            for (int val = 0; val <= 1; val++)
            {
                assignment[pos] = val;
                if (Consistent()) Dfs(pos + 1);
            }
            assignment[pos] = -1;
        }

        bool Consistent()
        {
            for (int i = 0; i < consCells.Length; i++)
            {
                int assigned = 0;
                int undecided = 0;
                var cells = consCells[i];
                for (int j = 0; j < cells.Length; j++)
                {
                    int a = assignment[cells[j]];
                    if (a == -1) undecided++;
                    else assigned += a;
                }
                if (assigned > consNeed[i]) return false;
                if (assigned + undecided < consNeed[i]) return false;
            }
            return true;
        }

        for (int i = 0; i < n; i++)
        {
            if (canSafe[i] && !canMine[i])
                AddSafe(comp[i], w, emitted, result);
        }
    }

    private static void ApplyLocalRules(
        List<int> comp,
        List<Constraint> constraints,
        HashSet<int> emitted,
        List<Cell> result,
        int w)
    {
        var compSet = new HashSet<int>(comp);
        for (int cid = 0; cid < constraints.Count; cid++)
        {
            var con = constraints[cid];
            int inside = 0;
            var cells = new List<int>();
            for (int i = 0; i < con.Cells.Length; i++)
            {
                int id = con.Cells[i];
                if (compSet.Contains(id))
                {
                    inside++;
                    cells.Add(id);
                }
            }
            if (inside == 0) continue;
            if (con.Remaining == 0)
            {
                for (int i = 0; i < cells.Count; i++)
                    AddSafe(cells[i], w, emitted, result);
            }
        }
    }

    private static void AddSafe(int id, int w, HashSet<int> emitted, List<Cell> result)
    {
        if (!emitted.Add(id)) return;
        result.Add(new Cell(id / w, id % w));
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do c = Read(); while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        public string NextBoardLine(int w)
        {
            var chars = new char[w];
            int i = 0;
            while (i < w)
            {
                int c = Read();
                if (c == '\r') continue;
                if (c == '\n')
                {
                    if (i == 0) continue;
                    while (i < w) chars[i++] = ' ';
                    break;
                }
                chars[i++] = (char)c;
            }
            return new string(chars);
        }
    }
}
