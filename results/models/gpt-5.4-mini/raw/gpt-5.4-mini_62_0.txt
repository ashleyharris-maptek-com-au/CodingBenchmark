declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

type Town = { id: string; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = data.grid;
const ignitions: number[][] = data.ignitions || [];
const towns: Town[] = data.towns || [];
const cutBudget: number = data.cutBudget | 0;

const n = width * height;
const passable = new Uint8Array(n);
for (let y = 0; y < height; y++) {
  const row = grid[y];
  for (let x = 0; x < width; x++) {
    passable[y * width + x] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const INF = 1e9;
const dist = new Int32Array(n);
dist.fill(INF);
const prev = new Int32Array(n);
prev.fill(-1);
const q = new Int32Array(n);
let qh = 0, qt = 0;

function idx(x: number, y: number): number {
  return y * width + x;
}

function push(v: number, d: number, p: number) {
  if (d < dist[v]) {
    dist[v] = d;
    prev[v] = p;
    q[qt++] = v;
  }
}

for (const s of ignitions) {
  const x = s[0] | 0;
  const y = s[1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const v = idx(x, y);
  if (!passable[v]) continue;
  push(v, 0, -1);
}

while (qh < qt) {
  const v = q[qh++];
  const d = dist[v];
  const x = v % width;
  const y = (v / width) | 0;
  const nd = d + 1;
  if (x > 0) {
    const u = v - 1;
    if (passable[u]) push(u, nd, v);
  }
  if (x + 1 < width) {
    const u = v + 1;
    if (passable[u]) push(u, nd, v);
  }
  if (y > 0) {
    const u = v - width;
    if (passable[u]) push(u, nd, v);
  }
  if (y + 1 < height) {
    const u = v + width;
    if (passable[u]) push(u, nd, v);
  }
}

const townValue = new Float64Array(n);
for (const t of towns) {
  if (t.x < 0 || t.x >= width || t.y < 0 || t.y >= height) continue;
  const v = idx(t.x, t.y);
  townValue[v] += t.value;
}

const score = new Float64Array(n);
for (let v = 0; v < n; v++) {
  if (!passable[v] || dist[v] >= INF) continue;
  const d = dist[v];
  let s = 0;
  const x = v % width;
  const y = (v / width) | 0;
  if (x > 0) {
    const u = v - 1;
    if (passable[u] && dist[u] === d + 1) s += townValue[u];
  }
  if (x + 1 < width) {
    const u = v + 1;
    if (passable[u] && dist[u] === d + 1) s += townValue[u];
  }
  if (y > 0) {
    const u = v - width;
    if (passable[u] && dist[u] === d + 1) s += townValue[u];
  }
  if (y + 1 < height) {
    const u = v + width;
    if (passable[u] && dist[u] === d + 1) s += townValue[u];
  }
  score[v] = s;
}

const candidates: number[] = [];
for (let v = 0; v < n; v++) {
  if (passable[v] && dist[v] < INF && score[v] > 0) candidates.push(v);
}

candidates.sort((a, b) => score[b] - score[a] || dist[a] - dist[b] || a - b);

const chosen: number[] = [];
const blocked = new Uint8Array(n);
for (const v of candidates) {
  if (chosen.length >= cutBudget) break;
  if (blocked[v]) continue;
  chosen.push(v);
  blocked[v] = 1;
  let u = prev[v];
  while (u !== -1 && !blocked[u]) {
    blocked[u] = 1;
    u = prev[u];
  }
  if (chosen.length >= cutBudget) break;
}

const firebreaks: number[][] = [];
for (const v of chosen) {
  firebreaks.push([v % width, (v / width) | 0]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
