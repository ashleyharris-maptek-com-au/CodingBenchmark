import * as fs from 'fs';

type Point = [number, number];

interface Town {
  id: string | number;
  x: number;
  y: number;
  value: number;
}

interface InputData {
  width: number;
  height: number;
  grid: string[];
  ignitions: Point[];
  towns: Town[];
  cutBudget: number;
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const W = data.width | 0;
const H = data.height | 0;
const N = W * H;
const grid = data.grid;
const ignitions = data.ignitions || [];
const towns = data.towns || [];
const budget = Math.max(0, data.cutBudget | 0);

const idx = (x: number, y: number) => y * W + x;
const inBounds = (x: number, y: number) => x >= 0 && y >= 0 && x < W && y < H;

const burnable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  for (let x = 0; x < W; x++) {
    burnable[idx(x, y)] = row.charCodeAt(x) === 46 ? 1 : 0;
  }
}

const fireSources: number[] = [];
for (const [x, y] of ignitions) {
  if (inBounds(x, y) && burnable[idx(x, y)]) fireSources.push(idx(x, y));
}

if (budget <= 0 || fireSources.length === 0 || towns.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const dist = new Int32Array(N);
dist.fill(-1);
const q = new Int32Array(N);
let qh = 0, qt = 0;
for (const s of fireSources) {
  if (dist[s] === -1) {
    dist[s] = 0;
    q[qt++] = s;
  }
}
while (qh < qt) {
  const v = q[qh++];
  const d = dist[v] + 1;
  const x = v % W;
  const y = (v / W) | 0;
  if (x > 0) {
    const u = v - 1;
    if (burnable[u] && dist[u] === -1) { dist[u] = d; q[qt++] = u; }
  }
  if (x + 1 < W) {
    const u = v + 1;
    if (burnable[u] && dist[u] === -1) { dist[u] = d; q[qt++] = u; }
  }
  if (y > 0) {
    const u = v - W;
    if (burnable[u] && dist[u] === -1) { dist[u] = d; q[qt++] = u; }
  }
  if (y + 1 < H) {
    const u = v + W;
    if (burnable[u] && dist[u] === -1) { dist[u] = d; q[qt++] = u; }
  }
}

interface Candidate {
  id: number;
  x: number;
  y: number;
  score: number;
}

const candidates: Candidate[] = [];
const seen = new Uint8Array(N);
for (const town of towns) {
  const tx = town.x | 0;
  const ty = town.y | 0;
  if (!inBounds(tx, ty)) continue;
  const tIdx = idx(tx, ty);
  if (!burnable[tIdx]) continue;
  const td = dist[tIdx];
  if (td < 0) continue;

  const radius = Math.min(td, 25);
  for (let r = 0; r <= radius; r++) {
    for (let dy = -r; dy <= r; dy++) {
      const dx = r - Math.abs(dy);
      const xs = [tx - dx, tx + dx];
      for (const x of xs) {
        const y = ty + dy;
        if (!inBounds(x, y)) continue;
        const p = idx(x, y);
        if (!burnable[p] || seen[p]) continue;
        const d = dist[p];
        if (d < 0 || d > td) continue;
        let s = town.value / (d + 1);
        if (x === 0 || y === 0 || x === W - 1 || y === H - 1) s *= 1.2;
        if (d <= 2) s *= 2.0;
        candidates.push({ id: p, x, y, score: s });
        seen[p] = 1;
      }
    }
  }
}

if (candidates.length === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

candidates.sort((a, b) => b.score - a.score);

const selected: Point[] = [];
for (const c of candidates) {
  if (selected.length >= budget) break;
  selected.push([c.x, c.y]);
}

process.stdout.write(JSON.stringify({ firebreaks: selected }));
