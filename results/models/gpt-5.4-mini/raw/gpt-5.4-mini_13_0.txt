struct RollingHash64 {
    static constexpr uint64_t base1 = 911382323ull;
    static constexpr uint64_t base2 = 972663749ull;
    vector<uint64_t> p1, p2, h1, h2;

    RollingHash64() = default;
    explicit RollingHash64(const string& s) { init(s); }

    void init(const string& s) {
        size_t n = s.size();
        p1.assign(n + 1, 1);
        p2.assign(n + 1, 1);
        h1.assign(n + 1, 0);
        h2.assign(n + 1, 0);
        for (size_t i = 0; i < n; ++i) {
            p1[i + 1] = p1[i] * base1;
            p2[i + 1] = p2[i] * base2;
            h1[i + 1] = h1[i] * base1 + static_cast<unsigned char>(s[i]) + 1;
            h2[i + 1] = h2[i] * base2 + static_cast<unsigned char>(s[i]) + 1;
        }
    }

    pair<uint64_t, uint64_t> get(size_t l, size_t r) const {
        uint64_t x1 = h1[r] - h1[l] * p1[r - l];
        uint64_t x2 = h2[r] - h2[l] * p2[r - l];
        return {x1, x2};
    }
};

struct Hasher {
    size_t operator()(const pair<uint64_t, uint64_t>& x) const noexcept {
        uint64_t v = x.first ^ (x.second + 0x9e3779b97f4a7c15ull + (x.first << 6) + (x.first >> 2));
        return static_cast<size_t>(v);
    }
};

static bool contains_substring(const string& text, const string& pat) {
    return text.find(pat) != string::npos;
}

static string solve_small(const vector<string>& a) {
    const string& base = *min_element(a.begin(), a.end(), [](const string& x, const string& y) {
        return x.size() < y.size();
    });
    for (size_t len = base.size(); len > 0; --len) {
        for (size_t i = 0; i + len <= base.size(); ++i) {
            string_view cand(base.data() + i, len);
            bool ok = true;
            for (const auto& s : a) {
                if (s.find(cand) == string::npos) { ok = false; break; }
            }
            if (ok) return string(cand);
        }
    }
    return "";
}

static bool common_of_length(const vector<string>& a, int len, string& out) {
    if (len == 0) { out.clear(); return true; }
    int idx = int(min_element(a.begin(), a.end(), [](const string& x, const string& y) { return x.size() < y.size(); }) - a.begin());
    const string& base = a[idx];
    if (base.size() < static_cast<size_t>(len)) return false;

    vector<RollingHash64> rh;
    rh.reserve(a.size());
    for (const auto& s : a) rh.emplace_back(s);

    unordered_map<pair<uint64_t, uint64_t>, vector<int>, Hasher> seen;
    seen.reserve(base.size() * 2 + 1);
    for (size_t i = 0; i + len <= base.size(); ++i) {
        auto key = rh[idx].get(i, i + len);
        seen[key].push_back(static_cast<int>(i));
    }

    for (const auto& s : a) {
        if (&s == &base) continue;
        unordered_set<pair<uint64_t, uint64_t>, Hasher> cur;
        cur.reserve(max<size_t>(1, s.size() - len + 1));
        for (size_t i = 0; i + len <= s.size(); ++i) cur.insert(rh[&s - &a[0]].get(i, i + len));
        for (auto it = seen.begin(); it != seen.end();) {
            if (!cur.count(it->first)) it = seen.erase(it);
            else ++it;
        }
        if (seen.empty()) return false;
    }

    auto key = seen.begin()->first;
    int pos = seen.begin()->second.front();
    out = base.substr(pos, len);
    for (const auto& s : a) {
        if (s.find(out) == string::npos) return false;
    }
    return true;
}

static string solve_hash_bs(const vector<string>& a) {
    int mn = (int)min_element(a.begin(), a.end(), [](const string& x, const string& y) { return x.size() < y.size(); })->size();
    string ans;
    string cur;
    int lo = 0, hi = mn;
    while (lo < hi) {
        int mid = (lo + hi + 1) / 2;
        if (common_of_length(a, mid, cur)) {
            lo = mid;
            ans = cur;
        } else {
            hi = mid - 1;
        }
    }
    if (lo == 0) return "";
    if (ans.size() != static_cast<size_t>(lo)) {
        common_of_length(a, lo, ans);
    }
    return ans;
}

static string solve_suffix_like(vector<string> a) {
    sort(a.begin(), a.end(), [](const string& x, const string& y) { return x.size() < y.size(); });
    string best = a[0];
    for (size_t k = 1; k < a.size(); ++k) {
        const string& s = a[k];
        string next;
        next.reserve(min(best.size(), s.size()));
        for (size_t i = 0; i < best.size(); ++i) {
            for (size_t len = best.size() - i; len > 0; --len) {
                if (i + len > best.size()) continue;
                string_view cand(best.data() + i, len);
                if (s.find(cand) != string::npos) {
                    bool ok = true;
                    for (size_t t = 0; t < k; ++t) {
                        if (a[t].find(cand) == string::npos) { ok = false; break; }
                    }
                    if (ok && len > next.size()) next = string(cand);
                    break;
                }
            }
        }
        best.swap(next);
        if (best.empty()) return "";
    }
    return best;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy);
    vector<string> a(N);
    for (int i = 0; i < N; ++i) getline(cin, a[i]);

    if (N == 0) {
        cout << '\n';
        return 0;
    }

    size_t total = 0, mx = 0;
    for (const auto& s : a) {
        total += s.size();
        mx = max(mx, s.size());
    }

    string ans;
    if (N <= 4 && total <= 5000) {
        ans = solve_small(a);
    } else if (N <= 20 && total <= 200000 && mx <= 50000) {
        ans = solve_hash_bs(a);
    } else {
        ans = solve_hash_bs(a);
    }

    cout << ans << '\n';
    return 0;
}
