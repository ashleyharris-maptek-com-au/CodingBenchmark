using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c != -1);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32 && c != -1)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        var cuts = new int[n];
        for (int i = 0; i < n; i++) cuts[i] = fs.NextInt();

        var order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) =>
        {
            int cmp = cuts[b].CompareTo(cuts[a]);
            return cmp != 0 ? cmp : a.CompareTo(b);
        });

        var bins = new List<List<int>>();
        var remaining = new List<int>();

        foreach (int idx in order)
        {
            int len = cuts[idx];
            if (len > stockLength)
            {
                Console.WriteLine("0");
                return;
            }

            int chosen = -1;
            int bestRem = int.MaxValue;

            for (int b = 0; b < bins.Count; b++)
            {
                int rem = remaining[b];
                if (len <= rem && rem - len < bestRem)
                {
                    bestRem = rem - len;
                    chosen = b;
                    if (bestRem == 0) break;
                }
            }

            if (chosen == -1)
            {
                bins.Add(new List<int>());
                remaining.Add(stockLength);
                chosen = bins.Count - 1;
            }

            bins[chosen].Add(idx);
            remaining[chosen] -= len;
        }

        var writer = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        writer.WriteLine(bins.Count);
        for (int i = 0; i < bins.Count; i++)
        {
            var bin = bins[i];
            for (int j = 0; j < bin.Count; j++)
            {
                if (j > 0) writer.Write(' ');
                writer.Write(bin[j]);
            }
            writer.WriteLine();
        }
        writer.Flush();
    }
}
