using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public class Program
{
    struct Edge
    {
        public int U;
        public int V;
        public long W;
    }

    sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int m = fs.NextInt();
        var edges = new Edge[m];
        var adj = new List<int>[n];
        for (int i = 0; i < n; i++) adj[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            int w = fs.NextInt();
            edges[i] = new Edge { U = u, V = v, W = w };
            adj[u].Add(i);
            adj[v].Add(i);
        }

        var used = new bool[m];
        var iter = new int[n];
        var stack = new Stack<int>();
        var route = new List<int>(m + 1);

        stack.Push(0);
        while (stack.Count > 0)
        {
            int v = stack.Peek();
            while (iter[v] < adj[v].Count && used[adj[v][iter[v]]]) iter[v]++;
            if (iter[v] == adj[v].Count)
            {
                route.Add(v);
                stack.Pop();
            }
            else
            {
                int id = adj[v][iter[v]++];
                if (used[id]) continue;
                used[id] = true;
                int to = edges[id].U == v ? edges[id].V : edges[id].U;
                stack.Push(to);
            }
        }

        route.Reverse();

        var sb = new StringBuilder();
        for (int i = 0; i < route.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb.ToString());
    }
}
