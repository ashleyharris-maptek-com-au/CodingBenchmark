```cpp
#include <algorithm>
#include <array>
#include <cassert>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <numeric>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct FullEdge {
    int u, v;
    long long w;
};

static constexpr long long INF = (1LL << 60);

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<FullEdge> edges;
    edges.reserve(m);
    vector<vector<Edge>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }
    if (k == 1) {
        cout << 0 << ' ' << 0 << '\n';
        return 0;
    }

    if (k <= 10 && n <= 80) {
        int S = 1 << k;
        vector<vector<long long>> dp(S, vector<long long>(n, INF));
        vector<vector<int>> parent_mask(S, vector<int>(n, -1));
        vector<vector<int>> parent_node(S, vector<int>(n, -1));

        for (int i = 0; i < k; ++i) dp[1 << i][terminals[i]] = 0;

        for (int mask = 1; mask < S; ++mask) {
            if ((mask & (mask - 1)) != 0) {
                for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                    int other = mask ^ sub;
                    if (other == 0) continue;
                    for (int v = 0; v < n; ++v) {
                        long long cand = dp[sub][v] + dp[other][v];
                        if (cand < dp[mask][v]) {
                            dp[mask][v] = cand;
                            parent_mask[mask][v] = sub;
                            parent_node[mask][v] = v;
                        }
                    }
                }
            }

            priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
            vector<char> vis(n, 0);
            for (int v = 0; v < n; ++v) {
                if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
            }
            while (!pq.empty()) {
                auto [d, v] = pq.top();
                pq.pop();
                if (vis[v]) continue;
                vis[v] = 1;
                if (d != dp[mask][v]) continue;
                for (const auto &e : g[v]) {
                    long long nd = d + e.w;
                    if (nd < dp[mask][e.to]) {
                        dp[mask][e.to] = nd;
                        parent_mask[mask][e.to] = mask;
                        parent_node[mask][e.to] = v;
                        pq.push({nd, e.to});
                    }
                }
            }
        }

        int full = S - 1;
        long long best = INF;
        int root = -1;
        for (int v = 0; v < n; ++v) {
            if (dp[full][v] < best) {
                best = dp[full][v];
                root = v;
            }
        }

        vector<int> used_edge(m, 0);
        vector<char> seen_state(S * n, 0);
        function<void(int,int)> reconstruct = [&](int mask, int v) {
            int idx = mask * n + v;
            if (seen_state[idx]) return;
            seen_state[idx] = 1;
            if ((mask & (mask - 1)) == 0 && v == terminals[__builtin_ctz(mask)]) return;
            int pm = parent_mask[mask][v];
            int pn = parent_node[mask][v];
            if (pm == -1) return;
            if (pm == mask) {
                if (pn != -1 && pn != v) {
                    for (const auto &e : g[pn]) {
                        if (e.to == v && dp[mask][v] == dp[mask][pn] + e.w) {
                            used_edge[e.id] = 1;
                            reconstruct(mask, pn);
                            break;
                        }
                    }
                }
            } else {
                int sub = pm;
                int other = mask ^ sub;
                reconstruct(sub, v);
                reconstruct(other, v);
            }
        };

        reconstruct(full, root);

        vector<int> ids;
        long long total = 0;
        for (int i = 0; i < m; ++i) {
            if (used_edge[i]) {
                ids.push_back(i);
                total += edges[i].w;
            }
        }

        cout << total << ' ' << ids.size() << '\n';
        for (int id : ids) {
            cout << edges[id].u << ' ' << edges[id].v << '\n';
        }
        return 0;
    }

    vector<vector<long long>> dist(k, vector<long long>(n, INF));
    vector<vector<int>> prev_node(k, vector<int>(n, -1));
    vector<vector<int>> prev_edge(k, vector<int>(n, -1));

    for (int s = 0; s < k; ++s) {
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        dist[s][terminals[s]] = 0;
        pq.push({0, terminals[s]});
        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dist[s][v]) continue;
            for (const auto &e : g[v]) {
                long long nd = d + e.w;
                if (nd < dist[s][e.to]) {
                    dist[s][e.to] = nd;
                    prev_node[s][e.to] = v;
                    prev_edge[s][e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    struct Candidate {
        long long w;
        int a, b;
    };
    vector<Candidate> cand;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            for (int v = 0; v < n; ++v) {
                if (dist[i][v] < INF && dist[j][v] < INF) {
                    cand.push_back({dist[i][v] + dist[j][v], i, j});
                }
            }
        }
    }
    sort(cand.begin(), cand.end(), [](const Candidate &x, const Candidate &y) {
        return x.w < y.w;
    });

    vector<int> parent(k);
    iota(parent.begin(), parent.end(), 0);
    function<int(int)> findp = [&](int x) {
        return parent[x] == x ? x : parent[x] = findp(parent[x]);
    };

    vector<int> used_edge(m, 0);
    auto mark_path = [&](int s, int v) {
        while (v != terminals[s]) {
            int e = prev_edge[s][v];
            if (e < 0) break;
            used_edge[e] = 1;
            v = prev_node[s][v];
        }
    };

    int comps = k;
    for (const auto &c : cand) {
        int pa = findp(c.a), pb = findp(c.b);
        if (pa == pb) continue;
        int best_v = -1;
        long long best_sum = INF;
        for (int v = 0; v < n; ++v) {
            if (dist[c.a][v] < INF && dist[c.b][v] < INF) {
                long long s = dist[c.a][v] + dist[c.b][v];
                if (s < best_sum) {
                    best_sum = s;
                    best_v = v;
                }
            }
        }
        if (best_v == -1) continue;
        mark_path(c.a, best_v);
        mark_path(c.b, best_v);
        parent[pa] = pb;
        --comps;
        if (comps == 1) break;
    }

    vector<int> ids;
    long long total = 0;
    for (int i = 0; i < m; ++i) {
        if (used_edge[i]) {
            ids.push_back(i);
            total += edges[i].w;
        }
    }

    cout << total << ' ' << ids.size() << '\n';
    for (int id : ids) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
    return 0;
}
```
