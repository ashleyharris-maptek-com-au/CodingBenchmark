```cpp
#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct MEdge {
    int u, v;
    long long w;
};

static const long long INF = numeric_limits<long long>::max() / 4;

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<MEdge> edges(m);
    vector<vector<Edge>> g(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges[i] = {u, v, w};
        g[u].push_back({v, w, i});
        g[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }
    if (k == 1) {
        cout << 0 << " " << 0 << "\n";
        return 0;
    }

    auto dijkstra = [&](int src) {
        vector<long long> dist(n, INF);
        vector<int> parent_v(n, -1), parent_e(n, -1);
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        dist[src] = 0;
        pq.push({0, src});
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (d != dist[u]) continue;
            for (const auto& e : g[u]) {
                long long nd = d + e.w;
                if (nd < dist[e.to]) {
                    dist[e.to] = nd;
                    parent_v[e.to] = u;
                    parent_e[e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
        return tuple<vector<long long>, vector<int>, vector<int>>(move(dist), move(parent_v), move(parent_e));
    };

    vector<vector<long long>> td(k);
    vector<vector<int>> tpv(k), tpe(k);
    for (int i = 0; i < k; ++i) {
        auto [dist, pv, pe] = dijkstra(terminals[i]);
        td[i] = move(dist);
        tpv[i] = move(pv);
        tpe[i] = move(pe);
    }

    vector<vector<long long>> dp(1 << k, vector<long long>(n, INF));
    vector<vector<int>> parent_mask(1 << k, vector<int>(n, -1));
    vector<vector<int>> parent_node(1 << k, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) dp[1 << i][terminals[i]] = 0;

    for (int mask = 1; mask < (1 << k); ++mask) {
        if ((mask & (mask - 1)) != 0) {
            for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
                int other = mask ^ sub;
                if (sub > other) continue;
                for (int v = 0; v < n; ++v) {
                    long long cand = dp[sub][v] + dp[other][v];
                    if (cand < dp[mask][v]) {
                        dp[mask][v] = cand;
                        parent_mask[mask][v] = sub;
                        parent_node[mask][v] = v;
                    }
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        vector<char> vis(n, 0);
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
        }
        while (!pq.empty()) {
            auto [d, u] = pq.top();
            pq.pop();
            if (vis[u]) continue;
            vis[u] = 1;
            for (const auto& e : g[u]) {
                long long nd = d + e.w;
                if (nd < dp[mask][e.to]) {
                    dp[mask][e.to] = nd;
                    parent_mask[mask][e.to] = -1;
                    parent_node[mask][e.to] = u;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int full = (1 << k) - 1;
    int best_v = 0;
    for (int v = 1; v < n; ++v) {
        if (dp[full][v] < dp[full][best_v]) best_v = v;
    }

    vector<char> used_edge(m, 0);
    vector<int> stack_mask_node;
    stack_mask_node.push_back(full * (n + 1) + best_v);

    auto mark_path = [&](int src_term_idx, int v) {
        int cur = v;
        while (cur != terminals[src_term_idx]) {
            int p = tpv[src_term_idx][cur];
            int eid = tpe[src_term_idx][cur];
            if (eid >= 0) used_edge[eid] = 1;
            cur = p;
        }
    };

    while (!stack_mask_node.empty()) {
        int code = stack_mask_node.back();
        stack_mask_node.pop_back();
        int mask = code / (n + 1);
        int v = code % (n + 1);
        if (mask == 0) continue;
        if ((mask & (mask - 1)) == 0) {
            int idx = __builtin_ctz(mask);
            mark_path(idx, v);
            continue;
        }
        int sub = parent_mask[mask][v];
        int pnode = parent_node[mask][v];
        if (sub != -1) {
            stack_mask_node.push_back(sub * (n + 1) + pnode);
            stack_mask_node.push_back((mask ^ sub) * (n + 1) + pnode);
        } else {
            int cur = v;
            while (cur != pnode) {
                int prev = parent_node[mask][cur];
                for (const auto& e : g[cur]) {
                    if (e.to == prev && dp[mask][prev] + e.w == dp[mask][cur]) {
                        used_edge[e.id] = 1;
                        break;
                    }
                }
                cur = prev;
            }
            stack_mask_node.push_back(mask * (n + 1) + pnode);
        }
    }

    vector<int> ans;
    long long total = 0;
    for (int i = 0; i < m; ++i) {
        if (used_edge[i]) {
            ans.push_back(i);
            total += edges[i].w;
        }
    }

    cout << total << " " << ans.size() << "\n";
    for (int id : ans) {
        cout << edges[id].u << " " << edges[id].v << "\n";
    }
    return 0;
}
```
