use std::io::{self, Read, Write, BufWriter};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

#[derive(Clone, Copy, Default)]
struct Centroid {
    x: f64,
    y: f64,
    z: f64,
}

fn dist2_point_centroid(p: Point, c: Centroid) -> f64 {
    let dx = p.x as f64 - c.x;
    let dy = p.y as f64 - c.y;
    let dz = p.z as f64 - c.z;
    dx * dx + dy * dy + dz * dz
}

fn dist2_point_point(a: Point, b: Point) -> f64 {
    let dx = a.x as f64 - b.x as f64;
    let dy = a.y as f64 - b.y as f64;
    let dz = a.z as f64 - b.z as f64;
    dx * dx + dy * dy + dz * dz
}

fn choose_initial_centroids(points: &[Point], k: usize) -> Vec<Centroid> {
    let n = points.len();
    let mut centroids = Vec::with_capacity(k);
    if n == 0 || k == 0 {
        return centroids;
    }
    centroids.push(Centroid {
        x: points[0].x as f64,
        y: points[0].y as f64,
        z: points[0].z as f64,
    });
    if k == 1 {
        return centroids;
    }

    let mut farthest_idx = 0usize;
    let mut farthest_dist = -1.0f64;
    for (i, &p) in points.iter().enumerate() {
        let d = dist2_point_point(p, points[0]);
        if d > farthest_dist {
            farthest_dist = d;
            farthest_idx = i;
        }
    }
    centroids.push(Centroid {
        x: points[farthest_idx].x as f64,
        y: points[farthest_idx].y as f64,
        z: points[farthest_idx].z as f64,
    });

    let mut min_dists = vec![0.0f64; n];
    for i in 0..n {
        min_dists[i] = dist2_point_centroid(points[i], centroids[0]).min(dist2_point_centroid(points[i], centroids[1]));
    }

    while centroids.len() < k {
        let mut best_idx = 0usize;
        let mut best_dist = -1.0f64;
        for i in 0..n {
            if min_dists[i] > best_dist {
                best_dist = min_dists[i];
                best_idx = i;
            }
        }
        centroids.push(Centroid {
            x: points[best_idx].x as f64,
            y: points[best_idx].y as f64,
            z: points[best_idx].z as f64,
        });
        let last = *centroids.last().unwrap();
        for i in 0..n {
            let d = dist2_point_centroid(points[i], last);
            if d < min_dists[i] {
                min_dists[i] = d;
            }
        }
    }

    centroids
}

fn assign_points(points: &[Point], centroids: &[Centroid], labels: &mut [u32]) -> bool {
    let mut changed = false;
    for (i, &p) in points.iter().enumerate() {
        let mut best = 0usize;
        let mut best_dist = dist2_point_centroid(p, centroids[0]);
        for c in 1..centroids.len() {
            let d = dist2_point_centroid(p, centroids[c]);
            if d < best_dist {
                best_dist = d;
                best = c;
            }
        }
        if labels[i] != best as u32 {
            labels[i] = best as u32;
            changed = true;
        }
    }
    changed
}

fn recompute_centroids(points: &[Point], k: usize, labels: &[u32], centroids: &mut [Centroid]) {
    let mut sum_x = vec![0.0f64; k];
    let mut sum_y = vec![0.0f64; k];
    let mut sum_z = vec![0.0f64; k];
    let mut count = vec![0u64; k];

    for (i, &p) in points.iter().enumerate() {
        let c = labels[i] as usize;
        sum_x[c] += p.x as f64;
        sum_y[c] += p.y as f64;
        sum_z[c] += p.z as f64;
        count[c] += 1;
    }

    for c in 0..k {
        if count[c] > 0 {
            let inv = 1.0f64 / (count[c] as f64);
            centroids[c] = Centroid {
                x: sum_x[c] * inv,
                y: sum_y[c] * inv,
                z: sum_z[c] * inv,
            };
        }
    }
}

fn fix_empty_clusters(points: &[Point], labels: &mut [u32], centroids: &mut [Centroid]) {
    let k = centroids.len();
    if k == 0 || points.is_empty() {
        return;
    }
    let mut counts = vec![0usize; k];
    for &l in labels.iter() {
        counts[l as usize] += 1;
    }

    let mut empties = Vec::new();
    for c in 0..k {
        if counts[c] == 0 {
            empties.push(c);
        }
    }
    if empties.is_empty() {
        return;
    }

    let mut farthest_points: Vec<(f64, usize)> = labels.iter().enumerate().map(|(i, &l)| {
        let d = dist2_point_centroid(points[i], centroids[l as usize]);
        (d, i)
    }).collect();
    farthest_points.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(std::cmp::Ordering::Equal));

    let mut idx = 0usize;
    for &empty in &empties {
        while idx < farthest_points.len() {
            let pi = farthest_points[idx].1;
            let old = labels[pi] as usize;
            if counts[old] > 1 {
                counts[old] -= 1;
                labels[pi] = empty as u32;
                centroids[empty] = Centroid {
                    x: points[pi].x as f64,
                    y: points[pi].y as f64,
                    z: points[pi].z as f64,
                };
                counts[empty] = 1;
                idx += 1;
                break;
            }
            idx += 1;
        }
    }
}

fn kmeans(points: &[Point], k: usize, max_iters: usize) -> Vec<u32> {
    let n = points.len();
    let mut labels = vec![0u32; n];
    if n == 0 || k == 0 {
        return labels;
    }
    if k >= n {
        for i in 0..n {
            labels[i] = i as u32;
        }
        return labels;
    }

    let mut centroids = choose_initial_centroids(points, k);
    if centroids.len() < k {
        centroids.resize(k, centroids[0]);
    }

    for _ in 0..max_iters {
        let changed = assign_points(points, &centroids, &mut labels);
        fix_empty_clusters(points, &mut labels, &mut centroids);
        recompute_centroids(points, k, &labels, &mut centroids);
        if !changed {
            break;
        }
    }

    labels
}

fn streaming_assign(points: &[Point], k: usize) -> Vec<u32> {
    let n = points.len();
    let mut labels = vec![0u32; n];
    if n == 0 || k == 0 {
        return labels;
    }
    if k >= n {
        for i in 0..n {
            labels[i] = i as u32;
        }
        return labels;
    }

    let mut centroids = Vec::with_capacity(k);
    let seed_count = n.min(k.max(1));
    for i in 0..seed_count {
        centroids.push(Centroid {
            x: points[i].x as f64,
            y: points[i].y as f64,
            z: points[i].z as f64,
        });
    }
    while centroids.len() < k {
        centroids.push(centroids[centroids.len() % seed_count]);
    }

    let mut counts = vec![1u64; k];
    for i in seed_count..n {
        let p = points[i];
        let mut best = 0usize;
        let mut best_dist = dist2_point_centroid(p, centroids[0]);
        for c in 1..k {
            let d = dist2_point_centroid(p, centroids[c]);
            if d < best_dist {
                best_dist = d;
                best = c;
            }
        }
        labels[i] = best as u32;
        let cnt = counts[best] as f64;
        let eta = 1.0 / (cnt + 1.0);
        centroids[best].x = centroids[best].x + (p.x as f64 - centroids[best].x) * eta;
        centroids[best].y = centroids[best].y + (p.y as f64 - centroids[best].y) * eta;
        centroids[best].z = centroids[best].z + (p.z as f64 - centroids[best].z) * eta;
        counts[best] += 1;
    }

    fix_empty_clusters(points, &mut labels, &mut centroids);
    labels
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    if input.len() < 8 {
        return;
    }

    let num_points = u32::from_le_bytes([input[0], input[1], input[2], input[3]]) as usize;
    let num_clusters = u32::from_le_bytes([input[4], input[5], input[6], input[7]]) as usize;
    let mut points = Vec::with_capacity(num_points);
    let mut offset = 8usize;
    for _ in 0..num_points {
        if offset + 12 > input.len() {
            break;
        }
        let x = f32::from_le_bytes([input[offset], input[offset + 1], input[offset + 2], input[offset + 3]]);
        let y = f32::from_le_bytes([input[offset + 4], input[offset + 5], input[offset + 6], input[offset + 7]]);
        let z = f32::from_le_bytes([input[offset + 8], input[offset + 9], input[offset + 10], input[offset + 11]]);
        points.push(Point { x, y, z });
        offset += 12;
    }

    let n = points.len();
    let k = num_clusters.max(1).min(n.max(1));

    let labels = if n <= 10_000_000 {
        kmeans(&points, k, 15)
    } else if n <= 100_000_000 {
        let mut lab = streaming_assign(&points, k);
        let mut centroids = choose_initial_centroids(&points, k);
        if centroids.len() < k && !centroids.is_empty() {
            centroids.resize(k, centroids[0]);
        }
        for _ in 0..5 {
            recompute_centroids(&points, k, &lab, &mut centroids);
            let changed = assign_points(&points, &centroids, &mut lab);
            fix_empty_clusters(&points, &mut lab, &mut centroids);
            if !changed {
                break;
            }
        }
        lab
    } else {
        streaming_assign(&points, k)
    };

    let mut out = BufWriter::new(io::stdout().lock());
    for label in labels {
        out.write_all(&label.to_le_bytes()).unwrap();
    }
    out.flush().unwrap();
}
