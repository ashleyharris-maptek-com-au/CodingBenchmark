import math


_STATE = {
    "prev": None,
    "vel_est": [0.0, 0.0, 0.0],
    "bias_thrust": [0.0, 0.0, 0.0],
    "cam_lock": False,
}


def _clamp(v, lo, hi):
    return lo if v < lo else hi if v > hi else v


def _valid(state, key, default=False):
    return bool(state.get(key, default))


def _num(state, key, default=0.0):
    v = state.get(key, default)
    try:
        if v is None:
            return default
        if isinstance(v, bool):
            return 1.0 if v else 0.0
        return float(v)
    except Exception:
        return default


def _median(vals):
    vals = sorted(vals)
    n = len(vals)
    if n == 0:
        return 0.0
    m = n // 2
    return vals[m] if n % 2 else 0.5 * (vals[m - 1] + vals[m])


def _fuse_axis(state, pos_keys, vel_keys, valid_pos_keys, valid_vel_keys):
    vals = []
    for pk, vk, pvk, vvk in zip(pos_keys, vel_keys, valid_pos_keys, valid_vel_keys):
        if _valid(state, pvk):
            vals.append(("p", _num(state, pk)))
        if _valid(state, vvk):
            vals.append(("v", _num(state, vk)))
    return vals


def _fuse_position(state):
    pos = [0.0, 0.0, 0.0]
    counts = [0, 0, 0]

    sources = [
        (
            ("radar_rel_pos_x_m", "radar_rel_pos_y_m", "radar_rel_pos_z_m"),
            ("radar_rel_x_valid", "radar_rel_y_valid", "radar_rel_z_valid"),
        ),
        (
            ("lidar_rel_pos_x_m", "lidar_rel_pos_y_m", "lidar_rel_pos_z_m"),
            ("lidar_rel_x_valid", "lidar_rel_y_valid", "lidar_rel_z_valid"),
        ),
        (
            ("gps_1_pos_x_m", "gps_1_pos_y_m", "gps_1_pos_z_m"),
            ("gps_1_x_valid", "gps_1_y_valid", "gps_1_z_valid"),
        ),
        (
            ("gps_2_pos_x_m", "gps_2_pos_y_m", "gps_2_pos_z_m"),
            ("gps_2_x_valid", "gps_2_y_valid", "gps_2_z_valid"),
        ),
    ]

    for keys, vkeys in sources:
        for i in range(3):
            if _valid(state, vkeys[i]):
                pos[i] += _num(state, keys[i])
                counts[i] += 1

    out = [None, None, None]
    for i in range(3):
        if counts[i]:
            out[i] = pos[i] / counts[i]

    return out


def _fuse_velocity(state):
    vel = [0.0, 0.0, 0.0]
    counts = [0, 0, 0]
    sources = [
        (
            ("radar_rel_vel_x_ms", "radar_rel_vel_y_ms", "radar_rel_vel_z_ms"),
            ("radar_relv_x_valid", "radar_relv_y_valid", "radar_relv_z_valid"),
        ),
        (
            ("lidar_rel_vel_x_ms", "lidar_rel_vel_y_ms", "lidar_rel_vel_z_ms"),
            ("lidar_relv_x_valid", "lidar_relv_y_valid", "lidar_relv_z_valid"),
        ),
    ]
    for keys, vkeys in sources:
        for i in range(3):
            if _valid(state, vkeys[i]):
                vel[i] += _num(state, keys[i])
                counts[i] += 1
    out = [None, None, None]
    for i in range(3):
        if counts[i]:
            out[i] = vel[i] / counts[i]
    return out


def _range_estimate(state):
    vals = []
    for i in (1, 2, 3):
        if _valid(state, f"range_{i}_valid"):
            vals.append(_num(state, f"range_{i}_m"))
    if vals:
        return _median(vals)
    return None


def _camera_bearing(state):
    if not _valid(state, "dock_camera_active"):
        return None
    vals = []
    for ax in ("x", "y", "z"):
        if _valid(state, f"dock_camera_bearing_{ax}_valid"):
            vals.append(_num(state, f"dock_camera_bearing_{ax}"))
        else:
            vals.append(None)
    if all(v is not None for v in vals):
        n = math.sqrt(sum(v * v for v in vals)) or 1.0
        return [v / n for v in vals]
    return None


def _infer_from_range_and_bearing(rng, bear):
    if rng is None or bear is None:
        return None
    return [rng * bear[0], rng * bear[1], rng * bear[2]]


def autopilot_step(state: dict, dt: float) -> dict:
    pos = _fuse_position(state)
    vel = _fuse_velocity(state)
    rng = _range_estimate(state)
    bear = _camera_bearing(state)
    cam_pos = _infer_from_range_and_bearing(rng, bear)
    if cam_pos is not None:
        _STATE["cam_lock"] = True

    if pos[0] is None and cam_pos is not None:
        pos = cam_pos[:]
    elif pos[0] is None and rng is not None:
        pos = [0.0, -rng, 0.0]

    if vel[0] is None:
        vel = _STATE["vel_est"][:]
    else:
        _STATE["vel_est"] = [0.85 * _STATE["vel_est"][i] + 0.15 * vel[i] for i in range(3)]

    if pos[1] is None:
        pos[1] = -rng if rng is not None else 0.0
    if pos[2] is None:
        pos[2] = 0.0
    if pos[0] is None:
        pos[0] = 0.0

    if vel[0] is None:
        vel[0] = _STATE["vel_est"][0]
    if vel[1] is None:
        vel[1] = _STATE["vel_est"][1]
    if vel[2] is None:
        vel[2] = _STATE["vel_est"][2]

    dist = math.sqrt(pos[0] ** 2 + pos[1] ** 2 + pos[2] ** 2)
    speed = math.sqrt(vel[0] ** 2 + vel[1] ** 2 + vel[2] ** 2)

    if _STATE["prev"] is not None and dt > 1e-6:
        dpos = [pos[i] - _STATE["prev"]["pos"][i] for i in range(3)]
        estv = [dpos[i] / dt for i in range(3)]
        vel = [0.7 * vel[i] + 0.3 * estv[i] for i in range(3)]
    _STATE["prev"] = {"pos": pos[:], "vel": vel[:]}

    fx = fy = fz = 0.0

    if dist > 200.0:
        if pos[1] < -20.0:
            fy = -1.0
        elif pos[1] > 20.0:
            fy = 0.7
        else:
            fy = _clamp(-0.015 * pos[1] - 0.6 * vel[1], -0.6, 0.6)
        fx = _clamp(-0.02 * pos[0] - 0.5 * vel[0], -0.5, 0.5)
        fz = _clamp(-0.02 * pos[2] - 0.5 * vel[2], -0.5, 0.5)
    elif dist > 50.0:
        fy = _clamp(-0.02 * pos[1] - 0.7 * vel[1], -0.45, 0.45)
        fx = _clamp(-0.03 * pos[0] - 0.8 * vel[0], -0.35, 0.35)
        fz = _clamp(-0.03 * pos[2] - 0.8 * vel[2], -0.35, 0.35)
        if pos[1] < 0:
            fy = min(fy, 0.15) if abs(pos[0]) < 30 else fy
    elif dist > 10.0:
        fx = _clamp(-0.05 * pos[0] - 1.0 * vel[0], -0.25, 0.25)
        fy = _clamp(-0.05 * pos[1] - 1.0 * vel[1], -0.25, 0.25)
        fz = _clamp(-0.05 * pos[2] - 1.0 * vel[2], -0.25, 0.25)
    else:
        fx = _clamp(-0.08 * pos[0] - 1.5 * vel[0], -0.12, 0.12)
        fy = _clamp(-0.08 * pos[1] - 1.5 * vel[1], -0.12, 0.12)
        fz = _clamp(-0.08 * pos[2] - 1.5 * vel[2], -0.12, 0.12)

    if dist > 30.0 and abs(pos[2]) > 5.0:
        fz = _clamp(fz + _clamp(-0.01 * pos[2], -0.2, 0.2), -1.0, 1.0)

    if _valid(state, "fuel_remaining_pct") and _num(state, "fuel_remaining_pct") < 15.0:
        fx *= 0.6
        fy *= 0.6
        fz *= 0.6

    if _valid(state, "fuel_remaining_kg") and _num(state, "fuel_remaining_kg") < 80.0:
        fx *= 0.7
        fy *= 0.7
        fz *= 0.7

    if abs(vel[1]) > 0.8 and pos[1] < 0:
        fy = max(fy, -0.9)
    if abs(vel[1]) > 0.8 and pos[1] > 0:
        fy = min(fy, 0.9)

    if rng is not None and rng < 10.0 and speed < 0.3:
        fx *= 0.5
        fy *= 0.5
        fz *= 0.5

    bias = _STATE["bias_thrust"]
    if _STATE["prev"] is not None:
        bias[0] = 0.98 * bias[0] + 0.02 * (-0.002 * pos[0] - 0.01 * vel[0])
        bias[1] = 0.98 * bias[1] + 0.02 * (-0.002 * pos[1] - 0.01 * vel[1])
        bias[2] = 0.98 * bias[2] + 0.02 * (-0.002 * pos[2] - 0.01 * vel[2])
    fx = _clamp(fx + bias[0], -1.0, 1.0)
    fy = _clamp(fy + bias[1], -1.0, 1.0)
    fz = _clamp(fz + bias[2], -1.0, 1.0)

    if _STATE["cam_lock"] and rng is not None and rng < 500.0:
        fx = _clamp(fx * 0.9, -1.0, 1.0)
        fy = _clamp(fy * 0.9, -1.0, 1.0)
        fz = _clamp(fz * 0.9, -1.0, 1.0)

    roll = _clamp(-0.015 * vel[2], -0.035, 0.035)
    pitch = _clamp(0.015 * vel[0], -0.035, 0.035)
    yaw = _clamp(0.015 * vel[1], -0.035, 0.035)

    return {
        "thrust_x": float(_clamp(fx, -1.0, 1.0)),
        "thrust_y": float(_clamp(fy, -1.0, 1.0)),
        "thrust_z": float(_clamp(fz, -1.0, 1.0)),
        "att_rate_roll": float(roll),
        "att_rate_pitch": float(pitch),
        "att_rate_yaw": float(yaw),
    }
