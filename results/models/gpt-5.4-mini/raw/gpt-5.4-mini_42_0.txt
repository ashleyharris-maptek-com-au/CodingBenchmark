OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpMemberName %ubo 0 "model"
OpMemberName %ubo 1 "view"
OpMemberName %ubo 2 "proj"
OpMemberName %ubo 3 "lightPos"
OpMemberName %ubo 4 "cameraPos"
OpMemberName %ubo 5 "params"

OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpDecorate %arr_pv_in ArrayStride 16

OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpMemberDecorate %ubo 0 ColMajor
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 0 MatrixStride 16
OpMemberDecorate %ubo 1 ColMajor
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 1 MatrixStride 16
OpMemberDecorate %ubo 2 ColMajor
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 2 MatrixStride 16
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %ubo Block
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_04 = OpConstant %float 0.04

%mat4 = OpTypeMatrix %v4float 4
%v4uint = OpTypeVector %uint 4
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_input_v3 = OpTypePointer Input %v3float
%ptr_uniform_ubo = OpTypePointer Uniform %ubo

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float

%worldPos = OpVariable %ptr_input_v3 Input
%normal = OpVariable %ptr_input_v3 Input
%color = OpVariable %ptr_input_v3 Input
%gl_in = OpVariable %ptr_in_arr_pv Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo_var = OpVariable %ptr_uniform_ubo Uniform

%main = OpFunction %void None %func
%label = OpLabel

%gl_in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%center = OpLoad %v4float %gl_in_pos_ptr
%center_x = OpCompositeExtract %float %center 0
%center_y = OpCompositeExtract %float %center 1
%color0 = OpLoad %v3float %color

%x_minus = OpFSub %float %center_x %float_0_04
%x_plus = OpFAdd %float %center_x %float_0_04
%y_minus = OpFSub %float %center_y %float_0_04
%y_plus = OpFAdd %float %center_y %float_0_04

%pos0_xy = OpCompositeConstruct %v2float %x_minus %y_minus
%pos1_xy = OpCompositeConstruct %v2float %x_plus %y_minus
%pos2_xy = OpCompositeConstruct %v2float %x_minus %y_plus
%pos3_xy = OpCompositeConstruct %v2float %x_plus %y_plus

%pos0_xy_x = OpCompositeExtract %float %pos0_xy 0
%pos0_xy_y = OpCompositeExtract %float %pos0_xy 1
%pos1_xy_x = OpCompositeExtract %float %pos1_xy 0
%pos1_xy_y = OpCompositeExtract %float %pos1_xy 1
%pos2_xy_x = OpCompositeExtract %float %pos2_xy 0
%pos2_xy_y = OpCompositeExtract %float %pos2_xy 1
%pos3_xy_x = OpCompositeExtract %float %pos3_xy 0
%pos3_xy_y = OpCompositeExtract %float %pos3_xy 1

%pos0 = OpCompositeConstruct %v4float %pos0_xy_x %pos0_xy_y %float_0 %float_1
%pos1 = OpCompositeConstruct %v4float %pos1_xy_x %pos1_xy_y %float_0 %float_1
%pos2 = OpCompositeConstruct %v4float %pos2_xy_x %pos2_xy_y %float_0 %float_1
%pos3 = OpCompositeConstruct %v4float %pos3_xy_x %pos3_xy_y %float_0 %float_1

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr %pos0
OpStore %outColor %color0
OpEmitVertex

OpStore %out_pos_ptr %pos1
OpStore %outColor %color0
OpEmitVertex

OpStore %out_pos_ptr %pos2
OpStore %outColor %color0
OpEmitVertex

OpStore %out_pos_ptr %pos3
OpStore %outColor %color0
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
