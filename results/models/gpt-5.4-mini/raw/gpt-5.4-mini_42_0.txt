OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

%void = OpTypeVoid
%fn = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%ptr_in_float = OpTypePointer Input %float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_out_v3float = OpTypePointer Output %v3float
%ptr_ubo_mat4 = OpTypePointer Uniform %mat4
%ptr_ubo_v4float = OpTypePointer Uniform %v4float
%ptr_ubo_v3float = OpTypePointer Uniform %v3float
%ptr_ubo_float = OpTypePointer Uniform %float
%ptr_ubo_int = OpTypePointer Uniform %int

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_1 = OpConstant %uint 1
%uint_4 = OpConstant %uint 4
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_04 = OpConstant %float 0.04

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
%arr_worldPos = OpTypeArray %v3float %uint_1
%arr_normal = OpTypeArray %v3float %uint_1
%arr_color = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_worldPos
%ptr_in_arr_n = OpTypePointer Input %arr_normal
%ptr_in_arr_c = OpTypePointer Input %arr_color
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_n Input
%color = OpVariable %ptr_in_arr_c Input

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
OpDecorate %outColor Location 0
%outColor = OpVariable %ptr_out_v3float Output

%ubo = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
OpMemberDecorate %ubo 0 ColMajor
OpMemberDecorate %ubo 0 Offset 0
OpMemberDecorate %ubo 0 MatrixStride 16
OpMemberDecorate %ubo 1 ColMajor
OpMemberDecorate %ubo 1 Offset 64
OpMemberDecorate %ubo 1 MatrixStride 16
OpMemberDecorate %ubo 2 ColMajor
OpMemberDecorate %ubo 2 Offset 128
OpMemberDecorate %ubo 2 MatrixStride 16
OpMemberDecorate %ubo 3 Offset 192
OpMemberDecorate %ubo 4 Offset 208
OpMemberDecorate %ubo 5 Offset 224
OpDecorate %ubo Block
%ptr_ubo = OpTypePointer Uniform %ubo
%uboVar = OpVariable %ptr_ubo Uniform
OpDecorate %uboVar DescriptorSet 0
OpDecorate %uboVar Binding 0

%main = OpFunction %void None %fn
%label = OpLabel
%p_ptr = OpAccessChain %ptr_in_v3float %gl_in %int_0 %int_0
%p = OpLoad %v3float %p_ptr
%c_ptr = OpAccessChain %ptr_in_v3float %color %int_0
%c = OpLoad %v3float %c_ptr
%x = OpCompositeExtract %float %p 0
%y = OpCompositeExtract %float %p 1
%z = OpCompositeExtract %float %p 2
%xh = OpFSub %float %x %float_04
%yh = OpFSub %float %y %float_04
%xp = OpFAdd %float %x %float_04
%yp = OpFAdd %float %y %float_04

%v0 = OpCompositeConstruct %v4float %xh %yh %float_0 %float_1
%p0 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %p0 %v0
OpStore %outColor %c
OpEmitVertex

%v1 = OpCompositeConstruct %v4float %xp %yh %float_0 %float_1
%p1 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %p1 %v1
OpStore %outColor %c
OpEmitVertex

%v2 = OpCompositeConstruct %v4float %xh %yp %float_0 %float_1
%p2 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %p2 %v2
OpStore %outColor %c
OpEmitVertex

%v3 = OpCompositeConstruct %v4float %xp %yp %float_0 %float_1
%p3 = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %p3 %v3
OpStore %outColor %c
OpEmitVertex
OpEndPrimitive
OpReturn
OpFunctionEnd
