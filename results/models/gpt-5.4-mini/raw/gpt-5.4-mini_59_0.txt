const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);
const h = grid.length;
const w = h ? grid[0].length : 0;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < w && y < h;
}

function cellValue(x, y) {
  return grid[y][x];
}

function key(x, y) { return x + ',' + y; }

const starts = drones.map(d => ({ x: d.x | 0, y: d.y | 0 }));
const n = starts.length;

// Precompute connected components of non-obstacle cells for reachability.
const comp = Array.from({ length: h }, () => Array(w).fill(-1));
const comps = [];
let compId = 0;
const dirs = [[1,0],[-1,0],[0,1],[0,-1]];
for (let y = 0; y < h; y++) {
  for (let x = 0; x < w; x++) {
    if (grid[y][x] < 0 || comp[y][x] !== -1) continue;
    const qx = [x], qy = [y];
    comp[y][x] = compId;
    const cells = [];
    for (let qi = 0; qi < qx.length; qi++) {
      const cx = qx[qi], cy = qy[qi];
      cells.push([cx, cy]);
      for (const [dx, dy] of dirs) {
        const nx = cx + dx, ny = cy + dy;
        if (!inBounds(nx, ny) || grid[ny][nx] < 0 || comp[ny][nx] !== -1) continue;
        comp[ny][nx] = compId;
        qx.push(nx); qy.push(ny);
      }
    }
    comps.push(cells);
    compId++;
  }
}

// BFS from each drone start within its component.
function bfsFrom(sx, sy) {
  const dist = Array.from({ length: h }, () => Array(w).fill(Infinity));
  const prev = Array.from({ length: h }, () => Array(w).fill(null));
  const qx = [sx], qy = [sy];
  dist[sy][sx] = 0;
  for (let qi = 0; qi < qx.length; qi++) {
    const x = qx[qi], y = qy[qi];
    for (const [dx, dy] of dirs) {
      const nx = x + dx, ny = y + dy;
      if (!inBounds(nx, ny) || grid[ny][nx] < 0) continue;
      if (dist[ny][nx] !== Infinity) continue;
      dist[ny][nx] = dist[y][x] + 1;
      prev[ny][nx] = [x, y];
      qx.push(nx); qy.push(ny);
    }
  }
  return { dist, prev };
}

const reach = starts.map(s => bfsFrom(s.x, s.y));

// Build cell list with a simple value heuristic favoring high value and accessibility.
const cells = [];
for (let y = 0; y < h; y++) {
  for (let x = 0; x < w; x++) {
    const v = grid[y][x];
    if (v <= 0) continue;
    cells.push({ x, y, v, degree: dirs.reduce((a, [dx, dy]) => a + (inBounds(x + dx, y + dy) && grid[y + dy][x + dx] >= 0 ? 1 : 0), 0) });
  }
}

cells.sort((a, b) => b.v - a.v || a.degree - b.degree);

// Assign each valuable cell to the drone that can reach it cheapest; break ties by current load.
const assigned = Array.from({ length: n }, () => []);
const load = Array(n).fill(0);
for (const c of cells) {
  let best = -1, bestScore = Infinity;
  for (let i = 0; i < n; i++) {
    const d = reach[i].dist[c.y][c.x];
    if (!Number.isFinite(d)) continue;
    const score = d + load[i] * 0.15 - c.v * 0.001;
    if (score < bestScore) {
      bestScore = score;
      best = i;
    }
  }
  if (best !== -1) {
    assigned[best].push(c);
    load[best] += c.v;
  }
}

function reconstructPath(prev, sx, sy, tx, ty) {
  const path = [];
  let x = tx, y = ty;
  while (!(x === sx && y === sy)) {
    path.push([x, y]);
    const p = prev[y][x];
    if (!p) break;
    [x, y] = p;
  }
  path.push([sx, sy]);
  path.reverse();
  return path;
}

// Create a compact walk for each drone by chaining shortest paths to assigned targets.
const paths = [];
const usedGlobal = new Set();
for (let i = 0; i < n; i++) {
  const s = starts[i];
  const order = assigned[i].sort((a, b) => reach[i].dist[a.y][a.x] - reach[i].dist[b.y][b.x] || b.v - a.v);
  let cur = [s.x, s.y];
  let path = [[cur[0], cur[1]]];
  let time = 0;
  const localVisited = new Set([key(cur[0], cur[1])]);
  for (const t of order) {
    if (time >= maxSteps) break;
    const targetPath = reconstructPath(reach[i].prev, cur[0], cur[1], t.x, t.y);
    for (let j = 1; j < targetPath.length && time < maxSteps; j++) {
      const [nx, ny] = targetPath[j];
      path.push([nx, ny]);
      localVisited.add(key(nx, ny));
      cur = [nx, ny];
      time++;
    }
    usedGlobal.add(key(t.x, t.y));
  }

  // If time remains, do a short local sweep around current position to pick up nearby value cells.
  if (time < maxSteps) {
    const q = [[cur[0], cur[1]]];
    const seen = new Set([key(cur[0], cur[1])]);
    const parent = new Map();
    let pick = null;
    while (q.length) {
      const [x, y] = q.shift();
      if (grid[y][x] > 0 && !localVisited.has(key(x, y))) { pick = [x, y]; break; }
      for (const [dx, dy] of dirs) {
        const nx = x + dx, ny = y + dy;
        const k = key(nx, ny);
        if (!inBounds(nx, ny) || grid[ny][nx] < 0 || seen.has(k)) continue;
        seen.add(k);
        parent.set(k, [x, y]);
        q.push([nx, ny]);
      }
    }
    if (pick) {
      const seg = [];
      let x = pick[0], y = pick[1];
      while (!(x === cur[0] && y === cur[1])) {
        seg.push([x, y]);
        [x, y] = parent.get(key(x, y));
      }
      seg.reverse();
      for (const p of seg) {
        if (time >= maxSteps) break;
        path.push(p);
        time++;
      }
    }
  }

  while (path.length < maxSteps + 1) path.push([cur[0], cur[1]]);
  paths.push(path);
}
