const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);
const grid = data.probability || [];
const drones = data.drones || [];
const maxSteps = Math.max(0, data.maxStepsPerDrone | 0);
const H = grid.length;
const W = H ? grid[0].length : 0;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function cellValue(x, y) {
  return grid[y][x];
}

function key(x, y) {
  return y * W + x;
}

const dirs = [
  [1, 0],
  [-1, 0],
  [0, 1],
  [0, -1],
];

// Precompute all positive, non-obstacle cells.
const cells = [];
for (let y = 0; y < H; y++) {
  for (let x = 0; x < W; x++) {
    const v = grid[y][x];
    if (v > 0) cells.push({ x, y, v, k: key(x, y) });
  }
}

// If there are no drones, return an empty structure.
if (!drones.length) {
  process.stdout.write(JSON.stringify({ paths: [] }));
  process.exit(0);
}

// Multi-source Manhattan distances from each cell to nearest drone start.
const dCount = drones.length;
const dist = Array.from({ length: dCount }, () => new Int32Array(H * W).fill(1e9));
for (let i = 0; i < dCount; i++) {
  const s = drones[i];
  if (inBounds(s.x, s.y) && grid[s.y][s.x] !== -1) {
    dist[i][key(s.x, s.y)] = 0;
    const q = [[s.x, s.y]];
    let head = 0;
    while (head < q.length) {
      const [x, y] = q[head++];
      const cd = dist[i][key(x, y)];
      for (const [dx, dy] of dirs) {
        const nx = x + dx, ny = y + dy;
        if (!inBounds(nx, ny) || grid[ny][nx] === -1) continue;
        const nk = key(nx, ny);
        if (dist[i][nk] > cd + 1) {
          dist[i][nk] = cd + 1;
          q.push([nx, ny]);
        }
      }
    }
  }
}

// Assign each positive cell to the most promising drone using value / travel cost.
const assigned = Array.from({ length: dCount }, () => []);
for (const c of cells) {
  let best = 0;
  let bestScore = -1;
  for (let i = 0; i < dCount; i++) {
    const d = dist[i][c.k];
    if (d >= 1e9) continue;
    // Higher value and shorter travel are both good; bias toward reachable close cells.
    const score = c.v / (d + 1);
    if (score > bestScore) {
      bestScore = score;
      best = i;
    }
  }
  assigned[best].push(c);
}

function nearestCurrentPosIndex(path, x, y) {
  let bestIdx = 0;
  let bestDist = 1e9;
  for (let i = 0; i < path.length; i++) {
    const p = path[i];
    const d = Math.abs(p[0] - x) + Math.abs(p[1] - y);
    if (d < bestDist) {
      bestDist = d;
      bestIdx = i;
    }
  }
  return bestIdx;
}

function buildPathForDrone(start, targets) {
  const path = [[start.x, start.y]];
  let cx = start.x, cy = start.y;
  const visited = new Set([key(cx, cy)]);

  // Prefer higher-value targets near the current path.
  targets.sort((a, b) => b.v - a.v || a.k - b.k);

  for (const t of targets) {
    if (path.length >= maxSteps + 1) break;
    // Greedy step-by-step move toward the target.
    while ((cx !== t.x || cy !== t.y) && path.length < maxSteps + 1) {
      let options = [];
      for (const [dx, dy] of dirs) {
        const nx = cx + dx, ny = cy + dy;
        if (!inBounds(nx, ny) || grid[ny][nx] === -1) continue;
        const nd = Math.abs(nx - t.x) + Math.abs(ny - t.y);
        options.push({ nx, ny, nd, v: grid[ny][nx] });
      }
      // Include stay-in-place as a fallback.
      options.push({ nx: cx, ny: cy, nd: Math.abs(cx - t.x) + Math.abs(cy - t.y), v: grid[cy][cx] });
      options.sort((a, b) => a.nd - b.nd || b.v - a.v);
      const chosen = options[0];
      cx = chosen.nx; cy = chosen.ny;
      path.push([cx, cy]);
      visited.add(key(cx, cy));
    }
    if (path.length >= maxSteps + 1) break;

    // If we reached the target, linger one step if that cell is valuable, otherwise continue.
    if (cx === t.x && cy === t.y) {
      if (path.length < maxSteps + 1) path.push([cx, cy]);
    }
  }

  // If there is remaining budget, do a local sweep of nearby high-value cells.
  while (path.length < maxSteps + 1) {
    let best = null;
    let bestScore = -1;
    for (const [dx, dy] of dirs) {
      const nx = cx + dx, ny = cy + dy;
      if (!inBounds(nx, ny) || grid[ny][nx] === -1) continue;
      const k = key(nx, ny);
      const gain = visited.has(k) ? 0.15 : Math.max(0.1, grid[ny][nx]);
      const score = gain - (Math.abs(dx) + Math.abs(dy)) * 0.01;
      if (score > bestScore) {
        bestScore = score;
        best = [nx, ny];
      }
    }
    if (!best) {
      path.push([cx, cy]);
    } else {
      cx = best[0]; cy = best[1];
      path.push([cx, cy]);
      visited.add(key(cx, cy));
    }
  }

  return path;
}

const paths = [];
for (let i = 0; i < dCount; i++) {
  const s = drones[i];
  const path = buildPathForDrone(s, assigned[i]);
  // Ensure exact length maxStepsPerDrone + 1 when possible; stay in place as needed.
  while (path.length < maxSteps + 1) path.push(path[path.length - 1].slice());
  paths.push(path.slice(0, maxSteps + 1));
}

process.stdout.write(JSON.stringify({ paths }));
