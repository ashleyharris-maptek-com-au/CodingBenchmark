using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;
using System.Threading;

public static class Program
{
    private static ulong[][] Adj;
    private static int[] Best;
    private static int BestSize;
    private static long DeadlineTicks;
    private static readonly object BestLock = new object();

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        int m = fs.NextInt();
        var adjLists = new List<int>[n];
        for (int i = 0; i < n; i++) adjLists[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n || u == v) continue;
            adjLists[u].Add(v);
            adjLists[v].Add(u);
        }

        Adj = BuildBitsets(adjLists, n);
        Best = new int[0];
        BestSize = 0;

        // Keep a hard wall-clock bound so we always emit a valid clique.
        DeadlineTicks = DateTime.UtcNow.Ticks + TimeSpan.FromMinutes(4.75).Ticks;

        // Cheap lower bound from greedy expansion.
        var greedy = GreedyClique(adjLists, n);
        UpdateBest(greedy);

        // Order vertices by descending degree to improve branching.
        int[] order = Enumerable.Range(0, n).OrderByDescending(x => adjLists[x].Count).ToArray();

        // Parallelize the root branching when there is enough work.
        if (n >= 40)
        {
            int rootCount = Math.Min(n, 8);
            var threads = new Thread[rootCount];
            for (int i = 0; i < rootCount; i++)
            {
                int idx = i;
                threads[i] = new Thread(() =>
                {
                    if (TimeExpired()) return;
                    int v = order[idx];
                    int[] cand = IntersectWithNeighbors(order, idx + 1, n, v);
                    var current = new int[1] { v };
                    Search(current, cand, cand.Length);
                });
                threads[i].IsBackground = true;
                threads[i].Start();
            }
            for (int i = 0; i < rootCount; i++) threads[i].Join();
        }
        else
        {
            for (int i = 0; i < n; i++)
            {
                if (TimeExpired()) break;
                int v = order[i];
                int[] cand = IntersectWithNeighbors(order, i + 1, n, v);
                Search(new int[1] { v }, cand, cand.Length);
            }
        }

        // Fallback in case search was cut short.
        if (BestSize == 0 && n > 0) Best = new[] { 0 }; // singleton is always a clique if graph non-empty
        BestSize = Best.Length;

        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        sw.WriteLine(BestSize);
        if (BestSize > 0)
        {
            for (int i = 0; i < BestSize; i++)
            {
                if (i > 0) sw.Write(' ');
                sw.Write(Best[i]);
            }
            sw.WriteLine();
        }
        else
        {
            sw.WriteLine();
        }
        sw.Flush();
    }

    private static ulong[][] BuildBitsets(List<int>[] adjLists, int n)
    {
        int words = (n + 63) >> 6;
        var bits = new ulong[n][];
        for (int i = 0; i < n; i++) bits[i] = new ulong[words];
        for (int u = 0; u < n; u++)
        {
            foreach (int v in adjLists[u])
            {
                bits[u][v >> 6] |= 1UL << (v & 63);
            }
        }
        return bits;
    }

    private static int[] GreedyClique(List<int>[] adjLists, int n)
    {
        int[] order = Enumerable.Range(0, n).OrderByDescending(x => adjLists[x].Count).ToArray();
        var clique = new List<int>();
        foreach (int v in order)
        {
            bool ok = true;
            for (int i = 0; i < clique.Count; i++)
            {
                int u = clique[i];
                if (((Adj[v][u >> 6] >> (u & 63)) & 1UL) == 0)
                {
                    ok = false;
                    break;
                }
            }
            if (ok) clique.Add(v);
        }
        return clique.ToArray();
    }

    private static void UpdateBest(int[] clique)
    {
        lock (BestLock)
        {
            if (clique.Length > BestSize)
            {
                BestSize = clique.Length;
                Best = (int[])clique.Clone();
            }
        }
    }

    private static bool TimeExpired() => DateTime.UtcNow.Ticks >= DeadlineTicks;

    private static int[] IntersectWithNeighbors(int[] order, int start, int n, int v)
    {
        var tmp = new int[n - start];
        int count = 0;
        for (int i = start; i < n; i++)
        {
            int u = order[i];
            if (((Adj[v][u >> 6] >> (u & 63)) & 1UL) != 0) tmp[count++] = u;
        }
        Array.Resize(ref tmp, count);
        return tmp;
    }

    private static void Search(int[] clique, int[] candidates, int candCount)
    {
        if (TimeExpired()) return;
        if (clique.Length + candCount <= BestSize) return;

        if (candCount == 0)
        {
            UpdateBest(clique);
            return;
        }

        // Color bound: greedy coloring over the candidate set.
        int[] colors = new int[candCount];
        int maxColor = ColorBound(candidates, candCount, colors);
        if (clique.Length + maxColor <= BestSize) return;

        for (int i = candCount - 1; i >= 0; i--)
        {
            if (TimeExpired()) return;
            if (clique.Length + i + 1 <= BestSize) return;

            int v = candidates[i];
            int[] nextClique = new int[clique.Length + 1];
            Array.Copy(clique, nextClique, clique.Length);
            nextClique[clique.Length] = v;

            int[] nextCand = new int[i];
            int nextCount = 0;
            for (int j = 0; j < i; j++)
            {
                int u = candidates[j];
                if (((Adj[v][u >> 6] >> (u & 63)) & 1UL) != 0)
                    nextCand[nextCount++] = u;
            }
            Array.Resize(ref nextCand, nextCount);
            Search(nextClique, nextCand, nextCount);
        }
    }

    private static int ColorBound(int[] candidates, int count, int[] colors)
    {
        int maxColor = 0;
        for (int i = 0; i < count; i++)
        {
            int v = candidates[i];
            int c = 1;
            while (c <= maxColor)
            {
                bool conflict = false;
                for (int j = 0; j < i; j++)
                {
                    if (colors[j] != c) continue;
                    int u = candidates[j];
                    if (((Adj[v][u >> 6] >> (u & 63)) & 1UL) != 0)
                    {
                        conflict = true;
                        break;
                    }
                }
                if (!conflict) break;
                c++;
            }
            if (c > maxColor) maxColor = c;
            colors[i] = c;
        }
        return maxColor;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }
    }
}
