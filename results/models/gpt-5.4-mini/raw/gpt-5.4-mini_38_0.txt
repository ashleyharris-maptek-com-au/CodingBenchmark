import sys


def main():
    data = sys.stdin.buffer.read().splitlines()
    if not data:
        return

    u_s = data[0].split()
    if len(u_s) < 2:
        sys.stdout.write("NO SOLUTION\n")
        return
    U = int(u_s[0])
    S = int(u_s[1])

    sets = []
    elem_to_sets = [[] for _ in range(U)]
    for idx in range(1, min(len(data), S + 1)):
        line = data[idx].strip()
        if not line:
            elems = []
        else:
            elems = list(map(int, line.split()))
        mask = 0
        ok = True
        for e in elems:
            if e < 0 or e >= U:
                ok = False
                break
            bit = 1 << e
            if mask & bit:
                ok = False
                break
            mask |= bit
        if not ok:
            sets.append(None)
            continue
        sets.append((mask, elems))
        for e in elems:
            elem_to_sets[e].append(idx - 1)

    while len(sets) < S:
        sets.append((0, []))

    full = (1 << U) - 1
    cover = [0] * U
    for i, item in enumerate(sets):
        if item is None:
            continue
        mask, elems = item
        for e in elems:
            cover[e] |= 1 << i

    solution = []

    def choose_element(uncovered):
        best_e = -1
        best_count = 10**9
        m = uncovered
        while m:
            lsb = m & -m
            e = lsb.bit_length() - 1
            count = 0
            mm = cover[e]
            while mm:
                sid = (mm & -mm).bit_length() - 1
                item = sets[sid]
                if item is not None and (item[0] & uncovered) == item[0]:
                    count += 1
                mm &= mm - 1
            if count < best_count:
                best_count = count
                best_e = e
                if count <= 1:
                    break
            m &= m - 1
        return best_e

    sys.setrecursionlimit(1000000)

    def search(uncovered):
        if uncovered == 0:
            return True
        e = choose_element(uncovered)
        if e < 0:
            return False
        candidates = []
        mm = cover[e]
        while mm:
            sid = (mm & -mm).bit_length() - 1
            item = sets[sid]
            if item is not None:
                mask, _ = item
                if (mask & uncovered) == mask:
                    candidates.append(sid)
            mm &= mm - 1
        candidates.sort(key=lambda sid: sets[sid][0].bit_count())
        for sid in candidates:
            mask, _ = sets[sid]
            solution.append(sid)
            if search(uncovered ^ mask):
                return True
            solution.pop()
        return False

    if search(full):
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write(" ".join(map(str, solution)) + "\n")
    else:
        sys.stdout.write("NO SOLUTION\n")


if __name__ == "__main__":
    main()
