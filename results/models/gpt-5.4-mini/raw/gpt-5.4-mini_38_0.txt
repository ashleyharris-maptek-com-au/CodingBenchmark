import sys


def solve():
    data = sys.stdin.buffer.read().splitlines()
    if not data:
        return

    first = data[0].split()
    if len(first) < 2:
        sys.stdout.write("NO SOLUTION\n")
        return

    U = int(first[0])
    S = int(first[1])

    sets = []
    element_to_sets = [[] for _ in range(U)]

    for idx in range(1, min(len(data), S + 1)):
        line = data[idx].split()
        elems = []
        seen = set()
        for x in line:
            e = int(x)
            if 0 <= e < U and e not in seen:
                seen.add(e)
                elems.append(e)
        if elems:
            sets.append((idx - 1, tuple(elems)))
            for e in elems:
                element_to_sets[e].append(idx - 1)
        else:
            sets.append((idx - 1, tuple()))

    # Keep original indices, but build fast lookup structures.
    index_to_elems = {idx: elems for idx, elems in sets}

    remaining = set(range(U))
    chosen = []

    # Cache coverage lists per element; ignore empty sets in search.
    def search():
        if not remaining:
            return True

        # Choose the uncovered element with the fewest compatible sets.
        target = None
        candidates = None
        best_len = 10**9
        for e in remaining:
            cur = []
            for sidx in element_to_sets[e]:
                elems = index_to_elems[sidx]
                ok = True
                for x in elems:
                    if x not in remaining:
                        ok = False
                        break
                if ok:
                    cur.append(sidx)
            if not cur:
                return False
            if len(cur) < best_len:
                best_len = len(cur)
                target = e
                candidates = cur
                if best_len == 1:
                    break

        for sidx in candidates:
            elems = index_to_elems[sidx]
            removed = []
            for x in elems:
                if x in remaining:
                    remaining.remove(x)
                    removed.append(x)
            chosen.append(sidx)
            if search():
                return True
            chosen.pop()
            for x in removed:
                remaining.add(x)
        return False

    if search():
        chosen.sort()
        sys.stdout.write("SOLUTION\n")
        sys.stdout.write(" ".join(map(str, chosen)) + "\n")
    else:
        sys.stdout.write("NO SOLUTION\n")


if __name__ == "__main__":
    solve()
