; SPIR-V
; Geometry shader: expand each input point into a diamond triangle strip

OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpSource GLSL 450

%void = OpTypeVoid
%3 = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%uint_1 = OpConstant %uint 1
%float_0 = OpConstant %float 0.0
%float_005 = OpConstant %float 0.05
%float_1 = OpConstant %float 1.0
%v3float_0 = OpConstantComposite %v3float %float_0 %float_0 %float_0
%v4float_0 = OpConstantComposite %v4float %float_0 %float_0 %float_0 %float_1

%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input
%ptr_in_pv = OpTypePointer Input %gl_PerVertex_in
%ptr_in_v4 = OpTypePointer Input %v4float

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
%arr_v3 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
OpDecorate %outColor Location 0
%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_v3float = OpTypePointer Uniform %v3float
%Params = OpTypeStruct %mat4 %mat4 %mat4 %v3float %v3float %v3float
OpMemberName %Params 0 "model"
OpMemberName %Params 1 "view"
OpMemberName %Params 2 "proj"
OpMemberName %Params 3 "lightPos"
OpMemberName %Params 4 "cameraPos"
OpMemberName %Params 5 "params"
OpMemberDecorate %Params 0 ColMajor
OpMemberDecorate %Params 0 Offset 0
OpMemberDecorate %Params 1 ColMajor
OpMemberDecorate %Params 1 Offset 64
OpMemberDecorate %Params 2 ColMajor
OpMemberDecorate %Params 2 Offset 128
OpMemberDecorate %Params 3 Offset 192
OpMemberDecorate %Params 4 Offset 208
OpMemberDecorate %Params 5 Offset 224
OpDecorate %Params Block
%_ptr_Uniform_Params = OpTypePointer Uniform %Params
%ubo = OpVariable %_ptr_Uniform_Params Uniform

%main = OpFunction %void None %3
%label = OpLabel
%p = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos4 = OpLoad %v4float %p
%wp = OpAccessChain %ptr_in_v3 %worldPos %int_0
%cptr = OpAccessChain %ptr_in_v3 %color %int_0
%col = OpLoad %v3float %cptr
%x = OpCompositeExtract %float %pos4 0
%y = OpCompositeExtract %float %pos4 1
%z = OpCompositeExtract %float %pos4 2
%w = OpCompositeExtract %float %pos4 3
%y_top = OpFAdd %float %y %float_005
%x_right = OpFAdd %float %x %float_005
%y_bottom = OpFSub %float %y %float_005
%x_left = OpFSub %float %x %float_005
%top = OpCompositeConstruct %v4float %x %y_top %z %w
%right = OpCompositeConstruct %v4float %x_right %y %z %w
%left = OpCompositeConstruct %v4float %x_left %y %z %w
%bottom = OpCompositeConstruct %v4float %x %y_bottom %z %w

OpStore %gl_out %top
OpStore %outColor %col
OpEmitVertex
OpStore %gl_out %right
OpStore %outColor %col
OpEmitVertex
OpStore %gl_out %left
OpStore %outColor %col
OpEmitVertex
OpStore %gl_out %bottom
OpStore %outColor %col
OpEmitVertex
OpEndPrimitive
OpFunctionEnd
