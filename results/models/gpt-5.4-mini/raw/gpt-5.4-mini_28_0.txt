use std::cmp::Reverse;
use std::collections::BinaryHeap;
use std::io::{self, Read, Write};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
        edges.push((u, v));
    }

    let mut active = vec![true; n];
    let mut deg = vec![0usize; n];
    for i in 0..n {
        deg[i] = adj[i].len();
    }

    let mut heap: BinaryHeap<(Reverse<usize>, usize)> = BinaryHeap::new();
    for v in 0..n {
        heap.push((Reverse(deg[v]), v));
    }

    let mut removed = vec![false; n];
    let mut cover = vec![false; n];

    while let Some((Reverse(d), v)) = heap.pop() {
        if removed[v] || !active[v] || deg[v] != d {
            continue;
        }
        if deg[v] == 0 {
            removed[v] = true;
            active[v] = false;
            continue;
        }

        let mut best_neighbor = None;
        let mut best_score = 0usize;
        for &to in &adj[v] {
            if active[to] {
                let score = deg[to];
                if score > best_score {
                    best_score = score;
                    best_neighbor = Some(to);
                }
            }
        }

        let take_v = best_neighbor.map_or(true, |u| deg[v] >= deg[u]);
        let chosen = if take_v {
            v
        } else {
            best_neighbor.unwrap()
        };

        if cover[chosen] || removed[chosen] {
            continue;
        }

        cover[chosen] = true;
        active[chosen] = false;
        removed[chosen] = true;

        for &to in &adj[chosen] {
            if active[to] {
                active[to] = false;
                removed[to] = true;
                for &w in &adj[to] {
                    if active[w] && deg[w] > 0 {
                        deg[w] -= 1;
                        heap.push((Reverse(deg[w]), w));
                    }
                }
            }
        }
    }

    let mut uncovered = Vec::new();
    for &(u, v) in &edges {
        if !(cover[u] || cover[v]) {
            uncovered.push((u, v));
        }
    }

    if !uncovered.is_empty() {
        let mut chosen = vec![false; n];
        for &(u, v) in &uncovered {
            if !chosen[u] && !chosen[v] {
                let pick = if deg[u] >= deg[v] { u } else { v };
                chosen[pick] = true;
            }
        }
        for i in 0..n {
            if chosen[i] {
                cover[i] = true;
            }
        }
    }

    let mut ans = Vec::new();
    for i in 0..n {
        if cover[i] {
            ans.push(i);
        }
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", ans.len()));
    for (i, v) in ans.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');

    let mut stdout = io::BufWriter::new(io::stdout());
    stdout.write_all(out.as_bytes()).unwrap();
}
