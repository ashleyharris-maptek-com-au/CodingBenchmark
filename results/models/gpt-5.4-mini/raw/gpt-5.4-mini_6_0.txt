using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private const double MuDefault = 398600.4418;
    private const double EarthRadius = 6371.0;
    private const double TwoPi = Math.PI * 2.0;

    private struct Vec3
    {
        public double X;
        public double Y;
        public double Z;

        public Vec3(double x, double y, double z)
        {
            X = x; Y = y; Z = z;
        }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public Vec3 Cross(Vec3 b) => new Vec3(
            Y * b.Z - Z * b.Y,
            Z * b.X - X * b.Z,
            X * b.Y - Y * b.X);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
    }

    private struct Orbit
    {
        public Vec3 R;
        public Vec3 V;
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n <= 0) return;
        double mu = fs.NextDouble();

        Orbit start = ReadOrbit(fs);
        var stations = new Orbit[n];
        for (int i = 0; i < n; i++) stations[i] = ReadOrbit(fs);

        int[] order = Solve(start, stations, mu);
        var sw = new StreamWriter(Console.OpenStandardOutput()) { AutoFlush = false };
        for (int i = 0; i < order.Length; i++)
        {
            if (i > 0) sw.Write(' ');
            sw.Write(order[i]);
        }
        sw.WriteLine();
        sw.Flush();
    }

    private static Orbit ReadOrbit(FastScanner fs)
    {
        return new Orbit
        {
            R = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble()),
            V = new Vec3(fs.NextDouble(), fs.NextDouble(), fs.NextDouble())
        };
    }

    private static int[] Solve(Orbit start, Orbit[] stations, double mu)
    {
        int n = stations.Length;
        if (n == 1) return new[] { 0 };

        var features = new StationFeature[n];
        for (int i = 0; i < n; i++) features[i] = MakeFeature(stations[i], mu);
        var startFeature = MakeFeature(start, mu);

        if (n <= 10)
        {
            return SolveSmall(startFeature, features, mu);
        }
        else if (n <= 25)
        {
            return ImproveTour(startFeature, features, mu, InitialTour(startFeature, features));
        }
        else
        {
            return LargeScaleTour(startFeature, features, mu);
        }
    }

    private struct StationFeature
    {
        public Vec3 R;
        public Vec3 V;
        public double Radius;
        public double Speed;
        public double Energy;
        public Vec3 H;
        public double Inclination;
        public double Omega;
        public double TrueAnomaly;
    }

    private static StationFeature MakeFeature(Orbit o, double mu)
    {
        double r = o.R.Norm();
        double v = o.V.Norm();
        Vec3 h = o.R.Cross(o.V);
        double hNorm = h.Norm();
        double incl = hNorm > 0 ? Math.Acos(Clamp(h.Z / hNorm, -1, 1)) : 0.0;
        Vec3 nVec = new Vec3(-h.Y, h.X, 0);
        double nNorm = nVec.Norm();
        double omega = 0.0;
        if (nNorm > 1e-12)
        {
            omega = Math.Atan2(nVec.Y, nVec.X);
            if (omega < 0) omega += TwoPi;
        }
        double nu = Math.Atan2(o.R.Dot(o.V) / Math.Sqrt(mu * Math.Max(r, 1e-9)), 1.0 - r * v * v / mu);
        if (double.IsNaN(nu)) nu = 0.0;
        double energy = 0.5 * v * v - mu / r;
        return new StationFeature
        {
            R = o.R,
            V = o.V,
            Radius = r,
            Speed = v,
            Energy = energy,
            H = h,
            Inclination = incl,
            Omega = omega,
            TrueAnomaly = nu
        };
    }

    private static int[] SolveSmall(StationFeature start, StationFeature[] stations, double mu)
    {
        int n = stations.Length;
        int[] perm = Enumerable.Range(0, n).ToArray();
        double best = double.PositiveInfinity;
        int[] bestPerm = (int[])perm.Clone();
        do
        {
            double cost = TourCost(start, stations, perm, mu);
            if (cost < best)
            {
                best = cost;
                Array.Copy(perm, bestPerm, n);
            }
        } while (NextPermutation(perm));
        return bestPerm;
    }

    private static int[] InitialTour(StationFeature start, StationFeature[] stations)
    {
        int n = stations.Length;
        var idx = Enumerable.Range(0, n)
            .OrderBy(i => stations[i].Radius)
            .ThenBy(i => stations[i].Inclination)
            .ThenBy(i => stations[i].Omega)
            .ThenBy(i => stations[i].TrueAnomaly)
            .ToArray();
        return idx;
    }

    private static int[] ImproveTour(StationFeature start, StationFeature[] stations, double mu, int[] order)
    {
        int n = order.Length;
        double best = TourCost(start, stations, order, mu);
        bool improved = true;
        int iter = 0;
        while (improved && iter++ < 200)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    Array.Reverse(order, i, j - i + 1);
                    double cost = TourCost(start, stations, order, mu);
                    if (cost + 1e-9 < best)
                    {
                        best = cost;
                        improved = true;
                    }
                    else
                    {
                        Array.Reverse(order, i, j - i + 1);
                    }
                }
            }
        }
        return order;
    }

    private static int[] LargeScaleTour(StationFeature start, StationFeature[] stations, double mu)
    {
        int n = stations.Length;
        var clusters = stations.Select((s, i) => (s, i))
            .OrderBy(t => t.s.Inclination)
            .ThenBy(t => t.s.Omega)
            .ThenBy(t => t.s.TrueAnomaly)
            .ToArray();

        int[] order = clusters.Select(t => t.i).ToArray();
        order = TwoOpt(start, stations, order, mu, 2);
        order = OrOpt(start, stations, order, mu, 1);
        order = OrOpt(start, stations, order, mu, 2);
        return order;
    }

    private static int[] TwoOpt(StationFeature start, StationFeature[] stations, int[] order, double mu, int passes)
    {
        int n = order.Length;
        double best = TourCost(start, stations, order, mu);
        for (int p = 0; p < passes; p++)
        {
            bool changed = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    Array.Reverse(order, i, j - i + 1);
                    double cost = TourCost(start, stations, order, mu);
                    if (cost + 1e-9 < best)
                    {
                        best = cost;
                        changed = true;
                    }
                    else
                    {
                        Array.Reverse(order, i, j - i + 1);
                    }
                }
            }
            if (!changed) break;
        }
        return order;
    }

    private static int[] OrOpt(StationFeature start, StationFeature[] stations, int[] order, double mu, int len)
    {
        int n = order.Length;
        double best = TourCost(start, stations, order, mu);
        bool improved = true;
        while (improved)
        {
            improved = false;
            for (int i = 0; i + len <= n; i++)
            {
                int[] segment = new int[len];
                Array.Copy(order, i, segment, 0, len);
                int[] remainder = new int[n - len];
                if (i > 0) Array.Copy(order, 0, remainder, 0, i);
                if (i + len < n) Array.Copy(order, i + len, remainder, i, n - i - len);
                for (int j = 0; j <= remainder.Length; j++)
                {
                    int[] candidate = new int[n];
                    if (j > 0) Array.Copy(remainder, 0, candidate, 0, j);
                    Array.Copy(segment, 0, candidate, j, len);
                    if (j < remainder.Length) Array.Copy(remainder, j, candidate, j + len, remainder.Length - j);
                    double cost = TourCost(start, stations, candidate, mu);
                    if (cost + 1e-9 < best)
                    {
                        Array.Copy(candidate, order, n);
                        best = cost;
                        improved = true;
                    }
                }
                if (improved) break;
            }
        }
        return order;
    }

    private static double TourCost(StationFeature start, StationFeature[] stations, int[] order, double mu)
    {
        double sum = 0.0;
        StationFeature cur = start;
        for (int i = 0; i < order.Length; i++)
        {
            StationFeature nxt = stations[order[i]];
            sum += TransferCost(cur, nxt, mu);
            cur = nxt;
        }
        return sum;
    }

    private static double TransferCost(StationFeature a, StationFeature b, double mu)
    {
        double dr = (b.R - a.R).Norm();
        double dv = (b.V - a.V).Norm();
        double ra = Math.Max(a.Radius, EarthRadius + 1.0);
        double rb = Math.Max(b.Radius, EarthRadius + 1.0);
        double da = Math.Abs(a.Inclination - b.Inclination);
        double doo = AngleDiff(a.Omega, b.Omega);
        double dnu = AngleDiff(a.TrueAnomaly, b.TrueAnomaly);
        double phase = dr / (ra + rb) + 0.15 * dv + 5.0 * da + 0.5 * doo + 0.2 * dnu;
        double grav = Math.Abs(a.Energy - b.Energy) / 1000.0;
        return phase + grav;
    }

    private static double AngleDiff(double a, double b)
    {
        double d = Math.Abs(a - b) % TwoPi;
        return d > Math.PI ? TwoPi - d : d;
    }

    private static double Clamp(double x, double lo, double hi) => x < lo ? lo : (x > hi ? hi : x);

    private static bool NextPermutation(int[] a)
    {
        int n = a.Length;
        int i = n - 2;
        while (i >= 0 && a[i] >= a[i + 1]) i--;
        if (i < 0) return false;
        int j = n - 1;
        while (a[j] <= a[i]) j--;
        (a[i], a[j]) = (a[j], a[i]);
        Array.Reverse(a, i + 1, n - i - 1);
        return true;
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buf[_ptr++];
        }

        public int NextInt() => (int)NextDouble();

        public double NextDouble()
        {
            byte c;
            do c = Read(); while (c <= 32 && c != 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            double val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            if (c == '.')
            {
                double frac = 1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    frac *= 0.1;
                    val += (c - '0') * frac;
                    c = Read();
                }
            }
            if (c == 'e' || c == 'E')
            {
                int esign = 1;
                int exp = 0;
                c = Read();
                if (c == '-')
                {
                    esign = -1;
                    c = Read();
                }
                else if (c == '+')
                {
                    c = Read();
                }
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                val *= Math.Pow(10, esign * exp);
            }
            return sign * val;
        }
    }
}
