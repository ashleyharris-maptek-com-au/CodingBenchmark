use std::collections::VecDeque;
use std::io::{self, Read};

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let h: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return,
    };
    let w: usize = it.next().unwrap().parse().unwrap();
    let n = h * w;

    let mut grid = Vec::with_capacity(n);
    let mut start = None;
    let mut goal = None;

    for y in 0..h {
        let row = it.next().unwrap_or("");
        let bytes = row.as_bytes();
        for x in 0..w {
            let c = *bytes.get(x).unwrap_or(&b'#');
            grid.push(c);
            if c == b'A' {
                start = Some(y * w + x);
            } else if c == b'B' {
                goal = Some(y * w + x);
            }
        }
    }

    let start = match start {
        Some(s) => s,
        None => {
            println!("0");
            return;
        }
    };
    let goal = match goal {
        Some(g) => g,
        None => {
            println!("0");
            return;
        }
    };

    if start == goal {
        println!("1\n0 0");
        return;
    }

    let mut visited = vec![false; n];
    let mut parent = vec![usize::MAX; n];
    let mut q = VecDeque::new();

    visited[start] = true;
    q.push_back(start);

    while let Some(u) = q.pop_front() {
        if u == goal {
            break;
        }
        let x = u % w;
        let y = u / w;

        if x > 0 {
            let v = u - 1;
            if !visited[v] && grid[v] != b'#' {
                visited[v] = true;
                parent[v] = u;
                q.push_back(v);
            }
        }
        if x + 1 < w {
            let v = u + 1;
            if !visited[v] && grid[v] != b'#' {
                visited[v] = true;
                parent[v] = u;
                q.push_back(v);
            }
        }
        if y > 0 {
            let v = u - w;
            if !visited[v] && grid[v] != b'#' {
                visited[v] = true;
                parent[v] = u;
                q.push_back(v);
            }
        }
        if y + 1 < h {
            let v = u + w;
            if !visited[v] && grid[v] != b'#' {
                visited[v] = true;
                parent[v] = u;
                q.push_back(v);
            }
        }
    }

    if !visited[goal] {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut cur = goal;
    while cur != usize::MAX {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = parent[cur];
    }
    path.reverse();

    let mut out = String::new();
    out.push_str(&format!("{}\n", path.len()));
    for idx in path {
        out.push_str(&format!("{} {}\n", idx % w, idx / w));
    }
    print!("{}", out);
}
