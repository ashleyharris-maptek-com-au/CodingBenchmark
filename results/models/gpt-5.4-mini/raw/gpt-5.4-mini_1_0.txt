```cpp
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using namespace std;

struct Point {
    double x, y;
};

static inline double dist2(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static vector<int> nearest_neighbor_small(const vector<Point>& p) {
    int n = (int)p.size();
    vector<int> route;
    route.reserve(n);
    vector<char> used(n, 0);
    route.push_back(0);
    used[0] = 1;
    int cur = 0;
    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestd = numeric_limits<double>::infinity();
        for (int i = 0; i < n; ++i) {
            if (!used[i]) {
                double d = dist2(p[cur], p[i]);
                if (d < bestd) {
                    bestd = d;
                    best = i;
                }
            }
        }
        used[best] = 1;
        route.push_back(best);
        cur = best;
    }
    return route;
}

static void two_opt(vector<int>& route, const vector<Point>& p, int max_passes) {
    int n = (int)route.size();
    if (n < 4) return;
    auto d = [&](int a, int b) { return dist2(p[a], p[b]); };
    for (int pass = 0; pass < max_passes; ++pass) {
        bool improved = false;
        for (int i = 1; i + 2 < n; ++i) {
            for (int k = i + 1; k + 1 < n; ++k) {
                int a = route[i - 1], b = route[i];
                int c = route[k], d2i = route[k + 1];
                double before = d(a, b) + d(c, d2i);
                double after = d(a, c) + d(b, d2i);
                if (after + 1e-12 < before) {
                    reverse(route.begin() + i, route.begin() + k + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

static vector<int> farthest_insertion(const vector<Point>& p) {
    int n = (int)p.size();
    vector<int> route;
    route.reserve(n);
    vector<char> used(n, 0);
    route.push_back(0);
    used[0] = 1;
    if (n == 1) return route;
    int far = 1;
    double bestd = dist2(p[0], p[1]);
    for (int i = 2; i < n; ++i) {
        double d = dist2(p[0], p[i]);
        if (d > bestd) bestd = d, far = i;
    }
    route.push_back(far);
    used[far] = 1;
    while ((int)route.size() < n) {
        int pick = -1;
        double pickScore = -1.0;
        for (int i = 0; i < n; ++i) if (!used[i]) {
            double mind = numeric_limits<double>::infinity();
            for (int v : route) mind = min(mind, dist2(p[i], p[v]));
            if (mind > pickScore) pickScore = mind, pick = i;
        }
        int bestPos = 0;
        double bestInc = numeric_limits<double>::infinity();
        int m = (int)route.size();
        for (int i = 0; i < m; ++i) {
            int a = route[i];
            int b = route[(i + 1) % m];
            double inc = dist2(p[a], p[pick]) + dist2(p[pick], p[b]) - dist2(p[a], p[b]);
            if (inc < bestInc) bestInc = inc, bestPos = i + 1;
        }
        route.insert(route.begin() + bestPos, pick);
        used[pick] = 1;
    }
    return route;
}

struct Grid {
    double minx, miny, invCell;
    int cells;
    vector<vector<int>> buckets;
    Grid(const vector<Point>& pts, int bucketTarget = 4096) {
        minx = miny = numeric_limits<double>::infinity();
        double maxx = -numeric_limits<double>::infinity();
        double maxy = -numeric_limits<double>::infinity();
        for (auto& pt : pts) {
            minx = min(minx, pt.x); miny = min(miny, pt.y);
            maxx = max(maxx, pt.x); maxy = max(maxy, pt.y);
        }
        double span = max(maxx - minx, maxy - miny);
        if (span <= 0) span = 1.0;
        cells = max(1, (int)sqrt((double)bucketTarget));
        invCell = cells / span;
        buckets.assign(cells * cells, {});
        for (int i = 0; i < (int)pts.size(); ++i) {
            int cx = min(cells - 1, max(0, (int)((pts[i].x - minx) * invCell)));
            int cy = min(cells - 1, max(0, (int)((pts[i].y - miny) * invCell)));
            buckets[cy * cells + cx].push_back(i);
        }
    }
    vector<int> nearestCandidates(const Point& q, int radius = 1) const {
        int cx = min(cells - 1, max(0, (int)((q.x - minx) * invCell)));
        int cy = min(cells - 1, max(0, (int)((q.y - miny) * invCell)));
        vector<int> out;
        for (int dy = -radius; dy <= radius; ++dy) {
            for (int dx = -radius; dx <= radius; ++dx) {
                int nx = cx + dx, ny = cy + dy;
                if (nx < 0 || ny < 0 || nx >= cells || ny >= cells) continue;
                const auto& b = buckets[ny * cells + nx];
                out.insert(out.end(), b.begin(), b.end());
            }
        }
        return out;
    }
};

static vector<int> grid_greedy(const vector<Point>& p) {
    int n = (int)p.size();
    Grid g(p, min(8192, max(64, n / 8 + 1)));
    vector<int> route;
    route.reserve(n);
    vector<char> used(n, 0);
    route.push_back(0);
    used[0] = 1;
    int cur = 0;
    for (int step = 1; step < n; ++step) {
        int best = -1;
        double bestd = numeric_limits<double>::infinity();
        auto cand = g.nearestCandidates(p[cur], 1);
        if (cand.empty()) cand = g.nearestCandidates(p[cur], 2);
        if (cand.empty()) {
            for (int i = 0; i < n; ++i) if (!used[i]) cand.push_back(i);
        }
        for (int i : cand) if (!used[i]) {
            double d = dist2(p[cur], p[i]);
            if (d < bestd) bestd = d, best = i;
        }
        if (best == -1) {
            for (int i = 0; i < n; ++i) if (!used[i]) {
                double d = dist2(p[cur], p[i]);
                if (d < bestd) bestd = d, best = i;
            }
        }
        used[best] = 1;
        route.push_back(best);
        cur = best;
    }
    return route;
}

static vector<int> multi_start_greedy(const vector<Point>& p) {
    int n = (int)p.size();
    vector<int> bestRoute;
    double bestLen = numeric_limits<double>::infinity();
    vector<int> starts = {0};
    if (n > 2) starts.push_back(n / 2);
    if (n > 3) starts.push_back(n - 1);
    for (int s : starts) {
        vector<int> route;
        route.reserve(n);
        vector<char> used(n, 0);
        route.push_back(0);
        used[0] = 1;
        if (s != 0 && !used[s]) {
            route.push_back(s);
            used[s] = 1;
        }
        while ((int)route.size() < n) {
            int cur = route.back();
            int best = -1;
            double bestd = numeric_limits<double>::infinity();
            for (int i = 0; i < n; ++i) if (!used[i]) {
                double d = dist2(p[cur], p[i]);
                if (d < bestd) bestd = d, best = i;
            }
            used[best] = 1;
            route.push_back(best);
        }
        double len = 0;
        for (int i = 0; i < n; ++i) len += sqrt(dist2(p[route[i]], p[route[(i + 1) % n]]));
        if (len < bestLen) bestLen = len, bestRoute = move(route);
    }
    return bestRoute;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    int N;
    if (!(cin >> N)) return 0;
    vector<Point> pts(N);
    for (int i = 0; i < N; ++i) cin >> pts[i].x >> pts[i].y;
    vector<int> route;
    if (N <= 40) {
        route = nearest_neighbor_small(pts);
        if (N >= 4) two_opt(route, pts, 3);
    } else if (N <= 5000) {
        route = farthest_insertion(pts);
        two_opt(route, pts, 1);
    } else if (N <= 200000) {
        route = grid_greedy(pts);
    } else {
        route = multi_start_greedy(pts);
    }
    if (route.empty()) {
        route.resize(N);
        iota(route.begin(), route.end(), 0);
    } else if (route[0] != 0) {
        auto it = find(route.begin(), route.end(), 0);
        if (it != route.end()) rotate(route.begin(), it, route.end());
    }
    for (int i = 0; i < N; ++i) {
        if (i) cout << ' ';
        cout << route[i];
    }
    cout << '\n';
    return 0;
}
```
