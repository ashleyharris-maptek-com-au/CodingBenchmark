struct FastOutput {

  ~FastOutput() { flush(); }

    if (pos) {
      std::fwrite(buf, 1, pos, stdout);
      pos = 0;
    }
  }

  inline void writeChar(char c) {
    if (pos >= kBufSize) flush();
    buf[pos++] = c;
  }

  inline void writeUInt32(uint32_t v) {
    char tmp[16];

    pos += n;
  }
};

static inline uint64_t splitBy1(uint32_t a) {
  uint64_t x = a;
  x = (x | (x << 16)) & 0x0000FFFF0000FFFFULL;
  x = (x | (x << 8)) & 0x00FF00FF00FF00FFULL;
  x = (x | (x << 4)) & 0x0F0F0F0F0F0F0F0FULL;
  x = (x | (x << 2)) & 0x3333333333333333ULL;
  x = (x | (x << 1)) & 0x5555555555555555ULL;
  return x;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
  return splitBy1(x) | (splitBy1(y) << 1);
}

static inline uint32_t quantize01(double v, double minV, double invRange) {
  if (invRange == 0.0) return 0u;
  double t = (v - minV) * invRange;
  if (t < 0.0) t = 0.0;
  if (t > 1.0) t = 1.0;
  double u = t * 4294967295.0 + 0.5;
  if (u < 0.0) u = 0.0;
  if (u > 4294967295.0) u = 4294967295.0;
  return static_cast<uint32_t>(static_cast<uint64_t>(u));
}

static inline double dist2(uint32_t a, uint32_t b, const std::vector<float>& x,
                           const std::vector<float>& y) {
  double dx = static_cast<double>(x[a]) - static_cast<double>(x[b]);
  double dy = static_cast<double>(y[a]) - static_cast<double>(y[b]);
  return dx * dx + dy * dy;
}

static inline double dist(uint32_t a, uint32_t b, const std::vector<float>& x,
                          const std::vector<float>& y) {
  return std::sqrt(dist2(a, b, x, y));
}

static void radixSort64(std::vector<KeyIndex>& a) {
  const size_t n = a.size();
  std::vector<KeyIndex> tmp(n);
  static constexpr int kBits = 16;
  static constexpr size_t kBuckets = 1u << kBits;
  std::vector<size_t> count(kBuckets);

  for (int pass = 0; pass < 4; ++pass) {
    std::fill(count.begin(), count.end(), 0);
    const int shift = pass * kBits;
    for (size_t i = 0; i < n; ++i) {
      ++count[static_cast<size_t>((a[i].key >> shift) & 0xFFFFULL)];
    }
    size_t sum = 0;
    for (size_t b = 0; b < kBuckets; ++b) {
      const size_t c = count[b];
      count[b] = sum;
      sum += c;
    }
    for (size_t i = 0; i < n; ++i) {
      const size_t b = static_cast<size_t>((a[i].key >> shift) & 0xFFFFULL);
      tmp[count[b]++] = a[i];
    }
    a.swap(tmp);
  }
}

static std::vector<uint32_t> solveHeldKarp(const std::vector<float>& x,
                                           const std::vector<float>& y,
                                           uint32_t n) {
  const uint32_t N = n;
  if (N == 1) return {0u};
  const uint32_t M = N - 1; // vertices 1..N-1
  const uint32_t masks = 1u << M;
  const double INF = std::numeric_limits<double>::infinity();

  std::vector<double> dp(static_cast<size_t>(masks) * M, INF);
  std::vector<int16_t> parent(static_cast<size_t>(masks) * M, -1);

  auto at = [&](uint32_t mask, uint32_t i) -> size_t {
    return static_cast<size_t>(mask) * M + i;
  };

  for (uint32_t i = 0; i < M; ++i) {
    const uint32_t mask = 1u << i;
    dp[at(mask, i)] = dist(0u, i + 1u, x, y);
    parent[at(mask, i)] = -1;
  }

  for (uint32_t mask = 1; mask < masks; ++mask) {
    for (uint32_t i = 0; i < M; ++i) {
      if ((mask & (1u << i)) == 0) continue;
      const uint32_t pm = mask ^ (1u << i);
      if (pm == 0) continue;

      double best = INF;
      int16_t bestPrev = -1;
      for (uint32_t j = 0; j < M; ++j) {
        if ((pm & (1u << j)) == 0) continue;
        const double cand = dp[at(pm, j)] + dist(j + 1u, i + 1u, x, y);
        if (cand < best) {
          best = cand;
          bestPrev = static_cast<int16_t>(j);
        }
      }
      dp[at(mask, i)] = best;
      parent[at(mask, i)] = bestPrev;
    }
  }

  const uint32_t full = masks - 1u;
  double bestTour = INF;
  int16_t last = -1;
  for (uint32_t i = 0; i < M; ++i) {
    const double cand = dp[at(full, i)] + dist(i + 1u, 0u, x, y);
    if (cand < bestTour) {
      bestTour = cand;
      last = static_cast<int16_t>(i);
    }
  }

  std::vector<uint32_t> rev;
  rev.reserve(M);
  uint32_t mask = full;
  int16_t cur = last;
  while (cur >= 0) {
    rev.push_back(static_cast<uint32_t>(cur) + 1u);
    const int16_t p = parent[at(mask, static_cast<uint32_t>(cur))];
    mask ^= (1u << static_cast<uint32_t>(cur));
    cur = p;
  }

  std::vector<uint32_t> tour;
  tour.reserve(N);
  tour.push_back(0u);
  for (size_t i = 0; i < rev.size(); ++i) {
    tour.push_back(rev[rev.size() - 1 - i]);
  }
  return tour;
}

static std::vector<uint32_t> solveNearestNeighbor(const std::vector<float>& x,
                                                  const std::vector<float>& y,
                                                  uint32_t n) {
  const uint32_t N = n;
  std::vector<uint32_t> tour(N);
  std::vector<uint8_t> used(N, 0);
  tour[0] = 0u;
  used[0] = 1;
  uint32_t cur = 0u;

  for (uint32_t t = 1; t < N; ++t) {
    uint32_t best = 0u;
    double bestD = std::numeric_limits<double>::infinity();
    for (uint32_t j = 1; j < N; ++j) {
      if (used[j]) continue;
      const double d = dist2(cur, j, x, y);
      if (d < bestD) {
        bestD = d;
        best = j;
      }
    }
    tour[t] = best;
    used[best] = 1;
    cur = best;
  }
  return tour;
}

static void twoOptWindow(std::vector<uint32_t>& tour, const std::vector<float>& x,
                         const std::vector<float>& y, int window,
                         int passes) {
  const size_t n = tour.size();
  if (n < 4) return;
  if (window < 4) return;

  for (int pass = 0; pass < passes; ++pass) {
    bool improved = false;
    for (size_t i = 0; i + 3 < n; ++i) {
      const uint32_t a = tour[i];
      const uint32_t b = tour[i + 1];
      const size_t jMax = std::min(n - 2, i + static_cast<size_t>(window));
      for (size_t j = i + 2; j <= jMax; ++j) {
        const uint32_t c = tour[j];
        const uint32_t d = tour[j + 1];
        const double before = dist2(a, b, x, y) + dist2(c, d, x, y);
        const double after = dist2(a, c, x, y) + dist2(b, d, x, y);
        if (after + 1e-12 < before) {
          std::reverse(tour.begin() + (i + 1), tour.begin() + (j + 1));
          improved = true;
        }
      }
    }
    if (!improved) break;
  }
}

static std::vector<uint32_t> solveMorton(const std::vector<float>& x,

  const uint32_t N = n;
  std::vector<KeyIndex> keys;
  keys.resize(N);

  const double rx = maxX - minX;
  const double ry = maxY - minY;
  const double invRx = (rx != 0.0) ? (1.0 / rx) : 0.0;
  const double invRy = (ry != 0.0) ? (1.0 / ry) : 0.0;

  for (uint32_t i = 0; i < N; ++i) {
    const uint32_t qx = quantize01(static_cast<double>(x[i]), minX, invRx);
    const uint32_t qy = quantize01(static_cast<double>(y[i]), minY, invRy);
    keys[i] = KeyIndex{morton2D(qx, qy), i};
  }

  if (N >= 200000u) {
    radixSort64(keys);
  } else {
    std::sort(keys.begin(), keys.end(),
              [](const KeyIndex& a, const KeyIndex& b) {
                if (a.key != b.key) return a.key < b.key;
                return a.idx < b.idx;
              });
  }

  size_t pos0 = 0;
  for (size_t i = 0; i < keys.size(); ++i) {

  }
  return tour;
}

static Face make_face(int a, int b, int c, const std::vector<Vec3>& pts, const Vec3& inside_point, double eps) {

  f.b = b;
  f.c = c;

static double polyhedron_volume(const std::vector<Face>& faces, const std::vector<Vec3>& pts, const Vec3& ref) {
  long double acc = 0.0L;
  for (const Face& f : faces) {
    const Vec3 a = pts[f.a] - ref;
    const Vec3 b = pts[f.b] - ref;
    const Vec3 c = pts[f.c] - ref;
    const Vec3 cr = cross(b, c);
    acc += static_cast<long double>(dot(a, cr));
  }
  long double v = std::fabs(acc) / 6.0L;
  return static_cast<double>(v);
}

}

    let len = neighbors(id, w, h, blocked, &mut nb);
    for i in 0..len {
        let v = nb[i];
        if visited[v] == 0 {
            cnt += 1;
        }
    }
    cnt
}

fn reachable_count_from_start(start: usize, w: usize, h: usize, blocked: &[u8]) -> usize {
    let total = blocked.len();
    let mut seen = vec![0u8; total];
    let mut q = Vec::with_capacity(total.min(4096));
    let mut head = 0usize;
    seen[start] = 1;
    q.push(start);
    let mut count = 1usize;
    while head < q.len() {
        let u = q[head];
        head += 1;
        let mut nb = [0usize; 4];
        let len = neighbors(u, w, h, blocked, &mut nb);
        for i in 0..len {
            let v = nb[i];
            if seen[v] == 0 {
                seen[v] = 1;
                q.push(v);
                count += 1;
            }
        }
    }
    count
}

fn remaining_connected(
    current: usize,
    w: usize,
    h: usize,
    blocked: &[u8],
    visited: &[u8],
    free_total: usize,
    path_len: usize,
    seen: &mut [u32],
    stamp: &mut u32,
    queue: &mut Vec<usize>,
    queue_head: &mut usize,
) -> bool {
    let target = free_total.saturating_sub(path_len).saturating_add(1);
    if target <= 1 {
        return true;
    }
    *stamp = stamp.wrapping_add(1);
    let s = *stamp;
    queue.clear();
    *queue_head = 0;
    seen[current] = s;
    queue.push(current);
    let mut count = 1usize;
    while *queue_head < queue.len() {
        let u = queue[*queue_head];
        *queue_head += 1;
        let mut nb = [0usize; 4];
        let len = neighbors(u, w, h, blocked, &mut nb);
        for i in 0..len {
            let v = nb[i];
            if v != current && visited[v] != 0 {
                continue;
            }
            if seen[v] == s {
                continue;
            }
            seen[v] = s;
            queue.push(v);
            count += 1;
        }
    }
    count == target
}

fn build_snake_path(w: usize, h: usize) -> Vec<usize> {
    let mut path = Vec::with_capacity(w.saturating_mul(h));
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                path.push(id_of(x, y, w));
            }
        } else {
            for x in (0..w).rev() {
                path.push(id_of(x, y, w));
            }
        }
    }
    path
}

fn build_cycle_even_height(w: usize, h: usize) -> Vec<(usize, usize)> {
    // Preconditions: w>=2, h>=2, h even, no obstacles.
    // Constructs a Hamiltonian cycle as a Hamiltonian path that ends adjacent to (0,0).
    //
    // Pattern:
    // - Traverse top row fully: (0,0) -> (w-1,0)
    // - For rows 1..h-1: snake across columns 1..w-1, leaving column 0 for the end
    // - Finally: step to (0,h-1) and go up column 0 to (0,1). End at (0,1).
    let mut path: Vec<(usize, usize)> = Vec::with_capacity(w * h);
    for x in 0..w {
        path.push((x, 0));
    }
    let mut cur_x = w - 1;
    for y in 1..h {
        // move down at fixed x
        path.push((cur_x, y));
        if y % 2 == 1 {
            // go left to x=1
            for x in (1..cur_x).rev() {
                path.push((x, y));
            }
            cur_x = 1;
        } else {
            // go right to x=w-1
            for x in (cur_x + 1)..w {
                path.push((x, y));
            }
            cur_x = w - 1;
        }
    }
    // At h even, last row is odd index => cur_x==1 here.
    path.push((0, h - 1));
    for y in (1..(h - 1)).rev() {
        path.push((0, y));
    }
    path
}

fn build_cycle_rect_no_obstacles(w: usize, h: usize) -> Option<Vec<usize>> {
    if w < 2 || h < 2 {
        return None;
    }
    if w % 2 == 1 && h % 2 == 1 {
        return None;
    }
    // Use even-height construction; if height is odd but width even, transpose, then swap back.
    if h % 2 == 0 {
        let coords = build_cycle_even_height(w, h);
        let mut path = Vec::with_capacity(w * h);
        for (x, y) in coords {
            path.push(id_of(x, y, w));
        }
        return Some(path);
    }
    // transpose: build on (h,w) where height=w is even.
    let coords_t = build_cycle_even_height(h, w); // width'=h, height'=w
    let mut path = Vec::with_capacity(w * h);
    for (x_t, y_t) in coords_t {
        // swap back: original (x,y) = (y_t, x_t)
        let x = y_t;
        let y = x_t;
        path.push(id_of(x, y, w));
    }
    Some(path)
}

fn validate_and_print(
    w: usize,
    _h: usize,
    blocked: &[u8],
    require_cycle: bool,
    free_count: usize,
    path: &[usize],
) -> Option<String> {
    if path.len() != free_count {
        return None;
    }
    let start = 0usize;
    if path.is_empty() || path[0] != start {
        return None;
    }
    if blocked[start] != 0 {
        return None;
    }
    let total = blocked.len();
    let mut seen = vec![0u8; total];
    seen[start] = 1;
    for i in 0..path.len() {
        let u = path[i];
        if u >= total || blocked[u] != 0 {
            return None;
        }
        if i > 0 {
            if seen[u] != 0 {
                return None;
            }
            let prev = path[i - 1];
            if !is_adjacent(prev, u, w) {
                return None;
            }
            seen[u] = 1;
        }
    }
    // ensure all free visited
    let mut visited_free = 0usize;
    for i in 0..total {
        if blocked[i] == 0 && seen[i] != 0 {
            visited_free += 1;
        }
    }
    if visited_free != free_count {
        return None;
    }
    if require_cycle {
        if free_count <= 1 {
            return None;
        }
        if !is_adjacent(*path.last().unwrap(), start, w) {
            return None;
        }
    }

    let mut out = String::with_capacity(free_count * 8);
    writeln!(&mut out, "{}", free_count).ok()?;
    for &u in path {
        let (x, y) = xy_of(u, w);
        writeln!(&mut out, "{} {}", x, y).ok()?;
    }
    Some(out)
}

fn precheck_impossible(
    w: usize,
    h: usize,
    blocked: &[u8],
    require_cycle: bool,
    free_count: usize,
) -> bool {
    let start = 0usize;
    if blocked[start] != 0 {
        return true;
    }
    if free_count == 0 {
        return true;
    }

    // Connectivity (necessary)
    if free_count > 1 {
        let reach = reachable_count_from_start(start, w, h, blocked);
        if reach != free_count {
            return true;
        }
    }

    // Degree-based quick rejects and bipartite parity constraints
    let mut deg1 = 0usize;
    let start_deg = degree(start, w, h, blocked) as usize;

    let mut black = 0usize;
    let mut white = 0usize;
    for y in 0..h {
        for x in 0..w {
            let id = id_of(x, y, w);
            if blocked[id] != 0 {
                continue;
            }
            let d = degree(id, w, h, blocked) as usize;
            if d == 0 && free_count > 1 {
                return true;
            }
            if d == 1 {
                deg1 += 1;
            }
            if (x + y) % 2 == 0 {
                black += 1;
            } else {
                white += 1;
            }
        }
    }

    if require_cycle {
        if free_count <= 1 {
            return true;
        }
        // Bipartite parity: cycle alternates colors => equal partition sizes.
        if black != white {
            return true;
        }
        // Every vertex in a Hamiltonian cycle must have degree >= 2.
        for y in 0..h {
            for x in 0..w {
                let id = id_of(x, y, w);
                if blocked[id] != 0 {
                    continue;
                }
                if degree(id, w, h, blocked) < 2 {
                    return true;
                }
            }
        }
        return false;
    }

    // Path (not cycle): too many graph degree-1 vertices makes it impossible.
    if deg1 > 2 {
        return true;
    }
    if free_count > 1 && start_deg == 0 {
        return true;
    }

    // If there are exactly two degree-1 vertices, start must be one of them (since start is fixed).
    if deg1 == 2 && start_deg != 1 {
        return true;
    }

    // Parity check when endpoints are forced by degree-1 vertices.
    // In bipartite graphs:
    // - If N is even, endpoints have opposite colors.
    // - If N is odd, endpoints have the same color.
    if deg1 >= 1 {
        let start_color_black = true; // (0+0)%2==0
        let end_must_be_black = if free_count % 2 == 0 {
            !start_color_black
        } else {
            start_color_black
        };

        if deg1 == 2 {
            // End is the other degree-1 vertex.
            for y in 0..h {
                for x in 0..w {
                    let id = id_of(x, y, w);
                    if blocked[id] != 0 {
                        continue;
                    }
                    if id != start && degree(id, w, h, blocked) == 1 {
                        let is_black = (x + y) % 2 == 0;
                        if is_black != end_must_be_black {
                            return true;
                        }
                        break;
                    }
                }
            }
        } else if deg1 == 1 && start_deg != 1 {
            // The unique degree-1 vertex must be the end.
            for y in 0..h {
                for x in 0..w {
                    let id = id_of(x, y, w);
                    if blocked[id] != 0 {
                        continue;
                    }
                    if degree(id, w, h, blocked) == 1 {
                        let is_black = (x + y) % 2 == 0;
                        if is_black != end_must_be_black {
                            return true;
                        }
                        break;
                    }
                }
            }
        }
    }

    false
}

fn exact_solve(
    w: usize,
    h: usize,
    blocked: &[u8],
    require_cycle: bool,
    free_count: usize,
    deadline: Instant,
) -> Option<Vec<usize>> {
    let total = blocked.len();
    let start = 0usize;
    let mut visited = vec![0u8; total];
    visited[start] = 1;
    let mut path = Vec::with_capacity(free_count);
    path.push(start);

    let mut seen_stamp = vec![0u32; total];
    let mut stamp: u32 = 1;
    let mut queue: Vec<usize> = Vec::with_capacity(total.min(4096));
    let mut queue_head: usize = 0;

    fn dfs(
        current: usize,
        w: usize,
        h: usize,
        blocked: &[u8],
        require_cycle: bool,
        free_count: usize,
        visited: &mut [u8],
        path: &mut Vec<usize>,
        seen_stamp: &mut [u32],
        stamp: &mut u32,
        queue: &mut Vec<usize>,
        queue_head: &mut usize,
        deadline: Instant,
    ) -> bool {
        if Instant::now() >= deadline {
            return false;
        }
        if path.len() == free_count {
            if !require_cycle || is_adjacent(current, 0usize, w) {
                return true;
            }
            return false;
        }

        // Connectivity pruning (necessary).
        if !remaining_connected(
            current,
            w,
            h,
            blocked,
            visited,
            free_count,
            path.len(),
            seen_stamp,
            stamp,
            queue,
            queue_head,
        ) {
            return false;
        }

        let mut nb = [0usize; 4];
        let mut cands = [0usize; 4];
        let mut clen = 0usize;
        let nlen = neighbors(current, w, h, blocked, &mut nb);
        for i in 0..nlen {
            let v = nb[i];
            if visited[v] == 0 {
                // If this would be the last step of a cycle, enforce adjacency to start.
                if require_cycle && path.len() + 1 == free_count && !is_adjacent(v, 0usize, w) {
                    continue;
                }
                cands[clen] = v;
                clen += 1;
            }
        }
        if clen == 0 {
            return false;
        }

        // Warnsdorff ordering: smallest onward unvisited degree first.
        let mut scores = [0u8; 4];
        for i in 0..clen {
            let v = cands[i];
            // Treat current as visited already; we count only unvisited neighbors.
            scores[i] = unvisited_degree(v, w, h, blocked, visited);
        }
        for i in 0..clen {
            for j in (i + 1)..clen {
                if scores[j] < scores[i] || (scores[j] == scores[i] && cands[j] < cands[i]) {
                    scores.swap(i, j);
                    cands.swap(i, j);
                }
            }
        }

        for i in 0..clen {
            let v = cands[i];
            visited[v] = 1;
            path.push(v);
            if dfs(
                v, w, h, blocked, require_cycle, free_count, visited, path, seen_stamp, stamp,
                queue, queue_head, deadline,
            ) {
                return true;
            }
            path.pop();
            visited[v] = 0;
        }
        false
    }

    if dfs(
        start,
        w,
        h,
        blocked,
        require_cycle,
        free_count,
        &mut visited,
        &mut path,
        &mut seen_stamp,
        &mut stamp,
        &mut queue,
        &mut queue_head,
        deadline,
    ) {
        return Some(path);
    }
    None
}

#[derive(Clone, Copy)]
struct Frame {
    candidates: [usize; 4],
    len: u8,
    next: u8,
}

fn heuristic_solve(
    w: usize,
    h: usize,
    blocked: &[u8],
    require_cycle: bool,
    free_count: usize,
    deadline: Instant,
) -> Option<Vec<usize>> {
    let total = blocked.len();
    let start = 0usize;

    let mut base_seed: u64 = (w as u64).wrapping_mul(0x9E37_79B9_7F4A_7C15)
        ^ (h as u64).wrapping_mul(0xD1B5_4A32_D192_ED03)
        ^ (free_count as u64).wrapping_mul(0x94D0_49BB_1331_11EB);
    // Mix in obstacle positions (cheaply).
    for i in (0..total).step_by((total / 32).max(1)) {
        base_seed ^= (blocked[i] as u64).wrapping_mul((i as u64).wrapping_add(0xA5A5_A5A5));
    }

    let mut attempt: u32 = 0;
    while Instant::now() < deadline {
        attempt = attempt.wrapping_add(1);
        let mut rng = Rng64::new(base_seed ^ (attempt as u64).wrapping_mul(0xBF58_476D_1CE4_E5B9));
        let mut tie = vec![0u32; total];
        for t in tie.iter_mut() {
            *t = rng.next_u32();
        }

        let mut visited = vec![0u8; total];
        visited[start] = 1;
        let mut path: Vec<usize> = Vec::with_capacity(free_count);
        path.push(start);
        let mut frames: Vec<Frame> = Vec::with_capacity(free_count);

        let build_frame = |current: usize, path_len: usize, visited: &Vec<u8>| -> Frame {
            let mut nb = [0usize; 4];
            let mut cands = [0usize; 4];
            let mut scores = [0u8; 4];
            let mut keys = [0u32; 4];
            let mut clen = 0usize;
            let nlen = neighbors(current, w, h, blocked, &mut nb);
            for i in 0..nlen {
                let v = nb[i];
                if visited[v] != 0 {
                    continue;
                }
                if require_cycle && path_len + 1 == free_count && !is_adjacent(v, start, w) {
                    continue;
                }
                cands[clen] = v;
                scores[clen] = unvisited_degree(v, w, h, blocked, visited);
                keys[clen] = tie[v];
                clen += 1;
            }
            for i in 0..clen {
                for j in (i + 1)..clen {
                    if scores[j] < scores[i]
                        || (scores[j] == scores[i] && keys[j] < keys[i])
                    {
                        scores.swap(i, j);
                        keys.swap(i, j);
                        cands.swap(i, j);
                    }
                }
            }
            Frame {
                candidates: cands,
                len: clen as u8,
                next: 0,
            }
        };

        frames.push(build_frame(start, 1, &visited));

        let mut backtracks: u64 = 0;
        const BACKTRACK_LIMIT: u64 = 20_000_000;

        while Instant::now() < deadline {
            if path.len() == free_count {
                if !require_cycle || is_adjacent(*path.last().unwrap(), start, w) {
                    return Some(path);
                }
                // Not a valid cycle end; force backtrack.
            }

            if frames.is_empty() {
                break;
            }
            let depth = path.len() - 1;
            if frames.len() != path.len() {
                // Should not happen; resync defensively.
                frames.truncate(path.len());
            }

            let mut frame = frames[depth];
            if frame.next >= frame.len {
                // backtrack
                if depth == 0 {
                    break;
                }
                let u = path.pop().unwrap();
                visited[u] = 0;
                frames.pop();
                backtracks += 1;
                if backtracks >= BACKTRACK_LIMIT {
                    break;
                }
                continue;
            }

            let v = frame.candidates[frame.next as usize];
            frame.next += 1;
            frames[depth] = frame;
            if visited[v] != 0 {
                continue;
            }
            visited[v] = 1;
            path.push(v);
            frames.push(build_frame(v, path.len(), &visited));
        }
    }
    None
}

fn main() {
    let mut p = Parser::new();
    let w = match p.next_usize() {
        Some(v) => v,
        None => return,
    };
    let h = match p.next_usize() {
        Some(v) => v,
        None => return,
    };
    let k = match p.next_usize() {
        Some(v) => v,
        None => return,
    };
    let require_cycle = match p.next_usize() {
        Some(v) => v != 0,
        None => return,
    };

    if w == 0 || h == 0 {
        print!("0\n");
        return;
    }
    let total_u128 = (w as u128).saturating_mul(h as u128);
    if total_u128 > (usize::MAX as u128) {
        print!("0\n");
        return;
    }
    let total = total_u128 as usize;
    if total == 0 {
        print!("0\n");
        return;
    }

    // Output would be enormous; also memory/time would be dominated by printing.
    const MAX_OUTPUT_CELLS: usize = 5_000_000;
    if total > MAX_OUTPUT_CELLS {
        print!("0\n");
        return;
    }

    let mut blocked = vec![0u8; total];
    for _ in 0..k {
        let x = match p.next_usize() {
            Some(v) => v,
            None => break,
        };
        let y = match p.next_usize() {
            Some(v) => v,
            None => break,
        };
        if x < w && y < h {
            blocked[id_of(x, y, w)] = 1;
        }
    }

    let mut blocked_count = 0usize;
    for &b in blocked.iter() {
        blocked_count += b as usize;
    }
    let free_count = total.saturating_sub(blocked_count);
    if free_count == 0 || blocked[0] != 0 {
        print!("0\n");
        return;
    }

    if precheck_impossible(w, h, &blocked, require_cycle, free_count) {
        print!("0\n");
        return;
    }

    // Fast paths for obstacle-free grids.
    if blocked_count == 0 {
        let path = if !require_cycle {
            build_snake_path(w, h)
        } else {
            match build_cycle_rect_no_obstacles(w, h) {
                Some(p) => p,
                None => {
                    print!("0\n");
                    return;
                }
            }
        };
        if let Some(out) = validate_and_print(w, h, &blocked, require_cycle, free_count, &path) {
            print!("{}", out);
        } else {
            print!("0\n");
        }
        return;
    }

    // Adaptive solving based on instance size.
    // - For small instances, do exact backtracking with strong pruning.
    // - For larger instances, use multi-start heuristic backtracking.
    let start_time = Instant::now();
    let deadline = start_time + Duration::from_secs(290);

    let exact_max = 80usize;
    let path_opt = if free_count <= exact_max {
        exact_solve(w, h, &blocked, require_cycle, free_count, deadline)
    } else {
        None
    };

    let path = if let Some(p) = path_opt {
        p
    } else {
        match heuristic_solve(w, h, &blocked, require_cycle, free_count, deadline) {
            Some(p) => p,
            None => {
                print!("0\n");
                return;
            }
        }
    };

    if let Some(out) = validate_and_print(w, h, &blocked, require_cycle, free_count, &path) {
        print!("{}", out);
    } else {
        print!("0\n");
    }
}

    private static double EstimateDeltaV(OrbitFeatures o1, OrbitFeatures o2, double mu)
    {
        // Coarse but fast, time-free rendezvous proxy:
        //   - Hohmann-like energy change between representative radii (a)
        //   - Plane change from angular-momentum vectors
        //   - Eccentricity vector mismatch penalty
        double r1 = SafeRadius(o1.A);
        double r2 = SafeRadius(o2.A);

        double vc1 = Math.Sqrt(mu / r1);
        double vc2 = Math.Sqrt(mu / r2);

        double dvH = HohmannDeltaV(mu, r1, r2);

        double dot = Vec3.Dot(o1.HHat, o2.HHat);
        if (double.IsNaN(dot)) dot = 1.0;
        dot = Clamp(dot, -1.0, 1.0);
        double planeAngle = Math.Acos(dot);
        double vPlane = Math.Min(vc1, vc2);
        double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * planeAngle);

        Vec3 de = o1.Evec - o2.Evec;
        double vE = 0.5 * (vc1 + vc2);
        double dvE = vE * de.Norm();

        double dv = dvH + dvPlane + 0.5 * dvE;
        if (!double.IsFinite(dv))
            return 1e9;
        return dv;
    }

    private static double HohmannDeltaV(double mu, double r1, double r2)
    {
        if (r1 <= 0 || r2 <= 0)
            return 1e9;

        if (Math.Abs(r1 - r2) < 1e-9)
            return 0.0;

        double aT = 0.5 * (r1 + r2);
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double vP = Math.Sqrt(mu * (2.0 / r1 - 1.0 / aT));
        double vA = Math.Sqrt(mu * (2.0 / r2 - 1.0 / aT));
        return Math.Abs(vP - v1) + Math.Abs(v2 - vA);
    }

    private static int[] SolveExact(int n, double[][] dist, double[] distStart)
    {
        int fullMask = (1 << n) - 1;
        int stateCount = (1 << n) * n;
        var dp = new double[stateCount];
        var parent = new int[stateCount];
        Array.Fill(dp, double.PositiveInfinity);
        Array.Fill(parent, -1);

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            int idx = mask * n + i;
            dp[idx] = distStart[i];
            parent[idx] = -1;
        }

        for (int mask = 1; mask <= fullMask; mask++)
        {
            int baseIdx = mask * n;
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0)
                    continue;

                double cur = dp[baseIdx + last];
                if (!double.IsFinite(cur))
                    continue;

                int remaining = fullMask ^ mask;
                while (remaining != 0)
                {
                    int lb = remaining & -remaining;
                    int next = BitIndex(lb);
                    remaining ^= lb;

                    int newMask = mask | (1 << next);
                    int newIdx = newMask * n + next;
                    double cand = cur + dist[last][next];
                    if (cand < dp[newIdx])
                    {
                        dp[newIdx] = cand;
                        parent[newIdx] = last;
                    }
                }
            }
        }

        double best = double.PositiveInfinity;
        int bestLast = 0;
        int fullBase = fullMask * n;
        for (int last = 0; last < n; last++)
        {
            double v = dp[fullBase + last];
            if (v < best)
            {
                best = v;
                bestLast = last;
            }
        }

        var route = new int[n];
        int curMask = fullMask;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int idx = curMask * n + curLast;
            int prev = parent[idx];
            curMask ^= 1 << curLast;
            if (prev < 0)
                break;
            curLast = prev;
        }

        return route;
    }

        // singleBit is power of two
        int idx = 0;
        while ((singleBit >>= 1) != 0)
            idx++;
        return idx;

    private static int PickFromRclStart(int n, double[] distStart, bool[] used, int rclSize, Random rng)
    {
        rclSize = Math.Max(1, Math.Min(rclSize, n));
        Span<int> cand = stackalloc int[8];
        Span<double> candD = stackalloc double[8];
        int k = Math.Min(rclSize, 8);
        for (int i = 0; i < k; i++) { cand[i] = -1; candD[i] = double.PositiveInfinity; }

        for (int j = 0; j < n; j++)
        {
            if (used[j]) continue;
            double d = distStart[j];
            InsertCandidate(cand, candD, k, j, d);
        }

        int count = 0;
        for (int i = 0; i < k; i++)
            if (cand[i] >= 0) count++;
        if (count == 0)
            return 0;
        return cand[rng.Next(count)];
    }

    private static int PickFromRclNext(int n, double[][] dist, int cur, bool[] used, int rclSize, Random rng)

    private static void InsertCandidate(Span<int> cand, Span<double> candD, int k, int id, double d)
    {
        for (int i = 0; i < k; i++)
        {
            if (d < candD[i])
            {
                for (int j = k - 1; j > i; j--)
                {
                    candD[j] = candD[j - 1];
                    cand[j] = cand[j - 1];
                }
                candD[i] = d;
                cand[i] = id;
                return;
            }
        }
    }

    private enum MoveType : byte { None = 0, TwoOpt = 1, Swap = 2 }

    private static void LocalSearch(int[] route, double[][] dist, double[] distStart, Stopwatch sw, double timeLimitSeconds, int maxIters)
    {
        int n = route.Length;
        const double eps = 1e-10;

        for (int iter = 0; iter < maxIters && sw.Elapsed.TotalSeconds < timeLimitSeconds; iter++)
        {
            double bestDelta = 0.0;
            int bestI = -1;
            int bestK = -1;
            MoveType bestMove = MoveType.None;

            // Best 2-opt (segment reversal) move
            for (int i = 0; i < n - 1; i++)
            {
                int prev = (i == 0) ? -1 : route[i - 1];
                int a = route[i];
                double prevToA = Edge(prev, a, dist, distStart);

                for (int k = i + 1; k < n; k++)
                {
                    int b = route[k];
                    int next = (k == n - 1) ? -1 : route[k + 1];
                    double delta = (Edge(prev, b, dist, distStart) + Edge(a, next, dist, distStart))
                                   - (prevToA + Edge(b, next, dist, distStart));
                    if (delta < bestDelta - eps)
                    {
                        bestDelta = delta;
                        bestI = i;
                        bestK = k;
                        bestMove = MoveType.TwoOpt;
                    }
                }
            }

            // Best swap move
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    double delta = SwapDelta(route, i, j, dist, distStart);
                    if (delta < bestDelta - eps)
                    {
                        bestDelta = delta;
                        bestI = i;
                        bestK = j;
                        bestMove = MoveType.Swap;
                    }
                }
            }

            if (bestMove == MoveType.None)
                return;

            if (bestMove == MoveType.TwoOpt)
            {
                Array.Reverse(route, bestI, bestK - bestI + 1);
            }
            else
            {
                int tmp = route[bestI];
                route[bestI] = route[bestK];
                route[bestK] = tmp;
            }
        }
    }

    private static double SwapDelta(int[] route, int i, int j, double[][] dist, double[] distStart)
    {
        if (i == j) return 0.0;
        if (i > j) { int t = i; i = j; j = t; }

        int n = route.Length;
        int a = route[i];
        int b = route[j];

        int prevA = (i == 0) ? -1 : route[i - 1];
        int nextA = (i == n - 1) ? -1 : route[i + 1];
        int prevB = (j == 0) ? -1 : route[j - 1];
        int nextB = (j == n - 1) ? -1 : route[j + 1];

        if (j == i + 1)
        {
            // prevA -> a -> b -> nextB  becomes  prevA -> b -> a -> nextB
            double oldCost = Edge(prevA, a, dist, distStart) + Edge(a, b, dist, distStart) + Edge(b, nextB, dist, distStart);
            double newCost = Edge(prevA, b, dist, distStart) + Edge(b, a, dist, distStart) + Edge(a, nextB, dist, distStart);
            return newCost - oldCost;
        }
        else
        {
            double oldCost = Edge(prevA, a, dist, distStart) + Edge(a, nextA, dist, distStart)
                           + Edge(prevB, b, dist, distStart) + Edge(b, nextB, dist, distStart);
            double newCost = Edge(prevA, b, dist, distStart) + Edge(b, nextA, dist, distStart)
                           + Edge(prevB, a, dist, distStart) + Edge(a, nextB, dist, distStart);
            return newCost - oldCost;
        }
    }

    private static double Edge(int from, int to, double[][] dist, double[] distStart)
    {
        if (to < 0) return 0.0;
        if (from < 0) return distStart[to];
        return dist[from][to];
    }

    private static double PathCost(int[] route, double[][] dist, double[] distStart)
    {
        if (route.Length == 0) return 0.0;
        double cost = distStart[route[0]];
        for (int i = 1; i < route.Length; i++)
            cost += dist[route[i - 1]][route[i]];
        return cost;
    }

        {

            Vec3 v = s.V;

            double rMag = r.Norm();
            double v2 = v.Dot(v);
            double denom = 2.0 / Math.Max(rMag, 1e-9) - v2 / mu;
            double a = denom != 0.0 ? 1.0 / denom : rMag;
            if (!double.IsFinite(a) || a <= 0.0)
                a = rMag;

            Vec3 h = Vec3.Cross(r, v);
            double hMag = h.Norm();

            else
            {
                evec = new Vec3(0, 0, 0);
            }
