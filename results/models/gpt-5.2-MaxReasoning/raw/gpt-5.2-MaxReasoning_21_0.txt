#include <algorithm>
#include <cstdint>
#include <deque>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::uint64_t;

struct Move {
    int axis = 0;
    int dir = 1; // -1 or +1
};

struct ParentInfo {
    uint64_t prev = 0;
    int axis = 0;
    int dir = 1;
};

struct Node {
    uint64_t pos = 0;
    int g = 0;
    int f = 0;
};

struct NodeCmp {
    bool operator()(const Node& a, const Node& b) const {
        if (a.f != b.f) return a.f > b.f;
        return a.g > b.g;
    }
};

class NDGrid {
public:
    int D = 0;
    std::vector<int> bounds;
    std::vector<uint64_t> stride;
    uint64_t totalCells = 0;

    explicit NDGrid(int dims, std::vector<int> b) : D(dims), bounds(std::move(b)) {
        stride.assign(D, 1);
        for (int i = 1; i < D; ++i) {
            stride[i] = stride[i - 1] * static_cast<uint64_t>(bounds[i - 1]);
        }
        totalCells = 1;
        for (int i = 0; i < D; ++i) totalCells *= static_cast<uint64_t>(bounds[i]);
    }

    uint64_t encode(const std::vector<int>& coord) const {
        uint64_t idx = 0;
        for (int i = 0; i < D; ++i) idx += static_cast<uint64_t>(coord[i]) * stride[i];
        return idx;
    }

    void decode(uint64_t idx, std::vector<int>& out) const {
        out.assign(D, 0);
        for (int i = 0; i < D; ++i) {
            out[i] = static_cast<int>((idx / stride[i]) % static_cast<uint64_t>(bounds[i]));
        }
    }

    int coordOf(uint64_t idx, int axis) const {
        return static_cast<int>((idx / stride[axis]) % static_cast<uint64_t>(bounds[axis]));
    }

    int manhattan(uint64_t aIdx, const std::vector<int>& bCoord) const {
        int sum = 0;
        for (int i = 0; i < D; ++i) {
            int a = static_cast<int>((aIdx / stride[i]) % static_cast<uint64_t>(bounds[i]));
            int diff = a - bCoord[i];
            sum += (diff < 0) ? -diff : diff;
        }
        return sum;
    }

    uint64_t step(uint64_t idx, int axis, int dir, bool& ok) const {
        const int c = coordOf(idx, axis);
        if (dir < 0) {
            if (c == 0) {
                ok = false;
                return idx;
            }
            ok = true;
            return idx - stride[axis];
        }
        if (c + 1 >= bounds[axis]) {
            ok = false;
            return idx;
        }
        ok = true;
        return idx + stride[axis];
    }
};

static std::vector<Move> astarPath(
    const NDGrid& g,
    uint64_t start,
    uint64_t goal,
    const std::function<bool(uint64_t)>& blocked,
    int nodeLimit
) {
    if (start == goal) return {};

    std::vector<int> goalCoord;
    g.decode(goal, goalCoord);

    std::priority_queue<Node, std::vector<Node>, NodeCmp> open;
    std::unordered_map<uint64_t, int> bestG;
    std::unordered_map<uint64_t, ParentInfo> parent;
    bestG.reserve(static_cast<size_t>(nodeLimit) * 2u + 16u);
    parent.reserve(static_cast<size_t>(nodeLimit) * 2u + 16u);

    const int h0 = g.manhattan(start, goalCoord);
    bestG[start] = 0;
    open.push(Node{start, 0, h0});

    int expanded = 0;
    while (!open.empty() && expanded < nodeLimit) {
        Node cur = open.top();
        open.pop();

        auto itg = bestG.find(cur.pos);
        if (itg == bestG.end() || cur.g != itg->second) continue; // stale

        if (cur.pos == goal) {
            std::vector<Move> path;
            uint64_t x = goal;
            while (x != start) {
                auto itp = parent.find(x);
                if (itp == parent.end()) break;
                path.push_back(Move{itp->second.axis, itp->second.dir});
                x = itp->second.prev;
            }
            std::reverse(path.begin(), path.end());
            return path;
        }

        ++expanded;

        for (int axis = 0; axis < g.D; ++axis) {
            for (int dir : {-1, +1}) {
                bool ok = false;
                uint64_t nxt = g.step(cur.pos, axis, dir, ok);
                if (!ok) continue;
                if (blocked(nxt) && nxt != goal) continue;

                const int ng = cur.g + 1;
                auto it = bestG.find(nxt);
                if (it != bestG.end() && it->second <= ng) continue;

                bestG[nxt] = ng;
                parent[nxt] = ParentInfo{cur.pos, axis, dir};
                const int nf = ng + g.manhattan(nxt, goalCoord);
                open.push(Node{nxt, ng, nf});
            }
        }
    }

    return {};
}

class SnakeSolver {
public:
    NDGrid grid;
    long long maxTurns = 0;

    std::unordered_set<uint64_t> obstacles;
    std::unordered_set<uint64_t> foods;

    std::deque<uint64_t> body;              // head at front
    std::unordered_set<uint64_t> occupied;  // all segments

    uint64_t head = 0;
    uint64_t tail = 0;

    std::deque<Move> plan;
    uint64_t planTarget = std::numeric_limits<uint64_t>::max();

    int wanderAxis = 0;
    int wanderDir = +1;

    explicit SnakeSolver(NDGrid g, long long turns) : grid(std::move(g)), maxTurns(turns) {}

    bool isBlockedStatic(uint64_t idx, bool allowTail) const {
        if (obstacles.find(idx) != obstacles.end()) return true;
        if (occupied.find(idx) == occupied.end()) return false;
        if (allowTail && idx == tail) return false;
        return true;
    }

    bool isMoveLegal(int axis, int dir) const {
        bool ok = false;
        uint64_t nxt = grid.step(head, axis, dir, ok);
        if (!ok) return false;
        if (obstacles.find(nxt) != obstacles.end()) return false;

        const bool willGrow = (foods.find(nxt) != foods.end());
        if (occupied.find(nxt) == occupied.end()) return true;

        if (!willGrow && nxt == tail) return true;
        return false;
    }

    uint64_t moveHeadIndex(int axis, int dir, bool& ok) const {
        return grid.step(head, axis, dir, ok);
    }

    int degreeIfMove(uint64_t newHead, bool newWillGrow, uint64_t oldTail) const {
        int deg = 0;
        for (int axis = 0; axis < grid.D; ++axis) {
            for (int dir : {-1, +1}) {
                bool ok = false;
                uint64_t nxt = grid.step(newHead, axis, dir, ok);
                if (!ok) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;

                bool occ = (occupied.find(nxt) != occupied.end());
                if (!occ) {
                    ++deg;
                    continue;
                }

                if (!newWillGrow && nxt == oldTail) {
                    ++deg;
                    continue;
                }
            }
        }
        return deg;
    }

    uint64_t pickClosestFood() const {
        uint64_t best = std::numeric_limits<uint64_t>::max();
        int bestDist = std::numeric_limits<int>::max();
        std::vector<int> foodCoord;
        for (uint64_t f : foods) {
            grid.decode(f, foodCoord);
            int d = grid.manhattan(head, foodCoord);
            if (d < bestDist) {
                bestDist = d;
                best = f;
            }
        }
        return best;
    }

    bool makePlanTo(uint64_t target) {
        if (target == std::numeric_limits<uint64_t>::max()) return false;
        const int nodeLimit = (grid.totalCells <= 500000ull) ? static_cast<int>(std::min<uint64_t>(grid.totalCells, 250000ull)) : 60000;

        auto blocked = [&](uint64_t idx) -> bool {
            return isBlockedStatic(idx, /*allowTail=*/true);
        };

        std::vector<Move> p = astarPath(grid, head, target, blocked, nodeLimit);
        if (p.empty()) return false;

        plan.clear();
        for (const auto& m : p) plan.push_back(m);
        planTarget = target;
        return true;
    }

    bool planToFood() {
        if (foods.empty()) return false;

        // Try up to a few nearest foods by Manhattan distance.
        std::vector<std::pair<int, uint64_t>> cands;
        cands.reserve(std::min<size_t>(foods.size(), 16u));
        std::vector<int> fcoord;
        for (uint64_t f : foods) {
            grid.decode(f, fcoord);
            cands.push_back({grid.manhattan(head, fcoord), f});
        }
        const size_t K = std::min<size_t>(cands.size(), 5u);
        if (cands.size() > K) {
            std::nth_element(cands.begin(), cands.begin() + K, cands.end(),
                             [](const auto& a, const auto& b) { return a.first < b.first; });
            cands.resize(K);
        }
        std::sort(cands.begin(), cands.end(), [](const auto& a, const auto& b) { return a.first < b.first; });

        for (const auto& [_, f] : cands) {
            if (makePlanTo(f)) return true;
        }
        return false;
    }

    bool planToTail() {
        if (body.empty()) return false;
        const uint64_t t = tail;

        const int nodeLimit = (grid.totalCells <= 500000ull) ? static_cast<int>(std::min<uint64_t>(grid.totalCells, 200000ull)) : 60000;
        auto blocked = [&](uint64_t idx) -> bool {
            if (idx == t) return false;
            return isBlockedStatic(idx, /*allowTail=*/true);
        };
        std::vector<Move> p = astarPath(grid, head, t, blocked, nodeLimit);
        if (p.empty()) return false;
        plan.clear();
        for (const auto& m : p) plan.push_back(m);
        planTarget = t;
        return true;
    }

    Move chooseFallbackMove() {
        // 1) If possible, follow a path to tail to stay safe.
        if (plan.empty() || planTarget != tail) {
            if (planToTail()) {
                Move m = plan.front();
                plan.pop_front();
                if (plan.empty()) planTarget = std::numeric_limits<uint64_t>::max();
                return m;
            }
        }

        // 2) Choose a locally safe move with best degree (avoid dead ends).
        Move bestMove{0, +1};
        int bestScore = std::numeric_limits<int>::min();
        bool found = false;

        const uint64_t oldTail = tail;
        for (int axis = 0; axis < grid.D; ++axis) {
            for (int dir : {-1, +1}) {
                bool ok = false;
                uint64_t nxt = grid.step(head, axis, dir, ok);
                if (!ok) continue;
                if (obstacles.find(nxt) != obstacles.end()) continue;

                const bool willGrow = (foods.find(nxt) != foods.end());
                if (occupied.find(nxt) != occupied.end()) {
                    if (willGrow || nxt != oldTail) continue;
                }

                const int deg = degreeIfMove(nxt, willGrow, oldTail);
                int score = deg * 10;

                // Slight bias towards "wandering" direction for stability.
                if (axis == wanderAxis && dir == wanderDir) score += 1;

                if (score > bestScore) {
                    bestScore = score;
                    bestMove = Move{axis, dir};
                    found = true;
                }
            }
        }

        if (found) {
            // Update wander direction if we hit a boundary in that axis.
            wanderAxis = bestMove.axis;
            wanderDir = bestMove.dir;
            return bestMove;
        }

        // 3) Should be rare: output something deterministic.
        return Move{0, +1};
    }

    Move chooseMove() {
        if (!plan.empty()) {
            Move m = plan.front();
            if (isMoveLegal(m.axis, m.dir)) {
                plan.pop_front();
                if (plan.empty()) planTarget = std::numeric_limits<uint64_t>::max();
                return m;
            }
            plan.clear();
            planTarget = std::numeric_limits<uint64_t>::max();
        }

        if (!foods.empty()) {
            if (planToFood() && !plan.empty()) {
                Move m = plan.front();
                if (isMoveLegal(m.axis, m.dir)) {
                    plan.pop_front();
                    if (plan.empty()) planTarget = std::numeric_limits<uint64_t>::max();
                    return m;
                }
                plan.clear();
                planTarget = std::numeric_limits<uint64_t>::max();
            }
        }

        return chooseFallbackMove();
    }

    void applyMove(const Move& m) {
        bool ok = false;
        uint64_t nxt = grid.step(head, m.axis, m.dir, ok);
        if (!ok) return; // would be dead; keep state unchanged
        if (obstacles.find(nxt) != obstacles.end()) return;

        const bool willGrow = (foods.find(nxt) != foods.end());

        // Collision check (tail is special when not growing).
        if (occupied.find(nxt) != occupied.end()) {
            if (willGrow || nxt != tail) return;
        }

        body.push_front(nxt);
        occupied.insert(nxt);
        head = nxt;

        if (willGrow) {
            foods.erase(nxt);
        } else {
            uint64_t oldTail = body.back();
            body.pop_back();
            occupied.erase(oldTail);
        }
        tail = body.empty() ? head : body.back();
    }
};

static bool buildInitialSnake(
    const NDGrid& grid,
    int length,
    uint64_t head,
    const std::unordered_set<uint64_t>& obstacles,
    const std::unordered_set<uint64_t>& foods,
    std::deque<uint64_t>& outBody,
    std::unordered_set<uint64_t>& outOcc
) {
    if (length <= 0) length = 1;

    std::vector<int> headCoord;
    grid.decode(head, headCoord);

    for (int axis = 0; axis < grid.D; ++axis) {
        const int hc = headCoord[axis];
        for (int dir : {-1, +1}) {
            const int tailCoord = hc + (length - 1) * dir;
            if (tailCoord < 0 || tailCoord >= grid.bounds[axis]) continue;

            bool ok = true;
            std::deque<uint64_t> b;
            std::unordered_set<uint64_t> occ;
            b.clear();
            occ.clear();
            b.push_back(head);
            occ.insert(head);

            uint64_t cur = head;
            for (int k = 1; k < length; ++k) {
                bool stepOk = false;
                cur = grid.step(cur, axis, dir, stepOk);
                if (!stepOk) {
                    ok = false;
                    break;
                }
                if (obstacles.find(cur) != obstacles.end()) {
                    ok = false;
                    break;
                }
                if (foods.find(cur) != foods.end()) {
                    ok = false;
                    break;
                }
                if (occ.find(cur) != occ.end()) {
                    ok = false;
                    break;
                }
                b.push_back(cur);
                occ.insert(cur);
            }

            if (ok) {
                // b currently head->tail along chosen axis; convert to head at front.
                outBody.clear();
                outOcc.clear();
                for (uint64_t x : b) outBody.push_back(x);
                outOcc = std::move(occ);
                // Ensure head at front.
                while (!outBody.empty() && outBody.front() != head) outBody.pop_front();
                return true;
            }
        }
    }

    // Fallback: length 1 at head.
    outBody.clear();
    outOcc.clear();
    outBody.push_back(head);
    outOcc.insert(head);
    return false;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int D = 0;
    long long maxTurns = 0;
    if (!(std::cin >> D >> maxTurns)) return 0;
    if (D <= 0) return 0;

    std::vector<int> bounds(D, 0);
    for (int i = 0; i < D; ++i) std::cin >> bounds[i];

    int snakeLen = 1;
    std::vector<int> headCoord(D, 0);
    std::cin >> snakeLen;
    for (int i = 0; i < D; ++i) std::cin >> headCoord[i];

    NDGrid grid(D, bounds);
    uint64_t head = grid.encode(headCoord);

    int F = 0;
    std::cin >> F;
    std::unordered_set<uint64_t> foods;
    foods.reserve(static_cast<size_t>(F) * 2u + 16u);
    for (int i = 0; i < F; ++i) {
        std::vector<int> c(D, 0);
        for (int j = 0; j < D; ++j) std::cin >> c[j];
        foods.insert(grid.encode(c));
    }

    int O = 0;
    std::cin >> O;
    std::unordered_set<uint64_t> obstacles;
    obstacles.reserve(static_cast<size_t>(O) * 2u + 16u);
    for (int i = 0; i < O; ++i) {
        std::vector<int> c(D, 0);
        for (int j = 0; j < D; ++j) std::cin >> c[j];
        obstacles.insert(grid.encode(c));
    }

    SnakeSolver solver(grid, maxTurns);
    solver.foods = std::move(foods);
    solver.obstacles = std::move(obstacles);

    buildInitialSnake(grid, snakeLen, head, solver.obstacles, solver.foods, solver.body, solver.occupied);
    solver.head = solver.body.front();
    solver.tail = solver.body.back();

    for (long long t = 0; t < maxTurns; ++t) {
        Move m = solver.chooseMove();
        std::cout << m.axis << ' ' << m.dir << "\n";
        solver.applyMove(m);
    }

    return 0;
}
