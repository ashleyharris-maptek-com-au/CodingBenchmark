@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, shift: u32) -> u32 {
  return (x << shift) | (x >> (32u - shift));
}

fn quarter_round(a_in: u32, b_in: u32, c_in: u32, d_in: u32) -> vec4<u32> {
  var a: u32 = a_in;
  var b: u32 = b_in;
  var c: u32 = c_in;
  var d: u32 = d_in;

  a = a + b;
  d = rotl32(d ^ a, 16u);
  c = c + d;
  b = rotl32(b ^ c, 12u);
  a = a + b;
  d = rotl32(d ^ a, 8u);
  c = c + d;
  b = rotl32(b ^ c, 7u);

  return vec4<u32>(a, b, c, d);
}

fn custom_hash(nonce: u32) -> vec2<u32> {
  var state: array<u32, 16>;

  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = baseData[i] ^ (nonce + i);
  }

  for (var r: u32 = 0u; r < 8u; r = r + 1u) {
    let t0: vec4<u32> = quarter_round(state[0], state[1], state[2], state[3]);
    state[0] = t0.x;
    state[1] = t0.y;
    state[2] = t0.z;
    state[3] = t0.w;

    let t1: vec4<u32> = quarter_round(state[4], state[5], state[6], state[7]);
    state[4] = t1.x;
    state[5] = t1.y;
    state[6] = t1.z;
    state[7] = t1.w;

    let t2: vec4<u32> = quarter_round(state[8], state[9], state[10], state[11]);
    state[8] = t2.x;
    state[9] = t2.y;
    state[10] = t2.z;
    state[11] = t2.w;

    let t3: vec4<u32> = quarter_round(state[12], state[13], state[14], state[15]);
    state[12] = t3.x;
    state[13] = t3.y;
    state[14] = t3.z;
    state[15] = t3.w;
  }

  let h0: u32 =
    state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
  let h1: u32 =
    state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

  return vec2<u32>(h0, h1);
}

@compute @workgroup_size(256u)
fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
  let idx: u32 = gid.x;
  if (idx >= params.z) {
    return;
  }

  let nonce: u32 = params.y + idx;
  let h: vec2<u32> = custom_hash(nonce);
  let h0: u32 = h.x;
  let h1: u32 = h.y;

  let c0: u32 = countTrailingZeros(h0);
  var trailing_zeros: u32 = 0u;
  if (c0 < 32u) {
    trailing_zeros = c0;
  } else {
    trailing_zeros = 32u + countTrailingZeros(h1);
  }

  if (trailing_zeros >= params.x) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h0;
    result[3] = h1;
  }
}
