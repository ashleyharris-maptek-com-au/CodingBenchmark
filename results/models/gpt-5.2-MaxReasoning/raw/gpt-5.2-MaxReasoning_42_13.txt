; SPIR-V Geometry Shader (Vulkan 1.0)
OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 12
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %emit_edge "emit_edge"
OpName %gl_PerVertex_in "gl_PerVertex"
OpName %gl_in "gl_in"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %gl_PerVertex_out "gl_PerVertex"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpName %ubo "ubo"

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%ptr_ubo = OpTypePointer Uniform %UBO

%gl_in = OpVariable %ptr_in_arr_pv Input
%in_worldPos = OpVariable %ptr_in_arr_v3 Input
%in_normal = OpVariable %ptr_in_arr_v3 Input
%in_color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output
%ubo = OpVariable %ptr_ubo Uniform

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_thickness = OpConstant %float 0.008

%fn_void = OpTypeFunction %void
%fn_emit_edge = OpTypeFunction %void %v4float %v4float %v3float %v3float

%emit_edge = OpFunction %void None %fn_emit_edge
%p0 = OpFunctionParameter %v4float
%p1 = OpFunctionParameter %v4float
%c0 = OpFunctionParameter %v3float
%c1 = OpFunctionParameter %v3float
%emit_edge_entry = OpLabel
%c_add = OpFAdd %v3float %c0 %c1
%c_avg = OpVectorTimesScalar %v3float %c_add %float_0_5

%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%w0 = OpCompositeExtract %float %p0 3
%w1 = OpCompositeExtract %float %p1 3
%inv_w0 = OpFDiv %float %float_1 %w0
%inv_w1 = OpFDiv %float %float_1 %w1
%ndc0_xy = OpVectorTimesScalar %v2float %p0_xy %inv_w0
%ndc1_xy = OpVectorTimesScalar %v2float %p1_xy %inv_w1
%delta = OpFSub %v2float %ndc1_xy %ndc0_xy
%dir = OpExtInst %v2float %glsl Normalize %delta
%dir_x = OpCompositeExtract %float %dir 0
%dir_y = OpCompositeExtract %float %dir 1
%neg_dir_y = OpFNegate %float %dir_y
%perp = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%offset_ndc = OpVectorTimesScalar %v2float %perp %float_thickness
%offset0_xy = OpVectorTimesScalar %v2float %offset_ndc %w0
%offset1_xy = OpVectorTimesScalar %v2float %offset_ndc %w1

%off0_x = OpCompositeExtract %float %offset0_xy 0
%off0_y = OpCompositeExtract %float %offset0_xy 1
%off1_x = OpCompositeExtract %float %offset1_xy 0
%off1_y = OpCompositeExtract %float %offset1_xy 1
%offset0_v4 = OpCompositeConstruct %v4float %off0_x %off0_y %float_0 %float_0
%offset1_v4 = OpCompositeConstruct %v4float %off1_x %off1_y %float_0 %float_0

%p0_plus = OpFAdd %v4float %p0 %offset0_v4
%p0_minus = OpFSub %v4float %p0 %offset0_v4
%p1_plus = OpFAdd %v4float %p1 %offset1_v4
%p1_minus = OpFSub %v4float %p1 %offset1_v4

%gl_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %gl_out_pos %p0_plus
OpStore %outColor %c_avg
OpEmitVertex

OpStore %gl_out_pos %p0_minus
OpStore %outColor %c_avg
OpEmitVertex

OpStore %gl_out_pos %p1_plus
OpStore %outColor %c_avg
OpEmitVertex

OpStore %gl_out_pos %p1_minus
OpStore %outColor %c_avg
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd

%main = OpFunction %void None %fn_void
%main_entry = OpLabel
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos0 = OpLoad %v4float %p0_ptr
%pos2 = OpLoad %v4float %p2_ptr
%pos4 = OpLoad %v4float %p4_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %in_color %int_0
%c2_ptr = OpAccessChain %ptr_in_v3 %in_color %int_2
%c4_ptr = OpAccessChain %ptr_in_v3 %in_color %int_4
%col0 = OpLoad %v3float %c0_ptr
%col2 = OpLoad %v3float %c2_ptr
%col4 = OpLoad %v3float %c4_ptr

OpFunctionCall %void %emit_edge %pos0 %pos2 %col0 %col2
OpFunctionCall %void %emit_edge %pos2 %pos4 %col2 %col4
OpFunctionCall %void %emit_edge %pos4 %pos0 %col4 %col0

OpReturn
OpFunctionEnd
