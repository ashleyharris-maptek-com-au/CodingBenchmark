using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private const double AabbPadFactor = 1e-9; // conservative padding to avoid boundary/overlap issues from FP noise

    private enum Scale
    {
        Trivial,
        Medium,
        Large,
        Extreme
    }

    private readonly struct Region
    {
        public readonly double Ox, Oy, Oz;
        public readonly double W, H, D;
        public Region(double ox, double oy, double oz, double w, double h, double d)
        {
            Ox = ox; Oy = oy; Oz = oz;
            W = w; H = h; D = d;
        }
    }

    private sealed class PlanNode
    {
        public Region Region;
        public int Ori;
        public long Nx, Ny, Nz;
        public double UsedW, UsedH, UsedD;
        public PlanNode? ChildX;
        public PlanNode? ChildY;
        public PlanNode? ChildZ;
    }

    private struct Orientation
    {
        public double Qw, Qx, Qy, Qz;
        public double M00, M01, M02;
        public double M10, M11, M12;
        public double M20, M21, M22;
        public double MinX, MinY, MinZ;
        public double SizeX, SizeY, SizeZ;
    }

    private struct AxisVec
    {
        public int X, Y, Z;
        public AxisVec(int x, int y, int z) { X = x; Y = y; Z = z; }
        public static int Dot(in AxisVec a, in AxisVec b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static AxisVec Cross(in AxisVec a, in AxisVec b)
            => new AxisVec(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        if (!fs.TryNextInt(out int vCount) || !fs.TryNextInt(out int fCount))
        {
            Console.WriteLine("0");
            return;
        }

        double[] xs = new double[vCount];
        double[] ys = new double[vCount];
        double[] zs = new double[vCount];
        for (int i = 0; i < vCount; i++)
        {
            xs[i] = fs.NextDouble();
            ys[i] = fs.NextDouble();
            zs[i] = fs.NextDouble();
        }

        // Faces are not required for the conservative AABB-based strategies, but we must consume input.
        for (int i = 0; i < fCount; i++)
        {
            fs.NextInt();
            fs.NextInt();
            fs.NextInt();
        }

        double W = fs.NextDouble();
        double H = fs.NextDouble();
        double D = fs.NextDouble();

        if (!(W > 0 && H > 0 && D > 0) || vCount <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        double maxDim = Math.Max(W, Math.Max(H, D));
        Scale scale = ClassifyScale(maxDim);

        // Adaptive: keep orientation sampling small when geometry is large or container is massive.
        int axisCount = 24;
        int extraRandom = scale switch
        {
            Scale.Trivial => 160,
            Scale.Medium => 96,
            Scale.Large => 32,
            _ => 0
        };

        if (vCount > 5000) extraRandom = (int)(extraRandom * (5000.0 / vCount));
        if (vCount > 50000) extraRandom = 0;
        extraRandom = Math.Clamp(extraRandom, 0, 256 - axisCount);

        // Extreme scale: use only axis-aligned rotations and single-block packing for speed.
        bool extremeFastPath = scale == Scale.Extreme || vCount > 50000 || maxDim >= 200;

        List<Orientation> orientations = BuildOrientations(axisCount, extraRandom, extremeFastPath);
        PrecomputeAabbs(orientations, xs, ys, zs);

        if (orientations.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }

        PlanNode? plan;
        if (extremeFastPath)
        {
            plan = BuildSingleBlockPlan(new Region(0, 0, 0, W, H, D), orientations);
        }
        else
        {
            int depthLimit = scale == Scale.Trivial ? 18 : (scale == Scale.Medium ? 14 : 10);
            plan = BuildGuillotinePlan(new Region(0, 0, 0, W, H, D), orientations, depth: 0, depthLimit);
        }

        long total = CountPlan(plan);
        var outStream = Console.OpenStandardOutput();
        using var writer = new StreamWriter(outStream, new UTF8Encoding(encoderShouldEmitUTF8Identifier: false), bufferSize: 1 << 20)
        {
            NewLine = "\n"
        };
        writer.WriteLine(total.ToString(CultureInfo.InvariantCulture));
        if (total == 0 || plan is null)
        {
            writer.Flush();
            return;
        }

        WritePlan(writer, plan, orientations);
        writer.Flush();
    }

    private static Scale ClassifyScale(double maxDim)
    {
        if (maxDim <= 6.0) return Scale.Trivial;
        if (maxDim <= 25.0) return Scale.Medium;
        if (maxDim <= 120.0) return Scale.Large;
        return Scale.Extreme;
    }

    private static List<Orientation> BuildOrientations(int axisCount, int extraRandom, bool extremeFastPath)
    {
        var list = new List<Orientation>(axisCount + extraRandom);

        foreach (var o in GenerateAxisOrientations())
            list.Add(o);

        if (extremeFastPath || extraRandom <= 0)
            return list;

        // A few deterministic "nice" angles (can help for some polyhedra).
        AddNiceAngleRotations(list);

        // Deterministic random samples.
        var rng = new XorShift64(0xC0FFEEu);
        for (int i = list.Count; i < axisCount + extraRandom; i++)
        {
            var q = RandomUnitQuaternion(ref rng);
            list.Add(FromQuaternion(q.qw, q.qx, q.qy, q.qz));
        }

        // Cap to requested count (axisCount + extraRandom).
        if (list.Count > axisCount + extraRandom)
            list.RemoveRange(axisCount + extraRandom, list.Count - (axisCount + extraRandom));

        return list;
    }

    private static IEnumerable<Orientation> GenerateAxisOrientations()
    {
        AxisVec[] axes =
        {
            new AxisVec( 1, 0, 0),
            new AxisVec(-1, 0, 0),
            new AxisVec( 0, 1, 0),
            new AxisVec( 0,-1, 0),
            new AxisVec( 0, 0, 1),
            new AxisVec( 0, 0,-1),
        };

        var seen = new HashSet<string>(StringComparer.Ordinal);
        for (int i = 0; i < axes.Length; i++)
        {
            for (int j = 0; j < axes.Length; j++)
            {
                if (AxisVec.Dot(axes[i], axes[j]) != 0) continue;
                AxisVec a = axes[i];
                AxisVec b = axes[j];
                AxisVec c = AxisVec.Cross(a, b); // right-handed
                if (c.X == 0 && c.Y == 0 && c.Z == 0) continue;

                // Columns are a,b,c
                double m00 = a.X, m10 = a.Y, m20 = a.Z;
                double m01 = b.X, m11 = b.Y, m21 = b.Z;
                double m02 = c.X, m12 = c.Y, m22 = c.Z;

                // Deduplicate (shouldn't be necessary, but cheap).
                string key = $"{m00},{m10},{m20},{m01},{m11},{m21},{m02},{m12},{m22}";
                if (!seen.Add(key)) continue;

                yield return FromMatrix(m00, m01, m02, m10, m11, m12, m20, m21, m22);
            }
        }
    }

    private static void AddNiceAngleRotations(List<Orientation> list)
    {
        // Rotations by 45 degrees around principal axes and a couple of diagonals.
        const double deg45 = Math.PI / 4.0;
        AddAxisAngle(list, 1, 0, 0, deg45);
        AddAxisAngle(list, 0, 1, 0, deg45);
        AddAxisAngle(list, 0, 0, 1, deg45);
        AddAxisAngle(list, 1, 1, 0, deg45);
        AddAxisAngle(list, 1, 0, 1, deg45);
        AddAxisAngle(list, 0, 1, 1, deg45);
    }

    private static void AddAxisAngle(List<Orientation> list, double ax, double ay, double az, double angle)
    {
        double len = Math.Sqrt(ax * ax + ay * ay + az * az);
        if (len <= 0) return;
        ax /= len; ay /= len; az /= len;
        double half = angle * 0.5;
        double s = Math.Sin(half);
        double qw = Math.Cos(half);
        double qx = ax * s, qy = ay * s, qz = az * s;
        list.Add(FromQuaternion(qw, qx, qy, qz));
    }

    private static (double qw, double qx, double qy, double qz) RandomUnitQuaternion(ref XorShift64 rng)
    {
        // Shoemake method.
        double u1 = rng.NextDouble01();
        double u2 = rng.NextDouble01();
        double u3 = rng.NextDouble01();

        double sqrt1MinusU1 = Math.Sqrt(1.0 - u1);
        double sqrtU1 = Math.Sqrt(u1);
        double twoPiU2 = 2.0 * Math.PI * u2;
        double twoPiU3 = 2.0 * Math.PI * u3;

        double qx = sqrt1MinusU1 * Math.Sin(twoPiU2);
        double qy = sqrt1MinusU1 * Math.Cos(twoPiU2);
        double qz = sqrtU1 * Math.Sin(twoPiU3);
        double qw = sqrtU1 * Math.Cos(twoPiU3);

        // Make qw non-negative for stable output.
        if (qw < 0) { qw = -qw; qx = -qx; qy = -qy; qz = -qz; }
        return (qw, qx, qy, qz);
    }

    private static Orientation FromQuaternion(double qw, double qx, double qy, double qz)
    {
        // Normalize defensively.
        double n = Math.Sqrt(qw * qw + qx * qx + qy * qy + qz * qz);
        if (n <= 0) { qw = 1; qx = qy = qz = 0; n = 1; }
        qw /= n; qx /= n; qy /= n; qz /= n;
        if (qw < 0) { qw = -qw; qx = -qx; qy = -qy; qz = -qz; }

        double xx = qx * qx, yy = qy * qy, zz = qz * qz;
        double xy = qx * qy, xz = qx * qz, yz = qy * qz;
        double wx = qw * qx, wy = qw * qy, wz = qw * qz;

        Orientation o = default;
        o.Qw = qw; o.Qx = qx; o.Qy = qy; o.Qz = qz;
        o.M00 = 1.0 - 2.0 * (yy + zz);
        o.M01 = 2.0 * (xy - wz);
        o.M02 = 2.0 * (xz + wy);
        o.M10 = 2.0 * (xy + wz);
        o.M11 = 1.0 - 2.0 * (xx + zz);
        o.M12 = 2.0 * (yz - wx);
        o.M20 = 2.0 * (xz - wy);
        o.M21 = 2.0 * (yz + wx);
        o.M22 = 1.0 - 2.0 * (xx + yy);
        return o;
    }

    private static Orientation FromMatrix(
        double m00, double m01, double m02,
        double m10, double m11, double m12,
        double m20, double m21, double m22)
    {
        // Convert matrix -> quaternion, then fill matrix back to keep it orthonormal.
        double qw, qx, qy, qz;
        double trace = m00 + m11 + m22;
        if (trace > 0)
        {
            double s = Math.Sqrt(trace + 1.0) * 2.0;
            qw = 0.25 * s;
            qx = (m21 - m12) / s;
            qy = (m02 - m20) / s;
            qz = (m10 - m01) / s;
        }
        else if (m00 > m11 && m00 > m22)
        {
            double s = Math.Sqrt(1.0 + m00 - m11 - m22) * 2.0;
            qw = (m21 - m12) / s;
            qx = 0.25 * s;
            qy = (m01 + m10) / s;
            qz = (m02 + m20) / s;
        }
        else if (m11 > m22)
        {
            double s = Math.Sqrt(1.0 + m11 - m00 - m22) * 2.0;
            qw = (m02 - m20) / s;
            qx = (m01 + m10) / s;
            qy = 0.25 * s;
            qz = (m12 + m21) / s;
        }
        else
        {
            double s = Math.Sqrt(1.0 + m22 - m00 - m11) * 2.0;
            qw = (m10 - m01) / s;
            qx = (m02 + m20) / s;
            qy = (m12 + m21) / s;
            qz = 0.25 * s;
        }

        return FromQuaternion(qw, qx, qy, qz);
    }

    private static void PrecomputeAabbs(List<Orientation> list, double[] xs, double[] ys, double[] zs)
    {
        for (int i = 0; i < list.Count; i++)
        {
            Orientation o = list[i];
            double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
            double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;

            double m00 = o.M00, m01 = o.M01, m02 = o.M02;
            double m10 = o.M10, m11 = o.M11, m12 = o.M12;
            double m20 = o.M20, m21 = o.M21, m22 = o.M22;

            for (int v = 0; v < xs.Length; v++)
            {
                double x = xs[v], y = ys[v], z = zs[v];
                double rx = m00 * x + m01 * y + m02 * z;
                double ry = m10 * x + m11 * y + m12 * z;
                double rz = m20 * x + m21 * y + m22 * z;

                if (rx < minX) minX = rx;
                if (ry < minY) minY = ry;
                if (rz < minZ) minZ = rz;
                if (rx > maxX) maxX = rx;
                if (ry > maxY) maxY = ry;
                if (rz > maxZ) maxZ = rz;
            }

            double spanX = maxX - minX;
            double spanY = maxY - minY;
            double spanZ = maxZ - minZ;

            if (!(spanX > 0 && spanY > 0 && spanZ > 0) || double.IsInfinity(spanX) || double.IsNaN(spanX))
            {
                // Degenerate; mark as unusable.
                o.SizeX = o.SizeY = o.SizeZ = double.PositiveInfinity;
                list[i] = o;
                continue;
            }

            double padX = AabbPadFactor * (Math.Abs(spanX) + 1.0);
            double padY = AabbPadFactor * (Math.Abs(spanY) + 1.0);
            double padZ = AabbPadFactor * (Math.Abs(spanZ) + 1.0);

            o.MinX = minX; o.MinY = minY; o.MinZ = minZ;
            o.SizeX = spanX + padX;
            o.SizeY = spanY + padY;
            o.SizeZ = spanZ + padZ;
            list[i] = o;
        }

        // Remove unusable orientations (keeps plan selection simple).
        for (int i = list.Count - 1; i >= 0; i--)
        {
            if (double.IsInfinity(list[i].SizeX) || double.IsInfinity(list[i].SizeY) || double.IsInfinity(list[i].SizeZ))
                list.RemoveAt(i);
        }
    }

    private static PlanNode? BuildSingleBlockPlan(Region region, List<Orientation> orientations)
    {
        var best = BestFit(region.W, region.H, region.D, orientations);
        if (best.count == 0) return null;

        Orientation o = orientations[best.ori];
        return new PlanNode
        {
            Region = region,
            Ori = best.ori,
            Nx = best.nx,
            Ny = best.ny,
            Nz = best.nz,
            UsedW = best.nx * o.SizeX,
            UsedH = best.ny * o.SizeY,
            UsedD = best.nz * o.SizeZ
        };
    }

    private static PlanNode? BuildGuillotinePlan(Region region, List<Orientation> orientations, int depth, int depthLimit)
    {
        if (depth >= depthLimit) return BuildSingleBlockPlan(region, orientations);

        var best = BestFit(region.W, region.H, region.D, orientations);
        if (best.count == 0) return null;

        Orientation o = orientations[best.ori];
        double usedW = best.nx * o.SizeX;
        double usedH = best.ny * o.SizeY;
        double usedD = best.nz * o.SizeZ;

        var node = new PlanNode
        {
            Region = region,
            Ori = best.ori,
            Nx = best.nx,
            Ny = best.ny,
            Nz = best.nz,
            UsedW = usedW,
            UsedH = usedH,
            UsedD = usedD
        };

        double remW = region.W - usedW;
        double remH = region.H - usedH;
        double remD = region.D - usedD;

        // Disjoint partition (X slab first covers all remaining X; then Y within usedW; then Z within usedW & usedH).
        if (remW > 1e-12 && usedW >= 0)
        {
            node.ChildX = BuildGuillotinePlan(
                new Region(region.Ox + usedW, region.Oy, region.Oz, remW, region.H, region.D),
                orientations, depth + 1, depthLimit);
        }
        if (remH > 1e-12 && usedW > 1e-12)
        {
            node.ChildY = BuildGuillotinePlan(
                new Region(region.Ox, region.Oy + usedH, region.Oz, usedW, remH, region.D),
                orientations, depth + 1, depthLimit);
        }
        if (remD > 1e-12 && usedW > 1e-12 && usedH > 1e-12)
        {
            node.ChildZ = BuildGuillotinePlan(
                new Region(region.Ox, region.Oy, region.Oz + usedD, usedW, usedH, remD),
                orientations, depth + 1, depthLimit);
        }

        return node;
    }

    private static (int ori, long nx, long ny, long nz, long count) BestFit(double W, double H, double D, List<Orientation> orientations)
    {
        long bestCount = 0;
        int bestOri = 0;
        long bestNx = 0, bestNy = 0, bestNz = 0;

        for (int i = 0; i < orientations.Count; i++)
        {
            Orientation o = orientations[i];

            long nx = FloorDiv(W, o.SizeX);
            if (nx <= 0) continue;
            long ny = FloorDiv(H, o.SizeY);
            if (ny <= 0) continue;
            long nz = FloorDiv(D, o.SizeZ);
            if (nz <= 0) continue;

            long c = Mul3(nx, ny, nz);
            if (c > bestCount)
            {
                bestCount = c;
                bestOri = i;
                bestNx = nx;
                bestNy = ny;
                bestNz = nz;
            }
        }

        return (bestOri, bestNx, bestNy, bestNz, bestCount);
    }

    private static long FloorDiv(double a, double b)
    {
        if (!(a > 0) || !(b > 0) || double.IsInfinity(b) || double.IsNaN(b)) return 0;
        // Conservative floor with tiny epsilon to counteract rounding.
        double eps = 1e-12 * (Math.Abs(a) + 1.0);
        double v = (a + eps) / b;
        if (v <= 0) return 0;
        double f = Math.Floor(v);
        if (f >= long.MaxValue) return long.MaxValue;
        return (long)f;
    }

    private static long Mul3(long a, long b, long c)
    {
        if (a <= 0 || b <= 0 || c <= 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        long ab = a * b;
        if (ab > long.MaxValue / c) return long.MaxValue;
        return ab * c;
    }

    private static long CountPlan(PlanNode? n)
    {
        if (n is null) return 0;
        long here = Mul3(n.Nx, n.Ny, n.Nz);
        long cx = CountPlan(n.ChildX);
        long cy = CountPlan(n.ChildY);
        long cz = CountPlan(n.ChildZ);
        long sum = here;
        sum = AddSat(sum, cx);
        sum = AddSat(sum, cy);
        sum = AddSat(sum, cz);
        return sum;
    }

    private static long AddSat(long a, long b)
    {
        if (b <= 0) return a;
        if (a > long.MaxValue - b) return long.MaxValue;
        return a + b;
    }

    private static void WritePlan(StreamWriter writer, PlanNode n, List<Orientation> orientations)
    {
        WriteNode(writer, n, orientations);
        if (n.ChildX is not null) WritePlan(writer, n.ChildX, orientations);
        if (n.ChildY is not null) WritePlan(writer, n.ChildY, orientations);
        if (n.ChildZ is not null) WritePlan(writer, n.ChildZ, orientations);
    }

    private static void WriteNode(StreamWriter writer, PlanNode n, List<Orientation> orientations)
    {
        Orientation o = orientations[n.Ori];
        string qw = o.Qw.ToString("G17", CultureInfo.InvariantCulture);
        string qx = o.Qx.ToString("G17", CultureInfo.InvariantCulture);
        string qy = o.Qy.ToString("G17", CultureInfo.InvariantCulture);
        string qz = o.Qz.ToString("G17", CultureInfo.InvariantCulture);

        double baseTx = n.Region.Ox - o.MinX;
        double baseTy = n.Region.Oy - o.MinY;
        double baseTz = n.Region.Oz - o.MinZ;

        double stepX = o.SizeX;
        double stepY = o.SizeY;
        double stepZ = o.SizeZ;

        // Loop order chosen to stream cache-friendly and avoid excessive string allocations.
        for (long iz = 0; iz < n.Nz; iz++)
        {
            double tz = baseTz + iz * stepZ;
            for (long iy = 0; iy < n.Ny; iy++)
            {
                double ty = baseTy + iy * stepY;
                for (long ix = 0; ix < n.Nx; ix++)
                {
                    double tx = baseTx + ix * stepX;
                    writer.Write(tx.ToString("G17", CultureInfo.InvariantCulture));
                    writer.Write(' ');
                    writer.Write(ty.ToString("G17", CultureInfo.InvariantCulture));
                    writer.Write(' ');
                    writer.Write(tz.ToString("G17", CultureInfo.InvariantCulture));
                    writer.Write(' ');
                    writer.Write(qw);
                    writer.Write(' ');
                    writer.Write(qx);
                    writer.Write(' ');
                    writer.Write(qy);
                    writer.Write(' ');
                    writer.Write(qz);
                    writer.Write('\n');
                }
            }
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        public bool TryNextInt(out int value)
        {
            value = 0;
            int c = ReadNonSpace();
            if (c < 0) return false;
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int v = 0;
            while (c >= '0' && c <= '9')
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            value = v * sign;
            return true;
        }

        public int NextInt()
        {
            if (!TryNextInt(out int v)) return 0;
            return v;
        }

        public double NextDouble()
        {
            int c = ReadNonSpace();
            if (c < 0) return 0.0;

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            double val = 0.0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10.0 + (c - '0');
                c = Read();
            }

            if (c == '.')
            {
                c = Read();
                double place = 0.1;
                while (c >= '0' && c <= '9')
                {
                    val += (c - '0') * place;
                    place *= 0.1;
                    c = Read();
                }
            }

            if (c == 'e' || c == 'E')
            {
                c = Read();
                int esign = 1;
                if (c == '-')
                {
                    esign = -1;
                    c = Read();
                }
                else if (c == '+')
                {
                    c = Read();
                }

                int exp = 0;
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                val *= Math.Pow(10.0, esign * exp);
            }

            return sign * val;
        }

        private int ReadNonSpace()
        {
            int c;
            do { c = Read(); } while (c >= 0 && c <= ' ');
            return c;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }
    }

    private struct XorShift64
    {
        private ulong _state;
        public XorShift64(uint seed)
        {
            _state = seed == 0 ? 0x9E3779B97F4A7C15UL : seed;
            // Scramble a bit.
            NextULong();
            NextULong();
        }

        private ulong NextULong()
        {
            ulong x = _state;
            x ^= x << 7;
            x ^= x >> 9;
            _state = x;
            return x;
        }

        public double NextDouble01()
        {
            // 53-bit precision in [0,1).
            ulong u = NextULong();
            u = (u >> 11) | 1UL; // avoid exact 0
            return (u & ((1UL << 53) - 1)) * (1.0 / (1UL << 53));
        }
    }
}
