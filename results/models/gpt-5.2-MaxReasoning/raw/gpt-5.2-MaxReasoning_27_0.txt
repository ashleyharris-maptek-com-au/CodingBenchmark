// Fast buffered input for large graphs
struct FastInput {
  static constexpr size_t BUFSIZE = 1u << 20;
  char buf[BUFSIZE];
  size_t idx = 0, size = 0;

  inline char read() {
    if (idx >= size) {
      size = std::fread(buf, 1, BUFSIZE, stdin);
      idx = 0;
      if (size == 0) return 0;
    }
    return buf[idx++];
  }

  bool readInt(int& out) {
    char c = 0;
    do {
      c = read();
      if (!c) return false;
    } while (c <= ' ');
    int sign = 1;
    if (c == '-') {
      sign = -1;
      c = read();
    }
    int x = 0;
    while (c > ' ') {
      x = x * 10 + (c - '0');
      c = read();
    }
    out = x * sign;
    return true;
  }
};

// Fast buffered output
struct FastOutput {
  static constexpr size_t BUFSIZE = 1u << 20;
  char buf[BUFSIZE];
  size_t idx = 0;

  ~FastOutput() { flush(); }

  inline void flush() {
    if (idx) {
      std::fwrite(buf, 1, idx, stdout);
      idx = 0;
    }
  }

  inline void pushChar(char c) {
    if (idx >= BUFSIZE) flush();
    buf[idx++] = c;
  }

  inline void writeInt(int x, char after) {
    if (x == 0) {
      pushChar('0');
      pushChar(after);
      return;
    }
    if (x < 0) {
      pushChar('-');
      x = -x;
    }
    char s[16];
    int n = 0;
    while (x) {
      s[n++] = char('0' + (x % 10));
      x /= 10;
    }
    while (n--) pushChar(s[n]);
    pushChar(after);
  }
};

struct SplitMix64 {
  uint64_t x;
  explicit SplitMix64(uint64_t seed) : x(seed) {}
  inline uint64_t next() {
    uint64_t z = (x += 0x9e3779b97f4a7c15ULL);
    z = (z ^ (z >> 30)) * 0xbf58476d1ce4e5b9ULL;
    z = (z ^ (z >> 27)) * 0x94d049bb133111ebULL;
    return z ^ (z >> 31);
  }
  inline uint32_t nextU32() { return (uint32_t)next(); }
  inline int uniformInt(int bound) { return bound > 0 ? (int)(next() % (uint64_t)bound) : 0; }
};

static inline int ctz64(uint64_t x) {
#if defined(_MSC_VER) && (defined(_M_X64) || defined(_M_ARM64))
  unsigned long r = 0;
  _BitScanForward64(&r, x);
  return (int)r;
#elif defined(_MSC_VER)
  // 32-bit fallback (rare for this benchmark)
  unsigned long r = 0;
  uint32_t lo = (uint32_t)x;
  if (lo) {
    _BitScanForward(&r, lo);
    return (int)r;
  }
  _BitScanForward(&r, (uint32_t)(x >> 32));
  return (int)r + 32;
#else
  return __builtin_ctzll(x);
#endif
}

struct GraphCSR {
  int n = 0;
  std::vector<int> offset;  // size n+1
  std::vector<int> adj;     // size 2*m (undirected)
  std::vector<int> degree;  // size n
  int maxDegree = 0;
};

static GraphCSR readGraphCSR(FastInput& in) {
  GraphCSR g;
  int n = 0, m = 0, k = 0;
  if (!in.readInt(n)) return g;
  in.readInt(m);
  in.readInt(k);  // consumed here; caller separately stores
  g.n = n;

  std::vector<int> eu;
  std::vector<int> ev;
  eu.reserve((size_t)m);
  ev.reserve((size_t)m);

  std::vector<int> deg(n, 0);
  for (int i = 0; i < m; i++) {
    int u = 0, v = 0;
    in.readInt(u);
    in.readInt(v);
    if (u < 0 || v < 0 || u >= n || v >= n) continue;
    if (u == v) {
      // Keep the self-loop; it makes the instance unsatisfiable, but we store it
      // so any checker will still see it.
      deg[u]++;
      eu.push_back(u);
      ev.push_back(v);
      continue;
    }
    deg[u]++;
    deg[v]++;
    eu.push_back(u);
    ev.push_back(v);
  }

  g.offset.assign(n + 1, 0);
  for (int i = 0; i < n; i++) g.offset[i + 1] = g.offset[i] + deg[i];
  g.adj.assign((size_t)g.offset[n], 0);
  std::vector<int> next = g.offset;
  for (size_t i = 0; i < eu.size(); i++) {
    int u = eu[i], v = ev[i];
    g.adj[next[u]++] = v;
    if (u != v) g.adj[next[v]++] = u;
  }

  g.degree.assign(n, 0);
  int maxDeg = 0;
  for (int i = 0; i < n; i++) {
    int d = g.offset[i + 1] - g.offset[i];
    g.degree[i] = d;
    if (d > maxDeg) maxDeg = d;
  }
  g.maxDegree = maxDeg;
  return g;
}

static std::vector<int> orderByDegreeDesc(const GraphCSR& g, SplitMix64& rng, bool randomTies) {
  std::vector<int> order(g.n);
  for (int i = 0; i < g.n; i++) order[i] = i;
  if (!randomTies) {
    std::sort(order.begin(), order.end(), [&](int a, int b) {
      int da = g.degree[a], db = g.degree[b];
      if (da != db) return da > db;
      return a < b;
    });
    return order;
  }

  // Randomize tie-breaking without re-sorting every attempt: stable sort by degree,
  // then shuffle within equal-degree buckets.
  std::sort(order.begin(), order.end(), [&](int a, int b) {
    int da = g.degree[a], db = g.degree[b];
    if (da != db) return da > db;
    return a < b;
  });
  size_t i = 0;
  while (i < order.size()) {
    size_t j = i + 1;
    while (j < order.size() && g.degree[order[j]] == g.degree[order[i]]) j++;
    for (size_t t = i; t + 1 < j; t++) {
      size_t r = t + (size_t)(rng.next() % (uint64_t)(j - t));
      std::swap(order[t], order[r]);
    }
    i = j;
  }
  return order;
}

// Smallest-last ordering (degeneracy-based), computed in O(n+m) using buckets.
static std::vector<int> smallestLastOrder(const GraphCSR& g) {
  const int n = g.n;
  std::vector<int> workDeg = g.degree;
  const int maxDeg = g.maxDegree;

  std::vector<int> head(maxDeg + 1, -1);
  std::vector<int> next(n, -1), prev(n, -1);
  std::vector<unsigned char> removed(n, 0);

  auto bucketRemove = [&](int v, int d) {
    int p = prev[v];
    int nx = next[v];
    if (p != -1)
      next[p] = nx;
    else
      head[d] = nx;
    if (nx != -1) prev[nx] = p;
    prev[v] = next[v] = -1;
  };

  auto bucketInsert = [&](int v, int d) {
    int h = head[d];
    next[v] = h;
    prev[v] = -1;
    if (h != -1) prev[h] = v;
    head[d] = v;
  };

  for (int v = 0; v < n; v++) bucketInsert(v, workDeg[v]);

  int minDeg = 0;
  std::vector<int> removal;
  removal.reserve((size_t)n);
  for (int iter = 0; iter < n; iter++) {
    while (minDeg <= maxDeg && head[minDeg] == -1) ++minDeg;
    int v = head[minDeg];
    // Remove v from bucket minDeg (as head)
    head[minDeg] = next[v];
    if (head[minDeg] != -1) prev[head[minDeg]] = -1;
    prev[v] = next[v] = -1;
    removed[v] = 1;
    removal.push_back(v);

    // Decrease degrees of remaining neighbors
    for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
      int u = g.adj[ei];
      if (u < 0 || u >= n) continue;
      if (removed[u]) continue;
      int du = workDeg[u];
      bucketRemove(u, du);
      int du2 = du - 1;
      workDeg[u] = du2;
      bucketInsert(u, du2);
      if (du2 < minDeg) minDeg = du2;
    }
  }

  std::reverse(removal.begin(), removal.end());  // smallest-last ordering
  return removal;
}

static bool greedyStrict(const GraphCSR& g, int k, const std::vector<int>& order, std::vector<int16_t>& color) {
  const int n = g.n;
  color.assign(n, -1);
  if (n == 0) return true;
  if (k <= 0) return false;

  if (k <= 64) {
    uint64_t fullMask = (k == 64) ? ~0ULL : ((1ULL << k) - 1ULL);
    for (int v : order) {
      uint64_t used = 0;
      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        int c = (int)color[u];
        if (c >= 0) used |= (1ULL << (unsigned)c);
      }
      uint64_t avail = fullMask & ~used;
      if (!avail) return false;
      int chosen = ctz64(avail);
      color[v] = (int16_t)chosen;
    }
    return true;
  }

  std::vector<int> lastSeen((size_t)k, -1);
  for (int v : order) {
    for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
      int u = g.adj[ei];
      int c = (int)color[u];
      if (c >= 0 && c < k) lastSeen[(size_t)c] = v;
    }
    int chosen = -1;
    for (int c = 0; c < k; c++) {
      if (lastSeen[(size_t)c] != v) {
        chosen = c;
        break;
      }
    }
    if (chosen < 0) return false;
    color[v] = (int16_t)chosen;
  }
  return true;
}

static void greedyAllowConflicts(const GraphCSR& g, int k, const std::vector<int>& order, SplitMix64& rng,
                                std::vector<int16_t>& color) {
  const int n = g.n;
  color.assign(n, -1);
  if (n == 0) return;

  if (k <= 64) {
    uint64_t fullMask = (k == 64) ? ~0ULL : ((1ULL << k) - 1ULL);
    for (int v : order) {
      uint64_t used = 0;
      int counts[64];
      for (int i = 0; i < k; i++) counts[i] = 0;
      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        int c = (int)color[u];
        if (c >= 0) {
          used |= (1ULL << (unsigned)c);
          counts[c]++;
        }
      }
      uint64_t avail = fullMask & ~used;
      if (avail) {
        int chosen = ctz64(avail);
        color[v] = (int16_t)chosen;
      } else {
        int bestColor = rng.uniformInt(k);
        int bestVal = counts[bestColor];
        int seen = 1;
        for (int c = 0; c < k; c++) {
          int val = counts[c];
          if (val < bestVal) {
            bestVal = val;
            bestColor = c;
            seen = 1;
          } else if (val == bestVal) {
            if ((int)(rng.nextU32() % (uint32_t)(++seen)) == 0) bestColor = c;
          }
        }
        color[v] = (int16_t)bestColor;
      }
    }
    return;
  }

  std::vector<int> counts((size_t)k);
  for (int v : order) {
    std::fill(counts.begin(), counts.end(), 0);
    std::vector<int> used((size_t)k, 0);
    for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
      int u = g.adj[ei];
      int c = (int)color[u];
      if (c >= 0 && c < k) {
        used[(size_t)c] = 1;
        counts[(size_t)c]++;
      }
    }
    int chosen = -1;
    for (int c = 0; c < k; c++) {
      if (!used[(size_t)c]) {
        chosen = c;
        break;
      }
    }
    if (chosen >= 0) {
      color[v] = (int16_t)chosen;
    } else {
      int bestColor = rng.uniformInt(k);
      int bestVal = counts[(size_t)bestColor];
      int seen = 1;
      for (int c = 0; c < k; c++) {
        int val = counts[(size_t)c];
        if (val < bestVal) {
          bestVal = val;
          bestColor = c;
          seen = 1;
        } else if (val == bestVal) {
          if ((int)(rng.nextU32() % (uint32_t)(++seen)) == 0) bestColor = c;
        }
      }
      color[v] = (int16_t)bestColor;
    }
  }
}

static bool tryBipartite2Coloring(const GraphCSR& g, std::vector<int16_t>& colorOut) {
  const int n = g.n;
  colorOut.assign(n, -1);
  std::vector<int> q;
  q.reserve((size_t)std::min<int>(n, 1'000'000));

  for (int s = 0; s < n; s++) {
    if (colorOut[s] != -1) continue;
    colorOut[s] = 0;
    q.clear();
    q.push_back(s);
    size_t head = 0;
    while (head < q.size()) {
      int v = q[head++];
      int cv = (int)colorOut[v];
      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        if (colorOut[u] == -1) {
          colorOut[u] = (int16_t)(cv ^ 1);
          q.push_back(u);
        } else if ((int)colorOut[u] == cv) {
          return false;
        }
      }
    }
  }
  return true;
}

static bool tabuSearch(const GraphCSR& g, int k, SplitMix64& rng, std::vector<int16_t>& color,
                      std::chrono::steady_clock::time_point deadline) {
  const int n = g.n;
  if (n == 0) return true;
  if (k <= 1) return false;

  std::vector<int> conflicts(n, 0);
  std::vector<int> pos(n, -1);
  std::vector<int> conflictList;
  conflictList.reserve((size_t)std::min<int>(n, 1'000'000));

  long long totalConf = 0;
  for (int v = 0; v < n; v++) {
    int cv = (int)color[v];
    for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
      int u = g.adj[ei];
      if (u > v && (int)color[u] == cv) {
        totalConf++;
        conflicts[v]++;
        conflicts[u]++;
      }
    }
  }
  for (int v = 0; v < n; v++) {
    if (conflicts[v] > 0) {
      pos[v] = (int)conflictList.size();
      conflictList.push_back(v);
    }
  }
  if (totalConf == 0) return true;

  auto updateMembership = [&](int v) {
    if (conflicts[v] > 0) {
      if (pos[v] == -1) {
        pos[v] = (int)conflictList.size();
        conflictList.push_back(v);
      }
    } else {
      int p = pos[v];
      if (p != -1) {
        int last = conflictList.back();
        conflictList[(size_t)p] = last;
        pos[last] = p;
        conflictList.pop_back();
        pos[v] = -1;
      }
    }
  };

  std::vector<int16_t> tabuColor(n, -1);
  std::vector<int> tabuUntil(n, 0);

  long long bestTotal = totalConf;
  int iter = 0;

  std::vector<int> countsDyn;
  if (k > 64) countsDyn.resize((size_t)k);

  while (totalConf > 0) {
    if ((iter & 255) == 0) {
      if (std::chrono::steady_clock::now() >= deadline) break;
      if (conflictList.empty()) break;
    }
    iter++;
    int v = conflictList[(size_t)rng.uniformInt((int)conflictList.size())];

    int oldC = (int)color[v];
    int oldConf = conflicts[v];

    int bestC = -1;
    int bestVal = std::numeric_limits<int>::max();
    int seen = 0;

    auto considerColor = [&](int c, int val) {
      if (c == oldC) return;
      bool isTabu = (tabuUntil[v] > iter && (int)tabuColor[v] == c);
      long long newTotal = totalConf + (long long)(val - oldConf);
      if (isTabu && newTotal >= bestTotal) return;
      if (val < bestVal) {
        bestVal = val;
        bestC = c;
        seen = 1;
      } else if (val == bestVal) {
        if ((int)(rng.nextU32() % (uint32_t)(++seen)) == 0) bestC = c;
      }
    };

    if (k <= 64) {
      int cnt[64];
      for (int i = 0; i < k; i++) cnt[i] = 0;
      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        int cu = (int)color[u];
        if (0 <= cu && cu < k) cnt[cu]++;
      }
      for (int c = 0; c < k; c++) considerColor(c, cnt[c]);

      if (bestC < 0) {
        // All moves tabu without aspiration; ignore tabu
        bestVal = std::numeric_limits<int>::max();
        seen = 0;
        for (int c = 0; c < k; c++) {
          if (c == oldC) continue;
          int val = cnt[c];
          if (val < bestVal) {
            bestVal = val;
            bestC = c;
            seen = 1;
          } else if (val == bestVal) {
            if ((int)(rng.nextU32() % (uint32_t)(++seen)) == 0) bestC = c;
          }
        }
      }

      // Apply move v: oldC -> bestC
      int tenure = 7 + (int)(rng.nextU32() % 10u) + (int)((conflictList.size() * 3u) / 100u);
      if (tenure > 250) tenure = 250 + (int)(rng.nextU32() % 50u);
      tabuColor[v] = (int16_t)oldC;
      tabuUntil[v] = iter + tenure;

      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        int cu = (int)color[u];
        if (cu == oldC) {
          conflicts[u]--;
          updateMembership(u);
        } else if (cu == bestC) {
          conflicts[u]++;
          updateMembership(u);
        }
      }
      color[v] = (int16_t)bestC;
      conflicts[v] = bestVal;
      updateMembership(v);

      totalConf += (long long)(bestVal - oldConf);
    } else {
      std::fill(countsDyn.begin(), countsDyn.end(), 0);
      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        int cu = (int)color[u];
        if (0 <= cu && cu < k) countsDyn[(size_t)cu]++;
      }
      for (int c = 0; c < k; c++) considerColor(c, countsDyn[(size_t)c]);
      if (bestC < 0) {
        bestVal = std::numeric_limits<int>::max();
        seen = 0;
        for (int c = 0; c < k; c++) {
          if (c == oldC) continue;
          int val = countsDyn[(size_t)c];
          if (val < bestVal) {
            bestVal = val;
            bestC = c;
            seen = 1;
          } else if (val == bestVal) {
            if ((int)(rng.nextU32() % (uint32_t)(++seen)) == 0) bestC = c;
          }
        }
      }

      int tenure = 7 + (int)(rng.nextU32() % 10u) + (int)((conflictList.size() * 3u) / 100u);
      if (tenure > 250) tenure = 250 + (int)(rng.nextU32() % 50u);
      tabuColor[v] = (int16_t)oldC;
      tabuUntil[v] = iter + tenure;

      for (int ei = g.offset[v]; ei < g.offset[v + 1]; ei++) {
        int u = g.adj[ei];
        int cu = (int)color[u];
        if (cu == oldC) {
          conflicts[u]--;
          updateMembership(u);
        } else if (cu == bestC) {
          conflicts[u]++;
          updateMembership(u);
        }
      }
      color[v] = (int16_t)bestC;
      conflicts[v] = bestVal;
      updateMembership(v);
      totalConf += (long long)(bestVal - oldConf);
    }

    if (totalConf < bestTotal) bestTotal = totalConf;
  }

  return totalConf == 0;
}

int main() {
  FastInput in;
  int n = 0, m = 0, k = 0;
  if (!in.readInt(n)) return 0;
  in.readInt(m);
  in.readInt(k);

  // Rewind: readGraphCSR expects to read n,m,k itself. We already consumed them.
  // So we reconstruct by reading edges manually here.
  // (This keeps FastInput simple and avoids storing the whole input.)
  GraphCSR g;
  g.n = n;
  std::vector<int> eu;
  std::vector<int> ev;
  eu.reserve((size_t)m);
  ev.reserve((size_t)m);
  std::vector<int> deg(n, 0);
  for (int i = 0; i < m; i++) {
    int u = 0, v = 0;
    in.readInt(u);
    in.readInt(v);
    if (u < 0 || v < 0 || u >= n || v >= n) continue;
    if (u == v) {
      deg[u]++;
      eu.push_back(u);
      ev.push_back(v);
      continue;
    }
    deg[u]++;
    deg[v]++;
    eu.push_back(u);
    ev.push_back(v);
  }

  g.offset.assign(n + 1, 0);
  for (int i = 0; i < n; i++) g.offset[i + 1] = g.offset[i] + deg[i];
  g.adj.assign((size_t)g.offset[n], 0);
  std::vector<int> next = g.offset;
  for (size_t i = 0; i < eu.size(); i++) {
    int u = eu[i], v = ev[i];
    g.adj[next[u]++] = v;
    if (u != v) g.adj[next[v]++] = u;
  }

  g.degree.assign(n, 0);
  int maxDeg = 0;
  for (int i = 0; i < n; i++) {
    int d = g.offset[i + 1] - g.offset[i];
    g.degree[i] = d;
    if (d > maxDeg) maxDeg = d;
  }
  g.maxDegree = maxDeg;

  // Seed RNG
  uint64_t seed = (uint64_t)std::chrono::high_resolution_clock::now().time_since_epoch().count();
  seed ^= (uint64_t)n * 0x9e3779b97f4a7c15ULL;
  seed ^= (uint64_t)m * 0xbf58476d1ce4e5b9ULL;
  seed ^= (uint64_t)k * 0x94d049bb133111ebULL;
  SplitMix64 rng(seed);

  std::vector<int16_t> color;
  color.reserve((size_t)n);

  if (n == 0) return 0;
  if (k <= 0) {
    // Degenerate: output zeros
    color.assign(n, 0);
  } else if (m == 0) {
    color.assign(n, 0);
  } else if (k == 1) {
    // Only possible for edgeless graphs; fall back to zeros.
    color.assign(n, 0);
  } else if (k == 2) {
    if (!tryBipartite2Coloring(g, color)) {
      // Likely unsatisfiable, but still attempt a repair search.
      auto order = orderByDegreeDesc(g, rng, true);
      greedyAllowConflicts(g, k, order, rng, color);
      auto deadline = std::chrono::steady_clock::now() + std::chrono::seconds(290);
      tabuSearch(g, k, rng, color, deadline);
    }
  } else {
    auto globalDeadline = std::chrono::steady_clock::now() + std::chrono::seconds(290);

    // Strategy 1 (fast, good on many graphs): Welsh-Powell (degree-desc) greedy
    // Guaranteed to succeed if k >= maxDegree+1.
    {
      auto order = orderByDegreeDesc(g, rng, false);
      if (greedyStrict(g, k, order, color)) {
        // success
      } else {
        // Strategy 2: smallest-last (degeneracy) greedy
        auto sl = smallestLastOrder(g);
        if (greedyStrict(g, k, sl, color)) {
          // success
        } else {
          // Strategy 3: repair search (TabuCol-style), with a few restarts
          bool ok = false;

          // Attempt A: constructive + tabu from degree order
          {
            auto orderA = orderByDegreeDesc(g, rng, true);
            greedyAllowConflicts(g, k, orderA, rng, color);
            auto slice = std::min(globalDeadline, std::chrono::steady_clock::now() + std::chrono::seconds(60));
            ok = tabuSearch(g, k, rng, color, slice);
          }

          // Attempt B: constructive + tabu from smallest-last order
          if (!ok && std::chrono::steady_clock::now() < globalDeadline) {
            auto orderB = sl;  // already computed
            greedyAllowConflicts(g, k, orderB, rng, color);
            auto slice = std::min(globalDeadline, std::chrono::steady_clock::now() + std::chrono::seconds(90));
            ok = tabuSearch(g, k, rng, color, slice);
          }

          // Attempt C+: random restarts until time is up
          int restart = 0;
          while (!ok && std::chrono::steady_clock::now() < globalDeadline) {
            std::vector<int> randOrder(n);
            for (int i = 0; i < n; i++) randOrder[i] = i;
            for (int i = 0; i + 1 < n; i++) {
              int j = i + rng.uniformInt(n - i);
              std::swap(randOrder[i], randOrder[j]);
            }
            greedyAllowConflicts(g, k, randOrder, rng, color);
            auto slice = std::min(globalDeadline, std::chrono::steady_clock::now() + std::chrono::seconds(30));
            ok = tabuSearch(g, k, rng, color, slice);
            restart++;
            if (restart >= 6) break;
          }
        }
      }
    }
  }

  FastOutput out;
  for (int i = 0; i < n; i++) {
    int c = (int)color[i];
    if (c < 0) c = 0;
    if (k > 0 && c >= k) c %= k;
    out.writeInt(c, (i + 1 == n) ? '\n' : ' ');
  }
  out.flush();
  return 0;
}
