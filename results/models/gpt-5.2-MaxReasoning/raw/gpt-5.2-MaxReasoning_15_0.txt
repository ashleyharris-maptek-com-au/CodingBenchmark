use std::cmp::Ordering;
use std::io::{self, BufWriter, Read, Write};

#[derive(Clone, Copy)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn new(x: f64, y: f64) -> Self {
        Self { x, y }
    }

    fn sub(self, other: Self) -> Self {
        Self::new(self.x - other.x, self.y - other.y)
    }

    fn len(self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }
}

#[derive(Clone, Copy)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

impl Vec3 {
    fn new(x: f64, y: f64, z: f64) -> Self {
        Self { x, y, z }
    }

    fn add(self, other: Self) -> Self {
        Self::new(self.x + other.x, self.y + other.y, self.z + other.z)
    }

    fn mul(self, s: f64) -> Self {
        Self::new(self.x * s, self.y * s, self.z * s)
    }

    fn dot(self, other: Self) -> f64 {
        self.x * other.x + self.y * other.y + self.z * other.z
    }

    fn cross(self, other: Self) -> Self {
        Self::new(
            self.y * other.z - self.z * other.y,
            self.z * other.x - self.x * other.z,
            self.x * other.y - self.y * other.x,
        )
    }

    fn norm(self) -> f64 {
        self.dot(self).sqrt()
    }

    fn normalize(self) -> Self {
        let n = self.norm();
        if n <= 0.0 {
            return Self::new(0.0, 0.0, 0.0);
        }
        self.mul(1.0 / n)
    }
}

#[derive(Clone, Copy)]
struct Quat {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quat {
    fn identity() -> Self {
        Self {
            w: 1.0,
            x: 0.0,
            y: 0.0,
            z: 0.0,
        }
    }

    fn from_axis_angle(axis: Vec3, angle: f64) -> Self {
        let axis_n = axis.normalize();
        let half = 0.5 * angle;
        let (s, c) = half.sin_cos();
        Self {
            w: c,
            x: axis_n.x * s,
            y: axis_n.y * s,
            z: axis_n.z * s,
        }
        .normalize()
    }

    fn normalize(self) -> Self {
        let n2 = self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z;
        if n2 <= 0.0 {
            return Self::identity();
        }
        let inv = 1.0 / n2.sqrt();
        Self {
            w: self.w * inv,
            x: self.x * inv,
            y: self.y * inv,
            z: self.z * inv,
        }
    }

    fn rotate_vec3(self, v: Vec3) -> Vec3 {
        // v' = v + 2*w*(q_vec x v) + 2*(q_vec x (q_vec x v))
        let qv = Vec3::new(self.x, self.y, self.z);
        let t = qv.cross(v).mul(2.0);
        v.add(t.mul(self.w)).add(qv.cross(t))
    }
}

fn cross2(a: Vec2, b: Vec2) -> f64 {
    a.x * b.y - a.y * b.x
}

fn hull_ccw(points: &[Vec2]) -> Vec<Vec2> {
    let mut pts: Vec<Vec2> = points.to_vec();
    pts.sort_by(|a, b| match a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal) {
        Ordering::Equal => a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal),
        o => o,
    });

    let eps = 1e-12_f64;
    pts.dedup_by(|a, b| (a.x - b.x).abs() <= eps && (a.y - b.y).abs() <= eps);
    if pts.len() <= 2 {
        return pts;
    }

    let mut lower: Vec<Vec2> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let n = lower.len();
            let a = lower[n - 2];
            let b = lower[n - 1];
            let ab = b.sub(a);
            let bc = p.sub(b);
            if cross2(ab, bc) <= eps {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }

    let mut upper: Vec<Vec2> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let n = upper.len();
            let a = upper[n - 2];
            let b = upper[n - 1];
            let ab = b.sub(a);
            let bc = p.sub(b);
            if cross2(ab, bc) <= eps {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }

    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn polygon_centroid(vertices: &[Vec2]) -> Vec2 {
    let mut sx = 0.0;
    let mut sy = 0.0;
    let n = vertices.len().max(1) as f64;
    for &p in vertices {
        sx += p.x;
        sy += p.y;
    }
    Vec2::new(sx / n, sy / n)
}

fn signed_inradius_at_point(hull: &[Vec2], p: Vec2) -> f64 {
    if hull.len() < 3 {
        return -1.0;
    }
    let mut best = f64::INFINITY;
    for i in 0..hull.len() {
        let a = hull[i];
        let b = hull[(i + 1) % hull.len()];
        let e = b.sub(a);
        let elen = e.len();
        if elen <= 1e-15 {
            continue;
        }
        let d = cross2(e, p.sub(a)) / elen;
        if d < best {
            best = d;
        }
    }
    best
}

struct Rng64 {
    state: u64,
}

impl Rng64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    fn next_u64(&mut self) -> u64 {
        const A: u64 = 6364136223846793005;
        const C: u64 = 1442695040888963407;
        self.state = self.state.wrapping_mul(A).wrapping_add(C);
        self.state
    }

    fn next_f64(&mut self) -> f64 {
        // [0,1)
        let v = self.next_u64() >> 11;
        (v as f64) * (1.0 / ((1_u64 << 53) as f64))
    }
}

fn best_orientation(
    dir_over_z: Vec2,
    samples: usize,
    seed: u64,
    ref_verts: &[Vec3; 4],
) -> (Quat, Vec2, f64) {
    let mut best_q = Quat::identity();
    let mut best_c = Vec2::new(0.0, 0.0);
    let mut best_r = -1.0;

    let mut consider = |q: Quat| {
        let mut proj: [Vec2; 4] = [Vec2::new(0.0, 0.0); 4];
        for (i, &rv) in ref_verts.iter().enumerate() {
            let r = q.rotate_vec3(rv);
            let x = r.x - r.z * dir_over_z.x;
            let y = r.y - r.z * dir_over_z.y;
            proj[i] = Vec2::new(x, y);
        }
        let hull = hull_ccw(&proj);
        if hull.len() < 3 {
            return;
        }
        let c = polygon_centroid(&hull);
        let r0 = signed_inradius_at_point(&hull, c);
        if r0.is_finite() && r0 > best_r {
            best_q = q.normalize();
            best_c = c;
            best_r = r0;
        }
    };

    // Deterministic baseline rotations.
    let pi = std::f64::consts::PI;
    consider(Quat::identity());
    consider(Quat::from_axis_angle(Vec3::new(1.0, 0.0, 0.0), 0.5 * pi));
    consider(Quat::from_axis_angle(Vec3::new(0.0, 1.0, 0.0), 0.5 * pi));
    consider(Quat::from_axis_angle(Vec3::new(0.0, 0.0, 1.0), 0.5 * pi));
    consider(Quat::from_axis_angle(Vec3::new(1.0, 0.0, 0.0), pi));
    consider(Quat::from_axis_angle(Vec3::new(0.0, 1.0, 0.0), pi));
    consider(Quat::from_axis_angle(Vec3::new(0.0, 0.0, 1.0), pi));

    // Random-ish sampling for improved robustness.
    let mut rng = Rng64::new(seed);
    for _ in 0..samples {
        let u1 = rng.next_f64();
        let u2 = rng.next_f64();
        let u3 = rng.next_f64();

        let theta = 2.0 * pi * u1;
        let z = 2.0 * u2 - 1.0;
        let r = (1.0 - z * z).max(0.0).sqrt();
        let (st, ct) = theta.sin_cos();
        let axis = Vec3::new(r * ct, r * st, z);

        let angle = 2.0 * pi * u3;
        consider(Quat::from_axis_angle(axis, angle));
    }

    (best_q, best_c, best_r)
}

fn base_center_z(scale: f64, dz: f64) -> f64 {
    let min_vertex_z = 0.7071067811865476 * scale;
    let margin = 10.0;
    if dz < 0.0 {
        min_vertex_z + margin
    } else if dz > 0.0 {
        -(min_vertex_z + margin)
    } else {
        min_vertex_z + margin
    }
}

fn bound_radius(scale: f64) -> f64 {
    // max |vertex| for reference tetrahedron is sqrt(1.5)
    1.224744871391589 * scale
}

fn generate_centers(min_x: f64, max_x: f64, min_y: f64, max_y: f64, r: f64) -> Vec<Vec2> {
    let mut out: Vec<Vec2> = Vec::new();
    if !r.is_finite() || r <= 0.0 {
        out.push(Vec2::new(0.5 * (min_x + max_x), 0.5 * (min_y + max_y)));
        return out;
    }

    let w = (max_x - min_x).abs();
    let h = (max_y - min_y).abs();

    if w <= 2.0 * r && h <= 2.0 * r {
        out.push(Vec2::new(0.5 * (min_x + max_x), 0.5 * (min_y + max_y)));
        return out;
    }

    if w <= 2.0 * r {
        let x = 0.5 * (min_x + max_x);
        let half_w = 0.5 * w;
        let eff = (r * r - half_w * half_w).max(0.0).sqrt();
        let step = (2.0 * eff * 0.999).max(r * 0.25);
        let mut y = min_y - eff;
        while y <= max_y + eff + 1e-12 {
            out.push(Vec2::new(x, y));
            y += step;
        }
        return out;
    }

    if h <= 2.0 * r {
        let y = 0.5 * (min_y + max_y);
        let half_h = 0.5 * h;
        let eff = (r * r - half_h * half_h).max(0.0).sqrt();
        let step = (2.0 * eff * 0.999).max(r * 0.25);
        let mut x = min_x - eff;
        while x <= max_x + eff + 1e-12 {
            out.push(Vec2::new(x, y));
            x += step;
        }
        return out;
    }

    // Hexagonal covering grid.
    let dx = (3.0_f64).sqrt() * r * 0.999;
    let dy = 1.5 * r * 0.999;
    let y_start = min_y - r;
    let y_end = max_y + r;
    let x_start = min_x - r - dx;
    let x_end = max_x + r + dx;

    let mut row = 0usize;
    let mut y = y_start;
    while y <= y_end + 1e-12 {
        let x_offset = if row % 2 == 0 { 0.0 } else { 0.5 * dx };
        let mut x = x_start + x_offset;
        while x <= x_end + 1e-12 {
            out.push(Vec2::new(x, y));
            x += dx;
        }
        row += 1;
        y += dy;
    }
    out
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;

    for _ in 0..nv {
        let x: f64 = it.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let y: f64 = it.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        min_x = min_x.min(x);
        max_x = max_x.max(x);
        min_y = min_y.min(y);
        max_y = max_y.max(y);
    }

    let sx: f64 = it.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
    let sy: f64 = it.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
    let mut sz: f64 = it.next().and_then(|s| s.parse().ok()).unwrap_or(-1.0);

    if nv == 0 || !min_x.is_finite() || !min_y.is_finite() {
        let mut out = BufWriter::new(io::stdout().lock());
        writeln!(out, "0").unwrap();
        return;
    }

    if sz.abs() < 1e-12 {
        sz = if sz.is_sign_negative() { -1e-12 } else { 1e-12 };
    }

    let d = Vec3::new(sx, sy, sz);
    let mut u = d.normalize();
    if u.norm() <= 0.0 {
        u = Vec3::new(0.0, 0.0, -1.0);
    }

    let dir_over_z = Vec2::new(sx / sz, sy / sz);

    let sqrt2_inv = 0.7071067811865476_f64;
    let ref_verts: [Vec3; 4] = [
        Vec3::new(1.0, 0.0, -sqrt2_inv),
        Vec3::new(-1.0, 0.0, -sqrt2_inv),
        Vec3::new(0.0, 1.0, sqrt2_inv),
        Vec3::new(0.0, -1.0, sqrt2_inv),
    ];

    let samples = if nv > 50_000 {
        12
    } else if nv > 5_000 {
        24
    } else if nv > 500 {
        40
    } else {
        64
    };

    let seed = (nv as u64)
        ^ (sx.to_bits().rotate_left(17))
        ^ (sy.to_bits().rotate_left(33))
        ^ (sz.to_bits().rotate_left(49));

    let (rot, shadow_centroid_unit, inr_unit) = best_orientation(dir_over_z, samples, seed, &ref_verts);

    let scale = 10.0_f64;
    let mut inr = inr_unit;
    let mut centroid = shadow_centroid_unit;
    let mut rotation = rot;
    if !inr.is_finite() || inr <= 1e-9 {
        rotation = Quat::identity();
        centroid = Vec2::new(0.0, 0.0);
        inr = 0.5;
    }

    let r_disk = (inr * scale).max(1e-6);
    let centers = generate_centers(min_x, max_x, min_y, max_y, r_disk);

    let z0 = base_center_z(scale, sz);
    let sep = 3.0 * bound_radius(scale) + 1.0;
    let sep_dir = u.mul(-1.0);

    let mut out = BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", centers.len()).unwrap();

    for (i, g) in centers.iter().enumerate() {
        let tx = g.x - scale * centroid.x + z0 * dir_over_z.x;
        let ty = g.y - scale * centroid.y + z0 * dir_over_z.y;
        let base = Vec3::new(tx, ty, z0);
        let off = sep_dir.mul(sep * (i as f64));
        let c = base.add(off);
        writeln!(
            out,
            "{} {} {} {} {} {} {} {}",
            c.x, c.y, c.z, rotation.w, rotation.x, rotation.y, rotation.z, scale
        )
        .unwrap();
    }
}
