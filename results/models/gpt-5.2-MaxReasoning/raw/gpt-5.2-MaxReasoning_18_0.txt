struct Edge {
  uint32_t u;
  uint32_t v;
};

class FastScanner {
public:
  explicit FastScanner(std::FILE* in = stdin) : in_(in) {}

  bool readU64(uint64_t& out) {
    out = 0;
    int c = readCharSkipWS();
    if (c == EOF) return false;
    while (c >= '0' && c <= '9') {
      out = out * 10 + static_cast<uint64_t>(c - '0');
      c = readChar();
    }
    return true;
  }

private:
  static constexpr size_t kBufSize = 1u << 20;
  std::FILE* in_;
  unsigned char buf_[kBufSize];
  size_t len_ = 0;
  size_t pos_ = 0;

  int readChar() {
    if (pos_ >= len_) {
      len_ = std::fread(buf_, 1, kBufSize, in_);
      pos_ = 0;
      if (len_ == 0) return EOF;
    }
    return buf_[pos_++];
  }

  int readCharSkipWS() {
    int c;
    do {
      c = readChar();
      if (c == EOF) return EOF;
    } while (c <= ' ');
    return c;
  }
};

class FastOutput {
public:
  ~FastOutput() { flush(); }

  void writeU64(uint64_t x) {
    char s[32];
    int n = 0;
    do {
      s[n++] = static_cast<char>('0' + (x % 10));
      x /= 10;
    } while (x);
    while (n--) pushChar(s[n]);
  }

  void writeU32(uint32_t x) { writeU64(x); }

  void writeChar(char c) { pushChar(c); }

  void writeStr(const char* s) {
    while (*s) pushChar(*s++);
  }

  void flush() {
    if (pos_ != 0) {
      std::fwrite(buf_, 1, pos_, stdout);
      pos_ = 0;
    }
  }

private:
  static constexpr size_t kBufSize = 1u << 20;
  char buf_[kBufSize];
  size_t pos_ = 0;

  void pushChar(char c) {
    if (pos_ == kBufSize) flush();
    buf_[pos_++] = c;
  }
};

struct DSU {
  std::vector<uint32_t> parent;
  std::vector<uint32_t> size;

  explicit DSU(uint32_t n) : parent(n), size(n, 1) {
    std::iota(parent.begin(), parent.end(), 0);
  }

  uint32_t find(uint32_t x) {
    while (parent[x] != x) {
      parent[x] = parent[parent[x]];
      x = parent[x];
    }
    return x;
  }

  bool unite(uint32_t a, uint32_t b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (size[a] < size[b]) std::swap(a, b);
    parent[b] = a;
    size[a] += size[b];
    return true;
  }
};

struct Graph {
  uint32_t n = 0;
  std::vector<Edge> edges;
};

struct CutResult {
  uint64_t cut = std::numeric_limits<uint64_t>::max();
  std::vector<char> side; // 0/1 for each vertex in current graph
};

static bool isNonTrivialPartition(const std::vector<char>& side);

static CutResult bruteMinCut(const Graph& g) {
  CutResult best;
  const uint32_t n = g.n;
  if (n < 2) {
    best.cut = 0;
    best.side.assign(n, 0);
    return best;
  }
  const uint32_t fullMask = (n >= 32) ? 0u : ((1u << n) - 1u);
  // n <= 6 in our usage.
  for (uint32_t mask = 1; mask + 1 <= fullMask; ++mask) {
    uint64_t c = 0;
    for (const auto& e : g.edges) {
      const bool a = (mask >> e.u) & 1u;
      const bool b = (mask >> e.v) & 1u;
      if (a != b) ++c;
    }
    if (c < best.cut) {
      best.cut = c;
      best.side.assign(n, 0);
      for (uint32_t i = 0; i < n; ++i) best.side[i] = ((mask >> i) & 1u) ? 1 : 0;
    }
  }
  if (best.side.empty()) {
    best.cut = 0;
    best.side.assign(n, 0);
    best.side[0] = 1;
  }
  return best;
}

struct ContractedGraph {
  Graph g;
  std::vector<uint32_t> mapOldToNew;
};

static ContractedGraph contractToT(const Graph& g, uint32_t t, std::mt19937_64& rng) {
  ContractedGraph out;
  out.mapOldToNew.resize(g.n);
  if (g.n <= t) {
    out.g = g;
    std::iota(out.mapOldToNew.begin(), out.mapOldToNew.end(), 0);
    return out;
  }

  DSU dsu(g.n);
  uint32_t comps = g.n;
  std::uniform_int_distribution<size_t> dist(0, g.edges.empty() ? 0 : (g.edges.size() - 1));
  while (comps > t) {
    const Edge& e = g.edges[dist(rng)];
    if (dsu.unite(e.u, e.v)) --comps;
  }

  std::vector<uint32_t> rootToNew(g.n, std::numeric_limits<uint32_t>::max());
  uint32_t next = 0;
  for (uint32_t i = 0; i < g.n; ++i) {
    const uint32_t r = dsu.find(i);
    uint32_t id = rootToNew[r];
    if (id == std::numeric_limits<uint32_t>::max()) {
      id = next++;
      rootToNew[r] = id;
    }
    out.mapOldToNew[i] = id;
  }

  out.g.n = next;
  out.g.edges.reserve(g.edges.size());
  for (const auto& e : g.edges) {
    const uint32_t a = out.mapOldToNew[e.u];
    const uint32_t b = out.mapOldToNew[e.v];
    if (a != b) out.g.edges.push_back({a, b});
  }
  return out;
}

static std::vector<char> liftPartition(const std::vector<char>& sideNew,
                                       const std::vector<uint32_t>& mapOldToNew) {
  std::vector<char> sideOld(mapOldToNew.size(), 0);
  for (size_t i = 0; i < mapOldToNew.size(); ++i) {
    sideOld[i] = sideNew[mapOldToNew[i]];
  }
  return sideOld;
}

static CutResult kargerSteinRec(const Graph& g, std::mt19937_64& rng) {
  if (g.n <= 6) return bruteMinCut(g);

  const double s2 = std::sqrt(2.0);
  const uint32_t t = static_cast<uint32_t>(std::ceil(1.0 + static_cast<double>(g.n) / s2));

  ContractedGraph c1 = contractToT(g, t, rng);
  CutResult r1 = kargerSteinRec(c1.g, rng);
  r1.side = liftPartition(r1.side, c1.mapOldToNew);

  ContractedGraph c2 = contractToT(g, t, rng);
  CutResult r2 = kargerSteinRec(c2.g, rng);
  r2.side = liftPartition(r2.side, c2.mapOldToNew);

  return (r1.cut <= r2.cut) ? std::move(r1) : std::move(r2);
}

static CutResult kargerStein(const Graph& g, std::mt19937_64& rng, int runs, uint64_t earlyStopAt) {
  CutResult best;
  for (int i = 0; i < runs; ++i) {
    CutResult cur = kargerSteinRec(g, rng);
    if (cur.cut < best.cut) best = std::move(cur);
    if (best.cut <= earlyStopAt) break;
  }
  return best;
}

static CutResult stoerWagnerMatrix(uint32_t n, const std::vector<Edge>& edges) {
  CutResult res;
  if (n < 2) {
    res.cut = 0;
    res.side.assign(n, 0);
    return res;
  }

  std::vector<uint64_t> w(static_cast<size_t>(n) * static_cast<size_t>(n), 0);
  for (const auto& e : edges) {
    if (e.u == e.v) continue;
    w[static_cast<size_t>(e.u) * n + e.v] += 1;
    w[static_cast<size_t>(e.v) * n + e.u] += 1;
  }

  std::vector<uint32_t> active(n);
  std::iota(active.begin(), active.end(), 0);

  std::vector<std::vector<uint32_t>> group(n);
  for (uint32_t i = 0; i < n; ++i) group[i].push_back(i);

  uint64_t best = std::numeric_limits<uint64_t>::max();
  std::vector<uint32_t> bestSide;

  std::vector<uint64_t> weights(n);
  std::vector<char> added(n);

  for (uint32_t phase = n; phase > 1; --phase) {
    std::fill(weights.begin(), weights.end(), 0);
    std::fill(added.begin(), added.end(), 0);

    uint32_t prev = std::numeric_limits<uint32_t>::max();
    uint32_t last = std::numeric_limits<uint32_t>::max();

    for (uint32_t it = 0; it < phase; ++it) {
      uint32_t sel = std::numeric_limits<uint32_t>::max();
      uint64_t selW = 0;
      for (uint32_t j = 0; j < phase; ++j) {
        const uint32_t v = active[j];
        if (added[v]) continue;
        if (sel == std::numeric_limits<uint32_t>::max() || weights[v] > selW) {
          sel = v;
          selW = weights[v];
        }
      }
      added[sel] = 1;
      prev = last;
      last = sel;

      for (uint32_t j = 0; j < phase; ++j) {
        const uint32_t v = active[j];
        if (added[v]) continue;
        weights[v] += w[static_cast<size_t>(sel) * n + v];
      }
    }

    const uint64_t cutOfPhase = weights[last];
    if (cutOfPhase < best) {
      best = cutOfPhase;
      bestSide = group[last];
    }

    // Merge last into prev.
    const uint32_t s = prev;
    const uint32_t t = last;
    if (s == std::numeric_limits<uint32_t>::max()) break; // should not happen

    for (uint32_t j = 0; j < phase; ++j) {
      const uint32_t v = active[j];
      if (v == s || v == t) continue;
      const uint64_t add = w[static_cast<size_t>(t) * n + v];
      if (add) {
        w[static_cast<size_t>(s) * n + v] += add;
        w[static_cast<size_t>(v) * n + s] += add;
      }
    }
    // Clear t connections
    for (uint32_t j = 0; j < phase; ++j) {
      const uint32_t v = active[j];
      w[static_cast<size_t>(t) * n + v] = 0;
      w[static_cast<size_t>(v) * n + t] = 0;
    }

    if (group[s].size() < group[t].size()) group[s].swap(group[t]);
    group[s].insert(group[s].end(), group[t].begin(), group[t].end());
    group[t].clear();

    // Remove t from active (swap with last active in range)
    uint32_t idxT = 0;
    while (idxT < phase && active[idxT] != t) ++idxT;
    if (idxT + 1 != phase) std::swap(active[idxT], active[phase - 1]);
  }

  res.cut = best;
  res.side.assign(n, 0);
  for (uint32_t v : bestSide) res.side[v] = 1;
  if (bestSide.empty() || bestSide.size() == n) {
    res.side.assign(n, 0);
    res.side[0] = 1;
    res.cut = 0;
  }
  return res;
}

static CutResult stoerWagnerSparse(uint32_t n,
                                  const std::vector<Edge>& edges,
                                  const std::vector<uint64_t>& deg) {
  CutResult res;
  if (n < 2) {
    res.cut = 0;
    res.side.assign(n, 0);
    return res;
  }

  std::vector<std::unordered_map<uint32_t, uint64_t>> adj(n);
  // Small capped reserve: helps on sparse graphs without risking massive overallocation.
  for (uint32_t i = 0; i < n; ++i) {
    const size_t r = static_cast<size_t>(std::min<uint64_t>(deg[i], 256));
    adj[i].reserve(r);
  }
  for (const auto& e : edges) {
    if (e.u == e.v) continue;
    adj[e.u][e.v] += 1;
    adj[e.v][e.u] += 1;
  }

  std::vector<uint32_t> active(n);
  std::iota(active.begin(), active.end(), 0);
  std::vector<char> alive(n, 1);

  std::vector<std::vector<uint32_t>> group(n);
  for (uint32_t i = 0; i < n; ++i) group[i].push_back(i);

  uint64_t best = std::numeric_limits<uint64_t>::max();
  std::vector<uint32_t> bestSide;

  std::vector<uint64_t> weight(n);
  std::vector<char> added(n);

  for (uint32_t phase = n; phase > 1; --phase) {
    std::fill(weight.begin(), weight.end(), 0);
    std::fill(added.begin(), added.end(), 0);

    std::priority_queue<std::pair<uint64_t, uint32_t>> pq;
    for (uint32_t j = 0; j < phase; ++j) pq.push({0, active[j]});

    uint32_t prev = std::numeric_limits<uint32_t>::max();
    uint32_t last = std::numeric_limits<uint32_t>::max();

    for (uint32_t it = 0; it < phase; ++it) {
      uint32_t sel = std::numeric_limits<uint32_t>::max();
      while (!pq.empty()) {
        const auto [key, v] = pq.top();
        pq.pop();
        if (!alive[v] || added[v]) continue;
        if (key != weight[v]) continue; // stale
        sel = v;
        break;
      }
      if (sel == std::numeric_limits<uint32_t>::max()) {
        // Shouldn't happen for connected graphs, but pick any remaining vertex.
        for (uint32_t j = 0; j < phase; ++j) {
          const uint32_t v = active[j];
          if (alive[v] && !added[v]) {
            sel = v;
            break;
          }
        }
      }

      added[sel] = 1;
      prev = last;
      last = sel;

      for (const auto& [to, w] : adj[sel]) {
        if (!alive[to] || added[to]) continue;
        weight[to] += w;
        pq.push({weight[to], to});
      }
    }

    const uint64_t cutOfPhase = weight[last];
    if (cutOfPhase < best) {
      best = cutOfPhase;
      bestSide = group[last];
    }

    const uint32_t s = prev;
    const uint32_t t = last;
    if (s == std::numeric_limits<uint32_t>::max()) break;

    // Merge t into s.
    if (group[s].size() < group[t].size()) group[s].swap(group[t]);
    group[s].insert(group[s].end(), group[t].begin(), group[t].end());
    group[t].clear();

    for (const auto& [to, w] : adj[t]) {
      if (!alive[to] || to == s) continue;
      adj[s][to] += w;
      adj[to][s] += w;
      adj[to].erase(t);
    }
    adj[s].erase(t);
    adj[t].clear();
    alive[t] = 0;

    // Remove t from active (swap with end of active window).
    uint32_t idxT = 0;
    while (idxT < phase && active[idxT] != t) ++idxT;
    if (idxT + 1 != phase) std::swap(active[idxT], active[phase - 1]);
  }

  res.cut = best;
  res.side.assign(n, 0);
  for (uint32_t v : bestSide) res.side[v] = 1;
  if (!isNonTrivialPartition(res.side)) {
    res.cut = 0;
    res.side.assign(n, 0);
    res.side[0] = 1;
  }
  return res;
}

static bool isNonTrivialPartition(const std::vector<char>& side) {
  bool has0 = false, has1 = false;
  for (char c : side) {
    has0 |= (c == 0);
    has1 |= (c != 0);
    if (has0 && has1) return true;
  }
  return false;
}

int main() {
  FastScanner fs;
  uint64_t n64 = 0, m64 = 0;
  if (!fs.readU64(n64)) return 0;
  if (!fs.readU64(m64)) return 0;
  if (n64 > std::numeric_limits<uint32_t>::max()) return 0;
  const uint32_t n = static_cast<uint32_t>(n64);

  FastOutput out;
  if (n < 2) {
    out.writeU64(0);
    out.writeChar('\n');
    return 0;
  }

  // Try to create an edge spill file for a second pass if needed.
  std::FILE* edgeFile = nullptr;
  std::filesystem::path edgeFilePath;
  {
    std::error_code ec;
    auto tmpdir = std::filesystem::temp_directory_path(ec);
    if (ec) tmpdir = std::filesystem::current_path(ec);
    const uint64_t seed =
        static_cast<uint64_t>(std::chrono::high_resolution_clock::now().time_since_epoch().count());
    edgeFilePath = tmpdir / ("mincut_edges_" + std::to_string(seed) + ".bin");
#if defined(_WIN32)
    edgeFile = _wfopen(edgeFilePath.c_str(), L"wb+");
#else
    edgeFile = std::fopen(edgeFilePath.string().c_str(), "wb+");
#endif
  }

  std::vector<uint64_t> deg(n, 0);
  std::vector<Edge> edges;
  edges.clear();
  bool storeInMemory = true;
  if (m64 > static_cast<uint64_t>(std::numeric_limits<size_t>::max())) storeInMemory = false;
  if (storeInMemory) {
    try {
      edges.reserve(static_cast<size_t>(m64));
    } catch (...) {
      storeInMemory = false;
    }
  }

  // Read edges; ignore self-loops (they don't affect edge cuts).
  for (uint64_t i = 0; i < m64; ++i) {
    uint64_t u64 = 0, v64 = 0;
    fs.readU64(u64);
    fs.readU64(v64);
    if (u64 >= n64 || v64 >= n64) continue;
    const uint32_t u = static_cast<uint32_t>(u64);
    const uint32_t v = static_cast<uint32_t>(v64);
    if (u == v) continue;
    deg[u] += 1;
    deg[v] += 1;

    if (storeInMemory) edges.push_back({u, v});
    if (edgeFile) {
      std::fwrite(&u, sizeof(u), 1, edgeFile);
      std::fwrite(&v, sizeof(v), 1, edgeFile);
    }
  }
  if (edgeFile) std::fflush(edgeFile);

  // Trivial cut: isolate a minimum-degree vertex.
  uint64_t minDeg = std::numeric_limits<uint64_t>::max();
  uint32_t minV = 0;
  for (uint32_t i = 0; i < n; ++i) {
    if (deg[i] < minDeg) {
      minDeg = deg[i];
      minV = i;
    }
  }

  std::vector<char> bestSide(n, 0);
  bestSide[minV] = 1;
  uint64_t bestCut = minDeg;

  if (storeInMemory) {
    if (n <= 2000) {
      CutResult sw = stoerWagnerMatrix(n, edges);
      if (isNonTrivialPartition(sw.side)) {
        // Recompute exact cut size on original edge list for safety.
        uint64_t c = 0;
        for (const auto& e : edges) c += (sw.side[e.u] != sw.side[e.v]);
        if (c < bestCut) {
          bestCut = c;
          bestSide = std::move(sw.side);
        }
      }
    } else {
      bool solved = false;
      try {
        CutResult sw = stoerWagnerSparse(n, edges, deg);
        if (isNonTrivialPartition(sw.side)) {
          uint64_t c = 0;
          for (const auto& e : edges) c += (sw.side[e.u] != sw.side[e.v]);
          if (c < bestCut) {
            bestCut = c;
            bestSide = std::move(sw.side);
          }
          solved = true;
        }
      } catch (...) {
        solved = false;
      }

      if (!solved) {
        // Last-resort fallback when the exact algorithm can't be built (e.g., OOM).
        // Uses Karger-Stein to try to improve over the trivial cut.
        Graph g;
        g.n = n;
        g.edges = edges;
        const uint64_t seed =
            static_cast<uint64_t>(
                std::chrono::high_resolution_clock::now().time_since_epoch().count());
        std::mt19937_64 rng(seed ^ (seed << 13));
        const double ln = std::log(std::max<uint32_t>(2u, n));
        int runs = static_cast<int>(std::ceil(ln * ln));
        runs = std::clamp(runs, 1, 100);
        CutResult ks = kargerStein(g, rng, runs, /*earlyStopAt=*/1);
        if (isNonTrivialPartition(ks.side)) {
          uint64_t c = 0;
          for (const auto& e : edges) c += (ks.side[e.u] != ks.side[e.v]);
          if (c < bestCut) {
            bestCut = c;
            bestSide = std::move(ks.side);
          }
        }
      }
    }
  }

  // Output cut edges crossing bestSide.
  out.writeU64(bestCut);
  out.writeChar('\n');

  if (storeInMemory) {
    for (const auto& e : edges) {
      if (bestSide[e.u] != bestSide[e.v]) {
        out.writeU32(e.u);
        out.writeChar(' ');
        out.writeU32(e.v);
        out.writeChar('\n');
      }
    }
  } else if (edgeFile) {
    std::rewind(edgeFile);
    while (true) {
      uint32_t u = 0, v = 0;
      if (std::fread(&u, sizeof(u), 1, edgeFile) != 1) break;
      if (std::fread(&v, sizeof(v), 1, edgeFile) != 1) break;
      if (u >= n || v >= n) continue;
      if (bestSide[u] != bestSide[v]) {
        out.writeU32(u);
        out.writeChar(' ');
        out.writeU32(v);
        out.writeChar('\n');
      }
    }
  }

  if (edgeFile) std::fclose(edgeFile);
  if (!edgeFilePath.empty()) {
    std::error_code ec;
    std::filesystem::remove(edgeFilePath, ec);
  }
  return 0;
}
