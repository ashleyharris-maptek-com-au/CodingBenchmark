using System;
using System.IO;
using System.Runtime.CompilerServices;

internal static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len;
        private int _ptr;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = Read();
                if (c == -1) break;
            }
            return value * sign;
        }
    }

    private sealed class FastWriter : IDisposable
    {
        private readonly Stream _stream = Console.OpenStandardOutput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _pos;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private void Ensure(int bytes)
        {
            if (_pos + bytes <= _buffer.Length) return;
            _stream.Write(_buffer, 0, _pos);
            _pos = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            Ensure(1);
            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteInt(int value)
        {
            Span<byte> tmp = stackalloc byte[11];
            value.TryFormat(tmp, out int written);
            Ensure(written);
            tmp.Slice(0, written).CopyTo(_buffer.AsSpan(_pos));
            _pos += written;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteNewLine() => WriteByte((byte)'\n');

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteSpace() => WriteByte((byte)' ');

        public void Flush()
        {
            if (_pos > 0)
            {
                _stream.Write(_buffer, 0, _pos);
                _pos = 0;
            }
        }

        public void Dispose() => Flush();
    }

    private sealed class Treap
    {
        private int _root;
        private int _nodeCount;
        private readonly int[] _key;
        private readonly int[] _prio;
        private readonly int[] _left;
        private readonly int[] _right;
        private readonly int[] _head;
        private readonly int[] _nextSame;
        private uint _rng;

        public Treap(int maxNodes, int[] nextSame)
        {
            _key = new int[maxNodes + 1];
            _prio = new int[maxNodes + 1];
            _left = new int[maxNodes + 1];
            _right = new int[maxNodes + 1];
            _head = new int[maxNodes + 1];
            _nextSame = nextSame;
            _rng = 2463534242u;
            _root = 0;
            _nodeCount = 0;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private uint NextRand()
        {
            uint x = _rng;
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng = x;
            return x;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int NewNode(int key, int binId)
        {
            int n = ++_nodeCount;
            _key[n] = key;
            _prio[n] = (int)(NextRand() & 0x7fffffff);
            _left[n] = 0;
            _right[n] = 0;
            _head[n] = binId;
            _nextSame[binId] = -1;
            return n;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int RotateRight(int r)
        {
            int l = _left[r];
            _left[r] = _right[l];
            _right[l] = r;
            return l;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int RotateLeft(int r)
        {
            int rr = _right[r];
            _right[r] = _left[rr];
            _left[rr] = r;
            return rr;
        }

        public void Add(int remaining, int binId)
        {
            _root = Insert(_root, remaining, binId);
        }

        private int Insert(int r, int remaining, int binId)
        {
            if (r == 0) return NewNode(remaining, binId);

            int k = _key[r];
            if (remaining == k)
            {
                _nextSame[binId] = _head[r];
                _head[r] = binId;
                return r;
            }

            if (remaining < k)
            {
                int l = Insert(_left[r], remaining, binId);
                _left[r] = l;
                if (_prio[l] > _prio[r]) r = RotateRight(r);
                return r;
            }
            else
            {
                int rr = Insert(_right[r], remaining, binId);
                _right[r] = rr;
                if (_prio[rr] > _prio[r]) r = RotateLeft(r);
                return r;
            }
        }

        private int Merge(int a, int b)
        {
            if (a == 0) return b;
            if (b == 0) return a;
            if (_prio[a] > _prio[b])
            {
                _right[a] = Merge(_right[a], b);
                return a;
            }
            else
            {
                _left[b] = Merge(a, _left[b]);
                return b;
            }
        }

        private int Erase(int r, int remaining)
        {
            if (r == 0) return 0;
            int k = _key[r];
            if (remaining < k)
            {
                _left[r] = Erase(_left[r], remaining);
                return r;
            }
            if (remaining > k)
            {
                _right[r] = Erase(_right[r], remaining);
                return r;
            }
            return Merge(_left[r], _right[r]);
        }

        public bool TryPopLowerBound(int needed, out int remaining, out int binId)
        {
            int cur = _root;
            int cand = 0;
            while (cur != 0)
            {
                if (_key[cur] >= needed)
                {
                    cand = cur;
                    cur = _left[cur];
                }
                else
                {
                    cur = _right[cur];
                }
            }

            if (cand == 0)
            {
                remaining = 0;
                binId = -1;
                return false;
            }

            remaining = _key[cand];
            binId = _head[cand];
            _head[cand] = _nextSame[binId];
            _nextSame[binId] = -1;

            if (_head[cand] == -1)
            {
                _root = Erase(_root, remaining);
            }

            return true;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        if (n <= 0 || n == int.MinValue)
        {
            Console.WriteLine(0);
            return;
        }

        int stockLength = fs.NextInt();
        if (stockLength <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        var items = new ulong[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.NextInt();
            items[i] = ((ulong)(uint.MaxValue - (uint)len) << 32) | (uint)i;
        }

        Array.Sort(items);

        var assignment = new int[n];
        var nextSame = new int[n];
        var treap = new Treap(n, nextSame);

        int binCount = 0;
        for (int i = 0; i < n; i++)
        {
            ulong v = items[i];
            int idx = (int)(uint)v;
            int len = (int)(uint.MaxValue - (uint)(v >> 32));

            if (treap.TryPopLowerBound(len, out int rem, out int binId))
            {
                assignment[idx] = binId;
                int newRem = rem - len;
                if (newRem > 0) treap.Add(newRem, binId);
            }
            else
            {
                binId = binCount++;
                assignment[idx] = binId;
                int newRem = stockLength - len;
                if (newRem > 0) treap.Add(newRem, binId);
            }
        }

        items = null!;
        nextSame = null!;
        treap = null!;

        var counts = new int[binCount];
        for (int i = 0; i < n; i++)
        {
            counts[assignment[i]]++;
        }

        var starts = new int[binCount + 1];
        int running = 0;
        for (int b = 0; b < binCount; b++)
        {
            starts[b] = running;
            running += counts[b];
            starts[b + 1] = running;
            counts[b] = starts[b]; // reuse counts as write pointer
        }

        var grouped = new int[n];
        for (int i = 0; i < n; i++)
        {
            int b = assignment[i];
            grouped[counts[b]++] = i;
        }

        assignment = null!;
        counts = null!;

        using var w = new FastWriter();
        w.WriteInt(binCount);
        w.WriteNewLine();
        for (int b = 0; b < binCount; b++)
        {
            int start = starts[b];
            int end = starts[b + 1];
            if (start < end)
            {
                w.WriteInt(grouped[start]);
                for (int j = start + 1; j < end; j++)
                {
                    w.WriteSpace();
                    w.WriteInt(grouped[j]);
                }
            }
            w.WriteNewLine();
        }
    }
}
