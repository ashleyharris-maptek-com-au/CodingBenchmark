; SPIR-V
; Version: 1.0
; Geometry shader: expand each input point into a hexagon (6 triangles)

OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %inWorldPos %inNormal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %inColor "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0
OpDecorate %arr_v3float_1 ArrayStride 16

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%uint_1 = OpConstant %uint 1
%int_0 = OpConstant %int 0

%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_n1 = OpConstant %float -1.0
%float_0_5 = OpConstant %float 0.5
%float_n0_5 = OpConstant %float -0.5
%float_0_8660254 = OpConstant %float 0.8660254
%float_n0_8660254 = OpConstant %float -0.8660254
%float_r = OpConstant %float 0.04
%float_edgeScale = OpConstant %float 0.8

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in

%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3float_1
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%inWorldPos = OpVariable %ptr_in_arr_v3 Input
%inNormal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

%in_pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%basePos = OpLoad %v4float %in_pos_ptr

%in_col_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%baseColor = OpLoad %v3float %in_col_ptr
%edgeColor = OpVectorTimesScalar %v3float %baseColor %float_edgeScale

%x = OpCompositeExtract %float %basePos 0
%y = OpCompositeExtract %float %basePos 1
%z = OpCompositeExtract %float %basePos 2
%w = OpCompositeExtract %float %basePos 3

; Precompute edge positions around the point using r = 0.04.
%dx0 = OpFMul %float %float_r %float_1
%dy0 = OpFMul %float %float_r %float_0
%ex0 = OpFAdd %float %x %dx0
%ey0 = OpFAdd %float %y %dy0
%edgePos0 = OpCompositeConstruct %v4float %ex0 %ey0 %z %w

%dx1 = OpFMul %float %float_r %float_0_5
%dy1 = OpFMul %float %float_r %float_0_8660254
%ex1 = OpFAdd %float %x %dx1
%ey1 = OpFAdd %float %y %dy1
%edgePos1 = OpCompositeConstruct %v4float %ex1 %ey1 %z %w

%dx2 = OpFMul %float %float_r %float_n0_5
%dy2 = OpFMul %float %float_r %float_0_8660254
%ex2 = OpFAdd %float %x %dx2
%ey2 = OpFAdd %float %y %dy2
%edgePos2 = OpCompositeConstruct %v4float %ex2 %ey2 %z %w

%dx3 = OpFMul %float %float_r %float_n1
%dy3 = OpFMul %float %float_r %float_0
%ex3 = OpFAdd %float %x %dx3
%ey3 = OpFAdd %float %y %dy3
%edgePos3 = OpCompositeConstruct %v4float %ex3 %ey3 %z %w

%dx4 = OpFMul %float %float_r %float_n0_5
%dy4 = OpFMul %float %float_r %float_n0_8660254
%ex4 = OpFAdd %float %x %dx4
%ey4 = OpFAdd %float %y %dy4
%edgePos4 = OpCompositeConstruct %v4float %ex4 %ey4 %z %w

%dx5 = OpFMul %float %float_r %float_0_5
%dy5 = OpFMul %float %float_r %float_n0_8660254
%ex5 = OpFAdd %float %x %dx5
%ey5 = OpFAdd %float %y %dy5
%edgePos5 = OpCompositeConstruct %v4float %ex5 %ey5 %z %w

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit 6 triangles (18 vertices) as separate triangle strips.
; Triangle 0: center, edge0, edge1
OpStore %out_pos_ptr %basePos
OpStore %outColor %baseColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos0
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos1
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 1: center, edge1, edge2
OpStore %out_pos_ptr %basePos
OpStore %outColor %baseColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos1
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos2
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 2: center, edge2, edge3
OpStore %out_pos_ptr %basePos
OpStore %outColor %baseColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos2
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos3
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 3: center, edge3, edge4
OpStore %out_pos_ptr %basePos
OpStore %outColor %baseColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos3
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos4
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 4: center, edge4, edge5
OpStore %out_pos_ptr %basePos
OpStore %outColor %baseColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos4
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos5
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

; Triangle 5: center, edge5, edge0 (close)
OpStore %out_pos_ptr %basePos
OpStore %outColor %baseColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos5
OpStore %outColor %edgeColor
OpEmitVertex
OpStore %out_pos_ptr %edgePos0
OpStore %outColor %edgeColor
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
