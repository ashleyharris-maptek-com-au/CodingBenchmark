declare const require: any;
declare const process: any;

const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: any = input && input.length ? JSON.parse(input) : {};

const prob2d: any = data.probability;
const dronesRaw: any = data.drones;
const maxStepsPerDrone: number = (data.maxStepsPerDrone | 0) >>> 0;

const drones: any[] = Array.isArray(dronesRaw) ? dronesRaw : [];

if (!Array.isArray(prob2d) || prob2d.length === 0) {
  // No grid; emit empty paths.
  process.stdout.write('{"paths":[' + drones.map(() => '[]').join(',') + ']}');
  process.exit(0);
}

const H: number = prob2d.length | 0;
let W = 0;
for (let y = 0; y < H; y++) {
  const row: any = prob2d[y];
  const len = Array.isArray(row) ? row.length | 0 : 0;
  if (len > W) W = len;
}

if (W <= 0 || H <= 0) {
  process.stdout.write('{"paths":[' + drones.map(() => '[]').join(',') + ']}');
  process.exit(0);
}

const N: number = (W * H) | 0;
const prob = new Float64Array(N);
for (let i = 0; i < N; i++) prob[i] = -1;

for (let y = 0; y < H; y++) {
  const row: any = prob2d[y];
  if (!Array.isArray(row)) continue;
  const base = y * W;
  const rowLen = row.length | 0;
  const lim = rowLen < W ? rowLen : W;
  for (let x = 0; x < lim; x++) {
    const v = +row[x];
    prob[base + x] = Number.isFinite(v) ? v : -1;
  }
}

function isObstacleIdx(idx: number): boolean {
  return prob[idx] < 0;
}

function idxOf(x: number, y: number): number {
  return (y * W + x) | 0;
}

function xOf(idx: number): number {
  return idx % W;
}

function yOf(idx: number): number {
  return (idx / W) | 0;
}

function clamp(v: number, lo: number, hi: number): number {
  return v < lo ? lo : v > hi ? hi : v;
}

// Choose unique, valid start cells (handles duplicates / invalids defensively).
const droneCount = drones.length | 0;
const startIdx = new Int32Array(droneCount);
for (let i = 0; i < droneCount; i++) startIdx[i] = -1;

let fallbackIdx = -1;
for (let i = 0; i < N; i++) {
  if (!isObstacleIdx(i)) {
    fallbackIdx = i;
    break;
  }
}

if (fallbackIdx === -1) {
  // Grid is all obstacles; every drone must trivially "stay" at (0,0).
  process.stdout.write('{"paths":[' + drones.map(() => '[[0,0]]').join(',') + ']}');
  process.exit(0);
}

const taken = new Uint8Array(N);

function isFreeUntaken(idx: number): boolean {
  return idx >= 0 && idx < N && taken[idx] === 0 && !isObstacleIdx(idx);
}

function findNearestFreeStart(x0: number, y0: number): number {
  const xC = clamp(x0 | 0, 0, W - 1);
  const yC = clamp(y0 | 0, 0, H - 1);
  const start = idxOf(xC, yC);
  if (isFreeUntaken(start)) return start;

  const maxD = 40;
  for (let d = 1; d <= maxD; d++) {
    for (let dx = -d; dx <= d; dx++) {
      const adx = dx < 0 ? -dx : dx;
      const dy = d - adx;

      const x1 = xC + dx;
      const y1 = yC + dy;
      if (x1 >= 0 && x1 < W && y1 >= 0 && y1 < H) {
        const idx1 = idxOf(x1, y1);
        if (isFreeUntaken(idx1)) return idx1;
      }

      if (dy !== 0) {
        const y2 = yC - dy;
        if (x1 >= 0 && x1 < W && y2 >= 0 && y2 < H) {
          const idx2 = idxOf(x1, y2);
          if (isFreeUntaken(idx2)) return idx2;
        }
      }
    }
  }

  // Fallback: scan for any free untaken cell.
  for (let idx = 0; idx < N; idx++) {
    if (isFreeUntaken(idx)) return idx;
  }
  return fallbackIdx;
}

for (let i = 0; i < droneCount; i++) {
  const d: any = drones[i] || {};
  const sx = d.x | 0;
  const sy = d.y | 0;
  const si = findNearestFreeStart(sx, sy);
  startIdx[i] = si;
  taken[si] = 1;
}

// Multi-source BFS to partition the grid into geodesic Voronoi regions.
const owner = new Int32Array(N);
const dist = new Int32Array(N);
const parent = new Int32Array(N);
const firstChild = new Int32Array(N);
const nextSibling = new Int32Array(N);
for (let i = 0; i < N; i++) {
  owner[i] = -1;
  dist[i] = -1;
  parent[i] = -1;
  firstChild[i] = -1;
  nextSibling[i] = -1;
}

const queue = new Int32Array(N);
const order = new Int32Array(N);
let qh = 0;
let qt = 0;
let ordLen = 0;

for (let i = 0; i < droneCount; i++) {
  const s = startIdx[i];
  if (s < 0 || s >= N || isObstacleIdx(s)) continue;
  owner[s] = i;
  dist[s] = 0;
  parent[s] = -1;
  queue[qt++] = s;
}

while (qh < qt) {
  const cur = queue[qh++];
  order[ordLen++] = cur;
  const curOwner = owner[cur];
  const nextD = (dist[cur] + 1) | 0;

  // Up
  if (cur >= W) {
    const nb = (cur - W) | 0;
    if (dist[nb] === -1 && !isObstacleIdx(nb)) {
      owner[nb] = curOwner;
      dist[nb] = nextD;
      parent[nb] = cur;
      queue[qt++] = nb;
    }
  }
  // Down
  if (cur < N - W) {
    const nb = (cur + W) | 0;
    if (dist[nb] === -1 && !isObstacleIdx(nb)) {
      owner[nb] = curOwner;
      dist[nb] = nextD;
      parent[nb] = cur;
      queue[qt++] = nb;
    }
  }
  // Left
  if (cur % W !== 0) {
    const nb = (cur - 1) | 0;
    if (dist[nb] === -1 && !isObstacleIdx(nb)) {
      owner[nb] = curOwner;
      dist[nb] = nextD;
      parent[nb] = cur;
      queue[qt++] = nb;
    }
  }
  // Right
  if ((cur + 1) % W !== 0) {
    const nb = (cur + 1) | 0;
    if (dist[nb] === -1 && !isObstacleIdx(nb)) {
      owner[nb] = curOwner;
      dist[nb] = nextD;
      parent[nb] = cur;
      queue[qt++] = nb;
    }
  }
}

// Build child lists for the BFS trees.
for (let i = 0; i < ordLen; i++) {
  const node = order[i];
  const p = parent[node];
  if (p >= 0) {
    nextSibling[node] = firstChild[p];
    firstChild[p] = node;
  }
}

// Compute subtree reward sums (sum of positive probabilities in a subtree).
const sub = new Float64Array(N);
for (let i = 0; i < ordLen; i++) {
  const node = order[i];
  const v = prob[node];
  sub[node] = v > 0 ? v : 0;
}
for (let i = ordLen - 1; i >= 0; i--) {
  const node = order[i];
  const p = parent[node];
  if (p >= 0) sub[p] += sub[node];
}

// Sort each node's children by descending subtree reward.
const childBuf = new Int32Array(4);
for (let i = 0; i < ordLen; i++) {
  const node = order[i];
  let child = firstChild[node];
  if (child === -1) continue;

  let c = 0;
  while (child !== -1 && c < 4) {
    childBuf[c++] = child;
    child = nextSibling[child];
  }
  if (c <= 1) continue;

  // Insertion sort for tiny arrays.
  for (let a = 1; a < c; a++) {
    const key = childBuf[a];
    const keyVal = sub[key];
    let b = a - 1;
    while (b >= 0) {
      const curIdx = childBuf[b];
      const curVal = sub[curIdx];
      if (curVal > keyVal) break;
      if (curVal === keyVal && curIdx < key) break;
      childBuf[b + 1] = curIdx;
      b--;
    }
    childBuf[b + 1] = key;
  }

  firstChild[node] = childBuf[0];
  for (let j = 0; j < c - 1; j++) nextSibling[childBuf[j]] = childBuf[j + 1];
  nextSibling[childBuf[c - 1]] = -1;
}

function firstUsefulChild(node: number): number {
  let c = firstChild[node];
  while (c !== -1 && sub[c] <= 0) c = nextSibling[c];
  return c;
}

function nextUsefulSibling(node: number): number {
  let s = nextSibling[node];
  while (s !== -1 && sub[s] <= 0) s = nextSibling[s];
  return s;
}

// Emit JSON progressively to avoid massive allocations.
process.stdout.write('{"paths":[');
for (let i = 0; i < droneCount; i++) {
  if (i > 0) process.stdout.write(',');
  process.stdout.write('[');

  let cur = startIdx[i];
  if (cur < 0 || cur >= N || isObstacleIdx(cur)) cur = fallbackIdx;
  let prev = parent[cur]; // -1 for roots/unreached nodes

  process.stdout.write('[' + xOf(cur) + ',' + yOf(cur) + ']');

  for (let step = 0; step < maxStepsPerDrone; step++) {
    const p = parent[cur];
    let next: number;
    if (prev === p) {
      const child = firstUsefulChild(cur);
      next = child !== -1 ? child : p;
    } else {
      const sib = prev >= 0 ? nextUsefulSibling(prev) : -1;
      next = sib !== -1 ? sib : p;
    }
    if (next === -1) next = cur;
    prev = cur;
    cur = next;
    process.stdout.write(',[' + xOf(cur) + ',' + yOf(cur) + ']');
  }

  process.stdout.write(']');
}
process.stdout.write(']}');
