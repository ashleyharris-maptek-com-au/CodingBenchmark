namespace chessbot {

using U64 = std::uint64_t;

static constexpr int WHITE = 0;
static constexpr int BLACK = 1;

static constexpr int CASTLE_WK = 1;
static constexpr int CASTLE_WQ = 2;
static constexpr int CASTLE_BK = 4;
static constexpr int CASTLE_BQ = 8;

static constexpr int MATE_SCORE = 30000;
static constexpr int MATE_THRESHOLD = 29000;
static constexpr int INF = 32000;
static constexpr int MAX_PLY = 64;

enum Piece : int {
    EMPTY = 0,
    WP = 1,
    WN = 2,
    WB = 3,
    WR = 4,
    WQ = 5,
    WK = 6,
    BP = 7,
    BN = 8,
    BB = 9,
    BR = 10,
    BQ = 11,
    BK = 12
};

inline bool isWhitePiece(int p) { return p >= WP && p <= WK; }
inline bool isBlackPiece(int p) { return p >= BP && p <= BK; }
inline int pieceColor(int p) { return isWhitePiece(p) ? WHITE : (isBlackPiece(p) ? BLACK : -1); }

inline int pieceTypeIndex(int p) {
    switch (p) {
    case WP:
    case BP: return 0; // pawn
    case WN:
    case BN: return 1; // knight
    case WB:
    case BB: return 2; // bishop
    case WR:
    case BR: return 3; // rook
    case WQ:
    case BQ: return 4; // queen
    case WK:
    case BK: return 5; // king
    default: return -1;
    }
}

inline int fileOf(int sq) { return sq & 7; }
inline int rankOf(int sq) { return sq >> 3; }
inline U64 sqBB(int sq) { return 1ULL << sq; }

inline int popLsb(U64& bb) {
    int idx = static_cast<int>(std::countr_zero(bb));
    bb &= bb - 1;
    return idx;
}

inline int flipSq(int sq) { return sq ^ 56; }

enum MoveFlags : std::uint8_t {
    MF_NONE = 0,
    MF_CAPTURE = 1 << 0,
    MF_ENPASSANT = 1 << 1,
    MF_CASTLING = 1 << 2,
    MF_PAWN_DOUBLE = 1 << 3
};

struct Move {
    std::uint8_t from = 0;
    std::uint8_t to = 0;
    std::uint8_t promo = 0; // Piece code of promoted piece (WQ/WR/WB/WN or BQ/BR/BB/BN), or 0
    std::uint8_t flags = 0;
};

inline std::uint32_t encodeMove(const Move& m) {
    return (std::uint32_t)m.from | ((std::uint32_t)m.to << 6) | ((std::uint32_t)m.promo << 12) |
        ((std::uint32_t)m.flags << 16);
}

inline Move decodeMove(std::uint32_t code) {
    Move m;
    m.from = (std::uint8_t)(code & 0x3F);
    m.to = (std::uint8_t)((code >> 6) & 0x3F);
    m.promo = (std::uint8_t)((code >> 12) & 0x0F);
    m.flags = (std::uint8_t)((code >> 16) & 0xFF);
    return m;
}

struct Undo {
    int capturedPiece = EMPTY;
    int castlingRights = 0;
    int epSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;
    U64 hash = 0;
    int mg0 = 0, mg1 = 0;
    int eg0 = 0, eg1 = 0;
    int phase = 0;
    int ksq0 = -1, ksq1 = -1;
};

static U64 KNIGHT_ATTACKS[64];
static U64 KING_ATTACKS[64];

static int MG_TABLE[13][64];
static int EG_TABLE[13][64];
static int PHASE_INC[13];

static U64 Z_PIECE[13][64];
static U64 Z_CASTLING[16];
static U64 Z_EPFILE[8];
static U64 Z_SIDE;

static int pieceValue(int p) {
    switch (p) {
    case WP:
    case BP: return 100;
    case WN:
    case BN: return 320;
    case WB:
    case BB: return 330;
    case WR:
    case BR: return 500;
    case WQ:
    case BQ: return 900;
    case WK:
    case BK: return 10000;
    default: return 0;
    }
}

static char pieceLetterForOutput(int p) {
    switch (p) {
    case WP:
    case BP: return 'p';
    case WN:
    case BN: return 'N';
    case WB:
    case BB: return 'B';
    case WR:
    case BR: return 'R';
    case WQ:
    case BQ: return 'Q';
    case WK:
    case BK: return 'K';
    default: return '?';
    }
}

static char promotionLetter(int promoPiece) {
    switch (promoPiece) {
    case WQ:
    case BQ: return 'Q';
    case WR:
    case BR: return 'R';
    case WB:
    case BB: return 'B';
    case WN:
    case BN: return 'N';
    default: return '?';
    }
}

static std::string squareToString(int sq) {
    std::string s;
    s.push_back(char('a' + fileOf(sq)));
    s.push_back(char('1' + rankOf(sq)));
    return s;
}

static inline int scoreToTT(int score, int ply) {
    if (score > MATE_THRESHOLD) return score + ply;
    if (score < -MATE_THRESHOLD) return score - ply;
    return score;
}

static inline int scoreFromTT(int score, int ply) {
    if (score > MATE_THRESHOLD) return score - ply;
    if (score < -MATE_THRESHOLD) return score + ply;
    return score;
}

struct Board {
    std::array<int, 64> piece{};
    std::array<U64, 13> bb{};
    std::array<U64, 2> occ{};
    U64 allOcc = 0;

    int sideToMove = WHITE;
    int castlingRights = 0;
    int epSquare = -1;
    int halfmoveClock = 0;
    int fullmoveNumber = 1;

    U64 hash = 0;

    int mgScore[2] = {0, 0};
    int egScore[2] = {0, 0};
    int phase = 0;
    int kingSq[2] = {-1, -1};

    void clear() {
        piece.fill(EMPTY);
        bb.fill(0);
        occ.fill(0);
        allOcc = 0;
        sideToMove = WHITE;
        castlingRights = 0;
        epSquare = -1;
        halfmoveClock = 0;
        fullmoveNumber = 1;
        hash = 0;
        mgScore[0] = mgScore[1] = 0;
        egScore[0] = egScore[1] = 0;
        phase = 0;
        kingSq[0] = kingSq[1] = -1;
    }

    void addPiece(int sq, int p) {
        piece[sq] = p;
        bb[p] |= sqBB(sq);
        int c = pieceColor(p);
        if (c != -1) {
            occ[c] |= sqBB(sq);
            allOcc |= sqBB(sq);
        }
        if (p == WK) kingSq[WHITE] = sq;
        if (p == BK) kingSq[BLACK] = sq;
        if (p != EMPTY) {
            int c2 = pieceColor(p);
            mgScore[c2] += MG_TABLE[p][sq];
            egScore[c2] += EG_TABLE[p][sq];
            phase += PHASE_INC[p];
        }
        hash ^= Z_PIECE[p][sq];
    }

    void removePiece(int sq) {
        int p = piece[sq];
        if (p == EMPTY) return;
        piece[sq] = EMPTY;
        bb[p] &= ~sqBB(sq);
        int c = pieceColor(p);
        if (c != -1) {
            occ[c] &= ~sqBB(sq);
            allOcc &= ~sqBB(sq);
        }
        if (p == WK) kingSq[WHITE] = -1;
        if (p == BK) kingSq[BLACK] = -1;
        mgScore[c] -= MG_TABLE[p][sq];
        egScore[c] -= EG_TABLE[p][sq];
        phase -= PHASE_INC[p];
        hash ^= Z_PIECE[p][sq];
    }

    void computeInitialHashAndEval() {
        hash = 0;
        mgScore[0] = mgScore[1] = 0;
        egScore[0] = egScore[1] = 0;
        phase = 0;
        kingSq[0] = kingSq[1] = -1;

        bb.fill(0);
        occ.fill(0);
        allOcc = 0;

        for (int sq = 0; sq < 64; ++sq) {
            int p = piece[sq];
            if (p == EMPTY) continue;
            hash ^= Z_PIECE[p][sq];
            bb[p] |= sqBB(sq);
            int c = pieceColor(p);
            occ[c] |= sqBB(sq);
            allOcc |= sqBB(sq);
            mgScore[c] += MG_TABLE[p][sq];
            egScore[c] += EG_TABLE[p][sq];
            phase += PHASE_INC[p];
            if (p == WK) kingSq[WHITE] = sq;
            if (p == BK) kingSq[BLACK] = sq;
        }

        hash ^= Z_CASTLING[castlingRights];
        if (epSquare != -1) hash ^= Z_EPFILE[fileOf(epSquare)];
        if (sideToMove == BLACK) hash ^= Z_SIDE;
    }

    bool setFromFEN(const std::string& fen) {
        clear();
        std::istringstream iss(fen);
        std::string placement, stm, castling, ep;
        if (!(iss >> placement >> stm >> castling >> ep >> halfmoveClock >> fullmoveNumber)) {
            return false;
        }

        int rank = 7;
        int file = 0;
        for (char ch : placement) {
            if (ch == '/') {
                --rank;
                file = 0;
                continue;
            }
            if (ch >= '1' && ch <= '8') {
                file += (ch - '0');
                continue;
            }
            if (file < 0 || file > 7 || rank < 0 || rank > 7) return false;
            int sq = rank * 8 + file;
            int p = EMPTY;
            switch (ch) {
            case 'P': p = WP; break;
            case 'N': p = WN; break;
            case 'B': p = WB; break;
            case 'R': p = WR; break;
            case 'Q': p = WQ; break;
            case 'K': p = WK; break;
            case 'p': p = BP; break;
            case 'n': p = BN; break;
            case 'b': p = BB; break;
            case 'r': p = BR; break;
            case 'q': p = BQ; break;
            case 'k': p = BK; break;
            default: return false;
            }
            piece[sq] = p;
            ++file;
        }

        sideToMove = (stm == "b") ? BLACK : WHITE;

        castlingRights = 0;
        if (castling != "-") {
            for (char c : castling) {
                switch (c) {
                case 'K': castlingRights |= CASTLE_WK; break;
                case 'Q': castlingRights |= CASTLE_WQ; break;
                case 'k': castlingRights |= CASTLE_BK; break;
                case 'q': castlingRights |= CASTLE_BQ; break;
                default: break;
                }
            }
        }

        epSquare = -1;
        if (ep != "-") {
            if (ep.size() == 2) {
                int f = ep[0] - 'a';
                int r = ep[1] - '1';
                if (f >= 0 && f < 8 && r >= 0 && r < 8) epSquare = r * 8 + f;
            }
        }

        computeInitialHashAndEval();
        return true;
    }

    bool isSquareAttacked(int sq, int byColor) const {
        int r = rankOf(sq);
        int f = fileOf(sq);

        // pawns
        if (byColor == WHITE) {
            if (r > 0) {
                if (f > 0 && piece[sq - 9] == WP) return true;
                if (f < 7 && piece[sq - 7] == WP) return true;
            }
        } else {
            if (r < 7) {
                if (f > 0 && piece[sq + 7] == BP) return true;
                if (f < 7 && piece[sq + 9] == BP) return true;
            }
        }

        // knights
        if (KNIGHT_ATTACKS[sq] & (byColor == WHITE ? bb[WN] : bb[BN])) return true;

        // bishops / queens (diagonals)
        static constexpr int ddr[4] = {1, 1, -1, -1};
        static constexpr int ddf[4] = {1, -1, 1, -1};
        for (int i = 0; i < 4; ++i) {
            int rr = r + ddr[i];
            int ff = f + ddf[i];
            while (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) {
                int to = rr * 8 + ff;
                int p = piece[to];
                if (p != EMPTY) {
                    if (pieceColor(p) == byColor) {
                        int t = pieceTypeIndex(p);
                        if (t == 2 || t == 4) return true;
                    }
                    break;
                }
                rr += ddr[i];
                ff += ddf[i];
            }
        }

        // rooks / queens (orthogonal)
        static constexpr int rdr[4] = {1, -1, 0, 0};
        static constexpr int rdf[4] = {0, 0, 1, -1};
        for (int i = 0; i < 4; ++i) {
            int rr = r + rdr[i];
            int ff = f + rdf[i];
            while (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) {
                int to = rr * 8 + ff;
                int p = piece[to];
                if (p != EMPTY) {
                    if (pieceColor(p) == byColor) {
                        int t = pieceTypeIndex(p);
                        if (t == 3 || t == 4) return true;
                    }
                    break;
                }
                rr += rdr[i];
                ff += rdf[i];
            }
        }

        // king
        if (KING_ATTACKS[sq] & (byColor == WHITE ? bb[WK] : bb[BK])) return true;

        return false;
    }

    bool inCheck(int color) const {
        int ksq = kingSq[color];
        if (ksq < 0) return false;
        return isSquareAttacked(ksq, color ^ 1);
    }

    void updateCastlingRightsAfterMove(int movingPiece, int from, int to, int capturedPiece) {
        if (movingPiece == WK) castlingRights &= ~(CASTLE_WK | CASTLE_WQ);
        else if (movingPiece == BK) castlingRights &= ~(CASTLE_BK | CASTLE_BQ);
        else if (movingPiece == WR) {
            if (from == 7) castlingRights &= ~CASTLE_WK;
            else if (from == 0) castlingRights &= ~CASTLE_WQ;
        } else if (movingPiece == BR) {
            if (from == 63) castlingRights &= ~CASTLE_BK;
            else if (from == 56) castlingRights &= ~CASTLE_BQ;
        }

        if (capturedPiece == WR) {
            if (to == 7) castlingRights &= ~CASTLE_WK;
            else if (to == 0) castlingRights &= ~CASTLE_WQ;
        } else if (capturedPiece == BR) {
            if (to == 63) castlingRights &= ~CASTLE_BK;
            else if (to == 56) castlingRights &= ~CASTLE_BQ;
        }
    }

    void makeMove(const Move& m, Undo& u) {
        u.capturedPiece = EMPTY;
        u.castlingRights = castlingRights;
        u.epSquare = epSquare;
        u.halfmoveClock = halfmoveClock;
        u.fullmoveNumber = fullmoveNumber;
        u.hash = hash;
        u.mg0 = mgScore[0];
        u.mg1 = mgScore[1];
        u.eg0 = egScore[0];
        u.eg1 = egScore[1];
        u.phase = phase;
        u.ksq0 = kingSq[0];
        u.ksq1 = kingSq[1];

        const int us = sideToMove;
        const int them = us ^ 1;

        const int from = m.from;
        const int to = m.to;
        const int movingPiece = piece[from];

        // remove ep / castling / side from hash (then add new ones later)
        if (epSquare != -1) hash ^= Z_EPFILE[fileOf(epSquare)];
        hash ^= Z_CASTLING[castlingRights];
        hash ^= Z_SIDE;

        epSquare = -1;

        // capture
        if (m.flags & MF_ENPASSANT) {
            int capSq = (us == WHITE) ? (to - 8) : (to + 8);
            u.capturedPiece = piece[capSq];
            removePiece(capSq);
        } else if (m.flags & MF_CAPTURE) {
            u.capturedPiece = piece[to];
            removePiece(to);
        }

        // move piece (and promotion)
        removePiece(from);

        int placedPiece = movingPiece;
        if (m.promo) placedPiece = m.promo;
        addPiece(to, placedPiece);

        // castling rook move
        if (m.flags & MF_CASTLING) {
            if (us == WHITE) {
                if (to == 6) { // O-O
                    removePiece(7);
                    addPiece(5, WR);
                } else if (to == 2) { // O-O-O
                    removePiece(0);
                    addPiece(3, WR);
                }
            } else {
                if (to == 62) { // O-O
                    removePiece(63);
                    addPiece(61, BR);
                } else if (to == 58) { // O-O-O
                    removePiece(56);
                    addPiece(59, BR);
                }
            }
        }

        // castling rights update
        updateCastlingRightsAfterMove(movingPiece, from, to, u.capturedPiece);

        // ep square update
        if (m.flags & MF_PAWN_DOUBLE) {
            epSquare = (us == WHITE) ? (from + 8) : (from - 8);
        }

        // halfmove clock
        if (movingPiece == WP || movingPiece == BP || (m.flags & MF_CAPTURE)) halfmoveClock = 0;
        else halfmoveClock++;
        if (us == BLACK) fullmoveNumber++;

        // add new castling/ep and side-to-move hash
        hash ^= Z_CASTLING[castlingRights];
        if (epSquare != -1) hash ^= Z_EPFILE[fileOf(epSquare)];

        sideToMove = them;
    }

    void unmakeMove(const Move& m, const Undo& u) {
        // restore simple state
        sideToMove ^= 1;
        castlingRights = u.castlingRights;
        epSquare = u.epSquare;
        halfmoveClock = u.halfmoveClock;
        fullmoveNumber = u.fullmoveNumber;
        hash = u.hash;
        mgScore[0] = u.mg0;
        mgScore[1] = u.mg1;
        egScore[0] = u.eg0;
        egScore[1] = u.eg1;
        phase = u.phase;
        kingSq[0] = u.ksq0;
        kingSq[1] = u.ksq1;

        const int from = m.from;
        const int to = m.to;
        const int us = sideToMove;
        const int them = us ^ 1;

        // undo castling rook move first (board currently in "after move" state)
        if (m.flags & MF_CASTLING) {
            if (us == WHITE) {
                if (to == 6) {
                    // rook f1 -> h1
                    piece[5] = EMPTY; bb[WR] &= ~sqBB(5); occ[WHITE] &= ~sqBB(5); allOcc &= ~sqBB(5);
                    piece[7] = WR; bb[WR] |= sqBB(7); occ[WHITE] |= sqBB(7); allOcc |= sqBB(7);
                } else if (to == 2) {
                    // rook d1 -> a1
                    piece[3] = EMPTY; bb[WR] &= ~sqBB(3); occ[WHITE] &= ~sqBB(3); allOcc &= ~sqBB(3);
                    piece[0] = WR; bb[WR] |= sqBB(0); occ[WHITE] |= sqBB(0); allOcc |= sqBB(0);
                }
            } else {
                if (to == 62) {
                    // rook f8 -> h8
                    piece[61] = EMPTY; bb[BR] &= ~sqBB(61); occ[BLACK] &= ~sqBB(61); allOcc &= ~sqBB(61);
                    piece[63] = BR; bb[BR] |= sqBB(63); occ[BLACK] |= sqBB(63); allOcc |= sqBB(63);
                } else if (to == 58) {
                    // rook d8 -> a8
                    piece[59] = EMPTY; bb[BR] &= ~sqBB(59); occ[BLACK] &= ~sqBB(59); allOcc &= ~sqBB(59);
                    piece[56] = BR; bb[BR] |= sqBB(56); occ[BLACK] |= sqBB(56); allOcc |= sqBB(56);
                }
            }
        }

        // move piece back (handle promotion)
        int movedPieceAfter = piece[to];
        int movedPieceBefore = movedPieceAfter;
        if (m.promo) {
            movedPieceBefore = (us == WHITE) ? WP : BP;
        }

        // clear destination
        piece[to] = EMPTY;
        bb[movedPieceAfter] &= ~sqBB(to);
        occ[us] &= ~sqBB(to);
        allOcc &= ~sqBB(to);

        // restore origin
        piece[from] = movedPieceBefore;
        bb[movedPieceBefore] |= sqBB(from);
        occ[us] |= sqBB(from);
        allOcc |= sqBB(from);

        // restore captured piece
        if (m.flags & MF_ENPASSANT) {
            int capSq = (us == WHITE) ? (to - 8) : (to + 8);
            piece[capSq] = (us == WHITE) ? BP : WP;
            int capPiece = piece[capSq];
            bb[capPiece] |= sqBB(capSq);
            occ[them] |= sqBB(capSq);
            allOcc |= sqBB(capSq);
        } else if (m.flags & MF_CAPTURE) {
            piece[to] = u.capturedPiece;
            if (u.capturedPiece != EMPTY) {
                bb[u.capturedPiece] |= sqBB(to);
                occ[them] |= sqBB(to);
                allOcc |= sqBB(to);
            }
        }
    }

    int evaluate() const {
        int mgPhase = phase;
        if (mgPhase > 24) mgPhase = 24;
        int egPhase = 24 - mgPhase;

        int us = sideToMove;
        int them = us ^ 1;

        int mgScoreDiff = mgScore[us] - mgScore[them];
        int egScoreDiff = egScore[us] - egScore[them];

        int score = (mgScoreDiff * mgPhase + egScoreDiff * egPhase) / 24;
        score += 8; // small tempo bonus for side to move
        return score;
    }

    void generatePseudoMoves(std::array<Move, 256>& out, int& count, bool capturesOnly) const {
        count = 0;
        const int us = sideToMove;
        const int them = us ^ 1;

        auto pushMove = [&](int from, int to, int promo, std::uint8_t flags) {
            Move m;
            m.from = (std::uint8_t)from;
            m.to = (std::uint8_t)to;
            m.promo = (std::uint8_t)promo;
            m.flags = flags;
            out[count++] = m;
        };

        // pawns
        U64 pawns = (us == WHITE) ? bb[WP] : bb[BP];
        while (pawns) {
            int from = popLsb(pawns);
            int r = rankOf(from);
            int f = fileOf(from);
            if (us == WHITE) {
                int one = from + 8;
                if (one < 64 && piece[one] == EMPTY) {
                    // Always include promotions (even in capture-only mode)
                    if (r == 6) {
                        pushMove(from, one, WQ, MF_NONE);
                        pushMove(from, one, WR, MF_NONE);
                        pushMove(from, one, WB, MF_NONE);
                        pushMove(from, one, WN, MF_NONE);
                    } else if (!capturesOnly) {
                        pushMove(from, one, 0, MF_NONE);
                        if (r == 1) {
                            int two = from + 16;
                            if (piece[two] == EMPTY) pushMove(from, two, 0, MF_PAWN_DOUBLE);
                        }
                    }
                }
                if (f > 0) {
                    int to = from + 7;
                    if (to < 64) {
                        if (piece[to] != EMPTY && pieceColor(piece[to]) == them) {
                            if (r == 6) {
                                pushMove(from, to, WQ, MF_CAPTURE);
                                pushMove(from, to, WR, MF_CAPTURE);
                                pushMove(from, to, WB, MF_CAPTURE);
                                pushMove(from, to, WN, MF_CAPTURE);
                            } else {
                                pushMove(from, to, 0, MF_CAPTURE);
                            }
                        } else if (epSquare == to) {
                            pushMove(from, to, 0, (std::uint8_t)(MF_CAPTURE | MF_ENPASSANT));
                        }
                    }
                }
                if (f < 7) {
                    int to = from + 9;
                    if (to < 64) {
                        if (piece[to] != EMPTY && pieceColor(piece[to]) == them) {
                            if (r == 6) {
                                pushMove(from, to, WQ, MF_CAPTURE);
                                pushMove(from, to, WR, MF_CAPTURE);
                                pushMove(from, to, WB, MF_CAPTURE);
                                pushMove(from, to, WN, MF_CAPTURE);
                            } else {
                                pushMove(from, to, 0, MF_CAPTURE);
                            }
                        } else if (epSquare == to) {
                            pushMove(from, to, 0, (std::uint8_t)(MF_CAPTURE | MF_ENPASSANT));
                        }
                    }
                }
            } else {
                int one = from - 8;
                if (one >= 0 && piece[one] == EMPTY) {
                    // Always include promotions (even in capture-only mode)
                    if (r == 1) {
                        pushMove(from, one, BQ, MF_NONE);
                        pushMove(from, one, BR, MF_NONE);
                        pushMove(from, one, BB, MF_NONE);
                        pushMove(from, one, BN, MF_NONE);
                    } else if (!capturesOnly) {
                        pushMove(from, one, 0, MF_NONE);
                        if (r == 6) {
                            int two = from - 16;
                            if (piece[two] == EMPTY) pushMove(from, two, 0, MF_PAWN_DOUBLE);
                        }
                    }
                }
                if (f > 0) {
                    int to = from - 9;
                    if (to >= 0) {
                        if (piece[to] != EMPTY && pieceColor(piece[to]) == them) {
                            if (r == 1) {
                                pushMove(from, to, BQ, MF_CAPTURE);
                                pushMove(from, to, BR, MF_CAPTURE);
                                pushMove(from, to, BB, MF_CAPTURE);
                                pushMove(from, to, BN, MF_CAPTURE);
                            } else {
                                pushMove(from, to, 0, MF_CAPTURE);
                            }
                        } else if (epSquare == to) {
                            pushMove(from, to, 0, (std::uint8_t)(MF_CAPTURE | MF_ENPASSANT));
                        }
                    }
                }
                if (f < 7) {
                    int to = from - 7;
                    if (to >= 0) {
                        if (piece[to] != EMPTY && pieceColor(piece[to]) == them) {
                            if (r == 1) {
                                pushMove(from, to, BQ, MF_CAPTURE);
                                pushMove(from, to, BR, MF_CAPTURE);
                                pushMove(from, to, BB, MF_CAPTURE);
                                pushMove(from, to, BN, MF_CAPTURE);
                            } else {
                                pushMove(from, to, 0, MF_CAPTURE);
                            }
                        } else if (epSquare == to) {
                            pushMove(from, to, 0, (std::uint8_t)(MF_CAPTURE | MF_ENPASSANT));
                        }
                    }
                }
            }
        }

        auto genLeapers = [&](int pieceCode, const U64 attacksTable[64]) {
            U64 bits = bb[pieceCode];
            while (bits) {
                int from = popLsb(bits);
                U64 targets = attacksTable[from] & ~occ[us];
                while (targets) {
                    int to = popLsb(targets);
                    bool cap = piece[to] != EMPTY;
                    if (capturesOnly && !cap) continue;
                    pushMove(from, to, 0, cap ? MF_CAPTURE : MF_NONE);
                }
            }
        };

        // knights
        genLeapers(us == WHITE ? WN : BN, KNIGHT_ATTACKS);

        // bishops / rooks / queens (sliders)
        auto genSliders = [&](int pieceCode, bool diag, bool ortho) {
            U64 bits = bb[pieceCode];
            while (bits) {
                int from = popLsb(bits);
                int r = rankOf(from);
                int f = fileOf(from);
                if (diag) {
                    static constexpr int ddr[4] = {1, 1, -1, -1};
                    static constexpr int ddf[4] = {1, -1, 1, -1};
                    for (int i = 0; i < 4; ++i) {
                        int rr = r + ddr[i];
                        int ff = f + ddf[i];
                        while (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) {
                            int to = rr * 8 + ff;
                            int p = piece[to];
                            if (p == EMPTY) {
                                if (!capturesOnly) pushMove(from, to, 0, MF_NONE);
                            } else {
                                if (pieceColor(p) == them) pushMove(from, to, 0, MF_CAPTURE);
                                break;
                            }
                            rr += ddr[i];
                            ff += ddf[i];
                        }
                    }
                }
                if (ortho) {
                    static constexpr int rdr[4] = {1, -1, 0, 0};
                    static constexpr int rdf[4] = {0, 0, 1, -1};
                    for (int i = 0; i < 4; ++i) {
                        int rr = r + rdr[i];
                        int ff = f + rdf[i];
                        while (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) {
                            int to = rr * 8 + ff;
                            int p = piece[to];
                            if (p == EMPTY) {
                                if (!capturesOnly) pushMove(from, to, 0, MF_NONE);
                            } else {
                                if (pieceColor(p) == them) pushMove(from, to, 0, MF_CAPTURE);
                                break;
                            }
                            rr += rdr[i];
                            ff += rdf[i];
                        }
                    }
                }
            }
        };

        genSliders(us == WHITE ? WB : BB, true, false);
        genSliders(us == WHITE ? WR : BR, false, true);
        genSliders(us == WHITE ? WQ : BQ, true, true);

        // king
        genLeapers(us == WHITE ? WK : BK, KING_ATTACKS);

        // castling
        if (!capturesOnly) {
            if (us == WHITE && kingSq[WHITE] == 4) {
                if ((castlingRights & CASTLE_WK) && piece[5] == EMPTY && piece[6] == EMPTY && piece[7] == WR) {
                    if (!isSquareAttacked(4, BLACK) && !isSquareAttacked(5, BLACK) && !isSquareAttacked(6, BLACK)) {
                        pushMove(4, 6, 0, MF_CASTLING);
                    }
                }
                if ((castlingRights & CASTLE_WQ) && piece[3] == EMPTY && piece[2] == EMPTY && piece[1] == EMPTY &&
                    piece[0] == WR) {
                    if (!isSquareAttacked(4, BLACK) && !isSquareAttacked(3, BLACK) && !isSquareAttacked(2, BLACK)) {
                        pushMove(4, 2, 0, MF_CASTLING);
                    }
                }
            } else if (us == BLACK && kingSq[BLACK] == 60) {
                if ((castlingRights & CASTLE_BK) && piece[61] == EMPTY && piece[62] == EMPTY && piece[63] == BR) {
                    if (!isSquareAttacked(60, WHITE) && !isSquareAttacked(61, WHITE) && !isSquareAttacked(62, WHITE)) {
                        pushMove(60, 62, 0, MF_CASTLING);
                    }
                }
                if ((castlingRights & CASTLE_BQ) && piece[59] == EMPTY && piece[58] == EMPTY && piece[57] == EMPTY &&
                    piece[56] == BR) {
                    if (!isSquareAttacked(60, WHITE) && !isSquareAttacked(59, WHITE) && !isSquareAttacked(58, WHITE)) {
                        pushMove(60, 58, 0, MF_CASTLING);
                    }
                }
            }
        }
    }
};

struct TTEntry {
    U64 key = 0;
    std::int16_t score = 0;
    std::int8_t depth = -1;
    std::uint8_t flag = 0; // 0=empty, 1=exact, 2=lower, 3=upper
    std::uint32_t move = 0;
};

struct TranspositionTable {
    std::vector<TTEntry> table;
    std::size_t mask = 0;

    explicit TranspositionTable(std::size_t pow2Size = (1u << 20)) {
        std::size_t size = 1;
        while (size < pow2Size) size <<= 1;
        table.resize(size);
        mask = size - 1;
    }

    bool probe(U64 key, int depth, int alpha, int beta, int ply, int& outScore, std::uint32_t& outMove) const {
        const TTEntry& e = table[static_cast<std::size_t>(key) & mask];
        if (e.key != key || e.flag == 0) {
            outMove = 0;
            return false;
        }
        outMove = e.move;
        if (e.depth < depth) return false;
        int score = scoreFromTT((int)e.score, ply);
        if (e.flag == 1) {
            outScore = score;
            return true;
        }
        if (e.flag == 2 && score >= beta) {
            outScore = score;
            return true;
        }
        if (e.flag == 3 && score <= alpha) {
            outScore = score;
            return true;
        }
        return false;
    }

    void store(U64 key, int depth, int score, std::uint8_t flag, std::uint32_t bestMove, int ply) {
        TTEntry& e = table[static_cast<std::size_t>(key) & mask];
        if (e.key != key || depth >= e.depth) {
            e.key = key;
            e.depth = (std::int8_t)depth;
            e.flag = flag;
            e.move = bestMove;
            e.score = (std::int16_t)scoreToTT(score, ply);
        }
    }
};

struct ScoredMove {
    Move move;
    int score = 0;
};

struct Search {
    Board board;
    TranspositionTable tt;

    std::uint64_t nodes = 0;
    bool stop = false;
    std::chrono::steady_clock::time_point start;
    double timeLimitSeconds = 0.0;

    std::array<std::uint32_t, MAX_PLY> killer1{};
    std::array<std::uint32_t, MAX_PLY> killer2{};
    int history[2][64][64]{};

    std::uint32_t rootBestMove = 0;

    explicit Search(std::size_t ttSizePow2 = (1u << 20)) : tt(ttSizePow2) {}

    bool timeUp() const {
        using namespace std::chrono;
        double elapsed = duration<double>(steady_clock::now() - start).count();
        return elapsed >= timeLimitSeconds;
    }

    void maybeCheckTime() {
        if ((nodes & 2047ULL) == 0ULL && timeUp()) stop = true;
    }

    int scoreMove(const Move& m, std::uint32_t ttMove, int ply) {
        std::uint32_t code = encodeMove(m);
        if (code == ttMove) return 10'000'000;

        if (m.promo) {
            return 900'000 + pieceValue((int)m.promo);
        }

        if (m.flags & MF_CAPTURE) {
            int attacker = board.piece[m.from];
            int victim = EMPTY;
            if (m.flags & MF_ENPASSANT) victim = (board.sideToMove == WHITE) ? BP : WP;
            else victim = board.piece[m.to];
            return 500'000 + pieceValue(victim) * 10 - pieceValue(attacker);
        }

        if (code == killer1[ply]) return 400'000;
        if (code == killer2[ply]) return 399'000;

        return history[board.sideToMove][m.from][m.to];
    }

    int quiescence(int alpha, int beta, int ply) {
        nodes++;
        maybeCheckTime();
        if (stop) return board.evaluate();

        int standPat = board.evaluate();
        if (standPat >= beta) return standPat;
        if (standPat > alpha) alpha = standPat;

        std::array<Move, 256> moves{};
        int moveCount = 0;
        board.generatePseudoMoves(moves, moveCount, true);

        // score and sort
        std::array<ScoredMove, 256> scored{};
        for (int i = 0; i < moveCount; ++i) {
            scored[i].move = moves[i];
            scored[i].score = scoreMove(moves[i], 0, ply);
        }
        std::sort(scored.begin(), scored.begin() + moveCount, [](const ScoredMove& a, const ScoredMove& b) {
            return a.score > b.score;
        });

        const int us = board.sideToMove;
        for (int i = 0; i < moveCount; ++i) {
            const Move& m = scored[i].move;
            Undo u;
            board.makeMove(m, u);
            if (board.inCheck(us)) {
                board.unmakeMove(m, u);
                continue;
            }
            int score = -quiescence(-beta, -alpha, ply + 1);
            board.unmakeMove(m, u);
            if (stop) return alpha;

            if (score >= beta) return score;
            if (score > alpha) alpha = score;
        }

        return alpha;
    }

    int negamax(int depth, int alpha, int beta, int ply) {
        nodes++;
        maybeCheckTime();
        if (stop) return 0;

        bool inChk = board.inCheck(board.sideToMove);
        if (inChk && depth < MAX_PLY - 1) depth++; // check extension

        if (depth <= 0) return quiescence(alpha, beta, ply);

        int ttScore = 0;
        std::uint32_t ttMove = 0;
        if (tt.probe(board.hash, depth, alpha, beta, ply, ttScore, ttMove)) {
            return ttScore;
        }

        std::array<Move, 256> moves{};
        int moveCount = 0;
        board.generatePseudoMoves(moves, moveCount, false);

        std::array<ScoredMove, 256> scored{};
        for (int i = 0; i < moveCount; ++i) {
            scored[i].move = moves[i];
            scored[i].score = scoreMove(moves[i], ttMove, ply);
        }
        std::sort(scored.begin(), scored.begin() + moveCount, [](const ScoredMove& a, const ScoredMove& b) {
            return a.score > b.score;
        });

        const int us = board.sideToMove;
        const int alphaOrig = alpha;
        std::uint32_t bestMoveCode = 0;
        int bestScore = -INF;
        bool anyLegal = false;

        for (int i = 0; i < moveCount; ++i) {
            const Move& m = scored[i].move;
            Undo u;
            board.makeMove(m, u);
            if (board.inCheck(us)) {
                board.unmakeMove(m, u);
                continue;
            }
            anyLegal = true;

            int score;
            if (i == 0) {
                score = -negamax(depth - 1, -beta, -alpha, ply + 1);
            } else {
                score = -negamax(depth - 1, -alpha - 1, -alpha, ply + 1);
                if (!stop && score > alpha && score < beta) {
                    score = -negamax(depth - 1, -beta, -alpha, ply + 1);
                }
            }

            board.unmakeMove(m, u);
            if (stop) return 0;

            if (score > bestScore) {
                bestScore = score;
                bestMoveCode = encodeMove(m);
            }

            if (score > alpha) {
                alpha = score;
                if (ply == 0) rootBestMove = bestMoveCode;
            }

            if (alpha >= beta) {
                // beta cutoff
                if (!(m.flags & MF_CAPTURE)) {
                    std::uint32_t code = encodeMove(m);
                    if (killer1[ply] != code) {
                        killer2[ply] = killer1[ply];
                        killer1[ply] = code;
                    }
                    int& h = history[us][m.from][m.to];
                    h += depth * depth;
                    if (h > 1'000'000) h = 1'000'000;
                }
                break;
            }
        }

        if (!anyLegal) {
            if (inChk) return -MATE_SCORE + ply;
            return 0; // stalemate
        }

        std::uint8_t flag = 1; // exact
        if (bestScore <= alphaOrig) flag = 3; // upper
        else if (bestScore >= beta) flag = 2; // lower

        tt.store(board.hash, depth, bestScore, flag, bestMoveCode, ply);
        return bestScore;
    }

    Move findBestMove(double timeoutSeconds) {
        nodes = 0;
        stop = false;
        start = std::chrono::steady_clock::now();
        timeLimitSeconds = timeoutSeconds * 0.98;

        rootBestMove = 0;

        Move best{};
        int bestScore = -INF;

        for (int depth = 1; depth <= MAX_PLY; ++depth) {
            int score = negamax(depth, -INF, INF, 0);
            if (stop) break;
            if (rootBestMove != 0) {
                best = decodeMove(rootBestMove);
                bestScore = score;
            }
            // If mate is found, no need to search deeper.
            if (bestScore > MATE_THRESHOLD || bestScore < -MATE_THRESHOLD) break;
            if (timeUp()) break;
        }

        // Fallback: choose any legal move if search didn't produce one (e.g., extremely short time)
        if (encodeMove(best) == 0) {
            std::array<Move, 256> moves{};
            int moveCount = 0;
            board.generatePseudoMoves(moves, moveCount, false);
            const int us = board.sideToMove;
            for (int i = 0; i < moveCount; ++i) {
                Undo u;
                board.makeMove(moves[i], u);
                bool legal = !board.inCheck(us);
                board.unmakeMove(moves[i], u);
                if (legal) return moves[i];
            }
        }

        return best;
    }
};

static std::string trim(const std::string& s) {
    std::size_t a = 0;
    while (a < s.size() && std::isspace((unsigned char)s[a])) a++;
    std::size_t b = s.size();
    while (b > a && std::isspace((unsigned char)s[b - 1])) b--;
    return s.substr(a, b - a);
}

static void initAttacks() {
    for (int sq = 0; sq < 64; ++sq) {
        int r = rankOf(sq);
        int f = fileOf(sq);

        U64 k = 0;
        for (int dr = -1; dr <= 1; ++dr) {
            for (int df = -1; df <= 1; ++df) {
                if (dr == 0 && df == 0) continue;
                int rr = r + dr, ff = f + df;
                if (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) k |= sqBB(rr * 8 + ff);
            }
        }
        KING_ATTACKS[sq] = k;

        U64 n = 0;
        static constexpr int ndr[8] = {2, 2, 1, 1, -1, -1, -2, -2};
        static constexpr int ndf[8] = {1, -1, 2, -2, 2, -2, 1, -1};
        for (int i = 0; i < 8; ++i) {
            int rr = r + ndr[i], ff = f + ndf[i];
            if (rr >= 0 && rr < 8 && ff >= 0 && ff < 8) n |= sqBB(rr * 8 + ff);
        }
        KNIGHT_ATTACKS[sq] = n;
    }
}

            MG_TABLE[whitePiece][sq] = mg_value[type] + mg_pesto[type][sq];
            EG_TABLE[whitePiece][sq] = eg_value[type] + eg_pesto[type][sq];
            MG_TABLE[blackPiece][sq] = mg_value[type] + mg_pesto[type][flipSq(sq)];
            EG_TABLE[blackPiece][sq] = eg_value[type] + eg_pesto[type][flipSq(sq)];
        }
    };

    // phase increments
    for (int p = 0; p < 13; ++p) PHASE_INC[p] = 0;
    PHASE_INC[WN] = PHASE_INC[BN] = 1;
    PHASE_INC[WB] = PHASE_INC[BB] = 1;
    PHASE_INC[WR] = PHASE_INC[BR] = 2;
    PHASE_INC[WQ] = PHASE_INC[BQ] = 4;

    initPiece(WP, BP, 0);
    initPiece(WN, BN, 1);
    initPiece(WB, BB, 2);
    initPiece(WR, BR, 3);
    initPiece(WQ, BQ, 4);
    initPiece(WK, BK, 5);

    for (int sq = 0; sq < 64; ++sq) {
        MG_TABLE[EMPTY][sq] = 0;
        EG_TABLE[EMPTY][sq] = 0;
    }
}

static void initZobrist() {
    std::mt19937_64 rng(20260520ULL);
    auto rnd = [&]() -> U64 { return rng(); };

    for (int p = 0; p < 13; ++p) {
        for (int sq = 0; sq < 64; ++sq) Z_PIECE[p][sq] = rnd();
    }
    for (int i = 0; i < 16; ++i) Z_CASTLING[i] = rnd();
    for (int i = 0; i < 8; ++i) Z_EPFILE[i] = rnd();
    Z_SIDE = rnd();
}

static void initAll() {
    initAttacks();
    initEvalTables();
    initZobrist();
}

static std::string formatMoveWithCheck(Board& board, const Move& m) {
    // Requires `m` to be legal in current position.
    std::string base;

    if (m.flags & MF_CASTLING) {
        base = (m.to == 6 || m.to == 62) ? "O-O" : "O-O-O";
    } else {
        int movingPiece = board.piece[m.from];
        base.push_back(pieceLetterForOutput(movingPiece));
        base += squareToString(m.from);
        if (m.flags & MF_CAPTURE) base.push_back('x');
        base += squareToString(m.to);
        if (m.promo) {
            base.push_back('/');
            base.push_back(promotionLetter((int)m.promo));
        }
        if (m.flags & MF_ENPASSANT) base += "ep";
    }

    Undo u;
    board.makeMove(m, u);
    bool check = board.inCheck(board.sideToMove);
    bool mate = false;
    if (check) {
        std::array<Move, 256> replies{};
        int rc = 0;
        board.generatePseudoMoves(replies, rc, false);
        const int defender = board.sideToMove;
        mate = true;
        for (int i = 0; i < rc; ++i) {
            Undo u2;
            board.makeMove(replies[i], u2);
            bool legal = !board.inCheck(defender);
            board.unmakeMove(replies[i], u2);
            if (legal) {
                mate = false;
                break;
            }
        }
    }
    board.unmakeMove(m, u);

    if (mate) base.push_back('#');
    else if (check) base.push_back('+');

    return base;
}

} // namespace chessbot

int main() {
    using namespace chessbot;
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    initAll();

    std::string line1;
    if (!std::getline(std::cin, line1)) return 0;
    line1 = trim(line1);
    double timeoutSeconds = 5.0;
    try {
        if (!line1.empty()) timeoutSeconds = std::stod(line1);
    } catch (...) {
        timeoutSeconds = 5.0;
    }

    std::string fen;
    if (!std::getline(std::cin, fen)) return 0;
    fen = trim(fen);

    Search search;
    if (!search.board.setFromFEN(fen)) {
        std::cout << "0000\n";
        return 0;
    }

    Move best = search.findBestMove(timeoutSeconds);

    // If no legal moves exist, output "0000"
    if (encodeMove(best) == 0) {
        std::cout << "0000\n";
        return 0;
    }

    // Ensure chosen move is legal; otherwise fallback to first legal move.
    {
        const int us = search.board.sideToMove;
        Undo u;
        search.board.makeMove(best, u);
        bool legal = !search.board.inCheck(us);
        search.board.unmakeMove(best, u);
        if (!legal) {
            std::array<Move, 256> moves{};
            int moveCount = 0;
            search.board.generatePseudoMoves(moves, moveCount, false);
            for (int i = 0; i < moveCount; ++i) {
                Undo u2;
                search.board.makeMove(moves[i], u2);
                bool ok = !search.board.inCheck(us);
                search.board.unmakeMove(moves[i], u2);
                if (ok) {
                    best = moves[i];
                    break;
                }
            }
        }
    }

    std::cout << formatMoveWithCheck(search.board, best) << '\n';
    return 0;
}
