from __future__ import annotations

import math
from typing import Any, Dict, Iterable, List, Optional, Tuple


def _clamp(x: float, lo: float, hi: float) -> float:
    if x < lo:
        return lo
    if x > hi:
        return hi
    return x


def _finite(x: Any) -> bool:
    try:
        return x is not None and math.isfinite(float(x))
    except Exception:
        return False


def _median(values: List[float]) -> Optional[float]:
    if not values:
        return None
    values = sorted(values)
    n = len(values)
    mid = n // 2
    if n % 2 == 1:
        return float(values[mid])
    return 0.5 * (values[mid - 1] + values[mid])


def _wrap_deg(angle: float) -> float:
    a = (angle + 180.0) % 360.0 - 180.0
    if a == -180.0:
        return 180.0
    return a


def _lpf(prev: float, new: float, tau_s: float, dt: float) -> float:
    if tau_s <= 0.0:
        return new
    alpha = _clamp(dt / (tau_s + dt), 0.0, 1.0)
    return prev + alpha * (new - prev)


def _vote_triplet(state: Dict[str, Any], name: str, field: str) -> Optional[float]:
    values: List[float] = []
    for i in (1, 2, 3):
        v = state.get(f"{name}_{i}_{field}")
        valid = state.get(f"{name}_{i}_valid", True)
        if bool(valid) and _finite(v):
            values.append(float(v))
    return _median(values)


def _best_of(state: Dict[str, Any], keys: Iterable[str]) -> Optional[float]:
    values: List[float] = []
    for k in keys:
        v = state.get(k)
        if _finite(v):
            values.append(float(v))
    return _median(values)


def _best_dme_nm(state: Dict[str, Any]) -> Optional[float]:
    dm = _best_of(state, ("dme_1_nm", "dme_2_nm"))
    if dm is None:
        return None
    if dm < 0.0:
        return None
    return dm


def _best_agl_ft(state: Dict[str, Any], dme_nm: Optional[float]) -> Optional[float]:
    ra = state.get("radio_alt_ft")
    if _finite(ra):
        ra_f = float(ra)
        if -20.0 <= ra_f <= 5000.0:
            return max(0.0, ra_f)
    if dme_nm is None:
        return None
    # Simple 3° ILS geometry: ~318 ft / NM. Add ~50 ft threshold crossing height.
    return max(0.0, 318.0 * float(dme_nm) + 50.0)


def _engine_running(state: Dict[str, Any], idx: int) -> bool:
    key = f"engine_{idx}_running"
    if key in state:
        return bool(state.get(key))
    return True


def _choose_speed_targets(state: Dict[str, Any]) -> Tuple[float, float]:
    vref = float(state.get("target_vref_kt", 137.0)) if _finite(state.get("target_vref_kt")) else 137.0
    tgt = state.get("target_ias_kt")
    if _finite(tgt):
        target_ias = float(tgt)
    else:
        target_ias = vref + 5.0
    # Keep some margin above Vref during the approach.
    target_ias = max(target_ias, vref + 3.0)
    return vref, target_ias


_AP: Dict[str, Any] = {
    "phase": "APPROACH",
    "t": 0.0,
    "i_loc": 0.0,
    "i_gs": 0.0,
    "i_spd": 0.0,
    "trim": 0.0,
    "unstable_t": 0.0,
    "loc_f": 0.0,
    "gs_f": 0.0,
    "ra_f": 1000.0,
    "flaps_est": 0.0,
    "last_roll": 0.0,
    "last_pitch": 0.0,
    "last_thr": 0.55,
    "last_ail": 0.0,
    "last_elv": 0.0,
    "last_rud": 0.0,
    "flare_vs_entry": -700.0,
}


def autopilot_step(state: dict, dt: float) -> dict:
    global _AP

    # Defensive dt handling (the sim should pass sane values, but keep stable anyway).
    try:
        dt_f = float(dt)
    except Exception:
        dt_f = 0.05
    if not math.isfinite(dt_f) or dt_f <= 0.0:
        dt_f = 0.05
    dt_f = _clamp(dt_f, 0.01, 0.2)
    _AP["t"] = float(_AP.get("t", 0.0)) + dt_f

    weight_on_wheels = bool(state.get("weight_on_wheels", False))
    on_ground = bool(state.get("on_ground", False)) or weight_on_wheels

    dme_nm = _best_dme_nm(state)
    ra_ft = _best_agl_ft(state, dme_nm)
    if ra_ft is None:
        ra_ft = float(_AP.get("ra_f", 1000.0))
    _AP["ra_f"] = _lpf(float(_AP.get("ra_f", ra_ft)), float(ra_ft), tau_s=0.6, dt=dt_f)
    ra_ft = float(_AP["ra_f"])

    eng1 = _engine_running(state, 1)
    eng2 = _engine_running(state, 2)
    dual_engine_out = (not eng1) and (not eng2)

    # Voted ILS signals (robust to single failures, bias, flicker).
    loc_raw = _vote_triplet(state, "localizer", "dots")
    gs_raw = _vote_triplet(state, "glideslope", "dots")

    if loc_raw is None:
        loc = float(_AP.get("loc_f", 0.0))
    else:
        loc = _lpf(float(_AP.get("loc_f", loc_raw)), float(loc_raw), tau_s=0.4, dt=dt_f)
    if gs_raw is None:
        gs = float(_AP.get("gs_f", 0.0))
    else:
        gs = _lpf(float(_AP.get("gs_f", gs_raw)), float(gs_raw), tau_s=0.4, dt=dt_f)

    _AP["loc_f"] = float(loc)
    _AP["gs_f"] = float(gs)

    # Aircraft attitude / rates.
    roll_deg = _vote_triplet(state, "irs", "roll_deg")
    pitch_deg = _vote_triplet(state, "irs", "pitch_deg")
    vs_fpm = _vote_triplet(state, "irs", "vs_fpm")
    if roll_deg is None:
        roll_deg = float(_AP.get("last_roll", 0.0))
    if pitch_deg is None:
        pitch_deg = float(_AP.get("last_pitch", 0.0))
    if vs_fpm is None:
        vs_fpm = -700.0

    roll_rate_dps = (float(roll_deg) - float(_AP.get("last_roll", float(roll_deg)))) / dt_f
    _AP["last_roll"] = float(roll_deg)
    _AP["last_pitch"] = float(pitch_deg)

    # Airspeed.
    ias_kt = _vote_triplet(state, "ias", "kt")
    if ias_kt is None:
        ias_kt = float(state.get("target_ias_kt", 140.0)) if _finite(state.get("target_ias_kt")) else 140.0

    # Groundspeed (prefer explicit, else fuse GPS/IRS).
    gs_kt = state.get("groundspeed_kt")
    if _finite(gs_kt):
        gnd_spd_kt = float(gs_kt)
    else:
        gnd_spd_kt = _best_of(
            state,
            (
                "gps_1_gs_kt",
                "gps_2_gs_kt",
                "gps_3_gs_kt",
                "irs_1_gs_kt",
                "irs_2_gs_kt",
                "irs_3_gs_kt",
            ),
        )
        if gnd_spd_kt is None:
            gnd_spd_kt = float(ias_kt)

    aoa_deg = _vote_triplet(state, "aoa", "deg")
    if aoa_deg is None:
        aoa_deg = 0.0

    # Headings for alignment (crosswind decrab).
    hdg_deg = _vote_triplet(state, "irs", "heading_deg")
    if hdg_deg is None:
        hdg_deg = float(state.get("target_hdg_deg", 0.0)) if _finite(state.get("target_hdg_deg")) else 0.0
    runway_hdg_deg = float(state.get("target_hdg_deg", hdg_deg)) if _finite(state.get("target_hdg_deg")) else float(hdg_deg)
    yaw_err_deg = _wrap_deg(runway_hdg_deg - float(hdg_deg))

    vref_kt, target_ias_kt = _choose_speed_targets(state)

    # Phase management / go-around logic.
    phase = str(_AP.get("phase", "APPROACH"))
    flare_h_ft = 50.0
    retard_h_ft = 25.0

    # Any explicit sim flag to force a missed approach.
    must_missed = bool(
        state.get("missed_approach_required", False)
        or state.get("go_around_required", False)
        or state.get("must_go_around", False)
    )

    dh_ft = float(state.get("decision_height_ft", 200.0)) if _finite(state.get("decision_height_ft")) else 200.0
    below_dh = bool(state.get("below_decision_height", False)) or (ra_ft <= dh_ft)

    # Unstable approach detection (time gated to avoid transient go-arounds).
    speed_err = float(target_ias_kt) - float(ias_kt)
    max_dev = max(abs(float(loc)), abs(float(gs)))
    unstable_now = False
    if ra_ft <= 600.0:
        if max_dev > 2.7:
            unstable_now = True
        if abs(speed_err) > 25.0:
            unstable_now = True
        if float(vs_fpm) < -2200.0:
            unstable_now = True
        if float(aoa_deg) > 18.0:
            unstable_now = True
    if ra_ft <= 250.0:
        if max_dev > 2.2:
            unstable_now = True
        if abs(speed_err) > 18.0:
            unstable_now = True
        if float(vs_fpm) < -1600.0:
            unstable_now = True
        if float(aoa_deg) > 17.0:
            unstable_now = True

    unstable_t = float(_AP.get("unstable_t", 0.0))
    if unstable_now and ra_ft <= 800.0:
        unstable_t += dt_f
    else:
        unstable_t = max(0.0, unstable_t - 2.0 * dt_f)
    _AP["unstable_t"] = unstable_t

    if on_ground:
        phase = "ROLLOUT"
    elif phase == "GOAROUND":
        # Let the sim reset us when it decides to; but if it doesn't, exit after 1500' AGL.
        if ra_ft >= 1500.0:
            phase = "APPROACH"
            _AP["i_loc"] = 0.0
            _AP["i_gs"] = 0.0
            _AP["i_spd"] = 0.0
            _AP["unstable_t"] = 0.0
    else:
        if (must_missed and (below_dh or ra_ft <= dh_ft + 50.0)) and (not dual_engine_out) and (ra_ft > 60.0):
            phase = "GOAROUND"
        elif (unstable_t >= 2.5 and (below_dh or ra_ft <= 300.0)) and (not dual_engine_out) and (ra_ft > 60.0):
            phase = "GOAROUND"
        elif ra_ft <= flare_h_ft:
            if phase != "FLARE":
                _AP["flare_vs_entry"] = float(vs_fpm)
            phase = "FLARE"
        else:
            phase = "APPROACH"

    _AP["phase"] = phase

    # Configuration scheduling.
    if _finite(state.get("flaps")):
        flaps_deg = float(state.get("flaps"))
    else:
        flaps_deg = float(_AP.get("flaps_est", 0.0))
    gear_cmd = 1.0

    if phase in ("GOAROUND",):
        # Go-around: clean up gently.
        gear_cmd = 1.0 if ra_ft < 100.0 else 0.0
        flaps_cmd = 15.0 if ra_ft < 400.0 else 10.0
    else:
        # Approach: aim for landing config by ~4 NM / 1500' AGL.
        gear_cmd = 1.0 if (ra_ft <= 2500.0 or (dme_nm is not None and dme_nm <= 8.0)) else 0.0
        if dme_nm is not None:
            if dme_nm > 10.0:
                flaps_cmd = 0.0
            elif dme_nm > 8.0:
                flaps_cmd = 5.0
            elif dme_nm > 6.0:
                flaps_cmd = 10.0
            elif dme_nm > 4.5:
                flaps_cmd = 20.0
            elif dme_nm > 2.5:
                flaps_cmd = 30.0
            else:
                flaps_cmd = 40.0 if abs(speed_err) < 10.0 else 30.0
        else:
            flaps_cmd = 30.0 if ra_ft < 1200.0 else 10.0

    # Limit flap command changes (avoid unrealistic instant jumps).
    flap_rate_dps = 10.0  # deg/sec
    flaps_deg = float(flaps_deg)
    flaps_deg = flaps_deg + _clamp(float(flaps_cmd) - flaps_deg, -flap_rate_dps * dt_f, flap_rate_dps * dt_f)
    _AP["flaps_est"] = float(flaps_deg)

    # Lateral control (LOC -> bank -> aileron), with bank limit scheduling near the ground.
    if ra_ft > 250.0:
        max_bank = 25.0
    elif ra_ft > 100.0:
        max_bank = 15.0
    elif ra_ft > 50.0:
        max_bank = 8.0
    else:
        max_bank = 5.0

    loc_target = float(state.get("target_loc_dots", 0.0)) if _finite(state.get("target_loc_dots")) else 0.0
    loc_err = loc_target - float(loc)

    i_loc = float(_AP.get("i_loc", 0.0))
    if loc_raw is not None:
        i_loc = _clamp(i_loc + _clamp(loc_err, -2.0, 2.0) * dt_f, -2.5, 2.5)
    else:
        i_loc *= 0.98
    _AP["i_loc"] = i_loc

    # Reduce lateral aggressiveness in flare/rollout.
    loc_gain_scale = 1.0
    if phase == "FLARE":
        loc_gain_scale = 0.7
    elif phase == "ROLLOUT":
        loc_gain_scale = 0.9

    desired_bank_deg = _clamp((10.0 * loc_gain_scale) * loc_err + (1.5 * loc_gain_scale) * i_loc, -max_bank, max_bank)
    roll_err = desired_bank_deg - float(roll_deg)
    aileron_cmd = _clamp(0.06 * roll_err - 0.02 * roll_rate_dps, -1.0, 1.0)

    # Vertical and speed control.
    trim = float(_AP.get("trim", 0.0))

    # Go-around control takes full authority.
    if phase == "GOAROUND":
        pitch_target_deg = 12.0 if ra_ft < 800.0 else 8.0
        pitch_err = pitch_target_deg - float(pitch_deg)
        elevator_total = _clamp(0.06 * pitch_err - 0.0002 * float(vs_fpm), -1.0, 1.0)

        # Autothrottle to TOGA (respect engine-out).
        thr_cmd = 1.0
        spdbrk_cmd = 0.0
        spoiler_cmd = 0.0
        autobrake_cmd = 0.0
        rev_cmd = 0.0

    elif phase == "ROLLOUT":
        # On ground: centerline + max decel.
        spd_kt = float(state.get("groundspeed_kt", ias_kt)) if _finite(state.get("groundspeed_kt")) else float(ias_kt)
        spd_kt = max(0.0, spd_kt)

        # Use localizer if available; otherwise use runway heading.
        if loc_raw is not None:
            rud = _clamp(-0.35 * float(loc), -0.8, 0.8)
        else:
            rud = _clamp(0.02 * yaw_err_deg, -0.6, 0.6)
        rudder_cmd = rud

        # Keep wings level (aileron can also help counter a stuck aileron).
        aileron_cmd = _clamp(-0.06 * float(roll_deg) - 0.02 * roll_rate_dps, -0.8, 0.8)

        # Keep nose gently up until ~80 kt to help weight transfer, then neutral.
        elevator_total = 0.15 if spd_kt > 80.0 else 0.0
        thr_cmd = 0.0
        spdbrk_cmd = 1.0
        spoiler_cmd = 1.0
        autobrake_cmd = 1.0 if spd_kt > 40.0 else 0.6

        if spd_kt > 70.0:
            rev_cmd = 1.0
        elif spd_kt > 40.0:
            rev_cmd = 0.6
        elif spd_kt > 20.0:
            rev_cmd = 0.3
        else:
            rev_cmd = 0.0

    elif phase == "FLARE":
        ra_n = _clamp(ra_ft, 0.0, flare_h_ft)
        vs_entry = float(_AP.get("flare_vs_entry", -700.0))
        # Target vs transitions from current approach vs at 50 ft to ~-100 fpm at touchdown.
        vs_target = -100.0 + (ra_n / flare_h_ft) * (vs_entry + 100.0)
        vs_err_fpm = vs_target - float(vs_fpm)

        pitch_target_deg = _clamp(2.0 + 4.0 * (1.0 - ra_n / flare_h_ft), -2.0, 7.0)
        pitch_err = pitch_target_deg - float(pitch_deg)

        elevator_total = _clamp(0.04 * pitch_err + 0.0007 * vs_err_fpm, -1.0, 1.0)

        # Retard thrust to idle through the flare (unless gliding).
        if dual_engine_out:
            thr_cmd = 0.0
        else:
            base_thr = float(_AP.get("last_thr", 0.55))
            if ra_n <= retard_h_ft:
                thr_cmd = 0.0
            else:
                f = _clamp((ra_n - retard_h_ft) / (flare_h_ft - retard_h_ft), 0.0, 1.0)
                thr_cmd = _clamp(base_thr * f, 0.0, 1.0)

        spdbrk_cmd = 0.0
        spoiler_cmd = 0.0
        autobrake_cmd = 0.0
        rev_cmd = 0.0

    else:
        # APPROACH
        gs_target = float(state.get("target_gs_dots", 0.0)) if _finite(state.get("target_gs_dots")) else 0.0
        gs_err = gs_target - float(gs)

        i_gs = float(_AP.get("i_gs", 0.0))
        if gs_raw is not None:
            i_gs = _clamp(i_gs + _clamp(gs_err, -2.0, 2.0) * dt_f, -2.5, 2.5)
        else:
            i_gs *= 0.98
        _AP["i_gs"] = i_gs

        # 3° descent rate estimate (fpm) ~ GS * 5.
        vs_target = -5.0 * float(gnd_spd_kt)
        vs_err = vs_target - float(vs_fpm)

        # Gain schedule: stronger as we get closer.
        gs_gain = 0.7 if ra_ft > 1000.0 else (0.9 if ra_ft > 300.0 else 1.0)

        elevator_total = _clamp(
            gs_gain * (0.32 * gs_err + 0.06 * i_gs) + 0.0004 * vs_err,
            -1.0,
            1.0,
        )

        # Autothrottle for speed (glidepath handled by pitch).
        i_spd = float(_AP.get("i_spd", 0.0))
        i_spd = _clamp(i_spd + _clamp(speed_err, -20.0, 20.0) * dt_f, -200.0, 200.0)
        _AP["i_spd"] = i_spd

        base_thr = 0.55
        thr_cmd = base_thr + 0.02 * speed_err + 0.0008 * i_spd

        # If high on glideslope, bleed energy a bit (helps "too fast" / high path cases).
        thr_cmd += -0.05 * float(gs)
        thr_cmd = _clamp(thr_cmd, 0.0, 1.0)

        # Use speedbrake sparingly if fast and above ~100 ft.
        spdbrk_cmd = 0.0
        if (float(ias_kt) - float(target_ias_kt)) > 20.0 and ra_ft > 300.0:
            spdbrk_cmd = 0.6
        elif (float(ias_kt) - float(target_ias_kt)) > 12.0 and ra_ft > 150.0:
            spdbrk_cmd = 0.35

        spoiler_cmd = 0.0
        autobrake_cmd = 0.0
        rev_cmd = 0.0

        # Dual engine failure: trade controls (pitch for speed, drag for glidepath).
        if dual_engine_out:
            thr_cmd = 0.0
            # Pitch for speed (if fast -> pitch up).
            spd_err_glide = float(ias_kt) - float(target_ias_kt)
            elevator_total = _clamp(0.03 * spd_err_glide - 0.0002 * float(vs_fpm), -1.0, 1.0)
            # Drag for path (if high -> more drag).
            spdbrk_cmd = _clamp(0.35 * max(0.0, float(gs)), 0.0, 1.0)

    # Basic AoA protection (wins over everything except rollout).
    if phase not in ("ROLLOUT",):
        if float(aoa_deg) > 16.0 and float(ias_kt) < (vref_kt + 15.0):
            # Reduce AoA aggressively: nose down + add thrust if available.
            elevator_total = min(elevator_total, -0.35)
            if not dual_engine_out and phase != "FLARE":
                thr_cmd = max(thr_cmd, 0.9)

    # Rudder: combine runway alignment near touchdown + engine asymmetry compensation.
    # In rollout we already computed a dedicated centerline steering rudder.
    if phase != "ROLLOUT":
        rudder_asym = 0.0
        if not dual_engine_out and (eng1 != eng2):
            # Positive means left producing more than right => yaw right => need left rudder (negative).
            rudder_asym = -0.25 * (float(eng1) - float(eng2))
        rudder_align = 0.0
        if phase in ("FLARE",) or (ra_ft < 120.0):
            rudder_align = _clamp(0.02 * yaw_err_deg, -0.4, 0.4)
        rudder_cmd = _clamp(rudder_asym + rudder_align, -1.0, 1.0)

    # Autotrim: slowly move trim toward the total elevator demand.
    trim = float(_AP.get("trim", 0.0))
    trim = _clamp(trim + 0.15 * elevator_total * dt_f, -1.0, 1.0)
    _AP["trim"] = trim
    elevator_cmd = _clamp(elevator_total - trim, -1.0, 1.0)

    # Slew-rate limit primary actuators for stability in turbulence.
    def _slew(prev: float, cmd: float, rate_per_s: float) -> float:
        return prev + _clamp(cmd - prev, -rate_per_s * dt_f, rate_per_s * dt_f)

    aileron_cmd = _slew(float(_AP.get("last_ail", 0.0)), float(aileron_cmd), rate_per_s=3.0)
    elevator_cmd = _slew(float(_AP.get("last_elv", 0.0)), float(elevator_cmd), rate_per_s=3.0)
    rudder_cmd = _slew(float(_AP.get("last_rud", 0.0)), float(rudder_cmd), rate_per_s=4.0)
    thr_cmd = _slew(float(_AP.get("last_thr", 0.55)), float(thr_cmd), rate_per_s=1.5)
    _AP["last_ail"] = float(aileron_cmd)
    _AP["last_elv"] = float(elevator_cmd)
    _AP["last_rud"] = float(rudder_cmd)
    _AP["last_thr"] = float(thr_cmd)

    # Final throttle split + reversers.
    throttle_1 = float(thr_cmd) if eng1 else 0.0
    throttle_2 = float(thr_cmd) if eng2 else 0.0
    reverse_1 = float(rev_cmd) if eng1 else 0.0
    reverse_2 = float(rev_cmd) if eng2 else 0.0

    # After touchdown, don't fight the trim.
    if on_ground:
        elevator_cmd = float(elevator_cmd)
        trim = float(trim)

    return {
        "elevator": _clamp(float(elevator_cmd), -1.0, 1.0),
        "aileron": _clamp(float(aileron_cmd), -1.0, 1.0),
        "rudder": _clamp(float(rudder_cmd), -1.0, 1.0),
        "throttle_1": _clamp(float(throttle_1), 0.0, 1.0),
        "throttle_2": _clamp(float(throttle_2), 0.0, 1.0),
        "elevator_trim": _clamp(float(trim), -1.0, 1.0),
        "flaps": _clamp(float(flaps_deg), 0.0, 40.0),
        "gear": 1.0 if float(gear_cmd) >= 0.5 else 0.0,
        "speedbrake": _clamp(float(spdbrk_cmd), 0.0, 1.0),
        "spoiler": 1.0 if float(spoiler_cmd) >= 0.5 else 0.0,
        "autobrake": _clamp(float(autobrake_cmd), 0.0, 1.0),
        "reverse_1": _clamp(float(reverse_1), 0.0, 1.0),
        "reverse_2": _clamp(float(reverse_2), 0.0, 1.0),
    }
