use std::collections::BinaryHeap;
use std::io::{self, Read};

struct Scanner {
    input: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.input.len() && self.input[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.input.len() {
            return None;
        }
        let mut val: usize = 0;
        while self.idx < self.input.len() && !self.input[self.idx].is_ascii_whitespace() {
            let c = self.input[self.idx];
            if (b'0'..=b'9').contains(&c) {
                val = val * 10 + (c - b'0') as usize;
            }
            self.idx += 1;
        }
        Some(val)
    }
}

fn make_tie(n: usize, seed: u64) -> Vec<u32> {
    let mut state = if seed == 0 { 0x9E3779B97F4A7C15 } else { seed };
    let mut tie = Vec::with_capacity(n);
    for _ in 0..n {
        state ^= state << 13;
        state ^= state >> 7;
        state ^= state << 17;
        tie.push(((state as u32) ^ ((state >> 32) as u32)).wrapping_mul(0x9E3779B9));
    }
    tie
}

fn cover_vertex(
    x: usize,
    adj: &Vec<Vec<usize>>,
    degree: &Vec<usize>,
    tie: &Vec<u32>,
    uncovered: &mut [bool],
    uncovered_remaining: &mut usize,
    gain: &mut [usize],
    stamp: &mut [u32],
    heap: &mut BinaryHeap<(usize, usize, u32, usize, u32)>,
) {
    if !uncovered[x] {
        return;
    }
    uncovered[x] = false;
    *uncovered_remaining -= 1;

    if gain[x] > 0 {
        gain[x] -= 1;
    }
    stamp[x] = stamp[x].wrapping_add(1);
    heap.push((gain[x], degree[x], tie[x], x, stamp[x]));

    for &w in adj[x].iter() {
        if gain[w] > 0 {
            gain[w] -= 1;
        }
        stamp[w] = stamp[w].wrapping_add(1);
        heap.push((gain[w], degree[w], tie[w], w, stamp[w]));
    }
}

fn greedy_dominating_set(adj: &Vec<Vec<usize>>, tie: &Vec<u32>) -> Vec<bool> {
    let n = adj.len();
    if n == 0 {
        return Vec::new();
    }

    let degree: Vec<usize> = adj.iter().map(|v| v.len()).collect();
    let mut uncovered = vec![true; n];
    let mut uncovered_remaining = n;

    let mut gain = vec![0usize; n];
    let mut stamp = vec![0u32; n];
    let mut heap = BinaryHeap::<(usize, usize, u32, usize, u32)>::new();

    for v in 0..n {
        gain[v] = degree[v] + 1;
        heap.push((gain[v], degree[v], tie[v], v, stamp[v]));
    }

    let mut in_set = vec![false; n];

    while uncovered_remaining > 0 {
        let (g, _deg, _tie, mut v, st) = loop {
            if let Some(e) = heap.pop() {
                let v = e.3;
                if e.4 == stamp[v] && e.0 == gain[v] {
                    break e;
                }
            } else {
                let fv = uncovered
                    .iter()
                    .position(|&u| u)
                    .unwrap_or(0);
                break (gain[fv], degree[fv], tie[fv], fv, stamp[fv]);
            }
        };

        if st != stamp[v] || g != gain[v] {
            continue;
        }

        if in_set[v] {
            continue;
        }

        if g == 0 {
            if let Some(fv) = uncovered.iter().position(|&u| u) {
                v = fv;
                if in_set[v] {
                    continue;
                }
            } else {
                break;
            }
        }

        in_set[v] = true;

        cover_vertex(
            v,
            adj,
            &degree,
            tie,
            &mut uncovered,
            &mut uncovered_remaining,
            &mut gain,
            &mut stamp,
            &mut heap,
        );
        for &u in adj[v].iter() {
            cover_vertex(
                u,
                adj,
                &degree,
                tie,
                &mut uncovered,
                &mut uncovered_remaining,
                &mut gain,
                &mut stamp,
                &mut heap,
            );
        }
    }

    in_set
}

fn prune_dominating_set(adj: &Vec<Vec<usize>>, in_set: &mut Vec<bool>) {
    let n = adj.len();
    if n == 0 {
        return;
    }

    let mut cover_count = vec![0u32; n];
    for v in 0..n {
        if in_set[v] {
            cover_count[v] += 1;
            for &u in adj[v].iter() {
                cover_count[u] += 1;
            }
        }
    }

    let mut changed = true;
    while changed {
        changed = false;
        let mut candidates: Vec<usize> = (0..n).filter(|&v| in_set[v]).collect();
        candidates.sort_by_key(|&v| adj[v].len());
        for v in candidates {
            if !in_set[v] {
                continue;
            }
            if cover_count[v] < 2 {
                continue;
            }
            let mut ok = true;
            for &u in adj[v].iter() {
                if cover_count[u] < 2 {
                    ok = false;
                    break;
                }
            }
            if !ok {
                continue;
            }
            in_set[v] = false;
            cover_count[v] -= 1;
            for &u in adj[v].iter() {
                cover_count[u] -= 1;
            }
            changed = true;
        }
    }
}

fn fixup_dominating_set(adj: &Vec<Vec<usize>>, in_set: &mut Vec<bool>) {
    let n = adj.len();
    if n == 0 {
        return;
    }

    let mut dominated = vec![false; n];
    for v in 0..n {
        if in_set[v] {
            dominated[v] = true;
            for &u in adj[v].iter() {
                dominated[u] = true;
            }
        }
    }

    for v in 0..n {
        if dominated[v] {
            continue;
        }
        let mut best = v;
        let mut best_deg = adj[v].len();
        for &u in adj[v].iter() {
            let d = adj[u].len();
            if d > best_deg {
                best = u;
                best_deg = d;
            }
        }
        if !in_set[best] {
            in_set[best] = true;
            dominated[best] = true;
            for &u in adj[best].iter() {
                dominated[u] = true;
            }
        }
    }
}

fn count_set(in_set: &Vec<bool>) -> usize {
    in_set.iter().filter(|&&b| b).count()
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => {
            print!("0\n\n");
            return;
        }
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let Some(u) = sc.next_usize() else { break };
        let Some(v) = sc.next_usize() else { break };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }

    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    let mut edge_count: usize = 0;
    for v in 0..n {
        edge_count += adj[v].len();
    }
    edge_count /= 2;

    let seed1 = 0x1234_5678_9ABC_DEF0u64;
    let seed2 = 0xFEDC_BA98_7654_3211u64;

    let tie1 = make_tie(n, seed1);
    let mut best = greedy_dominating_set(&adj, &tie1);
    prune_dominating_set(&adj, &mut best);
    fixup_dominating_set(&adj, &mut best);
    prune_dominating_set(&adj, &mut best);
    let mut best_size = count_set(&best);

    if n <= 200_000 && edge_count <= 2_000_000 {
        let tie2 = make_tie(n, seed2);
        let mut alt = greedy_dominating_set(&adj, &tie2);
        prune_dominating_set(&adj, &mut alt);
        fixup_dominating_set(&adj, &mut alt);
        prune_dominating_set(&adj, &mut alt);
        let alt_size = count_set(&alt);
        if alt_size < best_size {
            best = alt;
            best_size = alt_size;
        }
    }

    let mut result: Vec<usize> = (0..n).filter(|&v| best[v]).collect();
    result.sort_unstable();

    let mut out = String::new();
    out.push_str(&format!("{}\n", best_size));
    if !result.is_empty() {
        for (i, v) in result.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
    }
    out.push('\n');
    print!("{}", out);
}
