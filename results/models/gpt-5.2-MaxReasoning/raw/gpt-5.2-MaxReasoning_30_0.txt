import sys
import time
import random
from array import array


class FastScanner:
    __slots__ = ("buf", "data", "idx", "size")

    def __init__(self, bufsize: int = 1 << 20):
        self.buf = sys.stdin.buffer
        self.data = bytearray(bufsize)
        self.idx = 0
        self.size = 0

    def _fill(self) -> int:
        self.size = self.buf.readinto(self.data)
        self.idx = 0
        return self.size

    def next_int(self):
        data = self.data
        idx = self.idx
        size = self.size

        while True:
            while idx < size and data[idx] <= 32:
                idx += 1
            if idx < size:
                break
            if not self._fill():
                self.idx = 0
                return None
            data = self.data
            idx = 0
            size = self.size

        sign = 1
        if data[idx] == 45:  # '-'
            sign = -1
            idx += 1

        num = 0
        while True:
            if idx >= size:
                self.idx = idx
                if not self._fill():
                    self.idx = 0
                    return sign * num
                data = self.data
                idx = 0
                size = self.size
            b = data[idx]
            if b <= 32:
                idx += 1
                break
            num = num * 10 + (b - 48)
            idx += 1

        self.idx = idx
        return sign * num


def main() -> None:
    scan = FastScanner()
    n = scan.next_int()
    m = scan.next_int()
    if n is None or m is None:
        return

    # Store literals densely: clause i is lits[3*i : 3*i+3]
    lits = array("i")
    lits_append = lits.append

    # Count, for each variable, how many clauses contain it (dedup per clause)
    occ_ofs = array("I", [0]) * (n + 2)  # 1..n used; n+1 is sentinel
    abs_ = abs

    for _ in range(m):
        a = scan.next_int()
        b = scan.next_int()
        c = scan.next_int()
        # Input guaranteed well-formed; if EOF mid-stream, stop.
        if a is None or b is None or c is None:
            m = len(lits) // 3
            break
        lits_append(a)
        lits_append(b)
        lits_append(c)

        v0 = abs_(a)
        v1 = abs_(b)
        v2 = abs_(c)
        occ_ofs[v0] += 1
        if v1 != v0:
            occ_ofs[v1] += 1
        if v2 != v0 and v2 != v1:
            occ_ofs[v2] += 1

    # Build CSR-style occurrence list: occ_list[occ_ofs[v] : occ_ofs[v+1]] are clauses containing v.
    total = 0
    for v in range(1, n + 1):
        total += occ_ofs[v]
        occ_ofs[v] = total  # temporarily store end pointers
    occ_ofs[n + 1] = total

    occ_list = array("I", [0]) * total

    # Fill using end-pointers and decrementing (no extra cursor array needed).
    for clause_idx in range(m):
        base = 3 * clause_idx
        a = lits[base]
        b = lits[base + 1]
        c = lits[base + 2]
        v0 = abs_(a)
        v1 = abs_(b)
        v2 = abs_(c)

        pos = occ_ofs[v0] - 1
        occ_ofs[v0] = pos
        occ_list[pos] = clause_idx

        if v1 != v0:
            pos = occ_ofs[v1] - 1
            occ_ofs[v1] = pos
            occ_list[pos] = clause_idx

        if v2 != v0 and v2 != v1:
            pos = occ_ofs[v2] - 1
            occ_ofs[v2] = pos
            occ_list[pos] = clause_idx

    # Now occ_ofs[v] are start offsets (and occ_ofs[v+1] are ends).
    assignment = bytearray(n + 1)  # 0/1; index 0 unused
    clause_sat = bytearray(m)  # 0/1: clause satisfied under current assignment
    unsat = []  # list of clause indices that are (or used to be) unsatisfied

    rnd = random.Random(0xC0FFEE)
    getbits = rnd.getrandbits
    randrange = rnd.randrange
    random_ = rnd.random

    def randomize_assignment() -> None:
        i = 1
        while i <= n:
            bits = getbits(64)
            end = i + 64
            if end > n + 1:
                end = n + 1
            while i < end:
                assignment[i] = bits & 1
                bits >>= 1
                i += 1

    def recompute_clause_status():
        unsat.clear()
        num_unsatisfied = 0
        assn = assignment
        csat = clause_sat
        ls = lits
        for clause_idx in range(m):
            base = 3 * clause_idx
            lit = ls[base]
            satisfied = 0
            if lit > 0:
                if assn[lit]:
                    satisfied = 1
            else:
                if assn[-lit] == 0:
                    satisfied = 1
            if not satisfied:
                lit = ls[base + 1]
                if lit > 0:
                    if assn[lit]:
                        satisfied = 1
                else:
                    if assn[-lit] == 0:
                        satisfied = 1
            if not satisfied:
                lit = ls[base + 2]
                if lit > 0:
                    if assn[lit]:
                        satisfied = 1
                else:
                    if assn[-lit] == 0:
                        satisfied = 1

            csat[clause_idx] = satisfied
            if not satisfied:
                unsat.append(clause_idx)
                num_unsatisfied += 1
        return num_unsatisfied

    # Start from a random assignment.
    randomize_assignment()
    num_unsat = recompute_clause_status()

    # Stochastic local search (ProbSAT-style) with a small number of restarts.
    deadline = time.perf_counter() + 295.0
    noise_prob = 0.02
    weight_scale = 1 << 20

    assn = assignment
    csat = clause_sat
    ls = lits
    ofs = occ_ofs
    occ = occ_list
    unsat_append = unsat.append

    for restart in range(4):
        if num_unsat == 0 or time.perf_counter() >= deadline:
            break
        if restart:
            randomize_assignment()
            num_unsat = recompute_clause_status()
            if num_unsat == 0:
                break

        while num_unsat and time.perf_counter() < deadline:
            # Pick a random currently-unsatisfied clause (lazy-clean the list).
            while True:
                if not unsat:
                    # Shouldn't happen, but recover by full scan.
                    num_unsat = recompute_clause_status()
                    break
                pick = randrange(len(unsat))
                clause_idx = unsat[pick]
                if csat[clause_idx] == 0:
                    break
                unsat[pick] = unsat[-1]
                unsat.pop()

            if num_unsat == 0:
                break

            base = 3 * clause_idx
            l0 = ls[base]
            l1 = ls[base + 1]
            l2 = ls[base + 2]

            v0 = abs_(l0)
            v1 = abs_(l1)
            v2 = abs_(l2)
            candidates = [v0]
            if v1 != v0:
                candidates.append(v1)
            if v2 != v0 and v2 != v1:
                candidates.append(v2)

            if len(candidates) == 1 or random_() < noise_prob:
                var = candidates[randrange(len(candidates))]
            else:
                # Compute exact breakcount by simulating the flip on clauses containing each variable.
                cum0 = 0
                cum1 = 0
                cum2 = 0
                total_w = 0

                for j, var_j in enumerate(candidates):
                    br = 0
                    start = ofs[var_j]
                    end = ofs[var_j + 1]
                    for ii in range(start, end):
                        cl = occ[ii]
                        if csat[cl] == 0:
                            continue
                        b = 3 * cl

                        # Is the clause satisfied after flipping var_j?
                        lit = ls[b]
                        if lit > 0:
                            vv = lit
                            val = assn[vv]
                            if vv == var_j:
                                val ^= 1
                            if val:
                                continue
                        else:
                            vv = -lit
                            val = assn[vv]
                            if vv == var_j:
                                val ^= 1
                            if val == 0:
                                continue

                        lit = ls[b + 1]
                        if lit > 0:
                            vv = lit
                            val = assn[vv]
                            if vv == var_j:
                                val ^= 1
                            if val:
                                continue
                        else:
                            vv = -lit
                            val = assn[vv]
                            if vv == var_j:
                                val ^= 1
                            if val == 0:
                                continue

                        lit = ls[b + 2]
                        if lit > 0:
                            vv = lit
                            val = assn[vv]
                            if vv == var_j:
                                val ^= 1
                            if val:
                                continue
                        else:
                            vv = -lit
                            val = assn[vv]
                            if vv == var_j:
                                val ^= 1
                            if val == 0:
                                continue

                        br += 1

                    denom = (br + 1) * (br + 1)
                    w = weight_scale // denom
                    if w <= 0:
                        w = 1
                    total_w += w
                    if j == 0:
                        cum0 = total_w
                    elif j == 1:
                        cum1 = total_w
                    else:
                        cum2 = total_w

                r = randrange(total_w)
                if r < cum0:
                    var = candidates[0]
                elif len(candidates) == 2 or r < cum1:
                    var = candidates[1]
                else:
                    var = candidates[2]

            # Flip var and update all impacted clauses.
            assn[var] ^= 1
            start = ofs[var]
            end = ofs[var + 1]
            for ii in range(start, end):
                cl = occ[ii]
                old = csat[cl]
                b = 3 * cl

                lit = ls[b]
                new = 0
                if lit > 0:
                    if assn[lit]:
                        new = 1
                else:
                    if assn[-lit] == 0:
                        new = 1
                if not new:
                    lit = ls[b + 1]
                    if lit > 0:
                        if assn[lit]:
                            new = 1
                    else:
                        if assn[-lit] == 0:
                            new = 1
                if not new:
                    lit = ls[b + 2]
                    if lit > 0:
                        if assn[lit]:
                            new = 1
                    else:
                        if assn[-lit] == 0:
                            new = 1

                if new != old:
                    csat[cl] = new
                    if new == 0:
                        num_unsat += 1
                        unsat_append(cl)
                    else:
                        num_unsat -= 1

    # Output.
    out = sys.stdout.write
    out("SAT\n")

    chunk = []
    chunk_append = chunk.append
    CHUNK_SIZE = 200_000
    first = True
    for i in range(1, n + 1):
        chunk_append("1" if assn[i] else "0")
        if len(chunk) >= CHUNK_SIZE:
            s = " ".join(chunk)
            if first:
                out(s)
                first = False
            else:
                out(" " + s)
            chunk.clear()
    if chunk:
        s = " ".join(chunk)
        if first:
            out(s)
        else:
            out(" " + s)
    out("\n")


if __name__ == "__main__":
    main()
