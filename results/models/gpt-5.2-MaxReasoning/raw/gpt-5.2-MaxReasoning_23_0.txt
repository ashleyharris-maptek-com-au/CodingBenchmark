use std::io::{self, BufRead, Write};
use std::convert::TryFrom;
use std::sync::atomic::{AtomicBool, AtomicU64, Ordering};
use std::sync::Arc;
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug, Default)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

struct Input<R: BufRead> {
    reader: R,
    token: Vec<u8>,
}

impl<R: BufRead> Input<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            token: Vec::with_capacity(32),
        }
    }

    fn next_token_bytes(&mut self) -> Option<&[u8]> {
        self.token.clear();
        loop {
            let buf = self.reader.fill_buf().ok()?;
            let buf_len = buf.len();
            if buf.is_empty() {
                return if self.token.is_empty() {
                    None
                } else {
                    Some(&self.token)
                };
            }

            let mut i = 0usize;
            while i < buf.len() {
                let b = buf[i];
                if self.token.is_empty() {
                    if b.is_ascii_whitespace() {
                        i += 1;
                        continue;
                    }
                    self.token.push(b);
                    i += 1;
                } else {
                    if b.is_ascii_whitespace() {
                        break;
                    }
                    self.token.push(b);
                    i += 1;
                }
            }

            self.reader.consume(i);

            if !self.token.is_empty() {
                if i < buf_len {
                    return Some(&self.token);
                }
                // Token may continue in next buffer.
            }
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token_bytes()?;
        std::str::from_utf8(t).ok()?.parse::<usize>().ok()
    }

    fn next_f64(&mut self) -> Option<f64> {
        let t = self.next_token_bytes()?;
        std::str::from_utf8(t).ok()?.parse::<f64>().ok()
    }

    fn scan_map(&mut self, width: usize, height: usize, corridor_min_x: i64, corridor_max_x: i64) -> MapInfo {
        let mut info = MapInfo::default();
        let total_cells = match (width as u64).checked_mul(height as u64) {
            Some(v) => v,
            None => {
                // Unreasonably large; attempt to skip until EOF.
                info.skip_until_eof = true;
                return info;
            }
        };

        let mut x: i64 = 0;
        let mut y: i64 = 0;
        let mut cells_read: u64 = 0;

        while cells_read < total_cells {
            let buf = match self.reader.fill_buf() {
                Ok(b) => b,
                Err(_) => break,
            };
            if buf.is_empty() {
                break;
            }

            let mut consumed = 0usize;
            for &b in buf {
                consumed += 1;
                if b.is_ascii_whitespace() {
                    continue;
                }

                // Process a single map cell.
                match b {
                    b'L' => {
                        info.lz_found = true;
                        info.lz_min_x = info.lz_min_x.min(x);
                        info.lz_max_x = info.lz_max_x.max(x);
                        info.lz_min_y = info.lz_min_y.min(y);
                        info.lz_max_y = info.lz_max_y.max(y);
                        if x >= corridor_min_x && x <= corridor_max_x {
                            info.corridor_min_solid_y = info.corridor_min_solid_y.min(y);
                        }
                    }
                    b'#' => {
                        if x >= corridor_min_x && x <= corridor_max_x {
                            info.corridor_min_solid_y = info.corridor_min_solid_y.min(y);
                        }
                    }
                    _ => {}
                }

                cells_read += 1;
                x += 1;
                if x >= width as i64 {
                    x = 0;
                    y += 1;
                }
                if cells_read >= total_cells {
                    break;
                }
            }
            self.reader.consume(consumed);
        }

        info
    }
}

#[derive(Clone, Copy, Debug)]
struct MapInfo {
    lz_found: bool,
    lz_min_x: i64,
    lz_max_x: i64,
    lz_min_y: i64,
    lz_max_y: i64,
    corridor_min_solid_y: i64,
    skip_until_eof: bool,
}

impl Default for MapInfo {
    fn default() -> Self {
        Self {
            lz_found: false,
            lz_min_x: i64::MAX,
            lz_max_x: i64::MIN,
            lz_min_y: i64::MAX,
            lz_max_y: i64::MIN,
            corridor_min_solid_y: i64::MAX,
            skip_until_eof: false,
        }
    }
}

struct Controller {
    gravity: f64,
    max_thrust: f64,
    pad_x: f64,
    pad_y: f64,
    zone_min_x: f64,
    zone_max_x: f64,
    cruise_y: f64,
}

impl Controller {
    fn command(&self, s: State) -> (f64, f64) {
        let x_err = self.pad_x - s.x;
        let alt_to_pad = (self.pad_y - s.y).max(0.0);

        let zone_width = (self.zone_max_x - self.zone_min_x).max(0.0);
        let x_tol = (0.25 * zone_width).max(1.0);

        // Phase selection: hold a "cruise" altitude while moving horizontally, then descend.
        let desired_y = if x_err.abs() > x_tol { self.cruise_y } else { self.pad_y };
        let y_err = desired_y - s.y;

        // Horizontal speed limit shrinks near touchdown.
        let mut vx_max = (alt_to_pad.sqrt() * 1.5 + 1.0).clamp(1.0, 18.0);
        if alt_to_pad < 10.0 {
            vx_max = vx_max.min(4.0);
        }
        if alt_to_pad < 4.0 {
            vx_max = vx_max.min(1.5);
        }

        let mut vx_target = (x_err * 0.40).clamp(-vx_max, vx_max);
        if alt_to_pad < 6.0 && s.x >= self.zone_min_x && s.x <= self.zone_max_x {
            vx_target = 0.0;
        }

        // Vertical speed limit based on how quickly we can brake with max thrust.
        let a_up_max = (self.max_thrust - self.gravity).max(0.1);
        let v_allow = (2.0 * a_up_max * alt_to_pad.max(0.0)).sqrt();
        let mut vy_limit = (0.60 * v_allow + 0.5).clamp(0.5, 15.0);
        if desired_y != self.pad_y {
            vy_limit = vy_limit.min(5.0);
        }
        if alt_to_pad < 5.0 {
            vy_limit = vy_limit.min(2.0);
        }
        if alt_to_pad < 2.0 {
            vy_limit = vy_limit.min(0.9);
        }

        let mut vy_target = if y_err > 0.0 {
            (y_err * 0.55).min(vy_limit)
        } else {
            (y_err * 0.65).max(-4.0)
        };

        // Ensure we actually "touch down" instead of hovering just above the pad.
        if desired_y == self.pad_y && y_err.abs() < 1.0 {
            vy_target = vy_target.max(0.25);
        }

        // PD-style desired net accelerations.
        let mut ax_des = (vx_target - s.vx) * 1.20 + x_err * 0.06;
        let mut ay_des_net = (vy_target - s.vy) * 1.50 + y_err * 0.22;

        // Limit requested accelerations to sane ranges.
        let ax_lim = 0.7 * self.max_thrust;
        ax_des = ax_des.clamp(-ax_lim, ax_lim);
        ay_des_net = ay_des_net.clamp(-self.max_thrust, self.gravity + 2.0);

        // Convert desired net acceleration to required engine "upward" component.
        // Net ay (down) = gravity - engine_up.
        let engine_up = (self.gravity - ay_des_net).clamp(0.0, self.max_thrust);
        let mut engine_x = ax_des;

        // If we're basically not thrusting, don't bother rotating for lateral control.
        if engine_up < 0.2 {
            engine_x = 0.0;
        }

        let (angle_cmd, thrust_frac) = self.solve_thrust_vector(engine_x, engine_up, alt_to_pad);

        // Angle controller: command a turn rate to drive angle -> angle_cmd.
        let angle_error = angle_cmd - s.angle;
        let turn = (angle_error * 2.5 - s.angular_vel * 0.7).clamp(-1.0, 1.0);

        (sanitize01(thrust_frac), sanitize11(turn))
    }

    fn solve_thrust_vector(&self, engine_x: f64, engine_up: f64, alt_to_pad: f64) -> (f64, f64) {
        if engine_up <= 1e-9 || self.max_thrust <= 1e-9 {
            return (0.0, 0.0);
        }
        if engine_up >= self.max_thrust {
            // Need full vertical authority; keep upright.
            return (0.0, 1.0);
        }

        let global_max_tilt: f64 = 0.80; // rad
        let mut angle_des = engine_x.atan2(engine_up);

        // Reduce tilt near the ground.
        let tilt_scale = (alt_to_pad / 20.0).clamp(0.0, 1.0);
        angle_des *= tilt_scale;

        // Ensure the commanded tilt still allows producing the required vertical component.
        let max_tilt_vert = (engine_up / self.max_thrust).clamp(0.0, 1.0).acos();
        let tilt_limit = global_max_tilt.min(max_tilt_vert);
        let angle_cmd = angle_des.clamp(-tilt_limit, tilt_limit);

        let cos_a = angle_cmd.cos().max(1e-6);
        let required_mag = engine_up / cos_a; // <= max_thrust due to tilt_limit
        let thrust_frac = (required_mag / self.max_thrust).clamp(0.0, 1.0);

        (angle_cmd, thrust_frac)
    }
}

fn sanitize01(v: f64) -> f64 {
    if v.is_finite() {
        v.clamp(0.0, 1.0)
    } else {
        0.0
    }
}

fn sanitize11(v: f64) -> f64 {
    if v.is_finite() {
        v.clamp(-1.0, 1.0)
    } else {
        0.0
    }
}

fn store_f64(atom: &AtomicU64, v: f64) {
    atom.store(v.to_bits(), Ordering::Relaxed);
}

fn clamp_i64(v: i64, lo: i64, hi: i64) -> i64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

fn main() {
    let stdin = io::stdin();
    let mut input = Input::new(io::BufReader::new(stdin.lock()));

    let width = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let height = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let gravity = match input.next_f64() {
        Some(v) => v.abs(),
        None => return,
    };
    let max_thrust = match input.next_f64() {
        Some(v) => v.abs().max(1e-6),
        None => return,
    };
    let _fuel = match input.next_f64() {
        Some(v) => v,
        None => return,
    };

    let start_x = match input.next_f64() {
        Some(v) => v,
        None => return,
    };
    let start_y = match input.next_f64() {
        Some(v) => v,
        None => return,
    };
    let target_x = match input.next_f64() {
        Some(v) => v,
        None => return,
    };
    let target_y = match input.next_f64() {
        Some(v) => v,
        None => return,
    };

    let max_x_idx = match i64::try_from(width.saturating_sub(1)) {
        Ok(v) => v,
        Err(_) => i64::MAX,
    };
    let corridor_margin = 4.0;
    let corridor_min_x = clamp_i64((start_x.min(target_x) - corridor_margin).floor() as i64, 0, max_x_idx);
    let corridor_max_x = clamp_i64((start_x.max(target_x) + corridor_margin).ceil() as i64, 0, max_x_idx);

    let map_info = input.scan_map(width, height, corridor_min_x, corridor_max_x);

    // Landing zone bounds in world x coordinates. If no zone, fall back to a small window around target_x.
    let (zone_min_x, zone_max_x, pad_x) = if map_info.lz_found {
        let min_x = map_info.lz_min_x as f64;
        let max_x = map_info.lz_max_x as f64 + 1.0;
        let center = 0.5 * (map_info.lz_min_x as f64 + map_info.lz_max_x as f64 + 1.0);
        (min_x, max_x, center)
    } else {
        (target_x - 1.0, target_x + 1.0, target_x)
    };

    // Compute a conservative "cruise" ceiling (smaller y = higher altitude) over the corridor.
    let cruise_y = if map_info.corridor_min_solid_y != i64::MAX {
        ((map_info.corridor_min_solid_y as f64) - 2.5).max(0.0)
    } else {
        // No obstacles observed in corridor; don't force a climb.
        start_y
    };

    let controller = Controller {
        gravity,
        max_thrust,
        pad_x,
        pad_y: target_y,
        zone_min_x,
        zone_max_x,
        cruise_y,
    };

    let thrust_bits = Arc::new(AtomicU64::new(0.0f64.to_bits()));
    let turn_bits = Arc::new(AtomicU64::new(0.0f64.to_bits()));
    let stop = Arc::new(AtomicBool::new(false));

    // Output thread: always emit commands at >= 20 Hz.
    {
        let thrust_bits = Arc::clone(&thrust_bits);
        let turn_bits = Arc::clone(&turn_bits);
        let stop = Arc::clone(&stop);
        thread::spawn(move || {
            let stdout = io::stdout();
            let mut out = io::BufWriter::new(stdout.lock());

            let period = Duration::from_millis(20);
            let mut next = Instant::now();
            loop {
                if stop.load(Ordering::Relaxed) {
                    break;
                }
                let thrust = f64::from_bits(thrust_bits.load(Ordering::Relaxed));
                let turn = f64::from_bits(turn_bits.load(Ordering::Relaxed));
                let _ = writeln!(out, "{:.6} {:.6}", sanitize01(thrust), sanitize11(turn));
                let _ = out.flush();

                next += period;
                let now = Instant::now();
                if next > now {
                    thread::sleep(next - now);
                } else {
                    next = now;
                }
            }
        });
    }

    // Main loop: read states and update command atomics.
    loop {
        let x = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let y = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let vx = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let vy = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let angle = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let angular_vel = match input.next_f64() {
            Some(v) => v,
            None => break,
        };
        let fuel_remaining = match input.next_f64() {
            Some(v) => v,
            None => break,
        };

        let s = State {
            x,
            y,
            vx,
            vy,
            angle,
            angular_vel,
            fuel_remaining,
        };

        let (thrust, turn) = controller.command(s);
        store_f64(&thrust_bits, thrust);
        store_f64(&turn_bits, turn);
    }

    stop.store(true, Ordering::Relaxed);
}
