using System;
using System.IO;
using System.Text;
using System.Diagnostics;
using System.Collections.Generic;

public static class Program
{
    private struct Box
    {
        public int Id;
        public long W;
        public long H;
        public long D;
        public long Volume;
        public long MaxDim;
        public long MinDim;
    }

    private struct Placement
    {
        public int Id;
        public long X;
        public long Y;
        public long Z;
    }

    private struct Space
    {
        public long X;
        public long Y;
        public long Z;
        public long W;
        public long H;
        public long D;
    }

    private enum Axis : byte { X = 0, Y = 1, Z = 2 }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());

        int n;
        long containerW, containerH, containerD;
        try
        {
            n = (int)fs.NextLong();
            containerW = fs.NextLong();
            containerH = fs.NextLong();
            containerD = fs.NextLong();
        }
        catch
        {
            return;
        }

        if (n <= 0 || containerW <= 0 || containerH <= 0 || containerD <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            long w = fs.NextLong();
            long h = fs.NextLong();
            long d = fs.NextLong();
            if (w <= 0 || h <= 0 || d <= 0) continue;

            // Output format does not include orientation; treat input (w,h,d) as fixed.
            if (w > containerW || h > containerH || d > containerD) continue;

            long max = w;
            if (h > max) max = h;
            if (d > max) max = d;
            long min = w;
            if (h < min) min = h;
            if (d < min) min = d;

            boxes.Add(new Box
            {
                Id = i,
                W = w,
                H = h,
                D = d,
                Volume = Mul3Saturated(w, h, d),
                MaxDim = max,
                MinDim = min
            });
        }

        var packed = Solve(boxes.ToArray(), containerW, containerH, containerD);

        var sb = new StringBuilder(packed.Count * 24);
        sb.AppendLine(packed.Count.ToString());
        for (int i = 0; i < packed.Count; i++)
        {
            var p = packed[i];
            sb.Append(p.Id).Append(' ')
              .Append(p.X).Append(' ')
              .Append(p.Y).Append(' ')
              .Append(p.Z).Append('\n');
        }
        Console.Write(sb.ToString());
    }

    private static List<Placement> Solve(Box[] boxes, long W, long H, long D)
    {
        int n = boxes.Length;
        var best = new List<Placement>();
        int bestCount = 0;

        if (n == 0) return best;

        var sw = Stopwatch.StartNew();
        const int timeLimitMs = 28500;

        if (n <= 3000)
        {
            // More flexible (but slower): space-partitioning guillotine packing.
            var splitOrders = (n <= 800)
                ? AllSplitOrders
                : TwoSplitOrders;

            for (int strategy = 0; strategy < SmallComparers.Length; strategy++)
            {
                if (sw.ElapsedMilliseconds > timeLimitMs) break;

                var copy = (Box[])boxes.Clone();
                Array.Sort(copy, SmallComparers[strategy]);

                for (int so = 0; so < splitOrders.Length; so++)
                {
                    if (sw.ElapsedMilliseconds > timeLimitMs) break;

                    var o = splitOrders[so];
                    var placements = PackGuillotine(copy, W, H, D, o[0], o[1], o[2]);
                    if (placements.Count > bestCount)
                    {
                        best = placements;
                        bestCount = placements.Count;
                        if (bestCount == n) return best;
                    }
                }
            }
        }
        else
        {
            // Large-N: very fast layer/row packing; try several orderings.
            for (int strategy = 0; strategy < LargeComparers.Length; strategy++)
            {
                if (sw.ElapsedMilliseconds > timeLimitMs) break;

                var copy = (Box[])boxes.Clone();
                Array.Sort(copy, LargeComparers[strategy]);

                var placements = PackShelf(copy, W, H, D);
                if (placements.Count > bestCount)
                {
                    best = placements;
                    bestCount = placements.Count;
                    if (bestCount == n) return best;
                }
            }
        }

        return best;
    }

    private static List<Placement> PackShelf(Box[] boxes, long W, long H, long D)
    {
        var res = new List<Placement>(Math.Min(boxes.Length, 4096));

        long x = 0, y = 0, z = 0;
        long rowHeight = 0;
        long layerDepth = 0;

        for (int i = 0; i < boxes.Length; i++)
        {
            var b = boxes[i];
            if (TryPlaceShelf(in b, W, H, D, ref x, ref y, ref z, ref rowHeight, ref layerDepth, out long px, out long py, out long pz))
            {
                res.Add(new Placement { Id = b.Id, X = px, Y = py, Z = pz });
            }
        }

        return res;
    }

    private static bool TryPlaceShelf(in Box b, long W, long H, long D,
        ref long x, ref long y, ref long z, ref long rowHeight, ref long layerDepth,
        out long px, out long py, out long pz)
    {
        px = py = pz = 0;
        if (z + b.D > D) return false;

        while (true)
        {
            if (x + b.W <= W && y + b.H <= H)
            {
                px = x;
                py = y;
                pz = z;

                x += b.W;
                if (b.H > rowHeight) rowHeight = b.H;
                if (b.D > layerDepth) layerDepth = b.D;
                return true;
            }

            if (x + b.W > W)
            {
                x = 0;
                y += rowHeight;
                rowHeight = 0;
                continue;
            }

            // Height overflow: start a new layer.
            x = 0;
            y = 0;
            z += layerDepth;
            layerDepth = 0;
            rowHeight = 0;
            if (z + b.D > D) return false;
        }
    }

    private static List<Placement> PackGuillotine(Box[] boxes, long W, long H, long D, Axis first, Axis second, Axis third)
    {
        var free = new List<Space>(boxes.Length * 3 + 1)
        {
            new Space { X = 0, Y = 0, Z = 0, W = W, H = H, D = D }
        };

        var res = new List<Placement>(Math.Min(boxes.Length, 4096));

        for (int i = 0; i < boxes.Length; i++)
        {
            var b = boxes[i];
            int best = -1;
            long bestZ = 0, bestY = 0, bestX = 0;
            long bestSlackMax = 0, bestSlackSum = 0;
            bool bestInit = false;

            for (int si = 0; si < free.Count; si++)
            {
                var s = free[si];
                if (b.W > s.W || b.H > s.H || b.D > s.D) continue;

                long slackX = s.W - b.W;
                long slackY = s.H - b.H;
                long slackZ = s.D - b.D;
                long slackMax = Max3(slackX, slackY, slackZ);
                long slackSum = slackX + slackY + slackZ;

                if (!bestInit || IsBetterCandidate(s, slackMax, slackSum, bestZ, bestY, bestX, bestSlackMax, bestSlackSum))
                {
                    bestInit = true;
                    best = si;
                    bestZ = s.Z;
                    bestY = s.Y;
                    bestX = s.X;
                    bestSlackMax = slackMax;
                    bestSlackSum = slackSum;
                }
            }

            if (best < 0) continue;

            var chosen = free[best];
            res.Add(new Placement { Id = b.Id, X = chosen.X, Y = chosen.Y, Z = chosen.Z });

            int last = free.Count - 1;
            free[best] = free[last];
            free.RemoveAt(last);

            SplitSpace(in chosen, in b, first, second, third, free);
        }

        return res;
    }

    private static bool IsBetterCandidate(in Space s, long slackMax, long slackSum,
        long bestZ, long bestY, long bestX, long bestSlackMax, long bestSlackSum)
    {
        if (s.Z != bestZ) return s.Z < bestZ;
        if (s.Y != bestY) return s.Y < bestY;
        if (s.X != bestX) return s.X < bestX;
        if (slackMax != bestSlackMax) return slackMax < bestSlackMax;
        return slackSum < bestSlackSum;
    }

    private static void SplitSpace(in Space space, in Box box, Axis first, Axis second, Axis third, List<Space> free)
    {
        long boxFirst = GetDim(in box, first);
        long boxSecond = GetDim(in box, second);
        long boxThird = GetDim(in box, third);

        long remFirst = GetDim(in space, first) - boxFirst;
        long remSecond = GetDim(in space, second) - boxSecond;
        long remThird = GetDim(in space, third) - boxThird;

        if (remFirst > 0)
        {
            long nx = space.X, ny = space.Y, nz = space.Z;
            AddToCoord(ref nx, ref ny, ref nz, first, boxFirst);
            long nw = space.W, nh = space.H, nd = space.D;
            SetDim(ref nw, ref nh, ref nd, first, remFirst);
            if (nw > 0 && nh > 0 && nd > 0) free.Add(new Space { X = nx, Y = ny, Z = nz, W = nw, H = nh, D = nd });
        }

        if (remSecond > 0)
        {
            long nx = space.X, ny = space.Y, nz = space.Z;
            AddToCoord(ref nx, ref ny, ref nz, second, boxSecond);
            long nw = space.W, nh = space.H, nd = space.D;
            SetDim(ref nw, ref nh, ref nd, first, boxFirst);
            SetDim(ref nw, ref nh, ref nd, second, remSecond);
            if (nw > 0 && nh > 0 && nd > 0) free.Add(new Space { X = nx, Y = ny, Z = nz, W = nw, H = nh, D = nd });
        }

        if (remThird > 0)
        {
            long nx = space.X, ny = space.Y, nz = space.Z;
            AddToCoord(ref nx, ref ny, ref nz, third, boxThird);
            long nw = space.W, nh = space.H, nd = space.D;
            SetDim(ref nw, ref nh, ref nd, first, boxFirst);
            SetDim(ref nw, ref nh, ref nd, second, boxSecond);
            SetDim(ref nw, ref nh, ref nd, third, remThird);
            if (nw > 0 && nh > 0 && nd > 0) free.Add(new Space { X = nx, Y = ny, Z = nz, W = nw, H = nh, D = nd });
        }
    }

    private static long GetDim(in Box b, Axis axis)
    {
        return axis == Axis.X ? b.W : (axis == Axis.Y ? b.H : b.D);
    }

    private static long GetDim(in Space s, Axis axis)
    {
        return axis == Axis.X ? s.W : (axis == Axis.Y ? s.H : s.D);
    }

    private static void SetDim(ref long w, ref long h, ref long d, Axis axis, long value)
    {
        if (axis == Axis.X) w = value;
        else if (axis == Axis.Y) h = value;
        else d = value;
    }

    private static void AddToCoord(ref long x, ref long y, ref long z, Axis axis, long delta)
    {
        if (axis == Axis.X) x += delta;
        else if (axis == Axis.Y) y += delta;
        else z += delta;
    }

    private static long Max3(long a, long b, long c)
    {
        long m = a > b ? a : b;
        return m > c ? m : c;
    }

    private static long MulSaturated(long a, long b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    private static long Mul3Saturated(long a, long b, long c)
    {
        return MulSaturated(MulSaturated(a, b), c);
    }

    private static readonly IComparer<Box>[] SmallComparers = new IComparer<Box>[]
    {
        new CompareVolumeDesc(),
        new CompareMaxDimDescVolumeDesc(),
        new CompareVolumeAsc(),
        new CompareMinDimAscVolumeAsc()
    };

    private static readonly IComparer<Box>[] LargeComparers = new IComparer<Box>[]
    {
        new CompareVolumeAsc(),
        new CompareDepthAscThenVolumeAsc(),
        new CompareVolumeDesc(),
        new CompareHeightDescThenVolumeDesc()
    };

    private static readonly Axis[][] AllSplitOrders = new Axis[][]
    {
        new[] { Axis.X, Axis.Y, Axis.Z },
        new[] { Axis.X, Axis.Z, Axis.Y },
        new[] { Axis.Y, Axis.X, Axis.Z },
        new[] { Axis.Y, Axis.Z, Axis.X },
        new[] { Axis.Z, Axis.X, Axis.Y },
        new[] { Axis.Z, Axis.Y, Axis.X }
    };

    private static readonly Axis[][] TwoSplitOrders = new Axis[][]
    {
        new[] { Axis.Y, Axis.X, Axis.Z },
        new[] { Axis.X, Axis.Y, Axis.Z }
    };

    private sealed class CompareVolumeDesc : IComparer<Box>
    {
        public int Compare(Box a, Box b)
        {
            int c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            c = b.MaxDim.CompareTo(a.MaxDim);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    private sealed class CompareVolumeAsc : IComparer<Box>
    {
        public int Compare(Box a, Box b)
        {
            int c = a.Volume.CompareTo(b.Volume);
            if (c != 0) return c;
            c = a.MinDim.CompareTo(b.MinDim);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    private sealed class CompareMaxDimDescVolumeDesc : IComparer<Box>
    {
        public int Compare(Box a, Box b)
        {
            int c = b.MaxDim.CompareTo(a.MaxDim);
            if (c != 0) return c;
            c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    private sealed class CompareMinDimAscVolumeAsc : IComparer<Box>
    {
        public int Compare(Box a, Box b)
        {
            int c = a.MinDim.CompareTo(b.MinDim);
            if (c != 0) return c;
            c = a.Volume.CompareTo(b.Volume);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    private sealed class CompareDepthAscThenVolumeAsc : IComparer<Box>
    {
        public int Compare(Box a, Box b)
        {
            int c = a.D.CompareTo(b.D);
            if (c != 0) return c;
            c = a.H.CompareTo(b.H);
            if (c != 0) return c;
            c = a.W.CompareTo(b.W);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    private sealed class CompareHeightDescThenVolumeDesc : IComparer<Box>
    {
        public int Compare(Box a, Box b)
        {
            int c = b.H.CompareTo(a.H);
            if (c != 0) return c;
            c = b.D.CompareTo(a.D);
            if (c != 0) return c;
            c = b.W.CompareTo(a.W);
            if (c != 0) return c;
            return a.Id.CompareTo(b.Id);
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buf = new byte[bufferSize];
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = Read();
                if (c == -1) throw new EndOfStreamException();
            } while (c <= 32);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long val = 0;
            while (c > 32 && c != -1)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }
    }
}
