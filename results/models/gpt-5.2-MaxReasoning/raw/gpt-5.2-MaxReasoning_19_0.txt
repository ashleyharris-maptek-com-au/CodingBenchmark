using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    private readonly struct PlanOp
    {
        public readonly int Steps;
        public readonly int DigX;
        public readonly int DigY;
        public readonly int DigZ;
        public readonly int DestX;
        public readonly int DestY;

        public PlanOp(int steps, int digX, int digY, int digZ, int destX, int destY)
        {
            Steps = steps;
            DigX = digX;
            DigY = digY;
            DigZ = digZ;
            DestX = destX;
            DestY = destY;
        }
    }

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());

        if (!input.TryReadToken(out string? t0))
            return;

        int xCount = ParseInt(t0);
        int yCount = input.ReadIntToken();
        int zCount = input.ReadIntToken();
        _ = input.ReadDoubleToken(); // XBlockSize (not required for current heuristics)
        _ = input.ReadDoubleToken(); // YBlockSize
        _ = input.ReadDoubleToken(); // ZBlockSize

        // Adaptive strategy based on problem scale:
        // - Huge volumes: output no-ops immediately (cannot reasonably scan the full world).
        // - Otherwise: scan only the plant column (x=0,y=0) and move profitable exposed ore straight
        //   to the plant with a zero-length path (N=0), which avoids slope/path ambiguity.
        ulong volume = SafeMul(SafeMul((ulong)Math.Max(0, xCount), (ulong)Math.Max(0, yCount)), (ulong)Math.Max(0, zCount));

        List<PlanOp> ops;
        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            ops = new List<PlanOp>(capacity: 0);
        }
        else if (volume > 2_000_000_000UL || zCount > 2_000_000)
        {
            // Too large to process meaningfully within typical time limits.
            ops = new List<PlanOp>(capacity: 0);
        }
        else
        {
            // Ensure we are positioned at the first voxel byte (handle \r\n after header).
            input.SkipWhitespace();
            ops = PlanPlantColumnOnly(input, zCount);
        }

        using var writer = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), bufferSize: 1 << 16)
        {
            NewLine = "\n"
        };

        writer.WriteLine(ops.Count);
        for (int i = 0; i < ops.Count; i++)
        {
            // Format:
            // N digX digY digZ (N stepX stepY pairs) destX destY
            // Here we always use N=0 and dest at the processing plant (0,0).
            var op = ops[i];
            writer.Write(op.Steps);
            writer.Write(' ');
            writer.Write(op.DigX);
            writer.Write(' ');
            writer.Write(op.DigY);
            writer.Write(' ');
            writer.Write(op.DigZ);
            writer.Write(' ');
            writer.Write(op.DestX);
            writer.Write(' ');
            writer.Write(op.DestY);
            writer.Write('\n');
        }
    }

    private static List<PlanOp> PlanPlantColumnOnly(FastInput input, int zCount)
    {
        // Read the (x=0,y=0) column: z varies fastest in input format.
        // We only plan moves that require no horizontal transport (N=0), so we can safely
        // deliver ore already at the plant column.
        int effectiveZ = zCount;
        byte[] column = zCount <= 1_000_000 ? new byte[zCount] : Array.Empty<byte>();

        int topSolidZ = -1;
        for (int z = 0; z < zCount; z++)
        {
            int b = input.ReadByte();
            if (b < 0)
            {
                effectiveZ = z;
                break;
            }
            byte d = (byte)b;
            if (column.Length != 0) column[z] = d;
            if ((d & 0x0F) != 0) topSolidZ = z; // massInTonnes = d % 16
        }

        var ops = new List<PlanOp>(capacity: 32);
        if (topSolidZ < 0) return ops;

        // Mine valuable voxels from the top down, stopping at the first non-empty waste voxel
        // (valuePerTonne == 0) since dumping waste at the plant is penalized.
        for (int z = topSolidZ; z >= 0; z--)
        {
            byte d = column.Length != 0 ? column[z] : (byte)0;
            int mass = d & 0x0F;
            if (mass == 0) continue;

            int valuePerTonne = d >> 4;
            if (valuePerTonne == 0) break;

            // N=0 steps, dig at (0,0,z), dump at plant (0,0)
            ops.Add(new PlanOp(steps: 0, digX: 0, digY: 0, digZ: z, destX: 0, destY: 0));
        }

        return ops;
    }

    private static int ParseInt(string s)
        => int.Parse(s, NumberStyles.Integer, CultureInfo.InvariantCulture);

    private static ulong SafeMul(ulong a, ulong b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > ulong.MaxValue / b) return ulong.MaxValue;
        return a * b;
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;
        private char[] _tokenBuf;

        public FastInput(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
            _len = 0;
            _pos = 0;
            _tokenBuf = new char[64];
        }

        public int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        private int PeekByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos];
        }

        public void SkipWhitespace()
        {
            int b;
            while ((b = PeekByte()) >= 0 && b <= 32)
                _pos++;
        }

        public bool TryReadToken(out string? token)
        {
            SkipWhitespace();
            int b = ReadByte();
            if (b < 0)
            {
                token = null;
                return false;
            }

            int n = 0;
            while (b > 32)
            {
                if (n == _tokenBuf.Length)
                    Array.Resize(ref _tokenBuf, _tokenBuf.Length * 2);
                _tokenBuf[n++] = (char)b;

                b = ReadByte();
                if (b < 0) break;
            }

            token = new string(_tokenBuf, 0, n);
            return true;
        }

        public int ReadIntToken()
        {
            if (!TryReadToken(out string? token) || token is null)
                return 0;
            return int.Parse(token, NumberStyles.Integer, CultureInfo.InvariantCulture);
        }

        public double ReadDoubleToken()
        {
            if (!TryReadToken(out string? token) || token is null)
                return 0.0;
            return double.Parse(token, NumberStyles.Float, CultureInfo.InvariantCulture);
        }
    }
}
