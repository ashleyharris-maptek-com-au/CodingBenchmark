import math
from typing import Dict, Optional, Tuple

# Module-level persistence (per scenario).
_MEM: Dict[str, object] = {}


def _clamp(x: float, lo: float, hi: float) -> float:
    if x < lo:
        return lo
    if x > hi:
        return hi
    return x


def _sign(x: float) -> float:
    if x > 0.0:
        return 1.0
    if x < 0.0:
        return -1.0
    return 0.0


def _norm(v) -> float:
    return math.sqrt(v[0] * v[0] + v[1] * v[1] + v[2] * v[2])


def _sub(a, b):
    return [a[0] - b[0], a[1] - b[1], a[2] - b[2]]


def _add(a, b):
    return [a[0] + b[0], a[1] + b[1], a[2] + b[2]]


def _mul(s: float, v):
    return [s * v[0], s * v[1], s * v[2]]


def _unit(v):
    n = _norm(v)
    if n <= 1e-9:
        return [0.0, 0.0, 0.0]
    return [v[0] / n, v[1] / n, v[2] / n]


def _get_float(state: dict, key: str) -> Optional[float]:
    try:
        val = state.get(key, None)
        if val is None:
            return None
        return float(val)
    except Exception:
        return None


def _get_bool(state: dict, key: str) -> bool:
    try:
        return bool(state.get(key, False))
    except Exception:
        return False


def _median(vals):
    vals = [v for v in vals if v is not None]
    if not vals:
        return None
    vals = sorted(vals)
    m = len(vals) // 2
    return vals[m] if (len(vals) % 2 == 1) else 0.5 * (vals[m - 1] + vals[m])


def _fuse_relative_state(state: dict, dt: float) -> Tuple[Optional[list], Optional[list], Optional[float], float]:
    """
    Returns (pos_meas, vel_meas, range_scalar, quality in [0,1]).
    pos_meas/vel_meas are LVLH relative vectors.
    """
    # Scalar range (median of valid finders; fall back to camera range if active).
    ranges = []
    for i in (1, 2, 3):
        if _get_bool(state, f"range_{i}_valid"):
            ranges.append(_get_float(state, f"range_{i}_m"))
    range_scalar = _median(ranges)

    # Candidates for position.
    candidates = []

    def add_candidate(kind: str, base_w: float, pos: Optional[list]):
        if pos is None:
            return
        candidates.append((kind, base_w, pos))

    # Radar pos.
    if _get_bool(state, "radar_rel_x_valid") and _get_bool(state, "radar_rel_y_valid") and _get_bool(state, "radar_rel_z_valid"):
        add_candidate(
            "radar",
            0.85,
            [
                _get_float(state, "radar_rel_pos_x_m") or 0.0,
                _get_float(state, "radar_rel_pos_y_m") or 0.0,
                _get_float(state, "radar_rel_pos_z_m") or 0.0,
            ],
        )

    # Lidar pos.
    if _get_bool(state, "lidar_rel_x_valid") and _get_bool(state, "lidar_rel_y_valid") and _get_bool(state, "lidar_rel_z_valid"):
        add_candidate(
            "lidar",
            1.0,
            [
                _get_float(state, "lidar_rel_pos_x_m") or 0.0,
                _get_float(state, "lidar_rel_pos_y_m") or 0.0,
                _get_float(state, "lidar_rel_pos_z_m") or 0.0,
            ],
        )

    # Docking camera bearing + range.
    if _get_bool(state, "dock_camera_active"):
        if _get_bool(state, "dock_camera_bearing_x_valid") and _get_bool(state, "dock_camera_bearing_y_valid") and _get_bool(state, "dock_camera_bearing_z_valid"):
            br = [
                _get_float(state, "dock_camera_bearing_x") or 0.0,
                _get_float(state, "dock_camera_bearing_y") or 0.0,
                _get_float(state, "dock_camera_bearing_z") or 0.0,
            ]
            cam_r = _get_float(state, "dock_camera_range_m")
            if cam_r is not None and cam_r > 0.0:
                add_candidate("camera", 0.7, _mul(cam_r, _unit(br)))
                if range_scalar is None:
                    range_scalar = cam_r

    # Pick / blend position measurement.
    pos_meas = None
    pos_q = 0.0
    if candidates:
        # Score candidates by consistency with scalar range and with last estimate.
        pos_est = _MEM.get("pos_est", [0.0, 0.0, 0.0])
        scored = []
        for kind, base_w, pos in candidates:
            r = _norm(pos)
            score = base_w
            if range_scalar is not None and range_scalar > 1.0:
                mismatch = abs(r - range_scalar) / max(1.0, range_scalar)
                score -= 2.0 * mismatch
            jump = _norm(_sub(pos, pos_est))
            if range_scalar is not None and range_scalar > 5.0:
                score -= 0.15 * (jump / max(1.0, range_scalar))
            scored.append((score, kind, base_w, pos))
        scored.sort(reverse=True, key=lambda t: t[0])

        # If top 2 are both good and close, blend them.
        best = scored[0]
        if best[0] > 0.2:
            pos_meas = best[3]
            pos_q = _clamp((best[0] + 0.1) / 1.1, 0.0, 1.0)
            if len(scored) >= 2 and scored[1][0] > 0.15:
                p1 = scored[0][3]
                p2 = scored[1][3]
                if _norm(_sub(p1, p2)) < max(10.0, 0.1 * (_norm(p1) + _norm(p2))):
                    w1 = max(0.01, scored[0][2])
                    w2 = max(0.01, scored[1][2])
                    pos_meas = [
                        (w1 * p1[0] + w2 * p2[0]) / (w1 + w2),
                        (w1 * p1[1] + w2 * p2[1]) / (w1 + w2),
                        (w1 * p1[2] + w2 * p2[2]) / (w1 + w2),
                    ]
                    pos_q = _clamp(pos_q + 0.1, 0.0, 1.0)

    # Velocity measurement: prefer lidar, then radar.
    vel_meas = None
    vel_q = 0.0
    if _get_bool(state, "lidar_relv_x_valid") and _get_bool(state, "lidar_relv_y_valid") and _get_bool(state, "lidar_relv_z_valid"):
        vel_meas = [
            _get_float(state, "lidar_rel_vel_x_ms") or 0.0,
            _get_float(state, "lidar_rel_vel_y_ms") or 0.0,
            _get_float(state, "lidar_rel_vel_z_ms") or 0.0,
        ]
        vel_q = 1.0
    elif _get_bool(state, "radar_relv_x_valid") and _get_bool(state, "radar_relv_y_valid") and _get_bool(state, "radar_relv_z_valid"):
        vel_meas = [
            _get_float(state, "radar_rel_vel_x_ms") or 0.0,
            _get_float(state, "radar_rel_vel_y_ms") or 0.0,
            _get_float(state, "radar_rel_vel_z_ms") or 0.0,
        ]
        vel_q = 0.8

    # If no direct velocity measurement, finite-difference position measurements (noisy but helps).
    if vel_meas is None and pos_meas is not None:
        prev_pos_meas = _MEM.get("prev_pos_meas", None)
        if isinstance(prev_pos_meas, list) and dt > 1e-6:
            vel_meas = _mul(1.0 / dt, _sub(pos_meas, prev_pos_meas))
            vel_q = 0.35

    # Overall quality.
    quality = _clamp(0.55 * pos_q + 0.45 * vel_q, 0.0, 1.0)
    return pos_meas, vel_meas, range_scalar, quality


def _estimate_orbit_rate(state: dict) -> float:
    # Prefer ground semi-major axis + speed for n ~= v/a (circular).
    if _get_bool(state, "ground_speed_ms_valid") and _get_bool(state, "ground_semi_major_axis_m_valid"):
        v = _get_float(state, "ground_speed_ms")
        a = _get_float(state, "ground_semi_major_axis_m")
        if v is not None and a is not None and a > 1e3 and v > 10.0:
            n = abs(v) / a
            return _clamp(n, 5e-4, 2.2e-3)
    # Fallback: LEO ~ 93 minutes => n ~ 0.00113 rad/s.
    return 0.00113


def _fuse_attitude(state: dict, dt: float) -> Tuple[list, list]:
    """
    Returns (att_angles_rad [roll,pitch,yaw], gyro_rates_rad_s [roll,pitch,yaw]).
    Uses star trackers when available, otherwise integrates gyros.
    """
    if "att_est" not in _MEM:
        _MEM["att_est"] = [0.0, 0.0, 0.0]

    # Gyro rate fusion (deg/s).
    rates = {"roll": [], "pitch": [], "yaw": []}
    for i in (1, 2, 3):
        if _get_bool(state, f"gyro_{i}_roll_valid"):
            rates["roll"].append(_get_float(state, f"gyro_{i}_roll_dps"))
        if _get_bool(state, f"gyro_{i}_pitch_valid"):
            rates["pitch"].append(_get_float(state, f"gyro_{i}_pitch_dps"))
        if _get_bool(state, f"gyro_{i}_yaw_valid"):
            rates["yaw"].append(_get_float(state, f"gyro_{i}_yaw_dps"))
    gyro_dps = [
        _median(rates["roll"]) or 0.0,
        _median(rates["pitch"]) or 0.0,
        _median(rates["yaw"]) or 0.0,
    ]
    gyro = [math.radians(gyro_dps[0]), math.radians(gyro_dps[1]), math.radians(gyro_dps[2])]

    # Integrate.
    att_est = _MEM["att_est"]
    att_est = [att_est[0] + gyro[0] * dt, att_est[1] + gyro[1] * dt, att_est[2] + gyro[2] * dt]

    # Star tracker fusion (absolute angles deg).
    st_angles = []
    for i in (1, 2):
        if _get_bool(state, f"star_tracker_{i}_roll_valid") and _get_bool(state, f"star_tracker_{i}_pitch_valid") and _get_bool(state, f"star_tracker_{i}_yaw_valid"):
            st_angles.append(
                [
                    math.radians(_get_float(state, f"star_tracker_{i}_roll_deg") or 0.0),
                    math.radians(_get_float(state, f"star_tracker_{i}_pitch_deg") or 0.0),
                    math.radians(_get_float(state, f"star_tracker_{i}_yaw_deg") or 0.0),
                ]
            )
    if st_angles:
        # Robust average: if two trackers disagree a lot, trust the one closer to current estimate.
        if len(st_angles) == 1:
            st = st_angles[0]
        else:
            a0, a1 = st_angles[0], st_angles[1]
            d01 = _norm(_sub(a0, a1))
            if d01 < math.radians(5.0):
                st = [(a0[0] + a1[0]) * 0.5, (a0[1] + a1[1]) * 0.5, (a0[2] + a1[2]) * 0.5]
            else:
                d0 = _norm(_sub(a0, att_est))
                d1 = _norm(_sub(a1, att_est))
                st = a0 if d0 <= d1 else a1
        # Blend to star tracker to prevent gyro drift.
        alpha = 0.35
        att_est = [(1 - alpha) * att_est[0] + alpha * st[0], (1 - alpha) * att_est[1] + alpha * st[1], (1 - alpha) * att_est[2] + alpha * st[2]]

    _MEM["att_est"] = att_est
    return att_est, gyro


def autopilot_step(state: dict, dt: float) -> dict:
    dt = float(dt) if dt is not None else 0.1
    dt = _clamp(dt, 1e-3, 5.0)

    # Initialize memory.
    if "t" not in _MEM:
        _MEM["t"] = 0.0
        _MEM["pos_est"] = [0.0, 0.0, 0.0]
        _MEM["vel_est"] = [0.0, 0.0, 0.0]
        _MEM["vel_int"] = [0.0, 0.0, 0.0]
        _MEM["prev_pos_meas"] = None

    _MEM["t"] = float(_MEM["t"]) + dt

    # Orbit mean motion estimate (rad/s).
    n = _estimate_orbit_rate(state)

    # Sensor fusion -> measurement.
    pos_meas, vel_meas, range_scalar, meas_q = _fuse_relative_state(state, dt)
    if pos_meas is not None:
        _MEM["prev_pos_meas"] = pos_meas

    # State estimate propagate + update (simple alpha-beta).
    pos_est = _MEM["pos_est"]
    vel_est = _MEM["vel_est"]
    pos_pred = _add(pos_est, _mul(dt, vel_est))
    vel_pred = vel_est

    # If measurement jumps wildly, treat as reset.
    if pos_meas is not None and _norm(_sub(pos_meas, pos_pred)) > 5000.0:
        pos_pred = pos_meas
        vel_pred = vel_meas if vel_meas is not None else vel_pred

    # Update with measurement.
    if pos_meas is not None:
        k_pos = 0.15 + 0.65 * meas_q
        res = _sub(pos_meas, pos_pred)
        pos_est = _add(pos_pred, _mul(k_pos, res))
    else:
        pos_est = pos_pred

    if vel_meas is not None:
        k_vel = 0.1 + 0.7 * meas_q
        resv = _sub(vel_meas, vel_pred)
        vel_est = _add(vel_pred, _mul(k_vel, resv))
    else:
        vel_est = vel_pred

    _MEM["pos_est"] = pos_est
    _MEM["vel_est"] = vel_est

    # Range for mode selection.
    r_vec = pos_est
    r = _norm(r_vec)
    if range_scalar is not None:
        # Use scalar range to smooth gross noise, but don't break direction.
        r = 0.75 * r + 0.25 * float(range_scalar)

    # Targets.
    target_dock_speed = _get_float(state, "target_dock_speed_ms")
    if target_dock_speed is None or target_dock_speed <= 0.0:
        target_dock_speed = 0.3

    # Decide if we're in "phasing" (orbital mechanics-dominated) mode: large along-track separation.
    x, y, z = pos_est[0], pos_est[1], pos_est[2]
    vx, vy, vz = vel_est[0], vel_est[1], vel_est[2]
    along_track_dominant = abs(y) > (abs(x) + abs(z) + 20.0)
    phasing_mode = along_track_dominant and (abs(y) >= 180.0)

    # Speed caps as a function of range (m/s).
    def speed_limit(dist_m: float) -> float:
        if dist_m > 800.0:
            return 2.0
        if dist_m > 300.0:
            return 1.2
        if dist_m > 120.0:
            return 0.7
        if dist_m > 50.0:
            return 0.45
        if dist_m > 20.0:
            return 0.25
        if dist_m > 10.0:
            return 0.18
        if dist_m > 2.5:
            return min(0.12, 0.9 * target_dock_speed)
        return min(0.06, 0.7 * target_dock_speed)

    v_lim = speed_limit(r)

    # Thruster acceleration per full-scale command (unknown mass): assume ~0.25 m/s^2.
    # The controller is feedback-based and robust to moderate mismatch.
    a_max = 0.25

    # Fuel-based throttle limiting (avoid wasting propellant when low).
    fuel_kg = _get_float(state, "fuel_remaining_kg")
    throttle_limit = 1.0
    if fuel_kg is not None:
        if fuel_kg < 80.0:
            throttle_limit = 0.75
        if fuel_kg < 35.0:
            throttle_limit = 0.55

    # Measurement confidence limiting (if nav is blind, be gentle).
    throttle_limit *= (0.35 + 0.65 * meas_q)
    throttle_limit = _clamp(throttle_limit, 0.2, 1.0)

    # Translational guidance.
    u = [0.0, 0.0, 0.0]  # desired thrust-induced acceleration in LVLH (m/s^2)

    if phasing_mode:
        # FAR ALONG-TRACK PHASING:
        # If behind (y<0) we generally want retrograde thrust (u_y<0) to drop to a lower, faster orbit.
        # If ahead (y>0) we generally want prograde thrust (u_y>0) to rise to a higher, slower orbit.
        y_s = _sign(y)
        desired_closing = _clamp(abs(y) / 650.0, 0.12, 1.4)  # m/s, gentle closure
        closing = (-y_s) * vy  # positive when closing towards y=0

        # Phasing acceleration magnitudes.
        a_phase = 0.05 * throttle_limit
        a_brake = 0.05 * throttle_limit

        if y_s == 0.0:
            u[1] = 0.0
        elif closing < 0.85 * desired_closing:
            u[1] = y_s * a_phase
        elif closing > 1.25 * desired_closing:
            u[1] = -y_s * a_brake
        else:
            u[1] = 0.0

        # Keep cross-track under control even during phasing (plane mismatch is expensive later).
        # Use mild PD for z.
        u[2] = _clamp((-0.004 * z - 0.04 * vz), -0.06 * throttle_limit, 0.06 * throttle_limit)

        # Let x float unless it's getting large, to allow the orbit change to do its job.
        x_soft_limit = 180.0
        if abs(x) > x_soft_limit:
            u[0] = _clamp((-0.003 * x - 0.03 * vx), -0.05 * throttle_limit, 0.05 * throttle_limit)
        else:
            u[0] = 0.0

    else:
        # MID/NEAR/FINAL GLIDESLOPE (linear regime): track a velocity that shrinks with range.
        # Range-rate proportional guidance: v_des = -k * r, capped by v_lim.
        k_glide = 0.004  # 1/s
        v_des = _mul(-k_glide, r_vec)
        v_des_mag = _norm(v_des)
        if v_des_mag > v_lim and v_des_mag > 1e-9:
            v_des = _mul(v_lim / v_des_mag, v_des)

        # Acceleration to track v_des with a range-dependent time constant.
        tau_v = _clamp(r / 12.0, 1.8, 18.0)
        vel_err = _sub(v_des, vel_est)

        # Integral disturbance rejection (small, with anti-windup).
        vel_int = _MEM.get("vel_int", [0.0, 0.0, 0.0])
        if isinstance(vel_int, list):
            # Freeze integral if likely saturated (rough heuristic).
            sat_risk = v_des_mag > 0.9 * v_lim + 1e-6
            if not sat_risk:
                vel_int = _add(vel_int, _mul(dt, vel_err))
                # Clamp integral to prevent windup.
                vel_int = [
                    _clamp(vel_int[0], -2.0, 2.0),
                    _clamp(vel_int[1], -2.0, 2.0),
                    _clamp(vel_int[2], -2.0, 2.0),
                ]
        else:
            vel_int = [0.0, 0.0, 0.0]
        _MEM["vel_int"] = vel_int

        ki = 0.02
        a_des = _add(_mul(1.0 / tau_v, vel_err), _mul(ki, vel_int))

        # If we're going too fast overall, bias acceleration to brake relative velocity.
        speed = _norm(vel_est)
        if speed > 1.25 * v_lim and speed > 1e-6:
            brake = _mul(-1.0 / 4.0, vel_est)  # ~4 s to bleed speed
            a_des = _add(a_des, brake)

        # Convert desired linear accelerations into LVLH thrust accelerations using HCW dynamics feedforward.
        # HCW with control: x¨ = 3 n^2 x + 2 n y˙ + u_x; y¨ = -2 n x˙ + u_y; z¨ = -n^2 z + u_z
        u[0] = a_des[0] - 3.0 * (n * n) * x - 2.0 * n * vy
        u[1] = a_des[1] + 2.0 * n * vx
        u[2] = a_des[2] + (n * n) * z

        # Limit commanded acceleration (m/s^2).
        u_lim = a_max * throttle_limit
        u = [_clamp(u[0], -u_lim, u_lim), _clamp(u[1], -u_lim, u_lim), _clamp(u[2], -u_lim, u_lim)]

    # Map accelerations to normalized thruster commands.
    thrust_x = _clamp(u[0] / a_max, -throttle_limit, throttle_limit)
    thrust_y = _clamp(u[1] / a_max, -throttle_limit, throttle_limit)
    thrust_z = _clamp(u[2] / a_max, -throttle_limit, throttle_limit)

    # Attitude hold (optional; keep near LVLH = 0 angles).
    att_angles, gyro_rates = _fuse_attitude(state, dt)
    # Rate command (rad/s) to drive angles to 0 with damping.
    k_p_att = 0.8
    k_d_att = 0.25
    rate_roll = -k_p_att * att_angles[0] - k_d_att * gyro_rates[0]
    rate_pitch = -k_p_att * att_angles[1] - k_d_att * gyro_rates[1]
    rate_yaw = -k_p_att * att_angles[2] - k_d_att * gyro_rates[2]
    rate_roll = _clamp(rate_roll, -0.035, 0.035)
    rate_pitch = _clamp(rate_pitch, -0.035, 0.035)
    rate_yaw = _clamp(rate_yaw, -0.035, 0.035)

    return {
        "thrust_x": float(thrust_x),
        "thrust_y": float(thrust_y),
        "thrust_z": float(thrust_z),
        "att_rate_roll": float(rate_roll),
        "att_rate_pitch": float(rate_pitch),
        "att_rate_yaw": float(rate_yaw),
    }
