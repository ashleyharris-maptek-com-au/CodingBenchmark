; SPIR-V
; Version: 1.0
; Geometry shader: thick wireframe + dark fill (triangles adjacency)

OpCapability Shader
OpCapability Geometry

%ext = OpExtInstImport "GLSL.std.450"

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor

OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 15
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %inColor "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %arr_v3float_6 ArrayStride 16

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2

OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%bool = OpTypeBool

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3float_6 = OpTypeArray %v3float %uint_6

%ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_Input_arr_v3float_6 = OpTypePointer Input %arr_v3float_6
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv_in Input
%worldPos = OpVariable %ptr_Input_arr_v3float_6 Input
%normal = OpVariable %ptr_Input_arr_v3float_6 Input
%inColor = OpVariable %ptr_Input_arr_v3float_6 Input

%gl_out = OpVariable %ptr_Output_pv_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0_015 = OpConstant %float 0.015
%float_0_3 = OpConstant %float 0.3
%float_1 = OpConstant %float 1.0
%float_0 = OpConstant %float 0.0
%float_n1 = OpConstant %float -1.0
%float_0_0001 = OpConstant %float 0.0001

%v3_white = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %fn_void
%entry = OpLabel

%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Load main triangle vertices (0,2,4) from gl_in[].gl_Position
%pos0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos4_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %pos4_ptr

; Determine triangle winding in NDC so edge quads follow the same facing
%pos0_xy_wind = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos2_xy_wind = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos4_xy_wind = OpVectorShuffle %v2float %pos4 %pos4 0 1
%w0_wind = OpCompositeExtract %float %pos0 3
%w2_wind = OpCompositeExtract %float %pos2 3
%w4_wind = OpCompositeExtract %float %pos4 3
%invw0_wind = OpFDiv %float %float_1 %w0_wind
%invw2_wind = OpFDiv %float %float_1 %w2_wind
%invw4_wind = OpFDiv %float %float_1 %w4_wind
%ndc0_wind = OpVectorTimesScalar %v2float %pos0_xy_wind %invw0_wind
%ndc2_wind = OpVectorTimesScalar %v2float %pos2_xy_wind %invw2_wind
%ndc4_wind = OpVectorTimesScalar %v2float %pos4_xy_wind %invw4_wind
%e02_wind = OpFSub %v2float %ndc2_wind %ndc0_wind
%e04_wind = OpFSub %v2float %ndc4_wind %ndc0_wind
%e02_x = OpCompositeExtract %float %e02_wind 0
%e02_y = OpCompositeExtract %float %e02_wind 1
%e04_x = OpCompositeExtract %float %e04_wind 0
%e04_y = OpCompositeExtract %float %e04_wind 1
%cross_a = OpFMul %float %e02_x %e04_y
%cross_b = OpFMul %float %e02_y %e04_x
%tri_cross = OpFSub %float %cross_a %cross_b
%tri_ccw = OpFOrdGreaterThanEqual %bool %tri_cross %float_0
%perp_sign = OpSelect %float %tri_ccw %float_1 %float_n1

; ---- Filled triangle (dimmed vertex colors) ----
%col0_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_0
%col0 = OpLoad %v3float %col0_ptr
%col2_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_2
%col2 = OpLoad %v3float %col2_ptr
%col4_ptr = OpAccessChain %ptr_Input_v3float %inColor %int_4
%col4 = OpLoad %v3float %col4_ptr

%col0_dim = OpVectorTimesScalar %v3float %col0 %float_0_3
%col2_dim = OpVectorTimesScalar %v3float %col2 %float_0_3
%col4_dim = OpVectorTimesScalar %v3float %col4 %float_0_3

OpStore %gl_out_pos_ptr %pos0
OpStore %outColor %col0_dim
OpEmitVertex

OpStore %gl_out_pos_ptr %pos2
OpStore %outColor %col2_dim
OpEmitVertex

OpStore %gl_out_pos_ptr %pos4
OpStore %outColor %col4_dim
OpEmitVertex

OpEndPrimitive

; ---- Edge 0 -> 2 (emit quad as triangle strip) ----
%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%w0 = OpCompositeExtract %float %pos0 3
%w2 = OpCompositeExtract %float %pos2 3
%invw0 = OpFDiv %float %float_1 %w0
%invw2 = OpFDiv %float %float_1 %w2
%ndc0 = OpVectorTimesScalar %v2float %pos0_xy %invw0
%ndc2 = OpVectorTimesScalar %v2float %pos2_xy %invw2
%delta02 = OpFSub %v2float %ndc2 %ndc0
%dir02 = OpExtInst %v2float %ext Normalize %delta02
%dir02_x = OpCompositeExtract %float %dir02 0
%dir02_y = OpCompositeExtract %float %dir02 1
%dir02_y_neg = OpFNegate %float %dir02_y
%perp02 = OpCompositeConstruct %v2float %dir02_y_neg %dir02_x
%perp02_oriented = OpVectorTimesScalar %v2float %perp02 %perp_sign
%perp02_thick = OpVectorTimesScalar %v2float %perp02_oriented %float_0_015
%off0_02 = OpVectorTimesScalar %v2float %perp02_thick %w0
%off2_02 = OpVectorTimesScalar %v2float %perp02_thick %w2

%pos0_plus_xy_02 = OpFAdd %v2float %pos0_xy %off0_02
%pos0_minus_xy_02 = OpFSub %v2float %pos0_xy %off0_02
%pos2_plus_xy_02 = OpFAdd %v2float %pos2_xy %off2_02
%pos2_minus_xy_02 = OpFSub %v2float %pos2_xy %off2_02

%z0 = OpCompositeExtract %float %pos0 2
%z2 = OpCompositeExtract %float %pos2 2
%zbias0 = OpFMul %float %float_0_0001 %w0
%zbias2 = OpFMul %float %float_0_0001 %w2
%z0_edge = OpFSub %float %z0 %zbias0
%z2_edge = OpFSub %float %z2 %zbias2

%p0p_x_02 = OpCompositeExtract %float %pos0_plus_xy_02 0
%p0p_y_02 = OpCompositeExtract %float %pos0_plus_xy_02 1
%p0m_x_02 = OpCompositeExtract %float %pos0_minus_xy_02 0
%p0m_y_02 = OpCompositeExtract %float %pos0_minus_xy_02 1
%p2p_x_02 = OpCompositeExtract %float %pos2_plus_xy_02 0
%p2p_y_02 = OpCompositeExtract %float %pos2_plus_xy_02 1
%p2m_x_02 = OpCompositeExtract %float %pos2_minus_xy_02 0
%p2m_y_02 = OpCompositeExtract %float %pos2_minus_xy_02 1

%pos0_plus_02 = OpCompositeConstruct %v4float %p0p_x_02 %p0p_y_02 %z0_edge %w0
%pos0_minus_02 = OpCompositeConstruct %v4float %p0m_x_02 %p0m_y_02 %z0_edge %w0
%pos2_plus_02 = OpCompositeConstruct %v4float %p2p_x_02 %p2p_y_02 %z2_edge %w2
%pos2_minus_02 = OpCompositeConstruct %v4float %p2m_x_02 %p2m_y_02 %z2_edge %w2

OpStore %gl_out_pos_ptr %pos0_plus_02
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos0_minus_02
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos2_plus_02
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos2_minus_02
OpStore %outColor %v3_white
OpEmitVertex

OpEndPrimitive

; ---- Edge 2 -> 4 ----
%pos4_xy = OpVectorShuffle %v2float %pos4 %pos4 0 1
%w4 = OpCompositeExtract %float %pos4 3
%invw4 = OpFDiv %float %float_1 %w4
%ndc4 = OpVectorTimesScalar %v2float %pos4_xy %invw4
%delta24 = OpFSub %v2float %ndc4 %ndc2
%dir24 = OpExtInst %v2float %ext Normalize %delta24
%dir24_x = OpCompositeExtract %float %dir24 0
%dir24_y = OpCompositeExtract %float %dir24 1
%dir24_y_neg = OpFNegate %float %dir24_y
%perp24 = OpCompositeConstruct %v2float %dir24_y_neg %dir24_x
%perp24_oriented = OpVectorTimesScalar %v2float %perp24 %perp_sign
%perp24_thick = OpVectorTimesScalar %v2float %perp24_oriented %float_0_015
%off2_24 = OpVectorTimesScalar %v2float %perp24_thick %w2
%off4_24 = OpVectorTimesScalar %v2float %perp24_thick %w4

%pos2_plus_xy_24 = OpFAdd %v2float %pos2_xy %off2_24
%pos2_minus_xy_24 = OpFSub %v2float %pos2_xy %off2_24
%pos4_plus_xy_24 = OpFAdd %v2float %pos4_xy %off4_24
%pos4_minus_xy_24 = OpFSub %v2float %pos4_xy %off4_24

%z4 = OpCompositeExtract %float %pos4 2
%zbias4 = OpFMul %float %float_0_0001 %w4
%z4_edge = OpFSub %float %z4 %zbias4

%p2p_x_24 = OpCompositeExtract %float %pos2_plus_xy_24 0
%p2p_y_24 = OpCompositeExtract %float %pos2_plus_xy_24 1
%p2m_x_24 = OpCompositeExtract %float %pos2_minus_xy_24 0
%p2m_y_24 = OpCompositeExtract %float %pos2_minus_xy_24 1
%p4p_x_24 = OpCompositeExtract %float %pos4_plus_xy_24 0
%p4p_y_24 = OpCompositeExtract %float %pos4_plus_xy_24 1
%p4m_x_24 = OpCompositeExtract %float %pos4_minus_xy_24 0
%p4m_y_24 = OpCompositeExtract %float %pos4_minus_xy_24 1

%pos2_plus_24 = OpCompositeConstruct %v4float %p2p_x_24 %p2p_y_24 %z2_edge %w2
%pos2_minus_24 = OpCompositeConstruct %v4float %p2m_x_24 %p2m_y_24 %z2_edge %w2
%pos4_plus_24 = OpCompositeConstruct %v4float %p4p_x_24 %p4p_y_24 %z4_edge %w4
%pos4_minus_24 = OpCompositeConstruct %v4float %p4m_x_24 %p4m_y_24 %z4_edge %w4

OpStore %gl_out_pos_ptr %pos2_plus_24
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos2_minus_24
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos4_plus_24
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos4_minus_24
OpStore %outColor %v3_white
OpEmitVertex

OpEndPrimitive

; ---- Edge 4 -> 0 ----
%invw0_b = OpFDiv %float %float_1 %w0
%invw4_b = OpFDiv %float %float_1 %w4
%ndc0_b = OpVectorTimesScalar %v2float %pos0_xy %invw0_b
%ndc4_b = OpVectorTimesScalar %v2float %pos4_xy %invw4_b
%delta40 = OpFSub %v2float %ndc0_b %ndc4_b
%dir40 = OpExtInst %v2float %ext Normalize %delta40
%dir40_x = OpCompositeExtract %float %dir40 0
%dir40_y = OpCompositeExtract %float %dir40 1
%dir40_y_neg = OpFNegate %float %dir40_y
%perp40 = OpCompositeConstruct %v2float %dir40_y_neg %dir40_x
%perp40_oriented = OpVectorTimesScalar %v2float %perp40 %perp_sign
%perp40_thick = OpVectorTimesScalar %v2float %perp40_oriented %float_0_015
%off4_40 = OpVectorTimesScalar %v2float %perp40_thick %w4
%off0_40 = OpVectorTimesScalar %v2float %perp40_thick %w0

%pos4_plus_xy_40 = OpFAdd %v2float %pos4_xy %off4_40
%pos4_minus_xy_40 = OpFSub %v2float %pos4_xy %off4_40
%pos0_plus_xy_40 = OpFAdd %v2float %pos0_xy %off0_40
%pos0_minus_xy_40 = OpFSub %v2float %pos0_xy %off0_40

%p4p_x_40 = OpCompositeExtract %float %pos4_plus_xy_40 0
%p4p_y_40 = OpCompositeExtract %float %pos4_plus_xy_40 1
%p4m_x_40 = OpCompositeExtract %float %pos4_minus_xy_40 0
%p4m_y_40 = OpCompositeExtract %float %pos4_minus_xy_40 1
%p0p_x_40 = OpCompositeExtract %float %pos0_plus_xy_40 0
%p0p_y_40 = OpCompositeExtract %float %pos0_plus_xy_40 1
%p0m_x_40 = OpCompositeExtract %float %pos0_minus_xy_40 0
%p0m_y_40 = OpCompositeExtract %float %pos0_minus_xy_40 1

%pos4_plus_40 = OpCompositeConstruct %v4float %p4p_x_40 %p4p_y_40 %z4_edge %w4
%pos4_minus_40 = OpCompositeConstruct %v4float %p4m_x_40 %p4m_y_40 %z4_edge %w4
%pos0_plus_40 = OpCompositeConstruct %v4float %p0p_x_40 %p0p_y_40 %z0_edge %w0
%pos0_minus_40 = OpCompositeConstruct %v4float %p0m_x_40 %p0m_y_40 %z0_edge %w0

OpStore %gl_out_pos_ptr %pos4_plus_40
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos4_minus_40
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos0_plus_40
OpStore %outColor %v3_white
OpEmitVertex

OpStore %gl_out_pos_ptr %pos0_minus_40
OpStore %outColor %v3_white
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
