#include <algorithm>
#include <cstddef>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <vector>

using std::cin;
using std::cout;
using std::priority_queue;
using std::vector;

static inline long long area_ll(int w, int h) { return 1LL * w * h; }

struct RectInput {
    int id = -1;
    int w = 0;
    int h = 0;
    long long area = 0;
    int maxSide = 0;
    int minSide = 0;
};

struct Placement {
    int bin = -1;
    int x = 0;
    int y = 0;
    int rotated = 0; // 0/1
};

struct Node {
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    bool rotated = false;
};

static inline bool rectContains(const Node& a, const Node& b) {
    return b.x >= a.x && b.y >= a.y && b.x + b.w <= a.x + a.w && b.y + b.h <= a.y + a.h;
}

static inline bool betterCandidate(int s1,
                                   int s2,
                                   int y,
                                   int x,
                                   long long freeAfter,
                                   int bs1,
                                   int bs2,
                                   int by,
                                   int bx,
                                   long long bFreeAfter) {
    if (s1 != bs1) return s1 < bs1;
    if (s2 != bs2) return s2 < bs2;
    if (y != by) return y < by;
    if (x != bx) return x < bx;
    return freeAfter < bFreeAfter;
}

class MaxRectsBin {
public:
    explicit MaxRectsBin(int w, int h) : binW(w), binH(h), binArea(area_ll(w, h)) {
        freeRects.push_back(Node{0, 0, w, h, false});
    }

    long long freeArea() const { return binArea - usedArea; }

    struct Candidate {
        bool ok = false;
        Node node;
        int score1 = std::numeric_limits<int>::max();
        int score2 = std::numeric_limits<int>::max();
        long long freeAfter = 0;
    };

    Candidate scoreRectBSSF(int w, int h, bool allowRotate) const {
        Candidate best;
        const long long after = freeArea() - area_ll(w, h);

        for (const auto& fr : freeRects) {
            if (w <= fr.w && h <= fr.h) {
                const int leftoverW = fr.w - w;
                const int leftoverH = fr.h - h;
                const int shortSide = std::min(leftoverW, leftoverH);
                const int longSide = std::max(leftoverW, leftoverH);
                if (!best.ok || shortSide < best.score1 ||
                    (shortSide == best.score1 &&
                     (longSide < best.score2 ||
                      (longSide == best.score2 &&
                       (fr.y < best.node.y || (fr.y == best.node.y && fr.x < best.node.x)))))) {
                    best.ok = true;
                    best.score1 = shortSide;
                    best.score2 = longSide;
                    best.freeAfter = after;
                    best.node = Node{fr.x, fr.y, w, h, false};
                }
            }
            if (allowRotate && h <= fr.w && w <= fr.h) {
                const int leftoverW = fr.w - h;
                const int leftoverH = fr.h - w;
                const int shortSide = std::min(leftoverW, leftoverH);
                const int longSide = std::max(leftoverW, leftoverH);
                if (!best.ok || shortSide < best.score1 ||
                    (shortSide == best.score1 &&
                     (longSide < best.score2 ||
                      (longSide == best.score2 &&
                       (fr.y < best.node.y || (fr.y == best.node.y && fr.x < best.node.x)))))) {
                    best.ok = true;
                    best.score1 = shortSide;
                    best.score2 = longSide;
                    best.freeAfter = after;
                    best.node = Node{fr.x, fr.y, h, w, true};
                }
            }
            if (best.ok && best.score1 == 0 && best.score2 == 0) break;
        }

        if (!best.ok) best.freeAfter = 0;
        return best;
    }

    void placeNode(const Node& node) {
        size_t i = 0;
        size_t numToProcess = freeRects.size();
        while (i < numToProcess) {
            if (splitFreeNode(freeRects[i], node)) {
                freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(i));
                --numToProcess;
            } else {
                ++i;
            }
        }
        pruneFreeList();
        usedArea += area_ll(node.w, node.h);
    }

private:
    int binW = 0;
    int binH = 0;
    long long binArea = 0;
    long long usedArea = 0;
    vector<Node> freeRects;

    bool splitFreeNode(Node freeNode, const Node& usedNode) {
        if (usedNode.x >= freeNode.x + freeNode.w || usedNode.x + usedNode.w <= freeNode.x ||
            usedNode.y >= freeNode.y + freeNode.h || usedNode.y + usedNode.h <= freeNode.y) {
            return false;
        }

        if (usedNode.x < freeNode.x + freeNode.w && usedNode.x + usedNode.w > freeNode.x) {
            if (usedNode.y > freeNode.y && usedNode.y < freeNode.y + freeNode.h) {
                Node newNode = freeNode;
                newNode.h = usedNode.y - newNode.y;
                freeRects.push_back(newNode);
            }
            if (usedNode.y + usedNode.h < freeNode.y + freeNode.h) {
                Node newNode = freeNode;
                newNode.y = usedNode.y + usedNode.h;
                newNode.h = freeNode.y + freeNode.h - (usedNode.y + usedNode.h);
                freeRects.push_back(newNode);
            }
        }

        if (usedNode.y < freeNode.y + freeNode.h && usedNode.y + usedNode.h > freeNode.y) {
            if (usedNode.x > freeNode.x && usedNode.x < freeNode.x + freeNode.w) {
                Node newNode = freeNode;
                newNode.w = usedNode.x - newNode.x;
                freeRects.push_back(newNode);
            }
            if (usedNode.x + usedNode.w < freeNode.x + freeNode.w) {
                Node newNode = freeNode;
                newNode.x = usedNode.x + usedNode.w;
                newNode.w = freeNode.x + freeNode.w - (usedNode.x + usedNode.w);
                freeRects.push_back(newNode);
            }
        }

        return true;
    }

    void pruneFreeList() {
        for (int i = 0; i < static_cast<int>(freeRects.size()); ++i) {
            for (int j = i + 1; j < static_cast<int>(freeRects.size());) {
                if (rectContains(freeRects[i], freeRects[j])) {
                    freeRects[j] = freeRects.back();
                    freeRects.pop_back();
                } else if (rectContains(freeRects[j], freeRects[i])) {
                    freeRects[i] = freeRects.back();
                    freeRects.pop_back();
                    --i;
                    break;
                } else {
                    ++j;
                }
            }
        }
    }
};

class GuillotineBin {
public:
    GuillotineBin(int w, int h) : W(w), H(h), freeArea(area_ll(w, h)) { freeRects.push_back(Node{0, 0, w, h}); }

    struct Fit {
        bool ok = false;
        int freeIndex = -1;
        Node node;
        int score1 = std::numeric_limits<int>::max();
        int score2 = std::numeric_limits<int>::max();
        long long freeAfter = 0;
    };

    Fit findBestFitBSSF(int w, int h, bool allowRotate) const {
        Fit best;
        const long long after = freeArea - area_ll(w, h);
        for (int i = 0; i < static_cast<int>(freeRects.size()); ++i) {
            const Node& fr = freeRects[i];
            if (w <= fr.w && h <= fr.h) {
                const int leftoverW = fr.w - w;
                const int leftoverH = fr.h - h;
                const int shortSide = std::min(leftoverW, leftoverH);
                const int longSide = std::max(leftoverW, leftoverH);
                if (!best.ok || betterCandidate(shortSide, longSide, fr.y, fr.x, after, best.score1, best.score2,
                                                best.node.y, best.node.x, best.freeAfter)) {
                    best.ok = true;
                    best.freeIndex = i;
                    best.score1 = shortSide;
                    best.score2 = longSide;
                    best.freeAfter = after;
                    best.node = Node{fr.x, fr.y, w, h, false};
                    if (shortSide == 0 && longSide == 0) break;
                }
            }
            if (allowRotate && h <= fr.w && w <= fr.h) {
                const int leftoverW = fr.w - h;
                const int leftoverH = fr.h - w;
                const int shortSide = std::min(leftoverW, leftoverH);
                const int longSide = std::max(leftoverW, leftoverH);
                if (!best.ok || betterCandidate(shortSide, longSide, fr.y, fr.x, after, best.score1, best.score2,
                                                best.node.y, best.node.x, best.freeAfter)) {
                    best.ok = true;
                    best.freeIndex = i;
                    best.score1 = shortSide;
                    best.score2 = longSide;
                    best.freeAfter = after;
                    best.node = Node{fr.x, fr.y, h, w, true};
                    if (shortSide == 0 && longSide == 0) break;
                }
            }
        }
        return best;
    }

    void place(const Fit& fit) {
        const Node fr = freeRects[fit.freeIndex];
        freeRects[fit.freeIndex] = freeRects.back();
        freeRects.pop_back();

        const int placedW = fit.node.w;
        const int placedH = fit.node.h;

        // Shorter-Axis Split (SAS): split horizontally if freeRect.w < freeRect.h, else vertically.
        if (fr.w < fr.h) {
            const int topH = fr.h - placedH;
            if (topH > 0) freeRects.push_back(Node{fr.x, fr.y + placedH, fr.w, topH});
            const int rightW = fr.w - placedW;
            if (rightW > 0) freeRects.push_back(Node{fr.x + placedW, fr.y, rightW, placedH});
        } else {
            const int rightW = fr.w - placedW;
            if (rightW > 0) freeRects.push_back(Node{fr.x + placedW, fr.y, rightW, fr.h});
            const int topH = fr.h - placedH;
            if (topH > 0) freeRects.push_back(Node{fr.x, fr.y + placedH, placedW, topH});
        }

        freeArea -= area_ll(placedW, placedH);
        ++version;
    }

    long long freeArea = 0;
    int version = 0;

private:
    int W = 0;
    int H = 0;
    vector<Node> freeRects;
};

struct PQEntry {
    long long freeArea = 0;
    int idx = -1;
    int version = 0;
    bool operator<(const PQEntry& other) const { return freeArea < other.freeArea; } // max-heap by freeArea
};

int main() {
    std::ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N = 0;
    int boardW = 0, boardH = 0;
    if (!(cin >> N >> boardW >> boardH)) return 0;

    vector<RectInput> rects(N);
    long long totalArea = 0;
    for (int i = 0; i < N; ++i) {
        int w = 0, h = 0;
        cin >> w >> h;
        rects[i].id = i;
        rects[i].w = w;
        rects[i].h = h;
        rects[i].area = area_ll(w, h);
        rects[i].maxSide = std::max(w, h);
        rects[i].minSide = std::min(w, h);
        totalArea += rects[i].area;
    }

    vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = i;
    std::sort(order.begin(), order.end(), [&](int a, int b) {
        const RectInput& ra = rects[a];
        const RectInput& rb = rects[b];
        if (ra.area != rb.area) return ra.area > rb.area;
        if (ra.maxSide != rb.maxSide) return ra.maxSide > rb.maxSide;
        if (ra.minSide != rb.minSide) return ra.minSide > rb.minSide;
        return ra.id < rb.id;
    });

    vector<Placement> out(N);

    const int maxRectsThreshold = 4000;
    if (N <= maxRectsThreshold) {
        vector<MaxRectsBin> bins;
        bins.emplace_back(boardW, boardH);

        for (int idx : order) {
            const int w = rects[idx].w;
            const int h = rects[idx].h;
            const long long a = rects[idx].area;

            int bestBin = -1;
            MaxRectsBin::Candidate best;

            for (int b = 0; b < static_cast<int>(bins.size()); ++b) {
                if (a > bins[b].freeArea()) continue;
                auto cand = bins[b].scoreRectBSSF(w, h, true);
                if (!cand.ok) continue;
                if (bestBin == -1 ||
                    betterCandidate(cand.score1, cand.score2, cand.node.y, cand.node.x, cand.freeAfter, best.score1,
                                    best.score2, best.node.y, best.node.x, best.freeAfter)) {
                    bestBin = b;
                    best = cand;
                }
            }

            if (bestBin == -1) {
                bins.emplace_back(boardW, boardH);
                bestBin = static_cast<int>(bins.size()) - 1;
                best = bins[bestBin].scoreRectBSSF(w, h, true);
                // If a rectangle doesn't fit even in an empty bin, input is infeasible; emit something deterministic.
                if (!best.ok) {
                    out[idx] = Placement{bestBin, 0, 0, 0};
                    continue;
                }
            }

            bins[bestBin].placeNode(best.node);
            out[idx] = Placement{bestBin, best.node.x, best.node.y, best.node.rotated ? 1 : 0};
        }

        cout << bins.size() << "\n";
        for (int i = 0; i < N; ++i) {
            cout << out[i].bin << " " << out[i].x << " " << out[i].y << " " << out[i].rotated << "\n";
        }
        return 0;
    }

    // Large instances: Guillotine-BSSF + SAS with bounded bin search.
    vector<GuillotineBin> bins;
    bins.emplace_back(boardW, boardH);
    priority_queue<PQEntry> pq;
    pq.push(PQEntry{bins[0].freeArea, 0, bins[0].version});

    const int recentLimit = (N > 20000) ? 16 : 32;
    const int pqTries = (N > 20000) ? 8 : 16;

    vector<int> tried;
    tried.reserve(static_cast<size_t>(recentLimit + pqTries + 8));

    for (int idx : order) {
        const int w = rects[idx].w;
        const int h = rects[idx].h;
        const long long a = rects[idx].area;

        int bestBin = -1;
        GuillotineBin::Fit bestFit;

        tried.clear();
        auto alreadyTried = [&](int b) {
            for (int t : tried)
                if (t == b) return true;
            return false;
        };
        auto considerBin = [&](int b) {
            if (b < 0 || b >= static_cast<int>(bins.size())) return;
            if (alreadyTried(b)) return;
            tried.push_back(b);
            if (a > bins[b].freeArea) return;
            auto fit = bins[b].findBestFitBSSF(w, h, true);
            if (!fit.ok) return;
            if (bestBin == -1 ||
                betterCandidate(fit.score1, fit.score2, fit.node.y, fit.node.x, fit.freeAfter, bestFit.score1,
                                bestFit.score2, bestFit.node.y, bestFit.node.x, bestFit.freeAfter)) {
                bestBin = b;
                bestFit = fit;
            }
        };

        // Try most recent bins first (next-fit-ish, but pick best among them).
        for (int b = static_cast<int>(bins.size()) - 1, cnt = 0; b >= 0 && cnt < recentLimit; --b, ++cnt)
            considerBin(b);

        // Also try a few "most empty" bins (more likely to fit large rectangles).
        vector<PQEntry> popped;
        popped.reserve(static_cast<size_t>(pqTries));
        for (int t = 0; t < pqTries; ++t) {
            while (!pq.empty()) {
                const PQEntry top = pq.top();
                if (top.idx < 0 || top.idx >= static_cast<int>(bins.size())) {
                    pq.pop();
                    continue;
                }
                if (top.version != bins[top.idx].version || top.freeArea != bins[top.idx].freeArea) {
                    pq.pop();
                    continue;
                }
                break;
            }
            if (pq.empty()) break;
            PQEntry e = pq.top();
            pq.pop();
            popped.push_back(e);
            considerBin(e.idx);
        }
        for (const auto& e : popped) pq.push(e);

        if (bestBin == -1) {
            bins.emplace_back(boardW, boardH);
            bestBin = static_cast<int>(bins.size()) - 1;
            bestFit = bins[bestBin].findBestFitBSSF(w, h, true);
            if (!bestFit.ok) {
                out[idx] = Placement{bestBin, 0, 0, 0};
                continue;
            }
        }

        bins[bestBin].place(bestFit);
        pq.push(PQEntry{bins[bestBin].freeArea, bestBin, bins[bestBin].version});

        out[idx] = Placement{bestBin, bestFit.node.x, bestFit.node.y, bestFit.node.rotated ? 1 : 0};
    }

    cout << bins.size() << "\n";
    for (int i = 0; i < N; ++i) {
        cout << out[i].bin << " " << out[i].x << " " << out[i].y << " " << out[i].rotated << "\n";
    }
    return 0;
}
