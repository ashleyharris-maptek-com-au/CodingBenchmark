OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in = OpTypeArray %v3float %uint_2

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_v3 = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input
%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_eps = OpConstant %float 0.000001
%thick_start = OpConstant %float 0.04
%thick_end = OpConstant %float 0.005

%main = OpFunction %void None %fn_void
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr

%p0xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p1xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p0w = OpCompositeExtract %float %p0 3
%p1w = OpCompositeExtract %float %p1 3

%invW0 = OpFDiv %float %float_1 %p0w
%invW1 = OpFDiv %float %float_1 %p1w
%p0ndc = OpVectorTimesScalar %v2float %p0xy %invW0
%p1ndc = OpVectorTimesScalar %v2float %p1xy %invW1

%dir = OpFSub %v2float %p1ndc %p0ndc
%len = OpExtInst %float %glsl450 Length %dir
%len_safe = OpExtInst %float %glsl450 FMax %len %float_eps
%invLen = OpFDiv %float %float_1 %len_safe
%dirN = OpVectorTimesScalar %v2float %dir %invLen

%dirNx = OpCompositeExtract %float %dirN 0
%dirNy = OpCompositeExtract %float %dirN 1
%negDirNy = OpFNegate %float %dirNy
%perp = OpCompositeConstruct %v2float %negDirNy %dirNx

%offNdc0 = OpVectorTimesScalar %v2float %perp %thick_start
%offNdc1 = OpVectorTimesScalar %v2float %perp %thick_end
%offClip0 = OpVectorTimesScalar %v2float %offNdc0 %p0w
%offClip1 = OpVectorTimesScalar %v2float %offNdc1 %p1w

%off0x = OpCompositeExtract %float %offClip0 0
%off0y = OpCompositeExtract %float %offClip0 1
%off1x = OpCompositeExtract %float %offClip1 0
%off1y = OpCompositeExtract %float %offClip1 1

%off0v4 = OpCompositeConstruct %v4float %off0x %off0y %float_0 %float_0
%off1v4 = OpCompositeConstruct %v4float %off1x %off1y %float_0 %float_0

%p0_plus = OpFAdd %v4float %p0 %off0v4
%p0_minus = OpFSub %v4float %p0 %off0v4
%p1_plus = OpFAdd %v4float %p1 %off1v4
%p1_minus = OpFSub %v4float %p1 %off1v4

%outPosPtr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %outPosPtr %p0_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %outPosPtr %p0_minus
OpStore %outColor %c0
OpEmitVertex

OpStore %outPosPtr %p1_plus
OpStore %outColor %c1
OpEmitVertex

OpStore %outPosPtr %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
