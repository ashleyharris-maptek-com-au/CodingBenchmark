const fs = require('fs');

const raw = fs.readFileSync(0, 'utf8');
if (!raw || !raw.trim()) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const data = JSON.parse(raw);

const width = Math.trunc(Number(data.width));
const height = Math.trunc(Number(data.height));
const grid = Array.isArray(data.grid) ? data.grid : [];
const ignitions = Array.isArray(data.ignitions) ? data.ignitions : [];
const townsIn = Array.isArray(data.towns) ? data.towns : [];
let cutBudget = Math.trunc(Number(data.cutBudget));
if (!Number.isFinite(cutBudget) || cutBudget < 0) cutBudget = 0;

if (
  !Number.isFinite(width) ||
  !Number.isFinite(height) ||
  width <= 0 ||
  height <= 0 ||
  grid.length !== height
) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const W = width;
const H = height;
const N = W * H;

const burnable = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = String(grid[y] ?? '');
  const rowLen = row.length;
  const base = y * W;
  for (let x = 0; x < W; x++) {
    const c = x < rowLen ? row.charCodeAt(x) : 126; // '~'
    burnable[base + x] = c === 46 ? 1 : 0; // '.'
  }
}

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}

function idxOf(x, y) {
  return y * W + x;
}

const isIgnition = new Uint8Array(N);
for (let i = 0; i < ignitions.length; i++) {
  const pair = ignitions[i];
  if (!Array.isArray(pair) || pair.length < 2) continue;
  const x = Math.trunc(Number(pair[0]));
  const y = Math.trunc(Number(pair[1]));
  if (!inBounds(x, y)) continue;
  const idx = idxOf(x, y);
  if (burnable[idx]) isIgnition[idx] = 1;
}

const isTown = new Uint8Array(N);
const towns = [];
for (let i = 0; i < townsIn.length; i++) {
  const t = townsIn[i] || {};
  const x = Math.trunc(Number(t.x));
  const y = Math.trunc(Number(t.y));
  const value = Number(t.value);
  if (!Number.isFinite(value) || value <= 0) continue;
  if (!inBounds(x, y)) continue;
  const idx = idxOf(x, y);
  isTown[idx] = 1;
  towns.push({ id: t.id, x, y, idx, value });
}

// Multi-source BFS from ignitions through burnable cells.
const dist = new Int32Array(N);
dist.fill(-1);
const parent = new Int32Array(N);
parent.fill(-1);
const queue = new Int32Array(N);
let head = 0;
let tail = 0;

for (let i = 0; i < ignitions.length; i++) {
  const pair = ignitions[i];
  if (!Array.isArray(pair) || pair.length < 2) continue;
  const x = Math.trunc(Number(pair[0]));
  const y = Math.trunc(Number(pair[1]));
  if (!inBounds(x, y)) continue;
  const idx = idxOf(x, y);
  if (!burnable[idx]) continue;
  if (dist[idx] !== -1) continue;
  dist[idx] = 0;
  queue[tail++] = idx;
}

while (head < tail) {
  const idx = queue[head++];
  const nextD = dist[idx] + 1;
  const y = Math.trunc(idx / W);
  const x = idx - y * W;

  if (x > 0) {
    const n = idx - 1;
    if (burnable[n] && dist[n] === -1) {
      dist[n] = nextD;
      parent[n] = idx;
      queue[tail++] = n;
    }
  }
  if (x + 1 < W) {
    const n = idx + 1;
    if (burnable[n] && dist[n] === -1) {
      dist[n] = nextD;
      parent[n] = idx;
      queue[tail++] = n;
    }
  }
  if (y > 0) {
    const n = idx - W;
    if (burnable[n] && dist[n] === -1) {
      dist[n] = nextD;
      parent[n] = idx;
      queue[tail++] = n;
    }
  }
  if (y + 1 < H) {
    const n = idx + W;
    if (burnable[n] && dist[n] === -1) {
      dist[n] = nextD;
      parent[n] = idx;
      queue[tail++] = n;
    }
  }
}

const cutMask = new Uint8Array(N);
const cutIdxs = [];

function isCuttable(idx) {
  return burnable[idx] === 1 && cutMask[idx] === 0 && isIgnition[idx] === 0 && isTown[idx] === 0;
}

function addCut(idx) {
  cutMask[idx] = 1;
  cutIdxs.push(idx);
}

function requiredNeighborCutsForTown(town) {
  const { x, y, idx } = town;
  if (dist[idx] < 0) return null;
  if (isIgnition[idx]) return null;

  const req = [];
  // A town is protected if all *reachable* burnable neighbors are blocked.
  // Neighbors that are already unreachable from fire needn't be cut.
  if (x > 0) {
    const n = idx - 1;
    if (burnable[n] && dist[n] >= 0) {
      if (!isCuttable(n) && cutMask[n] === 0) return null;
      req.push(n);
    }
  }
  if (x + 1 < W) {
    const n = idx + 1;
    if (burnable[n] && dist[n] >= 0) {
      if (!isCuttable(n) && cutMask[n] === 0) return null;
      req.push(n);
    }
  }
  if (y > 0) {
    const n = idx - W;
    if (burnable[n] && dist[n] >= 0) {
      if (!isCuttable(n) && cutMask[n] === 0) return null;
      req.push(n);
    }
  }
  if (y + 1 < H) {
    const n = idx + W;
    if (burnable[n] && dist[n] >= 0) {
      if (!isCuttable(n) && cutMask[n] === 0) return null;
      req.push(n);
    }
  }
  return req;
}

// Phase 1: Greedily isolate high-value threatened towns by cutting their reachable neighbors.
const threatened = [];
for (let i = 0; i < towns.length; i++) {
  const t = towns[i];
  if (!burnable[t.idx]) continue;
  const d = dist[t.idx];
  if (d < 0) continue;
  const req = requiredNeighborCutsForTown(t);
  if (!req) continue;
  let cost = 0;
  for (let j = 0; j < req.length; j++) {
    if (cutMask[req[j]] === 0) cost++;
  }
  if (cost <= 0) continue;
  threatened.push({ town: t, req, cost, score: t.value / cost });
}

threatened.sort((a, b) => b.score - a.score || b.town.value - a.town.value);

for (let i = 0; i < threatened.length && cutBudget > 0; i++) {
  const item = threatened[i];
  const req = item.req;
  let addCost = 0;
  for (let j = 0; j < req.length; j++) {
    const idx = req[j];
    if (cutMask[idx] === 0) addCost++;
  }
  if (addCost <= 0) continue;
  if (addCost > cutBudget) continue;
  for (let j = 0; j < req.length; j++) {
    const idx = req[j];
    if (cutMask[idx] === 0) addCut(idx);
  }
  cutBudget -= addCost;
}

// Phase 2: Spend remaining budget on chokepoint candidates from BFS-tree paths to remaining threatened towns.
if (cutBudget > 0) {
  const townProtected = new Uint8Array(towns.length);
  // Mark protected towns from phase 1 by checking if any of their neighbor requirements were fully satisfied.
  // (Fast approximate: if all reachable burnable neighbors are now blocked by ~ or cuts.)
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    if (!burnable[t.idx]) continue;
    if (dist[t.idx] < 0) continue;
    if (isIgnition[t.idx]) continue;

    const { x, y, idx } = t;
    let safe = true;
    if (x > 0) {
      const n = idx - 1;
      if (burnable[n] && dist[n] >= 0 && cutMask[n] === 0) safe = false;
    }
    if (safe && x + 1 < W) {
      const n = idx + 1;
      if (burnable[n] && dist[n] >= 0 && cutMask[n] === 0) safe = false;
    }
    if (safe && y > 0) {
      const n = idx - W;
      if (burnable[n] && dist[n] >= 0 && cutMask[n] === 0) safe = false;
    }
    if (safe && y + 1 < H) {
      const n = idx + W;
      if (burnable[n] && dist[n] >= 0 && cutMask[n] === 0) safe = false;
    }
    if (safe) townProtected[i] = 1;
  }

  const remainingTowns = [];
  for (let i = 0; i < towns.length; i++) {
    if (townProtected[i]) continue;
    const t = towns[i];
    if (!burnable[t.idx]) continue;
    if (dist[t.idx] < 0) continue;
    if (isIgnition[t.idx]) continue;
    remainingTowns.push(t);
  }
  remainingTowns.sort((a, b) => b.value - a.value);

  const pathTownLimit = 600;
  const maxPathLen = 4096;
  const weights = new Map(); // idx -> accumulated value

  const limit = Math.min(pathTownLimit, remainingTowns.length);
  for (let i = 0; i < limit; i++) {
    const t = remainingTowns[i];
    let cur = t.idx;
    let steps = 0;
    while (steps++ < maxPathLen) {
      const p = parent[cur];
      if (p < 0) break;
      cur = p;
      if (dist[cur] <= 0) break;
      if (!isCuttable(cur)) continue;
      weights.set(cur, (weights.get(cur) || 0) + t.value);
    }
  }

  if (weights.size > 0) {
    const cand = [];
    for (const [idx, w] of weights.entries()) cand.push([idx, w]);
    cand.sort((a, b) => b[1] - a[1]);

    for (let i = 0; i < cand.length && cutBudget > 0; i++) {
      const idx = cand[i][0];
      if (!isCuttable(idx)) continue;
      addCut(idx);
      cutBudget--;
    }
  }
}

const firebreaks = [];
for (let i = 0; i < cutIdxs.length; i++) {
  const idx = cutIdxs[i];
  const y = Math.trunc(idx / W);
  const x = idx - y * W;
  firebreaks.push([x, y]);
}

process.stdout.write(JSON.stringify({ firebreaks }));
