; SPIR-V
; Version: 1.0
; Schema: 0
OpCapability Shader
OpCapability Geometry
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputTrianglesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0
OpDecorate %arr_pv_in ArrayStride 16

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_6 = OpConstant %uint 6

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%arr_v3_in = OpTypeArray %v3float %uint_6

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_in_v4 = OpTypePointer Input %v4float

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_2 = OpConstant %int 2
%int_4 = OpConstant %int 4

%float_0 = OpConstant %float 0.0
%float_0_2 = OpConstant %float 0.2
%float_0_8 = OpConstant %float 0.8
%float_1 = OpConstant %float 1.0
%float_0_33333334 = OpConstant %float 0.33333334

%v3_ones = OpConstantComposite %v3float %float_1 %float_1 %float_1

%main = OpFunction %void None %fn_void
%entry = OpLabel

%worldPos_0_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_0
%p0 = OpLoad %v3float %worldPos_0_ptr
%worldPos_2_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_2
%p2 = OpLoad %v3float %worldPos_2_ptr
%worldPos_4_ptr = OpAccessChain %ptr_in_v3 %worldPos %int_4
%p4 = OpLoad %v3float %worldPos_4_ptr

%a = OpFSub %v3float %p2 %p0
%b = OpFSub %v3float %p4 %p0

%a_x = OpCompositeExtract %float %a 0
%a_y = OpCompositeExtract %float %a 1
%a_z = OpCompositeExtract %float %a 2
%b_x = OpCompositeExtract %float %b 0
%b_y = OpCompositeExtract %float %b 1
%b_z = OpCompositeExtract %float %b 2

%cx_mul1 = OpFMul %float %a_y %b_z
%cx_mul2 = OpFMul %float %a_z %b_y
%cx = OpFSub %float %cx_mul1 %cx_mul2

%cy_mul1 = OpFMul %float %a_z %b_x
%cy_mul2 = OpFMul %float %a_x %b_z
%cy = OpFSub %float %cy_mul1 %cy_mul2

%cz_mul1 = OpFMul %float %a_x %b_y
%cz_mul2 = OpFMul %float %a_y %b_x
%cz = OpFSub %float %cz_mul1 %cz_mul2

%cross = OpCompositeConstruct %v3float %cx %cy %cz
%N = OpExtInst %v3float %GLSLstd450 Normalize %cross

%L = OpExtInst %v3float %GLSLstd450 Normalize %v3_ones
%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %GLSLstd450 FMax %ndotl %float_0

%color_0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c0 = OpLoad %v3float %color_0_ptr
%color_2_ptr = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %color_2_ptr
%color_4_ptr = OpAccessChain %ptr_in_v3 %color %int_4
%c4 = OpLoad %v3float %color_4_ptr

%c0_plus_c2 = OpFAdd %v3float %c0 %c2
%sumColor = OpFAdd %v3float %c0_plus_c2 %c4
%baseColor = OpVectorTimesScalar %v3float %sumColor %float_0_33333334

%diffuseScaled = OpFMul %float %float_0_8 %diffuse
%lightFactor = OpFAdd %float %float_0_2 %diffuseScaled
%finalColor = OpVectorTimesScalar %v3float %baseColor %lightFactor

%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

%in_pos_0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %in_pos_0_ptr
OpStore %out_pos_ptr %pos0
OpStore %outColor %finalColor
OpEmitVertex

%in_pos_2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %in_pos_2_ptr
OpStore %out_pos_ptr %pos2
OpStore %outColor %finalColor
OpEmitVertex

%in_pos_4_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
%pos4 = OpLoad %v4float %in_pos_4_ptr
OpStore %out_pos_ptr %pos4
OpStore %outColor %finalColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
