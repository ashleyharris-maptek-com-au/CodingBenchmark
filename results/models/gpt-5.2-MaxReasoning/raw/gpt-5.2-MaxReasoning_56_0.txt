'use strict';

const fs = require('fs');

const input = fs.readFileSync(0, 'utf8').trim();
if (!input) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

/** @type {{depot:[number,number], truckCount:number, packages:Array<{id:any,x:number,y:number}>}} */
const data = JSON.parse(input);

const depot = Array.isArray(data.depot) ? data.depot : [0, 0];
const depotX = +depot[0];
const depotY = +depot[1];

const packages = Array.isArray(data.packages) ? data.packages : [];
const n = packages.length;

if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const ids = new Array(n);
const xs = new Float64Array(n);
const ys = new Float64Array(n);

let minX = Infinity;
let maxX = -Infinity;
let minY = Infinity;
let maxY = -Infinity;
let sumX = 0;
let sumY = 0;

for (let i = 0; i < n; i++) {
  const p = packages[i];
  const x = +p.x;
  const y = +p.y;
  ids[i] = p.id;
  xs[i] = x;
  ys[i] = y;
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
  sumX += x;
  sumY += y;
}

function compareId(a, b) {
  if (a === b) return 0;
  if (typeof a === 'number' && typeof b === 'number') return a - b;
  const sa = String(a);
  const sb = String(b);
  return sa < sb ? -1 : sa > sb ? 1 : 0;
}

function distIdx(a, b) {
  const dx = xs[a] - xs[b];
  const dy = ys[a] - ys[b];
  return Math.sqrt(dx * dx + dy * dy);
}

function distDepotIdx(i) {
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  return Math.sqrt(dx * dx + dy * dy);
}

function clampU16(v) {
  if (v < 0) return 0;
  if (v > 65535) return 65535;
  return v | 0;
}

function part1By1(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n >>> 0;
}

function morton2D(x, y) {
  return (part1By1(x) | (part1By1(y) << 1)) >>> 0;
}

function bestDepotCut(order) {
  const m = order.length;
  if (m === 1) {
    const only = order[0];
    const d = distDepotIdx(only);
    return { start: 0, cost: 2 * d };
  }

  let cycleSum = 0;
  const edgeDist = new Float64Array(m);
  for (let i = 0; i < m; i++) {
    const a = order[i];
    const b = order[(i + 1) % m];
    const d = distIdx(a, b);
    edgeDist[i] = d;
    cycleSum += d;
  }

  let bestCost = Infinity;
  let bestStart = 0;
  for (let i = 0; i < m; i++) {
    const prevPos = (i - 1 + m) % m;
    const prev = order[prevPos];
    const curr = order[i];
    const cutEdge = edgeDist[prevPos]; // prev -> curr in the cycle
    const cost = cycleSum - cutEdge + distDepotIdx(curr) + distDepotIdx(prev);
    if (cost < bestCost) {
      bestCost = cost;
      bestStart = i;
    }
  }
  return { start: bestStart, cost: bestCost };
}

// Precompute candidate keys
const base = new Array(n);
for (let i = 0; i < n; i++) base[i] = i;

// Candidate 1: Morton (Z-order)
const mortonKeys = new Uint32Array(n);
const rangeX = maxX - minX;
const rangeY = maxY - minY;
const invRangeX = rangeX !== 0 ? 65535 / rangeX : 0;
const invRangeY = rangeY !== 0 ? 65535 / rangeY : 0;
for (let i = 0; i < n; i++) {
  const sx = rangeX !== 0 ? clampU16(Math.floor((xs[i] - minX) * invRangeX)) : 0;
  const sy = rangeY !== 0 ? clampU16(Math.floor((ys[i] - minY) * invRangeY)) : 0;
  mortonKeys[i] = morton2D(sx, sy);
}
const orderMorton = base.slice();
orderMorton.sort((i, j) => {
  const ki = mortonKeys[i];
  const kj = mortonKeys[j];
  if (ki !== kj) return ki < kj ? -1 : 1;
  const dx = xs[i] - xs[j];
  if (dx !== 0) return dx < 0 ? -1 : 1;
  const dy = ys[i] - ys[j];
  if (dy !== 0) return dy < 0 ? -1 : 1;
  return compareId(ids[i], ids[j]);
});
const cutMorton = bestDepotCut(orderMorton);

let bestOrder = orderMorton;
let bestStart = cutMorton.start;
let bestCost = cutMorton.cost;

// Candidate 2: polar angle around depot
const angleKey = new Float64Array(n);
const angleDist2 = new Float64Array(n);
for (let i = 0; i < n; i++) {
  const dx = xs[i] - depotX;
  const dy = ys[i] - depotY;
  angleKey[i] = Math.atan2(dy, dx);
  angleDist2[i] = dx * dx + dy * dy;
}
const orderAngle = base.slice();
orderAngle.sort((i, j) => {
  const ai = angleKey[i];
  const aj = angleKey[j];
  if (ai !== aj) return ai < aj ? -1 : 1;
  const di = angleDist2[i];
  const dj = angleDist2[j];
  if (di !== dj) return di < dj ? -1 : 1;
  return compareId(ids[i], ids[j]);
});
const cutAngle = bestDepotCut(orderAngle);
if (cutAngle.cost < bestCost) {
  bestOrder = orderAngle;
  bestStart = cutAngle.start;
  bestCost = cutAngle.cost;
}

// Candidate 3: PCA projection order
const meanX = sumX / n;
const meanY = sumY / n;
let sxx = 0;
let syy = 0;
let sxy = 0;
for (let i = 0; i < n; i++) {
  const dx = xs[i] - meanX;
  const dy = ys[i] - meanY;
  sxx += dx * dx;
  syy += dy * dy;
  sxy += dx * dy;
}
const a = sxx / n;
const c = syy / n;
const b = sxy / n;
let vx = 1;
let vy = 0;
if (b !== 0 || a !== c) {
  if (b === 0) {
    if (a >= c) {
      vx = 1;
      vy = 0;
    } else {
      vx = 0;
      vy = 1;
    }
  } else {
    const tr = a + c;
    const disc = Math.sqrt((a - c) * (a - c) + 4 * b * b);
    const lambda1 = (tr + disc) / 2;
    vx = b;
    vy = lambda1 - a;
  }
}
const vlen = Math.sqrt(vx * vx + vy * vy) || 1;
vx /= vlen;
vy /= vlen;

const proj = new Float64Array(n);
const ortho = new Float64Array(n);
for (let i = 0; i < n; i++) {
  const dx = xs[i] - meanX;
  const dy = ys[i] - meanY;
  proj[i] = dx * vx + dy * vy;
  ortho[i] = -dx * vy + dy * vx;
}

const orderPca = base.slice();
orderPca.sort((i, j) => {
  const pi = proj[i];
  const pj = proj[j];
  if (pi !== pj) return pi < pj ? -1 : 1;
  const oi = ortho[i];
  const oj = ortho[j];
  if (oi !== oj) return oi < oj ? -1 : 1;
  return compareId(ids[i], ids[j]);
});
const cutPca = bestDepotCut(orderPca);
if (cutPca.cost < bestCost) {
  bestOrder = orderPca;
  bestStart = cutPca.start;
  bestCost = cutPca.cost;
}

// Build a single (best) route; using more routes can only increase total distance in metric space.
const route = new Array(n);
for (let k = 0; k < n; k++) {
  route[k] = ids[bestOrder[(bestStart + k) % n]];
}

process.stdout.write(JSON.stringify({ routes: [route] }));
