using System;
using System.Diagnostics;
using System.Globalization;
using System.Text;

public static class Program
{
    private const double EarthRadiusKm = 6371.0;

    public static void Main()
    {
        var scanner = new FastScanner();
        if (!scanner.TryNextInt(out int n))
            return;

        double mu = scanner.NextDouble();
        State start = ReadState(scanner);

        var stations = new State[n];
        for (int i = 0; i < n; i++)
            stations[i] = ReadState(scanner);

        var startOrbit = OrbitFeatures.FromState(start, mu);
        var orbits = new OrbitFeatures[n];
        for (int i = 0; i < n; i++)
            orbits[i] = OrbitFeatures.FromState(stations[i], mu);

        var distStart = new double[n];
        for (int i = 0; i < n; i++)
            distStart[i] = EstimateDeltaV(startOrbit, orbits[i], mu);

        var dist = new double[n][];
        for (int i = 0; i < n; i++)
            dist[i] = new double[n];

        for (int i = 0; i < n; i++)
        {
            dist[i][i] = 0.0;
            for (int j = i + 1; j < n; j++)
            {
                double d = EstimateDeltaV(orbits[i], orbits[j], mu);
                dist[i][j] = d;
                dist[j][i] = d;
            }
        }

        int[] route;
        var sw = Stopwatch.StartNew();
        const double timeLimitSeconds = 290.0;

        if (n <= 18)
        {
            route = SolveExact(n, dist, distStart);
        }
        else
        {
            route = SolveHeuristic(n, dist, distStart, sw, timeLimitSeconds);
        }

        if (route.Length != n)
        {
            route = new int[n];
            for (int i = 0; i < n; i++)
                route[i] = i;
        }

        var sb = new StringBuilder(n * 3);
        for (int i = 0; i < n; i++)
        {
            if (i != 0) sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    private static State ReadState(FastScanner scanner)
    {
        double x = scanner.NextDouble();
        double y = scanner.NextDouble();
        double z = scanner.NextDouble();
        double vx = scanner.NextDouble();
        double vy = scanner.NextDouble();
        double vz = scanner.NextDouble();
        return new State(new Vec3(x, y, z), new Vec3(vx, vy, vz));
    }

    // --- Physics / Cost model ---
    private static double EstimateDeltaV(OrbitFeatures o1, OrbitFeatures o2, double mu)
    {
        // Coarse but fast, time-free rendezvous proxy:
        //   - Hohmann-like energy change between representative radii (a)
        //   - Plane change from angular-momentum vectors
        //   - Eccentricity vector mismatch penalty
        double r1 = SafeRadius(o1.A);
        double r2 = SafeRadius(o2.A);

        double vc1 = Math.Sqrt(mu / r1);
        double vc2 = Math.Sqrt(mu / r2);

        double dvH = HohmannDeltaV(mu, r1, r2);

        double dot = Vec3.Dot(o1.HHat, o2.HHat);
        if (double.IsNaN(dot)) dot = 1.0;
        dot = Clamp(dot, -1.0, 1.0);
        double planeAngle = Math.Acos(dot);
        double vPlane = Math.Min(vc1, vc2);
        double dvPlane = 2.0 * vPlane * Math.Sin(0.5 * planeAngle);

        Vec3 de = o1.Evec - o2.Evec;
        double vE = 0.5 * (vc1 + vc2);
        double dvE = vE * de.Norm();

        double dv = dvH + dvPlane + 0.5 * dvE;
        if (!double.IsFinite(dv))
            return 1e9;
        return dv;
    }

    private static double SafeRadius(double a)
    {
        double r = a;
        if (!double.IsFinite(r) || r <= 0)
            r = EarthRadiusKm + 500.0;
        r = Clamp(r, EarthRadiusKm + 150.0, 1_000_000.0);
        return r;
    }

    private static double HohmannDeltaV(double mu, double r1, double r2)
    {
        if (r1 <= 0 || r2 <= 0)
            return 1e9;

        if (Math.Abs(r1 - r2) < 1e-9)
            return 0.0;

        double aT = 0.5 * (r1 + r2);
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        double vP = Math.Sqrt(mu * (2.0 / r1 - 1.0 / aT));
        double vA = Math.Sqrt(mu * (2.0 / r2 - 1.0 / aT));
        return Math.Abs(vP - v1) + Math.Abs(v2 - vA);
    }

    // --- Exact solver (bitmask DP) for small N ---
    private static int[] SolveExact(int n, double[][] dist, double[] distStart)
    {
        int fullMask = (1 << n) - 1;
        int stateCount = (1 << n) * n;
        var dp = new double[stateCount];
        var parent = new int[stateCount];
        Array.Fill(dp, double.PositiveInfinity);
        Array.Fill(parent, -1);

        for (int i = 0; i < n; i++)
        {
            int mask = 1 << i;
            int idx = mask * n + i;
            dp[idx] = distStart[i];
            parent[idx] = -1;
        }

        for (int mask = 1; mask <= fullMask; mask++)
        {
            int baseIdx = mask * n;
            for (int last = 0; last < n; last++)
            {
                if ((mask & (1 << last)) == 0)
                    continue;

                double cur = dp[baseIdx + last];
                if (!double.IsFinite(cur))
                    continue;

                int remaining = fullMask ^ mask;
                while (remaining != 0)
                {
                    int lb = remaining & -remaining;
                    int next = BitIndex(lb);
                    remaining ^= lb;

                    int newMask = mask | (1 << next);
                    int newIdx = newMask * n + next;
                    double cand = cur + dist[last][next];
                    if (cand < dp[newIdx])
                    {
                        dp[newIdx] = cand;
                        parent[newIdx] = last;
                    }
                }
            }
        }

        double best = double.PositiveInfinity;
        int bestLast = 0;
        int fullBase = fullMask * n;
        for (int last = 0; last < n; last++)
        {
            double v = dp[fullBase + last];
            if (v < best)
            {
                best = v;
                bestLast = last;
            }
        }

        var route = new int[n];
        int curMask = fullMask;
        int curLast = bestLast;
        for (int pos = n - 1; pos >= 0; pos--)
        {
            route[pos] = curLast;
            int idx = curMask * n + curLast;
            int prev = parent[idx];
            curMask ^= 1 << curLast;
            if (prev < 0)
                break;
            curLast = prev;
        }

        return route;
    }

    private static int BitIndex(int singleBit)
    {
        // singleBit is power of two
        int idx = 0;
        while ((singleBit >>= 1) != 0)
            idx++;
        return idx;
    }

    // --- Heuristic solver for larger N ---
    private static int[] SolveHeuristic(int n, double[][] dist, double[] distStart, Stopwatch sw, double timeLimitSeconds)
    {
        var idsByStart = new int[n];
        for (int i = 0; i < n; i++) idsByStart[i] = i;
        Array.Sort(idsByStart, (a, b) => distStart[a].CompareTo(distStart[b]));

        int restarts = n <= 25 ? 70 : (n <= 60 ? 45 : 30);
        int maxLocalIters = n <= 40 ? 900 : (n <= 70 ? 600 : 450);

        var rng = new Random(1234567);

        int[] bestRoute = BuildNearestNeighborRcl(n, dist, distStart, forcedFirst: idsByStart[0], rclStart: 1, rclStep: 1, rng);
        LocalSearch(bestRoute, dist, distStart, sw, timeLimitSeconds, maxLocalIters);
        double bestCost = PathCost(bestRoute, dist, distStart);

        for (int t = 0; t < restarts && sw.Elapsed.TotalSeconds < timeLimitSeconds; t++)
        {
            int? forcedFirst = null;
            int rclStart = 4;
            int rclStep = 3;

            if (t < Math.Min(n, 12))
            {
                forcedFirst = idsByStart[t];
                rclStart = 1;
                rclStep = 1;
            }

            int[] route = BuildNearestNeighborRcl(n, dist, distStart, forcedFirst, rclStart, rclStep, rng);
            LocalSearch(route, dist, distStart, sw, timeLimitSeconds, maxLocalIters);
            double cost = PathCost(route, dist, distStart);
            if (cost < bestCost)
            {
                bestCost = cost;
                bestRoute = route;
            }
        }

        // Final polish pass if time permits.
        if (sw.Elapsed.TotalSeconds < timeLimitSeconds * 0.98)
            LocalSearch(bestRoute, dist, distStart, sw, timeLimitSeconds, maxLocalIters);

        return bestRoute;
    }

    private static int[] BuildNearestNeighborRcl(
        int n,
        double[][] dist,
        double[] distStart,
        int? forcedFirst,
        int rclStart,
        int rclStep,
        Random rng)
    {
        var used = new bool[n];
        var route = new int[n];

        int first;
        if (forcedFirst.HasValue)
        {
            first = forcedFirst.Value;
        }
        else
        {
            first = PickFromRclStart(n, distStart, used, rclStart, rng);
        }

        route[0] = first;
        used[first] = true;
        int cur = first;

        for (int pos = 1; pos < n; pos++)
        {
            int next = PickFromRclNext(n, dist, cur, used, rclStep, rng);
            route[pos] = next;
            used[next] = true;
            cur = next;
        }

        return route;
    }

    private static int PickFromRclStart(int n, double[] distStart, bool[] used, int rclSize, Random rng)
    {
        rclSize = Math.Max(1, Math.Min(rclSize, n));
        Span<int> cand = stackalloc int[8];
        Span<double> candD = stackalloc double[8];
        int k = Math.Min(rclSize, 8);
        for (int i = 0; i < k; i++) { cand[i] = -1; candD[i] = double.PositiveInfinity; }

        for (int j = 0; j < n; j++)
        {
            if (used[j]) continue;
            double d = distStart[j];
            InsertCandidate(cand, candD, k, j, d);
        }

        int count = 0;
        for (int i = 0; i < k; i++)
            if (cand[i] >= 0) count++;
        if (count == 0)
            return 0;
        return cand[rng.Next(count)];
    }

    private static int PickFromRclNext(int n, double[][] dist, int cur, bool[] used, int rclSize, Random rng)
    {
        rclSize = Math.Max(1, Math.Min(rclSize, n));
        Span<int> cand = stackalloc int[8];
        Span<double> candD = stackalloc double[8];
        int k = Math.Min(rclSize, 8);
        for (int i = 0; i < k; i++) { cand[i] = -1; candD[i] = double.PositiveInfinity; }

        var row = dist[cur];
        for (int j = 0; j < n; j++)
        {
            if (used[j]) continue;
            double d = row[j];
            InsertCandidate(cand, candD, k, j, d);
        }

        int count = 0;
        for (int i = 0; i < k; i++)
            if (cand[i] >= 0) count++;
        if (count == 0)
            return 0;
        return cand[rng.Next(count)];
    }

    private static void InsertCandidate(Span<int> cand, Span<double> candD, int k, int id, double d)
    {
        for (int i = 0; i < k; i++)
        {
            if (d < candD[i])
            {
                for (int j = k - 1; j > i; j--)
                {
                    candD[j] = candD[j - 1];
                    cand[j] = cand[j - 1];
                }
                candD[i] = d;
                cand[i] = id;
                return;
            }
        }
    }

    private enum MoveType : byte { None = 0, TwoOpt = 1, Swap = 2 }

    private static void LocalSearch(int[] route, double[][] dist, double[] distStart, Stopwatch sw, double timeLimitSeconds, int maxIters)
    {
        int n = route.Length;
        const double eps = 1e-10;

        for (int iter = 0; iter < maxIters && sw.Elapsed.TotalSeconds < timeLimitSeconds; iter++)
        {
            double bestDelta = 0.0;
            int bestI = -1;
            int bestK = -1;
            MoveType bestMove = MoveType.None;

            // Best 2-opt (segment reversal) move
            for (int i = 0; i < n - 1; i++)
            {
                int prev = (i == 0) ? -1 : route[i - 1];
                int a = route[i];
                double prevToA = Edge(prev, a, dist, distStart);

                for (int k = i + 1; k < n; k++)
                {
                    int b = route[k];
                    int next = (k == n - 1) ? -1 : route[k + 1];
                    double delta = (Edge(prev, b, dist, distStart) + Edge(a, next, dist, distStart))
                                   - (prevToA + Edge(b, next, dist, distStart));
                    if (delta < bestDelta - eps)
                    {
                        bestDelta = delta;
                        bestI = i;
                        bestK = k;
                        bestMove = MoveType.TwoOpt;
                    }
                }
            }

            // Best swap move
            for (int i = 0; i < n - 1; i++)
            {
                for (int j = i + 1; j < n; j++)
                {
                    double delta = SwapDelta(route, i, j, dist, distStart);
                    if (delta < bestDelta - eps)
                    {
                        bestDelta = delta;
                        bestI = i;
                        bestK = j;
                        bestMove = MoveType.Swap;
                    }
                }
            }

            if (bestMove == MoveType.None)
                return;

            if (bestMove == MoveType.TwoOpt)
            {
                Array.Reverse(route, bestI, bestK - bestI + 1);
            }
            else
            {
                int tmp = route[bestI];
                route[bestI] = route[bestK];
                route[bestK] = tmp;
            }
        }
    }

    private static double SwapDelta(int[] route, int i, int j, double[][] dist, double[] distStart)
    {
        if (i == j) return 0.0;
        if (i > j) { int t = i; i = j; j = t; }

        int n = route.Length;
        int a = route[i];
        int b = route[j];

        int prevA = (i == 0) ? -1 : route[i - 1];
        int nextA = (i == n - 1) ? -1 : route[i + 1];
        int prevB = (j == 0) ? -1 : route[j - 1];
        int nextB = (j == n - 1) ? -1 : route[j + 1];

        if (j == i + 1)
        {
            // prevA -> a -> b -> nextB  becomes  prevA -> b -> a -> nextB
            double oldCost = Edge(prevA, a, dist, distStart) + Edge(a, b, dist, distStart) + Edge(b, nextB, dist, distStart);
            double newCost = Edge(prevA, b, dist, distStart) + Edge(b, a, dist, distStart) + Edge(a, nextB, dist, distStart);
            return newCost - oldCost;
        }
        else
        {
            double oldCost = Edge(prevA, a, dist, distStart) + Edge(a, nextA, dist, distStart)
                           + Edge(prevB, b, dist, distStart) + Edge(b, nextB, dist, distStart);
            double newCost = Edge(prevA, b, dist, distStart) + Edge(b, nextA, dist, distStart)
                           + Edge(prevB, a, dist, distStart) + Edge(a, nextB, dist, distStart);
            return newCost - oldCost;
        }
    }

    private static double Edge(int from, int to, double[][] dist, double[] distStart)
    {
        if (to < 0) return 0.0;
        if (from < 0) return distStart[to];
        return dist[from][to];
    }

    private static double PathCost(int[] route, double[][] dist, double[] distStart)
    {
        if (route.Length == 0) return 0.0;
        double cost = distStart[route[0]];
        for (int i = 1; i < route.Length; i++)
            cost += dist[route[i - 1]][route[i]];
        return cost;
    }

    // --- Math helpers ---
    private static double Clamp(double x, double lo, double hi)
    {
        if (x < lo) return lo;
        if (x > hi) return hi;
        return x;
    }

    // --- Data types ---
    private readonly struct State
    {
        public readonly Vec3 R;
        public readonly Vec3 V;
        public State(Vec3 r, Vec3 v) { R = r; V = v; }
    }

    private readonly struct OrbitFeatures
    {
        public readonly double A;
        public readonly Vec3 Evec;
        public readonly Vec3 HHat;

        public OrbitFeatures(double a, Vec3 evec, Vec3 hHat)
        {
            A = a;
            Evec = evec;
            HHat = hHat;
        }

        public static OrbitFeatures FromState(State s, double mu)
        {
            Vec3 r = s.R;
            Vec3 v = s.V;

            double rMag = r.Norm();
            double v2 = v.Dot(v);
            double denom = 2.0 / Math.Max(rMag, 1e-9) - v2 / mu;
            double a = denom != 0.0 ? 1.0 / denom : rMag;
            if (!double.IsFinite(a) || a <= 0.0)
                a = rMag;

            Vec3 h = Vec3.Cross(r, v);
            double hMag = h.Norm();
            Vec3 hHat = hMag > 1e-12 ? h / hMag : new Vec3(0, 0, 1);

            Vec3 evec;
            if (hMag > 1e-12 && rMag > 1e-12)
            {
                evec = (Vec3.Cross(v, h) / mu) - (r / rMag);
                if (!evec.IsFinite())
                    evec = new Vec3(0, 0, 0);
            }
            else
            {
                evec = new Vec3(0, 0, 0);
            }

            return new OrbitFeatures(a, evec, hHat);
        }
    }

    private readonly struct Vec3
    {
        public readonly double X;
        public readonly double Y;
        public readonly double Z;

        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }

        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double k) => new Vec3(a.X * k, a.Y * k, a.Z * k);
        public static Vec3 operator /(Vec3 a, double k) => new Vec3(a.X / k, a.Y / k, a.Z / k);

        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public static double Dot(Vec3 a, Vec3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;

        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);

        public bool IsFinite() => double.IsFinite(X) && double.IsFinite(Y) && double.IsFinite(Z);

        public static Vec3 Cross(Vec3 a, Vec3 b)
        {
            return new Vec3(
                a.Y * b.Z - a.Z * b.Y,
                a.Z * b.X - a.X * b.Z,
                a.X * b.Y - a.Y * b.X
            );
        }
    }

    // --- Input ---
    private sealed class FastScanner
    {
        private readonly string _s;
        private int _pos;

        public FastScanner()
        {
            _s = Console.In.ReadToEnd();
            _pos = 0;
        }

        public bool TryNextInt(out int value)
        {
            if (!TryNextToken(out ReadOnlySpan<char> tok))
            {
                value = default;
                return false;
            }
            value = int.Parse(tok, NumberStyles.Integer, CultureInfo.InvariantCulture);
            return true;
        }

        public double NextDouble()
        {
            if (!TryNextToken(out ReadOnlySpan<char> tok))
                return 0.0;
            return double.Parse(tok, NumberStyles.Float, CultureInfo.InvariantCulture);
        }

        private bool TryNextToken(out ReadOnlySpan<char> tok)
        {
            while (_pos < _s.Length && char.IsWhiteSpace(_s[_pos])) _pos++;
            if (_pos >= _s.Length)
            {
                tok = default;
                return false;
            }
            int start = _pos;
            while (_pos < _s.Length && !char.IsWhiteSpace(_s[_pos])) _pos++;
            tok = _s.AsSpan(start, _pos - start);
            return true;
        }
    }
}
