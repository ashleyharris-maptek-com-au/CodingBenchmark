using System;
using System.IO;
using System.Runtime.CompilerServices;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private byte Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return 0;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            byte c;
            do
            {
                c = Read();
                if (c == 0) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == (byte)'-')
            {
                sign = -1;
                c = Read();
            }

            int value = 0;
            while (c > 32 && c != 0)
            {
                value = unchecked(value * 10 + (c - (byte)'0'));
                c = Read();
            }
            return value * sign;
        }
    }

    private sealed class FastWriter : IDisposable
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _pos;

        public FastWriter(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteByte(byte b)
        {
            if (_pos >= _buffer.Length)
            {
                _stream.Write(_buffer, 0, _pos);
                _pos = 0;
            }
            _buffer[_pos++] = b;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void WriteInt(int value)
        {
            if (value == 0)
            {
                WriteByte((byte)'0');
                return;
            }

            long v = value;
            if (v < 0)
            {
                WriteByte((byte)'-');
                v = -v;
            }

            Span<byte> tmp = stackalloc byte[20];
            int len = 0;
            while (v > 0)
            {
                tmp[len++] = (byte)('0' + (int)(v % 10));
                v /= 10;
            }
            for (int i = len - 1; i >= 0; i--)
                WriteByte(tmp[i]);
        }

        public void Flush()
        {
            if (_pos > 0)
            {
                _stream.Write(_buffer, 0, _pos);
                _pos = 0;
            }
        }

        public void Dispose() => Flush();
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();
        if (m == int.MinValue) return;

        if (n <= 0)
        {
            return;
        }

        int arcCapacity = m <= 0 ? 0 : checked(m * 2);

        int[] head = new int[n];
        Array.Fill(head, -1);

        int[] to = new int[arcCapacity];
        int[] next = new int[arcCapacity];
        int edgePtr = 0;

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        void AddArc(int u, int v)
        {
            to[edgePtr] = v;
            next[edgePtr] = head[u];
            head[u] = edgePtr;
            edgePtr++;
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            _ = fs.NextInt(); // weight (unused for feasibility)
            if (u == int.MinValue || v == int.MinValue) break;
            AddArc(u, v);
            AddArc(v, u);
        }

        int totalArcs = edgePtr;

        int[] iter = new int[n];
        Array.Copy(head, iter, n);

        int[] stack = new int[totalArcs + 1];
        int[] circuit = new int[totalArcs + 1];
        int sp = 0;
        int cp = 0;

        stack[sp++] = 0;
        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = iter[v];
            if (e != -1)
            {
                iter[v] = next[e];
                stack[sp++] = to[e];
            }
            else
            {
                circuit[cp++] = v;
                sp--;
            }
        }

        using var fw = new FastWriter(Console.OpenStandardOutput());

        // circuit is built in reverse; output reversed so it starts at 0 and ends at 0.
        for (int i = cp - 1; i >= 0; i--)
        {
            fw.WriteInt(circuit[i]);
            if (i != 0) fw.WriteByte((byte)' ');
        }
        fw.WriteByte((byte)'\n');
    }
}
