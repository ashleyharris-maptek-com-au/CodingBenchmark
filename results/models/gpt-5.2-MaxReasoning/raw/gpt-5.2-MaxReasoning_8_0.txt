use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read, Write};

struct FastReader<R: Read> {
    reader: R,
    buf: [u8; 1 << 16],
    len: usize,
    pos: usize,
    pushback: Option<u8>,
}

impl<R: Read> FastReader<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: [0; 1 << 16],
            len: 0,
            pos: 0,
            pushback: None,
        }
    }

    #[inline]
    fn refill(&mut self) -> bool {
        self.len = self.reader.read(&mut self.buf).unwrap_or(0);
        self.pos = 0;
        self.len != 0
    }

    #[inline]
    fn next_byte(&mut self) -> Option<u8> {
        if let Some(b) = self.pushback.take() {
            return Some(b);
        }
        if self.pos >= self.len && !self.refill() {
            return None;
        }
        let b = self.buf[self.pos];
        self.pos += 1;
        Some(b)
    }

    #[inline]
    fn unread_byte(&mut self, b: u8) {
        debug_assert!(self.pushback.is_none());
        self.pushback = Some(b);
    }

    fn skip_ascii_whitespace(&mut self) {
        while let Some(b) = self.next_byte() {
            if b.is_ascii_whitespace() {
                continue;
            }
            self.unread_byte(b);
            break;
        }
    }

    fn next_usize(&mut self) -> Option<usize> {
        self.skip_ascii_whitespace();
        let mut v: usize = 0;
        let mut any = false;
        while let Some(b) = self.next_byte() {
            if b.is_ascii_digit() {
                any = true;
                v = v.saturating_mul(10).saturating_add((b - b'0') as usize);
            } else {
                self.unread_byte(b);
                break;
            }
        }
        any.then_some(v)
    }

    fn skip_line_breaks(&mut self) {
        while let Some(b) = self.next_byte() {
            if b == b'\n' || b == b'\r' {
                continue;
            }
            self.unread_byte(b);
            break;
        }
    }
}

struct BitSet {
    words: Vec<u64>,
}

impl BitSet {
    fn new(nbits: usize) -> Self {
        Self {
            words: vec![0; (nbits + 63) / 64],
        }
    }

    #[inline]
    fn get(&self, i: usize) -> bool {
        let w = i >> 6;
        let b = i & 63;
        ((self.words[w] >> b) & 1) != 0
    }

    #[inline]
    fn set(&mut self, i: usize) {
        let w = i >> 6;
        let b = i & 63;
        self.words[w] |= 1u64 << b;
    }
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct State {
    f: u32,
    g: u32,
    idx: u32,
    x: u16,
    y: u16,
    parent_dir: u8, // direction from this node back to its parent
}

impl Ord for State {
    fn cmp(&self, other: &Self) -> Ordering {
        // Reverse ordering for min-heap behavior on BinaryHeap (which is max-heap).
        other
            .f
            .cmp(&self.f)
            .then_with(|| other.g.cmp(&self.g)) // tie-break: prefer larger g (closer to goal)
            .then_with(|| other.idx.cmp(&self.idx))
    }
}

impl PartialOrd for State {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

#[inline]
fn manhattan(x: u16, y: u16, gx: u16, gy: u16) -> u32 {
    x.abs_diff(gx) as u32 + y.abs_diff(gy) as u32
}

fn main() {
    let stdin = io::stdin();
    let mut input = FastReader::new(stdin.lock());

    let h = match input.next_usize() {
        Some(v) => v,
        None => return,
    };
    let w = match input.next_usize() {
        Some(v) => v,
        None => return,
    };

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            // Dimensions too large to represent.
            let mut out = io::BufWriter::new(io::stdout().lock());
            let _ = writeln!(out, "0");
            return;
        }
    };
    if n == 0 {
        let mut out = io::BufWriter::new(io::stdout().lock());
        let _ = writeln!(out, "0");
        return;
    }

    let w_u16 = w.min(u16::MAX as usize) as u16;
    let h_u16 = h.min(u16::MAX as usize) as u16;

    let mut walls = BitSet::new(n);
    let mut start: Option<(u16, u16)> = None;
    let mut goal: Option<(u16, u16)> = None;

    for y in 0..h {
        input.skip_line_breaks();
        let mut x = 0usize;
        let mut ended_early = false;

        while x < w {
            let b = match input.next_byte() {
                Some(b) => b,
                None => {
                    ended_early = true;
                    break;
                }
            };

            if b == b'\r' {
                continue;
            }
            if b == b'\n' {
                ended_early = true;
                break;
            }

            let idx = y * w + x;
            match b {
                b'#' => walls.set(idx),
                b'A' => start = Some((x as u16, y as u16)),
                b'B' => goal = Some((x as u16, y as u16)),
                _ => {}
            }
            x += 1;
        }

        // If we consumed exactly W cells, discard the rest of the line (if any) up to newline.
        if x == w && !ended_early {
            loop {
                match input.next_byte() {
                    Some(b'\n') => break,
                    Some(b'\r') => continue,
                    Some(_) => continue,
                    None => break,
                }
            }
        }
        // If the row ended early (missing trailing spaces), remaining cells are treated as open.
    }

    let (sx, sy) = match start {
        Some(p) => p,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            let _ = writeln!(out, "0");
            return;
        }
    };
    let (gx, gy) = match goal {
        Some(p) => p,
        None => {
            let mut out = io::BufWriter::new(io::stdout().lock());
            let _ = writeln!(out, "0");
            return;
        }
    };

    if sx >= w_u16 || sy >= h_u16 || gx >= w_u16 || gy >= h_u16 {
        let mut out = io::BufWriter::new(io::stdout().lock());
        let _ = writeln!(out, "0");
        return;
    }

    let start_idx = (sy as usize) * w + (sx as usize);
    let goal_idx = (gy as usize) * w + (gx as usize);

    if walls.get(start_idx) || walls.get(goal_idx) {
        let mut out = io::BufWriter::new(io::stdout().lock());
        let _ = writeln!(out, "0");
        return;
    }

    if start_idx == goal_idx {
        let mut out = io::BufWriter::new(io::stdout().lock());
        let _ = writeln!(out, "1");
        let _ = writeln!(out, "{} {}", sx as usize, sy as usize);
        return;
    }

    let mut closed = BitSet::new(n);
    let mut parent_dir: Vec<u8> = vec![0; n];

    let mut open = BinaryHeap::new();
    open.push(State {
        f: manhattan(sx, sy, gx, gy),
        g: 0,
        idx: start_idx as u32,
        x: sx,
        y: sy,
        parent_dir: 4, // start sentinel
    });

    let mut found_goal = false;

    while let Some(cur) = open.pop() {
        let idx = cur.idx as usize;
        if closed.get(idx) {
            continue;
        }

        closed.set(idx);
        parent_dir[idx] = cur.parent_dir;

        if idx == goal_idx {
            found_goal = true;
            break;
        }

        let x = cur.x as usize;
        let y = cur.y as usize;
        let g_next = cur.g + 1;

        // Up: neighbor's parent is Down (2)
        if y > 0 {
            let nidx = idx - w;
            if !walls.get(nidx) && !closed.get(nidx) {
                open.push(State {
                    f: g_next + manhattan(cur.x, cur.y - 1, gx, gy),
                    g: g_next,
                    idx: nidx as u32,
                    x: cur.x,
                    y: cur.y - 1,
                    parent_dir: 2,
                });
            }
        }
        // Down: neighbor's parent is Up (0)
        if y + 1 < h {
            let nidx = idx + w;
            if !walls.get(nidx) && !closed.get(nidx) {
                open.push(State {
                    f: g_next + manhattan(cur.x, cur.y + 1, gx, gy),
                    g: g_next,
                    idx: nidx as u32,
                    x: cur.x,
                    y: cur.y + 1,
                    parent_dir: 0,
                });
            }
        }
        // Left: neighbor's parent is Right (1)
        if x > 0 {
            let nidx = idx - 1;
            if !walls.get(nidx) && !closed.get(nidx) {
                open.push(State {
                    f: g_next + manhattan(cur.x - 1, cur.y, gx, gy),
                    g: g_next,
                    idx: nidx as u32,
                    x: cur.x - 1,
                    y: cur.y,
                    parent_dir: 1,
                });
            }
        }
        // Right: neighbor's parent is Left (3)
        if x + 1 < w {
            let nidx = idx + 1;
            if !walls.get(nidx) && !closed.get(nidx) {
                open.push(State {
                    f: g_next + manhattan(cur.x + 1, cur.y, gx, gy),
                    g: g_next,
                    idx: nidx as u32,
                    x: cur.x + 1,
                    y: cur.y,
                    parent_dir: 3,
                });
            }
        }
    }

    if !found_goal {
        let mut out = io::BufWriter::new(io::stdout().lock());
        let _ = writeln!(out, "0");
        return;
    }

    // Reconstruct path from goal to start using parent directions.
    let mut path: Vec<(usize, usize)> = Vec::new();
    let mut cx = gx as usize;
    let mut cy = gy as usize;
    let mut cidx = goal_idx;

    loop {
        path.push((cx, cy));
        if cidx == start_idx {
            break;
        }
        let dir = parent_dir[cidx];
        match dir {
            0 => {
                // Up
                cy -= 1;
                cidx -= w;
            }
            1 => {
                // Right
                cx += 1;
                cidx += 1;
            }
            2 => {
                // Down
                cy += 1;
                cidx += w;
            }
            3 => {
                // Left
                cx -= 1;
                cidx -= 1;
            }
            _ => {
                // Should not happen for a valid found path.
                let mut out = io::BufWriter::new(io::stdout().lock());
                let _ = writeln!(out, "0");
                return;
            }
        }
    }

    path.reverse();

    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());
    let _ = writeln!(out, "{}", path.len());
    for (x, y) in path {
        let _ = writeln!(out, "{} {}", x, y);
    }
    let _ = out.flush();
}
