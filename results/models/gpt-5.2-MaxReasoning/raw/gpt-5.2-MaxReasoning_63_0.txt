declare const require: any;
const fs = require("fs");

const input: string = fs.readFileSync(0, "utf8");
const data: any = input && input.trim().length ? JSON.parse(input) : { grid: [], robots: [], jobs: [] };

type Robot = { id: number; x: number; y: number };
type JobId = any;
type Job = { id: JobId; x: number; y: number; priority: number };

function toInt(v: any): number {
  const n = typeof v === "number" ? v : Number(v);
  return Number.isFinite(n) ? Math.trunc(n) : 0;
}

function part1By1(n: number): number {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n;
}

function morton2D(x: number, y: number): number {
  return (part1By1(y) << 1) | part1By1(x);
}

function compareId(a: any, b: any): number {
  if (a === b) return 0;
  const ta = typeof a;
  const tb = typeof b;
  if (ta === "number" && tb === "number") return a - b;
  const sa = String(a);
  const sb = String(b);
  return sa < sb ? -1 : sa > sb ? 1 : 0;
}

const grid: string[] = Array.isArray(data.grid) ? data.grid.map((s: any) => String(s)) : [];
const robotsRaw: any[] = Array.isArray(data.robots) ? data.robots : [];
const jobsRaw: any[] = Array.isArray(data.jobs) ? data.jobs : [];

const robots: Robot[] = [];
let minRobotId = Number.POSITIVE_INFINITY;
let maxRobotId = -1;
for (let i = 0; i < robotsRaw.length; i++) {
  const r = robotsRaw[i];
  if (!r) continue;
  const id = toInt(r.id);
  if (id < 0) continue;
  const x = toInt(r.x);
  const y = toInt(r.y);
  robots.push({ id, x, y });
  if (id < minRobotId) minRobotId = id;
  if (id > maxRobotId) maxRobotId = id;
}

if (robots.length === 0) {
  process.stdout.write(JSON.stringify({ assignments: [] }));
  process.exit(0);
}

const jobs: Job[] = [];
for (let i = 0; i < jobsRaw.length; i++) {
  const j = jobsRaw[i];
  if (!j) continue;
  jobs.push({ id: j.id, x: toInt(j.x), y: toInt(j.y), priority: toInt(j.priority) });
}

const assignments: JobId[][] = new Array(maxRobotId + 1);
for (let i = 0; i < assignments.length; i++) assignments[i] = [];

const h = grid.length;
const w = h > 0 ? grid[0].length : 0;
const nCells = w * h;

if (w <= 0 || h <= 0 || nCells <= 0) {
  for (let i = 0; i < jobs.length; i++) assignments[minRobotId].push(jobs[i].id);
  process.stdout.write(JSON.stringify({ assignments }));
  process.exit(0);
}

const passable = new Uint8Array(nCells);
for (let y = 0; y < h; y++) {
  const row = grid[y];
  for (let x = 0; x < w; x++) {
    passable[y * w + x] = row.charCodeAt(x) === 35 ? 0 : 1; // '#'
  }
}

const dist = new Int32Array(nCells);
dist.fill(-1);
const owner = new Int32Array(nCells);
owner.fill(-1);

const robotsSorted = robots.slice().sort((a, b) => a.id - b.id);
const q = new Int32Array(nCells);
let head = 0;
let tail = 0;

for (let i = 0; i < robotsSorted.length; i++) {
  const r = robotsSorted[i];
  if (r.x < 0 || r.x >= w || r.y < 0 || r.y >= h) continue;
  const idx = r.y * w + r.x;
  if (passable[idx] === 0) continue;
  if (dist[idx] === -1) {
    dist[idx] = 0;
    owner[idx] = r.id;
    q[tail++] = idx;
  } else if (dist[idx] === 0 && owner[idx] > r.id) {
    owner[idx] = r.id;
  }
}

while (head < tail) {
  const cur = q[head++];
  const d = dist[cur];
  const o = owner[cur];

  const col = cur % w;
  const nextD = d + 1;

  // left
  if (col !== 0) {
    const nb = cur - 1;
    if (passable[nb] === 1 && dist[nb] === -1) {
      dist[nb] = nextD;
      owner[nb] = o;
      q[tail++] = nb;
    }
  }
  // right
  if (col + 1 < w) {
    const nb = cur + 1;
    if (passable[nb] === 1 && dist[nb] === -1) {
      dist[nb] = nextD;
      owner[nb] = o;
      q[tail++] = nb;
    }
  }
  // up
  if (cur >= w) {
    const nb = cur - w;
    if (passable[nb] === 1 && dist[nb] === -1) {
      dist[nb] = nextD;
      owner[nb] = o;
      q[tail++] = nb;
    }
  }
  // down
  const down = cur + w;
  if (down < nCells) {
    const nb = down;
    if (passable[nb] === 1 && dist[nb] === -1) {
      dist[nb] = nextD;
      owner[nb] = o;
      q[tail++] = nb;
    }
  }
}

type JobEntry = { id: JobId; x: number; y: number; priority: number; d0: number; z: number };
const perRobot: JobEntry[][] = new Array(maxRobotId + 1);
for (let i = 0; i < perRobot.length; i++) perRobot[i] = [];

for (let i = 0; i < jobs.length; i++) {
  const j = jobs[i];
  let rid = minRobotId;
  let d0 = 2147483647;
  if (j.x >= 0 && j.x < w && j.y >= 0 && j.y < h) {
    const idx = j.y * w + j.x;
    const o = owner[idx];
    if (o >= 0 && o <= maxRobotId) {
      rid = o;
      const dj = dist[idx];
      if (dj >= 0) d0 = dj;
    }
  }
  perRobot[rid].push({ id: j.id, x: j.x, y: j.y, priority: j.priority, d0, z: morton2D(j.x, j.y) });
}

for (let rid = 0; rid <= maxRobotId; rid++) {
  const list = perRobot[rid];
  if (list.length === 0) continue;
  list.sort((a, b) => {
    if (a.priority !== b.priority) return b.priority - a.priority;
    if (a.d0 !== b.d0) return a.d0 - b.d0;
    if (a.z !== b.z) return a.z - b.z;
    return compareId(a.id, b.id);
  });
  const out = assignments[rid];
  for (let i = 0; i < list.length; i++) out.push(list[i].id);
}

process.stdout.write(JSON.stringify({ assignments }));
