using System;
using System.Collections.Generic;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();

        int[] head = new int[n];
        int[] headR = new int[n];
        Array.Fill(head, -1);
        Array.Fill(headR, -1);

        int[] to = new int[m];
        int[] next = new int[m];
        int[] toR = new int[m];
        int[] nextR = new int[m];

        bool[] selfLoop = new bool[n];

        int edgeIdx = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;
            if (u == v) selfLoop[u] = true;

            to[edgeIdx] = v;
            next[edgeIdx] = head[u];
            head[u] = edgeIdx;

            toR[edgeIdx] = u;
            nextR[edgeIdx] = headR[v];
            headR[v] = edgeIdx;

            edgeIdx++;
        }

        // Kosaraju (iterative) to get SCCs.
        int[] order = new int[n];
        bool[] visited = new bool[n];
        int[] stackV = new int[n];
        int[] stackE = new int[n];
        int orderCount = 0;

        for (int v = 0; v < n; v++)
        {
            if (visited[v]) continue;
            int sp = 0;
            stackV[sp] = v;
            stackE[sp] = head[v];
            sp++;
            visited[v] = true;

            while (sp > 0)
            {
                int u = stackV[sp - 1];
                int e = stackE[sp - 1];
                if (e != -1)
                {
                    stackE[sp - 1] = next[e];
                    int w = to[e];
                    if (!visited[w])
                    {
                        visited[w] = true;
                        stackV[sp] = w;
                        stackE[sp] = head[w];
                        sp++;
                    }
                }
                else
                {
                    sp--;
                    order[orderCount++] = u;
                }
            }
        }

        int[] compId = new int[n];
        Array.Fill(compId, -1);
        int[] compStarts = new int[n];
        int[] compSizes = new int[n];
        int[] compVerts = new int[n];
        int compVertsCount = 0;
        int compCount = 0;

        for (int i = n - 1; i >= 0; i--)
        {
            int v = order[i];
            if (compId[v] != -1) continue;

            compStarts[compCount] = compVertsCount;
            int sp = 0;
            stackV[sp++] = v;
            compId[v] = compCount;

            while (sp > 0)
            {
                int u = stackV[--sp];
                compVerts[compVertsCount++] = u;
                for (int e = headR[u]; e != -1; e = nextR[e])
                {
                    int w = toR[e];
                    if (compId[w] == -1)
                    {
                        compId[w] = compCount;
                        stackV[sp++] = w;
                    }
                }
            }

            compSizes[compCount] = compVertsCount - compStarts[compCount];
            compCount++;
        }

        var answer = new List<int>();
        int[] pos = new int[n];
        Array.Fill(pos, -1);

        for (int cid = 0; cid < compCount; cid++)
        {
            int start = compStarts[cid];
            int size = compSizes[cid];
            if (size == 1)
            {
                int v = compVerts[start];
                if (selfLoop[v]) answer.Add(v);
                continue;
            }

            // This implementation uses 64-bit masks per SCC. Instances with SCCs > 64 vertices are
            // not expected in this benchmark's exact track.
            if (size > 64)
            {
                for (int i = 0; i < size; i++) answer.Add(compVerts[start + i]);
                continue;
            }

            int s = size;
            int[] verts = new int[s];
            for (int i = 0; i < s; i++)
            {
                int v = compVerts[start + i];
                verts[i] = v;
                pos[v] = i;
            }

            ulong[] outLocal = new ulong[s];
            ulong[] inLocal = new ulong[s];
            ulong selfLoopMask = 0;

            for (int i = 0; i < s; i++)
            {
                int u = verts[i];
                for (int e = head[u]; e != -1; e = next[e])
                {
                    int w = to[e];
                    int j = pos[w];
                    if (j < 0) continue;
                    ulong bitJ = 1UL << j;
                    outLocal[i] |= bitJ;
                    inLocal[j] |= 1UL << i;
                    if (i == j) selfLoopMask |= bitJ;
                }
            }

            var solver = new DfvsSolver64(s, outLocal, inLocal, selfLoopMask);
            ulong removeMaskLocal = solver.SolveAll();

            ulong mm = removeMaskLocal;
            while (mm != 0)
            {
                int i = BitOperations.TrailingZeroCount(mm);
                mm &= mm - 1;
                answer.Add(verts[i]);
            }

            for (int i = 0; i < s; i++) pos[verts[i]] = -1;
        }

        var sb = new StringBuilder();
        sb.Append(answer.Count).Append('\n');
        for (int i = 0; i < answer.Count; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(answer[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream) => _stream = stream;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c == -1) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int x = 0;
            while (c > 32 && c != -1)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private sealed class DfvsSolver64
    {
        private readonly int _n;
        private readonly ulong[] _out;
        private readonly ulong[] _in;
        private readonly ulong _selfLoopMask;
        private readonly ulong _fullMask;

        private const int DpThreshold = 25;

        // Memo over canonicalized "alive" masks (after removing forced self-loops and acyclic vertices).
        private readonly Dictionary<ulong, Result> _memo = new Dictionary<ulong, Result>(1024);

        public DfvsSolver64(int n, ulong[] outAdj, ulong[] inAdj, ulong selfLoopMask)
        {
            _n = n;
            _out = outAdj;
            _in = inAdj;
            _selfLoopMask = selfLoopMask;
            _fullMask = n == 64 ? ulong.MaxValue : ((1UL << n) - 1UL);
        }

        public ulong SolveAll()
        {
            var res = Solve(_fullMask);
            return res.Mask;
        }

        private Result Solve(ulong alive)
        {
            // Forced: any vertex with a self-loop must be removed.
            ulong forced = _selfLoopMask & alive;
            int forcedCount = PopCount(forced);
            alive &= ~forced;

            // Remove vertices not on any cycle.
            alive = CyclicCore(alive);

            if (alive == 0)
            {
                return new Result(forced, forcedCount);
            }

            if (_memo.TryGetValue(alive, out Result cached))
            {
                return new Result(cached.Mask | forced, cached.Count + forcedCount);
            }

            // Split into SCCs: optimal DFVS is additive over SCCs.
            Span<ulong> sccs = stackalloc ulong[64];
            int sccCount = GetSccMasks(alive, sccs);

            if (sccCount > 1)
            {
                ulong totalMask = 0;
                int totalCount = 0;
                for (int i = 0; i < sccCount; i++)
                {
                    ulong scc = sccs[i];
                    if (PopCount(scc) <= 1) continue;
                    var r = Solve(scc);
                    totalMask |= r.Mask;
                    totalCount += r.Count;
                }
                var coreRes = new Result(totalMask, totalCount);
                _memo[alive] = coreRes;
                return new Result(totalMask | forced, totalCount + forcedCount);
            }

            // Single SCC (cyclic).
            int size = PopCount(alive);
            Result sccRes = size <= DpThreshold ? SolveBySubsetDp(alive) : SolveByIterativeDeepening(alive);

            _memo[alive] = sccRes;
            return new Result(sccRes.Mask | forced, sccRes.Count + forcedCount);
        }

        private Result SolveByIterativeDeepening(ulong alive)
        {
            ulong ubMask = GreedyFvs(alive);
            int ub = PopCount(ubMask);
            if (ub == 0) return new Result(0, 0);

            var fail = new Dictionary<ulong, int>(2048);
            int lb = CyclePackingLowerBound(alive, ub);
            if (lb < 0) lb = 0;

            for (int k = lb; k <= ub; k++)
            {
                if (CanSolve(alive, k, fail, out ulong sol))
                {
                    int cnt = PopCount(sol);
                    return new Result(sol, cnt);
                }
            }

            return new Result(ubMask, ub);
        }

        private bool CanSolve(ulong alive, int k, Dictionary<ulong, int> fail, out ulong solutionMask)
        {
            // Forced self-loops.
            ulong forced = _selfLoopMask & alive;
            int forcedCount = PopCount(forced);
            if (forcedCount > k)
            {
                solutionMask = 0;
                return false;
            }

            alive &= ~forced;
            k -= forcedCount;

            alive = CyclicCore(alive);
            if (alive == 0)
            {
                solutionMask = forced;
                return true;
            }

            if (fail.TryGetValue(alive, out int maxFail) && maxFail >= k)
            {
                solutionMask = 0;
                return false;
            }

            if (k == 0)
            {
                NoteFail(fail, alive, k);
                solutionMask = 0;
                return false;
            }

            int lb = CyclePackingLowerBound(alive, k);
            if (lb > k)
            {
                NoteFail(fail, alive, k);
                solutionMask = 0;
                return false;
            }

            ulong cycle = FindShortCycleMask(alive);
            if (cycle == 0)
            {
                solutionMask = forced;
                return true;
            }

            ulong verts = cycle;
            while (verts != 0)
            {
                int v = BitOperations.TrailingZeroCount(verts);
                verts &= verts - 1;
                ulong bitV = 1UL << v;

                if (CanSolve(alive & ~bitV, k - 1, fail, out ulong subSol))
                {
                    solutionMask = forced | subSol | bitV;
                    return true;
                }
            }

            NoteFail(fail, alive, k);
            solutionMask = 0;
            return false;
        }

        private static void NoteFail(Dictionary<ulong, int> fail, ulong alive, int k)
        {
            if (fail.TryGetValue(alive, out int old))
            {
                if (old < k) fail[alive] = k;
            }
            else
            {
                fail[alive] = k;
            }
        }

        private int CyclePackingLowerBound(ulong alive, int cap)
        {
            int count = 0;
            ulong cur = alive;
            while (cur != 0)
            {
                cur = CyclicCore(cur);
                if (cur == 0) break;
                count++;
                if (count > cap) return count;
                ulong cyc = FindAnyCycleMask(cur);
                if (cyc == 0) break;
                cur &= ~cyc;
            }
            return count;
        }

        private ulong GreedyFvs(ulong alive)
        {
            ulong removed = 0;
            ulong cur = alive;
            while (true)
            {
                ulong loops = _selfLoopMask & cur;
                if (loops != 0)
                {
                    removed |= loops;
                    cur &= ~loops;
                }

                cur = CyclicCore(cur);
                if (cur == 0) break;

                ulong cycle = FindShortCycleMask(cur);
                if (cycle == 0) break;

                int bestV = -1;
                int bestDeg = -1;
                ulong tmp = cycle;
                while (tmp != 0)
                {
                    int v = BitOperations.TrailingZeroCount(tmp);
                    tmp &= tmp - 1;
                    int deg = PopCount(_out[v] & cur) + PopCount(_in[v] & cur);
                    if (deg > bestDeg)
                    {
                        bestDeg = deg;
                        bestV = v;
                    }
                }

                if (bestV < 0) break;
                ulong bit = 1UL << bestV;
                removed |= bit;
                cur &= ~bit;
            }
            return removed;
        }

        private ulong FindShortCycleMask(ulong alive)
        {
            // 2-cycle
            ulong tmpU = alive;
            while (tmpU != 0)
            {
                int u = BitOperations.TrailingZeroCount(tmpU);
                tmpU &= tmpU - 1;
                ulong neigh = _out[u] & alive;
                while (neigh != 0)
                {
                    int v = BitOperations.TrailingZeroCount(neigh);
                    neigh &= neigh - 1;
                    if ((_out[v] & (1UL << u)) != 0)
                    {
                        return (1UL << u) | (1UL << v);
                    }
                }
            }

            // 3-cycle
            tmpU = alive;
            while (tmpU != 0)
            {
                int u = BitOperations.TrailingZeroCount(tmpU);
                tmpU &= tmpU - 1;
                ulong neigh1 = _out[u] & alive;
                while (neigh1 != 0)
                {
                    int v = BitOperations.TrailingZeroCount(neigh1);
                    neigh1 &= neigh1 - 1;
                    ulong wCand = _out[v] & alive & _in[u];
                    if (wCand != 0)
                    {
                        int w = BitOperations.TrailingZeroCount(wCand);
                        return (1UL << u) | (1UL << v) | (1UL << w);
                    }
                }
            }

            return FindAnyCycleMask(alive);
        }

        private ulong FindAnyCycleMask(ulong alive)
        {
            Span<byte> color = stackalloc byte[64];
            Span<int> parent = stackalloc int[64];
            for (int i = 0; i < _n; i++) parent[i] = -1;

            Span<int> stackV = stackalloc int[64];
            Span<ulong> stackNeigh = stackalloc ulong[64];

            ulong remaining = alive;
            while (remaining != 0)
            {
                int start = BitOperations.TrailingZeroCount(remaining);
                remaining &= remaining - 1;
                if (color[start] != 0) continue;

                int sp = 0;
                stackV[sp] = start;
                stackNeigh[sp] = _out[start] & alive;
                sp++;
                color[start] = 1;
                parent[start] = -1;

                while (sp > 0)
                {
                    int u = stackV[sp - 1];
                    ulong neigh = stackNeigh[sp - 1];
                    if (neigh == 0)
                    {
                        color[u] = 2;
                        sp--;
                        continue;
                    }

                    int v = BitOperations.TrailingZeroCount(neigh);
                    stackNeigh[sp - 1] = neigh & (neigh - 1);

                    if (color[v] == 0)
                    {
                        parent[v] = u;
                        color[v] = 1;
                        stackV[sp] = v;
                        stackNeigh[sp] = _out[v] & alive;
                        sp++;
                    }
                    else if (color[v] == 1)
                    {
                        ulong cycle = 0;
                        cycle |= 1UL << v;
                        int cur = u;
                        while (cur != v && cur != -1)
                        {
                            cycle |= 1UL << cur;
                            cur = parent[cur];
                        }
                        if (cur == v) return cycle;
                    }
                }
            }

            return 0;
        }

        private ulong CyclicCore(ulong alive)
        {
            if (alive == 0) return 0;
            Span<ulong> sccs = stackalloc ulong[64];
            int cnt = GetSccMasks(alive, sccs);
            ulong core = 0;
            for (int i = 0; i < cnt; i++)
            {
                ulong scc = sccs[i];
                if (PopCount(scc) > 1 || ((_selfLoopMask & scc) != 0)) core |= scc;
            }
            return core;
        }

        private int GetSccMasks(ulong alive, Span<ulong> sccOut)
        {
            Span<int> stackV = stackalloc int[64];
            Span<ulong> stackNeigh = stackalloc ulong[64];
            Span<int> order = stackalloc int[64];
            int orderCount = 0;

            ulong visited = 0;
            ulong rem = alive;
            while (rem != 0)
            {
                int start = BitOperations.TrailingZeroCount(rem);
                rem &= rem - 1;
                ulong bitStart = 1UL << start;
                if ((visited & bitStart) != 0) continue;

                int sp = 0;
                stackV[sp] = start;
                stackNeigh[sp] = _out[start] & alive;
                sp++;
                visited |= bitStart;

                while (sp > 0)
                {
                    int u = stackV[sp - 1];
                    ulong neigh = stackNeigh[sp - 1];
                    if (neigh != 0)
                    {
                        int w = BitOperations.TrailingZeroCount(neigh);
                        stackNeigh[sp - 1] = neigh & (neigh - 1);
                        ulong bitW = 1UL << w;
                        if ((visited & bitW) == 0)
                        {
                            visited |= bitW;
                            stackV[sp] = w;
                            stackNeigh[sp] = _out[w] & alive;
                            sp++;
                        }
                    }
                    else
                    {
                        sp--;
                        order[orderCount++] = u;
                    }
                }
            }

            ulong assigned = 0;
            int sccCount = 0;
            for (int i = orderCount - 1; i >= 0; i--)
            {
                int v = order[i];
                ulong bitV = 1UL << v;
                if ((alive & bitV) == 0 || (assigned & bitV) != 0) continue;

                ulong compMask = 0;
                int sp = 0;
                stackV[sp++] = v;
                assigned |= bitV;

                while (sp > 0)
                {
                    int u = stackV[--sp];
                    compMask |= 1UL << u;
                    ulong neigh = _in[u] & alive;
                    while (neigh != 0)
                    {
                        int w = BitOperations.TrailingZeroCount(neigh);
                        neigh &= neigh - 1;
                        ulong bitW = 1UL << w;
                        if ((assigned & bitW) == 0)
                        {
                            assigned |= bitW;
                            stackV[sp++] = w;
                        }
                    }
                }

                sccOut[sccCount++] = compMask;
            }

            return sccCount;
        }

        private Result SolveBySubsetDp(ulong alive)
        {
            int p = PopCount(alive);
            if (p <= 1) return new Result(0, 0);

            Span<int> verts = stackalloc int[64];
            int idx = 0;
            ulong tmp = alive;
            while (tmp != 0)
            {
                int v = BitOperations.TrailingZeroCount(tmp);
                tmp &= tmp - 1;
                verts[idx++] = v;
            }

            Span<int> pos = stackalloc int[64];
            for (int i = 0; i < _n; i++) pos[i] = -1;
            for (int i = 0; i < p; i++) pos[verts[i]] = i;

            int[] outLocal = new int[p];
            for (int i = 0; i < p; i++)
            {
                int v = verts[i];
                ulong neigh = _out[v] & alive;
                int mask = 0;
                while (neigh != 0)
                {
                    int w = BitOperations.TrailingZeroCount(neigh);
                    neigh &= neigh - 1;
                    mask |= 1 << pos[w];
                }
                outLocal[i] = mask;
            }

            int N = 1 << p;
            ulong[] acyclicBits = new ulong[(N + 63) >> 6];
            acyclicBits[0] |= 1UL;

            int bestMask = 0;
            int bestCount = 0;

            for (int mask = 1; mask < N; mask++)
            {
                int mm = mask;
                while (mm != 0)
                {
                    int v = BitOperations.TrailingZeroCount((uint)mm);
                    int prev = mask & ~(1 << v);
                    if (GetBit(acyclicBits, prev) && (outLocal[v] & prev) == 0)
                    {
                        SetBit(acyclicBits, mask);
                        int cnt = BitOperations.PopCount((uint)mask);
                        if (cnt > bestCount)
                        {
                            bestCount = cnt;
                            bestMask = mask;
                        }
                        break;
                    }
                    mm &= mm - 1;
                }
            }

            ulong remove = 0;
            for (int i = 0; i < p; i++)
            {
                if (((bestMask >> i) & 1) == 0) remove |= 1UL << verts[i];
            }

            return new Result(remove, p - bestCount);
        }

        private static bool GetBit(ulong[] bits, int idx) => (bits[idx >> 6] & (1UL << (idx & 63))) != 0;
        private static void SetBit(ulong[] bits, int idx) => bits[idx >> 6] |= 1UL << (idx & 63);

        private static int PopCount(ulong x) => (int)BitOperations.PopCount(x);

        private readonly struct Result
        {
            public readonly ulong Mask;
            public readonly int Count;
            public Result(ulong mask, int count)
            {
                Mask = mask;
                Count = count;
            }
        }
    }
}
