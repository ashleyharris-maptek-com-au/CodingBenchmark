using System;
using System.Collections.Generic;
using System.IO;

public static class Program
{
    private const byte Unknown = (byte)' ';
    private const byte Mine = (byte)'*';
    private const byte SafeMark = (byte)'s'; // internal marker: proven safe but unrevealed

    public static void Main()
    {
        var input = new FastInput(Console.OpenStandardInput());
        int height = input.ReadInt();
        int width = input.ReadInt();
        input.SkipToLineEndIfNeeded();

        if (height <= 0 || width <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        int cellCount = checked(height * width);
        var grid = new byte[cellCount];

        const int StoreUnknownLimit = 200_000;
        var unknownPositions = new List<int>(Math.Min(StoreUnknownLimit, cellCount));
        int unknownTotal = 0;

        for (int i = 0; i < cellCount; i++)
        {
            byte b = input.ReadBoardByte();
            grid[i] = b;
            if (b == Unknown)
            {
                unknownTotal++;
                if (unknownPositions.Count < StoreUnknownLimit)
                    unknownPositions.Add(i);
            }
        }

        if (unknownTotal == 0)
        {
            Console.WriteLine("0");
            return;
        }

        var safeCells = new List<int>(Math.Min(unknownTotal, 1_000_000));
        var queue = new IntQueue(Math.Min(1024, cellCount));
        var activeDigits = new List<int>();

        // Adaptive seeding:
        // - If unknown cells are few, seed from unknowns (cheaper than scanning all digits).
        // - Otherwise, scan digits and enqueue those with at least one unknown neighbor.
        if (unknownTotal <= StoreUnknownLimit)
        {
            var digitSet = new HashSet<int>();
            foreach (int uIdx in unknownPositions)
            {
                int uy = uIdx / width;
                int ux = uIdx - uy * width;
                for (int dy = -1; dy <= 1; dy++)
                {
                    int ny = uy + dy;
                    if ((uint)ny >= (uint)height) continue;
                    int rowBase = ny * width;
                    for (int dx = -1; dx <= 1; dx++)
                    {
                        if (dx == 0 && dy == 0) continue;
                        int nx = ux + dx;
                        if ((uint)nx >= (uint)width) continue;
                        int nIdx = rowBase + nx;
                        byte nb = grid[nIdx];
                        if (IsDigit(nb))
                            digitSet.Add(nIdx);
                    }
                }
            }

            foreach (int d in digitSet)
            {
                activeDigits.Add(d);
                queue.Enqueue(d);
            }
        }
        else
        {
            for (int idx = 0; idx < cellCount; idx++)
            {
                byte b = grid[idx];
                if (!IsDigit(b)) continue;
                if (HasUnknownNeighbor(grid, height, width, idx))
                {
                    activeDigits.Add(idx);
                    queue.Enqueue(idx);
                }
            }
        }

        // Fast deterministic propagation (rules: remaining=0 => all safe; remaining=U => all mines)
        PropagateBasic(grid, height, width, queue, safeCells);

        // Adaptive deeper reasoning on the remaining frontier.
        // Only attempt when the number of relevant digit constraints is manageable.
        const int AdvancedMaxDigits = 50_000;
        const int AdvancedMaxFrontierVars = 50_000;

        if (activeDigits.Count > 0 && activeDigits.Count <= AdvancedMaxDigits)
        {
            if (TryAdvancedSolve(grid, height, width, activeDigits, queue, safeCells, AdvancedMaxFrontierVars))
            {
                // Re-run basic propagation because advanced solving may have fixed extra mines/safes.
                PropagateBasic(grid, height, width, queue, safeCells);
            }
        }

        var output = new StreamWriter(Console.OpenStandardOutput()) { NewLine = "\n" };
        output.WriteLine(safeCells.Count);
        for (int i = 0; i < safeCells.Count; i++)
        {
            int idx = safeCells[i];
            int y = idx / width;
            int x = idx - y * width;
            output.Write(x);
            output.Write(' ');
            output.WriteLine(y);
        }
        output.Flush();
    }

    private static bool TryAdvancedSolve(
        byte[] grid,
        int height,
        int width,
        List<int> activeDigits,
        IntQueue queue,
        List<int> safeCells,
        int advancedMaxFrontierVars)
    {
        var varIdByCell = new Dictionary<int, int>(capacity: Math.Min(4096, activeDigits.Count * 2));
        var varCellIdx = new List<int>(capacity: Math.Min(4096, activeDigits.Count * 2));
        var constraints = new List<Constraint>(capacity: activeDigits.Count);

        // Build frontier constraints (each digit cell gives: sum(vars) == minesNeeded).
        for (int i = 0; i < activeDigits.Count; i++)
        {
            int dIdx = activeDigits[i];
            byte digitByte = grid[dIdx];
            if (!IsDigit(digitByte)) continue;

            int minesNeeded;
            Span<int> tmpVars = stackalloc int[8];
            int tmpCount = CollectUnknownNeighborVars(grid, height, width, dIdx, varIdByCell, varCellIdx, tmpVars, out minesNeeded);

            if (tmpCount <= 0) continue;
            if ((uint)minesNeeded > (uint)tmpCount) continue;

            int[] vars = new int[tmpCount];
            for (int k = 0; k < tmpCount; k++) vars[k] = tmpVars[k];
            Array.Sort(vars);
            constraints.Add(new Constraint(minesNeeded, vars));

            if (varCellIdx.Count > advancedMaxFrontierVars)
                return false;
        }

        int varCount = varCellIdx.Count;
        int consCount = constraints.Count;
        if (varCount == 0 || consCount == 0) return false;

        // Build variable -> constraints adjacency (CSR) for connected components.
        int totalEdges = 0;
        var edgeCounts = new int[varCount];
        for (int c = 0; c < consCount; c++)
        {
            int[] vars = constraints[c].Vars;
            totalEdges += vars.Length;
            for (int j = 0; j < vars.Length; j++) edgeCounts[vars[j]]++;
        }

        var offsets = new int[varCount + 1];
        for (int v = 0; v < varCount; v++) offsets[v + 1] = offsets[v] + edgeCounts[v];
        var edges = new int[totalEdges];
        var cursor = (int[])offsets.Clone();
        for (int c = 0; c < consCount; c++)
        {
            int[] vars = constraints[c].Vars;
            for (int j = 0; j < vars.Length; j++)
            {
                int v = vars[j];
                edges[cursor[v]++] = c;
            }
        }

        bool anyChange = false;
        var varVisited = new bool[varCount];
        var consVisited = new bool[consCount];

        const int ExactMaxVars = 22;
        const int ExactMaxConstraints = 64;
        const int SubsetMaxVars = 180;
        const int SubsetMaxConstraints = 300;

        // Global var state used by subset inference (only for components we attempt).
        // -1 unknown, 0 safe, 1 mine
        int[] varState = Array.Empty<int>();

        for (int startVar = 0; startVar < varCount; startVar++)
        {
            if (varVisited[startVar]) continue;

            var compVars = new List<int>();
            var compCons = new List<int>();

            varVisited[startVar] = true;
            compVars.Add(startVar);
            for (int qi = 0; qi < compVars.Count; qi++)
            {
                int v = compVars[qi];
                for (int ei = offsets[v]; ei < offsets[v + 1]; ei++)
                {
                    int c = edges[ei];
                    if (consVisited[c]) continue;
                    consVisited[c] = true;
                    compCons.Add(c);

                    int[] vars = constraints[c].Vars;
                    for (int j = 0; j < vars.Length; j++)
                    {
                        int v2 = vars[j];
                        if (varVisited[v2]) continue;
                        varVisited[v2] = true;
                        compVars.Add(v2);
                    }
                }
            }

            if (compVars.Count <= ExactMaxVars && compCons.Count <= ExactMaxConstraints)
            {
                anyChange |= SolveComponentExactly(grid, height, width, compVars, compCons, constraints, varCellIdx, queue, safeCells);
            }
            else if (compVars.Count <= SubsetMaxVars && compCons.Count <= SubsetMaxConstraints)
            {
                if (varState.Length == 0)
                {
                    varState = new int[varCount];
                    Array.Fill(varState, -1);
                }

                anyChange |= SolveComponentBySubsetInference(grid, height, width, compVars, compCons, constraints, varCellIdx, varState, queue, safeCells);
            }
        }

        return anyChange;
    }

    private static bool SolveComponentExactly(
        byte[] grid,
        int height,
        int width,
        List<int> compVarsGlobal,
        List<int> compConsGlobal,
        List<Constraint> constraints,
        List<int> varCellIdx,
        IntQueue queue,
        List<int> safeCells)
    {
        int nVars = compVarsGlobal.Count;
        int nCons = compConsGlobal.Count;

        var globalToLocal = new Dictionary<int, int>(nVars);
        for (int i = 0; i < nVars; i++) globalToLocal[compVarsGlobal[i]] = i;

        var localCons = new LocalConstraint[nCons];
        var localVarToCons = new List<int>[nVars];
        for (int i = 0; i < nVars; i++) localVarToCons[i] = new List<int>();

        for (int ci = 0; ci < nCons; ci++)
        {
            Constraint gc = constraints[compConsGlobal[ci]];
            int[] gVars = gc.Vars;
            var lVars = new int[gVars.Length];
            for (int j = 0; j < gVars.Length; j++) lVars[j] = globalToLocal[gVars[j]];
            localCons[ci] = new LocalConstraint(gc.Mines, lVars);
            for (int j = 0; j < lVars.Length; j++) localVarToCons[lVars[j]].Add(ci);
        }

        int[] order = new int[nVars];
        int[] degree = new int[nVars];
        for (int i = 0; i < nVars; i++)
        {
            order[i] = i;
            degree[i] = localVarToCons[i].Count;
        }
        Array.Sort(order, (a, b) => degree[b].CompareTo(degree[a]));

        int[] assign = new int[nVars];
        Array.Fill(assign, -1);
        int[] assignedMines = new int[nCons];
        int[] unassigned = new int[nCons];
        for (int i = 0; i < nCons; i++) unassigned[i] = localCons[i].Vars.Length;

        bool[] canBeSafe = new bool[nVars];
        bool[] canBeMine = new bool[nVars];
        int solutions = 0;

        void Dfs2(int depth)
        {
            if (depth == nVars)
            {
                solutions++;
                for (int i = 0; i < nVars; i++)
                {
                    if (assign[i] == 0) canBeSafe[i] = true;
                    else canBeMine[i] = true;
                }
                return;
            }

            int v = order[depth];
            for (int val = 0; val <= 1; val++)
            {
                if (!Apply(v, val)) continue;
                Dfs2(depth + 1);
                Revert(v, val);
            }
        }

        bool Apply(int v, int val)
        {
            assign[v] = val;
            List<int> consList = localVarToCons[v];
            for (int i = 0; i < consList.Count; i++)
            {
                int c = consList[i];
                unassigned[c]--;
                if (val == 1) assignedMines[c]++;

                int need = localCons[c].Mines;
                if (assignedMines[c] > need || assignedMines[c] + unassigned[c] < need)
                {
                    // partial revert
                    for (int j = 0; j <= i; j++)
                    {
                        int c2 = consList[j];
                        if (val == 1) assignedMines[c2]--;
                        unassigned[c2]++;
                    }
                    assign[v] = -1;
                    return false;
                }
            }
            return true;
        }

        void Revert(int v, int val)
        {
            List<int> consList = localVarToCons[v];
            for (int i = 0; i < consList.Count; i++)
            {
                int c = consList[i];
                if (val == 1) assignedMines[c]--;
                unassigned[c]++;
            }
            assign[v] = -1;
        }

        Dfs2(0);

        if (solutions == 0) return false;

        bool changed = false;
        for (int li = 0; li < nVars; li++)
        {
            if (canBeSafe[li] && !canBeMine[li])
            {
                int gVar = compVarsGlobal[li];
                changed |= SetSafe(grid, height, width, varCellIdx[gVar], queue, safeCells);
            }
            else if (canBeMine[li] && !canBeSafe[li])
            {
                int gVar = compVarsGlobal[li];
                changed |= SetMine(grid, height, width, varCellIdx[gVar], queue);
            }
        }

        return changed;
    }

    private static bool SolveComponentBySubsetInference(
        byte[] grid,
        int height,
        int width,
        List<int> compVars,
        List<int> compCons,
        List<Constraint> constraints,
        List<int> varCellIdx,
        int[] varState,
        IntQueue queue,
        List<int> safeCells)
    {
        bool changed = false;

        // Iterate a few rounds: simplify constraints with current assignments, apply direct rules, then subset rules.
        for (int iter = 0; iter < 6; iter++)
        {
            bool anyThisIter = false;
            var reduced = new List<Constraint>(compCons.Count);

            for (int i = 0; i < compCons.Count; i++)
            {
                Constraint c = constraints[compCons[i]];
                int mines = c.Mines;

                Span<int> tmp = stackalloc int[8];
                int tCount = 0;
                int[] vars = c.Vars;
                for (int j = 0; j < vars.Length; j++)
                {
                    int v = vars[j];
                    int st = varState[v];
                    if (st == 1) mines--;
                    else if (st == -1) tmp[tCount++] = v;
                }

                if (tCount == 0) continue;
                if ((uint)mines > (uint)tCount) continue;

                if (mines == 0)
                {
                    for (int k = 0; k < tCount; k++)
                        anyThisIter |= SetVarState(varState, tmp[k], 0);
                }
                else if (mines == tCount)
                {
                    for (int k = 0; k < tCount; k++)
                        anyThisIter |= SetVarState(varState, tmp[k], 1);
                }
                else
                {
                    int[] rv = new int[tCount];
                    for (int k = 0; k < tCount; k++) rv[k] = tmp[k];
                    Array.Sort(rv);
                    reduced.Add(new Constraint(mines, rv));
                }
            }

            // Subset reasoning within reduced constraints.
            for (int i = 0; i < reduced.Count; i++)
            {
                var a = reduced[i];
                for (int j = i + 1; j < reduced.Count; j++)
                {
                    var b = reduced[j];
                    if (a.Vars.Length <= b.Vars.Length)
                        anyThisIter |= ApplySubsetRule(a, b, varState);
                    if (b.Vars.Length <= a.Vars.Length)
                        anyThisIter |= ApplySubsetRule(b, a, varState);
                }
            }

            if (!anyThisIter) break;
            changed = true;
        }

        if (!changed) return false;

        // Materialize assignments into the grid + propagation queue.
        for (int i = 0; i < compVars.Count; i++)
        {
            int v = compVars[i];
            int st = varState[v];
            if (st == 0) SetSafe(grid, height, width, varCellIdx[v], queue, safeCells);
            else if (st == 1) SetMine(grid, height, width, varCellIdx[v], queue);
        }

        return true;
    }

    private static bool ApplySubsetRule(Constraint small, Constraint big, int[] varState)
    {
        // If Vars(small) ⊆ Vars(big), then the difference is constrained.
        int[] a = small.Vars;
        int[] b = big.Vars;

        Span<int> diff = stackalloc int[8];
        int diffCount = 0;

        int i = 0, j = 0;
        while (i < a.Length && j < b.Length)
        {
            int av = a[i];
            int bv = b[j];
            if (av == bv) { i++; j++; continue; }
            if (av > bv) { diff[diffCount++] = bv; j++; continue; }
            return false; // av < bv => not a subset of b
        }
        if (i < a.Length) return false;
        while (j < b.Length) diff[diffCount++] = b[j++];

        int diffMines = big.Mines - small.Mines;
        if (diffMines == 0)
        {
            bool changed = false;
            for (int k = 0; k < diffCount; k++) changed |= SetVarState(varState, diff[k], 0);
            return changed;
        }
        if (diffMines == diffCount)
        {
            bool changed = false;
            for (int k = 0; k < diffCount; k++) changed |= SetVarState(varState, diff[k], 1);
            return changed;
        }
        return false;
    }

    private static bool SetVarState(int[] varState, int v, int st)
    {
        int cur = varState[v];
        if (cur == st) return false;
        if (cur != -1) return false; // conflict => ignore for safety
        varState[v] = st;
        return true;
    }

    private static int CollectUnknownNeighborVars(
        byte[] grid,
        int height,
        int width,
        int digitIdx,
        Dictionary<int, int> varIdByCell,
        List<int> varCellIdx,
        Span<int> outVarIds,
        out int minesNeeded)
    {
        byte digitByte = grid[digitIdx];
        int digit = digitByte - (byte)'0';

        int y = digitIdx / width;
        int x = digitIdx - y * width;
        int flagged = 0;
        int count = 0;

        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)height) continue;
            int rowBase = ny * width;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)width) continue;
                int nIdx = rowBase + nx;

                byte b = grid[nIdx];
                if (b == Mine)
                {
                    flagged++;
                }
                else if (b == Unknown)
                {
                    if (!varIdByCell.TryGetValue(nIdx, out int id))
                    {
                        id = varCellIdx.Count;
                        varIdByCell[nIdx] = id;
                        varCellIdx.Add(nIdx);
                    }
                    outVarIds[count++] = id;
                }
            }
        }

        minesNeeded = digit - flagged;
        return count;
    }

    private static void PropagateBasic(byte[] grid, int height, int width, IntQueue queue, List<int> safeCells)
    {
        while (queue.TryDequeue(out int dIdx))
        {
            byte digitByte = grid[dIdx];
            if (!IsDigit(digitByte)) continue;

            int digit = digitByte - (byte)'0';
            int y = dIdx / width;
            int x = dIdx - y * width;

            int flagged = 0;
            Span<int> unknowns = stackalloc int[8];
            int uCount = 0;

            for (int dy = -1; dy <= 1; dy++)
            {
                int ny = y + dy;
                if ((uint)ny >= (uint)height) continue;
                int rowBase = ny * width;
                for (int dx = -1; dx <= 1; dx++)
                {
                    if (dx == 0 && dy == 0) continue;
                    int nx = x + dx;
                    if ((uint)nx >= (uint)width) continue;
                    int nIdx = rowBase + nx;

                    byte b = grid[nIdx];
                    if (b == Mine) flagged++;
                    else if (b == Unknown) unknowns[uCount++] = nIdx;
                }
            }

            if (uCount == 0) continue;

            int remaining = digit - flagged;
            if ((uint)remaining > (uint)uCount) continue;

            if (remaining == 0)
            {
                for (int i = 0; i < uCount; i++)
                    SetSafe(grid, height, width, unknowns[i], queue, safeCells);
            }
            else if (remaining == uCount)
            {
                for (int i = 0; i < uCount; i++)
                    SetMine(grid, height, width, unknowns[i], queue);
            }
        }
    }

    private static bool HasUnknownNeighbor(byte[] grid, int height, int width, int idx)
    {
        int y = idx / width;
        int x = idx - y * width;
        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)height) continue;
            int rowBase = ny * width;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)width) continue;
                if (grid[rowBase + nx] == Unknown) return true;
            }
        }
        return false;
    }

    private static bool SetSafe(byte[] grid, int height, int width, int idx, IntQueue queue, List<int> safeCells)
    {
        if (grid[idx] != Unknown) return false;
        grid[idx] = SafeMark;
        safeCells.Add(idx);
        EnqueueAdjacentDigits(grid, height, width, idx, queue);
        return true;
    }

    private static bool SetMine(byte[] grid, int height, int width, int idx, IntQueue queue)
    {
        if (grid[idx] != Unknown) return false;
        grid[idx] = Mine;
        EnqueueAdjacentDigits(grid, height, width, idx, queue);
        return true;
    }

    private static void EnqueueAdjacentDigits(byte[] grid, int height, int width, int idx, IntQueue queue)
    {
        int y = idx / width;
        int x = idx - y * width;
        for (int dy = -1; dy <= 1; dy++)
        {
            int ny = y + dy;
            if ((uint)ny >= (uint)height) continue;
            int rowBase = ny * width;
            for (int dx = -1; dx <= 1; dx++)
            {
                if (dx == 0 && dy == 0) continue;
                int nx = x + dx;
                if ((uint)nx >= (uint)width) continue;
                int nIdx = rowBase + nx;
                if (IsDigit(grid[nIdx]))
                    queue.Enqueue(nIdx);
            }
        }
    }

    private static bool IsDigit(byte b) => b >= (byte)'0' && b <= (byte)'8';

    private readonly struct Constraint
    {
        public readonly int Mines;
        public readonly int[] Vars;
        public Constraint(int mines, int[] vars)
        {
            Mines = mines;
            Vars = vars;
        }
    }

    private readonly struct LocalConstraint
    {
        public readonly int Mines;
        public readonly int[] Vars;
        public LocalConstraint(int mines, int[] vars)
        {
            Mines = mines;
            Vars = vars;
        }
    }

    private sealed class IntQueue
    {
        private int[] _buf;
        private int _head;
        private int _tail;

        public IntQueue(int capacity)
        {
            _buf = new int[Math.Max(4, capacity)];
        }

        public void Enqueue(int value)
        {
            if (_tail == _buf.Length)
            {
                if (_head > 0)
                {
                    int len = _tail - _head;
                    Array.Copy(_buf, _head, _buf, 0, len);
                    _head = 0;
                    _tail = len;
                }
                else
                {
                    Array.Resize(ref _buf, _buf.Length * 2);
                }
            }
            _buf[_tail++] = value;
        }

        public bool TryDequeue(out int value)
        {
            if (_head == _tail)
            {
                value = 0;
                return false;
            }
            value = _buf[_head++];
            if (_head == _tail)
            {
                _head = 0;
                _tail = 0;
            }
            return true;
        }
    }

    private sealed class FastInput
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;
        private int _lastDelimiter;

        public FastInput(Stream stream)
        {
            _stream = stream;
            _buffer = new byte[1 << 16];
        }

        private int ReadByteRaw()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int ReadInt()
        {
            int c;
            do { c = ReadByteRaw(); } while (c <= 32 && c >= 0);
            if (c < 0) return 0;

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByteRaw();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = ReadByteRaw();
            }
            _lastDelimiter = c;
            return val * sign;
        }

        // If the header line contains extra whitespace after the ints, discard it
        // without eating into the first board row (which can legitimately start with spaces).
        public void SkipToLineEndIfNeeded()
        {
            if (_lastDelimiter == '\n' || _lastDelimiter < 0) return;
            int c;
            do { c = ReadByteRaw(); } while (c >= 0 && c != '\n');
        }

        // Board bytes include spaces as data, so skip only CR/LF.
        public byte ReadBoardByte()
        {
            int c;
            do { c = ReadByteRaw(); } while (c == '\r' || c == '\n');
            return c < 0 ? (byte)0 : (byte)c;
        }
    }
}
