// Mission-control workstation scheduler (benchmark solution)
// Reads one JSON document from stdin, prints one JSON document to stdout.

declare const require: any;
declare const process: any;
const fs = require("fs");

type AnyId = string | number;

function trunc(n: number): number {
  return n < 0 ? Math.ceil(n) : Math.floor(n);
}

function isArray(value: any): value is any[] {
  return Object.prototype.toString.call(value) === "[object Array]";
}

function toInt(value: any, fallback: number): number {
  const n = Number(value);
  if (!isFinite(n)) return fallback;
  return trunc(n);
}

function toNonNegInt(value: any, fallback: number): number {
  const n = toInt(value, fallback);
  return n < 0 ? 0 : n;
}

function toNumber(value: any, fallback: number): number {
  const n = Number(value);
  return isFinite(n) ? n : fallback;
}

class Heap<T> {
  private a: T[] = [];
  constructor(private better: (x: T, y: T) => boolean) {}

  isEmpty(): boolean {
    return this.a.length === 0;
  }

  peek(): T | undefined {
    return this.a[0];
  }

  push(x: T): void {
    const a = this.a;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (this.better(a[p], a[i])) break;
      const tmp = a[p];
      a[p] = a[i];
      a[i] = tmp;
      i = p;
    }
  }

  pop(): T | undefined {
    const a = this.a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const last = a.pop() as T;
    if (n > 1) {
      a[0] = last;
      let i = 0;
      const nn = a.length;
      while (true) {
        const l = i * 2 + 1;
        if (l >= nn) break;
        const r = l + 1;
        let b = l;
        if (r < nn && this.better(a[r], a[l])) b = r;
        if (this.better(a[i], a[b])) break;
        const tmp = a[i];
        a[i] = a[b];
        a[b] = tmp;
        i = b;
      }
    }
    return top;
  }
}

type Task = {
  id: AnyId;
  duration: number;
  deps: AnyId[];
  deadline: number;
  weight: number;
};

const rawInput = fs.readFileSync(0, "utf8");
const trimmed = rawInput.trim();
if (!trimmed) {
  process.stdout.write(JSON.stringify({ order: [] }));
  process.exit(0);
}

const data = JSON.parse(trimmed);
const tasksIn: Task[] = isArray(data.tasks) ? data.tasks : [];
let machineCount = toInt(data.machineCount, 1);
if (!isFinite(machineCount) || machineCount < 1) machineCount = 1;

const n = tasksIn.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ order: [] }));
  process.exit(0);
}

// Normalize + index tasks
const idRaw: AnyId[] = new Array(n);
const duration: number[] = new Array(n);
const deadline: number[] = new Array(n);
const weight: number[] = new Array(n);
const depIdx: number[][] = new Array(n);
const succ: number[][] = new Array(n);
const indegree: number[] = new Array(n);
for (let i = 0; i < n; i++) indegree[i] = 0;

const idToIndex: { [k: string]: number } = {};
for (let i = 0; i < n; i++) {
  const t = tasksIn[i] as any;
  const id = (t && t.id !== undefined ? t.id : i) as AnyId;
  idRaw[i] = id;
  const key = "$" + String(id);
  if (idToIndex[key] === undefined) idToIndex[key] = i;
}

let totalDur = 0;
for (let i = 0; i < n; i++) {
  const t = tasksIn[i] as any;
  const p = toNonNegInt(t?.duration, 1);
  duration[i] = p > 0 ? p : 1;
  totalDur += duration[i];
  const d = toNumber(t?.deadline, Number.POSITIVE_INFINITY);
  deadline[i] = isFinite(d) ? d : Number.POSITIVE_INFINITY;
  const w = toNumber(t?.weight, 1);
  weight[i] = isFinite(w) ? w : 1;
  const depsArr = isArray(t?.deps) ? (t.deps as AnyId[]) : [];
  depIdx[i] = [];
  succ[i] = [];
  for (let k = 0; k < depsArr.length; k++) {
    const dep = depsArr[k];
    const j = idToIndex["$" + String(dep)];
    if (j === undefined || j === i) continue;
    depIdx[i].push(j);
    succ[j].push(i);
    indegree[i]++;
  }
}

const avgDur = totalDur > 0 ? totalDur / n : 1;
const avg = avgDur > 0 ? avgDur : 1;

// Base topological order (for computing levels). If cyclic, fall back to input order.
const indegTmp = indegree.slice();
const stack: number[] = [];
for (let i = 0; i < n; i++) if (indegTmp[i] === 0) stack.push(i);
const topo: number[] = [];
while (stack.length > 0) {
  const u = stack.pop() as number;
  topo.push(u);
  const su = succ[u];
  for (let k = 0; k < su.length; k++) {
    const v = su[k];
    const nv = --indegTmp[v];
    if (nv === 0) stack.push(v);
  }
}
if (topo.length !== n) {
  process.stdout.write(JSON.stringify({ order: idRaw }));
  process.exit(0);
}

// t-level (earliest start, ignoring machine constraints)
const topLevel: number[] = new Array(n);
for (let i = 0; i < n; i++) topLevel[i] = 0;
for (let t = 0; t < topo.length; t++) {
  const u = topo[t];
  const finishU = topLevel[u] + duration[u];
  const su = succ[u];
  for (let k = 0; k < su.length; k++) {
    const v = su[k];
    if (topLevel[v] < finishU) topLevel[v] = finishU;
  }
}

// b-level (bottom level / upward rank without comm)
const bottomLevel: number[] = new Array(n);
for (let i = 0; i < n; i++) bottomLevel[i] = duration[i];
for (let ti = topo.length - 1; ti >= 0; ti--) {
  const u = topo[ti];
  const su = succ[u];
  let maxSucc = 0;
  for (let k = 0; k < su.length; k++) {
    const v = su[k];
    const bl = bottomLevel[v];
    if (bl > maxSucc) maxSucc = bl;
  }
  bottomLevel[u] = duration[u] + maxSucc;
}

function buildOrderByScore(score: number[]): number[] {
  type Node = { i: number; key: number };
  const indeg = indegree.slice();
  const ready = new Heap<Node>((a, b) => {
    if (a.key !== b.key) return a.key > b.key;
    return a.i < b.i;
  });
  for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push({ i, key: score[i] });
  const order: number[] = new Array(n);
  let out = 0;
  while (!ready.isEmpty()) {
    const node = ready.pop() as Node;
    const u = node.i;
    order[out++] = u;
    const su = succ[u];
    for (let k = 0; k < su.length; k++) {
      const v = su[k];
      const nv = --indeg[v];
      if (nv === 0) ready.push({ i: v, key: score[v] });
    }
  }
  // DAG guaranteed here; still guard.
  if (out !== n) {
    const seen = new Array<boolean>(n);
    for (let i = 0; i < n; i++) seen[i] = false;
    for (let k = 0; k < out; k++) seen[order[k]] = true;
    for (let i = 0; i < n; i++) if (!seen[i]) order[out++] = i;
  }
  return order;
}

function evaluateGreedy(order: number[]): number {
  type M = { time: number; idx: number };
  const machines = new Heap<M>((a, b) => a.time < b.time || (a.time === b.time && a.idx < b.idx));
  for (let m = 0; m < machineCount; m++) machines.push({ time: 0, idx: m });

  const finish: number[] = new Array(n);
  for (let i = 0; i < n; i++) finish[i] = 0;
  let makespan = 0;
  let weightedTardiness = 0;

  for (let pos = 0; pos < order.length; pos++) {
    const u = order[pos];
    let readyTime = 0;
    const deps = depIdx[u];
    for (let k = 0; k < deps.length; k++) {
      const f = finish[deps[k]];
      if (f > readyTime) readyTime = f;
    }
    const m = machines.pop() as M;
    const start = readyTime > m.time ? readyTime : m.time;
    const f = start + duration[u];
    finish[u] = f;
    machines.push({ time: f, idx: m.idx });

    const late = f - deadline[u];
    if (late > 0) weightedTardiness += weight[u] * late;
    if (f > makespan) makespan = f;
  }

  return makespan + weightedTardiness;
}

// Candidate scoring rules (static, based on t-level estimate).
const scoreATC: number[] = new Array(n);
const scoreWMDD: number[] = new Array(n);
const k1 = 2.0;

for (let i = 0; i < n; i++) {
  const p = duration[i] > 0 ? duration[i] : 1;
  const w = weight[i];
  const d = deadline[i];
  const t = topLevel[i];
  const bl = bottomLevel[i];

  const blBoostATC = 1 + bl / (avg * 8);
  const slack = d - p - t;
  const expTerm = slack > 0 ? Math.exp(-slack / (k1 * avg)) : 1;
  scoreATC[i] = (w / p) * expTerm * blBoostATC;

  const blBoostWMDD = 1 + bl / (avg * 10);
  const denom = Math.max(p, d - t);
  const denomSafe = denom > 0 ? denom : p;
  scoreWMDD[i] = (w / denomSafe) * blBoostWMDD;
}

const candidates: { name: string; score: number[] }[] = [];
// Keep runtime bounded on very large instances.
if (n <= 200000) candidates.push({ name: "wmdd", score: scoreWMDD });
candidates.push({ name: "atc", score: scoreATC });

let bestOrder: number[] | null = null;
let bestObj = Number.POSITIVE_INFINITY;

for (let c = 0; c < candidates.length; c++) {
  const ord = buildOrderByScore(candidates[c].score);
  const obj = evaluateGreedy(ord);
  if (obj < bestObj) {
    bestObj = obj;
    bestOrder = ord;
  }
}

const finalOrder = (bestOrder ?? topo).map((i) => idRaw[i]);
process.stdout.write(JSON.stringify({ order: finalOrder }));
