(() => {
const fs = require('fs');

const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = Array.isArray(data.grid) ? data.grid : [];
const robotsIn = Array.isArray(data.robots) ? data.robots : [];
const jobsIn = Array.isArray(data.jobs) ? data.jobs : [];

const H = grid.length | 0;
const W = (H > 0 ? (grid[0] ? grid[0].length : 0) : 0) | 0;
const N = (W * H) | 0;

function inBounds(x, y) {
  return x >= 0 && y >= 0 && x < W && y < H;
}
function toIndex(x, y) {
  return (y * W + x) | 0;
}
function manhattan(ax, ay, bx, by) {
  const dx = ax - bx;
  const dy = ay - by;
  return (dx < 0 ? -dx : dx) + (dy < 0 ? -dy : dy);
}

let maxRobotId = -1;
for (let i = 0; i < robotsIn.length; i++) {
  const rid = Number(robotsIn[i]?.id);
  if (Number.isFinite(rid) && rid > maxRobotId) maxRobotId = rid | 0;
}

if (maxRobotId < 0) {
  process.stdout.write(JSON.stringify({ assignments: [] }));
  return;
}

const robotX = new Int32Array(maxRobotId + 1);
const robotY = new Int32Array(maxRobotId + 1);
const robotValid = new Uint8Array(maxRobotId + 1);
robotX.fill(-1);
robotY.fill(-1);

for (let i = 0; i < robotsIn.length; i++) {
  const r = robotsIn[i] || {};
  const rid = (Number(r.id) | 0);
  if (rid < 0 || rid > maxRobotId) continue;
  const x = Number(r.x) | 0;
  const y = Number(r.y) | 0;
  robotX[rid] = x;
  robotY[rid] = y;
  if (inBounds(x, y)) robotValid[rid] = 1;
}

const assignmentsBuckets = new Array(maxRobotId + 1);
for (let i = 0; i <= maxRobotId; i++) assignmentsBuckets[i] = [];

if (N === 0 || W === 0 || H === 0 || jobsIn.length === 0) {
  for (let j = 0; j < jobsIn.length; j++) {
    assignmentsBuckets[0].push(jobsIn[j]?.id);
  }
  process.stdout.write(JSON.stringify({ assignments: assignmentsBuckets }));
  return;
}

const open = new Uint8Array(N);
for (let y = 0; y < H; y++) {
  const row = grid[y] || '';
  const base = y * W;
  for (let x = 0; x < W; x++) {
    // Treat anything other than '#' as traversable.
    open[base + x] = row.charCodeAt(x) === 35 ? 0 : 1;
  }
}

// Multi-source BFS from robot start positions to assign each cell to its nearest robot.
let dist = null;
let owner = null;
if (robotsIn.length > 0) {
  const sources = [];
  for (let rid = 0; rid <= maxRobotId; rid++) {
    if (!robotValid[rid]) continue;
    const x = robotX[rid];
    const y = robotY[rid];
    if (!inBounds(x, y)) continue;
    const s = toIndex(x, y);
    if (open[s]) sources.push(rid);
  }
  sources.sort((a, b) => a - b);

  if (sources.length > 0) {
    dist = new Int32Array(N);
    owner = new Int32Array(N);
    dist.fill(-1);
    owner.fill(-1);
    const queue = new Int32Array(N);
    let head = 0;
    let tail = 0;

    for (let i = 0; i < sources.length; i++) {
      const rid = sources[i];
      const s = toIndex(robotX[rid], robotY[rid]);
      if (dist[s] === 0) {
        if (rid < owner[s]) owner[s] = rid;
        continue;
      }
      dist[s] = 0;
      owner[s] = rid;
      queue[tail++] = s;
    }

    while (head < tail) {
      const idx = queue[head++];
      const nextD = (dist[idx] + 1) | 0;
      const o = owner[idx] | 0;
      const x = idx % W;

      // left
      if (x > 0) {
        const n = idx - 1;
        if (open[n] && dist[n] === -1) {
          dist[n] = nextD;
          owner[n] = o;
          queue[tail++] = n;
        }
      }
      // right
      if (x + 1 < W) {
        const n = idx + 1;
        if (open[n] && dist[n] === -1) {
          dist[n] = nextD;
          owner[n] = o;
          queue[tail++] = n;
        }
      }
      // up
      if (idx >= W) {
        const n = idx - W;
        if (open[n] && dist[n] === -1) {
          dist[n] = nextD;
          owner[n] = o;
          queue[tail++] = n;
        }
      }
      // down
      const dn = idx + W;
      if (dn < N) {
        const n = dn;
        if (open[n] && dist[n] === -1) {
          dist[n] = nextD;
          owner[n] = o;
          queue[tail++] = n;
        }
      }
    }
  }
}

function part1By1(n) {
  n &= 0xffff;
  n = (n | (n << 8)) & 0x00ff00ff;
  n = (n | (n << 4)) & 0x0f0f0f0f;
  n = (n | (n << 2)) & 0x33333333;
  n = (n | (n << 1)) & 0x55555555;
  return n;
}
function morton2D16(x, y) {
  return (part1By1(x) | (part1By1(y) << 1)) >>> 0;
}

const xScale = W > 1 ? 65535 / (W - 1) : 0;
const yScale = H > 1 ? 65535 / (H - 1) : 0;
function mortonKey(x, y) {
  let xx = (xScale ? (x * xScale) | 0 : 0) & 0xffff;
  let yy = (yScale ? (y * yScale) | 0 : 0) & 0xffff;
  if (xx < 0) xx = 0;
  if (yy < 0) yy = 0;
  if (xx > 65535) xx = 65535;
  if (yy > 65535) yy = 65535;
  return morton2D16(xx, yy);
}

function nearestRobotByManhattan(x, y) {
  let bestRid = 0;
  let bestD = 2147483647;
  for (let rid = 0; rid <= maxRobotId; rid++) {
    if (!robotValid[rid]) continue;
    const d = manhattan(x, y, robotX[rid], robotY[rid]);
    if (d < bestD || (d === bestD && rid < bestRid)) {
      bestD = d;
      bestRid = rid;
    }
  }
  return bestRid;
}

// Bucket jobs to their nearest robot (by exact grid distance when available).
const jobRecsByRobot = new Array(maxRobotId + 1);
for (let rid = 0; rid <= maxRobotId; rid++) jobRecsByRobot[rid] = [];

for (let i = 0; i < jobsIn.length; i++) {
  const job = jobsIn[i] || {};
  const x = Number(job.x) | 0;
  const y = Number(job.y) | 0;
  const p = Number(job.priority);
  const priority = Number.isFinite(p) ? p : 0;
  const jobId = job.id;

  let rid = 0;
  let dKey = 2147483647;

  if (inBounds(x, y) && open[toIndex(x, y)] && owner) {
    const cell = toIndex(x, y);
    const o = owner[cell];
    if (o >= 0) {
      rid = o;
      dKey = dist ? dist[cell] : 0;
    } else {
      rid = nearestRobotByManhattan(x, y);
      dKey = manhattan(x, y, robotX[rid], robotY[rid]);
    }
  } else if (inBounds(x, y)) {
    rid = nearestRobotByManhattan(x, y);
    dKey = manhattan(x, y, robotX[rid], robotY[rid]);
  } else {
    rid = 0;
  }

  const idNumRaw = Number(jobId);
  const idNum = Number.isFinite(idNumRaw) ? idNumRaw : null;
  const idStr = typeof jobId === 'string' ? jobId : String(jobId);

  jobRecsByRobot[rid].push({
    id: jobId,
    p: priority,
    d: dKey | 0,
    m: mortonKey(x, y),
    n: idNum,
    s: idStr,
  });
}

function cmpJob(a, b) {
  if (a.p !== b.p) return b.p - a.p;
  if (a.d !== b.d) return a.d - b.d;
  if (a.m !== b.m) return a.m - b.m;
  if (a.n !== null && b.n !== null && a.n !== b.n) return a.n - b.n;
  if (a.s < b.s) return -1;
  if (a.s > b.s) return 1;
  return 0;
}

for (let rid = 0; rid <= maxRobotId; rid++) {
  const recs = jobRecsByRobot[rid];
  if (recs.length === 0) continue;
  recs.sort(cmpJob);
  const out = assignmentsBuckets[rid];
  out.length = recs.length;
  for (let i = 0; i < recs.length; i++) out[i] = recs[i].id;
}

process.stdout.write(JSON.stringify({ assignments: assignmentsBuckets }));
})();
