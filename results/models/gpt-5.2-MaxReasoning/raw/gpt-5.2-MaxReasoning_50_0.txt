#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint id = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (id >= N) {
        return;
    }

    uint base = id * 2u;
    vec3 pos = inData[base].xyz;
    vec3 vel = inData[base + 1u].xyz;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    float sepR2 = sepRadius * sepRadius;
    float alignR2 = alignRadius * alignRadius;
    float cohR2 = cohRadius * cohRadius;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);

    uint sepCount = 0u;
    uint alignCount = 0u;
    uint cohCount = 0u;

    const float EPS = 1e-8;

    for (uint j = 0u; j < N; ++j) {
        if (j == id) {
            continue;
        }

        uint jBase = j * 2u;
        vec3 otherPos = inData[jBase].xyz;
        vec3 otherVel = inData[jBase + 1u].xyz;

        vec3 diff = pos - otherPos;
        float dist2 = dot(diff, diff);
        if (dist2 <= EPS) {
            continue;
        }

        if (dist2 < sepR2) {
            float invDist = inversesqrt(dist2);
            sepSum += diff * invDist * invDist; // diff / dist^2
            sepCount += 1u;
        }

        if (dist2 < alignR2) {
            alignSum += otherVel;
            alignCount += 1u;
        }

        if (dist2 < cohR2) {
            cohSum += otherPos;
            cohCount += 1u;
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0u) {
        accel += (sepSum / float(sepCount)) * sepWeight;
    }
    if (alignCount > 0u) {
        vec3 avgVel = alignSum / float(alignCount);
        accel += (avgVel - vel) * alignWeight;
    }
    if (cohCount > 0u) {
        vec3 center = cohSum / float(cohCount);
        accel += (center - pos) * cohWeight;
    }

    // Boundary force: when a component exceeds boundSize*0.4, push toward center.
    float limit = boundSize * 0.4;
    float half = boundSize * 0.5;
    vec3 boundAccel = vec3(0.0);

    if (pos.x > limit) {
        boundAccel.x -= boundForce * (pos.x - limit) / half;
    } else if (pos.x < -limit) {
        boundAccel.x += boundForce * (-limit - pos.x) / half;
    }
    if (pos.y > limit) {
        boundAccel.y -= boundForce * (pos.y - limit) / half;
    } else if (pos.y < -limit) {
        boundAccel.y += boundForce * (-limit - pos.y) / half;
    }
    if (pos.z > limit) {
        boundAccel.z -= boundForce * (pos.z - limit) / half;
    } else if (pos.z < -limit) {
        boundAccel.z += boundForce * (-limit - pos.z) / half;
    }

    accel += boundAccel;

    vel += accel * dt;

    float speed = length(vel);
    if (speed > maxSpeed && speed > 0.0) {
        vel = (vel / speed) * maxSpeed;
    }

    pos += vel * dt;

    outData[base] = vec4(pos, 0.0);
    outData[base + 1u] = vec4(vel, 0.0);
}
