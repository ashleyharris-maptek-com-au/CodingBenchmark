; SPIR-V
; Version: 1.0
; Schema: 0

OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %in_worldPos %in_normal %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %in_worldPos "worldPos"
OpName %in_normal "normal"
OpName %in_color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0.0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in = OpTypeArray %v3float %uint_2

%ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_Input_arr_v3_in = OpTypePointer Input %arr_v3_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv_in Input
%in_worldPos = OpVariable %ptr_Input_arr_v3_in Input
%in_normal = OpVariable %ptr_Input_arr_v3_in Input
%in_color = OpVariable %ptr_Input_arr_v3_in Input

%gl_out = OpVariable %ptr_Output_pv_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_Input_v3float %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr

%dir4 = OpFSub %v4float %p1 %p0
%dx = OpCompositeExtract %float %dir4 0
%dy = OpCompositeExtract %float %dir4 1
%neg_dy = OpFNegate %float %dy
%perp_un = OpCompositeConstruct %v2float %neg_dy %dx
%perp = OpExtInst %v2float %glsl Normalize %perp_un

%offset2_thin = OpVectorTimesScalar %v2float %perp %float_0_015
%offx_thin = OpCompositeExtract %float %offset2_thin 0
%offy_thin = OpCompositeExtract %float %offset2_thin 1
%offset4_thin = OpCompositeConstruct %v4float %offx_thin %offy_thin %float_0 %float_0

%offset2_big = OpVectorTimesScalar %v2float %perp %float_0_05
%offx_big = OpCompositeExtract %float %offset2_big 0
%offy_big = OpCompositeExtract %float %offset2_big 1
%offset4_big = OpCompositeConstruct %v4float %offx_big %offy_big %float_0 %float_0

%dir4_075 = OpVectorTimesScalar %v4float %dir4 %float_0_75
%mid = OpFAdd %v4float %p0 %dir4_075

%p0_plus = OpFAdd %v4float %p0 %offset4_thin
%p0_minus = OpFSub %v4float %p0 %offset4_thin
%mid_plus = OpFAdd %v4float %mid %offset4_thin
%mid_minus = OpFSub %v4float %mid %offset4_thin

%mid_big_plus = OpFAdd %v4float %mid %offset4_big
%mid_big_minus = OpFSub %v4float %mid %offset4_big

%outPos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

OpStore %outPos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %outPos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

OpStore %outPos_ptr %mid_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %outPos_ptr %mid_minus
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

OpStore %outPos_ptr %mid_big_plus
OpStore %outColor %c1
OpEmitVertex

OpStore %outPos_ptr %mid_big_minus
OpStore %outColor %c1
OpEmitVertex

OpStore %outPos_ptr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
