/* eslint-disable no-unused-vars */
'use strict';

// Benchmark task: plan grid paths for search drones.
// Reads one JSON document from stdin, outputs {"paths": [...]} to stdout.

const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = input && input.trim().length ? JSON.parse(input) : {};

const probability = Array.isArray(data.probability) ? data.probability : [];
const drones = Array.isArray(data.drones) ? data.drones : [];
const maxStepsPerDrone = Math.max(0, (data.maxStepsPerDrone | 0) || 0);

const H = probability.length | 0;
const W = H ? (probability[0].length | 0) : 0;

function iabs(a) {
  return a < 0 ? -a : a;
}

function idxToX(idx) {
  return idx % W;
}
function idxToY(idx) {
  return (idx / W) | 0;
}

if (!H || !W || !drones.length || maxStepsPerDrone < 0) {
  const paths = drones.map(() => []);
  process.stdout.write(JSON.stringify({ paths }));
  process.exit(0);
}

const N = W * H;

// Flatten grid for speed. -1 => obstacle, positive => reward, 0 => free/no reward.
const val = new Float64Array(N);
{
  let k = 0;
  for (let y = 0; y < H; y++) {
    const row = probability[y] || [];
    for (let x = 0; x < W; x++) {
      const v = row[x];
      val[k++] = typeof v === 'number' ? v : -1;
    }
  }
}

// Smoothed value encourages heading toward dense regions of reward.
const smoothed = new Float64Array(N);
{
  for (let y = 0; y < H; y++) {
    const rowBase = y * W;
    for (let x = 0; x < W; x++) {
      const idx = rowBase + x;
      const v = val[idx];
      if (v < 0) {
        smoothed[idx] = 0;
        continue;
      }
      let s = v;
      if (x > 0) s += Math.max(0, val[idx - 1]) * 0.25;
      if (x + 1 < W) s += Math.max(0, val[idx + 1]) * 0.25;
      if (y > 0) s += Math.max(0, val[idx - W]) * 0.25;
      if (y + 1 < H) s += Math.max(0, val[idx + W]) * 0.25;
      smoothed[idx] = s;
    }
  }
}

// Marks whether a positive-value cell has been covered by any drone yet.
const covered = new Uint8Array(N);

// Build a list of top-K hotspots (by raw value) for long-range guidance.
function buildHotspots(maxK) {
  const heapIdx = [];
  const heapVal = [];
  let heapSize = 0;

  function swap(i, j) {
    const ti = heapIdx[i];
    heapIdx[i] = heapIdx[j];
    heapIdx[j] = ti;
    const tv = heapVal[i];
    heapVal[i] = heapVal[j];
    heapVal[j] = tv;
  }

  function bubbleUp(pos) {
    while (pos > 0) {
      const parent = (pos - 1) >> 1;
      if (heapVal[pos] >= heapVal[parent]) break;
      swap(pos, parent);
      pos = parent;
    }
  }

  function bubbleDown(pos) {
    for (;;) {
      const left = pos * 2 + 1;
      if (left >= heapSize) break;
      const right = left + 1;
      let best = left;
      if (right < heapSize && heapVal[right] < heapVal[left]) best = right;
      if (heapVal[pos] <= heapVal[best]) break;
      swap(pos, best);
      pos = best;
    }
  }

  function push(idx, v) {
    heapIdx[heapSize] = idx;
    heapVal[heapSize] = v;
    bubbleUp(heapSize);
    heapSize++;
  }

  function replaceRoot(idx, v) {
    heapIdx[0] = idx;
    heapVal[0] = v;
    bubbleDown(0);
  }

  for (let idx = 0; idx < N; idx++) {
    const v = val[idx];
    if (v <= 0) continue;
    if (heapSize < maxK) {
      push(idx, v);
    } else if (v > heapVal[0]) {
      replaceRoot(idx, v);
    }
  }

  const outIdx = new Int32Array(heapSize);
  const outVal = new Float64Array(heapSize);
  for (let i = 0; i < heapSize; i++) {
    outIdx[i] = heapIdx[i];
    outVal[i] = heapVal[i];
  }

  // Sort by descending value.
  const order = Array.from({ length: heapSize }, (_, i) => i);
  order.sort((a, b) => outVal[b] - outVal[a]);
  const sortedIdx = new Int32Array(heapSize);
  const sortedVal = new Float64Array(heapSize);
  for (let i = 0; i < heapSize; i++) {
    const j = order[i];
    sortedIdx[i] = outIdx[j];
    sortedVal[i] = outVal[j];
  }
  return { idx: sortedIdx, val: sortedVal };
}

const hotspots = buildHotspots(Math.min(5000, N));
const hotspotCount = hotspots.idx.length | 0;
const goalScanCount = Math.min(1000, hotspotCount) | 0;

function chooseGoal(currentIdx) {
  if (goalScanCount <= 0) return -1;
  const cx = currentIdx % W;
  const cy = (currentIdx / W) | 0;
  let bestIdx = -1;
  let bestScore = 0;
  for (let i = 0; i < goalScanCount; i++) {
    const idx = hotspots.idx[i];
    if (covered[idx] !== 0) continue;
    const x = idx % W;
    const y = (idx / W) | 0;
    const d = iabs(x - cx) + iabs(y - cy);
    const score = hotspots.val[i] / (d + 1);
    if (score > bestScore) {
      bestScore = score;
      bestIdx = idx;
    }
  }
  return bestIdx;
}

function nearestFreeIndex(x, y) {
  if (x >= 0 && x < W && y >= 0 && y < H) {
    const idx = y * W + x;
    if (val[idx] >= 0) return idx;
  }
  // Small outward diamond search; start positions should almost always be valid.
  const maxR = 50;
  for (let r = 1; r <= maxR; r++) {
    const y0 = y - r;
    const y1 = y + r;
    for (let xx = x - r; xx <= x + r; xx++) {
      if (y0 >= 0 && y0 < H && xx >= 0 && xx < W) {
        const idx0 = y0 * W + xx;
        if (val[idx0] >= 0) return idx0;
      }
      if (y1 >= 0 && y1 < H && xx >= 0 && xx < W) {
        const idx1 = y1 * W + xx;
        if (val[idx1] >= 0) return idx1;
      }
    }
    for (let yy = y - r + 1; yy <= y + r - 1; yy++) {
      const x0 = x - r;
      const x1 = x + r;
      if (yy >= 0 && yy < H && x0 >= 0 && x0 < W) {
        const idx0 = yy * W + x0;
        if (val[idx0] >= 0) return idx0;
      }
      if (yy >= 0 && yy < H && x1 >= 0 && x1 < W) {
        const idx1 = yy * W + x1;
        if (val[idx1] >= 0) return idx1;
      }
    }
  }
  // Fallback: scan for any free cell.
  for (let idx = 0; idx < N; idx++) {
    if (val[idx] >= 0) return idx;
  }
  return -1;
}

// BFS work buffers (reused across calls, using stamp technique).
const MAX_DEPTH = 200;
const maxQueue = 1 + 2 * MAX_DEPTH * (MAX_DEPTH + 1);
const queue = new Int32Array(maxQueue);
const seenStamp = new Int32Array(N);
const parent = new Int32Array(N);
const dist = new Uint16Array(N);
let stamp = 0;
const tmpPath = new Int32Array(MAX_DEPTH + 2);

function chooseTarget(startIdx, maxDepth, goalIdx) {
  stamp++;
  if (stamp >= 2147483646) {
    seenStamp.fill(0);
    stamp = 1;
  }

  let head = 0;
  let tail = 0;
  seenStamp[startIdx] = stamp;
  parent[startIdx] = -1;
  dist[startIdx] = 0;
  queue[tail++] = startIdx;

  let bestRewardIdx = -1;
  let bestRewardScore = 0;

  let bestGoalIdx = -1;
  let bestGoalMetric = 1e30;
  let goalX = 0;
  let goalY = 0;
  if (goalIdx >= 0) {
    goalX = goalIdx % W;
    goalY = (goalIdx / W) | 0;
  }

  while (head < tail) {
    const idx = queue[head++];
    const d = dist[idx];
    const v = val[idx];

    if (v > 0 && covered[idx] === 0) {
      const score = smoothed[idx] / Math.sqrt(d + 1);
      if (score > bestRewardScore) {
        bestRewardScore = score;
        bestRewardIdx = idx;
      }
    }

    if (goalIdx >= 0) {
      const y = (idx / W) | 0;
      const x = idx - y * W;
      const m = iabs(x - goalX) + iabs(y - goalY);
      const metric = m * 4 + d;
      if (metric < bestGoalMetric) {
        bestGoalMetric = metric;
        bestGoalIdx = idx;
      }
    }

    if (d === maxDepth) continue;

    const nd = d + 1;
    const y = (idx / W) | 0;
    const x = idx - y * W;

    if (x > 0) {
      const n = idx - 1;
      if (seenStamp[n] !== stamp && val[n] >= 0) {
        seenStamp[n] = stamp;
        parent[n] = idx;
        dist[n] = nd;
        queue[tail++] = n;
      }
    }
    if (x + 1 < W) {
      const n = idx + 1;
      if (seenStamp[n] !== stamp && val[n] >= 0) {
        seenStamp[n] = stamp;
        parent[n] = idx;
        dist[n] = nd;
        queue[tail++] = n;
      }
    }
    if (y > 0) {
      const n = idx - W;
      if (seenStamp[n] !== stamp && val[n] >= 0) {
        seenStamp[n] = stamp;
        parent[n] = idx;
        dist[n] = nd;
        queue[tail++] = n;
      }
    }
    if (y + 1 < H) {
      const n = idx + W;
      if (seenStamp[n] !== stamp && val[n] >= 0) {
        seenStamp[n] = stamp;
        parent[n] = idx;
        dist[n] = nd;
        queue[tail++] = n;
      }
    }
  }

  if (bestRewardIdx !== -1) return bestRewardIdx;
  if (goalIdx >= 0 && bestGoalIdx !== -1 && bestGoalIdx !== startIdx) return bestGoalIdx;
  return -1;
}

function collectPath(targetIdx, startIdx) {
  let len = 0;
  let cur = targetIdx;
  while (cur !== -1 && cur !== startIdx && len < tmpPath.length) {
    tmpPath[len++] = cur;
    cur = parent[cur];
  }
  return len;
}

function bestNeighborStep(currentIdx, goalIdx) {
  const cy = (currentIdx / W) | 0;
  const cx = currentIdx - cy * W;
  let gx = 0;
  let gy = 0;
  if (goalIdx >= 0) {
    gx = goalIdx % W;
    gy = (goalIdx / W) | 0;
  }

  let bestIdx = currentIdx;
  let bestScore = -1e30;

  function consider(nIdx, nx, ny, stayPenalty) {
    if (val[nIdx] < 0) return;
    let s = 0;
    const v = val[nIdx];
    if (v > 0 && covered[nIdx] === 0) {
      s += v * 2 + smoothed[nIdx];
    } else {
      s += smoothed[nIdx] * 0.05;
    }
    if (goalIdx >= 0) {
      const m = iabs(nx - gx) + iabs(ny - gy);
      s -= m * 0.02;
    }
    s -= stayPenalty;
    if (s > bestScore) {
      bestScore = s;
      bestIdx = nIdx;
    }
  }

  consider(currentIdx, cx, cy, 0.1);
  if (cx > 0) consider(currentIdx - 1, cx - 1, cy, 0);
  if (cx + 1 < W) consider(currentIdx + 1, cx + 1, cy, 0);
  if (cy > 0) consider(currentIdx - W, cx, cy - 1, 0);
  if (cy + 1 < H) consider(currentIdx + W, cx, cy + 1, 0);

  return bestIdx;
}

const paths = new Array(drones.length);

for (let i = 0; i < drones.length; i++) {
  const d = drones[i] || {};
  let sx = d.x | 0;
  let sy = d.y | 0;
  let current = nearestFreeIndex(sx, sy);
  if (current < 0) {
    paths[i] = [];
    continue;
  }

  if (val[current] > 0) covered[current] = 1;

  const pathLenTarget = maxStepsPerDrone + 1;
  const path = new Array(pathLenTarget);
  let step = 0;
  path[step++] = [current % W, (current / W) | 0];

  let goalIdx = chooseGoal(current);

  while (step < pathLenTarget) {
    const stepsLeft = pathLenTarget - step;

    if (goalIdx >= 0 && covered[goalIdx] !== 0) goalIdx = chooseGoal(current);

    // Segment depth cap keeps BFS bounded and scalable.
    const depth = stepsLeft < 60 ? stepsLeft : 60;
    const target = chooseTarget(current, depth, goalIdx);

    if (target < 0 || target === current) {
      const next = bestNeighborStep(current, goalIdx);
      current = next;
      path[step++] = [current % W, (current / W) | 0];
      if (val[current] > 0) covered[current] = 1;
      continue;
    }

    const segLen = collectPath(target, current);
    if (segLen <= 0) {
      const next = bestNeighborStep(current, goalIdx);
      current = next;
      path[step++] = [current % W, (current / W) | 0];
      if (val[current] > 0) covered[current] = 1;
      continue;
    }

    for (let j = segLen - 1; j >= 0 && step < pathLenTarget; j--) {
      current = tmpPath[j];
      path[step++] = [current % W, (current / W) | 0];
      if (val[current] > 0) covered[current] = 1;
    }
  }

  paths[i] = path;
}

process.stdout.write(JSON.stringify({ paths }));
