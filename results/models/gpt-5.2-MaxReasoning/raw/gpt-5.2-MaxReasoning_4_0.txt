// 3D Regular Tetrahedron Packing into a convex polyhedron (convex hull of input vertices).
// Strategy:
//  - Build a convex hull (incremental) from the input vertices.
//  - Pack non-overlapping tetrahedra by packing their circumscribed spheres on an FCC lattice.
//    (If spheres don't overlap and spheres stay inside the container, tetrahedra are guaranteed valid.)
//  - Adapt output size to the estimated max count from container volume, with a hard cap for I/O/runtime.
//
// Notes:
//  - Input provides only vertices; this solver assumes the container is the convex hull of those vertices.
//  - Euler angles are output as (0,0,0) since orientation is fixed; translation-only packing.

#include <algorithm>
#include <array>
#include <cmath>
#include <cstdint>
#include <iomanip>
#include <iostream>
#include <limits>
#include <random>
#include <unordered_map>
#include <utility>
#include <vector>

struct Vec3 {
  double x = 0, y = 0, z = 0;
  Vec3() = default;
  Vec3(double x_, double y_, double z_) : x(x_), y(y_), z(z_) {}
  Vec3 operator+(const Vec3& o) const { return Vec3{x + o.x, y + o.y, z + o.z}; }
  Vec3 operator-(const Vec3& o) const { return Vec3{x - o.x, y - o.y, z - o.z}; }
  Vec3 operator*(double s) const { return Vec3{x * s, y * s, z * s}; }
  Vec3 operator/(double s) const { return Vec3{x / s, y / s, z / s}; }
  Vec3& operator+=(const Vec3& o) {
    x += o.x;
    y += o.y;
    z += o.z;
    return *this;
  }
};

static inline double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static inline Vec3 cross(const Vec3& a, const Vec3& b) {
  return Vec3{a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}
static inline double norm2(const Vec3& v) { return dot(v, v); }
static inline double norm(const Vec3& v) { return std::sqrt(norm2(v)); }
static inline Vec3 normalize(const Vec3& v) {
  double n = norm(v);
  if (n <= 0) return Vec3{0, 0, 0};
  return v / n;
}

struct Face {
  int a = -1, b = -1, c = -1;
  Vec3 n;     // outward unit normal
  double d = 0;  // plane offset: n·x <= d for interior
  bool dead = false;
};

static Face make_face(int a, int b, int c, const std::vector<Vec3>& pts, const Vec3& inside_point, double eps) {
  Face f;
  f.a = a;
  f.b = b;
  f.c = c;
  const Vec3 pa = pts[a], pb = pts[b], pc = pts[c];
  Vec3 nn = cross(pb - pa, pc - pa);
  double len = norm(nn);
  if (len <= eps) {
    f.n = Vec3{0, 0, 0};
    f.d = 0;
    return f;
  }
  f.n = nn / len;
  f.d = dot(f.n, pa);
  // Ensure inside_point satisfies n·x <= d (interior side).
  if (dot(f.n, inside_point) > f.d + eps) {
    std::swap(f.b, f.c);
    f.n = f.n * -1.0;
    f.d = dot(f.n, pts[f.a]);
  }
  return f;
}

static inline double signed_plane_distance(const Face& f, const Vec3& p) { return dot(f.n, p) - f.d; }

// Incremental 3D convex hull for points in general position (triangulated surface).
// Returns empty on degeneracy.
static std::vector<Face> convex_hull_3d(const std::vector<Vec3>& pts, double eps) {
  const int n = static_cast<int>(pts.size());
  if (n < 4) return {};

  // Find an initial non-degenerate tetrahedron.
  int i0 = 0, i1 = 0;
  for (int i = 1; i < n; ++i) {
    if (pts[i].x < pts[i0].x) i0 = i;
    if (pts[i].x > pts[i1].x) i1 = i;
  }
  if (i0 == i1) return {};

  Vec3 dir = pts[i1] - pts[i0];
  double best_line = -1.0;
  int i2 = -1;
  for (int i = 0; i < n; ++i) {
    if (i == i0 || i == i1) continue;
    double v = norm2(cross(dir, pts[i] - pts[i0]));
    if (v > best_line) {
      best_line = v;
      i2 = i;
    }
  }
  if (i2 < 0 || best_line <= eps * eps) return {};

  Vec3 nrm = cross(pts[i1] - pts[i0], pts[i2] - pts[i0]);
  double best_plane = -1.0;
  int i3 = -1;
  for (int i = 0; i < n; ++i) {
    if (i == i0 || i == i1 || i == i2) continue;
    double v = std::fabs(dot(nrm, pts[i] - pts[i0]));
    if (v > best_plane) {
      best_plane = v;
      i3 = i;
    }
  }
  if (i3 < 0 || best_plane <= eps) return {};

  Vec3 inside_point = (pts[i0] + pts[i1] + pts[i2] + pts[i3]) / 4.0;

  std::vector<Face> faces;
  faces.reserve(8 * n);
  faces.push_back(make_face(i0, i1, i2, pts, inside_point, eps));
  faces.push_back(make_face(i0, i3, i1, pts, inside_point, eps));
  faces.push_back(make_face(i1, i3, i2, pts, inside_point, eps));
  faces.push_back(make_face(i2, i3, i0, pts, inside_point, eps));

  // Randomized incremental insertion to reduce worst-case behavior.
  std::vector<int> order;
  order.reserve(n);
  for (int i = 0; i < n; ++i) {
    if (i == i0 || i == i1 || i == i2 || i == i3) continue;
    order.push_back(i);
  }
  std::mt19937_64 rng(1337);
  std::shuffle(order.begin(), order.end(), rng);

  auto edge_key = [](int u, int v) -> std::uint64_t {
    return (static_cast<std::uint64_t>(static_cast<std::uint32_t>(u)) << 32) |
           static_cast<std::uint32_t>(v);
  };

  for (int idx : order) {
    const Vec3 p = pts[idx];
    bool outside = false;
    for (const Face& f : faces) {
      if (f.dead) continue;
      if (signed_plane_distance(f, p) > eps) {
        outside = true;
        break;
      }
    }
    if (!outside) continue;

    std::unordered_map<std::uint64_t, std::pair<int, int>> boundary_edges;
    boundary_edges.reserve(128);

    for (Face& f : faces) {
      if (f.dead) continue;
      if (signed_plane_distance(f, p) > eps) {
        f.dead = true;
        auto add_edge = [&](int u, int v) {
          std::uint64_t rev = edge_key(v, u);
          auto it = boundary_edges.find(rev);
          if (it != boundary_edges.end()) {
            boundary_edges.erase(it);
          } else {
            boundary_edges.emplace(edge_key(u, v), std::make_pair(u, v));
          }
        };
        add_edge(f.a, f.b);
        add_edge(f.b, f.c);
        add_edge(f.c, f.a);
      }
    }

    // Cone over the horizon edges.
    for (const auto& kv : boundary_edges) {
      const int u = kv.second.first;
      const int v = kv.second.second;
      Face nf = make_face(u, v, idx, pts, inside_point, eps);
      // Skip degenerate faces (can happen with near-coplanar inputs).
      if (norm2(nf.n) <= 0) continue;
      faces.push_back(nf);
    }
  }

  // Compact alive faces.
  std::vector<Face> hull;
  hull.reserve(faces.size());
  for (const Face& f : faces) {
    if (!f.dead && norm2(f.n) > 0) hull.push_back(f);
  }
  return hull;
}

static double polyhedron_volume(const std::vector<Face>& faces, const std::vector<Vec3>& pts, const Vec3& ref) {
  long double acc = 0.0L;
  for (const Face& f : faces) {
    const Vec3 a = pts[f.a] - ref;
    const Vec3 b = pts[f.b] - ref;
    const Vec3 c = pts[f.c] - ref;
    const Vec3 cr = cross(b, c);
    acc += static_cast<long double>(dot(a, cr));
  }
  long double v = std::fabs(acc) / 6.0L;
  return static_cast<double>(v);
}

static bool contains_sphere_strict(const std::vector<Face>& faces, const Vec3& c, double r, double margin) {
  // Strict containment: require a small negative margin to avoid numerical false-positives.
  // For all planes: n·c + r <= d - margin
  for (const Face& f : faces) {
    if (dot(f.n, c) > f.d - r - margin) return false;
  }
  return true;
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int V = 0;
  double edge = 0.0;
  if (!(std::cin >> V >> edge)) {
    return 0;
  }

  std::vector<Vec3> pts;
  pts.reserve(std::max(0, V));
  for (int i = 0; i < V; ++i) {
    double x, y, z;
    std::cin >> x >> y >> z;
    pts.emplace_back(x, y, z);
  }

  if (pts.size() < 4 || !(edge > 0.0)) {
    std::cout << 0 << "\n";
    return 0;
  }

  // Compute scale-based epsilon for robustness.
  Vec3 mn = pts[0], mx = pts[0];
  for (const Vec3& p : pts) {
    mn.x = std::min(mn.x, p.x);
    mn.y = std::min(mn.y, p.y);
    mn.z = std::min(mn.z, p.z);
    mx.x = std::max(mx.x, p.x);
    mx.y = std::max(mx.y, p.y);
    mx.z = std::max(mx.z, p.z);
  }
  const double scale = std::max({mx.x - mn.x, mx.y - mn.y, mx.z - mn.z, 1.0});
  const double eps_hull = 1e-10 * scale;
  const double eps_contain = 1e-9 * std::max(1.0, edge) + 1e-12 * scale;

  std::vector<Face> faces = convex_hull_3d(pts, eps_hull);
  if (faces.empty()) {
    std::cout << 0 << "\n";
    return 0;
  }

  // Estimate container volume (convex hull).
  Vec3 ref{0, 0, 0};
  for (const Vec3& p : pts) ref += p;
  ref = ref / static_cast<double>(pts.size());
  const double vol_container = polyhedron_volume(faces, pts, ref);

  // Regular tetrahedron volume and circumsphere radius.
  const double tet_volume = (edge * edge * edge) / (6.0 * std::sqrt(2.0));
  const double R = edge * std::sqrt(3.0 / 8.0);

  // Adaptive output cap: never exceed hard cap to keep I/O bounded.
  const std::size_t hard_cap = 200000;
  std::size_t cap = 0;
  if (tet_volume > 0) {
    double upper = vol_container / tet_volume;
    if (upper < 0) upper = 0;
    if (upper > static_cast<double>(hard_cap)) cap = hard_cap;
    else cap = static_cast<std::size_t>(upper + 1.0);  // +1 to avoid truncation to 0
  }
  if (cap == 0) {
    std::cout << 0 << "\n";
    return 0;
  }

  // FCC lattice for sphere centers (dense packing for spheres).
  const double neighbor = (2.0 * R) * (1.0 + 1e-6);  // slight separation margin
  const double a = neighbor * std::sqrt(2.0);        // conventional FCC cube side
  const std::array<Vec3, 4> basis = {Vec3{0, 0, 0}, Vec3{0, 0.5 * a, 0.5 * a}, Vec3{0.5 * a, 0, 0.5 * a},
                                     Vec3{0.5 * a, 0.5 * a, 0}};

  // Tighten search bounds using the AABB, expanded slightly for basis offsets.
  const double pad = a;
  const int ix0 = static_cast<int>(std::floor((mn.x - pad) / a));
  const int iy0 = static_cast<int>(std::floor((mn.y - pad) / a));
  const int iz0 = static_cast<int>(std::floor((mn.z - pad) / a));
  const int ix1 = static_cast<int>(std::ceil((mx.x + pad) / a));
  const int iy1 = static_cast<int>(std::ceil((mx.y + pad) / a));
  const int iz1 = static_cast<int>(std::ceil((mx.z + pad) / a));

  std::vector<Vec3> centers;
  centers.reserve(cap);

  // Simple scan; stops early once cap is reached.
  for (int ix = ix0; ix <= ix1 && centers.size() < cap; ++ix) {
    for (int iy = iy0; iy <= iy1 && centers.size() < cap; ++iy) {
      for (int iz = iz0; iz <= iz1 && centers.size() < cap; ++iz) {
        const Vec3 base{ix * a, iy * a, iz * a};
        for (const Vec3& b : basis) {
          if (centers.size() >= cap) break;
          const Vec3 c = base + b;
          if (contains_sphere_strict(faces, c, R, eps_contain)) {
            centers.push_back(c);
          }
        }
      }
    }
  }

  // Output: fixed orientation (Euler angles 0,0,0).
  std::cout << centers.size() << "\n";
  std::cout.setf(std::ios::fixed);
  std::cout << std::setprecision(10);
  for (const Vec3& c : centers) {
    std::cout << c.x << " " << c.y << " " << c.z << " 0 0 0\n";
  }
  return 0;
}
