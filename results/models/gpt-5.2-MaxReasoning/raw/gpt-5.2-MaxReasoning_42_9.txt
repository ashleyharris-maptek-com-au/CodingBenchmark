; SPIR-V
; Version: 1.0
; Geometry shader: thick line strip using adjacency

OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%uint_4 = OpConstant %uint 4

%thickness = OpConstant %float 0.025

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_Input_v4float = OpTypePointer Input %v4float

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v3float = OpTypePointer Input %v3float

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv_in Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input

%gl_out = OpVariable %ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

%pos0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%pos0 = OpLoad %v4float %pos0_ptr
%pos1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%pos1 = OpLoad %v4float %pos1_ptr
%pos2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%pos2 = OpLoad %v4float %pos2_ptr
%pos3_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%pos3 = OpLoad %v4float %pos3_ptr

%pos0_xy = OpVectorShuffle %v2float %pos0 %pos0 0 1
%pos1_xy = OpVectorShuffle %v2float %pos1 %pos1 0 1
%pos2_xy = OpVectorShuffle %v2float %pos2 %pos2 0 1
%pos3_xy = OpVectorShuffle %v2float %pos3 %pos3 0 1

%d0 = OpFSub %v2float %pos2_xy %pos0_xy
%t0 = OpExtInst %v2float %glsl Normalize %d0

%d1 = OpFSub %v2float %pos3_xy %pos1_xy
%t1 = OpExtInst %v2float %glsl Normalize %d1

%t0_x = OpCompositeExtract %float %t0 0
%t0_y = OpCompositeExtract %float %t0 1
%neg_t0_y = OpFNegate %float %t0_y
%perp0 = OpCompositeConstruct %v2float %neg_t0_y %t0_x

%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%perp1 = OpCompositeConstruct %v2float %neg_t1_y %t1_x

%off0 = OpVectorTimesScalar %v2float %perp0 %thickness
%off1 = OpVectorTimesScalar %v2float %perp1 %thickness

%start_zw = OpVectorShuffle %v2float %pos1 %pos1 2 3
%end_zw = OpVectorShuffle %v2float %pos2 %pos2 2 3

%start_plus_xy = OpFAdd %v2float %pos1_xy %off0
%start_minus_xy = OpFSub %v2float %pos1_xy %off0
%end_plus_xy = OpFAdd %v2float %pos2_xy %off1
%end_minus_xy = OpFSub %v2float %pos2_xy %off1

%start_plus_pos = OpVectorShuffle %v4float %start_plus_xy %start_zw 0 1 2 3
%start_minus_pos = OpVectorShuffle %v4float %start_minus_xy %start_zw 0 1 2 3
%end_plus_pos = OpVectorShuffle %v4float %end_plus_xy %end_zw 0 1 2 3
%end_minus_pos = OpVectorShuffle %v4float %end_minus_xy %end_zw 0 1 2 3

%col1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr

%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

OpStore %outColor %col1
OpStore %gl_out_pos_ptr %start_plus_pos
OpEmitVertex

OpStore %outColor %col1
OpStore %gl_out_pos_ptr %start_minus_pos
OpEmitVertex

OpStore %outColor %col2
OpStore %gl_out_pos_ptr %end_plus_pos
OpEmitVertex

OpStore %outColor %col2
OpStore %gl_out_pos_ptr %end_minus_pos
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
