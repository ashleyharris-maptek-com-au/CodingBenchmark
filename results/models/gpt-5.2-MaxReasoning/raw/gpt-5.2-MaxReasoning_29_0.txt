using System;
using System.Buffers;
using System.Diagnostics;
using System.IO;
using System.Numerics;
using System.Text;

public static class Program
{
    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        const int TotalTimeLimitMs = 290_000; // keep a little buffer under 5 minutes

        var scanner = new FastScanner(Console.OpenStandardInput());
        int n;
        try
        {
            n = scanner.NextInt();
        }
        catch
        {
            return;
        }

        long mLong = scanner.NextLong();
        if (n <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        if (mLong < 0) mLong = 0;
        if (mLong > int.MaxValue) mLong = int.MaxValue; // cannot allocate more anyway
        int mInput = (int)mLong;

        int[] degrees = new int[n];
        int[] eu = new int[mInput];
        int[] ev = new int[mInput];
        int m = 0;

        for (int i = 0; i < mInput; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if ((uint)u >= (uint)n || (uint)v >= (uint)n) continue;
            if (u == v) continue;
            eu[m] = u;
            ev[m] = v;
            m++;
            degrees[u]++;
            degrees[v]++;
        }

        // Build CSR adjacency
        long totalAdjLong = 0;
        int[] offsets = new int[n + 1];
        for (int v = 0; v < n; v++)
        {
            offsets[v] = (int)totalAdjLong;
            totalAdjLong += degrees[v];
            if (totalAdjLong > int.MaxValue)
            {
                // Too large to allocate; fall back to trivial solution.
                Console.WriteLine("1");
                Console.WriteLine("0");
                return;
            }
        }
        offsets[n] = (int)totalAdjLong;
        int totalAdj = (int)totalAdjLong;
        int[] adj = new int[totalAdj];
        int[] cursor = (int[])offsets.Clone();

        for (int i = 0; i < m; i++)
        {
            int u = eu[i];
            int v = ev[i];
            adj[cursor[u]++] = v;
            adj[cursor[v]++] = u;
        }

        // Sort neighbor lists in-place (needed for fast intersections / binary searches)
        for (int v = 0; v < n; v++)
        {
            int start = offsets[v];
            int len = offsets[v + 1] - start;
            if (len > 1) Array.Sort(adj, start, len);
        }

        eu = Array.Empty<int>();
        ev = Array.Empty<int>();

        int maxDegree = 0;
        for (int i = 0; i < n; i++) if (degrees[i] > maxDegree) maxDegree = degrees[i];

        // Choose vertices by degree for seeding heuristics and subset selection.
        int topCount = Math.Min(n, Math.Max(4096, Math.Min(12000, n)));
        int[] topVertices = GetTopByDegree(degrees, topCount);

        // Fast greedy multi-start heuristic on full graph.
        int[] bestClique = HeuristicClique(degrees, offsets, adj, topVertices, sw, TotalTimeLimitMs);
        if (bestClique.Length == maxDegree + 1)
        {
            Array.Sort(bestClique);
            WriteClique(bestClique);
            return;
        }

        // Optional exact search (full for small graphs, or exact-on-induced-subgraph for large graphs).
        int[] finalClique = bestClique;
        int timeLeft = TotalTimeLimitMs - (int)sw.ElapsedMilliseconds;
        if (timeLeft > 3000)
        {
            int subsetLimit = n <= 7000 ? n : 6000;
            int[] subset = n <= subsetLimit ? Range(n) : BuildSubset(bestClique, degrees, offsets, adj, topVertices, subsetLimit);
            if (subset.Length > 0)
            {
                int exactBudget = Math.Min(timeLeft - 1500, n <= 7000 ? 180_000 : 60_000);
                if (exactBudget > 1500)
                {
                    int[] exactClique = ExactOnSubset(subset, bestClique, n, degrees, offsets, adj, sw, exactBudget);
                    if (exactClique.Length > finalClique.Length) finalClique = exactClique;
                }
            }
        }

        finalClique = ValidateCliqueGreedy(finalClique, offsets, adj);
        Array.Sort(finalClique);
        WriteClique(finalClique);
    }

    private static void WriteClique(int[] clique)
    {
        using var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 16);
        output.WriteLine(clique.Length);
        if (clique.Length > 0)
        {
            for (int i = 0; i < clique.Length; i++)
            {
                if (i != 0) output.Write(' ');
                output.Write(clique[i]);
            }
        }
        output.WriteLine();
    }

    private static int[] Range(int n)
    {
        int[] arr = new int[n];
        for (int i = 0; i < n; i++) arr[i] = i;
        return arr;
    }

    private static int[] GetTopByDegree(int[] degrees, int count)
    {
        int n = degrees.Length;
        count = Math.Min(count, n);
        long[] keys = new long[n];
        for (int v = 0; v < n; v++)
        {
            keys[v] = ((long)degrees[v] << 32) | (uint)v;
        }
        Array.Sort(keys); // ascending
        int[] top = new int[count];
        for (int i = 0; i < count; i++)
        {
            top[i] = (int)(keys[n - 1 - i] & 0xffffffff);
        }
        return top;
    }

    private static int[] HeuristicClique(int[] degrees, int[] offsets, int[] adj, int[] topVertices, Stopwatch sw, int totalLimitMs)
    {
        int n = degrees.Length;
        int seedCount = Math.Min(topVertices.Length, n <= 20000 ? Math.Min(n, 8000) : 3000);
        int randomCount = n <= 200000 ? 512 : 256;
        int totalSeeds = seedCount + randomCount;
        int[] seeds = new int[totalSeeds];
        Array.Copy(topVertices, seeds, seedCount);

        var rng = new XorShift64Star(0x9e3779b97f4a7c15UL);
        for (int i = 0; i < randomCount; i++)
        {
            seeds[seedCount + i] = (int)(rng.NextUInt() % (uint)n);
        }

        int maxSeedDeg = 0;
        for (int i = 0; i < totalSeeds; i++)
        {
            int d = degrees[seeds[i]];
            if (d > maxSeedDeg) maxSeedDeg = d;
        }

        int[] cand = ArrayPool<int>.Shared.Rent(Math.Max(1, maxSeedDeg));
        int[] tmp = ArrayPool<int>.Shared.Rent(Math.Max(1, maxSeedDeg));
        int[] cliqueBuf = ArrayPool<int>.Shared.Rent(Math.Max(2, maxSeedDeg + 1));

        int bestSize = 0;
        int[] best = Array.Empty<int>();

        try
        {
            for (int si = 0; si < totalSeeds; si++)
            {
                if (sw.ElapsedMilliseconds > totalLimitMs - 10_000) break;

                int seed = seeds[si];
                int cliqueSize = GreedyCliqueFromSeed(seed, degrees, offsets, adj, cand, tmp, cliqueBuf);
                if (cliqueSize > bestSize)
                {
                    bestSize = cliqueSize;
                    best = new int[bestSize];
                    Array.Copy(cliqueBuf, 0, best, 0, bestSize);
                    if (bestSize == degrees[seed] + 1) { /* already maximum possible for this seed */ }
                }
            }
        }
        finally
        {
            ArrayPool<int>.Shared.Return(cand);
            ArrayPool<int>.Shared.Return(tmp);
            ArrayPool<int>.Shared.Return(cliqueBuf);
        }

        return bestSize == 0 ? Array.Empty<int>() : best;
    }

    private static int GreedyCliqueFromSeed(
        int seed,
        int[] degrees,
        int[] offsets,
        int[] adj,
        int[] cand,
        int[] tmp,
        int[] clique)
    {
        int cliqueSize = 0;
        clique[cliqueSize++] = seed;

        int candCount = CopyUniqueNeighbors(seed, offsets, adj, cand);
        if (candCount == 0) return cliqueSize;

        while (candCount > 0)
        {
            // Pick the highest-degree vertex among candidates.
            int bestV = cand[0];
            int bestD = degrees[bestV];
            for (int i = 1; i < candCount; i++)
            {
                int v = cand[i];
                int d = degrees[v];
                if (d > bestD)
                {
                    bestD = d;
                    bestV = v;
                }
            }

            clique[cliqueSize++] = bestV;
            candCount = IntersectCandWithNeighbors(cand, candCount, bestV, offsets, adj, tmp);

            // swap cand/tmp buffers
            var swap = cand;
            cand = tmp;
            tmp = swap;
        }

        return cliqueSize;
    }

    private static int CopyUniqueNeighbors(int v, int[] offsets, int[] adj, int[] dest)
    {
        int start = offsets[v];
        int end = offsets[v + 1];
        int count = 0;
        int last = int.MinValue;
        for (int i = start; i < end; i++)
        {
            int nb = adj[i];
            if (nb != last)
            {
                dest[count++] = nb;
                last = nb;
            }
        }
        return count;
    }

    private static int IntersectCandWithNeighbors(int[] cand, int candCount, int v, int[] offsets, int[] adj, int[] output)
    {
        int a = 0;
        int b = offsets[v];
        int bEnd = offsets[v + 1];
        int outCount = 0;
        int last = int.MinValue;

        while (a < candCount && b < bEnd)
        {
            int va = cand[a];
            int vb = adj[b];
            if (va == vb)
            {
                if (va != last)
                {
                    output[outCount++] = va;
                    last = va;
                }
                a++;
                b++;
                while (a < candCount && cand[a] == last) a++;
                while (b < bEnd && adj[b] == last) b++;
            }
            else if (va < vb)
            {
                a++;
            }
            else
            {
                b++;
            }
        }
        return outCount;
    }

    private static int[] BuildSubset(int[] bestClique, int[] degrees, int[] offsets, int[] adj, int[] topVertices, int limit)
    {
        int n = degrees.Length;
        if (limit <= 0) return Array.Empty<int>();

        int[] clique = bestClique.Length == 0 ? Array.Empty<int>() : (int[])bestClique.Clone();
        Array.Sort(clique);
        int cliqueUnique = 0;
        for (int i = 0; i < clique.Length; i++)
        {
            if (i == 0 || clique[i] != clique[i - 1]) clique[cliqueUnique++] = clique[i];
        }
        if (cliqueUnique != clique.Length) Array.Resize(ref clique, cliqueUnique);

        int[] seen = new int[n];
        int stamp = 1;

        var subset = new int[Math.Min(n, Math.Max(limit * 3, 1024))];
        int subsetCount = 0;

        void Add(int v)
        {
            if (seen[v] == stamp) return;
            seen[v] = stamp;
            if (subsetCount == subset.Length)
            {
                Array.Resize(ref subset, Math.Min(n, subset.Length + Math.Max(1024, subset.Length / 2)));
            }
            subset[subsetCount++] = v;
        }

        for (int i = 0; i < clique.Length; i++) Add(clique[i]);

        for (int i = 0; i < clique.Length; i++)
        {
            int v = clique[i];
            for (int e = offsets[v]; e < offsets[v + 1]; e++) Add(adj[e]);
        }

        // Prefer high-degree vertices too.
        for (int i = 0; i < topVertices.Length && subsetCount < limit * 3; i++) Add(topVertices[i]);

        if (subsetCount <= limit)
        {
            int[] res = new int[subsetCount];
            Array.Copy(subset, 0, res, 0, subsetCount);
            return res;
        }

        // Keep all clique vertices, then fill the rest by degree.
        int cliqueCount = clique.Length;
        int keepRest = Math.Max(0, limit - cliqueCount);

        long[] restKeys = new long[subsetCount - cliqueCount];
        int rk = 0;
        for (int i = 0; i < subsetCount; i++)
        {
            int v = subset[i];
            if (Array.BinarySearch(clique, v) >= 0) continue;
            restKeys[rk++] = ((long)degrees[v] << 32) | (uint)v;
        }
        Array.Sort(restKeys, 0, rk);

        int outCount = Math.Min(limit, cliqueCount + keepRest);
        int[] outVerts = new int[outCount];
        int oi = 0;
        for (int i = 0; i < cliqueCount && oi < outCount; i++) outVerts[oi++] = clique[i];
        for (int i = 0; i < keepRest && oi < outCount; i++)
        {
            int v = (int)(restKeys[rk - 1 - i] & 0xffffffff);
            outVerts[oi++] = v;
        }
        if (oi < outCount)
        {
            Array.Resize(ref outVerts, oi);
        }
        return outVerts;
    }

    private static int[] ExactOnSubset(
        int[] subsetVertices,
        int[] initialCliqueOriginal,
        int originalN,
        int[] originalDegrees,
        int[] offsets,
        int[] adj,
        Stopwatch sw,
        int budgetMs)
    {
        int k = subsetVertices.Length;
        if (k <= 1) return k == 1 ? new[] { subsetVertices[0] } : Array.Empty<int>();

        int[] map = new int[originalN];
        Array.Fill(map, -1);
        for (int i = 0; i < k; i++) map[subsetVertices[i]] = i;

        int words = (k + 63) >> 6;
        ulong[] adjFlat = new ulong[k * words];

        for (int i = 0; i < k; i++)
        {
            int v = subsetVertices[i];
            var row = adjFlat.AsSpan(i * words, words);
            for (int e = offsets[v]; e < offsets[v + 1]; e++)
            {
                int nb = adj[e];
                int j = (uint)nb < (uint)originalN ? map[nb] : -1;
                if (j >= 0)
                {
                    row[j >> 6] |= 1UL << (j & 63);
                }
            }
        }

        int[] initCliqueIdx = Array.Empty<int>();
        if (initialCliqueOriginal.Length > 0)
        {
            int[] tmp = new int[initialCliqueOriginal.Length];
            int cnt = 0;
            for (int i = 0; i < initialCliqueOriginal.Length; i++)
            {
                int idx = map[initialCliqueOriginal[i]];
                if (idx >= 0) tmp[cnt++] = idx;
            }
            if (cnt > 0)
            {
                initCliqueIdx = new int[cnt];
                Array.Copy(tmp, initCliqueIdx, cnt);
            }
        }

        var solver = new MaxCliqueBitsetSolver(k, words, adjFlat, sw, budgetMs, initCliqueIdx);
        int[] bestIdx = solver.Solve();
        int[] best = new int[bestIdx.Length];
        for (int i = 0; i < bestIdx.Length; i++) best[i] = subsetVertices[bestIdx[i]];
        return best;
    }

    private static int[] ValidateCliqueGreedy(int[] clique, int[] offsets, int[] adj)
    {
        if (clique.Length <= 1) return clique;

        // Remove duplicates first.
        int[] tmp = (int[])clique.Clone();
        Array.Sort(tmp);
        int unique = 0;
        for (int i = 0; i < tmp.Length; i++)
        {
            if (i == 0 || tmp[i] != tmp[i - 1]) tmp[unique++] = tmp[i];
        }

        int[] outClique = new int[unique];
        int outCount = 0;
        for (int i = 0; i < unique; i++)
        {
            int v = tmp[i];
            bool ok = true;
            for (int j = 0; j < outCount; j++)
            {
                if (!HasEdge(v, outClique[j], offsets, adj))
                {
                    ok = false;
                    break;
                }
            }
            if (ok) outClique[outCount++] = v;
        }

        if (outCount == outClique.Length) return outClique;
        Array.Resize(ref outClique, outCount);
        return outClique;
    }

    private static bool HasEdge(int u, int v, int[] offsets, int[] adj)
    {
        int start = offsets[u];
        int len = offsets[u + 1] - start;
        return Array.BinarySearch(adj, start, len, v) >= 0;
    }

    private sealed class MaxCliqueBitsetSolver
    {
        private readonly int _n;
        private readonly int _words;
        private readonly ulong[] _adj; // flattened n x words adjacency bitsets
        private readonly Stopwatch _sw;
        private readonly int _deadlineMs;

        private readonly ulong[] _pStack; // (n+1) * words
        private readonly ulong[] _tmpU;
        private readonly ulong[] _tmpQ;
        private readonly int[] _order;
        private readonly int[] _colors;

        private readonly int[] _current;
        private int[] _best;
        private int _bestSize;
        private bool _timedOut;

        public MaxCliqueBitsetSolver(int n, int words, ulong[] adjFlat, Stopwatch sw, int budgetMs, int[] initialClique)
        {
            _n = n;
            _words = words;
            _adj = adjFlat;
            _sw = sw;
            _deadlineMs = (int)sw.ElapsedMilliseconds + Math.Max(1, budgetMs);

            _pStack = new ulong[(n + 1) * words];
            _tmpU = new ulong[words];
            _tmpQ = new ulong[words];
            _order = new int[n];
            _colors = new int[n];
            _current = new int[n];

            _bestSize = initialClique.Length;
            _best = initialClique.Length == 0 ? Array.Empty<int>() : (int[])initialClique.Clone();
        }

        public int[] Solve()
        {
            if (_n == 0) return Array.Empty<int>();

            // P0 = all vertices
            var p0 = P(0);
            for (int i = 0; i < _n; i++) SetBit(p0, i);

            Expand(0);
            return _best;
        }

        private void Expand(int depth)
        {
            if (_timedOut) return;
            if ((int)_sw.ElapsedMilliseconds >= _deadlineMs)
            {
                _timedOut = true;
                return;
            }

            var p = P(depth);
            if (IsEmpty(p))
            {
                if (depth > _bestSize)
                {
                    _bestSize = depth;
                    _best = new int[depth];
                    Array.Copy(_current, 0, _best, 0, depth);
                }
                return;
            }

            while (!IsEmpty(p))
            {
                if (_timedOut) return;
                if ((int)_sw.ElapsedMilliseconds >= _deadlineMs)
                {
                    _timedOut = true;
                    return;
                }

                int count = ColorSort(p);
                if (count <= 0) break;
                int bound = _colors[count - 1];
                if (depth + bound <= _bestSize) return;

                int v = _order[count - 1];
                _current[depth] = v;

                var pNext = P(depth + 1);
                AndTo(p, AdjRow(v), pNext);
                Expand(depth + 1);

                ClearBit(p, v);
            }

            if (IsEmpty(p) && depth > _bestSize)
            {
                _bestSize = depth;
                _best = new int[depth];
                Array.Copy(_current, 0, _best, 0, depth);
            }
        }

        private int ColorSort(ReadOnlySpan<ulong> p)
        {
            // U = p
            p.CopyTo(_tmpU);
            int count = 0;
            int color = 0;

            while (!IsEmpty(_tmpU))
            {
                color++;
                // Q = U
                Array.Copy(_tmpU, _tmpQ, _words);

                while (!IsEmpty(_tmpQ))
                {
                    int v = PopLeastBit(_tmpQ);
                    _order[count] = v;
                    _colors[count] = color;
                    count++;
                    ClearBit(_tmpU, v);
                    AndNotSelf(_tmpQ, AdjRow(v));
                }
            }

            return count;
        }

        private Span<ulong> P(int depth) => _pStack.AsSpan(depth * _words, _words);

        private ReadOnlySpan<ulong> AdjRow(int v) => _adj.AsSpan(v * _words, _words);

        private static bool IsEmpty(ReadOnlySpan<ulong> a)
        {
            for (int i = 0; i < a.Length; i++)
            {
                if (a[i] != 0) return false;
            }
            return true;
        }

        private static void AndTo(ReadOnlySpan<ulong> a, ReadOnlySpan<ulong> b, Span<ulong> dest)
        {
            for (int i = 0; i < dest.Length; i++) dest[i] = a[i] & b[i];
        }

        private static void AndNotSelf(Span<ulong> a, ReadOnlySpan<ulong> b)
        {
            for (int i = 0; i < a.Length; i++) a[i] &= ~b[i];
        }

        private static void SetBit(Span<ulong> a, int bit)
        {
            a[bit >> 6] |= 1UL << (bit & 63);
        }

        private static void ClearBit(Span<ulong> a, int bit)
        {
            a[bit >> 6] &= ~(1UL << (bit & 63));
        }

        private static void ClearBit(ulong[] a, int bit)
        {
            a[bit >> 6] &= ~(1UL << (bit & 63));
        }

        private static int PopLeastBit(Span<ulong> a)
        {
            for (int i = 0; i < a.Length; i++)
            {
                ulong w = a[i];
                if (w == 0) continue;
                int tz = BitOperations.TrailingZeroCount(w);
                int bit = (i << 6) + tz;
                a[i] = w & (w - 1);
                return bit;
            }
            return -1;
        }
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) throw new EndOfStreamException();
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = (val * 10) + (c - '0');
                c = Read();
                if (c < 0) break;
            }
            return val * sign;
        }

        public long NextLong()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return 0;
            } while (c <= 32);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long val = 0;
            while (c > 32)
            {
                val = (val * 10) + (c - '0');
                c = Read();
                if (c < 0) break;
            }
            return val * sign;
        }
    }

    private struct XorShift64Star
    {
        private ulong _x;
        public XorShift64Star(ulong seed) => _x = seed == 0 ? 0xdeadbeefcafebabeUL : seed;
        public uint NextUInt()
        {
            ulong x = _x;
            x ^= x >> 12;
            x ^= x << 25;
            x ^= x >> 27;
            _x = x;
            return (uint)((x * 2685821657736338717UL) >> 32);
        }
    }
