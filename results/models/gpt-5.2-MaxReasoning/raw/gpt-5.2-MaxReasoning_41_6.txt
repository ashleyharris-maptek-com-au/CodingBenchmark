; SPIR-V
; Version: 1.0
OpCapability Shader
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %outColor %ubo
OpExecutionMode %main OriginUpperLeft
OpSource GLSL 450

OpName %main "main"
OpName %inWorldPos "worldPos"
OpName %inNormal "normal"
OpName %outColor "outColor"
OpName %ubo "ubo"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"

OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%int = OpTypeInt 32 1
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

%inWorldPos = OpVariable %_ptr_Input_v3float Input
%inNormal = OpVariable %_ptr_Input_v3float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8
%float_64 = OpConstant %float 64

%main = OpFunction %void None %fn_void
%label = OpLabel
%worldPos = OpLoad %v3float %inWorldPos
%normal = OpLoad %v3float %inNormal
%N = OpExtInst %v3float %glsl Normalize %normal

%ptr_lightPos = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
%lightPos4 = OpLoad %v4float %ptr_lightPos
%lightPos3 = OpVectorShuffle %v3float %lightPos4 %lightPos4 0 1 2
%Ldir = OpFSub %v3float %lightPos3 %worldPos
%L = OpExtInst %v3float %glsl Normalize %Ldir

%ptr_cameraPos = OpAccessChain %_ptr_Uniform_v4float %ubo %int_4
%cameraPos4 = OpLoad %v4float %ptr_cameraPos
%cameraPos3 = OpVectorShuffle %v3float %cameraPos4 %cameraPos4 0 1 2
%Vdir = OpFSub %v3float %cameraPos3 %worldPos
%V = OpExtInst %v3float %glsl Normalize %Vdir

%LplusV = OpFAdd %v3float %L %V
%H = OpExtInst %v3float %glsl Normalize %LplusV

%ndotl = OpDot %float %N %L
%diffuse = OpExtInst %float %glsl FMax %ndotl %float_0

%ndoth = OpDot %float %N %H
%ndoth_clamped = OpExtInst %float %glsl FMax %ndoth %float_0
%specular = OpExtInst %float %glsl Pow %ndoth_clamped %float_64

%diffuseTerm = OpFMul %float %float_0_6 %diffuse
%specularTerm = OpFMul %float %float_0_8 %specular
%ambientPlusDiffuse = OpFAdd %float %float_0_1 %diffuseTerm
%intensity = OpFAdd %float %ambientPlusDiffuse %specularTerm

%outVec4 = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
OpStore %outColor %outVec4
OpReturn
OpFunctionEnd
