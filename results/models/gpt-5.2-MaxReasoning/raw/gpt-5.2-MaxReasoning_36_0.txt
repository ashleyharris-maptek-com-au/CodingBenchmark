using std::int64_t;
using std::uint64_t;

struct FastScanner {
  static constexpr size_t kBufSize = 1 << 20;
  size_t idx = 0;
  size_t size = 0;
  char buf[kBufSize];

  inline char read() {
    if (idx >= size) {
      size = std::fread(buf, 1, kBufSize, stdin);
      idx = 0;
      if (size == 0) return 0;
    }
    return buf[idx++];
  }

  template <class T>
  bool nextInt(T& out) {
    char c = 0;
    do {
      c = read();
      if (!c) return false;
    } while (c <= ' ');

    T sign = 1;
    if (c == '-') {
      sign = -1;
      c = read();
    }

    T val = 0;
    while (c > ' ') {
      val = val * 10 + (c - '0');
      c = read();
    }
    out = val * sign;
    return true;
  }
};

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
  return static_cast<int>(__popcnt64(x));
#else
  return __builtin_popcountll(x);
#endif
}

static inline int ctz64(uint64_t x) {
#if defined(_MSC_VER)
  unsigned long idx;
  _BitScanForward64(&idx, x);
  return static_cast<int>(idx);
#else
  return __builtin_ctzll(x);
#endif
}

struct HopcroftKarp {
  int L = 0, R = 0;
  const std::vector<std::vector<int>>* adj = nullptr;  // left -> right indices
  std::vector<int> pairU, pairV, dist;

  HopcroftKarp(int L_, int R_, const std::vector<std::vector<int>>& adj_)
      : L(L_), R(R_), adj(&adj_), pairU(L, -1), pairV(R, -1), dist(L, 0) {}

  bool bfs() {
    static constexpr int INF = std::numeric_limits<int>::max();
    std::queue<int> q;
    for (int u = 0; u < L; ++u) {
      if (pairU[u] == -1) {
        dist[u] = 0;
        q.push(u);
      } else {
        dist[u] = INF;
      }
    }

    int distNil = INF;
    while (!q.empty()) {
      int u = q.front();
      q.pop();
      if (dist[u] >= distNil) continue;
      for (int v : (*adj)[u]) {
        int u2 = pairV[v];
        if (u2 == -1) {
          distNil = dist[u] + 1;
        } else if (dist[u2] == INF) {
          dist[u2] = dist[u] + 1;
          q.push(u2);
        }
      }
    }
    return distNil != INF;
  }

  bool dfs(int u) {
    static constexpr int INF = std::numeric_limits<int>::max();
    for (int v : (*adj)[u]) {
      int u2 = pairV[v];
      if (u2 == -1 || (dist[u2] == dist[u] + 1 && dfs(u2))) {
        pairU[u] = v;
        pairV[v] = u;
        return true;
      }
    }
    dist[u] = INF;
    return false;
  }

  int maxMatching() {
    int matching = 0;
    while (bfs()) {
      for (int u = 0; u < L; ++u) {
        if (pairU[u] == -1 && dfs(u)) ++matching;
      }
    }
    return matching;
  }
};

struct MaxCliqueSolver {
  int n = 0;
  int words = 0;
  uint64_t lastMask = ~0ULL;
  const std::vector<uint64_t>* adj = nullptr;  // n * words, row-major

  int bestSize = 0;
  std::vector<int> bestClique;

  MaxCliqueSolver(int n_, int words_, uint64_t lastMask_, const std::vector<uint64_t>& adj_)
      : n(n_), words(words_), lastMask(lastMask_), adj(&adj_) {}

  inline bool any(const std::vector<uint64_t>& bits) const {
    for (int i = 0; i < words; ++i) {
      if (bits[i]) return true;
    }
    return false;
  }

  inline int count(const std::vector<uint64_t>& bits) const {
    int total = 0;
    for (int i = 0; i < words; ++i) total += popcount64(bits[i]);
    return total;
  }

  inline int firstSet(const std::vector<uint64_t>& bits) const {
    for (int wi = 0; wi < words; ++wi) {
      uint64_t w = bits[wi];
      if (w) return (wi << 6) + ctz64(w);
    }
    return -1;
  }

  inline void reset(std::vector<uint64_t>& bits, int v) const {
    bits[v >> 6] &= ~(1ULL << (v & 63));
  }

  inline void andWithAdj(std::vector<uint64_t>& bits, int v) const {
    const uint64_t* row = adj->data() + static_cast<size_t>(v) * static_cast<size_t>(words);
    for (int i = 0; i < words; ++i) bits[i] &= row[i];
  }

  inline void andNotAdj(std::vector<uint64_t>& bits, int v) const {
    const uint64_t* row = adj->data() + static_cast<size_t>(v) * static_cast<size_t>(words);
    for (int i = 0; i < words; ++i) bits[i] &= ~row[i];
  }

  void colourOrder(const std::vector<uint64_t>& P, std::vector<int>& U, std::vector<int>& col) {
    std::vector<uint64_t> copyP = P;
    std::vector<uint64_t> Q(words, 0);
    int colourClass = 0;
    int i = 0;
    while (any(copyP)) {
      ++colourClass;
      Q = copyP;
      while (any(Q)) {
        int v = firstSet(Q);
        reset(copyP, v);
        reset(Q, v);
        andNotAdj(Q, v);
        U[i] = v;
        col[i] = colourClass;
        ++i;
      }
    }
  }

  void expand(std::vector<uint64_t> P, std::vector<int>& curr) {
    int m = count(P);
    if (m == 0) {
      if (static_cast<int>(curr.size()) > bestSize) {
        bestSize = static_cast<int>(curr.size());
        bestClique = curr;
      }
      return;
    }

    std::vector<int> U(m), col(m);
    colourOrder(P, U, col);

    for (int i = m - 1; i >= 0; --i) {
      if (static_cast<int>(curr.size()) + col[i] <= bestSize) return;

      int v = U[i];
      curr.push_back(v);

      std::vector<uint64_t> newP = P;
      andWithAdj(newP, v);
      if (!any(newP)) {
        if (static_cast<int>(curr.size()) > bestSize) {
          bestSize = static_cast<int>(curr.size());
          bestClique = curr;
        }
      } else {
        expand(std::move(newP), curr);
      }

      curr.pop_back();
      reset(P, v);
    }
  }

  std::vector<int> solve() {
    std::vector<uint64_t> P(words, ~0ULL);
    if (words > 0) P.back() &= lastMask;
    std::vector<int> curr;
    curr.reserve(bestSize + 16);
    expand(std::move(P), curr);
    return bestClique;
  }
};

static void printSet(const std::vector<int>& set) {
  std::cout << set.size() << "\n";
  for (size_t i = 0; i < set.size(); ++i) {
    if (i) std::cout << ' ';
    std::cout << set[i];
  }
  std::cout << "\n";
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  FastScanner fs;
  int n = 0;
  int64_t m = 0;
  if (!fs.nextInt(n)) return 0;
  fs.nextInt(m);

  if (n <= 0) {
    std::cout << 0 << "\n\n";
    return 0;
  }

  std::vector<std::vector<int>> g(n);
  for (int64_t i = 0; i < m; ++i) {
    int u = 0, v = 0;
    fs.nextInt(u);
    fs.nextInt(v);
    if (u < 0 || v < 0 || u >= n || v >= n) continue;
    if (u == v) continue;
    g[u].push_back(v);
    g[v].push_back(u);
  }

  if (m == 0) {
    std::vector<int> all(n);
    std::iota(all.begin(), all.end(), 0);
    printSet(all);
    return 0;
  }

  // Bipartite check (also handles very large graphs efficiently when bipartite).
  std::vector<int8_t> side(n, -1);
  std::queue<int> q;
  bool isBip = true;
  for (int s = 0; s < n && isBip; ++s) {
    if (side[s] != -1) continue;
    side[s] = 0;
    q.push(s);
    while (!q.empty() && isBip) {
      int u = q.front();
      q.pop();
      for (int v : g[u]) {
        if (side[v] == -1) {
          side[v] = static_cast<int8_t>(side[u] ^ 1);
          q.push(v);
        } else if (side[v] == side[u]) {
          isBip = false;
          break;
        }
      }
    }
  }

  if (isBip) {
    std::vector<int> leftVerts, rightVerts;
    leftVerts.reserve(n);
    rightVerts.reserve(n);
    std::vector<int> leftId(n, -1), rightId(n, -1);
    for (int v = 0; v < n; ++v) {
      if (side[v] == 0) {
        leftId[v] = static_cast<int>(leftVerts.size());
        leftVerts.push_back(v);
      } else {
        rightId[v] = static_cast<int>(rightVerts.size());
        rightVerts.push_back(v);
      }
    }

    std::vector<std::vector<int>> adjL(leftVerts.size());
    for (size_t i = 0; i < leftVerts.size(); ++i) {
      int u = leftVerts[i];
      auto& out = adjL[i];
      out.reserve(g[u].size());
      for (int v : g[u]) {
        int rid = rightId[v];
        if (rid >= 0) out.push_back(rid);
      }
    }

    HopcroftKarp hk(static_cast<int>(leftVerts.size()), static_cast<int>(rightVerts.size()), adjL);
    hk.maxMatching();

    // Recover maximum independent set as complement of min vertex cover.
    std::vector<char> visL(leftVerts.size(), 0), visR(rightVerts.size(), 0);
    std::queue<int> qq;
    for (int u = 0; u < hk.L; ++u) {
      if (hk.pairU[u] == -1) {
        visL[u] = 1;
        qq.push(u);
      }
    }
    while (!qq.empty()) {
      int u = qq.front();
      qq.pop();
      for (int v : adjL[u]) {
        if (hk.pairU[u] == v) continue;  // matched edge; follow only unmatched L->R
        if (visR[v]) continue;
        visR[v] = 1;
        int u2 = hk.pairV[v];
        if (u2 != -1 && !visL[u2]) {
          visL[u2] = 1;
          qq.push(u2);
        }
      }
    }

    std::vector<int> indep;
    indep.reserve(n);
    for (int u = 0; u < hk.L; ++u) {
      if (visL[u]) indep.push_back(leftVerts[u]);
    }
    for (int v = 0; v < hk.R; ++v) {
      if (!visR[v]) indep.push_back(rightVerts[v]);
    }
    printSet(indep);
    return 0;
  }

  // General graph: solve exactly via maximum clique in the complement graph (bitset branch-and-bound).
  // Order vertices by increasing original degree (equivalently decreasing degree in the complement graph).
  std::vector<int> order(n);
  std::iota(order.begin(), order.end(), 0);
  std::sort(order.begin(), order.end(), [&](int a, int b) {
    size_t da = g[a].size(), db = g[b].size();
    if (da != db) return da < db;
    return a < b;
  });

  std::vector<int> origOfInternal(n, 0), internalOfOrig(n, 0);
  for (int i = 0; i < n; ++i) {
    origOfInternal[i] = order[i];
    internalOfOrig[order[i]] = i;
  }

  // Greedy independent set in original graph for an initial lower bound.
  std::vector<int> greedyOrig;
  greedyOrig.reserve(n);
  std::vector<char> banned(n, 0);
  for (int v : order) {
    if (banned[v]) continue;
    greedyOrig.push_back(v);
    banned[v] = 1;
    for (int u : g[v]) banned[u] = 1;
  }
  std::vector<int> greedyInternal;
  greedyInternal.reserve(greedyOrig.size());
  for (int v : greedyOrig) greedyInternal.push_back(internalOfOrig[v]);

  const int words = (n + 63) >> 6;
  const uint64_t lastMask = (n & 63) ? ((1ULL << (n & 63)) - 1ULL) : ~0ULL;

  std::vector<uint64_t> origAdj(static_cast<size_t>(n) * static_cast<size_t>(words), 0ULL);
  for (int uOrig = 0; uOrig < n; ++uOrig) {
    int u = internalOfOrig[uOrig];
    uint64_t* row = origAdj.data() + static_cast<size_t>(u) * static_cast<size_t>(words);
    for (int vOrig : g[uOrig]) {
      int v = internalOfOrig[vOrig];
      row[v >> 6] |= 1ULL << (v & 63);
    }
  }
  std::vector<std::vector<int>>().swap(g);

  std::vector<uint64_t> compAdj(static_cast<size_t>(n) * static_cast<size_t>(words), 0ULL);
  for (int v = 0; v < n; ++v) {
    const uint64_t* in = origAdj.data() + static_cast<size_t>(v) * static_cast<size_t>(words);
    uint64_t* out = compAdj.data() + static_cast<size_t>(v) * static_cast<size_t>(words);
    for (int w = 0; w < words; ++w) out[w] = ~in[w];
    if (words > 0) out[words - 1] &= lastMask;
    out[v >> 6] &= ~(1ULL << (v & 63));  // clear self
  }
  std::vector<uint64_t>().swap(origAdj);

  MaxCliqueSolver solver(n, words, lastMask, compAdj);
  solver.bestClique = greedyInternal;
  solver.bestSize = static_cast<int>(greedyInternal.size());
  std::vector<int> bestInternal = solver.solve();

  std::vector<int> bestOrig;
  bestOrig.reserve(bestInternal.size());
  for (int vInt : bestInternal) bestOrig.push_back(origOfInternal[vInt]);
  printSet(bestOrig);
  return 0;
}
