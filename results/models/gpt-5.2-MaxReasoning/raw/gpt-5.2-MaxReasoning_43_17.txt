struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float3 n = normalize(input.normal);
    float3 l = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(n, l), 0.0);
    float shade = 0.4 + 0.6 * diffuse;

    float2 scaledUV = input.uv * float2(8.0, 16.0);
    float row = floor(scaledUV.y);
    float parity = fmod(row, 2.0);
    float2 cell = float2(frac(scaledUV.x + parity * 0.5), frac(scaledUV.y));

    const float mortarThicknessX = 0.05;
    const float mortarThicknessY = 0.1;
    float2 edgeDist = min(cell, 1.0 - cell);
    float mortarX = (edgeDist.x < mortarThicknessX * 0.5) ? 1.0 : 0.0;
    float mortarY = (edgeDist.y < mortarThicknessY * 0.5) ? 1.0 : 0.0;
    float mortarMask = clamp(mortarX + mortarY, 0.0, 1.0);

    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 baseColor = lerp(brickColor, mortarColor, mortarMask);

    return float4(baseColor * shade, 1.0);
}
