use std::io::{self, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline(always)]
    fn new(x: f32, y: f32, z: f32) -> Self {
        Self { x, y, z }
    }
}

#[inline(always)]
fn dist2(a: &Point, b: &Point) -> f32 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

struct FastReader<R: Read> {
    inner: R,
    buf: Vec<u8>,
    pos: usize,
    len: usize,
}

impl<R: Read> FastReader<R> {
    fn new(inner: R) -> Self {
        Self {
            inner,
            buf: vec![0u8; 1 << 20],
            pos: 0,
            len: 0,
        }
    }

    #[inline(always)]
    fn refill(&mut self) -> io::Result<()> {
        self.len = self.inner.read(&mut self.buf)?;
        self.pos = 0;
        Ok(())
    }

    #[inline(always)]
    fn read_u8(&mut self) -> io::Result<u8> {
        if self.pos >= self.len {
            self.refill()?;
            if self.len == 0 {
                return Err(io::Error::new(io::ErrorKind::UnexpectedEof, "EOF"));
            }
        }
        let b = self.buf[self.pos];
        self.pos += 1;
        Ok(b)
    }

    fn read_ascii_u64(&mut self) -> io::Result<u64> {
        let mut b = self.read_u8()?;
        while b.is_ascii_whitespace() {
            b = self.read_u8()?;
        }
        if !b.is_ascii_digit() {
            return Err(io::Error::new(
                io::ErrorKind::InvalidData,
                "expected ASCII integer",
            ));
        }
        let mut v: u64 = 0;
        loop {
            v = v
                .checked_mul(10)
                .and_then(|x| x.checked_add((b - b'0') as u64))
                .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "integer overflow"))?;
            if self.pos >= self.len {
                self.refill()?;
                if self.len == 0 {
                    break;
                }
            }
            b = self.buf[self.pos];
            if !b.is_ascii_digit() {
                break;
            }
            self.pos += 1;
        }
        Ok(v)
    }

    // Consume the remainder of the current header line (up to and including '\n').
    // This matches the documented format: two ASCII integers, newline, then binary points.
    fn consume_until_newline(&mut self) -> io::Result<()> {
        for _ in 0..1024 {
            if self.pos >= self.len {
                self.refill()?;
                if self.len == 0 {
                    return Err(io::Error::new(
                        io::ErrorKind::UnexpectedEof,
                        "missing newline after header",
                    ));
                }
            }
            let b = self.buf[self.pos];
            self.pos += 1;
            if b == b'\n' {
                return Ok(());
            }
        }
        Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "header line too long or missing newline",
        ))
    }

    fn read_exact_into(&mut self, out: &mut [u8]) -> io::Result<()> {
        let mut off = 0usize;
        while off < out.len() {
            if self.pos >= self.len {
                self.refill()?;
                if self.len == 0 {
                    return Err(io::Error::new(io::ErrorKind::UnexpectedEof, "EOF"));
                }
            }
            let n = (self.len - self.pos).min(out.len() - off);
            out[off..off + n].copy_from_slice(&self.buf[self.pos..self.pos + n]);
            self.pos += n;
            off += n;
        }
        Ok(())
    }

    #[inline(always)]
    fn read_f32_le(&mut self) -> io::Result<f32> {
        let mut b = [0u8; 4];
        self.read_exact_into(&mut b)?;
        Ok(f32::from_le_bytes(b))
    }

    #[inline(always)]
    fn read_point(&mut self) -> io::Result<Point> {
        let x = self.read_f32_le()?;
        let y = self.read_f32_le()?;
        let z = self.read_f32_le()?;
        Ok(Point::new(x, y, z))
    }
}

struct SplitMix64 {
    state: u64,
}

impl SplitMix64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }

    #[inline(always)]
    fn next_u64(&mut self) -> u64 {
        let mut z = self.state.wrapping_add(0x9E3779B97F4A7C15);
        self.state = z;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }

    #[inline(always)]
    fn next_usize(&mut self, upper: usize) -> usize {
        if upper <= 1 {
            return 0;
        }
        (self.next_u64() % (upper as u64)) as usize
    }

    #[inline(always)]
    fn next_f64(&mut self) -> f64 {
        // 53 bits of precision
        let x = self.next_u64() >> 11;
        (x as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

#[inline(always)]
fn nearest_exact(p: &Point, centroids: &[Point]) -> usize {
    let mut best = 0usize;
    let mut best_d = f32::INFINITY;
    for (i, c) in centroids.iter().enumerate() {
        let d = dist2(p, c);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

#[inline(always)]
fn nearest_two(p: &Point, centroids: &[Point]) -> (usize, usize) {
    let mut b1 = 0usize;
    let mut b2 = 0usize;
    let mut d1 = f32::INFINITY;
    let mut d2 = f32::INFINITY;
    for (i, c) in centroids.iter().enumerate() {
        let d = dist2(p, c);
        if d < d1 {
            d2 = d1;
            b2 = b1;
            d1 = d;
            b1 = i;
        } else if d < d2 {
            d2 = d;
            b2 = i;
        }
    }
    (b1, b2)
}

struct Hierarchy {
    coarse: Vec<Point>,
    groups: Vec<Vec<usize>>,
}

fn build_hierarchy(
    fine: &[Point],
    fine_counts: Option<&[u64]>,
    coarse_count: usize,
    rng: &mut SplitMix64,
) -> Hierarchy {
    let k = fine.len();
    let coarse_count = coarse_count.clamp(1, k.max(1));
    let mut idx: Vec<usize> = (0..k).collect();
    // Partial Fisher-Yates shuffle to pick coarse seeds.
    for i in 0..coarse_count {
        let j = i + rng.next_usize(k - i);
        idx.swap(i, j);
    }
    let mut coarse: Vec<Point> = idx[..coarse_count].iter().map(|&i| fine[i]).collect();

    let mut groups: Vec<Vec<usize>> = vec![Vec::new(); coarse_count];

    // Two rounds: assign -> recompute coarse -> reassign -> recompute coarse.
    for _round in 0..2 {
        for g in groups.iter_mut() {
            g.clear();
        }
        for (fi, fc) in fine.iter().enumerate() {
            let ci = nearest_exact(fc, &coarse);
            groups[ci].push(fi);
        }

        // Ensure non-empty groups.
        for ci in 0..coarse_count {
            if !groups[ci].is_empty() {
                continue;
            }
            let mut donor = None;
            let mut donor_len = 0usize;
            for dj in 0..coarse_count {
                let l = groups[dj].len();
                if l > donor_len {
                    donor_len = l;
                    donor = Some(dj);
                }
            }
            if let Some(dj) = donor {
                if let Some(moved) = groups[dj].pop() {
                    groups[ci].push(moved);
                    continue;
                }
            }
            groups[ci].push(rng.next_usize(k));
        }

        // Recompute coarse centroids as (weighted) mean of fine centroids in each group.
        for ci in 0..coarse_count {
            let mut sx = 0f64;
            let mut sy = 0f64;
            let mut sz = 0f64;
            let mut wsum = 0f64;
            for &fi in &groups[ci] {
                let w = fine_counts
                    .map(|c| c[fi] as f64)
                    .filter(|&w| w > 0.0)
                    .unwrap_or(1.0);
                sx += fine[fi].x as f64 * w;
                sy += fine[fi].y as f64 * w;
                sz += fine[fi].z as f64 * w;
                wsum += w;
            }
            if wsum > 0.0 {
                coarse[ci] = Point::new((sx / wsum) as f32, (sy / wsum) as f32, (sz / wsum) as f32);
            } else {
                coarse[ci] = fine[groups[ci][0]];
            }
        }
    }

    Hierarchy { coarse, groups }
}

#[inline(always)]
fn nearest_hierarchical(p: &Point, fine: &[Point], h: &Hierarchy) -> usize {
    let (c1, c2) = nearest_two(p, &h.coarse);
    let mut best = h.groups[c1][0];
    let mut best_d = dist2(p, &fine[best]);
    for &fi in &h.groups[c1] {
        let d = dist2(p, &fine[fi]);
        if d < best_d {
            best_d = d;
            best = fi;
        }
    }
    if c2 != c1 {
        for &fi in &h.groups[c2] {
            let d = dist2(p, &fine[fi]);
            if d < best_d {
                best_d = d;
                best = fi;
            }
        }
    }
    best
}

fn reservoir_sample(points: &[Point], sample_size: usize, rng: &mut SplitMix64) -> Vec<Point> {
    let n = points.len();
    let sample_size = sample_size.min(n);
    let mut sample = Vec::with_capacity(sample_size);
    for (i, &p) in points.iter().enumerate() {
        if i < sample_size {
            sample.push(p);
        } else {
            let j = (rng.next_u64() % ((i as u64) + 1)) as usize;
            if j < sample_size {
                sample[j] = p;
            }
        }
    }
    sample
}

fn init_centroids(sample: &[Point], k: usize, rng: &mut SplitMix64) -> Vec<Point> {
    if sample.is_empty() {
        return vec![Point::default(); k.max(1)];
    }

    const KMEANS_PP_MAX_K: usize = 512;
    const KMEANS_PP_MAX_SAMPLE: usize = 200_000;

    if k <= KMEANS_PP_MAX_K && sample.len() <= KMEANS_PP_MAX_SAMPLE {
        return kmeans_pp(sample, k, rng);
    }

    // Random distinct picks from sample when possible.
    let n = sample.len();
    let mut idx: Vec<usize> = (0..n).collect();
    let take = k.min(n);
    for i in 0..take {
        let j = i + rng.next_usize(n - i);
        idx.swap(i, j);
    }
    let mut centroids = Vec::with_capacity(k);
    for i in 0..k {
        centroids.push(sample[idx[i % take]]);
    }
    centroids
}

fn kmeans_pp(sample: &[Point], k: usize, rng: &mut SplitMix64) -> Vec<Point> {
    let n = sample.len();
    let mut centroids = Vec::with_capacity(k);
    centroids.push(sample[rng.next_usize(n)]);

    let mut d2: Vec<f32> = Vec::with_capacity(n);
    d2.extend(sample.iter().map(|p| dist2(p, &centroids[0])));

    while centroids.len() < k {
        let mut sum = 0f64;
        for &v in &d2 {
            sum += v as f64;
        }
        if !(sum > 0.0) {
            centroids.push(sample[rng.next_usize(n)]);
            continue;
        }
        let mut r = rng.next_f64() * sum;
        let mut pick = 0usize;
        for (i, &v) in d2.iter().enumerate() {
            r -= v as f64;
            if r <= 0.0 {
                pick = i;
                break;
            }
        }
        let new_c = sample[pick];
        centroids.push(new_c);
        for (i, p) in sample.iter().enumerate() {
            let nd = dist2(p, &new_c);
            if nd < d2[i] {
                d2[i] = nd;
            }
        }
    }
    centroids
}

fn lloyd_train(points: &[Point], k: usize, centroids: &mut [Point], max_iter: usize, rng: &mut SplitMix64) {
    let n = points.len();
    if n == 0 || k == 0 {
        return;
    }
    let use_hier = k >= 512 || (n >= 50_000_000 && k >= 256);
    let coarse_count = if use_hier {
        ((k as f64).sqrt().ceil() as usize).clamp(8, 2048).min(k)
    } else {
        0
    };

    let tol2 = 1e-8_f32;

    for _ in 0..max_iter {
        let hierarchy = if use_hier {
            Some(build_hierarchy(centroids, None, coarse_count, rng))
        } else {
            None
        };

        let mut sx = vec![0f64; k];
        let mut sy = vec![0f64; k];
        let mut sz = vec![0f64; k];
        let mut counts = vec![0u64; k];

        if let Some(h) = &hierarchy {
            for p in points {
                let ci = nearest_hierarchical(p, centroids, h);
                sx[ci] += p.x as f64;
                sy[ci] += p.y as f64;
                sz[ci] += p.z as f64;
                counts[ci] += 1;
            }
        } else {
            for p in points {
                let ci = nearest_exact(p, centroids);
                sx[ci] += p.x as f64;
                sy[ci] += p.y as f64;
                sz[ci] += p.z as f64;
                counts[ci] += 1;
            }
        }

        let mut max_shift = 0f32;
        for ci in 0..k {
            if counts[ci] == 0 {
                centroids[ci] = points[rng.next_usize(n)];
                continue;
            }
            let inv = 1.0 / (counts[ci] as f64);
            let newc = Point::new((sx[ci] * inv) as f32, (sy[ci] * inv) as f32, (sz[ci] * inv) as f32);
            let shift = dist2(&centroids[ci], &newc);
            if shift > max_shift {
                max_shift = shift;
            }
            centroids[ci] = newc;
        }

        if max_shift <= tol2 {
            break;
        }
    }
}

fn minibatch_train(
    points: &[Point],
    k: usize,
    centroids: &mut [Point],
    steps: usize,
    rng: &mut SplitMix64,
) {
    let n = points.len();
    if n == 0 || k == 0 || steps == 0 {
        return;
    }

    let use_hier = k >= 512 || (n >= 50_000_000 && k >= 256);
    let coarse_count = if use_hier {
        ((k as f64).sqrt().ceil() as usize).clamp(8, 2048).min(k)
    } else {
        0
    };
    let rebuild_every = (steps / 10).max(10_000).min(500_000);
    let mut hierarchy = if use_hier {
        Some(build_hierarchy(centroids, None, coarse_count, rng))
    } else {
        None
    };

    let mut counts = vec![0u64; k];
    for t in 0..steps {
        if use_hier && t % rebuild_every == 0 && t != 0 {
            hierarchy = Some(build_hierarchy(centroids, Some(&counts), coarse_count, rng));
        }
        let p = points[rng.next_usize(n)];
        let ci = if let Some(h) = &hierarchy {
            nearest_hierarchical(&p, centroids, h)
        } else {
            nearest_exact(&p, centroids)
        };
        counts[ci] = counts[ci].saturating_add(1);
        let eta = 1.0f32 / (counts[ci] as f32);
        let c = &mut centroids[ci];
        c.x += (p.x - c.x) * eta;
        c.y += (p.y - c.y) * eta;
        c.z += (p.z - c.z) * eta;
    }
}

fn fix_empty_clusters(assignments: &mut [u32], counts: &mut [u64]) {
    let k = counts.len();
    if k == 0 {
        return;
    }
    let mut empties: Vec<usize> = Vec::new();
    for (i, &c) in counts.iter().enumerate() {
        if c == 0 {
            empties.push(i);
        }
    }
    if empties.is_empty() {
        return;
    }

    let mut stash: Vec<Vec<usize>> = vec![Vec::new(); k];
    for (i, &a) in assignments.iter().enumerate() {
        let c = a as usize;
        if stash[c].len() < 2 {
            stash[c].push(i);
        }
    }

    for e in empties {
        let mut donor: Option<usize> = None;
        for d in 0..k {
            if counts[d] > 1 && stash[d].len() >= 2 {
                donor = Some(d);
                break;
            }
        }
        let Some(d) = donor else { break };
        let idx = stash[d].pop().unwrap();
        assignments[idx] = e as u32;
        counts[d] -= 1;
        counts[e] += 1;
        stash[e].push(idx);
    }
}

fn write_u32_le<W: Write>(w: &mut W, v: u32) -> io::Result<()> {
    w.write_all(&v.to_le_bytes())
}

fn in_memory_solve(mut reader: FastReader<io::StdinLock<'_>>, n_points: usize, k: usize) -> io::Result<()> {
    let mut points = Vec::with_capacity(n_points);
    for _ in 0..n_points {
        points.push(reader.read_point()?);
    }

    let seed = (n_points as u64)
        ^ ((k as u64).wrapping_mul(0x9E3779B97F4A7C15))
        ^ (points.get(0).map(|p| p.x.to_bits() as u64).unwrap_or(0));
    let mut rng = SplitMix64::new(seed);

    let sample_size = 100_000usize.min(points.len());
    let sample = reservoir_sample(&points, sample_size, &mut rng);
    let mut centroids = init_centroids(&sample, k, &mut rng);

    // Strategy: full Lloyd for smaller cases, otherwise mini-batch + 1 Lloyd pass.
    if points.len() <= 5_000_000 && k <= 2048 {
        let max_iter = 30;
        lloyd_train(&points, k, &mut centroids, max_iter, &mut rng);
    } else {
        // Mini-batch steps scale with k and sample size.
        let steps = (k.saturating_mul(2000)).clamp(200_000, 2_000_000);
        minibatch_train(&points, k, &mut centroids, steps, &mut rng);
        lloyd_train(&points, k, &mut centroids, 2, &mut rng);
    }

    let use_hier = k >= 512 || (points.len() >= 50_000_000 && k >= 256);
    let coarse_count = if use_hier {
        ((k as f64).sqrt().ceil() as usize).clamp(8, 2048).min(k)
    } else {
        0
    };
    let hierarchy = if use_hier {
        Some(build_hierarchy(&centroids, None, coarse_count, &mut rng))
    } else {
        None
    };

    let mut assignments: Vec<u32> = Vec::with_capacity(points.len());
    let mut counts = vec![0u64; k];
    if let Some(h) = &hierarchy {
        for p in &points {
            let ci = nearest_hierarchical(p, &centroids, h);
            assignments.push(ci as u32);
            counts[ci] += 1;
        }
    } else {
        for p in &points {
            let ci = nearest_exact(p, &centroids);
            assignments.push(ci as u32);
            counts[ci] += 1;
        }
    }

    if points.len() >= k {
        fix_empty_clusters(&mut assignments, &mut counts);
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());
    for &a in &assignments {
        write_u32_le(&mut out, a)?;
    }
    out.flush()?;
    Ok(())
}

fn streaming_solve(mut reader: FastReader<io::StdinLock<'_>>, n_points: u64, k: usize) -> io::Result<()> {
    let seed = n_points
        ^ ((k as u64).wrapping_mul(0x9E3779B97F4A7C15))
        ^ 0xD1B54A32D192ED03;
    let mut rng = SplitMix64::new(seed);

    let warmup = {
        let base = (k.saturating_mul(32)).max(50_000);
        base.min(1_000_000).min(n_points as usize)
    };

    let mut warm_points = Vec::with_capacity(warmup);
    for _ in 0..warmup {
        warm_points.push(reader.read_point()?);
    }

    if warm_points.is_empty() {
        return Ok(());
    }

    // Initialize centroids from a capped init sample to keep k-means++ affordable when used.
    let init_sample_size = warm_points.len().min(200_000);
    let init_sample = if warm_points.len() <= init_sample_size {
        warm_points.clone()
    } else {
        reservoir_sample(&warm_points, init_sample_size, &mut rng)
    };

    let mut centroids = init_centroids(&init_sample, k, &mut rng);

    // Quick refinement on warmup to stabilize.
    let refine_steps = (k.saturating_mul(400)).clamp(50_000, 500_000);
    minibatch_train(&warm_points, k, &mut centroids, refine_steps, &mut rng);

    let use_hier = k >= 512 || (n_points >= 50_000_000 && k >= 256);
    let coarse_count = if use_hier {
        ((k as f64).sqrt().ceil() as usize).clamp(8, 2048).min(k)
    } else {
        0
    };
    let mut hierarchy = if use_hier {
        Some(build_hierarchy(&centroids, None, coarse_count, &mut rng))
    } else {
        None
    };

    // Assign warmup points (buffered) so we can ensure all clusters are non-empty when possible.
    let mut warm_assign: Vec<u32> = Vec::with_capacity(warm_points.len());
    let mut counts: Vec<u64> = vec![0u64; k];
    if let Some(h) = &hierarchy {
        for p in &warm_points {
            let ci = nearest_hierarchical(p, &centroids, h);
            warm_assign.push(ci as u32);
            counts[ci] += 1;
        }
    } else {
        for p in &warm_points {
            let ci = nearest_exact(p, &centroids);
            warm_assign.push(ci as u32);
            counts[ci] += 1;
        }
    }

    if (n_points as usize) >= k {
        fix_empty_clusters(&mut warm_assign, &mut counts);
    }

    // Recompute centroids exactly from warmup assignments to make online updates consistent.
    let mut sx = vec![0f64; k];
    let mut sy = vec![0f64; k];
    let mut sz = vec![0f64; k];
    for (p, &a) in warm_points.iter().zip(warm_assign.iter()) {
        let ci = a as usize;
        sx[ci] += p.x as f64;
        sy[ci] += p.y as f64;
        sz[ci] += p.z as f64;
    }
    for ci in 0..k {
        if counts[ci] == 0 {
            centroids[ci] = warm_points[rng.next_usize(warm_points.len())];
        } else {
            let inv = 1.0 / (counts[ci] as f64);
            centroids[ci] = Point::new((sx[ci] * inv) as f32, (sy[ci] * inv) as f32, (sz[ci] * inv) as f32);
        }
    }

    if use_hier {
        hierarchy = Some(build_hierarchy(&centroids, Some(&counts), coarse_count, &mut rng));
    }

    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());

    for &a in &warm_assign {
        write_u32_le(&mut out, a)?;
    }

    let mut processed: u64 = warmup as u64;
    let rebuild_interval: u64 = if use_hier {
        // Rebuild coarse buckets periodically as centroids drift.
        (200_000u64).max((k as u64) * 50).min(2_000_000u64)
    } else {
        u64::MAX
    };
    let mut since_rebuild: u64 = 0;

    while processed < n_points {
        let p = reader.read_point()?;
        if use_hier {
            if since_rebuild >= rebuild_interval {
                hierarchy = Some(build_hierarchy(&centroids, Some(&counts), coarse_count, &mut rng));
                since_rebuild = 0;
            }
        }

        let ci = if let Some(h) = &hierarchy {
            nearest_hierarchical(&p, &centroids, h)
        } else {
            nearest_exact(&p, &centroids)
        };

        write_u32_le(&mut out, ci as u32)?;

        counts[ci] = counts[ci].saturating_add(1);
        let eta = 1.0f32 / (counts[ci] as f32);
        let c = &mut centroids[ci];
        c.x += (p.x - c.x) * eta;
        c.y += (p.y - c.y) * eta;
        c.z += (p.z - c.z) * eta;

        processed += 1;
        since_rebuild += 1;
    }

    out.flush()?;
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let reader = FastReader::new(stdin.lock());
    solve(reader)
}

fn solve(mut reader: FastReader<io::StdinLock<'_>>) -> io::Result<()> {
    let n_points_u64 = reader.read_ascii_u64()?;
    let k_u64 = reader.read_ascii_u64()?;
    reader.consume_until_newline()?;
    if n_points_u64 == 0 || k_u64 == 0 {
        return Ok(());
    }
    if k_u64 > (u32::MAX as u64) {
        return Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "num_clusters must fit in u32",
        ));
    }

    let k = k_u64 as usize;

    if k_u64 > n_points_u64 {
        // Not enough points to populate all clusters; emit unique labels for all points.
        let stdout = io::stdout();
        let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());
        for i in 0..(n_points_u64 as u32) {
            let _ = reader.read_point()?; // consume point
            write_u32_le(&mut out, i)?;
        }
        out.flush()?;
        return Ok(());
    }

    const MAX_INMEM_BYTES: u64 = 768 * 1024 * 1024;
    let bytes = n_points_u64.saturating_mul(12);
    let in_mem = bytes <= MAX_INMEM_BYTES && n_points_u64 <= (usize::MAX as u64);

    if in_mem {
        in_memory_solve(reader, n_points_u64 as usize, k)
    } else {
        streaming_solve(reader, n_points_u64, k)
    }
}
