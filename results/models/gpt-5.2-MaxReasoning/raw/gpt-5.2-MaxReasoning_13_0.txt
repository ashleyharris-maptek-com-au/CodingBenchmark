// Longest Common Substring across N strings (exact).
// Adaptive strategy:
//   - N == 1: output the string
//   - N == 2 and small: O(n*m) DP
//   - Otherwise: suffix automaton built on the shortest string, then intersected with others in O(total_length)
//
// MSVC /std:c++20 compatible, no non-standard headers.

#include <algorithm>
#include <cstdint>
#include <cstring>
#include <iostream>
#include <limits>
#include <string>
#include <utility>
#include <vector>

namespace {

struct SuffixAutomaton {
  struct Edge {
    int to;
    int next;
    unsigned char c;
  };
  struct State {
    int link;
    int len;
    int head;      // head of adjacency list in edges[]
    int first_pos; // end position (in base string) of some occurrence of this state
  };

  std::vector<State> st;
  std::vector<Edge> edges;
  int last = 0;

  explicit SuffixAutomaton(std::size_t max_len = 0) {
    // These are upper bounds for a suffix automaton of a single string:
    // states <= 2*|S|-1, transitions <= 3*|S|-4.
    if (max_len > 0) {
      const std::size_t max_states = 2 * max_len + 1;
      const std::size_t max_edges = 3 * max_len + 1;
      st.reserve(max_states);
      edges.reserve(max_edges);
    }
    st.push_back(State{-1, 0, -1, -1}); // root
    last = 0;
  }

  int size() const { return static_cast<int>(st.size()); }

  int get_next(int v, unsigned char c) const {
    for (int ei = st[v].head; ei != -1; ei = edges[ei].next) {
      if (edges[ei].c == c) return edges[ei].to;
    }
    return -1;
  }

  void add_edge(int v, unsigned char c, int to) {
    edges.push_back(Edge{to, st[v].head, c});
    st[v].head = static_cast<int>(edges.size()) - 1;
  }

  void set_next(int v, unsigned char c, int to) {
    for (int ei = st[v].head; ei != -1; ei = edges[ei].next) {
      if (edges[ei].c == c) {
        edges[ei].to = to;
        return;
      }
    }
    add_edge(v, c, to);
  }

  void copy_transitions(int from, int to) {
    st[to].head = -1;
    for (int ei = st[from].head; ei != -1; ei = edges[ei].next) {
      add_edge(to, edges[ei].c, edges[ei].to);
    }
  }

  int new_state() {
    st.push_back(State{-1, 0, -1, -1});
    return static_cast<int>(st.size()) - 1;
  }

  void extend(unsigned char c, int pos) {
    const int cur = new_state();
    st[cur].len = st[last].len + 1;
    st[cur].first_pos = pos;

    int p = last;
    while (p != -1 && get_next(p, c) == -1) {
      set_next(p, c, cur);
      p = st[p].link;
    }

    if (p == -1) {
      st[cur].link = 0;
      last = cur;
      return;
    }

    const int q = get_next(p, c);
    if (st[p].len + 1 == st[q].len) {
      st[cur].link = q;
      last = cur;
      return;
    }

    const int clone = new_state();
    st[clone].len = st[p].len + 1;
    st[clone].link = st[q].link;
    st[clone].first_pos = st[q].first_pos;
    copy_transitions(q, clone);

    while (p != -1 && get_next(p, c) == q) {
      set_next(p, c, clone);
      p = st[p].link;
    }

    st[q].link = clone;
    st[cur].link = clone;
    last = cur;
  }
};

std::string lcs_dp_two_strings(const std::string& s1, const std::string& s2) {
  // Space-optimized DP for longest common substring of two strings.
  // dp[j] = length of longest common suffix of s1[0..i) and s2[0..j).
  const std::string* a = &s1;
  const std::string* b = &s2;
  if (b->size() > a->size()) std::swap(a, b);

  const int n = static_cast<int>(a->size());
  const int m = static_cast<int>(b->size());
  std::vector<int> dp(m + 1, 0);

  int best_len = 0;
  int best_end = 0; // end position in *a (exclusive)

  for (int i = 1; i <= n; ++i) {
    for (int j = m; j >= 1; --j) {
      if ((*a)[i - 1] == (*b)[j - 1]) {
        dp[j] = dp[j - 1] + 1;
        if (dp[j] > best_len) {
          best_len = dp[j];
          best_end = i;
        }
      } else {
        dp[j] = 0;
      }
    }
  }

  if (best_len == 0) return std::string();
  return a->substr(best_end - best_len, best_len);
}

std::string lcs_suffix_automaton_multi(const std::vector<std::string>& strings, int base_idx) {
  const std::string& base = strings[base_idx];
  if (base.empty()) return std::string();

  SuffixAutomaton sam(base.size());
  for (int i = 0; i < static_cast<int>(base.size()); ++i) {
    sam.extend(static_cast<unsigned char>(base[i]), i);
  }

  const int sz = sam.size();

  // Counting sort states by length (ascending).
  const int max_len = sam.st[sam.last].len;
  std::vector<int> cnt(max_len + 1, 0);
  for (int v = 0; v < sz; ++v) ++cnt[sam.st[v].len];
  for (int i = 1; i <= max_len; ++i) cnt[i] += cnt[i - 1];
  std::vector<int> order(sz);
  for (int v = sz - 1; v >= 0; --v) {
    order[--cnt[sam.st[v].len]] = v;
  }

  // best[v] = best (max) match length for this state across all processed strings, aggregated as a min over strings.
  std::vector<int> best(sz, 0);
  for (int v = 0; v < sz; ++v) best[v] = sam.st[v].len; // base string contains all its substrings

  std::vector<int> tmp(sz, 0);

  for (int idx = 0; idx < static_cast<int>(strings.size()); ++idx) {
    if (idx == base_idx) continue;
    const std::string& t = strings[idx];
    if (t.empty()) return std::string();

    std::memset(tmp.data(), 0, tmp.size() * sizeof(tmp[0]));

    int v = 0;
    int l = 0;
    for (char ch : t) {
      const unsigned char c = static_cast<unsigned char>(ch);
      while (v != 0 && sam.get_next(v, c) == -1) {
        v = sam.st[v].link;
        l = sam.st[v].len;
      }

      const int nx = sam.get_next(v, c);
      if (nx != -1) {
        v = nx;
        ++l;
      } else {
        v = 0;
        l = 0;
      }

      if (l > tmp[v]) tmp[v] = l;
    }

    // Propagate match lengths to suffix links (descending lengths).
    for (int i = sz - 1; i >= 1; --i) { // skip root at i==0 (it has link=-1)
      const int x = order[i];
      const int p = sam.st[x].link;
      if (p != -1) {
        int clipped = tmp[x];
        if (clipped > sam.st[p].len) clipped = sam.st[p].len;
        if (clipped > tmp[p]) tmp[p] = clipped;
      }
      if (tmp[x] < best[x]) best[x] = tmp[x];
    }
  }

  int ans_len = 0;
  int ans_state = 0;
  for (int v = 1; v < sz; ++v) { // root contributes only length 0
    const int p = sam.st[v].link;
    const int min_len = (p == -1) ? 1 : (sam.st[p].len + 1);
    const int cand = best[v];
    if (cand < min_len) continue;
    if (cand > ans_len) {
      ans_len = cand;
      ans_state = v;
    }
  }

  if (ans_len == 0) return std::string();
  const int end_pos = sam.st[ans_state].first_pos; // inclusive index of last character
  return base.substr(static_cast<std::size_t>(end_pos - ans_len + 1), static_cast<std::size_t>(ans_len));
}

} // namespace

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int n = 0;
  if (!(std::cin >> n)) return 0;
  std::string line;
  std::getline(std::cin, line); // consume endline after n

  if (n <= 0) {
    std::cout << "\n";
    return 0;
  }

  std::vector<std::string> strings;
  strings.reserve(static_cast<std::size_t>(n));
  for (int i = 0; i < n; ++i) {
    std::getline(std::cin, line);
    if (!line.empty() && line.back() == '\r') line.pop_back(); // Windows CRLF
    strings.push_back(line);
  }

  for (const auto& s : strings) {
    if (s.empty()) {
      std::cout << "\n";
      return 0;
    }
  }

  if (n == 1) {
    std::cout << strings[0] << "\n";
    return 0;
  }

  // Choose base string (shortest) for SAM to minimize state count.
  int base_idx = 0;
  std::size_t min_len = strings[0].size();
  for (int i = 1; i < n; ++i) {
    if (strings[i].size() < min_len) {
      min_len = strings[i].size();
      base_idx = i;
    }
  }

  // Adaptive branch: for two small strings, use O(n*m) DP.
  if (n == 2) {
    const std::uint64_t prod =
        static_cast<std::uint64_t>(strings[0].size()) * static_cast<std::uint64_t>(strings[1].size());
    constexpr std::uint64_t DP_LIMIT = 50ULL * 1000ULL * 1000ULL; // ~50M cell updates
    if (prod <= DP_LIMIT) {
      std::cout << lcs_dp_two_strings(strings[0], strings[1]) << "\n";
      return 0;
    }
  }

  std::cout << lcs_suffix_automaton_multi(strings, base_idx) << "\n";
  return 0;
}
