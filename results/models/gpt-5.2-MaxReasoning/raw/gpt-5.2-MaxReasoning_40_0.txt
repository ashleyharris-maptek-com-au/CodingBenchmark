#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <utility>
#include <vector>

#if defined(_MSC_VER)
#include <intrin.h>
#endif

using namespace std;

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
  return (int)__popcnt64(x);
#else
  return __builtin_popcountll(x);
#endif
}

struct MaxWeightCliqueCoverBnB {
  int n = 0;
  int blocks = 0;
  vector<long long> w;
  vector<vector<uint64_t>> conflict;  // conflict[u][b] has bit v if u conflicts with v

  vector<long long> residual;         // workspace: residual weights for find_branching_nodes
  vector<int> best_clique;
  long long best_weight = 0;

  MaxWeightCliqueCoverBnB(int n_, vector<long long> w_, vector<vector<uint64_t>> conflict_)
      : n(n_),
        blocks((n_ + 63) / 64),
        w(std::move(w_)),
        conflict(std::move(conflict_)),
        residual(n_, 0) {}

  inline bool is_conflict(int u, int v) const {
    return (conflict[u][v >> 6] >> (v & 63)) & 1ULL;
  }

  static inline void set_bit(vector<uint64_t>& bs, int idx) {
    bs[idx >> 6] |= (1ULL << (idx & 63));
  }

  static inline bool test_bit(const vector<uint64_t>& bs, int idx) {
    return (bs[idx >> 6] >> (idx & 63)) & 1ULL;
  }

  void update_incumbent_if_improved(const vector<int>& clique, long long clique_weight) {
    if (clique_weight > best_weight) {
      best_weight = clique_weight;
      best_clique = clique;
    }
  }

  long long sum_weights(const vector<int>& P) const {
    long long s = 0;
    for (int v : P) s += w[v];
    return s;
  }

  // Greedy construction of a feasible solution (a clique in the compatibility graph,
  // i.e., an independent set in the conflict graph).
  void init_incumbent_greedy() {
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    stable_sort(order.begin(), order.end(), [&](int a, int b) {
      if (w[a] != w[b]) return w[a] > w[b];
      return a < b;
    });

    vector<uint64_t> forbidden(blocks, 0);
    vector<int> clique;
    clique.reserve(n);
    long long total = 0;

    for (int v : order) {
      if (test_bit(forbidden, v)) continue;
      clique.push_back(v);
      total += w[v];
      set_bit(forbidden, v);
      for (int b = 0; b < blocks; ++b) forbidden[b] |= conflict[v][b];
    }

    update_incumbent_if_improved(clique, total);
  }

  // Greedily find an independent set in the compatibility graph induced by P.
  // Compatibility edges are exactly the non-conflicts, so an independent set here
  // is a set of pairwise conflicting vertices.
  vector<int> greedily_find_independent_set(const vector<int>& P_in) const {
    vector<int> P = P_in;
    vector<int> indep;
    indep.reserve(P.size());

    while (!P.empty()) {
      int v = P.front();
      indep.push_back(v);
      vector<int> nextP;
      nextP.reserve(P.size());
      for (size_t i = 1; i < P.size(); ++i) {
        int u = P[i];
        if (is_conflict(v, u)) nextP.push_back(u);  // keep non-neighbors in compatibility graph
      }
      P.swap(nextP);
    }
    return indep;
  }

  // NetworkX-style branching set computation using a weighted independent set cover.
  vector<int> find_branching_nodes(const vector<int>& P_in, long long target) {
    vector<int> P = P_in;

    vector<int> touched;
    touched.reserve(P.size());
    for (int v : P) {
      residual[v] = w[v];
      touched.push_back(v);
    }

    long long total_wt = 0;
    while (!P.empty()) {
      vector<int> indep = greedily_find_independent_set(P);
      long long min_wt = std::numeric_limits<long long>::max();
      for (int v : indep) min_wt = std::min(min_wt, residual[v]);
      total_wt += min_wt;
      if (total_wt > target) break;

      for (int v : indep) residual[v] -= min_wt;

      vector<int> nextP;
      nextP.reserve(P.size());
      for (int v : P) {
        if (residual[v] != 0) nextP.push_back(v);
      }
      P.swap(nextP);
    }

    for (int v : touched) residual[v] = 0;
    return P;
  }

  void expand(vector<int>& clique, long long clique_weight, vector<int> P) {
    update_incumbent_if_improved(clique, clique_weight);

    if (P.empty()) return;
    if (clique_weight + sum_weights(P) <= best_weight) return;

    vector<int> branching_nodes = find_branching_nodes(P, best_weight - clique_weight);
    while (!branching_nodes.empty()) {
      int v = branching_nodes.back();
      branching_nodes.pop_back();

      auto it = std::find(P.begin(), P.end(), v);
      if (it == P.end()) continue;
      P.erase(it);

      vector<int> newP;
      newP.reserve(P.size());
      for (int u : P) {
        if (!is_conflict(v, u)) newP.push_back(u);  // edge in compatibility graph
      }

      clique.push_back(v);
      expand(clique, clique_weight + w[v], std::move(newP));
      clique.pop_back();
    }
  }

  pair<long long, vector<int>> solve() {
    best_weight = 0;
    best_clique.clear();
    init_incumbent_greedy();

    // Initial candidate order: descending degree in compatibility graph
    vector<int> nodes(n);
    iota(nodes.begin(), nodes.end(), 0);

    vector<int> conflict_deg(n, 0);
    for (int v = 0; v < n; ++v) {
      int deg = 0;
      for (int b = 0; b < blocks; ++b) deg += popcount64(conflict[v][b]);
      conflict_deg[v] = deg;
    }

    stable_sort(nodes.begin(), nodes.end(), [&](int a, int b) {
      int degA = (n - 1) - conflict_deg[a];
      int degB = (n - 1) - conflict_deg[b];
      if (degA != degB) return degA > degB;
      if (w[a] != w[b]) return w[a] > w[b];
      return a < b;
    });

    vector<int> clique;
    clique.reserve(n);
    expand(clique, 0, std::move(nodes));
    return {best_weight, best_clique};
  }
};

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m;
  if (!(cin >> n >> m)) return 0;

  vector<long long> c(n);
  for (int i = 0; i < n; ++i) cin >> c[i];

  vector<pair<int, int>> edges;
  edges.reserve((size_t)m);
  vector<char> forced_zero(n, 0);
  for (int i = 0; i < m; ++i) {
    int u, v;
    cin >> u >> v;
    if (u == v) {
      if (0 <= u && u < n) forced_zero[u] = 1;
      continue;
    }
    if (u > v) std::swap(u, v);
    edges.emplace_back(u, v);
  }

  // Keep only potentially useful vertices: positive weight and not forced to 0.
  vector<int> active;
  active.reserve(n);
  for (int i = 0; i < n; ++i) {
    if (!forced_zero[i] && c[i] > 0) active.push_back(i);
  }

  vector<int> x(n, 0);
  long long objective = 0;

  const int A = (int)active.size();
  if (A == 0) {
    cout << 0 << "\n";
    for (int i = 0; i < n; ++i) {
      if (i) cout << ' ';
      cout << 0;
    }
    cout << "\n";
    return 0;
  }

  vector<int> orig_to_active(n, -1);
  for (int i = 0; i < A; ++i) orig_to_active[active[i]] = i;

  vector<long long> wA(A);
  for (int i = 0; i < A; ++i) wA[i] = c[active[i]];

  vector<vector<int>> adjA(A);
  for (auto [u, v] : edges) {
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    int au = orig_to_active[u];
    int av = orig_to_active[v];
    if (au < 0 || av < 0) continue;
    adjA[au].push_back(av);
    adjA[av].push_back(au);
  }
  for (int i = 0; i < A; ++i) {
    auto& nb = adjA[i];
    sort(nb.begin(), nb.end());
    nb.erase(unique(nb.begin(), nb.end()), nb.end());
  }

  // Connected components in the conflict graph.
  vector<char> vis(A, 0);
  vector<int> local_of_active(A, -1);  // reused per component (sparse reset)

  for (int start = 0; start < A; ++start) {
    if (vis[start]) continue;

    vector<int> comp;
    queue<int> q;
    vis[start] = 1;
    q.push(start);
    while (!q.empty()) {
      int v = q.front();
      q.pop();
      comp.push_back(v);
      for (int u : adjA[v]) {
        if (!vis[u]) {
          vis[u] = 1;
          q.push(u);
        }
      }
    }

    const int k = (int)comp.size();
    if (k == 1) {
      int aidx = comp[0];
      int orig = active[aidx];
      x[orig] = 1;
      objective += c[orig];
      continue;
    }

    // Build local conflict bitsets for this component.
    const int blocks = (k + 63) / 64;
    vector<vector<uint64_t>> conflict(k, vector<uint64_t>(blocks, 0));
    vector<long long> wLocal(k);

    vector<int> touched;
    touched.reserve(k);
    for (int i = 0; i < k; ++i) {
      local_of_active[comp[i]] = i;
      touched.push_back(comp[i]);
      wLocal[i] = wA[comp[i]];
    }

    for (int i = 0; i < k; ++i) {
      int av = comp[i];
      for (int au : adjA[av]) {
        int j = local_of_active[au];
        if (j >= 0) {
          conflict[i][j >> 6] |= (1ULL << (j & 63));
        }
      }
    }

    for (int v : touched) local_of_active[v] = -1;

    MaxWeightCliqueCoverBnB solver(k, std::move(wLocal), std::move(conflict));
    auto [bestW, chosen] = solver.solve();
    (void)bestW;  // chosen vertices encode the solution.

    for (int lv : chosen) {
      int aidx = comp[lv];
      int orig = active[aidx];
      x[orig] = 1;
      objective += c[orig];
    }
  }

  cout << objective << "\n";
  for (int i = 0; i < n; ++i) {
    if (i) cout << ' ';
    cout << (x[i] ? 1 : 0);
  }
  cout << "\n";
  return 0;
}
