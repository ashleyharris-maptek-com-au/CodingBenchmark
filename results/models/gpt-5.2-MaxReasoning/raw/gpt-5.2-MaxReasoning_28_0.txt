use std::io::{self, Read, Write};

struct Scanner {
    input: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = Vec::new();
        io::stdin().read_to_end(&mut input).unwrap();
        Self { input, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.input.len() && self.input[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.input.len() {
            return None;
        }
        let mut val: usize = 0;
        while self.idx < self.input.len() {
            let b = self.input[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            val = val * 10 + (b - b'0') as usize;
            self.idx += 1;
        }
        Some(val)
    }
}

fn cover_size(in_cover: &[u8]) -> usize {
    in_cover.iter().map(|&b| b as usize).sum()
}

fn verify_cover(edges_u: &[u32], edges_v: &[u32], in_cover: &[u8], loop_fixed: &[u8]) -> bool {
    for (i, &is_fixed) in loop_fixed.iter().enumerate() {
        if is_fixed == 1 && in_cover[i] == 0 {
            return false;
        }
    }
    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;
        if in_cover[u] == 0 && in_cover[v] == 0 {
            return false;
        }
    }
    true
}

fn greedy_edge_picking_cover(
    n: usize,
    edges_u: &[u32],
    edges_v: &[u32],
    loop_fixed: &[u8],
) -> Vec<u8> {
    let mut in_cover = loop_fixed.to_vec();
    if in_cover.len() < n {
        in_cover.resize(n, 0);
    }

    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;
        if in_cover[u] == 0 && in_cover[v] == 0 {
            in_cover[u] = 1;
            in_cover[v] = 1;
        }
    }
    in_cover
}

fn build_csr(n: usize, edges_u: &[u32], edges_v: &[u32], deg: &[u32]) -> (Vec<usize>, Vec<u32>) {
    let mut offsets = vec![0usize; n + 1];
    for i in 0..n {
        offsets[i + 1] = offsets[i] + deg[i] as usize;
    }
    let mut neighbors = vec![0u32; offsets[n]];
    let mut cursor = offsets[..n].to_vec();
    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;
        neighbors[cursor[u]] = edges_v[i];
        cursor[u] += 1;
        neighbors[cursor[v]] = edges_u[i];
        cursor[v] += 1;
    }
    (offsets, neighbors)
}

fn prune_redundant_vertices(
    offsets: &[usize],
    neighbors: &[u32],
    deg: &[u32],
    in_cover: &mut [u8],
    fixed: &[u8],
) {
    let mut cover_vertices: Vec<u32> = Vec::new();
    cover_vertices.reserve(cover_size(in_cover));
    for v in 0..in_cover.len() {
        if in_cover[v] == 1 && fixed[v] == 0 {
            cover_vertices.push(v as u32);
        }
    }

    cover_vertices.sort_unstable_by_key(|&v| (deg[v as usize], v));

    for &v32 in &cover_vertices {
        let v = v32 as usize;
        if in_cover[v] == 0 {
            continue;
        }
        let mut redundant = true;
        let start = offsets[v];
        let end = offsets[v + 1];
        for i in start..end {
            let u = neighbors[i] as usize;
            if in_cover[u] == 0 {
                redundant = false;
                break;
            }
        }
        if redundant {
            in_cover[v] = 0;
        }
    }
}

fn leaf_then_edge_picking_cover(
    n: usize,
    edges_u: &[u32],
    edges_v: &[u32],
    offsets: &[usize],
    neighbors: &[u32],
    deg: &[u32],
    loop_fixed: &[u8],
) -> Vec<u8> {
    let mut in_cover = loop_fixed.to_vec();
    if in_cover.len() < n {
        in_cover.resize(n, 0);
    }
    let fixed = in_cover.clone();

    let mut active = vec![1u8; n];
    let mut rem_deg = deg.to_vec();

    // Remove already-fixed vertices from the remaining graph.
    for v in 0..n {
        if fixed[v] == 1 && active[v] == 1 {
            active[v] = 0;
            rem_deg[v] = 0;
            let start = offsets[v];
            let end = offsets[v + 1];
            for i in start..end {
                let u = neighbors[i] as usize;
                if active[u] == 1 {
                    rem_deg[u] = rem_deg[u].saturating_sub(1);
                }
            }
        }
    }

    let mut stack: Vec<usize> = Vec::new();
    stack.reserve(n / 16 + 1);
    for v in 0..n {
        if active[v] == 1 && rem_deg[v] == 1 {
            stack.push(v);
        }
    }

    while let Some(v) = stack.pop() {
        if active[v] == 0 || rem_deg[v] != 1 {
            continue;
        }

        let mut u_opt: Option<usize> = None;
        let start = offsets[v];
        let end = offsets[v + 1];
        for i in start..end {
            let u = neighbors[i] as usize;
            if active[u] == 1 {
                u_opt = Some(u);
                break;
            }
        }

        let Some(u) = u_opt else {
            active[v] = 0;
            rem_deg[v] = 0;
            continue;
        };

        if in_cover[u] == 0 {
            in_cover[u] = 1;
        }

        // Remove leaf v and its neighbor u (now in the cover).
        active[v] = 0;
        rem_deg[v] = 0;
        if active[u] == 1 {
            active[u] = 0;
            rem_deg[u] = 0;
            let start_u = offsets[u];
            let end_u = offsets[u + 1];
            for i in start_u..end_u {
                let w = neighbors[i] as usize;
                if active[w] == 1 {
                    rem_deg[w] = rem_deg[w].saturating_sub(1);
                    if rem_deg[w] == 1 {
                        stack.push(w);
                    }
                }
            }
        }
    }

    // Finish by covering any remaining uncovered edges with a 2-approx edge-picking pass.
    for i in 0..edges_u.len() {
        let u = edges_u[i] as usize;
        let v = edges_v[i] as usize;
        if in_cover[u] == 0 && in_cover[v] == 0 {
            in_cover[u] = 1;
            in_cover[v] = 1;
        }
    }

    in_cover
}

fn write_cover(in_cover: &[u8]) {
    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());

    let k = cover_size(in_cover);
    writeln!(out, "{}", k).unwrap();

    let mut first = true;
    for v in 0..in_cover.len() {
        if in_cover[v] == 1 {
            if !first {
                write!(out, " ").unwrap();
            }
            first = false;
            write!(out, "{}", v).unwrap();
        }
    }
    writeln!(out).unwrap();
}

fn main() {
    let mut sc = Scanner::new();
    let Some(n) = sc.next_usize() else {
        return;
    };
    let Some(m_in) = sc.next_usize() else {
        write_cover(&vec![0u8; n]);
        return;
    };

    let mut loop_fixed = vec![0u8; n];
    let mut deg = vec![0u32; n];
    let mut edges_u: Vec<u32> = Vec::with_capacity(m_in);
    let mut edges_v: Vec<u32> = Vec::with_capacity(m_in);

    for _ in 0..m_in {
        let Some(u0) = sc.next_usize() else { break };
        let Some(v0) = sc.next_usize() else { break };
        if u0 >= n || v0 >= n {
            continue;
        }
        if u0 == v0 {
            loop_fixed[u0] = 1;
            continue;
        }
        let u = u0 as u32;
        let v = v0 as u32;
        edges_u.push(u);
        edges_v.push(v);
        deg[u0] += 1;
        deg[v0] += 1;
    }

    if edges_u.is_empty() {
        // Only possible remaining constraints are self-loops.
        let mut in_cover = loop_fixed;
        if !verify_cover(&edges_u, &edges_v, &in_cover, &vec![0u8; 0]) {
            in_cover.fill(1);
        }
        write_cover(&in_cover);
        return;
    }

    let (offsets, neighbors) = build_csr(n, &edges_u, &edges_v, &deg);

    let mut cover_a = greedy_edge_picking_cover(n, &edges_u, &edges_v, &loop_fixed);
    prune_redundant_vertices(&offsets, &neighbors, &deg, &mut cover_a, &loop_fixed);

    let mut cover_b = leaf_then_edge_picking_cover(
        n,
        &edges_u,
        &edges_v,
        &offsets,
        &neighbors,
        &deg,
        &loop_fixed,
    );
    prune_redundant_vertices(&offsets, &neighbors, &deg, &mut cover_b, &loop_fixed);

    let size_a = cover_size(&cover_a);
    let size_b = cover_size(&cover_b);
    let mut best = if size_b < size_a { cover_b } else { cover_a };

    if !verify_cover(&edges_u, &edges_v, &best, &loop_fixed) {
        best.fill(1);
    }

    write_cover(&best);
}
