using std::cin;

using std::tuple;

struct EdgeInput {
  int u = -1;
  int v = -1;
  long long w = 0;
};

struct AdjEdge {
  int to = -1;
  long long w = 0;
  int id = -1;
};

struct DSU {
  vector<int> parent;
  vector<int> rankv;

  explicit DSU(int n) : parent(n), rankv(n, 0) {
    for (int i = 0; i < n; ++i) parent[i] = i;
  }

  int find(int x) {
    while (parent[x] != x) {
      parent[x] = parent[parent[x]];
      x = parent[x];
    }
    return x;
  }

  bool unite(int a, int b) {
    a = find(a);
    b = find(b);
    if (a == b) return false;
    if (rankv[a] < rankv[b]) std::swap(a, b);
    parent[b] = a;
    if (rankv[a] == rankv[b]) ++rankv[a];
    return true;
  }
};

static constexpr long long INF = (std::numeric_limits<long long>::max() / 4);

static inline uint64_t pack_u32_i32(uint32_t a, int32_t b) {
  return (uint64_t(a) << 32) | uint32_t(b);
}

static inline void unpack_u32_i32(uint64_t p, uint32_t &a, int32_t &b) {
  a = uint32_t(p >> 32);
  b = int32_t(uint32_t(p & 0xFFFFFFFFu));
}

static vector<int> shortest_path_edges(int n,
                                       const vector<vector<AdjEdge>> &g,
                                       int src,
                                       int dst) {
  vector<long long> dist(n, INF);
  vector<int> prev_v(n, -1);
  vector<int> prev_e(n, -1);
  priority_queue<pair<long long, int>, vector<pair<long long, int>>,
                 std::greater<pair<long long, int>>>
      pq;

  dist[src] = 0;
  pq.push({0, src});

  while (!pq.empty()) {
    auto [d, v] = pq.top();
    pq.pop();
    if (d != dist[v]) continue;
    if (v == dst) break;
    for (const auto &e : g[v]) {
      long long nd = d + e.w;
      if (nd < dist[e.to]) {
        dist[e.to] = nd;
        prev_v[e.to] = v;
        prev_e[e.to] = e.id;
        pq.push({nd, e.to});
      }
    }
  }

  if (dist[dst] >= INF) return {};
  vector<int> path_edges;
  for (int cur = dst; cur != src;) {
    int eid = prev_e[cur];
    if (eid < 0) break;
    path_edges.push_back(eid);
    cur = prev_v[cur];
  }
  std::reverse(path_edges.begin(), path_edges.end());
  return path_edges;
}

static bool terminals_connected(int n,
                                const vector<int> &terminals,
                                const vector<EdgeInput> &edges,
                                const vector<int> &edge_ids) {
  if (terminals.size() <= 1) return true;
  vector<vector<int>> adj(n);
  adj.reserve(n);
  for (int eid : edge_ids) {
    const auto &e = edges[eid];
    adj[e.u].push_back(e.v);
    adj[e.v].push_back(e.u);
  }
  vector<char> vis(n, 0);
  queue<int> q;
  q.push(terminals[0]);
  vis[terminals[0]] = 1;
  while (!q.empty()) {
    int v = q.front();
    q.pop();
    for (int to : adj[v]) {
      if (!vis[to]) {
        vis[to] = 1;
        q.push(to);
      }
    }
  }
  for (int t : terminals) {
    if (!vis[t]) return false;
  }
  return true;
}

static vector<int> mst_then_prune(int n,
                                 const vector<int> &terminals,
                                 const vector<char> &is_terminal,
                                 const vector<EdgeInput> &edges,
                                 vector<int> edge_ids) {
  if (terminals.size() <= 1) return {};
  if (edge_ids.empty()) return {};

  std::sort(edge_ids.begin(), edge_ids.end());
  edge_ids.erase(std::unique(edge_ids.begin(), edge_ids.end()), edge_ids.end());

  vector<int> order = edge_ids;
  std::sort(order.begin(), order.end(),
            [&](int a, int b) { return edges[a].w < edges[b].w; });

  DSU dsu(n);
  vector<int> tree_edges;
  tree_edges.reserve(order.size());
  for (int eid : order) {
    const auto &e = edges[eid];
    if (dsu.unite(e.u, e.v)) tree_edges.push_back(eid);
  }

  // Prune non-terminal leaves.
  vector<vector<int>> inc(n);
  vector<int> degree(n, 0);
  vector<char> alive(edges.size(), 0);
  for (int eid : tree_edges) {
    alive[eid] = 1;
    const auto &e = edges[eid];
    inc[e.u].push_back(eid);
    inc[e.v].push_back(eid);
    degree[e.u]++;
    degree[e.v]++;
  }

  queue<int> q;
  for (int v = 0; v < n; ++v) {
    if (degree[v] == 1 && !is_terminal[v]) q.push(v);
  }

  while (!q.empty()) {
    int v = q.front();
    q.pop();
    if (is_terminal[v] || degree[v] != 1) continue;

    int incident = -1;
    for (int eid : inc[v]) {
      if (alive[eid]) {
        incident = eid;
        break;
      }
    }
    if (incident < 0) continue;

    alive[incident] = 0;
    const auto &e = edges[incident];
    int u = (e.u == v) ? e.v : e.u;
    degree[v]--;
    degree[u]--;
    if (degree[u] == 1 && !is_terminal[u]) q.push(u);
  }

  vector<int> pruned;
  pruned.reserve(tree_edges.size());
  for (int eid : tree_edges) {
    if (alive[eid]) pruned.push_back(eid);
  }
  return pruned;
}

static bool steiner_exact_dp(int n,
                             const vector<vector<AdjEdge>> &g,
                             const vector<int> &terminals,
                             vector<int> &out_edge_ids) {
  const int k = int(terminals.size());
  if (k <= 1) {
    out_edge_ids.clear();
    return true;
  }
  if (k > 20) return false;

  const uint32_t masks = (1u << k);
  const uint32_t full = masks - 1u;

  const uint64_t states = uint64_t(masks) * uint64_t(n);
  const uint64_t max_states = 8'000'000ULL;
  if (states > max_states) return false;

  vector<long long> dp(states, INF);
  vector<uint64_t> pre(states, pack_u32_i32(UINT32_MAX, -2));

  for (int i = 0; i < k; ++i) {
    const uint32_t mask = (1u << i);
    dp[uint64_t(mask) * uint64_t(n) + uint64_t(terminals[i])] = 0;
  }

  priority_queue<pair<long long, int>, vector<pair<long long, int>>,
                 std::greater<pair<long long, int>>>
      pq;

  for (uint32_t mask = 1; mask <= full; ++mask) {
    // Combine submasks at the same root vertex.
    for (uint32_t sub = (mask - 1u) & mask; sub; sub = (sub - 1u) & mask) {
      const uint32_t other = mask ^ sub;
      if (sub > other) continue;  // avoid duplicate partitions
      const uint64_t base_sub = uint64_t(sub) * uint64_t(n);
      const uint64_t base_other = uint64_t(other) * uint64_t(n);
      const uint64_t base_mask = uint64_t(mask) * uint64_t(n);
      for (int v = 0; v < n; ++v) {
        const long long a = dp[base_sub + uint64_t(v)];
        const long long b = dp[base_other + uint64_t(v)];
        if (a >= INF || b >= INF) continue;
        const long long cand = a + b;
        long long &cur = dp[base_mask + uint64_t(v)];
        if (cand < cur) {
          cur = cand;
          pre[base_mask + uint64_t(v)] = pack_u32_i32(sub, -1);  // split
        }
      }
    }

    // Shortest-path closure (multi-source Dijkstra seeded by dp[mask][*]).
    while (!pq.empty()) pq.pop();
    const uint64_t base_mask = uint64_t(mask) * uint64_t(n);
    for (int v = 0; v < n; ++v) {
      long long d = dp[base_mask + uint64_t(v)];
      if (d < INF) pq.push({d, v});
    }

    while (!pq.empty()) {
      auto [d, v] = pq.top();
      pq.pop();
      if (d != dp[base_mask + uint64_t(v)]) continue;
      for (const auto &e : g[v]) {
        long long nd = d + e.w;
        long long &to_dist = dp[base_mask + uint64_t(e.to)];
        if (nd < to_dist) {
          to_dist = nd;
          pre[base_mask + uint64_t(e.to)] =
              pack_u32_i32(uint32_t(v), int32_t(e.id));  // edge relaxation
          pq.push({nd, e.to});
        }
      }
    }
  }

  // Choose best root vertex for the full mask.
  long long best_cost = INF;
  int best_v = -1;
  const uint64_t base_full = uint64_t(full) * uint64_t(n);
  for (int v = 0; v < n; ++v) {
    long long d = dp[base_full + uint64_t(v)];
    if (d < best_cost) {
      best_cost = d;
      best_v = v;
    }
  }
  if (best_v < 0 || best_cost >= INF) return false;

  // Reconstruct edge ids from predecessor info.
  vector<int> ret;
  queue<pair<uint32_t, int>> q;
  q.push({full, best_v});
  while (!q.empty()) {
    auto [mask, v] = q.front();
    q.pop();
    uint32_t a;
    int32_t b;
    unpack_u32_i32(pre[uint64_t(mask) * uint64_t(n) + uint64_t(v)], a, b);
    if (a == UINT32_MAX) continue;  // base/unset
    if (b == -1) {
      uint32_t sub = a;
      q.push({sub, v});
      q.push({mask ^ sub, v});
    } else {
      int prev_v = int(a);
      int eid = int(b);
      if (eid >= 0) ret.push_back(eid);
      q.push({mask, prev_v});
    }
  }
  out_edge_ids = std::move(ret);
  return true;
}

static vector<int> steiner_greedy_multisource(int n,
                                             const vector<vector<AdjEdge>> &g,
                                             const vector<int> &terminals,
                                             const vector<char> &is_terminal) {
  const int k = int(terminals.size());
  if (k <= 1) return {};

  vector<char> in_tree(n, 0);
  vector<char> need_terminal(n, 0);
  for (int t : terminals) need_terminal[t] = 1;

  vector<int> tree_vertices;
  tree_vertices.reserve(n);

  int remaining = k;
  int start = terminals[0];
  in_tree[start] = 1;
  tree_vertices.push_back(start);
  if (need_terminal[start]) {
    need_terminal[start] = 0;
    --remaining;
  }

  vector<long long> dist(n, INF);
  vector<int> prev_v(n, -1);
  vector<int> prev_e(n, -1);
  priority_queue<pair<long long, int>, vector<pair<long long, int>>,
                 std::greater<pair<long long, int>>>
      pq;

  vector<int> chosen_edges;

  while (remaining > 0) {
    std::fill(dist.begin(), dist.end(), INF);
    std::fill(prev_v.begin(), prev_v.end(), -1);
    std::fill(prev_e.begin(), prev_e.end(), -1);
    while (!pq.empty()) pq.pop();

    for (int v : tree_vertices) {
      dist[v] = 0;
      pq.push({0, v});
    }

    int target = -1;
    while (!pq.empty()) {
      auto [d, v] = pq.top();
      pq.pop();
      if (d != dist[v]) continue;
      if (need_terminal[v]) {
        target = v;
        break;
      }
      for (const auto &e : g[v]) {
        long long nd = d + e.w;
        if (nd < dist[e.to]) {
          dist[e.to] = nd;
          prev_v[e.to] = v;
          prev_e[e.to] = e.id;
          pq.push({nd, e.to});
        }
      }
    }

    if (target < 0) break;  // disconnected; give up

    // Reconstruct path from target back to the existing tree.
    vector<int> path_vertices;
    vector<int> path_edges;
    int cur = target;
    path_vertices.push_back(cur);
    while (!in_tree[cur]) {
      int eid = prev_e[cur];
      int pv = prev_v[cur];
      if (eid < 0 || pv < 0) break;
      path_edges.push_back(eid);
      cur = pv;
      path_vertices.push_back(cur);
    }

    // Add the path into the tree.
    for (int v : path_vertices) {
      if (!in_tree[v]) {
        in_tree[v] = 1;
        tree_vertices.push_back(v);
      }
      if (need_terminal[v]) {
        need_terminal[v] = 0;
        --remaining;
      }
    }
    for (int eid : path_edges) chosen_edges.push_back(eid);
  }

  // If still not connected, fall back to union of shortest paths from a root.
  if (remaining > 0) {
    chosen_edges.clear();
    for (int i = 1; i < k; ++i) {
      vector<int> p = shortest_path_edges(n, g, terminals[0], terminals[i]);
      chosen_edges.insert(chosen_edges.end(), p.begin(), p.end());
    }
  }

  (void)is_terminal;
  return chosen_edges;
}

int main() {
  std::ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m, k;
  if (!(cin >> n >> m >> k)) return 0;

  vector<EdgeInput> edges(m);
  vector<vector<AdjEdge>> g(n);
  g.assign(n, {});
  for (int i = 0; i < m; ++i) {
    int u, v;
    long long w;
    cin >> u >> v >> w;
    edges[i] = {u, v, w};
    g[u].push_back({v, w, i});
    g[v].push_back({u, w, i});
  }

  vector<int> terminals(k);
  for (int i = 0; i < k; ++i) cin >> terminals[i];

  if (k <= 1) {
    cout << 0 << " " << 0 << "\n";
    return 0;
  }

  vector<char> is_terminal(n, 0);
  for (int t : terminals) is_terminal[t] = 1;

  vector<int> candidate_edges;
  vector<int> exact_edges;
  if (steiner_exact_dp(n, g, terminals, exact_edges)) {
    candidate_edges = std::move(exact_edges);
  } else {
    candidate_edges = steiner_greedy_multisource(n, g, terminals, is_terminal);
  }

  vector<int> final_edges =
      mst_then_prune(n, terminals, is_terminal, edges, candidate_edges);

  // Ensure validity; if broken, fall back to connecting everything to terminals[0].
  if (!terminals_connected(n, terminals, edges, final_edges)) {
    vector<int> fallback;
    for (int i = 1; i < k; ++i) {
      vector<int> p = shortest_path_edges(n, g, terminals[0], terminals[i]);
      fallback.insert(fallback.end(), p.begin(), p.end());
    }
    final_edges = mst_then_prune(n, terminals, is_terminal, edges, fallback);
  }

  long long total_w = 0;
  for (int eid : final_edges) total_w += edges[eid].w;

  cout << total_w << " " << final_edges.size() << "\n";
  for (int eid : final_edges) {
    cout << edges[eid].u << " " << edges[eid].v << "\n";
  }
  return 0;
}
