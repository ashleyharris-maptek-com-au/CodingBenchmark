import sys


def main() -> None:
    data = sys.stdin.buffer
    first = data.readline().split()
    if not first:
        return

    U = int(first[0])
    S = int(first[1])

    if U == 0:
        sys.stdout.write("SOLUTION\n\n")
        return

    masks = [0] * S
    elem_sets = [[] for _ in range(U)]

    for i in range(S):
        line = data.readline()
        if not line:
            line = b""
        mask = 0
        for token in line.split():
            try:
                e = int(token)
            except ValueError:
                continue
            if 0 <= e < U:
                mask |= 1 << e
        masks[i] = mask

        m = mask
        while m:
            lsb = m & -m
            e = lsb.bit_length() - 1
            elem_sets[e].append(i)
            m ^= lsb

    for e in range(U):
        if not elem_sets[e]:
            sys.stdout.write("NO SOLUTION\n")
            return

    sys.setrecursionlimit(max(1000, U + 50))
    solution = []

    masks_local = masks
    elem_sets_local = elem_sets

    def search(remaining: int) -> bool:
        if remaining == 0:
            return True

        best_e = -1
        best_count = 1 << 30

        tmp = remaining
        while tmp:
            lsb = tmp & -tmp
            e = lsb.bit_length() - 1

            cnt = 0
            for idx in elem_sets_local[e]:
                m = masks_local[idx]
                if m & remaining == m:
                    cnt += 1
                    if cnt >= best_count:
                        break
            if cnt == 0:
                return False
            if cnt < best_count:
                best_count = cnt
                best_e = e

            tmp ^= lsb

        for idx in elem_sets_local[best_e]:
            m = masks_local[idx]
            if m & remaining != m:
                continue
            solution.append(idx)
            if search(remaining & ~m):
                return True
            solution.pop()

        return False

    full = (1 << U) - 1
    if not search(full):
        sys.stdout.write("NO SOLUTION\n")
        return

    out = " ".join(map(str, sorted(solution)))
    sys.stdout.write("SOLUTION\n")
    sys.stdout.write(out + "\n")


if __name__ == "__main__":
    main()
