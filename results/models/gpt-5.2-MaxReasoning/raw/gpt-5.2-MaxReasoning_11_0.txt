use std::cmp::Ordering;
use std::collections::{BinaryHeap, HashMap};
use std::io::{self, Read};

#[derive(Clone, Copy, Debug, Default)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug, Default)]
struct BBox {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl BBox {
    fn from_points(points: &[Pt]) -> Self {
        let mut b = BBox {
            minx: points[0].x,
            miny: points[0].y,
            maxx: points[0].x,
            maxy: points[0].y,
        };
        for &p in points.iter().skip(1) {
            if p.x < b.minx {
                b.minx = p.x;
            }
            if p.y < b.miny {
                b.miny = p.y;
            }
            if p.x > b.maxx {
                b.maxx = p.x;
            }
            if p.y > b.maxy {
                b.maxy = p.y;
            }
        }
        b
    }

    fn width(&self) -> f64 {
        self.maxx - self.minx
    }
    fn height(&self) -> f64 {
        self.maxy - self.miny
    }

    fn shifted(&self, dx: f64, dy: f64) -> Self {
        BBox {
            minx: self.minx + dx,
            miny: self.miny + dy,
            maxx: self.maxx + dx,
            maxy: self.maxy + dy,
        }
    }

    fn contains_bbox(&self, other: BBox, pad: f64) -> bool {
        other.minx >= self.minx + pad
            && other.miny >= self.miny + pad
            && other.maxx <= self.maxx - pad
            && other.maxy <= self.maxy - pad
    }

    fn intersects(&self, other: BBox, pad: f64) -> bool {
        !(self.maxx <= other.minx + pad
            || other.maxx <= self.minx + pad
            || self.maxy <= other.miny + pad
            || other.maxy <= self.miny + pad)
    }
}

fn sub(a: Pt, b: Pt) -> Pt {
    Pt {
        x: a.x - b.x,
        y: a.y - b.y,
    }
}

fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(sub(b, a), sub(c, a))
}

fn signum(v: f64, eps: f64) -> i32 {
    if v > eps {
        1
    } else if v < -eps {
        -1
    } else {
        0
    }
}

fn point_on_segment(a: Pt, b: Pt, p: Pt, eps: f64) -> bool {
    if orient(a, b, p).abs() > eps {
        return false;
    }
    let minx = a.x.min(b.x) - eps;
    let maxx = a.x.max(b.x) + eps;
    let miny = a.y.min(b.y) - eps;
    let maxy = a.y.max(b.y) + eps;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy
}

fn segments_intersect(a: Pt, b: Pt, c: Pt, d: Pt, eps: f64) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    let s1 = signum(o1, eps);
    let s2 = signum(o2, eps);
    let s3 = signum(o3, eps);
    let s4 = signum(o4, eps);

    if s1 * s2 < 0 && s3 * s4 < 0 {
        return true;
    }
    if s1 == 0 && point_on_segment(a, b, c, eps) {
        return true;
    }
    if s2 == 0 && point_on_segment(a, b, d, eps) {
        return true;
    }
    if s3 == 0 && point_on_segment(c, d, a, eps) {
        return true;
    }
    if s4 == 0 && point_on_segment(c, d, b, eps) {
        return true;
    }
    false
}

fn polygon_signed_area(poly: &[Pt]) -> f64 {
    let n = poly.len();
    if n < 3 {
        return 0.0;
    }
    let mut sum = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        sum += poly[i].x * poly[j].y - poly[j].x * poly[i].y;
    }
    0.5 * sum
}

fn is_convex(poly: &[Pt], eps: f64) -> bool {
    let n = poly.len();
    if n < 3 {
        return false;
    }
    let mut s: i32 = 0;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let c = poly[(i + 2) % n];
        let v = orient(a, b, c);
        let sv = signum(v, eps);
        if sv != 0 {
            if s == 0 {
                s = sv;
            } else if s != sv {
                return false;
            }
        }
    }
    true
}

fn point_in_polygon(poly: &[Pt], p: Pt, eps: f64) -> bool {
    let n = poly.len();
    if n < 3 {
        return false;
    }
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if point_on_segment(a, b, p, eps) {
            return true;
        }
    }
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let ay = a.y;
        let by = b.y;
        let py = p.y;
        let cond = (ay > py) != (by > py);
        if cond {
            let t = (py - ay) / (by - ay);
            let x_int = a.x + (b.x - a.x) * t;
            if p.x < x_int + eps {
                inside = !inside;
            }
        }
    }
    inside
}

fn polygon_intersects(a: &[Pt], ab: BBox, b: &[Pt], bb: BBox, eps: f64) -> bool {
    if !ab.intersects(bb, 0.0) {
        return false;
    }
    let na = a.len();
    let nb = b.len();
    for i in 0..na {
        let a0 = a[i];
        let a1 = a[(i + 1) % na];
        for j in 0..nb {
            let b0 = b[j];
            let b1 = b[(j + 1) % nb];
            if segments_intersect(a0, a1, b0, b1, eps) {
                return true;
            }
        }
    }
    // containment (one inside the other)
    if point_in_polygon(a, b[0], eps) {
        return true;
    }
    if point_in_polygon(b, a[0], eps) {
        return true;
    }
    false
}

#[derive(Clone)]
struct Piece {
    idx: usize,
    verts: Vec<Pt>,
    area: f64,
    bbox: BBox,
}

enum StockKind {
    AxisAlignedRect { minx: f64, miny: f64, maxx: f64, maxy: f64 },
    Convex { ccw: bool },
    Generic,
}

struct EdgeIndex {
    cell: f64,
    inv_cell: f64,
    edges: Vec<(Pt, Pt, BBox)>,
    map: HashMap<(i32, i32), Vec<usize>>,
}

impl EdgeIndex {
    fn new(poly: &[Pt], bbox: BBox, cell: f64) -> Self {
        let cell = cell.max(1e-9);
        let inv_cell = 1.0 / cell;
        let mut edges: Vec<(Pt, Pt, BBox)> = Vec::with_capacity(poly.len());
        for i in 0..poly.len() {
            let a = poly[i];
            let b = poly[(i + 1) % poly.len()];
            let eb = BBox {
                minx: a.x.min(b.x),
                miny: a.y.min(b.y),
                maxx: a.x.max(b.x),
                maxy: a.y.max(b.y),
            };
            edges.push((a, b, eb));
        }
        let mut map: HashMap<(i32, i32), Vec<usize>> = HashMap::new();
        for (eid, &(_, _, eb)) in edges.iter().enumerate() {
            let (x0, y0, x1, y1) = bbox_cells(eb, inv_cell);
            for gx in x0..=x1 {
                for gy in y0..=y1 {
                    map.entry((gx, gy)).or_default().push(eid);
                }
            }
        }
        EdgeIndex {
            cell,
            inv_cell,
            edges,
            map,
        }
    }

    fn any_intersection(&self, seg_a: Pt, seg_b: Pt, seg_bbox: BBox, eps: f64) -> bool {
        let (x0, y0, x1, y1) = bbox_cells(seg_bbox, self.inv_cell);
        let mut seen: HashMap<usize, ()> = HashMap::new();
        for gx in x0..=x1 {
            for gy in y0..=y1 {
                if let Some(list) = self.map.get(&(gx, gy)) {
                    for &eid in list {
                        if seen.insert(eid, ()).is_some() {
                            continue;
                        }
                        let (a, b, eb) = self.edges[eid];
                        if !seg_bbox.intersects(eb, 0.0) {
                            continue;
                        }
                        if segments_intersect(seg_a, seg_b, a, b, eps) {
                            return true;
                        }
                    }
                }
            }
        }
        false
    }
}

struct Stock {
    verts: Vec<Pt>,
    bbox: BBox,
    kind: StockKind,
    eps: f64,
    pad: f64,
    edge_index: Option<EdgeIndex>,
}

impl Stock {
    fn contains_point(&self, p: Pt) -> bool {
        match self.kind {
            StockKind::AxisAlignedRect { minx, miny, maxx, maxy } => {
                p.x >= minx - self.eps && p.x <= maxx + self.eps && p.y >= miny - self.eps && p.y <= maxy + self.eps
            }
            StockKind::Convex { ccw } => {
                let n = self.verts.len();
                for i in 0..n {
                    let a = self.verts[i];
                    let b = self.verts[(i + 1) % n];
                    let v = orient(a, b, p);
                    if ccw {
                        if v < -self.eps {
                            return false;
                        }
                    } else if v > self.eps {
                        return false;
                    }
                }
                true
            }
            StockKind::Generic => point_in_polygon(&self.verts, p, self.eps),
        }
    }

    fn contains_polygon(&self, poly: &[Pt], pb: BBox) -> bool {
        if !self.bbox.contains_bbox(pb, self.pad) {
            return false;
        }
        match self.kind {
            StockKind::AxisAlignedRect { .. } => true,
            StockKind::Convex { .. } => poly.iter().all(|&p| self.contains_point(p)),
            StockKind::Generic => {
                if !poly.iter().all(|&p| self.contains_point(p)) {
                    return false;
                }
                let Some(edge_index) = &self.edge_index else {
                    return true;
                };
                let n = poly.len();
                for i in 0..n {
                    let a = poly[i];
                    let b = poly[(i + 1) % n];
                    let seg_bbox = BBox {
                        minx: a.x.min(b.x),
                        miny: a.y.min(b.y),
                        maxx: a.x.max(b.x),
                        maxy: a.y.max(b.y),
                    };
                    if edge_index.any_intersection(a, b, seg_bbox, self.eps) {
                        return false;
                    }
                }
                true
            }
        }
    }
}

fn bbox_cells(b: BBox, inv_cell: f64) -> (i32, i32, i32, i32) {
    let x0 = (b.minx * inv_cell).floor() as i32;
    let y0 = (b.miny * inv_cell).floor() as i32;
    let x1 = (b.maxx * inv_cell).floor() as i32;
    let y1 = (b.maxy * inv_cell).floor() as i32;
    (x0, y0, x1, y1)
}

struct SpatialGrid {
    inv_cell: f64,
    map: HashMap<(i32, i32), Vec<usize>>,
}

impl SpatialGrid {
    fn new(cell: f64) -> Self {
        let cell = cell.max(1e-9);
        SpatialGrid {
            inv_cell: 1.0 / cell,
            map: HashMap::new(),
        }
    }

    fn insert(&mut self, id: usize, bbox: BBox) {
        let (x0, y0, x1, y1) = bbox_cells(bbox, self.inv_cell);
        for gx in x0..=x1 {
            for gy in y0..=y1 {
                self.map.entry((gx, gy)).or_default().push(id);
            }
        }
    }

    fn query(&self, bbox: BBox, seen: &mut Vec<u32>, generation: &mut u32, out: &mut Vec<usize>) {
        out.clear();
        let (x0, y0, x1, y1) = bbox_cells(bbox, self.inv_cell);
        *generation = generation.wrapping_add(1);
        if *generation == 0 {
            // Rare wrap-around; reset stamps.
            seen.fill(0);
            *generation = 1;
        }
        let mark = *generation;
        for gx in x0..=x1 {
            for gy in y0..=y1 {
                if let Some(list) = self.map.get(&(gx, gy)) {
                    for &id in list {
                        if id >= seen.len() {
                            seen.resize(id + 1, 0);
                        }
                        if seen[id] != mark {
                            seen[id] = mark;
                            out.push(id);
                        }
                    }
                }
            }
        }
    }
}

#[derive(Clone)]
struct Placed {
    verts: Vec<Pt>,
    bbox: BBox,
}

#[derive(Clone, Copy, Debug, Default)]
struct Placement {
    stock: usize,
    tx: f64,
    ty: f64,
    rot: f64,
}

#[derive(Clone, Copy)]
struct Candidate {
    tx: f64,
    ty: f64,
    rot: f64,
    bbox: BBox,
}

#[derive(Clone, Copy)]
struct AnchorKey {
    y: f64,
    x: f64,
    idx: usize,
}

impl PartialEq for AnchorKey {
    fn eq(&self, other: &Self) -> bool {
        self.y.to_bits() == other.y.to_bits() && self.x.to_bits() == other.x.to_bits() && self.idx == other.idx
    }
}
impl Eq for AnchorKey {}
impl PartialOrd for AnchorKey {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}
impl Ord for AnchorKey {
    fn cmp(&self, other: &Self) -> Ordering {
        match self.y.total_cmp(&other.y) {
            Ordering::Equal => match self.x.total_cmp(&other.x) {
                Ordering::Equal => self.idx.cmp(&other.idx),
                ord => ord,
            },
            ord => ord,
        }
    }
}

struct StockPack {
    placed: Vec<Placed>,
    anchors: Vec<Pt>,
    grid: SpatialGrid,
    tmp_ids: Vec<usize>,
    seen: Vec<u32>,
    seen_gen: u32,
}

impl StockPack {
    fn new(cell: f64, seed_anchors: Vec<Pt>) -> Self {
        StockPack {
            placed: Vec::new(),
            anchors: seed_anchors,
            grid: SpatialGrid::new(cell),
            tmp_ids: Vec::new(),
            seen: Vec::new(),
            seen_gen: 1,
        }
    }
}

fn normalize_angle(mut a: f64) -> f64 {
    let two_pi = std::f64::consts::TAU;
    while a <= -std::f64::consts::PI {
        a += two_pi;
    }
    while a > std::f64::consts::PI {
        a -= two_pi;
    }
    a
}

fn candidate_rotations(piece: &Piece, mode: u8) -> Vec<f64> {
    // mode: 0=large,1=medium,2=small
    let mut rots = Vec::new();
    rots.push(0.0);
    rots.push(std::f64::consts::FRAC_PI_2);
    rots.push(std::f64::consts::PI);
    rots.push(3.0 * std::f64::consts::FRAC_PI_2);
    if mode == 2 {
        // Add a few dominant edge orientations.
        let n = piece.verts.len();
        let mut edges: Vec<(f64, f64)> = Vec::with_capacity(n);
        for i in 0..n {
            let a = piece.verts[i];
            let b = piece.verts[(i + 1) % n];
            let dx = b.x - a.x;
            let dy = b.y - a.y;
            let len2 = dx * dx + dy * dy;
            if len2 > 0.0 {
                edges.push((len2, dy.atan2(dx)));
            }
        }
        edges.sort_unstable_by(|a, b| b.0.total_cmp(&a.0));
        for &(_, ang) in edges.iter().take(4) {
            rots.push(-ang);
            rots.push(std::f64::consts::FRAC_PI_2 - ang);
        }
    } else if mode == 1 {
        let d = 15.0_f64.to_radians();
        rots.push(d);
        rots.push(-d);
        rots.push(2.0 * d);
        rots.push(-2.0 * d);
    }
    // normalize + dedup
    for r in rots.iter_mut() {
        *r = normalize_angle(*r);
    }
    rots.sort_unstable_by(|a, b| a.total_cmp(b));
    rots.dedup_by(|a, b| (*a - *b).abs() < 1e-9);
    rots
}

fn transform_piece_into(src: &[Pt], rot: f64, out: &mut Vec<Pt>) -> BBox {
    let c = rot.cos();
    let s = rot.sin();
    out.clear();
    out.reserve(src.len());
    let mut bbox = BBox {
        minx: f64::INFINITY,
        miny: f64::INFINITY,
        maxx: f64::NEG_INFINITY,
        maxy: f64::NEG_INFINITY,
    };
    for &p in src {
        let x = p.x * c - p.y * s;
        let y = p.x * s + p.y * c;
        if x < bbox.minx {
            bbox.minx = x;
        }
        if y < bbox.miny {
            bbox.miny = y;
        }
        if x > bbox.maxx {
            bbox.maxx = x;
        }
        if y > bbox.maxy {
            bbox.maxy = y;
        }
        out.push(Pt { x, y });
    }
    bbox
}

fn shift_in_place(poly: &mut [Pt], dx: f64, dy: f64) {
    for p in poly {
        p.x += dx;
        p.y += dy;
    }
}

fn k_smallest_anchor_indices(anchors: &[Pt], k: usize) -> Vec<usize> {
    let n = anchors.len();
    if n <= k {
        let mut idxs: Vec<usize> = (0..n).collect();
        idxs.sort_unstable_by(|&i, &j| {
            let a = anchors[i];
            let b = anchors[j];
            match a.y.total_cmp(&b.y) {
                Ordering::Equal => a.x.total_cmp(&b.x).then(i.cmp(&j)),
                ord => ord,
            }
        });
        return idxs;
    }

    let mut heap: BinaryHeap<AnchorKey> = BinaryHeap::with_capacity(k + 1);
    for (i, &p) in anchors.iter().enumerate() {
        heap.push(AnchorKey { y: p.y, x: p.x, idx: i });
        if heap.len() > k {
            heap.pop();
        }
    }
    heap.into_sorted_vec().into_iter().map(|a| a.idx).collect()
}

fn try_place_general(
    stock: &Stock,
    pack: &mut StockPack,
    piece: &Piece,
    rots: &[f64],
    greedy_first: bool,
    anchor_limit: usize,
    scratch: &mut Vec<Pt>,
) -> Option<Candidate> {
    let anchor_idxs = k_smallest_anchor_indices(&pack.anchors, anchor_limit.max(1));

    let mut best: Option<(f64, f64, Candidate)> = None; // (cost_y, cost_x, cand)
    for ai in anchor_idxs {
        let anchor = pack.anchors[ai];
        for &rot in rots {
            let rot_bbox = transform_piece_into(&piece.verts, rot, scratch);
            let dx = anchor.x + stock.pad - rot_bbox.minx;
            let dy = anchor.y + stock.pad - rot_bbox.miny;
            shift_in_place(scratch, dx, dy);
            let pb = rot_bbox.shifted(dx, dy);

            if !stock.contains_polygon(scratch, pb) {
                // restore for next iteration (avoid realloc by shifting back)
                shift_in_place(scratch, -dx, -dy);
                continue;
            }

            // overlap check using spatial grid
            pack.grid.query(pb, &mut pack.seen, &mut pack.seen_gen, &mut pack.tmp_ids);
            let mut ok = true;
            for &pid in &pack.tmp_ids {
                let other = &pack.placed[pid];
                if !pb.intersects(other.bbox, 0.0) {
                    continue;
                }
                if polygon_intersects(scratch, pb, &other.verts, other.bbox, stock.eps) {
                    ok = false;
                    break;
                }
            }

            if ok {
                let cand = Candidate {
                    tx: dx,
                    ty: dy,
                    rot,
                    bbox: pb,
                };
                if greedy_first {
                    return Some(cand);
                }
                let cost_y = pb.maxy;
                let cost_x = pb.maxx;
                match best {
                    None => best = Some((cost_y, cost_x, cand)),
                    Some((by, bx, _)) => {
                        if cost_y < by - stock.eps || ((cost_y - by).abs() <= stock.eps && cost_x < bx - stock.eps) {
                            best = Some((cost_y, cost_x, cand));
                        }
                    }
                }
            }

            // restore scratch back to unshifted rotated coords
            shift_in_place(scratch, -dx, -dy);
        }
    }
    best.map(|(_, _, c)| c)
}

fn pack_general(stock: &Stock, pieces: &[Piece]) -> (usize, Vec<Placement>) {
    let np = pieces.len();
    let mut placements: Vec<Placement> = vec![Placement::default(); np];

    // Mode selection: 0=large,1=medium,2=small
    let mode = if np <= 200 { 2 } else if np <= 2000 { 1 } else { 0 };
    let greedy_first = mode == 0;
    let anchor_limit = if mode == 2 { 512 } else if mode == 1 { 192 } else { 64 };
    let stocks_to_try = if np <= 200 { usize::MAX } else if np <= 2000 { 8 } else { 4 };

    // Spatial grid cell size heuristic.
    let mut avg = 0.0;
    for p in pieces {
        let a = (p.bbox.width().abs() * p.bbox.height().abs()).max(0.0);
        avg += a.sqrt();
    }
    avg /= (np.max(1)) as f64;
    let cell = (avg * 1.25).max(stock.bbox.width().abs().min(stock.bbox.height().abs()) / 64.0).max(1e-6);

    // Seed anchors: a handful of stock vertices + bbox min corner.
    let mut seed: Vec<Pt> = Vec::new();
    seed.push(Pt { x: stock.bbox.minx, y: stock.bbox.miny });
    for &v in stock.verts.iter().take(32) {
        seed.push(v);
    }

    let mut packs: Vec<StockPack> = vec![StockPack::new(cell, seed)];
    let mut scratch: Vec<Pt> = Vec::new();

    for piece in pieces {
        let rots = candidate_rotations(piece, mode);
        let mut placed = false;

        let start = if packs.len() > stocks_to_try { packs.len() - stocks_to_try } else { 0 };
        for s in start..packs.len() {
            let cand = try_place_general(
                stock,
                &mut packs[s],
                piece,
                &rots,
                greedy_first,
                anchor_limit,
                &mut scratch,
            );
            if let Some(c) = cand {
                // Materialize verts for storage
                let rot_bbox = transform_piece_into(&piece.verts, c.rot, &mut scratch);
                shift_in_place(&mut scratch, c.tx, c.ty);

                let pid = packs[s].placed.len();
                packs[s].grid.insert(pid, c.bbox);
                packs[s].placed.push(Placed {
                    verts: std::mem::take(&mut scratch),
                    bbox: c.bbox,
                });

                // Add new anchors (bbox corners to the right and above)
                packs[s].anchors.push(Pt { x: c.bbox.maxx + stock.pad, y: c.bbox.miny });
                packs[s].anchors.push(Pt { x: c.bbox.minx, y: c.bbox.maxy + stock.pad });

                placements[piece.idx] = Placement {
                    stock: s,
                    tx: c.tx,
                    ty: c.ty,
                    rot: c.rot,
                };
                placed = true;
                break;
            }
        }

        if !placed {
            // Open a new stock and try again.
            let mut seed: Vec<Pt> = Vec::new();
            seed.push(Pt { x: stock.bbox.minx, y: stock.bbox.miny });
            for &v in stock.verts.iter().take(32) {
                seed.push(v);
            }
            packs.push(StockPack::new(cell, seed));
            let s = packs.len() - 1;
            let cand = try_place_general(
                stock,
                &mut packs[s],
                piece,
                &rots,
                false,
                anchor_limit.max(128),
                &mut scratch,
            );
            if let Some(c) = cand {
                let _rot_bbox = transform_piece_into(&piece.verts, c.rot, &mut scratch);
                shift_in_place(&mut scratch, c.tx, c.ty);
                let pid = packs[s].placed.len();
                packs[s].grid.insert(pid, c.bbox);
                packs[s].placed.push(Placed {
                    verts: std::mem::take(&mut scratch),
                    bbox: c.bbox,
                });
                packs[s].anchors.push(Pt { x: c.bbox.maxx + stock.pad, y: c.bbox.miny });
                packs[s].anchors.push(Pt { x: c.bbox.minx, y: c.bbox.maxy + stock.pad });

                placements[piece.idx] = Placement {
                    stock: s,
                    tx: c.tx,
                    ty: c.ty,
                    rot: c.rot,
                };
            } else {
                // As a last resort, output an identity transform in stock 0.
                placements[piece.idx] = Placement {
                    stock: 0,
                    tx: 0.0,
                    ty: 0.0,
                    rot: 0.0,
                };
            }
        }
    }

    (packs.len(), placements)
}

fn pack_axis_aligned_rect(stock: &Stock, pieces: &[Piece]) -> (usize, Vec<Placement>) {
    let np = pieces.len();
    let mut placements: Vec<Placement> = vec![Placement::default(); np];
    let StockKind::AxisAlignedRect { minx, miny, maxx, maxy } = stock.kind else {
        return pack_general(stock, pieces);
    };

    // NFDH-ish: sort by descending max(bbox height/width) with 0/90 considered.
    let mut order: Vec<usize> = (0..np).collect();
    order.sort_unstable_by(|&i, &j| {
        let pi = &pieces[i];
        let pj = &pieces[j];
        let hi = pi.bbox.height().abs().max(pi.bbox.width().abs());
        let hj = pj.bbox.height().abs().max(pj.bbox.width().abs());
        hj.total_cmp(&hi).then(pj.area.total_cmp(&pi.area))
    });

    let mut scratch: Vec<Pt> = Vec::new();
    let mut stock_idx: usize = 0;
    let mut cur_x = minx;
    let mut cur_y = miny;
    let mut row_h = 0.0_f64;
    let w_lim = maxx;
    let h_lim = maxy;

    for &pid in &order {
        let piece = &pieces[pid];
        let rots = if np <= 200 { vec![0.0, std::f64::consts::FRAC_PI_2, std::f64::consts::PI, 3.0 * std::f64::consts::FRAC_PI_2] } else { vec![0.0, std::f64::consts::FRAC_PI_2] };

        let mut placed = false;
        let mut tried_new_row = false;
        loop {
            let mut best: Option<Candidate> = None;
            for &rot in &rots {
                let rot_bbox = transform_piece_into(&piece.verts, rot, &mut scratch);
                let dx = cur_x - rot_bbox.minx;
                let dy = cur_y - rot_bbox.miny;
                let pb = rot_bbox.shifted(dx, dy);
                if pb.maxx <= w_lim && pb.maxy <= h_lim && pb.minx >= minx && pb.miny >= miny {
                    let cand = Candidate { tx: dx, ty: dy, rot, bbox: pb };
                    let score = pb.maxy;
                    match best {
                        None => best = Some(cand),
                        Some(b) => {
                            if score < b.bbox.maxy - stock.eps {
                                best = Some(cand);
                            }
                        }
                    }
                }
            }

            if let Some(c) = best {
                placements[piece.idx] = Placement { stock: stock_idx, tx: c.tx, ty: c.ty, rot: c.rot };
                cur_x = c.bbox.maxx;
                row_h = row_h.max(c.bbox.height());
                placed = true;
                break;
            }

            if !tried_new_row {
                // New row
                tried_new_row = true;
                cur_x = minx;
                cur_y += row_h;
                row_h = 0.0;
                if cur_y >= h_lim {
                    // New stock
                    stock_idx += 1;
                    cur_x = minx;
                    cur_y = miny;
                    row_h = 0.0;
                }
            } else {
                // Force new stock
                stock_idx += 1;
                cur_x = minx;
                cur_y = miny;
                row_h = 0.0;
                tried_new_row = false;
            }
            // Loop continues until it fits (assumed feasible).
        }

        if !placed {
            placements[piece.idx] = Placement { stock: 0, tx: 0.0, ty: 0.0, rot: 0.0 };
        }
    }

    (stock_idx + 1, placements)
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut input = String::new();
        io::stdin().read_to_string(&mut input).unwrap();
        Scanner {
            buf: input.into_bytes(),
            idx: 0,
        }
    }

    fn skip_ws(&mut self) {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
    }

    fn next_usize(&mut self) -> usize {
        self.skip_ws();
        let mut v: usize = 0;
        while self.idx < self.buf.len() {
            let c = self.buf[self.idx];
            if !c.is_ascii_digit() {
                break;
            }
            v = v * 10 + (c - b'0') as usize;
            self.idx += 1;
        }
        v
    }

    fn next_f64(&mut self) -> f64 {
        self.skip_ws();
        if self.idx >= self.buf.len() {
            return 0.0;
        }
        let mut sign = 1.0;
        if self.buf[self.idx] == b'-' {
            sign = -1.0;
            self.idx += 1;
        } else if self.buf[self.idx] == b'+' {
            self.idx += 1;
        }

        let mut int_part: u64 = 0;
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_digit() {
            int_part = int_part * 10 + (self.buf[self.idx] - b'0') as u64;
            self.idx += 1;
        }

        let mut frac_part: u64 = 0;
        let mut frac_div: f64 = 1.0;
        if self.idx < self.buf.len() && self.buf[self.idx] == b'.' {
            self.idx += 1;
            while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_digit() {
                frac_part = frac_part * 10 + (self.buf[self.idx] - b'0') as u64;
                frac_div *= 10.0;
                self.idx += 1;
            }
        }

        let mut v = int_part as f64 + (frac_part as f64) / frac_div;

        if self.idx < self.buf.len() && (self.buf[self.idx] == b'e' || self.buf[self.idx] == b'E') {
            self.idx += 1;
            let mut exp_sign: i32 = 1;
            if self.idx < self.buf.len() && self.buf[self.idx] == b'-' {
                exp_sign = -1;
                self.idx += 1;
            } else if self.idx < self.buf.len() && self.buf[self.idx] == b'+' {
                self.idx += 1;
            }
            let mut exp: i32 = 0;
            while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_digit() {
                exp = exp * 10 + (self.buf[self.idx] - b'0') as i32;
                self.idx += 1;
            }
            v *= 10f64.powi(exp_sign * exp);
        }

        sign * v
    }
}

fn detect_axis_aligned_rect(poly: &[Pt], eps: f64) -> Option<(f64, f64, f64, f64)> {
    if poly.len() < 4 {
        return None;
    }
    let bbox = BBox::from_points(poly);
    let w = bbox.width().abs();
    let h = bbox.height().abs();
    if w <= eps || h <= eps {
        return None;
    }

    // Quick necessary conditions:
    // - All edges axis-aligned
    // - All vertices lie on the bbox boundary
    // - Polygon area matches bbox area (within tolerance)
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        let dx = (a.x - b.x).abs();
        let dy = (a.y - b.y).abs();
        if dx > eps && dy > eps {
            return None;
        }
        let on_boundary = (a.x - bbox.minx).abs() <= eps
            || (a.x - bbox.maxx).abs() <= eps
            || (a.y - bbox.miny).abs() <= eps
            || (a.y - bbox.maxy).abs() <= eps;
        if !on_boundary {
            return None;
        }
    }

    let area = polygon_signed_area(poly).abs();
    let rect_area = w * h;
    let tol = (1e-6 * rect_area).max(1e-9);
    if (area - rect_area).abs() > tol {
        return None;
    }
    Some((bbox.minx, bbox.miny, bbox.maxx, bbox.maxy))
}

fn main() {
    let mut sc = Scanner::new();
    let sv = sc.next_usize();
    let np = sc.next_usize();

    let mut stock_verts: Vec<Pt> = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x = sc.next_f64();
        let y = sc.next_f64();
        stock_verts.push(Pt { x, y });
    }

    let stock_bbox = BBox::from_points(&stock_verts);
    let scale = stock_bbox
        .width()
        .abs()
        .max(stock_bbox.height().abs())
        .max(1.0);
    let eps = 1e-9 * scale;
    let pad = 5.0 * eps;

    let signed_area = polygon_signed_area(&stock_verts);
    let convex = is_convex(&stock_verts, eps);
    let kind = if let Some((minx, miny, maxx, maxy)) = detect_axis_aligned_rect(&stock_verts, eps) {
        StockKind::AxisAlignedRect { minx, miny, maxx, maxy }
    } else if convex {
        StockKind::Convex { ccw: signed_area >= 0.0 }
    } else {
        StockKind::Generic
    };

    let edge_index = match kind {
        StockKind::Generic => {
            // Edge index cell size: roughly stock size / 128 (bounded).
            let cell = (scale / 128.0).max(1e-6);
            Some(EdgeIndex::new(&stock_verts, stock_bbox, cell))
        }
        _ => None,
    };

    let stock = Stock {
        verts: stock_verts,
        bbox: stock_bbox,
        kind,
        eps,
        pad,
        edge_index,
    };

    let mut pieces: Vec<Piece> = Vec::with_capacity(np);
    for idx in 0..np {
        let pv = sc.next_usize();
        let mut verts: Vec<Pt> = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x = sc.next_f64();
            let y = sc.next_f64();
            verts.push(Pt { x, y });
        }
        let bbox = BBox::from_points(&verts);
        let area = polygon_signed_area(&verts).abs();
        pieces.push(Piece { idx, verts, area, bbox });
    }

    // Global ordering: largest first.
    pieces.sort_unstable_by(|a, b| b.area.total_cmp(&a.area));

    let (num_stocks, placements) = match stock.kind {
        StockKind::AxisAlignedRect { .. } => pack_axis_aligned_rect(&stock, &pieces),
        _ => pack_general(&stock, &pieces),
    };

    let mut out = String::new();
    out.push_str(&format!("{}\n", num_stocks));
    for i in 0..np {
        let p = placements[i];
        out.push_str(&format!(
            "{} {:.12} {:.12} {:.12}\n",
            p.stock, p.tx, p.ty, p.rot
        ));
    }
    print!("{out}");
}
