; SPIR-V
; Version: 1.0
; Note: Header comments (like Bound/Generator) are ignored by spirv-as.

OpCapability Shader
OpCapability Geometry

%glsl = OpExtInstImport "GLSL.std.450"

OpMemoryModel Logical GLSL450

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

OpSource GLSL 450

OpName %main "main"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %gl_PerVertex_in "gl_PerVertex"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_PerVertex_out "gl_PerVertex"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position

OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %arr_v3_in ArrayStride 16

%void = OpTypeVoid
%fn_void = OpTypeFunction %void

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_4 = OpConstant %uint 4

%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4

%arr_v3_in = OpTypeArray %v3float %uint_4

%ptr_Input_arr_pv_in = OpTypePointer Input %arr_pv_in
%ptr_Input_arr_v3_in = OpTypePointer Input %arr_v3_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float

%ptr_Output_pv_out = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv_in Input
%worldPos = OpVariable %ptr_Input_arr_v3_in Input
%normal = OpVariable %ptr_Input_arr_v3_in Input
%color = OpVariable %ptr_Input_arr_v3_in Input

%gl_out = OpVariable %ptr_Output_pv_out Output
%outColor = OpVariable %ptr_Output_v3float Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0.0
%float_0_02 = OpConstant %float 0.02
%float_0_33333334 = OpConstant %float 0.33333334
%float_0_66666669 = OpConstant %float 0.66666669

%main = OpFunction %void None %fn_void
%entry = OpLabel

%gl_out_pos_ptr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

; Read start/end (gl_in[1], gl_in[2]) and adjacency (gl_in[0], gl_in[3])
%p0_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_Input_v4float %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

%col1_ptr = OpAccessChain %ptr_Input_v3float %color %int_1
%col1 = OpLoad %v3float %col1_ptr
%col2_ptr = OpAccessChain %ptr_Input_v3float %color %int_2
%col2 = OpLoad %v3float %col2_ptr

; Convert to NDC for tangent/perp computation: ndc = xy / w
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p0_ww = OpVectorShuffle %v2float %p0 %p0 3 3
%p0_ndc = OpFDiv %v2float %p0_xy %p0_ww

%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p1_ww = OpVectorShuffle %v2float %p1 %p1 3 3
%p1_ndc = OpFDiv %v2float %p1_xy %p1_ww

%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p2_ww = OpVectorShuffle %v2float %p2 %p2 3 3
%p2_ndc = OpFDiv %v2float %p2_xy %p2_ww

%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%p3_ww = OpVectorShuffle %v2float %p3 %p3 3 3
%p3_ndc = OpFDiv %v2float %p3_xy %p3_ww

; Adjacency-based tangents for smooth perpendiculars
%v_prev = OpFSub %v2float %p1_ndc %p0_ndc
%v_cur = OpFSub %v2float %p2_ndc %p1_ndc
%v_next = OpFSub %v2float %p3_ndc %p2_ndc

%dir_prev = OpExtInst %v2float %glsl Normalize %v_prev
%dir_cur = OpExtInst %v2float %glsl Normalize %v_cur
%dir_next = OpExtInst %v2float %glsl Normalize %v_next

%tan_start_sum = OpFAdd %v2float %dir_prev %dir_cur
%tan_end_sum = OpFAdd %v2float %dir_cur %dir_next

%tan_start = OpExtInst %v2float %glsl Normalize %tan_start_sum
%tan_end = OpExtInst %v2float %glsl Normalize %tan_end_sum

; Perpendicular of current segment direction
%dir_cur_x = OpCompositeExtract %float %dir_cur 0
%dir_cur_y = OpCompositeExtract %float %dir_cur 1
%dir_cur_y_neg = OpFNegate %float %dir_cur_y
%n_cur = OpCompositeConstruct %v2float %dir_cur_y_neg %dir_cur_x

; Miter perpendiculars at start/end
%tan_start_x = OpCompositeExtract %float %tan_start 0
%tan_start_y = OpCompositeExtract %float %tan_start 1
%tan_start_y_neg = OpFNegate %float %tan_start_y
%miter_start = OpCompositeConstruct %v2float %tan_start_y_neg %tan_start_x

%tan_end_x = OpCompositeExtract %float %tan_end 0
%tan_end_y = OpCompositeExtract %float %tan_end 1
%tan_end_y_neg = OpFNegate %float %tan_end_y
%miter_end = OpCompositeConstruct %v2float %tan_end_y_neg %tan_end_x

%dot_start = OpDot %float %miter_start %n_cur
%dot_end = OpDot %float %miter_end %n_cur

%miter_len_start = OpFDiv %float %float_0_02 %dot_start
%miter_len_end = OpFDiv %float %float_0_02 %dot_end

%offset_start = OpVectorTimesScalar %v2float %miter_start %miter_len_start
%offset_end = OpVectorTimesScalar %v2float %miter_end %miter_len_end

; Split the segment into thirds (clip-space interpolation), and interpolate offsets/colors along the segment
%delta_clip = OpFSub %v4float %p2 %p1
%delta_13 = OpVectorTimesScalar %v4float %delta_clip %float_0_33333334
%delta_23 = OpVectorTimesScalar %v4float %delta_clip %float_0_66666669
%p13 = OpFAdd %v4float %p1 %delta_13
%p23 = OpFAdd %v4float %p1 %delta_23

%col1_23 = OpVectorTimesScalar %v3float %col1 %float_0_66666669
%col2_13 = OpVectorTimesScalar %v3float %col2 %float_0_33333334
%col_13 = OpFAdd %v3float %col1_23 %col2_13

%col1_13 = OpVectorTimesScalar %v3float %col1 %float_0_33333334
%col2_23 = OpVectorTimesScalar %v3float %col2 %float_0_66666669
%col_23 = OpFAdd %v3float %col1_13 %col2_23

%off_start_23 = OpVectorTimesScalar %v2float %offset_start %float_0_66666669
%off_end_13 = OpVectorTimesScalar %v2float %offset_end %float_0_33333334
%offset_13 = OpFAdd %v2float %off_start_23 %off_end_13

%off_start_13 = OpVectorTimesScalar %v2float %offset_start %float_0_33333334
%off_end_23 = OpVectorTimesScalar %v2float %offset_end %float_0_66666669
%offset_23 = OpFAdd %v2float %off_start_13 %off_end_23

; --- Quad 1: [p1 .. p13] (draw), colors [col1 .. col_13]
%p1_w = OpCompositeExtract %float %p1 3
%p1_off = OpVectorTimesScalar %v2float %offset_start %p1_w
%p1_off_neg = OpFNegate %v2float %p1_off

%p1_off_x = OpCompositeExtract %float %p1_off 0
%p1_off_y = OpCompositeExtract %float %p1_off 1
%p1_off4 = OpCompositeConstruct %v4float %p1_off_x %p1_off_y %float_0 %float_0
%p1_pos_plus = OpFAdd %v4float %p1 %p1_off4
OpStore %gl_out_pos_ptr %p1_pos_plus
OpStore %outColor %col1
OpEmitVertex

%p1_offn_x = OpCompositeExtract %float %p1_off_neg 0
%p1_offn_y = OpCompositeExtract %float %p1_off_neg 1
%p1_off4n = OpCompositeConstruct %v4float %p1_offn_x %p1_offn_y %float_0 %float_0
%p1_pos_minus = OpFAdd %v4float %p1 %p1_off4n
OpStore %gl_out_pos_ptr %p1_pos_minus
OpStore %outColor %col1
OpEmitVertex

%p13_w = OpCompositeExtract %float %p13 3
%p13_off = OpVectorTimesScalar %v2float %offset_13 %p13_w
%p13_off_neg = OpFNegate %v2float %p13_off

%p13_off_x = OpCompositeExtract %float %p13_off 0
%p13_off_y = OpCompositeExtract %float %p13_off 1
%p13_off4 = OpCompositeConstruct %v4float %p13_off_x %p13_off_y %float_0 %float_0
%p13_pos_plus = OpFAdd %v4float %p13 %p13_off4
OpStore %gl_out_pos_ptr %p13_pos_plus
OpStore %outColor %col_13
OpEmitVertex

%p13_offn_x = OpCompositeExtract %float %p13_off_neg 0
%p13_offn_y = OpCompositeExtract %float %p13_off_neg 1
%p13_off4n = OpCompositeConstruct %v4float %p13_offn_x %p13_offn_y %float_0 %float_0
%p13_pos_minus = OpFAdd %v4float %p13 %p13_off4n
OpStore %gl_out_pos_ptr %p13_pos_minus
OpStore %outColor %col_13
OpEmitVertex

OpEndPrimitive

; --- Quad 3: [p23 .. p2] (draw), colors [col_23 .. col2]
%p23_w = OpCompositeExtract %float %p23 3
%p23_off = OpVectorTimesScalar %v2float %offset_23 %p23_w
%p23_off_neg = OpFNegate %v2float %p23_off

%p23_off_x = OpCompositeExtract %float %p23_off 0
%p23_off_y = OpCompositeExtract %float %p23_off 1
%p23_off4 = OpCompositeConstruct %v4float %p23_off_x %p23_off_y %float_0 %float_0
%p23_pos_plus = OpFAdd %v4float %p23 %p23_off4
OpStore %gl_out_pos_ptr %p23_pos_plus
OpStore %outColor %col_23
OpEmitVertex

%p23_offn_x = OpCompositeExtract %float %p23_off_neg 0
%p23_offn_y = OpCompositeExtract %float %p23_off_neg 1
%p23_off4n = OpCompositeConstruct %v4float %p23_offn_x %p23_offn_y %float_0 %float_0
%p23_pos_minus = OpFAdd %v4float %p23 %p23_off4n
OpStore %gl_out_pos_ptr %p23_pos_minus
OpStore %outColor %col_23
OpEmitVertex

%p2_w = OpCompositeExtract %float %p2 3
%p2_off = OpVectorTimesScalar %v2float %offset_end %p2_w
%p2_off_neg = OpFNegate %v2float %p2_off

%p2_off_x = OpCompositeExtract %float %p2_off 0
%p2_off_y = OpCompositeExtract %float %p2_off 1
%p2_off4 = OpCompositeConstruct %v4float %p2_off_x %p2_off_y %float_0 %float_0
%p2_pos_plus = OpFAdd %v4float %p2 %p2_off4
OpStore %gl_out_pos_ptr %p2_pos_plus
OpStore %outColor %col2
OpEmitVertex

%p2_offn_x = OpCompositeExtract %float %p2_off_neg 0
%p2_offn_y = OpCompositeExtract %float %p2_off_neg 1
%p2_off4n = OpCompositeConstruct %v4float %p2_offn_x %p2_offn_y %float_0 %float_0
%p2_pos_minus = OpFAdd %v4float %p2 %p2_off4n
OpStore %gl_out_pos_ptr %p2_pos_minus
OpStore %outColor %col2
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
