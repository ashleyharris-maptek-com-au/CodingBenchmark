use std::cmp::Reverse;
use std::io::{self, Read};

#[derive(Clone)]
struct Solver {
    n: usize,
    half: usize,
    adj: Vec<Vec<usize>>,
    // Edge multiplicities (supports parallel edges). w[u][v] == number of edges {u,v}.
    w: Vec<Vec<i32>>,
    edges: Vec<(usize, usize)>,

    // Search state
    side: Vec<i8>, // -1 unassigned, 0 in A, 1 in B
    deg_to_a: Vec<i32>,
    deg_to_b: Vec<i32>,
    count_a: usize,
    count_b: usize,
    cut_fixed: i32, // cut among already assigned vertices

    // Best solution found
    best_cut: i32,
    best_side: Vec<i8>,

    // Scratch for bounds
    scratch: Vec<i32>,
}

impl Solver {
    fn cut_of(&self, side: &[i8]) -> i32 {
        let mut cut = 0i32;
        for &(u, v) in &self.edges {
            if side[u] != side[v] {
                cut += 1;
            }
        }
        cut
    }

    fn greedy_initial(&self) -> Vec<i8> {
        let mut side = vec![-1i8; self.n];
        side[0] = 0;
        let mut count_a = 1usize;
        let mut count_b = 0usize;

        let mut order: Vec<usize> = (0..self.n).collect();
        order.sort_unstable_by_key(|&v| Reverse(self.adj[v].len()));

        for v in order {
            if v == 0 {
                continue;
            }
            if count_a == self.half {
                side[v] = 1;
                count_b += 1;
                continue;
            }
            if count_b == self.half {
                side[v] = 0;
                count_a += 1;
                continue;
            }

            let mut edges_to_a = 0i32;
            let mut edges_to_b = 0i32;
            for &u in &self.adj[v] {
                if side[u] == 0 {
                    edges_to_a += 1;
                } else if side[u] == 1 {
                    edges_to_b += 1;
                }
            }
            // If v goes to A, it adds edges to already-in-B as cut; similarly for B.
            let cost_if_a = edges_to_b;
            let cost_if_b = edges_to_a;

            let choose_a = if cost_if_a < cost_if_b {
                true
            } else if cost_if_b < cost_if_a {
                false
            } else {
                count_a <= count_b
            };

            if choose_a {
                side[v] = 0;
                count_a += 1;
            } else {
                side[v] = 1;
                count_b += 1;
            }
        }

        side
    }

    fn swap_improve(&self, mut side: Vec<i8>) -> Vec<i8> {
        // Keep vertex 0 fixed in A (symmetry breaking).
        loop {
            let mut d = vec![0i32; self.n];
            for v in 0..self.n {
                let mut ext = 0i32;
                let mut int = 0i32;
                for &u in &self.adj[v] {
                    if side[u] == side[v] {
                        int += 1;
                    } else {
                        ext += 1;
                    }
                }
                d[v] = ext - int;
            }

            let mut best_gain = 0i32;
            let mut best_pair = (usize::MAX, usize::MAX);
            for a in 0..self.n {
                if side[a] != 0 || a == 0 {
                    continue;
                }
                for b in 0..self.n {
                    if side[b] != 1 {
                        continue;
                    }
                    let gain = d[a] + d[b] - 2 * self.w[a][b];
                    if gain > best_gain {
                        best_gain = gain;
                        best_pair = (a, b);
                    }
                }
            }

            if best_gain <= 0 {
                break;
            }

            let (a, b) = best_pair;
            side[a] = 1;
            side[b] = 0;
        }

        side
    }

    fn init_best(&mut self) {
        let side0 = self.greedy_initial();
        let side1 = self.swap_improve(side0);
        let cut = self.cut_of(&side1);
        self.best_cut = cut;
        self.best_side = side1;
    }

    fn lower_bound(&mut self) -> i32 {
        let need_a = self.half as i32 - self.count_a as i32;
        let need_b = self.half as i32 - self.count_b as i32;
        if need_a < 0 || need_b < 0 {
            return i32::MAX / 4;
        }

        self.scratch.clear();
        let mut base = 0i32; // assume all remaining go to B
        let mut sum_b = 0i32;
        for v in 0..self.n {
            if self.side[v] == -1 {
                base += self.deg_to_a[v];
                sum_b += self.deg_to_b[v];
                self.scratch.push(self.deg_to_b[v] - self.deg_to_a[v]);
            }
        }

        let k = need_a as usize;
        let r = self.scratch.len();
        if k == 0 {
            return self.cut_fixed + base;
        }
        if k == r {
            return self.cut_fixed + sum_b;
        }

        self.scratch.select_nth_unstable(k - 1);
        let delta_sum: i32 = self.scratch[..k].iter().sum();
        self.cut_fixed + base + delta_sum
    }

    fn choose_vertex(&self) -> usize {
        let mut best_v = usize::MAX;
        let mut best_score: i32 = -1;
        for v in 0..self.n {
            if self.side[v] != -1 {
                continue;
            }
            let da = self.deg_to_a[v];
            let db = self.deg_to_b[v];
            let deg = self.adj[v].len() as i32;
            let score = 8 * (da + db) + 5 * (da - db).abs() + 2 * deg;
            if score > best_score {
                best_score = score;
                best_v = v;
            }
        }
        best_v
    }

    fn assign_vertex(&mut self, v: usize, s: i8) -> i32 {
        let delta_cut = if s == 0 {
            self.deg_to_b[v]
        } else {
            self.deg_to_a[v]
        };

        self.cut_fixed += delta_cut;
        if s == 0 {
            self.count_a += 1;
        } else {
            self.count_b += 1;
        }
        self.side[v] = s;

        if s == 0 {
            for &u in &self.adj[v] {
                if self.side[u] == -1 {
                    self.deg_to_a[u] += 1;
                }
            }
        } else {
            for &u in &self.adj[v] {
                if self.side[u] == -1 {
                    self.deg_to_b[u] += 1;
                }
            }
        }

        delta_cut
    }

    fn unassign_vertex(&mut self, v: usize, s: i8, delta_cut: i32) {
        // Important: keep side[v] assigned while reverting neighbor counts (handles self-loops safely).
        if s == 0 {
            for &u in &self.adj[v] {
                if self.side[u] == -1 {
                    self.deg_to_a[u] -= 1;
                }
            }
        } else {
            for &u in &self.adj[v] {
                if self.side[u] == -1 {
                    self.deg_to_b[u] -= 1;
                }
            }
        }

        self.side[v] = -1;
        if s == 0 {
            self.count_a -= 1;
        } else {
            self.count_b -= 1;
        }
        self.cut_fixed -= delta_cut;
    }

    fn finalize_if_forced(&mut self) -> bool {
        let need_a = self.half - self.count_a;
        let need_b = self.half - self.count_b;
        if need_a == 0 || need_b == 0 {
            let fill_side = if need_a == 0 { 1i8 } else { 0i8 };
            let mut final_cut = self.cut_fixed;
            for v in 0..self.n {
                if self.side[v] == -1 {
                    final_cut += if fill_side == 1 {
                        self.deg_to_a[v]
                    } else {
                        self.deg_to_b[v]
                    };
                }
            }
            if final_cut < self.best_cut {
                self.best_cut = final_cut;
                let mut completed = self.side.clone();
                for v in 0..self.n {
                    if completed[v] == -1 {
                        completed[v] = fill_side;
                    }
                }
                self.best_side = completed;
            }
            return true;
        }
        false
    }

    fn dfs(&mut self) {
        if self.best_cut == 0 {
            return;
        }

        let lb = self.lower_bound();
        if lb >= self.best_cut {
            return;
        }

        if self.count_a + self.count_b == self.n {
            if self.cut_fixed < self.best_cut {
                self.best_cut = self.cut_fixed;
                self.best_side = self.side.clone();
            }
            return;
        }

        if self.finalize_if_forced() {
            return;
        }

        let v = self.choose_vertex();

        // Determine branching order: try the cheaper side first.
        let mut options: [(i8, i32); 2] = [(0, i32::MAX / 4), (1, i32::MAX / 4)];
        if self.count_a < self.half {
            options[0].1 = self.deg_to_b[v];
        }
        if self.count_b < self.half {
            options[1].1 = self.deg_to_a[v];
        }
        if options[1].1 < options[0].1 {
            options.swap(0, 1);
        }

        for (s, _) in options {
            if s == 0 && self.count_a == self.half {
                continue;
            }
            if s == 1 && self.count_b == self.half {
                continue;
            }

            let delta = self.assign_vertex(v, s);
            self.dfs();
            self.unassign_vertex(v, s, delta);
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = it.next().unwrap().parse().unwrap();
    let half = n / 2;

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    let mut w: Vec<Vec<i32>> = vec![vec![0i32; n]; n];
    let mut edges: Vec<(usize, usize)> = Vec::with_capacity(m);

    for _ in 0..m {
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        adj[u].push(v);
        adj[v].push(u);
        if u < n && v < n {
            w[u][v] += 1;
            w[v][u] += 1;
        }
        edges.push((u, v));
    }

    // Initialize solver state: fix vertex 0 in A to break symmetry.
    let mut solver = Solver {
        n,
        half,
        adj,
        w,
        edges,
        side: vec![-1i8; n],
        deg_to_a: vec![0i32; n],
        deg_to_b: vec![0i32; n],
        count_a: 0,
        count_b: 0,
        cut_fixed: 0,
        best_cut: i32::MAX / 4,
        best_side: vec![0i8; n],
        scratch: Vec::with_capacity(n),
    };

    solver.init_best();

    solver.side[0] = 0;
    solver.count_a = 1;
    for &u in &solver.adj[0] {
        if solver.side[u] == -1 {
            solver.deg_to_a[u] += 1;
        }
    }

    solver.dfs();

    let mut part_a: Vec<usize> = Vec::with_capacity(half);
    for v in 0..n {
        if solver.best_side[v] == 0 {
            part_a.push(v);
        }
    }

    if part_a.len() != half {
        // Fallback: output 0..half as A with the correct cut size.
        let mut fallback = vec![1i8; n];
        for v in 0..half {
            fallback[v] = 0;
        }
        let cut = solver.cut_of(&fallback);
        println!("{cut}");
        for i in 0..half {
            if i > 0 {
                print!(" ");
            }
            print!("{i}");
        }
        println!();
        return;
    }

    let cut = solver.best_cut;
    println!("{cut}");
    for (i, v) in part_a.into_iter().enumerate() {
        if i > 0 {
            print!(" ");
        }
        print!("{v}");
    }
    println!();
}
