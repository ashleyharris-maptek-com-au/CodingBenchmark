declare const require: any;
declare const process: any;
const fs = require("fs");
const input: string = fs.readFileSync(0, "utf8");
const data: any = JSON.parse(input);

type Town = { id: any; x: number; y: number; value: number };

const width: number = data.width | 0;
const height: number = data.height | 0;
const grid: string[] = Array.isArray(data.grid) ? data.grid : [];
const cutBudget: number = Math.max(0, data.cutBudget | 0);

const N = width * height;
if (width <= 0 || height <= 0 || grid.length !== height || N <= 0 || cutBudget <= 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const burnable = new Uint8Array(N); // 1 if '.', 0 otherwise
for (let y = 0; y < height; y++) {
  const row = grid[y] ?? "";
  for (let x = 0; x < width; x++) {
    const ch = row.charCodeAt(x);
    burnable[y * width + x] = ch === 46 /* '.' */ ? 1 : 0;
  }
}

const cut = new Uint8Array(N); // 1 if selected cut
const isIgnition = new Uint8Array(N);
const townValueAt = new Float64Array(N);

const ignitionsRaw: any[] = Array.isArray(data.ignitions) ? data.ignitions : [];
const ignitionList: number[] = [];
for (let i = 0; i < ignitionsRaw.length; i++) {
  const p = ignitionsRaw[i];
  if (!Array.isArray(p) || p.length < 2) continue;
  const x = p[0] | 0;
  const y = p[1] | 0;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const id = y * width + x;
  if (burnable[id] !== 1) continue;
  if (isIgnition[id] === 1) continue;
  isIgnition[id] = 1;
  ignitionList.push(id);
}

const townsRaw: any[] = Array.isArray(data.towns) ? data.towns : [];
const towns: Town[] = new Array(townsRaw.length);
let townsCount = 0;
for (let i = 0; i < townsRaw.length; i++) {
  const t = townsRaw[i];
  if (!t) continue;
  const x = (t.x as number) | 0;
  const y = (t.y as number) | 0;
  const value = Number(t.value);
  if (!Number.isFinite(value) || value <= 0) continue;
  if (x < 0 || x >= width || y < 0 || y >= height) continue;
  const id = y * width + x;
  if (burnable[id] !== 1) continue;
  townValueAt[id] += value;
  towns[townsCount++] = { id: t.id, x, y, value };
}
towns.length = townsCount;

if (ignitionList.length === 0 || townsCount === 0) {
  process.stdout.write(JSON.stringify({ firebreaks: [] }));
  process.exit(0);
}

const queue = new Int32Array(N);
const mark = new Int32Array(N);
let markId = 1;

function bfsMarkReachable(currentMarkId: number): number {
  let head = 0;
  let tail = 0;
  for (let i = 0; i < ignitionList.length; i++) {
    const s = ignitionList[i];
    if (burnable[s] !== 1 || cut[s] === 1) continue;
    if (mark[s] === currentMarkId) continue;
    mark[s] = currentMarkId;
    queue[tail++] = s;
  }
  while (head < tail) {
    const u = queue[head++];
    const y = (u / width) | 0;
    const x = u - y * width;

    // up
    if (y > 0) {
      const v = u - width;
      if (mark[v] !== currentMarkId && burnable[v] === 1 && cut[v] === 0) {
        mark[v] = currentMarkId;
        queue[tail++] = v;
      }
    }
    // down
    if (y + 1 < height) {
      const v = u + width;
      if (mark[v] !== currentMarkId && burnable[v] === 1 && cut[v] === 0) {
        mark[v] = currentMarkId;
        queue[tail++] = v;
      }
    }
    // left
    if (x > 0) {
      const v = u - 1;
      if (mark[v] !== currentMarkId && burnable[v] === 1 && cut[v] === 0) {
        mark[v] = currentMarkId;
        queue[tail++] = v;
      }
    }
    // right
    if (x + 1 < width) {
      const v = u + 1;
      if (mark[v] !== currentMarkId && burnable[v] === 1 && cut[v] === 0) {
        mark[v] = currentMarkId;
        queue[tail++] = v;
      }
    }
  }
  return tail; // visited count
}

// Baseline reachable region (no cuts yet) — restrict later graph work.
const baseMarkId = markId++;
bfsMarkReachable(baseMarkId);

// Early containment attempt: cut all burnable neighbors of ignitions.
if (cutBudget > 0) {
  const frontierMark = new Uint8Array(N);
  const frontier: number[] = [];
  for (let i = 0; i < ignitionList.length; i++) {
    const u = ignitionList[i];
    const y = (u / width) | 0;
    const x = u - y * width;
    // up
    if (y > 0) {
      const v = u - width;
      if (burnable[v] === 1 && isIgnition[v] === 0 && frontierMark[v] === 0) {
        frontierMark[v] = 1;
        frontier.push(v);
      }
    }
    // down
    if (y + 1 < height) {
      const v = u + width;
      if (burnable[v] === 1 && isIgnition[v] === 0 && frontierMark[v] === 0) {
        frontierMark[v] = 1;
        frontier.push(v);
      }
    }
    // left
    if (x > 0) {
      const v = u - 1;
      if (burnable[v] === 1 && isIgnition[v] === 0 && frontierMark[v] === 0) {
        frontierMark[v] = 1;
        frontier.push(v);
      }
    }
    // right
    if (x + 1 < width) {
      const v = u + 1;
      if (burnable[v] === 1 && isIgnition[v] === 0 && frontierMark[v] === 0) {
        frontierMark[v] = 1;
        frontier.push(v);
      }
    }
  }
  if (frontier.length > 0 && frontier.length <= cutBudget) {
    const firebreaks: number[][] = new Array(frontier.length);
    for (let i = 0; i < frontier.length; i++) {
      const id = frontier[i];
      cut[id] = 1;
      const y = (id / width) | 0;
      const x = id - y * width;
      firebreaks[i] = [x, y];
    }
    process.stdout.write(JSON.stringify({ firebreaks }));
    process.exit(0);
  }
}

function ringCellsFor(id: number, out: number[]): number {
  out.length = 0;
  const y = (id / width) | 0;
  const x = id - y * width;

  if (y > 0) {
    const v = id - width;
    if (burnable[v] === 1 && cut[v] === 0 && isIgnition[v] === 0) out.push(v);
  }
  if (y + 1 < height) {
    const v = id + width;
    if (burnable[v] === 1 && cut[v] === 0 && isIgnition[v] === 0) out.push(v);
  }
  if (x > 0) {
    const v = id - 1;
    if (burnable[v] === 1 && cut[v] === 0 && isIgnition[v] === 0) out.push(v);
  }
  if (x + 1 < width) {
    const v = id + 1;
    if (burnable[v] === 1 && cut[v] === 0 && isIgnition[v] === 0) out.push(v);
  }

  return out.length;
}

// Best ring density (value per cut cell) among towns at risk (baseline).
let bestTownDensity = 0;
{
  const tmp: number[] = [];
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    const id = t.y * width + t.x;
    if (mark[id] !== baseMarkId) continue;
    const cost = ringCellsFor(id, tmp);
    if (cost <= 0) continue;
    const density = t.value / cost;
    if (density > bestTownDensity) bestTownDensity = density;
  }
}

const selectedCuts: number[] = [];
let budgetLeft = cutBudget;

// Articulation-point heuristic on baseline reachable region.
if (budgetLeft > 0 && bestTownDensity > 0) {
  const root = N;
  const nodeCount = N + 1;

  const disc = new Int32Array(nodeCount);
  const low = new Int32Array(nodeCount);
  const parent = new Int32Array(nodeCount);
  parent.fill(-1);
  const iter = new Int32Array(nodeCount);
  const valueSum = new Float64Array(nodeCount);
  const sepSum = new Float64Array(nodeCount);

  for (let i = 0; i < N; i++) valueSum[i] = townValueAt[i];

  const stack = new Int32Array(nodeCount);
  let sp = 0;
  stack[sp++] = root;

  let time = 1;

  while (sp > 0) {
    const u = stack[sp - 1];
    if (disc[u] === 0) {
      disc[u] = time;
      low[u] = time;
      time++;
      iter[u] = 0;
    }

    let v = -1;
    if (u === root) {
      while (true) {
        const idx = iter[u]++;
        if (idx >= ignitionList.length) {
          v = -1;
          break;
        }
        const cand = ignitionList[idx];
        if (mark[cand] === baseMarkId) {
          v = cand;
          break;
        }
      }
    } else {
      // Only traverse within baseline reachable region, plus the root edge on ignition nodes.
      while (true) {
        const k = iter[u]++;
        if (k === 0) {
          if (u >= width) v = u - width;
          else continue;
        } else if (k === 1) {
          if (u < N - width) v = u + width;
          else continue;
        } else if (k === 2) {
          if (u % width !== 0) v = u - 1;
          else continue;
        } else if (k === 3) {
          if (u % width !== width - 1) v = u + 1;
          else continue;
        } else if (k === 4) {
          if (isIgnition[u] === 1) v = root;
          else continue;
        } else {
          v = -1;
          break;
        }

        if (v !== root && mark[v] !== baseMarkId) {
          v = -1;
          continue;
        }
        break;
      }
    }

    if (v === -1) {
      // finish u
      sp--;
      const p = parent[u];
      if (p !== -1) {
        // subtree sum accumulation
        valueSum[p] += valueSum[u];
        if (low[u] < low[p]) low[p] = low[u];
        if (low[u] >= disc[p]) sepSum[p] += valueSum[u];
      }
      continue;
    }

    if (disc[v] === 0) {
      parent[v] = u;
      stack[sp++] = v;
      continue;
    }

    // back-edge (ignore edge to parent)
    if (v !== parent[u] && disc[v] < low[u]) low[u] = disc[v];
  }

  const candidates: { id: number; benefit: number }[] = [];
  for (let id = 0; id < N; id++) {
    if (mark[id] !== baseMarkId) continue;
    if (burnable[id] !== 1 || cut[id] === 1 || isIgnition[id] === 1) continue;
    const benefit = sepSum[id];
    if (benefit > 0) candidates.push({ id, benefit });
  }
  candidates.sort((a, b) => b.benefit - a.benefit);

  for (let i = 0; i < candidates.length && budgetLeft > 0; i++) {
    const c = candidates[i];
    if (c.benefit < bestTownDensity) break;
    const id = c.id;
    if (cut[id] === 1) continue;
    cut[id] = 1;
    selectedCuts.push(id);
    budgetLeft--;
  }
}

// Town ring heuristic on the post-articulation reachability.
if (budgetLeft > 0) {
  const afterArtMarkId = markId++;
  bfsMarkReachable(afterArtMarkId);

  const ringTmp: number[] = [];
  const townCandidates: { value: number; cost: number; cells: number[] }[] = [];
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    const id = t.y * width + t.x;
    if (mark[id] !== afterArtMarkId) continue;
    ringCellsFor(id, ringTmp);
    const cost = ringTmp.length;
    if (cost <= 0) continue;
    townCandidates.push({ value: t.value, cost, cells: ringTmp.slice() });
  }

  townCandidates.sort((a, b) => {
    const da = a.value / a.cost;
    const db = b.value / b.cost;
    if (db !== da) return db - da;
    if (b.value !== a.value) return b.value - a.value;
    return a.cost - b.cost;
  });

  for (let i = 0; i < townCandidates.length && budgetLeft > 0; i++) {
    const cand = townCandidates[i];
    // marginal cost given existing cuts
    let marginal = 0;
    for (let j = 0; j < cand.cells.length; j++) {
      const id = cand.cells[j];
      if (cut[id] === 0) marginal++;
    }
    if (marginal === 0 || marginal > budgetLeft) continue;

    for (let j = 0; j < cand.cells.length && budgetLeft > 0; j++) {
      const id = cand.cells[j];
      if (cut[id] === 1) continue;
      cut[id] = 1;
      selectedCuts.push(id);
      budgetLeft--;
    }
  }
}

// Extra cuts: BFS-tree criticality heuristic (limited).
if (budgetLeft > 0) {
  const dist = new Int32Array(N);
  const parentTree = new Int32Array(N);
  parentTree.fill(-1);
  const predCount = new Uint8Array(N);
  const subtree = new Float64Array(N);

  const treeMarkId = markId++;
  let head = 0;
  let tail = 0;

  for (let i = 0; i < ignitionList.length; i++) {
    const s = ignitionList[i];
    if (burnable[s] !== 1 || cut[s] === 1) continue;
    if (mark[s] === treeMarkId) continue;
    mark[s] = treeMarkId;
    dist[s] = 0;
    parentTree[s] = -1;
    predCount[s] = 0;
    queue[tail++] = s;
  }

  while (head < tail) {
    const u = queue[head++];
    const du = dist[u];
    const y = (u / width) | 0;
    const x = u - y * width;

    // up
    if (y > 0) {
      const v = u - width;
      if (burnable[v] === 1 && cut[v] === 0) {
        if (mark[v] !== treeMarkId) {
          mark[v] = treeMarkId;
          dist[v] = du + 1;
          parentTree[v] = u;
          predCount[v] = 1;
          queue[tail++] = v;
        } else if (dist[v] === du + 1 && predCount[v] < 255) {
          predCount[v]++;
        }
      }
    }
    // down
    if (y + 1 < height) {
      const v = u + width;
      if (burnable[v] === 1 && cut[v] === 0) {
        if (mark[v] !== treeMarkId) {
          mark[v] = treeMarkId;
          dist[v] = du + 1;
          parentTree[v] = u;
          predCount[v] = 1;
          queue[tail++] = v;
        } else if (dist[v] === du + 1 && predCount[v] < 255) {
          predCount[v]++;
        }
      }
    }
    // left
    if (x > 0) {
      const v = u - 1;
      if (burnable[v] === 1 && cut[v] === 0) {
        if (mark[v] !== treeMarkId) {
          mark[v] = treeMarkId;
          dist[v] = du + 1;
          parentTree[v] = u;
          predCount[v] = 1;
          queue[tail++] = v;
        } else if (dist[v] === du + 1 && predCount[v] < 255) {
          predCount[v]++;
        }
      }
    }
    // right
    if (x + 1 < width) {
      const v = u + 1;
      if (burnable[v] === 1 && cut[v] === 0) {
        if (mark[v] !== treeMarkId) {
          mark[v] = treeMarkId;
          dist[v] = du + 1;
          parentTree[v] = u;
          predCount[v] = 1;
          queue[tail++] = v;
        } else if (dist[v] === du + 1 && predCount[v] < 255) {
          predCount[v]++;
        }
      }
    }
  }

  // Initialize subtree values with town values for towns still at risk.
  for (let i = 0; i < towns.length; i++) {
    const t = towns[i];
    const id = t.y * width + t.x;
    if (mark[id] === treeMarkId) subtree[id] += t.value;
  }

  // Accumulate subtree sums bottom-up (reverse BFS order).
  for (let i = tail - 1; i >= 0; i--) {
    const u = queue[i];
    const p = parentTree[u];
    if (p !== -1) subtree[p] += subtree[u];
  }

  // Fixed-size min-heap to keep best candidates.
  const cap = Math.min(budgetLeft, 5000);
  if (cap > 0) {
    const heapId = new Int32Array(cap);
    const heapScore = new Float64Array(cap);
    let hs = 0;

    function heapSwap(i: number, j: number): void {
      const ti = heapId[i];
      heapId[i] = heapId[j];
      heapId[j] = ti;
      const ts = heapScore[i];
      heapScore[i] = heapScore[j];
      heapScore[j] = ts;
    }
    function heapUp(i: number): void {
      while (i > 0) {
        const p = (i - 1) >> 1;
        if (heapScore[p] <= heapScore[i]) break;
        heapSwap(i, p);
        i = p;
      }
    }
    function heapDown(i: number): void {
      while (true) {
        const l = i * 2 + 1;
        if (l >= hs) break;
        const r = l + 1;
        let m = l;
        if (r < hs && heapScore[r] < heapScore[l]) m = r;
        if (heapScore[i] <= heapScore[m]) break;
        heapSwap(i, m);
        i = m;
      }
    }
    function heapConsider(id: number, score: number): void {
      if (hs < cap) {
        heapId[hs] = id;
        heapScore[hs] = score;
        heapUp(hs);
        hs++;
        return;
      }
      if (score <= heapScore[0]) return;
      heapId[0] = id;
      heapScore[0] = score;
      heapDown(0);
    }

    for (let i = 0; i < tail; i++) {
      const u = queue[i];
      if (cut[u] === 1 || burnable[u] !== 1 || isIgnition[u] === 1) continue;
      const val = subtree[u];
      if (val <= 0) continue;
      const pc = predCount[u];
      if (pc === 0 || pc > 2) continue;
      const score = val / pc;
      heapConsider(u, score);
    }

    const picks: { id: number; score: number }[] = new Array(hs);
    for (let i = 0; i < hs; i++) picks[i] = { id: heapId[i], score: heapScore[i] };
    picks.sort((a, b) => b.score - a.score);

    for (let i = 0; i < picks.length && budgetLeft > 0; i++) {
      const id = picks[i].id;
      if (cut[id] === 1) continue;
      cut[id] = 1;
      selectedCuts.push(id);
      budgetLeft--;
    }
  }
}

// Output.
const firebreaks: number[][] = new Array(selectedCuts.length);
for (let i = 0; i < selectedCuts.length; i++) {
  const id = selectedCuts[i];
  const y = (id / width) | 0;
  const x = id - y * width;
  firebreaks[i] = [x, y];
}
process.stdout.write(JSON.stringify({ firebreaks }));
