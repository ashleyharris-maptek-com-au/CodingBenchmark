use std::io::{self, Read, Write};

const UNVISITED: u32 = u32::MAX;

#[derive(Clone, Copy)]
struct BaryKey {
    num: i64,
    den: i64,
    orig: i32,
    node: u32,
}

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    fn next_usize(&mut self) -> Option<usize> {
        while self.idx < self.buf.len() && self.buf[self.idx].is_ascii_whitespace() {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let mut n: usize = 0;
        while self.idx < self.buf.len() {
            let b = self.buf[self.idx];
            if !b.is_ascii_digit() {
                break;
            }
            n = n.saturating_mul(10).saturating_add((b - b'0') as usize);
            self.idx += 1;
        }
        Some(n)
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = sc.next_usize().unwrap_or(0);
    if n == 0 {
        return;
    }

    // Pass 1: degrees
    let edges_start = sc.idx;
    let mut deg = vec![0usize; n];
    for _ in 0..m {
        let Some(u) = sc.next_usize() else { break };
        let Some(v) = sc.next_usize() else { break };
        if u < n && v < n {
            deg[u] = deg[u].saturating_add(1);
            deg[v] = deg[v].saturating_add(1);
        }
    }

    // CSR adjacency
    let mut off = vec![0usize; n + 1];
    for i in 0..n {
        off[i + 1] = off[i].saturating_add(deg[i]);
    }
    let total_adj = off[n];
    let mut adj = vec![0u32; total_adj];
    let mut next = off.clone();

    // Pass 2: fill adjacency
    sc.idx = edges_start;
    for _ in 0..m {
        let Some(u) = sc.next_usize() else { break };
        let Some(v) = sc.next_usize() else { break };
        if u >= n || v >= n {
            continue;
        }
        let u32u = u as u32;
        let u32v = v as u32;
        let pu = next[u];
        if pu < off[u + 1] {
            adj[pu] = u32v;
            next[u] = pu + 1;
        }
        let pv = next[v];
        if pv < off[v + 1] {
            adj[pv] = u32u;
            next[v] = pv + 1;
        }
    }

    // Output coords
    let mut x = vec![0i64; n];
    let mut y = vec![0i64; n];

    // Component + BFS tree workspace
    let mut comp_vis = vec![0u8; n];
    let mut comp_nodes: Vec<u32> = Vec::new();
    let mut queue: Vec<u32> = Vec::new();
    let mut parent = vec![UNVISITED; n];
    let mut depth = vec![0u32; n];

    // Layered crossing-reduction workspace
    let mut pos = vec![0i32; n];
    let mut preorder: Vec<u32> = Vec::new();
    let mut layers: Vec<Vec<u32>> = Vec::new();
    let mut keys_buf: Vec<BaryKey> = Vec::new();
    let mut layer_counts: Vec<usize> = Vec::new();

    // Pack components in rows
    let mut cursor_x: i64 = 0;
    let mut cursor_y: i64 = 0;
    let mut row_h: i64 = 0;
    const PAD: i64 = 8;
    const ROW_LIMIT: i64 = 8000;

    for start in 0..n {
        if comp_vis[start] != 0 {
            continue;
        }

        // Discover component, also pick a reasonable root (max degree)
        comp_nodes.clear();
        queue.clear();
        comp_vis[start] = 1;
        queue.push(start as u32);
        comp_nodes.push(start as u32);
        let mut qh = 0usize;
        let mut root = start as u32;
        let mut best_deg = deg[start];
        let mut sum_deg: u64 = 0;
        while qh < queue.len() {
            let u = queue[qh] as usize;
            qh += 1;
            let du = deg[u];
            sum_deg += du as u64;
            if du > best_deg {
                best_deg = du;
                root = u as u32;
            }
            let a = off[u];
            let b = off[u + 1];
            for &vv in &adj[a..b] {
                let v = vv as usize;
                if v >= n {
                    continue;
                }
                if comp_vis[v] == 0 {
                    comp_vis[v] = 1;
                    queue.push(v as u32);
                    comp_nodes.push(v as u32);
                }
            }
        }

        let comp_n = comp_nodes.len();
        let edges_in_comp = (sum_deg / 2) as i64;
        let is_tree = edges_in_comp == (comp_n as i64).saturating_sub(1);

        // Reset parent markers for this component
        for &v in &comp_nodes {
            parent[v as usize] = UNVISITED;
        }

        // BFS tree from chosen root
        queue.clear();
        let r = root as usize;
        parent[r] = root;
        depth[r] = 0;
        queue.push(root);
        let mut qh2 = 0usize;
        let mut max_depth: u32 = 0;
        while qh2 < queue.len() {
            let uu = queue[qh2];
            qh2 += 1;
            let u = uu as usize;
            let du = depth[u];
            if du > max_depth {
                max_depth = du;
            }
            let a = off[u];
            let b = off[u + 1];
            for &vv in &adj[a..b] {
                let v = vv as usize;
                if v >= n {
                    continue;
                }
                if parent[v] == UNVISITED {
                    parent[v] = uu;
                    depth[v] = du.saturating_add(1);
                    queue.push(v as u32);
                }
            }
        }

        // Compute DFS preorder of the BFS tree (used both for tree drawing and as init order)
        preorder.clear();
        {
            // Iterative DFS on the BFS-tree edges only: maintain a per-node cursor into its adjacency list.
            let mut stack_nodes: Vec<u32> = Vec::new();
            let mut stack_idx: Vec<usize> = Vec::new();

            stack_nodes.push(root);
            stack_idx.push(off[r]);
            preorder.push(root);

            while !stack_nodes.is_empty() {
                let uu = *stack_nodes.last().unwrap();
                let u = uu as usize;
                let top = stack_nodes.len() - 1;

                let mut idx = stack_idx[top];
                let end = off[u + 1];
                let mut child: Option<u32> = None;
                while idx < end {
                    let vv = adj[idx] as usize;
                    idx += 1;
                    if vv >= n {
                        continue;
                    }
                    if parent[vv] == uu && vv != u {
                        child = Some(vv as u32);
                        break;
                    }
                }
                stack_idx[top] = idx;

                if let Some(vv) = child {
                    stack_nodes.push(vv);
                    stack_idx.push(off[vv as usize]);
                    preorder.push(vv);
                } else {
                    stack_nodes.pop();
                    stack_idx.pop();
                }
            }
        }

        // Place component box in packing grid (avoid overlaps so components' edges can't cross each other).
        let max_layer_size: usize = if is_tree {
            comp_n.max(1)
        } else {
            layer_counts.clear();
            layer_counts.resize((max_depth as usize) + 1, 0);
            let mut max_layer = 1usize;
            for &v in &comp_nodes {
                let d = depth[v as usize] as usize;
                let c = &mut layer_counts[d];
                *c += 1;
                if *c > max_layer {
                    max_layer = *c;
                }
            }
            max_layer
        };

        let comp_w_est = (max_layer_size as i64) * 2 + PAD;
        let comp_h_est = (max_depth as i64 + 1) * 2 + PAD;
        if cursor_x != 0 && cursor_x.saturating_add(comp_w_est) > ROW_LIMIT {
            cursor_x = 0;
            cursor_y = cursor_y.saturating_add(row_h);
            row_h = 0;
        }
        let base_x = cursor_x;
        let base_y = cursor_y;
        cursor_x = cursor_x.saturating_add(comp_w_est);
        row_h = row_h.max(comp_h_est);

        if is_tree {
            // Simple planar tree drawing: x by preorder index, y by BFS depth.
            for (i, &v) in preorder.iter().enumerate() {
                let d = depth[v as usize] as i64;
                x[v as usize] = base_x + (i as i64) * 2 + (d & 1);
                y[v as usize] = base_y + d * 2;
            }
            continue;
        }

        // Layered layout: nodes grouped by BFS depth, ordered by barycenter sweeps.
        layers.clear();
        layers.resize((max_depth as usize) + 1, Vec::new());
        for &v in &preorder {
            let d = depth[v as usize] as usize;
            layers[d].push(v);
        }

        // Initialize positions
        for layer in &layers {
            for (i, &v) in layer.iter().enumerate() {
                pos[v as usize] = i as i32;
            }
        }

        let iters = match comp_n {
            0..=2000 => 4,
            2001..=20000 => 2,
            20001..=120000 => 1,
            _ => 0,
        };

        for _ in 0..iters {
            // Down sweep
            for li in 1..layers.len() {
                let ref_layer = li - 1;
                let layer = &mut layers[li];
                if layer.len() <= 1 {
                    continue;
                }
                keys_buf.clear();
                keys_buf.reserve(layer.len());
                for (orig_idx, &v) in layer.iter().enumerate() {
                    let mut sum: i64 = 0;
                    let mut cnt: i64 = 0;
                    let vv = v as usize;
                    for &u in &adj[off[vv]..off[vv + 1]] {
                        let uu = u as usize;
                        if uu >= n {
                            continue;
                        }
                        if depth[uu] as usize == ref_layer {
                            sum += pos[uu] as i64;
                            cnt += 1;
                        }
                    }
                    if cnt == 0 {
                        sum = orig_idx as i64;
                        cnt = 1;
                    }
                    keys_buf.push(BaryKey {
                        num: sum,
                        den: cnt,
                        orig: orig_idx as i32,
                        node: v,
                    });
                }
                keys_buf.sort_unstable_by(|a, b| {
                    let lhs = (a.num as i128) * (b.den as i128);
                    let rhs = (b.num as i128) * (a.den as i128);
                    lhs.cmp(&rhs).then(a.orig.cmp(&b.orig))
                });
                for (i, k) in keys_buf.iter().enumerate() {
                    layer[i] = k.node;
                    pos[k.node as usize] = i as i32;
                }
            }

            // Up sweep
            if layers.len() >= 2 {
                for li in (0..layers.len() - 1).rev() {
                    let ref_layer = li + 1;
                    let layer = &mut layers[li];
                    if layer.len() <= 1 {
                        continue;
                    }
                    keys_buf.clear();
                    keys_buf.reserve(layer.len());
                    for (orig_idx, &v) in layer.iter().enumerate() {
                        let mut sum: i64 = 0;
                        let mut cnt: i64 = 0;
                        let vv = v as usize;
                        for &u in &adj[off[vv]..off[vv + 1]] {
                            let uu = u as usize;
                            if uu >= n {
                                continue;
                            }
                            if depth[uu] as usize == ref_layer {
                                sum += pos[uu] as i64;
                                cnt += 1;
                            }
                        }
                        if cnt == 0 {
                            sum = orig_idx as i64;
                            cnt = 1;
                        }
                        keys_buf.push(BaryKey {
                            num: sum,
                            den: cnt,
                            orig: orig_idx as i32,
                            node: v,
                        });
                    }
                    keys_buf.sort_unstable_by(|a, b| {
                        let lhs = (a.num as i128) * (b.den as i128);
                        let rhs = (b.num as i128) * (a.den as i128);
                        lhs.cmp(&rhs).then(a.orig.cmp(&b.orig))
                    });
                    for (i, k) in keys_buf.iter().enumerate() {
                        layer[i] = k.node;
                        pos[k.node as usize] = i as i32;
                    }
                }
            }
        }

        // Assign final integer coordinates for this component
        for (li, layer) in layers.iter().enumerate() {
            let ly = li as i64;
            for (pi, &v) in layer.iter().enumerate() {
                x[v as usize] = base_x + (pi as i64) * 2 + (ly & 1);
                y[v as usize] = base_y + ly * 2;
            }
        }
    }

    // Output: integers are valid float literals.
    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 20, stdout.lock());
    for i in 0..n {
        let _ = writeln!(out, "{} {}", x[i], y[i]);
    }
}
