// Job-Shop Scheduling (JSSP) solver
// - Always outputs a feasible schedule
// - Uses adaptive strategies depending on instance scale
// - MSVC /std:c++20 compatible

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <utility>
#include <vector>

using i64 = long long;

struct Task {
    int machine = 0;
    i64 duration = 0;
};

struct Instance {
    int jobs = 0;
    int machines = 0;
    std::vector<std::vector<Task>> tasks;        // tasks[j][k]
    std::vector<std::vector<i64>> suffix_dur;    // suffix_dur[j][k] = sum_{t>=k} duration
    i64 total_ops = 0;
};

struct Schedule {
    std::vector<std::vector<i64>> start; // start[j][k]
    i64 makespan = 0;
};

static Schedule schedule_baseline_by_job_order(const Instance& inst, const std::vector<int>& job_order) {
    std::vector<i64> machine_ready(inst.machines, 0);
    std::vector<i64> job_ready(inst.jobs, 0);

    Schedule s;
    s.start.resize(inst.jobs);
    for (int j = 0; j < inst.jobs; ++j) s.start[j].assign(inst.tasks[j].size(), 0);

    for (int jj = 0; jj < (int)job_order.size(); ++jj) {
        int j = job_order[jj];
        const auto& jt = inst.tasks[j];
        for (int k = 0; k < (int)jt.size(); ++k) {
            const Task& t = jt[k];
            i64 st = std::max(job_ready[j], machine_ready[t.machine]);
            s.start[j][k] = st;
            i64 en = st + t.duration;
            job_ready[j] = en;
            machine_ready[t.machine] = en;
            if (en > s.makespan) s.makespan = en;
        }
    }
    return s;
}

enum class ConflictRule {
    MostWorkRemaining = 0,
    ShortestProcTime = 1,
    LongestProcTime = 2,
    EarliestStart = 3,
    PickHat = 4,
};

static int pick_from_conflict_set(const Instance& inst,
                                 const std::vector<int>& conflict_jobs,
                                 const std::vector<int>& next_op,
                                 const std::vector<i64>& est,
                                 ConflictRule rule,
                                 int job_hat) {
    auto better = [&](int a, int b) -> bool {
        if (a == b) return false;
        int oa = next_op[a], ob = next_op[b];
        const Task& ta = inst.tasks[a][oa];
        const Task& tb = inst.tasks[b][ob];
        i64 rema = inst.suffix_dur[a][oa];
        i64 remb = inst.suffix_dur[b][ob];

        switch (rule) {
            case ConflictRule::MostWorkRemaining:
                if (rema != remb) return rema > remb;
                if (ta.duration != tb.duration) return ta.duration > tb.duration;
                if (est[a] != est[b]) return est[a] < est[b];
                return a < b;
            case ConflictRule::ShortestProcTime:
                if (ta.duration != tb.duration) return ta.duration < tb.duration;
                if (est[a] != est[b]) return est[a] < est[b];
                if (rema != remb) return rema > remb;
                return a < b;
            case ConflictRule::LongestProcTime:
                if (ta.duration != tb.duration) return ta.duration > tb.duration;
                if (est[a] != est[b]) return est[a] < est[b];
                if (rema != remb) return rema > remb;
                return a < b;
            case ConflictRule::EarliestStart:
                if (est[a] != est[b]) return est[a] < est[b];
                if (rema != remb) return rema > remb;
                if (ta.duration != tb.duration) return ta.duration > tb.duration;
                return a < b;
            case ConflictRule::PickHat:
                // Only used as "always pick job_hat if present"; fallback to earliest start.
                if (a == job_hat) return true;
                if (b == job_hat) return false;
                if (est[a] != est[b]) return est[a] < est[b];
                return a < b;
        }
        return a < b;
    };

    int best = conflict_jobs.front();
    for (int idx = 1; idx < (int)conflict_jobs.size(); ++idx) {
        int j = conflict_jobs[idx];
        if (better(j, best)) best = j;
    }
    return best;
}

// Giffler-Thompson style appending schedule generation scheme (fast; scales well).
static Schedule schedule_gt_appending(const Instance& inst, ConflictRule rule) {
    std::vector<int> next_op(inst.jobs, 0);
    std::vector<i64> job_ready(inst.jobs, 0);
    std::vector<i64> machine_ready(inst.machines, 0);

    Schedule s;
    s.start.resize(inst.jobs);
    for (int j = 0; j < inst.jobs; ++j) s.start[j].assign(inst.tasks[j].size(), 0);

    // Active job list to avoid scanning finished jobs unnecessarily.
    std::vector<int> active;
    active.reserve(inst.jobs);
    for (int j = 0; j < inst.jobs; ++j) {
        if (!inst.tasks[j].empty()) active.push_back(j);
    }

    std::vector<i64> est(inst.jobs, 0);
    std::vector<i64> ect(inst.jobs, 0);
    std::vector<int> op_machine(inst.jobs, 0);

    i64 remaining_ops = inst.total_ops;
    while (remaining_ops > 0) {
        int job_hat = -1;
        i64 best_ect = std::numeric_limits<i64>::max();

        // Compute ESA/EC for available operations and pick hat op with smallest EC.
        for (int idx = 0; idx < (int)active.size(); ++idx) {
            int j = active[idx];
            int k = next_op[j];
            const Task& t = inst.tasks[j][k];
            op_machine[j] = t.machine;
            i64 st = std::max(job_ready[j], machine_ready[t.machine]);
            est[j] = st;
            i64 en = st + t.duration;
            ect[j] = en;

            if (en < best_ect || (en == best_ect && j < job_hat)) {
                best_ect = en;
                job_hat = j;
            }
        }

        // Build conflict set on the machine of hat operation.
        const int machine_hat = op_machine[job_hat];
        std::vector<int> conflict;
        conflict.reserve(8);
        for (int idx = 0; idx < (int)active.size(); ++idx) {
            int j = active[idx];
            if (op_machine[j] != machine_hat) continue;
            // Conflict definition: ESA_hat + p_hat > ESA_j  <=> EC_hat > ESA_j
            if (best_ect > est[j]) conflict.push_back(j);
        }
        // Corner case: duration==0 can make conflict empty; ensure progress.
        if (conflict.empty()) conflict.push_back(job_hat);

        int jsel = pick_from_conflict_set(inst, conflict, next_op, est, rule, job_hat);
        int ksel = next_op[jsel];
        const Task& tsel = inst.tasks[jsel][ksel];

        i64 st = std::max(job_ready[jsel], machine_ready[tsel.machine]);
        s.start[jsel][ksel] = st;
        i64 en = st + tsel.duration;

        job_ready[jsel] = en;
        machine_ready[tsel.machine] = en;
        if (en > s.makespan) s.makespan = en;

        ++next_op[jsel];
        --remaining_ops;

        if (next_op[jsel] >= (int)inst.tasks[jsel].size()) {
            // Remove finished job from active list (swap-remove).
            for (int idx = 0; idx < (int)active.size(); ++idx) {
                if (active[idx] == jsel) {
                    active[idx] = active.back();
                    active.pop_back();
                    break;
                }
            }
        }
    }

    return s;
}

struct Interval {
    i64 s = 0;
    i64 e = 0;
};

static std::pair<i64, int> earliest_insertion_slot(const std::vector<Interval>& ivs, i64 ready, i64 dur) {
    if (ivs.empty()) return {ready, 0};

    // Before first
    if (ready + dur <= ivs.front().s) return {ready, 0};

    i64 t = std::max(ready, ivs.front().e);
    for (int i = 0; i + 1 < (int)ivs.size(); ++i) {
        i64 gap_start = std::max(t, ivs[i].e);
        i64 gap_end = ivs[i + 1].s;
        if (gap_start + dur <= gap_end) return {gap_start, i + 1};
        t = std::max(t, ivs[i + 1].e);
    }
    // After last
    return {std::max(ready, ivs.back().e), (int)ivs.size()};
}

enum class InsertHeuristic {
    MinECT = 0,                // earliest completion time
    MinEST = 1,                // earliest start time
    MinECT_MaxRemainder = 2,   // tie-break by remaining work
    MinEST_MaxRemainder = 3,   // tie-break by remaining work
    MinECT_Shorter = 4,        // tie-break by shorter processing time
};

static Schedule schedule_insertion_greedy(const Instance& inst, InsertHeuristic h) {
    std::vector<int> next_op(inst.jobs, 0);
    std::vector<i64> job_ready(inst.jobs, 0);
    std::vector<std::vector<Interval>> machine_ivs(inst.machines);

    Schedule s;
    s.start.resize(inst.jobs);
    for (int j = 0; j < inst.jobs; ++j) s.start[j].assign(inst.tasks[j].size(), 0);

    std::vector<int> active;
    active.reserve(inst.jobs);
    for (int j = 0; j < inst.jobs; ++j) if (!inst.tasks[j].empty()) active.push_back(j);

    i64 remaining_ops = inst.total_ops;
    while (remaining_ops > 0) {
        int best_j = -1;
        i64 best_est = 0, best_ect = 0;
        int best_pos = 0;

        for (int idx = 0; idx < (int)active.size(); ++idx) {
            int j = active[idx];
            int k = next_op[j];
            const Task& t = inst.tasks[j][k];

            auto [est, pos] = earliest_insertion_slot(machine_ivs[t.machine], job_ready[j], t.duration);
            i64 ect = est + t.duration;
            i64 rem = inst.suffix_dur[j][k];

            auto better = [&](int cand_j, i64 cand_est, i64 cand_ect, int /*cand_pos*/,
                              int cur_j, i64 cur_est, i64 cur_ect, int /*cur_pos*/) -> bool {
                int ck = next_op[cand_j], ok = next_op[cur_j];
                const Task& ct = inst.tasks[cand_j][ck];
                const Task& ot = inst.tasks[cur_j][ok];
                i64 crem = inst.suffix_dur[cand_j][ck];
                i64 orem = inst.suffix_dur[cur_j][ok];

                switch (h) {
                    case InsertHeuristic::MinECT:
                        if (cand_ect != cur_ect) return cand_ect < cur_ect;
                        if (cand_est != cur_est) return cand_est < cur_est;
                        return cand_j < cur_j;
                    case InsertHeuristic::MinEST:
                        if (cand_est != cur_est) return cand_est < cur_est;
                        if (cand_ect != cur_ect) return cand_ect < cur_ect;
                        return cand_j < cur_j;
                    case InsertHeuristic::MinECT_MaxRemainder:
                        if (cand_ect != cur_ect) return cand_ect < cur_ect;
                        if (crem != orem) return crem > orem;
                        if (ct.duration != ot.duration) return ct.duration > ot.duration;
                        return cand_j < cur_j;
                    case InsertHeuristic::MinEST_MaxRemainder:
                        if (cand_est != cur_est) return cand_est < cur_est;
                        if (crem != orem) return crem > orem;
                        if (cand_ect != cur_ect) return cand_ect < cur_ect;
                        return cand_j < cur_j;
                    case InsertHeuristic::MinECT_Shorter:
                        if (cand_ect != cur_ect) return cand_ect < cur_ect;
                        if (ct.duration != ot.duration) return ct.duration < ot.duration;
                        if (cand_est != cur_est) return cand_est < cur_est;
                        return cand_j < cur_j;
                }
                return cand_j < cur_j;
            };

            if (best_j == -1 || better(j, est, ect, pos, best_j, best_est, best_ect, best_pos)) {
                best_j = j;
                best_est = est;
                best_ect = ect;
                best_pos = pos;
                (void)rem;
            }
        }

        int jsel = best_j;
        int ksel = next_op[jsel];
        const Task& tsel = inst.tasks[jsel][ksel];

        s.start[jsel][ksel] = best_est;
        job_ready[jsel] = best_ect;
        if (best_ect > s.makespan) s.makespan = best_ect;

        // Insert interval while preserving sort by start time.
        auto& ivs = machine_ivs[tsel.machine];
        ivs.insert(ivs.begin() + best_pos, Interval{best_est, best_ect});

        ++next_op[jsel];
        --remaining_ops;

        if (next_op[jsel] >= (int)inst.tasks[jsel].size()) {
            for (int idx = 0; idx < (int)active.size(); ++idx) {
                if (active[idx] == jsel) {
                    active[idx] = active.back();
                    active.pop_back();
                    break;
                }
            }
        }
    }

    return s;
}

static void print_schedule(const Instance& inst, const Schedule& s) {
    for (int j = 0; j < inst.jobs; ++j) {
        const int T = (int)inst.tasks[j].size();
        for (int k = 0; k < T; ++k) {
            if (k) std::cout << ' ';
            std::cout << s.start[j][k] << ' ' << inst.tasks[j][k].machine;
        }
        std::cout << '\n';
    }
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    Instance inst;
    int J = 0, M = 0;
    if (!(std::cin >> J >> M)) return 0;

    inst.jobs = J;
    inst.machines = M;
    inst.tasks.assign(J, {});

    int max_machine = M - 1;
    i64 total_ops = 0;
    for (int j = 0; j < J; ++j) {
        int T = 0;
        std::cin >> T;
        inst.tasks[j].resize(T);
        total_ops += T;
        for (int k = 0; k < T; ++k) {
            int machine = 0;
            i64 duration = 0;
            std::cin >> machine >> duration;
            inst.tasks[j][k] = Task{machine, duration};
            if (machine > max_machine) max_machine = machine;
        }
    }
    inst.total_ops = total_ops;
    if (max_machine + 1 > inst.machines) inst.machines = max_machine + 1;

    inst.suffix_dur.resize(J);
    for (int j = 0; j < J; ++j) {
        const int T = (int)inst.tasks[j].size();
        inst.suffix_dur[j].assign(T + 1, 0);
        for (int k = T - 1; k >= 0; --k) {
            inst.suffix_dur[j][k] = inst.suffix_dur[j][k + 1] + inst.tasks[j][k].duration;
        }
    }

    // Trivial no-op instance.
    if (inst.total_ops == 0) {
        for (int j = 0; j < inst.jobs; ++j) std::cout << '\n';
        return 0;
    }

    // Adaptive strategy selection (scale-aware):
    // - Small: try insertion-based schedules (left-shift / gap insertion) + GT
    // - Medium/Large: multi-start GT with different conflict rules
    // - Extreme: limited GT variants only
    const bool small = (inst.total_ops <= 8000 && inst.jobs <= 2000 && inst.machines <= 200);
    const bool medium = (inst.total_ops <= 200000);

    Schedule best;
    best.makespan = std::numeric_limits<i64>::max();

    auto consider = [&](Schedule cand) {
        if (cand.makespan < best.makespan) best = std::move(cand);
    };

    // Baselines.
    std::vector<int> order_id(inst.jobs);
    std::iota(order_id.begin(), order_id.end(), 0);
    consider(schedule_baseline_by_job_order(inst, order_id));

    std::vector<std::pair<i64, int>> by_work;
    by_work.reserve(inst.jobs);
    for (int j = 0; j < inst.jobs; ++j) by_work.push_back({inst.suffix_dur[j][0], j});
    std::sort(by_work.begin(), by_work.end(), [&](const auto& a, const auto& b) {
        if (a.first != b.first) return a.first > b.first;
        return a.second < b.second;
    });
    std::vector<int> order_work;
    order_work.reserve(inst.jobs);
    for (auto& p : by_work) order_work.push_back(p.second);
    consider(schedule_baseline_by_job_order(inst, order_work));

    if (small) {
        // Higher-quality (but slower) left-shift insertion construction, multiple heuristics.
        consider(schedule_insertion_greedy(inst, InsertHeuristic::MinECT));
        consider(schedule_insertion_greedy(inst, InsertHeuristic::MinEST));
        consider(schedule_insertion_greedy(inst, InsertHeuristic::MinECT_MaxRemainder));
        consider(schedule_insertion_greedy(inst, InsertHeuristic::MinEST_MaxRemainder));
        consider(schedule_insertion_greedy(inst, InsertHeuristic::MinECT_Shorter));
    }

    // Multi-start GT with different conflict rules; cap variants by scale.
    std::vector<ConflictRule> gt_rules = {
        ConflictRule::MostWorkRemaining,
        ConflictRule::EarliestStart,
        ConflictRule::LongestProcTime,
        ConflictRule::ShortestProcTime,
        ConflictRule::PickHat,
    };

    int gt_variants = 1;
    if (small) gt_variants = 5;
    else if (medium) gt_variants = 3;
    else gt_variants = 2;

    for (int i = 0; i < gt_variants && i < (int)gt_rules.size(); ++i) {
        consider(schedule_gt_appending(inst, gt_rules[i]));
    }

    print_schedule(inst, best);
    return 0;
}
