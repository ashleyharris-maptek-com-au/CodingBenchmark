const fs = require('fs');

function safeNumber(x, fallback) {
  const v = Number(x);
  return Number.isFinite(v) ? v : fallback;
}

class BinaryHeap {
  constructor(isHigherPriority) {
    this._a = [];
    this._higher = isHigherPriority; // (x,y) => true if x should be above y
  }
  size() {
    return this._a.length;
  }
  push(x) {
    const a = this._a;
    const higher = this._higher;
    a.push(x);
    let i = a.length - 1;
    while (i > 0) {
      const p = (i - 1) >> 1;
      if (!higher(a[i], a[p])) break;
      const tmp = a[i];
      a[i] = a[p];
      a[p] = tmp;
      i = p;
    }
  }
  pop() {
    const a = this._a;
    const n = a.length;
    if (n === 0) return undefined;
    const top = a[0];
    const last = a.pop();
    if (n > 1) {
      a[0] = last;
      const higher = this._higher;
      let i = 0;
      for (;;) {
        const l = i * 2 + 1;
        if (l >= a.length) break;
        let best = l;
        const r = l + 1;
        if (r < a.length && higher(a[r], a[l])) best = r;
        if (!higher(a[best], a[i])) break;
        const tmp = a[i];
        a[i] = a[best];
        a[best] = tmp;
        i = best;
      }
    }
    return top;
  }
}

class RunningHeap {
  constructor() {
    this.end = [];
    this.machine = [];
    this.task = [];
  }
  size() {
    return this.end.length;
  }
  peekEnd() {
    return this.end[0];
  }
  push(endTime, machineId, taskId) {
    const end = this.end;
    const machine = this.machine;
    const task = this.task;
    let i = end.length;
    end.push(endTime);
    machine.push(machineId);
    task.push(taskId);
    while (i > 0) {
      const p = (i - 1) >> 1;
      const pe = end[p];
      if (pe < endTime || (pe === endTime && machine[p] <= machineId)) break;
      end[i] = pe;
      machine[i] = machine[p];
      task[i] = task[p];
      i = p;
    }
    end[i] = endTime;
    machine[i] = machineId;
    task[i] = taskId;
  }
  pop() {
    const end = this.end;
    const machine = this.machine;
    const task = this.task;
    const n = end.length;
    if (n === 0) return null;
    const end0 = end[0];
    const machine0 = machine[0];
    const task0 = task[0];
    const lastEnd = end.pop();
    const lastMachine = machine.pop();
    const lastTask = task.pop();
    if (n > 1) {
      let i = 0;
      const size = end.length;
      for (;;) {
        const l = i * 2 + 1;
        if (l >= size) break;
        let best = l;
        const r = l + 1;
        if (
          r < size &&
          (end[r] < end[l] || (end[r] === end[l] && machine[r] < machine[l]))
        ) {
          best = r;
        }
        if (end[best] > lastEnd || (end[best] === lastEnd && machine[best] >= lastMachine)) break;
        end[i] = end[best];
        machine[i] = machine[best];
        task[i] = task[best];
        i = best;
      }
      end[i] = lastEnd;
      machine[i] = lastMachine;
      task[i] = lastTask;
    }
    return { end: end0, machine: machine0, task: task0 };
  }
}

function buildBottomLevels(n, duration, succ, indegree) {
  const indeg = indegree.slice();
  const q = [];
  for (let i = 0; i < n; i++) if (indeg[i] === 0) q.push(i);
  const topo = [];
  for (let qi = 0; qi < q.length; qi++) {
    const v = q[qi];
    topo.push(v);
    const s = succ[v];
    for (let k = 0; k < s.length; k++) {
      const u = s[k];
      indeg[u]--;
      if (indeg[u] === 0) q.push(u);
    }
  }
  const bottom = duration.slice();
  if (topo.length === n) {
    for (let ti = topo.length - 1; ti >= 0; ti--) {
      const v = topo[ti];
      let best = bottom[v];
      const s = succ[v];
      for (let k = 0; k < s.length; k++) {
        const u = s[k];
        const cand = duration[v] + bottom[u];
        if (cand > best) best = cand;
      }
      bottom[v] = best;
    }
  }
  return bottom;
}

function scheduleWithComparator({
  n,
  machineCount,
  ids,
  duration,
  deadline,
  weight,
  succ,
  indegree,
  higherPriority,
}) {
  if (!Number.isFinite(machineCount) || machineCount <= 0) return { feasible: false };

  const indeg = indegree.slice();
  const ready = new BinaryHeap(higherPriority);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) ready.push(i);

  const idle = [];
  for (let m = 0; m < machineCount; m++) idle.push(m);
  const running = new RunningHeap();

  let time = 0;
  let started = 0;

  const completion = new Array(n);
  const records = [];

  while (started < n) {
    while (idle.length > 0 && ready.size() > 0) {
      const taskId = ready.pop();
      const machineId = idle.pop();
      const start = time;
      const end = start + duration[taskId];

      records.push({ id: ids[taskId], start, machine: machineId });
      completion[taskId] = end;
      running.push(end, machineId, taskId);
      started++;
    }

    if (started >= n) break;
    if (running.size() === 0) return { feasible: false };

    time = running.peekEnd();
    while (running.size() > 0 && running.peekEnd() === time) {
      const ev = running.pop();
      idle.push(ev.machine);
      const s = succ[ev.task];
      for (let k = 0; k < s.length; k++) {
        const u = s[k];
        indeg[u]--;
        if (indeg[u] === 0) ready.push(u);
      }
    }
  }

  let makespan = 0;
  let weightedTardiness = 0;
  for (let i = 0; i < n; i++) {
    const c = completion[i];
    if (!Number.isFinite(c)) return { feasible: false };
    if (c > makespan) makespan = c;
    const d = deadline[i];
    if (Number.isFinite(d)) {
      const tard = c > d ? c - d : 0;
      weightedTardiness += weight[i] * tard;
    }
  }

  return { feasible: true, records, objective: weightedTardiness + makespan };
}

function topoOrderWithComparator(n, succ, indegree, higherPriority, ids) {
  const indeg = indegree.slice();
  const heap = new BinaryHeap(higherPriority);
  for (let i = 0; i < n; i++) if (indeg[i] === 0) heap.push(i);
  const order = [];
  while (heap.size() > 0) {
    const v = heap.pop();
    order.push(ids[v]);
    const s = succ[v];
    for (let k = 0; k < s.length; k++) {
      const u = s[k];
      indeg[u]--;
      if (indeg[u] === 0) heap.push(u);
    }
  }
  if (order.length !== n) {
    // Cycle or missing deps; fall back to input order.
    return ids.slice();
  }
  return order;
}

function main() {
  const input = fs.readFileSync(0, 'utf8');
  let data;
  try {
    data = JSON.parse(input);
  } catch {
    process.stdout.write(JSON.stringify({ order: [] }));
    return;
  }

  const tasks = Array.isArray(data.tasks) ? data.tasks : [];
  const n = tasks.length;
  if (n === 0) {
    process.stdout.write(JSON.stringify({ schedule: [] }));
    return;
  }

  const machineCount = safeNumber(data.machineCount, 1);

  const ids = new Array(n);
  const idStr = new Array(n);
  const duration = new Array(n);
  const deadline = new Array(n);
  const weight = new Array(n);
  const depsRaw = new Array(n);

  for (let i = 0; i < n; i++) {
    const t = tasks[i] || {};
    ids[i] = t.id;
    idStr[i] = String(t.id);
    duration[i] = safeNumber(t.duration, 0);
    deadline[i] =
      t.deadline === undefined || t.deadline === null ? Number.POSITIVE_INFINITY : safeNumber(t.deadline, Number.POSITIVE_INFINITY);
    weight[i] = t.weight === undefined || t.weight === null ? 1 : safeNumber(t.weight, 1);
    depsRaw[i] = Array.isArray(t.deps) ? t.deps : [];
  }

  const idxById = new Map();
  for (let i = 0; i < n; i++) idxById.set(idStr[i], i);

  const succ = Array.from({ length: n }, () => []);
  const indegree = new Array(n).fill(0);

  for (let i = 0; i < n; i++) {
    const deps = depsRaw[i];
    let cnt = 0;
    for (let k = 0; k < deps.length; k++) {
      const j = idxById.get(String(deps[k]));
      if (j === undefined) continue;
      succ[j].push(i);
      cnt++;
    }
    indegree[i] = cnt;
  }

  const bottom = buildBottomLevels(n, duration, succ, indegree);

  const eps = 1e-9;
  const density = new Array(n);
  const effSlack = new Array(n);
  for (let i = 0; i < n; i++) {
    const p = Math.max(duration[i], eps);
    const w = weight[i];
    const wPos = Math.max(w, eps);
    density[i] = w / p;
    effSlack[i] = (deadline[i] - bottom[i]) / wPos;
  }

  function higherSlack(a, b) {
    const sa = effSlack[a];
    const sb = effSlack[b];
    if (sa !== sb) return sa < sb;
    const da = deadline[a];
    const db = deadline[b];
    if (da !== db) return da < db;
    const ra = density[a];
    const rb = density[b];
    if (ra !== rb) return ra > rb;
    const ba = bottom[a];
    const bb = bottom[b];
    if (ba !== bb) return ba > bb;
    const pa = duration[a];
    const pb = duration[b];
    if (pa !== pb) return pa < pb;
    return idStr[a] < idStr[b];
  }

  function higherBottom(a, b) {
    const ba = bottom[a];
    const bb = bottom[b];
    if (ba !== bb) return ba > bb;
    const sa = effSlack[a];
    const sb = effSlack[b];
    if (sa !== sb) return sa < sb;
    const ra = density[a];
    const rb = density[b];
    if (ra !== rb) return ra > rb;
    const da = deadline[a];
    const db = deadline[b];
    if (da !== db) return da < db;
    const pa = duration[a];
    const pb = duration[b];
    if (pa !== pb) return pa < pb;
    return idStr[a] < idStr[b];
  }

  const attempt1 = scheduleWithComparator({
    n,
    machineCount,
    ids,
    duration,
    deadline,
    weight,
    succ,
    indegree,
    higherPriority: higherSlack,
  });

  const attempt2 = scheduleWithComparator({
    n,
    machineCount,
    ids,
    duration,
    deadline,
    weight,
    succ,
    indegree,
    higherPriority: higherBottom,
  });

  let best = null;
  if (attempt1.feasible) best = attempt1;
  if (attempt2.feasible && (!best || attempt2.objective < best.objective)) best = attempt2;

  if (best) {
    process.stdout.write(JSON.stringify({ schedule: best.records }));
    return;
  }

  const order = topoOrderWithComparator(n, succ, indegree, higherSlack, ids);
  process.stdout.write(JSON.stringify({ order }));
}

main();
