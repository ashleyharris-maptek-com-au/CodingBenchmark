using System;
using System.Diagnostics;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _ptr;

        public FastScanner(Stream stream, int bufferSize = 1 << 16)
        {
            _stream = stream;
            _buffer = new byte[bufferSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return int.MinValue;
            } while (c <= ' ');

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > ' ')
            {
                value = unchecked(value * 10 + (c - '0'));
                c = ReadByte();
            }
            return value * sign;
        }
    }

    private struct XorShift64Star
    {
        private ulong _state;

        public XorShift64Star(ulong seed)
        {
            _state = seed != 0 ? seed : 0x9E3779B97F4A7C15UL;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public ulong NextULong()
        {
            ulong x = _state;
            x ^= x >> 12;
            x ^= x << 25;
            x ^= x >> 27;
            _state = x;
            return x * 2685821657736338717UL;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int NextInt(int maxExclusive)
        {
            return (int)(NextULong() % (ulong)maxExclusive);
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public double NextDouble()
        {
            return (NextULong() >> 11) * (1.0 / (1UL << 53));
        }
    }

    private static int GetTimeLimitMs(int n)
    {
        string env = Environment.GetEnvironmentVariable("QAP_TIME_MS");
        int parsed;
        if (!string.IsNullOrWhiteSpace(env) && int.TryParse(env, out parsed) && parsed > 0)
            return parsed;

        int ms = 2000 + n * 12;
        if (ms < 3000) ms = 3000;
        if (ms > 120_000) ms = 120_000;
        return ms;
    }

    private static long ComputeCost(int n, int[] flow, int[] dist, int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int iRowA = i * n;
            int pi = perm[i];
            int piRowB = pi * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[iRowA + j] * dist[piRowB + perm[j]];
            }
        }
        return cost;
    }

    private static long DeltaSwap(int n, int[] flow, int[] dist, int[] perm, int u, int v)
    {
        int pu = perm[u];
        int pv = perm[v];

        int uRowA = u * n;
        int vRowA = v * n;
        int puRowB = pu * n;
        int pvRowB = pv * n;

        long delta =
            (long)(flow[uRowA + u] - flow[vRowA + v]) * (dist[pvRowB + pv] - dist[puRowB + pu]) +
            (long)(flow[uRowA + v] - flow[vRowA + u]) * (dist[pvRowB + pu] - dist[puRowB + pv]);

        for (int k = 0; k < n; k++)
        {
            if (k == u || k == v) continue;

            int pk = perm[k];
            int kRowA = k * n;
            int pkRowB = pk * n;

            delta +=
                (long)(flow[uRowA + k] - flow[vRowA + k]) * (dist[pvRowB + pk] - dist[puRowB + pk]) +
                (long)(flow[kRowA + u] - flow[kRowA + v]) * (dist[pkRowB + pv] - dist[pkRowB + pu]);
        }

        return delta;
    }

    private static void ApplyRandomSwaps(int[] perm, ref XorShift64Star rng, int swapCount)
    {
        int n = perm.Length;
        for (int t = 0; t < swapCount; t++)
        {
            int i = rng.NextInt(n);
            int j = rng.NextInt(n - 1);
            if (j >= i) j++;
            int tmp = perm[i];
            perm[i] = perm[j];
            perm[j] = tmp;
        }
    }

    private static int[] BuildGreedyPermutation(int n, int[] flow, int[] dist)
    {
        long[] rowSumF = new long[n];
        long[] colSumF = new long[n];
        for (int i = 0; i < n; i++)
        {
            int iRow = i * n;
            for (int j = 0; j < n; j++)
            {
                int val = flow[iRow + j];
                rowSumF[i] += val;
                colSumF[j] += val;
            }
        }

        long[] rowSumD = new long[n];
        long[] colSumD = new long[n];
        for (int i = 0; i < n; i++)
        {
            int iRow = i * n;
            for (int j = 0; j < n; j++)
            {
                int val = dist[iRow + j];
                rowSumD[i] += val;
                colSumD[j] += val;
            }
        }

        long[] facilityPot = new long[n];
        long[] locationPot = new long[n];
        for (int i = 0; i < n; i++)
        {
            facilityPot[i] = rowSumF[i] + colSumF[i];
            locationPot[i] = rowSumD[i] + colSumD[i];
        }

        int[] facilities = new int[n];
        int[] locations = new int[n];
        for (int i = 0; i < n; i++)
        {
            facilities[i] = i;
            locations[i] = i;
        }

        Array.Sort(facilities, (a, b) => facilityPot[b].CompareTo(facilityPot[a])); // descending
        Array.Sort(locations, (a, b) => locationPot[a].CompareTo(locationPot[b]));  // ascending

        int[] perm = new int[n];
        for (int idx = 0; idx < n; idx++)
        {
            perm[facilities[idx]] = locations[idx];
        }

        return perm;
    }

    public static void Main()
    {
        var scanner = new FastScanner(Console.OpenStandardInput());
        int n = scanner.NextInt();
        if (n <= 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }

        int nn = checked(n * n);
        int[] flow = new int[nn];
        int[] dist = new int[nn];

        for (int i = 0; i < nn; i++) flow[i] = scanner.NextInt();
        for (int i = 0; i < nn; i++) dist[i] = scanner.NextInt();

        if (n == 1)
        {
            long onlyCost = (long)flow[0] * dist[0];
            Console.WriteLine(onlyCost);
            Console.WriteLine("0");
            return;
        }

        int timeLimitMs = GetTimeLimitMs(n);
        var sw = Stopwatch.StartNew();

        // Seed RNG from a mix of clock and instance characteristics.
        ulong seed = (ulong)Stopwatch.GetTimestamp();
        seed ^= (ulong)n * 0x9E3779B97F4A7C15UL;
        seed ^= (ulong)(flow[0] * 1315423911) + (ulong)(dist[0] * 2654435761u);
        var rng = new XorShift64Star(seed);

        int[] basePerm = BuildGreedyPermutation(n, flow, dist);

        int[] perm = new int[n];
        Array.Copy(basePerm, perm, n);
        ApplyRandomSwaps(perm, ref rng, Math.Min(n, 64));

        long cost = ComputeCost(n, flow, dist, perm);

        long bestCost = cost;
        int[] bestPerm = new int[n];
        Array.Copy(perm, bestPerm, n);

        // Estimate scale for temperature.
        int sampleCount = Math.Min(64, n * (n - 1) / 2);
        double avgPosDelta = 0;
        int posCnt = 0;
        for (int s = 0; s < sampleCount; s++)
        {
            int i = rng.NextInt(n);
            int j = rng.NextInt(n - 1);
            if (j >= i) j++;
            long d = DeltaSwap(n, flow, dist, perm, i, j);
            if (d > 0)
            {
                avgPosDelta += d;
                posCnt++;
            }
        }
        if (posCnt > 0) avgPosDelta /= posCnt;
        if (avgPosDelta <= 0) avgPosDelta = 1.0;

        double t0 = avgPosDelta / Math.Log(2.0); // ~50% accept at start for avg uphill move.
        double tMin = Math.Max(1.0, t0 * 1e-4);
        double temperature = t0;

        long stage1Ms = (long)(timeLimitMs * 0.80);
        if (stage1Ms < 1) stage1Ms = timeLimitMs;

        int iter = 0;
        const int tempUpdateMask = 1023;

        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            bool hillClimbOnly = sw.ElapsedMilliseconds >= stage1Ms;

            if ((iter & tempUpdateMask) == 0 && !hillClimbOnly)
            {
                double frac = sw.ElapsedMilliseconds / (double)stage1Ms;
                if (frac > 1.0) frac = 1.0;
                temperature = tMin + (t0 - tMin) * (1.0 - frac); // linear cooling.
                if (temperature < tMin) temperature = tMin;
            }

            int u = rng.NextInt(n);
            int v;

            if (hillClimbOnly)
            {
                int bestV = -1;
                long bestDelta = 0;
                int trials = n <= 200 ? 32 : n <= 600 ? 20 : 14;
                for (int t = 0; t < trials; t++)
                {
                    int cand = rng.NextInt(n - 1);
                    if (cand >= u) cand++;
                    long d = DeltaSwap(n, flow, dist, perm, u, cand);
                    if (d < bestDelta)
                    {
                        bestDelta = d;
                        bestV = cand;
                    }
                }
                if (bestV < 0)
                {
                    iter++;
                    continue;
                }
                v = bestV;

                // Apply best improving sampled move.
                int tmp = perm[u];
                perm[u] = perm[v];
                perm[v] = tmp;
                cost += bestDelta;
                if (cost < bestCost)
                {
                    bestCost = cost;
                    Array.Copy(perm, bestPerm, n);
                }
                iter++;
                continue;
            }

            v = rng.NextInt(n - 1);
            if (v >= u) v++;

            long delta = DeltaSwap(n, flow, dist, perm, u, v);
            bool accept;
            if (delta <= 0)
            {
                accept = true;
            }
            else
            {
                double prob = Math.Exp(-delta / temperature);
                accept = rng.NextDouble() < prob;
            }

            if (accept)
            {
                int tmp = perm[u];
                perm[u] = perm[v];
                perm[v] = tmp;
                cost += delta;
                if (cost < bestCost)
                {
                    bestCost = cost;
                    Array.Copy(perm, bestPerm, n);
                }
            }

            iter++;
        }

        var sb = new StringBuilder(n * 4 + 64);
        sb.Append(bestCost).Append('\n');
        for (int i = 0; i < n; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(bestPerm[i]);
        }
        sb.Append('\n');
        Console.Write(sb.ToString());
    }
}
