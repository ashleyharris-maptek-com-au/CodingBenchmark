; SPIR-V
; Version: 1.0
; Shader: Geometry

OpCapability Shader
OpCapability Geometry
%glsl450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

OpDecorate %arr_v3float ArrayStride 16

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_3 = OpConstant %uint 3
%arr_gl_PerVertex_in = OpTypeArray %gl_PerVertex_in %uint_3
%arr_v3float = OpTypeArray %v3float %uint_3

%_ptr_Input_arr_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in
%_ptr_Input_arr_v3float = OpTypePointer Input %arr_v3float
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %_ptr_Input_arr_gl_PerVertex_in Input
%worldPos = OpVariable %_ptr_Input_arr_v3float Input
%normal = OpVariable %_ptr_Input_arr_v3float Input
%color = OpVariable %_ptr_Input_arr_v3float Input

%gl_out = OpVariable %_ptr_Output_gl_PerVertex_out Output
%outColor = OpVariable %_ptr_Output_v3float Output

%func = OpTypeFunction %void

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2

%float_0 = OpConstant %float 0.0
%float_0_08 = OpConstant %float 0.08
%float_0_33333334 = OpConstant %float 0.33333334

%main = OpFunction %void None %func
%entry = OpLabel

%p0_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr

%sum01 = OpFAdd %v4float %p0 %p1
%sum012 = OpFAdd %v4float %sum01 %p2
%center = OpVectorTimesScalar %v4float %sum012 %float_0_33333334

%p0_xyz = OpVectorShuffle %v3float %p0 %p0 0 1 2
%p1_xyz = OpVectorShuffle %v3float %p1 %p1 0 1 2
%p2_xyz = OpVectorShuffle %v3float %p2 %p2 0 1 2

%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz
%cross = OpExtInst %v3float %glsl450 Cross %e1 %e2
%nrm = OpExtInst %v3float %glsl450 Normalize %cross
%offset = OpVectorTimesScalar %v3float %nrm %float_0_08

%off_x = OpCompositeExtract %float %offset 0
%off_y = OpCompositeExtract %float %offset 1
%off_z = OpCompositeExtract %float %offset 2
%offset4 = OpCompositeConstruct %v4float %off_x %off_y %off_z %float_0

%out_pos_ptr = OpAccessChain %_ptr_Output_v4float %gl_out %int_0

%c0_ptr = OpAccessChain %_ptr_Input_v3float %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%p0_out = OpFAdd %v4float %p0 %offset4
OpStore %out_pos_ptr %p0_out
OpStore %outColor %c0
OpEmitVertex

%c1_ptr = OpAccessChain %_ptr_Input_v3float %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%p1_out = OpFAdd %v4float %p1 %offset4
OpStore %out_pos_ptr %p1_out
OpStore %outColor %c1
OpEmitVertex

%c2_ptr = OpAccessChain %_ptr_Input_v3float %color %int_2
%c2 = OpLoad %v3float %c2_ptr
%p2_out = OpFAdd %v4float %p2 %offset4
OpStore %out_pos_ptr %p2_out
OpStore %outColor %c2
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
