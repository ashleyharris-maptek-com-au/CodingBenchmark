use std::collections::HashMap;
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

const U32_SENTINEL: u32 = u32::MAX;

struct Scanner {
    buf: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new(buf: Vec<u8>) -> Self {
        Self { buf, pos: 0 }
    }

    fn next_i64(&mut self) -> Option<i64> {
        while self.pos < self.buf.len() && self.buf[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        if self.pos >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.pos] == b'-' {
            sign = -1;
            self.pos += 1;
        }
        let mut value: i64 = 0;
        while self.pos < self.buf.len() {
            let b = self.buf[self.pos];
            if !b.is_ascii_digit() {
                break;
            }
            value = value
                .saturating_mul(10)
                .saturating_add((b - b'0') as i64);
            self.pos += 1;
        }
        Some(value * sign)
    }
}

#[derive(Default)]
struct FastHasher {
    state: u64,
}

impl Hasher for FastHasher {
    fn finish(&self) -> u64 {
        self.state
    }

    fn write(&mut self, bytes: &[u8]) {
        let mut x: u64 = 0x9E37_79B9_7F4A_7C15;
        for &b in bytes {
            x ^= b as u64;
            x = x.wrapping_mul(0xBF58_476D_1CE4_E5B9);
            x ^= x >> 32;
        }
        self.state ^= x;
        self.state = self
            .state
            .wrapping_mul(0x94D0_49BB_1331_11EB)
            .rotate_left(27);
    }

    fn write_u64(&mut self, i: u64) {
        let mut x = i ^ 0x9E37_79B9_7F4A_7C15;
        x = x.wrapping_mul(0xBF58_476D_1CE4_E5B9);
        x ^= x >> 32;
        x = x.wrapping_mul(0x94D0_49BB_1331_11EB);
        x ^= x >> 29;
        self.state ^= x;
        self.state = self.state.rotate_left(17).wrapping_mul(0xD6E8_FEB8_6659_FD93);
    }
}

type FastHashMap<K, V> = HashMap<K, V, BuildHasherDefault<FastHasher>>;

#[derive(Clone, Copy)]
enum ChunkSrc {
    Single { index: usize },
    Bucket { w: usize, start: usize, len: usize },
}

#[derive(Clone, Copy)]
struct Chunk {
    weight: usize,
    src: ChunkSrc,
}

fn gcd_usize(mut a: usize, mut b: usize) -> usize {
    while b != 0 {
        let r = a % b;
        a = b;
        b = r;
    }
    a
}

fn dp_target_reached(dp: &[u64], target: usize) -> bool {
    let word = target / 64;
    let bit = target % 64;
    (dp[word] >> bit) & 1 == 1
}

fn mask_last_word(dp: &mut [u64], max_bit: usize) {
    let last = dp.len() - 1;
    let keep = (max_bit % 64) + 1;
    let mask = if keep == 64 { u64::MAX } else { (1u64 << keep) - 1 };
    dp[last] &= mask;
}

fn shift_left_or_record(
    dp: &mut [u64],
    shift: usize,
    max_bit: usize,
    item_id: u32,
    parent_item: &mut [u32],
    parent_prev: &mut [u32],
) {
    if shift == 0 {
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for i in (0..dp.len()).rev() {
        let mut shifted = 0u64;
        if i >= word_shift {
            let src = i - word_shift;
            let src_val = dp[src];
            if bit_shift == 0 {
                shifted = src_val;
            } else {
                let mut v = src_val << bit_shift;
                if src > 0 {
                    v |= dp[src - 1] >> (64 - bit_shift);
                }
                shifted = v;
            }
        }
        let old = dp[i];
        let new_bits = shifted & !old;
        dp[i] = old | shifted;
        if new_bits != 0 {
            let base = i * 64;
            let mut bits = new_bits;
            while bits != 0 {
                let tz = bits.trailing_zeros() as usize;
                let new_pos = base + tz;
                if new_pos <= max_bit {
                    parent_item[new_pos] = item_id;
                    parent_prev[new_pos] = (new_pos - shift) as u32;
                }
                bits &= bits - 1;
            }
        }
    }
    mask_last_word(dp, max_bit);
}

fn shift_right_or_record(
    dp: &mut [u64],
    shift: usize,
    max_bit: usize,
    item_id: u32,
    parent_item: &mut [u32],
    parent_prev: &mut [u32],
) {
    if shift == 0 {
        return;
    }
    let word_shift = shift / 64;
    let bit_shift = shift % 64;
    for i in 0..dp.len() {
        let mut shifted = 0u64;
        let src = i + word_shift;
        if src < dp.len() {
            let src_val = dp[src];
            if bit_shift == 0 {
                shifted = src_val;
            } else {
                let mut v = src_val >> bit_shift;
                if src + 1 < dp.len() {
                    v |= dp[src + 1] << (64 - bit_shift);
                }
                shifted = v;
            }
        }
        let old = dp[i];
        let new_bits = shifted & !old;
        dp[i] = old | shifted;
        if new_bits != 0 {
            let base = i * 64;
            let mut bits = new_bits;
            while bits != 0 {
                let tz = bits.trailing_zeros() as usize;
                let new_pos = base + tz;
                if new_pos <= max_bit {
                    parent_item[new_pos] = item_id;
                    parent_prev[new_pos] = (new_pos + shift) as u32;
                }
                bits &= bits - 1;
            }
        }
    }
    mask_last_word(dp, max_bit);
}

fn subset_sum_bitset(chunks: &[Chunk], max_bit: usize, target: usize) -> Option<Vec<usize>> {
    let words = (max_bit / 64) + 1;
    let mut dp = vec![0u64; words];
    dp[0] = 1;

    let mut parent_item = vec![U32_SENTINEL; max_bit + 1];
    let mut parent_prev = vec![U32_SENTINEL; max_bit + 1];

    for (cid, ch) in chunks.iter().enumerate() {
        let shift = ch.weight;
        if shift == 0 || shift > max_bit {
            continue;
        }
        shift_left_or_record(
            &mut dp,
            shift,
            max_bit,
            cid as u32,
            &mut parent_item,
            &mut parent_prev,
        );
        if dp_target_reached(&dp, target) {
            let mut used = Vec::new();
            let mut s = target;
            while s != 0 {
                let it = parent_item[s];
                if it == U32_SENTINEL {
                    return None;
                }
                used.push(it as usize);
                s = parent_prev[s] as usize;
            }
            return Some(used);
        }
    }
    None
}

fn subset_sum_offset_range(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let mut min_sum: i64 = 0;
    let mut max_sum: i64 = 0;
    for &x in nums {
        if x < 0 {
            min_sum = min_sum.saturating_add(x);
        } else {
            max_sum = max_sum.saturating_add(x);
        }
    }
    if target < min_sum || target > max_sum {
        return None;
    }
    let range_i64 = max_sum - min_sum;
    if range_i64 < 0 {
        return None;
    }
    let range = usize::try_from(range_i64).ok()?;
    const RANGE_LIMIT: usize = 5_000_000;
    if range > RANGE_LIMIT {
        return None;
    }
    let offset = usize::try_from(-min_sum).ok()?;
    let target_pos = usize::try_from(target - min_sum).ok()?;

    let words = (range / 64) + 1;
    let mut dp = vec![0u64; words];
    dp[offset / 64] |= 1u64 << (offset % 64);

    let mut parent_item = vec![U32_SENTINEL; range + 1];
    let mut parent_prev = vec![U32_SENTINEL; range + 1];

    for (idx, &x) in nums.iter().enumerate() {
        if x == 0 {
            continue;
        }
        if x > 0 {
            let shift = usize::try_from(x).ok()?;
            if shift > range {
                continue;
            }
            shift_left_or_record(
                &mut dp,
                shift,
                range,
                idx as u32,
                &mut parent_item,
                &mut parent_prev,
            );
        } else {
            let shift = usize::try_from(-x).ok()?;
            if shift > range {
                continue;
            }
            shift_right_or_record(
                &mut dp,
                shift,
                range,
                idx as u32,
                &mut parent_item,
                &mut parent_prev,
            );
        }
        if dp_target_reached(&dp, target_pos) {
            let mut res = Vec::new();
            let mut pos = target_pos;
            while pos != offset {
                let it = parent_item[pos];
                if it == U32_SENTINEL {
                    return None;
                }
                res.push(it as usize);
                pos = parent_prev[pos] as usize;
            }
            res.sort_unstable();
            return Some(res);
        }
    }
    None
}

fn solve_mitm(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = nums.len();
    let mid = n / 2;
    let left_len = mid;
    let right_len = n - mid;

    let left_total = 1usize << left_len;
    let right_total = 1usize << right_len;

    let mut left_sums: Vec<(i64, u32)> = Vec::with_capacity(left_total);
    for mask in 0..left_total {
        let mut sum = 0i64;
        for i in 0..left_len {
            if (mask >> i) & 1 == 1 {
                sum = sum.saturating_add(nums[i]);
            }
        }
        left_sums.push((sum, mask as u32));
    }
    left_sums.sort_unstable_by(|a, b| a.0.cmp(&b.0));

    for mask in 0..right_total {
        let mut sum = 0i64;
        for i in 0..right_len {
            if (mask >> i) & 1 == 1 {
                sum = sum.saturating_add(nums[mid + i]);
            }
        }
        let want = target.saturating_sub(sum);
        if let Ok(pos) = left_sums.binary_search_by(|(s, _)| s.cmp(&want)) {
            let lmask = left_sums[pos].1 as usize;
            let mut res = Vec::new();
            for i in 0..left_len {
                if (lmask >> i) & 1 == 1 {
                    res.push(i);
                }
            }
            for i in 0..right_len {
                if (mask >> i) & 1 == 1 {
                    res.push(mid + i);
                }
            }
            res.sort_unstable();
            return Some(res);
        }
    }
    None
}

fn two_sum_fast(items: &[(usize, usize)], target: usize) -> Option<Vec<usize>> {
    let mut seen: FastHashMap<u64, usize> = FastHashMap::default();
    seen.reserve(items.len().min(2_000_000));
    for &(w, idx) in items.iter().take(2_000_000) {
        if w > target {
            continue;
        }
        let need = target - w;
        if let Some(&j) = seen.get(&(need as u64)) {
            if j != idx {
                let mut res = vec![j, idx];
                res.sort_unstable();
                return Some(res);
            }
        }
        seen.entry(w as u64).or_insert(idx);
    }
    None
}

fn solve_nonnegative(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    let target_usize = usize::try_from(target).ok()?;
    if target_usize == 0 {
        return Some(Vec::new());
    }

    let mut items: Vec<(usize, usize)> = Vec::new();
    items.reserve(nums.len());

    let mut total_sum: usize = 0;
    let mut g: usize = 0;
    let mut max_w: usize = 0;

    for (idx, &x) in nums.iter().enumerate() {
        if x == target {
            return Some(vec![idx]);
        }
        if x <= 0 {
            continue;
        }
        let w = usize::try_from(x).ok()?;
        if w > target_usize {
            continue;
        }
        items.push((w, idx));
        total_sum = total_sum.saturating_add(w);
        g = if g == 0 { w } else { gcd_usize(g, w) };
        if w > max_w {
            max_w = w;
        }
    }

    if total_sum < target_usize {
        return None;
    }
    if g != 0 && target_usize % g != 0 {
        return None;
    }

    if let Some(res) = two_sum_fast(&items, target_usize) {
        return Some(res);
    }

    let max_bit = target_usize;
    if max_bit > u32::MAX as usize {
        return None;
    }

    let words = (max_bit / 64) + 1;
    let mut chunks: Vec<Chunk> = Vec::new();
    let mut buckets: Option<Vec<Vec<usize>>> = None;

    const BUCKET_MAX: usize = 200_000;
    let use_bucket = max_w <= BUCKET_MAX && items.len() > 200_000;
    if use_bucket {
        let mut bs: Vec<Vec<usize>> = vec![Vec::new(); max_w + 1];
        for (w, idx) in &items {
            bs[*w].push(*idx);
        }
        for w in 1..=max_w {
            let count = bs[w].len();
            if count == 0 {
                continue;
            }
            let usable = (target_usize / w).min(count);
            if usable == 0 {
                continue;
            }
            let mut start = 0usize;
            let mut p = 1usize;
            let mut remaining = usable;
            while remaining > 0 {
                let take = if p < remaining { p } else { remaining };
                let total_w = w * take;
                chunks.push(Chunk {
                    weight: total_w,
                    src: ChunkSrc::Bucket {
                        w,
                        start,
                        len: take,
                    },
                });
                start += take;
                remaining -= take;
                p = p.saturating_mul(2);
            }
        }
        buckets = Some(bs);
    } else {
        for (w, idx) in &items {
            chunks.push(Chunk {
                weight: *w,
                src: ChunkSrc::Single { index: *idx },
            });
        }
    }

    const MAX_BYTES: usize = 3_000_000_000;
    let bytes_needed = (max_bit + 1)
        .saturating_mul(8)
        .saturating_add(words.saturating_mul(8));
    if bytes_needed > MAX_BYTES {
        return None;
    }

    let ops = (chunks.len() as u128) * (words as u128);
    const OPS_LIMIT: u128 = 2_500_000_000;
    if ops > OPS_LIMIT {
        return None;
    }

    let used_chunks = subset_sum_bitset(&chunks, max_bit, target_usize)?;
    let mut res: Vec<usize> = Vec::new();
    for cid in used_chunks {
        match chunks[cid].src {
            ChunkSrc::Single { index } => res.push(index),
            ChunkSrc::Bucket { w, start, len } => {
                let bs = buckets.as_ref()?;
                res.extend_from_slice(&bs[w][start..start + len]);
            }
        }
    }
    res.sort_unstable();

    let mut check_sum: i64 = 0;
    for &idx in &res {
        check_sum = check_sum.saturating_add(nums[idx]);
    }
    if check_sum == target {
        Some(res)
    } else {
        None
    }
}

fn solve(nums: &[i64], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if nums.is_empty() {
        return None;
    }

    if nums.len() <= 40 {
        if let Some(ans) = solve_mitm(nums, target) {
            return Some(ans);
        }
    }

    let mut all_nonneg = true;
    let mut all_nonpos = true;
    for &x in nums {
        if x < 0 {
            all_nonneg = false;
        }
        if x > 0 {
            all_nonpos = false;
        }
    }

    if all_nonpos {
        let mut neg = Vec::with_capacity(nums.len());
        for &x in nums {
            neg.push(-x);
        }
        return solve_nonnegative(&neg, -target);
    }

    if all_nonneg && target >= 0 {
        if let Some(ans) = solve_nonnegative(nums, target) {
            return Some(ans);
        }
    }

    subset_sum_offset_range(nums, target)
}

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut sc = Scanner::new(input);
    let n = match sc.next_i64() {
        Some(v) if v > 0 => v as usize,
        Some(_) => 0usize,
        None => return,
    };
    let target = match sc.next_i64() {
        Some(v) => v,
        None => 0,
    };
    let mut nums = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(v) = sc.next_i64() {
            nums.push(v);
        } else {
            nums.push(0);
        }
    }

    let ans = solve(&nums, target);
    let mut out = String::new();
    match ans {
        Some(indices) => {
            out.push_str("YES\n");
            if !indices.is_empty() {
                for (i, idx) in indices.iter().enumerate() {
                    if i != 0 {
                        out.push(' ');
                    }
                    out.push_str(&idx.to_string());
                }
            }
            out.push('\n');
        }
        None => {
            out.push_str("NO\n");
        }
    }
    print!("{out}");
}
