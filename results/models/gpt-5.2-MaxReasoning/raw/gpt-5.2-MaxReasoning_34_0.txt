from __future__ import annotations

import sys
import math
import random
from time import perf_counter


def solve() -> None:
    data = sys.stdin.buffer.read().split()
    if not data:
        return

    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    capacity = float(next(it))

    depot_x = float(next(it))
    depot_y = float(next(it))

    xs = [0.0] * (n + 1)
    ys = [0.0] * (n + 1)
    demands = [0.0] * n

    for i in range(n):
        xs[i] = float(next(it))
        ys[i] = float(next(it))
        demands[i] = float(next(it))

    DEPOT = n
    xs[DEPOT] = depot_x
    ys[DEPOT] = depot_y

    eps = 1e-9
    for d in demands:
        if d > capacity + eps:
            # Infeasible under the problem rules; output a trivial format.
            # (Benchmarks normally avoid infeasible instances.)
            print("0.0")
            print("0")
            return

    sqrt = math.sqrt
    atan2 = math.atan2

    def dist(a: int, b: int) -> float:
        dx = xs[a] - xs[b]
        dy = ys[a] - ys[b]
        return sqrt(dx * dx + dy * dy)

    def route_load(route: list[int]) -> float:
        return sum(demands[c] for c in route)

    def route_distance(route: list[int]) -> float:
        if not route:
            return 0.0
        total = dist(DEPOT, route[0])
        prev = route[0]
        for c in route[1:]:
            total += dist(prev, c)
            prev = c
        total += dist(prev, DEPOT)
        return total

    def total_distance(routes: list[list[int]]) -> float:
        return sum(route_distance(r) for r in routes)

    def best_insertion(route: list[int], node: int) -> tuple[float, int]:
        # Returns (delta_cost, position) for best insertion.
        best_delta = float("inf")
        best_pos = 0
        m = len(route)
        for pos in range(m + 1):
            prev = DEPOT if pos == 0 else route[pos - 1]
            nxt = DEPOT if pos == m else route[pos]
            delta = dist(prev, node) + dist(node, nxt) - dist(prev, nxt)
            if delta < best_delta:
                best_delta = delta
                best_pos = pos
        return best_delta, best_pos

    def nearest_neighbor_order(nodes: list[int]) -> list[int]:
        if len(nodes) <= 2:
            return nodes[:]
        unvisited = nodes[:]
        route: list[int] = []
        current = DEPOT
        while unvisited:
            best_i = 0
            best_d = dist(current, unvisited[0])
            for i in range(1, len(unvisited)):
                c = unvisited[i]
                d = dist(current, c)
                if d < best_d:
                    best_d = d
                    best_i = i
            nxt = unvisited.pop(best_i)
            route.append(nxt)
            current = nxt
        return route

    def two_opt(route: list[int], time_limit: float) -> list[int]:
        # First-improvement 2-opt with a small time budget.
        m = len(route)
        if m < 4:
            return route
        start_t = perf_counter()
        while perf_counter() - start_t < time_limit:
            improved = False
            for i in range(m - 1):
                a = DEPOT if i == 0 else route[i - 1]
                b = route[i]
                for j in range(i + 1, m):
                    c = route[j]
                    d = DEPOT if j == m - 1 else route[j + 1]
                    delta = dist(a, c) + dist(b, d) - dist(a, b) - dist(c, d)
                    if delta < -1e-12:
                        route[i : j + 1] = reversed(route[i : j + 1])
                        improved = True
                        break
                if improved or (perf_counter() - start_t >= time_limit):
                    break
            if not improved:
                break
        return route

    def split_by_capacity(order: list[int]) -> list[list[int]]:
        routes: list[list[int]] = []
        cur: list[int] = []
        load = 0.0
        for c in order:
            d = demands[c]
            if cur and load + d > capacity + eps:
                routes.append(cur)
                cur = []
                load = 0.0
            cur.append(c)
            load += d
        if cur:
            routes.append(cur)
        return routes

    def reduce_to_k_routes(routes: list[list[int]]) -> list[list[int]] | None:
        # Tries to eliminate routes by inserting their customers into others.
        if len(routes) <= k:
            return routes

        routes = [r[:] for r in routes]
        while len(routes) > k:
            # Try a few smallest-load routes as elimination candidates.
            cand_indices = sorted(
                range(len(routes)),
                key=lambda idx: (route_load(routes[idx]), len(routes[idx])),
            )

            best_candidate: list[list[int]] | None = None
            best_cost = float("inf")
            for idx in cand_indices[: min(8, len(cand_indices))]:
                candidate = [r[:] for r in routes]
                removed = candidate.pop(idx)
                loads = [route_load(r) for r in candidate]

                # Harder-to-place first: larger demands first.
                removed_sorted = sorted(removed, key=lambda c: demands[c], reverse=True)
                ok = True
                for node in removed_sorted:
                    best_r = -1
                    best_pos = 0
                    best_delta = float("inf")
                    dmd = demands[node]
                    for r_i, r in enumerate(candidate):
                        if loads[r_i] + dmd <= capacity + eps:
                            delta, pos = best_insertion(r, node)
                            if delta < best_delta:
                                best_delta = delta
                                best_r = r_i
                                best_pos = pos
                    if best_r < 0:
                        ok = False
                        break
                    candidate[best_r].insert(best_pos, node)
                    loads[best_r] += dmd

                if ok:
                    cost = total_distance(candidate)
                    if cost < best_cost:
                        best_cost = cost
                        best_candidate = candidate

            if best_candidate is None:
                return None
            routes = best_candidate

        return routes

    def make_solution_from_routes(routes: list[list[int]], time_budget: float) -> list[list[int]]:
        # Reorder each route (NN + 2-opt) within a time budget.
        if not routes:
            return routes
        per_route = time_budget / max(1, len(routes))
        out: list[list[int]] = []
        for r in routes:
            rr = r[:] if len(r) > 120 else nearest_neighbor_order(r)
            rr = two_opt(rr, per_route * 0.9)
            out.append(rr)
        return out

    def build_sweep_candidates() -> list[list[list[int]]]:
        customers = list(range(n))
        angles = [0.0] * n
        for i in range(n):
            angles[i] = atan2(ys[i] - depot_y, xs[i] - depot_x)
        base = sorted(customers, key=lambda i: angles[i])

        starts = min(12, n) if n else 1
        offsets = []
        if n:
            for t in range(starts):
                offsets.append((t * n) // starts)
            offsets = sorted(set(offsets))
        else:
            offsets = [0]

        candidates: list[list[list[int]]] = []
        for off in offsets:
            order = base[off:] + base[:off]
            candidates.append(split_by_capacity(order))
            candidates.append(split_by_capacity(list(reversed(order))))
        return candidates

    def build_packing_candidates(rng: random.Random) -> list[list[list[int]]]:
        customers = list(range(n))
        if not customers:
            return [[]]

        # Precompute a couple of useful keys.
        d_to_depot = [dist(i, DEPOT) for i in range(n)]
        angles = [atan2(ys[i] - depot_y, xs[i] - depot_x) for i in range(n)]

        orders: list[list[int]] = []
        orders.append(sorted(customers, key=lambda i: demands[i], reverse=True))
        orders.append(sorted(customers, key=lambda i: (demands[i], d_to_depot[i]), reverse=True))
        orders.append(sorted(customers, key=lambda i: (demands[i], angles[i]), reverse=True))

        # A few randomized mixes.
        for _ in range(5):
            rng.shuffle(customers)
            orders.append(customers[:])

        def best_fit_decreasing(order: list[int]) -> list[list[int]] | None:
            routes: list[list[int]] = []
            loads: list[float] = []
            for node in order:
                dmd = demands[node]
                best_i = -1
                best_rem = float("inf")
                for i in range(len(routes)):
                    new_load = loads[i] + dmd
                    if new_load <= capacity + eps:
                        rem = capacity - new_load
                        if rem < best_rem:
                            best_rem = rem
                            best_i = i
                if best_i < 0:
                    if len(routes) >= k:
                        return None
                    routes.append([node])
                    loads.append(dmd)
                else:
                    routes[best_i].append(node)
                    loads[best_i] += dmd
            return routes

        candidates: list[list[list[int]]] = []
        for order in orders:
            packed = best_fit_decreasing(order)
            if packed is not None:
                candidates.append(packed)
        return candidates

    rng = random.Random(0)
    start_time = perf_counter()
    overall_budget = 1.2 if n <= 300 else (1.6 if n <= 800 else 2.0)

    best_routes: list[list[int]] | None = None
    best_cost = float("inf")

    all_candidates = []
    all_candidates.extend(build_sweep_candidates())
    all_candidates.extend(build_packing_candidates(rng))

    for routes0 in all_candidates:
        if perf_counter() - start_time > overall_budget:
            break

        reduced = reduce_to_k_routes(routes0)
        if reduced is None or len(reduced) > k:
            continue

        time_left = max(0.05, overall_budget - (perf_counter() - start_time))
        improved = make_solution_from_routes(reduced, time_left * 0.6)

        cost = total_distance(improved)
        if cost < best_cost:
            best_cost = cost
            best_routes = improved

    if best_routes is None:
        # Guaranteed-feasible greedy fallback (may use >k if instance is infeasible).
        customers = list(range(n))
        angles = [atan2(ys[i] - depot_y, xs[i] - depot_x) for i in range(n)]
        customers.sort(key=lambda i: angles[i])
        best_routes = split_by_capacity(customers)
        best_cost = total_distance(best_routes)

    # Final sanity: ensure we do not output empty routes.
    best_routes = [r for r in best_routes if r]
    if len(best_routes) > k:
        reduced = reduce_to_k_routes(best_routes)
        if reduced is not None and len(reduced) <= k:
            best_routes = make_solution_from_routes(reduced, 0.3)
            best_cost = total_distance(best_routes)

    def validate(routes: list[list[int]]) -> bool:
        if len(routes) > k:
            return False
        seen = [0] * n
        for r in routes:
            load = 0.0
            for c in r:
                if c < 0 or c >= n or seen[c]:
                    return False
                seen[c] = 1
                load += demands[c]
            if load > capacity + eps:
                return False
        return True if n == 0 else all(seen)

    if not validate(best_routes):
        customers = list(range(n))
        best_routes = split_by_capacity(customers)
        reduced = reduce_to_k_routes(best_routes)
        if reduced is not None:
            best_routes = reduced
        best_routes = [r for r in best_routes if r]
        best_cost = total_distance(best_routes)

    # Output
    out_lines = []
    out_lines.append(f"{best_cost:.10f}")
    out_lines.append(str(len(best_routes)))
    for r in best_routes:
        out_lines.append(" ".join(str(i) for i in r))
    sys.stdout.write("\n".join(out_lines))


if __name__ == "__main__":
    solve()
