OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1
OpSource GLSL 450

OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_in 0 Offset 0
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpMemberDecorate %gl_PerVertex_out 0 Offset 0
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 0
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%h = OpConstant %float 0.04

%arr_pv_in = OpTypeArray %gl_PerVertex_in %int_1
%arr_v3_in = OpTypeArray %v3float %int_1

%ptr_Input_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_Input_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Output_v3float = OpTypePointer Output %v3float

%gl_in = OpVariable %ptr_Input_arr_pv Input
%worldPos = OpVariable %ptr_Input_arr_v3 Input
%normal = OpVariable %ptr_Input_arr_v3 Input
%color = OpVariable %ptr_Input_arr_v3 Input
%gl_out = OpVariable %ptr_Output_pv Output
%outColor = OpVariable %ptr_Output_v3float Output

%fn_void = OpTypeFunction %void

%main = OpFunction %void None %fn_void
%entry = OpLabel

%inPosPtr = OpAccessChain %ptr_Input_v4float %gl_in %int_0 %int_0
%inPos = OpLoad %v4float %inPosPtr
%inColorPtr = OpAccessChain %ptr_Input_v3float %color %int_0
%inColor = OpLoad %v3float %inColorPtr

%x = OpCompositeExtract %float %inPos 0
%y = OpCompositeExtract %float %inPos 1

%xm = OpFSub %float %x %h
%xp = OpFAdd %float %x %h
%ym = OpFSub %float %y %h
%yp = OpFAdd %float %y %h

%outPosPtr = OpAccessChain %ptr_Output_v4float %gl_out %int_0

%pos0 = OpCompositeConstruct %v4float %xm %ym %float_0 %float_1
OpStore %outPosPtr %pos0
OpStore %outColor %inColor
OpEmitVertex

%pos1 = OpCompositeConstruct %v4float %xp %ym %float_0 %float_1
OpStore %outPosPtr %pos1
OpStore %outColor %inColor
OpEmitVertex

%pos2 = OpCompositeConstruct %v4float %xm %yp %float_0 %float_1
OpStore %outPosPtr %pos2
OpStore %outColor %inColor
OpEmitVertex

%pos3 = OpCompositeConstruct %v4float %xp %yp %float_0 %float_1
OpStore %outPosPtr %pos3
OpStore %outColor %inColor
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
