declare const require: any;
declare const process: any;
const fs = require('fs');

type PackageInput = { id: any; x: number; y: number };
type InputData = { depot: [number, number]; truckCount: number; packages: PackageInput[] };

function distXY(ax: number, ay: number, bx: number, by: number): number {
  const dx = ax - bx;
  const dy = ay - by;
  return Math.sqrt(dx * dx + dy * dy);
}

function part1By1_16(x: number): number {
  // Spreads the lower 16 bits of x so that there are zeros between the bits.
  // Returns a 32-bit int (use >>> 0 to interpret as unsigned).
  x &= 0xffff;
  x = (x | (x << 8)) & 0x00ff00ff;
  x = (x | (x << 4)) & 0x0f0f0f0f;
  x = (x | (x << 2)) & 0x33333333;
  x = (x | (x << 1)) & 0x55555555;
  return x >>> 0;
}

function morton2D_16(x: number, y: number): number {
  // Interleave two 16-bit values into one 32-bit Morton code.
  return (part1By1_16(x) | (part1By1_16(y) << 1)) >>> 0;
}

function buildMortonOrder(
  xs: Float64Array,
  ys: Float64Array,
  minX: number,
  maxX: number,
  minY: number,
  maxY: number
): number[] {
  const n = xs.length;
  const keys = new Uint32Array(n);
  const spanX = maxX - minX;
  const spanY = maxY - minY;
  const invSpanX = spanX > 0 ? 65535 / spanX : 0;
  const invSpanY = spanY > 0 ? 65535 / spanY : 0;

  for (let i = 0; i < n; i++) {
    const sx =
      spanX > 0
        ? Math.max(0, Math.min(65535, Math.floor((xs[i] - minX) * invSpanX)))
        : 0;
    const sy =
      spanY > 0
        ? Math.max(0, Math.min(65535, Math.floor((ys[i] - minY) * invSpanY)))
        : 0;
    keys[i] = morton2D_16(sx, sy);
  }

  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => {
    const ka = keys[a];
    const kb = keys[b];
    if (ka !== kb) return ka < kb ? -1 : 1;
    const ax = xs[a];
    const bx = xs[b];
    if (ax !== bx) return ax < bx ? -1 : 1;
    const ay = ys[a];
    const by = ys[b];
    if (ay !== by) return ay < by ? -1 : 1;
    return a - b;
  });
  return order;
}

function buildAngleOrder(xs: Float64Array, ys: Float64Array, depotX: number, depotY: number): number[] {
  const n = xs.length;
  const angles = new Float64Array(n);
  for (let i = 0; i < n; i++) angles[i] = Math.atan2(ys[i] - depotY, xs[i] - depotX);

  const order = new Array<number>(n);
  for (let i = 0; i < n; i++) order[i] = i;
  order.sort((a, b) => {
    const aa = angles[a];
    const ab = angles[b];
    if (aa !== ab) return aa < ab ? -1 : 1;
    const ax = xs[a];
    const bx = xs[b];
    if (ax !== bx) return ax < bx ? -1 : 1;
    const ay = ys[a];
    const by = ys[b];
    if (ay !== by) return ay < by ? -1 : 1;
    return a - b;
  });
  return order;
}

function rotateToBestDepotBreak(
  order: number[],
  xs: Float64Array,
  ys: Float64Array,
  depotX: number,
  depotY: number
): number[] {
  const n = order.length;
  if (n <= 1) return order.slice();

  let bestIdx = 0;
  let bestScore = Infinity;
  for (let i = 0; i < n; i++) {
    const cur = order[i];
    const prev = order[i === 0 ? n - 1 : i - 1];
    const score =
      distXY(depotX, depotY, xs[cur], ys[cur]) +
      distXY(depotX, depotY, xs[prev], ys[prev]) -
      distXY(xs[prev], ys[prev], xs[cur], ys[cur]);
    if (score < bestScore) {
      bestScore = score;
      bestIdx = i;
    }
  }

  if (bestIdx === 0) return order.slice();
  const rotated = new Array<number>(n);
  for (let i = 0; i < n; i++) rotated[i] = order[(bestIdx + i) % n];
  return rotated;
}

function routeDistance(order: number[], xs: Float64Array, ys: Float64Array, depotX: number, depotY: number): number {
  const n = order.length;
  if (n === 0) return 0;
  let total = distXY(depotX, depotY, xs[order[0]], ys[order[0]]);
  for (let i = 0; i < n - 1; i++) {
    const a = order[i];
    const b = order[i + 1];
    total += distXY(xs[a], ys[a], xs[b], ys[b]);
  }
  total += distXY(xs[order[n - 1]], ys[order[n - 1]], depotX, depotY);
  return total;
}

function reverseRange(order: number[], i: number, j: number): void {
  while (i < j) {
    const tmp = order[i];
    order[i] = order[j];
    order[j] = tmp;
    i++;
    j--;
  }
}

function twoOptWindowed(
  order: number[],
  xs: Float64Array,
  ys: Float64Array,
  depotX: number,
  depotY: number,
  window: number,
  passes: number
): void {
  const n = order.length;
  if (n < 4) return;

  const maxWindow = Math.max(4, Math.min(window, n - 1));
  const eps = 1e-12;

  for (let pass = 0; pass < passes; pass++) {
    let any = false;
    for (let i = 0; i < n - 2; i++) {
      const prevIdx = i === 0 ? -1 : order[i - 1];
      const aIdx = order[i];
      const prevX = prevIdx < 0 ? depotX : xs[prevIdx];
      const prevY = prevIdx < 0 ? depotY : ys[prevIdx];
      const aX = xs[aIdx];
      const aY = ys[aIdx];
      const distPrevA = distXY(prevX, prevY, aX, aY);

      const jMax = Math.min(n - 1, i + maxWindow);
      for (let j = i + 2; j <= jMax; j++) {
        const bIdx = order[j];
        const nextIdx = j === n - 1 ? -1 : order[j + 1];
        const bX = xs[bIdx];
        const bY = ys[bIdx];
        const nextX = nextIdx < 0 ? depotX : xs[nextIdx];
        const nextY = nextIdx < 0 ? depotY : ys[nextIdx];

        const oldDist = distPrevA + distXY(bX, bY, nextX, nextY);
        const newDist = distXY(prevX, prevY, bX, bY) + distXY(aX, aY, nextX, nextY);

        if (newDist + eps < oldDist) {
          reverseRange(order, i, j);
          any = true;
          break; // first-improvement
        }
      }
    }
    if (!any) break;
  }
}

const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const depotX = Number(data.depot?.[0] ?? 0);
const depotY = Number(data.depot?.[1] ?? 0);
const packages: PackageInput[] = Array.isArray(data.packages) ? data.packages : [];

const n = packages.length;
if (n === 0) {
  process.stdout.write(JSON.stringify({ routes: [] }));
  process.exit(0);
}

const ids = new Array<any>(n);
const xs = new Float64Array(n);
const ys = new Float64Array(n);

let minX = Infinity;
let minY = Infinity;
let maxX = -Infinity;
let maxY = -Infinity;

for (let i = 0; i < n; i++) {
  const p = packages[i];
  ids[i] = (p as any).id;
  const x = Number((p as any).x);
  const y = Number((p as any).y);
  xs[i] = x;
  ys[i] = y;
  if (x < minX) minX = x;
  if (x > maxX) maxX = x;
  if (y < minY) minY = y;
  if (y > maxY) maxY = y;
}

const candidates: number[][] = [];
try {
  candidates.push(buildMortonOrder(xs, ys, minX, maxX, minY, maxY));
} catch {
  // ignore
}
try {
  candidates.push(buildAngleOrder(xs, ys, depotX, depotY));
} catch {
  // ignore
}

let bestOrder: number[] = candidates.length ? candidates[0] : (() => {
  const o = new Array<number>(n);
  for (let i = 0; i < n; i++) o[i] = i;
  return o;
})();

let bestCost = Infinity;
for (let c = 0; c < candidates.length; c++) {
  const rotated = rotateToBestDepotBreak(candidates[c], xs, ys, depotX, depotY);
  const cost = routeDistance(rotated, xs, ys, depotX, depotY);
  if (cost < bestCost) {
    bestCost = cost;
    bestOrder = rotated;
  }
}

let searchWindow = 30;
let passes = 2;
if (n <= 400) {
  searchWindow = n;
  passes = 6;
} else if (n <= 2_000) {
  searchWindow = 150;
  passes = 4;
} else if (n <= 10_000) {
  searchWindow = 80;
  passes = 3;
} else if (n <= 50_000) {
  searchWindow = 35;
  passes = 2;
} else {
  searchWindow = 20;
  passes = 1;
}

twoOptWindowed(bestOrder, xs, ys, depotX, depotY, searchWindow, passes);

const route = new Array<any>(n);
for (let i = 0; i < n; i++) route[i] = ids[bestOrder[i]];

process.stdout.write(JSON.stringify({ routes: [route] }));
