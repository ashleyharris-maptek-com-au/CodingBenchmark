use std::cmp::{Ordering, Reverse};
use std::collections::{BinaryHeap, HashMap};
use std::hash::{BuildHasherDefault, Hasher};
use std::io::{self, Read};

#[derive(Default)]
struct FastHasher(u64);

#[inline]
fn mix64(mut x: u64) -> u64 {
    x = x.wrapping_add(0x9E37_79B9_7F4A_7C15);
    x = (x ^ (x >> 30)).wrapping_mul(0xBF58_476D_1CE4_E5B9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94D0_49BB_1331_11EB);
    x ^ (x >> 31)
}

impl Hasher for FastHasher {
    #[inline]
    fn finish(&self) -> u64 {
        self.0
    }

    #[inline]
    fn write(&mut self, bytes: &[u8]) {
        // Fallback: a tiny byte mixer. Most keys will use write_u64.
        let mut h = self.0 ^ 0xCBF2_9CE4_8422_2325;
        for &b in bytes {
            h ^= b as u64;
            h = h.wrapping_mul(0x1000_0000_01B3);
        }
        self.0 = mix64(h);
    }

    #[inline]
    fn write_u64(&mut self, i: u64) {
        self.0 = mix64(i);
    }

    #[inline]
    fn write_usize(&mut self, i: usize) {
        self.0 = mix64(i as u64);
    }

    #[inline]
    fn write_i64(&mut self, i: i64) {
        self.0 = mix64(i as u64);
    }
}

type FastBuildHasher = BuildHasherDefault<FastHasher>;
type FastHashMap<K, V> = HashMap<K, V, FastBuildHasher>;

struct Scanner {
    buf: Vec<u8>,
    idx: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, idx: 0 }
    }

    #[inline]
    fn is_ws(b: u8) -> bool {
        b <= b' '
    }

    fn next_token(&mut self) -> Option<&[u8]> {
        while self.idx < self.buf.len() && Self::is_ws(self.buf[self.idx]) {
            self.idx += 1;
        }
        if self.idx >= self.buf.len() {
            return None;
        }
        let start = self.idx;
        while self.idx < self.buf.len() && !Self::is_ws(self.buf[self.idx]) {
            self.idx += 1;
        }
        Some(&self.buf[start..self.idx])
    }

    #[inline]
    fn next_usize(&mut self) -> Option<usize> {
        let t = self.next_token()?;
        let s = unsafe { std::str::from_utf8_unchecked(t) };
        s.parse::<usize>().ok()
    }

    #[inline]
    fn next_f64(&mut self) -> Option<f64> {
        let t = self.next_token()?;
        let s = unsafe { std::str::from_utf8_unchecked(t) };
        s.parse::<f64>().ok()
    }

    #[inline]
    fn next_f32_allow_nan(&mut self) -> f32 {
        match self.next_token() {
            None => f32::NAN,
            Some(t) => {
                // Treat any parse failure as missing.
                let s = unsafe { std::str::from_utf8_unchecked(t) };
                s.parse::<f32>().unwrap_or(f32::NAN)
            }
        }
    }

    fn next_string(&mut self) -> Option<String> {
        let t = self.next_token()?;
        std::str::from_utf8(t).ok().map(|s| s.to_string())
    }
}

#[derive(Clone)]
struct HoleInfo {
    id: String,
    start_sample: usize,
    num_samples: usize,
}

#[derive(Clone, Copy)]
struct SampleMeta {
    hole_idx: u32,
    sample_idx: u32,
}

#[derive(Clone, Copy)]
struct CellEntry {
    key: u64,
    sample: u32,
}

#[derive(Clone, Copy)]
struct Scored {
    score: f32,
    sample: u32,
    prop: u32,
}

impl PartialEq for Scored {
    fn eq(&self, other: &Self) -> bool {
        self.score.to_bits() == other.score.to_bits()
            && self.sample == other.sample
            && self.prop == other.prop
    }
}
impl Eq for Scored {}

impl PartialOrd for Scored {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

impl Ord for Scored {
    fn cmp(&self, other: &Self) -> Ordering {
        self.score
            .total_cmp(&other.score)
            .then_with(|| self.sample.cmp(&other.sample))
            .then_with(|| self.prop.cmp(&other.prop))
    }
}

#[inline]
fn dist2(a: [f32; 3], b: [f32; 3]) -> f32 {
    let dx = a[0] - b[0];
    let dy = a[1] - b[1];
    let dz = a[2] - b[2];
    dx * dx + dy * dy + dz * dz
}

#[inline]
fn cell_key(ix: i32, iy: i32, iz: i32) -> u64 {
    // Coordinates are expected to be non-negative (we clamp at 0).
    let x = ix as u64;
    let y = iy as u64;
    let z = iz as u64;
    let h = x.wrapping_mul(0x9E37_79B9_7F4A_7C15)
        ^ y.wrapping_mul(0xBF58_476D_1CE4_E5B9)
        ^ z.wrapping_mul(0x94D0_49BB_1331_11EB);
    mix64(h)
}

#[inline]
fn to_cell_coord(pos: [f32; 3], origin: [f32; 3], inv_cell: f32) -> [i32; 3] {
    let mut ix = ((pos[0] - origin[0]) * inv_cell).floor() as i32;
    let mut iy = ((pos[1] - origin[1]) * inv_cell).floor() as i32;
    let mut iz = ((pos[2] - origin[2]) * inv_cell).floor() as i32;
    if ix < 0 {
        ix = 0;
    }
    if iy < 0 {
        iy = 0;
    }
    if iz < 0 {
        iz = 0;
    }
    [ix, iy, iz]
}

fn compute_cell_size(n_samples: usize, minp: [f32; 3], maxp: [f32; 3]) -> f32 {
    const TARGET_POINTS_PER_CELL: f64 = 32.0;
    let ex = (maxp[0] - minp[0]) as f64;
    let ey = (maxp[1] - minp[1]) as f64;
    let ez = (maxp[2] - minp[2]) as f64;
    let max_extent = ex.max(ey).max(ez).max(1e-3);
    let n = (n_samples as f64).max(1.0);

    let mut cell = if ex > 0.0 && ey > 0.0 && ez > 0.0 {
        let vol = ex * ey * ez;
        let target_vol = TARGET_POINTS_PER_CELL * vol / n;
        target_vol.cbrt()
    } else {
        max_extent / n.cbrt()
    };
    if !cell.is_finite() || cell <= 0.0 {
        cell = max_extent / n.cbrt();
    }

    // Small safety enlargement to pull in enough neighbors.
    cell *= 1.5;
    let min_cell = (max_extent / 10_000.0).max(1e-3);
    let max_cell = max_extent.max(1.0);
    (cell.clamp(min_cell, max_cell)) as f32
}

#[derive(Clone, Copy)]
struct RunningStat {
    n: u64,
    mean: f64,
    m2: f64,
}

impl RunningStat {
    fn new() -> Self {
        Self {
            n: 0,
            mean: 0.0,
            m2: 0.0,
        }
    }

    #[inline]
    fn push(&mut self, x: f64) {
        self.n += 1;
        let delta = x - self.mean;
        self.mean += delta / (self.n as f64);
        let delta2 = x - self.mean;
        self.m2 += delta * delta2;
    }

    fn stddev(&self) -> f64 {
        if self.n < 2 {
            return 0.0;
        }
        (self.m2 / ((self.n - 1) as f64)).sqrt()
    }
}

fn push_topk(heap: &mut BinaryHeap<Reverse<Scored>>, k: usize, cand: Scored) {
    if heap.len() < k {
        heap.push(Reverse(cand));
        return;
    }
    if let Some(&Reverse(smallest)) = heap.peek() {
        if cand.score > smallest.score {
            heap.pop();
            heap.push(Reverse(cand));
        }
    }
}

fn main() {
    let mut sc = Scanner::new();
    let num_holes = match sc.next_usize() {
        None => {
            println!("0");
            return;
        }
        Some(v) => v,
    };
    let num_props = match sc.next_usize() {
        None => {
            println!("0");
            return;
        }
        Some(v) => v,
    };

    let mut prop_names: Vec<String> = Vec::with_capacity(num_props);
    for _ in 0..num_props {
        if let Some(s) = sc.next_string() {
            prop_names.push(s);
        } else {
            // Missing property list; continue with what we have.
            break;
        }
    }
    let p = prop_names.len();
    if p == 0 {
        println!("0");
        return;
    }

    let mut holes: Vec<HoleInfo> = Vec::with_capacity(num_holes);
    let mut metas: Vec<SampleMeta> = Vec::new();
    let mut positions: Vec<[f32; 3]> = Vec::new();
    let mut values: Vec<Vec<f32>> = (0..p).map(|_| Vec::new()).collect();
    let mut stats: Vec<RunningStat> = (0..p).map(|_| RunningStat::new()).collect();

    let mut minp = [f32::INFINITY; 3];
    let mut maxp = [f32::NEG_INFINITY; 3];

    for hole_i in 0..num_holes {
        let hole_id = match sc.next_string() {
            None => break,
            Some(s) => s,
        };
        let start_x = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let start_y = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let start_z = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let dir_x = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let dir_y = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let dir_z = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let _length = match sc.next_f64() {
            None => break,
            Some(v) => v,
        };
        let num_samples = match sc.next_usize() {
            None => break,
            Some(v) => v,
        };

        let start_sample = positions.len();
        holes.push(HoleInfo {
            id: hole_id,
            start_sample,
            num_samples,
        });

        positions.reserve(num_samples);
        metas.reserve(num_samples);
        for col in values.iter_mut() {
            col.reserve(num_samples);
        }

        let norm = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (ux, uy, uz) = if norm.is_finite() && norm > 0.0 {
            let inv = 1.0 / norm;
            (dir_x * inv, dir_y * inv, dir_z * inv)
        } else {
            (0.0, 0.0, 0.0)
        };

        let hole_idx_u32 = hole_i as u32;
        for sidx in 0..num_samples {
            let depth = match sc.next_f64() {
                None => break,
                Some(v) => v,
            };

            let px = (start_x + ux * depth) as f32;
            let py = (start_y + uy * depth) as f32;
            let pz = (start_z + uz * depth) as f32;
            let pos = [px, py, pz];
            positions.push(pos);
            metas.push(SampleMeta {
                hole_idx: hole_idx_u32,
                sample_idx: sidx as u32,
            });

            for d in 0..3 {
                if pos[d] < minp[d] {
                    minp[d] = pos[d];
                }
                if pos[d] > maxp[d] {
                    maxp[d] = pos[d];
                }
            }

            for prop_i in 0..p {
                let v = sc.next_f32_allow_nan();
                values[prop_i].push(v);
                if v.is_finite() {
                    stats[prop_i].push(v as f64);
                }
            }
        }
    }

    let n = positions.len();
    if n == 0 {
        println!("0");
        return;
    }

    // Global per-property floors to prevent tiny local variance exploding scores.
    let mut std_floors: Vec<f64> = Vec::with_capacity(p);
    for st in &stats {
        let s = st.stddev();
        std_floors.push((s * 0.05).max(1e-6));
    }

    // Always do along-hole checks (very cheap and catches many typos).
    let mut best: FastHashMap<u64, f32> = FastHashMap::default();
    best.reserve(1024);

    let seq_z_thresh = if n <= 100_000 { 6.0_f32 } else { 7.0_f32 };
    for hole in &holes {
        let start = hole.start_sample;
        let ns = hole.num_samples;
        if start + ns > n {
            continue;
        }
        for s in 0..ns {
            let i = start + s;
            let iprev = if s > 0 { Some(start + s - 1) } else { None };
            let inext = if s + 1 < ns {
                Some(start + s + 1)
            } else {
                None
            };

            for prop_i in 0..p {
                let x = values[prop_i][i];
                if !x.is_finite() {
                    continue;
                }
                let mut have = 0u32;
                let mut sum = 0.0f64;
                let mut sumsq = 0.0f64;

                if let Some(j) = iprev {
                    let v = values[prop_i][j];
                    if v.is_finite() {
                        have += 1;
                        let vf = v as f64;
                        sum += vf;
                        sumsq += vf * vf;
                    }
                }
                if let Some(j) = inext {
                    let v = values[prop_i][j];
                    if v.is_finite() {
                        have += 1;
                        let vf = v as f64;
                        sum += vf;
                        sumsq += vf * vf;
                    }
                }
                if have == 0 {
                    continue;
                }
                let mean = sum / (have as f64);
                let var = (sumsq / (have as f64) - mean * mean).max(0.0);
                let std = var.sqrt() + std_floors[prop_i];
                let z = ((x as f64) - mean).abs() / std.max(1e-12);
                let zf = z as f32;
                if zf >= seq_z_thresh && zf.is_finite() {
                    let key = ((i as u64) << 32) | (prop_i as u64);
                    match best.get_mut(&key) {
                        Some(prev) => {
                            if zf > *prev {
                                *prev = zf;
                            }
                        }
                        None => {
                            best.insert(key, zf);
                        }
                    }
                }
            }
        }
    }

    // Spatial strategy selection (adaptive by dataset size).
    #[derive(Clone, Copy)]
    enum Strategy {
        Brute,
        GridSample,
        GridCell,
    }

    let strategy = if n <= 6_000 {
        Strategy::Brute
    } else if n <= 600_000 {
        Strategy::GridSample
    } else {
        Strategy::GridCell
    };

    let z_thresh = match strategy {
        Strategy::Brute => 4.5_f32,
        Strategy::GridSample => 5.0_f32,
        Strategy::GridCell => 6.0_f32,
    };
    let min_neighbors: u32 = match strategy {
        Strategy::Brute => 10,
        Strategy::GridSample => 10,
        Strategy::GridCell => 20,
    };

    // Build spatial grid for GridSample/GridCell (and also for Brute's radius selection).
    let cell_size = compute_cell_size(n, minp, maxp);
    let inv_cell = 1.0f32 / cell_size.max(1e-6);
    let eps2 = (cell_size * 0.1).max(1e-6);
    let eps2 = eps2 * eps2;

    let mut sample_cell_coords: Vec<[i32; 3]> = Vec::with_capacity(n);
    let mut entries: Vec<CellEntry> = Vec::with_capacity(n);
    for i in 0..n {
        let c = to_cell_coord(positions[i], minp, inv_cell);
        sample_cell_coords.push(c);
        let key = cell_key(c[0], c[1], c[2]);
        entries.push(CellEntry {
            key,
            sample: i as u32,
        });
    }
    entries.sort_unstable_by(|a, b| a.key.cmp(&b.key));

    let mut unique_keys: Vec<u64> = Vec::new();
    let mut cell_coords_unique: Vec<[i32; 3]> = Vec::new();
    let mut starts: Vec<usize> = Vec::new();

    let mut last_key: Option<u64> = None;
    for (idx, e) in entries.iter().enumerate() {
        if last_key != Some(e.key) {
            last_key = Some(e.key);
            unique_keys.push(e.key);
            starts.push(idx);
            cell_coords_unique.push(sample_cell_coords[e.sample as usize]);
        }
    }
    starts.push(entries.len());
    let cells = unique_keys.len();

    let mut key_to_cell: FastHashMap<u64, usize> = FastHashMap::default();
    key_to_cell.reserve(cells * 2 + 1);
    for (cell_idx, &k) in unique_keys.iter().enumerate() {
        key_to_cell.insert(k, cell_idx);
    }

    let mut sample_cell_index: Vec<u32> = vec![0; n];
    for cell_idx in 0..cells {
        for idx in starts[cell_idx]..starts[cell_idx + 1] {
            let sidx = entries[idx].sample as usize;
            sample_cell_index[sidx] = cell_idx as u32;
        }
    }

    match strategy {
        Strategy::Brute => {
            let ex = (maxp[0] - minp[0]).max(1e-3);
            let ey = (maxp[1] - minp[1]).max(1e-3);
            let ez = (maxp[2] - minp[2]).max(1e-3);
            let max_extent = ex.max(ey).max(ez);
            let approx_spacing = (max_extent as f64 / (n as f64).cbrt()).max(cell_size as f64);
            let radius = (approx_spacing * 3.0) as f32;
            let r2 = radius * radius;

            let mut sum_w = vec![0.0f64; p];
            let mut sum_wv = vec![0.0f64; p];
            let mut sum_wv2 = vec![0.0f64; p];
            let mut cnt = vec![0u32; p];

            for i in 0..n {
                for v in sum_w.iter_mut() {
                    *v = 0.0;
                }
                for v in sum_wv.iter_mut() {
                    *v = 0.0;
                }
                for v in sum_wv2.iter_mut() {
                    *v = 0.0;
                }
                for c in cnt.iter_mut() {
                    *c = 0;
                }

                let pi = positions[i];
                for j in 0..n {
                    if j == i {
                        continue;
                    }
                    let d2 = dist2(pi, positions[j]);
                    if d2 > r2 {
                        continue;
                    }
                    let w = 1.0f32 / (d2 + eps2);
                    let w64 = w as f64;
                    for prop_i in 0..p {
                        let v = values[prop_i][j];
                        if v.is_finite() {
                            let vf = v as f64;
                            sum_w[prop_i] += w64;
                            sum_wv[prop_i] += w64 * vf;
                            sum_wv2[prop_i] += w64 * vf * vf;
                            cnt[prop_i] += 1;
                        }
                    }
                }

                for prop_i in 0..p {
                    let x = values[prop_i][i];
                    if !x.is_finite() {
                        continue;
                    }
                    if cnt[prop_i] < min_neighbors || sum_w[prop_i] <= 0.0 {
                        continue;
                    }
                    let mean = sum_wv[prop_i] / sum_w[prop_i];
                    let var = (sum_wv2[prop_i] / sum_w[prop_i] - mean * mean).max(0.0);
                    let std = var.sqrt() + std_floors[prop_i];
                    let z = ((x as f64) - mean).abs() / std.max(1e-12);
                    let zf = z as f32;
                    if zf >= z_thresh && zf.is_finite() {
                        let key = ((i as u64) << 32) | (prop_i as u64);
                        match best.get_mut(&key) {
                            Some(prev) => {
                                if zf > *prev {
                                    *prev = zf;
                                }
                            }
                            None => {
                                best.insert(key, zf);
                            }
                        }
                    }
                }
            }
        }
        Strategy::GridSample => {
            const MAX_POINTS_PER_CELL_SCAN: usize = 128;
            const MAX_TOTAL_NEIGHBORS: usize = 1024;

            let mut sum_w = vec![0.0f64; p];
            let mut sum_wv = vec![0.0f64; p];
            let mut sum_wv2 = vec![0.0f64; p];
            let mut cnt = vec![0u32; p];

            for i in 0..n {
                for v in sum_w.iter_mut() {
                    *v = 0.0;
                }
                for v in sum_wv.iter_mut() {
                    *v = 0.0;
                }
                for v in sum_wv2.iter_mut() {
                    *v = 0.0;
                }
                for c in cnt.iter_mut() {
                    *c = 0;
                }

                let c0 = sample_cell_coords[i];
                let pi = positions[i];
                let mut scanned = 0usize;

                'cells: for dx in -1..=1 {
                    for dy in -1..=1 {
                        for dz in -1..=1 {
                            let nx = c0[0] + dx;
                            let ny = c0[1] + dy;
                            let nz = c0[2] + dz;
                            if nx < 0 || ny < 0 || nz < 0 {
                                continue;
                            }
                            let key = cell_key(nx, ny, nz);
                            let Some(&cell_idx) = key_to_cell.get(&key) else {
                                continue;
                            };
                            let start = starts[cell_idx];
                            let end = starts[cell_idx + 1];
                            let len = end - start;
                            let step = if len > MAX_POINTS_PER_CELL_SCAN {
                                (len / MAX_POINTS_PER_CELL_SCAN).max(1)
                            } else {
                                1
                            };
                            for idx in (start..end).step_by(step) {
                                let j = entries[idx].sample as usize;
                                if j == i {
                                    continue;
                                }
                                scanned += 1;
                                if scanned > MAX_TOTAL_NEIGHBORS {
                                    break 'cells;
                                }
                                let d2 = dist2(pi, positions[j]);
                                let w = 1.0f32 / (d2 + eps2);
                                let w64 = w as f64;
                                for prop_i in 0..p {
                                    let v = values[prop_i][j];
                                    if v.is_finite() {
                                        let vf = v as f64;
                                        sum_w[prop_i] += w64;
                                        sum_wv[prop_i] += w64 * vf;
                                        sum_wv2[prop_i] += w64 * vf * vf;
                                        cnt[prop_i] += 1;
                                    }
                                }
                            }
                        }
                    }
                }

                for prop_i in 0..p {
                    let x = values[prop_i][i];
                    if !x.is_finite() {
                        continue;
                    }
                    if cnt[prop_i] < min_neighbors || sum_w[prop_i] <= 0.0 {
                        continue;
                    }
                    let mean = sum_wv[prop_i] / sum_w[prop_i];
                    let var = (sum_wv2[prop_i] / sum_w[prop_i] - mean * mean).max(0.0);
                    let std = var.sqrt() + std_floors[prop_i];
                    let z = ((x as f64) - mean).abs() / std.max(1e-12);
                    let zf = z as f32;
                    if zf >= z_thresh && zf.is_finite() {
                        let key = ((i as u64) << 32) | (prop_i as u64);
                        match best.get_mut(&key) {
                            Some(prev) => {
                                if zf > *prev {
                                    *prev = zf;
                                }
                            }
                            None => {
                                best.insert(key, zf);
                            }
                        }
                    }
                }
            }
        }
        Strategy::GridCell => {
            // Cell stats per (cell, prop).
            let mut cell_count: Vec<u32> = vec![0; cells * p];
            let mut cell_sum: Vec<f64> = vec![0.0; cells * p];
            let mut cell_sumsq: Vec<f64> = vec![0.0; cells * p];

            for cell_idx in 0..cells {
                for idx in starts[cell_idx]..starts[cell_idx + 1] {
                    let sidx = entries[idx].sample as usize;
                    for prop_i in 0..p {
                        let v = values[prop_i][sidx];
                        if v.is_finite() {
                            let k = cell_idx * p + prop_i;
                            let vf = v as f64;
                            cell_count[k] += 1;
                            cell_sum[k] += vf;
                            cell_sumsq[k] += vf * vf;
                        }
                    }
                }
            }

            // Precompute aggregated neighbor-cell stats per cell (27 neighbors).
            let mut agg_count: Vec<u32> = vec![0; cells * p];
            let mut agg_sum: Vec<f64> = vec![0.0; cells * p];
            let mut agg_sumsq: Vec<f64> = vec![0.0; cells * p];

            for cell_idx in 0..cells {
                let c0 = cell_coords_unique[cell_idx];
                for dx in -1..=1 {
                    for dy in -1..=1 {
                        for dz in -1..=1 {
                            let nx = c0[0] + dx;
                            let ny = c0[1] + dy;
                            let nz = c0[2] + dz;
                            if nx < 0 || ny < 0 || nz < 0 {
                                continue;
                            }
                            let key = cell_key(nx, ny, nz);
                            let Some(&ncell) = key_to_cell.get(&key) else {
                                continue;
                            };
                            let base_dst = cell_idx * p;
                            let base_src = ncell * p;
                            for prop_i in 0..p {
                                let src = base_src + prop_i;
                                let dst = base_dst + prop_i;
                                agg_count[dst] += cell_count[src];
                                agg_sum[dst] += cell_sum[src];
                                agg_sumsq[dst] += cell_sumsq[src];
                            }
                        }
                    }
                }
            }

            // Score each sample against its cell's precomputed neighbor aggregates (excluding self).
            for i in 0..n {
                let cell_idx = sample_cell_index[i] as usize;
                let base = cell_idx * p;
                for prop_i in 0..p {
                    let x = values[prop_i][i];
                    if !x.is_finite() {
                        continue;
                    }
                    let k = base + prop_i;
                    let mut cnt = agg_count[k];
                    let mut sum = agg_sum[k];
                    let mut sumsq = agg_sumsq[k];
                    // Remove self contribution (from its own cell) if present.
                    if cnt > 0 {
                        cnt -= 1;
                        let xf = x as f64;
                        sum -= xf;
                        sumsq -= xf * xf;
                    }
                    if cnt < min_neighbors {
                        continue;
                    }
                    let mean = sum / (cnt as f64);
                    let var = (sumsq / (cnt as f64) - mean * mean).max(0.0);
                    let std = var.sqrt() + std_floors[prop_i];
                    let z = ((x as f64) - mean).abs() / std.max(1e-12);
                    let zf = z as f32;
                    if zf >= z_thresh && zf.is_finite() {
                        let key = ((i as u64) << 32) | (prop_i as u64);
                        match best.get_mut(&key) {
                            Some(prev) => {
                                if zf > *prev {
                                    *prev = zf;
                                }
                            }
                            None => {
                                best.insert(key, zf);
                            }
                        }
                    }
                }
            }
        }
    }

    // Keep only the top K (after de-duping by (sample, prop)).
    const MAX_RESULTS: usize = 20_000;
    let mut heap: BinaryHeap<Reverse<Scored>> = BinaryHeap::with_capacity(MAX_RESULTS + 1);
    for (key, score) in best {
        if !score.is_finite() {
            continue;
        }
        let sample = (key >> 32) as u32;
        let prop = (key & 0xFFFF_FFFF) as u32;
        if (sample as usize) >= n || (prop as usize) >= p {
            continue;
        }
        push_topk(
            &mut heap,
            MAX_RESULTS,
            Scored {
                score,
                sample,
                prop,
            },
        );
    }

    let mut out: Vec<Scored> = heap.into_iter().map(|Reverse(s)| s).collect();
    out.sort_unstable_by(|a, b| b.score.total_cmp(&a.score));

    println!("{}", out.len());
    for s in out {
        let meta = metas[s.sample as usize];
        let hole_id = &holes[meta.hole_idx as usize].id;
        let prop_name = &prop_names[s.prop as usize];
        println!(
            "{} {} {} {:.6}",
            hole_id, meta.sample_idx, prop_name, s.score
        );
    }
}
