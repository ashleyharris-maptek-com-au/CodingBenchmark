#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <limits>
#include <cstdio>
#include <string>

using namespace std;

/*--------------- fast input ---------------*/

class FastScanner {
    static const size_t BUFSIZE = 1 << 20; // 1 MB
    char buf[BUFSIZE];
    size_t pos, len;
public:
    FastScanner() : pos(0), len(0) {}
    inline char nextChar() {
        if (pos >= len) {
            len = fread(buf, 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }
    bool skip() {
        char c;
        do {
            c = nextChar();
            if (c == EOF) return false;
        } while (c <= ' ');
        pos--;
        return true;
    }
    bool readInt(int &out) {
        if (!skip()) return false;
        char c = nextChar();
        bool neg = false;
        if (c == '-') { neg = true; c = nextChar(); }
        int val = 0;
        while (c > ' ') {
            val = val * 10 + (c - '0');
            c = nextChar();
        }
        out = neg ? -val : val;
        return true;
    }
    bool readDouble(double &out) {
        if (!skip()) return false;
        char c = nextChar();
        bool neg = false;
        if (c == '-') { neg = true; c = nextChar(); }
        double val = 0;
        // integer part
        while (c > ' ' && c != '.') {
            val = val * 10 + (c - '0');
            c = nextChar();
        }
        // fractional part
        if (c == '.') {
            c = nextChar();
            double factor = 0.1;
            while (c > ' ') {
                if (c <= ' ') break;
                val += (c - '0') * factor;
                factor *= 0.1;
                c = nextChar();
            }
        }
        out = neg ? -val : val;
        return true;
    }
};

/*--------------- distance helpers ---------------*/

inline double dist2(int a, int b, const vector<double> &xs, const vector<double> &ys) {
    double dx = xs[a] - xs[b];
    double dy = ys[a] - ys[b];
    return dx * dx + dy * dy;
}

/*--------------- Hilbert curve mapping ---------------*/

static uint64_t xy2d(int n, int x, int y) {
    uint64_t rx, ry, s, d = 0;
    for (s = n / 2; s > 0; s /= 2) {
        rx = (x & s) > 0;
        ry = (y & s) > 0;
        d += (uint64_t)s * s * ((3 * rx) ^ ry);
        if (ry == 0) {
            if (rx == 1) {
                x = n - 1 - x;
                y = n - 1 - y;
            }
            // swap x and y
            int t = x;
            x = y;
            y = t;
        }
    }
    return d;
}

/*--------------- exact solver for tiny instances ---------------*/

vector<int> solveExact(int N, const vector<double> &xs, const vector<double> &ys) {
    if (N == 1) return {0};

    // pre‑compute Euclidean distances
    vector<vector<double>> dist(N, vector<double>(N));
    for (int i = 0; i < N; ++i) {
        dist[i][i] = 0.0;
        for (int j = i + 1; j < N; ++j) {
            double dx = xs[i] - xs[j];
            double dy = ys[i] - ys[j];
            double d = sqrt(dx * dx + dy * dy);
            dist[i][j] = dist[j][i] = d;
        }
    }

    const int FULL = (1 << N) - 1;
    const double INF = numeric_limits<double>::infinity();

    // dp[mask][i] = minimal cost to start at 0, visit 'mask', end at i
    vector<double> dp((FULL + 1) * N, INF);
    vector<int> parent((FULL + 1) * N, -1);
    auto idx = [&](int mask, int i) { return mask * N + i; };

    dp[idx(1 << 0, 0)] = 0.0;

    for (int mask = 1; mask <= FULL; ++mask) {
        for (int i = 0; i < N; ++i) {
            if (!(mask & (1 << i))) continue;
            double cur = dp[idx(mask, i)];
            if (cur >= INF) continue;
            for (int j = 0; j < N; ++j) {
                if (mask & (1 << j)) continue;
                int nmask = mask | (1 << j);
                double nd = cur + dist[i][j];
                double &ref = dp[idx(nmask, j)];
                if (nd < ref) {
                    ref = nd;
                    parent[idx(nmask, j)] = i;
                }
            }
        }
    }

    // choose best endpoint (must not be the start city 0)
    double best = INF;
    int bestEnd = -1;
    for (int i = 1; i < N; ++i) {
        double cur = dp[idx(FULL, i)];
        if (cur < best) {
            best = cur;
            bestEnd = i;
        }
    }

    // reconstruct path
    vector<int> route;
    int mask = FULL;
    int cur = bestEnd;
    while (cur != -1) {
        route.push_back(cur);
        int prev = parent[idx(mask, cur)];
        mask ^= (1 << cur);
        cur = prev;
    }
    reverse(route.begin(), route.end());          // now from 0 to end

    // ensure it really starts with 0 (should already)
    if (route.empty() || route[0] != 0) {
        for (size_t i = 0; i < route.size(); ++i) {
            if (route[i] == 0) {
                rotate(route.begin(), route.begin() + i, route.end());
                break;
            }
        }
    }
    return route;
}

/*--------------- 2‑opt improvement for moderate sizes ---------------*/

void twoOpt(vector<int> &route, const vector<double> &xs, const vector<double> &ys, int maxPasses) {
    int n = (int)route.size();
    if (n <= 3) return;
    bool improved = true;
    int passes = 0;
    while (improved && passes < maxPasses) {
        improved = false;
        for (int i = 0; i < n - 2; ++i) {
            for (int j = i + 2; j < n; ++j) {
                int i1 = i;
                int i2 = (i + 1) % n;
                int j1 = j;
                int j2 = (j + 1) % n;

                double cur = dist2(i1, i2, xs, ys) + dist2(j1, j2, xs, ys);
                double ndist = dist2(i1, j1, xs, ys) + dist2(i2, j2, xs, ys);

                if (ndist < cur) {
                    reverse(route.begin() + i + 1, route.begin() + j + 1);
                    improved = true;
                }
            }
        }
        ++passes;
    }
}

/*--------------- heuristic for larger instances ---------------*/

vector<int> solveHeuristic(int N, const vector<double> &xs, const vector<double> &ys) {
    // decide Hilbert curve order: smallest power of two >= N
    int order = 0;
    while ((1ULL << order) < (unsigned long long)N) ++order;
    if (order < 1) order = 1;
    int n = 1 << order;               // grid size 2^order × 2^order

    // bounding box
    double minX = xs[0], maxX = xs[0];
    double minY = ys[0], maxY = ys[0];
    for (int i = 1; i < N; ++i) {
        if (xs[i] < minX) minX = xs[i];
        if (xs[i] > maxX) maxX = xs[i];
        if (ys[i] < minY) minY = ys[i];
        if (ys[i] > maxY) maxY = ys[i];
    }
    double rangeX = maxX - minX;
    double rangeY = maxY - minY;
    double scaleX = (rangeX == 0.0) ? 0.0 : (n - 1) / rangeX;
    double scaleY = (rangeY == 0.0) ? 0.0 : (n - 1) / rangeY;

    struct Pair { uint64_t h; int idx; };
    vector<Pair> arr;
    arr.reserve(N);

    for (int i = 0; i < N; ++i) {
        int xi, yi;
        if (scaleX == 0.0) xi = 0;
        else {
            xi = (int)((xs[i] - minX) * scaleX);
            if (xi < 0) xi = 0;
            else if (xi >= n) xi = n - 1;
        }
        if (scaleY == 0.0) yi = 0;
        else {
            yi = (int)((ys[i] - minY) * scaleY);
            if (yi < 0) yi = 0;
            else if (yi >= n) yi = n - 1;
        }
        uint64_t h = xy2d(n, xi, yi);
        arr.push_back({h, i});
    }

    sort(arr.begin(), arr.end(),
         [](const Pair &a, const Pair &b) {
             if (a.h != b.h) return a.h < b.h;
             return a.idx < b.idx;
         });

    // locate city 0 and rotate
    int pos0 = -1;
    for (int i = 0; i < N; ++i) {
        if (arr[i].idx == 0) { pos0 = i; break; }
    }

    vector<int> route(N);
    for (int i = 0; i < N; ++i)
        route[i] = arr[(pos0 + i) % N].idx;

    // modest 2‑opt for moderate sizes
    if (N <= 5000) {
        twoOpt(route, xs, ys, 5);
        // re‑rotate so that city 0 is first
        for (int i = 0; i < N; ++i) {
            if (route[i] == 0) {
                rotate(route.begin(), route.begin() + i, route.end());
                break;
            }
        }
    }
    return route;
}

/*--------------- output routine ---------------*/

inline void appendInt(string &out, int x) {
    char buf[12];
    int len = 0;
    if (x == 0) {
        buf[len++] = '0';
    } else {
        int tmp = x;
        while (tmp > 0) {
            buf[len++] = char('0' + tmp % 10);
            tmp /= 10;
        }
        for (int i = 0; i < len / 2; ++i) {
            char t = buf[i];
            buf[i] = buf[len - 1 - i];
            buf[len - 1 - i] = t;
        }
    }
    out.append(buf, len);
    out.push_back(' ');
}

/*--------------- main ---------------*/

int main() {
    FastScanner fs;
    int N;
    if (!fs.readInt(N)) return 0;

    vector<double> xs(N), ys(N);
    for (int i = 0; i < N; ++i) {
        double x, y;
        fs.readDouble(x);
        fs.readDouble(y);
        xs[i] = x;
        ys[i] = y;
    }

    vector<int> route;
    if (N <= 20) {
        route = solveExact(N, xs, ys);
    } else {
        route = solveHeuristic(N, xs, ys);
    }

    // write output
    string out;
    out.reserve(N * 12);
    for (int v : route) appendInt(out, v);
    if (!out.empty() && out.back() == ' ') out.pop_back();
    out.push_back('\n');
    fwrite(out.c_str(), 1, out.size(), stdout);
    return 0;
}
