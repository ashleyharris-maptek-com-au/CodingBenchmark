#include <algorithm>
#include <atomic>
#include <cmath>
#include <cstdint>
#include <fstream>
#include <iomanip>
#include <iostream>
#include <limits>
#include <string>
#include <thread>

using namespace std;

static constexpr double PI = 3.141592653589793238462643383279502884;

static double clampd(double v, double lo, double hi) {
    return std::max(lo, std::min(hi, v));
}

static double wrapPi(double a) {
    while (a > PI) a -= 2.0 * PI;
    while (a < -PI) a += 2.0 * PI;
    return a;
}

static double norm2(double x, double y) {
    return std::sqrt(x * x + y * y);
}

static double norm3(double x, double y, double z) {
    return std::sqrt(x * x + y * y + z * z);
}

struct TerrainHint {
    atomic<bool> ready{false};
    atomic<double> targetGround{-1.0};
    atomic<double> localCeiling{numeric_limits<double>::infinity()};
};

static uint16_t read_u16le(const unsigned char* p) {
    return static_cast<uint16_t>(p[0] | (static_cast<unsigned int>(p[1]) << 8));
}

static void terrainProbeThread(string path, int width, int depth, int tx, int ty, TerrainHint* hint) {
    if (width <= 0 || depth <= 0 || tx < 0 || tx >= width || ty < 0 || ty >= depth) return;

    ifstream in(path, ios::binary);
    if (!in) return;

    const int radius = 8;
    const int x0 = max(0, tx - radius);
    const int x1 = min(width - 1, tx + radius);
    const int y0 = max(0, ty - radius);
    const int y1 = min(depth - 1, ty + radius);

    double highestGround = -1.0;
    double lowestFloatingBottom = numeric_limits<double>::infinity();
    unsigned char buf[6];

    for (int x = x0; x <= x1; ++x) {
        for (int y = y0; y <= y1; ++y) {
            uint64_t idx = (static_cast<uint64_t>(x) * static_cast<uint64_t>(depth) +
                            static_cast<uint64_t>(y)) * 6ull;
            in.seekg(static_cast<streamoff>(idx), ios::beg);
            if (!in.read(reinterpret_cast<char*>(buf), 6)) continue;
            uint16_t a = read_u16le(buf);
            uint16_t b = read_u16le(buf + 2);
            uint16_t c = read_u16le(buf + 4);
            highestGround = max(highestGround, static_cast<double>(a));
            if (b != c) lowestFloatingBottom = min(lowestFloatingBottom, static_cast<double>(b));
        }
    }

    hint->targetGround.store(highestGround, memory_order_relaxed);
    hint->localCeiling.store(lowestFloatingBottom, memory_order_relaxed);
    hint->ready.store(true, memory_order_release);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double widthD, depthD, heightD, gravity, maxThrust, fuel0;
    double startX, startY, startZ, targetX, targetY, targetZ;
    if (!(cin >> widthD >> depthD >> heightD >> gravity >> maxThrust >> fuel0)) return 0;
    if (!(cin >> startX >> startY >> startZ >> targetX >> targetY >> targetZ)) return 0;

    string terrainPath;
    getline(cin, terrainPath);
    if (terrainPath.empty()) getline(cin, terrainPath);

    string marker;
    cin >> marker;

    int width = static_cast<int>(llround(widthD));
    int depth = static_cast<int>(llround(depthD));

    TerrainHint hint;
    thread probe;
    int txCell = static_cast<int>(floor(targetX));
    int tyCell = static_cast<int>(floor(targetY));
    try {
        probe = thread(terrainProbeThread, terrainPath, width, depth, txCell, tyCell, &hint);
    } catch (...) {
    }

    cout.setf(ios::fixed);
    cout << setprecision(6);

    const double landingRadius = max(5.0, min(widthD, depthD) * 0.02);
    const double g = gravity > 0.0 ? gravity : 1.62;
    const double accelMax = max(maxThrust, g + 0.1);
    const double hover = clampd(g / accelMax, 0.0, 1.0);

    double x, y, z, vx, vy, vz, pitch, yaw, roll, pitchRate, yawRate, rollRate, fuel;
    while (cin >> x >> y >> z >> vx >> vy >> vz >> pitch >> yaw >> roll
               >> pitchRate >> yawRate >> rollRate >> fuel) {

        double dx = targetX - x;
        double dy = targetY - y;
        double hdist = norm2(dx, dy);
        double speedH = norm2(vx, vy);

        double ceiling = hint.localCeiling.load(memory_order_relaxed);
        double floorHint = hint.targetGround.load(memory_order_relaxed);
        double clearance = max(25.0, min(1500.0, max(widthD, depthD) * 0.01));
        double highAlt = targetZ + clearance + 0.25 * hdist;
        if (isfinite(ceiling)) highAlt = min(highAlt, ceiling - 12.0);
        if (floorHint >= 0.0) highAlt = max(highAlt, floorHint + 20.0);
        highAlt = clampd(highAlt, targetZ + 12.0, heightD - 20.0);

        double desiredVx = 0.0;
        double desiredVy = 0.0;
        double desiredVz = 0.0;

        if (hdist > landingRadius * 0.85) {
            double maxHSpeed = clampd(0.85 * sqrt(max(1.0, hdist)), 5.0, 95.0);
            double slowRadius = max(landingRadius * 4.0, speedH * speedH / 2.2 + 30.0);
            if (hdist < slowRadius) maxHSpeed = min(maxHSpeed, max(1.0, hdist * 0.18));
            desiredVx = dx / max(1.0, hdist) * maxHSpeed;
            desiredVy = dy / max(1.0, hdist) * maxHSpeed;

            double altErr = highAlt - z;
            desiredVz = clampd(0.35 * altErr, -18.0, 18.0);
            if (z < targetZ + 25.0 && hdist > landingRadius * 2.0) desiredVz = max(desiredVz, 6.0);
        } else {
            desiredVx = clampd(dx * 0.22, -1.5, 1.5);
            desiredVy = clampd(dy * 0.22, -1.5, 1.5);

            double alt = max(0.0, z - targetZ);
            double sink = -clampd(0.42 * sqrt(max(0.0, alt)), 0.25, 2.1);
            if (alt < 12.0) sink = -clampd(0.12 * alt + 0.15, 0.15, 1.2);
            if (alt < 3.0) sink = -0.25;
            desiredVz = sink;
        }

        double axCmd = clampd((desiredVx - vx) * 0.65 + dx * 0.015, -0.45 * accelMax, 0.45 * accelMax);
        double ayCmd = clampd((desiredVy - vy) * 0.65 + dy * 0.015, -0.45 * accelMax, 0.45 * accelMax);

        double alt = z - targetZ;
        double vzErr = desiredVz - vz;
        double zRef = (hdist > landingRadius * 0.85) ? highAlt : targetZ;
        double azCmd = g + clampd(0.85 * vzErr + 0.025 * (zRef - z), -0.65 * accelMax, 0.85 * accelMax);
        if (alt < 20.0 && hdist < landingRadius) {
            azCmd = g + clampd(1.35 * vzErr + 0.08 * (targetZ - z), -0.4 * accelMax, 0.95 * accelMax);
        }
        azCmd = max(0.05 * accelMax, azCmd);

        double totalA = norm3(axCmd, ayCmd, azCmd);
        if (totalA > accelMax) {
            double s = accelMax / totalA;
            axCmd *= s;
            ayCmd *= s;
            azCmd *= s;
            totalA = accelMax;
        }

        double thrust = clampd(totalA / accelMax, 0.0, 1.0);
        if (fuel <= 0.05) thrust = 0.0;

        double horizA = norm2(axCmd, ayCmd);
        double desiredTilt = atan2(horizA, max(0.01, azCmd));
        desiredTilt = clampd(desiredTilt, 0.0, 0.65);

        double desiredYaw = yaw;
        double desiredPitch = 0.0;
        if (horizA > 1e-5) {
            desiredYaw = atan2(-ayCmd, -axCmd);
            desiredPitch = desiredTilt;
        }
        if (hdist < landingRadius * 1.3 && z - targetZ < 35.0) {
            desiredPitch = 0.0;
            desiredYaw = yaw;
        }

        double pErr = wrapPi(desiredPitch - pitch);
        double yErr = wrapPi(desiredYaw - yaw);
        double rErr = wrapPi(0.0 - roll);

        double pitchCmd = clampd(2.7 * pErr - 1.1 * pitchRate, -1.0, 1.0);
        double yawCmd = clampd(2.2 * yErr - 1.0 * yawRate, -1.0, 1.0);
        double rollCmd = clampd(2.0 * rErr - 0.9 * rollRate, -1.0, 1.0);

        if (std::abs(pitch) > 1.35) {
            pitchCmd = clampd(-2.5 * pitch - 1.2 * pitchRate, -1.0, 1.0);
            thrust = max(thrust, hover);
        }

        cout << thrust << ' ' << pitchCmd << ' ' << yawCmd << ' ' << rollCmd << '\n';
        cout.flush();
    }

    if (probe.joinable()) probe.join();
    return 0;
}
