using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream = Console.OpenStandardInput();
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len, _ptr;

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
            } while (c <= 32 && c >= 0);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int stockLength = fs.NextInt();

        if (n <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        int[] lengths = new int[n];
        int[] indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            lengths[i] = fs.NextInt();
            indices[i] = i;
        }

        Array.Sort(lengths, indices);

        int size = 1;
        while (size < n) size <<= 1;

        int[] tree = new int[size << 1];
        int[] remaining = new int[n];
        int[] head = new int[n];
        int[] tail = new int[n];
        int[] itemIndex = new int[n];
        int[] next = new int[n];
        Array.Fill(head, -1);
        Array.Fill(tail, -1);
        Array.Fill(next, -1);

        int stockCount = 0;

        for (int sorted = n - 1; sorted >= 0; sorted--)
        {
            int len = lengths[sorted];
            int cutIndex = indices[sorted];
            int item = sorted;
            itemIndex[item] = cutIndex;

            int stock;
            if (tree[1] >= len)
            {
                int node = 1;
                while (node < size)
                {
                    int left = node << 1;
                    node = tree[left] >= len ? left : left + 1;
                }
                stock = node - size;
            }
            else
            {
                stock = stockCount++;
                remaining[stock] = stockLength;
            }

            remaining[stock] -= len;

            if (head[stock] < 0)
            {
                head[stock] = tail[stock] = item;
            }
            else
            {
                next[tail[stock]] = item;
                tail[stock] = item;
            }

            int pos = stock + size;
            tree[pos] = remaining[stock];
            pos >>= 1;
            while (pos > 0)
            {
                int left = tree[pos << 1];
                int right = tree[(pos << 1) + 1];
                int value = left >= right ? left : right;
                if (tree[pos] == value) break;
                tree[pos] = value;
                pos >>= 1;
            }
        }

        using var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        output.WriteLine(stockCount);

        for (int s = 0; s < stockCount; s++)
        {
            bool first = true;
            for (int item = head[s]; item >= 0; item = next[item])
            {
                if (!first) output.Write(' ');
                first = false;
                output.Write(itemIndex[item]);
            }
            output.WriteLine();
        }
    }
}
