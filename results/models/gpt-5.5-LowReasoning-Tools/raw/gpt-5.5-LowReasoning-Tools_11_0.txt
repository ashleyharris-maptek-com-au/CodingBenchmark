use std::cmp::Ordering;
use std::f64::consts::PI;
use std::io::{self, Read};

const EPS: f64 = 1.0e-9;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    min_x: f64,
    min_y: f64,
    max_x: f64,
    max_y: f64,
}

#[derive(Clone)]
struct Piece {
    id: usize,
    pts: Vec<Pt>,
    area: f64,
}

#[derive(Clone, Copy)]
struct Placement {
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
}

#[derive(Clone, Copy)]
struct PackedItem {
    id: usize,
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
}

#[derive(Clone, Copy)]
struct Shelf {
    y: f64,
    h: f64,
    used_w: f64,
}

struct StockBin {
    shelves: Vec<Shelf>,
    used_h: f64,
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn polygon_area(poly: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        s += a.x * b.y - a.y * b.x;
    }
    0.5 * s
}

fn bbox(poly: &[Pt]) -> BBox {
    let mut b = BBox {
        min_x: f64::INFINITY,
        min_y: f64::INFINITY,
        max_x: f64::NEG_INFINITY,
        max_y: f64::NEG_INFINITY,
    };
    for &p in poly {
        b.min_x = b.min_x.min(p.x);
        b.min_y = b.min_y.min(p.y);
        b.max_x = b.max_x.max(p.x);
        b.max_y = b.max_y.max(p.y);
    }
    b
}

fn rotate_points(poly: &[Pt], ang: f64) -> Vec<Pt> {
    let c = ang.cos();
    let s = ang.sin();
    poly.iter()
        .map(|p| Pt {
            x: p.x * c - p.y * s,
            y: p.x * s + p.y * c,
        })
        .collect()
}

fn on_segment(a: Pt, b: Pt, p: Pt) -> bool {
    cross(a, b, p).abs() <= 1.0e-8
        && p.x >= a.x.min(b.x) - 1.0e-8
        && p.x <= a.x.max(b.x) + 1.0e-8
        && p.y >= a.y.min(b.y) - 1.0e-8
        && p.y <= a.y.max(b.y) + 1.0e-8
}

fn point_in_poly(poly: &[Pt], p: Pt) -> bool {
    let mut inside = false;
    for i in 0..poly.len() {
        let a = poly[i];
        let b = poly[(i + 1) % poly.len()];
        if on_segment(a, b, p) {
            return true;
        }
        let crosses = (a.y > p.y) != (b.y > p.y);
        if crosses {
            let x_at_y = (b.x - a.x) * (p.y - a.y) / (b.y - a.y) + a.x;
            if x_at_y >= p.x - EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn segments_properly_intersect(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let c1 = cross(a, b, c);
    let c2 = cross(a, b, d);
    let c3 = cross(c, d, a);
    let c4 = cross(c, d, b);
    if c1.abs() <= 1.0e-8 && on_segment(a, b, c) {
        return false;
    }
    if c2.abs() <= 1.0e-8 && on_segment(a, b, d) {
        return false;
    }
    if c3.abs() <= 1.0e-8 && on_segment(c, d, a) {
        return false;
    }
    if c4.abs() <= 1.0e-8 && on_segment(c, d, b) {
        return false;
    }
    (c1 > EPS) != (c2 > EPS) && (c3 > EPS) != (c4 > EPS)
}

fn polygon_inside(container: &[Pt], inner: &[Pt]) -> bool {
    for &p in inner {
        if !point_in_poly(container, p) {
            return false;
        }
    }
    for i in 0..inner.len() {
        let a = inner[i];
        let b = inner[(i + 1) % inner.len()];
        let mid = Pt {
            x: (a.x + b.x) * 0.5,
            y: (a.y + b.y) * 0.5,
        };
        if !point_in_poly(container, mid) {
            return false;
        }
        for j in 0..container.len() {
            if segments_properly_intersect(a, b, container[j], container[(j + 1) % container.len()]) {
                return false;
            }
        }
    }
    true
}

fn translated(poly: &[Pt], dx: f64, dy: f64) -> Vec<Pt> {
    poly.iter().map(|p| Pt { x: p.x + dx, y: p.y + dy }).collect()
}

fn is_axis_aligned_rectangle(poly: &[Pt]) -> bool {
    if poly.len() != 4 {
        return false;
    }
    let b = bbox(poly);
    let mut corners = 0;
    for &p in poly {
        let x_ok = (p.x - b.min_x).abs() <= 1.0e-7 || (p.x - b.max_x).abs() <= 1.0e-7;
        let y_ok = (p.y - b.min_y).abs() <= 1.0e-7 || (p.y - b.max_y).abs() <= 1.0e-7;
        if x_ok && y_ok {
            corners += 1;
        }
    }
    corners == 4
}

fn try_place_one(stock: &[Pt], piece: &[Pt], rotations: &[f64]) -> (f64, f64, f64) {
    let sb = bbox(stock);
    let stock_c = Pt {
        x: (sb.min_x + sb.max_x) * 0.5,
        y: (sb.min_y + sb.max_y) * 0.5,
    };
    let grid = [0.5, 0.25, 0.75, 0.1, 0.9, 0.4, 0.6];
    for &rot in rotations {
        let rp = rotate_points(piece, rot);
        let rb = bbox(&rp);
        let w = rb.max_x - rb.min_x;
        let h = rb.max_y - rb.min_y;
        if w > (sb.max_x - sb.min_x) + 1.0e-8 || h > (sb.max_y - sb.min_y) + 1.0e-8 {
            continue;
        }
        let mut candidates = Vec::with_capacity(grid.len() * grid.len() + 5);
        candidates.push(Pt {
            x: stock_c.x - (rb.min_x + rb.max_x) * 0.5,
            y: stock_c.y - (rb.min_y + rb.max_y) * 0.5,
        });
        for &gx in &grid {
            for &gy in &grid {
                let left = sb.min_x + gx * ((sb.max_x - sb.min_x) - w);
                let bot = sb.min_y + gy * ((sb.max_y - sb.min_y) - h);
                candidates.push(Pt {
                    x: left - rb.min_x,
                    y: bot - rb.min_y,
                });
            }
        }
        for t in candidates {
            let moved = translated(&rp, t.x, t.y);
            if polygon_inside(stock, &moved) {
                return (t.x, t.y, rot);
            }
        }
    }

    let pb = bbox(piece);
    (stock_c.x - (pb.min_x + pb.max_x) * 0.5, stock_c.y - (pb.min_y + pb.max_y) * 0.5, 0.0)
}

fn pack_rectangular(stock: &[Pt], pieces: &[Piece]) -> Vec<Placement> {
    let sb = bbox(stock);
    let sw = sb.max_x - sb.min_x;
    let sh = sb.max_y - sb.min_y;
    let mut order: Vec<usize> = (0..pieces.len()).collect();
    order.sort_by(|&a, &b| {
        pieces[b]
            .area
            .partial_cmp(&pieces[a].area)
            .unwrap_or(Ordering::Equal)
    });

    let mut out = vec![
        Placement {
            stock: 0,
            x: 0.0,
            y: 0.0,
            rot: 0.0,
        };
        pieces.len()
    ];
    let mut bins: Vec<StockBin> = Vec::new();

    for idx in order {
        let piece = &pieces[idx];
        let b0 = bbox(&piece.pts);
        let dims0 = (b0.max_x - b0.min_x, b0.max_y - b0.min_y, 0.0, b0);
        let r90 = rotate_points(&piece.pts, PI * 0.5);
        let b90 = bbox(&r90);
        let dims90 = (b90.max_x - b90.min_x, b90.max_y - b90.min_y, PI * 0.5, b90);
        let mut variants = [dims0, dims90];
        variants.sort_by(|a, b| {
            (a.1.max(a.0))
                .partial_cmp(&b.1.max(b.0))
                .unwrap_or(Ordering::Equal)
        });

        let mut placed: Option<PackedItem> = None;
        'search: for (stock_i, bin) in bins.iter_mut().enumerate() {
            for &(w, h, rot, rb) in &variants {
                if w > sw + 1.0e-8 || h > sh + 1.0e-8 {
                    continue;
                }
                for shelf in &mut bin.shelves {
                    if h <= shelf.h + 1.0e-8 && shelf.used_w + w <= sw + 1.0e-8 {
                        let left = sb.min_x + shelf.used_w;
                        let bottom = sb.min_y + shelf.y;
                        shelf.used_w += w;
                        placed = Some(PackedItem {
                            id: piece.id,
                            stock: stock_i,
                            x: left - rb.min_x,
                            y: bottom - rb.min_y,
                            rot,
                        });
                        break 'search;
                    }
                }
                if bin.used_h + h <= sh + 1.0e-8 {
                    let left = sb.min_x;
                    let bottom = sb.min_y + bin.used_h;
                    bin.shelves.push(Shelf {
                        y: bin.used_h,
                        h,
                        used_w: w,
                    });
                    bin.used_h += h;
                    placed = Some(PackedItem {
                        id: piece.id,
                        stock: stock_i,
                        x: left - rb.min_x,
                        y: bottom - rb.min_y,
                        rot,
                    });
                    break 'search;
                }
            }
        }

        if placed.is_none() {
            let stock_i = bins.len();
            bins.push(StockBin {
                shelves: Vec::new(),
                used_h: 0.0,
            });
            for &(w, h, rot, rb) in &variants {
                if w <= sw + 1.0e-8 && h <= sh + 1.0e-8 {
                    bins[stock_i].shelves.push(Shelf { y: 0.0, h, used_w: w });
                    bins[stock_i].used_h = h;
                    placed = Some(PackedItem {
                        id: piece.id,
                        stock: stock_i,
                        x: sb.min_x - rb.min_x,
                        y: sb.min_y - rb.min_y,
                        rot,
                    });
                    break;
                }
            }
        }

        if let Some(p) = placed {
            out[p.id] = Placement {
                stock: p.stock,
                x: p.x,
                y: p.y,
                rot: p.rot,
            };
        } else {
            let (x, y, rot) = try_place_one(stock, &piece.pts, &[0.0, PI * 0.5, PI, PI * 1.5]);
            out[piece.id] = Placement {
                stock: bins.len(),
                x,
                y,
                rot,
            };
            bins.push(StockBin {
                shelves: Vec::new(),
                used_h: sh,
            });
        }
    }
    out
}

fn parse_input() -> Option<(Vec<Pt>, Vec<Piece>)> {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).ok()?;
    let mut it = s.split_whitespace();
    let sv: usize = it.next()?.parse().ok()?;
    let np: usize = it.next()?.parse().ok()?;
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next()?.parse().ok()?;
        let y: f64 = it.next()?.parse().ok()?;
        stock.push(Pt { x, y });
    }
    let mut pieces = Vec::with_capacity(np);
    for id in 0..np {
        let pv: usize = it.next()?.parse().ok()?;
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next()?.parse().ok()?;
            let y: f64 = it.next()?.parse().ok()?;
            pts.push(Pt { x, y });
        }
        pieces.push(Piece {
            id,
            area: polygon_area(&pts).abs(),
            pts,
        });
    }
    Some((stock, pieces))
}

fn main() {
    let Some((stock, pieces)) = parse_input() else {
        return;
    };
    if pieces.is_empty() {
        println!("0");
        return;
    }

    let placements = if is_axis_aligned_rectangle(&stock) {
        pack_rectangular(&stock, &pieces)
    } else {
        let rotations = if pieces.len() > 1000 {
            vec![0.0, PI * 0.5]
        } else {
            vec![0.0, PI * 0.5, PI, PI * 1.5, PI * 0.25, -PI * 0.25]
        };
        pieces
            .iter()
            .enumerate()
            .map(|(i, p)| {
                let (x, y, rot) = try_place_one(&stock, &p.pts, &rotations);
                Placement {
                    stock: i,
                    x,
                    y,
                    rot,
                }
            })
            .collect::<Vec<_>>()
    };

    let num_stocks = placements.iter().map(|p| p.stock).max().unwrap_or(0) + 1;
    println!("{}", num_stocks);
    for p in placements {
        println!("{} {:.10} {:.10} {:.10}", p.stock, p.x, p.y, p.rot);
    }
}
