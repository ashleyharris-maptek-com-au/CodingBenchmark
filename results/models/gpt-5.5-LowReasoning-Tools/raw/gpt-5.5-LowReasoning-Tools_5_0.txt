use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Solver {
    w: usize,
    h: usize,
    n: usize,
    blocked: Vec<bool>,
    used: Vec<bool>,
    path: Vec<usize>,
    require_cycle: bool,
    deadline: Instant,
}

impl Solver {
    fn id_xy(&self, id: usize) -> (usize, usize) { (id % self.w, id / self.w) }

    fn neighbors(&self, id: usize, out: &mut Vec<usize>) {
        out.clear();
        let (x, y) = self.id_xy(id);
        if x > 0 { out.push(id - 1); }
        if x + 1 < self.w { out.push(id + 1); }
        if y > 0 { out.push(id - self.w); }
        if y + 1 < self.h { out.push(id + self.w); }
    }

    fn adjacent(&self, a: usize, b: usize) -> bool {
        let (ax, ay) = self.id_xy(a);
        let (bx, by) = self.id_xy(b);
        ax.abs_diff(bx) + ay.abs_diff(by) == 1
    }

    fn free_count(&self) -> usize {
        self.blocked.iter().filter(|&&b| !b).count()
    }

    fn initial_checks(&self) -> bool {
        if self.w == 0 || self.h == 0 || self.blocked[0] { return false; }
        let total = self.free_count();
        if total == 0 { return false; }
        if self.require_cycle && total < 4 { return false; }

        let mut black = 0usize;
        let mut white = 0usize;
        for id in 0..self.n {
            if !self.blocked[id] {
                let (x, y) = self.id_xy(id);
                if (x + y) & 1 == 0 { black += 1; } else { white += 1; }
            }
        }
        if self.require_cycle {
            if black != white { return false; }
        } else if black.abs_diff(white) > 1 || white > black {
            return false;
        }

        self.connected_all_free()
    }

    fn connected_all_free(&self) -> bool {
        let total = self.free_count();
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        q.push_back(0);
        seen[0] = true;
        let mut cnt = 0usize;
        let mut ns = Vec::with_capacity(4);
        while let Some(v) = q.pop_front() {
            cnt += 1;
            self.neighbors(v, &mut ns);
            for &u in &ns {
                if !seen[u] && !self.blocked[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        cnt == total
    }

    fn reachable_remaining(&self, cur: usize) -> bool {
        let remaining = self.free_count() - self.path.len() + 1;
        let mut seen = vec![false; self.n];
        let mut q = VecDeque::new();
        q.push_back(cur);
        seen[cur] = true;
        let mut cnt = 0usize;
        let mut ns = Vec::with_capacity(4);
        while let Some(v) = q.pop_front() {
            cnt += 1;
            self.neighbors(v, &mut ns);
            for &u in &ns {
                if !seen[u] && !self.blocked[u] && (!self.used[u] || u == cur) {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        cnt == remaining
    }

    fn onward_degree(&self, v: usize, cur: usize) -> usize {
        let mut ns = Vec::with_capacity(4);
        self.neighbors(v, &mut ns);
        ns.into_iter()
            .filter(|&u| !self.blocked[u] && (!self.used[u] || u == cur))
            .count()
    }

    fn local_prune(&self, cur: usize) -> bool {
        if !self.reachable_remaining(cur) { return true; }
        let left = self.free_count() - self.path.len();
        if left == 0 {
            return self.require_cycle && !self.adjacent(cur, 0);
        }
        let mut forced = 0usize;
        for id in 0..self.n {
            if !self.blocked[id] && !self.used[id] {
                let d = self.onward_degree(id, cur);
                if d == 0 { return true; }
                if d == 1 { forced += 1; }
            }
        }
        forced > 1
    }

    fn search(&mut self, cur: usize) -> bool {
        if Instant::now() > self.deadline { return false; }
        if self.path.len() == self.free_count() {
            return !self.require_cycle || self.adjacent(cur, 0);
        }
        if self.local_prune(cur) { return false; }

        let mut ns = Vec::with_capacity(4);
        self.neighbors(cur, &mut ns);
        ns.retain(|&u| !self.blocked[u] && !self.used[u]);
        ns.sort_by_key(|&u| {
            let deg = self.onward_degree(u, cur);
            let (x, y) = self.id_xy(u);
            (deg, x.abs_diff(self.w / 2) + y.abs_diff(self.h / 2))
        });

        for u in ns {
            self.used[u] = true;
            self.path.push(u);
            if self.search(u) { return true; }
            self.path.pop();
            self.used[u] = false;
        }
        false
    }
}

fn empty_rect_path(w: usize, h: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    if w == 0 || h == 0 { return None; }
    if !require_cycle {
        let mut p = Vec::with_capacity(w * h);
        for y in 0..h {
            if y & 1 == 0 {
                for x in 0..w { p.push((x, y)); }
            } else {
                for x in (0..w).rev() { p.push((x, y)); }
            }
        }
        return Some(p);
    }
    if w < 2 || h < 2 || (w * h) & 1 == 1 { return None; }

    fn cycle_even_width(w: usize, h: usize) -> Vec<(usize, usize)> {
        let mut p = Vec::with_capacity(w * h);
        for x in 0..w { p.push((x, 0)); }
        for x in (0..w).rev() {
            let offset = w - 1 - x;
            if offset & 1 == 0 {
                for y in 1..h { p.push((x, y)); }
            } else {
                for y in (1..h).rev() { p.push((x, y)); }
            }
        }
        p
    }

    if w & 1 == 0 {
        Some(cycle_even_width(w, h))
    } else {
        Some(cycle_even_width(h, w).into_iter().map(|(y, x)| (x, y)).collect())
    }
}

fn validate(w: usize, h: usize, blocked: &[bool], require_cycle: bool, p: &[(usize, usize)]) -> bool {
    let free = blocked.iter().filter(|&&b| !b).count();
    if p.len() != free || p.first() != Some(&(0, 0)) { return false; }
    let mut seen = vec![false; w * h];
    for (i, &(x, y)) in p.iter().enumerate() {
        if x >= w || y >= h { return false; }
        let id = y * w + x;
        if blocked[id] || seen[id] { return false; }
        seen[id] = true;
        if i > 0 {
            let (px, py) = p[i - 1];
            if px.abs_diff(x) + py.abs_diff(y) != 1 { return false; }
        }
    }
    if require_cycle {
        let &(x, y) = p.last().unwrap();
        if x + y != 1 { return false; }
    }
    true
}

fn print_path(p: &[(usize, usize)]) {
    let mut out = String::new();
    out.push_str(&format!("{}\n", p.len()));
    for &(x, y) in p {
        out.push_str(&format!("{} {}\n", x, y));
    }
    print!("{}", out);
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let w: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let h: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let obstacle_count: usize = it.next().unwrap_or("0").parse().unwrap_or(0);
    let require_cycle = it.next().unwrap_or("0") == "1";
    if w == 0 || h == 0 {
        println!("0");
        return;
    }
    let n = match w.checked_mul(h) {
        Some(v) => v,
        None => { println!("0"); return; }
    };
    let mut blocked = vec![false; n];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap_or("0").parse().unwrap_or(usize::MAX);
        let y: usize = it.next().unwrap_or("0").parse().unwrap_or(usize::MAX);
        if x < w && y < h { blocked[y * w + x] = true; }
    }

    if obstacle_count == 0 {
        if let Some(p) = empty_rect_path(w, h, require_cycle) {
            if validate(w, h, &blocked, require_cycle, &p) {
                print_path(&p);
                return;
            }
        }
    }

    let mut solver = Solver {
        w,
        h,
        n,
        blocked: blocked.clone(),
        used: vec![false; n],
        path: Vec::new(),
        require_cycle,
        deadline: Instant::now() + Duration::from_secs(285),
    };

    if !solver.initial_checks() {
        println!("0");
        return;
    }
    solver.used[0] = true;
    solver.path.push(0);

    if solver.search(0) {
        let p: Vec<(usize, usize)> = solver.path.iter().map(|&id| solver.id_xy(id)).collect();
        if validate(w, h, &blocked, require_cycle, &p) {
            print_path(&p);
            return;
        }
    }
    println!("0");
}
