struct Vec3 {
    double x = 0, y = 0, z = 0;
};

struct Tri {
    int a = 0, b = 0, c = 0;
};

struct Aabb {
    Vec3 mn{std::numeric_limits<double>::infinity(),
            std::numeric_limits<double>::infinity(),
            std::numeric_limits<double>::infinity()};
    Vec3 mx{-std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity()};
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static double dot(const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; }
static Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y * b.z - a.z * b.y, a.z * b.x - a.x * b.z, a.x * b.y - a.y * b.x};
}

static void grow(Aabb& b, const Vec3& p) {
    b.mn.x = std::min(b.mn.x, p.x); b.mn.y = std::min(b.mn.y, p.y); b.mn.z = std::min(b.mn.z, p.z);
    b.mx.x = std::max(b.mx.x, p.x); b.mx.y = std::max(b.mx.y, p.y); b.mx.z = std::max(b.mx.z, p.z);
}

static bool overlaps(const Aabb& a, const Aabb& b) {
    return a.mn.x <= b.mx.x && a.mx.x >= b.mn.x &&
           a.mn.y <= b.mx.y && a.mx.y >= b.mn.y &&
           a.mn.z <= b.mx.z && a.mx.z >= b.mn.z;
}

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;

    Aabb bounds() const {
        Aabb b;
        for (const auto& p : v) grow(b, p);
        return b;
    }
};

struct Bvh {
    struct Node {
        Aabb box;
        int left = -1, right = -1;
        int begin = 0, end = 0;
        bool leaf() const { return left < 0; }
    };

    const Mesh* mesh = nullptr;
    std::vector<int> ids;
    std::vector<Node> nodes;
    std::vector<Vec3> centroids;
    std::vector<Aabb> triBoxes;

    explicit Bvh(const Mesh& m) : mesh(&m), ids(m.f.size()), centroids(m.f.size()), triBoxes(m.f.size()) {
        std::iota(ids.begin(), ids.end(), 0);
        for (size_t i = 0; i < m.f.size(); ++i) {
            const Tri& t = m.f[i];
            centroids[i] = (m.v[t.a] + m.v[t.b] + m.v[t.c]) * (1.0 / 3.0);
            grow(triBoxes[i], m.v[t.a]);
            grow(triBoxes[i], m.v[t.b]);
            grow(triBoxes[i], m.v[t.c]);
        }
        if (!ids.empty()) build(0, static_cast<int>(ids.size()));
    }

    int build(int begin, int end) {
        Node n;
        n.begin = begin; n.end = end;
        Aabb cb;
        for (int i = begin; i < end; ++i) {
            int id = ids[i];
            grow(n.box, triBoxes[id].mn);
            grow(n.box, triBoxes[id].mx);
            grow(cb, centroids[id]);
        }
        int idx = static_cast<int>(nodes.size());
        nodes.push_back(n);
        if (end - begin <= 12) return idx;

        Vec3 span = cb.mx - cb.mn;
        int axis = (span.x >= span.y && span.x >= span.z) ? 0 : (span.y >= span.z ? 1 : 2);
        int mid = (begin + end) / 2;
        std::nth_element(ids.begin() + begin, ids.begin() + mid, ids.begin() + end, [&](int a, int b) {
            if (axis == 0) return centroids[a].x < centroids[b].x;
            if (axis == 1) return centroids[a].y < centroids[b].y;
            return centroids[a].z < centroids[b].z;
        });
        nodes[idx].left = build(begin, mid);
        nodes[idx].right = build(mid, end);
        return idx;
    }
};

static bool rayBox(const Vec3& o, const Vec3& d, const Aabb& b) {
    double t0 = 0.0, t1 = std::numeric_limits<double>::infinity();
    const double ov[3] = {o.x, o.y, o.z};
    const double dv[3] = {d.x, d.y, d.z};
    const double mn[3] = {b.mn.x, b.mn.y, b.mn.z};
    const double mx[3] = {b.mx.x, b.mx.y, b.mx.z};
    for (int i = 0; i < 3; ++i) {
        if (std::abs(dv[i]) < 1e-30) {
            if (ov[i] < mn[i] || ov[i] > mx[i]) return false;
        } else {
            double inv = 1.0 / dv[i];
            double a = (mn[i] - ov[i]) * inv;
            double c = (mx[i] - ov[i]) * inv;
            if (a > c) std::swap(a, c);
            t0 = std::max(t0, a);
            t1 = std::min(t1, c);
            if (t0 > t1) return false;
        }
    }
    return true;
}

static bool rayTri(const Vec3& o, const Vec3& d, const Vec3& a, const Vec3& b, const Vec3& c, double& t) {
    const double eps = 1e-12;
    Vec3 e1 = b - a, e2 = c - a;
    Vec3 p = cross(d, e2);
    double det = dot(e1, p);
    if (std::abs(det) < eps) return false;
    double inv = 1.0 / det;
    Vec3 s = o - a;
    double u = dot(s, p) * inv;
    if (u < -eps || u > 1.0 + eps) return false;
    Vec3 q = cross(s, e1);
    double v = dot(d, q) * inv;
    if (v < -eps || u + v > 1.0 + eps) return false;
    t = dot(e2, q) * inv;
    return t > eps;
}

static bool insideMesh(const Mesh& m, const Bvh& bvh, const Vec3& p) {
    if (bvh.nodes.empty()) return false;
    const Vec3 dir{0.892137358, 0.371390676, 0.249136440};
    int hits = 0;
    std::vector<int> stack;
    stack.push_back(0);
    while (!stack.empty()) {
        int ni = stack.back();
        stack.pop_back();
        const auto& n = bvh.nodes[ni];
        if (!rayBox(p, dir, n.box)) continue;
        if (n.leaf()) {
            for (int i = n.begin; i < n.end; ++i) {
                int id = bvh.ids[i];
                const Tri& tr = m.f[id];
                double t = 0;
                if (rayTri(p, dir, m.v[tr.a], m.v[tr.b], m.v[tr.c], t)) ++hits;
            }
        } else {
            stack.push_back(n.left);
            stack.push_back(n.right);
        }
    }
    return (hits & 1) != 0;
}

struct QuantKey {
    long long x, y, z;
    bool operator==(const QuantKey& o) const { return x == o.x && y == o.y && z == o.z; }
};

struct QuantHash {
    size_t operator()(const QuantKey& k) const {
        uint64_t x = static_cast<uint64_t>(k.x) * 11995408973635179863ull;
        uint64_t y = static_cast<uint64_t>(k.y) * 10150724397891781847ull;
        uint64_t z = static_cast<uint64_t>(k.z) * 1442695040888963407ull;
        return static_cast<size_t>(x ^ (y << 1) ^ (z << 7));
    }
};

static QuantKey keyOf(const Vec3& p) {
    const double s = 1e9;
    return {static_cast<long long>(std::llround(p.x * s)),
            static_cast<long long>(std::llround(p.y * s)),
            static_cast<long long>(std::llround(p.z * s))};
}

static Mesh mergeMeshes(const Mesh& a, const Mesh& b) {
    Mesh out;
    out.v.reserve(a.v.size() + b.v.size());
    out.f.reserve(a.f.size() + b.f.size());
    std::unordered_map<QuantKey, int, QuantHash> map;
    map.reserve(a.v.size() + b.v.size());
    auto addV = [&](const Vec3& p) {
        QuantKey k = keyOf(p);
        auto it = map.find(k);
        if (it != map.end()) return it->second;
        int id = static_cast<int>(out.v.size());
        out.v.push_back(p);
        map.emplace(k, id);
        return id;
    };
    auto addMesh = [&](const Mesh& m) {
        std::vector<int> remap(m.v.size());
        for (size_t i = 0; i < m.v.size(); ++i) remap[i] = addV(m.v[i]);
        for (const Tri& t : m.f) out.f.push_back({remap[t.a], remap[t.b], remap[t.c]});
    };
    addMesh(a);
    addMesh(b);
    return out;
}

static Mesh classifiedUnion(const Mesh& a, const Mesh& b) {
    Bvh bvhA(a), bvhB(b);
    Mesh ao, bo;
    ao.v = a.v; bo.v = b.v;
    ao.f.reserve(a.f.size());
    bo.f.reserve(b.f.size());
    for (const Tri& t : a.f) {
        Vec3 c = (a.v[t.a] + a.v[t.b] + a.v[t.c]) * (1.0 / 3.0);
        if (!insideMesh(b, bvhB, c)) ao.f.push_back(t);
    }
    for (const Tri& t : b.f) {
        Vec3 c = (b.v[t.a] + b.v[t.b] + b.v[t.c]) * (1.0 / 3.0);
        if (!insideMesh(a, bvhA, c)) bo.f.push_back(t);
    }
    return mergeMeshes(ao, bo);
}

static bool allVerticesInside(const Mesh& inner, const Mesh& outer, const Bvh& outerBvh) {
    for (const Vec3& p : inner.v) {
        if (!insideMesh(outer, outerBvh, p)) return false;
    }
    return !inner.v.empty();
}

static Mesh readMesh(int nv, int nf) {
    Mesh m;
    m.v.resize(nv);
    m.f.resize(nf);
    for (auto& p : m.v) std::cin >> p.x >> p.y >> p.z;
    for (auto& t : m.f) std::cin >> t.a >> t.b >> t.c;
    return m;
}

static void writeMesh(const Mesh& m) {
    std::cout << m.v.size() << ' ' << m.f.size() << '\n';
    std::cout << std::setprecision(17);
    for (const auto& p : m.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : m.f) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int va, fa, vb, fb;
    if (!(std::cin >> va >> fa >> vb >> fb)) return 0;
    Mesh a = readMesh(va, fa);
    Mesh b = readMesh(vb, fb);

    Aabb ba = a.bounds(), bb = b.bounds();
    if (!overlaps(ba, bb)) {
        writeMesh(mergeMeshes(a, b));
        return 0;
    }

    Bvh bvhA(a), bvhB(b);
    bool bInA = allVerticesInside(b, a, bvhA);
    bool aInB = allVerticesInside(a, b, bvhB);
    if (bInA && !aInB) {
        writeMesh(a);
    } else if (aInB && !bInA) {
        writeMesh(b);
    } else {
        writeMesh(classifiedUnion(a, b));
    }
    return 0;
}
