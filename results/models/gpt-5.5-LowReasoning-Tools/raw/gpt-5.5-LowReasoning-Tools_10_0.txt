// 2D cutting stock / rectangle bin packing solver.
// C++20, standard library only.
//
// Strategy:
// - Sort rectangles by descending "hardness" (larger side and area first).
// - Pack each board with a skyline bottom-left heuristic.
// - For small/medium instances, test every existing board.
// - For very large instances, test a bounded set of promising boards so runtime
//   remains predictable for 50k rectangles.
//
// The algorithm is heuristic (the problem is NP-hard), but it always emits valid
// non-overlapping placements for every rectangle that fits on a board in at least
// one orientation.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

using std::int64_t;

struct Rect {
    int id = 0;
    int w = 0;
    int h = 0;
    int64_t area = 0;
};

struct Answer {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Node {
    int x = 0;
    int y = 0;
    int w = 0;
};

struct Candidate {
    int score = std::numeric_limits<int>::max();
    int waste = std::numeric_limits<int>::max();
    int y = 0;
    int x = 0;
    int board = -1;
    int index = -1;
    int w = 0;
    int h = 0;
    int rotated = 0;
};

class Board {
public:
    Board(int width, int height) : W(width), H(height), freeArea(int64_t(width) * height) {
        skyline.push_back({0, 0, width});
    }

    bool findPlacement(int rw, int rh, int rotated, Candidate& best) const {
        bool found = false;
        for (int i = 0; i < static_cast<int>(skyline.size()); ++i) {
            int y = 0;
            int waste = 0;
            if (!fitsAt(i, rw, rh, y, waste)) {
                continue;
            }

            // Prefer lower placement, then less covered skyline waste, then left.
            int score = y + rh;
            if (!found ||
                score < best.score ||
                (score == best.score && waste < best.waste) ||
                (score == best.score && waste == best.waste && y < best.y) ||
                (score == best.score && waste == best.waste && y == best.y && skyline[i].x < best.x)) {
                best.score = score;
                best.waste = waste;
                best.y = y;
                best.x = skyline[i].x;
                best.index = i;
                best.w = rw;
                best.h = rh;
                best.rotated = rotated;
                found = true;
            }
        }
        return found;
    }

    void place(const Candidate& c) {
        Node n{c.x, c.y + c.h, c.w};
        skyline.insert(skyline.begin() + c.index, n);

        int i = c.index + 1;
        while (i < static_cast<int>(skyline.size())) {
            Node& prev = skyline[i - 1];
            Node& cur = skyline[i];
            int overlap = (prev.x + prev.w) - cur.x;
            if (overlap <= 0) {
                break;
            }
            cur.x += overlap;
            cur.w -= overlap;
            if (cur.w <= 0) {
                skyline.erase(skyline.begin() + i);
            } else {
                break;
            }
        }

        merge();
        freeArea -= int64_t(c.w) * c.h;
        if (freeArea < 0) {
            freeArea = 0;
        }
    }

    int64_t remainingArea() const {
        return freeArea;
    }

    int nodeCount() const {
        return static_cast<int>(skyline.size());
    }

private:
    int W = 0;
    int H = 0;
    int64_t freeArea = 0;
    std::vector<Node> skyline;

    bool fitsAt(int idx, int rw, int rh, int& outY, int& outWaste) const {
        const int x = skyline[idx].x;
        if (x + rw > W) {
            return false;
        }

        int widthLeft = rw;
        int y = skyline[idx].y;
        int waste = 0;
        int i = idx;
        while (widthLeft > 0) {
            if (i >= static_cast<int>(skyline.size())) {
                return false;
            }
            y = std::max(y, skyline[i].y);
            if (y + rh > H) {
                return false;
            }
            int used = std::min(widthLeft, skyline[i].w);
            waste += used * (y - skyline[i].y);
            widthLeft -= used;
            ++i;
        }

        outY = y;
        outWaste = waste;
        return true;
    }

    void merge() {
        for (int i = 0; i + 1 < static_cast<int>(skyline.size());) {
            if (skyline[i].y == skyline[i + 1].y) {
                skyline[i].w += skyline[i + 1].w;
                skyline.erase(skyline.begin() + i + 1);
            } else {
                ++i;
            }
        }
    }
};

static bool canFitBoard(const Rect& r, int W, int H) {
    return (r.w <= W && r.h <= H) || (r.h <= W && r.w <= H);
}

static bool betterGlobalCandidate(const Candidate& a, const Candidate& b, const std::vector<Board>& boards) {
    if (a.index < 0) {
        return false;
    }
    if (b.index < 0) {
        return true;
    }
    if (a.score != b.score) {
        return a.score < b.score;
    }
    if (a.waste != b.waste) {
        return a.waste < b.waste;
    }
    int64_t areaA = int64_t(a.w) * a.h;
    int64_t areaB = int64_t(b.w) * b.h;
    int64_t remA = boards[a.board].remainingArea() - areaA;
    int64_t remB = boards[b.board].remainingArea() - areaB;
    if (remA != remB) {
        return remA < remB;
    }
    if (a.y != b.y) {
        return a.y < b.y;
    }
    if (a.x != b.x) {
        return a.x < b.x;
    }
    return a.board < b.board;
}

static int boardSearchLimit(int n, int boardCount) {
    if (n <= 2000 || boardCount <= 256) {
        return boardCount;
    }
    if (n <= 10000) {
        return std::min(boardCount, 512);
    }
    return std::min(boardCount, 192);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    int W = 0;
    int H = 0;
    if (!(std::cin >> N >> W >> H)) {
        return 0;
    }

    std::vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        int w, h;
        std::cin >> w >> h;
        rects[i] = {i, w, h, int64_t(w) * h};
    }

    std::vector<Answer> ans(N);
    std::vector<Rect> order = rects;
    std::stable_sort(order.begin(), order.end(), [](const Rect& a, const Rect& b) {
        int aLong = std::max(a.w, a.h), bLong = std::max(b.w, b.h);
        int aShort = std::min(a.w, a.h), bShort = std::min(b.w, b.h);
        if (aLong != bLong) {
            return aLong > bLong;
        }
        if (a.area != b.area) {
            return a.area > b.area;
        }
        if (aShort != bShort) {
            return aShort > bShort;
        }
        return a.id < b.id;
    });

    std::vector<Board> boards;
    boards.reserve(std::max(1, std::min(N, 4096)));

    for (const Rect& r : order) {
        if (!canFitBoard(r, W, H)) {
            // Invalid input under the stated problem constraints. Emit a harmless
            // placeholder rather than crashing; valid benchmark cases fit.
            ans[r.id] = {0, 0, 0, 0};
            continue;
        }

        Candidate best;
        best.board = -1;

        const int limit = boardSearchLimit(N, static_cast<int>(boards.size()));
        const int start = std::max(0, static_cast<int>(boards.size()) - limit);

        for (int b = start; b < static_cast<int>(boards.size()); ++b) {
            if (boards[b].remainingArea() < r.area) {
                continue;
            }

            Candidate c0;
            c0.board = b;
            if (r.w <= W && r.h <= H && boards[b].findPlacement(r.w, r.h, 0, c0)) {
                c0.board = b;
                if (betterGlobalCandidate(c0, best, boards)) {
                    best = c0;
                }
            }

            if (r.w != r.h && r.h <= W && r.w <= H) {
                Candidate c1;
                c1.board = b;
                if (boards[b].findPlacement(r.h, r.w, 1, c1)) {
                    c1.board = b;
                    if (betterGlobalCandidate(c1, best, boards)) {
                        best = c1;
                    }
                }
            }
        }

        if (best.board < 0) {
            boards.emplace_back(W, H);
            int b = static_cast<int>(boards.size()) - 1;
            Candidate c0;
            c0.board = b;
            bool ok = false;
            if (r.w <= W && r.h <= H && boards[b].findPlacement(r.w, r.h, 0, c0)) {
                c0.board = b;
                best = c0;
                ok = true;
            }
            if (r.w != r.h && r.h <= W && r.w <= H) {
                Candidate c1;
                c1.board = b;
                if (boards[b].findPlacement(r.h, r.w, 1, c1)) {
                    c1.board = b;
                    if (!ok || betterGlobalCandidate(c1, best, boards)) {
                        best = c1;
                    }
                    ok = true;
                }
            }
        }

        boards[best.board].place(best);
        ans[r.id] = {best.board, best.x, best.y, best.rotated};
    }

    std::cout << boards.size() << '\n';
    for (int i = 0; i < N; ++i) {
        std::cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }

    return 0;
}
