use std::cmp::min;
use std::io::{self, BufRead, BufReader, BufWriter, Read, Write};

#[derive(Clone, Copy, Default)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline]
    fn dist2(self, other: Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }

    #[inline]
    fn add(self, other: Point) -> Point {
        Point {
            x: self.x + other.x,
            y: self.y + other.y,
            z: self.z + other.z,
        }
    }

    #[inline]
    fn div(self, v: f32) -> Point {
        Point {
            x: self.x / v,
            y: self.y / v,
            z: self.z / v,
        }
    }
}

#[derive(Clone)]
struct Lcg(u64);

impl Lcg {
    fn new(seed: u64) -> Self {
        Self(seed | 1)
    }

    #[inline]
    fn next_u64(&mut self) -> u64 {
        self.0 = self
            .0
            .wrapping_mul(6364136223846793005)
            .wrapping_add(1442695040888963407);
        self.0
    }

}

#[inline]
fn read_point<R: Read>(r: &mut R, buf: &mut [u8; 12]) -> io::Result<Option<Point>> {
    match r.read_exact(buf) {
        Ok(()) => {
            let x = f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]);
            let y = f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]);
            let z = f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]);
            Ok(Some(Point { x, y, z }))
        }
        Err(e) if e.kind() == io::ErrorKind::UnexpectedEof => Ok(None),
        Err(e) => Err(e),
    }
}

#[inline]
fn write_u32<W: Write>(w: &mut W, v: u32) -> io::Result<()> {
    w.write_all(&v.to_le_bytes())
}

#[inline]
fn nearest(p: Point, centers: &[Point]) -> usize {
    let mut best = 0usize;
    let mut best_d = f32::INFINITY;
    for (i, &c) in centers.iter().enumerate() {
        let d = p.dist2(c);
        if d < best_d {
            best_d = d;
            best = i;
        }
    }
    best
}

fn farthest_first_seed(points: &[Point], k: usize) -> Vec<Point> {
    let n = points.len();
    let mut centers = Vec::with_capacity(k);
    if n == 0 || k == 0 {
        return centers;
    }

    centers.push(points[0]);
    let mut min_d = vec![f32::INFINITY; n];
    while centers.len() < k {
        let last = *centers.last().unwrap();
        let mut far_i = centers.len() % n;
        let mut far_d = -1.0f32;
        for (i, &p) in points.iter().enumerate() {
            let d = p.dist2(last);
            if d < min_d[i] {
                min_d[i] = d;
            }
            if min_d[i] > far_d {
                far_d = min_d[i];
                far_i = i;
            }
        }
        centers.push(points[far_i]);
    }
    centers
}

fn kmeans_in_memory(points: &[Point], k: usize) -> Vec<Point> {
    if k == 0 {
        return Vec::new();
    }
    let mut centers = farthest_first_seed(points, k);
    let iters = if points.len() <= 1_000_000 {
        18
    } else if points.len() <= 10_000_000 {
        10
    } else {
        6
    };

    for _ in 0..iters {
        let mut sums = vec![Point::default(); k];
        let mut counts = vec![0u64; k];
        for &p in points {
            let c = nearest(p, &centers);
            sums[c] = sums[c].add(p);
            counts[c] += 1;
        }

        let mut max_shift = 0.0f32;
        for i in 0..k {
            if counts[i] != 0 {
                let new_c = sums[i].div(counts[i] as f32);
                max_shift = max_shift.max(centers[i].dist2(new_c));
                centers[i] = new_c;
            } else {
                centers[i] = points[(i * 1_000_003 + centers.len()) % points.len()];
            }
        }
        if max_shift < 1.0e-12 {
            break;
        }
    }
    centers
}

fn run_in_memory<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    n: usize,
    k: usize,
) -> io::Result<()> {
    let mut points = Vec::with_capacity(n);
    let mut buf = [0u8; 12];
    for _ in 0..n {
        if let Some(p) = read_point(reader, &mut buf)? {
            points.push(p);
        }
    }

    if k == 0 || points.is_empty() {
        return Ok(());
    }

    let centers = kmeans_in_memory(&points, k);
    for p in points {
        write_u32(writer, nearest(p, &centers) as u32)?;
    }
    Ok(())
}

fn init_online_centers(sample: &[Point], k: usize) -> Vec<Point> {
    if sample.is_empty() || k == 0 {
        return Vec::new();
    }
    let mut centers = farthest_first_seed(sample, min(k, sample.len()));
    while centers.len() < k {
        centers.push(sample[centers.len() % sample.len()]);
    }

    let rounds = if sample.len() <= 200_000 { 8 } else { 5 };
    for _ in 0..rounds {
        let mut sums = vec![Point::default(); k];
        let mut counts = vec![0u32; k];
        for &p in sample {
            let c = nearest(p, &centers);
            sums[c] = sums[c].add(p);
            counts[c] += 1;
        }
        for i in 0..k {
            if counts[i] > 0 {
                centers[i] = sums[i].div(counts[i] as f32);
            } else {
                centers[i] = sample[(i * 2654435761usize) % sample.len()];
            }
        }
    }
    centers
}

fn assign_and_update(p: Point, centers: &mut [Point], seen: &mut [u32]) -> usize {
    let c = nearest(p, centers);
    let s = seen[c].saturating_add(1);
    seen[c] = s;
    let eta = if s <= 64 {
        1.0 / s as f32
    } else {
        0.05 / (1.0 + (s as f32 / 20000.0).sqrt())
    };
    centers[c].x += eta * (p.x - centers[c].x);
    centers[c].y += eta * (p.y - centers[c].y);
    centers[c].z += eta * (p.z - centers[c].z);
    c
}

fn run_streaming<R: Read, W: Write>(
    reader: &mut R,
    writer: &mut W,
    n: u64,
    k: usize,
) -> io::Result<()> {
    if k == 0 {
        return Ok(());
    }

    let sample_cap = if n <= 100_000_000 {
        min(1_000_000usize, k.saturating_mul(200).max(20_000))
    } else if n <= 1_000_000_000 {
        min(400_000usize, k.saturating_mul(80).max(20_000))
    } else {
        min(200_000usize, k.saturating_mul(30).max(20_000))
    };

    let mut sample = Vec::with_capacity(sample_cap);
    let mut rng = Lcg::new(n ^ ((k as u64) << 32) ^ 0x9e3779b97f4a7c15);
    let mut buf = [0u8; 12];
    let mut read = 0u64;

    while read < n && sample.len() < sample_cap {
        if let Some(p) = read_point(reader, &mut buf)? {
            sample.push(p);
            read += 1;
        } else {
            break;
        }
    }

    let mut centers = init_online_centers(&sample, k);
    let mut seen = vec![1u32; k];

    for &p in &sample {
        let c = assign_and_update(p, &mut centers, &mut seen);
        write_u32(writer, c as u32)?;
    }

    while read < n {
        match read_point(reader, &mut buf)? {
            Some(p) => {
                if read < 10_000_000 || (rng.next_u64() & 15) == 0 {
                    let c = assign_and_update(p, &mut centers, &mut seen);
                    write_u32(writer, c as u32)?;
                } else {
                    write_u32(writer, nearest(p, &centers) as u32)?;
                }
                read += 1;
            }
            None => break,
        }
    }
    Ok(())
}

fn parse_header(line: &[u8]) -> io::Result<(u64, usize)> {
    let s = std::str::from_utf8(line)
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "header is not UTF-8"))?;
    let mut it = s.split_whitespace();
    let n = it
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing point count"))?
        .parse::<u64>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "bad point count"))?;
    let k = it
        .next()
        .ok_or_else(|| io::Error::new(io::ErrorKind::InvalidData, "missing cluster count"))?
        .parse::<usize>()
        .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "bad cluster count"))?;
    if k > u32::MAX as usize {
        return Err(io::Error::new(
            io::ErrorKind::InvalidData,
            "cluster count exceeds u32 output range",
        ));
    }
    Ok((n, k))
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
    let mut writer = BufWriter::with_capacity(1 << 20, stdout.lock());

    let mut header = Vec::with_capacity(64);
    reader.read_until(b'\n', &mut header)?;
    let (n, k) = parse_header(&header)?;

    if n == 0 || k == 0 {
        writer.flush()?;
        return Ok(());
    }

    let small_enough = n <= 10_000_000 && (n as usize).saturating_mul(12) <= 512 * 1024 * 1024;
    if small_enough {
        run_in_memory(&mut reader, &mut writer, n as usize, k)?;
    } else {
        run_streaming(&mut reader, &mut writer, n, k)?;
    }

    writer.flush()
}
