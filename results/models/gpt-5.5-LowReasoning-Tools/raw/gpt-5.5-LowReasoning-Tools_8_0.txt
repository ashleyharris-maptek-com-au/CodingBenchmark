use std::collections::VecDeque;
use std::io::{self, BufRead, BufWriter, Write};

const UNSEEN: u8 = 255;
const START: u8 = 254;
const WALL: u8 = 253;

fn main() {
    let stdin = io::stdin();
    let mut input = io::BufReader::new(stdin.lock());

    let mut first = String::new();
    if input.read_line(&mut first).unwrap_or(0) == 0 {
        return;
    }
    let mut it = first.split_whitespace();
    let h: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let w: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    if h == 0 || w == 0 {
        println!("0");
        return;
    }

    let cells = match h.checked_mul(w) {
        Some(v) if v <= u32::MAX as usize => v,
        _ => {
            println!("0");
            return;
        }
    };

    let mut state = vec![UNSEEN; cells];
    let mut start: Option<u32> = None;
    let mut goal: Option<u32> = None;
    let mut row = Vec::new();

    for y in 0..h {
        row.clear();
        if input.read_until(b'\n', &mut row).unwrap_or(0) == 0 {
            break;
        }
        while matches!(row.last(), Some(b'\n' | b'\r')) {
            row.pop();
        }

        let base = y * w;
        for x in 0..w {
            let ch = row.get(x).copied().unwrap_or(b'#');
            let idx = base + x;
            match ch {
                b'#' => state[idx] = WALL,
                b'A' => {
                    state[idx] = START;
                    start = Some(idx as u32);
                }
                b'B' => {
                    goal = Some(idx as u32);
                }
                _ => {}
            }
        }
    }

    let (start, goal) = match (start, goal) {
        (Some(s), Some(g)) => (s, g),
        _ => {
            println!("0");
            return;
        }
    };

    if start == goal {
        println!("1\n{} {}", start as usize % w, start as usize / w);
        return;
    }

    let mut q = VecDeque::new();
    q.push_back(start);

    let mut found = false;
    while let Some(cur) = q.pop_front() {
        let i = cur as usize;
        let x = i % w;

        if i >= w {
            let n = i - w;
            if state[n] == UNSEEN {
                state[n] = 1; // parent is one row down
                if n as u32 == goal {
                    found = true;
                    break;
                }
                q.push_back(n as u32);
            }
        }
        if i + w < cells {
            let n = i + w;
            if state[n] == UNSEEN {
                state[n] = 0; // parent is one row up
                if n as u32 == goal {
                    found = true;
                    break;
                }
                q.push_back(n as u32);
            }
        }
        if x > 0 {
            let n = i - 1;
            if state[n] == UNSEEN {
                state[n] = 3; // parent is one column right
                if n as u32 == goal {
                    found = true;
                    break;
                }
                q.push_back(n as u32);
            }
        }
        if x + 1 < w {
            let n = i + 1;
            if state[n] == UNSEEN {
                state[n] = 2; // parent is one column left
                if n as u32 == goal {
                    found = true;
                    break;
                }
                q.push_back(n as u32);
            }
        }
    }
    drop(q);

    if !found {
        println!("0");
        return;
    }

    let mut path: Vec<u32> = Vec::new();
    let mut cur = goal as usize;
    loop {
        path.push(cur as u32);
        if cur as u32 == start {
            break;
        }
        cur = match state[cur] {
            0 => cur - w,
            1 => cur + w,
            2 => cur - 1,
            3 => cur + 1,
            _ => {
                println!("0");
                return;
            }
        };
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    writeln!(out, "{}", path.len()).unwrap();
    for &idx in path.iter().rev() {
        let i = idx as usize;
        writeln!(out, "{} {}", i % w, i / w).unwrap();
    }
}
