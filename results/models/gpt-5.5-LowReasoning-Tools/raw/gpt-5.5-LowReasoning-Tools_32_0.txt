#include <algorithm>
#include <cstdint>
#include <functional>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using namespace std;

struct Edge {
    int to;
    long long w;
    int id;
};

struct InputEdge {
    int u;
    int v;
    long long w;
};

struct DSU {
    vector<int> p, r;
    explicit DSU(int n = 0) : p(n), r(n, 0) {
        iota(p.begin(), p.end(), 0);
    }
    int find(int x) {
        while (p[x] != x) {
            p[x] = p[p[x]];
            x = p[x];
        }
        return x;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

static const long long INF = numeric_limits<long long>::max() / 4;

int single_bit_index(int mask) {
    int idx = 0;
    while ((mask & 1) == 0) {
        mask >>= 1;
        ++idx;
    }
    return idx;
}

void add_path_edges(int s, int t, const vector<int>& parent_edge,
                    const vector<InputEdge>& edges, vector<char>& chosen) {
    int cur = t;
    while (cur != s && cur >= 0) {
        int eid = parent_edge[cur];
        if (eid < 0) break;
        chosen[eid] = 1;
        cur = edges[eid].u ^ edges[eid].v ^ cur;
    }
}

vector<char> prune_to_terminal_tree(int n, const vector<InputEdge>& edges,
                                    const vector<char>& chosen,
                                    const vector<int>& terminals) {
    vector<vector<pair<int, int>>> tree(n);
    int cnt = 0;
    for (int i = 0; i < (int)edges.size(); ++i) {
        if (!chosen[i]) continue;
        ++cnt;
        tree[edges[i].u].push_back({edges[i].v, i});
        tree[edges[i].v].push_back({edges[i].u, i});
    }

    vector<char> is_terminal(n, 0), alive_edge = chosen, alive_vertex(n, 0);
    vector<int> degree(n, 0);
    for (int t : terminals) is_terminal[t] = 1;
    for (int v = 0; v < n; ++v) {
        degree[v] = (int)tree[v].size();
        alive_vertex[v] = degree[v] > 0;
    }

    queue<int> q;
    for (int v = 0; v < n; ++v) {
        if (!is_terminal[v] && degree[v] <= 1) q.push(v);
    }

    while (!q.empty()) {
        int v = q.front();
        q.pop();
        if (is_terminal[v] || degree[v] != 1) continue;
        alive_vertex[v] = 0;
        for (auto [to, eid] : tree[v]) {
            if (!alive_edge[eid]) continue;
            alive_edge[eid] = 0;
            --degree[v];
            --degree[to];
            if (!is_terminal[to] && degree[to] == 1) q.push(to);
            break;
        }
    }
    (void)cnt;
    return alive_edge;
}

void output_tree(int n, const vector<InputEdge>& edges, vector<char> chosen,
                 const vector<int>& terminals) {
    struct Item {
        long long w;
        int id;
    };
    vector<Item> items;
    for (int i = 0; i < (int)edges.size(); ++i) {
        if (chosen[i]) items.push_back({edges[i].w, i});
    }
    sort(items.begin(), items.end(), [](const Item& a, const Item& b) {
        if (a.w != b.w) return a.w < b.w;
        return a.id < b.id;
    });

    DSU dsu(n);
    vector<char> acyclic(edges.size(), 0);
    for (const auto& item : items) {
        int id = item.id;
        if (dsu.unite(edges[id].u, edges[id].v)) acyclic[id] = 1;
    }
    acyclic = prune_to_terminal_tree(n, edges, acyclic, terminals);

    long long total = 0;
    vector<int> out;
    for (int i = 0; i < (int)edges.size(); ++i) {
        if (acyclic[i]) {
            total += edges[i].w;
            out.push_back(i);
        }
    }

    cout << total << ' ' << out.size() << '\n';
    for (int id : out) {
        cout << edges[id].u << ' ' << edges[id].v << '\n';
    }
}

void solve_heuristic(int n, const vector<vector<Edge>>& graph,
                     const vector<InputEdge>& edges,
                     const vector<int>& terminals) {
    int k = (int)terminals.size();
    vector<vector<long long>> dist(k, vector<long long>(n, INF));
    vector<vector<int>> parent_edge(k, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) {
        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        dist[i][terminals[i]] = 0;
        pq.push({0, terminals[i]});
        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dist[i][v]) continue;
            for (const Edge& e : graph[v]) {
                long long nd = d + e.w;
                if (nd < dist[i][e.to]) {
                    dist[i][e.to] = nd;
                    parent_edge[i][e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    vector<tuple<long long, int, int>> metric_edges;
    for (int i = 0; i < k; ++i) {
        for (int j = i + 1; j < k; ++j) {
            if (dist[i][terminals[j]] < INF) {
                metric_edges.push_back({dist[i][terminals[j]], i, j});
            }
        }
    }
    sort(metric_edges.begin(), metric_edges.end());

    DSU dsu(k);
    vector<char> chosen(edges.size(), 0);
    int used = 0;
    for (auto [w, a, b] : metric_edges) {
        (void)w;
        if (!dsu.unite(a, b)) continue;
        add_path_edges(terminals[a], terminals[b], parent_edge[a], edges, chosen);
        if (++used == k - 1) break;
    }
    output_tree(n, edges, chosen, terminals);
}

void solve_exact(int n, const vector<vector<Edge>>& graph,
                 const vector<InputEdge>& edges,
                 const vector<int>& terminals) {
    int k = (int)terminals.size();
    int full = 1 << k;
    vector<vector<long long>> dp(full, vector<long long>(n, INF));
    vector<vector<int>> split(full, vector<int>(n, 0));
    vector<vector<int>> parent_edge(full, vector<int>(n, -1));

    for (int i = 0; i < k; ++i) dp[1 << i][terminals[i]] = 0;

    for (int mask = 1; mask < full; ++mask) {
        for (int sub = (mask - 1) & mask; sub; sub = (sub - 1) & mask) {
            int other = mask ^ sub;
            if (sub > other) continue;
            for (int v = 0; v < n; ++v) {
                if (dp[sub][v] == INF || dp[other][v] == INF) continue;
                long long nd = dp[sub][v] + dp[other][v];
                if (nd < dp[mask][v]) {
                    dp[mask][v] = nd;
                    split[mask][v] = sub;
                    parent_edge[mask][v] = -1;
                }
            }
        }

        priority_queue<pair<long long, int>, vector<pair<long long, int>>, greater<pair<long long, int>>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] < INF) pq.push({dp[mask][v], v});
        }
        while (!pq.empty()) {
            auto [d, v] = pq.top();
            pq.pop();
            if (d != dp[mask][v]) continue;
            for (const Edge& e : graph[v]) {
                long long nd = d + e.w;
                if (nd < dp[mask][e.to]) {
                    dp[mask][e.to] = nd;
                    split[mask][e.to] = 0;
                    parent_edge[mask][e.to] = e.id;
                    pq.push({nd, e.to});
                }
            }
        }
    }

    int best_root = 0;
    for (int v = 1; v < n; ++v) {
        if (dp[full - 1][v] < dp[full - 1][best_root]) best_root = v;
    }

    vector<char> chosen(edges.size(), 0);
    vector<pair<int, int>> stack;
    stack.push_back({full - 1, best_root});
    while (!stack.empty()) {
        auto [mask, v] = stack.back();
        stack.pop_back();
        if ((mask & (mask - 1)) == 0 && v == terminals[single_bit_index(mask)]) continue;

        if (split[mask][v]) {
            int sub = split[mask][v];
            stack.push_back({sub, v});
            stack.push_back({mask ^ sub, v});
        } else {
            int eid = parent_edge[mask][v];
            if (eid < 0) continue;
            chosen[eid] = 1;
            int prev = edges[eid].u ^ edges[eid].v ^ v;
            stack.push_back({mask, prev});
        }
    }

    output_tree(n, edges, chosen, terminals);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;

    vector<InputEdge> edges;
    edges.reserve(m);
    vector<vector<Edge>> graph(n);
    for (int i = 0; i < m; ++i) {
        int u, v;
        long long w;
        cin >> u >> v >> w;
        edges.push_back({u, v, w});
        graph[u].push_back({v, w, i});
        graph[v].push_back({u, w, i});
    }

    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) cin >> terminals[i];

    if (k == 0) {
        cout << "0 0\n";
        return 0;
    }
    if (k == 1) {
        cout << "0 0\n";
        return 0;
    }

    const long long max_exact_states = 25000000LL;
    if (k <= 20 && (1LL << k) * n <= max_exact_states) {
        solve_exact(n, graph, edges, terminals);
    } else {
        solve_heuristic(n, graph, edges, terminals);
    }

    return 0;
}
