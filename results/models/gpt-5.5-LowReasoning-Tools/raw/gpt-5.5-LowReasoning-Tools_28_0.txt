use std::collections::VecDeque;
use std::io::{self, Read, Write};

struct Scanner {
    data: Vec<u8>,
    pos: usize,
}

impl Scanner {
    fn new() -> Self {
        let mut data = Vec::new();
        io::stdin().read_to_end(&mut data).unwrap();
        Self { data, pos: 0 }
    }

    fn next_usize(&mut self) -> usize {
        while self.pos < self.data.len() && self.data[self.pos].is_ascii_whitespace() {
            self.pos += 1;
        }
        let mut value = 0usize;
        while self.pos < self.data.len() && !self.data[self.pos].is_ascii_whitespace() {
            value = value * 10 + usize::from(self.data[self.pos] - b'0');
            self.pos += 1;
        }
        value
    }
}

fn build_csr(n: usize, edges: &[(usize, usize)]) -> Vec<Vec<usize>> {
    let mut adj = vec![Vec::<usize>::new(); n];
    for &(u, v) in edges {
        if u == v {
            adj[u].push(v);
        } else {
            adj[u].push(v);
            adj[v].push(u);
        }
    }
    adj
}

fn try_bipartite_cover(n: usize, adj: &[Vec<usize>]) -> Option<Vec<bool>> {
    let mut color = vec![2u8; n];
    let mut q = VecDeque::new();
    for start in 0..n {
        if color[start] != 2 {
            continue;
        }
        color[start] = 0;
        q.push_back(start);
        while let Some(u) = q.pop_front() {
            for &v in &adj[u] {
                if color[v] == 2 {
                    color[v] = color[u] ^ 1;
                    q.push_back(v);
                } else if color[v] == color[u] {
                    return None;
                }
            }
        }
    }

    let mut left_id = vec![usize::MAX; n];
    let mut right_id = vec![usize::MAX; n];
    let mut left = Vec::new();
    let mut right = Vec::new();
    for v in 0..n {
        if color[v] == 0 {
            left_id[v] = left.len();
            left.push(v);
        } else {
            right_id[v] = right.len();
            right.push(v);
        }
    }

    let l_len = left.len();
    let r_len = right.len();
    let mut ladj = vec![Vec::<usize>::new(); l_len];
    for (i, &u) in left.iter().enumerate() {
        for &v in &adj[u] {
            let rid = right_id[v];
            if rid != usize::MAX {
                ladj[i].push(rid);
            }
        }
    }

    let mut pair_l = vec![usize::MAX; l_len];
    let mut pair_r = vec![usize::MAX; r_len];
    let mut dist = vec![0usize; l_len];

    loop {
        let mut queue = VecDeque::new();
        for i in 0..l_len {
            if pair_l[i] == usize::MAX {
                dist[i] = 0;
                queue.push_back(i);
            } else {
                dist[i] = usize::MAX;
            }
        }

        while let Some(i) = queue.pop_front() {
            for &j in &ladj[i] {
                let p = pair_r[j];
                if p != usize::MAX && dist[p] == usize::MAX {
                    dist[p] = dist[i] + 1;
                    queue.push_back(p);
                }
            }
        }

        let mut advanced = false;
        for i in 0..l_len {
            if pair_l[i] == usize::MAX && dfs_hk(i, &ladj, &mut pair_l, &mut pair_r, &mut dist) {
                advanced = true;
            }
        }
        if !advanced {
            break;
        }
    }

    let mut seen_l = vec![false; l_len];
    let mut seen_r = vec![false; r_len];
    let mut stack = Vec::new();
    for i in 0..l_len {
        if pair_l[i] == usize::MAX {
            seen_l[i] = true;
            stack.push(i);
        }
    }
    while let Some(i) = stack.pop() {
        for &j in &ladj[i] {
            if pair_l[i] != j && !seen_r[j] {
                seen_r[j] = true;
                let p = pair_r[j];
                if p != usize::MAX && !seen_l[p] {
                    seen_l[p] = true;
                    stack.push(p);
                }
            }
        }
    }

    let mut cover = vec![false; n];
    for i in 0..l_len {
        if !seen_l[i] {
            cover[left[i]] = true;
        }
    }
    for j in 0..r_len {
        if seen_r[j] {
            cover[right[j]] = true;
        }
    }
    Some(cover)
}

fn dfs_hk(
    i: usize,
    ladj: &[Vec<usize>],
    pair_l: &mut [usize],
    pair_r: &mut [usize],
    dist: &mut [usize],
) -> bool {
    for &j in &ladj[i] {
        let p = pair_r[j];
        if p == usize::MAX || (dist[p] == dist[i] + 1 && dfs_hk(p, ladj, pair_l, pair_r, dist)) {
            pair_l[i] = j;
            pair_r[j] = i;
            return true;
        }
    }
    dist[i] = usize::MAX;
    false
}

fn greedy_cover(n: usize, adj: &[Vec<usize>], edges: &[(usize, usize)]) -> Vec<bool> {
    let mut covered = vec![false; edges.len()];
    let mut cover = vec![false; n];
    let mut remaining_degree: Vec<usize> = adj.iter().map(Vec::len).collect();

    for (idx, &(u, v)) in edges.iter().enumerate() {
        if covered[idx] {
            continue;
        }
        let choose = if remaining_degree[u] >= remaining_degree[v] { u } else { v };
        if !cover[choose] {
            cover[choose] = true;
            for &x in &adj[choose] {
                remaining_degree[x] = remaining_degree[x].saturating_sub(1);
            }
        }
        covered[idx] = true;
    }

    for &(u, v) in edges {
        if !cover[u] && !cover[v] {
            if adj[u].len() >= adj[v].len() {
                cover[u] = true;
            } else {
                cover[v] = true;
            }
        }
    }
    cover
}

fn maximal_matching_cover(n: usize, edges: &[(usize, usize)]) -> Vec<bool> {
    let mut matched = vec![false; n];
    let mut cover = vec![false; n];
    for &(u, v) in edges {
        if !matched[u] && !matched[v] {
            matched[u] = true;
            matched[v] = true;
            cover[u] = true;
            cover[v] = true;
        }
    }
    cover
}

fn prune_cover(cover: &mut [bool], adj: &[Vec<usize>]) {
    let n = cover.len();
    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by_key(|&v| adj[v].len());
    for &v in &order {
        if !cover[v] {
            continue;
        }
        let mut removable = true;
        for &u in &adj[v] {
            if !cover[u] {
                removable = false;
                break;
            }
        }
        if removable {
            cover[v] = false;
        }
    }
}

fn ensure_valid(cover: &mut [bool], adj: &[Vec<usize>], edges: &[(usize, usize)]) {
    for &(u, v) in edges {
        if !cover[u] && !cover[v] {
            if adj[u].len() >= adj[v].len() {
                cover[u] = true;
            } else {
                cover[v] = true;
            }
        }
    }
}

fn exact_small(n: usize, edges: &[(usize, usize)]) -> Option<Vec<bool>> {
    if n > 34 || edges.len() > 120 {
        return None;
    }
    let mut masks = Vec::new();
    for &(u, v) in edges {
        if u == v {
            masks.push(1u64 << u);
        } else {
            masks.push((1u64 << u) | (1u64 << v));
        }
    }
    let mut best = if n == 64 { u64::MAX } else { (1u64 << n) - 1 };
    branch(0, &masks, 0, &mut best);
    let mut cover = vec![false; n];
    for (i, slot) in cover.iter_mut().enumerate() {
        *slot = (best & (1u64 << i)) != 0;
    }
    Some(cover)
}

fn branch(chosen: u64, masks: &[u64], start: usize, best: &mut u64) {
    if chosen.count_ones() >= best.count_ones() {
        return;
    }
    let mut edge_pos = usize::MAX;
    for (i, &mask) in masks.iter().enumerate().skip(start) {
        if (chosen & mask) == 0 {
            edge_pos = i;
            break;
        }
    }
    if edge_pos == usize::MAX {
        *best = chosen;
        return;
    }
    let mask = masks[edge_pos];
    let a = mask.trailing_zeros();
    branch(chosen | (1u64 << a), masks, edge_pos + 1, best);
    let bmask = mask & !(1u64 << a);
    if bmask != 0 {
        let b = bmask.trailing_zeros();
        branch(chosen | (1u64 << b), masks, edge_pos + 1, best);
    }
}

fn main() {
    let mut scanner = Scanner::new();
    if scanner.data.is_empty() {
        return;
    }
    let n = scanner.next_usize();
    let m = scanner.next_usize();
    let mut edges = Vec::with_capacity(m);
    for _ in 0..m {
        let u = scanner.next_usize();
        let v = scanner.next_usize();
        if u < n && v < n {
            edges.push((u, v));
        }
    }

    let cover = if let Some(exact) = exact_small(n, &edges) {
        exact
    } else {
        let adj = build_csr(n, &edges);
        let mut candidate = if let Some(bip) = try_bipartite_cover(n, &adj) {
            bip
        } else {
            let mut greedy = greedy_cover(n, &adj, &edges);
            let mut matching = maximal_matching_cover(n, &edges);
            prune_cover(&mut greedy, &adj);
            prune_cover(&mut matching, &adj);
            ensure_valid(&mut greedy, &adj, &edges);
            ensure_valid(&mut matching, &adj, &edges);
            if matching.iter().filter(|&&x| x).count() < greedy.iter().filter(|&&x| x).count() {
                matching
            } else {
                greedy
            }
        };
        ensure_valid(&mut candidate, &adj, &edges);
        candidate
    };

    let vertices: Vec<usize> = cover
        .iter()
        .enumerate()
        .filter_map(|(i, &selected)| if selected { Some(i) } else { None })
        .collect();
    let mut out = String::new();
    out.push_str(&vertices.len().to_string());
    out.push('\n');
    for (i, v) in vertices.iter().enumerate() {
        if i > 0 {
            out.push(' ');
        }
        out.push_str(&v.to_string());
    }
    out.push('\n');
    io::stdout().write_all(out.as_bytes()).unwrap();
}
