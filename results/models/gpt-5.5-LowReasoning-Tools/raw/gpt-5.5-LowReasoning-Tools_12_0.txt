using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    const double Eps = 1e-9;

    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Ori
    {
        public double Dx, Dy, Dz, Ox, Oy, Oz;
        public double Qw, Qx, Qy, Qz;
        public Ori(double qw, double qx, double qy, double qz)
        {
            Dx = Dy = Dz = Ox = Oy = Oz = 0;
            Qw = qw; Qx = qx; Qy = qy; Qz = qz;
        }
    }

    struct Box
    {
        public double X, Y, Z, Qw, Qx, Qy, Qz;
        public Box(double x, double y, double z, double qw, double qx, double qy, double qz)
        {
            X = x; Y = y; Z = z; Qw = qw; Qx = qx; Qy = qy; Qz = qz;
        }
    }

    sealed class FastScanner
    {
        readonly Stream input = Console.OpenStandardInput();
        readonly byte[] buffer = new byte[1 << 16];
        int len, ptr;

        int Read()
        {
            if (ptr >= len)
            {
                len = input.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buffer[ptr++];
        }

        public int NextInt() => (int)NextDouble();

        public double NextDouble()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            double val = 0;
            while (c >= '0' && c <= '9')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            if (c == '.')
            {
                double mul = 0.1;
                c = Read();
                while (c >= '0' && c <= '9')
                {
                    val += (c - '0') * mul;
                    mul *= 0.1;
                    c = Read();
                }
            }
            if (c == 'e' || c == 'E')
            {
                c = Read();
                int esign = 1;
                if (c == '-') { esign = -1; c = Read(); }
                else if (c == '+') c = Read();
                int exp = 0;
                while (c >= '0' && c <= '9')
                {
                    exp = exp * 10 + (c - '0');
                    c = Read();
                }
                val *= Math.Pow(10, esign * exp);
            }
            return sign * val;
        }
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var fs = new FastScanner();
        int v = fs.NextInt();
        int f = fs.NextInt();

        var pts = new Vec[v];
        double minX = double.PositiveInfinity, minY = double.PositiveInfinity, minZ = double.PositiveInfinity;
        double maxX = double.NegativeInfinity, maxY = double.NegativeInfinity, maxZ = double.NegativeInfinity;
        for (int i = 0; i < v; i++)
        {
            double x = fs.NextDouble(), y = fs.NextDouble(), z = fs.NextDouble();
            pts[i] = new Vec(x, y, z);
            if (x < minX) minX = x; if (x > maxX) maxX = x;
            if (y < minY) minY = y; if (y > maxY) maxY = y;
            if (z < minZ) minZ = z; if (z > maxZ) maxZ = z;
        }

        for (int i = 0; i < f; i++)
        {
            fs.NextInt(); fs.NextInt(); fs.NextInt();
        }

        double W = fs.NextDouble(), H = fs.NextDouble(), D = fs.NextDouble();
        double sx = maxX - minX, sy = maxY - minY, sz = maxZ - minZ;
        if (v == 0 || sx <= Eps || sy <= Eps || sz <= Eps || W <= Eps || H <= Eps || D <= Eps)
        {
            Console.WriteLine(0);
            return;
        }

        Ori[] all = Orientations(pts);
        long outputCap = ChooseOutputCap(W, H, D, v, f);
        var placements = new List<Box>((int)Math.Min(outputCap, 1_000_000));

        long bestCount = -1;
        Ori bestOri = all[0];
        double bestDx = sx, bestDy = sy, bestDz = sz;
        foreach (var o in all)
        {
            double dx = o.Dx, dy = o.Dy, dz = o.Dz;
            long nx = Fit(W, dx), ny = Fit(H, dy), nz = Fit(D, dz);
            long cnt = SaturatingMul(SaturatingMul(nx, ny), nz);
            if (cnt > bestCount || (cnt == bestCount && dx * dy * dz < bestDx * bestDy * bestDz))
            {
                bestCount = cnt;
                bestOri = o;
                bestDx = dx; bestDy = dy; bestDz = dz;
            }
        }

        if (bestCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        bool tiny = W * H * D <= 8000 && v <= 64;
        bool medium = Math.Max(W, Math.Max(H, D)) <= 30 && v <= 256;
        bool huge = Math.Max(W, Math.Max(H, D)) >= 150 || bestCount > 2_000_000;

        if (tiny || medium)
        {
            GenerateLayered(placements, all, W, H, D, outputCap, tryAllOrientations: tiny);
        }
        else if (huge)
        {
            GenerateGrid(placements, bestOri, W, H, D, outputCap);
        }
        else
        {
            GenerateLayered(placements, new[] { bestOri }, W, H, D, outputCap, tryAllOrientations: false);
        }

        if (placements.Count == 0)
            GenerateGrid(placements, bestOri, W, H, D, outputCap);

        var sb = new StringBuilder(Math.Min(placements.Count, 100000) * 48 + 32);
        sb.AppendLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        foreach (var p in placements)
        {
            sb.Append(Format(p.X)).Append(' ')
              .Append(Format(p.Y)).Append(' ')
              .Append(Format(p.Z)).Append(' ')
              .Append(Format(p.Qw)).Append(' ')
              .Append(Format(p.Qx)).Append(' ')
              .Append(Format(p.Qy)).Append(' ')
              .Append(Format(p.Qz)).AppendLine();
        }
        Console.Write(sb.ToString());
    }

    static long ChooseOutputCap(double w, double h, double d, int v, int f)
    {
        double maxSide = Math.Max(w, Math.Max(h, d));
        if (maxSide >= 200) return 1_000_000;
        if (maxSide >= 80) return 1_500_000;
        return 2_000_000;
    }

    static long Fit(double outer, double inner)
    {
        if (inner <= Eps) return 0;
        return Math.Max(0, (long)Math.Floor((outer + Eps) / inner));
    }

    static long SaturatingMul(long a, long b)
    {
        if (a == 0 || b == 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    static void GenerateGrid(List<Box> res, Ori o, double W, double H, double D, long cap)
    {
        double dx = o.Dx, dy = o.Dy, dz = o.Dz;
        long nx = Fit(W, dx), ny = Fit(H, dy), nz = Fit(D, dz);
        double ox = o.Ox, oy = o.Oy, oz = o.Oz;
        for (long z = 0; z < nz && res.Count < cap; z++)
            for (long y = 0; y < ny && res.Count < cap; y++)
                for (long x = 0; x < nx && res.Count < cap; x++)
                    res.Add(new Box(x * dx + ox, y * dy + oy, z * dz + oz, o.Qw, o.Qx, o.Qy, o.Qz));
    }

    static void GenerateLayered(List<Box> res, Ori[] orientations, double W, double H, double D, long cap, bool tryAllOrientations)
    {
        double z = 0;
        while (z < D - Eps && res.Count < cap)
        {
            Ori best = orientations[0];
            long bestLayer = -1;
            double bestDx = 0, bestDy = 0, bestDz = 0;
            foreach (var o in orientations)
            {
                double dx = o.Dx, dy = o.Dy, dz = o.Dz;
                if (z + dz > D + Eps) continue;
                long cnt = SaturatingMul(Fit(W, dx), Fit(H, dy));
                if (cnt > bestLayer)
                {
                    bestLayer = cnt; best = o; bestDx = dx; bestDy = dy; bestDz = dz;
                }
            }
            if (bestLayer <= 0) break;

            long nx = Fit(W, bestDx), ny = Fit(H, bestDy);
            double ox = best.Ox, oy = best.Oy, oz = best.Oz;
            for (long y = 0; y < ny && res.Count < cap; y++)
                for (long x = 0; x < nx && res.Count < cap; x++)
                    res.Add(new Box(x * bestDx + ox, y * bestDy + oy, z + oz, best.Qw, best.Qx, best.Qy, best.Qz));
            z += bestDz;
            if (!tryAllOrientations && orientations.Length == 1 && z + bestDz > D + Eps) break;
        }
    }

    static Ori[] Orientations(Vec[] pts)
    {
        double s = Math.Sqrt(0.5);
        var list = new[]
        {
            new Ori(1,0,0,0),
            new Ori(s,0,0,s),
            new Ori(s,s,0,0),
            new Ori(s,0,s,0),
            new Ori(0.5,0.5,0.5,0.5),
            new Ori(0.5,0.5,0.5,-0.5)
        };
        for (int i = 0; i < list.Length; i++)
        {
            double mnx = double.PositiveInfinity, mny = double.PositiveInfinity, mnz = double.PositiveInfinity;
            double mxx = double.NegativeInfinity, mxy = double.NegativeInfinity, mxz = double.NegativeInfinity;
            foreach (var p in pts)
            {
                Vec r = Rotate(p, list[i]);
                if (r.X < mnx) mnx = r.X; if (r.X > mxx) mxx = r.X;
                if (r.Y < mny) mny = r.Y; if (r.Y > mxy) mxy = r.Y;
                if (r.Z < mnz) mnz = r.Z; if (r.Z > mxz) mxz = r.Z;
            }
            list[i].Dx = mxx - mnx;
            list[i].Dy = mxy - mny;
            list[i].Dz = mxz - mnz;
            list[i].Ox = -mnx;
            list[i].Oy = -mny;
            list[i].Oz = -mnz;
        }
        return list;
    }

    static Vec Rotate(Vec p, Ori q)
    {
        double tx = 2 * (q.Qy * p.Z - q.Qz * p.Y);
        double ty = 2 * (q.Qz * p.X - q.Qx * p.Z);
        double tz = 2 * (q.Qx * p.Y - q.Qy * p.X);
        return new Vec(
            p.X + q.Qw * tx + (q.Qy * tz - q.Qz * ty),
            p.Y + q.Qw * ty + (q.Qz * tx - q.Qx * tz),
            p.Z + q.Qw * tz + (q.Qx * ty - q.Qy * tx));
    }

    static string Format(double x)
    {
        if (Math.Abs(x) < 0.0000000005) x = 0;
        return x.ToString("0.#########", CultureInfo.InvariantCulture);
    }
}
