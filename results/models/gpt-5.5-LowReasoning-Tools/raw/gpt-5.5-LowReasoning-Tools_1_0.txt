struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    unsigned char buf[BUFSIZE];
    size_t pos = 0, len = 0;

    int getch() {
        if (pos >= len) {
            len = std::fread(buf, 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return EOF;
        }
        return buf[pos++];
    }

    bool readSize(size_t& out) {
        int c;
        do {
            c = getch();
            if (c == EOF) return false;
        } while (c <= ' ');
        size_t v = 0;
        while (c > ' ') {
            v = v * 10 + static_cast<size_t>(c - '0');
            c = getch();
        }
        out = v;
        return true;
    }

    bool readDouble(double& out) {
        int c;
        do {
            c = getch();
            if (c == EOF) return false;
        } while (c <= ' ');

        double sign = 1.0;
        if (c == '-') {
            sign = -1.0;
            c = getch();
        } else if (c == '+') {
            c = getch();
        }

        double v = 0.0;
        while (c >= '0' && c <= '9') {
            v = v * 10.0 + static_cast<double>(c - '0');
            c = getch();
        }
        if (c == '.') {
            double place = 0.1;
            c = getch();
            while (c >= '0' && c <= '9') {
                v += static_cast<double>(c - '0') * place;
                place *= 0.1;
                c = getch();
            }
        }
        if (c == 'e' || c == 'E') {
            int esign = 1;
            int expv = 0;
            c = getch();
            if (c == '-') {
                esign = -1;
                c = getch();
            } else if (c == '+') {
                c = getch();
            }
            while (c >= '0' && c <= '9') {
                expv = expv * 10 + (c - '0');
                c = getch();
            }
            v *= std::pow(10.0, esign * expv);
        }
        out = sign * v;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t pos = 0;

    ~FastOutput() { flush(); }

    void flush() {
        if (pos) {
            std::fwrite(buf, 1, pos, stdout);
            pos = 0;
        }
    }

    void putChar(char c) {
        if (pos == BUFSIZE) flush();
        buf[pos++] = c;
    }

    void writeSize(size_t v) {
        char tmp[32];
        int n = 0;
        do {
            tmp[n++] = static_cast<char>('0' + (v % 10));
            v /= 10;
        } while (v);
        while (n--) putChar(tmp[n]);
    }
};

struct City {
    float x;
    float y;
    uint32_t id;
    uint64_t key;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline uint64_t splitBy1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFull;
    v = (v | (v << 8)) & 0x00FF00FF00FF00FFull;
    v = (v | (v << 4)) & 0x0F0F0F0F0F0F0F0Full;
    v = (v | (v << 2)) & 0x3333333333333333ull;
    v = (v | (v << 1)) & 0x5555555555555555ull;
    return v;
}

static inline uint64_t morton2D(uint32_t x, uint32_t y) {
    return splitBy1(x) | (splitBy1(y) << 1);
}

static std::vector<uint32_t> heldKarp(const std::vector<City>& cities) {
    const size_t n = cities.size();
    const size_t full = size_t(1) << n;
    const double INF = std::numeric_limits<double>::infinity();
    std::vector<double> dp(full * n, INF);
    std::vector<uint8_t> parent(full * n, 255);
    dp[1] = 0.0;

    for (size_t mask = 1; mask < full; ++mask) {
        if ((mask & 1) == 0) continue;
        for (size_t last = 0; last < n; ++last) {
            const double cur = dp[mask * n + last];
            if (!std::isfinite(cur)) continue;
            const size_t remaining = (~mask) & (full - 1);
            for (size_t bits = remaining; bits; bits &= bits - 1) {
                const size_t lsb = bits & (~bits + 1);
                unsigned long nxtUL = 0;
#if defined(_MSC_VER)
                _BitScanForward64(&nxtUL, static_cast<unsigned long long>(lsb));
#else
                nxtUL = static_cast<unsigned long>(__builtin_ctzll(lsb));
#endif
                const size_t nxt = static_cast<size_t>(nxtUL);
                const size_t nm = mask | lsb;
                const double cand = cur + std::sqrt(dist2(cities[last], cities[nxt]));
                double& cell = dp[nm * n + nxt];
                if (cand < cell) {
                    cell = cand;
                    parent[nm * n + nxt] = static_cast<uint8_t>(last);
                }
            }
        }
    }

    size_t bestLast = 0;
    double best = INF;
    const size_t all = full - 1;
    for (size_t last = 1; last < n; ++last) {
        const double cand = dp[all * n + last] + std::sqrt(dist2(cities[last], cities[0]));
        if (cand < best) {
            best = cand;
            bestLast = last;
        }
    }

    std::vector<uint32_t> route(n);
    size_t mask = all;
    size_t cur = bestLast;
    for (size_t pos = n; pos-- > 1;) {
        route[pos] = cities[cur].id;
        const uint8_t p = parent[mask * n + cur];
        mask ^= size_t(1) << cur;
        cur = p;
    }
    route[0] = 0;
    return route;
}

static std::vector<uint32_t> nearestWindowRoute(const std::vector<City>& sorted, size_t zeroPos, int window) {
    const size_t n = sorted.size();
    std::vector<uint32_t> route;
    route.reserve(n);
    std::vector<unsigned char> used(n, 0);
    std::vector<uint32_t> rankById(n);
    for (uint32_t i = 0; i < static_cast<uint32_t>(n); ++i) rankById[sorted[i].id] = i;

    uint32_t cur = static_cast<uint32_t>(zeroPos);
    for (size_t step = 0; step < n; ++step) {
        used[cur] = 1;
        route.push_back(sorted[cur].id);
        if (step + 1 == n) break;

        double bestD = std::numeric_limits<double>::infinity();
        uint32_t best = UINT32_MAX;
        const int c = static_cast<int>(cur);
        const int lo = std::max(0, c - window);
        const int hi = std::min<int>(static_cast<int>(n) - 1, c + window);
        for (int i = lo; i <= hi; ++i) {
            if (!used[static_cast<size_t>(i)]) {
                const double d = dist2(sorted[cur], sorted[static_cast<size_t>(i)]);
                if (d < bestD) {
                    bestD = d;
                    best = static_cast<uint32_t>(i);
                }
            }
        }
        if (best == UINT32_MAX) {
            for (uint32_t radius = static_cast<uint32_t>(window + 1); best == UINT32_MAX; radius *= 2) {
                const uint32_t l = (cur > radius) ? cur - radius : 0;
                const uint32_t r = std::min<uint32_t>(static_cast<uint32_t>(n - 1), cur + radius);
                for (uint32_t i = l; i <= r; ++i) {
                    if (!used[i]) {
                        const double d = dist2(sorted[cur], sorted[i]);
                        if (d < bestD) {
                            bestD = d;
                            best = i;
                        }
                    }
                }
                if (l == 0 && r + 1 == n) break;
            }
        }
        cur = best;
    }
    (void)rankById;
    return route;
}

static void twoOptLimited(std::vector<uint32_t>& route, const std::vector<City>& byId, int passes, int span) {
    const size_t n = route.size();
    if (n < 4) return;
    for (int pass = 0; pass < passes; ++pass) {
        bool changed = false;
        for (size_t i = 1; i + 2 < n; ++i) {
            const City& a = byId[route[i - 1]];
            const City& b = byId[route[i]];
            const size_t maxJ = std::min(n - 1, i + static_cast<size_t>(span));
            for (size_t j = i + 1; j < maxJ; ++j) {
                const City& c = byId[route[j]];
                const City& d = byId[route[j + 1]];
                const double oldLen = std::sqrt(dist2(a, b)) + std::sqrt(dist2(c, d));
                const double newLen = std::sqrt(dist2(a, c)) + std::sqrt(dist2(b, d));
                if (newLen + 1e-9 < oldLen) {
                    std::reverse(route.begin() + static_cast<std::ptrdiff_t>(i),
                                 route.begin() + static_cast<std::ptrdiff_t>(j + 1));
                    changed = true;
                    break;
                }
            }
        }
        if (!changed) break;
    }
}

int main() {
    FastInput in;
    size_t n = 0;
    if (!in.readSize(n)) return 0;

    std::vector<City> cities;
    cities.resize(n);
    double minX = std::numeric_limits<double>::infinity();
    double minY = std::numeric_limits<double>::infinity();
    double maxX = -std::numeric_limits<double>::infinity();
    double maxY = -std::numeric_limits<double>::infinity();

    for (size_t i = 0; i < n; ++i) {
        double x = 0.0, y = 0.0;
        in.readDouble(x);
        in.readDouble(y);
        cities[i].x = static_cast<float>(x);
        cities[i].y = static_cast<float>(y);
        cities[i].id = static_cast<uint32_t>(i);
        cities[i].key = 0;
        minX = std::min(minX, x);
        minY = std::min(minY, y);
        maxX = std::max(maxX, x);
        maxY = std::max(maxY, y);
    }

    std::vector<uint32_t> route;
    if (n <= 1) {
        route.resize(n);
        if (n == 1) route[0] = 0;
    } else if (n <= 20) {
        route = heldKarp(cities);
    } else {
        const double sx = (maxX > minX) ? (4294967295.0 / (maxX - minX)) : 0.0;
        const double sy = (maxY > minY) ? (4294967295.0 / (maxY - minY)) : 0.0;
        for (size_t i = 0; i < n; ++i) {
            const uint32_t qx = static_cast<uint32_t>(std::max(0.0, std::min(4294967295.0, (static_cast<double>(cities[i].x) - minX) * sx)));
            const uint32_t qy = static_cast<uint32_t>(std::max(0.0, std::min(4294967295.0, (static_cast<double>(cities[i].y) - minY) * sy)));
            cities[i].key = morton2D(qx, qy);
        }

        std::sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
            if (a.key != b.key) return a.key < b.key;
            return a.id < b.id;
        });

        size_t zeroPos = 0;
        while (zeroPos < n && cities[zeroPos].id != 0) ++zeroPos;

        if (n <= 5000) {
            const int window = (n <= 200) ? 80 : 48;
            route = nearestWindowRoute(cities, zeroPos, window);

            std::vector<City> byId(n);
            for (const City& c : cities) byId[c.id] = c;
            twoOptLimited(route, byId, (n <= 1000) ? 3 : 1, (n <= 1000) ? 64 : 24);
        } else {
            route.reserve(n);
            for (size_t k = 0; k < n; ++k) {
                route.push_back(cities[(zeroPos + k) % n].id);
            }
        }

        if (!route.empty() && route[0] != 0) {
            const auto it = std::find(route.begin(), route.end(), 0u);
            if (it != route.end()) std::rotate(route.begin(), it, route.end());
        }
    }

    FastOutput out;
    for (size_t i = 0; i < route.size(); ++i) {
        if (i) out.putChar(' ');
        out.writeSize(route[i]);
    }
    out.putChar('\n');
    return 0;
}
