#include <algorithm>
#include <array>
#include <cstdint>
#include <cmath>
#include <iostream>
#include <limits>
#include <numeric>
#include <string>
#include <vector>

struct City {
    float x = 0.0f;
    float y = 0.0f;
    uint32_t id = 0;
    uint64_t key = 0;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = static_cast<double>(a.x) - static_cast<double>(b.x);
    const double dy = static_cast<double>(a.y) - static_cast<double>(b.y);
    return dx * dx + dy * dy;
}

static inline double dist2_xy(float ax, float ay, float bx, float by) {
    const double dx = static_cast<double>(ax) - static_cast<double>(bx);
    const double dy = static_cast<double>(ay) - static_cast<double>(by);
    return dx * dx + dy * dy;
}

static inline uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 32)) & 0x00000000ffffffffULL;
    v = (v | (v << 16)) & 0x0000ffff0000ffffULL;
    v = (v | (v << 8))  & 0x00ff00ff00ff00ffULL;
    v = (v | (v << 4))  & 0x0f0f0f0f0f0f0f0fULL;
    v = (v | (v << 2))  & 0x3333333333333333ULL;
    v = (v | (v << 1))  & 0x5555555555555555ULL;
    return v;
}

static inline uint64_t morton2(uint32_t x, uint32_t y) {
    return part1by1(x) | (part1by1(y) << 1);
}

static void write_route(const std::vector<uint32_t>& route) {
    std::string out;
    out.reserve(route.size() * 11ULL);
    for (size_t i = 0; i < route.size(); ++i) {
        if (i) out.push_back(' ');
        out += std::to_string(route[i]);
        if (out.size() > (1U << 20)) {
            std::cout << out;
            out.clear();
        }
    }
    out.push_back('\n');
    std::cout << out;
}

static void write_identity_route(uint64_t n) {
    std::string out;
    out.reserve(1U << 20);
    for (uint64_t i = 0; i < n; ++i) {
        if (i) out.push_back(' ');
        out += std::to_string(i);
        if (out.size() > (1U << 20)) {
            std::cout << out;
            out.clear();
        }
    }
    out.push_back('\n');
    std::cout << out;
}

static std::vector<uint32_t> exact_tiny_route(const std::vector<City>& c) {
    const int n = static_cast<int>(c.size());
    std::vector<uint32_t> p;
    for (int i = 1; i < n; ++i) p.push_back(static_cast<uint32_t>(i));

    std::vector<uint32_t> best = p;
    double best_len = std::numeric_limits<double>::infinity();
    do {
        double len = std::sqrt(dist2(c[0], c[p[0]]));
        for (int i = 1; i < n - 1; ++i) {
            len += std::sqrt(dist2(c[p[i - 1]], c[p[i]]));
        }
        len += std::sqrt(dist2(c[p.back()], c[0]));
        if (len < best_len) {
            best_len = len;
            best = p;
        }
    } while (std::next_permutation(p.begin(), p.end()));

    std::vector<uint32_t> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), best.begin(), best.end());
    return route;
}

static std::vector<uint32_t> nearest_route(const std::vector<City>& c) {
    const uint32_t n = static_cast<uint32_t>(c.size());
    std::vector<uint32_t> route;
    std::vector<uint8_t> used(n, 0);
    route.reserve(n);
    route.push_back(0);
    used[0] = 1;

    uint32_t cur = 0;
    for (uint32_t step = 1; step < n; ++step) {
        uint32_t best = 0;
        double best_d = std::numeric_limits<double>::infinity();
        for (uint32_t j = 1; j < n; ++j) {
            if (!used[j]) {
                const double d = dist2(c[cur], c[j]);
                if (d < best_d) {
                    best_d = d;
                    best = j;
                }
            }
        }
        used[best] = 1;
        route.push_back(best);
        cur = best;
    }
    return route;
}

static void two_opt_limited(std::vector<uint32_t>& route, const std::vector<City>& c, int passes, int span_limit) {
    const int n = static_cast<int>(route.size());
    if (n < 5) return;

    for (int pass = 0; pass < passes; ++pass) {
        bool changed = false;
        for (int i = 1; i < n - 2; ++i) {
            const int max_k = std::min(n - 1, i + span_limit);
            for (int k = i + 1; k <= max_k; ++k) {
                const City& a = c[route[i - 1]];
                const City& b = c[route[i]];
                const City& d = c[route[k]];
                const City& e = c[route[(k + 1) % n]];

                const double old_len = std::sqrt(dist2(a, b)) + std::sqrt(dist2(d, e));
                const double new_len = std::sqrt(dist2(a, d)) + std::sqrt(dist2(b, e));
                if (new_len + 1e-12 < old_len) {
                    std::reverse(route.begin() + i, route.begin() + k + 1);
                    changed = true;
                }
            }
        }
        if (!changed) break;
    }
}

static std::vector<uint32_t> morton_route(std::vector<City>& c) {
    float minx = c[0].x, maxx = c[0].x, miny = c[0].y, maxy = c[0].y;
    for (const City& p : c) {
        minx = std::min(minx, p.x);
        maxx = std::max(maxx, p.x);
        miny = std::min(miny, p.y);
        maxy = std::max(maxy, p.y);
    }

    const double sx = (maxx > minx) ? (2097151.0 / (static_cast<double>(maxx) - minx)) : 0.0;
    const double sy = (maxy > miny) ? (2097151.0 / (static_cast<double>(maxy) - miny)) : 0.0;
    for (City& p : c) {
        const uint32_t ix = static_cast<uint32_t>(std::max(0.0, std::min(2097151.0, (static_cast<double>(p.x) - minx) * sx)));
        const uint32_t iy = static_cast<uint32_t>(std::max(0.0, std::min(2097151.0, (static_cast<double>(p.y) - miny) * sy)));
        p.key = morton2(ix, iy);
    }

    std::sort(c.begin(), c.end(), [](const City& a, const City& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.id < b.id;
    });

    size_t zero_pos = 0;
    for (size_t i = 0; i < c.size(); ++i) {
        if (c[i].id == 0) {
            zero_pos = i;
            break;
        }
    }

    std::vector<uint32_t> route;
    route.reserve(c.size());
    for (size_t t = 0; t < c.size(); ++t) route.push_back(c[(zero_pos + t) % c.size()].id);

    if (route.size() > 2) {
        auto by_id = [&c](uint32_t id) -> const City& {
            return *std::find_if(c.begin(), c.end(), [id](const City& p) { return p.id == id; });
        };
        const City& z = by_id(0);
        const City& forward = by_id(route[1]);
        const City& backward = by_id(route.back());
        if (dist2(z, backward) < dist2(z, forward)) {
            std::reverse(route.begin() + 1, route.end());
        }
    }
    return route;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    uint64_t n64 = 0;
    if (!(std::cin >> n64)) return 0;
    if (n64 == 0) {
        std::cout << '\n';
        return 0;
    }

    if (n64 > 20000000ULL) {
        write_identity_route(n64);
        return 0;
    }

    const uint64_t max_supported = static_cast<uint64_t>(std::numeric_limits<uint32_t>::max());
    if (n64 > max_supported) {
        write_identity_route(n64);
        return 0;
    }
    const uint32_t n = static_cast<uint32_t>(n64);

    std::vector<City> cities;
    cities.reserve(n);
    for (uint32_t i = 0; i < n; ++i) {
        City p;
        std::cin >> p.x >> p.y;
        p.id = i;
        cities.push_back(p);
    }

    std::vector<uint32_t> route;
    if (n <= 1) {
        route.push_back(0);
    } else if (n <= 10) {
        route = exact_tiny_route(cities);
    } else if (n <= 2500) {
        route = nearest_route(cities);
        two_opt_limited(route, cities, n <= 200 ? 8 : 3, n <= 200 ? n : 80);
    } else {
        route = morton_route(cities);
        if (n <= 100000) {
            std::vector<City> by_id(n);
            for (const City& p : cities) by_id[p.id] = p;
            two_opt_limited(route, by_id, n <= 20000 ? 2 : 1, 32);
        }
    }

    if (!route.empty() && route[0] != 0) {
        auto it = std::find(route.begin(), route.end(), 0);
        if (it != route.end()) std::rotate(route.begin(), it, route.end());
    }

    write_route(route);
    return 0;
}
