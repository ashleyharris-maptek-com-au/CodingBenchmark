use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone)]
struct Sample {
    hole: String,
    idx: usize,
    x: f64,
    y: f64,
    z: f64,
    vals: Vec<f64>,
}

#[derive(Clone)]
struct Suspect {
    hole: String,
    idx: usize,
    prop: String,
    confidence: f64,
}

fn parse_f64(s: Option<&str>) -> f64 {
    match s {
        Some(t) => t.parse::<f64>().unwrap_or(f64::NAN),
        None => f64::NAN,
    }
}

fn parse_usize(s: Option<&str>) -> usize {
    match s {
        Some(t) => t.parse::<usize>().unwrap_or(0),
        None => 0,
    }
}

fn clean(v: f64) -> bool {
    v.is_finite()
}

fn median_sorted(v: &[f64]) -> f64 {
    if v.is_empty() {
        return f64::NAN;
    }
    let n = v.len();
    if n & 1 == 1 {
        v[n / 2]
    } else {
        (v[n / 2 - 1] + v[n / 2]) * 0.5
    }
}

fn robust_stats(samples: &[Sample], props: usize) -> (Vec<f64>, Vec<f64>) {
    let mut med = vec![0.0; props];
    let mut scale = vec![1.0; props];
    for p in 0..props {
        let mut a = Vec::new();
        for s in samples {
            let v = s.vals[p];
            if clean(v) {
                a.push(v);
            }
        }
        if a.is_empty() {
            med[p] = 0.0;
            scale[p] = 1.0;
            continue;
        }
        a.sort_by(|x, y| x.partial_cmp(y).unwrap_or(Ordering::Equal));
        let m = median_sorted(&a);
        let mut dev: Vec<f64> = a.iter().map(|x| (x - m).abs()).collect();
        dev.sort_by(|x, y| x.partial_cmp(y).unwrap_or(Ordering::Equal));
        let mad = median_sorted(&dev) * 1.4826;
        let q1 = a[a.len() / 4];
        let q3 = a[(a.len() * 3) / 4];
        let iqr_scale = (q3 - q1).abs() / 1.349;
        let mut sc = mad.max(iqr_scale).max(1e-12);
        if !sc.is_finite() {
            sc = 1.0;
        }
        med[p] = m;
        scale[p] = sc;
    }
    (med, scale)
}

fn dist2(a: &Sample, b: &Sample) -> f64 {
    let dx = a.x - b.x;
    let dy = a.y - b.y;
    let dz = a.z - b.z;
    dx * dx + dy * dy + dz * dz
}

fn local_score(target: &Sample, neigh: &[usize], samples: &[Sample], prop: usize, global_med: f64, global_scale: f64) -> Option<f64> {
    let v = target.vals[prop];
    if !clean(v) {
        return Some(12.0);
    }
    let global_z = (v - global_med).abs() / global_scale.max(1e-12);
    if global_z >= 12.0 {
        return Some(global_z);
    }
    let mut vals = Vec::with_capacity(neigh.len());
    for &j in neigh {
        let nv = samples[j].vals[prop];
        if clean(nv) {
            vals.push(nv);
        }
    }
    if vals.len() < 4 {
        return if global_z >= 8.0 { Some(global_z) } else { None };
    }
    vals.sort_by(|x, y| x.partial_cmp(y).unwrap_or(Ordering::Equal));
    let lm = median_sorted(&vals);
    let mut dev: Vec<f64> = vals.iter().map(|x| (x - lm).abs()).collect();
    dev.sort_by(|x, y| x.partial_cmp(y).unwrap_or(Ordering::Equal));
    let ls = (median_sorted(&dev) * 1.4826).max(global_scale * 0.05).max(1e-12);
    let local_z = (v - lm).abs() / ls;
    let score = 0.75 * local_z + 0.25 * global_z;
    if score >= 5.5 && local_z >= 4.5 {
        Some(score)
    } else {
        None
    }
}

fn cell_key(x: f64, y: f64, z: f64, cell: f64) -> (i64, i64, i64) {
    (
        (x / cell).floor() as i64,
        (y / cell).floor() as i64,
        (z / cell).floor() as i64,
    )
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let holes = parse_usize(it.next());
    let props = parse_usize(it.next());
    if holes == 0 || props == 0 {
        println!("0");
        return;
    }

    let mut prop_names = Vec::with_capacity(props);
    for i in 0..props {
        prop_names.push(it.next().map(|s| s.to_string()).unwrap_or_else(|| format!("prop{}", i + 1)));
    }

    let mut samples = Vec::new();
    for _ in 0..holes {
        let hole = match it.next() {
            Some(h) => h.to_string(),
            None => break,
        };
        let sx = parse_f64(it.next());
        let sy = parse_f64(it.next());
        let sz = parse_f64(it.next());
        let dx = parse_f64(it.next());
        let dy = parse_f64(it.next());
        let dz = parse_f64(it.next());
        let len = parse_f64(it.next());
        let ns = parse_usize(it.next());
        let norm = (dx * dx + dy * dy + dz * dz).sqrt();
        let (ux, uy, uz) = if clean(norm) && norm > 0.0 {
            (dx / norm, dy / norm, dz / norm)
        } else {
            (0.0, 0.0, 1.0)
        };
        let max_depth = if clean(len) && len >= 0.0 { len } else { f64::INFINITY };
        for si in 0..ns {
            let mut depth = parse_f64(it.next());
            if !clean(depth) {
                depth = 0.0;
            }
            if depth < 0.0 {
                depth = 0.0;
            }
            if depth > max_depth {
                depth = max_depth;
            }
            let mut vals = Vec::with_capacity(props);
            for _ in 0..props {
                vals.push(parse_f64(it.next()));
            }
            samples.push(Sample {
                hole: hole.clone(),
                idx: si,
                x: sx + ux * depth,
                y: sy + uy * depth,
                z: sz + uz * depth,
                vals,
            });
        }
    }

    let n = samples.len();
    if n == 0 {
        println!("0");
        return;
    }

    let (gmed, gscale) = robust_stats(&samples, props);
    let k = if n < 2_000 { 16 } else if n < 100_000 { 24 } else { 32 };
    let mut suspects = Vec::new();

    if n <= 12_000 {
        for i in 0..n {
            let mut near: Vec<(f64, usize)> = Vec::with_capacity(k + 1);
            for j in 0..n {
                if i == j {
                    continue;
                }
                let d = dist2(&samples[i], &samples[j]);
                if near.len() < k {
                    near.push((d, j));
                    if near.len() == k {
                        near.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(Ordering::Equal));
                    }
                } else if d < near[0].0 {
                    near[0] = (d, j);
                    near.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(Ordering::Equal));
                }
            }
            let ids: Vec<usize> = near.iter().map(|x| x.1).collect();
            for p in 0..props {
                if let Some(sc) = local_score(&samples[i], &ids, &samples, p, gmed[p], gscale[p]) {
                    suspects.push(Suspect { hole: samples[i].hole.clone(), idx: samples[i].idx, prop: prop_names[p].clone(), confidence: sc });
                }
            }
        }
    } else {
        let mut minx = samples[0].x;
        let mut maxx = samples[0].x;
        let mut miny = samples[0].y;
        let mut maxy = samples[0].y;
        let mut minz = samples[0].z;
        let mut maxz = samples[0].z;
        for s in &samples {
            minx = minx.min(s.x); maxx = maxx.max(s.x);
            miny = miny.min(s.y); maxy = maxy.max(s.y);
            minz = minz.min(s.z); maxz = maxz.max(s.z);
        }
        let volume = ((maxx - minx).abs() * (maxy - miny).abs() * (maxz - minz).abs()).max(1.0);
        let target_per_cell = if n > 500_000 { 48.0 } else { 32.0 };
        let mut cell = (volume * target_per_cell / n as f64).cbrt();
        if !cell.is_finite() || cell <= 0.0 {
            cell = 10.0;
        }
        let mut grid: HashMap<(i64, i64, i64), Vec<usize>> = HashMap::with_capacity(n / 8 + 1);
        for (i, s) in samples.iter().enumerate() {
            grid.entry(cell_key(s.x, s.y, s.z, cell)).or_default().push(i);
        }

        for i in 0..n {
            let key = cell_key(samples[i].x, samples[i].y, samples[i].z, cell);
            let mut near: Vec<(f64, usize)> = Vec::with_capacity(k + 1);
            let mut radius = 1_i64;
            while near.len() < k && radius <= 4 {
                for ax in -radius..=radius {
                    for ay in -radius..=radius {
                        for az in -radius..=radius {
                            if ax.abs().max(ay.abs()).max(az.abs()) != radius {
                                continue;
                            }
                            if let Some(bucket) = grid.get(&(key.0 + ax, key.1 + ay, key.2 + az)) {
                                for &j in bucket {
                                    if i == j {
                                        continue;
                                    }
                                    let d = dist2(&samples[i], &samples[j]);
                                    if near.len() < k {
                                        near.push((d, j));
                                        if near.len() == k {
                                            near.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(Ordering::Equal));
                                        }
                                    } else if d < near[0].0 {
                                        near[0] = (d, j);
                                        near.sort_by(|a, b| b.0.partial_cmp(&a.0).unwrap_or(Ordering::Equal));
                                    }
                                }
                            }
                        }
                    }
                }
                radius += 1;
            }
            if near.len() < 4 {
                for j in i.saturating_sub(3)..(i + 4).min(n) {
                    if i != j {
                        near.push((dist2(&samples[i], &samples[j]), j));
                    }
                }
            }
            near.sort_by(|a, b| a.0.partial_cmp(&b.0).unwrap_or(Ordering::Equal));
            near.truncate(k);
            let ids: Vec<usize> = near.iter().map(|x| x.1).collect();
            for p in 0..props {
                if let Some(sc) = local_score(&samples[i], &ids, &samples, p, gmed[p], gscale[p]) {
                    suspects.push(Suspect { hole: samples[i].hole.clone(), idx: samples[i].idx, prop: prop_names[p].clone(), confidence: sc });
                }
            }
        }
    }

    suspects.sort_by(|a, b| {
        b.confidence
            .partial_cmp(&a.confidence)
            .unwrap_or(Ordering::Equal)
            .then_with(|| a.hole.cmp(&b.hole))
            .then_with(|| a.idx.cmp(&b.idx))
            .then_with(|| a.prop.cmp(&b.prop))
    });
    let cap = suspects.len().min(10_000);
    println!("{}", cap);
    for s in suspects.into_iter().take(cap) {
        println!("{} {} {} {:.6}", s.hole, s.idx, s.prop, s.confidence);
    }
}
