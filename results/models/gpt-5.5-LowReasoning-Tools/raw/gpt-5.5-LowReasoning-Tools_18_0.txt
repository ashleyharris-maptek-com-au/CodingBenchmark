// Minimum Edge Cut in an undirected connected graph.
// C++20, standard library only, tested-oriented implementation for MSVC.
//
// Strategy:
//   1. Read the graph once into compact edge arrays.
//   2. If there is a bridge, the answer is immediately one edge.
//   3. If the current upper bound (minimum degree) is small, run exact
//      unit-capacity s-t max-flow from a minimum-degree vertex to other
//      vertices until the global cut is proved/found.
//   4. Otherwise run exact Stoer-Wagner global min-cut on a sparse
//      adjacency-map representation. It supports parallel edges.
//
// Exact global min-cut for an arbitrary multi-GB graph is not a one-pass
// streaming problem in the comparison/RAM model; an exact solver must retain
// or externally index enough incidence information to verify cuts. This
// program is written to be adaptive and compact, while still always using
// exact algorithms for the graph it processes.

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <unordered_map>
#include <utility>
#include <vector>

using std::int64_t;

struct FastInput {
    static constexpr size_t BUFSIZE = 1u << 20;
    char buf[BUFSIZE];
    size_t pos = 0, len = 0;

    char getChar() {
        if (pos == len) {
            len = std::fread(buf, 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return 0;
        }
        return buf[pos++];
    }

    bool readUInt(uint32_t& out) {
        char c = getChar();
        while (c && (c < '0' || c > '9')) c = getChar();
        if (!c) return false;
        uint64_t x = 0;
        while (c >= '0' && c <= '9') {
            x = x * 10 + static_cast<unsigned>(c - '0');
            c = getChar();
        }
        out = static_cast<uint32_t>(x);
        return true;
    }

    bool readULL(uint64_t& out) {
        char c = getChar();
        while (c && (c < '0' || c > '9')) c = getChar();
        if (!c) return false;
        uint64_t x = 0;
        while (c >= '0' && c <= '9') {
            x = x * 10 + static_cast<unsigned>(c - '0');
            c = getChar();
        }
        out = x;
        return true;
    }
};

struct Edge {
    uint32_t u, v;
};

struct CutAnswer {
    uint64_t value = 0;
    std::vector<unsigned char> side;
};

static void printCutFromSide(const std::vector<Edge>& edges, const std::vector<unsigned char>& side) {
    std::vector<Edge> cut;
    cut.reserve(1024);
    for (const Edge& e : edges) {
        if (e.u != e.v && side[e.u] != side[e.v]) cut.push_back(e);
    }
    std::printf("%zu\n", cut.size());
    for (const Edge& e : cut) std::printf("%u %u\n", e.u, e.v);
}

static bool findBridge(uint32_t n, const std::vector<Edge>& edges, const std::vector<std::vector<uint32_t>>& adjEdge,
                       Edge& bridge) {
    std::vector<uint32_t> tin(n, 0), low(n, 0), parentEdge(n, std::numeric_limits<uint32_t>::max());
    std::vector<uint32_t> it(n, 0), parent(n, std::numeric_limits<uint32_t>::max());
    uint32_t timer = 0;

    for (uint32_t root = 0; root < n; ++root) {
        if (tin[root]) continue;
        std::vector<uint32_t> st;
        st.push_back(root);
        parent[root] = root;
        while (!st.empty()) {
            uint32_t u = st.back();
            if (!tin[u]) tin[u] = low[u] = ++timer;

            if (it[u] < adjEdge[u].size()) {
                uint32_t eid = adjEdge[u][it[u]++];
                const Edge& e = edges[eid];
                uint32_t v = e.u ^ e.v ^ u;
                if (eid == parentEdge[u]) continue;
                if (!tin[v]) {
                    parent[v] = u;
                    parentEdge[v] = eid;
                    st.push_back(v);
                } else {
                    low[u] = std::min(low[u], tin[v]);
                }
            } else {
                st.pop_back();
                if (u != root) {
                    uint32_t p = parent[u];
                    low[p] = std::min(low[p], low[u]);
                    if (low[u] > tin[p]) {
                        bridge = edges[parentEdge[u]];
                        return true;
                    }
                }
            }
        }
    }
    return false;
}

struct Dinic {
    struct Arc {
        uint32_t to, rev;
        int cap;
    };
    uint32_t n = 0;
    std::vector<std::vector<Arc>> g;
    std::vector<int> level;
    std::vector<uint32_t> ptr;

    explicit Dinic(uint32_t n_) : n(n_), g(n_), level(n_), ptr(n_) {}

    void addArc(uint32_t u, uint32_t v, int c) {
        Arc a{v, static_cast<uint32_t>(g[v].size()), c};
        Arc b{u, static_cast<uint32_t>(g[u].size()), 0};
        g[u].push_back(a);
        g[v].push_back(b);
    }

    bool bfs(uint32_t s, uint32_t t) {
        std::fill(level.begin(), level.end(), -1);
        std::queue<uint32_t> q;
        level[s] = 0;
        q.push(s);
        while (!q.empty()) {
            uint32_t u = q.front();
            q.pop();
            for (const Arc& a : g[u]) {
                if (a.cap > 0 && level[a.to] < 0) {
                    level[a.to] = level[u] + 1;
                    if (a.to == t) return true;
                    q.push(a.to);
                }
            }
        }
        return level[t] >= 0;
    }

    int dfs(uint32_t u, uint32_t t, int pushed) {
        if (u == t || pushed == 0) return pushed;
        for (uint32_t& cid = ptr[u]; cid < g[u].size(); ++cid) {
            Arc& a = g[u][cid];
            if (a.cap <= 0 || level[a.to] != level[u] + 1) continue;
            int tr = dfs(a.to, t, std::min(pushed, a.cap));
            if (!tr) continue;
            a.cap -= tr;
            g[a.to][a.rev].cap += tr;
            return tr;
        }
        return 0;
    }

    int maxflow(uint32_t s, uint32_t t, int limit) {
        int flow = 0;
        while (flow < limit && bfs(s, t)) {
            std::fill(ptr.begin(), ptr.end(), 0);
            while (flow < limit) {
                int pushed = dfs(s, t, limit - flow);
                if (!pushed) break;
                flow += pushed;
            }
        }
        return flow;
    }

    std::vector<unsigned char> reachable(uint32_t s) const {
        std::vector<unsigned char> seen(n, 0);
        std::queue<uint32_t> q;
        seen[s] = 1;
        q.push(s);
        while (!q.empty()) {
            uint32_t u = q.front();
            q.pop();
            for (const Arc& a : g[u]) {
                if (a.cap > 0 && !seen[a.to]) {
                    seen[a.to] = 1;
                    q.push(a.to);
                }
            }
        }
        return seen;
    }
};

static bool trySmallDegreeFlows(uint32_t n, const std::vector<Edge>& edges, const std::vector<uint32_t>& degree,
                                CutAnswer& ans) {
    uint32_t s = 0;
    for (uint32_t i = 1; i < n; ++i) {
        if (degree[i] < degree[s]) s = i;
    }
    uint32_t upper = degree[s];
    if (upper <= 2) {
        ans.value = upper;
        ans.side.assign(n, 0);
        ans.side[s] = 1;
        return true;
    }
    if (upper > 16 || n > 200000 || edges.size() > 1000000) return false;

    ans.value = upper;
    ans.side.assign(n, 0);
    ans.side[s] = 1;

    for (uint32_t t = 0; t < n; ++t) {
        if (t == s) continue;
        Dinic din(n);
        for (const Edge& e : edges) {
            if (e.u == e.v) continue;
            din.addArc(e.u, e.v, 1);
            din.addArc(e.v, e.u, 1);
        }
        int f = din.maxflow(s, t, static_cast<int>(ans.value));
        if (static_cast<uint64_t>(f) < ans.value) {
            ans.value = static_cast<uint64_t>(f);
            ans.side = din.reachable(s);
            if (ans.value <= 2) return true;
        }
    }
    return true;
}

static CutAnswer stoerWagner(uint32_t n, const std::vector<Edge>& edges) {
    std::vector<std::unordered_map<uint32_t, uint64_t>> g(n);
    for (const Edge& e : edges) {
        if (e.u == e.v) continue;
        ++g[e.u][e.v];
        ++g[e.v][e.u];
    }

    std::vector<uint32_t> vert(n);
    std::iota(vert.begin(), vert.end(), 0);
    std::vector<std::vector<uint32_t>> members(n);
    for (uint32_t i = 0; i < n; ++i) members[i].push_back(i);

    uint64_t best = std::numeric_limits<uint64_t>::max();
    std::vector<unsigned char> bestSide(n, 0);

    while (vert.size() > 1) {
        std::vector<uint64_t> w(n, 0);
        std::vector<unsigned char> added(n, 0);
        uint32_t prev = vert[0], sel = vert[0];

        for (size_t i = 0; i < vert.size(); ++i) {
            sel = vert[0];
            bool have = false;
            for (uint32_t v : vert) {
                if (!added[v] && (!have || w[v] > w[sel])) {
                    sel = v;
                    have = true;
                }
            }
            if (i + 1 == vert.size()) {
                if (w[sel] < best) {
                    best = w[sel];
                    std::fill(bestSide.begin(), bestSide.end(), 0);
                    for (uint32_t original : members[sel]) bestSide[original] = 1;
                }

                for (const auto& kv : g[sel]) {
                    uint32_t to = kv.first;
                    uint64_t wt = kv.second;
                    if (to == prev) continue;
                    g[prev][to] += wt;
                    g[to][prev] += wt;
                    g[to].erase(sel);
                }
                g[prev].erase(sel);
                g[sel].clear();
                members[prev].insert(members[prev].end(), members[sel].begin(), members[sel].end());
                vert.erase(std::find(vert.begin(), vert.end(), sel));
                break;
            }

            added[sel] = 1;
            prev = sel;
            for (const auto& kv : g[sel]) {
                if (!added[kv.first]) w[kv.first] += kv.second;
            }
        }
    }

    CutAnswer ans;
    ans.value = best;
    ans.side = std::move(bestSide);
    return ans;
}

int main() {
    FastInput in;
    uint64_t n64 = 0, m64 = 0;
    if (!in.readULL(n64) || !in.readULL(m64)) return 0;
    if (n64 == 0 || n64 > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max())) {
        std::printf("0\n");
        return 0;
    }
    const uint32_t n = static_cast<uint32_t>(n64);

    std::vector<Edge> edges;
    edges.reserve(static_cast<size_t>(std::min<uint64_t>(m64, static_cast<uint64_t>(std::numeric_limits<size_t>::max() / sizeof(Edge)))));
    std::vector<uint32_t> degree(n, 0);

    for (uint64_t i = 0; i < m64; ++i) {
        uint32_t u = 0, v = 0;
        if (!in.readUInt(u) || !in.readUInt(v)) break;
        if (u >= n || v >= n) continue;
        edges.push_back({u, v});
        if (u != v) {
            ++degree[u];
            ++degree[v];
        }
    }

    if (n <= 1) {
        std::printf("0\n");
        return 0;
    }

    std::vector<std::vector<uint32_t>> adjEdge(n);
    for (uint32_t i = 0; i < n; ++i) adjEdge[i].reserve(degree[i]);
    for (uint32_t i = 0; i < edges.size(); ++i) {
        const Edge& e = edges[i];
        if (e.u == e.v) continue;
        adjEdge[e.u].push_back(i);
        adjEdge[e.v].push_back(i);
    }

    Edge bridge{0, 0};
    if (findBridge(n, edges, adjEdge, bridge)) {
        std::printf("1\n%u %u\n", bridge.u, bridge.v);
        return 0;
    }

    CutAnswer ans;
    if (!trySmallDegreeFlows(n, edges, degree, ans)) {
        ans = stoerWagner(n, edges);
    }

    printCutFromSide(edges, ans.side);
    return 0;
}
