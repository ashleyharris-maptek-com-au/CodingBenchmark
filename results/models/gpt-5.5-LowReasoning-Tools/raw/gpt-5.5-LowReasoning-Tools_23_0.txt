use std::io::{self, BufRead, BufReader, BufWriter, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
    seen: bool,
}

#[derive(Clone, Copy, Debug)]
struct World {
    gravity: f64,
    max_thrust: f64,
    initial_fuel: f64,
    start_x: f64,
    start_y: f64,
    target_x: f64,
    target_y: f64,
    pad_left: f64,
    pad_right: f64,
}

struct Scanner<R: BufRead> {
    reader: R,
    buf: Vec<u8>,
}

impl<R: BufRead> Scanner<R> {
    fn new(reader: R) -> Self {
        Self {
            reader,
            buf: Vec::with_capacity(128),
        }
    }

    fn token(&mut self) -> Option<String> {
        self.buf.clear();
        loop {
            let available = self.reader.fill_buf().ok()?;
            if available.is_empty() {
                if self.buf.is_empty() {
                    return None;
                }
                return String::from_utf8(std::mem::take(&mut self.buf)).ok();
            }

            let available_len = available.len();
            let mut consumed = 0usize;
            while consumed < available_len && available[consumed].is_ascii_whitespace() {
                consumed += 1;
            }
            let found_token = consumed < available_len || !self.buf.is_empty();
            self.reader.consume(consumed);

            if found_token {
                break;
            }
        }

        loop {
            let available = self.reader.fill_buf().ok()?;
            if available.is_empty() {
                return String::from_utf8(std::mem::take(&mut self.buf)).ok();
            }

            let available_len = available.len();
            let mut end = 0usize;
            while end < available_len && !available[end].is_ascii_whitespace() {
                end += 1;
            }
            self.buf.extend_from_slice(&available[..end]);
            let token_done = end < available_len;
            self.reader.consume(end);

            if token_done {
                return String::from_utf8(std::mem::take(&mut self.buf)).ok();
            }
        }
    }

    fn f64(&mut self) -> Option<f64> {
        self.token()?.parse::<f64>().ok()
    }

    fn usize(&mut self) -> Option<usize> {
        self.token()?.parse::<usize>().ok()
    }
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    v.max(lo).min(hi)
}

fn read_world<R: BufRead>(scan: &mut Scanner<R>) -> Option<World> {
    let _width = scan.usize()?;
    let height = scan.usize()?;
    let gravity = scan.f64()?;
    let max_thrust = scan.f64()?;
    let initial_fuel = scan.f64()?;
    let start_x = scan.f64()?;
    let start_y = scan.f64()?;
    let target_x = scan.f64()?;
    let target_y = scan.f64()?;

    let mut pad_left = target_x.floor();
    let mut pad_right = target_x.ceil();

    for _row in 0..height {
        let row = scan.token()?;
        let bytes = row.as_bytes();
        let mut i = 0usize;
        while i < bytes.len() {
            if bytes[i] == b'L' || bytes[i] == b'T' {
                let left = i;
                let mut right = i;
                while right + 1 < bytes.len()
                    && (bytes[right + 1] == b'L' || bytes[right + 1] == b'T')
                {
                    right += 1;
                }
                if (left as f64 - 2.0) <= target_x && target_x <= (right as f64 + 2.0) {
                    pad_left = left as f64 + 0.5;
                    pad_right = right as f64 + 0.5;
                }
                i = right + 1;
            } else {
                i += 1;
            }
        }
    }

    Some(World {
        gravity,
        max_thrust,
        initial_fuel,
        start_x,
        start_y,
        target_x,
        target_y,
        pad_left,
        pad_right,
    })
}

fn controller(world: World, s: State) -> (f64, f64) {
    if !s.seen {
        return (0.0, 0.0);
    }

    let max_t = world.max_thrust.max(0.001);
    let g = world.gravity.max(0.0);
    let altitude = (world.target_y - s.y).max(0.0);
    let x_error = world.target_x - s.x;

    let inside_pad = s.x >= world.pad_left - 0.3 && s.x <= world.pad_right + 0.3;
    let pad_half = ((world.pad_right - world.pad_left).abs() * 0.5).max(1.0);

    let desired_vx = if altitude > 120.0 {
        clamp(0.22 * x_error, -22.0, 22.0)
    } else if altitude > 45.0 {
        clamp(0.18 * x_error, -12.0, 12.0)
    } else if altitude > 14.0 {
        clamp(0.13 * x_error, -5.0, 5.0)
    } else {
        clamp(0.08 * x_error, -1.0, 1.0)
    };

    let lateral_gain = if altitude < 12.0 { 0.55 } else { 0.8 };
    let mut want_ax = lateral_gain * (desired_vx - s.vx);
    if altitude < 8.0 && inside_pad {
        want_ax = -0.35 * s.vx;
    }
    want_ax = clamp(want_ax, -0.55 * max_t, 0.55 * max_t);

    let desired_vy = if altitude > 180.0 {
        24.0
    } else if altitude > 80.0 {
        16.0
    } else if altitude > 35.0 {
        8.0
    } else if altitude > 14.0 {
        4.0
    } else if altitude > 5.0 {
        1.8
    } else {
        0.6
    };

    let closing_margin = (2.0 * g.max(0.1) * altitude).sqrt();
    let emergency_brake = s.vy > desired_vy + closing_margin * 0.35 || (altitude < 25.0 && s.vy > 6.0);
    let vy_gain = if emergency_brake { 0.95 } else { 0.55 };
    let desired_net_ay = clamp(vy_gain * (desired_vy - s.vy), -0.85 * max_t, 0.35 * max_t);
    let mut required_up = g - desired_net_ay;

    if altitude < 10.0 {
        required_up += 0.15 * (2.0 - s.vy).max(0.0);
    }
    if emergency_brake {
        required_up = required_up.max(0.92 * max_t);
    }

    let max_angle = if altitude > 80.0 {
        0.75
    } else if altitude > 30.0 {
        0.48
    } else if altitude > 10.0 {
        0.25
    } else {
        0.10
    };

    if altitude < 7.0 || x_error.abs() < pad_half * 0.35 {
        want_ax *= 0.45;
    }

    let desired_angle = clamp(want_ax.atan2(required_up.max(0.1)), -max_angle, max_angle);
    let angle_error = desired_angle - s.angle;
    let turn = clamp(4.0 * angle_error - 1.4 * s.angular_vel, -1.0, 1.0);

    let up_from_angle = s.angle.cos().max(0.15);
    let mut thrust = (required_up / (max_t * up_from_angle)).max(0.0);
    thrust = thrust.max((want_ax.abs() / max_t).min(1.0));

    if altitude < 3.0 && s.vy < 0.2 && s.vx.abs() < 0.8 {
        thrust *= 0.82;
    }
    if s.fuel <= 0.05 {
        thrust = 0.0;
    } else if s.fuel < 2.0 && altitude > 20.0 && !emergency_brake {
        thrust = thrust.min(0.65);
    }

    (clamp(thrust, 0.0, 1.0), turn)
}

fn main() {
    let stdin = io::stdin();
    let mut scan = Scanner::new(BufReader::with_capacity(1 << 20, stdin.lock()));
    let Some(world) = read_world(&mut scan) else {
        return;
    };

    let latest = Arc::new(Mutex::new(State {
        x: world.start_x,
        y: world.start_y,
        vx: 0.0,
        vy: 0.0,
        angle: 0.0,
        angular_vel: 0.0,
        fuel: world.initial_fuel,
        seen: false,
    }));

    let output_state = Arc::clone(&latest);
    thread::spawn(move || {
        let stdout = io::stdout();
        let mut out = BufWriter::with_capacity(4096, stdout.lock());
        let tick = Duration::from_millis(40);
        let mut next = Instant::now();

        loop {
            let state = match output_state.lock() {
                Ok(st) => *st,
                Err(_) => break,
            };
            let (thrust, turn) = controller(world, state);

            if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
                break;
            }
            if out.flush().is_err() {
                break;
            }

            next += tick;
            let now = Instant::now();
            if next > now {
                thread::sleep(next - now);
            } else {
                next = now;
            }
        }
    });

    loop {
        let Some(x) = scan.f64() else { break };
        let Some(y) = scan.f64() else { break };
        let Some(vx) = scan.f64() else { break };
        let Some(vy) = scan.f64() else { break };
        let Some(angle) = scan.f64() else { break };
        let Some(angular_vel) = scan.f64() else { break };
        let Some(fuel) = scan.f64() else { break };

        if let Ok(mut st) = latest.lock() {
            *st = State {
                x,
                y,
                vx,
                vy,
                angle,
                angular_vel,
                fuel,
                seen: true,
            };
        } else {
            break;
        }
    }
}
