#include <algorithm>
#include <chrono>
#include <cmath>
#include <iomanip>
#include <iostream>
#include <limits>
#include <vector>

struct Vec3 {
    double x, y, z;
    Vec3() : x(0), y(0), z(0) {}
    Vec3(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec3 operator+(const Vec3& o) const { return Vec3(x + o.x, y + o.y, z + o.z); }
    Vec3 operator-(const Vec3& o) const { return Vec3(x - o.x, y - o.y, z - o.z); }
    Vec3 operator*(double s) const { return Vec3(x * s, y * s, z * s); }
};

static double dotp(const Vec3& a, const Vec3& b) {
    return a.x * b.x + a.y * b.y + a.z * b.z;
}

static Vec3 crossp(const Vec3& a, const Vec3& b) {
    return Vec3(a.y * b.z - a.z * b.y,
                a.z * b.x - a.x * b.z,
                a.x * b.y - a.y * b.x);
}

static double norm(const Vec3& a) {
    return std::sqrt(dotp(a, a));
}

struct Plane {
    Vec3 n;
    double d;
};

static bool inside_planes(const std::vector<Plane>& planes, const Vec3& p, double margin) {
    const double eps = 1e-9;
    for (const Plane& pl : planes) {
        if (dotp(pl.n, p) + pl.d > -margin + eps) return false;
    }
    return true;
}

static void add_box_planes(std::vector<Plane>& planes, const Vec3& mn, const Vec3& mx) {
    planes.push_back({Vec3(-1, 0, 0), mn.x});
    planes.push_back({Vec3(1, 0, 0), -mx.x});
    planes.push_back({Vec3(0, -1, 0), mn.y});
    planes.push_back({Vec3(0, 1, 0), -mx.y});
    planes.push_back({Vec3(0, 0, -1), mn.z});
    planes.push_back({Vec3(0, 0, 1), -mx.z});
}

static std::vector<Plane> build_planes(const std::vector<Vec3>& pts, const Vec3& mn, const Vec3& mx) {
    std::vector<Plane> planes;
    const int n = static_cast<int>(pts.size());
    const double scale = std::max({mx.x - mn.x, mx.y - mn.y, mx.z - mn.z, 1.0});
    bool axisBox = true;
    for (const Vec3& p : pts) {
        bool onX = std::abs(p.x - mn.x) < scale * 1e-9 || std::abs(p.x - mx.x) < scale * 1e-9;
        bool onY = std::abs(p.y - mn.y) < scale * 1e-9 || std::abs(p.y - mx.y) < scale * 1e-9;
        bool onZ = std::abs(p.z - mn.z) < scale * 1e-9 || std::abs(p.z - mx.z) < scale * 1e-9;
        if (!(onX && onY && onZ)) {
            axisBox = false;
            break;
        }
    }
    if (axisBox) {
        add_box_planes(planes, mn, mx);
        return planes;
    }

    if (n < 4) {
        add_box_planes(planes, mn, mx);
        return planes;
    }

    const long long work = 1LL * n * n * n * n;
    if (n > 180 || work > 120000000LL) {
        add_box_planes(planes, mn, mx);
        return planes;
    }

    Vec3 c;
    for (const Vec3& p : pts) c = c + p;
    c = c * (1.0 / n);

    const double eps = scale * 1e-9;

    for (int i = 0; i < n; ++i) {
        for (int j = i + 1; j < n; ++j) {
            for (int k = j + 1; k < n; ++k) {
                Vec3 normal = crossp(pts[j] - pts[i], pts[k] - pts[i]);
                double len = norm(normal);
                if (len <= eps) continue;
                normal = normal * (1.0 / len);
                double d = -dotp(normal, pts[i]);

                int pos = 0, neg = 0;
                for (int q = 0; q < n; ++q) {
                    double s = dotp(normal, pts[q]) + d;
                    if (s > eps) ++pos;
                    else if (s < -eps) ++neg;
                    if (pos && neg) break;
                }
                if (pos && neg) continue;

                if (dotp(normal, c) + d > 0) {
                    normal = normal * -1.0;
                    d = -d;
                }

                bool duplicate = false;
                for (const Plane& p : planes) {
                    if (std::abs(dotp(p.n, normal) - 1.0) < 1e-7 && std::abs(p.d - d) < scale * 1e-7) {
                        duplicate = true;
                        break;
                    }
                }
                if (!duplicate) planes.push_back({normal, d});
            }
        }
    }

    if (planes.empty()) add_box_planes(planes, mn, mx);
    return planes;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int V;
    double edge;
    if (!(std::cin >> V >> edge) || V <= 0 || edge <= 0.0) {
        std::cout << 0 << '\n';
        return 0;
    }

    std::vector<Vec3> pts;
    pts.reserve(static_cast<size_t>(V));
    Vec3 mn(std::numeric_limits<double>::infinity(),
            std::numeric_limits<double>::infinity(),
            std::numeric_limits<double>::infinity());
    Vec3 mx(-std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity(),
            -std::numeric_limits<double>::infinity());

    for (int i = 0; i < V; ++i) {
        Vec3 p;
        std::cin >> p.x >> p.y >> p.z;
        pts.push_back(p);
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }

    const double R = edge * std::sqrt(3.0 / 8.0);
    const double sep = 2.0 * R * 1.00001;
    const double dx = sep;
    const double dy = sep * std::sqrt(3.0) * 0.5;
    const double dz = sep * std::sqrt(2.0 / 3.0);
    const size_t maxOut = 250000;

    std::vector<Plane> planes = build_planes(pts, mn, mx);
    std::vector<Vec3> centers;
    centers.reserve(4096);

    const double lx = mx.x - mn.x, ly = mx.y - mn.y, lz = mx.z - mn.z;
    if (lx < 2 * R || ly < 2 * R || lz < 2 * R) {
        std::cout << 0 << '\n';
        return 0;
    }

    long long nx = static_cast<long long>(std::floor(lx / dx)) + 3;
    long long ny = static_cast<long long>(std::floor(ly / dy)) + 3;
    long long nz = static_cast<long long>(std::floor(lz / dz)) + 3;
    long double estimated = static_cast<long double>(nx) * static_cast<long double>(ny) * static_cast<long double>(nz);
    long long stride = 1;
    if (estimated > static_cast<long double>(maxOut) * 5.0L) {
        stride = static_cast<long long>(std::ceil(std::cbrt(static_cast<double>(estimated / static_cast<long double>(maxOut)))));
        if (stride < 1) stride = 1;
    }

    auto start = std::chrono::steady_clock::now();
    for (long long iz = 0; iz < nz && centers.size() < maxOut; iz += stride) {
        double z = mn.z + R + iz * dz;
        if (z > mx.z - R) break;
        for (long long iy = 0; iy < ny && centers.size() < maxOut; iy += stride) {
            double y = mn.y + R + iy * dy;
            if (y > mx.y - R) break;
            double rowOffset = ((iy + iz) & 1LL) ? 0.5 * dx : 0.0;
            for (long long ix = 0; ix < nx && centers.size() < maxOut; ix += stride) {
                double x = mn.x + R + ix * dx + rowOffset;
                if (x > mx.x - R) break;
                Vec3 p(x, y, z);
                if (inside_planes(planes, p, R)) centers.push_back(p);
            }
        }
        auto now = std::chrono::steady_clock::now();
        if (std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count() > 27000) break;
    }

    if (centers.empty()) {
        Vec3 c((mn.x + mx.x) * 0.5, (mn.y + mx.y) * 0.5, (mn.z + mx.z) * 0.5);
        if (inside_planes(planes, c, R)) centers.push_back(c);
    }

    std::cout << centers.size() << '\n';
    std::cout << std::setprecision(12);
    for (const Vec3& p : centers) {
        std::cout << p.x << ' ' << p.y << ' ' << p.z << " 0 0 0\n";
    }
    return 0;
}
