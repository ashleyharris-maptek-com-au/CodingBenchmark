using namespace std;

struct VecHash {
    size_t operator()(const vector<int>& v) const noexcept {
        uint64_t h = 1469598103934665603ull;
        for (int x : v) {
            uint64_t y = static_cast<uint64_t>(static_cast<int64_t>(x) + 1000003);
            h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
            h *= 1099511628211ull;
        }
        return static_cast<size_t>(h);
    }
};

struct Move {
    int axis;
    int dir;
};

static int D;
static long long maxTurns;
static vector<int> boundsV;
static vector<int> headPos;
static deque<vector<int>> snake;
static unordered_set<vector<int>, VecHash> occupied;
static unordered_set<vector<int>, VecHash> obstacles;
static unordered_set<vector<int>, VecHash> foods;

static bool inBounds(const vector<int>& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= boundsV[i]) return false;
    }
    return true;
}

static long long manhattan(const vector<int>& a, const vector<int>& b) {
    long long s = 0;
    for (int i = 0; i < D; ++i) s += llabs(static_cast<long long>(a[i]) - b[i]);
    return s;
}

static vector<int> stepPos(const vector<int>& p, Move m) {
    vector<int> q = p;
    q[m.axis] += m.dir;
    return q;
}

static bool blockedNow(const vector<int>& p) {
    return !inBounds(p) || obstacles.find(p) != obstacles.end() || occupied.find(p) != occupied.end();
}

static vector<int> nearestFood(const vector<int>& from) {
    vector<int> best;
    long long bd = numeric_limits<long long>::max();
    for (const auto& f : foods) {
        long long d = manhattan(from, f);
        if (d < bd) {
            bd = d;
            best = f;
        }
    }
    return best;
}

static vector<Move> directGreedyPath(const vector<int>& start, const vector<int>& target) {
    vector<Move> path;
    vector<int> cur = start;
    unordered_set<vector<int>, VecHash> tempOcc = occupied;
    deque<vector<int>> tempSnake = snake;

    while (cur != target && static_cast<int>(path.size()) < 100000) {
        vector<Move> candidates;
        for (int i = 0; i < D; ++i) {
            if (cur[i] < target[i]) candidates.push_back({i, 1});
            if (cur[i] > target[i]) candidates.push_back({i, -1});
        }
        sort(candidates.begin(), candidates.end(), [&](const Move& a, const Move& b) {
            vector<int> pa = stepPos(cur, a), pb = stepPos(cur, b);
            return manhattan(pa, target) < manhattan(pb, target);
        });

        bool moved = false;
        for (Move m : candidates) {
            vector<int> nxt = stepPos(cur, m);
            bool tailFree = !tempSnake.empty() && nxt == tempSnake.back() && foods.find(nxt) == foods.end();
            if (inBounds(nxt) && obstacles.find(nxt) == obstacles.end() &&
                (tailFree || tempOcc.find(nxt) == tempOcc.end())) {
                path.push_back(m);
                tempSnake.push_front(nxt);
                tempOcc.insert(nxt);
                if (foods.find(nxt) == foods.end() && !tempSnake.empty()) {
                    tempOcc.erase(tempSnake.back());
                    tempSnake.pop_back();
                }
                cur = nxt;
                moved = true;
                break;
            }
        }
        if (!moved) return {};
    }
    return cur == target ? path : vector<Move>{};
}

static vector<Move> astarPath(const vector<int>& start, const vector<int>& target, int expansionLimit) {
    struct Node {
        long long f, g;
        vector<int> p;
        bool operator<(const Node& other) const {
            if (f != other.f) return f > other.f;
            return g < other.g;
        }
    };

    priority_queue<Node> pq;
    unordered_map<vector<int>, long long, VecHash> dist;
    unordered_map<vector<int>, pair<vector<int>, Move>, VecHash> parent;
    dist[start] = 0;
    pq.push({manhattan(start, target), 0, start});
    int expanded = 0;

    while (!pq.empty() && expanded < expansionLimit) {
        Node cur = pq.top();
        pq.pop();
        auto it = dist.find(cur.p);
        if (it == dist.end() || it->second != cur.g) continue;
        if (cur.p == target) {
            vector<Move> path;
            vector<int> p = target;
            while (p != start) {
                auto pr = parent[p];
                path.push_back(pr.second);
                p = pr.first;
            }
            reverse(path.begin(), path.end());
            return path;
        }
        ++expanded;

        for (int axis = 0; axis < D; ++axis) {
            for (int dir : {-1, 1}) {
                Move mv{axis, dir};
                vector<int> nxt = stepPos(cur.p, mv);
                if (!inBounds(nxt) || obstacles.find(nxt) != obstacles.end()) continue;
                if (occupied.find(nxt) != occupied.end() && nxt != target) continue;
                long long ng = cur.g + 1;
                auto dit = dist.find(nxt);
                if (dit == dist.end() || ng < dit->second) {
                    dist[nxt] = ng;
                    parent[nxt] = {cur.p, mv};
                    pq.push({ng + manhattan(nxt, target), ng, nxt});
                }
            }
        }
    }
    return {};
}

static Move safeFallback(const vector<int>& target) {
    Move best{0, 1};
    long long bestScore = numeric_limits<long long>::min();
    for (int axis = 0; axis < D; ++axis) {
        for (int dir : {-1, 1}) {
            Move mv{axis, dir};
            vector<int> nxt = stepPos(headPos, mv);
            bool tailFree = !snake.empty() && nxt == snake.back() && foods.find(nxt) == foods.end();
            if (!inBounds(nxt) || obstacles.find(nxt) != obstacles.end() ||
                (!tailFree && occupied.find(nxt) != occupied.end())) {
                continue;
            }
            long long options = 0;
            for (int a = 0; a < D; ++a) {
                for (int d : {-1, 1}) {
                    vector<int> nn = nxt;
                    nn[a] += d;
                    if (inBounds(nn) && obstacles.find(nn) == obstacles.end() &&
                        occupied.find(nn) == occupied.end()) {
                        ++options;
                    }
                }
            }
            long long score = options * 1000000 - (foods.empty() ? 0 : manhattan(nxt, target));
            if (score > bestScore) {
                bestScore = score;
                best = mv;
            }
        }
    }
    return best;
}

static void applyMove(Move mv) {
    vector<int> nxt = stepPos(headPos, mv);
    bool eat = foods.find(nxt) != foods.end();
    snake.push_front(nxt);
    occupied.insert(nxt);
    headPos = nxt;
    if (eat) {
        foods.erase(nxt);
    } else if (!snake.empty()) {
        vector<int> tail = snake.back();
        snake.pop_back();
        bool stillPresent = false;
        for (const auto& p : snake) {
            if (p == tail) {
                stillPresent = true;
                break;
            }
        }
        if (!stillPresent) occupied.erase(tail);
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    if (!(cin >> D >> maxTurns)) return 0;
    boundsV.assign(D, 0);
    for (int i = 0; i < D; ++i) cin >> boundsV[i];

    int snakeLength;
    cin >> snakeLength;
    headPos.assign(D, 0);
    for (int i = 0; i < D; ++i) cin >> headPos[i];

    snake.clear();
    for (int i = 0; i < max(1, snakeLength); ++i) {
        snake.push_back(headPos);
    }
    occupied.insert(headPos);

    int F;
    cin >> F;
    for (int i = 0; i < F; ++i) {
        vector<int> p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        if (p != headPos) foods.insert(p);
    }

    int O;
    cin >> O;
    for (int i = 0; i < O; ++i) {
        vector<int> p(D);
        for (int j = 0; j < D; ++j) cin >> p[j];
        obstacles.insert(p);
    }

    vector<Move> output;
    output.reserve(static_cast<size_t>(min<long long>(maxTurns, 1000000)));

    for (long long turn = 0; turn < maxTurns && !foods.empty(); ++turn) {
        vector<int> target = nearestFood(headPos);
        vector<Move> plan;

        long long nearest = manhattan(headPos, target);
        int limit = 0;
        if (D <= 4) limit = 250000;
        else if (D <= 8) limit = 80000;
        else limit = 15000;

        if (nearest <= 250 || D <= 8) {
            plan = astarPath(headPos, target, limit);
        }
        if (plan.empty()) {
            plan = directGreedyPath(headPos, target);
        }

        Move mv = plan.empty() ? safeFallback(target) : plan.front();
        vector<int> nxt = stepPos(headPos, mv);
        bool tailFree = !snake.empty() && nxt == snake.back() && foods.find(nxt) == foods.end();
        if (!inBounds(nxt) || obstacles.find(nxt) != obstacles.end() ||
            (!tailFree && occupied.find(nxt) != occupied.end())) {
            mv = safeFallback(target);
        }

        output.push_back(mv);
        applyMove(mv);
    }

    for (const Move& mv : output) {
        cout << mv.axis << ' ' << mv.dir << '\n';
    }
    return 0;
}
