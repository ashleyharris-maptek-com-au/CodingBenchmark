using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    struct Candidate
    {
        public int X, Y, Z;
        public byte Data;
        public long Score;
    }

    sealed class FastInput
    {
        readonly Stream _s = Console.OpenStandardInput();
        readonly byte[] _buf = new byte[1 << 20];
        int _len, _pos;

        int Read()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len == 0) return -1;
            }
            return _buf[_pos++];
        }

        public string Token()
        {
            var sb = new StringBuilder();
            int c;
            do { c = Read(); } while (c >= 0 && c <= 32);
            while (c > 32)
            {
                sb.Append((char)c);
                c = Read();
            }
            return sb.ToString();
        }

        public int Byte()
        {
            return Read();
        }

        public void Skip(long n)
        {
            while (n > 0)
            {
                if (_pos < _len)
                {
                    int take = (int)Math.Min(n, _len - _pos);
                    _pos += take;
                    n -= take;
                }
                else
                {
                    int want = (int)Math.Min(n, _buf.Length);
                    int got = _s.Read(_buf, 0, want);
                    if (got <= 0) return;
                    n -= got;
                }
            }
        }
    }

    sealed class TopKeeper
    {
        readonly Candidate[] _heap;
        public int Count { get; private set; }

        public TopKeeper(int capacity)
        {
            _heap = new Candidate[Math.Max(1, capacity)];
        }

        static bool Less(Candidate a, Candidate b)
        {
            if (a.Score != b.Score) return a.Score < b.Score;
            int da = a.X + a.Y, db = b.X + b.Y;
            if (da != db) return da > db;
            if (a.X != b.X) return a.X > b.X;
            return a.Y > b.Y;
        }

        public void Add(Candidate c)
        {
            if (Count < _heap.Length)
            {
                int i = Count++;
                _heap[i] = c;
                Up(i);
            }
            else if (Less(_heap[0], c))
            {
                _heap[0] = c;
                Down(0);
            }
        }

        void Up(int i)
        {
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (!Less(_heap[i], _heap[p])) break;
                (_heap[i], _heap[p]) = (_heap[p], _heap[i]);
                i = p;
            }
        }

        void Down(int i)
        {
            for (;;)
            {
                int l = i * 2 + 1, r = l + 1, m = i;
                if (l < Count && Less(_heap[l], _heap[m])) m = l;
                if (r < Count && Less(_heap[r], _heap[m])) m = r;
                if (m == i) break;
                (_heap[i], _heap[m]) = (_heap[m], _heap[i]);
                i = m;
            }
        }

        public Candidate[] ToSortedArray()
        {
            var a = new Candidate[Count];
            Array.Copy(_heap, a, Count);
            Array.Sort(a, (u, v) =>
            {
                int c = v.Score.CompareTo(u.Score);
                if (c != 0) return c;
                c = (u.X + u.Y).CompareTo(v.X + v.Y);
                if (c != 0) return c;
                c = u.X.CompareTo(v.X);
                if (c != 0) return c;
                return u.Y.CompareTo(v.Y);
            });
            return a;
        }
    }

    public static void Main()
    {
        var input = new FastInput();
        string first = input.Token();
        if (first.Length == 0)
        {
            Console.WriteLine("0");
            return;
        }

        int xCount = int.Parse(first, CultureInfo.InvariantCulture);
        int yCount = int.Parse(input.Token(), CultureInfo.InvariantCulture);
        int zCount = int.Parse(input.Token(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.Token(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.Token(), CultureInfo.InvariantCulture);
        _ = double.Parse(input.Token(), CultureInfo.InvariantCulture);

        long columns = (long)xCount * yCount;
        long voxels = columns * zCount;
        if (xCount <= 0 || yCount <= 0 || zCount <= 0 || voxels <= 0)
        {
            Console.WriteLine("0");
            return;
        }

        int maxOps;
        if (columns <= 250_000) maxOps = 80_000;
        else if (columns <= 2_000_000) maxOps = 35_000;
        else maxOps = 8_000;
        maxOps = Math.Min(maxOps, (int)Math.Min(columns, int.MaxValue));

        var keeper = new TopKeeper(maxOps);

        for (int x = 0; x < xCount; x++)
        {
            for (int y = 0; y < yCount; y++)
            {
                if (zCount > 1) input.Skip(zCount - 1L);
                int d = input.Byte();
                if (d < 0) d = 0;

                int mass = d & 15;
                int value = d >> 4;
                if (mass == 0 || value == 0) continue;

                int dist = x + y;
                long fuel = dist == 0 ? 0L : mass * (long)(dist + 4);
                long revenue = mass * value * 1000L;
                long score = revenue - fuel;
                if (score <= 0) continue;

                keeper.Add(new Candidate
                {
                    X = x,
                    Y = y,
                    Z = zCount - 1,
                    Data = (byte)d,
                    Score = score
                });
            }
        }

        Candidate[] candidates = keeper.ToSortedArray();
        var sb = new StringBuilder(Math.Min(1 << 24, Math.Max(1024, candidates.Length * 64)));
        sb.Append(candidates.Length).Append('\n');

        foreach (var c in candidates)
        {
            AppendOperation(sb, c.X, c.Y, c.Z);
            if (sb.Length > (1 << 22))
            {
                Console.Write(sb.ToString());
                sb.Clear();
            }
        }

        Console.Write(sb.ToString());
    }

    static void AppendOperation(StringBuilder sb, int x, int y, int z)
    {
        int dist = x + y;
        int intermediate = Math.Max(0, dist - 1);
        sb.Append(intermediate).Append(' ')
          .Append(x).Append(' ')
          .Append(y).Append(' ')
          .Append(z);

        int cx = x, cy = y;
        while (cx > 0)
        {
            cx--;
            if (cx == 0 && cy == 0) break;
            sb.Append(' ').Append(cx).Append(' ').Append(cy);
        }
        while (cy > 0)
        {
            cy--;
            if (cx == 0 && cy == 0) break;
            sb.Append(' ').Append(cx).Append(' ').Append(cy);
        }

        sb.Append(" 0 0\n");
    }
}
