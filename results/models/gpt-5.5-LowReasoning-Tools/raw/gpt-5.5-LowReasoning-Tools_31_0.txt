use std::collections::{HashMap, HashSet};
use std::io::{self, Read, Write};

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return,
    };
    let target = match it.next().and_then(|s| s.parse::<i64>().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut nums = Vec::with_capacity(n);
    for idx in 0..n {
        if let Some(s) = it.next() {
            if let Ok(value) = s.parse::<i64>() {
                nums.push(Item { value, index: idx });
            }
        }
    }

    let ans = solve(&nums, target);
    let mut out = String::new();
    match ans {
        Some(mut indices) => {
            indices.sort_unstable();
            out.push_str("YES\n");
            for (i, idx) in indices.iter().enumerate() {
                if i > 0 {
                    out.push(' ');
                }
                out.push_str(&idx.to_string());
            }
            out.push('\n');
        }
        None => out.push_str("NO\n"),
    }
    io::stdout().write_all(out.as_bytes()).unwrap();
}

fn solve(items: &[Item], target: i64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    if items.is_empty() {
        return None;
    }
    for item in items {
        if item.value == target {
            return Some(vec![item.index]);
        }
    }

    let total = checked_sum(items)?;
    if total == target {
        return Some(items.iter().map(|x| x.index).collect());
    }

    let (neg, non_neg): (Vec<Item>, Vec<Item>) = items.iter().copied().partition(|x| x.value < 0);
    if neg.is_empty() {
        if target < 0 {
            return None;
        }
        return solve_nonnegative(&non_neg, target as u64);
    }
    if non_neg.is_empty() {
        if target > 0 {
            return None;
        }
        let flipped: Vec<Item> = neg
            .iter()
            .map(|x| Item {
                value: -x.value,
                index: x.index,
            })
            .collect();
        return solve_nonnegative(&flipped, (-target) as u64);
    }

    if items.len() <= 44 {
        return meet_in_middle(items, target);
    }
    sparse_dp_signed(items, target, 2_000_000)
}

fn checked_sum(items: &[Item]) -> Option<i64> {
    let mut sum = 0_i64;
    for item in items {
        sum = sum.checked_add(item.value)?;
    }
    Some(sum)
}

fn solve_nonnegative(items: &[Item], target: u64) -> Option<Vec<usize>> {
    if target == 0 {
        return Some(Vec::new());
    }
    let filtered: Vec<Item> = items
        .iter()
        .copied()
        .filter(|x| x.value > 0 && (x.value as u64) <= target)
        .collect();
    if filtered.is_empty() {
        return None;
    }

    let mut total = 0_u64;
    for item in &filtered {
        total = total.saturating_add(item.value as u64);
        if total >= target {
            break;
        }
    }
    if total < target {
        return None;
    }

    if filtered.len() <= 44 {
        return meet_in_middle(&filtered, target as i64);
    }

    let limit = target as usize;
    let memory_ok = limit <= 80_000_000;
    if memory_ok {
        return dense_dp_nonnegative(&filtered, limit);
    }

    sparse_dp_nonnegative(&filtered, target, 2_500_000)
}

fn dense_dp_nonnegative(items: &[Item], target: usize) -> Option<Vec<usize>> {
    let mut prev_sum = vec![usize::MAX; target + 1];
    let mut prev_idx = vec![usize::MAX; target + 1];
    prev_sum[0] = 0;

    for item in items {
        let w = item.value as usize;
        if w > target {
            continue;
        }
        let mut s = target - w + 1;
        while s > 0 {
            s -= 1;
            if prev_sum[s] != usize::MAX && prev_sum[s + w] == usize::MAX {
                prev_sum[s + w] = s;
                prev_idx[s + w] = item.index;
                if s + w == target {
                    return reconstruct_dense(&prev_sum, &prev_idx, target);
                }
            }
        }
    }
    reconstruct_dense(&prev_sum, &prev_idx, target)
}

fn reconstruct_dense(prev_sum: &[usize], prev_idx: &[usize], target: usize) -> Option<Vec<usize>> {
    if prev_sum[target] == usize::MAX {
        return None;
    }
    let mut out = Vec::new();
    let mut cur = target;
    while cur != 0 {
        out.push(prev_idx[cur]);
        cur = prev_sum[cur];
    }
    Some(out)
}

fn sparse_dp_nonnegative(items: &[Item], target: u64, cap: usize) -> Option<Vec<usize>> {
    let mut parent: HashMap<u64, (u64, usize)> = HashMap::with_capacity(1024);
    let mut sums = Vec::with_capacity(1024);
    parent.insert(0, (0, usize::MAX));
    sums.push(0_u64);

    for item in items {
        let w = item.value as u64;
        if w == 0 || w > target {
            continue;
        }
        let snapshot_len = sums.len();
        for i in 0..snapshot_len {
            let ns = sums[i] + w;
            if ns <= target && !parent.contains_key(&ns) {
                parent.insert(ns, (sums[i], item.index));
                sums.push(ns);
                if ns == target {
                    return reconstruct_sparse_u64(&parent, target);
                }
                if sums.len() > cap {
                    return None;
                }
            }
        }
    }
    reconstruct_sparse_u64(&parent, target)
}

fn reconstruct_sparse_u64(parent: &HashMap<u64, (u64, usize)>, target: u64) -> Option<Vec<usize>> {
    if !parent.contains_key(&target) {
        return None;
    }
    let mut cur = target;
    let mut out = Vec::new();
    while cur != 0 {
        let &(prev, idx) = parent.get(&cur)?;
        out.push(idx);
        cur = prev;
    }
    Some(out)
}

fn sparse_dp_signed(items: &[Item], target: i64, cap: usize) -> Option<Vec<usize>> {
    let mut parent: HashMap<i64, (i64, usize)> = HashMap::with_capacity(1024);
    let mut sums = Vec::with_capacity(1024);
    parent.insert(0, (0, usize::MAX));
    sums.push(0_i64);

    for item in items {
        if item.value == 0 {
            continue;
        }
        let snapshot_len = sums.len();
        for i in 0..snapshot_len {
            if let Some(ns) = sums[i].checked_add(item.value) {
                if !parent.contains_key(&ns) {
                    parent.insert(ns, (sums[i], item.index));
                    sums.push(ns);
                    if ns == target {
                        return reconstruct_sparse_i64(&parent, target);
                    }
                    if sums.len() > cap {
                        return None;
                    }
                }
            }
        }
    }
    reconstruct_sparse_i64(&parent, target)
}

fn reconstruct_sparse_i64(parent: &HashMap<i64, (i64, usize)>, target: i64) -> Option<Vec<usize>> {
    if !parent.contains_key(&target) {
        return None;
    }
    let mut cur = target;
    let mut out = Vec::new();
    while cur != 0 {
        let &(prev, idx) = parent.get(&cur)?;
        out.push(idx);
        cur = prev;
    }
    Some(out)
}

fn meet_in_middle(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let mid = items.len() / 2;
    let left = enum_subsets(&items[..mid]);
    let right = enum_subsets(&items[mid..]);

    let mut seen_right: HashMap<i64, u64> = HashMap::with_capacity(right.len());
    for (sum, mask) in right {
        seen_right.entry(sum).or_insert(mask);
    }

    for (lsum, lmask) in left {
        let need = target - lsum;
        if let Some(&rmask) = seen_right.get(&need) {
            let mut out = Vec::new();
            for i in 0..mid {
                if ((lmask >> i) & 1) != 0 {
                    out.push(items[i].index);
                }
            }
            for i in mid..items.len() {
                if ((rmask >> (i - mid)) & 1) != 0 {
                    out.push(items[i].index);
                }
            }
            return Some(out);
        }
    }
    None
}

fn enum_subsets(items: &[Item]) -> Vec<(i64, u64)> {
    let count = 1_usize << items.len();
    let mut out: Vec<(i64, u64)> = Vec::with_capacity(count);
    out.push((0, 0));
    for (pos, item) in items.iter().enumerate() {
        let bit = 1_u64 << pos;
        let old_len = out.len();
        for i in 0..old_len {
            if let Some(sum) = out[i].0.checked_add(item.value) {
                out.push((sum, out[i].1 | bit));
            }
        }
    }

    let mut unique = Vec::with_capacity(out.len());
    let mut seen = HashSet::with_capacity(out.len());
    for pair in out {
        if seen.insert(pair.0) {
            unique.push(pair);
        }
    }
    unique
}
