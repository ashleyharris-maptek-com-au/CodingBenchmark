using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    const double Day = 86400.0;
    const double G = 6.67430e-11;
    const double Tiny = 1e-12;

    struct V3
    {
        public double X, Y, Z;
        public V3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static V3 operator +(V3 a, V3 b) => new V3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static V3 operator -(V3 a, V3 b) => new V3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static V3 operator *(V3 a, double s) => new V3(a.X * s, a.Y * s, a.Z * s);
        public static V3 operator /(V3 a, double s) => new V3(a.X / s, a.Y / s, a.Z / s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public static double Dot(V3 a, V3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static V3 Cross(V3 a, V3 b) => new V3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
        public V3 Unit()
        {
            double n = Norm();
            return n > Tiny ? this / n : new V3(1, 0, 0);
        }
    }

    sealed class Body
    {
        public string Name = "";
        public double Mass;
        public V3 R;
        public V3 V;
    }

    sealed class State
    {
        public double T;
        public V3 EarthR, EarthV, AstR, AstV;
    }

    sealed class Candidate
    {
        public double Launch, Impact, Cost, MissRisk;
        public V3 DvLaunch, RelAtImpact;
    }

    static readonly CultureInfo Inv = CultureInfo.InvariantCulture;

    public static void Main()
    {
        try
        {
            string input = Console.In.ReadToEnd() ?? "";
            var tok = input.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            int p = 0;
            if (tok.Length < 7) { Fallback(); return; }

            int n = int.Parse(tok[p++], Inv);
            double warningDays = double.Parse(tok[p++], Inv);
            double impactorMass = double.Parse(tok[p++], Inv);
            double dvBudget = double.Parse(tok[p++], Inv);
            double asteroidMass = double.Parse(tok[p++], Inv);
            p++; // t0_unix_utc, not needed because the supplied state defines t=0.

            var bodies = new List<Body>();
            Body? earth = null, asteroid = null, sun = null;
            for (int i = 0; i < n && p + 7 < tok.Length; i++)
            {
                var b = new Body();
                b.Name = tok[p++];
                b.Mass = double.Parse(tok[p++], Inv);
                b.R = new V3(double.Parse(tok[p++], Inv), double.Parse(tok[p++], Inv), double.Parse(tok[p++], Inv));
                b.V = new V3(double.Parse(tok[p++], Inv), double.Parse(tok[p++], Inv), double.Parse(tok[p++], Inv));
                bodies.Add(b);
                if (string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase)) earth = b;
                if (string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase)) asteroid = b;
                if (string.Equals(b.Name, "Sun", StringComparison.OrdinalIgnoreCase)) sun = b;
            }

            if (earth == null || asteroid == null)
            {
                Fallback();
                return;
            }
            if (sun == null)
            {
                sun = new Body { Name = "Sun", Mass = 1.98847e30, R = new V3(0, 0, 0), V = new V3(0, 0, 0) };
                bodies.Insert(0, sun);
            }
            if (asteroid.Mass <= 0 && asteroidMass > 0) asteroid.Mass = asteroidMass;

            double impactEstimate = EstimateEarthImpactDays(earth, asteroid, warningDays);
            double latestImpact = Math.Max(5.0, Math.Min(warningDays * 0.98, impactEstimate * 0.92));
            var ephem = BuildEphemeris(CloneBodies(bodies), latestImpact, ChooseStepDays(warningDays));

            Candidate best = FindCandidate(ephem, warningDays, latestImpact, dvBudget, impactorMass, Math.Max(asteroidMass, asteroid.Mass));
            Emit(best, dvBudget);
        }
        catch
        {
            Fallback();
        }
    }

    static List<Body> CloneBodies(List<Body> src)
    {
        var r = new List<Body>(src.Count);
        foreach (var b in src) r.Add(new Body { Name = b.Name, Mass = b.Mass, R = b.R, V = b.V });
        return r;
    }

    static double ChooseStepDays(double warningDays)
    {
        if (warningDays <= 900) return 0.25;
        if (warningDays <= 3000) return 0.75;
        if (warningDays <= 9000) return 2.0;
        return 4.0;
    }

    static List<State> BuildEphemeris(List<Body> bodies, double maxDays, double stepDays)
    {
        var states = new List<State>();
        double dt = stepDays * Day;
        double t = 0;
        AddState(states, bodies, t);
        int steps = Math.Max(1, (int)Math.Ceiling(maxDays / stepDays));
        for (int s = 0; s < steps; s++)
        {
            double actual = Math.Min(dt, (maxDays - t) * Day);
            if (actual <= 0) break;
            KdkStep(bodies, actual);
            t += actual / Day;
            AddState(states, bodies, t);
        }
        return states;
    }

    static void AddState(List<State> states, List<Body> bodies, double t)
    {
        Body? e = null, a = null;
        foreach (var b in bodies)
        {
            if (string.Equals(b.Name, "Earth", StringComparison.OrdinalIgnoreCase)) e = b;
            else if (string.Equals(b.Name, "Asteroid", StringComparison.OrdinalIgnoreCase)) a = b;
        }
        if (e != null && a != null) states.Add(new State { T = t, EarthR = e.R, EarthV = e.V, AstR = a.R, AstV = a.V });
    }

    static void KdkStep(List<Body> b, double dt)
    {
        V3[] acc = Accelerations(b);
        for (int i = 0; i < b.Count; i++) b[i].V += acc[i] * (0.5 * dt);
        for (int i = 0; i < b.Count; i++) b[i].R += b[i].V * dt;
        acc = Accelerations(b);
        for (int i = 0; i < b.Count; i++) b[i].V += acc[i] * (0.5 * dt);
    }

    static V3[] Accelerations(List<Body> b)
    {
        int n = b.Count;
        var a = new V3[n];
        for (int i = 0; i < n; i++)
        {
            V3 sum = new V3();
            for (int j = 0; j < n; j++)
            {
                if (i == j || b[j].Mass <= 0) continue;
                V3 d = b[j].R - b[i].R;
                double r2 = Math.Max(1.0, V3.Dot(d, d));
                double invR = 1.0 / Math.Sqrt(r2);
                sum += d * (G * b[j].Mass * invR * invR * invR);
            }
            a[i] = sum;
        }
        return a;
    }

    static State Sample(List<State> e, double t)
    {
        if (t <= e[0].T) return e[0];
        if (t >= e[e.Count - 1].T) return e[e.Count - 1];
        int lo = 0, hi = e.Count - 1;
        while (hi - lo > 1)
        {
            int mid = (lo + hi) >> 1;
            if (e[mid].T <= t) lo = mid; else hi = mid;
        }
        double f = (t - e[lo].T) / Math.Max(Tiny, e[hi].T - e[lo].T);
        return new State
        {
            T = t,
            EarthR = Lerp(e[lo].EarthR, e[hi].EarthR, f),
            EarthV = Lerp(e[lo].EarthV, e[hi].EarthV, f),
            AstR = Lerp(e[lo].AstR, e[hi].AstR, f),
            AstV = Lerp(e[lo].AstV, e[hi].AstV, f)
        };
    }

    static V3 Lerp(V3 a, V3 b, double f) => a * (1.0 - f) + b * f;

    static Candidate FindCandidate(List<State> eph, double warning, double latestImpact, double budget, double impactorMass, double asteroidMass)
    {
        Candidate? best = null;
        double launchMax = Math.Min(warning, Math.Max(0.0, latestImpact - 2.0));
        int launchSamples = warning > 5000 ? 44 : 56;
        int impactSamples = warning > 5000 ? 90 : 72;

        for (int li = 0; li < launchSamples; li++)
        {
            double lf = launchSamples == 1 ? 0 : (double)li / (launchSamples - 1);
            double launch = launchMax * lf * lf; // favor early launches but keep late options.
            State sl = Sample(eph, launch);
            double minTof = Math.Max(2.0, warning < 1000 ? 20.0 : 60.0);
            for (int ii = 0; ii < impactSamples; ii++)
            {
                double f = impactSamples == 1 ? 1.0 : (double)ii / (impactSamples - 1);
                double impact = launch + minTof + (latestImpact - launch - minTof) * f;
                if (impact <= launch + 1.0 || impact > latestImpact) continue;
                State si = Sample(eph, impact);
                Candidate c = Evaluate(sl, si, launch, impact, budget, impactorMass, asteroidMass);
                if (best == null || c.Cost + c.MissRisk < best.Cost + best.MissRisk) best = c;
            }
        }

        if (best == null)
        {
            State s0 = Sample(eph, 0);
            State si = Sample(eph, Math.Max(1.0, latestImpact * 0.5));
            best = Evaluate(s0, si, 0, si.T, budget, impactorMass, asteroidMass);
        }
        return best;
    }

    static Candidate Evaluate(State launchState, State impactState, double launch, double impact, double budget, double impactorMass, double asteroidMass)
    {
        double tof = Math.Max(Day, (impact - launch) * Day);
        V3 chordVel = (impactState.AstR - launchState.EarthR) / tof;

        // A light two-body correction around the Sun improves the straight-line Lambert guess
        // without making the program depend on external SPICE bindings.
        V3 sunR = new V3(0, 0, 0);
        double muSun = G * 1.98847e30;
        V3 mid = (launchState.EarthR + impactState.AstR) * 0.5;
        V3 solarSag = (sunR - mid) * (muSun / Math.Pow(Math.Max(1.0, mid.Norm()), 3.0)) * (0.5 * tof);
        V3 required = chordVel - solarSag * 0.35;
        V3 dv = required - launchState.EarthV;
        double mag = dv.Norm();
        if (mag > budget && mag > 0) dv = dv * (budget / mag);

        V3 arrivalVelocity = launchState.EarthV + dv;
        V3 rel = arrivalVelocity - impactState.AstV;
        double momentumScore = impactorMass * Math.Max(0.0, rel.Norm()) / Math.Max(1.0, asteroidMass);
        double timeScore = Math.Sqrt(Math.Max(1.0, impact));
        double missRisk = 1.0e6 / Math.Max(1.0, momentumScore * timeScore);

        return new Candidate
        {
            Launch = launch,
            Impact = impact,
            DvLaunch = dv,
            RelAtImpact = rel,
            Cost = Math.Abs(budget - dv.Norm()) * 0.02 - rel.Norm() * 0.001,
            MissRisk = missRisk
        };
    }

    static double EstimateEarthImpactDays(Body earth, Body ast, double warning)
    {
        V3 dr = ast.R - earth.R;
        V3 dv = ast.V - earth.V;
        double vv = Math.Max(Tiny, V3.Dot(dv, dv));
        double t = -V3.Dot(dr, dv) / vv / Day;
        if (double.IsNaN(t) || double.IsInfinity(t) || t < 1 || t > warning * 1.5) return warning;
        return t;
    }

    static void Emit(Candidate c, double budget)
    {
        // The final entry is a terminal zero correction at impact. It tells the simulator the
        // intended impact time while preserving all available kinetic energy in the spacecraft.
        double launch = Clamp(c.Launch, 0, Math.Max(0, c.Impact - 1e-6));
        double impact = Math.Max(launch + 1e-6, c.Impact);
        V3 dv = Limit(c.DvLaunch, budget);
        Console.WriteLine("2");
        Console.WriteLine(Format(launch) + " " + Format(dv.X) + " " + Format(dv.Y) + " " + Format(dv.Z));
        Console.WriteLine(Format(impact) + " 0 0 0");
    }

    static V3 Limit(V3 v, double max)
    {
        double n = v.Norm();
        if (max >= 0 && n > max && n > 0) return v * (max / n);
        return v;
    }

    static double Clamp(double x, double lo, double hi) => Math.Max(lo, Math.Min(hi, x));
    static string Format(double x) => x.ToString("G17", Inv);

    static void Fallback()
    {
        Console.WriteLine("2");
        Console.WriteLine("0 0 0 0");
        Console.WriteLine("1 0 0 0");
    }
}
