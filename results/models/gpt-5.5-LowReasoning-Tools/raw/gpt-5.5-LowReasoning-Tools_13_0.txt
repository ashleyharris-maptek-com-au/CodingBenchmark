#include <algorithm>
#include <iostream>
#include <string>
#include <vector>

class SuffixAutomaton {
    struct State {
        int link = -1;
        int len = 0;
        int head = -1;
    };

    struct Edge {
        int to = 0;
        int next = -1;
        unsigned char ch = 0;
    };

    std::vector<State> st;
    std::vector<Edge> edges;
    int last = 0;

    int find_edge_index(int v, unsigned char c) const {
        for (int e = st[v].head; e != -1; e = edges[e].next) {
            if (edges[e].ch == c) return e;
        }
        return -1;
    }

public:
    explicit SuffixAutomaton(const std::string& s) {
        st.reserve(std::max<size_t>(1, 2 * s.size()));
        edges.reserve(std::max<size_t>(1, 3 * s.size()));
        st.push_back(State{});
        for (unsigned char c : s) extend(c);
    }

    void add_edge(int from, unsigned char c, int to) {
        edges.push_back(Edge{to, st[from].head, c});
        st[from].head = static_cast<int>(edges.size()) - 1;
    }

    void set_transition(int from, unsigned char c, int to) {
        int e = find_edge_index(from, c);
        if (e == -1) add_edge(from, c, to);
        else edges[e].to = to;
    }

    int transition(int from, unsigned char c) const {
        int e = find_edge_index(from, c);
        return e == -1 ? -1 : edges[e].to;
    }

    void copy_transitions(int from, int to) {
        std::vector<Edge> copied;
        for (int e = st[from].head; e != -1; e = edges[e].next) copied.push_back(edges[e]);
        for (const Edge& e : copied) add_edge(to, e.ch, e.to);
    }

    void extend(unsigned char c) {
        int cur = static_cast<int>(st.size());
        st.push_back(State{});
        st[cur].len = st[last].len + 1;

        int p = last;
        while (p != -1 && transition(p, c) == -1) {
            add_edge(p, c, cur);
            p = st[p].link;
        }

        if (p == -1) {
            st[cur].link = 0;
        } else {
            int q = transition(p, c);
            if (st[p].len + 1 == st[q].len) {
                st[cur].link = q;
            } else {
                int clone = static_cast<int>(st.size());
                st.push_back(State{});
                st[clone].len = st[p].len + 1;
                st[clone].link = st[q].link;
                copy_transitions(q, clone);

                while (p != -1 && transition(p, c) == q) {
                    set_transition(p, c, clone);
                    p = st[p].link;
                }
                st[q].link = st[cur].link = clone;
            }
        }
        last = cur;
    }

    int size() const { return static_cast<int>(st.size()); }
    int length(int v) const { return st[v].len; }
    int suffix_link(int v) const { return st[v].link; }
};

static std::vector<int> states_by_length(const SuffixAutomaton& sam) {
    int n = sam.size();
    int max_len = 0;
    for (int i = 0; i < n; ++i) max_len = std::max(max_len, sam.length(i));

    std::vector<int> count(max_len + 1, 0), order(n);
    for (int i = 0; i < n; ++i) ++count[sam.length(i)];
    for (int i = 1; i <= max_len; ++i) count[i] += count[i - 1];
    for (int i = n - 1; i >= 0; --i) order[--count[sam.length(i)]] = i;
    return order;
}

static std::string brute_force_small(const std::vector<std::string>& strings) {
    size_t base_index = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].size() < strings[base_index].size()) base_index = i;
    }

    const std::string& base = strings[base_index];
    for (size_t len = base.size(); len > 0; --len) {
        for (size_t start = 0; start + len <= base.size(); ++start) {
            std::string candidate = base.substr(start, len);
            bool ok = true;
            for (size_t i = 0; i < strings.size(); ++i) {
                if (i != base_index && strings[i].find(candidate) == std::string::npos) {
                    ok = false;
                    break;
                }
            }
            if (ok) return candidate;
        }
    }
    return "";
}

static std::string longest_common_substring(std::vector<std::string>& strings) {
    if (strings.empty()) return "";
    if (strings.size() == 1) return strings[0];

    size_t total_size = 0;
    size_t shortest_size = strings[0].size();
    for (const std::string& s : strings) {
        total_size += s.size();
        shortest_size = std::min(shortest_size, s.size());
    }

    // Tiny cases are faster with direct library substring search and avoid index setup overhead.
    if (shortest_size <= 256 && total_size <= 4096) {
        return brute_force_small(strings);
    }

    size_t base_index = 0;
    for (size_t i = 1; i < strings.size(); ++i) {
        if (strings[i].size() < strings[base_index].size()) base_index = i;
    }
    if (strings[base_index].empty()) return "";
    std::swap(strings[0], strings[base_index]);

    const std::string& base = strings[0];
    SuffixAutomaton sam(base);
    const int states = sam.size();

    std::vector<int> common(states), match(states), order = states_by_length(sam);
    for (int v = 0; v < states; ++v) common[v] = sam.length(v);

    for (size_t si = 1; si < strings.size(); ++si) {
        std::fill(match.begin(), match.end(), 0);
        int v = 0;
        int l = 0;

        for (unsigned char c : strings[si]) {
            int to = sam.transition(v, c);
            if (to != -1) {
                v = to;
                ++l;
            } else {
                while (v != -1 && sam.transition(v, c) == -1) v = sam.suffix_link(v);
                if (v == -1) {
                    v = 0;
                    l = 0;
                    continue;
                }
                l = sam.length(v) + 1;
                v = sam.transition(v, c);
            }
            if (l > match[v]) match[v] = l;
        }

        for (int i = states - 1; i > 0; --i) {
            int state = order[i];
            int parent = sam.suffix_link(state);
            if (parent >= 0) {
                int inherited = std::min(match[state], sam.length(parent));
                if (inherited > match[parent]) match[parent] = inherited;
            }
        }

        int best_possible = 0;
        for (int state = 1; state < states; ++state) {
            if (match[state] < common[state]) common[state] = match[state];
            if (common[state] > best_possible) best_possible = common[state];
        }
        if (best_possible == 0) return "";
    }

    int best_len = 0;
    for (int v = 1; v < states; ++v) best_len = std::max(best_len, common[v]);
    if (best_len == 0) return "";

    // Recover the earliest base-string occurrence of any substring of best_len that survived all strings.
    int v = 0;
    int l = 0;
    size_t best_pos = std::string::npos;
    for (size_t i = 0; i < base.size(); ++i) {
        v = sam.transition(v, static_cast<unsigned char>(base[i]));
        ++l;
        while (v != 0 && sam.length(sam.suffix_link(v)) >= best_len) {
            v = sam.suffix_link(v);
            l = sam.length(v);
        }
        if (l >= best_len && common[v] >= best_len) {
            best_pos = i + 1 - best_len;
            break;
        }
    }

    if (best_pos == std::string::npos) return "";
    return base.substr(best_pos, static_cast<size_t>(best_len));
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n = 0;
    if (!(std::cin >> n)) return 0;
    std::string line;
    std::getline(std::cin, line);

    std::vector<std::string> strings;
    strings.reserve(std::max(0, n));
    for (int i = 0; i < n; ++i) {
        std::getline(std::cin, line);
        strings.push_back(line);
    }

    std::cout << longest_common_substring(strings) << '\n';
    return 0;
}
