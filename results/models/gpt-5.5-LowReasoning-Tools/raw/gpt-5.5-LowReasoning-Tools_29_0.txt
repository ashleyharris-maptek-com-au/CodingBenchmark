using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Text;

public static class Program
{
    private const int ExactVertexLimit = 5000;
    private const long ExactEdgeLimit = 8000000;
    private static int N;
    private static int M;
    private static int[][] Adj = Array.Empty<int[]>();
    private static int[] Degree = Array.Empty<int>();
    private static int[] Best = Array.Empty<int>();
    private static readonly Stopwatch Watch = new Stopwatch();
    private static long TimeLimitMs = 285000;

    public static void Main()
    {
        var fs = Console.OpenStandardInput();
        var scanner = new FastScanner(fs);
        N = scanner.NextInt();
        if (N < 0) return;
        M = scanner.NextInt();
        if (M < 0) M = 0;

        var lists = new List<int>[N];
        for (int i = 0; i < N; i++) lists[i] = new List<int>();

        for (int i = 0; i < M; i++)
        {
            int u = scanner.NextInt();
            int v = scanner.NextInt();
            if ((uint)u >= (uint)N || (uint)v >= (uint)N || u == v) continue;
            lists[u].Add(v);
            lists[v].Add(u);
        }

        Adj = new int[N][];
        Degree = new int[N];
        long uniqueTwice = 0;
        for (int i = 0; i < N; i++)
        {
            var a = lists[i];
            a.Sort();
            int w = 0, last = -1;
            for (int j = 0; j < a.Count; j++)
            {
                int x = a[j];
                if (x != last)
                {
                    a[w++] = x;
                    last = x;
                }
            }
            if (w < a.Count) a.RemoveRange(w, a.Count - w);
            Adj[i] = a.ToArray();
            Degree[i] = Adj[i].Length;
            uniqueTwice += Degree[i];
        }
        M = (int)Math.Min(int.MaxValue, uniqueTwice / 2);
        Watch.Start();

        Best = GreedyClique();

        if (N == 0)
        {
            Print(Array.Empty<int>());
            return;
        }

        if (M == 0)
        {
            Print(new[] { 0 });
            return;
        }

        long completeEdges = (long)N * (N - 1) / 2;
        if (completeEdges == M)
        {
            var all = new int[N];
            for (int i = 0; i < N; i++) all[i] = i;
            Print(all);
            return;
        }

        if (N <= ExactVertexLimit && M <= ExactEdgeLimit)
        {
            RunExact();
        }
        else
        {
            ImproveLargeGraphHeuristic();
        }

        Array.Sort(Best);
        Print(Best);
    }

    private static void RunExact()
    {
        int[] vertices = Enumerable.Range(0, N).Where(v => Degree[v] >= Best.Length - 1).ToArray();
        Array.Sort(vertices, (a, b) =>
        {
            int c = Degree[b].CompareTo(Degree[a]);
            return c != 0 ? c : a.CompareTo(b);
        });
        Expand(Array.Empty<int>(), vertices);
    }

    private static void Expand(int[] clique, int[] cand)
    {
        if (cand.Length == 0)
        {
            SaveBest(clique);
            return;
        }
        if (Watch.ElapsedMilliseconds > TimeLimitMs) return;
        if (clique.Length + cand.Length <= Best.Length) return;

        int[] order, colors;
        ColorSort(cand, out order, out colors);

        for (int i = order.Length - 1; i >= 0; i--)
        {
            if (clique.Length + colors[i] <= Best.Length) return;
            if (Watch.ElapsedMilliseconds > TimeLimitMs) return;

            int v = order[i];
            var nextClique = new int[clique.Length + 1];
            Array.Copy(clique, nextClique, clique.Length);
            nextClique[clique.Length] = v;
            if (nextClique.Length > Best.Length) SaveBest(nextClique);

            var next = IntersectPrefixWithNeighbors(order, i, v);
            if (next.Length > 0 && nextClique.Length + next.Length > Best.Length)
                Expand(nextClique, next);
        }
    }

    private static void ColorSort(int[] cand, out int[] order, out int[] colors)
    {
        int n = cand.Length;
        order = new int[n];
        colors = new int[n];
        var uncolored = new List<int>(cand);
        int pos = 0, color = 0;

        while (uncolored.Count > 0)
        {
            color++;
            var stable = new List<int>();
            for (int i = 0; i < uncolored.Count; i++)
            {
                int v = uncolored[i];
                bool ok = true;
                for (int j = 0; j < stable.Count; j++)
                {
                    if (AreAdjacent(v, stable[j]))
                    {
                        ok = false;
                        break;
                    }
                }
                if (ok)
                {
                    stable.Add(v);
                    uncolored.RemoveAt(i);
                    i--;
                    order[pos] = v;
                    colors[pos] = color;
                    pos++;
                }
            }
        }
    }

    private static int[] IntersectPrefixWithNeighbors(int[] order, int endExclusive, int v)
    {
        var res = new List<int>();
        for (int i = 0; i < endExclusive; i++)
        {
            int u = order[i];
            if (Degree[u] >= Best.Length && AreAdjacent(u, v))
                res.Add(u);
        }
        return res.ToArray();
    }

    private static int[] GreedyClique()
    {
        if (N == 0) return Array.Empty<int>();
        int attempts = Math.Min(N, 256);
        int[] starts = Enumerable.Range(0, N).OrderByDescending(v => Degree[v]).Take(attempts).ToArray();
        int[] best = new[] { starts.Length == 0 ? 0 : starts[0] };

        foreach (int s in starts)
        {
            var clique = new List<int> { s };
            var candidates = Adj[s].ToList();
            candidates.Sort((a, b) => Degree[b].CompareTo(Degree[a]));

            while (candidates.Count > 0)
            {
                int v = candidates[0];
                clique.Add(v);
                var next = new List<int>();
                for (int i = 1; i < candidates.Count; i++)
                {
                    int u = candidates[i];
                    if (AreAdjacent(u, v)) next.Add(u);
                }
                candidates = next;
            }
            if (clique.Count > best.Length) best = clique.ToArray();
            if (Watch.ElapsedMilliseconds > 10000) break;
        }
        return best;
    }

    private static void ImproveLargeGraphHeuristic()
    {
        int limit = Math.Min(N, 2048);
        int[] top = Enumerable.Range(0, N).OrderByDescending(v => Degree[v]).Take(limit).ToArray();
        var map = new Dictionary<int, int>(limit);
        for (int i = 0; i < top.Length; i++) map[top[i]] = i;

        int[] bestLocal = Best;
        foreach (int start in top)
        {
            if (Watch.ElapsedMilliseconds > TimeLimitMs) break;
            var c = new List<int> { start };
            var cand = new List<int>();
            foreach (int u in Adj[start])
                if (map.ContainsKey(u)) cand.Add(u);
            cand.Sort((a, b) => Degree[b].CompareTo(Degree[a]));
            while (cand.Count > 0)
            {
                int v = cand[0];
                c.Add(v);
                var next = new List<int>();
                for (int i = 1; i < cand.Count; i++)
                    if (AreAdjacent(cand[i], v)) next.Add(cand[i]);
                cand = next;
            }
            if (c.Count > bestLocal.Length) bestLocal = c.ToArray();
        }
        SaveBest(bestLocal);
    }

    private static bool AreAdjacent(int a, int b)
    {
        var arr = Adj[a];
        int lo = 0, hi = arr.Length - 1;
        while (lo <= hi)
        {
            int mid = (lo + hi) >> 1;
            int x = arr[mid];
            if (x == b) return true;
            if (x < b) lo = mid + 1;
            else hi = mid - 1;
        }
        return false;
    }

    private static void SaveBest(int[] clique)
    {
        if (clique.Length > Best.Length)
        {
            var copy = new int[clique.Length];
            Array.Copy(clique, copy, clique.Length);
            Best = copy;
        }
    }

    private static void Print(int[] clique)
    {
        var sb = new StringBuilder();
        sb.AppendLine(clique.Length.ToString());
        for (int i = 0; i < clique.Length; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(clique[i]);
        }
        sb.AppendLine();
        Console.Write(sb.ToString());
    }

    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 20];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return -1;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            int val = 0;
            while (c > 32)
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
