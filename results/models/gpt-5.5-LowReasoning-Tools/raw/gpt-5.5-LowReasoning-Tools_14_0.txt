using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    struct Constraint
    {
        public int Need;
        public int[] Vars;
        public HashSet<int>? Set;
    }

    static readonly int[] DX = { -1, 0, 1, -1, 1, -1, 0, 1 };
    static readonly int[] DY = { -1, -1, -1, 0, 0, 1, 1, 1 };

    static int H, W;
    static string[] Board = Array.Empty<string>();
    static readonly HashSet<long> Safe = new HashSet<long>();

    static long Key(int x, int y) => ((long)y << 32) | (uint)x;
    static bool In(int x, int y) => (uint)x < (uint)W && (uint)y < (uint)H;
    static bool IsDigit(char c) => c >= '0' && c <= '8';

    public static void Main()
    {
        var first = Console.ReadLine()!.TrimStart('\uFEFF').Split(' ', StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(first[0]);
        W = int.Parse(first[1]);
        Board = new string[H];
        for (int y = 0; y < H; y++)
        {
            string? s = Console.ReadLine();
            Board[y] = s ?? "";
            if (Board[y].Length < W) Board[y] = Board[y].PadRight(W, ' ');
        }

        var constraints = BuildConstraintsAndDoLocal();

        // Adaptive escalation: subset logic is cheap because every Minesweeper clue has at most 8 vars.
        if (constraints.Count > 1)
            PairwiseSubsetDeductions(constraints);

        // Exact CSP is valuable on small/medium tangled frontiers, but deliberately bounded for huge boards.
        int frontierVars = CountDistinctVars(constraints);
        if (constraints.Count <= 200000 && frontierVars <= 200000)
            ExactSmallComponents(constraints);

        var sb = new StringBuilder();
        sb.AppendLine(Safe.Count.ToString());
        foreach (long k in Safe)
        {
            int x = (int)(uint)k;
            int y = (int)(k >> 32);
            sb.Append(x).Append(' ').Append(y).AppendLine();
        }
        Console.Write(sb.ToString());
    }

    static List<Constraint> BuildConstraintsAndDoLocal()
    {
        var idOf = new Dictionary<long, int>();
        var constraints = new List<Constraint>();

        for (int y = 0; y < H; y++)
        {
            string row = Board[y];
            long[] unkKeys = new long[8];
            for (int x = 0; x < W; x++)
            {
                char c = row[x];
                if (!IsDigit(c)) continue;

                int need = c - '0';
                int flags = 0;
                int unkCount = 0;

                for (int d = 0; d < 8; d++)
                {
                    int nx = x + DX[d], ny = y + DY[d];
                    if (!In(nx, ny)) continue;
                    char nc = Board[ny][nx];
                    if (nc == '*') flags++;
                    else if (nc == ' ') unkKeys[unkCount++] = Key(nx, ny);
                }

                int rem = need - flags;
                if (unkCount == 0 || rem < 0 || rem > unkCount) continue;

                if (rem == 0)
                {
                    for (int i = 0; i < unkCount; i++) Safe.Add(unkKeys[i]);
                    continue;
                }

                var vars = new int[unkCount];
                for (int i = 0; i < unkCount; i++)
                {
                    long k = unkKeys[i];
                    if (!idOf.TryGetValue(k, out int id))
                    {
                        id = idOf.Count;
                        idOf.Add(k, id);
                    }
                    vars[i] = id;
                }
                constraints.Add(new Constraint { Need = rem, Vars = vars });
            }
        }

        IdToKey = new long[idOf.Count];
        foreach (var kv in idOf) IdToKey[kv.Value] = kv.Key;
        return constraints;
    }

    static long[] IdToKey = Array.Empty<long>();

    static int CountDistinctVars(List<Constraint> constraints) => IdToKey.Length;

    static void PairwiseSubsetDeductions(List<Constraint> constraints)
    {
        for (int i = 0; i < constraints.Count; i++)
            constraints[i] = WithSet(constraints[i]);

        const int maxComparisons = 30000000;
        long comparisons = 0;
        for (int i = 0; i < constraints.Count; i++)
        {
            var a = constraints[i];
            for (int j = i + 1; j < constraints.Count; j++)
            {
                if (++comparisons > maxComparisons) return;
                var b = constraints[j];
                TrySubset(a, b);
                TrySubset(b, a);
            }
        }
    }

    static Constraint WithSet(Constraint c)
    {
        c.Set = new HashSet<int>(c.Vars);
        return c;
    }

    static void TrySubset(Constraint small, Constraint big)
    {
        if (small.Vars.Length >= big.Vars.Length || small.Set == null || big.Set == null) return;
        foreach (int v in small.Vars)
            if (!big.Set.Contains(v)) return;

        int diffNeed = big.Need - small.Need;
        int diffCount = big.Vars.Length - small.Vars.Length;

        if (diffNeed == 0)
        {
            foreach (int v in big.Vars)
                if (!small.Set.Contains(v)) Safe.Add(IdToKey[v]);
        }
        // If diffNeed == diffCount then the difference is all mines. That can help future play,
        // but the requested output is only safe cells, so no output action is needed here.
    }

    static void ExactSmallComponents(List<Constraint> constraints)
    {
        var byVar = new List<int>[IdToKey.Length];
        for (int i = 0; i < byVar.Length; i++) byVar[i] = new List<int>();
        for (int ci = 0; ci < constraints.Count; ci++)
            foreach (int v in constraints[ci].Vars)
                byVar[v].Add(ci);

        var seenVar = new bool[IdToKey.Length];
        var seenCon = new bool[constraints.Count];
        var qv = new Queue<int>();
        var qc = new Queue<int>();

        for (int start = 0; start < IdToKey.Length; start++)
        {
            if (seenVar[start]) continue;
            var vars = new List<int>();
            var cons = new List<int>();
            seenVar[start] = true;
            qv.Enqueue(start);

            while (qv.Count > 0 || qc.Count > 0)
            {
                while (qv.Count > 0)
                {
                    int v = qv.Dequeue();
                    vars.Add(v);
                    foreach (int ci in byVar[v])
                        if (!seenCon[ci]) { seenCon[ci] = true; qc.Enqueue(ci); }
                }
                while (qc.Count > 0)
                {
                    int ci = qc.Dequeue();
                    cons.Add(ci);
                    foreach (int v in constraints[ci].Vars)
                        if (!seenVar[v]) { seenVar[v] = true; qv.Enqueue(v); }
                }
            }

            if (vars.Count <= 24 && cons.Count <= 80)
                SolveComponent(vars, cons, constraints);
        }
    }

    static void SolveComponent(List<int> vars, List<int> cons, List<Constraint> all)
    {
        int n = vars.Count, m = cons.Count;
        var local = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) local[vars[i]] = i;

        var cVars = new int[m][];
        var need = new int[m];
        var rem = new int[m];
        var cnt = new int[m];
        var varCons = new List<int>[n];
        for (int i = 0; i < n; i++) varCons[i] = new List<int>();

        for (int i = 0; i < m; i++)
        {
            var c = all[cons[i]];
            need[i] = c.Need;
            rem[i] = c.Vars.Length;
            cVars[i] = new int[c.Vars.Length];
            for (int j = 0; j < c.Vars.Length; j++)
            {
                int lv = local[c.Vars[j]];
                cVars[i][j] = lv;
                varCons[lv].Add(i);
            }
        }

        long solutions = 0;
        var mineInSolutions = new long[n];
        var assign = new int[n];
        Array.Fill(assign, -1);

        void Dfs(int p)
        {
            if (solutions > 2000000) return;
            if (p == n)
            {
                for (int i = 0; i < m; i++) if (cnt[i] != need[i]) return;
                solutions++;
                for (int i = 0; i < n; i++) if (assign[i] == 1) mineInSolutions[i]++;
                return;
            }

            for (int val = 0; val <= 1; val++)
            {
                bool ok = true;
                assign[p] = val;
                foreach (int ci in varCons[p])
                {
                    rem[ci]--;
                    cnt[ci] += val;
                    if (cnt[ci] > need[ci] || cnt[ci] + rem[ci] < need[ci]) ok = false;
                }
                if (ok) Dfs(p + 1);
                foreach (int ci in varCons[p])
                {
                    cnt[ci] -= val;
                    rem[ci]++;
                }
                assign[p] = -1;
            }
        }

        Dfs(0);
        if (solutions == 0 || solutions > 2000000) return;
        for (int i = 0; i < n; i++)
            if (mineInSolutions[i] == 0)
                Safe.Add(IdToKey[vars[i]]);
    }
}
