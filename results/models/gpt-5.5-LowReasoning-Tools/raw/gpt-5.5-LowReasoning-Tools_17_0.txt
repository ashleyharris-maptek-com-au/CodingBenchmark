using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    struct Box
    {
        public int Index;
        public long W, H, D, V;
        public long MaxSide;
    }

    struct Space
    {
        public long X, Y, Z, W, H, D;
        public long V => W * H * D;
    }

    struct Placement
    {
        public int Index;
        public long X, Y, Z;
    }

    struct Candidate
    {
        public int SpaceIndex;
        public long W, H, D;
        public long Waste;
        public long LongestLeftover;
        public long ShortestLeftover;
        public long SpaceVolume;
    }

    static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n < 0) return;

        long containerW = fs.NextLong();
        long containerH = fs.NextLong();
        long containerD = fs.NextLong();

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            long w = fs.NextLong();
            long h = fs.NextLong();
            long d = fs.NextLong();
            if (w <= containerW && h <= containerH && d <= containerD)
            {
                boxes.Add(new Box
                {
                    Index = i,
                    W = w,
                    H = h,
                    D = d,
                    V = w * h * d,
                    MaxSide = Math.Max(w, Math.Max(h, d))
                });
            }
        }

        boxes.Sort((a, b) =>
        {
            int c = a.V.CompareTo(b.V);
            if (c != 0) return c;
            c = a.MaxSide.CompareTo(b.MaxSide);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        });

        var spaces = new List<Space>(Math.Max(16, n * 2));
        spaces.Add(new Space { X = 0, Y = 0, Z = 0, W = containerW, H = containerH, D = containerD });

        var placements = new List<Placement>(boxes.Count);
        foreach (var box in boxes)
        {
            if (TryFindPlacement(box, spaces, out Candidate best))
            {
                Space used = spaces[best.SpaceIndex];
                placements.Add(new Placement { Index = box.Index, X = used.X, Y = used.Y, Z = used.Z });
                SplitSpace(spaces, best.SpaceIndex, used, best.W, best.H, best.D);
            }
        }

        var sb = new StringBuilder();
        sb.AppendLine(placements.Count.ToString());
        foreach (var p in placements)
            sb.Append(p.Index).Append(' ').Append(p.X).Append(' ').Append(p.Y).Append(' ').Append(p.Z).AppendLine();
        Console.Write(sb.ToString());
    }

    static bool TryFindPlacement(Box box, List<Space> spaces, out Candidate best)
    {
        best = default;
        bool found = false;

        for (int si = 0; si < spaces.Count; si++)
        {
            Space s = spaces[si];
            TryCandidate(box.W, box.H, box.D, s, si, ref best, ref found);
        }

        return found;
    }

    static void TryCandidate(long w, long h, long d, Space s, int si, ref Candidate best, ref bool found)
    {
        if (w > s.W || h > s.H || d > s.D) return;

        long lw = s.W - w, lh = s.H - h, ld = s.D - d;
        long longest = Math.Max(lw, Math.Max(lh, ld));
        long shortest = Math.Min(lw, Math.Min(lh, ld));
        var c = new Candidate
        {
            SpaceIndex = si,
            W = w,
            H = h,
            D = d,
            Waste = s.V - w * h * d,
            LongestLeftover = longest,
            ShortestLeftover = shortest,
            SpaceVolume = s.V
        };

        if (!found || IsBetter(c, best))
        {
            best = c;
            found = true;
        }
    }

    static bool IsBetter(Candidate a, Candidate b)
    {
        int c = a.Waste.CompareTo(b.Waste);
        if (c != 0) return c < 0;
        c = a.LongestLeftover.CompareTo(b.LongestLeftover);
        if (c != 0) return c < 0;
        c = b.ShortestLeftover.CompareTo(a.ShortestLeftover);
        if (c != 0) return c < 0;
        return a.SpaceVolume < b.SpaceVolume;
    }

    static void SplitSpace(List<Space> spaces, int index, Space s, long w, long h, long d)
    {
        int last = spaces.Count - 1;
        spaces[index] = spaces[last];
        spaces.RemoveAt(last);

        long rw = s.W - w, rh = s.H - h, rd = s.D - d;

        // The three residual cuboids partition the original free cuboid around
        // the placed box, so generated spaces remain disjoint.
        AddSpace(spaces, new Space { X = s.X + w, Y = s.Y, Z = s.Z, W = rw, H = s.H, D = s.D });
        AddSpace(spaces, new Space { X = s.X, Y = s.Y + h, Z = s.Z, W = w, H = rh, D = s.D });
        AddSpace(spaces, new Space { X = s.X, Y = s.Y, Z = s.Z + d, W = w, H = h, D = rd });
    }

    static void AddSpace(List<Space> spaces, Space s)
    {
        if (s.W <= 0 || s.H <= 0 || s.D <= 0) return;
        spaces.Add(s);
    }

    sealed class FastScanner
    {
        readonly Stream _stream;
        readonly byte[] _buffer = new byte[1 << 16];
        int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt() => (int)NextLong();

        public long NextLong()
        {
            int c;
            do
            {
                c = Read();
                if (c < 0) return -1;
            } while (c <= 32);

            long sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }

            long value = 0;
            while (c > 32 && c >= 0)
            {
                value = value * 10 + (c - '0');
                c = Read();
            }
            return value * sign;
        }
    }
}
