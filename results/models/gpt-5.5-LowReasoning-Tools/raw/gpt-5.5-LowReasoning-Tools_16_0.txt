#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <queue>
#include <string>
#include <utility>
#include <vector>

using namespace std;

struct Task {
    int machine = 0;
    long long duration = 0;
};

struct Interval {
    long long start = 0;
    long long end = 0;
};

struct Instance {
    int jobs = 0;
    int machines = 0;
    vector<vector<Task>> job;
    vector<long long> totalWork;
    vector<vector<long long>> remainingFrom;
    int64_t taskCount = 0;
};

struct Schedule {
    vector<vector<long long>> start;
    long long makespan = numeric_limits<long long>::max();
};

enum class Rule {
    EarliestStart,
    ShortestProcessing,
    LongestProcessing,
    MostWorkRemaining,
    LeastWorkRemaining,
    ShortestJobReady,
    LongestJobReady
};

static long long earliestInGaps(const vector<Interval>& intervals, long long ready, long long dur) {
    long long t = ready;
    for (const Interval& in : intervals) {
        if (t + dur <= in.start) return t;
        if (t < in.end) t = in.end;
    }
    return t;
}

static void insertInterval(vector<Interval>& intervals, long long s, long long e) {
    auto it = lower_bound(intervals.begin(), intervals.end(), s,
        [](const Interval& a, long long value) { return a.start < value; });
    intervals.insert(it, {s, e});
}

struct Candidate {
    long long est = 0;
    long long primary = 0;
    long long secondary = 0;
    int job = 0;
};

struct CandidateGreater {
    bool operator()(const Candidate& a, const Candidate& b) const {
        if (a.est != b.est) return a.est > b.est;
        if (a.primary != b.primary) return a.primary > b.primary;
        if (a.secondary != b.secondary) return a.secondary > b.secondary;
        return a.job > b.job;
    }
};

static pair<long long, long long> priorityValues(const Instance& in, Rule rule, int j, int k, long long jobReady) {
    const Task& t = in.job[j][k];
    long long rem = in.remainingFrom[j][k];
    switch (rule) {
        case Rule::ShortestProcessing: return {t.duration, rem};
        case Rule::LongestProcessing: return {-t.duration, -rem};
        case Rule::MostWorkRemaining: return {-rem, -t.duration};
        case Rule::LeastWorkRemaining: return {rem, t.duration};
        case Rule::ShortestJobReady: return {jobReady, t.duration};
        case Rule::LongestJobReady: return {-jobReady, -t.duration};
        case Rule::EarliestStart:
        default: return {0, t.duration};
    }
}

static Schedule buildSchedule(const Instance& in, Rule rule, bool fillGaps) {
    Schedule out;
    out.start.resize(in.jobs);
    for (int j = 0; j < in.jobs; ++j) out.start[j].assign(in.job[j].size(), 0);

    vector<int> nextTask(in.jobs, 0);
    vector<long long> jobReady(in.jobs, 0), machineReady(in.machines, 0);
    vector<vector<Interval>> machineIntervals(fillGaps ? in.machines : 0);
    priority_queue<Candidate, vector<Candidate>, CandidateGreater> pq;

    auto pushJob = [&](int j) {
        if (nextTask[j] >= (int)in.job[j].size()) return;
        int k = nextTask[j];
        const Task& t = in.job[j][k];
        long long est = max(jobReady[j], machineReady[t.machine]);
        auto [p, s] = priorityValues(in, rule, j, k, jobReady[j]);
        pq.push({est, p, s, j});
    };

    for (int j = 0; j < in.jobs; ++j) pushJob(j);

    int64_t scheduled = 0;
    while (scheduled < in.taskCount && !pq.empty()) {
        Candidate c = pq.top();
        pq.pop();
        int j = c.job;
        int k = nextTask[j];
        if (k >= (int)in.job[j].size()) continue;

        const Task& t = in.job[j][k];
        long long expected = max(jobReady[j], machineReady[t.machine]);
        auto [p, s] = priorityValues(in, rule, j, k, jobReady[j]);
        if (c.est != expected || c.primary != p || c.secondary != s) {
            pq.push({expected, p, s, j});
            continue;
        }

        long long st;
        if (fillGaps) {
            st = earliestInGaps(machineIntervals[t.machine], jobReady[j], t.duration);
            insertInterval(machineIntervals[t.machine], st, st + t.duration);
            machineReady[t.machine] = max(machineReady[t.machine], st + t.duration);
        } else {
            st = expected;
            machineReady[t.machine] = st + t.duration;
        }

        out.start[j][k] = st;
        jobReady[j] = st + t.duration;
        out.makespan = max(out.makespan == numeric_limits<long long>::max() ? 0 : out.makespan, jobReady[j]);
        ++nextTask[j];
        ++scheduled;
        pushJob(j);
    }

    if (in.taskCount == 0) out.makespan = 0;
    return out;
}

static bool better(const Schedule& a, const Schedule& b) {
    return a.makespan < b.makespan;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    Instance in;
    if (!(cin >> in.jobs >> in.machines)) return 0;
    if (in.jobs < 0 || in.machines <= 0) return 0;

    in.job.resize(in.jobs);
    in.totalWork.assign(in.jobs, 0);
    in.remainingFrom.resize(in.jobs);

    for (int j = 0; j < in.jobs; ++j) {
        int T;
        cin >> T;
        in.job[j].resize(max(0, T));
        in.taskCount += max(0, T);
        for (int k = 0; k < T; ++k) {
            cin >> in.job[j][k].machine >> in.job[j][k].duration;
            if (in.job[j][k].machine < 0) in.job[j][k].machine = 0;
            if (in.job[j][k].machine >= in.machines) in.job[j][k].machine %= in.machines;
            if (in.job[j][k].duration < 0) in.job[j][k].duration = 0;
            in.totalWork[j] += in.job[j][k].duration;
        }
        in.remainingFrom[j].assign(in.job[j].size() + 1, 0);
        for (int k = (int)in.job[j].size() - 1; k >= 0; --k) {
            in.remainingFrom[j][k] = in.remainingFrom[j][k + 1] + in.job[j][k].duration;
        }
    }

    vector<Rule> rules;
    bool fillGaps = false;

    if (in.taskCount <= 2000) {
        fillGaps = true;
        rules = {
            Rule::EarliestStart,
            Rule::ShortestProcessing,
            Rule::LongestProcessing,
            Rule::MostWorkRemaining,
            Rule::LeastWorkRemaining,
            Rule::ShortestJobReady,
            Rule::LongestJobReady
        };
    } else if (in.taskCount <= 20000) {
        fillGaps = true;
        rules = {
            Rule::EarliestStart,
            Rule::ShortestProcessing,
            Rule::MostWorkRemaining,
            Rule::LongestProcessing
        };
    } else {
        fillGaps = false;
        rules = {
            Rule::EarliestStart,
            Rule::MostWorkRemaining,
            Rule::ShortestProcessing
        };
    }

    Schedule best = buildSchedule(in, rules.front(), fillGaps);
    for (size_t i = 1; i < rules.size(); ++i) {
        Schedule cand = buildSchedule(in, rules[i], fillGaps);
        if (better(cand, best)) best = std::move(cand);
    }

    for (int j = 0; j < in.jobs; ++j) {
        for (int k = 0; k < (int)in.job[j].size(); ++k) {
            if (k) cout << ' ';
            cout << best.start[j][k] << ' ' << in.job[j][k].machine;
        }
        cout << '\n';
    }

    return 0;
}
