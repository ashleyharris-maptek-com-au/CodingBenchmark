using System;
using System.IO;
using System.Text;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }

            int value = 0;
            while (c > 32)
            {
                value = value * 10 + (c - '0');
                c = ReadByte();
            }
            return value * sign;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;

        int m = fs.NextInt();
        if (n <= 0)
        {
            Console.WriteLine();
            return;
        }

        int arcCount = checked(m * 2);
        int[] head = new int[n];
        int[] iter = new int[n];
        int[] to = new int[arcCount];
        int[] next = new int[arcCount];
        Array.Fill(head, -1);

        int edgeIndex = 0;
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            fs.NextInt(); // Weight is irrelevant for a guaranteed covering route.

            to[edgeIndex] = v;
            next[edgeIndex] = head[u];
            head[u] = edgeIndex++;

            to[edgeIndex] = u;
            next[edgeIndex] = head[v];
            head[v] = edgeIndex++;
        }

        Array.Copy(head, iter, n);

        // Each input edge was replaced by two directed arcs, one each way.
        // Every vertex is balanced, so a directed Euler tour gives a closed
        // route from 0 that traverses every original edge at least once.
        int[] stack = new int[arcCount + 1];
        int[] route = new int[arcCount + 1];
        int sp = 0, rp = 0;
        stack[sp++] = 0;

        while (sp > 0)
        {
            int v = stack[sp - 1];
            int e = iter[v];
            if (e != -1)
            {
                iter[v] = next[e];
                stack[sp++] = to[e];
            }
            else
            {
                route[rp++] = v;
                sp--;
            }
        }

        var output = new StringBuilder(Math.Min(1 << 26, Math.Max(16, rp * 4)));
        for (int i = rp - 1; i >= 0; i--)
        {
            if (output.Length > 0) output.Append(' ');
            output.Append(route[i]);
        }

        Console.Write(output.ToString());
    }
}
