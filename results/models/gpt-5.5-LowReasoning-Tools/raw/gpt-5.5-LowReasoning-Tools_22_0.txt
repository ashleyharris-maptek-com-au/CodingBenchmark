using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private struct Cell
    {
        public int X;
        public int Y;
        public Cell(int x, int y) { X = x; Y = y; }
    }

    private sealed class Orientation
    {
        public int Rotation;
        public int Width;
        public int Height;
        public Cell[] Cells;

        public Orientation(int rotation, Cell[] cells)
        {
            Rotation = rotation;
            Cells = Normalize(cells, out Width, out Height);
        }

        private static Cell[] Normalize(Cell[] cells, out int width, out int height)
        {
            int minX = int.MaxValue, minY = int.MaxValue;
            for (int i = 0; i < cells.Length; i++)
            {
                if (cells[i].X < minX) minX = cells[i].X;
                if (cells[i].Y < minY) minY = cells[i].Y;
            }

            Cell[] result = new Cell[cells.Length];
            int maxX = 0, maxY = 0;
            for (int i = 0; i < cells.Length; i++)
            {
                int x = cells[i].X - minX;
                int y = cells[i].Y - minY;
                result[i] = new Cell(x, y);
                if (x > maxX) maxX = x;
                if (y > maxY) maxY = y;
            }

            Array.Sort(result, (a, b) => a.X == b.X ? a.Y.CompareTo(b.Y) : a.X.CompareTo(b.X));
            width = maxX + 1;
            height = maxY + 1;
            return result;
        }
    }

    private sealed class Move
    {
        public Orientation Shape = null!;
        public int X;
        public int Y;
        public double Score;
        public int Lines;
        public bool Valid;
    }

    private sealed class FastScanner
    {
        private readonly System.IO.TextReader _reader;

        public FastScanner(System.IO.TextReader reader)
        {
            _reader = reader;
        }

        public string? Next()
        {
            var chars = new List<char>(16);
            int c;
            do
            {
                c = Read();
                if (c < 0) return null;
            } while (char.IsWhiteSpace((char)c) || c == 0xFEFF);

            do
            {
                chars.Add((char)c);
                c = Read();
            } while (c >= 0 && !char.IsWhiteSpace((char)c) && c != 0xFEFF);

            return new string(chars.ToArray());
        }

        private int Read()
        {
            return _reader.Read();
        }
    }

    private static int _width;
    private static int _height;
    private static bool[,] _board = null!;
    private static readonly Dictionary<char, Orientation[]> Pieces = BuildPieces();

    public static void Main()
    {
        var input = new FastScanner(Console.In);
        string? wTok = input.Next();
        if (wTok == null) return;

        _width = int.Parse(wTok, CultureInfo.InvariantCulture);
        _height = int.Parse(input.Next() ?? "20", CultureInfo.InvariantCulture);
        int numPieces = int.Parse(input.Next() ?? "0", CultureInfo.InvariantCulture);
        _board = new bool[_height, _width];

        for (int turn = 0; numPieces <= 0 || turn < numPieces; turn++)
        {
            string? pieceToken = input.Next();
            if (pieceToken == null) break;
            char piece = char.ToUpperInvariant(pieceToken[0]);

            Move move = ChooseMove(piece, turn, numPieces);
            Console.WriteLine(move.Shape.Rotation.ToString(CultureInfo.InvariantCulture) + " " + move.X.ToString(CultureInfo.InvariantCulture));
            Console.Out.Flush();

            string? result = input.Next();
            if (result == null) break;
            if (result == "gameover") break;

            if (move.Valid)
            {
                ApplyMove(move);
            }
            else
            {
                RebuildAfterUnknownOutcome(result);
            }
        }
    }

    private static Move ChooseMove(char piece, int turn, int totalPieces)
    {
        if (!Pieces.TryGetValue(piece, out Orientation[]? orientations))
            orientations = Pieces['O'];

        bool marathon = totalPieces <= 0 || totalPieces > _height * _width * 3;
        bool danger = MaxColumnHeight() > _height * 3 / 4;
        double best = double.NegativeInfinity;
        Move bestMove = new Move { Shape = orientations[0], X = 0, Y = 0, Score = best, Valid = false };

        for (int i = 0; i < orientations.Length; i++)
        {
            Orientation shape = orientations[i];
            for (int x = 0; x <= _width - shape.Width; x++)
            {
                int y = DropY(shape, x);
                if (y < 0 || y + shape.Height > _height) continue;

                bool[,] after = CopyBoard();
                Place(after, shape, x, y);
                int lines = ClearLines(after);
                double score = ScoreBoard(after, lines, shape, x, y, marathon, danger);

                if (score > best || (Math.Abs(score - best) < 1e-9 && TieBreak(shape, x, bestMove.Shape, bestMove.X)))
                {
                    best = score;
                    bestMove = new Move { Shape = shape, X = x, Y = y, Score = score, Lines = lines, Valid = true };
                }
            }
        }

        if (!bestMove.Valid)
        {
            Orientation shape = orientations[0];
            bestMove = new Move { Shape = shape, X = Math.Max(0, Math.Min(_width - 1, (_width - shape.Width) / 2)), Y = 0, Valid = false };
        }

        return bestMove;
    }

    private static double ScoreBoard(bool[,] b, int lines, Orientation shape, int x, int y, bool marathon, bool danger)
    {
        int[] heights = Heights(b);
        int aggregateHeight = 0, maxHeight = 0, bumpiness = 0, wells = 0;
        for (int i = 0; i < _width; i++)
        {
            aggregateHeight += heights[i];
            if (heights[i] > maxHeight) maxHeight = heights[i];
            int left = i == 0 ? heights[i + 1] : heights[i - 1];
            int right = i == _width - 1 ? heights[i - 1] : heights[i + 1];
            int rim = Math.Min(left, right);
            if (rim > heights[i]) wells += (rim - heights[i]) * (rim - heights[i] + 1) / 2;
            if (i > 0) bumpiness += Math.Abs(heights[i] - heights[i - 1]);
        }

        int holes = 0, covered = 0, rowTransitions = 0, colTransitions = 0;
        for (int c = 0; c < _width; c++)
        {
            bool seenBlock = false;
            int cover = 0;
            bool previous = true;
            for (int r = _height - 1; r >= 0; r--)
            {
                bool filled = b[r, c];
                if (filled)
                {
                    seenBlock = true;
                    cover++;
                }
                else if (seenBlock)
                {
                    holes++;
                    covered += cover;
                }
                if (filled != previous) colTransitions++;
                previous = filled;
            }
            if (!previous) colTransitions++;
        }

        for (int r = 0; r < _height; r++)
        {
            bool previous = true;
            for (int c = 0; c < _width; c++)
            {
                bool filled = b[r, c];
                if (filled != previous) rowTransitions++;
                previous = filled;
            }
            if (!previous) rowTransitions++;
        }

        int landingHeight = _height - y - shape.Height / 2;
        double lineReward = marathon ? 760.0 : 520.0;
        double score =
            lines * lineReward
            - aggregateHeight * (marathon ? 5.2 : 3.9)
            - maxHeight * (danger ? 12.0 : 4.5)
            - holes * (danger ? 95.0 : 62.0)
            - covered * 5.0
            - bumpiness * (marathon ? 8.0 : 5.2)
            - wells * (marathon ? 6.0 : 2.0)
            - rowTransitions * 1.6
            - colTransitions * 2.8
            - landingHeight * (danger ? 5.5 : 1.2);

        if (lines == 4 && shape.Width == 4) score += marathon ? 420.0 : 120.0;
        if (x == 0 || x + shape.Width == _width) score += 3.0;
        return score;
    }

    private static bool TieBreak(Orientation candidate, int candidateX, Orientation incumbent, int incumbentX)
    {
        int center = _width / 2;
        int candidateDist = Math.Abs((candidateX * 2 + candidate.Width) - center * 2);
        int incumbentDist = Math.Abs((incumbentX * 2 + incumbent.Width) - center * 2);
        if (candidateDist != incumbentDist) return candidateDist < incumbentDist;
        if (candidate.Rotation != incumbent.Rotation) return candidate.Rotation < incumbent.Rotation;
        return candidateX < incumbentX;
    }

    private static int DropY(Orientation shape, int x)
    {
        for (int y = _height - shape.Height; y >= -shape.Height; y--)
        {
            if (Collides(shape, x, y))
            {
                int settled = y + 1;
                return settled + shape.Height <= _height ? settled : -1;
            }
        }
        return 0;
    }

    private static bool Collides(Orientation shape, int x, int y)
    {
        for (int i = 0; i < shape.Cells.Length; i++)
        {
            int bx = x + shape.Cells[i].X;
            int by = y + shape.Cells[i].Y;
            if (bx < 0 || bx >= _width || by < 0) return true;
            if (by < _height && _board[by, bx]) return true;
        }
        return false;
    }

    private static void ApplyMove(Move move)
    {
        Place(_board, move.Shape, move.X, move.Y);
        ClearLines(_board);
    }

    private static void Place(bool[,] b, Orientation shape, int x, int y)
    {
        for (int i = 0; i < shape.Cells.Length; i++)
        {
            int bx = x + shape.Cells[i].X;
            int by = y + shape.Cells[i].Y;
            if (0 <= bx && bx < _width && 0 <= by && by < _height)
                b[by, bx] = true;
        }
    }

    private static int ClearLines(bool[,] b)
    {
        int write = 0;
        int cleared = 0;
        for (int read = 0; read < _height; read++)
        {
            bool full = true;
            for (int c = 0; c < _width; c++)
            {
                if (!b[read, c]) { full = false; break; }
            }

            if (full)
            {
                cleared++;
            }
            else
            {
                if (write != read)
                {
                    for (int c = 0; c < _width; c++)
                        b[write, c] = b[read, c];
                }
                write++;
            }
        }

        for (int r = write; r < _height; r++)
            for (int c = 0; c < _width; c++)
                b[r, c] = false;

        return cleared;
    }

    private static int[] Heights(bool[,] b)
    {
        int[] heights = new int[_width];
        for (int c = 0; c < _width; c++)
        {
            for (int r = _height - 1; r >= 0; r--)
            {
                if (b[r, c])
                {
                    heights[c] = r + 1;
                    break;
                }
            }
        }
        return heights;
    }

    private static int MaxColumnHeight()
    {
        int max = 0;
        int[] heights = Heights(_board);
        for (int i = 0; i < heights.Length; i++)
            if (heights[i] > max) max = heights[i];
        return max;
    }

    private static bool[,] CopyBoard()
    {
        var copy = new bool[_height, _width];
        Array.Copy(_board, copy, _board.Length);
        return copy;
    }

    private static void RebuildAfterUnknownOutcome(string result)
    {
        if (!result.StartsWith("cleared_", StringComparison.Ordinal)) return;
        if (!int.TryParse(result.Substring(8), NumberStyles.Integer, CultureInfo.InvariantCulture, out int count) || count <= 0) return;
        ClearLowestCompleteOrDenseRows(count);
    }

    private static void ClearLowestCompleteOrDenseRows(int count)
    {
        for (int k = 0; k < count; k++)
        {
            int bestRow = -1;
            int bestFill = -1;
            for (int r = 0; r < _height; r++)
            {
                int fill = 0;
                for (int c = 0; c < _width; c++)
                    if (_board[r, c]) fill++;
                if (fill > bestFill)
                {
                    bestFill = fill;
                    bestRow = r;
                }
            }

            if (bestRow < 0) return;
            for (int r = bestRow; r < _height - 1; r++)
                for (int c = 0; c < _width; c++)
                    _board[r, c] = _board[r + 1, c];
            for (int c = 0; c < _width; c++)
                _board[_height - 1, c] = false;
        }
    }

    private static Dictionary<char, Orientation[]> BuildPieces()
    {
        var result = new Dictionary<char, Orientation[]>();
        result['I'] = Unique(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(3, 0) });
        result['O'] = Unique(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(0, 1), new Cell(1, 1) });
        result['T'] = Unique(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(1, 1) });
        result['S'] = Unique(new[] { new Cell(1, 0), new Cell(2, 0), new Cell(0, 1), new Cell(1, 1) });
        result['Z'] = Unique(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(1, 1), new Cell(2, 1) });
        result['J'] = Unique(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(0, 1) });
        result['L'] = Unique(new[] { new Cell(0, 0), new Cell(1, 0), new Cell(2, 0), new Cell(2, 1) });
        return result;
    }

    private static Orientation[] Unique(Cell[] baseCells)
    {
        var list = new List<Orientation>();
        var seen = new HashSet<string>();
        Cell[] cells = baseCells;
        for (int r = 0; r < 4; r++)
        {
            var orientation = new Orientation(r, cells);
            string key = Key(orientation.Cells);
            if (seen.Add(key))
                list.Add(orientation);
            cells = RotateClockwise(cells);
        }
        return list.ToArray();
    }

    private static Cell[] RotateClockwise(Cell[] cells)
    {
        var rotated = new Cell[cells.Length];
        for (int i = 0; i < cells.Length; i++)
            rotated[i] = new Cell(cells[i].Y, -cells[i].X);
        return rotated;
    }

    private static string Key(Cell[] cells)
    {
        char[] chars = new char[cells.Length * 2];
        for (int i = 0; i < cells.Length; i++)
        {
            chars[i * 2] = (char)('0' + cells[i].X);
            chars[i * 2 + 1] = (char)('0' + cells[i].Y);
        }
        return new string(chars);
    }
}
