using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public class Program
{
    struct V3
    {
        public double X, Y, Z;
        public V3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static V3 operator +(V3 a, V3 b) => new V3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static V3 operator -(V3 a, V3 b) => new V3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static V3 operator *(V3 a, double s) => new V3(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public double Norm2() => X * X + Y * Y + Z * Z;
        public static double Dot(V3 a, V3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static V3 Cross(V3 a, V3 b) => new V3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    struct State
    {
        public V3 R, V, H;
        public double Rn, Vn, A, Energy, Incl, Raan, ArgLat;
        public int Id;
    }

    static int N;
    static double Mu;
    static State Start;
    static State[] S = Array.Empty<State>();
    static double[,] C = new double[0, 0];
    static double[] StartCost = Array.Empty<double>();

    public static void Main()
    {
        var nums = ReadAllDoubles();
        if (nums.Count < 8) return;
        int p = 0;
        N = (int)Math.Round(nums[p++]);
        Mu = nums[p++];
        if (Mu <= 0) Mu = 398600.4418;

        Start = MakeState(-1, nums[p++], nums[p++], nums[p++], nums[p++], nums[p++], nums[p++]);
        S = new State[N];
        for (int i = 0; i < N; i++)
            S[i] = MakeState(i, nums[p++], nums[p++], nums[p++], nums[p++], nums[p++], nums[p++]);

        BuildCosts();
        int[] order = N <= 1 ? Enumerable.Range(0, N).ToArray()
                    : N <= 11 ? SolveExactSmall()
                    : SolveHeuristic();
        Console.WriteLine(string.Join(" ", order));
    }

    static List<double> ReadAllDoubles()
    {
        string text = Console.In.ReadToEnd();
        var list = new List<double>();
        foreach (string token in text.Split((char[])null, StringSplitOptions.RemoveEmptyEntries))
            if (double.TryParse(token, NumberStyles.Float, CultureInfo.InvariantCulture, out double v))
                list.Add(v);
        return list;
    }

    static State MakeState(int id, double x, double y, double z, double vx, double vy, double vz)
    {
        var r = new V3(x, y, z);
        var v = new V3(vx, vy, vz);
        var h = V3.Cross(r, v);
        double rn = Math.Max(1e-9, r.Norm());
        double vn = v.Norm();
        double energy = 0.5 * vn * vn - Mu / rn;
        double a = energy < -1e-12 ? -Mu / (2.0 * energy) : rn;
        double hn = Math.Max(1e-12, h.Norm());
        double incl = Math.Acos(Clamp(h.Z / hn));
        double raan = Math.Atan2(h.X, -h.Y);
        double argLat = Math.Atan2(r.Z / Math.Max(1e-12, Math.Sin(incl)), r.X * Math.Cos(raan) + r.Y * Math.Sin(raan));
        if (double.IsNaN(argLat)) argLat = Math.Atan2(r.Y, r.X);
        return new State { Id = id, R = r, V = v, H = h, Rn = rn, Vn = vn, A = a, Energy = energy, Incl = incl, Raan = raan, ArgLat = argLat };
    }

    static void BuildCosts()
    {
        StartCost = new double[N];
        C = new double[N, N];
        for (int i = 0; i < N; i++)
        {
            StartCost[i] = TransferCost(Start, S[i]);
            for (int j = 0; j < N; j++)
                C[i, j] = i == j ? 1e100 : TransferCost(S[i], S[j]);
        }
    }

    static double TransferCost(State a, State b)
    {
        double r1 = a.Rn, r2 = b.Rn;
        double vc1 = Math.Sqrt(Mu / r1), vc2 = Math.Sqrt(Mu / r2);
        double hoh = Math.Abs(Math.Sqrt(Mu / r1) * (Math.Sqrt(2 * r2 / (r1 + r2)) - 1.0)) +
                     Math.Abs(Math.Sqrt(Mu / r2) * (1.0 - Math.Sqrt(2 * r1 / (r1 + r2))));

        double hden = Math.Max(1e-12, a.H.Norm() * b.H.Norm());
        double plane = Math.Acos(Clamp(V3.Dot(a.H, b.H) / hden));
        double planeDv = 2.0 * Math.Min(vc1, vc2) * Math.Sin(plane * 0.5);

        double relV = (a.V - b.V).Norm();
        double posChord = (a.R - b.R).Norm() / Math.Max(r1 + r2, 1.0);
        double phase = Math.Abs(NormalizeAngle(b.ArgLat - a.ArgLat));
        double energy = Math.Abs(a.Energy - b.Energy) / Math.Max(Math.Max(vc1, vc2), 1e-6);
        double eccShape = Math.Abs((a.Vn - vc1) - (b.Vn - vc2));

        return 1.45 * hoh + 0.95 * planeDv + 0.18 * relV + 0.08 * energy + 0.06 * eccShape
             + 0.015 * Math.Sqrt(r1 * r2) * posChord / 1000.0 + 0.01 * phase;
    }

    static int[] SolveExactSmall()
    {
        int total = 1 << N;
        var dp = new double[total, N];
        var prev = new int[total, N];
        for (int m = 0; m < total; m++)
            for (int i = 0; i < N; i++) { dp[m, i] = 1e100; prev[m, i] = -1; }
        for (int i = 0; i < N; i++) dp[1 << i, i] = StartCost[i];

        for (int mask = 1; mask < total; mask++)
            for (int last = 0; last < N; last++)
            {
                double cur = dp[mask, last];
                if (cur >= 1e90) continue;
                int rem = (total - 1) ^ mask;
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    double val = cur + C[last, nxt];
                    int nm = mask | bit;
                    if (val < dp[nm, nxt]) { dp[nm, nxt] = val; prev[nm, nxt] = last; }
                    rem -= bit;
                }
            }

        int full = total - 1, best = 0;
        for (int i = 1; i < N; i++) if (dp[full, i] < dp[full, best]) best = i;
        var ans = new int[N];
        int curMask = full, curLast = best;
        for (int k = N - 1; k >= 0; k--)
        {
            ans[k] = curLast;
            int pr = prev[curMask, curLast];
            curMask ^= 1 << curLast;
            curLast = pr;
        }
        return ans;
    }

    static int[] SolveHeuristic()
    {
        var best = NearestNeighbor(ChooseSeed());
        Improve(best, Environment.TickCount64 + (N <= 25 ? 2500 : N <= 60 ? 4500 : 7500));

        int starts = Math.Min(N, N <= 30 ? 12 : 7);
        var candidates = Enumerable.Range(0, N).OrderBy(i => StartCost[i]).Take(starts).ToArray();
        foreach (int s in candidates)
        {
            var path = NearestNeighbor(s);
            Improve(path, Environment.TickCount64 + (N <= 30 ? 500 : 800));
            if (PathCost(path) < PathCost(best)) best = path;
        }
        return best;
    }

    static int ChooseSeed()
    {
        int best = 0;
        for (int i = 1; i < N; i++) if (StartCost[i] < StartCost[best]) best = i;
        return best;
    }

    static int[] NearestNeighbor(int first)
    {
        var path = new int[N];
        var used = new bool[N];
        path[0] = first; used[first] = true;
        for (int k = 1; k < N; k++)
        {
            int last = path[k - 1], best = -1;
            double bc = 1e100;
            for (int i = 0; i < N; i++)
                if (!used[i])
                {
                    double look = C[last, i];
                    if (k + 1 < N)
                    {
                        double near = 1e100;
                        for (int j = 0; j < N; j++) if (!used[j] && j != i && C[i, j] < near) near = C[i, j];
                        look += 0.12 * near;
                    }
                    if (look < bc) { bc = look; best = i; }
                }
            path[k] = best; used[best] = true;
        }
        return path;
    }

    static void Improve(int[] path, long deadline)
    {
        bool changed = true;
        while (changed && Environment.TickCount64 < deadline)
        {
            changed = false;
            for (int i = 0; i < N - 2 && Environment.TickCount64 < deadline; i++)
            {
                int a = i == 0 ? -1 : path[i - 1];
                int b = path[i], c = path[i + 1];
                double beforeLeft = a < 0 ? StartCost[b] : C[a, b];
                for (int j = i + 1; j < N - 1; j++)
                {
                    int d = path[j], e = path[j + 1];
                    double before = beforeLeft + C[d, e];
                    double after = (a < 0 ? StartCost[d] : C[a, d]) + C[b, e];
                    if (after + 1e-12 < before)
                    {
                        Array.Reverse(path, i, j - i + 1);
                        changed = true;
                        break;
                    }
                }
                if (changed) break;
            }
        }

        bool moved = true;
        while (moved && Environment.TickCount64 < deadline)
        {
            moved = false;
            for (int i = 0; i < N && !moved && Environment.TickCount64 < deadline; i++)
            {
                int node = path[i];
                double removeDelta = RemovedDelta(path, i);
                for (int j = 0; j <= N - 1; j++)
                {
                    if (j == i || j == i + 1) continue;
                    double addDelta = InsertDelta(path, i, j, node);
                    if (removeDelta + addDelta < -1e-10)
                    {
                        Move(path, i, j);
                        moved = true;
                        break;
                    }
                }
            }
        }
    }

    static double RemovedDelta(int[] p, int i)
    {
        int x = p[i];
        double old = (i == 0 ? StartCost[x] : C[p[i - 1], x]) + (i == N - 1 ? 0 : C[x, p[i + 1]]);
        double neu = i == N - 1 ? 0 : (i == 0 ? StartCost[p[i + 1]] : C[p[i - 1], p[i + 1]]);
        return neu - old;
    }

    static double InsertDelta(int[] p, int from, int beforeIndex, int x)
    {
        int adjBefore = beforeIndex;
        if (from < beforeIndex) adjBefore--;
        if (adjBefore <= 0) return StartCost[x] + C[x, p[0]] - StartCost[p[0]];
        if (adjBefore >= N - 1) return C[p[N - 1], x];
        int a = p[adjBefore - 1], b = p[adjBefore];
        return C[a, x] + C[x, b] - C[a, b];
    }

    static void Move(int[] p, int from, int before)
    {
        int x = p[from];
        if (from < before)
        {
            Array.Copy(p, from + 1, p, from, before - from - 1);
            p[before - 1] = x;
        }
        else
        {
            Array.Copy(p, before, p, before + 1, from - before);
            p[before] = x;
        }
    }

    static double PathCost(int[] p)
    {
        if (p.Length == 0) return 0;
        double sum = StartCost[p[0]];
        for (int i = 1; i < p.Length; i++) sum += C[p[i - 1], p[i]];
        return sum;
    }

    static int BitIndex(int bit)
    {
        int idx = 0;
        while ((bit >>= 1) != 0) idx++;
        return idx;
    }

    static double Clamp(double x) => x < -1 ? -1 : (x > 1 ? 1 : x);
    static double NormalizeAngle(double a)
    {
        while (a <= -Math.PI) a += 2 * Math.PI;
        while (a > Math.PI) a -= 2 * Math.PI;
        return a;
    }
}
