using namespace std;

struct Timer {
    chrono::steady_clock::time_point start = chrono::steady_clock::now();
    double seconds() const {
        return chrono::duration<double>(chrono::steady_clock::now() - start).count();
    }
};

struct FastInput {
    static const int BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    int idx = 0, size = 0;

    char getChar() {
        if (idx >= size) {
            cin.read(buf, BUFSIZE);
            size = static_cast<int>(cin.gcount());
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    bool readInt(int &out) {
        char c;
        int sign = 1;
        int value = 0;
        do {
            c = getChar();
            if (!c) return false;
        } while (c <= ' ');
        if (c == '-') {
            sign = -1;
            c = getChar();
        }
        while (c > ' ') {
            value = value * 10 + (c - '0');
            c = getChar();
        }
        out = value * sign;
        return true;
    }
};

struct Graph {
    int n = 0;
    int k = 0;
    vector<int> offset;
    vector<int> adj;
    vector<pair<int, int> > edges;

    int degree(int v) const {
        return offset[v + 1] - offset[v];
    }
};

static Graph readGraph() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    FastInput in;
    int n = 0, m = 0, k = 0;
    if (!in.readInt(n)) return Graph();
    in.readInt(m);
    in.readInt(k);

    Graph g;
    g.n = max(0, n);
    g.k = max(0, k);
    g.edges.reserve(static_cast<size_t>(max(0, m)));
    vector<int> deg(g.n, 0);

    for (int i = 0; i < m; ++i) {
        int u, v;
        in.readInt(u);
        in.readInt(v);
        if (u < 0 || v < 0 || u >= g.n || v >= g.n || u == v) continue;
        g.edges.push_back(make_pair(u, v));
        ++deg[u];
        ++deg[v];
    }

    g.offset.assign(static_cast<size_t>(g.n) + 1, 0);
    for (int i = 0; i < g.n; ++i) g.offset[i + 1] = g.offset[i] + deg[i];
    g.adj.assign(static_cast<size_t>(g.offset[g.n]), 0);
    vector<int> cur = g.offset;
    for (size_t i = 0; i < g.edges.size(); ++i) {
        int u = g.edges[i].first;
        int v = g.edges[i].second;
        g.adj[cur[u]++] = v;
        g.adj[cur[v]++] = u;
    }
    return g;
}

static vector<int> largestDegreeOrder(const Graph &g) {
    vector<int> order(g.n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        int da = g.degree(a), db = g.degree(b);
        if (da != db) return da > db;
        return a < b;
    });
    return order;
}

static vector<int> smallestLastOrder(const Graph &g) {
    int n = g.n;
    vector<int> degree(n), removed(n, 0), order;
    order.reserve(n);
    int maxDegree = 0;
    for (int v = 0; v < n; ++v) {
        degree[v] = g.degree(v);
        maxDegree = max(maxDegree, degree[v]);
    }

    vector<vector<int> > buckets(static_cast<size_t>(maxDegree) + 1);
    for (int v = 0; v < n; ++v) buckets[degree[v]].push_back(v);

    int current = 0;
    for (int taken = 0; taken < n; ++taken) {
        while (current <= maxDegree) {
            while (!buckets[current].empty() && (removed[buckets[current].back()] || degree[buckets[current].back()] != current)) {
                buckets[current].pop_back();
            }
            if (!buckets[current].empty()) break;
            ++current;
        }
        if (current > maxDegree) break;

        int v = buckets[current].back();
        buckets[current].pop_back();
        if (removed[v]) {
            --taken;
            continue;
        }
        removed[v] = 1;
        order.push_back(v);

        for (int p = g.offset[v]; p < g.offset[v + 1]; ++p) {
            int u = g.adj[p];
            if (!removed[u]) {
                int old = degree[u]--;
                buckets[old - 1].push_back(u);
                if (old - 1 < current) current = old - 1;
            }
        }
    }

    reverse(order.begin(), order.end());
    if (static_cast<int>(order.size()) != n) return largestDegreeOrder(g);
    return order;
}

static void greedyColor(const Graph &g, const vector<int> &order, vector<int> &color) {
    color.assign(g.n, -1);
    if (g.k <= 0) return;

    vector<int> seen(static_cast<size_t>(g.k), -1);
    for (size_t oi = 0; oi < order.size(); ++oi) {
        int v = order[oi];
        for (int p = g.offset[v]; p < g.offset[v + 1]; ++p) {
            int c = color[g.adj[p]];
            if (c >= 0) seen[c] = v;
        }
        int chosen = -1;
        for (int c = 0; c < g.k; ++c) {
            if (seen[c] != v) {
                chosen = c;
                break;
            }
        }
        if (chosen < 0) {
            vector<int> cnt(static_cast<size_t>(g.k), 0);
            for (int p = g.offset[v]; p < g.offset[v + 1]; ++p) {
                int c = color[g.adj[p]];
                if (c >= 0) ++cnt[c];
            }
            chosen = static_cast<int>(min_element(cnt.begin(), cnt.end()) - cnt.begin());
        }
        color[v] = chosen;
    }
}

static vector<int> conflictedVertices(const Graph &g, const vector<int> &color, int &conflicts) {
    vector<int> bad;
    vector<char> inBad(g.n, 0);
    conflicts = 0;
    for (size_t i = 0; i < g.edges.size(); ++i) {
        int u = g.edges[i].first, v = g.edges[i].second;
        if (color[u] == color[v]) {
            ++conflicts;
            if (!inBad[u]) {
                inBad[u] = 1;
                bad.push_back(u);
            }
            if (!inBad[v]) {
                inBad[v] = 1;
                bad.push_back(v);
            }
        }
    }
    return bad;
}

static bool repairColoring(const Graph &g, vector<int> &color, const Timer &timer) {
    if (g.k <= 0) return g.n == 0;
    if (g.k == 1) return g.edges.empty();

    mt19937 rng(1234567u + static_cast<unsigned>(g.n) * 17u + static_cast<unsigned>(g.edges.size()));
    vector<int> cnt(static_cast<size_t>(g.k), 0);

    int conflicts = 0;
    vector<int> bad = conflictedVertices(g, color, conflicts);
    if (conflicts == 0) return true;

    int bestConflicts = conflicts;
    vector<int> bestColor = color;
    const double timeLimit = 285.0;
    int maxPasses = (g.n > 300000 ? 30 : 400);

    for (int pass = 0; pass < maxPasses && timer.seconds() < timeLimit; ++pass) {
        shuffle(bad.begin(), bad.end(), rng);

        for (size_t bi = 0; bi < bad.size() && timer.seconds() < timeLimit; ++bi) {
            int v = bad[bi];
            fill(cnt.begin(), cnt.end(), 0);
            for (int p = g.offset[v]; p < g.offset[v + 1]; ++p) {
                int c = color[g.adj[p]];
                if (c >= 0) ++cnt[c];
            }

            int old = color[v];
            int chosen = old;
            int best = cnt[old];
            int start = static_cast<int>(rng() % static_cast<unsigned>(g.k));
            for (int t = 0; t < g.k; ++t) {
                int c = (start + t) % g.k;
                if (cnt[c] < best) {
                    best = cnt[c];
                    chosen = c;
                    if (best == 0) break;
                }
            }
            color[v] = chosen;
        }

        bad = conflictedVertices(g, color, conflicts);
        if (conflicts == 0) return true;
        if (conflicts < bestConflicts) {
            bestConflicts = conflicts;
            bestColor = color;
        } else if ((pass % 7) == 6 && !bad.empty()) {
            int kicks = min<int>(static_cast<int>(bad.size()), max(8, g.n / 2000));
            for (int i = 0; i < kicks; ++i) {
                int v = bad[static_cast<size_t>(rng() % bad.size())];
                color[v] = static_cast<int>(rng() % static_cast<unsigned>(g.k));
            }
        }
    }

    color.swap(bestColor);
    return bestConflicts == 0;
}

static bool dsaturSmall(const Graph &g, vector<int> &color, const Timer &timer) {
    int n = g.n;
    if (n > 160 || g.k > 64 || timer.seconds() > 250.0) return false;

    color.assign(n, -1);
    vector<uint64_t> neighMask(n, 0);
    vector<int> sat(n, 0);
    int colored = 0;
    const uint64_t fullMask = (g.k == 64) ? numeric_limits<uint64_t>::max() : ((uint64_t(1) << g.k) - 1);

    function<bool()> dfs = [&]() -> bool {
        if (colored == n) return true;
        if (timer.seconds() > 285.0) return false;

        int v = -1;
        for (int i = 0; i < n; ++i) {
            if (color[i] >= 0) continue;
            if (v < 0 || sat[i] > sat[v] || (sat[i] == sat[v] && g.degree(i) > g.degree(v))) v = i;
        }

        uint64_t available = fullMask & ~neighMask[v];
        while (available) {
            uint64_t bit = available & (~available + 1);
            int c = 0;
            uint64_t temp = bit;
            while ((temp >>= 1) != 0) ++c;
            available &= available - 1;

            vector<pair<int, uint64_t> > changed;
            color[v] = c;
            ++colored;
            for (int p = g.offset[v]; p < g.offset[v + 1]; ++p) {
                int u = g.adj[p];
                if (color[u] < 0 && (neighMask[u] & bit) == 0) {
                    changed.push_back(make_pair(u, neighMask[u]));
                    neighMask[u] |= bit;
                    ++sat[u];
                }
            }

            if (dfs()) return true;

            for (size_t i = 0; i < changed.size(); ++i) {
                int u = changed[i].first;
                neighMask[u] = changed[i].second;
                --sat[u];
            }
            --colored;
            color[v] = -1;
        }
        return false;
    };

    return dfs();
}

static bool verify(const Graph &g, const vector<int> &color) {
    if (static_cast<int>(color.size()) != g.n) return false;
    for (int v = 0; v < g.n; ++v) {
        if (color[v] < 0 || color[v] >= g.k) return false;
    }
    for (size_t i = 0; i < g.edges.size(); ++i) {
        if (color[g.edges[i].first] == color[g.edges[i].second]) return false;
    }
    return true;
}

int main() {
    Timer timer;
    Graph g = readGraph();
    vector<int> color;

    if (g.n == 0) {
        cout << '\n';
        return 0;
    }
    if (g.k <= 0) {
        for (int i = 0; i < g.n; ++i) cout << (i ? " 0" : "0");
        cout << '\n';
        return 0;
    }
    if (g.k == 1) {
        for (int i = 0; i < g.n; ++i) cout << (i ? " 0" : "0");
        cout << '\n';
        return 0;
    }

    bool solved = false;
    if (g.n <= 160 && g.edges.size() <= 6000) {
        solved = dsaturSmall(g, color, timer);
    }

    if (!solved) {
        vector<int> order;
        double avgDegree = g.n ? (2.0 * static_cast<double>(g.edges.size()) / static_cast<double>(g.n)) : 0.0;
        if (g.n <= 250000 && avgDegree < 200.0) {
            order = smallestLastOrder(g);
        } else {
            order = largestDegreeOrder(g);
        }
        greedyColor(g, order, color);
        solved = verify(g, color);
    }

    if (!solved) {
        solved = repairColoring(g, color, timer);
    }

    if (!verify(g, color)) {
        vector<int> order = largestDegreeOrder(g);
        greedyColor(g, order, color);
        repairColoring(g, color, timer);
    }

    for (int i = 0; i < g.n; ++i) {
        if (i) cout << ' ';
        int c = (i < static_cast<int>(color.size()) && color[i] >= 0 && color[i] < g.k) ? color[i] : (i % g.k);
        cout << c;
    }
    cout << '\n';
    return 0;
}
