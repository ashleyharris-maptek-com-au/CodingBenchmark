use std::collections::VecDeque;
use std::io::{self, Read, Write};

#[derive(Clone)]
struct Edge {
    u: usize,
    v: usize,
}

#[derive(Clone)]
struct Component {
    nodes: Vec<usize>,
    edge_count: usize,
}

struct Rng {
    s: u64,
}

impl Rng {
    fn new(seed: u64) -> Self {
        Self { s: seed | 1 }
    }
    fn next_u64(&mut self) -> u64 {
        let mut x = self.s;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.s = x;
        x.wrapping_mul(2685821657736338717)
    }
    fn f64(&mut self) -> f64 {
        ((self.next_u64() >> 11) as f64) * (1.0 / ((1u64 << 53) as f64))
    }
}

fn hash01(mut x: u64) -> f64 {
    x = (x ^ (x >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
    x = (x ^ (x >> 27)).wrapping_mul(0x94d049bb133111eb);
    x ^= x >> 31;
    ((x >> 11) as f64) * (1.0 / ((1u64 << 53) as f64))
}

fn bfs_order(start: usize, in_comp: &[bool], adj: &[Vec<usize>], n_hint: usize) -> (Vec<usize>, Vec<i32>) {
    let mut dist = vec![-1i32; adj.len()];
    let mut q = VecDeque::with_capacity(n_hint);
    let mut order = Vec::with_capacity(n_hint);
    dist[start] = 0;
    q.push_back(start);
    while let Some(u) = q.pop_front() {
        order.push(u);
        for &v in &adj[u] {
            if in_comp[v] && dist[v] < 0 {
                dist[v] = dist[u] + 1;
                q.push_back(v);
            }
        }
    }
    (order, dist)
}

fn farthest_from(start: usize, in_comp: &[bool], adj: &[Vec<usize>], n_hint: usize) -> usize {
    let (order, dist) = bfs_order(start, in_comp, adj, n_hint);
    let mut best = start;
    let mut best_d = -1;
    for u in order {
        if dist[u] > best_d || (dist[u] == best_d && adj[u].len() > adj[best].len()) {
            best = u;
            best_d = dist[u];
        }
    }
    best
}

fn circular_component(comp: &Component, adj: &[Vec<usize>], pos: &mut [(f64, f64)], ox: f64, oy: f64, scale: f64) {
    let k = comp.nodes.len();
    if k == 1 {
        pos[comp.nodes[0]] = (ox, oy);
        return;
    }
    let mut nodes = comp.nodes.clone();
    nodes.sort_by_key(|&u| (std::cmp::Reverse(adj[u].len()), u));
    let r = scale * (k as f64).sqrt().max(1.0);
    for (i, &u) in nodes.iter().enumerate() {
        let a = 2.0 * std::f64::consts::PI * (i as f64) / (k as f64);
        pos[u] = (ox + r * a.cos(), oy + r * a.sin());
    }
}

fn tree_or_sparse_layout(comp: &Component, adj: &[Vec<usize>], pos: &mut [(f64, f64)], ox: f64, oy: f64, scale: f64) {
    let k = comp.nodes.len();
    if k <= 2 {
        for (i, &u) in comp.nodes.iter().enumerate() {
            pos[u] = (ox + (i as f64 - 0.5) * scale, oy);
        }
        return;
    }
    if k <= 16 {
        circular_component(comp, adj, pos, ox, oy, scale);
        return;
    }
    let mut in_comp = vec![false; adj.len()];
    for &u in &comp.nodes {
        in_comp[u] = true;
    }
    let seed = comp.nodes.iter().copied().max_by_key(|&u| adj[u].len()).unwrap_or(comp.nodes[0]);
    let a = farthest_from(seed, &in_comp, adj, k);
    let b = farthest_from(a, &in_comp, adj, k);
    let (_, da) = bfs_order(a, &in_comp, adj, k);
    let (_, db) = bfs_order(b, &in_comp, adj, k);
    let diam = da[b].max(1) as f64;
    let mut layers: Vec<Vec<usize>> = vec![Vec::new(); diam as usize + 1];
    for &u in &comp.nodes {
        let t = if da[u] >= 0 { da[u] as usize } else { 0 };
        let idx = t.min(layers.len() - 1);
        layers[idx].push(u);
    }
    for layer in &mut layers {
        layer.sort_by_key(|&u| (db[u].max(0), u));
    }
    let width = scale * diam.max(1.0);
    for (i, layer) in layers.iter().enumerate() {
        let x = ox + ((i as f64 / diam) - 0.5) * width;
        let cnt = layer.len();
        for (j, &u) in layer.iter().enumerate() {
            let y = oy + (j as f64 - (cnt.saturating_sub(1) as f64) * 0.5) * scale;
            pos[u] = (x, y);
        }
    }
}

fn spectral_layout(comp: &Component, adj: &[Vec<usize>], pos: &mut [(f64, f64)], ox: f64, oy: f64, scale: f64) {
    let k = comp.nodes.len();
    if k <= 200 && comp.edge_count <= k + 2 {
        tree_or_sparse_layout(comp, adj, pos, ox, oy, scale);
        return;
    }
    if k <= 3 {
        circular_component(comp, adj, pos, ox, oy, scale);
        return;
    }

    let mut local = vec![usize::MAX; adj.len()];
    for (i, &u) in comp.nodes.iter().enumerate() {
        local[u] = i;
    }
    let mut x = vec![0.0; k];
    let mut y = vec![0.0; k];
    for (i, &u) in comp.nodes.iter().enumerate() {
        let h = u as u64 + 0x9e3779b97f4a7c15;
        x[i] = hash01(h) - 0.5;
        y[i] = hash01(h ^ 0xd1b54a32d192ed03) - 0.5;
    }
    let iters = if k <= 100 { 80 } else if k <= 1000 { 35 } else if k <= 10000 { 14 } else { 6 };
    let alpha = 0.58;
    let beta = 0.035;
    for _ in 0..iters {
        let mean_x = x.iter().sum::<f64>() / k as f64;
        let mean_y = y.iter().sum::<f64>() / k as f64;
        for i in 0..k {
            x[i] -= mean_x;
            y[i] -= mean_y;
        }
        let mut nx = vec![0.0; k];
        let mut ny = vec![0.0; k];
        for (i, &u) in comp.nodes.iter().enumerate() {
            let deg = adj[u].len().max(1) as f64;
            let mut sx = 0.0;
            let mut sy = 0.0;
            for &v in &adj[u] {
                let j = local[v];
                if j != usize::MAX {
                    sx += x[j];
                    sy += y[j];
                }
            }
            nx[i] = alpha * x[i] + (1.0 - alpha) * sx / deg + beta * (hash01((u as u64) ^ 17) - 0.5);
            ny[i] = alpha * y[i] + (1.0 - alpha) * sy / deg + beta * (hash01((u as u64) ^ 31) - 0.5);
        }
        x = nx;
        y = ny;
    }
    let mut max_r: f64 = 1e-9;
    for i in 0..k {
        max_r = max_r.max((x[i] * x[i] + y[i] * y[i]).sqrt());
    }
    let rscale = scale * (k as f64).sqrt().max(2.0) / max_r;
    for (i, &u) in comp.nodes.iter().enumerate() {
        let jitter = scale * 0.02;
        pos[u] = (
            ox + x[i] * rscale + jitter * (hash01(u as u64) - 0.5),
            oy + y[i] * rscale + jitter * (hash01((u as u64) << 1) - 0.5),
        );
    }
}

fn force_refine(comp: &Component, edges: &[Edge], pos: &mut [(f64, f64)], ox: f64, oy: f64, scale: f64) {
    let k = comp.nodes.len();
    if k < 2 || k > 700 {
        return;
    }
    let mut local = vec![usize::MAX; pos.len()];
    for (i, &u) in comp.nodes.iter().enumerate() {
        local[u] = i;
    }
    let area = (scale * scale * k as f64).max(1.0);
    let ideal = (area / k as f64).sqrt();
    let iters = if k <= 60 { 170 } else if k <= 180 { 90 } else { 35 };
    let mut disp = vec![(0.0, 0.0); k];
    let mut rng = Rng::new((k as u64).wrapping_mul(11995408973635179863));
    for it in 0..iters {
        for d in &mut disp {
            *d = (0.0, 0.0);
        }
        let temp = scale * (1.0 - it as f64 / iters as f64).max(0.05);
        for i in 0..k {
            for j in (i + 1)..k {
                let ui = comp.nodes[i];
                let uj = comp.nodes[j];
                let mut dx = pos[ui].0 - pos[uj].0;
                let mut dy = pos[ui].1 - pos[uj].1;
                let mut d2 = dx * dx + dy * dy;
                if d2 < 1e-9 {
                    dx = rng.f64() - 0.5;
                    dy = rng.f64() - 0.5;
                    d2 = dx * dx + dy * dy + 1e-9;
                }
                let d = d2.sqrt();
                let f = (ideal * ideal / d).min(scale * 4.0);
                let fx = dx / d * f;
                let fy = dy / d * f;
                disp[i].0 += fx;
                disp[i].1 += fy;
                disp[j].0 -= fx;
                disp[j].1 -= fy;
            }
        }
        for e in edges {
            let i = local[e.u];
            let j = local[e.v];
            if i == usize::MAX || j == usize::MAX || i == j {
                continue;
            }
            let dx = pos[e.u].0 - pos[e.v].0;
            let dy = pos[e.u].1 - pos[e.v].1;
            let d = (dx * dx + dy * dy).sqrt().max(1e-6);
            let f = (d * d / ideal) * 0.06;
            let fx = dx / d * f;
            let fy = dy / d * f;
            disp[i].0 -= fx;
            disp[i].1 -= fy;
            disp[j].0 += fx;
            disp[j].1 += fy;
        }
        for (i, &u) in comp.nodes.iter().enumerate() {
            let (dx, dy) = disp[i];
            let len = (dx * dx + dy * dy).sqrt().max(1e-9);
            let step = temp.min(len);
            pos[u].0 += dx / len * step;
            pos[u].1 += dy / len * step;
            pos[u].0 = ox + (pos[u].0 - ox) * 0.995;
            pos[u].1 = oy + (pos[u].1 - oy) * 0.995;
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next().and_then(|s| s.parse().ok()) {
        Some(v) => v,
        None => return,
    };
    let m: usize = it.next().and_then(|s| s.parse().ok()).unwrap_or(0);
    let mut adj = vec![Vec::<usize>::new(); n];
    let mut edges = Vec::with_capacity(m.min(10_000_000));
    for _ in 0..m {
        let Some(us) = it.next() else { break };
        let Some(vs) = it.next() else { break };
        let Ok(u) = us.parse::<usize>() else { continue };
        let Ok(v) = vs.parse::<usize>() else { continue };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
        edges.push(Edge { u, v });
    }

    let mut seen = vec![false; n];
    let mut comps = Vec::new();
    for s in 0..n {
        if seen[s] {
            continue;
        }
        seen[s] = true;
        let mut q = VecDeque::new();
        q.push_back(s);
        let mut nodes = Vec::new();
        let mut deg_sum = 0usize;
        while let Some(u) = q.pop_front() {
            nodes.push(u);
            deg_sum += adj[u].len();
            for &v in &adj[u] {
                if !seen[v] {
                    seen[v] = true;
                    q.push_back(v);
                }
            }
        }
        comps.push(Component {
            nodes,
            edge_count: deg_sum / 2,
        });
    }
    comps.sort_by_key(|c| std::cmp::Reverse(c.nodes.len()));

    let mut pos = vec![(0.0f64, 0.0f64); n];
    let mut cursor_x = 0.0f64;
    let mut cursor_y = 0.0f64;
    let mut row_h = 0.0f64;
    let total_side = (n as f64).sqrt().max(1.0) * 12.0;
    for comp in &comps {
        let k = comp.nodes.len();
        let radius = (k as f64).sqrt().max(1.0) * 8.0;
        if cursor_x + 2.0 * radius > total_side && cursor_x > 0.0 {
            cursor_x = 0.0;
            cursor_y += row_h + 25.0;
            row_h = 0.0;
        }
        let ox = cursor_x + radius;
        let oy = cursor_y + radius;
        let density = if k > 1 {
            comp.edge_count as f64 / k as f64
        } else {
            0.0
        };
        if k <= 700 {
            spectral_layout(comp, &adj, &mut pos, ox, oy, 9.0);
            if density <= 12.0 && comps.len() <= 64 {
                force_refine(comp, &edges, &mut pos, ox, oy, 9.0);
            }
        } else if density <= 1.3 {
            tree_or_sparse_layout(comp, &adj, &mut pos, ox, oy, 8.0);
        } else {
            spectral_layout(comp, &adj, &mut pos, ox, oy, 8.0);
        }
        cursor_x += 2.0 * radius + 25.0;
        row_h = row_h.max(2.0 * radius);
    }

    let mut out = String::with_capacity(n * 24);
    for (x, y) in pos {
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    io::stdout().write_all(out.as_bytes()).unwrap();
}
