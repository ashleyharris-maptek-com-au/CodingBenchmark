use std::cmp::Ordering;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    fn sub(self, other: Pt) -> Pt {
        Pt {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }
}

fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

fn convex_hull(mut pts: Vec<Pt>) -> Vec<Pt> {
    pts.sort_by(|a, b| {
        a.x.partial_cmp(&b.x)
            .unwrap_or(Ordering::Equal)
            .then(a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal))
    });
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1.0e-10 && (a.y - b.y).abs() < 1.0e-10);
    if pts.len() <= 2 {
        return pts;
    }
    let mut lower: Vec<Pt> = Vec::new();
    for p in &pts {
        while lower.len() >= 2 {
            let n = lower.len();
            if cross(lower[n - 1].sub(lower[n - 2]), (*p).sub(lower[n - 1])) > 1.0e-12 {
                break;
            }
            lower.pop();
        }
        lower.push(*p);
    }
    let mut upper: Vec<Pt> = Vec::new();
    for p in pts.iter().rev() {
        while upper.len() >= 2 {
            let n = upper.len();
            if cross(upper[n - 1].sub(upper[n - 2]), (*p).sub(upper[n - 1])) > 1.0e-12 {
                break;
            }
            upper.pop();
        }
        upper.push(*p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let mut j = poly.len() - 1;
    for i in 0..poly.len() {
        let pi = poly[i];
        let pj = poly[j];
        let crosses = (pi.y > p.y) != (pj.y > p.y);
        if crosses {
            let x_at_y = (pj.x - pi.x) * (p.y - pi.y) / (pj.y - pi.y) + pi.x;
            if p.x <= x_at_y {
                inside = !inside;
            }
        }
        j = i;
    }
    inside
}

fn point_on_segment(p: Pt, a: Pt, b: Pt) -> bool {
    let ab = b.sub(a);
    let ap = p.sub(a);
    cross(ab, ap).abs() < 1.0e-9
        && p.x >= a.x.min(b.x) - 1.0e-9
        && p.x <= a.x.max(b.x) + 1.0e-9
        && p.y >= a.y.min(b.y) - 1.0e-9
        && p.y <= a.y.max(b.y) + 1.0e-9
}

fn segment_intersects_poly(a: Pt, b: Pt, poly: &[Pt]) -> bool {
    for i in 0..poly.len() {
        let c = poly[i];
        let d = poly[(i + 1) % poly.len()];
        let ab = b.sub(a);
        let cd = d.sub(c);
        let ac = c.sub(a);
        let ad = d.sub(a);
        let ca = a.sub(c);
        let cb = b.sub(c);
        let c1 = cross(ab, ac);
        let c2 = cross(ab, ad);
        let c3 = cross(cd, ca);
        let c4 = cross(cd, cb);
        if (c1 >= -1.0e-9 && c2 <= 1.0e-9 || c1 <= 1.0e-9 && c2 >= -1.0e-9)
            && (c3 >= -1.0e-9 && c4 <= 1.0e-9 || c3 <= 1.0e-9 && c4 >= -1.0e-9)
        {
            return true;
        }
        if point_on_segment(a, c, d) || point_on_segment(b, c, d) {
            return true;
        }
    }
    false
}

fn rect_touches_poly(cx: f64, cy: f64, half: f64, poly: &[Pt]) -> bool {
    let corners = [
        Pt {
            x: cx - half,
            y: cy - half,
        },
        Pt {
            x: cx + half,
            y: cy - half,
        },
        Pt {
            x: cx + half,
            y: cy + half,
        },
        Pt {
            x: cx - half,
            y: cy + half,
        },
    ];
    for p in &corners {
        if point_in_poly(*p, poly) {
            return true;
        }
    }
    for p in poly {
        if p.x >= cx - half && p.x <= cx + half && p.y >= cy - half && p.y <= cy + half {
            return true;
        }
    }
    for i in 0..4 {
        if segment_intersects_poly(corners[i], corners[(i + 1) % 4], poly) {
            return true;
        }
    }
    false
}

fn footprint_inradius(sx: f64, sy: f64, sz: f64, scale: f64) -> f64 {
    let inv_sz = if sz.abs() < 1.0e-9 {
        if sz.is_sign_negative() {
            -1.0e9
        } else {
            1.0e9
        }
    } else {
        1.0 / sz
    };
    let verts = [
        (1.0, 0.0, -0.707),
        (-1.0, 0.0, -0.707),
        (0.0, 1.0, 0.707),
        (0.0, -1.0, 0.707),
    ];
    let mut pts = Vec::with_capacity(4);
    for (x, y, z) in verts {
        let t = -z * inv_sz;
        pts.push(Pt {
            x: scale * (x + t * sx),
            y: scale * (y + t * sy),
        });
    }
    let hull = convex_hull(pts);
    let mut best = f64::INFINITY;
    for i in 0..hull.len() {
        let a = hull[i];
        let b = hull[(i + 1) % hull.len()];
        let e = b.sub(a);
        let len = (e.x * e.x + e.y * e.y).sqrt();
        if len > 1.0e-12 {
            best = best.min(cross(e, Pt { x: -a.x, y: -a.y }).abs() / len);
        }
    }
    best.max(0.05)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let nv = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return,
    };
    let mut poly = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
        let y = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
        poly.push(Pt { x, y });
    }
    let sx = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
    let sy = it.next().unwrap_or("0").parse::<f64>().unwrap_or(0.0);
    let sz = it.next().unwrap_or("-1").parse::<f64>().unwrap_or(-1.0);
    if poly.is_empty() {
        println!("0");
        return;
    }

    let mut min_x = f64::INFINITY;
    let mut max_x = f64::NEG_INFINITY;
    let mut min_y = f64::INFINITY;
    let mut max_y = f64::NEG_INFINITY;
    for p in &poly {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }

    let width = max_x - min_x;
    let height = max_y - min_y;
    let scale = 10.0_f64;
    let inradius = footprint_inradius(sx, sy, sz, scale) * 0.92;
    let single_limit = inradius * 2.0_f64.sqrt();
    let mut centers: Vec<Pt> = Vec::new();

    if width <= single_limit && height <= single_limit && nv <= 64 {
        centers.push(Pt {
            x: (min_x + max_x) * 0.5,
            y: (min_y + max_y) * 0.5,
        });
    } else {
        let step = (inradius * 2.0_f64.sqrt()).max(0.08);
        let half = step * 0.5;
        let start_x = min_x - half;
        let end_x = max_x + half;
        let start_y = min_y - half;
        let end_y = max_y + half;
        let dense_check = nv <= 20_000 && ((width / step + 3.0) * (height / step + 3.0)) <= 2_000_000.0;
        let mut y = start_y;
        while y <= end_y + 1.0e-9 {
            let mut x = start_x;
            while x <= end_x + 1.0e-9 {
                if !dense_check || rect_touches_poly(x, y, half, &poly) {
                    centers.push(Pt { x, y });
                }
                x += step;
            }
            y += step;
        }
    }

    println!("{}", centers.len());
    let z_gap = 3.0 * scale + 10.0;
    let inv_sz = if sz.abs() < 1.0e-9 { 0.0 } else { 1.0 / sz };
    for (i, c) in centers.iter().enumerate() {
        let z = scale * 0.707 + 1.0 + i as f64 * z_gap;
        let px = c.x + z * sx * inv_sz;
        let py = c.y + z * sy * inv_sz;
        println!("{:.9} {:.9} {:.9} 1.0 0.0 0.0 0.0 {:.9}", px, py, z, scale);
    }
}
