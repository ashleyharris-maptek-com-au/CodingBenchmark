    Move searchBestMove(double timeLimitSeconds) {
        startTime = chrono::steady_clock::now();
        double allocated = max(0.01, timeLimitSeconds * 0.95);
        stopTime = startTime + chrono::duration<double>(allocated);
        timeUp = false;

        pvMove = Move();

        Move bestOverall;

        for (int depth = 1; depth <= MAX_DEPTH; ++depth) {
            if (timeUp) break;
            Move iterationBest;
            int score = searchRoot(depth, iterationBest);
            if (timeUp) break;
            if (iterationBest.from != -1) {
                bestOverall = iterationBest;
                haveBest = true;
                pvMove = iterationBest;
                lastScore = score;
            }
            if (score > MATE_SCORE - 1000 || score < -MATE_SCORE + 1000) {
                break; // found mate
            }

        if (!haveBest) {
            vector<Move> moves;
            generatePseudoMoves(moves, false);
            for (const Move& m : moves) {
                if (makeMove(m)) {
                    undoMove();
                    bestOverall = m;
                    haveBest = true;
                    break;
                }
            }
        }

        if (!haveBest) {
            Move nullMove;
            nullMove.from = -1;
            return nullMove;
        }
        return bestOverall;
    }

    string moveToNotation(const Move& move) {
        if (move.from < 0) {
            return "0000";

                notation = "O-O";
            } else {
                notation = "O-O-O";
            }
        } else {
            char pieceChar = 'p';
            switch (abs(move.piece)) {
            case 1: pieceChar = 'p'; break;
            case 2: pieceChar = 'N'; break;
            case 3: pieceChar = 'B'; break;
            case 4: pieceChar = 'R'; break;
            case 5: pieceChar = 'Q'; break;
            case 6: pieceChar = 'K'; break;
            }
            notation += pieceChar;
            notation += squareToString(move.from);
            if (move.flags & FLAG_CAPTURE) {
                notation += 'x';
            }
            notation += squareToString(move.to);
            if (move.flags & FLAG_PROMOTION) {
                char promoChar = 'Q';
                switch (move.promotion) {
                case 2: promoChar = 'N'; break;
                case 3: promoChar = 'B'; break;
                case 4: promoChar = 'R'; break;
                case 5: promoChar = 'Q'; break;
                default: promoChar = 'Q'; break;
                }
                notation += '/';
                notation += promoChar;
            }
            if (move.flags & FLAG_ENPASSANT) {
                notation += "ep";
            }
        }

        if (makeMove(move)) {
            bool givesCheck = inCheck();
            bool mate = false;
            if (givesCheck) {
                mate = !hasLegalMoves();
            }
            undoMove();
            if (mate) {
                notation += '#';
            } else if (givesCheck) {
                notation += '+';
            }
        }
        return notation;
    }

    bool makeMove(const Move& move) {
        Undo undo;
        doMove(move, undo);

        int movedSide = -pos.side; // after doMove, side has been flipped
        int kingIndex = (movedSide == 1) ? 0 : 1;
        if (isSquareAttacked(pos.kingSq[kingIndex], pos.side)) {
            revertMove(undo);
            return false;
        }

        history.push_back(undo);
        return true;
    }

    void undoMove() {
        if (history.empty()) return;
        Undo undo = history.back();
        history.pop_back();
        revertMove(undo);
    }

    void doMove(const Move& move, Undo& undo) {
        undo.move = move;
        undo.side = pos.side;
        undo.castlingRights = pos.castlingRights;
        undo.enPassant = pos.enPassant;
        undo.halfmove = pos.halfmove;
        undo.fullmove = pos.fullmove;
        undo.kingSq[0] = pos.kingSq[0];
        undo.kingSq[1] = pos.kingSq[1];

        int side = pos.side;
        int piece = pos.board[move.from];
        undo.movedPiece = piece;

        pos.enPassant = -1;

        if (abs(piece) == 1 || (move.flags & FLAG_CAPTURE)) {
            pos.halfmove = 0;
        } else {
            pos.halfmove++;
        }

        pos.board[move.from] = 0;

        int capturedPiece = 0;
        if (move.flags & FLAG_ENPASSANT) {
            int capSq = (side == 1) ? move.to - 8 : move.to + 8;
            capturedPiece = pos.board[capSq];
            pos.board[capSq] = 0;
        } else {
            capturedPiece = pos.board[move.to];
        }
        undo.capturedPiece = capturedPiece;

        int newPiece = piece;
        if (move.flags & FLAG_PROMOTION) {
            int promoted = move.promotion;
            newPiece = (side == 1) ? promoted : -promoted;
        }
        pos.board[move.to] = newPiece;

        if (move.flags & FLAG_CASTLE) {
            if (side == 1) {
                if (move.to == 6) { // g1
                    pos.board[5] = pos.board[7];
                    pos.board[7] = 0;
                } else if (move.to == 2) { // c1
                    pos.board[3] = pos.board[0];
                    pos.board[0] = 0;
                }
            } else {
                if (move.to == 62) { // g8
                    pos.board[61] = pos.board[63];
                    pos.board[63] = 0;
                } else if (move.to == 58) { // c8
                    pos.board[59] = pos.board[56];
                    pos.board[56] = 0;
                }
            }
        }

        if (abs(piece) == 6) {
            pos.kingSq[(side == 1) ? 0 : 1] = move.to;
        }

        updateCastlingRights(move, piece, capturedPiece);

        if (move.flags & FLAG_DOUBLE_PAWN) {
            pos.enPassant = (side == 1) ? move.from + 8 : move.from - 8;
        }

        pos.side = -side;
        if (side == -1) {
            pos.fullmove++;
        }
    }

    void revertMove(const Undo& undo) {
        const Move& move = undo.move;

        pos.side = undo.side;
        pos.castlingRights = undo.castlingRights;
        pos.enPassant = undo.enPassant;
        pos.halfmove = undo.halfmove;
        pos.fullmove = undo.fullmove;
        pos.kingSq[0] = undo.kingSq[0];
        pos.kingSq[1] = undo.kingSq[1];

        pos.board[move.from] = undo.movedPiece;

            pos.board[move.to] = 0;
            int capSq = (undo.side == 1) ? move.to - 8 : move.to + 8;
            pos.board[capSq] = undo.capturedPiece;
        } else {
            pos.board[move.to] = undo.capturedPiece;
        }

        if (move.flags & FLAG_CASTLE) {
            if (undo.side == 1) {
                if (move.to == 6) {
                    pos.board[7] = pos.board[5];
                    pos.board[5] = 0;
                } else if (move.to == 2) {
                    pos.board[0] = pos.board[3];

                    pos.board[59] = 0;
                }
            }
        }
    }

    void updateCastlingRights(const Move& move, int piece, int capturedPiece) {
        int from = move.from;
        int to = move.to;
        int absPiece = abs(piece);

        if (absPiece == 6) {
            if (piece > 0) {
                pos.castlingRights &= ~(1 | 2);
            } else {
                pos.castlingRights &= ~(4 | 8);
            }
        }

        if (absPiece == 4) {
            if (piece > 0) {
                if (from == 0) pos.castlingRights &= ~2;
                if (from == 7) pos.castlingRights &= ~1;
            } else {
                if (from == 56) pos.castlingRights &= ~8;
                if (from == 63) pos.castlingRights &= ~4;
            }
        }

        if (capturedPiece != 0 && abs(capturedPiece) == 4) {
            if (to == 0) pos.castlingRights &= ~2;
            if (to == 7) pos.castlingRights &= ~1;
            if (to == 56) pos.castlingRights &= ~8;
            if (to == 63) pos.castlingRights &= ~4;
        }
    }

    bool isSquareAttacked(int square, int bySide) const {
        const int file = square % 8;
        const int rank = square / 8;

        if (bySide == 1) {
            if (file > 0) {
                int from = square - 9;
                if (from >= 0 && pos.board[from] == 1) return true;
            }
            if (file < 7) {
                int from = square - 7;
                if (from >= 0 && pos.board[from] == 1) return true;
            }
        } else {
            if (file > 0) {
                int from = square + 7;
                if (from < 64 && pos.board[from] == -1) return true;
            }
            if (file < 7) {
                int from = square + 9;
                if (from < 64 && pos.board[from] == -1) return true;
            }
        }

        static const int knightOffsets[8] = { 15, 17, 10, 6, -6, -10, -15, -17 };
        for (int offset : knightOffsets) {
            int fromSq = square + offset;
            if (fromSq < 0 || fromSq >= 64) continue;
            int fromFile = fromSq % 8;
            int fromRank = fromSq / 8;
            int dx = abs(fromFile - file);
            int dy = abs(fromRank - rank);
            if (!((dx == 1 && dy == 2) || (dx == 2 && dy == 1))) continue;
            int piece = pos.board[fromSq];
            if (piece != 0 && ((piece > 0) == (bySide == 1)) && abs(piece) == 2) {
                return true;
            }
        }

        static const int bishopDirs[4] = { 9, 7, -7, -9 };
        for (int dir : bishopDirs) {
            int current = square;
            while (true) {
                int next = current + dir;
                if (next < 0 || next >= 64) break;
                int currentFile = current % 8;
                int nextFile = next % 8;
                if (abs(nextFile - currentFile) != 1) break;
                int piece = pos.board[next];
                if (piece != 0) {
                    if ((piece > 0) == (bySide == 1)) {
                        int type = abs(piece);
                        if (type == 3 || type == 5) return true;
                    }
                    break;
                }
                current = next;
            }
        }

        static const int rookDirs[4] = { 1, -1, 8, -8 };
        for (int dir : rookDirs) {
            int current = square;
            while (true) {
                int next = current + dir;
                if (next < 0 || next >= 64) break;
                int currentFile = current % 8;
                int nextFile = next % 8;
                if (dir == 1 && nextFile - currentFile != 1) break;
                if (dir == -1 && nextFile - currentFile != -1) break;
                int piece = pos.board[next];
                if (piece != 0) {
                    if ((piece > 0) == (bySide == 1)) {
                        int type = abs(piece);
                        if (type == 4 || type == 5) return true;
                    }
                    break;
                }
                current = next;
            }
        }

        static const int kingOffsets[8] = { 1, -1, 8, -8, 9, 7, -9, -7 };
        for (int offset : kingOffsets) {
            int fromSq = square + offset;
            if (fromSq < 0 || fromSq >= 64) continue;
            int fromFile = fromSq % 8;
            int fromRank = fromSq / 8;
            int dx = abs(fromFile - file);
            int dy = abs(fromRank - rank);
            if (dx > 1 || dy > 1) continue;
            int piece = pos.board[fromSq];
            if (piece != 0 && ((piece > 0) == (bySide == 1)) && abs(piece) == 6) {
                return true;
            }
        }

        return false;
    }

                                moves.push_back(m);
                            }
                        }
                    }
                    if (file < 7) {
                        int target = sq - 7;
                        if (target >= 0 && pos.board[target] > 0) {
                            if (rank == 1) {
                                for (int promo : promotionPieces) {
                                    Move m;
                                    m.from = sq;
                                    m.to = target;
                                    m.piece = piece;
                                    m.promotion = promo;
                                    m.captured = pos.board[target];
                                    m.flags = FLAG_CAPTURE | FLAG_PROMOTION;
                                    moves.push_back(m);
                                }
                            } else {
                                Move m;
                                m.from = sq;
                                m.to = target;
                                m.piece = piece;
                                m.captured = pos.board[target];
                                m.flags = FLAG_CAPTURE;
                                moves.push_back(m);
                            }
                        }
                    }
                    if (pos.enPassant != -1) {
                        if (file > 0 && pos.enPassant == sq - 9) {
                            Move m;
                            m.from = sq;
                            m.to = pos.enPassant;
                            m.piece = piece;
                            m.captured = 1;
                            m.flags = FLAG_CAPTURE | FLAG_ENPASSANT;
                            moves.push_back(m);
                        }
                        if (file < 7 && pos.enPassant == sq - 7) {
                            Move m;
                            m.from = sq;
                            m.to = pos.enPassant;
                            m.piece = piece;
                            m.captured = 1;
                            m.flags = FLAG_CAPTURE | FLAG_ENPASSANT;
                            moves.push_back(m);
                        }
                    }
                }
            } else if (type == 2) {
                for (int offset : knightOffsets) {
                    int target = sq + offset;
                    if (target < 0 || target >= 64) continue;
                    int targetFile = target % 8;
                    int targetRank = target / 8;
                    int dx = abs(targetFile - file);
                    int dy = abs(targetRank - rank);
                    if (!((dx == 1 && dy == 2) || (dx == 2 && dy == 1))) continue;
                    int destPiece = pos.board[target];
                    if (destPiece == 0) {
                        if (captureOnly) continue;
                        Move m;
                        m.from = sq;
                        m.to = target;
                        m.piece = piece;
                        m.captured = 0;
                        moves.push_back(m);
                    } else if ((destPiece > 0) != (piece > 0)) {
                        Move m;
                        m.from = sq;
                        m.to = target;
                        m.piece = piece;
                        m.captured = destPiece;
                        m.flags = FLAG_CAPTURE;
                        moves.push_back(m);
                    }
                }
            } else if (type == 3 || type == 5) {
                static const int dirs[4] = { 9, 7, -7, -9 };
                for (int dir : dirs) {
                    int current = sq;
                    while (true) {
                        int next = current + dir;
                        if (next < 0 || next >= 64) break;
                        int currentFile = current % 8;
                        int nextFile = next % 8;
                        if (abs(nextFile - currentFile) != 1) break;
                        int destPiece = pos.board[next];
                        if (destPiece == 0) {
                            if (!captureOnly) {
                                Move m;
                                m.from = sq;
                                m.to = next;
                                m.piece = piece;
                                m.captured = 0;
                                moves.push_back(m);
                            }
                        } else {
                            if ((destPiece > 0) != (piece > 0)) {
                                Move m;
                                m.from = sq;
                                m.to = next;
                                m.piece = piece;
                                m.captured = destPiece;
                                m.flags = FLAG_CAPTURE;
                                moves.push_back(m);
                            }
                            break;
                        }
                        current = next;
                    }
                }
            }

            if (type == 4 || type == 5) {
                static const int dirs[4] = { 1, -1, 8, -8 };
                for (int dir : dirs) {
                    int current = sq;
                    while (true) {
                        int next = current + dir;
                        if (next < 0 || next >= 64) break;
                        int currentFile = current % 8;
                        int nextFile = next % 8;
                        if (dir == 1 && nextFile - currentFile != 1) break;
                        if (dir == -1 && nextFile - currentFile != -1) break;
                        int destPiece = pos.board[next];
                        if (destPiece == 0) {
                            if (!captureOnly) {
                                Move m;
                                m.from = sq;
                                m.to = next;
                                m.piece = piece;
                                m.captured = 0;
                                moves.push_back(m);
                            }
                        } else {
                            if ((destPiece > 0) != (piece > 0)) {
                                Move m;
                                m.from = sq;
                                m.to = next;
                                m.piece = piece;
                                m.captured = destPiece;
                                m.flags = FLAG_CAPTURE;
                                moves.push_back(m);
                            }
                            break;
                        }
                        current = next;
                    }
                }
            }

            if (type == 6) {
                for (int offset : kingOffsets) {
                    int target = sq + offset;
                    if (target < 0 || target >= 64) continue;
                    int targetFile = target % 8;
                    int targetRank = target / 8;
                    int dx = abs(targetFile - file);
                    int dy = abs(targetRank - rank);
                    if (dx > 1 || dy > 1) continue;
                    int destPiece = pos.board[target];
                    if (destPiece == 0) {
                        if (!captureOnly) {
                            Move m;
                            m.from = sq;
                            m.to = target;
                            m.piece = piece;
                            m.captured = 0;
                            moves.push_back(m);
                        }
                    } else if ((destPiece > 0) != (piece > 0)) {
                        Move m;
                        m.from = sq;
                        m.to = target;
                        m.piece = piece;
                        m.captured = destPiece;
                        m.flags = FLAG_CAPTURE;
                        moves.push_back(m);
                    }
                }
                if (!captureOnly) {
                    if (piece > 0) {
                        if ((pos.castlingRights & 1) && pos.board[5] == 0 && pos.board[6] == 0) {
                            if (!isSquareAttacked(sq, -pos.side) && !isSquareAttacked(5, -pos.side) && !isSquareAttacked(6, -pos.side)) {
                                Move m;
                                m.from = sq;
                                m.to = 6;
                                m.piece = piece;
                                m.captured = 0;
                                m.flags = FLAG_CASTLE;
                                moves.push_back(m);
                            }
                        }
                        if ((pos.castlingRights & 2) && pos.board[1] == 0 && pos.board[2] == 0 && pos.board[3] == 0) {
                            if (!isSquareAttacked(sq, -pos.side) && !isSquareAttacked(3, -pos.side) && !isSquareAttacked(2, -pos.side)) {
                                Move m;
                                m.from = sq;
                                m.to = 2;
                                m.piece = piece;
                                m.captured = 0;
                                m.flags = FLAG_CASTLE;
                                moves.push_back(m);
                            }
                        }
                    } else {
                        if ((pos.castlingRights & 4) && pos.board[61] == 0 && pos.board[62] == 0) {
                            if (!isSquareAttacked(sq, -pos.side) && !isSquareAttacked(61, -pos.side) && !isSquareAttacked(62, -pos.side)) {
                                Move m;
                                m.from = sq;
                                m.to = 62;
                                m.piece = piece;
                                m.captured = 0;
                                m.flags = FLAG_CASTLE;
                                moves.push_back(m);
                            }
                        }
                        if ((pos.castlingRights & 8) && pos.board[57] == 0 && pos.board[58] == 0 && pos.board[59] == 0) {
                            if (!isSquareAttacked(sq, -pos.side) && !isSquareAttacked(59, -pos.side) && !isSquareAttacked(58, -pos.side)) {
                                Move m;
                                m.from = sq;
                                m.to = 58;
                                m.piece = piece;
                                m.captured = 0;
                                m.flags = FLAG_CASTLE;
                                moves.push_back(m);
                            }
                        }
                    }
                }
            }
        }
    }

    int evaluate() const {
        int score = 0;
        for (int sq = 0; sq < 64; ++sq) {
            int piece = pos.board[sq];
            if (piece == 0) continue;
            int type = abs(piece);
            int val = pieceValues[type];
            int pst = 0;
            switch (type) {
            case 1:
                pst = (piece > 0) ? pawnTable[sq] : -pawnTable[mirrorSquare(sq)];
                break;
            case 2:
                pst = (piece > 0) ? knightTable[sq] : -knightTable[mirrorSquare(sq)];
                break;
            case 3:
                pst = (piece > 0) ? bishopTable[sq] : -bishopTable[mirrorSquare(sq)];
                break;
            case 4:
                pst = (piece > 0) ? rookTable[sq] : -rookTable[mirrorSquare(sq)];
                break;
            case 5:
                pst = (piece > 0) ? queenTable[sq] : -queenTable[mirrorSquare(sq)];
                break;
            case 6:
                pst = (piece > 0) ? kingTable[sq] : -kingTable[mirrorSquare(sq)];
                break;
            default:
                break;
            }
            score += (piece > 0) ? (val + pst) : -(val + pst);
        }
        return (pos.side == 1) ? score : -score;
    }

    int quiescence(int alpha, int beta, int ply) {
        if ((++nodes & 1023) == 0) {
            if (checkTime()) return evaluate();
        }
        int standPat = evaluate();
        if (standPat >= beta) return standPat;
        if (standPat > alpha) alpha = standPat;

        vector<Move> moves;
        generatePseudoMoves(moves, true);
        sortMoves(moves, Move());

        for (const Move& m : moves) {
            if (!makeMove(m)) continue;
            int score = -quiescence(-beta, -alpha, ply + 1);
            undoMove();
            if (timeUp) return alpha;
            if (score >= beta) return score;
            if (score > alpha) alpha = score;
        }
        return alpha;
    }

    int negamax(int depth, int alpha, int beta, int ply) {
        if (timeUp) return evaluate();
        if ((++nodes & 1023) == 0) {
            if (checkTime()) return evaluate();
        }

        bool inChk = inCheck();
        if (depth <= 0) {
            if (inChk) depth = 1;
            else return quiescence(alpha, beta, ply);
        }

        vector<Move> moves;
        generatePseudoMoves(moves, false);
        sortMoves(moves, pvMove);

        int bestScore = -INF;
        bool foundMove = false;

        for (const Move& m : moves) {
            if (!makeMove(m)) continue;
            foundMove = true;
            int score = -negamax(depth - 1, -beta, -alpha, ply + 1);
            undoMove();

            if (timeUp) return alpha;

            if (score > bestScore) {
                bestScore = score;
            }
            if (score > alpha) {
                alpha = score;
            }
            if (alpha >= beta) {
                int sideIdx = (pos.side == 1) ? 1 : 0; // pos.side has been restored to side to move
                if (!(m.flags & FLAG_CAPTURE)) {
                    historyTable[sideIdx][m.from][m.to] += depth * depth;
                    if (historyTable[sideIdx][m.from][m.to] > 400000) {
                        historyTable[sideIdx][m.from][m.to] = 400000;
                    }
                }
                return beta;
            }
        }

        if (!foundMove) {
            if (inChk) {
                return -MATE_SCORE + ply;
            }
            return 0;
        }

        return alpha;
    }

    int moveScore(const Move& move, const Move& bestHint) const {
        int score = 0;
        if (sameMove(move, bestHint)) {
            score += 1'000'000;
        }
        if (move.flags & FLAG_CAPTURE) {
            int capturedVal = pieceValues[abs(move.captured)];
            int moverVal = pieceValues[abs(move.piece)];
            score += 500'000 + 10 * capturedVal - moverVal;
        }
        if (move.flags & FLAG_PROMOTION) {
            score += 400'000 + pieceValues[move.promotion];
        }
        if (move.flags & FLAG_CASTLE) {
            score += 50;
        }
        if (!(move.flags & (FLAG_CAPTURE | FLAG_PROMOTION))) {
            int sideIdx = (pos.side == 1) ? 0 : 1;
            score += historyTable[sideIdx][move.from][move.to];
        }
        return score;
    }

    void sortMoves(vector<Move>& moves, const Move& bestHint) const {
        sort(moves.begin(), moves.end(), [&](const Move& a, const Move& b) {
            return moveScore(a, bestHint) > moveScore(b, bestHint);
        });
    }

    int searchRoot(int depth, Move& outBest) {
        vector<Move> moves;
        generatePseudoMoves(moves, false);
        sortMoves(moves, pvMove);

        int alpha = -INF;
        int beta = INF;
        int bestScore = -INF;
        Move bestMove;

        bool foundLegal = false;

        for (const Move& m : moves) {
            if (!makeMove(m)) continue;
            foundLegal = true;
            int score = -negamax(depth - 1, -beta, -alpha, 1);
            undoMove();

            if (timeUp) return bestScore;

            if (score > bestScore || !bestMove.from) {
                bestScore = score;
                bestMove = m;
            }
            if (score > alpha) {
                alpha = score;
                bestMove = m;
            }
        }

        if (!foundLegal) {
            outBest.from = -1;
            if (inCheck()) {
                return -MATE_SCORE;
            }
            return 0;
        }

        outBest = bestMove;
        return bestScore;
    }
};

Position parseFEN(const string& fen) {
    Position pos{};
    fill(begin(pos.board), end(pos.board), 0);

    istringstream iss(fen);
    vector<string> tokens;
    string token;
    while (iss >> token) {
        tokens.push_back(token);
    }

    if (tokens.size() < 2) {
        return pos;
    }

    string boardPart = tokens[0];
    vector<string> ranks;
    string current;
    for (char c : boardPart) {
        if (c == '/') {
            ranks.push_back(current);
            current.clear();
        } else {
            current += c;
        }
    }
    ranks.push_back(current);

    for (int r = 0; r < 8 && r < static_cast<int>(ranks.size()); ++r) {
        const string& rank = ranks[r];
        int file = 0;
        for (char c : rank) {
            if (isdigit(static_cast<unsigned char>(c))) {
                file += c - '0';
            } else {
                int index = (7 - r) * 8 + file;
                int piece = 0;
                switch (c) {
                case 'P': piece = 1; break;
                case 'N': piece = 2; break;
                case 'B': piece = 3; break;
                case 'R': piece = 4; break;
                case 'Q': piece = 5; break;
                case 'K': piece = 6; break;
                case 'p': piece = -1; break;
                case 'n': piece = -2; break;
                case 'b': piece = -3; break;
                case 'r': piece = -4; break;
                case 'q': piece = -5; break;
                case 'k': piece = -6; break;
                default: break;
                }
                pos.board[index] = piece;
                if (piece == 6) pos.kingSq[0] = index;
                if (piece == -6) pos.kingSq[1] = index;
                file++;
            }
        }
    }

    pos.side = (tokens.size() >= 2 && tokens[1] == "b") ? -1 : 1;

    pos.castlingRights = 0;
    if (tokens.size() >= 3) {
        const string& cast = tokens[2];
        if (cast.find('K') != string::npos) pos.castlingRights |= 1;
        if (cast.find('Q') != string::npos) pos.castlingRights |= 2;
        if (cast.find('k') != string::npos) pos.castlingRights |= 4;
        if (cast.find('q') != string::npos) pos.castlingRights |= 8;
    }

    pos.enPassant = -1;
    if (tokens.size() >= 4 && tokens[3] != "-") {
        string ep = tokens[3];
        if (ep.size() == 2) {
            int file = ep[0] - 'a';
            int rank = ep[1] - '1';
            if (file >= 0 && file < 8 && rank >= 0 && rank < 8) {
                pos.enPassant = rank * 8 + file;
            }
        }
    }

    pos.halfmove = (tokens.size() >= 5) ? stoi(tokens[4]) : 0;
    pos.fullmove = (tokens.size() >= 6) ? stoi(tokens[5]) : 1;

    if (pos.kingSq[0] == -1 || pos.kingSq[1] == -1) {
        for (int sq = 0; sq < 64; ++sq) {
            if (pos.board[sq] == 6) pos.kingSq[0] = sq;
            if (pos.board[sq] == -6) pos.kingSq[1] = sq;
        }
    }

    return pos;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    double timeLimit;
    if (!(cin >> timeLimit)) {
        return 0;
    }
    string dummy;
    getline(cin, dummy); // consume rest of line

    string fen;
    getline(cin, fen);

    Position pos = parseFEN(fen);
    Engine engine;
    engine.setPosition(pos);

    Move bestMove = engine.searchBestMove(timeLimit);
    cout << engine.moveToNotation(bestMove) << "\n";
    return 0;
}
