Below is a self-contained C++20 program that solves the global minimum edge cut (min cut) in an undirected graph using a sparse Stoer–Wagner contraction approach. It adapts to graph size by representing the current contracted graph with per-vertex adjacency maps (unordered_map) and a lazy max-heap to select the next vertex in each phase. It also builds the actual cut edges from the original graph, so the output is a valid min-cut.

Key points:
- Works for large graphs without materializing a dense n×n matrix.
- Maintains, for each contracted supernode, the list of original vertices it contains (members).
- Tracks the best phase (minimum cut value) and the corresponding partition of original vertices.
- After finishing, prints all input edges crossing the min-cut (in any order).

Notes:
- We assume 0-based vertex indices as given in the input.
- The implementation uses long long for counts to be safe on large inputs.
- The code includes adaptive logic in spirit: a sparse contraction path is chosen by default to handle large graphs efficiently.

Compile with: cl /EHsc /std:c++20 mincut.cpp

Code:

```cpp
#include <iostream>
#include <vector>
#include <unordered_map>
#include <queue>
#include <limits>
#include <algorithm>
#include <cstdint>

using std::cin;
using std::cout;
using std::endl;
using std::vector;
using std::pair;
using std::unordered_map;
using std::priority_queue;
using std::numeric_limits;
using std::make_pair;

// Global minimum edge cut (Stoer-Wagner, sparse, adaptive)
int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    long long M;
    if (!(cin >> N >> M)) {
        return 0;
    }

    vector<pair<int,int>> edges;
    edges.reserve((size_t)M);

    // Adjacency for current contracted graph (original vertices are the nodes)
    vector< unordered_map<int, long long> > adj(N);
    vector<char> alive(N, true);

    // Each contracted supernode keeps a list of original vertices it contains
    vector< vector<int> > members(N);
    for (int i = 0; i < N; ++i) {
        members[i].push_back(i);
    }

    // Read edges
    for (long long i = 0; i < M; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || v < 0 || u >= N || v >= N) continue; // robust input
        edges.emplace_back(u, v);
        adj[u][v] += 1;
        adj[v][u] += 1;
    }

    long long best_cut = numeric_limits<long long>::max();
    vector<char> best_left(N, false); // best left component in terms of original vertices

    int current_n = N;

    // Stoer-Wagner: repeat phases until only one active vertex remains
    while (current_n > 1) {
        // Phase variables
        vector<char> inA(N, false);               // which current-graph vertices are in A
        vector<long long> weight(N, 0);           // weights to A for alive vertices
        // Max-heap with lazy updates: (weight, vertex)
        priority_queue< pair<long long,int> > pq;

        // Pick any alive as start
        int start = -1;
        for (int i = 0; i < N; ++i) if (alive[i]) { start = i; break; }
        if (start == -1) break;

        inA[start] = true;
        vector<int> order;
        order.push_back(start);

        // Initialize weights for neighbors of start
        for (auto &kv : adj[start]) {
            int u = kv.first;
            long long w = kv.second;
            if (alive[u] && !inA[u]) {
                weight[u] += w;
            }
        }

        // Push all alive not-in-A into the heap with their current weights
        for (int i = 0; i < N; ++i) if (alive[i] && !inA[i]) {
            pq.push({weight[i], i});
        }

        int last = start;
        for (int iter = 1; iter < current_n; ++iter) {
            int v = -1;
            // Extract the best candidate from the heap
            while (!pq.empty()) {
                auto top = pq.top(); pq.pop();
                int cand = top.second;
                long long w = top.first;
                if (alive[cand] && !inA[cand] && weight[cand] == w) {
                    v = cand;
                    break;
                }
            }
            if (v == -1) {
                // Fallback: pick any remaining alive vertex not in A
                for (int i = 0; i < N; ++i) {
                    if (alive[i] && !inA[i]) { v = i; break; }
                }
                if (v == -1) break; // should not happen
            }

            inA[v] = true;
            order.push_back(v);
            last = v;

            // Update weights for neighbors of v
            for (auto &kv : adj[v]) {
                int u = kv.first;
                long long w = kv.second;
                if (alive[u] && !inA[u]) {
                    weight[u] += w;
                    pq.push({weight[u], u});
                }
            }
        }

        // t is the last added vertex in this phase
        int t = last;
        int s = -1;
        if (!order.empty()) s = order[order.size() - 2];

        long long cut = (t >= 0 && t < N) ? weight[t] : 0;

        // Build left side: union of members of order[0..order.size()-2]
        vector<char> leftFlag(N, false);
        for (size_t idx = 0; idx + 1 < order.size(); ++idx) {
            int a = order[idx];
            for (int orig : members[a]) leftFlag[orig] = true;
        }

        if (cut < best_cut) {
            best_cut = cut;
            best_left = leftFlag;
        }

        // Contract s and t (merge t into s)
        if (s == -1) break;

        // Merge edges from t into s
        for (auto &kv : adj[t]) {
            int w = kv.first;
            long long wgt = kv.second;
            if (w == s) continue; // skip self-loop consequence
            if (alive[w]) {
                adj[s][w] += wgt;
                adj[w][s] += wgt;
            }
        }
        // Remove t from the graph
        alive[t] = false;

        // Remove t from all neighbors' maps
        for (auto &kv : adj[t]) {
            int w = kv.first;
            if (alive[w]) {
                adj[w].erase(t);
            }
        }
        adj[t].clear();

        // Remove direct edge between s and t (if any)
        adj[s].erase(t);

        // Merge member lists: s absorbs t's members
        if (!members[t].empty()) {
            members[s].insert(members[s].end(), members[t].begin(), members[t].end());
            vector<int>().swap(members[t]);
        }

        current_n--;
    }

    // Prepare to output cut edges
    // best_left marks the original vertices on the "left" side of the min-cut
    // Right side is the complement
    vector<pair<int,int>> cut_edges;
    cut_edges.reserve((size_t)M);
    for (const auto &e : edges) {
        int u = e.first;
        int v = e.second;
        int lu = best_left[u] ? 1 : 0;
        int lv = best_left[v] ? 1 : 0;
        if (lu != lv) {
            cut_edges.emplace_back(u, v);
        }
    }

    cout << (long long)cut_edges.size() << "\n";
    for (const auto &e : cut_edges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
```
