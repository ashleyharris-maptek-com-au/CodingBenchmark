import sys

def main():
    data = sys.stdin.read().splitlines()
    if not data:
        print("NO SOLUTION")
        return

    # Skip leading empty lines
    while data and data[0].strip() == "":
        data.pop(0)
    if not data:
        print("NO SOLUTION")
        return

    # Parse U and S
    first_line = data[0].strip().split()
    if len(first_line) < 2:
        print("NO SOLUTION")
        return
    U = int(first_line[0])
    S = int(first_line[1])

    # Read S lines of sets
    lines = data[1:]
    sets_masks = []
    for idx in range(S):
        if idx < len(lines):
            line = lines[idx].strip()
            if line == "":
                elements = []
            else:
                elements = list(map(int, line.split()))
        else:
            elements = []
        mask = 0
        for e in elements:
            if 0 <= e < U:
                mask |= 1 << e
        sets_masks.append(mask)

    UniverseMask = (1 << U) - 1 if U > 0 else 0

    # Build element -> list of set indices that contain the element
    element_to_sets = [[] for _ in range(U)]
    for i, mask in enumerate(sets_masks):
        m = mask
        epos = 0
        while m:
            if m & 1:
                element_to_sets[epos].append(i)
            m >>= 1
            epos += 1

    sys.setrecursionlimit(10000)

    best_solution = None

    def dfs(used_mask, last_index, chosen_indices):
        nonlocal best_solution
        if best_solution is not None:
            return True
        if used_mask == UniverseMask:
            best_solution = chosen_indices[:]
            return True

        remaining = UniverseMask & ~used_mask
        # Pick the remaining element with the fewest available (disjoint) sets
        best_e = -1
        best_count = None
        rem = remaining
        epos = 0
        while rem:
            if rem & 1:
                cnt = 0
                for si in element_to_sets[epos]:
                    if si > last_index and (sets_masks[si] & used_mask) == 0:
                        cnt += 1
                if cnt == 0:
                    return False  # dead end
                if best_count is None or cnt < best_count:
                    best_count = cnt
                    best_e = epos
                    if cnt == 1:
                        break
            rem >>= 1
            epos += 1

        if best_e == -1:
            return False

        # Try candidate sets that cover best_e
        for si in element_to_sets[best_e]:
            if si > last_index and (sets_masks[si] & used_mask) == 0:
                chosen_indices.append(si)
                if dfs(used_mask | sets_masks[si], si, chosen_indices):
                    return True
                chosen_indices.pop()
        return False

    if U == 0:
        print("SOLUTION")
        print()
        return

    dfs(0, -1, [])

    if best_solution is None:
        print("NO SOLUTION")
    else:
        print("SOLUTION")
        print(" ".join(str(i) for i in best_solution))

if __name__ == "__main__":
    main()
