; SPIR-V assembly for a simple triangle-shrinking geometry shader
; Purpose: shrink each input triangle toward its centroid by s = 0.7
; Produces 3 vertices per triangle, with per-vertex color passed through.

OpCapability Geometry
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0

; Constants
%uint_3 = OpConstant %uint 3
%float_0_7 = OpConstant %float 0.7
%float_inv3 = OpConstant %float 0.33333334
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%float_3 = OpConstant %float 3.0
%float_1 = OpConstant %float 1.0

; Array types for per-vertex inputs (triangles -> 3 vertices)
%arr_v3 = OpTypeArray %v3float %uint_3

; gl_PerVertex_in block (input positions from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%ptr_in_arr_pv = OpTypePointer Input %gl_PerVertex_in
%gl_in = OpVariable %ptr_in_arr_pv Input
; Access to gl_in[i].gl_Position: need a pointer to v4float
%ptr_gl_in_v4 = OpTypePointer Input %v4float

; Input arrays from vertex shader: worldPos[], normal[], color[]
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal   = OpVariable %ptr_in_arr_v3 Input
%color    = OpVariable %ptr_in_arr_v3 Input

; gl_PerVertex_out block (output gl_Position)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
; gl_Position is member 0 of gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

; Output color to fragment shader
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0

; Function type and entry point
%func_type_main = OpTypeFunction %void
%main = OpFunction %void None %func_type_main
%entry = OpLabel

; ------------------------------------------------------------
; Read input positions p0, p1, p2 from gl_in[0..2].gl_Position
; p_i are vec4
%p0_ptr = OpAccessChain %ptr_gl_in_v4 %gl_in %uint_0 %uint_0
%p0 = OpLoad %v4float %p0_ptr

%p1_ptr = OpAccessChain %ptr_gl_in_v4 %gl_in %uint_1 %uint_0
%p1 = OpLoad %v4float %p1_ptr

%p2_ptr = OpAccessChain %ptr_gl_in_v4 %gl_in %uint_2 %uint_0
%p2 = OpLoad %v4float %p2_ptr

; ------------------------------------------------------------
; Compute centroid C = (p0 + p1 + p2) / 3
%sum01 = OpFAdd %v4float %p0 %p1
%sum    = OpFAdd %v4float %sum01 %p2
%centroid = OpVectorTimesScalar %v4float %sum %float_inv3

; ------------------------------------------------------------
; Shrink factor s = 0.7
; new_p0 = centroid + s * (p0 - centroid)
%p0_minus_centroid = OpFSub %v4float %p0 %centroid
%scaled0 = OpVectorTimesScalar %v4float %p0_minus_centroid %float_0_7
%new_p0 = OpFAdd %v4float %centroid %scaled0

; new_p1 = centroid + s * (p1 - centroid)
%p1_minus_centroid = OpFSub %v4float %p1 %centroid
%scaled1 = OpVectorTimesScalar %v4float %p1_minus_centroid %float_0_7
%new_p1 = OpFAdd %v4float %centroid %scaled1

; new_p2 = centroid + s * (p2 - centroid)
%p2_minus_centroid = OpFSub %v4float %p2 %centroid
%scaled2 = OpVectorTimesScalar %v4float %p2_minus_centroid %float_0_7
%new_p2 = OpFAdd %v4float %centroid %scaled2

; ------------------------------------------------------------
; Read per-vertex colors: color[0], color[1], color[2]
%color0_ptr = OpAccessChain %ptr_in_arr_v3 %color %uint_0
%color0 = OpLoad %v3float %color0_ptr

%color1_ptr = OpAccessChain %ptr_in_arr_v3 %color %uint_1
%color1 = OpLoad %v3float %color1_ptr

%color2_ptr = OpAccessChain %ptr_in_arr_v3 %color %uint_2
%color2 = OpLoad %v3float %color2_ptr

; ------------------------------------------------------------
; Emit vertex 0
; gl_Position = new_p0
%gl_out_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_ptr0 %new_p0
; outColor = color0
OpStore %outColor %color0
OpEmitVertex

; Emit vertex 1
%gl_out_ptr1 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_ptr1 %new_p1
OpStore %outColor %color1
OpEmitVertex

; Emit vertex 2
%gl_out_ptr2 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_out_ptr2 %new_p2
OpStore %outColor %color2
OpEmitVertex

; End primitive
OpEndPrimitive

; ------------------------------------------------------------
; Finish
OpReturn
OpFunctionEnd

; Execution and interface declarations
OpEntryPoint Geometry %main "main" %worldPos %normal %color %outColor
OpExecutionMode %main Triangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1
