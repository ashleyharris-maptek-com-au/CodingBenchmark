#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <utility>
#include <cstdint>

using ll = long long;

// A single task definition: which machine and how long it takes
struct TaskDef {
    int machine;
    ll duration;
};

// Type alias for a schedule: for each job, a vector of (start_time, machine)
using Schedule = std::vector<std::vector<std::pair<ll,int>>>;

// Strategy A: Earliest-Start-Time based greedy scheduling (EST)
static Schedule schedule_EST(const std::vector<std::vector<TaskDef>>& jobs, ll& makespan_out) {
    int J = (int)jobs.size();
    // Determine how many machines exist (machines are 0-based)
    int maxMachine = -1;
    ll totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        totalTasks += (ll)jobs[j].size();
        for (const auto& t : jobs[j]) maxMachine = std::max(maxMachine, t.machine);
    }
    if (maxMachine < 0) maxMachine = 0;
    std::vector<ll> machine_available(maxMachine + 1, 0LL);

    // Result container
    Schedule result(J);
    for (int j = 0; j < J; ++j) {
        result[j].resize(jobs[j].size());
    }

    // Next task index per job
    std::vector<int> next_index(J, 0);

    // Ready-task node for EST
    struct ReadyNodeEST {
        ll est_start;
        int job;
        int task_idx;
        int machine;
        ll duration;
    };
    struct CmpEST {
        bool operator()(const ReadyNodeEST& a, const ReadyNodeEST& b) const {
            if (a.est_start != b.est_start) return a.est_start > b.est_start; // min-heap by est_start
            if (a.duration != b.duration) return a.duration > b.duration;       // tie-breaker: shorter duration first
            if (a.job != b.job) return a.job > b.job;
            return a.task_idx > b.task_idx;
        }
    };
    std::priority_queue<ReadyNodeEST, std::vector<ReadyNodeEST>, CmpEST> pq;

    // Initialize: push all first tasks
    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            const TaskDef& t = jobs[j][0];
            ll est = std::max(0LL, machine_available[t.machine]);
            pq.push({est, j, 0, t.machine, t.duration});
        }
    }

    makespan_out = 0;
    while (!pq.empty()) {
        ReadyNodeEST cur = pq.top(); pq.pop();
        // If this task is no longer the next for its job (could happen in more complex variants)
        if (cur.task_idx != next_index[cur.job]) continue;

        // Actual start is max of est and machine availability
        ll actual_start = std::max(cur.est_start, machine_available[cur.machine]);
        ll end_time     = actual_start + cur.duration;

        // Record
        result[cur.job][cur.task_idx] = {actual_start, cur.machine};

        // Update bookkeeping
        makespan_out = std::max(makespan_out, end_time);
        next_index[cur.job]++;

        // Push the next task for this job if any
        if (next_index[cur.job] < (int)jobs[cur.job].size()) {
            int next_k = next_index[cur.job];
            const TaskDef& nextTask = jobs[cur.job][next_k];
            ll next_est = std::max(end_time, machine_available[nextTask.machine]);
            pq.push({next_est, cur.job, next_k, nextTask.machine, nextTask.duration});
        }

        // Update machine availability
        if (cur.machine >= 0 && cur.machine <= maxMachine) {
            machine_available[cur.machine] = end_time;
        }
    }

    return result;
}

// Strategy B: Longest Remaining Time First (LRTF) heuristic
static Schedule schedule_LRTF(const std::vector<std::vector<TaskDef>>& jobs, ll& makespan_out) {
    int J = (int)jobs.size();
    int maxMachine = -1;
    ll totalTasks = 0;
    for (int j = 0; j < J; ++j) {
        totalTasks += (ll)jobs[j].size();
        for (const auto& t : jobs[j]) maxMachine = std::max(maxMachine, t.machine);
    }
    if (maxMachine < 0) maxMachine = 0;
    std::vector<ll> machine_available(maxMachine + 1, 0LL);

    Schedule result(J);
    for (int j = 0; j < J; ++j) {
        result[j].resize(jobs[j].size());
    }

    std::vector<int> next_index(J, 0);
    std::vector<ll> remaining_time(J, 0);
    for (int j = 0; j < J; ++j) {
        ll s = 0;
        for (const auto& t : jobs[j]) s += t.duration;
        remaining_time[j] = s;
    }

    // Ready task with priority: higher remaining_time -> higher priority
    struct ReadyNodeLR {
        ll est_start;
        int job;
        int task_idx;
        int machine;
        ll duration;
        ll remaining_time;
    };
    struct CmpLR {
        bool operator()(const ReadyNodeLR& a, const ReadyNodeLR& b) const {
            if (a.est_start != b.est_start) return a.est_start > b.est_start; // min est_start
            if (a.remaining_time != b.remaining_time) return a.remaining_time < b.remaining_time; // larger remaining_time first
            if (a.job != b.job) return a.job > b.job;
            if (a.task_idx != b.task_idx) return a.task_idx > b.task_idx;
            return false;
        }
    };
    std::priority_queue<ReadyNodeLR, std::vector<ReadyNodeLR>, CmpLR> pq;

    // Initialize: push all first tasks with their current remaining_time
    for (int j = 0; j < J; ++j) {
        if (!jobs[j].empty()) {
            const TaskDef& t0 = jobs[j][0];
            ll est = std::max(0LL, machine_available[t0.machine]);
            pq.push({est, j, 0, t0.machine, t0.duration, remaining_time[j]});
        }
    }

    makespan_out = 0;
    while (!pq.empty()) {
        ReadyNodeLR cur = pq.top(); pq.pop();
        if (cur.task_idx != next_index[cur.job]) continue;

        ll actual_start = std::max(cur.est_start, machine_available[cur.machine]);
        ll end_time     = actual_start + cur.duration;

        // Record
        result[cur.job][cur.task_idx] = {actual_start, cur.machine};

        // Update makespan
        makespan_out = std::max(makespan_out, end_time);

        // Update machine and job state
        machine_available[cur.machine] = end_time;
        next_index[cur.job]++;

        remaining_time[cur.job] -= cur.duration;

        // Push next task for this job if exists
        if (next_index[cur.job] < (int)jobs[cur.job].size()) {
            int next_k = next_index[cur.job];
            const TaskDef& nextTask = jobs[cur.job][next_k];
            ll next_est = std::max(end_time, machine_available[nextTask.machine]);
            pq.push({next_est, cur.job, next_k, nextTask.machine, nextTask.duration, remaining_time[cur.job]});
        }
    }

    return result;
}

int main() {
    // Fast IO (optional for MSVC)
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int J, M;
    if (!(std::cin >> J >> M)) {
        return 0;
    }

    std::vector<std::vector<TaskDef>> jobs(J);
    int maxMachineIndex = -1;
    ll totalTasks = 0;

    for (int j = 0; j < J; ++j) {
        int T;
        std::cin >> T;
        jobs[j].resize(T);
        totalTasks += (ll)T;
        for (int t = 0; t < T; ++t) {
            int m;
            ll dur;
            std::cin >> m >> dur;
            jobs[j][t] = {m, dur};
            if (m > maxMachineIndex) maxMachineIndex = m;
        }
    }

    // Adaptive strategy choice
    // Define a conservative threshold for trying multiple strategies.
    // For very small problems, compare EST and LRTF and pick the better (smaller makespan).
    // For larger problems, rely on EST (fast and scalable) but still consider EST as baseline.
    Schedule bestSchedule;
    ll makespan_EST = 0;
    ll makespan_LRTF = 0;

    bool try_both = false;
    // Simple heuristic: if total tasks are small and machines/jobs are few, try both
    if (totalTasks <= 60 && J <= 6 && M <= 6) {
        try_both = true;
    }

    if (try_both) {
        Schedule schedEST = schedule_EST(jobs, makespan_EST);
        Schedule schedLRTF = schedule_LRTF(jobs, makespan_LRTF);

        bestSchedule = schedEST;
        ll bestMakespan = makespan_EST;

        if (makespan_LRTF < bestMakespan) {
            bestSchedule = schedLRTF;
            bestMakespan = makespan_LRTF;
        }

        // If there is a tie, prefer EST (deterministic)
        bestSchedule = bestSchedule;
        // Output based on bestSchedule below
    } else {
        // Default to EST for large problems
        ll tmpMakespan = 0;
        bestSchedule = schedule_EST(jobs, tmpMakespan);
        makespan_EST = tmpMakespan;
    }

    // Print the schedule: for each job, print "start_time machine" pairs per task
    for (size_t j = 0; j < bestSchedule.size(); ++j) {
        const auto& line = bestSchedule[j];
        for (size_t k = 0; k < line.size(); ++k) {
            std::cout << line[k].first << " " << line[k].second;
            if (k + 1 < line.size()) std::cout << ' ';
        }
        if (j + 1 < bestSchedule.size()) std::cout << "\n";
    }

    return 0;
}
