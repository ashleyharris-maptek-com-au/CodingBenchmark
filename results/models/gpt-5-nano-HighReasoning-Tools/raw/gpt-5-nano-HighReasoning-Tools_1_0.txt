#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <limits>
#include <ctime>

// 2D point
struct Point {
    double x, y;
};

// Euclidean distance between two points
inline double dist(const Point& a, const Point& b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return std::sqrt(dx*dx + dy*dy);
}

// Solve TSP with adaptive strategy
// - N <= 20: exact Held-Karp DP
// - 21 <= N <= 3000: heuristic NN + 2-opt optimization
// - N > 3000: simple angular sort around centroid (scales to huge N)
std::vector<int> solveTSP(const std::vector<Point>& pts) {
    const int N = (int)pts.size();
    if (N <= 1) {
        return std::vector<int>(N, 0);
    }

    // For small N, use Held-Karp DP (exact)
    if (N <= 20) {
        // Precompute all-pairs distances
        std::vector<std::vector<double>> d(N, std::vector<double>(N, 0.0));
        for (int i = 0; i < N; ++i) {
            for (int j = 0; j < N; ++j) {
                d[i][j] = dist(pts[i], pts[j]);
            }
        }

        int m = N - 1; // we will consider nodes 1..N-1
        size_t full = 1u << m;
        const double INF = 1e100;

        // dp[mask][last] = minimal cost to start at 0, visit nodes in 'mask'
        // (subset of {1..N-1}), and end at node (last+1)
        auto idx = [m](size_t mask, int last) { return mask * m + last; };

        std::vector<double> dp(full * m, INF);
        std::vector<int> parent(full * m, -1);

        // base: going from 0 to each i+1
        for (int i = 0; i < m; ++i) {
            dp[idx(1u << i, i)] = d[0][i+1];
        }

        for (size_t mask = 1; mask < full; ++mask) {
            for (int last = 0; last < m; ++last) {
                if (!(mask & (1u << last))) continue;
                double cur = dp[idx(mask, last)];
                if (cur >= INF/2) continue;
                for (int nxt = 0; nxt < m; ++nxt) {
                    if (mask & (1u << nxt)) continue;
                    size_t newMask = mask | (1u << nxt);
                    double nv = cur + d[last+1][nxt+1];
                    double& ref = dp[idx(newMask, nxt)];
                    if (nv < ref) {
                        ref = nv;
                        parent[idx(newMask, nxt)] = last;
                    }
                }
            }
        }

        // close the tour back to 0
        double best = INF;
        int lastBest = -1;
        size_t fullMask = full - 1;
        for (int last = 0; last < m; ++last) {
            double val = dp[idx(fullMask, last)] + d[last+1][0];
            if (val < best) {
                best = val;
                lastBest = last;
            }
        }

        // reconstruct path
        std::vector<int> path;
        int cur = lastBest;
        size_t mask = fullMask;
        for (int i = 0; i < m; ++i) {
            path.push_back(cur + 1);
            int p = parent[idx(mask, cur)];
            mask ^= (1u << cur);
            cur = p;
        }
        std::reverse(path.begin(), path.end());

        std::vector<int> order;
        order.reserve(N);
        order.push_back(0);
        order.insert(order.end(), path.begin(), path.end());
        return order;
    }

    // Medium to large: Nearest Neighbor + 2-opt (reasonable quality with linear-ish memory)
    if (N <= 3000) {
        // Build a naive NN tour starting from 0
        std::vector<int> tour(N);
        std::iota(tour.begin(), tour.end(), 0);
        std::vector<char> used(N, 0);

        int cur = 0;
        tour[0] = 0;
        used[0] = 1;
        int idx = 1;

        while (idx < N) {
            int best = -1;
            double bestD = std::numeric_limits<double>::infinity();
            for (int j = 0; j < N; ++j) {
                if (used[j]) continue;
                double dcur = dist(pts[cur], pts[j]);
                if (dcur < bestD) {
                    bestD = dcur;
                    best = j;
                }
            }
            tour[idx++] = best;
            used[best] = 1;
            cur = best;
        }

        // 2-opt local improvement (bounded iterations for scalability)
        const int maxPasses = 7;
        for (int pass = 0; pass < maxPasses; ++pass) {
            bool improved = false;
            for (int i = 0; i < N - 1; ++i) {
                for (int k = i + 2; k < N; ++k) {
                    int a = tour[i];
                    int b = tour[(i + 1) % N];
                    int c = tour[k];
                    int d = tour[(k + 1) % N];
                    double delta = dist(pts[a], pts[c]) + dist(pts[b], pts[d])
                                 - dist(pts[a], pts[b]) - dist(pts[c], pts[d]);
                    if (delta < -1e-12) {
                        // reverse segment [i+1 .. k]
                        std::reverse(tour.begin() + i + 1, tour.begin() + k + 1);
                        improved = true;
                        break;
                    }
                }
                if (improved) break;
            }
            if (!improved) break;
        }

        return tour;
    }

    // Large to extreme: simple angular order around centroid (fast, scalable)
    // This yields a valid permutation starting with 0
    double cx = 0.0, cy = 0.0;
    for (const auto& p : pts) { cx += p.x; cy += p.y; }
    cx /= N; cy /= N;

    std::vector<std::pair<double,int>> angles(N);
    for (int i = 0; i < N; ++i) {
        double ang = std::atan2(pts[i].y - cy, pts[i].x - cx);
        angles[i] = {ang, i};
    }

    std::sort(angles.begin(), angles.end(),
              [](const std::pair<double,int>& a, const std::pair<double,int>& b){
                  if (a.first != b.first) return a.first < b.first;
                  return a.second < b.second;
              });

    std::vector<int> order(N);
    for (int i = 0; i < N; ++i) order[i] = angles[i].second;

    // rotate to start at city 0
    auto it = std::find(order.begin(), order.end(), 0);
    std::rotate(order.begin(), it, order.end());

    return order;
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N;
    if (!(std::cin >> N)) return 0;
    std::vector<Point> pts(N);
    for (int i = 0; i < N; ++i) {
        std::cin >> pts[i].x >> pts[i].y;
    }

    std::vector<int> order = solveTSP(pts);

    // Output: N integers, permutation starting with 0
    for (int i = 0; i < (int)order.size(); ++i) {
        if (i) std::cout << ' ';
        std::cout << order[i];
    }
    std::cout << '\n';
    return 0;
}
