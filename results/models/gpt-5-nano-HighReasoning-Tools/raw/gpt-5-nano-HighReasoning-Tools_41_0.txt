; SPIR-V
; Version: 1.0
; Generator: custom
OpCapability Shader
OpMemoryModel Logical GLSL450

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform Block (UBO)
%type_UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO_uniform = OpTypePointer Uniform %type_UBO
%ubo = OpVariable %ptr_UBO_uniform Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %type_UBO Block
OpMemberDecorate %type_UBO 0 Offset 0
OpMemberDecorate %type_UBO 1 Offset 64
OpMemberDecorate %type_UBO 2 Offset 128
OpMemberDecorate %type_UBO 3 Offset 192
OpMemberDecorate %type_UBO 4 Offset 208
OpMemberDecorate %type_UBO 5 Offset 224

; Inputs
%ptr_input_vec3 = OpTypePointer Input %vec3
%in_worldPos = OpVariable %ptr_input_vec3 Input
%in_normal = OpVariable %ptr_input_vec3 Input
%ptr_input_vec2 = OpTypePointer Input %vec2
%in_uv = OpVariable %ptr_input_vec2 Input
%in_tangent = OpVariable %ptr_input_vec3 Input
%in_color = OpVariable %ptr_input_vec3 Input
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

; Output
%ptr_output_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_output_vec4 Output
OpDecorate %outColor Location 0

; Global constants for solid red color
%const_1 = OpConstant %float 1.0
%const_0 = OpConstant %float 0.0
%vec4_red = OpConstantComposite %vec4 %const_1 %const_0 %const_0 %const_1

; Entry points / functions
%void = OpTypeVoid
%func_void = OpTypeFunction %void

%main = OpFunction %void None %func_void
%lab = OpLabel
OpStore %outColor %vec4_red
OpReturn
OpFunctionEnd

OpExecutionMode %main OriginUpperLeft
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
