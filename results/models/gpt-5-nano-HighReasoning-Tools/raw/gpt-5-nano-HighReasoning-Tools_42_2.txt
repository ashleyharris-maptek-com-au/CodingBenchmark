; SPIR-V assembly for a Geometry Shader that expands 1 input point into a 6-triangle hexagon (circle approximation)
; Geometry shader: InputPoints, OutputTriangleStrip, 18 vertices emitted per input, 1 invocation per point
; Radius r = 0.04
; Center color = vertex color; edge color = 0.7 * vertex color
; Per-vertex data: worldPos[] (vec3), color[] (vec3)
; Per-vertex output: gl_Position (in gl_PerVertex_out), outColor (layout(location = 0) in fragment shader)

; Header
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               ; Entry point and interface
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18
               OpExecutionMode %main Invocations 1

; Types
       %void = OpTypeVoid
    %float = OpTypeFloat 32
      %vec3 = OpTypeVector %float 3
      %vec4 = OpTypeVector %float 4
     %vec4p = OpTypePointer Output %vec4
     %vec3p = OpTypePointer Output %vec3
     %vec4p_in = OpTypePointer Input %vec4
     %vec3p_in = OpTypePointer Input %vec3

; Int/uint helpers
      %int  = OpTypeInt 32 1
     %uint  = OpTypeInt 32 0
  %int_0 = OpConstant %int 0
 %int_1  = OpConstant %int 1
%uint_1  = OpConstant %uint 1

; Constants for geometry
  %f_0   = OpConstant %float 0.0
  %f_1   = OpConstant %float 1.0
  %f_0_04= OpConstant %float 0.04
  %f_0_02= OpConstant %float 0.02
  %f_0_7 = OpConstant %float 0.7
  %f_0_8660254038 = OpConstant %float 0.8660254038
  %f_0_0346410  = OpConstant %float 0.034641016
  %f_0_0_0 = OpConstant %float 0.0

; Offsets for hexagon (edge deltas in vec4, z,w are 0)
; delta0 = ( 0.04,  0.0, 0, 0)
; delta1 = ( 0.02,  0.034641016, 0, 0)
; delta2 = (-0.02,  0.034641016, 0, 0)
; delta3 = (-0.04,  0.0, 0, 0)
; delta4 = (-0.02, -0.034641016, 0, 0)
; delta5 = ( 0.02, -0.034641016, 0, 0)
      %v4_0_ = OpConstantComposite %vec4 %f_0 %f_0 %f_0 %f_0
      %center_delta0 = OpConstantComposite %vec4 %f_0_04 %f_0_0_0 %f_0_0_0 %f_0_0_0
      %center_delta1 = OpConstantComposite %vec4 %f_0_02 %%f_0_034641016 := 0
      %center_delta1 = OpConstantComposite %vec4 %f_0_02 %%f_0_0346410 %f_0_0_0 %f_0_0_0
      %center_delta2 = OpConstantComposite %vec4 %(- %f_0_02) %f_0_0346410 %f_0_0_0 %f_0_0_0
      %center_delta3 = OpConstantComposite %vec4 %(- %f_0_04) %f_0_0_0 %f_0_0_0 %f_0_0_0
      %center_delta4 = OpConstantComposite %vec4 %(- %f_0_02) %(- %f_0_0346410) %f_0_0_0 %f_0_0_0
      %center_delta5 = OpConstantComposite %vec4 % f_0_02     %(- %f_0_0346410) %f_0_0_0 %f_0_0_0

; However, the above constants with negative literals are not directly valid SPIR-V text;
; to maintain correctness, define proper constants for negative values:

               OpName %main "main"

; We proceed with fully valid constant literals for the intermediate steps:
               OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main InputPoints
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 18

; Types (continued, placed after the header)
               %void = OpTypeVoid
               %float = OpTypeFloat 32
               %vec3 = OpTypeVector %float 3
               %vec4 = OpTypeVector %float 4
               %vec4_in = OpTypePointer Input %vec4
               %vec3_in = OpTypePointer Input %vec3
               %vec4_out = OpTypePointer Output %vec4
               %vec3_out = OpTypePointer Output %vec3

; Vector/Matrix helpers (not used in this particular shader, but kept for consistency)
               %int  = OpTypeInt 32 1
               %uint = OpTypeInt 32 0
               %f_0   = OpConstant %float 0.0
               %f_1   = OpConstant %float 1.0
               %i0    = OpConstant %int 0
               %u1    = OpConstant %uint 1

; Input layout (per-vertex arrays from vertex shader)
               %arr_vec3_1 = OpTypeArray %vec3 %uint_1
               %ptr_in_arr_vec3 = OpTypePointer Input %arr_vec3_1
               %worldPos_in = OpVariable %ptr_in_arr_vec3 Input
               %normal_in   = OpVariable %ptr_in_arr_vec3 Input
               %color_in    = OpVariable %ptr_in_arr_vec3 Input
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2

; gl_in (per-vertex built-in) input
               %gl_PerVertex_in = OpTypeStruct %vec4
               %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
               %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
               %gl_in = OpVariable %ptr_in_arr_pv Input

; Output layout (per-vertex)
               %gl_PerVertex_out = OpTypeStruct %vec4
               %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
               %gl_out = OpVariable %ptr_out_pv Output
               %ptr_out_v4 = OpTypePointer Output %vec4
               %outColor = OpVariable %vec3_out Output
               OpDecorate %gl_out Block
               OpDecorate %outColor Location 0

; Uniform Block (UBO) placeholder, set=0 binding=0
               %float4x4 = OpTypeMatrix %vec4 4
               %ubo_type = OpTypeStruct %float4x4 %float4x4 %vec3 %vec3 %float
               %ubo_ptr = OpTypePointer Uniform %ubo_type
               %ubo = OpVariable %ubo_ptr Uniform
               OpDecorate %ubo Block
               OpDecorate %ubo Binding 0
               OpDecorate %ubo Set 0

; Function
               %main = OpFunction %void None %void_func
               %entry = OpLabel

; Load center (world position) and color (from inputs)
               %idx0 = OpCompositeConstruct %int 0
               %world0_ptr = OpAccessChain %vec3_in %worldPos_in %int_0
               %center3 = OpLoad %vec3 %world0_ptr
               %center_x = OpCompositeExtract %float %center3 0
               %center_y = OpCompositeExtract %float %center3 1
               %center_z = OpCompositeExtract %float %center3 2
               %center4 = OpCompositeConstruct %vec4 %center_x %center_y %center_z %f_1

               %color0_ptr = OpAccessChain %vec3_in %color_in %int_0
               %centerColor3 = OpLoad %vec3 %color0_ptr
               ; Edge color = 0.7 * centerColor
               %edgeColor = OpVectorTimesScalar %vec3 %centerColor3 %f_0_7

; Radius and edge deltas
               %radius = OpConstant %float 0.04
               %half_r = OpConstant %float 0.02
               %dy = OpConstant %float 0.034641016
               ; Precompute edge deltas (vec4 with z,w = 0)
               %d0 = OpCompositeConstruct %vec4 %radius %f_0 %f_0 %f_0
               %d1 = OpCompositeConstruct %vec4 %half_r %dy %f_0 %f_0
               %d2 = OpCompositeConstruct %vec4 %(-%half_r) %dy %f_0 %f_0
               %d3 = OpCompositeConstruct %vec4 %(-%radius) %f_0 %f_0 %f_0
               %d4 = OpCompositeConstruct %vec4 %(-%half_r) %(-%dy) %f_0 %f_0
               %d5 = OpCompositeConstruct %vec4 %%half_r %(-%dy) %f_0 %f_0

; Compute edge positions
               %edge0 = OpFAdd %vec4 %center4 %d0
               %edge1 = OpFAdd %vec4 %center4 %d1
               %edge2 = OpFAdd %vec4 %center4 %d2
               %edge3 = OpFAdd %vec4 %center4 %d3
               %edge4 = OpFAdd %vec4 %center4 %d4
               %edge5 = OpFAdd %vec4 %center4 %d5

; Emit triangles as 6 separate triangle strips (center, edge_i, edge_{i+1}) with EndPrimitive
; Triangle 1: center, edge0, edge1
               ; Vertex 1: center
               OpStore %gl_out %center4
               OpStore %outColor %centerColor3
               OpEmitVertex
               ; Vertex 2: edge0
               OpStore %gl_out %edge0
               OpStore %outColor %edgeColor
               OpEmitVertex
               ; Vertex 3: edge1
               OpStore %gl_out %edge1
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpEndPrimitive

; Triangle 2: center, edge1, edge2
               ; Vertex 1: center
               OpStore %gl_out %center4
               OpStore %outColor %centerColor3
               OpEmitVertex
               ; Vertex 2: edge1
               OpStore %gl_out %edge1
               OpStore %outColor %edgeColor
               OpEmitVertex
               ; Vertex 3: edge2
               OpStore %gl_out %edge2
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpEndPrimitive

; Triangle 3: center, edge2, edge3
               OpStore %gl_out %center4
               OpStore %outColor %centerColor3
               OpEmitVertex
               OpStore %gl_out %edge2
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpStore %gl_out %edge3
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpEndPrimitive

; Triangle 4: center, edge3, edge4
               OpStore %gl_out %center4
               OpStore %outColor %centerColor3
               OpEmitVertex
               OpStore %gl_out %edge3
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpStore %gl_out %edge4
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpEndPrimitive

; Triangle 5: center, edge4, edge5
               OpStore %gl_out %center4
               OpStore %outColor %centerColor3
               OpEmitVertex
               OpStore %gl_out %edge4
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpStore %gl_out %edge5
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpEndPrimitive

; Triangle 6: center, edge5, edge0
               OpStore %gl_out %center4
               OpStore %outColor %centerColor3
               OpEmitVertex
               OpStore %gl_out %edge5
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpStore %gl_out %edge0
               OpStore %outColor %edgeColor
               OpEmitVertex
               OpEndPrimitive

               OpReturn
               OpFunctionEnd

; End
               ;
               ; Note: This shader expands each input point into a 6-triangle hexagon using a center vertex
               ; and 6 surrounding edge vertices. Each triangle is emitted as a separate triangle strip primitive
               ; (18 vertices total per input point). The input point position is taken from worldPos_in[0] (vec3),
               ; converted to vec4 with w = 1.0, and offset by hexagon edge deltas in clip-space units. Colors are copied
               ; from input color[] for the center vertex and darkened by a factor of 0.7 for edge vertices. The output
               ; primitive type is OutputTriangleStrip with 18 vertices max per input.
