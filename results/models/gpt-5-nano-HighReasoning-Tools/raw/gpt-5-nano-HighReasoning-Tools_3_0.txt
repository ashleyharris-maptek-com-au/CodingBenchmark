use std::collections::HashSet;
use std::f64::consts::PI;
use std::io::{self, Read};

// A simple 2D vector
#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

fn clamp(v: f64, lo: f64, hi: f64) -> f64 {
    if v < lo {
        lo
    } else if v > hi {
        hi
    } else {
        v
    }
}

// Initial positions: evenly place nodes on a circle inside the given width/height box
fn initial_positions(n: usize, width: f64, height: f64) -> Vec<Vec2> {
    let cx = width * 0.5;
    let cy = height * 0.5;
    let radius = (width.min(height)) * 0.4;
    let mut pos = Vec::with_capacity(n);
    for i in 0..n {
        let ang = 2.0 * PI * (i as f64) / (n.max(1) as f64);
        pos.push(Vec2 {
            x: cx + radius * ang.cos(),
            y: cy + radius * ang.sin(),
        });
    }
    pos
}

// Exact Fruchterman–Reingold layout (O(n^2) repulsion) for small to moderate graphs
fn layout_fr_exact(n: usize, edges: &[(usize, usize)], iterations: usize, width: f64, height: f64) -> Vec<Vec2> {
    let mut pos = initial_positions(n, width, height);
    let mut disp = vec![Vec2 { x: 0.0, y: 0.0 }; n];
    let area = width * height;
    let k = (area / (n as f64)).sqrt();

    for iter in 0..iterations {
        // reset displacements
        for d in &mut disp {
            d.x = 0.0;
            d.y = 0.0;
        }

        // Repulsive forces (all pairs)
        if n > 1 {
            for i in 0..n {
                for j in (i + 1)..n {
                    let dx = pos[i].x - pos[j].x;
                    let dy = pos[i].y - pos[j].y;
                    let mut dist = (dx * dx + dy * dy).sqrt();
                    if dist < 1e-9 {
                        dist = 1e-9;
                    }
                    let force = (k * k) / dist;
                    let ux = dx / dist;
                    let uy = dy / dist;
                    disp[i].x += ux * force;
                    disp[i].y += uy * force;
                    disp[j].x -= ux * force;
                    disp[j].y -= uy * force;
                }
            }
        }

        // Attractive forces along edges
        for &(u, v) in edges {
            let dx = pos[u].x - pos[v].x;
            let dy = pos[u].y - pos[v].y;
            let mut dist = (dx * dx + dy * dy).sqrt();
            if dist < 1e-9 {
                dist = 1e-9;
            }
            let force = (dist * dist) / k;
            let ux = dx / dist;
            let uy = dy / dist;
            disp[u].x -= ux * force;
            disp[u].y -= uy * force;
            disp[v].x += ux * force;
            disp[v].y += uy * force;
        }

        // Move nodes by a damped amount (temperature decreases over time)
        let t = (width.min(height)) * 0.5 * (1.0 - iter as f64 / iterations as f64);
        for i in 0..n {
            let dx = disp[i].x;
            let dy = disp[i].y;
            let dist = (dx * dx + dy * dy).sqrt();
            if dist > 0.0 {
                let factor = t / dist;
                pos[i].x += dx * factor;
                pos[i].y += dy * factor;
            } else {
                // tiny deterministic jitter to avoid stagnation
                let jitter = (iter as f64) * 0.01 + (i as f64) * 0.01;
                pos[i].x += jitter.sin() * 0.001;
                pos[i].y += jitter.cos() * 0.001;
            }

            // Keep inside the box to avoid runaway layouts
            pos[i].x = clamp(pos[i].x, 0.0, width);
            pos[i].y = clamp(pos[i].y, 0.0, height);
        }
    }

    pos
}

// Approximate layout for very large graphs:
// - uses grid-based (cell) bucketing for local repulsion (O(n) per iter)
// - edge attractions as in exact FR
fn layout_fr_approx(n: usize, edges: &[(usize, usize)], iterations: usize, width: f64, height: f64) -> Vec<Vec2> {
    let mut pos = initial_positions(n, width, height);
    let mut disp = vec![Vec2 { x: 0.0, y: 0.0 }; n];
    let area = width * height;
    let k = (area / (n as f64)).sqrt();

    // Grid parameters (adapt to graph size)
    // cell_size chosen to balance cell count and neighbor checks
    let cell_size = ((area / (n as f64)).sqrt()) * 0.75;
    let cols = ((width / cell_size).ceil() as usize).max(1);
    let rows = ((height / cell_size).ceil() as usize).max(1);
    let grid_len = cols * rows;

    for iter in 0..iterations {
        // reset displacements
        for d in &mut disp {
            d.x = 0.0;
            d.y = 0.0;
        }

        // Build grid cells with current positions
        let mut grid: Vec<Vec<usize>> = vec![Vec::new(); grid_len];
        for i in 0..n {
            let mut cx = (pos[i].x / cell_size).floor() as isize;
            let mut cy = (pos[i].y / cell_size).floor() as isize;
            if cx < 0 { cx = 0; }
            if cy < 0 { cy = 0; }
            if (cx as usize) >= cols { cx = (cols - 1) as isize; }
            if (cy as usize) >= rows { cy = (rows - 1) as isize; }
            let idx = (cy as usize) * cols + (cx as usize);
            grid[idx].push(i);
        }

        // Repulsive forces from nearby grid cells (process i<j only once)
        if n > 1 {
            for i in 0..n {
                let mut cx = (pos[i].x / cell_size).floor() as isize;
                let mut cy = (pos[i].y / cell_size).floor() as isize;
                if cx < 0 { cx = 0; }
                if cy < 0 { cy = 0; }
                if (cx as usize) >= cols { cx = (cols - 1) as isize; }
                if (cy as usize) >= rows { cy = (rows - 1) as isize; }

                for dxcell in -1..=1 {
                    for dycell in -1..=1 {
                        let nx = cx + dxcell;
                        let ny = cy + dycell;
                        if nx < 0 || ny < 0 { continue; }
                        if (nx as usize) >= cols || (ny as usize) >= rows { continue; }
                        let nidx = (ny as usize) * cols + (nx as usize);
                        for &j in &grid[nidx] {
                            if j <= i { continue; }
                            let dx = pos[i].x - pos[j].x;
                            let dy = pos[i].y - pos[j].y;
                            let mut dist = (dx * dx + dy * dy).sqrt();
                            if dist < 1e-9 { dist = 1e-9; }
                            let force = (k * k) / dist;
                            let ux = dx / dist;
                            let uy = dy / dist;
                            disp[i].x += ux * force;
                            disp[i].y += uy * force;
                            disp[j].x -= ux * force;
                            disp[j].y -= uy * force;
                        }
                    }
                }
            }
        }

        // Attractive forces along edges
        for &(u, v) in edges {
            let dx = pos[u].x - pos[v].x;
            let dy = pos[u].y - pos[v].y;
            let mut dist = (dx * dx + dy * dy).sqrt();
            if dist < 1e-9 { dist = 1e-9; }
            let force = (dist * dist) / k;
            let ux = dx / dist;
            let uy = dy / dist;
            disp[u].x -= ux * force;
            disp[u].y -= uy * force;
            disp[v].x += ux * force;
            disp[v].y += uy * force;
        }

        // Move nodes with damping
        let t = (width.min(height)) * 0.5 * (1.0 - iter as f64 / iterations as f64);
        for i in 0..n {
            let dx = disp[i].x;
            let dy = disp[i].y;
            let dist = (dx * dx + dy * dy).sqrt();
            if dist > 0.0 {
                let factor = t / dist;
                pos[i].x += dx * factor;
                pos[i].y += dy * factor;
            } else {
                // tiny deterministic jitter
                let jitter = (iter as f64) * 0.01 + (i as f64) * 0.01;
                pos[i].x += jitter.sin() * 0.001;
                pos[i].y += jitter.cos() * 0.001;
            }

            // Keep inside box
            pos[i].x = clamp(pos[i].x, 0.0, width);
            pos[i].y = clamp(pos[i].y, 0.0, height);
        }
    }

    pos
}

fn main() {
    // Read entire stdin
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Read N and M
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    // Read edges, deduplicate, ignore self-loops
    let mut seen = HashSet<(usize, usize)>();
    for _ in 0..m {
        let a: usize = match it.next() { Some(v) => v.parse().unwrap(), None => break };
        let b: usize = match it.next() { Some(v) => v.parse().unwrap(), None => break };
        if a == b { continue; }
        let (u, v) = if a < b { (a, b) } else { (b, a) };
        seen.insert((u, v));
    }

    let edges: Vec<(usize, usize)> = seen.into_iter().collect();

    // Layout area
    let width = 1000.0;
    let height = 1000.0;

    // Choose a strategy based on graph size
    // Small/moderate graphs: exact repulsion
    // Large graphs: approximate, grid-based repulsion
    let positions = if n <= 1500 {
        layout_fr_exact(n, &edges, 200, width, height)
    } else {
        layout_fr_approx(n, &edges, 200, width, height)
    };

    // Output: N lines, one per node: x y
    for i in 0..n {
        println!("{:.6} {:.6}", positions[i].x, positions[i].y);
    }
}
