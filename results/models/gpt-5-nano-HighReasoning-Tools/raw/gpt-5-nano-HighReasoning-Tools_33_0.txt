Here's a self-contained C# program that reads N, the flow and distance matrices, builds an initial permutation by a simple centrality heuristic, and then improves it with a lightweight randomized local search. It uses only the standard library and is designed to handle up to about N = 2000 within reasonable time.

- Heuristic: assign facilities with larger total flow to locations that are more central (smaller sum of distances).
- Improvement: a fast, stochastic pairwise-swap search. Each swap delta is computed in O(N) time, allowing many improvements within a short time budget.

Compile with: csc Program.cs (or your preferred .NET build method)

```csharp
using System;
using System.IO;
using System.Text;
using System.Diagnostics;

class FastScanner
{
    private readonly Stream _in;
    private readonly byte[] _buffer = new byte[1 << 16];
    private int _ptr = 0, _len = 0;

    public FastScanner(Stream inStream)
    {
        _in = inStream;
    }

    private int Read()
    {
        if (_ptr >= _len)
        {
            _len = _in.Read(_buffer, 0, _buffer.Length);
            _ptr = 0;
            if (_len == 0) return -1;
        }
        return _buffer[_ptr++];
    }

    public int NextInt()
    {
        int c;
        do
        {
            c = Read();
        } while (c <= ' ' && c != -1);

        int sign = 1;
        if (c == '-')
        {
            sign = -1;
            c = Read();
        }

        int val = 0;
        while (c > ' ')
        {
            val = val * 10 + (c - '0');
            c = Read();
        }
        return val * sign;
    }
}

class Program
{
    static void Main(string[] args)
    {
        var scanner = new FastScanner(Console.OpenStandardInput());

        int N = scanner.NextInt();
        if (N <= 0)
        {
            Console.WriteLine("0");
            Console.WriteLine();
            return;
        }

        // Read flow matrix
        int[][] flow = new int[N][];
        for (int i = 0; i < N; i++)
        {
            flow[i] = new int[N];
            for (int j = 0; j < N; j++) flow[i][j] = scanner.NextInt();
        }

        // Read dist matrix
        int[][] dist = new int[N][];
        for (int i = 0; i < N; i++)
        {
            dist[i] = new int[N];
            for (int j = 0; j < N; j++) dist[i][j] = scanner.NextInt();
        }

        // Precompute facility centralities: F[i] = sum_j flow[i][j] + sum_j flow[j][i]
        long[] outSum = new long[N];
        long[] inSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            long s = 0;
            for (int j = 0; j < N; j++) s += flow[i][j];
            outSum[i] = s;
        }
        for (int j = 0; j < N; j++)
        {
            long s = 0;
            for (int i = 0; i < N; i++) s += flow[i][j];
            inSum[j] = s;
        }
        long[] F = new long[N];
        for (int i = 0; i < N; i++) F[i] = outSum[i] + inSum[i];

        // Location centralities: Dloc[l] = sum_m dist[l][m]
        long[] Dloc = new long[N];
        for (int l = 0; l < N; l++)
        {
            long s = 0;
            for (int m = 0; m < N; m++) s += dist[l][m];
            Dloc[l] = s;
        }

        // Initial permutation p[i] = location assigned to facility i
        int[] idxFlow = new int[N];
        for (int i = 0; i < N; i++) idxFlow[i] = i;
        int[] idxLoc = new int[N];
        for (int i = 0; i < N; i++) idxLoc[i] = i;

        // Sort facilities by decreasing F and locations by increasing Dloc
        Array.Sort(idxFlow, (a, b) => F[b].CompareTo(F[a])); // desc
        Array.Sort(idxLoc, (a, b) => Dloc[a].CompareTo(Dloc[b])); // asc

        int[] p = new int[N];
        for (int k = 0; k < N; k++)
        {
            p[idxFlow[k]] = idxLoc[k];
        }

        long cost = ComputeCost(flow, dist, p);

        // Lightweight randomized local search
        Random rng = new Random(123456);
        Stopwatch sw = Stopwatch.StartNew();

        // Time budget in seconds. Adjust as needed.
        double timeBudget = 3.0; // seconds
        const int maxIters = 100000; // fallback cap

        int iter = 0;
        while (sw.Elapsed.TotalSeconds < timeBudget && iter < maxIters)
        {
            iter++;

            int a = rng.Next(N);
            int b = rng.Next(N);
            if (a == b) continue;

            long delta = DeltaSwap(flow, dist, p, a, b);
            if (delta < 0)
            {
                // apply swap
                int tmp = p[a];
                p[a] = p[b];
                p[b] = tmp;

                cost += delta;
            }
        }

        // Output
        Console.WriteLine(cost);
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < N; i++)
        {
            if (i > 0) sb.Append(' ');
            sb.Append(p[i]);
        }
        Console.WriteLine(sb.ToString());
    }

    // Compute total cost of a permutation p
    static long ComputeCost(int[][] flow, int[][] dist, int[] p)
    {
        int N = p.Length;
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int di = dist[p[i]][0]; // just to avoid unused warning (not used, but keeps compiler happy)
            for (int j = 0; j < N; j++)
            {
                cost += (long)flow[i][j] * dist[p[i]][p[j]];
            }
        }
        return cost;
    }

    // Delta of swapping facilities a and b in permutation p
    // Delta = cost(new) - cost(old)
    static long DeltaSwap(int[][] flow, int[][] dist, int[] p, int a, int b)
    {
        int N = p.Length;
        int x = p[a];
        int y = p[b];

        long term1 = 0, term2 = 0, term3 = 0, term4 = 0;

        // i dimension: contributions where i is any facility
        for (int i = 0; i < N; i++)
        {
            term1 += (long)flow[i][a] * (dist[p[i]][y] - dist[p[i]][x]);
            term2 += (long)flow[i][b] * (dist[p[i]][x] - dist[p[i]][y]);
        }

        // j dimension: contributions where j is any facility
        for (int j = 0; j < N; j++)
        {
            term3 += (long)flow[a][j] * (dist[y][p[j]] - dist[x][p[j]]);
            term4 += (long)flow[b][j] * (dist[x][p[j]] - dist[y][p[j]]);
        }

        return term1 + term2 + term3 + term4;
    }
}
```
