Here's a self-contained C++17 solution that solves the 0/1 ILP by reducing it to a maximum-weight independent set (MWIS) on a graph and solving exactly via branch-and-bound with pruning.

- We keep only vertices with positive weight (x_i = 1 can only help if c_i > 0).
- Edges connect pairs (u, v) that must not be both chosen.
- The solver uses a 64-bit bitmask (works for up to 64 active vertices). It performs a DFS with pruning and includes isolated vertices greedily to speed up search.
- The final output prints the objective value and the binary assignment for all original variables (non-active ones are 0).

Code (C++17):

```cpp
#include <bits/stdc++.h>
using namespace std;

// Global state for the MWIS solver (fits up to 64 vertices after filtering)
static int K; // number of active vertices (K <= 64)
static vector<long long> w;              // weights of active vertices
static vector<unsigned long long> adj;     // adjacency masks for active vertices
static long long best;                    // best objective found
static unsigned long long bestMask;         // mask of vertices chosen in best solution

// Recursive branch-and-bound solver
void dfs(unsigned long long P, long long curW, unsigned long long chosen) {
    // Greedily include all vertices in P that are isolated within P
    unsigned long long Pcopy = P;
    long long cw = curW;
    unsigned long long ch = chosen;
    while (Pcopy) {
        int v = __builtin_ctzll(Pcopy);
        // If v has no neighbors in Pcopy, it can be safely included
        if ((adj[v] & Pcopy) == 0ULL) {
            Pcopy &= ~(1ULL << v);
            cw += w[v];
            ch |= (1ULL << v);
        } else {
            break;
        }
    }

    if (Pcopy == 0ULL) {
        if (cw > best) {
            best = cw;
            bestMask = ch;
        }
        return;
    }

    // Simple upper bound: current weight + sum of weights in Pcopy
    long long sumP = 0;
    unsigned long long tmp = Pcopy;
    while (tmp) {
        int v = __builtin_ctzll(tmp);
        sumP += w[v];
        tmp &= tmp - 1;
    }
    if (cw + sumP <= best) return; // prune

    // Choose a vertex to branch on (LSB strategy)
    int v = __builtin_ctzll(Pcopy);

    // Include v
    unsigned long long P_incl = Pcopy & ~((1ULL << v) | adj[v]);
    dfs(P_incl, cw + w[v], ch | (1ULL << v));

    // Exclude v
    unsigned long long P_excl = Pcopy & ~(1ULL << v);
    dfs(P_excl, cw, ch);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n, m;
    if (!(cin >> n >> m)) return 0;

    vector<long long> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    vector<pair<int,int>> edges;
    edges.reserve(m);
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        edges.emplace_back(u, v);
    }

    // Keep only vertices with positive weight
    vector<int> idxInActive(n, -1);
    vector<int> act; // mapping from active index -> original index
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            idxInActive[i] = (int)act.size();
            act.push_back(i);
        }
    }

    int k = (int)act.size(); // number of active vertices
    vector<long long> w_local(k, 0);
    vector<unsigned long long> adj_local;
    if (k > 0) adj_local.assign(k, 0ULL);
    for (int i = 0; i < k; ++i) w_local[i] = c[act[i]];

    // Build adjacency among active vertices
    for (auto &e : edges) {
        int u = e.first, v = e.second;
        if (u < 0 || v < 0 || u >= n || v >= n) continue;
        int iu = idxInActive[u];
        int iv = idxInActive[v];
        if (iu != -1 && iv != -1) {
            adj_local[iu] |= (1ULL << iv);
            adj_local[iv] |= (1ULL << iu);
        }
    }

    // Prepare global state
    K = k;
    w = w_local;
    adj = adj_local;
    best = 0;
    bestMask = 0;

    unsigned long long P0 = 0;
    if (k > 0) {
        P0 = (k == 64) ? ~0ULL : ((1ULL << k) - 1);
        dfs(P0, 0, 0);
    }

    // Build final answer for all n variables
    vector<int> ans(n, 0);
    for (int i = 0; i < k; ++i) {
        if ((bestMask >> i) & 1ULL) {
            ans[act[i]] = 1;
        }
    }

    cout << best << "\n";
    for (int i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << ans[i];
    }
    cout << "\n";

    return 0;
}
```
