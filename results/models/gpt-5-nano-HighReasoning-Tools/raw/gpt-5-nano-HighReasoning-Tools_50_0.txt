Here is a complete GLSL 450 compute shader that performs one timestep of Reynolds-style 3D boid flocking. It reads from the input buffer, computes separation, alignment, and cohesion (averaged over neighbors within the provided radii), applies a boundary push toward the center when positions exceed a 0.4–0.5 boundSize window, and writes updated position/velocity to the output buffer. It guards against out-of-bounds work-items, ignores self and zero-distance neighbors, and clamps speed to maxSpeed.

```glsl
#version 450

layout(local_size_x = 256) in;

// Input boids: each boid is two vec4s: [pos.xyz, 0], [vel.xyz, 0]
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

// Output boids (ping-pong): same layout as input
layout(set = 0, binding = 1) writeonly buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

// Parameters (one uniform block for all subpasses)
layout(set = 0, binding = 2) uniform ParamsBlock {
    // counts.x = N
    uvec4 counts;

    // radii.x = dt, radii.y = sepRadius, radii.z = alignRadius, radii.w = cohRadius
    vec4 radii;

    // weights.x = sepWeight, weights.y = alignWeight, weights.z = cohWeight, weights.w = maxSpeed
    vec4 weights;

    // bounds.x = boundSize, bounds.y = boundForce
    vec4 bounds;
} params;

void main() {
    uint N = params.counts.x;
    uint gid = gl_GlobalInvocationID.x;

    // Guard: do not process out-of-bounds work items
    if (gid >= N) {
        return;
    }

    int i = int(gid);
    int base_i = i * 2;

    // Read current boid state
    vec3 pos = inData[base_i].xyz;
    vec3 vel = inData[base_i + 1].xyz;

    // Timestepping and neighborhood radii
    float dt       = params.radii.x;
    float sepR     = params.radii.y;
    float alignR   = params.radii.z;
    float cohR     = params.radii.w;

    // Weights
    float sepW = params.weights.x;
    float alignW = params.weights.y;
    float cohW = params.weights.z;
    float maxSpeed = params.weights.w;

    // Boundary
    float boundSize = params.bounds.x;
    float boundForce = params.bounds.y;

    // Accumulators for flocking forces
    vec3 sepAcc = vec3(0.0);
    vec3 alignAcc = vec3(0.0);
    vec3 cohAcc = vec3(0.0);
    int neighborCount = 0;

    // Precompute max radius for a quick skip
    float maxR = max(sepR, max(alignR, cohR));

    // Iterate over all neighbors
    for (int j = 0; j < int(N); ++j) {
        if (j == i) continue; // skip self

        int base_j = j * 2;
        vec3 otherPos = inData[base_j].xyz;
        vec3 toOther = otherPos - pos;
        float dist = length(toOther);

        // Skip zero-distance neighbors
        if (dist <= 0.0) continue;

        // Consider only neighbors within the largest neighborhood radius
        if (dist <= maxR) {
            // Separation: push away from very close neighbors
            if (dist < sepR) {
                vec3 away = -normalize(toOther) * (sepR - dist);
                sepAcc += away;
            }

            // Alignment: steer toward neighbor's velocity
            if (dist < alignR) {
                vec3 otherVel = inData[base_j + 1].xyz;
                alignAcc += otherVel;
            }

            // Cohesion: steer toward the neighbor's position (center of mass)
            if (dist < cohR) {
                cohAcc += otherPos;
            }

            neighborCount += 1;
        }
    }

    // Compute acceleration from neighbors (average by neighbor count)
    vec3 accel = vec3(0.0);
    if (neighborCount > 0) {
        // Cohesion direction toward the average neighbor position
        vec3 cohesionDir = (cohAcc / float(neighborCount)) - pos;
        accel = (sepAcc * sepW) + (alignAcc * alignW) + (cohesionDir * cohW);
    }

    // Boundary handling: push toward center when any component exceeds 0.4*boundSize
    // Scaling is based on (abs(pos) - 0.4*boundSize) / (0.5*boundSize - 0.4*boundSize)
    // i.e., 0 at 0.4*boundSize and 1 at 0.5*boundSize (clamped)
    float T = boundSize * 0.4;
    float L = boundSize * 0.5;
    vec3 boundaryPush = vec3(0.0);

    // X component
    float ax = abs(pos.x);
    if (ax > T) {
        float scale = (ax - T) / (L - T);
        scale = clamp(scale, 0.0, 1.0);
        boundaryPush.x += -sign(pos.x) * boundForce * scale;
    }
    // Y component
    float ay = abs(pos.y);
    if (ay > T) {
        float scale = (ay - T) / (L - T);
        scale = clamp(scale, 0.0, 1.0);
        boundaryPush.y += -sign(pos.y) * boundForce * scale;
    }
    // Z component
    float az = abs(pos.z);
    if (az > T) {
        float scale = (az - T) / (L - T);
        scale = clamp(scale, 0.0, 1.0);
        boundaryPush.z += -sign(pos.z) * boundForce * scale;
    }

    accel += boundaryPush;

    // Integrate velocity and position
    vec3 newVel = vel + accel * dt;

    // Clamp speed
    float speed = length(newVel);
    if (speed > maxSpeed && speed > 0.0) {
        newVel *= (maxSpeed / speed);
    }

    vec3 newPos = pos + (newVel * dt);

    // Write results back to output buffers
    outData[base_i]     = vec4(newPos, 0.0);
    outData[base_i + 1] = vec4(newVel, 0.0);
}
```
