struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    // Procedural brick pattern with 8x16 UV tiling and staggered rows
    float uScaled = input.uv.x * 8.0;
    float vScaled = input.uv.y * 16.0;

    int row = (int)floor(vScaled);
    bool oddRow = (row % 2) != 0;
    float uInBrick = frac(uScaled + (oddRow ? 0.5 : 0.0));
    float vInBrick = frac(vScaled);

    const float mortarX = 0.05;
    const float mortarY = 0.10;

    bool isMortar = (uInBrick < mortarX) || (uInBrick > 1.0 - mortarX) || (vInBrick < mortarY) || (vInBrick > 1.0 - mortarY);

    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 color = isMortar ? mortarColor : brickColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diff = saturate(dot(N, L));
    float lambert = 0.4 + 0.6 * diff;

    float3 finalColor = color * lambert;

    return float4(finalColor, 1.0);
}
