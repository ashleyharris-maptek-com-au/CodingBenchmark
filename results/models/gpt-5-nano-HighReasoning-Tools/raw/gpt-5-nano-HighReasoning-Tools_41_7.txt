; SPIR-V
; Version: 1.0
; Generator: OpenAI
; Bound: 9999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
%fn_void = OpTypeFunction %void

; GLSL std450 extension
%GLSL = OpExtInstImport "GLSL.std.450"

; Input variables (Vertex shader outputs as inputs to fragment shader)
%ptr_Input_vec3 = OpTypePointer Input %vec3
%ptr_Input_vec2 = OpTypePointer Input %vec2

%in_worldPos = OpVariable %ptr_Input_vec3 Input
%in_normal   = OpVariable %ptr_Input_vec3 Input
%in_uv       = OpVariable %ptr_Input_vec2 Input
%in_tangent  = OpVariable %ptr_Input_vec3 Input
%in_color    = OpVariable %ptr_Input_vec3 Input

OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4

; Output variable
%ptr_Output_vec4 = OpTypePointer Output %vec4
%outColor = OpVariable %ptr_Output_vec4 Output
OpDecorate %outColor Location 0

; Uniform buffer (DescriptorSet 0, Binding 0)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

%ptr_UBO = OpTypePointer Uniform %UBO
%ubo = OpVariable %ptr_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Pointers to UBO members
%ptr_mat4_Uniform = OpTypePointer Uniform %mat4
%ptr_vec4_Uniform = OpTypePointer Uniform %vec4

%modelPtr = OpAccessChain %ptr_mat4_Uniform %ubo 0
%viewPtr  = OpAccessChain %ptr_mat4_Uniform %ubo 1
%projPtr  = OpAccessChain %ptr_mat4_Uniform %ubo 2
%lightPosPtr = OpAccessChain %ptr_vec4_Uniform %ubo 3
%cameraPosPtr = OpAccessChain %ptr_vec4_Uniform %ubo 4
%paramsPtr = OpAccessChain %ptr_vec4_Uniform %ubo 5

; Constants
%one = OpConstant %float 1.0
%zero = OpConstant %float 0.0
%half = OpConstant %float 0.5
%three = OpConstant %float 3.0
%C2 = OpConstant %float 0.2
%C3 = OpConstant %float 0.3
%C8 = OpConstant %float 0.8
%C5 = OpConstant %float 0.5

; Entry point
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft

; Function declarations
%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %in_worldPos
%worldNormal = OpLoad %vec3 %in_normal

; Load UBO data
%lightPosVec4 = OpLoad %vec4 %lightPosPtr
%lightPosXYZ = OpCompositeExtract %vec3 %lightPosVec4 0 1 2

; Compute L = normalize(lightPos.xyz - worldPos)
%Ldir = OpFSub %vec3 %lightPosXYZ %worldPos
%L = OpExtInst %vec3 %GLSL Normalize %Ldir

; Normalize N
%N = OpExtInst %vec3 %GLSL Normalize %worldNormal

; N · L
%dotNL = OpExtInst %float %GLSL Dot %N %L
; diffuse = max(dot(N,L), 0)
%diffuse = OpExtInst %float %GLSL Max %dotNL %zero

; baseFactor = 0.3 + 0.5 * diffuse
%halfDiffuse = OpFMul %float %half %diffuse
%baseFactor = OpFAdd %float %C3 %halfDiffuse

; baseColor = vec3(0.2, 0.3, 0.8) * baseFactor
%baseColorVec3 = OpCompositeConstruct %vec3 %C2 %C3 %C8
%modulatedBase = OpVectorTimesScalar %vec3 %baseColorVec3 %baseFactor

; Rim calculation: rim = pow(1 - diffuse, 3)
%oneMinus = OpFSub %float %one %diffuse
%rim = OpExtInst %float %GLSL Pow %oneMinus %three

; Rim color = vec3(1.0, 0.8, 0.5) * rim
%rimVec3 = OpCompositeConstruct %vec3 %one %C8 %C5
%rimColor = OpVectorTimesScalar %vec3 %rimVec3 %rim

; Final color = modulatedBase + rimColor
%finalColor3 = OpFAdd %vec3 %modulatedBase %rimColor
; Output alpha = 1.0
%finalColor4 = OpCompositeConstruct %vec4 %finalColor3 %one

; Store to output
OpStore %outColor %finalColor4

OpReturn
OpFunctionEnd
