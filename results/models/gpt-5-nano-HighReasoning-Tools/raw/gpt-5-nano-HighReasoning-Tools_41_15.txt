; SPIR-V
; Version: 1.0
; Generator: 0
; Bound: 9999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Pointer types
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %ubo_struct
%_ptr_Uniform_vec4 = OpTypePointer Uniform %vec4

; UBO struct
%ubo_struct = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
OpMemberDecorate %ubo_struct 0 Offset 0
OpMemberDecorate %ubo_struct 1 Offset 64
OpMemberDecorate %ubo_struct 2 Offset 128
OpMemberDecorate %ubo_struct 3 Offset 192
OpMemberDecorate %ubo_struct 4 Offset 208
OpMemberDecorate %ubo_struct 5 Offset 224
OpDecoration %ubo_struct Block

; Inputs/Outputs
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
%in_normal   = OpVariable %_ptr_Input_vec3 Input
%in_uv       = OpVariable %_ptr_Input_vec2 Input
%in_tangent  = OpVariable %_ptr_Input_vec3 Input
%in_color    = OpVariable %_ptr_Input_vec3 Input
%outColor    = OpVariable %_ptr_Output_vec4 Output
%ubo         = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %in_worldPos Location 0
OpDecorate %in_normal Location 1
OpDecorate %in_uv Location 2
OpDecorate %in_tangent Location 3
OpDecorate %in_color Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; GLSL std450 extension
%ext = OpExtInstImport "GLSL.std.450"

; Constants
%int = OpTypeInt 32 Signed
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_2 = OpConstant %float 0.2
%float_0_4 = OpConstant %float 0.4
%float_0_5 = OpConstant %float 0.5
%float_0_7 = OpConstant %float 0.7
%float_0_75 = OpConstant %float 0.75
%float_1_0 = OpConstant %float 1.0
%int_0 = OpConstant %int 0
%int_3 = OpConstant %int 3

; Function declaration
%func_type = OpTypeFunction %void
%main = OpFunction %void None %func_type
%entry = OpLabel

; Local temps
%worldPos = OpLoad %vec3 %in_worldPos
%normalRaw = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %ext GLSLstd450Normalize %normalRaw

%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %int_3
%lightPos = OpLoad %vec4 %lightPos_ptr
%x = OpCompositeExtract %float %lightPos 0
%y = OpCompositeExtract %float %lightPos 1
%z = OpCompositeExtract %float %lightPos 2
%lightPos_xyz = OpCompositeConstruct %vec3 %x %y %z

%wx = OpCompositeExtract %float %worldPos 0
%wy = OpCompositeExtract %float %worldPos 1
%wz = OpCompositeExtract %float %worldPos 2
%worldPos_xyz = OpCompositeConstruct %vec3 %wx %wy %wz

%diff = OpFSub %vec3 %lightPos_xyz %worldPos_xyz
%L = OpExtInst %vec3 %ext GLSLstd450Normalize %diff

%dotNL = OpExtInst %float %ext GLSLstd450Dot %N %L
%dotNL_half = OpFMul %float %dotNL %float_0_5
%t = OpFAdd %float %dotNL_half %float_0_5

; Gooch colors
%cool = OpConstantComposite %vec3 %float_0_2 %float_0_2 %float_0_75
%warm = OpConstantComposite %vec3 %float_0_7 %float_0_7 %float_0_4

%oneMinusT = OpFSub %float %float_1 %t
%coolScaled = OpVectorTimesScalar %vec3 %cool %oneMinusT
%warmScaled = OpVectorTimesScalar %vec3 %warm %t
%gooch3 = OpFAdd %vec3 %coolScaled %warmScaled

%outVec4 = OpCompositeConstruct %vec4 %gooch3 %float_1
OpStore %outColor %outVec4

OpReturn
OpFunctionEnd

; Entry point and execution mode
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %outColor
OpExecutionMode %main OriginUpperLeft
