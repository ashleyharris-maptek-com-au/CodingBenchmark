; SPIR-V geometry shader implementing 8 radial line shafts with arrowheads
; Input primitive: InputLines (2 vertices per invocation)
; Output: 7 vertices total (4 for shaft as a quad-strip, 3 for arrowhead)
; Shaft color: color[0], Arrow color: color[1]

               OpCapability Geometry
               OpMemoryModel Logical GLSL450

; GLSL.std.450 ext for sqrt
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 1

; Convenience constants
%float_1 = OpConstant %float 1.0
%float_0 = OpConstant %float 0.0
%float_015 = OpConstant %float 0.015
%float_075 = OpConstant %float 0.75
%float_050 = OpConstant %float 0.05
%float_1_0 = OpConstant %float 1.0
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3

; Vector types aliases
%vec3 = %v3float
%vec4 = %v4float

; Inputs from vertex shader (per-vertex arrays)
; gl_in (array of gl_PerVertex_in) for two vertices
; gl_PerVertex_in { vec4 gl_Position; } is the per-vertex input block
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block

; Array of 2 elements (InputLines -> 2 vertices)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Point-wise inputs from vertex shader (arrays of vec3, size 2)
%arr_vec3_2 = OpTypeArray %vec3 %uint_2
%ptr_in_arr_worldPos = OpTypePointer Input %arr_vec3_2
%worldPos_in = OpVariable %ptr_in_arr_worldPos Input

%ptr_in_arr_normal = OpTypePointer Input %arr_vec3_2
%normal_in = OpVariable %ptr_in_arr_normal Input

%ptr_in_arr_color = OpTypePointer Input %arr_vec3_2
%color_in = OpVariable %ptr_in_arr_color Input

; Output to fragment shader and per-vertex position
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block

%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

; Output per-vertex color (to fragment shader)
%ptr_out_color = OpTypePointer Output %vec3
%outColor = OpVariable %ptr_out_color Output
; Also expose the color as an interface (location 0)
OpDecorate %outColor Location 0

; Read-only UBO (set=0, binding=0) placeholder (layout matches test 41 roughly)
%UBOType = OpTypeStruct %float
%ptr_ubo = OpTypePointer Uniform %UBOType
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; Execution model interface (OpEntryPoint must list ALL interface vars)
; We'll include all relevant interface vars: gl_in, worldPos_in, normal_in, color_in, gl_out, outColor
; Entry point
%main = OpFunction %void None %void_fn
%entry = OpLabel
; (Main function body will be emitted below)

; Function declarations
; Function type and a minimal void function type
%void_fn = OpTypeFunction %void
; Local IDs used in code
; Indices for array elements
%idx_0 = OpConstant %uint 0
%idx_1 = OpConstant %uint 1
%idx_2 = OpConstant %uint 2
%idx_3 = OpConstant %uint 3

; Pointers to per-vertex inputs
%ptr_in_v3 = OpTypePointer Input %vec3
%ptr_in_v4 = OpTypePointer Input %v4float
; Pointers to per-vertex inputs (worldPos, color)
%ptr_worldPos = OpTypePointer Input %arr_vec3_2
%ptr_color = OpTypePointer Input %arr_vec3_2

; Pointers to outputs
%ptr_out_v4 = OpTypePointer Output %v4float

; Initialize: declare the entry point interface
; OpEntryPoint Geometry %main "main" gl_in worldPos_in normal_in color_in gl_out outColor
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor

; Execution modes
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

; The actual entry point implementation (body)
; We'll redefine the function body now with the actual body
; (End of header; begin actual body)
OpName %main "main"

; Begin function body (redeclare function header with actual body)
%main = OpFunction %void None %void_fn
%entry = OpLabel

; Load endpoints p0 and p1 from worldPos_in
%p0 = OpLoad %arr_vec3_2 %worldPos_in
%p1 = OpLoad %arr_vec3_2 %worldPos_in
; But to access a specific element, use OpAccessChain-like approach:
; Access gl_in[0].gl_Position? Our p0/p1 come from worldPos_in[0] and worldPos_in[1]
%p0_ptr = OpAccessChain %ptr_in_v3 %worldPos_in %idx_0
%p0_vec = OpLoad %vec3 %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v3 %worldPos_in %idx_1
%p1_vec = OpLoad %vec3 %p1_ptr

; Load colors c0, c1
%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %idx_0
%c0_vec = OpLoad %vec3 %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %idx_1
%c1_vec = OpLoad %vec3 %c1_ptr

; Extract components of p0 and p1
%p0_x = OpCompositeExtract %float %p0_vec 0
%p0_y = OpCompositeExtract %float %p0_vec 1
%p0_z = OpCompositeExtract %float %p0_vec 2

%p1_x = OpCompositeExtract %float %p1_vec 0
%p1_y = OpCompositeExtract %float %p1_vec 1
%p1_z = OpCompositeExtract %float %p1_vec 2

; dir = p1 - p0
%dx = OpFSub %float %p1_x % %p0_x
%dy = OpFSub %float %p1_y % %p0_y

; perp = normalize(-dir.y, dir.x)
%neg_dy = OpFNegate %float %dy
%perp_x = %neg_dy
%perp_y = %dx

; length = sqrt(perp_x^2 + perp_y^2)
%perp_x2 = OpFMul %float %perp_x %perp_x
%perp_y2 = OpFMul %float %perp_y %perp_y
%len = OpFAdd %float %perp_x2 %perp_y2
%len_sqrt = OpExtInst %float %GLSLstd450 "sqrt" %len
; Normalized perpendicular
%perp_nx = OpFDiv %float %perp_x % %len_sqrt
%perp_ny = OpFDiv %float %perp_y % %len_sqrt

; mid = p0 + 0.75 * (p1 - p0)
%dx075 = OpFMul %float %dx % %float_075
%dy075 = OpFMul %float %dy % %float_075
%mid_x = OpFAdd %float %p0_x % %dx075
%mid_y = OpFAdd %float %p0_y % %dy075
%dz = OpFSub %float %p1_z % %p0_z
%mid_z = OpFAdd %float %p0_z % (OpFMul %float %dz % %float_075)

; Shaft thickness
%thick = %float_015
; offset = perp * thick
%offset_x = OpFMul %float %perp_nx % %float_015
%offset_y = OpFMul %float %perp_ny % %float_015

; W = 1.0
%w1 = %float_1

; ---- Emit shaft vertices (4 verts) ----
; v0 = p0 - offset
%v0x = OpFSub %float %p0_x % %offset_x
%v0y = OpFSub %float %p0_y % %offset_y
%v0z = %p0_z
%v0 = OpCompositeConstruct %v4float %v0x %v0y %v0z % %float_1
; write gl_Position
%gl_pos0 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos0 %v0
; color = c0
%c0x = OpCompositeExtract %float %c0_vec 0
%c0y = OpCompositeExtract %float %c0_vec 1
%c0z = OpCompositeExtract %float %c0_vec 2
%c0_out = OpCompositeConstruct %vec3 %c0x %c0y %c0z
OpStore %outColor %c0_out
OpEmitVertex

; v1 = p0 + offset
%v1x = OpFAdd %float %p0_x % %offset_x
%v1y = OpFAdd %float %p0_y % %offset_y
%v1z = %p0_z
%v1 = OpCompositeConstruct %v4float %v1x %v1y %v1z % %float_1
%gl_pos1 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos1 %v1
OpStore %outColor %c0_out
OpEmitVertex

; v2 = mid - offset
%v2x = OpFSub %float %mid_x % %offset_x
%v2y = OpFSub %float %mid_y % %offset_y
%v2z = %mid_z
%v2 = OpCompositeConstruct %v4float %v2x %v2y %v2z % %float_1
%gl_pos2 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos2 %v2
OpStore %outColor %c0_out
OpEmitVertex

; v3 = mid + offset
%v3x = OpFAdd %float %mid_x % %offset_x
%v3y = OpFAdd %float %mid_y % %offset_y
%v3z = %mid_z
%v3 = OpCompositeConstruct %v4float %v3x %v3y %v3z % %float_1
%gl_pos3 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos3 %v3
OpStore %outColor %c0_out
OpEmitVertex

; End shaft primitive
OpEndPrimitive

; ---- Emit arrowhead (triangle strip with 3 verts) ----
; arrow length factor
%arrow = %float_050
%arrow_x = OpFMul %float %perp_nx % %arrow
%arrow_y = OpFMul %float %perp_ny % %arrow

; v4 = mid + perp*0.05
%v4x = OpFAdd %float %mid_x % %arrow_x
%v4y = OpFAdd %float %mid_y % %arrow_y
%v4z = %mid_z
%v4 = OpCompositeConstruct %v4float %v4x %v4y %v4z % %float_1
%gl_pos4 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos4 %v4
OpStore %outColor %c1_out
OpEmitVertex

; v5 = mid - perp*0.05
%v5x = OpFSub %float %mid_x % %arrow_x
%v5y = OpFSub %float %mid_y % %arrow_y
%v5z = %mid_z
%v5 = OpCompositeConstruct %v4float %v5x %v5y %v5z % %float_1
%gl_pos5 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos5 %v5
OpStore %outColor %c1_out
OpEmitVertex

; v6 = p1
%v6x = %p1_x
%v6y = %p1_y
%v6z = %p1_z
%v6 = OpCompositeConstruct %v4float %v6x %v6y %v6z % %float_1
%gl_pos6 = OpAccessChain %ptr_out_v4 %gl_out %uint_0
OpStore %gl_pos6 %v6
Op StoreColor?_note: color for v6 (arrowhead tip) uses c1
OpStore %outColor %c1_out
OpEmitVertex

; End arrowhead primitive
OpEndPrimitive

; Finish main
OpReturn
OpFunctionEnd
