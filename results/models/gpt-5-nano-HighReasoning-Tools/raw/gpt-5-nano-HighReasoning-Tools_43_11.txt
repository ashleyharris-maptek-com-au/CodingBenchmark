struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;    // (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;   // (0.0, 0.0, 3.0, 1.0)
    float4 params;      // (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0
{
    // 6x6 UV polka-dot pattern with circular dots of radius 0.3 in cell space
    const float cells = 6.0;
    float2 uvScaled = input.uv * cells;
    float2 local = frac(uvScaled); // coordinates inside each 1/6 cell [0,1)
    float2 toCenter = local - float2(0.5, 0.5);
    float dist = length(toCenter);

    float3 bg  = float3(0.1, 0.1, 0.3);
    float3 dotC = float3(1.0, 0.8, 0.0);
    float3 color = (dist <= 0.3) ? dotC : bg;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float lambert = 0.3 + 0.7 * diffuse;

    float3 finalColor = color * lambert;
    return float4(finalColor, 1.0);
}
