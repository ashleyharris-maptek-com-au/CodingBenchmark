; SPIR-V
; Version: 1.0
; Bound: 999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%funcType = OpTypeFunction %void

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Uniform Buffer Object (std140-like layout offsets)
%UBOType = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%UBOPtr = OpTypePointer Uniform %UBOType

; IO types and pointers
%InputWorldPosPtr = OpTypePointer Input %vec3
%InputNormalPtr   = OpTypePointer Input %vec3
%InputUVPtr       = OpTypePointer Input %vec2
%InputTangentPtr  = OpTypePointer Input %vec3
%InputColorPtr    = OpTypePointer Input %vec3
%OutColorPtr      = OpTypePointer Output %vec4

; Variables
%InWorldPos = OpVariable %InputWorldPosPtr Input
%InNormal   = OpVariable %InputNormalPtr Input
%InUV       = OpVariable %InputUVPtr Input
%InTangent  = OpVariable %InputTangentPtr Input
%InColor    = OpVariable %InputColorPtr Input
%OutColor   = OpVariable %OutColorPtr Output
%UBO        = OpVariable %UBOPtr Uniform

; Decorations
OpDecorate %InWorldPos Location 0
OpDecorate %InNormal Location 1
OpDecorate %InUV Location 2
OpDecorate %InTangent Location 3
OpDecorate %InColor Location 4
OpDecorate %OutColor Location 0
OpDecorate %UBO DescriptorSet 0
OpDecorate %UBO Binding 0

OpMemberDecorate %UBOType 0 Offset 0
OpMemberDecorate %UBOType 1 Offset 64
OpMemberDecorate %UBOType 2 Offset 128
OpMemberDecorate %UBOType 3 Offset 192
OpMemberDecorate %UBOType 4 Offset 208
OpMemberDecorate %UBOType 5 Offset 224
OpDecorate %UBOType Block

OpEntryPoint Fragment %main "main" %InWorldPos %InNormal %InUV %InTangent %InColor %OutColor
OpExecutionMode %main OriginUpperLeft

; GLSL std450 extension (for builtins)
%GLSL = OpExtInstImport "GLSL.std.450"

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_0_3 = OpConstant %float 0.3
%float_0_7 = OpConstant %float 0.7
%float_0_5 = OpConstant %float 0.5
%float_0_1 = OpConstant %float 0.1
%float_6 = OpConstant %float 6.0
%float_0_8 = OpConstant %float 0.8
%float_0_0 = OpConstant %float 0.0

; Vectors/constants for polka-dot colors
%vec3_bg = OpConstantComposite %vec3 %float_0_1 %float_0_1 %float_0_3
%vec3_yellow = OpConstantComposite %vec3 %float_1 %%float_0_8 %%float_0_0
; Center and radius in cell space
%vec2_0_5 = OpConstantComposite %vec2 %float_0_5 %float_0_5
%vec2_6 = OpConstantComposite %vec2 %float_6 %float_6
%vec2Frag = OpConstantComposite %vec2 %float_6 %float_6  ; alias for uv*6

; Main function
%main = OpFunction %void None %funcType
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %InWorldPos
%nrmIn    = OpLoad %vec3 %InNormal
;n = normalize(normal)
%n = OpExtInst %vec3 %GLSL Normalize %nrmIn

; Load UBO and light position
%uboVal = OpLoad %UBOType %UBO
%lightPos4 = OpCompositeExtract %vec4 %uboVal 3
%lpx = OpCompositeExtract %float %lightPos4 0
%lpy = OpCompositeExtract %float %lightPos4 1
%lpz = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lpx %lpy %lpz
; lightDir = normalize(lightPos - worldPos)
%toLight = OpFSub %vec3 %lightPos3 % worldPos
%lightDir = OpExtInst %vec3 %GLSL Normalize %toLight

; Lambertian diffuse
%dotNL = OpExtInst %float %GLSL Dot %n %lightDir
%zero = %float_0
%diffuse = OpFMax %float %dotNL % %zero
%part = OpFMul %float %diffuse %float_0_7
%shade = OpFAdd %float %float_0_3 % %part

; UV processing for 6x6 grid with dots of radius 0.3 in cell space
%uv = OpLoad %vec2 %InUV
%uv6 = OpFMul %vec2 %uv % vec2Frag
%uv6fract = OpExtInst %vec2 %GLSL Fract %uv6
%center = %vec2_0_5
%diffCenter = OpFSub %vec2 %uv6fract % center
%dist = OpExtInst %float %GLSL Length %diffCenter
%radius = %float_0_3
%dotMask = OpExtInst %float %GLSL Step %radius %dist

; Dot color blending with background using dotMask
%colorMix = OpExtInst %vec3 %GLSL Mix %vec3_bg %vec3_yellow %dotMask
%colorShaded = OpFMul %vec3 %colorMix % shade

; Output with alpha = 1.0
%color4 = OpCompositeConstruct %vec4 %colorShaded %float_1
OpStore %OutColor %color4

OpReturn
OpFunctionEnd

; End of shader
