cbuffer UBO : register(b0)
{
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

float4 main(PSInput input) : SV_Target0
{
    // Normalize world-space normal
    float3 N = normalize(input.normal);

    // Light direction in world space
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Gooch shading parameter t (provided in the task)
    float t = dot(N, L) * 0.5 + 0.5;
    t = clamp(t, 0.0, 1.0);

    // Cool and warm Gooch colors
    float3 coolColor = float3(0.2, 0.2, 0.75);
    float3 warmColor = float3(0.7, 0.7, 0.4);

    // Gooch color
    float3 gooColor = lerp(coolColor, warmColor, t);

    // Optional per-vertex color modulation
    float3 finalColor = gooColor * input.color;

    return float4(finalColor, 1.0);
}
