using System;
using System.Collections.Generic;
using System.Diagnostics;

namespace TetrisOptimalPlayer
{
    // Simple, fast stdin/stdout based Tetris AI per problem spec
    // Reads: width height numPieces
    // For each piece: reads pieceType (I,O,T,S,Z,J,L)
    // Writes: "rotation column" (rotation: 0-3, column: 0..width-1)
    // Reads: "ok" | "cleared_N" | "gameover"
    // Maintains its own board state and clears lines after each placement.

    class Program
    {
        // Piece type enum
        enum PieceType { I, O, T, S, Z, J, L }

        // Rotation shape container
        struct RotationShape
        {
            public int Width;
            public int Height;
            // Blocks as [x0,y0,x1,y1,...]
            public int[] Blocks;
            public RotationShape(int w, int h, int[] blocks)
            {
                Width = w; Height = h; Blocks = blocks;
            }
        }

        // Static rotations database: one 2D array per piece type, with 4 rotations
        // Each rotation stores width, height and block coordinates (minX,minY = 0)
        static readonly RotationShape[][] Rotations = InitRotations();

        // Init all piece rotations (minX/minY anchored at 0)
        static RotationShape[][] InitRotations()
        {
            // Helper to create rotation quickly
            RotationShape rs(int w, int h, int[] b) => new RotationShape(w, h, b);

            // I
            var I = new RotationShape[]
            {
                new RotationShape(4,1, new int[]{0,0,1,0,2,0,3,0}), // rot0
                new RotationShape(1,4, new int[]{0,0,0,1,0,2,0,3}), // rot1
                new RotationShape(4,1, new int[]{0,0,1,0,2,0,3,0}), // rot2
                new RotationShape(1,4, new int[]{0,0,0,1,0,2,0,3})  // rot3
            };

            // O
            var O = new RotationShape[]

                new RotationShape(2,2, new int[]{0,0,1,0,0,1,1,1}),
                new RotationShape(2,2, new int[]{0,0,1,0,0,1,1,1}),
                new RotationShape(2,2, new int[]{0,0,1,0,0,1,1,1}),
            };

            // T
            var T = new RotationShape[]
            {
                // rot0
                new RotationShape(3,2, new int[]{0,0,1,0,2,0,1,1}),
                // rot1
                new RotationShape(2,3, new int[]{0,0,0,1,0,2,1,1}),
                // rot2
                new RotationShape(3,2, new int[]{0,1,1,1,2,1,1,0}),
                // rot3
                new RotationShape(2,3, new int[]{0,1,1,0,1,1,1,2})

            // S
            var S = new RotationShape[]

                new RotationShape(3,2, new int[]{1,0,2,0,0,1,1,1}),
                // rot1
                new RotationShape(2,3, new int[]{0,0,0,1,1,1,1,2}),
                // rot2
                new RotationShape(3,2, new int[]{0,1,1,1,1,0,2,0}),

                new RotationShape(3,2, new int[]{0,0,1,0,1,1,2,1}),

                new RotationShape(2,3, new int[]{1,0,1,1,1,2,0,0}),

                new RotationShape(3,2, new int[]{0,0,0,1,1,0,2,0}),

                new RotationShape(2,3, new int[]{0,0,0,1,0,2,1,2}),

                new RotationShape(2,3, new int[]{0,2,1,2,1,1,1,0}),

            // Assemble into 7-piece array with 4 rotations each
            RotationShape[][] all = new RotationShape[7][];
            all[(int)PieceType.I] = I;
            all[(int)PieceType.O] = O;
            all[(int)PieceType.T] = T;

        // Fast scanner for whitespace-delimited tokens
        class FastScanner
        {
            private readonly System.IO.StreamReader _r;
            private readonly Queue<string> _q = new Queue<string>();
            public FastScanner()
            {
                _r = new System.IO.StreamReader(Console.OpenStandardInput());
            }
            public string Next()
            {
                while (_q.Count == 0)
                {
                    string line = _r.ReadLine();
                    if (line == null) return null;
                    var parts = line.Split((char[])null, StringSplitOptions.RemoveEmptyEntries);
                    foreach (var p in parts) _q.Enqueue(p);
                }
                return _q.Dequeue();
            }
        }

        static void Main(string[] args)
        {
            var scanner = new FastScanner();

            // Read board and game meta
            string tok = scanner.Next();
            if (tok == null) return;
            int width = int.Parse(tok);
            int height = int.Parse(scanner.Next());
            int totalPieces = int.Parse(scanner.Next());

            // Initialize empty board: board[y][x], y=0 bottom
            bool[][] board = new bool[height][];
            for (int y = 0; y < height; y++) board[y] = new bool[width];

            // Helper: clone board
            bool[][] CloneBoard(bool[][] src)
            {
                bool[][] c = new bool[height][];
                for (int y = 0; y < height; y++) { c[y] = new bool[width]; Array.Copy(src[y], c[y], width); }
                return c;
            }

            // Helper: collision test for a given rotation shape and placement
            bool CollisionAt(bool[][] b, RotationShape rs, int col, int yOrigin)
            {
                int bl = rs.Blocks.Length;
                for (int i = 0; i < bl; i += 2)
                {
                    int bx = rs.Blocks[i];
                    int by = rs.Blocks[i + 1];
                    int x = col + bx;
                    int y = yOrigin + by;
                    if (x < 0 || x >= width || y < 0 || y >= height) return true;
                    if (b[y][x]) return true;
                }
                return false;
            }

            // Try to drop piece; returns (success, landingY)
            bool TryDrop(bool[][] b, RotationShape rs, int col, out int landingY)
            {
                landingY = -1;
                // yOrigin is top-left anchor's bottom-left origin in our coordinate system
                // The piece's height is rs.Height; lowest yOrigin is 0; highest is height - rs.Height
                for (int y = height - rs.Height; y >= 0; y--)
                {
                    if (CollisionAt(b, rs, col, y)) continue;
                    // If we move one more down, would collide or be out of bounds
                    if (y == 0 || CollisionAt(b, rs, col, y - 1))
                    {
                        landingY = y;
                        return true;
                    }
                }
                return false;
            }

            // Place piece on board (assumes valid landing)
            void PlaceOnBoard(bool[][] b, RotationShape rs, int col, int landingY)
            {
                for (int i = 0; i < rs.Blocks.Length; i += 2)
                {
                    int x = col + rs.Blocks[i];
                    int y = landingY + rs.Blocks[i + 1];
                    b[y][x] = true;
                }
            }

            // Clear full lines; return number of lines cleared
            int ClearLines(bool[][] b)
            {
                List<bool[]> newRows = new List<bool[]>();
                for (int y = 0; y < height; y++)
                {
                    bool full = true;
                    for (int x = 0; x < width; x++) { if (!b[y][x]) { full = false; break; } }
                    if (!full) newRows.Add(b[y]);
                }
                int cleared = height - newRows.Count;
                // Fill from bottom with remaining rows, then empty on top
                for (int y = 0; y < newRows.Count; y++) b[y] = newRows[y];
                for (int y = newRows.Count; y < height; y++) b[y] = new bool[width];
                return cleared;
            }

            // Evaluation function: simple heuristic scoring
            int EvaluateBoard(bool[][] b)
            {
                // heights per column
                int[] colH = new int[width];
                for (int x = 0; x < width; x++)
                {
                    int h = 0;
                    for (int y = height - 1; y >= 0; y--)
                    {
                        if (b[y][x])
                        {
                            h = y + 1;
                            break;
                        }
                    }
                    colH[x] = h;
                }

                int aggregateHeight = 0;
                for (int x = 0; x < width; x++) aggregateHeight += colH[x];

                int holes = 0;
                for (int x = 0; x < width; x++)
                {
                    bool seenBlock = false;
                    for (int y = height - 1; y >= 0; y--)
                    {
                        if (b[y][x]) seenBlock = true;
                        else if (seenBlock) holes++;
                    }
                }

                int bumpiness = 0;
                for (int x = 1; x < width; x++) bumpiness += Math.Abs(colH[x] - colH[x - 1]);

                // count complete lines
                int completeLines = 0;
                for (int y = 0; y < height; y++)
                {
                    bool full = true;
                    for (int x = 0; x < width; x++) if (!b[y][x]) { full = false; break; }
                    if (full) completeLines++;
                }

                // Score: prioritize lines, penalize height/holes/bumpiness
                int score = completeLines * 1000 - aggregateHeight * 4 - holes * 6 - bumpiness * 2;
                // Add a tiny bonus for more completed lines in a row (not strictly necessary)
                score += completeLines * 5;
                return score;
            }

            // Decode piece type from string
            PieceType ParseType(string s)
            {
                switch (s)
                {
                    case "I": return PieceType.I;
                    case "O": return PieceType.O;
                    case "T": return PieceType.T;
                    case "S": return PieceType.S;
                    case "Z": return PieceType.Z;
                    case "J": return PieceType.J;
                    case "L": return PieceType.L;
                    default: throw new ArgumentException("Unknown piece type: " + s);
                }
            }

            // Decision: choose best move using a time budget
            (int bestRot, int bestCol) DecideMove(PieceType t, bool[][] curBoard, int w, int h, int piecesLeft)
            {
                RotationShape[] rots = Rotations[(int)t];
                // Time budget per piece
                // Simple adaptive budget: a bit more when many pieces are left
                int budgetMs = Math.Max(60, Math.Min(800, piecesLeft * 8));

                var sw = Stopwatch.StartNew();
                int bestRot = 0;
                int bestCol = 0;
                long bestScore = long.MinValue;
                bool foundAny = false;
                bool[][] bestBoardSnapshot = null;

                // Enumerate candidate moves
                for (int rot = 0; rot < 4; rot++)
                {
                    var rs = rots[rot];
                    if (rs.Width > w || rs.Height > h) continue;
                    int colMax = w - rs.Width;
                    for (int col = 0; col <= colMax; col++)
                    {
                        // Quick skip if even the top-left would collide immediately
                        if (CollisionAt(curBoard, rs, col, height - rs.Height)) continue;

                        // Try drop
                        if (!TryDrop(curBoard, rs, col, out int landingY)) continue;

                        // Copy board and place
                        bool[][] tmp = CloneBoard(curBoard);
                        PlaceOnBoard(tmp, rs, col, landingY);
                        int cleared = ClearLines(tmp);
                        int score = EvaluateBoard(tmp);

                        if (!foundAny || score > bestScore)
                        {
                            bestScore = score;
                            bestRot = rot;
                            bestCol = col;
                            bestBoardSnapshot = tmp;
                            foundAny = true;
                        }

                        if (sw.ElapsedMilliseconds >= budgetMs) goto DONE; // time budget reached
                    }
                }

            DONE:
                // If nothing found (should be rare), fall back to a safe default: first valid position
                if (!foundAny)
                {
                    // Try any rotation/col randomly (deterministic fallback)
                    for (int rot = 0; rot < 4; rot++)
                    {
                        var rs = rots[rot];
                        if ( rs.Width > w || rs.Height > h ) continue;
                        int colMax = w - rs.Width;
                        for (int col = 0; col <= colMax; col++)
                        {
                            if (CollisionAt(curBoard, rs, col, height - rs.Height)) continue;
                            if (TryDrop(curBoard, rs, col, out int landingY))
                            {
                                bestRot = rot;
                                bestCol = col;
                                foundAny = true;
                                bestBoardSnapshot = CloneBoard(curBoard);
                                PlaceOnBoard(bestBoardSnapshot, rs, col, landingY);
                                ClearLines(bestBoardSnapshot);
                                goto END_SAFE;
                            }
                        }
                    }
                }
            END_SAFE:
                // If still not found, default to rotation 0 column 0 (best effort)
                if (!foundAny) { bestRot = 0; bestCol = 0; }

                return (bestRot, bestCol);
            }

            // Local helpers using captured variables (already defined above to reduce param noise)

            // Start of main loop
            for (int iPiece = 0; iPiece < totalPieces; iPiece++)
            {
                string pieceToken = scanner.Next();
                if (pieceToken == null) break;
                PieceType t = ParseType(pieceToken);

                int piecesLeft = totalPieces - iPiece;
                var move = DecideMove(t, board, width, height, piecesLeft);

                // Output rotation and column
                Console.WriteLine($"{move.bestRot} {move.bestCol}");
                Console.Out.Flush();

                // Read result from judge
                string result = scanner.Next();
                if (result != null && result.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
                {
                    // Game over reported by judge; stop processing
                    break;
                }

                // Apply move to local board (simulate)
                // Recompute landing with chosen move to ensure consistent board state
                RotationShape chosenRs = Rotations[(int)t][move.bestRot];
                if (!TryDrop(board, chosenRs, move.bestCol, out int landing))
                {
                    // If we cannot drop (shouldn't happen), end gracefully
                    break;
                }
                PlaceOnBoard(board, chosenRs, move.bestCol, landing);
                ClearLines(board);
            }

            // End of program
        }

    }
