use std::collections::HashMap;
use std::io::{self, Read};

fn mitm(a: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = a.len();
    if n == 0 {
        return if target == 0 { Some(vec![]) } else { None };
    }

    let n1 = n / 2;
    let n2 = n - n1;
    let left = &a[..n1];
    let right = &a[n1..];

    let lim1 = 1usize << n1;
    let mut lefts: Vec<(i64, u64)> = Vec::with_capacity(lim1);
    for mask in 0..lim1 {
        let mut sum = 0i64;
        for i in 0..n1 {
            if (mask >> i) & 1 == 1 {
                sum += left[i];
            }
        }
        lefts.push((sum, mask as u64));
    }

    let lim2 = 1usize << n2;
    let mut rights: Vec<(i64, u64)> = Vec::with_capacity(lim2);
    for mask in 0..lim2 {
        let mut sum = 0i64;
        for i in 0..n2 {
            if (mask >> i) & 1 == 1 {
                sum += right[i];
            }
        }
        rights.push((sum, mask as u64));
    }

    // Build map for right sums: sum -> mask
    let mut map_right: HashMap<i64, u64> = HashMap::with_capacity(rights.len());
    for (sum, mask) in rights {
        // keep the first mask encountered for a given sum
        map_right.entry(sum).or_insert(mask);
    }

    // Try to find a pair that sums to target
    for (sum_l, mask_l) in lefts {
        let need = target - sum_l;
        if let Some(mask_r) = map_right.get(&need) {
            // Reconstruct indices from mask_l and mask_r
            let mut indices: Vec<usize> = Vec::new();

            for i in 0..n1 {
                if ((mask_l >> i) & 1) == 1 {
                    indices.push(i);
                }
            }
            for i in 0..n2 {
                if ((mask_r >> i) & 1) == 1 {
                    indices.push(n1 + i);
                }
            }
            return Some(indices);
        }
    }

    None
}

fn subset_sum_non_negative_dp(a: &[i64], target: i64) -> Option<Vec<usize>> {
    if target < 0 {
        return None;
    }
    let t = target as usize;
    if t == 0 {
        return Some(vec![]);
    }

    // dp[s] = last index used to reach sum s, -2 means unreachable, -1 means base (sum 0)
    let mut dp = vec![-2i32; t + 1];
    dp[0] = -1;

    for i in 0..a.len() {
        let w = a[i] as usize;
        if w > t {
            continue;
        }
        // iterate downward to avoid reusing the same element
        for s in (w..=t).rev() {
            if dp[s] == -2 && dp[s - w] != -2 {
                dp[s] = i as i32;
            }
        }
    }

    if dp[t] == -2 {
        return None;
    }

    // reconstruct
    let mut res: Vec<usize> = Vec::new();
    let mut s = t;
    while s > 0 {
        let idx = dp[s] as usize;
        res.push(idx);
        let w = a[idx] as usize;
        s -= w;
    }
    res.reverse();
    Some(res)
}

fn reconstruct_path(target: i64, path: &HashMap<i64, (i64, usize)>) -> Option<Vec<usize>> {
    let mut indices: Vec<usize> = Vec::new();
    let mut cur = target;
    while cur != 0 {
        let entry = path.get(&cur)?;
        let prev = entry.0;
        let idx = entry.1;
        indices.push(idx);
        cur = prev;
    }
    indices.reverse();
    Some(indices)
}

fn subset_sum_hash_dp(a: &[i64], target: i64) -> Option<Vec<usize>> {
    // Path compression via backpointers: sum -> (prev_sum, index_used)
    let mut path: HashMap<i64, (i64, usize)> = HashMap::new();
    path.insert(0, (0, usize::MAX)); // base

    let mut sums: Vec<i64> = vec![0];

    for i in 0..a.len() {
        let val = a[i];
        let mut new_sums: Vec<i64> = Vec::new();

        for &s in sums.iter() {
            let ns = s + val;
            if path.contains_key(&ns) {
                continue;
            }
            path.insert(ns, (s, i));
            new_sums.push(ns);

            if ns == target {
                // reconstruct path
                return reconstruct_path(target, &path);
            }
        }
        sums.extend(new_sums);
    }

    None
}

fn main() {
    // Read entire input
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    // Read n and target
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("NO");
            return;
        }
    };
    let target: i64 = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => {
            println!("NO");
            return;
        }
    };

    // Read numbers
    let mut a: Vec<i64> = Vec::with_capacity(n);
    for _ in 0..n {
        if let Some(tok) = it.next() {
            let val: i64 = tok.parse().unwrap();
            a.push(val);
        } else {
            break;
        }
    }

    // If actual number count differs, adjust n
    let n = a.len();

    // Quick edge case: target 0 -> empty subset is valid
    if target == 0 {
        println!("YES");
        println!("[]");
        return;
    }

    // Decide strategy
    let result: Option<Vec<usize>> = if n == 0 {
        None
    } else if a.iter().all(|&x| x >= 0) && target >= 0 {
        subset_sum_non_negative_dp(&a, target)
    } else if n <= 40 {
        mitm(&a, target)
    } else {
        // Fallback for mixed signs and larger inputs
        subset_sum_hash_dp(&a, target)
    };

    match result {
        Some(indices) => {
            println!("YES");
            // Build output like [i j k]
            let mut s = String::from("[");
            for (idx, val) in indices.iter().enumerate() {
                if idx > 0 {
                    s.push(' ');
                }
                s.push_str(&val.to_string());
            }
            s.push(']');
            println!("{}", s);
        }
        None => {
            println!("NO");
        }
    }
}
