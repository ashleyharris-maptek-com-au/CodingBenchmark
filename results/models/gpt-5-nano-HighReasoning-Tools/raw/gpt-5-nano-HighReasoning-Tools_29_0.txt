using System;
using System.IO;
using System.Numerics;
using System.Threading.Tasks;
using System.Diagnostics;
using System.Collections.Generic;

class Program
{
    // Fast input reader to handle large inputs efficiently
    private class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len = 0;
        private int _ptr = 0;

        public FastScanner()
        {
            _stream = Console.OpenStandardInput();
        }

        private int Read()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public bool ReadInt(out int value)
        {
            value = 0;
            int c;
            do
            {
                c = Read();
                if (c == -1) return false;
            } while (c <= ' ');
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            for (; c > ' '; c = Read())
            {
                value = value * 10 + (c - '0');
            }
            value *= sign;
            return true;
        }
    }

    // Exact maximum clique solver for n <= 64 using Tomita-style color pruning
    private static class ExactCliqueSolver
    {
        static ulong[] Adj;
        static int N;
        static int BestSize;
        static ulong BestMask;

        public static void Run(ulong[] adj, int n, out int bestSize, out ulong bestMask)
        {
            Adj = adj;
            N = n;
            BestSize = 0;
            BestMask = 0;

            ulong P = (N == 64) ? ulong.MaxValue : ((1UL << N) - 1);
            Expand(0UL, P, 0UL);

            bestSize = BestSize;
            bestMask = BestMask;
        }

        // Expand with current clique R, candidates P, excluded X
        private static void Expand(ulong R, ulong P, ulong X)
        {
            if (P == 0)
            {
                int sz = PopCount(R);
                if (sz > BestSize)
                {
                    BestSize = sz;
                    BestMask = R;
                }
                return;
            }

            // Build candidate list from P with their degrees inside P
            int nCand = 0;
            int[] candVerts = new int[64];
            int[] candDeg = new int[64];

            ulong tmp = P;
            while (tmp != 0)
            {
                int v = TrailingZeros(tmp);
                tmp &= tmp - 1;
                candVerts[nCand] = v;
                candDeg[nCand] = PopCount(Adj[v] & P);
                nCand++;
            }

            // If no candidates, update best and return
            if (nCand == 0) {
                int sz = PopCount(R);
                if (sz > BestSize)
                {
                    BestSize = sz;
                    BestMask = R;
                }
                return;
            }

            // Sort candVerts by deg descending
            for (int i = 0; i < nCand - 1; i++)
            {
                int maxIdx = i;
                for (int j = i + 1; j < nCand; j++)
                {
                    if (candDeg[j] > candDeg[maxIdx]) maxIdx = j;
                }
                if (maxIdx != i)
                {
                    int tv = candVerts[i]; candVerts[i] = candVerts[maxIdx]; candVerts[maxIdx] = tv;
                    int td = candDeg[i]; candDeg[i] = candDeg[maxIdx]; candDeg[maxIdx] = td;
                }
            }

            // Greedy coloring to bound
            int maxColor = 0;
            int[] color = new int[nCand];
            for (int i = 0; i < nCand; i++)
            {
                int v = candVerts[i];
                bool[] forbidden = new bool[maxColor + 2];
                for (int j = 0; j < i; j++)
                {
                    int u = candVerts[j];
                    if ((Adj[v] & (1UL << u)) != 0) forbidden[color[j]] = true;
                }
                int c = 1;
                while (forbidden[c]) c++;
                color[i] = c;
                if (c > maxColor) maxColor = c;
            }

            // Branch in reverse order (largest color first)
            for (int idx = nCand - 1; idx >= 0; idx--)
            {
                int v = candVerts[idx];
                if (PopCount(R) + color[idx] <= BestSize)
                    return;

                Expand(R | (1UL << v), P & Adj[v], X & Adj[v]);
                P &= ~(1UL << v);
                X |= (1UL << v);
            }
        }

        // Helpers
        private static int PopCount(ulong x) => (int)BitOperations.PopCount(x);
        private static int TrailingZeros(ulong x) => BitOperations.TrailingZeroCount(x);
        private static int TrailingZeros(uint x) => (int)BitOperations.TrailingZeroCount(x);
        private static int TrailingZeros(ushort x) => (int)BitOperations.TrailingZeroCount(x);
        private static int TrailingZeros(byte x) => (int)BitOperations.TrailingZeroCount(x);
    }

    // Large graph heuristic: CSR-based adjacency with parallel attempts
    private static class LargeGraphHeuristic
    {
        public static int[] Off;
        public static int[] Neigh;
        public static int N;
        public static int BestSize = 0;
        public static int[] BestClique = null;
        private static object _lock = new object();

        public static void Init(int n, int[] off, int[] neigh)
        {
            N = n;
            Off = off;
            Neigh = neigh;
        }

        public static void RunPar(int numTrials)
        {
            BestSize = 0;
            BestClique = null;

            // Base orderings
            int[] baseOrder = new int[N];
            for (int i = 0; i < N; i++) baseOrder[i] = i;

            List<int[]> orders = new List<int[]>();

            // Degree-aware orders
            int[] deg = new int[N];
            for (int i = 0; i < N; i++) deg[i] = Degree(i);
            int[] ordDesc = (int[])baseOrder.Clone();
            Array.Sort(ordDesc, (a, b) => deg[b].CompareTo(deg[a]));
            int[] ordAsc = (int[])baseOrder.Clone();
            Array.Sort(ordAsc, (a, b) => deg[a].CompareTo(deg[b]));
            int[] ordOrig = (int[])baseOrder.Clone();
            int[] ordRev = (int[])baseOrder.Clone();
            Array.Reverse(ordRev);

            orders.Add(ordDesc);
            orders.Add(ordAsc);
            orders.Add(ordOrig);
            orders.Add(ordRev);

            // Randomized orders
            var rndSeeds = new int[] { 42, 123, 999, 7777 };
            foreach (var s in rndSeeds)
            {
                int[] ord = (int[])baseOrder.Clone();
                Shuffle(ord, new Random(s));
                orders.Add(ord);
            }

            List<Task> tasks = new List<Task>();
            foreach (var order in orders)
            {
                tasks.Add(Task.Run(() => {
                    int[] clique = BuildCliqueFromOrder(order);
                    int sz = clique?.Length ?? 0;
                    if (sz > BestSize)
                    {
                        lock (_lock)
                        {
                            if (sz > BestSize)
                            {
                                BestSize = sz;
                                BestClique = clique;
                            }
                        }
                    }
                }));
            }

            // Wait all tasks
            Task.WaitAll(tasks.ToArray());

            // If nothing found, ensure at least one vertex
            if (BestClique == null || BestClique.Length == 0)
            {
                BestSize = 1;
                BestClique = new int[] { 0 };
            }
        }

        private static int Degree(int v)
        {
            int l = Off[v];
            int r = Off[v + 1];
            return r - l;
        }

        private static int[] BuildCliqueFromOrder(int[] order)
        {
            List<int> clique = new List<int>();
            foreach (int v in order)
            {
                bool ok = true;
                foreach (int u in clique)
                {
                    if (!EdgeExists(v, u)) { ok = false; break; }
                }
                if (ok) clique.Add(v);
            }
            if (clique.Count == 0) return new int[0];
            int[] res = clique.ToArray();
            Array.Sort(res); // return sorted for stable output
            return res;
        }

        private static bool EdgeExists(int a, int b)
        {
            int l = Off[a], r = Off[a + 1] - 1;
            while (l <= r)
            {
                int m = (l + r) >> 1;
                int val = Neigh[m];
                if (val == b) return true;
                if (val < b) l = m + 1;
                else r = m - 1;
            }
            return false;
        }

        private static void Shuffle(int[] arr, Random rng)
        {
            for (int i = arr.Length - 1; i > 0; i--)
            {
                int j = rng.Next(i + 1);
                int tmp = arr[i];
                arr[i] = arr[j];
                arr[j] = tmp;
            }
        }
    }

    static void Main(string[] args)
    {
        var fs = new FastScanner();

        if (!fs.ReadInt(out int n)) return;
        if (!fs.ReadInt(out int m)) return;

        int[] U = new int[m];
        int[] V = new int[m];
        int[] deg = new int[n];

        for (int i = 0; i < m; i++)
        {
            int u, v;
            fs.ReadInt(out u);
            fs.ReadInt(out v);
            U[i] = u;
            V[i] = v;
            deg[u]++;
            deg[v]++;
        }

        if (n <= 64)
        {
            // Build adjacency masks
            ulong[] adjMask = new ulong[n];
            for (int i = 0; i < m; i++)
            {
                int u = U[i], v = V[i];
                adjMask[u] |= (1UL << v);
                adjMask[v] |= (1UL << u);
            }

            ExactCliqueSolver.Run(adjMask, n, out int bestSize, out ulong bestMask);

            Console.WriteLine(bestSize);
            if (bestSize > 0)
            {
                bool first = true;
                for (int i = 0; i < n; i++)
                {
                    if ((bestMask & (1UL << i)) != 0)
                    {
                        if (!first) Console.Write(' ');
                        Console.Write(i);
                        first = false;
                    }
                }
            }
            Console.WriteLine();
        }
        else
        {
            // Large graph: Build CSR representation to save memory on per-vertex objects
            int[] off = new int[n + 1];
            for (int i = 0; i < n; i++) off[i + 1] = off[i] + deg[i];
            int totalNeighbors = off[n];

            int[] neigh = new int[2 * m];
            int[] pos = new int[n];
            for (int i = 0; i < n; i++) pos[i] = off[i];

            for (int i = 0; i < m; i++)
            {
                int u = U[i], v = V[i];
                neigh[pos[u]++] = v;
                neigh[pos[v]++] = u;
            }

            // Sort neighbor lists for binary search
            for (int i = 0; i < n; i++)
            {
                int l = off[i];
                int r = off[i + 1] - 1;
                if (l <= r)
                    Array.Sort(neigh, l, r - l + 1);
            }

            // Initialize heuristic solver
            LargeGraphHeuristic.Init(n, off, neigh);

            // Run several parallel heuristic attempts
            LargeGraphHeuristic.RunPar(8);

            int[] bestClique = LargeGraphHeuristic.BestClique;
            int bestSize = LargeGraphHeuristic.BestSize;

            // Output
            Console.WriteLine(bestSize);
            if (bestClique != null && bestClique.Length > 0)
            {
                // Ensure ascending order for output
                Array.Sort(bestClique);
                for (int i = 0; i < bestClique.Length; i++)
                {
                    if (i > 0) Console.Write(' ');
                    Console.Write(bestClique[i]);
                }
            }
            Console.WriteLine();
        }
    }
}
