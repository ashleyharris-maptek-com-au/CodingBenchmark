OpCapability Geometry
OpCapability Shader
OpExtension "GLSL.std.450"
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

; gl_in input layout (per-vertex data from vertex shader)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
%ptr_in_arr_pv = OpTypePointer Input %gl_PerVertex_in
%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%gl_in = OpVariable %ptr_in_arr_pv Input

; User inputs (arrays sized by vertex count)
%ptr_in_color = OpTypePointer Input %v3float
%color = OpVariable %ptr_in_color Input
%ptr_in_worldPos = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_worldPos Input
%ptr_in_normal = OpTypePointer Input %v3float
%normal = OpVariable %ptr_in_normal Input

; gl_out output layout
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output

; UBO placeholder (set=0, binding=0) with GLSL layout similarity
%mat4 = OpTypeMatrix %v4float 4
%UBOType = OpTypeStruct %mat4 %mat4 %v3float %v3float %float %float
OpDecorate %UBOType Block
%ptr_ubo = OpTypePointer Uniform %UBOType
%ubo = OpVariable %ptr_ubo Uniform

; Execution model entities
%main = OpFunction %void None %void_fn
%void_fn = OpTypeFunction %void
%entry = OpLabel

; Load input positions (p0 = gl_in[0].gl_Position, p1 = gl_in[1].gl_Position)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load input colors (c0 = color[0], c1 = color[1])
%c0_ptr = OpAccessChain %ptr_in_color %color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_color %color %int_1
%c1 = OpLoad %v3float %c1_ptr

; Compute 2D direction and perpendicular for ribbon thickness
; dx = p1.x - p0.x, dy = p1.y - p0.y
%p0_x = OpCompositeExtract %float %p0 0
%p0_y = OpCompositeExtract %float %p0 1
%p1_x = OpCompositeExtract %float %p1 0
%p1_y = OpCompositeExtract %float %p1 1
%dx = OpFSub %float %p1_x %p0_x
%dy = OpFSub %float %p1_y %p0_y

; len = sqrt(dx*dx + dy*dy)
%dx2 = OpFMul %float %dx %dx
%dy2 = OpFMul %float %dy %dy
%len_sq = OpFAdd %float %dx2 %dy2
%len = OpExtInst %float %GLSLstd450 Sqrt %len_sq

; nx = (-dy) / len, ny = dx / len
%neg_dy = OpFNegate %float %dy
%nx = OpFDiv %float %neg_dy %len
%ny = OpFDiv %float %dx %len

; thickness = 0.03
%thick = OpConstant %float 0.03
%perp_x = OpFMul %float %nx %thick
%perp_y = OpFMul %float %ny %thick
%zero = OpConstant %float 0.0
%perp_vec4 = OpCompositeConstruct %v4float %perp_x %perp_y %zero %zero

; p0_plus, p0_minus, p1_plus, p1_minus
%p0_plus = OpFAdd %v4float %p0 %perp_vec4
%p0_minus = OpFSub %v4float %p0 %perp_vec4
%p1_plus = OpFAdd %v4float %p1 %perp_vec4
%p1_minus = OpFSub %v4float %p1 %perp_vec4

; Emit 4 vertices for a strip: p0_plus, p0_minus, p1_plus, p1_minus
; Vertex 1
%pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %pos_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

; Vertex 2
OpStore %pos_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

; Vertex 3
OpStore %pos_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

; Vertex 4
OpStore %pos_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex

; End primitive and finish
OpEndPrimitive
OpReturn
OpFunctionEnd

; Execution modes for geometry shader
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Entry point
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color
