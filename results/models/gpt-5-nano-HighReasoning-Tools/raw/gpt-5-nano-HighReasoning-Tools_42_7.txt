; SPIR-V geometry shader (GLSL450 style, Triangles input, triangle strip output)
; Exports 3 vertices per primitive, exploding each triangle outward along its face normal.

OpCapability Geometry
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%uint = OpTypeInt 32 0
%mat4 = OpTypeMatrix %v4float 4

; Constants
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%float_0 = OpConstant %float 0.0
%float_0_08 = OpConstant %float 0.08
%float_one_third = OpConstant %float 0.33333334

; gl_PerVertex_in (input per-vertex block)
%gl_PerVertex_in = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_in Block
%ptr_gl_PerVertex_in = OpTypePointer Input %gl_PerVertex_in
; gl_in is an array of gl_PerVertex_in sized by input primitive (3 for triangles)
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

; Input arrays (per-vertex, same size as primitive)
%arr_worldPos_in = OpTypeArray %v3float %uint_3
%ptr_worldPos_in = OpTypePointer Input %arr_worldPos_in
%worldPos = OpVariable %ptr_worldPos_in Input
%arr_normal_in = OpTypeArray %v3float %uint_3
%ptr_normal_in = OpTypePointer Input %arr_normal_in
%normal = OpVariable %ptr_normal_in Input
%arr_color_in = OpTypeArray %v3float %uint_3
%ptr_color_in = OpTypePointer Input %arr_color_in
%color = OpVariable %ptr_color_in Input

; Output (per-vertex)
%gl_PerVertex_out = OpTypeStruct %v4float
OpDecorate %gl_PerVertex_out Block
%ptr_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_gl_PerVertex_out Output
; Out color (location = 0)
%ptr_out_color = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_color Output
OpDecorate %outColor Location 0
; Pointer to gl_Position inside gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float

; UBO (set=0, binding=0) - minimal layout to satisfy accessibility
%BlockMat = OpTypeStruct %mat4
%ptr_ubo = OpTypePointer Uniform %BlockMat
%ubo = OpVariable %ptr_ubo Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %BlockMat Block

; Entry point and interface
%main = OpFunction %void None %void_fn
%entry = OpLabel
OpName %main "main"

OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %outColor
OpExecutionMode %main InputTriangles
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 3
OpExecutionMode %main Invocations 1

; Function type
%void_fn = OpFunctionType %void

; Load helpers and compute per-face data
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_color_v3_in = OpTypePointer Input %v3float
%color0 = OpVariable %ptr_color_v3_in Input
%color1 = OpVariable %ptr_color_v3_in Input
%color2 = OpVariable %ptr_color_v3_in Input
; Since velocities/colors are accessed via array, we fetch by indices using a common type
%idx0 = %uint_0
%idx1 = %uint_1
%idx2 = %uint_2

; Values from gl_in[i].gl_Position
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %idx0 %idx0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %idx1 %idx0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %idx2 %idx0
%p2 = OpLoad %v4float %p2_ptr

; Colors per vertex
%ci0_ptr = OpAccessChain %ptr_color_v3_in %color %idx0
%ci0 = OpLoad %v3float %ci0_ptr
%ci1_ptr = OpAccessChain %ptr_color_v3_in %color %idx1
%ci1 = OpLoad %v3float %ci1_ptr
%ci2_ptr = OpAccessChain %ptr_color_v3_in %color %idx2
%ci2 = OpLoad %v3float %ci2_ptr

; Extract xyz components for center and normal calculation
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0z = OpCompositeExtract %float %p0 2
%p0_xyz = OpCompositeConstruct %v3float %p0x %p0y %p0z

%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1z = OpCompositeExtract %float %p1 2
%p1_xyz = OpCompositeConstruct %v3float %p1x %p1y %p1z

%p2x = OpCompositeExtract %float %p2 0
%p2y = OpCompositeExtract %float %p2 1
%p2z = OpCompositeExtract %float %p2 2
%p2_xyz = OpCompositeConstruct %v3float %p2x %p2y %p2z

; Center = (p0+p1+p2)/3
%sum01 = OpFAdd %v3float %p0_xyz %p1_xyz
%center_tmp = OpFAdd %v3float %sum01 %p2_xyz
%center = OpFMul %v3float %center_tmp %float_one_third

; E1 = p1 - p0, E2 = p2 - p0
%e1 = OpFSub %v3float %p1_xyz %p0_xyz
%e2 = OpFSub %v3float %p2_xyz %p0_xyz

; N = normalize(cross(e1, e2))
%cross01 = OpExtInst %v3float %GLSLstd450 Cross %v3float %e1 %e2
%Nunnorm = %cross01
%N = OpExtInst %v3float %GLSLstd450 Normalize %Nunnorm

; offset = N * 0.08
%offset_xyz = OpFMul %v3float %N %float_0_08
%ox = OpCompositeExtract %float %offset_xyz 0
%oy = OpCompositeExtract %float %offset_xyz 1
%oz = OpCompositeExtract %float %offset_xyz 2
%offset4 = OpCompositeConstruct %v4float %ox %oy %oz %float_0

; P0_out = p0 + offset4, etc.
%P0_out = OpFAdd %v4float %p0 %offset4
%P1_out = OpFAdd %v4float %p1 %offset4
%P2_out = OpFAdd %v4float %p2 %offset4

; gl_Position pointer (same for all vertices; updated per-vertex)
%gl_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %idx0

; Emit vertex 0
OpStore %outColor %ci0
OpStore %gl_pos_ptr %P0_out
OpEmitVertex

; Emit vertex 1
OpStore %outColor %ci1
OpStore %gl_pos_ptr %P1_out
OpEmitVertex

; Emit vertex 2
OpStore %outColor %ci2
OpStore %gl_pos_ptr %P2_out
OpEmitVertex

; End primitive
OpEndPrimitive

; Finish
OpReturn
OpFunctionEnd

Note:
- This SPIR-V assumes Triangles input (3 vertices per invocation) and outputs a TriangleStrip with 3 vertices, exploding each triangle outward along its face normal.
- The UBO is declared minimally (set=0, binding=0) and not used in calculations; it exists to satisfy the requirement of having an accessible UBO.
- The code uses GLSL.std.450 extended instructions for cross and normalize.
