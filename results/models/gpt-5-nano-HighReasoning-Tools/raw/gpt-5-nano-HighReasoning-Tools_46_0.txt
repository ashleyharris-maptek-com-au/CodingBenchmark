```glsl
#version 450

// Each work item handles one body (index = gl_GlobalInvocationID.x)
layout(local_size_x = 256) in;

// Input: previous timestep data (read-only)
//   For body i, at index i*2:
//     inData[i*2 + 0] = vec4(pos.x, pos.y, pos.z, mass)
//     inData[i*2 + 1] = vec4(vel.x, vel.y, vel.z, 0.0)
layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];
};

// Output: updated timestep data (write-only)
layout(set = 0, binding = 1) writeonly buffer OutputBodies {
    vec4 outData[];
};

// Uniform-like parameters block
//   counts.x = N (number of bodies)
//   physics.x = dt, physics.y = softening, physics.z = G
layout(set = 0, binding = 2) uniform ParamsBlock {
    uvec4 counts;   // counts.x = N
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
} params;

void main() {
    uint N = params.counts.x;
    uint gid = gl_GlobalInvocationID.x;

    // Guard against out-of-bounds invocations
    if (gid >= N) {
        return;
    }

    int base = int(gid) * 2;

    // Read this body's current state from input
    vec3 pos_i = inData[base].xyz;
    float mass_i = inData[base].w;

    vec3 vel_i = inData[base + 1].xyz;

    // Physics constants for this timestep
    float dt  = params.physics.x;
    float soft = params.physics.y;
    float G    = params.physics.z;

    // Accumulator for acceleration on body i due to all others
    vec3 acc = vec3(0.0);

    // Loop over all bodies to accumulate gravitational acceleration
    for (uint j = 0u; j < N; ++j) {
        if (j == gid) {
            continue; // skip self-interaction
        }

        int jbase = int(j) * 2;
        vec3 pos_j = inData[jbase].xyz;
        float mass_j = inData[jbase].w;

        vec3 delta = pos_j - pos_i;

        // dist^2 + soft^2 (softening to avoid singularities)
        float dist2 = dot(delta, delta) + soft * soft;

        // inv(dist) and inv(dist^3)
        float invDist = inversesqrt(dist2);
        float invDist3 = invDist * invDist * invDist;

        // Accumulate acceleration contribution
        acc += G * mass_j * delta * invDist3;
    }

    // Semi-implicit (symplectic) Euler integration:
    // v_{t+dt} = v_t + a * dt
    // x_{t+dt} = x_t + v_{t+dt} * dt
    vec3 vel_new = vel_i + acc * dt;
    vec3 pos_new = pos_i + vel_new * dt;

    // Write updated position+mass and velocity to output
    outData[base]     = vec4(pos_new, mass_i);
    outData[base + 1] = vec4(vel_new, 0.0);
}
```
