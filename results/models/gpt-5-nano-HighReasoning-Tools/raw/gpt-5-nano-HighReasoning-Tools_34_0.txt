#!/usr/bin/env python3
"""
Simple heuristic solver for the Capacitated Vehicle Routing Problem (CVRP).

Input (stdin):
N K CAPACITY
depot_x depot_y
x1 y1 demand1
x2 y2 demand2
...
xN yN demandN

Output (stdout):
total_distance
num_routes
route_1
route_2
...

- Each route is a line listing 0-based customer indices in visit order.
- Up to K routes are used. Each route's total demand does not exceed CAPACITY.
- All customers are visited exactly once if the instance is feasible (sum(demands) <= K*CAPACITY).

This is a constructive heuristic:
1) Build up to K routes greedily using a nearest-feasible-next approach from the depot.
2) If there are unassigned customers after K routes, try to insert them into existing routes
   at the best position (minimizing distance increase) while respecting capacity.

Distances are Euclidean. Distances are printed with 6 decimal places.

Note: This is not guaranteed to be optimal, but it is fast and produces valid solutions for large instances.
"""

import sys
import math

def read_input():
    data = sys.stdin.read().strip().split()
    if not data:
        return None

    it = iter(data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAPACITY = int(next(it))

        depot_x = float(next(it))
        depot_y = float(next(it))

        xs = []
        ys = []
        ds = []
        for _ in range(N):
            x = float(next(it)); y = float(next(it)); d = int(next(it))
            xs.append(x)
            ys.append(y)
            ds.append(d)

        return {
            'N': N,
            'K': K,
            'CAPACITY': CAPACITY,
            'depot': (depot_x, depot_y),
            'xs': xs,
            'ys': ys,
            'ds': ds
        }
    except StopIteration:
        raise ValueError("Input is incomplete or malformed.")

def distance_points(p1, p2):
    return math.hypot(p1[0] - p2[0], p1[1] - p2[1])

def dist_between_nodes(i, j, depot, xs, ys):
    # i, j can be -1 to denote the depot
    c1 = depot if i == -1 else (xs[i], ys[i])
    c2 = depot if j == -1 else (xs[j], ys[j])
    return distance_points(c1, c2)

def route_distance(route_nodes, depot, xs, ys):
    if not route_nodes:
        return 0.0
    total = dist_between_nodes(-1, route_nodes[0], depot, xs, ys)
    for a, b in zip(route_nodes, route_nodes[1:]):
        total += dist_between_nodes(a, b, depot, xs, ys)
    total += dist_between_nodes(route_nodes[-1], -1, depot, xs, ys)
    return total

def solve_cvrp_heuristic(cfg):
    N = cfg['N']
    K = cfg['K']
    CAP = cfg['CAPACITY']
    depot = cfg['depot']
    xs = cfg['xs']
    ys = cfg['ys']
    ds = cfg['ds']

    if N == 0:
        return 0.0, 0, []

    max_demand = max(ds)
    if max_demand > CAP:
        raise ValueError("Unsolvable: a customer's demand exceeds vehicle capacity.")

    # Unvisited customers set
    unvisited = set(range(N))

    routes = []  # each route: {'nodes': [...], 'load': int}

    # Helper: distance from depot to customer i
    def depot_to_i(i):
        return distance_points(depot, (xs[i], ys[i]))

    # Phase 1: Build up to K routes greedily
    while unvisited and len(routes) < K:
        route_nodes = []
        current = -1  # start at depot
        remaining = CAP
        # Choose first customer (nearest to depot among feasible)
        candidates = [i for i in unvisited if ds[i] <= remaining]
        if not candidates:
            break  # cannot start any more route
        next_cand = min(candidates, key=lambda i: depot_to_i(i))
        route_nodes.append(next_cand)
        unvisited.remove(next_cand)
        remaining -= ds[next_cand]
        current = next_cand

        # Extend greedily by nearest feasible neighbor
        while True:
            best_next = None
            best_dist = float('inf')
            for j in unvisited:
                if ds[j] <= remaining:
                    d = distance_points((xs[current], ys[current]), (xs[j], ys[j]))
                    if d < best_dist:
                        best_dist = d
                        best_next = j
            if best_next is None:
                break
            route_nodes.append(best_next)
            unvisited.remove(best_next)
            remaining -= ds[best_next]
            current = best_next

        # Record route
        route_load = sum(ds[i] for i in route_nodes)
        routes.append({'nodes': route_nodes, 'load': route_load})

    # Phase 2: If customers remain, try to insert into existing routes
    # (Keep at most K routes; do not create new ones in this phase)
    if unvisited:
        # For deterministic behavior, process in sorted order
        for c in sorted(unvisited):
            best_route_idx = None
            best_pos = None
            best_delta = float('inf')
            for r_idx, r in enumerate(routes):
                if r['load'] + ds[c] > CAP:
                    continue
                nodes = r['nodes']
                if not nodes:
                    # Inserting into empty route
                    delta = distance_points(depot, (xs[c], ys[c])) + distance_points((xs[c], ys[c]), depot)
                    if delta < best_delta:
                        best_delta = delta
                        best_route_idx = r_idx
                        best_pos = 0
                else:
                    # Try all insertion positions
                    for pos in range(0, len(nodes) + 1):
                        if pos == 0:
                            prev = depot
                            next_node = nodes[0]
                            delta = distance_points(depot, (xs[c], ys[c])) + distance_points((xs[c], ys[c]), (xs[next_node], ys[next_node])) - distance_points(depot, (xs[next_node], ys[next_node]))
                        elif pos == len(nodes):
                            prev_node = nodes[-1]
                            delta = distance_points((xs[prev_node], ys[prev_node]), (xs[c], ys[c])) + distance_points((xs[c], ys[c]), depot) - distance_points((xs[prev_node], ys[prev_node]), depot)
                        else:
                            prev_node = nodes[pos-1]
                            next_node = nodes[pos]
                            delta = (distance_points((xs[prev_node], ys[prev_node]), (xs[c], ys[c]))
                                     + distance_points((xs[c], ys[c]), (xs[next_node], ys[next_node]))
                                     - distance_points((xs[prev_node], ys[prev_node]), (xs[next_node], ys[next_node])))
                        if delta < best_delta:
                            best_delta = delta
                            best_route_idx = r_idx
                            best_pos = pos
            if best_route_idx is not None:
                # Perform insertion
                routes[best_route_idx]['nodes'].insert(best_pos, c)
                routes[best_route_idx]['load'] += ds[c]
                unvisited.remove(c)
            else:
                # No feasible insertion into existing routes (shouldn't happen if sum(d) <= K*CAP)
                pass

    # Optional: Quick sanity check. If still unvisited and total capacity insufficient, raise.
    if unvisited:
        total_cap = K * CAP
        total_demand = sum(ds[i] for i in range(N))
        if total_demand > total_cap:
            raise ValueError("No feasible solution: total demand exceeds total vehicle capacity (K*CAPACITY).")

        # If still unvisited (rare), try a simple fallback: append remaining customers to the first route if possible.
        # This is a last-resort to ensure all customers are visited. Not optimal, but ensures feasibility.
        for c in sorted(unvisited):
            if routes:
                if routes[0]['load'] + ds[c] <= CAP:
                    routes[0]['nodes'].append(c)
                    routes[0]['load'] += ds[c]
                    unvisited.remove(c)
            if not unvisited:
                break

    # Compute total distance
    total_distance = sum(route_distance(r['nodes'], depot, xs, ys) for r in routes)

    # Build output
    num_routes = len(routes)
    output_lines = []
    output_lines.append("{:.6f}".format(total_distance))
    output_lines.append(str(num_routes))
    for r in routes:
        if r['nodes']:
            output_lines.append(" ".join(str(i) for i in r['nodes']))
        else:
            output_lines.append("")  # empty route (edge case)

    return "\n".join(output_lines)

def main():
    cfg = read_input()
    if cfg is None:
        return
    try:
        result = solve_cvrp_heuristic(cfg)
        print(result)
    except ValueError as e:
        # Print error to stdout as per the problem's input/output contract.
        print(str(e))

if __name__ == "__main__":
    main()
