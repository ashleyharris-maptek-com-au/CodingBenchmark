```csharp
using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    // Global graph
    static int n;
    static List<int>[] adj;
    static List<int>[] radj;

    // Best solution found so far
    static List<int> bestSet = new List<int>();
    static int bestK = int.MaxValue;

    // Current path in DFS
    static List<int> curr = new List<int>();

    static void Main()
    {
        var fs = new FastScanner();
        if (!fs.HasNext()) return;

        n = fs.NextInt();
        int m = fs.NextInt();

        adj = new List<int>[n];
        radj = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            adj[i] = new List<int>();
            radj[i] = new List<int>();
        }

        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            adj[u].Add(v);
            radj[v].Add(u);
        }

        // Seed a reasonable upper bound with a greedy FVS
        List<int> greedy = GreedyFVS();
        bestK = greedy.Count;
        bestSet = new List<int>(greedy);

        // Start exact search (branching on cycles)
        bool[] removed = new bool[n];
        DFS(0, removed);

        // Output
        Console.WriteLine(bestK);
        if (bestK > 0)
            Console.WriteLine(string.Join(" ", bestSet));
        else
            Console.WriteLine();
    }

    // Greedy heuristic to get an initial upper bound
    static List<int> GreedyFVS()
    {
        bool[] removed = new bool[n];
        List<int> res = new List<int>();

        while (true)
        {
            List<int> cycle;
            if (!FindCycle(removed, out cycle))
                break; // already a DAG

            // Pick a vertex in the cycle with maximum total degree within the remaining graph
            int bestV = cycle[0];
            int bestDeg = -1;
            foreach (int v in cycle)
            {
                int deg = 0;
                foreach (int w in adj[v]) if (!removed[w]) deg++;
                foreach (int u in radj[v]) if (!removed[u]) deg++;
                if (deg > bestDeg)
                {
                    bestDeg = deg;
                    bestV = v;
                }
            }

            removed[bestV] = true;
            res.Add(bestV);
        }

        return res;
    }

    // Find a directed cycle in the subgraph induced by not-removed vertices
    // Returns true if a cycle is found and outputs one cycle (list of vertices)
    // If no cycle exists, returns false with cycle == null
    static bool FindCycle(bool[] removed, out List<int> cycle)
    {
        cycle = null;

        int[] color = new int[n];   // 0 = unvisited, 1 = in stack, 2 = done
        int[] parent = new int[n];
        for (int i = 0; i < n; i++) parent[i] = -1;

        bool found = false;

        void Dfs(int v)
        {
            color[v] = 1;
            foreach (int w in adj[v])
            {
                if (removed[w]) continue;
                if (color[w] == 0)
                {
                    parent[w] = v;
                    Dfs(w);
                    if (found) return;
                }
                else if (color[w] == 1)
                {
                    // Found a back-edge v -> w, reconstruct cycle
                    if (w == v)
                    {
                        // Self-loop
                        cycle = new List<int> { v };
                    }
                    else
                    {
                        List<int> cyc = new List<int>();
                        cyc.Add(w);
                        int x = v;
                        while (x != w)
                        {
                            cyc.Add(x);
                            x = parent[x];
                        }
                        cycle = cyc;
                    }
                    found = true;
                    return;
                }
            }
            color[v] = 2;
        }

        for (int i = 0; i < n && !found; i++)
        {
            if (!removed[i] && color[i] == 0)
                Dfs(i);
        }

        return found;
    }

    // Exact DFS search for minimum DFVS
    static void DFS(int depth, bool[] removed)
    {
        if (depth >= bestK) return;

        List<int> cycle;
        if (!FindCycle(removed, out cycle))
        {
            // DAG achieved
            if (depth < bestK)
            {
                bestK = depth;
                bestSet = new List<int>(curr);
            }
            return;
        }

        // Branch on vertices of the detected cycle
        // Heuristic: try vertices with higher local degree first
        var order = new List<(int v, int score)>();
        foreach (int v in cycle)
        {
            int score = 0;
            foreach (int w in adj[v]) if (!removed[w]) score++;
            foreach (int u in radj[v]) if (!removed[u]) score++;
            order.Add((v, score));
        }
        order.Sort((a, b) => b.score.CompareTo(a.score));

        foreach (var t in order)
        {
            int v = t.v;
            removed[v] = true;
            curr.Add(v);
            DFS(depth + 1, removed);
            curr.RemoveAt(curr.Count - 1);
            removed[v] = false;
        }
    }

    // Simple fast scanner for stdin
    class FastScanner
    {
        private Stream stream;
        private byte[] buffer = new byte[1 << 16];
        private int len = 0;
        private int ptr = 0;

        public FastScanner()
        {
            stream = Console.OpenStandardInput();
        }

        private int Read()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr++];
        }

        public bool HasNext()
        {
            int c;
            while ((c = Peek()) != -1 && c <= ' ') Consume();
            return c != -1;
        }

        private int Peek()
        {
            if (ptr >= len)
            {
                len = stream.Read(buffer, 0, buffer.Length);
                ptr = 0;
                if (len <= 0) return -1;
            }
            return buffer[ptr];
        }

        private void Consume()
        {
            int c = Peek();
            if (c == -1) return;
            ptr++;
        }

        public int NextInt()
        {
            int c;
            while ((c = Read()) != -1 && c <= ' ') { }
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = Read();
            }
            int val = 0;
            while (c > ' ')
            {
                val = val * 10 + (c - '0');
                c = Read();
            }
            return val * sign;
        }
    }
}
```
