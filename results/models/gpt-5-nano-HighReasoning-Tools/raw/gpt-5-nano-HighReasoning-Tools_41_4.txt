; SPIR-V
; Version: 1.0
; Generator: manual
; Bound: 999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%func_type = OpTypeFunction %void
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; Uniform block
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %v4float

; Inputs/Outputs pointers
%_ptr_Input_vec3 = OpTypePointer Input %v3float
%_ptr_Input_vec2 = OpTypePointer Input %v2float
%_ptr_Output_vec4 = OpTypePointer Output %v4float

; Global vars
worldPos = OpVariable %_ptr_Input_vec3 Input
normal = OpVariable %_ptr_Input_vec3 Input
uv = OpVariable %_ptr_Input_vec2 Input
tangent = OpVariable %_ptr_Input_vec3 Input
color = OpVariable %_ptr_Input_vec3 Input
outColor = OpVariable %_ptr_Output_vec4 Output
ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Decorations
OpDecorate worldPos Location 0
OpDecorate normal Location 1
OpDecorate uv Location 2
OpDecorate tangent Location 3
OpDecorate color Location 4
OpDecorate outColor Location 0
OpDecorate ubo DescriptorSet 0
OpDecorate ubo Binding 0
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; GLSL std 450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Function type
%main = OpFunction %void None %func_type
%entry = OpLabel

; Constants and indices
%const_0 = OpConstant %int 0
%const_1 = OpConstant %int 1
%const_2 = OpConstant %int 2
%const_3 = OpConstant %int 3
%const_4 = OpConstant %int 4
%const_5 = OpConstant %int 5

; Float constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0

; Access UBO members
%model_ptr = OpAccessChain %_ptr_Uniform_mat4 %ubo %const_0
%view_ptr  = OpAccessChain %_ptr_Uniform_mat4 %ubo %const_1
%proj_ptr  = OpAccessChain %_ptr_Uniform_mat4 %ubo %const_2
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %const_3
%cameraPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %const_4
%params_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %const_5

; Load uniforms
%model = OpLoad %mat4 %model_ptr
%view = OpLoad %mat4 %view_ptr
%projection = OpLoad %mat4 %proj_ptr
%lightPos4 = OpLoad %v4float %lightPos_ptr
%cameraPos4 = OpLoad %v4float %cameraPos_ptr
%params = OpLoad %v4float %params_ptr

; Load inputs
worldPos_in = OpLoad %v3float %worldPos
normal_in = OpLoad %v3float %normal

; Lambertian diffuse
; N = normalize(normal)
%dotNN = OpVectorDot %float %normal_in %normal_in
%sqrtNN = OpExtInst %float %GLSLstd450 Sqrt %dotNN
%invNN = OpFDiv %float %float_1 % sqrtNN
%N_norm = OpVectorTimesScalar %v3float %normal_in %invNN

; L = normalize(lightPos.xyz - worldPos)
%Lx = OpCompositeExtract %float %lightPos4 0
%Ly = OpCompositeExtract %float %lightPos4 1
%Lz = OpCompositeExtract %float %lightPos4 2
%lightPos_xyz = OpCompositeConstruct %v3float %Lx %Ly %Lz
%L = OpVectorSub %v3float %lightPos_xyz worldPos_in
%dotL = OpVectorDot %float %L %L
%sqrtL = OpExtInst %float %GLSLstd450 Sqrt %dotL
%invL = OpFDiv %float %float_1 % sqrtL
%L_norm = OpVectorTimesScalar %v3float %L %invL

; d = max(dot(N_norm, L_norm), 0.0)
%dNL = OpVectorDot %float %N_norm %L_norm
%diff = OpExtInst %float %GLSLstd450 FMax %dNL %float_0

; Output: grayscale with alpha 1
%dvec3 = OpCompositeConstruct %v3float %diff %diff %diff
%out4 = OpCompositeConstruct %v4float %dvec3 %float_1
OpStore outColor %out4

; Return
OpReturn
OpFunctionEnd
