 OpCapability Geometry
 OpMemoryModel Logical GLSL450

; Types
 %void = OpTypeVoid
 %float = OpTypeFloat 32
 %v3float = OpTypeVector %float 3
 %v4float = OpTypeVector %float 4
 %uint = OpTypeInt 32 0
 %int = OpTypeInt 32 1

; Constants
 %u32_0 = OpConstant %uint 0
 %u32_1 = OpConstant %uint 1
 %u32_2 = OpConstant %uint 2
 %u32_3 = OpConstant %uint 3

; gl_PerVertex_in block (input per-vertex from vertex shader)
 %gl_PerVertex_in = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_in Block
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %u32_3
 %ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
 %gl_in = OpVariable %ptr_in_arr_pv Input

; Input arrays from vertex shader
 %worldPos_type  = OpTypeArray %v3float %u32_3
 %normal_type    = OpTypeArray %v3float %u32_3
 %color_type     = OpTypeArray %v3float %u32_3

 %ptr_worldPos = OpTypePointer Input %worldPos_type
 %worldPos     = OpVariable %ptr_worldPos Input

 %ptr_normal = OpTypePointer Input %normal_type
 %normal       = OpVariable %ptr_normal Input

 %ptr_color = OpTypePointer Input %color_type
 %color        = OpVariable %ptr_color Input

; gl_PerVertex_out block (output per-vertex position)
 %gl_PerVertex_out = OpTypeStruct %v4float
 OpDecorate %gl_PerVertex_out Block
 %ptr_gl_out = OpTypePointer Output %gl_PerVertex_out
 %gl_out = OpVariable %ptr_gl_out Output

; Output per-vertex color
 %ptr_out_color = OpTypePointer Output %v3float
 %outColor = OpVariable %ptr_out_color Output
 OpDecorate %outColor Location 0

; Also decorate inputs with locations (arrays per-vertex)
 OpDecorate %worldPos Location 0
 OpDecorate %normal Location 1
 OpDecorate %color Location 2

; Uniform Block (UBO) - set=0, binding=0 (accessible to geometry shader)
 %f32 = OpTypeFloat 32
 %UBO_Block = OpTypeStruct %f32 %f32 %f32  ; minimal 3-float block to satisfy binding
 OpDecorate %UBO_Block Block
 %ubo_ptr = OpTypePointer Uniform %UBO_Block
 %ubo = OpVariable %ubo_ptr Uniform
 OpDecorate %ubo DescriptorSet 0
 OpDecorate %ubo Binding 0

; Pointer types for convenience
 %ptr_v4_in  = OpTypePointer Input %v4float
 %ptr_v3_in  = OpTypePointer Input %v3float
 %ptr_v4_out = OpTypePointer Output %v4float
 %ptr_v3_out = OpTypePointer Output %v3float

 ; Function declaration
 %void_f = %void
 %func_type = OpTypeFunction %void_f

 ; Main function
 %main = OpFunction %void_f None %func_type
 %entry = OpLabel

 ; Vertex 0:
 ; gl_in[0].gl_Position
 %gl_in_pos0_ptr = OpAccessChain %ptr_v4_in %gl_in %u32_0 %u32_0
 %gl_in_pos0     = OpLoad %v4float %gl_in_pos0_ptr

 ; world/color for vertex 0
 %color0_ptr = OpAccessChain %ptr_worldPos %color %u32_0
 %color0     = OpLoad %v3float %color0_ptr

 ; Emit vertex 0
 %gl_out_pos_ptr0 = OpAccessChain %ptr_v4_out %gl_out %u32_0
 OpStore %gl_out_pos_ptr0 %gl_in_pos0
 OpStore %outColor %color0
 OpEmitVertex

 ; Vertex 1:
 %gl_in_pos1_ptr = OpAccessChain %ptr_v4_in %gl_in %u32_1 %u32_0
 %gl_in_pos1     = OpLoad %v4float %gl_in_pos1_ptr

 %color1_ptr = OpAccessChain %ptr_worldPos %color %u32_1
 %color1     = OpLoad %v3float %color1_ptr

 %gl_out_pos_ptr1 = OpAccessChain %ptr_v4_out %gl_out %u32_0
 OpStore %gl_out_pos_ptr1 %gl_in_pos1
 OpStore %outColor %color1
 OpEmitVertex

 ; Vertex 2:
 %gl_in_pos2_ptr = OpAccessChain %ptr_v4_in %gl_in %u32_2 %u32_0
 %gl_in_pos2     = OpLoad %v4float %gl_in_pos2_ptr

 %color2_ptr = OpAccessChain %ptr_worldPos %color %u32_2
 %color2     = OpLoad %v3float %color2_ptr

 %gl_out_pos_ptr2 = OpAccessChain %ptr_v4_out %gl_out %u32_0
 OpStore %gl_out_pos_ptr2 %gl_in_pos2
 OpStore %outColor %color2
 OpEmitVertex

 ; End primitive
 OpEndPrimitive

 ; End function
 OpReturn
 OpFunctionEnd

; Entry point declaration and execution modes
 OpEntryPoint Geometry %main "main" %worldPos %normal %color %outColor
 OpExecutionMode %main InputTriangles
 OpExecutionMode %main OutputTriangleStrip
 OpExecutionMode %main OutputVertices 3
 OpExecutionMode %main Invocations 1
