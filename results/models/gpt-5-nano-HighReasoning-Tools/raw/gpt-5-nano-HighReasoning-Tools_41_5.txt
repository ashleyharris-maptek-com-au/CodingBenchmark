; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Types
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4

; Pointers and UBO
%uint = OpTypeInt 32 0
%UBOtype = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%ptr_UBO = OpTypePointer Uniform %UBOtype
%ptr_Vec4_Uniform = OpTypePointer Uniform %vec4
%ptr_Vec3_Input = OpTypePointer Input %vec3
%ptr_Vec2_Input = OpTypePointer Input %vec2
%ptr_Vec4_Input = OpTypePointer Input %vec4
%ptr_Vec4_Output = OpTypePointer Output %vec4

; Variables (inputs, outputs, uniform)
%inWorldPos = OpVariable %ptr_Vec3_Input Input
%inNormal   = OpVariable %ptr_Vec3_Input Input
%inUV       = OpVariable %ptr_Vec2_Input Input
%inTangent  = OpVariable %ptr_Vec3_Input Input
%inColor    = OpVariable %ptr_Vec3_Input Input
%outColor   = OpVariable %ptr_Vec4_Output Output
%ubo        = OpVariable %ptr_UBO Uniform

; Decorations
OpDecorate %inWorldPos Location 0
OpDecorate %inNormal Location 1
OpDecorate %inUV Location 2
OpDecorate %inTangent Location 3
OpDecorate %inColor Location 4
OpDecorate %outColor Location 0
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBOtype Block
OpMemberDecorate %UBOtype 0 Offset 0
OpMemberDecorate %UBOtype 1 Offset 64
OpMemberDecorate %UBOtype 2 Offset 128
OpMemberDecorate %UBOtype 3 Offset 192
OpMemberDecorate %UBOtype 4 Offset 208
OpMemberDecorate %UBOtype 5 Offset 224

; Constants
%zero = OpConstant %float 0.0
%oneC = OpConstant %float 1.0
%halfC = OpConstant %float 0.5
%eightC = OpConstant %float 0.8
%twoC = OpConstant %float 0.2
%powExp = OpConstant %float 32.0
%powTwo = OpConstant %float 32.0

; Entry function and main
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%main = OpFunction %void None %funcType
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %vec3 %inWorldPos
%normIn = OpLoad %vec3 %inNormal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %normIn

; Read UBO fields (lightPos, cameraPos)
%idx3 = OpConstant %uint 3
%idx4 = OpConstant %uint 4
%lightPosPtr = OpAccessChain %ptr_Vec4_Uniform %ubo %idx3
%lightPos4 = OpLoad %vec4 %lightPosPtr
%lpX = OpCompositeExtract %float %lightPos4 0
%lpY = OpCompositeExtract %float %lightPos4 1
%lpZ = OpCompositeExtract %float %lightPos4 2
%lightPos = OpCompositeConstruct %vec3 %lpX %lpY %lpZ

%camPosPtr = OpAccessChain %ptr_Vec4_Uniform %ubo %idx4
%camPos4 = OpLoad %vec4 %camPosPtr
%cpX = OpCompositeExtract %float %camPos4 0
%cpY = OpCompositeExtract %float %camPos4 1
%cpZ = OpCompositeExtract %float %camPos4 2
%camPos = OpCompositeConstruct %vec3 %cpX %cpY %cpZ

; Compute L and V
%L = OpFSub %vec3 %lightPos %worldPos
%Ln = OpExtInst %vec3 %GLSLstd450 Normalize %L

%Vt = OpFSub %vec3 %camPos %worldPos
%V = OpExtInst %vec3 %GLSLstd450 Normalize %Vt

; Reflection R = reflect(-L, N)
%negL = OpFNegate %vec3 %Ln
%R = OpExtInst %vec3 %GLSLstd450 Reflect %negL %N

; Phong components
%dNL = OpExtInst %float %GLSLstd450 Dot %N %Ln
%diff = OpExtInst %float %GLSLstd450 FMax %dNL %zero

%dotRV = OpExtInst %float %GLSLstd450 Dot %R %V
%specBase = OpExtInst %float %GLSLstd450 FMax %dotRV %zero
%spec = OpExtInst %float %GLSLstd450 Pow %specBase %powExp

; Intensity = 0.2 + 0.5*diff + 0.8*spec
%halfDiff = OpExtInst %float %GLSLstd450 FMul %diff %halfC
%scaledSpec = OpExtInst %float %GLSLstd450 FMul %spec %eightC
%sum = OpExtInst %float %GLSLstd450 FAdd %halfDiff %scaledSpec
%intensity = OpExtInst %float %GLSLstd450 FAdd %sum %twoC

; Output color (grayscale), alpha = 1.0
$outCol = OpCompositeConstruct %vec4 %intensity %intensity %intensity %oneC
OpStore %outColor %outCol

OpReturn
OpFunctionEnd

; Entry point and mode
OpEntryPoint Fragment %main "main" %inWorldPos %inNormal %inUV %inTangent %inColor %outColor
OpExecutionMode %main OriginUpperLeft
