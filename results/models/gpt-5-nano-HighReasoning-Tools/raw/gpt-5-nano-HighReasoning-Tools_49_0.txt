GLSL compute shader (compatible with SPIR-V 1.0/GLSL450)
- Binding 0 (set=0): Storage buffer (read-only) - uint array data[]
- Binding 1 (set=0): Storage buffer (read-write) - uint array data[]
- Binding 2 (set=0): Uniform buffer - uint4 config (config.x = N)
- Local size: 256 x 1 x 1

GLSL:
-------
#version 450

layout(local_size_x = 256, local_size_y = 1, local_size_z = 1) in;

layout(set = 0, binding = 0) readonly buffer InBuffer {
    uint data[];
} In;

layout(set = 0, binding = 1) writeonly buffer OutBuffer {
    uint data[];
} Out;

layout(set = 0, binding = 2) uniform Config {
    uvec4 config; // config.x = N
} U;

void main() {
    uint gid = gl_GlobalInvocationID.x;
    uint N = U.config.x;
    if (gid >= N) {
        return;
    }
    uint sum = 0u;
    for (uint i = 0u; i <= gid; ++i) {
        sum += In.data[i];
    }
    Out.data[gid] = sum;
}
