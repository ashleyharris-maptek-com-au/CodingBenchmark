// WGSL compute shader: ChaCha-like hash-based PoW (as specified)
// - baseData: input 16 u32 words
// - result: [ {found flag}, nonce, h0, h1 ]
// - params: vec4<u32> with (x=difficulty, y=range_start, z=range_size)
// - Each thread tests nonce = range_start + gid.x, for gid.x < range_size

@group(0) @binding(0) var<storage, read> baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result  : array<u32, 4>;
@group(0) @binding(2) var<uniform>     params  : vec4<u32>;

// Rotate left helper for 32-bit values
fn rotl32(x: u32, n: u32) -> u32 {
  return (x << n) | (x >> (32u - n));
}

// 16-word input -> 64-bit hash output (h0 in low 32 bits, h1 in high 32 bits)
fn custom_hash(data: array<u32, 16>, nonce: u32) -> vec2<u32> {
  var state: array<u32, 16>;

  // 1) Initialize state
  var i: u32;
  i = 0u;
  while (i < 16u) {
    state[i] = data[i];
    i = i + 1u;
  }

  // 2) XOR with nonce+i
  i = 0u;
  while (i < 16u) {
    state[i] = state[i] ^ (nonce + i);
    i = i + 1u;
  }

  // 3) 8 rounds of 4-way ChaCha-like mixing (unrolled per group)
  for (var round: u32 = 0u; round < 8u; round = round + 1u) {
    // Group 0: indices [0..3]
    var a = state[0u];
    var b = state[1u];
    var c = state[2u];
    var d = state[3u];

    a = a + b;       d = d ^ a; d = rotl32(d, 16u);
    c = c + d;       b = b ^ c; b = rotl32(b, 12u);
    a = a + b;       d = d ^ a; d = rotl32(d, 8u);
    c = c + d;       b = b ^ c; b = rotl32(b, 7u);

    state[0u] = a; state[1u] = b; state[2u] = c; state[3u] = d;

    // Group 1: indices [4..7]
    a = state[4u];
    b = state[5u];
    c = state[6u];
    d = state[7u];

    a = a + b;       d = d ^ a; d = rotl32(d, 16u);
    c = c + d;       b = b ^ c; b = rotl32(b, 12u);
    a = a + b;       d = d ^ a; d = rotl32(d, 8u);
    c = c + d;       b = b ^ c; b = rotl32(b, 7u);

    state[4u] = a; state[5u] = b; state[6u] = c; state[7u] = d;

    // Group 2: indices [8..11]
    a = state[8u];
    b = state[9u];
    c = state[10u];
    d = state[11u];

    a = a + b;       d = d ^ a; d = rotl32(d, 16u);
    c = c + d;       b = b ^ c; b = rotl32(b, 12u);
    a = a + b;       d = d ^ a; d = rotl32(d, 8u);
    c = c + d;       b = b ^ c; b = rotl32(b, 7u);

    state[8u] = a; state[9u] = b; state[10u] = c; state[11u] = d;

    // Group 3: indices [12..15]
    a = state[12u];
    b = state[13u];
    c = state[14u];
    d = state[15u];

    a = a + b;       d = d ^ a; d = rotl32(d, 16u);
    c = c + d;       b = b ^ c; b = rotl32(b, 12u);
    a = a + b;       d = d ^ a; d = rotl32(d, 8u);
    c = c + d;       b = b ^ c; b = rotl32(b, 7u);

    state[12u] = a; state[13u] = b; state[14u] = c; state[15u] = d;
  }

  // 4) Fold to 2 words
  let h0 = state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^
           state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
  let h1 = state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^
           state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

  return vec2<u32>(h0, h1);
}

// Count trailing zeros for 32-bit value
fn ctz32(x: u32) -> u32 {
  var v = x;
  var cnt = 0u;
  if (v == 0u) { return 32u; }
  while ((v & 1u) == 0u) {
    cnt = cnt + 1u;
    v = v >> 1u;
  }
  return cnt;
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
  let difficulty: u32 = params.x;
  let range_start: u32 = params.y;
  let range_size: u32 = params.z;

  let idx: u32 = gid.x;
  if (idx >= range_size) { return; }

  let nonce: u32 = range_start + idx;

  // Compute hash
  let hv: vec2<u32> = custom_hash(baseData, nonce);
  let h0 = hv.x;
  let h1 = hv.y;

  // Trailing zero count on 64-bit value (h0 | (h1 << 32))
  var zeros: u32;
  if (h0 != 0u) {
    zeros = ctz32(h0);
  } else if (h1 != 0u) {
    zeros = 32u + ctz32(h1);
  } else {
    zeros = 64u;
  }

  // If meets difficulty, write result (race writes allowed)
  if (zeros >= difficulty) {
    result[0] = 1u;
    result[1] = nonce;
    result[2] = h0;
    result[3] = h1;
  }
}
