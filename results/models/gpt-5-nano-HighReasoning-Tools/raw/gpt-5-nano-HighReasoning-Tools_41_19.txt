; SPIR-V
; Version: 1.0
; Generator: hand-written
OpCapability Shader
OpMemoryModel Logical GLSL450
%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4
; UBO block type
%UBO_type = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO_type
%_ptr_Mat4_Uniform = OpTypePointer Uniform %mat4
%_ptr_Vec4_Uniform = OpTypePointer Uniform %vec4
; Pointer/IO types
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Output_vec4 = OpTypePointer Output %vec4
; Ext import for GLSL std450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"
; Uniform block variable
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
; Inputs
%in_worldPos = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_worldPos Location 0
%in_normal = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_normal Location 1
%in_uv = OpVariable %_ptr_Input_vec2 Input
OpDecorate %in_uv Location 2
%in_tangent = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_tangent Location 3
%in_color = OpVariable %_ptr_Input_vec3 Input
OpDecorate %in_color Location 4
; Output
%outColor = OpVariable %_ptr_Output_vec4 Output
OpDecorate %outColor Location 0
; Constants
%zero_f = OpConstant %float 0.0
%one_f = OpConstant %float 1.0
%two_f = OpConstant %float 2.0
%shininess = OpConstant %float 32.0
; ambient color (0.1, 0.05, 0.05)
%ambient_r = OpConstant %float 0.1
%ambient_g = OpConstant %float 0.05
%ambient_b = OpConstant %float 0.05
%ambient_vec = OpConstantComposite %vec3 %ambient_r %ambient_g %ambient_b
%const_1 = OpConstant %float 1.0
; vec3/vec4 constants for clamp
%zero_vec3 = OpConstantComposite %vec3 %zero_f %zero_f %zero_f
%one_vec3  = OpConstantComposite %vec3 %const_1 %const_1 %const_1
; Entry point and function types
%void = OpTypeVoid
%funcType = OpTypeFunction %void
%main = OpFunction %void None %funcType
%entry = OpLabel
; Load inputs and uniforms
%worldPos = OpLoad %vec3 %in_worldPos
%normal = OpLoad %vec3 %in_normal
%N = OpExtInst %vec3 %GLSLstd450 Normalize %normal
%color = OpLoad %vec3 %in_color
; Light position (world space) from UBO (vec4), convert to vec3
%lightPos_ptr = OpAccessChain %_ptr_Vec4_Uniform %ubo 3
%lightPos4 = OpLoad %vec4 %lightPos_ptr
%lpx = OpCompositeExtract %float %lightPos4 0
%lpy = OpCompositeExtract %float %lightPos4 1
%lpz = OpCompositeExtract %float %lightPos4 2
%lightPos3 = OpCompositeConstruct %vec3 %lpx %lpy %lpz
; L = normalize(lightPos - worldPos)
%toLight = OpFSub %vec3 %lightPos3 % worldPos
%L = OpExtInst %vec3 %GLSLstd450 Normalize %toLight
; Diffuse term
%dotNL = OpDot %float %N %L
%NL = OpFMax %float %dotNL % zero_f
%diffuse = OpVectorTimesScalar %vec3 %color % NL
; Specular term
%negL = OpFNeg %vec3 %L
%scaled = OpVectorTimesScalar %vec3 %N % dotNL
%scaled2 = OpVectorTimesScalar %vec3 %scaled % two_f
%R = OpFAdd %vec3 %negL % scaled2
; View direction
%camPos_ptr = OpAccessChain %_ptr_Vec4_Uniform %ubo 4
%camPos4 = OpLoad %vec4 %camPos_ptr
%cx = OpCompositeExtract %float %camPos4 0
%cy = OpCompositeExtract %float %camPos4 1
%cz = OpCompositeExtract %float %camPos4 2
%camPos3 = OpCompositeConstruct %vec3 %cx %cy %cz
%toCamera = OpFSub %vec3 % camPos3 % worldPos
%V = OpExtInst %vec3 %GLSLstd450 Normalize %toCamera
%dotRV = OpDot %float %R %V
%dotRV_clamped = OpFMax %float %dotRV % zero_f
%spec_factor = OpExtInst %float %GLSLstd450 Pow %dotRV_clamped % shininess
%white_vec3 = OpConstantComposite %vec3 %const_1 %const_1 %const_1
%specular = OpVectorTimesScalar %vec3 %white_vec3 % spec_factor
; Ambient + Diffuse + Specular
%ambient_plus_diffuse = OpFAdd %vec3 % ambient_vec % diffuse
%color_sum = OpFAdd %vec3 % ambient_plus_diffuse % specular
; Clamp to [0,1]
%clamp_tmp = OpFMin %vec3 % color_sum % one_vec3
%clamp_final = OpFMax %vec3 % clamp_tmp % zero_vec3
; Build final vec4 with alpha = 1.0
%fr = OpCompositeExtract %float % clamp_final 0
%fg = OpCompositeExtract %float % clamp_final 1
%fb = OpCompositeExtract %float % clamp_final 2
%outColor_vec4 = OpCompositeConstruct %vec4 %fr %fg %fb % const_1
OpStore %outColor %outColor_vec4
OpReturn
OpFunctionEnd
; Entry point decorated
OpEntryPoint Fragment %main "main" %in_worldPos %in_normal %in_uv %in_tangent %in_color %outColor
