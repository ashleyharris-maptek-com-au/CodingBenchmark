Here is a complete, compilable C++17 solution that finds an exact Maximum Independent Set using a meet-in-the-middle approach. It works well for graphs with up to about 50 vertices and outputs one maximum independent set (k and the vertex indices).

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <climits>

#if defined(_MSC_VER)
#include <intrin.h>
#pragma intrinsic(_BitScanForward)
#endif

// Portable function: index (0-based) of least significant set bit
static inline int lsb_index(unsigned int x) {
#if defined(_MSC_VER)
    unsigned long idx;
    _BitScanForward(&idx, x);
    return (int)idx;
#else
    return __builtin_ctz(x);
#endif
}

// Portable population count for 32-bit integers
static inline int popcount32(unsigned int x) {
#if defined(_MSC_VER)
    return (int)__popcnt(x);
#else
    return __builtin_popcount(x);
#endif
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, m;
    if (!(std::cin >> n >> m)) return 0;

    // Adjacency matrix for original graph
    std::vector<std::vector<char>> adj(n, std::vector<char>(n, 0));

    for (int i = 0; i < m; ++i) {
        int u, v;
        std::cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n || u == v) continue;
        adj[u][v] = adj[v][u] = 1;
    }

    // Meet-in-the-middle split
    int n1 = n / 2;
    int n2 = n - n1;

    // First half adjacency (n1 x n1)
    std::vector<unsigned int> adj1(n1, 0);
    // Cross edges from first half to second half (n1 elements, each with n2-bit mask)
    std::vector<unsigned int> crossMask(n1, 0);
    for (int i = 0; i < n1; ++i) {
        unsigned int mask = 0;
        for (int j = 0; j < n1; ++j) if (adj[i][j]) mask |= (1u << j);
        adj1[i] = mask;

        unsigned int cm = 0;
        for (int j = 0; j < n2; ++j) if (adj[i][n1 + j]) cm |= (1u << j);
        crossMask[i] = cm;
    }

    // Second half adjacency (n2 x n2)
    std::vector<unsigned int> adj2(n2, 0);
    for (int j = 0; j < n2; ++j) {
        unsigned int mask = 0;
        for (int k = 0; k < n2; ++k) if (adj[n1 + j][n1 + k]) mask |= (1u << k);
        adj2[j] = mask;
    }

    // DP on second half: enumerate all masks, determine independence and best submask
    unsigned int maxMask2 = (n2 >= 32) ? 0u : (1u << n2);
    // If n2 == 0, handle gracefully
    std::vector<char> indep2(maxMask2, 0);
    std::vector<int8_t> dp2(maxMask2, (int8_t) -100); // -100 acts as -INF
    std::vector<uint32_t> choice(maxMask2, 0);

    indep2[0] = 1;
    dp2[0] = 0;
    choice[0] = 0;

    for (unsigned int mask = 1; mask < maxMask2; ++mask) {
        unsigned int bit = mask & -mask;
        int i = lsb_index(bit);
        unsigned int rest = mask ^ bit;
        if (indep2[rest] && ((adj2[i] & rest) == 0)) {
            indep2[mask] = 1;
            dp2[mask] = (int8_t)popcount32(mask);
            choice[mask] = mask;
        } else {
            indep2[mask] = 0;
            dp2[mask] = (int8_t)-100;
            choice[mask] = 0;
        }
    }

    // SOS DP: for every mask, store the best (maximum size) independent submask contained in it
    for (int i = 0; i < (int)n2; ++i) {
        for (unsigned int mask = 0; mask < maxMask2; ++mask) {
            if (mask & (1u << i)) {
                unsigned int prev = mask ^ (1u << i);
                if (dp2[prev] > dp2[mask]) {
                    dp2[mask] = dp2[prev];
                    choice[mask] = choice[prev];
                }
            }
        }
    }

    // First half: DP to find independent subsets and best combination with second half
    unsigned int maxMask1 = (n1 >= 32) ? 0u : (1u << n1);
    std::vector<char> indep1(maxMask1, 0);
    std::vector<unsigned char> pop1(maxMask1, 0);

    indep1[0] = 1;
    pop1[0] = 0;

    for (unsigned int mask = 1; mask < maxMask1; ++mask) {
        unsigned int bit = mask & -mask;
        int i = lsb_index(bit);
        unsigned int rest = mask ^ bit;
        indep1[mask] = indep1[rest] && ((adj1[i] & rest) == 0);
        pop1[mask] = (unsigned char)(pop1[rest] + 1);
    }

    int bestK = 0;
    unsigned int bestMask1 = 0;
    unsigned int bestMask2 = 0;

    for (unsigned int mask1 = 0; mask1 < maxMask1; ++mask1) {
        if (!indep1[mask1]) continue;

        // Compute union of cross-edges from vertices in mask1
        unsigned int crossUnion = 0;
        unsigned int mm = mask1;
        while (mm) {
            unsigned int bit = mm & -mm;
            int i = lsb_index(bit);
            crossUnion |= crossMask[i];
            mm ^= bit;
        }

        unsigned int allowed2 = (~crossUnion) & ( (n2 == 32) ? 0xFFFFFFFFu : ((1u << n2) - 1) );
        int total = (int)pop1[mask1] + (int)dp2[allowed2];
        if (total > bestK) {
            bestK = total;
            bestMask1 = mask1;
            bestMask2 = choice[allowed2];
        }
    }

    // Reconstruct answer
    std::vector<int> ans;
    for (int i = 0; i < n1; ++i) if (bestMask1 & (1u << i)) ans.push_back(i);
    for (int j = 0; j < n2; ++j) if (bestMask2 & (1u << j)) ans.push_back(n1 + j);

    std::sort(ans.begin(), ans.end());

    std::cout << bestK << "\n";
    if (!ans.empty()) {
        for (size_t i = 0; i < ans.size(); ++i) {
            if (i) std::cout << ' ';
            std::cout << ans[i];
        }
        std::cout << "\n";
    } else {
        std::cout << "\n";
    }

    return 0;
}
```
