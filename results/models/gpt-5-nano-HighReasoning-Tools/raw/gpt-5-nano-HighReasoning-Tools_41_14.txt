; SPIR-V
; Version: 1.0
; Generator: custom
; Bound: 99999
; Schema: 0
OpCapability Shader
OpMemoryModel Logical GLSL450

%void = OpTypeVoid
%float = OpTypeFloat 32
%uint = OpTypeInt 32 0

; Basic types
%v2 = OpTypeVector %float 2
%v3 = OpTypeVector %float 3
%v4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4 4

; UBO struct
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4 %v4 %v4
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
%_ptr_Uniform_vec4 = OpTypePointer Uniform %v4
; Inputs/Outputs
%_ptr_Input_v3 = OpTypePointer Input %v3
%_ptr_Input_v2 = OpTypePointer Input %v2
%_ptr_Output_v4 = OpTypePointer Output %v4

; Extensions
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Global variables (I/O and Uniforms)
%worldPos = OpVariable %_ptr_Input_v3 Input
%normal = OpVariable %_ptr_Input_v3 Input
%uv = OpVariable %_ptr_Input_v2 Input
%tangent = OpVariable %_ptr_Input_v3 Input
%color = OpVariable %_ptr_Input_v3 Input
%outColor = OpVariable %_ptr_Output_v4 Output
%ubo = OpVariable %_ptr_Uniform_UBO Uniform

; Decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; Pointer types for struct and members
%_ptr_Uniform_vec4 = OpTypePointer Uniform %v4
%_ptr_Uniform_UBO_for_mat4 = OpTypePointer Uniform %UBO
%_ptr_Uniform_mat4 = OpTypePointer Uniform %mat4
; Offsets helpers
%const_0 = OpConstant %float 0.0
%const_1 = OpConstant %float 1.0
%const_0_3 = OpConstant %float 0.3
%const_0_7 = OpConstant %float 0.7
%const_0_8 = OpConstant %float 0.8
%const_0_2 = OpConstant %float 0.2
%const_1_5 = OpConstant %float 1.5
%const_3 = OpConstant %float 3.0

; Misc constants
%idx_0 = OpConstant %uint 0
%idx_1 = OpConstant %uint 1
%idx_2 = OpConstant %uint 2
%idx_3 = OpConstant %uint 3
%idx_4 = OpConstant %uint 4

; Function types
%_funcType = OpTypeFunction %void
; Main function
%main = OpFunction %void None %_funcType
%entry = OpLabel

; Load inputs
%worldPosVal = OpLoad %v3 %worldPos
%normalVal = OpLoad %v3 %normal
%uvVal = OpLoad %v2 %uv
%colorVal = OpLoad %v3 %color

; Normalize inputs
%normNorm = OpExtInst %v3 %GLSLstd450 Normalize %normalVal

; Light position from UBO
%lightPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %idx_3
%lightPosVec4 = OpLoad %v4 %lightPos_ptr
%lightPosX = OpCompositeExtract %float %lightPosVec4 0
%lightPosY = OpCompositeExtract %float %lightPosVec4 1
%lightPosZ = OpCompositeExtract %float %lightPosVec4 2
%lightPosVec3 = OpCompositeConstruct %v3 %lightPosX %lightPosY %lightPosZ
; Light direction
%lightDir = OpFSub %v3 %lightPosVec3 %worldPosVal
%lightDirNorm = OpExtInst %v3 %GLSLstd450 Normalize %lightDir

; Diffuse term (Lambertian)
%dotNL = OpExtInst %float %GLSLstd450 Dot %normNorm %lightDirNorm
%diffuse = OpExtInst %float %GLSLstd450 FMax %dotNL %const_0
; Lambertian factor: 0.3 + 0.7*diffuse
%lambert = OpFMul %float %diffuse %const_0_7
%lambertFactor = OpFAdd %float %lambert %const_0_3

; Base color (0.8, 0.2, 0.2) scaled by lambertFactor
%baseColorVec3 = OpCompositeConstruct %v3 %const_0_8 %const_0_2 %const_0_2
%baseColor = OpVectorTimesScalar %v3 %baseColorVec3 %lambertFactor

; Fog calculations
%camPos_ptr = OpAccessChain %_ptr_Uniform_vec4 %ubo %idx_4
%camPosVec4 = OpLoad %v4 %camPos_ptr
%camPosX = OpCompositeExtract %float %camPosVec4 0
%camPosY = OpCompositeExtract %float %camPosVec4 1
%camPosZ = OpCompositeExtract %float %camPosVec4 2
%camPosVec3 = OpCompositeConstruct %v3 %camPosX %camPosY %camPosZ
%worldCamDiff = OpFSub %v3 %worldPosVal %camPosVec3
%dist = OpExtInst %float %GLSLstd450 Length %worldCamDiff
%fogRaw = OpFSub %float %dist %const_1_5
%fogDiv = OpFDiv %float %fogRaw %const_3
%fogFactor = OpExtInst %float %GLSLstd450 FClamp %fogDiv %const_0 %const_1

; Fog color (0.7, 0.7, 0.8)
%fogColorVec3 = OpCompositeConstruct %v3 %const_0_7 %const_0_7 %const_0_8
; Mix baseColor and fogColor according to fogFactor
%oneMinusFog = OpFSub %float %const_1 %fogFactor
%part1 = OpVectorTimesScalar %v3 %baseColor %oneMinusFog
%part2 = OpVectorTimesScalar %v3 %fogColorVec3 %fogFactor
%finalRGB = OpFAdd %v3 %part1 %part2
%finalColor4 = OpCompositeConstruct %v4 %finalRGB %const_1

; Store and finish
OpStore %outColor %finalColor4
OpReturn
OpFunctionEnd

; Entry points and execution mode
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %ubo
OpExecutionMode %main OriginUpperLeft
