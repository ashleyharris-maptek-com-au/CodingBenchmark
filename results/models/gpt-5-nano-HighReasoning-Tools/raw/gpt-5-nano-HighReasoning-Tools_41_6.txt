; SPIR-V
; Version: 1.0
; Generator: OpenAI
; Bound: 999
; Schema: 0

OpCapability Shader
OpMemoryModel Logical GLSL450

; Types
%void = OpTypeVoid
%voidfn = OpTypeFunction %void
%float = OpTypeFloat 32
%_u32 = OpTypeInt 32 0
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4

; UBO (Block)
%UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
; Offsets will be used by host; referenced via Block/Offsets
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %UBO Block

; Input/Output pointers
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_V4float = OpTypePointer Uniform %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
; Inputs
%worldPosIn = OpVariable %_ptr_Input_v3float Input
OpDecorate %worldPosIn Location 0
%normalIn = OpVariable %_ptr_Input_v3float Input
OpDecorate %normalIn Location 1
%uvIn = OpVariable %_ptr_Input_v2float Input
OpDecorate %uvIn Location 2
%tangentIn = OpVariable %_ptr_Input_v3float Input
OpDecorate %tangentIn Location 3
%colorIn = OpVariable %_ptr_Input_v3float Input
OpDecorate %colorIn Location 4
; Output
$outColor = OpVariable %_ptr_Output_v4float Output
OpDecorate %outColor Location 0
; Uniform
%ubo = OpVariable %_ptr_Uniform_UBO Uniform
OpDecorate %ubo DescriptorSet 0
OpDecorate %ubo Binding 0

; GLSL std 450
%GLSLstd450 = OpExtInstImport "GLSL.std.450"

; Constants
%float_0 = OpConstant %float 0.0
%float_1 = OpConstant %float 1.0
%float_64 = OpConstant %float 64.0
%float_0_1 = OpConstant %float 0.1
%float_0_5 = OpConstant %float 0.5
%float_0_6 = OpConstant %float 0.6
%float_0_8 = OpConstant %float 0.8

; Indices for uniform block access
%idx3 = OpConstant %_u32 3
%idx4 = OpConstant %_u32 4

; Main function
%main = OpFunction %void None %voidfn
%entry = OpLabel

; Load inputs
%worldPos = OpLoad %v3float %worldPosIn
%norm = OpLoad %v3float %normalIn
; Normalize N
%N = OpExtInst %v3float %GLSLstd450 Normalize %norm

; Load lightPos from UBO (member 3)
%lightPosPtr = OpAccessChain %_ptr_Uniform_V4float %ubo %idx3
%lightPosVal = OpLoad %v4float %lightPosPtr
; Extract xyz
%lp_x = OpCompositeExtract %float %lightPosVal 0
%lp_y = OpCompositeExtract %float %lightPosVal 1
%lp_z = OpCompositeExtract %float %lightPosVal 2
%lightPosXYZ = OpCompositeConstruct %v3float %lp_x %lp_y %lp_z

; Load cameraPos from UBO (member 4)
%camPosPtr = OpAccessChain %_ptr_Uniform_V4float %ubo %idx4
%camPosVal = OpLoad %v4float %camPosPtr
%cp_x = OpCompositeExtract %float %camPosVal 0
%cp_y = OpCompositeExtract %float %camPosVal 1
%cp_z = OpCompositeExtract %float %camPosVal 2
%camPosXYZ = OpCompositeConstruct %v3float %cp_x %cp_y %cp_z

; L = normalize(lightPos.xyz - worldPos)
%Ldir = OpFSub %v3float %lightPosXYZ %worldPos
%L = OpExtInst %v3float %GLSLstd450 Normalize %Ldir

; V = normalize(cameraPos - worldPos)
%Vdir = OpFSub %v3float %camPosXYZ %worldPos
%V = OpExtInst %v3float %GLSLstd450 Normalize %Vdir

; H = normalize(L + V)
%LplusV = OpFAdd %v3float %L %V
%H = OpExtInst %v3float %GLSLstd450 Normalize %LplusV

; Diffuse: diffuse = max(dot(N,L), 0)
%dotNL = OpExtInst %float %GLSLstd450 Dot %N %L
%absDotNL = OpExtInst %float %GLSLstd450 Abs %dotNL
%sumNL = OpFAdd %float %dotNL %absDotNL
%diffuse = OpFMul %float %float_0_5 %sumNL

; Specular: specular = pow(max(dot(N,H),0), 64)
%dotNH = OpExtInst %float %GLSLstd450 Dot %N %H
%absDotNH = OpExtInst %float %GLSLstd450 Abs %dotNH
%sumNH = OpFAdd %float %dotNH %absDotNH
%clampedNH = OpFMul %float %float_0_5 %sumNH
%exp64 = OpConstant %float 64.0
%spec = OpExtInst %float %GLSLstd450 Pow %clampedNH %exp64

; Intensity = 0.1 + 0.6*diffuse + 0.8*spec
%term1 = OpFMul %float %diffuse %float_0_6
%term2 = OpFMul %float %spec %float_0_8
%sumTerms = OpFAdd %float %term1 %term2
%intensity = OpFAdd %float %sumTerms % %float_0_1

; Output color as grayscale with alpha = 1.0
%outColorVec = OpCompositeConstruct %v4float %intensity %intensity %intensity %float_1
OpStore %outColor %outColorVec

OpReturn
OpFunctionEnd

; Entry point declarations
OpEntryPoint Fragment %main "main" %worldPosIn %normalIn %outColor
OpExecutionMode %main OriginUpperLeft
