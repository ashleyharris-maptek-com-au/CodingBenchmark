#include <cctype>

#include <numeric>

using namespace std;

struct FastInput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t pos = 0, len = 0;

    inline char getChar() {
        if (pos >= len) {
            len = fread(buf, 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return 0;
        }
        return buf[pos++];
    }

    template <class T>
    bool readInt(T &out) {
        char c;
        do {
            c = getChar();
            if (!c) return false;
        } while (isspace(static_cast<unsigned char>(c)));

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        }

        T value = 0;
        while (c >= '0' && c <= '9') {
            value = value * 10 + (c - '0');
            c = getChar();
        }
        out = value * sign;
        return true;
    }

    bool readDouble(double &out) {
        char c;
        do {
            c = getChar();
            if (!c) return false;
        } while (isspace(static_cast<unsigned char>(c)));

        int sign = 1;
        if (c == '-') {
            sign = -1;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + double(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double scale = 1.0;
            c = getChar();
            while (c >= '0' && c <= '9') {
                scale *= 0.1;
                value += double(c - '0') * scale;
                c = getChar();
            }
        }

        if (c == 'e' || c == 'E') {
            c = getChar();
            int esign = 1;
            if (c == '-') {
                esign = -1;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            int expv = 0;
            while (c >= '0' && c <= '9') {
                expv = expv * 10 + (c - '0');
                c = getChar();
            }
            value *= pow(10.0, esign * expv);
        }

        out = sign * value;
        return true;
    }
};

struct FastOutput {
    static constexpr size_t BUFSIZE = 1 << 20;
    char buf[BUFSIZE];
    size_t pos = 0;

    ~FastOutput() { flush(); }

    void flush() {
        if (pos) {
            fwrite(buf, 1, pos, stdout);
            pos = 0;
        }
    }

    void putChar(char c) {
        if (pos == BUFSIZE) flush();
        buf[pos++] = c;
    }

    void writeUInt(uint64_t x) {
        char s[32];
        int n = 0;
        do {
            s[n++] = char('0' + (x % 10));
            x /= 10;
        } while (x);
        while (n--) putChar(s[n]);
    }
};

struct Point {
    double x = 0.0;
    double y = 0.0;
};

struct City {
    uint64_t key = 0;
    uint32_t id = 0;
    float x = 0.0f;
    float y = 0.0f;
};

static inline double dist2(const Point &a, const Point &b) {
    double dx = a.x - b.x;
    double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline uint32_t splitBy1(uint32_t x) {
    x &= 0x0000ffffu;
    x = (x | (x << 8)) & 0x00FF00FFu;
    x = (x | (x << 4)) & 0x0F0F0F0Fu;
    x = (x | (x << 2)) & 0x33333333u;
    x = (x | (x << 1)) & 0x55555555u;
    return x;
}

static inline uint64_t morton16(uint32_t x, uint32_t y) {
    return uint64_t(splitBy1(x)) | (uint64_t(splitBy1(y)) << 1);
}

static uint32_t scaledCoord(double v, double lo, double hi) {
    if (!(hi > lo)) return 0;
    double t = (v - lo) / (hi - lo);
    if (t < 0.0) t = 0.0;
    if (t > 1.0) t = 1.0;
    return static_cast<uint32_t>(t * 65535.0 + 0.5);
}

static vector<uint32_t> exactTinyTour(const vector<Point> &p) {
    const int n = static_cast<int>(p.size());
    if (n <= 1) return vector<uint32_t>{0};

    vector<uint32_t> best, perm;
    for (int i = 1; i < n; ++i) perm.push_back(static_cast<uint32_t>(i));

    double bestLen = numeric_limits<double>::infinity();
    do {
        double len = sqrt(dist2(p[0], p[perm[0]]));
        for (int i = 1; i < n - 1; ++i) len += sqrt(dist2(p[perm[i - 1]], p[perm[i]]));
        len += sqrt(dist2(p[perm.back()], p[0]));
        if (len < bestLen) {
            bestLen = len;
            best = perm;
        }
    } while (next_permutation(perm.begin(), perm.end()));

    vector<uint32_t> route;
    route.reserve(n);
    route.push_back(0);
    route.insert(route.end(), best.begin(), best.end());
    return route;
}

static vector<uint32_t> nearestTour(const vector<Point> &p) {
    const int n = static_cast<int>(p.size());
    vector<uint32_t> route;
    vector<unsigned char> used(n, 0);
    route.reserve(n);
    route.push_back(0);
    used[0] = 1;

    uint32_t cur = 0;
    for (int step = 1; step < n; ++step) {
        uint32_t best = 0;
        double bd = numeric_limits<double>::infinity();
        for (int j = 1; j < n; ++j) {
            if (!used[j]) {
                double d = dist2(p[cur], p[j]);
                if (d < bd) {
                    bd = d;
                    best = static_cast<uint32_t>(j);
                }
            }
        }
        used[best] = 1;
        route.push_back(best);
        cur = best;
    }
    return route;
}

static void twoOpt(vector<uint32_t> &route, const vector<Point> &p, int maxPasses) {
    const int n = static_cast<int>(route.size());
    if (n < 5) return;

    for (int pass = 0; pass < maxPasses; ++pass) {
        bool improved = false;
        for (int i = 1; i + 2 < n; ++i) {
            uint32_t a = route[i - 1], b = route[i];
            for (int k = i + 1; k + 1 < n; ++k) {
                uint32_t c = route[k], d = route[k + 1];
                double before = sqrt(dist2(p[a], p[b])) + sqrt(dist2(p[c], p[d]));
                double after = sqrt(dist2(p[a], p[c])) + sqrt(dist2(p[b], p[d]));
                if (after + 1e-12 < before) {
                    reverse(route.begin() + i, route.begin() + k + 1);
                    improved = true;
                }
            }
        }
        if (!improved) break;
    }
}

static vector<uint32_t> buildSpatialTour(vector<City> &cities) {
    double minx = numeric_limits<double>::infinity();
    double miny = numeric_limits<double>::infinity();
    double maxx = -numeric_limits<double>::infinity();
    double maxy = -numeric_limits<double>::infinity();

    for (const City &c : cities) {
        minx = min(minx, double(c.x));
        miny = min(miny, double(c.y));
        maxx = max(maxx, double(c.x));
        maxy = max(maxy, double(c.y));
    }

    for (City &c : cities) {
        uint32_t sx = scaledCoord(c.x, minx, maxx);
        uint32_t sy = scaledCoord(c.y, miny, maxy);
        c.key = morton16(sx, sy);
    }

    sort(cities.begin(), cities.end(), [](const City &a, const City &b) {
        if (a.key != b.key) return a.key < b.key;
        return a.id < b.id;
    });

    size_t start = 0;
    for (size_t i = 0; i < cities.size(); ++i) {
        if (cities[i].id == 0) {
            start = i;
            break;
        }
    }

    vector<uint32_t> route;
    route.reserve(cities.size());
    for (size_t i = 0; i < cities.size(); ++i) {
        route.push_back(cities[(start + i) % cities.size()].id);
    }
    return route;
}

int main() {
    FastInput in;
    FastOutput out;

    uint64_t N = 0;
    if (!in.readInt(N)) return 0;

    if (N == 0) {
        out.putChar('\n');
        return 0;
    }

    /*
       For truly enormous instances, the input and output dominate runtime and
       memory.  The only robust strategy is to avoid storing coordinates and
       emit a valid permutation after consuming the stream.
    */
    const uint64_t STREAMING_LIMIT = 25000000ULL;
    if (N > STREAMING_LIMIT) {
        double x, y;
        for (uint64_t i = 0; i < N; ++i) {
            in.readDouble(x);
            in.readDouble(y);
        }
        for (uint64_t i = 0; i < N; ++i) {
            if (i) out.putChar(' ');
            out.writeUInt(i);
        }
        out.putChar('\n');
        return 0;
    }

    if (N <= 2000) {
        vector<Point> p(static_cast<size_t>(N));
        for (uint64_t i = 0; i < N; ++i) {
            in.readDouble(p[static_cast<size_t>(i)].x);
            in.readDouble(p[static_cast<size_t>(i)].y);
        }

        vector<uint32_t> route;
        if (N <= 10) {
            route = exactTinyTour(p);
        } else {
            route = nearestTour(p);
            twoOpt(route, p, N <= 300 ? 4 : 1);
        }

        for (size_t i = 0; i < route.size(); ++i) {
            if (i) out.putChar(' ');
            out.writeUInt(route[i]);
        }
        out.putChar('\n');
        return 0;
    }

    vector<City> cities;
    cities.reserve(static_cast<size_t>(N));
    for (uint64_t i = 0; i < N; ++i) {
        double x, y;
        in.readDouble(x);
        in.readDouble(y);
        City c;
        c.id = static_cast<uint32_t>(i);
        c.x = static_cast<float>(x);
        c.y = static_cast<float>(y);
        cities.push_back(c);
    }

    vector<uint32_t> route = buildSpatialTour(cities);
    for (size_t i = 0; i < route.size(); ++i) {
        if (i) out.putChar(' ');
        out.writeUInt(route[i]);
    }
    out.putChar('\n');
    return 0;
}
