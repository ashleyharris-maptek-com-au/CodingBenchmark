#include <algorithm>
#include <cctype>
#include <cfloat>
#include <cmath>
#include <cstdio>
#include <cstdint>
#include <limits>
#include <numeric>
#include <string>
#include <vector>

struct FastInput {
    static constexpr std::size_t BUFSIZE = 1u << 20;
    std::vector<char> buffer;
    std::size_t pos = 0, len = 0;

    FastInput() : buffer(BUFSIZE) {}

    int getChar() {
        if (pos >= len) {
            len = std::fread(buffer.data(), 1, BUFSIZE, stdin);
            pos = 0;
            if (len == 0) return EOF;
        }
        return buffer[pos++];
    }

    template <class T>
    bool readUnsigned(T& out) {
        int c;
        do {
            c = getChar();
            if (c == EOF) return false;
        } while (std::isspace(static_cast<unsigned char>(c)));

        T value = 0;
        while (c != EOF && !std::isspace(static_cast<unsigned char>(c))) {
            value = static_cast<T>(value * 10 + (c - '0'));
            c = getChar();
        }
        out = value;
        return true;
    }

    bool readDouble(double& out) {
        int c;
        do {
            c = getChar();
            if (c == EOF) return false;
        } while (std::isspace(static_cast<unsigned char>(c)));

        bool neg = false;
        if (c == '-') {
            neg = true;
            c = getChar();
        } else if (c == '+') {
            c = getChar();
        }

        double value = 0.0;
        while (c >= '0' && c <= '9') {
            value = value * 10.0 + static_cast<double>(c - '0');
            c = getChar();
        }

        if (c == '.') {
            double place = 0.1;
            c = getChar();
            while (c >= '0' && c <= '9') {
                value += place * static_cast<double>(c - '0');
                place *= 0.1;
                c = getChar();
            }
        }

        if (c == 'e' || c == 'E') {
            c = getChar();
            bool expNeg = false;
            if (c == '-') {
                expNeg = true;
                c = getChar();
            } else if (c == '+') {
                c = getChar();
            }
            int expValue = 0;
            while (c >= '0' && c <= '9') {
                expValue = expValue * 10 + (c - '0');
                c = getChar();
            }
            value *= std::pow(10.0, expNeg ? -expValue : expValue);
        }

        out = neg ? -value : value;
        return true;
    }
};

struct FastOutput {
    static constexpr std::size_t BUFSIZE = 1u << 20;
    std::vector<char> buffer;
    std::size_t pos = 0;

    FastOutput() : buffer(BUFSIZE) {}
    ~FastOutput() { flush(); }

    void flush() {
        if (pos) {
            std::fwrite(buffer.data(), 1, pos, stdout);
            pos = 0;
        }
    }

    void putChar(char c) {
        if (pos == BUFSIZE) flush();
        buffer[pos++] = c;
    }

    void writeInt(std::uint32_t value) {
        char tmp[16];
        int n = 0;
        do {
            tmp[n++] = static_cast<char>('0' + (value % 10));
            value /= 10;
        } while (value);
        while (n--) putChar(tmp[n]);
    }
};

struct City {
    double x = 0.0;
    double y = 0.0;
    std::uint32_t id = 0;
    std::uint64_t key = 0;
};

static inline double dist2(const City& a, const City& b) {
    const double dx = a.x - b.x;
    const double dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline double dist(const City& a, const City& b) {
    return std::sqrt(dist2(a, b));
}

static void rotateHilbert(std::uint32_t n, std::uint32_t& x, std::uint32_t& y,
                          std::uint32_t rx, std::uint32_t ry) {
    if (ry == 0) {
        if (rx != 0) {
            x = n - 1u - x;
            y = n - 1u - y;
        }
        std::uint32_t t = x;
        x = y;
        y = t;
    }
}

static std::uint64_t hilbertIndex(std::uint32_t x, std::uint32_t y) {
    constexpr std::uint32_t BITS = 31;
    constexpr std::uint32_t SIDE = 1u << BITS;
    std::uint64_t d = 0;
    for (std::uint32_t s = SIDE >> 1; s > 0; s >>= 1) {
        const std::uint32_t rx = (x & s) ? 1u : 0u;
        const std::uint32_t ry = (y & s) ? 1u : 0u;
        d += static_cast<std::uint64_t>(s) * static_cast<std::uint64_t>(s) *
             static_cast<std::uint64_t>((3u * rx) ^ ry);
        rotateHilbert(s, x, y, rx, ry);
    }
    return d;
}

static std::vector<std::uint32_t> exactTiny(const std::vector<City>& cities) {
    const std::uint32_t n = static_cast<std::uint32_t>(cities.size());
    std::vector<std::uint32_t> perm;
    for (std::uint32_t i = 1; i < n; ++i) perm.push_back(i);

    double bestLen = std::numeric_limits<double>::infinity();
    std::vector<std::uint32_t> best = perm;
    do {
        double total = dist(cities[0], cities[perm[0]]);
        for (std::uint32_t i = 1; i < perm.size(); ++i) {
            total += dist(cities[perm[i - 1]], cities[perm[i]]);
        }
        total += dist(cities[perm.back()], cities[0]);
        if (total < bestLen) {
            bestLen = total;
            best = perm;
        }
    } while (std::next_permutation(perm.begin(), perm.end()));

    std::vector<std::uint32_t> order;
    order.reserve(n);
    order.push_back(0);
    order.insert(order.end(), best.begin(), best.end());
    return order;
}

static std::vector<std::uint32_t> nearestNeighbor(const std::vector<City>& cities) {
    const std::uint32_t n = static_cast<std::uint32_t>(cities.size());
    std::vector<unsigned char> used(n, 0);
    std::vector<std::uint32_t> order;
    order.reserve(n);
    std::uint32_t cur = 0;
    used[cur] = 1;
    order.push_back(cur);

    for (std::uint32_t step = 1; step < n; ++step) {
        std::uint32_t best = 0;
        double bestD = std::numeric_limits<double>::infinity();
        for (std::uint32_t j = 0; j < n; ++j) {
            if (!used[j]) {
                const double d = dist2(cities[cur], cities[j]);
                if (d < bestD) {
                    bestD = d;
                    best = j;
                }
            }
        }
        used[best] = 1;
        order.push_back(best);
        cur = best;
    }
    return order;
}

static void twoOpt(std::vector<std::uint32_t>& order, const std::vector<City>& cities, int maxPasses) {
    const std::uint32_t n = static_cast<std::uint32_t>(order.size());
    if (n < 4) return;

    for (int pass = 0; pass < maxPasses; ++pass) {
        bool changed = false;
        for (std::uint32_t i = 1; i + 1 < n; ++i) {
            const City& a = cities[order[i - 1]];
            const City& b = cities[order[i]];
            for (std::uint32_t k = i + 1; k < n; ++k) {
                const City& c = cities[order[k]];
                const City& d = cities[order[(k + 1) % n]];
                const double before = dist(a, b) + dist(c, d);
                const double after = dist(a, c) + dist(b, d);
                if (after + 1e-12 < before) {
                    std::reverse(order.begin() + i, order.begin() + k + 1);
                    changed = true;
                    break;
                }
            }
        }
        if (!changed) break;
    }
}

static void windowedTwoOpt(std::vector<std::uint32_t>& order, const std::vector<City>& cities,
                           std::uint32_t window) {
    const std::uint32_t n = static_cast<std::uint32_t>(order.size());
    if (n < 4) return;
    for (std::uint32_t i = 1; i + 3 < n; ++i) {
        const std::uint32_t upper = std::min<std::uint32_t>(n - 1, i + window);
        for (std::uint32_t k = i + 1; k <= upper; ++k) {
            const City& a = cities[order[i - 1]];
            const City& b = cities[order[i]];
            const City& c = cities[order[k]];
            const City& d = cities[order[(k + 1) % n]];
            if (dist(a, c) + dist(b, d) + 1e-12 < dist(a, b) + dist(c, d)) {
                std::reverse(order.begin() + i, order.begin() + k + 1);
                break;
            }
        }
    }
}

static std::vector<std::uint32_t> hilbertTour(std::vector<City>& cities,
                                              double minX, double maxX,
                                              double minY, double maxY) {
    constexpr double SCALE = 2147483647.0;
    const double rangeX = maxX - minX;
    const double rangeY = maxY - minY;

    for (City& c : cities) {
        std::uint32_t ix = 0;
        std::uint32_t iy = 0;
        if (rangeX > 0.0) {
            double v = (c.x - minX) / rangeX;
            if (v < 0.0) v = 0.0;
            if (v > 1.0) v = 1.0;
            ix = static_cast<std::uint32_t>(v * SCALE);
        }
        if (rangeY > 0.0) {
            double v = (c.y - minY) / rangeY;
            if (v < 0.0) v = 0.0;
            if (v > 1.0) v = 1.0;
            iy = static_cast<std::uint32_t>(v * SCALE);
        }
        c.key = hilbertIndex(ix, iy);
    }

    std::sort(cities.begin(), cities.end(), [](const City& a, const City& b) {
        if (a.key != b.key) return a.key < b.key;
        return a.id < b.id;
    });

    std::vector<std::uint32_t> order;
    order.reserve(cities.size());
    for (const City& c : cities) order.push_back(c.id);

    const auto it = std::find(order.begin(), order.end(), 0u);
    if (it != order.end()) std::rotate(order.begin(), it, order.end());
    return order;
}

int main() {
    FastInput in;
    std::uint32_t n = 0;
    if (!in.readUnsigned(n)) return 0;

    std::vector<City> cities;
    cities.resize(n);
    double minX = std::numeric_limits<double>::infinity();
    double maxX = -std::numeric_limits<double>::infinity();
    double minY = std::numeric_limits<double>::infinity();
    double maxY = -std::numeric_limits<double>::infinity();

    for (std::uint32_t i = 0; i < n; ++i) {
        double x = 0.0, y = 0.0;
        in.readDouble(x);
        in.readDouble(y);
        cities[i].x = x;
        cities[i].y = y;
        cities[i].id = i;
        if (x < minX) minX = x;
        if (x > maxX) maxX = x;
        if (y < minY) minY = y;
        if (y > maxY) maxY = y;
    }

    std::vector<std::uint32_t> order;
    if (n <= 1) {
        order.reserve(n);
        if (n == 1) order.push_back(0);
    } else if (n <= 10) {
        order = exactTiny(cities);
    } else if (n <= 5000) {
        order = nearestNeighbor(cities);
        if (n <= 2000) twoOpt(order, cities, 2);
    } else {
        order = hilbertTour(cities, minX, maxX, minY, maxY);
        if (n <= 200000) windowedTwoOpt(order, cities, 32);
    }

    FastOutput out;
    for (std::uint32_t i = 0; i < order.size(); ++i) {
        if (i) out.putChar(' ');
        out.writeInt(order[i]);
    }
    out.putChar('\n');
    return 0;
}
