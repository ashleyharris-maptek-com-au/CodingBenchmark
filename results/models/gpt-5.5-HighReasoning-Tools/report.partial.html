static bool isWhitePiece(char p) { return p >= 'A' && p <= 'Z'; }

static int pieceVal(char p) {
    switch (upperPiece(p)) {
    case 'P': return 100;
    case 'N': return 320;
    case 'B': return 330;
    case 'R': return 500;
    case 'Q': return 900;
    case 'K': return 20000;
    default: return 0;
    }
}

static int pst(char p, int s) {
    int f = fileOf(s), r = rankOf(s);
    if (isBlackPiece(p)) r = 7 - r;
    int cf = 3 - abs(f - 3), cr = 3 - abs(r - 3);
    switch (upperPiece(p)) {
    case 'P': return r * 10 - abs(f - 3) * 2;
    case 'N': return (cf + cr) * 10 - (f == 0 || f == 7 || r == 0 || r == 7 ? 20 : 0);
    case 'B': return (cf + cr) * 6;
    case 'R': return r * 2;
    case 'Q': return (cf + cr) * 3;
    case 'K': return r < 2 ? 12 - abs(f - 3) * 2 : -(cf + cr) * 8;
    default: return 0;
    }
}

static Pos parseFen(const string& fen) {
    Pos p;
    p.b.fill('.');
    string board, stm, cast, ep;
    stringstream ss(fen);
    ss >> board >> stm >> cast >> ep >> p.halfmove >> p.fullmove;
    int r = 7, f = 0;
    for (char c : board) {
        if (c == '/') { --r; f = 0; }
        else if (isdigit((unsigned char)c)) { f += c - '0'; }
        else if (r >= 0 && f < 8) { p.b[r * 8 + f++] = c; }
    }
    p.white = (stm != "b");
    p.castle = 0;
    if (cast.find('K') != string::npos) p.castle |= 1;
    if (cast.find('Q') != string::npos) p.castle |= 2;
    if (cast.find('k') != string::npos) p.castle |= 4;
    if (cast.find('q') != string::npos) p.castle |= 8;
    if (ep != "-" && ep.size() == 2) p.ep = (ep[0] - 'a') + (ep[1] - '1') * 8;
    return p;
}

static Pos makeMove(const Pos& p, const Move& m) {
    Pos q = p;
    char pc = q.b[m.from];
    char target = q.b[m.to];
    q.b[m.from] = '.';
    if (m.ep) q.b[m.to + (p.white ? -8 : 8)] = '.';
    if (m.castle) {
        if (m.to == 6) { q.b[5] = 'R'; q.b[7] = '.'; }
        else if (m.to == 2) { q.b[3] = 'R'; q.b[0] = '.'; }
        else if (m.to == 62) { q.b[61] = 'r'; q.b[63] = '.'; }
        else if (m.to == 58) { q.b[59] = 'r'; q.b[56] = '.'; }
    }
    q.b[m.to] = m.promo ? (p.white ? m.promo : char(tolower((unsigned char)m.promo))) : pc;
    if (pc == 'K') q.castle &= ~3;
    if (pc == 'k') q.castle &= ~12;
    if (pc == 'R' && m.from == 0) q.castle &= ~2;
    if (pc == 'R' && m.from == 7) q.castle &= ~1;
    if (pc == 'r' && m.from == 56) q.castle &= ~8;
    if (pc == 'r' && m.from == 63) q.castle &= ~4;
    if (target == 'R' && m.to == 0) q.castle &= ~2;
    if (target == 'R' && m.to == 7) q.castle &= ~1;
    if (target == 'r' && m.to == 56) q.castle &= ~8;
    if (target == 'r' && m.to == 63) q.castle &= ~4;
    q.ep = -1;
    if (upperPiece(pc) == 'P' && abs(m.to - m.from) == 16) q.ep = (m.from + m.to) / 2;
    q.white = !p.white;
    if (!q.white) ++q.fullmove;
    return q;
}

static vector<Move> legalMoves(const Pos& p, bool capturesOnly = false) {
    vector<Move> ps, out;
    genPseudo(p, ps, capturesOnly);
    out.reserve(ps.size());
    for (const Move& m : ps) {
        Pos q = makeMove(p, m);
        if (!inCheck(q, !q.white)) out.push_back(m);
    }
    return out;
}

static int evaluate(const Pos& p) {
    int score = 0;
    for (int i = 0; i < 64; ++i) {
        char pc = p.b[i];
        if (pc == '.') continue;
        int v = pieceVal(pc) + pst(pc, i);
        score += isWhitePiece(pc) ? v : -v;
    }
    return p.white ? score : -score;
}

static bool timeUp() {
    if (timedOut) return true;
    if (chrono::steady_clock::now() >= deadline) {
        timedOut = true;
        return true;
    }
    return false;
}

static int moveScore(const Pos& p, const Move& m) {
    char pc = p.b[m.from];
    char cap = m.ep ? (p.white ? 'p' : 'P') : p.b[m.to];
    int s = 0;
    if (cap != '.') s += 10000 + pieceVal(cap) - pieceVal(pc) / 10;
    if (m.promo) s += pieceVal(m.promo);
    if (m.castle) s += 50;
    int cf = abs(fileOf(m.to) - 3), cr = abs(rankOf(m.to) - 3);
    s += 20 - cf - cr;
    return s;
}

static int quiesce(const Pos& p, int alpha, int beta) {
    if (timeUp()) return evaluate(p);
    int stand = evaluate(p);
    if (stand >= beta) return beta;
    if (stand > alpha) alpha = stand;
    vector<Move> ms = legalMoves(p, true);
    sort(ms.begin(), ms.end(), [&](const Move& a, const Move& b) { return moveScore(p, a) > moveScore(p, b); });
    for (const Move& m : ms) {
        Pos q = makeMove(p, m);
        int score = -quiesce(q, -beta, -alpha);
        if (score >= beta) return beta;
        if (score > alpha) alpha = score;
    }
    return alpha;
}

static int negamax(const Pos& p, int depth, int alpha, int beta) {
    if (timeUp()) return evaluate(p);
    if (depth <= 0) return quiesce(p, alpha, beta);
    vector<Move> ms = legalMoves(p);
    if (ms.empty()) return inCheck(p, p.white) ? -10000000 - depth : 0;
    sort(ms.begin(), ms.end(), [&](const Move& a, const Move& b) { return moveScore(p, a) > moveScore(p, b); });
    int best = -INF;
    for (const Move& m : ms) {
        Pos q = makeMove(p, m);
        int score = -negamax(q, depth - 1, -beta, -alpha);
        if (timeUp()) return best;
        if (score > best) best = score;
        if (score > alpha) alpha = score;
        if (alpha >= beta) break;
    }
    return best;
}

static string notation(const Pos& p, const Move& m) {
    if (m.castle) {
        string s = m.to > m.from ? "O-O" : "O-O-O";
        Pos q = makeMove(p, m);
        if (inCheck(q, q.white)) {
            vector<Move> replies = legalMoves(q);
            s += replies.empty() ? '#' : '+';
        }
        return s;
    }
    char pc = p.b[m.from];
    string s;
    s += (upperPiece(pc) == 'P') ? 'p' : upperPiece(pc);
    s += sqName(m.from);
    if (m.capture || m.ep) s += 'x';
    s += sqName(m.to);
    if (m.promo) { s += '/'; s += m.promo; }
    if (m.ep) s += "ep";
    Pos q = makeMove(p, m);
    if (inCheck(q, q.white)) {
        vector<Move> replies = legalMoves(q);
        s += replies.empty() ? '#' : '+';
    }
    return s;
}
