#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <set>
#include <bit>
#include <utility>

using namespace std;

/**
 * Steiner Tree Problem Solver (NP-Hard)
 * 
 * Algorithm: Dreyfus-Wagner (Dynamic Programming)
 * Complexity: O(3^K * N + 2^K * (N log N + M))
 * where K = number of terminals, N = number of vertices, M = number of edges.
 * 
 * This exact algorithm is suitable for small terminal sets (K up to 12-15).
 * It uses DP to find the minimum weight tree connecting subsets of terminals.
 */

typedef long long ll;
const ll INF_LL = 1e16; // Sufficiently large to avoid overflow while representing infinity

struct Edge {
    int to;
    int weight;
};

struct Node {
    int v;
    ll dist;
    bool operator>(const Node& other) const {
        return dist > other.dist;
    }
};

struct Parent {
    int submask; // -1 if the state was reached via a Dijkstra edge extension
    int prev_v;  // -1 if the state was reached via a mask split (merging two trees)
};

int main() {
    // Optimization for fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m, k;
    if (!(cin >> n >> m >> k)) return 0;
    if (n <= 0) return 0;

    // Adjacency list representation of the graph
    vector<vector<Edge>> adj(n);
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        if (u >= 0 && u < n && v >= 0 && v < n) {
            adj[u].push_back({v, w});
            adj[v].push_back({u, w});
        }
    }

    // Read and preprocess terminals
    vector<int> t_raw(k);
    for (int i = 0; i < k; ++i) {
        cin >> t_raw[i];
    }
    sort(t_raw.begin(), t_raw.end());
    t_raw.erase(unique(t_raw.begin(), t_raw.end()), t_raw.end());
    
    vector<int> terminals;
    for (int t : t_raw) {
        if (t >= 0 && t < n) {
            terminals.push_back(t);
        }
    }
    k = (int)terminals.size();

    // Handle trivial cases
    if (k == 0) {
        cout << "0 0" << endl;
        return 0;
    }
    if (k == 1) {
        cout << "0 0" << endl;
        return 0;
    }

    int num_masks = 1 << k;
    // dp[mask * n + v] stores the min weight to connect terminals in 'mask' with v in the tree.
    vector<ll> dp(num_masks * n, INF_LL);
    // parent table used for backtracking and reconstructing the tree edges.
    vector<Parent> parent(num_masks * n, {-1, -1});

    // Initialize DP with single-terminal states
    for (int i = 0; i < k; ++i) {
        dp[(1 << i) * n + terminals[i]] = 0;
    }

    // Main Dreyfus-Wagner DP loop
    for (int mask = 1; mask < num_masks; ++mask) {
        int mask_offset = mask * n;
        
        // Transition Type 1: Merging two trees at a common vertex v
        // Iterate over all partitions of the current terminal subset 'mask'.
        for (int submask = (mask - 1) & mask; submask > (mask ^ submask); submask = (submask - 1) & mask) {
            int other = mask ^ submask;
            int sub_offset = submask * n;
            int oth_offset = other * n;
            
            for (int v = 0; v < n; ++v) {
                if (dp[sub_offset + v] < INF_LL && dp[oth_offset + v] < INF_LL) {
                    ll combined = dp[sub_offset + v] + dp[oth_offset + v];
                    if (combined < dp[mask_offset + v]) {
                        dp[mask_offset + v] = combined;
                        parent[mask_offset + v] = {submask, -1};
                    }
                }
            }
        }

        // Transition Type 2: Extending a tree using an edge (similar to Dijkstra)
        priority_queue<Node, vector<Node>, greater<Node>> pq;
        for (int v = 0; v < n; ++v) {
            if (dp[mask_offset + v] < INF_LL) {
                pq.push({v, dp[mask_offset + v]});
            }
        }

        while (!pq.empty()) {
            Node top = pq.top();
            pq.pop();

            if (top.dist > dp[mask_offset + top.v]) continue;

            for (const auto& edge : adj[top.v]) {
                if (dp[mask_offset + top.v] + (ll)edge.weight < dp[mask_offset + edge.to]) {
                    dp[mask_offset + edge.to] = dp[mask_offset + top.v] + (ll)edge.weight;
                    parent[mask_offset + edge.to] = {-1, top.v};
                    pq.push({edge.to, dp[mask_offset + edge.to]});
                }
            }
        }
    }

    // Find the minimum weight across all vertices for the full set of terminals
    ll min_w = INF_LL;
    int root = -1;
    int full_mask = num_masks - 1;
    for (int v = 0; v < n; ++v) {
        if (dp[full_mask * n + v] < min_w) {
            min_w = dp[full_mask * n + v];
            root = v;
        }
    }

    // If no root is found, terminals are likely in different connected components
    if (root == -1 || min_w >= INF_LL) {
        return 0;
    }

    // Reconstruct the edges of the Steiner Tree using a stack-based approach
    set<pair<int, int>> res_edges;
    vector<char> visited(num_masks * n, 0);
    vector<pair<int, int>> st;
    st.push_back({full_mask, root});

    while (!st.empty()) {
        auto [msk, v] = st.back();
        st.pop_back();
        if (visited[msk * n + v]) continue;
        visited[msk * n + v] = 1;

        Parent p = parent[msk * n + v];
        if (p.submask != -1) {
            // Reconstruct the two subtrees merged at v
            st.push_back({p.submask, v});
            st.push_back({msk ^ p.submask, v});
        } else if (p.prev_v != -1) {
            // Reconstruct the path extension
            int a = v, b = p.prev_v;
            if (a > b) swap(a, b);
            res_edges.insert({a, b});
            st.push_back({msk, p.prev_v});
        }
    }

    // Final output: total weight and number of edges, followed by the edges themselves
    cout << min_w << " " << (int)res_edges.size() << "\n";
    for (const auto& e : res_edges) {
        cout << e.first << " " << e.second << "\n";
    }

    return 0;
}
