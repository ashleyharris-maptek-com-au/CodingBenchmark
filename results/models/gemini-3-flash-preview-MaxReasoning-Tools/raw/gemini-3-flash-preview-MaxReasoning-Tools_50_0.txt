#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    if (i >= N) return;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    vec3 posI = inData[i * 2].xyz;
    vec3 velI = inData[i * 2 + 1].xyz;

    vec3 separation = vec3(0.0);
    vec3 alignment = vec3(0.0);
    vec3 cohesion = vec3(0.0);

    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;

    for (uint j = 0; j < N; j++) {
        if (i == j) continue;

        vec3 posJ = inData[j * 2].xyz;
        vec3 velJ = inData[j * 2 + 1].xyz;

        float d = distance(posI, posJ);
        if (d > 0.0) {
            if (d < sepRadius) {
                separation += (posI - posJ) / d;
                sepCount++;
            }
            if (d < alignRadius) {
                alignment += velJ;
                alignCount++;
            }
            if (d < cohRadius) {
                cohesion += posJ;
                cohCount++;
            }
        }
    }

    vec3 accel = vec3(0.0);

    if (sepCount > 0) {
        accel += (separation / float(sepCount)) * sepWeight;
    }
    if (alignCount > 0) {
        accel += (alignment / float(alignCount)) * alignWeight;
    }
    if (cohCount > 0) {
        vec3 centerOfMass = cohesion / float(cohCount);
        accel += (centerOfMass - posI) * cohWeight;
    }

    // Boundary force
    // apply the boundary force when a component exceeds boundSize*0.4 
    // (push toward center using boundForce and boundSize*0.5 scaling)
    if (abs(posI.x) > boundSize * 0.4) accel.x -= (posI.x / (boundSize * 0.5)) * boundForce;
    if (abs(posI.y) > boundSize * 0.4) accel.y -= (posI.y / (boundSize * 0.5)) * boundForce;
    if (abs(posI.z) > boundSize * 0.4) accel.z -= (posI.z / (boundSize * 0.5)) * boundForce;

    // Update velocity
    velI += accel * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(velI);
    if (speed > maxSpeed) {
        velI = (velI / speed) * maxSpeed;
    }

    // Update position
    posI += velI * dt;

    // Write back
    outData[i * 2] = vec4(posI, 0.0);
    outData[i * 2 + 1] = vec4(velI, 0.0);
}
