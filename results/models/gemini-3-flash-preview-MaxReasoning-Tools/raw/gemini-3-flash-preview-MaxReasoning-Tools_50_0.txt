#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against gl_GlobalInvocationID.x >= N
    if (i >= N) {
        return;
    }

    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;

    vec3 sepSum = vec3(0.0);
    vec3 alignSum = vec3(0.0);
    vec3 cohSum = vec3(0.0);
    
    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;

    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    // Iterate through all boids to find neighbors
    for (uint j = 0; j < N; j++) {
        // Skip self
        if (i == j) {
            continue;
        }

        vec3 neighborPos = inData[j * 2].xyz;
        vec3 neighborVel = inData[j * 2 + 1].xyz;
        
        float dist = distance(pos, neighborPos);

        // Skip zero-distance neighbors to avoid division by zero
        if (dist > 0.0) {
            if (dist < sepRadius) {
                // Separation: Accumulate direction away from neighbor
                sepSum += (pos - neighborPos) / dist;
                sepCount++;
            }
            if (dist < alignRadius) {
                // Alignment: Accumulate neighbor velocity
                alignSum += neighborVel;
                alignCount++;
            }
            if (dist < cohRadius) {
                // Cohesion: Accumulate neighbor position
                cohSum += neighborPos;
                cohCount++;
            }
        }
    }

    vec3 accel = vec3(0.0);

    // Average contributions (avoiding division by zero) and apply weights
    if (sepCount > 0) {
        accel += (sepSum / float(sepCount)) * weights.x;
    }
    if (alignCount > 0) {
        // Steer towards average velocity
        accel += (alignSum / float(alignCount) - vel) * weights.y;
    }
    if (cohCount > 0) {
        // Steer towards average position
        accel += (cohSum / float(cohCount) - pos) * weights.z;
    }

    // Boundary force: push toward center when outside 80% of the box
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    vec3 bForce = vec3(0.0);
    
    // Threshold is 0.4 * boundSize, scaling is by 0.5 * boundSize (the edge)
    if (abs(pos.x) > boundSize * 0.4) {
        bForce.x -= (pos.x / (boundSize * 0.5)) * boundForce;
    }
    if (abs(pos.y) > boundSize * 0.4) {
        bForce.y -= (pos.y / (boundSize * 0.5)) * boundForce;
    }
    if (abs(pos.z) > boundSize * 0.4) {
        bForce.z -= (pos.z / (boundSize * 0.5)) * boundForce;
    }
    accel += bForce;

    // Update velocity
    vel += accel * dt;

    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > weights.w && speed > 0.0) {
        vel = (vel / speed) * weights.w;
    }

    // Update position
    pos += vel * dt;

    // Store results
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
