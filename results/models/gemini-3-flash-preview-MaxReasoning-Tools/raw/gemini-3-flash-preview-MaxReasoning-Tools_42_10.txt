```spirv
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputVertices 10

; Decorations
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%uint_4 = OpConstant %uint 4
%int = OpTypeInt 32 1
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%int_4 = OpConstant %int 4
%bool = OpTypeBool

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%arr_v3_4 = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_4
%color_in = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_func_int = OpTypePointer Function %int

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_25 = OpConstant %float 0.25
%float_0_01 = OpConstant %float 0.01

; Main Function
%main = OpFunction %void None %void_func
%entry = OpLabel
%i_var = OpVariable %ptr_func_int Function

; Load Control Points P0, P1, P2, P3
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

; Load Colors C1, C2
%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
%c2 = OpLoad %v3float %c2_ptr

; Tangents at start and end
%tstart_v4 = OpFSub %v4float %p2 %p0
%tstart = OpVectorShuffle %v2float %tstart_v4 %tstart_v4 0 1
%tend_v4 = OpFSub %v4float %p3 %p1
%tend = OpVectorShuffle %v2float %tend_v4 %tend_v4 0 1

OpStore %i_var %int_0
OpBranch %loop_header

%loop_header = OpLabel
%i = OpLoad %int %i_var
%cond = OpSLessThanEqual %bool %i %int_4
OpBranchConditional %cond %loop_body %loop_merge

%loop_body = OpLabel
%i_f = OpConvertSToF %float %i
%t = OpFMul %float %i_f %float_0_25

; Linear interpolation for positions
%p2_p1 = OpFSub %v4float %p2 %p1
%t_p2_p1 = OpVectorTimesScalar %v4float %p2_p1 %t
%pos = OpFAdd %v4float %p1 %t_p2_p1

; Lerp tangents for smooth perpendiculars
%one_minus_t = OpFSub %float %float_1 %t
%term1 = OpVectorTimesScalar %v2float %tstart %one_minus_t
%term2 = OpVectorTimesScalar %v2float %tend %t
%tang = OpFAdd %v2float %term1 %term2

; Calculate normalized perpendicular in 2D
%tang_norm = OpExtInst %v2float %1 Normalize %tang
%tang_norm_y = OpCompositeExtract %float %tang_norm 1
%tang_norm_x = OpCompositeExtract %float %tang_norm 0
%neg_tang_norm_y = OpFNegate %float %tang_norm_y
%perp_dir = OpCompositeConstruct %v2float %neg_tang_norm_y %tang_norm_x
%perp = OpVectorTimesScalar %v2float %perp_dir %float_0_01

; Interpolate color
%c1_term = OpVectorTimesScalar %v3float %c1 %one_minus_t
%c2_term = OpVectorTimesScalar %v3float %c2 %t
%color = OpFAdd %v3float %c1_term %c2_term

; Prepare perp offset as v4
%perp_x = OpCompositeExtract %float %perp 0
%perp_y = OpCompositeExtract %float %perp 1
%perp_v4 = OpCompositeConstruct %v4float %perp_x %perp_y %float_0 %float_0

; Output vertex 1: pos + perp
%v1_pos = OpFAdd %v4float %pos %perp_v4
OpStore %outColor %color
%gl_out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %gl_out_pos_ptr %v1_pos
OpEmitVertex

; Output vertex 2: pos - perp
%v2_pos = OpFSub %v4float %pos %perp_v4
OpStore %outColor %color
OpStore %gl_out_pos_ptr %v2_pos
OpEmitVertex

; Increment loop counter
%next_i = OpIAdd %int %i %int_1
OpStore %i_var %next_i
OpBranch %loop_header

%loop_merge = OpLabel
OpEndPrimitive
OpReturn
OpFunctionEnd
```
