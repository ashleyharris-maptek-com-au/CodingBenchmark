; SPIR-V
; Version: 1.0
; Generator: Khronos; 1
; Bound: 150
; Schema: 0
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor %worldPos %normal
               OpExecutionMode %main InputLinesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 10

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1

; Types
       %void = OpTypeVoid
    %void_fn = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_4 = OpConstant %uint 4
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
       %bool = OpTypeBool

; PerVertex blocks
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

; User inputs/outputs
 %arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

; Pointer types
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_int_Function = OpTypePointer Function %int

; Constants
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
 %float_0_25 = OpConstant %float 0.25
 %float_0_01 = OpConstant %float 0.01

; Main Function
       %main = OpFunction %void None %void_fn
 %label_main = OpLabel
        %idx = OpVariable %ptr_int_Function Function

; Load Input Positions
     %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %ptr_p0
     %ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %ptr_p1
     %ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %ptr_p2
     %ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
         %p3 = OpLoad %v4float %ptr_p3

; Load Input Colors
     %ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %ptr_c1
     %ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %ptr_c2

; Calculate Tangents T1 = P2 - P0, T2 = P3 - P1
    %t1_full = OpFSub %v4float %p2 %p0
    %t2_full = OpFSub %v4float %p3 %p1

; Loop Start
               OpStore %idx %int_0
               OpBranch %loop_cond

  %loop_cond = OpLabel
 %curr_idx_i = OpLoad %int %idx
       %cond = OpSLessThanEqual %bool %curr_idx_i %int_4
               OpBranchConditional %cond %loop_body %loop_merge

  %loop_body = OpLabel
     %curr_f = OpConvertSToF %float %curr_idx_i
          %t = OpFMul %float %curr_f %float_0_25
 %one_minus_t = OpFSub %float %float_1 %t

; pos = mix(p1, p2, t)
      %pos_a = OpVectorTimesScalar %v4float %p1 %one_minus_t
      %pos_b = OpVectorTimesScalar %v4float %p2 %t
        %pos = OpFAdd %v4float %pos_a %pos_b

; tangent = mix(t1, t2, t)
     %tang_a = OpVectorTimesScalar %v4float %t1_full %one_minus_t
     %tang_b = OpVectorTimesScalar %v4float %t2_full %t
       %tang = OpFAdd %v4float %tang_a %tang_b

; Perpendicular from tangent.xy
    %tang_xy = OpVectorShuffle %v2float %tang %tang 0 1
   %tang_len = OpExtInst %float %1 Length %tang_xy
  %tang_norm = OpFDiv %v2float %tang_xy %tang_len

    %tang_nx = OpCompositeExtract %float %tang_norm 0
    %tang_ny = OpCompositeExtract %float %tang_norm 1
%neg_tang_ny = OpFNegate %float %tang_ny

; perp_v2 = vec2(-tang_ny, tang_nx) * 0.01
%perp_x_sc = OpFMul %float %neg_tang_ny %float_0_01
%perp_y_sc = OpFMul %float %tang_nx %float_0_01

; perp_v4_pos = (perp_x_sc, perp_y_sc, 0, 0)
%perp_v4_pos = OpCompositeConstruct %v4float %perp_x_sc %perp_y_sc %float_0 %float_0
%perp_v4_neg = OpFNegate %v4float %perp_v4_pos

; outColor = mix(c1, c2, t)
      %col_a = OpVectorTimesScalar %v3float %c1 %one_minus_t
      %col_b = OpVectorTimesScalar %v3float %c2 %t
   %curr_col = OpFAdd %v3float %col_a %col_b

; Emit Vertex 1: pos + perp
    %v1_pos = OpFAdd %v4float %pos %perp_v4_pos
               OpStore %outColor %curr_col
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %ptr_out_pos %v1_pos
               OpEmitVertex

; Emit Vertex 2: pos - perp
    %v2_pos = OpFAdd %v4float %pos %perp_v4_neg
               OpStore %outColor %curr_col
               OpStore %ptr_out_pos %v2_pos
               OpEmitVertex

; Increment index
   %next_idx = OpIAdd %int %curr_idx_i %int_1
               OpStore %idx %next_idx
               OpBranch %loop_cond

 %loop_merge = OpLabel
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
