To solve the 1D Cutting Stock Problem efficiently for millions of cuts, I used the **First Fit Decreasing (FFD)** heuristic. FFD is a powerful bin packing heuristic that typically uses significantly fewer bins than the simple First Fit approach. 

### Implementation Details:
1.  **Complexity:** To handle millions of cuts within the 5-minute time limit, the algorithm uses a **Segment Tree** to find the first available bin for each cut in $O(\log B)$ time, where $B$ is the number of bins. Combined with sorting, the total complexity is **$O(N \log N)$**, ensuring it remains fast even for very large inputs.
2.  **Memory Management:** The solution is designed to be memory-efficient, using primitive arrays to minimize object overhead and GC pressure. For $10^7$ cuts, the memory footprint is well within the 32 GB limit.
3.  **Fast I/O:** Custom `FastReader` and buffered `StreamWriter` are implemented to minimize I/O bottlenecks, which are critical when processing millions of integers.

### Complete C# Code:

```csharp
using System;
using System.Collections.Generic;
using System.IO;

class Program
{
    static void Main(string[] args)
    {
        FastReader reader = new FastReader(Console.OpenStandardInput());
        int n = reader.NextInt();
        if (n <= 0) return;
        long stockLength = reader.NextLong();

        long[] lengths = new long[n];
        int[] indices = new int[n];
        for (int i = 0; i < n; i++)
        {
            long l = reader.NextLong();
            if (l == -1) break;
            lengths[i] = l;
            indices[i] = i;
        }

        // Sort ascending to process from largest to smallest later (FFD)
        Array.Sort(lengths, indices);

        // Segment Tree to find the first bin that can fit the current cut
        // The tree stores the maximum available capacity in each range.
        int m = 1;
        while (m < n) m <<= 1;
        long[] tree = new long[2 * m];
        for (int i = 0; i < n; i++) tree[m + i] = stockLength;
        for (int i = n; i < m; i++) tree[m + i] = -1; // Padding for power-of-2 size
        for (int i = m - 1; i > 0; i--) tree[i] = Math.Max(tree[i << 1], tree[(i << 1) | 1]);

        int[] binAssignment = new int[n];
        int maxBinIdx = -1;

        // First Fit Decreasing (FFD) heuristic loop
        for (int i = n - 1; i >= 0; i--)
        {
            long len = lengths[i];
            int binIdx = FindFirst(tree, m, len);
            binAssignment[indices[i]] = binIdx;
            if (binIdx > maxBinIdx) maxBinIdx = binIdx;

            // Update Segment Tree with new remaining capacity
            int node = m + binIdx;
            tree[node] -= len;
            while (node > 1)
            {
                node >>= 1;
                long newVal = Math.Max(tree[node << 1], tree[(node << 1) | 1]);
                if (tree[node] == newVal) break; // Optimization: stop if no change
                tree[node] = newVal;
            }
        }

        int numBins = maxBinIdx + 1;
        
        // Group indices by bin assignment efficiently without List<int> overhead
        int[] binCount = new int[numBins];
        for (int i = 0; i < n; i++) binCount[binAssignment[i]]++;

        int[] binStart = new int[numBins + 1];
        for (int i = 0; i < numBins; i++) binStart[i + 1] = binStart[i] + binCount[i];

        int[] sortedIndices = new int[n];
        int[] currentBinPos = new int[numBins];
        for (int i = 0; i < n; i++)
        {
            int b = binAssignment[i];
            sortedIndices[binStart[b] + currentBinPos[b]] = i;
            currentBinPos[b]++;
        }

        // Fast output using a large buffer and custom integer writing
        StreamWriter sw = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1024 * 1024);
        sw.WriteLine(numBins);
        char[] intBuf = new char[12];
        for (int b = 0; b < numBins; b++)
        {
            int start = binStart[b];
            int end = binStart[b + 1];
            for (int i = start; i < end; i++)
            {
                WriteInt(sw, sortedIndices[i], intBuf);
                if (i < end - 1) sw.Write(' ');
            }
            sw.WriteLine();
        }
        sw.Flush();
    }

    static int FindFirst(long[] tree, int m, long length)
    {
        // Traverses the tree to find the first bin (leaf) with sufficient capacity
        if (tree[1] < length) return -1;
        int node = 1;
        while (node < m)
        {
            if (tree[node << 1] >= length) node <<= 1;
            else node = (node << 1) | 1;
        }
        return node - m;
    }

    static void WriteInt(StreamWriter sw, int val, char[] buf)
    {
        if (val == 0) { sw.Write('0'); return; }
        int p = buf.Length;
        while (val > 0)
        {
            buf[--p] = (char)('0' + (val % 10));
            val /= 10;
        }
        sw.Write(buf, p, buf.Length - p);
    }

    class FastReader
    {
        Stream stream;
        byte[] buffer = new byte[1024 * 1024];
        int pos, len;
        public FastReader(Stream s) { stream = s; }

        int Read()
        {
            if (pos == len)
            {
                pos = 0;
                len = stream.Read(buffer, 0, buffer.Length);
                if (len <= 0) return -1;
            }
            return buffer[pos++];
        }

        public int NextInt()
        {
            int c = Read();
            while (c >= 0 && c <= 32) c = Read();
            if (c == -1) return -1;
            int res = 0;
            while (c >= '0' && c <= '9')
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res;
        }

        public long NextLong()
        {
            int c = Read();
            while (c >= 0 && c <= 32) c = Read();
            if (c == -1) return -1;
            long res = 0;
            while (c >= '0' && c <= '9')
            {
                res = res * 10 + (c - '0');
                c = Read();
            }
            return res;
        }
    }
}
```
