To solve the Maximum Clique problem efficiently across varying graph sizes and densities, I have implemented a parallelized branch-and-bound solver in C#. 

### Key Features:
1.  **Degeneracy Ordering:** The graph is pre-processed to compute a degeneracy ordering. Processing vertices in reverse degeneracy order is a highly effective heuristic for Maximum Clique solvers, as it often finds the largest cliques early and allows for aggressive pruning.
2.  **Parallel Search:** The top-level search is parallelized using `Parallel.For`, leveraging all 12 available cores to explore different branches of the search tree simultaneously.
3.  **Hybrid Data Representation:** 
    *   For smaller or moderately sized graphs (up to 25,000 vertices), the solver builds a bitset-based adjacency matrix (`ulong[][]`). This allows for extremely fast set intersections and connectivity checks.
    *   For very large graphs (up to 1,000,000 vertices), it uses memory-efficient adjacency lists (sorted `int[]`) and performs intersections using a two-pointer approach.
4.  **Pruning Techniques:**
    *   **Greedy Initialization:** A fast greedy heuristic finds a large initial clique to set a high lower bound for pruning.
    *   **Size Pruning:** The search prunes any branch where the current clique plus the number of potential candidate vertices cannot exceed the best found so far.
    *   **Coloring Bound:** For smaller candidate sets, the solver uses a greedy coloring bound. Since a clique cannot contain more vertices than there are colors in a valid coloring of its induced subgraph, this provides a stronger pruning condition than simple counting.
5.  **Performance Optimization:** A custom `FastReader` class is used to handle large-scale input (millions of edges) efficiently, and the search incorporates a time-out mechanism to ensure the program terminates within the 5-minute limit.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;
using System.Threading;
using System.Threading.Tasks;
using System.IO;
using System.Runtime.CompilerServices;
using System.Numerics;

public class MaxCliqueSolver {
    static int V, E;
    static int[][] adj;
    static ulong[][] bitAdj;
    static int[] maxClique = new int[0];
    static int maxCliqueSize = 0;
    static object lockObj = new object();
    static DateTime startTime;
    static bool timeOut = false;
    static int[] rank;
    static int[] order;

    class FastReader {
        Stream s;
        byte[] buf = new byte[1 << 16];
        int pos, len;
        public FastReader(Stream s) { this.s = s; }
        int ReadByte() {
            if (pos >= len) {
                pos = 0;
                len = s.Read(buf, 0, buf.Length);
                if (len <= 0) return -1;
            }
            return buf[pos++];
        }
        public int NextInt() {
            int c = ReadByte();
            while (c >= 0 && c <= 32) c = ReadByte();
            if (c == -1) return -1;
            int res = 0;
            while (c > 32) {
                res = res * 10 + (c - '0');
                c = ReadByte();
            }
            return res;
        }
    }

    public static void Main() {
        startTime = DateTime.Now;
        FastReader fr = new FastReader(Console.OpenStandardInput());
        V = fr.NextInt();
        E = fr.NextInt();
        if (V <= 0) {
            Console.WriteLine(0);
            return;
        }

        List<int>[] adjList = new List<int>[V];
        for (int i = 0; i < V; i++) adjList[i] = new List<int>();

        for (int i = 0; i < E; i++) {
            int u = fr.NextInt();
            int v = fr.NextInt();
            if (u < 0 || v < 0 || u >= V || v >= V || u == v) continue;
            adjList[u].Add(v);
            adjList[v].Add(u);
        }

        adj = new int[V][];
        int[] degrees = new int[V];
        for (int i = 0; i < V; i++) {
            adj[i] = adjList[i].Distinct().ToArray();
            Array.Sort(adj[i]);
            degrees[i] = adj[i].Length;
        }
        adjList = null; 

        Greedy(degrees);

        order = GetDegeneracyOrder(degrees);
        rank = new int[V];
        for (int i = 0; i < V; i++) rank[order[i]] = i;

        if (V <= 25000) {
            BuildBitAdj();
        }

        ParallelOptions options = new ParallelOptions { MaxDegreeOfParallelism = 12 };
        try {
            Parallel.For(0, V, options, (i, state) => {
                if (timeOut || (DateTime.Now - startTime).TotalSeconds > 290) {
                    timeOut = true;
                    state.Stop();
                    return;
                }

                int v = order[V - 1 - i];
                List<int> P = new List<int>();
                foreach (int neighbor in adj[v]) {
                    if (rank[neighbor] > rank[v]) {
                        P.Add(neighbor);
                    }
                }

                if (P.Count + 1 > maxCliqueSize) {
                    List<int> R = new List<int> { v };
                    if (bitAdj != null) {
                        SearchBit(P, R);
                    } else {
                        SearchList(P, R);
                    }
                }
            });
        } catch (Exception) { }

        Console.WriteLine(maxCliqueSize);
        Array.Sort(maxClique);
        Console.WriteLine(string.Join(" ", maxClique));
    }

    static void Greedy(int[] degrees) {
        int[] nodes = Enumerable.Range(0, V).ToArray();
        Array.Sort(nodes, (a, b) => degrees[b].CompareTo(degrees[a]));

        for (int start = 0; start < Math.Min(V, 20); start++) {
            List<int> current = new List<int> { nodes[start] };
            for (int i = start + 1; i < Math.Min(V, start + 10000); i++) {
                int v = nodes[i];
                bool ok = true;
                foreach (int u in current) {
                    if (Array.BinarySearch(adj[v], u) < 0) { ok = false; break; }
                }
                if (ok) current.Add(v);
                if ((DateTime.Now - startTime).TotalSeconds > 5) break;
            }
            lock (lockObj) {
                if (current.Count > maxCliqueSize) {
                    maxCliqueSize = current.Count;
                    maxClique = current.ToArray();
                }
            }
            if ((DateTime.Now - startTime).TotalSeconds > 10) break;
        }
    }

    static int[] GetDegeneracyOrder(int[] degrees) {
        int n = V;
        int[] d = (int[])degrees.Clone();
        bool[] removed = new bool[n];
        int[] res = new int[n];
        
        List<int>[] bins = new List<int>[n];
        for (int i = 0; i < n; i++) bins[i] = new List<int>();
        for (int i = 0; i < n; i++) bins[d[i]].Add(i);
        
        int minD = 0;
        for (int i = 0; i < n; i++) {
            while (minD < n && bins[minD].Count == 0) minD++;
            if (minD == n) break;
            
            int v = -1;
            while (v == -1 && minD < n) {
                int lastIdx = bins[minD].Count - 1;
                int candidate = bins[minD][lastIdx];
                bins[minD].RemoveAt(lastIdx);
                if (!removed[candidate]) v = candidate;
                else if (bins[minD].Count == 0) minD++;
            }
            if (v == -1) break;
            
            removed[v] = true;
            res[i] = v;
            foreach (int u in adj[v]) {
                if (!removed[u]) {
                    d[u]--;
                    bins[d[u]].Add(u);
                    if (d[u] < minD) minD = d[u];
                }
            }
        }
        return res;
    }

    static void BuildBitAdj() {
        int words = (V + 63) / 64;
        bitAdj = new ulong[V][];
        for (int i = 0; i < V; i++) {
            bitAdj[i] = new ulong[words];
            foreach (int neighbor in adj[i]) {
                bitAdj[i][neighbor >> 6] |= (1UL << (neighbor & 63));
            }
        }
    }

    static void SearchList(List<int> P, List<int> R) {
        if (timeOut) return;
        if (P.Count == 0) {
            lock (lockObj) {
                if (R.Count > maxCliqueSize) {
                    maxCliqueSize = R.Count;
                    maxClique = R.ToArray();
                }
            }
            return;
        }

        if (R.Count + P.Count <= maxCliqueSize) return;

        P.Sort((a, b) => rank[b].CompareTo(rank[a]));

        while (P.Count > 0) {
            if (R.Count + P.Count <= maxCliqueSize) return;
            if (timeOut) return;

            int v = P[P.Count - 1];
            P.RemoveAt(P.Count - 1);

            List<int> newP = Intersect(adj[v], P);
            if (R.Count + newP.Count + 1 > maxCliqueSize) {
                R.Add(v);
                SearchList(newP, R);
                R.RemoveAt(R.Count - 1);
            }
        }
    }

    static void SearchBit(List<int> P, List<int> R) {
        if (timeOut) return;
        if (P.Count == 0) {
            lock (lockObj) {
                if (R.Count > maxCliqueSize) {
                    maxCliqueSize = R.Count;
                    maxClique = R.ToArray();
                }
            }
            return;
        }

        int bound = (P.Count < 500) ? GetColorBound(P) : P.Count;
        if (R.Count + bound <= maxCliqueSize) return;

        P.Sort((a, b) => rank[b].CompareTo(rank[a]));

        while (P.Count > 0) {
            if (R.Count + P.Count <= maxCliqueSize) return;
            if (timeOut) return;

            int v = P[P.Count - 1];
            P.RemoveAt(P.Count - 1);

            List<int> newP = IntersectBit(v, P);
            if (R.Count + newP.Count + 1 > maxCliqueSize) {
                R.Add(v);
                SearchBit(newP, R);
                R.RemoveAt(R.Count - 1);
            }
        }
    }

    static List<int> Intersect(int[] sortedAdj, List<int> sortedP) {
        List<int> res = new List<int>();
        int i = 0, j = 0;
        while (i < sortedAdj.Length && j < sortedP.Count) {
            if (sortedAdj[i] == sortedP[j]) {
                res.Add(sortedAdj[i]);
                i++; j++;
            } else if (sortedAdj[i] < sortedP[j]) {
                i++;
            } else {
                j++;
            }
        }
        return res;
    }

    static List<int> IntersectBit(int v, List<int> P) {
        List<int> res = new List<int>();
        ulong[] bitsV = bitAdj[v];
        foreach (int u in P) {
            if ((bitsV[u >> 6] & (1UL << (u & 63))) != 0) {
                res.Add(u);
            }
        }
        return res;
    }

    static int GetColorBound(List<int> P) {
        int n = P.Count;
        int[] colors = new int[n];
        int maxColor = 0;
        for (int i = 0; i < n; i++) {
            int v = P[i];
            ulong usedMask = 0;
            for (int j = 0; j < i; j++) {
                if (colors[j] < 64) {
                    int u = P[j];
                    if ((bitAdj[v][u >> 6] & (1UL << (u & 63))) != 0) {
                        usedMask |= (1UL << colors[j]);
                    }
                }
            }
            int c = BitOperations.TrailingZeroCount(~usedMask);
            if (c >= 64) return P.Count;
            colors[i] = c;
            if (c > maxColor) maxColor = c;
        }
        return maxColor + 1;
    }
}
```
