using System;
using System.Collections.Generic;
using System.Linq;

#nullable disable

namespace DFVS {
    class Graph {
        public HashSet<int> Nodes = new HashSet<int>();
        public Dictionary<int, HashSet<int>> Adj = new Dictionary<int, HashSet<int>>();
        public Dictionary<int, HashSet<int>> RevAdj = new Dictionary<int, HashSet<int>>();

        public Graph() { }

        public Graph Clone() {
            var g = new Graph();
            g.Nodes = new HashSet<int>(Nodes);
            foreach (var kvp in Adj) g.Adj[kvp.Key] = new HashSet<int>(kvp.Value);
            foreach (var kvp in RevAdj) g.RevAdj[kvp.Key] = new HashSet<int>(kvp.Value);
            return g;
        }

        public bool AddEdge(int u, int v) {
            if (!Adj.ContainsKey(u)) Adj[u] = new HashSet<int>();
            if (!RevAdj.ContainsKey(v)) RevAdj[v] = new HashSet<int>();
            
            bool added = Adj[u].Add(v);
            RevAdj[v].Add(u);
            Nodes.Add(u);
            Nodes.Add(v);
            return added;
        }

        public void RemoveVertex(int v) {
            if (!Nodes.Contains(v)) return;
            if (Adj.ContainsKey(v)) {
                foreach (var neighbor in Adj[v]) {
                    if (neighbor != v) RevAdj[neighbor].Remove(v);
                }
                Adj.Remove(v);
            }
            if (RevAdj.ContainsKey(v)) {
                foreach (var neighbor in RevAdj[v]) {
                    if (neighbor != v) Adj[neighbor].Remove(v);
                }
                RevAdj.Remove(v);
            }
            Nodes.Remove(v);
        }
    }

    class Program {
        static void Main() {
            var inputRaw = Console.In.ReadToEnd();
            if (string.IsNullOrWhiteSpace(inputRaw)) return;
            var input = inputRaw.Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (input.Length < 2) return;
            int n = int.Parse(input[0]);
            int m = int.Parse(input[1]);

            var g = new Graph();
            for (int i = 0; i < n; i++) g.Nodes.Add(i);

            for (int i = 0; i < m; i++) {
                int idx = 2 + i * 2;
                if (idx + 1 >= input.Length) break;
                int u = int.Parse(input[idx]);
                int v = int.Parse(input[idx + 1]);
                g.AddEdge(u, v);
            }

            var result = Solve(g);
            Console.WriteLine(result.Count);
            Console.WriteLine(string.Join(" ", result));
        }

        static List<int> Solve(Graph g) {
            var fvs = new List<int>();
            Reduce(g, fvs);
            if (g.Nodes.Count == 0) return fvs;

            var sccs = GetSCCs(g);
            if (sccs.Count > 1) {
                foreach (var scc in sccs) {
                    if (scc.Count == 1) {
                        int v = scc[0];
                        if (g.Adj.ContainsKey(v) && g.Adj[v].Contains(v)) fvs.Add(v);
                        continue;
                    }
                    var sub = ExtractSubgraph(g, scc);
                    fvs.AddRange(Solve(sub));
                }
                return fvs;
            }

            // Single SCC
            for (int k = 0; ; k++) {
                var res = Branch(g, k);
                if (res != null) {
                    fvs.AddRange(res);
                    return fvs;
                }
            }
        }

        static void Reduce(Graph g, List<int> fvs) {
            var dirty = new Queue<int>(g.Nodes);
            var inDirty = new HashSet<int>(g.Nodes);

            while (dirty.Count > 0) {
                int v = dirty.Dequeue();
                inDirty.Remove(v);
                if (!g.Nodes.Contains(v)) continue;

                // 1. Self-loop
                if (g.Adj.ContainsKey(v) && g.Adj[v].Contains(v)) {
                    fvs.Add(v);
                    var neighbors = GetNeighbors(g, v);
                    g.RemoveVertex(v);
                    MarkDirty(neighbors, g, dirty, inDirty);
                    continue;
                }

                // 2. Sink/Source
                int inD = g.RevAdj.ContainsKey(v) ? g.RevAdj[v].Count : 0;
                int outD = g.Adj.ContainsKey(v) ? g.Adj[v].Count : 0;
                if (inD == 0 || outD == 0) {
                    var neighbors = GetNeighbors(g, v);
                    g.RemoveVertex(v);
                    MarkDirty(neighbors, g, dirty, inDirty);
                    continue;
                }

                // 3. IN-1
                if (inD == 1) {
                    int u = g.RevAdj[v].First();
                    if (u != v) {
                        var outNeighbors = g.Adj.ContainsKey(v) ? g.Adj[v].ToList() : new List<int>();
                        bool anyNew = false;
                        foreach (var w in outNeighbors) {
                            if (w != v) {
                                if (g.AddEdge(u, w)) anyNew = true;
                            }
                        }
                        g.RemoveVertex(v);
                        if (!inDirty.Contains(u)) { dirty.Enqueue(u); inDirty.Add(u); }
                        if (anyNew) {
                            foreach(var w in outNeighbors) if (g.Nodes.Contains(w) && !inDirty.Contains(w)) { dirty.Enqueue(w); inDirty.Add(w); }
                        }
                        continue;
                    }
                }

                // 4. OUT-1
                if (outD == 1) {
                    int w = g.Adj[v].First();
                    if (w != v) {
                        var inNeighbors = g.RevAdj.ContainsKey(v) ? g.RevAdj[v].ToList() : new List<int>();
                        bool anyNew = false;
                        foreach (var u in inNeighbors) {
                            if (u != v) {
                                if (g.AddEdge(u, w)) anyNew = true;
                            }
                        }
                        g.RemoveVertex(v);
                        if (!inDirty.Contains(w)) { dirty.Enqueue(w); inDirty.Add(w); }
                        if (anyNew) {
                            foreach(var u in inNeighbors) if (g.Nodes.Contains(u) && !inDirty.Contains(u)) { dirty.Enqueue(u); inDirty.Add(u); }
                        }
                        continue;
                    }
                }
            }
        }

        static List<int> GetNeighbors(Graph g, int v) {
            var res = new HashSet<int>();
            if (g.Adj.ContainsKey(v)) foreach (var n in g.Adj[v]) if (n != v) res.Add(n);
            if (g.RevAdj.ContainsKey(v)) foreach (var n in g.RevAdj[v]) if (n != v) res.Add(n);
            return res.ToList();
        }

        static void MarkDirty(List<int> nodes, Graph g, Queue<int> dirty, HashSet<int> inDirty) {
            foreach (var n in nodes) {
                if (g.Nodes.Contains(n) && !inDirty.Contains(n)) {
                    dirty.Enqueue(n);
                    inDirty.Add(n);
                }
            }
        }

        static List<int> Branch(Graph g, int k) {
            var fvs = new List<int>();
            var gCopy = g.Clone();
            Reduce(gCopy, fvs);
            if (fvs.Count > k) return null;
            if (gCopy.Nodes.Count == 0) return fvs;

            int remainingK = k - fvs.Count;
            if (LowerBound(gCopy) > remainingK) return null;

            var sccs = GetSCCs(gCopy);
            if (sccs.Count > 1) {
                var totalFvs = new List<int>(fvs);
                int totalMin = 0;
                foreach (var scc in sccs) {
                    var sub = ExtractSubgraph(gCopy, scc);
                    var res = Solve(sub);
                    totalMin += res.Count;
                    totalFvs.AddRange(res);
                }
                if (totalMin <= remainingK) return totalFvs;
                return null;
            }

            var cycle = FindSmallCycle(gCopy);
            if (cycle == null) return fvs;
            if (remainingK == 0) return null;

            var sortedCycle = cycle.OrderByDescending(v => (gCopy.Adj.ContainsKey(v) ? gCopy.Adj[v].Count : 0) * (gCopy.RevAdj.ContainsKey(v) ? gCopy.RevAdj[v].Count : 0)).ToList();

            foreach (var v in sortedCycle) {
                var nextG = gCopy.Clone();
                nextG.RemoveVertex(v);
                var res = Branch(nextG, remainingK - 1);
                if (res != null) {
                    var combined = new List<int>(fvs);
                    combined.Add(v);
                    combined.AddRange(res);
                    return combined;
                }
            }

            return null;
        }

        static int LowerBound(Graph g) {
            int count = 0;
            var removed = new HashSet<int>();
            var visited = new HashSet<int>();
            foreach (var node in g.Nodes) {
                if (removed.Contains(node) || visited.Contains(node)) continue;
                var cycle = DfsCycle(node, g, visited, removed);
                if (cycle != null) {
                    count++;
                    foreach (var v in cycle) removed.Add(v);
                }
            }
            return count;
        }

        static List<int> FindSmallCycle(Graph g) {
            foreach (var u in g.Nodes) {
                if (g.Adj.ContainsKey(u)) {
                    foreach (var v in g.Adj[u]) {
                        if (u != v && g.Adj.ContainsKey(v) && g.Adj[v].Contains(u)) return new List<int> { u, v };
                    }
                }
            }
            var nodes = g.Nodes.Take(50).ToList();
            int bestLen = int.MaxValue;
            List<int> bestCycle = null;
            foreach (var start in nodes) {
                var q = new Queue<int>();
                var parent = new Dictionary<int, int>();
                var depth = new Dictionary<int, int>();
                q.Enqueue(start);
                parent[start] = -1;
                depth[start] = 0;
                while (q.Count > 0) {
                    int u = q.Dequeue();
                    int d = depth[u];
                    if (d >= bestLen - 1) continue;
                    if (g.Adj.ContainsKey(u)) {
                        foreach (var v in g.Adj[u]) {
                            if (v == start) {
                                var cycle = new List<int>();
                                int curr = u;
                                while (curr != -1) { cycle.Add(curr); curr = parent[curr]; }
                                if (cycle.Count < bestLen) { bestLen = cycle.Count; bestCycle = cycle; }
                                goto next_start;
                            }
                            if (!depth.ContainsKey(v)) {
                                depth[v] = d + 1;
                                parent[v] = u;
                                q.Enqueue(v);
                            }
                        }
                    }
                }
                next_start:;
                if (bestLen == 2) return bestCycle;
            }
            if (bestCycle != null) return bestCycle;
            
            var visited = new HashSet<int>();
            foreach (var node in g.Nodes) {
                if (!visited.Contains(node)) {
                    var res = DfsCycle(node, g, visited, null);
                    if (res != null) return res;
                }
            }
            return null;
        }

        static List<int> DfsCycle(int startNode, Graph g, HashSet<int> globalVisited, HashSet<int> removed) {
            var stack = new Stack<(int u, List<int> adj, int edgeIdx)>();
            var currentPath = new List<int>();
            var onPath = new HashSet<int>();
            var localVisited = new HashSet<int>();

            stack.Push((startNode, g.Adj.ContainsKey(startNode) ? g.Adj[startNode].ToList() : null, 0));

            while (stack.Count > 0) {
                var (u, adj, edgeIdx) = stack.Pop();

                if (edgeIdx == 0) {
                    if (localVisited.Contains(u)) continue;
                    localVisited.Add(u);
                    globalVisited.Add(u);
                    currentPath.Add(u);
                    onPath.Add(u);
                }

                bool pushed = false;
                if (adj != null) {
                    for (int i = edgeIdx; i < adj.Count; i++) {
                        int v = adj[i];
                        if (removed != null && removed.Contains(v)) continue;
                        
                        if (onPath.Contains(v)) {
                            var cycle = new List<int>();
                            bool found = false;
                            foreach (var n in currentPath) { if (n == v) found = true; if (found) cycle.Add(n); }
                            return cycle;
                        }
                        
                        if (!localVisited.Contains(v)) {
                            stack.Push((u, adj, i + 1));
                            stack.Push((v, g.Adj.ContainsKey(v) ? g.Adj[v].ToList() : null, 0));
                            pushed = true;
                            break;
                        }
                    }
                }

                if (!pushed) {
                    onPath.Remove(u);
                    currentPath.RemoveAt(currentPath.Count - 1);
                }
            }
            return null;
        }

        static List<List<int>> GetSCCs(Graph g) {
            var sccs = new List<List<int>>();
            var index = new Dictionary<int, int>();
            var lowlink = new Dictionary<int, int>();
            var stack = new Stack<int>();
            var onStack = new HashSet<int>();
            int time = 0;

            foreach (var startNode in g.Nodes) {
                if (index.ContainsKey(startNode)) continue;

                var work = new Stack<(int node, List<int> adj, int edgeIdx)>();
                work.Push((startNode, g.Adj.ContainsKey(startNode) ? g.Adj[startNode].ToList() : null, 0));

                while (work.Count > 0) {
                    var (u, adj, edgeIdx) = work.Pop();
                    if (edgeIdx == 0) {
                        index[u] = lowlink[u] = time++;
                        stack.Push(u);
                        onStack.Add(u);
                    }

                    bool pushed = false;
                    if (adj != null) {
                        for (int i = edgeIdx; i < adj.Count; i++) {
                            int v = adj[i];
                            if (!index.ContainsKey(v)) {
                                work.Push((u, adj, i + 1));
                                work.Push((v, g.Adj.ContainsKey(v) ? g.Adj[v].ToList() : null, 0));
                                pushed = true;
                                break;
                            } else if (onStack.Contains(v)) {
                                lowlink[u] = Math.Min(lowlink[u], index[v]);
                            }
                        }
                    }

                    if (pushed) continue;

                    if (lowlink[u] == index[u]) {
                        var scc = new List<int>();
                        int w;
                        do {
                            w = stack.Pop();
                            onStack.Remove(w);
                            scc.Add(w);
                        } while (w != u);
                        sccs.Add(scc);
                    }

                    if (work.Count > 0) {
                        var (parent, _, _) = work.Peek();
                        lowlink[parent] = Math.Min(lowlink[parent], lowlink[u]);
                    }
                }
            }
            return sccs;
        }

        static Graph ExtractSubgraph(Graph g, List<int> nodes) {
            var sub = new Graph();
            var nodeSet = new HashSet<int>(nodes);
            foreach (var v in nodes) sub.Nodes.Add(v);
            foreach (var u in nodes) {
                if (g.Adj.ContainsKey(u)) {
                    foreach (var v in g.Adj[u]) if (nodeSet.Contains(v)) sub.AddEdge(u, v);
                }
            }
            return sub;
        }
    }
