using System;
using System.Collections.Generic;
using System.Linq;
using System.IO;

namespace DFVS
{
    class Program
    {
        static void Main(string[] args)
        {
            var reader = new FastReader(Console.OpenStandardInput());
            int n = reader.NextInt();
            if (n < 0) return;
            int m = reader.NextInt();
            if (m < 0) return;

            var adj = new HashSet<int>[n];
            var rev = new HashSet<int>[n];
            for (int i = 0; i < n; i++)
            {
                adj[i] = new HashSet<int>();
                rev[i] = new HashSet<int>();
            }

            for (int i = 0; i < m; i++)
            {
                int u = reader.NextInt();
                int v = reader.NextInt();
                if (u < 0 || u >= n || v < 0 || v >= n) continue;
                if (adj[u].Add(v))
                {
                    rev[v].Add(u);
                }
            }

            var solver = new Solver(n, adj, rev);
            var result = solver.Solve();

            using (var writer = new StreamWriter(Console.OpenStandardOutput()))
            {
                writer.WriteLine(result.Count);
                if (result.Count > 0)
                {
                    for (int i = 0; i < result.Count; i++)
                    {
                        writer.Write(result[i]);
                        if (i < result.Count - 1) writer.Write(' ');
                    }
                    writer.WriteLine();
                }
                else
                {
                    writer.WriteLine();
                }
            }
        }
    }

    class FastReader
    {
        private Stream _stream;
        private byte[] _buf = new byte[65536];
        private int _pos, _len;

        public FastReader(Stream stream) { _stream = stream; }

        private byte Read()
        {
            if (_pos >= _len)
            {
                _pos = 0;
                _len = _stream.Read(_buf, 0, _buf.Length);
                if (_len <= 0) return 0;
            }
            return _buf[_pos++];
        }

        public int NextInt()
        {
            int res = 0;
            byte b = Read();
            while (b != 0 && (b < '0' || b > '9')) b = Read();
            if (b == 0) return -1;
            while (b >= '0' && b <= '9')
            {
                res = res * 10 + (b - '0');
                b = Read();
            }
            return res;
        }
    }

    class Solver
    {
        private int _n;
        private HashSet<int>[] _adj;
        private HashSet<int>[] _rev;

        public Solver(int n, HashSet<int>[] adj, HashSet<int>[] rev)
        {
            _n = n;
            _adj = adj;
            _rev = rev;
        }

        public List<int> Solve()
        {
            var active = new HashSet<int>();
            for (int i = 0; i < _n; i++)
            {
                if (_adj[i].Count > 0 || _rev[i].Count > 0)
                    active.Add(i);
            }

            return SolveRecursive(active, _adj, _rev);
        }

        private List<int> SolveRecursive(HashSet<int> active, HashSet<int>[] adj, HashSet<int>[] rev)
        {
            var sccs = GetSCCs(active, adj);
            var result = new List<int>();
            foreach (var scc in sccs)
            {
                if (scc.Count == 0) continue;
                if (scc.Count == 1)
                {
                    int v = scc[0];
                    if (adj[v].Contains(v)) result.Add(v);
                    continue;
                }

                var sccActive = new HashSet<int>(scc);
                var sccFvs = SolveSCC(sccActive, adj, rev);
                result.AddRange(sccFvs);
            }
            return result;
        }

        private List<int> SolveSCC(HashSet<int> active, HashSet<int>[] adj, HashSet<int>[] rev)
        {
            var currentFvs = new List<int>();
            Reduce(active, currentFvs, adj, rev);
            if (active.Count == 0) return currentFvs;

            int lb = LowerBound(active, adj);
            for (int k = lb; ; k++)
            {
                var res = Search(new Graph(active, adj, rev), k);
                if (res != null)
                {
                    currentFvs.AddRange(res);
                    return currentFvs;
                }
            }
        }

        private List<int> Search(Graph g, int k)
        {
            var fvs = new List<int>();
            Reduce(g.Active, fvs, g.Adj, g.Rev);
            k -= fvs.Count;

            if (g.Active.Count == 0) return fvs;
            if (k < 0) return null;
            if (k < LowerBound(g.Active, g.Adj)) return null;

            var cycle = GetShortCycle(g.Active, g.Adj, g.Rev);
            if (cycle == null) return fvs;

            foreach (var v in cycle)
            {
                var gNext = g.Clone();
                RemoveVertex(v, gNext.Active, gNext.Adj, gNext.Rev);
                var res = Search(gNext, k - 1);
                if (res != null)
                {
                    fvs.Add(v);
                    fvs.AddRange(res);
                    return fvs;
                }
            }

            return null;
        }

        private void Reduce(HashSet<int> active, List<int> fvs, HashSet<int>[] adj, HashSet<int>[] rev)
        {
            var queue = new Queue<int>(active);
            var inQueue = new HashSet<int>(active);

            while (queue.Count > 0)
            {
                int v = queue.Dequeue();
                inQueue.Remove(v);
                if (!active.Contains(v)) continue;

                if (adj[v].Contains(v))
                {
                    fvs.Add(v);
                    var neighbors = adj[v].Concat(rev[v]).ToList();
                    RemoveVertex(v, active, adj, rev);
                    foreach (var n in neighbors) if (active.Contains(n) && inQueue.Add(n)) queue.Enqueue(n);
                    continue;
                }

                if (adj[v].Count == 0 || rev[v].Count == 0)
                {
                    var neighbors = adj[v].Concat(rev[v]).ToList();
                    RemoveVertex(v, active, adj, rev);
                    foreach (var n in neighbors) if (active.Contains(n) && inQueue.Add(n)) queue.Enqueue(n);
                    continue;
                }

                if (rev[v].Count == 1)
                {
                    int u = rev[v].First();
                    if (u != v)
                    {
                        var outs = adj[v].ToList();
                        RemoveVertex(v, active, adj, rev);
                        foreach (var w in outs)
                        {
                            if (adj[u].Add(w)) rev[w].Add(u);
                        }
                        if (inQueue.Add(u)) queue.Enqueue(u);
                        foreach (var w in outs) if (active.Contains(w) && inQueue.Add(w)) queue.Enqueue(w);
                        continue;
                    }
                }

                if (adj[v].Count == 1)
                {
                    int w = adj[v].First();
                    if (w != v)
                    {
                        var ins = rev[v].ToList();
                        RemoveVertex(v, active, adj, rev);
                        foreach (var u in ins)
                        {
                            if (adj[u].Add(w)) rev[w].Add(u);
                        }
                        if (inQueue.Add(w)) queue.Enqueue(w);
                        foreach (var u in ins) if (active.Contains(u) && inQueue.Add(u)) queue.Enqueue(u);
                        continue;
                    }
                }
            }
        }

        private void RemoveVertex(int v, HashSet<int> active, HashSet<int>[] adj, HashSet<int>[] rev)
        {
            foreach (var neighbor in adj[v]) rev[neighbor].Remove(v);
            foreach (var neighbor in rev[v]) adj[neighbor].Remove(v);
            adj[v].Clear();
            rev[v].Clear();
            active.Remove(v);
        }

        private int LowerBound(HashSet<int> active, HashSet<int>[] adj)
        {
            int lb = 0;
            var temp = new HashSet<int>(active);
            foreach (var v in active)
            {
                if (!temp.Contains(v)) continue;
                foreach (var u in adj[v])
                {
                    if (u != v && temp.Contains(u) && adj[u].Contains(v))
                    {
                        lb++;
                        temp.Remove(v);
                        temp.Remove(u);
                        break;
                    }
                }
            }
            return lb;
        }

        private int[] _dist;
        private int[] _distStart;
        private int[] _parent;
        private int _currentStartId = 0;

        private List<int> GetShortCycle(HashSet<int> active, HashSet<int>[] adj, HashSet<int>[] rev)
        {
            int n = adj.Length;
            if (_dist == null || _dist.Length < n)
            {
                _dist = new int[n];
                _distStart = new int[n];
                _parent = new int[n];
            }

            int minLen = int.MaxValue;
            List<int> bestCycle = null;

            var candidates = active.OrderByDescending(v => adj[v].Count + rev[v].Count).Take(50).ToList();
            if (active.Count < 200) candidates = active.ToList();

            foreach (var start in candidates)
            {
                _currentStartId++;
                var queue = new Queue<int>();

                foreach (var next in adj[start])
                {
                    if (active.Contains(next))
                    {
                        if (next == start) return new List<int> { start };
                        _dist[next] = 1;
                        _distStart[next] = _currentStartId;
                        _parent[next] = start;
                        queue.Enqueue(next);
                    }
                }

                while (queue.Count > 0)
                {
                    int u = queue.Dequeue();
                    if (_dist[u] >= minLen) break;

                    foreach (var v in adj[u])
                    {
                        if (!active.Contains(v)) continue;
                        if (v == start)
                        {
                            if (_dist[u] + 1 < minLen)
                            {
                                minLen = _dist[u] + 1;
                                bestCycle = new List<int>();
                                int curr = u;
                                while (curr != start) { bestCycle.Add(curr); curr = _parent[curr]; }
                                bestCycle.Add(start);
                            }
                            goto next_start;
                        }
                        if (_distStart[v] != _currentStartId)
                        {
                            _dist[v] = _dist[u] + 1;
                            _distStart[v] = _currentStartId;
                            _parent[v] = u;
                            queue.Enqueue(v);
                        }
                    }
                }
            next_start:;
                if (minLen <= 2) break;
            }

            if (bestCycle == null && active.Count > 0)
            {
                var remaining = active.Where(v => !candidates.Contains(v)).Take(500);
                foreach (var start in remaining)
                {
                    _currentStartId++;
                    var queue = new Queue<int>();
                    foreach (var next in adj[start]) {
                        if (active.Contains(next)) {
                            if (next == start) return new List<int> { start };
                            _dist[next] = 1; _distStart[next] = _currentStartId; _parent[next] = start;
                            queue.Enqueue(next);
                        }
                    }
                    while (queue.Count > 0) {
                        int u = queue.Dequeue();
                        if (_dist[u] >= minLen) break;
                        foreach (var v in adj[u]) {
                            if (!active.Contains(v)) continue;
                            if (v == start) {
                                if (_dist[u] + 1 < minLen) {
                                    minLen = _dist[u] + 1;
                                    bestCycle = new List<int>();
                                    int curr = u;
                                    while (curr != start) { bestCycle.Add(curr); curr = _parent[curr]; }
                                    bestCycle.Add(start);
                                }
                                goto next_start_2;
                            }
                            if (_distStart[v] != _currentStartId) {
                                _dist[v] = _dist[u] + 1; _distStart[v] = _currentStartId; _parent[v] = u;
                                queue.Enqueue(v);
                            }
                        }
                    }
                    next_start_2:;
                    if (bestCycle != null) break;
                }
            }

            return bestCycle;
        }

        private List<List<int>> GetSCCs(HashSet<int> active, HashSet<int>[] adj)
        {
            int n = adj.Length;
            int[] disc = new int[n];
            int[] low = new int[n];
            bool[] onStack = new bool[n];
            int time = 0;
            var stack = new Stack<int>();
            var result = new List<List<int>>();

            foreach (var v in active)
            {
                if (disc[v] == 0) Tarjan(v, active, adj, disc, low, stack, onStack, ref time, result);
            }
            return result;
        }

        private void Tarjan(int u, HashSet<int> active, HashSet<int>[] adj, int[] disc, int[] low, Stack<int> stack, bool[] onStack, ref int time, List<List<int>> result)
        {
            var work = new Stack<(int u, List<int> neighbors, int idx)>();
            work.Push((u, adj[u].ToList(), 0));

            while (work.Count > 0)
            {
                var current = work.Pop();
                int node = current.u;
                var neighbors = current.neighbors;
                int i = current.idx;

                if (i == 0)
                {
                    disc[node] = low[node] = ++time;
                    stack.Push(node);
                    onStack[node] = true;
                }

                bool pushed = false;
                for (; i < neighbors.Count; i++)
                {
                    int v = neighbors[i];
                    if (!active.Contains(v)) continue;

                    if (disc[v] == 0)
                    {
                        work.Push((node, neighbors, i + 1));
                        work.Push((v, adj[v].ToList(), 0));
                        pushed = true;
                        break;
                    }
                    else if (onStack[v])
                    {
                        low[node] = Math.Min(low[node], disc[v]);
                    }
                }

                if (pushed) continue;

                if (low[node] == disc[node])
                {
                    var scc = new List<int>();
                    while (true)
                    {
                        int v = stack.Pop();
                        onStack[v] = false;
                        scc.Add(v);
                        if (node == v) break;
                    }
                    result.Add(scc);
                }

                if (work.Count > 0)
                {
                    var parentNode = work.Peek().u;
                    low[parentNode] = Math.Min(low[parentNode], low[node]);
                }
            }
        }

        class Graph
        {
            public HashSet<int> Active;
            public HashSet<int>[] Adj;
            public HashSet<int>[] Rev;
            public Graph(HashSet<int> active, HashSet<int>[] adj, HashSet<int>[] rev)
            {
                Active = new HashSet<int>(active);
                int n = adj.Length;
                Adj = new HashSet<int>[n];
                Rev = new HashSet<int>[n];
                foreach (var v in active)
                {
                    Adj[v] = new HashSet<int>(adj[v]);
                    Rev[v] = new HashSet<int>(rev[v]);
                }
                for(int i=0; i<n; i++) {
                    if (Adj[i] == null) { Adj[i] = new HashSet<int>(); Rev[i] = new HashSet<int>(); }
                }
            }
            public Graph Clone()
            {
                var g = (Graph)this.MemberwiseClone();
                g.Active = new HashSet<int>(Active);
                int n = Adj.Length;
                g.Adj = new HashSet<int>[n];
                g.Rev = new HashSet<int>[n];
                foreach (var v in Active)
                {
                    g.Adj[v] = new HashSet<int>(Adj[v]);
                    g.Rev[v] = new HashSet<int>(Rev[v]);
                }
                for(int i=0; i<n; i++) {
                    if (g.Adj[i] == null) { g.Adj[i] = new HashSet<int>(); g.Rev[i] = new HashSet<int>(); }
                }
                return g;
            }
        }
    }
