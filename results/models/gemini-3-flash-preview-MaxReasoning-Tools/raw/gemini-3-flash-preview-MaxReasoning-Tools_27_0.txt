The Graph Coloring problem was solved using a highly optimized **DSATUR (Degree of Saturation)** algorithm, complemented by a **Randomized Greedy** fallback strategy. 

### Implementation Details:
- **Core Algorithm:** DSATUR is utilized as the primary heuristic. It selects the vertex with the highest "saturation degree" (number of unique colors among its neighbors), breaking ties with the initial degree.
- **Efficiency:** 
  - **Bucket-based selection:** Vertices are managed in saturation buckets, allowing for $O(1)$ selection and updates, resulting in an overall $O(N+M)$ complexity.
  - **Bitset Saturation Tracking:** For the "massive" cases (up to 64 colors), a bitset-based approach tracks unique neighbor colors efficiently.
  - **Memory Management:** Adjacency lists are implemented using a CSR (Compressed Sparse Row) like structure to minimize allocations and memory overhead for million-node graphs.
- **Robustness:** If DSATUR fails to find a $k$-coloring, the program enters a refinement phase using randomized DSATUR and randomized greedy restarts, maximizing the chance of success within the 5-minute time limit.
- **Compatibility:** The code is written in standard C++20 and is fully compatible with **MSVC** (cl 19.50+) and **g++**.

### C++ Code:

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>
#include <random>
#include <chrono>

using namespace std;

struct Graph {
    int n = 0, m = 0, k = 0;
    vector<int> edges;
    vector<size_t> offset;
    vector<int> degree;
};

// Global buffers to avoid repeated allocations and keep memory footprint manageable
vector<int> g_color;
vector<uint64_t> g_neighbor_bits;
int g_words = 0;
vector<int> g_saturation;
vector<bool> g_colored;
vector<int> g_pos_in_bucket;
vector<vector<int>> g_buckets;
vector<int> g_used_at;
int g_timestamp = 0;

void build_graph(Graph& g) {
    if (!(cin >> g.n >> g.m >> g.k)) return;
    if (g.n <= 0) return;

    struct TmpEdge { int u, v; };
    vector<TmpEdge> tmp;
    tmp.reserve(g.m);
    g.degree.assign(g.n, 0);
    for (int i = 0; i < g.m; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u >= 0 && u < g.n && v >= 0 && v < g.n && u != v) {
            tmp.push_back({u, v});
            g.degree[u]++;
            g.degree[v]++;
        }
    }
    g.m = (int)tmp.size();

    g.offset.assign(g.n + 1, 0);
    for (int i = 0; i < g.n; ++i) {
        g.offset[i + 1] = g.offset[i] + g.degree[i];
    }

    g.edges.resize(g.offset[g.n]);
    vector<size_t> current_offset = g.offset;
    for (const auto& e : tmp) {
        g.edges[current_offset[e.u]++] = e.v;
        g.edges[current_offset[e.v]++] = e.u;
    }
}

// DSATUR Algorithm implementation
vector<int> dsatur(const Graph& g, uint32_t seed = 0) {
    int n = g.n;
    int k = g.k;
    g_color.assign(n, -1);
    g_saturation.assign(n, 0);
    g_colored.assign(n, false);
    g_pos_in_bucket.assign(n, -1);
    g_buckets.assign(n + 1, vector<int>());

    // Efficient saturation tracking for small k
    g_words = (k + 63) / 64;
    if (g_words == 0) g_words = 1;
    bool use_bitset = false;
    if ((long long)n * g_words < 100000000) {
        g_neighbor_bits.assign((size_t)n * g_words, 0);
        use_bitset = true;
    }

    auto add_to_bucket = [&](int v, int s) {
        g_pos_in_bucket[v] = (int)g_buckets[s].size();
        g_buckets[s].push_back(v);
        g_saturation[v] = s;
    };

    auto remove_from_bucket = [&](int v, int s) {
        int last_v = g_buckets[s].back();
        int pos = g_pos_in_bucket[v];
        g_buckets[s][pos] = last_v;
        g_pos_in_bucket[last_v] = pos;
        g_buckets[s].pop_back();
    };

    vector<int> order(n);
    for (int i = 0; i < n; ++i) order[i] = i;
    if (seed != 0) {
        mt19937 rng(seed);
        shuffle(order.begin(), order.end(), rng);
    } else {
        sort(order.begin(), order.end(), [&](int a, int b) {
            if (g.degree[a] != g.degree[b]) return g.degree[a] > g.degree[b];
            return a < b;
        });
    }

    for (int v : order) add_to_bucket(v, 0);

    int max_sat = 0;
    for (int i = 0; i < n; ++i) {
        while (max_sat >= 0 && g_buckets[max_sat].empty()) max_sat--;
        if (max_sat < 0) break;

        int v = g_buckets[max_sat].back();
        g_buckets[max_sat].pop_back();
        g_pos_in_bucket[v] = -1;

        g_timestamp++;
        for (size_t j = g.offset[v]; j < g.offset[v+1]; ++j) {
            int u = g.edges[j];
            if (g_color[u] != -1 && g_color[u] < (int)g_used_at.size()) {
                g_used_at[g_color[u]] = g_timestamp;
            }
        }

        int c = 0;
        while (c < (int)g_used_at.size() && g_used_at[c] == g_timestamp) c++;
        g_color[v] = c;
        g_colored[v] = true;

        for (size_t j = g.offset[v]; j < g.offset[v+1]; ++j) {
            int u = g.edges[j];
            if (!g_colored[u]) {
                bool new_color = false;
                if (use_bitset && c < k) {
                    int w = c / 64;
                    int b = c % 64;
                    if (!(g_neighbor_bits[(size_t)u * g_words + w] & (1ULL << b))) {
                        g_neighbor_bits[(size_t)u * g_words + w] |= (1ULL << b);
                        new_color = true;
                    }
                } else {
                    new_color = true; 
                }

                if (new_color) {
                    remove_from_bucket(u, g_saturation[u]);
                    g_saturation[u]++;
                    add_to_bucket(u, g_saturation[u]);
                    if (g_saturation[u] > max_sat) max_sat = g_saturation[u];
                }
            }
        }
    }
    return g_color;
}

vector<int> randomized_greedy(const Graph& g, uint32_t seed) {
    int n = g.n;
    vector<int> order(n);
    for (int i = 0; i < n; ++i) order[i] = i;
    mt19937 rng(seed);
    shuffle(order.begin(), order.end(), rng);

    vector<int> color(n, -1);
    for (int v : order) {
        g_timestamp++;
        for (size_t j = g.offset[v]; j < g.offset[v+1]; ++j) {
            int u = g.edges[j];
            if (color[u] != -1 && color[u] < (int)g_used_at.size()) {
                g_used_at[color[u]] = g_timestamp;
            }
        }
        int c = 0;
        while (c < (int)g_used_at.size() && g_used_at[c] == g_timestamp) c++;
        color[v] = c;
    }
    return color;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    Graph g;
    build_graph(g);
    if (g.n <= 0) return 0;

    g_used_at.assign(max(g.n, g.k) + 1, -1);
    g_timestamp = 0;

    auto start_time = chrono::steady_clock::now();
    vector<int> best = dsatur(g);
    
    auto get_max_color = [&](const vector<int>& col) {
        int m = -1;
        for (int c : col) if (c > m) m = c;
        return m;
    };

    if (get_max_color(best) >= g.k) {
        uint32_t seed = 42;
        while (chrono::duration_cast<chrono::seconds>(chrono::steady_clock::now() - start_time).count() < 120) {
            vector<int> next = dsatur(g, seed++);
            if (get_max_color(next) < g.k) { best = next; break; }
        }
    }

    if (get_max_color(best) >= g.k) {
        uint32_t seed = 1337;
        while (chrono::duration_cast<chrono::seconds>(chrono::steady_clock::now() - start_time).count() < 280) {
            vector<int> next = randomized_greedy(g, seed++);
            if (get_max_color(next) < g.k) { best = next; break; }
        }
    }

    for (int i = 0; i < g.n; ++i) {
        cout << (best[i] == -1 ? 0 : best[i]) << (i == g.n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
