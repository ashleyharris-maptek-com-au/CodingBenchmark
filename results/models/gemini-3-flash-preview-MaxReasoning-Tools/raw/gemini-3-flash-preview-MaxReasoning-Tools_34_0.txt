I have implemented a solution for the Capacitated Vehicle Routing Problem (CVRP) in `solution.py`.

### Implementation Details:
- **Heuristic**: I used the **Clarke-Wright Savings algorithm**, which is a classical and efficient heuristic for CVRP.
- **Algorithm Flow**:
    1.  Starts with each customer in an individual route (Depot -> Customer -> Depot).
    2.  Calculates "savings" for merging any two routes, based on the distance reduced by visiting customers sequentially rather than returning to the depot.
    3.  Iteratively merges routes with the highest savings, provided they respect the vehicle's capacity and the customers are at the endpoints of their respective routes.
- **Complexity**: The algorithm handles distance matrix calculation in $O(N^2)$ and sorting savings in $O(N^2 \log N)$, making it suitable for relatively large instances.
- **Compliance**: The solution reads from `stdin`, writes to `stdout`, ensures all customers are visited once, and respects vehicle capacity.

### Final Code:
```python
import sys
import math

def solve():
    # Read all input and split into tokens
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    it = iter(input_data)
    
    try:
        n = int(next(it))
        k = int(next(it))
        capacity = int(next(it))

        depot_x = float(next(it))
        depot_y = float(next(it))

        customers = []
        for i in range(n):
            x = float(next(it))
            y = float(next(it))
            demand = int(next(it))
            customers.append({'id': i, 'x': x, 'y': y, 'demand': demand})
    except StopIteration:
        pass

    def dist(x1, y1, x2, y2):
        return math.sqrt((x1 - x2)**2 + (y1 - y2)**2)

    # Distance from depot to each customer
    dist_depot = [dist(depot_x, depot_y, c['x'], c['y']) for c in customers]

    # Distance between customers
    dist_matrix = [[0.0] * n for _ in range(n)]
    for i in range(n):
        for j in range(i + 1, n):
            d = dist(customers[i]['x'], customers[i]['y'], customers[j]['x'], customers[j]['y'])
            dist_matrix[i][j] = dist_matrix[j][i] = d

    # Clarke-Wright Savings
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            s = dist_depot[i] + dist_depot[j] - dist_matrix[i][j]
            savings.append((s, i, j))

    # Sort savings in descending order
    savings.sort(key=lambda x: x[0], reverse=True)

    # Initialize routes: each customer in their own route
    routes = [[i] for i in range(n)]
    route_demands = [c['demand'] for c in customers]
    
    # Map customer to route index
    customer_to_route = list(range(n))

    for s, i, j in savings:
        ri = customer_to_route[i]
        rj = customer_to_route[j]

        if ri == rj:
            continue

        # Check if i and j are at the ends of their routes
        is_i_start = (routes[ri][0] == i)
        is_i_end = (routes[ri][-1] == i)
        is_j_start = (routes[rj][0] == j)
        is_j_end = (routes[rj][-1] == j)

        if (is_i_start or is_i_end) and (is_j_start or is_j_end):
            # Check capacity
            if route_demands[ri] + route_demands[rj] <= capacity:
                # Merge routes
                if is_i_end and is_j_start:
                    new_route = routes[ri] + routes[rj]
                elif is_i_end and is_j_end:
                    new_route = routes[ri] + routes[rj][::-1]
                elif is_i_start and is_j_start:
                    new_route = routes[ri][::-1] + routes[rj]
                elif is_i_start and is_j_end:
                    new_route = routes[rj] + routes[ri]
                else:
                    continue

                # Update routes
                routes[ri] = new_route
                route_demands[ri] += route_demands[rj]
                for customer_id in routes[rj]:
                    customer_to_route[customer_id] = ri
                routes[rj] = [] # Mark as merged
                route_demands[rj] = 0

    # Filter out empty routes
    final_routes = [r for r in routes if r]

    # Calculate total distance
    total_dist = 0.0
    for r in final_routes:
        if not r: continue
        total_dist += dist_depot[r[0]]
        for idx in range(len(r) - 1):
            total_dist += dist_matrix[r[idx]][r[idx+1]]
        total_dist += dist_depot[r[-1]]

    # Output results
    print(f"{total_dist:.6f}")
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    solve()
```
