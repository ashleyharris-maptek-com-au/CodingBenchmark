import sys
import math

def solve():
    input_data = sys.stdin.read().split()
    if not input_data:
        return
    
    it = iter(input_data)
    try:
        N = int(next(it))
        K = int(next(it))
        CAPACITY = float(next(it))
        
        depot_x = float(next(it))
        depot_y = float(next(it))
        
        customers = []
        for i in range(N):
            x = float(next(it))
            y = float(next(it))
            d = float(next(it))
            customers.append({'id': i, 'x': x, 'y': y, 'demand': d})
    except StopIteration:
        return
        
    def dist_sq(p1, p2):
        return (p1[0] - p2[0])**2 + (p1[1] - p2[1])**2
    
    def dist(p1, p2):
        return math.sqrt(dist_sq(p1, p2))
    
    depot = (depot_x, depot_y)
    dist_to_depot = [dist(depot, (c['x'], c['y'])) for c in customers]
    
    savings = []
    for i in range(N):
        p_i = (customers[i]['x'], customers[i]['y'])
        for j in range(i + 1, N):
            p_j = (customers[j]['x'], customers[j]['y'])
            d_ij = dist(p_i, p_j)
            s = dist_to_depot[i] + dist_to_depot[j] - d_ij
            if s > 0:
                savings.append((s, i, j))
            
    savings.sort(key=lambda x: x[0], reverse=True)
    
    # Each customer starts in its own route
    # routes[idx] will store the list of customer indices
    routes = [[i] for i in range(N)]
    route_demands = [c['demand'] for c in customers]
    # customer_to_route_idx[cust_id] = index in routes list
    customer_to_route_idx = list(range(N))
    
    for s, i, j in savings:
        ri_idx = customer_to_route_idx[i]
        rj_idx = customer_to_route_idx[j]
        
        if ri_idx == rj_idx:
            continue
            
        r_i = routes[ri_idx]
        r_j = routes[rj_idx]
        
        # Check if i and j are at the ends of their routes
        # End means first or last element of the list (since they connect to depot)
        i_at_start = (r_i[0] == i)
        i_at_end = (r_i[-1] == i)
        j_at_start = (r_j[0] == j)
        j_at_end = (r_j[-1] == j)
        
        if not ((i_at_start or i_at_end) and (j_at_start or j_at_end)):
            continue
            
        # Check capacity
        if route_demands[ri_idx] + route_demands[rj_idx] <= CAPACITY:
            # Merge routes
            if i_at_end and j_at_start:
                new_route = r_i + r_j
            elif i_at_end and j_at_end:
                new_route = r_i + r_j[::-1]
            elif i_at_start and j_at_start:
                new_route = r_i[::-1] + r_j
            elif i_at_start and j_at_end:
                new_route = r_j + r_i
            else:
                continue
                
            # Update routes
            new_demand = route_demands[ri_idx] + route_demands[rj_idx]
            routes[ri_idx] = new_route
            route_demands[ri_idx] = new_demand
            routes[rj_idx] = None
            
            # Update customer mapping
            for cust_id in r_j:
                customer_to_route_idx[cust_id] = ri_idx

    # Filter active routes
    final_routes = [r for r in routes if r is not None]
    
    # Calculate total distance
    total_dist = 0
    for r in final_routes:
        if not r: continue
        # depot -> first customer
        c_first = customers[r[0]]
        total_dist += dist(depot, (c_first['x'], c_first['y']))
        # between customers
        for idx in range(len(r) - 1):
            c1 = customers[r[idx]]
            c2 = customers[r[idx+1]]
            total_dist += dist((c1['x'], c1['y']), (c2['x'], c2['y']))
        # last customer -> depot
        c_last = customers[r[-1]]
        total_dist += dist((c_last['x'], c_last['y']), depot)
        
    print(total_dist)
    print(len(final_routes))
    for r in final_routes:
        print(" ".join(map(str, r)))

if __name__ == "__main__":
    solve()
