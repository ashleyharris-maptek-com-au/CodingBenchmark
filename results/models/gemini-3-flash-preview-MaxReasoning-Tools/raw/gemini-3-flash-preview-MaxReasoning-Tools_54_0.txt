import math

# Persistent state across calls within a scenario.
# Reset automatically between scenarios and go-arounds.
_p_int = 0.0  # Pitch integral
_r_int = 0.0  # Roll integral
_t_int = 0.0  # Throttle integral
_l_p_err = 0.0  # Last pitch error
_l_r_err = 0.0  # Last roll error
_l_t_err = 0.0  # Last throttle error
_initialized = False

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Autopilot for airliner landing using ILS approach and autoland.
    Handles sensor redundancy, failure scenarios, and environmental factors.
    """
    global _p_int, _r_int, _t_int, _l_p_err, _l_r_err, _l_t_err, _initialized

    # Initialize or reset PID states if necessary
    if not _initialized or dt <= 0:
        _p_int = 0.0
        _r_int = 0.0
        _t_int = 0.0
        _l_p_err = 0.0
        _l_r_err = 0.0
        _l_t_err = 0.0
        _initialized = True
    
    # --- Sensor Data Fusion ---
    # Combine redundant sensors using validation flags.
    def get_val(prefix, suffix, count=3, valid_sfx="_valid"):
        vals = []
        for i in range(1, count + 1):
            v = state.get(f"{prefix}_{i}_{suffix}")
            is_valid = state.get(f"{prefix}_{i}{valid_sfx}")
            if is_valid and v is not None:
                vals.append(v)
        if not vals:
            # Fallback to the first sensor if none are flagged as valid.
            return state.get(f"{prefix}_1_{suffix}")
        return sum(vals) / len(vals)

    # Primary navigation and flight parameters
    loc_dots = get_val("localizer", "dots") or 0.0
    gs_dots = get_val("glideslope", "dots") or 0.0
    ias = get_val("ias", "kt") or 140.0
    pitch = get_val("irs", "pitch_deg") or 0.0
    roll = get_val("irs", "roll_deg") or 0.0
    vs = get_val("irs", "vs_fpm") or 0.0
    radio_alt = state.get("radio_alt_ft")
    
    # Fallback for Radio Altitude (critical for flare)
    if radio_alt is None or radio_alt > 100000:
        # If all radar altimeters failed, we use baro or GPS altitude.
        radio_alt = get_val("baro_alt", "ft") or get_val("gps", "alt_ft") or 5000.0
    
    gs_kt = state.get("groundspeed_kt") or ias
    on_ground = state.get("on_ground", False) or state.get("weight_on_wheels", False)
    
    # Target speeds
    target_vref = state.get("target_vref_kt", 137.0)
    target_ias = state.get("target_ias_kt", target_vref + 5.0)

    cmd = {}

    # --- Go-Around Logic ---
    # Abort landing if approach is dangerously unstable or unsafe at low altitude.
    is_go_around = False
    if 50 < radio_alt < 500:
        # Excessive deviation or low speed triggers a go-around.
        if abs(loc_dots) > 2.0 or abs(gs_dots) > 2.0 or ias < target_vref - 7:
            is_go_around = True
    
    if is_go_around:
        # TOGA (Takeoff/Go-Around) sequence
        cmd['throttle_1'] = 1.0
        cmd['throttle_2'] = 1.0
        cmd['elevator'] = 0.3  # Positive pitch for climb
        cmd['gear'] = 0.0
        cmd['flaps'] = 15.0
        cmd['speedbrake'] = 0.0
        return cmd

    # --- Landing Configuration ---
    if not on_ground:
        cmd['gear'] = 1.0
        # Stepwise flap extension based on altitude
        if radio_alt < 2000:
            cmd['flaps'] = 40.0
        elif radio_alt < 4000:
            cmd['flaps'] = 20.0
        else:
            cmd['flaps'] = 5.0
    else:
        # Post-touchdown sequence
        cmd['flaps'] = 0.0
        cmd['spoiler'] = 1.0
        cmd['autobrake'] = 1.0
        cmd['speedbrake'] = 1.0
        cmd['reverse_1'] = 1.0
        cmd['reverse_2'] = 1.0

    # --- Lateral Control (Localizer) ---
    # Track the localizer centerline using bank angle.
    # Limit bank to 30 degrees during approach.
    target_roll = max(-30.0, min(30.0, -loc_dots * 15.0))
    roll_err = target_roll - roll
    _r_int += roll_err * dt
    _r_int = max(-10.0, min(10.0, _r_int))  # Anti-windup
    roll_d = (roll_err - _l_r_err) / dt if dt > 0 else 0
    _l_r_err = roll_err
    cmd['aileron'] = roll_err * 0.06 + _r_int * 0.01 + roll_d * 0.02

    # Rudder for coordinated flight and ground tracking.
    if radio_alt > 100:
        cmd['rudder'] = -roll * 0.03  # Simple coordination
    else:
        # Align heading with runway centerline on short final and rollout.
        target_hdg = state.get("target_hdg_deg", state.get("irs_1_heading_deg", 0))
        hdg_err = (target_hdg - state.get("irs_1_heading_deg", target_hdg) + 180) % 360 - 180
        cmd['rudder'] = hdg_err * 0.05

    # --- Vertical Control (Glideslope / Flare) ---
    if on_ground:
        # Hold nose down during rollout
        cmd['elevator'] = -0.2
        cmd['throttle_1'] = 0.0
        cmd['throttle_2'] = 0.0
    elif radio_alt < 50:
        # Flare Phase: Reduce descent rate for a smooth touchdown.
        target_vs = -130.0  # Target touchdown VS (fpm)
        vs_err = target_vs - vs
        _p_int += vs_err * dt
        cmd['elevator'] = vs_err * 0.003 + _p_int * 0.01 + 0.12
        # Retard thrust to idle during flare.
        cmd['throttle_1'] = 0.0
        cmd['throttle_2'] = 0.0
    else:
        # Approach Phase: Track the 3-degree glideslope.
        # Normal descent rate (VS) for 3-deg slope is roughly -5.2 * Groundspeed.
        base_vs = -gs_kt * 5.25
        target_vs = base_vs - (gs_dots * 250.0)
        target_vs = max(-1600.0, min(-100.0, target_vs))
        
        vs_err = target_vs - vs
        _p_int += vs_err * dt
        _p_int = max(-50.0, min(50.0, _p_int))
        cmd['elevator'] = vs_err * 0.002 + _p_int * 0.006 + 0.05

    # --- Speed Control (Autothrottle) ---
    if not on_ground and radio_alt >= 50:
        e1_running = state.get("engine_1_running", True)
        e2_running = state.get("engine_2_running", True)
        
        if not e1_running and not e2_running:
            # Dual engine failure (Glide approach): Control speed with pitch.
            cmd['throttle_1'] = 0.0
            cmd['throttle_2'] = 0.0
            cmd['elevator'] = (ias - target_ias) * 0.06
        else:
            # PID Speed control
            speed_err = target_ias - ias
            _t_int += speed_err * dt
            _t_int = max(-2.0, min(2.0, _t_int))
            speed_d = (speed_err - _l_t_err) / dt if dt > 0 else 0
            _l_t_err = speed_err
            
            base_throttle = 0.5 + speed_err * 0.08 + _t_int * 0.03 + speed_d * 0.02
            
            # Compensation for asymmetric thrust (Single engine failure).
            if not e1_running:
                cmd['throttle_1'] = 0.0
                cmd['throttle_2'] = max(0.0, min(1.0, base_throttle * 2.0))
                cmd['rudder'] += cmd['throttle_2'] * 0.2
            elif not e2_running:
                cmd['throttle_1'] = max(0.0, min(1.0, base_throttle * 2.0))
                cmd['throttle_2'] = 0.0
                cmd['rudder'] -= cmd['throttle_1'] * 0.2
            else:
                cmd['throttle_1'] = max(0.0, min(1.0, base_throttle))
                cmd['throttle_2'] = max(0.0, min(1.0, base_throttle))

    # --- Output Clamping ---
    # Ensure all control surface and engine commands are within valid ranges.
    for k in ['elevator', 'aileron', 'rudder', 'elevator_trim']:
        if k in cmd:
            cmd[k] = max(-1.0, min(1.0, cmd[k]))
    for k in ['throttle_1', 'throttle_2', 'reverse_1', 'reverse_2', 'autobrake', 'speedbrake', 'spoiler']:
        if k in cmd:
            cmd[k] = max(0.0, min(1.0, cmd[k]))

    return cmd
