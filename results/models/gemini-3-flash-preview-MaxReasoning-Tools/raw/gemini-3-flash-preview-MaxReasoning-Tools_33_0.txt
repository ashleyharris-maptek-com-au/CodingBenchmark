The following C# code solves the Quadratic Assignment Problem (QAP) using a combination of a flow-distance correlation construction heuristic and an iterated local search (Hill Climbing with 2-opt swaps and random restarts). It is designed to handle $N$ up to 2000 efficiently within the 5-minute time limit by using optimized incremental cost updates and fast I/O.

```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Text;

class QapSolver
{
    static int N;
    static int[] flow;
    static int[] dist;
    static int[] flowT;
    static int[] distT;
    static int[] p;
    static long currentCost;
    static Random rng = new Random();
    static StreamReader reader = new StreamReader(Console.OpenStandardInput());

    static void Main()
    {
        if (!ReadInput()) return;
        if (N <= 0) return;

        Stopwatch sw = Stopwatch.StartNew();
        long timeLimitMs = 290 * 1000; // 4 minutes 50 seconds

        InitializeSolution();

        long bestCost = currentCost;
        int[] bestP = (int[])p.Clone();

        while (sw.ElapsedMilliseconds < timeLimitMs)
        {
            HillClimb(sw, timeLimitMs);
            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                bestP = (int[])p.Clone();
            }

            if (sw.ElapsedMilliseconds < timeLimitMs)
            {
                Shake();
                currentCost = CalculateCost(p);
            }
            else
            {
                break;
            }
        }

        var output = new StringBuilder();
        output.AppendLine(bestCost.ToString());
        for (int i = 0; i < N; i++)
        {
            output.Append(bestP[i]);
            if (i < N - 1) output.Append(" ");
        }
        Console.WriteLine(output.ToString());
    }

    static bool ReadInput()
    {
        N = NextInt();
        if (N == -1) return false;

        flow = new int[N * N];
        for (int i = 0; i < N * N; i++) flow[i] = NextInt();

        dist = new int[N * N];
        for (int i = 0; i < N * N; i++) dist[i] = NextInt();

        // Pre-transpose matrices for faster row-access in the inner loop
        flowT = new int[N * N];
        distT = new int[N * N];
        for (int i = 0; i < N; i++)
        {
            int iN = i * N;
            for (int j = 0; j < N; j++)
            {
                flowT[j * N + i] = flow[iN + j];
                distT[j * N + i] = dist[iN + j];
            }
        }
        return true;
    }

    static int NextInt()
    {
        int res = 0;
        int c = reader.Read();
        while (c != -1 && (c < '0' || c > '9') && c != '-') c = reader.Read();
        if (c == -1) return -1;
        bool neg = false;
        if (c == '-') { neg = true; c = reader.Read(); }
        while (c >= '0' && c <= '9')
        {
            res = res * 10 + (c - '0');
            c = reader.Read();
        }
        return neg ? -res : res;
    }

    static void InitializeSolution()
    {
        // Construction heuristic based on total flow and distance correlation
        long[] totalFlow = new long[N];
        long[] totalDist = new long[N];
        for (int i = 0; i < N; i++)
        {
            int iN = i * N;
            for (int j = 0; j < N; j++)
            {
                totalFlow[i] += (long)flow[iN + j] + flowT[iN + j];
                totalDist[i] += (long)dist[iN + j] + distT[iN + j];
            }
        }

        var facilities = Enumerable.Range(0, N).OrderByDescending(i => totalFlow[i]).ToArray();
        var locations = Enumerable.Range(0, N).OrderBy(i => totalDist[i]).ToArray();

        p = new int[N];
        for (int i = 0; i < N; i++)
        {
            p[facilities[i]] = locations[i];
        }

        currentCost = CalculateCost(p);
    }

    static long CalculateCost(int[] perm)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int iN = i * N;
            int piN = perm[i] * N;
            for (int j = 0; j < N; j++)
            {
                cost += (long)flow[iN + j] * dist[piN + perm[j]];
            }
        }
        return cost;
    }

    static void HillClimb(Stopwatch sw, long limitMs)
    {
        bool improved = true;
        int[] indices = Enumerable.Range(0, N).ToArray();

        while (improved && sw.ElapsedMilliseconds < limitMs)
        {
            improved = false;
            
            // Randomize exploration order
            for (int i = 0; i < N; i++)
            {
                int r = i + rng.Next(N - i);
                int tmp = indices[i];
                indices[i] = indices[r];
                indices[r] = tmp;
            }

            for (int i = 0; i < N && sw.ElapsedMilliseconds < limitMs; i++)
            {
                int u = indices[i];
                for (int j = i + 1; j < N && sw.ElapsedMilliseconds < limitMs; j++)
                {
                    int v = indices[j];
                    long delta = ComputeDelta(u, v);
                    if (delta < 0)
                    {
                        currentCost += delta;
                        int tmp = p[u];
                        p[u] = p[v];
                        p[v] = tmp;
                        improved = true;
                        if (N > 100) break; // First improvement for large N
                    }
                }
                if (improved && N > 100) break;
            }
        }
    }

    static void Shake()
    {
        // Shaking to escape local optima
        int numSwaps = Math.Max(2, N / 20);
        for (int i = 0; i < numSwaps; i++)
        {
            int u = rng.Next(N);
            int v = rng.Next(N);
            if (u == v) continue;
            int tmp = p[u];
            p[u] = p[v];
            p[v] = tmp;
        }
    }

    static long ComputeDelta(int r, int s)
    {
        int pr = p[r];
        int ps = p[s];
        long delta = 0;
        int rN = r * N;
        int sN = s * N;
        int prN = pr * N;
        int psN = ps * N;

        // O(N) delta computation for swapping two facilities
        for (int j = 0; j < N; j++)
        {
            if (j == r || j == s) continue;
            int pj = p[j];
            int pjN = pj * N;
            
            delta += (long)(flow[rN + j] - flow[sN + j]) * (dist[psN + pj] - dist[prN + pj]);
            delta += (long)(flowT[rN + j] - flowT[sN + j]) * (distT[psN + pj] - distT[prN + pj]);
        }
        
        delta += (long)(flow[rN + s] - flow[sN + r]) * (dist[psN + pr] - dist[prN + ps]);
        delta += (long)(flow[rN + r] - flow[sN + s]) * (dist[psN + ps] - dist[prN + pr]);
        
        return delta;
    }
}
```
