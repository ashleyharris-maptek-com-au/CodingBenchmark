using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;
using System.IO;
using System.Text;

class QAPSolver
{
    static int N;
    static int[][] Flow;
    static int[][] FlowT;
    static int[][] Dist;
    static int[][] DistT;

    static void Main(string[] args)
    {
        if (!ReadInput()) return;

        Stopwatch sw = Stopwatch.StartNew();
        Random rng = new Random(42);

        // Initial construction heuristic
        int[] p = InitialSolution();
        long currentCost = CalculateCost(p);

        int[] bestP = (int[])p.Clone();
        long bestCost = currentCost;

        bool[] dlb = new bool[N];
        int[] facilities = Enumerable.Range(0, N).ToArray();

        // Iterated Local Search (ILS)
        // We run local search and perturbation until near the 5-minute limit.
        while (sw.Elapsed.TotalMinutes < 4.75)
        {
            bool improved = true;
            // Local Search (2-opt with First Improvement and Don't Look Bits)
            while (improved && sw.Elapsed.TotalMinutes < 4.75)
            {
                improved = false;
                
                // Shuffle facilities to avoid scan bias
                for (int i = 0; i < N; i++)
                {
                    int r = i + rng.Next(N - i);
                    int tmp = facilities[i]; 
                    facilities[i] = facilities[r]; 
                    facilities[r] = tmp;
                }

                for (int ii = 0; ii < N && sw.Elapsed.TotalMinutes < 4.75; ii++)
                {
                    int i = facilities[ii];
                    if (dlb[i]) continue;

                    bool foundForI = false;
                    int pi = p[i];
                    int[] fI = Flow[i];
                    int[] fTI = FlowT[i];
                    int[] dPi = Dist[pi];
                    int[] dTPi = DistT[pi];

                    for (int j = 0; j < N; j++)
                    {
                        if (i == j) continue;

                        int pj = p[j];
                        int[] fJ = Flow[j];
                        int[] fTJ = FlowT[j];
                        int[] dPj = Dist[pj];
                        int[] dTPj = DistT[pj];

                        // Calculate change in cost for swapping p[i] and p[j]
                        // Delta formula handles non-symmetric matrices efficiently in O(N)
                        long delta = (long)(fI[i] - fJ[j]) * (dTPj[pj] - dTPi[pi])
                                   + (long)(fI[j] - fJ[i]) * (dTPi[pj] - dTPj[pi]);

                        for (int k = 0; k < N; k++)
                        {
                            if (k == i || k == j) continue;
                            int pk = p[k];
                            // Use pre-computed transposes for cache-friendly access
                            delta += (long)(fI[k] - fJ[k]) * (dPj[pk] - dPi[pk])
                                   + (long)(fTI[k] - fTJ[k]) * (dTPj[pk] - dTPi[pk]);
                        }

                        if (delta < 0)
                        {
                            currentCost += delta;
                            p[i] = pj;
                            p[j] = pi;
                            improved = true;
                            foundForI = true;
                            
                            // Update local variables for facility i
                            pi = p[i];
                            dPi = Dist[pi];
                            dTPi = DistT[pi];

                            // Reset DLB for all as any swap can affect potential improvements
                            // For large N, we could reset more selectively, but O(N) scan is fast enough.
                            for (int k = 0; k < N; k++) dlb[k] = false;
                            break; 
                        }
                    }
                    if (!foundForI) dlb[i] = true;
                }
            }

            // Update global best
            if (currentCost < bestCost)
            {
                bestCost = currentCost;
                bestP = (int[])p.Clone();
            }

            // Perturbation: few random swaps to escape local optimum
            p = (int[])bestP.Clone();
            int numSwaps = Math.Max(2, N / 100);
            if (numSwaps > 8) numSwaps = 8; 
            for (int s = 0; s < numSwaps; s++)
            {
                int i = rng.Next(N);
                int j = rng.Next(N);
                if (i != j) { int tmp = p[i]; p[i] = p[j]; p[j] = tmp; }
            }
            currentCost = CalculateCost(p);
            for (int k = 0; k < N; k++) dlb[k] = false;
        }

        // Output results
        Console.WriteLine(bestCost);
        Console.WriteLine(string.Join(" ", bestP));
    }

    static bool ReadInput()
    {
        FastReader fr = new FastReader();
        string s = fr.Next();
        if (s == null) return false;
        if (!int.TryParse(s, out N)) return false;

        Flow = new int[N][];
        FlowT = new int[N][];
        for (int i = 0; i < N; i++) { Flow[i] = new int[N]; FlowT[i] = new int[N]; }
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                Flow[i][j] = fr.NextInt();

        Dist = new int[N][];
        DistT = new int[N][];
        for (int i = 0; i < N; i++) { Dist[i] = new int[N]; DistT[i] = new int[N]; }
        for (int i = 0; i < N; i++)
            for (int j = 0; j < N; j++)
                Dist[i][j] = fr.NextInt();

        // Pre-compute transposes
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                FlowT[j][i] = Flow[i][j];
                DistT[j][i] = Dist[i][j];
            }
        }
        return true;
    }

    static int[] InitialSolution()
    {
        // Hager's construction: match high-flow facilities with central locations
        long[] flowSum = new long[N];
        long[] distSum = new long[N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flowSum[i] += (long)Flow[i][j] + FlowT[i][j];
                distSum[i] += (long)Dist[i][j] + DistT[i][j];
            }
        }

        int[] facilities = Enumerable.Range(0, N).ToArray();
        Array.Sort(facilities, (a, b) => flowSum[b].CompareTo(flowSum[a]));
        
        int[] locations = Enumerable.Range(0, N).ToArray();
        Array.Sort(locations, (a, b) => distSum[a].CompareTo(distSum[b]));

        int[] p = new int[N];
        for (int i = 0; i < N; i++) p[facilities[i]] = locations[i];
        return p;
    }

    static long CalculateCost(int[] p)
    {
        long cost = 0;
        for (int i = 0; i < N; i++)
        {
            int pi = p[i];
            int[] fI = Flow[i];
            for (int j = 0; j < N; j++)
            {
                cost += (long)fI[j] * Dist[pi][p[j]];
            }
        }
        return cost;
    }

    class FastReader
    {
        TextReader reader = Console.In;
        public string Next()
        {
            int c;
            while ((c = reader.Read()) != -1 && char.IsWhiteSpace((char)c)) ;
            if (c == -1) return null;
            StringBuilder sb = new StringBuilder();
            sb.Append((char)c);
            while ((c = reader.Read()) != -1 && !char.IsWhiteSpace((char)c))
                sb.Append((char)c);
            return sb.ToString();
        }

        public int NextInt()
        {
            string s = Next();
            return s == null ? 0 : int.Parse(s);
        }
    }
}
