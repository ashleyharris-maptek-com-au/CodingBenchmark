#version 450 core

/**
 * GLSL Compute Shader for Gravitational N-Body Simulation
 * 
 * This shader implements an O(n^2) gravitational force calculation
 * optimized using shared memory tiling.
 */

layout(local_size_x = 256) in;

struct Particle {
    vec4 pos; // x, y, z, mass
    vec4 vel; // vx, vy, vz, (unused)
};

// Input buffer containing current particle states
layout(std430, binding = 0) buffer ParticlesIn {
    Particle particlesIn[];
};

// Output buffer for updated particle states
layout(std430, binding = 1) buffer ParticlesOut {
    Particle particlesOut[];
};

// Simulation parameters
layout(location = 0) uniform float dt;         // Time step
layout(location = 1) uniform int numParticles; // Total number of particles
layout(location = 2) uniform float G;          // Gravitational constant
layout(location = 3) uniform float softening;  // Softening factor to prevent singularities

// Shared memory for tiling optimization
shared vec4 sharedPos[256];

void main() {
    uint gti = gl_GlobalInvocationID.x; // Global thread index
    uint lti = gl_LocalInvocationID.x;  // Local thread index (within workgroup)

    if (gti >= numParticles) return;

    // Cache current particle data in registers
    vec4 myPos = particlesIn[gti].pos;
    vec3 myVel = particlesIn[gti].vel.xyz;
    vec3 acceleration = vec3(0.0);

    // Iterate over all particles in tiles of 256
    for (int i = 0; i < numParticles; i += 256) {
        // Collaborative load: each thread in the workgroup loads one particle into shared memory
        int sourceIdx = i + int(lti);
        if (sourceIdx < numParticles) {
            sharedPos[lti] = particlesIn[sourceIdx].pos;
        } else {
            // If out of bounds, set mass to 0 so it exerts no force
            sharedPos[lti] = vec4(0.0);
        }

        // Synchronize to ensure all threads have finished loading the tile
        barrier();

        // Calculate gravitational influence from all particles in the current tile
        // This loop is unrolled by the compiler for performance
        for (int j = 0; j < 256; ++j) {
            vec4 otherPos = sharedPos[j];
            
            vec3 diff = otherPos.xyz - myPos.xyz;
            float distSq = dot(diff, diff) + softening;
            
            // Newton's Law of Universal Gravitation: F = G * (m1 * m2) / r^2
            // Acceleration: a = F / m1 = G * m2 / r^2
            // Vector form: a = G * m2 * r_vec / |r|^3
            
            float invDist = inversesqrt(distSq);
            float invDist3 = invDist * invDist * invDist;
            
            acceleration += G * otherPos.w * diff * invDist3;
        }

        // Synchronize before loading the next tile
        barrier();
    }

    // Integration: Simple Semi-implicit Euler
    // v(t + dt) = v(t) + a(t) * dt
    // p(t + dt) = p(t) + v(t + dt) * dt
    vec3 newVel = myVel + acceleration * dt;
    vec3 newPos = myPos.xyz + newVel * dt;

    // Write results to the output buffer
    particlesOut[gti].pos = vec4(newPos, myPos.w);
    particlesOut[gti].vel = vec4(newVel, 0.0);
}
