const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const depot = data.depot;
const truckCount = data.truckCount;
const packages = data.packages;

if (!packages || packages.length === 0) {
    process.stdout.write(JSON.stringify({ routes: [] }) + '\n');
    process.exit(0);
}

// Calculate angle of each package relative to the depot
for (let i = 0; i < packages.length; i++) {
    const pkg = packages[i];
    pkg.angle = Math.atan2(pkg.y - depot[1], pkg.x - depot[0]);
}

// Sort packages by angle (Sweep algorithm basis)
packages.sort((a, b) => a.angle - b.angle);

const routes = [];
const actualTruckCount = Math.min(truckCount, packages.length);
const packagesPerTruck = Math.ceil(packages.length / actualTruckCount);

for (let i = 0; i < actualTruckCount; i++) {
    const group = packages.slice(i * packagesPerTruck, (i + 1) * packagesPerTruck);
    if (group.length === 0) continue;

    // Nearest Neighbor within each group for better local optimization
    const route = [];
    let currentX = depot[0];
    let currentY = depot[1];
    const unvisited = group;

    while (unvisited.length > 0) {
        let nearestIdx = 0;
        let minDistSq = Infinity;

        for (let j = 0; j < unvisited.length; j++) {
            const pkg = unvisited[j];
            const dx = currentX - pkg.x;
            const dy = currentY - pkg.y;
            const d2 = dx * dx + dy * dy;
            if (d2 < minDistSq) {
                minDistSq = d2;
                nearestIdx = j;
            }
        }

        const nextPkg = unvisited.splice(nearestIdx, 1)[0];
        route.push(nextPkg.id);
        currentX = nextPkg.x;
        currentY = nextPkg.y;
    }
    routes.push(route);
}

process.stdout.write(JSON.stringify({ routes }) + '\n');
