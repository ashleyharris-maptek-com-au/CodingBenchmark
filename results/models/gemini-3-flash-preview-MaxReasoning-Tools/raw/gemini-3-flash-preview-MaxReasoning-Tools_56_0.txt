```javascript
"use strict";

const fs = require('fs');

/**
 * Plan delivery routes for a fleet of vans.
 * 
 * Strategy:
 * 1. Sort packages by angle around the depot (Sweep algorithm).
 * 2. Partition the sorted packages into `truckCount` routes.
 * 3. For each route, optimize the delivery order:
 *    - Use Nearest Neighbor for smaller routes (up to 10,000 packages).
 *    - Use Z-order sort (Quadtree traversal) for larger routes to ensure O(N log N) scaling.
 */

function solve() {
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (err) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }
    
    const depot = data.depot || [0, 0];
    const truckCount = data.truckCount || 1;
    const packages = data.packages || [];
    
    if (packages.length === 0) {
        process.stdout.write(JSON.stringify({ routes: [] }) + '\n');
        return;
    }

    const depotX = depot[0];
    const depotY = depot[1];

    // 1. Sort packages by angle from depot to create sectors/wedges
    const pkgsWithAngle = packages.map(p => {
        return {
            id: p.id,
            x: p.x,
            y: p.y,
            angle: Math.atan2(p.y - depotY, p.x - depotX)
        };
    });
    
    pkgsWithAngle.sort((a, b) => a.angle - b.angle);
    
    // 2. Partition into routes
    // We use the maximum allowed number of trucks, capped by the number of packages.
    const numRoutes = Math.min(truckCount, pkgsWithAngle.length);
    const routes = Array.from({ length: numRoutes }, () => []);
    
    for (let i = 0; i < pkgsWithAngle.length; i++) {
        const routeIdx = Math.floor(i * numRoutes / pkgsWithAngle.length);
        routes[routeIdx].push(pkgsWithAngle[i]);
    }
    
    // 3. For each route, determine the best order
    const finalRoutes = [];
    for (let i = 0; i < routes.length; i++) {
        const routePkgs = routes[i];
        if (routePkgs.length === 0) continue;
        
        // Threshold for switching from O(N^2) Nearest Neighbor to O(N log N) Z-order
        if (routePkgs.length < 10000) {
            finalRoutes.push(nearestNeighbor(routePkgs, [depotX, depotY]));
        } else {
            finalRoutes.push(zOrderSort(routePkgs));
        }
    }
    
    process.stdout.write(JSON.stringify({ routes: finalRoutes }) + '\n');
}

/**
 * Nearest Neighbor heuristic (O(N^2))
 */
function nearestNeighbor(pkgs, startPos) {
    const n = pkgs.length;
    const result = [];
    const visited = new Uint8Array(n);
    let currentX = startPos[0];
    let currentY = startPos[1];
    
    for (let i = 0; i < n; i++) {
        let bestIdx = -1;
        let minDistSq = Infinity;
        
        for (let j = 0; j < n; j++) {
            if (visited[j]) continue;
            const p = pkgs[j];
            const dx = p.x - currentX;
            const dy = p.y - currentY;
            const d2 = dx * dx + dy * dy;
            if (d2 < minDistSq) {
                minDistSq = d2;
                bestIdx = j;
            }
        }
        
        if (bestIdx !== -1) {
            visited[bestIdx] = 1;
            const p = pkgs[bestIdx];
            result.push(p.id);
            currentX = p.x;
            currentY = p.y;
        }
    }
    return result;
}

/**
 * Z-order sort (Quadtree-style) for large scale scenarios (O(N log N))
 */
function zOrderSort(pkgs) {
    if (pkgs.length === 0) return [];
    
    const result = [];
    function sortRecursive(arr) {
        if (arr.length === 0) return;
        if (arr.length === 1) {
            result.push(arr[0].id);
            return;
        }
        
        let x0 = Infinity, x1 = -Infinity, y0 = Infinity, y1 = -Infinity;
        for (let i = 0; i < arr.length; i++) {
            const p = arr[i];
            if (p.x < x0) x0 = p.x;
            if (p.x > x1) x1 = p.x;
            if (p.y < y0) y0 = p.y;
            if (p.y > y1) y1 = p.y;
        }
        
        if (x0 === x1 && y0 === y1) {
            for (let i = 0; i < arr.length; i++) result.push(arr[i].id);
            return;
        }
        
        const xMid = (x0 + x1) / 2;
        const yMid = (y0 + y1) / 2;
        const q0 = [], q1 = [], q2 = [], q3 = [];
        for (let i = 0; i < arr.length; i++) {
            const p = arr[i];
            if (p.x < xMid) {
                if (p.y < yMid) q0.push(p); else q1.push(p);
            } else {
                if (p.y < yMid) q3.push(p); else q2.push(p);
            }
        }
        // Traverse in Z-order pattern
        sortRecursive(q0);
        sortRecursive(q1);
        sortRecursive(q2);
        sortRecursive(q3);
    }
    
    sortRecursive(pkgs);
    return result;
}

// Execute the solver
solve();
```
