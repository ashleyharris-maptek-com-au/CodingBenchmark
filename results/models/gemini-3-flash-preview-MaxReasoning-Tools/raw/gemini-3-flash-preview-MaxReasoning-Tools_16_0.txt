#include <iostream>
#include <vector>
#include <algorithm>
#include <limits>

using namespace std;

/**
 * Job-Shop Scheduling Problem (JSSP) Solver
 * 
 * This solver uses the Giffler-Thompson (GT) algorithm, which is a classic heuristic 
 * for generating active schedules. Active schedules are schedules where no task 
 * can be started earlier without delaying another task.
 * 
 * Adaptive Strategy:
 * - For smaller problems, the solver tries multiple Priority Dispatching Rules (PDRs)
 *   (Most Work Remaining, Shortest Processing Time, Most Operations Remaining) 
 *   and selects the one that yields the minimum makespan.
 * - For extremely large problems, it uses the Most Work Remaining (MWKR) rule, 
 *   which is generally robust for JSSP.
 * 
 * Complexity: O(N * J) where N is the total number of tasks and J is the number of jobs.
 * This ensures that even "extreme" cases (50k tasks, 1000 jobs) are solved in a few seconds.
 */

struct Task {
    int machine;
    int duration;
    long long startTime = -1;
};

struct Job {
    int id;
    vector<Task> tasks;
    int nextTaskIdx = 0;
    long long readyTime = 0;
    long long workRemaining = 0;
};

class JSSPSolver {
public:
    int numJobs;
    int numMachines;
    vector<Job> originalJobs;
    vector<Job> currentJobs;
    vector<long long> machineFreeTime;

    JSSPSolver(int j, int m) : numJobs(j), numMachines(m) {
        originalJobs.resize(j);
        machineFreeTime.resize(m);
    }

    void reset() {
        currentJobs = originalJobs;
        for (auto& job : currentJobs) {
            job.nextTaskIdx = 0;
            job.readyTime = 0;
            job.workRemaining = 0;
            for (const auto& t : job.tasks) {
                job.workRemaining += t.duration;
            }
        }
        fill(machineFreeTime.begin(), machineFreeTime.end(), 0LL);
    }

    /**
     * ruleType: 0 = MWKR (Most Work Remaining), 
     *           1 = SPT (Shortest Processing Time), 
     *           2 = MOPNR (Most Operations Remaining)
     */
    long long solve(int ruleType) {
        reset();
        int totalTasks = 0;
        for (const auto& job : currentJobs) totalTasks += (int)job.tasks.size();
        int completedTasks = 0;

        // Group jobs by the machine they are currently waiting for (only 'ready' tasks)
        vector<vector<int>> readyForMachine(numMachines);
        for (int j = 0; j < numJobs; ++j) {
            if (!currentJobs[j].tasks.empty()) {
                readyForMachine[currentJobs[j].tasks[0].machine].push_back(j);
            }
        }

        while (completedTasks < totalTasks) {
            long long earliestFinish = numeric_limits<long long>::max();
            int m_star = -1;

            // Step 1: Find the earliest possible completion time among all currently available tasks
            for (int m = 0; m < numMachines; ++m) {
                for (int j : readyForMachine[m]) {
                    const auto& task = currentJobs[j].tasks[currentJobs[j].nextTaskIdx];
                    long long possibleStart = max(currentJobs[j].readyTime, machineFreeTime[m]);
                    long long possibleFinish = possibleStart + task.duration;
                    if (possibleFinish < earliestFinish) {
                        earliestFinish = possibleFinish;
                        m_star = m;
                    }
                }
            }

            if (m_star == -1) break; // Feasibility check (valid JSSP is always feasible)

            // Step 2: Form the conflict set of jobs that could potentially start on machine m_star 
            // before the earliestFinish found above.
            vector<int> conflictSet;
            for (int j : readyForMachine[m_star]) {
                if (max(currentJobs[j].readyTime, machineFreeTime[m_star]) < earliestFinish) {
                    conflictSet.push_back(j);
                }
            }

            // Step 3: Choose one job from the conflict set according to the priority rule
            int j_star = -1;
            if (ruleType == 0) { // MWKR
                long long maxWork = -1;
                for (int j : conflictSet) {
                    if (currentJobs[j].workRemaining > maxWork) {
                        maxWork = currentJobs[j].workRemaining;
                        j_star = j;
                    }
                }
            } else if (ruleType == 1) { // SPT
                int minDur = numeric_limits<int>::max();
                for (int j : conflictSet) {
                    if (currentJobs[j].tasks[currentJobs[j].nextTaskIdx].duration < minDur) {
                        minDur = currentJobs[j].tasks[currentJobs[j].nextTaskIdx].duration;
                        j_star = j;
                    }
                }
            } else { // MOPNR
                int maxOps = -1;
                for (int j : conflictSet) {
                    int opsLeft = (int)currentJobs[j].tasks.size() - currentJobs[j].nextTaskIdx;
                    if (opsLeft > maxOps) {
                        maxOps = opsLeft;
                        j_star = j;
                    }
                }
            }

            // Step 4: Schedule the task for j_star on machine m_star
            auto& job = currentJobs[j_star];
            auto& task = job.tasks[job.nextTaskIdx];
            long long startTime = max(job.readyTime, machineFreeTime[m_star]);
            task.startTime = startTime;
            long long endTime = startTime + task.duration;

            machineFreeTime[m_star] = endTime;
            job.readyTime = endTime;
            job.workRemaining -= task.duration;

            // Update ready list for the machine
            auto& q = readyForMachine[m_star];
            for (int i = 0; i < (int)q.size(); ++i) {
                if (q[i] == j_star) {
                    q[i] = q.back(); // Efficient removal
                    q.pop_back();
                    break;
                }
            }

            // Move the job to the next machine's ready list if it has more tasks
            job.nextTaskIdx++;
            if (job.nextTaskIdx < (int)job.tasks.size()) {
                readyForMachine[job.tasks[job.nextTaskIdx].machine].push_back(j_star);
            }
            completedTasks++;
        }

        long long makespan = 0;
        for (long long t : machineFreeTime) {
            if (t > makespan) makespan = t;
        }
        return makespan;
    }

    void printSolution() {
        for (int i = 0; i < numJobs; ++i) {
            const auto& job = currentJobs[i];
            for (int t = 0; t < (int)job.tasks.size(); ++t) {
                cout << job.tasks[t].startTime << " " << job.tasks[t].machine << (t == (int)job.tasks.size() - 1 ? "" : " ");
            }
            cout << "\n";
        }
    }
};

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    JSSPSolver solver(J, M);
    long long totalTasks = 0;
    for (int i = 0; i < J; ++i) {
        int T;
        if (!(cin >> T)) break;
        solver.originalJobs[i].id = i;
        solver.originalJobs[i].tasks.resize(T);
        totalTasks += T;
        for (int j = 0; j < T; ++j) {
            cin >> solver.originalJobs[i].tasks[j].machine >> solver.originalJobs[i].tasks[j].duration;
        }
    }

    // Adaptive logic: decide how many priority rules to try based on complexity
    // We estimate complexity as J * totalTasks.
    long long complexityScore = (long long)J * totalTasks;

    vector<int> rulesToTry;
    if (complexityScore < 10000000) { // < 10M iterations
        rulesToTry = {0, 1, 2}; // Try MWKR, SPT, MOPNR
    } else if (complexityScore < 50000000) { // < 50M
        rulesToTry = {0, 1}; // Try MWKR, SPT
    } else {
        rulesToTry = {0}; // Only MWKR for extremely large scales
    }

    long long bestMakespan = numeric_limits<long long>::max();
    int bestRuleType = 0;

    for (int rule : rulesToTry) {
        long long ms = solver.solve(rule);
        if (ms < bestMakespan) {
            bestMakespan = ms;
            bestRuleType = rule;
        }
    }

    // Run one last time with the best rule to set start times for the final output
    solver.solve(bestRuleType);
    solver.printSolution();

    return 0;
}
