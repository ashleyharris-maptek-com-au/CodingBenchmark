using System;
using System.Collections.Generic;
using System.Linq;
using System.Numerics;
using System.Globalization;
using System.IO;

namespace Solver
{
    /// <summary>
    /// Represents an Axis-Aligned Bounding Box.
    /// </summary>
    public struct AABB
    {
        public Vector3 Min;
        public Vector3 Max;
        public Vector3 Size => Max - Min;
        public float Volume => Size.X * Size.Y * Size.Z;

        public static AABB FromVertices(Vector3[] vertices)
        {
            if (vertices.Length == 0) return new AABB();
            Vector3 min = new Vector3(float.MaxValue);
            Vector3 max = new Vector3(float.MinValue);
            foreach (var v in vertices)
            {
                min = Vector3.Min(min, v);
                max = Vector3.Max(max, v);
            }
            return new AABB { Min = min, Max = max };
        }
    }

    /// <summary>
    /// Represents a polyhedron in a specific orientation.
    /// </summary>
    public class OrientedPolyhedron
    {
        public Quaternion Rotation;
        public AABB AABB;
        public Vector3 Offset; // Translation to bring the AABB min to the origin
    }

    class Program
    {
        static void Main(string[] args)
        {
            // Ensure consistent parsing and formatting
            CultureInfo.DefaultThreadCurrentCulture = CultureInfo.InvariantCulture;
            
            // Use a custom fast reader for performance at ludicrous scales
            var reader = new InputReader(Console.OpenStandardInput());
            
            int V = reader.NextInt();
            if (V <= 0) return;
            int F = reader.NextInt();

            Vector3[] vertices = new Vector3[V];
            for (int i = 0; i < V; i++)
            {
                vertices[i] = new Vector3(reader.NextFloat(), reader.NextFloat(), reader.NextFloat());
            }

            // Skip face indices (not strictly needed for AABB-based packing)
            for (int i = 0; i < F * 3; i++) reader.NextInt();

            float containerW = reader.NextFloat();
            float containerH = reader.NextFloat();
            float containerD = reader.NextFloat();
            Vector3 container = new Vector3(containerW, containerH, containerD);

            // Pre-calculate 24 standard rotations and their AABBs
            var orientations = Get24Orientations(vertices);

            // Adaptive packing strategy:
            // For any scale, we use a recursive grid-based packer.
            // 1. Find the best orientation for a grid that fits in the current box.
            // 2. Pack as many items as possible in that grid.
            // 3. Recursively pack the remaining 3 disjoint boxes that form the L-shaped leftover space.
            
            // Phase 1: Count items to output the total count on the first line
            long totalCount = CountRecursive(container, orientations, 0);

            // Phase 2: Print translations and rotations
            using (var sw = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1024 * 1024))
            {
                sw.WriteLine(totalCount);
                if (totalCount > 0)
                {
                    PrintRecursive(container, orientations, Vector3.Zero, sw, 0);
                }
            }
        }

        /// <summary>
        /// Generates the 24 standard 90-degree rotations and filters them by unique AABB sizes.
        /// </summary>
        static List<OrientedPolyhedron> Get24Orientations(Vector3[] originalVertices)
        {
            var results = new List<OrientedPolyhedron>();
            Vector3[] dirs = { Vector3.UnitX, Vector3.UnitY, Vector3.UnitZ, -Vector3.UnitX, -Vector3.UnitY, -Vector3.UnitZ };
            
            foreach (var fwd in dirs)
            {
                foreach (var up in dirs)
                {
                    if (Math.Abs(Vector3.Dot(fwd, up)) > 0.001f) continue;
                    var right = Vector3.Cross(up, fwd);
                    // Create a rotation matrix where basis vectors are rows (for Vector3.Transform)
                    var m = new Matrix4x4(
                        right.X, right.Y, right.Z, 0,
                        up.X, up.Y, up.Z, 0,
                        fwd.X, fwd.Y, fwd.Z, 0,
                        0, 0, 0, 1
                    );
                    var q = Quaternion.CreateFromRotationMatrix(m);
                    q = Quaternion.Normalize(q);

                    var rotated = originalVertices.Select(v => Vector3.Transform(v, q)).ToArray();
                    var aabb = AABB.FromVertices(rotated);
                    results.Add(new OrientedPolyhedron { Rotation = q, AABB = aabb, Offset = -aabb.Min });
                }
            }
            
            // Deduplicate by AABB dimensions to simplify search
            var unique = new List<OrientedPolyhedron>();
            foreach (var op in results)
            {
                if (!unique.Any(u => 
                    Math.Abs(u.AABB.Size.X - op.AABB.Size.X) < 1e-4f && 
                    Math.Abs(u.AABB.Size.Y - op.AABB.Size.Y) < 1e-4f && 
                    Math.Abs(u.AABB.Size.Z - op.AABB.Size.Z) < 1e-4f))
                {
                    unique.Add(op);
                }
            }
            return unique;
        }

        /// <summary>
        /// Recursively calculates the number of items that can be packed.
        /// </summary>
        static long CountRecursive(Vector3 box, List<OrientedPolyhedron> orientations, int depth)
        {
            if (depth > 12) return 0; // Prevent excessive recursion depth
            if (box.X < 0 || box.Y < 0 || box.Z < 0) return 0;

            long bestCount = 0;
            OrientedPolyhedron bestOp = null;
            int bx = 0, by = 0, bz = 0;

            // Find the orientation that allows the largest grid count
            foreach (var op in orientations)
            {
                int nx = (int)(box.X / op.AABB.Size.X + 1e-6f);
                int ny = (int)(box.Y / op.AABB.Size.Y + 1e-6f);
                int nz = (int)(box.Z / op.AABB.Size.Z + 1e-6f);
                if (nx <= 0 || ny <= 0 || nz <= 0) continue;
                long count = (long)nx * ny * nz;
                if (count > bestCount) {
                    bestCount = count; bestOp = op;
                    bx = nx; by = ny; bz = nz;
                }
            }

            if (bestOp == null) return 0;

            Vector3 s = bestOp.AABB.Size;
            float lx = box.X - bx * s.X;
            float ly = box.Y - by * s.Y;
            float lz = box.Z - bz * s.Z;

            long total = bestCount;
            // Pack leftovers in 3 disjoint boxes
            total += CountRecursive(new Vector3(lx, box.Y, box.Z), orientations, depth + 1);
            total += CountRecursive(new Vector3(bx * s.X, ly, box.Z), orientations, depth + 1);
            total += CountRecursive(new Vector3(bx * s.X, by * s.Y, lz), orientations, depth + 1);
            return total;
        }

        /// <summary>
        /// Recursively prints the packing solutions.
        /// </summary>
        static void PrintRecursive(Vector3 box, List<OrientedPolyhedron> orientations, Vector3 offset, StreamWriter sw, int depth)
        {
            if (depth > 12) return;
            if (box.X < 0 || box.Y < 0 || box.Z < 0) return;

            long bestCount = 0;
            OrientedPolyhedron bestOp = null;
            int bx = 0, by = 0, bz = 0;

            foreach (var op in orientations)
            {
                int nx = (int)(box.X / op.AABB.Size.X + 1e-6f);
                int ny = (int)(box.Y / op.AABB.Size.Y + 1e-6f);
                int nz = (int)(box.Z / op.AABB.Size.Z + 1e-6f);
                if (nx <= 0 || ny <= 0 || nz <= 0) continue;
                long count = (long)nx * ny * nz;
                if (count > bestCount) {
                    bestCount = count; bestOp = op;
                    bx = nx; by = ny; bz = nz;
                }
            }

            if (bestOp == null) return;

            Vector3 s = bestOp.AABB.Size;
            Quaternion rot = bestOp.Rotation;
            Vector3 off = bestOp.Offset;
            string rotStr = $"{rot.W:F4} {rot.X:F4} {rot.Y:F4} {rot.Z:F4}";
            
            // Print the current grid
            for (int x = 0; x < bx; x++) {
                float tx = offset.X + x * s.X + off.X;
                for (int y = 0; y < by; y++) {
                    float ty = offset.Y + y * s.Y + off.Y;
                    for (int z = 0; z < bz; z++) {
                        float tz = offset.Z + z * s.Z + off.Z;
                        sw.Write(tx.ToString("F3")); sw.Write(' ');
                        sw.Write(ty.ToString("F3")); sw.Write(' ');
                        sw.Write(tz.ToString("F3")); sw.Write(' ');
                        sw.WriteLine(rotStr);
                    }
                }
            }

            float lx = box.X - bx * s.X;
            float ly = box.Y - by * s.Y;
            float lz = box.Z - bz * s.Z;

            // Recurse into leftovers
            PrintRecursive(new Vector3(lx, box.Y, box.Z), orientations, offset + new Vector3(bx * s.X, 0, 0), sw, depth + 1);
            PrintRecursive(new Vector3(bx * s.X, ly, box.Z), orientations, offset + new Vector3(0, by * s.Y, 0), sw, depth + 1);
            PrintRecursive(new Vector3(bx * s.X, by * s.Y, lz), orientations, offset + new Vector3(0, 0, bz * s.Z), sw, depth + 1);
        }
    }

    /// <summary>
    /// Fast input reader to handle large amounts of data efficiently.
    /// </summary>
    class InputReader
    {
        private Stream stream;
        private byte[] buffer = new byte[1024 * 1024];
        private int pos = 0;
        private int limit = 0;

        public InputReader(Stream s) { stream = s; }

        private int ReadByte()
        {
            if (pos >= limit)
            {
                limit = stream.Read(buffer, 0, buffer.Length);
                pos = 0;
                if (limit == 0) return -1;
            }
            return buffer[pos++];
        }

        public int NextInt()
        {
            int c = ReadByte();
            while (c != -1 && c <= ' ') c = ReadByte();
            if (c == -1) return -1;
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int res = 0;
            while (c >= '0' && c <= '9') { res = res * 10 + (c - '0'); c = ReadByte(); }
            return res * sign;
        }

        public float NextFloat()
        {
            int c = ReadByte();
            while (c != -1 && c <= ' ') c = ReadByte();
            if (c == -1) return 0;
            bool neg = false;
            if (c == '-') { neg = true; c = ReadByte(); }
            double res = 0;
            while (c >= '0' && c <= '9') { res = res * 10 + (c - '0'); c = ReadByte(); }
            if (c == '.') {
                double div = 10;
                c = ReadByte();
                while (c >= '0' && c <= '9') { res += (c - '0') / div; div *= 10; c = ReadByte(); }
            }
            if (c == 'e' || c == 'E') {
                c = ReadByte();
                int expSign = 1;
                if (c == '-') { expSign = -1; c = ReadByte(); }
                else if (c == '+') { c = ReadByte(); }
                int exp = 0;
                while (c >= '0' && c <= '9') { exp = exp * 10 + (c - '0'); c = ReadByte(); }
                res *= Math.Pow(10, expSign * exp);
            }
            return (float)(neg ? -res : res);
        }
    }
}
