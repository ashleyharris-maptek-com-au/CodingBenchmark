#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Calculate 8x8 checkerboard parity
    vec2 grid = floor(uv * 8.0);
    float parity = mod(grid.x + grid.y, 2.0);
    float checkColor = (parity < 0.5) ? 0.9 : 0.2;

    // Standard Lambertian diffuse factor
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);

    // Apply lighting factor: (0.5 + 0.5 * diffuse)
    float shadingFactor = 0.5 + 0.5 * diffuse;

    // Final color with alpha 1.0
    outColor = vec4(vec3(checkColor * shadingFactor), 1.0);
}
