use std::io::{self, Read, Write, BufReader, BufWriter, BufRead};

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f32,
    y: f32,
    z: f32,
}

impl Point {
    #[inline(always)]
    fn dist_sq(&self, other: &Point) -> f32 {
        let dx = self.x - other.x;
        let dy = self.y - other.y;
        let dz = self.z - other.z;
        dx * dx + dy * dy + dz * dz
    }
}

#[derive(Clone, Copy)]
struct AccumPoint {
    x: f64,
    y: f64,
    z: f64,
}

struct KdNode {
    point: Point,
    id: u32,
    left: i32,
    right: i32,
}

fn build_kd_tree_flat(centroids: &[Point]) -> Vec<KdNode> {
    let mut indexed: Vec<(Point, u32)> = centroids.iter().enumerate().map(|(i, &p)| (p, i as u32)).collect();
    let mut nodes = Vec::with_capacity(centroids.len());
    if !centroids.is_empty() {
        build_recursive(&mut indexed, 0, &mut nodes);
    }
    nodes
}

fn build_recursive(centroids: &mut [(Point, u32)], depth: usize, nodes: &mut Vec<KdNode>) -> i32 {
    if centroids.is_empty() { return -1; }
    
    let axis = depth % 3;
    centroids.sort_by(|a, b| {
        let val_a = match axis { 0 => a.0.x, 1 => a.0.y, _ => a.0.z };
        let val_b = match axis { 0 => b.0.x, 1 => b.0.y, _ => b.0.z };
        val_a.partial_cmp(&val_b).unwrap_or(std::cmp::Ordering::Equal)
    });
    
    let mid = centroids.len() / 2;
    let point = centroids[mid].0;
    let id = centroids[mid].1;
    
    let node_idx = nodes.len() as i32;
    nodes.push(KdNode { point, id, left: -1, right: -1 });
    
    let left = build_recursive(&mut centroids[..mid], depth + 1, nodes);
    let right = build_recursive(&mut centroids[mid+1..], depth + 1, nodes);
    
    nodes[node_idx as usize].left = left;
    nodes[node_idx as usize].right = right;
    
    node_idx
}

#[inline(always)]
fn find_nearest_flat(nodes: &[KdNode], node_idx: i32, p: &Point, depth: usize, best: &mut (f32, u32)) {
    if node_idx == -1 { return; }
    let node = &nodes[node_idx as usize];
    
    let d = p.dist_sq(&node.point);
    if d < best.0 {
        best.0 = d;
        best.1 = node.id;
    }
    
    let axis = depth % 3;
    let (val_p, val_n) = match axis {
        0 => (p.x, node.point.x),
        1 => (p.y, node.point.y),
        _ => (p.z, node.point.z),
    };
    
    let diff = val_p - val_n;
    let (near, far) = if diff < 0.0 {
        (node.left, node.right)
    } else {
        (node.right, node.left)
    };
    
    find_nearest_flat(nodes, near, p, depth + 1, best);
    if diff * diff < best.0 {
        find_nearest_flat(nodes, far, p, depth + 1, best);
    }
}

fn k_means_limited(points: &[Point], k: u32, max_iters: usize) -> Vec<Point> {
    let n = points.len();
    if n == 0 { return vec![]; }
    let k_usize = k as usize;
    
    let mut centroids = Vec::with_capacity(k_usize);
    for i in 0..k_usize {
        // Use a deterministic pseudo-random sequence to pick initial centroids
        let idx = (i as u64 * 1103515245 + 12345) % n as u64;
        centroids.push(points[idx as usize]);
    }
    
    for _ in 0..max_iters {
        let mut new_centroids = vec![AccumPoint { x: 0.0, y: 0.0, z: 0.0 }; k_usize];
        let mut counts = vec![0u64; k_usize];
        
        if k > 128 {
            let kd_tree = build_kd_tree_flat(&centroids);
            for p in points {
                let mut best = (f32::MAX, 0u32);
                find_nearest_flat(&kd_tree, 0, p, 0, &mut best);
                let best_k = best.1 as usize;
                new_centroids[best_k].x += p.x as f64;
                new_centroids[best_k].y += p.y as f64;
                new_centroids[best_k].z += p.z as f64;
                counts[best_k] += 1;
            }
        } else {
            for p in points {
                let mut min_dist = f32::MAX;
                let mut best_k = 0;
                for (j, c) in centroids.iter().enumerate() {
                    let d = p.dist_sq(c);
                    if d < min_dist {
                        min_dist = d;
                        best_k = j;
                    }
                }
                new_centroids[best_k].x += p.x as f64;
                new_centroids[best_k].y += p.y as f64;
                new_centroids[best_k].z += p.z as f64;
                counts[best_k] += 1;
            }
        }
        
        let mut max_move = 0.0f32;
        for j in 0..k_usize {
            if counts[j] > 0 {
                let next_c = Point {
                    x: (new_centroids[j].x / counts[j] as f64) as f32,
                    y: (new_centroids[j].y / counts[j] as f64) as f32,
                    z: (new_centroids[j].z / counts[j] as f64) as f32,
                };
                max_move = max_move.max(centroids[j].dist_sq(&next_c));
                centroids[j] = next_c;
            } else {
                // Empty cluster: re-initialize with another point
                let idx = (j as u64 * 1103515245 + 12345 + counts.iter().sum::<u64>()) % n as u64;
                centroids[j] = points[idx as usize];
            }
        }
        if max_move < 1e-6 { break; }
    }
    centroids
}

#[inline(always)]
fn read_point<R: Read>(reader: &mut R) -> io::Result<Point> {
    let mut buf = [0u8; 12];
    reader.read_exact(&mut buf)?;
    Ok(Point {
        x: f32::from_le_bytes([buf[0], buf[1], buf[2], buf[3]]),
        y: f32::from_le_bytes([buf[4], buf[5], buf[6], buf[7]]),
        z: f32::from_le_bytes([buf[8], buf[9], buf[10], buf[11]]),
    })
}

fn solve_in_memory<R: Read, W: Write>(num_points: u64, k: u32, reader: &mut R, writer: &mut W) -> io::Result<()> {
    let mut points = Vec::with_capacity(num_points as usize);
    for _ in 0..num_points {
        points.push(read_point(reader)?);
    }
    
    let centroids = k_means_limited(&points, k, 20);
    
    if k > 128 {
        let kd_tree = build_kd_tree_flat(&centroids);
        for p in &points {
            let mut best = (f32::MAX, 0u32);
            find_nearest_flat(&kd_tree, 0, p, 0, &mut best);
            writer.write_all(&best.1.to_le_bytes())?;
        }
    } else {
        for p in &points {
            let mut min_dist = f32::MAX;
            let mut best_k = 0;
            for (j, c) in centroids.iter().enumerate() {
                let d = p.dist_sq(c);
                if d < min_dist {
                    min_dist = d;
                    best_k = j;
                }
            }
            writer.write_all(&(best_k as u32).to_le_bytes())?;
        }
    }
    Ok(())
}

fn solve_streaming<R: Read, W: Write>(num_points: u64, k: u32, reader: &mut R, writer: &mut W) -> io::Result<()> {
    // For large datasets, train centroids on a significant subset
    let training_size = 5_000_000.min(num_points) as usize;
    let mut training_points = Vec::with_capacity(training_size);
    for _ in 0..training_size {
        training_points.push(read_point(reader)?);
    }
    
    let centroids = k_means_limited(&training_points, k, 15);
    
    if k > 128 {
        let kd_tree = build_kd_tree_flat(&centroids);
        // Output for training points
        for p in &training_points {
            let mut best = (f32::MAX, 0u32);
            find_nearest_flat(&kd_tree, 0, p, 0, &mut best);
            writer.write_all(&best.1.to_le_bytes())?;
        }
        // Stream remaining points
        for _ in training_size as u64..num_points {
            let p = read_point(reader)?;
            let mut best = (f32::MAX, 0u32);
            find_nearest_flat(&kd_tree, 0, &p, 0, &mut best);
            writer.write_all(&best.1.to_le_bytes())?;
        }
    } else {
        // Output for training points
        for p in &training_points {
            let mut min_dist = f32::MAX;
            let mut best_k = 0;
            for (j, c) in centroids.iter().enumerate() {
                let d = p.dist_sq(c);
                if d < min_dist {
                    min_dist = d;
                    best_k = j;
                }
            }
            writer.write_all(&(best_k as u32).to_le_bytes())?;
        }
        // Stream remaining points
        for _ in training_size as u64..num_points {
            let p = read_point(reader)?;
            let mut min_dist = f32::MAX;
            let mut best_k = 0;
            for (j, c) in centroids.iter().enumerate() {
                let d = p.dist_sq(c);
                if d < min_dist {
                    min_dist = d;
                    best_k = j;
                }
            }
            writer.write_all(&(best_k as u32).to_le_bytes())?;
        }
    }
    Ok(())
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut reader = BufReader::with_capacity(1024 * 1024, stdin.lock());
    
    let mut line = String::new();
    if reader.read_line(&mut line)? == 0 {
        return Ok(());
    }
    
    let parts: Vec<&str> = line.split_whitespace().collect();
    if parts.len() < 2 { return Ok(()); }
    let num_points: u64 = parts[0].parse().unwrap_or(0);
    let num_clusters: u32 = parts[1].parse().unwrap_or(0);
    
    if num_points == 0 || num_clusters == 0 { return Ok(()); }

    let stdout = io::stdout();
    let mut writer = BufWriter::with_capacity(1024 * 1024, stdout.lock());
    
    // Threshold for in-memory processing (e.g., 64M points ~ 768MB + indices)
    if num_points <= 64_000_000 {
        solve_in_memory(num_points, num_clusters, &mut reader, &mut writer)?;
    } else {
        solve_streaming(num_points, num_clusters, &mut reader, &mut writer)?;
    }
    writer.flush()?;
    Ok(())
}
