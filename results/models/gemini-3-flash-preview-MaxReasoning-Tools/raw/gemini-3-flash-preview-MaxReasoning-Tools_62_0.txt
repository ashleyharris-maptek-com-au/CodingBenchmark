"use strict";

declare const require: any;
declare const process: any;
const fs = require('fs');

function solve() {
    let input: string;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data: any;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const { width, height, grid, ignitions, towns, cutBudget } = data;

    if (width === undefined || height === undefined || !grid || !towns) {
        process.stdout.write(JSON.stringify({ firebreaks: [] }));
        return;
    }

    const size = width * height;
    const g = new Uint8Array(size);
    for (let y = 0; y < height; y++) {
        const row = grid[y];
        for (let x = 0; x < width; x++) {
            if (row[x] === '.') g[y * width + x] = 1;
        }
    }

    // 1. Initial fire simulation (BFS) to identify which cells will burn without firebreaks
    const burned = new Uint8Array(size);
    const queue = new Int32Array(size);
    let head = 0;
    let tail = 0;

    const ignitionSet = new Set<number>();
    if (Array.isArray(ignitions)) {
        for (const [ix, iy] of ignitions) {
            if (ix >= 0 && ix < width && iy >= 0 && iy < height) {
                const idx = iy * width + ix;
                ignitionSet.add(idx);
                if (g[idx] && !burned[idx]) {
                    burned[idx] = 1;
                    queue[tail++] = idx;
                }
            }
        }
    }

    const dirs = [-1, 1, -width, width];
    while (head < tail) {
        const u = queue[head++];
        const ux = u % width;
        for (const d of dirs) {
            const v = u + d;
            if (v >= 0 && v < size) {
                // Bounds check for left/right movement
                if (d === -1 && ux === 0) continue;
                if (d === 1 && ux === width - 1) continue;
                
                if (!burned[v] && g[v]) {
                    burned[v] = 1;
                    queue[tail++] = v;
                }
            }
        }
    }

    // 2. Identify towns at risk and their potential local cuts
    const atRiskTowns = (towns as any[]).filter(t => {
        const idx = t.y * width + t.x;
        return idx >= 0 && idx < size && burned[idx];
    });

    if (atRiskTowns.length === 0 || cutBudget <= 0) {
        process.stdout.write(JSON.stringify({ firebreaks: [] }));
        return;
    }

    // Candidate firebreaks for each town: its burned forest neighbors.
    // This is a robust strategy for protecting specific assets.
    const townToNeighbors: any[] = [];
    for (const town of atRiskTowns) {
        const idx = town.y * width + town.x;
        
        // If a town is at an ignition point, it cannot be saved.
        if (ignitionSet.has(idx)) continue;
        
        const neighbors: number[] = [];
        const tx = town.x;
        for (const d of dirs) {
            const v = idx + d;
            if (v >= 0 && v < size) {
                if (d === -1 && tx === 0) continue;
                if (d === 1 && tx === width - 1) continue;
                if (g[v] && burned[v]) {
                    neighbors.push(v);
                }
            }
        }
        
        if (neighbors.length > 0) {
            townToNeighbors.push({
                id: town.id,
                value: town.value,
                originalNeighbors: neighbors
            });
        }
    }

    // 3. Greedy selection of towns to save
    // We prioritize towns with the highest value-to-cost ratio, 
    // where cost is the number of currently-uncut forest neighbors.
    const firebreaks = new Set<number>();
    let budget = cutBudget;
    
    // Pre-calculate cell-to-town mapping and initial costs
    const cellToTowns = new Map<number, number[]>();
    const costs = new Int32Array(townToNeighbors.length);
    for (let i = 0; i < townToNeighbors.length; i++) {
        const tn = townToNeighbors[i];
        costs[i] = tn.originalNeighbors.length;
        for (const n of tn.originalNeighbors) {
            let list = cellToTowns.get(n);
            if (!list) {
                list = [];
                cellToTowns.set(n, list);
            }
            list.push(i);
        }
    }

    const activeTowns = new Set<number>();
    for (let i = 0; i < townToNeighbors.length; i++) activeTowns.add(i);

    while (budget > 0 && activeTowns.size > 0) {
        let bestTownIdx = -1;
        let bestRatio = -1;
        
        for (const i of activeTowns) {
            const cost = costs[i];
            if (cost === 0) {
                // Already saved by firebreaks placed for other towns
                activeTowns.delete(i);
                continue;
            }
            
            if (cost <= budget) {
                const ratio = townToNeighbors[i].value / cost;
                if (ratio > bestRatio) {
                    bestRatio = ratio;
                    bestTownIdx = i;
                }
            }
        }

        if (bestTownIdx === -1) break;

        // Apply necessary firebreaks to save this town completely
        const tn = townToNeighbors[bestTownIdx];
        for (const n of tn.originalNeighbors) {
            if (!firebreaks.has(n)) {
                firebreaks.add(n);
                budget--;
                
                // Update costs of all towns that share this neighbor
                const sharingTowns = cellToTowns.get(n);
                if (sharingTowns) {
                    for (const stIdx of sharingTowns) {
                        costs[stIdx]--;
                    }
                }
                if (budget === 0) break;
            }
        }
        activeTowns.delete(bestTownIdx);
    }

    // 4. Output results as JSON
    const firebreakCoords = Array.from(firebreaks).map(idx => [idx % width, Math.floor(idx / width)]);
    process.stdout.write(JSON.stringify({ firebreaks: firebreakCoords }));
}

solve();
