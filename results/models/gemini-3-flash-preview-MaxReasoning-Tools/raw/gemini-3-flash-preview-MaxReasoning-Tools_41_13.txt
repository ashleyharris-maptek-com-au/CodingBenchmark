; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %outColor
               OpExecutionMode %main OriginUpperLeft

; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %outColor Location 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %mat4 = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
   %worldPos = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output
       %uint = OpTypeInt 32 0
     %uint_1 = OpConstant %uint 1
%_ptr_Input_float = OpTypePointer Input %float

; Constants
  %float_0_5 = OpConstant %float 0.5
  %float_0_1 = OpConstant %float 0.1
  %float_0_4 = OpConstant %float 0.4
    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0
     %bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
        %top = OpConstantComposite %v3float %float_1 %float_1 %float_1

; Main
       %main = OpFunction %void None %3
          %5 = OpLabel
     %pos_ptr = OpAccessChain %_ptr_Input_float %worldPos %uint_1
          %y = OpLoad %float %pos_ptr
        %mul = OpFMul %float %y %float_0_5
%t_unclamped = OpFAdd %float %mul %float_0_5
          %t = OpExtInst %float %1 FClamp %t_unclamped %float_0 %float_1
      %t_vec = OpCompositeConstruct %v3float %t %t %t
        %rgb = OpExtInst %v3float %1 FMix %bottom %top %t_vec
          %r = OpCompositeExtract %float %rgb 0
          %g = OpCompositeExtract %float %rgb 1
          %b = OpCompositeExtract %float %rgb 2
      %color = OpCompositeConstruct %v4float %r %g %b %float_1
               OpStore %outColor %color
               OpReturn
               OpFunctionEnd
