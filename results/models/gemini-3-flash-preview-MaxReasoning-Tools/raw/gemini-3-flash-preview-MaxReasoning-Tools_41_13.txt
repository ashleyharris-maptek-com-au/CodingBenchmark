; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor
               OpExecutionMode %main OriginUpperLeft
               OpName %main "main"
               OpName %worldPos "worldPos"
               OpName %normal "normal"
               OpName %uv "uv"
               OpName %tangent "tangent"
               OpName %color "color"
               OpName %outColor "outColor"
               OpName %UBO "UBO"
               OpMemberName %UBO 0 "model"
               OpMemberName %UBO 1 "view"
               OpMemberName %UBO 2 "projection"
               OpMemberName %UBO 3 "lightPos"
               OpMemberName %UBO 4 "cameraPos"
               OpMemberName %UBO 5 "params"
               OpName %ubo "ubo"
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %uv Location 2
               OpDecorate %tangent Location 3
               OpDecorate %color Location 4
               OpDecorate %outColor Location 0
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %UBO Block
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4
        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform
%_ptr_Input_v3float = OpTypePointer Input %v3float
   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input
%_ptr_Input_v2float = OpTypePointer Input %v2float
         %uv = OpVariable %_ptr_Input_v2float Input
    %tangent = OpVariable %_ptr_Input_v3float Input
      %color = OpVariable %_ptr_Input_v3float Input
%_ptr_Output_v4float = OpTypePointer Output %v4float
   %outColor = OpVariable %_ptr_Output_v4float Output
  %float_0_5 = OpConstant %float 0.5
  %float_0_0 = OpConstant %float 0.0
  %float_1_0 = OpConstant %float 1.0
  %float_0_1 = OpConstant %float 0.1
  %float_0_4 = OpConstant %float 0.4
%vec3_bottom = OpConstantComposite %v3float %float_0_1 %float_0_4 %float_0_1
%vec3_top    = OpConstantComposite %v3float %float_1_0 %float_1_0 %float_1_0
       %main = OpFunction %void None %3
          %5 = OpLabel
    %pos_val = OpLoad %v3float %worldPos
      %pos_y = OpCompositeExtract %float %pos_val 1
      %val_0 = OpFMul %float %pos_y %float_0_5
      %val_1 = OpFAdd %float %val_0 %float_0_5
          %t = OpExtInst %float %1 FClamp %val_1 %float_0_0 %float_1_0
      %t_vec = OpCompositeConstruct %v3float %t %t %t
    %mix_res = OpExtInst %v3float %1 FMix %vec3_bottom %vec3_top %t_vec
      %res_x = OpCompositeExtract %float %mix_res 0
      %res_y = OpCompositeExtract %float %mix_res 1
      %res_z = OpCompositeExtract %float %mix_res 2
    %out_val = OpCompositeConstruct %v4float %res_x %res_y %res_z %float_1_0
               OpStore %outColor %out_val
               OpReturn
               OpFunctionEnd
