This script implements Donald Knuth's **Algorithm X** to solve the **Exact Cover** problem. It uses a dictionary-based approach to simulate the efficient "Dancing Links" data structure, enabling fast backtracking with pruning. The solver includes a heuristic to select the most constrained elements first, significantly reducing the search space.

```python
import sys

# Increase recursion depth for deep backtracking trees
sys.setrecursionlimit(10000)

def solve_exact_cover(u_count, sets):
    """
    Solves the Exact Cover problem using Algorithm X.
    
    Args:
        u_count: Number of elements in the universe (0..U-1).
        sets: A list of lists, where each sublist contains element IDs.
        
    Returns:
        A sorted list of set indices that form an exact cover, or None if no solution exists.
    """
    # columns[e] = set of indices of rows (sets) that cover element e
    columns = {i: set() for i in range(u_count)}
    for i, s in enumerate(sets):
        for e in s:
            if 0 <= e < u_count:
                columns[e].add(i)

    rows = {i: s for i, s in enumerate(sets)}
    
    ans = []

    def select(r):
        """Selects a row and removes overlapping rows and covered columns."""
        cols_removed = {}
        for j in rows[r]:
            if j not in columns:
                continue
            for i in columns[j]:
                for k in rows[i]:
                    if k != j:
                        if i in columns[k]:
                            columns[k].remove(i)
            cols_removed[j] = columns.pop(j)
        return cols_removed

    def deselect(r, cols_removed):
        """Restores the state after backtracking from a selected row."""
        for j in reversed(rows[r]):
            columns[j] = cols_removed[j]
            for i in columns[j]:
                for k in rows[i]:
                    if k != j:
                        columns[k].add(i)

    def solve():
        """Recursive backtracking function."""
        if not columns:
            return True
        
        # Heuristic: choose the column covered by the fewest remaining rows
        c = min(columns, key=lambda x: len(columns[x]))
        if not columns[c]:
            return False
        
        # Try each row that covers column c
        # Sorting ensures deterministic behavior
        row_choices = sorted(list(columns[c]))
        for r in row_choices:
            ans.append(r)
            cols_removed = select(r)
            if solve():
                return True
            deselect(r, cols_removed)
            ans.pop()
        
        return False

    if solve():
        return sorted(ans)
    else:
        return None

def main():
    # Read U and S
    line1 = sys.stdin.readline()
    if not line1:
        return
    parts = line1.split()
    if len(parts) < 2:
        return
    try:
        u_count = int(parts[0])
        s_count = int(parts[1])
    except ValueError:
        return
    
    # Read the sets
    sets = []
    for _ in range(s_count):
        line = sys.stdin.readline()
        if not line:
            sets.append([])
            continue
        try:
            sets.append([int(x) for x in line.split()])
        except ValueError:
            sets.append([])
    
    # Solve and output result
    result = solve_exact_cover(u_count, sets)
    
    if result is not None:
        print("SOLUTION")
        print(" ".join(map(str, result)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    main()
```
