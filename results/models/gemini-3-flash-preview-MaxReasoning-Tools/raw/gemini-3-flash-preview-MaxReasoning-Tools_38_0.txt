import sys

# Increase recursion depth for deep searches
sys.setrecursionlimit(100000)

def solve():
    input_iter = sys.stdin
    
    # Read U and S
    line = next(input_iter, None)
    while line is not None and not line.strip():
        line = next(input_iter, None)
    if line is None:
        return
    
    try:
        parts = line.split()
        if len(parts) < 2:
            return
        u, s = int(parts[0]), int(parts[1])
    except ValueError:
        return
    
    # X: element -> set of set_indices that cover it
    X = {i: set() for i in range(u)}
    # Y: set_index -> list of elements it covers (within 0..u-1)
    Y = {}
    
    for i in range(s):
        line = next(input_iter, None)
        if line is not None:
            # Use a set to remove duplicates and ignore elements outside universe
            try:
                elements = {int(x) for x in line.split() if x.strip()}
                Y[i] = [e for e in elements if 0 <= e < u]
                for e in Y[i]:
                    X[e].add(i)
            except ValueError:
                Y[i] = []
        else:
            Y[i] = []

    solution = []

    def select(r):
        """Selects set r and covers all elements it satisfies."""
        cols = {}
        for j in Y[r]:
            # For each element j covered by set r
            for i in list(X[j]):
                # For each set i that also covers element j
                for k in Y[i]:
                    # For each element k covered by set i
                    if k != j and k in X:
                        X[k].remove(i)
            # Remove element j from the universe
            cols[j] = X.pop(j)
        return cols

    def deselect(r, cols):
        """Deselects set r and restores all covered elements."""
        for j in reversed(Y[r]):
            # Restore element j to the universe
            X[j] = cols[j]
            for i in X[j]:
                # Restore set i to all elements k it covers
                for k in Y[i]:
                    if k != j and k in X:
                        X[k].add(i)

    def algo_x():
        """Recursive Algorithm X with backtracking and MRV heuristic."""
        if not X:
            return True
        
        # MRV: pick column (element) with fewest covering rows (sets)
        best_c = -1
        min_len = float('inf')
        for c in X:
            l = len(X[c])
            if l < min_len:
                min_len = l
                best_c = c
                if l == 0:
                    break
        
        if best_c == -1 or min_len == 0:
            return False
        
        # Try each set r that covers best_c
        for r in list(X[best_c]):
            solution.append(r)
            cols_removed = select(r)
            if algo_x():
                return True
            deselect(r, cols_removed)
            solution.pop()
        
        return False

    if algo_x():
        print("SOLUTION")
        # Output sorted 0-based indices
        print(*(sorted(solution)))
    else:
        print("NO SOLUTION")

if __name__ == "__main__":
    solve()
