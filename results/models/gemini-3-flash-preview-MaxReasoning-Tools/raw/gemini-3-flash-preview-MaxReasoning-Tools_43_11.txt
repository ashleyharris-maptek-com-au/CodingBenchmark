struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // 6x6 grid
    float2 grid = input.uv * 6.0;
    // Coordinates within each cell centered at (0,0)
    float2 localCoord = frac(grid) - 0.5;
    // Distance from the center of the cell
    float dist = length(localCoord);
    
    // Polka dot pattern: yellow dots on dark blue background
    float3 baseColor = (dist < 0.3) ? float3(1.0, 0.8, 0.0) : float3(0.1, 0.1, 0.3);
    
    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(0.0, dot(N, L));
    float factor = 0.3 + 0.7 * diffuse;
    
    return float4(baseColor * factor, 1.0);
}
