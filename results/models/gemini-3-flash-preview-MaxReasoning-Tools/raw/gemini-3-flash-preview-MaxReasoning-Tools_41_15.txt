; SPIR-V
; Version: 1.0
               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %inNormal %inWorldPos %outColor
               OpExecutionMode %main OriginUpperLeft

               ; Decorations
               OpDecorate %inNormal Location 1
               OpDecorate %inWorldPos Location 0
               OpDecorate %outColor Location 0
               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %uboVar DescriptorSet 0
               OpDecorate %uboVar Binding 0

               ; Types
       %void = OpTypeVoid
    %fn_type = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

               ; Variables
   %inNormal = OpVariable %ptr_Input_v3float Input
 %inWorldPos = OpVariable %ptr_Input_v3float Input
   %outColor = OpVariable %ptr_Output_v4float Output
     %uboVar = OpVariable %ptr_Uniform_UBO Uniform

               ; Constants
        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3
  %float_0_2 = OpConstant %float 0.2
 %float_0_75 = OpConstant %float 0.75
  %float_0_7 = OpConstant %float 0.7
  %float_0_4 = OpConstant %float 0.4
  %float_0_5 = OpConstant %float 0.5
    %float_1 = OpConstant %float 1.0

 %cool_color = OpConstantComposite %v3float %float_0_2 %float_0_2 %float_0_75
 %warm_color = OpConstantComposite %v3float %float_0_7 %float_0_7 %float_0_4

               ; Main function
       %main = OpFunction %void None %fn_type
 %main_label = OpLabel

               ; Normalize surface normal
   %normal_in = OpLoad %v3float %inNormal
          %N = OpExtInst %v3float %1 Normalize %normal_in

               ; Load world position
 %worldPos_in = OpLoad %v3float %inWorldPos

               ; Calculate light direction (L)
  %light_ptr = OpAccessChain %ptr_Uniform_v4float %uboVar %int_3
   %light_v4 = OpLoad %v4float %light_ptr
   %light_v3 = OpVectorShuffle %v3float %light_v4 %light_v4 0 1 2
      %L_dir = OpFSub %v3float %light_v3 %worldPos_in
          %L = OpExtInst %v3float %1 Normalize %L_dir

               ; Gooch warm/cool shading calculation
      %dotNL = OpDot %float %N %L
      %t_mul = OpFMul %float %dotNL %float_0_5
          %t = OpFAdd %float %t_mul %float_0_5
     %mix_v3 = OpExtInst %v3float %1 FMix %cool_color %warm_color %t

               ; Final color output with alpha 1.0
     %out_v4 = OpCompositeConstruct %v4float %mix_v3 %float_1
               OpStore %outColor %out_v4

               OpReturn
               OpFunctionEnd
