using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Linq;

namespace OrbitalTSP
{
    class Program
    {
        struct Vector3
        {
            public double X, Y, Z;
            public Vector3(double x, double y, double z) { X = x; Y = y; Z = z; }
            public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
            public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
            public static Vector3 operator *(Vector3 a, double b) => new Vector3(a.X * b, a.Y * b, a.Z * b);
            public double LengthSquared() => X * X + Y * Y + Z * Z;
            public double Length() => Math.Sqrt(LengthSquared());
        }

        struct State
        {
            public Vector3 Pos;
            public Vector3 Vel;
        }

        static int N;
        static double Mu;
        static State StartState;
        static State[] Stations;
        static double[][] Dist;
        static double[] StartDist;

        static void Main(string[] args)
        {
            Stopwatch sw = Stopwatch.StartNew();

            string line1 = Console.ReadLine();
            while (line1 != null && string.IsNullOrWhiteSpace(line1)) line1 = Console.ReadLine();
            if (line1 == null) return;

            string[] parts1 = line1.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts1.Length < 2) return;
            N = int.Parse(parts1[0]);
            Mu = double.Parse(parts1[1]);

            StartState = ReadState();
            Stations = new State[N];
            for (int i = 0; i < N; i++)
            {
                Stations[i] = ReadState();
            }

            // Calculate tau (time scale) based on average radius to balance position and velocity differences.
            // tau ~ sqrt(r^3 / mu)
            double sumR = StartState.Pos.Length();
            for (int i = 0; i < N; i++) sumR += Stations[i].Pos.Length();
            double avgR = sumR / (N + 1);
            double tau = Math.Sqrt(Math.Pow(avgR, 3) / Mu);

            // Precompute distance matrix
            Dist = new double[N][];
            for (int i = 0; i < N; i++)
            {
                Dist[i] = new double[N];
            }

            for (int i = 0; i < N; i++)
            {
                for (int j = i + 1; j < N; j++)
                {
                    double d = CalculateDist(Stations[i], Stations[j], tau);
                    Dist[i][j] = d;
                    Dist[j][i] = d;
                }
            }

            StartDist = new double[N];
            for (int i = 0; i < N; i++)
            {
                StartDist[i] = CalculateDist(StartState, Stations[i], tau);
            }

            int[] bestPath = null;
            double bestCost = double.MaxValue;

            // Strategy: Try multiple starting heuristics and refine with 2-opt.
            // 1. Try each node as the first hop for a Greedy Nearest Neighbor path.
            for (int firstHop = 0; firstHop < N; firstHop++)
            {
                if (sw.ElapsedMilliseconds > 60000) break; // 1 minute limit for Phase 1

                int[] path = GreedyNN(firstHop);
                path = Optimize2Opt(path);
                double cost = GetPathCost(path);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestPath = (int[])path.Clone();
                }
            }

            // 2. Random restarts with 2-opt if time permits.
            Random rand = new Random(42);
            int restartLimit = N <= 10 ? 100 : 100000;
            int restarts = 0;
            while (sw.ElapsedMilliseconds < 240000 && restarts < restartLimit) // 4 minute total limit
            {
                restarts++;
                int[] path = Enumerable.Range(0, N).ToArray();
                for (int i = N - 1; i > 0; i--)
                {
                    int k = rand.Next(i + 1);
                    int tmp = path[i];
                    path[i] = path[k];
                    path[k] = tmp;
                }

                path = Optimize2Opt(path);
                double cost = GetPathCost(path);
                if (cost < bestCost)
                {
                    bestCost = cost;
                    bestPath = (int[])path.Clone();
                }
            }

            if (bestPath != null)
            {
                Console.WriteLine(string.Join(" ", bestPath));
            }
            else if (N > 0)
            {
                // Fallback for extreme cases
                Console.WriteLine(string.Join(" ", Enumerable.Range(0, N)));
            }
        }

        static State ReadState()
        {
            while (true)
            {
                string line = Console.ReadLine();
                if (line == null) return new State();
                string[] parts = line.Split(new[] { ' ', '\t', ':' }, StringSplitOptions.RemoveEmptyEntries);
                if (parts.Length >= 6)
                {
                    int s = parts.Length - 6;
                    return new State
                    {
                        Pos = new Vector3(double.Parse(parts[s]), double.Parse(parts[s+1]), double.Parse(parts[s+2])),
                        Vel = new Vector3(double.Parse(parts[s+3]), double.Parse(parts[s+4]), double.Parse(parts[s+5]))
                    };
                }
            }
        }

        static double CalculateDist(State s1, State s2, double tau)
        {
            double dx = s1.Pos.X - s2.Pos.X;
            double dy = s1.Pos.Y - s2.Pos.Y;
            double dz = s1.Pos.Z - s2.Pos.Z;
            double dvx = s1.Vel.X - s2.Vel.X;
            double dvy = s1.Vel.Y - s2.Vel.Y;
            double dvz = s1.Vel.Z - s2.Vel.Z;
            // Weighted Euclidean distance in phase space.
            return Math.Sqrt(dx * dx + dy * dy + dz * dz + tau * tau * (dvx * dvx + dvy * dvy + dvz * dvz));
        }

        static int[] GreedyNN(int firstNode)
        {
            int[] path = new int[N];
            bool[] visited = new bool[N];
            path[0] = firstNode;
            visited[firstNode] = true;

            int current = firstNode;
            for (int i = 1; i < N; i++)
            {
                int next = -1;
                double minDist = double.MaxValue;
                for (int j = 0; j < N; j++)
                {
                    if (!visited[j])
                    {
                        double d = Dist[current][j];
                        if (d < minDist)
                        {
                            minDist = d;
                            next = j;
                        }
                    }
                }
                path[i] = next;
                visited[next] = true;
                current = next;
            }
            return path;
        }

        static int[] Optimize2Opt(int[] path)
        {
            bool improved = true;
            while (improved)
            {
                improved = false;
                for (int i = 0; i < N; i++)
                {
                    for (int j = i + 1; j < N; j++)
                    {
                        // Cost before swap:
                        double d_prev_i = (i == 0) ? StartDist[path[i]] : Dist[path[i - 1]][path[i]];
                        double d_j_next = (j == N - 1) ? 0 : Dist[path[j]][path[j + 1]];

                        // Cost after swap (path segment i..j is reversed):
                        double d_prev_j = (i == 0) ? StartDist[path[j]] : Dist[path[i - 1]][path[j]];
                        double d_i_next = (j == N - 1) ? 0 : Dist[path[i]][path[j + 1]];

                        if (d_prev_j + d_i_next < d_prev_i + d_j_next - 1e-9)
                        {
                            Array.Reverse(path, i, j - i + 1);
                            improved = true;
                        }
                    }
                }
            }
            return path;
        }

        static double GetPathCost(int[] path)
        {
            double cost = StartDist[path[0]];
            for (int i = 0; i < N - 1; i++)
            {
                cost += Dist[path[i]][path[i + 1]];
            }
            return cost;
        }
    }
}
