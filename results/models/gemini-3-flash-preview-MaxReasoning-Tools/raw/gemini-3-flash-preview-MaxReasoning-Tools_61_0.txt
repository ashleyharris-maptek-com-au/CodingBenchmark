"use strict";

declare const require: any;
declare const process: any;
const fs = require('fs');

/**
 * Reads the input JSON from stdin, plans delivery routes for a fleet of vans,
 * and prints the result as a JSON document to stdout.
 * 
 * Strategy:
 * 1. Partition packages into angular slices around the depot.
 * 2. Within each slice, use a Greedy Nearest Neighbor heuristic to form a route.
 * 3. Optimize the search for the nearest neighbor using a windowed approach
 *    to ensure the algorithm scales to large datasets.
 */
function main() {
    let input: string;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;

    let data: any;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const { depot, truckCount, packages } = data;
    if (!packages || packages.length === 0) {
        process.stdout.write(JSON.stringify({ routes: [] }) + "\n");
        return;
    }

    const [depotX, depotY] = depot;
    const n = packages.length;

    // 1. Calculate angles for angular sweep
    const pData = packages.map((p: any) => ({
        id: p.id,
        x: p.x,
        y: p.y,
        angle: Math.atan2(p.y - depotY, p.x - depotX)
    }));

    // 2. Sort by angle to partition the space into "slices"
    // This provides a good global structure for the routes.
    pData.sort((a: any, b: any) => a.angle - b.angle);

    // 3. Partition into routes based on the maximum number of vans
    const m = Math.max(1, Math.min(truckCount, n));
    const resultRoutes: any[][] = [];

    for (let i = 0; i < m; i++) {
        // Calculate the range of packages for this route
        const start = Math.floor((i * n) / m);
        const end = Math.floor(((i + 1) * n) / m);
        const group = pData.slice(start, end);
        if (group.length === 0) continue;

        const route: any[] = [];
        let currX = depotX;
        let currY = depotY;
        const used = new Uint8Array(group.length);
        let remainingCount = group.length;
        let lastIdx = 0;

        // Greedy Nearest Neighbor within each angular slice
        while (remainingCount > 0) {
            let bestIdx = -1;
            let minDistSq = Infinity;

            // Efficiently search for the nearest neighbor within a window in the angularly sorted list.
            // This maintains O(N) scaling for the routing phase in the common case.
            const searchWindow = 500;
            const sStart = Math.max(0, lastIdx - searchWindow);
            const sEnd = Math.min(group.length, lastIdx + searchWindow);

            for (let j = sStart; j < sEnd; j++) {
                if (used[j]) continue;
                const p = group[j];
                const dx = p.x - currX;
                const dy = p.y - currY;
                const d2 = dx * dx + dy * dy;
                if (d2 < minDistSq) {
                    minDistSq = d2;
                    bestIdx = j;
                }
            }

            // Fallback: if all packages in the window are used, search the entire group.
            if (bestIdx === -1) {
                for (let j = 0; j < group.length; j++) {
                    if (used[j]) continue;
                    const p = group[j];
                    const dx = p.x - currX;
                    const dy = p.y - currY;
                    const d2 = dx * dx + dy * dy;
                    if (d2 < minDistSq) {
                        minDistSq = d2;
                        bestIdx = j;
                    }
                }
            }

            if (bestIdx !== -1) {
                used[bestIdx] = 1;
                const p = group[bestIdx];
                route.push(p.id);
                currX = p.x;
                currY = p.y;
                lastIdx = bestIdx;
                remainingCount--;
            } else {
                break;
            }
        }
        resultRoutes.push(route);
    }

    // 4. Output the result JSON
    process.stdout.write(JSON.stringify({ routes: resultRoutes }) + "\n");
}

main();
