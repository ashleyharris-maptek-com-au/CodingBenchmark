This TypeScript program implements an efficient vehicle routing solver using the **Clarke-Wright Savings algorithm** for small-to-medium scenarios (up to 5,000 packages) and a highly scalable **Sweep algorithm** for larger datasets.

The Clarke-Wright heuristic iteratively merges routes that provide the greatest distance savings while respecting the maximum van count. For extremely large scenarios, the Sweep algorithm partitions packages by their polar coordinates relative to the depot, ensuring $O(N \log N)$ performance and minimal memory overhead.

```typescript
"use strict";

declare const require: any;
const fs = require('fs');

/**
 * Plans delivery routes for a fleet of vans to minimize total travel distance.
 * Uses Clarke-Wright Savings algorithm for optimal heuristics and a Sweep algorithm for scalability.
 */
function solve() {
    let input: string;
    try {
        // Read the entire input from stdin
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;

    let data: any;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const depot: [number, number] = data.depot;
    const truckCount: number = data.truckCount;
    const packages: { id: string, x: number, y: number }[] = data.packages;

    // Handle edge case of no packages
    if (!packages || packages.length === 0) {
        console.log(JSON.stringify({ routes: [] }));
        return;
    }

    const n = packages.length;
    const depotX = depot[0];
    const depotY = depot[1];

    /**
     * Calculates Euclidean distance between two points.
     */
    function getDist(x1: number, y1: number, x2: number, y2: number) {
        const dx = x1 - x2;
        const dy = y1 - y2;
        return Math.sqrt(dx * dx + dy * dy);
    }

    // SCALABILITY FALLBACK: Use Sweep Algorithm for very large N (> 5000)
    // Complexity: O(N log N)
    if (n > 5000) {
        const packagesWithAngle = packages.map(p => ({
            id: p.id,
            angle: Math.atan2(p.y - depotY, p.x - depotX)
        }));
        // Sort by angle around the depot
        packagesWithAngle.sort((a, b) => a.angle - b.angle);

        const routes: string[][] = [];
        const m = Math.min(truckCount, n);
        for (let i = 0; i < m; i++) {
            const start = Math.floor((i * n) / m);
            const end = Math.floor(((i + 1) * n) / m);
            const slice = packagesWithAngle.slice(start, end);
            if (slice.length > 0) {
                routes.push(slice.map(p => p.id));
            }
        }
        console.log(JSON.stringify({ routes }));
        return;
    }

    // MAIN ALGORITHM: Clarke-Wright Savings
    // Complexity: O(N^2 log N) for sorting pairs
    const distToDepot = new Float64Array(n);
    for (let i = 0; i < n; i++) {
        distToDepot[i] = getDist(packages[i].x, packages[i].y, depotX, depotY);
    }

    // Calculate savings for all pairs of packages
    // Saving S(i, j) = dist(D, i) + dist(D, j) - dist(i, j)
    const savings: { i: number, j: number, s: number }[] = [];
    for (let i = 0; i < n; i++) {
        const pi = packages[i];
        for (let j = i + 1; j < n; j++) {
            const pj = packages[j];
            const d_ij = getDist(pi.x, pi.y, pj.x, pj.y);
            const s = distToDepot[i] + distToDepot[j] - d_ij;
            savings.push({ i, j, s });
        }
    }

    // Sort savings in descending order
    savings.sort((a, b) => b.s - a.s);

    // Initialize each package in its own route
    const routes: number[][] = [];
    const packageToRouteId = new Int32Array(n);
    for (let i = 0; i < n; i++) {
        routes.push([i]);
        packageToRouteId[i] = i;
    }

    let routeCount = n;
    const targetRouteCount = Math.min(truckCount, n);

    // Process merges based on highest savings
    for (let k = 0; k < savings.length; k++) {
        if (routeCount <= targetRouteCount) break;

        const { i, j } = savings[k];
        const r1Id = packageToRouteId[i];
        const r2Id = packageToRouteId[j];

        // Skip if they are already in the same route
        if (r1Id === r2Id) continue;

        const r1 = routes[r1Id];
        const r2 = routes[r2Id];

        // i and j must be at the endpoints of their respective routes to allow merging
        const isIStart = r1[0] === i;
        const isIEnd = r1[r1.length - 1] === i;
        const isJStart = r2[0] === j;
        const isJEnd = r2[r2.length - 1] === j;

        if ((isIStart || isIEnd) && (isJStart || isJEnd)) {
            let merged: number[];
            // Determine relative orientation for the merge
            if (isIEnd && isJStart) {
                merged = r1.concat(r2);
            } else if (isIEnd && isJEnd) {
                const len2 = r2.length;
                const reversedR2 = new Array(len2);
                for (let l = 0; l < len2; l++) reversedR2[l] = r2[len2 - 1 - l];
                merged = r1.concat(reversedR2);
            } else if (isIStart && isJStart) {
                const len1 = r1.length;
                const reversedR1 = new Array(len1);
                for (let l = 0; l < len1; l++) reversedR1[l] = r1[len1 - 1 - l];
                merged = reversedR1.concat(r2);
            } else { // isIStart && isJEnd
                merged = r2.concat(r1);
            }

            // Update route mapping and count
            routes[r1Id] = merged;
            for (let pIdx = 0; pIdx < merged.length; pIdx++) {
                packageToRouteId[merged[pIdx]] = r1Id;
            }
            routes[r2Id] = [];
            routeCount--;
        }
    }

    // Format output JSON
    const finalRoutes = routes
        .filter(r => r.length > 0)
        .map(r => r.map(pIdx => packages[pIdx].id));

    console.log(JSON.stringify({ routes: finalRoutes }));
}

// Execute the solver
solve();
```
