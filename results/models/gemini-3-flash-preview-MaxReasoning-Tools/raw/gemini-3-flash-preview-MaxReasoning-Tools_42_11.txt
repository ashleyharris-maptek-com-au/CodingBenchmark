```spirv
OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%glsl = OpExtInstImport "GLSL.std.450"

%void = OpTypeVoid
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%void_fn = OpTypeFunction %void

%uint_4 = OpConstant %uint 4
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3

%float_0 = OpConstant %float 0
%float_0_01 = OpConstant %float 0.01
%float_0_333333 = OpConstant %float 0.33333334
%float_0_666667 = OpConstant %float 0.66666667

; Interface Variables
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_4
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_4
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointers
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%main = OpFunction %void None %void_fn
%label = OpLabel

; Load positions p0, p1, p2, p3 from gl_in
%ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %ptr_p0
%ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %ptr_p1
%ptr_p2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %ptr_p2
%ptr_p3 = OpAccessChain %ptr_in_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %ptr_p3

; Load colors c1, c2
%ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
%c1 = OpLoad %v3float %ptr_c1
%ptr_c2 = OpAccessChain %ptr_in_v3 %color %int_2
%c2 = OpLoad %v3float %ptr_c2

; Compute smooth perpendicular N1 at p1 using adjacency p0, p2
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%diff1 = OpFSub %v2float %p2_xy %p0_xy
%T1 = OpExtInst %v2float %glsl Normalize %diff1
%T1x = OpCompositeExtract %float %T1 0
%T1y = OpCompositeExtract %float %T1 1
%N1x = OpFNegate %float %T1y
%N1y = %T1x
%N1_dir = OpCompositeConstruct %v2float %N1x %N1y
%N1 = OpVectorTimesScalar %v2float %N1_dir %float_0_01

; Compute smooth perpendicular N2 at p2 using adjacency p1, p3
%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%diff2 = OpFSub %v2float %p3_xy %p1_xy
%T2 = OpExtInst %v2float %glsl Normalize %diff2
%T2x = OpCompositeExtract %float %T2 0
%T2y = OpCompositeExtract %float %T2 1
%N2x = OpFNegate %float %T2y
%N2y = %T2x
%N2_dir = OpCompositeConstruct %v2float %N2x %N2y
%N2 = OpVectorTimesScalar %v2float %N2_dir %float_0_01

; Split segment [p1, p2] into 3 equal parts: [p1, pA], [pA, pB], [pB, p2]
%seg = OpFSub %v4float %p2 %p1
%pA_off = OpVectorTimesScalar %v4float %seg %float_0_333333
%pA = OpFAdd %v4float %p1 %pA_off
%pB_off = OpVectorTimesScalar %v4float %seg %float_0_666667
%pB = OpFAdd %v4float %p1 %pB_off

; Interpolate colors and perpendiculars for internal points A and B
%cdiff = OpFSub %v3float %c2 %c1
%cA_off = OpVectorTimesScalar %v3float %cdiff %float_0_333333
%cA = OpFAdd %v3float %c1 %cA_off
%cB_off = OpVectorTimesScalar %v3float %cdiff %float_0_666667
%cB = OpFAdd %v3float %c1 %cB_off

%Ndiff = OpFSub %v2float %N2 %N1
%NA_off = OpVectorTimesScalar %v2float %Ndiff %float_0_333333
%NA = OpFAdd %v2float %N1 %NA_off
%NB_off = OpVectorTimesScalar %v2float %Ndiff %float_0_666667
%NB = OpFAdd %v2float %N1 %NB_off

%ptr_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; --- Quad 1: Dash 1 (p1 to pA) ---
%p1w = OpCompositeExtract %float %p1 3
%o1 = OpVectorTimesScalar %v2float %N1 %p1w
%o1x = OpCompositeExtract %float %o1 0
%o1y = OpCompositeExtract %float %o1 1
%vo1 = OpCompositeConstruct %v4float %o1x %o1y %float_0 %float_0
%pos1 = OpFAdd %v4float %p1 %vo1
OpStore %outColor %c1
OpStore %ptr_pos %pos1
OpEmitVertex

%pos2 = OpFSub %v4float %p1 %vo1
OpStore %outColor %c1
OpStore %ptr_pos %pos2
OpEmitVertex

%pAw = OpCompositeExtract %float %pA 3
%oA = OpVectorTimesScalar %v2float %NA %pAw
%oAx = OpCompositeExtract %float %oA 0
%oAy = OpCompositeExtract %float %oA 1
%voA = OpCompositeConstruct %v4float %oAx %oAy %float_0 %float_0
%pos3 = OpFAdd %v4float %pA %voA
OpStore %outColor %cA
OpStore %ptr_pos %pos3
OpEmitVertex

%pos4 = OpFSub %v4float %pA %voA
OpStore %outColor %cA
OpStore %ptr_pos %pos4
OpEmitVertex

OpEndPrimitive

; --- Quad 2: Dash 2 (pB to p2) ---
%pBw = OpCompositeExtract %float %pB 3
%oB = OpVectorTimesScalar %v2float %NB %pBw
%oBx = OpCompositeExtract %float %oB 0
%oBy = OpCompositeExtract %float %oB 1
%voB = OpCompositeConstruct %v4float %oBx %oBy %float_0 %float_0
%pos5 = OpFAdd %v4float %pB %voB
OpStore %outColor %cB
OpStore %ptr_pos %pos5
OpEmitVertex

%pos6 = OpFSub %v4float %pB %voB
OpStore %outColor %cB
OpStore %ptr_pos %pos6
OpEmitVertex

%p2w = OpCompositeExtract %float %p2 3
%o2 = OpVectorTimesScalar %v2float %N2 %p2w
%o2x = OpCompositeExtract %float %o2 0
%o2y = OpCompositeExtract %float %o2 1
%vo2 = OpCompositeConstruct %v4float %o2x %o2y %float_0 %float_0
%pos7 = OpFAdd %v4float %p2 %vo2
OpStore %outColor %c2
OpStore %ptr_pos %pos7
OpEmitVertex

%pos8 = OpFSub %v4float %p2 %vo2
OpStore %outColor %c2
OpStore %ptr_pos %pos8
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
