OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor
OpExecutionMode %main InputLinesAdjacency
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 8
OpExecutionMode %main Invocations 1

; Decorations
OpMemberDecorate %gl_PerVertex 0 BuiltIn Position
OpDecorate %gl_PerVertex Block
OpDecorate %color Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%gl_PerVertex = OpTypeStruct %v4float
%uint_4 = OpConstant %uint 4
%arr4_pv = OpTypeArray %gl_PerVertex %uint_4
%ptr_Input_arr4_pv = OpTypePointer Input %arr4_pv
%gl_in = OpVariable %ptr_Input_arr4_pv Input

%ptr_Output_pv = OpTypePointer Output %gl_PerVertex
%gl_out = OpVariable %ptr_Output_pv Output

%arr4_v3 = OpTypeArray %v3float %uint_4
%ptr_Input_arr4_v3 = OpTypePointer Input %arr4_v3
%color = OpVariable %ptr_Input_arr4_v3 Input

%ptr_Output_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_Output_v3 Output

%ptr_Input_v4 = OpTypePointer Input %v4float
%ptr_Output_v4 = OpTypePointer Output %v4float
%ptr_Input_v3 = OpTypePointer Input %v3float

; Constants
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_2 = OpConstant %int 2
%int_3 = OpConstant %int 3
%float_0_01 = OpConstant %float 0.01
%float_0_333333 = OpConstant %float 0.33333334
%float_0_666666 = OpConstant %float 0.66666667

%main = OpFunction %void None %void_func
%label = OpLabel

; Load positions
%p0_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr
%p2_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_2 %int_0
%p2 = OpLoad %v4float %p2_ptr
%p3_ptr = OpAccessChain %ptr_Input_v4 %gl_in %int_3 %int_0
%p3 = OpLoad %v4float %p3_ptr

; Load colors
%c1_ptr = OpAccessChain %ptr_Input_v3 %color %int_1
%c1 = OpLoad %v3float %c1_ptr
%c2_ptr = OpAccessChain %ptr_Input_v3 %color %int_2
%c2 = OpLoad %v3float %c2_ptr

; NDC conversion for tangent calculation
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p0_w = OpCompositeExtract %float %p0 3
%p0_w_v2 = OpCompositeConstruct %v2float %p0_w %p0_w
%p0_ndc = OpFDiv %v2float %p0_xy %p0_w_v2

%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p1_w = OpCompositeExtract %float %p1 3
%p1_w_v2 = OpCompositeConstruct %v2float %p1_w %p1_w
%p1_ndc = OpFDiv %v2float %p1_xy %p1_w_v2

%p2_xy = OpVectorShuffle %v2float %p2 %p2 0 1
%p2_w = OpCompositeExtract %float %p2 3
%p2_w_v2 = OpCompositeConstruct %v2float %p2_w %p2_w
%p2_ndc = OpFDiv %v2float %p2_xy %p2_w_v2

%p3_xy = OpVectorShuffle %v2float %p3 %p3 0 1
%p3_w = OpCompositeExtract %float %p3 3
%p3_w_v2 = OpCompositeConstruct %v2float %p3_w %p3_w
%p3_ndc = OpFDiv %v2float %p3_xy %p3_w_v2

; Normal at P1 (from P2 - P0)
%v1 = OpFSub %v2float %p2_ndc %p0_ndc
%t1 = OpExtInst %v2float %1 Normalize %v1
%t1_x = OpCompositeExtract %float %t1 0
%t1_y = OpCompositeExtract %float %t1 1
%neg_t1_y = OpFNegate %float %t1_y
%n1_dir = OpCompositeConstruct %v2float %neg_t1_y %t1_x
%n1 = OpVectorTimesScalar %v2float %n1_dir %float_0_01

; Normal at P2 (from P3 - P1)
%v2 = OpFSub %v2float %p3_ndc %p1_ndc
%t2 = OpExtInst %v2float %1 Normalize %v2
%t2_x = OpCompositeExtract %float %t2 0
%t2_y = OpCompositeExtract %float %t2 1
%neg_t2_y = OpFNegate %float %t2_y
%n2_dir = OpCompositeConstruct %v2float %neg_t2_y %t2_x
%n2 = OpVectorTimesScalar %v2float %n2_dir %float_0_01

; Output pointer for gl_Position
%pos_out_ptr = OpAccessChain %ptr_Output_v4 %gl_out %int_0

; --- Segment 1 (t=0 to t=1/3) ---
%n1_s0 = OpVectorTimesScalar %v2float %n1 %p1_w
%p1_xy_v1 = OpFAdd %v2float %p1_xy %n1_s0
%p1_z = OpCompositeExtract %float %p1 2
%pos1 = OpCompositeConstruct %v4float %p1_xy_v1 %p1_z %p1_w
OpStore %outColor %c1
OpStore %pos_out_ptr %pos1
OpEmitVertex

%p1_xy_v2 = OpFSub %v2float %p1_xy %n1_s0
%pos2 = OpCompositeConstruct %v4float %p1_xy_v2 %p1_z %p1_w
OpStore %outColor %c1
OpStore %pos_out_ptr %pos2
OpEmitVertex

%p_t1 = OpExtInst %v4float %1 FMix %p1 %p2 %float_0_333333
%n_t1 = OpExtInst %v2float %1 FMix %n1 %n2 %float_0_333333
%c_t1 = OpExtInst %v3float %1 FMix %c1 %c2 %float_0_333333

%pt1_xy = OpVectorShuffle %v2float %p_t1 %p_t1 0 1
%pt1_z = OpCompositeExtract %float %p_t1 2
%pt1_w = OpCompositeExtract %float %p_t1 3
%nt1_s = OpVectorTimesScalar %v2float %n_t1 %pt1_w
%pt1_xy_v3 = OpFAdd %v2float %pt1_xy %nt1_s
%pos3 = OpCompositeConstruct %v4float %pt1_xy_v3 %pt1_z %pt1_w
OpStore %outColor %c_t1
OpStore %pos_out_ptr %pos3
OpEmitVertex

%pt1_xy_v4 = OpFSub %v2float %pt1_xy %nt1_s
%pos4 = OpCompositeConstruct %v4float %pt1_xy_v4 %pt1_z %pt1_w
OpStore %outColor %c_t1
OpStore %pos_out_ptr %pos4
OpEmitVertex

OpEndPrimitive

; --- Segment 3 (t=2/3 to t=1) ---
%p_t2 = OpExtInst %v4float %1 FMix %p1 %p2 %float_0_666666
%n_t2 = OpExtInst %v2float %1 FMix %n1 %n2 %float_0_666666
%c_t2 = OpExtInst %v3float %1 FMix %c1 %c2 %float_0_666666

%pt2_xy = OpVectorShuffle %v2float %p_t2 %p_t2 0 1
%pt2_z = OpCompositeExtract %float %p_t2 2
%pt2_w = OpCompositeExtract %float %p_t2 3
%nt2_s = OpVectorTimesScalar %v2float %n_t2 %pt2_w
%pt2_xy_v5 = OpFAdd %v2float %pt2_xy %nt2_s
%pos5 = OpCompositeConstruct %v4float %pt2_xy_v5 %pt2_z %pt2_w
OpStore %outColor %c_t2
OpStore %pos_out_ptr %pos5
OpEmitVertex

%pt2_xy_v6 = OpFSub %v2float %pt2_xy %nt2_s
%pos6 = OpCompositeConstruct %v4float %pt2_xy_v6 %pt2_z %pt2_w
OpStore %outColor %c_t2
OpStore %pos_out_ptr %pos6
OpEmitVertex

%n2_s = OpVectorTimesScalar %v2float %n2 %p2_w
%p2_xy_v7 = OpFAdd %v2float %p2_xy %n2_s
%p2_z = OpCompositeExtract %float %p2 2
%pos7 = OpCompositeConstruct %v4float %p2_xy_v7 %p2_z %p2_w
OpStore %outColor %c2
OpStore %pos_out_ptr %pos7
OpEmitVertex

%p2_xy_v8 = OpFSub %v2float %p2_xy %n2_s
%pos8 = OpCompositeConstruct %v4float %p2_xy_v8 %p2_z %p2_w
OpStore %outColor %c2
OpStore %pos_out_ptr %pos8
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
