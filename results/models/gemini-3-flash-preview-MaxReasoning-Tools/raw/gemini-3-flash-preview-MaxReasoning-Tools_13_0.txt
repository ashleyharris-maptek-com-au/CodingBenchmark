The Longest Common Substring problem for multiple strings is solved using an optimized rolling hash approach with binary search on the substring length. This implementation is designed to handle scales from trivial to "extreme" (100 strings of 10MB+ each) by utilizing O(ΣL log L) time complexity and O(ΣL) memory complexity.

### Technical Details:
- **Scalability**: The algorithm adapts to any number and length of strings. By using the shortest string as a baseline, the search space for candidate substrings is minimized.
- **Rolling Hash**: A 61-bit rolling hash using the Mersenne prime $2^{61}-1$ provides a massive hash space, making collisions extremely unlikely even across billions of substrings. MSVC's `_umul128` intrinsic is used for efficient 128-bit multiplication modulo $2^{61}-1$.
- **Performance**:
    - **Binary Search**: Efficiently finds the maximum possible length.
    - **Candidate Pruning**: A custom open-addressed hash table tracks common substrings across strings. After each string is processed, candidates that do not appear in it are pruned, significantly speeding up subsequent steps.
    - **Optimized Hash Table**: Power-of-two sizing and bitwise masking ensure fast lookups, while memory caps prevent exceeding the 32GB RAM limit.
- **Robustness**: Handles edge cases such as $N=1$, no common substrings, and varied character sets (via byte-level hashing).

```cpp
#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
#include <random>
#include <cstdint>
#include <intrin.h>

using namespace std;

typedef uint64_t u64;
typedef uint32_t u32;

// MOD is 2^61 - 1, a Mersenne prime.
const u64 MOD = (1ULL << 61) - 1;

#pragma intrinsic(_umul128)

/**
 * Multiplies two 64-bit integers and returns the result modulo 2^61 - 1.
 * Uses the property that 2^61 \equiv 1 (mod 2^61 - 1).
 */
inline u64 mul_mod(u64 a, u64 b) {
    u64 high, low;
    low = _umul128(a, b, &high);
    // q = (a*b) >> 61, r = (a*b) & (2^61 - 1)
    // a*b = q * 2^61 + r \equiv q + r (mod 2^61 - 1)
    u64 res = (high << 3) | (low >> 61);
    res += low & MOD;
    if (res >= MOD) res -= MOD;
    return res;
}

inline u64 add_mod(u64 a, u64 b) {
    a += b;
    if (a >= MOD) a -= MOD;
    return a;
}

inline u64 sub_mod(u64 a, u64 b) {
    return a >= b ? a - b : a + MOD - b;
}

/**
 * MurmurHash3 finalizer for mixing 64-bit hashes.
 */
inline u32 hash_u64(u64 h) {
    h ^= h >> 33;
    h *= 0xff51afd7ed558ccdULL;
    h ^= h >> 33;
    h *= 0xc4ceb9fe1a85ec53ULL;
    h ^= h >> 33;
    return (u32)h;
}

struct Entry {
    u64 h;
    int last_id;
    int pos;
};

// Global table and mask to avoid repeated allocations where possible.
vector<Entry> table;
u32 table_mask;

void init_table(u32 size) {
    u64 n = 1;
    // Aim for a load factor around 0.5 - 0.7 for performance.
    while (n < (u64)size * 15 / 10) {
        n <<= 1;
        if (n >= (1ULL << 30)) break; // Cap at 1B entries (~16GB) to avoid OOM
    }
    if (n < (u64)size + 1) n = (u64)size + 1;

    table.assign((size_t)n, {0, -1, -1});
    if ((n & (n - 1)) == 0) {
        table_mask = (u32)n - 1;
    } else {
        table_mask = 0; // Use modulo instead of mask for non-power-of-2
    }
}

inline void insert(u64 h, int pos) {
    u32 p;
    if (table_mask) p = hash_u64(h) & table_mask;
    else p = hash_u64(h) % (u32)table.size();

    while (table[p].last_id != -1) {
        if (table[p].h == h) return;
        p++;
        if (table_mask) p &= table_mask;
        else if (p >= table.size()) p = 0;
    }
    table[p] = {h, 0, pos};
}

inline void probe(u64 h, int id) {
    u32 p;
    if (table_mask) p = hash_u64(h) & table_mask;
    else p = hash_u64(h) % (u32)table.size();

    while (table[p].last_id != -1) {
        if (table[p].h == h) {
            if (table[p].last_id == id - 1) {
                table[p].last_id = id;
            }
            return;
        }
        p++;
        if (table_mask) p &= table_mask;
        else if (p >= table.size()) p = 0;
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    if (!(cin >> N)) return 0;
    string dummy;
    getline(cin, dummy); 

    if (N <= 0) {
        cout << endl;
        return 0;
    }

    vector<string> strings(N);
    int shortest_idx = 0;
    for (int i = 0; i < N; ++i) {
        if (!getline(cin, strings[i])) break;
        if (!strings[i].empty() && strings[i].back() == '\r') strings[i].pop_back();
        if (strings[i].length() < strings[shortest_idx].length()) {
            shortest_idx = i;
        }
    }

    if (N == 1) {
        cout << strings[0] << endl;
        return 0;
    }

    u64 base = 13131;
    random_device rd;
    mt19937_64 gen(rd());
    uniform_int_distribution<u64> dis(256, MOD - 1);
    base = dis(gen);
    if (base % 2 == 0) base--;

    int low = 1, high = (int)strings[shortest_idx].length();
    int ans_len = 0;
    int ans_pos = -1;

    while (low <= high) {
        int mid = low + (high - low) / 2;
        u64 powL = 1;
        for (int i = 0; i < mid; ++i) powL = mul_mod(powL, base);
        u64 char_powL[256];
        for (int c = 0; c < 256; ++c) char_powL[c] = mul_mod((u64)c, powL);
        
        init_table((u32)strings[shortest_idx].length() - mid + 1);
        const string& s_min = strings[shortest_idx];
        u64 h = 0;
        for (int i = 0; i < mid; ++i) h = add_mod(mul_mod(h, base), (u64)(unsigned char)s_min[i]);
        insert(h, 0);
        for (int i = mid; i < (int)s_min.length(); ++i) {
            h = mul_mod(h, base);
            h = add_mod(h, (u64)(unsigned char)s_min[i]);
            h = sub_mod(h, char_powL[(unsigned char)s_min[i - mid]]);
            insert(h, i - mid + 1);
        }
        
        vector<u32> active_indices;
        for (u32 j = 0; j < (u32)table.size(); ++j) {
            if (table[j].last_id == 0) active_indices.push_back(j);
        }
        
        bool possible = !active_indices.empty();
        int current_id = 1;
        if (possible) {
            for (int i = 0; i < N; ++i) {
                if (i == shortest_idx) continue;
                const string& s = strings[i];
                u64 ch = 0;
                for (int j = 0; j < mid; ++j) ch = add_mod(mul_mod(ch, base), (u64)(unsigned char)s[j]);
                probe(ch, current_id);
                for (int j = mid; j < (int)s.length(); ++j) {
                    ch = mul_mod(ch, base);
                    ch = add_mod(ch, (u64)(unsigned char)s[j]);
                    ch = sub_mod(ch, char_powL[(unsigned char)s[j - mid]]);
                    probe(ch, current_id);
                }
                
                u32 write_ptr = 0;
                for (u32 idx : active_indices) {
                    if (table[idx].last_id == current_id) active_indices[write_ptr++] = idx;
                }
                active_indices.resize(write_ptr);
                if (active_indices.empty()) { possible = false; break; }
                current_id++;
            }
        }
        
        if (possible && !active_indices.empty()) {
            ans_len = mid;
            ans_pos = table[active_indices[0]].pos;
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    if (ans_len > 0) cout << strings[shortest_idx].substr(ans_pos, ans_len) << endl;
    else cout << endl;

    return 0;
}
```
