               OpSource GLSL 450

               OpDecorate %in_normal Location 1
               OpDecorate %in_color Location 4
               OpDecorate %outColor Location 0
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16

               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 2 Offset 128

; Types
      %void = OpTypeVoid
         %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %v4float 4
       %bool = OpTypeBool

        %UBO = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float

%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

; Variables
%in_worldPos = OpVariable %ptr_Input_v3float Input
%in_normal = OpVariable %ptr_Input_v3float Input
%in_color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%ubo_var = OpVariable %ptr_Uniform_UBO Uniform

; Constants
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
 %float_0_75 = OpConstant %float 0.75
  %float_0_5 = OpConstant %float 0.5
 %float_0_25 = OpConstant %float 0.25
  %float_0_7 = OpConstant %float 0.7
  %float_0_4 = OpConstant %float 0.4
  %float_0_2 = OpConstant %float 0.2

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3

; Main entry point
       %main = OpFunction %void None %3
          %5 = OpLabel
          
; Normalize the interpolated surface normal
     %n_val = OpLoad %v3float %in_normal
    %n_norm = OpExtInst %v3float %1 Normalize %n_val

; Get light position from UBO and extract XYZ
    %lp_ptr = OpAccessChain %ptr_Uniform_v4float %ubo_var %int_3
   %lp_val4 = OpLoad %v4float %lp_ptr
   %lp_val3 = OpVectorShuffle %v3float %lp_val4 %lp_val4 0 1 2

; Calculate direction from surface to light
    %wp_val = OpLoad %v3float %in_worldPos
   %ld_diff = OpFSub %v3float %lp_val3 %wp_val
   %ld_norm = OpExtInst %v3float %1 Normalize %ld_diff

; Compute Lambertian term d = max(dot(N, L), 0.0)
   %dot_nl = OpDot %float %n_norm %ld_norm
        %d = OpExtInst %float %1 FMax %dot_nl %float_0

; Quantize d into bands: 0.75/0.5/0.25 -> 1.0/0.7/0.4/0.2
       %c1 = OpFOrdGreaterThan %bool %d %float_0_75
       %c2 = OpFOrdGreaterThan %bool %d %float_0_5
       %c3 = OpFOrdGreaterThan %bool %d %float_0_25
       
       %b3 = OpSelect %float %c3 %float_0_4 %float_0_2
       %b2 = OpSelect %float %c2 %float_0_7 %b3
     %band = OpSelect %float %c1 %float_1 %b2

; Final color calculation
   %v_color = OpLoad %v3float %in_color
   %res_rgb = OpVectorTimesScalar %v3float %v_color %band
   %res_v4 = OpCompositeConstruct %v4float %res_rgb %float_1

; Output
             OpStore %outColor %res_v4
             OpReturn
             OpFunctionEnd
