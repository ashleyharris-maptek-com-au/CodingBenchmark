The N-Dimensional Snake Game solver is implemented using an efficient A* pathfinding algorithm that generalizes across any number of dimensions. The solver maintains a local state of the snake, food, and obstacles, using hash-based collections (`unordered_map` and `unordered_set`) for fast spatial lookups in high-dimensional space.

### Key Features:
- **Arbitrary Dimensionality**: Handles dimensions from 2D up to 20D by representing coordinates as `std::vector<int>` and using custom hashing.
- **Pathfinding**: Employs A* search with Manhattan distance as a heuristic to navigate the hypercube efficiently.
- **Dynamic Growth**: Correctly tracks snake length and body segments, accounting for growth upon eating food.
- **Survival Logic**: If no path to food is found, the snake uses a survival heuristic to move towards areas with the most free neighbors, extending its lifespan.
- **Performance**: Optimized with fast I/O and localized search limits to ensure responsiveness even in extreme dimensionalities.

```cpp
#include <iostream>
#include <vector>
#include <deque>
#include <unordered_set>
#include <unordered_map>
#include <queue>
#include <algorithm>

using namespace std;

// Position representation
typedef vector<int> Pos;

// Hash function for Pos (vector<int>) to use in unordered maps/sets
struct PosHash {
    size_t operator()(const Pos& v) const {
        size_t seed = 0;
        for (int i : v) {
            seed ^= hash<int>{}(i) + 0x9e3779b9 + (seed << 6) + (seed >> 2);
        }
        return seed;
    }
};

// Global game state
int D;
long long max_turns;
Pos bounds;
unordered_set<Pos, PosHash> food_set;
unordered_set<Pos, PosHash> obstacles;
deque<Pos> snake;
unordered_map<Pos, int, PosHash> body_counts;

// Check if a position is within the hypercube bounds
bool is_in_bounds(const Pos& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    }
    return true;
}

// Check if a position is valid (no obstacle, no body)
bool is_valid(const Pos& p) {
    if (!is_in_bounds(p)) return false;
    if (obstacles.count(p)) return false;
    if (body_counts.count(p)) return false;
    return true;
}

// Manhattan distance heuristic for A*
long long get_h(const Pos& p, const Pos& target) {
    long long h = 0;
    for (int i = 0; i < D; ++i) {
        long long diff = (long long)p[i] - target[i];
        if (diff < 0) diff = -diff;
        h += diff;
    }
    return h;
}

// A* Node structure
struct Node {
    Pos p;
    long long g, h;
    int axis, dir;
    int parent_idx;

    bool operator>(const Node& other) const {
        if ((g + h) != (other.g + other.h))
            return (g + h) > (other.g + other.h);
        return h > other.h; // Tie-breaker: prefer nodes closer to target
    }
};

// Pathfinding using A*
vector<pair<int, int>> find_path(const Pos& start, const Pos& target) {
    priority_queue<Node, vector<Node>, greater<Node>> pq;
    unordered_map<Pos, long long, PosHash> visited;

    pq.push({start, 0, get_h(start, target), -1, 0, -1});
    visited[start] = 0;

    vector<Node> all_nodes;
    
    while (!pq.empty()) {
        Node current = pq.top();
        pq.pop();

        if (current.p == target) {
            vector<pair<int, int>> path;
            Node temp = current;
            while (temp.axis != -1) {
                path.push_back({temp.axis, temp.dir});
                temp = all_nodes[temp.parent_idx];
            }
            reverse(path.begin(), path.end());
            return path;
        }

        if (all_nodes.size() >= 100000) break; // Memory/Time safety limit

        int current_idx = (int)all_nodes.size();
        all_nodes.push_back(current);

        for (int a = 0; a < D; ++a) {
            for (int d : {-1, 1}) {
                Pos next_p = current.p;
                next_p[a] += d;
                if (is_in_bounds(next_p) && !obstacles.count(next_p)) {
                    // Static body pathfinding: avoid current body positions
                    if (!body_counts.count(next_p) || next_p == target) {
                        long long next_g = current.g + 1;
                        if (visited.find(next_p) == visited.end() || next_g < visited[next_p]) {
                            visited[next_p] = next_g;
                            pq.push({next_p, next_g, get_h(next_p, target), a, d, current_idx});
                        }
                    }
                }
            }
        }
    }
    return {};
}

// Update local snake state after a move
void apply_move(int axis, int dir) {
    Pos new_head = snake.front();
    new_head[axis] += dir;
    
    snake.push_front(new_head);
    body_counts[new_head]++;
    
    if (food_set.count(new_head)) {
        food_set.erase(new_head);
        // Grows: don't pop tail
    } else {
        Pos tail = snake.back();
        snake.pop_back();
        body_counts[tail]--;
        if (body_counts[tail] == 0) body_counts.erase(tail);
    }
}

// Survival strategy when no path to food is found
bool can_survive(pair<int, int>& move) {
    Pos head = snake.front();
    int best_free = -1;
    bool found = false;
    for (int a = 0; a < D; ++a) {
        for (int d : {-1, 1}) {
            Pos next_p = head;
            next_p[a] += d;
            if (is_valid(next_p)) {
                int free_neighbors = 0;
                for (int na = 0; na < D; ++na) {
                    for (int nd : {-1, 1}) {
                        Pos nnp = next_p;
                        nnp[na] += nd;
                        if (is_valid(nnp)) free_neighbors++;
                    }
                }
                if (free_neighbors > best_free) {
                    best_free = free_neighbors;
                    move = {a, d};
                    found = true;
                }
            }
        }
    }
    return found;
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    if (!(cin >> D >> max_turns)) return 0;
    bounds.resize(D);
    for (int i = 0; i < D; ++i) cin >> bounds[i];
    
    int start_len;
    cin >> start_len;
    Pos start_pos(D);
    for (int i = 0; i < D; ++i) cin >> start_pos[i];
    
    // Initial snake state: start_len segments stacked at start_pos
    for (int i = 0; i < start_len; ++i) {
        snake.push_back(start_pos);
        body_counts[start_pos]++;
    }
    
    int F;
    if (!(cin >> F)) F = 0;
    for (int i = 0; i < F; ++i) {
        Pos f(D);
        for (int j = 0; j < D; ++j) cin >> f[j];
        food_set.insert(f);
    }
    
    int O;
    if (!(cin >> O)) O = 0;
    for (int i = 0; i < O; ++i) {
        Pos o(D);
        for (int j = 0; j < D; ++j) cin >> o[j];
        obstacles.insert(o);
    }
    
    long long turn = 0;
    while (!food_set.empty() && turn < max_turns) {
        Pos head = snake.front();
        
        // Find candidate food items sorted by proximity
        vector<pair<long long, Pos>> sorted_food;
        for (const auto& f : food_set) {
            sorted_food.push_back({get_h(head, f), f});
        }
        sort(sorted_food.begin(), sorted_food.end(), [](const auto& a, const auto& b) {
            return a.first < b.first;
        });

        bool found_any_path = false;
        // Check top candidate foods for reachability
        int limit = min((int)sorted_food.size(), 8);
        for (int i = 0; i < limit; ++i) {
            vector<pair<int, int>> path = find_path(head, sorted_food[i].second);
            if (!path.empty()) {
                for (auto m : path) {
                    if (turn >= max_turns) break;
                    cout << m.first << " " << m.second << "\n";
                    apply_move(m.first, m.second);
                    turn++;
                    if (food_set.empty()) break;
                    // Re-evaluate if we ate the current target food
                    if (food_set.count(sorted_food[i].second) == 0) break;
                }
                found_any_path = true;
                break;
            }
        }

        if (!found_any_path) {
            pair<int, int> move;
            if (can_survive(move)) {
                cout << move.first << " " << move.second << "\n";
                apply_move(move.first, move.second);
                turn++;
            } else {
                // No valid moves, snake is trapped
                break;
            }
        }
    }
    
    return 0;
}
```
