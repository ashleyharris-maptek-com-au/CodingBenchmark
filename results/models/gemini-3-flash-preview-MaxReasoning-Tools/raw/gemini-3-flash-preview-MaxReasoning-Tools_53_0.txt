import math

class Autopilot:
    def __init__(self):
        self.v_int = [0.0, 0.0, 0.0]
        self.last_pos = [0.0, 0.0, 0.0]
        self.last_vel = [0.0, 0.0, 0.0]
        self.initialized = False

    def fuse_pos(self, state, dt):
        sources = []
        # Relative position sources (LVLH: x=radial, y=along, z=cross)
        if state.get('radar_rel_x_valid') and state.get('radar_rel_y_valid') and state.get('radar_rel_z_valid'):
            sources.append([state['radar_rel_pos_x_m'], state['radar_rel_pos_y_m'], state['radar_rel_pos_z_m']])
        if state.get('lidar_rel_x_valid') and state.get('lidar_rel_y_valid') and state.get('lidar_rel_z_valid'):
            sources.append([state['lidar_rel_pos_x_m'], state['lidar_rel_pos_y_m'], state['lidar_rel_pos_z_m']])
        if state.get('gps_1_x_valid') and state.get('gps_1_y_valid') and state.get('gps_1_z_valid'):
            sources.append([state['gps_1_pos_x_m'], state['gps_1_pos_y_m'], state['gps_1_pos_z_m']])
        if state.get('gps_2_x_valid') and state.get('gps_2_y_valid') and state.get('gps_2_z_valid'):
            sources.append([state['gps_2_pos_x_m'], state['gps_2_pos_y_m'], state['gps_2_pos_z_m']])
        
        if state.get('dock_camera_active') and state.get('dock_camera_bearing_z_valid'):
            r = state['dock_camera_range_m']
            bx, by, bz = state['dock_camera_bearing_x'], state['dock_camera_bearing_y'], state['dock_camera_bearing_z']
            mag = math.sqrt(bx*bx + by*by + bz*bz)
            if mag > 1e-6:
                sources.append([bx*r/mag, by*r/mag, bz*r/mag])

        # Range finders (scalar distance)
        ranges = []
        for i in range(1, 4):
            if state.get(f'range_{i}_valid'):
                ranges.append(state[f'range_{i}_m'])
        
        if not sources:
            if self.initialized:
                # Dead reckoning
                est_pos = [self.last_pos[i] + self.last_vel[i] * dt for i in range(3)]
                return est_pos
            return [0.0, 0.0, 0.0]
        
        pos = [0.0, 0.0, 0.0]
        for i in range(3):
            vals = [s[i] for s in sources]
            vals.sort()
            pos[i] = vals[len(vals)//2]
            
        # Optional: consistency check with range finders
        if ranges:
            r_avg = sum(ranges) / len(ranges)
            r_est = math.sqrt(sum(p*p for p in pos))
            if r_est > 1e-3:
                scale = r_avg / r_est
                # If they differ significantly, maybe trust ranges more for magnitude?
                # But bearing from sensors is better. We'll trust 3D sensors mostly.
                pass
                
        return pos

    def fuse_vel(self, state, pos, dt):
        sources = []
        if state.get('radar_relv_x_valid') and state.get('radar_relv_y_valid') and state.get('radar_relv_z_valid'):
            sources.append([state['radar_rel_vel_x_ms'], state['radar_rel_vel_y_ms'], state['radar_rel_vel_z_ms']])
        if state.get('lidar_relv_x_valid') and state.get('lidar_relv_y_valid') and state.get('lidar_relv_z_valid'):
            sources.append([state['lidar_rel_vel_x_ms'], state['lidar_rel_vel_y_ms'], state['lidar_rel_vel_z_ms']])
        
        if not sources:
            if self.initialized and dt > 0:
                return [(pos[i] - self.last_pos[i]) / dt for i in range(3)]
            return self.last_vel
        
        vel = [0.0, 0.0, 0.0]
        for i in range(3):
            vals = [s[i] for s in sources]
            vals.sort()
            vel[i] = vals[len(vals)//2]
        return vel

    def step(self, state, dt):
        if dt <= 0: dt = 0.1
        
        pos = self.fuse_pos(state, dt)
        vel = self.fuse_vel(state, pos, dt)
        
        if not self.initialized:
            self.last_pos = pos
            self.last_vel = vel
            self.initialized = True

        dist_total = math.sqrt(sum(p*p for p in pos))
        dist_y = pos[1]
        
        target_vel = [0.0, 0.0, 0.0]
        thrust = [0.0, 0.0, 0.0]
        
        # Strategy
        if dist_total > 50:
            # Orbital Mechanics Regime
            # Cross-track (z): independent PD
            thrust[2] = -0.1 * pos[2] - 0.6 * vel[2]
            
            # Along-track (y) and Radial (x)
            if dist_y < -50: # Behind
                # Catch up: thrust retrograde to lower orbit
                if vel[1] < 2.0:
                    thrust[1] = -0.4 
                elif vel[1] > 3.0:
                    thrust[1] = 0.2
                # Keep radial near 0 or slightly below
                thrust[0] = -0.05 * (pos[0] + 50) - 0.4 * vel[0]
            elif dist_y > 50: # Ahead
                # Slow down: thrust prograde to raise orbit
                if vel[1] > -2.0:
                    thrust[1] = 0.4
                elif vel[1] < -3.0:
                    thrust[1] = -0.2
                thrust[0] = -0.05 * (pos[0] - 50) - 0.4 * vel[0]
            else:
                # Transition to linear
                target_vel[0] = -0.1 * pos[0]
                target_vel[1] = -0.1 * pos[1]
                for i in range(2):
                    ev = target_vel[i] - vel[i]
                    self.v_int[i] += ev * dt
                    thrust[i] = 1.0 * ev + 0.02 * self.v_int[i]
        else:
            # Linear Regime (<50m)
            speed = max(0.05, min(0.8, 0.08 * dist_total))
            if dist_total > 0.01:
                target_vel = [-(p/dist_total)*speed for p in pos]
            
            for i in range(3):
                ev = target_vel[i] - vel[i]
                self.v_int[i] += ev * dt
                self.v_int[i] = max(-20.0, min(20.0, self.v_int[i]))
                thrust[i] = 1.5 * ev + 0.03 * self.v_int[i]

        # Attitude Control
        gyro_rates = [0.0, 0.0, 0.0]
        for i, axis in enumerate(['pitch', 'roll', 'yaw']):
            v = [state[f'gyro_{j}_{axis}_dps'] for j in range(1, 4) if state.get(f'gyro_{j}_{axis}_valid')]
            if v: 
                v.sort()
                gyro_rates[i] = v[len(v)//2]
                
        star_angles = [None, None, None]
        for i, axis in enumerate(['pitch', 'roll', 'yaw']):
            v = [state[f'star_tracker_{j}_{axis}_deg'] for j in range(1, 3) if state.get(f'star_tracker_{j}_{axis}_valid')]
            if v:
                v.sort()
                star_angles[i] = v[len(v)//2]
        
        att_rates = [0.0, 0.0, 0.0]
        for i in range(3):
            rate_dps = gyro_rates[i]
            if star_angles[i] is not None:
                # PD on attitude angle
                target_rate_dps = -0.2 * star_angles[i]
                target_rate_dps = max(-1.8, min(1.8, target_rate_dps))
                att_rates[i] = (target_rate_dps - rate_dps) * (math.pi/180.0)
            else:
                # Rate damping only
                att_rates[i] = -rate_dps * (math.pi/180.0) * 0.5

        self.last_pos = pos
        self.last_vel = vel
        
        def cl(v, l): return max(-l, min(l, v))
        return {
            'thrust_x': cl(thrust[0], 1.0),
            'thrust_y': cl(thrust[1], 1.0),
            'thrust_z': cl(thrust[2], 1.0),
            'att_rate_roll': cl(att_rates[1], 0.035),
            'att_rate_pitch': cl(att_rates[0], 0.035),
            'att_rate_yaw': cl(att_rates[2], 0.035)
        }

_auto = Autopilot()

def autopilot_step(state: dict, dt: float) -> dict:
    """
    Main autopilot function called every dt seconds.
    Handles sensor fusion, orbital mechanics, and precision docking.
    """
    return _auto.step(state, dt)
