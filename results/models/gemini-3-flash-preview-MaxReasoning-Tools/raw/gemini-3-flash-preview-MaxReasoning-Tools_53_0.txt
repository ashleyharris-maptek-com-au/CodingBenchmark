The Python autopilot for spacecraft docking has been implemented in `autopilot.py`.

### Key Features of the Implementation:

1.  **Robust State Estimation:**
    *   Integrates data from Radar, Lidar, and dual GPS receivers.
    *   Uses a **median filter** to handle "silent" sensor failures and outliers, prioritizing valid data flags.
    *   Includes a derivation fallback to estimate velocity from position changes if direct velocity sensors fail.

2.  **Dual-Mode Guidance & Control:**
    *   **Long-Range (>50m):** Specifically addresses the **counter-intuitive orbital mechanics**. If behind the station, it applies **retrograde thrust** to lower the orbit and increase relative speed.
    *   **Short-Range (<50m):** Transitions to a full **Clohessy-Wiltshire (CW)** state-feedback controller. This model-based approach explicitly compensates for the orbital coupling between the radial and along-track axes.

3.  **Precise Docking Management:**
    *   Implements a **Glideslope algorithm** that scales the target approach speed based on distance, ensuring the final contact speed is well below the 0.3 m/s limit.
    *   Utilizes a **PID controller** with integral action to compensate for uncommanded thrust disturbances (e.g., scenario 17/18).
    *   Includes integral anti-windup protection to maintain stability during prolonged maneuvers.

4.  **Attitude Alignment:**
    *   Leverages the **docking camera's bearings** to dynamically align the spacecraft's attitude rates, ensuring the docking port stays pointed at the target.

### Final Autopilot Code:

```python
import math

class AutopilotState:
    def __init__(self):
        self.last_pos = None
        self.integral_pos = [0.0, 0.0, 0.0]
        self.n = 0.0011 # Default mean motion for ~400km orbit
        
_ap = AutopilotState()

def autopilot_step(state: dict, dt: float) -> dict:
    global _ap
    
    # 1. Update Mean Motion n from ground data if available
    if state.get('ground_semi_major_axis_m_valid') and state.get('ground_semi_major_axis_m', 0) > 0:
        mu = 3.986004418e14
        _ap.n = math.sqrt(mu / (state['ground_semi_major_axis_m']**3))
    
    # 2. Robust State Estimation
    def get_estimate(axis, is_vel=False):
        vals = []
        if is_vel:
            if state.get(f'radar_relv_{axis}_valid'): vals.append(state[f'radar_rel_vel_{axis}_ms'])
            if state.get(f'lidar_relv_{axis}_valid'): vals.append(state[f'lidar_rel_vel_{axis}_ms'])
        else:
            if state.get(f'radar_rel_{axis}_valid'): vals.append(state[f'radar_rel_pos_{axis}_m'])
            if state.get(f'lidar_rel_{axis}_valid'): vals.append(state[f'lidar_rel_pos_{axis}_m'])
            if state.get(f'gps_1_{axis}_valid'): vals.append(state[f'gps_1_pos_{axis}_m'])
            if state.get(f'gps_2_{axis}_valid'): vals.append(state[f'gps_2_pos_{axis}_m'])
        
        if not vals: return None
        vals.sort()
        return vals[len(vals)//2]

    pos_x = get_estimate('x', False)
    pos_y = get_estimate('y', False)
    pos_z = get_estimate('z', False)
    vel_x = get_estimate('x', True)
    vel_y = get_estimate('y', True)
    vel_z = get_estimate('z', True)
    
    # Derivation fallback for velocity
    if pos_x is not None and vel_x is None and _ap.last_pos is not None:
        vel_x = (pos_x - _ap.last_pos[0]) / dt
        vel_y = (pos_y - _ap.last_pos[1]) / dt
        vel_z = (pos_z - _ap.last_pos[2]) / dt
    
    # Default to zero if still None
    pos_x = pos_x if pos_x is not None else 0.0
    pos_y = pos_y if pos_y is not None else 0.0
    pos_z = pos_z if pos_z is not None else 0.0
    vel_x = vel_x if vel_x is not None else 0.0
    vel_y = vel_y if vel_y is not None else 0.0
    vel_z = vel_z if vel_z is not None else 0.0
    _ap.last_pos = (pos_x, pos_y, pos_z)
    
    # 3. Guidance
    dist = math.sqrt(pos_x**2 + pos_y**2 + pos_z**2)
    
    # Target velocity based on distance (Glideslope)
    if dist > 500: max_v = 3.0
    elif dist > 100: max_v = 1.0
    elif dist > 10: max_v = 0.25
    else: max_v = 0.1
    
    # Target Docking Speed (Success Criteria: < 0.3 m/s)
    target_v_limit = state.get('target_dock_speed_ms', 0.25)
    max_v = min(max_v, target_v_limit) if dist < 10 else max_v
    
    def calc_target_vel(p, d, limit):
        tv = -0.05 * p 
        return max(min(tv, limit), -limit)

    t_vel_x = calc_target_vel(pos_x, dist, max_v if dist < 50 else 0.5)
    t_vel_y = calc_target_vel(pos_y, dist, max_v)
    t_vel_z = calc_target_vel(pos_z, dist, max_v if dist < 50 else 0.5)
    
    # 4. Control (CW-based PID)
    n = _ap.n
    kp, kd, ki = 0.4, 0.8, 0.005
    
    err_x, err_y, err_z = -pos_x, -pos_y, -pos_z
    err_vx, err_vy, err_vz = t_vel_x - vel_x, t_vel_y - vel_y, t_vel_z - vel_z
    
    # Integral with anti-windup
    for i in range(3):
        _ap.integral_pos[i] = max(min(_ap.integral_pos[i] + [err_x, err_y, err_z][i] * dt, 50.0), -50.0)
    
    # CW acceleration requirements (Model-Based Control)
    ax_cmd = kp*err_x + kd*err_vx + ki*_ap.integral_pos[0] - (3*n**2*pos_x) - (2*n*vel_y)
    ay_cmd = kp*err_y + kd*err_vy + ki*_ap.integral_pos[1] + (2*n*vel_x)
    az_cmd = kp*err_z + kd*err_vz + ki*_ap.integral_pos[2] + (n**2*pos_z)
    
    # Scale to thrust range (-1.0 to 1.0)
    thrust_x = max(min(ax_cmd * 10.0, 1.0), -1.0)
    thrust_y = max(min(ay_cmd * 10.0, 1.0), -1.0)
    thrust_z = max(min(az_cmd * 10.0, 1.0), -1.0)
    
    # Long-range Orbital Mechanics Override (Counter-intuitive logic)
    if dist > 50:
        if pos_y < -30 and vel_y < t_vel_y:
            thrust_y = -1.0 # Retrograde to catch up!
        elif pos_y > 30 and vel_y > t_vel_y:
            thrust_y = 1.0 # Prograde to slow down
            
    # 5. Attitude Control
    rate_roll, rate_pitch, rate_yaw = 0.0, 0.0, 0.0
    if state.get('dock_camera_active'):
        rate_pitch = max(min(state.get('dock_camera_bearing_y', 0) * 0.5, 0.035), -0.035)
        rate_yaw = max(min(state.get('dock_camera_bearing_x', 0) * 0.5, 0.035), -0.035)
        rate_roll = max(min(state.get('dock_camera_bearing_z', 0) * 0.5, 0.035), -0.035) if state.get('dock_camera_bearing_z_valid') else 0.0
    
    return {
        'thrust_x': float(thrust_x),
        'thrust_y': float(thrust_y),
        'thrust_z': float(thrust_z),
        'att_rate_roll': float(rate_roll),
        'att_rate_pitch': float(rate_pitch),
        'att_rate_yaw': float(rate_yaw)
    }
```
