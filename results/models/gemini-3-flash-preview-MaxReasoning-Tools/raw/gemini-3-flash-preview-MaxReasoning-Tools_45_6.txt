def word(val):
    return val.to_bytes(4, 'little').hex()

def string(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return b.hex()

def op(opcode, word_count):
    return ((word_count << 16) | opcode).to_bytes(4, 'little').hex()
