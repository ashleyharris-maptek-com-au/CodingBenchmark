using System;
using System.Collections.Generic;
using System.Linq;

namespace TetrisAI
{
    class Program
    {
        static void Main(string[] args)
        {
            Scanner sc = new Scanner();
            string wStr = sc.Next();
            if (wStr == null) return;
            int width = int.Parse(wStr);
            int height = sc.NextInt();
            int numPieces = sc.NextInt();

            AIPlayer player = new AIPlayer(width, height, numPieces);
            player.Play(sc);
        }
    }

    class Scanner
    {
        private string[] _tokens;
        private int _next;

        public string Next()
        {
            while (_tokens == null || _next >= _tokens.Length)
            {
                string line = Console.ReadLine();
                if (line == null) return null;
                _tokens = line.Split(new[] { ' ', '\t', '\n', '\r' }, StringSplitOptions.RemoveEmptyEntries);
                _next = 0;
            }
            return _tokens[_next++];
        }

        public int NextInt()
        {
            string s = Next();
            return s == null ? 0 : int.Parse(s);
        }
    }

    class AIPlayer
    {
        public int Width, Height, NumPieces;
        public Board CurrentBoard;
        public Dictionary<char, Piece> Pieces;

        // Weights for heuristic (Dellacherie algorithm base)
        private double _wLandingHeight = -1.0;
        private double _wLinesCleared = 1.0;
        private double _wRowTrans = -1.0;
        private double _wColTrans = -1.0;
        private double _wHoles = -4.0;
        private double _wWells = -1.0;

        public AIPlayer(int w, int h, int n)
        {
            Width = w;
            Height = h;
            NumPieces = n;
            CurrentBoard = new Board(w, h);
            InitializePieces();
            AdjustStrategy();
        }

        private void InitializePieces()
        {
            Pieces = new Dictionary<char, Piece>
            {
                { 'I', new Piece(new int[][] { new int[] { 0, 0 }, new int[] { 0, 1 }, new int[] { 0, 2 }, new int[] { 0, 3 } }, 4) },
                { 'O', new Piece(new int[][] { new int[] { 0, 0 }, new int[] { 0, 1 }, new int[] { 1, 0 }, new int[] { 1, 1 } }, 2) },
                { 'T', new Piece(new int[][] { new int[] { 0, 0 }, new int[] { 0, 1 }, new int[] { 0, 2 }, new int[] { 1, 1 } }, 3) },
                { 'S', new Piece(new int[][] { new int[] { 0, 0 }, new int[] { 0, 1 }, new int[] { 1, 1 }, new int[] { 1, 2 } }, 3) },
                { 'Z', new Piece(new int[][] { new int[] { 0, 1 }, new int[] { 0, 2 }, new int[] { 1, 0 }, new int[] { 1, 1 } }, 3) },
                { 'J', new Piece(new int[][] { new int[] { 0, 0 }, new int[] { 0, 1 }, new int[] { 0, 2 }, new int[] { 1, 0 } }, 3) },
                { 'L', new Piece(new int[][] { new int[] { 0, 0 }, new int[] { 0, 1 }, new int[] { 0, 2 }, new int[] { 1, 2 } }, 3) }
            };
        }

        private void AdjustStrategy()
        {
            // Adaptive logic based on game complexity and piece count
            double complexity = (double)Width * Height;
            if (NumPieces > 500 || complexity > 200)
            {
                // Marathon/Large board: survival is key
                _wHoles = -10.0;
                _wWells = -2.0;
                _wLandingHeight = -2.0;
            }
            if (Width < 8)
            {
                // Narrow boards: holes are catastrophic
                _wHoles = -15.0;
            }
        }

        public void Play(Scanner sc)
        {
            for (int i = 0; i < NumPieces; i++)
            {
                string pStr = sc.Next();
                if (pStr == null) break;
                char type = pStr[0];

                Move best = FindBestMove(type);
                if (best == null)
                {
                    // Desperation move
                    Console.WriteLine("0 0");
                    break;
                }
                else
                {
                    Console.WriteLine($"{best.Rotation} {best.Column}");
                    
                    string result = sc.Next();
                    if (result == null || result == "gameover") break;

                    CurrentBoard.ApplyMove(Pieces[type].Rotations[best.Rotation], best.RPos, best.Column);
                }
            }
        }

        private Move FindBestMove(char type)
        {
            Piece p = Pieces[type];
            Move bestMove = null;
            double bestScore = double.NegativeInfinity;

            for (int rot = 0; rot < 4; rot++)
            {
                var shape = p.Rotations[rot];
                int pWidth = p.Widths[rot];
                int pHeight = p.Heights[rot];

                for (int col = 0; col <= Width - pWidth; col++)
                {
                    int rPos = CurrentBoard.GetRPos(shape, col);
                    if (rPos + pHeight > Height) continue;

                    Board sim = CurrentBoard.Clone();
                    int linesCleared = sim.ApplyMove(shape, rPos, col);
                    
                    double score = Evaluate(sim, rPos, linesCleared);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestMove = new Move { Rotation = rot, Column = col, RPos = rPos, Score = score };
                    }
                }
            }

            return bestMove;
        }

        private double Evaluate(Board board, int landingHeight, int linesCleared)
        {
            int holes = board.CountHoles();
            int rowTrans = board.CalculateRowTransitions();
            int colTrans = board.CalculateColTransitions();
            int wells = board.CalculateWells();

            double score = _wLandingHeight * landingHeight
                         + _wLinesCleared * linesCleared
                         + _wRowTrans * rowTrans
                         + _wColTrans * colTrans
                         + _wHoles * holes
                         + _wWells * wells;

            // Panic mode: if stack is too high, prioritize clearing lines even more
            int maxHeight = board.GetMaxHeight();
            if (maxHeight > Height * 0.8)
            {
                score += linesCleared * 5.0;
            }

            return score;
        }
    }

    class Board
    {
        public int Width, Height;
        public long[] Rows;

        public Board(int w, int h)
        {
            Width = w;
            Height = h;
            Rows = new long[h];
        }

        public Board Clone()
        {
            Board b = new Board(Width, Height);
            Array.Copy(Rows, b.Rows, Height);
            return b;
        }

        public int GetRPos(List<int[]> shape, int col)
        {
            int rPos = 0;
            foreach (var p in shape)
            {
                int c = col + p[1];
                int rOffset = p[0];
                int h = GetColumnHeight(c);
                rPos = Math.Max(rPos, h - rOffset);
            }
            return rPos;
        }

        private int GetColumnHeight(int c)
        {
            for (int r = Height - 1; r >= 0; r--)
            {
                if ((Rows[r] & (1L << c)) != 0) return r + 1;
            }
            return 0;
        }

        public int GetMaxHeight()
        {
            for (int r = Height - 1; r >= 0; r--)
            {
                if (Rows[r] != 0) return r + 1;
            }
            return 0;
        }

        public int ApplyMove(List<int[]> shape, int rPos, int col)
        {
            foreach (var p in shape)
            {
                int r = rPos + p[0];
                int c = col + p[1];
                if (r < Height) Rows[r] |= (1L << c);
            }

            int cleared = 0;
            long fullRow = (1L << Width) - 1;
            int target = 0;
            for (int source = 0; source < Height; source++)
            {
                if (Rows[source] == fullRow)
                {
                    cleared++;
                }
                else
                {
                    Rows[target++] = Rows[source];
                }
            }
            while (target < Height) Rows[target++] = 0;
            return cleared;
        }

        public int CountHoles()
        {
            int holes = 0;
            for (int c = 0; c < Width; c++)
            {
                bool blockFound = false;
                for (int r = Height - 1; r >= 0; r--)
                {
                    if ((Rows[r] & (1L << c)) != 0) blockFound = true;
                    else if (blockFound) holes++;
                }
            }
            return holes;
        }

        public int CalculateRowTransitions()
        {
            int trans = 0;
            for (int r = 0; r < Height; r++)
            {
                bool prev = true; // Left wall
                for (int c = 0; c < Width; c++)
                {
                    bool curr = (Rows[r] & (1L << c)) != 0;
                    if (curr != prev) trans++;
                    prev = curr;
                }
                if (!prev) trans++; // Right wall
            }
            return trans;
        }

        public int CalculateColTransitions()
        {
            int trans = 0;
            for (int c = 0; c < Width; c++)
            {
                bool prev = true; // Floor
                for (int r = 0; r < Height; r++)
                {
                    bool curr = (Rows[r] & (1L << c)) != 0;
                    if (curr != prev) trans++;
                    prev = curr;
                }
            }
            return trans;
        }

        public int CalculateWells()
        {
            int sum = 0;
            for (int c = 0; c < Width; c++)
            {
                int depth = 0;
                for (int r = Height - 1; r >= 0; r--)
                {
                    bool curr = (Rows[r] & (1L << c)) != 0;
                    if (curr)
                    {
                        depth = 0;
                    }
                    else
                    {
                        bool left = (c == 0) || (Rows[r] & (1L << (c - 1))) != 0;
                        bool right = (c == Width - 1) || (Rows[r] & (1L << (c + 1))) != 0;
                        if (left && right)
                        {
                            depth++;
                            sum += depth;
                        }
                        else
                        {
                            depth = 0;
                        }
                    }
                }
            }
            return sum;
        }
    }

    class Piece
    {
        public List<int[]>[] Rotations = new List<int[]>[4];
        public int[] Widths = new int[4];
        public int[] Heights = new int[4];

        public Piece(int[][] baseShape, int size)
        {
            int[][] current = baseShape;
            for (int i = 0; i < 4; i++)
            {
                // Normalize
                int minR = current.Min(p => p[0]);
                int minC = current.Min(p => p[1]);
                var norm = current.Select(p => new int[] { p[0] - minR, p[1] - minC }).ToList();
                Rotations[i] = norm;
                Widths[i] = norm.Max(p => p[1]) + 1;
                Heights[i] = norm.Max(p => p[0]) + 1;
                current = RotateCW(current, size);
            }
        }

        private int[][] RotateCW(int[][] shape, int size)
        {
            int[][] rotated = new int[shape.Length][];
            for (int i = 0; i < shape.Length; i++)
            {
                rotated[i] = new int[] { shape[i][1], size - 1 - shape[i][0] };
            }
            return rotated;
        }
    }

    class Move
    {
        public int Rotation;
        public int Column;
        public int RPos;
        public double Score;
    }
}
