use std::io::{self, Read};

fn solve() {
    let mut input = String::new();
    if io::stdin().read_to_string(&mut input).is_err() {
        return;
    }
    let mut words = input.split_whitespace();

    // Read n and m
    let n: usize = match words.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = match words.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => 0,
    };

    if n == 0 {
        println!("0\n");
        return;
    }

    // Read edges and build adjacency list
    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        let u: usize = match words.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        let v: usize = match words.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        if u < n && v < n && u != v {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    // Pre-calculate edges to already processed vertices for faster cut calculation
    let mut adj_back = vec![vec![]; n];
    for u in 0..n {
        for &v in &adj[u] {
            if v < u {
                adj_back[u].push(v);
            }
        }
    }

    let mut best_cut = 0;
    let mut best_partition = vec![false; n];
    
    // Initial guess: first n/2 vertices in partition A
    for i in 0..n/2 {
        best_partition[i] = true;
    }
    for u in 0..n {
        for &v in &adj_back[u] {
            if best_partition[u] != best_partition[v] {
                best_cut += 1;
            }
        }
    }

    let mut current_partition = vec![false; n];
    // Symmetry breaking: fix vertex 0 in partition A
    current_partition[0] = true;

    // Branch and Bound recursive search
    fn backtrack(
        u: usize,
        count_a: usize,
        current_cut: usize,
        n: usize,
        adj_back: &Vec<Vec<usize>>,
        current_partition: &mut Vec<bool>,
        best_cut: &mut usize,
        best_partition: &mut Vec<bool>,
    ) {
        // Pruning: if current cut already exceeds best found, stop this branch
        if current_cut >= *best_cut {
            return;
        }

        // Base case: all vertices assigned
        if u == n {
            if count_a == n / 2 {
                *best_cut = current_cut;
                *best_partition = current_partition.clone();
            }
            return;
        }

        let remaining = n - u;
        let count_b = u - count_a;
        let half = n / 2;

        // Calculate contribution to the cut if u is in A vs B
        let mut d_a = 0;
        for &v in adj_back[u].iter() {
            if current_partition[v] {
                d_a += 1;
            }
        }
        let d_b = adj_back[u].len() - d_a;

        // Try assigning vertex u to partition A
        if count_a < half && remaining >= (half - count_a) {
            // Check if we can still fill partition B
            if remaining - 1 >= (half - count_b) {
                current_partition[u] = true;
                backtrack(u + 1, count_a + 1, current_cut + d_b, n, adj_back, current_partition, best_cut, best_partition);
            }
        }

        // Try assigning vertex u to partition B
        if count_b < half && remaining >= (half - count_b) {
            // Check if we can still fill partition A
            if remaining - 1 >= (half - count_a) {
                current_partition[u] = false;
                backtrack(u + 1, count_a, current_cut + d_a, n, adj_back, current_partition, best_cut, best_partition);
            }
        }
    }

    // Start backtracking from vertex 1 (vertex 0 is fixed in partition A)
    backtrack(1, 1, 0, n, &adj_back, &mut current_partition, &mut best_cut, &mut best_partition);

    // Output result
    println!("{}", best_cut);
    let mut first = true;
    for i in 0..n {
        if best_partition[i] {
            if !first {
                print!(" ");
            }
            print!("{}", i);
            first = false;
        }
    }
    println!();
}

fn main() {
    solve();
}
