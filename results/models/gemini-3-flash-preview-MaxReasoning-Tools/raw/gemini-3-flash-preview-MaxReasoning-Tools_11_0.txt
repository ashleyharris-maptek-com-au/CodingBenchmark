use std::io::{self, BufRead};
use std::time::Instant;

#[derive(Clone, Copy, Debug)]
struct Point {
    x: f64,
    y: f64,
}

struct Polygon {
    vertices: Vec<Point>,
    min_x: f64,
    max_x: f64,
    min_y: f64,
    max_y: f64,
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let mut min_x = f64::INFINITY;
        let mut max_x = f64::NEG_INFINITY;
        let mut min_y = f64::INFINITY;
        let mut max_y = f64::NEG_INFINITY;
        for p in &vertices {
            if p.x < min_x { min_x = p.x; }
            if p.x > max_x { max_x = p.x; }
            if p.y < min_y { min_y = p.y; }
            if p.y > max_y { max_y = p.y; }
        }
        Polygon { vertices, min_x, max_x, min_y, max_y }
    }

    fn area(&self) -> f64 {
        let mut area = 0.0;
        for i in 0..self.vertices.len() {
            let p1 = &self.vertices[i];
            let p2 = &self.vertices[(i + 1) % self.vertices.len()];
            area += p1.x * p2.y - p2.x * p1.y;
        }
        area.abs() / 2.0
    }

    fn rotate(&self, angle: f64) -> Polygon {
        if angle == 0.0 { return Polygon::new(self.vertices.clone()); }
        let cos_a = angle.cos();
        let sin_a = angle.sin();
        let rotated_vertices = self.vertices.iter().map(|p| {
            Point {
                x: p.x * cos_a - p.y * sin_a,
                y: p.x * sin_a + p.y * cos_a,
            }
        }).collect();
        Polygon::new(rotated_vertices)
    }

    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated_vertices = self.vertices.iter().map(|p| {
            Point {
                x: p.x + dx,
                y: p.y + dy,
            }
        }).collect();
        Polygon::new(translated_vertices)
    }

    fn contains_point(&self, p: Point) -> bool {
        if p.x < self.min_x - 1e-7 || p.x > self.max_x + 1e-7 || p.y < self.min_y - 1e-7 || p.y > self.max_y + 1e-7 {
            return false;
        }
        let mut inside = false;
        let mut j = self.vertices.len() - 1;
        for i in 0..self.vertices.len() {
            if ((self.vertices[i].y > p.y) != (self.vertices[j].y > p.y)) &&
               (p.x < (self.vertices[j].x - self.vertices[i].x) * (p.y - self.vertices[i].y) / (self.vertices[j].y - self.vertices[i].y) + self.vertices[i].x) {
                inside = !inside;
            }
            j = i;
        }
        inside
    }
}

fn dist_sq(p1: Point, p2: Point) -> f64 {
    (p1.x - p2.x).powi(2) + (p1.y - p2.y).powi(2)
}

fn dist_point_to_segment_sq(p: Point, a: Point, b: Point) -> f64 {
    let l2 = dist_sq(a, b);
    if l2 == 0.0 { return dist_sq(p, a); }
    let t = ((p.x - a.x) * (b.x - a.x) + (p.y - a.y) * (b.y - a.y)) / l2;
    let t = t.max(0.0).min(1.0);
    dist_sq(p, Point { x: a.x + t * (b.x - a.x), y: a.y + t * (b.y - a.y) })
}

fn segments_intersect(p1: Point, p2: Point, p3: Point, p4: Point) -> bool {
    fn ccw(a: Point, b: Point, c: Point) -> f64 {
        (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
    }
    let eps = 1e-8;
    let ccw1 = ccw(p1, p2, p3);
    let ccw2 = ccw(p1, p2, p4);
    let ccw3 = ccw(p3, p4, p1);
    let ccw4 = ccw(p3, p4, p2);

    if ((ccw1 > eps && ccw2 < -eps) || (ccw1 < -eps && ccw2 > eps)) &&
       ((ccw3 > eps && ccw4 < -eps) || (ccw3 < -eps && ccw4 > eps)) {
        return true;
    }
    false
}

fn polygons_overlap(poly1: &Polygon, poly2: &Polygon) -> bool {
    if poly1.min_x > poly2.max_x - 1e-8 || poly1.max_x < poly2.min_x + 1e-8 ||
       poly1.min_y > poly2.max_y - 1e-8 || poly1.max_y < poly2.min_y + 1e-8 {
        return false;
    }

    if (poly1.min_x - poly2.min_x).abs() < 1e-9 && (poly1.min_y - poly2.min_y).abs() < 1e-9 &&
       (poly1.max_x - poly2.max_x).abs() < 1e-9 && (poly1.max_y - poly2.max_y).abs() < 1e-9 &&
       poly1.vertices.len() == poly2.vertices.len() {
        return true;
    }

    for i in 0..poly1.vertices.len() {
        let p1 = poly1.vertices[i];
        let p2 = poly1.vertices[(i + 1) % poly1.vertices.len()];
        for j in 0..poly2.vertices.len() {
            let p3 = poly2.vertices[j];
            let p4 = poly2.vertices[(j + 1) % poly2.vertices.len()];
            if segments_intersect(p1, p2, p3, p4) {
                return true;
            }
        }
    }
    
    let mut test_points1 = Vec::new();
    let n1 = poly1.vertices.len();
    let mut cx1 = 0.0; let mut cy1 = 0.0;
    for &v in &poly1.vertices { cx1 += v.x; cy1 += v.y; }
    test_points1.push(Point { x: cx1 / n1 as f64, y: cy1 / n1 as f64 });
    for i in 0..n1 {
        let v1 = poly1.vertices[i];
        let v2 = poly1.vertices[(i + 1) % n1];
        test_points1.push(Point { x: (v1.x + v2.x) / 2.0, y: (v1.y + v2.y) / 2.0 });
    }

    let mut test_points2 = Vec::new();
    let n2 = poly2.vertices.len();
    let mut cx2 = 0.0; let mut cy2 = 0.0;
    for &v in &poly2.vertices { cx2 += v.x; cy2 += v.y; }
    test_points2.push(Point { x: cx2 / n2 as f64, y: cy2 / n2 as f64 });
    for i in 0..n2 {
        let v1 = poly2.vertices[i];
        let v2 = poly2.vertices[(i + 1) % n2];
        test_points2.push(Point { x: (v1.x + v2.x) / 2.0, y: (v1.y + v2.y) / 2.0 });
    }

    let eps_sq = 1e-11;
    for &p in &test_points1 {
        if poly2.contains_point(p) {
            let mut on_boundary = false;
            for j in 0..n2 {
                if dist_point_to_segment_sq(p, poly2.vertices[j], poly2.vertices[(j + 1) % n2]) < eps_sq {
                    on_boundary = true; break;
                }
            }
            if !on_boundary { return true; }
        }
    }
    for &p in &test_points2 {
        if poly1.contains_point(p) {
            let mut on_boundary = false;
            for j in 0..n1 {
                if dist_point_to_segment_sq(p, poly1.vertices[j], poly1.vertices[(j + 1) % n1]) < eps_sq {
                    on_boundary = true; break;
                }
            }
            if !on_boundary { return true; }
        }
    }

    false
}

fn is_poly_inside_stock(piece: &Polygon, stock: &Polygon) -> bool {
    if piece.min_x < stock.min_x - 1e-8 || piece.max_x > stock.max_x + 1e-8 ||
       piece.min_y < stock.min_y - 1e-8 || piece.max_y > stock.max_y + 1e-8 {
        return false;
    }
    for &v in &piece.vertices {
        if !stock.contains_point(v) {
            return false;
        }
    }
    for i in 0..piece.vertices.len() {
        let p1 = piece.vertices[i];
        let p2 = piece.vertices[(i + 1) % piece.vertices.len()];
        for j in 0..stock.vertices.len() {
            let p3 = stock.vertices[j];
            let p4 = stock.vertices[(j + 1) % stock.vertices.len()];
            if segments_intersect(p1, p2, p3, p4) {
                return false;
            }
        }
    }
    true
}

struct SpatialIndex {
    cells: Vec<Vec<usize>>,
    cell_size_x: f64,
    cell_size_y: f64,
    min_x: f64,
    min_y: f64,
    nx: usize,
    ny: usize,
}

impl SpatialIndex {
    fn new(stock_poly: &Polygon, nx: usize, ny: usize) -> Self {
        let cell_size_x = (stock_poly.max_x - stock_poly.min_x) / nx as f64;
        let cell_size_y = (stock_poly.max_y - stock_poly.min_y) / ny as f64;
        SpatialIndex {
            cells: vec![vec![]; nx * ny],
            cell_size_x,
            cell_size_y,
            min_x: stock_poly.min_x,
            min_y: stock_poly.min_y,
            nx,
            ny,
        }
    }

    fn add(&mut self, piece_poly: &Polygon, piece_idx: usize) {
        let i_min = (((piece_poly.min_x - self.min_x) / self.cell_size_x).floor() as isize).max(0) as usize;
        let i_max = (((piece_poly.max_x - self.min_x) / self.cell_size_x).floor() as isize).min(self.nx as isize - 1) as usize;
        let j_min = (((piece_poly.min_y - self.min_y) / self.cell_size_y).floor() as isize).max(0) as usize;
        let j_max = (((piece_poly.max_y - self.min_y) / self.cell_size_y).floor() as isize).min(self.ny as isize - 1) as usize;
        for i in i_min..=i_max {
            for j in j_min..=j_max {
                self.cells[i + j * self.nx].push(piece_idx);
            }
        }
    }

    fn get_candidates(&self, piece_poly: &Polygon) -> Vec<usize> {
        let i_min = (((piece_poly.min_x - self.min_x) / self.cell_size_x).floor() as isize).max(0) as usize;
        let i_max = (((piece_poly.max_x - self.min_x) / self.cell_size_x).floor() as isize).min(self.nx as isize - 1) as usize;
        let j_min = (((piece_poly.min_y - self.min_y) / self.cell_size_y).floor() as isize).max(0) as usize;
        let j_max = (((piece_poly.max_y - self.min_y) / self.cell_size_y).floor() as isize).min(self.ny as isize - 1) as usize;
        let mut candidates = Vec::new();
        for i in i_min..=i_max {
            for j in j_min..=j_max {
                candidates.extend(&self.cells[i + j * self.nx]);
            }
        }
        candidates.sort();
        candidates.dedup();
        candidates
    }
}

struct PlacedPiece {
    poly: Polygon,
}

struct StockInstance {
    placed: Vec<PlacedPiece>,
    spatial_index: SpatialIndex,
}

struct RotatedPiece {
    rotation: f64,
    poly: Polygon,
}

struct Piece {
    id: usize,
    area: f64,
    rotations: Vec<RotatedPiece>,
}

fn find_position(piece: &Polygon, stock_poly: &Polygon, stock: &StockInstance, steps: usize) -> Option<(f64, f64)> {
    let dx_min = stock_poly.min_x - piece.min_x;
    let dx_max = stock_poly.max_x - piece.max_x;
    let dy_min = stock_poly.min_y - piece.min_y;
    let dy_max = stock_poly.max_y - piece.max_y;
    if dx_min > dx_max + 1e-8 || dy_min > dy_max + 1e-8 { return None; }
    let dx_step = if steps > 0 { (dx_max - dx_min) / steps as f64 } else { 0.0 };
    let dy_step = if steps > 0 { (dy_max - dy_min) / steps as f64 } else { 0.0 };

    for j in 0..=steps {
        let dy = dy_min + j as f64 * dy_step;
        for i in 0..=steps {
            let dx = dx_min + i as f64 * dx_step;
            let translated_piece = piece.translate(dx, dy);
            if !is_poly_inside_stock(&translated_piece, stock_poly) { continue; }
            let mut overlap = false;
            let candidates = stock.spatial_index.get_candidates(&translated_piece);
            for &idx in &candidates {
                if polygons_overlap(&translated_piece, &stock.placed[idx].poly) {
                    overlap = true;
                    break;
                }
            }
            if !overlap { return Some((dx, dy)); }
        }
    }
    None
}

fn main() {
    let start_time = Instant::now();
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    let first_line = match lines.next() {
        Some(Ok(l)) => l,
        _ => return,
    };
    let parts: Vec<usize> = first_line.split_whitespace().map(|s| s.parse().unwrap()).collect();
    let sv = parts[0];
    let np = parts[1];
    let mut stock_vertices = Vec::new();
    for _ in 0..sv {
        let line = lines.next().unwrap().unwrap();
        let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
        stock_vertices.push(Point { x: coords[0], y: coords[1] });
    }
    let stock_poly = Polygon::new(stock_vertices);
    let mut pieces = Vec::new();
    for i in 0..np {
        let pv: usize = lines.next().unwrap().unwrap().trim().parse().unwrap();
        let mut vertices = Vec::new();
        for _ in 0..pv {
            let line = lines.next().unwrap().unwrap();
            let coords: Vec<f64> = line.split_whitespace().map(|s| s.parse().unwrap()).collect();
            vertices.push(Point { x: coords[0], y: coords[1] });
        }
        let poly = Polygon::new(vertices);
        let mut rotations = Vec::new();
        for &r in &[0.0, std::f64::consts::FRAC_PI_2, std::f64::consts::PI, 3.0 * std::f64::consts::FRAC_PI_2] {
            rotations.push(RotatedPiece { rotation: r, poly: poly.rotate(r) });
        }
        pieces.push(Piece { id: i, area: poly.area(), rotations });
    }
    pieces.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap());
    let mut stocks: Vec<StockInstance> = Vec::new();
    let mut results = vec![(0, 0.0, 0.0, 0.0); np];
    let nx = 50;
    let ny = 50;
    for piece in pieces {
        let mut placed_info = None;
        let elapsed = start_time.elapsed().as_secs();
        let steps = if np > 1000 {
            if elapsed > 250 { 5 } else if elapsed > 150 { 10 } else { 20 }
        } else {
            if elapsed > 250 { 10 } else if elapsed > 150 { 20 } else { 40 }
        };

        'rot_loop: for rp in &piece.rotations {
            for (s_idx, stock) in stocks.iter_mut().enumerate() {
                if let Some((dx, dy)) = find_position(&rp.poly, &stock_poly, stock, steps) {
                    placed_info = Some((s_idx, dx, dy, rp.rotation, &rp.poly));
                    break 'rot_loop;
                }
            }
            let new_stock = StockInstance {
                placed: Vec::new(),
                spatial_index: SpatialIndex::new(&stock_poly, nx, ny),
            };
            if let Some((dx, dy)) = find_position(&rp.poly, &stock_poly, &new_stock, steps) {
                placed_info = Some((stocks.len(), dx, dy, rp.rotation, &rp.poly));
                break 'rot_loop;
            }
        }
        if let Some((s_idx, dx, dy, rot, poly)) = placed_info {
            let final_poly = poly.translate(dx, dy);
            if s_idx == stocks.len() {
                let mut stock = StockInstance {
                    placed: Vec::new(),
                    spatial_index: SpatialIndex::new(&stock_poly, nx, ny),
                };
                stock.spatial_index.add(&final_poly, 0);
                stock.placed.push(PlacedPiece { poly: final_poly });
                stocks.push(stock);
            } else {
                let stock = &mut stocks[s_idx];
                stock.spatial_index.add(&final_poly, stock.placed.len());
                stock.placed.push(PlacedPiece { poly: final_poly });
            }
            results[piece.id] = (s_idx, dx, dy, rot);
        }
    }
    println!("{}", stocks.len());
    for i in 0..np {
        let (s_idx, x, y, rot) = results[i];
        println!("{} {} {} {}", s_idx, x, y, rot);
    }
}
