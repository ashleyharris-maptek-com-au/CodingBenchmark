// WGSL Compute Shader for Hash Mining (SHA-256 / SHA256d)

struct MiningInput {
    midstate: array<u32, 8>, // Optional midstate to resume hashing
    data: array<u32, 16>,     // 64 bytes of data to hash (including padding)
    target: array<u32, 8>,   // 256-bit target (difficulty)
    nonce_start: u32,
    nonce_offset: u32,       // Index in 'data' where the nonce should be placed
};

struct MiningOutput {
    found: u32,
    nonce: u32,
    hash: array<u32, 8>,
};

@group(0) @binding(0) var<uniform> input: MiningInput;
@group(0) @binding(1) var<storage, read_write> output: MiningOutput;

const K = array<u32, 64>(
    0x428a2f98u, 0x71374491u, 0xb5c0fbcfu, 0xe9b5dba5u, 0x3956c25bu, 0x59f111f1u, 0x923f82a4u, 0xab1c5ed5u,

    0xe49b69c1u, 0xefbe4786u, 0x0fc19dc6u, 0x240ca1ccu, 0x2de92c6fu, 0x4a7484aau, 0x5cb0a9dcu, 0x76f988dau,
    0x983e5152u, 0xa831c66du, 0xb00327c8u, 0xbf597fc7u, 0xc6e00bf3u, 0xd5a79147u, 0x06ca6351u, 0x14292967u,
    0x27b70a85u, 0x2e1b2138u, 0x4d2c6dfcu, 0x53380d13u, 0x650a7354u, 0x766a0abbu, 0x81c2c92eu, 0x92722c85u,

    0x19a4c116u, 0x1e376c08u, 0x2748774cu, 0x34b0bcb5u, 0x391c0cb3u, 0x4ed8aa4au, 0x5b9cca4fu, 0x682e6ff3u,
    0x748f82eeu, 0x78a5636fu, 0x84c87814u, 0x8cc70208u, 0x90befffeu, 0xa4506cebu, 0xbef9a3f7u, 0xc67178f2u
);

const H_INIT = array<u32, 8>(
    0x6a09e667u, 0xbb67ae85u, 0x3c6ef372u, 0xa54ff53au,
    0x510e527fu, 0x9b05688cu, 0x1f83d9abu, 0x5be0cd19u
);

fn rotr(x: u32, n: u32) -> u32 {
    return (x >> n) | (x << (32u - n));
}

fn sha256_block(state: array<u32, 8>, data: array<u32, 16>) -> array<u32, 8> {
    var w: array<u32, 64>;
    for (var i = 0u; i < 16u; i = i + 1u) {
        w[i] = data[i];
    }
    for (var i = 16u; i < 64u; i = i + 1u) {
        let s0 = rotr(w[i - 15u], 7u) ^ rotr(w[i - 15u], 18u) ^ (w[i - 15u] >> 3u);
        let s1 = rotr(w[i - 2u], 17u) ^ rotr(w[i - 2u], 19u) ^ (w[i - 2u] >> 10u);
        w[i] = w[i - 16u] + s0 + w[i - 7u] + s1;
    }

    var a = state[0];
    var b = state[1];
    var c = state[2];
    var d = state[3];
    var e = state[4];
    var f = state[5];
    var g = state[6];
    var h = state[7];

    for (var i = 0u; i < 64u; i = i + 1u) {
        let s1 = rotr(e, 6u) ^ rotr(e, 11u) ^ rotr(e, 25u);
        let ch = (e & f) ^ (~e & g);
        let temp1 = h + s1 + ch + K[i] + w[i];
        let s0 = rotr(a, 2u) ^ rotr(a, 13u) ^ rotr(a, 22u);
        let maj = (a & b) ^ (a & c) ^ (b & c);
        let temp2 = s0 + maj;

        h = g;
        g = f;
        f = e;
        e = d + temp1;
        d = c;
        c = b;
        b = a;
        a = temp1 + temp2;
    }

    var result: array<u32, 8>;
    result[0] = a + state[0];
    result[1] = b + state[1];
    result[2] = c + state[2];
    result[3] = d + state[3];
    result[4] = e + state[4];
    result[5] = f + state[5];
    result[6] = g + state[6];
    result[7] = h + state[7];
    return result;
}

fn is_less_than_or_equal(a: array<u32, 8>, b: array<u32, 8>) -> bool {
    for (var i = 0u; i < 8u; i = i + 1u) {
        if (a[i] < b[i]) { return true; }
        if (a[i] > b[i]) { return false; }
    }
    return true;
}

@compute @workgroup_size(64)
fn main(@builtin(global_invocation_id) global_id: vec3<u32>) {
    let nonce = input.nonce_start + global_id.x;
    
    // 1. Prepare data with nonce
    var data = input.data;
    data[input.nonce_offset % 16u] = nonce;

    // 2. First SHA-256 Pass
    // Use input.midstate if provided (not all zeros), otherwise use H_INIT
    var state = H_INIT;
    var use_midstate = false;
    for(var i=0u; i<8u; i=i+1u) { if(input.midstate[i] != 0u) { use_midstate = true; break; } }
    if(use_midstate) { state = input.midstate; }
    
    let hash1 = sha256_block(state, data);

    // 3. Second SHA-256 Pass (SHA256d)
    // Common in Bitcoin: hash the result of the first pass
    var data2: array<u32, 16>;
    for(var i=0u; i<8u; i=i+1u) { data2[i] = hash1[i]; }
    data2[8] = 0x80000000u; // Padding: 1 followed by zeros
    for(var i=9u; i<15u; i=i+1u) { data2[i] = 0u; }
    data2[15] = 256u;       // Length in bits (32 bytes * 8 = 256)

    let final_hash = sha256_block(H_INIT, data2);

    // 4. Check against target
    if (is_less_than_or_equal(final_hash, input.target)) {
        // Atomic check to avoid multiple threads writing simultaneously if found
        // For simplicity in this example, we just write.
        output.found = 1u;
        output.nonce = nonce;
        output.hash = final_hash;
    }
}
