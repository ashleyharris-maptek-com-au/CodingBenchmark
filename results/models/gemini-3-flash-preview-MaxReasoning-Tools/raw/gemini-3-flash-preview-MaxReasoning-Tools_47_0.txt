@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write>  result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>              params   : vec4<u32>;

// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl(x: u32, shift: u32) -> u32 {
    return (x << shift) | (x >> (32u - shift));
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid : vec3<u32>) {
    let difficulty = params.x;
    let range_start = params.y;
    let range_size = params.z;

    if (gid.x >= range_size) {
        return;
    }

    let nonce = range_start + gid.x;

    var state : array<u32, 16>;
    for (var i = 0u; i < 16u; i = i + 1u) {
        state[i] = baseData[i] ^ (nonce + i);
    }

    for (var round = 0u; round < 8u; round = round + 1u) {
        // Group 0: 0, 1, 2, 3
        state[0] += state[1]; state[3] ^= state[0]; state[3] = rotl(state[3], 16u);
        state[2] += state[3]; state[1] ^= state[2]; state[1] = rotl(state[1], 12u);
        state[0] += state[1]; state[3] ^= state[0]; state[3] = rotl(state[3], 8u);
        state[2] += state[3]; state[1] ^= state[2]; state[1] = rotl(state[1], 7u);

        // Group 1: 4, 5, 6, 7
        state[4] += state[5]; state[7] ^= state[4]; state[7] = rotl(state[7], 16u);
        state[6] += state[7]; state[5] ^= state[6]; state[5] = rotl(state[5], 12u);
        state[4] += state[5]; state[7] ^= state[4]; state[7] = rotl(state[7], 8u);
        state[6] += state[7]; state[5] ^= state[6]; state[5] = rotl(state[5], 7u);

        // Group 2: 8, 9, 10, 11
        state[8] += state[9]; state[11] ^= state[8]; state[11] = rotl(state[11], 16u);
        state[10] += state[11]; state[9] ^= state[10]; state[9] = rotl(state[9], 12u);
        state[8] += state[9]; state[11] ^= state[8]; state[11] = rotl(state[11], 8u);
        state[10] += state[11]; state[9] ^= state[10]; state[9] = rotl(state[9], 7u);

        // Group 3: 12, 13, 14, 15
        state[12] += state[13]; state[15] ^= state[12]; state[15] = rotl(state[15], 16u);
        state[14] += state[15]; state[13] ^= state[14]; state[13] = rotl(state[13], 12u);
        state[12] += state[13]; state[15] ^= state[12]; state[15] = rotl(state[15], 8u);
        state[14] += state[15]; state[13] ^= state[14]; state[13] = rotl(state[13], 7u);
    }

    let h0 = state[0] ^ state[2] ^ state[4] ^ state[6] ^ state[8] ^ state[10] ^ state[12] ^ state[14];
    let h1 = state[1] ^ state[3] ^ state[5] ^ state[7] ^ state[9] ^ state[11] ^ state[13] ^ state[15];

    var tz: u32 = 0u;
    if (h0 == 0u) {
        tz = 32u + countTrailingZeros(h1);
    } else {
        tz = countTrailingZeros(h0);
    }

    if (tz >= difficulty) {
        result[0] = 1u;
        result[1] = nonce;
        result[2] = h0;
        result[3] = h1;
    }
}
