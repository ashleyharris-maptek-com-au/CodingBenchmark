OpCapability Geometry
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpSource GLSL 450
               OpName %main "main"
               OpName %gl_in "gl_in"
               OpName %worldPos_in "worldPos"
               OpName %normal_in "normal"
               OpName %color_in "color"
               OpName %gl_out "gl_out"
               OpName %outColor "outColor"
               OpDecorate %gl_in_block Block
               OpMemberDecorate %gl_in_block 0 BuiltIn Position
               OpDecorate %gl_out_block Block
               OpMemberDecorate %gl_out_block 0 BuiltIn Position
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_1 = OpConstant %uint 1
%gl_in_block = OpTypeStruct %v4float
%gl_in_array = OpTypeArray %gl_in_block %uint_1
%ptr_in_gl_in_array = OpTypePointer Input %gl_in_array
      %gl_in = OpVariable %ptr_in_gl_in_array Input
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
%ptr_in_v4float = OpTypePointer Input %v4float
%v3float_arr = OpTypeArray %v3float %uint_1
%ptr_in_v3float_arr = OpTypePointer Input %v3float_arr
%worldPos_in = OpVariable %ptr_in_v3float_arr Input
 %normal_in  = OpVariable %ptr_in_v3float_arr Input
   %color_in = OpVariable %ptr_in_v3float_arr Input
%ptr_in_v3float = OpTypePointer Input %v3float
%gl_out_block = OpTypeStruct %v4float
%ptr_out_gl_out_block = OpTypePointer Output %gl_out_block
     %gl_out = OpVariable %ptr_out_gl_out_block Output
%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output
%ptr_out_v4float = OpTypePointer Output %v4float
    %h_const = OpConstant %float 0.04
    %z_const = OpConstant %float 0.0
    %w_const = OpConstant %float 1.0
       %main = OpFunction %void None %3
          %5 = OpLabel
     %ptr_pos = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
        %pos = OpLoad %v4float %ptr_pos
          %x = OpCompositeExtract %float %pos 0
          %y = OpCompositeExtract %float %pos 1
   %ptr_color = OpAccessChain %ptr_in_v3float %color_in %int_0
      %color = OpLoad %v3float %ptr_color
         %xm = OpFSub %float %x %h_const
         %xp = OpFAdd %float %x %h_const
         %ym = OpFSub %float %y %h_const
         %yp = OpFAdd %float %y %h_const
%ptr_out_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0
     %v1_pos = OpCompositeConstruct %v4float %xm %ym %z_const %w_const
               OpStore %ptr_out_pos %v1_pos
               OpStore %outColor %color
               OpEmitVertex
     %v2_pos = OpCompositeConstruct %v4float %xp %ym %z_const %w_const
               OpStore %ptr_out_pos %v2_pos
               OpStore %outColor %color
               OpEmitVertex
     %v3_pos = OpCompositeConstruct %v4float %xm %yp %z_const %w_const
               OpStore %ptr_out_pos %v3_pos
               OpStore %outColor %color
               OpEmitVertex
     %v4_pos = OpCompositeConstruct %v4float %xp %yp %z_const %w_const
               OpStore %ptr_out_pos %v4_pos
               OpStore %outColor %color
               OpEmitVertex
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
