               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
     %uint_1 = OpConstant %uint 1
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input
 %arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
 %ptr_in_v4 = OpTypePointer Input %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float
    %float_h = OpConstant %float 0.04
    %float_0 = OpConstant %float 0
    %float_1 = OpConstant %float 1
       %main = OpFunction %void None %3
          %5 = OpLabel
    %p_color = OpAccessChain %ptr_in_v3 %color %int_0
  %color_val = OpLoad %v3float %p_color
      %p_pos = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
    %pos_val = OpLoad %v4float %p_pos
      %pos_x = OpCompositeExtract %float %pos_val 0
      %pos_y = OpCompositeExtract %float %pos_val 1
         %x1 = OpFSub %float %pos_x %float_h
         %y1 = OpFSub %float %pos_y %float_h
         %v1 = OpCompositeConstruct %v4float %x1 %y1 %float_0 %float_1
               OpStore %outColor %color_val
         %o1 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %o1 %v1
               OpEmitVertex
         %x2 = OpFAdd %float %pos_x %float_h
         %y2 = OpFSub %float %pos_y %float_h
         %v2 = OpCompositeConstruct %v4float %x2 %y2 %float_0 %float_1
               OpStore %outColor %color_val
         %o2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %o2 %v2
               OpEmitVertex
         %x3 = OpFSub %float %pos_x %float_h
         %y3 = OpFAdd %float %pos_y %float_h
         %v3 = OpCompositeConstruct %v4float %x3 %y3 %float_0 %float_1
               OpStore %outColor %color_val
         %o3 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %o3 %v3
               OpEmitVertex
         %x4 = OpFAdd %float %pos_x %float_h
         %y4 = OpFAdd %float %pos_y %float_h
         %v4 = OpCompositeConstruct %v4float %x4 %y4 %float_0 %float_1
               OpStore %outColor %color_val
         %o4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %o4 %v4
               OpEmitVertex
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
