#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // 8x16 UV scale for the brick pattern
    vec2 scaledUV = uv * vec2(8.0, 16.0);
    
    // Staggered rows: offset every other row by half a brick (0.5)
    if (mod(floor(scaledUV.y), 2.0) > 0.5) {
        scaledUV.x += 0.5;
    }
    
    // Fractional position within each brick cell [0, 1]
    vec2 f = fract(scaledUV);
    
    // Mortar thickness: 0.05 in X, 0.1 in Y
    // brick color: (0.7, 0.2, 0.1), mortar color: (0.7, 0.7, 0.7)
    float isBrick = step(0.05, f.x) * step(0.1, f.y);
    vec3 baseColor = mix(vec3(0.7, 0.7, 0.7), vec3(0.7, 0.2, 0.1), isBrick);
    
    // Lambertian shading: factor = (0.4 + 0.6 * diffuse)
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.4 + 0.6 * diffuse;
    
    // Output final color with alpha 1.0
    outColor = vec4(baseColor * shadingFactor, 1.0);
}
