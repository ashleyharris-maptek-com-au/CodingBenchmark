               OpCapability Shader
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Fragment %main "main" %worldPos %normal %outColor
               OpExecutionMode %main OriginUpperLeft

               ; Decorations
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %outColor Location 0

               OpDecorate %UBO Block
               OpMemberDecorate %UBO 0 ColMajor
               OpMemberDecorate %UBO 0 Offset 0
               OpMemberDecorate %UBO 0 MatrixStride 16
               OpMemberDecorate %UBO 1 ColMajor
               OpMemberDecorate %UBO 1 Offset 64
               OpMemberDecorate %UBO 1 MatrixStride 16
               OpMemberDecorate %UBO 2 ColMajor
               OpMemberDecorate %UBO 2 Offset 128
               OpMemberDecorate %UBO 2 MatrixStride 16
               OpMemberDecorate %UBO 3 Offset 192
               OpMemberDecorate %UBO 4 Offset 208
               OpMemberDecorate %UBO 5 Offset 224
               OpDecorate %ubo DescriptorSet 0
               OpDecorate %ubo Binding 0

               ; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
%mat4v4float = OpTypeMatrix %v4float 4

        %UBO = OpTypeStruct %mat4v4float %mat4v4float %mat4v4float %v4float %v4float %v4float

%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%_ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float

   %worldPos = OpVariable %_ptr_Input_v3float Input
     %normal = OpVariable %_ptr_Input_v3float Input
   %outColor = OpVariable %_ptr_Output_v4float Output
        %ubo = OpVariable %_ptr_Uniform_UBO Uniform

        %int = OpTypeInt 32 1
      %int_3 = OpConstant %int 3

               ; Constants
    %float_0 = OpConstant %float 0
  %float_0_5 = OpConstant %float 0.5
    %float_1 = OpConstant %float 1
  %float_0_3 = OpConstant %float 0.3
  %float_0_4 = OpConstant %float 0.4
  %float_0_6 = OpConstant %float 0.6
 %float_0_15 = OpConstant %float 0.15
 %float_0_05 = OpConstant %float 0.05

 %skyColor = OpConstantComposite %v3float %float_0_4 %float_0_6 %float_1
%groundColor = OpConstantComposite %v3float %float_0_3 %float_0_15 %float_0_05

       %main = OpFunction %void None %3
      %label = OpLabel

               ; 1. Normalize normal
   %norm_val = OpLoad %v3float %normal
          %N = OpExtInst %v3float %1 Normalize %norm_val

               ; 2. Hemisphere ambient
         %Ny = OpCompositeExtract %float %N 1
       %tmp1 = OpFMul %float %Ny %float_0_5
     %factor = OpFAdd %float %tmp1 %float_0_5

               ; mix(groundColor, skyColor, factor) = groundColor*(1-factor) + skyColor*factor
     %f_inv = OpFSub %float %float_1 %factor
     %tmp2 = OpVectorTimesScalar %v3float %groundColor %f_inv
     %tmp3 = OpVectorTimesScalar %v3float %skyColor %factor
    %ambient = OpFAdd %v3float %tmp2 %tmp3

               ; 3. Diffuse term
     %lp_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3
    %lp_vec4 = OpLoad %v4float %lp_ptr
     %lp_xyz = OpVectorShuffle %v3float %lp_vec4 %lp_vec4 0 1 2
     %wp_val = OpLoad %v3float %worldPos
   %L_unnorm = OpFSub %v3float %lp_xyz %wp_val
          %L = OpExtInst %v3float %1 Normalize %L_unnorm

      %dotNL = OpDot %float %N %L
          %d = OpExtInst %float %1 FMax %dotNL %float_0
    %diffuse_val = OpFMul %float %float_0_3 %d
    %diffuse = OpCompositeConstruct %v3float %diffuse_val %diffuse_val %diffuse_val

               ; 4. Final color
  %final_rgb = OpFAdd %v3float %ambient %diffuse
 %final_rgba = OpCompositeConstruct %v4float %final_rgb %float_1
               OpStore %outColor %final_rgba

               OpReturn
               OpFunctionEnd
