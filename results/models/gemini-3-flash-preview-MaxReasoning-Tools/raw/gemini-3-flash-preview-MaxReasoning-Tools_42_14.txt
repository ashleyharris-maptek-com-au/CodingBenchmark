               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor %worldPos %normal
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
     %uint_6 = OpConstant %uint 6
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_2 = OpConstant %int 2
      %int_4 = OpConstant %int 4

%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

 %arr_v3_in = OpTypeArray %v3float %uint_6
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
%ptr_in_v3 = OpTypePointer Input %v3float

; Constants
    %float_0 = OpConstant %float 0
    %float_0_2 = OpConstant %float 0.2
    %float_0_8 = OpConstant %float 0.8
    %float_0_577 = OpConstant %float 0.577350269
      %vecL = OpConstantComposite %v3float %float_0_577 %float_0_577 %float_0_577
 %float_inv3 = OpConstant %float 0.333333333
   %inv3vec = OpConstantComposite %v3float %float_inv3 %float_inv3 %float_inv3

       %void = OpTypeVoid
   %main_type = OpTypeFunction %void

       %main = OpFunction %void None %main_type
 %main_label = OpLabel

; Get P0, P2, P4 from gl_in
      %ptr0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0_v4 = OpLoad %v4float %ptr0
        %p0 = OpVectorShuffle %v3float %p0_v4 %p0_v4 0 1 2

      %ptr2 = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
        %p2_v4 = OpLoad %v4float %ptr2
        %p2 = OpVectorShuffle %v3float %p2_v4 %p2_v4 0 1 2

      %ptr4 = OpAccessChain %ptr_in_v4 %gl_in %int_4 %int_0
        %p4_v4 = OpLoad %v4float %ptr4
        %p4 = OpVectorShuffle %v3float %p4_v4 %p4_v4 0 1 2

; Compute face normal: N = normalize(cross(p2-p0, p4-p0))
         %A = OpFSub %v3float %p2 %p0
         %B = OpFSub %v3float %p4 %p0

        %ax = OpCompositeExtract %float %A 0
        %ay = OpCompositeExtract %float %A 1
        %az = OpCompositeExtract %float %A 2
        %bx = OpCompositeExtract %float %B 0
        %by = OpCompositeExtract %float %B 1
        %bz = OpCompositeExtract %float %B 2

       %aybz = OpFMul %float %ay %bz
       %azby = OpFMul %float %az %by
       %nx = OpFSub %float %aybz %azby

       %azbx = OpFMul %float %az %bx
       %axbz = OpFMul %float %ax %bz
       %ny = OpFSub %float %azbx %axbz

       %axby = OpFMul %float %ax %by
       %aybx = OpFMul %float %ay %bx
       %nz = OpFSub %float %axby %aybx

    %crossAB = OpCompositeConstruct %v3float %nx %ny %nz
      %normN = OpExtInst %v3float %1 Normalize %crossAB

; Diffuse = max(dot(N, L), 0.0), where L is normalize(vec3(1,1,1))
      %dotNL = OpDot %float %normN %vecL
    %diffuse = OpExtInst %float %1 FMax %dotNL %float_0

; Intensity = 0.2 + 0.8 * diffuse
   %diff_08 = OpFMul %float %diffuse %float_0_8
  %intensity = OpFAdd %float %float_0_2 %diff_08

; Base color = (color[0] + color[2] + color[4]) / 3.0
      %c0ptr = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %c0ptr
      %c2ptr = OpAccessChain %ptr_in_v3 %color %int_2
         %c2 = OpLoad %v3float %c2ptr
      %c4ptr = OpAccessChain %ptr_in_v3 %color %int_4
         %c4 = OpLoad %v3float %c4ptr

      %c02 = OpFAdd %v3float %c0 %c2
      %c024 = OpFAdd %v3float %c02 %c4
 %avgColor = OpFMul %v3float %c024 %inv3vec

; Final color = base * intensity
 %finalColor = OpVectorTimesScalar %v3float %avgColor %intensity

; Output vertex 0
               OpStore %outColor %finalColor
    %out_pos0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos0 %p0_v4
               OpEmitVertex

; Output vertex 2
               OpStore %outColor %finalColor
    %out_pos2 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos2 %p2_v4
               OpEmitVertex

; Output vertex 4
               OpStore %outColor %finalColor
    %out_pos4 = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %out_pos4 %p4_v4
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
