struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // 8x16 UV scale
    float2 brickUV = input.uv * float2(8.0, 16.0);
    
    // Staggered rows: half-brick offset every other row
    float row = floor(brickUV.y);
    brickUV.x += step(1.0, fmod(row, 2.0)) * 0.5;
    
    // Local coordinates within the brick cell
    float2 cellUV = frac(brickUV);
    
    // Mortar check: thickness 0.05 in X, 0.1 in Y
    // step(edge, x) returns 1.0 if x >= edge, else 0.0
    float2 isBrick2D = step(float2(0.05, 0.1), cellUV);
    float isBrick = isBrick2D.x * isBrick2D.y;
    
    // Colors
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = lerp(mortarColor, brickColor, isBrick);
    
    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float factor = 0.4 + 0.6 * diffuse;
    
    return float4(baseColor * factor, 1.0);
}
