This Node.js TypeScript program implements a **List Scheduler** using an event-based simulation to efficiently assign mission-control tasks to parallel workstations.

### Key Strategies:
1.  **Effective Deadlines & Latest Start Times**: The program computes `effectiveDeadline` for each task by propagating deadline constraints backwards through the dependency graph. This ensures that tasks are prioritized based on the urgency of their entire downstream path. The `latestStart` time (effective deadline minus duration) is used as the primary priority metric to minimize weighted lateness.
2.  **Weighted Priority (Smith's Rule)**: As a secondary priority, the program uses the ratio of `weight / duration`. This is an extension of Smith's Rule for minimizing weighted completion time, adapted for a multi-machine environment with dependencies.
3.  **Critical Path Tie-breaking**: For tasks with similar deadline constraints, the program prioritizes those on the critical path (longest downstream duration) to minimize the overall makespan.
4.  **Scalability**: The implementation utilizes a binary heap-based **Priority Queue** for managing both ready tasks and scheduling events. This ensures $O(N \log N)$ complexity, making it suitable for large-scale scenarios without redundant machine scanning or time-step iterations.

```typescript
declare const require: any;
declare const process: any;
const fs = require('fs');

interface Task {
    id: string;
    duration: number;
    deps: string[];
    deadline: number;
    weight: number;
}

interface InputData {
    machineCount: number;
    tasks: Task[];
}

interface ScheduledTask {
    id: string;
    start: number;
    machine: number;
}

/**
 * Binary Heap based Priority Queue for efficient task management.
 */
class PriorityQueue<T> {
    private heap: T[] = [];
    constructor(private compare: (a: T, b: T) => number) {}

    push(val: T) {
        this.heap.push(val);
        this.bubbleUp();
    }

    pop(): T | undefined {
        if (this.size() === 0) return undefined;
        const top = this.heap[0];
        const bottom = this.heap.pop()!;
        if (this.size() > 0) {
            this.heap[0] = bottom;
            this.bubbleDown();
        }
        return top;
    }

    size() {
        return this.heap.length;
    }

    peek(): T | undefined {
        return this.heap[0];
    }

    private bubbleUp() {
        let index = this.heap.length - 1;
        while (index > 0) {
            let parentIndex = (index - 1) >> 1;
            if (this.compare(this.heap[index], this.heap[parentIndex]) < 0) {
                [this.heap[index], this.heap[parentIndex]] = [this.heap[parentIndex], this.heap[index]];
                index = parentIndex;
            } else {
                break;
            }
        }
    }

    private bubbleDown() {
        let index = 0;
        while (true) {
            let left = (index << 1) + 1;
            let right = (index << 1) + 2;
            let smallest = index;

            if (left < this.heap.length && this.compare(this.heap[left], this.heap[smallest]) < 0) {
                smallest = left;
            }
            if (right < this.heap.length && this.compare(this.heap[right], this.heap[smallest]) < 0) {
                smallest = right;
            }

            if (smallest !== index) {
                [this.heap[index], this.heap[smallest]] = [this.heap[smallest], this.heap[index]];
                index = smallest;
            } else {
                break;
            }
        }
    }
}

function solve() {
    // Read input from stdin
    const inputStr: string = fs.readFileSync(0, 'utf8');
    if (!inputStr) return;
    const data: InputData = JSON.parse(inputStr);

    const { machineCount, tasks } = data;
    const taskMap = new Map<string, Task>();
    for (const t of tasks) taskMap.set(t.id, t);

    // Build dependency graph
    const adj = new Map<string, string[]>();
    const inDegree = new Map<string, number>();

    for (const t of tasks) {
        adj.set(t.id, []);
        inDegree.set(t.id, t.deps.length);
    }

    for (const t of tasks) {
        for (const depId of t.deps) {
            if (!adj.has(depId)) adj.set(depId, []);
            adj.get(depId)!.push(t.id);
        }
    }

    // Topological Sort to calculate heuristics
    const topoOrder: string[] = [];
    const q: string[] = [];
    const tempInDegree = new Map(inDegree);
    for (const t of tasks) {
        if (tempInDegree.get(t.id) === 0) q.push(t.id);
    }

    let head = 0;
    while(head < q.length) {
        const u = q[head++];
        topoOrder.push(u);
        const children = adj.get(u);
        if (children) {
            for (const v of children) {
                const newDeg = tempInDegree.get(v)! - 1;
                tempInDegree.set(v, newDeg);
                if (newDeg === 0) q.push(v);
            }
        }
    }

    // Heuristics: Critical Path and Effective Deadlines
    const criticalPathLength = new Map<string, number>();
    const effectiveDeadline = new Map<string, number>();

    // Iterate backwards through topological order
    for (let i = topoOrder.length - 1; i >= 0; i--) {
        const u = topoOrder[i];
        const task = taskMap.get(u)!;
        let maxChildCP = 0;
        let minChildED = Infinity;
        const children = adj.get(u);
        if (children) {
            for (const v of children) {
                maxChildCP = Math.max(maxChildCP, criticalPathLength.get(v)!);
                minChildED = Math.min(minChildED, effectiveDeadline.get(v)! - taskMap.get(v)!.duration);
            }
        }
        criticalPathLength.set(u, task.duration + maxChildCP);
        effectiveDeadline.set(u, Math.min(task.deadline, minChildED));
    }

    const latestStart = new Map<string, number>();
    for (const t of tasks) {
        latestStart.set(t.id, effectiveDeadline.get(t.id)! - t.duration);
    }

    // Scheduling Priority: Latest Start (ASC), Smith's Rule (DESC), Critical Path (DESC)
    const readyQueue = new PriorityQueue<string>((a, b) => {
        const lsA = latestStart.get(a)!;
        const lsB = latestStart.get(b)!;
        if (lsA !== lsB) return lsA - lsB;

        const tA = taskMap.get(a)!;
        const tB = taskMap.get(b)!;
        const smithA = tA.weight / tA.duration;
        const smithB = tB.weight / tB.duration;
        if (smithA !== smithB) return smithB - smithA;

        return criticalPathLength.get(b)! - criticalPathLength.get(a)!;
    });

    const initialReady = tasks.filter(t => inDegree.get(t.id) === 0).map(t => t.id);
    for (const id of initialReady) readyQueue.push(id);

    // Simulation Loop
    const freeMachines: number[] = [];
    for (let m = 0; m < machineCount; m++) freeMachines.push(m);

    const finishEvents = new PriorityQueue<{time: number, id: string, machine: number}>((a, b) => a.time - b.time);
    const schedule: ScheduledTask[] = [];
    let scheduledCount = 0;
    let currentTime = 0;

    while (scheduledCount < tasks.length) {
        // 1. Process all task completion events at the current simulation time
        while (finishEvents.size() > 0 && finishEvents.peek()!.time <= currentTime) {
            const f = finishEvents.pop()!;
            freeMachines.push(f.machine);
            const children = adj.get(f.id);
            if (children) {
                for (const v of children) {
                    const newDeg = inDegree.get(v)! - 1;
                    inDegree.set(v, newDeg);
                    if (newDeg === 0) readyQueue.push(v);
                }
            }
        }

        // 2. Greedy assignment: match ready tasks to free machines
        while (freeMachines.length > 0 && readyQueue.size() > 0) {
            const m = freeMachines.pop()!;
            const taskId = readyQueue.pop()!;
            const task = taskMap.get(taskId)!;
            
            schedule.push({ id: taskId, start: currentTime, machine: m });
            finishEvents.push({ time: currentTime + task.duration, id: taskId, machine: m });
            scheduledCount++;
        }

        // 3. Advance the simulation clock to the next event time
        if (scheduledCount < tasks.length) {
            const nextEventTime = finishEvents.peek()?.time;
            if (nextEventTime !== undefined) {
                currentTime = Math.max(currentTime, nextEventTime);
            }
        }
    }

    // Output the resulting schedule as JSON
    process.stdout.write(JSON.stringify({ schedule }));
}

solve();
```
