declare const require: any;
declare const process: any;
const fs = require('fs');

interface Task {
    id: string | number;
    duration: number;
    deps: (string | number)[];
    deadline: number;
    weight: number;
}

interface InputData {
    machineCount: number;
    tasks: Task[];
}

class PriorityQueue<T> {
    private heap: T[] = [];
    constructor(private compare: (a: T, b: T) => number) {}
    push(val: T) {
        this.heap.push(val);
        this.bubbleUp();
    }
    pop(): T | undefined {
        if (this.heap.length === 0) return undefined;
        const top = this.heap[0];
        const bottom = this.heap.pop()!;
        if (this.heap.length > 0) {
            this.heap[0] = bottom;
            this.bubbleDown();
        }
        return top;
    }
    size() { return this.heap.length; }
    private bubbleUp() {
        let idx = this.heap.length - 1;
        while (idx > 0) {
            let parentIdx = (idx - 1) >> 1;
            if (this.compare(this.heap[idx], this.heap[parentIdx]) < 0) {
                let tmp = this.heap[idx];
                this.heap[idx] = this.heap[parentIdx];
                this.heap[parentIdx] = tmp;
                idx = parentIdx;
            } else break;
        }
    }
    private bubbleDown() {
        let idx = 0;
        while (true) {
            let left = (idx << 1) + 1;
            let right = (idx << 1) + 2;
            let smallest = idx;
            if (left < this.heap.length && this.compare(this.heap[left], this.heap[smallest]) < 0) smallest = left;
            if (right < this.heap.length && this.compare(this.heap[right], this.heap[smallest]) < 0) smallest = right;
            if (smallest !== idx) {
                let tmp = this.heap[idx];
                this.heap[idx] = this.heap[smallest];
                this.heap[smallest] = tmp;
                idx = smallest;
            } else break;
        }
    }
}

function solve() {
    let input: string;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data: InputData;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }
    
    const { tasks } = data;
    if (!tasks || tasks.length === 0) {
        process.stdout.write(JSON.stringify({ order: [] }));
        return;
    }

    const idToTask = new Map<string | number, Task>();
    const children = new Map<string | number, (string | number)[]>();
    const parents = new Map<string | number, (string | number)[]>();
    const outDegree = new Map<string | number, number>();
    const forwardInDegree = new Map<string | number, number>();

    for (const task of tasks) {
        idToTask.set(task.id, task);
        children.set(task.id, []);
        parents.set(task.id, []);
        outDegree.set(task.id, 0);
        forwardInDegree.set(task.id, 0);
    }

    for (const task of tasks) {
        const uniqueDeps = new Set(task.deps);
        for (const depId of uniqueDeps) {
            if (!idToTask.has(depId)) continue;
            
            children.get(depId)!.push(task.id);
            parents.get(task.id)!.push(depId);
            
            outDegree.set(depId, (outDegree.get(depId) || 0) + 1);
            forwardInDegree.set(task.id, (forwardInDegree.get(task.id) || 0) + 1);
        }
    }

    // Step 1: Compute Effective Deadlines using reverse topological sort
    const effectiveDeadline = new Map<string | number, number>();
    const revReady: (string | number)[] = [];
    for (const task of tasks) {
        if (outDegree.get(task.id) === 0) {
            revReady.push(task.id);
        }
    }

    let head = 0;
    while (head < revReady.length) {
        const id = revReady[head++];
        const task = idToTask.get(id)!;
        
        let minChildVal = Infinity;
        const taskChildren = children.get(id)!;
        for (const childId of taskChildren) {
            const childTask = idToTask.get(childId)!;
            const val = effectiveDeadline.get(childId)! - childTask.duration;
            if (val < minChildVal) minChildVal = val;
        }
        
        effectiveDeadline.set(id, Math.min(task.deadline, minChildVal));
        
        const taskParents = parents.get(id)!;
        for (const parentId of taskParents) {
            const currentOut = outDegree.get(parentId)! - 1;
            outDegree.set(parentId, currentOut);
            if (currentOut === 0) {
                revReady.push(parentId);
            }
        }
    }

    // Ensure all tasks have an effective deadline (in case of cycles, though not expected)
    for (const task of tasks) {
        if (!effectiveDeadline.has(task.id)) {
            effectiveDeadline.set(task.id, task.deadline);
        }
    }

    // Step 2: Forward topological sort with Priority Queue
    const pq = new PriorityQueue<Task>((a, b) => {
        const da = effectiveDeadline.get(a.id)!;
        const db = effectiveDeadline.get(b.id)!;
        if (da !== db) return da - db;
        if (a.weight !== b.weight) return b.weight - a.weight;
        return a.duration - b.duration;
    });

    for (const task of tasks) {
        if (forwardInDegree.get(task.id) === 0) {
            pq.push(task);
        }
    }

    const order: (string | number)[] = [];
    while (pq.size() > 0) {
        const task = pq.pop()!;
        order.push(task.id);
        
        const taskChildren = children.get(task.id)!;
        for (const childId of taskChildren) {
            const currentIn = forwardInDegree.get(childId)! - 1;
            forwardInDegree.set(childId, currentIn);
            if (currentIn === 0) {
                pq.push(idToTask.get(childId)!);
            }
        }
    }

    // If there's a cycle, we might have tasks left over.
    // But for a valid topological order, we should include them anyway (best effort).
    if (order.length < tasks.length) {
        const seen = new Set(order);
        for (const task of tasks) {
            if (!seen.has(task.id)) {
                order.push(task.id);
            }
        }
    }

    process.stdout.write(JSON.stringify({ order }));
}

solve();
