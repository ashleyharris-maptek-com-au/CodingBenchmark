"use strict";

const fs = require('fs');

/**
 * A simple MinHeap implementation for priority queueing.
 */
class MinHeap {
    constructor(compare) {
        this.heap = [];
        this.compare = compare;
    }
    push(val) {
        this.heap.push(val);
        let idx = this.heap.length - 1;
        while (idx > 0) {
            let parent = (idx - 1) >> 1;
            if (this.compare(this.heap[idx], this.heap[parent]) < 0) {
                const tmp = this.heap[idx];
                this.heap[idx] = this.heap[parent];
                this.heap[parent] = tmp;
                idx = parent;
            } else break;
        }
    }
    pop() {
        if (this.heap.length === 0) return null;
        if (this.heap.length === 1) return this.heap.pop();
        const top = this.heap[0];
        this.heap[0] = this.heap.pop();
        let idx = 0;
        const len = this.heap.length;
        while (true) {
            let left = (idx << 1) + 1;
            let right = (idx << 1) + 2;
            let smallest = idx;
            if (left < len && this.compare(this.heap[left], this.heap[smallest]) < 0) smallest = left;
            if (right < len && this.compare(this.heap[right], this.heap[smallest]) < 0) smallest = right;
            if (smallest !== idx) {
                const tmp = this.heap[idx];
                this.heap[idx] = this.heap[smallest];
                this.heap[smallest] = tmp;
                idx = smallest;
            } else break;
        }
        return top;
    }
    size() { return this.heap.length; }
}

/**
 * Main solution function.
 */
function solve() {
    let input;
    try {
        // Read the entire input from stdin.
        input = fs.readFileSync(0, 'utf8');
    } catch (err) { return; }
    if (!input) return;

    let data;
    try {
        data = JSON.parse(input);
    } catch (err) { return; }

    const tasks = data.tasks;
    if (!tasks || tasks.length === 0) {
        process.stdout.write(JSON.stringify({ order: [] }) + '\n');
        return;
    }

    const taskMap = new Map();
    const adj = new Map();
    const inDegree = new Map();
    const ids = [];

    // Initialize data structures.
    for (let i = 0; i < tasks.length; i++) {
        const task = tasks[i];
        taskMap.set(task.id, task);
        adj.set(task.id, []);
        inDegree.set(task.id, 0);
        ids.push(task.id);
    }

    // Build the dependency graph.
    for (let i = 0; i < tasks.length; i++) {
        const task = tasks[i];
        if (task.deps) {
            for (let j = 0; j < task.deps.length; j++) {
                const depId = task.deps[j];
                if (taskMap.has(depId)) {
                    adj.get(depId).push(task.id);
                    inDegree.set(task.id, inDegree.get(task.id) + 1);
                }
            }
        }
    }

    // Generate a simple topological order using Kahn's algorithm.
    // This order will be used for iterative calculation of rank and effective deadline.
    const topo = [];
    const q = [];
    const tempInDegree = new Map(inDegree);
    for (let i = 0; i < ids.length; i++) {
        if (tempInDegree.get(ids[i]) === 0) q.push(ids[i]);
    }
    let head = 0;
    while (head < q.length) {
        const id = q[head++];
        topo.push(id);
        const successors = adj.get(id);
        for (let i = 0; i < successors.length; i++) {
            const nextId = successors[i];
            const d = tempInDegree.get(nextId) - 1;
            tempInDegree.set(nextId, d);
            if (d === 0) q.push(nextId);
        }
    }

    const rank = new Map();
    const effectiveDeadline = new Map();

    // Compute rank (critical path duration to end) and effective deadline
    // in reverse topological order to handle dependencies and avoid stack overflow.
    for (let i = topo.length - 1; i >= 0; i--) {
        const id = topo[i];
        const task = taskMap.get(id);
        const successors = adj.get(id);
        
        let maxNextRank = 0;
        let minNextED = Infinity;
        
        for (let j = 0; j < successors.length; j++) {
            const nextId = successors[j];
            const nextTask = taskMap.get(nextId);
            
            // Rank is the max duration path from here to any sink.
            const r = rank.get(nextId) || 0;
            if (r > maxNextRank) maxNextRank = r;
            
            // Effective deadline is the latest possible finish time to not miss any downstream deadline.
            const ed = (effectiveDeadline.get(nextId) ?? Infinity) - (nextTask.duration || 0);
            if (ed < minNextED) minNextED = ed;
        }
        
        rank.set(id, (task.duration || 0) + maxNextRank);
        effectiveDeadline.set(id, Math.min(task.deadline ?? Infinity, minNextED));
    }

    // Provide default values for any tasks not covered by topo (e.g. cycles).
    for (let i = 0; i < ids.length; i++) {
        const id = ids[i];
        if (!rank.has(id)) {
            rank.set(id, taskMap.get(id).duration || 0);
            effectiveDeadline.set(id, taskMap.get(id).deadline ?? Infinity);
        }
    }

    /**
     * Comparison function for the priority queue.
     * Prioritizes tasks that are most critical for reducing weighted lateness and makespan.
     */
    const compareTasks = (idA, idB) => {
        // 1. Primary priority: Effective Deadline (earlier first)
        const edA = effectiveDeadline.get(idA);
        const edB = effectiveDeadline.get(idB);
        if (edA < edB) return -1;
        if (edA > edB) return 1;

        // 2. Secondary priority: Weight (higher first)
        const wA = taskMap.get(idA).weight || 0;
        const wB = taskMap.get(idB).weight || 0;
        if (wA > wB) return -1;
        if (wA < wB) return 1;

        // 3. Tertiary priority: Rank (longer path first to keep machines busy)
        const rA = rank.get(idA);
        const rB = rank.get(idB);
        if (rA > rB) return -1;
        if (rA < rB) return 1;

        // 4. Quaternary priority: Duration (shorter first - SPT heuristic)
        const dA = taskMap.get(idA).duration || 0;
        const dB = taskMap.get(idB).duration || 0;
        if (dA < dB) return -1;
        if (dA > dB) return 1;

        // Tie-breaker for deterministic output.
        if (idA < idB) return -1;
        if (idA > idB) return 1;
        return 0;
    };

    // Perform the final topological sort using the heuristic priority queue.
    const heap = new MinHeap(compareTasks);
    for (let i = 0; i < ids.length; i++) {
        if (inDegree.get(ids[i]) === 0) {
            heap.push(ids[i]);
        }
    }

    const order = [];
    while (heap.size() > 0) {
        const id = heap.pop();
        order.push(id);
        const successors = adj.get(id);
        for (let i = 0; i < successors.length; i++) {
            const nextId = successors[i];
            const d = inDegree.get(nextId) - 1;
            inDegree.set(nextId, d);
            if (d === 0) heap.push(nextId);
        }
    }
    
    // If there were cycles, add remaining tasks to ensure a complete order.
    if (order.length < ids.length) {
        const seen = new Set(order);
        for (let i = 0; i < ids.length; i++) {
            if (!seen.has(ids[i])) order.push(ids[i]);
        }
    }

    // Output the resulting topological order as JSON.
    process.stdout.write(JSON.stringify({ order }) + '\n');
}

solve();
