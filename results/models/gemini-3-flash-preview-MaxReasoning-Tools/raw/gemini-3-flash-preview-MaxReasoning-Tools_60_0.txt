"use strict";

const fs = require('fs');

function solve() {
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }
    
    const { tasks } = data;
    if (!tasks || !Array.isArray(tasks)) return;

    const tasksMap = new Map();
    for (let i = 0; i < tasks.length; i++) {
        const t = tasks[i];
        tasksMap.set(t.id, {
            id: t.id,
            duration: t.duration,
            deadline: t.deadline,
            weight: t.weight,
            successors: [],
            inDegree: 0,
            initialInDegree: 0,
            effDeadline: 0
        });
    }

    for (let i = 0; i < tasks.length; i++) {
        const t = tasks[i];
        const currentTask = tasksMap.get(t.id);
        if (t.deps && Array.isArray(t.deps)) {
            for (let j = 0; j < t.deps.length; j++) {
                const depId = t.deps[j];
                const depTask = tasksMap.get(depId);
                if (depTask) {
                    depTask.successors.push(t.id);
                    currentTask.inDegree++;
                    currentTask.initialInDegree++;
                }
            }
        }
    }

    const topoOrder = [];
    const q = [];
    for (const t of tasksMap.values()) {
        if (t.inDegree === 0) {
            q.push(t.id);
        }
    }

    let head = 0;
    while (head < q.length) {
        const id = q[head++];
        topoOrder.push(id);
        const t = tasksMap.get(id);
        for (let i = 0; i < t.successors.length; i++) {
            const succId = t.successors[i];
            const succ = tasksMap.get(succId);
            succ.inDegree--;
            if (succ.inDegree === 0) {
                q.push(succId);
            }
        }
    }

    for (const t of tasksMap.values()) {
        t.inDegree = t.initialInDegree;
    }

    for (let i = topoOrder.length - 1; i >= 0; i--) {
        const id = topoOrder[i];
        const t = tasksMap.get(id);
        let minSuccStart = Infinity;
        for (let j = 0; j < t.successors.length; j++) {
            const succId = t.successors[j];
            const succ = tasksMap.get(succId);
            const succStart = succ.effDeadline - succ.duration;
            if (succStart < minSuccStart) minSuccStart = succStart;
        }
        t.effDeadline = Math.min(t.deadline, minSuccStart);
    }

    class MinHeap {
        constructor(compare) {
            this.heap = [];
            this.compare = compare;
        }
        push(val) {
            this.heap.push(val);
            this.bubbleUp(this.heap.length - 1);
        }
        pop() {
            if (this.heap.length === 0) return null;
            const top = this.heap[0];
            const last = this.heap.pop();
            if (this.heap.length > 0) {
                this.heap[0] = last;
                this.bubbleDown(0);
            }
            return top;
        }
        bubbleUp(idx) {
            while (idx > 0) {
                let pIdx = (idx - 1) >> 1;
                if (this.compare(this.heap[idx], this.heap[pIdx]) < 0) {
                    let tmp = this.heap[idx];
                    this.heap[idx] = this.heap[pIdx];
                    this.heap[pIdx] = tmp;
                    idx = pIdx;
                } else break;
            }
        }
        bubbleDown(idx) {
            const len = this.heap.length;
            while (true) {
                let left = (idx << 1) + 1;
                let right = (idx << 1) + 2;
                let smallest = idx;
                if (left < len && this.compare(this.heap[left], this.heap[smallest]) < 0) smallest = left;
                if (right < len && this.compare(this.heap[right], this.heap[smallest]) < 0) smallest = right;
                if (smallest !== idx) {
                    let tmp = this.heap[idx];
                    this.heap[idx] = this.heap[smallest];
                    this.heap[smallest] = tmp;
                    idx = smallest;
                } else break;
            }
        }
    }

    const compareTasks = (a, b) => {
        if (a.effDeadline !== b.effDeadline) return a.effDeadline - b.effDeadline;
        if (a.weight !== b.weight) return b.weight - a.weight;
        return a.duration - b.duration;
    };

    const readyHeap = new MinHeap(compareTasks);
    for (const t of tasksMap.values()) {
        if (t.inDegree === 0) {
            readyHeap.push(t);
        }
    }

    const order = [];
    while (readyHeap.heap.length > 0) {
        const t = readyHeap.pop();
        order.push(t.id);
        for (let i = 0; i < t.successors.length; i++) {
            const succId = t.successors[i];
            const succ = tasksMap.get(succId);
            succ.inDegree--;
            if (succ.inDegree === 0) {
                readyHeap.push(succ);
            }
        }
    }

    if (order.length < tasks.length) {
        const seen = new Set(order);
        for (let i = 0; i < tasks.length; i++) {
            if (!seen.has(tasks[i].id)) order.push(tasks[i].id);
        }
    }

    process.stdout.write(JSON.stringify({ order }));
}

solve();
