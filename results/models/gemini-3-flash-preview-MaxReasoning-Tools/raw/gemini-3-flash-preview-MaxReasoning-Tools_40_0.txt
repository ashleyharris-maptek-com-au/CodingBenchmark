To solve the 0/1 Integer Linear Program (which is equivalent to the **Maximum Weight Independent Set** problem), we implement an exact solver using a bitset-optimized branch and bound algorithm.

### Key Features of the Implementation:
1.  **Node Pre-processing**: Nodes with non-positive weights are filtered out, as they do not contribute to the maximum objective and only add constraints.
2.  **Heuristic Reordering**: Nodes are sorted by weight (descending) and degree (ascending). This reordering significantly improves the efficiency of the pruning logic by prioritizing "heavy" nodes.
3.  **Bitset Optimization**: A custom `FastBitset` using 64-bit integers and the `ctzll` (count trailing zeros) intrinsic provides high-speed bitwise operations for graph traversal and state management.
4.  **Pruning Logic**: The solver uses an upper bound based on the sum of weights of remaining candidate nodes to prune branches that cannot exceed the current best objective value.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cstdint>

using namespace std;

typedef long long ll;

// Intrinsic for counting trailing zeros (bitscan)
#ifdef _MSC_VER
#include <intrin.h>
#pragma intrinsic(_BitScanForward64)
inline int my_ctzll(unsigned __int64 x) {
    unsigned long index;
    if (_BitScanForward64(&index, x)) return (int)index;
    return 64;
}
#else
inline int my_ctzll(unsigned long long x) {
    return __builtin_ctzll(x);
}
#endif

// Fast bitset implementation for graph operations
struct FastBitset {
    vector<uint64_t> data;
    int n;
    FastBitset(int n = 0) : n(n) {
        data.assign((n + 63) / 64, 0);
    }
    void set(int i) { data[i / 64] |= (1ULL << (i % 64)); }
    void reset(int i) { data[i / 64] &= ~(1ULL << (i % 64)); }
    bool test(int i) const { return (data[i / 64] >> (i % 64)) & 1; }
    bool any() const {
        for (uint64_t x : data) if (x) return true;
        return false;
    }
    int first_set() const {
        for (int i = 0; i < (int)data.size(); ++i) {
            if (data[i]) return i * 64 + my_ctzll(data[i]);
        }
        return -1;
    }
    void and_not(const FastBitset& other) {
        int sz = min((int)data.size(), (int)other.data.size());
        for (int i = 0; i < sz; ++i) data[i] &= ~other.data[i];
    }
};

struct Node {
    int id;
    ll weight;
    int degree;
};

int FN; // Filtered N
vector<ll> W;
vector<FastBitset> G;
ll max_weight = 0;
FastBitset best_assignment(0);

// Branch and Bound recursive solver
void solve(FastBitset P, ll current_w, ll remaining_w, FastBitset& current_set) {
    if (!P.any()) {
        if (current_w > max_weight) {
            max_weight = current_w;
            best_assignment = current_set;
        }
        return;
    }

    // Pruning: if current + potential weight <= best found, skip
    if (current_w + remaining_w <= max_weight) return;

    int v = P.first_set();

    // Branch 1: Include v
    ll weight_conflicts = 0;
    for (int i = 0; i < (int)P.data.size(); ++i) {
        uint64_t mask = P.data[i] & G[v].data[i];
        if (mask) {
            uint64_t val = mask;
            while (val) {
                int bit = my_ctzll(val);
                weight_conflicts += W[i * 64 + bit];
                val &= ~(1ULL << bit);
            }
        }
    }

    FastBitset next_P_with_v = P;
    next_P_with_v.reset(v);
    next_P_with_v.and_not(G[v]); // Neighbors of v cannot be included
    
    current_set.set(v);
    solve(next_P_with_v, current_w + W[v], remaining_w - W[v] - weight_conflicts, current_set);
    current_set.reset(v);

    // Branch 2: Exclude v
    FastBitset next_P_without_v = P;
    next_P_without_v.reset(v);
    solve(next_P_without_v, current_w, remaining_w - W[v], current_set);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int orig_N, orig_M;
    if (!(cin >> orig_N >> orig_M)) return 0;

    vector<ll> orig_weights(orig_N);
    for (int i = 0; i < orig_N; ++i) cin >> orig_weights[i];

    vector<vector<int>> orig_adj(orig_N);
    for (int i = 0; i < orig_M; ++i) {
        int u, v;
        if (!(cin >> u >> v)) break;
        if (u >= 0 && u < orig_N && v >= 0 && v < orig_N) {
            orig_adj[u].push_back(v);
            orig_adj[v].push_back(u);
        }
    }

    // Filter nodes with positive weights and calculate local degrees
    vector<Node> filtered_nodes;
    for (int i = 0; i < orig_N; ++i) {
        if (orig_weights[i] > 0) {
            int deg = 0;
            for (int neighbor : orig_adj[i]) {
                if (orig_weights[neighbor] > 0) deg++;
            }
            filtered_nodes.push_back({i, orig_weights[i], deg});
        }
    }

    // Heuristic: Weight descending, then degree ascending
    sort(filtered_nodes.begin(), filtered_nodes.end(), [](const Node& a, const Node& b) {
        if (a.weight != b.weight) return a.weight > b.weight;
        return a.degree < b.degree;
    });

    FN = filtered_nodes.size();
    W.resize(FN);
    G.assign(FN, FastBitset(FN));
    vector<int> orig_to_new(orig_N, -1);
    for (int i = 0; i < FN; ++i) {
        orig_to_new[filtered_nodes[i].id] = i;
        W[i] = filtered_nodes[i].weight;
    }

    for (int i = 0; i < FN; ++i) {
        for (int neighbor : orig_adj[filtered_nodes[i].id]) {
            int ni = orig_to_new[neighbor];
            if (ni != -1) G[i].set(ni);
        }
    }

    ll total_remaining = 0;
    FastBitset P(FN);
    for (int i = 0; i < FN; ++i) {
        P.set(i);
        total_remaining += W[i];
    }

    max_weight = 0;
    best_assignment = FastBitset(FN);
    FastBitset current_set(FN);
    
    if (FN > 0) {
        solve(P, 0, total_remaining, current_set);
    }

    // Output results
    cout << max_weight << "\n";
    vector<int> x(orig_N, 0);
    for (int i = 0; i < FN; ++i) {
        if (best_assignment.test(i)) {
            x[filtered_nodes[i].id] = 1;
        }
    }

    for (int i = 0; i < orig_N; ++i) {
        cout << x[i] << (i == orig_N - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
```
