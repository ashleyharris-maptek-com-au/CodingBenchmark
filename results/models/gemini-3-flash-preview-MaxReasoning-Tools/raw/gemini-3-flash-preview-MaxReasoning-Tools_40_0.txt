#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

typedef long long ll;

int N_orig;
vector<ll> weights;
vector<int> original_indices;
vector<vector<bool>> is_adj;
ll max_obj = 0;
vector<int> best_x_filtered;
vector<int> current_x_filtered;

/**
 * solve: Recursive backtracking to find Maximum Weight Independent Set.
 * @param candidates: Current set of vertices that can be included.
 * @param start_idx: Current vertex index in candidates to consider.
 * @param current_weight: Total weight of vertices currently in the set.
 * @param remaining_weight: Sum of weights of all vertices in candidates from start_idx to the end.
 */
void solve(const vector<int>& candidates, int start_idx, ll current_weight, ll remaining_weight) {
    // Pruning: if even including all remaining candidates doesn't beat max_obj, return.
    if (current_weight + remaining_weight <= max_obj) return;

    // Base case: all candidates considered.
    if (start_idx == (int)candidates.size()) {
        if (current_weight > max_obj) {
            max_obj = current_weight;
            best_x_filtered = current_x_filtered;
        }
        return;
    }

    int v = candidates[start_idx];

    // Optimization: if v has no neighbors in the remaining candidates, it's always optimal to include it
    // because its weight is positive and it won't block any future potential vertices.
    bool has_neighbors_in_rem = false;
    for (int i = start_idx + 1; i < (int)candidates.size(); ++i) {
        if (is_adj[v][candidates[i]]) {
            has_neighbors_in_rem = true;
            break;
        }
    }

    if (!has_neighbors_in_rem) {
        current_x_filtered[v] = 1;
        solve(candidates, start_idx + 1, current_weight + weights[v], remaining_weight - weights[v]);
        current_x_filtered[v] = 0;
        return;
    }

    // Branch 1: Include v in the Independent Set
    vector<int> next_candidates_in;
    ll next_rem_in = 0;
    for (int i = start_idx + 1; i < (int)candidates.size(); ++i) {
        int u = candidates[i];
        if (!is_adj[v][u]) {
            next_candidates_in.push_back(u);
            next_rem_in += weights[u];
        }
    }
    current_x_filtered[v] = 1;
    // Note: next_candidates_in already filters out neighbors of v.
    solve(next_candidates_in, 0, current_weight + weights[v], next_rem_in);
    current_x_filtered[v] = 0;

    // Branch 2: Exclude v from the Independent Set
    solve(candidates, start_idx + 1, current_weight, remaining_weight - weights[v]);
}

int main() {
    // Fast I/O
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n, m;
    if (!(cin >> n >> m)) return 0;
    N_orig = n;
    vector<ll> c(n);
    for (int i = 0; i < n; ++i) cin >> c[i];

    // Pre-processing:
    // 1. Only consider vertices with c_i > 0.
    // 2. Handle constraints u = v (which implies x_u = 0).
    vector<int> mapping(n, -1);
    vector<ll> filtered_c;
    vector<int> orig_idx;
    for (int i = 0; i < n; ++i) {
        if (c[i] > 0) {
            mapping[i] = (int)filtered_c.size();
            filtered_c.push_back(c[i]);
            orig_idx.push_back(i);
        }
    }

    int n_prime = (int)filtered_c.size();
    weights = filtered_c;
    original_indices = orig_idx;
    is_adj.assign(n_prime, vector<bool>(n_prime, false));

    vector<bool> force_zero(n_prime, false);

    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        if (u < 0 || u >= n || v < 0 || v >= n) continue;
        if (u == v) {
            if (mapping[u] != -1) force_zero[mapping[u]] = true;
            continue;
        }
        int mu = mapping[u];
        int mv = mapping[v];
        if (mu != -1 && mv != -1) {
            is_adj[mu][mv] = is_adj[mv][mu] = true;
        }
    }

    // Initial candidates are those with positive weights not forced to zero.
    vector<int> initial_candidates;
    ll total_rem = 0;
    for (int i = 0; i < n_prime; ++i) {
        if (!force_zero[i]) {
            initial_candidates.push_back(i);
            total_rem += weights[i];
        }
    }

    // Heuristic: Sort candidates by weight descending to find good solutions early.
    sort(initial_candidates.begin(), initial_candidates.end(), [&](int a, int b) {
        return weights[a] > weights[b];
    });

    current_x_filtered.assign(n_prime, 0);
    best_x_filtered.assign(n_prime, 0);
    max_obj = 0; // Objective is at least 0 (empty set)

    if (!initial_candidates.empty()) {
        solve(initial_candidates, 0, 0, total_rem);
    }

    // Output the results
    cout << max_obj << "\n";
    vector<int> final_x(n, 0);
    for (int i = 0; i < n_prime; ++i) {
        if (best_x_filtered[i]) {
            final_x[original_indices[i]] = 1;
        }
    }
    for (int i = 0; i < n; ++i) {
        cout << final_x[i] << (i == n - 1 ? "" : " ");
    }
    cout << endl;

    return 0;
}
