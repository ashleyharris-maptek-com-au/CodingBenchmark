using namespace std;

enum PieceType { EMPTY = 0, PAWN = 1, KNIGHT = 2, BISHOP = 3, ROOK = 4, QUEEN = 5, KING = 6 };
enum Color { WHITE = 8, BLACK = 16 };

struct Move {
    int from, to, piece, captured, promote;
    bool isCastling, isEnPassant;
    int score;
    Move() : from(-1), to(-1), piece(0), captured(0), promote(0), isCastling(false), isEnPassant(false), score(0) {}
};

const int PIECE_VALUES[] = { 0, 100, 320, 330, 500, 900, 20000 };
const int PAWN_PST[64] = { 0,0,0,0,0,0,0,0, 50,50,50,50,50,50,50,50, 10,10,20,30,30,20,10,10, 5,5,10,25,25,10,5,5, 0,0,0,20,20,0,0,0, 5,-5,-10,0,0,-10,-5,5, 5,10,10,-20,-20,10,10,5, 0,0,0,0,0,0,0,0 };
const int KNIGHT_PST[64] = { -50,-40,-30,-30,-30,-30,-40,-50, -40,-20,0,0,0,0,-20,-40, -30,0,10,15,15,10,0,-30, -30,5,15,20,20,15,5,-30, -30,0,15,20,20,15,0,-30, -30,5,10,15,15,10,5,-30, -40,-20,0,5,5,0,-20,-40, -50,-40,-30,-30,-30,-30,-40,-50 };
const int BISHOP_PST[64] = { -20,-10,-10,-10,-10,-10,-10,-20, -10,0,0,0,0,0,0,-10, -10,0,5,10,10,5,0,-10, -10,5,5,10,10,5,5,-10, -10,0,10,10,10,10,0,-10, -10,10,10,10,10,10,10,-10, -10,5,0,0,0,0,5,-10, -20,-10,-10,-10,-10,-10,-10,-20 };
const int ROOK_PST[64] = { 0,0,0,0,0,0,0,0, 5,10,10,10,10,10,10,5, -5,0,0,0,0,0,0,-5, -5,0,0,0,0,0,0,-5, -5,0,0,0,0,0,0,-5, -5,0,0,0,0,0,0,-5, -5,0,0,0,0,0,0,-5, 0,0,0,5,5,0,0,0 };
const int QUEEN_PST[64] = { -20,-10,-10,-5,-5,-10,-10,-20, -10,0,0,0,0,0,0,-10, -10,0,5,5,5,5,0,-10, -5,0,5,5,5,5,0,-5, 0,0,5,5,5,5,0,-5, -10,5,5,5,5,5,0,-10, -10,0,5,0,0,0,0,-10, -20,-10,-10,-5,-5,-10,-10,-20 };
const int KING_PST[64] = { -30,-40,-40,-50,-50,-40,-40,-30, -30,-40,-40,-50,-50,-40,-40,-30, -30,-40,-40,-50,-50,-40,-40,-30, -30,-40,-40,-50,-50,-40,-40,-30, -20,-30,-30,-40,-40,-30,-30,-20, -10,-20,-20,-20,-20,-20,-20,-10, 20,20,0,0,0,0,20,20, 20,30,10,0,0,10,30,20 };

class Board {
public:
    int pieces[64]; int turn, castling, epSquare;
    Board() { for (int i=0; i<64; ++i) pieces[i]=EMPTY; turn=WHITE; castling=0; epSquare=-1; }
    void loadFEN(string fen) {
        stringstream ss(fen); string b_s, t_s, c_s, e_s; ss >> b_s >> t_s >> c_s >> e_s;
        int r=7, c=0; for (char ch : b_s) {
            if (ch=='/') { r--; c=0; } else if (isdigit(ch)) c+=ch-'0';
            else { int clr=isupper(ch)?WHITE:BLACK; int ty; char lch=tolower(ch);
                if (lch=='p') ty=PAWN; else if (lch=='n') ty=KNIGHT; else if (lch=='b') ty=BISHOP;
                else if (lch=='r') ty=ROOK; else if (lch=='q') ty=QUEEN; else ty=KING;
                pieces[r*8+c]=ty|clr; c++; }
        }
        turn=(t_s=="w")?WHITE:BLACK; castling=0;
        if (c_s!="-") for (char ch : c_s) {
            if (ch=='K') castling|=1; else if (ch=='Q') castling|=2; else if (ch=='k') castling|=4; else if (ch=='q') castling|=8;
        }
        if (e_s=="-") epSquare=-1; else epSquare=(e_s[1]-'1')*8+(e_s[0]-'a');
    }
    bool isSquareAttacked(int sq, int attackerColor) const {
        if (sq<0 || sq>=64) return false;
        if (attackerColor==WHITE) { if (sq>=8) { if (sq%8>0 && pieces[sq-9]==(PAWN|WHITE)) return true; if (sq%8<7 && pieces[sq-7]==(PAWN|WHITE)) return true; } }
        else { if (sq<56) { if (sq%8>0 && pieces[sq+7]==(PAWN|BLACK)) return true; if (sq%8<7 && pieces[sq+9]==(PAWN|BLACK)) return true; } }
        static const int kn_st[]={-17,-15,-10,-6,6,10,15,17}; for (int m : kn_st) {
            int nsq=sq+m; if (nsq>=0 && nsq<64 && abs((nsq%8)-(sq%8))<=2 && abs((nsq%8)-(sq%8))>=1) if (pieces[nsq]==(KNIGHT|attackerColor)) return true;
        }
        static const int k_st[]={-9,-8,-7,-1,1,7,8,9}; for (int m : k_st) {
            int nsq=sq+m; if (nsq>=0 && nsq<64 && abs((nsq%8)-(sq%8))<=1) if (pieces[nsq]==(KING|attackerColor)) return true;
        }
        static const int dirs[]={-8,8,-1,1,-9,-7,7,9}; for (int i=0; i<8; ++i) {
            int d=dirs[i]; int nsq=sq+d; while (nsq>=0 && nsq<64 && abs((nsq%8)-((nsq-d)%8))<=1) {
                int p=pieces[nsq]; if (p!=EMPTY) { if ((p&(WHITE|BLACK))==attackerColor) { int ty=p&7; if (i<4 && (ty==ROOK || ty==QUEEN)) return true; if (i>=4 && (ty==BISHOP || ty==QUEEN)) return true; } break; }
                nsq+=d;
            }
        }
        return false;
    }
    bool inCheck() const { int ksq=-1; for (int i=0; i<64; ++i) if (pieces[i]==(KING|turn)) { ksq=i; break; } return isSquareAttacked(ksq, turn==WHITE?BLACK:WHITE); }
    void makeMove(const Move& m, int& oc, int& oe) {
        oc=castling; oe=epSquare; int p=pieces[m.from]; pieces[m.to]=p; pieces[m.from]=EMPTY;
        if (m.promote) pieces[m.to]=m.promote|turn;
        if (m.isEnPassant) { if (turn==WHITE) pieces[m.to-8]=EMPTY; else pieces[m.to+8]=EMPTY; }
        if (m.isCastling) { if (m.to==6) { pieces[5]=pieces[7]; pieces[7]=EMPTY; } else if (m.to==2) { pieces[3]=pieces[0]; pieces[0]=EMPTY; } else if (m.to==62) { pieces[61]=pieces[63]; pieces[63]=EMPTY; } else if (m.to==58) { pieces[59]=pieces[56]; pieces[56]=EMPTY; } }
        if (m.from==0 || m.to==0) castling&=~2; if (m.from==7 || m.to==7) castling&=~1; if (m.from==56 || m.to==56) castling&=~8; if (m.from==63 || m.to==63) castling&=~4;
        if (m.from==4 || p==(KING|WHITE)) castling&=~3; if (m.from==60 || p==(KING|BLACK)) castling&=~12;
        epSquare=-1; if ((p&7)==PAWN && abs(m.to-m.from)==16) epSquare=(m.from+m.to)/2; turn=(turn==WHITE)?BLACK:WHITE;
    }
    void unmakeMove(const Move& m, int oc, int oe) {
        turn=(turn==WHITE)?BLACK:WHITE; int p=pieces[m.to]; if (m.promote) p=PAWN|turn;
        pieces[m.from]=p; pieces[m.to]=m.captured;
        if (m.isEnPassant) { int epClr=(turn==WHITE)?BLACK:WHITE; if (turn==WHITE) pieces[m.to-8]=PAWN|epClr; else pieces[m.to+8]=PAWN|epClr; pieces[m.to]=EMPTY; }
        if (m.isCastling) { if (m.to==6) { pieces[7]=pieces[5]; pieces[5]=EMPTY; } else if (m.to==2) { pieces[0]=pieces[3]; pieces[3]=EMPTY; } else if (m.to==62) { pieces[63]=pieces[61]; pieces[61]=EMPTY; } else if (m.to==58) { pieces[56]=pieces[59]; pieces[59]=EMPTY; } }
        castling=oc; epSquare=oe;
    }
};

struct MoveGen {
    static void generate(Board& b, vector<Move>& moves, bool onlyCaps=false) {
        for (int i=0; i<64; ++i) {
            int p=b.pieces[i]; if (p==EMPTY || (p&(WHITE|BLACK))!=b.turn) continue;
            int ty=p&7; int clr=b.turn;
            if (ty==PAWN) {
                int d=(clr==WHITE)?8:-8; int sR=(clr==WHITE)?1:6; int pR=(clr==WHITE)?7:0; int nsq=i+d;
                if (!onlyCaps && nsq>=0 && nsq<64 && b.pieces[nsq]==EMPTY) {
                    if (nsq/8==pR) { for (int pr:{QUEEN,ROOK,BISHOP,KNIGHT}) { Move m; m.from=i; m.to=nsq; m.piece=p; m.promote=pr; moves.push_back(m); } }
                    else { Move m; m.from=i; m.to=nsq; m.piece=p; moves.push_back(m); if (i/8==sR && b.pieces[nsq+d]==EMPTY) { Move m2; m2.from=i; m2.to=nsq+d; m2.piece=p; moves.push_back(m2); } }
                }
                for (int c:{-1,1}) {
                    int nsq=i+d+c; if (nsq>=0 && nsq<64 && abs((nsq%8)-(i%8))==1) {
                        if (b.pieces[nsq]!=EMPTY && (b.pieces[nsq]&(WHITE|BLACK))!=clr) {
                            if (nsq/8==pR) { for (int pr:{QUEEN,ROOK,BISHOP,KNIGHT}) { Move m; m.from=i; m.to=nsq; m.piece=p; m.captured=b.pieces[nsq]; m.promote=pr; moves.push_back(m); } }
                            else { Move m; m.from=i; m.to=nsq; m.piece=p; m.captured=b.pieces[nsq]; moves.push_back(m); }
                        } else if (nsq==b.epSquare) { Move m; m.from=i; m.to=nsq; m.piece=p; m.captured=PAWN|(clr==WHITE?BLACK:WHITE); m.isEnPassant=true; moves.push_back(m); }
                    }
                }
            } else if (ty==KNIGHT || ty==KING) {
                static const int kn_st[]={-17,-15,-10,-6,6,10,15,17}; static const int k_st[]={-9,-8,-7,-1,1,7,8,9};
                const int* st=(ty==KNIGHT)?kn_st:k_st; for (int j=0; j<8; ++j) {
                    int nsq=i+st[j]; if (nsq>=0 && nsq<64 && abs((nsq%8)-(i%8))<=2) {
                        if (b.pieces[nsq]==EMPTY) { if (!onlyCaps) { Move m; m.from=i; m.to=nsq; m.piece=p; moves.push_back(m); } }
                        else if ((b.pieces[nsq]&(WHITE|BLACK))!=clr) { Move m; m.from=i; m.to=nsq; m.piece=p; m.captured=b.pieces[nsq]; moves.push_back(m); }
                    }
                }
                if (!onlyCaps && ty==KING) {
                    if (clr==WHITE) {
                        if ((b.castling&1) && b.pieces[5]==EMPTY && b.pieces[6]==EMPTY && !b.isSquareAttacked(4,BLACK) && !b.isSquareAttacked(5,BLACK) && !b.isSquareAttacked(6,BLACK)) { Move m; m.from=4; m.to=6; m.piece=p; m.isCastling=true; moves.push_back(m); }
                        if ((b.castling&2) && b.pieces[3]==EMPTY && b.pieces[2]==EMPTY && b.pieces[1]==EMPTY && !b.isSquareAttacked(4,BLACK) && !b.isSquareAttacked(3,BLACK) && !b.isSquareAttacked(2,BLACK)) { Move m; m.from=4; m.to=2; m.piece=p; m.isCastling=true; moves.push_back(m); }
                    } else {
                        if ((b.castling&4) && b.pieces[61]==EMPTY && b.pieces[62]==EMPTY && !b.isSquareAttacked(60,WHITE) && !b.isSquareAttacked(61,WHITE) && !b.isSquareAttacked(62,WHITE)) { Move m; m.from=60; m.to=62; m.piece=p; m.isCastling=true; moves.push_back(m); }
                        if ((b.castling&8) && b.pieces[59]==EMPTY && b.pieces[58]==EMPTY && b.pieces[57]==EMPTY && !b.isSquareAttacked(60,WHITE) && !b.isSquareAttacked(59,WHITE) && !b.isSquareAttacked(58,WHITE)) { Move m; m.from=60; m.to=58; m.piece=p; m.isCastling=true; moves.push_back(m); }
                    }
                }
            } else {
                static const int d_r[]={-8,8,-1,1}; static const int d_b[]={-9,-7,7,9}; static const int d_q[]={-8,8,-1,1,-9,-7,7,9};
                const int* dirs=(ty==ROOK)?d_r:(ty==BISHOP?d_b:d_q); int nD=(ty==ROOK||ty==BISHOP)?4:8; for (int j=0; j<nD; ++j) {
                    int d=dirs[j]; int nsq=i+d; while (nsq>=0 && nsq<64 && abs((nsq%8)-((nsq-d)%8))<=1) {
                        if (b.pieces[nsq]==EMPTY) { if (!onlyCaps) { Move m; m.from=i; m.to=nsq; m.piece=p; moves.push_back(m); } }
                        else { if ((b.pieces[nsq]&(WHITE|BLACK))!=clr) { Move m; m.from=i; m.to=nsq; m.piece=p; m.captured=b.pieces[nsq]; moves.push_back(m); } break; }
                        nsq+=d;
                    }
                }
            }
        }
        vector<Move> legal; for (const auto& m : moves) {
            int oc, oe; int side=b.turn; b.makeMove(m, oc, oe); int ksq=-1; for (int i=0; i<64; ++i) if (b.pieces[i]==(KING|side)) { ksq=i; break; }
            if (!b.isSquareAttacked(ksq, (side==WHITE?BLACK:WHITE))) legal.push_back(m); b.unmakeMove(m, oc, oe);
        }
        moves=legal;
    }
};

int evaluate(const Board& b) {
    int sc=0; for (int i=0; i<64; ++i) {
        int p=b.pieces[i]; if (p==EMPTY) continue; int clr=p&(WHITE|BLACK); int ty=p&7; int val=PIECE_VALUES[ty]; int psq=0; int sq=(clr==WHITE)?(i^56):i;
        if (ty==PAWN) psq=PAWN_PST[sq]; else if (ty==KNIGHT) psq=KNIGHT_PST[sq]; else if (ty==BISHOP) psq=BISHOP_PST[sq]; else if (ty==ROOK) psq=ROOK_PST[sq]; else if (ty==QUEEN) psq=QUEEN_PST[sq]; else psq=KING_PST[sq];
        if (clr==WHITE) sc+=val+psq; else sc-=val+psq;
    }
    return (b.turn==WHITE)?sc:-sc;
}

atomic<bool> stopSearch(false);
int quiesce(Board& b, int alpha, int beta) {
    int stand_pat=evaluate(b); if (stand_pat>=beta) return beta; if (alpha<stand_pat) alpha=stand_pat;
    vector<Move> moves; MoveGen::generate(b, moves, true); for (auto& m : moves) m.score=10*PIECE_VALUES[m.captured&7]-PIECE_VALUES[m.piece&7];
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b){ return a.score>b.score; });
    for (const auto& m : moves) {
        int oc, oe; b.makeMove(m, oc, oe); int sc=-quiesce(b, -beta, -alpha); b.unmakeMove(m, oc, oe);
        if (sc>=beta) return beta; if (sc>alpha) alpha=sc; if (stopSearch) break;
    }
    return alpha;
}
int alphaBeta(Board& b, int alpha, int beta, int depth) {
    if (stopSearch) return 0; if (depth<=0) return quiesce(b, alpha, beta);
    vector<Move> moves; MoveGen::generate(b, moves); if (moves.empty()) { if (b.inCheck()) return -30000+10-depth; return 0; }
    for (auto& m : moves) { m.score=0; if (m.captured) m.score=10*PIECE_VALUES[m.captured&7]-PIECE_VALUES[m.piece&7]; if (m.promote) m.score+=PIECE_VALUES[m.promote]; }
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b){ return a.score>b.score; });
    for (const auto& m : moves) {
        int oc, oe; b.makeMove(m, oc, oe); int sc=-alphaBeta(b, -beta, -alpha, depth-1); b.unmakeMove(m, oc, oe);
        if (sc>=beta) return beta; if (sc>alpha) alpha=sc; if (stopSearch) break;
    }
    return alpha;
}

string moveToString(Board& b, Move m) {
    string s=""; if (m.isCastling) { if (m.to==6 || m.to==62) s="O-O"; else s="O-O-O"; }
    else { char p; int ty=m.piece&7; if (ty==PAWN) p='p'; else if (ty==KNIGHT) p='N'; else if (ty==BISHOP) p='B'; else if (ty==ROOK) p='R'; else if (ty==QUEEN) p='Q'; else p='K';
        s+=p; s+=(char)('a'+(m.from%8)); s+=(char)('1'+(m.from/8)); if (m.captured || m.isEnPassant) s+='x';
        s+=(char)('a'+(m.to%8)); s+=(char)('1'+(m.to/8));
        if (m.promote) { s+='/'; if (m.promote==QUEEN) s+='Q'; else if (m.promote==ROOK) s+='R'; else if (m.promote==BISHOP) s+='B'; else s+='N'; }
        if (m.isEnPassant) s+="ep";
    }
    int oc, oe; b.makeMove(m, oc, oe); if (b.inCheck()) { vector<Move> next; MoveGen::generate(b, next); if (next.empty()) s+='#'; else s+='+'; }
    b.unmakeMove(m, oc, oe); return s;
}

int main() {
    int timeout; string fen; if (!(cin >> timeout)) return 0; getline(cin, fen); if (!getline(cin, fen)) return 0;
    Board b; b.loadFEN(fen); auto start=chrono::steady_clock::now();
    vector<Move> moves; MoveGen::generate(b, moves); if (moves.empty()) return 0;
    Move bestM=moves[0]; thread t([start, timeout](){ while (true) { if (chrono::duration_cast<chrono::seconds>(chrono::steady_clock::now()-start).count()>=timeout-1) { stopSearch=true; break; } this_thread::sleep_for(chrono::milliseconds(50)); } });
    for (int d=1; d<=12; ++d) {
        vector<future<pair<int, Move>>> futs; for (const auto& m : moves) futs.push_back(async(launch::async, [b, m, d]() mutable { int oc, oe; b.makeMove(m, oc, oe); return make_pair(-alphaBeta(b, -1000000, 1000000, d-1), m); }));
        int bestS=-1000000; Move currB=moves[0]; for (auto& f : futs) { auto res=f.get(); if (res.first>bestS) { bestS=res.first; currB=res.second; } }
        if (!stopSearch) bestM=currB; else break;
        for (auto& m : moves) if (m.from==bestM.from && m.to==bestM.to && m.promote==bestM.promote) m.score=10000000; sort(moves.begin(), moves.end(), [](const Move& a, const Move& b){ return a.score>b.score; });
    }
    stopSearch=true; if (t.joinable()) t.join(); cout << moveToString(b, bestM) << endl; return 0;
}
