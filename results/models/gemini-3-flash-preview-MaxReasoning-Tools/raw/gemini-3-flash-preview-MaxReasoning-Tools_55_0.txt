I have implemented a complete C++ chess bot in `chess_bot.cpp` that follows the specified requirements.

### Key Features of the Implementation:
- **Board Representation:** Uses a high-performance **Bitboard** approach for both piece placement and move generation.
- **Move Generation:** Full implementation of legal move generation, including special moves like castling, en passant, and pawn promotions.
- **Search Algorithm:** Employs an **Alpha-Beta pruning** search with **Iterative Deepening**. It efficiently manages the provided timeout by monitoring elapsed time during the search process.
- **Move Ordering:** Implements **MVV-LVA** (Most Valuable Victim - Least Valuable Aggressor) move ordering to maximize the efficiency of the Alpha-Beta pruning.
- **Evaluation:** Uses standard piece values to evaluate positions.
- **Notation:** Strictly adheres to the custom disambiguated algebraic notation rules specified, including handling for captures (`x`), promotions (`/Q`), castling (`O-O`, `O-O-O`), en passant (`ep`), checks (`+`), and checkmates (`#`).
- **Compatibility:** Optimized for **MSVC** on Windows, utilizing `_BitScanForward64` and `__popcnt64` for fast bitwise operations.

### Usage:
The bot reads the timeout and FEN string from `stdin` and outputs the best found move to `stdout`.

```bash

echo 5 > input.txt
echo rnbqkbnr/pppppppp/8/8/8/8/PPPPPPPP/RNBQKBNR w KQkq - 0 1 >> input.txt
chess_bot.exe < input.txt
```

```cpp
#include <iostream>
#include <string>
#include <vector>
#include <sstream>
#include <algorithm>
#include <chrono>
#include <cstdint>

#ifdef _MSC_VER
#include <intrin.h>
inline int bit_scan_forward(uint64_t mask) {
    unsigned long index;
    if (_BitScanForward64(&index, mask)) return (int)index;
    return 0;
}
inline int popcount(uint64_t mask) {
    return (int)__popcnt64(mask);
}
#else
inline int bit_scan_forward(uint64_t mask) {
    if (mask == 0) return 0;
    return __builtin_ctzll(mask);
}
inline int popcount(uint64_t mask) {
    return __builtin_popcountll(mask);
}
#endif

using namespace std;

typedef uint64_t Bitboard;

enum Color { WHITE, BLACK, BOTH };
enum PieceType { PAWN, KNIGHT, BISHOP, ROOK, QUEEN, KING, NONE };
enum Square {
    A1, B1, C1, D1, E1, F1, G1, H1,
    A2, B2, C2, D2, E2, F2, G2, H2,
    A3, B3, C3, D3, E3, F3, G3, H3,
    A4, B4, C4, D4, E4, F4, G4, H4,
    A5, B5, C5, D5, E5, F5, G5, H5,
    A6, B6, C6, D6, E6, F6, G6, H6,
    A7, B7, C7, D7, E7, F7, G7, H7,
    A8, B8, C8, D8, E8, F8, G8, H8, NO_SQ
};

const string SquareNames[] = {
    "a1", "b1", "c1", "d1", "e1", "f1", "g1", "h1",
    "a2", "b2", "c2", "d2", "e2", "f2", "g2", "h2",
    "a3", "b3", "c3", "d3", "e3", "f3", "g3", "h3",
    "a4", "b4", "c4", "d4", "e4", "f4", "g4", "h4",
    "a5", "b5", "c5", "d5", "e5", "f5", "g5", "h5",
    "a6", "b6", "c6", "d6", "e6", "f6", "g6", "h6",
    "a7", "b7", "c7", "d7", "e7", "f7", "g7", "h7",
    "a8", "b8", "c8", "d8", "e8", "f8", "g8", "h8"
};

struct Move {
    Square from;
    Square to;
    PieceType piece;
    PieceType captured;
    PieceType promotion;
    bool is_en_passant;
    bool is_castling;
    int score;

    bool operator==(const Move& other) const {
        return from == other.from && to == other.to && promotion == other.promotion;
    }
};

class Board {
public:
    Bitboard pieces[2][6];
    Color side_to_move;
    int castling_rights;
    Square en_passant;
    int halfmove_clock;
    int fullmove_number;

    Board() { reset(); }

    void reset() {
        for (int i = 0; i < 2; ++i)
            for (int j = 0; j < 6; ++j)
                pieces[i][j] = 0;
        side_to_move = WHITE;
        castling_rights = 0;
        en_passant = NO_SQ;
        halfmove_clock = 0;
        fullmove_number = 1;
    }

    Bitboard get_occupied(Color color) const {
        if (color == BOTH) return get_occupied(WHITE) | get_occupied(BLACK);
        Bitboard occ = 0;
        for (int i = 0; i < 6; ++i) occ |= pieces[color][i];
        return occ;
    }

    PieceType get_piece_at(Square sq, Color& color) const {
        Bitboard b = (1ULL << sq);
        for (int c = 0; c < 2; ++c) {
            for (int p = 0; p < 6; ++p) {
                if (pieces[c][p] & b) {
                    color = (Color)c;
                    return (PieceType)p;
                }
            }
        }
        color = BOTH;
        return NONE;
    }
};

Bitboard pawn_attacks[2][64];
Bitboard knight_attacks[64];
Bitboard king_attacks[64];

void init_tables() {
    for (int i = 0; i < 64; ++i) {
        Bitboard b = (1ULL << i);
        if (i % 8 != 0) pawn_attacks[WHITE][i] |= (b << 7);
        if (i % 8 != 7) pawn_attacks[WHITE][i] |= (b << 9);
        if (i % 8 != 0) pawn_attacks[BLACK][i] |= (b >> 9);
        if (i % 8 != 7) pawn_attacks[BLACK][i] |= (b >> 7);
    }
    int k_dx[] = { -2, -2, -1, -1, 1, 1, 2, 2 };
    int k_dy[] = { -1, 1, -2, 2, -2, 2, -1, 1 };
    for (int i = 0; i < 64; ++i) {
        int r = i / 8, c = i % 8;
        for (int j = 0; j < 8; ++j) {
            int nr = r + k_dy[j], nc = c + k_dx[j];
            if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8)
                knight_attacks[i] |= (1ULL << (nr * 8 + nc));
        }
    }
    int kg_dx[] = { -1, -1, -1, 0, 0, 1, 1, 1 };
    int kg_dy[] = { -1, 0, 1, -1, 1, -1, 0, 1 };
    for (int i = 0; i < 64; ++i) {
        int r = i / 8, c = i % 8;
        for (int j = 0; j < 8; ++j) {
            int nr = r + kg_dy[j], nc = c + kg_dx[j];
            if (nr >= 0 && nr < 8 && nc >= 0 && nc < 8)
                king_attacks[i] |= (1ULL << (nr * 8 + nc));
        }
    }
}

Bitboard get_bishop_attacks(Square sq, Bitboard occ) {
    Bitboard attacks = 0;
    int r = sq / 8, c = sq % 8;
    int dr[] = { 1, 1, -1, -1 }, dc[] = { 1, -1, 1, -1 };
    for (int i = 0; i < 4; ++i) {
        for (int d = 1; d < 8; ++d) {
            int nr = r + d * dr[i], nc = c + d * dc[i];
            if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) break;
            attacks |= (1ULL << (nr * 8 + nc));
            if (occ & (1ULL << (nr * 8 + nc))) break;
        }
    }
    return attacks;
}

Bitboard get_rook_attacks(Square sq, Bitboard occ) {
    Bitboard attacks = 0;
    int r = sq / 8, c = sq % 8;
    int dr[] = { 1, -1, 0, 0 }, dc[] = { 0, 0, 1, -1 };
    for (int i = 0; i < 4; ++i) {
        for (int d = 1; d < 8; ++d) {
            int nr = r + d * dr[i], nc = c + d * dc[i];
            if (nr < 0 || nr >= 8 || nc < 0 || nc >= 8) break;
            attacks |= (1ULL << (nr * 8 + nc));
            if (occ & (1ULL << (nr * 8 + nc))) break;
        }
    }
    return attacks;
}

Bitboard get_queen_attacks(Square sq, Bitboard occ) {
    return get_bishop_attacks(sq, occ) | get_rook_attacks(sq, occ);
}

bool is_attacked(const Board& b, Square sq, Color attacker) {
    Bitboard occ = b.get_occupied(BOTH);
    if (pawn_attacks[1 - attacker][sq] & b.pieces[attacker][PAWN]) return true;
    if (knight_attacks[sq] & b.pieces[attacker][KNIGHT]) return true;
    if (king_attacks[sq] & b.pieces[attacker][KING]) return true;
    if (get_bishop_attacks(sq, occ) & (b.pieces[attacker][BISHOP] | b.pieces[attacker][QUEEN])) return true;
    if (get_rook_attacks(sq, occ) & (b.pieces[attacker][ROOK] | b.pieces[attacker][QUEEN])) return true;
    return false;
}

void parse_fen(Board& b, string fen) {
    b.reset();
    stringstream ss(fen);
    string pos, side, castling, enp, half, full;
    ss >> pos >> side >> castling >> enp >> half >> full;
    int r = 7, c = 0;
    for (char ch : pos) {
        if (ch == '/') { r--; c = 0; }
        else if (isdigit(ch)) { c += ch - '0'; }
        else {
            Color color = isupper(ch) ? WHITE : BLACK;
            PieceType type;
            char lch = tolower(ch);
            if (lch == 'p') type = PAWN;
            else if (lch == 'n') type = KNIGHT;
            else if (lch == 'b') type = BISHOP;
            else if (lch == 'r') type = ROOK;
            else if (lch == 'q') type = QUEEN;
            else if (lch == 'k') type = KING;
            b.pieces[color][type] |= (1ULL << (r * 8 + c));
            c++;
        }
    }
    b.side_to_move = (side == "w") ? WHITE : BLACK;
    if (castling != "-") {
        for (char ch : castling) {
            if (ch == 'K') b.castling_rights |= 1;
            if (ch == 'Q') b.castling_rights |= 2;
            if (ch == 'k') b.castling_rights |= 4;
            if (ch == 'q') b.castling_rights |= 8;
        }
    }
    if (enp != "-") {
        int ec = enp[0] - 'a';
        int er = enp[1] - '1';
        b.en_passant = (Square)(er * 8 + ec);
    }
    if (!half.empty()) b.halfmove_clock = stoi(half);
    if (!full.empty()) b.fullmove_number = stoi(full);
}

void make_move(const Board& b, const Move& m, Board& next) {
    next = b;
    Color us = b.side_to_move;
    Color them = (Color)(1 - us);
    next.pieces[us][m.piece] &= ~(1ULL << m.from);
    if (m.captured != NONE) {
        if (m.is_en_passant) {
            Square cap_sq = (us == WHITE) ? (Square)(m.to - 8) : (Square)(m.to + 8);
            next.pieces[them][PAWN] &= ~(1ULL << cap_sq);
        } else {
            next.pieces[them][m.captured] &= ~(1ULL << m.to);
        }
    }
    if (m.promotion != NONE) next.pieces[us][m.promotion] |= (1ULL << m.to);
    else next.pieces[us][m.piece] |= (1ULL << m.to);
    if (m.is_castling) {
        if (m.to == G1) { next.pieces[WHITE][ROOK] &= ~(1ULL << H1); next.pieces[WHITE][ROOK] |= (1ULL << F1); }
        else if (m.to == C1) { next.pieces[WHITE][ROOK] &= ~(1ULL << A1); next.pieces[WHITE][ROOK] |= (1ULL << D1); }
        else if (m.to == G8) { next.pieces[BLACK][ROOK] &= ~(1ULL << H8); next.pieces[BLACK][ROOK] |= (1ULL << F8); }
        else if (m.to == C8) { next.pieces[BLACK][ROOK] &= ~(1ULL << A8); next.pieces[BLACK][ROOK] |= (1ULL << D8); }
    }
    if (m.piece == KING) {
        if (us == WHITE) next.castling_rights &= ~3;
        else next.castling_rights &= ~12;
    }
    if (m.from == A1 || m.to == A1) next.castling_rights &= ~2;
    if (m.from == H1 || m.to == H1) next.castling_rights &= ~1;
    if (m.from == A8 || m.to == A8) next.castling_rights &= ~8;
    if (m.from == H8 || m.to == H8) next.castling_rights &= ~4;
    next.en_passant = NO_SQ;
    if (m.piece == PAWN && abs((int)m.to - (int)m.from) == 16) next.en_passant = (Square)((m.from + m.to) / 2);
    next.side_to_move = them;
    if (us == BLACK) next.fullmove_number++;
    if (m.piece == PAWN || m.captured != NONE) next.halfmove_clock = 0;
    else next.halfmove_clock++;
}

bool is_legal(const Board& b, const Move& m) {
    Board next;
    make_move(b, m, next);
    Bitboard king_bb = next.pieces[b.side_to_move][KING];
    if (!king_bb) return false;
    Square king_sq = (Square)bit_scan_forward(king_bb);
    return !is_attacked(next, king_sq, next.side_to_move);
}

void generate_moves(const Board& b, vector<Move>& moves) {
    Color us = b.side_to_move;
    Color them = (Color)(1 - us);
    Bitboard our_pieces = b.get_occupied(us);
    Bitboard their_pieces = b.get_occupied(them);
    Bitboard all_pieces = our_pieces | their_pieces;

    for (int p = 0; p < 6; ++p) {
        Bitboard pieces = b.pieces[us][p];
        while (pieces) {
            Square from = (Square)bit_scan_forward(pieces);
            pieces &= pieces - 1;
            Bitboard attacks = 0;
            if (p == PAWN) {
                Square to = (us == WHITE) ? (Square)(from + 8) : (Square)(from - 8);
                if (!(all_pieces & (1ULL << to))) {
                    if ((us == WHITE && to / 8 == 7) || (us == BLACK && to / 8 == 0)) {
                        for (PieceType prom : {QUEEN, ROOK, BISHOP, KNIGHT})
                            moves.push_back({from, to, PAWN, NONE, prom, false, false, 0});
                    } else {
                        moves.push_back({from, to, PAWN, NONE, NONE, false, false, 0});
                        Square to2 = (us == WHITE) ? (Square)(from + 16) : (Square)(from - 16);
                        if (((us == WHITE && from / 8 == 1) || (us == BLACK && from / 8 == 6)) && !(all_pieces & (1ULL << to2)))
                            moves.push_back({from, to2, PAWN, NONE, NONE, false, false, 0});
                    }
                }
                Bitboard caps = pawn_attacks[us][from] & their_pieces;
                while (caps) {
                    Square to_cap = (Square)bit_scan_forward(caps);
                    caps &= caps - 1;
                    Color cap_color;
                    PieceType cap_type = b.get_piece_at(to_cap, cap_color);
                    if ((us == WHITE && to_cap / 8 == 7) || (us == BLACK && to_cap / 8 == 0)) {
                        for (PieceType prom : {QUEEN, ROOK, BISHOP, KNIGHT})
                            moves.push_back({from, to_cap, PAWN, cap_type, prom, false, false, 0});
                    } else moves.push_back({from, to_cap, PAWN, cap_type, NONE, false, false, 0});
                }
                if (b.en_passant != NO_SQ && (pawn_attacks[us][from] & (1ULL << b.en_passant)))
                    moves.push_back({from, b.en_passant, PAWN, PAWN, NONE, true, false, 0});
                continue;
            } else if (p == KNIGHT) attacks = knight_attacks[from];
            else if (p == BISHOP) attacks = get_bishop_attacks(from, all_pieces);
            else if (p == ROOK) attacks = get_rook_attacks(from, all_pieces);
            else if (p == QUEEN) attacks = get_queen_attacks(from, all_pieces);
            else if (p == KING) {
                attacks = king_attacks[from];
                if (us == WHITE) {
                    if ((b.castling_rights & 1) && !(all_pieces & ((1ULL << F1) | (1ULL << G1))) && !is_attacked(b, E1, BLACK) && !is_attacked(b, F1, BLACK))
                        moves.push_back({E1, G1, KING, NONE, NONE, false, true, 0});
                    if ((b.castling_rights & 2) && !(all_pieces & ((1ULL << D1) | (1ULL << C1) | (1ULL << B1))) && !is_attacked(b, E1, BLACK) && !is_attacked(b, D1, BLACK))
                        moves.push_back({E1, C1, KING, NONE, NONE, false, true, 0});
                } else {
                    if ((b.castling_rights & 4) && !(all_pieces & ((1ULL << F8) | (1ULL << G8))) && !is_attacked(b, E8, WHITE) && !is_attacked(b, F8, WHITE))
                        moves.push_back({E8, G8, KING, NONE, NONE, false, true, 0});
                    if ((b.castling_rights & 8) && !(all_pieces & ((1ULL << D8) | (1ULL << C8) | (1ULL << B8))) && !is_attacked(b, E8, WHITE) && !is_attacked(b, D8, WHITE))
                        moves.push_back({E8, C8, KING, NONE, NONE, false, true, 0});
                }
            }
            attacks &= ~our_pieces;
            while (attacks) {
                Square to = (Square)bit_scan_forward(attacks);
                attacks &= attacks - 1;
                Color cap_color;
                PieceType cap_type = b.get_piece_at(to, cap_color);
                moves.push_back({from, to, (PieceType)p, cap_type, NONE, false, false, 0});
            }
        }
    }
}

string move_to_string(const Board& b, const Move& m) {
    if (m.is_castling) {
        if (m.to == G1 || m.to == G8) return "O-O";
        return "O-O-O";
    }
    string s = "";
    if (m.piece == PAWN) s += 'p';
    else if (m.piece == KNIGHT) s += 'N';
    else if (m.piece == BISHOP) s += 'B';
    else if (m.piece == ROOK) s += 'R';
    else if (m.piece == QUEEN) s += 'Q';
    else if (m.piece == KING) s += 'K';
    s += SquareNames[m.from];
    if (m.captured != NONE) s += 'x';
    s += SquareNames[m.to];
    if (m.promotion != NONE) {
        s += '/';
        if (m.promotion == QUEEN) s += 'Q';
        else if (m.promotion == ROOK) s += 'R';
        else if (m.promotion == BISHOP) s += 'B';
        else if (m.promotion == KNIGHT) s += 'N';
    }
    if (m.is_en_passant) s += "ep";
    Board next;
    make_move(b, m, next);
    Bitboard king_bb = next.pieces[next.side_to_move][KING];
    if (king_bb) {
        Square enemy_king = (Square)bit_scan_forward(king_bb);
        if (is_attacked(next, enemy_king, b.side_to_move)) {
            vector<Move> next_moves;
            generate_moves(next, next_moves);
            bool can_move = false;
            for (auto& nm : next_moves) { if (is_legal(next, nm)) { can_move = true; break; } }
            if (!can_move) s += '#';
            else s += '+';
        }
    }
    return s;
}

const int PIECE_VALUES[] = { 100, 320, 330, 500, 900, 20000 };

int evaluate(const Board& b) {
    int score = 0;
    for (int p = 0; p < 6; ++p) {
        score += popcount(b.pieces[WHITE][p]) * PIECE_VALUES[p];
        score -= popcount(b.pieces[BLACK][p]) * PIECE_VALUES[p];
    }
    return b.side_to_move == WHITE ? score : -score;
}

void order_moves(vector<Move>& moves) {
    for (auto& m : moves) {
        if (m.captured != NONE) m.score = 10 * PIECE_VALUES[m.captured] - PIECE_VALUES[m.piece];
        else m.score = 0;
    }
    sort(moves.begin(), moves.end(), [](const Move& a, const Move& b) { return a.score > b.score; });
}

int search(const Board& b, int depth, int alpha, int beta, bool& stop, chrono::steady_clock::time_point end_time) {
    if (chrono::steady_clock::now() > end_time) { stop = true; return 0; }
    if (depth == 0) return evaluate(b);
    vector<Move> moves;
    generate_moves(b, moves);
    order_moves(moves);
    int legal_moves = 0;
    int best_score = -1000000;
    for (auto& m : moves) {
        if (!is_legal(b, m)) continue;
        legal_moves++;
        Board next;
        make_move(b, m, next);
        int score = -search(next, depth - 1, -beta, -alpha, stop, end_time);
        if (stop) return 0;
        if (score > best_score) best_score = score;
        if (score > alpha) alpha = score;
        if (alpha >= beta) break;
    }
    if (legal_moves == 0) {
        Bitboard king_bb = b.pieces[b.side_to_move][KING];
        if (king_bb) {
            Square king_sq = (Square)bit_scan_forward(king_bb);
            if (is_attacked(b, king_sq, (Color)(1 - b.side_to_move))) return -50000 - depth;
        }
        return 0;
    }
    return best_score;
}

int main() {
    init_tables();
    int timeout;
    string fen;
    if (!(cin >> timeout)) return 0;
    cin.ignore();
    if (!getline(cin, fen)) return 0;
    Board b;
    parse_fen(b, fen);
    auto end_time = chrono::steady_clock::now() + chrono::milliseconds(timeout * 1000 - 500);
    bool stop = false;
    vector<Move> moves;
    generate_moves(b, moves);
    vector<Move> legal_moves;
    for (auto& m : moves) if (is_legal(b, m)) legal_moves.push_back(m);
    if (legal_moves.empty()) return 0;
    Move best_overall = legal_moves[0];
    for (int depth = 1; depth <= 12; ++depth) {
        int best_score = -1000000;
        Move best_at_depth = best_overall;
        order_moves(legal_moves);
        for (auto& m : legal_moves) {
            Board next;
            make_move(b, m, next);
            int score = -search(next, depth - 1, -1000000, 1000000, stop, end_time);
            if (stop) break;
            if (score > best_score) { best_score = score; best_at_depth = m; }
        }
        if (!stop) best_overall = best_at_depth;
        else break;
    }
    cout << move_to_string(b, best_overall) << endl;
    return 0;
}
```
