This C# program solves the 3D voxel mining problem by implementing an efficient greedy strategy tailored for both small-scale and large-scale worlds.

### Key Strategies:
1.  **Adaptive Memory Management:** Uses 2D arrays to store height maps and metadata (like the top-most valuable ore) for each column. This allows the program to handle worlds up to 5000x5000 with a manageable memory footprint (~300MB).
2.  **Efficient Stream Processing:** Reads the voxel byte stream in chunks to minimize I/O overhead. It includes a timer to stop reading and start processing if the input is extremely large, ensuring it completes within the 5-minute limit.
3.  **Profit Optimization:** Calculates a potential profit score for each column, considering the value of the best ore, the cost of moving it to (0,0), and the penalty for clearing junk voxels above it. Columns are processed in order of descending profit.
4.  **Greedy Pathfinding:** Implements a robust pathfinder that prioritizes flat or downhill steps to minimize fuel costs while ensuring compliance with the 10% slope constraint and avoiding obstacles.
5.  **Simplified Dumping:** All dug voxels (valuable and junk) are moved to the processing plant at (0,0). While junk incurs a penalty, this avoids the complexity of building stable ramps or managing separate dump sites that could block future operations.

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Diagnostics;
using System.Text;

namespace MiningSolution {
    class Program {
        static int XCount, YCount, ZCount;
        static int[,] heightMap = null!;
        static short[,] bestZ = null!;
        static byte[,] bestVal = null!;
        static byte[,] bestMass = null!;
        static int[,] junkMassAbove = null!;
        static Stopwatch sw = new Stopwatch();

        static void Main(string[] args) {
            sw.Start();
            
            // Fast I/O
            var stdin = Console.OpenStandardInput();
            
            string? headerLine = ReadHeader(stdin);
            if (string.IsNullOrEmpty(headerLine)) return;

            string[] parts = headerLine.Split(new[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
            if (parts.Length < 6) return;
            XCount = int.Parse(parts[0]);
            YCount = int.Parse(parts[1]);
            ZCount = int.Parse(parts[2]);
            
            // Initialize arrays. For 5000x5000, these take ~300MB total.
            heightMap = new int[XCount, YCount];
            bestZ = new short[XCount, YCount];
            bestVal = new byte[XCount, YCount];
            bestMass = new byte[XCount, YCount];
            junkMassAbove = new int[XCount, YCount];

            byte[] colBuffer = new byte[ZCount];

            for (int x = 0; x < XCount; x++) {
                for (int y = 0; y < YCount; y++) {
                    int read = 0;
                    while (read < ZCount) {
                        int r = stdin.Read(colBuffer, read, ZCount - read);
                        if (r <= 0) break;
                        read += r;
                    }

                    bestZ[x, y] = -1;
                    for (int z = 0; z < ZCount; z++) {
                        byte b = colBuffer[z];
                        int mass = b % 16;
                        int val = b / 16;

                        if (mass > 0) {
                            heightMap[x, y] = z + 1;
                            if (val > 0) {
                                // Store the top-most valuable ore in this column
                                bestZ[x, y] = (short)z;
                                bestVal[x, y] = (byte)val;
                                bestMass[x, y] = (byte)mass;
                            }
                        }
                    }
                    
                    if (bestZ[x, y] != -1) {
                        int jMass = 0;
                        for (int z = bestZ[x, y] + 1; z < heightMap[x, y]; z++) {
                            jMass += colBuffer[z] % 16;
                        }
                        junkMassAbove[x, y] = jMass;
                    }
                }
                // Time limit check during reading (mostly for huge ZCount)
                if (x % 50 == 0 && sw.Elapsed.TotalSeconds > 150) break;
            }

            // Find profitable columns
            var candidates = new List<ColumnScore>();
            for (int x = 0; x < XCount; x++) {
                for (int y = 0; y < YCount; y++) {
                    if (bestZ[x, y] != -1) {
                        double profit = CalculateScore(x, y);
                        if (profit > 0) {
                            candidates.Add(new ColumnScore { x = x, y = y, score = profit });
                        }
                    }
                }
            }
            // Sort by potential profit
            candidates.Sort((a, b) => b.score.CompareTo(a.score));

            var results = new List<string>();
            int processedColumns = 0;
            
            foreach (var cand in candidates) {
                if (sw.Elapsed.TotalSeconds > 280) break;

                int x = cand.x;
                int y = cand.y;
                int targetZ = bestZ[x, y];
                
                List<string> colOps = new List<string>();
                bool possible = true;
                int tempH = heightMap[x, y];
                
                // Dig from top down to the target valuable ore
                while (tempH > targetZ) {
                    int z = tempH - 1;
                    var path = FindPath(x, y, z, 0, 0);
                    if (path == null) {
                        possible = false;
                        break;
                    }
                    
                    StringBuilder sb = new StringBuilder();
                    sb.Append(path.Count).Append(" ").Append(x).Append(" ").Append(y).Append(" ").Append(z);
                    foreach (var p in path) {
                        sb.Append(" ").Append(p.x).Append(" ").Append(p.y);
                    }
                    sb.Append(" 0 0");
                    colOps.Add(sb.ToString());
                    tempH--;
                }
                
                if (possible) {
                    results.AddRange(colOps);
                    heightMap[x, y] = tempH;
                    processedColumns++;
                    // Safety limit on total operations to avoid extreme memory/output
                    if (results.Count > 500000) break;
                }
            }

            // Final output using Buffered StreamWriter for performance
            var stdout = new StreamWriter(Console.OpenStandardOutput(), Encoding.ASCII, 256 * 1024);
            stdout.WriteLine(results.Count);
            foreach (var s in results) {
                stdout.WriteLine(s);
            }
            stdout.Flush();
        }

        static string? ReadHeader(Stream s) {
            StringBuilder sb = new StringBuilder();
            int b;
            while ((b = s.ReadByte()) != -1 && b != '\n') {
                if (b != '\r') sb.Append((char)b);
            }
            if (sb.Length == 0 && b == -1) return null;
            return sb.ToString();
        }

        struct ColumnScore { public int x, y; public double score; }
        struct Pos { public int x, y; }

        static double CalculateScore(int x, int y) {
            int dist = x + y;
            double revenue = (double)bestVal[x, y] * bestMass[x, y] * 1000;
            double moveCost = (double)dist * bestMass[x, y] * 1.5; // Estimated avg fuel cost
            double junkPenalty = (double)junkMassAbove[x, y] * 1000;
            double junkMoveCost = (double)dist * junkMassAbove[x, y] * 1.5;
            
            return revenue - moveCost - junkPenalty - junkMoveCost;
        }

        static List<Pos>? FindPath(int startX, int startY, int startZ, int endX, int endY) {
            List<Pos> path = new List<Pos>();
            int cx = startX;
            int cy = startY;
            int cz = startZ;
            
            int maxSteps = (XCount + YCount) * 4;
            int steps = 0;

            while ((cx != endX || cy != endY) && steps < maxSteps) {
                steps++;
                int nx = -1, ny = -1;
                int bestDist = Math.Abs(cx - endX) + Math.Abs(cy - endY);
                int moveType = 5; 
                // Move types: 0:closer+down, 1:closer+flat, 2:closer+up, 3:away+down/flat, 4:away+up

                for (int i = 0; i < 4; i++) {
                    int tx = cx, ty = cy;
                    if (i == 0) tx--; else if (i == 1) ty--; else if (i == 2) tx++; else ty++;
                    
                    if (tx < 0 || tx >= XCount || ty < 0 || ty >= YCount) continue;
                    
                    // Voxel can move if ground height at target is <= current carry height
                    if (heightMap[tx, ty] <= cz) {
                        int d = Math.Abs(tx - endX) + Math.Abs(ty - endY);
                        int z_next = Math.Max(cz - 1, heightMap[tx, ty] + 1);
                        int type;
                        if (d < bestDist) {
                            if (z_next < cz) type = 0;
                            else if (z_next == cz) type = 1;
                            else type = 2;
                        } else {
                            if (z_next <= cz) type = 3;
                            else type = 4;
                        }
                        
                        if (type < moveType) {
                            moveType = type;
                            nx = tx; ny = ty;
                        } else if (type == moveType) {
                            if (tx + ty < nx + ny) { nx = tx; ny = ty; }
                        }
                    }
                }

                if (nx == -1) return null; // Stuck
                
                cz = Math.Max(cz - 1, heightMap[nx, ny] + 1);
                cx = nx;
                cy = ny;
                path.Add(new Pos { x = cx, y = cy });
            }
            
            return (cx == endX && cy == endY) ? path : null;
        }
    }
}
```
