"use strict";

const fs = require('fs');

/**
 * Benchmark Task: Wildfire Firebreaks
 * 
 * Strategy: Greedy importance scoring.
 * For each iteration:
 * 1. Run BFS from all ignitions to find shortest distances to all reachable forest cells.
 * 2. If no towns are reachable, we are done.
 * 3. Propagate town values backwards from reachable towns to ignitions.
 *    A cell's score represents the "flow" of town value passing through it on shortest paths.
 *    score(C) = (townValue(C) + sum(score(S) / num_preds(S))) 
 *    where S are successors of C in the BFS shortest-path DAG.
 * 4. Pick the forest cell with the highest score that is not an ignition.
 * 5. Update the board (cut the cell) and repeat until budget is exhausted.
 */

function solve() {
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const { width, height, grid, ignitions, towns, cutBudget } = data;
    const size = width * height;

    // board: 0 for forest, 1 for blocked (water, rock, or firebreak)
    const board = new Uint8Array(size);
    const townValue = new Float64Array(size);
    const isIgnition = new Uint8Array(size);
    const isTownCell = new Uint8Array(size);

    for (let y = 0; y < height; y++) {
        const row = grid[y];
        for (let x = 0; x < width; x++) {
            const idx = y * width + x;
            board[idx] = row[x] === '.' ? 0 : 1;
        }
    }

    for (const town of towns) {
        const idx = town.y * width + town.x;
        townValue[idx] += town.value;
        isTownCell[idx] = 1;
    }

    for (const [ix, iy] of ignitions) {
        isIgnition[iy * width + ix] = 1;
    }

    const firebreaks = [];
    let budget = cutBudget;

    const dx = [0, 0, 1, -1];
    const dy = [1, -1, 0, 0];

    const distFire = new Int32Array(size);
    const queue = new Int32Array(size);
    const score = new Float64Array(size);
    const num_preds = new Int32Array(size);

    while (budget > 0) {
        // 1. BFS from all ignitions to find the "fire front" distances
        distFire.fill(-1);
        let head = 0;
        let tail = 0;
        for (const [ix, iy] of ignitions) {
            const idx = iy * width + ix;
            distFire[idx] = 0;
            queue[tail++] = idx;
        }

        let anyTownReachable = false;
        while (head < tail) {
            const curr = queue[head++];
            if (isTownCell[curr]) anyTownReachable = true;
            
            const cx = curr % width;
            const cy = (curr / width) | 0;
            const d = distFire[curr];

            for (let i = 0; i < 4; i++) {
                const nx = cx + dx[i];
                const ny = cy + dy[i];
                if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                    const nidx = ny * width + nx;
                    if (board[nidx] === 0 && distFire[nidx] === -1) {
                        distFire[nidx] = d + 1;
                        queue[tail++] = nidx;
                    }
                }
            }
        }

        // If no town is reachable by fire, we've saved everything we can
        if (!anyTownReachable) break;

        // 2. Score propagation
        score.fill(0);
        num_preds.fill(0);
        
        // First, identify number of predecessors in the BFS DAG for each reachable cell
        for (let i = 0; i < tail; i++) {
            const curr = queue[i];
            if (townValue[curr] > 0) score[curr] = townValue[curr];
            
            const d = distFire[curr];
            if (d === 0) continue; // Ignitions have no predecessors in this context

            const cx = curr % width;
            const cy = (curr / width) | 0;
            let p = 0;
            for (let j = 0; j < 4; j++) {
                const nx = cx + dx[j];
                const ny = cy + dy[j];
                if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                    const nidx = ny * width + nx;
                    if (distFire[nidx] === d - 1) p++;
                }
            }
            num_preds[curr] = p;
        }

        // Propagate score backwards from furthest reachable nodes to ignitions
        for (let i = tail - 1; i >= 0; i--) {
            const curr = queue[i];
            const d = distFire[curr];
            if (d === 0 || score[curr] === 0) continue;

            const cx = curr % width;
            const cy = (curr / width) | 0;
            
            // Distribute current score among predecessors. 
            // This highlights bottlenecks where paths converge.
            const s = score[curr] / (num_preds[curr] || 1);

            for (let j = 0; j < 4; j++) {
                const nx = cx + dx[j];
                const ny = cy + dy[j];
                if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
                    const nidx = ny * width + nx;
                    if (distFire[nidx] === d - 1) {
                        score[nidx] += s;
                    }
                }
            }
        }

        // 3. Select the best cell to cut
        // We prefer non-town cells, but will cut a town cell if it's the only option to block fire.
        let bestIdx = -1;
        let maxScore = -1;
        
        // Pass 1: Look for non-town forest cells
        for (let i = 0; i < size; i++) {
            if (board[i] === 0 && !isIgnition[i] && !isTownCell[i]) {
                if (score[i] > maxScore) {
                    maxScore = score[i];
                    bestIdx = i;
                }
            }
        }
        
        // Pass 2: If no non-town cell is helpful, check town cells (though usually less ideal)
        if (bestIdx === -1 || maxScore === 0) {
            for (let i = 0; i < size; i++) {
                if (board[i] === 0 && !isIgnition[i]) {
                    if (score[i] > maxScore) {
                        maxScore = score[i];
                        bestIdx = i;
                    }
                }
            }
        }

        if (bestIdx === -1 || maxScore === 0) break;

        // Apply the cut
        const bx = bestIdx % width;
        const by = (bestIdx / width) | 0;
        firebreaks.push([bx, by]);
        board[bestIdx] = 1;
        budget--;
    }

    process.stdout.write(JSON.stringify({ firebreaks }));
}

solve();
