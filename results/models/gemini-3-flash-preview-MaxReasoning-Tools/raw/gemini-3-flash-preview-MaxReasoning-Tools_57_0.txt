const fs = require('fs');

/**
 * Wildfire Firebreak Placement Solver
 * 
 * Strategy:
 * 1. Find all towns that would burn (reachable from ignitions).
 * 2. Sort these towns by value descending.
 * 3. Greedily select towns to save. For each town, cut its own cell to save it.
 *    This is highly efficient (cost 1) and guaranteed to save the town.
 * 4. To handle large scenarios and accidental saves, re-run reachability periodically.
 */

function solve() {
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const { width, height, grid, ignitions, towns, cutBudget } = data;
    if (!grid || !ignitions || !towns) return;

    const n = width * height;
    let budget = cutBudget;
    let firebreaks = [];
    let broken = new Uint8Array(n);

    // Initial reachability analysis
    let reachableSet = getReachableSet(width, height, grid, ignitions, broken);
    
    // Only consider towns that are actually at risk (reachable from fire)
    let candidates = towns
        .filter(t => t.x >= 0 && t.x < width && t.y >= 0 && t.y < height && reachableSet[t.y * width + t.x])
        .sort((a, b) => b.value - a.value);

    let breakCount = 0;
    for (let i = 0; i < candidates.length; i++) {
        if (budget <= 0) break;

        const town = candidates[i];
        const idx = town.y * width + town.x;

        // Periodically refresh reachability to see if any town was already saved by previous breaks
        // This is much faster than re-running BFS for every single break.
        if (breakCount > 0 && breakCount % 50 === 0) {
            reachableSet = getReachableSet(width, height, grid, ignitions, broken);
        }

        if (reachableSet[idx] && !broken[idx]) {
            // Cut the town cell itself - cost 1, guaranteed to block fire from entering this cell
            broken[idx] = 1;
            firebreaks.push([town.x, town.y]);
            budget--;
            breakCount++;
        }
    }

    process.stdout.write(JSON.stringify({ firebreaks }));
}

// Reusable buffers to minimize garbage collection in large scenarios
let reachableBuffer = null;
let queueBuffer = null;

/**
 * Computes a bitset of all cells reachable from fire sources using BFS.
 */
function getReachableSet(width, height, grid, ignitions, broken) {
    const n = width * height;
    if (!reachableBuffer || reachableBuffer.length < n) {
        reachableBuffer = new Uint8Array(n);
        queueBuffer = new Int32Array(n);
    } else {
        reachableBuffer.fill(0);
    }
    const reachable = reachableBuffer;
    const queue = queueBuffer;
    let head = 0, tail = 0;

    for (let i = 0; i < ignitions.length; i++) {
        const [ix, iy] = ignitions[i];
        if (iy >= 0 && iy < height && ix >= 0 && ix < width) {
            const idx = iy * width + ix;
            // Fire starts on burnable forest that hasn't been cut
            if (grid[iy][ix] === '.' && !broken[idx]) {
                if (!reachable[idx]) {
                    reachable[idx] = 1;
                    queue[tail++] = idx;
                }
            }
        }
    }

    while (head < tail) {
        const curr = queue[head++];
        const cx = curr % width;
        const cy = (curr / width) | 0;

        // Check four-directional neighbors
        // Right
        if (cx + 1 < width) {
            const nidx = curr + 1;
            if (!reachable[nidx] && !broken[nidx] && grid[cy][cx + 1] === '.') {
                reachable[nidx] = 1;
                queue[tail++] = nidx;
            }
        }
        // Left
        if (cx - 1 >= 0) {
            const nidx = curr - 1;
            if (!reachable[nidx] && !broken[nidx] && grid[cy][cx - 1] === '.') {
                reachable[nidx] = 1;
                queue[tail++] = nidx;
            }
        }
        // Down
        if (cy + 1 < height) {
            const nidx = curr + width;
            if (!reachable[nidx] && !broken[nidx] && grid[cy + 1][cx] === '.') {
                reachable[nidx] = 1;
                queue[tail++] = nidx;
            }
        }
        // Up
        if (cy - 1 >= 0) {
            const nidx = curr - width;
            if (!reachable[nidx] && !broken[nidx] && grid[cy - 1][cx] === '.') {
                reachable[nidx] = 1;
                queue[tail++] = nidx;
            }
        }
    }
    return reachable;
}

solve();
