import struct

def word(opcode, count):
    return (count << 16) | opcode

def words_to_hex(words):
    return "".join(struct.pack("<I", w).hex() for w in words)

def string_to_words(s):
    b = s.encode('utf-8') + b'\x00'
    while len(b) % 4 != 0:
        b += b'\x00'
    return [struct.unpack("<I", b[i:i+4])[0] for i in range(0, len(b), 4)]

def float_to_u32(f):
    return struct.unpack("<I", struct.pack("<f", f))[0]

class SPIRVAssembler:
    def __init__(self):
        self.next_id = 1
        self.instructions = []
        self.types_and_constants = []
        self.decorations = []
        self.header_insts = []
        self.entry_points = []
        self.execution_modes = []
        self.functions = []

    def get_id(self):
        id = self.next_id
        self.next_id += 1
        return id

    def add_inst(self, list, opcode, *operands):
        list.append([word(opcode, 1 + len(operands))] + list(operands))

    def assemble(self):
        # Bound is next_id
        header = [0x07230203, 0x00010000, 0, self.next_id, 0]
        res = words_to_hex(header)
        for inst_list in [self.header_insts, self.entry_points, self.execution_modes, self.decorations, self.types_and_constants, self.functions]:
            for inst in inst_list:
                res += words_to_hex(inst)
        return res
