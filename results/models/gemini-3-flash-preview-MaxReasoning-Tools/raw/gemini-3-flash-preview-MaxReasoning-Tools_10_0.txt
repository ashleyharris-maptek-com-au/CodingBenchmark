To solve the 2D Cutting Stock Problem at both trivial and ludicrous scales, we employ an efficient **First-Fit Decreasing Height (FFDH)** shelf-based heuristic. This algorithm is particularly well-suited for large-scale problems due to its $O(N \log N)$ complexity (from sorting) and linear-time packing phase, ensuring it completes within the 30-second limit even for 50,000 rectangles.

### Algorithm Strategy
1.  **Initial Orientation**: We orient each rectangle to be "tall" (height $\ge$ width) whenever possible within the board's dimensions. This maximizes shelf height for the sorting phase.
2.  **Sorting**: Rectangles are sorted by height in descending order. This ensures that the first rectangle in any shelf defines the maximum height for that shelf, allowing subsequent shorter rectangles to fit.
3.  **First-Fit Packing**:
    *   We maintain a list of boards, each containing a list of "shelves."
    *   For each rectangle, we search for the first available shelf in any existing board where it fits (considering both its current and rotated orientations).
    *   If no shelf fits, we attempt to start a new shelf in an existing board.
    *   If no board can accommodate a new shelf, we open a new board.
4.  **Performance Optimization**: We use a `max_rem_w` pruning technique for each board to quickly skip boards that cannot accommodate the current rectangle, significantly accelerating the search on the "Ludicrous" scale.

```cpp
#include <iostream>
#include <vector>
#include <algorithm>

using namespace std;

/**
 * Problem: 2D Cutting Stock (Rectangle Bin Packing)
 * Goal: Minimize number of boards, handle up to 50,000 rectangles in 30s.
 * Approach: First-Fit Decreasing Height (FFDH) Shelf Algorithm with rotation support.
 */

struct Rect {
    int id;
    int w, h;          // Dimensions as placed in the shelf
    int board_idx;
    int x, y;
    int rotated;       // 1 if dimensions were swapped relative to original input
    int w_orig, h_orig;
};

struct Shelf {
    int y;
    int height;
    int rem_w;
    int next_x;
};

struct Board {
    int rem_h;
    int next_y;
    int max_rem_w;     // Max remaining width in any existing shelf of this board
    vector<Shelf> shelves;

    Board(int bh) : rem_h(bh), next_y(0), max_rem_w(0) {}
};

int main() {
    // Optimize I/O for high-volume data
    ios::sync_with_stdio(false);
    cin.tie(NULL);

    int N, BW, BH;
    if (!(cin >> N >> BW >> BH)) return 0;

    vector<Rect> rects(N);
    for (int i = 0; i < N; ++i) {
        rects[i].id = i;
        if (!(cin >> rects[i].w_orig >> rects[i].h_orig)) break;
        
        // Initial orientation strategy:
        // Prefer the "tall" orientation (h >= w) to maximize shelf height consistency,
        // while strictly respecting the board's boundaries (BW, BH).
        int w0 = rects[i].w_orig;
        int h0 = rects[i].h_orig;
        
        bool fits_normal = (w0 <= BW && h0 <= BH);
        bool fits_rotated = (h0 <= BW && w0 <= BH);

        if (fits_normal && fits_rotated) {
            if (h0 >= w0) {
                rects[i].w = w0; rects[i].h = h0; rects[i].rotated = 0;
            } else {
                rects[i].w = h0; rects[i].h = w0; rects[i].rotated = 1;
            }
        } else if (fits_normal) {
            rects[i].w = w0; rects[i].h = h0; rects[i].rotated = 0;
        } else if (fits_rotated) {
            rects[i].w = h0; rects[i].h = w0; rects[i].rotated = 1;
        } else {
            // Safety fallback for items that exceed board limits
            rects[i].w = w0; rects[i].h = h0; rects[i].rotated = 0;
        }
    }

    // Sort by height descending: Tallest items define the shelves
    sort(rects.begin(), rects.end(), [](const Rect& a, const Rect& b) {
        if (a.h != b.h) return a.h > b.h;
        return a.w > b.w;
    });

    vector<Board> boards;
    vector<Rect> results(N);

    for (int i = 0; i < N; ++i) {
        Rect& r = rects[i];
        bool placed = false;

        // Try to place in existing boards using First-Fit
        for (int bi = 0; bi < (int)boards.size(); ++bi) {
            Board& b = boards[bi];

            // Spatial Pruning: Skip board if neither orientation fits existing shelves or rem_h
            int min_r_dim = (r.w < r.h) ? r.w : r.h;
            if (b.rem_h < min_r_dim && b.max_rem_w < min_r_dim) continue;

            // 1. Check existing shelves
            for (auto& s : b.shelves) {
                // Try current orientation
                if (r.w <= s.rem_w && r.h <= s.height) {
                    r.board_idx = bi;
                    r.x = s.next_x;
                    r.y = s.y;
                    s.next_x += r.w;
                    s.rem_w -= r.w;
                    placed = true;
                    break;
                }
                // Try rotated orientation if it fits the existing shelf better
                int rw = r.h;
                int rh = r.w;
                if (rw <= s.rem_w && rh <= s.height) {
                    r.board_idx = bi;
                    r.x = s.next_x;
                    r.y = s.y;
                    r.w = rw; r.h = rh;
                    r.rotated = 1 - r.rotated;
                    s.next_x += r.w;
                    s.rem_w -= r.w;
                    placed = true;
                    break;
                }
            }
            if (placed) {
                // Update board's max width metric for future pruning
                int mw = 0;
                for (const auto& s : b.shelves) if (s.rem_w > mw) mw = s.rem_w;
                b.max_rem_w = mw;
                break;
            }

            // 2. Try starting a new shelf in this board
            // Try current orientation
            if (r.h <= b.rem_h && r.w <= BW) {
                Shelf ns;
                ns.y = b.next_y;
                ns.height = r.h;
                ns.rem_w = BW - r.w;
                ns.next_x = r.w;
                r.board_idx = bi;
                r.x = 0;
                r.y = ns.y;
                b.shelves.push_back(ns);
                b.next_y += r.h;
                b.rem_h -= r.h;
                if (ns.rem_w > b.max_rem_w) b.max_rem_w = ns.rem_w;
                placed = true;
                break;
            }
            // Try rotated orientation for a new shelf
            int rw = r.h;
            int rh = r.w;
            if (rh <= b.rem_h && rw <= BW) {
                Shelf ns;
                ns.y = b.next_y;
                ns.height = rh;
                ns.rem_w = BW - rw;
                ns.next_x = rw;
                r.board_idx = bi;
                r.x = 0;
                r.y = ns.y;
                r.w = rw; r.h = rh;
                r.rotated = 1 - r.rotated;
                b.shelves.push_back(ns);
                b.next_y += rh;
                b.rem_h -= rh;
                if (ns.rem_w > b.max_rem_w) b.max_rem_w = ns.rem_w;
                placed = true;
                break;
            }
        }

        if (!placed) {
            // 3. Open a new board if item fits nowhere else
            Board nb(BH);
            Shelf ns;
            ns.y = 0;
            ns.height = r.h;
            ns.rem_w = BW - r.w;
            ns.next_x = r.w;
            r.board_idx = (int)boards.size();
            r.x = 0;
            r.y = 0;
            nb.shelves.push_back(ns);
            nb.next_y = r.h;
            nb.rem_h = BH - r.h;
            nb.max_rem_w = ns.rem_w;
            boards.push_back(nb);
        }

        // Store result based on original input ID
        results[r.id] = r;
    }

    // Final Output: Total boards and placement coordinates
    cout << boards.size() << "\n";
    for (int i = 0; i < N; ++i) {
        cout << results[i].board_idx << " " << results[i].x << " " << results[i].y << " " << results[i].rotated << "\n";
    }

    return 0;
}
```
