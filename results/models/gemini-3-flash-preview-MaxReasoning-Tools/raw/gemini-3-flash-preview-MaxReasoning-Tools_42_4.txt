OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

               ; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

               ; Types and Constants
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
       %uint = OpTypeInt 32 0
     %uint_2 = OpConstant %uint 2
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

 %arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1

 %ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float

 %float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005
    %float_0 = OpConstant %float 0

       %main = OpFunction %void None %3
          %5 = OpLabel

               ; Load input positions p0, p1
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr

               ; Load input colors c0, c1
     %c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
         %c0 = OpLoad %v3float %c0_ptr
     %c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
         %c1 = OpLoad %v3float %c1_ptr

               ; Compute perpendicular direction in XY
      %p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
      %p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
        %dir = OpFSub %v2float %p1_xy %p0_xy
        %len = OpExtInst %float %1 Length %dir
       %norm = OpFDiv %v2float %dir %len
     %norm_y = OpCompositeExtract %float %norm 1
     %norm_x = OpCompositeExtract %float %norm 0
 %neg_norm_y = OpFNegate %float %norm_y
       %perp = OpCompositeConstruct %v2float %neg_norm_y %norm_x

               ; Vertex 1: p0 + perp * 0.04
    %offset0 = OpVectorTimesScalar %v2float %perp %float_0_04
 %offset0_v4 = OpCompositeConstruct %v4float %offset0 %float_0 %float_0
     %v1_pos = OpFAdd %v4float %p0 %offset0_v4
               OpStore %outColor %c0
     %v1_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %v1_ptr %v1_pos
               OpEmitVertex

               ; Vertex 2: p0 - perp * 0.04
     %v2_pos = OpFSub %v4float %p0 %offset0_v4
               OpStore %outColor %c0
     %v2_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %v2_ptr %v2_pos
               OpEmitVertex

               ; Vertex 3: p1 + perp * 0.005
    %offset1 = OpVectorTimesScalar %v2float %perp %float_0_005
 %offset1_v4 = OpCompositeConstruct %v4float %offset1 %float_0 %float_0
     %v3_pos = OpFAdd %v4float %p1 %offset1_v4
               OpStore %outColor %c1
     %v3_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %v3_ptr %v3_pos
               OpEmitVertex

               ; Vertex 4: p1 - perp * 0.005
     %v4_pos = OpFSub %v4float %p1 %offset1_v4
               OpStore %outColor %c1
     %v4_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0
               OpStore %v4_ptr %v4_pos
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
