```spirv
OpCapability Geometry
%1 = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main Invocations 1
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4

; Decorations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos_in Location 0
OpDecorate %normal_in Location 1
OpDecorate %color_in Location 2
OpDecorate %outColor Location 0

; Types
%void = OpTypeVoid
%func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1

%uint_2 = OpConstant %uint 2
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

; Input interface
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
%normal_in = OpVariable %ptr_in_arr_v3 Input
%color_in = OpVariable %ptr_in_arr_v3 Input

; Output interface
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

; Pointers for access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Constants
%float_0_04 = OpConstant %float 0.04
%float_0_005 = OpConstant %float 0.005

; Main logic
%main = OpFunction %void None %func
%label = OpLabel

; Load input positions from gl_in[0] and gl_in[1]
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

; Load input colors from color[0] and color[1]
%c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
%c1 = OpLoad %v3float %c1_ptr

; Perspective divide to NDC for computing screen-space direction
%p0_xy = OpVectorShuffle %v2float %p0 %p0 0 1
%p0_w = OpCompositeExtract %float %p0 3
%p0_ndc = OpFDiv %v2float %p0_xy %p0_w

%p1_xy = OpVectorShuffle %v2float %p1 %p1 0 1
%p1_w = OpCompositeExtract %float %p1 3
%p1_ndc = OpFDiv %v2float %p1_xy %p1_w

; Compute perpendicular direction in NDC
%dir = OpFSub %v2float %p1_ndc %p0_ndc
%dx = OpCompositeExtract %float %dir 0
%dy = OpCompositeExtract %float %dir 1
%neg_dy = OpFNegate %float %dy
%perp_raw = OpCompositeConstruct %v2float %neg_dy %dx
%perp = OpExtInst %v2float %1 Normalize %perp_raw

; Output vertex shared pointers
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Vertex 0: p0 + perp * thick_start (corrected for perspective)
%s0 = OpFMul %float %float_0_04 %p0_w
%offset0 = OpVectorTimesScalar %v2float %perp %s0
%v0_xy = OpFAdd %v2float %p0_xy %offset0
%p0_z = OpCompositeExtract %float %p0 2
%v0_pos = OpCompositeConstruct %v4float %v0_xy %p0_z %p0_w
OpStore %out_pos_ptr %v0_pos
OpStore %outColor %c0
OpEmitVertex

; Vertex 1: p0 - perp * thick_start
%v1_xy = OpFSub %v2float %p0_xy %offset0
%v1_pos = OpCompositeConstruct %v4float %v1_xy %p0_z %p0_w
OpStore %out_pos_ptr %v1_pos
OpStore %outColor %c0
OpEmitVertex

; Vertex 2: p1 + perp * thick_end
%s1 = OpFMul %float %float_0_005 %p1_w
%offset1 = OpVectorTimesScalar %v2float %perp %s1
%v2_xy = OpFAdd %v2float %p1_xy %offset1
%p1_z = OpCompositeExtract %float %p1 2
%v2_pos = OpCompositeConstruct %v4float %v2_xy %p1_z %p1_w
OpStore %out_pos_ptr %v2_pos
OpStore %outColor %c1
OpEmitVertex

; Vertex 3: p1 - perp * thick_end
%v3_xy = OpFSub %v2float %p1_xy %offset1
%v3_pos = OpCompositeConstruct %v4float %v3_xy %p1_z %p1_w
OpStore %out_pos_ptr %v3_pos
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
```
