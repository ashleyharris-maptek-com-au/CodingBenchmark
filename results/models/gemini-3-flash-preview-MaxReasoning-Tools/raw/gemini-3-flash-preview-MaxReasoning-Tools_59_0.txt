"use strict";

const fs = require('fs');

/**
 * Plans grid paths for search drones to maximize unique probability coverage.
 * Uses a greedy approach with target prioritization and coordination.
 */
function solve() {
    // Read input from stdin
    let input;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;

    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const { probability, drones, maxStepsPerDrone } = data;
    if (!probability || !drones) return;

    const rows = probability.length;
    if (rows === 0) {
        process.stdout.write(JSON.stringify({ paths: [] }) + '\n');
        return;
    }
    const cols = probability[0].length;
    const size = rows * cols;

    // Convert probability to Float32Array for performance and memory efficiency
    const probData = new Float32Array(size);
    for (let y = 0; y < rows; y++) {
        const row = probability[y];
        for (let x = 0; x < cols; x++) {
            probData[y * cols + x] = row[x];
        }
    }

    // State tracking
    const visited = new Uint8Array(size);
    const targeted = new Uint8Array(size);
    const paths = drones.map(d => [[d.x, d.y]]);
    
    // Initial positions are visited
    for (const d of drones) {
        visited[d.y * cols + d.x] = 1;
    }

    // Pre-allocate BFS structures
    const seen = new Uint32Array(size);
    let seenCounter = 0;
    const parent = new Int32Array(size);
    const dists = new Int32Array(size);
    const queue = new Int32Array(size * 2);

    /**
     * Finds the shortest path to the best reachable unvisited cell with positive probability.
     */
    function findPathToTarget(sx, sy, remainingSteps) {
        if (remainingSteps <= 0) return null;

        seenCounter++;
        let head = 0;
        let tail = 0;
        
        const startIdx = sy * cols + sx;
        queue[tail++] = sx;
        queue[tail++] = sy;
        seen[startIdx] = seenCounter;
        parent[startIdx] = -1;
        dists[startIdx] = 0;
        
        let bestX = -1, bestY = -1, bestVal = -1;
        let foundDist = -1;

        while (head < tail) {
            const x = queue[head++];
            const y = queue[head++];
            const idx = y * cols + x;
            const d = dists[idx];

            // If we already found a target at a closer distance, stop exploring further levels
            if (foundDist !== -1 && d > foundDist) break;

            // Check if current cell is a good target
            if (visited[idx] === 0 && targeted[idx] === 0 && probData[idx] > 0) {
                if (probData[idx] > bestVal) {
                    bestVal = probData[idx];
                    bestX = x;
                    bestY = y;
                    foundDist = d;
                }
            }

            // Continue BFS exploration if no target found yet or at this distance
            if (foundDist === -1 && d < remainingSteps) {
                const xP1 = x + 1, xM1 = x - 1, yP1 = y + 1, yM1 = y - 1;
                
                // Right
                if (xP1 < cols) {
                    const nidx = idx + 1;
                    if (seen[nidx] !== seenCounter && probData[nidx] !== -1) {
                        seen[nidx] = seenCounter;
                        parent[nidx] = idx;
                        dists[nidx] = d + 1;
                        queue[tail++] = xP1;
                        queue[tail++] = y;
                    }
                }
                // Left
                if (xM1 >= 0) {
                    const nidx = idx - 1;
                    if (seen[nidx] !== seenCounter && probData[nidx] !== -1) {
                        seen[nidx] = seenCounter;
                        parent[nidx] = idx;
                        dists[nidx] = d + 1;
                        queue[tail++] = xM1;
                        queue[tail++] = y;
                    }
                }
                // Down
                if (yP1 < rows) {
                    const nidx = idx + cols;
                    if (seen[nidx] !== seenCounter && probData[nidx] !== -1) {
                        seen[nidx] = seenCounter;
                        parent[nidx] = idx;
                        dists[nidx] = d + 1;
                        queue[tail++] = x;
                        queue[tail++] = yP1;
                    }
                }
                // Up
                if (yM1 >= 0) {
                    const nidx = idx - cols;
                    if (seen[nidx] !== seenCounter && probData[nidx] !== -1) {
                        seen[nidx] = seenCounter;
                        parent[nidx] = idx;
                        dists[nidx] = d + 1;
                        queue[tail++] = x;
                        queue[tail++] = yM1;
                    }
                }
            }
        }

        if (bestX !== -1) {
            const path = [];
            let curr = bestY * cols + bestX;
            while (curr !== -1 && curr !== startIdx) {
                path.push({ x: curr % cols, y: (curr / cols) | 0 });
                curr = parent[curr];
            }
            return path.reverse();
        }
        return null;
    }

    const droneTargetPaths = drones.map(() => []);

    // Main step-by-step loop
    for (let s = 0; s < maxStepsPerDrone; s++) {
        for (let i = 0; i < drones.length; i++) {
            const dronePath = paths[i];
            const [cx, cy] = dronePath[dronePath.length - 1];
            const remainingSteps = maxStepsPerDrone - s;

            let targetPath = droneTargetPaths[i];

            // If current target path is empty or target already visited by someone, find a new target
            let needNewTarget = targetPath.length === 0;
            if (!needNewTarget) {
                const target = targetPath[targetPath.length - 1];
                if (visited[target.y * cols + target.x]) {
                    needNewTarget = true;
                }
            }

            if (needNewTarget) {
                const res = findPathToTarget(cx, cy, remainingSteps);
                if (res) {
                    targetPath = res;
                    droneTargetPaths[i] = res;
                    const target = res[res.length - 1];
                    targeted[target.y * cols + target.x] = 1;
                } else {
                    targetPath = [];
                    droneTargetPaths[i] = [];
                }
            }

            if (targetPath.length > 0) {
                const next = targetPath.shift();
                dronePath.push([next.x, next.y]);
                visited[next.y * cols + next.x] = 1;
            } else {
                // Exploration fallback: try to move to any unvisited non-obstacle neighbor
                let moved = false;
                const dirs = [[1, 0], [-1, 0], [0, 1], [0, -1]];
                for (let j = 0; j < 4; j++) {
                    const dx = dirs[j][0], dy = dirs[j][1];
                    const nx = cx + dx, ny = cy + dy;
                    if (nx >= 0 && nx < cols && ny >= 0 && ny < rows) {
                        const nidx = ny * cols + nx;
                        if (probData[nidx] !== -1 && visited[nidx] === 0) {
                            dronePath.push([nx, ny]);
                            visited[nidx] = 1;
                            moved = true;
                            break;
                        }
                    }
                }
                if (!moved) {
                    // Stay in place if no valid unvisited moves
                    dronePath.push([cx, cy]);
                }
            }
        }
    }

    // Output JSON
    process.stdout.write(JSON.stringify({ paths }) + '\n');
}

// Execute solver
solve();
