OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color_in %color_out
               OpExecutionMode %main Triangles
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 3

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %color_in Location 2
               OpDecorate %color_out Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
%float_0_3333333 = OpConstant %float 0.333333333
%float_0_7 = OpConstant %float 0.7

; Interface Variables
%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
     %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_pv Output

 %arr_v3_in = OpTypeArray %v3float %uint_3
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
  %color_in = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
 %color_out = OpVariable %ptr_out_v3 Output

; Reusable Pointers
 %ptr_in_v4 = OpTypePointer Input %v4float
%ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float

; Main function
       %main = OpFunction %void None %3
          %5 = OpLabel

; 1. Read the 3 vertex positions and colors
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr
     %p2_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %p2_ptr

     %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr
     %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr
     %c2_ptr = OpAccessChain %ptr_in_v3 %color_in %int_2
         %c2 = OpLoad %v3float %c2_ptr

; 2. Compute centroid: C = (p0 + p1 + p2) / 3
      %sum01 = OpFAdd %v4float %p0 %p1
       %sum = OpFAdd %v4float %sum01 %p2
          %C = OpVectorTimesScalar %v4float %sum %float_0_3333333

; Pointer to output gl_Position
  %out_p_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; 3-5. For each vertex i: new_pos = C + 0.7 * (pI - C) and Emit

; Vertex 0
      %diff0 = OpFSub %v4float %p0 %C
     %scale0 = OpVectorTimesScalar %v4float %diff0 %float_0_7
      %new0  = OpFAdd %v4float %C %scale0
               OpStore %color_out %c0
               OpStore %out_p_ptr %new0
               OpEmitVertex

; Vertex 1
      %diff1 = OpFSub %v4float %p1 %C
     %scale1 = OpVectorTimesScalar %v4float %diff1 %float_0_7
      %new1  = OpFAdd %v4float %C %scale1
               OpStore %color_out %c1
               OpStore %out_p_ptr %new1
               OpEmitVertex

; Vertex 2
      %diff2 = OpFSub %v4float %p2 %C
     %scale2 = OpVectorTimesScalar %v4float %diff2 %float_0_7
      %new2  = OpFAdd %v4float %C %scale2
               OpStore %color_out %c2
               OpStore %out_p_ptr %new2
               OpEmitVertex

; 6. End primitive
               OpEndPrimitive
               OpReturn
               OpFunctionEnd
