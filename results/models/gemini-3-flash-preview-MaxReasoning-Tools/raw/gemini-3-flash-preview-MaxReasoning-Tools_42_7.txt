OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %color %outColor %worldPos %normal
               OpExecutionMode %main Triangles
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 3

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types, Constants and Variables
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_3 = OpConstant %uint 3
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
 %float_0_08 = OpConstant %float 0.08
    %float_0 = OpConstant %float 0
%float_0_333 = OpConstant %float 0.333333333

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_3
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%arr3_v3float = OpTypeArray %v3float %uint_3
%ptr_in_arr3_v3float = OpTypePointer Input %arr3_v3float
   %worldPos = OpVariable %ptr_in_arr3_v3float Input
     %normal = OpVariable %ptr_in_arr3_v3float Input
      %color = OpVariable %ptr_in_arr3_v3float Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3float = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_in_v3float = OpTypePointer Input %v3float
%ptr_out_v4float = OpTypePointer Output %v4float

; Main Function
       %main = OpFunction %void None %3
          %5 = OpLabel

; 1. Read the 3 vertex positions
         %10 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %10
         %11 = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %11
         %12 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
         %p2 = OpLoad %v4float %12

; 2. Compute face center (p0+p1+p2)/3
       %p0v3 = OpVectorShuffle %v3float %p0 %p0 0 1 2
       %p1v3 = OpVectorShuffle %v3float %p1 %p1 0 1 2
       %p2v3 = OpVectorShuffle %v3float %p2 %p2 0 1 2
        %sum = OpFAdd %v3float %p0v3 %p1v3
       %sum2 = OpFAdd %v3float %sum %p2v3
     %center = OpVectorTimesScalar %v3float %sum2 %float_0_333

; 3. Compute face normal using cross product
         %e1 = OpFSub %v3float %p1v3 %p0v3
         %e2 = OpFSub %v3float %p2v3 %p0v3
      %cross = OpExtInst %v3float %1 Cross %e1 %e2
     %normal_v = OpExtInst %v3float %1 Normalize %cross

; 4. Offset = Normal * 0.08
     %offset = OpVectorTimesScalar %v3float %normal_v %float_0_08
  %offset_v4 = OpCompositeConstruct %v4float %offset %float_0

; 5. Emit 3 vertices shifted by offset
; Vertex 0
     %p0_new = OpFAdd %v4float %p0 %offset_v4
     %o0_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %o0_ptr %p0_new
     %c0_ptr = OpAccessChain %ptr_in_v3float %color %int_0
         %c0 = OpLoad %v3float %c0_ptr
               OpStore %outColor %c0
               OpEmitVertex

; Vertex 1
     %p1_new = OpFAdd %v4float %p1 %offset_v4
     %o1_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %o1_ptr %p1_new
     %c1_ptr = OpAccessChain %ptr_in_v3float %color %int_1
         %c1 = OpLoad %v3float %c1_ptr
               OpStore %outColor %c1
               OpEmitVertex

; Vertex 2
     %p2_new = OpFAdd %v4float %p2 %offset_v4
     %o2_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
               OpStore %o2_ptr %p2_new
     %c2_ptr = OpAccessChain %ptr_in_v3float %color %int_2
         %c2 = OpLoad %v3float %c2_ptr
               OpStore %outColor %c2
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
