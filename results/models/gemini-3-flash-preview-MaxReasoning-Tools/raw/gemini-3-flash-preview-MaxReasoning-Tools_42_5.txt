OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos_in %color_in %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 7

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos_in Location 0
               OpDecorate %color_in Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_2 = OpConstant %uint 2
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1

%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

 %arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
   %color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
 %ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

 %ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

 %ptr_out_v4 = OpTypePointer Output %v4float
 %ptr_in_v3 = OpTypePointer Input %v3float

; Constants
%float_0_015 = OpConstant %float 0.015
 %float_0_75 = OpConstant %float 0.75
 %float_0_05 = OpConstant %float 0.05
    %float_0 = OpConstant %float 0.0
    %float_1 = OpConstant %float 1.0

  %main_func = OpTypeFunction %void

       %main = OpFunction %void None %main_func
      %label = OpLabel

; 1. Read line endpoints p0, p1 and colors c0, c1
    %p0_ptr = OpAccessChain %ptr_in_v3 %worldPos_in %int_0
        %p0 = OpLoad %v3float %p0_ptr
    %p1_ptr = OpAccessChain %ptr_in_v3 %worldPos_in %int_1
        %p1 = OpLoad %v3float %p1_ptr

    %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
        %c0 = OpLoad %v3float %c0_ptr
    %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
        %c1 = OpLoad %v3float %c1_ptr

; 2. Compute direction dir = p1 - p0, and perpendicular perp = normalize(-dir.y, dir.x)
       %dir = OpFSub %v3float %p1 %p0
     %dir_x = OpCompositeExtract %float %dir 0
     %dir_y = OpCompositeExtract %float %dir 1
 %neg_dir_y = OpFNegate %float %dir_y
%perp_unnorm = OpCompositeConstruct %v2float %neg_dir_y %dir_x
      %perp = OpExtInst %v2float %1 Normalize %perp_unnorm

; 3. Shaft: thickness = 0.015. Arrow starts at 75% along the line.
; mid = p0 + 0.75 * (p1 - p0)
%scaled_dir = OpVectorTimesScalar %v3float %dir %float_0_75
       %mid = OpFAdd %v3float %p0 %scaled_dir

; thickness offset
    %perp_x = OpCompositeExtract %float %perp 0
    %perp_y = OpCompositeExtract %float %perp 1
     %off_x = OpFMul %float %perp_x %float_0_015
     %off_y = OpFMul %float %perp_y %float_0_015
    %offset = OpCompositeConstruct %v3float %off_x %off_y %float_0

; 4. Emit shaft as triangle strip (4 verts): p0±perp*0.015, mid±perp*0.015.
               OpStore %outColor %c0
%pos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; v1 = p0 + offset
    %v1_pos3 = OpFAdd %v3float %p0 %offset
      %v1_x = OpCompositeExtract %float %v1_pos3 0
      %v1_y = OpCompositeExtract %float %v1_pos3 1
      %v1_z = OpCompositeExtract %float %v1_pos3 2
    %v1_pos4 = OpCompositeConstruct %v4float %v1_x %v1_y %v1_z %float_1
               OpStore %pos_out_ptr %v1_pos4
               OpEmitVertex

; v2 = p0 - offset
    %v2_pos3 = OpFSub %v3float %p0 %offset
      %v2_x = OpCompositeExtract %float %v2_pos3 0
      %v2_y = OpCompositeExtract %float %v2_pos3 1
      %v2_z = OpCompositeExtract %float %v2_pos3 2
    %v2_pos4 = OpCompositeConstruct %v4float %v2_x %v2_y %v2_z %float_1
               OpStore %pos_out_ptr %v2_pos4
               OpEmitVertex

; v3 = mid + offset
    %v3_pos3 = OpFAdd %v3float %mid %offset
      %v3_x = OpCompositeExtract %float %v3_pos3 0
      %v3_y = OpCompositeExtract %float %v3_pos3 1
      %v3_z = OpCompositeExtract %float %v3_pos3 2
    %v3_pos4 = OpCompositeConstruct %v4float %v3_x %v3_y %v3_z %float_1
               OpStore %pos_out_ptr %v3_pos4
               OpEmitVertex

; v4 = mid - offset
    %v4_pos3 = OpFSub %v3float %mid %offset
      %v4_x = OpCompositeExtract %float %v4_pos3 0
      %v4_y = OpCompositeExtract %float %v4_pos3 1
      %v4_z = OpCompositeExtract %float %v4_pos3 2
    %v4_pos4 = OpCompositeConstruct %v4float %v4_x %v4_y %v4_z %float_1
               OpStore %pos_out_ptr %v4_pos4
               OpEmitVertex

               OpEndPrimitive

; 5. Emit arrowhead as triangle strip (3 verts): mid + perp*0.05, mid - perp*0.05, p1 (tip).
               OpStore %outColor %c1

    %aoff_x = OpFMul %float %perp_x %float_0_05
    %aoff_y = OpFMul %float %perp_y %float_0_05
   %aoffset = OpCompositeConstruct %v3float %aoff_x %aoff_y %float_0

; v5 = mid + aoffset
    %v5_pos3 = OpFAdd %v3float %mid %aoffset
      %v5_x = OpCompositeExtract %float %v5_pos3 0
      %v5_y = OpCompositeExtract %float %v5_pos3 1
      %v5_z = OpCompositeExtract %float %v5_pos3 2
    %v5_pos4 = OpCompositeConstruct %v4float %v5_x %v5_y %v5_z %float_1
               OpStore %pos_out_ptr %v5_pos4
               OpEmitVertex

; v6 = mid - aoffset
    %v6_pos3 = OpFSub %v3float %mid %aoffset
      %v6_x = OpCompositeExtract %float %v6_pos3 0
      %v6_y = OpCompositeExtract %float %v6_pos3 1
      %v6_z = OpCompositeExtract %float %v6_pos3 2
    %v6_pos4 = OpCompositeConstruct %v4float %v6_x %v6_y %v6_z %float_1
               OpStore %pos_out_ptr %v6_pos4
               OpEmitVertex

; v7 = p1
      %v7_x = OpCompositeExtract %float %p1 0
      %v7_y = OpCompositeExtract %float %p1 1
      %v7_z = OpCompositeExtract %float %p1 2
    %v7_pos4 = OpCompositeConstruct %v4float %v7_x %v7_y %v7_z %float_1
               OpStore %pos_out_ptr %v7_pos4
               OpEmitVertex

               OpEndPrimitive

               OpReturn
               OpFunctionEnd
