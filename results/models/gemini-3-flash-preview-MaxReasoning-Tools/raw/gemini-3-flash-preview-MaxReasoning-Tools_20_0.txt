The following Rust program implements a spatial grid-based outlier detection system designed to work efficiently across varying scales of drillhole datasets. It identifies samples that deviate significantly from their spatial neighbors using a local Z-score analysis.

### Technical Approach
1.  **Adaptive Spatial Indexing**: The program calculates a bounding box and determines a grid cell size dynamically based on the sample density and total dataset volume. For massive datasets, it increases the target samples per cell to maintain a manageable number of cells and reduce memory overhead.
2.  **Efficient Data Management**: Samples are stored using a flat vector structure to minimize allocations. Property values are stored as `f32` to optimize memory usage (crucial for the 200,000-hole scale), while spatial coordinates use `f64` for precision.
3.  **Local Contextual Analysis**: Instead of a global search, it compares each sample to its 27 immediate neighboring grid cells (3x3x3 neighborhood). This ensures $O(N)$ complexity, making it suitable for datasets with millions of points.
4.  **Optimized Neighborhood Calculation**: Cell-level statistics (sums and sums of squares) are precomputed once. When evaluating individual samples, the program aggregates these cell stats and subtracts the target sample's contribution, drastically reducing the number of operations per sample.
5.  **Robustness**: The logic handles missing data (NaN), varying property counts, and edge cases like zero-variance neighborhoods or 1D/2D spatial distributions.

```rust
use std::io::{self, BufRead, Write};
use std::collections::HashMap;

/// Represents a single sample's spatial and metadata.
struct Sample {
    hole_idx: u32,
    sample_idx: u32,
    x: f64,
    y: f64,
    z: f64,
}

/// Represents a detected suspect outlier.
struct Suspect {
    hole_id: String,
    sample_idx: usize,
    prop_name: String,
    confidence: f64,
}

/// Key for the spatial grid.
#[derive(PartialEq, Eq, Hash, Clone, Copy)]
struct CellKey(i32, i32, i32);

/// Data stored in each spatial grid cell.
struct CellData {
    indices: Vec<usize>,
    sums: Vec<f64>,
    sums_sq: Vec<f64>,
    count: usize,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    // Using a large buffer for fast reading of massive datasets.
    let mut reader = io::BufReader::with_capacity(1024 * 1024, stdin.lock());

    let mut line = String::new();
    if reader.read_line(&mut line)? == 0 { return Ok(()); }
    let mut parts = line.split_ascii_whitespace();
    let num_holes: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return Ok(()),
    };
    let num_props: usize = match parts.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return Ok(()),
    };

    line.clear();
    if reader.read_line(&mut line)? == 0 { return Ok(()); }
    let prop_names: Vec<String> = line.split_ascii_whitespace().map(|s| s.to_string()).collect();

    let mut hole_ids = Vec::with_capacity(num_holes);
    let mut all_samples = Vec::new();
    let mut all_values = Vec::new();

    // Parse holes and their samples.
    for h_idx in 0..num_holes {
        line.clear();
        if reader.read_line(&mut line)? == 0 { break; }
        let mut parts = line.split_ascii_whitespace();
        let hole_id = match parts.next() {
            Some(s) => s.to_string(),
            None => break,
        };
        let start_x: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let start_y: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let start_z: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let dir_x: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let dir_y: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let dir_z: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let _length: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);
        let num_samples: usize = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0);

        hole_ids.push(hole_id);

        // Normalize direction vector to unit vector for spatial calculation.
        let mag = (dir_x * dir_x + dir_y * dir_y + dir_z * dir_z).sqrt();
        let (dx, dy, dz) = if mag > 1e-9 {
            (dir_x / mag, dir_y / mag, dir_z / mag)
        } else {
            (0.0, 0.0, 0.0)
        };

        for s_idx in 0..num_samples {
            line.clear();
            if reader.read_line(&mut line)? == 0 { break; }
            let mut parts = line.split_ascii_whitespace();
            let depth: f64 = parts.next().and_then(|s| s.parse().ok()).unwrap_or(0.0);

            all_samples.push(Sample {
                hole_idx: h_idx as u32,
                sample_idx: s_idx as u32,
                x: start_x + dx * depth,
                y: start_y + dy * depth,
                z: start_z + dz * depth,
            });

            // Store values as f32 to save memory for massive datasets.
            for _ in 0..num_props {
                let val_str = parts.next().unwrap_or("NaN");
                let val: f32 = val_str.parse().unwrap_or(f32::NAN);
                all_values.push(val);
            }
        }
    }

    if all_samples.is_empty() {
        println!("0");
        return Ok(());
    }

    // Execute adaptive outlier detection.
    let mut suspects = detect_outliers_optimized(&hole_ids, &all_samples, &all_values, &prop_names, num_props);

    // Sort suspects by confidence descending.
    suspects.sort_by(|a, b| b.confidence.partial_cmp(&a.confidence).unwrap_or(std::cmp::Ordering::Equal));

    // Output suspects.
    println!("{}", suspects.len());
    let stdout = io::stdout();
    let mut writer = io::BufWriter::with_capacity(1024 * 1024, stdout.lock());
    for s in suspects {
        let _ = writeln!(writer, "{} {} {} {:.4}", s.hole_id, s.sample_idx, s.prop_name, s.confidence);
    }

    Ok(())
}

/// Detects outliers using a spatial grid and Z-score analysis.
/// Adaptive logic adjusts cell size based on dataset scale and distribution.
fn detect_outliers_optimized(
    hole_ids: &[String],
    samples: &[Sample],
    values: &[f32],
    prop_names: &[String],
    num_props: usize,
) -> Vec<Suspect> {
    // 1. Calculate Bounding Box.
    let mut min_x = samples[0].x;
    let mut max_x = samples[0].x;
    let mut min_y = samples[0].y;
    let mut max_y = samples[0].y;
    let mut min_z = samples[0].z;
    let mut max_z = samples[0].z;

    for s in samples {
        if s.x < min_x { min_x = s.x; } if s.x > max_x { max_x = s.x; }
        if s.y < min_y { min_y = s.y; } if s.y > max_y { max_y = s.y; }
        if s.z < min_z { min_z = s.z; } if s.z > max_z { max_z = s.z; }
    }

    let n = samples.len() as f64;
    let dx = max_x - min_x;
    let dy = max_y - min_y;
    let dz = max_z - min_z;
    
    // 2. Adaptive cell size calculation based on density.
    let samples_per_cell = if n > 1_000_000.0 { 40.0 } else { 12.0 };
    let vol = dx * dy * dz;
    let mut cell_size;
    if vol > 1e-6 {
        cell_size = (vol / (n / samples_per_cell)).powf(1.0/3.0);
    } else {
        let area = (dx * dy).max(dx * dz).max(dy * dz);
        if area > 1e-6 {
            cell_size = (area / (n / samples_per_cell)).powf(1.0/2.0);
        } else {
            let line = dx.max(dy).max(dz);
            cell_size = line / (n / samples_per_cell).max(1.0);
        }
    }
    
    // Clamp cell size to ensure it's not too small or too large.
    let max_dim = dx.max(dy).max(dz);
    if cell_size < max_dim / 2000.0 { cell_size = max_dim / 2000.0; }
    if cell_size < 1e-4 { cell_size = 1.0; }

    // 3. Build Spatial Grid.
    let mut grid: HashMap<CellKey, CellData> = HashMap::with_capacity(
        if n > 1_000_000.0 { 100_000 } else { (n as usize) / 8 + 1 }
    );

    for (i, s) in samples.iter().enumerate() {
        let gx = (s.x / cell_size).floor() as i32;
        let gy = (s.y / cell_size).floor() as i32;
        let gz = (s.z / cell_size).floor() as i32;
        let key = CellKey(gx, gy, gz);

        let entry = grid.entry(key).or_insert_with(|| CellData {
            indices: Vec::new(),
            sums: vec![0.0; num_props],
            sums_sq: vec![0.0; num_props],
            count: 0,
        });

        entry.indices.push(i);
        entry.count += 1;
        for p in 0..num_props {
            let v = values[i * num_props + p];
            if !v.is_nan() {
                let v64 = v as f64;
                entry.sums[p] += v64;
                entry.sums_sq[p] += v64 * v64;
            }
        }
    }

    // 4. Identify Outliers.
    let mut suspects = Vec::new();
    let threshold = 5.0; // Statistical confidence threshold.

    let mut neighbor_cells = Vec::with_capacity(27);
    let mut total_sums = vec![0.0; num_props];
    let mut total_sums_sq = vec![0.0; num_props];

    for (key, cell_data) in &grid {
        neighbor_cells.clear();
        for ox in -1..=1 {
            for oy in -1..=1 {
                for oz in -1..=1 {
                    let neighbor_key = CellKey(key.0 + ox, key.1 + oy, key.2 + oz);
                    if let Some(nd) = grid.get(&neighbor_key) {
                        neighbor_cells.push(nd);
                    }
                }
            }
        }

        for p in 0..num_props {
            total_sums[p] = 0.0;
            total_sums_sq[p] = 0.0;
        }
        let mut total_count_all = 0;
        for nd in &neighbor_cells {
            total_count_all += nd.count;
            for p in 0..num_props {
                total_sums[p] += nd.sums[p];
                total_sums_sq[p] += nd.sums_sq[p];
            }
        }

        for &i in &cell_data.indices {
            let s = &samples[i];
            for p in 0..num_props {
                let val_f32 = values[i * num_props + p];
                if val_f32.is_nan() { continue; }
                let val = val_f32 as f64;

                let local_sum = total_sums[p] - val;
                let local_sum_sq = total_sums_sq[p] - val * val;
                let local_count = total_count_all - 1;

                // Require sufficient neighbors for valid statistical comparison.
                if local_count >= 4 {
                    let mean = local_sum / local_count as f64;
                    let var = (local_sum_sq / local_count as f64) - mean * mean;
                    let std_dev = var.max(0.0).sqrt();

                    let confidence = if std_dev < 1e-8 {
                        if (val - mean).abs() > 1e-4 { 20.0 } else { 0.0 }
                    } else {
                        (val - mean).abs() / std_dev
                    };

                    if confidence > threshold {
                        suspects.push(Suspect {
                            hole_id: hole_ids[s.hole_idx as usize].clone(),
                            sample_idx: s.sample_idx as usize,
                            prop_name: prop_names[p].clone(),
                            confidence,
                        });
                    }
                }
            }
        }
    }

    suspects
}
```
