; SPIR-V

; Generator: Khronos; 1

               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4
               OpExecutionMode %main Invocations 1

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
     %uint_2 = OpConstant %uint 2
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1

%gl_PerVertex_in = OpTypeStruct %v4float
 %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
   %worldPos = OpVariable %ptr_in_arr_v3 Input
     %normal = OpVariable %ptr_in_arr_v3 Input
      %color = OpVariable %ptr_in_arr_v3 Input

%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

 %float_0_03 = OpConstant %float 0.03
    %float_0 = OpConstant %float 0.0

; Main Function
       %main = OpFunction %void None %3
         %lb = OpLabel

; 1. Read p0, p1 from gl_in
    %ptr_p0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
        %p0 = OpLoad %v4float %ptr_p0
    %ptr_p1 = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
        %p1 = OpLoad %v4float %ptr_p1

; 2. Read c0, c1 from color
    %ptr_c0 = OpAccessChain %ptr_in_v3 %color %int_0
        %c0 = OpLoad %v3float %ptr_c0
    %ptr_c1 = OpAccessChain %ptr_in_v3 %color %int_1
        %c1 = OpLoad %v3float %ptr_c1

; 3. Compute line direction: dir = p1 - p0
       %dir = OpFSub %v4float %p1 %p0

; 4. Compute perpendicular: perp = normalize(-dir.y, dir.x) * 0.03
    %dir_xy = OpVectorShuffle %v2float %dir %dir 0 1
    %dir_x  = OpCompositeExtract %float %dir_xy 0
    %dir_y  = OpCompositeExtract %float %dir_xy 1
 %neg_dir_y = OpFNegate %float %dir_y
%perp_xy_raw = OpCompositeConstruct %v2float %neg_dir_y %dir_x
%perp_xy_norm = OpExtInst %v2float %1 Normalize %perp_xy_raw
    %px_norm = OpCompositeExtract %float %perp_xy_norm 0
    %py_norm = OpCompositeExtract %float %perp_xy_norm 1
        %px = OpFMul %float %px_norm %float_0_03
        %py = OpFMul %float %py_norm %float_0_03
   %perp_v4 = OpCompositeConstruct %v4float %px %py %float_0 %float_0

; Output Position pointer
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; 5. Emit 4 vertices as triangle strip
; Vertex 0: p0 + perp, color c0
    %v0_pos = OpFAdd %v4float %p0 %perp_v4
              OpStore %ptr_out_pos %v0_pos
              OpStore %outColor %c0
              OpEmitVertex

; Vertex 1: p0 - perp, color c0
    %v1_pos = OpFSub %v4float %p0 %perp_v4
              OpStore %ptr_out_pos %v1_pos
              OpStore %outColor %c0
              OpEmitVertex

; Vertex 2: p1 + perp, color c1
    %v2_pos = OpFAdd %v4float %p1 %perp_v4
              OpStore %ptr_out_pos %v2_pos
              OpStore %outColor %c1
              OpEmitVertex

; Vertex 3: p1 - perp, color c1
    %v3_pos = OpFSub %v4float %p1 %perp_v4
              OpStore %ptr_out_pos %v3_pos
              OpStore %outColor %c1
              OpEmitVertex

              OpEndPrimitive
              OpReturn
              OpFunctionEnd
