; SPIR-V
; Version: 1.0
; Generator: Khronos; 1
; Bound: 100
; Schema: 0
               OpCapability Shader
               OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %worldPos_in %normal_in %color_in %gl_out %color_out
               OpExecutionMode %main InputLines
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 4

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %worldPos_in Location 0
               OpDecorate %normal_in Location 1
               OpDecorate %color_in Location 2
               OpDecorate %color_out Location 0

; Types
       %void = OpTypeVoid
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1

     %uint_2 = OpConstant %uint 2
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
 %float_0_03 = OpConstant %float 0.03

%gl_PerVertex_in = OpTypeStruct %v4float
  %arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

  %arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%worldPos_in = OpVariable %ptr_in_arr_v3 Input
 %normal_in = OpVariable %ptr_in_arr_v3 Input
   %color_in = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

 %ptr_out_v3 = OpTypePointer Output %v3float
  %color_out = OpVariable %ptr_out_v3 Output

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

  %main_type = OpTypeFunction %void
       %main = OpFunction %void None %main_type
      %entry = OpLabel

; Load positions from gl_in
     %p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
         %p0 = OpLoad %v4float %p0_ptr
     %p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
         %p1 = OpLoad %v4float %p1_ptr

; Load colors from color_in
     %c0_ptr = OpAccessChain %ptr_in_v3 %color_in %int_0
         %c0 = OpLoad %v3float %c0_ptr
     %c1_ptr = OpAccessChain %ptr_in_v3 %color_in %int_1
         %c1 = OpLoad %v3float %c1_ptr

; Extract components for 2D line calculations (using clip space xy)
       %p0_x = OpCompositeExtract %float %p0 0
       %p0_y = OpCompositeExtract %float %p0 1
       %p1_x = OpCompositeExtract %float %p1 0
       %p1_y = OpCompositeExtract %float %p1 1

; Compute direction vector (p1 - p0).xy
      %dir_x = OpFSub %float %p1_x %p0_x
      %dir_y = OpFSub %float %p1_y %p0_y

; Compute perpendicular vector: perp = normalize(-dir.y, dir.x) * 0.03
 %perp_raw_x = OpFNegate %float %dir_y
   %perp_raw = OpCompositeConstruct %v2float %perp_raw_x %dir_x
  %perp_norm = OpExtInst %v2float %1 Normalize %perp_raw
   %perp_n_x = OpCompositeExtract %float %perp_norm 0
   %perp_n_y = OpCompositeExtract %float %perp_norm 1
     %perp_x = OpFMul %float %perp_n_x %float_0_03
     %perp_y = OpFMul %float %perp_n_y %float_0_03

; Extract Z and W to maintain perspective/depth from original vertices
       %p0_z = OpCompositeExtract %float %p0 2
       %p0_w = OpCompositeExtract %float %p0 3
       %p1_z = OpCompositeExtract %float %p1 2
       %p1_w = OpCompositeExtract %float %p1 3

; Output position pointer (member 0 of gl_PerVertex struct)
%out_pos_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

; Emit Triangle Strip: v0, v1, v2, v3
; v0: p0 + perp, color c0
      %v0_x = OpFAdd %float %p0_x %perp_x
      %v0_y = OpFAdd %float %p0_y %perp_y
     %v0_pos = OpCompositeConstruct %v4float %v0_x %v0_y %p0_z %p0_w
               OpStore %color_out %c0
               OpStore %out_pos_ptr %v0_pos
               OpEmitVertex

; v1: p0 - perp, color c0
      %v1_x = OpFSub %float %p0_x %perp_x
      %v1_y = OpFSub %float %p0_y %perp_y
     %v1_pos = OpCompositeConstruct %v4float %v1_x %v1_y %p0_z %p0_w
               OpStore %color_out %c0
               OpStore %out_pos_ptr %v1_pos
               OpEmitVertex

; v2: p1 + perp, color c1
      %v2_x = OpFAdd %float %p1_x %perp_x
      %v2_y = OpFAdd %float %p1_y %perp_y
     %v2_pos = OpCompositeConstruct %v4float %v2_x %v2_y %p1_z %p1_w
               OpStore %color_out %c1
               OpStore %out_pos_ptr %v2_pos
               OpEmitVertex

; v3: p1 - perp, color c1
      %v3_x = OpFSub %float %p1_x %perp_x
      %v3_y = OpFSub %float %p1_y %perp_y
     %v3_pos = OpCompositeConstruct %v4float %v3_x %v3_y %p1_z %p1_w
               OpStore %color_out %c1
               OpStore %out_pos_ptr %v3_pos
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
