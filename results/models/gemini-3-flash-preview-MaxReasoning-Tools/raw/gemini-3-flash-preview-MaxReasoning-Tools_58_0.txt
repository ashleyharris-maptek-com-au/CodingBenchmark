"use strict";

const fs = require('fs');

/**
 * Assign warehouse pick jobs to robots.
 * 
 * Algorithm:
 * 1. Read input JSON from stdin.
 * 2. Collect all points of interest (robot start positions and job positions).
 * 3. Sort jobs by priority (descending) and then ID (ascending).
 * 4. For each job in sorted order:
 *    a. Calculate (or use cached) grid distances from the job's position to all points of interest.
 *    b. Find the robot whose current position is closest to the job.
 *    c. Tie-break using the smallest robot ID.
 *    d. Assign the job to that robot and update the robot's current position to the job's position.
 * 5. Print assignments as JSON to stdout.
 */
function solve() {
    let input;
    try {
        // Read input from stdin
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;
    
    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const { grid, robots, jobs } = data;
    if (!grid || grid.length === 0 || !robots || !jobs) {
        process.stdout.write(JSON.stringify({ assignments: [] }) + '\n');
        return;
    }
    
    const rows = grid.length;
    const cols = grid[0].length;

    // We use a map to uniquely identify every (x, y) position of interest.
    // Robot starts and job locations are the only positions robots will ever be at.
    const posToIdxMap = new Map();
    const allPositions = [];
    function getPosIdx(x, y) {
        const key = `${x},${y}`;
        if (posToIdxMap.has(key)) return posToIdxMap.get(key);
        const idx = allPositions.length;
        allPositions.push({ x, y });
        posToIdxMap.set(key, idx);
        return idx;
    }

    // Track robot start positions
    for (const r of robots) getPosIdx(r.x, r.y);
    // Track job positions
    for (const j of jobs) getPosIdx(j.x, j.y);

    // Initial state for each robot
    const robotState = robots.map(r => ({
        id: r.id,
        posIdx: posToIdxMap.get(`${r.x},${r.y}`)
    })).sort((a, b) => a.id - b.id);

    // Prepare assignments array indexed by robot ID
    let maxRobotId = -1;
    for (const r of robots) {
        if (r.id > maxRobotId) maxRobotId = r.id;
    }
    const assignments = Array.from({ length: maxRobotId + 1 }, () => []);

    // Deterministic greedy: process jobs by priority then ID
    const sortedJobs = [...jobs].sort((a, b) => {
        if (b.priority !== a.priority) return b.priority - a.priority;
        if (a.id < b.id) return -1;
        if (a.id > b.id) return 1;
        return 0;
    });

    // Cache distances from unique job positions to all other POIs
    const distMatrixCache = new Map();
    // Pre-allocate buffers for BFS to avoid repeated allocation
    const bfsDists = new Int32Array(rows * cols);
    const bfsQueue = new Uint32Array(rows * cols * 2);

    /**
     * Performs BFS from a starting (x, y) and returns distances to all points of interest.
     */
    function getDistancesFrom(startX, startY) {
        const cacheKey = `${startX},${startY}`;
        if (distMatrixCache.has(cacheKey)) return distMatrixCache.get(cacheKey);

        bfsDists.fill(-1);
        let head = 0;
        let tail = 0;

        bfsQueue[tail++] = startX;
        bfsQueue[tail++] = startY;
        bfsDists[startY * cols + startX] = 0;

        while (head < tail) {
            const x = bfsQueue[head++];
            const y = bfsQueue[head++];
            const d = bfsDists[y * cols + x];
            const d1 = d + 1;

            // Neighbors: Up, Down, Left, Right
            // Check boundaries, blocked cells (#), and already visited cells
            if (y - 1 >= 0 && grid[y - 1][x] === '.' && bfsDists[(y - 1) * cols + x] === -1) {
                bfsDists[(y - 1) * cols + x] = d1;
                bfsQueue[tail++] = x;
                bfsQueue[tail++] = y - 1;
            }
            if (y + 1 < rows && grid[y + 1][x] === '.' && bfsDists[(y + 1) * cols + x] === -1) {
                bfsDists[(y + 1) * cols + x] = d1;
                bfsQueue[tail++] = x;
                bfsQueue[tail++] = y + 1;
            }
            if (x - 1 >= 0 && grid[y][x - 1] === '.' && bfsDists[y * cols + (x - 1)] === -1) {
                bfsDists[y * cols + (x - 1)] = d1;
                bfsQueue[tail++] = x - 1;
                bfsQueue[tail++] = y;
            }
            if (x + 1 < cols && grid[y][x + 1] === '.' && bfsDists[y * cols + (x + 1)] === -1) {
                bfsDists[y * cols + (x + 1)] = d1;
                bfsQueue[tail++] = x + 1;
                bfsQueue[tail++] = y;
            }
        }

        // We only care about distances to the specific POIs we tracked
        const result = new Int32Array(allPositions.length);
        for (let i = 0; i < allPositions.length; i++) {
            const p = allPositions[i];
            result[i] = bfsDists[p.y * cols + p.x];
        }
        distMatrixCache.set(cacheKey, result);
        return result;
    }

    // Assign each job to the best robot
    for (const job of sortedJobs) {
        const jobDists = getDistancesFrom(job.x, job.y);
        
        let bestRobot = null;
        let minDistance = Infinity;

        // Find the robot that can reach this job the fastest (shortest distance)
        for (let i = 0; i < robotState.length; i++) {
            const rs = robotState[i];
            const d = jobDists[rs.posIdx];
            
            // If the job is reachable from this robot's current position
            if (d !== -1) {
                if (d < minDistance) {
                    minDistance = d;
                    bestRobot = rs;
                }
                // Tie-breaking: since robotState is sorted by ID, the first robot 
                // encountered with minDistance is the one with the smallest ID.
            }
        }

        // Fallback: if no robot can reach the job (unlikely in a warehouse scenario),
        // we must still assign it to fulfill the "every job must appear exactly once" rule.
        if (!bestRobot) {
            bestRobot = robotState[0];
        }

        assignments[bestRobot.id].push(job.id);
        
        // Update robot's current position to the job location
        bestRobot.posIdx = posToIdxMap.get(`${job.x},${job.y}`);
    }

    // Print the final assignments JSON
    process.stdout.write(JSON.stringify({ assignments }) + '\n');
}

// Execute the solver
solve();
