OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %color %gl_out %outColor
               OpExecutionMode %main InputPoints
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main OutputVertices 4

; Decorations
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpDecorate %color Location 2
               OpDecorate %outColor Location 0

; Types
       %void = OpTypeVoid
          %3 = OpTypeFunction %void
      %float = OpTypeFloat 32
    %v4float = OpTypeVector %float 4
    %v3float = OpTypeVector %float 3
       %uint = OpTypeInt 32 0
     %uint_1 = OpConstant %uint 1
        %int = OpTypeInt 32 1
      %int_0 = OpConstant %int 0

; Input gl_PerVertex block
%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
      %gl_in = OpVariable %ptr_in_arr_pv Input

; Input color array
%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
      %color = OpVariable %ptr_in_arr_v3 Input

; Output gl_PerVertex block
%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
     %gl_out = OpVariable %ptr_out_pv Output

; Output color
%ptr_out_v3 = OpTypePointer Output %v3float
   %outColor = OpVariable %ptr_out_v3 Output

; Constants
    %float_h = OpConstant %float 0.05

; Pointers for access
%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

; Main function
       %main = OpFunction %void None %3
      %label = OpLabel

; Load input color from color[0]
%ptr_in_color_0 = OpAccessChain %ptr_in_v3 %color %int_0
     %inColor = OpLoad %v3float %ptr_in_color_0

; Load input position from gl_in[0].gl_Position
%ptr_in_pos_0 = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
      %pos_in = OpLoad %v4float %ptr_in_pos_0

; Extract x, y, z, w
          %x = OpCompositeExtract %float %pos_in 0
          %y = OpCompositeExtract %float %pos_in 1
          %z = OpCompositeExtract %float %pos_in 2
          %w = OpCompositeExtract %float %pos_in 3

; Output color is same for all vertices
               OpStore %outColor %inColor

; Common pointer for gl_out.gl_Position
%ptr_out_pos = OpAccessChain %ptr_out_v4 %gl_out %int_0

; v0: Top (x, y+h, z, w)
      %y_top = OpFAdd %float %y %float_h
      %pos_0 = OpCompositeConstruct %v4float %x %y_top %z %w
               OpStore %ptr_out_pos %pos_0
               OpEmitVertex

; v1: Right (x+h, y, z, w)
     %x_right = OpFAdd %float %x %float_h
      %pos_1 = OpCompositeConstruct %v4float %x_right %y %z %w
               OpStore %ptr_out_pos %pos_1
               OpEmitVertex

; v2: Left (x-h, y, z, w)
      %x_left = OpFSub %float %x %float_h
      %pos_2 = OpCompositeConstruct %v4float %x_left %y %z %w
               OpStore %ptr_out_pos %pos_2
               OpEmitVertex

; v3: Bottom (x, y-h, z, w)
    %y_bottom = OpFSub %float %y %float_h
      %pos_3 = OpCompositeConstruct %v4float %x %y_bottom %z %w
               OpStore %ptr_out_pos %pos_3
               OpEmitVertex

               OpEndPrimitive
               OpReturn
               OpFunctionEnd
