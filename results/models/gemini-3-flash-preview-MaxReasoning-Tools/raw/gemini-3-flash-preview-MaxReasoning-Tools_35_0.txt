To solve the Minimum Dominating Set problem, I implemented a greedy heuristic in Rust. This approach is highly effective for NP-hard problems on large graphs within a time-constrained environment.

### Algorithm Strategy
1.  **Greedy Construction**: The core of the algorithm is a greedy selection process. In each step, it picks the vertex that covers the maximum number of currently uncovered vertices. This is implemented efficiently using a priority queue with lazy updates, resulting in $O(M \log N)$ complexity.
2.  **Randomized Tie-Breaking**: To improve the result quality, I incorporated randomized tie-breaking. By running the greedy algorithm multiple times with different random seeds, the solver can explore various parts of the solution space and retain the best result found.
3.  **Pruning Phase**: After each greedy construction, a pruning step is performed. It iterates through the selected set and removes any vertex that is redundant (i.e., all vertices it covers are already covered by other members of the set). Vertices are checked for removal in increasing order of their degree to prioritize keeping high-impact vertices.
4.  **Performance Optimization**:
    *   **Fast I/O**: A custom `FastScanner` using `BufRead::fill_buf` and `consume` was implemented to handle large inputs efficiently.
    *   **Memory Efficiency**: The graph is stored as an adjacency list, and memory usage is kept minimal to accommodate very large datasets.
    *   **Time Management**: The solver monitors its elapsed time and continues running randomized iterations for up to 280 seconds, ensuring it finishes within the 5-minute limit while maximizing the search for a smaller set.

### Final Implementation

```rust
use std::io::{self, BufRead, Write, BufWriter};
use std::collections::BinaryHeap;
use std::time::{Instant, SystemTime};

/// A simple SplitMix64 RNG for randomized tie-breaking in the greedy heuristic.
struct SimpleRng {
    state: u64,
}

impl SimpleRng {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next(&mut self) -> u64 {
        let mut z = self.state.wrapping_add(0x9e3779b97f4a7c15);
        self.state = z;
        z = (z ^ (z >> 30)).wrapping_mul(0xbf58476d1ce4e5b9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94d049bb133111eb);
        z ^ (z >> 31)
    }
}

/// A fast scanner to read large graph inputs efficiently.
struct FastScanner<R: BufRead> {
    reader: R,
}

impl<R: BufRead> FastScanner<R> {
    fn next_usize(&mut self) -> Option<usize> {
        let mut val = 0usize;
        let mut started = false;
        loop {
            let (consumed, result) = {
                let buf = match self.reader.fill_buf() {
                    Ok(b) => b,
                    Err(_) => return None,
                };
                if buf.is_empty() {
                    return if started { Some(val) } else { None };
                }
                let mut c = 0;
                let mut res = None;
                for &b in buf {
                    c += 1;
                    if b.is_ascii_whitespace() {
                        if started {
                            res = Some(val);
                            break;
                        }
                    } else if b.is_ascii_digit() {
                        started = true;
                        val = val.wrapping_mul(10).wrapping_add((b - b'0') as usize);
                    }
                }
                (c, res)
            };
            self.reader.consume(consumed);
            if result.is_some() {
                return result;
            }
        }
    }
}

/// Adds a vertex to the dominating set and updates the coverage and gains.
fn add_to_set(
    v: usize,
    result: &mut Vec<usize>,
    is_covered: &mut Vec<bool>,
    covered_count: &mut usize,
    gain: &mut Vec<i32>,
    adj: &Vec<Vec<usize>>,
) {
    result.push(v);
    
    // Check vertex v itself
    if !is_covered[v] {
        is_covered[v] = true;
        *covered_count += 1;
        gain[v] -= 1;
        for &w in &adj[v] {
            gain[w] -= 1;
        }
    }
    // Check neighbors of v
    for &neighbor in &adj[v] {
        if !is_covered[neighbor] {
            is_covered[neighbor] = true;
            *covered_count += 1;
            gain[neighbor] -= 1;
            for &w in &adj[neighbor] {
                gain[w] -= 1;
            }
        }
    }
}

/// Runs a greedy heuristic to find a dominating set.
/// If rng is provided, tie-breaks are randomized.
fn run_greedy(n: usize, adj: &Vec<Vec<usize>>, mut rng: Option<&mut SimpleRng>) -> Vec<usize> {
    let mut is_covered = vec![false; n];
    let mut covered_count = 0;
    let mut result = Vec::new();
    let mut gain = vec![0i32; n];
    for i in 0..n {
        gain[i] = (adj[i].len() + 1) as i32;
    }

    let mut pq = BinaryHeap::new();
    for i in 0..n {
        let priority = if let Some(ref mut r) = rng {
            (gain[i] as i64) << 32 | (r.next() as u32 as i64)
        } else {
            (gain[i] as i64) << 32
        };
        pq.push((priority, i));
    }

    while covered_count < n {
        let (priority, v) = match pq.pop() {
            Some(x) => x,
            None => {
                let mut found = false;
                for i in 0..n {
                    if !is_covered[i] {
                        add_to_set(i, &mut result, &mut is_covered, &mut covered_count, &mut gain, adj);
                        found = true;
                        break;
                    }
                }
                if !found { break; }
                continue;
            }
        };
        
        let g = (priority >> 32) as i32;

        if g != gain[v] {
            if gain[v] > 0 {
                let new_priority = if let Some(ref mut r) = rng {
                    (gain[v] as i64) << 32 | (r.next() as u32 as i64)
                } else {
                    (gain[v] as i64) << 32
                };
                pq.push((new_priority, v));
            }
            continue;
        }

        if g <= 0 {
            let mut found = false;
            for i in 0..n {
                if !is_covered[i] {
                    add_to_set(i, &mut result, &mut is_covered, &mut covered_count, &mut gain, adj);
                    found = true;
                    break;
                }
            }
            if !found { break; }
            continue;
        }

        add_to_set(v, &mut result, &mut is_covered, &mut covered_count, &mut gain, adj);
    }
    result
}

/// Prunes redundant vertices from the dominating set.
fn prune(n: usize, adj: &Vec<Vec<usize>>, result: Vec<usize>) -> Vec<usize> {
    if result.is_empty() { return result; }
    let mut cover_count = vec![0; n];
    for &v in &result {
        cover_count[v] += 1;
        for &neighbor in &adj[v] {
            cover_count[neighbor] += 1;
        }
    }

    let mut sorted_result = result.clone();
    sorted_result.sort_by_key(|&v| adj[v].len());

    let mut final_set = Vec::new();
    for v in sorted_result {
        let mut redundant = true;
        if cover_count[v] <= 1 {
            redundant = false;
        } else {
            for &neighbor in &adj[v] {
                if cover_count[neighbor] <= 1 {
                    redundant = false;
                    break;
                }
            }
        }

        if redundant {
            cover_count[v] -= 1;
            for &neighbor in &adj[v] {
                cover_count[neighbor] -= 1;
            }
        } else {
            final_set.push(v);
        }
    }
    final_set
}

fn main() {
    let start_time = Instant::now();
    let stdin = io::stdin();
    let mut scanner = FastScanner { reader: stdin.lock() };

    let n = match scanner.next_usize() { Some(v) => v, None => return };
    let m = match scanner.next_usize() { Some(v) => v, None => return };

    let mut adj = vec![vec![]; n];
    for _ in 0..m {
        let u = match scanner.next_usize() { Some(v) => v, None => break };
        let v = match scanner.next_usize() { Some(v) => v, None => break };
        if u < n && v < n {
            adj[u].push(v);
            adj[v].push(u);
        }
    }

    let mut best_set = run_greedy(n, &adj, None);
    best_set = prune(n, &adj, best_set);
    let mut best_size = best_set.len();

    let seed = SystemTime::now().duration_since(SystemTime::UNIX_EPOCH)
        .map(|d| d.as_nanos() as u64)
        .unwrap_or(42);
    let mut rng = SimpleRng::new(seed);

    while start_time.elapsed().as_secs() < 280 {
        let mut current_set = run_greedy(n, &adj, Some(&mut rng));
        current_set = prune(n, &adj, current_set);
        if current_set.len() < best_size {
            best_size = current_set.len();
            best_set = current_set;
        }
        if n > 200000 { break; }
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let _ = writeln!(out, "{}", best_set.len());
    for (i, &v) in best_set.iter().enumerate() {
        if i > 0 { let _ = write!(out, " "); }
        let _ = write!(out, "{}", v);
    }
    let _ = writeln!(out);
}
```
