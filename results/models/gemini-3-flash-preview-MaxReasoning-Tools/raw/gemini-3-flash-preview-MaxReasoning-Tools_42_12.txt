OpCapability Geometry
          %1 = OpExtInstImport "GLSL.std.450"
               OpMemoryModel Logical GLSL450
               OpEntryPoint Geometry %main "main" %gl_in %gl_out %inColor %outColor %worldPos %normal
               OpExecutionMode %main InputTrianglesAdjacency
               OpExecutionMode %main OutputTriangleStrip
               OpExecutionMode %main Invocations 1
               OpExecutionMode %main OutputVertices 15

; Decorations
               OpDecorate %gl_PerVertex_in Block
               OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
               OpDecorate %gl_PerVertex_out Block
               OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
               OpDecorate %inColor Location 2
               OpDecorate %outColor Location 0
               OpDecorate %worldPos Location 0
               OpDecorate %normal Location 1

; Types
       %void = OpTypeVoid
       %bool = OpTypeBool
      %float = OpTypeFloat 32
    %v2float = OpTypeVector %float 2
    %v3float = OpTypeVector %float 3
    %v4float = OpTypeVector %float 4
       %uint = OpTypeInt 32 0
        %int = OpTypeInt 32 1
     %uint_6 = OpConstant %uint 6
      %int_0 = OpConstant %int 0
      %int_1 = OpConstant %int 1
      %int_2 = OpConstant %int 2
      %int_3 = OpConstant %int 3
      %int_4 = OpConstant %int 4
      %int_5 = OpConstant %int 5

%gl_PerVertex_in = OpTypeStruct %v4float
%arr6_gl_pv_in = OpTypeArray %gl_PerVertex_in %uint_6
%ptr_in_arr6_gl_pv = OpTypePointer Input %arr6_gl_pv_in
     %gl_in = OpVariable %ptr_in_arr6_gl_pv Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_gl_pv = OpTypePointer Output %gl_PerVertex_out
    %gl_out = OpVariable %ptr_out_gl_pv Output

%arr6_v3float = OpTypeArray %v3float %uint_6
%ptr_in_arr6_v3float = OpTypePointer Input %arr6_v3float
   %inColor = OpVariable %ptr_in_arr6_v3float Input
  %worldPos = OpVariable %ptr_in_arr6_v3float Input
    %normal = OpVariable %ptr_in_arr6_v3float Input

%ptr_out_v3float = OpTypePointer Output %v3float
  %outColor = OpVariable %ptr_out_v3float Output

%ptr_in_v4float = OpTypePointer Input %v4float
%ptr_out_v4float = OpTypePointer Output %v4float
%ptr_in_v3float = OpTypePointer Input %v3float

%float_0_3 = OpConstant %float 0.3
%float_0_0075 = OpConstant %float 0.0075
    %float_1 = OpConstant %float 1.0
     %white = OpConstantComposite %v3float %float_1 %float_1 %float_1

  %main_func = OpTypeFunction %void

       %main = OpFunction %void None %main_func
      %entry = OpLabel

; --- Load Triangle Vertices ---
     %ptr_p0 = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
      %pos0 = OpLoad %v4float %ptr_p0
     %ptr_p2 = OpAccessChain %ptr_in_v4float %gl_in %int_2 %int_0
      %pos2 = OpLoad %v4float %ptr_p2
     %ptr_p4 = OpAccessChain %ptr_in_v4float %gl_in %int_4 %int_0
      %pos4 = OpLoad %v4float %ptr_p4

; --- Emit Filled Triangle (Dimmed) ---
     %ptr_op_pos = OpAccessChain %ptr_out_v4float %gl_out %int_0

     %ptr_c0 = OpAccessChain %ptr_in_v3float %inColor %int_0
       %c0 = OpLoad %v3float %ptr_c0
     %c0_dim = OpVectorTimesScalar %v3float %c0 %float_0_3
               OpStore %outColor %c0_dim
               OpStore %ptr_op_pos %pos0
               OpEmitVertex

     %ptr_c2 = OpAccessChain %ptr_in_v3float %inColor %int_2
       %c2 = OpLoad %v3float %ptr_c2
     %c2_dim = OpVectorTimesScalar %v3float %c2 %float_0_3
               OpStore %outColor %c2_dim
               OpStore %ptr_op_pos %pos2
               OpEmitVertex

     %ptr_c4 = OpAccessChain %ptr_in_v3float %inColor %int_4
       %c4 = OpLoad %v3float %ptr_c4
     %c4_dim = OpVectorTimesScalar %v3float %c4 %float_0_3
               OpStore %outColor %c4_dim
               OpStore %ptr_op_pos %pos4
               OpEmitVertex

               OpEndPrimitive

; --- Edge Calculations ---
     %w0 = OpCompositeExtract %float %pos0 3
    %xy0 = OpVectorShuffle %v2float %pos0 %pos0 0 1
  %w0vec = OpCompositeConstruct %v2float %w0 %w0
%p0_final = OpFDiv %v2float %xy0 %w0vec

     %w2 = OpCompositeExtract %float %pos2 3
    %xy2 = OpVectorShuffle %v2float %pos2 %pos2 0 1
  %w2vec = OpCompositeConstruct %v2float %w2 %w2
%p2_final = OpFDiv %v2float %xy2 %w2vec

     %w4 = OpCompositeExtract %float %pos4 3
    %xy4 = OpVectorShuffle %v2float %pos4 %pos4 0 1
  %w4vec = OpCompositeConstruct %v2float %w4 %w4
%p4_final = OpFDiv %v2float %xy4 %w4vec

; Edge 0-2
  %dir02 = OpFSub %v2float %p2_final %p0_final
%norm02 = OpExtInst %v2float %1 Normalize %dir02
     %dx = OpCompositeExtract %float %norm02 0
     %dy = OpCompositeExtract %float %norm02 1
 %neg_dy = OpFNegate %float %dy
 %perp02 = OpCompositeConstruct %v2float %neg_dy %dx
%off02 = OpVectorTimesScalar %v2float %perp02 %float_0_0075

               OpStore %outColor %white
    %o0p = OpVectorTimesScalar %v2float %off02 %w0
  %xy0p = OpFAdd %v2float %xy0 %o0p
   %x0p = OpCompositeExtract %float %xy0p 0
   %y0p = OpCompositeExtract %float %xy0p 1
   %z0 = OpCompositeExtract %float %pos0 2
  %v0p = OpCompositeConstruct %v4float %x0p %y0p %z0 %w0
               OpStore %ptr_op_pos %v0p
               OpEmitVertex
    %o0m = OpVectorTimesScalar %v2float %off02 %w0
  %xy0m = OpFSub %v2float %xy0 %o0m
   %x0m = OpCompositeExtract %float %xy0m 0
   %y0m = OpCompositeExtract %float %xy0m 1
  %v0m = OpCompositeConstruct %v4float %x0m %y0m %z0 %w0
               OpStore %ptr_op_pos %v0m
               OpEmitVertex
    %o2p = OpVectorTimesScalar %v2float %off02 %w2
  %xy2p = OpFAdd %v2float %xy2 %o2p
   %x2p = OpCompositeExtract %float %xy2p 0
   %y2p = OpCompositeExtract %float %xy2p 1
   %z2 = OpCompositeExtract %float %pos2 2
  %v2p = OpCompositeConstruct %v4float %x2p %y2p %z2 %w2
               OpStore %ptr_op_pos %v2p
               OpEmitVertex
    %o2m = OpVectorTimesScalar %v2float %off02 %w2
  %xy2m = OpFSub %v2float %xy2 %o2m
   %x2m = OpCompositeExtract %float %xy2m 0
   %y2m = OpCompositeExtract %float %xy2m 1
  %v2m = OpCompositeConstruct %v4float %x2m %y2m %z2 %w2
               OpStore %ptr_op_pos %v2m
               OpEmitVertex
               OpEndPrimitive

; Edge 2-4
  %dir24 = OpFSub %v2float %p4_final %p2_final
%norm24 = OpExtInst %v2float %1 Normalize %dir24
    %dx2 = OpCompositeExtract %float %norm24 0
    %dy2 = OpCompositeExtract %float %norm24 1
%neg_dy2 = OpFNegate %float %dy2
%perp24 = OpCompositeConstruct %v2float %neg_dy2 %dx2
%off24 = OpVectorTimesScalar %v2float %perp24 %float_0_0075

               OpStore %outColor %white
   %o2p2 = OpVectorTimesScalar %v2float %off24 %w2
 %xy2p2 = OpFAdd %v2float %xy2 %o2p2
  %x2p2 = OpCompositeExtract %float %xy2p2 0
  %y2p2 = OpCompositeExtract %float %xy2p2 1
 %v2p2 = OpCompositeConstruct %v4float %x2p2 %y2p2 %z2 %w2
               OpStore %ptr_op_pos %v2p2
               OpEmitVertex
   %o2m2 = OpVectorTimesScalar %v2float %off24 %w2
 %xy2m2 = OpFSub %v2float %xy2 %o2m2
  %x2m2 = OpCompositeExtract %float %xy2m2 0
  %y2m2 = OpCompositeExtract %float %xy2m2 1
 %v2m2 = OpCompositeConstruct %v4float %x2m2 %y2m2 %z2 %w2
               OpStore %ptr_op_pos %v2m2
               OpEmitVertex
   %o4p = OpVectorTimesScalar %v2float %off24 %w4
 %xy4p = OpFAdd %v2float %xy4 %o4p
  %x4p = OpCompositeExtract %float %xy4p 0
  %y4p = OpCompositeExtract %float %xy4p 1
  %z4 = OpCompositeExtract %float %pos4 2
 %v4p = OpCompositeConstruct %v4float %x4p %y4p %z4 %w4
               OpStore %ptr_op_pos %v4p
               OpEmitVertex
   %o4m = OpVectorTimesScalar %v2float %off24 %w4

  %x4m = OpCompositeExtract %float %xy4m 0
  %y4m = OpCompositeExtract %float %xy4m 1

               OpEndPrimitive

; Edge 4-0
  %dir40 = OpFSub %v2float %p0_final %p4_final

    %dx3 = OpCompositeExtract %float %norm40 0
