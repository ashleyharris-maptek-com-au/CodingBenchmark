This C++ solver implements an adaptive approach to the Travelling Salesman Problem (TSP) that scales efficiently from a few cities to billions. 

### Strategy
1.  **Small Scale ($N < 2,000$):** Combines a Morton-ordered initial tour with an exhaustive **2-opt** local search. This ensures high-quality results for trivial and medium cases.
2.  **Medium Scale ($2,000 \le N < 1,000,000$):** Uses **Morton order (Z-order curve)** sorting to create a spatially aware initial tour, then applies a **windowed 2-opt** heuristic. By only checking swaps within a sliding window of 400 cities, it significantly improves tour quality while maintaining near-linear time complexity.
3.  **Large & Extreme Scale ($N \ge 1,000,000$):** Employs a pure **Morton order** sorting approach. This $O(N \log N)$ algorithm is extremely fast and provides a mathematically sound approximation of the optimal tour by preserving spatial locality.
4.  **Performance & Memory:**
    *   **Fast I/O:** Custom `FastIO` class for parsing floats and integers using `fread` and `fwrite`, critical for processing $10^7+$ cities within 300 seconds.
    *   **Memory Efficiency:** Uses a compact 24-byte `CityData` struct. For 1 billion cities, this consumes $\sim$22.4 GB, fitting within the 32 GB RAM limit.
    *   **Bit Manipulation:** Uses bit interleaving with magic numbers for fast Z-order calculation.

### Implementation

```cpp
#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <cstdint>
#include <cstdio>

using namespace std;

// Fast I/O for floats and ints
struct FastIO {
    static const int BUF_SIZE = 1 << 20;
    char buf[BUF_SIZE];
    int pos = 0, len = 0;

    char readChar() {
        if (pos == len) {
            pos = 0;
            len = (int)fread(buf, 1, BUF_SIZE, stdin);
            if (!len) return EOF;
        }
        return buf[pos++];
    }

    int readInt() {
        int x = 0;
        char c = readChar();
        while (c != EOF && (c < '0' || c > '9')) c = readChar();
        if (c == EOF) return -1;
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = readChar();
        }
        return x;
    }

    float readFloat() {
        double x = 0;
        char c = readChar();
        while (c != EOF && c != '-' && (c < '0' || c > '9') && c != '.') c = readChar();
        if (c == EOF) return 0.0f;
        bool neg = false;
        if (c == '-') { neg = true; c = readChar(); }
        while (c >= '0' && c <= '9') {
            x = x * 10 + (c - '0');
            c = readChar();
        }
        if (c == '.') {
            double f = 0.1;
            c = readChar();
            while (c >= '0' && c <= '9') {
                x += (c - '0') * f;
                f *= 0.1;
                c = readChar();
            }
        }
        if (c == 'e' || c == 'E') {
            c = readChar();
            int exp = 0;
            bool eneg = false;
            if (c == '-') { eneg = true; c = readChar(); }
            else if (c == '+') { c = readChar(); }
            while (c >= '0' && c <= '9') {
                exp = exp * 10 + (c - '0');
                c = readChar();
            }
            if (eneg) x /= pow(10, exp);
            else x *= pow(10, exp);
        }
        return (float)(neg ? -x : x);
    }

    char out_buf[BUF_SIZE];
    int out_pos = 0;

    void writeInt(int x) {
        if (out_pos + 12 >= BUF_SIZE) flush();
        if (x == 0) {
            out_buf[out_pos++] = '0';
        } else {
            char s[12];
            int i = 0;
            while (x > 0) {
                s[i++] = (x % 10) + '0';
                x /= 10;
            }
            while (i > 0) out_buf[out_pos++] = s[--i];
        }
    }

    void writeChar(char c) {
        if (out_pos + 1 >= BUF_SIZE) flush();
        out_buf[out_pos++] = c;
    }

    void flush() {
        if (out_pos > 0) {
            fwrite(out_buf, 1, out_pos, stdout);
            out_pos = 0;
        }
    }
} io;

struct CityData {
    float x, y;
    int id;
    uint64_t z;
};

// Morton order interleave bits
inline uint64_t split_bits(uint32_t x) {
    uint64_t r = x;
    r = (r | (r << 16)) & 0x0000FFFF0000FFFF;
    r = (r | (r << 8)) & 0x00FF00FF00FF00FF;
    r = (r | (r << 4)) & 0x0F0F0F0F0F0F0F0F;
    r = (r | (r << 2)) & 0x3333333333333333;
    r = (r | (r << 1)) & 0x5555555555555555;
    return r;
}

inline uint64_t get_z(uint32_t x, uint32_t y) {
    return (split_bits(x) << 1) | split_bits(y);
}

inline double distSq(const CityData& a, const CityData& b) {
    double dx = (double)a.x - b.x;
    double dy = (double)a.y - b.y;
    return dx * dx + dy * dy;
}

int main() {
    int N = io.readInt();
    if (N <= 0) return 0;
    if (N == 1) {
        io.writeInt(0);
        io.writeChar('\n');
        io.flush();
        return 0;
    }

    // Read cities and calculate bounding box
    vector<CityData> cities(N);
    float minX = 1e30f, maxX = -1e30f, minY = 1e30f, maxY = -1e30f;
    for (int i = 0; i < N; ++i) {
        cities[i].x = io.readFloat();
        cities[i].y = io.readFloat();
        cities[i].id = i;
        if (cities[i].x < minX) minX = cities[i].x;
        if (cities[i].x > maxX) maxX = cities[i].x;
        if (cities[i].y < minY) minY = cities[i].y;
        if (cities[i].y > maxY) maxY = cities[i].y;
    }

    double rangeX = (double)maxX - minX;
    double rangeY = (double)maxY - minY;
    if (rangeX <= 0) rangeX = 1.0;
    if (rangeY <= 0) rangeY = 1.0;

    // Calculate Z-order for each city
    for (int i = 0; i < N; ++i) {
        uint32_t ux = (uint32_t)(((double)cities[i].x - minX) / rangeX * 4294967295.0);
        uint32_t uy = (uint32_t)(((double)cities[i].y - minY) / rangeY * 4294967295.0);
        cities[i].z = get_z(ux, uy);
    }

    // Initial tour via Space-Filling Curve sorting
    sort(cities.begin(), cities.end(), [](const CityData& a, const CityData& b) {
        return a.z < b.z;
    });

    // Adaptive local refinement
    if (N < 2000) {
        bool improved = true;
        while (improved) {
            improved = false;
            for (int i = 0; i < N; ++i) {
                for (int j = i + 2; j < N; ++j) {
                    if (i == 0 && j == N - 1) continue;
                    int next_i = (i + 1) % N;
                    int next_j = (j + 1) % N;
                    double d_old = distSq(cities[i], cities[next_i]) + distSq(cities[j], cities[next_j]);
                    double d_new = distSq(cities[i], cities[j]) + distSq(cities[next_i], cities[next_j]);
                    if (d_new < d_old - 1e-9) {
                        reverse(cities.begin() + i + 1, cities.begin() + j + 1);
                        improved = true;
                    }
                }
            }
        }
    } else if (N < 1000000) {
        const int WINDOW = 400;
        for (int iter = 0; iter < 3; ++iter) {
            bool improved_any = false;
            for (int i = 0; i < N; ++i) {
                int limit = min(N, i + WINDOW);
                for (int j = i + 2; j < limit; ++j) {
                    if (i == 0 && j == N - 1) continue;
                    int next_i = (i + 1) % N;
                    int next_j = (j + 1) % N;
                    double d_old = distSq(cities[i], cities[next_i]) + distSq(cities[j], cities[next_j]);
                    double d_new = distSq(cities[i], cities[j]) + distSq(cities[next_i], cities[next_j]);
                    if (d_new < d_old - 1e-9) {
                        reverse(cities.begin() + i + 1, cities.begin() + j + 1);
                        improved_any = true;
                    }
                }
            }
            if (!improved_any) break;
        }
    }

    // Rotate tour to start at city 0
    int start_pos = 0;
    for (int i = 0; i < N; ++i) {
        if (cities[i].id == 0) {
            start_pos = i;
            break;
        }
    }

    // Output tour
    for (int i = 0; i < N; ++i) {
        io.writeInt(cities[(start_pos + i) % N].id);
        if (i < N - 1) io.writeChar(' ');
    }
    io.writeChar('\n');
    io.flush();

    return 0;
}
```
