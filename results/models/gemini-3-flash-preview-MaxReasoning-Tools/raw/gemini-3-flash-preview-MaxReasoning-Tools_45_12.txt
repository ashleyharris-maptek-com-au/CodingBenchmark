import struct

def word(w):
    return struct.pack('<I', w).hex()

def fword(f):
    return struct.pack('<f', f).hex()

def instruction(opcode, *operands):
    ops = []
    for op in operands:
        if isinstance(op, int):
            ops.append(word(op))
        elif isinstance(op, float):
            ops.append(fword(op))
        elif isinstance(op, str):
            b = op.encode('utf-8') + b'\x00'
            while len(b) % 4 != 0:
                b += b'\x00'
            for i in range(0, len(b), 4):
                ops.append(struct.pack('<I', struct.unpack('<I', b[i:i+4])[0]).hex())
    
    word_count = 1 + len(ops)
    first_word = (word_count << 16) | opcode
    return word(first_word) + "".join(ops)
