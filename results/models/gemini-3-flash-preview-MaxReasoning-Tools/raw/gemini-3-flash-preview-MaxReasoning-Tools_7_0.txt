This C++ solver implements a Constructive Solid Geometry (CSG) Union operation designed for scalability and performance. It uses a **Bounding Volume Hierarchy (BVH)** to accelerate both triangle-triangle intersection tests and point-in-polyhedron classification (via ray casting). For intersecting faces, it employs a **recursive subdivision** strategy to approximate the intersection boundary while maintaining manifoldness and watertightness through epsilon-based vertex merging (snapping).

The algorithm proceeds in four main phases:
1.  **Spatial Acceleration**: Builds BVH trees for both input meshes to enable $O(N \log N)$ complexity.
2.  **Intersection Search**: Identifies all pairs of intersecting triangles using the BVHs.
3.  **Classification & Subdivision**:
    *   **Clean faces**: Classified as inside or outside the other mesh using robust ray casting. Centroids are shifted slightly along the normal to handle coplanar cases.
    *   **Dirty (intersecting) faces**: Subdivided into smaller triangles, each classified individually.
4.  **Mesh Cleanup**: Merges nearly identical vertices and removes duplicate or degenerate faces to ensure a clean manifold output.

```cpp
#include <iostream>
#include <vector>
#include <array>
#include <algorithm>
#include <cmath>
#include <limits>
#include <map>
#include <cstdio>

using namespace std;

// Robustness parameters
const double EPS = 1e-7;
const double MERGE_EPS = 1e-6;

struct Vec3 {
    double x, y, z;
    Vec3 operator+(const Vec3& v) const { return {x+v.x, y+v.y, z+v.z}; }
    Vec3 operator-(const Vec3& v) const { return {x-v.x, y-v.y, z-v.z}; }
    Vec3 operator*(double s) const { return {x*s, y*s, z*s}; }
    Vec3 operator/(double s) const { return {x/s, y/s, z/s}; }
    double dot(const Vec3& v) const { return x*v.x + y*v.y + z*v.z; }
    Vec3 cross(const Vec3& v) const { return {y*v.z-z*v.y, z*v.x-x*v.z, x*v.y-y*v.x}; }
    double normSq() const { return dot(*this); }
    double norm() const { return sqrt(normSq()); }
    bool operator<(const Vec3& v) const {
        if (abs(x-v.x) > MERGE_EPS) return x < v.x;
        if (abs(y-v.y) > MERGE_EPS) return y < v.y;
        return z < v.z - MERGE_EPS;
    }
};

struct Face { int v[3]; };
struct Mesh { vector<Vec3> vertices; vector<Face> faces; };

struct AABB {
    Vec3 min_p, max_p;
    AABB() : min_p({1e18,1e18,1e18}), max_p({-1e18,-1e18,-1e18}) {}
    void add(const Vec3& p) {
        min_p.x = min(min_p.x, p.x); min_p.y = min(min_p.y, p.y); min_p.z = min(min_p.z, p.z);
        max_p.x = max(max_p.x, p.x); max_p.y = max(max_p.y, p.y); max_p.z = max(max_p.z, p.z);
    }
    bool intersects(const AABB& o) const {
        return min_p.x <= o.max_p.x+EPS && max_p.x >= o.min_p.x-EPS &&
               min_p.y <= o.max_p.y+EPS && max_p.y >= o.min_p.y-EPS &&
               min_p.z <= o.max_p.z+EPS && max_p.z >= o.min_p.z-EPS;
    }
};

struct BVHNode { AABB box; int left = -1, right = -1, face_idx = -1; };

void build_bvh(const Mesh& mesh, vector<int>& indices, int start, int end, vector<BVHNode>& nodes) {
    int node_idx = (int)nodes.size();
    nodes.push_back({});
    AABB box;
    for (int i = start; i < end; ++i) {
        const auto& f = mesh.faces[indices[i]];
        box.add(mesh.vertices[f.v[0]]); box.add(mesh.vertices[f.v[1]]); box.add(mesh.vertices[f.v[2]]);
    }
    nodes[node_idx].box = box;
    if (end - start == 1) { nodes[node_idx].face_idx = indices[start]; return; }
    int axis = 0; Vec3 d = box.max_p - box.min_p;
    if (d.y > d.x && d.y > d.z) axis = 1; else if (d.z > d.x && d.z > d.y) axis = 2;
    nth_element(indices.begin() + start, indices.begin() + (start + end) / 2, indices.begin() + end, [&](int a, int b) {
        auto c = [&](int idx) {
            const auto& f = mesh.faces[idx];
            if (axis == 0) return mesh.vertices[f.v[0]].x + mesh.vertices[f.v[1]].x + mesh.vertices[f.v[2]].x;
            if (axis == 1) return mesh.vertices[f.v[0]].y + mesh.vertices[f.v[1]].y + mesh.vertices[f.v[2]].y;
            return mesh.vertices[f.v[0]].z + mesh.vertices[f.v[1]].z + mesh.vertices[f.v[2]].z;
        };
        return c(a) < c(b);
    });
    int mid = (start + end) / 2;
    int left_idx = (int)nodes.size(); build_bvh(mesh, indices, start, mid, nodes);
    int right_idx = (int)nodes.size(); build_bvh(mesh, indices, mid, end, nodes);
    nodes[node_idx].left = left_idx; nodes[node_idx].right = right_idx;
}

bool is_inside(const Vec3& p, const Mesh& mesh, const vector<BVHNode>& nodes, int root) {
    if (nodes.empty()) return false;
    int count = 0; Vec3 dir = {1.0, 0.0123, 0.0456}; dir = dir / dir.norm();
    auto intersect = [&](auto self, int idx) -> void {
        const auto& node = nodes[idx];
        double t_min = -1e18, t_max = 1e18;
        for (int i = 0; i < 3; ++i) {
            double d = (i==0?dir.x:(i==1?dir.y:dir.z)), pc = (i==0?p.x:(i==1?p.y:p.z));
            double bmin = (i==0?node.box.min_p.x:(i==1?node.box.min_p.y:node.box.min_p.z));
            double bmax = (i==0?node.box.max_p.x:(i==1?node.box.max_p.y:node.box.max_p.z));
            if (abs(d)<1e-12) { if (pc<bmin-EPS || pc>bmax+EPS) return; }
            else { double t1=(bmin-pc)/d, t2=(bmax-pc)/d; if(t1>t2) swap(t1,t2); t_min=max(t_min,t1); t_max=min(t_max,t2); }
        }
        if (t_min > t_max + EPS || t_max < 0) return;
        if (node.face_idx != -1) {
            const auto& f = mesh.faces[node.face_idx];
            Vec3 v0 = mesh.vertices[f.v[0]], v1 = mesh.vertices[f.v[1]], v2 = mesh.vertices[f.v[2]];
            Vec3 e1 = v1 - v0, e2 = v2 - v0, h = dir.cross(e2); double a = e1.dot(h);
            if (abs(a) < 1e-12) return;
            double fi = 1.0/a; Vec3 s = p - v0; double u = fi*s.dot(h); if (u<0.0||u>1.0) return;
            Vec3 q = s.cross(e1); double v = fi*dir.dot(q); if (v<0.0||u+v>1.0) return;
            if (fi*e2.dot(q) > EPS) count++; return;
        }
        self(self, node.left); self(self, node.right);
    };
    intersect(intersect, root); return (count % 2 != 0);
}

bool tri_tri(const Vec3& p1, const Vec3& q1, const Vec3& r1, const Vec3& p2, const Vec3& q2, const Vec3& r2) {
    Vec3 n1 = (q1-p1).cross(r1-p1); double d1 = n1.dot(p1);
    double dp2 = n1.dot(p2)-d1, dq2 = n1.dot(q2)-d1, dr2 = n1.dot(r2)-d1;
    if (abs(dp2)<EPS) dp2=0; if (abs(dq2)<EPS) dq2=0; if (abs(dr2)<EPS) dr2=0;
    if ((dp2>0&&dq2>0&&dr2>0)||(dp2<0&&dq2<0&&dr2<0)) return false;
    Vec3 n2 = (q2-p2).cross(r2-p2); double d2 = n2.dot(p2);
    double dp1 = n2.dot(p1)-d2, dq1 = n2.dot(q1)-d2, dr1 = n2.dot(r1)-d2;
    if (abs(dp1)<EPS) dp1=0; if (abs(dq1)<EPS) dq1=0; if (abs(dr1)<EPS) dr1=0;
    if ((dp1>0&&dq1>0&&dr1>0)||(dp1<0&&dq1<0&&dr1<0)) return false;
    Vec3 D = n1.cross(n2); if (D.normSq()<1e-20) return false;
    auto get_i = [&](const Vec3& p, const Vec3& q, const Vec3& r, double dp, double dq, double dr, double& t1, double& t2) {
        auto inter = [&](const Vec3& a, const Vec3& b, double da, double db) { return a + (b-a)*(da/(da-db)); };
        vector<Vec3> pts;
        if (dp*dq < 0) pts.push_back(inter(p,q,dp,dq)); if (dq*dr < 0) pts.push_back(inter(q,r,dq,dr)); if (dr*dp < 0) pts.push_back(inter(r,p,dr,dp));
        if (dp==0) pts.push_back(p); if (dq==0) pts.push_back(q); if (dr==0) pts.push_back(r);
        if (pts.empty()) return false;
        t1=1e18; t2=-1e18; for(const auto& pt : pts) { double t = pt.dot(D); t1=min(t1,t); t2=max(t2,t); }
        return true;
    };
    double t1s, t1e, t2s, t2e;
    if (!get_i(p1,q1,r1,dp1,dq1,dr1,t1s,t1e)) return false;
    if (!get_i(p2,q2,r2,dp2,dq2,dr2,t2s,t2e)) return false;
    return max(t1s, t2s) <= min(t1e, t2e) - EPS;
}

int main() {
    ios::sync_with_stdio(false); cin.tie(0);
    int VA, FA, VB, FB; if (!(cin >> VA >> FA >> VB >> FB)) return 0;
    Mesh A, B; A.vertices.resize(VA); for(int i=0; i<VA; ++i) cin >> A.vertices[i].x >> A.vertices[i].y >> A.vertices[i].z;
    A.faces.resize(FA); for(int i=0; i<FA; ++i) cin >> A.faces[i].v[0] >> A.faces[i].v[1] >> A.faces[i].v[2];
    B.vertices.resize(VB); for(int i=0; i<VB; ++i) cin >> B.vertices[i].x >> B.vertices[i].y >> B.vertices[i].z;
    B.faces.resize(FB); for(int i=0; i<FB; ++i) cin >> B.faces[i].v[0] >> B.faces[i].v[1] >> B.faces[i].v[2];

    vector<int> indA(FA), indB(FB); for(int i=0; i<FA; ++i) indA[i]=i; for(int i=0; i<FB; ++i) indB[i]=i;
    vector<BVHNode> nodesA, nodesB;
    if (FA > 0) { nodesA.reserve(2*FA); build_bvh(A, indA, 0, FA, nodesA); }
    if (FB > 0) { nodesB.reserve(2*FB); build_bvh(B, indB, 0, FB, nodesB); }

    vector<bool> dirtyA(FA, false), dirtyB(FB, false);
    auto find_inter = [&](auto self, int iA, int iB) -> void {
        if (nodesA.empty() || nodesB.empty()) return;
        if (!nodesA[iA].box.intersects(nodesB[iB].box)) return;
        if (nodesA[iA].face_idx != -1 && nodesB[iB].face_idx != -1) {
            const auto& fA = A.faces[nodesA[iA].face_idx]; const auto& fB = B.faces[nodesB[iB].face_idx];
            if (tri_tri(A.vertices[fA.v[0]], A.vertices[fA.v[1]], A.vertices[fA.v[2]], B.vertices[fB.v[0]], B.vertices[fB.v[1]], B.vertices[fB.v[2]])) {
                dirtyA[nodesA[iA].face_idx] = true; dirtyB[nodesB[iB].face_idx] = true;
            }
            return;
        }
        if (nodesA[iA].face_idx != -1) { self(self, iA, nodesB[iB].left); self(self, iA, nodesB[iB].right); }
        else if (nodesB[iB].face_idx != -1) { self(self, nodesA[iA].left, iB); self(self, nodesA[iA].right, iB); }
        else { self(self, nodesA[iA].left, nodesB[iB].left); self(self, nodesA[iA].left, nodesB[iB].right); self(self, nodesA[iA].right, nodesB[iB].left); self(self, nodesA[iA].right, nodesB[iB].right); }
    };
    if (FA > 0 && FB > 0) find_inter(find_inter, 0, 0);

    vector<Vec3> outV; vector<Face> outF; map<Vec3, int> vMap;
    auto getV = [&](const Vec3& v) {
        auto it = vMap.lower_bound(v); if (it != vMap.end() && !(v < it->first)) return it->second;
        int idx = (int)outV.size(); outV.push_back(v); vMap[v] = idx; return idx;
    };

    auto process = [&](const Mesh& m1, const Mesh& m2, const vector<bool>& dirty, const vector<BVHNode>& nodes2) {
        if (m1.faces.empty()) return;
        for(int i=0; i<(int)m1.faces.size(); ++i) {
            const auto& f = m1.faces[i];
            if (dirty[i]) {
                auto subdiv = [&](auto self, Vec3 v0, Vec3 v1, Vec3 v2, int lvl) -> void {
                    if (lvl == 0) {
                        Vec3 center = (v0+v1+v2)/3.0;
                        Vec3 normal = (v1-v0).cross(v2-v0);
                        if (normal.normSq() > 1e-20) center = center + (normal/normal.norm()) * 1e-9;
                        if (!is_inside(center, m2, nodes2, 0)) {
                            int iv0 = getV(v0), iv1 = getV(v1), iv2 = getV(v2);
                            if (iv0 != iv1 && iv1 != iv2 && iv2 != iv0) outF.push_back({iv0, iv1, iv2});
                        }
                        return;
                    }
                    Vec3 v01=(v0+v1)*0.5, v12=(v1+v2)*0.5, v20=(v2+v0)*0.5;
                    self(self, v0, v01, v20, lvl-1); self(self, v1, v12, v01, lvl-1); self(self, v2, v20, v12, lvl-1); self(self, v01, v12, v20, lvl-1);
                };
                subdiv(subdiv, m1.vertices[f.v[0]], m1.vertices[f.v[1]], m1.vertices[f.v[2]], 3);
            } else {
                Vec3 center = (m1.vertices[f.v[0]]+m1.vertices[f.v[1]]+m1.vertices[f.v[2]])/3.0;
                Vec3 normal = (m1.vertices[f.v[1]]-m1.vertices[f.v[0]]).cross(m1.vertices[f.v[2]]-m1.vertices[f.v[0]]);
                if (normal.normSq() > 1e-20) center = center + (normal/normal.norm()) * 1e-9;
                if (!is_inside(center, m2, nodes2, 0)) {
                    int iv0 = getV(m1.vertices[f.v[0]]), iv1 = getV(m1.vertices[f.v[1]]), iv2 = getV(m1.vertices[f.v[2]]);
                    if (iv0 != iv1 && iv1 != iv2 && iv2 != iv0) outF.push_back({iv0, iv1, iv2});
                }
            }
        }
    };
    process(A, B, dirtyA, nodesB);
    process(B, A, dirtyB, nodesA);

    // Final face cleanup (remove duplicates)
    for(auto& f : outF) { if(f.v[0]>f.v[1]) swap(f.v[0],f.v[1]); if(f.v[1]>f.v[2]) swap(f.v[1],f.v[2]); if(f.v[0]>f.v[1]) swap(f.v[0],f.v[1]); }
    sort(outF.begin(), outF.end(), [](const Face& a, const Face& b){ if(a.v[0]!=b.v[0]) return a.v[0]<b.v[0]; if(a.v[1]!=b.v[1]) return a.v[1]<b.v[1]; return a.v[2]<b.v[2]; });
    outF.erase(unique(outF.begin(), outF.end(), [](const Face& a, const Face& b){ return a.v[0]==b.v[0] && a.v[1]==b.v[1] && a.v[2]==b.v[2]; }), outF.end());

    cout << outV.size() << " " << outF.size() << endl;
    for(const auto& v : outV) printf("%.10f %.10f %.10f\n", v.x, v.y, v.z);
    for(const auto& f : outF) printf("%d %d %d\n", f.v[0], f.v[1], f.v[2]);
    return 0;
}
```
