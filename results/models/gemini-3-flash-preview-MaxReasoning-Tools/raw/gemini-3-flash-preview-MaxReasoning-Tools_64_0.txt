"use strict";

declare const require: any;
const fs = require('fs');

function solve() {
    let input: string;
    try {
        input = fs.readFileSync(0, 'utf8');
    } catch (e) {
        return;
    }
    if (!input) return;

    let data;
    try {
        data = JSON.parse(input);
    } catch (e) {
        return;
    }

    const probability: number[][] = data.probability;
    const drones: { id: number, x: number, y: number }[] = data.drones;
    const maxStepsPerDrone: number = data.maxStepsPerDrone;

    if (!probability || !probability.length || !drones) return;

    const rows = probability.length;
    const cols = probability[0].length;

    const covered = new Uint8Array(rows * cols);
    const visited = new Int32Array(rows * cols);
    let visitedToken = 0;

    const paths: [number, number][][] = drones.map(d => [[d.x, d.y]]);

    for (const d of drones) {
        if (d.y >= 0 && d.y < rows && d.x >= 0 && d.x < cols) {
            covered[d.y * cols + d.x] = 1;
        }
    }

    const currentPositions = drones.map(d => ({ x: d.x, y: d.y }));

    const mx = [0, 0, 1, -1];
    const my = [1, -1, 0, 0];

    // BFS Queue structures
    const qX = new Int32Array(5000);
    const qY = new Int32Array(5000);
    const qD = new Int32Array(5000);
    const qFX = new Int32Array(5000);
    const qFY = new Int32Array(5000);

    function findBestMoveBFS(sx: number, sy: number, maxDist: number) {
        visitedToken++;
        let qHead = 0;
        let qTail = 0;

        visited[sy * cols + sx] = visitedToken;

        for (let m = 0; m < 4; m++) {
            const nx = sx + mx[m];
            const ny = sy + my[m];
            if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                const idx = ny * cols + nx;
                if (!covered[idx] && probability[ny][nx] > 0) return { x: nx, y: ny };
                visited[idx] = visitedToken;
                qX[qTail] = nx; qY[qTail] = ny; qD[qTail] = 1; qFX[qTail] = nx; qFY[qTail] = ny;
                qTail++;
            }
        }

        while (qHead < qTail) {
            const x = qX[qHead];
            const y = qY[qHead];
            const d = qD[qHead];
            const fx = qFX[qHead];
            const fy = qFY[qHead];
            qHead++;

            if (d >= maxDist) continue;

            for (let m = 0; m < 4; m++) {
                const nx = x + mx[m];
                const ny = y + my[m];
                if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                    const idx = ny * cols + nx;
                    if (visited[idx] !== visitedToken) {
                        if (!covered[idx] && probability[ny][nx] > 0) {
                            return { x: fx, y: fy };
                        }
                        visited[idx] = visitedToken;
                        if (qTail < qX.length) {
                            qX[qTail] = nx; qY[qTail] = ny; qD[qTail] = d + 1; qFX[qTail] = fx; qFY[qTail] = fy;
                            qTail++;
                        }
                    }
                }
            }
        }
        return null;
    }

    for (let step = 0; step < maxStepsPerDrone; step++) {
        for (let i = 0; i < drones.length; i++) {
            const cp = currentPositions[i];

            let bestMove = { x: cp.x, y: cp.y };
            let foundUncovered = false;
            let bestUncoveredProb = -1;

            // Check neighbors greedily
            for (let m = 0; m < 4; m++) {
                const nx = cp.x + mx[m];
                const ny = cp.y + my[m];

                if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                    const idx = ny * cols + nx;
                    if (!covered[idx]) {
                        if (probability[ny][nx] > bestUncoveredProb) {
                            bestUncoveredProb = probability[ny][nx];
                            bestMove = { x: nx, y: ny };
                            foundUncovered = true;
                        }
                    }
                }
            }

            if (!foundUncovered) {
                const bfsMove = findBestMoveBFS(cp.x, cp.y, 40);
                if (bfsMove) {
                    bestMove = bfsMove;
                } else {
                    let maxP = -1;
                    if (probability[cp.y][cp.x] !== -1) {
                        maxP = probability[cp.y][cp.x];
                        bestMove = { x: cp.x, y: cp.y };
                    }
                    for (let m = 0; m < 4; m++) {
                        const nx = cp.x + mx[m];
                        const ny = cp.y + my[m];
                        if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && probability[ny][nx] !== -1) {
                            if (probability[ny][nx] > maxP) {
                                maxP = probability[ny][nx];
                                bestMove = { x: nx, y: ny };
                            }
                        }
                    }
                }
            }

            currentPositions[i] = bestMove;
            covered[bestMove.y * cols + bestMove.x] = 1;
            paths[i].push([bestMove.x, bestMove.y]);
        }
    }

    process.stdout.write(JSON.stringify({ paths }));
}

solve();
