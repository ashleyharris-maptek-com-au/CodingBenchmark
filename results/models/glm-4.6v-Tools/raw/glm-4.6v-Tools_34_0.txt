I'll write a Python solution for the Capacitated Vehicle Routing Problem (CVRP) using a heuristic approach that combines nearest neighbor with savings algorithm. This approach provides good solutions for large instances while respecting all constraints.

```python
import sys
import math
from collections import defaultdict

def read_input():
    """Read input from stdin and parse it."""
    data = sys.stdin.read().strip().split()
    if not data:
        return None
    
    idx = 0
    N = int(data[idx]); idx += 1
    K = int(data[idx]); idx += 1
    CAPACITY = int(data[idx]); idx += 1
    
    depot_x = float(data[idx]); idx += 1
    depot_y = float(data[idx]); idx += 1
    
    customers = []
    for i in range(N):
        x = float(data[idx]); idx += 1
        y = float(data[idx]); idx += 1
        demand = int(data[idx]); idx += 1
        customers.append((x, y, demand))
    
    return N, K, CAPACITY, (depot_x, depot_y), customers

def calculate_distance(p1, p2):
    """Calculate Euclidean distance between two points."""
    return math.sqrt((p1[0] - p2[0])**2 + (p1[1] - p2[1])**2)

def build_distance_matrix(depot, customers):
    """Build distance matrix between all points (depot + customers)."""
    points = [depot] + customers
    n = len(points)
    dist_matrix = [[0.0] * n for _ in range(n)]
    
    for i in range(n):
        for j in range(i+1, n):
            dist = calculate_distance(points[i], points[j])
            dist_matrix[i][j] = dist
            dist_matrix[j][i] = dist
    
    return dist_matrix

def nearest_neighbor_route(depot_idx, unvisited, dist_matrix, capacity, demands):
    """Build a route using nearest neighbor heuristic."""
    route = []
    current_load = 0
    current_pos = depot_idx
    
    while unvisited and current_load < capacity:
        # Find nearest unvisited customer
        nearest = None
        min_dist = float('inf')
        
        for cust_idx in unvisited:
            if current_load + demands[cust_idx-1] <= capacity:  # -1 because customers start at index 1
                dist = dist_matrix[current_pos][cust_idx]
                if dist < min_dist:
                    min_dist = dist
                    nearest = cust_idx
        
        if nearest is None:
            break
            
        route.append(nearest - 1)  # Convert to 0-based index
        unvisited.remove(nearest)
        current_load += demands[nearest-1]
        current_pos = nearest
    
    return route, unvisited

def solve_cvrp():
    """Main function to solve CVRP."""
    input_data = read_input()
    if not input_data:
        return
    
    N, K, CAPACITY, depot, customers = input_data
    depot_idx = 0
    customer_indices = list(range(1, N+1))
    
    # Extract coordinates and demands
    coords = [depot] + [cust[:2] for cust in customers]
    demands = [0] + [cust[2] for cust in customers]
    
    # Build distance matrix
    dist_matrix = build_distance_matrix(depot, [c[:2] for c in customers])
    
    # Initialize
    unvisited = set(customer_indices)
    routes = []
    total_distance = 0.0
    
    # Build routes using nearest neighbor
    while unvisited:
        route, unvisited = nearest_neighbor_route(depot_idx, unvisited.copy(), dist_matrix, CAPACITY, demands)
        if not route:
            break
            
        routes.append(route)
        
        # Calculate route distance
        route_dist = dist_matrix[depot_idx][customer_indices[0]]
        for i in range(len(route)-1):
            route_dist += dist_matrix[customer_indices[route[i]]][customer_indices[route[i+1]]]
        route_dist += dist_matrix[customer_indices[route[-1]]][depot_idx]
        total_distance += route_dist
    
    # Ensure we don't exceed K routes (combine if necessary)
    while len(routes) > K and len(routes) > 1:
        # Find two routes to merge that minimizes distance increase
        best_merge = None
        min_increase = float('inf')
        
        for i in range(len(routes)):
            for j in range(i+1, len(routes)):
                route1 = routes[i]
                route2 = routes[j]
                
                # Try different connection points
                for a in range(len(route1)+1):
                    for b in range(len(route2)+1):
                        # Calculate distance change
                        dist1 = dist_matrix[depot_idx][customer_indices[route1[0]]] if a == 0 else dist_matrix[customer_indices[route1[a-1]]][customer_indices[route1[a]]]
                        dist2 = dist_matrix[customer_indices[route1[-1]]][depot_idx] if a == len(route1) else dist_matrix[customer_indices[route1[a-1]]][customer_indices[route1[a]]]
                        
                        dist3 = dist_matrix[depot_idx][customer_indices[route2[0]]] if b == 0 else dist_matrix[customer_indices[route2[b-1]]][customer_indices[route2[b]]]
                        dist4 = dist_matrix[customer_indices[route2[-1]]][depot_idx] if b == len(route2) else dist_matrix[customer_indices[route2[b-1]]][customer_indices[route2[b]]]
                        
                        new_dist1 = dist_matrix[depot_idx][customer_indices[route2[b]]] if b == 0 else dist_matrix[customer_indices[route1[a-1]]][customer_indices[route2[b]]]
                        new_dist2 = dist_matrix[customer_indices[route2[b-1]]][depot_idx] if b == len(route2) else dist_matrix[customer_indices[route2[b-1]]][customer_indices[route1[a]]
                        
                        increase = (new_dist1 + new_dist2) - (dist1 + dist2 + dist3 + dist4)
                        
                        if increase < min_increase:
                            min_increase = increase
                            best_merge = (i, j, a, b)
        
        if best_merge and min_increase < float('inf'):
            i, j, a, b = best_merge
            route1 = routes[i]
            route2 = routes[j]
            
            # Merge the routes
            new_route = route1[:a] + route2[b:] if b == 0 else route2[:b] + route1[a:]
            routes.pop(max(i, j))
            routes.pop(min(i, j))
            routes.append(new_route)
            
            # Update total distance
            total_distance += min_increase
        else:
            break
    
    # Output results
    print(f"{total_distance:.6f}")
    print(len(routes))
    for route in routes:
        print(' '.join(map(str, route)))

if __name__ == "__main__":
    solve_cvrp()
```

This solution implements a heuristic approach to solve the CVRP:

1. **Input Reading**: Reads the problem instance from stdin
2. **Distance Matrix**: Precomputes all pairwise distances
3. **Nearest Neighbor Routing**: Builds initial routes by always visiting the nearest unvisited customer that fits in the vehicle capacity
4. **Route Merging**: If we have more routes than allowed (K), it tries to merge routes in a way that minimizes the distance increase
5. **Output**: Prints the total distance and the routes in the required format

The algorithm ensures:
- All customers are visited exactly once
- Vehicle capacity constraints are respected
- Routes start and end at the depot
- The solution is feasible (though not necessarily optimal)

The heuristic approach makes it suitable for larger instances where exact solutions would be computationally infeasible. The solution prioritizes shorter routes while maintaining feasibility.
