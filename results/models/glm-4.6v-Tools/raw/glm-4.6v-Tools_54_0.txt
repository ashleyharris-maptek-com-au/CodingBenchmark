import math

# Module-level variables for persistence
class AutopilotState:
    def __init__(self):
        self.phase = "intercept"  # intercept, approach, flare, rollout
        self.go_around_count = 0
        self.last_radio_alt = 0
        self.flaps_configured = False
        self.gear_down = False
        self.speedbrake_deployed = False
        self.spoilers_deployed = False
        self.reverse_deployed = False
        self.autobrake_active = False

        self.rudder_trim = 0.0
        self.last_glideslope = 0.0
        self.last_localizer = 0.0
        self.last_ias = 0.0
        self.target_ias = 0.0
        self.vref = 0.0
        self.decision_height = 200  # CAT I
        self.flare_start_alt = 50
        self.flare_end_alt = 5
        self.go_around_altitude = 1500
        self.go_around_pitch = 15
        self.stabilized = False
        self.unstable_reason = ""
        self.engine_failure = False
        self.elevator_jammed = False
        self.aileron_jammed = False
        self.aileron_jam_angle = 0
        self.glideslope_valid = [True, True, True]
        self.localizer_valid = [True, True, True]
        self.radar_alt_valid = True
        self.last_touchdown_vspeed = 0
        self.runway_remaining = 100

def autopilot_step(state: dict, dt: float) -> dict:
    # Initialize or reset state
    if not hasattr(autopilot_step, "autopilot_state"):
        autopilot_step.autopilot_state = AutopilotState()
    
    aps = autopilot_step.autopilot_state
    
    # Extract key parameters
    radio_alt = state.get('radio_alt_ft', 0)
    ias = get_valid_ias(state)
    target_ias = state.get('target_ias_kt', 137)
    target_vref = state.get('target_vref_kt', 137)
    target_vs = state.get('target_vs_fpm', -700)
    glideslope = get_valid_ils_value(state, 'glideslope')
    localizer = get_valid_ils_value(state, 'localizer')
    on_ground = state.get('on_ground', False)
    weight_on_wheels = state.get('weight_on_wheels', False)
    below_dh = state.get('below_decision_height', False)
    dme = get_valid_dme(state)
    groundspeed = state.get('groundspeed_kt', 0)
    
    # Check for engine failure
    aps.engine_failure = (state.get('engine_1_running', 1) == 0 or 
                        state.get('engine_2_running', 1) == 0)
    
    # Check for control surface failures
    aps.elevator_jammed = (state.get('elevator_jammed', False) if 'elevator_jammed' in state else False)
    aps.aileron_jammed = (state.get('aileron_jammed', False) if 'aileron_jammed' in state else False)
    aps.aileron_jam_angle = state.get('aileron_jam_angle', 0) if 'aileron_jam_angle' in state else 0
    
    # Phase management
    if aps.phase == "intercept":
        handle_intercept_phase(state, aps, radio_alt, ias, target_ias, glideslope, localizer, dme)
    elif aps.phase == "approach":
        handle_approach_phase(state, aps, radio_alt, ias, target_ias, target_vs, glideslope, localizer, below_dh)
    elif aps.phase == "flare":
        handle_flare_phase(state, aps, radio_alt, ias, target_ias, glideslope)
    elif aps.phase == "rollout":
        handle_rollout_phase(state, aps, radio_alt, ias, groundspeed, on_ground, weight_on_wheels)
    
    # Go-around logic
    if should_go_around(state, aps, radio_alt, ias, glideslope, localizer, on_ground):
        return initiate_go_around(state, aps)
    
    # Generate control outputs
    controls = generate_controls(state, aps, radio_alt, ias, target_ias, glideslope, localizer, on_ground)
    
    # Update state tracking
    aps.last_radio_alt = radio_alt
    aps.last_ias = ias
    aps.last_glideslope = glideslope
    aps.last_localizer = localizer
    
    return controls

def get_valid_ias(state: dict) -> float:
    """Get valid indicated airspeed from redundant sensors"""
    for i in range(1, 4):
        key = f'ias_{i}_kt'
        valid_key = f'ias_{i}_valid'
        if state.get(valid_key, False):
            return state.get(key, 0)
    return state.get('target_ias_kt', 137)

def get_valid_ils_value(state: dict, type: str) -> float:
    """Get valid ILS deviation from redundant receivers"""
    values = []
    for i in range(1, 4):
        dots_key = f'{type}_{i}_dots'
        valid_key = f'{type}_{i}_valid'
        if state.get(valid_key, False):
            values.append(state.get(dots_key, 0))
    
    if not values:
        return 0.0
    
    # Use median value for robustness
    values.sort()
    return values[len(values) // 2]

def get_valid_dme(state: dict) -> float:
    """Get valid DME distance"""
    dme1 = state.get('dme_1_nm', 999)
    dme2 = state.get('dme_2_nm', 999)
    return min(dme1, dme2) if dme1 < 999 and dme2 < 999 else max(dme1, dme2)

def handle_intercept_phase(state: dict, aps: AutopilotState, radio_alt: float, ias: float, 
                        target_ias: float, glideslope: float, localizer: float, dme: float):
    """Handle intercept phase (>1000ft)"""
    # Configure for landing
    if not aps.flaps_configured and dme > 5:
        aps.flaps_configured = True
        aps.target_ias = target_ias
    
    # Check if we should transition to approach
    if radio_alt < 1000:
        aps.phase = "approach"
        aps.stabilized = False

def handle_approach_phase(state: dict, aps: AutopilotState, radio_alt: float, ias: float, 
                       target_ias: float, target_vs: float, glideslope: float, localizer: float, below_dh: bool):
    """Handle approach phase (1000ft-200ft)"""
    # Check stabilization criteria
    if not aps.stabilized:
        speed_error = abs(ias - target_ias) / target_ias
        gs_error = abs(glideslope)
        loc_error = abs(localizer)
        
        if speed_error < 0.05 and gs_error < 0.5 and loc_error < 0.5 and radio_alt < 800:
            aps.stabilized = True
        elif radio_alt < 500:
            aps.unstable_reason = "unstabilized_approach"
            return
    
    # Transition to flare
    if radio_alt < 50:
        aps.phase = "flare"

def handle_flare_phase(state: dict, aps: AutopilotState, radio_alt: float, ias: float, 
                     target_ias: float, glideslope: float):
    """Handle flare phase (<50ft radio alt)"""
    # Gradually reduce descent rate
    if radio_alt > 20:
        # Continue approach speed
        pass
    else:
        # Idle thrust and prepare for touchdown
        if not aps.spoilers_deployed and radio_alt < 10:
            aps.spoilers_deployed = True

def handle_rollout_phase(state: dict, aps: AutopilotState, radio_alt: float, ias: float, 
                      groundspeed: float, on_ground: bool, weight_on_wheels: bool):
    """Handle rollout phase after touchdown"""
    if not aps.reverse_deployed and weight_on_wheels:
        aps.reverse_deployed = True
        aps.autobrake_active = True
    
    # Stop when slow enough or runway end reached
    if groundspeed < 50:
        aps.autobrake_active = False

def should_go_around(state: dict, aps: AutopilotState, radio_alt: float, ias: float, 
                   glideslope: float, localizer: float, on_ground: bool) -> bool:
    """Determine if go-around is required"""
    # Already in go-around
    if aps.phase == "go_around":
        return True
    
    # Below decision height and not stabilized
    if radio_alt < aps.decision_height and not aps.stabilized:
        aps.unstable_reason = "below_dh_unstabilized"
        return True
    
    # Excessive deviations
    if abs(glideslope) > 2.0 or abs(localizer) > 3.0:
        aps.unstable_reason = "excessive_deviation"
        return True
    
    # Engine failure during critical phase
    if aps.engine_failure and radio_alt < 500:
        aps.unstable_reason = "engine_failure"
        return True
    
    # Control surface failures
    if aps.elevator_jammed or aps.aileron_jammed:
        aps.unstable_reason = "control_failure"
        return True
    
    # Touchdown criteria (should not be in flare phase when on ground)
    if on_ground and aps.phase == "flare":
        aps.unstable_reason = "touchdown_too_high"
        return True
    
    return False

def initiate_go_around(state: dict, aps: AutopilotState) -> dict:
    """Initiate go-around procedure"""
    aps.go_around_count += 1
    aps.phase = "go_around"
    
    # Reset for new approach
    aps.flaps_configured = False
    aps.gear_down = False
    aps.speedbrake_deployed = False
    aps.spoilers_deployed = False
    aps.reverse_deployed = False
    aps.autobrake_active = False
    aps.stabilized = False
    
    # Go-around commands
    return {
        'elevator': 0.5,  # Pitch up
        'throttle_1': 1.0,  # TOGA thrust
        'throttle_2': 1.0,
        'speedbrake': 0.0,  # Retract
        'flaps': 15,  # Intermediate flap setting
        'gear': 0.0,  # Retract gear
        'spoiler': 0.0,
        'reverse_1': 0.0,
        'reverse_2': 0.0,
        'autobrake': 0.0
    }

def generate_controls(state: dict, aps: AutopilotState, radio_alt: float, ias: float, 
                   target_ias: float, glideslope: float, localizer: float, on_ground: bool) -> dict:
    """Generate control outputs based on current phase"""
    controls = {}
    
    if aps.phase == "go_around":
        # Climb to 1500ft AGL
        if radio_alt < aps.go_around_altitude:
            controls['elevator'] = 0.5  # Maintain climb attitude
            controls['throttle_1'] = 1.0
            controls['throttle_2'] = 1.0
            controls['speedbrake'] = 0.0
        else:
            # Return to approach phase
            aps.phase = "intercept"
            return controls
    
    # Basic flight control gains
    pitch_gain = 0.15
    roll_gain = 0.08
    yaw_gain = 0.05
    speed_gain = 0.02
    
    # Pitch control (elevator)
    if aps.phase == "intercept" or aps.phase == "approach":
        # Track glideslope
        pitch_command = -glideslope * pitch_gain * 3  # 3 dots = 3 degrees
        # Speed control
        speed_error = target_ias - ias
        pitch_command += speed_error * speed_gain
        
        controls['elevator'] = pitch_command
    
    elif aps.phase == "flare":
        # Reduce descent rate
        if radio_alt > 20:
            controls['elevator'] = 0.2  # Gentle pitch up
        else:
            controls['elevator'] = 0.4  # Stronger pitch up for flare
            controls['throttle_1'] = 0.0  # Idle thrust
            controls['throttle_2'] = 0.0
    
    elif aps.phase == "rollout":
        if not on_ground:
            controls['elevator'] = 0.0  # Level pitch
        else:
            controls['elevator'] = -0.1  # Nose down for rollout
    
    # Roll control (aileron)
    if aps.phase in ["intercept", "approach", "flare"]:
        roll_command = -localizer * roll_gain * 3  # 3 dots = 3 degrees
        
        # Apply aileron jam if present
        if aps.aileron_jammed:
            roll_command += aps.aileron_jam_angle * 0.05
        
        controls['aileron'] = roll_command
    
    # Yaw control (rudder)
    if aps.phase in ["intercept", "approach", "flare"]:
        # Crosswind correction
        crosswind = localizer * 0.3  # Approximate crosswind from localizer
        controls['rudder'] = -crosswind * yaw_gain
    
    # Throttle control
    if aps.phase == "intercept":
        # Maintain approach speed
        speed_error = target_ias - ias
        throttle = 0.5 + speed_error * 0.01
        controls['throttle_1'] = max(0.2, min(0.8, throttle))
        controls['throttle_2'] = controls['throttle_1']
    
    elif aps.phase == "approach":
        # Fine tune speed
        speed_error = target_ias - ias
        throttle = 0.5 + speed_error * 0.005
        controls['throttle_1'] = max(0.3, min(0.7, throttle))
        controls['throttle_2'] = controls['throttle_1']
    
    elif aps.phase == "flare":
        # Idle thrust
        controls['throttle_1'] = 0.0
        controls['throttle_2'] = 0.0
    
    elif aps.phase == "rollout":
        if aps.reverse_deployed:
            # Apply reverse thrust
            controls['throttle_1'] = 0.5
            controls['throttle_2'] = 0.5
            controls['reverse_1'] = 0.5
            controls['reverse_2'] = 0.5
        else:
            controls['throttle_1'] = 0.1
            controls['throttle_2'] = 0.1
    
    # Configuration controls
    if aps.phase == "intercept" and radio_alt > 1500:
        # Configure for approach
        controls['flaps'] = 5
        controls['gear'] = 0.0
        controls['speedbrake'] = 0.0
    
    elif aps.phase == "intercept" and radio_alt < 1500:
        # Extend gear and flaps
        controls['flaps'] = 15
        controls['gear'] = 1.0
        controls['speedbrake'] = 0.0
    
    elif aps.phase == "approach" and radio_alt < 500:
        # Full landing configuration
        controls['flaps'] = 40
        controls['gear'] = 1.0
        controls['speedbrake'] = 0.0
    
    # Spoilers and autobrake
    if aps.spoilers_deployed:
        controls['spoiler'] = 1.0
    
    if aps.autobrake_active:
        controls['autobrake'] = 0.7  # Moderate braking
    
    # Apply elevator trim for long-term stability
    if 'elevator' in controls:
        aps.elevator_trim += controls['elevator'] * 0.001
        aps.elevator_trim = max(-1.0, min(1.0, aps.elevator_trim))
        controls['elevator_trim'] = aps.elevator_trim
    
    return controls
