```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate diffuse lighting (Lambertian)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    
    // Procedural brick pattern
    float2 uv = input.uv * float2(8.0, 16.0); // 8x16 scale
    float2 brick = floor(uv);
    float2 mortar = frac(uv);
    
    // Staggered rows (half-brick offset every other row)
    if (frac(brick.y * 0.5) > 0.5) {
        brick.x += 0.5;
    }
    
    // Mortar thickness: 0.05 in X, 0.1 in Y
    float2 mortarSize = float2(0.05, 0.1);
    bool isMortar = mortar.x < mortarSize.x || mortar.y < mortarSize.y;
    
    // Brick and mortar colors
    float3 brickColor = float3(0.7, 0.2, 0.1);
    float3 mortarColor = float3(0.7, 0.7, 0.7);
    float3 baseColor = isMortar ? mortarColor : brickColor;
    
    // Apply Lambertian shading factor
    float lightingFactor = 0.4 + 0.6 * diffuse;
    float3 finalColor = baseColor * lightingFactor;
    
    return float4(finalColor, 1.0);
}
```
