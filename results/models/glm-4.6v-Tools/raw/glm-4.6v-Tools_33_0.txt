I'll write a C# solution for the Quadratic Assignment Problem (QAP) that uses a heuristic approach suitable for N up to around 2000. The solution will implement a simulated annealing algorithm with a greedy initialization, which provides a good balance between solution quality and computation time.

```csharp
using System;
using System.Diagnostics;
using System.IO;

class QAPSolver
{
    static void Main()
    {
        // Read input
        var input = Console.In.ReadToEnd().Trim().Split(new[] { ' ', '\n', '\r', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        int idx = 0;
        int N = int.Parse(input[idx++]);
        
        // Read flow matrix
        int[,] flow = new int[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                flow[i, j] = int.Parse(input[idx++]);
            }
        }
        
        // Read distance matrix
        int[,] dist = new int[N, N];
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                dist[i, j] = int.Parse(input[idx++]);
            }
        }
        
        // Solve QAP
        var result = SolveQAP(N, flow, dist);
        
        // Output result
        Console.WriteLine(result.cost);
        Console.WriteLine(string.Join(" ", result.permutation));
    }
    
    static (int cost, int[] permutation) SolveQAP(int N, int[,] flow, int[,] dist)
    {
        // Initialize with greedy solution
        int[] currentPermutation = GreedyInitialization(N, flow, dist);
        int currentCost = CalculateCost(currentPermutation, flow, dist);
        
        // Simulated annealing parameters
        double temperature = 1000.0;
        double coolingRate = 0.9995;
        int iterations = 50000;
        int noImprovementCount = 0;
        int maxNoImprovement = 10000;
        
        Random rand = new Random();
        
        for (int iter = 0; iter < iterations && noImprovementCount < maxNoImprovement; iter++)
        {
            // Generate neighbor by swapping two random elements
            int i = rand.Next(N);
            int j = rand.Next(N);
            while (j == i) j = rand.Next(N);
            
            // Create new permutation
            int[] newPermutation = (int[])currentPermutation.Clone();
            int temp = newPermutation[i];
            newPermutation[i] = newPermutation[j];
            newPermutation[j] = temp;
            
            // Calculate cost difference
            int delta = CalculateDelta(currentPermutation, newPermutation, flow, dist, i, j);
            int newCost = currentCost + delta;
            
            // Accept if better or with probability
            if (delta < 0 || rand.NextDouble() < Math.Exp(-delta / temperature))
            {
                currentPermutation = newPermutation;
                currentCost = newCost;
                noImprovementCount = 0;
            }
            else
            {
                noImprovementCount++;
            }
            
            // Cool down
            temperature *= coolingRate;
        }
        
        return (currentCost, currentPermutation);
    }
    
    static int[] GreedyInitialization(int N, int[,] flow, int[,] dist)
    {
        int[] permutation = new int[N];
        bool[] used = new bool[N];
        
        // Start with the first facility
        permutation[0] = 0;
        used[0] = true;
        
        // Assign remaining facilities greedily
        for (int i = 1; i < N; i++)
        {
            int bestLoc = -1;
            int bestCost = int.MaxValue;
            
            for (int loc = 0; loc < N; loc++)
            {
                if (!used[loc])
                {
                    permutation[i] = loc;
                    int cost = CalculateCost(permutation, flow, dist);
                    if (cost < bestCost)
                    {
                        bestCost = cost;
                        bestLoc = loc;
                    }
                }
            }
            
            permutation[i] = bestLoc;
            used[bestLoc] = true;
        }
        
        return permutation;
    }
    
    static int CalculateCost(int[] permutation, int[,] flow, int[,] dist)
    {
        int cost = 0;
        int N = permutation.Length;
        
        for (int i = 0; i < N; i++)
        {
            for (int j = 0; j < N; j++)
            {
                cost += flow[i, j] * dist[permutation[i], permutation[j]];
            }
        }
        
        return cost;
    }
    
    static int CalculateDelta(int[] oldPerm, int[] newPerm, int[,] flow, int[,] dist, int i, int j)
    {
        int delta = 0;
        int N = oldPerm.Length;
        
        // Calculate the difference when swapping elements i and j
        for (int k = 0; k < N; k++)
        {
            if (k != i && k != j)
            {
                int oldI = oldPerm[i];
                int oldJ = oldPerm[j];
                int newI = newPerm[i];
                int newJ = newPerm[j];
                
                delta += flow[i, k] * (dist[newI, newPerm[k]] - dist[oldI, oldPerm[k]]) +
                         flow[j, k] * (dist[newJ, newPerm[k]] - dist[oldJ, oldPerm[k]]) +
                         flow[k, i] * (dist[newPerm[k], newI] - dist[oldPerm[k], oldI]) +
                         flow[k, j] * (dist[newPerm[k], newJ] - dist[oldPerm[k], oldJ]);
            }
        }
        
        // Add the diagonal terms
        delta += flow[i, j] * (dist[newPerm[i], newPerm[j]] - dist[oldPerm[i], oldPerm[j]]) +
                 flow[j, i] * (dist[newPerm[j], newPerm[i]] - dist[oldPerm[j], oldPerm[i]]);
        
        return delta;
    }
}
```

This solution implements:

1. **Greedy Initialization**: Starts with a simple greedy algorithm to get a reasonable initial solution
2. **Simulated Annealing**: Uses a probabilistic technique to escape local minima and find better solutions
3. **Efficient Cost Calculation**: Includes an optimized delta calculation that only computes the cost difference when swapping two elements
4. **Scalability**: The algorithm is designed to handle N up to around 2000 within reasonable time (5 minutes)

The simulated annealing approach provides a good balance between solution quality and computation time, making it suitable for the problem constraints. The greedy initialization helps to start with a reasonable solution, and the annealing process refines it over time.
