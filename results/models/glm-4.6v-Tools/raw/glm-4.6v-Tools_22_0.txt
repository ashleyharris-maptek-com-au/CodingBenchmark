```csharp
using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.IO;
using System.Linq;
using System.Threading;

public class TetrisAI
{
    private const int BOARD_WIDTH = 10;
    private const int BOARD_HEIGHT = 20;
    private const int MAX_ROTATIONS = 4;
    private const int MAX_SEARCH_DEPTH = 2;
    private const int MAX_TIME_MS = 4800; // 4.8 seconds to be safe

    private int width;
    private int height;
    private int[] board;
    private int clearedLines = 0;
    private int totalPieces = 0;
    private int currentPiece = 0;
    private Stopwatch timer;
    private bool isMarathonGame = false;
    private bool useSimpleStrategy = false;

    private readonly Dictionary<char, int[,]> pieceTemplates = new Dictionary<char, int[,]>
    {
        // I piece (4x1)
        { 'I', new int[,] { { 1, 1, 1, 1 } } },
        // O piece (2x2)
        { 'O', new int[,] { { 1, 1 }, { 1, 1 } } },
        // T piece
        { 'T', new int[,] { { 0, 1, 0 }, { 1, 1, 1 } } },
        // S piece (right-leaning)
        { 'S', new int[,] { { 0, 1, 1 }, { 1, 1, 0 } } },
        // Z piece (left-leaning)
        { 'Z', new int[,] { { 1, 1, 0 }, { 0, 1, 1 } } },
        // J piece (left-leaning L)
        { 'J', new int[,] { { 1, 0, 0 }, { 1, 1, 1 } } },
        // L piece (right-leaning L)
        { 'L', new int[,] { { 0, 0, 1 }, { 1, 1, 1 } } }
    };

    public static void Main(string[] args)
    {
        var ai = new TetrisAI();
        ai.Run();
    }

    public void Run()
    {
        var input = Console.ReadLine().Split(' ');
        width = int.Parse(input[0]);
        height = int.Parse(input[1]);
        totalPieces = int.Parse(input[2]);

        // Initialize board
        board = new int[width * height];

        timer = Stopwatch.StartNew();

        // Determine if this is likely a marathon game
        isMarathonGame = totalPieces > 100;
        useSimpleStrategy = isMarathonGame || width > 15;

        for (currentPiece = 0; currentPiece < totalPieces; currentPiece++)
        {
            var pieceType = Console.ReadLine()[0];
            var bestMove = FindBestMove(pieceType);
            
            Console.WriteLine($"{bestMove.rotation} {bestMove.column}");
            Console.Out.Flush();

            var result = Console.ReadLine();
            if (result.StartsWith("gameover"))
            {
                break;
            }
            else if (result.StartsWith("cleared"))
            {
                clearedLines += int.Parse(result.Split('_')[1]);
            }
        }
    }

    private Move FindBestMove(char pieceType)
    {
        if (useSimpleStrategy)
        {
            return FindSimpleMove(pieceType);
        }

        int bestScore = int.MinValue;
        Move bestMove = new Move(0, 0);
        var piece = pieceTemplates[pieceType];

        // Try all rotations
        for (int rotation = 0; rotation < MAX_ROTATIONS; rotation++)
        {
            var rotatedPiece = RotatePiece(piece, rotation);
            int pieceWidth = rotatedPiece.GetLength(1);
            int pieceHeight = rotatedPiece.GetLength(0);

            // Try all possible columns
            for (int col = 0; col <= width - pieceWidth; col++)
            {
                // Check time limit
                if (timer.ElapsedMilliseconds > MAX_TIME_MS)
                {
                    return bestMove;
                }

                // Simulate placing the piece
                var testBoard = (int[])board.Clone();
                int landingHeight = SimulatePlacement(testBoard, rotatedPiece, col, out int linesCleared);

                // Evaluate the move
                int score = EvaluateMove(testBoard, landingHeight, linesCleared, pieceType);

                // Look ahead for better moves
                if (!useSimpleStrategy && linesCleared == 0 && timer.ElapsedMilliseconds < MAX_TIME_MS / 2)
                {
                    var lookaheadScore = LookAhead(testBoard, pieceType, rotation, col);
                    score += lookaheadScore;
                }

                if (score > bestScore)
                {
                    bestScore = score;
                    bestMove = new Move(rotation, col);
                }
            }
        }

        return bestMove;
    }

    private Move FindSimpleMove(char pieceType)
    {
        // Simple strategy for marathon games or wide boards
        var piece = pieceTemplates[pieceType];
        int bestScore = int.MinValue;
        Move bestMove = new Move(0, width / 2);

        for (int rotation = 0; rotation < 2; rotation++)
        {
            var rotatedPiece = RotatePiece(piece, rotation);
            int pieceWidth = rotatedPiece.GetLength(1);

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                var testBoard = (int[])board.Clone();
                int landingHeight = SimulatePlacement(testBoard, rotatedPiece, col, out int linesCleared);
                
                int score = EvaluateSimpleMove(testBoard, landingHeight, linesCleared);
                
                if (score > bestScore)
                {
                    bestScore = score;
                    bestMove = new Move(rotation, col);
                }
            }
        }

        return bestMove;
    }

    private int LookAhead(int[] boardState, char currentPieceType, int currentRotation, int currentColumn)
    {
        int lookaheadScore = 0;
        int lookaheadDepth = Math.Min(1, MAX_SEARCH_DEPTH - 1);

        // Simplified lookahead - just consider the next piece
        var nextPieceType = GetNextPieceType(currentPieceType);
        if (nextPieceType == '\0') return 0;

        var nextPiece = pieceTemplates[nextPieceType];
        int bestNextScore = int.MinValue;

        for (int rotation = 0; rotation < MAX_ROTATIONS; rotation++)
        {
            var rotatedPiece = RotatePiece(nextPiece, rotation);
            int pieceWidth = rotatedPiece.GetLength(1);

            for (int col = 0; col <= width - pieceWidth; col++)
            {
                var testBoard = (int[])boardState.Clone();
                int landingHeight = SimulatePlacement(testBoard, rotatedPiece, col, out int linesCleared);
                
                int score = EvaluateMove(testBoard, landingHeight, linesCleared, nextPieceType);
                if (score > bestNextScore)
                {
                    bestNextScore = score;
                }
            }
        }

        return bestNextScore / 2; // Weight lookahead less than current move
    }

    private char GetNextPieceType(char currentPieceType)
    {
        // Simple cycle through piece types
        var pieces = new char[] { 'I', 'O', 'T', 'S', 'Z', 'J', 'L' };
        int currentIndex = Array.IndexOf(pieces, currentPieceType);
        return currentIndex < pieces.Length - 1 ? pieces[currentIndex + 1] : pieces[0];
    }

    private int[,] RotatePiece(int[,] piece, int rotation)
    {
        int rows = piece.GetLength(0);
        int cols = piece.GetLength(1);

        if (rotation == 0) return piece;
        if (rotation == 2) return RotatePiece(RotatePiece(piece, 1), 1);

        var rotated = new int[cols, rows];
        for (int i = 0; i < rows; i++)
        {
            for (int j = 0; j < cols; j++)
            {
                rotated[j, rows - 1 - i] = piece[i, j];
            }
        }
        return rotated;
    }

    private int SimulatePlacement(int[] board, int[,] piece, int column, out int linesCleared)
    {
        int pieceHeight = piece.GetLength(0);
        int pieceWidth = piece.GetLength(1);

        // Find the lowest valid position
        int bestY = -1;
        for (int y = 0; y <= height - pieceHeight; y++)
        {
            bool valid = true;
            for (int dy = 0; dy < pieceHeight; dy++)
            {
                for (int dx = 0; dx < pieceWidth; dx++)
                {
                    if (piece[dy, dx] == 1 && board[(y + dy) * width + column + dx] != 0)
                    {
                        valid = false;
                        break;
                    }
                }
                if (!valid) break;
            }
            if (!valid) break;
            bestY = y;
        }

        if (bestY == -1)
        {
            linesCleared = 0;
            return height;
        }

        // Place the piece
        for (int dy = 0; dy < pieceHeight; dy++)
        {
            for (int dx = 0; dx < pieceWidth; dx++)
            {
                if (piece[dy, dx] == 1)
                {
                    board[(bestY + dy) * width + column + dx] = 1;
                }
            }
        }

        // Check for cleared lines
        linesCleared = 0;
        for (int y = 0; y < height; y++)
        {
            bool fullLine = true;
            for (int x = 0; x < width; x++)
            {
                if (board[y * width + x] == 0)
                {
                    fullLine = false;
                    break;
                }
            }
            if (fullLine)
            {
                linesCleared++;
                // Clear the line
                for (int x = 0; x < width; x++)
                {
                    board[y * width + x] = 0;
                }
                // Move lines down
                for (int moveY = y; moveY > 0; moveY--)
                {
                    for (int x = 0; x < width; x++)
                    {
                        board[moveY * width + x] = board[(moveY - 1) * width + x];
                    }
                }
                // Clear top row
                for (int x = 0; x < width; x++)
                {
                    board[x] = 0;
                }
            }
        }

        return bestY;
    }

    private int EvaluateMove(int[] board, int landingHeight, int linesCleared, char pieceType)
    {
        int score = 0;

        // Reward for clearing lines (most important)
        score += linesCleared * 1000;

        // Punish for high landing heights
        score -= landingHeight * 10;

        // Reward for keeping the board flat
        int maxHeight = GetMaxHeight(board);
        score -= maxHeight * 5;

        // Reward for creating holes
        int holes = CountHoles(board);
        score -= holes * 25;

        // Reward for creating opportunities for future clears
        int bumpiness = CalculateBumpiness(board);
        score -= bumpiness * 3;

        // Special handling for different piece types
        if (pieceType == 'I' || pieceType == 'O')
        {
            score += 20; // These pieces are more versatile
        }

        return score;
    }

    private int EvaluateSimpleMove(int[] board, int landingHeight, int linesCleared)
    {
        int score = linesCleared * 500 - landingHeight * 8 - CountHoles(board) * 20;
        return score;
    }

    private int GetMaxHeight(int[] board)
    {
        for (int y = 0; y < height; y++)
        {
            for (int x = 0; x < width; x++)
            {
                if (board[y * width + x] != 0)
                {
                    return height - y;
                }
            }
        }
        return 0;
    }

    private int CountHoles(int[] board)
    {
        int holes = 0;
        for (int x = 0; x < width; x++)
        {
            bool blockFound = false;
            for (int y = 0; y < height; y++)
            {
                if (board[y * width + x] != 0)
                {
                    blockFound = true;
                }
                else if (blockFound)
                {
                    holes++;
                }
            }
        }
        return holes;
    }

    private int CalculateBumpiness(int[] board)
    {
        int[] columnHeights = new int[width];
        for (int x = 0; x < width; x++)
        {
            for (int y = 0; y < height; y++)
            {
                if (board[y * width + x] != 0)
                {
                    columnHeights[x] = height - y;
                    break;
                }
            }
        }

        int bumpiness = 0;
        for (int x = 0; x < width - 1; x++)
        {
            bumpiness += Math.Abs(columnHeights[x] - columnHeights[x + 1]);
        }
        return bumpiness;
    }

    private struct Move
    {
        public int rotation;
        public int column;

        public Move(int rotation, int column)
        {
            this.rotation = rotation;
            this.column = column;
        }
    }
}
```
