I'll create a complete Node.js program that plans optimal paths for search drones to maximize unique probability coverage. This solution uses a heuristic approach with path planning and greedy selection to balance performance and coverage.

```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Main function to solve the problem
function solve() {
    const { probability, drones, maxStepsPerDrone } = data;
    const grid = probability;
    const rows = grid.length;
    const cols = grid[0].length;
    
    // Initialize visited grid to track covered probabilities
    const visited = Array(rows).fill().map(() => Array(cols).fill(false));
    
    // Initialize paths for each drone
    const paths = drones.map(() => []);
    
    // Track current positions of drones
    const positions = drones.map(d => ({ x: d.x, y: d.y }));
    
    // Mark initial positions as visited
    positions.forEach(pos => {
        if (grid[pos.y][pos.x] > -1) {
            visited[pos.y][pos.x] = true;
        }
    });
    
    // Function to get valid neighbors
    function getNeighbors(x, y) {
        const neighbors = [];
        const moves = [[0, 1], [0, -1], [1, 0], [-1, 0], [0, 0]]; // right, left, down, up, stay
        
        for (const [dx, dy] of moves) {
            const nx = x + dx;
            const ny = y + dy;
            
            if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && grid[ny][nx] > -1) {
                neighbors.push({ x: nx, y: ny });
            }
        }
        
        return neighbors;
    }
    
    // Function to calculate heuristic value for a position
    function heuristic(x, y, visitedGrid) {
        let score = 0;
        const moves = [[0, 1], [0, -1], [1, 0], [-1, 0]];
        
        for (const [dx, dy] of moves) {
            const nx = x + dx;
            const ny = y + dy;
            
            if (nx >= 0 && nx < cols && ny >= 0 && ny < rows && 
                grid[ny][nx] > -1 && !visitedGrid[ny][nx]) {
                score += grid[ny][nx];
            }
        }
        
        return score;
    }
    
    // Main planning loop
    for (let step = 0; step < maxStepsPerDrone; step++) {
        let bestMove = null;
        let bestDrone = -1;
        let bestScore = -Infinity;
        
        // Try all possible moves for all drones
        for (let droneIdx = 0; droneIdx < drones.length; droneIdx++) {
            const pos = positions[droneIdx];
            const neighbors = getNeighbors(pos.x, pos.y);
            
            for (const neighbor of neighbors) {
                // Create temporary visited grid to simulate the move
                const tempVisited = visited.map(row => [...row]);
                
                // Mark the neighbor as visited
                tempVisited[neighbor.y][neighbor.x] = true;
                
                // Calculate score for this move
                const score = heuristic(neighbor.x, neighbor.y, tempVisited);
                
                // If this is the best move so far, record it
                if (score > bestScore) {
                    bestScore = score;
                    bestMove = neighbor;
                    bestDrone = droneIdx;
                }
            }
        }
        
        // If no good move found, break early
        if (bestScore === -Infinity) {
            break;
        }
        
        // Apply the best move
        const dronePos = positions[bestDrone];
        paths[bestDrone].push([bestMove.x, bestMove.y]);
        positions[bestDrone] = bestMove;
        visited[bestMove.y][bestMove.x] = true;
    }
    
    // Ensure all paths have the correct length (maxStepsPerDrone)
    paths.forEach(path => {
        while (path.length < maxStepsPerDrone) {
            path.push([path[path.length - 1][0], path[path.length - 1][1]]);
        }
    });
    
    return { paths };
}

// Execute and output the result
const result = solve();
console.log(JSON.stringify(result));
```

This program implements a heuristic-based path planning algorithm that:

1. **Initializes** the grid and drone positions
2. **Tracks visited cells** to avoid revisiting the same locations
3. **Evaluates possible moves** for each drone using a heuristic that considers:
   - The probability value of neighboring cells
   - Whether those cells have been visited
4. **Selects the best move** that maximizes uncovered probability
5. **Updates positions** and continues for the specified number of steps
6. **Returns complete paths** with the required format

The algorithm scales well for large scenarios by:
- Using efficient grid operations
- Limiting the search space with heuristics
- Avoiding expensive computations
- Using simple data structures

The solution balances exploration and exploitation by prioritizing moves that lead to high-value, unvisited cells while respecting obstacles and grid boundaries.
