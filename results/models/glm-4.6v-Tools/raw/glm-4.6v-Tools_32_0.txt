I'll write a C++ program that solves the Steiner Tree problem using the Dreyfus-Wagner algorithm, which is suitable for small terminal sets (up to around 20-25 terminals). This algorithm uses dynamic programming with bitmask states to find the exact minimum Steiner tree.

```cpp
#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
#include <climits>
#include <cstring>
#include <bitset>

using namespace std;

const int INF = 1e9;

// Dreyfus-Wagner algorithm for Steiner Tree
pair<int, vector<pair<int, int>>> steiner_tree(const vector<vector<pair<int, int>>>& graph, const vector<int>& terminals) {
    int n = graph.size();
    int k = terminals.size();
    
    if (k == 0) return {0, {}};
    if (k == 1) return {0, {}};
    
    // Precompute all-pairs shortest paths
    vector<vector<int>> dist(n, vector<int>(n, INF));
    for (int i = 0; i < n; ++i) {
        dist[i][i] = 0;
        for (auto& edge : graph[i]) {
            dist[i][edge.first] = min(dist[i][edge.first], edge.second);
        }
    }
    
    for (int m = 0; m < n; ++m) {
        for (int i = 0; i < n; ++i) {
            for (int j = 0; j < n; ++j) {
                if (dist[i][m] < INF && dist[m][j] < INF) {
                    dist[i][j] = min(dist[i][j], dist[i][m] + dist[m][j]);
                }
            }
        }
    }
    
    // DP table: dp[mask][v] = minimum cost to connect terminals in 'mask' ending at vertex v
    vector<vector<int>> dp(1 << k, vector<int>(n, INF));
    
    // Initialize: single terminal states
    for (int i = 0; i < k; ++i) {
        dp[1 << i][terminals[i]] = 0;
    }
    
    // Main DP
    for (int mask = 1; mask < (1 << k); ++mask) {
        // Try to split the mask into two subsets
        for (int submask = (mask - 1) & mask; submask; submask = (submask - 1) & mask) {
            int other = mask ^ submask;
            if (other >= submask) continue; // Avoid duplicate work
            
            // Find best connection between the two subsets
            for (int v = 0; v < n; ++v) {
                if (dp[submask][v] < INF && dp[other][v] < INF) {
                    dp[mask][v] = min(dp[mask][v], dp[submask][v] + dp[other][v]);
                }
            }
        }
        
        // Try to improve by connecting to a non-terminal vertex
        for (int v = 0; v < n; ++v) {
            if (dp[mask][v] == INF) continue;
            
            for (int u = 0; u < n; ++u) {
                if (dist[v][u] < INF) {
                    dp[mask][u] = min(dp[mask][u], dp[mask][v] + dist[v][u]);
                }
            }
        }
    }
    
    // Find the minimum cost and the connecting vertex
    int min_cost = INF;
    int best_vertex = -1;
    for (int v = 0; v < n; ++v) {
        if (dp[(1 << k) - 1][v] < min_cost) {
            min_cost = dp[(1 << k) - 1][v];
            best_vertex = v;
        }
    }
    
    // Reconstruct the tree
    vector<pair<int, int>> edges;
    if (min_cost == INF) return {0, {}}; // No solution
    
    // Backtrack to reconstruct the tree
    vector<vector<int>> parent(1 << k, vector<int>(n, -1));
    vector<vector<int>> parent_vertex(1 << k, vector<int>(n, -1));
    
    // Re-initialize DP with parent tracking
    vector<vector<int>> dp2(1 << k, vector<int>(n, INF));
    for (int i = 0; i < k; ++i) {
        dp2[1 << i][terminals[i]] = 0;
    }
    
    for (int mask = 1; mask < (1 << k); ++mask) {
        for (int submask = (mask - 1) & mask; submask; submask = (submask - 1) & mask) {
            int other = mask ^ submask;
            if (other >= submask) continue;
            
            for (int v = 0; v < n; ++v) {
                if (dp2[submask][v] < INF && dp2[other][v] < INF) {
                    if (dp2[submask][v] + dp2[other][v] < dp2[mask][v]) {
                        dp2[mask][v] = dp2[submask][v] + dp2[other][v];
                        parent[mask][v] = submask;
                        parent_vertex[mask][v] = v;
                    }
                }
            }
        }
        
        for (int v = 0; v < n; ++v) {
            if (dp2[mask][v] == INF) continue;
            
            for (int u = 0; u < n; ++u) {
                if (dist[v][u] < INF && dp2[mask][v] + dist[v][u] < dp2[mask][u]) {
                    dp2[mask][u] = dp2[mask][v] + dist[v][u];
                    parent[mask][u] = mask;
                    parent_vertex[mask][u] = v;
                }
            }
        }
    }
    
    // Reconstruct the tree using the parent information
    function<void(int, int)> reconstruct = [&](int mask, int v) {
        if (mask == 0) return;
        
        int p = parent[mask][v];
        if (p == -1) return;
        
        if (p == mask) {
            // This was a connection to a non-terminal
            int pv = parent_vertex[mask][v];
            reconstruct(mask, pv);
            edges.emplace_back(pv, v);
        } else {
            // This was a split into two subsets
            reconstruct(p, v);
            reconstruct(mask ^ p, v);
        }
    };
    
    reconstruct((1 << k) - 1, best_vertex);
    
    return {min_cost, edges};
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);
    
    int n, m, k;
    cin >> n >> m >> k;
    
    vector<vector<pair<int, int>>> graph(n);
    
    for (int i = 0; i < m; ++i) {
        int u, v, w;
        cin >> u >> v >> w;
        graph[u].emplace_back(v, w);
        graph[v].emplace_back(u, w);
    }
    
    vector<int> terminals(k);
    for (int i = 0; i < k; ++i) {
        cin >> terminals[i];
    }
    
    auto result = steiner_tree(graph, terminals);
    
    cout << result.first << " " << result.second.size() << "\n";
    for (auto& edge : result.second) {
        cout << edge.first << " " << edge.second << "\n";
    }
    
    return 0;
}
```

This program implements the Dreyfus-Wagner algorithm, which is an exact algorithm for the Steiner Tree problem. Here's how it works:

1. **Precompute shortest paths**: It first computes all-pairs shortest paths using the Floyd-Warshall algorithm.

2. **Dynamic programming with bitmask states**: It uses a DP table where `dp[mask][v]` represents the minimum cost to connect all terminals in the bitmask `mask` ending at vertex `v`.

3. **State transitions**: For each state, it tries to:
   - Split the current set of terminals into two subsets and connect them
   - Connect to additional non-terminal vertices to potentially reduce the cost

4. **Tree reconstruction**: After finding the minimum cost, it backtracks through the DP table to reconstruct the actual tree structure.

The algorithm has a time complexity of O(3^k * n^2 + n^3), where k is the number of terminals and n is the number of vertices. This makes it suitable for small terminal sets (typically k ≤ 20-25).

The program reads the input in the specified format and outputs the minimum weight and the edges of the Steiner tree. If no solution exists (which shouldn't happen in a connected graph), it will output 0 and an empty edge list.
