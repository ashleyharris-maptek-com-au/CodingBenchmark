I'll create a Node.js program that solves the vehicle routing problem using a heuristic approach. This solution will use a nearest neighbor algorithm with route optimization to create efficient delivery routes for the vans.

```javascript
const fs = require('fs');
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

// Calculate Euclidean distance between two points
function distance(p1, p2) {
    return Math.sqrt(Math.pow(p1[0] - p2[0], 2) + Math.pow(p1[1] - p2[1], 2));
}

// Main routing function
function planRoutes(data) {
    const depot = data.depot;
    const truckCount = data.truckCount;
    const packages = data.packages;
    
    // If no packages, return empty routes
    if (packages.length === 0) {
        return { routes: [] };
    }
    
    // If only one truck, create one route visiting all packages
    if (truckCount === 1) {
        return { routes: [packages.map(p => p.id)] };
    }
    
    // Sort packages by distance from depot (closest first)
    const sortedPackages = [...packages].sort((a, b) => {
        return distance(depot, a) - distance(depot, b);
    });
    
    // Initialize routes
    const routes = Array(truckCount).fill().map(() => []);
    
    // Assign packages to routes using a simple load balancing approach
    for (let i = 0; i < sortedPackages.length; i++) {
        const package_ = sortedPackages[i];
        const truckIndex = i % truckCount;
        routes[truckIndex].push(package_.id);
    }
    
    // Optimize each route using nearest neighbor algorithm
    for (let i = 0; i < routes.length; i++) {
        if (routes[i].length === 0) continue;
        
        const currentRoute = routes[i].map(id => 
            packages.find(p => p.id === id)
        );
        
        // Simple nearest neighbor optimization
        const optimizedRoute = [currentRoute[0]];
        let remaining = currentRoute.slice(1);
        
        while (remaining.length > 0) {
            const last = optimizedRoute[optimizedRoute.length - 1];
            let nearestIndex = 0;
            let minDist = distance(last, remaining[0]);
            
            for (let j = 1; j < remaining.length; j++) {
                const dist = distance(last, remaining[j]);
                if (dist < minDist) {
                    minDist = dist;
                    nearestIndex = j;
                }
            }
            
            optimizedRoute.push(remaining[nearestIndex]);
            remaining.splice(nearestIndex, 1);
        }
        
        routes[i] = optimizedRoute.map(p => p.id);
    }
    
    // Ensure all routes start and end at depot (conceptually)
    // The actual distance calculation would include depot at start/end
    
    return { routes };
}

// Execute and output result
const result = planRoutes(data);
console.log(JSON.stringify(result));
```

This program implements a solution for the vehicle routing problem with the following approach:

1. **Distance Calculation**: Uses Euclidean distance between points
2. **Package Sorting**: Sorts packages by proximity to the depot
3. **Load Balancing**: Distributes packages evenly across available trucks
4. **Route Optimization**: Applies a nearest neighbor algorithm to optimize each route
5. **Output Format**: Returns the routes in the required JSON format

The solution scales well for large scenarios by:
- Using efficient sorting (O(n log n))
- Implementing a simple nearest neighbor algorithm (O(n²) per route, but with practical optimizations)
- Avoiding external dependencies
- Using built-in Node.js functionality

The program reads from stdin and writes to stdout as required, making it suitable for the benchmark environment.
