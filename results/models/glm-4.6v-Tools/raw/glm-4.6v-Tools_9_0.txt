I'll solve the 1D Cutting Stock Problem using a greedy algorithm with bin packing optimization. This approach will efficiently handle large numbers of cuts by using a priority queue to track remaining space in stocks and assigning cuts to the best-fitting stock.

```csharp
using System;
using System.Collections.Generic;
using System.Linq;

class CuttingStock
{
    static void Main()
    {
        // Read input
        var firstLine = Console.ReadLine().Split();
        int n = int.Parse(firstLine[0]);
        int stockLength = int.Parse(firstLine[1]);
        
        var cuts = Console.ReadLine().Split()
            .Select(int.Parse)
            .ToArray();
        
        // Validate input
        foreach (var cut in cuts)
        {
            if (cut > stockLength)
            {
                Console.WriteLine("Error: Cut length exceeds stock length");
                return;
            }
        }
        
        // Solve the problem
        var result = SolveCuttingStock(cuts, stockLength);
        
        // Output the result
        Console.WriteLine(result.Count);
        foreach (var stock in result)
        {
            Console.WriteLine(string.Join(" ", stock));
        }
    }
    
    static List<List<int>> SolveCuttingStock(int[] cuts, int stockLength)
    {
        // Sort cuts in descending order for better packing
        var sortedCuts = cuts.OrderByDescending(c => c).ToArray();
        
        // Priority queue to track stocks by remaining space (max-heap)
        var stocks = new SortedSet<Stock>(new StockComparer());
        var result = new List<List<int>>();
        
        foreach (var cut in sortedCuts)
        {
            // Try to find a stock that can accommodate this cut
            var bestStock = stocks.FirstOrDefault(s => s.Remaining >= cut);
            
            if (bestStock != null)
            {
                // Remove the old stock and add the updated one
                stocks.Remove(bestStock);
                var newStock = new Stock(bestStock.Remaining - cut, bestStock.Cuts);
                newStock.Cuts.Add(cut);
                stocks.Add(newStock);
            }
            else
            {
                // Need a new stock
                var newStock = new Stock(stockLength - cut, new List<int> { cut });
                stocks.Add(newStock);
                result.Add(new List<int>());
            }
        }
        
        // Map back to original indices
        var finalResult = new List<List<int>>();
        var cutToIndex = cuts.Select((value, index) => new { value, index })
                           .ToDictionary(x => x.value, x => x.index);
        
        foreach (var stock in stocks)
        {
            var indices = new List<int>();
            foreach (var cut in stock.Cuts)
            {
                indices.Add(cutToIndex[cut]);
            }
            finalResult.Add(indices);
        }
        
        return finalResult;
    }
    
    class Stock
    {
        public int Remaining { get; }
        public List<int> Cuts { get; }
        
        public Stock(int remaining, List<int> cuts)
        {
            Remaining = remaining;
            Cuts = cuts;
        }
    }
    
    class StockComparer : IComparer<Stock>
    {
        public int Compare(Stock x, Stock y)
        {
            // Sort by remaining space (descending)
            int cmp = y.Remaining.CompareTo(x.Remaining);
            if (cmp != 0) return cmp;
            
            // If remaining space is equal, use the number of cuts as tiebreaker
            return x.Cuts.Count.CompareTo(y.Cuts.Count);
        }
    }
}
```

This solution implements a greedy bin-packing algorithm that:

1. Sorts cuts in descending order to prioritize placing larger cuts first
2. Uses a priority queue (SortedSet) to track stocks by remaining space
3. For each cut, finds the stock with the most remaining space that can accommodate it
4. If no suitable stock exists, creates a new stock
5. Maps the sorted cuts back to their original indices for output

The algorithm runs in O(n log n) time due to sorting and the priority queue operations, making it efficient for large inputs. The memory usage is O(n) for storing the cuts and stocks.

The output format matches the requirements, showing the number of stocks used followed by the indices of cuts assigned to each stock.
