```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the normal (may be needed due to interpolation)
    vec3 N = normalize(normal);
    
    // Calculate light direction (from surface to light)
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse term (Lambertian)
    float diffuse = max(dot(N, L), 0.0);
    
    // Create checkerboard pattern at 8x8 frequency
    float checker = mod(floor(uv.x * 8.0) + floor(uv.y * 8.0), 2.0);
    float checkerColor = mix(0.2, 0.9, checker);
    
    // Apply Lambertian shading factor
    float shading = 0.5 + 0.5 * diffuse;
    
    // Combine checkerboard with shading
    vec3 finalColor = checkerColor * shading;
    
    // Set output color with alpha = 1.0
    outColor = vec4(finalColor, 1.0);
}
```
