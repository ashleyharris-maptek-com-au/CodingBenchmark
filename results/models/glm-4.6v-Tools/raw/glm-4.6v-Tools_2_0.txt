using System.Linq;
using System.Diagnostics;

class ChinesePostmanSolver

    private static int N, M;

    private static List<int>[] adj;
    private static int[] degree;
    private static int[] nodeIndex;
    private static int[] comp;
    private static int compCount;
    private static long totalWeight;

    public static void Main()
    {
        var sw = Stopwatch.StartNew();
        ReadInput();
        
        // Determine graph complexity and choose appropriate algorithm
        if (N <= 20 || M <= 45)
        {
            // Trivial: Use exact algorithm with DFS
            SolveTrivial();
        }
        else if (N <= 200 || M <= 600)
        {
            // Medium: Use Hierholzer's algorithm with minimal matching
            SolveMedium();
        }
        else if (N <= 1000 || M <= 5000)
        {
            // Large: Optimized Hierholzer with efficient matching
            SolveLarge();
        }
        else if (N <= 10000 || M <= 80000)
        {
            // Extreme: Highly optimized with adjacency compression
            SolveExtreme();
        }
        else
        {
            // Epic: Maximum optimization for massive graphs
            SolveEpic();
        }
        
        sw.Stop();
        Console.Error.WriteLine($"Total time: {sw.ElapsedMilliseconds} ms");
    }

    private static void ReadInput()
    {
        var parts = Console.ReadLine().Split();
        N = int.Parse(parts[0]);
        M = int.Parse(parts[1]);
        
        edges = new List<(int, int, int)>(M);
        adj = new List<int>[N];
        degree = new int[N];
        nodeIndex = new int[N];
        comp = new int[N];
        
        for (int i = 0; i < N; i++)
        {
            adj[i] = new List<int>();
        }
        
        for (int i = 0; i < M; i++)
        {
            parts = Console.ReadLine().Split();
            int u = int.Parse(parts[0]);
            int v = int.Parse(parts[1]);
            int w = int.Parse(parts[2]);
            
            edges.Add((u, v, w));
            adj[u].Add(v);
            adj[v].Add(u);
            degree[u]++;
            degree[v]++;
            totalWeight += w;
        }
    }

    private static void SolveTrivial()
    {
        // For very small graphs, we can use exact algorithms
        if (IsEulerian())
        {
            FindEulerianTour();
        }
        else
        {
            // Add minimum weight matching for odd-degree vertices
            AddMinimumWeightMatching();
            FindEulerianTour();
        }
        PrintTour();
    }

    private static void SolveMedium()
    {
        // Medium graphs: Hierholzer's algorithm with optimized matching
        if (!IsEulerian())
        {
            AddMinimumWeightMatching();
        }
        FindEulerianTour();
        PrintTour();
    }

    private static void SolveLarge()
    {
        // Large graphs: Optimized implementation
        if (!IsEulerian())
        {
            AddOptimizedMatching();
        }
        FindEulerianTourOptimized();
        PrintTour();
    }

    private static void SolveExtreme()
    {
        // Extreme graphs: Highly optimized with adjacency compression
        CompressGraph();
        if (!IsEulerian())
        {
            AddOptimizedMatching();
        }
        FindEulerianTourOptimized();
        DecompressTour();
        PrintTour();
    }

    private static void SolveEpic()
    {
        // Epic graphs: Maximum optimization
        CompressGraph();
        if (!IsEulerian())
        {
            AddParallelMatching();
        }
        FindEulerianTourParallel();
        DecompressTour();
        PrintTour();
    }

    private static bool IsEulerian()
    {
        int oddCount = 0;
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0)
            {
                oddCount++;
            }
        }
        return oddCount == 0 || oddCount == 2;
    }

    private static void AddMinimumWeightMatching()
    {
        // Find all odd-degree vertices
        var oddVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0)
            {
                oddVertices.Add(i);
            }
        }
        
        // Find minimum weight perfect matching
        var matching = new List<(int, int)>();
        if (oddVertices.Count > 0)
        {
            matching = FindMinimumWeightPerfectMatching(oddVertices);
        }
        
        // Add matching edges to the graph
        foreach (var (u, v) in matching)
        {
            adj[u].Add(v);
            adj[v].Add(u);
            degree[u]++;
            degree[v]++;
            totalWeight += GetEdgeWeight(u, v);
        }
    }

    private static List<(int, int)> FindMinimumWeightPerfectMatching(List<int> oddVertices)
    {
        // For small odd vertex sets, use exact matching
        if (oddVertices.Count <= 20)
        {
            return FindExactMinimumWeightMatching(oddVertices);
        }
        
        // For larger sets, use greedy approximation
        return FindGreedyMinimumWeightMatching(oddVertices);
    }

    private static List<(int, int)> FindExactMinimumWeightMatching(List<int> oddVertices)
    {
        int k = oddVertices.Count;
        var dist = new int[k, k];
        var matching = new List<(int, int)>();
        
        // Build distance matrix
        for (int i = 0; i < k; i++)
        {
            for (int j = i + 1; j < k; j++)
            {
                int u = oddVertices[i];
                int v = oddVertices[j];
                dist[i, j] = GetEdgeWeight(u, v);
                dist[j, i] = dist[i, j];
            }
        }
        
        // Use DP for exact matching
        var dp = new int[1 << k];
        var parent = new int[1 << k];
        
        for (int mask = 0; mask < (1 << k); mask++)
        {
            dp[mask] = int.MaxValue;
        }
        dp[0] = 0;
        
        for (int mask = 0; mask < (1 << k); mask++)
        {
            if (dp[mask] == int.MaxValue) continue;
            
            for (int i = 0; i < k; i++)
            {
                if ((mask & (1 << i)) == 0)
                {
                    for (int j = i + 1; j < k; j++)
                    {
                        if ((mask & (1 << j)) == 0)
                        {
                            int newMask = mask | (1 << i) | (1 << j);
                            if (dp[newMask] > dp[mask] + dist[i, j])
                            {
                                dp[newMask] = dp[mask] + dist[i, j];
                                parent[newMask] = mask;
                            }
                        }
                    }
                }
            }
        }
        
        // Reconstruct matching
        int currentMask = (1 << k) - 1;
        while (currentMask != 0)
        {
            int prevMask = parent[currentMask];
            int diff = currentMask ^ prevMask;
            int i = 0;
            while ((diff & (1 << i)) == 0) i++;
            int j = i + 1;
            while ((diff & (1 << j)) == 0) j++;
            
            matching.Add((oddVertices[i], oddVertices[j]));
            currentMask = prevMask;
        }
        
        return matching;
    }

    private static List<(int, int)> FindGreedyMinimumWeightMatching(List<int> oddVertices)
    {
        var matching = new List<(int, int)>();
        var remaining = new HashSet<int>(oddVertices);
        
        while (remaining.Count > 0)
        {
            int u = remaining.First();
            remaining.Remove(u);
            
            int bestV = -1;
            int minWeight = int.MaxValue;
            
            foreach (int v in remaining)
            {
                int weight = GetEdgeWeight(u, v);
                if (weight < minWeight)
                {
                    minWeight = weight;
                    bestV = v;
                }
            }
            
            if (bestV != -1)
            {
                matching.Add((u, bestV));
                remaining.Remove(bestV);
            }
        }
        
        return matching;
    }

    private static void AddOptimizedMatching()
    {
        // Optimized matching for large graphs
        var oddVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0)
            {
                oddVertices.Add(i);
            }
        }
        
        if (oddVertices.Count > 0)
        {
            var matching = FindOptimizedMatching(oddVertices);
            foreach (var (u, v) in matching)
            {
                adj[u].Add(v);
                adj[v].Add(u);
                degree[u]++;
                degree[v]++;
                totalWeight += GetEdgeWeight(u, v);
            }
        }
    }

    private static List<(int, int)> FindOptimizedMatching(List<int> oddVertices)
    {
        // Use heuristic matching for large graphs
        var matching = new List<(int, int)>();
        var remaining = new HashSet<int>(oddVertices);
        
        // Sort by degree to improve matching quality
        var sorted = oddVertices.OrderBy(v => degree[v]).ToList();
        
        foreach (int u in sorted)
        {
            if (!remaining.Contains(u)) continue;
            
            remaining.Remove(u);
            int bestV = -1;
            int minWeight = int.MaxValue;
            
            // Try to find best match among neighbors
            foreach (int v in adj[u])
            {
                if (remaining.Contains(v) && GetEdgeWeight(u, v) < minWeight)
                {
                    minWeight = GetEdgeWeight(u, v);
                    bestV = v;
                }
            }
            
            if (bestV != -1)
            {
                matching.Add((u, bestV));
                remaining.Remove(bestV);
            }
        }
        
        return matching;
    }

    private static void AddParallelMatching()
    {
        // Parallel matching for epic graphs
        var oddVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degree[i] % 2 != 0)
            {
                oddVertices.Add(i);
            }
        }
        
        if (oddVertices.Count > 0)
        {
            var matching = FindParallelMatching(oddVertices);
            foreach (var (u, v) in matching)
            {
                adj[u].Add(v);
                adj[v].Add(u);
                degree[u]++;
                degree[v]++;
                totalWeight += GetEdgeWeight(u, v);
            }
        }
    }

    private static List<(int, int)> FindParallelMatching(List<int> oddVertices)
    {
        var matching = new List<(int, int)>();
        var remaining = new HashSet<int>(oddVertices);
        
        // Use parallel processing for large sets
        var tasks = new List<System.Threading.Tasks.Task>();
        var localMatchings = new List<List<(int, int)>>();
        
        int chunkSize = Math.Max(1, oddVertices.Count / 64);
        for (int i = 0; i < oddVertices.Count; i += chunkSize)
        {
            int start = i;
            int end = Math.Min(i + chunkSize, oddVertices.Count);
            tasks.Add(System.Threading.Tasks.Task.Run(() => {
                var local = new List<(int, int)>();
                var localRemaining = new HashSet<int>();
                
                for (int j = start; j < end; j++)
                {
                    localRemaining.Add(oddVertices[j]);
                }
                
                while (localRemaining.Count > 0)
                {
                    int u = localRemaining.First();
                    localRemaining.Remove(u);
                    
                    int bestV = -1;
                    int minWeight = int.MaxValue;
                    
                    foreach (int v in localRemaining)
                    {
                        int weight = GetEdgeWeight(u, v);
                        if (weight < minWeight)
                        {
                            minWeight = weight;
                            bestV = v;
                        }
                    }
                    
                    if (bestV != -1)
                    {
                        local.Add((u, bestV));
                        localRemaining.Remove(bestV);
                    }
                }
                
                lock (localMatchings)
                {
                    localMatchings.Add(local);
                }
            }));
        }
        
        System.Threading.Tasks.Task.WaitAll(tasks.ToArray());
        
        foreach (var local in localMatchings)
        {
            matching.AddRange(local);
        }
        
        return matching;
    }

    private static int GetEdgeWeight(int u, int v)
    {
        // Find weight of edge between u and v
        foreach (var (a, b, w) in edges)
        {
            if ((a == u && b == v) || (a == v && b == u))
            {
                return w;
            }
        }
        return int.MaxValue;
    }

    private static void FindEulerianTour()
    {
        eulerianTour = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);
        
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (adj[u].Count > 0)
            {
                int v = adj[u][0];
                adj[u].RemoveAt(0);
                adj[v].RemoveAt(adj[v].IndexOf(u));
                stack.Push(v);
            }
            else
            {
                eulerianTour.Add(stack.Pop());
            }
        }
        
        eulerianTour.Reverse();
    }

    private static void FindEulerianTourOptimized()
    {
        eulerianTour = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);
        
        // Use adjacency lists with index tracking for faster removal
        var adjIndex = new int[N][];
        for (int i = 0; i < N; i++)
        {
            adjIndex[i] = new int[adj[i].Count];
            for (int j = 0; j < adj[i].Count; j++)
            {
                adjIndex[i][j] = adj[i][j];
            }
        }
        
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (adjIndex[u].Length > 0)
            {
                int v = adjIndex[u][0];
                // Remove edge from both adjacency lists
                var vList = adjIndex[v];
                int vIndex = Array.IndexOf(vList, u);
                if (vIndex >= 0)
                {
                    var newVList = new int[vList.Length - 1];
                    Array.Copy(vList, 0, newVList, 0, vIndex);
                    Array.Copy(vList, vIndex + 1, newVList, vIndex, vList.Length - vIndex - 1);
                    adjIndex[v] = newVList;
                }
                
                var newUList = new int[adjIndex[u].Length - 1];
                Array.Copy(adjIndex[u], 1, newUList, 0, newUList.Length);
                adjIndex[u] = newUList;
                
                stack.Push(v);
            }
            else
            {
                eulerianTour.Add(stack.Pop());
            }
        }
        
        eulerianTour.Reverse();
    }

    private static void FindEulerianTourParallel()
    {
        eulerianTour = new List<int>();
        var stack = new Stack<int>();
        stack.Push(0);
        
        // Use parallel processing for tour finding
        var adjIndex = new int[N][];
        for (int i = 0; i < N; i++)
        {
            adjIndex[i] = new int[adj[i].Count];
            for (int j = 0; j < adj[i].Count; j++)
            {
                adjIndex[i][j] = adj[i][j];
            }
        }
        
        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (adjIndex[u].Length > 0)
            {
                int v = adjIndex[u][0];
                
                // Parallel edge removal
                System.Threading.Tasks.Parallel.Invoke(
                    () => {
                        var vList = adjIndex[v];
                        int vIndex = Array.IndexOf(vList, u);
                        if (vIndex >= 0)
                        {
                            var newVList = new int[vList.Length - 1];
                            Array.Copy(vList, 0, newVList, 0, vIndex);
                            Array.Copy(vList, vIndex + 1, newVList, vIndex, vList.Length - vIndex - 1);
                            adjIndex[v] = newVList;
                        }
                    },
                    () => {
                        var newUList = new int[adjIndex[u].Length - 1];
                        Array.Copy(adjIndex[u], 1, newUList, 0, newUList.Length);
                        adjIndex[u] = newUList;
                    }
                );
                
                stack.Push(v);
            }
            else
            {
                eulerianTour.Add(stack.Pop());
            }
        }
        
        eulerianTour.Reverse();
    }

    private static void CompressGraph()
    {
        // Find connected components
        compCount = 0;
        for (int i = 0; i < N; i++)
        {
            comp[i] = -1;
        }
        
        for (int i = 0; i < N; i++)
        {
            if (comp[i] == -1)
            {
                DFS(i, compCount);
                compCount++;
            }
        }
        
        // Create mapping from original nodes to compressed nodes
        var compNodes = new List<int>[compCount];
        for (int i = 0; i < compCount; i++)
        {
            compNodes[i] = new List<int>();
        }
        
        for (int i = 0; i < N; i++)
        {
            compNodes[comp[i]].Add(i);
        }
        
        // Create compressed graph
        var compAdj = new List<int>[compCount];
        for (int i = 0; i < compCount; i++)
        {
            compAdj[i] = new List<int>();
        }
        
        var compEdges = new List<(int, int, int)>();
        
        foreach (var (u, v, w) in edges)
        {
            int cu = comp[u];
            int cv = comp[v];
            if (cu != cv)
            {
                compAdj[cu].Add(cv);
                compAdj[cv].Add(cu);
                compEdges.Add((cu, cv, w));
            }
        }
        
        // Replace original graph with compressed graph
        N = compCount;
        adj = compAdj;
        edges = compEdges;
        degree = new int[N];
        
        for (int i = 0; i < N; i++)
        {
            degree[i] = adj[i].Count;
        }
    }

    private static void DFS(int u, int component)
    {
        comp[u] = component;
        foreach (int v in adj[u])
        {
            if (comp[v] == -1)
            {
                DFS(v, component);
            }
        }
    }

    private static void DecompressTour()
    {
        // Decompress the tour back to original node indices
        var decompressed = new List<int>();
        foreach (int node in eulerianTour)
        {
            if (comp[node] == node) // This is a representative node
            {
                decompressed.Add(node);
            }
            else
            {
                // Find a representative node in the same component
                for (int i = 0; i < N; i++)
                {
                    if (comp[i] == node)
                    {
                        decompressed.Add(i);
                        break;
                    }
                }
            }
        }
        eulerianTour = decompressed;
    }

    private static void PrintTour()
    {
        if (eulerianTour == null || eulerianTour.Count == 0)
        {
            Console.WriteLine("0");
            return;
        }
        
        // Optionally print tour length
        Console.Write(eulerianTour.Count + " ");
        
        // Print the tour
        Console.WriteLine(string.Join(" ", eulerianTour));
    }
}
