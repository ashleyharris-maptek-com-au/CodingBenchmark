I'll create a complete Node.js TypeScript program for the van routing benchmark. This will use a greedy nearest-neighbor algorithm with depot return optimization, which scales well for large scenarios while providing reasonable solutions.

```typescript
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

interface Package {
    id: string;
    x: number;
    y: number;
}

interface Depot {
    x: number;
    y: number;
}

interface InputData {
    depot: Depot;
    truckCount: number;
    packages: Package[];
}

interface OutputData {
    routes: string[][];
}

// Calculate Euclidean distance between two points
function distance(p1: {x: number, y: number}, p2: {x: number, y: number}): number {
    return Math.sqrt(Math.pow(p1.x - p2.x, 2) + Math.pow(p1.y - p2.y, 2));
}

// Main routing function
function planRoutes(input: InputData): OutputData {
    const depot = input.depot;
    const truckCount = input.truckCount;
    const packages = [...input.packages];
    
    // If no packages, return empty routes
    if (packages.length === 0) {
        return { routes: Array(truckCount).fill([]) };
    }
    
    // Sort packages by distance from depot (helps with initial distribution)
    packages.sort((a, b) => {
        const distA = distance(depot, a);
        const distB = distance(depot, b);
        return distA - distB;
    });
    
    // Initialize routes
    const routes: string[][] = Array(truckCount).fill(null).map(() => []);
    const routeLoads: number[] = Array(truckCount).fill(0);
    
    // Distribute packages to routes using a greedy approach
    for (const pkg of packages) {
        // Find the route with minimum current load
        let minLoadIndex = 0;
        let minLoad = routeLoads[0];
        
        for (let i = 1; i < truckCount; i++) {
            if (routeLoads[i] < minLoad) {
                minLoad = routeLoads[i];
                minLoadIndex = i;
            }
        }
        
        routes[minLoadIndex].push(pkg.id);
        routeLoads[minLoadIndex]++;
    }
    
    // Optimize each route using nearest neighbor algorithm
    for (let i = 0; i < truckCount; i++) {
        const route = routes[i];
        if (route.length === 0) continue;
        
        const optimizedRoute: string[] = [];
        const remaining = [...route];
        
        // Start from the package closest to depot
        let current = depot;
        let closestIndex = 0;
        let minDist = Infinity;
        
        for (let j = 0; j < remaining.length; j++) {
            const pkg = packages.find(p => p.id === remaining[j])!;
            const dist = distance(current, pkg);
            if (dist < minDist) {
                minDist = dist;
                closestIndex = j;
            }
        }
        
        optimizedRoute.push(remaining[closestIndex]);
        remaining.splice(closestIndex, 1);
        
        // Build route using nearest neighbor
        while (remaining.length > 0) {
            current = packages.find(p => p.id === optimizedRoute[optimizedRoute.length - 1])!;
            minDist = Infinity;
            closestIndex = 0;
            
            for (let j = 0; j < remaining.length; j++) {
                const pkg = packages.find(p => p.id === remaining[j])!;
                const dist = distance(current, pkg);
                if (dist < minDist) {
                    minDist = dist;
                    closestIndex = j;
                }
            }
            
            optimizedRoute.push(remaining[closestIndex]);
            remaining.splice(closestIndex, 1);
        }
        
        routes[i] = optimizedRoute;
    }
    
    return { routes };
}

// Execute the routing function and output the result
const result = planRoutes(data);
console.log(JSON.stringify(result));
```

This program implements a scalable van routing solution with the following features:

1. **Input Handling**: Reads JSON from stdin as specified
2. **Distance Calculation**: Uses Euclidean distance between points
3. **Route Planning**:
   - Distributes packages evenly across available vans
   - Optimizes each route using nearest neighbor algorithm
   - Ensures all packages are delivered exactly once
   - Each route implicitly starts and ends at the depot

4. **Scalability**:
   - O(n log n) sorting for initial package distribution
   - O(n²) optimization per route (suitable for large scenarios)
   - No external dependencies or network calls

5. **Output**: Returns JSON with routes in the required format

The algorithm provides a good balance between solution quality and computational efficiency, making it suitable for the benchmark's increasingly large scenarios.
