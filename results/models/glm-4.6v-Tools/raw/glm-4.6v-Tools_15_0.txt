use std::f64::consts::SQRT_2;
use std::io::{self, BufRead, Write};

// Quaternion struct for rotations
#[derive(Debug, Clone, Copy)]
struct Quaternion {
    w: f64,
    x: f64,
    y: f64,
    z: f64,
}

impl Quaternion {
    fn new(w: f64, x: f64, y: f64, z: f64) -> Self {
        Self { w, x, y, z }
    }

    fn from_axis_angle(axis: [f64; 3], angle: f64) -> Self {
        let half_angle = angle / 2.0;
        let sin_half = half_angle.sin();
        let cos_half = half_angle.cos();
        
        Self {
            w: cos_half,
            x: axis[0] * sin_half,
            y: axis[1] * sin_half,
            z: axis[2] * sin_half,
        }
    }

    fn normalize(&self) -> Self {
        let norm = (self.w * self.w + self.x * self.x + self.y * self.y + self.z * self.z).sqrt();
        if norm < 1e-10 {
            Self::new(1.0, 0.0, 0.0, 0.0)
        } else {
            Self {
                w: self.w / norm,
                x: self.x / norm,
                y: self.y / norm,
                z: self.z / norm,
            }
        }
    }

    fn conjugate(&self) -> Self {
        Self {
            w: self.w,
            x: -self.x,
            y: -self.y,
            z: -self.z,
        }
    }

    fn mul(&self, other: &Self) -> Self {
        Self {
            w: self.w * other.w - self.x * other.x - self.y * other.y - self.z * other.z,
            x: self.w * other.x + self.x * other.w + self.y * other.z - self.z * other.y,
            y: self.w * other.y - self.x * other.z + self.y * other.w + self.z * other.x,
            z: self.w * other.z + self.x * other.y - self.y * other.x + self.z * other.w,
        }
    }

    fn rotate_vector(&self, v: [f64; 3]) -> [f64; 3] {
        let qv = Self::new(0.0, v[0], v[1], v[2]);
        let q_conj = self.conjugate();
        let qr = self.mul(&qv).mul(&q_conj);
        [qr.x, qr.y, qr.z]
    }
}

// Vector operations
fn dot(a: [f64; 3], b: [f64; 3]) -> f64 {
    a[0] * b[0] + a[1] * b[1] + a[2] * b[2]
}

fn cross(a: [f64; 3], b: [f64; 3]) -> [f64; 3] {
    [
        a[1] * b[2] - a[2] * b[1],
        a[2] * b[0] - a[0] * b[2],
        a[0] * b[1] - a[1] * b[0],
    ]
}

fn normalize(v: [f64; 3]) -> [f64; 3] {
    let norm = (v[0] * v[0] + v[1] * v[1] + v[2] * v[2]).sqrt();
    if norm < 1e-10 {
        [0.0, 0.0, 0.0]
    } else {
        [v[0] / norm, v[1] / norm, v[2] / norm]
    }
}

fn scale(v: [f64; 3], s: f64) -> [f64; 3] {
    [v[0] * s, v[1] * s, v[2] * s]
}

fn add(a: [f64; 3], b: [f64; 3]) -> [f64; 3] {
    [a[0] + b[0], a[1] + b[1], a[2] + b[2]]
}

fn sub(a: [f64; 3], b: [f64; 3]) -> [f64; 3] {
    [a[0] - b[0], a[1] - b[1], a[2] - b[2]]
}

// Tetrahedron struct
#[derive(Debug, Clone, Copy)]
struct Tetrahedron {
    center: [f64; 3],
    rotation: Quaternion,
    scale: f64,
}

impl Tetrahedron {
    fn new(center: [f64; 3], rotation: Quaternion, scale: f64) -> Self {
        Self {
            center,
            rotation,
            scale,
        }
    }

    fn transform_vertices(&self) -> [[f64; 3]; 4] {
        let vertices = [
            [1.0, 0.0, -0.707],
            [-1.0, 0.0, -0.707],
            [0.0, 1.0, 0.707],
            [0.0, -1.0, 0.707],
        ];
        
        let mut transformed = [[0.0; 3]; 4];
        for i in 0..4 {
            let scaled = scale(vertices[i], self.scale);
            let rotated = self.rotation.rotate_vector(scaled);
            transformed[i] = add(self.center, rotated);
        }
        transformed
    }
}

// Main solution structure
struct ShadowCoveringSolver {
    polygon: Vec<[f64; 2]>,
    sun_direction: [f64; 3],
}

impl ShadowCoveringSolver {
    fn new(polygon: Vec<[f64; 2]>, sun_direction: [f64; 3]) -> Self {
        Self {
            polygon,
            sun_direction: normalize(sun_direction),
        }
    }

    // Check if a point is inside the polygon (2D)
    fn point_in_polygon(&self, point: [f64; 2]) -> bool {
        let mut inside = false;
        let n = self.polygon.len();
        
        for i in 0..n {
            let j = (i + 1) % n;
            let xi = self.polygon[i][0];
            let yi = self.polygon[i][1];
            let xj = self.polygon[j][0];
            let yj = self.polygon[j][1];
            
            let intersect = ((yi > point[1]) != (yj > point[1])) &&
                (point[0] < (xj - xi) * (point[1] - yi) / (yj - yi) + xi);
            
            if intersect {
                inside = !inside;
            }
        }
        
        inside
    }

    // Check if a tetrahedron's shadow covers a point
    fn shadow_covers_point(&self, tetra: &Tetrahedron, point: [f64; 2]) -> bool {
        let vertices = tetra.transform_vertices();
        
        // Project vertices onto ground plane
        let projected: Vec<[f64; 2]> = vertices
            .iter()
            .map(|v| [v[0], v[1]])
            .collect();
        
        // Simple coverage check: if point is inside the convex hull of projected vertices
        self.point_in_convex_hull(&projected, point)
    }

    // Check if point is in convex hull (2D)
    fn point_in_convex_hull(&self, hull: &[ [f64; 2] ], point: [f64; 2]) -> bool {
        let n = hull.len();
        if n < 3 {
            return false;
        }
        
        for i in 0..n {
            let j = (i + 1) % n;
            let vi = hull[i];
            let vj = hull[j];
            
            let edge = sub(vj, vi);
            let to_point = sub(point, vi);
            
            let cross_z = edge[0] * to_point[1] - edge[1] * to_point[0];
            if cross_z < 0.0 {
                return false;
            }
        }
        
        true
    }

    // Check if two tetrahedrons intersect
    fn tetrahedrons_intersect(&self, t1: &Tetrahedron, t2: &Tetrahedron) -> bool {
        // Simple bounding box check for performance
        let v1 = t1.transform_vertices();
        let v2 = t2.transform_vertices();
        
        let (min1, max1) = self.bounding_box(&v1);
        let (min2, max2) = self.bounding_box(&v2);
        
        // Check if bounding boxes overlap
        if min1[0] > max2[0] || max1[0] < min2[0] ||
           min1[1] > max2[1] || max1[1] < min2[1] ||
           min1[2] > max2[2] || max1[2] < min2[2] {
            return false;
        }
        
        // More detailed intersection check could be added here
        // For now, we'll use a simplified approach
        false
    }

    // Get bounding box of vertices
    fn bounding_box(&self, vertices: &[[f64; 3]]) -> ([f64; 3], [f64; 3]) {
        let mut min = [f64::MAX, f64::MAX, f64::MAX];
        let mut max = [f64::MIN, f64::MIN, f64::MIN];
        
        for v in vertices {
            for i in 0..3 {
                min[i] = min[i].min(v[i]);
                max[i] = max[i].max(v[i]);
            }
        }
        
        (min, max)
    }

    // Generate tetrahedrons for the polygon
    fn generate_tetrahedrons(&self) -> Vec<Tetrahedron> {
        let n = self.polygon.len();
        
        // Adaptive strategy based on polygon size
        if n <= 10 {
            // Small polygon: use simple grid-based approach
            self.generate_small_polygon_solution()
        } else if n <= 100 {
            // Medium polygon: use edge-based approach
            self.generate_medium_polygon_solution()
        } else {
            // Large polygon: use hierarchical approach
            self.generate_large_polygon_solution()
        }
    }

    // Solution for small polygons
    fn generate_small_polygon_solution(&self) -> Vec<Tetrahedron> {
        let mut tetrahedrons = Vec::new();
        let center = self.polygon_center();
        
        // Create a single large tetrahedron
        let rotation = self.sun_aligned_rotation();
        let scale = 10.0; // Maximum allowed scale
        
        let tetra = Tetrahedron::new(center, rotation, scale);
        tetrahedrons.push(tetra);
        
        tetrahedrons
    }

    // Solution for medium polygons
    fn generate_medium_polygon_solution(&self) -> Vec<Tetrahedron> {
        let mut tetrahedrons = Vec::new();
        let center = self.polygon_center();
        
        // Create tetrahedrons along the polygon edges
        let n = self.polygon.len();
        let edge_length = self.average_edge_length();
        
        for i in 0..n {
            let j = (i + 1) % n;
            let edge_center = [
                (self.polygon[i][0] + self.polygon[j][0]) / 2.0,
                (self.polygon[i][1] + self.polygon[j][1]) / 2.0,
                0.0,
            ];
            
            let direction = normalize([
                self.polygon[j][0] - self.polygon[i][0],
                self.polygon[j][1] - self.polygon[i][1],
                0.0,
            ]);
            
            let rotation = self.rotation_from_direction(direction);
            let scale = edge_length.min(10.0);
            
            let tetra = Tetrahedron::new(edge_center, rotation, scale);
            tetrahedrons.push(tetra);
        }
        
        tetrahedrons
    }

    // Solution for large polygons
    fn generate_large_polygon_solution(&self) -> Vec<Tetrahedron> {
        let mut tetrahedrons = Vec::new();
        let center = self.polygon_center();
        
        // Use a grid-based approach with hierarchical clustering
        let grid_size = 5.0;
        let (min_x, max_x, min_y, max_y) = self.polygon_bounds();
        let width = max_x - min_x;
        let height = max_y - min_y;
        
        let cols = ((width / grid_size).ceil() as usize).max(1);
        let rows = ((height / grid_size).ceil() as usize).max(1);
        
        for row in 0..rows {
            for col in 0..cols {
                let x = min_x + col as f64 * grid_size + grid_size / 2.0;
                let y = min_y + row as f64 * grid_size + grid_size / 2.0;
                
                // Check if grid cell is inside polygon
                if self.point_in_polygon([x, y]) {
                    let center = [x, y, 5.0]; // Height above ground
                    let rotation = self.sun_aligned_rotation();
                    let scale = grid_size.min(10.0);
                    
                    let tetra = Tetrahedron::new(center, rotation, scale);
                    tetrahedrons.push(tetra);
                }
            }
        }
        
        // Optimize by removing redundant tetrahedrons
        self.optimize_tetrahedrons(&mut tetrahedrons);
        
        tetrahedrons
    }

    // Optimize tetrahedrons by removing redundant ones
    fn optimize_tetrahedrons(&self, tetrahedrons: &mut Vec<Tetrahedron>) {
        let mut to_remove = Vec::new();
        
        for i in 0..tetrahedrons.len() {
            let mut covered = true;
            for j in 0..tetrahedrons.len() {
                if i != j && self.shadow_covers_point(&tetrahedrons[i], self.polygon_center()) {
                    covered = false;
                    break;
                }
            }
            
            if covered {
                to_remove.push(i);
            }
        }
        
        // Remove in reverse order to avoid index issues
        for &i in to_remove.iter().rev() {
            tetrahedrons.remove(i);
        }
    }

    // Calculate polygon center
    fn polygon_center(&self) -> [f64; 3] {
        let mut sum_x = 0.0;
        let mut sum_y = 0.0;
        
        for vertex in &self.polygon {
            sum_x += vertex[0];
            sum_y += vertex[1];
        }
        
        let n = self.polygon.len() as f64;
        [sum_x / n, sum_y / n, 5.0] // Height above ground
    }

    // Calculate average edge length
    fn average_edge_length(&self) -> f64 {
        let n = self.polygon.len();
        let mut total_length = 0.0;
        
        for i in 0..n {
            let j = (i + 1) % n;
            let dx = self.polygon[j][0] - self.polygon[i][0];
            let dy = self.polygon[j][1] - self.polygon[i][1];
            total_length += (dx * dx + dy * dy).sqrt();
        }
        
        total_length / n as f64
    }

    // Get polygon bounds
    fn polygon_bounds(&self) -> (f64, f64, f64, f64) {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        
        for vertex in &self.polygon {
            min_x = min_x.min(vertex[0]);
            max_x = max_x.max(vertex[0]);
            min_y = min_y.min(vertex[1]);
            max_y = max_y.max(vertex[1]);
        }
        
        (min_x, max_x, min_y, max_y)
    }

    // Get sun-aligned rotation
    fn sun_aligned_rotation(&self) -> Quaternion {
        // Rotate so that tetrahedron faces the sun
        let up = [0.0, 0.0, 1.0];
        let right = cross(self.sun_direction, up);
        let forward = cross(right, self.sun_direction);
        
        let rotation_axis = normalize(cross(up, forward));
        let angle = (dot(up, forward)).acos();
        
        Quaternion::from_axis_angle(rotation_axis, angle)
    }

    // Get rotation from direction vector
    fn rotation_from_direction(&self, direction: [f64; 3]) -> Quaternion {
        let up = [0.0, 0.0, 1.0];
        let right = cross(direction, up);
        let forward = cross(right, direction);
        
        let rotation_axis = normalize(cross(up, forward));
        let angle = (dot(up, forward)).acos();
        
        Quaternion::from_axis_angle(rotation_axis, angle)
    }
}

    let sun_direction = [parts[0], parts[1], parts[2]];

    }
}
