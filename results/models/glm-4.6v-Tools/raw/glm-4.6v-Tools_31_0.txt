// Meet-in-the-middle algorithm for small sets (n <= 40)
fn solve_meet_in_middle(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let half = n / 2;
    
    // Generate all subsets for first half
    let mut left_sums = Vec::new();
    let mut left_indices = Vec::new();
    
    for mask in 0..(1 << half) {
        let mut sum = 0;
        let mut indices = Vec::new();
        for i in 0..half {
            if mask & (1 << i) != 0 {
                sum += numbers[i];
                indices.push(i);
            }
        }
        left_sums.push(sum);
        left_indices.push(indices);
    }
    
    // Generate all subsets for second half
    let mut right_sums = Vec::new();
    let mut right_indices = Vec::new();
    
    for mask in 0..(1 << (n - half)) {
        let mut sum = 0;
        let mut indices = Vec::new();
        for i in 0..(n - half) {
            if mask & (1 << i) != 0 {
                sum += numbers[half + i];
                indices.push(half + i);
            }
        }
        right_sums.push(sum);
        right_indices.push(indices);
    }
    
    // Sort right side for binary search
    let mut right_sorted: Vec<(i64, usize)> = right_sums.iter()
        .enumerate()
        .map(|(i, &sum)| (sum, i))
        .collect();
    right_sorted.sort_by_key(|&(sum, _)| sum);
    
    // Check for complement
    for (i, &sum) in left_sums.iter().enumerate() {
        let complement = target - sum;
        
        // Binary search in right side
        let mut low = 0;
        let mut high = right_sorted.len();
        
        while low < high {
            let mid = (low + high) / 2;
            if right_sorted[mid].0 < complement {
                low = mid + 1;
            } else {
                high = mid;
            }
        }
        
        if low < right_sorted.len() && right_sorted[low].0 == complement {
            let right_idx = right_sorted[low].1;
            let mut result = left_indices[i].clone();
            result.extend_from_slice(&right_indices[right_idx]);
            return Some(result);
        }
    }
    
    None
}

// Bitset DP for medium-sized problems
fn solve_bitset_dp(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let abs_target = target.abs() as usize;
    
    // Use bitset for DP state
    let mut dp = vec![0u64; (abs_target + 63) / 64];
    dp[0 / 64] |= 1 << (0 % 64);
    
    // Track which numbers contribute to each sum
    let mut prev = vec![None; abs_target + 1];
    
    for (i, &num) in numbers.iter().enumerate() {
        let num = num.abs() as usize;
        if num > abs_target {
            continue;
        }
        
        // Update DP backwards to avoid reuse
        for sum in (num..=abs_target).rev() {
            let bit = 1 << (sum % 64);
            let idx = sum / 64;
            
            if (dp[idx] & bit) == 0 && (dp[(sum - num) / 64] & (1 << ((sum - num) % 64))) != 0 {
                dp[idx] |= bit;
                prev[sum] = Some((sum - num, i));
            }
        }
        
        // Early termination if target reached
        if (dp[abs_target / 64] & (1 << (abs_target % 64))) != 0 {
            // Reconstruct solution
            let mut result = Vec::new();
            let mut current = abs_target;
            
            while let Some((prev_sum, idx)) = prev[current] {
                result.push(idx);
                current = prev_sum;
            }
            
            result.reverse();
            return Some(result);
        }
    }
    
    // Check if target is reachable
    if (dp[abs_target / 64] & (1 << (abs_target % 64))) != 0 {
        // Reconstruct solution
        let mut result = Vec::new();
        let mut current = abs_target;
        
        while let Some((prev_sum, idx)) = prev[current] {
            result.push(idx);
            current = prev_sum;
        }
        
        result.reverse();
        return Some(result);
    }
    
    None
}

// Parallel DP for large sets
fn solve_parallel_dp(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let abs_target = target.abs() as usize;
    let num_threads = std::cmp::min(8, n);
    
    // Split numbers into chunks for parallel processing
    let chunk_size = (n + num_threads - 1) / num_threads;
    let chunks: Vec<&[i64]> = numbers.chunks(chunk_size).collect();
    
    let mut dp = vec![0u64; (abs_target + 63) / 64];
    dp[0 / 64] |= 1 << (0 % 64);
    
    let mut prev = vec![None; abs_target + 1];
    
    // Process chunks sequentially but with parallel updates within each chunk
    for chunk in chunks {
        let mut new_dp = dp.clone();
        let mut new_prev = prev.clone();
        
        for (i, &num) in chunk.iter().enumerate() {
            let num = num.abs() as usize;
            if num > abs_target {
                continue;
            }
            
            // Parallelize the inner loop
            let chunk_len = chunk.len();
            let num_threads_inner = std::cmp::min(4, chunk_len);
            let chunk_size_inner = (chunk_len + num_threads_inner - 1) / num_threads_inner;
            
            let mut handles = Vec::new();
            
            for t in 0..num_threads_inner {
                let start = t * chunk_size_inner;
                let end = std::cmp::min((t + 1) * chunk_size_inner, chunk_len);
                
                let dp_clone = dp.clone();
                let prev_clone = prev.clone();
                
                handles.push(thread::spawn(move || {
                    let mut local_dp = dp_clone;
                    let mut local_prev = prev_clone;
                    
                    for sum in (num..=abs_target).rev() {
                        let bit = 1 << (sum % 64);
                        let idx = sum / 64;
                        
                        if (local_dp[idx] & bit) == 0 && (local_dp[(sum - num) / 64] & (1 << ((sum - num) % 64))) != 0 {
                            local_dp[idx] |= bit;
                            local_prev[sum] = Some((sum - num, i + start));
                        }
                    }
                    
                    (local_dp, local_prev)
                }));
            }
            
            // Join threads and merge results
            for handle in handles {
                let (thread_dp, thread_prev) = handle.join().unwrap();
                
                for i in 0..dp.len() {
                    new_dp[i] |= thread_dp[i];
                }
                
                for i in 0..prev.len() {
                    if new_prev[i].is_none() {
                        new_prev[i] = thread_prev[i];
                    }
                }
            }
        }
        
        dp = new_dp;
        prev = new_prev;
        
        // Early termination
        if (dp[abs_target / 64] & (1 << (abs_target % 64))) != 0 {
            // Reconstruct solution
            let mut result = Vec::new();
            let mut current = abs_target;
            
            while let Some((prev_sum, idx)) = prev[current] {
                result.push(idx);
                current = prev_sum;
            }
            
            result.reverse();
            return Some(result);
        }
    }
    
    if (dp[abs_target / 64] & (1 << (abs_target % 64))) != 0 {
        // Reconstruct solution
        let mut result = Vec::new();
        let mut current = abs_target;
        
        while let Some((prev_sum, idx)) = prev[current] {
            result.push(idx);
            current = prev_sum;
        }
        
        result.reverse();
        return Some(result);
    }
    
    None
}

// Heuristic approach for very large sets
fn solve_heuristic(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let abs_target = target.abs() as usize;
    
    // Use greedy approach with backtracking for small subsets
    let mut sorted_indices: Vec<usize> = (0..n).collect();
    sorted_indices.sort_by_key(|&i| numbers[i].abs());
    
    // Try to find subset using BFS with pruning
    let mut queue = VecDeque::new();
    queue.push_back((0, Vec::new(), 0)); // (current_sum, indices, start_index)
    
    let max_queue_size = 1000000;
    
    while let Some((current_sum, indices, start_idx)) = queue.pop_front() {
        if current_sum == target {
            return Some(indices);
        }
        
        if queue.len() > max_queue_size {
            // Too many possibilities, try approximation
            return solve_approximation(numbers, target);
        }
        
        for i in start_idx..n {
            let num = numbers[sorted_indices[i]];
            let new_sum = current_sum + num;
            
            // Prune if sum exceeds target (assuming positive numbers for simplicity)
            if new_sum > target && num > 0 {
                continue;
            }
            
            let mut new_indices = indices.clone();
            new_indices.push(sorted_indices[i]);
            queue.push_back((new_sum, new_indices, i + 1));
        }
    }
    
    None
}

// Approximation algorithm for very large instances
fn solve_approximation(numbers: &[i64], target: i64) -> Option<Vec<usize>> {
    let n = numbers.len();
    let mut current_sum = 0;
    let mut result = Vec::new();
    let mut used = HashSet::new();
    
    // Sort by absolute value
    let mut sorted_indices: Vec<usize> = (0..n).collect();
    sorted_indices.sort_by_key(|&i| numbers[i].abs());
    
    for &i in &sorted_indices {
        if used.contains(&i) {
            continue;
        }
        
        let num = numbers[i];
        let new_sum = current_sum + num;
        
        if new_sum == target {
            result.push(i);
            return Some(result);
        }
        
        if (new_sum < target && num > 0) || (new_sum > target && num < 0) {
            result.push(i);
            current_sum = new_sum;
            used.insert(i);
        }
        
        // Early exit if we're close enough
        if current_sum == target {
            return Some(result);
        }
    }
    
    // Try to adjust by removing some elements
    for &i in &result {
        let num = numbers[i];
        let new_sum = current_sum - num;
        
        if new_sum == target {
            let mut adjusted = result.clone();
            adjusted.retain(|&x| x != i);
            return Some(adjusted);
        }
    }
    
    None
}
