```hlsl
struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);
    
    // Calculate light direction (from surface to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate diffuse term (Lambertian)
    float diffuse = max(dot(N, L), 0.0);
    
    // Polka dot pattern
    float2 cell = floor(input.uv * 6.0);
    float2 center = (cell + 0.5) / 6.0;
    float2 toCenter = input.uv - center;
    float dist = length(toCenter);
    
    float3 patternColor;
    if (dist < 0.3) {
        patternColor = float3(1.0, 0.8, 0.0); // Yellow dots
    } else {
        patternColor = float3(0.1, 0.1, 0.3); // Dark blue background
    }
    
    // Apply Lambertian shading factor
    float3 shadedColor = patternColor * (0.3 + 0.7 * diffuse);
    
    // Combine with per-vertex color
    float3 finalColor = shadedColor * input.color;
    
    return float4(finalColor, 1.0);
}
```
