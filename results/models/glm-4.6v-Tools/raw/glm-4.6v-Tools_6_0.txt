    public int[] Solve(Orbit startOrbit, List<Orbit> stations, double mu)
    {
        int n = stations.Count;
        var distanceMatrix = new double[n, n];
        var toStartDistances = new double[n];

        // Precompute all distances
        Parallel.For(0, n, i =>
        {
            toStartDistances[i] = ComputeDeltaV(startOrbit, stations[i], mu);
            for (int j = i + 1; j < n; j++)
            {
                distanceMatrix[i, j] = distanceMatrix[j, i] = ComputeDeltaV(stations[i], stations[j], mu);
            }
        });

        // Choose algorithm based on problem size
        if (n <= 5)
        {
            return SolveExact(startOrbit, stations, toStartDistances, distanceMatrix);
        }
        else if (n <= 10)
        {
            return SolveHeuristic(startOrbit, stations, toStartDistances, distanceMatrix, use2Opt: true);
        }
        else if (n <= 20)
        {
            return SolveLarge(startOrbit, stations, toStartDistances, distanceMatrix);
        }
        else
        {
            return SolveExtreme(startOrbit, stations, toStartDistances, distanceMatrix);
        }
    }

    private int[] SolveExact(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, double[,] distanceMatrix)
    {
        int n = stations.Count;
        var bestPath = new int[n];
        var currentPath = new int[n];
        var visited = new bool[n];
        double bestCost = double.MaxValue;

        // Try all permutations starting from each station
        for (int start = 0; start < n; start++)
        {
            Array.Fill(visited, false);
            currentPath[0] = start;
            visited[start] = true;
            FindBestPermutation(startOrbit, stations, toStartDistances, distanceMatrix, currentPath, visited, 1, 0, ref bestCost, bestPath);
        }

        return bestPath;
    }

    private void FindBestPermutation(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, 
        double[,] distanceMatrix, int[] currentPath, bool[] visited, int depth, double currentCost, 
        ref double bestCost, int[] bestPath)
    {
        int n = stations.Count;
        if (depth == n)
        {
            // Complete the tour by returning to start
            double totalCost = currentCost + toStartDistances[currentPath[n - 1]];
            if (totalCost < bestCost)
            {
                bestCost = totalCost;
                Array.Copy(currentPath, bestPath, n);
            }
            return;
        }

        for (int i = 0; i < n; i++)
        {
            if (!visited[i])
            {
                visited[i] = true;
                currentPath[depth] = i;
                
                double additionalCost = (depth == 0) ? toStartDistances[i] : distanceMatrix[currentPath[depth - 1], i];
                FindBestPermutation(startOrbit, stations, toStartDistances, distanceMatrix, 
                    currentPath, visited, depth + 1, currentCost + additionalCost, 
                    ref bestCost, bestPath);
                
                visited[i] = false;
            }
        }
    }

    private int[] SolveHeuristic(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, double[,] distanceMatrix, bool use2Opt)

        if (use2Opt)

            path = TwoOptOptimization(path, distanceMatrix);
        }
        
        return path;
    }

    private int[] SolveLarge(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, double[,] distanceMatrix)

    private int[] NearestNeighbor(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, double[,] distanceMatrix)
    {
        int n = stations.Count;
        var path = new int[n];
        var visited = new bool[n];
        int current = -1;
        double minDist = double.MaxValue;

        // Find nearest station to start
        for (int i = 0; i < n; i++)
        {
            if (toStartDistances[i] < minDist)
            {
                minDist = toStartDistances[i];
                current = i;
            }
        }

        path[0] = current;
        visited[current] = true;

        // Build path by always going to nearest unvisited station
        for (int i = 1; i < n; i++)
        {
            minDist = double.MaxValue;
            int next = -1;
            for (int j = 0; j < n; j++)
            {
                if (!visited[j] && distanceMatrix[path[i - 1], j] < minDist)
                {
                    minDist = distanceMatrix[path[i - 1], j];
                    next = j;
                }
            }
            path[i] = next;
            visited[next] = true;
        }

        return path;
    }

    private int[] ChristofidesAlgorithm(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, double[,] distanceMatrix)
    {
        int n = stations.Count;
        var mst = BuildMST(distanceMatrix, n);
        var oddDegreeNodes = FindOddDegreeNodes(mst, n);
        var perfectMatching = FindMinimumWeightPerfectMatching(oddDegreeNodes, distanceMatrix);
        
        // Combine MST and perfect matching to form Eulerian circuit
        var eulerianCircuit = CombineMSTAndMatching(mst, perfectMatching, n);
        
        // Convert Eulerian circuit to Hamiltonian path
        var path = ConvertEulerianToHamiltonian(eulerianCircuit, n);
        return path;
    }

    private int[,] BuildMST(double[,] distanceMatrix, int n)
    {
        var mst = new int[n, 2];
        var inMST = new bool[n];
        var minEdge = new double[n];
        var parent = new int[n];

        for (int i = 0; i < n; i++)
        {
            minEdge[i] = double.MaxValue;
            inMST[i] = false;
        }

        minEdge[0] = 0;
        parent[0] = -1;

        for (int count = 0; count < n - 1; count++)
        {
            int u = -1;
            double min = double.MaxValue;

            for (int v = 0; v < n; v++)
            {
                if (!inMST[v] && minEdge[v] < min)
                {
                    min = minEdge[v];
                    u = v;
                }
            }

            inMST[u] = true;

            for (int v = 0; v < n; v++)
            {
                if (!inMST[v] && distanceMatrix[u, v] < minEdge[v])
                {
                    minEdge[v] = distanceMatrix[u, v];
                    parent[v] = u;
                }
            }
        }

        for (int i = 1; i < n; i++)
        {
            mst[i, 0] = parent[i];
            mst[i, 1] = i;
        }

        return mst;
    }

    private List<int> FindOddDegreeNodes(int[,] mst, int n)
    {
        var degree = new int[n];
        for (int i = 1; i < n; i++)
        {
            degree[mst[i, 0]]++;
            degree[mst[i, 1]]++;
        }

        var oddNodes = new List<int>();
        for (int i = 0; i < n; i++)
        {
            if (degree[i] % 2 != 0)
            {
                oddNodes.Add(i);
            }
        }
        return oddNodes;
    }

    private List<(int, int)> FindMinimumWeightPerfectMatching(List<int> oddNodes, double[,] distanceMatrix)
    {
        var matching = new List<(int, int)>();
        var visited = new bool[oddNodes.Count];

        while (matching.Count < oddNodes.Count / 2)
        {
            double minWeight = double.MaxValue;
            int u = -1, v = -1;

            for (int i = 0; i < oddNodes.Count; i++)
            {
                if (!visited[i])
                {
                    for (int j = i + 1; j < oddNodes.Count; j++)
                    {
                        if (!visited[j] && distanceMatrix[oddNodes[i], oddNodes[j]] < minWeight)
                        {
                            minWeight = distanceMatrix[oddNodes[i], oddNodes[j]];
                            u = i;
                            v = j;
                        }
                    }
                }
            }

            visited[u] = true;
            visited[v] = true;
            matching.Add((oddNodes[u], oddNodes[v]));
        }

        return matching;
    }

    private List<int> CombineMSTAndMatching(int[,] mst, List<(int, int)> matching, int n)
    {
        var graph = new List<int>[n];
        for (int i = 0; i < n; i++)
        {
            graph[i] = new List<int>();
        }

        for (int i = 1; i < n; i++)
        {
            int u = mst[i, 0];
            int v = mst[i, 1];
            graph[u].Add(v);
            graph[v].Add(u);
        }

        foreach (var (u, v) in matching)
        {
            graph[u].Add(v);
            graph[v].Add(u);
        }

        // Find Eulerian circuit
        return FindEulerianCircuit(graph, 0);
    }

    private List<int> FindEulerianCircuit(List<int>[] graph, int start)
    {
        var circuit = new List<int>();
        var stack = new Stack<int>();
        stack.Push(start);

        while (stack.Count > 0)
        {
            int u = stack.Peek();
            if (graph[u].Count == 0)
            {
                circuit.Add(u);
                stack.Pop();
            }
            else
            {
                int v = graph[u][0];
                graph[u].Remove(v);
                graph[v].Remove(u);
                stack.Push(v);
            }
        }

        circuit.Reverse();
        return circuit;
    }

    private int[] ConvertEulerianToHamiltonian(List<int> eulerianCircuit, int n)
    {
        var visited = new bool[n];
        var path = new int[n];
        int pathIndex = 0;

        foreach (var node in eulerianCircuit)
        {
            if (!visited[node])
            {
                visited[node] = true;
                path[pathIndex++] = node;
            }
        }

        return path;
    }

    private int[] AntColonyOptimization(Orbit startOrbit, List<Orbit> stations, double[] toStartDistances, double[,] distanceMatrix)
    {
        int n = stations.Count;
        int numAnts = Math.Min(50, n * 2);
        int maxIterations = 100;
        double alpha = 1.0;
        double beta = 2.0;
        double evaporationRate = 0.5;
        double Q = 1.0;

        var pheromone = new double[n, n];
        for (int i = 0; i < n; i++)
        {
            for (int j = 0; j < n; j++)
            {
                pheromone[i, j] = 1.0;
            }
        }

        var bestPath = new int[n];
        double bestCost = double.MaxValue;

        for (int iter = 0; iter < maxIterations; iter++)
        {
            var allPaths = new List<int[]>();
            var allCosts = new List<double>();

            Parallel.For(0, numAnts, ant =>
            {
                var path = new int[n];
                var visited = new bool[n];
                int current = -1;
                double minDist = double.MaxValue;

                // Find nearest station to start
                for (int i = 0; i < n; i++)
                {
                    if (toStartDistances[i] < minDist)
                    {
                        minDist = toStartDistances[i];
                        current = i;
                    }
                }

                path[0] = current;
                visited[current] = true;

                for (int i = 1; i < n; i++)
                {
                    double[] probabilities = new double[n];
                    double total = 0.0;

                    for (int j = 0; j < n; j++)
                    {
                        if (!visited[j])
                        {
                            probabilities[j] = Math.Pow(pheromone[current, j], alpha) * 
                                             Math.Pow(1.0 / (distanceMatrix[current, j] + 1e-10), beta);
                            total += probabilities[j];
                        }
                    }

                    double rand = new Random(ant + iter).NextDouble() * total;
                    double cumulative = 0.0;

                    for (int j = 0; j < n; j++)
                    {
                        if (!visited[j])
                        {
                            cumulative += probabilities[j];
                            if (cumulative >= rand)
                            {
                                path[i] = j;
                                visited[j] = true;
                                current = j;
                                break;
                            }
                        }
                    }
                }

                allPaths.Add(path);
                allCosts.Add(CalculatePathCost(path, toStartDistances, distanceMatrix));
            });

            // Update pheromones
            for (int i = 0; i < n; i++)
            {
                for (int j = 0; j < n; j++)
                {
                    pheromone[i, j] *= (1.0 - evaporationRate);
                }
            }

            for (int a = 0; a < numAnts; a++)
            {
                double cost = allCosts[a];
                for (int i = 0; i < n - 1; i++)
                {
                    pheromone[allPaths[a][i], allPaths[a][i + 1]] += Q / cost;
                }
                pheromone[allPaths[a][n - 1], allPaths[a][0]] += Q / cost;
            }

            // Find best path this iteration
            for (int a = 0; a < numAnts; a++)
            {
                if (allCosts[a] < bestCost)
                {
                    bestCost = allCosts[a];
                    Array.Copy(allPaths[a], bestPath, n);
                }
            }
        }

        return bestPath;
    }

    private int[] TwoOptOptimization(int[] path, double[,] distanceMatrix)
    {
        int n = path.Length;
        bool improved = true;
        var newPath = (int[])path.Clone();

        while (improved)
        {
            improved = false;
            for (int i = 0; i < n - 1; i++)
            {
                for (int k = i + 2; k < n; k++)
                {
                    double delta = -distanceMatrix[newPath[i], newPath[i + 1]] 
                                - distanceMatrix[newPath[k], newPath[(k + 1) % n]]
                                + distanceMatrix[newPath[i], newPath[k]] 
                                + distanceMatrix[newPath[i + 1], newPath[(k + 1) % n]];

                    if (delta < -1e-10)
                    {
                        // Reverse the segment from i+1 to k
                        ReverseSegment(newPath, i + 1, k);
                        improved = true;
                    }
                }
            }
        }

        return newPath;
    }

    private void ReverseSegment(int[] path, int start, int end)
    {
        while (start < end)
        {
            int temp = path[start];
            path[start] = path[end];
            path[end] = temp;
            start++;
            end--;
        }
    }

    private double CalculatePathCost(int[] path, double[] toStartDistances, double[,] distanceMatrix)
    {
        double cost = 0.0;
        int n = path.Length;

        for (int i = 0; i < n - 1; i++)
        {
            cost += distanceMatrix[path[i], path[i + 1]];
        }
        cost += toStartDistances[path[n - 1]];

        return cost;
    }

    private double ComputeDeltaV(Orbit from, Orbit to, double mu)
    {
        // Simplified delta-V calculation using Lambert's problem approximation
        // In a real implementation, you would solve Lambert's problem here
        double r1 = Math.Sqrt(from.X * from.X + from.Y * from.Y + from.Z * from.Z);
        double r2 = Math.Sqrt(to.X * to.X + to.Y * to.Y + to.Z * to.Z);
        
        // Approximate transfer time (simplified)
        double transferTime = Math.Sqrt(Math.Pow(r1 + r2, 3) / mu) * Math.PI;
        
        // Approximate delta-V (simplified)
        double v1 = Math.Sqrt(mu / r1);
        double v2 = Math.Sqrt(mu / r2);
        
        return Math.Abs(v1 - v2) * 0.5; // Simplified approximation
    }
