I'll write a C++ program that solves the Maximum Independent Set problem exactly using a branch-and-bound algorithm with bitmask representation. This approach is efficient for graphs with up to around 60 vertices, which should handle most practical cases within time limits.

```cpp
#include <bits/stdc++.h>
using namespace std;

// Global variables
int n, m;
vector<int> adj[60];
int best_size = 0;
vector<int> best_solution;

// Bitmask operations
using ull = unsigned long long;

// Check if vertex v is adjacent to any vertex in the current set
bool is_adjacent_to_set(int v, ull current_set) {
    for (int u : adj[v]) {
        if ((current_set >> u) & 1) return true;
    }
    return false;
}

// Branch and bound algorithm
void branch_and_bound(int v, ull current_set, int current_size) {
    // Prune if current set cannot beat the best solution
    int remaining_vertices = n - v;
    if (current_size + remaining_vertices <= best_size) return;
    
    // Try to include vertex v
    if (v < n && !is_adjacent_to_set(v, current_set)) {
        ull new_set = current_set | (1ULL << v);
        if (current_size + 1 > best_size) {
            best_size = current_size + 1;
            best_solution.clear();
            for (int i = 0; i < n; ++i) {
                if ((new_set >> i) & 1) best_solution.push_back(i);
            }
        }
        branch_and_bound(v + 1, new_set, current_size + 1);
    }
    
    // Try to exclude vertex v
    branch_and_bound(v + 1, current_set, current_size);
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);
    
    // Read input
    cin >> n >> m;
    for (int i = 0; i < m; ++i) {
        int u, v;
        cin >> u >> v;
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
    
    // Special case: empty graph
    if (n == 0) {
        cout << "0\n\n";
        return 0;
    }
    
    // Initialize best solution
    best_size = 0;
    best_solution.clear();
    
    // Run branch and bound
    branch_and_bound(0, 0ULL, 0);
    
    // Output result
    cout << best_size << "\n";
    for (int i = 0; i < best_size; ++i) {
        if (i > 0) cout << " ";
        cout << best_solution[i];
    }
    cout << "\n";
    
    return 0;
}
```

This program implements an exact algorithm for the Maximum Independent Set problem using branch and bound with bitmask representation. Here's how it works:

1. **Input Handling**: Reads the graph structure from standard input.
2. **Branch and Bound Algorithm**: 
   - Uses a recursive approach to explore both including and excluding each vertex
   - Uses bitmask operations for efficient set manipulation
   - Prunes branches that cannot possibly improve the current best solution
3. **Output**: Prints the size of the maximum independent set and the vertices in the solution

The algorithm is efficient for graphs with up to around 60 vertices, which should be sufficient for most practical cases. For very large graphs, exact solutions may be computationally expensive, but this implementation provides the guaranteed optimal solution as required.
