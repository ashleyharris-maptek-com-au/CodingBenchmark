[[vk::location(2)]] float2 uv       : TEXCOORD2;
[[vk::location(3)]] float3 tangent  : TEXCOORD3;

    float4x4 model;
    float4x4 view;

float4 main(PSInput input) : SV_Target0 {
    float2 c = float2(uv.x * 3 - 2, uv.y * 2 - 1);
    float2 z = float2(0, 0);
    int iterations = 0;
    const int maxIterations = 20;
    
    for (int i = 0; i < maxIterations; i++) {
        float zr2 = z.x * z.x;
        float zi2 = z.y * z.y;
        
        if (zr2 + zi2 > 4.0) break;
        
        z.y = 2 * z.x * z.y + c.y;
        z.x = zr2 - zi2 + c.x;
        iterations++;
    }
    
    float t = float(iterations) / float(maxIterations);
    return float4(t, t * 0.5, 1.0 - t, 1.0);
}
