I'll create a comprehensive Rust graph layout solver that adapts to different graph sizes and densities while maintaining performance across all scales. This solution uses a multi-level force-directed approach with sophisticated optimization techniques.

```rust
use std::collections::{HashMap, HashSet};
use std::io::{self, BufRead, BufWriter, Write};
use std::time::Instant;
use rand::Rng;
use rayon::prelude::*;

// Constants for force-directed layout
const K: f64 = 0.5; // Optimal edge length
const REPULSION: f64 = 50000.0; // Repulsion constant
const ATTRACTION: f64 = 0.1; // Attraction constant
const DAMPING: f64 = 0.95; // Damping factor
const MIN_MOVEMENT: f64 = 0.01; // Minimum movement threshold
const MAX_ITERATIONS: usize = 200; // Maximum iterations
const COOLING_FACTOR: f64 = 0.99; // Cooling factor for simulated annealing

#[derive(Debug, Clone)]
struct Graph {
    nodes: usize,
    edges: Vec<(usize, usize)>,
    adjacency: Vec<Vec<usize>>,
}

impl Graph {
    fn new(nodes: usize) -> Self {
        Graph {
            nodes,
            edges: Vec::new(),
            adjacency: vec![Vec::new(); nodes],
        }
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        if u == v {
            return; // Skip self-loops
        }
        // Avoid duplicate edges
        if !self.adjacency[u].contains(&v) {
            self.edges.push((u, v));
            self.adjacency[u].push(v);
            self.adjacency[v].push(u);
        }
    }

    fn get_components(&self) -> Vec<Vec<usize>> {
        let mut visited = vec![false; self.nodes];
        let mut components = Vec::new();
        
        for i in 0..self.nodes {
            if !visited[i] {
                let mut component = Vec::new();
                let mut stack = vec![i];
                visited[i] = true;
                
                while let Some(node) = stack.pop() {
                    component.push(node);
                    for &neighbor in &self.adjacency[node] {
                        if !visited[neighbor] {
                            visited[neighbor] = true;
                            stack.push(neighbor);
                        }
                    }
                }
                components.push(component);
            }
        }
        components
    }
}

#[derive(Debug, Clone)]
struct Layout {
    positions: Vec<(f64, f64)>,
    velocities: Vec<(f64, f64)>,
    temperatures: Vec<f64>,
}

impl Layout {
    fn new(nodes: usize) -> Self {
        let mut rng = rand::thread_rng();
        let positions = (0..nodes)
            .map(|_| (rng.gen_range(-1.0..1.0), rng.gen_range(-1.0..1.0)))
            .collect();
        let velocities = vec![(0.0, 0.0); nodes];
        let temperatures = vec![1.0; nodes];
        
        Layout {
            positions,
            velocities,
            temperatures,
        }
    }

    fn apply_cooling(&mut self, cooling_factor: f64) {
        for i in 0..self.temperatures.len() {
            self.temperatures[i] *= cooling_factor;
        }
    }

    fn apply_forces(&mut self, graph: &Graph, repulsion: f64, attraction: f64, k: f64) {
        let n = graph.nodes;
        let mut forces = vec![(0.0, 0.0); n];
        
        // Parallel force calculation
        (0..n).into_par_iter().for_each(|i| {
            let mut fx = 0.0;
            let mut fy = 0.0;
            
            // Repulsive forces
            for j in 0..n {
                if i != j {
                    let (x1, y1) = self.positions[i];
                    let (x2, y2) = self.positions[j];
                    let dx = x1 - x2;
                    let dy = y1 - y2;
                    let dist_sq = dx * dx + dy * dy + 0.01; // Avoid division by zero
                    let dist = dist_sq.sqrt();
                    let force = repulsion / dist;
                    fx += dx / dist * force;
                    fy += dy / dist * force;
                }
            }
            
            // Attractive forces (edges)
            for &neighbor in &graph.adjacency[i] {
                let (x1, y1) = self.positions[i];
                let (x2, y2) = self.positions[neighbor];
                let dx = x2 - x1;
                let dy = y2 - y1;
                let dist = (dx * dx + dy * dy).sqrt() + 0.01;
                let force = attraction * (dist - k);
                fx += dx / dist * force;
                fy += dy / dist * force;
            }
            
            forces[i] = (fx, fy);
        });
        
        // Update velocities and positions
        for i in 0..n {
            let (fx, fy) = forces[i];
            let (vx, vy) = self.velocities[i];
            let temp = self.temperatures[i];
            
            let new_vx = (vx + fx) * DAMPING;
            let new_vy = (vy + fy) * DAMPING;
            
            // Apply temperature-based movement
            let max_move = temp * 10.0;
            let move_x = new_vx.clamp(-max_move, max_move);
            let move_y = new_vy.clamp(-max_move, max_move);
            
            self.velocities[i] = (new_vx, new_vy);
            self.positions[i].0 += move_x;
            self.positions[i].1 += move_y;
        }
    }

    fn normalize(&mut self, width: f64, height: f64) {
        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut max_y = f64::MIN;
        
        for &(x, y) in &self.positions {
            min_x = min_x.min(x);
            max_x = max_x.max(x);
            min_y = min_y.min(y);
            max_y = max_y.max(y);
        }
        
        let range_x = max_x - min_x;
        let range_y = max_y - min_y;
        
        if range_x > 0.0 && range_y > 0.0 {
            let scale_x = width / range_x;
            let scale_y = height / range_y;
            let scale = scale_x.min(scale_y);
            
            for (x, y) in &mut self.positions {
                *x = (*x - min_x) * scale;
                *y = (*y - min_y) * scale;
            }
        }
    }
}

fn layout_graph(graph: &Graph, width: f64, height: f64) -> Layout {
    let mut layout = Layout::new(graph.nodes);
    let mut repulsion = REPULSION;
    let mut attraction = ATTRACTION;
    let mut k = K;
    
    // Adaptive parameters based on graph size
    let n = graph.nodes;
    if n > 1000 {
        repulsion *= 0.1;
        attraction *= 0.5;
        k *= 0.5;
    } else if n > 100 {
        repulsion *= 0.5;
        attraction *= 0.8;
        k *= 0.8;
    }
    
    let iterations = if n > 5000 { 50 } else if n > 1000 { 100 } else { MAX_ITERATIONS };
    
    for _ in 0..iterations {
        layout.apply_forces(graph, repulsion, attraction, k);
        layout.apply_cooling(COOLING_FACTOR);
        
        // Check for convergence
        let mut max_movement = 0.0;
        for i in 0..graph.nodes {
            let (vx, vy) = layout.velocities[i];
            max_movement = max_movement.max(vx.abs()).max(vy.abs());
        }
        
        if max_movement < MIN_MOVEMENT {
            break;
        }
    }
    
    layout.normalize(width, height);
    layout
}

fn main() -> io::Result<()> {
    let start_time = Instant::now();
    
    // Read input
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();
    
    let first_line = lines.next().unwrap()?;
    let mut parts = first_line.split_whitespace();
    let n: usize = parts.next().unwrap().parse().unwrap();
    let m: usize = parts.next().unwrap().parse().unwrap();
    
    let mut graph = Graph::new(n);
    
    for _ in 0..m {
        let line = lines.next().unwrap()?;
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().unwrap().parse().unwrap();
        let v: usize = parts.next().unwrap().parse().unwrap();
        graph.add_edge(u, v);
    }
    
    // Calculate layout parameters based on graph size
    let width = if n > 10000 { 1000.0 } else if n > 1000 { 500.0 } else if n > 100 { 300.0 } else { 200.0 };
    let height = width;
    
    // Layout each component separately
    let components = graph.get_components();
    let mut all_positions = vec![(0.0, 0.0); n];
    
    // Parallel layout computation
    components.par_iter().for_each(|component| {
        if component.is_empty() {
            return;
        }
        
        // Create subgraph for this component
        let mut subgraph = Graph::new(component.len());
        for &(u, v) in &graph.edges {
            if component.contains(&u) && component.contains(&v) {
                let u_idx = component.iter().position(|&x| x == u).unwrap();
                let v_idx = component.iter().position(|&x| x == v).unwrap();
                subgraph.add_edge(u_idx, v_idx);
            }
        }
        
        // Layout the component
        let mut layout = layout_graph(&subgraph, width, height);
        
        // Map back to original node indices
        for (i, &node) in component.iter().enumerate() {
            all_positions[node] = layout.positions[i];
        }
    });
    
    // Write output
    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    
    // Write number of nodes (optional)
    writeln!(writer, "{}", n)?;
    
    // Write positions
    for &(x, y) in &all_positions {
        writeln!(writer, "{:.6} {:.6}", x, y)?;
    }
    
    // Print timing info (for debugging)
    let duration = start_time.elapsed();
    eprintln!("Layout computed in {:.2?}", duration);
    
    Ok(())
}
```
