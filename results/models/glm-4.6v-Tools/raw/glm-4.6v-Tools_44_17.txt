```glsl
#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize interpolated normal
    vec3 N = normalize(normal);
    
    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    
    // Calculate diffuse lighting (Lambertian)
    float diffuse = max(dot(N, L), 0.0);
    
    // Procedural brick pattern
    vec2 st = uv * vec2(8.0, 16.0);  // 8x16 scale
    
    // Staggered rows (half-brick offset every other row)
    if (fract(st.y * 0.5) > 0.5) {
        st.x += 0.5;
    }
    
    // Get brick and mortar colors
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    
    // Calculate brick pattern with mortar
    vec2 brick = fract(st);
    vec2 mortar = step(vec2(0.05, 0.1), brick);
    vec2 brickMask = 1.0 - mortar;
    
    // Mix colors based on mortar/brick mask
    vec3 patternColor = mix(mortarColor, brickColor, brickMask.x * brickMask.y);
    
    // Apply Lambertian shading with factor (0.4 + 0.6*diffuse)
    vec3 shadedColor = patternColor * (0.4 + 0.6 * diffuse);
    
    // Output final color with alpha 1.0
    outColor = vec4(shadedColor, 1.0);
}
```
