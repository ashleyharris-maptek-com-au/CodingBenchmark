# GLSL Compute Shader for 3D Boid Flocking Simulation

```glsl
#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint i = gl_GlobalInvocationID.x;
    uint N = counts.x;
    
    // Guard against out-of-bounds
    if (i >= N) {
        return;
    }
    
    // Read boid data
    vec3 pos = inData[i * 2].xyz;
    vec3 vel = inData[i * 2 + 1].xyz;
    
    // Initialize forces
    vec3 sepForce = vec3(0.0);
    vec3 alignForce = vec3(0.0);
    vec3 cohForce = vec3(0.0);
    
    uint sepCount = 0;
    uint alignCount = 0;
    uint cohCount = 0;
    
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;
    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;
    float boundSize = bounds.x;
    float boundForce = bounds.y;
    
    // Check all other boids
    for (uint j = 0; j < N; j++) {
        if (i == j) continue; // Skip self
        
        vec3 otherPos = inData[j * 2].xyz;
        vec3 otherVel = inData[j * 2 + 1].xyz;
        
        // Calculate distance
        vec3 diff = pos - otherPos;
        float dist = length(diff);
        
        // Skip if too far away
        if (dist > cohRadius) continue;
        
        // Separation: avoid crowding neighbors
        if (dist < sepRadius && dist > 0.0001) {
            vec3 repel = normalize(diff) / dist; // Weight by distance
            sepForce += repel;
            sepCount++;
        }
        
        // Alignment: steer towards average velocity of neighbors
        if (dist < alignRadius) {
            alignForce += otherVel;
            alignCount++;
        }
        
        // Cohesion: steer towards average position of neighbors
        if (dist < cohRadius) {
            cohForce += otherPos;
            cohCount++;
        }
    }
    
    // Average the forces by neighbor count
    if (sepCount > 0) {
        sepForce = sepForce / float(sepCount) * sepWeight;
    }
    
    if (alignCount > 0) {
        alignForce = (alignForce / float(alignCount) - vel) * alignWeight;
    }
    
    if (cohCount > 0) {
        vec3 center = cohForce / float(cohCount);
        cohForce = (center - pos) * cohWeight;
    }
    
    // Apply boundary force (push towards center when near edges)
    for (int axis = 0; axis < 3; axis++) {
        if (pos[axis] > boundSize * 0.4) {
            float distFromEdge = boundSize - pos[axis];
            float force = boundForce * (distFromEdge / (boundSize * 0.5));
            vel[axis] -= force;
        } else if (pos[axis] < -boundSize * 0.4) {
            float distFromEdge = -boundSize - pos[axis];
            float force = boundForce * (distFromEdge / (boundSize * 0.5));
            vel[axis] -= force;
        }
    }
    
    // Combine all forces
    vec3 accel = sepForce + alignForce + cohForce;
    
    // Update velocity
    vel += accel * dt;
    
    // Clamp speed to maxSpeed
    float speed = length(vel);
    if (speed > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }
    
    // Update position
    pos += vel * dt;
    
    // Write updated boid data
    outData[i * 2] = vec4(pos, 0.0);
    outData[i * 2 + 1] = vec4(vel, 0.0);
}
```

## Key Features:

1. **Reynolds' Three Rules Implementation:**
   - **Separation**: Avoids crowding neighbors within `sepRadius` (2.0)
   - **Alignment**: Steers towards average velocity of neighbors within `alignRadius` (5.0)
   - **Cohesion**: Steers towards average position of neighbors within `cohRadius` (8.0)

2. **Boundary Handling**: 
   - Applies force when boids are within 40% of boundary
   - Pushes boids toward center using `boundForce` (0.5) and scaled distance

3. **Force Averaging**: 
   - Each force is divided by neighbor count to prevent accumulation bias
   - Zero-division protection with neighbor count checks

4. **Velocity/Position Update**:
   - Velocity updated by acceleration × dt
   - Speed clamped to `maxSpeed` (5.0)
   - Position updated by velocity × dt

5. **Performance Optimizations**:
   - Early exit for out-of-bounds boids
   - Skip self and distant neighbors
   - Uses efficient distance calculations

The shader performs one timestep of the simulation and can be dispatched repeatedly with ping-pong buffers for the required number of timesteps (50-200 depending on configuration).
