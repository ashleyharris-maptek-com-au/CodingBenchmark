struct Point {
    x: usize,
    y: usize,
}

impl Point {
    fn new(x: usize, y: usize) -> Self {
        Point { x, y }
    }

    fn neighbors(&self, width: usize, height: usize) -> Vec<Point> {
        let mut result = Vec::with_capacity(4);
        if self.x > 0 {
            result.push(Point::new(self.x - 1, self.y));
        }
        if self.x < width - 1 {
            result.push(Point::new(self.x + 1, self.y));
        }
        if self.y > 0 {
            result.push(Point::new(self.x, self.y - 1));
        }
        if self.y < height - 1 {
            result.push(Point::new(self.x, self.y + 1));
        }
        result
    }
}

struct Grid {

    height: usize,

impl Grid {
    fn new(width: usize, height: usize, obstacles: HashSet<Point>, require_cycle: bool) -> Self {
        let total_cells = width * height - obstacles.len();
        Grid {
            width,
            height,
            obstacles,
            require_cycle,
            total_cells,
        }
    }

    fn is_obstacle(&self, p: &Point) -> bool {
        self.obstacles.contains(p)
    }

    fn is_valid(&self, p: &Point) -> bool {
        p.x < self.width && p.y < self.height && !self.is_obstacle(p)
    }

    fn is_corner(&self, p: &Point) -> bool {
        (p.x == 0 || p.x == self.width - 1) && (p.y == 0 || p.y == self.height - 1)
    }

    fn get_degree(&self, p: &Point) -> usize {
        p.neighbors(self.width, self.height)
            .iter()
            .filter(|n| self.is_valid(n))
            .count()
    }

    fn get_dead_ends(&self) -> Vec<Point> {

                if !self.is_obstacle(&p) && self.get_degree(&p) == 1 {

    forward_queue.push_back(start);
    backward_queue.push_back(end);

    let mut backward_visited = HashMap::new();
    
    forward_visited.insert(start, None);
    backward_visited.insert(end, None);

    let directions = [(1, 0), (-1, 0), (0, 1), (0, -1)];

            &mut forward_queue,
            &mut forward_visited,
            &backward_visited,
            &directions,
            w,
            h,
        ) {
            meeting_point = Some(meet);
            break;

            &maze,
            &mut backward_queue,
            &mut backward_visited,
            &forward_visited,
            &directions,

            h,

            break;

        let mut path = reconstruct_path(&forward_visited, start, meeting_point);
        let mut backward_path = reconstruct_path(&backward_visited, end, meeting_point);

fn expand_search(
    maze: &[Vec<char>],
    queue: &mut VecDeque<(usize, usize)>,
    visited: &mut HashMap<(usize, usize), Option<(usize, usize)>>,
    other_visited: &HashMap<(usize, usize), Option<(usize, usize)>>,
    directions: &[(isize, isize)],
    w: usize,
    h: usize,
) -> Option<(usize, usize)> {
    let (x, y) = queue.pop_front().unwrap();
    
    for &(dx, dy) in directions {
        let nx = x as isize + dx;
        let ny = y as isize + dy;
        
        if nx < 0 || ny < 0 || nx >= w as isize || ny >= h as isize {
            continue;

        let ny = ny as usize;

            continue;

            continue;

            return Some((nx, ny));
        }

        queue.push_back((nx, ny));
    }

}

struct Polygon {
    vertices: Vec<Point>,
    bounding_box: (f64, f64, f64, f64), // min_x, min_y, max_x, max_y
}

impl Polygon {
    fn new(vertices: Vec<Point>) -> Self {
        let mut min_x = f64::MAX;
        let mut min_y = f64::MAX;
        let mut max_x = f64::MIN;
        let mut max_y = f64::MIN;
        
        for v in &vertices {
            min_x = min_x.min(v.x);
            min_y = min_y.min(v.y);
            max_x = max_x.max(v.x);
            max_y = max_y.max(v.y);
        }
        
        Polygon {
            vertices,
            bounding_box: (min_x, min_y, max_x, max_y),
        }
    }
    
    fn area(&self) -> f64 {
        let mut area = 0.0;
        let n = self.vertices.len();
        for i in 0..n {
            let j = (i + 1) % n;
            area += self.vertices[i].x * self.vertices[j].y;
            area -= self.vertices[j].x * self.vertices[i].y;
        }
        area.abs() / 2.0
    }
    
    fn contains_point(&self, point: &Point) -> bool {
        let mut inside = false;
        let n = self.vertices.len();
        
        for i in 0..n {
            let j = (i + 1) % n;
            let (vi, vj) = (&self.vertices[i], &self.vertices[j]);
            
            if ((vi.y > point.y) != (vj.y > point.y)) &&
               (point.x < (vj.x - vi.x) * (point.y - vi.y) / (vj.y - vi.y) + vi.x) {
                inside = !inside;
            }
        }
        
        inside
    }
    
    fn intersects(&self, other: &Polygon) -> bool {
        // Check bounding box overlap first
        let (min_x1, min_y1, max_x1, max_y1) = self.bounding_box;
        let (min_x2, min_y2, max_x2, max_y2) = other.bounding_box;
        
        if max_x1 < min_x2 || max_x2 < min_x1 || max_y1 < min_y2 || max_y2 < min_y1 {
            return false;
        }
        
        // Check if any edge of self intersects with any edge of other
        let n1 = self.vertices.len();
        let n2 = other.vertices.len();
        
        for i in 0..n1 {
            let j = (i + 1) % n1;
            let edge1 = (&self.vertices[i], &self.vertices[j]);
            
            for k in 0..n2 {
                let l = (k + 1) % n2;
                let edge2 = (&other.vertices[k], &other.vertices[l]);
                
                if self.lines_intersect(edge1, edge2) {
                    return true;
                }
            }
        }
        
        // Check if one polygon contains the other
        if self.contains_polygon(other) || other.contains_polygon(self) {
            return true;
        }
        
        false
    }
    
    fn lines_intersect(&self, (p1, p2): (&Point, &Point), (p3, p4): (&Point, &Point)) -> bool {
        let d1 = self.direction(p3, p4, p1);
        let d2 = self.direction(p3, p4, p2);
        let d3 = self.direction(p1, p2, p3);
        let d4 = self.direction(p1, p2, p4);
        
        ((d1 > 0.0 && d2 < 0.0) || (d1 < 0.0 && d2 > 0.0)) &&
        ((d3 > 0.0 && d4 < 0.0) || (d3 < 0.0 && d4 > 0.0))
    }
    
    fn direction(&self, p1: &Point, p2: &Point, p3: &Point) -> f64 {
        (p3.x - p1.x) * (p2.y - p1.y) - (p3.y - p1.y) * (p2.x - p1.x)
    }
    
    fn contains_polygon(&self, other: &Polygon) -> bool {
        for vertex in &other.vertices {
            if !self.contains_point(vertex) {
                return false;
            }
        }
        true
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Polygon {
        let translated_vertices = self.vertices.iter()
            .map(|v| Point::new(v.x + dx, v.y + dy))
            .collect();
        Polygon::new(translated_vertices)
    }
    
    fn rotate(&self, center: &Point, angle: f64) -> Polygon {
        let rotated_vertices = self.vertices.iter()
            .map(|v| v.rotate(center, angle))
            .collect();
        Polygon::new(rotated_vertices)
    }
}

                    placed = true;
                }

                self.stocks_used.push(new_stock);
                self.place_in_stock(piece_idx, self.stocks_used.len() - 1, piece, 0.0, 0.0, 0.0);
            }
            
            // Check time limit
            if start_time.elapsed() > max_time {
                break;
            }
        }
    }

    fn is_valid_placement(&self, stock_idx: usize, piece: &Polygon) -> bool {

        for vertex in &piece.vertices {

                return false;
            }
        }

        for placement in &self.placements {

                let placed_piece = &self.pieces[placement.piece_index];
                let translated = placed_piece.translate(placement.x, placement.y);
                let rotated = translated.rotate(&Point::new(0.0, 0.0), placement.rotation);

                }
            }
        }
        
        true
    }

        self.placements.push(Placement {
            piece_index: piece_idx,
            stock_index: stock_idx,
            x,
            y,
            rotation,
        });
    }
    
    fn output_results(&self) -> String {
        let mut output = String::new();
        output.push_str(&format!("{}\n", self.stocks_used.len()));

        let mut placement_map: HashMap<usize, &Placement> = HashMap::new();

                    "{} {} {} {}\n",
                    placement.stock_index,
                    placement.x,
                    placement.y,
                    placement.rotation
                ));
            } else {
                // Should not happen if all pieces are placed
                output.push_str(&format!("0 0 0 0\n"));
            }
        }
        
        output
    }
}

        let half_angle = angle / 2.0;
        let sin_half = half_angle.sin();
        let cos_half = half_angle.cos();

            w: cos_half,
            x: axis[0] * sin_half,
            y: axis[1] * sin_half,
            z: axis[2] * sin_half,
        }
    }

                x: self.x / norm,
                y: self.y / norm,
                z: self.z / norm,
            }

            w: self.w,
            x: -self.x,
            y: -self.y,
            z: -self.z,

class Karger {
private:

    vector<pair<int, int>> edges;
    mt19937 rng;

    int find(vector<int>& parent, int x) {
        if (parent[x] != x) {
            parent[x] = find(parent, parent[x]);
        }
        return parent[x];
    }

    void unite(vector<int>& parent, vector<int>& rank, int x, int y) {
        int xroot = find(parent, x);
        int yroot = find(parent, y);
        
        if (xroot == yroot) return;
        
        if (rank[xroot] < rank[yroot]) {
            parent[xroot] = yroot;
        } else if (rank[xroot] > rank[yroot]) {
            parent[yroot] = xroot;
        } else {
            parent[yroot] = xroot;
            rank[xroot]++;
        }
    }

public:
    Karger(size_t nodes, const vector<pair<int, int>>& edge_list) 
        : n(nodes), edges(edge_list), rng(chrono::steady_clock::now().time_since_epoch().count()) {}

class StoerWagner {
private:
    const vector<vector<int>>& adj;
    size_t n;

    pair<int, vector<int>> min_cut_phase(int a, vector<int>& in_set, vector<bool>& vis) {
        vector<int> added;
        vector<int> w(n, 0);
        int last = -1, prev = -1;
        
        for (int i = 0; i < n; ++i) {
            int max_v = -1;
            for (int v = 0; v < n; ++v) {
                if (!vis[v] && (max_v == -1 || w[v] > w[max_v])) {
                    max_v = v;
                }
            }
            
            if (i == n - 1) {
                if (added.size() < 2) return {numeric_limits<int>::max(), {}};
                return {w[max_v], added};
            }
            
            vis[max_v] = true;
            added.push_back(max_v);
            
            for (int v : adj[max_v]) {
                if (!vis[v]) {
                    w[v] += adj[max_v][v];
                }
            }
            
            prev = last;
            last = max_v;
        }
        
        return {numeric_limits<int>::max(), {}};
    }

public:
    StoerWagner(const vector<vector<int>>& adjacency) : adj(adjacency), n(adjacency.size()) {}

    pair<int, vector<pair<int, int>>> compute() {
        vector<bool> vis(n, false);
        vector<int> in_set;
        int best = numeric_limits<int>::max();
        vector<pair<int, int>> best_cut;
        
        for (int phase = 0; phase < n - 1; ++phase) {
            auto [cut_weight, cut_nodes] = min_cut_phase(phase == 0 ? 0 : -1, in_set, vis);
            
            if (cut_weight < best) {
                best = cut_weight;
                best_cut.clear();
                for (int i = 0; i < cut_nodes.size(); ++i) {
                    for (int j = i + 1; j < cut_nodes.size(); ++j) {
                        if (adj[cut_nodes[i]][cut_nodes[j]] > 0) {
                            best_cut.emplace_back(cut_nodes[i], cut_nodes[j]);
                        }
                    }
                }
            }
            
            if (phase == n - 2) break;
            
            int s = cut_nodes.back();
            for (int t : cut_nodes) {
                if (t != s) {
                    adj[s][t] += adj[t][s];
                    adj[t][s] = adj[s][t];
                }
            }
        }
        
        return {best, best_cut};
    }
};

// Karger's algorithm for large graphs
class Karger {
private:
    size_t n;
    vector<pair<int, int>> edges;
    mt19937 rng;

    int find(vector<int>& parent, int x) {
        if (parent[x] != x) {
            parent[x] = find(parent, parent[x]);
        }
        return parent[x];
    }

    void unite(vector<int>& parent, vector<int>& rank, int x, int y) {
        int xroot = find(parent, x);
        int yroot = find(parent, y);
        
        if (xroot == yroot) return;
        
        if (rank[xroot] < rank[yroot]) {
            parent[xroot] = yroot;
        } else if (rank[xroot] > rank[yroot]) {
            parent[yroot] = xroot;
        } else {
            parent[yroot] = xroot;
            rank[xroot]++;
        }
    }

public:
    Karger(size_t nodes, const vector<pair<int, int>>& edge_list) 
        : n(nodes), edges(edge_list), rng(chrono::steady_clock::now().time_since_epoch().count()) {}

    pair<int, vector<pair<int, int>>> compute(int iterations = 100) {
        int best_cut = numeric_limits<int>::max();
        vector<pair<int, int>> best_cut_edges;
        
        for (int iter = 0; iter < iterations; ++iter) {
            vector<int> parent(n);
            vector<int> rank(n, 0);
            iota(parent.begin(), parent.end(), 0);
            
            vector<pair<int, int>> current_edges = edges;
            int current_n = n;
            
            while (current_n > 2) {
                uniform_int_distribution<int> dist(0, current_edges.size() - 1);
                int idx = dist(rng);
                auto [u, v] = current_edges[idx];
                
                int uroot = find(parent, u);
                int vroot = find(parent, v);
                
                if (uroot != vroot) {
                    unite(parent, rank, uroot, vroot);
                    current_n--;
                }
                
                current_edges.erase(current_edges.begin() + idx);
            }
            
            int cut_size = 0;
            vector<pair<int, int>> cut_edges;
            
            for (const auto& edge : edges) {
                int uroot = find(parent, edge.first);
                int vroot = find(parent, edge.second);
                
                if (uroot != vroot) {
                    cut_size++;
                    cut_edges.emplace_back(edge.first, edge.second);
                }
            }
            
            if (cut_size < best_cut) {
                best_cut = cut_size;
                best_cut_edges = cut_edges;
            }

            return GetAccessibilityScore(a.x, a.y, a.z).CompareTo(GetAccessibilityScore(b.x, b.y, b.z));
        });
        
        // Process each ore voxel
        foreach (var ore in oreVoxels)
        {
            if (!isSolid[ore.x, ore.y, ore.z]) continue; // Already processed
            
            // Find best destination
            (int destX, int destY, int destZ) = FindBestDestination(ore.x, ore.y, ore.z, isLargeProblem);
            
            // Dig and move the ore
            MoveVoxel(ore.x, ore.y, ore.z, destX, destY, destZ);
        }
        
        // Process remaining dirt if profitable
        if (!isLargeProblem) // Only process dirt in small problems
        {
            ProcessRemainingDirt();
        }
    }
    
    private int GetAccessibilityScore(int x, int y, int z)
    {
        // Calculate accessibility score based on distance to processing plant and exposed faces
        int distance = Math.Abs(x - PROCESSING_X) + Math.Abs(y - PROCESSING_Y);
        int exposedFaces = CountExposedFaces(x, y, z);
        return exposedFaces * 1000 - distance;
    }
    
    private int CountExposedFaces(int x, int y, int z)
    {
        int exposed = 0;
        if (x == 0 || !isSolid[x - 1, y, z]) exposed++;
        if (x == xVoxelCount - 1 || !isSolid[x + 1, y, z]) exposed++;
        if (y == 0 || !isSolid[x, y - 1, z]) exposed++;
        if (y == yVoxelCount - 1 || !isSolid[x, y + 1, z]) exposed++;
        if (z == zVoxelCount - 1 || !isSolid[x, y, z + 1]) exposed++;
        if (z == 0 || !isSolid[x, y, z - 1]) exposed++;
        return exposed;
    }
    
    private (int destX, int destY, int destZ) FindBestDestination(int sourceX, int sourceY, int sourceZ, bool isLargeProblem)
    {
        // For large problems, prioritize processing plant
        if (isLargeProblem)
        {
            return (PROCESSING_X, PROCESSING_Y, 0); // Ground level at processing plant
        }
        
        // For small problems, consider profitability
        int value = voxelData[sourceX, sourceY, sourceZ] // 16;
        int mass = voxelData[sourceX, sourceY, sourceZ] % 16;
        long potentialRevenue = (long)value * mass * ORE_VALUE_MULTIPLIER;
        
        // If highly valuable, send to processing plant
        if (value >= 10)
        {
            return (PROCESSING_X, PROCESSING_Y, 0);
        }
        
        // Otherwise, find a nearby dump location that doesn't block future access
        return FindOptimalDumpLocation(sourceX, sourceY, sourceZ);
    }
    
    private (int destX, int destY, int destZ) FindOptimalDumpLocation(int sourceX, int sourceY, int sourceZ)
    {
        // Find the nearest location that:
        // 1. Is at ground level or can be reached with valid slope
        // 2. Doesn't block access to more valuable ore
        // 3. Is not too far from source
        
        int bestX = sourceX, bestY = sourceY, bestZ = 0;
        long bestScore = long.MinValue;
        
        // Search in expanding radius
        for (int radius = 0; radius <= 10; radius++)
        {
            bool foundBetter = false;
            for (int dx = -radius; dx <= radius; dx++)
            {
                for (int dy = -radius; dy <= radius; dy++)
                {
                    if (Math.Abs(dx) + Math.Abs(dy) != radius) continue;
                    
                    int x = sourceX + dx;
                    int y = sourceY + dy;
                    
                    if (x < 0 || x >= xVoxelCount || y < 0 || y >= yVoxelCount) continue;
                    
                    // Find ground level at this location
                    int z = FindGroundLevel(x, y);
                    if (z < 0) continue; // No ground found
                    
                    // Check if path is valid
                    if (!IsPathValid(sourceX, sourceY, sourceZ, x, y, z))
                    {
                        continue;
                    }
                    
                    // Calculate score
                    long score = CalculateDumpScore(sourceX, sourceY, sourceZ, x, y, z);
                    if (score > bestScore)
                    {
                        bestScore = score;
                        bestX = x;
                        bestY = y;
                        bestZ = z;
                        foundBetter = true;
                    }
                }
            }
            if (foundBetter) break; // Found best within this radius
        }
        
        return (bestX, bestY, bestZ);
    }
    
    private int FindGroundLevel(int x, int y)
    {
        for (int z = 0; z < zVoxelCount; z++)
        {
            if (isSolid[x, y, z])
            {
                return z;
            }
        }
        return -1; // No solid voxel found
    }
    
    private bool IsPathValid(int startX, int startY, int startZ, int endX, int endY, int endZ)
    {
        // Simple check: ensure we don't go uphill more than allowed
        int dx = Math.Abs(endX - startX);
        int dy = Math.Abs(endY - startY);
        int dz = endZ - startZ;
        
        if (dz > 0 && dz > (dx + dy) * MAX_SLOPE)
        {
            return false;
        }
        
        return true;
    }
    
    private long CalculateDumpScore(int sourceX, int sourceY, int sourceZ, int destX, int destY, int destZ)
    {
        int value = voxelData[sourceX, sourceY, sourceZ] // 16;
        int mass = voxelData[sourceX, sourceY, sourceZ] % 16;
        long revenue = (long)value * mass * ORE_VALUE_MULTIPLIER;
        
        // Calculate fuel cost
        int dx = Math.Abs(destX - sourceX);
        int dy = Math.Abs(destY - sourceY);
        int dz = destZ - sourceZ;
        
        long fuelCost = 0;
        if (dz > 0)
        {
            fuelCost = (long)(dx + dy) * FUEL_COST_UPHILL;
        }
        else
        {
            fuelCost = (long)(dx + dy) * FUEL_COST_FLAT;
        }
        
        // Check if destination blocks access to valuable ore
        long blockingCost = EstimateBlockingCost(destX, destY, destZ);
        
        return revenue - fuelCost - blockingCost;
    }
    
    private long EstimateBlockingCost(int x, int y, int z)
    {
        // Simple heuristic: higher ground blocks more access
        return z * 10;
    }
    
    private void MoveVoxel(int sourceX, int sourceY, int sourceZ, int destX, int destY, int destZ)
    {
        // Check if we're moving to processing plant
        bool toProcessingPlant = (destX == PROCESSING_X && destY == PROCESSING_Y);
        
        // Find path
        var path = FindPath(sourceX, sourceY, sourceZ, destX, destY, destZ);
        if (path == null || path.Count == 0)
        {
            return; // No valid path found
        }
        
        // Execute move
        operationsCount++;
        var steps = new List<(int x, int y)>();
        
        for (int i = 0; i < path.Count - 1; i++)
        {
            steps.Add((path[i + 1].x - path[i].x, path[i + 1].y - path[i].y));
        }
        
        // Calculate fuel cost
        int dx = Math.Abs(destX - sourceX);
        int dy = Math.Abs(destY - sourceY);
        int dz = destZ - sourceZ;
        long fuelCost = 0;
        
        if (dz > 0)
        {
            fuelCost = (long)(dx + dy) * FUEL_COST_UPHILL;
        }
        else
        {
            fuelCost = (long)(dx + dy) * FUEL_COST_FLAT;
        }
        
        totalFuelCost += fuelCost;
        
        // Calculate revenue
        int value = voxelData[sourceX, sourceY, sourceZ] // 16;
        int mass = voxelData[sourceX, sourceY, sourceZ] % 16;
        long revenue = (long)value * mass * ORE_VALUE_MULTIPLIER;

    }
    
    private List<(int x, int y, int z)> FindPath(int startX, int startY, int startZ, int endX, int endY, int endZ)
    {
        // Simple A* pathfinding with slope constraints
        var openSet = new PriorityQueue<(int x, int y, int z), int>();
        var cameFrom = new Dictionary<(int x, int y, int z), (int x, int y, int z)>();
        var gScore = new Dictionary<(int x, int y, int z), int>();
        var fScore = new Dictionary<(int x, int y, int z), int>();
        
        var start = (startX, startY, startZ);
        openSet.Enqueue(start, 0);
        gScore[start] = 0;
        fScore[start] = Heuristic(start, (endX, endY, endZ));
        
        while (openSet.Count > 0)
        {
            var current = openSet.Dequeue();
            
            if (current.x == endX && current.y == endY && current.z == endZ)
            {
                return ReconstructPath(cameFrom, current);
            }
            
            foreach (var neighbor in GetNeighbors(current))
            {
                if (!IsAccessible(neighbor.x, neighbor.y, neighbor.z)) continue;
                
                int tentativeGScore = gScore[current] + 1;
                
                if (!gScore.ContainsKey(neighbor) || tentativeGScore < gScore[neighbor])
                {
                    cameFrom[neighbor] = current;
                    gScore[neighbor] = tentativeGScore;
                    fScore[neighbor] = tentativeGScore + Heuristic(neighbor, (endX, endY, endZ));
                    openSet.Enqueue(neighbor, fScore[neighbor]);
                }
            }
        }
        
        return null; // No path found
    }
    
    private List<(int x, int y, int z)> ReconstructPath(Dictionary<(int x, int y, int z), (int x, int y, int z)> cameFrom, (int x, int y, int z) current)
    {
        var path = new List<(int x, int y, int z)> { current };
        while (cameFrom.ContainsKey(current))
        {
            current = cameFrom[current];
            path.Add(current);
        }
        path.Reverse();
        return path;
    }
    
    private int Heuristic((int x, int y, int z) a, (int x, int y, int z) b)
    {
        return Math.Abs(a.x - b.x) + Math.Abs(a.y - b.y) + Math.Abs(a.z - b.z) * 2;
    }
    
    private IEnumerable<(int x, int y, int z)> GetNeighbors((int x, int y, int z) pos)
    {
        var neighbors = new List<(int x, int y, int z)>();
        
        // 4-connected neighbors in X/Y plane
        var directions = new[] { (1, 0, 0), (-1, 0, 0), (0, 1, 0), (0, -1, 0) };
        
        foreach (var (dx, dy, dz) in directions)
        {
            int nx = pos.x + dx;
            int ny = pos.y + dy;
            int nz = pos.z + dz;
            
            if (nx >= 0 && nx < xVoxelCount && ny >= 0 && ny < yVoxelCount && nz >= 0 && nz < zVoxelCount)
            {
                neighbors.Add((nx, ny, nz));
            }
        }
        
        return neighbors;
    }
    
    private bool IsAccessible(int x, int y, int z)
    {
        // Check if voxel is accessible (not solid or already dug)
        return !isSolid[x, y, z] || isDug[x, y, z];
    }
    
    private void ProcessRemainingDirt()
    {
        // In small problems, process remaining dirt for additional profit
        var dirtVoxels = new List<(int x, int y, int z, int mass)>();
        
        for (int x = 0; x < xVoxelCount; x++)
        {
            for (int y = 0; y < yVoxelCount; y++)
            {
                for (int z = 0; z < zVoxelCount; z++)
                {
                    if (isSolid[x, y, z] && !isOre[x, y, z] && !isDug[x, y, z])
                    {
                        dirtVoxels.Add((x, y, z, voxelData[x, y, z] % 16));
                    }
                }
            }
        }
        
        // Sort by mass (descending) and accessibility
        dirtVoxels.Sort((a, b) => 
        {
            int massCompare = b.mass.CompareTo(a.mass);
            if (massCompare != 0) return massCompare;
            return GetAccessibilityScore(a.x, a.y, a.z).CompareTo(GetAccessibilityScore(b.x, b.y, b.z));
        });
        
        foreach (var dirt in dirtVoxels)
        {
            if (!isSolid[dirt.x, dirt.y, dirt.z]) continue;
            
            // Find nearest dump location
            var dest = FindOptimalDumpLocation(dirt.x, dirt.y, dirt.z);
            MoveVoxel(dirt.x, dirt.y, dirt.z, dest.destX, dest.destY, dest.destZ);
        }
    }
