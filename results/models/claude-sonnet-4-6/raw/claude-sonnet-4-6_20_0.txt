use std::collections::HashMap;
use std::io::{self, BufRead, Write, BufWriter};

fn median_and_mad(sorted_vals: &[f64]) -> (f64, f64) {
    let n = sorted_vals.len();
    if n == 0 {
        return (0.0, 0.0);
    }
    let med = if n % 2 == 0 {
        (sorted_vals[n / 2 - 1] + sorted_vals[n / 2]) / 2.0
    } else {
        sorted_vals[n / 2]
    };
    let mut devs: Vec<f64> = sorted_vals.iter().map(|&v| (v - med).abs()).collect();
    devs.sort_unstable_by(|a, b| a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal));
    (med, devs[devs.len() / 2])
}

fn mz_score(value: f64, med: f64, mad: f64) -> f64 {
    if mad > 1e-10 {
        (value - med).abs() * 0.6745 / mad
    } else if (value - med).abs() > 1e-6 {
        8.0
    } else {
        0.0
    }
}

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

    let first = match lines.next().and_then(|r| r.ok()) {
        Some(s) => s,
        None => {
            writeln!(out, "0").unwrap();
            return;
        }
    };

    let mut p = first.split_ascii_whitespace();
    let num_holes: usize = p.next().unwrap_or("0").parse().unwrap_or(0);
    let num_props: usize = p.next().unwrap_or("0").parse().unwrap_or(0);

    let prop_line = match lines.next().and_then(|r| r.ok()) {
        Some(s) => s,
        None => {
            writeln!(out, "0").unwrap();
            return;
        }
    };
    let prop_names: Vec<String> = prop_line
        .split_ascii_whitespace()
        .map(|s| s.to_string())
        .collect();
    let num_props = num_props.min(prop_names.len());

    let mut hole_ids: Vec<String> = Vec::with_capacity(num_holes);
    let mut positions: Vec<[f64; 3]> = Vec::new();
    let mut hole_of: Vec<u32> = Vec::new();
    let mut idx_in_hole: Vec<u32> = Vec::new();
    let mut values: Vec<Vec<f64>> = Vec::new();

    for _ in 0..num_holes {
        let hl = match lines.next().and_then(|r| r.ok()) {
            Some(s) => s,
            None => break,
        };
        let mut hp = hl.split_ascii_whitespace();
        let hole_id = hp.next().unwrap_or("").to_string();
        let sx: f64 = hp.next().unwrap_or("0").parse().unwrap_or(0.0);
        let sy: f64 = hp.next().unwrap_or("0").parse().unwrap_or(0.0);
        let sz: f64 = hp.next().unwrap_or("0").parse().unwrap_or(0.0);
        let dx: f64 = hp.next().unwrap_or("0").parse().unwrap_or(0.0);
        let dy: f64 = hp.next().unwrap_or("0").parse().unwrap_or(0.0);
        let dz: f64 = hp.next().unwrap_or("1").parse().unwrap_or(1.0);
        let _len: f64 = hp.next().unwrap_or("0").parse().unwrap_or(0.0);
        let ns: usize = hp.next().unwrap_or("0").parse().unwrap_or(0);

        let mag = (dx * dx + dy * dy + dz * dz).sqrt();
        let (nx, ny, nz) = if mag > 1e-10 {
            (dx / mag, dy / mag, dz / mag)
        } else {
            (0.0, 0.0, -1.0)
        };

        let hidx = hole_ids.len() as u32;
        hole_ids.push(hole_id);

        for si in 0..ns {
            let sl = match lines.next().and_then(|r| r.ok()) {
                Some(s) => s,
                None => break,
            };
            let mut sp = sl.split_ascii_whitespace();
            let depth: f64 = sp.next().unwrap_or("0").parse().unwrap_or(0.0);

            positions.push([
                sx + nx * depth,
                sy + ny * depth,
                sz + nz * depth,
            ]);
            hole_of.push(hidx);
            idx_in_hole.push(si as u32);

            let vals: Vec<f64> = (0..num_props)
                .map(|_| {
                    sp.next()
                        .unwrap_or("")
                        .parse::<f64>()
                        .unwrap_or(f64::NAN)
                })
                .collect();
            values.push(vals);
        }
    }

    let n = positions.len();
    if n == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    // Adaptive parameters based on dataset size
    let k: usize = if n < 50 {
        (n - 1).max(2).min(8)
    } else if n < 10_000 {
        20
    } else if n < 100_000 {
        15
    } else {
        10
    };

    let z_thresh: f64 = if n < 200 { 3.0 } else { 3.5 };

    // Compute bounding box
    let mut mn = [f64::INFINITY; 3];
    let mut mx = [f64::NEG_INFINITY; 3];
    for pos in &positions {
        for i in 0..3 {
            if pos[i].is_finite() {
                if pos[i] < mn[i] {
                    mn[i] = pos[i];
                }
                if pos[i] > mx[i] {
                    mx[i] = pos[i];
                }
            }
        }
    }
    for i in 0..3 {
        if !mn[i].is_finite() {
            mn[i] = 0.0;
        }
        if !mx[i].is_finite() {
            mx[i] = 1.0;
        }
        if mx[i] <= mn[i] {
            mx[i] = mn[i] + 1.0;
        }
    }

    let mut suspects: Vec<(u32, u32, usize, f64)> = Vec::new();

    if n <= 2000 {
        // Brute force: O(n^2) exact k-nearest for small datasets
        let mut nbuf: Vec<(f64, usize)> = Vec::with_capacity(n);
        let mut nvals: Vec<f64> = Vec::with_capacity(k + 5);

        for i in 0..n {
            nbuf.clear();
            for j in 0..n {
                if j == i {
                    continue;
                }
                let d2: f64 = (0..3)
                    .map(|d| (positions[i][d] - positions[j][d]).powi(2))
                    .sum();
                nbuf.push((d2, j));
            }
            nbuf.sort_unstable_by(|a, b| {
                a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal)
            });
            let uk = k.min(nbuf.len());
            if uk < 2 {
                continue;
            }

            for pi in 0..num_props {
                let v = values[i][pi];
                if !v.is_finite() {
                    continue;
                }

                nvals.clear();
                for &(_, j) in &nbuf[..uk] {
                    let vj = values[j][pi];
                    if vj.is_finite() {
                        nvals.push(vj);
                    }
                }
                if nvals.len() < 3 {
                    continue;
                }

                nvals.sort_unstable_by(|a, b| {
                    a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal)
                });
                let (med, mad) = median_and_mad(&nvals);
                let score = mz_score(v, med, mad);

                if score > z_thresh {
                    suspects.push((hole_of[i], idx_in_hole[i], pi, score));
                }
            }
        }
    } else {
        // Grid-based spatial index for larger datasets
        let vol = (mx[0] - mn[0]).max(1.0)
            * (mx[1] - mn[1]).max(1.0)
            * (mx[2] - mn[2]).max(1.0);
        let density = n as f64 / vol;
        // Cell size so that 27 cells hold ~2k neighbors on average
        let cell_vol = (2.0 * k as f64) / (27.0 * density);
        let cell_size = cell_vol.powf(1.0 / 3.0).max(1e-9);

        let mut grid: HashMap<(i32, i32, i32), Vec<u32>> =
            HashMap::with_capacity(n / 4 + 1);
        for (i, pos) in positions.iter().enumerate() {
            let key = (
                ((pos[0] - mn[0]) / cell_size).floor() as i32,
                ((pos[1] - mn[1]) / cell_size).floor() as i32,
                ((pos[2] - mn[2]) / cell_size).floor() as i32,
            );
            grid.entry(key).or_default().push(i as u32);
        }

        let mut nbuf: Vec<(f64, u32)> = Vec::with_capacity(k * 20);
        let mut nvals: Vec<f64> = Vec::with_capacity(k + 5);

        for i in 0..n {
            let pos = &positions[i];
            let cx = ((pos[0] - mn[0]) / cell_size).floor() as i32;
            let cy = ((pos[1] - mn[1]) / cell_size).floor() as i32;
            let cz = ((pos[2] - mn[2]) / cell_size).floor() as i32;

            nbuf.clear();

            // Expand search shell by shell until enough neighbors found
            'outer: for r in 1i32..=5 {
                for dz in -r..=r {
                    for dy in -r..=r {
                        for dx in -r..=r {
                            // Only process cells in the new outer shell at this radius
                            if r > 1 && dx.abs() < r && dy.abs() < r && dz.abs() < r {
                                continue;
                            }
                            if let Some(cell) = grid.get(&(cx + dx, cy + dy, cz + dz)) {
                                for &sidx in cell {
                                    if sidx as usize == i {
                                        continue;
                                    }
                                    let sp = &positions[sidx as usize];
                                    let d2 = (0..3)
                                        .map(|d| (pos[d] - sp[d]).powi(2))
                                        .sum::<f64>();
                                    nbuf.push((d2, sidx));
                                }
                            }
                        }
                    }
                }
                if nbuf.len() >= k {
                    break 'outer;
                }
            }

            if nbuf.len() < 3 {
                continue;
            }

            nbuf.sort_unstable_by(|a, b| {
                a.0.partial_cmp(&b.0).unwrap_or(std::cmp::Ordering::Equal)
            });
            let uk = k.min(nbuf.len());

            for pi in 0..num_props {
                let v = values[i][pi];
                if !v.is_finite() {
                    continue;
                }

                nvals.clear();
                for &(_, j) in &nbuf[..uk] {
                    let vj = values[j as usize][pi];
                    if vj.is_finite() {
                        nvals.push(vj);
                    }
                }
                if nvals.len() < 3 {
                    continue;
                }

                nvals.sort_unstable_by(|a, b| {
                    a.partial_cmp(b).unwrap_or(std::cmp::Ordering::Equal)
                });
                let (med, mad) = median_and_mad(&nvals);
                let score = mz_score(v, med, mad);

                if score > z_thresh {
                    suspects.push((hole_of[i], idx_in_hole[i], pi, score));
                }
            }
        }
    }

    suspects.sort_unstable_by(|a, b| {
        b.3.partial_cmp(&a.3).unwrap_or(std::cmp::Ordering::Equal)
    });

    writeln!(out, "{}", suspects.len()).unwrap();
    for &(hidx, sidx, pidx, score) in &suspects {
        writeln!(
            out,
            "{} {} {} {:.4}",
            hole_ids[hidx as usize], sidx, prop_names[pidx], score
        )
        .unwrap();
    }
}
