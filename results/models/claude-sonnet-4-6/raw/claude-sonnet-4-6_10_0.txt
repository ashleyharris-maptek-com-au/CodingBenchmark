#include <iostream>
#include <vector>
#include <algorithm>
#include <climits>
#include <numeric>

using namespace std;

struct FR {
    int x, y, w, h;
};

class Bin {
public:
    static const int MAX_FR = 256;
    long long freeArea;
    vector<FR> frs;

    Bin(int W, int H) : freeArea((long long)W * H) {
        frs.push_back({0, 0, W, H});
    }

    // Best Short Side Fit score; sets rx,ry on success. Returns INT_MAX if no fit.
    int bssf(int w, int h, int& rx, int& ry) const {
        int b1 = INT_MAX, b2 = INT_MAX;
        for (const FR& f : frs) {
            if (f.w >= w && f.h >= h) {
                int s1 = min(f.w - w, f.h - h);
                int s2 = max(f.w - w, f.h - h);
                if (s1 < b1 || (s1 == b1 && s2 < b2)) {
                    b1 = s1; b2 = s2;
                    rx = f.x; ry = f.y;
                }
            }
        }
        return b1;
    }

    void commit(int px, int py, int pw, int ph) {
        vector<FR> nf;
        nf.reserve(frs.size() * 4);

        for (const FR& f : frs) {
            // No intersection: keep unchanged
            if (px >= f.x + f.w || px + pw <= f.x ||
                py >= f.y + f.h || py + ph <= f.y) {
                nf.push_back(f);
                continue;
            }
            // Split into up to 4 new free rects
            if (px > f.x)            nf.push_back({f.x, f.y, px - f.x, f.h});
            if (px + pw < f.x + f.w) nf.push_back({px + pw, f.y, f.x + f.w - px - pw, f.h});
            if (py > f.y)            nf.push_back({f.x, f.y, f.w, py - f.y});
            if (py + ph < f.y + f.h) nf.push_back({f.x, py + ph, f.w, f.y + f.h - py - ph});
        }

        // Prune: remove rects fully contained within another
        int n = (int)nf.size();
        vector<bool> del(n, false);
        for (int i = 0; i < n; i++) {
            if (del[i]) continue;
            for (int j = 0; j < n && !del[i]; j++) {
                if (j == i || del[j]) continue;
                if (nf[j].x <= nf[i].x && nf[j].y <= nf[i].y &&
                    nf[j].x + nf[j].w >= nf[i].x + nf[i].w &&
                    nf[j].y + nf[j].h >= nf[i].y + nf[i].h)
                    del[i] = true;
            }
        }

        frs.clear();
        for (int i = 0; i < n; i++)
            if (!del[i]) frs.push_back(nf[i]);

        // Cap free rect list to bound prune cost on subsequent placements
        if ((int)frs.size() > MAX_FR) {
            nth_element(frs.begin(), frs.begin() + MAX_FR, frs.end(),
                [](const FR& a, const FR& b) {
                    return (long long)a.w * a.h > (long long)b.w * b.h;
                });
            frs.resize(MAX_FR);
        }

        freeArea -= (long long)pw * ph;
    }
};

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    int N, BW, BH;
    cin >> N >> BW >> BH;

    vector<int> rw(N), rh(N);
    for (int i = 0; i < N; i++) cin >> rw[i] >> rh[i];

    // First Fit Decreasing by area for better bin utilization
    vector<int> ord(N);
    iota(ord.begin(), ord.end(), 0);
    sort(ord.begin(), ord.end(), [&](int a, int b) {
        return (long long)rw[a] * rh[a] > (long long)rw[b] * rh[b];
    });

    struct Placement { int bin, x, y, rot; };
    vector<Placement> result(N);
    vector<Bin> bins;
    bins.reserve(8192);

    for (int idx : ord) {
        int w = rw[idx], h = rh[idx];

        // Best Fit: find bin+orientation with minimum BSSF score
        int bb = -1, bx = 0, by = 0, brot = 0, bs = INT_MAX;

        for (int b = 0; b < (int)bins.size() && bs > 0; b++) {
            // Quick area check to skip obviously full bins
            if (bins[b].freeArea < (long long)w * h) continue;

            int rx, ry, s;
            s = bins[b].bssf(w, h, rx, ry);
            if (s < bs) { bs = s; bb = b; bx = rx; by = ry; brot = 0; }

            if (bs > 0 && w != h) {
                s = bins[b].bssf(h, w, rx, ry);
                if (s < bs) { bs = s; bb = b; bx = rx; by = ry; brot = 1; }
            }
        }

        if (bb == -1) {
            // No existing bin fits; open a new one
            bins.emplace_back(BW, BH);
            bb = (int)bins.size() - 1;
            int rx, ry;
            int s1 = bins[bb].bssf(w, h, rx, ry);
            if (s1 != INT_MAX) { bx = rx; by = ry; brot = 0; bs = s1; }
            if (w != h) {
                int rx2, ry2;
                int s2 = bins[bb].bssf(h, w, rx2, ry2);
                if (s2 < bs) { bx = rx2; by = ry2; brot = 1; }
            }
        }

        int pw = brot ? h : w;
        int ph = brot ? w : h;
        bins[bb].commit(bx, by, pw, ph);
        result[idx] = {bb, bx, by, brot};
    }

    cout << bins.size() << "\n";
    for (int i = 0; i < N; i++)
        cout << result[i].bin << " " << result[i].x << " "
             << result[i].y << " " << result[i].rot << "\n";

    return 0;
}
