use std::io::{self, Read, Write, BufWriter};
use std::time::Instant;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut words = input.split_ascii_whitespace();

    macro_rules! nxt {
        ($t:ty) => {
            words.next().and_then(|s| s.parse::<$t>().ok()).unwrap_or_default()
        };
    }

    let n: usize = nxt!(usize);
    let m: usize = nxt!(usize);

    if n == 0 {
        return;
    }

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m.min(8_000_000));
    for _ in 0..m {
        let u = nxt!(u32) as usize;
        let v = nxt!(u32) as usize;
        if u < n && v < n && u != v {
            edges.push((u as u32, v as u32));
        }
    }

    let timer = Instant::now();
    let time_limit = 28.5_f64;

    let mut px = vec![0.0_f64; n];
    let mut py = vec![0.0_f64; n];
    let k = 100.0_f64;
    let k2 = k * k;

    // Circular init for small n (avoids collinearity, good for planar graphs)
    // Grid init for large n (compact, efficient for grid-based FR)
    if n <= 5000 {
        let r = n as f64 * k / std::f64::consts::TAU;
        for i in 0..n {
            let a = std::f64::consts::TAU * i as f64 / n as f64;
            px[i] = r * a.cos();
            py[i] = r * a.sin();
        }
    } else {
        let cols = (n as f64).sqrt().ceil() as usize + 1;
        for i in 0..n {
            let h = i.wrapping_mul(2654435769_usize);
            let jx = (h & 0xFF) as f64 / 256.0 - 0.5;
            let jy = ((h >> 8) & 0xFF) as f64 / 256.0 - 0.5;
            px[i] = (i % cols) as f64 * k + jx;
            py[i] = (i / cols) as f64 * k + jy;
        }
    }

    let iters: usize = if n <= 30 { 700 }
        else if n <= 100 { 500 }
        else if n <= 300 { 300 }
        else if n <= 800 { 180 }
        else if n <= 2000 { 110 }
        else if n <= 5000 { 70 }
        else if n <= 20_000 { 45 }
        else if n <= 100_000 { 30 }
        else { 20 };

    let area_side = (n as f64).sqrt() * k;
    let mut temp = area_side * 0.1;
    let cool = (temp - 0.01) / iters as f64;

    let mut fdx = vec![0.0_f64; n];
    let mut fdy = vec![0.0_f64; n];

    if n <= 2000 {
        // Full O(n^2) Fruchterman-Reingold
        for _ in 0..iters {
            if timer.elapsed().as_secs_f64() > time_limit { break; }
            fdx.fill(0.0);
            fdy.fill(0.0);

            for i in 0..n {
                for j in i + 1..n {
                    let dx = px[i] - px[j];
                    let dy = py[i] - py[j];
                    let d2 = dx * dx + dy * dy;
                    if d2 < 1e-8 {
                        fdx[i] += 0.1;
                        fdx[j] -= 0.1;
                        continue;
                    }
                    let f = k2 / d2;
                    fdx[i] += dx * f;
                    fdy[i] += dy * f;
                    fdx[j] -= dx * f;
                    fdy[j] -= dy * f;
                }
            }

            for &(u, v) in &edges {
                let (u, v) = (u as usize, v as usize);
                let dx = px[u] - px[v];
                let dy = py[u] - py[v];
                let d2 = dx * dx + dy * dy;
                if d2 < 1e-8 { continue; }
                let d = d2.sqrt();
                let f = d / k;
                fdx[u] -= dx * f;
                fdy[u] -= dy * f;
                fdx[v] += dx * f;
                fdy[v] += dy * f;
            }

            for i in 0..n {
                let mag2 = fdx[i] * fdx[i] + fdy[i] * fdy[i];
                if mag2 > 1e-16 {
                    let s = (temp / mag2.sqrt()).min(1.0);
                    px[i] += fdx[i] * s;
                    py[i] += fdy[i] * s;
                }
            }
            temp = (temp - cool).max(0.01);
        }
    } else {
        // Grid-accelerated Fruchterman-Reingold
        let cell_size = k * 2.0;
        let max_g: usize = 1000;
        let max_c = max_g * max_g;
        let mut gcnt = vec![0_u32; max_c];
        let mut gstart = vec![0_u32; max_c + 1];
        let mut ncell = vec![0_u32; n];
        let mut bucket = vec![0_u32; n];

        for _ in 0..iters {
            if timer.elapsed().as_secs_f64() > time_limit { break; }
            fdx.fill(0.0);
            fdy.fill(0.0);

            let mut mnx = f64::INFINITY;
            let mut mxx = f64::NEG_INFINITY;
            let mut mny = f64::INFINITY;
            let mut mxy = f64::NEG_INFINITY;
            for i in 0..n {
                if px[i] < mnx { mnx = px[i]; }
                if px[i] > mxx { mxx = px[i]; }
                if py[i] < mny { mny = py[i]; }
                if py[i] > mxy { mxy = py[i]; }
            }
            let sx = (mxx - mnx).max(cell_size);
            let sy = (mxy - mny).max(cell_size);
            let gw = ((sx / cell_size).ceil() as usize).max(1).min(max_g);
            let gh = ((sy / cell_size).ceil() as usize).max(1).min(max_g);
            let tc = gw * gh;
            let cw = sx / gw as f64;
            let ch = sy / gh as f64;

            gcnt[..tc].fill(0);
            for i in 0..n {
                let cx = ((px[i] - mnx) / cw) as usize;
                let cy = ((py[i] - mny) / ch) as usize;
                let ci = cy.min(gh - 1) * gw + cx.min(gw - 1);
                ncell[i] = ci as u32;
                gcnt[ci] += 1;
            }

            gstart[0] = 0;
            for i in 0..tc {
                gstart[i + 1] = gstart[i] + gcnt[i];
            }

            // Copy start positions into gcnt to use as write pointer
            gcnt[..tc].copy_from_slice(&gstart[..tc]);
            for i in 0..n {
                let ci = ncell[i] as usize;
                bucket[gcnt[ci] as usize] = i as u32;
                gcnt[ci] += 1;
            }

            // Check 5x5 cell neighborhood (~2k radius)
            let cr: isize = 2;
            for i in 0..n {
                let ci = ncell[i] as usize;
                let cx = (ci % gw) as isize;
                let cy = (ci / gw) as isize;
                let gx0 = (cx - cr).max(0) as usize;
                let gx1 = (cx + cr + 1).min(gw as isize) as usize;
                let gy0 = (cy - cr).max(0) as usize;
                let gy1 = (cy + cr + 1).min(gh as isize) as usize;

                for gx in gx0..gx1 {
                    for gy in gy0..gy1 {
                        let ci2 = gy * gw + gx;
                        let s = gstart[ci2] as usize;
                        let e = gstart[ci2 + 1] as usize;
                        for bi in s..e {
                            let j = bucket[bi] as usize;
                            if j <= i { continue; }
                            let dx = px[i] - px[j];
                            let dy = py[i] - py[j];
                            let d2 = dx * dx + dy * dy;
                            if d2 < 1e-8 {
                                fdx[i] += 0.1;
                                fdx[j] -= 0.1;
                                continue;
                            }
                            let f = k2 / d2;
                            fdx[i] += dx * f;
                            fdy[i] += dy * f;
                            fdx[j] -= dx * f;
                            fdy[j] -= dy * f;
                        }
                    }
                }
            }

            for &(u, v) in &edges {
                let (u, v) = (u as usize, v as usize);
                let dx = px[u] - px[v];
                let dy = py[u] - py[v];
                let d2 = dx * dx + dy * dy;
                if d2 < 1e-8 { continue; }
                let d = d2.sqrt();
                let f = d / k;
                fdx[u] -= dx * f;
                fdy[u] -= dy * f;
                fdx[v] += dx * f;
                fdy[v] += dy * f;
            }

            for i in 0..n {
                let mag2 = fdx[i] * fdx[i] + fdy[i] * fdy[i];
                if mag2 > 1e-16 {
                    let s = (temp / mag2.sqrt()).min(1.0);
                    px[i] += fdx[i] * s;
                    py[i] += fdy[i] * s;
                }
            }
            temp = (temp - cool).max(0.01);
        }
    }

    let stdout = io::stdout();
    let mut writer = BufWriter::new(stdout.lock());
    for i in 0..n {
        writeln!(writer, "{:.6} {:.6}", px[i], py[i]).unwrap();
    }
}
