1. **Small N (≤12)**: Exact bitmask DP
2. **Medium N (≤200k)**: Greedy nearest neighbor + 2-opt improvements
3. **Large N**: Greedy nearest neighbor with spatial indexing (k-d tree / grid)
4. **Extreme N**: Fast greedy with grid-based nearest neighbor

Key design decisions:
- Use a grid-based spatial index for fast nearest neighbor queries at large scales
- Apply 2-opt improvements when time budget allows
- Parallel processing where beneficial

// ==================== Exact TSP for tiny N ====================

vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y) {
    // Bitmask DP, n <= 20
    int states = 1 << n;
    vector<vector<float>> dp(states, vector<float>(n, 1e30f));
    vector<vector<int>> parent(states, vector<int>(n, -1));
    
    dp[1][0] = 0.0f;
    
    for (int mask = 1; mask < states; mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            if (dp[mask][u] >= 1e29f) continue;
            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;
                int nmask = mask | (1 << v);
                float dx = x[u] - x[v], dy = y[u] - y[v];
                float d = sqrtf(dx*dx + dy*dy);
                float nd = dp[mask][u] + d;
                if (nd < dp[nmask][v]) {
                    dp[nmask][v] = nd;
                    parent[nmask][v] = u;
                }
            }
        }
    }
    
    int fullMask = states - 1;
    float best = 1e30f;
    int last = -1;
    for (int v = 1; v < n; v++) {
        float dx = x[v] - x[0], dy = y[v] - y[0];
        float total = dp[fullMask][v] + sqrtf(dx*dx + dy*dy);
        if (total < best) { best = total; last = v; }
    }
    
    vector<int> tour;
    int mask = fullMask, cur = last;
    while (cur != -1) {
        tour.push_back(cur);
        int prev = parent[mask][cur];
        mask ^= (1 << cur);
        cur = prev;
    }
    reverse(tour.begin(), tour.end());
    return tour;
}

// ==================== Grid-based spatial index ====================

struct GridIndex {
    float minX, minY, maxX, maxY;
    float cellW, cellH;
    int gridW, gridH;
    vector<vector<int>> cells;
    const vector<float>* px;
    const vector<float>* py;
    int n;
    
    void build(const vector<float>& x, const vector<float>& y, int N, int targetCellSize = 4) {
        px = &x; py = &y; n = N;
        minX = *min_element(x.begin(), x.end());
        maxX = *max_element(x.begin(), x.end());
        minY = *min_element(y.begin(), y.end());
        maxY = *max_element(y.begin(), y.end());
        
        float rangeX = maxX - minX + 1e-6f;
        float rangeY = maxY - minY + 1e-6f;
        
        // Choose grid size
        int totalCells = max(1, (int)(N / targetCellSize));
        float ratio = rangeX / rangeY;
        gridH = max(1, (int)sqrt((double)totalCells / ratio));
        gridW = max(1, totalCells / gridH);
        // Cap grid size to avoid memory issues
        long long maxGrid = 50000000LL;
        while ((long long)gridW * gridH > maxGrid) {
            gridW = max(1, gridW/2);
            gridH = max(1, gridH/2);
        }
        
        cellW = rangeX / gridW;
        cellH = rangeY / gridH;
        
        cells.assign((long long)gridW * gridH, vector<int>());
        for (int i = 0; i < N; i++) {
            int cx = min((int)((x[i]-minX)/cellW), gridW-1);
            int cy = min((int)((y[i]-minY)/cellH), gridH-1);
            cells[cy*gridW+cx].push_back(i);
        }
    }
    
    // Find nearest unvisited neighbor
    int nearestUnvisited(int query, const vector<bool>& visited) const {
        float qx = (*px)[query], qy = (*py)[query];
        int cx = min((int)((qx-minX)/cellW), gridW-1);
        int cy = min((int)((qy-minY)/cellH), gridH-1);
        
        float bestDist = 1e30f;
        int bestIdx = -1;
        
        int maxRadius = max(gridW, gridH);
        for (int r = 0; r <= maxRadius && bestIdx == -1; r++) {
            // Early termination: if min possible dist in ring > bestDist
            float minCellDist = max(0.0f, (r-1) * min(cellW, cellH));
            if (minCellDist * minCellDist > bestDist && bestIdx != -1) break;
            
            int x0 = max(0, cx-r), x1 = min(gridW-1, cx+r);
            int y0 = max(0, cy-r), y1 = min(gridH-1, cy+r);
            
            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    // Only process border of ring when r>0
                    if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                    for (int idx : cells[gy*gridW+gx]) {
                        if (visited[idx]) continue;
                        float dx = (*px)[idx]-qx, dy = (*py)[idx]-qy;
                        float d = dx*dx+dy*dy;
                        if (d < bestDist) { bestDist = d; bestIdx = idx; }
                    }
                }
            }
            
            if (bestIdx != -1) {
                // Check if we can do better in next ring
                float nextMinDist = max(0.0f, r * min(cellW, cellH));
                if (nextMinDist * nextMinDist > bestDist) break;
            }
        }
        return bestIdx;
    }
};

// ==================== Greedy nearest neighbor ====================

vector<int> greedyNN(int n, const vector<float>& x, const vector<float>& y) {
    GridIndex grid;
    int cellSize = (n > 1000000) ? 8 : 4;
    grid.build(x, y, n, cellSize);
    
    vector<bool> visited(n, false);
    vector<int> tour;
    tour.reserve(n);
    
    int cur = 0;
    visited[0] = true;
    tour.push_back(0);
    
    for (int step = 1; step < n; step++) {
        int next = grid.nearestUnvisited(cur, visited);
        if (next == -1) {
            // fallback: find any unvisited
            for (int i = 0; i < n; i++) if (!visited[i]) { next = i; break; }
        }
        visited[next] = true;
        tour.push_back(next);
        cur = next;
    }
    return tour;
}

inline float dist2(int a, int b, const vector<float>& x, const vector<float>& y) {
    float dx = x[a]-x[b], dy = y[a]-y[b];
    return dx*dx+dy*dy;
}

inline float distf(int a, int b, const vector<float>& x, const vector<float>& y) {
    float dx = x[a]-x[b], dy = y[a]-y[b];
    return sqrtf(dx*dx+dy*dy);
}

// 2-opt with neighbor lists
void twoOpt(vector<int>& tour, const vector<float>& x, const vector<float>& y, double timeLimit, int maxNeighbors = 10) {
    int n = tour.size();
    if (n < 4) return;
    
    // Build position array: pos[city] = index in tour
    vector<int> pos(n);
    for (int i = 0; i < n; i++) pos[tour[i]] = i;
    
    // Build neighbor lists using grid
    GridIndex grid;
    grid.build(x, y, n, 4);
    
    int knn = min(maxNeighbors, n-1);
    vector<vector<int>> neighbors(n);
    
    // For each city, find k nearest neighbors
    vector<bool> dummy(n, false);
    // Use grid to find approximate neighbors - but grid's nearestUnvisited needs modification
    // Instead, build neighbor list differently
    
    for (int i = 0; i < n && elapsed() < timeLimit * 0.3; i++) {
        float qx = x[i], qy = y[i];
        int cx = min((int)((qx-grid.minX)/grid.cellW), grid.gridW-1);
        int cy = min((int)((qy-grid.minY)/grid.cellH), grid.gridH-1);
        
        // collect candidates from expanding rings
        vector<pair<float,int>> cands;
        int maxR = max(grid.gridW, grid.gridH);
        for (int r = 0; r <= maxR; r++) {
            int x0 = max(0, cx-r), x1 = min(grid.gridW-1, cx+r);
            int y0 = max(0, cy-r), y1 = min(grid.gridH-1, cy+r);
            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                    for (int idx : grid.cells[gy*grid.gridW+gx]) {
                        if (idx == i) continue;
                        float dx = x[idx]-qx, dy = y[idx]-qy;
                        cands.push_back({dx*dx+dy*dy, idx});
                    }
                }
            }
            if ((int)cands.size() >= knn*2) break;
            float nd = r * min(grid.cellW, grid.cellH);
            if (!cands.empty() && nd*nd > cands[0].first * 4) break;
        }
        partial_sort(cands.begin(), cands.begin()+min((int)cands.size(),knn), cands.end());
        for (int j = 0; j < min((int)cands.size(), knn); j++)
            neighbors[i].push_back(cands[j].second);
    }
    
    bool improved = true;
    while (improved && elapsed() < timeLimit) {
        improved = false;
        for (int idx = 0; idx < n && elapsed() < timeLimit; idx++) {
            int i = tour[idx];
            int i1 = tour[(idx+1)%n];
            float d_i = distf(i, i1, x, y);
            
            for (int nb : neighbors[i]) {
                int j = pos[nb];
                int j1_idx = (j+1)%n;
                int j1 = tour[j1_idx];
                if (j == idx || j1 == i) continue;
                
                float gain = d_i + distf(j, j1, x, y)
                           - distf(i, j, x, y) - distf(i1, j1, x, y);
                if (gain > 1e-6f) {
                    // Reverse segment between i1 and j (or j1 and i depending on order)
                    int a = (idx+1)%n, b = j;
                    if (a <= b) {
                        reverse(tour.begin()+a, tour.begin()+b+1);
                    } else {
                        // wrap-around reverse - more complex
                        // just skip for simplicity
                        continue;
                    }
                    for (int k = min(a,b); k <= max(a,b); k++) pos[tour[k]] = k;
                    i1 = tour[(idx+1)%n];
                    d_i = distf(i, i1, x, y);
                    improved = true;
                }
            }
        }
    }
}

// Simple 2-opt without neighbor lists (for medium N)
void twoOptSimple(vector<int>& tour, const vector<float>& x, const vector<float>& y, double timeLimit) {
    int n = tour.size();
    bool improved = true;
    while (improved && elapsed() < timeLimit) {
        improved = false;
        for (int i = 0; i < n-1 && elapsed() < timeLimit; i++) {
            for (int j = i+2; j < n; j++) {
                if (i == 0 && j == n-1) continue;
                float d1 = distf(tour[i], tour[i+1], x, y) + distf(tour[j], tour[(j+1)%n], x, y);
                float d2 = distf(tour[i], tour[j], x, y) + distf(tour[i+1], tour[(j+1)%n], x, y);
                if (d2 < d1 - 1e-6f) {
                    reverse(tour.begin()+i+1, tour.begin()+j+1);
                    improved = true;
                }
            }
        }
    }
}

// ==================== Or-opt (move single cities) ====================
void orOpt(vector<int>& tour, const vector<float>& x, const vector<float>& y, double timeLimit) {
    int n = tour.size();
    bool improved = true;
    while (improved && elapsed() < timeLimit) {
        improved = false;
        for (int i = 0; i < n && elapsed() < timeLimit; i++) {
            int prev = (i-1+n)%n, next = (i+1)%n;
            int city = tour[i];
            int pc = tour[prev], nc = tour[next];
            float removeCost = distf(pc, city, x, y) + distf(city, nc, x, y) - distf(pc, nc, x, y);
            
            // Try inserting after each other position
            for (int j = 0; j < n; j++) {
                if (j == prev || j == i) continue;
                int jn = (j+1)%n;
                if (jn == i) continue;
                int jc = tour[j], jnc = tour[jn];
                float insertCost = distf(jc, city, x, y) + distf(city, jnc, x, y) - distf(jc, jnc, x, y);
                if (insertCost < removeCost - 1e-6f) {
                    // Remove city from position i and insert after j
                    tour.erase(tour.begin()+i);
                    int ins = (j > i) ? j : j+1;
                    if (ins >= (int)tour.size()) ins = tour.size();
                    tour.insert(tour.begin()+ins, city);
                    improved = true;
                    // Recompute i
                    i = max(0, i-1);
                    break;
                }
            }
        }
    }
}

// ==================== Exact TSP (bitmask DP) for n <= 20 ====================
static vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y) {
    int S = 1 << n;
    // dp[mask][v] = min cost to visit exactly cities in mask, ending at v, starting from 0
    vector<vector<float>> dp(S, vector<float>(n, 1e30f));
    vector<vector<short>> par(S, vector<short>(n, -1));
    dp[1][0] = 0.0f;

    auto d = [&](int a, int b) {
        float dx = x[a]-x[b], dy = y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };

    for (int mask = 1; mask < S; mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask >> u & 1)) continue;
            if (dp[mask][u] > 1e29f) continue;
            float cur = dp[mask][u];
            for (int v = 0; v < n; v++) {
                if (mask >> v & 1) continue;
                int nm = mask | (1<<v);
                float nd = cur + d(u, v);
                if (nd < dp[nm][v]) {
                    dp[nm][v] = nd;
                    par[nm][v] = (short)u;
                }
            }
        }
    }

    int full = S-1;
    float best = 1e30f; int last = 1;
    for (int v = 1; v < n; v++) {
        float t = dp[full][v] + d(v, 0);
        if (t < best) { best = t; last = v; }
    }

    vector<int> tour;
    int mask = full, cur = last;
    while (cur >= 0) {
        tour.push_back(cur);
        int p = par[mask][cur];
        mask ^= (1<<cur);
        cur = p;
    }
    reverse(tour.begin(), tour.end());
    return tour;
}

// ==================== Grid spatial index ====================
struct Grid {
    float mnX, mnY, mxX, mxY, cW, cH;
    int gW, gH;
    vector<vector<int>> cells;
    const float* px; const float* py;

    void build(const float* x, const float* y, int n, float loadFactor = 4.0f) {
        px = x; py = y;
        mnX = mnY = 1e30f; mxX = mxY = -1e30f;
        for (int i = 0; i < n; i++) {
            if (x[i]<mnX) mnX=x[i]; if (x[i]>mxX) mxX=x[i];
            if (y[i]<mnY) mnY=y[i]; if (y[i]>mxY) mxY=y[i];
        }
        float rx = mxX-mnX+1e-6f, ry = mxY-mnY+1e-6f;
        int total = max(1, (int)(n / loadFactor));
        float ratio = rx/ry;
        gH = max(1,(int)sqrt(total/ratio));
        gW = max(1, total/gH);
        // memory cap: ~40M cells
        while ((long long)gW*gH > 40000000LL) { gW=max(1,gW/2); gH=max(1,gH/2); }
        cW = rx/gW; cH = ry/gH;
        cells.assign((size_t)gW*gH, {});
        for (int i = 0; i < n; i++) {
            int cx = min((int)((x[i]-mnX)/cW), gW-1);
            int cy = min((int)((y[i]-mnY)/cH), gH-1);
            cells[(size_t)cy*gW+cx].push_back(i);
        }
    }

    // nearest unvisited to city q
    int nearest(int q, const vector<bool>& vis) const {
        float qx=px[q], qy=py[q];
        int cx=min((int)((qx-mnX)/cW),gW-1);
        int cy=min((int)((qy-mnY)/cH),gH-1);
        float best=1e30f; int bi=-1;
        int maxR=max(gW,gH);
        for (int r=0; r<=maxR; r++) {
            // check if ring can improve
            if (r > 0 && bi >= 0) {
                float md = (r-1)*min(cW,cH);
                if (md*md > best) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for (int gy=y0;gy<=y1;gy++) for (int gx=x0;gx<=x1;gx++) {
                if (r>0 && gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                for (int idx : cells[(size_t)gy*gW+gx]) {
                    if (vis[idx]) continue;
                    float dx=px[idx]-qx, dy=py[idx]-qy;
                    float d=dx*dx+dy*dy;
                    if (d<best){best=d;bi=idx;}
                }
            }
        }
        return bi;
    }

    // get k nearest to point (qx,qy), excluding self
    void kNearest(int q, int k, vector<int>& out) const {
        float qx=px[q], qy=py[q];
        int cx=min((int)((qx-mnX)/cW),gW-1);
        int cy=min((int)((qy-mnY)/cH),gH-1);
        
        vector<pair<float,int>> cands;
        cands.reserve(k*4);
        int maxR=max(gW,gH);
        
        for (int r=0; r<=maxR; r++) {
            if (r>0 && (int)cands.size()>=k) {
                float md=(r-1)*min(cW,cH);
                // best k-th so far
                if ((int)cands.size()>=k) {
                    // check if nth_element done
                    float kth = cands[k-1].first; // not sorted yet
                    if (md*md > kth) break;
                }
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for (int gy=y0;gy<=y1;gy++) for (int gx=x0;gx<=x1;gx++) {
                if (r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                for (int idx : cells[(size_t)gy*gW+gx]) {
                    if (idx==q) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy;
                    cands.push_back({dx*dx+dy*dy,idx});
                }
            }
            if ((int)cands.size()>=k*3 && r>0) {
                // sort and check
                nth_element(cands.begin(),cands.begin()+k-1,cands.end());
                float kth=cands[k-1].first;
                float md=r*min(cW,cH);
                if (md*md > kth) break;
                cands.resize(k);
            }
        }
        if ((int)cands.size()>k) {
            partial_sort(cands.begin(),cands.begin()+k,cands.end());
            cands.resize(k);
        } else {
            sort(cands.begin(),cands.end());
        }
        out.clear();
        for (auto& p:cands) out.push_back(p.second);
    }
};

// ==================== Greedy nearest neighbor tour ====================
static vector<int> greedyNN(int n, const float* x, const float* y) {
    float lf = 4.0f;
    if (n > 500000) lf = 8.0f;
    if (n > 2000000) lf = 16.0f;
    
    Grid grid;
    grid.build(x, y, n, lf);
    
    vector<bool> vis(n, false);
    vector<int> tour;
    tour.reserve(n);
    
    int cur = 0;
    vis[0]=true; tour.push_back(0);
    
    for (int step=1; step<n; step++) {
        int nxt = grid.nearest(cur, vis);
        if (nxt < 0) {
            for (int i=0;i<n;i++) if (!vis[i]){nxt=i;break;}
        }
        vis[nxt]=true;
        tour.push_back(nxt);
        cur=nxt;
        
        if (step % 1000000 == 0) {
            // progress check
            if (elapsed() > 250.0) {
                // fill remaining randomly
                for (int i=0;i<n;i++) if (!vis[i]) { tour.push_back(i); }
                break;
            }
        }
    }
    return tour;

// ==================== 2-opt with neighbor lists ====================
static void twoOptNeighbors(vector<int>& tour, const float* x, const float* y, 
                             double tLimit, int K=10) {
    int n=(int)tour.size();
    if (n<4) return;
    
    // Build position lookup
    vector<int> pos(n);
    for (int i=0;i<n;i++) pos[tour[i]]=i;
    
    // Build neighbor lists
    Grid grid;
    grid.build(x,y,n,4.0f);
    
    vector<vector<int>> nb(n);
    for (int i=0;i<n && elapsed()<tLimit*0.4;i++) {
        grid.kNearest(i, K, nb[i]);
    }
    
    auto D=[&](int a,int b){
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };
    
    bool improved=true;
    while(improved && elapsed()<tLimit) {
        improved=false;
        for (int idx=0;idx<n && elapsed()<tLimit;idx++) {
            int t1=tour[idx];
            int t2=tour[(idx+1)%n];
            float d_t1t2=D(t1,t2);
            
            for (int t3 : nb[t1]) {
                // t3 is a neighbor of t1
                // try reversing segment between t2 and t3 (or t4 and t1)
                int j=pos[t3];
                int t4=tour[(j+1)%n];
                if (t3==t2||t4==t1||t3==t1) continue;
                
                float gain = d_t1t2 + D(t3,t4) - D(t1,t3) - D(t2,t4);
                if (gain > 1e-6f) {
                    // Reverse the segment from min(idx+1,j+1) to max(idx,j)
                    // Standard 2-opt: if idx < j, reverse tour[idx+1..j]
                    int a, b;
                    if (idx < j) { a=idx+1; b=j; }
                    else { a=j+1; b=idx; }
                    
                    while(a<b) {
                        swap(tour[a],tour[b]);
                        pos[tour[a]]=a;
                        pos[tour[b]]=b;
                        a++; b--;
                    }
                    if(a==b) pos[tour[a]]=a;
                    
                    t2=tour[(idx+1)%n];
                    d_t1t2=D(t1,t2);
                    improved=true;
                }
            }
        }
    }
}

static void twoOptSimple(vector<int>& tour, const float* x, const float* y, double tLimit) {
    int n=(int)tour.size();
    auto D=[&](int a,int b){
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);

    bool imp=true;
    while(imp && elapsed()<tLimit) {
        imp=false;
        for (int i=0;i<n-1 && elapsed()<tLimit;i++) {

                    reverse(tour.begin()+i+1,tour.begin()+j+1);
                    imp=true;
                }
            }
        }
    }
}

// ==================== Or-opt: move segments of length 1,2,3 ====================
static void orOpt(vector<int>& tour, const float* x, const float* y, double tLimit, int segLen=1) {
    int n=(int)tour.size();
    if(n<4) return;
    auto D=[&](int a,int b){
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };
    
    bool imp=true;
    while(imp && elapsed()<tLimit) {
        imp=false;
        for (int i=0;i<n && elapsed()<tLimit;i++) {
            // segment: tour[i], tour[i+1], ..., tour[i+segLen-1]
            int prev=(i-1+n)%n;
            int last=(i+segLen-1)%n;
            int after=(i+segLen)%n;
            if(prev==last||prev==after) continue;
            
            int cp=tour[prev], ci=tour[i], cl=tour[last], ca=tour[after];
            float rem=D(cp,ci)+D(cl,ca)-D(cp,ca);
            
            // try inserting between j and j+1
            int best_j=-1; float best_gain=-1e-9f;
            for (int j=0;j<n;j++) {
                // don't insert adjacent to current position
                if (j==prev||j==i||j==last) continue;
                bool inSeg=false;
                // check if j is within the segment
                for(int k=0;k<segLen;k++) if(j==(i+k)%n){inSeg=true;break;}
                if(inSeg) continue;
                int jn=(j+1)%n;
                bool inSeg2=false;
                for(int k=0;k<segLen;k++) if(jn==(i+k)%n){inSeg2=true;break;}
                if(inSeg2) continue;
                
                int cj=tour[j],cjn=tour[jn];
                float ins=D(cj,ci)+D(cl,cjn)-D(cj,cjn);
                float gain=rem-ins;
                if(gain>best_gain){best_gain=gain;best_j=j;}
            }
            
            if(best_j>=0) {
                // Extract segment and reinsert
                // Build new tour
                vector<int> seg;
                for(int k=0;k<segLen;k++) seg.push_back(tour[(i+k)%n]);
                
                // Remove segment from tour
                vector<int> rest;
                rest.reserve(n-segLen);
                // We need to handle wrap-around carefully
                // Mark positions
                vector<bool> inS(n,false);
                for(int k=0;k<segLen;k++) inS[(i+k)%n]=true;
                for(int k=0;k<n;k++) if(!inS[k]) rest.push_back(tour[k]);
                
                // Find best_j city in rest
                int insertAfter=-1;
                int bj_city=tour[best_j];
                for(int k=0;k<(int)rest.size();k++) if(rest[k]==bj_city){insertAfter=k;break;}
                
                // Insert segment after insertAfter
                vector<int> newt;
                newt.reserve(n);
                for(int k=0;k<=insertAfter;k++) newt.push_back(rest[k]);
                for(int c:seg) newt.push_back(c);
                for(int k=insertAfter+1;k<(int)rest.size();k++) newt.push_back(rest[k]);
                
                tour=newt;
                imp=true;
                i=max(0,i-1);
            }
        }
    }
}

vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y) {

3. **2-opt neighbor**: The reversal when `ii>jj` case needs thought. In standard 2-opt on a circular tour, reversing segment (t2..t3) or (t4..t1) - we pick the shorter one or the one that doesn't wrap
4. **Memory**: For 10M cities, `vector<vector<int>> nb(n)` with K=7 neighbors = 70M ints = ~280MB, which should be OK

Actually for 10M+ cities, even building neighbor lists will be slow and memory-heavy. Let me adjust the strategy:
- For n > 5M: just greedy NN, no 2-opt
- For n > 1M: 2-opt with very few neighbors and limited passes

// Exact TSP via bitmask DP (only for n <= 20)

static vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y){
    int S=1<<n;
    vector<vector<float>> dp(S, vector<float>(n,1e30f));
    vector<vector<int>>   pa(S, vector<int>(n,-1));
    dp[1][0]=0.0f;
    for(int m=1;m<S;m++) for(int u=0;u<n;u++){
        if(!(m>>u&1)||dp[m][u]>1e29f) continue;
        for(int v=0;v<n;v++){
            if(m>>v&1) continue;
            float dx=x[u]-x[v],dy=y[u]-y[v];
            float nd=dp[m][u]+sqrtf(dx*dx+dy*dy);
            if(nd<dp[m|(1<<v)][v]){dp[m|(1<<v)][v]=nd; pa[m|(1<<v)][v]=u;}
        }
    }
    int F=S-1; float best=1e30f; int last=0;
    for(int v=1;v<n;v++){
        float dx=x[v]-x[0],dy=y[v]-y[0];
        float t=dp[F][v]+sqrtf(dx*dx+dy*dy);
        if(t<best){best=t;last=v;}
    }
    vector<int> tour; int m=F,c=last;
    while(c>=0){tour.push_back(c);int p=pa[m][c];m^=(1<<c);c=p;}
    reverse(tour.begin(),tour.end());
    return tour;
}

//===========================================================================
// Grid-based spatial index (compact, flat storage)
//===========================================================================
struct Grid {
    float mnX,mnY,cW,cH;
    int   gW,gH,N;
    const float *px,*py;
    vector<int> data,offset; // CSR format

    void build(const float* x, const float* y, int n, float lf=4.0f){
        px=x; py=y; N=n;
        float mxX=x[0],mxY=y[0]; mnX=x[0]; mnY=y[0];
        for(int i=1;i<n;i++){
            if(x[i]<mnX)mnX=x[i]; if(x[i]>mxX)mxX=x[i];
            if(y[i]<mnY)mnY=y[i]; if(y[i]>mxY)mxY=y[i];
        }
        float rx=mxX-mnX+1e-6f, ry=mxY-mnY+1e-6f;
        long long total=max(1LL,(long long)(n/lf));
        float ratio=sqrtf(rx/ry);
        gW=max(1,(int)(total*ratio)); // roughly gW*gH ~ total, gW/gH ~ rx/ry
        // actually:  gW = sqrt(total * rx/ry), gH = sqrt(total * ry/rx)
        // simpler:
        gH=max(1,(int)sqrtf((float)total/sqrtf(rx/ry)));
        gW=max(1,(int)(total/gH));
        // memory cap ~30M cells
        while((long long)gW*gH>30000000LL){gW=max(1,gW/2);gH=max(1,gH/2);}
        cW=rx/gW; cH=ry/gH;
        long long nc=(long long)gW*gH;
        
        // count phase
        offset.assign(nc+1,0);
        vector<int> cellOf(n);
        for(int i=0;i<n;i++){
            int cx=min((int)((x[i]-mnX)/cW),gW-1);
            int cy=min((int)((y[i]-mnY)/cH),gH-1);
            cellOf[i]=cy*gW+cx;
            offset[cellOf[i]+1]++;
        }
        for(long long i=0;i<nc;i++) offset[i+1]+=offset[i];
        data.resize(n);
        vector<int> tmp(nc,0);
        for(int i=0;i<n;i++){
            int c=cellOf[i];
            data[offset[c]+tmp[c]]=i;
            tmp[c]++;
        }
    }

    inline int cellID(float qx,float qy) const {
        int cx=min((int)((qx-mnX)/cW),gW-1);
        int cy=min((int)((qy-mnY)/cH),gH-1);
        return cy*gW+cx;
    }

    // nearest unvisited city to city q
    int nearestUnvisited(int q, const vector<bool>& vis) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float best=1e30f; int bi=-1;
        for(int r=0;r<=gW+gH;r++){
            // early exit: nearest possible city in next ring
            if(bi>=0){
                float md=(float)(r-1)*min(cW,cH);
                if(md>0&&md*md>=best) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            if(x0>x1||y0>y1) { if(bi>=0) break; continue; }
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k];
                    if(vis[idx]) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy;
                    float d=dx*dx+dy*dy;
                    if(d<best){best=d;bi=idx;}
                }
            }
        }
        return bi;
    }

    // k nearest to city q (not including itself)
    void kNearest(int q, int K, vector<int>& out) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        
        vector<pair<float,int>> cands;
        cands.reserve(K*4);
        float kthDist=1e30f;
        int found=0;
        
        for(int r=0;r<=gW+gH;r++){
            if(found>=K){
                float md=(float)(r-1)*min(cW,cH);
                if(md>0&&md*md>=kthDist) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            if(x0>x1||y0>y1) {if(found>=K) break; continue;}
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k];
                    if(idx==q) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy;
                    float d=dx*dx+dy*dy;
                    cands.push_back({d,idx});
                    found++;
                }
            }
            // periodically trim
            if((int)cands.size()>=K*8){
                nth_element(cands.begin(),cands.begin()+K,cands.end());
                cands.resize(K);
                kthDist=0;
                for(auto&p:cands) if(p.first>kthDist) kthDist=p.first;
                found=K;
            }
        }
        
        if((int)cands.size()>K){
            partial_sort(cands.begin(),cands.begin()+K,cands.end());
            cands.resize(K);
        } else sort(cands.begin(),cands.end());
        
        out.clear();
        for(auto&p:cands) out.push_back(p.second);
    }
};

//===========================================================================
// Greedy nearest-neighbor tour
//===========================================================================
static vector<int> greedyNN(int n, const float* x, const float* y){
    float lf=(n>5000000)?16.0f:(n>1000000)?8.0f:4.0f;
    Grid g; g.build(x,y,n,lf);
    vector<bool> vis(n,false);
    vector<int> tour; tour.reserve(n);
    int cur=0; vis[0]=true; tour.push_back(0);
    for(int s=1;s<n;s++){
        int nxt=g.nearestUnvisited(cur,vis);
        if(nxt<0){for(int i=0;i<n;i++)if(!vis[i]){nxt=i;break;}}
        vis[nxt]=true; tour.push_back(nxt); cur=nxt;
        if((s&0xFFFFF)==0&&s>0){
            if(elapsed()>245.0){
                // emergency fill
                for(int i=0;i<n;i++)if(!vis[i])tour.push_back(i);
                return tour;
            }
        }
    }
    return tour;
}

//===========================================================================
// 2-opt with neighbor candidate lists
//===========================================================================
static void twoOptNN(vector<int>& tour, const float* x, const float* y, 
                     double tLim, int K=10){
    int n=(int)tour.size();
    if(n<4) return;
    
    vector<int> pos(n);
    for(int i=0;i<n;i++) pos[tour[i]]=i;
    
    auto D=[&](int a,int b)->float{
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };
    
    // Build neighbor lists (use fraction of time)
    double buildFrac=0.3;
    double buildEnd=elapsed()+(tLim-elapsed())*buildFrac;
    
    Grid g; g.build(x,y,n,4.0f);
    
    int actualK=K;
    vector<vector<int>> nb(n);
    for(int i=0;i<n;i++){
        if(elapsed()>buildEnd) { actualK=max(3,actualK-1); }
        g.kNearest(i,actualK,nb[i]);
    }
    
    // 2-opt passes
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int ii=0;ii<n&&elapsed()<tLim;ii++){
            int t1=tour[ii];
            int t2=tour[(ii+1)%n];
            float d_t1_t2=D(t1,t2);
            
            for(int t3:nb[t1]){
                // potential improvement requires d(t1,t3) < d(t1,t2)+d(t3,t4)
                // quick check: if d(t1,t3) >= d(t1,t2), can still improve if d(t3,t4) is large
                // but since nb is sorted by distance, once d(t1,t3) > d(t1,t2)+something, break
                float d13=D(t1,t3);
                if(d13>=d_t1_t2+1.0f) break; // roughly: if t3 is far, no improvement possible
                
                int jj=pos[t3];
                int t4=tour[(jj+1)%n];
                if(t3==t2||t4==t1||t3==t1||t4==t2) continue;
                
                float gain=d_t1_t2+D(t3,t4)-d13-D(t2,t4);
                if(gain>1e-6f){
                    // reverse appropriate segment
                    // In a linear 2-opt: reverse tour[min+1..max]
                    int a,b;
                    if(ii<jj){a=ii+1;b=jj;}
                    else{a=jj+1;b=ii;}
                    while(a<b){
                        swap(tour[a],tour[b]);
                        pos[tour[a]]=a; pos[tour[b]]=b;
                        a++;b--;
                    }
                    if(a==b) pos[tour[a]]=a;
                    t2=tour[(ii+1)%n];
                    d_t1_t2=D(t1,t2);
                    imp=true;
                }
            }
        }
    }
}

//===========================================================================
// Simple O(n^2) 2-opt for small N
//===========================================================================
static void twoOptFull(vector<int>& tour, const float* x, const float* y, double tLim){
    int n=(int)tour.size();
    auto D=[&](int a,int b)->float{
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int i=0;i<n-1&&elapsed()<tLim;i++){
            for(int j=i+2;j<n;j++){
                if(i==0&&j==n-1) continue;
                float b4=D(tour[i],tour[i+1])+D(tour[j],tour[(j+1)%n]);
                float af=D(tour[i],tour[j])+D(tour[i+1],tour[(j+1)%n]);
                if(af<b4-1e-6f){
                    reverse(tour.begin()+i+1,tour.begin()+j+1);
                    imp=true;
                }
            }
        }
    }
}

There are still some issues I'm thinking about:

1. In `kNearest`, when I trim `cands`, I need to properly recompute `kthDist` as the max of the remaining K elements, not iterate to find max
2. The early termination in 2-opt with `if(d13>=d_t1_t2+1.0f) break;` - this is wrong because we'd need the actual distance units, plus it assumes sorted order in `nb` which kNearest provides
3. Memory: For 10M cities, `vector<vector<int>> nb(n)` = 10M vectors = maybe 240 bytes each = 2.4GB just for the outer vectors. I need a flat neighbor array instead.

//===========================================================================
// Exact TSP bitmask DP  (n <= 20)
//===========================================================================
static vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y){
    int S=1<<n;
    vector<vector<float>> dp(S, vector<float>(n,1e30f));
    vector<vector<int>>   pa(S, vector<int>(n,-1));
    dp[1][0]=0.0f;

//===========================================================================
// Grid spatial index (flat CSR storage)
//===========================================================================
struct Grid {
    float mnX,mnY,cW,cH;
    int   gW,gH,N;
    const float *px,*py;
    vector<int> data,offset;

    void build(const float* x, const float* y, int n, float lf=4.0f){
        px=x; py=y; N=n;
        float mxX=x[0],mxY=y[0]; mnX=x[0]; mnY=y[0];
        for(int i=1;i<n;i++){
            mnX=min(mnX,x[i]); mxX=max(mxX,x[i]);
            mnY=min(mnY,y[i]); mxY=max(mxY,y[i]);
        }
        float rx=mxX-mnX+1e-6f, ry=mxY-mnY+1e-6f;
        double total=max(1.0,(double)n/lf);
        double ratio=rx/ry;
        gH=max(1,(int)sqrt(total/ratio));
        gW=max(1,(int)(total/gH));
        while((long long)gW*gH>30000000LL){gW=max(1,gW*7/8);gH=max(1,gH*7/8);}
        cW=rx/gW; cH=ry/gH;
        
        long long nc=(long long)gW*gH;
        offset.assign((size_t)(nc+1),0);
        
        // First pass: count
        vector<int> cof(n);
        for(int i=0;i<n;i++){
            int cx=min((int)((x[i]-mnX)/cW),gW-1);
            int cy=min((int)((y[i]-mnY)/cH),gH-1);
            cof[i]=cy*gW+cx;
            offset[(size_t)cof[i]+1]++;
        }
        for(size_t i=0;i<(size_t)nc;i++) offset[i+1]+=offset[i];
        
        data.resize(n);
        vector<int> tmp((size_t)nc,0);
        for(int i=0;i<n;i++){
            int c=cof[i];
            data[offset[c]+tmp[c]]=i;
            tmp[c]++;
        }
    }

    int nearestUnvisited(int q, const vector<bool>& vis) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float best=1e30f; int bi=-1;
        float mc=min(cW,cH);
        for(int r=0;;r++){
            if(bi>=0){
                float md=max(0.0f,(float)(r-1))*mc;
                if(md*md>=best) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            if(x0>cx+r||y0>cy+r) {if(bi>=0) break; continue;}
            bool anyNew=false;
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                anyNew=true;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k];
                    if(vis[idx]) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(d<best){best=d;bi=idx;}
                }
            }
            if(!anyNew&&r>0) break;
            if(r>gW+gH) break;
        }
        return bi;
    }

    // Returns K nearest (sorted) into out
    void kNearest(int q, int K, vector<int>& out) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float mc=min(cW,cH);
        
        // We'll collect in a small buffer and nth_element periodically
        vector<pair<float,int>> cands;
        cands.reserve(K*8);
        float kthDist=1e30f;
        int found=0;
        
        for(int r=0;;r++){
            if(found>=K){
                float md=max(0.0f,(float)(r-1))*mc;
                if(md*md>=kthDist) break;
            }
            if(r>gW+gH) break;
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            bool anyNew=false;
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                anyNew=true;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k];
                    if(idx==q) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(d<kthDist||(int)cands.size()<K){
                        cands.push_back({d,idx});
                        found++;
                    }
                }
            }
            // Trim periodically
            if(found>=K*8||(found>=K&&!anyNew)){
                if((int)cands.size()>K){
                    nth_element(cands.begin(),cands.begin()+K-1,cands.end());
                    kthDist=cands[K-1].first;
                    // Remove elements beyond K
                    int sz=(int)cands.size();
                    int kept=0;
                    for(int i=0;i<sz;i++) if(cands[i].first<=kthDist) cands[kept++]=cands[i];
                    cands.resize(min(kept,(int)K*2));
                    found=(int)cands.size();
                }
            }
            if(!anyNew&&r>0) break;
        }
        
        if((int)cands.size()>K){
            partial_sort(cands.begin(),cands.begin()+K,cands.end());
            cands.resize(K);
        } else sort(cands.begin(),cands.end());

static void twoOptNN(vector<int>& tour, const float* x, const float* y,
                     double tLim, int K){
    int n=(int)tour.size();
    if(n<4) return;
    
    vector<int> pos(n);
    for(int i=0;i<n;i++) pos[tour[i]]=i;
    
    auto D=[&](int a,int b)->float{
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };
    
    // Build flat neighbor array
    double buildEnd=min(elapsed()+(tLim-elapsed())*0.35, tLim-1.0);
    Grid g; g.build(x,y,n,4.0f);
    
    // Flat storage: nbData[i*K .. i*K+K-1] = neighbors of city i
    vector<int> nbData((size_t)n*K, -1);
    vector<int> tmp;
    
    for(int i=0;i<n;i++){
        int ki=(elapsed()<buildEnd)?K:min(K,5);
        g.kNearest(i,ki,tmp);
        int base=i*K;
        for(int j=0;j<(int)tmp.size()&&j<K;j++) nbData[base+j]=tmp[j];
    }
    
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int ii=0;ii<n&&elapsed()<tLim;ii++){
            int t1=tour[ii];
            int t2=tour[(ii+1)%n];
            float d12=D(t1,t2);
            int base=t1*K;
            
            for(int ki=0;ki<K;ki++){
                int t3=nbData[base+ki];
                if(t3<0) break;
                float d13=D(t1,t3);
                // If d(t1,t3) >= d(t1,t2), reversing can only help if d(t3,t4) > d(t2,t4)+d(t1,t2)-d(t1,t3)
                // Simple bound: if d13 >= d12 + current best possible d34, break
                // Since neighbors sorted by distance, if d13 > d12 we can't guarantee improvement
                // but let's not break early here since it might still improve
                
                int jj=pos[t3];
                int t4=tour[(jj+1)%n];
                if(t3==t2||t4==t1||t3==t1) continue;
                
                float gain=d12+D(t3,t4)-d13-D(t2,t4);
                if(gain>1e-6f){
                    int a,b;
                    if(ii<jj){a=ii+1;b=jj;}
                    else{a=jj+1;b=ii;}
                    // reverse tour[a..b]
                    int l=a,r2=b;
                    while(l<r2){
                        swap(tour[l],tour[r2]);
                        pos[tour[l]]=l; pos[tour[r2]]=r2;
                        l++;r2--;
                    }
                    if(l==r2) pos[tour[l]]=l;
                    t2=tour[(ii+1)%n];
                    d12=D(t1,t2);
                    imp=true;
                }
            }
        }
    }
}

// Full O(n^2) 2-opt
//===========================================================================
static void twoOptFull(vector<int>& tour, const float* x, const float* y, double tLim){
    int n=(int)tour.size();
    auto D=[&](int a,int b)->float{
        float dx=x[a]-x[b],dy=y[a]-y[b];
        return sqrtf(dx*dx+dy*dy);
    };
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int i=0;i<n-1&&elapsed()<tLim;i++){
            for(int j=i+2;j<n;j++){
                if(i==0&&j==n-1) continue;
                float b4=D(tour[i],tour[i+1])+D(tour[j],tour[(j+1)%n]);
                float af=D(tour[i],tour[j])+D(tour[i+1],tour[(j+1)%n]);
                if(af<b4-1e-6f){
                    reverse(tour.begin()+i+1,tour.begin()+j+1);
                    imp=true;
                }
            }
        }
    }
}

static vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y){
    int S=1<<n;
    vector<vector<float>> dp(S, vector<float>(n,1e30f));
    vector<vector<int>> pa(S, vector<int>(n,-1));
    dp[1][0]=0.0f;
    for(int m=1;m<S;m++) for(int u=0;u<n;u++){
        if(!(m>>u&1)||dp[m][u]>1e29f) continue;
        for(int v=0;v<n;v++){
            if(m>>v&1) continue;
            float dx=x[u]-x[v],dy=y[u]-y[v];
            float nd=dp[m][u]+sqrtf(dx*dx+dy*dy);
            int nm=m|(1<<v);
            if(nd<dp[nm][v]){dp[nm][v]=nd;pa[nm][v]=u;}
        }
    }
    int F=S-1; float best=1e30f; int last=0;
    for(int v=1;v<n;v++){
        float dx=x[v]-x[0],dy=y[v]-y[0];
        float t=dp[F][v]+sqrtf(dx*dx+dy*dy);
        if(t<best){best=t;last=v;}
    }
    vector<int> t; int m=F,c=last;
    while(c>=0){t.push_back(c);int p=pa[m][c];m^=(1<<c);c=p;}
    reverse(t.begin(),t.end());
    return t;
}

// Grid index with CSR storage
struct Grid {
    float mnX,mnY,cW,cH;
    int gW,gH;
    const float *px,*py;
    vector<int> data, offset;

    void build(const float* x,const float* y,int n,float lf=4.0f){
        px=x; py=y;
        float mxX=x[0],mxY=y[0]; mnX=x[0]; mnY=y[0];
        for(int i=1;i<n;i++){
            if(x[i]<mnX)mnX=x[i]; if(x[i]>mxX)mxX=x[i];
            if(y[i]<mnY)mnY=y[i]; if(y[i]>mxY)mxY=y[i];
        }
        float rx=mxX-mnX+1e-6f,ry=mxY-mnY+1e-6f;
        double tot=max(1.0,(double)n/lf);
        gH=max(1,(int)sqrt(tot*ry/rx));
        gW=max(1,(int)(tot/gH));
        while((long long)gW*gH>20000000LL){gW=max(1,gW*3/4);gH=max(1,gH*3/4);}
        cW=rx/gW; cH=ry/gH;
        
        size_t nc=(size_t)gW*gH;
        offset.assign(nc+1,0);
        vector<int> cof(n);
        for(int i=0;i<n;i++){
            int cx=min((int)((x[i]-mnX)/cW),gW-1);
            int cy=min((int)((y[i]-mnY)/cH),gH-1);
            cof[i]=cy*gW+cx;
            offset[cof[i]+1]++;
        }
        for(size_t i=0;i<nc;i++) offset[i+1]+=offset[i];
        data.resize(n);
        vector<int> tmp(nc,0);
        for(int i=0;i<n;i++){
            int c=cof[i];
            data[offset[c]+tmp[c]++]=i;
        }
    }

    int nearestUnvisited(int q, const vector<bool>& vis) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float best=1e30f; int bi=-1;
        float mc=min(cW,cH);
        for(int r=0;r<=gW+gH;r++){
            if(bi>=0){
                float md=(r>0?(r-1)*mc:0.0f);
                if(md*md>=best) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k]; if(vis[idx]) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(d<best){best=d;bi=idx;}
                }
            }
        }
        return bi;
    }

    void kNearest(int q,int K,vector<int>& out) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float mc=min(cW,cH);
        
        vector<pair<float,int>> cands;
        cands.reserve(K*4);
        float kthDist=1e30f;
        
        for(int r=0;r<=gW+gH;r++){
            if((int)cands.size()>=K){
                float md=(r>0?(r-1)*mc:0.0f);
                if(md>0&&md*md>=kthDist) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k]; if(idx==q) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(d<kthDist||(int)cands.size()<K){
                        cands.push_back({d,idx});
                    }
                }
            }
            // Trim if grown too large
            if((int)cands.size()>K*6){
                nth_element(cands.begin(),cands.begin()+K-1,cands.end());
                kthDist=cands[K-1].first;
                int sz=0;
                for(auto& p:cands) if(p.first<=kthDist&&sz<K*2) cands[sz++]=p;
                cands.resize(sz);
            }
        }
        
        if((int)cands.size()>K){
            partial_sort(cands.begin(),cands.begin()+K,cands.end());
            cands.resize(K);
        } else sort(cands.begin(),cands.end());
        
        out.clear(); out.reserve(cands.size());
        for(auto& p:cands) out.push_back(p.second);
    }
};

static vector<int> exactTSP(int n,const vector<float>&x,const vector<float>&y){
    int S=1<<n;
    vector<vector<float>> dp(S,vector<float>(n,1e30f));
    vector<vector<int>> pa(S,vector<int>(n,-1));
    dp[1][0]=0;
    for(int m=1;m<S;m++) for(int u=0;u<n;u++){
        if(!(m>>u&1)||dp[m][u]>1e29f) continue;
        for(int v=0;v<n;v++){
            if(m>>v&1) continue;
            float dx=x[u]-x[v],dy=y[u]-y[v];
            float nd=dp[m][u]+sqrtf(dx*dx+dy*dy);
            if(nd<dp[m|(1<<v)][v]){dp[m|(1<<v)][v]=nd;pa[m|(1<<v)][v]=u;}
        }
    }
    int F=S-1; float best=1e30f; int last=0;
    for(int v=1;v<n;v++){
        float dx=x[v]-x[0],dy=y[v]-y[0];
        float t=dp[F][v]+sqrtf(dx*dx+dy*dy);
        if(t<best){best=t;last=v;}
    }
    vector<int> t; int m=F,c=last;
    while(c>=0){t.push_back(c);int p=pa[m][c];m^=(1<<c);c=p;}
    reverse(t.begin(),t.end());
    return t;
}

//=== Morton code for space-filling curve ordering ===
static uint32_t expandBits(uint32_t v){
    v&=0xFFFF;
    v=(v|(v<<8))&0x00FF00FF;
    v=(v|(v<<4))&0x0F0F0F0F;
    v=(v|(v<<2))&0x33333333;
    v=(v|(v<<1))&0x55555555;
    return v;
}
static uint32_t mortonCode(uint32_t x,uint32_t y){
    return (expandBits(x)|(expandBits(y)<<1));
}

static vector<int> mortonSort(int n,const float* x,const float* y){
    float mnX=x[0],mnY=y[0],mxX=x[0],mxY=y[0];
    for(int i=1;i<n;i++){
        if(x[i]<mnX)mnX=x[i]; if(x[i]>mxX)mxX=x[i];
        if(y[i]<mnY)mnY=y[i]; if(y[i]>mxY)mxY=y[i];
    }
    float rx=mxX-mnX+1e-6f,ry=mxY-mnY+1e-6f;
    float scaleX=65535.0f/rx, scaleY=65535.0f/ry;
    
    vector<pair<uint32_t,int>> mc(n);
    for(int i=0;i<n;i++){
        uint32_t ix=(uint32_t)((x[i]-mnX)*scaleX);
        uint32_t iy=(uint32_t)((y[i]-mnY)*scaleY);
        mc[i]={mortonCode(ix,iy),i};
    }
    sort(mc.begin(),mc.end());
    
    vector<int> tour(n);
    for(int i=0;i<n;i++) tour[i]=mc[i].second;
    return tour;
}

//=== Grid index ===
struct Grid {
    float mnX,mnY,cW,cH;
    int gW,gH;
    const float *px,*py;
    vector<int> data, offset;

    void build(const float* x,const float* y,int n,float lf=4.0f){
        px=x; py=y;
        float mxX=x[0],mxY=y[0]; mnX=x[0]; mnY=y[0];
        for(int i=1;i<n;i++){
            if(x[i]<mnX)mnX=x[i]; if(x[i]>mxX)mxX=x[i];
            if(y[i]<mnY)mnY=y[i]; if(y[i]>mxY)mxY=y[i];
        }
        float rx=mxX-mnX+1e-6f,ry=mxY-mnY+1e-6f;
        double tot=max(1.0,(double)n/lf);
        gH=max(1,(int)sqrt(tot*ry/rx));
        gW=max(1,(int)(tot/gH));
        while((long long)gW*gH>20000000LL){gW=max(1,gW*4/5);gH=max(1,gH*4/5);}
        cW=rx/gW; cH=ry/gH;
        
        size_t nc=(size_t)gW*gH;
        offset.assign(nc+1,0);
        vector<int> cof(n);
        for(int i=0;i<n;i++){
            int cx=min((int)((x[i]-mnX)/cW),gW-1);
            int cy=min((int)((y[i]-mnY)/cH),gH-1);
            cof[i]=cy*gW+cx;
            offset[cof[i]+1]++;
        }
        for(size_t i=0;i<nc;i++) offset[i+1]+=offset[i];
        data.resize(n);
        vector<int> tmp(nc,0);
        for(int i=0;i<n;i++){int c=cof[i];data[offset[c]+tmp[c]++]=i;}
    }

    int nearestUnvisited(int q,const vector<bool>& vis) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float best=1e30f; int bi=-1;
        float mc=min(cW,cH);
        for(int r=0;r<=gW+gH;r++){
            if(bi>=0){
                float md=(r>1?(float)(r-1)*mc:0.0f);
                if(md*md>=best) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k]; if(vis[idx]) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(d<best){best=d;bi=idx;}
                }
            }
        }
        return bi;
    }

    void kNearest(int q,int K,vector<int>& out) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float mc=min(cW,cH);
        vector<pair<float,int>> cands;
        cands.reserve(K*4);
        float kthDist=1e30f;
        
        for(int r=0;r<=gW+gH;r++){
            if((int)cands.size()>=K&&r>1){
                float md=(float)(r-1)*mc;
                if(md*md>=kthDist) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=offset[c];k<offset[c+1];k++){
                    int idx=data[k]; if(idx==q) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if((int)cands.size()<K||d<kthDist)
                        cands.push_back({d,idx});
                }
            }
            if((int)cands.size()>K*8){
                nth_element(cands.begin(),cands.begin()+K,cands.end());
                cands.resize(K+1);
                kthDist=0; for(auto&p:cands)if(p.first>kthDist)kthDist=p.first;
            }
        }
        if((int)cands.size()>K){
            partial_sort(cands.begin(),cands.begin()+K,cands.end());
            cands.resize(K);
        } else sort(cands.begin(),cands.end());
        out.clear();
        for(auto&p:cands) out.push_back(p.second);
    }
};

static vector<int> greedyNN(int n,const float* x,const float* y){
    float lf=(n>5000000)?16.0f:(n>500000)?8.0f:4.0f;
    Grid g; g.build(x,y,n,lf);
    vector<bool> vis(n,false);
    vector<int> tour; tour.reserve(n);
    int cur=0; vis[0]=true; tour.push_back(0);
    for(int s=1;s<n;s++){
        int nxt=g.nearestUnvisited(cur,vis);
        if(nxt<0){for(int i=0;i<n;i++)if(!vis[i]){nxt=i;break;}}
        vis[nxt]=true; tour.push_back(nxt); cur=nxt;
        if((s&0xFFFFF)==0&&elapsed()>240.0){
            for(int i=0;i<n;i++)if(!vis[i])tour.push_back(i);
            return tour;
        }
    }
    return tour;
}

static void twoOptNN(vector<int>& tour,const float* x,const float* y,
                     double tLim,int K){
    int n=(int)tour.size(); if(n<4) return;
    vector<int> pos(n);
    for(int i=0;i<n;i++) pos[tour[i]]=i;
    auto D=[&](int a,int b)->float{
        float dx=x[a]-x[b],dy=y[a]-y[b]; return sqrtf(dx*dx+dy*dy);
    };
    double buildLim=elapsed()+(tLim-elapsed())*0.35;
    Grid g; g.build(x,y,n,4.0f);
    vector<int> nb((size_t)n*K,-1); vector<int> tmp;
    for(int i=0;i<n;i++){
        int k=(elapsed()<buildLim)?K:max(3,K/2);
        g.kNearest(i,k,tmp);
        size_t base=(size_t)i*K;
        for(int j=0;j<(int)tmp.size()&&j<K;j++) nb[base+j]=tmp[j];
    }
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int ii=0;ii<n&&elapsed()<tLim;ii++){
            int t1=tour[ii],t2=tour[(ii+1)%n];
            float d12=D(t1,t2);
            size_t base=(size_t)t1*K;
            for(int ki=0;ki<K;ki++){
                int t3=nb[base+ki]; if(t3<0) break;
                int jj=pos[t3],t4=tour[(jj+1)%n];
                if(t3==t2||t4==t1||t3==t1) continue;
                float gain=d12+D(t3,t4)-D(t1,t3)-D(t2,t4);
                if(gain>1e-6f){
                    int a,b;
                    if(ii<jj){a=ii+1;b=jj;}else{a=jj+1;b=ii;}
                    int l=a,r=b;
                    while(l<r){swap(tour[l],tour[r]);pos[tour[l]]=l;pos[tour[r]]=r;l++;r--;}
                    if(l==r)pos[tour[l]]=l;
                    t2=tour[(ii+1)%n]; d12=D(t1,t2); imp=true;
                }
            }
        }
    }
}

static void twoOptFull(vector<int>& tour,const float* x,const float* y,double tLim){
    int n=(int)tour.size();
    auto D=[&](int a,int b)->float{float dx=x[a]-x[b],dy=y[a]-y[b];return sqrtf(dx*dx+dy*dy);};
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int i=0;i<n-1&&elapsed()<tLim;i++)
            for(int j=i+2;j<n;j++){
                if(i==0&&j==n-1) continue;

                if(af<b4-1e-6f){reverse(tour.begin()+i+1,tour.begin()+j+1);imp=true;}
            }
    }
}

static vector<int> exactTSP(int n,const float* x,const float* y){
    int S=1<<n;
    vector<vector<float>> dp(S,vector<float>(n,1e30f));
    vector<vector<int>>   pa(S,vector<int>(n,-1));
    dp[1][0]=0;
    for(int m=1;m<S;m++) for(int u=0;u<n;u++){
        if(!(m>>u&1)||dp[m][u]>1e29f) continue;
        for(int v=0;v<n;v++){
            if(m>>v&1) continue;
            float dx=x[u]-x[v],dy=y[u]-y[v];
            float nd=dp[m][u]+sqrtf(dx*dx+dy*dy);
            int nm=m|(1<<v);
            if(nd<dp[nm][v]){dp[nm][v]=nd;pa[nm][v]=u;}
        }
    }
    int F=S-1; float best=1e30f; int last=0;
    for(int v=1;v<n;v++){
        float dx=x[v]-x[0],dy=y[v]-y[0];
        float t=dp[F][v]+sqrtf(dx*dx+dy*dy);
        if(t<best){best=t;last=v;}
    }
    vector<int> tour; int m=F,c=last;
    while(c>=0){tour.push_back(c);int p=pa[m][c];m^=(1<<c);c=p;}
    reverse(tour.begin(),tour.end());
    return tour;
}

static inline uint32_t mortonExpand(uint32_t x){
    x&=0xFFFF;
    x=(x|(x<<8))&0x00FF00FF;
    x=(x|(x<<4))&0x0F0F0F0F;
    x=(x|(x<<2))&0x33333333;
    x=(x|(x<<1))&0x55555555;
    return x;
}
static inline uint32_t morton(uint32_t x,uint32_t y){
    return mortonExpand(x)|(mortonExpand(y)<<1);
}

static vector<int> mortonTour(int n,const float* x,const float* y){
    float mnX=x[0],mnY=y[0],mxX=x[0],mxY=y[0];
    for(int i=1;i<n;i++){
        if(x[i]<mnX)mnX=x[i]; if(x[i]>mxX)mxX=x[i];
        if(y[i]<mnY)mnY=y[i]; if(y[i]>mxY)mxY=y[i];
    }
    float sx=65535.0f/(mxX-mnX+1e-6f),sy=65535.0f/(mxY-mnY+1e-6f);
    
    vector<pair<uint32_t,int>> mc(n);
    for(int i=0;i<n;i++){
        uint32_t ix=(uint32_t)((x[i]-mnX)*sx);
        uint32_t iy=(uint32_t)((y[i]-mnY)*sy);
        mc[i]={morton(ix,iy),i};
    }
    sort(mc.begin(),mc.end());
    
    vector<int> tour(n);
    for(int i=0;i<n;i++) tour[i]=mc[i].second;
    return tour;
}

struct Grid {
    float mnX,mnY,cW,cH;
    int gW,gH;
    const float *px,*py;
    vector<int> data, off; // CSR
    
    void build(const float* x,const float* y,int n,float lf=4.0f){
        px=x; py=y;
        float mxX=x[0],mxY=y[0]; mnX=x[0]; mnY=y[0];
        for(int i=1;i<n;i++){
            if(x[i]<mnX)mnX=x[i]; if(x[i]>mxX)mxX=x[i];
            if(y[i]<mnY)mnY=y[i]; if(y[i]>mxY)mxY=y[i];
        }
        float rx=mxX-mnX+1e-6f,ry=mxY-mnY+1e-6f;
        double tot=(double)n/lf;
        if(tot<1) tot=1;
        gH=max(1,(int)sqrt(tot*ry/rx));
        gW=max(1,(int)(tot/gH));
        // memory cap: ~20M cells
        while((long long)gW*gH>20000000LL){
            if(gW>gH) gW=gW*4/5; else gH=gH*4/5;
            gW=max(1,gW); gH=max(1,gH);
        }
        cW=rx/gW; cH=ry/gH;
        long long nc=(long long)gW*gH;
        off.assign((size_t)(nc+1),0);
        vector<int> cof(n);
        for(int i=0;i<n;i++){
            int cx=min((int)((x[i]-mnX)/cW),gW-1);
            int cy=min((int)((y[i]-mnY)/cH),gH-1);
            cof[i]=cy*gW+cx;
            off[(size_t)cof[i]+1]++;
        }
        for(size_t i=0;i<(size_t)nc;i++) off[i+1]+=off[i];
        data.resize(n);
        vector<int> tmp((size_t)nc,0);
        for(int i=0;i<n;i++){
            int c=cof[i];
            data[(size_t)off[c]+tmp[c]++]=i;
        }
    }
    
    int nearestUnvisited(int q,const vector<bool>& vis) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float best=1e30f; int bi=-1;
        float mc2=min(cW,cH);
        for(int r=0;r<=gW+gH;r++){
            if(bi>=0&&r>1){
                float md=(float)(r-1)*mc2;
                if(md*md>=best) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c2=gy*gW+gx;
                int s=(int)off[c2],e=(int)off[(size_t)c2+1];
                for(int k=s;k<e;k++){
                    int idx=data[k]; if(vis[idx]) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(d<best){best=d;bi=idx;}
                }
            }
        }
        return bi;
    }
    
    void kNearest(int q,int K,vector<int>& out) const {
        float qx=px[q],qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)),gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)),gH-1);
        float mc2=min(cW,cH);
        vector<pair<float,int>> cands;
        cands.reserve(K*4);
        float kthDist=1e30f; int found=0;
        for(int r=0;r<=gW+gH;r++){
            if(found>=K&&r>1){
                float md=(float)(r-1)*mc2;
                if(md*md>=kthDist) break;
            }
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1) continue;
                int c2=gy*gW+gx;
                int s=(int)off[c2],e=(int)off[(size_t)c2+1];
                for(int k=s;k<e;k++){
                    int idx=data[k]; if(idx==q) continue;
                    float dx=px[idx]-qx,dy=py[idx]-qy,d=dx*dx+dy*dy;
                    if(found<K||d<kthDist){cands.push_back({d,idx});found++;}
                }
            }
            if(found>K*8){
                nth_element(cands.begin(),cands.begin()+K-1,cands.end());
                kthDist=cands[K-1].first;
                // Keep only elements <= kthDist
                int sz=0;
                for(auto& p:cands) if(p.first<=kthDist) cands[sz++]=p;
                cands.resize(sz);
                found=sz;
            }
        }
        if((int)cands.size()>K){
            partial_sort(cands.begin(),cands.begin()+K,cands.end());
            cands.resize(K);
        } else sort(cands.begin(),cands.end());
        out.clear();
        for(auto& p:cands) out.push_back(p.second);
    }
};

static vector<int> greedyNN(int n,const float* x,const float* y){
    float lf=(n>5000000)?16.0f:(n>500000)?8.0f:4.0f;
    Grid g; g.build(x,y,n,lf);
    vector<bool> vis(n,false);
    vector<int> tour; tour.reserve(n);
    int cur=0; vis[0]=true; tour.push_back(0);
    for(int s=1;s<n;s++){
        int nxt=g.nearestUnvisited(cur,vis);
        if(nxt<0){for(int i=0;i<n;i++)if(!vis[i]){nxt=i;break;}}
        vis[nxt]=true; tour.push_back(nxt); cur=nxt;
        if((s&0xFFFFF)==0&&elapsed()>240.0){
            for(int i=0;i<n;i++)if(!vis[i])tour.push_back(i);
            return tour;
        }
    }
    return tour;
}

static void twoOptFull(vector<int>& tour,const float* x,const float* y,double tLim){
    int n=(int)tour.size();
    auto D=[&](int a,int b)->float{float dx=x[a]-x[b],dy=y[a]-y[b];return sqrtf(dx*dx+dy*dy);};
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int i=0;i<n-1&&elapsed()<tLim;i++)
            for(int j=i+2;j<n;j++){
                if(i==0&&j==n-1) continue;
                float b4=D(tour[i],tour[i+1])+D(tour[j],tour[(j+1)%n]);
                float af=D(tour[i],tour[j])+D(tour[i+1],tour[(j+1)%n]);
                if(af<b4-1e-6f){reverse(tour.begin()+i+1,tour.begin()+j+1);imp=true;}
            }
    }
}

static void twoOptNN(vector<int>& tour,const float* x,const float* y,
                     double tLim,int K){
    int n=(int)tour.size(); if(n<4) return;
    vector<int> pos(n);
    for(int i=0;i<n;i++) pos[tour[i]]=i;
    auto D=[&](int a,int b)->float{
        float dx=x[a]-x[b],dy=y[a]-y[b]; return sqrtf(dx*dx+dy*dy);
    };
    
    double buildLim=elapsed()+(tLim-elapsed())*0.4;
    Grid g; g.build(x,y,n,4.0f);
    vector<int> nb((size_t)n*K,-1);
    {
        vector<int> tmp;
        for(int i=0;i<n;i++){
            int k=(elapsed()<buildLim)?K:max(3,K/2);
            g.kNearest(i,k,tmp);
            size_t base=(size_t)i*K;
            for(int j=0;j<(int)tmp.size()&&j<K;j++) nb[base+j]=tmp[j];
        }
    }
    
    bool imp=true;
    while(imp&&elapsed()<tLim){
        imp=false;
        for(int ii=0;ii<n&&elapsed()<tLim;ii++){
            int t1=tour[ii],t2=tour[(ii+1)%n];
            float d12=D(t1,t2);
            size_t base=(size_t)t1*K;
            for(int ki=0;ki<K;ki++){
                int t3=nb[base+ki]; if(t3<0) break;
                int jj=pos[t3],t4=tour[(jj+1)%n];
                if(t3==t2||t4==t1||t3==t1) continue;
                float gain=d12+D(t3,t4)-D(t1,t3)-D(t2,t4);
                if(gain>1e-6f){
                    int a,b;
                    if(ii<jj){a=ii+1;b=jj;}
                    else{a=jj+1;b=ii;}
                    if(a<b){
                        int l=a,r=b;
                        while(l<r){
                            swap(tour[l],tour[r]);
                            pos[tour[l]]=l; pos[tour[r]]=r;
                            l++;r--;
                        }
                        if(l==r)pos[tour[l]]=l;
                    }
                    t2=tour[(ii+1)%n]; d12=D(t1,t2); imp=true;
                }
            }
        }
    }
}

static vector<int> exactTSP(int n,const float* x,const float* y){

static vector<int> mortonTour(int n,const float* x,const float* y){

        if(y[i]<mnY)mnY=y[i];if(y[i]>mxY)mxY=y[i];
    }
    float sx=65535.0f/(mxX-mnX+1e-6f),sy=65535.0f/(mxY-mnY+1e-6f);
    vector<pair<uint32_t,int>> mc(n);
    for(int i=0;i<n;i++)
        mc[i]={mcod((uint32_t)((x[i]-mnX)*sx),(uint32_t)((y[i]-mnY)*sy)),i};
    
    // Sort: radix for large n, std::sort for smaller
    if(n>1000000){
        vector<pair<uint32_t,int>> tmp(n);
        const int B=8,M=(1<<B)-1;
        vector<int> cnt(1<<B,0);
        for(int sh=0;sh<32;sh+=B){
            fill(cnt.begin(),cnt.end(),0);
            for(auto&p:mc)cnt[(p.first>>sh)&M]++;
            for(int i=1;i<(1<<B);i++)cnt[i]+=cnt[i-1];
            for(int i=n-1;i>=0;i--){int b=(mc[i].first>>sh)&M;tmp[--cnt[b]]=mc[i];}
            swap(mc,tmp);
        }
    } else {
        sort(mc.begin(),mc.end());
    }
    
    vector<int> tour(n);
    for(int i=0;i<n;i++) tour[i]=mc[i].second;
    return tour;
}

struct Grid {

    void build(const float* x,const float* y,int n,float lf=4.0f){

        for(int i=1;i<n;i++){
            if(x[i]<mnX)mnX=x[i];if(x[i]>mxX)mxX=x[i];
            if(y[i]<mnY)mnY=y[i];if(y[i]>mxY)mxY=y[i];
        }
        float rx=mxX-mnX+1e-6f,ry=mxY-mnY+1e-6f;
        double tot=max(1.0,(double)n/lf);
        gH=max(1,(int)sqrt(tot*(double)ry/rx));

        while((long long)gW*gH>20000000LL){gW=max(1,gW*4/5);gH=max(1,gH*4/5);}

            if(r>gW+gH)break;

                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1)continue;

        for(int r=0;;r++){
            if(found>=K&&r>1){float md=(float)(r-1)*mc2;if(md*md>=kd)break;}
            if(r>gW+gH)break;
            int x0=max(0,cx-r),x1=min(gW-1,cx+r);
            int y0=max(0,cy-r),y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0&&gx>x0&&gx<x1&&gy>y0&&gy<y1)continue;

        else sort(cands.begin(),cands.end());
        out.clear();for(auto&p:cands)out.push_back(p.second);
    }
};

static vector<int> exactTSP(int n, const float* x, const float* y){
    int S=1<<n;
    vector<vector<float>> dp(S, vector<float>(n, 1e30f));

            int c=cof[i];
            data[(size_t)off[c]+tmp[c]++]=i;
        }
    }
    
    int nearestUnvis(int q, const vector<bool>& vis) const {
        float qx=px[q], qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)), gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)), gH-1);
        float best=1e30f; int bi=-1;
        float mc2=min(cW,cH);
        for(int r=0;;r++){
            if(bi>=0 && r>1){
                float md=(float)(r-1)*mc2;
                if(md*md>=best) break;
            }
            if(r>gW+gH) break;
            int x0=max(0,cx-r), x1=min(gW-1,cx+r);
            int y0=max(0,cy-r), y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0 && gx>x0 && gx<x1 && gy>y0 && gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=(int)off[c]; k<(int)off[(size_t)c+1]; k++){
                    int idx=data[k];
                    if(vis[idx]) continue;
                    float dx=px[idx]-qx, dy=py[idx]-qy, d=dx*dx+dy*dy;
                    if(d<best){ best=d; bi=idx; }
                }
            }
        }
        return bi;
    }
    
    void kNN(int q, int K, vector<int>& out) const {
        float qx=px[q], qy=py[q];
        int cx=min(max(0,(int)((qx-mnX)/cW)), gW-1);
        int cy=min(max(0,(int)((qy-mnY)/cH)), gH-1);
        float mc2=min(cW,cH);
        vector<pair<float,int>> cands;
        cands.reserve(K*4);
        float kd=1e30f; int found=0;
        for(int r=0;;r++){
            if(found>=K && r>1){
                float md=(float)(r-1)*mc2;
                if(md*md>=kd) break;
            }
            if(r>gW+gH) break;
            int x0=max(0,cx-r), x1=min(gW-1,cx+r);
            int y0=max(0,cy-r), y1=min(gH-1,cy+r);
            for(int gy=y0;gy<=y1;gy++) for(int gx=x0;gx<=x1;gx++){
                if(r>0 && gx>x0 && gx<x1 && gy>y0 && gy<y1) continue;
                int c=gy*gW+gx;
                for(int k=(int)off[c]; k<(int)off[(size_t)c+1]; k++){
                    int idx=data[k]; if(idx==q) continue;
                    float dx=px[idx]-qx, dy=py[idx]-qy, d=dx*dx+dy*dy;
                    if(found<K || d<kd){ cands.push_back({d,idx}); found++; }
                }
            }
            if(found>K*8){
                nth_element(cands.begin(), cands.begin()+K-1, cands.end());
                kd=cands[K-1].first;
                int sz=0;
                for(auto& p:cands) if(p.first<=kd) cands[sz++]=p;
                cands.resize(sz); found=sz;
            }
        }
        if((int)cands.size()>K){
            partial_sort(cands.begin(), cands.begin()+K, cands.end());
            cands.resize(K);
        } else sort(cands.begin(), cands.end());
        out.clear();
        for(auto& p:cands) out.push_back(p.second);
    }
};

// ==================== Greedy nearest-neighbor ====================
static vector<int> greedyNN(int n, const float* x, const float* y){
    float lf=(n>5000000)?16.0f:(n>500000)?8.0f:4.0f;
    Grid g; g.build(x,y,n,lf);
    vector<bool> vis(n, false);
    vector<int> tour; tour.reserve(n);
    int cur=0; vis[0]=true; tour.push_back(0);
    for(int s=1;s<n;s++){
        int nxt=g.nearestUnvis(cur, vis);
        if(nxt<0){
            for(int i=0;i<n;i++) if(!vis[i]){ nxt=i; break; }
        }
        vis[nxt]=true; tour.push_back(nxt); cur=nxt;
        if((s&0xFFFFF)==0 && elapsed()>240.0){
            for(int i=0;i<n;i++) if(!vis[i]) tour.push_back(i);
            return tour;
        }
    }
    return tour;
}

// ==================== Full 2-opt (O(n^2)) ====================
static void twoOptFull(vector<int>& tour, const float* x, const float* y, double tL){
    int n=(int)tour.size();
    auto D=[&](int a, int b)->float{
        float dx=x[a]-x[b], dy=y[a]-y[b]; return sqrtf(dx*dx+dy*dy);
    };
    bool imp=true;
    while(imp && elapsed()<tL){
        imp=false;
        for(int i=0;i<n-1 && elapsed()<tL;i++)
            for(int j=i+2;j<n;j++){
                if(i==0 && j==n-1) continue;
                float b4=D(tour[i],tour[i+1])+D(tour[j],tour[(j+1)%n]);
                float af=D(tour[i],tour[j])+D(tour[i+1],tour[(j+1)%n]);
                if(af<b4-1e-6f){
                    reverse(tour.begin()+i+1, tour.begin()+j+1);
                    imp=true;
                }
            }
    }
}

// ==================== 2-opt with neighbor lists ====================
static void twoOptNN(vector<int>& tour, const float* x, const float* y, double tL, int K){
    int n=(int)tour.size(); if(n<4) return;
    vector<int> pos(n);
    for(int i=0;i<n;i++) pos[tour[i]]=i;
    auto D=[&](int a, int b)->float{
        float dx=x[a]-x[b], dy=y[a]-y[b]; return sqrtf(dx*dx+dy*dy);
    };
    
    double bLim=elapsed()+(tL-elapsed())*0.4;
    Grid g; g.build(x,y,n,4.0f);
    vector<int> nb((size_t)n*K, -1);
    {
        vector<int> tmp;
        for(int i=0;i<n;i++){
            int k=(elapsed()<bLim) ? K : max(3,K/2);
            g.kNN(i, k, tmp);
            size_t base=(size_t)i*K;
            for(int j=0;j<(int)tmp.size() && j<K;j++) nb[base+j]=tmp[j];
        }
    }
    
    bool imp=true;
    while(imp && elapsed()<tL){
        imp=false;
        for(int ii=0;ii<n && elapsed()<tL;ii++){
            int t1=tour[ii], t2=tour[(ii+1)%n];
            float d12=D(t1,t2);
            size_t base=(size_t)t1*K;
            for(int ki=0;ki<K;ki++){
                int t3=nb[base+ki]; if(t3<0) break;
                int jj=pos[t3];
                int t4=tour[(jj+1)%n];
                if(t3==t2 || t4==t1 || t3==t1) continue;
                float gain=d12+D(t3,t4)-D(t1,t3)-D(t2,t4);
                if(gain>1e-6f){
                    int a,b;
                    if(ii<jj){ a=ii+1; b=jj; }
                    else{ a=jj+1; b=ii; }
                    if(a<b){
                        int l=a, r=b;
                        while(l<r){
                            swap(tour[l],tour[r]);
                            pos[tour[l]]=l; pos[tour[r]]=r;
                            l++; r--;
                        }
                        if(l==r) pos[tour[l]]=l;
                    }
                    t2=tour[(ii+1)%n]; d12=D(t1,t2); imp=true;
                }
            }
        }
    }
}

}

static char obuf[OBSZ];
static int opos = 0;

static void oflush() {
    if (opos) { fwrite(obuf, 1, opos, stdout); opos = 0; }
}
static inline void ochar(char c) {
    if (opos >= OBSZ) oflush();
    obuf[opos++] = c;
}
static inline void ouint(unsigned v) {
    if (opos + 12 >= OBSZ) oflush();
    if (!v) { obuf[opos++] = '0'; return; }
    char t[10]; int l = 0;
    while (v) { t[l++] = '0' + v % 10; v /= 10; }
    for (int i = l - 1; i >= 0; i--) obuf[opos++] = t[i];
}

//=============================================================================
// Exact TSP via bitmask DP (n <= 20)
//=============================================================================
static vector<int> exactTSP(int n, const float* x, const float* y) {
    int S = 1 << n;
    vector<vector<float>> dp(S, vector<float>(n, 1e30f));
    vector<vector<int>>   pa(S, vector<int>(n, -1));
    dp[1][0] = 0.0f;
    for (int m = 1; m < S; m++) for (int u = 0; u < n; u++) {
        if (!(m >> u & 1) || dp[m][u] > 1e29f) continue;
        for (int v = 0; v < n; v++) {
            if (m >> v & 1) continue;
            float dx = x[u] - x[v], dy = y[u] - y[v];
            float nd = dp[m][u] + sqrtf(dx * dx + dy * dy);
            int nm = m | (1 << v);
            if (nd < dp[nm][v]) { dp[nm][v] = nd; pa[nm][v] = u; }
        }
    }
    int F = S - 1; float best = 1e30f; int last = 0;
    for (int v = 1; v < n; v++) {
        float dx = x[v] - x[0], dy = y[v] - y[0];
        float t = dp[F][v] + sqrtf(dx * dx + dy * dy);
        if (t < best) { best = t; last = v; }
    }
    vector<int> t; int m = F, c = last;
    while (c >= 0) { t.push_back(c); int p = pa[m][c]; m ^= (1 << c); c = p; }
    reverse(t.begin(), t.end());
    return t;
}

//=============================================================================
// Morton curve ordering for large N
//=============================================================================
static inline uint32_t mexp(uint32_t v) {
    v &= 0xFFFF;
    v = (v | (v << 8))  & 0x00FF00FF;
    v = (v | (v << 4))  & 0x0F0F0F0F;
    v = (v | (v << 2))  & 0x33333333;
    return  (v | (v << 1)) & 0x55555555;
}
static inline uint32_t mcod(uint32_t a, uint32_t b) {
    return mexp(a) | (mexp(b) << 1);
}

static vector<int> mortonTour(int n, const float* x, const float* y) {
    float mnX = x[0], mnY = y[0], mxX = x[0], mxY = y[0];
    for (int i = 1; i < n; i++) {
        if (x[i] < mnX) mnX = x[i]; if (x[i] > mxX) mxX = x[i];
        if (y[i] < mnY) mnY = y[i]; if (y[i] > mxY) mxY = y[i];
    }
    float sx = 65535.0f / (mxX - mnX + 1e-6f);
    float sy = 65535.0f / (mxY - mnY + 1e-6f);

    vector<pair<uint32_t, int>> mc(n);
    for (int i = 0; i < n; i++) {
        mc[i] = { mcod((uint32_t)((x[i] - mnX) * sx), (uint32_t)((y[i] - mnY) * sy)), i };
    }

    if (n > 1000000) {
        // Radix sort (4 passes, 8-bit radix)
        vector<pair<uint32_t, int>> tmp(n);
        const int B = 8, MSK = (1 << B) - 1;
        vector<int> cnt(1 << B);
        for (int sh = 0; sh < 32; sh += B) {
            fill(cnt.begin(), cnt.end(), 0);
            for (auto& p : mc) cnt[(p.first >> sh) & MSK]++;
            for (int i = 1; i < (1 << B); i++) cnt[i] += cnt[i - 1];
            for (int i = n - 1; i >= 0; i--) {
                int b = (mc[i].first >> sh) & MSK;
                tmp[--cnt[b]] = mc[i];
            }
            swap(mc, tmp);
        }
    } else {
        sort(mc.begin(), mc.end());
    }

    vector<int> tour(n);
    for (int i = 0; i < n; i++) tour[i] = mc[i].second;
    return tour;
}

// Grid-based spatial index (flat CSR storage)
//=============================================================================
struct Grid {
    float mnX, mnY, cW, cH;
    int gW, gH;
    const float *px, *py;
    vector<int> data, off;

    void build(const float* x, const float* y, int n, float lf = 4.0f) {
        px = x; py = y;
        float mxX = x[0], mxY = y[0]; mnX = x[0]; mnY = y[0];
        for (int i = 1; i < n; i++) {
            if (x[i] < mnX) mnX = x[i]; if (x[i] > mxX) mxX = x[i];
            if (y[i] < mnY) mnY = y[i]; if (y[i] > mxY) mxY = y[i];
        }
        float rx = mxX - mnX + 1e-6f, ry = mxY - mnY + 1e-6f;
        double tot = max(1.0, (double)n / lf);
        gH = max(1, (int)sqrt(tot * (double)ry / rx));
        gW = max(1, (int)(tot / gH));
        // Cap at 20M cells to limit memory
        while ((long long)gW * gH > 20000000LL) {
            gW = max(1, gW * 4 / 5);
            gH = max(1, gH * 4 / 5);
        }
        cW = rx / gW; cH = ry / gH;

        size_t nc = (size_t)gW * gH;
        off.assign(nc + 1, 0);
        vector<int> cof(n);
        for (int i = 0; i < n; i++) {
            int cx = min((int)((x[i] - mnX) / cW), gW - 1);
            int cy = min((int)((y[i] - mnY) / cH), gH - 1);
            cof[i] = cy * gW + cx;
            off[(size_t)cof[i] + 1]++;
        }
        for (size_t i = 0; i < nc; i++) off[i + 1] += off[i];
        data.resize(n);
        vector<int> tmp(nc, 0);
        for (int i = 0; i < n; i++) {
            int c = cof[i];
            data[(size_t)off[c] + tmp[c]++] = i;
        }
    }

    int nearestUnvis(int q, const vector<bool>& vis) const {
        float qx = px[q], qy = py[q];
        int cx = min(max(0, (int)((qx - mnX) / cW)), gW - 1);
        int cy = min(max(0, (int)((qy - mnY) / cH)), gH - 1);
        float best = 1e30f; int bi = -1;
        float mc2 = min(cW, cH);
        for (int r = 0; ; r++) {
            if (bi >= 0 && r > 1) {
                float md = (float)(r - 1) * mc2;
                if (md * md >= best) break;
            }
            if (r > gW + gH) break;
            int x0 = max(0, cx - r), x1 = min(gW - 1, cx + r);
            int y0 = max(0, cy - r), y1 = min(gH - 1, cy + r);
            for (int gy = y0; gy <= y1; gy++) for (int gx = x0; gx <= x1; gx++) {
                if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                int c = gy * gW + gx;
                for (int k = (int)off[c]; k < (int)off[(size_t)c + 1]; k++) {
                    int idx = data[k];
                    if (vis[idx]) continue;
                    float dx = px[idx] - qx, dy = py[idx] - qy, d = dx * dx + dy * dy;
                    if (d < best) { best = d; bi = idx; }
                }
            }
        }
        return bi;
    }

    void kNN(int q, int K, vector<int>& out) const {
        float qx = px[q], qy = py[q];
        int cx = min(max(0, (int)((qx - mnX) / cW)), gW - 1);
        int cy = min(max(0, (int)((qy - mnY) / cH)), gH - 1);
        float mc2 = min(cW, cH);
        vector<pair<float, int>> cands;
        cands.reserve(K * 4);
        float kd = 1e30f; int found = 0;
        for (int r = 0; ; r++) {
            if (found >= K && r > 1) {
                float md = (float)(r - 1) * mc2;
                if (md * md >= kd) break;
            }
            if (r > gW + gH) break;
            int x0 = max(0, cx - r), x1 = min(gW - 1, cx + r);
            int y0 = max(0, cy - r), y1 = min(gH - 1, cy + r);
            for (int gy = y0; gy <= y1; gy++) for (int gx = x0; gx <= x1; gx++) {
                if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                int c = gy * gW + gx;
                for (int k = (int)off[c]; k < (int)off[(size_t)c + 1]; k++) {
                    int idx = data[k]; if (idx == q) continue;
                    float dx = px[idx] - qx, dy = py[idx] - qy, d = dx * dx + dy * dy;
                    if (found < K || d < kd) { cands.push_back({ d, idx }); found++; }
                }
            }
            if (found > K * 8) {
                nth_element(cands.begin(), cands.begin() + K - 1, cands.end());
                kd = cands[K - 1].first;
                int sz = 0;
                for (auto& p : cands) if (p.first <= kd && sz < K * 2) cands[sz++] = p;
                cands.resize(sz); found = sz;
            }
        }
        if ((int)cands.size() > K) {
            partial_sort(cands.begin(), cands.begin() + K, cands.end());
            cands.resize(K);
        } else sort(cands.begin(), cands.end());
        out.clear();
        for (auto& p : cands) out.push_back(p.second);
    }
};

//=============================================================================
// Greedy nearest-neighbor tour
//=============================================================================
static vector<int> greedyNN(int n, const float* x, const float* y) {
    float lf = (n > 5000000) ? 16.0f : (n > 500000) ? 8.0f : 4.0f;
    Grid g; g.build(x, y, n, lf);
    vector<bool> vis(n, false);
    vector<int> tour; tour.reserve(n);
    int cur = 0; vis[0] = true; tour.push_back(0);
    for (int s = 1; s < n; s++) {
        int nxt = g.nearestUnvis(cur, vis);
        if (nxt < 0) {
            for (int i = 0; i < n; i++) if (!vis[i]) { nxt = i; break; }
        }
        vis[nxt] = true; tour.push_back(nxt); cur = nxt;
        if ((s & 0xFFFFF) == 0 && elapsed() > 240.0) {
            for (int i = 0; i < n; i++) if (!vis[i]) tour.push_back(i);
            return tour;
        }
    }
    return tour;
}

//=============================================================================
// Full O(n^2) 2-opt (for small n)
//=============================================================================
static void twoOptFull(vector<int>& tour, const float* x, const float* y, double tL) {
    int n = (int)tour.size();
    auto D = [&](int a, int b) -> float {
        float dx = x[a] - x[b], dy = y[a] - y[b]; return sqrtf(dx * dx + dy * dy);
    };
    bool imp = true;
    while (imp && elapsed() < tL) {
        imp = false;
        for (int i = 0; i < n - 1 && elapsed() < tL; i++)
            for (int j = i + 2; j < n; j++) {
                if (i == 0 && j == n - 1) continue;
                float b4 = D(tour[i], tour[i + 1]) + D(tour[j], tour[(j + 1) % n]);
                float af = D(tour[i], tour[j]) + D(tour[i + 1], tour[(j + 1) % n]);
                if (af < b4 - 1e-6f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    imp = true;
                }
            }
    }
}

//=============================================================================
// 2-opt with nearest-neighbor candidate lists
//=============================================================================
static void twoOptNN(vector<int>& tour, const float* x, const float* y, double tL, int K) {
    int n = (int)tour.size(); if (n < 4) return;
    vector<int> pos(n);
    for (int i = 0; i < n; i++) pos[tour[i]] = i;
    auto D = [&](int a, int b) -> float {
        float dx = x[a] - x[b], dy = y[a] - y[b]; return sqrtf(dx * dx + dy * dy);
    };

    double bLim = elapsed() + (tL - elapsed()) * 0.4;
    Grid g; g.build(x, y, n, 4.0f);
    vector<int> nb((size_t)n * K, -1);
    {
        vector<int> tmp;
        for (int i = 0; i < n; i++) {
            int k = (elapsed() < bLim) ? K : max(3, K / 2);
            g.kNN(i, k, tmp);
            size_t base = (size_t)i * K;
            for (int j = 0; j < (int)tmp.size() && j < K; j++) nb[base + j] = tmp[j];
        }
    }

    bool imp = true;
    while (imp && elapsed() < tL) {
        imp = false;
        for (int ii = 0; ii < n && elapsed() < tL; ii++) {
            int t1 = tour[ii], t2 = tour[(ii + 1) % n];
            float d12 = D(t1, t2);
            size_t base = (size_t)t1 * K;
            for (int ki = 0; ki < K; ki++) {
                int t3 = nb[base + ki]; if (t3 < 0) break;
                int jj = pos[t3];
                int t4 = tour[(jj + 1) % n];
                if (t3 == t2 || t4 == t1 || t3 == t1) continue;
                float gain = d12 + D(t3, t4) - D(t1, t3) - D(t2, t4);
                if (gain > 1e-6f) {
                    int a, b;
                    if (ii < jj) { a = ii + 1; b = jj; }
                    else { a = jj + 1; b = ii; }
                    if (a < b) {
                        int l = a, r = b;
                        while (l < r) {
                            swap(tour[l], tour[r]);
                            pos[tour[l]] = l; pos[tour[r]] = r;
                            l++; r--;
                        }
                        if (l == r) pos[tour[l]] = l;
                    }
                    t2 = tour[(ii + 1) % n]; d12 = D(t1, t2); imp = true;
                }
            }
        }
    }
}

                if (r>0 && gx>x0 && gx<x1 && gy>y0 && gy<y1) continue;
                int c=gy*gW+gx;
                for (int k=(int)off[c];k<(int)off[(size_t)c+1];k++) {
                    int idx=data[k]; if (idx==q) continue;
                    float dx=px[idx]-qx, dy=py[idx]-qy, d=dx*dx+dy*dy;
                    if (found<K || d<kd) { cands.push_back({d,idx}); found++; }
                }
            }
            if (found>K*8) {
                nth_element(cands.begin(),cands.begin()+K-1,cands.end());
                kd=cands[K-1].first;
                int sz=0; for (auto& p:cands) if (p.first<=kd && sz<K*2) cands[sz++]=p;
                cands.resize(sz); found=sz;
            }
        }
        if ((int)cands.size()>K) { partial_sort(cands.begin(),cands.begin()+K,cands.end()); cands.resize(K); }
        else sort(cands.begin(),cands.end());
        out.clear(); for (auto& p:cands) out.push_back(p.second);
    }
};

// ─────────────────────────────────────────────────────────────────────────────
// Greedy nearest-neighbor tour
// ─────────────────────────────────────────────────────────────────────────────
static vector<int> greedyNN(int n, const float* x, const float* y) {
    float lf=(n>5000000)?16.0f:(n>500000)?8.0f:4.0f;
    Grid g; g.build(x,y,n,lf);
    vector<bool> vis(n,false); vector<int> tour; tour.reserve(n);
    int cur=0; vis[0]=true; tour.push_back(0);
    for (int s=1;s<n;s++) {
        int nxt=g.nearestUnvis(cur,vis);
        if (nxt<0) for (int i=0;i<n;i++) if (!vis[i]) { nxt=i; break; }
        vis[nxt]=true; tour.push_back(nxt); cur=nxt;
        if ((s&0xFFFFF)==0 && elapsed()>240.0) {
            for (int i=0;i<n;i++) if (!vis[i]) tour.push_back(i); return tour;
        }
    }
    return tour;
}

// ─────────────────────────────────────────────────────────────────────────────
// Full O(n²) 2-opt  (for small n)
// ─────────────────────────────────────────────────────────────────────────────
static void twoOptFull(vector<int>& tour, const float* x, const float* y, double tL) {
    int n=(int)tour.size();
    auto D=[&](int a,int b){ float dx=x[a]-x[b],dy=y[a]-y[b]; return sqrtf(dx*dx+dy*dy); };
    bool imp=true;
    while (imp && elapsed()<tL) {
        imp=false;
        for (int i=0;i<n-1 && elapsed()<tL;i++)
            for (int j=i+2;j<n;j++) {
                if (i==0 && j==n-1) continue;
                float b4=D(tour[i],tour[i+1])+D(tour[j],tour[(j+1)%n]);
                float af=D(tour[i],tour[j])+D(tour[i+1],tour[(j+1)%n]);
                if (af<b4-1e-6f) { reverse(tour.begin()+i+1,tour.begin()+j+1); imp=true; }
            }
    }
}

// ─────────────────────────────────────────────────────────────────────────────
// 2-opt with nearest-neighbour candidate lists  (for medium/large n)
// ─────────────────────────────────────────────────────────────────────────────
static void twoOptNN(vector<int>& tour, const float* x, const float* y, double tL, int K) {
    int n=(int)tour.size(); if (n<4) return;
    vector<int> pos(n); for (int i=0;i<n;i++) pos[tour[i]]=i;
    auto D=[&](int a,int b){ float dx=x[a]-x[b],dy=y[a]-y[b]; return sqrtf(dx*dx+dy*dy); };

    double bLim=elapsed()+(tL-elapsed())*0.4;
    Grid g; g.build(x,y,n,4.0f);
    vector<int> nb((size_t)n*K,-1);
    {
        vector<int> tmp;
        for (int i=0;i<n;i++) {
            int k=(elapsed()<bLim)?K:max(3,K/2);
            g.kNN(i,k,tmp);
            size_t base=(size_t)i*K;
            for (int j=0;j<(int)tmp.size()&&j<K;j++) nb[base+j]=tmp[j];
        }
    }

    bool imp=true;
    while (imp && elapsed()<tL) {
        imp=false;
        for (int ii=0;ii<n && elapsed()<tL;ii++) {
            int t1=tour[ii], t2=tour[(ii+1)%n]; float d12=D(t1,t2);
            size_t base=(size_t)t1*K;
            for (int ki=0;ki<K;ki++) {
                int t3=nb[base+ki]; if (t3<0) break;
                int jj=pos[t3], t4=tour[(jj+1)%n];
                if (t3==t2||t4==t1||t3==t1) continue;
                float gain=d12+D(t3,t4)-D(t1,t3)-D(t2,t4);
                if (gain>1e-6f) {
                    int a,b;
                    if (ii<jj) { a=ii+1; b=jj; } else { a=jj+1; b=ii; }
                    if (a<b) {
                        int l=a,r=b;
                        while (l<r) { swap(tour[l],tour[r]); pos[tour[l]]=l; pos[tour[r]]=r; l++;r--; }
                        if (l==r) pos[tour[l]]=l;
                    }
                    t2=tour[(ii+1)%n]; d12=D(t1,t2); imp=true;
                }
            }
        }
    }
}

// ─────────────────────────────────────────────────────────────────────────────
// main
// ─────────────────────────────────────────────────────────────────────────────
int main() {
    g_start=steady_clock::now();
    int n=readInt();
    vector<float> x(n),y(n);
    for (int i=0;i<n;i++) { x[i]=readFloat(); y[i]=readFloat(); }

    const double TL=270.0;
    vector<int> tour;

    if (n==1) { ouint(0); ochar('\n'); oflush(); return 0; }
    if (n==2) { ouint(0); ochar(' '); ouint(1); ochar('\n'); oflush(); return 0; }
    if (n==3) { ouint(0); ochar(' '); ouint(1); ochar(' '); ouint(2); ochar('\n'); oflush(); return 0; }

    if (n<=20) {
        tour=exactTSP(n,x.data(),y.data());
    } else if (n>5000000) {
        tour=mortonTour(n,x.data(),y.data());
    } else {
        tour=greedyNN(n,x.data(),y.data());
        double rem=TL-elapsed();
        if (rem>2.0) {
            if (n<=3000) twoOptFull(tour,x.data(),y.data(),elapsed()+rem*0.95);
            else {
                int K=(n<=50000)?15:(n<=500000)?12:(n<=2000000)?7:5;
                twoOptNN(tour,x.data(),y.data(),elapsed()+rem*0.88,K);
            }
        }
    }

    // Rotate so city 0 is first
    { int zp=0; for (int i=0;i<n;i++) if (tour[i]==0) { zp=i; break; } if (zp) rotate(tour.begin(),tour.begin()+zp,tour.end()); }

    for (int i=0;i<n;i++) { ouint((unsigned)tour[i]); ochar(i<n-1?' ':'\n'); }
    oflush(); return 0;
}
