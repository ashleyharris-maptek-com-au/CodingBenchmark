#include <cstdio>
#include <vector>
#include <cmath>
#include <algorithm>
#include <chrono>

using namespace std;
using namespace std::chrono;

static steady_clock::time_point g_start = steady_clock::now();
static inline double elapsed() {
    return duration<double>(steady_clock::now() - g_start).count();
}

static int N;
static vector<float> X, Y;

static inline double dist2(int a, int b) {
    double dx = X[a]-X[b], dy = Y[a]-Y[b];
    return dx*dx+dy*dy;
}
static inline double dist(int a, int b) { return sqrt(dist2(a,b)); }

// ── Held-Karp exact TSP (N <= 20) ────────────────────────────────────────────
static vector<int> held_karp() {
    int n = N, states = 1<<n;
    vector<float> dp((long long)states*n, 1e30f);
    vector<int>   par((long long)states*n, -1);
    dp[(long long)1*n+0] = 0.0f;
    for (int mask = 1; mask < states; mask++) {
        if (!(mask&1)) continue;
        for (int u = 0; u < n; u++) {
            if (!(mask&(1<<u))) continue;
            float cur = dp[(long long)mask*n+u];
            if (cur > 1e29f) continue;
            for (int v = 0; v < n; v++) {
                if (mask&(1<<v)) continue;
                int nm = mask|(1<<v);
                float nd = cur+(float)dist(u,v);
                long long idx = (long long)nm*n+v;
                if (nd < dp[idx]) { dp[idx]=nd; par[idx]=u; }
            }
        }
    }
    int full = states-1, last = -1;
    float best = 1e30f;
    for (int u = 1; u < n; u++) {
        float d = dp[(long long)full*n+u]+(float)dist(u,0);
        if (d < best) { best=d; last=u; }
    }
    vector<int> path;
    for (int mask=full, cur=last; cur!=0; ) {
        path.push_back(cur);
        int prev = par[(long long)mask*n+cur];
        mask ^= (1<<cur);
        cur = prev;
    }
    path.push_back(0);
    reverse(path.begin(), path.end());
    return path;
}

// ── Grid spatial index (CSR format) ──────────────────────────────────────────
struct Grid {
    float mnx, mny, cs;
    int cols, rows;
    vector<int> S, D;

    void build() {
        float mxx=X[0], mxy=Y[0]; mnx=X[0]; mny=Y[0];
        for (int i=1;i<N;i++) {
            if(X[i]<mnx) mnx=X[i]; if(Y[i]<mny) mny=Y[i];
            if(X[i]>mxx) mxx=X[i]; if(Y[i]>mxy) mxy=Y[i];
        }
        float w=mxx-mnx, h=mxy-mny, area=w*h;
        cs = (area>0.f) ? sqrtf(area/N)*2.f : 1.f;

        auto dims=[&](){
            cols=(int)((mxx-mnx)/cs)+2;
            rows=(int)((mxy-mny)/cs)+2;
        };
        dims();
        while((long long)cols*rows > 10000000LL){ cs*=1.5f; dims(); }

        long long nc=(long long)cols*rows;
        vector<int> cnt(nc,0);
        for(int i=0;i<N;i++){
            int c=min((int)((X[i]-mnx)/cs),cols-1);
            int r=min((int)((Y[i]-mny)/cs),rows-1);
            cnt[(long long)r*cols+c]++;
        }
        S.resize(nc+1,0);
        for(long long i=0;i<nc;i++) S[i+1]=S[i]+cnt[i];
        D.resize(N);
        fill(cnt.begin(),cnt.end(),0);
        for(int i=0;i<N;i++){
            int c=min((int)((X[i]-mnx)/cs),cols-1);
            int r=min((int)((Y[i]-mny)/cs),rows-1);
            long long cell=(long long)r*cols+c;
            D[S[cell]+cnt[cell]++]=i;
        }
    }

    int nearest(int q, const vector<bool>& vis) const {
        float qx=X[q], qy=Y[q];
        int qc=min((int)((qx-mnx)/cs),cols-1);
        int qr=min((int)((qy-mny)/cs),rows-1);
        int best=-1; double bd=1e30;
        for(int rad=0;;rad++){
            double mp=(rad>1)?(double)(rad-1)*cs:0.0;
            if(best>=0 && mp*mp>=bd) break;
            int r0=max(0,qr-rad), r1=min(rows-1,qr+rad);
            int c0=max(0,qc-rad), c1=min(cols-1,qc+rad);
            for(int r=r0;r<=r1;r++) for(int c=c0;c<=c1;c++){
                if(rad>0 && r>r0 && r<r1 && c>c0 && c<c1) continue;
                long long cell=(long long)r*cols+c;
                for(int k=S[cell];k<S[cell+1];k++){
                    int idx=D[k];
                    if(idx==q||vis[idx]) continue;
                    double d=dist2(q,idx);
                    if(d<bd){bd=d;best=idx;}
                }
            }
            if(r0==0&&c0==0&&r1==rows-1&&c1==cols-1) break;
        }
        return best;
    }
} g_grid;

// ── Nearest-neighbor heuristic ───────────────────────────────────────────────
static vector<int> nn_tour() {
    g_grid.build();
    vector<bool> vis(N,false);
    vector<int> tour; tour.reserve(N);
    int cur=0; vis[0]=true; tour.push_back(0);
    for(int s=1;s<N;s++){
        int nxt=g_grid.nearest(cur,vis);
        if(nxt<0){ for(int i=0;i<N;i++) if(!vis[i]){nxt=i;break;} }
        vis[nxt]=true; tour.push_back(nxt); cur=nxt;
    }
    return tour;
}

// ── 2-opt improvement ────────────────────────────────────────────────────────
static void two_opt(vector<int>& t, double tlim) {
    int chk = max(1, N/200);
    bool imp=true;
    while(imp && elapsed()<tlim){
        imp=false;
        for(int i=0;i<N-1;i++){
            if(i%chk==0 && elapsed()>=tlim) break;
            int a=t[i], b=t[i+1];
            double dab=dist(a,b);
            for(int j=i+2;j<N;j++){
                if(i==0&&j==N-1) continue;
                int c=t[j], d=t[(j+1)%N];
                double gain=dab+dist(c,d)-dist(a,c)-dist(b,d);
                if(gain>1e-10){
                    reverse(t.begin()+i+1, t.begin()+j+1);
                    b=t[i+1]; dab=dist(a,b); imp=true;
                }
            }
        }
    }
}

// ── Or-opt: relocate single cities ───────────────────────────────────────────
static void or_opt(vector<int>& t, double tlim) {
    bool imp=true;
    int chk = max(1, N/100);
    while(imp && elapsed()<tlim){
        imp=false;
        for(int i=0;i<N && elapsed()<tlim;i++){
            if(i%chk==0 && elapsed()>=tlim) break;
            int prev=(i-1+N)%N, nxt=(i+1)%N;
            int A=t[prev], B=t[i], C=t[nxt];
            double remove_gain = dist(A,B)+dist(B,C)-dist(A,C);
            if(remove_gain < 1e-10) continue;
            int best_j=-1; double best_ins=1e30;
            for(int j=0;j<N;j++){
                if(j==prev||j==i) continue;
                int jn=(j+1)%N;
                if(jn==i) continue;
                int P=t[j], Q=t[jn];
                double ins=dist(P,B)+dist(B,Q)-dist(P,Q);
                if(ins < best_ins){ best_ins=ins; best_j=j; }
            }
            if(best_j>=0 && best_ins < remove_gain - 1e-10){
                int city = t[i];
                t.erase(t.begin()+i);
                int ins_pos = (best_j < i) ? best_j+1 : best_j;
                t.insert(t.begin()+ins_pos, city);
                imp=true;
                i = max(0, i-1);
            }
        }
    }
}

// ── Fast buffered output ─────────────────────────────────────────────────────
static char obuf[1<<22];
static int opos=0;
static void oflush(){ fwrite(obuf,1,opos,stdout); opos=0; }
static void write_int(int v){
    if(opos+16>(int)sizeof(obuf)) oflush();
    if(v==0){obuf[opos++]='0';return;}
    char tmp[12]; int len=0;
    while(v){tmp[len++]='0'+v%10;v/=10;}
    for(int i=len-1;i>=0;i--) obuf[opos++]=tmp[i];
}

// ── main ─────────────────────────────────────────────────────────────────────
int main(){
    scanf("%d",&N);
    X.resize(N); Y.resize(N);
    for(int i=0;i<N;i++) scanf("%f %f",&X[i],&Y[i]);

    if(N==1){puts("0");return 0;}
    if(N==2){puts("0 1");return 0;}

    vector<int> tour;
    const double TLIM = 280.0;

    if(N<=20){
        tour = held_karp();
    } else {
        tour = nn_tour();
        auto it = find(tour.begin(), tour.end(), 0);
        if(it != tour.begin()) rotate(tour.begin(), it, tour.end());

        if(N <= 8000){
            two_opt(tour, TLIM - 2.0);
            or_opt(tour, TLIM - 1.0);
        } else if(N <= 100000){
            two_opt(tour, TLIM);
        }
    }

    {
        auto it = find(tour.begin(), tour.end(), 0);
        if(it != tour.begin()) rotate(tour.begin(), it, tour.end());
    }

    for(int i=0;i<N;i++){
        if(i){ if(opos>=(int)sizeof(obuf)-20) oflush(); obuf[opos++]=' '; }
        write_int(tour[i]);
    }
    obuf[opos++]='\n';
    oflush();
    return 0;
}
