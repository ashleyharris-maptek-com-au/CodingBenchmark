use std::f64::consts::PI;
use std::io::{self, Read, Write};

type Pt = (f64, f64);

#[inline]
fn cross(o: Pt, a: Pt, b: Pt) -> f64 {
    (a.0 - o.0) * (b.1 - o.1) - (a.1 - o.1) * (b.0 - o.0)
}

fn on_seg(p: Pt, q: Pt, r: Pt) -> bool {
    q.0 >= p.0.min(r.0) - 1e-12
        && q.0 <= p.0.max(r.0) + 1e-12
        && q.1 >= p.1.min(r.1) - 1e-12
        && q.1 <= p.1.max(r.1) + 1e-12
}

fn seg_cross(p: Pt, q: Pt, r: Pt, s: Pt) -> bool {
    let d1 = cross(r, s, p);
    let d2 = cross(r, s, q);
    let d3 = cross(p, q, r);
    let d4 = cross(p, q, s);
    if (d1 > 0.0) != (d2 > 0.0) && (d3 > 0.0) != (d4 > 0.0) {
        return true;
    }
    (d1.abs() < 1e-12 && on_seg(r, p, s))
        || (d2.abs() < 1e-12 && on_seg(r, q, s))
        || (d3.abs() < 1e-12 && on_seg(p, r, q))
        || (d4.abs() < 1e-12 && on_seg(p, s, q))
}

fn pip(px: f64, py: f64, poly: &[Pt]) -> bool {
    let n = poly.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let (xi, yi) = poly[i];
        let (xj, yj) = poly[j];
        if (yi > py) != (yj > py) && px < (xj - xi) * (py - yi) / (yj - yi) + xi {
            inside = !inside;
        }
        j = i;
    }
    inside
}

fn bbox(pts: &[Pt]) -> (f64, f64, f64, f64) {
    pts.iter().fold(
        (f64::INFINITY, f64::INFINITY, f64::NEG_INFINITY, f64::NEG_INFINITY),
        |(x0, y0, x1, y1), &(x, y)| (x0.min(x), y0.min(y), x1.max(x), y1.max(y)),
    )
}

fn poly_area(p: &[Pt]) -> f64 {
    let n = p.len();
    (0..n)
        .map(|i| {
            let j = (i + 1) % n;
            p[i].0 * p[j].1 - p[j].0 * p[i].1
        })
        .sum::<f64>()
        .abs()
        * 0.5
}

fn rotate_pts(p: &[Pt], a: f64) -> Vec<Pt> {
    let (c, s) = (a.cos(), a.sin());
    p.iter().map(|&(x, y)| (x * c - y * s, x * s + y * c)).collect()
}

fn fits_in(piece: &[Pt], stock: &[Pt]) -> bool {
    for &(x, y) in piece {
        if !pip(x, y, stock) {
            return false;
        }
    }
    let np = piece.len();
    let ns = stock.len();
    for i in 0..np {
        for k in 0..ns {
            if seg_cross(piece[i], piece[(i + 1) % np], stock[k], stock[(k + 1) % ns]) {
                return false;
            }
        }
    }
    true
}

fn overlaps(a: &[Pt], ab: (f64, f64, f64, f64), b: &[Pt], bb: (f64, f64, f64, f64)) -> bool {
    if ab.2 < bb.0 || bb.2 < ab.0 || ab.3 < bb.1 || bb.3 < ab.1 {
        return false;
    }
    if a.iter().any(|&(x, y)| pip(x, y, b)) {
        return true;
    }
    if b.iter().any(|&(x, y)| pip(x, y, a)) {
        return true;
    }
    let na = a.len();
    let nb = b.len();
    for i in 0..na {
        for k in 0..nb {
            if seg_cross(a[i], a[(i + 1) % na], b[k], b[(k + 1) % nb]) {
                return true;
            }
        }
    }
    false
}

struct Bin {
    polys: Vec<Vec<Pt>>,
    bbs: Vec<(f64, f64, f64, f64)>,
}

impl Bin {
    fn new() -> Self {
        Bin {
            polys: Vec::new(),
            bbs: Vec::new(),
        }
    }
    fn no_overlap(&self, c: &[Pt], cb: (f64, f64, f64, f64)) -> bool {
        self.polys
            .iter()
            .zip(&self.bbs)
            .all(|(p, &pb)| !overlaps(c, cb, p, pb))
    }
    fn push(&mut self, p: Vec<Pt>) {
        self.bbs.push(bbox(&p));
        self.polys.push(p);
    }
}

fn place(
    piece: &[Pt],
    stock: &[Pt],
    sbb: (f64, f64, f64, f64),
    bin: &Bin,
    angs: &[f64],
    gn: usize,
) -> Option<(Vec<Pt>, f64, f64, f64)> {
    let (sx0, sy0, sx1, sy1) = sbb;
    let sw = sx1 - sx0;
    let sh = sy1 - sy0;

    for &ang in angs {
        let rot = rotate_pts(piece, ang);
        let (rx0, ry0, rx1, ry1) = bbox(&rot);
        let pw = rx1 - rx0;
        let ph = ry1 - ry0;
        if pw > sw + 1e-9 || ph > sh + 1e-9 {
            continue;
        }

        let norm: Vec<Pt> = rot.iter().map(|&(x, y)| (x - rx0, y - ry0)).collect();
        let rx = (sw - pw).max(0.0);
        let ry = (sh - ph).max(0.0);
        let nx = if rx < 1e-9 { 1 } else { gn };
        let ny = if ry < 1e-9 { 1 } else { gn };
        let dx = if nx <= 1 { 0.0 } else { rx / (nx - 1) as f64 };
        let dy = if ny <= 1 { 0.0 } else { ry / (ny - 1) as f64 };

        for iy in 0..ny {
            let ty = sy0 + iy as f64 * dy;
            for ix in 0..nx {
                let tx = sx0 + ix as f64 * dx;
                let cand: Vec<Pt> = norm.iter().map(|&(x, y)| (x + tx, y + ty)).collect();
                let cb = (tx, ty, tx + pw, ty + ph);
                if fits_in(&cand, stock) && bin.no_overlap(&cand, cb) {
                    return Some((cand, tx - rx0, ty - ry0, ang));
                }
            }
        }
    }
    None
}

fn main() {
    let mut src = String::new();
    io::stdin().lock().read_to_string(&mut src).unwrap();
    let mut it = src.split_ascii_whitespace();
    macro_rules! g {
        ($t:ty) => {
            it.next().unwrap().parse::<$t>().unwrap()
        };
    }

    let sv = g!(usize);
    let np = g!(usize);
    let stock: Vec<Pt> = (0..sv).map(|_| (g!(f64), g!(f64))).collect();
    let pieces: Vec<Vec<Pt>> = (0..np)
        .map(|_| {
            let pv = g!(usize);
            (0..pv).map(|_| (g!(f64), g!(f64))).collect()
        })
        .collect();

    let sbb = bbox(&stock);

    let (angs, gn): (Vec<f64>, usize) = match np {
        0..=20 => ((0..24).map(|i| i as f64 * PI / 12.0).collect(), 20),
        21..=100 => ((0..8).map(|i| i as f64 * PI / 4.0).collect(), 15),
        101..=2000 => (vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0], 8),
        _ => (vec![0.0, PI / 2.0, PI, 3.0 * PI / 2.0], 5),
    };

    let mut ord: Vec<usize> = (0..np).collect();
    ord.sort_unstable_by(|&a, &b| {
        poly_area(&pieces[b])
            .partial_cmp(&poly_area(&pieces[a]))
            .unwrap_or(std::cmp::Ordering::Equal)
    });

    let max_bins: usize = match np {
        0..=100 => usize::MAX,
        101..=1000 => 15,
        _ => 7,
    };

    let mut bins: Vec<Bin> = Vec::new();
    let mut res: Vec<(usize, f64, f64, f64)> = vec![(0, 0.0, 0.0, 0.0); np];

    for &pi in &ord {
        let piece = &pieces[pi];
        let mut done = false;

        let start = bins.len().saturating_sub(max_bins);
        for si in start..bins.len() {
            if let Some((c, x, y, a)) = place(piece, &stock, sbb, &bins[si], &angs, gn) {
                res[pi] = (si, x, y, a);
                bins[si].push(c);
                done = true;
                break;
            }
        }

        if !done {
            let si = bins.len();
            bins.push(Bin::new());
            match place(piece, &stock, sbb, &bins[si], &angs, gn) {
                Some((c, x, y, a)) => {
                    res[pi] = (si, x, y, a);
                    bins[si].push(c);
                }
                None => {
                    res[pi] = (si, sbb.0, sbb.1, 0.0);
                }
            }
        }
    }

    let out = io::stdout();
    let mut w = io::BufWriter::new(out.lock());
    writeln!(w, "{}", bins.len()).unwrap();
    for i in 0..np {
        let (si, x, y, a) = res[i];
        writeln!(w, "{} {} {} {}", si, x, y, a).unwrap();
    }
}
