    fn bbox(&self) -> [f64; 4] {
        let mut min_x = f64::MAX; let mut min_y = f64::MAX;
        let mut max_x = f64::MIN; let mut max_y = f64::MIN;
        for &[x, y] in &self.verts {
            if x < min_x { min_x = x; }
            if y < min_y { min_y = y; }
            if x > max_x { max_x = x; }
            if y > max_y { max_y = y; }
        }
        [min_x, min_y, max_x, max_y]
    }
    
    fn translate(&self, dx: f64, dy: f64) -> Self {
        Poly { verts: self.verts.iter().map(|&[x,y]| [x+dx, y+dy]).collect() }
    }
    
    fn rotate(&self, angle: f64) -> Self {
        let (s, c) = angle.sin_cos();
        Poly { verts: self.verts.iter().map(|&[x,y]| [x*c - y*s, x*s + y*c]).collect() }
    }
    
    fn area(&self) -> f64 {
        let n = self.verts.len();
        let mut a = 0.0f64;
        for i in 0..n {
            let [x1,y1] = self.verts[i];
            let [x2,y2] = self.verts[(i+1)%n];
            a += x1*y2 - x2*y1;
        }
        (a/2.0).abs()
    }
}

fn sat_axes(poly: &Poly) -> Vec<[f64;2]> {
    let n = poly.verts.len();

        let [x2,y2] = poly.verts[(i+1)%n];
        let dx = x2-x1; let dy = y2-y1;
        let len = (dx*dx+dy*dy).sqrt();
        if len > 1e-10 { axes.push([-dy/len, dx/len]); }
    }
    axes
}

fn project(poly: &Poly, axis: [f64;2]) -> (f64,f64) {
    let mut mn = f64::MAX; let mut mx = f64::MIN;
    for &[x,y] in &poly.verts {
        let d = x*axis[0] + y*axis[1];
        if d < mn { mn = d; }
        if d > mx { mx = d; }
    }
    (mn, mx)
}

fn overlaps_sat(a: &Poly, b: &Poly) -> bool {
    let mut axes = sat_axes(a);
    axes.extend(sat_axes(b));
    for ax in axes {
        let (mn_a, mx_a) = project(a, ax);
        let (mn_b, mx_b) = project(b, ax);
        if mx_a < mn_b - 1e-9 || mx_b < mn_a - 1e-9 { return false; }
    }
    true
}

fn point_in_poly(px: f64, py: f64, poly: &Poly) -> bool {
    let verts = &poly.verts;
    let n = verts.len();
    let mut inside = false;
    let mut j = n - 1;
    for i in 0..n {
        let [xi, yi] = verts[i];
        let [xj, yj] = verts[j];
        if ((yi > py) != (yj > py)) && (px < (xj-xi)*(py-yi)/(yj-yi)+xi) {
            inside = !inside;
        }
        j = i;
    }
    inside
}

    true

fn segments_intersect(p1: [f64;2], p2: [f64;2], p3: [f64;2], p4: [f64;2]) -> bool {
    let d1 = [p2[0]-p1[0], p2[1]-p1[1]];
    let d2 = [p4[0]-p3[0], p4[1]-p3[1]];
    let cross = d1[0]*d2[1] - d1[1]*d2[0];
    if cross.abs() < 1e-10 { return false; }
    let t = ((p3[0]-p1[0])*d2[1] - (p3[1]-p1[1])*d2[0]) / cross;
    let u = ((p3[0]-p1[0])*d1[1] - (p3[1]-p1[1])*d1[0]) / cross;
    t > 1e-9 && t < 1.0-1e-9 && u > 1e-9 && u < 1.0-1e-9
}

fn poly_edges_intersect(a: &Poly, b: &Poly) -> bool {

                if let Some((tx, ty)) = try_place(&norm, &stock, &stocks[si], step) {
                    let placed_poly = norm.translate(tx, ty);
                    let final_tx = tx - bb[0];
                    let final_ty = ty - bb[1];
                    results[pi] = (si, final_tx, final_ty, rot);
                    let pp_bb = placed_poly.bbox();
                    stocks[si].push(placed_poly);
                    stock_placed_bbs[si].push(pp_bb);
                    found = true;
                    break;
                }
            }

            if !found {
                // Fallback: place at stock bbox origin
                let s_bb = stock.bbox();
                results[pi] = (si, s_bb[0], s_bb[1], 0.0);
                let placed_poly = piece.translate(s_bb[0], s_bb[1]);
                let pp_bb = placed_poly.bbox();
                stocks[si].push(placed_poly);
                stock_placed_bbs[si].push(pp_bb);
            }
        }
    }

    let num_stocks = stocks.len();
    println!("{}", num_stocks);
    for i in 0..np {
        let (si, tx, ty, rot) = results[i];
        println!("{} {} {} {}", si, tx, ty, rot);
    }
}
