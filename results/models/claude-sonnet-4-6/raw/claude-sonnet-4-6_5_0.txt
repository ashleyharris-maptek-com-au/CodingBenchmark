use std::io::{self, BufRead, Write};

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = io::BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let header = lines.next().unwrap().unwrap();
    let mut it = header.split_whitespace();
    let width: usize = it.next().unwrap().parse().unwrap();
    let height: usize = it.next().unwrap().parse().unwrap();
    let obstacle_count: usize = it.next().unwrap().parse().unwrap();
    let require_cycle: bool = it.next().unwrap().trim() == "1";

    let n = width * height;
    let mut obstacle = vec![false; n];
    for _ in 0..obstacle_count {
        let line = lines.next().unwrap().unwrap();
        let mut it = line.split_whitespace();
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if y < height && x < width {
            obstacle[y * width + x] = true;
        }
    }

    if obstacle[0] {
        writeln!(out, "0").unwrap();
        return;
    }

    let total = obstacle.iter().filter(|&&b| !b).count();
    if total == 0 {
        writeln!(out, "0").unwrap();
        return;
    }

    // Build flat adjacency list
    let adj: Vec<Vec<usize>> = (0..n)
        .map(|i| {
            if obstacle[i] {
                return vec![];
            }
            let x = i % width;
            let y = i / width;
            let mut v = Vec::with_capacity(4);
            if x > 0 && !obstacle[i - 1] {
                v.push(i - 1);
            }
            if x + 1 < width && !obstacle[i + 1] {
                v.push(i + 1);
            }
            if y > 0 && !obstacle[i - width] {
                v.push(i - width);
            }
            if y + 1 < height && !obstacle[i + width] {
                v.push(i + width);
            }
            v
        })
        .collect();

    // Connectivity check: all non-obstacle cells reachable from 0
    {
        let mut vis = vec![false; n];
        vis[0] = true;
        let mut q = vec![0usize];
        let mut qi = 0;
        let mut cnt = 1usize;
        while qi < q.len() {
            let c = q[qi];
            qi += 1;
            for &nb in &adj[c] {
                if !vis[nb] {
                    vis[nb] = true;
                    q.push(nb);
                    cnt += 1;
                }
            }
        }
        if cnt != total {
            writeln!(out, "0").unwrap();
            return;
        }
    }

    // Early exit: require_cycle with degree-1 start and total > 2 is impossible
    if require_cycle && adj[0].len() < 2 && total > 2 {
        writeln!(out, "0").unwrap();
        return;
    }

    // degree[i] = count of unvisited non-obstacle neighbors
    let mut deg: Vec<i32> = adj.iter().map(|v| v.len() as i32).collect();
    let mut vis = vec![false; n];
    vis[0] = true;
    for &nb in &adj[0] {
        deg[nb] -= 1;
    }

    let mut path = vec![0usize];
    let mut sg = vec![0u32; n]; // generation marks for BFS reuse
    let mut gen = 0u32;
    let mut bq: Vec<usize> = Vec::with_capacity(n);

    if bt(
        &adj,
        &mut vis,
        &mut path,
        &mut deg,
        total,
        require_cycle,
        &mut sg,
        &mut gen,
        &mut bq,
    ) {
        writeln!(out, "{}", path.len()).unwrap();
        for &i in &path {
            writeln!(out, "{} {}", i % width, i / width).unwrap();
        }
    } else {
        writeln!(out, "0").unwrap();
    }
}

/// BFS connectivity check: can we reach all `rem` unvisited cells from `cur`?
/// Uses generation-stamped array to avoid allocation on every call.
fn conn_ok(
    adj: &[Vec<usize>],
    vis: &[bool],
    cur: usize,
    rem: usize,
    sg: &mut Vec<u32>,
    gen: &mut u32,
    q: &mut Vec<usize>,
) -> bool {
    if rem == 0 {
        return true;
    }
    *gen = gen.wrapping_add(1);
    let g = *gen;
    sg[cur] = g;
    q.clear();
    let mut cnt = 0usize;
    for &nb in &adj[cur] {
        if !vis[nb] && sg[nb] != g {
            sg[nb] = g;
            q.push(nb);
            cnt += 1;
        }
    }
    let mut qi = 0;
    while qi < q.len() {
        let c = q[qi];
        qi += 1;
        for &nb in &adj[c] {
            if !vis[nb] && sg[nb] != g {
                sg[nb] = g;
                q.push(nb);
                cnt += 1;
            }
        }
    }
    cnt == rem
}

fn bt(
    adj: &[Vec<usize>],
    vis: &mut Vec<bool>,
    path: &mut Vec<usize>,
    deg: &mut Vec<i32>,
    total: usize,
    cyc: bool,
    sg: &mut Vec<u32>,
    gen: &mut u32,
    bq: &mut Vec<usize>,
) -> bool {
    let rem = total - path.len();
    if rem == 0 {
        if cyc {
            let &last = path.last().unwrap();
            return adj[last].iter().any(|&nb| nb == 0);
        }
        return true;
    }

    let &cur = path.last().unwrap();

    // Connectivity pruning: all remaining unvisited cells must be reachable from cur
    if rem > 1 && !conn_ok(adj, vis, cur, rem, sg, gen, bq) {
        return false;
    }

    // Collect candidate moves with Warnsdorff degree (ascending = most constrained first)
    let mut moves: Vec<(usize, i32)> = adj[cur]
        .iter()
        .filter(|&&nb| !vis[nb])
        .map(|&nb| (nb, deg[nb]))
        .collect();

    if moves.is_empty() {
        return false;
    }

    moves.sort_unstable_by_key(|&(_, d)| d);

    for (nb, _) in moves {
        // Visit nb
        vis[nb] = true;
        path.push(nb);
        for &x in &adj[nb] {
            deg[x] -= 1;
        }

        if bt(adj, vis, path, deg, total, cyc, sg, gen, bq) {
            return true;
        }

        // Un-visit nb
        for &x in &adj[nb] {
            deg[x] += 1;
        }
        path.pop();
        vis[nb] = false;
    }

    false
}
