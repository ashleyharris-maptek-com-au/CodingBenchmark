using System;
using System.Collections.Generic;

class Tetris
{
    static int W, H;
    static long[] board;
    static long fullMask;

    // El-Tetris weights (well-tuned for standard play)
    const double WH = -0.510066;
    const double WL =  0.760666;
    const double WO = -0.35663;
    const double WB = -0.184483;

    static (int r, int c)[][][] shapes;

    static void Main()
    {
        W = int.Parse(Next()); H = int.Parse(Next()); int N = int.Parse(Next());
        fullMask = (1L << W) - 1;
        board = new long[H];
        shapes = MakeShapes();

        for (int i = 0; i < N; i++)
        {
            string tok = Next();
            if (tok == null) break;
            int pt = "IOTSZJL".IndexOf(tok[0]);

            (int rot, int col) = BestMove(pt);
            Console.WriteLine(rot + " " + col);
            Console.Out.Flush();

            Apply(board, shapes[pt][rot], col);
            string res = Next();
            if (res == null || res == "gameover") break;
        }
    }

    static (int, int) BestMove(int pt)
    {
        double best = double.NegativeInfinity;
        int br = 0, bc = 0;
        for (int rot = 0; rot < 4; rot++)
        {
            var sh = shapes[pt][rot];
            int maxC = 0;
            foreach (var (_, c) in sh) if (c > maxC) maxC = c;
            for (int col = 0; col + maxC < W; col++)
            {
                int dr = GetDrop(board, sh, col);
                if (dr < 0) continue;
                long[] sim = (long[])board.Clone();
                foreach (var (r, c) in sh) sim[dr + r] |= 1L << (col + c);
                int lines = ClearLines(sim);
                double s = Eval(sim, lines);
                if (s > best) { best = s; br = rot; bc = col; }
            }
        }
        return (br, bc);
    }

    // Column-top approach: for each piece cell, find topmost occupied cell in that column.
    // The piece cannot pass through existing blocks, so the landing row is constrained per column.
    static int GetDrop(long[] b, (int r, int c)[] sh, int col)
    {
        int land = H;
        foreach (var (dr, dc) in sh)
        {
            int bc = col + dc, top = H;
            for (int r = 0; r < H; r++)
                if (((b[r] >> bc) & 1) != 0) { top = r; break; }
            int lim = top - dr - 1;
            if (lim < land) land = lim;
        }
        return land;
    }

    static int ClearLines(long[] b)
    {
        int cl = 0, wr = H - 1;
        for (int r = H - 1; r >= 0; r--)
            if (b[r] == fullMask) cl++;
            else b[wr--] = b[r];
        while (wr >= 0) b[wr--] = 0;
        return cl;
    }

    static void Apply(long[] b, (int r, int c)[] sh, int col)
    {
        int dr = GetDrop(b, sh, col);
        if (dr < 0) return;
        foreach (var (r, c) in sh) b[dr + r] |= 1L << (col + c);
        ClearLines(b);
    }

    // Adaptive evaluation: scale penalties based on how full the board is
    static double Eval(long[] b, int lines)
    {
        int agg = 0, holes = 0, bump = 0, ph = 0, maxH = 0;
        for (int c = 0; c < W; c++)
        {
            int h = 0; bool fil = false;
            for (int r = 0; r < H; r++)
            {
                bool on = ((b[r] >> c) & 1) != 0;
                if (on) { if (h == 0) h = H - r; fil = true; }
                else if (fil) holes++;
            }
            agg += h;
            if (h > maxH) maxH = h;
            if (c > 0) bump += Math.Abs(h - ph);
            ph = h;
        }
        // Increase height/hole penalties as board approaches the top
        double danger = (double)maxH / H;
        double wh = WH * (1.0 + danger * 0.5);
        double wo = WO * (1.0 + danger);
        return wh * agg + WL * lines + wo * holes + WB * bump;
    }

    // Piece shapes: [type][rotation] -> cells as (row, col) offsets
    // All pieces normalized so min col = 0, min row = 0
    // Rotations: 0=spawn, 1=90 CW, 2=180 CW, 3=270 CW
    // Piece order: I=0 O=1 T=2 S=3 Z=4 J=5 L=6
    static (int r, int c)[][][] MakeShapes()
    {
        (int r, int c)[] h4 = { (0,0),(0,1),(0,2),(0,3) };
        (int r, int c)[] v4 = { (0,0),(1,0),(2,0),(3,0) };
        (int r, int c)[] sq = { (0,0),(0,1),(1,0),(1,1) };
        return new (int r, int c)[][][]
        {
            new[]{ h4, v4, h4, v4 }, // I
            new[]{ sq, sq, sq, sq }, // O
            new[]{
                // T: spawn=.X./XXX, 90CW=X./XX/X., 180=XXX/.X., 270=.X/XX/.X
                new (int r, int c)[]{ (0,1),(1,0),(1,1),(1,2) },
                new (int r, int c)[]{ (0,0),(1,0),(1,1),(2,0) },
                new (int r, int c)[]{ (0,0),(0,1),(0,2),(1,1) },
                new (int r, int c)[]{ (0,1),(1,0),(1,1),(2,1) },
            }, // T
            new[]{
                // S: spawn=.XX/XX., 90CW=X./XX/.X
                new (int r, int c)[]{ (0,1),(0,2),(1,0),(1,1) },
                new (int r, int c)[]{ (0,0),(1,0),(1,1),(2,1) },
                new (int r, int c)[]{ (0,1),(0,2),(1,0),(1,1) },
                new (int r, int c)[]{ (0,0),(1,0),(1,1),(2,1) },
            }, // S
            new[]{
                // Z: spawn=XX./.XX, 90CW=.X/XX/X.
                new (int r, int c)[]{ (0,0),(0,1),(1,1),(1,2) },
                new (int r, int c)[]{ (0,1),(1,0),(1,1),(2,0) },
                new (int r, int c)[]{ (0,0),(0,1),(1,1),(1,2) },
                new (int r, int c)[]{ (0,1),(1,0),(1,1),(2,0) },
            }, // Z
            new[]{
                // J: spawn=X../XXX, 90CW=XX/X./X., 180=XXX/..X, 270=.X/.X/XX
                new (int r, int c)[]{ (0,0),(1,0),(1,1),(1,2) },
                new (int r, int c)[]{ (0,0),(0,1),(1,0),(2,0) },
                new (int r, int c)[]{ (0,0),(0,1),(0,2),(1,2) },
                new (int r, int c)[]{ (0,1),(1,1),(2,0),(2,1) },
            }, // J
            new[]{
                // L: spawn=..X/XXX, 90CW=X./X./XX, 180=XXX/X.., 270=XX/.X/.X
                new (int r, int c)[]{ (0,2),(1,0),(1,1),(1,2) },
                new (int r, int c)[]{ (0,0),(1,0),(2,0),(2,1) },
                new (int r, int c)[]{ (0,0),(0,1),(0,2),(1,0) },
                new (int r, int c)[]{ (0,0),(0,1),(1,1),(2,1) },
            }, // L
        };
    }

    static readonly Queue<string> _q = new Queue<string>();
    static string Next()
    {
        while (_q.Count == 0)
        {
            string line = Console.ReadLine();
            if (line == null) return null;
            foreach (string t in line.Split(new char[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries))
                _q.Enqueue(t);
        }
        return _q.Dequeue();
    }
}
