#include <iostream>
#include <vector>
#include <algorithm>
#include <chrono>
#include <random>
#include <cmath>
#include <cstdint>
using namespace std;
using clk = chrono::steady_clock;

static int J, M, N;
static vector<vector<int>> machOf, durOf;
static vector<int> gJob, gTask;
static vector<vector<int>> gId;
static vector<int> bmp, bms, bIn, bQ;

// Evaluate machine-order sequences via topological BFS.
// Returns makespan; fills st[i] = start time of global task i.
static int evalSeq(const vector<vector<int>>& seq, vector<int>& st) {
    fill(bmp.begin(), bmp.end(), -1);
    fill(bms.begin(), bms.end(), -1);
    for (int m = 0; m < M; m++) {
        const auto& s = seq[m];
        for (int k = 0, sz = (int)s.size()-1; k < sz; k++) {
            bms[s[k]] = s[k+1];
            bmp[s[k+1]] = s[k];
        }
    }
    for (int i = 0; i < N; i++)
        bIn[i] = (gTask[i] > 0 ? 1 : 0) + (bmp[i] >= 0 ? 1 : 0);

    fill(st.begin(), st.end(), 0);
    bQ.clear();
    for (int i = 0; i < N; i++) if (!bIn[i]) bQ.push_back(i);

    for (int qi = 0, qn = (int)bQ.size(); qi < qn; qi++) {
        int u = bQ[qi];
        int j = gJob[u], t = gTask[u];
        int e = st[u] + durOf[j][t];
        if (t+1 < (int)machOf[j].size()) {
            int v = gId[j][t+1];
            if (e > st[v]) st[v] = e;
            if (!--bIn[v]) { bQ.push_back(v); ++qn; }
        }
        if (bms[u] >= 0) {
            int v = bms[u];
            if (e > st[v]) st[v] = e;
            if (!--bIn[v]) { bQ.push_back(v); ++qn; }
        }
    }

    int mk = 0;
    for (int i = 0; i < N; i++) {
        int e = st[i] + durOf[gJob[i]][gTask[i]];
        if (e > mk) mk = e;
    }
    return mk;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(nullptr);

    cin >> J >> M;
    machOf.resize(J); durOf.resize(J); gId.resize(J);
    N = 0;
    for (int j = 0; j < J; j++) {
        int T; cin >> T;
        machOf[j].resize(T); durOf[j].resize(T); gId[j].resize(T);
        for (int t = 0; t < T; t++) {
            cin >> machOf[j][t] >> durOf[j][t];
            gId[j][t] = N++;
            gJob.push_back(j);
            gTask.push_back(t);
        }
    }

    if (N == 0) {
        for (int j = 0; j < J; j++) cout << '\n';
        return 0;
    }

    bmp.assign(N, -1); bms.assign(N, -1);
    bIn.resize(N); bQ.reserve(N);

    // Suffix duration for MWKR (Most Work Remaining) heuristic
    vector<int> sfx(N);
    for (int j = 0; j < J; j++) {
        int s = 0;
        for (int t = (int)machOf[j].size()-1; t >= 0; t--) {
            s += durOf[j][t];
            sfx[gId[j][t]] = s;
        }
    }

    // Build per-machine task lists
    vector<vector<int>> base(M);
    for (int j = 0; j < J; j++)
        for (int t = 0; t < (int)machOf[j].size(); t++)
            base[machOf[j][t]].push_back(gId[j][t]);

    // Build two candidate initial sequences: MWKR (descending sfx) and LWR (ascending sfx)
    auto makeSeq = [&](bool mwkr) {
        auto s = base;
        for (int m = 0; m < M; m++) {
            sort(s[m].begin(), s[m].end(), [&](int a, int b) {
                return mwkr ? (sfx[a] > sfx[b]) : (sfx[a] < sfx[b]);
            });
        }
        return s;
    };

    vector<int> st1(N), st2(N);
    auto s1 = makeSeq(true);
    auto s2 = makeSeq(false);
    int mk1 = evalSeq(s1, st1);
    int mk2 = evalSeq(s2, st2);

    vector<vector<int>> curSeq, bestSeq;
    vector<int> curSt, bestSt, tmpSt(N);
    int curMk, bestMk;

    if (mk1 <= mk2) {
        curSeq = bestSeq = s1;
        curSt = bestSt = st1;
        curMk = bestMk = mk1;
    } else {
        curSeq = bestSeq = s2;
        curSt = bestSt = st2;
        curMk = bestMk = mk2;
    }

    // Collect machines with 2+ tasks (eligible for swaps)
    vector<int> am;
    for (int m = 0; m < M; m++)
        if ((int)curSeq[m].size() > 1) am.push_back(m);

    if (!am.empty()) {
        mt19937 rng(42);
        const int AM = (int)am.size();
        const auto deadline = clk::now() + chrono::milliseconds(4800);

        const double T0 = max(1.0, (double)bestMk * 0.1);
        double temp = T0;

        // Cooling rate: slower for small N (more iterations feasible)
        const double cool = N <= 200   ? 0.999995 :
                            N <= 1000  ? 0.99999  :
                            N <= 5000  ? 0.9999   :
                            N <= 20000 ? 0.9998   : 0.9997;

        // Periodic restart period (iterations)
        const long long restartPeriod = N <= 1000 ? 500000LL :
                                        N <= 5000 ? 100000LL : 20000LL;

        for (long long iter = 0; ; ++iter) {
            // Check deadline every 256 iterations to amortize clock overhead
            if ((iter & 0xFF) == 0 && clk::now() >= deadline) break;

            // Periodic restart from best to escape local optima
            if (iter > 0 && iter % restartPeriod == 0) {
                curSeq = bestSeq; curSt = bestSt; curMk = bestMk;
                temp = T0 * 0.3;
            }

            const int m = am[(uint32_t)rng() % (uint32_t)AM];
            const int sz = (int)curSeq[m].size();
            const int k = (int)((uint32_t)rng() % (uint32_t)(sz-1));

            swap(curSeq[m][k], curSeq[m][k+1]);
            const int newMk = evalSeq(curSeq, tmpSt);
            const int delta = newMk - curMk;

            bool accept = (delta <= 0);
            if (!accept && temp > 0.01) {
                const double r = (double)(uint32_t)(rng()) / 4294967296.0;
                accept = (r < exp(-(double)delta / temp));
            }

            if (accept) {
                curMk = newMk;
                swap(curSt, tmpSt);
                if (curMk < bestMk) {
                    bestMk = curMk;
                    bestSeq = curSeq;
                    bestSt = curSt;
                }
            } else {
                swap(curSeq[m][k], curSeq[m][k+1]);
            }

            temp *= cool;
        }
    }

    // Output: for each job, print "start_time machine" pairs for each task
    for (int j = 0; j < J; j++) {
        const int T = (int)machOf[j].size();
        for (int t = 0; t < T; t++) {
            if (t) cout << ' ';
            cout << bestSt[gId[j][t]] << ' ' << machOf[j][t];
        }
        cout << '\n';
    }

    return 0;
}
