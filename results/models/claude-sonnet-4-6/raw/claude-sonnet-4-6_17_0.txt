using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

class Program
{
    struct Space
    {
        public long X, Y, Z, W, H, D;
        public long Volume => W * H * D;
    }

    static int N;
    static long CW, CH, CD;
    static long[] BW, BH, BD;
    const int MAX_SPACES = 8000;

    static void Main()
    {
        string[] tok = Console.In.ReadToEnd()
            .Split(new char[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int t = 0;

        N = int.Parse(tok[t++]);
        CW = long.Parse(tok[t++]);
        CH = long.Parse(tok[t++]);
        CD = long.Parse(tok[t++]);

        BW = new long[N]; BH = new long[N]; BD = new long[N];
        for (int i = 0; i < N; i++)
        {
            BW[i] = long.Parse(tok[t++]);
            BH[i] = long.Parse(tok[t++]);
            BD[i] = long.Parse(tok[t++]);
        }

        // Try multiple box orderings and take the best result
        int[][] orders = new int[][]
        {
            MakeOrder(i => -BW[i] * BH[i] * BD[i]),                              // Volume desc
            MakeOrder(i => BW[i] * BH[i] * BD[i]),                               // Volume asc
            MakeOrder(i => -Math.Max(BW[i], Math.Max(BH[i], BD[i]))),             // Max-dim desc
            MakeOrder(i => Math.Min(BW[i], Math.Min(BH[i], BD[i]))),              // Min-dim asc
        };

        (int idx, long x, long y, long z)[] best = new (int, long, long, long)[0];
        foreach (int[] order in orders)
        {
            var res = Pack(order);
            if (res.Length > best.Length)
                best = res;
        }

        var sb = new StringBuilder();
        sb.Append(best.Length).Append('\n');
        foreach (var (idx, x, y, z) in best)
            sb.Append(idx).Append(' ').Append(x).Append(' ').Append(y).Append(' ').Append(z).Append('\n');
        Console.Write(sb);
    }

    static int[] MakeOrder(Func<int, long> key)
        => Enumerable.Range(0, N).OrderBy(key).ToArray();

    static (int, long, long, long)[] Pack(int[] order)
    {
        var spaces = new List<Space>(MAX_SPACES + 4);
        spaces.Add(new Space { X = 0, Y = 0, Z = 0, W = CW, H = CH, D = CD });

        var result = new (int idx, long x, long y, long z)[N];
        int count = 0;

        foreach (int i in order)
        {
            long dw = BW[i], dh = BH[i], dd = BD[i];
            long boxVol = dw * dh * dd;

            int bestSi = -1;
            long bestFit = long.MaxValue;
            long bestW = 0, bestH = 0, bestD = 0;

            for (int si = 0; si < spaces.Count; si++)
            {
                Space sp = spaces[si];
                long sv = sp.Volume;
                if (sv < boxVol || sv >= bestFit) continue;

                if (TryFit(dw, dh, dd, sp.W, sp.H, sp.D, out long fw, out long fh, out long fd))
                {
                    bestFit = sv;
                    bestSi = si;
                    bestW = fw; bestH = fh; bestD = fd;
                }
            }

            if (bestSi < 0) continue;

            Space chosen = spaces[bestSi];
            result[count++] = (i, chosen.X, chosen.Y, chosen.Z);

            // Swap-remove chosen space (O(1))
            int last = spaces.Count - 1;
            if (bestSi != last) spaces[bestSi] = spaces[last];
            spaces.RemoveAt(last);

            // Add split sub-spaces
            Split(spaces, chosen, bestW, bestH, bestD);

            // Prune spaces if too many, keeping the largest
            if (spaces.Count > MAX_SPACES)
            {
                spaces.Sort((a, b) => b.Volume.CompareTo(a.Volume));
                spaces.RemoveRange(MAX_SPACES, spaces.Count - MAX_SPACES);
            }
        }

        return result[..count];
    }

    static bool TryFit(long dw, long dh, long dd, long sw, long sh, long sd,
        out long fw, out long fh, out long fd)
    {
        // Sort box dims: a <= b <= c, then try all 6 permutations against space
        long a = dw, b = dh, c = dd;
        if (a > b) { long t = a; a = b; b = t; }
        if (b > c) { long t = b; b = c; c = t; }
        if (a > b) { long t = a; a = b; b = t; }

        if (a <= sw && b <= sh && c <= sd) { fw = a; fh = b; fd = c; return true; }
        if (a <= sw && c <= sh && b <= sd) { fw = a; fh = c; fd = b; return true; }
        if (b <= sw && a <= sh && c <= sd) { fw = b; fh = a; fd = c; return true; }
        if (b <= sw && c <= sh && a <= sd) { fw = b; fh = c; fd = a; return true; }
        if (c <= sw && a <= sh && b <= sd) { fw = c; fh = a; fd = b; return true; }
        if (c <= sw && b <= sh && a <= sd) { fw = c; fh = b; fd = a; return true; }

        fw = fh = fd = 0; return false;
    }

    // Guillotine split: 3 strategies, pick the one that maximizes the largest sub-space.
    // Each strategy partitions remaining volume (sp minus placed box) into 3 non-overlapping spaces.
    static void Split(List<Space> spaces, Space sp, long bw, long bh, long bd)
    {
        long rx = sp.W - bw, ry = sp.H - bh, rz = sp.D - bd;

        // Strategy 0 (split X first): sub-volumes = rx*H*D, bw*ry*D, bw*bh*rz
        long m0 = Max3(rx * sp.H * sp.D, bw * ry * sp.D, bw * bh * rz);
        // Strategy 1 (split Y first): sub-volumes = W*ry*D, rx*bh*D, bw*bh*rz
        long m1 = Max3(sp.W * ry * sp.D, rx * bh * sp.D, bw * bh * rz);
        // Strategy 2 (split Z first): sub-volumes = W*H*rz, rx*H*bd, bw*ry*bd
        long m2 = Max3(sp.W * sp.H * rz, rx * sp.H * bd, bw * ry * bd);

        if (m0 >= m1 && m0 >= m2)
        {
            if (rx > 0) spaces.Add(new Space { X = sp.X + bw, Y = sp.Y, Z = sp.Z, W = rx, H = sp.H, D = sp.D });
            if (ry > 0) spaces.Add(new Space { X = sp.X, Y = sp.Y + bh, Z = sp.Z, W = bw, H = ry, D = sp.D });
            if (rz > 0) spaces.Add(new Space { X = sp.X, Y = sp.Y, Z = sp.Z + bd, W = bw, H = bh, D = rz });
        }
        else if (m1 >= m0 && m1 >= m2)
        {
            if (ry > 0) spaces.Add(new Space { X = sp.X, Y = sp.Y + bh, Z = sp.Z, W = sp.W, H = ry, D = sp.D });
            if (rx > 0) spaces.Add(new Space { X = sp.X + bw, Y = sp.Y, Z = sp.Z, W = rx, H = bh, D = sp.D });
            if (rz > 0) spaces.Add(new Space { X = sp.X, Y = sp.Y, Z = sp.Z + bd, W = bw, H = bh, D = rz });
        }
        else
        {
            if (rz > 0) spaces.Add(new Space { X = sp.X, Y = sp.Y, Z = sp.Z + bd, W = sp.W, H = sp.H, D = rz });
            if (rx > 0) spaces.Add(new Space { X = sp.X + bw, Y = sp.Y, Z = sp.Z, W = rx, H = sp.H, D = bd });
            if (ry > 0) spaces.Add(new Space { X = sp.X, Y = sp.Y + bh, Z = sp.Z, W = bw, H = ry, D = bd });
        }
    }

    static long Max3(long a, long b, long c) => a >= b ? (a >= c ? a : c) : (b >= c ? b : c);
}
