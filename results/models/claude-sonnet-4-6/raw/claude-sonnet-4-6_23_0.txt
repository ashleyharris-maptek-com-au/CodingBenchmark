- Slow down as we approach, align angle to face up (0 radians)

- Thrust applies in lander's facing direction

```rust
    updated: bool,

impl Default for State {
    fn default() -> Self {

fn main() {
    let state = Arc::new(Mutex::new(State::default()));
    let map_info = Arc::new(Mutex::new(MapInfo::default()));

    let state_writer = Arc::clone(&state);
    let map_writer = Arc::clone(&map_info);

    // Reader thread
    thread::spawn(move || {
        let stdin = io::stdin();
        let mut reader = stdin.lock();
        let mut line = String::new();
        
        // Read header: width height gravity max_thrust fuel
        line.clear();
        if reader.read_line(&mut line).unwrap_or(0) == 0 { return; }
        let header: Vec<f64> = line.split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if header.len() < 5 { return; }
        let width = header[0] as usize;
        let height = header[1] as usize;
        let gravity = header[2];
        let max_thrust = header[3];
        // let initial_fuel = header[4];

        // Read start/target line
        line.clear();
        if reader.read_line(&mut line).unwrap_or(0) == 0 { return; }
        let coords: Vec<f64> = line.split_whitespace()
            .filter_map(|s| s.parse().ok())
            .collect();
        if coords.len() < 4 { return; }
        let target_x = coords[2];
        let target_y = coords[3];

        // Skip map rows (must read them to drain buffer, but efficiently)
        for _ in 0..height {
            line.clear();
            if reader.read_line(&mut line).unwrap_or(0) == 0 { break; }
            // Don't store map data (could be huge)
        }
        let _ = width; // suppress warning

        {
            let mut mi = map_writer.lock().unwrap();
            mi.gravity = gravity;
            mi.max_thrust = max_thrust;
            mi.target_x = target_x;
            mi.target_y = target_y;
            mi.ready = true;
        }

        // Read continuous state updates
        loop {
            line.clear();
            match reader.read_line(&mut line) {
                Ok(0) => break,
                Ok(_) => {},
                Err(_) => break,
            }
            let parts: Vec<f64> = line.split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if parts.len() >= 7 {

                s.x = parts[0];
                s.y = parts[1];
                s.vx = parts[2];
                s.vy = parts[3];
                s.angle = parts[4];
                s.angular_vel = parts[5];
                s.fuel = parts[6];
                s.valid = true;
                s.updated = true;
            }
        }
    });

    // Wait for map info to be ready
    loop {
        {
            let mi = map_info.lock().unwrap();
            if mi.ready { break; }

        thread::sleep(Duration::from_millis(10));

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    // Control loop at 20Hz
    let interval = Duration::from_millis(50);
    
    loop {
        let start = std::time::Instant::now();
        
        let (s, mi) = {
            let s = state.lock().unwrap().clone();
            let mi = map_info.lock().unwrap().clone();
            (s, mi)
        };

        let (thrust, turn) = if !s.valid {

            compute_control(&s, &mi)
        };

        out.flush().unwrap();

        let elapsed = start.elapsed();
        if elapsed < interval {
            thread::sleep(interval - elapsed);
        }
    }
}

fn compute_control(s: &State, mi: &MapInfo) -> (f64, f64) {
    let gravity = mi.gravity;
    let max_thrust = mi.max_thrust;
    let target_x = mi.target_x;
    let target_y = mi.target_y;

    // Position errors (y increases downward)
    let dx = target_x - s.x;
    let dy = target_y - s.y; // positive means target is below us

    let dist = (dx * dx + dy * dy).sqrt();

    // Velocity
    let vx = s.vx;
    let vy = s.vy; // positive = moving down

    // Desired velocities: move toward target, but limit speed
    let max_horiz_speed = (dist * 0.5 + 0.5).min(8.0);
    let max_vert_speed_approach = if dy > 5.0 {
        // Target is below us, we can descend
        (dy * 0.3 + 0.5).min(6.0)
    } else if dy > 1.0 {
        2.0
    } else {
        1.0
    };

    // If we're above the target (dy < 0), we need to go up... 
    // but wait: y increases downward, so dy = target_y - s.y
    // If target_y > s.y (target is lower), dy > 0, we need to descend
    // If target_y < s.y (target is higher), dy < 0... but target should be on ground

    // Desired vx: proportional to dx, clamped
    let desired_vx = (dx * 0.4).clamp(-max_horiz_speed, max_horiz_speed);
    // Desired vy: we want to descend at controlled rate
    let desired_vy = if dy > 2.0 {
        max_vert_speed_approach
    } else if dy > 0.0 {
        dy * 0.5
    } else {
        // We're below target? Emergency: thrust up
        -2.0
    };

    // Velocity errors
    let vx_err = desired_vx - vx;
    let vy_err = desired_vy - vy;

    // Desired accelerations (PD control)
    // For horizontal: ax = Kp * vx_err
    // For vertical: ay = gravity + Kp * vy_err (ay is net, thrust provides -thrust_component)
    let ax_desired = (vx_err * 1.5).clamp(-max_thrust * 0.8, max_thrust * 0.8);
    
    // Vertical: net_ay = gravity - thrust * cos(angle)
    // We want net_ay = vy_err * Kp
    // So thrust * cos(angle) = gravity - vy_err * Kp_v
    let vy_kp = 2.0;
    let thrust_vert = gravity - vy_err * vy_kp;

    // We need: thrust * sin(angle) = ax_desired (horizontal component)
    //          thrust * cos(angle) = thrust_vert (vertical component, note: angle=0 faces up)
    // Wait: angle=0 faces up. In our coordinate system (y down):
    // thrust_x = thrust * sin(angle)  (positive angle tilts right, thrust goes right)
    // thrust_y = -thrust * cos(angle) (negative because thrust opposes gravity when angle=0)
    // 
    // Net ay = gravity - thrust * cos(angle) * (sign adjustment)
    // When angle=0, lander faces up, thrust is in -y direction (upward)
    // So thrust_y_component = -max_thrust * thrust_frac * cos(angle)
    // Net vertical accel = gravity + thrust_y_component = gravity - max_thrust * T * cos(angle)
    //
    // Net horizontal accel = max_thrust * T * sin(angle)
    //
    // Desired: max_thrust * T * sin(angle) = ax_desired
    //          max_thrust * T * cos(angle) = thrust_vert (we want this to = gravity - vy_correction)
    //
    // So: desired_angle = atan2(ax_desired, thrust_vert)
    // And: total_thrust_accel = sqrt(ax_desired^2 + thrust_vert^2)
    // T = total_thrust_accel / max_thrust

    let desired_angle = ax_desired.atan2(thrust_vert);
    // Clamp desired angle to reasonable range (avoid flipping)
    let desired_angle = desired_angle.clamp(-PI / 2.5, PI / 2.5);

    let total_accel = (ax_desired * ax_desired + thrust_vert * thrust_vert).sqrt();
    let thrust_frac = (total_accel / max_thrust).clamp(0.0, 1.0);

    // Fuel conservation: if we're close to landing and moving slowly, reduce thrust
    let near_target = dist < 5.0;
    let slow_enough = vx.abs() < 1.5 && vy.abs() < 1.5 && near_target;
    let thrust_frac = if slow_enough && dy.abs() < 3.0 {
        // Just maintain hover or let gravity help
        thrust_frac * 0.8
    } else {
        thrust_frac
    };

    // Angle control: turn toward desired_angle
    let angle_err = desired_angle - s.angle;
    // Normalize angle error
    let angle_err = wrap_angle(angle_err);

    // PD control for angular velocity
    let desired_angular_vel = (angle_err * 3.0).clamp(-2.0, 2.0);
    let angular_vel_err = desired_angular_vel - s.angular_vel;
    let turn_cmd = (angular_vel_err * 1.0).clamp(-1.0, 1.0);

    // If very low on fuel, just try to angle upward and thrust
    let thrust_frac = if s.fuel < 5.0 {
        0.3
    } else {
        thrust_frac
    };

    (thrust_frac, turn_cmd)
}

fn wrap_angle(a: f64) -> f64 {
    let mut a = a;
    while a > PI { a -= 2.0 * PI; }
    while a < -PI { a += 2.0 * PI; }
    a
}
```

```rust
fn compute_control(s: &State, mi: &MapInfo) -> (f64, f64) {
    let g = mi.gravity;
    let max_thr = mi.max_thrust;

    // Target position (y increases downward)
    let tx = mi.target_x;
    let ty = mi.target_y;

    let dx = tx - s.x; // positive = target is to the right
    let dy = ty - s.y; // positive = target is below us

    let dist_horiz = dx.abs();
    let dist_vert = dy.abs();
    let dist = (dx * dx + dy * dy).sqrt();

    // === Desired velocity computation ===
    // Horizontal: move toward target, proportional to distance, clamped
    let approach_speed_h = (dist_horiz * 0.5).clamp(0.2, 8.0);
    let desired_vx = if dist_horiz < 0.5 {
        0.0
    } else {
        dx.signum() * approach_speed_h
    };

    // Vertical: descend toward target
    // If target is below (dy > 0): descend at controlled rate
    // If target is above (dy < 0): ascend slowly (shouldn't happen much)
    let desired_vy = if dy > 2.0 {
        // Descend toward target
        let approach_speed_v = (dy * 0.4).clamp(0.3, 5.0);
        approach_speed_v
    } else if dy > 0.3 {
        // Very close vertically, slow down
        dy * 0.5
    } else if dy > -1.0 {
        // At or slightly above target
        0.3 // small positive = gentle descent
    } else {
        // Above target, ascend slightly
        -1.0
    };

    // === PD velocity control ===
    let vx_err = desired_vx - s.vx;
    let vy_err = desired_vy - s.vy;

    // Kp gains
    let kp_h = 2.5;
    let kp_v = 2.0;

    // Desired accelerations
    // Horizontal: ax = kp_h * vx_err
    // Vertical (y-down): ay_net = g - thrust_y_component
    //   thrust_y_component = T * max_thr * cos(angle)  (positive when angle=0 = facing up = thrust opposes gravity)
    //   we want ay_net = kp_v * vy_err  => thrust_y = g - kp_v * vy_err

    let ax = (kp_h * vx_err).clamp(-max_thr * 0.7, max_thr * 0.7);
    let thrust_y_needed = g - kp_v * vy_err;
    // thrust_y_needed should be positive (upward force in y-down coords means thrust_y > 0)
    // clamp to achievable range
    let thrust_y_needed = thrust_y_needed.clamp(0.1, max_thr * 1.0);

    // Desired angle: atan2(ax, thrust_y_needed)
    // angle = 0 -> facing up -> thrust in -y direction -> thrust_x=0, thrust_y=max_thr
    // angle > 0 (right tilt) -> thrust_x = max_thr*sin(angle), thrust_y = max_thr*cos(angle)
    // So: ax = T * max_thr * sin(angle), thrust_y = T * max_thr * cos(angle)
    // => desired_angle = atan2(ax, thrust_y_needed)
    let desired_angle = ax.atan2(thrust_y_needed);
    // Limit angle to avoid dangerous tilts
    let max_angle = PI / 3.0; // 60 degrees max tilt
    let desired_angle = desired_angle.clamp(-max_angle, max_angle);

    // Total thrust magnitude
    let total_accel = (ax * ax + thrust_y_needed * thrust_y_needed).sqrt();
    let thrust_frac = (total_accel / max_thr).clamp(0.0, 1.0);

    // === Angle control ===
    let angle_err = wrap_angle(desired_angle - s.angle);
    let desired_ang_vel = (angle_err * 4.0).clamp(-3.0, 3.0);
    let ang_vel_err = desired_ang_vel - s.angular_vel;
    let turn_cmd = (ang_vel_err * 0.8).clamp(-1.0, 1.0);

    // === Landing approach: be gentler when close ===
    let landing_phase = dist < 8.0 && dy.abs() < 5.0;
    let thrust_frac = if landing_phase {
        // Fine control
        thrust_frac
    } else {
        thrust_frac
    };

    // Fuel emergency
    let thrust_frac = if s.fuel < 3.0 {
        0.2_f64.min(thrust_frac)
    } else {
        thrust_frac
    };

    let _ = dist;
    let _ = dist_vert;
    let _ = landing_phase;

    (thrust_frac, turn_cmd)
}
```

```rust
fn compute_control(s: &State, mi: &MapInfo) -> (f64, f64) {
    let g        = mi.gravity;
    let max_thr  = mi.max_thrust;
    let dx       = mi.target_x - s.x;
    let dy       = mi.target_y - s.y; // +ve = target is below

    // --- desired velocities ---
    // Horizontal: proportional to distance, capped
    let spd_h = (dx.abs() * 0.5).clamp(0.0, 7.0);
    let desired_vx = dx.signum() * spd_h * if dx.abs() > 0.5 { 1.0 } else { 0.0 };

    // Vertical: descend toward target, softly
    let desired_vy = if dy > 2.0 {
        (dy * 0.35).clamp(0.3, 4.5)
    } else if dy > 0.0 {
        dy * 0.4
    } else {
        -1.0 // rise if overshot below
    };

    // --- PD → desired accelerations ---
    let ax_des = ((desired_vx - s.vx) * 2.5).clamp(-max_thr * 0.65, max_thr * 0.65);
    let thr_y  = (g - (desired_vy - s.vy) * 2.0).clamp(0.05, max_thr);

    // --- desired angle ---
    // thrust_x = T*sin(a), thrust_y = T*cos(a)  (y-down, a=0 faces up)
    let desired_angle = ax_des.atan2(thr_y).clamp(-PI / 2.8, PI / 2.8);

    // --- thrust magnitude ---
    let total = (ax_des * ax_des + thr_y * thr_y).sqrt();
    let mut thrust = (total / max_thr).clamp(0.0, 1.0);

    // low-fuel override
    if s.fuel < 2.0 { thrust = thrust.min(0.15); }

    // --- angular control ---
    let a_err    = wrap_angle(desired_angle - s.angle);
    let want_av  = (a_err * 4.5).clamp(-3.5, 3.5);
    let turn_cmd = ((want_av - s.angular_vel) * 0.9).clamp(-1.0, 1.0);

    (thrust, turn_cmd)
}

fn main() {
    let state   = Arc::new(Mutex::new(State::default()));
    let mapinfo = Arc::new(Mutex::new(MapInfo::default()));

    let state_w  = Arc::clone(&state);
    let mapinfo_w = Arc::clone(&mapinfo);

    thread::spawn(move || {

        let h: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
        if h.len() < 5 { return; }
        let height    = h[1] as usize;
        let gravity   = h[2];
        let max_thr   = h[3];

        line.clear();
        if rdr.read_line(&mut line).unwrap_or(0) == 0 { return; }
        let c: Vec<f64> = line.split_whitespace().filter_map(|s| s.parse().ok()).collect();
        if c.len() < 4 { return; }

            let mut mi = mapinfo_w.lock().unwrap();

            mi.max_thrust = max_thr;
            mi.target_x  = c[2];
            mi.target_y  = c[3];
            mi.ready     = true;
        }

        // drain map rows
        for _ in 0..height {
            line.clear();
            if rdr.read_line(&mut line).unwrap_or(0) == 0 { break; }
        }

        loop {
            line.clear();
            match rdr.read_line(&mut line) {
                Ok(0) | Err(_) => break,
                Ok(_) => {}
            }

                s.vx = p[2]; s.vy = p[3];
                s.angle = p[4]; s.angular_vel = p[5];
                s.fuel = p[6]; s.valid = true;
            }
        }

        if mapinfo.lock().unwrap().ready { break; }
        thread::sleep(Duration::from_millis(5));

        if out.flush().is_err() { break; }
```

```rust
use std::f64::consts::PI;
use std::io::{self, BufRead, BufWriter, Write};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone)]
struct State {
    x: f64, y: f64,
    vx: f64, vy: f64,
    angle: f64, angular_vel: f64,
    fuel: f64,
    valid: bool,
}

impl Default for State {
    fn default() -> Self {
        Self { x:0., y:0., vx:0., vy:0., angle:0., angular_vel:0., fuel:100., valid:false }
    }
}

#[derive(Clone)]
struct Map {
    gravity: f64,
    max_thrust: f64,
    tx: f64,
    ty: f64,
    ready: bool,
}

impl Default for Map {
    fn default() -> Self {
        Self { gravity:1.6, max_thrust:10., tx:50., ty:40., ready:false }
    }
}

fn wrap(mut a: f64) -> f64 {
    while a >  PI { a -= 2.*PI; }
    while a < -PI { a += 2.*PI; }
    a
}

fn control(s: &State, m: &Map) -> (f64, f64) {
    let g   = m.gravity;
    let mt  = m.max_thrust;
    let dx  = m.tx - s.x;
    let dy  = m.ty - s.y;  // +ve = target below us

    // desired velocities
    let vx_d = (dx.signum() * (dx.abs() * 0.5).clamp(0., 7.))
        * if dx.abs() > 0.4 { 1. } else { 0. };
    let vy_d = if dy > 2.   { (dy * 0.35).clamp(0.3, 4.5) }
               else if dy > 0. { dy * 0.4 }
               else            { -1. };

    // desired accelerations
    let ax = ((vx_d - s.vx) * 2.5).clamp(-mt*0.65, mt*0.65);
    let ay = (g - (vy_d - s.vy) * 2.0).clamp(0.05, mt);

    // desired angle (angle=0 faces up; y-down frame)
    // thrust_x = T sin(a), thrust_y_up = T cos(a)
    let ang_d = ax.atan2(ay).clamp(-PI/2.8, PI/2.8);

    // thrust magnitude
    let mut thr = ((ax*ax + ay*ay).sqrt() / mt).clamp(0., 1.);
    if s.fuel < 2. { thr = thr.min(0.15); }

    // turn command
    let aerr = wrap(ang_d - s.angle);
    let av_d = (aerr * 4.5).clamp(-3.5, 3.5);
    let turn = ((av_d - s.angular_vel) * 0.9).clamp(-1., 1.);

    (thr, turn)
}

fn main() {
    let state = Arc::new(Mutex::new(State::default()));
    let map   = Arc::new(Mutex::new(Map::default()));

    let s2 = Arc::clone(&state);
    let m2 = Arc::clone(&map);

    thread::spawn(move || {
        let stdin = io::stdin();
        let mut r = stdin.lock();
        let mut line = String::new();

        // header: width height gravity max_thrust fuel
        line.clear();
        if r.read_line(&mut line).unwrap_or(0) == 0 { return; }
        let h: Vec<f64> = line.split_whitespace().filter_map(|x| x.parse().ok()).collect();
        if h.len() < 5 { return; }
        let rows   = h[1] as usize;
        let grav   = h[2];
        let mthrust = h[3];

        // start_x start_y target_x target_y
        line.clear();
        if r.read_line(&mut line).unwrap_or(0) == 0 { return; }
        let c: Vec<f64> = line.split_whitespace().filter_map(|x| x.parse().ok()).collect();
        if c.len() < 4 { return; }
        {
            let mut mg = m2.lock().unwrap();
            mg.gravity    = grav;
            mg.max_thrust = mthrust;
            mg.tx = c[2];
            mg.ty = c[3];
            mg.ready = true;
        }

        // drain map
        for _ in 0..rows {
            line.clear();
            if r.read_line(&mut line).unwrap_or(0) == 0 { break; }
        }

        // state updates
        loop {
            line.clear();
            match r.read_line(&mut line) {
                Ok(0) | Err(_) => break,
                Ok(_) => {}
            }
            let p: Vec<f64> = line.split_whitespace().filter_map(|x| x.parse().ok()).collect();
            if p.len() >= 7 {
                let mut sg = s2.lock().unwrap();
                sg.x=p[0]; sg.y=p[1]; sg.vx=p[2]; sg.vy=p[3];
                sg.angle=p[4]; sg.angular_vel=p[5]; sg.fuel=p[6];
                sg.valid=true;
            }
        }
    });

    // wait for map ready
    loop {
        if map.lock().unwrap().ready { break; }
        thread::sleep(Duration::from_millis(5));
    }

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let dt = Duration::from_millis(40);

    loop {
        let t0 = Instant::now();
        let s = state.lock().unwrap().clone();
        let m = map.lock().unwrap().clone();

        let (thr, turn) = if s.valid {
            control(&s, &m)
        } else {
            let h = (m.gravity / m.max_thrust).clamp(0., 1.);
            (h, 0.)
        };

        if writeln!(out, "{:.4} {:.4}", thr.clamp(0.,1.), turn.clamp(-1.,1.)).is_err() { break; }
        if out.flush().is_err() { break; }

        let el = t0.elapsed();
        if el < dt { thread::sleep(dt - el); }
    }
}
```
