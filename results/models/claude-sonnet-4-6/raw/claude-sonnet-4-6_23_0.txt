use std::io::{self, BufRead, BufReader, Write, BufWriter};
use std::sync::{Arc, Mutex};
use std::thread;
use std::time::{Duration, Instant};

#[derive(Clone)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel: f64,
}

#[derive(Clone)]
struct Params {
    gravity: f64,
    max_thrust: f64,
    target_x: f64,
    target_y: f64,
}

fn control(s: &State, p: &Params) -> (f64, f64) {
    let dx = p.target_x - s.x;
    let dy = p.target_y - s.y; // positive = target is below us

    // Vertical velocity target: safely descend towards landing pad
    // Braking capacity (net upward deceleration at near-vertical attitude)
    let max_brake = (p.max_thrust * 0.9 - p.gravity).max(0.5);
    let desired_vy = if dy > 0.0 {
        // Max safe descent speed from which we can brake to zero in distance dy
        let safe_max = (2.0 * max_brake * dy).sqrt().min(5.0);
        (dy * 0.4).min(safe_max)
    } else {
        // Below or at target height: rise gently
        -0.3
    };

    // PD thrust control for vertical velocity
    // Physics: dvy/dt = gravity - cos(angle)*thrust*max_thrust
    // We want: dvy/dt = (desired_vy - vy) * Kp
    // => thrust = (gravity - vy_err*Kp) / (cos(angle)*max_thrust)
    let vy_err = desired_vy - s.vy;
    let cos_a = s.angle.cos().max(0.2);
    let thrust = ((p.gravity - vy_err * 4.0) / (cos_a * p.max_thrust)).clamp(0.0, 1.0);

    // Horizontal velocity target: move towards target_x
    let target_vx = (dx * 0.5).clamp(-4.0, 4.0);
    let vx_err = target_vx - s.vx;

    // Desired tilt angle to generate horizontal acceleration
    // Physics: dvx/dt = sin(angle)*thrust*max_thrust = desired_ax
    // => angle = asin(desired_ax / (thrust*max_thrust))
    let thrust_force = (thrust * p.max_thrust).max(0.1);
    let desired_angle = (vx_err * 2.0 / thrust_force).clamp(-0.85, 0.85).asin();

    // PD turn control to achieve desired angle
    let turn = ((desired_angle - s.angle) * 4.0 - s.angular_vel * 0.7).clamp(-1.0, 1.0);

    (thrust, turn)
}

// Efficiently skip n lines from a BufReader without large allocations
fn skip_lines<R: io::Read>(reader: &mut BufReader<R>, n: usize) {
    let mut remaining = n;
    while remaining > 0 {
        let consumed = {
            let buf = match reader.fill_buf() {
                Ok(b) if !b.is_empty() => b,
                _ => return,
            };
            let mut count = 0usize;
            let mut end = buf.len();
            let mut done = false;
            for (i, &b) in buf.iter().enumerate() {
                if b == b'\n' {
                    count += 1;
                    if count == remaining {
                        end = i + 1;
                        done = true;
                        break;
                    }
                }
            }
            if done {
                remaining = 0;
            } else {
                remaining -= count;
            }
            end
        };
        reader.consume(consumed);
    }
}

fn main() {
    let shared_state: Arc<Mutex<Option<State>>> = Arc::new(Mutex::new(None));
    let shared_params: Arc<Mutex<Option<Params>>> = Arc::new(Mutex::new(None));

    let state_tx = Arc::clone(&shared_state);
    let params_tx = Arc::clone(&shared_params);

    // Reader thread: parse all stdin
    thread::spawn(move || {
        let stdin = io::stdin();
        // 1MB read buffer for efficient large-map I/O
        let mut reader = BufReader::with_capacity(1 << 20, stdin.lock());
        let mut line = String::new();

        // Parse header: width height gravity max_thrust fuel
        let (map_height, gravity, max_thrust) = loop {
            line.clear();
            if reader.read_line(&mut line).unwrap_or(0) == 0 {
                return;
            }
            let v: Vec<f64> = line
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if v.len() >= 5 {
                break (v[1] as usize, v[2], v[3]);
            }
        };

        // Parse start/target positions: start_x start_y target_x target_y
        let (target_x, target_y) = loop {
            line.clear();
            if reader.read_line(&mut line).unwrap_or(0) == 0 {
                return;
            }
            let v: Vec<f64> = line
                .split_whitespace()
                .filter_map(|s| s.parse().ok())
                .collect();
            if v.len() >= 4 {
                break (v[2], v[3]);
            }
        };

        // Params available before map is parsed so control loop can start immediately
        *params_tx.lock().unwrap() = Some(Params {
            gravity,
            max_thrust,
            target_x,
            target_y,
        });

        // Skip map rows without storing them (supports multi-GB maps)
        skip_lines(&mut reader, map_height);

        // Process state updates indefinitely
        loop {
            line.clear();
            match reader.read_line(&mut line) {
                Ok(0) | Err(_) => break,
                Ok(_) => {
                    let v: Vec<f64> = line
                        .split_whitespace()
                        .filter_map(|s| s.parse().ok())
                        .collect();
                    if v.len() >= 7 {
                        *state_tx.lock().unwrap() = Some(State {
                            x: v[0],
                            y: v[1],
                            vx: v[2],
                            vy: v[3],
                            angle: v[4],
                            angular_vel: v[5],
                            fuel: v[6],
                        });
                    }
                }
            }
        }
    });

    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    // Wait for params (map header must be parsed first)
    let deadline = Instant::now() + Duration::from_secs(120);
    loop {
        if shared_params.lock().unwrap().is_some() {
            break;
        }
        if Instant::now() > deadline {
            return;
        }
        thread::sleep(Duration::from_millis(5));
    }

    // Control output loop at ~25Hz (well above the 20Hz minimum requirement)
    loop {
        let t0 = Instant::now();

        let (thrust, turn) = {
            let params = shared_params.lock().unwrap().clone();
            let state = shared_state.lock().unwrap().clone();
            match (params, state) {
                (Some(p), Some(s)) => control(&s, &p),
                // State not yet received: output hover thrust at angle=0
                (Some(p), None) => ((p.gravity / p.max_thrust).clamp(0.0, 1.0), 0.0),
                _ => (0.5, 0.0),
            }
        };

        if writeln!(out, "{:.6} {:.6}", thrust, turn).is_err() {
            break;
        }
        if out.flush().is_err() {
            break;
        }

        // Sleep for remainder of 40ms period to maintain ~25Hz output
        let elapsed = t0.elapsed();
        let period = Duration::from_millis(40);
        if elapsed < period {
            thread::sleep(period - elapsed);
        }
    }
}
