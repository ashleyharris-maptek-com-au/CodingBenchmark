use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());
    let mut reader = io::BufReader::new(stdin.lock());

    let mut first_line = String::new();
    reader.read_line(&mut first_line).unwrap();
    let mut parts = first_line.split_whitespace();
    let h: usize = parts.next().unwrap().parse().unwrap();
    let w: usize = parts.next().unwrap().parse().unwrap();

    let total = h * w;
    let num_u64 = (total + 63) / 64;

    // Compact bitset for walls
    let mut walls = vec![0u64; num_u64];
    let mut start = 0usize;
    let mut end = 0usize;

    let mut buf = Vec::with_capacity(w + 4);
    for row in 0..h {
        buf.clear();
        reader.read_until(b'\n', &mut buf).unwrap();
        let len = buf.len();
        let line_len = if len > 0 && buf[len - 1] == b'\n' {
            if len > 1 && buf[len - 2] == b'\r' { len - 2 } else { len - 1 }
        } else {
            len
        };
        let base = row * w;
        for col in 0..w.min(line_len) {
            let idx = base + col;
            match buf[col] {
                b'#' => walls[idx >> 6] |= 1u64 << (idx & 63),
                b'A' => start = idx,
                b'B' => end = idx,
                _ => {}
            }
        }
    }

    if start == end {
        writeln!(out, "1").unwrap();
        writeln!(out, "{} {}", start % w, start / w).unwrap();
        return;
    }

    // BFS: parent[i] encodes which direction we came from:
    //   0 = unvisited, 255 = start
    //   1 = came from North (parent is curr - w)
    //   2 = came from South (parent is curr + w)
    //   3 = came from East  (parent is curr + 1)
    //   4 = came from West  (parent is curr - 1)
    let mut parent = vec![0u8; total];
    parent[start] = 255;

    let mut queue: VecDeque<u32> = VecDeque::with_capacity(65536);
    queue.push_back(start as u32);

    let mut found = false;

    'bfs: while let Some(curr_u32) = queue.pop_front() {
        let curr = curr_u32 as usize;
        let row = curr / w;
        let col = curr % w;

        // North
        if row > 0 {
            let next = curr - w;
            if walls[next >> 6] & (1u64 << (next & 63)) == 0 && parent[next] == 0 {
                parent[next] = 2; // came from South (curr is to its South)
                if next == end { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
        // South
        if row + 1 < h {
            let next = curr + w;
            if walls[next >> 6] & (1u64 << (next & 63)) == 0 && parent[next] == 0 {
                parent[next] = 1; // came from North
                if next == end { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
        // West
        if col > 0 {
            let next = curr - 1;
            if walls[next >> 6] & (1u64 << (next & 63)) == 0 && parent[next] == 0 {
                parent[next] = 3; // came from East
                if next == end { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
        // East
        if col + 1 < w {
            let next = curr + 1;
            if walls[next >> 6] & (1u64 << (next & 63)) == 0 && parent[next] == 0 {
                parent[next] = 4; // came from West
                if next == end { found = true; break 'bfs; }
                queue.push_back(next as u32);
            }
        }
    }

    if !found {
        writeln!(out, "0").unwrap();
        return;
    }

    // Reconstruct path from end back to start (stored in reverse order)
    let mut path_rev: Vec<u32> = Vec::new();
    let mut curr = end;
    loop {
        path_rev.push(curr as u32);
        if curr == start { break; }
        curr = match parent[curr] {
            1 => curr - w, // came from North
            2 => curr + w, // came from South
            3 => curr + 1, // came from East
            4 => curr - 1, // came from West
            _ => unreachable!(),
        };
    }

    writeln!(out, "{}", path_rev.len()).unwrap();
    for &idx in path_rev.iter().rev() {
        let idx = idx as usize;
        writeln!(out, "{} {}", idx % w, idx / w).unwrap();
    }
}
