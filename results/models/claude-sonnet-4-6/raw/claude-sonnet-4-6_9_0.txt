using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var stdin = new StreamReader(new BufferedStream(Console.OpenStandardInput(), 1 << 20));

        int n = ReadInt(stdin);
        int stockLength = ReadInt(stdin);

        int[] cuts = new int[n];
        for (int i = 0; i < n; i++)
            cuts[i] = ReadInt(stdin);

        // Sort indices by decreasing cut size (Best Fit Decreasing)
        int[] order = new int[n];
        for (int i = 0; i < n; i++) order[i] = i;
        Array.Sort(order, (a, b) => cuts[b].CompareTo(cuts[a]));

        // SortedSet<(remaining, binId)> — ValueTuple compares lexicographically
        var available = new SortedSet<(int rem, int bin)>();
        int[] assign = new int[n];
        int numBins = 0;

        foreach (int i in order)
        {
            int sz = cuts[i];

            bool found = false;
            (int rem, int bin) best = default;

            // Find bin with smallest remaining capacity >= sz
            foreach (var item in available.GetViewBetween((sz, int.MinValue), (stockLength, int.MaxValue)))
            {
                best = item;
                found = true;
                break;
            }

            int b, newRem;
            if (found)
            {
                available.Remove(best);
                b = best.bin;
                newRem = best.rem - sz;
            }
            else
            {
                b = numBins++;
                newRem = stockLength - sz;
            }

            assign[i] = b;
            if (newRem > 0) available.Add((newRem, b));
        }

        // Build linked lists per bin (iterate reverse so output is ascending index order)
        int[] binHead = new int[numBins];
        int[] next = new int[n];
        for (int i = 0; i < numBins; i++) binHead[i] = -1;

        for (int i = n - 1; i >= 0; i--)
        {
            int b = assign[i];
            next[i] = binHead[b];
            binHead[b] = i;
        }

        var stdout = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1 << 20));
        stdout.WriteLine(numBins);

        var sb = new StringBuilder(256);
        for (int b = 0; b < numBins; b++)
        {
            sb.Clear();
            int idx = binHead[b];
            while (idx != -1)
            {
                sb.Append(idx);
                int nx = next[idx];
                if (nx != -1) sb.Append(' ');
                idx = nx;
            }
            stdout.WriteLine(sb);
        }

        stdout.Flush();
    }

    static int ReadInt(StreamReader r)
    {
        int c = r.Read();
        while (c == ' ' || c == '\n' || c == '\r' || c == '\t') c = r.Read();
        int v = 0;
        while (c >= '0' && c <= '9')
        {
            v = v * 10 + (c - '0');
            c = r.Read();
        }
        return v;
    }
}
