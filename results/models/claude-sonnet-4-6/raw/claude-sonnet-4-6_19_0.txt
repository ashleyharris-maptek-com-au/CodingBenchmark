using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static void Main()
    {
        var stdin = new BufferedStream(Console.OpenStandardInput(), 1 << 20);

        // Read text header line
        var hSb = new StringBuilder();
        int c;
        while ((c = stdin.ReadByte()) != '\n' && c != -1)
            if (c != '\r') hSb.Append((char)c);

        var tok = hSb.ToString().Trim().Split(new char[] { ' ' }, StringSplitOptions.RemoveEmptyEntries);
        int XC = int.Parse(tok[0]);
        int YC = int.Parse(tok[1]);
        int ZC = int.Parse(tok[2]);

        int N = XC * YC;
        int[] surZ = new int[N];
        byte[] surV = new byte[N];
        for (int i = 0; i < N; i++) surZ[i] = -1;

        // Stream binary voxels, track highest solid voxel (surface) per column
        // z iterates 0..ZC-1 low-to-high; last overwrite = highest solid = surface
        byte[] ibuf = new byte[65536];
        int ip = 0, il = 0;

        for (int x = 0; x < XC; x++)
        for (int y = 0; y < YC; y++)
        {
            int si = x * YC + y;
            for (int z = 0; z < ZC; z++)
            {
                if (ip >= il)
                {
                    il = stdin.Read(ibuf, 0, ibuf.Length);
                    ip = 0;
                    if (il == 0) goto doneRead;
                }
                byte vox = ibuf[ip++];
                if ((vox & 0xF) != 0) { surZ[si] = z; surV[si] = vox; }
            }
        }
        doneRead:;

        // Ensure processing plant origin (0,0) has accessible terrain
        if (surZ[0] < 0) surZ[0] = 0;

        // Reverse Dijkstra from (0,0):
        // dist[i] = min per-unit-mass fuel cost to transport ore from cell i to (0,0)
        // Forward edge A->B cost: 5 if surZ[B]>surZ[A] (uphill), 1 if equal (flat), 0 if lower (downhill)
        // Slope constraint: |surZ[B]-surZ[A]| <= 1
        // In reverse Dijkstra at node cur exploring neighbor ni:
        //   cost = forward edge cost of ni->cur = 5 if cz>nz, 1 if cz==nz, 0 if cz<nz
        int[] dist = new int[N];
        int[] prv = new int[N];
        for (int i = 0; i < N; i++) { dist[i] = int.MaxValue; prv[i] = -1; }
        dist[0] = 0;

        var pq = new PriorityQueue<int, int>();
        pq.Enqueue(0, 0);

        int[] DX = { 1, -1, 0, 0 };
        int[] DY = { 0, 0, 1, -1 };

        while (pq.Count > 0)
        {
            pq.TryDequeue(out int cur, out int d);
            if (d > dist[cur]) continue;
            int cz = surZ[cur];
            if (cz < 0) continue;
            int cx = cur / YC, cy = cur % YC;

            for (int dir = 0; dir < 4; dir++)
            {
                int nx = cx + DX[dir], ny = cy + DY[dir];
                if ((uint)nx >= (uint)XC || (uint)ny >= (uint)YC) continue;
                int ni = nx * YC + ny;
                int nz = surZ[ni];
                if (nz < 0) continue;
                int dz = cz - nz; // positive: cur is higher than neighbor
                if (dz > 1 || dz < -1) continue; // slope constraint
                int sc = dz > 0 ? 5 : dz == 0 ? 1 : 0;
                int nd = d + sc;
                if (nd < dist[ni]) { dist[ni] = nd; prv[ni] = cur; pq.Enqueue(ni, nd); }
            }
        }

        // Find profitable surface ore deposits
        // Revenue per op = 1000 * mass * val; Fuel = mass * dist[i]
        // Profitable iff 1000 * val > dist[i] (mass > 0 so cancel it)
        var ores = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (surZ[i] < 0) continue;
            byte v = surV[i];
            int mass = v & 0xF, val = (v >> 4) & 0xF;
            if (val == 0 || mass == 0 || dist[i] == int.MaxValue) continue;
            if (1000 * val > dist[i]) ores.Add(i);
        }

        // Sort by total profit descending
        ores.Sort((a, b) =>
        {
            byte va = surV[a], vb = surV[b];
            long pa = (long)(va & 0xF) * (1000 * ((va >> 4) & 0xF) - dist[a]);
            long pb = (long)(vb & 0xF) * (1000 * ((vb >> 4) & 0xF) - dist[b]);
            return pb.CompareTo(pa);
        });

        // Build all operation lines then output count + lines
        var opBuf = new StringBuilder(1 << 20);
        int opCnt = 0;

        foreach (int si in ores)
        {
            int oz = surZ[si], ox = si / YC, oy = si % YC;

            // Reconstruct path from (ox,oy) to (0,0) by following prv[] chain
            // prv[v] = next node on cheapest path from v toward (0,0)
            var path = new List<(int dx, int dy)>();
            int cur = si;
            bool ok = true;
            while (cur != 0)
            {
                int nxt = prv[cur];
                if (nxt < 0) { ok = false; break; }
                path.Add((nxt / YC - cur / YC, nxt % YC - cur % YC));
                cur = nxt;
            }
            if (!ok) continue;

            // Output format: N digX digY digZ stepX1 stepY1 ... stepXN stepYN destX destY
            opBuf.Append(path.Count).Append(' ')
                 .Append(ox).Append(' ').Append(oy).Append(' ').Append(oz);
            foreach (var (pdx, pdy) in path)
                opBuf.Append(' ').Append(pdx).Append(' ').Append(pdy);
            opBuf.Append(" 0 0\n");
            opCnt++;
        }

        using var stdout = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1 << 20));
        stdout.Write(opCnt);
        stdout.Write('\n');
        stdout.Write(opBuf);
        stdout.Flush();
    }
}
