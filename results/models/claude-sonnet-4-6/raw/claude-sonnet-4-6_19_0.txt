```csharp
    static bool IsConnectedToGround(int startX, int startY, int startZ)
    {

        if (!present[startX, startY, startZ]) return false;

        var visited = new HashSet<(int, int, int)>();
        var queue = new Queue<(int, int, int)>();
        queue.Enqueue((startX, startY, startZ));
        visited.Add((startX, startY, startZ));

        int[] dy = { 0, 0, 1, -1, 0, 0 };
        int[] dz = { 0, 0, 0, 0, 1, -1 };

        {
            var (cx, cy, cz) = queue.Dequeue();
            if (cz == 0) return true;

            {

                int ny = cy + dy[d];

    static double MoveCost(List<(int x, int y)> path, int startZ, int mass)
    {
        if (path.Count == 0) return 0;
        double cost = 0;
        int curZ = startZ;
        int prevX = -1, prevY = -1;
        
        // Get surface Z along path
        for (int i = 0; i < path.Count; i++)
        {
            int nx = path[i].x;
            int ny = path[i].y;
            int nz = surface[nx, ny];
            if (nz < 0) nz = 0;
            
            int dz = nz - curZ;
            if (dz > 0)
            {
                // Uphill: cost is $5 per unit mass per step * slope factor
                // Actually: $5 per unit mass to go up 10% slope = 1Z/10XY
                // Here Z voxel = 1/10 of X/Y voxel in real size
                // So 1 Z voxel per X/Y voxel = 10% slope
                // Cost: $1 flat, $5 for 10% slope per mass per XY step
                // The slope here is dz * ZBlockSize / XBlockSize per step
                // With 10x Z precision: ZBlockSize ~ XBlockSize/10
                // So 1 Z voxel per 1 XY voxel = ZBlockSize/XBlockSize = 10% slope
                // Cost = mass * (1 + (dz * 4)) approximately, but let's use $5 for uphill
                cost += mass * 5.0 * Math.Abs(dz); // simplified
            }
            else
            {
                cost += mass * 1.0; // flat cost per XY step
            }
            curZ = nz;
        }
        return cost;
    }
    
    // Find path from (sx,sy) to (dx,dy) respecting slope constraint
    // Uses A* with slope constraint
    static List<(int x, int y)> FindPath(int sx, int sy, int sz, int dx, int dy)
    {
        if (sx == dx && sy == dy) return new List<(int, int)>();
        
        // A* search
        var dist = new Dictionary<(int, int, int), double>();
        var prev = new Dictionary<(int, int, int), (int, int, int)>();
        var pq = new SortedSet<(double cost, int x, int y, int z)>(
            Comparer<(double, int, int, int)>.Create((a, b) => {
                int c = a.Item1.CompareTo(b.Item1);
                if (c != 0) return c;
                c = a.Item2.CompareTo(b.Item2);
                if (c != 0) return c;
                c = a.Item3.CompareTo(b.Item3);
                if (c != 0) return c;
                return a.Item4.CompareTo(b.Item4);
            }));
        
        var startKey = (sx, sy, sz);
        dist[startKey] = 0;
        pq.Add((0, sx, sy, sz));
        
        int[] ddx = { 1, -1, 0, 0 };
        int[] ddy = { 0, 0, 1, -1 };
        
        while (pq.Count > 0)
        {
            var (curCost, cx, cy, cz) = pq.Min;
            pq.Remove(pq.Min);
            
            if (cx == dx && cy == dy)
            {
                // Reconstruct path
                var path = new List<(int, int)>();
                var cur = (cx, cy, cz);
                while (cur != startKey)
                {
                    path.Add((cur.Item1, cur.Item2));
                    cur = prev[cur];
                }
                path.Reverse();
                return path;
            }
            
            var curKey = (cx, cy, cz);
            if (dist.TryGetValue(curKey, out double d) && d < curCost) continue;
            
            for (int dir = 0; dir < 4; dir++)
            {
                int nx = cx + ddx[dir];
                int ny = cy + ddy[dir];
                if (nx < 0 || nx >= XCount || ny < 0 || ny >= YCount) continue;
                
                int nz = surface[nx, ny];
                if (nz < 0) nz = 0;
                
                // Check slope constraint
                int dzAbs = Math.Abs(nz - cz);
                if (dzAbs > 1) continue; // too steep
                
                double stepCost;
                if (nz > cz) stepCost = 5.0; // uphill
                else stepCost = 1.0; // flat or downhill
                
                double newCost = curCost + stepCost;
                double heuristic = Math.Abs(nx - dx) + Math.Abs(ny - dy);
                
                var nKey = (nx, ny, nz);
                if (!dist.TryGetValue(nKey, out double nd) || newCost < nd)
                {
                    dist[nKey] = newCost;
                    prev[nKey] = curKey;
                    pq.Add((newCost + heuristic, nx, ny, nz));
                }
            }
        }
        
        return null; // no path found
    }
    
    // Find best dump location for waste
    // Should be accessible and not block future mining
    static (int x, int y) FindDumpLocation(int fromX, int fromY, int fromZ)
    {
        // Look for a location that is accessible and far from valuable ore
        // Simple: look for edges of the map or already low-value areas
        
        // Try to find a nearby flat area that won't interfere
        int bestX = -1, bestY = -1;
        double bestScore = double.MaxValue;
        
        // Search in expanding rings
        for (int r = 1; r < Math.Max(XCount, YCount); r++)
        {
            for (int dx = -r; dx <= r; dx++)
            {
                for (int dy = -r; dy <= r; dy++)
                {
                    if (Math.Abs(dx) != r && Math.Abs(dy) != r) continue;
                    int nx = fromX + dx;
                    int ny = fromY + dy;
                    if (nx < 0 || nx >= XCount || ny < 0 || ny >= YCount) continue;
                    
                    // Skip (0,0) processing plant
                    if (nx == 0 && ny == 0) continue;
                    
                    int nz = surface[nx, ny];
                    // Find path
                    var path = FindPath(fromX, fromY, fromZ, nx, ny);
                    if (path != null)
                    {
                        double score = path.Count + (nz < 0 ? 0 : nz);
                        if (score < bestScore)
                        {
                            bestScore = score;
                            bestX = nx;
                            bestY = ny;
                        }
                        goto foundDump;
                    }
                }
            }
            if (bestX >= 0) break;
        }
        foundDump:
        
        if (bestX < 0)
        {
            // Fallback: use corner
            bestX = XCount - 1;
            bestY = YCount - 1;
        }
        
        return (bestX, bestY);
    }
    
    // Main greedy strategy: process most valuable ore first
    static void RunGreedyStrategy()
    {
        // Collect all voxels with value > 0
        var valuableVoxels = new List<(int x, int y, int z, int mass, int value, double profit)>();
        
        for (int x = 0; x < XCount; x++)
        {
            for (int y = 0; y < YCount; y++)
            {
                for (int z = 0; z < ZCount; z++)
                {
                    if (present[x, y, z])
                    {
                        byte d = voxelData[x, y, z];
                        int mass = GetMass(d);
                        int val = GetValue(d);
                        if (val > 0 && mass > 0)
                        {
                            double gross = 1000.0 * mass * val;
                            valuableVoxels.Add((x, y, z, mass, val, gross));
                        }
                    }
                }
            }
        }
        
        // Sort by value descending
        valuableVoxels.Sort((a, b) => b.profit.CompareTo(a.profit));
        
        foreach (var (vx, vy, vz, mass, val, profit) in valuableVoxels)
        {
            if (!present[vx, vy, vz]) continue;
            
            // Need to expose this voxel - remove voxels above it first
            // Then move it to (0,0)
            bool exposed = TryExposeAndMine(vx, vy, vz, true);
            if (!exposed) continue;
        }
    }

    static bool TryExposeAndMine(int tx, int ty, int tz, bool sendToPlant)
    {
        // First check if already exposed
        // If not, need to dig down from surface
        
        int curSurf = surface[tx, ty];
        
        // Remove voxels above target
        for (int z = curSurf; z > tz; z--)
        {
            if (!present[tx, ty, z]) continue;
            
            byte d = voxelData[tx, ty, z];
            int mass = GetMass(d);
            int val = GetValue(d);
            
            // This is a waste voxel or lower-value - dump it nearby
            bool digResult = DigAndDump(tx, ty, z, false);
            if (!digResult) return false;
        }
        
        // Now mine the target voxel
        if (!present[tx, ty, tz]) return false;
        if (!IsExposed(tx, ty, tz)) return false;
        
        return DigAndDump(tx, ty, tz, sendToPlant);
    }
    
    static bool DigAndDump(int dx, int dy, int dz, bool toPlant)
    {
        if (!present[dx, dy, dz]) return false;
        if (!IsExposed(dx, dy, dz)) return false;
        
        byte d = voxelData[dx, dy, dz];
        int mass = GetMass(d);
        int val = GetValue(d);
        
        int destX, destY;
        List<(int x, int y)> path;
        
        if (toPlant && val > 0)
        {
            destX = 0;
            destY = 0;
        }
        else
        {
            // Find dump location
            var dump = FindDumpSite(dx, dy, dz);
            if (dump == (-1, -1)) return false;
            destX = dump.Item1;
            destY = dump.Item2;
        }
        
        // Find path from dig location to destination
        path = FindPath(dx, dy, dz, destX, destY);
        if (path == null) 
        {
            // Try to find alternative
            if (!toPlant || val == 0)
            {
                var alt = FindDumpSite(dx, dy, dz);
                if (alt != (-1, -1))
                {
                    destX = alt.Item1;
                    destY = alt.Item2;
                    path = FindPath(dx, dy, dz, destX, destY);
                }
            }
            if (path == null) return false;
        }
        
        // Record operation
        var op = new Operation
        {
            digX = dx,
            digY = dy,
            digZ = dz,
            steps = path,
            destX = destX,
            destY = destY
        };
        operations.Add(op);
        
        // Update world state
        present[dx, dy, dz] = false;
        voxelData[dx, dy, dz] = 0;
        
        // Update surface for dig location
        if (surface[dx, dy] == dz)
        {
            surface[dx, dy] = -1;
            for (int z = dz - 1; z >= 0; z--)
            {
                if (present[dx, dy, z])
                {
                    surface[dx, dy] = z;
                    break;
                }
            }
        }
        
        // Update destination (unless plant)
        if (!(destX == 0 && destY == 0 && val > 0))
        {
            // Dump on top of destination
            int dstSurf = surface[destX, destY];
            int newZ = (dstSurf >= 0) ? dstSurf + 1 : 0;

                voxelData[destX, destY, newZ] = d;
```

```csharp
            }
        }

    static bool Exposed(int x, int y, int z)
    {

        if (x == 0 || !present[x-1, y, z]) return true;

    // 4-connected path on XY plane with slope <= 1 Z per step
    // Returns list of (stepX, stepY) Manhattan steps, or null if unreachable
    static List<(int sx, int sy)> AStarPath(int x0, int y0, int z0, int xd, int yd)
    {
        if (x0 == xd && y0 == yd) return new List<(int,int)>();
        
        // State: (x, y, z) where z is current height on surface
        // We navigate on the surface
        
        // Use (x,y) -> best known cost
        var gCost = new float[XN, YN];
        for (int x = 0; x < XN; x++) for (int y = 0; y < YN; y++) gCost[x, y] = float.MaxValue;
        gCost[x0, y0] = 0;
        
        var prevXY = new (int px, int py)[XN, YN];
        for (int x = 0; x < XN; x++) for (int y = 0; y < YN; y++) prevXY[x, y] = (-1, -1);
        
        // Priority queue: (f, x, y)
        var pq = new PriorityQueue<(int x, int y), float>();
        pq.Enqueue((x0, y0), MHDist(x0, y0, xd, yd));
        
        int[] ddx = {1,-1,0,0};
        int[] ddy = {0,0,1,-1};
        
        while (pq.Count > 0)
        {
            var (cx, cy) = pq.Dequeue();
            float cg = gCost[cx, cy];
            int cz = surfZ[cx, cy]; if (cz < 0) cz = 0;
            
            if (cx == xd && cy == yd)
            {
                // Reconstruct
                var path = new List<(int,int)>();
                int rx = xd, ry = yd;
                while (rx != x0 || ry != y0)
                {
                    var (px, py) = prevXY[rx, ry];
                    path.Add((rx - px, ry - py));
                    rx = px; ry = py;
                }
                path.Reverse();
                return path;
            }
            
            for (int d = 0; d < 4; d++)
            {
                int nx = cx + ddx[d];
                int ny = cy + ddy[d];
                if (nx < 0 || nx >= XN || ny < 0 || ny >= YN) continue;
                int nz = surfZ[nx, ny]; if (nz < 0) nz = 0;
                
                int dz = nz - cz;
                if (Math.Abs(dz) > 1) continue; // slope too steep
                
                float step = (dz > 0) ? 5f : 1f;
                float ng = cg + step;
                if (ng < gCost[nx, ny])
                {
                    gCost[nx, ny] = ng;
                    prevXY[nx, ny] = (cx, cy);
                    float h = MHDist(nx, ny, xd, yd);
                    pq.Enqueue((nx, ny), ng + h);
                }
            }
        }
        return null; // unreachable
    }
    
    static float MHDist(int x1, int y1, int x2, int y2) => Math.Abs(x1-x2) + Math.Abs(y1-y2);

        rawData[dx, dy, dz] = 0;

            surfZ[dx, dy] = -1;
            for (int z = dz - 1; z >= 0; z--)

                surfZ[destX, destY] = newZ;

    // Check if removing voxel at (x,y,z) disconnects anything
    // Conservative: only remove if it's the topmost or has no voxel directly above it
    static bool CanSafelyRemove(int x, int y, int z)
    {
        if (!present[x, y, z]) return false;
        if (!Exposed(x, y, z)) return false;
        
        // If it's the surface voxel, definitely safe
        if (surfZ[x, y] == z) return true;
        
        // Check if voxel above exists
        if (z + 1 < ZN && present[x, y, z + 1])
        {
            // Voxel above - check if it has other support via sides
            int ax = x, ay = y, az = z + 1;
            // Does voxel above have neighbor that connects it to ground?
            // Simple: does it have a neighbor at same z level that's also present?
            int[] ddx = {1,-1,0,0};
            int[] ddy = {0,0,1,-1};
            bool hasSideNeighbor = false;
            for (int d = 0; d < 4; d++)
            {
                int nx = ax + ddx[d];
                int ny = ay + ddy[d];
                if (nx >= 0 && nx < XN && ny >= 0 && ny < YN && present[nx, ny, az])
                {
                    hasSideNeighbor = true;
                    break;
                }
            }
            if (!hasSideNeighbor)
            {
                // The voxel above only has us as support - removing us disconnects it
                // Unless it's at z=0 (impossible if z+1 above us means z >= 0)
                return false;
            }
        }
        return true;

    static (int x, int y) FindBestDumpSite(int fx, int fy, int fz)
    {
        // Find reachable location not at (0,0), not blocking critical path
        // Simple: find nearest reachable cell with lowest surface height
        var visited = new HashSet<long>();
        var queue = new Queue<(int x, int y)>();
        
        void Enq(int x, int y) {
            long k = (long)x * YN + y;
            if (!visited.Contains(k)) { visited.Add(k); queue.Enqueue((x,y)); }
        }
        
        Enq(fx, fy);
        int[] ddx = {1,-1,0,0};
        int[] ddy = {0,0,1,-1};
        
        int fzSurf = surfZ[fx, fy]; if (fzSurf < 0) fzSurf = fz;
        
        var candidates = new List<(int dist, int surfH, int x, int y)>();
        
        while (queue.Count > 0 && visited.Count < 500)
        {
            var (cx, cy) = queue.Dequeue();
            int cz = surfZ[cx, cy]; if (cz < 0) cz = 0;
            
            if (!(cx == fx && cy == fy) && !(cx == 0 && cy == 0))
            {
                int dist = Math.Abs(cx - fx) + Math.Abs(cy - fy);
                candidates.Add((dist, cz, cx, cy));
            }
            
            for (int d = 0; d < 4; d++)
            {
                int nx = cx + ddx[d];
                int ny = cy + ddy[d];
                if (nx < 0 || nx >= XN || ny < 0 || ny >= YN) continue;
                int nz = surfZ[nx, ny]; if (nz < 0) nz = 0;
                if (Math.Abs(nz - cz) <= 1) Enq(nx, ny);
            }
        }
        
        if (candidates.Count == 0) return (-1, -1);
        
        // Pick lowest surface + farthest from (0,0)
        candidates.Sort((a, b) => {
            int sa = a.surfH, sb = b.surfH;
            if (sa != sb) return sa.CompareTo(sb);
            return a.dist.CompareTo(b.dist);
        });
        
        // Return first that has a valid path
        foreach (var (dist, h, cx, cy) in candidates)
        {
            var path = AStarPath(fx, fy, fzSurf, cx, cy);
            if (path != null) return (cx, cy);
        }
        return (-1, -1);
    }
    
    static void Solve()
    {
        long totalVoxels = (long)XN * YN * ZN;
        bool isSmall = totalVoxels <= 2_000_000;
        
        // Gather candidates: voxels with value > 0
        var candidates = new List<(double score, int x, int y, int z)>();
        
        int maxSearch = isSmall ? XN : Math.Min(XN, 100);
        int maxSearchY = isSmall ? YN : Math.Min(YN, 100);
        
        for (int x = 0; x < maxSearch; x++)
        {
            for (int y = 0; y < maxSearchY; y++)
            {
                for (int z = 0; z < ZN; z++)
                {
                    if (!present[x, y, z]) continue;
                    byte d = rawData[x, y, z];
                    int mass = d & 0xF;
                    int val = d >> 4;
                    if (val > 0 && mass > 0)
                    {
                        double gross = 1000.0 * mass * val;
                        double estMoveCost = (x + y) * mass * 1.5;
                        double net = gross - estMoveCost;
                        if (net > 0 || isSmall)
                            candidates.Add((gross, x, y, z));
                    }
                }
            }
        }
        
        // Sort by score descending
        candidates.Sort((a, b) => b.score.CompareTo(a.score));
        
        int maxOps = isSmall ? candidates.Count : Math.Min(candidates.Count, 10000);
        
        foreach (var (score, tx, ty, tz) in candidates.Take(maxOps))
        {
            if (!present[tx, ty, tz]) continue;
            
            byte d = rawData[tx, ty, tz];
            int mass = d & 0xF;
            int val = d >> 4;
            if (val == 0 || mass == 0) continue;
            
            // Need to expose target: remove voxels above it
            int curTop = surfZ[tx, ty];
            bool canReach = true;
            
            for (int z = curTop; z > tz && canReach; z--)
            {
                if (!present[tx, ty, z]) continue;
                
                // Remove this waste voxel
                if (!CanSafelyRemove(tx, ty, z))
                {
                    canReach = false;
                    break;
                }
                
                // Find path to dump site
                var (dx, dy) = FindBestDumpSite(tx, ty, z);
                if (dx < 0) { canReach = false; break; }
                
                int sz = surfZ[tx, ty]; if (sz < 0) sz = z;
                var path = AStarPath(tx, ty, sz, dx, dy);
                if (path == null) { canReach = false; break; }
                
                PerformDig(tx, ty, z, dx, dy, path);
            }
            
            if (!canReach) continue;
            if (!present[tx, ty, tz]) continue;
            if (!Exposed(tx, ty, tz)) continue;
            if (!CanSafelyRemove(tx, ty, tz)) continue;
            
            // Mine to plant
            int srcZ = surfZ[tx, ty]; if (srcZ < 0) srcZ = tz;
            var pathToPlant = AStarPath(tx, ty, srcZ, 0, 0);
            if (pathToPlant == null) continue;
            
            PerformDig(tx, ty, tz, 0, 0, pathToPlant);
        }
    }

    {
        var sb = new StringBuilder();
        sb.AppendLine(outputOps.Count.ToString());
        foreach (var op in outputOps) sb.AppendLine(op);
        Console.Write(sb);
    }
```

```csharp
    class MinHeap<T>
    {
        List<(float pri, T val)> heap = new List<(float, T)>();
        public int Count => heap.Count;
        
        public void Push(float priority, T value)
        {
            heap.Add((priority, value));
            int i = heap.Count - 1;
            while (i > 0)
            {
                int parent = (i - 1) / 2;
                if (heap[parent].pri <= heap[i].pri) break;
                var tmp = heap[parent]; heap[parent] = heap[i]; heap[i] = tmp;
                i = parent;
            }
        }
        
        public (float pri, T val) Pop()
        {
            var top = heap[0];
            int last = heap.Count - 1;
            heap[0] = heap[last];
            heap.RemoveAt(last);
            int i = 0;
            while (true)
            {
                int left = 2*i+1, right = 2*i+2, smallest = i;
                if (left < heap.Count && heap[left].pri < heap[smallest].pri) smallest = left;
                if (right < heap.Count && heap[right].pri < heap[smallest].pri) smallest = right;
                if (smallest == i) break;
                var tmp = heap[smallest]; heap[smallest] = heap[i]; heap[i] = tmp;
                i = smallest;

            return top;

            if ((d & 0xF) == 0 || (d >> 4) == 0) continue;
            
            // Remove overburden
            int top = surfZ[tx, ty];
            bool ok = true;
            for (int z = top; z > tz && ok; z--)
            {
                if (!present[tx, ty, z]) continue;
                if (!SafeToRemove(tx, ty, z)) { ok = false; break; }
                
                var (dx, dy) = PickDumpSite(tx, ty);
                if (dx < 0) { ok = false; break; }
                
                var path = FindPath(tx, ty, dx, dy);
                if (path == null) { ok = false; break; }
                
                DoMove(tx, ty, z, dx, dy, path);
            }
            
            if (!ok || !present[tx, ty, tz]) continue;
            if (!SafeToRemove(tx, ty, tz)) continue;
            
            var plantPath = FindPath(tx, ty, 0, 0);
            if (plantPath == null) continue;
            
            DoMove(tx, ty, tz, 0, 0, plantPath);
        }
    }
    
    static void PrintOutput()
    {
        var sb = new StringBuilder();
        sb.Append(ops.Count).Append('\n');
        foreach (var op in ops) sb.Append(op).Append('\n');
        Console.Write(sb);
    }
}
```

```csharp
using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Mining
{
    static int XN, YN, ZN;
    static double XBS, YBS, ZBS;
    static byte[,,] raw;
    static bool[,,] has;
    static int[,] top; // surface Z per (x,y), -1 if empty
    
    static readonly int[] DX = {1,-1,0,0};
    static readonly int[] DY = {0,0,1,-1};
    
    static List<string> opLines = new List<string>();
    
    static void Main()
    {
        ReadInput();
        ComputeSurface();
        RunMining();
        EmitOutput();
    }
    
    static void ReadInput()
    {
        var sin = Console.OpenStandardInput();
        var hdr = new List<byte>(128);
        int b;
        while ((b = sin.ReadByte()) >= 0 && b != '\n')
            if (b != '\r') hdr.Add((byte)b);

            .Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        XN = int.Parse(tok[0]); YN = int.Parse(tok[1]); ZN = int.Parse(tok[2]);
        XBS = double.Parse(tok[3]); YBS = double.Parse(tok[4]); ZBS = double.Parse(tok[5]);

        has = new bool[XN, YN, ZN];
        
        for (int x = 0; x < XN; x++)
        for (int y = 0; y < YN; y++)
        for (int z = 0; z < ZN; z++)
        {
            int v = sin.ReadByte(); if (v < 0) v = 0;
            raw[x,y,z] = (byte)v;
            has[x,y,z] = v != 0;
        }
    }
    
    static void ComputeSurface()
    {
        top = new int[XN, YN];
        for (int x = 0; x < XN; x++)
        for (int y = 0; y < YN; y++)
        {
            top[x,y] = -1;
            for (int z = ZN-1; z >= 0; z--)
                if (has[x,y,z]) { top[x,y] = z; break; }
        }
    }
    
    static bool Exposed(int x, int y, int z)
    {
        if (!has[x,y,z]) return false;
        if (z+1 >= ZN || !has[x,y,z+1]) return true;
        for (int d = 0; d < 4; d++)
        {
            int nx = x+DX[d], ny = y+DY[d];
            if (nx < 0 || nx >= XN || ny < 0 || ny >= YN) return true;
            if (!has[nx,ny,z]) return true;
        }
        return false;
    }
    
    static bool SafeRemove(int x, int y, int z)
    {
        if (!has[x,y,z] || !Exposed(x,y,z)) return false;
        if (top[x,y] == z) return true;
        if (z+1 < ZN && has[x,y,z+1])
        {
            for (int d = 0; d < 4; d++)
            {
                int nx = x+DX[d], ny = y+DY[d];
                if (nx >= 0 && nx < XN && ny >= 0 && ny < YN && has[nx,ny,z+1])
                    return true;
            }
            return false;
        }
        return true;
    }
    
    // Min-heap
    struct HeapItem : IComparable<HeapItem>
    {
        public float f;
        public int x, y;
        public int CompareTo(HeapItem o) => f.CompareTo(o.f);
    }
    
    static List<HeapItem> _heap = new List<HeapItem>(4096);
    
    static void HeapPush(float f, int x, int y)
    {
        _heap.Add(new HeapItem{f=f,x=x,y=y});
        int i = _heap.Count-1;
        while (i > 0) {
            int p = (i-1)>>1;
            if (_heap[p].f <= _heap[i].f) break;
            var t = _heap[p]; _heap[p] = _heap[i]; _heap[i] = t;
            i = p;
        }
    }
    
    static HeapItem HeapPop()
    {
        var r = _heap[0];
        int last = _heap.Count-1;
        _heap[0] = _heap[last]; _heap.RemoveAt(last);
        int i = 0, n = _heap.Count;
        while (true) {
            int l = 2*i+1, ri2 = 2*i+2, s = i;
            if (l < n && _heap[l].f < _heap[s].f) s = l;
            if (ri2 < n && _heap[ri2].f < _heap[s].f) s = ri2;
            if (s == i) break;
            var t = _heap[s]; _heap[s] = _heap[i]; _heap[i] = t;
            i = s;
        }
        return r;
    }
    
    // Reusable arrays for A*
    static float[] gArr;
    static int[] prevDir;
    static bool[] inClose;
    
    static void EnsureArrays()
    {
        int sz = XN * YN;
        if (gArr == null || gArr.Length < sz)
        {
            gArr = new float[sz];
            prevDir = new int[sz];
            inClose = new bool[sz];
        }
    }
    
    static List<(int sx,int sy)> FindPath(int x0, int y0, int xd, int yd)
    {
        if (x0 == xd && y0 == yd) return new List<(int,int)>();
        EnsureArrays();
        int sz = XN * YN;
        
        // Init
        for (int i = 0; i < sz; i++) { gArr[i] = float.MaxValue; inClose[i] = false; prevDir[i] = -1; }
        _heap.Clear();
        
        int s0 = x0*YN+y0;
        gArr[s0] = 0;
        float h0 = Math.Abs(x0-xd) + Math.Abs(y0-yd);
        HeapPush(h0, x0, y0);
        
        while (_heap.Count > 0)
        {
            var cur = HeapPop();
            int cx = cur.x, cy = cur.y;
            int ci = cx*YN+cy;
            if (inClose[ci]) continue;
            inClose[ci] = true;
            
            if (cx == xd && cy == yd)
            {
                var path = new List<(int,int)>(Math.Abs(xd-x0)+Math.Abs(yd-y0)+4);
                int rx = xd, ry = yd;
                while (rx != x0 || ry != y0)
                {
                    int dir = prevDir[rx*YN+ry];
                    path.Add((DX[dir], DY[dir]));
                    rx -= DX[dir]; ry -= DY[dir];
                }
                path.Reverse();
                return path;
            }
            
            float cg = gArr[ci];
            int cz = top[cx,cy]; if (cz < 0) cz = 0;
            
            for (int d = 0; d < 4; d++)
            {
                int nx = cx+DX[d], ny = cy+DY[d];
                if (nx < 0 || nx >= XN || ny < 0 || ny >= YN) continue;
                int ni = nx*YN+ny;
                if (inClose[ni]) continue;
                int nz = top[nx,ny]; if (nz < 0) nz = 0;
                if (Math.Abs(nz-cz) > 1) continue;
                
                float step = (nz > cz) ? 5f : 1f;
                float ng = cg + step;
                if (ng < gArr[ni])
                {
                    gArr[ni] = ng;
                    prevDir[ni] = d;
                    float h = Math.Abs(nx-xd) + Math.Abs(ny-yd);
                    HeapPush(ng+h, nx, ny);
                }
            }
        }
        return null;
    }
    
    static void Perform(int x, int y, int z, int dx, int dy, List<(int,int)> path)
    {
        byte d = raw[x,y,z];
        int val = d >> 4;
        
        var sb = new StringBuilder();
        sb.Append(path.Count).Append(' ')
          .Append(x).Append(' ').Append(y).Append(' ').Append(z);
        foreach (var (sx,sy) in path) sb.Append(' ').Append(sx).Append(' ').Append(sy);
        sb.Append(' ').Append(dx).Append(' ').Append(dy);
        opLines.Add(sb.ToString());
        
        has[x,y,z] = false; raw[x,y,z] = 0;
        if (top[x,y] == z)
        {
            top[x,y] = -1;
            for (int zz = z-1; zz >= 0; zz--)
                if (has[x,y,zz]) { top[x,y] = zz; break; }
        }
        
        bool plant = (dx == 0 && dy == 0 && val > 0);
        if (!plant)
        {
            int t = top[dx,dy];
            int nz = t >= 0 ? t+1 : 0;
            if (nz < ZN) { has[dx,dy,nz] = true; raw[dx,dy,nz] = d; top[dx,dy] = nz; }
        }
    }
    
    static (int x, int y) FindDump(int fx, int fy)
    {
        int fz = top[fx,fy]; if (fz < 0) fz = 0;
        var vis = new HashSet<int>();
        var q = new Queue<(int,int)>();
        void Enq(int x, int y) {
            int k = x*YN+y;
            if (!vis.Contains(k)) { vis.Add(k); q.Enqueue((x,y)); }
        }
        Enq(fx,fy);
        
        var cands = new List<(int h, int dist, int x, int y)>();
        
        while (q.Count > 0 && cands.Count < 30)
        {
            var (cx,cy) = q.Dequeue();
            int cz = top[cx,cy]; if (cz < 0) cz = 0;
            if (!(cx == fx && cy == fy) && !(cx == 0 && cy == 0))
            {
                int dist = Math.Abs(cx-fx)+Math.Abs(cy-fy);
                cands.Add((cz, dist, cx, cy));
            }
            for (int d = 0; d < 4; d++)
            {
                int nx = cx+DX[d], ny = cy+DY[d];
                if (nx < 0 || nx >= XN || ny < 0 || ny >= YN) continue;
                int nz = top[nx,ny]; if (nz < 0) nz = 0;
                if (Math.Abs(nz-cz) <= 1) Enq(nx,ny);
            }
        }
        
        cands.Sort((a,b) => a.h != b.h ? a.h.CompareTo(b.h) : a.dist.CompareTo(b.dist));
        
        foreach (var (_,_,cx,cy) in cands)
        {
            var p = FindPath(fx,fy,cx,cy);
            if (p != null) return (cx,cy);
        }
        return (-1,-1);
    }
    
    static void RunMining()
    {
        long totalVox = (long)XN * YN * ZN;
        int limX = totalVox > 4_000_000 ? Math.Min(XN, 300) : XN;
        int limY = totalVox > 4_000_000 ? Math.Min(YN, 300) : YN;
        
        var cands = new List<(double score, int x, int y, int z)>();
        
        for (int x = 0; x < limX; x++)
        for (int y = 0; y < limY; y++)
        {
            int sz = top[x,y]; if (sz < 0) continue;
            for (int z = sz; z >= 0; z--)
            {
                if (!has[x,y,z]) continue;
                byte d = raw[x,y,z];
                int mass = d & 0xF, val = d >> 4;
                if (val > 0 && mass > 0)
                {
                    double gross = 1000.0 * mass * val;
                    int overburden = sz - z;
                    double moveCost = (x + y + overburden * 2) * mass * 2.0;
                    double net = gross - moveCost;
                    if (net > 0) cands.Add((net, x, y, z));
                }
            }
        }
        
        cands.Sort((a,b) => b.score.CompareTo(a.score));
        
        int opsLimit = totalVox > 4_000_000 ? 8000 : Math.Min(cands.Count, 50000);
        
        foreach (var (_,tx,ty,tz) in cands)
        {
            if (opLines.Count >= opsLimit) break;
            if (!has[tx,ty,tz]) continue;
            byte d2 = raw[tx,ty,tz];
            if ((d2 & 0xF) == 0 || (d2 >> 4) == 0) continue;
            
            // Remove overburden
            int cur = top[tx,ty];
            bool ok = true;
            for (int z = cur; z > tz && ok; z--)
            {
                if (!has[tx,ty,z]) continue;
                if (!SafeRemove(tx,ty,z)) { ok = false; break; }
                var (dx,dy) = FindDump(tx,ty);
                if (dx < 0) { ok = false; break; }
                var pth = FindPath(tx,ty,dx,dy);
                if (pth == null) { ok = false; break; }
                Perform(tx,ty,z,dx,dy,pth);
            }
            
            if (!ok || !has[tx,ty,tz]) continue;
            if (!SafeRemove(tx,ty,tz)) continue;
            var pp = FindPath(tx,ty,0,0);
            if (pp == null) continue;
            Perform(tx,ty,tz,0,0,pp);
        }
    }
    
    static void EmitOutput()
    {
        var sb = new StringBuilder();
        sb.Append(opLines.Count).Append('\n');
        foreach (var line in opLines) sb.Append(line).Append('\n');
        var bytes = Encoding.UTF8.GetBytes(sb.ToString());
        Console.OpenStandardOutput().Write(bytes, 0, bytes.Length);
    }
```

```csharp
using System;
using System.Collections.Generic;
using System.Text;

/// <summary>
/// Mining optimizer: extracts valuable ore and delivers to processing plant at (0,0).
/// Strategy:
///   1. Find all valuable voxels, estimate net profit
///   2. Sort by profit descending
///   3. For each candidate: remove overburden (dump to low-cost sites), then path to plant
///   4. Use A* for pathfinding on surface with slope <= 1Z/step constraint
/// </summary>
class M
{
    // Grid dimensions and block sizes
    static int XN, YN, ZN;
    static double XBS, YBS, ZBS;
    
    // World state
    static byte[,,] V;   // raw voxel data: lo4=mass, hi4=value
    static bool[,,] P;   // present[x,y,z]
    static int[,] T;     // top surface Z for each (x,y), -1 = empty column
    
    // Direction arrays (4-connected XY)
    static readonly int[] DX = {1,-1,0,0};
    static readonly int[] DY = {0,0,1,-1};
    
    // Output operations buffer
    static readonly List<byte[]> OPS = new List<byte[]>(65536);
    
    // A* reusable arrays (allocated once)
    static float[] GA;    // g-cost per node
    static int[] PD;      // previous direction
    static bool[] CL;     // closed set
    
    // Min-heap for A*
    static int[] HX, HY;
    static float[] HF;
    static int HSZ;
    
    static void Main()
    {
        ReadInput();
        InitSurface();
        AllocAStar();
        Mine();
        Output();
    }
    
    // ── Input ─────────────────────────────────────────────────────────────────
    static void ReadInput()
    {
        var sin = Console.OpenStandardInput();
        
        // Read header line
        var hb = new List<byte>(64);
        int c;
        while ((c = sin.ReadByte()) > 0 && c != '\n')
            if (c != '\r') hb.Add((byte)c);
        
        var tk = Encoding.ASCII.GetString(hb.ToArray())
                  .Split(new[]{' ','\t'}, StringSplitOptions.RemoveEmptyEntries);
        XN = int.Parse(tk[0]);
        YN = int.Parse(tk[1]);
        ZN = int.Parse(tk[2]);
        XBS = double.Parse(tk[3]);
        YBS = double.Parse(tk[4]);
        ZBS = double.Parse(tk[5]);
        
        V = new byte[XN, YN, ZN];
        P = new bool[XN, YN, ZN];
        
        // Bulk-read voxel data
        long total = (long)XN * YN * ZN;
        byte[] buf = new byte[Math.Min(total, 1 << 20)];
        long read = 0;
        int x = 0, y = 0, z = 0;
        
        while (read < total)
        {
            int got = sin.Read(buf, 0, (int)Math.Min(buf.Length, total - read));
            if (got <= 0) break;
            for (int i = 0; i < got; i++)
            {
                byte d = buf[i];
                V[x, y, z] = d;
                P[x, y, z] = d != 0;
                if (++z == ZN) { z = 0; if (++y == YN) { y = 0; x++; } }
            }
            read += got;
        }
    }
    
    // ── Surface ───────────────────────────────────────────────────────────────
    static void InitSurface()
    {
        T = new int[XN, YN];
        for (int x = 0; x < XN; x++)
        for (int y = 0; y < YN; y++)
        {
            T[x, y] = -1;
            for (int z = ZN-1; z >= 0; z--)
                if (P[x, y, z]) { T[x, y] = z; break; }
        }
    }
    
    // ── Voxel helpers ─────────────────────────────────────────────────────────
    static bool Exposed(int x, int y, int z)
    {
        if (!P[x, y, z]) return false;
        if (z+1 >= ZN || !P[x, y, z+1]) return true;
        for (int d = 0; d < 4; d++)
        {
            int nx = x+DX[d], ny = y+DY[d];
            if ((uint)nx >= (uint)XN || (uint)ny >= (uint)YN) return true;
            if (!P[nx, ny, z]) return true;
        }
        return false;
    }
    
    // Removing (x,y,z) is safe if voxel above (if any) has lateral support
    static bool CanDig(int x, int y, int z)
    {
        if (!Exposed(x, y, z)) return false;
        if (T[x, y] == z) return true;
        // Check if voxel directly above would be left floating
        if (z+1 < ZN && P[x, y, z+1])
        {
            for (int d = 0; d < 4; d++)
            {
                int nx = x+DX[d], ny = y+DY[d];
                if ((uint)nx < (uint)XN && (uint)ny < (uint)YN && P[nx, ny, z+1])
                    return true; // side support exists
            }
            return false; // would disconnect voxel above
        }
        return true;
    }
    
    // ── A* pathfinding ────────────────────────────────────────────────────────
    static void AllocAStar()
    {
        int n = XN * YN;
        GA = new float[n];
        PD = new int[n];
        CL = new bool[n];
        // Heap sized generously
        int cap = Math.Max(n * 4, 4096);
        HX = new int[cap]; HY = new int[cap]; HF = new float[cap];
    }
    
    static void HPush(float f, int x, int y)
    {
        if (HSZ >= HX.Length)
        {
            int nc = HX.Length * 2;
            Array.Resize(ref HX, nc);
            Array.Resize(ref HY, nc);
            Array.Resize(ref HF, nc);
        }
        int i = HSZ++;
        HX[i] = x; HY[i] = y; HF[i] = f;
        while (i > 0)
        {
            int p = (i-1) >> 1;
            if (HF[p] <= HF[i]) break;
            Swap(p, i); i = p;
        }
    }
    
    static void Swap(int a, int b)
    {
        float tf = HF[a]; HF[a] = HF[b]; HF[b] = tf;
        int ti = HX[a]; HX[a] = HX[b]; HX[b] = ti;
        ti = HY[a]; HY[a] = HY[b]; HY[b] = ti;
    }
    
    static (int x, int y) HPop()
    {
        int rx = HX[0], ry = HY[0];
        int last = --HSZ;
        HX[0] = HX[last]; HY[0] = HY[last]; HF[0] = HF[last];
        int i = 0;
        while (true)
        {
            int l = 2*i+1, r = 2*i+2, s = i;
            if (l < HSZ && HF[l] < HF[s]) s = l;
            if (r < HSZ && HF[r] < HF[s]) s = r;
            if (s == i) break;
            Swap(i, s); i = s;
        }
        return (rx, ry);
    }
    
    // Returns null if no path; otherwise list of (stepX, stepY) deltas
    static List<(int,int)> AStar(int x0, int y0, int xd, int yd)
    {
        if (x0 == xd && y0 == yd) return new List<(int,int)>();
        
        int n = XN * YN;
        // Reset arrays (use loop for cache efficiency)
        for (int i = 0; i < n; i++) { GA[i] = float.MaxValue; CL[i] = false; PD[i] = -1; }
        HSZ = 0;
        
        int s0 = x0*YN + y0;
        GA[s0] = 0;
        float h0 = (float)(Math.Abs(x0-xd) + Math.Abs(y0-yd));
        HPush(h0, x0, y0);
        
        while (HSZ > 0)
        {
            var (cx, cy) = HPop();
            int ci = cx*YN + cy;
            if (CL[ci]) continue;
            CL[ci] = true;
            
            if (cx == xd && cy == yd)
            {
                // Reconstruct path
                var path = new List<(int,int)>(Math.Abs(xd-x0)+Math.Abs(yd-y0)+4);
                int rx = xd, ry = yd;
                while (rx != x0 || ry != y0)
                {
                    int dir = PD[rx*YN+ry];
                    path.Add((DX[dir], DY[dir]));
                    rx -= DX[dir]; ry -= DY[dir];
                }
                path.Reverse();
                return path;
            }
            
            float cg = GA[ci];
            int cz = T[cx, cy]; if (cz < 0) cz = 0;
            
            for (int d = 0; d < 4; d++)
            {
                int nx = cx+DX[d], ny = cy+DY[d];
                if ((uint)nx >= (uint)XN || (uint)ny >= (uint)YN) continue;
                int ni = nx*YN + ny;
                if (CL[ni]) continue;
                int nz = T[nx, ny]; if (nz < 0) nz = 0;
                if (Math.Abs(nz-cz) > 1) continue;
                
                float step = (nz > cz) ? 5f : 1f;
                float ng = cg + step;
                if (ng < GA[ni])
                {
                    GA[ni] = ng;
                    PD[ni] = d;
                    float h = (float)(Math.Abs(nx-xd) + Math.Abs(ny-yd));
                    HPush(ng+h, nx, ny);
                }
            }
        }
        return null;
    }
    
    // ── World update ──────────────────────────────────────────────────────────
    static void DoMove(int x, int y, int z, int dx, int dy, List<(int,int)> path)
    {
        byte d = V[x, y, z];
        int val = d >> 4;
        
        // Build output line
        int sz = 3 + 3 + 3 + path.Count*4 + 3 + 3 + path.Count + 10;
        var sb = new StringBuilder(sz);
        sb.Append(path.Count).Append(' ')
          .Append(x).Append(' ').Append(y).Append(' ').Append(z);
        foreach (var (sx, sy) in path) sb.Append(' ').Append(sx).Append(' ').Append(sy);
        sb.Append(' ').Append(dx).Append(' ').Append(dy);
        OPS.Add(Encoding.ASCII.GetBytes(sb.ToString()));
        
        // Remove from source
        P[x, y, z] = false;
        V[x, y, z] = 0;
        if (T[x, y] == z)
        {
            T[x, y] = -1;
            for (int zz = z-1; zz >= 0; zz--)
                if (P[x, y, zz]) { T[x, y] = zz; break; }
        }
        
        // Deposit at destination (unless plant receiving valuable ore)
        if (!(dx == 0 && dy == 0 && val > 0))
        {
            int t = T[dx, dy];
            int nz = t >= 0 ? t+1 : 0;
            if (nz < ZN) { P[dx, dy, nz] = true; V[dx, dy, nz] = d; T[dx, dy] = nz; }
        }
    }
    
    // ── Dump site finder ─────────────────────────────────────────────────────
```

```csharp
        aGenCur++;
        hSz = 0;

        int s0 = x0 * YN + y0;
        aG[s0]   = 0;
        aGen[s0] = aGenCur;
        aPD[s0]  = -1;
        HPush(Heur(x0, y0, xd, yd), s0);

        while (hSz > 0)
        {
            int ci = HPop();
            int cx = ci / YN, cy = ci % YN;
            float cg = aG[ci];

            if (cx == xd && cy == yd)
                return Reconstruct(x0, y0, xd, yd);

            int cz = T[cx, cy]; if (cz < 0) cz = 0;

            for (int d = 0; d < 4; d++)
            {
                int nx = cx + DX[d], ny = cy + DY[d];
                if ((uint)nx >= (uint)XN || (uint)ny >= (uint)YN) continue;

                int nz = T[nx, ny]; if (nz < 0) nz = 0;
                if (Math.Abs(nz - cz) > 1) continue;   // slope too steep

                float step = nz > cz ? 5f : 1f;
                float ng   = cg + step;
                int   ni   = nx * YN + ny;

                bool fresh = aGen[ni] != aGenCur;
                if (fresh || ng < aG[ni])
                {
                    aG[ni]  = ng;
                    aPD[ni] = d;
                    if (fresh) aGen[ni] = aGenCur;
                    HPush(ng + Heur(nx, ny, xd, yd), ni);
                }
            }
        }
        return null;
    }

    static float Heur(int x, int y, int xd, int yd) =>
        (float)(Math.Abs(x - xd) + Math.Abs(y - yd));

    static List<(int, int)> Reconstruct(int x0, int y0, int xd, int yd)
    {
        var path = new List<(int, int)>(Math.Abs(xd - x0) + Math.Abs(yd - y0) + 4);
        int rx = xd, ry = yd;
        while (rx != x0 || ry != y0)
        {
            int dir = aPD[rx * YN + ry];
            path.Add((DX[dir], DY[dir]));
            rx -= DX[dir]; ry -= DY[dir];
        }
        path.Reverse();
        return path;
    }

    static void HPush(float key, int idx)
    {
        if (hSz >= hIdx.Length)
        {
            int nc = hIdx.Length * 2;
            Array.Resize(ref hIdx, nc);
            Array.Resize(ref hKey, nc);
        }
        int i = hSz++;
        hIdx[i] = idx; hKey[i] = key;
        while (i > 0)
        {
            int p = (i - 1) >> 1;
            if (hKey[p] <= hKey[i]) break;
            Swap(i, p); i = p;
        }
    }

    static int HPop()
    {
        int r = hIdx[0];
        hIdx[0] = hIdx[--hSz];
        hKey[0] = hKey[hSz];
        int i = 0;
        while (true)
        {
            int l = 2*i+1, r2 = 2*i+2, s = i;
            if (l < hSz && hKey[l] < hKey[s]) s = l;
            if (r2 < hSz && hKey[r2] < hKey[s]) s = r2;
            if (s == i) break;
            Swap(i, s); i = s;
        }
        return r;
    }

    static void Swap(int a, int b)
    {
        int  ti = hIdx[a]; hIdx[a] = hIdx[b]; hIdx[b] = ti;
        float tf = hKey[a]; hKey[a] = hKey[b]; hKey[b] = tf;
    }

    // ═══════════════════════════════════════════════════════════════════════════
    // World mutation
    // ═══════════════════════════════════════════════════════════════════════════
    static void Move(int x, int y, int z, int dx, int dy, List<(int, int)> path)
    {
        byte raw = V[x, y, z];
        int  val = raw >> 4;

        // Emit operation line
        var sb = new StringBuilder(32 + path.Count * 5);
        sb.Append(path.Count).Append(' ').Append(x).Append(' ').Append(y).Append(' ').Append(z);
        foreach (var (sx, sy) in path)
            sb.Append(' ').Append(sx).Append(' ').Append(sy);
        sb.Append(' ').Append(dx).Append(' ').Append(dy);
        OPS.Add(Encoding.ASCII.GetBytes(sb.ToString()));

        // Remove voxel from source
        P[x, y, z] = false;
        V[x, y, z] = 0;
        RecalcSurface(x, y, z);

        // Deposit at destination (unless plant receiving valuable ore)
        bool toPlant = dx == 0 && dy == 0 && val > 0;
        if (!toPlant)
        {
            int t  = T[dx, dy];
            int nz = t >= 0 ? t + 1 : 0;
            if (nz < ZN)
            {
                P[dx, dy, nz] = true;
                V[dx, dy, nz] = raw;
                T[dx, dy]     = nz;
            }
        }
    }

    // ═══════════════════════════════════════════════════════════════════════════
    // Dump site search — BFS from source to find accessible low-height cell
    // ═══════════════════════════════════════════════════════════════════════════
    static (int x, int y) FindDump(int fx, int fy)
    {
        int N  = XN * YN;
        var vis = new bool[N];

        var q    = new Queue<int>(128);
        var cands = new List<(int h, int dist, int idx)>(32);

        int s0 = fx * YN + fy;
        vis[s0] = true;
        q.Enqueue(s0);

        while (q.Count > 0 && cands.Count < 30)
        {
            int ci = q.Dequeue();
            int cx = ci / YN, cy = ci % YN;
            int cz = T[cx, cy]; if (cz < 0) cz = 0;

            if ((cx != fx || cy != fy) && (cx != 0 || cy != 0))
            {
                int dist = Math.Abs(cx - fx) + Math.Abs(cy - fy);
                cands.Add((cz, dist, ci));
            }

            for (int d = 0; d < 4; d++)
            {
                int nx = cx + DX[d], ny = cy + DY[d];
                if ((uint)nx >= (uint)XN || (uint)ny >= (uint)YN) continue;
                int ni = nx * YN + ny;
                if (vis[ni]) continue;
                int nz = T[nx, ny]; if (nz < 0) nz = 0;
                if (Math.Abs(nz - cz) > 1) continue;
                vis[ni] = true;
                q.Enqueue(ni);
            }
        }

        cands.Sort((a, b) => a.h != b.h ? a.h.CompareTo(b.h) : a.dist.CompareTo(b.dist));

        foreach (var (_, _, idx) in cands)
        {
            int cx = idx / YN, cy = idx % YN;
            var p = AStar(fx, fy, cx, cy);
            if (p != null) return (cx, cy);
        }
        return (-1, -1);
    }

    // ═══════════════════════════════════════════════════════════════════════════
    // Main mining loop
    // ═══════════════════════════════════════════════════════════════════════════
    static void Mine()
    {
        long tv  = (long)XN * YN * ZN;
        bool big = tv > 5_000_000;

        int limX = big ? Math.Min(XN, 500) : XN;
        int limY = big ? Math.Min(YN, 500) : YN;

        // ── Collect profitable candidates ──────────────────────────────────────
        var cands = new List<(double net, int x, int y, int z)>(16384);

        for (int x = 0; x < limX; x++)
        for (int y = 0; y < limY; y++)
        {
            int surf = T[x, y];
            if (surf < 0) continue;
            for (int z = surf; z >= 0; z--)
            {
                if (!P[x, y, z]) continue;
                byte d    = V[x, y, z];
                int  mass = d & 0xF, val = d >> 4;
                if (val <= 0 || mass <= 0) continue;

                double gross    = 1000.0 * mass * val;
                int    overb    = surf - z;
                // Rough cost estimate: overburden removal + travel to plant
                double estCost  = overb * mass * 3.0 + (x + y) * mass * 1.5;
                double net      = gross - estCost;
                if (net > 100)
                    cands.Add((net, x, y, z));
            }
        }

        cands.Sort((a, b) => b.net.CompareTo(a.net));

        int opLimit = big ? 10000 : Math.Min(cands.Count * 5 + 1000, 500000);

        // ── Process candidates ─────────────────────────────────────────────────
        foreach (var (_, tx, ty, tz) in cands)
        {
            if (OPS.Count >= opLimit) break;
            if (!P[tx, ty, tz]) continue;

            var dv = V[tx, ty, tz];
            if ((dv & 0xF) == 0 || (dv >> 4) == 0) continue;

            // Remove overburden above target voxel
            int  curTop = T[tx, ty];
            bool ok     = true;

            for (int z = curTop; z > tz && ok; z--)
            {
                if (!P[tx, ty, z]) continue;
                if (!CanDig(tx, ty, z)) { ok = false; break; }

                var (ddx, ddy) = FindDump(tx, ty);
                if (ddx < 0) { ok = false; break; }

                var pth = AStar(tx, ty, ddx, ddy);
                if (pth == null) { ok = false; break; }

                Move(tx, ty, z, ddx, ddy, pth);
                if (OPS.Count >= opLimit) { ok = false; break; }
            }

            if (!ok || !P[tx, ty, tz]) continue;
            if (!CanDig(tx, ty, tz)) continue;

            // Ship ore to plant
            var pp = AStar(tx, ty, 0, 0);
            if (pp == null) continue;

            Move(tx, ty, tz, 0, 0, pp);
        }
    }
```
