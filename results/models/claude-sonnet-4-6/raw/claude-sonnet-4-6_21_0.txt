#include <iostream>
#include <vector>
#include <deque>
#include <unordered_map>
#include <unordered_set>
#include <queue>
#include <algorithm>
#include <climits>
#include <tuple>

using namespace std;
using Pos = vector<int>;

struct PosHash {
    size_t operator()(const Pos& p) const noexcept {
        size_t h = 14695981039346656037ULL;
        for (int v : p) {
            h ^= (size_t)(unsigned)v;
            h *= 1099511628211ULL;
        }
        return h;
    }
};

int D;
long long max_turns;
vector<int> bounds;
deque<Pos> snake;
unordered_map<Pos, int, PosHash> snake_occ;
unordered_set<Pos, PosHash> obstacles;
unordered_set<Pos, PosHash> food_set;
vector<Pos> foods;
deque<pair<int,int>> planned_path;

bool inBounds(const Pos& p) {
    for (int i = 0; i < D; i++)
        if (p[i] < 0 || p[i] >= bounds[i]) return false;
    return true;
}

// Can the snake move to position np this turn?
// Allows entering the tail cell if it has exactly one occupant (it will vacate).
bool canMove(const Pos& np) {
    if (!inBounds(np)) return false;
    if (obstacles.count(np)) return false;
    auto it = snake_occ.find(np);
    if (it == snake_occ.end()) return true;
    return snake.back() == np && it->second == 1;
}

int manhattan(const Pos& a, const Pos& b) {
    int d = 0;
    for (int i = 0; i < D; i++) d += abs(a[i] - b[i]);
    return d;
}

deque<pair<int,int>> bfsPath(const Pos& target, int limit) {
    const Pos& head = snake[0];
    if (head == target) return {};
    struct CE { Pos from; int a, d; };
    unordered_map<Pos, CE, PosHash> came;
    queue<Pos> q;
    came[head] = {{}, -1, 0};
    q.push(head);
    const int dirs[2] = {-1, 1};
    while (!q.empty() && (int)came.size() <= limit) {
        Pos cur = q.front(); q.pop();
        for (int a = 0; a < D; a++) {
            for (int di = 0; di < 2; di++) {
                int d = dirs[di];
                Pos np = cur;
                np[a] += d;
                if (came.count(np) || !canMove(np)) continue;
                came[np] = {cur, a, d};
                if (np == target) {
                    deque<pair<int,int>> path;
                    Pos c = np;
                    while (c != head) {
                        auto& e = came[c];
                        path.push_front({e.a, e.d});
                        c = e.from;
                    }
                    return path;
                }
                q.push(np);
            }
        }
    }
    return {};
}

deque<pair<int,int>> astarPath(const Pos& target, int limit) {
    const Pos& head = snake[0];
    if (head == target) return {};
    struct CE { Pos from; int a, d; };
    using St = tuple<int,int,Pos>;
    priority_queue<St, vector<St>, greater<St>> pq;
    unordered_map<Pos, int, PosHash> g;
    unordered_map<Pos, CE, PosHash> came;
    g[head] = 0;
    pq.push({manhattan(head, target), 0, head});
    int vis = 0;
    const int dirs[2] = {-1, 1};
    while (!pq.empty() && vis <= limit) {
        auto top = pq.top(); pq.pop();
        int f = get<0>(top), cost = get<1>(top);
        Pos cur = get<2>(top);
        (void)f;
        if (g.count(cur) && g[cur] < cost) continue;
        vis++;
        if (cur == target) {
            deque<pair<int,int>> path;
            Pos c = cur;
            while (c != head) {
                auto& e = came[c];
                path.push_front({e.a, e.d});
                c = e.from;
            }
            return path;
        }
        for (int a = 0; a < D; a++) {
            for (int di = 0; di < 2; di++) {
                int d = dirs[di];
                Pos np = cur;
                np[a] += d;
                if (!canMove(np)) continue;
                int ng = cost + 1;
                if (!g.count(np) || g[np] > ng) {
                    g[np] = ng;
                    came[np] = {cur, a, d};
                    pq.push({ng + manhattan(np, target), ng, np});
                }
            }
        }
    }
    return {};
}

pair<int,int> computeNewPath() {
    if (foods.empty()) return {0, 1};
    const Pos& h = snake[0];
    const Pos* tgt = nullptr;
    int bestD = INT_MAX;
    for (const auto& f : foods) {
        int d = manhattan(h, f);
        if (d < bestD) { bestD = d; tgt = &f; }
    }
    // Estimate space size to choose algorithm
    long long total = 1;
    for (int b : bounds) {
        if (total > 1000000LL / b) { total = 1000001LL; break; }
        total *= b;
    }
    deque<pair<int,int>> path;
    if (total <= 1000000LL) {
        int lim = (int)min(total * 2, (long long)600000);
        path = bfsPath(*tgt, lim);
        if (path.empty()) {
            path = astarPath(*tgt, 200000);
        }
    } else {
        path = astarPath(*tgt, 200000);
    }
    if (!path.empty()) {
        pair<int,int> first = path.front();
        path.pop_front();
        planned_path = move(path);
        return first;
    }
    // Greedy fallback: move toward nearest food
    int bd = INT_MAX;
    pair<int,int> best = {-1, 0};
    const int dirs[2] = {-1, 1};
    for (int a = 0; a < D; a++) {
        for (int di = 0; di < 2; di++) {
            int d = dirs[di];
            Pos np = h;
            np[a] += d;
            if (!canMove(np)) continue;
            int dist = manhattan(np, *tgt);
            if (dist < bd) { bd = dist; best = {a, d}; }
        }
    }
    if (best.first != -1) return best;
    // Any valid move (survival)
    for (int a = 0; a < D; a++) {
        for (int di = 0; di < 2; di++) {
            int d = dirs[di];
            Pos np = h;
            np[a] += d;
            if (canMove(np)) return {a, d};
        }
    }
    return {0, 1};
}

pair<int,int> getMove() {
    if (!planned_path.empty()) {
        pair<int,int> mv = planned_path.front();
        Pos np = snake[0];
        np[mv.first] += mv.second;
        if (canMove(np)) {
            planned_path.pop_front();
            return mv;
        }
        planned_path.clear();
    }
    return computeNewPath();
}

void doMove(int axis, int dir) {
    Pos nh = snake[0];
    nh[axis] += dir;
    bool ate = food_set.count(nh) > 0;
    snake.push_front(nh);
    snake_occ[nh]++;
    if (ate) {
        food_set.erase(nh);
        auto it = find(foods.begin(), foods.end(), nh);
        if (it != foods.end()) foods.erase(it);
        planned_path.clear(); // Recompute path after eating food
    } else {
        Pos tail = snake.back();
        snake.pop_back();
        auto it = snake_occ.find(tail);
        if (--(it->second) == 0) snake_occ.erase(it);
    }
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    cin >> D >> max_turns;
    bounds.resize(D);
    for (int i = 0; i < D; i++) cin >> bounds[i];

    int len; cin >> len;
    Pos head(D);
    for (int i = 0; i < D; i++) cin >> head[i];
    // Initialize snake coiled at head position
    for (int i = 0; i < len; i++) snake.push_back(head);
    snake_occ[head] = len;

    int F; cin >> F;
    for (int i = 0; i < F; i++) {
        Pos fp(D);
        for (int j = 0; j < D; j++) cin >> fp[j];
        foods.push_back(fp);
        food_set.insert(fp);
    }

    int O; cin >> O;
    for (int i = 0; i < O; i++) {
        Pos op(D);
        for (int j = 0; j < D; j++) cin >> op[j];
        obstacles.insert(op);
    }

    for (long long t = 0; t < max_turns && !food_set.empty(); t++) {
        pair<int,int> mv = getMove();
        cout << mv.first << ' ' << mv.second << '\n';
        cout.flush();
        doMove(mv.first, mv.second);
    }

    return 0;
}
