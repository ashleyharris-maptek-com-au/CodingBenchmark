using System;
using System.Collections.Generic;
using System.IO;

class CP
{
    static int N, OrigM;
    static List<(int to, long w, int eid)>[] G;
    static int[] EU, EV;
    static long[] EW;

    static long[] D;
    static int[] Dt, Pv, Pe;
    static int Round;

    static byte[] Buf;
    static int Pos;

    static int RI()
    {
        while (Pos < Buf.Length && (Buf[Pos] < '0' || Buf[Pos] > '9') && Buf[Pos] != '-') Pos++;
        bool neg = Pos < Buf.Length && Buf[Pos] == '-'; if (neg) Pos++;
        int v = 0;
        while (Pos < Buf.Length && Buf[Pos] >= '0' && Buf[Pos] <= '9') v = v * 10 + (Buf[Pos++] - '0');
        return neg ? -v : v;
    }

    static long RL()
    {
        while (Pos < Buf.Length && (Buf[Pos] < '0' || Buf[Pos] > '9') && Buf[Pos] != '-') Pos++;
        bool neg = Pos < Buf.Length && Buf[Pos] == '-'; if (neg) Pos++;
        long v = 0;
        while (Pos < Buf.Length && Buf[Pos] >= '0' && Buf[Pos] <= '9') v = v * 10 + (Buf[Pos++] - '0');
        return neg ? -v : v;
    }

    static void Main()
    {
        using (var ms = new MemoryStream()) { Console.OpenStandardInput().CopyTo(ms); Buf = ms.ToArray(); }

        N = RI(); int M = RI(); OrigM = M;

        G = new List<(int, long, int)>[N];
        for (int i = 0; i < N; i++) G[i] = new List<(int, long, int)>();

        int initCap = Math.Max(M * 2, 16);
        EU = new int[initCap]; EV = new int[initCap]; EW = new long[initCap];

        int[] deg = new int[N];
        for (int i = 0; i < M; i++)
        {
            int u = RI(), v = RI(); long w = RL();
            EU[i] = u; EV[i] = v; EW[i] = w;
            G[u].Add((v, w, i)); G[v].Add((u, w, i));
            deg[u]++; deg[v]++;
        }

        D = new long[N]; Dt = new int[N]; Pv = new int[N]; Pe = new int[N];

        var odds = new List<int>();
        for (int i = 0; i < N; i++) if (deg[i] % 2 == 1) odds.Add(i);

        int nextEid = M;
        int k = odds.Count;

        if (k > 0)
        {
            if (k <= 20)
            {
                long[,] pd = new long[k, k];
                int[][] spn = new int[k][], spe = new int[k][];
                for (int i = 0; i < k; i++)
                {
                    FullDijk(odds[i]);
                    spn[i] = (int[])Pv.Clone(); spe[i] = (int[])Pe.Clone();
                    for (int j = 0; j < k; j++) pd[i, j] = D[odds[j]];
                }
                int[] match = Bmatch(k, pd);
                for (int i = 0; i < k; i++)
                    if (match[i] > i) AddPath(spn[i], spe[i], odds[i], odds[match[i]], ref nextEid);
            }
            else
            {
                bool[] rem = new bool[N];
                for (int i = 0; i < k; i++) rem[odds[i]] = true;
                for (int i = 0; i < k; i++)
                {
                    if (!rem[odds[i]]) continue;
                    rem[odds[i]] = false;
                    int v = LazyDijk(odds[i], rem);
                    if (v >= 0) { rem[v] = false; AddPath(Pv, Pe, odds[i], v, ref nextEid); }
                }
            }
        }

        bool[] used = new bool[nextEid];
        int[] ptr = new int[N];
        var circ = new List<int>(nextEid + 2);
        var stk = new Stack<int>();
        stk.Push(0);
        while (stk.Count > 0)
        {
            int v = stk.Peek(); bool ok = false;
            while (ptr[v] < G[v].Count)
            {
                var (to, _, eid) = G[v][ptr[v]++];
                if (!used[eid]) { used[eid] = true; stk.Push(to); ok = true; break; }
            }
            if (!ok) circ.Add(stk.Pop());
        }
        circ.Reverse();

        using var sw = new StreamWriter(new BufferedStream(Console.OpenStandardOutput(), 1 << 20));
        for (int i = 0; i < circ.Count; i++) { if (i > 0) sw.Write(' '); sw.Write(circ[i]); }
        sw.WriteLine();
    }

    static void FullDijk(int src)
    {
        Array.Fill(D, long.MaxValue / 2); Array.Fill(Pv, -1); Array.Fill(Pe, -1);
        D[src] = 0;
        var pq = new PriorityQueue<int, long>(); pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > D[u]) continue;
            foreach (var (v, w, e) in G[u])
            {
                if (e >= OrigM) continue;
                long nd = d + w;
                if (nd < D[v]) { D[v] = nd; Pv[v] = u; Pe[v] = e; pq.Enqueue(v, nd); }
            }
        }
    }

    static int LazyDijk(int src, bool[] tgt)
    {
        int r = ++Round;
        D[src] = 0; Dt[src] = r; Pv[src] = -1; Pe[src] = -1;
        var pq = new PriorityQueue<int, long>(); pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            long du = Dt[u] == r ? D[u] : long.MaxValue / 2;
            if (d > du) continue;
            if (tgt[u]) return u;
            foreach (var (v, w, e) in G[u])
            {
                if (e >= OrigM) continue;
                long nd = d + w;
                long dv = Dt[v] == r ? D[v] : long.MaxValue / 2;
                if (nd < dv) { D[v] = nd; Dt[v] = r; Pv[v] = u; Pe[v] = e; pq.Enqueue(v, nd); }
            }
        }
        return -1;
    }

    static void AddPath(int[] pv, int[] pe, int src, int dst, ref int nextEid)
    {
        var eids = new Stack<int>();
        int cur = dst;
        while (cur != src) { eids.Push(pe[cur]); cur = pv[cur]; }
        while (eids.Count > 0)
        {
            int eid = eids.Pop();
            int u = EU[eid], v = EV[eid]; long w = EW[eid];
            if (nextEid >= EU.Length)
            {
                int ns = EU.Length * 2;
                Array.Resize(ref EU, ns); Array.Resize(ref EV, ns); Array.Resize(ref EW, ns);
            }
            EU[nextEid] = u; EV[nextEid] = v; EW[nextEid] = w;
            G[u].Add((v, w, nextEid)); G[v].Add((u, w, nextEid));
            nextEid++;
        }
    }

    static int[] Bmatch(int k, long[,] d)
    {
        int sz = 1 << k;
        long[] dp = new long[sz]; int[] ch = new int[sz];
        for (int s = 1; s < sz; s++) dp[s] = long.MaxValue / 2;
        for (int mask = 0; mask < sz; mask++)
        {
            if (dp[mask] >= long.MaxValue / 2) continue;
            int f = -1; for (int i = 0; i < k; i++) if ((mask & (1 << i)) == 0) { f = i; break; }
            if (f < 0) continue;
            for (int j = f + 1; j < k; j++)
            {
                if ((mask & (1 << j)) != 0) continue;
                int nm = mask | (1 << f) | (1 << j); long nc = dp[mask] + d[f, j];
                if (nc < dp[nm]) { dp[nm] = nc; ch[nm] = f * k + j; }
            }
        }
        int[] res = new int[k]; for (int i = 0; i < k; i++) res[i] = -1;
        int cur2 = sz - 1;
        while (cur2 > 0) { int p = ch[cur2]; int a = p / k, b = p % k; res[a] = b; res[b] = a; cur2 ^= (1 << a) | (1 << b); }
        return res;
    }
}
