using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

class Program
{
    static int H, W;
    static byte[] board;
    static bool[] isSafe;
    static bool[] isMine;
    static bool[] inQueue;
    static Queue<int> workQueue = new Queue<int>();

    static readonly int[] DR = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] DC = { -1, 0, 1, -1, 1, -1, 0, 1 };

    // Reusable arrays to avoid GC pressure in tight loops
    static int[] _cellsA = new int[8];
    static int[] _cellsB = new int[8];

    const byte B_SPACE = (byte)' ';
    const byte B_ZERO  = (byte)'0';
    const byte B_EIGHT = (byte)'8';
    const byte B_STAR  = (byte)'*';

    static void Main()
    {
        // Adaptive strategy:
        // Phase 1 (all boards): basic constraint propagation via work queue - O(cells) total
        // Phase 2 (complex boards): pairwise subset constraint elimination - O(cells * 25) per pass
        // For extreme boards the active frontier is << total cells, so subset passes are fast in practice.

        var stdin = new StreamReader(Console.OpenStandardInput(), Encoding.UTF8, false, 1 << 16);

        string[] parts = stdin.ReadLine().Trim().Split();
        H = int.Parse(parts[0]);
        W = int.Parse(parts[1]);

        int total = H * W;
        board   = new byte[total];
        isSafe  = new bool[total];
        isMine  = new bool[total];
        inQueue = new bool[total];

        for (int r = 0; r < H; r++)
        {
            string line = stdin.ReadLine() ?? "";
            int baseIdx = r * W;
            int len = Math.Min(line.Length, W);
            for (int c = 0; c < len; c++)
                board[baseIdx + c] = (byte)line[c];
            for (int c = len; c < W; c++)
                board[baseIdx + c] = B_SPACE;
        }

        // Initialize: flag mines and enqueue all numbered cells
        for (int i = 0; i < total; i++)
        {
            byte b = board[i];
            if (b == B_STAR)
                isMine[i] = true;
            else if (b >= B_ZERO && b <= B_EIGHT && !inQueue[i])
            {
                workQueue.Enqueue(i);
                inQueue[i] = true;
            }
        }

        // Phase 1: basic constraint propagation
        ProcessQueue();

        // Phase 2: subset constraint elimination, repeat until no new deductions
        bool changed = true;
        while (changed)
        {
            changed = SubsetConstraintsPass();
            if (changed) ProcessQueue();
        }

        // Collect and output results
        int count = 0;
        for (int i = 0; i < total; i++)
            if (isSafe[i]) count++;

        var sb = new StringBuilder(count * 10 + 20);
        sb.Append(count).Append('\n');
        for (int r = 0; r < H; r++)
        {
            int baseIdx = r * W;
            for (int c = 0; c < W; c++)
            {
                if (isSafe[baseIdx + c])
                    sb.Append(c).Append(' ').Append(r).Append('\n');
            }
        }
        Console.Write(sb);
    }

    // Enqueue all numbered cells adjacent to the given (unknown) cell for reprocessing
    static void EnqueueNeighboringNumberCells(int idx)
    {
        int r = idx / W, c = idx % W;
        for (int d = 0; d < 8; d++)
        {
            int nr = r + DR[d], nc = c + DC[d];
            if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
            int nidx = nr * W + nc;
            byte nb = board[nidx];
            if (nb >= B_ZERO && nb <= B_EIGHT && !inQueue[nidx])
            {
                workQueue.Enqueue(nidx);
                inQueue[nidx] = true;
            }
        }
    }

    // Basic constraint propagation:
    // If remaining_mines == 0: all unknown neighbors are safe
    // If remaining_mines == unknown_count: all unknown neighbors are mines
    static void ProcessQueue()
    {
        while (workQueue.Count > 0)
        {
            int idx = workQueue.Dequeue();
            inQueue[idx] = false;

            int r = idx / W, c = idx % W;
            int target = board[idx] - B_ZERO;

            int knownMines = 0, unknownCount = 0, unknownBits = 0;
            for (int d = 0; d < 8; d++)
            {
                int nr = r + DR[d], nc = c + DC[d];
                if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                int nidx = nr * W + nc;
                if (isMine[nidx])
                    knownMines++;
                else if (board[nidx] == B_SPACE && !isSafe[nidx])
                {
                    unknownBits |= 1 << d;
                    unknownCount++;
                }
            }

            int remaining = target - knownMines;
            if (remaining < 0) continue;

            bool markAllSafe = remaining == 0 && unknownCount > 0;
            bool markAllMine = remaining > 0 && remaining == unknownCount;
            if (!markAllSafe && !markAllMine) continue;

            for (int d = 0; d < 8; d++)
            {
                if ((unknownBits & (1 << d)) == 0) continue;
                int nr = r + DR[d], nc = c + DC[d];
                int nidx = nr * W + nc;

                if (markAllSafe && !isSafe[nidx])
                {
                    isSafe[nidx] = true;
                    EnqueueNeighboringNumberCells(nidx);
                }
                else if (markAllMine && !isMine[nidx])
                {
                    isMine[nidx] = true;
                    EnqueueNeighboringNumberCells(nidx);
                }
            }
        }
    }

    // Subset constraint elimination:
    // If constraint A (cells_A, mines_A) is a proper subset of constraint B (cells_B, mines_B),
    // then cells_B \ cells_A contains exactly (mines_B - mines_A) mines.
    // Two constraints can share cells only if their source numbered cells are within Chebyshev distance 2.
    static bool SubsetConstraintsPass()
    {
        bool anyChanged = false;

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                int idx = r * W + c;
                byte bA = board[idx];
                if (bA < B_ZERO || bA > B_EIGHT) continue;

                // Build constraint A
                int countA = 0, minesA = 0;
                for (int d = 0; d < 8; d++)
                {
                    int nr = r + DR[d], nc = c + DC[d];
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    int nidx = nr * W + nc;
                    if (isMine[nidx]) minesA++;
                    else if (board[nidx] == B_SPACE && !isSafe[nidx]) _cellsA[countA++] = nidx;
                }
                if (countA == 0) continue;
                int remA = (bA - B_ZERO) - minesA;

                // Compare with all numbered cells in the 5x5 neighborhood
                for (int dr2 = -2; dr2 <= 2; dr2++)
                {
                    for (int dc2 = -2; dc2 <= 2; dc2++)
                    {
                        if (dr2 == 0 && dc2 == 0) continue;
                        int nr2 = r + dr2, nc2 = c + dc2;
                        if ((uint)nr2 >= (uint)H || (uint)nc2 >= (uint)W) continue;
                        int idx2 = nr2 * W + nc2;
                        byte bB = board[idx2];
                        if (bB < B_ZERO || bB > B_EIGHT) continue;

                        // Build constraint B
                        int countB = 0, minesB = 0;
                        for (int d = 0; d < 8; d++)
                        {
                            int nnr = nr2 + DR[d], nnc = nc2 + DC[d];
                            if ((uint)nnr >= (uint)H || (uint)nnc >= (uint)W) continue;
                            int nnidx = nnr * W + nnc;
                            if (isMine[nnidx]) minesB++;
                            else if (board[nnidx] == B_SPACE && !isSafe[nnidx]) _cellsB[countB++] = nnidx;
                        }
                        // We only benefit if A is a proper subset of B (|A| < |B|)
                        if (countB == 0 || countA >= countB) continue;
                        int remB = (bB - B_ZERO) - minesB;

                        // Check A is a subset of B
                        bool subset = true;
                        for (int i = 0; i < countA && subset; i++)
                        {
                            bool found = false;
                            for (int j = 0; j < countB; j++)
                                if (_cellsA[i] == _cellsB[j]) { found = true; break; }
                            if (!found) subset = false;
                        }
                        if (!subset) continue;

                        int diffMines = remB - remA;
                        int diffCount = countB - countA;
                        if (diffMines < 0 || diffMines > diffCount) continue;

                        // Only actionable if all diff cells are safe (0 mines) or all are mines
                        if (diffMines != 0 && diffMines != diffCount) continue;

                        for (int j = 0; j < countB; j++)
                        {
                            // Skip cells that are in A
                            bool inA = false;
                            for (int i = 0; i < countA; i++)
                                if (_cellsB[j] == _cellsA[i]) { inA = true; break; }
                            if (inA) continue;

                            int didx = _cellsB[j];
                            if (diffMines == 0 && !isSafe[didx])
                            {
                                isSafe[didx] = true;
                                anyChanged = true;
                                EnqueueNeighboringNumberCells(didx);
                            }
                            else if (diffMines > 0 && !isMine[didx])
                            {
                                isMine[didx] = true;
                                anyChanged = true;
                                EnqueueNeighboringNumberCells(didx);
                            }
                        }
                    }
                }
            }
        }

        return anyChanged;
    }
}
