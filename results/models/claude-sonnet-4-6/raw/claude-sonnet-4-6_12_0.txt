using System;
using System.IO;
using System.Text;

class Solution
{
    static void Rotate(double vx, double vy, double vz,
                       double qw, double qx, double qy, double qz,
                       out double rx, out double ry, out double rz)
    {
        // Rodrigues' rotation: v' = v + 2*qw*(q.xyz x v) + 2*(q.xyz x (q.xyz x v))
        double t1 = 2 * (qy * vz - qz * vy);
        double t2 = 2 * (qz * vx - qx * vz);
        double t3 = 2 * (qx * vy - qy * vx);
        rx = vx + qw * t1 + qy * t3 - qz * t2;
        ry = vy + qw * t2 + qz * t1 - qx * t3;
        rz = vz + qw * t3 + qx * t2 - qy * t1;
    }

    static void Main()
    {
        // ---- Parse input ----
        string[] p = Console.ReadLine().Trim()
            .Split(new[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        int V = int.Parse(p[0]), F = int.Parse(p[1]);

        double[] vX = new double[V], vY = new double[V], vZ = new double[V];
        for (int i = 0; i < V; i++) {
            p = Console.ReadLine().Trim()
                .Split(new[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
            vX[i] = double.Parse(p[0]);
            vY[i] = double.Parse(p[1]);
            vZ[i] = double.Parse(p[2]);
        }
        for (int i = 0; i < F; i++) Console.ReadLine();

        p = Console.ReadLine().Trim()
            .Split(new[] {' ', '\t'}, StringSplitOptions.RemoveEmptyEntries);
        double W = double.Parse(p[0]), H = double.Parse(p[1]), Dp = double.Parse(p[2]);

        // ---- All 24 axis-aligned rotation quaternions (qw, qx, qy, qz) ----
        // Covers: identity, 90/180/270 around X/Y/Z axes,
        //         180 around face diagonals, 120/240 around body diagonals
        double s = Math.Sqrt(0.5), h = 0.5;
        double[][] quats = {
            new[]{1d,  0d,  0d,  0d},
            // rotations around X
            new[]{s,   s,   0d,  0d},
            new[]{0d,  1d,  0d,  0d},
            new[]{s,  -s,   0d,  0d},
            // rotations around Y
            new[]{s,   0d,  s,   0d},
            new[]{0d,  0d,  1d,  0d},
            new[]{s,   0d, -s,   0d},
            // rotations around Z
            new[]{s,   0d,  0d,  s},
            new[]{0d,  0d,  0d,  1d},
            new[]{s,   0d,  0d, -s},
            // 180 around face diagonals
            new[]{0d,  s,   s,   0d},
            new[]{0d,  s,  -s,   0d},
            new[]{0d,  0d,  s,   s},
            new[]{0d,  0d,  s,  -s},
            new[]{0d,  s,   0d,  s},
            new[]{0d,  s,   0d, -s},
            // 120/240 around body diagonals
            new[]{h,   h,   h,   h},
            new[]{h,  -h,  -h,  -h},
            new[]{h,  -h,   h,   h},
            new[]{h,   h,  -h,  -h},
            new[]{h,   h,  -h,   h},
            new[]{h,  -h,   h,  -h},
            new[]{h,   h,   h,  -h},
            new[]{h,  -h,  -h,   h}
        };

        // ---- Evaluate each rotation: compute AABB and count grid placements ----
        long bestN = 0;
        double[] bestQ = quats[0];
        double bDx = 1, bDy = 1, bDz = 1;
        double bMx = 0, bMy = 0, bMz = 0;
        long bNx = 0, bNy = 0, bNz = 0;

        foreach (var q in quats) {
            double qw = q[0], qx = q[1], qy = q[2], qz = q[3];
            double mnX =  1e18, mnY =  1e18, mnZ =  1e18;
            double mxX = -1e18, mxY = -1e18, mxZ = -1e18;

            for (int i = 0; i < V; i++) {
                Rotate(vX[i], vY[i], vZ[i], qw, qx, qy, qz,
                       out double rx, out double ry, out double rz);
                if (rx < mnX) mnX = rx;  if (rx > mxX) mxX = rx;
                if (ry < mnY) mnY = ry;  if (ry > mxY) mxY = ry;
                if (rz < mnZ) mnZ = rz;  if (rz > mxZ) mxZ = rz;
            }

            double dx = mxX - mnX, dy = mxY - mnY, dz = mxZ - mnZ;
            if (dx < 1e-12 || dy < 1e-12 || dz < 1e-12) continue;

            long nx = (long)(W / dx), ny = (long)(H / dy), nz = (long)(Dp / dz);
            long n  = nx * ny * nz;
            if (n > bestN) {
                bestN = n;  bestQ = q;
                bDx = dx;   bDy = dy;   bDz = dz;
                bMx = mnX;  bMy = mnY;  bMz = mnZ;
                bNx = nx;   bNy = ny;   bNz = nz;
            }
        }

        // ---- Output all placements with buffered I/O ----
        using var sw = new StreamWriter(
            new BufferedStream(Console.OpenStandardOutput(), 1 << 20), Encoding.ASCII);

        sw.WriteLine(bestN);

        // Pre-format the constant quaternion string
        string qstr = bestQ[0].ToString("F6") + " " + bestQ[1].ToString("F6") + " " +
                      bestQ[2].ToString("F6") + " " + bestQ[3].ToString("F6");

        // Grid placement: copy (ix,iy,iz) has its AABB starting at (ix*dx, iy*dy, iz*dz).
        // Translation = -(min of rotated AABB) + grid offset, so AABB aligns to grid cell.
        for (long ix = 0; ix < bNx; ix++) {
            string txs = (-bMx + ix * bDx).ToString("F6");
            for (long iy = 0; iy < bNy; iy++) {
                string tys = (-bMy + iy * bDy).ToString("F6");
                for (long iz = 0; iz < bNz; iz++) {
                    string tzs = (-bMz + iz * bDz).ToString("F6");
                    sw.Write(txs);  sw.Write(' ');
                    sw.Write(tys);  sw.Write(' ');
                    sw.Write(tzs);  sw.Write(' ');
                    sw.WriteLine(qstr);
                }
            }
        }
        sw.Flush();
    }
}
