struct V3 {
    double x,y,z;
    V3():x(0),y(0),z(0){}
    V3(double x,double y,double z):x(x),y(y),z(z){}
    V3 operator+(V3 b)const{return{x+b.x,y+b.y,z+b.z};}
    V3 operator-(V3 b)const{return{x-b.x,y-b.y,z-b.z};}
    V3 operator*(double t)const{return{x*t,y*t,z*t};}
    V3 operator/(double t)const{return{x/t,y/t,z/t};}
    double dot(V3 b)const{return x*b.x+y*b.y+z*b.z;}
    V3 cross(V3 b)const{return{y*b.z-z*b.y,z*b.x-x*b.z,x*b.y-y*b.x};}
    double norm2()const{return x*x+y*y+z*z;}
    double norm()const{return sqrt(norm2());}
    V3 unit()const{double n=norm();return n>1e-15?*this/n:V3();}
};

struct AABB {
    V3 lo,hi;
    AABB():lo(1e30,1e30,1e30),hi(-1e30,-1e30,-1e30){}
    AABB(V3 a,V3 b,V3 c){
        lo.x=min({a.x,b.x,c.x});lo.y=min({a.y,b.y,c.y});lo.z=min({a.z,b.z,c.z});
        hi.x=max({a.x,b.x,c.x});hi.y=max({a.y,b.y,c.y});hi.z=max({a.z,b.z,c.z});
    }
    void pad(double e){lo=lo-V3(e,e,e);hi=hi+V3(e,e,e);}
    bool overlaps(AABB b)const{
        return lo.x<=b.hi.x&&hi.x>=b.lo.x&&lo.y<=b.hi.y&&hi.y>=b.lo.y&&lo.z<=b.hi.z&&hi.z>=b.lo.z;
    }
    AABB merge(AABB b)const{
        AABB r;
        r.lo.x=min(lo.x,b.lo.x);r.lo.y=min(lo.y,b.lo.y);r.lo.z=min(lo.z,b.lo.z);
        r.hi.x=max(hi.x,b.hi.x);r.hi.y=max(hi.y,b.hi.y);r.hi.z=max(hi.z,b.hi.z);
        return r;
    }
    V3 cen()const{return(lo+hi)*0.5;}
};

struct BVH {
    struct Node{AABB box;int l=-1,r=-1,tri=-1;bool leaf()const{return tri>=0;}};
    vector<Node> nd;
    const vector<V3>* vp=nullptr;
    const vector<array<int,3>>* fp=nullptr;

    AABB tb(int i)const{auto&f=(*fp)[i];return AABB((*vp)[f[0]],(*vp)[f[1]],(*vp)[f[2]]);}

    void build(const vector<V3>&v,const vector<array<int,3>>&f){
        vp=&v;fp=&f;
        if(f.empty())return;
        vector<int>ids(f.size());iota(ids.begin(),ids.end(),0);
        nd.reserve(2*f.size()+1);
        rec(ids,0,ids.size());
    }

    int rec(vector<int>&ids,int lo,int hi){
        int ni=nd.size();nd.emplace_back();
        nd[ni].box=tb(ids[lo]);
        for(int i=lo+1;i<hi;i++)nd[ni].box=nd[ni].box.merge(tb(ids[i]));
        if(hi-lo==1){nd[ni].tri=ids[lo];return ni;}
        V3 d=nd[ni].box.hi-nd[ni].box.lo;
        int ax=(d.y>d.x)?((d.z>d.y)?2:1):((d.z>d.x)?2:0);
        int mid=(lo+hi)/2;
        nth_element(ids.begin()+lo,ids.begin()+mid,ids.begin()+hi,
            [&](int a,int b){auto ca=tb(a).cen(),cb=tb(b).cen();
                return ax==0?ca.x<cb.x:ax==1?ca.y<cb.y:ca.z<cb.z;});
        nd[ni].l=rec(ids,lo,mid);
        nd[ni].r=rec(ids,mid,hi);
        return ni;
    }

    void query(int ni,AABB box,vector<int>&res)const{
        if(ni<0||!nd[ni].box.overlaps(box))return;
        if(nd[ni].leaf()){res.push_back(nd[ni].tri);return;}
        query(nd[ni].l,box,res);query(nd[ni].r,box,res);
    }

    bool rayBox(V3 o,V3 d,AABB box)const{
        double tmin=0,tmax=1e30;
        for(int i=0;i<3;i++){
            double oi=(i==0)?o.x:(i==1)?o.y:o.z;
            double di=(i==0)?d.x:(i==1)?d.y:d.z;
            double mni=(i==0)?box.lo.x:(i==1)?box.lo.y:box.lo.z;
            double mxi=(i==0)?box.hi.x:(i==1)?box.hi.y:box.hi.z;
            if(fabs(di)<1e-15){if(oi<mni||oi>mxi)return false;}
            else{double t1=(mni-oi)/di,t2=(mxi-oi)/di;
                if(t1>t2)swap(t1,t2);tmin=max(tmin,t1);tmax=min(tmax,t2);
                if(tmin>tmax)return false;}
        }
        return tmax>1e-12;
    }

    bool rayTri(V3 o,V3 d,V3 v0,V3 v1,V3 v2,double&t)const{
        V3 e1=v1-v0,e2=v2-v0,h=d.cross(e2);
        double a=e1.dot(h);if(fabs(a)<1e-15)return false;
        double f=1.0/a;V3 s=o-v0;
        double u=f*s.dot(h);if(u<-1e-9||u>1+1e-9)return false;
        V3 q=s.cross(e1);double v=f*d.dot(q);
        if(v<-1e-9||u+v>1+1e-9)return false;
        t=f*e2.dot(q);return true;
    }

    int rayCnt(int ni,V3 o,V3 d)const{
        if(ni<0||!rayBox(o,d,nd[ni].box))return 0;
        if(nd[ni].leaf()){
            auto&f=(*fp)[nd[ni].tri];double t;
            if(rayTri(o,d,(*vp)[f[0]],(*vp)[f[1]],(*vp)[f[2]],t)&&t>1e-9)return 1;
            return 0;
        }
        return rayCnt(nd[ni].l,o,d)+rayCnt(nd[ni].r,o,d);
    }

    bool inside(V3 p)const{
        if(nd.empty())return false;
        int v=0;
        v+=(rayCnt(0,p,V3(1,0,0))%2);
        v+=(rayCnt(0,p,V3(0,1,0))%2);
        v+=(rayCnt(0,p,V3(0,0,1))%2);
        return v>=2;
    }
};

bool triIsect(V3 p0,V3 p1,V3 p2,V3 q0,V3 q1,V3 q2,V3&a,V3&b){
    const double E=1e-10;
    V3 N1=(p1-p0).cross(p2-p0);
    if(N1.norm2()<E*E)return false;
    double dp0=N1.dot(q0-p0),dp1=N1.dot(q1-p0),dp2=N1.dot(q2-p0);
    if((dp0>E&&dp1>E&&dp2>E)||(dp0<-E&&dp1<-E&&dp2<-E))return false;
    V3 N2=(q1-q0).cross(q2-q0);
    if(N2.norm2()<E*E)return false;
    double dq0=N2.dot(p0-q0),dq1=N2.dot(p1-q0),dq2=N2.dot(p2-q0);
    if((dq0>E&&dq1>E&&dq2>E)||(dq0<-E&&dq1<-E&&dq2<-E))return false;
    V3 D=N1.cross(N2);
    if(D.norm2()<E*E)return false;

    auto getIsect=[&](V3 v0,V3 v1,V3 v2,double d0,double d1,double d2,V3&pa,V3&pb)->bool{
        const double EE=1e-9;
        vector<V3> pts;
        auto addEdge=[&](V3 ea,V3 eb,double da,double db){
            if(da*db<0){double t=da/(da-db);pts.push_back(ea+(eb-ea)*t);}
            else if(fabs(da)<EE){pts.push_back(ea);}
            else if(fabs(db)<EE){pts.push_back(eb);}
        };
        addEdge(v0,v1,d0,d1);addEdge(v1,v2,d1,d2);addEdge(v2,v0,d2,d0);
        vector<V3> uniq;
        for(auto&p:pts){bool dup=false;
            for(auto&u:uniq)if((u-p).norm2()<EE*EE*100){dup=true;break;}
            if(!dup)uniq.push_back(p);}
        if((int)uniq.size()<2)return false;
        pa=uniq[0];pb=uniq[1];return true;
    };

    V3 pa,pb,qa,qb;
    if(!getIsect(p0,p1,p2,dq0,dq1,dq2,pa,pb))return false;
    if(!getIsect(q0,q1,q2,dp0,dp1,dp2,qa,qb))return false;

    double tpa=D.dot(pa),tpb=D.dot(pb);
    double tqa=D.dot(qa),tqb=D.dot(qb);
    if(tpa>tpb){swap(tpa,tpb);swap(pa,pb);}
    if(tqa>tqb){swap(tqa,tqb);swap(qa,qb);}
    double ts=max(tpa,tqa),te=min(tpb,tqb);
    if(te<ts+E)return false;

    double lp=tpb-tpa,lq=tqb-tqa;
    V3 ptP1=pa+(pb-pa)*(lp>E?(ts-tpa)/lp:0.0);
    V3 ptP2=pa+(pb-pa)*(lp>E?(te-tpa)/lp:1.0);
    V3 ptQ1=qa+(qb-qa)*(lq>E?(ts-tqa)/lq:0.0);
    V3 ptQ2=qa+(qb-qa)*(lq>E?(te-tqa)/lq:1.0);
    a=(ptP1+ptQ1)*0.5;b=(ptP2+ptQ2)*0.5;
    return(b-a).norm2()>E*E;
}

struct V2{
    double x,y;
    V2():x(0),y(0){}
    V2(double x,double y):x(x),y(y){}
    V2 operator-(V2 b)const{return{x-b.x,y-b.y};}
    V2 operator+(V2 b)const{return{x+b.x,y+b.y};}
    double cross(V2 b)const{return x*b.y-y*b.x;}
    double norm2()const{return x*x+y*y;}
};

bool ptInTri2D(V2 p,V2 a,V2 b,V2 c){
    double d1=(b-a).cross(p-a),d2=(c-b).cross(p-b),d3=(a-c).cross(p-c);
    return!((d1<0||d2<0||d3<0)&&(d1>0||d2>0||d3>0));
}

void earClip(const vector<V2>&pts,vector<int>poly,vector<array<int,3>>&tris){
    int n=(int)poly.size();if(n<3)return;
    double area=0;
    for(int i=0;i<n;i++){int j=(i+1)%n;area+=pts[poly[i]].x*pts[poly[j]].y-pts[poly[j]].x*pts[poly[i]].y;}
    if(area<0)reverse(poly.begin(),poly.end());
    if(n==3){tris.push_back({poly[0],poly[1],poly[2]});return;}
    int iter=0,mi=n*n+100,idx=0;
    while((int)poly.size()>3&&iter++<mi){
        int sz=(int)poly.size();
        int prev=(idx-1+sz)%sz,curr=idx,next=(idx+1)%sz;
        V2 a=pts[poly[prev]],b=pts[poly[curr]],c=pts[poly[next]];
        double cr=(b-a).cross(c-a);
        if(cr>1e-12){
            bool ear=true;
            for(int k=0;k<sz&&ear;k++){
                if(k==prev||k==curr||k==next)continue;
                if(ptInTri2D(pts[poly[k]],a,b,c))ear=false;
            }
            if(ear){
                tris.push_back({poly[prev],poly[curr],poly[next]});
                poly.erase(poly.begin()+curr);
                idx=prev%(int)poly.size();iter=0;continue;
            }
        }
        idx=(idx+1)%(int)poly.size();
    }
    if((int)poly.size()==3)tris.push_back({poly[0],poly[1],poly[2]});
}

void triPoly(const vector<V2>&pts,vector<int>boundary,vector<pair<int,int>>cons,vector<array<int,3>>&tris){
    if((int)boundary.size()<3)return;
    if(cons.empty()){earClip(pts,boundary,tris);return;}
    for(int ci=0;ci<(int)cons.size();ci++){
        int a=cons[ci].first,b=cons[ci].second;
        int ap=-1,bp=-1;
        for(int k=0;k<(int)boundary.size();k++){
            if(boundary[k]==a)ap=k;if(boundary[k]==b)bp=k;
        }
        if(ap<0||bp<0||ap==bp)continue;
        if(ap>bp){swap(ap,bp);swap(a,b);}
        vector<int>p1,p2;
        for(int k=ap;k<=bp;k++)p1.push_back(boundary[k]);
        for(int k=bp;k<(int)boundary.size();k++)p2.push_back(boundary[k]);
        for(int k=0;k<=ap;k++)p2.push_back(boundary[k]);
        auto inP=[](const vector<int>&p,int v){for(int x:p)if(x==v)return true;return false;};
        vector<pair<int,int>>c1,c2;
        for(int k=0;k<(int)cons.size();k++){
            if(k==ci)continue;
            auto[ca,cb]=cons[k];
            if(inP(p1,ca)&&inP(p1,cb))c1.push_back({ca,cb});
            else if(inP(p2,ca)&&inP(p2,cb))c2.push_back({ca,cb});
        }
        triPoly(pts,p1,c1,tris);triPoly(pts,p2,c2,tris);return;
    }
    earClip(pts,boundary,tris);
}

struct Mesh{vector<V3>verts;vector<array<int,3>>faces;};

Mesh weld(const Mesh&m,double eps){
    double cs=max(eps*2.0,1e-20);
    auto hc=[](int x,int y,int z)->size_t{
        return(size_t)((long long)x*2654435761LL)^(size_t)((long long)y*805459861LL)^(size_t)((long long)z*3674653429LL);
    };
    unordered_map<size_t,vector<int>>grid;
    vector<V3>nv;nv.reserve(m.verts.size());
    vector<int>mp(m.verts.size());
    for(int i=0;i<(int)m.verts.size();i++){
        V3 v=m.verts[i];
        int cx=(int)floor(v.x/cs),cy=(int)floor(v.y/cs),cz=(int)floor(v.z/cs);
        int found=-1;
        for(int dx=-1;dx<=1&&found<0;dx++)
        for(int dy=-1;dy<=1&&found<0;dy++)
        for(int dz=-1;dz<=1&&found<0;dz++){
            auto it=grid.find(hc(cx+dx,cy+dy,cz+dz));if(it==grid.end())continue;
            for(int ni:it->second)if((nv[ni]-v).norm2()<eps*eps){found=ni;break;}
        }
        if(found>=0){mp[i]=found;}
        else{int ni=(int)nv.size();nv.push_back(v);mp[i]=ni;grid[hc(cx,cy,cz)].push_back(ni);}
    }
    Mesh r;r.verts=nv;
    for(auto&f:m.faces){int a=mp[f[0]],b=mp[f[1]],c=mp[f[2]];
        if(a!=b&&b!=c&&a!=c)r.faces.push_back({a,b,c});}
    return r;
}

struct TriBasis{
    V3 o,e1,e2,n;
    void set(V3 a,V3 b,V3 c){
        o=a;V3 ab=b-a;
        n=ab.cross(c-a).unit();
        e1=ab.unit();e2=n.cross(e1);
    }
    V2 to2D(V3 p)const{V3 d=p-o;return{d.dot(e1),d.dot(e2)};}
};

Mesh csgUnion(const Mesh&A,const Mesh&B){
    int VA=(int)A.verts.size(),FA=(int)A.faces.size();
    int VB=(int)B.verts.size();

    vector<V3>vp;
    vp.reserve(VA+VB+1000);
    for(auto&v:A.verts)vp.push_back(v);
    for(auto&v:B.verts)vp.push_back(v);

    BVH bA,bB;
    bA.build(A.verts,A.faces);
    bB.build(B.verts,B.faces);

    struct TD{
        array<vector<pair<double,int>>,3>ep;
        vector<pair<int,int>>seg;
    };
    vector<TD>dA(FA),dB((int)B.faces.size());

    for(int fi=0;fi<FA;fi++){
        auto&fa=A.faces[fi];
        V3 va0=A.verts[fa[0]],va1=A.verts[fa[1]],va2=A.verts[fa[2]];
        AABB boxA(va0,va1,va2);boxA.pad(1e-7);
        vector<int>cands;bB.query(0,boxA,cands);
        for(int gj:cands){
            auto&fb=B.faces[gj];
            V3 vb0=B.verts[fb[0]],vb1=B.verts[fb[1]],vb2=B.verts[fb[2]];
            V3 pt1,pt2;
            if(!triIsect(va0,va1,va2,vb0,vb1,vb2,pt1,pt2))continue;
            int i1=(int)vp.size();vp.push_back(pt1);
            int i2=(int)vp.size();vp.push_back(pt2);

            auto addToTri=[&](TD&td,V3 v0,V3 v1,V3 v2,int vi){
                V3 pt=vp[vi];
                const V3 from3[3]={v0,v1,v2};
                const V3 to3[3]={v1,v2,v0};
                for(int e=0;e<3;e++){
                    V3 from=from3[e],to=to3[e],edge=to-from;
                    double len2=edge.norm2();if(len2<1e-20)continue;
                    double t=edge.dot(pt-from)/len2;
                    if(t<-1e-5||t>1+1e-5)continue;
                    V3 proj=from+edge*max(0.0,min(1.0,t));
                    if((proj-pt).norm2()<1e-10){
                        td.ep[e].push_back({max(0.0,min(1.0,t)),vi});return;
                    }
                }
            };

            addToTri(dA[fi],va0,va1,va2,i1);
            addToTri(dA[fi],va0,va1,va2,i2);
            dA[fi].seg.push_back({i1,i2});
            addToTri(dB[gj],vb0,vb1,vb2,i1);
            addToTri(dB[gj],vb0,vb1,vb2,i2);
            dB[gj].seg.push_back({i1,i2});
        }
    }

    for(auto&d:dA)for(auto&e:d.ep)sort(e.begin(),e.end());
    for(auto&d:dB)for(auto&e:d.ep)sort(e.begin(),e.end());

    vector<array<int,3>>outF;

    auto process=[&](const Mesh&mesh,const vector<TD>&data,int voff,const BVH&obvh){
        for(int fi=0;fi<(int)mesh.faces.size();fi++){
            auto&f=mesh.faces[fi];
            int v0=f[0]+voff,v1=f[1]+voff,v2=f[2]+voff;
            auto&d=data[fi];
            bool hasI=!d.seg.empty();

            if(!hasI){
                V3 c=(vp[v0]+vp[v1]+vp[v2])/3.0;
                if(!obvh.inside(c))outF.push_back({v0,v1,v2});
                continue;
            }

            TriBasis basis;
            basis.set(vp[v0],vp[v1],vp[v2]);

            vector<V2>pts2d;
            map<int,int>vToL;
            auto getL=[&](int vi)->int{
                auto it=vToL.find(vi);if(it!=vToL.end())return it->second;
                int li=(int)pts2d.size();pts2d.push_back(basis.to2D(vp[vi]));vToL[vi]=li;return li;
            };

            int l0=getL(v0),l1=getL(v1),l2=getL(v2);

            vector<int>boundary;
            boundary.push_back(l0);
            for(auto&[t,vi]:d.ep[0])boundary.push_back(getL(vi));
            boundary.push_back(l1);
            for(auto&[t,vi]:d.ep[1])boundary.push_back(getL(vi));
            boundary.push_back(l2);
            for(auto&[t,vi]:d.ep[2])boundary.push_back(getL(vi));

            // Remove consecutive duplicates (by index or position)
            {
                vector<int>nb;
                for(int vi:boundary){
                    if(nb.empty()){nb.push_back(vi);continue;}
                    if(nb.back()==vi)continue;
                    if((pts2d[nb.back()]-pts2d[vi]).norm2()<1e-20)continue;
                    nb.push_back(vi);
                }
                while(nb.size()>1&&(nb.front()==nb.back()||
                    (pts2d[nb.front()]-pts2d[nb.back()]).norm2()<1e-20))nb.pop_back();
                boundary=nb;
            }

            if((int)boundary.size()<3){
                V3 c=(vp[v0]+vp[v1]+vp[v2])/3.0;
                if(!obvh.inside(c))outF.push_back({v0,v1,v2});
                continue;
            }

            vector<pair<int,int>>cons;
            for(auto&[ga,gb]:d.seg){
                auto ia=vToL.find(ga),ib=vToL.find(gb);
                if(ia!=vToL.end()&&ib!=vToL.end()&&ia->second!=ib->second)
                    cons.push_back({ia->second,ib->second});
            }

            // Deduplicate constraints (normalize order)
            for(auto&[ca,cb]:cons)if(ca>cb)swap(ca,cb);
            sort(cons.begin(),cons.end());
            cons.erase(unique(cons.begin(),cons.end()),cons.end());

            vector<array<int,3>>localTris;
            triPoly(pts2d,boundary,cons,localTris);

            vector<int>lToG((int)pts2d.size());
            for(auto&[gi,li]:vToL)lToG[li]=gi;

            for(auto&lt:localTris){
                int gi0=lToG[lt[0]],gi1=lToG[lt[1]],gi2=lToG[lt[2]];
                V3 c=(vp[gi0]+vp[gi1]+vp[gi2])/3.0;
                V3 normal=(vp[gi1]-vp[gi0]).cross(vp[gi2]-vp[gi0]).unit();
                c=c+normal*1e-7;
                if(!obvh.inside(c))outF.push_back({gi0,gi1,gi2});
            }
        }
    };

    process(A,dA,0,bB);
    process(B,dB,VA,bA);

    Mesh result;result.verts=vp;result.faces=outF;
    return result;
}

int main(){
    ios_base::sync_with_stdio(false);cin.tie(nullptr);
    int VA,FA,VB,FB;
    cin>>VA>>FA>>VB>>FB;
    Mesh A,B;
    A.verts.resize(VA);A.faces.resize(FA);
    B.verts.resize(VB);B.faces.resize(FB);
    for(auto&v:A.verts)cin>>v.x>>v.y>>v.z;
    for(auto&f:A.faces)cin>>f[0]>>f[1]>>f[2];
    for(auto&v:B.verts)cin>>v.x>>v.y>>v.z;
    for(auto&f:B.faces)cin>>f[0]>>f[1]>>f[2];

    Mesh result=csgUnion(A,B);
    result=weld(result,1e-7);

    cout<<result.verts.size()<<" "<<result.faces.size()<<"\n";
    cout<<fixed<<setprecision(10);
    for(auto&v:result.verts)cout<<v.x<<" "<<v.y<<" "<<v.z<<"\n";
    for(auto&f:result.faces)cout<<f[0]<<" "<<f[1]<<" "<<f[2]<<"\n";
    return 0;
}
