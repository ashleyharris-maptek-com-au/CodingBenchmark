using System;
using System.Diagnostics;
using System.Globalization;
using System.Linq;

class Program
{
    static int N;
    static double[] Pvx, Pvy, Pvz;
    static double[,] D;

    static void Main()
    {
        var t = NextLine();
        N = int.Parse(t[0]);
        // mu = double.Parse(t[1]); // not used directly in cost proxy

        var startTok = NextLine();
        double[] allVx = new double[N + 1];
        double[] allVy = new double[N + 1];
        double[] allVz = new double[N + 1];

        // Start position is index N
        allVx[N] = Pd(startTok[3]); allVy[N] = Pd(startTok[4]); allVz[N] = Pd(startTok[5]);

        for (int i = 0; i < N; i++)
        {
            var st = NextLine();
            allVx[i] = Pd(st[3]); allVy[i] = Pd(st[4]); allVz[i] = Pd(st[5]);
        }

        Pvx = allVx; Pvy = allVy; Pvz = allVz;

        D = new double[N + 1, N + 1];
        for (int i = 0; i <= N; i++)
            for (int j = 0; j <= N; j++)
                D[i, j] = CalcCost(i, j);

        var sw = Stopwatch.StartNew();
        int[] result = N <= 11 ? SolveSmall(sw) : SolveLarge(sw);
        Console.WriteLine(string.Join(" ", result));
    }

    static string[] NextLine()
    {
        string line;
        while ((line = Console.ReadLine()) != null)
        {
            line = line.Trim();
            if (line.Length > 0)
                return line.Split(new char[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        }
        return Array.Empty<string>();
    }

    static double Pd(string s) => double.Parse(s, CultureInfo.InvariantCulture);

    static double CalcCost(int i, int j)
    {
        if (i == j) return 0;
        double dvx = Pvx[i] - Pvx[j];
        double dvy = Pvy[i] - Pvy[j];
        double dvz = Pvz[i] - Pvz[j];
        return Math.Sqrt(dvx * dvx + dvy * dvy + dvz * dvz);
    }

    static double TourLen(int[] tour)
    {
        double c = D[N, tour[0]];
        for (int i = 0; i < N - 1; i++) c += D[tour[i], tour[i + 1]];
        return c;
    }

    static int[] NN(int from)
    {
        bool[] vis = new bool[N];
        int[] tour = new int[N];
        int cur = from;
        int start = 0;
        if (from < N) { vis[from] = true; tour[0] = from; start = 1; }
        for (int k = start; k < N; k++)
        {
            double best = double.MaxValue; int next = -1;
            for (int j = 0; j < N; j++)
                if (!vis[j] && D[cur, j] < best) { best = D[cur, j]; next = j; }
            vis[next] = true; tour[k] = next; cur = next;
        }
        return tour;
    }

    static int[] TwoOpt(int[] tour)
    {
        bool imp = true;
        while (imp)
        {
            imp = false;
            for (int i = 0; i < N - 1; i++)
                for (int j = i + 2; j < N; j++)
                {
                    int a = i == 0 ? N : tour[i - 1], b = tour[i], c = tour[j];
                    int d = j < N - 1 ? tour[j + 1] : -1;
                    double gain = D[a, b] - D[a, c] + (d >= 0 ? D[c, d] - D[b, d] : 0);
                    if (gain > 1e-10) { Array.Reverse(tour, i, j - i + 1); imp = true; }
                }
        }
        return tour;
    }

    static int[] SolveSmall(Stopwatch sw)
    {
        var best = TwoOpt(NN(N));
        double bestC = TourLen(best);
        for (int s = 0; s < N; s++)
        {
            var t = TwoOpt(NN(s));
            double c = TourLen(t);
            if (c < bestC) { bestC = c; best = t; }
        }
        // Brute force via Heap's algorithm
        var perm = Enumerable.Range(0, N).ToArray();
        int[] ci = new int[N];
        int ii = 0;
        while (ii < N && sw.Elapsed.TotalSeconds < 55)
        {
            if (ci[ii] < ii)
            {
                if (ii % 2 == 0) { int tmp = perm[0]; perm[0] = perm[ii]; perm[ii] = tmp; }
                else { int tmp = perm[ci[ii]]; perm[ci[ii]] = perm[ii]; perm[ii] = tmp; }
                double tc = TourLen(perm);
                if (tc < bestC) { bestC = tc; best = (int[])perm.Clone(); }
                ci[ii]++; ii = 0;
            }
            else { ci[ii] = 0; ii++; }
        }
        return best;
    }

    static int[] DoubleBridge(int[] tour, Random rng)
    {
        int n = tour.Length;
        if (n < 8) return (int[])tour.Clone();
        int p1 = rng.Next(1, n - 2);
        int p2 = rng.Next(p1 + 1, n - 1);
        int p3 = rng.Next(p2 + 1, n);
        var r = new int[n]; int idx = 0;
        for (int i = 0; i < p1; i++) r[idx++] = tour[i];
        for (int i = p2; i < p3; i++) r[idx++] = tour[i];
        for (int i = p1; i < p2; i++) r[idx++] = tour[i];
        for (int i = p3; i < n; i++) r[idx++] = tour[i];
        return r;
    }

    static int[] SolveLarge(Stopwatch sw)
    {
        var best = TwoOpt(NN(N));
        double bestC = TourLen(best);
        int nnLimit = Math.Min(N, 60);
        for (int s = 0; s < nnLimit && sw.Elapsed.TotalSeconds < 30; s++)
        {
            var t = TwoOpt(NN(s));
            double c = TourLen(t);
            if (c < bestC) { bestC = c; best = t; }
        }
        var rng = new Random(42);
        var cur = (int[])best.Clone();
        double curC = bestC;
        while (sw.Elapsed.TotalSeconds < 260)
        {
            var cand = TwoOpt(DoubleBridge(cur, rng));
            double cc = TourLen(cand);
            if (cc < curC)
            {
                cur = cand; curC = cc;
                if (cc < bestC) { bestC = cc; best = (int[])cand.Clone(); }
            }
            else
            {
                cur = (int[])best.Clone();
                curC = bestC;
            }
        }
        return best;
    }
}
