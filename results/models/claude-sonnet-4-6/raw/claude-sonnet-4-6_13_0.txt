#include <iostream>
#include <string>
#include <vector>
#include <unordered_set>
#include <unordered_map>
#include <algorithm>
#include <numeric>
#include <cstdint>

using namespace std;
typedef uint64_t u64;

static const u64 MOD1 = 1000000007ULL;
static const u64 MOD2 = 998244353ULL;
static const u64 BASE1 = 131ULL;
static const u64 BASE2 = 137ULL;
static const int SWITCH_THRESHOLD = 30;

static u64 fast_pow(u64 base, int exp, u64 mod) {
    u64 result = 1;
    base %= mod;
    while (exp > 0) {
        if (exp & 1) result = result * base % mod;
        base = base * base % mod;
        exp >>= 1;
    }
    return result;
}

bool check(const vector<string>& strs, const vector<int>& order, int L, string& result) {
    if (L == 0) { result = ""; return true; }

    u64 pow1 = fast_pow(BASE1, L, MOD1);
    u64 pow2 = fast_pow(BASE2, L, MOD2);

    int first = order[0];
    const string& s0 = strs[first];
    int n0 = (int)s0.size();
    if (n0 < L) return false;

    u64 h1 = 0, h2 = 0;
    for (int i = 0; i < L; i++) {
        h1 = (h1 * BASE1 + (u64)(unsigned char)s0[i]) % MOD1;
        h2 = (h2 * BASE2 + (u64)(unsigned char)s0[i]) % MOD2;
    }

    unordered_map<u64, int> candidates;
    candidates.reserve((size_t)(n0 - L + 2) * 2);
    candidates.emplace(h1 * MOD2 + h2, 0);

    for (int i = L; i < n0; i++) {
        h1 = (h1 * BASE1 % MOD1 + (u64)(unsigned char)s0[i] + MOD1
              - (u64)(unsigned char)s0[i - L] * pow1 % MOD1) % MOD1;
        h2 = (h2 * BASE2 % MOD2 + (u64)(unsigned char)s0[i] + MOD2
              - (u64)(unsigned char)s0[i - L] * pow2 % MOD2) % MOD2;
        candidates.emplace(h1 * MOD2 + h2, i - L + 1);
    }

    unordered_set<u64> present;

    for (int ki = 1; ki < (int)order.size() && !candidates.empty(); ki++) {
        // Once few candidates remain, switch to direct substring search
        if ((int)candidates.size() <= SWITCH_THRESHOLD) {
            vector<string> cands_str;
            cands_str.reserve(candidates.size());
            for (auto& [h, pos] : candidates)
                cands_str.push_back(s0.substr(pos, L));

            for (int ki2 = ki; ki2 < (int)order.size() && !cands_str.empty(); ki2++) {
                const string& sk2 = strs[order[ki2]];
                for (auto it = cands_str.begin(); it != cands_str.end(); ) {
                    if (sk2.find(*it) == string::npos)
                        it = cands_str.erase(it);
                    else
                        ++it;
                }
            }

            if (!cands_str.empty()) { result = cands_str[0]; return true; }
            return false;
        }

        const string& sk = strs[order[ki]];
        int nk = (int)sk.size();
        if (nk < L) return false;

        present.clear();
        present.reserve((size_t)(nk - L + 2) * 2);

        u64 k1 = 0, k2 = 0;
        for (int i = 0; i < L; i++) {
            k1 = (k1 * BASE1 + (u64)(unsigned char)sk[i]) % MOD1;
            k2 = (k2 * BASE2 + (u64)(unsigned char)sk[i]) % MOD2;
        }
        present.insert(k1 * MOD2 + k2);

        for (int i = L; i < nk; i++) {
            k1 = (k1 * BASE1 % MOD1 + (u64)(unsigned char)sk[i] + MOD1
                  - (u64)(unsigned char)sk[i - L] * pow1 % MOD1) % MOD1;
            k2 = (k2 * BASE2 % MOD2 + (u64)(unsigned char)sk[i] + MOD2
                  - (u64)(unsigned char)sk[i - L] * pow2 % MOD2) % MOD2;
            present.insert(k1 * MOD2 + k2);
        }

        for (auto it = candidates.begin(); it != candidates.end(); ) {
            if (!present.count(it->first))
                it = candidates.erase(it);
            else
                ++it;
        }
    }

    if (candidates.empty()) return false;

    // Verify survivors to handle any hash collisions
    for (auto& [h, pos] : candidates) {
        string sub = s0.substr(pos, L);
        bool ok = true;
        for (const auto& s : strs) {
            if (s.find(sub) == string::npos) { ok = false; break; }
        }
        if (ok) { result = sub; return true; }
    }

    return false;
}

int main() {
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);

    int n;
    cin >> n;

    vector<string> strs(n);
    for (int i = 0; i < n; i++) cin >> strs[i];

    if (n == 0) { cout << "\n"; return 0; }

    // Process shortest string first to minimize initial candidate set
    vector<int> order(n);
    iota(order.begin(), order.end(), 0);
    sort(order.begin(), order.end(), [&](int a, int b) {
        return strs[a].size() < strs[b].size();
    });

    int minLen = (int)strs[order[0]].size();

    // Binary search on answer length
    int lo = 0, hi = minLen;
    string result;

    while (lo <= hi) {
        int mid = (lo + hi) / 2;
        string candidate;
        if (check(strs, order, mid, candidate)) {
            result = candidate;
            lo = mid + 1;
        } else {
            hi = mid - 1;
        }
    }

    cout << result << "\n";
    return 0;
}
