// ==================== Exact TSP for tiny N ====================

vector<int> exactTSP(int n, const vector<float>& x, const vector<float>& y) {
    // Bitmask DP, n <= 20
    int states = 1 << n;
    vector<vector<float>> dp(states, vector<float>(n, 1e30f));
    vector<vector<int>> parent(states, vector<int>(n, -1));
    
    dp[1][0] = 0.0f;
    
    for (int mask = 1; mask < states; mask++) {
        for (int u = 0; u < n; u++) {
            if (!(mask & (1 << u))) continue;
            if (dp[mask][u] >= 1e29f) continue;
            for (int v = 0; v < n; v++) {
                if (mask & (1 << v)) continue;
                int nmask = mask | (1 << v);
                float dx = x[u] - x[v], dy = y[u] - y[v];
                float d = sqrtf(dx*dx + dy*dy);
                float nd = dp[mask][u] + d;
                if (nd < dp[nmask][v]) {
                    dp[nmask][v] = nd;
                    parent[nmask][v] = u;
                }
            }
        }
    }
    
    int fullMask = states - 1;
    float best = 1e30f;
    int last = -1;
    for (int v = 1; v < n; v++) {
        float dx = x[v] - x[0], dy = y[v] - y[0];
        float total = dp[fullMask][v] + sqrtf(dx*dx + dy*dy);
        if (total < best) { best = total; last = v; }
    }
    
    vector<int> tour;
    int mask = fullMask, cur = last;
    while (cur != -1) {
        tour.push_back(cur);
        int prev = parent[mask][cur];
        mask ^= (1 << cur);
        cur = prev;
    }
    reverse(tour.begin(), tour.end());
    return tour;
}

// ==================== Grid-based spatial index ====================

struct GridIndex {
    float minX, minY, maxX, maxY;
    float cellW, cellH;
    int gridW, gridH;
    vector<vector<int>> cells;
    const vector<float>* px;
    const vector<float>* py;
    int n;
    
    void build(const vector<float>& x, const vector<float>& y, int N, int targetCellSize = 4) {
        px = &x; py = &y; n = N;
        minX = *min_element(x.begin(), x.end());
        maxX = *max_element(x.begin(), x.end());
        minY = *min_element(y.begin(), y.end());
        maxY = *max_element(y.begin(), y.end());
        
        float rangeX = maxX - minX + 1e-6f;
        float rangeY = maxY - minY + 1e-6f;
        
        // Choose grid size
        int totalCells = max(1, (int)(N / targetCellSize));
        float ratio = rangeX / rangeY;
        gridH = max(1, (int)sqrt((double)totalCells / ratio));
        gridW = max(1, totalCells / gridH);
        // Cap grid size to avoid memory issues
        long long maxGrid = 50000000LL;
        while ((long long)gridW * gridH > maxGrid) {
            gridW = max(1, gridW/2);
            gridH = max(1, gridH/2);
        }
        
        cellW = rangeX / gridW;
        cellH = rangeY / gridH;
        
        cells.assign((long long)gridW * gridH, vector<int>());
        for (int i = 0; i < N; i++) {
            int cx = min((int)((x[i]-minX)/cellW), gridW-1);
            int cy = min((int)((y[i]-minY)/cellH), gridH-1);
            cells[cy*gridW+cx].push_back(i);
        }
    }
    
    // Find nearest unvisited neighbor
    int nearestUnvisited(int query, const vector<bool>& visited) const {
        float qx = (*px)[query], qy = (*py)[query];
        int cx = min((int)((qx-minX)/cellW), gridW-1);
        int cy = min((int)((qy-minY)/cellH), gridH-1);
        
        float bestDist = 1e30f;
        int bestIdx = -1;
        
        int maxRadius = max(gridW, gridH);
        for (int r = 0; r <= maxRadius && bestIdx == -1; r++) {
            // Early termination: if min possible dist in ring > bestDist
            float minCellDist = max(0.0f, (r-1) * min(cellW, cellH));
            if (minCellDist * minCellDist > bestDist && bestIdx != -1) break;
            
            int x0 = max(0, cx-r), x1 = min(gridW-1, cx+r);
            int y0 = max(0, cy-r), y1 = min(gridH-1, cy+r);
            
            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    // Only process border of ring when r>0
                    if (r > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                    for (int idx : cells[gy*gridW+gx]) {
                        if (visited[idx]) continue;
                        float dx = (*px)[idx]-qx, dy = (*py)[idx]-qy;
                        float d = dx*dx+dy*dy;
                        if (d < bestDist) { bestDist = d; bestIdx = idx; }
                    }
                }
            }
            
            if (bestIdx != -1) {
                // Check if we can do better in next ring
                float nextMinDist = max(0.0f, r * min(cellW, cellH));
                if (nextMinDist * nextMinDist > bestDist) break;
            }
        }
        return bestIdx;
    }
};

// ==================== Greedy nearest neighbor ====================

vector<int> greedyNN(int n, const vector<float>& x, const vector<float>& y) {
    GridIndex grid;
    int cellSize = (n > 1000000) ? 8 : 4;
    grid.build(x, y, n, cellSize);
    
    vector<bool> visited(n, false);
    vector<int> tour;
    tour.reserve(n);
    
    int cur = 0;
    visited[0] = true;
    tour.push_back(0);
    
    for (int step = 1; step < n; step++) {
        int next = grid.nearestUnvisited(cur, visited);
        if (next == -1) {
            // fallback: find any unvisited
            for (int i = 0; i < n; i++) if (!visited[i]) { next = i; break; }
        }
        visited[next] = true;
        tour.push_back(next);
        cur = next;
    }
    return tour;
}

inline float dist2(int a, int b, const vector<float>& x, const vector<float>& y) {
    float dx = x[a]-x[b], dy = y[a]-y[b];

    float dx = x[a]-x[b], dy = y[a]-y[b];

}

void twoOptSimple(vector<int>& tour, const vector<float>& x, const vector<float>& y, double timeLimit) {

}

// ==================== Or-opt (move single cities) ====================
void orOpt(vector<int>& tour, const vector<float>& x, const vector<float>& y, double timeLimit) {
    int n = tour.size();
    bool improved = true;
    while (improved && elapsed() < timeLimit) {
        improved = false;
        for (int i = 0; i < n && elapsed() < timeLimit; i++) {
            int prev = (i-1+n)%n, next = (i+1)%n;
            int city = tour[i];
            int pc = tour[prev], nc = tour[next];
            float removeCost = distf(pc, city, x, y) + distf(city, nc, x, y) - distf(pc, nc, x, y);
            
            // Try inserting after each other position
            for (int j = 0; j < n; j++) {
                if (j == prev || j == i) continue;
                int jn = (j+1)%n;
                if (jn == i) continue;
                int jc = tour[j], jnc = tour[jn];
                float insertCost = distf(jc, city, x, y) + distf(city, jnc, x, y) - distf(jc, jnc, x, y);
                if (insertCost < removeCost - 1e-6f) {
                    // Remove city from position i and insert after j
                    tour.erase(tour.begin()+i);
                    int ins = (j > i) ? j : j+1;
                    if (ins >= (int)tour.size()) ins = tour.size();
                    tour.insert(tour.begin()+ins, city);
                    improved = true;
                    // Recompute i
                    i = max(0, i-1);
                    break;
                }
            }
        }
    }
}

use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();

                    parent[nb] = node;
                    queue.push_back(nb);

                    let fx = ddx / dist * force;
                    let fy = ddy / dist * force;
                    dx[i] += fx;
                    dy[i] += fy;
                    dx[j] -= fx;
                    dy[j] -= fy;

            let fy = ddy / dist * force;
            dx[u] += fx;
            dy[u] += fy;
            dx[v] -= fx;
            dy[v] -= fy;
        }

            let scale = (disp.min(temp)) / disp;
            x[i] += dx[i] * scale;
            y[i] += dy[i] * scale;
        }

        let _ = iter; // suppress warning
    }

        }).collect();

                let dx = x[nd] - cx;
                let dy = y[nd] - cy;

        let mut row_idx = 0;
        let mut row_max_r = 0.0f64;
        let mut cur_x = 0.0f64;
        let mut cur_y = 0.0f64;

        for ci in 0..num_comps {
            let r = comp_radii[ci];
            if col_idx == grid_cols {
                col_idx = 0;
                cur_y += row_max_r * 2.0 + k * 2.0;
                cur_x = 0.0;
                row_max_r = 0.0;
                row_idx += 1;
            }
            offsets.push((cur_x + r, cur_y + r));
            cur_x += r * 2.0 + k * 2.0;
            if r > row_max_r { row_max_r = r; }
            col_idx += 1;
        }

        for ci in 0..num_comps {
            let (cx, cy) = comp_centers[ci];
            let (tx, ty) = offsets[ci];
            let ox = tx - cx;
            let oy = ty - cy;
            for &nd in &comp_nodes[ci] {
                x[nd] += ox;
                y[nd] += oy;
            }
        }
    }

    // Output
    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}
