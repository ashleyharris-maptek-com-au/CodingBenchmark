using System;
using System.Globalization;
using System.IO;
using System.Text;

public static class Program
{
    // Strategy:
    // - Always: consider only the vertical column at the processing plant (x=0, y=0).
    //   The topmost non-empty voxel is guaranteed "exposed from above".
    //   If it's valuable (value>0), "dump" it at the plant with a zero-length path (0 steps).
    //
    // Adaptive logic:
    // - For larger instances, do exactly one best-safe move (or none).
    // - For smaller instances, optionally mine multiple valuable voxels in the plant column
    //   by repeatedly taking the next exposed voxel below (still 0-step moves).
    //
    // This stays memory-safe for extreme world sizes because it only reads the header and the first Z bytes
    // (the (0,0) column) and then exits.
    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput(1 << 20));

        int xCount = reader.ReadInt();
        if (xCount == int.MinValue) return;
        int yCount = reader.ReadInt();
        int zCount = reader.ReadInt();

        // Block sizes exist for completeness; not required for this strategy.
        reader.ReadToken(); // XBlockSize
        reader.ReadToken(); // YBlockSize
        reader.ReadToken(); // ZBlockSize

        if (xCount <= 0 || yCount <= 0 || zCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        // By input order (x major, y next, z minor), the next zCount bytes are exactly the column at (0,0).
        byte[] col00 = reader.ReadBytesExact(zCount);

        // Find the topmost occupied voxel (mass>0). It's exposed from above by definition.
        int topZ = -1;
        for (int z = zCount - 1; z >= 0; z--)
        {
            byte d = col00[z];
            int mass = d & 0x0F;
            if (mass > 0)
            {
                topZ = z;
                break;
            }
        }

        if (topZ < 0)
        {
            Console.WriteLine(0);
            return;
        }

        // Adaptive: if tiny, mine all valuable voxels in this column (top-down) with 0-step moves.
        // Otherwise, mine only the topmost valuable voxel.
        long xy = (long)xCount * (long)yCount;
        bool mineMultiple = xy <= 50_000 && zCount <= 50_000; // conservative

        var sb = new StringBuilder();
        int opCount = 0;

        for (int z = topZ; z >= 0; z--)
        {
            byte d = col00[z];
            int mass = d & 0x0F;
            if (mass <= 0) continue;

            int value = (d >> 4) & 0x0F;

            // Avoid damaging the plant with junk ore (value==0).
            if (value <= 0) { if (!mineMultiple) break; else continue; }

            // Operation line with N=0 steps:
            // "0 digX digY digZ destX destY"
            sb.Append("0 0 0 ").Append(z).Append(" 0 0\n");
            opCount++;

            if (!mineMultiple) break;
        }

        if (opCount == 0)
        {
            Console.WriteLine(0);
            return;
        }

        Console.WriteLine(opCount);
        Console.Write(sb.ToString());
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer;
        private int _len;
        private int _pos;

        public FastReader(Stream stream, int bufferSize = 1 << 20)
        {
            _stream = stream;
            _buffer = new byte[Math.Max(4096, bufferSize)];
        }

        private int ReadRawByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_pos++];
        }

        public int ReadInt()
        {
            int c;
            do
            {
                c = ReadRawByte();
                if (c < 0) return int.MinValue;
            } while (c <= 32);

            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadRawByte();
            }

            int val = 0;
            while (c > 32 && c >= 0)
            {
                val = val * 10 + (c - '0');
                c = ReadRawByte();
            }
            return val * sign;
        }

        public string ReadToken()
        {
            int c;
            do
            {
                c = ReadRawByte();
                if (c < 0) return string.Empty;
            } while (c <= 32);

            var sb = new StringBuilder(32);
            while (c > 32 && c >= 0)
            {
                sb.Append((char)c);
                c = ReadRawByte();
            }
            return sb.ToString();
        }

        public byte[] ReadBytesExact(int count)
        {
            if (count <= 0) return Array.Empty<byte>();
            byte[] result = new byte[count];
            int written = 0;

            // First exhaust any remaining buffered data.
            while (written < count)
            {
                if (_pos >= _len)
                {
                    _len = _stream.Read(_buffer, 0, _buffer.Length);
                    _pos = 0;
                    if (_len <= 0) break;
                }

                int take = Math.Min(count - written, _len - _pos);
                Buffer.BlockCopy(_buffer, _pos, result, written, take);
                _pos += take;
                written += take;
            }

            if (written != count)
            {
                // Input truncated; still output a valid (no-op) solution.
                Array.Resize(ref result, written);
            }

            return result;
        }
    }
}
