// Approximate CSG Union via adaptive voxelization + flood fill + surface extraction.
// - Handles very large meshes by trading geometric fidelity for robustness and scalability.
// - Produces a watertight, manifold triangle mesh surface of the voxel-union volume.
//
// Input: two watertight triangle meshes A and B (0-indexed faces).
// Output: triangle mesh approximating A ∪ B.
//
// Notes:
// - This is NOT an exact triangle-mesh boolean; it is a volumetric boolean.
// - Resolution adapts to input size and available memory; higher resolution yields better fidelity.
// - Uses only C++20 standard library; compiles with MSVC.

#include <algorithm>
#include <array>
#include <cassert>
#include <cfloat>
#include <cstdint>
#include <cstdlib>
#include <deque>
#include <iomanip>
#include <iostream>
#include <limits>
#include <string>
#include <tuple>
#include <utility>
#include <unordered_map>
#include <vector>

struct Vec3 {
  double x = 0, y = 0, z = 0;
  Vec3() = default;
  Vec3(double xx, double yy, double zz) : x(xx), y(yy), z(zz) {}
  Vec3 operator+(const Vec3& o) const { return {x + o.x, y + o.y, z + o.z}; }
  Vec3 operator-(const Vec3& o) const { return {x - o.x, y - o.y, z - o.z}; }
  Vec3 operator*(double s) const { return {x * s, y * s, z * s}; }
};

static inline Vec3 vmin(const Vec3& a, const Vec3& b) {
  return {std::min(a.x, b.x), std::min(a.y, b.y), std::min(a.z, b.z)};
}
static inline Vec3 vmax(const Vec3& a, const Vec3& b) {
  return {std::max(a.x, b.x), std::max(a.y, b.y), std::max(a.z, b.z)};
}

struct Tri {
  int a = 0, b = 0, c = 0;
};

struct AABB {
  Vec3 lo{+DBL_MAX, +DBL_MAX, +DBL_MAX};
  Vec3 hi{-DBL_MAX, -DBL_MAX, -DBL_MAX};
  void expand(const Vec3& p) { lo = vmin(lo, p); hi = vmax(hi, p); }
  void expand(const AABB& b) { lo = vmin(lo, b.lo); hi = vmax(hi, b.hi); }
};

// Triangle-box overlap test (Akenine-Möller) adapted to double.
// Returns true if triangle intersects axis-aligned box (center, half-size).
static inline bool triBoxOverlap(const Vec3& boxCenter, const Vec3& boxHalfSize,
                                 const Vec3& v0, const Vec3& v1, const Vec3& v2) {
  // Translate triangle as conceptually moving box to origin.
  Vec3 tv0 = v0 - boxCenter;
  Vec3 tv1 = v1 - boxCenter;
  Vec3 tv2 = v2 - boxCenter;

  Vec3 e0 = tv1 - tv0;
  Vec3 e1 = tv2 - tv1;
  Vec3 e2 = tv0 - tv2;

  auto axisTest = [&](double a, double b, double fa, double fb, double v0a, double v0b,
                      double v1a, double v1b, double v2a, double v2b, double boxha,
                      double boxhb) -> bool {
    double p0 = a * v0a + b * v0b;
    double p1 = a * v1a + b * v1b;
    double p2 = a * v2a + b * v2b;
    double minp = std::min(p0, std::min(p1, p2));
    double maxp = std::max(p0, std::max(p1, p2));
    double rad = fa * boxha + fb * boxhb;
    return !(minp > rad || maxp < -rad);
  };

  // 9 tests for cross-products of triangle edges and the box axes.
  {
    double fex = std::abs(e0.x), fey = std::abs(e0.y), fez = std::abs(e0.z);
    if (!axisTest( e0.z, -e0.y, fez, fey, tv0.y, tv0.z, tv1.y, tv1.z, tv2.y, tv2.z, boxHalfSize.y, boxHalfSize.z)) return false;
    if (!axisTest(-e0.z,  e0.x, fez, fex, tv0.x, tv0.z, tv1.x, tv1.z, tv2.x, tv2.z, boxHalfSize.x, boxHalfSize.z)) return false;
    if (!axisTest( e0.y, -e0.x, fey, fex, tv0.x, tv0.y, tv1.x, tv1.y, tv2.x, tv2.y, boxHalfSize.x, boxHalfSize.y)) return false;
  }
  {
    double fex = std::abs(e1.x), fey = std::abs(e1.y), fez = std::abs(e1.z);
    if (!axisTest( e1.z, -e1.y, fez, fey, tv0.y, tv0.z, tv1.y, tv1.z, tv2.y, tv2.z, boxHalfSize.y, boxHalfSize.z)) return false;
    if (!axisTest(-e1.z,  e1.x, fez, fex, tv0.x, tv0.z, tv1.x, tv1.z, tv2.x, tv2.z, boxHalfSize.x, boxHalfSize.z)) return false;
    if (!axisTest( e1.y, -e1.x, fey, fex, tv0.x, tv0.y, tv1.x, tv1.y, tv2.x, tv2.y, boxHalfSize.x, boxHalfSize.y)) return false;
  }
  {
    double fex = std::abs(e2.x), fey = std::abs(e2.y), fez = std::abs(e2.z);
    if (!axisTest( e2.z, -e2.y, fez, fey, tv0.y, tv0.z, tv1.y, tv1.z, tv2.y, tv2.z, boxHalfSize.y, boxHalfSize.z)) return false;
    if (!axisTest(-e2.z,  e2.x, fez, fex, tv0.x, tv0.z, tv1.x, tv1.z, tv2.x, tv2.z, boxHalfSize.x, boxHalfSize.z)) return false;
    if (!axisTest( e2.y, -e2.x, fey, fex, tv0.x, tv0.y, tv1.x, tv1.y, tv2.x, tv2.y, boxHalfSize.x, boxHalfSize.y)) return false;
  }

  // Test overlap in the {x,y,z}-directions.
  auto min3 = [](double a, double b, double c) { return std::min(a, std::min(b, c)); };
  auto max3 = [](double a, double b, double c) { return std::max(a, std::max(b, c)); };
  if (min3(tv0.x, tv1.x, tv2.x) > boxHalfSize.x || max3(tv0.x, tv1.x, tv2.x) < -boxHalfSize.x) return false;
  if (min3(tv0.y, tv1.y, tv2.y) > boxHalfSize.y || max3(tv0.y, tv1.y, tv2.y) < -boxHalfSize.y) return false;
  if (min3(tv0.z, tv1.z, tv2.z) > boxHalfSize.z || max3(tv0.z, tv1.z, tv2.z) < -boxHalfSize.z) return false;

  // Test if the box intersects the plane of the triangle.
  // Compute plane equation of triangle: normal * x + d = 0
  Vec3 a = tv1 - tv0;
  Vec3 b = tv2 - tv0;
  Vec3 normal{
      a.y * b.z - a.z * b.y,
      a.z * b.x - a.x * b.z,
      a.x * b.y - a.y * b.x,
  };

  auto planeBoxOverlap = [&](const Vec3& n, const Vec3& v, const Vec3& maxbox) -> bool {
    Vec3 vminp, vmaxp;
    if (n.x > 0.0) { vminp.x = -maxbox.x - v.x; vmaxp.x =  maxbox.x - v.x; }
    else          { vminp.x =  maxbox.x - v.x; vmaxp.x = -maxbox.x - v.x; }
    if (n.y > 0.0) { vminp.y = -maxbox.y - v.y; vmaxp.y =  maxbox.y - v.y; }
    else          { vminp.y =  maxbox.y - v.y; vmaxp.y = -maxbox.y - v.y; }
    if (n.z > 0.0) { vminp.z = -maxbox.z - v.z; vmaxp.z =  maxbox.z - v.z; }
    else          { vminp.z =  maxbox.z - v.z; vmaxp.z = -maxbox.z - v.z; }
    auto dot = [](const Vec3& a, const Vec3& b) { return a.x * b.x + a.y * b.y + a.z * b.z; };
    if (dot(n, vminp) > 0.0) return false;
    if (dot(n, vmaxp) >= 0.0) return true;
    return false;
  };

  return planeBoxOverlap(normal, tv0, boxHalfSize);
}

struct Mesh {
  std::vector<Vec3> v;
  std::vector<Tri> f;
  AABB bounds() const {
    AABB b;
    for (const auto& p : v) b.expand(p);
    return b;
  }
};

struct Grid {
  int nx = 0, ny = 0, nz = 0;     // voxel cells
  Vec3 origin;                   // min corner of voxel grid in world space
  Vec3 cell;                     // cell size
};

// Bitset-like storage for large 3D grids.
struct BitGrid {
  int nx = 0, ny = 0, nz = 0;
  std::vector<uint64_t> data;
  void init(int x, int y, int z) {
    nx = x; ny = y; nz = z;
    const uint64_t bits = static_cast<uint64_t>(nx) * ny * nz;
    data.assign(static_cast<size_t>((bits + 63) / 64), 0ULL);
  }
  inline uint64_t idx(int x, int y, int z) const {
    return (static_cast<uint64_t>(z) * ny + y) * nx + x;
  }
  inline bool get(int x, int y, int z) const {
    uint64_t i = idx(x, y, z);
    return (data[static_cast<size_t>(i >> 6)] >> (i & 63)) & 1ULL;
  }
  inline void set(int x, int y, int z, bool val) {
    uint64_t i = idx(x, y, z);
    uint64_t& w = data[static_cast<size_t>(i >> 6)];
    uint64_t mask = 1ULL << (i & 63);
    if (val) w |= mask;
    else w &= ~mask;
  }
};

static inline int clampi(int v, int lo, int hi) { return std::max(lo, std::min(hi, v)); }

static Grid makeGrid(const AABB& sceneBounds, uint64_t approxTriCount) {
  // Pad bounds to ensure an outside region for flood-fill.
  Vec3 size = sceneBounds.hi - sceneBounds.lo;
  double maxDim = std::max(size.x, std::max(size.y, size.z));
  if (!(maxDim > 0)) maxDim = 1.0;

  // Heuristic: choose resolution based on triangle count, bounded for memory/time.
  // - small meshes get higher res (better fidelity)
  // - huge meshes get moderate res to remain fast
  int base = 128;
  if (approxTriCount < 5'000) base = 256;
  else if (approxTriCount < 100'000) base = 192;
  else if (approxTriCount < 1'000'000) base = 160;
  else base = 128;

  // Keep voxel count reasonable: cap largest axis to <= 256, floor to >= 64.
  int maxAxis = clampi(base, 64, 256);

  // Scale per-axis to match scene aspect ratio.
  double sx = size.x / maxDim;
  double sy = size.y / maxDim;
  double sz = size.z / maxDim;
  int nx = std::max(16, static_cast<int>(std::round(maxAxis * sx)));
  int ny = std::max(16, static_cast<int>(std::round(maxAxis * sy)));
  int nz = std::max(16, static_cast<int>(std::round(maxAxis * sz)));

  // Add a 2-voxel padding border for flood fill safety.
  nx += 4; ny += 4; nz += 4;

  Vec3 paddedLo = sceneBounds.lo - Vec3{maxDim * 0.02, maxDim * 0.02, maxDim * 0.02};
  Vec3 paddedHi = sceneBounds.hi + Vec3{maxDim * 0.02, maxDim * 0.02, maxDim * 0.02};
  Vec3 paddedSize = paddedHi - paddedLo;

  Grid g;
  g.nx = nx; g.ny = ny; g.nz = nz;
  g.origin = paddedLo;
  g.cell = Vec3{paddedSize.x / nx, paddedSize.y / ny, paddedSize.z / nz};
  return g;
}

static inline Vec3 cellCenter(const Grid& g, int x, int y, int z) {
  return g.origin + Vec3{(x + 0.5) * g.cell.x, (y + 0.5) * g.cell.y, (z + 0.5) * g.cell.z};
}

// Rasterize surface triangles into a boundary occupancy grid: boundary=1 for any voxel intersected by triangles.
static void rasterizeSurfaceBoundary(const Mesh& m, const Grid& g, BitGrid& boundary) {
  boundary.init(g.nx, g.ny, g.nz);

  Vec3 half{g.cell.x * 0.5, g.cell.y * 0.5, g.cell.z * 0.5};

  for (const auto& t : m.f) {
    const Vec3& p0 = m.v[t.a];
    const Vec3& p1 = m.v[t.b];
    const Vec3& p2 = m.v[t.c];
    AABB tb;
    tb.expand(p0); tb.expand(p1); tb.expand(p2);

    // Convert tri AABB to voxel range.
    auto toCell = [&](double coord, double o, double cs, int n) -> int {
      double u = (coord - o) / cs;
      return static_cast<int>(std::floor(u));
    };
    int x0 = clampi(toCell(tb.lo.x, g.origin.x, g.cell.x, g.nx) - 1, 0, g.nx - 1);
    int y0 = clampi(toCell(tb.lo.y, g.origin.y, g.cell.y, g.ny) - 1, 0, g.ny - 1);
    int z0 = clampi(toCell(tb.lo.z, g.origin.z, g.cell.z, g.nz) - 1, 0, g.nz - 1);
    int x1 = clampi(toCell(tb.hi.x, g.origin.x, g.cell.x, g.nx) + 1, 0, g.nx - 1);
    int y1 = clampi(toCell(tb.hi.y, g.origin.y, g.cell.y, g.ny) + 1, 0, g.ny - 1);
    int z1 = clampi(toCell(tb.hi.z, g.origin.z, g.cell.z, g.nz) + 1, 0, g.nz - 1);

    for (int z = z0; z <= z1; ++z) {
      for (int y = y0; y <= y1; ++y) {
        for (int x = x0; x <= x1; ++x) {
          if (boundary.get(x, y, z)) continue;
          Vec3 c = cellCenter(g, x, y, z);
          if (triBoxOverlap(c, half, p0, p1, p2)) {
            boundary.set(x, y, z, true);
          }
        }
      }
    }
  }
}

// Given boundary voxels for a watertight surface, classify solid interior by flood filling "outside" in empty space.
// Convention:
// - boundary=1 blocks flood fill
// - outside=visited empty cells reachable from borders
// - solid = !outside && !boundary
static void classifySolidFromBoundary(const Grid& g, const BitGrid& boundary, BitGrid& solid) {
  solid.init(g.nx, g.ny, g.nz);
  BitGrid outside;
  outside.init(g.nx, g.ny, g.nz);

  auto pushIf = [&](std::deque<std::array<int,3>>& q, int x, int y, int z) {
    if (x < 0 || y < 0 || z < 0 || x >= g.nx || y >= g.ny || z >= g.nz) return;
    if (boundary.get(x, y, z)) return;
    if (outside.get(x, y, z)) return;
    outside.set(x, y, z, true);
    q.push_back({x, y, z});
  };

  std::deque<std::array<int,3>> q;
  // Seed from all border cells.
  for (int x = 0; x < g.nx; ++x) {
    for (int y = 0; y < g.ny; ++y) {
      pushIf(q, x, y, 0);
      pushIf(q, x, y, g.nz - 1);
    }
  }
  for (int x = 0; x < g.nx; ++x) {
    for (int z = 0; z < g.nz; ++z) {
      pushIf(q, x, 0, z);
      pushIf(q, x, g.ny - 1, z);
    }
  }
  for (int y = 0; y < g.ny; ++y) {
    for (int z = 0; z < g.nz; ++z) {
      pushIf(q, 0, y, z);
      pushIf(q, g.nx - 1, y, z);
    }
  }

  const int dx[6] = {1,-1,0,0,0,0};
  const int dy[6] = {0,0,1,-1,0,0};
  const int dz[6] = {0,0,0,0,1,-1};

  while (!q.empty()) {
    auto [x,y,z] = q.front();
    q.pop_front();
    for (int k = 0; k < 6; ++k) {
      pushIf(q, x + dx[k], y + dy[k], z + dz[k]);
    }
  }

  for (int z = 0; z < g.nz; ++z) {
    for (int y = 0; y < g.ny; ++y) {
      for (int x = 0; x < g.nx; ++x) {
        bool isSolid = (!boundary.get(x, y, z)) && (!outside.get(x, y, z));
        solid.set(x, y, z, isSolid);
      }
    }
  }
}

// Extract surface of binary voxel volume using a simple "block surface" triangulation.
// For each solid voxel, emit two triangles per exposed face. This is watertight and manifold.
struct OutMesh {
  std::vector<Vec3> v;
  std::vector<Tri> f;
};

static OutMesh extractSurface(const Grid& g, const BitGrid& solid) {
  OutMesh out;
  out.v.reserve(1024);
  out.f.reserve(1024);

  // Deduplicate vertices: all vertices are grid corners.
  std::unordered_map<uint64_t, int> cornerToIndex;
  cornerToIndex.reserve(1024);

  auto key3 = [](uint32_t x, uint32_t y, uint32_t z) -> uint64_t {
    return static_cast<uint64_t>(x) | (static_cast<uint64_t>(y) << 21) | (static_cast<uint64_t>(z) << 42);
  };

  auto corner = [&](int x, int y, int z) -> Vec3 {
    return g.origin + Vec3{x * g.cell.x, y * g.cell.y, z * g.cell.z};
  };

  auto cornerIndex = [&](int x, int y, int z) -> int {
    uint64_t k = key3(static_cast<uint32_t>(x), static_cast<uint32_t>(y), static_cast<uint32_t>(z));
    auto it = cornerToIndex.find(k);
    if (it != cornerToIndex.end()) return it->second;
    int idx = static_cast<int>(out.v.size());
    out.v.push_back(corner(x, y, z));
    cornerToIndex.emplace(k, idx);
    return idx;
  };

  auto addQuad = [&](int x0, int y0, int z0, int x1, int y1, int z1, int x2, int y2, int z2, int x3, int y3, int z3) {
    int i0 = cornerIndex(x0, y0, z0);
    int i1 = cornerIndex(x1, y1, z1);
    int i2 = cornerIndex(x2, y2, z2);
    int i3 = cornerIndex(x3, y3, z3);
    out.f.push_back({i0, i1, i2});
    out.f.push_back({i0, i2, i3});
  };

  for (int z = 0; z < g.nz; ++z) {
    for (int y = 0; y < g.ny; ++y) {
      for (int x = 0; x < g.nx; ++x) {
        if (!solid.get(x, y, z)) continue;

        // Check each of 6 faces; if neighbor is empty/outside, emit that face.
        auto empty = [&](int nx, int ny, int nz) -> bool {
          if (nx < 0 || ny < 0 || nz < 0 || nx >= g.nx || ny >= g.ny || nz >= g.nz) return true;
          return !solid.get(nx, ny, nz);
        };

        // Voxel cube corners: (x,y,z) to (x+1,y+1,z+1) in grid-corner space.
        // Wind faces CCW as viewed from outside.
        int x1 = x + 1, y1 = y + 1, z1 = z + 1;

        // -X face (normal -X): (x,y,z)->(x,y,z1)->(x,y1,z1)->(x,y1,z)
        if (empty(x - 1, y, z)) addQuad(x, y, z,  x, y, z1,  x, y1, z1,  x, y1, z);
        // +X face (normal +X): (x1,y,z)->(x1,y1,z)->(x1,y1,z1)->(x1,y,z1)
        if (empty(x + 1, y, z)) addQuad(x1, y, z,  x1, y1, z,  x1, y1, z1,  x1, y, z1);
        // -Y face (normal -Y): (x,y,z)->(x1,y,z)->(x1,y,z1)->(x,y,z1)
        if (empty(x, y - 1, z)) addQuad(x, y, z,  x1, y, z,  x1, y, z1,  x, y, z1);
        // +Y face (normal +Y): (x,y1,z)->(x,y1,z1)->(x1,y1,z1)->(x1,y1,z)
        if (empty(x, y + 1, z)) addQuad(x, y1, z,  x, y1, z1,  x1, y1, z1,  x1, y1, z);
        // -Z face (normal -Z): (x,y,z)->(x,y1,z)->(x1,y1,z)->(x1,y,z)
        if (empty(x, y, z - 1)) addQuad(x, y, z,  x, y1, z,  x1, y1, z,  x1, y, z);
        // +Z face (normal +Z): (x,y,z1)->(x1,y,z1)->(x1,y1,z1)->(x,y1,z1)
        if (empty(x, y, z + 1)) addQuad(x, y, z1,  x1, y, z1,  x1, y1, z1,  x, y1, z1);
      }
    }
  }

  return out;
}

static Mesh readMesh(std::istream& in, int V, int F) {
  Mesh m;
  m.v.resize(V);
  m.f.resize(F);
  for (int i = 0; i < V; ++i) {
    in >> m.v[i].x >> m.v[i].y >> m.v[i].z;
  }
  for (int i = 0; i < F; ++i) {
    in >> m.f[i].a >> m.f[i].b >> m.f[i].c;
  }
  return m;
}

static void writeMesh(std::ostream& out, const OutMesh& m) {
  out << m.v.size() << " " << m.f.size() << "\n";
  out << std::setprecision(17);
  for (const auto& p : m.v) {
    out << p.x << " " << p.y << " " << p.z << "\n";
  }
  for (const auto& t : m.f) {
    out << t.a << " " << t.b << " " << t.c << "\n";
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  int VA = 0, FA = 0, VB = 0, FB = 0;
  if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;
  Mesh A = readMesh(std::cin, VA, FA);
  Mesh B = readMesh(std::cin, VB, FB);

  AABB scene;
  scene.expand(A.bounds());
  scene.expand(B.bounds());

  const uint64_t triCount = static_cast<uint64_t>(FA) + static_cast<uint64_t>(FB);
  Grid g = makeGrid(scene, triCount);

  BitGrid boundaryA, boundaryB;
  rasterizeSurfaceBoundary(A, g, boundaryA);
  rasterizeSurfaceBoundary(B, g, boundaryB);

  BitGrid solidA, solidB;
  classifySolidFromBoundary(g, boundaryA, solidA);
  classifySolidFromBoundary(g, boundaryB, solidB);

  BitGrid solidU;
  solidU.init(g.nx, g.ny, g.nz);
  for (int z = 0; z < g.nz; ++z) {
    for (int y = 0; y < g.ny; ++y) {
      for (int x = 0; x < g.nx; ++x) {
        solidU.set(x, y, z, solidA.get(x, y, z) || solidB.get(x, y, z));
      }
    }
  }

  OutMesh out = extractSurface(g, solidU);
  writeMesh(std::cout, out);
  return 0;
}
