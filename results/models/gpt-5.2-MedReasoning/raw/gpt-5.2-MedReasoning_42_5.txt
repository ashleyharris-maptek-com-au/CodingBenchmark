OpCapability Shader
OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %in_color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 7
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %in_color "color"
OpName %outColor "outColor"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %in_color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn_void = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%uint_2 = OpConstant %uint 2

%float_0 = OpConstant %float 0
%float_0_015 = OpConstant %float 0.015
%float_0_05 = OpConstant %float 0.05
%float_0_75 = OpConstant %float 0.75

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4float = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_2
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3float = OpTypePointer Input %v3float
%in_color = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4float = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3float = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3float Output

%main = OpFunction %void None %fn_void
%entry = OpLabel

; Load p0, p1 (vec4) and colors c0, c1 (vec3)
%p0_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_1 %int_0
%p1 = OpLoad %v4float %p1_ptr

%c0_ptr = OpAccessChain %ptr_in_v3float %in_color %int_0
%c0 = OpLoad %v3float %c0_ptr
%c1_ptr = OpAccessChain %ptr_in_v3float %in_color %int_1
%c1 = OpLoad %v3float %c1_ptr

; dir4 = p1 - p0
%dir4 = OpFSub %v4float %p1 %p0

; dir2 = dir4.xy
%dirx = OpCompositeExtract %float %dir4 0
%diry = OpCompositeExtract %float %dir4 1
%dir2 = OpCompositeConstruct %v2float %dirx %diry

; perp = normalize(vec2(-dir.y, dir.x))
%neg_diry = OpFNegate %float %diry
%perp_raw = OpCompositeConstruct %v2float %neg_diry %dirx
%perp = OpExtInst %v2float %glsl Normalize %perp_raw

; mid = p0 + 0.75 * dir4
%dir4_075 = OpVectorTimesScalar %v4float %dir4 %float_0_75
%mid = OpFAdd %v4float %p0 %dir4_075

; Offsets
%perp_thick = OpVectorTimesScalar %v2float %perp %float_0_015
%perp_wide = OpVectorTimesScalar %v2float %perp %float_0_05

%ptx = OpCompositeExtract %float %perp_thick 0
%pty = OpCompositeExtract %float %perp_thick 1
%offset_thick4 = OpCompositeConstruct %v4float %ptx %pty %float_0 %float_0
%offset_thick4_neg = OpFNegate %v4float %offset_thick4

%pwx = OpCompositeExtract %float %perp_wide 0
%pwy = OpCompositeExtract %float %perp_wide 1
%offset_wide4 = OpCompositeConstruct %v4float %pwx %pwy %float_0 %float_0
%offset_wide4_neg = OpFNegate %v4float %offset_wide4

; Shaft quad positions: p0?, mid?
%shaft0 = OpFAdd %v4float %p0 %offset_thick4
%shaft1 = OpFAdd %v4float %p0 %offset_thick4_neg
%shaft2 = OpFAdd %v4float %mid %offset_thick4
%shaft3 = OpFAdd %v4float %mid %offset_thick4_neg

; Emit shaft as triangle strip (4 verts)
%out_pos_ptr = OpAccessChain %ptr_out_v4float %gl_out %int_0
OpStore %out_pos_ptr %shaft0
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %shaft1
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %shaft2
OpStore %outColor %c0
OpEmitVertex

OpStore %out_pos_ptr %shaft3
OpStore %outColor %c0
OpEmitVertex

OpEndPrimitive

; Arrowhead triangle strip (3 verts): mid?wide, p1 tip
%arrow0 = OpFAdd %v4float %mid %offset_wide4
%arrow1 = OpFAdd %v4float %mid %offset_wide4_neg

OpStore %out_pos_ptr %arrow0
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %arrow1
OpStore %outColor %c1
OpEmitVertex

OpStore %out_pos_ptr %p1
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
