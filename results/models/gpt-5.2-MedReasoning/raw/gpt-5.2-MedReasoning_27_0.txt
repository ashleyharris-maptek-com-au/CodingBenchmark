// Graph k-coloring heuristic solver (C++17/20 compatible, MSVC-friendly).
// Reads: n m k, then m undirected edges (u v), 0-indexed.
// Writes: n integers in [0, k-1] - a valid k-coloring (assumes instance is k-colorable).

#include <algorithm>
#include <chrono>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <vector>

#if defined(_MSC_VER)
#  include <intrin.h>
#endif

using std::uint32_t;
using std::uint64_t;

static inline int popcount64(uint64_t x) {
#if defined(_MSC_VER)
  return static_cast<int>(__popcnt64(x));
#else
  return __builtin_popcountll(x);
#endif
}

static inline int ctz64(uint64_t x) {
#if defined(_MSC_VER)
  unsigned long idx = 0;
  _BitScanForward64(&idx, x);
  return static_cast<int>(idx);
#else
  return __builtin_ctzll(x);
#endif
}

struct GraphCSR {
  uint32_t n = 0;
  uint64_t m_undirected = 0;
  std::vector<uint64_t> off;    // size n+1
  std::vector<uint32_t> adj;    // size 2*m
  std::vector<uint32_t> degree; // size n

  uint64_t begin(uint32_t v) const { return off[v]; }
  uint64_t end(uint32_t v) const { return off[v + 1]; }
};

struct ParsedInput {
  GraphCSR g;
  int k = 0;
};

static ParsedInput read_graph() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  uint32_t n = 0;
  uint64_t m = 0;
  int k = 0;
  if (!(std::cin >> n >> m >> k)) {
    return {};
  }

  std::vector<uint32_t> eu;
  std::vector<uint32_t> ev;
  eu.reserve(static_cast<size_t>(m));
  ev.reserve(static_cast<size_t>(m));

  GraphCSR g;
  g.n = n;
  g.m_undirected = m;
  g.degree.assign(n, 0);

  for (uint64_t i = 0; i < m; ++i) {
    uint32_t u, v;
    std::cin >> u >> v;
    eu.push_back(u);
    ev.push_back(v);
    if (u < n) ++g.degree[u];
    if (v < n) ++g.degree[v];
  }

  g.off.assign(static_cast<size_t>(n) + 1, 0);
  for (uint32_t i = 0; i < n; ++i) g.off[i + 1] = g.off[i] + g.degree[i];

  const uint64_t total_adj = g.off[n];
  g.adj.assign(static_cast<size_t>(total_adj), 0);

  std::vector<uint64_t> cur = g.off;
  for (uint64_t i = 0; i < m; ++i) {
    const uint32_t u = eu[static_cast<size_t>(i)];
    const uint32_t v = ev[static_cast<size_t>(i)];
    if (u < n) g.adj[static_cast<size_t>(cur[u]++)] = v;
    if (v < n) g.adj[static_cast<size_t>(cur[v]++)] = u;
  }

  ParsedInput out;
  out.g = std::move(g);
  out.k = k;
  return out;
}

static inline uint64_t all_color_mask(int k) {
  if (k <= 0) return 0;
  if (k >= 64) return ~0ULL; // only meaningful when callers ensure colors < 64
  return (1ULL << k) - 1ULL;
}

static bool try_repair_one_step(const GraphCSR& g, int k, uint32_t v, std::vector<uint8_t>& color,
                                uint64_t allMask) {
  if (k > 63) return false; // repair logic uses 64-bit masks
  // v currently uncolored. Try to free a color by recoloring one neighbor.
  // This is a fast, local heuristic; it succeeds often on "nearly greedy-colorable" instances.
  // Compute used colors in v's neighborhood.
  uint64_t usedV = 0;
  for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
    uint32_t u = g.adj[static_cast<size_t>(it)];
    uint8_t cu = color[u];
    if (cu != 0xFF) usedV |= (1ULL << cu);
  }
  uint64_t availV = allMask & ~usedV;
  if (availV) {
    color[v] = static_cast<uint8_t>(ctz64(availV));
    return true;
  }

  for (int desired = 0; desired < k; ++desired) {
    // find a neighbor u with color 'desired'
    uint32_t u_found = std::numeric_limits<uint32_t>::max();
    for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
      uint32_t u = g.adj[static_cast<size_t>(it)];
      if (color[u] == static_cast<uint8_t>(desired)) {
        u_found = u;
        break;
      }
    }
    if (u_found == std::numeric_limits<uint32_t>::max()) continue;

    // compute available colors for u_found (excluding v which is currently uncolored)
    uint64_t usedU = 0;
    for (uint64_t jt = g.begin(u_found); jt < g.end(u_found); ++jt) {
      uint32_t w = g.adj[static_cast<size_t>(jt)];
      if (w == v) continue;
      uint8_t cw = color[w];
      if (cw != 0xFF) usedU |= (1ULL << cw);
    }
    uint64_t availU = allMask & ~usedU;
    // Must avoid reusing 'desired' (otherwise no progress)
    availU &= ~(1ULL << desired);
    if (!availU) continue;

    const uint8_t newColorU = static_cast<uint8_t>(ctz64(availU));
    color[u_found] = newColorU;
    color[v] = static_cast<uint8_t>(desired);
    return true;
  }

  return false;
}

static bool greedy_with_restarts(const GraphCSR& g, int k, std::vector<uint8_t>& color,
                                 std::vector<uint32_t> baseOrder, uint64_t timeLimitMs) {
  const uint32_t n = g.n;
  const bool useMask = (k <= 63);
  const uint64_t allMask = useMask ? all_color_mask(k) : 0ULL;

  std::mt19937_64 rng(static_cast<uint64_t>(
      std::chrono::high_resolution_clock::now().time_since_epoch().count()));
  auto start = std::chrono::steady_clock::now();

  // Attempt schedule: mostly deterministic, but with increasing perturbations.
  // Keep it simple: try base order, then shuffled blocks, then full shuffle.
  for (int attempt = 0; attempt < 64; ++attempt) {
    auto now = std::chrono::steady_clock::now();
    uint64_t elapsed =
        static_cast<uint64_t>(std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count());
    if (elapsed > timeLimitMs) break;

    std::vector<uint32_t> order = std::move(baseOrder);
    if (attempt > 0) {
      if (attempt < 8) {
        const size_t block = 2048;
        for (size_t i = 0; i < order.size(); i += block) {
          size_t j = std::min(order.size(), i + block);
          std::shuffle(order.begin() + static_cast<ptrdiff_t>(i), order.begin() + static_cast<ptrdiff_t>(j), rng);
        }
      } else if (attempt < 24) {
        std::shuffle(order.begin(), order.end(), rng);
      } else {
        // Degree-biased shuffle: swap some random pairs.
        std::uniform_int_distribution<uint32_t> dist(0, n - 1);
        for (uint32_t s = 0; s < n / 64 + 1; ++s) {
          uint32_t a = dist(rng), b = dist(rng);
          std::swap(order[a], order[b]);
        }
      }
    }

    color.assign(static_cast<size_t>(n), 0xFF);
    std::vector<uint32_t> seen;
    uint32_t stamp = 1;
    if (!useMask) seen.assign(static_cast<size_t>(k), 0);
    bool ok = true;
    for (uint32_t idx = 0; idx < n; ++idx) {
      const uint32_t v = order[idx];
      if (useMask) {
        uint64_t used = 0;
        for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
          uint32_t u = g.adj[static_cast<size_t>(it)];
          uint8_t cu = color[u];
          if (cu != 0xFF) used |= (1ULL << cu);
        }
        uint64_t avail = allMask & ~used;
        if (avail) {
          color[v] = static_cast<uint8_t>(ctz64(avail));
          continue;
        }
        if (!try_repair_one_step(g, k, v, color, allMask)) {
          ok = false;
          break;
        }
      } else {
        if (++stamp == 0) {
          std::fill(seen.begin(), seen.end(), 0);
          stamp = 1;
        }
        for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
          uint32_t u = g.adj[static_cast<size_t>(it)];
          uint8_t cu = color[u];
          if (cu != 0xFF && cu < static_cast<uint8_t>(k)) seen[cu] = stamp;
        }
        int chosen = -1;
        for (int c = 0; c < k; ++c) {
          if (seen[static_cast<size_t>(c)] != stamp) {
            chosen = c;
            break;
          }
        }
        if (chosen < 0) {
          ok = false;
          break;
        }
        color[v] = static_cast<uint8_t>(chosen);
      }
    }
    if (ok) return true;

    baseOrder = std::move(order); // carry forward the last perturbation as a new base
  }
  return false;
}

struct DSATKey {
  uint32_t v;
  uint16_t sat;
  uint32_t deg;
  uint32_t tie;
};
struct DSATCmp {
  bool operator()(const DSATKey& a, const DSATKey& b) const {
    if (a.sat != b.sat) return a.sat < b.sat;   // max-heap by sat
    if (a.deg != b.deg) return a.deg < b.deg;   // then degree
    return a.tie < b.tie;                       // then tiebreaker
  }
};

static bool dsatur_small(const GraphCSR& g, int k, std::vector<uint8_t>& color, uint64_t timeLimitMs) {
  const uint32_t n = g.n;
  if (k <= 0) return false;
  if (k > 63) return false; // DSATUR implementation uses 64-bit masks.

  const uint64_t allMask = all_color_mask(k);
  auto start = std::chrono::steady_clock::now();

  color.assign(static_cast<size_t>(n), 0xFF);
  std::vector<uint64_t> satMask(static_cast<size_t>(n), 0);

  std::mt19937_64 rng(static_cast<uint64_t>(
      std::chrono::high_resolution_clock::now().time_since_epoch().count()));
  std::uniform_int_distribution<uint32_t> tieDist(0, std::numeric_limits<uint32_t>::max());

  std::priority_queue<DSATKey, std::vector<DSATKey>, DSATCmp> pq;
  for (uint32_t v = 0; v < n; ++v) {
    pq.push(DSATKey{v, 0, g.degree[v], tieDist(rng)});
  }

  uint32_t coloredCount = 0;
  while (coloredCount < n) {
    auto now = std::chrono::steady_clock::now();
    uint64_t elapsed =
        static_cast<uint64_t>(std::chrono::duration_cast<std::chrono::milliseconds>(now - start).count());
    if (elapsed > timeLimitMs) return false;

    DSATKey top{};
    do {
      if (pq.empty()) return false;
      top = pq.top();
      pq.pop();
    } while (color[top.v] != 0xFF || top.sat != static_cast<uint16_t>(popcount64(satMask[top.v])));

    const uint32_t v = top.v;
    uint64_t used = satMask[v];
    uint64_t avail = allMask & ~used;
    if (!avail) return false;

    const uint8_t cv = static_cast<uint8_t>(ctz64(avail));
    color[v] = cv;
    ++coloredCount;

    // Update saturation masks for uncolored neighbors.
    for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
      uint32_t u = g.adj[static_cast<size_t>(it)];
      if (color[u] != 0xFF) continue;
      uint64_t before = satMask[u];
      uint64_t after = before | (1ULL << cv);
      if (after != before) {
        satMask[u] = after;
        pq.push(DSATKey{u, static_cast<uint16_t>(popcount64(after)), g.degree[u], tieDist(rng)});
      }
    }
  }

  return true;
}

static bool verify_coloring(const GraphCSR& g, int k, const std::vector<uint8_t>& color) {
  const uint32_t n = g.n;
  if (static_cast<uint32_t>(color.size()) != n) return false;
  for (uint32_t v = 0; v < n; ++v) {
    if (color[v] == 0xFF || color[v] >= static_cast<uint8_t>(k)) return false;
    for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
      uint32_t u = g.adj[static_cast<size_t>(it)];
      if (u < n && color[u] == color[v]) return false;
    }
  }
  return true;
}

int main() {
  ParsedInput in = read_graph();
  GraphCSR& g = in.g;
  const int k = in.k;
  if (g.n == 0) return 0;

  std::vector<uint8_t> color;

  // Strategy selection.
  // - Small/medium graphs: DSATUR often finds a solution quickly.
  // - Large graphs: fast greedy with restarts and a local 1-step repair.
  const uint32_t n = g.n;
  const uint64_t m2 = g.off.empty() ? 0 : g.off[n];

  const uint64_t totalBudgetMs = 295000; // keep a little margin under 5 minutes
  const uint64_t dsaturBudgetMs = 70000;
  const uint64_t greedyBudgetMs = totalBudgetMs;

  bool ok = false;

  if (k == 1) {
    // Only valid if no edges.
    if (m2 == 0) {
      color.assign(static_cast<size_t>(n), 0);
      ok = true;
    }
  } else if (k >= 2 && k <= 63 && n <= 50000) {
    ok = dsatur_small(g, k, color, dsaturBudgetMs);
  }

  if (!ok) {
    // Build base order: descending degree.
    std::vector<uint32_t> order(static_cast<size_t>(n));
    std::iota(order.begin(), order.end(), 0);
    std::sort(order.begin(), order.end(), [&](uint32_t a, uint32_t b) {
      uint32_t da = g.degree[a], db = g.degree[b];
      if (da != db) return da > db;
      return a < b;
    });
    ok = greedy_with_restarts(g, k, color, std::move(order), greedyBudgetMs);
  }

  // As a last resort, output something (benchmark instances are expected k-colorable).
  if (!ok) {
    // Fallback: plain greedy in input order (may fail; but ensures bounded runtime).
    color.assign(static_cast<size_t>(n), 0xFF);
    if (k <= 63) {
      const uint64_t allMask = all_color_mask(k);
      for (uint32_t v = 0; v < n; ++v) {
        uint64_t used = 0;
        for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
          uint32_t u = g.adj[static_cast<size_t>(it)];
          uint8_t cu = color[u];
          if (cu != 0xFF) used |= (1ULL << cu);
        }
        uint64_t avail = allMask & ~used;
        color[v] = avail ? static_cast<uint8_t>(ctz64(avail)) : 0;
      }
    } else {
      std::vector<uint32_t> seen(static_cast<size_t>(k), 0);
      uint32_t stamp = 1;
      for (uint32_t v = 0; v < n; ++v) {
        if (++stamp == 0) {
          std::fill(seen.begin(), seen.end(), 0);
          stamp = 1;
        }
        for (uint64_t it = g.begin(v); it < g.end(v); ++it) {
          uint32_t u = g.adj[static_cast<size_t>(it)];
          uint8_t cu = color[u];
          if (cu != 0xFF && cu < static_cast<uint8_t>(k)) seen[cu] = stamp;
        }
        int chosen = 0;
        for (int c = 0; c < k; ++c) {
          if (seen[static_cast<size_t>(c)] != stamp) {
            chosen = c;
            break;
          }
        }
        color[v] = static_cast<uint8_t>(chosen);
      }
    }
  }

  // Output colors as integers.
  for (uint32_t v = 0; v < n; ++v) {
    if (v) std::cout << ' ';
    uint32_t cv = (v < color.size() && color[v] != 0xFF) ? static_cast<uint32_t>(color[v]) : 0U;
    if (cv >= static_cast<uint32_t>(k)) cv = 0;
    std::cout << cv;
  }
  std::cout << '\n';

  return 0;
}
