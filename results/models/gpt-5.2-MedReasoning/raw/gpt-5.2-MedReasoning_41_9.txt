%worldPos = OpLoad %v3float %in_worldPos

; Checkerboard: parity = (floor(uv.x*8) + floor(uv.y*8)) & 1
%uv8 = OpVectorTimesScalar %v2float %uv %float_08
%uvFloor = OpExtInst %v2float %glsl Floor %uv8
%uFloor_f = OpCompositeExtract %float %uvFloor 0
%vFloor_f = OpCompositeExtract %float %uvFloor 1

%is_even = OpIEqual %bool %parity_i %int_0
%base = OpSelect %float %is_even %float_09 %float_02
%base_rgb = OpCompositeConstruct %v3float %base %base %base

; Lambertian: factor = 0.5 + 0.5*max(dot(normalize(n), normalize(lightPos-worldPos)), 0)

%lightPtr = OpAccessChain %_ptr_Uniform_v4float %ubo %int_3

%lightVec = OpFSub %v3float %lightPos3 %worldPos
%l = OpExtInst %v3float %glsl Normalize %lightVec
%dotNL = OpDot %float %n %l
%diffuse = OpExtInst %float %glsl FMax %dotNL %float_0
%halfDiffuse = OpFMul %float %float_05 %diffuse
%factor = OpFAdd %float %float_05 %halfDiffuse
%lit_rgb = OpVectorTimesScalar %v3float %base_rgb %factor

%lit_r = OpCompositeExtract %float %lit_rgb 0
%lit_g = OpCompositeExtract %float %lit_rgb 1
%lit_b = OpCompositeExtract %float %lit_rgb 2
%out = OpCompositeConstruct %v4float %lit_r %lit_g %lit_b %float_1
