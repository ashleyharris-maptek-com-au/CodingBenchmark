OpCapability Shader
OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %inColor %gl_out %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 18
OpExecutionMode %main Invocations 1
OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %inColor "color"
OpName %outColor "outColor"
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %inColor Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%int_0 = OpConstant %int 0
%uint_1 = OpConstant %uint 1

%float_0 = OpConstant %float 0
%float_r = OpConstant %float 0.04
%float_edge_scale = OpConstant %float 0.7

%float_dx0 = OpConstant %float 0.04
%float_dy0 = OpConstant %float 0
%float_dx1 = OpConstant %float 0.02
%float_dy1 = OpConstant %float 0.034641016
%float_dx2 = OpConstant %float -0.02
%float_dy2 = OpConstant %float 0.034641016
%float_dx3 = OpConstant %float -0.04
%float_dy3 = OpConstant %float 0
%float_dx4 = OpConstant %float -0.02
%float_dy4 = OpConstant %float -0.034641016
%float_dx5 = OpConstant %float 0.02
%float_dy5 = OpConstant %float -0.034641016

%v3_scale = OpConstantComposite %v3float %float_edge_scale %float_edge_scale %float_edge_scale

%gl_PerVertex_in = OpTypeStruct %v4float
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_1
%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_v4 = OpTypePointer Input %v4float
%gl_in = OpVariable %ptr_in_arr_pv Input

%arr_v3_in = OpTypeArray %v3float %uint_1
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_in_v3 = OpTypePointer Input %v3float
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%inColor = OpVariable %ptr_in_arr_v3 Input

%gl_PerVertex_out = OpTypeStruct %v4float
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v4 = OpTypePointer Output %v4float
%gl_out = OpVariable %ptr_out_pv Output

%ptr_out_v3 = OpTypePointer Output %v3float
%outColor = OpVariable %ptr_out_v3 Output

%main = OpFunction %void None %fn
%entry = OpLabel

; Load center position and color for this point (index 0)
%pos_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%centerPos = OpLoad %v4float %pos_ptr
%col_ptr = OpAccessChain %ptr_in_v3 %inColor %int_0
%centerCol = OpLoad %v3float %col_ptr
%edgeCol = OpFMul %v3float %centerCol %v3_scale

%cx = OpCompositeExtract %float %centerPos 0
%cy = OpCompositeExtract %float %centerPos 1
%cz = OpCompositeExtract %float %centerPos 2
%cw = OpCompositeExtract %float %centerPos 3

; Triangle 0: center, edge0, edge1
%cpos0 = OpCompositeConstruct %v4float %cx %cy %cz %cw
%out_pos_ptr0 = OpAccessChain %ptr_out_v4 %gl_out %int_0
OpStore %out_pos_ptr0 %cpos0
OpStore %outColor %centerCol
OpEmitVertex

%ex0 = OpFAdd %float %cx %float_dx0
%ey0 = OpFAdd %float %cy %float_dy0
%epos0 = OpCompositeConstruct %v4float %ex0 %ey0 %cz %cw
OpStore %out_pos_ptr0 %epos0
OpStore %outColor %edgeCol
OpEmitVertex

%ex1 = OpFAdd %float %cx %float_dx1
%ey1 = OpFAdd %float %cy %float_dy1
%epos1 = OpCompositeConstruct %v4float %ex1 %ey1 %cz %cw
OpStore %out_pos_ptr0 %epos1
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; Triangle 1: center, edge1, edge2
OpStore %out_pos_ptr0 %cpos0
OpStore %outColor %centerCol
OpEmitVertex
OpStore %out_pos_ptr0 %epos1
OpStore %outColor %edgeCol
OpEmitVertex
%ex2 = OpFAdd %float %cx %float_dx2
%ey2 = OpFAdd %float %cy %float_dy2
%epos2 = OpCompositeConstruct %v4float %ex2 %ey2 %cz %cw
OpStore %out_pos_ptr0 %epos2
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; Triangle 2: center, edge2, edge3
OpStore %out_pos_ptr0 %cpos0
OpStore %outColor %centerCol
OpEmitVertex
OpStore %out_pos_ptr0 %epos2
OpStore %outColor %edgeCol
OpEmitVertex
%ex3 = OpFAdd %float %cx %float_dx3
%ey3 = OpFAdd %float %cy %float_dy3
%epos3 = OpCompositeConstruct %v4float %ex3 %ey3 %cz %cw
OpStore %out_pos_ptr0 %epos3
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; Triangle 3: center, edge3, edge4
OpStore %out_pos_ptr0 %cpos0
OpStore %outColor %centerCol
OpEmitVertex
OpStore %out_pos_ptr0 %epos3
OpStore %outColor %edgeCol
OpEmitVertex
%ex4 = OpFAdd %float %cx %float_dx4
%ey4 = OpFAdd %float %cy %float_dy4
%epos4 = OpCompositeConstruct %v4float %ex4 %ey4 %cz %cw
OpStore %out_pos_ptr0 %epos4
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; Triangle 4: center, edge4, edge5
OpStore %out_pos_ptr0 %cpos0
OpStore %outColor %centerCol
OpEmitVertex
OpStore %out_pos_ptr0 %epos4
OpStore %outColor %edgeCol
OpEmitVertex
%ex5 = OpFAdd %float %cx %float_dx5
%ey5 = OpFAdd %float %cy %float_dy5
%epos5 = OpCompositeConstruct %v4float %ex5 %ey5 %cz %cw
OpStore %out_pos_ptr0 %epos5
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

; Triangle 5: center, edge5, edge0 (wrap)
OpStore %out_pos_ptr0 %cpos0
OpStore %outColor %centerCol
OpEmitVertex
OpStore %out_pos_ptr0 %epos5
OpStore %outColor %edgeCol
OpEmitVertex
OpStore %out_pos_ptr0 %epos0
OpStore %outColor %edgeCol
OpEmitVertex
OpEndPrimitive

OpReturn
OpFunctionEnd
