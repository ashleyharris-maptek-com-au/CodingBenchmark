using System;
using System.Collections.Generic;
using System.Globalization;
using System.IO;
using System.Linq;

public static class Program
{
    private readonly struct Box
    {
        public readonly int Index;
        public readonly long W;
        public readonly long H;
        public readonly long D;
        public Box(int index, long w, long h, long d) { Index = index; W = w; H = h; D = d; }
        public long Volume => W * H * D;
        public long MaxDim => Math.Max(W, Math.Max(H, D));
    }

    private readonly struct Placement
    {
        public readonly int Index;
        public readonly long X, Y, Z;
        public Placement(int index, long x, long y, long z) { Index = index; X = x; Y = y; Z = z; }
        public override string ToString() => Index.ToString(CultureInfo.InvariantCulture) + " " +
                                             X.ToString(CultureInfo.InvariantCulture) + " " +
                                             Y.ToString(CultureInfo.InvariantCulture) + " " +
                                             Z.ToString(CultureInfo.InvariantCulture);
    }

    private readonly struct Orient
    {
        public readonly long W, H, D;
        public Orient(long w, long h, long d) { W = w; H = h; D = d; }
        public long Volume => W * H * D;
    }

    private struct Space
    {
        public long X, Y, Z;
        public long W, H, D;
        public long Volume => W * H * D;
        public Space(long x, long y, long z, long w, long h, long d) { X = x; Y = y; Z = z; W = w; H = h; D = d; }
    }

    public static void Main()
    {
        var input = Console.In;
        string? header = input.ReadLine();
        if (header == null)
            return;

        var parts = SplitInts(header);
        if (parts.Count < 4)
            return;

        int n = (int)parts[0];
        long containerW = parts[1], containerH = parts[2], containerD = parts[3];

        var boxes = new List<Box>(n);
        for (int i = 0; i < n; i++)
        {
            string? line = input.ReadLine();
            if (line == null) break;
            var p = SplitInts(line);
            if (p.Count < 3) continue;
            long w = p[0], h = p[1], d = p[2];
            if (w <= 0 || h <= 0 || d <= 0) continue;
            boxes.Add(new Box(i, w, h, d));
        }

        // Filter boxes that cannot possibly fit in any orientation.
        boxes = boxes.Where(b => CanFitAnyOrientation(b, containerW, containerH, containerD)).ToList();

        // Adaptive strategy: fast shelves for large N, guillotine for smaller N.
        // The guillotine approach yields higher quality but can generate many spaces.
        // For very large instances we keep it simple and fast.
        List<Placement> placements;
        if (boxes.Count >= 3500 || (containerW * containerH * containerD) >= 5_000_000_000L)
            placements = PackShelves3D(boxes, containerW, containerH, containerD);
        else
            placements = PackGuillotine3D(boxes, containerW, containerH, containerD);

        var output = Console.Out;
        output.WriteLine(placements.Count.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < placements.Count; i++)
            output.WriteLine(placements[i].ToString());
    }

    private static bool CanFitAnyOrientation(Box b, long W, long H, long D)
    {
        foreach (var o in EnumerateOrientations(b.W, b.H, b.D))
        {
            if (o.W <= W && o.H <= H && o.D <= D) return true;
        }
        return false;
    }

    private static List<long> SplitInts(string s)
    {
        var res = new List<long>(6);
        int i = 0;
        while (i < s.Length)
        {
            while (i < s.Length && char.IsWhiteSpace(s[i])) i++;
            if (i >= s.Length) break;
            bool neg = false;
            if (s[i] == '-') { neg = true; i++; }
            long val = 0;
            bool any = false;
            while (i < s.Length && (uint)(s[i] - '0') <= 9)
            {
                any = true;
                val = val * 10 + (s[i] - '0');
                i++;
            }
            if (any) res.Add(neg ? -val : val);
            while (i < s.Length && !char.IsWhiteSpace(s[i])) i++;
        }
        return res;
    }

    private static IEnumerable<Orient> EnumerateOrientations(long a, long b, long c)
    {
        // Return up to 6 orientations; de-dup when dimensions repeat.
        // Order matters (we often want "largest face" aligned to X/Y in shelves; and tight fits in guillotine).
        Span<long> dims = stackalloc long[3] { a, b, c };
        // We'll generate unique permutations via a small set and a HashSet of packed keys.
        var seen = new HashSet<long>();
        long[][] perms = new[]
        {
            new[]{dims[0],dims[1],dims[2]},
            new[]{dims[0],dims[2],dims[1]},
            new[]{dims[1],dims[0],dims[2]},
            new[]{dims[1],dims[2],dims[0]},
            new[]{dims[2],dims[0],dims[1]},
            new[]{dims[2],dims[1],dims[0]},
        };
        for (int i = 0; i < perms.Length; i++)
        {
            long w = perms[i][0], h = perms[i][1], d = perms[i][2];
            long key = (w & 0x1FFFFFL) | ((h & 0x1FFFFFL) << 21) | ((d & 0x1FFFFFL) << 42);
            // If dims exceed 2^21, fallback to full key via hashing.
            if (w >= (1L << 21) || h >= (1L << 21) || d >= (1L << 21))
                key = HashDims(w, h, d);
            if (seen.Add(key))
                yield return new Orient(w, h, d);
        }
    }

    private static long HashDims(long w, long h, long d)
    {
        unchecked
        {
            long x = 1469598103934665603L;
            x = (x ^ w) * 1099511628211L;
            x = (x ^ h) * 1099511628211L;
            x = (x ^ d) * 1099511628211L;
            return x;
        }
    }

    // -------------------------------
    // Mode 1: 3D Shelves (fast)
    // -------------------------------
    private static List<Placement> PackShelves3D(List<Box> boxes, long W, long H, long D)
    {
        // Heuristic: sort by decreasing volume, then by max dimension.
        boxes.Sort((a, b) =>
        {
            int c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            return b.MaxDim.CompareTo(a.MaxDim);
        });

        var placements = new List<Placement>(Math.Min(boxes.Count, 10000));

        long z = 0;
        long layerHeight = 0;

        long y = 0;
        long rowHeight = 0;

        long x = 0;

        for (int i = 0; i < boxes.Count; i++)
        {
            Box b = boxes[i];

            // Try to place in current row -> new row -> new layer.
            if (TryPlaceInShelvesAt(b, W, H, D, x, y, z, W - x, H - y, D - z, out var o))
            {
                placements.Add(new Placement(b.Index, x, y, z));
                x += o.W;
                rowHeight = Math.Max(rowHeight, o.H);
                layerHeight = Math.Max(layerHeight, o.D);
                continue;
            }

            // New row within same layer
            if (rowHeight > 0)
            {
                x = 0;
                y += rowHeight;
                rowHeight = 0;
            }

            if (TryPlaceInShelvesAt(b, W, H, D, x, y, z, W - x, H - y, D - z, out o))
            {
                placements.Add(new Placement(b.Index, x, y, z));
                x += o.W;
                rowHeight = Math.Max(rowHeight, o.H);
                layerHeight = Math.Max(layerHeight, o.D);
                continue;
            }

            // New layer
            if (layerHeight > 0)
            {
                x = 0; y = 0; rowHeight = 0;
                z += layerHeight;
                layerHeight = 0;
            }

            if (TryPlaceInShelvesAt(b, W, H, D, x, y, z, W - x, H - y, D - z, out o))
            {
                placements.Add(new Placement(b.Index, x, y, z));
                x += o.W;
                rowHeight = Math.Max(rowHeight, o.H);
                layerHeight = Math.Max(layerHeight, o.D);
                continue;
            }
        }

        return placements;
    }

    private static bool TryPlaceInShelvesAt(Box b, long W, long H, long D, long x, long y, long z, long remW, long remH, long remD, out Orient chosen)
    {
        chosen = default;
        if (x < 0 || y < 0 || z < 0) return false;
        if (remW <= 0 || remH <= 0 || remD <= 0) return false;

        // Prefer orientations that "fill" width in X, then height in Y, with depth in Z.
        // Score: minimize leftover volume; tie-break by larger footprint (w*h).
        bool found = false;
        long bestScore = long.MaxValue;
        long bestFoot = -1;
        foreach (var o in EnumerateOrientations(b.W, b.H, b.D))
        {
            if (o.W > remW || o.H > remH || o.D > remD) continue;
            long score = (remW - o.W) + (remH - o.H) * 4 + (remD - o.D) * 2;
            long foot = o.W * o.H;
            if (!found || score < bestScore || (score == bestScore && foot > bestFoot))
            {
                found = true;
                bestScore = score;
                bestFoot = foot;
                chosen = o;
            }
        }
        return found;
    }

    // -----------------------------------------
    // Mode 2: 3D Guillotine Free-Space (quality)
    // -----------------------------------------
    private static List<Placement> PackGuillotine3D(List<Box> boxes, long W, long H, long D)
    {
        boxes.Sort((a, b) =>
        {
            int c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            c = b.MaxDim.CompareTo(a.MaxDim);
            if (c != 0) return c;
            return a.Index.CompareTo(b.Index);
        });

        var placements = new List<Placement>(boxes.Count);
        var spaces = new List<Space>(Math.Min(4096, boxes.Count * 2));
        spaces.Add(new Space(0, 0, 0, W, H, D));

        // Bound space growth (dominance pruning + trimming to top-K by volume).
        const int MaxSpacesHard = 25000;
        const int MaxSpacesSoft = 12000;

        for (int i = 0; i < boxes.Count; i++)
        {
            Box b = boxes[i];
            int bestSpaceIdx = -1;
            Orient bestOrient = default;
            long bestScore = long.MaxValue;

            // Evaluate candidate spaces; iterate larger spaces first for quicker fit.
            // For speed, we don't fully sort each time; instead do a lightweight scan.
            for (int si = 0; si < spaces.Count; si++)
            {
                var s = spaces[si];
                if (s.W <= 0 || s.H <= 0 || s.D <= 0) continue;

                // Fast reject: if even max dimension can't fit any axis.
                if (b.MaxDim > Math.Max(s.W, Math.Max(s.H, s.D))) continue;

                foreach (var o in EnumerateOrientations(b.W, b.H, b.D))
                {
                    if (o.W > s.W || o.H > s.H || o.D > s.D) continue;
                    // Best short-side fit in 3D: minimize sum of residuals, weighted to encourage tight packing.
                    long rx = s.W - o.W, ry = s.H - o.H, rz = s.D - o.D;
                    long score = rx + ry + rz * 2;
                    if (score < bestScore)
                    {
                        bestScore = score;
                        bestSpaceIdx = si;
                        bestOrient = o;
                        if (bestScore == 0) break;
                    }
                }
                if (bestScore == 0) break;
            }

            if (bestSpaceIdx < 0) continue;

            var chosenSpace = spaces[bestSpaceIdx];
            placements.Add(new Placement(b.Index, chosenSpace.X, chosenSpace.Y, chosenSpace.Z));

            // Remove used space by swap-remove.
            int last = spaces.Count - 1;
            spaces[bestSpaceIdx] = spaces[last];
            spaces.RemoveAt(last);

            // Split chosen space into residual spaces (guillotine-ish).
            SplitSpace(chosenSpace, bestOrient, spaces);

            // Prune dominated / tiny spaces periodically.
            if ((i & 63) == 63)
            {
                PruneSpaces(spaces, MaxSpacesSoft);
            }
            if (spaces.Count > MaxSpacesHard)
            {
                PruneSpaces(spaces, MaxSpacesSoft);
                if (spaces.Count > MaxSpacesHard)
                {
                    // Hard trim by keeping largest volumes.
                    spaces.Sort((a, b2) => b2.Volume.CompareTo(a.Volume));
                    if (spaces.Count > MaxSpacesSoft)
                        spaces.RemoveRange(MaxSpacesSoft, spaces.Count - MaxSpacesSoft);
                }
            }
        }

        return placements;
    }

    private static void SplitSpace(Space s, Orient o, List<Space> spaces)
    {
        // Place box at (s.X,s.Y,s.Z) with size o.
        long rw = s.W - o.W;
        long rh = s.H - o.H;
        long rd = s.D - o.D;

        // Choose a splitting scheme to reduce fragmentation:
        // - Create up to 3 spaces: right (X), front (Y), above (Z), but ensure they are disjoint.
        // We'll create "right" and "front" in the remaining height/depth of the placed box region,
        // and "above" covering full base area.
        if (rd > 0)
        {
            // Above: same X/Y footprint as original, remaining depth.
            spaces.Add(new Space(s.X, s.Y, s.Z + o.D, s.W, s.H, rd));
        }

        if (rw > 0)
        {
            // Right: remaining width, limited to box height and box depth (to keep disjoint with "front" below).
            spaces.Add(new Space(s.X + o.W, s.Y, s.Z, rw, o.H, o.D));
        }

        if (rh > 0)
        {
            // Front: remaining height, full original width, limited to box depth.
            spaces.Add(new Space(s.X, s.Y + o.H, s.Z, s.W, rh, o.D));
        }

        // Additionally, to reduce waste, add "right-above" and "front-above" are covered by the above space already,
        // so no need. The above space can later host any placement.
    }

    private static void PruneSpaces(List<Space> spaces, int softLimit)
    {
        // Remove invalid and sort larger first.
        int write = 0;
        for (int i = 0; i < spaces.Count; i++)
        {
            var s = spaces[i];
            if (s.W <= 0 || s.H <= 0 || s.D <= 0) continue;
            spaces[write++] = s;
        }
        if (write < spaces.Count) spaces.RemoveRange(write, spaces.Count - write);

        spaces.Sort((a, b) =>
        {
            int c = b.Volume.CompareTo(a.Volume);
            if (c != 0) return c;
            c = a.X.CompareTo(b.X);
            if (c != 0) return c;
            c = a.Y.CompareTo(b.Y);
            if (c != 0) return c;
            return a.Z.CompareTo(b.Z);
        });

        // Dominance pruning: remove spaces fully contained in another at same/lower origin.
        // O(m^2) worst-case, but m is bounded by softLimit and pruning runs periodically.
        int m = spaces.Count;
        var keep = new bool[m];
        Array.Fill(keep, true);

        for (int i = 0; i < m; i++)
        {
            if (!keep[i]) continue;
            var a = spaces[i];
            for (int j = i + 1; j < m; j++)
            {
                if (!keep[j]) continue;
                var b = spaces[j];
                if (IsContained(b, a)) keep[j] = false;
            }
        }

        write = 0;
        for (int i = 0; i < m; i++)
            if (keep[i]) spaces[write++] = spaces[i];
        if (write < spaces.Count) spaces.RemoveRange(write, spaces.Count - write);

        if (spaces.Count > softLimit)
            spaces.RemoveRange(softLimit, spaces.Count - softLimit);
    }

    private static bool IsContained(Space inner, Space outer)
    {
        return inner.X >= outer.X && inner.Y >= outer.Y && inner.Z >= outer.Z &&
               inner.X + inner.W <= outer.X + outer.W &&
               inner.Y + inner.H <= outer.Y + outer.H &&
               inner.Z + inner.D <= outer.Z + outer.D;
    }
}
