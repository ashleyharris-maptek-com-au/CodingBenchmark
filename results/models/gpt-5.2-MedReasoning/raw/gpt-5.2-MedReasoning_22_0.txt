using System;
using System.Collections.Generic;
using System.Diagnostics;
using System.Globalization;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream)
        {
            _stream = stream;
        }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public string Next()
        {
            int c;
            do
            {
                c = ReadByte();
                if (c < 0) return null;
            } while (c <= 32);

            var sb = new System.Text.StringBuilder();
            while (c > 32)
            {
                sb.Append((char)c);
                c = ReadByte();
                if (c < 0) break;
            }
            return sb.ToString();
        }

        public int NextInt()
        {
            var s = Next();
            if (s == null) throw new EndOfStreamException();
            return int.Parse(s, CultureInfo.InvariantCulture);
        }
    }

    private struct PieceRotation
    {
        public int Width;
        public int Height;
        public ulong[] RowMasks; // length Height, bottom-up, already unshifted (x=0)
        public int[] FilledPerRow;

        public PieceRotation(int width, int height, ulong[] rowMasks)
        {
            Width = width;
            Height = height;
            RowMasks = rowMasks;
            FilledPerRow = new int[height];
            for (int i = 0; i < height; i++)
                FilledPerRow[i] = PopCount(rowMasks[i]);
        }
    }

    private sealed class PieceDef
    {
        public readonly PieceRotation[] Rotations;
        public PieceDef(PieceRotation[] rotations)
        {
            Rotations = rotations;
        }
    }

    private sealed class Board
    {
        public readonly int Width;
        public readonly int Height;
        public ulong[] Rows; // bottom-up
        public ulong FullMask;
        public int[] ColHeights; // cached heights (0..Height)

        public Board(int width, int height)
        {
            Width = width;
            Height = height;
            Rows = new ulong[height];
            FullMask = width == 64 ? ulong.MaxValue : ((1UL << width) - 1UL);
            ColHeights = new int[width];
        }

        public void RecomputeColHeights()
        {
            Array.Clear(ColHeights, 0, ColHeights.Length);
            for (int x = 0; x < Width; x++)
            {
                int h = 0;
                for (int y = Height - 1; y >= 0; y--)
                {
                    if (((Rows[y] >> x) & 1UL) != 0)
                    {
                        h = y + 1;
                        break;
                    }
                }
                ColHeights[x] = h;
            }
        }
    }

    private struct Move
    {
        public int Rotation;
        public int X;
        public int Y;
        public int LinesCleared;
        public double Score;

        public Move(int rotation, int x, int y, int linesCleared, double score)
        {
            Rotation = rotation;
            X = x;
            Y = y;
            LinesCleared = linesCleared;
            Score = score;
        }
    }

    private static readonly Dictionary<char, PieceDef> Pieces = BuildPieces();

    public static void Main()
    {
        Console.OutputEncoding = System.Text.Encoding.ASCII;
        var scanner = new FastScanner(Console.OpenStandardInput());

        string wTok = scanner.Next();
        if (wTok == null) return;
        int width = int.Parse(wTok, CultureInfo.InvariantCulture);
        int height = scanner.NextInt();
        int numPieces = scanner.NextInt();

        var board = new Board(width, height);
        board.RecomputeColHeights();

        var swGlobal = Stopwatch.StartNew();
        double emaMs = 0;
        int processed = 0;

        for (int i = 0; i < numPieces; i++)
        {
            string pTok = scanner.Next();
            if (pTok == null) break;
            char pType = pTok[0];

            var start = Stopwatch.StartNew();
            Move best = ChooseMove(board, pType, processed, numPieces, emaMs, swGlobal.Elapsed.TotalSeconds);

            // Output move
            Console.Write(best.Rotation);
            Console.Write(' ');
            Console.WriteLine(best.X);
            Console.Out.Flush();

            // Read result token (interactive feedback). We keep our own simulation for robustness,
            // but also accept "gameover" as termination.
            string res = scanner.Next();
            if (res == null) return;

            // Update our internal board by simulating the placement we just chose.
            bool ok = ApplyMove(board, pType, best.Rotation, best.X, out int cleared);

            if (!ok || res.StartsWith("gameover", StringComparison.OrdinalIgnoreCase))
                return;

            // If judge reports clears, it should match, but we don't rely on it.
            _ = cleared;

            processed++;
            double ms = start.Elapsed.TotalMilliseconds;
            emaMs = emaMs <= 0 ? ms : (0.85 * emaMs + 0.15 * ms);
        }
    }

    private static Move ChooseMove(Board board, char pieceType, int processed, int totalPieces, double emaMs, double elapsedSeconds)
    {
        if (!Pieces.TryGetValue(pieceType, out var piece))
        {
            // Fallback: do nothing-ish.
            return new Move(0, 0, 0, 0, double.NegativeInfinity);
        }

        // Adaptive strategy:
        // - Early game / low stack: prioritize clean surfaces and building wells.
        // - Mid/late or high stack: survival weights (avoid holes & high max).
        // - If we're running slow, simplify (no extra tie-break probing).
        int maxHeight = GetMaxHeight(board);
        int stackDanger = (int)Math.Round(100.0 * maxHeight / Math.Max(1, board.Height));
        bool early = processed < 30 && maxHeight <= board.Height / 4;
        bool danger = maxHeight >= (board.Height * 7) / 10;
        bool timeTight = emaMs > 12.0; // very conservative; we should be far below this

        Weights weights = GetWeights(early, danger, stackDanger);

        double bestScore = double.NegativeInfinity;
        Move best = new Move(0, 0, 0, 0, double.NegativeInfinity);

        // Enumerate all legal placements.
        for (int r = 0; r < piece.Rotations.Length; r++)
        {
            var rot = piece.Rotations[r];
            int maxX = board.Width - rot.Width;
            for (int x = 0; x <= maxX; x++)
            {
                if (!TryDropY(board, rot, x, out int y)) continue;
                if (!TrySimulate(board, rot, x, y, out int linesCleared, out Features f)) continue;

                double score = Evaluate(f, linesCleared, weights);

                // Small tie-break: prefer leftmost in danger (more predictable), otherwise prefer higher clears.
                if (score > bestScore ||
                    (Math.Abs(score - bestScore) < 1e-9 && !timeTight && BetterTieBreak(linesCleared, x, best.LinesCleared, best.X, danger)))
                {
                    bestScore = score;
                    best = new Move(r, x, y, linesCleared, score);
                }
            }
        }

        // If no legal moves found, output something deterministic.
        if (double.IsNegativeInfinity(best.Score))
            return new Move(0, 0, 0, 0, best.Score);

        return best;
    }

    private static bool BetterTieBreak(int clearedA, int xA, int clearedB, int xB, bool danger)
    {
        if (clearedA != clearedB) return clearedA > clearedB;
        if (danger) return xA < xB;
        return xA < xB;
    }

    private struct Weights
    {
        public double Lines;
        public double AggregateHeight;
        public double MaxHeight;
        public double Holes;
        public double Bumpiness;
        public double WellSum;
        public double RowTransitions;
        public double ColTransitions;
        public double CoveredCells;

        public Weights(
            double lines,
            double aggregateHeight,
            double maxHeight,
            double holes,
            double bumpiness,
            double wellSum,
            double rowTransitions,
            double colTransitions,
            double coveredCells)
        {
            Lines = lines;
            AggregateHeight = aggregateHeight;
            MaxHeight = maxHeight;
            Holes = holes;
            Bumpiness = bumpiness;
            WellSum = wellSum;
            RowTransitions = rowTransitions;
            ColTransitions = colTransitions;
            CoveredCells = coveredCells;
        }
    }

    private static Weights GetWeights(bool early, bool danger, int stackDangerPct)
    {
        // Coefficients are heuristic, based on common Tetris AI feature sets (Dellacherie-style),
        // with simple adaptation rather than hard-coding a single personality.
        if (danger)
        {
            return new Weights(
                lines: 2.5,
                aggregateHeight: -0.55,
                maxHeight: -1.30,
                holes: -5.50,
                bumpiness: -0.40,
                wellSum: -0.15,
                rowTransitions: -0.25,
                colTransitions: -0.25,
                coveredCells: -0.65);
        }

        if (early)
        {
            return new Weights(
                lines: 1.6,
                aggregateHeight: -0.35,
                maxHeight: -0.60,
                holes: -4.20,
                bumpiness: -0.25,
                wellSum: 0.10, // allow gentle wells early for Tetris setups
                rowTransitions: -0.15,
                colTransitions: -0.15,
                coveredCells: -0.35);
        }

        // Mid game: scale hole/max penalties slightly with danger percentage.
        double dangerScale = 1.0 + (stackDangerPct / 100.0) * 0.35;
        return new Weights(
            lines: 2.0,
            aggregateHeight: -0.42,
            maxHeight: -0.85 * dangerScale,
            holes: -4.70 * dangerScale,
            bumpiness: -0.32,
            wellSum: -0.05,
            rowTransitions: -0.20,
            colTransitions: -0.20,
            coveredCells: -0.45 * dangerScale);
    }

    private struct Features
    {
        public int AggregateHeight;
        public int MaxHeight;
        public int Holes;
        public int Bumpiness;
        public int WellSum;
        public int RowTransitions;
        public int ColTransitions;
        public int CoveredCells;

        public Features(
            int aggregateHeight,
            int maxHeight,
            int holes,
            int bumpiness,
            int wellSum,
            int rowTransitions,
            int colTransitions,
            int coveredCells)
        {
            AggregateHeight = aggregateHeight;
            MaxHeight = maxHeight;
            Holes = holes;
            Bumpiness = bumpiness;
            WellSum = wellSum;
            RowTransitions = rowTransitions;
            ColTransitions = colTransitions;
            CoveredCells = coveredCells;
        }
    }

    private static double Evaluate(Features f, int linesCleared, Weights w)
    {
        // Reward multi-line clears superlinearly to encourage tetrises when safe.
        double lineReward;
        switch (linesCleared)
        {
            case 4: lineReward = 8.0; break;
            case 3: lineReward = 4.5; break;
            case 2: lineReward = 2.5; break;
            case 1: lineReward = 1.0; break;
            default: lineReward = 0.0; break;
        }

        return
            w.Lines * lineReward +
            w.AggregateHeight * f.AggregateHeight +
            w.MaxHeight * f.MaxHeight +
            w.Holes * f.Holes +
            w.Bumpiness * f.Bumpiness +
            w.WellSum * f.WellSum +
            w.RowTransitions * f.RowTransitions +
            w.ColTransitions * f.ColTransitions +
            w.CoveredCells * f.CoveredCells;
    }

    private static bool ApplyMove(Board board, char pieceType, int rotation, int x, out int cleared)
    {
        cleared = 0;
        if (!Pieces.TryGetValue(pieceType, out var piece)) return false;
        rotation = ((rotation % piece.Rotations.Length) + piece.Rotations.Length) % piece.Rotations.Length;
        var rot = piece.Rotations[rotation];
        if (!TryDropY(board, rot, x, out int y)) return false;

        // Place
        for (int dy = 0; dy < rot.Height; dy++)
        {
            ulong mask = rot.RowMasks[dy] << x;
            int ry = y + dy;
            if (ry < 0 || ry >= board.Height) return false;
            if ((board.Rows[ry] & mask) != 0) return false;
            board.Rows[ry] |= mask;
        }

        // Clear lines
        int write = 0;
        for (int read = 0; read < board.Height; read++)
        {
            if ((board.Rows[read] & board.FullMask) == board.FullMask)
            {
                cleared++;
                continue;
            }
            if (write != read) board.Rows[write] = board.Rows[read];
            write++;
        }
        for (int k = write; k < board.Height; k++) board.Rows[k] = 0;

        board.RecomputeColHeights();
        return true;
    }

    private static bool TrySimulate(Board board, PieceRotation rot, int x, int y, out int linesCleared, out Features features)
    {
        linesCleared = 0;

        // Copy affected board into temp.
        // For simplicity and robustness, copy full array (height is small).
        int h = board.Height;
        ulong[] tmp = new ulong[h];
        Array.Copy(board.Rows, tmp, h);

        // Place
        for (int dy = 0; dy < rot.Height; dy++)
        {
            ulong mask = rot.RowMasks[dy] << x;
            int ry = y + dy;
            if (ry < 0 || ry >= h)
            {
                features = default;
                return false;
            }
            if ((tmp[ry] & mask) != 0)
            {
                features = default;
                return false;
            }
            tmp[ry] |= mask;
        }

        // Clear lines
        int write = 0;
        for (int read = 0; read < h; read++)
        {
            if ((tmp[read] & board.FullMask) == board.FullMask)
            {
                linesCleared++;
                continue;
            }
            if (write != read) tmp[write] = tmp[read];
            write++;
        }
        for (int k = write; k < h; k++) tmp[k] = 0;

        // Compute features
        int[] colHeights = new int[board.Width];
        for (int cx = 0; cx < board.Width; cx++)
        {
            int ch = 0;
            for (int yy = h - 1; yy >= 0; yy--)
            {
                if (((tmp[yy] >> cx) & 1UL) != 0)
                {
                    ch = yy + 1;
                    break;
                }
            }
            colHeights[cx] = ch;
        }

        int aggregateHeight = 0;
        int maxHeight = 0;
        for (int cx = 0; cx < board.Width; cx++)
        {
            aggregateHeight += colHeights[cx];
            if (colHeights[cx] > maxHeight) maxHeight = colHeights[cx];
        }

        int holes = 0;
        int covered = 0;
        for (int cx = 0; cx < board.Width; cx++)
        {
            bool blockSeen = false;
            for (int yy = h - 1; yy >= 0; yy--)
            {
                bool filled = (((tmp[yy] >> cx) & 1UL) != 0);
                if (filled)
                {
                    blockSeen = true;
                }
                else if (blockSeen)
                {
                    holes++;
                }
            }

            // Covered cells: filled cells above lowest hole in the column (penalize burying).
            int holeY = -1;
            blockSeen = false;
            for (int yy = h - 1; yy >= 0; yy--)
            {
                bool filled = (((tmp[yy] >> cx) & 1UL) != 0);
                if (filled) blockSeen = true;
                else if (blockSeen)
                {
                    holeY = yy;
                    break;
                }
            }
            if (holeY >= 0)
            {
                for (int yy = holeY + 1; yy < h; yy++)
                {
                    if (((tmp[yy] >> cx) & 1UL) != 0) covered++;
                }
            }
        }

        int bumpiness = 0;
        for (int cx = 0; cx + 1 < board.Width; cx++)
            bumpiness += Math.Abs(colHeights[cx] - colHeights[cx + 1]);

        int wellSum = 0;
        for (int cx = 0; cx < board.Width; cx++)
        {
            int left = cx == 0 ? int.MaxValue : colHeights[cx - 1];
            int right = cx == board.Width - 1 ? int.MaxValue : colHeights[cx + 1];
            int minNeighbor = Math.Min(left, right);
            if (minNeighbor > colHeights[cx])
            {
                int depth = minNeighbor - colHeights[cx];
                // Triangular sum favors deeper wells more.
                wellSum += depth * (depth + 1) / 2;
            }
        }

        // Row transitions: count changes filled<->empty across each row, including edges as filled.
        int rowTransitions = 0;
        for (int yy = 0; yy < h; yy++)
        {
            ulong row = tmp[yy] & board.FullMask;
            int prev = 1; // left wall treated as filled
            for (int cx = 0; cx < board.Width; cx++)
            {
                int cur = (int)((row >> cx) & 1UL);
                if (cur != prev) rowTransitions++;
                prev = cur;
            }
            if (prev == 0) rowTransitions++; // right wall treated as filled
        }

        // Column transitions: count changes filled<->empty down each column, including floor as filled.
        int colTransitions = 0;
        for (int cx = 0; cx < board.Width; cx++)
        {
            int prev = 1; // floor treated as filled below row 0
            for (int yy = 0; yy < h; yy++)
            {
                int cur = (int)((tmp[yy] >> cx) & 1UL);
                if (cur != prev) colTransitions++;
                prev = cur;
            }
            if (prev == 0) colTransitions++; // above top treated as filled
        }

        features = new Features(
            aggregateHeight: aggregateHeight,
            maxHeight: maxHeight,
            holes: holes,
            bumpiness: bumpiness,
            wellSum: wellSum,
            rowTransitions: rowTransitions,
            colTransitions: colTransitions,
            coveredCells: covered);
        return true;
    }

    private static int GetMaxHeight(Board board)
    {
        int max = 0;
        for (int i = 0; i < board.Width; i++)
            if (board.ColHeights[i] > max) max = board.ColHeights[i];
        return max;
    }

    private static bool TryDropY(Board board, PieceRotation rot, int x, out int y)
    {
        y = board.Height - rot.Height;
        if (x < 0 || x + rot.Width > board.Width) return false;
        if (y < 0) return false;

        // If even at the top it collides, it's gameover.
        if (Collides(board, rot, x, y)) return false;

        while (y > 0 && !Collides(board, rot, x, y - 1))
            y--;
        return true;
    }

    private static bool Collides(Board board, PieceRotation rot, int x, int y)
    {
        for (int dy = 0; dy < rot.Height; dy++)
        {
            ulong mask = rot.RowMasks[dy] << x;
            if ((board.Rows[y + dy] & mask) != 0) return true;
        }
        return false;
    }

    private static Dictionary<char, PieceDef> BuildPieces()
    {
        // Shapes are defined in a small grid with (0,0) at bottom-left of the shape's bounding box.
        // Each rotation is a bottom-up list of row bitmasks.
        var dict = new Dictionary<char, PieceDef>();

        dict['I'] = new PieceDef(new[]
        {
            Rot(4,1, "####"),
            Rot(1,4, "#\n#\n#\n#"),
            Rot(4,1, "####"),
            Rot(1,4, "#\n#\n#\n#"),
        });

        dict['O'] = new PieceDef(new[]
        {
            Rot(2,2, "##\n##"),
            Rot(2,2, "##\n##"),
            Rot(2,2, "##\n##"),
            Rot(2,2, "##\n##"),
        });

        dict['T'] = new PieceDef(new[]
        {
            Rot(3,2, ".#.\n###"),
            Rot(2,3, ".#\n##\n.#"),
            Rot(3,2, "###\n.#."),
            Rot(2,3, "#.\n##\n#."),
        });

        dict['S'] = new PieceDef(new[]
        {
            Rot(3,2, "##.\n.##"),
            Rot(2,3, ".#\n##\n#."),
            Rot(3,2, "##.\n.##"),
            Rot(2,3, ".#\n##\n#."),
        });

        dict['Z'] = new PieceDef(new[]
        {
            Rot(3,2, ".##\n##."),
            Rot(2,3, "#.\n##\n.#"),
            Rot(3,2, ".##\n##."),
            Rot(2,3, "#.\n##\n.#"),
        });

        dict['J'] = new PieceDef(new[]
        {
            Rot(3,2, "###\n#.."),
            Rot(2,3, "##\n.#\n.#"),
            Rot(3,2, "..#\n###"),
            Rot(2,3, ".#\n.#\n##"),
        });

        dict['L'] = new PieceDef(new[]
        {
            Rot(3,2, "###\n..#"),
            Rot(2,3, ".#\n.#\n##"),
            Rot(3,2, "#..\n###"),
            Rot(2,3, "##\n#.\n#."),
        });

        return dict;
    }

    private static PieceRotation Rot(int w, int h, string ascii)
    {
        // ascii rows are BOTTOM-UP separated by '\n' (first row is the bottom of the piece).
        string[] rows = ascii.Split('\n');
        if (rows.Length != h)
            throw new ArgumentException("Bad rotation height");

        ulong[] masks = new ulong[h];
        for (int r = 0; r < h; r++)
        {
            string row = rows[r];
            if (row.Length != w)
                throw new ArgumentException("Bad rotation width");
            ulong m = 0;
            for (int c = 0; c < w; c++)
                if (row[c] == '#') m |= 1UL << c;
            masks[r] = m;
        }
        return new PieceRotation(w, h, masks);
    }

    private static int PopCount(ulong x)
    {
        // Builtin BitOperations.PopCount exists, but keep compatibility minimal.
        x = x - ((x >> 1) & 0x5555555555555555UL);
        x = (x & 0x3333333333333333UL) + ((x >> 2) & 0x3333333333333333UL);
        return (int)((((x + (x >> 4)) & 0x0F0F0F0F0F0F0F0FUL) * 0x0101010101010101UL) >> 56);
    }
}
