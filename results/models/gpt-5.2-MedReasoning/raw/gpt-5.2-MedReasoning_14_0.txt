
using System;
using System.Collections;
using System.Collections.Generic;
using System.IO;
using System.Runtime.CompilerServices;
using System.Text;

public static class Program
{
    // Cell state encoding in `cells`:
    // -2 = unknown (' ')
    // -1 = flagged mine ('*')
    //  0..8 = revealed digit
    private const sbyte Unknown = -2;
    private const sbyte Mine = -1;

    public static void Main()
    {
        var reader = new FastReader(Console.OpenStandardInput());
        int h = reader.ReadInt();
        int w = reader.ReadInt();
        if (h <= 0 || w <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        int n = checked(h * w);
        var cells = new sbyte[n];

        for (int r = 0; r < h; r++)
        {
            reader.ReadBoardRowInto(cells, r, w);
        }

        var safe = new BitArray(n);
        var output = new List<int>(1024); // flattened indices

        var inQueue = new BitArray(n);
        var queue = new IntQueue(1024);

        int revealedNumbers = 0;
        int unknownCells = 0;
        for (int i = 0; i < n; i++)
        {
            if (cells[i] >= 0) revealedNumbers++;
            else if (cells[i] == Unknown) unknownCells++;
        }

        bool useSubset = revealedNumbers <= 2_000_000;
        bool useExact = revealedNumbers <= 150_000 && unknownCells <= 5_000_000;

        for (int idx = 0; idx < n; idx++)
        {
            if (cells[idx] >= 0)
            {
                queue.Enqueue(idx);
                inQueue[idx] = true;
            }
        }

        while (queue.Count > 0)
        {
            int idx = queue.Dequeue();
            inQueue[idx] = false;
            if (cells[idx] < 0) continue;

            ApplyLocalRules(idx, h, w, cells, safe, output, queue, inQueue);
            if (useSubset) ApplySubsetRules(idx, h, w, cells, safe, output, queue, inQueue);
        }

        if (useExact)
        {
            RunBoundedExactEnumeration(h, w, cells, safe, output);
        }

        var sb = new StringBuilder();
        int count = 0;
        for (int i = 0; i < output.Count; i++)
        {
            int id = output[i];
            if ((uint)id >= (uint)n) continue;
            if (!safe[id]) continue;
            if (cells[id] != Unknown) continue;
            count++;
        }
        sb.Append(count).Append('\n');
        for (int i = 0; i < output.Count; i++)
        {
            int id = output[i];
            if ((uint)id >= (uint)n) continue;
            if (!safe[id]) continue;
            if (cells[id] != Unknown) continue;
            int y = id / w;
            int x = id - y * w;
            sb.Append(x).Append(' ').Append(y).Append('\n');
            safe[id] = false; // dedupe prints
        }
        Console.Write(sb.ToString());
    }

    private static readonly int[] Dr = new int[] { -1, -1, -1, 0, 0, 1, 1, 1 };
    private static readonly int[] Dc = new int[] { -1, 0, 1, -1, 1, -1, 0, 1 };

    private static void ApplyLocalRules(
        int idx,
        int h,
        int w,
        sbyte[] cells,
        BitArray safe,
        List<int> output,
        IntQueue queue,
        BitArray inQueue)
    {
        int y = idx / w;
        int x = idx - y * w;
        int digit = cells[idx];
        if (digit < 0) return;

        int flagged = 0;
        int unknownCount = 0;
        Span<int> unknownNeighbors = stackalloc int[8];

        for (int k = 0; k < 8; k++)
        {
            int ny = y + Dr[k];
            int nx = x + Dc[k];
            if ((uint)ny >= (uint)h || (uint)nx >= (uint)w) continue;
            int nid = ny * w + nx;
            sbyte st = cells[nid];
            if (st == Mine) flagged++;
            else if (st == Unknown && !safe[nid]) unknownNeighbors[unknownCount++] = nid;
        }

        int need = digit - flagged;
        if (need < 0 || need > unknownCount) return;

        if (need == 0)
        {
            for (int i = 0; i < unknownCount; i++)
            {
                int sid = unknownNeighbors[i];
                if (cells[sid] == Unknown && !safe[sid])
                {
                    safe[sid] = true;
                    output.Add(sid);
                }
            }
            return;
        }

        if (need == unknownCount)
        {
            for (int i = 0; i < unknownCount; i++)
            {
                int mid = unknownNeighbors[i];
                if (cells[mid] == Unknown)
                {
                    cells[mid] = Mine;
                    EnqueueAdjacentNumbers(mid, h, w, cells, queue, inQueue);
                }
            }
        }
    }

    private static void EnqueueAdjacentNumbers(int idx, int h, int w, sbyte[] cells, IntQueue queue, BitArray inQueue)
    {
        int y = idx / w;
        int x = idx - y * w;
        for (int k = 0; k < 8; k++)
        {
            int ny = y + Dr[k];
            int nx = x + Dc[k];
            if ((uint)ny >= (uint)h || (uint)nx >= (uint)w) continue;
            int nid = ny * w + nx;
            if (cells[nid] >= 0 && !inQueue[nid])
            {
                inQueue[nid] = true;
                queue.Enqueue(nid);
            }
        }
        if (cells[idx] >= 0 && !inQueue[idx])
        {
            inQueue[idx] = true;
            queue.Enqueue(idx);
        }
    }

    private static void ApplySubsetRules(
        int idx,
        int h,
        int w,
        sbyte[] cells,
        BitArray safe,
        List<int> output,
        IntQueue queue,
        BitArray inQueue)
    {
        int y = idx / w;
        int x = idx - y * w;
        int aDigit = cells[idx];
        if (aDigit < 0) return;

        Span<int> aSet = stackalloc int[8];
        int aLen = 0;
        int aFlagged = 0;
        for (int k = 0; k < 8; k++)
        {
            int ny = y + Dr[k];
            int nx = x + Dc[k];
            if ((uint)ny >= (uint)h || (uint)nx >= (uint)w) continue;
            int nid = ny * w + nx;
            sbyte st = cells[nid];
            if (st == Mine) aFlagged++;
            else if (st == Unknown && !safe[nid]) aSet[aLen++] = nid;
        }
        int aNeed = aDigit - aFlagged;
        if (aNeed < 0 || aNeed > aLen) return;
        SortSmall(aSet, aLen);

        int y0 = Math.Max(0, y - 2), y1 = Math.Min(h - 1, y + 2);
        int x0 = Math.Max(0, x - 2), x1 = Math.Min(w - 1, x + 2);
        for (int ny = y0; ny <= y1; ny++)
        {
            int rowBase = ny * w;
            for (int nx = x0; nx <= x1; nx++)
            {
                int j = rowBase + nx;
                if (j == idx) continue;
                int bDigit = cells[j];
                if (bDigit < 0) continue;

                Span<int> bSet = stackalloc int[8];
                int bLen = 0;
                int bFlagged = 0;
                for (int k = 0; k < 8; k++)
                {
                    int ky = ny + Dr[k];
                    int kx = nx + Dc[k];
                    if ((uint)ky >= (uint)h || (uint)kx >= (uint)w) continue;
                    int nid = ky * w + kx;
                    sbyte st = cells[nid];
                    if (st == Mine) bFlagged++;
                    else if (st == Unknown && !safe[nid]) bSet[bLen++] = nid;
                }
                int bNeed = bDigit - bFlagged;
                if (bNeed < 0 || bNeed > bLen) continue;
                SortSmall(bSet, bLen);

                ApplySubsetOneWay(aSet, aLen, aNeed, bSet, bLen, bNeed, h, w, cells, safe, output, queue, inQueue);
                ApplySubsetOneWay(bSet, bLen, bNeed, aSet, aLen, aNeed, h, w, cells, safe, output, queue, inQueue);
            }
        }
    }

    private static void ApplySubsetOneWay(
        Span<int> smallSet,
        int smallLen,
        int smallNeed,
        Span<int> bigSet,
        int bigLen,
        int bigNeed,
        int h,
        int w,
        sbyte[] cells,
        BitArray safe,
        List<int> output,
        IntQueue queue,
        BitArray inQueue)
    {
        if (smallLen == 0 || bigLen == 0) return;
        if (smallLen > bigLen) return;
        if (!IsSubsetSorted(smallSet, smallLen, bigSet, bigLen)) return;

        Span<int> diff = stackalloc int[8];
        int dLen = 0;
        int i = 0;
        for (int j = 0; j < bigLen; j++)
        {
            int v = bigSet[j];
            while (i < smallLen && smallSet[i] < v) i++;
            if (i < smallLen && smallSet[i] == v) continue;
            diff[dLen++] = v;
        }

        if (dLen == 0) return;
        int needDiff = bigNeed - smallNeed;
        if (needDiff < 0 || needDiff > dLen) return;

        if (needDiff == 0)
        {
            for (int k = 0; k < dLen; k++)
            {
                int sid = diff[k];
                if (cells[sid] == Unknown && !safe[sid])
                {
                    safe[sid] = true;
                    output.Add(sid);
                }
            }
        }
        else if (needDiff == dLen)
        {
            for (int k = 0; k < dLen; k++)
            {
                int mid = diff[k];
                if (cells[mid] == Unknown)
                {
                    cells[mid] = Mine;
                    EnqueueAdjacentNumbers(mid, h, w, cells, queue, inQueue);
                }
            }
        }
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static bool IsSubsetSorted(Span<int> a, int aLen, Span<int> b, int bLen)
    {
        int i = 0, j = 0;
        while (i < aLen && j < bLen)
        {
            int av = a[i], bv = b[j];
            if (av == bv) { i++; j++; }
            else if (av > bv) { j++; }
            else return false;
        }
        return i == aLen;
    }

    [MethodImpl(MethodImplOptions.AggressiveInlining)]
    private static void SortSmall(Span<int> a, int len)
    {
        for (int i = 1; i < len; i++)
        {
            int key = a[i];
            int j = i - 1;
            while (j >= 0 && a[j] > key)
            {
                a[j + 1] = a[j];
                j--;
            }
            a[j + 1] = key;
        }
    }

    private static void RunBoundedExactEnumeration(int h, int w, sbyte[] cells, BitArray safe, List<int> output)
    {
        int n = cells.Length;
        var isFrontier = new BitArray(n);
        var frontierIndices = new List<int>(1024);

        for (int idx = 0; idx < n; idx++)
        {
            if (cells[idx] != Unknown) continue;
            int y = idx / w;
            int x = idx - y * w;
            bool touches = false;
            for (int k = 0; k < 8; k++)
            {
                int ny = y + Dr[k];
                int nx = x + Dc[k];
                if ((uint)ny >= (uint)h || (uint)nx >= (uint)w) continue;
                int nid = ny * w + nx;
                if (cells[nid] >= 0) { touches = true; break; }
            }
            if (touches)
            {
                isFrontier[idx] = true;
                frontierIndices.Add(idx);
            }
        }

        if (frontierIndices.Count == 0) return;
        if (frontierIndices.Count > 50_000) return;

        var frontierId = new Dictionary<int, int>(frontierIndices.Count);
        for (int i = 0; i < frontierIndices.Count; i++) frontierId[frontierIndices[i]] = i;

        var frontierToNumbers = new int[frontierIndices.Count][];
        for (int i = 0; i < frontierIndices.Count; i++)
        {
            int idx = frontierIndices[i];
            int y = idx / w;
            int x = idx - y * w;
            Span<int> nums = stackalloc int[8];
            int cnt = 0;
            for (int k = 0; k < 8; k++)
            {
                int ny = y + Dr[k];
                int nx = x + Dc[k];
                if ((uint)ny >= (uint)h || (uint)nx >= (uint)w) continue;
                int nid = ny * w + nx;
                if (cells[nid] >= 0) nums[cnt++] = nid;
            }
            var arr = new int[cnt];
            for (int t = 0; t < cnt; t++) arr[t] = nums[t];
            frontierToNumbers[i] = arr;
        }

        var numberToVars = new Dictionary<int, List<int>>(frontierIndices.Count);
        for (int vi = 0; vi < frontierIndices.Count; vi++)
        {
            foreach (int num in frontierToNumbers[vi])
            {
                if (!numberToVars.TryGetValue(num, out var list))
                {
                    list = new List<int>(8);
                    numberToVars[num] = list;
                }
                list.Add(vi);
            }
        }

        var visited = new BitArray(frontierIndices.Count);
        var q = new IntQueue(1024);

        const int MaxVars = 24;
        const int MaxCons = 48;
        const int MaxNodes = 2_000_000;

        for (int start = 0; start < frontierIndices.Count; start++)
        {
            if (visited[start]) continue;

            var compVars = new List<int>(64);
            var compConsSet = new HashSet<int>();
            visited[start] = true;
            q.Enqueue(start);
            while (q.Count > 0)
            {
                int v = q.Dequeue();
                compVars.Add(v);
                foreach (int num in frontierToNumbers[v])
                {
                    if (compConsSet.Add(num) && numberToVars.TryGetValue(num, out var vars))
                    {
                        for (int t = 0; t < vars.Count; t++)
                        {
                            int u = vars[t];
                            if (!visited[u])
                            {
                                visited[u] = true;
                                q.Enqueue(u);
                            }
                        }
                    }
                }
                if (compVars.Count > MaxVars || compConsSet.Count > MaxCons) break;
            }

            if (compVars.Count == 0) continue;
            if (compVars.Count > MaxVars || compConsSet.Count > MaxCons) continue;

            var compVarMask = new BitArray(frontierIndices.Count);
            for (int i = 0; i < compVars.Count; i++) compVarMask[compVars[i]] = true;

            var constraints = new List<Constraint>(compConsSet.Count);
            bool inconsistent = false;
            foreach (int numIdx in compConsSet)
            {
                int y = numIdx / w;
                int x = numIdx - y * w;
                int flagged = 0;
                Span<int> vars = stackalloc int[8];
                int vcnt = 0;
                for (int k = 0; k < 8; k++)
                {
                    int ny = y + Dr[k];
                    int nx = x + Dc[k];
                    if ((uint)ny >= (uint)h || (uint)nx >= (uint)w) continue;
                    int nid = ny * w + nx;
                    sbyte st = cells[nid];
                    if (st == Mine) { flagged++; continue; }
                    if (st != Unknown) continue;
                    if (safe[nid]) continue;
                    if (!isFrontier[nid]) continue;
                    if (!frontierId.TryGetValue(nid, out int fid)) continue;
                    if (compVarMask[fid]) vars[vcnt++] = fid;
                }
                int need = cells[numIdx] - flagged;
                if (need < 0 || need > vcnt) { inconsistent = true; break; }
                if (vcnt == 0) continue;
                var arr = new int[vcnt];
                for (int t = 0; t < vcnt; t++) arr[t] = vars[t];
                constraints.Add(new Constraint(arr, need));
            }
            if (inconsistent) continue;
            if (constraints.Count == 0) continue;

            var varMap = new Dictionary<int, int>(compVars.Count);
            for (int i = 0; i < compVars.Count; i++) varMap[compVars[i]] = i;
            for (int ci = 0; ci < constraints.Count; ci++)
            {
                var c = constraints[ci];
                for (int t = 0; t < c.Vars.Length; t++) c.Vars[t] = varMap[c.Vars[t]];
                constraints[ci] = c;
            }

            int m = compVars.Count;
            var assign = new sbyte[m];
            for (int i = 0; i < m; i++) assign[i] = -1;
            var mineCount = new int[m];
            long models = 0;
            int nodes = 0;

            var varToCons = new List<int>[m];
            for (int i = 0; i < m; i++) varToCons[i] = new List<int>(8);
            for (int ci = 0; ci < constraints.Count; ci++)
            {
                foreach (int v in constraints[ci].Vars) varToCons[v].Add(ci);
            }

            var curNeed = new int[constraints.Count];
            var curUnk = new int[constraints.Count];
            for (int ci = 0; ci < constraints.Count; ci++)
            {
                curNeed[ci] = constraints[ci].Need;
                curUnk[ci] = constraints[ci].Vars.Length;
            }

            void Dfs(int varIndex)
            {
                if (++nodes > MaxNodes) return;
                if (varIndex == m)
                {
                    for (int ci = 0; ci < constraints.Count; ci++) if (curNeed[ci] != 0) return;
                    models++;
                    for (int i = 0; i < m; i++) if (assign[i] == 1) mineCount[i]++;
                    return;
                }

                int v = varIndex;
                for (sbyte val = 0; val <= 1; val++)
                {
                    assign[v] = val;
                    var affected = varToCons[v];
                    Span<int> changed = stackalloc int[64];
                    int ch = 0;
                    bool ok = true;
                    for (int ii = 0; ii < affected.Count; ii++)
                    {
                        int ci = affected[ii];
                        changed[ch++] = ci;
                        curUnk[ci]--;
                        if (val == 1) curNeed[ci]--;
                        if (curNeed[ci] < 0 || curNeed[ci] > curUnk[ci]) { ok = false; break; }
                    }
                    if (ok) Dfs(varIndex + 1);
                    for (int t = 0; t < ch; t++)
                    {
                        int ci = changed[t];
                        if (val == 1) curNeed[ci]++;
                        curUnk[ci]++;
                    }
                    assign[v] = -1;
                    if (nodes > MaxNodes) return;
                }
            }

            Dfs(0);
            if (models == 0) continue;

            for (int i = 0; i < m; i++)
            {
                if (mineCount[i] == 0)
                {
                    int globalVar = compVars[i];
                    int cellIdx = frontierIndices[globalVar];
                    if (cells[cellIdx] == Unknown && !safe[cellIdx])
                    {
                        safe[cellIdx] = true;
                        output.Add(cellIdx);
                    }
                }
            }
        }
    }

    private readonly struct Constraint
    {
        public readonly int[] Vars;
        public readonly int Need;
        public Constraint(int[] vars, int need) { Vars = vars; Need = need; }
    }

    private sealed class FastReader
    {
        private readonly Stream _stream;
        private readonly byte[] _buf;
        private int _len;
        private int _pos;

        public FastReader(Stream stream, int bufSize = 1 << 16)
        {
            _stream = stream;
            _buf = new byte[bufSize];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _stream.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int ReadInt()
        {
            int c;
            do { c = ReadByte(); if (c < 0) return 0; } while (c <= ' ');
            int sign = 1;
            if (c == '-') { sign = -1; c = ReadByte(); }
            int v = 0;
            while (c > ' ')
            {
                v = v * 10 + (c - '0');
                c = ReadByte();
            }
            return v * sign;
        }

        public void ReadBoardRowInto(sbyte[] cells, int row, int w)
        {
            int baseIdx = row * w;
            int col = 0;

            int c = ReadByte();
            while (c == '\r' || c == '\n') c = ReadByte();

            while (c >= 0 && c != '\n' && c != '\r')
            {
                if (col < w)
                {
                    char ch = (char)c;
                    cells[baseIdx + col] = ch switch
                    {
                        '*' => Mine,
                        ' ' => Unknown,
                        >= '0' and <= '8' => (sbyte)(ch - '0'),
                        _ => Unknown
                    };
                }
                col++;
                c = ReadByte();
            }

            if (c == '\r')
            {
                int nxt = ReadByte();
                if (nxt != '\n' && nxt >= 0) _pos--;
            }

            for (; col < w; col++) cells[baseIdx + col] = Unknown;
        }
    }

    private sealed class IntQueue
    {
        private int[] _a;
        private int _head;
        private int _tail;
        public int Count { get; private set; }

        public IntQueue(int capacity)
        {
            _a = new int[Math.Max(4, capacity)];
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public void Enqueue(int v)
        {
            if (Count == _a.Length) Grow();
            _a[_tail++] = v;
            if (_tail == _a.Length) _tail = 0;
            Count++;
        }

        [MethodImpl(MethodImplOptions.AggressiveInlining)]
        public int Dequeue()
        {
            int v = _a[_head++];
            if (_head == _a.Length) _head = 0;
            Count--;
            return v;
        }

        private void Grow()
        {
            int[] b = new int[_a.Length * 2];
            int j = 0;
            while (Count > 0) b[j++] = Dequeue();
            _a = b;
            _head = 0;
            _tail = j;
            Count = j;
        }
    }
}
