// C++17 exact solver for:
//   maximize sum(c_i * x_i)
//   subject to x_u + x_v <= 1 for each (u,v), x_i in {0,1}
// This is a Maximum Weight Independent Set problem on the conflict graph.
// We solve it as a Maximum Weight Clique on the complement graph using
// branch-and-bound with a greedy coloring upper bound (weight-aware).

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <numeric>
#include <utility>
#include <vector>

#if defined(_MSC_VER)
#include <intrin.h>
#endif

using namespace std;

static inline int ctz64(uint64_t x) {
#if defined(_MSC_VER)
  unsigned long idx;
  _BitScanForward64(&idx, x);
  return (int)idx;
#else
  return __builtin_ctzll(x);
#endif
}

struct Bitset {
  int n = 0;
  int words = 0;
  vector<uint64_t> w;

  Bitset() = default;
  explicit Bitset(int n_) { init(n_); }

  void init(int n_) {
    n = n_;
    words = (n + 63) >> 6;
    w.assign(words, 0);
  }

  inline void set(int i) { w[i >> 6] |= (1ULL << (i & 63)); }
  inline void reset(int i) { w[i >> 6] &= ~(1ULL << (i & 63)); }
  inline bool test(int i) const { return (w[i >> 6] >> (i & 63)) & 1ULL; }

  inline bool any() const {
    for (uint64_t x : w) if (x) return true;
    return false;
  }

  inline void clear() { for (auto &x : w) x = 0; }

  inline void and_with(const Bitset &o) {
    for (int i = 0; i < words; i++) w[i] &= o.w[i];
  }

  inline void andnot_with(const Bitset &o) {
    for (int i = 0; i < words; i++) w[i] &= ~o.w[i];
  }

  inline void or_with(const Bitset &o) {
    for (int i = 0; i < words; i++) w[i] |= o.w[i];
  }

  inline int pop_lsb() {
    for (int i = 0; i < words; i++) {
      uint64_t x = w[i];
      if (!x) continue;
      int b = ctz64(x);
      w[i] = x & (x - 1);
      return (i << 6) + b;
    }
    return -1;
  }

  inline int peek_lsb() const {
    for (int i = 0; i < words; i++) {
      uint64_t x = w[i];
      if (!x) continue;
      int b = ctz64(x);
      return (i << 6) + b;
    }
    return -1;
  }

  inline void mask_to_n() {
    int extra = (words << 6) - n;
    if (extra > 0) {
      uint64_t mask = (extra == 64) ? 0ULL : (~0ULL >> extra);
      w[words - 1] &= mask;
    }
  }
};

struct Solver {
  int n = 0;
  vector<long long> weight;
  vector<Bitset> adjG;   // conflict graph adjacency (includes self bit = 1)
  vector<Bitset> adjC;   // complement graph adjacency (no self bit)

  long long bestW = 0;
  vector<int> bestClique;

  // Greedy coloring of candidates in complement graph:
  // produces an ordering "order" and an upper-bound array "wbound" such that
  // when processing vertices from the end, currentW + wbound.back() is an upper
  // bound on any clique extension.
  void colorSortWeight(const Bitset &cand, vector<int> &order, vector<long long> &wbound) {
    Bitset U = cand;
    U.mask_to_n();

    order.clear();
    wbound.clear();

    long long cumulative = 0;
    vector<int> classVerts;

    while (U.any()) {
      Bitset Q = U;
      classVerts.clear();

      long long classMax = 0;

      while (Q.any()) {
        int v = Q.pop_lsb();
        if (v < 0 || v >= n) break;
        // Remove v from U (so it won't be colored again)
        U.reset(v);

        classVerts.push_back(v);
        classMax = max(classMax, weight[v]);

        // Keep only vertices non-adjacent to v (to build an independent color class)
        Bitset notNbr = Q;
        notNbr.andnot_with(adjC[v]); // remove neighbors of v
        Q = std::move(notNbr);
      }

      cumulative += max(0LL, classMax);
      for (int v : classVerts) {
        order.push_back(v);
        wbound.push_back(cumulative);
      }
    }

    // Small heuristic: within same bound, prefer heavier vertices later (picked earlier by DFS from back)
    // Stable sort by (wbound asc, weight asc) to keep structure while helping search.
    vector<int> idx(order.size());
    iota(idx.begin(), idx.end(), 0);
    stable_sort(idx.begin(), idx.end(), [&](int a, int b) {
      if (wbound[a] != wbound[b]) return wbound[a] < wbound[b];
      return weight[order[a]] < weight[order[b]];
    });
    vector<int> newOrder;
    vector<long long> newBound;
    newOrder.reserve(order.size());
    newBound.reserve(order.size());
    for (int i : idx) {
      newOrder.push_back(order[i]);
      newBound.push_back(wbound[i]);
    }
    order.swap(newOrder);
    wbound.swap(newBound);
  }

  void expand(Bitset cand, vector<int> &clique, long long curW) {
    if (!cand.any()) {
      if (curW > bestW) {
        bestW = curW;
        bestClique = clique;
      }
      return;
    }

    vector<int> order;
    vector<long long> wbound;
    colorSortWeight(cand, order, wbound);

    for (int i = (int)order.size() - 1; i >= 0; i--) {
      if (curW + wbound[i] <= bestW) return; // prune: even optimistic bound can't beat best

      int v = order[i];
      if (!cand.test(v)) continue; // may have been removed by earlier branches

      // Branch: include v
      clique.push_back(v);
      long long newW = curW + weight[v];

      Bitset newCand = cand;
      newCand.and_with(adjC[v]);   // must be adjacent to v in complement to keep clique
      newCand.mask_to_n();

      if (newW > bestW && !newCand.any()) {
        bestW = newW;
        bestClique = clique;
      } else {
        expand(std::move(newCand), clique, newW);
      }

      clique.pop_back();

      // Exclude v from candidates
      cand.reset(v);
    }
  }

  pair<long long, vector<int>> solve() {
    // Precompute complement adjacency from adjG
    adjC.assign(n, Bitset(n));
    Bitset all(n);
    for (int i = 0; i < n; i++) all.set(i);
    for (int i = 0; i < n; i++) {
      adjC[i] = all;
      adjC[i].andnot_with(adjG[i]); // keep non-neighbors in conflict graph (including self removed via adjG self bit)
      adjC[i].reset(i);
      adjC[i].mask_to_n();
    }

    // Candidate vertices: only positive weight can improve over 0
    Bitset cand(n);
    for (int i = 0; i < n; i++) if (weight[i] > 0) cand.set(i);
    cand.mask_to_n();

    bestW = 0;
    bestClique.clear();

    vector<int> clique;
    expand(cand, clique, 0);

    // bestClique is a max-weight clique in complement => max-weight independent set in original
    return {bestW, bestClique};
  }
};

int main() {
  ios::sync_with_stdio(false);
  cin.tie(nullptr);

  int n, m;
  if (!(cin >> n >> m)) return 0;
  vector<long long> c(n);
  for (int i = 0; i < n; i++) cin >> c[i];

  Solver solver;
  solver.n = n;
  solver.weight = c;
  solver.adjG.assign(n, Bitset(n));

  // conflict graph: self-bit on, plus undirected edges from constraints
  for (int i = 0; i < n; i++) solver.adjG[i].set(i);
  for (int i = 0; i < m; i++) {
    int u, v;
    cin >> u >> v;
    if (u < 0 || u >= n || v < 0 || v >= n) continue;
    if (u == v) {
      // x_u + x_u <= 1 implies x_u must be 0; treat as self-conflict with itself (already set)
      continue;
    }
    solver.adjG[u].set(v);
    solver.adjG[v].set(u);
  }
  for (int i = 0; i < n; i++) solver.adjG[i].mask_to_n();

  auto [bestW, bestSet] = solver.solve();

  vector<int> x(n, 0);
  for (int v : bestSet) if (0 <= v && v < n) x[v] = 1;

  // Verify objective (for safety and to match output check #3)
  long long obj = 0;
  for (int i = 0; i < n; i++) if (x[i]) obj += c[i];

  cout << obj << "\n";
  for (int i = 0; i < n; i++) {
    if (i) cout << ' ';
    cout << x[i];
  }
  cout << "\n";
  return 0;
}
