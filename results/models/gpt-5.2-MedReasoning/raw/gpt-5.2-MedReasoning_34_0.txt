#!/usr/bin/env python3
"""
CVRP heuristic solver.

Input:
N K CAPACITY
depot_x depot_y
x1 y1 demand1
...

Output:
total_distance
num_routes
route_1 (space-separated 0-based customer indices)
...
"""

from __future__ import annotations

import math
import random
import sys
from typing import List, Tuple


def dist(a: Tuple[float, float], b: Tuple[float, float]) -> float:
    dx = a[0] - b[0]
    dy = a[1] - b[1]
    return math.hypot(dx, dy)


def route_distance(route: List[int], depot: Tuple[float, float], pts: List[Tuple[float, float]]) -> float:
    if not route:
        return 0.0
    d = dist(depot, pts[route[0]])
    for i in range(len(route) - 1):
        d += dist(pts[route[i]], pts[route[i + 1]])
    d += dist(pts[route[-1]], depot)
    return d


def compute_savings(
    depot: Tuple[float, float],
    pts: List[Tuple[float, float]],
) -> List[Tuple[float, int, int]]:
    n = len(pts)
    d0 = [dist(depot, pts[i]) for i in range(n)]
    savings = []
    for i in range(n):
        for j in range(i + 1, n):
            sij = d0[i] + d0[j] - dist(pts[i], pts[j])
            savings.append((sij, i, j))
    savings.sort(reverse=True)
    return savings


def clarke_wright(
    n: int,
    k: int,
    cap: float,
    depot: Tuple[float, float],
    pts: List[Tuple[float, float]],
    dem: List[float],
) -> List[List[int]]:
    # Start with one route per customer.
    routes: List[List[int]] = [[i] for i in range(n)]
    rload: List[float] = [dem[i] for i in range(n)]
    # Track which route each customer is currently in.
    where = list(range(n))

    savings = compute_savings(depot, pts)
    for sij, i, j in savings:
        ri = where[i]
        rj = where[j]
        if ri == rj:
            continue
        A = routes[ri]
        B = routes[rj]
        if not A or not B:
            continue
        # Only merge if i is at an end of A and j is at an end of B.
        i_is_start = (A[0] == i)
        i_is_end = (A[-1] == i)
        j_is_start = (B[0] == j)
        j_is_end = (B[-1] == j)
        if not ((i_is_start or i_is_end) and (j_is_start or j_is_end)):
            continue
        if rload[ri] + rload[rj] > cap + 1e-9:
            continue

        # Create merged route with best orientation.
        candidates = []
        if i_is_end and j_is_start:
            candidates.append(A + B)
        if i_is_end and j_is_end:
            candidates.append(A + list(reversed(B)))
        if i_is_start and j_is_start:
            candidates.append(list(reversed(A)) + B)
        if i_is_start and j_is_end:
            candidates.append(list(reversed(A)) + list(reversed(B)))
        if not candidates:
            continue

        # Choose merged route that minimizes route distance.
        best = min(candidates, key=lambda r: route_distance(r, depot, pts))

        # Merge rj into ri (keep ri index for stability).
        routes[ri] = best
        rload[ri] = rload[ri] + rload[rj]
        routes[rj] = []
        rload[rj] = 0.0
        for c in best:
            where[c] = ri

    # Collect non-empty routes.
    out = [r for r in routes if r]
    return out


def two_opt(route: List[int], depot: Tuple[float, float], pts: List[Tuple[float, float]]) -> List[int]:
    if len(route) < 4:
        return route
    best = route
    best_d = route_distance(best, depot, pts)
    improved = True
    while improved:
        improved = False
        m = len(best)
        for i in range(m - 2):
            a = depot if i == 0 else pts[best[i - 1]]
            b = pts[best[i]]
            for j in range(i + 1, m - 1):
                c = pts[best[j]]
                d = depot if j == m - 1 else pts[best[j + 1]]
                # delta of reversing segment [i..j]
                before = dist(a, b) + dist(c, d)
                after = dist(a, c) + dist(b, d)
                if after + 1e-12 < before:
                    cand = best[:i] + list(reversed(best[i : j + 1])) + best[j + 1 :]
                    cand_d = best_d - before + after
                    best = cand
                    best_d = cand_d
                    improved = True
                    break
            if improved:
                break
    return best


def best_insertion_assignment(
    n: int,
    k: int,
    cap: float,
    depot: Tuple[float, float],
    pts: List[Tuple[float, float]],
    dem: List[float],
) -> List[List[int]]:
    # Fixed number of routes <= K, using cheapest feasible insertion.
    routes: List[List[int]] = []
    loads: List[float] = []

    # Start order: sweep by angle, but place larger demands first within similar angles
    angles = [math.atan2(pts[i][1] - depot[1], pts[i][0] - depot[0]) for i in range(n)]
    order = sorted(range(n), key=lambda i: (angles[i], -dem[i]))

    for cust in order:
        placed = False
        best_choice = None  # (delta, route_idx, insert_pos)
        for r_idx, r in enumerate(routes):
            if loads[r_idx] + dem[cust] > cap + 1e-9:
                continue
            # try all insertion positions
            if not r:
                delta = 2.0 * dist(depot, pts[cust])
                best_choice = min(best_choice, (delta, r_idx, 0)) if best_choice else (delta, r_idx, 0)
                continue
            for pos in range(len(r) + 1):
                prev_pt = depot if pos == 0 else pts[r[pos - 1]]
                next_pt = depot if pos == len(r) else pts[r[pos]]
                delta = dist(prev_pt, pts[cust]) + dist(pts[cust], next_pt) - dist(prev_pt, next_pt)
                if best_choice is None or delta < best_choice[0]:
                    best_choice = (delta, r_idx, pos)

        if best_choice is not None:
            _, r_idx, pos = best_choice
            routes[r_idx].insert(pos, cust)
            loads[r_idx] += dem[cust]
            placed = True
        else:
            if len(routes) >= k:
                # No feasible placement without exceeding K routes => instance likely infeasible for given K/CAP.
                raise ValueError("Unable to assign all customers within K routes and capacity constraints.")
            routes.append([cust])
            loads.append(dem[cust])
            placed = True

        if not placed:
            raise ValueError("Unassigned customer.")

    return routes


def solve(n: int, k: int, cap: float, depot: Tuple[float, float], pts: List[Tuple[float, float]], dem: List[float]) -> List[List[int]]:
    # Basic feasibility check.
    if any(d > cap + 1e-9 for d in dem):
        raise ValueError("A customer demand exceeds vehicle capacity.")
    if sum(dem) > k * cap + 1e-9:
        raise ValueError("Total demand exceeds K * capacity (infeasible).")

    # Try Clarke-Wright first for route-quality; if it uses too many routes, fall back to fixed-K insertion.
    routes = clarke_wright(n, k, cap, depot, pts, dem)
    if len(routes) > k:
        routes = best_insertion_assignment(n, k, cap, depot, pts, dem)

    # Improve each route with 2-opt.
    routes = [two_opt(r, depot, pts) for r in routes]

    # If fewer than K routes is fine. Ensure no empties.
    routes = [r for r in routes if r]

    # Final validation: cover all customers once.
    seen = [0] * n
    loads = []
    for r in routes:
        load = 0.0
        for c in r:
            if c < 0 or c >= n:
                raise ValueError("Invalid customer index.")
            seen[c] += 1
            load += dem[c]
        if load > cap + 1e-6:
            raise ValueError("Capacity violated.")
        loads.append(load)
    if any(s != 1 for s in seen):
        raise ValueError("Not a partition of customers.")
    if len(routes) > k:
        raise ValueError("Too many routes.")

    return routes


def main() -> None:
    data = sys.stdin.read().strip().split()
    if not data:
        return
    it = iter(data)
    n = int(next(it))
    k = int(next(it))
    cap = float(next(it))
    depot = (float(next(it)), float(next(it)))
    pts: List[Tuple[float, float]] = []
    dem: List[float] = []
    for _ in range(n):
        x = float(next(it))
        y = float(next(it))
        d = float(next(it))
        pts.append((x, y))
        dem.append(d)

    # Small randomization to avoid worst-case tie patterns (deterministic seed).
    random.seed(0)
    # Slightly shuffle equal-angle order by adding tiny noise to angles inside assignment when used.

    routes = solve(n, k, cap, depot, pts, dem)

    total = sum(route_distance(r, depot, pts) for r in routes)
    out_lines = []
    out_lines.append(f"{total:.10f}")
    out_lines.append(str(len(routes)))
    for r in routes:
        out_lines.append(" ".join(map(str, r)))
    sys.stdout.write("\n".join(out_lines))


if __name__ == "__main__":
    main()
