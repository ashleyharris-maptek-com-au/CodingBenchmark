// N-Dimensional Snake Solver
// Compile with: cl /std:c++20 /O2 solver.cpp
//
// Strategy (works across dimensions without full grid enumeration):
// - Maintain snake body as a deque of positions; start body as repeated head position.
// - Greedy/A*-lite planning:
//   1) Try to pick a "current target" food item (closest by Manhattan distance).
//   2) Choose the next move by scoring all legal neighbor moves:
//        - Primary: reduce distance to target food (Manhattan).
//        - Secondary: maximize local safety (available legal moves from next cell),
//                    avoid "tight" spots and immediate dead-ends.
//        - Tertiary: slight bias to stay away from obstacles/body (neighbor clearance).
//   3) If no move improves food distance, still pick safest move.
// - Legality accounts for obstacles and snake self-collision, allowing moving into the
//   tail cell because it vacates when the snake doesn't grow on that step.
//
// This is a robust baseline designed for very high dimensions where exhaustive search
// is impossible; it favors survival and incremental progress toward food.

#include <algorithm>
#include <array>
#include <cstdint>
#include <deque>
#include <iostream>
#include <limits>
#include <numeric>
#include <optional>
#include <string>
#include <unordered_map>
#include <unordered_set>
#include <utility>
#include <vector>

using std::int64_t;

struct VecHash {
  size_t operator()(const std::vector<int>& v) const noexcept {
    // 64-bit FNV-1a-ish hash.
    uint64_t h = 1469598103934665603ull;
    for (int x : v) {
      uint64_t y = static_cast<uint64_t>(static_cast<uint32_t>(x));
      h ^= y + 0x9e3779b97f4a7c15ull + (h << 6) + (h >> 2);
      h *= 1099511628211ull;
    }
    return static_cast<size_t>(h);
  }
};

static inline int64_t manhattan(const std::vector<int>& a, const std::vector<int>& b) {
  int64_t s = 0;
  for (size_t i = 0; i < a.size(); ++i) s += std::llabs((int64_t)a[i] - (int64_t)b[i]);
  return s;
}

static inline bool in_bounds(const std::vector<int>& p, const std::vector<int>& bounds) {
  for (size_t i = 0; i < p.size(); ++i) {
    if (p[i] < 0 || p[i] >= bounds[i]) return false;
  }
  return true;
}

struct Game {
  int D = 0;
  int max_turns = 0;
  std::vector<int> bounds;

  int snake_len = 1;
  std::deque<std::vector<int>> body;  // front=head
  std::unordered_map<std::vector<int>, int, VecHash> body_count;

  std::vector<std::vector<int>> foods;
  std::unordered_set<std::vector<int>, VecHash> food_set;

  std::unordered_set<std::vector<int>, VecHash> obstacles;

  int64_t turn = 0;
};

static inline void inc_count(std::unordered_map<std::vector<int>, int, VecHash>& m,
                             const std::vector<int>& p) {
  auto it = m.find(p);
  if (it == m.end()) m.emplace(p, 1);
  else ++it->second;
}

static inline void dec_count(std::unordered_map<std::vector<int>, int, VecHash>& m,
                             const std::vector<int>& p) {
  auto it = m.find(p);
  if (it == m.end()) return;
  if (--it->second <= 0) m.erase(it);
}

static bool is_occupied(const Game& g, const std::vector<int>& p) {
  return g.body_count.find(p) != g.body_count.end();
}

static int legal_degree_from(const Game& g, const std::vector<int>& pos, bool will_grow,
                             const std::vector<int>& tail_pos) {
  // Count legal immediate moves from pos if we were to move there.
  // We approximate future body by considering occupancy as current body, and allowing
  // stepping onto the tail if it will vacate (i.e., !will_grow).
  int deg = 0;
  for (int axis = 0; axis < g.D; ++axis) {
    for (int dir : {-1, +1}) {
      std::vector<int> nxt = pos;
      nxt[axis] += dir;
      if (!in_bounds(nxt, g.bounds)) continue;
      if (g.obstacles.find(nxt) != g.obstacles.end()) continue;

      bool occ = is_occupied(g, nxt);
      if (occ) {
        if (!will_grow && nxt == tail_pos) {
          // Tail vacates, so treat as free.
        } else {
          continue;
        }
      }
      ++deg;
    }
  }
  return deg;
}

static int neighbor_clearance(const Game& g, const std::vector<int>& pos,
                              const std::vector<int>& tail_pos, bool will_grow) {
  // Count how many of the 2D neighbors are "open" (not obstacle, and not body except tail-vacate).
  int open = 0;
  for (int axis = 0; axis < g.D; ++axis) {
    for (int dir : {-1, +1}) {
      std::vector<int> q = pos;
      q[axis] += dir;
      if (!in_bounds(q, g.bounds)) continue;
      if (g.obstacles.find(q) != g.obstacles.end()) continue;
      bool occ = is_occupied(g, q);
      if (occ) {
        if (!will_grow && q == tail_pos) {
          // ok
        } else {
          continue;
        }
      }
      ++open;
    }
  }
  return open;
}

static std::optional<int> pick_target_food_index(const Game& g) {
  if (g.foods.empty()) return std::nullopt;
  const auto& head = g.body.front();
  int best_i = -1;
  int64_t best_d = std::numeric_limits<int64_t>::max();
  for (int i = 0; i < (int)g.foods.size(); ++i) {
    int64_t d = manhattan(head, g.foods[i]);
    if (d < best_d) {
      best_d = d;
      best_i = i;
    }
  }
  if (best_i < 0) return std::nullopt;
  return best_i;
}

static bool would_grow(const Game& g, const std::vector<int>& next_head) {
  return g.food_set.find(next_head) != g.food_set.end();
}

static bool is_legal_move(const Game& g, int axis, int dir) {
  const auto& head = g.body.front();
  std::vector<int> nxt = head;
  nxt[axis] += dir;
  if (!in_bounds(nxt, g.bounds)) return false;
  if (g.obstacles.find(nxt) != g.obstacles.end()) return false;

  bool grow = would_grow(g, nxt);
  const auto& tail = g.body.back();
  bool occ = is_occupied(g, nxt);
  if (occ) {
    if (!grow && nxt == tail) return true;
    return false;
  }
  return true;
}

struct MoveChoice {
  int axis = 0;
  int dir = +1;
  int64_t score = std::numeric_limits<int64_t>::min();
};

static MoveChoice choose_move(const Game& g, const std::optional<int>& target_food_index) {
  const auto& head = g.body.front();
  const auto& tail = g.body.back();
  std::vector<int> target = head;
  bool has_target = false;
  if (target_food_index && *target_food_index >= 0 && *target_food_index < (int)g.foods.size()) {
    target = g.foods[*target_food_index];
    has_target = true;
  }

  const int64_t cur_food_dist = has_target ? manhattan(head, target) : 0;

  MoveChoice best;

  for (int axis = 0; axis < g.D; ++axis) {
    for (int dir : {-1, +1}) {
      if (!is_legal_move(g, axis, dir)) continue;

      std::vector<int> nxt = head;
      nxt[axis] += dir;

      bool grow = would_grow(g, nxt);
      int64_t next_food_dist = has_target ? manhattan(nxt, target) : 0;
      int food_progress = has_target ? (int)(cur_food_dist - next_food_dist) : 0;  // >0 is good

      int deg = legal_degree_from(g, nxt, grow, tail);
      int clearance = neighbor_clearance(g, nxt, tail, grow);

      // Score design:
      // - Food progress dominates when possible.
      // - Avoid dead-ends: prefer higher degree.
      // - Prefer more clearance (space around).
      // - Tiny bias against moving away from target.
      //
      // Use wide integers to avoid overflow.
      int64_t score = 0;
      score += (int64_t)food_progress * 1'000'000;
      score += (int64_t)deg * 10'000;
      score += (int64_t)clearance * 1'000;
      if (has_target) score += (cur_food_dist - next_food_dist);  // small tie-breaker

      // Prefer not to oscillate in tiny spaces:
      // slight penalty for moves that don't change distance at all when target exists.
      if (has_target && next_food_dist == cur_food_dist) score -= 50;

      // Deterministic tie-breaker: lowest axis then +1 over -1.
      if (score > best.score ||
          (score == best.score && (axis < best.axis || (axis == best.axis && dir > best.dir)))) {
        best.axis = axis;
        best.dir = dir;
        best.score = score;
      }
    }
  }

  // If no legal moves found (should mean death), output something deterministic.
  return best;
}

static void apply_move(Game& g, int axis, int dir) {
  const auto head = g.body.front();
  std::vector<int> nxt = head;
  nxt[axis] += dir;

  bool grow = would_grow(g, nxt);

  // Advance head.
  g.body.push_front(nxt);
  inc_count(g.body_count, nxt);

  // Eat food if present.
  if (grow) {
    g.food_set.erase(nxt);
    // Remove from foods vector lazily (keep vector but let target picking ignore removed by set check).
    // We'll compact occasionally to keep it small.
  } else {
    // Pop tail.
    auto tail = g.body.back();
    g.body.pop_back();
    dec_count(g.body_count, tail);
  }

  // Keep foods vector reasonably compact.
  if (!g.foods.empty() && (g.turn % 64 == 0)) {
    std::vector<std::vector<int>> kept;
    kept.reserve(g.foods.size());
    for (auto& f : g.foods) {
      if (g.food_set.find(f) != g.food_set.end()) kept.push_back(f);
    }
    g.foods.swap(kept);
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  Game g;
  if (!(std::cin >> g.D >> g.max_turns)) return 0;
  g.bounds.resize(g.D);
  for (int i = 0; i < g.D; ++i) std::cin >> g.bounds[i];

  g.snake_len = 1;
  std::vector<int> head(g.D, 0);
  std::cin >> g.snake_len;
  for (int i = 0; i < g.D; ++i) std::cin >> head[i];

  int F = 0;
  std::cin >> F;
  g.foods.reserve(std::max(0, F));
  for (int i = 0; i < F; ++i) {
    std::vector<int> f(g.D, 0);
    for (int j = 0; j < g.D; ++j) std::cin >> f[j];
    g.foods.push_back(f);
    g.food_set.insert(f);
  }

  int O = 0;
  std::cin >> O;
  for (int i = 0; i < O; ++i) {
    std::vector<int> o(g.D, 0);
    for (int j = 0; j < g.D; ++j) std::cin >> o[j];
    g.obstacles.insert(o);
  }

  // Initialize body as repeated head positions (length g.snake_len).
  // This makes self-occupancy well-defined even though the input doesn't provide full body coordinates.
  g.body.clear();
  g.body_count.clear();
  for (int k = 0; k < std::max(1, g.snake_len); ++k) {
    g.body.push_back(head);
    inc_count(g.body_count, head);
  }

  // If head is invalid, still output deterministic moves.
  for (g.turn = 0; g.turn < g.max_turns; ++g.turn) {
    if (g.food_set.empty()) break;

    // Pick a target food that still exists.
    std::optional<int> target_i = pick_target_food_index(g);
    if (target_i) {
      // If chosen target already eaten (foods vector stale), re-pick by scanning set.
      if (g.food_set.find(g.foods[*target_i]) == g.food_set.end()) {
        // Refresh foods from set (bounded by remaining foods).
        g.foods.assign(g.food_set.begin(), g.food_set.end());
        target_i = pick_target_food_index(g);
      }
    }

    MoveChoice mv = choose_move(g, target_i);

    // If no legal move found, output a default and stop.
    if (mv.score == std::numeric_limits<int64_t>::min()) {
      std::cout << 0 << " " << 1 << "\n";
      std::cout.flush();
      break;
    }

    std::cout << mv.axis << " " << mv.dir << "\n";
    std::cout.flush();

    apply_move(g, mv.axis, mv.dir);
  }

  return 0;
}
