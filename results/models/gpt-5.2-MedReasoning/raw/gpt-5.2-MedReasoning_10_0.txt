#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <string>
#include <tuple>
#include <utility>
#include <vector>

struct RectInput {
    int id{};
    int w{};
    int h{};
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

static inline bool fits(int W, int H, int w, int h) {
    return w <= W && h <= H;
}

// ------------------------ MaxRects (small/medium N) ------------------------
struct FreeRect {
    int x{};
    int y{};
    int w{};
    int h{};
};

static inline bool rectContains(const FreeRect& a, const FreeRect& b) {
    return b.x >= a.x && b.y >= a.y && (b.x + b.w) <= (a.x + a.w) && (b.y + b.h) <= (a.y + a.h);
}

static void pruneFreeList(std::vector<FreeRect>& freeRects) {
    // Remove contained rectangles (O(m^2), m is typically small).
    for (size_t i = 0; i < freeRects.size(); ++i) {
        for (size_t j = i + 1; j < freeRects.size();) {
            if (rectContains(freeRects[i], freeRects[j])) {
                freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(j));
            } else if (rectContains(freeRects[j], freeRects[i])) {
                freeRects.erase(freeRects.begin() + static_cast<std::ptrdiff_t>(i));
                --i;
                break;
            } else {
                ++j;
            }
        }
    }
}

static void splitFreeRect(std::vector<FreeRect>& freeRects, const FreeRect& fr, int px, int py, int pw, int ph) {
    // If no intersection, keep as-is.
    int frRight = fr.x + fr.w;
    int frTop = fr.y + fr.h;
    int pRight = px + pw;
    int pTop = py + ph;
    if (pRight <= fr.x || px >= frRight || pTop <= fr.y || py >= frTop) {
        freeRects.push_back(fr);
        return;
    }

    // Split into up to 4 rectangles.
    // Left
    if (px > fr.x) {
        freeRects.push_back(FreeRect{fr.x, fr.y, px - fr.x, fr.h});
    }
    // Right
    if (pRight < frRight) {
        freeRects.push_back(FreeRect{pRight, fr.y, frRight - pRight, fr.h});
    }
    // Bottom
    if (py > fr.y) {
        int nx = std::max(fr.x, px);
        int nw = std::min(frRight, pRight) - nx;
        if (nw > 0) {
            freeRects.push_back(FreeRect{nx, fr.y, nw, py - fr.y});
        }
    }
    // Top
    if (pTop < frTop) {
        int nx = std::max(fr.x, px);
        int nw = std::min(frRight, pRight) - nx;
        if (nw > 0) {
            freeRects.push_back(FreeRect{nx, pTop, nw, frTop - pTop});
        }
    }
}

struct MaxRectsBin {
    int W{};
    int H{};
    std::vector<FreeRect> freeRects;

    explicit MaxRectsBin(int w, int h) : W(w), H(h) {
        freeRects.push_back(FreeRect{0, 0, W, H});
    }

    // Best Short Side Fit with tie-breaker on long side fit.
    bool place(int rw, int rh, int& outX, int& outY, int& outRot) {
        int bestIdx = -1;
        int bestRot = 0;
        int bestX = 0, bestY = 0;
        int bestShort = std::numeric_limits<int>::max();
        int bestLong = std::numeric_limits<int>::max();

        for (int i = 0; i < static_cast<int>(freeRects.size()); ++i) {
            const auto& fr = freeRects[i];
            // Not rotated
            if (rw <= fr.w && rh <= fr.h) {
                int leftoverH = fr.h - rh;
                int leftoverW = fr.w - rw;
                int shortSide = std::min(leftoverH, leftoverW);
                int longSide = std::max(leftoverH, leftoverW);
                if (shortSide < bestShort || (shortSide == bestShort && longSide < bestLong)) {
                    bestIdx = i;
                    bestRot = 0;
                    bestX = fr.x;
                    bestY = fr.y;
                    bestShort = shortSide;
                    bestLong = longSide;
                }
            }
            // Rotated
            if (rh <= fr.w && rw <= fr.h) {
                int leftoverH = fr.h - rw;
                int leftoverW = fr.w - rh;
                int shortSide = std::min(leftoverH, leftoverW);
                int longSide = std::max(leftoverH, leftoverW);
                if (shortSide < bestShort || (shortSide == bestShort && longSide < bestLong)) {
                    bestIdx = i;
                    bestRot = 1;
                    bestX = fr.x;
                    bestY = fr.y;
                    bestShort = shortSide;
                    bestLong = longSide;
                }
            }
        }

        if (bestIdx < 0) return false;

        int pw = bestRot ? rh : rw;
        int ph = bestRot ? rw : rh;

        // Split all free rects against the placed rectangle.
        std::vector<FreeRect> newFree;
        newFree.reserve(freeRects.size() * 2);
        for (const auto& fr : freeRects) {
            splitFreeRect(newFree, fr, bestX, bestY, pw, ph);
        }
        freeRects.swap(newFree);
        pruneFreeList(freeRects);

        outX = bestX;
        outY = bestY;
        outRot = bestRot;
        return true;
    }
};

// ------------------------ Shelf (large N) ------------------------
struct Shelf {
    int y{};
    int height{};
    int usedW{};
};

struct ShelfBoard {
    int W{};
    int H{};
    int usedH{};
    std::vector<Shelf> shelves;

    explicit ShelfBoard(int w, int h) : W(w), H(h), usedH(0) {}

    bool tryPlace(int rw, int rh, int& outX, int& outY, int& outRot) {
        // Best-fit shelf by remaining width; allow rotation. Only place into shelves that can fit height.
        int bestShelf = -1;
        int bestRot = 0;
        int bestRem = std::numeric_limits<int>::max();

        for (int i = 0; i < static_cast<int>(shelves.size()); ++i) {
            const auto& s = shelves[i];
            int remW = W - s.usedW;
            // not rotated
            if (rh <= s.height && rw <= remW) {
                int r = remW - rw;
                if (r < bestRem) {
                    bestRem = r;
                    bestShelf = i;
                    bestRot = 0;
                }
            }
            // rotated
            if (rw <= s.height && rh <= remW) {
                int r = remW - rh;
                if (r < bestRem) {
                    bestRem = r;
                    bestShelf = i;
                    bestRot = 1;
                }
            }
        }

        if (bestShelf >= 0) {
            auto& s = shelves[bestShelf];
            outX = s.usedW;
            outY = s.y;
            outRot = bestRot;
            s.usedW += (bestRot ? rh : rw);
            return true;
        }

        // Open a new shelf.
        // Choose orientation that minimizes shelf height while fitting board width.
        int chooseRot = -1;
        int sh = 0;
        if (fits(W, H - usedH, rw, rh)) {
            chooseRot = 0;
            sh = rh;
        }
        if (fits(W, H - usedH, rh, rw)) {
            int candH = rw;
            if (chooseRot < 0 || candH < sh || (candH == sh && rh > rw)) {
                chooseRot = 1;
                sh = candH;
            }
        }
        if (chooseRot < 0) return false;
        if (usedH + sh > H) return false;

        Shelf ns;
        ns.y = usedH;
        ns.height = sh;
        ns.usedW = (chooseRot ? rh : rw);
        if (ns.usedW > W) return false;

        shelves.push_back(ns);
        outX = 0;
        outY = ns.y;
        outRot = chooseRot;
        usedH += sh;
        return true;
    }
};

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int N = 0;
    int BW = 0, BH = 0;
    if (!(std::cin >> N >> BW >> BH)) {
        return 0;
    }

    std::vector<RectInput> rects;
    rects.reserve(static_cast<size_t>(N));
    for (int i = 0; i < N; ++i) {
        int w = 0, h = 0;
        std::cin >> w >> h;
        rects.push_back(RectInput{i, w, h});
    }

    // Basic infeasibility check: any rectangle that can't fit in either orientation => impossible.
    // Still output something deterministic: each unplaceable goes on board 0 at (0,0), rotated=0.
    // (Benchmarks typically ensure feasibility; this avoids UB and keeps runtime bounded.)
    std::vector<Placement> ans(static_cast<size_t>(N));
    for (const auto& r : rects) {
        if (!fits(BW, BH, r.w, r.h) && !fits(BW, BH, r.h, r.w)) {
            ans[static_cast<size_t>(r.id)] = Placement{0, 0, 0, 0};
        }
    }

    // Sort by descending area, then max side, then id for determinism.
    std::sort(rects.begin(), rects.end(), [](const RectInput& a, const RectInput& b) {
        long long areaA = 1LL * a.w * a.h;
        long long areaB = 1LL * b.w * b.h;
        if (areaA != areaB) return areaA > areaB;
        int ma = std::max(a.w, a.h);
        int mb = std::max(b.w, b.h);
        if (ma != mb) return ma > mb;
        return a.id < b.id;
    });

    // Heuristic switch: MaxRects for smaller instances; shelf packing for large.
    const bool useMaxRects = (N <= 8000);

    int boardsUsed = 0;
    if (useMaxRects) {
        std::vector<MaxRectsBin> boards;
        boards.reserve(std::max(1, N / 8));

        for (const auto& r : rects) {
            // Skip already-marked unplaceables (rare).
            if (!fits(BW, BH, r.w, r.h) && !fits(BW, BH, r.h, r.w)) continue;

            bool placed = false;
            for (int bi = 0; bi < static_cast<int>(boards.size()); ++bi) {
                int x = 0, y = 0, rot = 0;
                if (boards[bi].place(r.w, r.h, x, y, rot)) {
                    ans[static_cast<size_t>(r.id)] = Placement{bi, x, y, rot};
                    placed = true;
                    break;
                }
            }
            if (!placed) {
                boards.emplace_back(BW, BH);
                int x = 0, y = 0, rot = 0;
                // Must succeed given feasibility.
                if (!boards.back().place(r.w, r.h, x, y, rot)) {
                    // Fallback (shouldn't happen): force board 0.
                    ans[static_cast<size_t>(r.id)] = Placement{0, 0, 0, 0};
                } else {
                    int bi = static_cast<int>(boards.size()) - 1;
                    ans[static_cast<size_t>(r.id)] = Placement{bi, x, y, rot};
                }
            }
        }
        boardsUsed = static_cast<int>(boards.size());
    } else {
        std::vector<ShelfBoard> boards;
        boards.reserve(std::max(1, N / 64));

        for (const auto& r : rects) {
            if (!fits(BW, BH, r.w, r.h) && !fits(BW, BH, r.h, r.w)) continue;

            bool placed = false;
            for (int bi = 0; bi < static_cast<int>(boards.size()); ++bi) {
                int x = 0, y = 0, rot = 0;
                if (boards[bi].tryPlace(r.w, r.h, x, y, rot)) {
                    ans[static_cast<size_t>(r.id)] = Placement{bi, x, y, rot};
                    placed = true;
                    break;
                }
            }
            if (!placed) {
                boards.emplace_back(BW, BH);
                int x = 0, y = 0, rot = 0;
                if (!boards.back().tryPlace(r.w, r.h, x, y, rot)) {
                    ans[static_cast<size_t>(r.id)] = Placement{0, 0, 0, 0};
                } else {
                    int bi = static_cast<int>(boards.size()) - 1;
                    ans[static_cast<size_t>(r.id)] = Placement{bi, x, y, rot};
                }
            }
        }
        boardsUsed = static_cast<int>(boards.size());
    }

    // Ensure all placements have a non-negative board index.
    for (auto& p : ans) {
        if (p.board < 0) p.board = 0;
    }
    // Compute boardsUsed if some were forced to board 0 without allocating.
    int maxBoard = 0;
    for (const auto& p : ans) maxBoard = std::max(maxBoard, p.board);
    boardsUsed = std::max(boardsUsed, maxBoard + 1);

    std::cout << boardsUsed << "\n";
    for (int i = 0; i < N; ++i) {
        const auto& p = ans[static_cast<size_t>(i)];
        std::cout << p.board << " " << p.x << " " << p.y << " " << p.rotated << "\n";
    }
    return 0;
}
