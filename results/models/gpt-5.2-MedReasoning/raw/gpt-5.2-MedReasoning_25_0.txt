// Asteroid interception trajectory solver (heuristic baseline)
// Reads scenario from stdin and outputs a small set of burns.
// This is a pragmatic, compilation-safe solver intended for benchmark harnesses.
//
// Strategy (simplified inertial model):
// - Treat each body as moving with constant velocity in the barycentric frame.
// - Estimate the asteroid–Earth "impact" time as the time of closest approach.
// - Search over feasible launch time within [0, warning_days] and intercept time
//   before closest approach to find a single-burn intercept that stays within
//   delta-v budget (or uses as much as available if needed).
// - Output 2 burns: launch burn and an "impact" line with zero dv at intercept time.
//
// Notes:
// - The host simulator uses full n-body + DE421; this solver intentionally uses a
//   lightweight approximation to ensure it always runs fast and produces valid output.
// - Output times are in days since t=0, as required.

using System;
using System.Collections.Generic;
using System.Globalization;

public static class Program
{
    private struct Vec3
    {
        public double X, Y, Z;
        public Vec3(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec3 operator +(Vec3 a, Vec3 b) => new Vec3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec3 operator -(Vec3 a, Vec3 b) => new Vec3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec3 operator *(Vec3 a, double s) => new Vec3(a.X * s, a.Y * s, a.Z * s);
        public static Vec3 operator /(Vec3 a, double s) => new Vec3(a.X / s, a.Y / s, a.Z / s);
        public double Dot(Vec3 b) => X * b.X + Y * b.Y + Z * b.Z;
        public double Norm() => Math.Sqrt(Dot(this));
        public Vec3 Normalized()
        {
            var n = Norm();
            if (n <= 0) return new Vec3(0, 0, 0);
            return this / n;
        }
    }

    private sealed class TokenReader
    {
        private readonly string[] _tokens;
        private int _i;

        public TokenReader(string input)
        {
            _tokens = input.Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
            _i = 0;
        }

        public bool HasMore => _i < _tokens.Length;

        public string Next()
        {
            if (_i >= _tokens.Length) throw new InvalidOperationException("Unexpected end of input.");
            return _tokens[_i++];
        }

        public int NextInt() => int.Parse(Next(), CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(Next(), CultureInfo.InvariantCulture);
    }

    private sealed class Body
    {
        public string Name = "";
        public double Mass;
        public Vec3 R0;
        public Vec3 V0;
        public Vec3 PosAt(double tSeconds) => R0 + V0 * tSeconds;
        public Vec3 VelAt(double _tSeconds) => V0;
    }

    public static void Main()
    {
        // Read entire stdin as whitespace tokens.
        var input = Console.In.ReadToEnd();
        if (string.IsNullOrWhiteSpace(input))
            return;

        var tr = new TokenReader(input);

        int numBodies = tr.NextInt();
        double warningDays = tr.NextDouble();
        double impactorMass = tr.NextDouble();
        double dvBudget = tr.NextDouble();
        double asteroidMass = tr.NextDouble();
        double t0UnixUtc = tr.NextDouble(); // unused; days since scenario start are required

        var bodies = new Dictionary<string, Body>(StringComparer.OrdinalIgnoreCase);
        for (int i = 0; i < numBodies; i++)
        {
            var b = new Body();
            b.Name = tr.Next();
            b.Mass = tr.NextDouble();
            b.R0 = new Vec3(tr.NextDouble(), tr.NextDouble(), tr.NextDouble());
            b.V0 = new Vec3(tr.NextDouble(), tr.NextDouble(), tr.NextDouble());
            bodies[b.Name] = b;
        }

        // de421.bsp_file_path is provided but not used by this baseline solver.
        // Consume it if present (may contain characters without spaces, per prompt).
        string de421Path = tr.HasMore ? tr.Next() : "";
        _ = de421Path;
        _ = impactorMass;
        _ = asteroidMass;
        _ = t0UnixUtc;

        if (!bodies.TryGetValue("Earth", out var earth) || !bodies.TryGetValue("Asteroid", out var asteroid))
        {
            // Fallback: emit a valid minimal output (one burn at t=0 with zero dv).
            Console.WriteLine("1");
            Console.WriteLine("0 0 0 0");
            return;
        }

        const double Day = 86400.0;
        double warningSeconds = warningDays * Day;

        // Estimate "impact time" as time of closest approach under constant velocity.
        // r(t) = (rA0 - rE0) + (vA0 - vE0) t
        // minimize |r(t)|^2 => t* = - (r0·vrel) / |vrel|^2
        Vec3 r0 = asteroid.R0 - earth.R0;
        Vec3 vrel = asteroid.V0 - earth.V0;
        double vrel2 = vrel.Dot(vrel);
        double tClosest = 0.0;
        if (vrel2 > 0)
            tClosest = -r0.Dot(vrel) / vrel2;
        if (tClosest < 0) tClosest = 0;

        // If closest approach is beyond warning horizon, cap to warning horizon for planning.
        // We still allow intercept slightly after warningSeconds (launch must be within warning,
        // intercept can be later), but we avoid extremely large values.
        double tPlanHorizon = Math.Max(warningSeconds, tClosest);
        if (tPlanHorizon < Day) tPlanHorizon = Day;
        // Hard cap to keep search bounded (50 years ~ 1.6e9 s); cap at 60 years.
        double tCap = 60.0 * 365.25 * Day;
        if (tPlanHorizon > tCap) tPlanHorizon = tCap;

        // Search for best (launchTime, interceptTime) pair.
        // Launch within [0, warningSeconds]. Intercept must be after launch.
        // We aim to arrive before closest approach if possible; otherwise any time within horizon.
        double interceptLatest = tClosest > 0 ? Math.Min(tClosest, tPlanHorizon) : tPlanHorizon;
        if (interceptLatest < 2 * Day) interceptLatest = 2 * Day;

        // Candidate launch times: include endpoints and a coarse grid.
        // Candidate intercept times: for each launch, search over intercept grid.
        // Use adaptive step sizes to cover long horizons quickly.
        var launchCandidates = BuildTimeGrid(0.0, warningSeconds, 16);
        if (launchCandidates.Count == 0) launchCandidates.Add(0.0);
        if (launchCandidates[launchCandidates.Count - 1] != warningSeconds) launchCandidates.Add(warningSeconds);

        double bestScore = double.PositiveInfinity;
        double bestLaunch = 0.0;
        double bestIntercept = Math.Max(2 * Day, Math.Min(interceptLatest, 30 * Day));
        Vec3 bestDv = new Vec3(0, 0, 0);

        foreach (var tLaunch in launchCandidates)
        {
            // Intercept must be at least 1 hour after launch to avoid singular dt.
            double tMin = tLaunch + 3600.0;
            double tMax = Math.Max(tMin + Day, interceptLatest);

            // Build intercept grid (log-ish spacing for coverage).
            var interceptCandidates = BuildTimeGrid(tMin, tMax, 40);

            foreach (var tIntercept in interceptCandidates)
            {
                double dt = tIntercept - tLaunch;
                if (dt <= 0) continue;

                Vec3 rE = earth.PosAt(tLaunch);
                Vec3 vE = earth.VelAt(tLaunch);
                Vec3 rA = asteroid.PosAt(tIntercept);

                // Required inertial spacecraft velocity at launch to hit asteroid at tIntercept:
                // rS(tIntercept) = rE + (vE + dv) * dt = rA  => dv = (rA - rE)/dt - vE
                Vec3 dv = (rA - rE) / dt - vE;
                double dvMag = dv.Norm();

                // Objective: prefer within budget; otherwise minimize over-budget amount,
                // then prefer earlier intercept and smaller dv.
                double over = Math.Max(0.0, dvMag - dvBudget);
                double timePenalty = (tIntercept / Day) * 1e-4; // mild preference for earlier
                double score = over * 1e6 + dvMag + timePenalty;

                if (score < bestScore)
                {
                    bestScore = score;
                    bestLaunch = tLaunch;
                    bestIntercept = tIntercept;
                    bestDv = dv;
                }
            }
        }

        // Enforce delta-v budget by scaling if needed.
        double bestDvMag = bestDv.Norm();
        if (bestDvMag > dvBudget && bestDvMag > 0)
        {
            bestDv = bestDv * (dvBudget / bestDvMag);
        }

        // Output:
        // - 2 burns: launch + impact (impact dv is 0; kinetic impact occurs by collision).
        // Times must be in days since scenario start.
        Console.WriteLine("2");
        Console.WriteLine(FormLine(bestLaunch / Day, bestDv));
        Console.WriteLine(FormLine(bestIntercept / Day, new Vec3(0, 0, 0)));
    }

    private static string FormLine(double timeDays, Vec3 dv)
    {
        return string.Format(CultureInfo.InvariantCulture, "{0:R} {1:R} {2:R} {3:R}", timeDays, dv.X, dv.Y, dv.Z);
    }

    private static List<double> BuildTimeGrid(double tStart, double tEnd, int n)
    {
        var list = new List<double>(n + 2);
        if (tEnd <= tStart)
        {
            list.Add(tStart);
            return list;
        }

        // Blend linear and logarithmic spacing by sampling in log-time relative to start.
        // This gives coverage near tStart while still reaching tEnd.
        double span = tEnd - tStart;
        double eps = Math.Max(1.0, span * 1e-9);
        double a = eps;
        double b = span + eps;

        for (int i = 0; i < n; i++)
        {
            double u = n == 1 ? 0.0 : (double)i / (n - 1);
            // log interpolation
            double x = a * Math.Pow(b / a, u) - eps; // in [0, span]
            double t = tStart + x;
            if (t < tStart) t = tStart;
            if (t > tEnd) t = tEnd;
            // avoid duplicates
            if (list.Count == 0 || Math.Abs(t - list[list.Count - 1]) > 1e-6)
                list.Add(t);
        }

        // Ensure endpoints
        if (list.Count == 0 || list[0] != tStart) list.Insert(0, tStart);
        if (list[list.Count - 1] != tEnd) list.Add(tEnd);
        return list;
    }
}
