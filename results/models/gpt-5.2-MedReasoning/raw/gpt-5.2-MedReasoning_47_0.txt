@group(0) @binding(0) var<storage, read>       baseData : array<u32, 16>;
@group(0) @binding(1) var<storage, read_write> result   : array<u32, 4>;
@group(0) @binding(2) var<uniform>             params   : vec4<u32>;
// params.x = difficulty, params.y = range_start, params.z = range_size

fn rotl32(x: u32, n: u32) -> u32 {
  return (x << n) | (x >> (32u - n));
}

fn trailing_zeros_64(h0: u32, h1: u32) -> u32 {
  if (h0 != 0u) {
    return countTrailingZeros(h0);
  }
  if (h1 != 0u) {
    return 32u + countTrailingZeros(h1);
  }
  return 64u;
}

fn custom_hash(nonce: u32) -> vec2<u32> {
  var state: array<u32, 16>;

  // state[0..15] = baseData[0..15]; then state[i] ^= (nonce + i)
  for (var i: u32 = 0u; i < 16u; i = i + 1u) {
    state[i] = baseData[i] ^ (nonce + i);
  }

  // 8 rounds of quarter-round mixing (ChaCha-like), unrolled groups of 4.
  for (var r: u32 = 0u; r < 8u; r = r + 1u) {
    // Group 0: (0, 1, 2, 3)
    var a0: u32 = state[0u];
    var b0: u32 = state[1u];
    var c0: u32 = state[2u];
    var d0: u32 = state[3u];
    a0 = a0 + b0; d0 = rotl32(d0 ^ a0, 16u);
    c0 = c0 + d0; b0 = rotl32(b0 ^ c0, 12u);
    a0 = a0 + b0; d0 = rotl32(d0 ^ a0, 8u);
    c0 = c0 + d0; b0 = rotl32(b0 ^ c0, 7u);
    state[0u] = a0; state[1u] = b0; state[2u] = c0; state[3u] = d0;

    // Group 1: (4, 5, 6, 7)
    var a1: u32 = state[4u];
    var b1: u32 = state[5u];
    var c1: u32 = state[6u];
    var d1: u32 = state[7u];
    a1 = a1 + b1; d1 = rotl32(d1 ^ a1, 16u);
    c1 = c1 + d1; b1 = rotl32(b1 ^ c1, 12u);
    a1 = a1 + b1; d1 = rotl32(d1 ^ a1, 8u);
    c1 = c1 + d1; b1 = rotl32(b1 ^ c1, 7u);
    state[4u] = a1; state[5u] = b1; state[6u] = c1; state[7u] = d1;

    // Group 2: (8, 9, 10, 11)
    var a2: u32 = state[8u];
    var b2: u32 = state[9u];
    var c2: u32 = state[10u];
    var d2: u32 = state[11u];
    a2 = a2 + b2; d2 = rotl32(d2 ^ a2, 16u);
    c2 = c2 + d2; b2 = rotl32(b2 ^ c2, 12u);
    a2 = a2 + b2; d2 = rotl32(d2 ^ a2, 8u);
    c2 = c2 + d2; b2 = rotl32(b2 ^ c2, 7u);
    state[8u] = a2; state[9u] = b2; state[10u] = c2; state[11u] = d2;

    // Group 3: (12, 13, 14, 15)
    var a3: u32 = state[12u];
    var b3: u32 = state[13u];
    var c3: u32 = state[14u];
    var d3: u32 = state[15u];
    a3 = a3 + b3; d3 = rotl32(d3 ^ a3, 16u);
    c3 = c3 + d3; b3 = rotl32(b3 ^ c3, 12u);
    a3 = a3 + b3; d3 = rotl32(d3 ^ a3, 8u);
    c3 = c3 + d3; b3 = rotl32(b3 ^ c3, 7u);
    state[12u] = a3; state[13u] = b3; state[14u] = c3; state[15u] = d3;
  }

  // Fold 16 words to 2.
  let h0: u32 =
    state[0u] ^ state[2u] ^ state[4u] ^ state[6u] ^
    state[8u] ^ state[10u] ^ state[12u] ^ state[14u];
  let h1: u32 =
    state[1u] ^ state[3u] ^ state[5u] ^ state[7u] ^
    state[9u] ^ state[11u] ^ state[13u] ^ state[15u];

  return vec2<u32>(h0, h1);
}

@compute @workgroup_size(256)
fn main(@builtin(global_invocation_id) gid: vec3<u32>) {
  // Guard out of range.
  if (gid.x >= params.z) {
    return;
  }

  let difficulty: u32 = params.x;
  let nonce: u32 = params.y + gid.x;

  let h: vec2<u32> = custom_hash(nonce);
  let h0: u32 = h.x;
  let h1: u32 = h.y;

  let tz: u32 = trailing_zeros_64(h0, h1);
  if (tz >= difficulty) {
    result[0u] = 1u;
    result[1u] = nonce;
    result[2u] = h0;
    result[3u] = h1;
  }
}
