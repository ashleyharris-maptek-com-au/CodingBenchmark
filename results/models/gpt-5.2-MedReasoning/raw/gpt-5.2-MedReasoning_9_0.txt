using System;
using System.IO;
using System.Text;

public static class Program
{
    private readonly struct Cut
    {
        public readonly int Len;
        public readonly int Id;
        public Cut(int len, int id) { Len = len; Id = id; }
    }

    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 20];
        private int _len, _pos;
        public FastScanner(Stream s) { _s = s; }

        private int ReadByte()
        {
            if (_pos >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _pos = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_pos++];
        }

        public int ReadInt()
        {
            int c;
            do { c = ReadByte(); if (c < 0) return int.MinValue; } while (c <= 32);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int val = 0;
            while (c > 32)
            {
                val = (val * 10) + (c - '0');
                c = ReadByte();
            }
            return val * sign;
        }
    }

    private sealed class Treap
    {
        private sealed class Node
        {
            public int Key;
            public int Prio;
            public Node? L;
            public Node? R;
            public int HeadBin; // head of linked list of bin ids for this Key, -1 if none
            public Node(int key, int prio, int headBin) { Key = key; Prio = prio; HeadBin = headBin; }
        }

        private Node? _root;
        private readonly uint[] _rng = new uint[1];

        public Treap(uint seed)
        {
            _rng[0] = seed != 0 ? seed : 2463534242u;
        }

        private int NextPrio()
        {
            // xorshift32
            uint x = _rng[0];
            x ^= x << 13;
            x ^= x >> 17;
            x ^= x << 5;
            _rng[0] = x;
            return (int)(x & 0x7fffffff);
        }

        private static void RotateRight(ref Node p)
        {
            Node q = p.L!;
            p.L = q.R;
            q.R = p;
            p = q;
        }

        private static void RotateLeft(ref Node p)
        {
            Node q = p.R!;
            p.R = q.L;
            q.L = p;
            p = q;
        }

        public void InsertOrPush(int key, int binId, int[] nextSame)
        {
            InsertOrPush(ref _root, key, binId, nextSame);
        }

        private void InsertOrPush(ref Node? cur, int key, int binId, int[] nextSame)
        {
            if (cur == null)
            {
                cur = new Node(key, NextPrio(), binId);
                nextSame[binId] = -1;
                return;
            }

            if (key == cur.Key)
            {
                nextSame[binId] = cur.HeadBin;
                cur.HeadBin = binId;
                return;
            }

            if (key < cur.Key)
            {
                InsertOrPush(ref cur.L, key, binId, nextSame);
                if (cur.L!.Prio > cur.Prio) RotateRight(ref cur);
            }
            else
            {
                InsertOrPush(ref cur.R, key, binId, nextSame);
                if (cur.R!.Prio > cur.Prio) RotateLeft(ref cur);
            }
        }

        public bool TryPopLowerBound(int needed, int[] nextSame, out int key, out int binId)
        {
            Node? cur = _root;
            Node? best = null;
            while (cur != null)
            {
                if (cur.Key >= needed)
                {
                    best = cur;
                    cur = cur.L;
                }
                else cur = cur.R;
            }

            if (best == null)
            {
                key = 0;
                binId = -1;
                return false;
            }

            key = best.Key;
            binId = best.HeadBin;
            best.HeadBin = nextSame[binId];
            nextSame[binId] = -1;

            if (best.HeadBin == -1)
            {
                _root = Delete(_root, best.Key);
            }
            return true;
        }

        private static Node? Delete(Node? cur, int key)
        {
            if (cur == null) return null;
            if (key < cur.Key)
            {
                cur.L = Delete(cur.L, key);
                return cur;
            }
            if (key > cur.Key)
            {
                cur.R = Delete(cur.R, key);
                return cur;
            }

            // cur.Key == key
            if (cur.L == null) return cur.R;
            if (cur.R == null) return cur.L;

            if (cur.L.Prio > cur.R.Prio)
            {
                RotateRight(ref cur);
                cur.R = Delete(cur.R, key);
            }
            else
            {
                RotateLeft(ref cur);
                cur.L = Delete(cur.L, key);
            }
            return cur;
        }
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.ReadInt();
        if (n == int.MinValue) return;
        int stockLen = fs.ReadInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
        {
            int len = fs.ReadInt();
            if ((uint)len > (uint)stockLen) len = stockLen; // assume valid, clamp defensively
            cuts[i] = new Cut(len, i);
        }

        Array.Sort(cuts, static (a, b) => b.Len.CompareTo(a.Len));

        int[] binRemaining = new int[Math.Max(4, Math.Min(n, 1024))];
        int[] nextSame = new int[binRemaining.Length];
        for (int i = 0; i < nextSame.Length; i++) nextSame[i] = -1;

        int[] binOfCut = new int[n];
        int binCount = 0;

        var treap = new Treap((uint)Environment.TickCount);

        for (int i = 0; i < n; i++)
        {
            int len = cuts[i].Len;
            if (!treap.TryPopLowerBound(len, nextSame, out int remKey, out int binId))
            {
                binId = binCount++;
                if (binId >= binRemaining.Length)
                {
                    int oldCap = binRemaining.Length;
                    int newCap = binRemaining.Length < (1 << 20) ? binRemaining.Length * 2 : binRemaining.Length + (binRemaining.Length >> 1);
                    if (newCap < binId + 1) newCap = binId + 1;
                    Array.Resize(ref binRemaining, newCap);
                    Array.Resize(ref nextSame, newCap);
                    for (int j = oldCap; j < newCap; j++) nextSame[j] = -1;
                }
                int newRem = stockLen - len;
                binRemaining[binId] = newRem;
                binOfCut[cuts[i].Id] = binId;
                if (newRem > 0) treap.InsertOrPush(newRem, binId, nextSame);
            }
            else
            {
                int newRem = remKey - len;
                binRemaining[binId] = newRem;
                binOfCut[cuts[i].Id] = binId;
                if (newRem > 0) treap.InsertOrPush(newRem, binId, nextSame);
            }
        }

        // Build per-bin lists using counting + prefix sums.
        int[] counts = new int[binCount];
        for (int i = 0; i < n; i++) counts[binOfCut[i]]++;
        int[] offsets = new int[binCount + 1];
        for (int b = 0; b < binCount; b++) offsets[b + 1] = offsets[b] + counts[b];
        int[] writePos = new int[binCount];
        Array.Copy(offsets, 0, writePos, 0, binCount);
        int[] allCuts = new int[n];
        for (int i = 0; i < n; i++)
        {
            int b = binOfCut[i];
            allCuts[writePos[b]++] = i;
        }

        var stdout = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        stdout.WriteLine(binCount);
        for (int b = 0; b < binCount; b++)
        {
            int start = offsets[b];
            int end = offsets[b + 1];
            if (start >= end)
            {
                stdout.WriteLine();
                continue;
            }
            stdout.Write(allCuts[start]);
            for (int p = start + 1; p < end; p++)
            {
                stdout.Write(' ');
                stdout.Write(allCuts[p]);
            }
            stdout.WriteLine();
        }
        stdout.Flush();
    }
}
