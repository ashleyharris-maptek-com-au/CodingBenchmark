using System;
using System.Collections.Generic;
using System.IO;
using System.Linq;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _s;
        private readonly byte[] _buf = new byte[1 << 16];
        private int _len, _ptr;
        public FastScanner(Stream s) { _s = s; }
        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buf[_ptr++];
        }
        public int NextInt()
        {
            int c;
            do { c = ReadByte(); if (c == -1) return int.MinValue; } while (c <= ' '); // skip
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int x = 0;
            while (c > ' ')
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private sealed class Graph
    {
        public readonly int N;
        public readonly List<int>[] Adj;
        public readonly List<int>[] Radj;
        public Graph(int n)
        {
            N = n;
            Adj = new List<int>[n];
            Radj = new List<int>[n];
            for (int i = 0; i < n; i++)
            {
                Adj[i] = new List<int>();
                Radj[i] = new List<int>();
            }
        }
        public void AddEdge(int u, int v)
        {
            Adj[u].Add(v);
            Radj[v].Add(u);
        }
    }

    private static List<int> KosarajuScc(Graph g, out int[] compOf, out List<int>[] comps)
    {
        int n = g.N;
        bool[] vis = new bool[n];
        int[] order = new int[n];
        int ordSz = 0;

        // iterative DFS for finish order
        int[] it = new int[n];
        int[] stack = new int[n];
        int sp = 0;
        for (int start = 0; start < n; start++)
        {
            if (vis[start]) continue;
            stack[sp++] = start;
            while (sp > 0)
            {
                int v = stack[sp - 1];
                if (!vis[v])
                {
                    vis[v] = true;
                    it[v] = 0;
                }
                var adj = g.Adj[v];
                if (it[v] < adj.Count)
                {
                    int to = adj[it[v]++];
                    if (!vis[to]) stack[sp++] = to;
                }
                else
                {
                    sp--;
                    order[ordSz++] = v;
                }
            }
        }

        compOf = new int[n];
        Array.Fill(compOf, -1);
        List<List<int>> compLists = new List<List<int>>();

        // second pass on reverse graph
        int[] st2 = new int[n];
        for (int i = ordSz - 1; i >= 0; i--)
        {
            int v = order[i];
            if (compOf[v] != -1) continue;
            int cid = compLists.Count;
            var list = new List<int>();
            int top = 0;
            st2[top++] = v;
            compOf[v] = cid;
            while (top > 0)
            {
                int x = st2[--top];
                list.Add(x);
                foreach (int to in g.Radj[x])
                {
                    if (compOf[to] == -1)
                    {
                        compOf[to] = cid;
                        st2[top++] = to;
                    }
                }
            }
            compLists.Add(list);
        }

        comps = new List<int>[compLists.Count];
        for (int i = 0; i < compLists.Count; i++) comps[i] = compLists[i];
        // Return component ids in topological reverse finishing (not needed)
        return new List<int>(Enumerable.Range(0, compLists.Count));
    }

    private static bool HasSelfLoop(Graph g, int v)
    {
        var adj = g.Adj[v];
        for (int i = 0; i < adj.Count; i++) if (adj[i] == v) return true;
        return false;
    }

    private sealed class InducedSubgraph
    {
        public readonly int[] Nodes;          // global vertex ids
        public readonly int Size;
        public readonly int[] MapToLocal;     // global->local, -1 if not in
        public readonly int[][] Adj;          // local adjacency within induced
        public InducedSubgraph(Graph g, List<int> nodes)
        {
            Nodes = nodes.ToArray();
            Size = Nodes.Length;
            MapToLocal = new int[g.N];
            Array.Fill(MapToLocal, -1);
            for (int i = 0; i < Size; i++) MapToLocal[Nodes[i]] = i;
            Adj = new int[Size][];
            for (int i = 0; i < Size; i++)
            {
                int gv = Nodes[i];
                var list = new List<int>();
                foreach (int toG in g.Adj[gv])
                {
                    int to = MapToLocal[toG];
                    if (to != -1) list.Add(to);
                }
                Adj[i] = list.ToArray();
            }
        }
    }

    private static bool FindAnyCycle(InducedSubgraph sg, bool[] removed, List<int> cycleOut)
    {
        int n = sg.Size;
        int[] color = new int[n]; // 0=unseen,1=visiting,2=done
        int[] parent = new int[n];
        Array.Fill(parent, -1);

        // iterative DFS with explicit stack of (v, nextIdx)
        int[] stackV = new int[n];
        int[] stackI = new int[n];

        for (int s = 0; s < n; s++)
        {
            if (removed[s] || color[s] != 0) continue;
            int sp = 0;
            stackV[sp] = s;
            stackI[sp] = 0;
            parent[s] = -1;
            color[s] = 1;
            sp++;

            while (sp > 0)
            {
                int v = stackV[sp - 1];
                int idx = stackI[sp - 1];
                var adj = sg.Adj[v];
                bool advanced = false;
                while (idx < adj.Length)
                {
                    int to = adj[idx++];
                    if (removed[to]) continue;
                    if (color[to] == 0)
                    {
                        stackI[sp - 1] = idx;
                        parent[to] = v;
                        color[to] = 1;
                        stackV[sp] = to;
                        stackI[sp] = 0;
                        sp++;
                        advanced = true;
                        break;
                    }
                    if (color[to] == 1)
                    {
                        // back edge v->to, reconstruct cycle
                        cycleOut.Clear();
                        cycleOut.Add(to);
                        int cur = v;
                        while (cur != to && cur != -1)
                        {
                            cycleOut.Add(cur);
                            cur = parent[cur];
                        }
                        // cycleOut currently [to, v, ..., to excluded end]; make unique, ok for branching
                        return true;
                    }
                }
                if (advanced) continue;
                // finished
                color[v] = 2;
                sp--;
            }
        }
        return false;
    }

    private static int ReduceTrivial(InducedSubgraph sg, bool[] removed, List<int> forcedRemoved)
    {
        // Repeatedly remove vertices with indeg==0 or outdeg==0 within remaining subgraph.
        // This does not change minimum FVS for the SCC induced subproblem.
        int n = sg.Size;
        int[] indeg = new int[n];
        int[] outdeg = new int[n];
        Array.Clear(indeg, 0, n);
        Array.Clear(outdeg, 0, n);
        for (int v = 0; v < n; v++)
        {
            if (removed[v]) continue;
            var adj = sg.Adj[v];
            for (int i = 0; i < adj.Length; i++)
            {
                int to = adj[i];
                if (removed[to]) continue;
                outdeg[v]++;
                indeg[to]++;
            }
        }

        var q = new Queue<int>();
        for (int v = 0; v < n; v++)
        {
            if (removed[v]) continue;
            if (indeg[v] == 0 || outdeg[v] == 0) q.Enqueue(v);
        }

        int changed = 0;
        while (q.Count > 0)
        {
            int v = q.Dequeue();
            if (removed[v]) continue;
            if (indeg[v] != 0 && outdeg[v] != 0) continue;
            removed[v] = true; // "deleted" from consideration without counting to FVS
            changed++;
            var adj = sg.Adj[v];
            for (int i = 0; i < adj.Length; i++)
            {
                int to = adj[i];
                if (removed[to]) continue;
                indeg[to]--;
                if (indeg[to] == 0) q.Enqueue(to);
            }
            // need update indeg/outdeg for predecessors too; compute predecessors on the fly (O(E) worst).
            // For SCC sizes we solve exactly, this is acceptable.
            for (int u = 0; u < n; u++)
            {
                if (removed[u]) continue;
                // check if u->v exists
                var a = sg.Adj[u];
                for (int j = 0; j < a.Length; j++)
                {
                    if (a[j] == v)
                    {
                        outdeg[u]--;
                        if (outdeg[u] == 0) q.Enqueue(u);
                        break;
                    }
                }
            }
        }
        return changed;
    }

    private static bool SearchFvsK(InducedSubgraph sg, bool[] removed, int k, List<int> picked)
    {
        // removed: current removed-from-graph flags (includes "deleted" trivial vertices too)
        // picked: vertices picked into FVS (local indices)
        if (picked.Count > k) return false;

        // Try reduce trivial vertices (indeg==0 or outdeg==0) to shrink cycle space
        // We do it on a copy of removed in recursion frames by manual undo stack.
        // For simplicity/safety, do one cheap reduction pass using undo list.
        var undone = new List<int>();
        {
            int n = sg.Size;
            // compute indeg/outdeg for remaining
            int[] indeg = new int[n];
            int[] outdeg = new int[n];
            for (int v = 0; v < n; v++)
            {
                if (removed[v]) continue;
                var adj = sg.Adj[v];
                for (int i = 0; i < adj.Length; i++)
                {
                    int to = adj[i];
                    if (removed[to]) continue;
                    outdeg[v]++;
                    indeg[to]++;
                }
            }
            var q = new Queue<int>();
            for (int v = 0; v < n; v++)
                if (!removed[v] && (indeg[v] == 0 || outdeg[v] == 0))
                    q.Enqueue(v);
            while (q.Count > 0)
            {
                int v = q.Dequeue();
                if (removed[v]) continue;
                if (indeg[v] != 0 && outdeg[v] != 0) continue;
                removed[v] = true;
                undone.Add(v);
                var adj = sg.Adj[v];
                for (int i = 0; i < adj.Length; i++)
                {
                    int to = adj[i];
                    if (removed[to]) continue;
                    indeg[to]--;
                    if (indeg[to] == 0) q.Enqueue(to);
                }
                // predecessors update (scan)
                for (int u = 0; u < n; u++)
                {
                    if (removed[u]) continue;
                    var a = sg.Adj[u];
                    for (int j = 0; j < a.Length; j++)
                    {
                        if (a[j] == v)
                        {
                            outdeg[u]--;
                            if (outdeg[u] == 0) q.Enqueue(u);
                            break;
                        }
                    }
                }
            }
        }

        try
        {
            var cycle = new List<int>();
            if (!FindAnyCycle(sg, removed, cycle))
                return true;

            if (picked.Count == k)
                return false;

            // Branch on vertices in found cycle (small in practice for these instances)
            // Heuristic: try higher degree first within remaining subgraph.
            cycle.Sort((a, b) =>
            {
                int da = 0, db = 0;
                var aa = sg.Adj[a];
                var bb = sg.Adj[b];
                for (int i = 0; i < aa.Length; i++) if (!removed[aa[i]]) da++;
                for (int i = 0; i < bb.Length; i++) if (!removed[bb[i]]) db++;
                return db.CompareTo(da);
            });

            foreach (int v in cycle)
            {
                if (removed[v]) continue;
                removed[v] = true;
                picked.Add(v);
                if (SearchFvsK(sg, removed, k, picked))
                    return true;
                picked.RemoveAt(picked.Count - 1);
                removed[v] = false;
            }
            return false;
        }
        finally
        {
            // undo trivial removals
            for (int i = 0; i < undone.Count; i++)
                removed[undone[i]] = false;
        }
    }

    private static List<int> SolveSccFvs(Graph g, List<int> sccNodes)
    {
        // If SCC is trivial, return empty or self-loop node.
        if (sccNodes.Count == 1)
        {
            int v = sccNodes[0];
            if (HasSelfLoop(g, v)) return new List<int> { v };
            return new List<int>();
        }

        var sg = new InducedSubgraph(g, sccNodes);
        int n = sg.Size;
        bool[] removed = new bool[n];

        // Fast check: if already acyclic (rare inside SCC but can happen if SCC built from whole graph? no)
        var tmpCycle = new List<int>();
        if (!FindAnyCycle(sg, removed, tmpCycle))
            return new List<int>();

        // Iterative deepening on k
        for (int k = 1; k <= n; k++)
        {
            Array.Clear(removed, 0, n);
            var pickedLocal = new List<int>(k);
            if (SearchFvsK(sg, removed, k, pickedLocal))
            {
                // Map back to global ids
                var res = new List<int>(pickedLocal.Count);
                for (int i = 0; i < pickedLocal.Count; i++)
                    res.Add(sg.Nodes[pickedLocal[i]]);
                // ensure distinct
                return res;
            }
        }

        // Should never reach; worst-case remove all vertices
        return new List<int>(sccNodes);
    }

    public static void Main()
    {
        var fs = new FastScanner(Console.OpenStandardInput());
        int n = fs.NextInt();
        if (n == int.MinValue) return;
        int m = fs.NextInt();
        var g = new Graph(n);
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt();
            int v = fs.NextInt();
            if ((uint)u < (uint)n && (uint)v < (uint)n) g.AddEdge(u, v);
        }

        KosarajuScc(g, out int[] compOf, out List<int>[] comps);

        var answer = new List<int>();
        for (int cid = 0; cid < comps.Length; cid++)
        {
            var nodes = comps[cid];
            if (nodes.Count <= 1)
            {
                if (nodes.Count == 1 && HasSelfLoop(g, nodes[0]))
                    answer.Add(nodes[0]);
                continue;
            }
            // Solve SCC exactly and add union.
            var part = SolveSccFvs(g, nodes);
            if (part.Count > 0) answer.AddRange(part);
        }

        // Output
        // Ensure distinct (SCCs disjoint so should already be distinct)
        if (answer.Count == 0)
        {
            Console.WriteLine(0);
            Console.WriteLine();
            return;
        }
        answer = answer.Distinct().ToList();
        Console.WriteLine(answer.Count);
        for (int i = 0; i < answer.Count; i++)
        {
            if (i > 0) Console.Write(' ');
            Console.Write(answer[i]);
        }
        Console.WriteLine();
    }
}
