import sys


def solve_exact_cover(U, sets_as_lists):
    all_mask = (1 << U) - 1 if U > 0 else 0
    row_mask = []
    rows_of = [[] for _ in range(U)]

    for i, elems in enumerate(sets_as_lists):
        m = 0
        ok = True
        for e in elems:
            if e < 0 or e >= U:
                ok = False
                break
            bit = 1 << e
            if m & bit:
                continue
            m |= bit
        if not ok:
            m = 0
        row_mask.append(m)
        for e in elems:
            if 0 <= e < U:
                rows_of[e].append(i)

    for e in range(U):
        rows_of[e].sort()

    sys.setrecursionlimit(10**7)

    def pick_column(uncovered_mask):
        best_e = None
        best_count = None
        m = uncovered_mask
        covered = all_mask ^ uncovered_mask
        while m:
            lsb = m & -m
            e = lsb.bit_length() - 1
            m ^= lsb
            cnt = 0
            for r in rows_of[e]:
                rm = row_mask[r]
                if rm and (rm & covered) == 0:
                    cnt += 1
            if best_count is None or cnt < best_count or (cnt == best_count and e < best_e):
                best_count = cnt
                best_e = e
                if best_count == 0:
                    break
        return best_e, best_count

    def search(uncovered_mask, chosen):
        if uncovered_mask == 0:
            return chosen
        e, cnt = pick_column(uncovered_mask)
        if cnt == 0 or e is None:
            return None
        covered = all_mask ^ uncovered_mask
        candidates = []
        for r in rows_of[e]:
            rm = row_mask[r]
            if rm and (rm & covered) == 0:
                candidates.append(r)
        for r in candidates:
            new_uncovered = uncovered_mask & ~row_mask[r]
            res = search(new_uncovered, chosen + [r])
            if res is not None:
                return res
        return None

    return search(all_mask, [])


def main():
    data = sys.stdin.read().splitlines()
    if not data:
        return
    first = data[0].strip().split()
    if len(first) < 2:
        return
    U = int(first[0])
    S = int(first[1])
    sets_as_lists = []
    for i in range(S):
        if 1 + i >= len(data):
            line = ""
        else:
            line = data[1 + i].strip()
        if not line:
            sets_as_lists.append([])
        else:
            sets_as_lists.append([int(x) for x in line.split()])

    sol = solve_exact_cover(U, sets_as_lists)
    out = sys.stdout
    if sol is None:
        out.write("NO SOLUTION\n")
    else:
        out.write("SOLUTION\n")
        if sol:
            out.write(" ".join(map(str, sol)) + "\n")
        else:
            out.write("\n")


if __name__ == "__main__":
    main()
