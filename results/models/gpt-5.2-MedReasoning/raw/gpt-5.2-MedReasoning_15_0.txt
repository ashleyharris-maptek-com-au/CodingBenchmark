use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Vec2 {
    x: f64,
    y: f64,
}

impl Vec2 {
    fn sub(self, other: Vec2) -> Vec2 {
        Vec2 {
            x: self.x - other.x,
            y: self.y - other.y,
        }
    }
    fn cross(self, other: Vec2) -> f64 {
        self.x * other.y - self.y * other.x
    }
    fn len(self) -> f64 {
        (self.x * self.x + self.y * self.y).sqrt()
    }
}

#[derive(Clone, Copy, Debug)]
struct Vec3 {
    x: f64,
    y: f64,
    z: f64,
}

fn rotate_z(v: Vec3, ang: f64) -> Vec3 {
    let (s, c) = ang.sin_cos();
    Vec3 {
        x: c * v.x - s * v.y,
        y: s * v.x + c * v.y,
        z: v.z,
    }
}

fn polygon_area(points: &[Vec2]) -> f64 {
    if points.len() < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..points.len() {
        let j = (i + 1) % points.len();
        a += points[i].x * points[j].y - points[j].x * points[i].y;
    }
    0.5 * a
}

fn convex_hull(mut pts: Vec<Vec2>) -> Vec<Vec2> {
    // Andrew monotone chain; handles duplicates.
    pts.sort_by(|a, b| {
        let ox = a.x.partial_cmp(&b.x).unwrap_or(std::cmp::Ordering::Equal);
        if ox != std::cmp::Ordering::Equal {
            return ox;
        }
        a.y.partial_cmp(&b.y).unwrap_or(std::cmp::Ordering::Equal)
    });
    pts.dedup_by(|a, b| (a.x == b.x) && (a.y == b.y));
    if pts.len() <= 1 {
        return pts;
    }

    fn build_half(pts: &[Vec2]) -> Vec<Vec2> {
        let mut half: Vec<Vec2> = Vec::new();
        for &p in pts {
            while half.len() >= 2 {
                let b = half[half.len() - 1];
                let a = half[half.len() - 2];
                let ab = b.sub(a);
                let bp = p.sub(b);
                if ab.cross(bp) <= 0.0 {
                    half.pop();
                } else {
                    break;
                }
            }
            half.push(p);
        }
        half
    }

    let lower = build_half(&pts);
    let mut upper_pts = pts;
    upper_pts.reverse();
    let upper = build_half(&upper_pts);

    let mut hull = Vec::new();
    for p in lower.iter().take(lower.len().saturating_sub(1)) {
        hull.push(*p);
    }
    for p in upper.iter().take(upper.len().saturating_sub(1)) {
        hull.push(*p);
    }
    hull
}

fn centered_inradius_ccw(hull_ccw: &[Vec2]) -> f64 {
    // hull must be convex, CCW, and contain the origin.
    if hull_ccw.len() < 3 {
        return 0.0;
    }
    let mut best = f64::INFINITY;
    for i in 0..hull_ccw.len() {
        let a = hull_ccw[i];
        let b = hull_ccw[(i + 1) % hull_ccw.len()];
        let e = b.sub(a);
        // Left normal points inward for CCW polygon.
        let n = Vec2 { x: e.y, y: -e.x };
        let denom = n.len();
        if denom == 0.0 {
            continue;
        }
        // Distance from origin to supporting line in direction n:
        // dot(n, x - a) >= 0 for inside; for x=0 => -dot(n,a) >= 0.
        let d = (-n.x * a.x - n.y * a.y) / denom;
        if d < best {
            best = d;
        }
    }
    if best.is_finite() && best > 0.0 {
        best
    } else {
        0.0
    }
}

fn main() {
    // Read all input.
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let nv_opt = it.next();
    if nv_opt.is_none() {
        println!("0");
        return;
    }
    let nv: usize = nv_opt.unwrap().parse().unwrap_or(0);
    if nv < 3 {
        println!("0");
        return;
    }

    let mut poly: Vec<Vec2> = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        poly.push(Vec2 { x, y });
    }

    let sx: f64 = it.next().unwrap().parse().unwrap();
    let sy: f64 = it.next().unwrap().parse().unwrap();
    let mut sz: f64 = it.next().unwrap().parse().unwrap();
    if sz.abs() < 1e-12 {
        sz = if sz.is_sign_negative() { -1e-12 } else { 1e-12 };
    }
    let kx = sx / sz;
    let ky = sy / sz;

    // Reference tetra vertices (object space).
    let base: [Vec3; 4] = [
        Vec3 {
            x: 1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: -1.0,
            y: 0.0,
            z: -0.707,
        },
        Vec3 {
            x: 0.0,
            y: 1.0,
            z: 0.707,
        },
        Vec3 {
            x: 0.0,
            y: -1.0,
            z: 0.707,
        },
    ];

    // Choose a yaw (rotation around Z) that maximizes the centered inradius of the 2D shadow
    // (i.e., largest guaranteed disk contained in each tetra shadow).
    let mut best_ang = 0.0;
    let mut best_r = 0.0;
    let samples = 16usize;
    for i in 0..samples {
        let ang = (i as f64) * PI / (samples as f64);
        let mut pts2: Vec<Vec2> = Vec::with_capacity(4);
        for &v in &base {
            let r = rotate_z(v, ang);
            pts2.push(Vec2 {
                x: r.x - r.z * kx,
                y: r.y - r.z * ky,
            });
        }
        let mut hull = convex_hull(pts2);
        if hull.len() < 3 {
            continue;
        }
        if polygon_area(&hull) < 0.0 {
            hull.reverse();
        }
        let r = centered_inradius_ccw(&hull);
        if r > best_r {
            best_r = r;
            best_ang = ang;
        }
    }

    let scale = 10.0_f64;
    let disk_r = (best_r * scale).max(1e-6);
    let step = disk_r * 2f64.sqrt(); // square grid covering condition

    // Bounding box of polygon.
    let mut minx = poly[0].x;
    let mut maxx = poly[0].x;
    let mut miny = poly[0].y;
    let mut maxy = poly[0].y;
    for p in &poly[1..] {
        if p.x < minx {
            minx = p.x;
        }
        if p.x > maxx {
            maxx = p.x;
        }
        if p.y < miny {
            miny = p.y;
        }
        if p.y > maxy {
            maxy = p.y;
        }
    }

    // Small-instance shortcut: if all vertices lie in the guaranteed disk, use one tetrahedron.
    let mut cx = 0.0;
    let mut cy = 0.0;
    for p in &poly {
        cx += p.x;
        cy += p.y;
    }
    cx /= nv as f64;
    cy /= nv as f64;
    let mut max_dist = 0.0;
    for p in &poly {
        let dx = p.x - cx;
        let dy = p.y - cy;
        let d = (dx * dx + dy * dy).sqrt();
        if d > max_dist {
            max_dist = d;
        }
    }
    if max_dist <= disk_r {
        // One tetrahedron centered so its shadow disk is centered at (cx,cy).
        let cz = 0.0;
        let px = cx + cz * kx;
        let py = cy + cz * ky;
        let qw = (best_ang * 0.5).cos();
        let qz = (best_ang * 0.5).sin();
        println!("1");
        println!(
            "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}",
            px, py, cz, qw, 0.0, 0.0, qz, scale
        );
        return;
    }

    // Grid cover for large instances: cover polygon by covering its bounding box with the
    // guaranteed inscribed disks, which are contained within each tetra shadow.
    let start_x = minx - step;
    let end_x = maxx + step;
    let start_y = miny - step;
    let end_y = maxy + step;

    let nx = ((end_x - start_x) / step).ceil() as i64 + 1;
    let ny = ((end_y - start_y) / step).ceil() as i64 + 1;
    let count_i128 = (nx as i128) * (ny as i128);
    let count = if count_i128 > (i64::MAX as i128) {
        i64::MAX
    } else {
        count_i128 as i64
    };

    let qw = (best_ang * 0.5).cos();
    let qz = (best_ang * 0.5).sin();

    // Z stacking for guaranteed non-intersection.
    let circ_r = 1.224744871391589_f64 * scale; // max distance from origin to a reference vertex
    let z_sep = (3.0 * circ_r + 1.0).max(50.0);

    // Build output efficiently.
    let mut out = String::new();
    out.push_str(&format!("{}\n", count));

    let mut idx: i64 = 0;
    for ix in 0..nx {
        let gx = start_x + (ix as f64) * step;
        for iy in 0..ny {
            let gy = start_y + (iy as f64) * step;
            let cz = (idx as f64) * z_sep;
            let px = gx + cz * kx;
            let py = gy + cz * ky;
            out.push_str(&format!(
                "{:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10} {:.10}\n",
                px, py, cz, qw, 0.0, 0.0, qz, scale
            ));
            idx += 1;
            if idx >= count {
                break;
            }
        }
        if idx >= count {
            break;
        }
    }

    print!("{out}");
}
