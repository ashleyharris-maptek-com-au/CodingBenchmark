use std::collections::VecDeque;
use std::io::{self, Read, Write};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        Self { state: seed.max(1) }
    }
    fn next_u64(&mut self) -> u64 {
        // xorshift64*
        let mut x = self.state;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.state = x;
        x.wrapping_mul(0x2545F4914F6CDD1D)
    }
    fn gen_range(&mut self, n: u64) -> u64 {
        if n <= 1 {
            return 0;
        }
        self.next_u64() % n
    }
    fn shuffle<T>(&mut self, a: &mut [T]) {
        for i in (1..a.len()).rev() {
            let j = self.gen_range((i + 1) as u64) as usize;
            a.swap(i, j);
        }
    }
}

fn idx(x: usize, y: usize, w: usize) -> usize {
    y * w + x
}

fn xy(id: usize, w: usize) -> (usize, usize) {
    (id % w, id / w)
}

fn neighbors(id: usize, w: usize, h: usize) -> [Option<usize>; 4] {
    let (x, y) = xy(id, w);
    let up = if y > 0 { Some(idx(x, y - 1, w)) } else { None };
    let down = if y + 1 < h { Some(idx(x, y + 1, w)) } else { None };
    let left = if x > 0 { Some(idx(x - 1, y, w)) } else { None };
    let right = if x + 1 < w { Some(idx(x + 1, y, w)) } else { None };
    [up, down, left, right]
}

fn write_snake_no_obstacles(mut out: impl Write, w: u64, h: u64, require_cycle: bool) -> io::Result<()> {
    let total: u128 = (w as u128) * (h as u128);
    if total == 0 {
        writeln!(out, "0")?;
        return Ok(());
    }

    // Start must be (0,0), so if empty grid not possible anyway.
    if require_cycle {
        // End must be adjacent to (0,0). With no obstacles, a construction exists for all
        // rectangles with at least one even dimension (and both dimensions >= 2), plus the 2-cell line.
        if (w == 1 && h == 2) || (w == 2 && h == 1) {
            writeln!(out, "2")?;
            writeln!(out, "0 0")?;
            if w == 2 {
                writeln!(out, "1 0")?;
            } else {
                writeln!(out, "0 1")?;
            }
            return Ok(());
        }
        if w < 2 || h < 2 {
            writeln!(out, "0")?;
            return Ok(());
        }
        if (w % 2 == 1) && (h % 2 == 1) {
            // Necessary condition for endpoints adjacent in bipartite grid: total even.
            writeln!(out, "0")?;
            return Ok(());
        }

        // Construction that ends at (1,0) (adjacent to start), works when h is even and w>=2.
        // If h is odd but w is even, we transpose (swap x/y) and use the same logic.
        let transpose = (h % 2 == 1) && (w % 2 == 0);
        let (ww, hh) = if transpose { (h, w) } else { (w, h) };
        // Now hh is even.
        if hh % 2 == 1 {
            writeln!(out, "0")?;
            return Ok(());
        }

        writeln!(out, "{total}")?;
        // Walk down first column, then snake the remaining columns bottom->top, ending at (1,0).
        for y in 0..hh {
            let (x, y) = if transpose { (y, 0) } else { (0, y) };
            writeln!(out, "{x} {y}")?;
        }
        // Now at (0, hh-1). Move into (1, hh-1), then snake rows from bottom to top over x=1..ww-1.
        for (k, y) in (0..hh).rev().enumerate() {
            let forward = k % 2 == 0;
            if forward {
                for x in 1..ww {
                    let (ox, oy) = if transpose { (y, x) } else { (x, y) };
                    writeln!(out, "{ox} {oy}")?;
                }
            } else {
                for x in (1..ww).rev() {
                    let (ox, oy) = if transpose { (y, x) } else { (x, y) };
                    writeln!(out, "{ox} {oy}")?;
                }
            }
        }
        return Ok(());
    }

    // Simple row-snake Hamiltonian path for obstacle-free grid.
    writeln!(out, "{total}")?;
    for y in 0..h {
        if y % 2 == 0 {
            for x in 0..w {
                writeln!(out, "{x} {y}")?;
            }
        } else {
            for x in (0..w).rev() {
                writeln!(out, "{x} {y}")?;
            }
        }
    }
    Ok(())
}

fn bfs_connected_unvisited(start: usize, w: usize, h: usize, blocked: &[bool], visited: &[bool], reserved_end: Option<usize>) -> usize {
    let mut q = VecDeque::new();
    let mut seen = vec![false; blocked.len()];
    q.push_back(start);
    seen[start] = true;
    let mut count = 0usize;
    while let Some(v) = q.pop_front() {
        // Count unvisited free vertices (including reserved_end if unvisited).
        if !blocked[v] && !visited[v] {
            count += 1;
        }
        for nb in neighbors(v, w, h).into_iter().flatten() {
            if seen[nb] {
                continue;
            }
            if blocked[nb] || visited[nb] {
                continue;
            }
            // reserved_end remains part of connectivity even if we are currently "avoiding" it.
            let _ = reserved_end;
            seen[nb] = true;
            q.push_back(nb);
        }
    }
    count
}

fn is_globally_connected_from_start(w: usize, h: usize, blocked: &[bool]) -> bool {
    let start = idx(0, 0, w);
    if blocked[start] {
        return false;
    }
    let mut q = VecDeque::new();
    let mut seen = vec![false; blocked.len()];
    q.push_back(start);
    seen[start] = true;
    let mut reached = 0usize;
    while let Some(v) = q.pop_front() {
        reached += 1;
        for nb in neighbors(v, w, h).into_iter().flatten() {
            if seen[nb] || blocked[nb] {
                continue;
            }
            seen[nb] = true;
            q.push_back(nb);
        }
    }
    let total_free = blocked.iter().filter(|&&b| !b).count();
    reached == total_free
}

fn parity_feasible(w: usize, h: usize, blocked: &[bool], require_cycle: bool) -> bool {
    let mut black = 0i64;
    let mut white = 0i64;
    for y in 0..h {
        for x in 0..w {
            let id = idx(x, y, w);
            if blocked[id] {
                continue;
            }
            if ((x + y) & 1) == 0 {
                black += 1;
            } else {
                white += 1;
            }
        }
    }
    if require_cycle {
        black == white
    } else {
        (black - white).abs() <= 1
    }
}

fn try_warnsdorff(
    w: usize,
    h: usize,
    blocked: &[bool],
    free_count: usize,
    reserved_end: Option<usize>,
    seed: u64,
) -> Option<Vec<usize>> {
    let start = idx(0, 0, w);
    if blocked[start] {
        return None;
    }
    if let Some(e) = reserved_end {
        if blocked[e] || e == start {
            return None;
        }
    }

    let mut rng = XorShift64::new(seed);
    let mut visited = vec![false; blocked.len()];
    visited[start] = true;
    let mut path = Vec::with_capacity(free_count);
    path.push(start);

    // Candidate list reused to avoid allocations.
    let mut candidates: Vec<usize> = Vec::with_capacity(4);

    while path.len() < free_count {
        let cur = *path.last().unwrap();
        let remaining = free_count - path.len();
        let must_take_reserved = reserved_end.is_some() && remaining == 1;
        candidates.clear();
        for nb in neighbors(cur, w, h).into_iter().flatten() {
            if blocked[nb] || visited[nb] {
                continue;
            }
            if let Some(e) = reserved_end {
                if !must_take_reserved && nb == e {
                    continue;
                }
            }
            candidates.push(nb);
        }
        if candidates.is_empty() {
            return None;
        }

        // Rank by Warnsdorff: fewest onward moves in the remaining graph.
        // Tie-break randomly for diversification.
        rng.shuffle(&mut candidates);
        candidates.sort_by(|&a, &b| {
            let da = onward_degree(a, w, h, blocked, &visited, reserved_end, remaining);
            let db = onward_degree(b, w, h, blocked, &visited, reserved_end, remaining);
            da.cmp(&db)
        });

        // Try candidates in order, but reject moves that disconnect the remaining unvisited set.
        // This adds cost, so only do it occasionally and when branching > 1.
        let do_connectivity_check = remaining <= 256 || (path.len() % 64 == 0);

        let mut moved = false;
        'cand: for (i, &next) in candidates.iter().enumerate() {
            let _ = i;
            // Tentatively move.
            visited[next] = true;

            if do_connectivity_check && remaining > 2 {
                // Pick a start vertex in the remaining unvisited set.
                let mut start_unvisited = None;
                if let Some(e) = reserved_end {
                    if !visited[e] && !blocked[e] {
                        start_unvisited = Some(e);
                    }
                }
                if start_unvisited.is_none() {
                    for nb2 in 0..blocked.len() {
                        if !blocked[nb2] && !visited[nb2] {
                            start_unvisited = Some(nb2);
                            break;
                        }
                    }
                }
                if let Some(su) = start_unvisited {
                    let reachable = bfs_connected_unvisited(su, w, h, blocked, &visited, reserved_end);
                    let unvisited_total = remaining - 1; // we just visited one
                    if reachable != unvisited_total {
                        visited[next] = false;
                        continue 'cand;
                    }
                }
            }

            path.push(next);
            moved = true;
            break 'cand;
        }

        if !moved {
            return None;
        }
    }

    Some(path)
}

fn onward_degree(
    v: usize,
    w: usize,
    h: usize,
    blocked: &[bool],
    visited: &[bool],
    reserved_end: Option<usize>,
    remaining: usize,
) -> u8 {
    let must_take_reserved = reserved_end.is_some() && remaining == 1;
    let mut deg: u8 = 0;
    for nb in neighbors(v, w, h).into_iter().flatten() {
        if blocked[nb] || visited[nb] {
            continue;
        }
        if let Some(e) = reserved_end {
            if !must_take_reserved && nb == e {
                continue;
            }
        }
        deg += 1;
    }
    deg
}

fn validate_path(w: usize, h: usize, blocked: &[bool], require_cycle: bool, path: &[usize]) -> bool {
    if path.is_empty() {
        return false;
    }
    if path[0] != idx(0, 0, w) {
        return false;
    }
    let mut seen = vec![false; blocked.len()];
    for &v in path {
        if v >= blocked.len() || blocked[v] || seen[v] {
            return false;
        }
        seen[v] = true;
    }
    let total_free = blocked.iter().filter(|&&b| !b).count();
    if path.len() != total_free {
        return false;
    }
    for win in path.windows(2) {
        let a = win[0];
        let b = win[1];
        let mut ok = false;
        for nb in neighbors(a, w, h).into_iter().flatten() {
            if nb == b {
                ok = true;
                break;
            }
        }
        if !ok {
            return false;
        }
    }
    if require_cycle {
        let end = *path.last().unwrap();
        let start = idx(0, 0, w);
        let mut adjacent = false;
        for nb in neighbors(end, w, h).into_iter().flatten() {
            if nb == start {
                adjacent = true;
                break;
            }
        }
        if !adjacent {
            return false;
        }
    }
    true
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let w_u64: u64 = match it.next() {
        Some(v) => v.parse().unwrap_or(0),
        None => return,
    };
    let h_u64: u64 = it.next().unwrap_or("0").parse().unwrap_or(0);
    let obs_u64: u64 = it.next().unwrap_or("0").parse().unwrap_or(0);
    let require_cycle = it.next().unwrap_or("0").parse::<u32>().unwrap_or(0) != 0;

    let mut out = io::BufWriter::new(io::stdout().lock());

    if obs_u64 == 0 {
        // Handle ludicrous sizes without building a grid.
        let _ = write_snake_no_obstacles(&mut out, w_u64, h_u64, require_cycle);
        return;
    }

    // For obstacle cases, we materialize the grid. Guard against overflow/absurd sizes.
    let total_cells_u128 = (w_u64 as u128) * (h_u64 as u128);
    if total_cells_u128 == 0 || total_cells_u128 > (usize::MAX as u128) {
        writeln!(out, "0").unwrap();
        return;
    }
    let w = w_u64 as usize;
    let h = h_u64 as usize;
    let total_cells = total_cells_u128 as usize;

    // Refuse truly enormous grids with obstacles (output itself would be infeasible).
    if total_cells > 5_000_000 {
        writeln!(out, "0").unwrap();
        return;
    }

    let mut blocked = vec![false; total_cells];
    for _ in 0..obs_u64 {
        let x: i64 = it.next().unwrap_or("0").parse().unwrap_or(-1);
        let y: i64 = it.next().unwrap_or("0").parse().unwrap_or(-1);
        if x < 0 || y < 0 {
            continue;
        }
        let (x, y) = (x as usize, y as usize);
        if x < w && y < h {
            blocked[idx(x, y, w)] = true;
        }
    }

    if blocked[idx(0, 0, w)] {
        writeln!(out, "0").unwrap();
        return;
    }

    if !is_globally_connected_from_start(w, h, &blocked) {
        writeln!(out, "0").unwrap();
        return;
    }

    if !parity_feasible(w, h, &blocked, require_cycle) {
        writeln!(out, "0").unwrap();
        return;
    }

    let free_count = blocked.iter().filter(|&&b| !b).count();
    if free_count == 1 {
        if require_cycle {
            // End adjacent to start is impossible with single vertex.
            writeln!(out, "0").unwrap();
            return;
        }
        writeln!(out, "1").unwrap();
        writeln!(out, "0 0").unwrap();
        return;
    }

    // If require_cycle, reserve one neighbor of start as the forced end.
    let start = idx(0, 0, w);
    let start_neighbors: Vec<usize> = neighbors(start, w, h).into_iter().flatten().filter(|&v| !blocked[v]).collect();
    if require_cycle && start_neighbors.is_empty() {
        writeln!(out, "0").unwrap();
        return;
    }

    // Multi-try randomized greedy with mild pruning.
    let mut best: Option<Vec<usize>> = None;
    let max_tries: usize = if free_count <= 200 { 20_000 } else if free_count <= 2500 { 5_000 } else { 1_000 };
    let mut seed = 0xC0FFEE_u64 ^ (w_u64.wrapping_shl(32) ^ h_u64) ^ obs_u64;

    let reserved_choices: Vec<Option<usize>> = if require_cycle {
        start_neighbors.iter().copied().map(Some).collect()
    } else {
        vec![None]
    };

    'outer: for attempt in 0..max_tries {
        let reserved_end = reserved_choices[attempt % reserved_choices.len()];
        let try_seed = seed.wrapping_add((attempt as u64).wrapping_mul(0x9E3779B97F4A7C15));
        if let Some(path) = try_warnsdorff(w, h, &blocked, free_count, reserved_end, try_seed) {
            if validate_path(w, h, &blocked, require_cycle, &path) {
                best = Some(path);
                break 'outer;
            }
        }
        // slight seed mixing
        seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1);
    }

    let Some(path) = best else {
        writeln!(out, "0").unwrap();
        return;
    };

    writeln!(out, "{}", path.len()).unwrap();
    for v in path {
        let (x, y) = xy(v, w);
        writeln!(out, "{x} {y}").unwrap();
    }
}
