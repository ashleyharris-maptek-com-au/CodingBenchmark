// HLSL pixel shader for Vulkan (glslangValidator HLSL mode)

struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0
{
    float2 uv = input.uv;

    // 6x6 polka-dot pattern in UV space
    float2 gridUV = uv * 6.0;
    float2 cellUV = frac(gridUV);
    float2 toCenter = cellUV - float2(0.5, 0.5);
    float dist = length(toCenter);

    float3 dotColor = float3(1.0, 0.8, 0.0);
    float3 bgColor  = float3(0.1, 0.1, 0.3);
    float3 baseColor = (dist <= 0.3) ? dotColor : bgColor;

    // Lambertian shading
    float3 N = normalize(input.normal);
    float3 L = normalize(lightPos.xyz - input.worldPos);
    float diffuse = max(dot(N, L), 0.0);
    float shade = 0.3 + 0.7 * diffuse;

    return float4(baseColor * shade, 1.0);
}
