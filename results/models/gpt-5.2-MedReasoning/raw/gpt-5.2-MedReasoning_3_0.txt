use std::collections::VecDeque;
use std::io::{self, Read};

// Fast input parser (whitespace-separated integers).
struct Scanner {
    buf: Vec<u8>,
    i: usize,
}
impl Scanner {
    fn new() -> Self {
        let mut buf = Vec::new();
        io::stdin().read_to_end(&mut buf).unwrap();
        Self { buf, i: 0 }
    }
    fn next_usize(&mut self) -> Option<usize> {
        while self.i < self.buf.len() && self.buf[self.i].is_ascii_whitespace() {
            self.i += 1;
        }
        if self.i >= self.buf.len() {
            return None;
        }
        let mut sign = 1i64;
        if self.buf[self.i] == b'-' {
            sign = -1;
            self.i += 1;
        }
        let mut v: i64 = 0;
        while self.i < self.buf.len() && !self.buf[self.i].is_ascii_whitespace() {
            v = v * 10 + (self.buf[self.i] - b'0') as i64;
            self.i += 1;
        }
        if sign < 0 {
            Some(0usize.saturating_sub((-v) as usize))
        } else {
            Some(v as usize)
        }
    }
}

// Deterministic RNG (SplitMix64).
#[derive(Clone)]
struct Rng64 {
    state: u64,
}
impl Rng64 {
    fn new(seed: u64) -> Self {
        Self { state: seed }
    }
    fn next_u64(&mut self) -> u64 {
        let mut z = self.state.wrapping_add(0x9E3779B97F4A7C15);
        self.state = z;
        z = (z ^ (z >> 30)).wrapping_mul(0xBF58476D1CE4E5B9);
        z = (z ^ (z >> 27)).wrapping_mul(0x94D049BB133111EB);
        z ^ (z >> 31)
    }
    fn next_f32(&mut self) -> f32 {
        // [0,1)
        let v = self.next_u64() >> 40;
        (v as f32) / ((1u64 << 24) as f32)
    }
    fn next_signed(&mut self) -> f32 {
        self.next_f32() * 2.0 - 1.0
    }
}

// CSR graph.
struct Graph {
    n: usize,
    off: Vec<u32>,
    adj: Vec<u32>,
    deg: Vec<u32>,
}
impl Graph {
    fn neighbors(&self, u: usize) -> &[u32] {
        let a = self.off[u] as usize;
        let b = self.off[u + 1] as usize;
        &self.adj[a..b]
    }
}

fn build_graph(n: usize, edges: &[(u32, u32)]) -> Graph {
    let mut deg = vec![0u32; n];
    for &(u, v) in edges {
        if u as usize >= n || v as usize >= n {
            continue;
        }
        if u == v {
            // self-loop: ignore for adjacency but keep degree signal low-cost
            deg[u as usize] = deg[u as usize].saturating_add(1);
        } else {
            deg[u as usize] = deg[u as usize].saturating_add(1);
            deg[v as usize] = deg[v as usize].saturating_add(1);
        }
    }
    let mut off = vec![0u32; n + 1];
    let mut sum: u64 = 0;
    for i in 0..n {
        off[i] = sum as u32;
        sum += deg[i] as u64;
    }
    off[n] = sum as u32;
    let mut cur = off.clone();
    let mut adj = vec![0u32; sum as usize];
    for &(u, v) in edges {
        let uu = u as usize;
        let vv = v as usize;
        if uu >= n || vv >= n {
            continue;
        }
        if uu == vv {
            continue;
        }
        let pu = cur[uu] as usize;
        adj[pu] = v;
        cur[uu] += 1;
        let pv = cur[vv] as usize;
        adj[pv] = u;
        cur[vv] += 1;
    }
    Graph { n, off, adj, deg }
}

fn component_nodes(graph: &Graph) -> (Vec<u32>, Vec<Vec<u32>>) {
    // Returns comp_id per node and node lists per component.
    let n = graph.n;
    let mut comp = vec![u32::MAX; n];
    let mut comps: Vec<Vec<u32>> = Vec::new();
    let mut q = VecDeque::<u32>::new();
    for s in 0..n {
        if comp[s] != u32::MAX {
            continue;
        }
        let cid = comps.len() as u32;
        comp[s] = cid;
        q.push_back(s as u32);
        let mut nodes = Vec::<u32>::new();
        while let Some(u) = q.pop_front() {
            nodes.push(u);
            for &v in graph.neighbors(u as usize) {
                let vi = v as usize;
                if comp[vi] == u32::MAX {
                    comp[vi] = cid;
                    q.push_back(v);
                }
            }
        }
        comps.push(nodes);
    }
    (comp, comps)
}

fn bfs_layers(graph: &Graph, nodes: &[u32], sort_small: bool) -> (Vec<u32>, Vec<u32>, Vec<u32>) {
    // For a component, returns:
    // - order: BFS order (node ids)
    // - depth[node_index_in_nodes]: depth
    // - parent[node_index_in_nodes]: parent (u32::MAX for root)
    let mut idx = vec![u32::MAX; graph.n];
    for (i, &u) in nodes.iter().enumerate() {
        idx[u as usize] = i as u32;
    }
    let mut start = nodes[0];
    // pick a high-degree start for stability
    let mut best_deg = graph.deg[start as usize];
    for &u in nodes.iter().take(64).chain(nodes.iter().rev().take(64)) {
        let d = graph.deg[u as usize];
        if d > best_deg {
            best_deg = d;
            start = u;
        }
    }
    let mut q = VecDeque::<u32>::new();
    let mut seen = vec![false; nodes.len()];
    let mut depth = vec![u32::MAX; nodes.len()];
    let mut parent = vec![u32::MAX; nodes.len()];
    let si = idx[start as usize] as usize;
    seen[si] = true;
    depth[si] = 0;
    q.push_back(start);
    let mut order = Vec::<u32>::with_capacity(nodes.len());

    let mut tmp_neighbors: Vec<u32> = Vec::new();
    while let Some(u) = q.pop_front() {
        order.push(u);
        let ui = idx[u as usize] as usize;
        if sort_small {
            tmp_neighbors.clear();
            tmp_neighbors.extend_from_slice(graph.neighbors(u as usize));
            tmp_neighbors.sort_unstable_by(|&a, &b| {
                let da = graph.deg[a as usize];
                let db = graph.deg[b as usize];
                da.cmp(&db)
            });
            for &v in &tmp_neighbors {
                let vi = idx[v as usize];
                if vi == u32::MAX {
                    continue;
                }
                let vii = vi as usize;
                if !seen[vii] {
                    seen[vii] = true;
                    depth[vii] = depth[ui].saturating_add(1);
                    parent[vii] = u;
                    q.push_back(v);
                }
            }
        } else {
            for &v in graph.neighbors(u as usize) {
                let vi = idx[v as usize];
                if vi == u32::MAX {
                    continue;
                }
                let vii = vi as usize;
                if !seen[vii] {
                    seen[vii] = true;
                    depth[vii] = depth[ui].saturating_add(1);
                    parent[vii] = u;
                    q.push_back(v);
                }
            }
        }
    }
    // disconnected within "component" shouldn't happen, but guard by appending unseen nodes.
    for &u in nodes {
        let ui = idx[u as usize] as usize;
        if !seen[ui] {
            seen[ui] = true;
            depth[ui] = 0;
            parent[ui] = u32::MAX;
            order.push(u);
        }
    }
    (order, depth, parent)
}

fn initial_layout_layers(graph: &Graph, nodes: &[u32]) -> (Vec<f32>, Vec<f32>, f32, f32) {
    // Produce an initial (x,y) for this component using BFS layers.
    let sort_small = nodes.len() <= 50_000;
    let (order, depth, _parent) = bfs_layers(graph, nodes, sort_small && nodes.len() <= 10_000);

    let mut idx = vec![u32::MAX; graph.n];
    for (i, &u) in nodes.iter().enumerate() {
        idx[u as usize] = i as u32;
    }

    let mut maxd = 0u32;
    for &d in &depth {
        if d != u32::MAX && d > maxd {
            maxd = d;
        }
    }
    let layers = (maxd as usize).saturating_add(1).max(1);
    let mut layer_counts = vec![0u32; layers];
    for &u in &order {
        let ui = idx[u as usize] as usize;
        let d = depth[ui];
        if d != u32::MAX {
            layer_counts[d as usize] += 1;
        }
    }
    let mut layer_pos = vec![0u32; layers];
    let mut maxw = 1u32;
    for &c in &layer_counts {
        if c > maxw {
            maxw = c;
        }
    }
    let mut x = vec![0f32; nodes.len()];
    let mut y = vec![0f32; nodes.len()];
    for &u in &order {
        let ui = idx[u as usize] as usize;
        let d = depth[ui];
        let w = layer_counts[d as usize].max(1);
        let p = layer_pos[d as usize];
        layer_pos[d as usize] += 1;
        // Center each layer horizontally; normalize widths to maxw for consistent scaling.
        let fx = (p as f32 + 0.5) / (w as f32) - 0.5;
        x[ui] = fx * (w as f32 / maxw as f32);
        y[ui] = -(d as f32);
    }
    // Normalize to roughly unit scale.
    let mut minx = x[0];
    let mut maxx = x[0];
    let mut miny = y[0];
    let mut maxy = y[0];
    for i in 0..x.len() {
        minx = minx.min(x[i]);
        maxx = maxx.max(x[i]);
        miny = miny.min(y[i]);
        maxy = maxy.max(y[i]);
    }
    let w = (maxx - minx).max(1e-3);
    let h = (maxy - miny).max(1e-3);
    for i in 0..x.len() {
        x[i] = (x[i] - (minx + maxx) * 0.5) / w;
        y[i] = (y[i] - (miny + maxy) * 0.5) / h;
    }
    (x, y, w, h)
}

fn force_refine(
    graph: &Graph,
    nodes: &[u32],
    x: &mut [f32],
    y: &mut [f32],
    iters: usize,
    rng: &mut Rng64,
) {
    // Grid-approximated repulsion + edge springs; designed for n <= ~4000.
    let n = nodes.len();
    if n <= 1 {
        return;
    }
    let area = 4.0f32; // roughly [-1,1]^2
    let k = (area / n as f32).sqrt().max(1e-3);
    let mut disp_x = vec![0f32; n];
    let mut disp_y = vec![0f32; n];

    // Component-local index mapping.
    let mut idx = vec![u32::MAX; graph.n];
    for (i, &u) in nodes.iter().enumerate() {
        idx[u as usize] = i as u32;
    }

    let mut temperature = 0.2f32;
    for _ in 0..iters {
        disp_x.fill(0.0);
        disp_y.fill(0.0);

        // Build uniform grid of points for approximate repulsion.
        let cell = (k * 2.0).max(1e-3);
        let gx0 = -1.2f32;
        let gy0 = -1.2f32;
        let gw = 2.4f32;
        let gh = 2.4f32;
        let cols = ((gw / cell).ceil() as i32).max(3);
        let rows = ((gh / cell).ceil() as i32).max(3);
        let mut head = vec![-1i32; (cols * rows) as usize];
        let mut next = vec![-1i32; n];
        for i in 0..n {
            let mut cx = ((x[i] - gx0) / cell).floor() as i32;
            let mut cy = ((y[i] - gy0) / cell).floor() as i32;
            cx = cx.clamp(0, cols - 1);
            cy = cy.clamp(0, rows - 1);
            let c = (cy * cols + cx) as usize;
            next[i] = head[c];
            head[c] = i as i32;
        }

        // Repulsion: check same and neighbor cells.
        for cy in 0..rows {
            for cx in 0..cols {
                let c = (cy * cols + cx) as usize;
                let mut a = head[c];
                while a != -1 {
                    let ai = a as usize;
                    for ny in (cy - 1).max(0)..=(cy + 1).min(rows - 1) {
                        for nx in (cx - 1).max(0)..=(cx + 1).min(cols - 1) {
                            let cc = (ny * cols + nx) as usize;
                            let mut b = head[cc];
                            while b != -1 {
                                let bi = b as usize;
                                if bi > ai {
                                    let dx = x[ai] - x[bi];
                                    let dy = y[ai] - y[bi];
                                    let dist2 = dx * dx + dy * dy + 1e-6;
                                    let inv = 1.0 / dist2.sqrt();
                                    let force = (k * k) * inv;
                                    let fx = dx * inv * force;
                                    let fy = dy * inv * force;
                                    disp_x[ai] += fx;
                                    disp_y[ai] += fy;
                                    disp_x[bi] -= fx;
                                    disp_y[bi] -= fy;
                                }
                                b = next[bi];
                            }
                        }
                    }
                    a = next[ai];
                }
            }
        }

        // Springs along edges (only intra-component).
        for (i, &u) in nodes.iter().enumerate() {
            for &v in graph.neighbors(u as usize) {
                let vi = idx[v as usize];
                if vi == u32::MAX {
                    continue;
                }
                let j = vi as usize;
                if j <= i {
                    continue;
                }
                let dx = x[i] - x[j];
                let dy = y[i] - y[j];
                let dist = (dx * dx + dy * dy + 1e-6).sqrt();
                let force = (dist * dist) / k;
                let fx = dx / dist * force;
                let fy = dy / dist * force;
                disp_x[i] -= fx;
                disp_y[i] -= fy;
                disp_x[j] += fx;
                disp_y[j] += fy;
            }
        }

        // Apply displacements with cooling.
        for i in 0..n {
            let mut dx = disp_x[i];
            let mut dy = disp_y[i];
            let d = (dx * dx + dy * dy).sqrt();
            if d > 1e-6 {
                let s = (temperature / d).min(1.0);
                dx *= s;
                dy *= s;
            }
            x[i] = (x[i] + dx).clamp(-1.0, 1.0);
            y[i] = (y[i] + dy).clamp(-1.0, 1.0);
        }

        // Tiny jitter to break perfect symmetry / collisions.
        for _ in 0..n.min(64) {
            let j = (rng.next_u64() as usize) % n;
            x[j] = (x[j] + rng.next_signed() * 1e-3).clamp(-1.0, 1.0);
            y[j] = (y[j] + rng.next_signed() * 1e-3).clamp(-1.0, 1.0);
        }

        temperature *= 0.95;
    }
}

fn edge_relax(
    graph: &Graph,
    nodes: &[u32],
    x: &mut [f32],
    y: &mut [f32],
    iters: usize,
    rng: &mut Rng64,
) {
    // Fast edge-only smoothing + gravity, for medium components.
    let n = nodes.len();
    if n <= 1 {
        return;
    }
    let mut idx = vec![u32::MAX; graph.n];
    for (i, &u) in nodes.iter().enumerate() {
        idx[u as usize] = i as u32;
    }

    let mut dx = vec![0f32; n];
    let mut dy = vec![0f32; n];
    let step0 = 0.08f32;
    for t in 0..iters {
        dx.fill(0.0);
        dy.fill(0.0);
        let step = step0 * (1.0 - (t as f32 / iters as f32) * 0.7);
        for (i, &u) in nodes.iter().enumerate() {
            let xi = x[i];
            let yi = y[i];
            for &v in graph.neighbors(u as usize) {
                let vi = idx[v as usize];
                if vi == u32::MAX {
                    continue;
                }
                let j = vi as usize;
                if j <= i {
                    continue;
                }
                let xj = x[j];
                let yj = y[j];
                let vx = xj - xi;
                let vy = yj - yi;
                let dist2 = vx * vx + vy * vy + 1e-6;
                let inv = 1.0 / dist2.sqrt();
                // Pull endpoints together to a target length ~ 0.05..0.15 depending on n.
                let target = (0.12f32 * (10_000f32 / n as f32).sqrt()).clamp(0.03, 0.15);
                let err = dist2.sqrt() - target;
                let fx = vx * inv * err;
                let fy = vy * inv * err;
                dx[i] += fx;
                dy[i] += fy;
                dx[j] -= fx;
                dy[j] -= fy;
            }
        }
        for i in 0..n {
            // Gravity towards origin to keep bounded.
            dx[i] += -x[i] * 0.01;
            dy[i] += -y[i] * 0.01;
            x[i] = (x[i] + dx[i] * step).clamp(-1.0, 1.0);
            y[i] = (y[i] + dy[i] * step).clamp(-1.0, 1.0);
        }
        // occasional tiny jitter to escape local overlaps
        if t % 25 == 0 {
            for _ in 0..(n / 2000).max(1).min(32) {
                let j = (rng.next_u64() as usize) % n;
                x[j] = (x[j] + rng.next_signed() * 5e-3).clamp(-1.0, 1.0);
                y[j] = (y[j] + rng.next_signed() * 5e-3).clamp(-1.0, 1.0);
            }
        }
    }
}

fn main() {
    let mut sc = Scanner::new();
    let n = match sc.next_usize() {
        Some(v) => v,
        None => return,
    };
    let m = sc.next_usize().unwrap_or(0);

    let mut edges: Vec<(u32, u32)> = Vec::with_capacity(m);
    for _ in 0..m {
        let u = sc.next_usize().unwrap_or(0) as u32;
        let v = sc.next_usize().unwrap_or(0) as u32;
        edges.push((u, v));
    }

    let graph = build_graph(n, &edges);
    let (_comp_id, mut comps) = component_nodes(&graph);

    // Sort components by size descending for nicer packing.
    comps.sort_unstable_by(|a, b| b.len().cmp(&a.len()));

    let mut out_x = vec![0f32; n];
    let mut out_y = vec![0f32; n];
    let mut rng = Rng64::new((n as u64).wrapping_mul(0xD1B54A32D192ED03) ^ (m as u64));

    // Pack components on a coarse grid.
    let mut cursor_x = 0f32;
    let mut cursor_y = 0f32;
    let mut row_h = 0f32;
    let canvas_w = 10.0f32; // arbitrary; scaled by component sizes
    for nodes in comps {
        let cn = nodes.len();
        if cn == 0 {
            continue;
        }
        let (mut x, mut y, _w0, _h0) = initial_layout_layers(&graph, &nodes);

        // Scale based on component size so big components get more space.
        let base = (cn as f32).sqrt().max(1.0);
        let scale = (base.ln_1p() * 1.3 + 1.0).min(12.0);

        // Refinement strategy by component size.
        if cn <= 4000 {
            let iters = if cn <= 200 { 160 } else if cn <= 800 { 120 } else { 80 };
            force_refine(&graph, &nodes, &mut x, &mut y, iters, &mut rng);
        } else if cn <= 60_000 {
            let iters = if cn <= 12_000 { 80 } else { 40 };
            edge_relax(&graph, &nodes, &mut x, &mut y, iters, &mut rng);
        } else {
            // Very large: no refinement; BFS layers only (already computed).
        }

        // Compute component bounds for packing.
        let mut minx = x[0];
        let mut maxx = x[0];
        let mut miny = y[0];
        let mut maxy = y[0];
        for i in 0..cn {
            minx = minx.min(x[i]);
            maxx = maxx.max(x[i]);
            miny = miny.min(y[i]);
            maxy = maxy.max(y[i]);
        }
        let cw = (maxx - minx).max(1e-3) * scale;
        let ch = (maxy - miny).max(1e-3) * scale;

        if cursor_x + cw > canvas_w {
            cursor_x = 0.0;
            cursor_y -= row_h + 1.0;
            row_h = 0.0;
        }
        row_h = row_h.max(ch);

        let ox = cursor_x - minx * scale;
        let oy = cursor_y - miny * scale;

        for (i, &u) in nodes.iter().enumerate() {
            out_x[u as usize] = x[i] * scale + ox;
            out_y[u as usize] = y[i] * scale + oy;
        }

        cursor_x += cw + 1.0;
    }

    // Output N lines (no prefix).
    let mut s = String::with_capacity(n * 24);
    for i in 0..n {
        // Format as floats; keep concise for speed.
        s.push_str(&format!("{:.6} {:.6}\n", out_x[i], out_y[i]));
    }
    print!("{s}");
}
