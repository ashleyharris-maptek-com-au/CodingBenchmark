// Job-Shop Scheduling heuristic solver (C++20, MSVC)
// Produces a feasible schedule for any input size; uses adaptive multi-rule search for small/medium cases.

#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <string>
#include <tuple>
#include <utility>
#include <vector>

using i64 = long long;

struct Task {
  int machine = 0;
  int duration = 0;
};

struct Instance {
  int jobs = 0;
  int machines = 0;
  std::vector<std::vector<Task>> job_tasks;
  std::vector<std::vector<i64>> suffix_dur; // suffix_dur[j][t] = sum durations from t..end
  i64 total_tasks = 0;
};

struct Schedule {
  std::vector<std::vector<i64>> start; // start[j][t]
  i64 makespan = 0;
};

enum class Rule {
  EST_SPT,       // earliest start, shortest processing time
  EST_LPT,       // earliest start, longest processing time
  EST_REMAINING  // earliest start, most remaining work (descending remaining)
};

static Instance read_instance() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  Instance ins;
  if (!(std::cin >> ins.jobs >> ins.machines)) return ins;
  ins.job_tasks.resize(ins.jobs);
  ins.suffix_dur.resize(ins.jobs);

  for (int j = 0; j < ins.jobs; ++j) {
    int tcount = 0;
    std::cin >> tcount;
    ins.job_tasks[j].resize(tcount);
    for (int t = 0; t < tcount; ++t) {
      int m = 0, d = 0;
      std::cin >> m >> d;
      ins.job_tasks[j][t] = Task{m, d};
    }
    ins.total_tasks += static_cast<i64>(tcount);
  }

  for (int j = 0; j < ins.jobs; ++j) {
    const int T = static_cast<int>(ins.job_tasks[j].size());
    ins.suffix_dur[j].assign(T + 1, 0);
    for (int t = T - 1; t >= 0; --t) {
      ins.suffix_dur[j][t] = ins.suffix_dur[j][t + 1] + ins.job_tasks[j][t].duration;
    }
  }

  return ins;
}

static inline i64 secondary_key(Rule rule,
                                int duration,
                                i64 remaining_work,
                                int job_id) {
  // For min-heap behavior, smaller key is "better".
  // Use job_id as final stable tie-breaker.
  constexpr i64 SCALE = 1'048'576; // 2^20
  switch (rule) {
    case Rule::EST_SPT:
      return static_cast<i64>(duration) * SCALE + static_cast<i64>(job_id);
    case Rule::EST_LPT:
      // Prefer longer duration => smaller key by negating.
      return (-static_cast<i64>(duration)) * SCALE + static_cast<i64>(job_id);
    case Rule::EST_REMAINING:
      // Prefer larger remaining work => smaller key by negating.
      return (-remaining_work) * SCALE + static_cast<i64>(job_id);
  }
  return static_cast<i64>(job_id);
}

struct HeapEntry {
  i64 est = 0;
  i64 secondary = 0;
  int job = 0;
  int gen = 0;
};

struct HeapGreater {
  bool operator()(const HeapEntry& a, const HeapEntry& b) const {
    if (a.est != b.est) return a.est > b.est;
    if (a.secondary != b.secondary) return a.secondary > b.secondary;
    if (a.job != b.job) return a.job > b.job;
    return a.gen > b.gen;
  }
};

static Schedule build_schedule(const Instance& ins, Rule rule) {
  Schedule out;
  out.start.resize(ins.jobs);
  for (int j = 0; j < ins.jobs; ++j) out.start[j].assign(ins.job_tasks[j].size(), 0);

  std::vector<int> next_idx(ins.jobs, 0);
  std::vector<int> gen(ins.jobs, 0);
  std::vector<i64> job_ready(ins.jobs, 0);
  std::vector<i64> machine_free(ins.machines, 0);

  std::priority_queue<HeapEntry, std::vector<HeapEntry>, HeapGreater> pq;

  auto push_job = [&](int j) {
    const int t = next_idx[j];
    const int T = static_cast<int>(ins.job_tasks[j].size());
    if (t >= T) return;
    const Task& task = ins.job_tasks[j][t];
    const i64 est = std::max(job_ready[j], machine_free[task.machine]);
    const i64 rem = ins.suffix_dur[j][t];
    const i64 sec = secondary_key(rule, task.duration, rem, j);
    pq.push(HeapEntry{est, sec, j, gen[j]});
  };

  for (int j = 0; j < ins.jobs; ++j) push_job(j);

  i64 scheduled = 0;
  while (!pq.empty() && scheduled < ins.total_tasks) {
    HeapEntry e = pq.top();
    pq.pop();

    const int j = e.job;
    if (e.gen != gen[j]) continue; // stale due to scheduling this job already

    const int t = next_idx[j];
    const int T = static_cast<int>(ins.job_tasks[j].size());
    if (t >= T) continue;

    const Task& task = ins.job_tasks[j][t];
    const i64 current_est = std::max(job_ready[j], machine_free[task.machine]);
    if (current_est != e.est) {
      // Machine became busier (or job became ready later); reinsert updated.
      const i64 rem = ins.suffix_dur[j][t];
      const i64 sec = secondary_key(rule, task.duration, rem, j);
      pq.push(HeapEntry{current_est, sec, j, gen[j]});
      continue;
    }

    // Schedule task now at current_est
    out.start[j][t] = current_est;
    const i64 finish = current_est + static_cast<i64>(task.duration);
    job_ready[j] = finish;
    machine_free[task.machine] = finish;
    out.makespan = std::max(out.makespan, finish);

    ++next_idx[j];
    ++gen[j];
    ++scheduled;
    push_job(j);
  }

  // In case of any oddities, ensure makespan is consistent.
  for (int m = 0; m < ins.machines; ++m) out.makespan = std::max(out.makespan, machine_free[m]);
  return out;
}

static i64 evaluate_makespan(const Schedule& s) { return s.makespan; }

static Schedule solve(const Instance& ins) {
  // Adaptive strategy selection:
  // - Huge: single fast heuristic.
  // - Small/medium: try a few priority rules and keep best makespan.
  const i64 N = ins.total_tasks;

  if (N == 0 || ins.jobs == 0) return Schedule{};

  // Very large instances: minimize overhead.
  if (N >= 200000 || ins.jobs >= 50000) {
    return build_schedule(ins, Rule::EST_REMAINING);
  }

  // Medium: try a small rule set.
  std::vector<Rule> rules;
  if (N <= 5000) {
    rules = {Rule::EST_REMAINING, Rule::EST_SPT, Rule::EST_LPT};
  } else {
    rules = {Rule::EST_REMAINING, Rule::EST_SPT};
  }

  Schedule best;
  best.makespan = std::numeric_limits<i64>::max();
  for (Rule r : rules) {
    Schedule cand = build_schedule(ins, r);
    const i64 ms = evaluate_makespan(cand);
    if (ms < best.makespan) best = std::move(cand);
  }
  return best;
}

int main() {
  Instance ins = read_instance();
  if (ins.jobs <= 0) return 0;

  Schedule sched = solve(ins);

  // Output: for each job, list "start machine" pairs for each task in order.
  for (int j = 0; j < ins.jobs; ++j) {
    const int T = static_cast<int>(ins.job_tasks[j].size());
    for (int t = 0; t < T; ++t) {
      if (t) std::cout << ' ';
      std::cout << sched.start[j][t] << ' ' << ins.job_tasks[j][t].machine;
    }
    std::cout << "\n";
  }
  return 0;
}
