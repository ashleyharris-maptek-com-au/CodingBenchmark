using System;
using System.Diagnostics;
using System.Globalization;
using System.IO;

public static class Program
{
    private sealed class FastScanner
    {
        private readonly Stream _stream;
        private readonly byte[] _buffer = new byte[1 << 16];
        private int _len, _ptr;

        public FastScanner(Stream stream) { _stream = stream; }

        private int ReadByte()
        {
            if (_ptr >= _len)
            {
                _len = _stream.Read(_buffer, 0, _buffer.Length);
                _ptr = 0;
                if (_len <= 0) return -1;
            }
            return _buffer[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = ReadByte(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-')
            {
                sign = -1;
                c = ReadByte();
            }
            int x = 0;
            while (c > 32)
            {
                x = x * 10 + (c - '0');
                c = ReadByte();
            }
            return x * sign;
        }
    }

    private static int Idx(int i, int j, int n) => i * n + j;

    private static long ComputeCost(int[] flow, int[] dist, int[] p, int n)
    {
        long cost = 0;
        for (int i = 0; i < n; i++)
        {
            int pi = p[i];
            int iOff = i * n;
            int piOff = pi * n;
            for (int j = 0; j < n; j++)
            {
                cost += (long)flow[iOff + j] * dist[piOff + p[j]];
            }
        }
        return cost;
    }

    private static long SwapDelta(int[] flow, int[] dist, int[] p, int a, int b, int n)
    {
        if (a == b) return 0;
        int pa = p[a];
        int pb = p[b];
        if (pa == pb) return 0;

        long delta = 0;

        // Self/cross terms.
        delta += (long)(flow[Idx(a, a, n)] - flow[Idx(b, b, n)]) * (dist[Idx(pb, pb, n)] - dist[Idx(pa, pa, n)]);
        delta += (long)(flow[Idx(a, b, n)] - flow[Idx(b, a, n)]) * (dist[Idx(pb, pa, n)] - dist[Idx(pa, pb, n)]);

        int aOff = a * n;
        int bOff = b * n;
        for (int k = 0; k < n; k++)
        {
            if (k == a || k == b) continue;
            int pk = p[k];

            // (F[a,k] - F[b,k]) * (D[pb,pk] - D[pa,pk])
            delta += (long)(flow[aOff + k] - flow[bOff + k]) * (dist[pb * n + pk] - dist[pa * n + pk]);

            // (F[k,a] - F[k,b]) * (D[pk,pb] - D[pk,pa])
            int kOff = k * n;
            delta += (long)(flow[kOff + a] - flow[kOff + b]) * (dist[pk * n + pb] - dist[pk * n + pa]);
        }

        return delta;
    }

    private static long SolveExactBruteforce(int[] flow, int[] dist, int n, out int[] bestP)
    {
        // Backtracking with incremental objective; n is small.
        bestP = new int[n];
        int[] p = new int[n];
        bool[] used = new bool[n];
        long best = long.MaxValue;

        long D(int li, int lj) => dist[li * n + lj];
        long F(int i, int j) => flow[i * n + j];

        void Dfs(int i, long partial)
        {
            if (partial >= best) return;
            if (i == n)
            {
                best = partial;
                Array.Copy(p, bestP, n);
                return;
            }

            for (int loc = 0; loc < n; loc++)
            {
                if (used[loc]) continue;

                long add = 0;
                for (int j = 0; j < i; j++)
                {
                    int locj = p[j];
                    add += F(i, j) * D(loc, locj) + F(j, i) * D(locj, loc);
                }
                add += F(i, i) * D(loc, loc);

                used[loc] = true;
                p[i] = loc;
                Dfs(i + 1, partial + add);
                used[loc] = false;
            }
        }

        Dfs(0, 0);
        return best;
    }

    private static void BuildHeuristicInitial(int[] flow, int[] dist, int n, int[] p)
    {
        long[] fScore = new long[n];
        long[] dScore = new long[n];

        for (int i = 0; i < n; i++)
        {
            int iOff = i * n;
            long s = 0;
            for (int j = 0; j < n; j++) s += flow[iOff + j] + flow[j * n + i];
            fScore[i] = s;
        }
        for (int l = 0; l < n; l++)
        {
            int lOff = l * n;
            long s = 0;
            for (int k = 0; k < n; k++) s += dist[lOff + k] + dist[k * n + l];
            dScore[l] = s;
        }

        int[] facilities = new int[n];
        int[] locations = new int[n];
        for (int i = 0; i < n; i++) { facilities[i] = i; locations[i] = i; }

        Array.Sort(facilities, (a, b) =>
        {
            int c = fScore[b].CompareTo(fScore[a]); // descending
            return c != 0 ? c : a.CompareTo(b);
        });
        Array.Sort(locations, (a, b) =>
        {
            int c = dScore[a].CompareTo(dScore[b]); // ascending (more central first)
            return c != 0 ? c : a.CompareTo(b);
        });

        for (int i = 0; i < n; i++) p[facilities[i]] = locations[i];
    }

    public static void Main()
    {
        CultureInfo.CurrentCulture = CultureInfo.InvariantCulture;
        var fs = new FastScanner(Console.OpenStandardInput());

        int n;
        try { n = fs.NextInt(); }
        catch { return; }
        if (n <= 0) return;

        int[] flow = new int[n * n];
        int[] dist = new int[n * n];

        for (int i = 0; i < n * n; i++) flow[i] = fs.NextInt();
        for (int i = 0; i < n * n; i++) dist[i] = fs.NextInt();

        int[] bestP;
        long bestCost;

        if (n <= 11)
        {
            bestCost = SolveExactBruteforce(flow, dist, n, out bestP);
        }
        else
        {
            int[] p = new int[n];
            BuildHeuristicInitial(flow, dist, n, p);
            long cost = ComputeCost(flow, dist, p, n);

            bestP = (int[])p.Clone();
            bestCost = cost;

            var sw = Stopwatch.StartNew();
            long timeLimitMs = 295_000; // ~4m55s
            var rng = new Random(1234567);

            // Estimate a reasonable initial temperature from random swap deltas.
            double t = 1.0;
            {
                double sumPos = 0;
                int cntPos = 0;
                int samples = Math.Min(200, n * 2);
                for (int s = 0; s < samples; s++)
                {
                    int a = rng.Next(n);
                    int b = rng.Next(n - 1);
                    if (b >= a) b++;
                    long d = SwapDelta(flow, dist, p, a, b, n);
                    if (d > 0)
                    {
                        sumPos += d;
                        cntPos++;
                    }
                }
                if (cntPos > 0) t = sumPos / cntPos;
                if (t < 1.0) t = 1.0;
            }

            double cooling = 0.99995;
            int iter = 0;
            int sinceBest = 0;

            while (sw.ElapsedMilliseconds < timeLimitMs)
            {
                int a = rng.Next(n);
                int b = rng.Next(n - 1);
                if (b >= a) b++;

                long delta = SwapDelta(flow, dist, p, a, b, n);
                bool accept = false;
                if (delta <= 0) accept = true;
                else
                {
                    double prob = Math.Exp(-(double)delta / t);
                    if (rng.NextDouble() < prob) accept = true;
                }

                if (accept)
                {
                    int tmp = p[a];
                    p[a] = p[b];
                    p[b] = tmp;
                    cost += delta;

                    if (cost < bestCost)
                    {
                        bestCost = cost;
                        Array.Copy(p, bestP, n);
                        sinceBest = 0;
                    }
                    else
                    {
                        sinceBest++;
                    }
                }
                else
                {
                    sinceBest++;
                }

                iter++;
                if ((iter & 1023) == 0)
                {
                    t *= cooling;
                    if (t < 1e-6) t = 1e-6;

                    // Simple reheat if stuck a long time.
                    if (sinceBest > 200_000)
                    {
                        t *= 2.0;
                        sinceBest = 0;
                    }
                }
            }
        }

        using var output = new StreamWriter(Console.OpenStandardOutput());
        output.WriteLine(bestCost.ToString(CultureInfo.InvariantCulture));
        for (int i = 0; i < n; i++)
        {
            if (i != 0) output.Write(' ');
            output.Write(bestP[i]);
        }
        output.WriteLine();
    }
}
