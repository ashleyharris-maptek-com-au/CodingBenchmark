use std::io::{self, Read, Write};

fn main() {
    // Read all input.
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut idx: usize = 0;

    fn next_u64(buf: &[u8], idx: &mut usize) -> Option<u64> {
        while *idx < buf.len() && buf[*idx].is_ascii_whitespace() {
            *idx += 1;
        }
        if *idx >= buf.len() {
            return None;
        }
        let mut v: u64 = 0;
        while *idx < buf.len() {
            let c = buf[*idx];
            if !c.is_ascii_digit() {
                break;
            }
            v = v * 10 + u64::from(c - b'0');
            *idx += 1;
        }
        Some(v)
    }

    let n = match next_u64(&input, &mut idx) {
        Some(v) => v as usize,
        None => return,
    };
    let m = match next_u64(&input, &mut idx) {
        Some(v) => v as usize,
        None => return,
    };

    // Greedy maximal matching (single pass). Endpoints of a maximal matching form a valid
    // vertex cover (2-approximation).
    let mut mate: Vec<i32> = vec![-1; n];
    let mut cover: Vec<u32> = Vec::new();
    cover.reserve(n.min(2_000_000));

    for _ in 0..m {
        let u = match next_u64(&input, &mut idx) {
            Some(v) => v as usize,
            None => break,
        };
        let v = match next_u64(&input, &mut idx) {
            Some(vv) => vv as usize,
            None => break,
        };

        if u >= n || v >= n {
            // Skip malformed edges rather than panicking; benchmark inputs should be valid.
            continue;
        }

        if u == v {
            if mate[u] < 0 {
                mate[u] = u as i32;
                cover.push(u as u32);
            }
            continue;
        }

        if mate[u] < 0 && mate[v] < 0 {
            mate[u] = v as i32;
            mate[v] = u as i32;
            cover.push(u as u32);
            cover.push(v as u32);
        }
    }

    let mut out = io::BufWriter::new(io::stdout().lock());
    writeln!(out, "{}", cover.len()).unwrap();
    if !cover.is_empty() {
        for (i, &x) in cover.iter().enumerate() {
            if i != 0 {
                out.write_all(b" ").unwrap();
            }
            write!(out, "{}", x).unwrap();
        }
    }
    out.write_all(b"\n").unwrap();
}
