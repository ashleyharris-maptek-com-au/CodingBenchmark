use std::cmp::Ordering;
use std::collections::HashMap;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Item {
    value: i64,
    index: usize,
}

fn parse_input() -> (usize, i64, Vec<Item>) {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(v) => v.parse().unwrap(),
        None => return (0, 0, Vec::new()),
    };
    let target: i64 = it.next().unwrap_or("0").parse().unwrap();
    let mut items = Vec::with_capacity(n);
    for idx in 0..n {
        if let Some(s) = it.next() {
            let v: i64 = s.parse().unwrap();
            items.push(Item { value: v, index: idx });
        } else {
            break;
        }
    }
    (items.len(), target, items)
}

fn output_yes(indices: &[usize]) {
    println!("YES");
    if indices.is_empty() {
        println!("[]");
        return;
    }
    print!("[");
    for (i, idx) in indices.iter().enumerate() {
        if i != 0 {
            print!(" ");
        }
        print!("{idx}");
    }
    println!("]");
}

fn output_no() {
    println!("NO");
}

fn meet_in_middle(items: &[Item], target: i64) -> Option<Vec<usize>> {
    let n = items.len();
    let mid = n / 2;
    let left = &items[..mid];
    let right = &items[mid..];

    let mut left_sums: Vec<(i64, u64)> = Vec::with_capacity(1usize << left.len());
    for mask in 0u64..(1u64 << left.len()) {
        let mut sum = 0i64;
        for (i, it) in left.iter().enumerate() {
            if (mask >> i) & 1 == 1 {
                sum = sum.wrapping_add(it.value);
            }
        }
        left_sums.push((sum, mask));
    }
    left_sums.sort_by(|a, b| a.0.cmp(&b.0));

    let mut right_sums: Vec<(i64, u64)> = Vec::with_capacity(1usize << right.len());
    for mask in 0u64..(1u64 << right.len()) {
        let mut sum = 0i64;
        for (i, it) in right.iter().enumerate() {
            if (mask >> i) & 1 == 1 {
                sum = sum.wrapping_add(it.value);
            }
        }
        right_sums.push((sum, mask));
    }
    right_sums.sort_by(|a, b| a.0.cmp(&b.0));

    let mut i = 0usize;
    let mut j = right_sums.len();
    while i < left_sums.len() && j > 0 {
        let ls = left_sums[i].0;
        let rs = right_sums[j - 1].0;
        let s = ls.wrapping_add(rs);
        match s.cmp(&target) {
            Ordering::Equal => {
                let lmask = left_sums[i].1;
                let rmask = right_sums[j - 1].1;
                let mut ans = Vec::new();
                for (k, it) in left.iter().enumerate() {
                    if (lmask >> k) & 1 == 1 {
                        ans.push(it.index);
                    }
                }
                for (k, it) in right.iter().enumerate() {
                    if (rmask >> k) & 1 == 1 {
                        ans.push(it.index);
                    }
                }
                ans.sort_unstable();
                return Some(ans);
            }
            Ordering::Less => i += 1,
            Ordering::Greater => j -= 1,
        }
    }
    None
}

fn bitset_subset_sum_nonneg(items: &[Item], target: usize) -> Option<Vec<usize>> {
    let word_len = (target + 64) / 64;
    let mut bits = vec![0u64; word_len];
    bits[0] = 1;

    let mut prev_idx = vec![-1i32; target + 1];
    let mut prev_sum = vec![-1i32; target + 1];

    for (cur_i, it) in items.iter().enumerate() {
        let v = it.value as usize;
        if v == 0 || v > target {
            continue;
        }
        let word_shift = v / 64;
        let bit_shift = (v % 64) as u32;
        let mut shifted = vec![0u64; word_len];
        for w in 0..word_len {
            let src = bits[w];
            if src == 0 {
                continue;
            }
            let dw = w + word_shift;
            if dw >= word_len {
                continue;
            }
            shifted[dw] |= src << bit_shift;
            if bit_shift != 0 && dw + 1 < word_len {
                shifted[dw + 1] |= src >> (64 - bit_shift);
            }
        }
        // Clear bits beyond target.
        let extra_bits = (word_len * 64).saturating_sub(target + 1);
        if extra_bits > 0 {
            let mask = (!0u64) >> extra_bits;
            shifted[word_len - 1] &= mask;
        }

        let mut newly = vec![0u64; word_len];
        for w in 0..word_len {
            newly[w] = shifted[w] & !bits[w];
            bits[w] |= shifted[w];
        }

        // Record parents for new sums.
        for w in 0..word_len {
            let mut x = newly[w];
            while x != 0 {
                let tz = x.trailing_zeros() as usize;
                let sum = w * 64 + tz;
                if sum <= target && prev_idx[sum] == -1 {
                    prev_idx[sum] = cur_i as i32;
                    prev_sum[sum] = (sum - v) as i32;
                }
                x &= x - 1;
            }
        }

        if (bits[target / 64] >> (target % 64)) & 1 == 1 {
            break;
        }
    }

    if ((bits[target / 64] >> (target % 64)) & 1) == 0 {
        return None;
    }

    // Reconstruct indices.
    let mut res = Vec::new();
    let mut s = target as i32;
    while s > 0 {
        let pi = prev_idx[s as usize];
        if pi < 0 {
            // Should not happen for reachable sum; treat as failure.
            return None;
        }
        let item = items[pi as usize];
        res.push(item.index);
        s = prev_sum[s as usize];
    }
    res.sort_unstable();
    Some(res)
}

fn bitset_subset_sum_bounded_counts(
    by_value: &[(i64, Vec<usize>)],
    target: usize,
) -> Option<Vec<usize>> {
    // Binary decomposition of multiplicities into 0/1 chunks.
    #[derive(Clone)]
    struct Chunk {
        weight: usize,
        value: i64,
        take: usize,
    }
    let mut chunks: Vec<Chunk> = Vec::new();
    for (value, idxs) in by_value {
        if *value <= 0 {
            continue;
        }
        let v = *value as usize;
        if v == 0 || v > target {
            continue;
        }
        let mut c = idxs.len();
        let mut k = 1usize;
        while c > 0 {
            let take = k.min(c);
            let weight = v.saturating_mul(take);
            if weight <= target {
                chunks.push(Chunk {
                    weight,
                    value: *value,
                    take,
                });
            }
            c -= take;
            k <<= 1;
        }
    }

    // Standard bitset DP on chunks (weights up to target).
    let word_len = (target + 64) / 64;
    let mut bits = vec![0u64; word_len];
    bits[0] = 1;
    let mut prev_chunk = vec![-1i32; target + 1];
    let mut prev_sum = vec![-1i32; target + 1];

    for (ci, ch) in chunks.iter().enumerate() {
        let w = ch.weight;
        if w == 0 || w > target {
            continue;
        }
        let word_shift = w / 64;
        let bit_shift = (w % 64) as u32;
        let mut shifted = vec![0u64; word_len];
        for ww in 0..word_len {
            let src = bits[ww];
            if src == 0 {
                continue;
            }
            let dw = ww + word_shift;
            if dw >= word_len {
                continue;
            }
            shifted[dw] |= src << bit_shift;
            if bit_shift != 0 && dw + 1 < word_len {
                shifted[dw + 1] |= src >> (64 - bit_shift);
            }
        }
        let extra_bits = (word_len * 64).saturating_sub(target + 1);
        if extra_bits > 0 {
            let mask = (!0u64) >> extra_bits;
            shifted[word_len - 1] &= mask;
        }

        let mut newly = vec![0u64; word_len];
        for ww in 0..word_len {
            newly[ww] = shifted[ww] & !bits[ww];
            bits[ww] |= shifted[ww];
        }

        for ww in 0..word_len {
            let mut x = newly[ww];
            while x != 0 {
                let tz = x.trailing_zeros() as usize;
                let sum = ww * 64 + tz;
                if sum <= target && prev_chunk[sum] == -1 {
                    prev_chunk[sum] = ci as i32;
                    prev_sum[sum] = (sum - w) as i32;
                }
                x &= x - 1;
            }
        }

        if (bits[target / 64] >> (target % 64)) & 1 == 1 {
            break;
        }
    }

    if ((bits[target / 64] >> (target % 64)) & 1) == 0 {
        return None;
    }

    // Reconstruct chunk selection -> counts per value.
    let mut take_counts: HashMap<i64, usize> = HashMap::new();
    let mut s = target as i32;
    while s > 0 {
        let cidx = prev_chunk[s as usize];
        if cidx < 0 {
            return None;
        }
        let ch = &chunks[cidx as usize];
        *take_counts.entry(ch.value).or_insert(0) += ch.take;
        s = prev_sum[s as usize];
    }

    let mut res = Vec::new();
    for (value, idxs) in by_value {
        if let Some(&need) = take_counts.get(value) {
            if need > idxs.len() {
                return None;
            }
            res.extend_from_slice(&idxs[..need]);
        }
    }
    res.sort_unstable();
    Some(res)
}

fn main() {
    let (n, target, mut items) = parse_input();
    if n == 0 {
        output_no();
        return;
    }

    if target == 0 {
        output_yes(&[]);
        return;
    }

    // Quick direct hit.
    for it in &items {
        if it.value == target {
            output_yes(&[it.index]);
            return;
        }
    }

    let has_negative = items.iter().any(|it| it.value < 0);
    if has_negative || target < 0 {
        // Exact solver for small N only.
        if items.len() <= 44 {
            if let Some(ans) = meet_in_middle(&items, target) {
                output_yes(&ans);
            } else {
                output_no();
            }
            return;
        }

        // Best-effort for large-N with negatives: exact solution is generally infeasible.
        // Still try to find a solution using a capped hash-DP on sums near the target.
        let cap_states: usize = 2_000_000;
        let mut map: HashMap<i64, (i64, usize)> = HashMap::new();
        map.insert(0, (i64::MIN, usize::MAX));
        for (i, it) in items.iter().enumerate() {
            if map.contains_key(&target) {
                break;
            }
            let mut updates: Vec<(i64, (i64, usize))> = Vec::new();
            for (&sum, _) in map.iter() {
                let ns = sum.wrapping_add(it.value);
                if !map.contains_key(&ns) {
                    updates.push((ns, (sum, i)));
                }
                if updates.len() > 50_000 {
                    break;
                }
            }
            for (k, v) in updates {
                map.insert(k, v);
                if map.len() > cap_states {
                    // Stop growing; keep correctness best-effort only.
                    break;
                }
            }
            if map.len() > cap_states {
                break;
            }
        }
        if !map.contains_key(&target) {
            output_no();
            return;
        }
        // Reconstruct.
        let mut res = Vec::new();
        let mut cur = target;
        while cur != 0 {
            let (ps, idx) = map[&cur];
            if idx == usize::MAX {
                break;
            }
            res.push(items[idx].index);
            cur = ps;
        }
        res.sort_unstable();
        output_yes(&res);
        return;
    }

    // Non-negative case: use bitset DP when target is moderate.
    let t_usize = target as usize;

    // Filter items with value in 1..=target and keep their original indices.
    items.retain(|it| it.value > 0 && (it.value as usize) <= t_usize);
    if items.is_empty() {
        output_no();
        return;
    }

    // If there are many items, compress duplicates to reduce DP iterations.
    // Also prefer multiplicity DP when n is large.
    let bitset_limit: usize = 50_000_000; // bits; ~6.25MB bitset + parent arrays
    if t_usize > bitset_limit {
        // Target too large for pseudopolynomial DP within memory/time.
        if items.len() <= 44 {
            if let Some(ans) = meet_in_middle(&items, target) {
                output_yes(&ans);
            } else {
                output_no();
            }
        } else {
            output_no();
        }
        return;
    }

    let large_n = items.len() >= 200_000;
    if large_n {
        let mut groups: HashMap<i64, Vec<usize>> = HashMap::new();
        for it in &items {
            groups.entry(it.value).or_default().push(it.index);
        }
        let mut by_value: Vec<(i64, Vec<usize>)> = groups.into_iter().collect();
        by_value.sort_by(|a, b| a.0.cmp(&b.0));
        if let Some(ans) = bitset_subset_sum_bounded_counts(&by_value, t_usize) {
            output_yes(&ans);
        } else {
            output_no();
        }
        return;
    }

    if let Some(ans) = bitset_subset_sum_nonneg(&items, t_usize) {
        output_yes(&ans);
    } else {
        output_no();
    }
}
