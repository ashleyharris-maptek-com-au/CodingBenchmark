// Travelling Salesman Problem (TSP) heuristic solver
// Outputs a permutation of 0..N-1 starting at 0 (cycle implied).
// Strategy:
// - Read all points, compute a 2D Hilbert (space-filling curve) order as a fast, scalable tour seed.
// - For small N, optionally apply a light local improvement (bounded 2-opt window).

#include <algorithm>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct FastScanner {
  static constexpr size_t kBufSize = 1 << 20;
  char buf[kBufSize];
  size_t idx = 0, size = 0;

  inline char readChar() {
    if (idx >= size) {
      size = std::fread(buf, 1, kBufSize, stdin);
      idx = 0;
      if (size == 0) return 0;
    }
    return buf[idx++];
  }

  template <class T>
  bool readInt(T& out) {
    char c;
    do {
      c = readChar();
      if (!c) return false;
    } while (c <= ' ');  // skip whitespace
    T sign = 1;
    if (c == '-') {
      sign = -1;
      c = readChar();
    }
    T val = 0;
    while (c > ' ') {
      if (c < '0' || c > '9') return false;
      val = val * 10 + (c - '0');
      c = readChar();
    }
    out = val * sign;
    return true;
  }

  bool readFloat(float& out) {
    char c;
    do {
      c = readChar();
      if (!c) return false;
    } while (c <= ' ');

    bool neg = false;
    if (c == '-') {
      neg = true;
      c = readChar();
    }

    double val = 0.0;
    while (c >= '0' && c <= '9') {
      val = val * 10.0 + (c - '0');
      c = readChar();
    }

    if (c == '.') {
      c = readChar();
      double place = 0.1;
      while (c >= '0' && c <= '9') {
        val += (c - '0') * place;
        place *= 0.1;
        c = readChar();
      }
    }

    if (c == 'e' || c == 'E') {
      c = readChar();
      bool expNeg = false;
      if (c == '-') {
        expNeg = true;
        c = readChar();
      } else if (c == '+') {
        c = readChar();
      }
      int expVal = 0;
      while (c >= '0' && c <= '9') {
        expVal = expVal * 10 + (c - '0');
        c = readChar();
      }
      double pow10 = 1.0;
      double base = 10.0;
      int e = expVal;
      while (e) {
        if (e & 1) pow10 *= base;
        base *= base;
        e >>= 1;
      }
      val = expNeg ? (val / pow10) : (val * pow10);
    }

    out = static_cast<float>(neg ? -val : val);
    return true;
  }
};

static inline uint32_t rot(uint32_t n, uint32_t& x, uint32_t& y, uint32_t rx, uint32_t ry) {
  if (ry == 0) {
    if (rx == 1) {
      x = n - 1 - x;
      y = n - 1 - y;
    }
    uint32_t t = x;
    x = y;
    y = t;
  }
  return 0;
}

// Hilbert curve index for 2D coordinates in [0, 2^bits).
static inline uint32_t hilbertIndex(uint32_t x, uint32_t y, int bits) {
  uint32_t d = 0;
  uint32_t n = 1u << bits;
  for (uint32_t s = n >> 1; s > 0; s >>= 1) {
    uint32_t rx = (x & s) ? 1u : 0u;
    uint32_t ry = (y & s) ? 1u : 0u;
    d += s * s * ((3u * rx) ^ ry);
    rot(n, x, y, rx, ry);
  }
  return d;
}

struct KeyIdx {
  uint32_t key;
  uint32_t idx;
};

static void radixSort32(std::vector<KeyIdx>& a) {
  const size_t n = a.size();
  std::vector<KeyIdx> tmp(n);
  constexpr int kPasses = 4;
  constexpr int kRadix = 256;
  for (int pass = 0; pass < kPasses; ++pass) {
    size_t cnt[kRadix];
    std::memset(cnt, 0, sizeof(cnt));
    const int shift = pass * 8;
    for (size_t i = 0; i < n; ++i) {
      ++cnt[(a[i].key >> shift) & 255u];
    }
    size_t sum = 0;
    for (int i = 0; i < kRadix; ++i) {
      size_t c = cnt[i];
      cnt[i] = sum;
      sum += c;
    }
    for (size_t i = 0; i < n; ++i) {
      tmp[cnt[(a[i].key >> shift) & 255u]++] = a[i];
    }
    a.swap(tmp);
  }
}

static inline double dist2(const float* xs, const float* ys, uint32_t a, uint32_t b) {
  double dx = static_cast<double>(xs[a]) - static_cast<double>(xs[b]);
  double dy = static_cast<double>(ys[a]) - static_cast<double>(ys[b]);
  return dx * dx + dy * dy;
}

static void rotateToStart0(std::vector<uint32_t>& order, const float* xs, const float* ys) {
  const size_t n = order.size();
  if (n == 0) return;
  size_t pos0 = 0;
  for (size_t i = 0; i < n; ++i) {
    if (order[i] == 0u) {
      pos0 = i;
      break;
    }
  }
  if (pos0 != 0) {
    std::rotate(order.begin(), order.begin() + static_cast<std::ptrdiff_t>(pos0), order.end());
  }
  if (n >= 3) {
    // Choose direction (keep cyclic order) to make the first step from 0 shorter.
    // Compare dist(0->next) vs dist(0->prev) if we reversed the cycle.
    uint32_t next = order[1];
    uint32_t prev = order[n - 1];
    if (dist2(xs, ys, 0u, prev) < dist2(xs, ys, 0u, next)) {
      // Reverse the cycle while keeping order[0] == 0
      std::reverse(order.begin() + 1, order.end());
    }
  }
}

static void bounded2opt(std::vector<uint32_t>& order, const float* xs, const float* ys, int window) {
  const int n = static_cast<int>(order.size());
  if (n < 4) return;
  for (int i = 0; i < n - 3; ++i) {
    int maxJ = std::min(n - 2, i + 1 + window);
    uint32_t a = order[i];
    uint32_t b = order[i + 1];
    for (int j = i + 2; j <= maxJ; ++j) {
      uint32_t c = order[j];
      uint32_t d = order[j + 1];
      double cur = dist2(xs, ys, a, b) + dist2(xs, ys, c, d);
      double alt = dist2(xs, ys, a, c) + dist2(xs, ys, b, d);
      if (alt + 1e-12 < cur) {
        std::reverse(order.begin() + (i + 1), order.begin() + (j + 1));
        b = order[i + 1];
      }
    }
  }
}

int main() {
  std::ios::sync_with_stdio(false);
  std::cin.tie(nullptr);

  FastScanner fs;
  int nInt = 0;
  if (!fs.readInt(nInt)) return 0;
  if (nInt <= 0) return 0;
  const uint32_t N = static_cast<uint32_t>(nInt);

  std::vector<float> xs(N), ys(N);
  float minX = std::numeric_limits<float>::infinity();
  float maxX = -std::numeric_limits<float>::infinity();
  float minY = std::numeric_limits<float>::infinity();
  float maxY = -std::numeric_limits<float>::infinity();

  for (uint32_t i = 0; i < N; ++i) {
    float x = 0.0f, y = 0.0f;
    if (!fs.readFloat(x)) x = 0.0f;
    if (!fs.readFloat(y)) y = 0.0f;
    xs[i] = x;
    ys[i] = y;
    if (x < minX) minX = x;
    if (x > maxX) maxX = x;
    if (y < minY) minY = y;
    if (y > maxY) maxY = y;
  }

  const float dx = (maxX > minX) ? (maxX - minX) : 1.0f;
  const float dy = (maxY > minY) ? (maxY - minY) : 1.0f;
  constexpr int kBits = 16;                 // 65536 x 65536 grid
  constexpr float kScale = 65535.0f;

  std::vector<KeyIdx> keyed;
  keyed.resize(N);
  for (uint32_t i = 0; i < N; ++i) {
    float nx = (xs[i] - minX) / dx;
    float ny = (ys[i] - minY) / dy;
    if (nx < 0.0f) nx = 0.0f;
    if (nx > 1.0f) nx = 1.0f;
    if (ny < 0.0f) ny = 0.0f;
    if (ny > 1.0f) ny = 1.0f;
    uint32_t qx = static_cast<uint32_t>(nx * kScale + 0.5f);
    uint32_t qy = static_cast<uint32_t>(ny * kScale + 0.5f);
    keyed[i].key = hilbertIndex(qx, qy, kBits);
    keyed[i].idx = i;
  }

  // Adaptive sorting: radix sort for large N, std::sort for small N.
  if (N >= 200000) {
    radixSort32(keyed);
  } else {
    std::sort(keyed.begin(), keyed.end(), [](const KeyIdx& a, const KeyIdx& b) {
      if (a.key != b.key) return a.key < b.key;
      return a.idx < b.idx;
    });
  }

  std::vector<uint32_t> order;
  order.resize(N);
  for (uint32_t i = 0; i < N; ++i) order[i] = keyed[i].idx;
  rotateToStart0(order, xs.data(), ys.data());

  // Small-instance improvements.
  if (N <= 5000) {
    bounded2opt(order, xs.data(), ys.data(), 32);
    rotateToStart0(order, xs.data(), ys.data());
  } else if (N <= 50000) {
    bounded2opt(order, xs.data(), ys.data(), 8);
    rotateToStart0(order, xs.data(), ys.data());
  }

  // Output permutation
  for (uint32_t i = 0; i < N; ++i) {
    if (i) std::cout << ' ';
    std::cout << order[i];
  }
  std::cout << '\n';
  return 0;
}
