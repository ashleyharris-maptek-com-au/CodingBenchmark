use std::cmp::Ordering;
use std::collections::{HashMap, HashSet};
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

impl Pt {
    #[inline]
    fn add(self, o: Pt) -> Pt {
        Pt {
            x: self.x + o.x,
            y: self.y + o.y,
        }
    }
    #[inline]
    fn sub(self, o: Pt) -> Pt {
        Pt {
            x: self.x - o.x,
            y: self.y - o.y,
        }
    }
}

#[derive(Clone, Copy, Debug)]
struct BBox {
    minx: f64,
    miny: f64,
    maxx: f64,
    maxy: f64,
}

impl BBox {
    fn new() -> Self {
        Self {
            minx: f64::INFINITY,
            miny: f64::INFINITY,
            maxx: f64::NEG_INFINITY,
            maxy: f64::NEG_INFINITY,
        }
    }
    fn from_pts(pts: &[Pt]) -> Self {
        let mut b = Self::new();
        for &p in pts {
            b.minx = b.minx.min(p.x);
            b.miny = b.miny.min(p.y);
            b.maxx = b.maxx.max(p.x);
            b.maxy = b.maxy.max(p.y);
        }
        b
    }
    #[inline]
    fn w(&self) -> f64 {
        self.maxx - self.minx
    }
    #[inline]
    fn h(&self) -> f64 {
        self.maxy - self.miny
    }
    #[inline]
    fn area(&self) -> f64 {
        self.w().max(0.0) * self.h().max(0.0)
    }
    #[inline]
    fn intersects(&self, o: &BBox) -> bool {
        self.minx <= o.maxx && self.maxx >= o.minx && self.miny <= o.maxy && self.maxy >= o.miny
    }
}

#[derive(Clone)]
struct Poly {
    v: Vec<Pt>,
    bbox: BBox,
    area: f64,
}

impl Poly {
    fn new(v: Vec<Pt>) -> Self {
        let bbox = BBox::from_pts(&v);
        let area = poly_area(&v).abs();
        Self { v, bbox, area }
    }
    fn rotated(&self, ang: f64) -> Poly {
        let (s, c) = ang.sin_cos();
        let mut out = Vec::with_capacity(self.v.len());
        for &p in &self.v {
            out.push(Pt {
                x: p.x * c - p.y * s,
                y: p.x * s + p.y * c,
            });
        }
        Poly::new(out)
    }
    fn translated(&self, dx: f64, dy: f64) -> Poly {
        let mut out = Vec::with_capacity(self.v.len());
        for &p in &self.v {
            out.push(Pt { x: p.x + dx, y: p.y + dy });
        }
        Poly::new(out)
    }
}

#[derive(Clone)]
struct Placed {
    poly: Poly,
    bbox: BBox,
}

struct StockState {
    placed: Vec<Placed>,
    grid: HashMap<(i32, i32), Vec<usize>>,
    cell: f64,
    bbox: BBox,
}

fn poly_area(v: &[Pt]) -> f64 {
    let n = v.len();
    if n < 3 {
        return 0.0;
    }
    let mut a = 0.0;
    for i in 0..n {
        let j = (i + 1) % n;
        a += v[i].x * v[j].y - v[j].x * v[i].y;
    }
    0.5 * a
}

#[inline]
fn cross(a: Pt, b: Pt) -> f64 {
    a.x * b.y - a.y * b.x
}

#[inline]
fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(b.sub(a), c.sub(a))
}

fn on_segment(a: Pt, b: Pt, p: Pt, eps: f64) -> bool {
    let minx = a.x.min(b.x) - eps;
    let maxx = a.x.max(b.x) + eps;
    let miny = a.y.min(b.y) - eps;
    let maxy = a.y.max(b.y) + eps;
    p.x >= minx && p.x <= maxx && p.y >= miny && p.y <= maxy && orient(a, b, p).abs() <= eps
}

fn seg_intersect(a1: Pt, a2: Pt, b1: Pt, b2: Pt, eps: f64) -> bool {
    let o1 = orient(a1, a2, b1);
    let o2 = orient(a1, a2, b2);
    let o3 = orient(b1, b2, a1);
    let o4 = orient(b1, b2, a2);

    if (o1 > eps && o2 < -eps) || (o1 < -eps && o2 > eps) {
        if (o3 > eps && o4 < -eps) || (o3 < -eps && o4 > eps) {
            return true;
        }
    }

    if o1.abs() <= eps && on_segment(a1, a2, b1, eps) {
        return true;
    }
    if o2.abs() <= eps && on_segment(a1, a2, b2, eps) {
        return true;
    }
    if o3.abs() <= eps && on_segment(b1, b2, a1, eps) {
        return true;
    }
    if o4.abs() <= eps && on_segment(b1, b2, a2, eps) {
        return true;
    }
    false
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    // Ray casting; boundary counts as inside.
    let n = poly.len();
    if n < 3 {
        return false;
    }
    let eps = 1e-12;
    let mut inside = false;
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if on_segment(a, b, p, eps) {
            return true;
        }
        let intersects = (a.y > p.y) != (b.y > p.y)
            && p.x < (b.x - a.x) * (p.y - a.y) / (b.y - a.y + 0.0) + a.x;
        if intersects {
            inside = !inside;
        }
    }
    inside
}

fn polys_overlap(a: &Poly, b: &Poly) -> bool {
    let eps = 1e-10;
    if !a.bbox.intersects(&b.bbox) {
        return false;
    }

    let an = a.v.len();
    let bn = b.v.len();
    for i in 0..an {
        let a1 = a.v[i];
        let a2 = a.v[(i + 1) % an];
        for j in 0..bn {
            let b1 = b.v[j];
            let b2 = b.v[(j + 1) % bn];
            if seg_intersect(a1, a2, b1, b2, eps) {
                return true;
            }
        }
    }
    // Containment
    if point_in_poly(a.v[0], &b.v) {
        return true;
    }
    if point_in_poly(b.v[0], &a.v) {
        return true;
    }
    false
}

fn poly_inside_stock(piece: &Poly, stock: &Poly) -> bool {
    // Conservative but robust: all vertices inside + no edge crosses stock boundary.
    let eps = 1e-10;
    for &p in &piece.v {
        if !point_in_poly(p, &stock.v) {
            return false;
        }
    }
    let sn = stock.v.len();
    let pn = piece.v.len();
    for i in 0..pn {
        let p1 = piece.v[i];
        let p2 = piece.v[(i + 1) % pn];
        for j in 0..sn {
            let s1 = stock.v[j];
            let s2 = stock.v[(j + 1) % sn];
            if seg_intersect(p1, p2, s1, s2, eps) {
                // Touching boundary is fine; crossing is not. We approximate by allowing colinear touches.
                let o1 = orient(p1, p2, s1).abs() <= eps && on_segment(p1, p2, s1, eps);
                let o2 = orient(p1, p2, s2).abs() <= eps && on_segment(p1, p2, s2, eps);
                let o3 = orient(s1, s2, p1).abs() <= eps && on_segment(s1, s2, p1, eps);
                let o4 = orient(s1, s2, p2).abs() <= eps && on_segment(s1, s2, p2, eps);
                if !(o1 || o2 || o3 || o4) {
                    return false;
                }
            }
        }
    }
    true
}

fn grid_cells_for(b: &BBox, cell: f64) -> (i32, i32, i32, i32) {
    let inv = 1.0 / cell.max(1e-9);
    let x0 = (b.minx * inv).floor() as i32;
    let x1 = (b.maxx * inv).floor() as i32;
    let y0 = (b.miny * inv).floor() as i32;
    let y1 = (b.maxy * inv).floor() as i32;
    (x0, x1, y0, y1)
}

fn stock_cell_size(stock_bbox: &BBox) -> f64 {
    // Heuristic: about 32x32 grid across the stock bbox.
    let s = ((stock_bbox.w().abs() * stock_bbox.h().abs()).max(1.0) / 1024.0).sqrt();
    s.max(1e-6)
}

fn can_place_in_stock(stock: &StockState, stock_poly: &Poly, candidate: &Poly) -> bool {
    if !poly_inside_stock(candidate, stock_poly) {
        return false;
    }

    let mut candidates = HashSet::<usize>::new();
    let (x0, x1, y0, y1) = grid_cells_for(&candidate.bbox, stock.cell);
    for gx in x0..=x1 {
        for gy in y0..=y1 {
            if let Some(v) = stock.grid.get(&(gx, gy)) {
                for &idx in v {
                    candidates.insert(idx);
                }
            }
        }
    }
    for idx in candidates {
        let other = &stock.placed[idx].poly;
        if polys_overlap(candidate, other) {
            return false;
        }
    }
    true
}

fn insert_placed(stock: &mut StockState, placed: Placed) {
    let idx = stock.placed.len();
    let (x0, x1, y0, y1) = grid_cells_for(&placed.bbox, stock.cell);
    for gx in x0..=x1 {
        for gy in y0..=y1 {
            stock.grid.entry((gx, gy)).or_default().push(idx);
        }
    }
    stock.placed.push(placed);
}

fn halton(mut i: u32, base: u32) -> f64 {
    let mut f = 1.0;
    let mut r = 0.0;
    while i > 0 {
        f /= base as f64;
        r += f * (i % base) as f64;
        i /= base;
    }
    r
}

fn rotation_set(np: usize, total_vertices: usize) -> Vec<f64> {
    // Adapt based on scale: fewer angles for bigger instances.
    let complexity = total_vertices / np.max(1);
    if np >= 2000 || complexity >= 64 {
        vec![0.0]
    } else if np >= 500 {
        vec![0.0, PI * 0.5, PI, PI * 1.5]
    } else if np >= 100 {
        let k = 8;
        (0..k).map(|i| (i as f64) * 2.0 * PI / (k as f64)).collect()
    } else {
        let k = 16;
        (0..k).map(|i| (i as f64) * 2.0 * PI / (k as f64)).collect()
    }
}

fn try_place_piece(
    stock_state: &StockState,
    stock_poly: &Poly,
    piece: &Poly,
    rotations: &[f64],
    max_tries: usize,
    rng_seed: u32,
) -> Option<(f64, f64, f64, Poly)> {
    let stock_bb = stock_poly.bbox;
    let mut best: Option<(f64, f64, f64, Poly)> = None;

    // Anchor points: bottom-left of stock bbox and vertices of existing placements.
    let mut anchors: Vec<Pt> = Vec::new();
    anchors.push(Pt {
        x: stock_bb.minx,
        y: stock_bb.miny,
    });
    // Add a few anchors derived from placed pieces (kept small for scale).
    if !stock_state.placed.is_empty() {
        let mut added = 0usize;
        for pl in stock_state.placed.iter().rev().take(64) {
            for &p in pl.poly.v.iter().take(16) {
                anchors.push(p);
                added += 1;
                if added >= 256 {
                    break;
                }
            }
            if added >= 256 {
                break;
            }
        }
    }

    for &ang in rotations {
        let rotated = piece.rotated(ang);
        let bb = rotated.bbox;
        let shift_to_origin = Pt {
            x: -bb.minx,
            y: -bb.miny,
        };

        // Deterministic anchor placements (bottom-left style).
        let mut anchor_positions: Vec<(f64, f64)> = Vec::with_capacity(anchors.len());
        for &a in &anchors {
            anchor_positions.push((a.x + shift_to_origin.x, a.y + shift_to_origin.y));
        }
        anchor_positions.sort_by(|(ax, ay), (bx, by)| {
            ay.partial_cmp(by)
                .unwrap_or(Ordering::Equal)
                .then_with(|| ax.partial_cmp(bx).unwrap_or(Ordering::Equal))
        });

        for (dx, dy) in anchor_positions {
            let cand = rotated.translated(dx, dy);
            if cand.bbox.minx < stock_bb.minx - 1e-9
                || cand.bbox.miny < stock_bb.miny - 1e-9
                || cand.bbox.maxx > stock_bb.maxx + 1e-9
                || cand.bbox.maxy > stock_bb.maxy + 1e-9
            {
                continue;
            }
            if can_place_in_stock(stock_state, stock_poly, &cand) {
                let score = (dy, dx);
                match &best {
                    None => best = Some((dx, dy, ang, cand)),
                    Some((bdx, bdy, _bang, _bpoly)) => {
                        if score < (*bdy, *bdx) {
                            best = Some((dx, dy, ang, cand));
                        }
                    }
                }
                // Good enough for large instances: take first feasible.
                if stock_state.placed.len() > 200 || rotations.len() <= 4 {
                    return best;
                }
            }
        }

        // Random-ish low-discrepancy sampling within stock bbox for robustness.
        // Tries are capped and depend on scale.
        let tries = max_tries;
        for t in 1..=tries {
            let u = halton(rng_seed.wrapping_add(t as u32), 2);
            let v = halton(rng_seed.wrapping_add(t as u32), 3);
            let dx = stock_bb.minx + u * (stock_bb.w() - rotated.bbox.w());
            let dy = stock_bb.miny + v * (stock_bb.h() - rotated.bbox.h());
            let cand = rotated.translated(dx + shift_to_origin.x, dy + shift_to_origin.y);
            if cand.bbox.minx < stock_bb.minx - 1e-9
                || cand.bbox.miny < stock_bb.miny - 1e-9
                || cand.bbox.maxx > stock_bb.maxx + 1e-9
                || cand.bbox.maxy > stock_bb.maxy + 1e-9
            {
                continue;
            }
            if can_place_in_stock(stock_state, stock_poly, &cand) {
                let score = (cand.bbox.miny, cand.bbox.minx);
                match &best {
                    None => best = Some((cand.bbox.minx - rotated.bbox.minx, cand.bbox.miny - rotated.bbox.miny, ang, cand)),
                    Some((bdx, bdy, _bang, _bpoly)) => {
                        if score < (*bdy, *bdx) {
                            best = Some((
                                cand.bbox.minx - rotated.bbox.minx,
                                cand.bbox.miny - rotated.bbox.miny,
                                ang,
                                cand,
                            ));
                        }
                    }
                }
            }
        }
    }

    best
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();

    let sv: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let np: usize = it.next().unwrap().parse().unwrap();

    let mut stock_pts = Vec::with_capacity(sv);
    for _ in 0..sv {
        let x: f64 = it.next().unwrap().parse().unwrap();
        let y: f64 = it.next().unwrap().parse().unwrap();
        stock_pts.push(Pt { x, y });
    }
    let stock_poly = Poly::new(stock_pts);
    let stock_cell = stock_cell_size(&stock_poly.bbox);

    let mut pieces: Vec<Poly> = Vec::with_capacity(np);
    let mut total_vertices = 0usize;
    for _ in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        total_vertices += pv;
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            let x: f64 = it.next().unwrap().parse().unwrap();
            let y: f64 = it.next().unwrap().parse().unwrap();
            pts.push(Pt { x, y });
        }
        pieces.push(Poly::new(pts));
    }

    let rotations = rotation_set(np, total_vertices);

    let mut order: Vec<usize> = (0..np).collect();
    order.sort_by(|&i, &j| {
        let a = &pieces[i];
        let b = &pieces[j];
        b.area
            .partial_cmp(&a.area)
            .unwrap_or(Ordering::Equal)
            .then_with(|| {
                let ad = a.bbox.w().max(a.bbox.h());
                let bd = b.bbox.w().max(b.bbox.h());
                bd.partial_cmp(&ad).unwrap_or(Ordering::Equal)
            })
    });

    let mut stocks: Vec<StockState> = Vec::new();
    stocks.push(StockState {
        placed: Vec::new(),
        grid: HashMap::new(),
        cell: stock_cell,
        bbox: stock_poly.bbox,
    });

    let mut result_stock = vec![0usize; np];
    let mut result_x = vec![0.0f64; np];
    let mut result_y = vec![0.0f64; np];
    let mut result_r = vec![0.0f64; np];

    for (k, &idx) in order.iter().enumerate() {
        let piece = &pieces[idx];
        let mut placed = None;

        // Scale-aware effort.
        let max_tries = if np >= 2000 {
            12
        } else if np >= 500 {
            48
        } else if np >= 100 {
            200
        } else {
            600
        };
        let seed = (idx as u32).wrapping_mul(2654435761u32) ^ (k as u32);

        for s in 0..stocks.len() {
            if let Some((dx, dy, ang, cand_poly)) =
                try_place_piece(&stocks[s], &stock_poly, piece, &rotations, max_tries, seed)
            {
                placed = Some((s, dx, dy, ang, cand_poly));
                break;
            }
        }

        if placed.is_none() {
            // Open a new stock and try there.
            stocks.push(StockState {
                placed: Vec::new(),
                grid: HashMap::new(),
                cell: stock_cell,
                bbox: stock_poly.bbox,
            });
            let s = stocks.len() - 1;
            if let Some((dx, dy, ang, cand_poly)) =
                try_place_piece(&stocks[s], &stock_poly, piece, &rotations, max_tries * 2, seed ^ 0x9e3779b9)
            {
                placed = Some((s, dx, dy, ang, cand_poly));
            } else {
                // Last-resort fallback (should be rare): place at stock bbox min with 0 rotation.
                let rot = piece.rotated(0.0);
                let bb = rot.bbox;
                let dx = stock_poly.bbox.minx - bb.minx;
                let dy = stock_poly.bbox.miny - bb.miny;
                let cand = rot.translated(dx, dy);
                placed = Some((s, dx, dy, 0.0, cand));
            }
        }

        let (s, dx, dy, ang, cand_poly) = placed.unwrap();
        result_stock[idx] = s;
        result_x[idx] = dx;
        result_y[idx] = dy;
        result_r[idx] = ang;

        let placed_rec = Placed {
            bbox: cand_poly.bbox,
            poly: cand_poly,
        };
        insert_placed(&mut stocks[s], placed_rec);
    }

    // Output
    let mut out = String::new();
    out.push_str(&format!("{}\n", stocks.len()));
    for i in 0..np {
        out.push_str(&format!(
            "{} {:.10} {:.10} {:.10}\n",
            result_stock[i], result_x[i], result_y[i], result_r[i]
        ));
    }
    print!("{out}");
}
