use std::f64::consts::PI;
use std::io::{self, Read, Write};
use std::time::{Duration, Instant};

struct FastInput<R: Read> {
    r: R,
    buf: [u8; 1 << 16],
    idx: usize,
    len: usize,
}

impl<R: Read> FastInput<R> {
    fn new(r: R) -> Self {
        Self {
            r,
            buf: [0u8; 1 << 16],
            idx: 0,
            len: 0,
        }
    }

    fn refill(&mut self) -> io::Result<bool> {
        if self.idx < self.len {
            return Ok(true);
        }
        self.len = self.r.read(&mut self.buf)?;
        self.idx = 0;
        Ok(self.len != 0)
    }

    fn peek_byte(&mut self) -> io::Result<Option<u8>> {
        if !self.refill()? {
            return Ok(None);
        }
        Ok(Some(self.buf[self.idx]))
    }

    fn next_byte(&mut self) -> io::Result<Option<u8>> {
        if !self.refill()? {
            return Ok(None);
        }
        let b = self.buf[self.idx];
        self.idx += 1;
        Ok(Some(b))
    }

    fn skip_ws(&mut self) -> io::Result<()> {
        while let Some(b) = self.peek_byte()? {
            if b.is_ascii_whitespace() {
                self.idx += 1;
            } else {
                break;
            }
        }
        Ok(())
    }

    fn next_f64(&mut self) -> io::Result<Option<f64>> {
        self.skip_ws()?;
        let mut s = Vec::<u8>::with_capacity(32);
        while let Some(b) = self.peek_byte()? {
            if b.is_ascii_whitespace() {
                break;
            }
            s.push(b);
            self.idx += 1;
            if s.len() > 128 {
                break;
            }
        }
        if s.is_empty() {
            return Ok(None);
        }
        let st = std::str::from_utf8(&s)
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "non-utf8 number"))?;
        let v = st
            .parse::<f64>()
            .map_err(|_| io::Error::new(io::ErrorKind::InvalidData, "bad number"))?;
        Ok(Some(v))
    }

    fn next_usize(&mut self) -> io::Result<Option<usize>> {
        Ok(self.next_f64()?.map(|v| v as usize))
    }
}

fn clamp(x: f64, lo: f64, hi: f64) -> f64 {
    if x < lo {
        lo
    } else if x > hi {
        hi
    } else {
        x
    }
}

fn wrap_pi(mut a: f64) -> f64 {
    while a > PI {
        a -= 2.0 * PI;
    }
    while a < -PI {
        a += 2.0 * PI;
    }
    a
}

#[derive(Clone, Copy, Debug)]
struct Env {
    gravity: f64,
    max_thrust: f64,
}

#[derive(Clone, Copy, Debug)]
struct Target {
    x: f64,
    y: f64,
    l_min_x: f64,
    l_max_x: f64,
    l_min_y: f64,
    l_max_y: f64,
}

#[derive(Clone, Copy, Debug)]
struct State {
    x: f64,
    y: f64,
    vx: f64,
    vy: f64,
    angle: f64,
    angular_vel: f64,
    fuel_remaining: f64,
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut input = FastInput::new(stdin.lock());

    let width = match input.next_usize()? {
        Some(v) => v,
        None => return Ok(()),
    };
    let height = input
        .next_usize()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing height"))?;
    let gravity = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing gravity"))?;
    let max_thrust = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing max_thrust"))?;
    let _fuel_total = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing fuel"))?;

    let _start_x = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing start_x"))?;
    let _start_y = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing start_y"))?;
    let target_x = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing target_x"))?;
    let target_y = input
        .next_f64()?
        .ok_or_else(|| io::Error::new(io::ErrorKind::UnexpectedEof, "missing target_y"))?;

    // Scan the map in a streaming manner; keep only landing-zone extents and a small
    // "ground height" profile near the target.
    let window_half = 1024usize;
    let txi = clamp(target_x.floor(), 0.0, (width.saturating_sub(1)) as f64) as usize;
    let win_lo = txi.saturating_sub(window_half);
    let win_hi = (txi + window_half).min(width.saturating_sub(1));
    let win_w = win_hi - win_lo + 1;
    let mut ground_y = vec![f64::INFINITY; win_w]; // min y of any solid cell in that column

    let mut l_min_x: Option<usize> = None;
    let mut l_max_x: Option<usize> = None;
    let mut l_min_y: Option<usize> = None;
    let mut l_max_y: Option<usize> = None;

    for y in 0..height {
        input.skip_ws()?;
        let mut x = 0usize;
        while x < width {
            let b = match input.next_byte()? {
                Some(v) => v,
                None => break,
            };
            if b.is_ascii_whitespace() {
                continue;
            }

            let is_solid = matches!(b, b'#' | b'L' | b'T');
            if is_solid {
                if x >= win_lo && x <= win_hi {
                    let idx = x - win_lo;
                    let yf = y as f64;
                    if yf < ground_y[idx] {
                        ground_y[idx] = yf;
                    }
                }
            }

            if b == b'L' {
                l_min_x = Some(l_min_x.map_or(x, |v| v.min(x)));
                l_max_x = Some(l_max_x.map_or(x, |v| v.max(x)));
                l_min_y = Some(l_min_y.map_or(y, |v| v.min(y)));
                l_max_y = Some(l_max_y.map_or(y, |v| v.max(y)));
            }

            x += 1;
        }
    }

    let (l_min_x, l_max_x, l_min_y, l_max_y) = match (l_min_x, l_max_x, l_min_y, l_max_y) {
        (Some(a), Some(b), Some(c), Some(d)) => (a, b, c, d),
        _ => {
            // If no explicit landing zone is present, treat the single target cell as the zone.
            let tx = clamp(target_x.floor(), 0.0, (width.saturating_sub(1)) as f64) as usize;
            let ty = clamp(target_y.floor(), 0.0, (height.saturating_sub(1)) as f64) as usize;
            (tx, tx, ty, ty)
        }
    };

    let env = Env {
        gravity,
        max_thrust,
    };
    let target = Target {
        x: target_x,
        y: target_y,
        l_min_x: l_min_x as f64,
        l_max_x: (l_max_x + 1) as f64, // treat as [min, max+1] in meters
        l_min_y: l_min_y as f64,
        l_max_y: (l_max_y + 1) as f64,
    };

    let stdout = io::stdout();
    let mut out = io::BufWriter::with_capacity(1 << 16, stdout.lock());

    // Control: run at ~20Hz even if state updates are slower.
    let mut last_cmd = (0.0f64, 0.0f64);
    let mut last_out_t = Instant::now() - Duration::from_millis(100);
    let min_period = Duration::from_millis(50);

    // Read continuous state updates until EOF.
    loop {
        let x = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let y = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let vx = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let vy = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let angle = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let angular_vel = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };
        let fuel_remaining = match input.next_f64()? {
            Some(v) => v,
            None => break,
        };

        let s = State {
            x,
            y,
            vx,
            vy,
            angle,
            angular_vel,
            fuel_remaining,
        };

        let now = Instant::now();
        if now.duration_since(last_out_t) >= min_period {
            last_cmd = compute_command(env, target, &ground_y, win_lo, win_hi, s);
            writeln!(out, "{:.6} {:.6}", last_cmd.0, last_cmd.1)?;
            // Avoid frequent flush; keep buffering but push out periodically.
            let _ = out.flush();
            last_out_t = now;
        } else {
            // If updates arrive faster than 20Hz, still refresh command cheaply.
            last_cmd = compute_command(env, target, &ground_y, win_lo, win_hi, s);
        }
    }

    // On EOF, emit a final safe command.
    writeln!(out, "{:.6} {:.6}", last_cmd.0, last_cmd.1)?;
    out.flush()?;
    Ok(())
}

fn compute_command(
    env: Env,
    target: Target,
    ground_y: &[f64],
    win_lo: usize,
    win_hi: usize,
    s: State,
) -> (f64, f64) {
    // Heuristic controller:
    // 1) Use lateral PD to drive x into landing zone center.
    // 2) Use vertical PD to slow descent; aim for gentle touchdown.
    // 3) Convert desired acceleration into a desired tilt + thrust.
    //
    // Angles: 0 faces up. Positive angle rotates clockwise.

    let zone_center_x = 0.5 * (target.l_min_x + target.l_max_x);
    let zone_half_w = 0.5 * (target.l_max_x - target.l_min_x).max(1.0);
    let mut x_goal = zone_center_x;
    if s.x < target.l_min_x {
        x_goal = target.l_min_x + zone_half_w * 0.25;
    } else if s.x > target.l_max_x {
        x_goal = target.l_max_x - zone_half_w * 0.25;
    }

    // Estimate altitude above "ground" near current x (only within our window).
    let xi = clamp(s.x.floor(), win_lo as f64, win_hi as f64) as usize;
    let mut ground_est = target.y.max(target.l_min_y);
    if xi >= win_lo && xi <= win_hi {
        let gy = ground_y[xi - win_lo];
        if gy.is_finite() {
            ground_est = gy;
        }
    }
    let altitude = (ground_est - s.y).max(0.0);

    // Lateral control: stronger when high, gentler near the ground.
    let horiz_kp = 0.08;
    let horiz_kd = 0.25;
    let slow_factor = clamp((altitude - 5.0) / 40.0, 0.2, 1.0);
    let ax_des = slow_factor * (horiz_kp * (x_goal - s.x) - horiz_kd * s.vx);

    // Vertical target: keep descent rate limited, with a flare close to ground.
    let vy_target = if altitude > 60.0 {
        15.0
    } else if altitude > 20.0 {
        8.0
    } else if altitude > 8.0 {
        4.0
    } else if altitude > 2.0 {
        1.5
    } else {
        0.5
    };

    let vert_kp = 0.9;
    let vert_kd = 0.35;
    let ay_des = vert_kp * (vy_target - s.vy) - vert_kd * s.angular_vel.abs().min(2.0);

    // Convert desired accelerations into desired thrust direction.
    // In world coords: +x right, +y down. Gravity is +g (down).
    // Thrust acceleration is along lander's facing direction, where angle=0 is up (negative y).
    // Desired net acceleration (excluding gravity) should be (ax_des, ay_des - g).
    let ax = ax_des;
    let ay = ay_des - env.gravity;

    // Desired thrust direction opposite of desired acceleration sign conventions:
    // If lander points with unit vector u = (sin(angle), -cos(angle)), then thrust accel = u * a_thrust.
    // Choose angle so u aligns with vector (ax, ay) (desired thrust accel).
    let desired_angle = clamp(ax.atan2(-ay), -PI * 0.95, PI * 0.95);
    let angle_err = wrap_pi(desired_angle - s.angle);

    let turn_kp = 2.8;
    let turn_kd = 0.8;
    let mut turn = turn_kp * angle_err - turn_kd * s.angular_vel;
    turn = clamp(turn, -1.0, 1.0);

    // Required thrust magnitude to achieve (ax, ay) given current angle.
    let ux = s.angle.sin();
    let uy = -s.angle.cos();
    let a_along = ax * ux + ay * uy;
    let mut thrust = if a_along <= 0.0 {
        0.0
    } else {
        a_along / env.max_thrust
    };

    // Safety: near ground, bias to keep upright and limit horizontal speed.
    if altitude < 10.0 {
        let upright_err = wrap_pi(0.0 - s.angle);
        let upright_turn = clamp(3.5 * upright_err - 0.7 * s.angular_vel, -1.0, 1.0);
        // Blend toward upright as we approach.
        let w = clamp((10.0 - altitude) / 10.0, 0.0, 1.0);
        turn = (1.0 - w) * turn + w * upright_turn;
        // Add extra thrust if descending too fast.
        if s.vy > 2.0 {
            thrust = thrust.max((env.gravity + 2.0) / env.max_thrust);
        }
        // If drifting sideways near touchdown, reduce tilt commands by clamping ax.
        if s.vx.abs() > 2.0 {
            thrust = thrust.max((env.gravity + 1.0) / env.max_thrust);
        }
    }

    // Fuel-aware cap: if almost out of fuel, prioritize soft impact over lateral correction.
    if s.fuel_remaining < 1.0 {
        turn = clamp(turn, -0.4, 0.4);
        thrust = thrust.max(env.gravity / env.max_thrust);
    }

    thrust = clamp(thrust, 0.0, 1.0);
    (thrust, turn)
}
