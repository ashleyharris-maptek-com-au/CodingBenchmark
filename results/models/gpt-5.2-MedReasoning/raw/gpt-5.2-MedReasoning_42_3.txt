; SPIR-V
; Version: 1.0
; Geometry shader: expand line into thick ribbon (quad)

OpCapability Geometry
%glsl = OpExtInstImport "GLSL.std.450"
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_out %outColor
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

OpName %main "main"
OpName %gl_in "gl_in"
OpName %gl_out "gl_out"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %outColor "outColor"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpMemberName %gl_PerVertex_out 0 "gl_Position"

OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position

OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %outColor Location 0

%void = OpTypeVoid
%fn = OpTypeFunction %void

%float = OpTypeFloat 32
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4

%gl_PerVertex_in = OpTypeStruct %v4float
%gl_PerVertex_out = OpTypeStruct %v4float

%uint_2 = OpConstant %uint 2
%arr_pv_in = OpTypeArray %gl_PerVertex_in %uint_2
%arr_v3_in = OpTypeArray %v3float %uint_2

%ptr_in_arr_pv = OpTypePointer Input %arr_pv_in
%ptr_in_arr_v3 = OpTypePointer Input %arr_v3_in
%ptr_out_pv = OpTypePointer Output %gl_PerVertex_out
%ptr_out_v3 = OpTypePointer Output %v3float

%ptr_in_v4 = OpTypePointer Input %v4float
%ptr_in_v3 = OpTypePointer Input %v3float
%ptr_out_v4 = OpTypePointer Output %v4float

%gl_in = OpVariable %ptr_in_arr_pv Input
%worldPos = OpVariable %ptr_in_arr_v3 Input
%normal = OpVariable %ptr_in_arr_v3 Input
%color = OpVariable %ptr_in_arr_v3 Input

%gl_out = OpVariable %ptr_out_pv Output
%outColor = OpVariable %ptr_out_v3 Output

%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1

%float_0 = OpConstant %float 0
%float_0_03 = OpConstant %float 0.03

%main = OpFunction %void None %fn
%entry = OpLabel

; Load line endpoints (clip-space positions)
%p0_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_0 %int_0
%p1_ptr = OpAccessChain %ptr_in_v4 %gl_in %int_1 %int_0
%p0 = OpLoad %v4float %p0_ptr
%p1 = OpLoad %v4float %p1_ptr

; Load per-vertex colors
%c0_ptr = OpAccessChain %ptr_in_v3 %color %int_0
%c1_ptr = OpAccessChain %ptr_in_v3 %color %int_1
%c0 = OpLoad %v3float %c0_ptr
%c1 = OpLoad %v3float %c1_ptr

; dir = p1 - p0
%dir4 = OpFSub %v4float %p1 %p0
%dirx = OpCompositeExtract %float %dir4 0
%diry = OpCompositeExtract %float %dir4 1

; perp = normalize( vec2(-dir.y, dir.x) ) * thickness
%neg_diry = OpFNegate %float %diry
%perp_raw = OpCompositeConstruct %v2float %neg_diry %dirx
%perp_norm = OpExtInst %v2float %glsl Normalize %perp_raw
%perp2 = OpVectorTimesScalar %v2float %perp_norm %float_0_03
%perpx = OpCompositeExtract %float %perp2 0
%perpy = OpCompositeExtract %float %perp2 1

; Extract z/w
%p0z = OpCompositeExtract %float %p0 2
%p0w = OpCompositeExtract %float %p0 3
%p1z = OpCompositeExtract %float %p1 2
%p1w = OpCompositeExtract %float %p1 3

; p0 +/- perp
%p0x = OpCompositeExtract %float %p0 0
%p0y = OpCompositeExtract %float %p0 1
%p0x_p = OpFAdd %float %p0x %perpx
%p0y_p = OpFAdd %float %p0y %perpy
%p0x_m = OpFSub %float %p0x %perpx
%p0y_m = OpFSub %float %p0y %perpy
%p0_plus = OpCompositeConstruct %v4float %p0x_p %p0y_p %p0z %p0w
%p0_minus = OpCompositeConstruct %v4float %p0x_m %p0y_m %p0z %p0w

; p1 +/- perp
%p1x = OpCompositeExtract %float %p1 0
%p1y = OpCompositeExtract %float %p1 1
%p1x_p = OpFAdd %float %p1x %perpx
%p1y_p = OpFAdd %float %p1y %perpy
%p1x_m = OpFSub %float %p1x %perpx
%p1y_m = OpFSub %float %p1y %perpy
%p1_plus = OpCompositeConstruct %v4float %p1x_p %p1y_p %p1z %p1w
%p1_minus = OpCompositeConstruct %v4float %p1x_m %p1y_m %p1z %p1w

; Emit triangle strip: p0+perp, p0-perp, p1+perp, p1-perp
%glpos_out_ptr = OpAccessChain %ptr_out_v4 %gl_out %int_0

OpStore %glpos_out_ptr %p0_plus
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_out_ptr %p0_minus
OpStore %outColor %c0
OpEmitVertex

OpStore %glpos_out_ptr %p1_plus
OpStore %outColor %c1
OpEmitVertex

OpStore %glpos_out_ptr %p1_minus
OpStore %outColor %c1
OpEmitVertex

OpEndPrimitive
OpReturn
OpFunctionEnd
