use std::cmp::Ordering;
use std::collections::BinaryHeap;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct XorShift64 {
    state: u64,
}

impl XorShift64 {
    fn new(seed: u64) -> Self {
        let mut s = seed;
        if s == 0 {
            s = 0x9E37_79B9_7F4A_7C15;
        }
        Self { state: s }
    }

    fn next_u64(&mut self) -> u64 {
        // xorshift64*
        let mut x = self.state;
        x ^= x >> 12;
        x ^= x << 25;
        x ^= x >> 27;
        self.state = x;
        x.wrapping_mul(0x2545_F491_4F6C_DD1D)
    }

    fn next_u32(&mut self) -> u32 {
        (self.next_u64() >> 32) as u32
    }
}

#[derive(Copy, Clone, Eq, PartialEq)]
struct HeapItem {
    gain: u32,
    tie: u32,
    v: usize,
}

impl Ord for HeapItem {
    fn cmp(&self, other: &Self) -> Ordering {
        // Max-heap by gain, then tie, then smaller vertex id (arbitrary)
        self.gain
            .cmp(&other.gain)
            .then_with(|| self.tie.cmp(&other.tie))
            .then_with(|| other.v.cmp(&self.v))
    }
}

impl PartialOrd for HeapItem {
    fn partial_cmp(&self, other: &Self) -> Option<Ordering> {
        Some(self.cmp(other))
    }
}

fn compute_gain(v: usize, adj: &[Vec<usize>], dom_count: &[u32]) -> u32 {
    let mut gain = 0u32;
    if dom_count[v] == 0 {
        gain += 1;
    }
    for &u in &adj[v] {
        if dom_count[u] == 0 {
            gain += 1;
        }
    }
    gain
}

fn add_vertex(
    v: usize,
    adj: &[Vec<usize>],
    in_set: &mut [bool],
    dom_count: &mut [u32],
    uncovered: &mut usize,
    set: &mut Vec<usize>,
) {
    if in_set[v] {
        return;
    }
    in_set[v] = true;
    set.push(v);
    if dom_count[v] == 0 {
        *uncovered -= 1;
    }
    dom_count[v] += 1;
    for &u in &adj[v] {
        if dom_count[u] == 0 {
            *uncovered -= 1;
        }
        dom_count[u] += 1;
    }
}

fn remove_vertex(
    v: usize,
    adj: &[Vec<usize>],
    in_set: &mut [bool],
    dom_count: &mut [u32],
    uncovered: &mut usize,
) {
    if !in_set[v] {
        return;
    }
    in_set[v] = false;
    dom_count[v] -= 1;
    if dom_count[v] == 0 {
        *uncovered += 1;
    }
    for &u in &adj[v] {
        dom_count[u] -= 1;
        if dom_count[u] == 0 {
            *uncovered += 1;
        }
    }
}

fn is_removable(v: usize, adj: &[Vec<usize>], dom_count: &[u32]) -> bool {
    if dom_count[v] == 1 {
        return false;
    }
    for &u in &adj[v] {
        if dom_count[u] == 1 {
            return false;
        }
    }
    true
}

fn greedy_dominating_set(adj: &[Vec<usize>], rng: &mut XorShift64) -> Vec<usize> {
    let n = adj.len();
    let mut dom_count = vec![0u32; n];
    let mut in_set = vec![false; n];
    let mut set: Vec<usize> = Vec::new();
    let mut uncovered = n;

    let mut heap: BinaryHeap<HeapItem> = BinaryHeap::with_capacity(n);
    for v in 0..n {
        let gain = (adj[v].len() as u32) + 1;
        heap.push(HeapItem {
            gain,
            tie: rng.next_u32(),
            v,
        });
    }

    while uncovered > 0 {
        let item = match heap.pop() {
            Some(x) => x,
            None => break,
        };
        if in_set[item.v] {
            continue;
        }
        let actual_gain = compute_gain(item.v, adj, &dom_count);
        if actual_gain != item.gain {
            heap.push(HeapItem {
                gain: actual_gain,
                tie: rng.next_u32(),
                v: item.v,
            });
            continue;
        }

        // If gain is 0, we still need to cover remaining vertices: pick an uncovered vertex itself.
        if actual_gain == 0 {
            // Find any uncovered vertex u and add it (guaranteed gain >= 1 for itself).
            let mut picked = None;
            for u in 0..n {
                if dom_count[u] == 0 {
                    picked = Some(u);
                    break;
                }
            }
            if let Some(u) = picked {
                add_vertex(u, adj, &mut in_set, &mut dom_count, &mut uncovered, &mut set);
                continue;
            } else {
                break;
            }
        }

        add_vertex(
            item.v,
            adj,
            &mut in_set,
            &mut dom_count,
            &mut uncovered,
            &mut set,
        );
    }

    // Prune redundant vertices (single pass in randomized order)
    // Build list to iterate without mutating `set` in-place.
    let mut order = set.clone();
    // Fisher-Yates shuffle
    for i in (1..order.len()).rev() {
        let j = (rng.next_u64() as usize) % (i + 1);
        order.swap(i, j);
    }
    for v in order {
        if in_set[v] && is_removable(v, adj, &dom_count) {
            remove_vertex(v, adj, &mut in_set, &mut dom_count, &mut uncovered);
        }
    }

    // Rebuild final set vector
    let mut result = Vec::new();
    result.reserve(n.min(1024));
    for v in 0..n {
        if in_set[v] {
            result.push(v);
        }
    }
    result
}

fn is_valid_dominating_set(adj: &[Vec<usize>], set: &[usize]) -> bool {
    let n = adj.len();
    let mut dominated = vec![false; n];
    for &v in set {
        if v >= n {
            return false;
        }
        dominated[v] = true;
        for &u in &adj[v] {
            dominated[u] = true;
        }
    }
    dominated.into_iter().all(|x| x)
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    if input.trim().is_empty() {
        return;
    }
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };
    let m: usize = match it.next() {
        Some(x) => x.parse().unwrap(),
        None => return,
    };

    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        let v: usize = match it.next() {
            Some(x) => x.parse().unwrap(),
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        adj[u].push(v);
        adj[v].push(u);
    }
    for v in 0..n {
        adj[v].sort_unstable();
        adj[v].dedup();
    }

    let start = Instant::now();
    let time_budget = Duration::from_secs(288); // leave a little slack under 5 minutes

    let mut rng = XorShift64::new((n as u64) ^ ((m as u64) << 32) ^ 0xD1B5_4A32_192E_3A59);

    let max_iters = if n <= 20_000 {
        60
    } else if n <= 100_000 {
        12
    } else {
        2
    };

    let mut best: Vec<usize> = Vec::new();
    let mut best_size = usize::MAX;

    for _ in 0..max_iters {
        if start.elapsed() + Duration::from_millis(200) >= time_budget {
            break;
        }
        let candidate = greedy_dominating_set(&adj, &mut rng);
        if candidate.len() < best_size && is_valid_dominating_set(&adj, &candidate) {
            best_size = candidate.len();
            best = candidate;
            if best_size <= 1 {
                break;
            }
        }
    }

    if best.is_empty() && n > 0 {
        // Fallback: all vertices (always valid)
        best = (0..n).collect();
        best_size = n;
    }

    // Output
    let mut out = String::new();
    out.push_str(&format!("{}\n", best_size));
    if best_size > 0 {
        for (i, v) in best.iter().enumerate() {
            if i > 0 {
                out.push(' ');
            }
            out.push_str(&v.to_string());
        }
    }
    out.push('\n');
    print!("{out}");
}
