// Steiner Tree heuristic (metric-closure MST + path expansion + Kruskal cleanup)
// C++20 (MSVC /std:c++20)

#include <algorithm>
#include <cctype>
#include <cstdint>
#include <cstdio>
#include <cstdlib>
#include <functional>
#include <iostream>
#include <limits>
#include <queue>
#include <tuple>
#include <utility>
#include <vector>

using std::int64_t;

struct FastScanner {
    static constexpr size_t BUFSIZE = 1 << 16;
    char buf[BUFSIZE];
    size_t idx = 0, size = 0;

    inline char readChar() {
        if (idx >= size) {
            size = std::fread(buf, 1, BUFSIZE, stdin);
            idx = 0;
            if (size == 0) return 0;
        }
        return buf[idx++];
    }

    template <class T>
    bool readInt(T& out) {
        char c;
        do {
            c = readChar();
            if (!c) return false;
        } while (std::isspace(static_cast<unsigned char>(c)));
        bool neg = false;
        if (c == '-') {
            neg = true;
            c = readChar();
        }
        T val = 0;
        while (c && !std::isspace(static_cast<unsigned char>(c))) {
            val = static_cast<T>(val * 10 + (c - '0'));
            c = readChar();
        }
        out = neg ? -val : val;
        return true;
    }
};

struct DSU {
    std::vector<int> p, r;
    explicit DSU(int n) : p(n), r(n, 0) {
        for (int i = 0; i < n; ++i) p[i] = i;
    }
    int find(int a) {
        while (p[a] != a) {
            p[a] = p[p[a]];
            a = p[a];
        }
        return a;
    }
    bool unite(int a, int b) {
        a = find(a);
        b = find(b);
        if (a == b) return false;
        if (r[a] < r[b]) std::swap(a, b);
        p[b] = a;
        if (r[a] == r[b]) ++r[a];
        return true;
    }
};

struct AdjEdge {
    int to;
    int id;
    int64_t w;
};

static constexpr int64_t INF = (std::numeric_limits<int64_t>::max() / 4);

static std::vector<int64_t> dijkstra_dist(int n, const std::vector<std::vector<AdjEdge>>& adj, int source) {
    std::vector<int64_t> dist(n, INF);
    dist[source] = 0;
    using P = std::pair<int64_t, int>;
    std::priority_queue<P, std::vector<P>, std::greater<P>> pq;
    pq.push({0, source});
    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (d != dist[v]) continue;
        for (const auto& e : adj[v]) {
            int to = e.to;
            int64_t nd = d + e.w;
            if (nd < dist[to]) {
                dist[to] = nd;
                pq.push({nd, to});
            }
        }
    }
    return dist;
}

static bool dijkstra_path_edges(
    int n,
    const std::vector<std::vector<AdjEdge>>& adj,
    int source,
    int target,
    std::vector<int>& parentV,
    std::vector<int>& parentE
) {
    parentV.assign(n, -1);
    parentE.assign(n, -1);
    std::vector<int64_t> dist(n, INF);
    dist[source] = 0;
    using P = std::pair<int64_t, int>;
    std::priority_queue<P, std::vector<P>, std::greater<P>> pq;
    pq.push({0, source});
    while (!pq.empty()) {
        auto [d, v] = pq.top();
        pq.pop();
        if (d != dist[v]) continue;
        if (v == target) return true;
        for (const auto& e : adj[v]) {
            int to = e.to;
            int64_t nd = d + e.w;
            if (nd < dist[to]) {
                dist[to] = nd;
                parentV[to] = v;
                parentE[to] = e.id;
                pq.push({nd, to});
            }
        }
    }
    return (source == target);
}

int main() {
    FastScanner fs;
    int n = 0, m = 0, k = 0;
    if (!fs.readInt(n)) return 0;
    fs.readInt(m);
    fs.readInt(k);

    std::vector<int> eu(m), ev(m);
    std::vector<int64_t> ew(m);
    std::vector<std::vector<AdjEdge>> adj(n);

    for (int i = 0; i < m; ++i) {
        int u, v;
        int64_t w;
        fs.readInt(u);
        fs.readInt(v);
        fs.readInt(w);
        eu[i] = u;
        ev[i] = v;
        ew[i] = w;
        adj[u].push_back({v, i, w});
        adj[v].push_back({u, i, w});
    }

    std::vector<int> terminals;
    terminals.reserve(std::max(0, k));
    for (int i = 0; i < k; ++i) {
        int t;
        if (!fs.readInt(t)) break;
        terminals.push_back(t);
    }
    k = static_cast<int>(terminals.size());

    if (k <= 1) {
        std::cout << 0 << " " << 0 << "\n";
        return 0;
    }

    // Compute distances between terminals (metric closure weights).
    std::vector<int64_t> distTT(static_cast<size_t>(k) * static_cast<size_t>(k), INF);
    for (int i = 0; i < k; ++i) {
        auto dist = dijkstra_dist(n, adj, terminals[i]);
        for (int j = 0; j < k; ++j) {
            distTT[static_cast<size_t>(i) * k + j] = dist[terminals[j]];
        }
    }

    // Prim MST on the complete graph of terminals.
    std::vector<int64_t> best(k, INF);
    std::vector<int> parent(k, -1);
    std::vector<char> used(k, 0);
    best[0] = 0;
    for (int it = 0; it < k; ++it) {
        int v = -1;
        for (int i = 0; i < k; ++i) {
            if (!used[i] && (v == -1 || best[i] < best[v])) v = i;
        }
        if (v == -1 || best[v] >= INF / 2) break;
        used[v] = 1;
        for (int to = 0; to < k; ++to) {
            if (used[to]) continue;
            int64_t w = distTT[static_cast<size_t>(v) * k + to];
            if (w < best[to]) {
                best[to] = w;
                parent[to] = v;
            }
        }
    }

    // Expand MST edges back into original graph paths; collect candidate edge ids.
    std::vector<char> inCandidate(m, 0);
    std::vector<int> parentV, parentE;
    for (int i = 1; i < k; ++i) {
        int p = parent[i];
        if (p < 0) continue; // unreachable; best effort.
        int s = terminals[i];
        int t = terminals[p];
        if (s == t) continue;
        bool ok = dijkstra_path_edges(n, adj, s, t, parentV, parentE);
        if (!ok) continue;
        int cur = t;
        while (cur != s && cur != -1) {
            int pe = parentE[cur];
            if (pe >= 0) inCandidate[pe] = 1;
            cur = parentV[cur];
        }
    }

    std::vector<int> candidateEdges;
    candidateEdges.reserve(m);
    for (int id = 0; id < m; ++id) {
        if (inCandidate[id]) candidateEdges.push_back(id);
    }

    // If somehow empty, fall back to a spanning tree of the whole graph (connects all vertices => includes terminals).
    if (candidateEdges.empty()) {
        candidateEdges.resize(m);
        for (int i = 0; i < m; ++i) candidateEdges[i] = i;
    }

    std::sort(candidateEdges.begin(), candidateEdges.end(), [&](int a, int b) {
        if (ew[a] != ew[b]) return ew[a] < ew[b];
        if (eu[a] != eu[b]) return eu[a] < eu[b];
        return ev[a] < ev[b];
    });

    DSU dsu(n);
    std::vector<int> chosen;
    chosen.reserve(std::min(n - 1, static_cast<int>(candidateEdges.size())));
    int64_t total = 0;
    for (int id : candidateEdges) {
        if (dsu.unite(eu[id], ev[id])) {
            chosen.push_back(id);
            total += ew[id];
            if (static_cast<int>(chosen.size()) == n - 1) break;
        }
    }

    // Validate that all terminals are connected in the chosen forest; if not, connect components with extra paths.
    auto all_connected = [&]() -> bool {
        int root = dsu.find(terminals[0]);
        for (int i = 1; i < k; ++i) {
            if (dsu.find(terminals[i]) != root) return false;
        }
        return true;
    };

    if (!all_connected()) {
        // Add shortest paths between terminal components (greedy).
        // Repeat until connected or no progress.
        std::vector<char> inChosen(m, 0);
        for (int id : chosen) inChosen[id] = 1;

        for (int iter = 0; iter < k && !all_connected(); ++iter) {
            // pick two terminals in different components
            int a = -1, b = -1;
            int ra = -1;
            for (int i = 0; i < k && a == -1; ++i) {
                int r = dsu.find(terminals[i]);
                for (int j = i + 1; j < k; ++j) {
                    if (dsu.find(terminals[j]) != r) {
                        a = terminals[i];
                        b = terminals[j];
                        ra = r;
                        break;
                    }
                }
            }
            if (a == -1) break;

            bool ok = dijkstra_path_edges(n, adj, a, b, parentV, parentE);
            if (!ok) break;
            int cur = b;
            while (cur != a && cur != -1) {
                int pe = parentE[cur];
                if (pe >= 0 && !inChosen[pe]) {
                    inChosen[pe] = 1;
                }
                cur = parentV[cur];
            }

            // Re-run Kruskal on the expanded chosen edge set to ensure acyclic.
            std::vector<int> edges2;
            edges2.reserve(m);
            for (int id = 0; id < m; ++id) {
                if (inChosen[id]) edges2.push_back(id);
            }
            std::sort(edges2.begin(), edges2.end(), [&](int x, int y) {
                if (ew[x] != ew[y]) return ew[x] < ew[y];
                if (eu[x] != eu[y]) return eu[x] < eu[y];
                return ev[x] < ev[y];
            });
            dsu = DSU(n);
            chosen.clear();
            total = 0;
            for (int id : edges2) {
                if (dsu.unite(eu[id], ev[id])) {
                    chosen.push_back(id);
                    total += ew[id];
                }
            }
        }
    }

    std::cout << total << " " << chosen.size() << "\n";
    for (int id : chosen) {
        std::cout << eu[id] << " " << ev[id] << "\n";
    }
    return 0;
}
