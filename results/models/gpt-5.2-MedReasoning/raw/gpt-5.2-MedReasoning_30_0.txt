#!/usr/bin/env python3
import sys
import os
import struct
import tempfile
from array import array


def _iter_ints_from_stdin():
    # Fast-ish integer parser over bytes from stdin.
    data = sys.stdin.buffer.read()
    num = 0
    sign = 1
    in_num = False
    for b in data:
        if 48 <= b <= 57:
            num = num * 10 + (b - 48)
            in_num = True
        else:
            if b == 45:  # '-'
                sign = -1
            elif in_num:
                yield sign * num
                num = 0
                sign = 1
                in_num = False
            else:
                sign = 1
    if in_num:
        yield sign * num


def solve():
    it = _iter_ints_from_stdin()
    try:
        n = next(it)
        m = next(it)
    except StopIteration:
        return

    # Store clauses on disk (3 int32 per clause) so we can do 1-2 linear passes
    # without keeping all clauses in memory.
    tmp = tempfile.NamedTemporaryFile(prefix="sat3_", suffix=".bin", delete=False)
    tmp_path = tmp.name

    # diff[i] > 0 suggests xi=True, diff[i] < 0 suggests xi=False.
    diff = array("i", [0]) * (n + 1)

    pack = struct.Struct("<iii").pack
    write = tmp.write

    # Read clauses, update diffs, and write to temp.
    for _ in range(m):
        try:
            a = next(it); b = next(it); c = next(it)
        except StopIteration:
            break
        write(pack(a, b, c))
        va = a if a >= 0 else -a
        vb = b if b >= 0 else -b
        vc = c if c >= 0 else -c
        diff[va] += 1 if a > 0 else -1
        diff[vb] += 1 if b > 0 else -1
        diff[vc] += 1 if c > 0 else -1

    tmp.flush()
    tmp.close()

    # Initial assignment from majority vote over literal signs.
    # assignment[i] == 1 means True.
    assignment = bytearray(n + 1)
    for i in range(1, n + 1):
        assignment[i] = 1 if diff[i] >= 0 else 0

    # Local search refinement: a couple of linear passes, flipping one literal
    # from each currently-unsatisfied clause.
    # Deterministic xorshift RNG (no imports, reproducible).
    rng = 2463534242

    def rand_u32():
        nonlocal rng
        rng ^= (rng << 13) & 0xFFFFFFFF
        rng ^= (rng >> 17) & 0xFFFFFFFF
        rng ^= (rng << 5) & 0xFFFFFFFF
        return rng & 0xFFFFFFFF

    unpack_from = struct.Struct("<iii").unpack_from

    def lit_value(lit: int) -> int:
        v = lit if lit >= 0 else -lit
        val = assignment[v]
        return val if lit > 0 else (1 - val)

    # For very large instances, keep passes small to stay within time.
    # Two passes is often enough for planted/random-sat style instances.
    max_passes = 2
    clause_bytes = 12
    try:
        with open(tmp_path, "rb", buffering=1024 * 1024) as f:
            for _pass in range(max_passes):
                f.seek(0)
                buf = f.read(1024 * 1024 * 8)
                offset = 0
                while buf:
                    blen = len(buf)
                    # process full clauses in this buffer
                    limit = blen - (blen % clause_bytes)
                    while offset < limit:
                        a, b, c = unpack_from(buf, offset)
                        offset += clause_bytes
                        if (lit_value(a) | lit_value(b) | lit_value(c)) == 0:
                            # flip one random literal's variable to satisfy it
                            r = rand_u32() % 3
                            lit = a if r == 0 else (b if r == 1 else c)
                            v = lit if lit >= 0 else -lit
                            assignment[v] = 1 if lit > 0 else 0
                    # carry remainder to next read
                    rem = buf[limit:]
                    buf = f.read(1024 * 1024 * 8)
                    if rem:
                        buf = rem + buf
                    offset = 0
    finally:
        try:
            os.remove(tmp_path)
        except OSError:
            pass

    out = sys.stdout.write
    out("SAT\n")
    # Output 1/0 assignment for variables 1..n
    # Chunked write to avoid huge join memory.
    CHUNK = 1_000_000
    parts = []
    for i in range(1, n + 1):
        parts.append("1" if assignment[i] else "0")
        if i % CHUNK == 0:
            out(" ".join(parts))
            out(" " if i != n else "")
            parts.clear()
    if parts:
        out(" ".join(parts))
    out("\n")


if __name__ == "__main__":
    solve()
