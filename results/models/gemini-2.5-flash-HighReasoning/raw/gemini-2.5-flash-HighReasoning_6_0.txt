using System;
using System.Collections.Generic;
using System.Linq;

public class OTSPSolver
{
    // Physics constants
    private const double MU_EARTH = 398600.4418; // km^3/s^2

    public struct Vector3
    {
        public double X, Y, Z;

        public Vector3(double x, double y, double z)
        {
            X = x;
            Y = y;
            Z = z;
        }

        public double Magnitude => Math.Sqrt(X * X + Y * Y + Z * Z);
        public Vector3 Normalize() => this / Magnitude;

        public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vector3 operator *(Vector3 a, double scalar) => new Vector3(a.X * scalar, a.Y * scalar, a.Z * scalar);
        public static Vector3 operator *(double scalar, Vector3 a) => new Vector3(a.X * scalar, a.Y * scalar, a.Z * scalar);
        public static Vector3 operator /(Vector3 a, double scalar) => new Vector3(a.X / scalar, a.Y / scalar, a.Z / scalar);
        public static double Dot(Vector3 a, Vector3 b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vector3 Cross(Vector3 a, Vector3 b) => new Vector3(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    public struct OrbitState
    {
        public Vector3 Position; // km
        public Vector3 Velocity; // km/s

        public OrbitState(Vector3 position, Vector3 velocity)
        {
            Position = position;
            Velocity = velocity;
        }

        // Calculate specific energy
        public double SpecificEnergy => (0.5 * Velocity.Magnitude * Velocity.Magnitude) - (MU_EARTH / Position.Magnitude);

        // Calculate semi-major axis
        public double SemiMajorAxis
        {
            get
            {
                double r = Position.Magnitude;
                double v = Velocity.Magnitude;
                // Handle edge cases for a and ensure it's not near-zero denominator
                double denominator = (2.0 / r) - (v * v / MU_EARTH);
                if (Math.Abs(denominator) < 1e-10) // Near parabolic/hyperbolic, semi-major axis tends to infinity
                {
                    return double.PositiveInfinity;
                }
                return 1.0 / denominator;
            }
        }
        
        // Calculate specific angular momentum
        public Vector3 SpecificAngularMomentum => Vector3.Cross(Position, Velocity);
    }

    public static double CalculateDeltaV(OrbitState start, OrbitState target)
    {
        // Approximation for delta-V, considering changes in both energy and direction.
        // This is a heuristic to provide "reasonable delta-V estimates" without solving Lambert's problem.

        // 1. Delta-V for Energy Change (to match target semi-major axis at current radius)
        double r_start_mag = start.Position.Magnitude;
        double v_start_mag = start.Velocity.Magnitude;

        // Velocity required at r_start_mag to be on an orbit with target.SemiMajorAxis
        // Using Vis-Viva equation: v = sqrt(mu * (2/r - 1/a))
        double v_target_sma_at_start_r;
        if (double.IsPositiveInfinity(target.SemiMajorAxis) || r_start_mag == 0) // Parabolic or Hyperbolic, assume high energy transfer, or if at origin
        {
            // For highly energetic target orbits, we assume a burn to escape velocity
            // Or if start position is origin, this formula doesn't make sense. Fallback.
            v_target_sma_at_start_r = Math.Sqrt(2 * MU_EARTH / r_start_mag); 
        }
        else if (target.SemiMajorAxis <= 0) // Highly degenerate or impossible orbit, avoid division by zero or negative root
        {
            // This case should ideally not happen for valid orbital states, but for robustness
            // We can assume a very high cost or a simple velocity difference if semi-major axis is non-positive
            v_target_sma_at_start_r = target.Velocity.Magnitude; // Fallback: just match target velocity magnitude
        }
        else
        {
            double arg = (2.0 / r_start_mag - 1.0 / target.SemiMajorAxis);
            if (arg < 0) // Cannot achieve target semi-major axis at this radius, implies hyperbolic escape required
            {
                 v_target_sma_at_start_r = Math.Sqrt(2 * MU_EARTH / r_start_mag); // Assume escape velocity
            }
            else
            {
                 v_target_sma_at_start_r = Math.Sqrt(MU_EARTH * arg);
            }
           
        }

        double dV_energy = Math.Abs(v_target_sma_at_start_r - v_start_mag);

        // 2. Delta-V for Directional/Plane Change
        // Approximated by the angle between the normalized velocity vectors.
        // This is a simple approximation and doesn't account for complex plane changes.
        double dotProduct = Vector3.Dot(start.Velocity.Normalize(), target.Velocity.Normalize());
        // Ensure dotProduct is within valid range for Acos due to potential floating point inaccuracies
        if (dotProduct > 1.0) dotProduct = 1.0;
        if (dotProduct < -1.0) dotProduct = -1.0;

        double angle = Math.Acos(dotProduct);
        double dV_direction = 2 * start.Velocity.Magnitude * Math.Sin(angle / 2.0); // Tsiolkovsky rocket equation simplified for plane change

        return dV_energy + dV_direction;
    }

    public static void Main(string[] args)
    {
        // Read input
        string[] line1 = Console.ReadLine().Split(' ');
        int numStations = int.Parse(line1[0]);
        // double mu = double.Parse(line1[1]); // Using fixed MU_EARTH

        List<OrbitState> stations = new List<OrbitState>();

        // Start orbit
        string[] startOrbitLine = Console.ReadLine().Split(' ');
        Vector3 startPos = new Vector3(double.Parse(startOrbitLine[2]), double.Parse(startOrbitLine[3]), double.Parse(startOrbitLine[4]));
        Vector3 startVel = new Vector3(double.Parse(startOrbitLine[5]), double.Parse(startOrbitLine[6]), double.Parse(startOrbitLine[7]));
        OrbitState startState = new OrbitState(startPos, startVel);
        stations.Add(startState); // Add start as station 0

        // Station orbits
        for (int i = 0; i < numStations; i++)
        {
            string[] stationOrbitLine = Console.ReadLine().Split(' ');
            // The input format might have "station orbit:" or similar prefix.
            // Assuming the actual numerical data starts from index 2 based on previous interpretation.
            Vector3 pos = new Vector3(double.Parse(stationOrbitLine[2]), double.Parse(stationOrbitLine[3]), double.Parse(stationOrbitLine[4]));
            Vector3 vel = new Vector3(double.Parse(stationOrbitLine[5]), double.Parse(stationOrbitLine[6]), double.Parse(stationOrbitLine[7]));
            stations.Add(new OrbitState(pos, vel));
        }

        // Create the full list of orbits, including the start orbit at index 0
        List<OrbitState> allOrbits = new List<OrbitState>();
        allOrbits.Add(startState); // Station 0 is the start orbit
        allOrbits.AddRange(stations); // Stations 1 to numStations are the target stations

        // Calculate Delta-V Cost Matrix
        double[,] deltaVCostMatrix = new double[numStations + 1, numStations + 1];
        for (int i = 0; i < numStations + 1; i++)
        {
            for (int j = 0; j < numStations + 1; j++)
            {
                if (i == j)
                {
                    deltaVCostMatrix[i, j] = 0; // Cost to go to itself is zero
                }
                else
                {
                    deltaVCostMatrix[i, j] = CalculateDeltaV(allOrbits[i], allOrbits[j]);
                }
            }
        }

        // --- TSP Solver ---
        List<int> bestPath = new List<int>();
        double minTotalDeltaV = double.MaxValue;

        const int MAX_BRUTE_FORCE_N = 10; // Threshold for brute-force vs. heuristic

        if (numStations <= MAX_BRUTE_FORCE_N)
        {
            // Brute-force for small number of stations
            // Stations to visit are 1 to numStations
            int[] stationIndicesToVisit = Enumerable.Range(1, numStations).ToArray();
            
            FindBestPathBruteForce(0, stationIndicesToVisit, deltaVCostMatrix, new List<int> { 0 }, 0, ref bestPath, ref minTotalDeltaV);
        }
        else
        {
            // Nearest Neighbor heuristic for larger number of stations
            bestPath = FindBestPathNearestNeighbor(0, numStations, deltaVCostMatrix);
        }
        
        // Output the result
        Console.WriteLine(string.Join(" ", bestPath));
    }

    // --- TSP Solver Implementations ---

    // Brute-force TSP for small number of stations (exact solution)
    private static void FindBestPathBruteForce(int currentStation, int[] remainingStations, double[,] costMatrix, List<int> currentPath, double currentTotalDeltaV, ref List<int> bestPath, ref double minTotalDeltaV)
    {
        if (remainingStations.Length == 0)
        {
            // All stations visited, check if this path is better
            if (currentTotalDeltaV < minTotalDeltaV)
            {
                minTotalDeltaV = currentTotalDeltaV;
                bestPath = new List<int>(currentPath);
            }
            return;
        }

        for (int i = 0; i < remainingStations.Length; i++)
        {
            int nextStation = remainingStations[i];
            double cost = costMatrix[currentStation, nextStation];

            List<int> newRemainingStations = remainingStations.Where((val, idx) => idx != i).ToList().ToArray();
            List<int> newPath = new List<int>(currentPath) { nextStation };

            FindBestPathBruteForce(nextStation, newRemainingStations, costMatrix, newPath, currentTotalDeltaV + cost, ref bestPath, ref minTotalDeltaV);
        }
    }

    // Nearest Neighbor TSP heuristic for larger number of stations
    private static List<int> FindBestPathNearestNeighbor(int startNode, int numStations, double[,] costMatrix)
    {
        List<int> path = new List<int>();
        bool[] visited = new bool[numStations + 1]; // +1 for the start orbit

        int currentNode = startNode;
        path.Add(currentNode);
        visited[currentNode] = true;

        for (int i = 0; i < numStations; i++) // We need to visit numStations other stations
        {
            double minCost = double.MaxValue;
            int nextNode = -1;

            // Find the nearest unvisited neighbor
            for (int j = 1; j < numStations + 1; j++) // Stations are 1 to numStations
            {
                if (!visited[j])
                {
                    if (costMatrix[currentNode, j] < minCost)
                    {
                        minCost = costMatrix[currentNode, j];
                        nextNode = j;
                    }
                }
            }

            if (nextNode != -1)
            {
                path.Add(nextNode);
                visited[nextNode] = true;
                currentNode = nextNode;
            }
            else
            {
                // This case should ideally not happen if there are unvisited nodes
                // but adding a break to prevent infinite loop in case of logic error.
                break; 
            }
        }
        return path;
    }
}
