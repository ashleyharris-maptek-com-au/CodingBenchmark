using System;
using System.Collections.Generic;
using System.Linq;

public class ChinesePostmanSolver
{
    private int N; // Number of nodes
    private List<(int to, int weight)>[] initialGraphAdj; // Stores the original graph structure for Dijkstra's
    private List<(int from, int to, int weight)> originalEdgesList; // Stores all original edges

    public ChinesePostmanSolver(int n)
    {
        N = n;
        initialGraphAdj = new List<(int to, int weight)>[N];
        for (int i = 0; i < N; i++)
        {
            initialGraphAdj[i] = new List<(int to, int weight)>();
        }
        originalEdgesList = new List<(int from, int to, int weight)>();
    }

    public void AddEdge(int u, int v, int w)
    {
        initialGraphAdj[u].Add((v, w));
        initialGraphAdj[v].Add((u, w)); // Undirected graph
        originalEdgesList.Add((u, v, w)); // Store for degree calculation and total weight
    }

    // Dijkstra's algorithm to find shortest paths and predecessors from a source node
    private (Dictionary<int, int> distances, Dictionary<int, int> predecessors) Dijkstra(int startNode)
    {
        var distances = new Dictionary<int, int>();
        var predecessors = new Dictionary<int, int>();

        for (int i = 0; i < N; i++)
        {
            distances[i] = int.MaxValue;
        }
        distances[startNode] = 0;

        var pq = new PriorityQueue<int, int>();
        pq.Enqueue(startNode, 0);

        while (pq.Count > 0)
        {
            var u = pq.Dequeue();
            var d = distances[u];

            // If we found a shorter path to 'u' since it was enqueued, ignore this entry
            if (d > distances[u]) continue; 

            foreach (var edge in initialGraphAdj[u]) // Use initialGraphAdj
            {
                var v = edge.to;
                var weight = edge.weight;

                // Check for overflow before adding to prevent `distances[v]` from becoming negative
                // Also, ensure `distances[u]` is not `int.MaxValue` which means unreachable
                if (distances[u] != int.MaxValue && (long)distances[u] + weight < distances[v]) 
                {
                    distances[v] = distances[u] + weight;
                    predecessors[v] = u;
                    pq.Enqueue(v, distances[v]);
                }
            }
        }
        return (distances, predecessors);
    }

    // Finds minimum weight perfect matching for odd degree vertices
    // This uses a recursive backtracking approach, suitable for small to medium V_odd
    private List<(int u, int v)> FindMinWeightPerfectMatching(List<int> oddDegreeVertices, Dictionary<int, (Dictionary<int, int> distances, Dictionary<int, int> predecessors)> dijkstraResults)
    {
        var matching = new List<(int u, int v)>();
        if (oddDegreeVertices.Count == 0)
        {
            return matching;
        }

        long minCost = long.MaxValue;
        List<(int u, int v)> bestMatching = new List<(int u, int v)>();

        // Recursive function to find matching
        // `currentMatching` stores the pairs already matched
        // `unmatchedVertices` is a boolean array indicating if a vertex is unmatched
        void Solve(List<(int u, int v)> currentMatching, bool[] unmatchedVertices, long currentCost)
        {
            if (currentCost >= minCost) return; // Pruning: if current path is already more expensive, stop

            int firstUnmatchedIdx = -1;
            for (int i = 0; i < unmatchedVertices.Length; i++)
            {
                if (unmatchedVertices[i])
                {
                    firstUnmatchedIdx = i;
                    break;
                }
            }

            if (firstUnmatchedIdx == -1) // All vertices matched
            {
                if (currentCost < minCost)
                {
                    minCost = currentCost;
                    bestMatching = new List<(int u, int v)>(currentMatching);
                }
                return;
            }

            unmatchedVertices[firstUnmatchedIdx] = false; // Mark as matched

            int u_odd = oddDegreeVertices[firstUnmatchedIdx];

            // Try to match u_odd with every other unmatched vertex
            for (int i = firstUnmatchedIdx + 1; i < unmatchedVertices.Length; i++)
            {
                if (unmatchedVertices[i])
                {
                    unmatchedVertices[i] = false; // Mark as matched
                    int v_odd = oddDegreeVertices[i];
                    int pathCost = dijkstraResults[u_odd].distances[v_odd]; // Cost between u_odd and v_odd

                    currentMatching.Add((u_odd, v_odd));
                    Solve(currentMatching, unmatchedVertices, currentCost + pathCost);
                    currentMatching.RemoveAt(currentMatching.Count - 1); // Backtrack
                    unmatchedVertices[i] = true; // Unmark
                }
            }
            unmatchedVertices[firstUnmatchedIdx] = true; // Unmark
        }

        // Initialize unmatchedVertices array
        bool[] unmatched = new bool[oddDegreeVertices.Count];
        for (int i = 0; i < oddDegreeVertices.Count; i++)
        {
            unmatched[i] = true;
        }

        Solve(new List<(int u, int v)>(), unmatched, 0);

        return bestMatching;
    }

    // Hierholzer's Algorithm to find Eulerian Circuit
    // Operates on the combined graph (original + duplicated shortest paths)
    private List<int> FindEulerianCircuit(
        int startNode,
        List<Tuple<int, int, int>> allGraphEdges, // (from, to, weight) - represents all edges in the Eulerian graph
        List<Tuple<int, int, int>>[] adjForEuler // (neighbor, edgeIndex, weight) - adjacency list for Eulerian graph
    )
    {
        // Make a mutable copy of the adjacency list for Hierholzer's
        // Each node's list will contain (neighbor, global_edge_index, weight)
        var currentAdj = new List<Tuple<int, int, int>>[N];
        for (int i = 0; i < N; i++)
        {
            currentAdj[i] = new List<Tuple<int, int, int>>(adjForEuler[i]);
        }

        Stack<int> currentPath = new Stack<int>();
        List<int> circuit = new List<int>();

        currentPath.Push(startNode);
        int currentNode = startNode;

        // Use a boolean array to track if an edge (by its global index) has been used
        bool[] isEdgeUsed = new bool[allGraphEdges.Count];

        while (currentPath.Count > 0)
        {
            int edgeToTakeIndex = -1;
            int nextNode = -1;

            // Find an unused edge from the currentNode
            // We iterate backwards to efficiently remove elements from the list without invalidating indices
            for (int i = currentAdj[currentNode].Count - 1; i >= 0; i--)
            {
                var edgeInfo = currentAdj[currentNode][i]; // (neighbor, edgeIndex, weight)
                if (!isEdgeUsed[edgeInfo.Item2])
                {
                    edgeToTakeIndex = edgeInfo.Item2;
                    nextNode = edgeInfo.Item1;
                    
                    // Remove this specific edge entry from currentAdj[currentNode]
                    // This is for efficiency so we don't re-check this edge for this node
                    currentAdj[currentNode].RemoveAt(i); 
                    break;
                }
            }

            if (edgeToTakeIndex != -1) // Found an unused edge
            {
                isEdgeUsed[edgeToTakeIndex] = true;
                currentPath.Push(currentNode);
                currentNode = nextNode;
            }
            else // No unused edges from currentNode, backtrack
            {
                circuit.Add(currentNode);
                currentNode = currentPath.Pop();
            }
        }
        circuit.Reverse(); // The circuit is built in reverse order during backtracking
        return circuit;
    }

    public (List<int> circuit, long totalLength) Solve()
    {
        // 1. Calculate degrees for initial graph and identify odd-degree vertices
        var degrees = new int[N];
        foreach (var edge in originalEdgesList)
        {
            degrees[edge.from]++;
            degrees[edge.to]++;
        }

        var oddDegreeVertices = new List<int>();
        for (int i = 0; i < N; i++)
        {
            if (degrees[i] % 2 != 0)
            {
                oddDegreeVertices.Add(i);
            }
        }

        // 2. Find shortest paths between all pairs of odd-degree vertices
        var dijkstraResults = new Dictionary<int, (Dictionary<int, int> distances, Dictionary<int, int> predecessors)>();
        foreach (var u_odd in oddDegreeVertices)
        {
            dijkstraResults[u_odd] = Dijkstra(u_odd);
        }

        // 3. Find Minimum Weight Perfect Matching for odd-degree vertices
        var matchingEdges = FindMinWeightPerfectMatching(oddDegreeVertices, dijkstraResults);

        // 4. Construct the Eulerian graph: Combine original edges with duplicated shortest path edges
        List<Tuple<int, int, int>> allGraphEdges = new List<Tuple<int, int, int>>();
        // Add all original edges first
        foreach (var edge in originalEdgesList)
        {
            allGraphEdges.Add(Tuple.Create(edge.from, edge.to, edge.weight));
        }

        long addedPathWeight = 0;
        foreach (var match in matchingEdges)
        {
            int u = match.u;
            int v = match.v;

            addedPathWeight += dijkstraResults[u].distances[v];

            // Reconstruct path from u to v using predecessors from Dijkstra starting at u
            var path = new List<int>();
            int current = v;
            while (current != u)
            {
                path.Insert(0, current); // Insert at beginning to build path in correct order
                current = dijkstraResults[u].predecessors[current];
            }
            path.Insert(0, u);

            // Add the edges along this shortest path to `allGraphEdges` (effectively duplicating them)
            for (int i = 0; i < path.Count - 1; i++)
            {
                int node1 = path[i];
                int node2 = path[i + 1];

                // Find the weight of the original edge (node1, node2) from the `initialGraphAdj`
                int edgeWeight = -1;
                foreach (var adjEdge in initialGraphAdj[node1])
                {
                    if (adjEdge.to == node2)
                    {
                        edgeWeight = adjEdge.weight;
                        break;
                    }
                }
                // Add this edge (which is a segment of the shortest path) to the list of all edges
                allGraphEdges.Add(Tuple.Create(node1, node2, edgeWeight));
            }
        }

        // Calculate initial total weight (sum of all original edge weights)
        long initialTotalTotalWeight = 0;
        foreach (var edge in originalEdgesList)
        {
            initialTotalTotalWeight += edge.weight;
        }

        // 5. Build the adjacency list for Hierholzer's using the combined edge list (`allGraphEdges`)
        List<Tuple<int, int, int>>[] adjForEuler = new List<Tuple<int, int, int>>[N];
        for(int i=0; i<N; i++) adjForEuler[i] = new List<Tuple<int, int, int>>();

        for(int i=0; i<allGraphEdges.Count; i++)
        {
            var edge = allGraphEdges[i];
            // Store (neighbor, global_edge_index, weight)
            adjForEuler[edge.Item1].Add(Tuple.Create(edge.Item2, i, edge.Item3));
            adjForEuler[edge.Item2].Add(Tuple.Create(edge.Item1, i, edge.Item3)); // Undirected graph
        }

        // 6. Find Eulerian Circuit starting from node 0
        List<int> circuit = FindEulerianCircuit(0, allGraphEdges, adjForEuler);

        // 7. Calculate total length of the circuit
        // The total length is the sum of all original edge weights plus the sum of added path weights.
        long totalLength = initialTotalTotalWeight + addedPathWeight;

        return (circuit, totalLength);
    }

    // A custom PriorityQueue implementation for Dijkstra's algorithm
    // It's a min-heap based on the priority (distance)
    public class PriorityQueue<TElement, TPriority>
    {
        private List<(TElement element, TPriority priority)> heap = new List<(TElement element, TPriority priority)>();
        private Comparer<TPriority> comparer = Comparer<TPriority>.Default;

        public int Count => heap.Count;

        // Adds an element with its priority to the priority queue
        public void Enqueue(TElement element, TPriority priority)
        {
            heap.Add((element, priority));
            BubbleUp(heap.Count - 1); // Maintain heap property
        }

        // Removes and returns the element with the smallest priority
        public TElement Dequeue()
        {
            if (heap.Count == 0) throw new InvalidOperationException("Priority queue is empty.");

            TElement result = heap[0].element; // Element with smallest priority is at the root
            int lastIndex = heap.Count - 1;
            heap[0] = heap[lastIndex]; // Move last element to root
            heap.RemoveAt(lastIndex); // Remove last element

            if (heap.Count > 0)
            {
                SinkDown(0); // Maintain heap property
            }
            return result;
        }

        // Moves an element up the heap to restore heap property
        private void BubbleUp(int index)
        {
            while (index > 0)
            {
                int parentIndex = (index - 1) / 2;
                // If current element's priority is smaller than its parent's, swap
                if (comparer.Compare(heap[index].priority, heap[parentIndex].priority) < 0)
                {
                    Swap(index, parentIndex);
                    index = parentIndex;
                }
                else
                {
                    break;
                }
            }
        }

        // Moves an element down the heap to restore heap property
        private void SinkDown(int index)
        {
            int lastIndex = heap.Count - 1;
            while (true)
            {
                int leftChildIndex = 2 * index + 1;
                int rightChildIndex = 2 * index + 2;
                int smallestChildIndex = index;

                // Find the child with the smallest priority
                if (leftChildIndex <= lastIndex && comparer.Compare(heap[leftChildIndex].priority, heap[smallestChildIndex].priority) < 0)
                {
                    smallestChildIndex = leftChildIndex;
                }

                if (rightChildIndex <= lastIndex && comparer.Compare(heap[rightChildIndex].priority, heap[smallestChildIndex].priority) < 0)
                {
                    smallestChildIndex = rightChildIndex;
                }

                // If the smallest child is not the current element, swap and continue sinking
                if (smallestChildIndex != index)
                {
                    Swap(index, smallestChildIndex);
                    index = smallestChildIndex;
                }
                else
                {
                    break; // Heap property restored
                }
            }
        }

        // Helper method to swap two elements in the heap
        private void Swap(int i, int j)
        {
            var temp = heap[i];
            heap[i] = heap[j];
            heap[j] = temp;
        }
    }

    public static void Main(string[] args)
    {
        // Read graph input from standard input
        string[] line1 = Console.ReadLine().Split(' ');
        int N = int.Parse(line1[0]); // Number of nodes
        int M = int.Parse(line1[1]); // Number of edges

        var solver = new ChinesePostmanSolver(N);

        for (int i = 0; i < M; i++)
        {
            string[] edgeLine = Console.ReadLine().Split(' ');
            int u = int.Parse(edgeLine[0]);
            int v = int.Parse(edgeLine[1]);
            int w = int.Parse(edgeLine[2]);
            solver.AddEdge(u, v, w);
        }

        // Solve the Chinese Postman Problem
        (List<int> circuit, long totalLength) result = solver.Solve();

        // Output the total length followed by the circuit nodes
        Console.WriteLine($"{result.totalLength} {string.Join(" ", result.circuit)}");
    }
}
