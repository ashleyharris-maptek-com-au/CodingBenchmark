// Basic HLSL Pixel Shader for Vulkan

// Input structure from the vertex shader
struct PSInput
{
    float4 Position : SV_POSITION;
    // Add other interpolated values here, e.g.,
    // float3 Color : COLOR0;
    // float2 TexCoord : TEXCOORD0;
};

// Output structure for the fragment color
struct PSOutput
{
    float4 Color : SV_TARGET0;
};

// Main pixel shader function
PSOutput main(PSInput input)
{
    PSOutput output;
    // For a basic passthrough, we can just output a fixed color or use interpolated values
    output.Color = float4(1.0f, 0.0f, 0.0f, 1.0f); // Red color
    // If you had an interpolated color from the vertex shader:
    // output.Color = float4(input.Color, 1.0f);
    return output;
}
