// Structure to represent a plane
struct Plane {
    Vec3 normal;
    Vec3 point_on_plane; // Any point on the plane

    Plane(const Vec3& p1, const Vec3& p2, const Vec3& p3) {
        normal = (p2 - p1).cross(p3 - p1).normalize();
        point_on_plane = p1;
    }

    // Distance from a point to the plane
    double distance(const Vec3& p) const {
        return normal.dot(p - point_on_plane);
    }
};

// Forward declaration for intersection functions
// Line-plane intersection
inline Vec3 line_plane_intersection(const Vec3& p1, const Vec3& p2, const Plane& plane) {
    Vec3 direction = p2 - p1;
    double denominator = plane.normal.dot(direction);

    // If denominator is close to zero, line is parallel to plane
    if (std::abs(denominator) < EPSILON) {
        // Handle parallel case, ideally this shouldn't happen for intersecting lines
        // or caller should handle it. For now, return p1.
        return p1; 
    }

    double t = -plane.distance(p1) / denominator;
    return p1 + direction * t;
}

// Clips a triangle by a plane. Returns new triangles that are on the 'positive' side of the plane.
// 'clipped_triangles' will contain the triangles that remain.
inline void clip_triangle_by_plane(const Triangle& tri, const Plane& plane, std::vector<Triangle>& clipped_triangles) {
    std::vector<Vec3> poly_vertices; // Vertices forming the clipped polygon
    std::vector<PointClassification> classifications(3);
    Vec3 vertices[3] = {tri.v0, tri.v1, tri.v2};

    for (int i = 0; i < 3; ++i) {
        classifications[i] = classify_point_to_plane(vertices[i], plane.normal, plane.point_on_plane);
    }

    for (int i = 0; i < 3; ++i) {
        int j = (i + 1) % 3; // Next vertex in the triangle
        
        PointClassification current_class = classifications[i];
        PointClassification next_class = classifications[j];

        if (current_class == PointClassification::OUTSIDE || current_class == PointClassification::ON_PLANE) {
            poly_vertices.push_back(vertices[i]);
        }

        if (current_class != next_class && current_class != PointClassification::ON_PLANE && next_class != PointClassification::ON_PLANE) {
            // Edge crosses the plane
            Vec3 intersection_point = line_plane_intersection(vertices[i], vertices[j], plane);
            poly_vertices.push_back(intersection_point);
        }
    }

    // Triangulate the resulting polygon (can be a triangle or a quadrilateral)
    if (poly_vertices.size() >= 3) {
        // First triangle uses the first three vertices
        clipped_triangles.emplace_back(poly_vertices[0], poly_vertices[1], poly_vertices[2]);

        // If it's a quadrilateral, form a second triangle
        if (poly_vertices.size() == 4) {
            clipped_triangles.emplace_back(poly_vertices[0], poly_vertices[2], poly_vertices[3]);
        }
    }
}

enum class PointClassification {
    INSIDE,    // Point is on the "negative" side of the plane
    OUTSIDE,   // Point is on the "positive" side of the plane
    ON_PLANE   // Point is on the plane
};

// Classifies a point with respect to a plane (defined by normal and a point on the plane)
inline PointClassification classify_point_to_plane(const Vec3& point, const Vec3& plane_normal, const Vec3& point_on_plane) {
    double distance = plane_normal.dot(point - point_on_plane);
    if (distance > EPSILON) {
        return PointClassification::OUTSIDE;
    } else if (distance < -EPSILON) {
        return PointClassification::INSIDE;
    } else {
        return PointClassification::ON_PLANE;
    }
}

// Ray-AABB intersection
inline bool intersect_ray_aabb(const Ray& ray, const BoundingBox& bbox, double& t_min_out, double& t_max_out) {
    t_min_out = -std::numeric_limits<double>::infinity();
    t_max_out = std::numeric_limits<double>::infinity();

    for (int i = 0; i < 3; ++i) {
        double inv_d = 1.0 / (&ray.direction.x)[i];
        double t0 = ((&bbox.min_point.x)[i] - (&ray.origin.x)[i]) * inv_d;
        double t1 = ((&bbox.max_point.x)[i] - (&ray.origin.x)[i]) * inv_d;

        if (inv_d < 0.0) std::swap(t0, t1);

        t_min_out = std::max(t0, t_min_out);
        t_max_out = std::min(t1, t_max_out);

        if (t_max_out < t_min_out) return false;
    }
    return t_max_out > EPSILON; // Ensure intersection is not behind the ray origin
}


// Ray-Triangle intersection using Möller–Trumbore algorithm
// Returns true if intersection occurs, updates t, u, v with barycentric coordinates
inline bool intersect_ray_triangle(const Ray& ray, const Triangle& tri, double& t_out, double& u_out, double& v_out) {
    Vec3 edge1 = tri.v1 - tri.v0;
    Vec3 edge2 = tri.v2 - tri.v0;
    Vec3 pvec = ray.direction.cross(edge2);
    double det = edge1.dot(pvec);

    // If det is close to 0, ray is parallel to triangle or lies in its plane
    if (std::abs(det) < EPSILON) return false;

    double inv_det = 1.0 / det;
    Vec3 tvec = ray.origin - tri.v0;
    u_out = tvec.dot(pvec) * inv_det;
    if (u_out < 0.0 || u_out > 1.0) return false;

    Vec3 qvec = tvec.cross(edge1);
    v_out = ray.direction.dot(qvec) * inv_det;
    if (v_out < 0.0 || u_out + v_out > 1.0) return false;

    t_out = edge2.dot(qvec) * inv_det;

    return t_out > EPSILON; // Intersection must be in front of the ray origin
}

// Point-in-mesh test using ray casting
// Counts intersections with triangles to determine if a point is inside.
inline bool is_point_inside_mesh(const Vec3& point, const AABBTree& tree) {
    // Generate a random ray direction to avoid issues with rays hitting edges/vertices
    static std::random_device rd;
    static std::mt19937 gen(rd());
    static std::uniform_real_distribution<> dis(-1.0, 1.0);

    Vec3 random_direction;
    // Ensure the direction is not a zero vector
    do {
        random_direction = Vec3(dis(gen), dis(gen), dis(gen));
    } while (random_direction.length() < EPSILON);

    Ray ray(point, random_direction.normalize());

    int intersection_count = 0;

    // Traverse AABB tree for ray-triangle intersections
    std::vector<int> nodes_to_visit;
    if (!tree.nodes.empty()) {
        nodes_to_visit.push_back(0); // Start with root
    }

    while (!nodes_to_visit.empty()) {
        int current_node_idx = nodes_to_visit.back();
        nodes_to_visit.pop_back();

        const AABBTreeNode& node = tree.nodes[current_node_idx];

        double t_min_aabb, t_max_aabb;
        if (!intersect_ray_aabb(ray, node.bbox, t_min_aabb, t_max_aabb)) {
            continue; // Ray misses AABB
        }

        if (node.isLeaf()) {
            for (int tri_idx : node.triangle_indices) {
                double t, u, v;
                if (intersect_ray_triangle(ray, tree.triangles[tri_idx], t, u, v)) {
                    intersection_count++;
                }
            }
        } else {
            nodes_to_visit.push_back(node.left_child);
            nodes_to_visit.push_back(node.right_child);
        }
    }

    return intersection_count % 2 == 1; // Odd number of intersections means inside
}

// Main CSG Union function (placeholder for now)
Mesh union_meshes(const Mesh& meshA, const Mesh& meshB) {
    // Build AABB trees for both meshes
    AABBTree treeA;
    treeA.build(meshA);

    AABBTree treeB;
    treeB.build(meshB);

    Mesh result_mesh;

    // Process triangles from mesh A
    for (const auto& tri_a : treeA.triangles) {
        // For union, keep triangles of A that are outside B
        if (!is_point_inside_mesh(tri_a.v0, treeB) &&
            !is_point_inside_mesh(tri_a.v1, treeB) &&
            !is_point_inside_mesh(tri_a.v2, treeB)) {
            
            result_mesh.vertices.push_back(tri_a.v0);
            result_mesh.vertices.push_back(tri_a.v1);
            result_mesh.vertices.push_back(tri_a.v2);
            result_mesh.faces.push_back({
                (int)result_mesh.vertices.size() - 3,
                (int)result_mesh.vertices.size() - 2,
                (int)result_mesh.vertices.size() - 1
            });
        }
    }
    
    // Process triangles from mesh B
    for (const auto& tri_b : treeB.triangles) {
        // For union, keep triangles of B that are outside A
        if (!is_point_inside_mesh(tri_b.v0, treeA) &&
            !is_point_inside_mesh(tri_b.v1, treeA) &&
            !is_point_inside_mesh(tri_b.v2, treeA)) {

            result_mesh.vertices.push_back(tri_b.v0);
            result_mesh.vertices.push_back(tri_b.v1);
            result_mesh.vertices.push_back(tri_b.v2);
            result_mesh.faces.push_back({
                (int)result_mesh.vertices.size() - 3,
                (int)result_mesh.vertices.size() - 2,
                (int)result_mesh.vertices.size() - 1
            });
        }
    }

    // TODO: Handle intersecting triangles and merging logic
    // The current approach is a simplified boolean operation that doesn't handle intersections
    // This will lead to holes and non-manifold geometry for intersecting meshes.

    return result_mesh;
}
