#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    vec3 N = normalize(normal);

    // Checkerboard pattern
    float checkerX = floor(uv.x * 8.0);
    float checkerY = floor(uv.y * 8.0);
    float checkerboardIntensity;
    if (mod(checkerX + checkerY, 2.0) == 0.0) {
        checkerboardIntensity = 0.9;
    } else {
        checkerboardIntensity = 0.2;
    }
    vec3 checkerboardColor = vec3(checkerboardIntensity);

    // Lambertian shading
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);
    float diffuseFactor = max(dot(N, lightDir), 0.0);

    // Apply shading to checkerboard color
    vec3 finalColor = checkerboardColor * (0.5 + 0.5 * diffuseFactor);

    outColor = vec4(finalColor, 1.0);
}
