%v4float_0_7_0_7_0_8_1 = OpConstantComposite %v4float %float_0_7 %float_0_7 %float_0_8 %float_1 ; Fog color

; Global variables
%_entryPoint_worldPos = OpVariable %_ptr_Input_v3float Input
%_entryPoint_normal = OpVariable %_ptr_Input_v3float Input
%_entryPoint_outColor = OpVariable %_ptr_Output_v4float Output
%ubo_var = OpVariable %_ptr_Uniform_UBO Uniform

; Main function
%main = OpFunction %void None %voidfn
%entry = OpLabel
    ; Get inputs
    %worldPos = OpLoad %v3float %_entryPoint_worldPos
    %normal = OpLoad %v3float %_entryPoint_normal

    ; Access UBO members
    %lightPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo_var %uint_3
    %lightPos = OpLoad %v4float %lightPos_ptr
    %cameraPos_ptr = OpAccessChain %_ptr_Uniform_v4float %ubo_var %uint_4
    %cameraPos = OpLoad %v4float %cameraPos_ptr

    ; Convert lightPos and cameraPos to vec3
    %lightPos_vec3 = OpVectorShuffle %v3float %lightPos %lightPos %uint_0 %uint_1 %uint_2
    %cameraPos_vec3 = OpVectorShuffle %v3float %cameraPos %cameraPos %uint_0 %uint_1 %uint_2

    ; Calculate distance for fog
    %distVec = OpFSub %v3float %cameraPos_vec3 %worldPos
    %dist = OpSNorm %float %distVec ; Using SNorm for length, as Length instruction needs extension
    
    ; Fog factor = clamp((dist - 1.5) / 3.0, 0.0, 1.0)
    %dist_minus_1_5 = OpFSub %float %dist %float_1_5
    %fogFactor_unclamped = OpFDiv %float %dist_minus_1_5 %float_3
    %fogFactor_clamped_min = OpFMax %float %fogFactor_unclamped %float_0
    %fogFactor = OpFMin %float %fogFactor_clamped_min %float_1

    ; Calculate Lambertian factor
    %lightDir = OpNormalize %v3float %lightPos_vec3
    %normal_normalized = OpNormalize %v3float %normal
    %diffuse = OpDot %float %lightDir %normal_normalized
    %clampedDiffuse = OpFMax %float %diffuse %float_0
    %lambertianFactor_unweighted = OpFMul %float %float_0_7 %clampedDiffuse
    %lambertianFactor = OpFAdd %float %float_0_3 %lambertianFactor_unweighted

    ; Base color * lambertianFactor
    %baseColor_rgb = OpVectorShuffle %v3float %v4float_0_8_0_2_0_2_1 %v4float_0_8_0_2_0_2_1 %uint_0 %uint_1 %uint_2
    %finalBaseColor_rgb = OpVectorTimesScalar %v3float %baseColor_rgb %lambertianFactor
    
    ; Mix base color and fog color: a * (1 - t) + b * t
    %fogColor_rgb = OpVectorShuffle %v3float %v4float_0_7_0_7_0_8_1 %v4float_0_7_0_7_0_8_1 %uint_0 %uint_1 %uint_2
    %one_minus_fogFactor = OpFSub %float %float_1 %fogFactor
    %term1 = OpVectorTimesScalar %v3float %finalBaseColor_rgb %one_minus_fogFactor
    %term2 = OpVectorTimesScalar %v3float %fogColor_rgb %fogFactor
    %mixed_rgb = OpFAdd %v3float %term1 %term2

    %outColor_final = OpCompositeConstruct %v4float %mixed_rgb %float_1 ; Alpha 1.0

    OpStore %_entryPoint_outColor %outColor_final
    OpReturn
OpFunctionEnd
