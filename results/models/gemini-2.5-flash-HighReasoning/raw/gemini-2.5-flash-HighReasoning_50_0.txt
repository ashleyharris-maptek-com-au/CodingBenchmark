#version 450

layout(local_size_x = 256) in;

// Each boid stored as two vec4s: [pos.xyz, 0], [vel.xyz, 0]
// Boid i is at index i*2 and i*2+1
layout(set = 0, binding = 0) readonly buffer InputBoids {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBoids {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;      // counts.x = N
    vec4 radii;        // radii.x=dt, radii.y=sepRadius, radii.z=alignRadius, radii.w=cohRadius
    vec4 weights;      // weights.x=sepWeight, weights.y=alignWeight, weights.z=cohWeight, weights.w=maxSpeed
    vec4 bounds;       // bounds.x=boundSize, bounds.y=boundForce
};

void main() {
    uint N = counts.x;
    uint boidIdx = gl_GlobalInvocationID.x;

    // Critical Requirement: Guard against gl_GlobalInvocationID.x >= N
    if (boidIdx >= N) {
        return;
    }

    // Unpack parameters from uniforms
    float dt = radii.x;
    float sepRadius = radii.y;
    float alignRadius = radii.z;
    float cohRadius = radii.w;

    float sepWeight = weights.x;
    float alignWeight = weights.y;
    float cohWeight = weights.z;
    float maxSpeed = weights.w;

    float boundSize = bounds.x;
    float boundForce = bounds.y;

    // Current boid's position and velocity
    vec3 pos = inData[boidIdx * 2].xyz;
    vec3 vel = inData[boidIdx * 2 + 1].xyz;

    vec3 separationSteer = vec3(0.0);
    vec3 alignmentSteer = vec3(0.0);
    vec3 cohesionSteer = vec3(0.0);

    int sepCount = 0;
    int alignCount = 0;
    int cohCount = 0;

    // Accumulators for alignment and cohesion
    vec3 avgNeighborVel = vec3(0.0);
    vec3 avgNeighborPos = vec3(0.0);

    // Iterate over all other boids to calculate flocking forces
    for (uint i = 0; i < N; ++i) {
        // Critical Requirement: Skip self (i == j)
        if (i == boidIdx) {
            continue;
        }

        vec3 neighborPos = inData[i * 2].xyz;
        vec3 neighborVel = inData[i * 2 + 1].xyz;

        vec3 diff = pos - neighborPos;
        float dist = length(diff);

        // Critical Requirement: Skip zero-distance neighbors
        if (dist > 0.0) {
            // Separation: Steer to avoid crowding local flockmates
            if (dist < sepRadius) {
                // Force is stronger when closer, steer away from neighbor
                separationSteer += normalize(diff) / dist;
                sepCount++;
            }

            // Alignment: Steer towards the average heading of local flockmates
            if (dist < alignRadius) {
                avgNeighborVel += neighborVel;
                alignCount++;
            }

            // Cohesion: Steer to move towards the average position of local flockmates
            if (dist < cohRadius) {
                avgNeighborPos += neighborPos;
                cohCount++;
            }
        }
    }

    // Critical Requirement: Average the accumulated forces by neighbor count (avoid division by zero)
    if (sepCount > 0) {
        separationSteer /= float(sepCount);
        // Normalize and scale to maxSpeed for steering force
        if (length(separationSteer) > 0.0) {
            separationSteer = normalize(separationSteer) * maxSpeed - vel;
        }
    }

    if (alignCount > 0) {
        avgNeighborVel /= float(alignCount);
        // Steering force towards average velocity
        if (length(avgNeighborVel) > 0.0) {
            alignmentSteer = normalize(avgNeighborVel) * maxSpeed - vel;
        }
    }

    if (cohCount > 0) {
        avgNeighborPos /= float(cohCount);
        // Steering force towards average position
        vec3 toCenter = avgNeighborPos - pos;
        if (length(toCenter) > 0.0) {
            cohesionSteer = normalize(toCenter) * maxSpeed - vel;
        }
    }

    // Boundary force: push toward center using boundForce when component exceeds boundSize*0.4
    vec3 boundaryAccel = vec3(0.0);
    float boundaryThreshold = boundSize * 0.4;
    float maxBoundaryDistance = boundSize * 0.5; // The point at which max boundForce is applied

    // Helper to calculate linear ramped force
    // Ensures no division by zero if threshold and max distance are too close or equal
    float forceRampFactor = (maxBoundaryDistance > boundaryThreshold) ?
                            ((abs(pos.x) - boundaryThreshold) / (maxBoundaryDistance - boundaryThreshold)) : 1.0;

    if (abs(pos.x) > boundaryThreshold) {
        if (abs(pos.x) > maxBoundaryDistance) {
             boundaryAccel.x = -sign(pos.x) * boundForce;
        } else {
             boundaryAccel.x = -sign(pos.x) * boundForce * forceRampFactor;
        }
    }
    if (abs(pos.y) > boundaryThreshold) {
        forceRampFactor = (maxBoundaryDistance > boundaryThreshold) ?
                          ((abs(pos.y) - boundaryThreshold) / (maxBoundaryDistance - boundaryThreshold)) : 1.0;
        if (abs(pos.y) > maxBoundaryDistance) {
             boundaryAccel.y = -sign(pos.y) * boundForce;
        } else {
             boundaryAccel.y = -sign(pos.y) * boundForce * forceRampFactor;
        }
    }
    if (abs(pos.z) > boundaryThreshold) {
        forceRampFactor = (maxBoundaryDistance > boundaryThreshold) ?
                          ((abs(pos.z) - boundaryThreshold) / (maxBoundaryDistance - boundaryThreshold)) : 1.0;
        if (abs(pos.z) > maxBoundaryDistance) {
             boundaryAccel.z = -sign(pos.z) * boundForce;
        } else {
             boundaryAccel.z = -sign(pos.z) * boundForce * forceRampFactor;
        }
    }

    // Combine forces. Note: Flocking behaviors are steering forces (desired_vel - current_vel)
    // Boundary force is an acceleration.
    vec3 acceleration = vec3(0.0);
    acceleration += separationSteer * sepWeight;
    acceleration += alignmentSteer * alignWeight;
    acceleration += cohesionSteer * cohWeight;
    acceleration += boundaryAccel; // Boundary force directly adds to acceleration

    // Update velocity by accel*dt
    vel += acceleration * dt;

    // Critical Requirement: Clamp velocity magnitude to maxSpeed
    if (length(vel) > maxSpeed) {
        vel = normalize(vel) * maxSpeed;
    }

    // Update position by vel*dt
    pos += vel * dt;

    // Write updated boid data to output buffer
    outData[boidIdx * 2] = vec4(pos, 0.0);
    outData[boidIdx * 2 + 1] = vec4(vel, 0.0);
}
