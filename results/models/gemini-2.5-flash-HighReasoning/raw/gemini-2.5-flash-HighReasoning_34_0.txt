import sys
import math

class Customer:
    def __init__(self, x, y, demand, index):
        self.x = x
        self.y = y
        self.demand = demand
        self.index = index
        self.is_visited = False

def calculate_distance(p1_x, p1_y, p2_x, p2_y):
    return math.sqrt((p1_x - p2_x)**2 + (p1_y - p2_y)**2)

def solve_cvrp():
    N, K, CAPACITY = map(int, sys.stdin.readline().split())

    depot_x, depot_y = map(int, sys.stdin.readline().split())

    customers = []
    for i in range(N):
        x, y, demand = map(int, sys.stdin.readline().split())
        customers.append(Customer(x, y, demand, i))

    routes = []
    total_distance = 0.0
    num_visited_customers = 0

    while num_visited_customers < N:
        current_route = []
        current_capacity = CAPACITY
        current_position_x, current_position_y = depot_x, depot_y
        route_distance = 0.0
        
        unvisited_customers = [c for c in customers if not c.is_visited]
        
        if not unvisited_customers:
            break

        best_customer_for_route = None
        min_dist_to_start = float('inf')

        # Find the nearest unvisited customer from the depot to start the route
        for customer in unvisited_customers:
            if customer.demand <= current_capacity:
                dist = calculate_distance(depot_x, depot_y, customer.x, customer.y)
                if dist < min_dist_to_start:
                    min_dist_to_start = dist
                    best_customer_for_route = customer

        if best_customer_for_route:
            current_route.append(best_customer_for_route.index)
            best_customer_for_route.is_visited = True
            current_capacity -= best_customer_for_route.demand
            route_distance += min_dist_to_start
            current_position_x, current_position_y = best_customer_for_route.x, best_customer_for_route.y
            num_visited_customers += 1

            # Continue adding nearest unvisited customers to this route
            while True:
                next_customer = None
                min_dist = float('inf')

                for customer in customers:
                    if not customer.is_visited and customer.demand <= current_capacity:
                        dist = calculate_distance(current_position_x, current_position_y, customer.x, customer.y)
                        if dist < min_dist:
                            min_dist = dist
                            next_customer = customer
                
                if next_customer:
                    current_route.append(next_customer.index)
                    next_customer.is_visited = True
                    current_capacity -= next_customer.demand
                    route_distance += min_dist
                    current_position_x, current_position_y = next_customer.x, next_customer.y
                    num_visited_customers += 1
                else:
                    break 
        else:
            # If no customer can be added to a new route, it might mean remaining customers
            # have demands exceeding capacity, or all customers are visited.
            break 
            
        # Return to depot for this route
        route_distance += calculate_distance(current_position_x, current_position_y, depot_x, depot_y)
        total_distance += route_distance
        routes.append(current_route)
        
        # If we have exceeded K routes and there are still unvisited customers,
        # it means the greedy approach might not respect K as a hard constraint.
        # However, the problem says "up to K", implying it's a soft upper bound on vehicles
        # not a strict requirement to use exactly K or less if more are needed.
        # The primary goal is to cover all customers and minimize distance.

    sys.stdout.write(f"{total_distance}
")
    sys.stdout.write(f"{len(routes)}
")
    for route in routes:
        sys.stdout.write(" ".join(map(str, route)) + "
")

if __name__ == "__main__":
    solve_cvrp()
