#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Material properties
    vec3 materialAmbient = vec3(0.1, 0.05, 0.05);
    vec3 materialDiffuse = color; // Use per-vertex color
    vec3 materialSpecular = vec3(1.0, 1.0, 1.0);
    float shininess = 32.0;

    // Light color (assuming white light as not specified)
    vec3 lightColor = vec3(1.0, 1.0, 1.0);

    // Calculate vectors
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    vec3 V = normalize(ubo.cameraPos.xyz - worldPos);
    vec3 R = reflect(-L, N);

    // Ambient component
    vec3 ambient = materialAmbient * lightColor;

    // Diffuse component
    float diff = max(dot(N, L), 0.0);
    vec3 diffuse = materialDiffuse * diff * lightColor;

    // Specular component
    float spec = pow(max(dot(R, V), 0.0), shininess);
    vec3 specular = materialSpecular * spec * lightColor;

    // Total light
    vec3 result = ambient + diffuse + specular;

    // Clamp the result to [0, 1]
    result = clamp(result, 0.0, 1.0);

    // Output color with alpha 1.0
    outColor = vec4(result, 1.0);
}
