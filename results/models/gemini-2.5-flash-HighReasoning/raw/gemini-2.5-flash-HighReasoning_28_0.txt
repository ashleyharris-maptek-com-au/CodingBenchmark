use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read num_vertices and num_edges
    let first_line = lines.next().expect("Expected graph dimensions").expect("Failed to read line");
    let mut parts = first_line.split_whitespace();
    let num_vertices: usize = parts.next().expect("Expected num_vertices").parse().expect("Invalid num_vertices");
    let _num_edges: usize = parts.next().expect("Expected num_edges").parse().expect("Invalid num_edges");

    // Adjacency list (not strictly needed for maximal matching, but good for general graph representation)
    // let mut adj: Vec<Vec<usize>> = vec![vec![]; num_vertices];
    let mut edges: Vec<(usize, usize)> = Vec::new();

    // Read edges
    for line in lines {
        let line = line.expect("Failed to read line");
        let mut parts = line.split_whitespace();
        let u: usize = parts.next().expect("Expected u").parse().expect("Invalid u");
        let v: usize = parts.next().expect("Expected v").parse().expect("Invalid v");
        edges.push((u, v));
        // adj[u].push(v);
        // adj[v].push(u); // Assuming an undirected graph
    }

    // --- Maximal Matching Algorithm ---
    let mut matched_nodes: HashSet<usize> = HashSet::new();
    let mut vertex_cover: HashSet<usize> = HashSet::new();

    for (u, v) in edges {
        if !matched_nodes.contains(&u) && !matched_nodes.contains(&v) {
            // This edge can be part of the matching
            matched_nodes.insert(u);
            matched_nodes.insert(v);
            vertex_cover.insert(u);
            vertex_cover.insert(v);
        }
    }

    // Output the vertex cover
    println!("{}", vertex_cover.len());
    let mut sorted_cover: Vec<usize> = vertex_cover.into_iter().collect();
    sorted_cover.sort_unstable();
    for (i, &v) in sorted_cover.iter().enumerate() {
        print!("{}", v);
        if i < sorted_cover.len() - 1 {
            print!(" ");
        }
    }
    println!();
}
