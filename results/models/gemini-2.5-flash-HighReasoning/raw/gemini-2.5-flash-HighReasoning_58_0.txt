const fs = require('fs');

// Read input from stdin
const input = fs.readFileSync(0, 'utf8');
const data = JSON.parse(input);

const grid = data.grid;
const robots = data.robots;
const jobs = data.jobs;

const rows = grid.length;
const cols = grid[0].length;

// Pre-process jobs for easy lookup
const jobsById = {};
jobs.forEach(job => {
    jobsById[job.id] = job;
});

// Sort robots by ID to ensure deterministic iteration order for tie-breaking
robots.sort((a, b) => a.id - b.id);

// Initialize robot current positions and assignments
const robotCurrentPositions = new Array(robots.length);
const robotAssignments = new Array(robots.length);

robots.forEach(r => {
    robotCurrentPositions[r.id] = { x: r.x, y: r.y };
    robotAssignments[r.id] = [];
});

/**
 * Performs a Breadth-First Search (BFS) to find the shortest path distance
 * between a start point and an end point in the grid, avoiding obstacles.
 * @param {number} startX - The starting X coordinate.
 * @param {number} startY - The starting Y coordinate.
 * @param {number} endX - The ending X coordinate.
 * @param {number} endY - The ending Y coordinate.
 * @returns {number} The shortest distance, or Infinity if the end point is unreachable.
 */
function bfs(startX, startY, endX, endY) {
    // If start or end is an obstacle, or out of bounds, it's unreachable
    if (startX < 0 || startX >= cols || startY < 0 || startY >= rows || grid[startY][startX] === '#') {
        return Infinity;
    }
    if (endX < 0 || endX >= cols || endY < 0 || endY >= rows || grid[endY][endX] === '#') {
        return Infinity;
    }

    // If start and end are the same, distance is 0
    if (startX === endX && startY === endY) {
        return 0;
    }

    const queue = [[startX, startY, 0]]; // [x, y, distance]
    const visited = new Set();
    const getCoordKey = (x, y) => `${x},${y}`;
    visited.add(getCoordKey(startX, startY));

    const directions = [[0, 1], [0, -1], [1, 0], [-1, 0]]; // Right, Left, Down, Up

    while (queue.length > 0) {
        const [currX, currY, dist] = queue.shift();

        // Check neighbors
        for (const [dx, dy] of directions) {
            const nextX = currX + dx;
            const nextY = currY + dy;

            // Check boundaries, obstacles, and if already visited
            if (nextX >= 0 && nextX < cols && nextY >= 0 && nextY < rows &&
                grid[nextY][nextX] === '.' && !visited.has(getCoordKey(nextX, nextY))) {

                if (nextX === endX && nextY === endY) {
                    return dist + 1; // Found the target
                }

                visited.add(getCoordKey(nextX, nextY));
                queue.push([nextX, nextY, dist + 1]);
            }
        }
    }
    return Infinity; // End point is unreachable
}

// Sort jobs for deterministic greedy assignment
// Primary: priority (descending, higher priority first)
// Secondary: id (ascending, smaller id first)
const sortedJobs = [...jobs].sort((a, b) => {
    if (b.priority !== a.priority) {
        return b.priority - a.priority;
    }
    return a.id - b.id;
});

// Main assignment loop: assign jobs one by one based on the greedy strategy
for (const job of sortedJobs) {
    let minDistance = Infinity;
    let bestRobotId = null;

    // Iterate through robots (already sorted by ID) to find the best one for this job
    for (const robot of robots) {
        const robotId = robot.id;
        const robotPos = robotCurrentPositions[robotId];
        const currentDistance = bfs(robotPos.x, robotPos.y, job.x, job.y);

        // Tie-breaking logic:
        // 1. Minimum distance
        // 2. Minimum robotId (implicitly handled by iterating through sorted robots)
        if (currentDistance < minDistance) {
            minDistance = currentDistance;
            bestRobotId = robotId;
        }
        // If currentDistance === minDistance, the robot with the smaller ID
        // (due to sorted `robots` array) would have been picked first.
    }

    if (bestRobotId !== null && minDistance !== Infinity) {
        // Assign the job to the best robot
        robotAssignments[bestRobotId].push(job.id);
        // Update the robot's current position to the assigned job's location
        robotCurrentPositions[bestRobotId] = { x: job.x, y: job.y };
    } else {
        // This case should ideally not be reached if all jobs are reachable.
        // If it is, it means a job cannot be reached by any robot.
        // For this benchmark, we assume all jobs are reachable.
        // If unreachable jobs are possible, a different handling (e.g., skipping) might be needed.
    }
}

// Output the assignments in the specified JSON format
console.log(JSON.stringify({ assignments: robotAssignments }));
