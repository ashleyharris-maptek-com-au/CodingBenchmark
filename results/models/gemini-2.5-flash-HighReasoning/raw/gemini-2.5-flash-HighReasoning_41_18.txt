; SPIR-V
; Version: 1.0
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %color %outColor %_ %_descriptor_set_0_binding_0
OpExecutionMode %main OriginUpperLeft

; Debug Information
OpSource GLSL 450
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %tangent "tangent"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %_descriptor_set_0_binding_0 "UBO_binding"

; Annotations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 ColMajor
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 0 MatrixStride 16
OpMemberDecorate %UBO 1 ColMajor
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 1 MatrixStride 16
OpMemberDecorate %UBO 2 ColMajor
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 2 MatrixStride 16
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224
OpDecorate %_descriptor_set_0_binding_0 DescriptorSet 0
OpDecorate %_descriptor_set_0_binding_0 Binding 0

; Types
%void = OpTypeVoid
%void_func = OpTypeFunction %void
%float = OpTypeFloat 32
%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4 = OpTypeMatrix %v4float 4
%_ptr_Input_v3float = OpTypePointer Input %v3float
%_ptr_Input_v2float = OpTypePointer Input %v2float
%_ptr_Input_v4float = OpTypePointer Input %v4float
%_ptr_Output_v4float = OpTypePointer Output %v4float
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%_struct_10 = OpTypeStruct %mat4 %mat4 %mat4 %v4float %v4float %v4float
%_ptr_Uniform__struct_10 = OpTypePointer Uniform %_struct_10
%_ptr_Uniform_v4float = OpTypePointer Uniform %v4float
%_ptr_Function_v3float = OpTypePointer Function %v3float
%_ptr_Function_float = OpTypePointer Function %float

; Constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_5 = OpConstant %float 0.5
%float_2 = OpConstant %float 2
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_2 = OpConstant %uint 2
%uint_3 = OpConstant %uint 3
%uint_4 = OpConstant %uint 4
%uint_5 = OpConstant %uint 5

; Global Variables
%worldPos = OpVariable %_ptr_Input_v3float Input
%normal = OpVariable %_ptr_Input_v3float Input
%uv = OpVariable %_ptr_Input_v2float Input
%tangent = OpVariable %_ptr_Input_v3float Input
%color = OpVariable %_ptr_Input_v4float Input
%outColor = OpVariable %_ptr_Output_v4float Output
%_descriptor_set_0_binding_0 = OpVariable %_ptr_Uniform__struct_10 Uniform

%main = OpFunction %void None %void_func
%entry = OpLabel
  ; Get inputs
  %_normal = OpLoad %v3float %normal
  %_worldPos = OpLoad %v3float %worldPos

  ; Get cameraPos from UBO
  %cameraPos_ptr = OpAccessChain %_ptr_Uniform_v4float %_descriptor_set_0_binding_0 %uint_4
  %cameraPos = OpLoad %v4float %cameraPos_ptr

  ; Calculate V (view vector)
  %cameraPos_xyz = OpVectorExtractDynamic %v3float %cameraPos %uint_0
  %V_raw = OpFSub %v3float %cameraPos_xyz %_worldPos
  %V = OpNormalize %v3float %V_raw

  ; Normalize N
  %N = OpNormalize %v3float %_normal

  ; Calculate NdotV
  %dotNV = OpDot %float %N %V
  %NdotV = OpExtInst %float %1 GLSL.fmax %dotNV %float_0

  ; Initialize outColor
  %finalColor = OpVariable %_ptr_Function_v4float Function
  OpStore %finalColor %v4float (1.0, 1.0, 1.0, 1.0) ; Default to white (or any other initial color)

  ; If NdotV < 0.5
  %condition_lt_0_5 = OpFOrdLessThan %bool %NdotV %float_0_5
  OpSelectionMerge %merge_block None
  OpIf %condition_lt_0_5
  %if_true_block = OpLabel
    %two_times_NdotV = OpFMul %float %float_2 %NdotV
    %color_lt_0_5 = OpCompositeConstruct %v4float %float_1 %two_times_NdotV %float_0 %float_1
    OpStore %finalColor %color_lt_0_5
    OpBranch %merge_block
  %if_false_block = OpLabel
    ; Else (NdotV >= 0.5)
    %NdotV_minus_0_5 = OpFSub %float %NdotV %float_0_5
    %temp = OpFMul %float %NdotV_minus_0_5 %float_2
    %one_minus_temp = OpFSub %float %float_1 %temp
    %color_ge_0_5 = OpCompositeConstruct %v4float %one_minus_temp %one_minus_temp %temp %float_1
    OpStore %finalColor %color_ge_0_5
    OpBranch %merge_block
  %merge_block = OpLabel

  %out = OpLoad %v4float %finalColor
  OpStore %outColor %out

  OpReturn
OpFunctionEnd
