#include <iostream>
#include <vector>
#include <deque>
#include <set>
#include <algorithm>
#include <numeric>
#include <map> // For BFS path reconstruction

// Represents an N-dimensional point
using Point = std::vector<int>;

// Custom comparison for Point for use in std::set and std::map keys
struct PointComparator {
    bool operator()(const Point& a, const Point& b) const {
        if (a.size() != b.size()) {
            return a.size() < b.size();
        }
        for (size_t i = 0; i < a.size(); ++i) {
            if (a[i] != b[i]) {
                return a[i] < b[i];
            }
        }
        return false; // Points are equal
    }
};

// Global game state variables
int D; // Number of dimensions
int max_turns;
Point bounds; // Size of space in each dimension
std::deque<Point> snake;
std::set<Point, PointComparator> food_positions;
std::set<Point, PointComparator> obstacle_positions;
int initial_snake_length; // Not strictly needed, snake.size() will track current length

// Helper function to read an N-dimensional point
Point read_point() {
    Point p(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> p[i];
    }
    return p;
}

// Function to parse all input
void parse_input() {
    std::cin >> D >> max_turns;

    bounds.resize(D);
    for (int i = 0; i < D; ++i) {
        std::cin >> bounds[i];
    }

    std::cin >> initial_snake_length; // This is the initial length, not current length for input
    snake.push_back(read_point()); // Snake head position

    // Add initial snake segments to match initial_snake_length
    // The problem statement specifies "snake_length start_pos", but only one start_pos is provided.
    // Assuming for now that initial_snake_length is 1 for the provided start_pos, and the snake grows from there.
    // If a longer initial snake is implied, the problem is underspecified about the other segment positions.
    // For a benchmark, it's safer to assume the simplest valid interpretation: initial snake is just the head.

    int num_food;
    std::cin >> num_food;
    for (int i = 0; i < num_food; ++i) {
        food_positions.insert(read_point());
    }

    int num_obstacles;
    std::cin >> num_obstacles;
    for (int i = 0; i < num_obstacles; ++i) {
        obstacle_positions.insert(read_point());
    }
}

// Function to check if a point is within the N-dimensional bounds
bool is_within_bounds(const Point& p) {
    for (int i = 0; i < D; ++i) {
        if (p[i] < 0 || p[i] >= bounds[i]) {
            return false;
        }
    }
    return true;
}

// Function to check if a point is an obstacle or part of the snake's body
// 'consider_tail_as_free' is true for pathfinding, false for actual collision check
bool is_occupied(const Point& p, bool consider_tail_as_free = true) {
    if (obstacle_positions.count(p)) {
        return true;
    }

    // Check if it's part of the snake's body
    size_t snake_body_size = snake.size();
    if (consider_tail_as_free && snake_body_size > 0) {
        // For pathfinding, the tail will move, so it's not an obstacle.
        // If the snake has grown due to eating food in this turn, the tail won't move,
        // so this logic would need to be slightly adjusted if the snake can grow mid-path.
        // Assuming movement happens at the end of a turn based on one chosen move.
        if (snake.back() == p) { // If the point is the current tail position
            return false;
        }
    }
    
    // Check all parts of the snake except the one potentially freed by the tail moving.
    for (size_t i = 0; i < snake_body_size; ++i) {
        if (snake[i] == p) {
            return true;
        }
    }
    return false;
}

// BFS to find the shortest path from start to target, avoiding obstacles and snake body
std::vector<Point> find_path_bfs(const Point& start, const Point& target) {
    std::deque<Point> q;
    std::map<Point, Point, PointComparator> parent_map; // To reconstruct path
    std::set<Point, PointComparator> visited;

    q.push_back(start);
    visited.insert(start);

    while (!q.empty()) {
        Point current = q.front();
        q.pop_front();

        if (current == target) {
            // Path found, reconstruct it
            std::vector<Point> path;
            Point temp = target;
            while (!(temp == start)) { // operator== for vector
                path.push_back(temp);
                temp = parent_map[temp];
            }
            std::reverse(path.begin(), path.end());
            return path;
        }

        // Generate possible next moves
        for (int axis = 0; axis < D; ++axis) {
            for (int direction_val = -1; direction_val <= 1; direction_val += 2) { // +1 and -1
                Point next_p = current;
                next_p[axis] += direction_val;

                if (is_within_bounds(next_p) && !is_occupied(next_p) && !visited.count(next_p)) {
                    visited.insert(next_p);
                    parent_map[next_p] = current;
                    q.push_back(next_p);
                }
            }
        }
    }
    return {}; // No path found
}


int main() {
    // Optimize C++ standard streams for competitive programming
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    parse_input();

    int current_turn = 0;
    while (current_turn < max_turns && !food_positions.empty()) {
        current_turn++;

        Point current_head = snake.front();

        // Find the nearest food (currently just picks the first one from the set)
        Point target_food;
        bool food_found = false;
        if (!food_positions.empty()) {
            target_food = *food_positions.begin(); // Simple strategy: target the first food found
            food_found = true;
        }

        if (!food_found) {
            // No food left, game should end (win condition handled by loop)
            break;
        }

        std::vector<Point> path = find_path_bfs(current_head, target_food);

        if (path.empty()) {
            // No path to food found. Snake is trapped or there's no way.
            // Try to make a "panic move" to any adjacent empty cell to avoid immediate death or deadlock.
            // This is a very basic fallback. A more advanced AI would try to find a path to its tail or a safe empty space.
            bool moved = false;
            for (int axis = 0; axis < D; ++axis) {
                for (int direction_val = -1; direction_val <= 1; direction_val += 2) {
                    Point next_p = current_head;
                    next_p[axis] += direction_val;
                    if (is_within_bounds(next_p) && !is_occupied(next_p, false)) { // Consider tail as obstacle for this panic move
                        // Found a safe move
                        snake.push_front(next_p);
                        std::cout << axis << " " << direction_val << std::endl;
                        snake.pop_back(); // Shrink tail since no food eaten
                        moved = true;
                        break;
                    }
                }
                if (moved) break;
            }

            if (!moved) {
                // Completely trapped, can't even make a panic move. Game over.
                // This scenario means snake hits itself or obstacle in the next possible move.
                break;
            }
            continue; // Go to next turn
        }

        Point next_move = path[0];

        // Determine axis and direction for output
        int move_axis = -1;
        int move_direction = 0;
        for (int i = 0; i < D; ++i) {
            if (current_head[i] != next_move[i]) {
                move_axis = i;
                move_direction = next_move[i] > current_head[i] ? 1 : -1;
                break;
            }
        }

        // Apply the move to the snake's body
        snake.push_front(next_move);

        // Output the move
        std::cout << move_axis << " " << move_direction << std::endl;

        // Check if food was eaten
        if (food_positions.count(next_move)) {
            food_positions.erase(next_move); // Remove eaten food
            // Snake grows, tail is NOT removed
        } else {
            snake.pop_back(); // Snake moves, tail IS removed
        }
    }

    return 0;
}
