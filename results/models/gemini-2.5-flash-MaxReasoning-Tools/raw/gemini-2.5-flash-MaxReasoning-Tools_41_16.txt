; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 
; Capabilities: Shader
; Memory Model: Logical GLSL450

OpCapability Shader
OpMemoryModel Logical GLSL450

; --- SPIR-V Decorations ---
; Inputs from Vertex Shader
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4

; Output to Fragment Buffer
OpDecorate %outColor Location 0

; Uniform Buffer Object (UBO)
OpDecorate %_ `_`UBO Block
OpMemberDecorate %_ `_`UBO 0 Offset 0
OpMemberDecorate %_ `_`UBO 1 Offset 64
OpMemberDecorate %_ `_`UBO 2 Offset 128
OpMemberDecorate %_ `_`UBO 3 Offset 192
OpMemberDecorate %_ `_`UBO 4 Offset 208
OpMemberDecorate %_ `_`UBO 5 Offset 224
OpDecorate %UBO_instance DescriptorSet 0
OpDecorate %UBO_instance Binding 0


; --- SPIR-V Types ---
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%vec2 = OpTypeVector %float 2
%vec3 = OpTypeVector %float 3
%vec4 = OpTypeVector %float 4
%mat4 = OpTypeMatrix %vec4 4 ; column-major

%uint = OpTypeInt 32 0
%int = OpTypeInt 32 1
%_ptr_Function_float = OpTypePointer Function %float
%_ptr_Input_vec3 = OpTypePointer Input %vec3
%_ptr_Input_vec2 = OpTypePointer Input %vec2
%_ptr_Input_vec4 = OpTypePointer Input %vec4
%_ptr_Output_vec4 = OpTypePointer Output %vec4
%_ptr_Uniform_UBO = OpTypePointer Uniform %_ `_`UBO

; UBO structure
%_ `_`UBO = OpTypeStruct %mat4 %mat4 %mat4 %vec4 %vec4 %vec4

; --- SPIR-V Global Variables ---
%worldPos = OpVariable %_ptr_Input_vec3 Input
%normal = OpVariable %_ptr_Input_vec3 Input
%uv = OpVariable %_ptr_Input_vec2 Input
%tangent = OpVariable %_ptr_Input_vec3 Input
%color = OpVariable %_ptr_Input_vec4 Input

%outColor = OpVariable %_ptr_Output_vec4 Output

%UBO_instance = OpVariable %_ptr_Uniform_UBO Uniform

; --- SPIR-V Constants ---
%float_0 = OpConstant %float 0.0
%float_0_5 = OpConstant %float 0.5
%float_1 = OpConstant %float 1.0
%float_2 = OpConstant %float 2.0
%float_3 = OpConstant %float 3.0
%float_4 = OpConstant %float 4.0
%int_0 = OpConstant %int 0
%int_1 = OpConstant %int 1
%int_20 = OpConstant %int 20
%float_m2 = OpConstant %float -2.0
%float_m1 = OpConstant %float -1.0

; --- SPIR-V Entry Point ---
%main = OpEntryPoint Fragment %main_fn "main" %worldPos %normal %uv %tangent %color %outColor
OpExecutionMode %main_fn OriginUpperLeft

; --- SPIR-V Main Function ---
%main_fn = OpFunction %void None %void_fn
%entry = OpLabel
  ; Get UV coordinates
  %in_uv = OpLoad %vec2 %uv

  ; Mandelbrot mapping: c = (uv.x*3-2, uv.y*2-1)
  %uv_x = OpCompositeExtract %float %in_uv 0
  %uv_y = OpCompositeExtract %float %in_uv 1

  %c_x_mul = OpFMul %float %uv_x %float_3
  %c_x = OpFAdd %float %c_x_mul %float_m2
  
  %c_y_mul = OpFMul %float %uv_y %float_2
  %c_y = OpFAdd %float %c_y_mul %float_m1

  %c = OpCompositeConstruct %vec2 %c_x %c_y

  ; Initialize z = (0, 0)
  %z_x = OpCopyObject %float %float_0
  %z_y = OpCopyObject %float %float_0
  %z = OpCompositeConstruct %vec2 %z_x %z_y

  %iterations = OpVariable %_ptr_Function_int Function
  OpStore %iterations %int_0

  ; Loop for Mandelbrot iterations
  %loop_header = OpLabel
  %current_iterations = OpLoad %int %iterations
  %continue_loop = OpSLessThan %bool %current_iterations %int_20
  OpLoopMerge %loop_end %loop_header_continue None
  OpBranchConditional %continue_loop %loop_body %loop_end

  %loop_body = OpLabel
    ; z_x = OpCompositeExtract %float %z 0 ; z.x
    ; z_y = OpCompositeExtract %float %z 1 ; z.y
    %z_x_current = OpCompositeExtract %float %z 0
    %z_y_current = OpCompositeExtract %float %z 1

    ; z.x_new = z.x*z.x - z.y*z.y + c.x
    %z_x_sq = OpFMul %float %z_x_current %z_x_current
    %z_y_sq = OpFMul %float %z_y_current %z_y_current
    %z_x_sq_minus_z_y_sq = OpFSub %float %z_x_sq %z_y_sq
    %new_z_x = OpFAdd %float %z_x_sq_minus_z_y_sq %c_x

    ; z.y_new = 2*z.x*z.y + c.y
    %two_z_x_z_y = OpFMul %float %float_2 %z_x_current
    %two_z_x_z_y_ = OpFMul %float %two_z_x_z_y %z_y_current
    %new_z_y = OpFAdd %float %two_z_x_z_y_ %c_y

    %z = OpCompositeConstruct %vec2 %new_z_x %new_z_y

    ; |z|^2 = z.x*z.x + z.y*z.y
    %abs_z_sq_x = OpFMul %float %new_z_x %new_z_x
    %abs_z_sq_y = OpFMul %float %new_z_y %new_z_y
    %abs_z_sq = OpFAdd %float %abs_z_sq_x %abs_z_sq_y

    ; Escape condition: |z|^2 > 4
    %escaped = OpFOrdGreaterThan %bool %abs_z_sq %float_4
    OpBranchConditional %escaped %loop_end %loop_header_continue

  %loop_header_continue = OpLabel
    ; Increment iterations
    %next_iterations = OpIAdd %int %current_iterations %int_1
    OpStore %iterations %next_iterations
    OpBranch %loop_header

  %loop_end = OpLabel
  %final_iterations = OpLoad %int %iterations
  
  ; Calculate color based on iterations: t = iterations/20
  %float_final_iterations = OpConvertSToF %float %final_iterations
  %t = OpFDiv %float %float_final_iterations %float_20

  ; Color: (t, t*0.5, 1.0-t), alpha 1.0
  %t_half = OpFMul %float %t %float_0_5
  %one_minus_t = OpFSub %float %float_1 %t

  %final_color = OpCompositeConstruct %vec4 %t %t_half %one_minus_t %float_1
  OpStore %outColor %final_color

  OpReturn
OpFunctionEnd
