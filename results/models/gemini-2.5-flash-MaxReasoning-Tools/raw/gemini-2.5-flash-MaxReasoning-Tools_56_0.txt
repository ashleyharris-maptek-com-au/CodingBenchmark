const fs = require('fs');

function calculateDistance(p1, p2) {
    const dx = p1[0] - p2[0];
    const dy = p1[1] - p2[1];
    return Math.sqrt(dx * dx + dy * dy);
}

function solve() {
    const input = fs.readFileSync(0, 'utf8');
    const data = JSON.parse(input);

    const depot = data.depot;
    const truckCount = data.truckCount;
    const packages = data.packages.map(p => ({ ...p, visited: false }));

    const routes = Array.from({ length: truckCount }, () => []);
    const truckCurrentPositions = Array.from({ length: truckCount }, () => [...depot]);

    let unvisitedCount = packages.length;

    while (unvisitedCount > 0) {
        for (let i = 0; i < truckCount && unvisitedCount > 0; i++) {
            let closestPackage = null;
            let minDistance = Infinity;

            // Find the closest unvisited package for the current truck
            for (const pkg of packages) {
                if (!pkg.visited) {
                    const dist = calculateDistance(truckCurrentPositions[i], [pkg.x, pkg.y]);
                    if (dist < minDistance) {
                        minDistance = dist;
                        closestPackage = pkg;
                    }
                }
            }

            if (closestPackage) {
                routes[i].push(closestPackage.id);
                truckCurrentPositions[i] = [closestPackage.x, closestPackage.y];
                closestPackage.visited = true;
                unvisitedCount--;
            }
        }
    }

    const output = { routes: routes };
    console.log(JSON.stringify(output));
}

solve();
