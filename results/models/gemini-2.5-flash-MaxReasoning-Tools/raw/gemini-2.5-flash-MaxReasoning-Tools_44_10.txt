#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
// layout(location = 3) in vec3 tangent; // Not used
// layout(location = 4) in vec3 color;   // Not used

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Normalize the interpolated normal
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 lightDir = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate diffuse component (Lambertian)
    float diffuse = max(0.0, dot(N, lightDir));

    // Calculate the stripe pattern
    float stripeFactor = sin(uv.y * 3.14159 * 16.0);
    vec3 red = vec3(0.9, 0.1, 0.1);
    vec3 blue = vec3(0.1, 0.1, 0.9);

    // Mix between red and blue based on the stripe factor
    // If stripeFactor > 0, it's more red; if <= 0, more blue.
    vec3 stripeColor = mix(blue, red, step(0.0, stripeFactor));

    // Apply Lambertian shading
    vec3 finalColorRGB = stripeColor * (0.4 + 0.6 * diffuse);

    outColor = vec4(finalColorRGB, 1.0);
}
