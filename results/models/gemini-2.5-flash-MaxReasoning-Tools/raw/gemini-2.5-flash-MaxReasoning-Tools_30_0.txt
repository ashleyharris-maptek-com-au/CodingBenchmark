import sys
import collections

def solve_3sat():
    num_variables, num_clauses = map(int, sys.stdin.readline().split())

    clauses = []
    for _ in range(num_clauses):
        literals = list(map(int, sys.stdin.readline().split()))
        clauses.append(literals)

    # assignment: 0=unassigned, 1=True, -1=False
    assignment = [0] * (num_variables + 1)
    
    # watches: Maps a literal to a list of clause indices that are watching it.
    # To handle both positive and negative literals:
    #   Positive literal `L` maps to index `L`.
    #   Negative literal `-L` maps to index `num_variables + abs(L)`.
    watches = [[] for _ in range(2 * num_variables + 1)]

    # watched_literals[c_idx] stores the two literals from clauses[c_idx] that are being watched.
    watched_literals = [[] for _ in range(num_clauses)]

    # trail: stack of assignments made during the search.
    # Each element: (variable_idx, assigned_value, decision_level)
    trail = []
    
    # prop_queue: Queue of literals that have just been assigned and need to be propagated.
    prop_queue = collections.deque()

    # decision_level: Current recursion depth / decision level.
    decision_level = 0
    
    # --- Helper Functions ---
    def get_watches_idx(literal):
        """Maps a literal to its index in the `watches` list."""
        if literal > 0:
            return literal
        else:
            return num_variables + abs(literal)

    def is_literal_true(literal, current_assignment):
        """Checks if a literal evaluates to True given the current assignment."""
        var = abs(literal)
        if current_assignment[var] == 0:  # Unassigned
            return False
        if literal > 0:
            return current_assignment[var] == 1
        else:  # literal < 0
            return current_assignment[var] == -1
            
    def is_literal_false(literal, current_assignment):
        """Checks if a literal evaluates to False given the current assignment."""
        var = abs(literal)
        if current_assignment[var] == 0:  # Unassigned
            return False
        if literal > 0:
            return current_assignment[var] == -1
        else:  # literal < 0
            return current_assignment[var] == 1

    # --- Initialization of watched literals ---
    for c_idx, clause in enumerate(clauses):
        lits_to_watch = []
        for lit in clause:
            lits_to_watch.append(lit)
            if len(lits_to_watch) == 2:
                break
        
        # If a clause has less than 2 literals (e.g., unit clause), watch the available literal(s)
        while len(lits_to_watch) < 2:
            if not lits_to_watch: # Empty clause, indicates unsat
                # This case should not be hit as per problem statement "All test instances are guaranteed to be satisfiable."
                # However, for robustness, if an empty clause is found, it's UNSAT.
                sys.stdout.write("UNSAT
") 
                sys.exit(0)
            lits_to_watch.append(lits_to_watch[0]) # Watch the same literal twice
            
        watched_literals[c_idx] = lits_to_watch
        
        # Add clause to watches for its initial watched literals
        for lit in lits_to_watch:
            watches[get_watches_idx(lit)].append(c_idx)

    # --- DPLL Core Functions ---
    
    def assign_literal_and_propagate(literal, current_decision_level):
        """
        Assigns a literal (variable) and performs unit propagation.
        Returns True if successful, False if a conflict is detected.
        """
        var = abs(literal)
        value = 1 if literal > 0 else -1

        if assignment[var] == 0: # Variable is unassigned
            assignment[var] = value
            trail.append((var, value, current_decision_level))
            prop_queue.append(literal) # Add to propagation queue

        elif assignment[var] == value: # Variable already assigned this value
            return True
        else: # Conflict: variable assigned opposite value
            return False

        # --- Unit Propagation ---
        while prop_queue:
            propagated_lit = prop_queue.popleft() # Literal that was just assigned True
            
            # We need to process clauses watching the *negation* of `propagated_lit`
            # because that negative literal just became False.
            negated_propagated_lit = -propagated_lit
            
            # Make a copy of the list of clause indices watching `negated_propagated_lit`
            # as we'll modify the original `watches` list by moving clauses.
            clauses_to_update = watches[get_watches_idx(negated_propagated_lit)][:]
            watches[get_watches_idx(negated_propagated_lit)].clear() # Clear and re-populate as we go

            for c_idx in clauses_to_update:
                clause = clauses[c_idx]
                
                # Identify the two watched literals in this clause
                w1 = watched_literals[c_idx][0]
                w2 = watched_literals[c_idx][1]
                
                # Determine which watched literal became false (it's `negated_propagated_lit`)
                # and which is the other watched literal.
                other_watched = w2 if w1 == negated_propagated_lit else w1
                
                # If the clause is already satisfied by the `other_watched` literal (it's True)
                if is_literal_true(other_watched, assignment):
                    # Keep watching `other_watched` and `negated_propagated_lit`
                    watches[get_watches_idx(other_watched)].append(c_idx)
                    watches[get_watches_idx(negated_propagated_lit)].append(c_idx)
                    continue
                
                # Try to find a new literal to watch in the clause
                new_watch_found = False
                for lit_in_clause in clause:
                    # A new watch candidate must not be one of the current watched literals
                    # and must not be assigned False.
                    if lit_in_clause != w1 and 
                       lit_in_clause != w2 and 
                       not is_literal_false(lit_in_clause, assignment):
                        
                        # Found a new literal to watch!
                        # Update `watched_literals` for this clause
                        if w1 == negated_propagated_lit:
                            watched_literals[c_idx][0] = lit_in_clause
                        else:
                            watched_literals[c_idx][1] = lit_in_clause
                        
                        # Add this clause to the watches of the new literal
                        watches[get_watches_idx(lit_in_clause)].append(c_idx)
                        watches[get_watches_idx(other_watched)].append(c_idx) # Re-add other_watched
                        new_watch_found = True
                        break
                
                if new_watch_found:
                    continue # Clause handled, move to next clause_to_update
                
                # If no new watch found, the clause is either unit or conflicting.
                
                if assignment[abs(other_watched)] == 0: # `other_watched` is unassigned, so it's a unit clause!
                    # The clause still watches `other_watched` and `negated_propagated_lit`
                    watches[get_watches_idx(other_watched)].append(c_idx)
                    watches[get_watches_idx(negated_propagated_lit)].append(c_idx)

                    # Assign `other_watched` to satisfy the clause and propagate
                    if not assign_literal_and_propagate(other_watched, current_decision_level):
                        return False # Conflict during recursive propagation
                else: # `other_watched` is also False, and no new watch found. This is a conflict!
                    # This clause cannot be satisfied by any of its literals.
                    return False
        
        return True # Propagation successful, no conflict

    # --- Backtracking ---
    def backtrack(target_decision_level):
        """Reverts assignments from the trail until the target_decision_level is reached."""
        nonlocal decision_level
        
        while trail and trail[-1][2] > target_decision_level:
            var_to_unassign, _, _ = trail.pop() # Only need to unassign variable
            assignment[var_to_unassign] = 0 # Unassign the variable
            
        prop_queue.clear() # Clear propagation queue as all assignments at higher levels are undone
        decision_level = target_decision_level

    # --- Main DPLL Solver Function ---
    def solve():
        nonlocal decision_level
        
        # Check if all clauses are satisfied
        all_satisfied = True
        for clause in clauses:
            if not any(is_literal_true(lit, assignment) for lit in clause):
                all_satisfied = False
                break
        if all_satisfied:
            return True

        # Check for conflicts - an empty clause (all literals false)
        for clause in clauses:
            all_false = True
            for lit in clause:
                if not is_literal_false(lit, assignment):
                    all_false = False
                    break
            if all_false:
                return False # Conflict found

        # Find an unassigned variable to make a decision (simple heuristic: first unassigned)
        decision_var = -1
        for i in range(1, num_variables + 1):
            if assignment[i] == 0:
                decision_var = i
                break
        
        if decision_var == -1: # No unassigned variables, but not all clauses satisfied
            return False

        # --- Decision Branching ---
        decision_level += 1

        # Try assigning True
        if assign_literal_and_propagate(decision_var, decision_level):
            if solve(): # Recursive call
                return True
        
        # If trying True failed, backtrack and try False
        backtrack(decision_level - 1)
        
        # Try assigning False
        if assign_literal_and_propagate(-decision_var, decision_level):
            if solve(): # Recursive call
                return True

        # If both True and False branches failed, backtrack beyond this decision level
        backtrack(decision_level - 1)
        return False

    # Execute the solver
    if solve():
        sys.stdout.write("SAT
")
        sys.stdout.write(" ".join(["1" if x == 1 else "0" for x in assignment[1:]]))
        sys.stdout.write("
")
    else:
        # This branch should ideally not be reached given the problem statement
        # "All test instances are guaranteed to be satisfiable."
        sys.stdout.write("UNSAT
")

solve_3sat()
