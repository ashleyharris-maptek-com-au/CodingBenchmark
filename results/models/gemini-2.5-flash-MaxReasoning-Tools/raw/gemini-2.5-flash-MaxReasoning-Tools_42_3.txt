OpCapability Geometry
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %gl_out %worldPos %normal %color %outColor %UBO_variable
OpExecutionMode %main InputLines
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices 4
OpExecutionMode %main Invocations 1

; Debug info
OpSource GLSL 450
OpName %main "main"
OpName %gl_PerVertex_in "gl_PerVertex_in"
OpMemberName %gl_PerVertex_in 0 "gl_Position"
OpName %arr_pv_in "arr_pv_in"
OpName %gl_in "gl_in"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %color "color"
OpName %gl_PerVertex_out "gl_PerVertex_out"
OpMemberName %gl_PerVertex_out 0 "gl_Position"
OpName %gl_out "gl_out"
OpName %outColor "outColor"
OpName %UBO_params_block "UBO_params_block"
OpMemberName %UBO_params_block 0 "model"
OpMemberName %UBO_params_block 1 "view"
OpMemberName %UBO_params_block 2 "proj"
OpMemberName %UBO_params_block 3 "lightPos"
OpMemberName %UBO_params_block 4 "cameraPos"
OpMemberName %UBO_params_block 5 "params"
OpName %UBO_variable "UBO_variable"

; Annotations
OpDecorate %gl_PerVertex_in Block
OpMemberDecorate %gl_PerVertex_in 0 BuiltIn Position
OpDecorate %arr_pv_in ArrayStride 16
OpDecorate %gl_in Block
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %color Location 2
OpDecorate %gl_PerVertex_out Block
OpMemberDecorate %gl_PerVertex_out 0 BuiltIn Position
OpDecorate %gl_out Block
OpDecorate %outColor Location 0

; UBO decorations - as per test 41.
OpDecorate %UBO_params_block Block
OpMemberDecorate %UBO_params_block 0 ColMajor
OpMemberDecorate %UBO_params_block 0 Offset 0
OpMemberDecorate %UBO_params_block 0 MatrixStride 16
OpMemberDecorate %UBO_params_block 1 ColMajor
OpMemberDecorate %UBO_params_block 1 Offset 64
OpMemberDecorate %UBO_params_block 1 MatrixStride 16
OpMemberDecorate %UBO_params_block 2 ColMajor
OpMemberDecorate %UBO_params_block 2 Offset 128
OpMemberDecorate %UBO_params_block 2 MatrixStride 16
OpMemberDecorate %UBO_params_block 3 Offset 192
OpMemberDecorate %UBO_params_block 4 Offset 208
OpMemberDecorate %UBO_params_block 5 Offset 224
OpDecorate %UBO_variable DescriptorSet 0
OpDecorate %UBO_variable Binding 0

; Types
%1 = OpExtInstImport "GLSL.std.450"
%2 = OpTypeVoid
%3 = OpTypeFunction %2
%4 = OpTypeFloat 32
%5 = OpTypeVector %4 2
%6 = OpTypeVector %4 3
%7 = OpTypeVector %4 4
%8 = OpTypeInt 32 1
%9 = OpTypeInt 32 0
%10 = OpTypeBool
%11 = OpTypeMatrix %7 4

; Constants
%13 = OpConstant %9 2 ; for array size 2 (InputLines)
%18 = OpConstant %9 0
%19 = OpConstant %9 1
%20 = OpConstant %9 2
%21 = OpConstant %9 3
%22 = OpConstant %8 0
%23 = OpConstant %8 1
%24 = OpConstant %4 0.0
%25 = OpConstant %4 1.0
%26 = OpConstant %4 0.03

; UBO Struct and Variable
%12 = OpTypeStruct %7 ; gl_PerVertex_in member 0
%14 = OpTypeArray %12 %13 ; arr_pv_in
%15 = OpTypePointer Input %14 ; ptr_Input_arr_pv_in
%27 = OpTypeStruct %7 ; gl_PerVertex_out member 0
%28 = OpTypePointer Output %27 ; ptr_Output_gl_PerVertex_out

%16 = OpTypePointer Input %7 ; For gl_Position access chain
%17 = OpTypePointer Input %6 ; For worldPos, normal, color access chain
%29 = OpTypePointer Output %6 ; For outColor

%30 = OpTypeStruct %11 %11 %11 %7 %7 %7 ; UBO_params_block
%31 = OpTypePointer Uniform %30 ; ptr_UBO_params
%32 = OpVariable %31 Uniform ; UBO_variable

; Global Variables
%33 = OpVariable %15 Input ; gl_in
%34 = OpVariable %17 Input ; worldPos
%35 = OpVariable %17 Input ; normal
%36 = OpVariable %17 Input ; color
%37 = OpVariable %28 Output ; gl_out
%38 = OpVariable %29 Output ; outColor

%main = OpFunction %2 None %3
%39 = OpLabel ; entry basic block

; Read gl_in[0].gl_Position and gl_in[1].gl_Position
%40 = OpAccessChain %16 %33 %22 %22 ; p0_ptr = &gl_in[0].gl_Position
%41 = OpLoad %7 %40 ; p0_v4
%42 = OpAccessChain %16 %33 %23 %22 ; p1_ptr = &gl_in[1].gl_Position
%43 = OpLoad %7 %42 ; p1_v4

; Read color[0] and color[1]
%44 = OpAccessChain %17 %36 %22 ; c0_ptr = &color[0]
%45 = OpLoad %6 %44 ; c0_v3
%46 = OpAccessChain %17 %36 %23 ; c1_ptr = &color[1]
%47 = OpLoad %6 %46 ; c1_v3

; Extract vec3 from v4float positions (assuming w=1.0 for clip space positions)
%48 = OpVectorShuffle %6 %41 %41 %18 %19 %20 ; p0_v3
%49 = OpVectorShuffle %6 %43 %43 %18 %19 %20 ; p1_v3

; Compute line direction: dir = p1 - p0
%50 = OpFSub %6 %49 %48 ; dir_v3

; Compute perpendicular: perp = normalize(vec2(-dir.y, dir.x)) * thickness
; Need to extract x and y from dir_v3
%51 = OpCompositeExtract %4 %50 1 ; dir_y
%52 = OpCompositeExtract %4 %50 0 ; dir_x
%53 = OpFNegate %4 %51 ; neg_dir_y
%54 = OpCompositeConstruct %5 %53 %52 ; (-dir.y, dir.x)
%55 = OpDot %4 %54 %54 ; len_perp_2d_sq
%56 = OpExtInst %4 %1 Fsqrt %55 ; len_perp_2d
%57 = OpFDiv %5 %54 %56 ; norm_perp_2d

%58 = OpVectorTimesScalar %5 %57 %26 ; scaled_perp_2d (normalize * thickness)

; Convert scaled_perp_2d to v4float for addition with positions.
; Assuming perp should be applied in XY plane, Z=0, W=0.
%59 = OpCompositeExtract %4 %58 0 ; perp_v4_x
%60 = OpCompositeExtract %4 %58 1 ; perp_v4_y
%61 = OpCompositeConstruct %7 %59 %60 %24 %24 ; perp_v4 (x, y, 0.0, 0.0)

; Emit 4 vertices as triangle strip: p0 + perp, p0 - perp, p1 + perp, p1 - perp.

; Vertex 0: p0 + perp
%62 = OpFAdd %7 %41 %61 ; v0_pos = p0_v4 + perp_v4
OpStore %37 %62
OpStore %38 %45 ; Use c0 for first two
OpEmitVertex

; Vertex 1: p0 - perp
%63 = OpFNegate %7 %61 ; neg_perp_v4
%64 = OpFAdd %7 %41 %63 ; v1_pos = p0_v4 - perp_v4
OpStore %37 %64
OpStore %38 %45 ; Use c0 for first two
OpEmitVertex

; Vertex 2: p1 + perp
%65 = OpFAdd %7 %43 %61 ; v2_pos = p1_v4 + perp_v4
OpStore %37 %65
OpStore %38 %47 ; Use c1 for last two
OpEmitVertex

; Vertex 3: p1 - perp
%66 = OpFAdd %7 %43 %63 ; v3_pos = p1_v4 - perp_v4
OpStore %37 %66
OpStore %38 %47 ; Use c1 for last two
OpEmitVertex

OpEndPrimitive

OpReturn
OpFunctionEnd
