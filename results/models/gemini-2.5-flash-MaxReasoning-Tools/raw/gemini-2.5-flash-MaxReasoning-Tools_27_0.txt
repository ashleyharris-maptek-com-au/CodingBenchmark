#include <iostream>
#include <vector>
#include <numeric>
#include <algorithm>
#include <set>

// Structure to represent the graph
struct Graph {
    int num_vertices;
    std::vector<std::vector<int>> adj;

    Graph(int n) : num_vertices(n), adj(n) {}

    void add_edge(int u, int v) {
        adj[u].push_back(v);
        adj[v].push_back(u);
    }
};

// Function to check if a color assignment is safe for a given vertex
bool is_safe(int v, const Graph& graph, const std::vector<int>& colors, int c) {
    for (int neighbor : graph.adj[v]) {
        if (colors[neighbor] == c) {
            return false;
        }
    }
    return true;
}

// Backtracking function for graph coloring
bool graph_coloring_backtracking(int v, const Graph& graph, int num_colors, std::vector<int>& colors) {
    if (v == graph.num_vertices) {
        return true; // All vertices colored
    }

    // Heuristic: Try to color vertices with higher degree first? (Simple ordering for now)
    // For proper optimization, consider MRV and LCV.

    for (int c = 0; c < num_colors; ++c) {
        if (is_safe(v, graph, colors, c)) {
            colors[v] = c;
            if (graph_coloring_backtracking(v + 1, graph, num_colors, colors)) {
                return true;
            }
            colors[v] = -1; // Backtrack
        }
    }
    return false; // No color can be assigned
}

// DSATUR algorithm for graph coloring (greedy heuristic)
// Returns true if a coloring is found, false otherwise.
bool graph_coloring_dsatur(const Graph& graph, int num_colors, std::vector<int>& colors) {
    colors.assign(graph.num_vertices, -1); // Initialize colors to -1 (uncolored)

    std::vector<int> saturation_degree(graph.num_vertices, 0); // Not directly used in the current loop, but conceptual for DSATUR
    std::vector<int> degree(graph.num_vertices, 0);
    for (int i = 0; i < graph.num_vertices; ++i) {
        degree[i] = graph.adj[i].size();
    }

    std::set<int> uncolored_vertices;
    for (int i = 0; i < graph.num_vertices; ++i) {
        uncolored_vertices.insert(i);
    }

    while (!uncolored_vertices.empty()) {
        int u_to_color = -1;
        int max_saturation = -1;
        int max_degree = -1;

        // Find the uncolored vertex with the highest saturation degree (number of colors used by its neighbors)
        // Tie-break with highest degree
        for (int v : uncolored_vertices) {
            int current_saturation = 0;
            std::set<int> neighbor_colors;
            for (int neighbor : graph.adj[v]) {
                if (colors[neighbor] != -1) {
                    neighbor_colors.insert(colors[neighbor]);
                }
            }
            current_saturation = neighbor_colors.size();

            if (u_to_color == -1 || current_saturation > max_saturation ||
                (current_saturation == max_saturation && degree[v] > max_degree)) {
                max_saturation = current_saturation;
                max_degree = degree[v];
                u_to_color = v;
            }
        }

        if (u_to_color == -1) {
            // This should not happen if uncolored_vertices is not empty
            break;
        }

        // Assign the smallest available color to u_to_color
        std::vector<bool> available_colors(num_colors, true);
        for (int neighbor : graph.adj[u_to_color]) {
            if (colors[neighbor] != -1) {
                if (colors[neighbor] < num_colors) { // Ensure color is within bounds
                    available_colors[colors[neighbor]] = false;
                }
            }
        }

        int assigned_color = -1;
        for (int c = 0; c < num_colors; ++c) {
            if (available_colors[c]) {
                assigned_color = c;
                break;
            }
        }

        if (assigned_color == -1) {
            // Cannot color this vertex with the given number of colors
            // This means DSATUR failed to find a coloring with num_colors
            return false;
        }

        colors[u_to_color] = assigned_color;
        uncolored_vertices.erase(u_to_color);
    }

    // After coloring all vertices, verify the coloring
    for (int i = 0; i < graph.num_vertices; ++i) {
        if (colors[i] == -1) return false; // Not all vertices colored
        for (int neighbor : graph.adj[i]) {
            if (colors[i] == colors[neighbor]) {
                return false; // Adjacent vertices have same color
            }
        }
    }
    return true;
}


int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_colors;
    std::cin >> num_vertices >> num_edges >> num_colors;

    Graph graph(num_vertices);
    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        std::cin >> u >> v;
        graph.add_edge(u, v);
    }

    std::vector<int> colors(num_vertices);
    bool colored = false;

    // Adaptive logic: Choose algorithm based on graph size/complexity
    // These thresholds are illustrative and might need tuning based on actual performance
    // Backtracking is exponential, DSATUR is greedy (polynomial but not optimal)
    // A more sophisticated approach might involve a time limit for backtracking
    // or further heuristics for DSATUR.
    if (num_vertices <= 15 && num_edges < 50) { // Small graphs, try backtracking for optimal solution
        // Initialize colors for backtracking
        std::fill(colors.begin(), colors.end(), -1);
        colored = graph_coloring_backtracking(0, graph, num_colors, colors);
    } else { // Larger graphs, use DSATUR
        colored = graph_coloring_dsatur(graph, num_colors, colors);
    }


    if (colored) {
        for (int i = 0; i < num_vertices; ++i) {
            std::cout << colors[i] << (i == num_vertices - 1 ? "" : " ");
        }
        std::cout << std::endl;
    } else {
        // If no coloring found, output -1 for all vertices.
        // This is not explicitly specified in the output format but is a common way
        // to indicate failure or no solution found in competitive programming contexts.
        // The problem description implies that if a coloring is found, it should be printed.
        // If not, the program likely just terminates or prints nothing (which would be interpreted as failure).
        for (int i = 0; i < num_vertices; ++i) {
            std::cout << -1 << (i == num_vertices - 1 ? "" : " ");
        }
        std::cout << std::endl;
    }

    return 0;
}
