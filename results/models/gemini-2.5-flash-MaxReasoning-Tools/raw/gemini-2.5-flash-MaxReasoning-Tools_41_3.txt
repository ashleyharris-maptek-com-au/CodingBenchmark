; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL; 
; Capabilities: Shader
; Memory Model: Logical GLSL450

OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %color %outColor
OpExecutionMode %main OriginUpperLeft

OpSource GLSL 450
OpName %main "main"
OpName %color "color"
OpName %outColor "outColor"

OpDecorate %color Location 4
OpDecorate %outColor Location 0

%void = OpTypeVoid
%void_fn = OpTypeFunction %void
%float = OpTypeFloat 32
%v4float = OpTypeVector %float 4
%v3float = OpTypeVector %float 3
%float_1 = OpConstant %float 1

%ptr_input_v3float = OpTypePointer Input %v3float
%ptr_output_v4float = OpTypePointer Output %v4float

%color = OpVariable %ptr_input_v3float Input
%outColor = OpVariable %ptr_output_v4float Output

%main = OpFunction %void None %void_fn
%entry = OpLabel
%color_val = OpLoad %v3float %color
%outColor_val = OpCompositeConstruct %v4float %color_val %float_1
OpStore %outColor %outColor_val
OpReturn
OpFunctionEnd
