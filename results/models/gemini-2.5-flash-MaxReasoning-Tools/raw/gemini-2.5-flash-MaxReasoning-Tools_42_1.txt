; Main function
%main = OpFunction %void None %fn
%main_label = OpLabel

  ; Get input gl_Position (vec4) for gl_in[0].gl_Position
  %ge_gl_in_0_0 = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
  %input_pos = OpLoad %v4float %ge_gl_in_0_0

  ; Get input color (vec3) for color_in[0]
  %ge_color_in_0 = OpAccessChain %_ptr_Input_v3float %color_in %int_0
  %input_color = OpLoad %v3float %ge_color_in_0

  ; Extract x, y, z, w from input_pos
  %input_x = OpCompositeExtract %float %input_pos 0
  %input_y = OpCompositeExtract %float %input_pos 1
  %input_z = OpCompositeExtract %float %input_pos 2
  %input_w = OpCompositeExtract %float %input_pos 3

  ; Calculate diamond vertices:
  ; Top: (x, y+h)
  %y_plus_h = OpFAdd %float %input_y %float_0_05
  %top_pos_v4 = OpCompositeConstruct %v4float %input_x %y_plus_h %input_z %input_w

  ; Right: (x+h, y)
  %x_plus_h = OpFAdd %float %input_x %float_0_05
  %right_pos_v4 = OpCompositeConstruct %v4float %x_plus_h %input_y %input_z %input_w

  ; Left: (x-h, y)
  %x_minus_h = OpFSub %float %input_x %float_0_05
  %left_pos_v4 = OpCompositeConstruct %v4float %x_minus_h %input_y %input_z %input_w

  ; Bottom: (x, y-h)
  %y_minus_h = OpFSub %float %input_y %float_0_05
  %bottom_pos_v4 = OpCompositeConstruct %v4float %input_x %y_minus_h %input_z %input_w

  ; Emit vertices: top, right, left, bottom
  ; Top
  %ptr_out_pos = OpAccessChain %_ptr_Output_v4float %gl_out %int_0
  OpStore %ptr_out_pos %top_pos_v4
  OpStore %outColor_out %input_color
  OpEmitVertex

  ; Right
  OpStore %ptr_out_pos %right_pos_v4
  OpStore %outColor_out %input_color
  OpEmitVertex

  ; Left
  OpStore %ptr_out_pos %left_pos_v4
  OpStore %outColor_out %input_color
  OpEmitVertex

  ; Bottom
  OpStore %ptr_out_pos %bottom_pos_v4
  OpStore %outColor_out %input_color
  OpEmitVertex

  OpEndPrimitive
  OpReturn
