// Define a small epsilon for floating-point comparisons
const double EPSILON = 1e-9;
const int MAX_V_FOR_V3_HULL = 100; // Threshold for V^3 convex hull face generation

// --- Geometric Structures ---

struct Vector3 {
    double x, y, z;

    Vector3(double x = 0.0, double y = 0.0, double z = 0.0) : x(x), y(y), z(z) {}

    // Basic arithmetic operations
    Vector3 operator+(const Vector3& other) const { return Vector3(x + other.x, y + other.y, z + other.z); }
    Vector3 operator-(const Vector3& other) const { return Vector3(x - other.x, y - other.y, z - other.z); }
    Vector3 operator*(double scalar) const { return Vector3(x * scalar, y * scalar, z * scalar); }
    Vector3 operator/(double scalar) const { return Vector3(x / scalar, y / scalar, z / scalar); }

    // Dot product
    double dot(const Vector3& other) const { return x * other.x + y * other.y + z * other.z; }

    // Cross product
    Vector3 cross(const Vector3& other) const {
        return Vector3(y * other.z - z * other.y,
                       z * other.x - x * other.z,
                       x * other.y - y * other.x);
    }

    // Magnitude (length)
    double length() const { return std::sqrt(x * x + y * y + z * z); }

    // Squared magnitude (for performance, avoid sqrt if possible)
    double lengthSq() const { return x * x + y * y + z * z; }

    // Normalization
    Vector3 normalize() const {
        double len = length();
        if (len < EPSILON) return Vector3(); // Avoid division by zero
        return Vector3(x / len, y / len, z / len);
    }
};

// Represents a 3x3 rotation matrix
struct Matrix3x3 {
    double m[3][3];

    Matrix3x3() {
        // Initialize as identity matrix
        for (int i = 0; i < 3; ++i) {
            for (int j = 0; j < 3; ++j) {
                m[i][j] = (i == j) ? 1.0 : 0.0;
            }
        }
    }

    // Create a rotation matrix from Euler angles (Z-Y-X convention often used in aerospace,
    // or X-Y-Z for some graphics systems. Let's assume a common convention like ZYX or XYZ.
    // The problem doesn't specify, so I'll pick a common one (e.g., yaw-pitch-roll, ZYX)
    // and note it. If I encounter issues, I'll revise.)
    // For now, let's use a simple XYZ order (roll, pitch, yaw)
    static Matrix3x3 fromEulerAngles(double roll_x, double pitch_y, double yaw_z) {
        Matrix3x3 R_x, R_y, R_z;

        double cosX = std::cos(roll_x);
        double sinX = std::sin(roll_x);
        double cosY = std::cos(pitch_y);
        double sinY = std::sin(pitch_y);
        double cosZ = std::cos(yaw_z);
        double sinZ = std::sin(yaw_z);

        // Rotation around X axis
        R_x.m[0][0] = 1.0; R_x.m[0][1] = 0.0;   R_x.m[0][2] = 0.0;
        R_x.m[1][0] = 0.0; R_x.m[1][1] = cosX;  R_x.m[1][2] = -sinX;
        R_x.m[2][0] = 0.0; R_x.m[2][1] = sinX;  R_x.m[2][2] = cosX;

        // Rotation around Y axis
        R_y.m[0][0] = cosY;  R_y.m[0][1] = 0.0; R_y.m[0][2] = sinY;
        R_y.m[1][0] = 0.0;   R_y.m[1][1] = 1.0; R_y.m[1][2] = 0.0;
        R_y.m[2][0] = -sinY; R_y.m[2][1] = 0.0; R_y.m[2][2] = cosY;

        // Rotation around Z axis
        R_z.m[0][0] = cosZ;  R_z.m[0][1] = -sinZ; R_z.m[0][2] = 0.0;
        R_z.m[1][0] = sinZ;  R_z.m[1][1] = cosZ;  R_z.m[1][2] = 0.0;;
        R_z.m[2][0] = 0.0;   R_z.m[2][1] = 0.0;   R_z.m[2][2] = 1.0;

        // Combine rotations: R = Rz * Ry * Rx (standard aerospace convention)
        // Or R = Rx * Ry * Rz (common in graphics)
        // Let's use Rx * Ry * Rz for now, and check if it aligns with typical Euler angle definitions
        // for object orientation.
        Matrix3x3 temp;
        // temp = R_y * R_x (this is matrix multiplication)
        // R = R_z * temp
        // For simplicity, let's use the explicit multiplication for now.
        // A more robust implementation would involve operator overloading for matrix multiplication.
        // For now, I'll implement a helper for vector * matrix.
        // Let's go with the standard ZYX (yaw, pitch, roll) which is R = R_z * R_y * R_x
        // This is commonly used in robotics and aerospace.
        // R(yaw, pitch, roll) = R_z(yaw) * R_y(pitch) * R_x(roll)
        // rx = roll (around x)
        // ry = pitch (around y)
        // rz = yaw (around z)

        // R = R_z * R_y * R_x
        // Assuming the input Euler angles (rx, ry, rz) correspond to (roll, pitch, yaw) respectively.
        // rx -> roll (X)
        // ry -> pitch (Y)
        // rz -> yaw (Z)

        // Rotation matrix elements for ZYX (yaw-pitch-roll)
        // R(rz, ry, rx) =
        // [ cZcY   cZ sY sX - sZ cX    cZ sY cX + sZ sX ]
        // [ sZcY   sZ sY sX + cZ cX    sZ sY cX - cZ sX ]
        // [ -sY    cY sX              cY cX            ]

        Matrix3x3 result;
        result.m[0][0] = cosZ * cosY;
        result.m[0][1] = cosZ * sinY * sinX - sinZ * cosX;
        result.m[0][2] = cosZ * sinY * cosX + sinZ * sinX;

        result.m[1][0] = sinZ * cosY;
        result.m[1][1] = sinZ * sinY * sinX + cosZ * cosX;
        result.m[1][2] = sinZ * sinY * cosX - cosZ * sinX;

        result.m[2][0] = -sinY;
        result.m[2][1] = cosY * sinX;
        result.m[2][2] = cosY * cosX;

        return result;
    }

    // Transform a vector by this matrix
    Vector3 transform(const Vector3& v) const {
        return Vector3(
            m[0][0] * v.x + m[0][1] * v.y + m[0][2] * v.z,
            m[1][0] * v.x + m[1][1] * v.y + m[1][2] * v.z,
            m[2][0] * v.x + m[2][1] * v.y + m[2][2] * v.z
        );
    }
};

// Represents a regular tetrahedron
struct Tetrahedron {
    Vector3 center;
    Matrix3x3 rotation; // Orientation
    double edge_length;
    std::vector<Vector3> local_vertices; // Vertices in local space, then transformed
    double rx, ry, rz; // Euler angles for output

    Tetrahedron(const Vector3& c, double roll_x, double pitch_y, double yaw_z, double e)
        : center(c), rotation(Matrix3x3::fromEulerAngles(roll_x, pitch_y, yaw_z)), edge_length(e), rx(roll_x), ry(pitch_y), rz(yaw_z) {
        
        // Calculate the vertices of a regular tetrahedron centered at origin
        double scale = edge_length / std::sqrt(8.0); // Factor to get correct edge length

        local_vertices.push_back(Vector3(scale, scale, scale));
        local_vertices.push_back(Vector3(scale, -scale, -scale));
        local_vertices.push_back(Vector3(-scale, scale, -scale));
        local_vertices.push_back(Vector3(-scale, -scale, scale));
    }

    // Get the global coordinates of the tetrahedron's vertices
    std::vector<Vector3> getGlobalVertices() const {
        std::vector<Vector3> global_verts;
        for (const auto& local_v : local_vertices) {
            global_verts.push_back(center + rotation.transform(local_v));
        }
        return global_verts;
    }
};

struct Plane {
    Vector3 normal;
    double distance; // distance from origin along normal (normal.dot(point_on_plane))

    Plane(const Vector3& p1, const Vector3& p2, const Vector3& p3) {
        // Calculate normal from three points
        Vector3 edge1 = p2 - p1;
        Vector3 edge2 = p3 - p1;
        normal = edge1.cross(edge2).normalize();

        // If normal is zero vector (collinear points), handle it
        if (normal.lengthSq() < EPSILON) {
            // This case should ideally be filtered out before creating a plane
            // For now, set to arbitrary non-zero to avoid division by zero later,
            // though this plane wouldn't be useful.
            normal = Vector3(1.0, 0.0, 0.0);
        }
        
        distance = normal.dot(p1); // D in Ax + By + Cz = D
    }

    // Signed distance from point P to the plane
    double signedDistance(const Vector3& p) const {
        return normal.dot(p) - distance;
    }
};

// Represents the container as a convex polyhedron
struct Container {
    std::vector<Vector3> vertices;
    std::vector<Plane> faces; // List of planes defining the faces of the convex hull

    // Bounding box for quick culling
    Vector3 min_bounds = Vector3(std::numeric_limits<double>::max(), std::numeric_limits<double>::max(), std::numeric_limits<double>::max());
    Vector3 max_bounds = Vector3(std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest(), std::numeric_limits<double>::lowest());

    Container(const std::vector<Vector3>& container_verts) : vertices(container_verts) {
        // Calculate bounding box
        for (const auto& v : vertices) {
            min_bounds.x = std::min(min_bounds.x, v.x);
            min_bounds.y = std::min(min_bounds.y, v.y);
            min_bounds.z = std::min(min_bounds.z, v.z);

            max_bounds.x = std::max(max_bounds.x, v.x);
            max_bounds.y = std::max(max_bounds.y, v.y);
            max_bounds.z = std::max(max_bounds.z, v.z);
        }

        // --- Convex Hull Face Generation ---
        if (vertices.size() < 4) {
            std::cerr << "Error: Container needs at least 4 vertices to form a polyhedron. Provided " << vertices.size() << std::endl;
            // Fallback: If not enough vertices, rely on bounding box only. This is imperfect.
            // A more robust solution might involve throwing an exception or returning an invalid container.
        } else if (vertices.size() > MAX_V_FOR_V3_HULL) {
            std::cerr << "Warning: Number of container vertices (" << vertices.size() << ") is too large for O(V^3) hull generation. Using bounding box for containment check." << std::endl;
            // In this case, `faces` will remain empty, and `contains` will rely only on the bounding box.
            // This is a trade-off for performance.
        } else {
            // Compute centroid of the polyhedron
            Vector3 centroid;
            for (const auto& v : vertices) {
                centroid = centroid + v;
            }
            centroid = centroid / (double)vertices.size();

            // Iterate through all combinations of 3 vertices to form potential faces
            for (size_t i = 0; i < vertices.size(); ++i) {
                for (size_t j = i + 1; j < vertices.size(); ++j) {
                    for (size_t k = j + 1; k < vertices.size(); ++k) {
                        Vector3 p1 = vertices[i];
                        Vector3 p2 = vertices[j];
                        Vector3 p3 = vertices[k];

                        // Check if points are collinear to avoid degenerate planes
                        if ((p2 - p1).cross(p3 - p1).lengthSq() < EPSILON * EPSILON) {
                            continue; // Collinear, skip
                        }

                        Plane candidate_face(p1, p2, p3);

                        // Ensure normal points outwards. Check if centroid is on the negative side.
                        if (candidate_face.signedDistance(centroid) < 0) {
                            candidate_face.normal = candidate_face.normal * -1.0;
                            candidate_face.distance = candidate_face.normal.dot(p1); // Recalculate distance with new normal
                        }
                        
                        bool is_face = true;
                        for (size_t l = 0; l < vertices.size(); ++l) {
                            // All other vertices must be on or "behind" this plane (inside side).
                            // If any vertex is "outside" (positive signed distance), this is not a face.
                            if (l == i || l == j || l == k) continue;

                            if (candidate_face.signedDistance(vertices[l]) > EPSILON) {
                                is_face = false;
                                break;
                            }
                        }

                        if (is_face) {
                            // Add face if not already present (checking for duplicates of planes)
                            bool found_duplicate = false;
                            for (const auto& existing_face : faces) {
                                // Check if normals are parallel and distances are close
                                if (existing_face.normal.dot(candidate_face.normal) > 1.0 - EPSILON &&
                                    std::abs(existing_face.distance - candidate_face.distance) < EPSILON) {
                                    found_duplicate = true;
                                    break;
                                }
                            }
                            if (!found_duplicate) {
                                faces.push_back(candidate_face);
                            }
                        }
                    }
                }
            }
        }
    }

    // Check if a point is inside the container (proper point-in-convex-polyhedron test)
    bool contains(const Vector3& p) const {
        // First, check bounding box for quick rejection
        if (p.x < min_bounds.x - EPSILON || p.x > max_bounds.x + EPSILON ||
            p.y < min_bounds.y - EPSILON || p.y > max_bounds.y + EPSILON ||
            p.z < min_bounds.z - EPSILON || p.z > max_bounds.z + EPSILON) {
            return false;
        }

        // If faces were generated, use them for a precise check
        if (!faces.empty()) {
            // For a convex polyhedron, a point is inside if it's on the "inside" side
            // of all defining faces (planes). Since we oriented normals outwards,
            // the signed distance should be negative (or zero if on the face).
            for (const auto& face_plane : faces) {
                if (face_plane.signedDistance(p) > EPSILON) { // If point is outside any face
                    return false;
                }
            }
            return true;
        } else {
            // Fallback: If no faces were generated (e.g., V too large or V < 4),
            // just rely on the bounding box. This is less accurate for non-rectangular containers.
            return true;
        }
    }
};

// --- Collision Detection ---

// Separating Axis Theorem (SAT) for two tetrahedrons
// This is a complex function. A full SAT implementation for tetrahedrons involves:
// 1. Checking axes normal to each face of both tetrahedrons (8 axes).
// 2. Checking axes formed by the cross product of an edge from one tetrahedron and an edge from the other (36 axes).
// Projecting each tetrahedron onto these axes and checking for overlap.

// Helper to project a tetrahedron onto an axis
struct Projection {
    double min_val;
    double max_val;
};

Projection projectTetrahedron(const std::vector<Vector3>& verts, const Vector3& axis) {
    double min_proj = verts[0].dot(axis);
    double max_proj = min_proj;

    for (size_t i = 1; i < verts.size(); ++i) {
        double proj = verts[i].dot(axis);
        min_proj = std::min(min_proj, proj);
        max_proj = std::max(max_proj, proj);
    }
    return {min_proj, max_proj};
}

// Check for overlap between two projections
bool checkOverlap(const Projection& p1, const Projection& p2) {
    return (p1.max_val >= p2.min_val - EPSILON && p2.max_val >= p1.min_val - EPSILON);
}

// Get face normals for a tetrahedron (local space)
std::vector<Vector3> getTetrahedronFaceNormals(const std::vector<Vector3>& local_verts) {
    std::vector<Vector3> normals;
    // Canonical vertices: (s,s,s), (s,-s,-s), (-s,s,-s), (-s,-s,s)
    // The normals can be derived relative to the center (0,0,0)

    // Face 0: formed by (v0, v1, v2) (local_verts[0], local_verts[1], local_verts[2])
    normals.push_back((local_verts[1] - local_verts[0]).cross(local_verts[2] - local_verts[0]).normalize());
    // Face 1: formed by (v0, v3, v1)
    normals.push_back((local_verts[3] - local_verts[0]).cross(local_verts[1] - local_verts[0]).normalize());
    // Face 2: formed by (v0, v2, v3)
    normals.push_back((local_verts[2] - local_verts[0]).cross(local_verts[3] - local_verts[0]).normalize());
    // Face 3: formed by (v1, v3, v2)
    normals.push_back((local_verts[3] - local_verts[1]).cross(local_verts[2] - local_verts[1]).normalize());

    return normals;
}

// Function to check for collision between two tetrahedrons using SAT
bool checkTetrahedronCollision(const Tetrahedron& t1, const Tetrahedron& t2) {
    std::vector<Vector3> verts1 = t1.getGlobalVertices();
    std::vector<Vector3> verts2 = t2.getGlobalVertices();

    std::vector<Vector3> axes;

    // Add face normals of t1 (4 axes)
    std::vector<Vector3> local_normals1 = getTetrahedronFaceNormals(t1.local_vertices);
    for (const auto& n : local_normals1) {
        axes.push_back(t1.rotation.transform(n).normalize()); // Transform local normal to global
    }

    // Add face normals of t2 (4 axes)
    std::vector<Vector3> local_normals2 = getTetrahedronFaceNormals(t2.local_vertices);
    for (const auto& n : local_normals2) {
        axes.push_back(t2.rotation.transform(n).normalize()); // Transform local normal to global
    }

    // Add axes from cross products of one edge from t1 and one edge from t2 (36 axes)
    // Edges of t1
    std::vector<Vector3> edges1;
    edges1.push_back(verts1[1] - verts1[0]);
    edges1.push_back(verts1[2] - verts1[0]);
    edges1.push_back(verts1[3] - verts1[0]);
    edges1.push_back(verts1[2] - verts1[1]);
    edges1.push_back(verts1[3] - verts1[1]);
    edges1.push_back(verts1[3] - verts1[2]);

    // Edges of t2
    std::vector<Vector3> edges2;
    edges2.push_back(verts2[1] - verts2[0]);
    edges2.push_back(verts2[2] - verts2[0]);
    edges2.push_back(verts2[3] - verts2[0]);
    edges2.push_back(verts2[2] - verts2[1]);
    edges2.push_back(verts2[3] - verts2[1]);
    edges2.push_back(verts2[3] - verts2[2]);

    for (const auto& e1 : edges1) {
        for (const auto& e2 : edges2) {
            Vector3 axis = e1.cross(e2).normalize();
            if (axis.lengthSq() > EPSILON) { // Only add if not a zero vector
                axes.push_back(axis);
            }
        }
    }

    // Check for overlap on all axes
    for (const auto& axis : axes) {
        Projection p1 = projectTetrahedron(verts1, axis);
        Projection p2 = projectTetrahedron(verts2, axis);
        if (!checkOverlap(p1, p2)) {
            return false; // Found a separating axis, no collision
        }
    }

    return true; // No separating axis found, collision
}

// --- Spatial Partitioning Grid for efficient collision checks ---
struct GridCell {
    std::vector<int> tetrahedron_indices; // Indices of tetrahedrons in this cell
};

struct Grid {
    std::vector<GridCell> cells;
    Vector3 min_bounds;
    Vector3 max_bounds;
    Vector3 cell_size;
    int dim_x, dim_y, dim_z;

    Grid(const Vector3& container_min, const Vector3& container_max, double resolution)
        : min_bounds(container_min), max_bounds(container_max) {

        dim_x = static_cast<int>(std::ceil((max_bounds.x - min_bounds.x) / resolution));
        dim_y = static_cast<int>(std::ceil((max_bounds.y - min_bounds.y) / resolution));
        dim_z = static_cast<int>(std::ceil((max_bounds.z - min_bounds.z) / resolution));

        if (dim_x <= 0) dim_x = 1;
        if (dim_y <= 0) dim_y = 1;
        if (dim_z <= 0) dim_z = 1;

        cell_size = Vector3(
            (max_bounds.x - min_bounds.x) / dim_x,
            (max_bounds.y - min_bounds.y) / dim_y,
            (max_bounds.z - min_bounds.z) / dim_z
        );

        cells.resize(dim_x * dim_y * dim_z);
    }

    // Get grid index for a point
    Vector3 getCellCoords(const Vector3& p) const {
        int x = static_cast<int>((p.x - min_bounds.x) / cell_size.x);
        int y = static_cast<int>((p.y - min_bounds.y) / cell_size.y);
        int z = static_cast<int>((p.z - min_bounds.z) / cell_size.z);

        x = std::max(0, std::min(x, dim_x - 1));
        y = std::max(0, std::min(y, dim_y - 1));
        z = std::max(0, std::min(z, dim_z - 1));
        return Vector3(static_cast<double>(x), static_cast<double>(y), static_cast<double>(z));
    }

    int getCellIndex(const Vector3& p) const {
        Vector3 coords = getCellCoords(p);
        return static_cast<int>(coords.x + coords.y * dim_x + coords.z * dim_x * dim_y);
    }

    // Add tetrahedron to grid
    void addTetrahedron(int tetra_idx, const Tetrahedron& tetra) {
        // Find min/max cell coordinates that the tetrahedron occupies
        Vector3 tetra_min_bound = tetra.getGlobalVertices()[0];
        Vector3 tetra_max_bound = tetra.getGlobalVertices()[0];
        for (const auto& v : tetra.getGlobalVertices()) {
            tetra_min_bound.x = std::min(tetra_min_bound.x, v.x);
            tetra_min_bound.y = std::min(tetra_min_bound.y, v.y);
            tetra_min_bound.z = std::min(tetra_min_bound.z, v.z);

            tetra_max_bound.x = std::max(tetra_max_bound.x, v.x);
            tetra_max_bound.y = std::max(tetra_max_bound.y, v.y);
            tetra_max_bound.z = std::max(tetra_max_bound.z, v.z);
        }

        Vector3 min_coords = getCellCoords(tetra_min_bound);
        Vector3 max_coords = getCellCoords(tetra_max_bound);

        for (int z = static_cast<int>(min_coords.z); z <= max_coords.z; ++z) {
            for (int y = static_cast<int>(min_coords.y); y <= max_coords.y; ++y) {
                for (int x = static_cast<int>(min_coords.x); x <= max_coords.x; ++x) {
                    int idx = static_cast<int>(x + y * dim_x + z * dim_x * dim_y);
                    cells[idx].tetrahedron_indices.push_back(tetra_idx);
                }
            }
        }
    }
};

// --- Solver Logic ---

bool isValidPlacement(const Tetrahedron& new_tetra,
                     const Container& container,
                     const std::vector<Tetrahedron>& placed_tetrahedrons,
                     const Grid& grid) {
    // 1. Check collision with container boundaries
    for (const auto& v : new_tetra.getGlobalVertices()) {
        if (!container.contains(v)) {
            return false;
        }
    }

    // 2. Check collision with already placed tetrahedrons using the grid
    // Find the cells the new tetrahedron occupies
    Vector3 tetra_min_bound = new_tetra.getGlobalVertices()[0];
    Vector3 tetra_max_bound = new_tetra.getGlobalVertices()[0];
    for (const auto& v : new_tetra.getGlobalVertices()) {
            tetra_min_bound.x = std::min(tetra_min_bound.x, v.x);
            tetra_min_bound.y = std::min(tetra_min_bound.y, v.y);
            tetra_min_bound.z = std::min(tetra_min_bound.z, v.z);

            tetra_max_bound.x = std::max(tetra_max_bound.x, v.x);
            tetra_max_bound.y = std::max(tetra_max_bound.y, v.y);
            tetra_max_bound.z = std::max(tetra_max_bound.z, v.z);
    }

    Vector3 min_coords = grid.getCellCoords(tetra_min_bound);
    Vector3 max_coords = grid.getCellCoords(tetra_max_bound);

    std::vector<int> potential_collisions;
    for (int z = static_cast<int>(min_coords.z); z <= max_coords.z; ++z) {
        for (int y = static_cast<int>(min_coords.y); y <= max_coords.y; ++y) {
            for (int x = static_cast<int>(min_coords.x); x <= max_coords.x; ++x) {
                int idx = static_cast<int>(x + y * grid.dim_x + z * grid.dim_x * grid.dim_y);
                for (int existing_tetra_idx : grid.cells[idx].tetrahedron_indices) {
                    potential_collisions.push_back(existing_tetra_idx);
                }
            }
        }
    }

    // Sort and unique the indices to avoid redundant checks
    std::sort(potential_collisions.begin(), potential_collisions.end());
    potential_collisions.erase(std::unique(potential_collisions.begin(), potential_collisions.end()), potential_collisions.end());

    for (int existing_tetra_idx : potential_collisions) {
        if (checkTetrahedronCollision(new_tetra, placed_tetrahedrons[existing_tetra_idx])) {
            return false;
        }
    }

    return true;
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int V;
    double edge_length;
    std::cin >> V >> edge_length;

    std::vector<Vector3> container_vertices_input(V);
    for (int i = 0; i < V; ++i) {
        std::cin >> container_vertices_input[i].x >> container_vertices_input[i].y >> container_vertices_input[i].z;
    }

    Container container(container_vertices_input);
    std::vector<Tetrahedron> placed_tetrahedrons;

    // Initialize random number generator
    std::mt19937 rng(std::chrono::steady_clock::now().time_since_epoch().count());
    std::uniform_real_distribution<double> dist_angle(0.0, 2.0 * M_PI); // M_PI is defined in cmath

    // Initialize grid for spatial partitioning
    // Choose a resolution roughly based on tetrahedron size
    double grid_resolution = edge_length * 2.0; // At least twice the edge length
    Grid grid(container.min_bounds, container.max_bounds, grid_resolution);

    // --- Placement Strategy: Iterative Grid Sampling with Random Orientations ---
    auto start_time = std::chrono::steady_clock::now();
    const std::chrono::duration<double> TIME_LIMIT(29.0); // 29 seconds to be safe

    // Granularity of position candidates
    double position_grid_resolution = edge_length / 4.0; // Can be tuned
    int max_orientations_per_pos = 500; // Number of random orientations to try at each grid position

    // Calculate grid for position candidates over the full container bounding box
    Vector3 actual_container_size = container.max_bounds - container.min_bounds;

    int pos_dim_x = static_cast<int>(std::ceil(actual_container_size.x / position_grid_resolution));
    int pos_dim_y = static_cast<int>(std::ceil(actual_container_size.y / position_grid_resolution));
    int pos_dim_z = static_cast<int>(std::ceil(actual_container_size.z / position_grid_resolution));

    // Ensure minimum dimension is 1 for iteration, even if size is 0 or negative
    if (pos_dim_x <= 0) pos_dim_x = 1;
    if (pos_dim_y <= 0) pos_dim_y = 1;
    if (pos_dim_z <= 0) pos_dim_z = 1;

    // Loop until timeout
    while (std::chrono::steady_clock::now() - start_time < TIME_LIMIT) {
        bool placed_in_this_iteration = false;

        // Iterate through grid positions
        for (int z_idx = 0; z_idx < pos_dim_z; ++z_idx) {
            for (int y_idx = 0; y_idx < pos_dim_y; ++y_idx) {
                for (int x_idx = 0; x_idx < pos_dim_x; ++x_idx) {
                    // Calculate center of grid cell within original container bounds
                    Vector3 candidate_center(
                        container.min_bounds.x + x_idx * position_grid_resolution + position_grid_resolution / 2.0,
                        container.min_bounds.y + y_idx * position_grid_resolution + position_grid_resolution / 2.0,
                        container.min_bounds.z + z_idx * position_grid_resolution + position_grid_resolution / 2.0
                    );

                    // Try multiple random orientations for this position
                    for (int orient_attempt = 0; orient_attempt < max_orientations_per_pos; ++orient_attempt) {
                        double candidate_rx = dist_angle(rng);
                        double candidate_ry = dist_angle(rng);
                        double candidate_rz = dist_angle(rng);

                        Tetrahedron candidate_tetra(candidate_center, candidate_rx, candidate_ry, candidate_rz, edge_length);

                        if (isValidPlacement(candidate_tetra, container, placed_tetrahedrons, grid)) {
                            placed_tetrahedrons.push_back(candidate_tetra);
                            grid.addTetrahedron(static_cast<int>(placed_tetrahedrons.size()) - 1, candidate_tetra);
                            placed_in_this_iteration = true;
                            // Found a spot, break from orientation loop and restart position search (new global state)
                            goto next_iteration_after_placement; 
                        }
                    }
                }
            }
        }

        next_iteration_after_placement:;
        if (!placed_in_this_iteration) {
            // No new tetrahedrons could be placed in this full scan
            break;
        }
    }

    // Output K and the placed tetrahedrons
    std::cout << placed_tetrahedrons.size() << std::endl;
    for (const auto& tetra : placed_tetrahedrons) {
        std::cout << tetra.center.x << " " << tetra.center.y << " " << tetra.center.z << " "
                  << tetra.rx << " " << tetra.ry << " " << tetra.rz << std::endl;
    }

    return 0;
}
