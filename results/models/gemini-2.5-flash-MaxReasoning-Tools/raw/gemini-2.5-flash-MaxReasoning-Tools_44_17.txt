#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Brick pattern parameters
    vec3 brickColor = vec3(0.7, 0.2, 0.1);
    vec3 mortarColor = vec3(0.7, 0.7, 0.7);
    float mortarThicknessX = 0.05;
    float mortarThicknessY = 0.1;
    vec2 brickScale = vec2(8.0, 16.0);

    // Calculate scaled UV and apply staggering
    vec2 scaledUV = uv * brickScale;
    float row = floor(scaledUV.y);
    if (mod(row, 2.0) == 1.0) {
        scaledUV.x += 0.5; // Staggered rows
    }

    // Determine if it's mortar or brick
    vec2 brickUV = fract(scaledUV);
    vec3 finalColor = brickColor;
    if (brickUV.x < mortarThicknessX || brickUV.y < mortarThicknessY) {
        finalColor = mortarColor;
    }

    // Lambertian Shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);
    float diffuse = max(dot(N, L), 0.0);
    finalColor *= (0.4 + 0.6 * diffuse);

    outColor = vec4(finalColor, 1.0);
}
