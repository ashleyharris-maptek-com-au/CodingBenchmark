struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize the world-space normal
    float3 N = normalize(input.normal);

    // Hemisphere Ambient Lighting
    float3 skyColor = float3(0.4, 0.6, 1.0);
    float3 groundColor = float3(0.3, 0.15, 0.05);

    // Blend factor based on normal's Y component (upwards direction)
    float blendFactor = N.y * 0.5 + 0.5;
    float3 ambientColor = lerp(groundColor, skyColor, blendFactor);

    // Diffuse Term
    float3 lightDirection = normalize(lightPos.xyz - input.worldPos);
    float diffuseFactor = max(0.0f, dot(N, lightDirection));
    float3 diffuseTerm = 0.3f * diffuseFactor; // 0.3 * d as requested, assuming light color is incorporated or effectively white

    // Final color calculation
    float3 finalColorRGB = ambientColor + diffuseTerm;

    return float4(finalColorRGB, 1.0f);
}
