%entry = OpLabel
%uv_val = OpLoad %v2float %uv

; Polka dot pattern calculation
%uv_scaled = OpFmul %v2float %uv_val %float_6_0 ; Scale UV to 0-6 range
%uv_fract = OpFract %v2float %uv_scaled        ; Get fractional part (0-1 within each cell)
%uv_minus_center = OpFSub %v2float %uv_fract %float_0_5 ; Offset to center at 0.0,0.0 for radius check

%dist_sq = OpDot %float %uv_minus_center %uv_minus_center ; Squared distance from cell center
%dot_radius_sq = OpFmul %float %float_0_3 %float_0_3 ; 0.3 * 0.3 = 0.09

%is_in_dot = OpFOrdLessThan %bool %dist_sq %dot_radius_sq ; Check if inside dot radius

%base_color_vec3 = OpSelect %v3float %is_in_dot %c_yellow %c_dark_blue

; Lambertian shading
%worldPos_val = OpLoad %v3float %worldPos
%normal_val = OpLoad %v3float %normal
%normal_norm = OpNormalize %v3float %normal_val

%lightPos_vec4 = OpLoad %v4float %lightPos_ubo
%lightPos_vec3 = OpVectorShuffle %v3float %lightPos_vec4 %lightPos_vec4 0 1 2 ; Extract xyz

%light_dir = OpFSub %v3float %lightPos_vec3 %worldPos_val
%light_dir_norm = OpNormalize %v3float %light_dir

%diffuse = OpDot %float %normal_norm %light_dir_norm
%diffuse_clamped = OpFNegate %float %diffuse ; This is for light coming from inside the surface, usually OpMax %float %diffuse %float_0_0 is used. I'll assume standard dot product for now, clamped at 0.
%diffuse_clamped = OpExtInst %float %1 Max %diffuse %float_0_0 ; GLSL.std.450 Max

%shading_factor_term1 = OpFmul %float %float_0_7 %diffuse_clamped
%shading_factor = OpFAdd %float %float_0_3 %shading_factor_term1

%final_color_rgb = OpVectorTimesScalar %v3float %base_color_vec3 %shading_factor

%final_color_vec4 = OpCompositeConstruct %v4float %final_color_rgb %c_alpha_1
