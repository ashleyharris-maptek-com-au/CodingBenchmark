using System;
using System.Collections.Generic;
using System.Linq;

public class Program
{
    public class Box
    {
        public int Id { get; set; }
        public int W { get; set; }
        public int H { get; set; }
        public int D { get; set; }
        public long Volume => (long)W * H * D;

        public Box(int id, int w, int h, int d)
        {
            Id = id;
            W = w;
            H = h;
            D = d;
        }
    }

    public class PackedBox
    {
        public int BoxId { get; set; }
        public int X { get; set; }
        public int Y { get; set; }
        public int Z { get; set; }
        public int W { get; set; } // Actual placed width
        public int H { get; set; } // Actual placed height
        public int D { get; set; } // Actual placed depth

        public PackedBox(int boxId, int x, int y, int z, int w, int h, int d)
        {
            BoxId = boxId;
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }

        // Checks for overlap with another packed box
        public bool OverlapsWith(PackedBox other)
        {
            return X < other.X + other.W && X + W > other.X &&
                   Y < other.Y + other.H && Y + H > other.Y &&
                   Z < other.Z + other.D && Z + D > other.Z;
        }
    }

    public class FreeSpace
    {
        public int X { get; set; }
        public int Y { get; set; }
        public int Z { get; set; }
        public int W { get; set; }
        public int H { get; set; }
        public int D { get; set; }
        public long Volume => (long)W * H * D;

        public FreeSpace(int x, int y, int z, int w, int h, int d)
        {
            X = x;
            Y = y;
            Z = z;
            W = w;
            H = h;
            D = d;
        }

        // Checks if a box of given dimensions fits in this free space
        public bool CanFit(int boxW, int boxH, int boxD)
        {
            return boxW <= W && boxH <= H && boxD <= D;
        }

        // Returns true if this free space completely contains another free space
        public bool Contains(FreeSpace other)
        {
            return other.X >= X && other.Y >= Y && other.Z >= Z &&
                   other.X + other.W <= X + W &&
                   other.Y + other.H <= Y + H &&
                   other.Z + other.D <= Z + D;
        }
    }

    // Helper to get all 6 axis-aligned orientations for a box
    private static IEnumerable<(int w, int h, int d)> GetOrientations(Box box)
    {
        // w, h, d
        yield return (box.W, box.H, box.D);
        yield return (box.W, box.D, box.H);
        yield return (box.H, box.W, box.D);
        yield return (box.H, box.D, box.W);
        yield return (box.D, box.W, box.H);
        yield return (box.D, box.H, box.W);
    }

    public static void Main(string[] args)
    {
        string[] line1 = Console.ReadLine()?.Split(' ') ?? throw new InvalidOperationException("Input line 1 cannot be null.");
        int N = int.Parse(line1[0]);
        int containerW = int.Parse(line1[1]);
        int containerH = int.Parse(line1[2]);
        int containerD = int.Parse(line1[3]);

        List<Box> boxes = new List<Box>();
        for (int i = 0; i < N; i++)
        {
            string[] boxDims = Console.ReadLine()?.Split(' ') ?? throw new InvalidOperationException($"Input line for box {i} cannot be null.");
            boxes.Add(new Box(i, int.Parse(boxDims[0]), int.Parse(boxDims[1]), int.Parse(boxDims[2])));
        }

        // Sort boxes by decreasing volume for better packing heuristic
        boxes = boxes.OrderByDescending(b => b.Volume).ToList();

        List<PackedBox> packedBoxes = new List<PackedBox>();
        List<FreeSpace> freeSpaces = new List<FreeSpace>
        {
            new FreeSpace(0, 0, 0, containerW, containerH, containerD)
        };

        foreach (var boxToPlace in boxes)
        {
            int bestSpaceIndex = -1;
            int bestOrientW = -1, bestOrientH = -1, bestOrientD = -1;
            int bestPlacementX = -1, bestPlacementY = -1, bestPlacementZ = -1;
            long minWaste = long.MaxValue; // Try to minimize the wasted space in the chosen free space

            // Iterate through free spaces
            for (int i = 0; i < freeSpaces.Count; i++)
            {
                var currentSpace = freeSpaces[i];

                // Iterate through all orientations of the box
                foreach (var orientation in GetOrientations(boxToPlace))
                {
                    int ow = orientation.w;
                    int oh = orientation.h;
                    int od = orientation.d;

                    if (currentSpace.CanFit(ow, oh, od))
                    {
                        // Calculate the wasted volume if this box is placed in this space
                        // Using (remaining space volume - box volume)
                        long currentWaste = (long)currentSpace.W * currentSpace.H * currentSpace.D - (long)ow * oh * od;

                        // Prioritize smaller waste
                        if (currentWaste < minWaste)
                        {
                            minWaste = currentWaste;
                            bestSpaceIndex = i;
                            bestOrientW = ow;
                            bestOrientH = oh;
                            bestOrientD = od;
                            bestPlacementX = currentSpace.X;
                            bestPlacementY = currentSpace.Y;
                            bestPlacementZ = currentSpace.Z;
                        }
                    }
                }
            }

            if (bestSpaceIndex != -1) // Found a place for the box
            {
                var placedBox = new PackedBox(boxToPlace.Id, bestPlacementX, bestPlacementY, bestPlacementZ, bestOrientW, bestOrientH, bestOrientD);
                packedBoxes.Add(placedBox);

                var usedSpace = freeSpaces[bestSpaceIndex];
                freeSpaces.RemoveAt(bestSpaceIndex);

                // Generate new free spaces by splitting the usedSpace
                // This is a simplified split, creating up to 3 new spaces
                // Space 1: remaining space on X axis (right of placed box)
                if (usedSpace.W > placedBox.W)
                {
                    freeSpaces.Add(new FreeSpace(
                        usedSpace.X + placedBox.W, usedSpace.Y, usedSpace.Z,
                        usedSpace.W - placedBox.W, usedSpace.H, usedSpace.D
                    ));
                }
                // Space 2: remaining space on Y axis (above placed box)
                if (usedSpace.H > placedBox.H)
                {
                    freeSpaces.Add(new FreeSpace(
                        usedSpace.X, usedSpace.Y + placedBox.H, usedSpace.Z,
                        placedBox.W, usedSpace.H - placedBox.H, usedSpace.D
                    ));
                }
                // Space 3: remaining space on Z axis (in front of placed box)
                if (usedSpace.D > placedBox.D)
                {
                    freeSpaces.Add(new FreeSpace(
                        usedSpace.X, usedSpace.Y, usedSpace.Z + placedBox.D,
                        placedBox.W, placedBox.H, usedSpace.D - placedBox.D
                    ));
                }
                
                List<FreeSpace> newFreeSpaces = new List<FreeSpace>();
                foreach (var fs1 in freeSpaces)
                {
                    bool isContained = false;
                    foreach (var fs2 in freeSpaces)
                    {
                        if (fs1 != fs2 && fs2.Contains(fs1))
                        {
                            isContained = true;
                            break;
                        }
                    }
                    if (!isContained)
                    {
                        newFreeSpaces.Add(fs1);
                    }
                }
                freeSpaces = newFreeSpaces;
            }
        }

        Console.WriteLine(packedBoxes.Count);
        foreach (var pBox in packedBoxes)
        {
            Console.WriteLine($"{pBox.BoxId} {pBox.X} {pBox.Y} {pBox.Z}");
        }
    }
}
