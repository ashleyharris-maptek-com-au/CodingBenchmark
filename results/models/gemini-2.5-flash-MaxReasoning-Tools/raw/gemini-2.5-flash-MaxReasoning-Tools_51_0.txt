import math
import statistics

# Module-level state for persistence
_autopilot_state = None

class PIDController:
    """
    A generic PID controller.
    """
    def __init__(self, kp, ki, kd, setpoint, output_limits=(-1.0, 1.0)):
        self.kp = kp
        self.ki = ki
        self.kd = kd
        self.setpoint = setpoint
        self.output_limits = output_limits

        self.integral = 0.0
        self.prev_error = 0.0
        self.last_update_time = None

    def update(self, current_value, dt):
        error = self.setpoint - current_value

        # Proportional term
        p_term = self.kp * error

        # Integral term
        self.integral += error * dt
        i_term = self.ki * self.integral

        # Derivative term
        d_term = self.kd * (error - self.prev_error) / dt if dt > 0 else 0.0

        output = p_term + i_term + d_term

        # Apply output limits
        output = max(self.output_limits[0], min(self.output_limits[1], output))

        self.prev_error = error
        return output

class AutopilotState:
    """
    Stores persistent variables across autopilot_step calls.
    """
    def __init__(self):
        self.mode = 'OFF'  # Default mode
        self.target_alt_ft = None
        self.target_hdg_deg = None
        self.target_ias_kt = None
        self.target_vs_fpm = None

        # PID controllers for various axes
        self.pid_pitch = PIDController(kp=0.1, ki=0.01, kd=0.05, setpoint=0.0) # Placeholder gains
        self.pid_roll = PIDController(kp=0.1, ki=0.01, kd=0.05, setpoint=0.0)  # Placeholder gains
        self.pid_yaw = PIDController(kp=0.1, ki=0.01, kd=0.05, setpoint=0.0)   # Placeholder gains
        self.pid_thrust = PIDController(kp=0.01, ki=0.001, kd=0.005, setpoint=0.0, output_limits=(0.0, 1.0)) # Placeholder gains

        self.previous_values = {} # To store previous sensor readings for rate calculations
        self.engine_failure_detected = [False, False] # [engine1, engine2]
        self.trim_state = 0.0 # Initial elevator trim

def get_valid_sensor_value(state: dict, sensor_prefix: str, key_suffix: str, default_value=None):
    """
    Safely retrieve a sensor value, checking its validity flag.
    """
    valid_key = f"{sensor_prefix}_valid"
    if valid_key in state and not state[valid_key]:
        return default_value
    
    full_key = f"{sensor_prefix}{key_suffix}"
    return state.get(full_key, default_value)

def fuse_sensors(state: dict, sensor_base_name: str, key_suffix: str, num_sensors: int, tolerance=None):
    """
    Fuse redundant sensor data, handling invalid flags and discrepancies.
    Returns the median of valid, non-outlier readings.
    """
    readings = []
    for i in range(1, num_sensors + 1):
        prefix = f"{sensor_base_name}_{i}"
        value = get_valid_sensor_value(state, prefix, key_suffix)
        if value is not None:
            readings.append(value)
    
    if not readings:
        return None
    
    if tolerance is not None and len(readings) > 2:
        # Simple outlier detection: remove values far from the median
        median_val = statistics.median(readings)
        filtered_readings = [r for r in readings if abs(r - median_val) <= tolerance]
        if filtered_readings:
            return statistics.median(filtered_readings)
        else:
            return median_val # If all filtered out, return original median
    
    return statistics.median(readings)

def autopilot_step(state: dict, dt: float) -> dict:
    global _autopilot_state

    if _autopilot_state is None:
        _autopilot_state = AutopilotState()

    output_controls = {}

    # --- 1. Data Preprocessing and Sensor Fusion ---
    fused_data = {}

    # Altitude
    fused_data['baro_alt_ft'] = fuse_sensors(state, 'baro_alt', '_ft', 3, tolerance=500) # 500 ft tolerance
    fused_data['radar_alt_ft'] = fuse_sensors(state, 'radar_alt', '_ft', 3, tolerance=50) # 50 ft tolerance
    fused_data['gps_alt_ft'] = fuse_sensors(state, 'gps', '_alt_ft', 3, tolerance=100)
    # Placeholder for IRS altitude (complex, might involve lat/lon to geoid)
    # fused_data['irs_alt_ft'] = ... 

    # Prioritize altitude sources
    if fused_data['radar_alt_ft'] is not None and fused_data['radar_alt_ft'] < 2500: # Below 2500 ft, trust radar alt more
        fused_data['current_alt_ft'] = fused_data['radar_alt_ft']
    elif fused_data['baro_alt_ft'] is not None:
        fused_data['current_alt_ft'] = fused_data['baro_alt_ft']
    elif fused_data['gps_alt_ft'] is not None:
        fused_data['current_alt_ft'] = fused_data['gps_alt_ft']
    else:
        fused_data['current_alt_ft'] = None # No reliable altitude

    # Airspeed
    fused_data['ias_kt'] = fuse_sensors(state, 'ias', '_kt', 3, tolerance=10) # 10 kt tolerance
    fused_data['mach'] = fuse_sensors(state, 'mach', '', 3, tolerance=0.02) # 0.02 Mach tolerance
    fused_data['tas_kt'] = fuse_sensors(state, 'tas', '_kt', 3, tolerance=15) # 15 kt tolerance

    # Attitude
    fused_data['pitch_deg'] = fuse_sensors(state, 'irs', '_pitch_deg', 3, tolerance=2.0)
    fused_data['roll_deg'] = fuse_sensors(state, 'irs', '_roll_deg', 3, tolerance=2.0)
    fused_data['heading_deg'] = fuse_sensors(state, 'irs', '_heading_deg', 3, tolerance=5.0)

    # Vertical Speed
    fused_data['vs_fpm'] = fuse_sensors(state, 'irs', '_vs_fpm', 3, tolerance=500)

    # AoA
    fused_data['aoa_deg'] = fuse_sensors(state, 'aoa', '_deg', 3, tolerance=1.0)

    # Gyro rates
    fused_data['pitch_rate_dps'] = fuse_sensors(state, 'gyro', '_pitch_rate_dps', 3, tolerance=0.5)
    fused_data['roll_rate_dps'] = fuse_sensors(state, 'gyro', '_roll_rate_dps', 3, tolerance=0.5)
    fused_data['yaw_rate_dps'] = fuse_sensors(state, 'gyro', '_yaw_rate_dps', 3, tolerance=0.5)

    # Engine status
    fused_data['engine_1_running'] = state.get('engine_1_running', False)
    fused_data['engine_2_running'] = state.get('engine_2_running', False)

    # --- 2. Autopilot Mode Selection ---
    ap_engaged = state.get('ap_engaged', False)
    at_engaged = state.get('at_engaged', False)

    if not ap_engaged:
        _autopilot_state.mode = 'OFF'
        # Reset PID controllers when AP is disengaged
        _autopilot_state.pid_pitch.integral = 0.0
        _autopilot_state.pid_pitch.prev_error = 0.0
        _autopilot_state.pid_roll.integral = 0.0
        _autopilot_state.pid_roll.prev_error = 0.0
        _autopilot_state.pid_yaw.integral = 0.0
        _autopilot_state.pid_yaw.prev_error = 0.0
        _autopilot_state.pid_thrust.integral = 0.0
        _autopilot_state.pid_thrust.prev_error = 0.0
    else:
        # Update targets
        _autopilot_state.target_alt_ft = state.get('target_alt_ft', fused_data['current_alt_ft'])
        _autopilot_state.target_hdg_deg = state.get('target_hdg_deg', fused_data['heading_deg'])
        _autopilot_state.target_ias_kt = state.get('target_ias_kt', fused_data['ias_kt'])
        _autopilot_state.target_vs_fpm = state.get('target_vs_fpm', fused_data['vs_fpm'])

        # Simple mode logic (can be expanded)
        if _autopilot_state.target_alt_ft is not None and fused_data['current_alt_ft'] is not None:
            if abs(_autopilot_state.target_alt_ft - fused_data['current_alt_ft']) > 100: # If far from target alt, go to ALT_HOLD
                _autopilot_state.mode = 'ALT_HOLD'
                _autopilot_state.pid_pitch.setpoint = 0.0 # Aim for level flight initially
            else: # Within 100 ft, fine-tune
                _autopilot_state.mode = 'ALT_HOLD'
        elif _autopilot_state.target_hdg_deg is not None:
            _autopilot_state.mode = 'HDG_HOLD'
        elif _autopilot_state.target_ias_kt is not None:
            _autopilot_state.mode = 'IAS_HOLD'
        elif _autopilot_state.target_vs_fpm is not None:
            _autopilot_state.mode = 'VS_HOLD'
        else:
            _autopilot_state.mode = 'OFF' # Should not happen if ap_engaged is true

    # --- 3. Calculate Control Efforts based on current_mode and fused_data ---
    elevator_cmd = 0.0
    aileron_cmd = 0.0
    rudder_cmd = 0.0
    throttle_cmd = 0.0

    if _autopilot_state.mode == 'ALT_HOLD' and fused_data['current_alt_ft'] is not None:
        alt_error = _autopilot_state.target_alt_ft - fused_data['current_alt_ft']
        # Simple P-controller for vertical speed target
        vs_target = alt_error * 0.5 # Proportional to altitude error
        vs_target = max(-2000, min(2000, vs_target)) # Limit vertical speed target

        if fused_data['vs_fpm'] is not None:
            _autopilot_state.pid_pitch.setpoint = vs_target # PID for pitch to achieve VS target
            elevator_cmd = _autopilot_state.pid_pitch.update(fused_data['vs_fpm'], dt)
        
        # Thrust for altitude hold (climb/descent)
        if at_engaged and fused_data['ias_kt'] is not None:
            # Maintain target IAS during climb/descent
            _autopilot_state.pid_thrust.setpoint = _autopilot_state.target_ias_kt if _autopilot_state.target_ias_kt is not None else 250 # Default to 250 KIAS
            throttle_cmd = _autopilot_state.pid_thrust.update(fused_data['ias_kt'], dt)
            
    elif _autopilot_state.mode == 'HDG_HOLD' and fused_data['heading_deg'] is not None:
        hdg_error = _autopilot_state.target_hdg_deg - fused_data['heading_deg']
        # Normalize heading error to -180 to 180
        if hdg_error > 180:
            hdg_error -= 360
        elif hdg_error < -180:
            hdg_error += 360
        
        # Simple P-controller for roll target
        roll_target = -hdg_error * 0.5 # Proportional to heading error, negative for left turn
        roll_target = max(-30, min(30, roll_target)) # Limit roll target to +/- 30 degrees

        if fused_data['roll_deg'] is not None:
            _autopilot_state.pid_roll.setpoint = roll_target
            aileron_cmd = _autopilot_state.pid_roll.update(fused_data['roll_deg'], dt)
        
        # Maintain current altitude and airspeed during heading changes
        if fused_data['vs_fpm'] is not None:
            _autopilot_state.pid_pitch.setpoint = 0.0 # Maintain level flight
            elevator_cmd = _autopilot_state.pid_pitch.update(fused_data['vs_fpm'], dt)
        
        if at_engaged and fused_data['ias_kt'] is not None:
            _autopilot_state.pid_thrust.setpoint = _autopilot_state.target_ias_kt if _autopilot_state.target_ias_kt is not None else 250
            throttle_cmd = _autopilot_state.pid_thrust.update(fused_data['ias_kt'], dt)

    elif _autopilot_state.mode == 'IAS_HOLD' and fused_data['ias_kt'] is not None:
        if at_engaged:
            _autopilot_state.pid_thrust.setpoint = _autopilot_state.target_ias_kt if _autopilot_state.target_ias_kt is not None else 250
            throttle_cmd = _autopilot_state.pid_thrust.update(fused_data['ias_kt'], dt)
        
        # Maintain current altitude
        if fused_data['vs_fpm'] is not None:
            _autopilot_state.pid_pitch.setpoint = 0.0
            elevator_cmd = _autopilot_state.pid_pitch.update(fused_data['vs_fpm'], dt)

    elif _autopilot_state.mode == 'VS_HOLD' and fused_data['vs_fpm'] is not None:
        _autopilot_state.pid_pitch.setpoint = _autopilot_state.target_vs_fpm if _autopilot_state.target_vs_fpm is not None else 0.0
        elevator_cmd = _autopilot_state.pid_pitch.update(fused_data['vs_fpm'], dt)

        if at_engaged and fused_data['ias_kt'] is not None:
            _autopilot_state.pid_thrust.setpoint = _autopilot_state.target_ias_kt if _autopilot_state.target_ias_kt is not None else 250
            throttle_cmd = _autopilot_state.pid_thrust.update(fused_data['ias_kt'], dt)

    # Yaw dampening (always active when AP is engaged)
    if ap_engaged and fused_data['yaw_rate_dps'] is not None:
        _autopilot_state.pid_yaw.setpoint = 0.0 # Aim for zero yaw rate
        rudder_cmd = _autopilot_state.pid_yaw.update(fused_data['yaw_rate_dps'], dt)

    # --- 4. Aerodynamic Protections (Stall, Vne) ---
    # Stall protection
    if fused_data['aoa_deg'] is not None and fused_data['ias_kt'] is not None:
        # Placeholder values for critical AoA and stall speed
        CRITICAL_AOA = 15.0 # degrees
        STALL_SPEED_KT = 120 # knots
        
        if fused_data['aoa_deg'] > CRITICAL_AOA - 2 or fused_data['ias_kt'] < STALL_SPEED_KT + 10: # Approaching stall
            elevator_cmd = max(elevator_cmd, -0.5) # Push nose down
            throttle_cmd = max(throttle_cmd, 0.8) # Increase thrust
            # Potentially disengage altitude hold if it's preventing recovery
            if _autopilot_state.mode == 'ALT_HOLD':
                _autopilot_state.mode = 'OFF' # Or a dedicated 'STALL_RECOVERY' mode

    # Vne protection (Vne is ~365 KIAS / Mach 0.82)
    VNE_IAS = 365
    VNE_MACH = 0.82
    
    if fused_data['ias_kt'] is not None and fused_data['mach'] is not None:
        if fused_data['ias_kt'] > VNE_IAS - 20 or fused_data['mach'] > VNE_MACH - 0.05: # Approaching Vne
            elevator_cmd = min(elevator_cmd, 0.5) # Pull nose up
            throttle_cmd = min(throttle_cmd, 0.2) # Reduce thrust
            output_controls['speedbrake'] = 0.5 # Deploy speedbrakes
            # Potentially disengage altitude hold if it's preventing recovery
            if _autopilot_state.mode == 'ALT_HOLD':
                _autopilot_state.mode = 'OFF' # Or a dedicated 'OVERSPEED_RECOVERY' mode

    # Asymmetric thrust handling
    if not fused_data['engine_1_running'] and fused_data['engine_2_running']:
        # Left engine failed, apply right rudder
        rudder_cmd = max(rudder_cmd, 0.3) # Placeholder value
        throttle_cmd_1 = 0.0
        throttle_cmd_2 = throttle_cmd # Use calculated throttle for working engine
    elif fused_data['engine_1_running'] and not fused_data['engine_2_running']:
        # Right engine failed, apply left rudder
        rudder_cmd = min(rudder_cmd, -0.3) # Placeholder value
        throttle_cmd_1 = throttle_cmd # Use calculated throttle for working engine
        throttle_cmd_2 = 0.0
    else:
        # Both engines running or both failed (handled by other logic)
        throttle_cmd_1 = throttle_cmd
        throttle_cmd_2 = throttle_cmd

    # --- 5. Output Mapping and Limiting ---
    output_controls['elevator'] = max(-1.0, min(1.0, elevator_cmd))
    output_controls['aileron'] = max(-1.0, min(1.0, aileron_cmd))
    output_controls['rudder'] = max(-1.0, min(1.0, rudder_cmd))
    output_controls['throttle_1'] = max(0.0, min(1.0, throttle_cmd_1))
    output_controls['throttle_2'] = max(0.0, min(1.0, throttle_cmd_2))

    # Elevator trim (simple proportional to elevator command to reduce steady state error)
    _autopilot_state.trim_state += elevator_cmd * 0.01 * dt # Adjust trim slowly
    _autopilot_state.trim_state = max(-1.0, min(1.0, _autopilot_state.trim_state))
    output_controls['elevator_trim'] = _autopilot_state.trim_state

    # Flaps and Gear logic (simplified)
    output_controls['flaps'] = 0 # Default to flaps up
    output_controls['gear'] = 0 # Default to gear up

    if fused_data['current_alt_ft'] is not None and fused_data['ias_kt'] is not None:
        if fused_data['current_alt_ft'] < 2000 and fused_data['ias_kt'] < 200:
            output_controls['gear'] = 1 # Gear down for approach/landing
            output_controls['flaps'] = 20 # Flaps for approach

        if fused_data['current_alt_ft'] < 500 and fused_data['ias_kt'] < 160:
            output_controls['flaps'] = 40 # Full flaps for landing

    return output_controls
