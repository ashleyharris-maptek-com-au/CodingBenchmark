%main = OpFunction %void None %void
%main_label = OpLabel
  %i = OpVariable %_ptr_Function_int Function
  OpStore %i %int_0

  ; Loop for 5 sub-segments
  OpBranch %loop_header
%loop_header = OpLabel
  %loop_cond = OpSLessThan %bool (OpLoad %int %i) %int_5
  OpLoopMerge %loop_merge %loop_continue %loop_cond
  OpBranchConditional %loop_cond %loop_body %loop_merge

%loop_body = OpLabel
    %t_int = OpLoad %int %i
    %t_float = OpConvertUToF %float %t_int
    %t = OpFDiv %float %t_float %float_4_0

    ; Load P0, P1, P2, P3 from gl_in
    %idx_0 = OpAccessChain %_ptr_Input_v4float %gl_in %int_0 %int_0
    %P0_v4 = OpLoad %v4float %idx_0
    %idx_1 = OpAccessChain %_ptr_Input_v4float %gl_in %int_1 %int_0
    %P1_v4 = OpLoad %v4float %idx_1
    %idx_2 = OpAccessChain %_ptr_Input_v4float %gl_in %int_2 %int_0
    %P2_v4 = OpLoad %v4float %idx_2
    %idx_3 = OpAccessChain %_ptr_Input_v4float %gl_in %int_3 %int_0
    %P3_v4 = OpLoad %v4float %idx_3

    ; P_current = P1 * (1-t) + P2 * t
    %one_minus_t = OpFSub %float %float_1 %t
    %P1_scaled = OpVectorTimesScalar %v4float %P1_v4 %one_minus_t
    %P2_scaled = OpVectorTimesScalar %v4float %P2_v4 %t
    %P_current = OpFAdd %v4float %P1_scaled %P2_scaled

    ; Tangent calculation using xyz components
    %P0_xyz = OpVectorShuffle %v3float %P0_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2
    %P1_xyz = OpVectorShuffle %v3float %P1_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2
    %P2_xyz = OpVectorShuffle %v3float %P2_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2
    %P3_xyz = OpVectorShuffle %v3float %P3_v4 %v4float_0_0_0_0 %uint_0 %uint_1 %uint_2

    %tangent_start = OpFSub %v3float %P2_xyz %P0_xyz
    %tangent_end = OpFSub %v3float %P3_xyz %P1_xyz
    %tangent_start_scaled = OpVectorTimesScalar %v3float %tangent_start %one_minus_t
    %tangent_end_scaled = OpVectorTimesScalar %v3float %tangent_end %t
    %tangent_current = OpFAdd %v3float %tangent_start_scaled %tangent_end_scaled
    %normalized_tangent_current = OpNormalize %v3float %tangent_current

    ; Perpendicular calculation
    ; A simple, robust perpendicular for a 3D vector V:
    ; P = cross(V, (0,1,0)) if V is not (0,1,0) parallel
    ; else P = cross(V, (0,0,1))
    ; To avoid branching, we can try to derive a perpendicular.
    ; For a tangent T, a vector orthogonal to T and (0,1,0) is cross(T, (0,1,0)).
    ; If T is parallel to (0,1,0), this would be zero.
    ; Let's use a simpler approach assuming the curve is mostly on XY.
    %up_vector = OpConstantComposite %v3float %float_0 %float_0 %float_1
    %perp_direction_raw = OpCrossProduct %v3float %normalized_tangent_current %up_vector
    %perp_direction = OpNormalize %v3float %perp_direction_raw
    %perp_scaled_xyz = OpVectorTimesScalar %v3float %perp_direction %float_0_02
    %perp_scaled = OpCompositeConstruct %v4float %perp_scaled_xyz %float_0

    ; Color interpolation
    %color_1_ptr = OpAccessChain %_ptr_Input_v3float %color_in %int_1
    %color_1 = OpLoad %v3float %color_1_ptr
    %color_2_ptr = OpAccessChain %_ptr_Input_v3float %color_in %int_2
    %color_2 = OpLoad %v3float %color_2_ptr
    %current_color_scaled_1 = OpVectorTimesScalar %v3float %color_1 %one_minus_t
    %current_color_scaled_2 = OpVectorTimesScalar %v3float %color_2 %t
    %current_color = OpFAdd %v3float %current_color_scaled_1 %current_color_scaled_2

    ; Emit vertices: P_current + perp_scaled
    %pos_plus_perp = OpFAdd %v4float %P_current %perp_scaled
    OpStore %gl_Position_out %pos_plus_perp
    OpStore %outColor_out %current_color
    OpEmitVertex

    ; Emit vertices: P_current - perp_scaled
    %pos_minus_perp = OpFSub %v4float %P_current %perp_scaled
    OpStore %gl_Position_out %pos_minus_perp
    OpStore %outColor_out %current_color
    OpEmitVertex

%loop_continue = OpLabel
  %next_i = OpIAdd %int %t_int %int_1
  OpStore %i %next_i
  OpBranch %loop_header

%loop_merge = OpLabel
  OpEndPrimitive
  OpReturn
OpFunctionEnd
