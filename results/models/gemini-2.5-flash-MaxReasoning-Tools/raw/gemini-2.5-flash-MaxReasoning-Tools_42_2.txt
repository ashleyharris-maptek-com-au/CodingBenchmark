OpCapability Geometry
OpMemoryModel Logical GLSL450
OpEntryPoint Geometry %main "main" %gl_in %worldPos %normal %color %gl_Position %outColor
OpExecutionMode %main InputPoints
OpExecutionMode %main OutputTriangleStrip
OpExecutionMode %main OutputVertices %uint_18
OpExecutionMode %main Invocations %uint_1

; Types
%void = OpTypeVoid
%void_fn = OpTypeFunction %void

%float = OpTypeFloat 32
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%mat4float = OpTypeMatrix %float 4 4

%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%uint_0 = OpConstant %uint 0
%uint_1 = OpConstant %uint 1
%uint_18 = OpConstant %uint 18 ; OutputVertices

%ptr_input_v4 = OpTypePointer Input %v4float
%ptr_output_v4 = OpTypePointer Output %v4float
%ptr_output_v3 = OpTypePointer Output %v3float

; UBO structure (kept for completeness as per problem, but not accessed in this specific shader task)
; struct UBO {
;     mat4 model;
;     mat4 view;
;     mat4 proj;
;     vec4 lightPos;
;     vec4 cameraPos;
;     vec4 params;
; };
%type_ubo = OpTypeStruct %mat4float %mat4float %mat4float %v4float %v4float %v4float
OpMemberDecorate %type_ubo 0 Offset 0
OpMemberDecorate %type_ubo 1 Offset 64
OpMemberDecorate %type_ubo 2 Offset 128
OpMemberDecorate %type_ubo 3 Offset 192
OpMemberDecorate %type_ubo 4 Offset 208
OpMemberDecorate %type_ubo 5 Offset 224
OpDecorate %type_ubo Block
%ptr_uniform_ubo = OpTypePointer Uniform %type_ubo
%ubo_var = OpVariable %ptr_uniform_ubo Uniform DescriptorSet 0 Binding 0

; gl_PerVertex_in
%gl_PerVertex_in_type = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_in_type 0 BuiltIn Position
OpDecorate %gl_PerVertex_in_type Block

; Input array for gl_PerVertex_in (size 1 for InputPoints)
%arr_gl_PerVertex_in_1 = OpTypeArray %gl_PerVertex_in_type %uint_1
%ptr_input_gl_PerVertex_in = OpTypePointer Input %arr_gl_PerVertex_in_1
%gl_in = OpVariable %ptr_input_gl_PerVertex_in Input BuiltIn Position

; gl_PerVertex_out
%gl_PerVertex_out_type = OpTypeStruct %v4float
OpMemberDecorate %gl_PerVertex_out_type 0 BuiltIn Position
OpDecorate %gl_PerVertex_out_type Block
%ptr_output_gl_PerVertex_out = OpTypePointer Output %gl_PerVertex_out_type
%gl_Position = OpVariable %ptr_output_gl_PerVertex_out Output BuiltIn Position

; User-defined inputs
%arr_v3float_1 = OpTypeArray %v3float %uint_1
%ptr_input_v3 = OpTypePointer Input %arr_v3float_1

%worldPos = OpVariable %ptr_input_v3 Input
OpDecorate %worldPos Location 0
%normal = OpVariable %ptr_input_v3 Input
OpDecorate %normal Location 1
%color = OpVariable %ptr_input_v3 Input
OpDecorate %color Location 2

; User-defined output
%outColor = OpVariable %ptr_output_v3 Output
OpDecorate %outColor Location 0

%main = OpFunction %void None %void_fn
%entry = OpLabel
  ; Read input position and color
  %input_gl_pos_ptr = OpAccessChain %ptr_input_v4 %gl_in %uint_0 %uint_0
  %center_pos_in = OpLoad %v4float %input_gl_pos_ptr

  %input_color_ptr = OpAccessChain %ptr_input_v3 %color %uint_0
  %in_color = OpLoad %v3float %input_color_ptr

  ; Constants for hexagon calculations
  %radius = OpConstant %float 0.04
  %f_0_0 = OpConstant %float 0.0
  %f_1_0 = OpConstant %float 1.0

  ; Trigonometric constants for 0, 60, 120, 180, 240, 300 degrees
  %cos0 = OpConstant %float 1.0
  %sin0 = OpConstant %float 0.0
  %cos60 = OpConstant %float 0.5
  %sin60 = OpConstant %float 0.866025404
  %cos120 = OpConstant %float -0.5
  %sin120 = OpConstant %float 0.866025404
  %cos180 = OpConstant %float -1.0
  %sin180 = OpConstant %float 0.0
  %cos240 = OpConstant %float -0.5
  %sin240 = OpConstant %float -0.866025404
  %cos300 = OpConstant %float 0.5
  %sin300 = OpConstant %float -0.866025404

  ; Store calculated edge offsets (radius * cos/sin)
  %edge_dx_0 = OpFMul %float %radius %cos0
  %edge_dy_0 = OpFMul %float %radius %sin0
  %edge_dx_60 = OpFMul %float %radius %cos60
  %edge_dy_60 = OpFMul %float %radius %sin60
  %edge_dx_120 = OpFMul %float %radius %cos120
  %edge_dy_120 = OpFMul %float %radius %sin120
  %edge_dx_180 = OpFMul %float %radius %cos180
  %edge_dy_180 = OpFMul %float %radius %sin180
  %edge_dx_240 = OpFMul %float %radius %cos240
  %edge_dy_240 = OpFMul %float %radius %sin240
  %edge_dx_300 = OpFMul %float %radius %cos300
  %edge_dy_300 = OpFMul %float %radius %sin300

  ; Edge color (slightly darker)
  %darken_factor = OpConstant %float 0.8
  %edge_color = OpVectorTimesScalar %v3float %in_color %darken_factor

  ; Extract components of center_pos_in for calculations
  %center_x = OpCompositeExtract %float %center_pos_in 0
  %center_y = OpCompositeExtract %float %center_pos_in 1
  %center_z = OpCompositeExtract %float %center_pos_in 2
  %center_w = OpCompositeExtract %float %center_pos_in 3

  ; Calculate edge positions
  ; E0
  %edge_pos_x_0 = OpFAdd %float %center_x %edge_dx_0
  %edge_pos_y_0 = OpFAdd %float %center_y %edge_dy_0
  %edge_pos_v4_0 = OpCompositeConstruct %v4float %edge_pos_x_0 %edge_pos_y_0 %center_z %center_w
  ; E60
  %edge_pos_x_60 = OpFAdd %float %center_x %edge_dx_60
  %edge_pos_y_60 = OpFAdd %float %center_y %edge_dy_60
  %edge_pos_v4_60 = OpCompositeConstruct %v4float %edge_pos_x_60 %edge_pos_y_60 %center_z %center_w
  ; E120
  %edge_pos_x_120 = OpFAdd %float %center_x %edge_dx_120
  %edge_pos_y_120 = OpFAdd %float %center_y %edge_dy_120
  %edge_pos_v4_120 = OpCompositeConstruct %v4float %edge_pos_x_120 %edge_pos_y_120 %center_z %center_w
  ; E180
  %edge_pos_x_180 = OpFAdd %float %center_x %edge_dx_180
  %edge_pos_y_180 = OpFAdd %float %center_y %edge_dy_180
  %edge_pos_v4_180 = OpCompositeConstruct %v4float %edge_pos_x_180 %edge_pos_y_180 %center_z %center_w
  ; E240
  %edge_pos_x_240 = OpFAdd %float %center_x %edge_dx_240
  %edge_pos_y_240 = OpFAdd %float %center_y %edge_dy_240
  %edge_pos_v4_240 = OpCompositeConstruct %v4float %edge_pos_x_240 %edge_pos_y_240 %center_z %center_w
  ; E300
  %edge_pos_x_300 = OpFAdd %float %center_x %edge_dx_300
  %edge_pos_y_300 = OpFAdd %float %center_y %edge_dy_300
  %edge_pos_v4_300 = OpCompositeConstruct %v4float %edge_pos_x_300 %edge_pos_y_300 %center_z %center_w

  %gl_Position_out_ptr = OpAccessChain %ptr_output_v4 %gl_Position %uint_0

  ; Triangle 1: center, E0, E60
  OpStore %gl_Position_out_ptr %center_pos_in
  OpStore %outColor %in_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_0
  OpStore %outColor %edge_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_60
  OpStore %outColor %edge_color
  OpEmitVertex
  OpEndPrimitive

  ; Triangle 2: center, E60, E120
  OpStore %gl_Position_out_ptr %center_pos_in
  OpStore %outColor %in_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_60
  OpStore %outColor %edge_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_120
  OpStore %outColor %edge_color
  OpEmitVertex
  OpEndPrimitive

  ; Triangle 3: center, E120, E180
  OpStore %gl_Position_out_ptr %center_pos_in
  OpStore %outColor %in_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_120
  OpStore %outColor %edge_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_180
  OpStore %outColor %edge_color
  OpEmitVertex
  OpEndPrimitive

  ; Triangle 4: center, E180, E240
  OpStore %gl_Position_out_ptr %center_pos_in
  OpStore %outColor %in_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_180
  OpStore %outColor %edge_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_240
  OpStore %outColor %edge_color
  OpEmitVertex
  OpEndPrimitive

  ; Triangle 5: center, E240, E300
  OpStore %gl_Position_out_ptr %center_pos_in
  OpStore %outColor %in_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_240
  OpStore %outColor %edge_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_300
  OpStore %outColor %edge_color
  OpEmitVertex
  OpEndPrimitive

  ; Triangle 6: center, E300, E0 (closing)
  OpStore %gl_Position_out_ptr %center_pos_in
  OpStore %outColor %in_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_300
  OpStore %outColor %edge_color
  OpEmitVertex

  OpStore %gl_Position_out_ptr %edge_pos_v4_0
  OpStore %outColor %edge_color
  OpEmitVertex
  OpEndPrimitive

  OpReturn
OpFunctionEnd
