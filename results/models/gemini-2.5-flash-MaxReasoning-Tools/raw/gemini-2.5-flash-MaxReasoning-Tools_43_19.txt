struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;
    [[vk::location(1)]] float3 normal   : TEXCOORD1;
    [[vk::location(2)]] float2 uv       : TEXCOORD2;
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;
    [[vk::location(4)]] float3 color    : TEXCOORD4;
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;
    float4x4 view;
    float4x4 projection;
    float4 lightPos;
    float4 cameraPos;
    float4 params;
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction (from fragment to light)
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // View direction (from fragment to camera)
    float3 V = normalize(cameraPos.xyz - input.worldPos);

    // Ambient component
    float3 ambientColor = float3(0.1, 0.05, 0.05);
    float3 ambient = ambientColor;

    // Diffuse component
    float diffuseStrength = max(dot(N, L), 0.0);
    float3 diffuse = input.color * diffuseStrength;

    // Specular component
    float3 R = reflect(-L, N); // reflected light vector
    float shininess = 32.0;
    float specularStrength = pow(max(dot(R, V), 0.0), shininess);
    float3 specularColor = float3(1.0, 1.0, 1.0);
    float3 specular = specularColor * specularStrength;

    // Combine components
    float3 finalColor = ambient + diffuse + specular;

    // Clamp the sum to [0,1]
    finalColor = saturate(finalColor); // saturate clamps to [0,1]

    // Output with alpha 1.0
    return float4(finalColor, 1.0);
}
