use std::io::{self, BufRead};
use std::collections::HashSet;

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read num_vertices and num_edges
    let first_line = lines.next().expect("Expected graph dimensions").unwrap();
    let parts: Vec<usize> = first_line
        .split_whitespace()
        .map(|s| s.parse().expect("Invalid number format"))
        .collect();
    let num_vertices = parts[0];
    let num_edges = parts[1];

    // Build adjacency list
    let mut adj: Vec<Vec<usize>> = vec![vec![]; num_vertices];
    for _ in 0..num_edges {
        let line = lines.next().expect("Expected edge").unwrap();
        let edge_parts: Vec<usize> = line
            .split_whitespace()
            .map(|s| s.parse().expect("Invalid number format"))
            .collect();
        let u = edge_parts[0];
        let v = edge_parts[1];
        adj[u].push(v);
        adj[v].push(u);
    }

    let mut dominating_set: HashSet<usize> = HashSet::new();
    let mut dominated_vertices: Vec<bool> = vec![false; num_vertices];
    let mut undominated_count = num_vertices;

    while undominated_count > 0 {
        let mut best_vertex: usize = 0;
        let mut max_dominated_this_turn: usize = 0;

        // Find the vertex that dominates the most new undominated vertices
        for i in 0..num_vertices {
            if dominating_set.contains(&i) {
                continue; // Skip if already in dominating set
            }

            let mut current_dominated_count = 0;
            // Count 'i' itself if it's undominated
            if !dominated_vertices[i] {
                current_dominated_count += 1;
            }

            // Count undominated neighbors
            for &neighbor in &adj[i] {
                if !dominated_vertices[neighbor] {
                    current_dominated_count += 1;
                }
            }

            // If a better vertex is found, update best_vertex and max_dominated_this_turn
            // Using '>' ensures that if multiple vertices have the same max_dominated_this_turn,
            // the one with the smallest index is chosen, which is fine for greedy.
            if current_dominated_count > max_dominated_this_turn {
                max_dominated_this_turn = current_dominated_count;
                best_vertex = i;
            }
        }
        
        // This break statement is a safety measure.
        // It prevents infinite loops if, due to some unexpected state,
        // no vertex can dominate any new undominated vertices,
        // although with correct logic and undominated_count > 0, this should not happen.
        // If max_dominated_this_turn is 0, it means no new vertices can be dominated.
        // This implies an issue with graph structure or logic if undominated_count > 0.
        if max_dominated_this_turn == 0 && undominated_count > 0 {
             break; 
        }

        // Add the best_vertex to the dominating set and update dominated vertices
        dominating_set.insert(best_vertex);

        // Update count for best_vertex itself
        if !dominated_vertices[best_vertex] {
            dominated_vertices[best_vertex] = true;
            undominated_count -= 1;
        }

        // Update count for neighbors of best_vertex
        for &neighbor in &adj[best_vertex] {
            if !dominated_vertices[neighbor] {
                dominated_vertices[neighbor] = true;
                undominated_count -= 1;
            }
        }
    }

    // Output results
    println!("{}", dominating_set.len());
    let mut sorted_set: Vec<usize> = dominating_set.into_iter().collect();
    sorted_set.sort_unstable();
    for (i, v) in sorted_set.iter().enumerate() {
        print!("{}", v);
        if i < sorted_set.len() - 1 {
            print!(" ");
        }
    }
    println!();
}
