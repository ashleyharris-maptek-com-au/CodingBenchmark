struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64
    float4x4 projection;     // offset 128
    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0 {
    // Polka dot pattern
    float2 uv_scaled = input.uv * 6.0; // Scale UV to 6x6 grid
    float2 cell_coords = frac(uv_scaled); // Coordinates within a single cell [0,1]

    // Center of each cell in [0,1] space is (0.5, 0.5)
    float2 center_dist = cell_coords - float2(0.5, 0.5);
    float dist_from_center = length(center_dist);

    float radius = 0.3;
    float3 pattern_color;
    if (dist_from_center < radius) {
        pattern_color = float3(1.0, 0.8, 0.0); // Yellow
    } else {
        pattern_color = float3(0.1, 0.1, 0.3); // Dark blue
    }

    // Lambertian shading
    float3 normal = normalize(input.normal);
    float3 lightDir = normalize(lightPos.xyz - input.worldPos);
    
    // Calculate diffuse component
    float diffuse = max(0.0, dot(normal, lightDir));
    
    // Apply Lambertian factor: (0.3 + 0.7 * diffuse)
    float lighting_factor = 0.3 + 0.7 * diffuse;

    // Final color
    float4 final_color = float4(pattern_color * lighting_factor, 1.0);

    return final_color;
}
