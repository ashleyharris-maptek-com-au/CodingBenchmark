# autopilot.py

import math

# Module-level variables for persistence across calls
_last_state = None
_integral_pos_y = 0.0
_integral_pos_x = 0.0
_integral_pos_z = 0.0
_last_timestamp = 0.0 # Placeholder for time tracking, assume 0 for now.

# PID gains - these will need tuning
# Y-axis (along-track) control for closing distance
KP_POS_Y_FAR = -0.005  # Proportional gain for position Y (far away)
KD_VEL_Y_FAR = -0.2   # Derivative gain for velocity Y (far away)
KP_POS_Y_MED = -0.05
KD_VEL_Y_MED = -0.5
KP_POS_Y_CLOSE = -0.1
KD_VEL_Y_CLOSE = -0.8

# X-axis (radial) control
KP_POS_X = -0.1
KD_VEL_X = -0.5

# Z-axis (cross-track) control
KP_POS_Z = -0.1
KD_VEL_Z = -0.5

# Attitude control gains
KP_ATTITUDE = 0.001

# Max thrust and attitude rates
MAX_THRUST = 1.0
MAX_ATT_RATE = 0.035 # rad/s

# Approach speeds for different phases
TARGET_VEL_Y_FAR = -0.5 # m/s
TARGET_VEL_Y_MED = -0.2 # m/s
TARGET_VEL_Y_CLOSE = -0.05 # m/s
TARGET_VEL_Y_FINAL = -0.01 # m/s

# Thresholds for docking criteria
DOCK_RANGE_THRESHOLD = 2.0
DOCK_SPEED_THRESHOLD = 0.3

def get_best_relative_pos_vel(state: dict):
    """
    Fuses sensor data to get the best estimate of relative position and velocity.
    Prioritizes Lidar > Radar. Combines data if one sensor provides valid pos and another valid vel.
    Returns (pos_x, pos_y, pos_z, vel_x, vel_y, vel_z) or (None, ...) if no valid 3D data.
    """
    fused_pos_x, fused_pos_y, fused_pos_z = None, None, None
    fused_vel_x, fused_vel_y, fused_vel_z = None, None, None

    # --- Lidar Data ---
    lidar_pos_valid = state.get('lidar_rel_x_valid', False) and 
                      state.get('lidar_rel_y_valid', False) and 
                      state.get('lidar_rel_z_valid', False)
    lidar_vel_valid = state.get('lidar_relv_x_valid', False) and 
                      state.get('lidar_relv_y_valid', False) and 
                      state.get('lidar_relv_z_valid', False)

    if lidar_pos_valid:
        fused_pos_x = state['lidar_rel_pos_x_m']
        fused_pos_y = state['lidar_rel_pos_y_m']
        fused_pos_z = state['lidar_rel_pos_z_m']
    
    if lidar_vel_valid:
        fused_vel_x = state['lidar_rel_vel_x_ms']
        fused_vel_y = state['lidar_rel_vel_y_ms']
        fused_vel_z = state['lidar_rel_vel_z_ms']

    # --- Radar Data (if Lidar is incomplete for pos or vel) ---
    radar_pos_valid = state.get('radar_rel_x_valid', False) and 
                      state.get('radar_rel_y_valid', False) and 
                      state.get('radar_rel_z_valid', False)
    radar_vel_valid = state.get('radar_relv_x_valid', False) and 
                      state.get('radar_relv_y_valid', False) and 
                      state.get('radar_relv_z_valid', False)

    if fused_pos_x is None and radar_pos_valid: # Only use radar pos if lidar pos was not valid
        fused_pos_x = state['radar_rel_pos_x_m']
        fused_pos_y = state['radar_rel_pos_y_m']
        fused_pos_z = state['radar_rel_pos_z_m']
    
    if fused_vel_x is None and radar_vel_valid: # Only use radar vel if lidar vel was not valid
        fused_vel_x = state['radar_rel_vel_x_ms']
        fused_vel_y = state['radar_rel_vel_y_ms']
        fused_vel_z = state['radar_rel_vel_z_ms']
    
    # Check if we have at least 3D position and velocity from any source
    if all(v is not None for v in [fused_pos_x, fused_pos_y, fused_pos_z, fused_vel_x, fused_vel_y, fused_vel_z]):
        return fused_pos_x, fused_pos_y, fused_pos_z, fused_vel_x, fused_vel_y, fused_vel_z
    
    # If we don't have a complete 3D pos/vel estimate, return Nones
    return None, None, None, None, None, None


def autopilot_step(state: dict, dt: float) -> dict:
    global _last_state, _integral_pos_y, _integral_pos_x, _integral_pos_z, _last_timestamp

    # Initialize outputs
    thrust_x, thrust_y, thrust_z = 0.0, 0.0, 0.0
    att_rate_roll, att_rate_pitch, att_rate_yaw = 0.0, 0.0, 0.0

    # Sensor Fusion
    pos_x, pos_y, pos_z, vel_x, vel_y, vel_z = get_best_relative_pos_vel(state)

    # If no valid position or velocity, do nothing (or try to stabilize)
    if pos_y is None or vel_y is None:
        # If no reliable 3D relative position/velocity, prioritize stability by issuing no thrusts
        # and continue with attitude control if applicable.
        
        # Attitude Control (pointing towards docking port if camera is active)
        if state.get('dock_camera_active', False) and 
           state.get('dock_camera_bearing_x_valid', False) and 
           state.get('dock_camera_bearing_y_valid', False):
            att_rate_pitch = -state['dock_camera_bearing_y'] * KP_ATTITUDE
            att_rate_yaw = state['dock_camera_bearing_x'] * KP_ATTITUDE
            
            att_rate_pitch = max(-MAX_ATT_RATE, min(MAX_ATT_RATE, att_rate_pitch))
            att_rate_yaw = max(-MAX_ATT_RATE, min(MAX_ATT_RATE, att_rate_yaw))

        _last_state = state
        _last_timestamp = state.get('timestamp', 0)
        return {
            'thrust_x': 0.0,
            'thrust_y': 0.0,
            'thrust_z': 0.0,
            'att_rate_roll': att_rate_roll,
            'att_rate_pitch': att_rate_pitch,
            'att_rate_yaw': att_rate_yaw,
        }

    # Calculate overall range for phase determination
    current_range = math.sqrt(pos_x**2 + pos_y**2 + pos_z**2)
    # The current_range value is critical. If for some reason pos_x,y,z were valid
    # but range finders were not, this would still be valid.
    
    # --- Phase-based control ---
    phase = 'far'
    if current_range <= 10.0:
        phase = 'final'
    elif current_range <= 50.0:
        phase = 'close'
    elif current_range <= 200.0:
        phase = 'medium'
    
    # Update integral terms (these are simple and might need bounding/resetting in a full PID)
    _integral_pos_y += pos_y * dt
    _integral_pos_x += pos_x * dt
    _integral_pos_z += pos_z * dt

    # Y-axis (along-track) control: Primary for closing distance
    # Use target_dock_range_m and target_dock_speed_ms as setpoints where applicable
    target_rel_pos_y = -state.get('target_dock_range_m', 0.0) # Target range should ideally be along the y-axis (approach)
    target_rel_vel_y = -state.get('target_dock_speed_ms', TARGET_VEL_Y_FINAL) # Target approach speed

    # Adjust target_rel_pos_y based on phase for smooth transition
    if phase == 'far':
        # In far phase, we focus on closing the large y-gap, not precise docking range
        target_rel_pos_y = -20.0 # A reasonable intermediate target behind the station to aim for
        target_rel_vel_y = TARGET_VEL_Y_FAR
        kp_y, kd_y = KP_POS_Y_FAR, KD_VEL_Y_FAR
        # Special logic for far phase: if behind (neg y), thrust retrograde to speed up
        # If ahead (pos y), thrust prograde to slow down
        if pos_y < -5.0: # If significantly behind
            # Apply a thrust proportional to distance to target velocity
            thrust_y = kp_y * (pos_y - target_rel_pos_y) + kd_y * (vel_y - target_rel_vel_y)
        elif pos_y > 5.0: # If significantly ahead
            # Apply prograde thrust to raise orbit and slow down
            thrust_y = kp_y * (pos_y - target_rel_pos_y) + kd_y * (vel_y - target_rel_vel_y)
        else: # Near Y target, try to match velocity
            thrust_y = kd_y * (vel_y - target_rel_vel_y)

    elif phase == 'medium':
        target_rel_pos_y = -5.0 # A closer intermediate target
        target_rel_vel_y = TARGET_VEL_Y_MED
        kp_y, kd_y = KP_POS_Y_MED, KD_VEL_Y_MED
        thrust_y = kp_y * (pos_y - target_rel_pos_y) + kd_y * (vel_y - target_rel_vel_y)

    elif phase == 'close':
        target_rel_pos_y = -1.0 # Very close target
        target_rel_vel_y = TARGET_VEL_Y_CLOSE
        kp_y, kd_y = KP_POS_Y_CLOSE, KD_VEL_Y_CLOSE
        thrust_y = kp_y * (pos_y - target_rel_pos_y) + kd_y * (vel_y - target_rel_vel_y)

    elif phase == 'final':
        # In the final phase, use the actual target_dock_range_m and target_dock_speed_ms
        target_rel_pos_y = -state.get('target_dock_range_m', 0.0)
        target_rel_vel_y = -state.get('target_dock_speed_ms', TARGET_VEL_Y_FINAL)
        
        # Use very small gains for final approach
        kp_y, kd_y = KP_POS_Y_CLOSE * 0.1, KD_VEL_Y_CLOSE * 0.1 # Reduced gains for fine control
        thrust_y = kp_y * (pos_y - target_rel_pos_y) + kd_y * (vel_y - target_rel_vel_y)

        # Ensure extremely slow final approach
        # If current velocity is too high, aggressively reduce it
        if abs(vel_y) > DOCK_SPEED_THRESHOLD / 2.0:
            thrust_y = min(MAX_THRUST, max(-MAX_THRUST, -vel_y * 1.5)) # Apply opposing thrust stronger

    # X-axis (radial) control
    thrust_x = KP_POS_X * pos_x + KD_VEL_X * vel_x

    # Z-axis (cross-track) control
    thrust_z = KP_POS_Z * pos_z + KD_VEL_Z * vel_z
    
    # Clamp thrust values
    thrust_x = max(-MAX_THRUST, min(MAX_THRUST, thrust_x))
    thrust_y = max(-MAX_THRUST, min(MAX_THRUST, thrust_y))
    thrust_z = max(-MAX_THRUST, min(MAX_THRUST, thrust_z))

    # Attitude Control (pointing towards docking port if camera is active)
    if state.get('dock_camera_active', False) and 
       state.get('dock_camera_bearing_x_valid', False) and 
       state.get('dock_camera_bearing_y_valid', False): # Assuming Z is always forward for now
        
        # Bearings are deviations from desired orientation. Apply proportional control.
        # Assuming bearing_x affects pitch, bearing_y affects yaw.
        # This is a simplification and might need adjustment based on vehicle's actual orientation.
        att_rate_pitch = -state['dock_camera_bearing_y'] * KP_ATTITUDE # Y bearing often maps to pitch
        att_rate_yaw = state['dock_camera_bearing_x'] * KP_ATTITUDE   # X bearing often maps to yaw
        
        # Clamp attitude rates
        att_rate_pitch = max(-MAX_ATT_RATE, min(MAX_ATT_RATE, att_rate_pitch))
        att_rate_yaw = max(-MAX_ATT_RATE, min(MAX_ATT_RATE, att_rate_yaw))

    # Store current state for next iteration (e.g., for derivative calculations if not using sensor vel)
    _last_state = state
    _last_timestamp = state.get('timestamp', 0) # Update dummy timestamp

    return {
        'thrust_x': thrust_x,
        'thrust_y': thrust_y,
        'thrust_z': thrust_z,
        'att_rate_roll': att_rate_roll,
        'att_rate_pitch': att_rate_pitch,
        'att_rate_yaw': att_rate_yaw,
    }
