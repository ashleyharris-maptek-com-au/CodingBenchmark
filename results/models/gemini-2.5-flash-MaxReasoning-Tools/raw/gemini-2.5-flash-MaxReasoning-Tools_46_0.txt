#version 450

layout(local_size_x = 256) in;

// Each body is stored as two consecutive vec4s:
//   vec4(position.xyz, mass)
//   vec4(velocity.xyz, 0.0)
// So body i starts at index i*2 in the buffer.

layout(set = 0, binding = 0) readonly buffer InputBodies {
    vec4 inData[];   // length = N * 2
};

layout(set = 0, binding = 1) buffer OutputBodies {
    vec4 outData[];  // length = N * 2
};

layout(set = 0, binding = 2) uniform Params {
    uvec4 counts;    // counts.x = N (number of bodies)
    vec4 physics;    // physics.x = dt, physics.y = softening, physics.z = G
};

void main() {
    uint N = counts.x;
    float dt = physics.x;
    float softeningSq = physics.y * physics.y; // softening is passed, but we need softening^2
    float G = physics.z;

    uint i = gl_GlobalInvocationID.x;

    if (i >= N) {
        return;
    }

    // Read current body's data
    vec4 posMass_i = inData[i * 2];
    vec3 pos_i = posMass_i.xyz;
    float mass_i = posMass_i.w;

    vec3 vel_i = inData[i * 2 + 1].xyz;

    vec3 acceleration = vec3(0.0);

    for (uint j = 0; j < N; ++j) {
        if (i == j) {
            continue; // Skip self-interaction
        }

        vec4 posMass_j = inData[j * 2];
        vec3 pos_j = posMass_j.xyz;
        float mass_j = posMass_j.w;

        vec3 r_vec = pos_j - pos_i;
        float distSq = dot(r_vec, r_vec) + softeningSq;
        float distInv = inversesqrt(distSq); // 1 / sqrt(distSq)
        float distInv3 = distInv * distInv * distInv; // 1 / dist^3

        acceleration += G * mass_j * r_vec * distInv3;
    }

    // Update velocity (Leapfrog / Euler integration)
    vec3 new_vel_i = vel_i + acceleration * dt;

    // Update position (Leapfrog / Euler integration)
    vec3 new_pos_i = pos_i + new_vel_i * dt;

    // Write updated data to output buffer
    outData[i * 2] = vec4(new_pos_i, mass_i);
    outData[i * 2 + 1] = vec4(new_vel_i, 0.0);
}
