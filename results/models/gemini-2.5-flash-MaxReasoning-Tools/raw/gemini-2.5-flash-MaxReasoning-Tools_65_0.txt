interface Task {
    id: string;
    duration: number;
    deps: string[];
    deadline: number;
    weight: number;
}

interface ScheduleEntry {
    id: string;
    start: number;
    machine: number;
}

interface InputData {
    machineCount: number;
    tasks: Task[];
}

interface OutputSchedule {
    schedule: ScheduleEntry[];
}

interface OutputOrder {
    order: string[];
}

// Function to perform topological sort
function topologicalSort(tasks: Task[]): string[] | null {
    const adj: Map<string, string[]> = new Map(); // Adjacency list for graph
    const inDegree: Map<string, number> = new Map(); // In-degree of each node
    const taskMap: Map<string, Task> = new Map(); // Map task ID to Task object

    // Initialize adj, inDegree, and taskMap
    for (const task of tasks) {
        adj.set(task.id, []);
        inDegree.set(task.id, 0);
        taskMap.set(task.id, task);
    }

    // Build graph and calculate in-degrees
    for (const task of tasks) {
        for (const depId of task.deps) {
            if (!taskMap.has(depId)) {
                // Dependency task not found, this indicates an invalid input or external dependency.
                // For this benchmark, we assume all dependencies are within the provided task list.
                // Depending on requirements, one might throw an error or log a warning here.
                // For now, we'll skip invalid dependencies to prevent crashes.
                continue;
            }
            adj.get(depId)?.push(task.id);
            inDegree.set(task.id, (inDegree.get(task.id) || 0) + 1);
        }
    }

    // Initialize queue with tasks that have no dependencies
    const queue: string[] = [];
    for (const [taskId, degree] of inDegree.entries()) {
        if (degree === 0) {
            queue.push(taskId);
        }
    }

    const result: string[] = [];
    let head = 0;
    while (head < queue.length) {
        const u = queue[head++];
        result.push(u);

        // For each neighbor of u, decrease its in-degree
        for (const v of adj.get(u) || []) {
            inDegree.set(v, (inDegree.get(v) || 0) - 1);
            if (inDegree.get(v) === 0) {
                queue.push(v);
            }
        }
    }

    // If the number of tasks in the result is not equal to the total number of tasks, a cycle was detected.
    if (result.length !== tasks.length) {
        return null; // Cycle detected
    }

    return result;
}

// Main program logic
declare const require: any;
const fs = require('fs');
const input: string = fs.readFileSync(0, 'utf8');
const data: InputData = JSON.parse(input);

const tasks = data.tasks;
const machineCount = data.machineCount;

const orderedTaskIds = topologicalSort(tasks);

if (orderedTaskIds === null) {
    console.error("Error: Cycle detected in task dependencies. Cannot schedule.");
    process.exit(1);
}

// Create a map for quick access to Task objects by ID
const taskMap = new Map<string, Task>();
for (const task of tasks) {
    taskMap.set(task.id, task);
}

// Track when each task finishes
const taskCompletionTime = new Map<string, number>();
// Track when each machine becomes free
const machineFreeTime = new Array<number>(machineCount).fill(0);

const schedule: ScheduleEntry[] = [];

// Iterate through tasks in topological order
for (const taskId of orderedTaskIds) {
    const task = taskMap.get(taskId)!;

    // Determine when this task can start due to dependencies
    let earliestStartTimeDueToDeps = 0;
    for (const depId of task.deps) {
        if (taskCompletionTime.has(depId)) {
            earliestStartTimeDueToDeps = Math.max(earliestStartTimeDueToDeps, taskCompletionTime.get(depId)!);
        }
    }

    // Find the machine that can start this task earliest
    let bestMachineIndex = -1;
    let earliestMachineAvailableTime = Infinity;

    for (let i = 0; i < machineCount; i++) {
        // The task can't start before its dependencies are met AND the machine is free
        const machineAvailableTime = Math.max(machineFreeTime[i], earliestStartTimeDueToDeps);
        
        // This greedy choice aims to fill the machines as early as possible
        if (machineAvailableTime < earliestMachineAvailableTime) {
            earliestMachineAvailableTime = machineAvailableTime;
            bestMachineIndex = i;
        }
    }

    if (bestMachineIndex === -1) {
        // This should not happen if machineCount is > 0 and tasks are schedulable
        console.error("Error: No suitable machine found for task " + taskId);
        process.exit(1);
    }

    // Schedule the task
    const startTime = earliestMachineAvailableTime;
    const completionTime = startTime + task.duration;

    schedule.push({
        id: task.id,
        start: startTime,
        machine: bestMachineIndex
    });

    // Update machine free time and task completion time
    machineFreeTime[bestMachineIndex] = completionTime;
    taskCompletionTime.set(task.id, completionTime);
}

const output: OutputSchedule = {
    schedule: schedule
};

console.log(JSON.stringify(output));
