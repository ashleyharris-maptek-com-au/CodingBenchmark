using System;
using System.Collections.Generic;
using System.Linq;

public struct Vector3
{
    public float X, Y, Z;

    public Vector3(float x, float y, float z)
    {
        X = x;
        Y = y;
        Z = z;
    }

    public static Vector3 operator +(Vector3 a, Vector3 b) => new Vector3(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
    public static Vector3 operator -(Vector3 a, Vector3 b) => new Vector3(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
    public static Vector3 operator *(Vector3 a, float s) => new Vector3(a.X * s, a.Y * s, a.Z * s);
    public static Vector3 operator /(Vector3 a, float s) => new Vector3(a.X / s, a.Y / s, a.Z / s);
    public float Dot(Vector3 other) => X * other.X + Y * other.Y + Z * other.Z;
    public Vector3 Cross(Vector3 other) => new Vector3(Y * other.Z - Z * other.Y, Z * other.X - X * other.Z, X * other.Y - Y * other.X);
    public float LengthSquared() => X * X + Y * Y + Z * Z;
    public float Length() => (float)Math.Sqrt(LengthSquared());
    public Vector3 Normalize()
    {
        float l = Length();
        return l == 0 ? new Vector3(0, 0, 0) : this / l;
    }
    public override string ToString() => $"({X}, {Y}, {Z})";
}

public struct Quaternion
{
    public float W, X, Y, Z;

    public Quaternion(float w, float x, float y, float z)
    {
        W = w;
        X = x;
        Y = y;
        Z = z;
    }

    public static Quaternion Identity => new Quaternion(1, 0, 0, 0);

    public float LengthSquared() => W * W + X * X + Y * Y + Z * Z;
    public float Length() => (float)Math.Sqrt(LengthSquared());
    public Quaternion Normalize()
    {
        float l = Length();
        return l == 0 ? new Quaternion(1, 0, 0, 0) : new Quaternion(W / l, X / l, Y / l, Z / l);
    }

    public Quaternion Conjugate() => new Quaternion(W, -X, -Y, -Z);

    public static Quaternion operator *(Quaternion a, Quaternion b)
    {
        return new Quaternion(
            a.W * b.W - a.X * b.X - a.Y * b.Y - a.Z * b.Z,
            a.W * b.X + a.X * b.W + a.Y * b.Z - a.Z * b.Y,
            a.W * b.Y - a.X * b.Z + a.Y * b.W + a.Z * b.X,
            a.W * b.Z + a.X * b.Y - a.Y * b.X + a.Z * b.W
        );
    }

    public static Vector3 operator *(Quaternion q, Vector3 v)
    {
        Quaternion p = new Quaternion(0, v.X, v.Y, v.Z);
        Quaternion qInv = q.Conjugate();
        Quaternion result = q * p * qInv;
        return new Vector3(result.X, result.Y, result.Z);
    }

    public static Quaternion FromAxisAngle(Vector3 axis, float angle)
    {
        float halfAngle = angle * 0.5f;
        float s = (float)Math.Sin(halfAngle);
        return new Quaternion((float)Math.Cos(halfAngle), axis.X * s, axis.Y * s, axis.Z * s).Normalize();
    }
    public override string ToString() => $"({W}, {X}, {Y}, {Z})";
}

public struct AABB
{
    public Vector3 Min;
    public Vector3 Max;

    public AABB(Vector3 min, Vector3 max)
    {
        Min = min;
        Max = max;
    }

    public bool Intersects(AABB other)
    {
        return Min.X <= other.Max.X && Max.X >= other.Min.X &&
               Min.Y <= other.Max.Y && Max.Y >= other.Min.Y &&
               Min.Z <= other.Max.Z && Max.Z >= other.Min.Z;
    }

    public bool Contains(Vector3 point)
    {
        return point.X >= Min.X && point.X <= Max.X &&
               point.Y >= Min.Y && point.Y <= Max.Y &&
               point.Z >= Min.Z && point.Z <= Max.Z;
    }

    public bool IsCompletelyInside(AABB container)
    {
        return Min.X >= container.Min.X && Max.X <= container.Max.X &&
               Min.Y >= container.Min.Y && Max.Y <= container.Max.Y &&
               Min.Z >= container.Min.Z && Max.Z <= container.Max.Z;
    }
}

public class Polyhedron
{
    public Vector3[] OriginalVertices { get; private set; } // Store original vertices for transformations
    public Vector3[] TransformedVertices { get; private set; } // Store current transformed vertices
    public int[][] Faces { get; private set; }

    public AABB CurrentAABB { get; private set; }

    public Polyhedron(Vector3[] vertices, int[][] faces)
    {
        OriginalVertices = vertices;
        Faces = faces;
        TransformedVertices = (Vector3[])vertices.Clone(); // Start with original
        UpdateCurrentAABB();
    }

    private void UpdateCurrentAABB()
    {
        if (TransformedVertices == null || TransformedVertices.Length == 0)
        {
            CurrentAABB = new AABB(new Vector3(0, 0, 0), new Vector3(0, 0, 0));
            return;
        }

        Vector3[] vertices = TransformedVertices; // Local variable to help compiler with null analysis

        float minX = vertices[0].X, minY = vertices[0].Y, minZ = vertices[0].Z;
        float maxX = vertices[0].X, maxY = vertices[0].Y, maxZ = vertices[0].Z;

        for (int i = 1; i < vertices.Length; i++)
        {
            Vector3 v = vertices[i];
            if (v.X < minX) minX = v.X;
            if (v.Y < minY) minY = v.Y;
            if (v.Z < minZ) minZ = v.Z;
            if (v.X > maxX) maxX = v.X;
            if (v.Y > maxY) maxY = v.Y;
            if (v.Z > maxZ) maxZ = v.Z;
        }

        CurrentAABB = new AABB(new Vector3(minX, minY, minZ), new Vector3(maxX, maxY, maxZ));
    }

    // Applies a transformation to the polyhedron's vertices
    public Polyhedron Transform(Vector3 translation, Quaternion rotation)
    {
        Vector3[] newTransformedVertices = new Vector3[OriginalVertices.Length];
        for (int i = 0; i < OriginalVertices.Length; i++)
        {
            newTransformedVertices[i] = (rotation * OriginalVertices[i]) + translation;
        }
        Polyhedron newPoly = new Polyhedron(OriginalVertices, Faces);
        newPoly.TransformedVertices = newTransformedVertices;
        newPoly.UpdateCurrentAABB();
        return newPoly;
    }
}

public struct PackedPolyhedron
{
    public Polyhedron Polyhedron { get; }
    public Vector3 Translation { get; }
    public Quaternion Rotation { get; }

    public PackedPolyhedron(Polyhedron polyhedron, Vector3 translation, Quaternion rotation)
    {
        Polyhedron = polyhedron;
        Translation = translation;
        Rotation = rotation;
    }

    public override string ToString()
    {
        return $"{Translation.X:F3} {Translation.Y:F3} {Translation.Z:F3} {Rotation.W:F3} {Rotation.X:F3} {Rotation.Y:F3} {Rotation.Z:F3}";
    }
}


public static class Collision
{
    // Check if a polyhedron's AABB is completely within the container AABB
    public static bool IsPolyhedronAABBInsideContainer(Polyhedron p, Vector3 containerSize)
    {
        AABB containerAABB = new AABB(new Vector3(0, 0, 0), containerSize);
        return p.CurrentAABB.IsCompletelyInside(containerAABB);
    }

    // Check for AABB intersection between two polyhedra
    public static bool IntersectsAABB(Polyhedron p1, Polyhedron p2)
    {
        return p1.CurrentAABB.Intersects(p2.CurrentAABB);
    }

    // Placeholder for actual SAT collision detection.
    // For now, it will just use AABB intersection.
    // Full SAT implementation is complex and will be added if necessary for higher accuracy.
    public static bool Intersects(Polyhedron p1, Polyhedron p2)
    {
        return IntersectsAABB(p1, p2);
    }
}


public class Program
{
    static void Main(string[] args)
    {
        // Read Polyhedron definition
        string[] line1 = Console.ReadLine().Split(' ');
        int vCount = int.Parse(line1[0]);
        int fCount = int.Parse(line1[1]);

        Vector3[] vertices = new Vector3[vCount];
        for (int i = 0; i < vCount; i++)
        {
            string[] coords = Console.ReadLine().Split(' ');
            vertices[i] = new Vector3(float.Parse(coords[0]), float.Parse(coords[1]), float.Parse(coords[2]));
        }

        int[][] faces = new int[fCount][];
        for (int i = 0; i < fCount; i++)
        {
            string[] indices = Console.ReadLine().Split(' ');
            faces[i] = new int[indices.Length];
            for (int j = 0; j < indices.Length; j++)
            {
                faces[i][j] = int.Parse(indices[j]);
            }
        }

        Polyhedron itemPolyhedron = new Polyhedron(vertices, faces);

        // Read Container dimensions
        string[] containerDims = Console.ReadLine().Split(' '); 
        float containerW = float.Parse(containerDims[0]);
        float containerH = float.Parse(containerDims[1]);
        float containerD = float.Parse(containerDims[2]);
        Vector3 containerSize = new Vector3(containerW, containerH, containerD);

        List<PackedPolyhedron> packedItems = new List<PackedPolyhedron>();

        // Define a set of standard rotations (90-degree increments around primary axes)
        List<Quaternion> rotationOptions = GetRotationOptions();

        // Generate all potential placement points once
        List<Vector3> allPlacementPoints = GeneratePlacementPoints(containerSize, itemPolyhedron.CurrentAABB);
        // Sort placement points to encourage a "bottom-left-fill" strategy
        allPlacementPoints.Sort((a, b) =>
        {
            int cmpX = a.X.CompareTo(b.X);
            if (cmpX != 0) return cmpX;
            int cmpY = a.Y.CompareTo(b.Y);
            if (cmpY != 0) return cmpY;
            return a.Z.CompareTo(b.Z);
        });

        bool itemPlaced;
        do
        {
            itemPlaced = false;
            foreach (Vector3 placementOrigin in allPlacementPoints)
            {
                foreach (Quaternion rotation in rotationOptions)
                {
                    Polyhedron transformedItem = itemPolyhedron.Transform(placementOrigin, rotation);

                    if (!Collision.IsPolyhedronAABBInsideContainer(transformedItem, containerSize))
                    {
                        continue;
                    }

                    bool collidesWithPacked = false;
                    foreach (PackedPolyhedron packed in packedItems)
                    {
                        if (Collision.Intersects(transformedItem, packed.Polyhedron))
                        {
                            collidesWithPacked = true;
                            break;
                        }
                    }

                    if (!collidesWithPacked)
                    {
                        packedItems.Add(new PackedPolyhedron(transformedItem, placementOrigin, rotation));
                        itemPlaced = true;
                        break; // Break from rotations loop, a placement found for current item
                    }
                }
                if (itemPlaced)
                {
                    break; // Break from placementPoints loop, a placement found for current item
                }
            }
        } while (itemPlaced);

        Console.WriteLine(packedItems.Count);
        foreach (PackedPolyhedron packed in packedItems)
        {
            Console.WriteLine(packed.ToString());
        }
    }

    private static List<Quaternion> GetRotationOptions()
    {
        List<Quaternion> rotations = new List<Quaternion>();
        // Identity (no rotation)
        rotations.Add(Quaternion.Identity);

        // 90-degree rotations around X-axis
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(1, 0, 0), (float)Math.PI / 2));
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(1, 0, 0), (float)Math.PI));
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(1, 0, 0), (float)Math.PI * 3 / 2));

        // 90-degree rotations around Y-axis
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(0, 1, 0), (float)Math.PI / 2));
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(0, 1, 0), (float)Math.PI));
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(0, 1, 0), (float)Math.PI * 3 / 2));

        // 90-degree rotations around Z-axis
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(0, 0, 1), (float)Math.PI / 2));
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(0, 0, 1), (float)Math.PI));
        rotations.Add(Quaternion.FromAxisAngle(new Vector3(0, 0, 1), (float)Math.PI * 3 / 2));

        // For shapes that are not highly symmetrical, one might need to combine rotations
        // For example, rotate around X, then Y, then Z. This quickly expands the number of rotations to check.
        // For now, these 9 rotations plus identity cover the primary axis-aligned orientations.
        // A full set of 24 rotations for a general cuboid (e.g., Box packing) would involve:
        // X-rotations (0, 90, 180, 270) * Y-rotations (0, 90, 180, 270) * Z-rotations (0, 90, 180, 270)
        // But many of these are redundant for simple shapes or covered by earlier rotations if done systematically.
        // The current set handles 6 primary face orientations.

        return rotations;
    }

    private static List<Vector3> GeneratePlacementPoints(Vector3 containerSize, AABB itemAABB)
    {
        List<Vector3> points = new List<Vector3>();

        // Determine grid density based on container size and item dimensions.
        // This is a crucial area for optimization and adaptive logic.
        float itemWidth = itemAABB.Max.X - itemAABB.Min.X;
        float itemHeight = itemAABB.Max.Y - itemAABB.Min.Y;
        float itemDepth = itemAABB.Max.Z - itemAABB.Min.Z;

        // Use a finer step for smaller containers/items, coarser for larger ones.
        // The goal is to keep the number of points manageable.
        float stepX = Math.Max(0.1f, itemWidth / 5f);
        float stepY = Math.Max(0.1f, itemHeight / 5f);
        float stepZ = Math.Max(0.1f, itemDepth / 5f);

        // Scale step size based on container dimensions
        stepX = Math.Min(stepX, containerSize.X / 20f);
        stepY = Math.Min(stepY, containerSize.Y / 20f);
        stepZ = Math.Min(stepZ, containerSize.Z / 20f);

        if (containerSize.X > 100) stepX = Math.Max(stepX, containerSize.X / 100f);
        if (containerSize.Y > 100) stepY = Math.Max(stepY, containerSize.Y / 100f);
        if (containerSize.Z > 100) stepZ = Math.Max(stepZ, containerSize.Z / 100f);

        // Ensure step sizes are reasonable and don't create an excessive number of points
        stepX = Math.Max(0.1f, stepX);
        stepY = Math.Max(0.1f, stepY);
        stepZ = Math.Max(0.1f, stepZ);

        // Iterate through a grid within the container, considering item dimensions
        for (float x = 0; x <= containerSize.X - itemWidth; x += stepX)
        {
            for (float y = 0; y <= containerSize.Y - itemHeight; y += stepY)
            {
                for (float z = 0; z <= containerSize.Z - itemDepth; z += stepZ)
                {
                    points.Add(new Vector3(x, y, z));
                }
            }
        }
        return points;
    }
}
