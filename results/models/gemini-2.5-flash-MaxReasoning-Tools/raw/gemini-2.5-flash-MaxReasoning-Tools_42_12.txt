; --- Main Geometry Shader Logic ---
; Emit original triangle with dimmed colors
%v0_pos_ptr = OpAccessChain %ptr_in_v4float %gl_in %int_0 %int_0

  OpCall %void %emit_vertex %v0_pos %v0_color_dim
  OpCall %void %emit_vertex %v2_pos %v2_color_dim
  OpCall %void %emit_vertex %v4_pos %v4_color_dim
  OpEndPrimitive

; Emit thick lines for all 3 edges (simplified approach)
; Edge 0-2 (vertices 0 and 2 from gl_in)
; Edge 2-4 (vertices 2 and 4 from gl_in)
; Edge 4-0 (vertices 4 and 0 from gl_in)

; Thickness of 0.015, half thickness 0.0075

; Edge 0-2
%edge02_dir = OpFSub %v4float %v2_pos %v0_pos
%edge02_perp_xy = OpCompositeConstruct %v2float
    (OpVectorExtractDynamic %float %edge02_dir %uint_1) ; -dir.y
    (OpVectorExtractDynamic %float %edge02_dir %uint_0) ; dir.x
    ; Invert y-component for perpendicular
%neg_edge02_dir_y = OpFNegate %float (OpVectorExtractDynamic %float %edge02_dir %uint_1)
%edge02_perp_xy_corrected = OpCompositeConstruct %v2float %neg_edge02_dir_y (OpVectorExtractDynamic %float %edge02_dir %uint_0)

; Normalize perpendicular vector
%edge02_perp_len_sq = OpDot %float %edge02_perp_xy_corrected %edge02_perp_xy_corrected
%edge02_perp_len = OpExtInst %float %GLSLstd450 Sqrt %edge02_perp_len_sq
%edge02_perp_norm_x = OpFDiv %float (OpVectorExtractDynamic %float %edge02_perp_xy_corrected %uint_0) %edge02_perp_len
%edge02_perp_norm_y = OpFDiv %float (OpVectorExtractDynamic %float %edge02_perp_xy_corrected %uint_1) %edge02_perp_len
%edge02_perp_norm_v2 = OpCompositeConstruct %v2float %edge02_perp_norm_x %edge02_perp_norm_y

; Scale by half thickness
%edge02_offset_xy = OpFMul %v2float %edge02_perp_norm_v2 %float_0_0075

; Construct 4-component offset vectors
%offset02_p = OpCompositeConstruct %v4float
    (OpVectorExtractDynamic %float %edge02_offset_xy %uint_0) ; x
    (OpVectorExtractDynamic %float %edge02_offset_xy %uint_1) ; y
    %float_0_0 ; z
    %float_0_0 ; w
%offset02_n = OpFNegate %v4float %offset02_p

; Emit quad for edge 0-2
; v0 - offset, v0 + offset, v2 - offset, v2 + offset
%v0_pos_m_off = OpFSub %v4float %v0_pos %offset02_n
%v0_pos_p_off = OpFAdd %v4float %v0_pos %offset02_n
%v2_pos_m_off = OpFSub %v4float %v2_pos %offset02_n
%v2_pos_p_off = OpFAdd %v4float %v2_pos %offset02_n

  OpCall %void %emit_vertex %v0_pos_m_off %v3float_1_1_1
  OpCall %void %emit_vertex %v0_pos_p_off %v3float_1_1_1
  OpCall %void %emit_vertex %v2_pos_m_off %v3float_1_1_1
  OpCall %void %emit_vertex %v2_pos_p_off %v3float_1_1_1
  OpEndPrimitive

; Edge 2-4
%edge24_dir = OpFSub %v4float %v4_pos %v2_pos
%neg_edge24_dir_y = OpFNegate %float (OpVectorExtractDynamic %float %edge24_dir %uint_1)
%edge24_perp_xy_corrected = OpCompositeConstruct %v2float %neg_edge24_dir_y (OpVectorExtractDynamic %float %edge24_dir %uint_0)

%edge24_perp_len_sq = OpDot %float %edge24_perp_xy_corrected %edge24_perp_xy_corrected
%edge24_perp_len = OpExtInst %float %GLSLstd450 Sqrt %edge24_perp_len_sq
%edge24_perp_norm_x = OpFDiv %float (OpVectorExtractDynamic %float %edge24_perp_xy_corrected %uint_0) %edge24_perp_len
%edge24_perp_norm_y = OpFDiv %float (OpVectorExtractDynamic %float %edge24_perp_xy_corrected %uint_1) %edge24_perp_len
%edge24_perp_norm_v2 = OpCompositeConstruct %v2float %edge24_perp_norm_x %edge24_perp_norm_y

%edge24_offset_xy = OpFMul %v2float %edge24_perp_norm_v2 %float_0_0075

%offset24_p = OpCompositeConstruct %v4float
    (OpVectorExtractDynamic %float %edge24_offset_xy %uint_0)
    (OpVectorExtractDynamic %float %edge24_offset_xy %uint_1)
    %float_0_0
    %float_0_0
%offset24_n = OpFNegate %v4float %offset24_p

%v2_pos_m_off_2 = OpFSub %v4float %v2_pos %offset24_n
%v2_pos_p_off_2 = OpFAdd %v4float %v2_pos %offset24_n
%v4_pos_m_off_2 = OpFSub %v4float %v4_pos %offset24_n
%v4_pos_p_off_2 = OpFAdd %v4float %v4_pos %offset24_n

  OpCall %void %emit_vertex %v2_pos_m_off_2 %v3float_1_1_1
  OpCall %void %emit_vertex %v2_pos_p_off_2 %v3float_1_1_1
  OpCall %void %emit_vertex %v4_pos_m_off_2 %v3float_1_1_1
  OpCall %void %emit_vertex %v4_pos_p_off_2 %v3float_1_1_1
  OpEndPrimitive

; Edge 4-0
%edge40_dir = OpFSub %v4float %v0_pos %v4_pos
%neg_edge40_dir_y = OpFNegate %float (OpVectorExtractDynamic %float %edge40_dir %uint_1)
%edge40_perp_xy_corrected = OpCompositeConstruct %v2float %neg_edge40_dir_y (OpVectorExtractDynamic %float %edge40_dir %uint_0)

%edge40_perp_len_sq = OpDot %float %edge40_perp_xy_corrected %edge40_perp_xy_corrected

%edge40_offset_xy = OpFMul %v2float %edge40_perp_norm_v2 %float_0_0075

%offset40_p = OpCompositeConstruct %v4float
    (OpVectorExtractDynamic %float %edge40_offset_xy %uint_0)
    (OpVectorExtractDynamic %float %edge40_offset_xy %uint_1)
    %float_0_0
    %float_0_0
%offset40_n = OpFNegate %v4float %offset40_p

%v4_pos_m_off_3 = OpFSub %v4float %v4_pos %offset40_n
%v4_pos_p_off_3 = OpFAdd %v4float %v4_pos %offset40_n
%v0_pos_m_off_3 = OpFSub %v4float %v0_pos %offset40_n
%v0_pos_p_off_3 = OpFAdd %v4float %v0_pos %offset40_n

  OpCall %void %emit_vertex %v4_pos_m_off_3 %v3float_1_1_1
  OpCall %void %emit_vertex %v4_pos_p_off_3 %v3float_1_1_1
  OpCall %void %emit_vertex %v0_pos_m_off_3 %v3float_1_1_1
  OpCall %void %emit_vertex %v0_pos_p_off_3 %v3float_1_1_1
  OpEndPrimitive

OpReturn
OpFunctionEnd
