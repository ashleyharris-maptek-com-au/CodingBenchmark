#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
layout(location = 2) in vec2 uv;
layout(location = 3) in vec3 tangent;
layout(location = 4) in vec3 color;

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Polka-dot pattern
    vec2 scaledUV = uv * 6.0;
    vec2 cellID = floor(scaledUV);
    vec2 cellUV = fract(scaledUV); // UV within the current cell [0,1]

    // Calculate center of the current cell in cellUV space
    vec2 cellCenter = vec2(0.5, 0.5);

    // Distance from fragment UV to cell center
    float dist = distance(cellUV, cellCenter);

    vec3 patternColor;
    if (dist < 0.3) { // Dot radius 0.3
        patternColor = vec3(1.0, 0.8, 0.0); // Yellow
    } else {
        patternColor = vec3(0.1, 0.1, 0.3); // Dark blue
    }

    // Lambertian shading
    vec3 N = normalize(normal);
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    float diffuse = max(dot(N, L), 0.0);
    float shadingFactor = 0.3 + 0.7 * diffuse;

    // Combine pattern and shading
    outColor = vec4(patternColor * shadingFactor, 1.0);
}
