#version 450

layout(location = 0) in vec3 worldPos;
layout(location = 1) in vec3 normal;
// layout(location = 2) in vec2 uv; // Not used for this task
// layout(location = 3) in vec3 tangent; // Not used for this task
// layout(location = 4) in vec3 color; // Not used for this task

layout(set = 0, binding = 0) uniform UBO {
    mat4 model;
    mat4 view;
    mat4 projection;
    vec4 lightPos;
    vec4 cameraPos;
    vec4 params;
} ubo;

layout(location = 0) out vec4 outColor;

void main() {
    // Define Gooch colors
    vec3 coolColor = vec3(0.2, 0.2, 0.75);
    vec3 warmColor = vec3(0.7, 0.7, 0.4);

    // Normalize normal vector
    vec3 N = normalize(normal);

    // Calculate light direction
    vec3 L = normalize(ubo.lightPos.xyz - worldPos);

    // Calculate t for mixing warm and cool colors
    float t = dot(N, L) * 0.5 + 0.5;

    // Mix warm and cool colors based on t
    vec3 finalColor = mix(coolColor, warmColor, t);

    // Output the final color with alpha 1.0
    outColor = vec4(finalColor, 1.0);
}
