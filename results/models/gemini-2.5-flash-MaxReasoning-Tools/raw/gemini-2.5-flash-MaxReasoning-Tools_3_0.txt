// graph_layout_solver/src/main.rs
use std::io::{self, BufRead};
use rand::{thread_rng, Rng};
mod quadtree;
use crate::quadtree::{Quadtree, Rect};

// Constants for Fruchterman-Reingold algorithm
const K_REPULSIVE: f32 = 1.0; // Repulsive force constant, adjusted for better behavior with Quadtree
const K_ATTRACTIVE: f32 = 0.1; // Attractive force constant
const ITERATIONS: usize = 100; // Number of simulation iterations
const MAX_DISPLACEMENT_MAGNITUDE: f32 = 100.0; // Limit node displacement per iteration
const THETA: f32 = 1.2; // Barnes-Hut opening angle parameter. Lower values mean more accurate but slower.

#[derive(Debug, Clone, Copy, PartialEq)]
pub struct Node {
    pub x: f32,
    pub y: f32,
    pub fx: f32, // Force x
    pub fy: f32, // Force y
}

impl Node {
    fn new(x: f32, y: f32) -> Self {
        Node { x, y, fx: 0.0, fy: 0.0 }
    }
}

pub struct Graph {
    num_nodes: usize,
    num_edges: usize,
    adj: Vec<Vec<usize>>, // Adjacency list
    pub nodes: Vec<Node>,
    area_width: f32,
    area_height: f32,
}

impl Graph {
    fn new(num_nodes: usize, num_edges: usize, area_width: f32, area_height: f32) -> Self {
        let mut graph = Graph {
            num_nodes,
            num_edges,
            adj: vec![Vec::new(); num_nodes],
            nodes: Vec::with_capacity(num_nodes),
            area_width,
            area_height,
        };
        graph.initialize_nodes(); // Initialize nodes upon creation
        graph
    }

    fn add_edge(&mut self, u: usize, v: usize) {
        self.adj[u].push(v);
        self.adj[v].push(u); // Undirected graph
    }

    fn initialize_nodes(&mut self) {
        let mut rng = thread_rng();
        for _ in 0..self.num_nodes {
            self.nodes.push(Node::new(
                rng.gen_range(0.0..self.area_width),
                rng.gen_range(0.0..self.area_height),
            ));
        }
    }

    fn apply_repulsive_forces(&mut self) {
        // Build the quadtree for the current node positions
        let boundary = Rect { x: 0.0, y: 0.0, width: self.area_width, height: self.area_height };
        let quadtree = Quadtree::build(&self.nodes, boundary);

        for i in 0..self.num_nodes {
            quadtree.apply_repulsive_forces_to_node(i, &mut self.nodes, THETA, K_REPULSIVE);
        }
    }

    fn apply_attractive_forces(&mut self) {
        for u in 0..self.num_nodes {
            for &v in &self.adj[u] {
                // Apply attractive force only once per edge
                if u < v { // To avoid processing the same edge twice
                    let dx = self.nodes[u].x - self.nodes[v].x;
                    let dy = self.nodes[u].y - self.nodes[v].y;
                    let dist_sq = dx * dx + dy * dy;
                    let dist = dist_sq.sqrt();

                    if dist < 0.1 { // Prevent division by zero
                        continue;
                    }

                    // F_attractive = dist * dist / k_attractive
                    let force = dist * dist / K_ATTRACTIVE;

                    self.nodes[u].fx -= dx / dist * force;
                    self.nodes[u].fy -= dy / dist * force;
                    self.nodes[v].fx += dx / dist * force;
                    self.nodes[v].fy += dy / dist * force;
                }
            }
        }
    }

    fn update_node_positions(&mut self) {
        for i in 0..self.num_nodes {
            let displacement_x = self.nodes[i].fx;
            let displacement_y = self.nodes[i].fy;

            // Limit displacement to prevent nodes from flying off
            let displacement_magnitude = (displacement_x * displacement_x + displacement_y * displacement_y).sqrt();
            let capped_displacement_x;
            let capped_displacement_y;

            if displacement_magnitude > MAX_DISPLACEMENT_MAGNITUDE {
                capped_displacement_x = displacement_x / displacement_magnitude * MAX_DISPLACEMENT_MAGNITUDE;
                capped_displacement_y = displacement_y / displacement_magnitude * MAX_DISPLACEMENT_MAGNITUDE;
            } else {
                capped_displacement_x = displacement_x;
                capped_displacement_y = displacement_y;
            }

            self.nodes[i].x += capped_displacement_x;
            self.nodes[i].y += capped_displacement_y;

            // Keep nodes within the bounding box
            self.nodes[i].x = self.nodes[i].x.max(0.0).min(self.area_width);
            self.nodes[i].y = self.nodes[i].y.max(0.0).min(self.area_height);

            // Reset forces for the next iteration
            self.nodes[i].fx = 0.0;
            self.nodes[i].fy = 0.0;
        }
    }

    fn layout(&mut self) {
        for _ in 0..ITERATIONS {
            self.apply_repulsive_forces();
            self.apply_attractive_forces();
            self.update_node_positions();
        }
    }
}

fn main() -> io::Result<()> {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read N and M
    let first_line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "Missing first line"))??;
    let parts: Vec<&str> = first_line.split_whitespace().collect();
    if parts.len() != 2 {
        return Err(io::Error::new(io::ErrorKind::InvalidInput, "Expected two numbers on the first line (N M)"));
    }
    let n: usize = parts[0].parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidInput, format!("Invalid N: {}", e)))?;
    let m: usize = parts[1].parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidInput, format!("Invalid M: {}", e)))?;

    // Dynamically adjust area size based on the number of nodes
    let area_width = 1000.0 + (n as f32).sqrt() * 10.0; 
    let area_height = 1000.0 + (n as f32).sqrt() * 10.0;

    let mut graph = Graph::new(n, m, area_width, area_height); 
    
    // Read M edges
    for _ in 0..m {
        let line = lines.next().ok_or_else(|| io::Error::new(io::ErrorKind::InvalidInput, "Missing edge line"))??;
        let parts: Vec<&str> = line.split_whitespace().collect();
        if parts.len() != 2 {
            return Err(io::Error::new(io::ErrorKind::InvalidInput, "Expected two numbers per edge line (u v)"));
        }
        let u: usize = parts[0].parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidInput, format!("Invalid u: {}", e)))?;
        let v: usize = parts[1].parse().map_err(|e| io::Error::new(io::ErrorKind::InvalidInput, format!("Invalid v: {}", e)))?;
        
        if u >= n || v >= n {
            return Err(io::Error::new(io::ErrorKind::InvalidInput, "Node index out of bounds"));
        }
        graph.add_edge(u, v);
    }

    graph.layout(); // Perform the layout

    // Output N and then the coordinates
    println!("{}", graph.num_nodes);
    for node in &graph.nodes {
        println!("{} {}", node.x, node.y);
    }

    Ok(())
}

// graph_layout_solver/src/quadtree.rs
use super::Node;

pub struct Rect {
    pub x: f32,
    pub y: f32,
    pub width: f32,
    pub height: f32,
}

impl Rect {
    pub fn contains(&self, node_x: f32, node_y: f32) -> bool {
        node_x >= self.x && node_x < self.x + self.width &&
        node_y >= self.y && node_y < self.y + self.height
    }
}

pub enum QuadtreeNode {
    Empty,
    Leaf(usize), // Stores the index of the node it contains
    Internal {
        center_of_mass_x: f32,
        center_of_mass_y: f32,
        total_mass: usize, // Represents the number of nodes in this subtree
        north_west: Box<Quadtree>,
        north_east: Box<Quadtree>,
        south_west: Box<Quadtree>,
        south_east: Box<Quadtree>,
    },
}

pub struct Quadtree {
    pub boundary: Rect,
    pub node: QuadtreeNode,
}

impl Quadtree {
    pub fn new(boundary: Rect) -> Self {
        Quadtree {
            boundary,
            node: QuadtreeNode::Empty,
        }
    }

    // Inserts a node into the quadtree
    pub fn insert(&mut self, node_idx: usize, nodes_vec: &Vec<Node>) {
        let node_x = nodes_vec[node_idx].x;
        let node_y = nodes_vec[node_idx].y;

        if !self.boundary.contains(node_x, node_y) {
            return; // Node is outside this quadtree's boundary
        }

        match &mut self.node {
            QuadtreeNode::Empty => {
                self.node = QuadtreeNode::Leaf(node_idx);
            },
            QuadtreeNode::Leaf(existing_node_idx) => {
                let existing_idx = *existing_node_idx;
                
                // If the leaf contains the same node, do nothing (e.g., duplicate insertion)
                if existing_idx == node_idx {
                    return;
                }

                // Subdivide and re-insert the existing node and the new node
                self.subdivide();
                
                // After subdivide, self.node is now Internal
                // Re-insert existing node
                self.insert_into_child(existing_idx, nodes_vec[existing_idx].x, nodes_vec[existing_idx].y, nodes_vec);

                // Insert new node
                self.insert_into_child(node_idx, node_x, node_y, nodes_vec);
            },
            QuadtreeNode::Internal { .. } => {
                self.insert_into_child(node_idx, node_x, node_y, nodes_vec);
            },
        }
    }

    fn insert_into_child(&mut self, node_idx: usize, node_x: f32, node_y: f32, nodes_vec: &Vec<Node>) {
        // This function assumes self.node is QuadtreeNode::Internal
        if let QuadtreeNode::Internal { north_west, north_east, south_west, south_east, .. } = &mut self.node {
            let mid_x = self.boundary.x + self.boundary.width / 2.0;
            let mid_y = self.boundary.y + self.boundary.height / 2.0;

            if node_x < mid_x && node_y < mid_y {
                north_west.insert(node_idx, nodes_vec);
            } else if node_x >= mid_x && node_y < mid_y {
                north_east.insert(node_idx, nodes_vec);
            } else if node_x < mid_x && node_y >= mid_y {
                south_west.insert(node_idx, nodes_vec);
            } else {
                south_east.insert(node_idx, nodes_vec);
            }
        }
    }

    // Helper to subdivide an empty or leaf node into an internal node with empty children
    fn subdivide(&mut self) {
        let x = self.boundary.x;
        let y = self.boundary.y;
        let half_width = self.boundary.width / 2.0;
        let half_height = self.boundary.height / 2.0;

        self.node = QuadtreeNode::Internal {
            center_of_mass_x: 0.0, // Will be calculated later
            center_of_mass_y: 0.0, // Will be calculated later
            total_mass: 0,         // Will be calculated later
            north_west: Box::new(Quadtree::new(Rect { x, y, width: half_width, height: half_height })),
            north_east: Box::new(Quadtree::new(Rect { x: x + half_width, y, width: half_width, height: half_height })),
            south_west: Box::new(Quadtree::new(Rect { x, y: y + half_height, width: half_width, height: half_height })),
            south_east: Box::new(Quadtree::new(Rect { x: x + half_width, y: y + half_height, width: half_width, height: half_height })),
        };
    }

    // Apply repulsive forces to a target node using the quadtree
    pub fn apply_repulsive_forces_to_node(&self, target_node_idx: usize, nodes: &mut Vec<Node>, theta: f32, k_repulsive: f32) {
        match &self.node {
            QuadtreeNode::Empty => {},
            QuadtreeNode::Leaf(idx) => {
                if *idx == target_node_idx { return; } // Don't apply force to self

                let dx = nodes[target_node_idx].x - nodes[*idx].x;
                let dy = nodes[target_node_idx].y - nodes[*idx].y;
                let dist_sq = dx * dx + dy * dy;
                let dist = dist_sq.sqrt();

                if dist < 0.1 { // Prevent division by zero
                    return;
                }

                let force = k_repulsive * k_repulsive / dist;
                nodes[target_node_idx].fx += dx / dist * force;
                nodes[target_node_idx].fy += dy / dist * force;
            },
            QuadtreeNode::Internal {
                center_of_mass_x, center_of_mass_y, total_mass,
                north_west, north_east, south_west, south_east,
            } => {
                let dx = nodes[target_node_idx].x - center_of_mass_x;
                let dy = nodes[target_node_idx].y - center_of_mass_y;
                let dist = (dx * dx + dy * dy).sqrt();

                let s = self.boundary.width.max(self.boundary.height); // Size of the region
                if *total_mass == 0 { return; } // No nodes in this subtree

                // Barnes-Hut criterion: if far enough, treat as single body
                if s / dist < theta && dist > 0.1 { // dist > 0.1 to prevent division by near-zero
                    let force = k_repulsive * k_repulsive * (*total_mass as f32) / dist; // Force proportional to mass
                    nodes[target_node_idx].fx += dx / dist * force;
                    nodes[target_node_idx].fy += dy / dist * force;
                } else { // Recurse
                    north_west.apply_repulsive_forces_to_node(target_node_idx, nodes, theta, k_repulsive);
                    north_east.apply_repulsive_forces_to_node(target_node_idx, nodes, theta, k_repulsive);
                    south_west.apply_repulsive_forces_to_node(target_node_idx, nodes, theta, k_repulsive);
                    south_east.apply_repulsive_forces_to_node(target_node_idx, nodes, theta, k_repulsive);
                }
            },
        }
    }

    // Function to build the quadtree for all nodes in the graph
    pub fn build(nodes_vec: &Vec<Node>, boundary: Rect) -> Self {
        let mut quadtree = Quadtree::new(boundary);
        for i in 0..nodes_vec.len() {
            quadtree.insert(i, nodes_vec);
        }
        // After all nodes are inserted, calculate center of mass and total mass recursively
        quadtree.calculate_center_of_mass_and_total_mass(nodes_vec);
        quadtree
    }

    // Helper to calculate center of mass and total mass for internal nodes recursively
    // Returns (com_x, com_y, total_mass) for the subtree rooted at this node
    fn calculate_center_of_mass_and_total_mass(&mut self, nodes_vec: &Vec<Node>) -> (f32, f32, usize) {
        match &mut self.node {
            QuadtreeNode::Empty => {
                (0.0, 0.0, 0)
            },
            QuadtreeNode::Leaf(idx) => {
                let node = &nodes_vec[*idx];
                (node.x, node.y, 1)
            },
            QuadtreeNode::Internal {
                center_of_mass_x, center_of_mass_y, total_mass,
                north_west, north_east, south_west, south_east,
            } => {
                let (nw_com_x, nw_com_y, nw_mass) = north_west.calculate_center_of_mass_and_total_mass(nodes_vec);
                let (ne_com_x, ne_com_y, ne_mass) = north_east.calculate_center_of_mass_and_total_mass(nodes_vec);
                let (sw_com_x, sw_com_y, sw_mass) = south_west.calculate_center_of_mass_and_total_mass(nodes_vec);
                let (se_com_x, se_com_y, se_mass) = south_east.calculate_center_of_mass_and_total_mass(nodes_vec);

                let mut com_x_sum = 0.0;
                let mut com_y_sum = 0.0;
                let mut mass_sum = 0;

                com_x_sum += nw_com_x * (nw_mass as f32);
                com_y_sum += nw_com_y * (nw_mass as f32);
                mass_sum += nw_mass;

                com_x_sum += ne_com_x * (ne_mass as f32);
                com_y_sum += ne_com_y * (ne_mass as f32);
                mass_sum += ne_mass;

                com_x_sum += sw_com_x * (sw_mass as f32);
                com_y_sum += sw_com_y * (sw_mass as f32);
                mass_sum += sw_mass;

                com_x_sum += se_com_x * (se_mass as f32);
                com_y_sum += se_com_y * (se_mass as f32);
                mass_sum += se_mass;
                
                if mass_sum > 0 {
                    *center_of_mass_x = com_x_sum / (mass_sum as f32);
                    *center_of_mass_y = com_y_sum / (mass_sum as f32);
                    *total_mass = mass_sum;
                } else {
                    *center_of_mass_x = 0.0; // Reset if no mass
                    *center_of_mass_y = 0.0;
                    *total_mass = 0;
                }
                (*center_of_mass_x, *center_of_mass_y, *total_mass)
            },
        }
    }
}
