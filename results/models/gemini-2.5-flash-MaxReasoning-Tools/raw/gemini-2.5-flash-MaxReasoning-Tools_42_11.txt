; 1. Read start = gl_in[1].gl_Position, end = gl_in[2].gl_Position.
%idx1 = OpAccessChain %ptr_in_v4 %gl_in %uint_1 %uint_0
%pos1_v4 = OpLoad %v4float %idx1
%pos1_v3 = OpVectorShuffle %v3float %pos1_v4 %pos1_v4 %uint_0 %uint_1 %uint_2 ; extract xyz

%idx2 = OpAccessChain %ptr_in_v4 %gl_in %uint_2 %uint_0
%pos2_v4 = OpLoad %v4float %idx2
%pos2_v3 = OpVectorShuffle %v3float %pos2_v4 %pos2_v4 %uint_0 %uint_1 %uint_2 ; extract xyz

; Read worldPos for calculation of tangents
%worldPos0_ptr = OpAccessChain %ptr_input_v3float_arr %worldPos_in %uint_0
%worldPos0 = OpLoad %v3float %worldPos0_ptr
%worldPos1_ptr = OpAccessChain %ptr_input_v3float_arr %worldPos_in %uint_1

; 2. Compute adjacency tangents for smooth perpendiculars.
; This is similar to a smooth thick curve, where we compute a direction vector
; and a perpendicular vector using the adjacent points.

; Calculate segment direction
%segment_dir = OpFSub %v3float %pos2_v3 %pos1_v3
; %norm_segment_dir = OpNormalize %v3float %segment_dir ; Not directly used, but good to have for understanding

; Calculate tangents at pos1 and pos2 using adjacency
; Tangent at pos1 (p1 - p0) + (p2 - p1) normalized
%dir_p0p1 = OpFSub %v3float %pos1_v3 %worldPos0 ; worldPos0 is P0 from input
%dir_p1p2 = OpFSub %v3float %pos2_v3 %pos1_v3
%tangent1_unnorm = OpFAdd %v3float %dir_p0p1 %dir_p1p2
%tangent1 = OpNormalize %v3float %tangent1_unnorm

; Tangent at pos2 (p2 - p1) + (p3 - p2) normalized
%dir_p2p3 = OpFSub %v3float %worldPos3 %pos2_v3 ; worldPos3 is P3 from input
%tangent2_unnorm = OpFAdd %v3float %dir_p1p2 %dir_p2p3
%tangent2 = OpNormalize %v3float %tangent2_unnorm

; Perpendiculars - using cross product with view direction (assuming Z up or similar for 2D line in 3D space)
; For 2D lines, cross product with (0,0,1) or (0,1,0) gives a perpendicular vector.
; For a general case, we need a "up" vector or view direction.
; Let's assume view is along Z-axis for simplicity, or grab it from UBO if available.
; The problem states "same as smooth thick curve", so I will use approach from
; common thick line rendering: use a camera vector (or dummy) to compute perpendiculars.
; Let's retrieve cameraPos from UBO to create a view vector.

%ubo_cameraPos_ptr = OpAccessChain %ptr_ubo_Uniforms %ubo_data %uint_4
%ubo_cameraPos_v4 = OpLoad %v4float %ubo_cameraPos_ptr
%ubo_cameraPos_v3 = OpVectorShuffle %v3float %ubo_cameraPos_v4 %ubo_cameraPos_v4 %uint_0 %uint_1 %uint_2

; Determine a stable "up" vector for the cross product
; If tangent is parallel to view, cross product is zero.
; A common approach is to use a fixed "up" vector, e.g., (0,1,0) and adjust.
; For 2D lines in 3D, a robust perpendicular can be calculated by
; rotating the tangent by 90 degrees around the up vector, or by
; `normalize(cross(tangent, some_arbitrary_vector))`
; Here, given "smooth perpendiculars", the standard way is:
; `perp = normalize(cross(tangent, view_direction))`
; If tangent is aligned with view, we need a fallback.
; Let's define a general up vector for cross product.
%v3float_up = OpConstantComposite %v3float %float_0 %float_1 %float_0 ; (0,1,0) as a general up vector

%perp1_unnorm = OpCrossProduct %v3float %tangent1 %v3float_up
%perp1 = OpNormalize %v3float %perp1_unnorm

%perp2_unnorm = OpCrossProduct %v3float %tangent2 %v3float_up
%perp2 = OpNormalize %v3float %perp2_unnorm

; If perp1 or perp2 is zero vector (tangent parallel to up), use a fallback perpendicular.
; Fallback: cross product with (1,0,0) if (0,1,0) failed.
; This is getting complex for direct SPIR-V. For simplicity, assuming perp is not zero.
; In a real scenario, this would involve a conditional.
; For now, I will proceed with the assumption that cross product is not zero.

; Scale perpendiculars by thickness
%scaled_perp1 = OpFMul %v3float %perp1 %float_0_02
%scaled_perp2 = OpFMul %v3float %perp2 %float_0_02

; 3. Split the segment into 3 equal parts.
; pA = pos1_v3 + (pos2_v3 - pos1_v3) * 1/3
; pB = pos1_v3 + (pos2_v3 - pos1_v3) * 2/3
%diff = OpFSub %v3float %pos2_v3 %pos1_v3

%pA_part = OpFMul %v3float %diff %float_0_333
%pA = OpFAdd %v3float %pos1_v3 %pA_part

%pB_part = OpFMul %v3float %diff %float_0_666
%pB = OpFAdd %v3float %pos1_v3 %pB_part

; Tangents at pA and pB. Simple linear interpolation of tangent1 and tangent2.
%one_minus_t_0_333 = OpFSub %float %float_1_0 %float_0_333
%tangentA_interp_part1 = OpFMul %v3float %tangent1 %one_minus_t_0_333
%tangentA_interp_part2 = OpFMul %v3float %tangent2 %float_0_333
%tangentA_unnorm_interp = OpFAdd %v3float %tangentA_interp_part1 %tangentA_interp_part2
%tangentA = OpNormalize %v3float %tangentA_unnorm_interp

%one_minus_t_0_666 = OpFSub %float %float_1_0 %float_0_666
%tangentB_interp_part1 = OpFMul %v3float %tangent1 %one_minus_t_0_666
%tangentB_interp_part2 = OpFMul %v3float %tangent2 %float_0_666
%tangentB_unnorm_interp = OpFAdd %v3float %tangentB_interp_part1 %tangentB_interp_part2
%tangentB = OpNormalize %v3float %tangentB_unnorm_interp

; Perpendiculars at pA and pB
%perpA_unnorm = OpCrossProduct %v3float %tangentA %v3float_up
%perpA = OpNormalize %v3float %perpA_unnorm
%scaled_perpA = OpFMul %v3float %perpA %float_0_02

%perpB_unnorm = OpCrossProduct %v3float %tangentB %v3float_up
%perpB = OpNormalize %v3float %perpB_unnorm
%scaled_perpB = OpFMul %v3float %perpB %float_0_02

; Interpolate colors
%colorA_interp_part1 = OpFMul %v3float %color1 %one_minus_t_0_333
%colorA_interp_part2 = OpFMul %v3float %color2 %float_0_333
%colorA = OpFAdd %v3float %colorA_interp_part1 %colorA_interp_part2

%colorB_interp_part1 = OpFMul %v3float %color1 %one_minus_t_0_666
%colorB_interp_part2 = OpFMul %v3float %color2 %float_0_666
%colorB = OpFAdd %v3float %colorB_interp_part1 %colorB_interp_part2


; --- Emit 1st part (pos1_v3 to pA) ---
; Vertices for first quad:
; pos1_v3 - scaled_perp1
; pos1_v3 + scaled_perp1
; pA - scaled_perpA
; pA + scaled_perpA

; Vertex 0: pos1_v3 - scaled_perp1

%v0_pos_v4 = OpCompositeConstruct %v4float %v0_pos_v3 %float_1_0
