; SPIR-V
; Version: 1.0
; Generator: Google Shaderc over GLSL HPP
; Capabilities: Shader
; Memory Model: Logical GLSL450
OpCapability Shader
OpMemoryModel Logical GLSL450
OpEntryPoint Fragment %main "main" %worldPos %normal %uv %tangent %color %outColor %_ %_param_var_UBO
OpExecutionMode %main OriginUpperLeft

; GLSL.std.450 extended instruction set
%glsl_std_450 = OpExtInstImport "GLSL.std.450"

; All debug names
OpName %main "main"
OpName %worldPos "worldPos"
OpName %normal "normal"
OpName %uv "uv"
OpName %tangent "tangent"
OpName %color "color"
OpName %outColor "outColor"
OpName %UBO "UBO"
OpMemberName %UBO 0 "model"
OpMemberName %UBO 1 "view"
OpMemberName %UBO 2 "projection"
OpMemberName %UBO 3 "lightPos"
OpMemberName %UBO 4 "cameraPos"
OpMemberName %UBO 5 "params"
OpName %_param_var_UBO "UBO" ; Renamed to avoid clash
OpName %_ "UBO" ; Placeholder for the Block type

; All decorations
OpDecorate %worldPos Location 0
OpDecorate %normal Location 1
OpDecorate %uv Location 2
OpDecorate %tangent Location 3
OpDecorate %color Location 4
OpDecorate %outColor Location 0

OpDecorate %_ DescriptorSet 0
OpDecorate %_ Binding 0
OpDecorate %UBO Block
OpMemberDecorate %UBO 0 Offset 0
OpMemberDecorate %UBO 1 Offset 64
OpMemberDecorate %UBO 2 Offset 128
OpMemberDecorate %UBO 3 Offset 192
OpMemberDecorate %UBO 4 Offset 208
OpMemberDecorate %UBO 5 Offset 224

; All types, specifically those with storage class
%void = OpTypeVoid
%bool = OpTypeBool
%int = OpTypeInt 32 1
%uint = OpTypeInt 32 0
%float = OpTypeFloat 32

%v2float = OpTypeVector %float 2
%v3float = OpTypeVector %float 3
%v4float = OpTypeVector %float 4
%m4v4float = OpTypeMatrix %v4float 4

; UBO structure type
%UBO = OpTypeStruct %m4v4float %m4v4float %m4v4float %v4float %v4float %v4float

; Pointers for storage classes
%ptr_Input_v3float = OpTypePointer Input %v3float
%ptr_Input_v2float = OpTypePointer Input %v2float
%ptr_Input_v4float = OpTypePointer Input %v4float
%ptr_Output_v4float = OpTypePointer Output %v4float
%ptr_Uniform_UBO = OpTypePointer Uniform %UBO
%ptr_Uniform_v4float = OpTypePointer Uniform %v4float

; All constants
%float_0 = OpConstant %float 0
%float_1 = OpConstant %float 1
%float_0_1 = OpConstant %float 0.1
%float_0_05 = OpConstant %float 0.05
%float_32 = OpConstant %float 32

%v3float_ambient = OpConstantComposite %v3float %float_0_1 %float_0_05 %float_0_05 ; (0.1, 0.05, 0.05)
%v3float_specular = OpConstantComposite %v3float %float_1 %float_1 %float_1 ; (1,1,1)
%v3float_zero = OpConstantComposite %v3float %float_0 %float_0 %float_0

%v4float_alpha_1 = OpConstantComposite %v4float %float_1 %float_1 %float_1 %float_1

; Global variables
%worldPos = OpVariable %ptr_Input_v3float Input
%normal = OpVariable %ptr_Input_v3float Input
%uv = OpVariable %ptr_Input_v2float Input
%tangent = OpVariable %ptr_Input_v3float Input
%color = OpVariable %ptr_Input_v3float Input
%outColor = OpVariable %ptr_Output_v4float Output
%_param_var_UBO = OpVariable %ptr_Uniform_UBO Uniform

%main = OpFunction %void None %void_func
%void_func = OpTypeFunction %void
%entry = OpLabel
; Load inputs
%_worldPos = OpLoad %v3float %worldPos
%_normal = OpLoad %v3float %normal
%_color = OpLoad %v3float %color

; Load UBO members
%lightPos_ptr = OpAccessChain %ptr_Uniform_v4float %_param_var_UBO %uint_3
%_lightPos = OpLoad %v4float %lightPos_ptr
%cameraPos_ptr = OpAccessChain %ptr_Uniform_v4float %_param_var_UBO %uint_4
%_cameraPos = OpLoad %v4float %cameraPos_ptr

; Convert lightPos and cameraPos to vec3
%_lightPos_vec3 = OpVectorExtractDynamic %v3float %_lightPos %uint_0
%_cameraPos_vec3 = OpVectorExtractDynamic %v3float %_cameraPos %uint_0

; Normalize normal vector
%norm_normal = OpExtInst %v3float %glsl_std_450 Normalize %_normal

; Calculate Light direction (L)
%light_dir_unnorm = OpFSub %v3float %_lightPos_vec3 %_worldPos
%light_dir = OpExtInst %v3float %glsl_std_450 Normalize %light_dir_unnorm

; Calculate View direction (V)
%view_dir_unnorm = OpFSub %v3float %_cameraPos_vec3 %_worldPos
%view_dir = OpExtInst %v3float %glsl_std_450 Normalize %view_dir_unnorm

; Ambient component
%ambient_comp = OpCopyObject %v3float %v3float_ambient

; Diffuse component: vertex color * max(dot(N,L),0)
%N_dot_L = OpDot %float %norm_normal %light_dir
%max_N_dot_L = OpExtInst %float %glsl_std_450 Max %N_dot_L %float_0
%diffuse_comp = OpFMul %v3float %_color %max_N_dot_L

; Specular component: (1,1,1) * shininess 32 * R=reflect(-L,N)
%neg_light_dir = OpFNegate %v3float %light_dir
%reflect_vec = OpExtInst %v3float %glsl_std_450 Reflect %neg_light_dir %norm_normal
%R_dot_V = OpDot %float %reflect_vec %view_dir
%max_R_dot_V = OpExtInst %float %glsl_std_450 Max %R_dot_V %float_0
%pow_spec = OpExtInst %float %glsl_std_450 Pow %max_R_dot_V %float_32
%specular_comp = OpFMul %v3float %v3float_specular %pow_spec

; Sum components
%sum_rgb_1 = OpFAdd %v3float %ambient_comp %diffuse_comp
%final_rgb = OpFAdd %v3float %sum_rgb_1 %specular_comp

; Clamp to [0,1]
%clamped_rgb_min = OpExtInst %v3float %glsl_std_450 Max %final_rgb %v3float_zero
%clamped_rgb = OpExtInst %v3float %glsl_std_450 Min %clamped_rgb_min %v3float_specular ; Using v3float_specular as (1,1,1)

; Output final color (RGBA float, alpha 1.0)
%final_outColor = OpVectorShuffle %v4float %clamped_rgb %v4float_alpha_1 0 1 2 7
OpStore %outColor %final_outColor

OpReturn
OpFunctionEnd
