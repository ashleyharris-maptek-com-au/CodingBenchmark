#version 450
#extension GL_KHR_shader_subgroup_ballot: enable

layout(local_size_x = 16, local_size_y = 16, local_size_z = 1) in;

layout(std430, binding = 0) readonly buffer InputBuffer {
    float f_in[];
};

layout(std430, binding = 1) buffer OutputBuffer {
    float f_out[];
};

layout(std140, binding = 2) uniform UniformBuffer {
    uvec4 grid_dims; // W, H
    vec4 params;     // omega
};

void main() {
    uint x = gl_GlobalInvocationID.x;
    uint y = gl_GlobalInvocationID.y;

    uint W = grid_dims.x;
    uint H = grid_dims.y;
    float omega = params.x;

    // Guard against out-of-bounds
    if (x >= W || y >= H) {
        return;
    }

    // D2Q9 constants
    // (cx, cy)
    const int cx[] = { 0,  1,  0, -1,  0,  1, -1, -1,  1 };
    const int cy[] = { 0,  0,  1,  0, -1,  1,  1, -1, -1 };
    // weights
    const float w[] = { 4.0/9.0, 1.0/9.0, 1.0/9.0, 1.0/9.0, 1.0/9.0, 1.0/36.0, 1.0/36.0, 1.0/36.0, 1.0/36.0 };

    float f[9];
    uint base_idx = (y * W + x) * 9;
    for (int k = 0; k < 9; ++k) {
        f[k] = f_in[base_idx + k];
    }

    // Compute macroscopic density rho and velocity (ux, uy)
    float rho = 0.0;
    float ux = 0.0;
    float uy = 0.0;

    for (int k = 0; k < 9; ++k) {
        rho += f[k];
        ux += (float)cx[k] * f[k];
        uy += (float)cy[k] * f[k];
    }

    ux /= rho;
    uy /= rho;

    // BGK collision update
    float f_collide[9];
    for (int k = 0; k < 9; ++k) {
        float cu = (float)cx[k] * ux + (float)cy[k] * uy;
        float feq = w[k] * rho * (1.0 + 3.0 * cu + 4.5 * cu * cu - 1.5 * (ux * ux + uy * uy));
        f_collide[k] = f[k] + omega * (feq - f[k]);
    }

    // Streaming
    for (int k = 0; k < 9; ++k) {
        // Stream to neighbor cell (x+cx[k], y+cy[k]) with periodic wraparound
        uint next_x = (x + cx[k] + W) % W;
        uint next_y = (y + cy[k] + H) % H;

        uint output_idx = (next_y * W + next_x) * 9 + k;
        f_out[output_idx] = f_collide[k];
    }
}
