use std::io::{self, BufRead};
use std::collections::HashMap;

fn generate_subset_sums(numbers: &[i64], offset: usize) -> HashMap<i64, Vec<usize>> {
    let mut sums_map: HashMap<i64, Vec<usize>> = HashMap::new();
    sums_map.insert(0, Vec::new()); // Empty subset has sum 0 and no elements

    for (i, &num) in numbers.iter().enumerate() {
        let current_index = offset + i;
        let mut entries_to_add: Vec<(i64, Vec<usize>)> = Vec::new();

        for (&sum, indices) in &sums_map {
            let new_sum = sum + num;
            let mut new_indices = indices.clone();
            new_indices.push(current_index);
            entries_to_add.push((new_sum, new_indices));
        }

        for (new_sum, new_indices) in entries_to_add {
            // Only insert if this sum hasn't been seen, or if we want to store an alternative.
            // For this problem, any valid subset is fine, so we just take the first one.
            sums_map.entry(new_sum).or_insert(new_indices);
        }
    }
    sums_map
}

fn main() {
    let stdin = io::stdin();
    let mut lines = stdin.lock().lines();

    // Read num_numbers and target
    let first_line = lines.next().expect("Expected first line").expect("Failed to read first line");
    let parts: Vec<i64> = first_line
        .split_whitespace()
        .map(|s| s.parse().expect("Failed to parse integer"))
        .collect();
    let num_numbers = parts[0] as usize;
    let target = parts[1];

    // Read the numbers
    let numbers: Vec<i64> = if num_numbers > 0 {
        let numbers_line = lines.next().expect("Expected numbers line").expect("Failed to read numbers line");
        numbers_line
            .split_whitespace()
            .map(|s| s.parse().expect("Failed to parse integer"))
            .collect()
    } else {
        Vec::new()
    };

    // The provided num_numbers might not match the actual count if input is malformed,
    // but the problem implies we should use the provided numbers, so we'll use `numbers.len()`
    // for array operations.
    if numbers.len() != num_numbers {
        eprintln!("Warning: num_numbers in input ({}) does not match actual number of elements ({}). Using actual number of elements.", num_numbers, numbers.len());
        // For robustness, let's use the actual length of numbers.
        // num_numbers = numbers.len(); // This would require num_numbers to be mutable.
        // For now, we'll proceed using numbers.len() in array slicing.
    }


    // Handle target == 0 always first, as an empty subset always sums to 0.
    if target == 0 {
        println!("YES\n[]");
        return;
    }

    if numbers.is_empty() {
        // We already handled target == 0, so if numbers is empty and target is not 0, it's NO.
        println!("NO");
        return;
    }

    // Determine algorithm based on input size.
    // Meet-in-the-middle is good for N up to around 40-45.
    // For N > 40-45, 2^(N/2) becomes too large.
    // For now, if N is very large, we'll simply state NO, as per instructions for larger cases
    // we would need approximation algorithms, threading etc. which will be
    // added if the current approach times out.
    if numbers.len() > 45 { // A heuristic for when meet-in-the-middle becomes too slow
        eprintln!("Info: Number of elements ({}) is too large for current exact algorithm. Returning NO.", numbers.len());
        println!("NO");
        return;
    }

    // Split the numbers into two halves
    let mid = numbers.len() / 2;
    let first_half = &numbers[0..mid];
    let second_half = &numbers[mid..numbers.len()];

    let sums1 = generate_subset_sums(first_half, 0);
    let sums2 = generate_subset_sums(second_half, mid);

    // Check for target sum
    // Case 1: Target is directly found in the first half's sums
    if let Some(indices) = sums1.get(&target) {
        println!("YES");
        println!("{}", indices.iter().map(|i| i.to_string()).collect::<Vec<String>>().join(" "));
        return;
    }

    // Case 2: Target is directly found in the second half's sums
    if let Some(indices) = sums2.get(&target) {
        println!("YES");
        println!("{}", indices.iter().map(|i| i.to_string()).collect::<Vec<String>>().join(" "));
        return;
    }

    // Case 3: A sum from the first half and a complement from the second half sum to target
    for (&sum1, indices1) in &sums1 {
        let complement = target - sum1;
        if let Some(indices2) = sums2.get(&complement) {
            let mut result_indices = indices1.clone();
            result_indices.extend(indices2.iter());
            result_indices.sort_unstable(); // Sort to ensure consistent output

            println!("YES");
            println!("{}", result_indices.iter().map(|i| i.to_string()).collect::<Vec<String>>().join(" "));
            return; // Found a solution, exit
        }
    }

    println!("NO");
}
