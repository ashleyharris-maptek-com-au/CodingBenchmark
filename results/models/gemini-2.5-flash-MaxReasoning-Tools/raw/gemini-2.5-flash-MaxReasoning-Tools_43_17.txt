struct PSInput {
    [[vk::location(0)]] float3 worldPos : TEXCOORD0;   // world-space position
    [[vk::location(1)]] float3 normal   : TEXCOORD1;   // world-space normal (interpolated, may need normalize)
    [[vk::location(2)]] float2 uv       : TEXCOORD2;   // texture coordinates in [0,1]
    [[vk::location(3)]] float3 tangent  : TEXCOORD3;   // tangent vector
    [[vk::location(4)]] float3 color    : TEXCOORD4;   // per-vertex color (smoothly varies: rgb = 0.5+0.5*xyz)
};

[[vk::binding(0, 0)]]
cbuffer UBO : register(b0) {
    float4x4 model;          // offset 0   (column-major)
    float4x4 view;           // offset 64
    float4x4 projection;     // offset 128
    float4 lightPos;         // value: (5.0, 5.0, 5.0, 1.0)
    float4 cameraPos;        // value: (0.0, 0.0, 3.0, 1.0)
    float4 params;           // value: (0.0, 0.5, 0.0, 0.0)
};

float4 main(PSInput input) : SV_Target0 {
    // Normalize interpolated normal
    float3 N = normalize(input.normal);

    // Light direction
    float3 L = normalize(lightPos.xyz - input.worldPos);

    // Lambertian diffuse component
    float diffuse = max(0.0f, dot(N, L));
    float shadingFactor = 0.4f + 0.6f * diffuse;

    // Brick pattern parameters
    float2  scaledUV = input.uv * float2(8.0f, 16.0f);
    float   mortarThicknessX = 0.05f;
    float   mortarThicknessY = 0.1f;
    float3  mortarColor = float3(0.7f, 0.7f, 0.7f);
    float3  brickColor  = float3(0.7f, 0.2f, 0.1f);

    // Apply staggering
    if (fmod(floor(scaledUV.y), 2.0f) < 0.5f) { // Every other row (even rows)
        scaledUV.x += 0.5f;
    }

    // Get fractional part of UVs for brick/mortar decision
    float2  fUV = frac(scaledUV);

    // Determine if it's mortar or brick
    bool isMortarX = (fUV.x < mortarThicknessX) || (fUV.x > (1.0f - mortarThicknessX));
    bool isMortarY = (fUV.y < mortarThicknessY) || (fUV.y > (1.0f - mortarThicknessY));
    bool isMortar = isMortarX || isMortarY;

    float3 finalColor = isMortar ? mortarColor : brickColor;

    // Apply shading
    finalColor *= shadingFactor;

    return float4(finalColor, 1.0f);
}
