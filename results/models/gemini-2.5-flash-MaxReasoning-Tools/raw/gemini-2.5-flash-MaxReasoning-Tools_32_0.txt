#include <iostream>
#include <vector>
#include <algorithm>
#include <queue>
#include <map>
#include <set>
#include <tuple> // For std::tie

const long long INF = 1e18; // Use a large enough value for infinity

// Function to reconstruct path from u to v using next_node matrix
void get_path_edges(int u, int v, const std::vector<std::vector<int>>& next_node, std::set<std::pair<int, int>>& result_edges) {
    if (u == v) return;
    while (u != v) {
        int next = next_node[u][v];
        if (next == -1) { // No path or issue with reconstruction
            // This should ideally not happen if paths exist
            break;
        }
        result_edges.insert({std::min(u, next), std::max(u, next)});
        u = next;
    }
}

int main() {
    std::ios_base::sync_with_stdio(false);
    std::cin.tie(NULL);

    int num_vertices, num_edges, num_terminals;
    std::cin >> num_vertices >> num_edges >> num_terminals;

    std::vector<std::vector<long long>> adj_matrix(num_vertices, std::vector<long long>(num_vertices, INF));
    std::vector<std::vector<int>> next_node(num_vertices, std::vector<int>(num_vertices, -1)); // For path reconstruction in Floyd-Warshall

    for (int i = 0; i < num_vertices; ++i) {
        adj_matrix[i][i] = 0;
    }

    for (int i = 0; i < num_edges; ++i) {
        int u, v;
        long long weight;
        std::cin >> u >> v >> weight;
        // For undirected graph, add edges in both directions
        if (weight < adj_matrix[u][v]) { // Only update if new weight is smaller (handling parallel edges)
            adj_matrix[u][v] = weight;
            adj_matrix[v][u] = weight;
            next_node[u][v] = v; // Direct edge, next node from u to v is v
            next_node[v][u] = u; // Direct edge, next node from v to u is u
        }
    }

    std::vector<int> terminals(num_terminals);
    std::map<int, int> terminal_to_idx; // Map terminal vertex to its index in 'terminals' array
    for (int i = 0; i < num_terminals; ++i) {
        std::cin >> terminals[i];
        terminal_to_idx[terminals[i]] = i;
    }

    // Floyd-Warshall to find all-pairs shortest paths
    for (int k = 0; k < num_vertices; ++k) {
        for (int i = 0; i < num_vertices; ++i) {
            for (int j = 0; j < num_vertices; ++j) {
                if (adj_matrix[i][k] != INF && adj_matrix[k][j] != INF) {
                    if (adj_matrix[i][j] > adj_matrix[i][k] + adj_matrix[k][j]) {
                        adj_matrix[i][j] = adj_matrix[i][k] + adj_matrix[k][j];
                        next_node[i][j] = next_node[i][k]; // Shortest path from i to j goes through k, so next node from i is same as next node from i to k
                    }
                }
            }
        }
    }

    // DP state: dp[mask][v] = minimum cost to connect terminals in 'mask' rooted at 'v'
    std::vector<std::vector<long long>> dp(1 << num_terminals, std::vector<long long>(num_vertices, INF));
    // For reconstruction:
    // type: 0 for base (terminal itself), 1 for split (from combining submasks), 2 for path (from extending from another vertex u)
    // value: if split, stores submask1; if path, stores parent vertex u
    std::vector<std::vector<std::pair<int, int>>> dp_reconstruct(1 << num_terminals, std::vector<std::pair<int, int>>(num_vertices, {-1, -1}));

    // Base cases: single terminal masks
    for (int i = 0; i < num_terminals; ++i) {
        dp[1 << i][terminals[i]] = 0;
        dp_reconstruct[1 << i][terminals[i]] = {0, -1}; // Type 0: base case, no parent info needed
    }

    // DP transitions
    for (int mask = 1; mask < (1 << num_terminals); ++mask) {
        // Step 1: For each vertex v, calculate dp[mask][v] by combining two smaller Steiner trees
        // that both contain v and cover complementary submasks of terminals.
        for (int v = 0; v < num_vertices; ++v) {
            // Iterate over all proper submasks of 'mask'
            for (int submask1 = (mask - 1) & mask; submask1 > 0; submask1 = (submask1 - 1) & mask) {
                int submask2 = mask ^ submask1; // submask2 is the complement of submask1 within mask
                if (dp[submask1][v] != INF && dp[submask2][v] != INF) {
                    if (dp[mask][v] > dp[submask1][v] + dp[submask2][v]) {
                        dp[mask][v] = dp[submask1][v] + dp[submask2][v];
                        dp_reconstruct[mask][v] = {1, submask1}; // Type 1: split, value is submask1
                    }
                }
            }
        }

        // Step 2: For each mask, relax dp[mask][v] values by considering shortest paths
        // from other vertices u. This means a Steiner tree for 'mask' rooted at 'v'
        // could be formed by taking a Steiner tree for 'mask' rooted at 'u' and
        // extending it with the shortest path from 'u' to 'v'.
        for (int u = 0; u < num_vertices; ++u) { // u is the current root for mask
            if (dp[mask][u] == INF) continue; 
            for (int v = 0; v < num_vertices; ++v) { // v is the potential new root for mask
                if (u == v) continue;
                // adj_matrix[u][v] contains the shortest path between u and v
                if (adj_matrix[u][v] != INF) {
                    if (dp[mask][v] > dp[mask][u] + adj_matrix[u][v]) {
                        dp[mask][v] = dp[mask][u] + adj_matrix[u][v];
                        dp_reconstruct[mask][v] = {2, u}; // Type 2: path, value is parent vertex u
                    }
                }
            }
        }
    }

    // Find the minimum total weight for connecting all terminals and the final root vertex
    long long min_total_weight = INF;
    int final_root = -1;
    int full_mask = (1 << num_terminals) - 1; // Mask representing all terminals

    for (int v = 0; v < num_vertices; ++v) {
        if (min_total_weight > dp[full_mask][v]) {
            min_total_weight = dp[full_mask][v];
            final_root = v;
        }
    }

    std::cout << min_total_weight << " ";

    std::set<std::pair<int, int>> result_edges; // Store unique edges for the Steiner tree
    std::vector<std::pair<int, int>> q_reconstruct; // Queue for BFS-like reconstruction: {mask, vertex}
    if (final_root != -1) {
        q_reconstruct.push_back({full_mask, final_root});
    }

    std::set<std::pair<int, int>> visited_reconstruct; // To prevent redundant processing and infinite loops

    int head = 0;
    while(head < q_reconstruct.size()){
        int current_mask = q_reconstruct[head].first;
        int current_v = q_reconstruct[head].second;
        head++;

        if (current_mask == 0) continue; // Base case for mask (no terminals to connect)
        if (visited_reconstruct.count({current_mask, current_v})) continue; // Already processed this state
        visited_reconstruct.insert({current_mask, current_v});
        
        auto [type, val] = dp_reconstruct[current_mask][current_v];

        if (type == 0) { // Base case: current_v is a terminal in this mask, no edges to add from DP
            // The cost is 0, implying current_v itself fulfills the mask.
        } else if (type == 1) { // Split: dp[current_mask][current_v] came from combining two subtrees
            int submask1 = val;
            int submask2 = current_mask ^ submask1;
            q_reconstruct.push_back({submask1, current_v}); // Recurse for submask1 rooted at current_v
            q_reconstruct.push_back({submask2, current_v}); // Recurse for submask2 rooted at current_v
        } else if (type == 2) { // Path: dp[current_mask][current_v] came from extending from parent_u
            int parent_u = val;
            // Add all edges forming the shortest path from parent_u to current_v
            get_path_edges(parent_u, current_v, next_node, result_edges);
            // Continue reconstruction from parent_u for the same mask
            q_reconstruct.push_back({current_mask, parent_u});
        }
    }

    std::cout << result_edges.size() << std::endl;
    for (const auto& edge : result_edges) {
        std::cout << edge.first << " " << edge.second << std::endl;
    }

    return 0;
}
