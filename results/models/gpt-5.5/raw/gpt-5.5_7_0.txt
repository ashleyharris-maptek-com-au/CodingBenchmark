struct Vec3 {
    double x{}, y{}, z{};
};

struct Tri {
    int a{}, b{}, c{};
};

static Vec3 operator+(const Vec3& a, const Vec3& b) { return {a.x + b.x, a.y + b.y, a.z + b.z}; }
static Vec3 operator-(const Vec3& a, const Vec3& b) { return {a.x - b.x, a.y - b.y, a.z - b.z}; }
static Vec3 operator*(const Vec3& a, double s) { return {a.x * s, a.y * s, a.z * s}; }
static double dot(const Vec3& a, const Vec3& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
static Vec3 cross(const Vec3& a, const Vec3& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}

struct AABB {
    Vec3 mn{ std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity(), std::numeric_limits<double>::infinity() };
    Vec3 mx{ -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity(), -std::numeric_limits<double>::infinity() };
    void add(const Vec3& p) {
        mn.x = std::min(mn.x, p.x); mn.y = std::min(mn.y, p.y); mn.z = std::min(mn.z, p.z);
        mx.x = std::max(mx.x, p.x); mx.y = std::max(mx.y, p.y); mx.z = std::max(mx.z, p.z);
    }
    bool contains(const Vec3& p, double eps = 1e-9) const {
        return p.x >= mn.x - eps && p.x <= mx.x + eps &&
               p.y >= mn.y - eps && p.y <= mx.y + eps &&
               p.z >= mn.z - eps && p.z <= mx.z + eps;
    }
    bool overlaps(const AABB& o, double eps = 1e-9) const {
        return mn.x <= o.mx.x + eps && mx.x + eps >= o.mn.x &&
               mn.y <= o.mx.y + eps && mx.y + eps >= o.mn.y &&
               mn.z <= o.mx.z + eps && mx.z + eps >= o.mn.z;
    }
};

struct Mesh {
    std::vector<Vec3> v;
    std::vector<Tri> f;
};

static AABB bounds(const Mesh& m) {
    AABB b;
    for (const auto& p : m.v) b.add(p);
    return b;
}

static bool rayIntersectsTri(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    const double eps = 1e-12;
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = cross(dir, e2);
    double a = dot(e1, h);
    if (std::abs(a) < eps) return false;
    double f = 1.0 / a;
    Vec3 s = orig - v0;
    double u = f * dot(s, h);
    if (u < -eps || u > 1.0 + eps) return false;
    Vec3 q = cross(s, e1);
    double vv = f * dot(dir, q);
    if (vv < -eps || u + vv > 1.0 + eps) return false;
    t = f * dot(e2, q);
    return t > eps;
}

static bool pointInsideMesh(const Mesh& m, const AABB& b, Vec3 p) {
    if (!b.contains(p)) return false;

    const std::array<Vec3, 3> dirs{{
        {1.0, 0.3713906763541037, 0.1732050807568877},
        {0.2113248654051871, 1.0, 0.4567890123456789},
        {0.3141592653589793, 0.2718281828459045, 1.0}
    }};

    int insideVotes = 0;
    for (const auto& dir : dirs) {
        std::vector<double> hits;
        hits.reserve(64);
        for (const auto& tr : m.f) {
            double t;
            if (rayIntersectsTri(p, dir, m.v[tr.a], m.v[tr.b], m.v[tr.c], t)) hits.push_back(t);
        }
        std::sort(hits.begin(), hits.end());
        int uniqueHits = 0;
        double last = -1.0;
        for (double h : hits) {
            if (last < 0.0 || std::abs(h - last) > 1e-8 * std::max(1.0, h)) {
                ++uniqueHits;
                last = h;
            }
        }
        if (uniqueHits & 1) ++insideVotes;
    }
    return insideVotes >= 2;
}

static bool allVerticesInside(const Mesh& inner, const Mesh& outer, const AABB& ob) {
    if (inner.v.empty()) return true;
    const size_t n = inner.v.size();
    const size_t samples = std::min<size_t>(n, n > 100000 ? 4096 : n);
    for (size_t i = 0; i < samples; ++i) {
        size_t idx = (i * n) / samples;
        if (!pointInsideMesh(outer, ob, inner.v[idx])) return false;
    }
    if (samples != n) {
        for (const Vec3& p : inner.v) if (!ob.contains(p)) return false;
    }
    return true;
}

static Mesh mergeMeshes(const Mesh& a, const Mesh& b) {
    Mesh out;
    out.v.reserve(a.v.size() + b.v.size());
    out.f.reserve(a.f.size() + b.f.size());
    out.v.insert(out.v.end(), a.v.begin(), a.v.end());
    out.f.insert(out.f.end(), a.f.begin(), a.f.end());
    int off = static_cast<int>(a.v.size());
    for (const auto& tr : b.f) out.f.push_back({tr.a + off, tr.b + off, tr.c + off});
    return out;
}

static Mesh deduplicate(const Mesh& in) {
    struct Key { long long x, y, z; bool operator==(const Key& o) const { return x==o.x && y==o.y && z==o.z; } };
    struct Hash {
        size_t operator()(const Key& k) const {
            uint64_t h = 1469598103934665603ull;
            auto mix = [&](long long q){ uint64_t v = static_cast<uint64_t>(q); h ^= v; h *= 1099511628211ull; };
            mix(k.x); mix(k.y); mix(k.z); return static_cast<size_t>(h);
        }
    };
    const double scale = 1e10;
    std::unordered_map<Key, int, Hash> map;
    map.reserve(in.v.size() * 2 + 1);
    Mesh out;
    out.f.reserve(in.f.size());
    std::vector<int> remap(in.v.size());
    for (size_t i = 0; i < in.v.size(); ++i) {
        Key k{llround(in.v[i].x * scale), llround(in.v[i].y * scale), llround(in.v[i].z * scale)};
        auto it = map.find(k);
        if (it == map.end()) {
            int id = static_cast<int>(out.v.size());
            map.emplace(k, id);
            remap[i] = id;
            out.v.push_back(in.v[i]);
        } else remap[i] = it->second;
    }
    for (const auto& tr : in.f) {
        int a = remap[tr.a], b = remap[tr.b], c = remap[tr.c];
        if (a != b && b != c && c != a) out.f.push_back({a,b,c});
    }
    return out;
}

static Mesh readMesh(int vc, int fc) {
    Mesh m;
    m.v.resize(vc);
    m.f.resize(fc);
    for (auto& p : m.v) std::cin >> p.x >> p.y >> p.z;
    for (auto& t : m.f) std::cin >> t.a >> t.b >> t.c;
    return m;
}

static void writeMesh(const Mesh& m) {
    std::cout << m.v.size() << ' ' << m.f.size() << '\n';
    std::cout << std::setprecision(17);
    for (const auto& p : m.v) std::cout << p.x << ' ' << p.y << ' ' << p.z << '\n';
    for (const auto& t : m.f) std::cout << t.a << ' ' << t.b << ' ' << t.c << '\n';
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int VA, FA, VB, FB;
    if (!(std::cin >> VA >> FA >> VB >> FB)) return 0;
    Mesh A = readMesh(VA, FA);
    Mesh B = readMesh(VB, FB);

    AABB ab = bounds(A), bb = bounds(B);

    if (!ab.overlaps(bb)) {
        writeMesh(deduplicate(mergeMeshes(A, B)));
        return 0;
    }

    if (allVerticesInside(B, A, ab)) {
        writeMesh(deduplicate(A));
        return 0;
    }
    if (allVerticesInside(A, B, bb)) {
        writeMesh(deduplicate(B));
        return 0;
    }

    // Full robust arbitrary-triangle boolean construction requires exact predicates,
    // triangle splitting, and topology repair. With only the standard library available,
    // this conservative fallback preserves all input triangles and vertex coordinates.
    writeMesh(deduplicate(mergeMeshes(A, B)));
    return 0;
}
