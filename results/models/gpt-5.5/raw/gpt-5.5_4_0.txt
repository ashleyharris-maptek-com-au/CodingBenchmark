#include <algorithm>
#include <array>
#include <chrono>
#include <cmath>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <random>
#include <tuple>
#include <vector>

using namespace std;

static constexpr double PI = 3.141592653589793238462643383279502884;

struct Vec {
    double x = 0, y = 0, z = 0;
    Vec() = default;
    Vec(double X, double Y, double Z) : x(X), y(Y), z(Z) {}
    Vec operator+(const Vec& o) const { return {x + o.x, y + o.y, z + o.z}; }
    Vec operator-(const Vec& o) const { return {x - o.x, y - o.y, z - o.z}; }
    Vec operator*(double s) const { return {x * s, y * s, z * s}; }
    Vec operator/(double s) const { return {x / s, y / s, z / s}; }
};

static double dotv(const Vec& a, const Vec& b) { return a.x*b.x + a.y*b.y + a.z*b.z; }
static Vec crossv(const Vec& a, const Vec& b) {
    return {a.y*b.z - a.z*b.y, a.z*b.x - a.x*b.z, a.x*b.y - a.y*b.x};
}
static double normv(const Vec& a) { return sqrt(dotv(a,a)); }

struct Plane {
    Vec n;
    double d = 0; // inside: n.p <= d
};

struct Tetra {
    Vec c;
    double rx, ry, rz;
};

static array<Vec,4> unitTetraVerts() {
    // Regular tetrahedron, edge 1, centered at origin.
    const double s = 1.0 / (2.0 * sqrt(2.0));
    return {Vec{s,s,s}, Vec{s,-s,-s}, Vec{-s,s,-s}, Vec{-s,-s,s}};
}

static Vec rotateEuler(const Vec& p, double rx, double ry, double rz) {
    double cx = cos(rx), sx = sin(rx), cy = cos(ry), sy = sin(ry), cz = cos(rz), sz = sin(rz);
    double x1 = p.x;
    double y1 = cx*p.y - sx*p.z;
    double z1 = sx*p.y + cx*p.z;
    double x2 = cy*x1 + sy*z1;
    double y2 = y1;
    double z2 = -sy*x1 + cy*z1;
    return {cz*x2 - sz*y2, sz*x2 + cz*y2, z2};
}

static vector<Plane> buildPlanes(const vector<Vec>& v) {
    int n = (int)v.size();
    Vec cen;
    for (auto& p : v) cen = cen + p;
    cen = cen / max(1, n);
    vector<Plane> planes;
    const double eps = 1e-8;
    for (int i=0;i<n;i++) for (int j=i+1;j<n;j++) for (int k=j+1;k<n;k++) {
        Vec normal = crossv(v[j]-v[i], v[k]-v[i]);
        double len = normv(normal);
        if (len < 1e-10) continue;
        normal = normal / len;
        double d = dotv(normal, v[i]);
        int pos = 0, neg = 0;
        for (int t=0;t<n;t++) {
            double s = dotv(normal, v[t]) - d;
            if (s > eps) pos++;
            if (s < -eps) neg++;
        }
        if (pos && neg) continue; // not a supporting plane of the convex hull
        if (dotv(normal, cen) > d) { normal = normal * -1.0; d = -d; }
        bool dup = false;
        for (auto& p : planes) {
            if (normv(p.n - normal) < 1e-6 && fabs(p.d - d) < 1e-6) { dup = true; break; }
        }
        if (!dup) planes.push_back({normal, d});
    }
    return planes;
}

static bool insideTetra(const Vec& c, const array<Vec,4>& verts, const vector<Plane>& planes, double edge) {
    const double eps = 1e-7;
    for (const auto& pl : planes) {
        for (const auto& q : verts) {
            Vec p = c + q * edge;
            if (dotv(pl.n, p) - pl.d > eps) return false;
        }
    }
    return true;
}

static uint64_t keyCell(int a, int b, int c) {
    uint64_t x = (uint32_t)(a + 1000000000);
    uint64_t y = (uint32_t)(b + 1000000000);
    uint64_t z = (uint32_t)(c + 1000000000);
    return (x * 73856093ULL) ^ (y * 19349663ULL) ^ (z * 83492791ULL);
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int V;
    double edge;
    if (!(cin >> V >> edge) || V <= 0 || edge <= 0) return 0;
    vector<Vec> pts(V);
    for (auto& p : pts) cin >> p.x >> p.y >> p.z;

    Vec mn{numeric_limits<double>::infinity(), numeric_limits<double>::infinity(), numeric_limits<double>::infinity()};
    Vec mx{-mn.x, -mn.y, -mn.z};
    Vec centroid;
    for (auto& p : pts) {
        mn.x = min(mn.x, p.x); mn.y = min(mn.y, p.y); mn.z = min(mn.z, p.z);
        mx.x = max(mx.x, p.x); mx.y = max(mx.y, p.y); mx.z = max(mx.z, p.z);
        centroid = centroid + p;
    }
    centroid = centroid / V;

    vector<Plane> planes = buildPlanes(pts);
    if (planes.empty()) { cout << 0 << "\n"; return 0; }

    const auto baseVerts = unitTetraVerts();
    vector<tuple<double,double,double>> angles = {
        {0,0,0}, {0,0,PI/2}, {0,PI/2,0}, {PI/2,0,0},
        {0.61548,0.0,0.785398}, {0.95532,0.61548,0.785398},
        {0.3,0.7,0.2}, {1.1,0.4,0.9}
    };

    double dx = mx.x - mn.x, dy = mx.y - mn.y, dz = mx.z - mn.z;
    double step = edge * 0.74;              // intentionally dense; collision guarded by spacing
    double sep2 = edge * edge * 0.999999;   // center distance sufficient for this conservative set
    long long nx = max(0LL, (long long)floor(dx / step) + 3);
    long long ny = max(0LL, (long long)floor(dy / step) + 3);
    long long nz = max(0LL, (long long)floor(dz / step) + 3);
    long long gridCount = nx * max(1LL, ny) * max(1LL, nz);

    int phaseCount = gridCount > 2000000 ? 1 : (gridCount > 250000 ? 2 : 4);
    int orientCount = gridCount > 1000000 ? 2 : (gridCount > 120000 ? 4 : (int)angles.size());
    const int maxOut = 200000;

    vector<Tetra> best;
    auto start = chrono::steady_clock::now();

    for (int oi=0; oi<orientCount; ++oi) {
        auto [rx, ry, rz] = angles[oi];
        array<Vec,4> verts;
        for (int i=0;i<4;i++) verts[i] = rotateEuler(baseVerts[i], rx, ry, rz);

        for (int ph=0; ph<phaseCount; ++ph) {
            vector<Tetra> cur;
            cur.reserve((size_t)min<long long>(maxOut, max(1LL, gridCount / 4)));
            double ox = ((ph & 1) ? 0.5 : 0.0) * step;
            double oy = ((ph & 2) ? 0.5 : 0.0) * step;
            double oz = ((ph == 3) ? 0.5 : 0.0) * step;

            for (long long ix=0; ix<nx && (int)cur.size()<maxOut; ++ix) {
                if ((ix & 63) == 0 && chrono::duration<double>(chrono::steady_clock::now()-start).count() > 28.0) break;
                for (long long iy=0; iy<ny && (int)cur.size()<maxOut; ++iy) {
                    double rowShift = ((iy & 1) ? 0.5 * step : 0.0);
                    for (long long iz=0; iz<nz && (int)cur.size()<maxOut; ++iz) {
                        Vec c{mn.x + ox + ix*step + rowShift,
                              mn.y + oy + iy*step,
                              mn.z + oz + iz*step + (((ix+iy)&1) ? 0.5*step : 0.0)};
                        if (c.x < mn.x || c.y < mn.y || c.z < mn.z || c.x > mx.x || c.y > mx.y || c.z > mx.z) continue;
                        if (!insideTetra(c, verts, planes, edge)) continue;
                        bool ok = true;
                        for (const auto& t : cur) {
                            Vec diff = c - t.c;
                            if (dotv(diff,diff) < sep2) { ok = false; break; }
                        }
                        if (ok) cur.push_back({c, rx, ry, rz});
                    }
                }
            }
            if (cur.size() > best.size()) best.swap(cur);
            if (chrono::duration<double>(chrono::steady_clock::now()-start).count() > 28.5) break;
        }
        if (chrono::duration<double>(chrono::steady_clock::now()-start).count() > 28.5) break;
    }

    // Fallback for very small containers where the grid phase misses the only feasible center.
    if (best.empty()) {
        auto [rx, ry, rz] = angles[0];
        array<Vec,4> verts = baseVerts;
        if (insideTetra(centroid, verts, planes, edge)) best.push_back({centroid, rx, ry, rz});
    }

    cout.setf(ios::fixed);
    cout.precision(10);
    cout << best.size() << "\n";
    for (const auto& t : best) {
        cout << t.c.x << ' ' << t.c.y << ' ' << t.c.z << ' '
             << t.rx << ' ' << t.ry << ' ' << t.rz << "\n";
    }
    return 0;
}
