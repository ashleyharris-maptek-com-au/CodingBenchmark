use std::cmp::Reverse;
use std::collections::VecDeque;
use std::io::{self, Read};
use std::time::{Duration, Instant};

#[derive(Clone)]
struct Solver {
    w: usize,
    h: usize,
    n: usize,
    cells: Vec<(usize, usize)>,
    adj: Vec<Vec<usize>>,
    start: usize,
    require_cycle: bool,
    path: Vec<usize>,
    used: Vec<bool>,
    deadline: Instant,
}

fn main() {
    let mut s = String::new();
    io::stdin().read_to_string(&mut s).unwrap();
    let mut it = s.split_whitespace();
    let w: usize = it.next().unwrap_or("0").parse().unwrap();
    let h: usize = it.next().unwrap_or("0").parse().unwrap();
    let obstacle_count: usize = it.next().unwrap_or("0").parse().unwrap();
    let require_cycle = it.next().unwrap_or("0") == "1";

    if w == 0 || h == 0 {
        println!("0");
        return;
    }

    let mut blocked = vec![false; w * h];
    for _ in 0..obstacle_count {
        let x: usize = it.next().unwrap().parse().unwrap();
        let y: usize = it.next().unwrap().parse().unwrap();
        if x < w && y < h {
            blocked[y * w + x] = true;
        }
    }

    if blocked[0] {
        println!("0");
        return;
    }

    let free_count = blocked.iter().filter(|&&b| !b).count();
    if free_count == 0 {
        println!("0");
        return;
    }

    if obstacle_count == 0 {
        if let Some(path) = empty_rectangle_path(w, h, require_cycle) {
            print_path(&path);
        } else {
            println!("0");
        }
        return;
    }

    let mut solver = Solver::new(w, h, blocked, require_cycle);
    if !solver.quick_impossible() {
        if solver.n <= 2500 {
            if let Some(path) = solver.solve() {
                let coords: Vec<_> = path.into_iter().map(|v| solver.cells[v]).collect();
                print_path(&coords);
                return;
            }
        }
    }
    println!("0");
}

impl Solver {
    fn new(w: usize, h: usize, blocked: Vec<bool>, require_cycle: bool) -> Self {
        let mut id = vec![None; w * h];
        let mut cells = Vec::new();
        for y in 0..h {
            for x in 0..w {
                let p = y * w + x;
                if !blocked[p] {
                    id[p] = Some(cells.len());
                    cells.push((x, y));
                }
            }
        }
        let n = cells.len();
        let mut adj = vec![Vec::new(); n];
        for i in 0..n {
            let (x, y) = cells[i];
            for (nx, ny) in neighbors_xy(x, y, w, h) {
                if let Some(j) = id[ny * w + nx] {
                    adj[i].push(j);
                }
            }
        }
        let start = id[0].unwrap();
        Self {
            w,
            h,
            n,
            cells,
            adj,
            start,
            require_cycle,
            path: Vec::with_capacity(n),
            used: vec![false; n],
            deadline: Instant::now() + Duration::from_secs(285),
        }
    }

    fn quick_impossible(&self) -> bool {
        if self.n == 1 {
            return self.require_cycle;
        }

        let mut q = VecDeque::new();
        let mut seen = vec![false; self.n];
        q.push_back(self.start);
        seen[self.start] = true;
        while let Some(v) = q.pop_front() {
            for &u in &self.adj[v] {
                if !seen[u] {
                    seen[u] = true;
                    q.push_back(u);
                }
            }
        }
        if seen.iter().any(|&x| !x) {
            return true;
        }

        let black = self.cells.iter().filter(|&&(x, y)| (x + y) % 2 == 0).count();
        let white = self.n - black;
        if self.require_cycle {
            if self.n % 2 == 1 || black != white || self.adj[self.start].len() < 2 {
                return true;
            }
        } else if black.abs_diff(white) > 1 {
            return true;
        }

        let leaves = self.adj.iter().filter(|a| a.len() <= 1).count();
        if self.require_cycle && leaves > 0 {
            return true;
        }
        if !self.require_cycle && leaves > 2 {
            return true;
        }
        false
    }

    fn solve(&mut self) -> Option<Vec<usize>> {
        if self.n == 1 {
            return Some(vec![self.start]);
        }

        self.path.clear();
        self.used.fill(false);
        self.path.push(self.start);
        self.used[self.start] = true;

        if self.n <= 180 || obstacle_density(self) > 0.10 {
            if self.dfs(self.start) {
                return Some(self.path.clone());
            }
        }

        self.greedy_restarts()
    }

    fn dfs(&mut self, v: usize) -> bool {
        if Instant::now() >= self.deadline {
            return false;
        }
        if self.path.len() == self.n {
            return !self.require_cycle || self.adj[v].contains(&self.start);
        }

        let mut cand: Vec<usize> = self.adj[v].iter().copied().filter(|&u| !self.used[u]).collect();
        cand.sort_by_key(|&u| self.onward_degree(u));

        for u in cand {
            self.used[u] = true;
            self.path.push(u);
            if self.local_feasible(u) && self.dfs(u) {
                return true;
            }
            self.path.pop();
            self.used[u] = false;
        }
        false
    }

    fn onward_degree(&self, v: usize) -> usize {
        self.adj[v].iter().filter(|&&u| !self.used[u]).count()
    }

    fn local_feasible(&self, head: usize) -> bool {
        let rem = self.n - self.path.len();
        if rem == 0 {
            return !self.require_cycle || self.adj[head].contains(&self.start);
        }

        let mut forced = 0usize;
        for v in 0..self.n {
            if self.used[v] {
                continue;
            }
            let d = self.adj[v]
                .iter()
                .filter(|&&u| !self.used[u] || u == head)
                .count();
            if d == 0 {
                return false;
            }
            if d == 1 {
                forced += 1;
                if forced > 1 {
                    return false;
                }
            }
        }

        true
    }

    fn greedy_restarts(&self) -> Option<Vec<usize>> {
        let mut seed = 0x9e3779b97f4a7c15u64;
        while Instant::now() < self.deadline {
            let mut used = vec![false; self.n];
            let mut path = Vec::with_capacity(self.n);
            let mut cur = self.start;
            used[cur] = true;
            path.push(cur);
            while path.len() < self.n {
                let mut cand: Vec<_> = self.adj[cur].iter().copied().filter(|&u| !used[u]).collect();
                if cand.is_empty() {
                    break;
                }
                cand.sort_by_key(|&u| {
                    let deg = self.adj[u].iter().filter(|&&z| !used[z]).count();
                    let edge = edge_bias(self.cells[u], self.w, self.h);
                    (deg, Reverse(edge), rand64(&mut seed) & 7)
                });
                cur = cand[0];
                used[cur] = true;
                path.push(cur);
            }
            if path.len() == self.n && (!self.require_cycle || self.adj[cur].contains(&self.start)) {
                return Some(path);
            }
        }
        None
    }
}

fn empty_rectangle_path(w: usize, h: usize, require_cycle: bool) -> Option<Vec<(usize, usize)>> {
    let n = w * h;
    if n == 1 {
        return if require_cycle { None } else { Some(vec![(0, 0)]) };
    }
    if require_cycle {
        if n % 2 == 1 || w == 1 || h == 1 {
            return None;
        }
        let mut p = Vec::with_capacity(n);
        if h % 2 == 0 {
            for y in 0..h {
                if y % 2 == 0 {
                    for x in 0..w {
                        p.push((x, y));
                    }
                } else {
                    for x in (0..w).rev() {
                        p.push((x, y));
                    }
                }
            }
        } else {
            for x in 0..w {
                if x % 2 == 0 {
                    for y in 0..h {
                        p.push((x, y));
                    }
                } else {
                    for y in (0..h).rev() {
                        p.push((x, y));
                    }
                }
            }
        }
        return Some(p);
    }

    let mut p = Vec::with_capacity(n);
    if w == 1 {
        for y in 0..h {
            p.push((0, y));
        }
    } else {
        for y in 0..h {
            if y % 2 == 0 {
                for x in 0..w {
                    p.push((x, y));
                }
            } else {
                for x in (0..w).rev() {
                    p.push((x, y));
                }
            }
        }
    }
    Some(p)
}

fn neighbors_xy(x: usize, y: usize, w: usize, h: usize) -> Vec<(usize, usize)> {
    let mut r = Vec::with_capacity(4);
    if x > 0 {
        r.push((x - 1, y));
    }
    if x + 1 < w {
        r.push((x + 1, y));
    }
    if y > 0 {
        r.push((x, y - 1));
    }
    if y + 1 < h {
        r.push((x, y + 1));
    }
    r
}

fn print_path(path: &[(usize, usize)]) {
    println!("{}", path.len());
    for &(x, y) in path {
        println!("{} {}", x, y);
    }
}

fn obstacle_density(s: &Solver) -> f64 {
    1.0 - (s.n as f64 / (s.w * s.h) as f64)
}

fn edge_bias((x, y): (usize, usize), w: usize, h: usize) -> usize {
    x.min(w - 1 - x).min(y.min(h - 1 - y))
}

fn rand64(seed: &mut u64) -> u64 {
    *seed ^= *seed >> 12;
    *seed ^= *seed << 25;
    *seed ^= *seed >> 27;
    seed.wrapping_mul(2685821657736338717)
}
