using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct V3
    {
        public double X, Y, Z;
        public V3(double x, double y, double z) { X = x; Y = y; Z = z; }
    }

    struct Rot
    {
        public double W, X, Y, Z;
        public int Ax, Ay, Az;
        public int Sx, Sy, Sz;
        public Rot(double w, double x, double y, double z, int ax, int ay, int az, int sx, int sy, int sz)
        {
            W = w; X = x; Y = y; Z = z;
            Ax = ax; Ay = ay; Az = az;
            Sx = sx; Sy = sy; Sz = sz;
        }
    }

    struct Placement
    {
        public double X, Y, Z;
        public Rot R;
        public Placement(double x, double y, double z, Rot r) { X = x; Y = y; Z = z; R = r; }
    }

    static readonly CultureInfo CI = CultureInfo.InvariantCulture;

    public static void Main()
    {
        var tokens = Console.In.ReadToEnd()
            .Split(new[] { ' ', '\t', '\r', '\n' }, StringSplitOptions.RemoveEmptyEntries);
        int p = 0;
        if (tokens.Length < 2) return;

        int vCount = int.Parse(tokens[p++], CI);
        int fCount = int.Parse(tokens[p++], CI);

        var vertices = new V3[vCount];
        for (int i = 0; i < vCount; i++)
            vertices[i] = new V3(Parse(tokens[p++]), Parse(tokens[p++]), Parse(tokens[p++]));

        // Faces are not needed by this conservative non-overlap strategy, but must be consumed.
        p += fCount * 3;

        double W = Parse(tokens[p++]), H = Parse(tokens[p++]), D = Parse(tokens[p++]);
        if (vCount == 0 || W <= 0 || H <= 0 || D <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        double[] min = { double.PositiveInfinity, double.PositiveInfinity, double.PositiveInfinity };
        double[] max = { double.NegativeInfinity, double.NegativeInfinity, double.NegativeInfinity };
        foreach (var v in vertices)
        {
            min[0] = Math.Min(min[0], v.X); min[1] = Math.Min(min[1], v.Y); min[2] = Math.Min(min[2], v.Z);
            max[0] = Math.Max(max[0], v.X); max[1] = Math.Max(max[1], v.Y); max[2] = Math.Max(max[2], v.Z);
        }

        double[] size = { Math.Max(0, max[0] - min[0]), Math.Max(0, max[1] - min[1]), Math.Max(0, max[2] - min[2]) };
        const double eps = 1e-9;
        if (size[0] < eps || size[1] < eps || size[2] < eps)
        {
            Console.WriteLine(0);
            return;
        }

        var rotations = BuildRotations();
        Rot best = rotations[0];
        long bestCount = -1;
        double[] bestDims = new double[3];
        foreach (var r in rotations)
        {
            double[] dims = { size[r.Ax], size[r.Ay], size[r.Az] };
            long nx = FloorCount(W, dims[0]), ny = FloorCount(H, dims[1]), nz = FloorCount(D, dims[2]);
            long count = SaturatingMul(SaturatingMul(nx, ny), nz);
            if (count > bestCount)
            {
                bestCount = count;
                best = r;
                bestDims = dims;
            }
        }

        if (bestCount <= 0)
        {
            Console.WriteLine(0);
            return;
        }

        long maxOutput = ChooseOutputLimit(W, H, D, vCount, fCount, bestCount);
        var placements = GeneratePlacements(vertices, min, max, size, best, bestDims, W, H, D, maxOutput);

        Console.WriteLine(placements.Count.ToString(CI));
        foreach (var pl in placements)
        {
            Console.WriteLine(string.Join(" ",
                F(pl.X), F(pl.Y), F(pl.Z),
                F(pl.R.W), F(pl.R.X), F(pl.R.Y), F(pl.R.Z)));
        }
    }

    static List<Placement> GeneratePlacements(V3[] vertices, double[] min, double[] max, double[] size,
        Rot r, double[] dims, double W, double H, double D, long limit)
    {
        long nx = FloorCount(W, dims[0]), ny = FloorCount(H, dims[1]), nz = FloorCount(D, dims[2]);
        var result = new List<Placement>((int)Math.Min(limit, 1000000));

        double[] rotMin = { double.PositiveInfinity, double.PositiveInfinity, double.PositiveInfinity };
        foreach (var v in vertices)
        {
            double[] a = { v.X, v.Y, v.Z };
            double[] rv = { r.Sx * a[r.Ax], r.Sy * a[r.Ay], r.Sz * a[r.Az] };
            rotMin[0] = Math.Min(rotMin[0], rv[0]);
            rotMin[1] = Math.Min(rotMin[1], rv[1]);
            rotMin[2] = Math.Min(rotMin[2], rv[2]);
        }

        bool extreme = nx * ny > 2000000L || limit > 2000000L;
        if (extreme)
        {
            // Stream-like generation order: fill broad layers without expensive occupancy structures.
            for (long z = 0; z < nz && result.Count < limit; z++)
                for (long y = 0; y < ny && result.Count < limit; y++)
                    for (long x = 0; x < nx && result.Count < limit; x++)
                        result.Add(new Placement(x * dims[0] - rotMin[0], y * dims[1] - rotMin[1], z * dims[2] - rotMin[2], r));
        }
        else
        {
            // For smaller cases, central layers first often leaves a better partial prefix if output is capped.
            long cx = nx / 2, cy = ny / 2, cz = nz / 2;
            var xs = OrderedRange(nx, cx);
            var ys = OrderedRange(ny, cy);
            var zs = OrderedRange(nz, cz);
            foreach (long z in zs)
                foreach (long y in ys)
                    foreach (long x in xs)
                    {
                        if (result.Count >= limit) return result;
                        result.Add(new Placement(x * dims[0] - rotMin[0], y * dims[1] - rotMin[1], z * dims[2] - rotMin[2], r));
                    }
        }
        return result;
    }

    static IEnumerable<long> OrderedRange(long n, long c)
    {
        if (n <= 0) yield break;
        yield return c;
        for (long d = 1; d < n; d++)
        {
            long a = c - d, b = c + d;
            if (a >= 0) yield return a;
            if (b < n) yield return b;
        }
    }

    static long ChooseOutputLimit(double W, double H, double D, int v, int f, long available)
    {
        double volume = W * H * D;
        long cap;
        if (volume <= 8000 && v + f <= 200) cap = 1000000;
        else if (volume <= 1000000) cap = 500000;
        else if (volume <= 27000000) cap = 200000;
        else cap = 100000;
        return Math.Min(available, cap);
    }

    static long FloorCount(double outer, double step)
    {
        if (step <= 0 || outer + 1e-9 < step) return 0;
        return Math.Max(0, (long)Math.Floor((outer + 1e-9) / step));
    }

    static long SaturatingMul(long a, long b)
    {
        if (a <= 0 || b <= 0) return 0;
        if (a > long.MaxValue / b) return long.MaxValue;
        return a * b;
    }

    static List<Rot> BuildRotations()
    {
        return new List<Rot>
        {
            new Rot(1,0,0,0, 0,1,2, 1,1,1),
            new Rot(0,1,0,0, 0,1,2, 1,-1,-1),
            new Rot(0,0,1,0, 0,1,2, -1,1,-1),
            new Rot(0,0,0,1, 0,1,2, -1,-1,1),
            new Rot(Math.Sqrt(0.5), Math.Sqrt(0.5),0,0, 0,2,1, 1,-1,1),
            new Rot(Math.Sqrt(0.5),-Math.Sqrt(0.5),0,0, 0,2,1, 1,1,-1),
            new Rot(Math.Sqrt(0.5),0,Math.Sqrt(0.5),0, 2,1,0, 1,1,-1),
            new Rot(Math.Sqrt(0.5),0,-Math.Sqrt(0.5),0, 2,1,0, -1,1,1),
            new Rot(Math.Sqrt(0.5),0,0,Math.Sqrt(0.5), 1,0,2, -1,1,1),
            new Rot(Math.Sqrt(0.5),0,0,-Math.Sqrt(0.5), 1,0,2, 1,-1,1),
        };
    }

    static double Parse(string s) => double.Parse(s, CI);
    static string F(double x) => x.ToString("0.#########", CI);
}
