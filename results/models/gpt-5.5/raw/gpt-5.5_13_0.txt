using namespace std;

struct PairHash {
    size_t operator()(const pair<uint64_t, uint64_t>& p) const noexcept {
        uint64_t x = p.first ^ (p.second + 0x9e3779b97f4a7c15ULL + (p.first << 6) + (p.first >> 2));
        x ^= x >> 30;
        x *= 0xbf58476d1ce4e5b9ULL;
        x ^= x >> 27;
        x *= 0x94d049bb133111ebULL;
        x ^= x >> 31;
        return static_cast<size_t>(x);
    }
};

struct RH {
    static constexpr uint64_t B1 = 911382323ULL;
    static constexpr uint64_t B2 = 972663749ULL;
    vector<uint64_t> h1, h2;

    explicit RH(const string& s) : h1(s.size() + 1, 0), h2(s.size() + 1, 0) {
        for (size_t i = 0; i < s.size(); ++i) {
            uint64_t v = static_cast<unsigned char>(s[i]) + 1ULL;
            h1[i + 1] = h1[i] * B1 + v;
            h2[i + 1] = h2[i] * B2 + v;
        }
    }

    pair<uint64_t, uint64_t> get(size_t pos, size_t len,
                                 const vector<uint64_t>& p1,
                                 const vector<uint64_t>& p2) const {
        return {h1[pos + len] - h1[pos] * p1[len],
                h2[pos + len] - h2[pos] * p2[len]};
    }
};

static bool same_substr_all(const vector<string>& s, const string& candidate) {
    for (size_t i = 1; i < s.size(); ++i) {
        if (s[i].find(candidate) == string::npos) return false;
    }
    return true;
}

static bool exists_length(size_t len,
                          const vector<string>& s,
                          const vector<RH>& rh,
                          const vector<uint64_t>& p1,
                          const vector<uint64_t>& p2,
                          size_t base_index,
                          size_t& answer_pos) {
    if (len == 0) {
        answer_pos = 0;
        return true;
    }

    unordered_map<pair<uint64_t, uint64_t>, uint32_t, PairHash> seen;
    const string& base = s[base_index];
    size_t base_windows = base.size() - len + 1;
    seen.reserve(base_windows * 2 + 1);
    for (size_t i = 0; i < base_windows; ++i) {
        auto key = rh[base_index].get(i, len, p1, p2);
        seen.emplace(key, 0U);
    }

    uint32_t stamp = 1;
    for (size_t si = 0; si < s.size(); ++si) {
        if (si == base_index) continue;
        if (s[si].size() < len) return false;

        bool any = false;
        unordered_set<pair<uint64_t, uint64_t>, PairHash> touched;
        size_t windows = s[si].size() - len + 1;
        touched.reserve(min(windows, seen.size()) * 2 + 1);

        for (size_t j = 0; j < windows; ++j) {
            auto key = rh[si].get(j, len, p1, p2);
            auto it = seen.find(key);
            if (it != seen.end() && it->second + 1 == stamp) {
                it->second = stamp;
                touched.insert(key);
                any = true;
            }
        }
        if (!any) return false;

        for (auto it = seen.begin(); it != seen.end(); ) {
            if (it->second == stamp) ++it;
            else it = seen.erase(it);
        }
        ++stamp;
    }

    for (size_t i = 0; i < base_windows; ++i) {
        auto key = rh[base_index].get(i, len, p1, p2);
        if (seen.find(key) != seen.end()) {
            string cand = base.substr(i, len);
            if (same_substr_all(s, cand)) {
                answer_pos = i;
                return true;
            }
        }
    }
    return false;
}

static string solve_binary_hash(vector<string>& s) {
    size_t base_index = 0;
    for (size_t i = 1; i < s.size(); ++i) {
        if (s[i].size() < s[base_index].size()) base_index = i;
    }
    size_t min_len = s[base_index].size();
    if (min_len == 0) return "";

    vector<uint64_t> p1(min_len + 1), p2(min_len + 1);
    p1[0] = p2[0] = 1;
    for (size_t i = 0; i < min_len; ++i) {
        p1[i + 1] = p1[i] * RH::B1;
        p2[i + 1] = p2[i] * RH::B2;
    }

    vector<RH> rh;
    rh.reserve(s.size());
    for (const auto& str : s) rh.emplace_back(str);

    size_t lo = 0, hi = min_len, best_len = 0, best_pos = 0;
    while (lo <= hi) {
        size_t mid = lo + (hi - lo) / 2;
        size_t pos = 0;
        if (exists_length(mid, s, rh, p1, p2, base_index, pos)) {
            best_len = mid;
            best_pos = pos;
            lo = mid + 1;
        } else {
            if (mid == 0) break;
            hi = mid - 1;
        }
    }
    return s[base_index].substr(best_pos, best_len);
}

static string solve_suffix_array(const vector<string>& input) {
    string text;
    vector<int> owner;
    text.reserve(1024);
    owner.reserve(1024);
    for (int i = 0; i < static_cast<int>(input.size()); ++i) {
        for (unsigned char c : input[i]) {
            text.push_back(static_cast<char>(c == 0 ? 1 : c));
            owner.push_back(i);
        }
        text.push_back(static_cast<char>(i + 1));
        owner.push_back(-1);
    }
    int n = static_cast<int>(text.size());
    vector<int> sa(n), r(n), tmp(n);
    for (int i = 0; i < n; ++i) sa[i] = i, r[i] = static_cast<unsigned char>(text[i]);
    for (int k = 1;; k <<= 1) {
        auto cmp = [&](int a, int b) {
            if (r[a] != r[b]) return r[a] < r[b];
            int ra = a + k < n ? r[a + k] : -1;
            int rb = b + k < n ? r[b + k] : -1;
            return ra < rb;
        };
        sort(sa.begin(), sa.end(), cmp);
        tmp[sa[0]] = 0;
        for (int i = 1; i < n; ++i) tmp[sa[i]] = tmp[sa[i - 1]] + (cmp(sa[i - 1], sa[i]) ? 1 : 0);
        r.swap(tmp);
        if (r[sa[n - 1]] == n - 1) break;
    }

    vector<int> rank(n), lcp(max(0, n - 1));
    for (int i = 0; i < n; ++i) rank[sa[i]] = i;
    for (int i = 0, h = 0; i < n; ++i) {
        int pos = rank[i];
        if (pos == n - 1) { h = 0; continue; }
        int j = sa[pos + 1];
        while (i + h < n && j + h < n && text[i + h] == text[j + h] && owner[i + h] >= 0 && owner[j + h] >= 0) ++h;
        lcp[pos] = h;
        if (h) --h;
    }

    int need = static_cast<int>(input.size()), have = 0, best = 0, best_pos = 0;
    vector<int> cnt(need, 0);
    int left = 0;
    for (int right = 0; right < n; ++right) {
        int ow = owner[sa[right]];
        if (ow >= 0 && cnt[ow]++ == 0) ++have;
        while (have == need && left <= right) {
            int mn = numeric_limits<int>::max();
            for (int i = left; i < right; ++i) mn = min(mn, lcp[i]);
            if (mn > best) {
                best = mn;
                best_pos = sa[left];
            }
            int low = owner[sa[left]];
            if (low >= 0 && --cnt[low] == 0) --have;
            ++left;
        }
    }
    return best > 0 ? text.substr(best_pos, best) : string();
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int n;
    if (!(cin >> n)) return 0;
    string dummy;
    getline(cin, dummy);
    vector<string> s(n);
    for (int i = 0; i < n; ++i) getline(cin, s[i]);
    if (n <= 0) { cout << '\n'; return 0; }
    if (n == 1) { cout << s[0] << '\n'; return 0; }
    for (const auto& x : s) if (x.empty()) { cout << '\n'; return 0; }

    size_t total = 0, min_len = s[0].size();
    for (const auto& x : s) {
        total += x.size();
        min_len = min(min_len, x.size());
    }

    string ans;
    if (n <= 10 && total <= 200000) ans = solve_suffix_array(s);
    else ans = solve_binary_hash(s);
    cout << ans << '\n';
    return 0;
}
