use std::collections::VecDeque;
use std::io::{self, Read, Write};

fn main() {
    let mut input = Vec::new();
    io::stdin().read_to_end(&mut input).unwrap();
    let mut p = 0usize;

    while p < input.len() && input[p].is_ascii_whitespace() {
        p += 1;
    }
    let h = read_usize(&input, &mut p);
    let w = read_usize(&input, &mut p);
    if h == 0 || w == 0 {
        return;
    }

    while p < input.len() && (input[p] == b'\r' || input[p] == b'\n') {
        p += 1;
    }

    let n = match h.checked_mul(w) {
        Some(v) => v,
        None => {
            println!("0");
            return;
        }
    };

    let mut grid = vec![b'#'; n];
    let mut start = usize::MAX;
    let mut goal = usize::MAX;

    for y in 0..h {
        for x in 0..w {
            if p >= input.len() {
                println!("0");
                return;
            }
            let c = input[p];
            p += 1;
            let idx = y * w + x;
            grid[idx] = c;
            if c == b'A' {
                start = idx;
            } else if c == b'B' {
                goal = idx;
            }
        }
        if p < input.len() && input[p] == b'\r' {
            p += 1;
        }
        if p < input.len() && input[p] == b'\n' {
            p += 1;
        }
    }

    if start == usize::MAX || goal == usize::MAX {
        println!("0");
        return;
    }
    if start == goal {
        println!("1\n{} {}", start % w, start / w);
        return;
    }

    // Direction used to first enter each cell: 1=up, 2=down, 3=left, 4=right.
    // This single byte array is both the visited set and the predecessor map.
    let mut parent_dir = vec![0u8; n];
    parent_dir[start] = 255;

    let mut q = VecDeque::new();
    q.push_back(start);

    while let Some(cur) = q.pop_front() {
        let x = cur % w;
        let y = cur / w;

        if y > 0 {
            let ni = cur - w;
            if parent_dir[ni] == 0 && grid[ni] != b'#' {
                parent_dir[ni] = 1;
                if ni == goal { break; }
                q.push_back(ni);
            }
        }
        if y + 1 < h {
            let ni = cur + w;
            if parent_dir[ni] == 0 && grid[ni] != b'#' {
                parent_dir[ni] = 2;
                if ni == goal { break; }
                q.push_back(ni);
            }
        }
        if x > 0 {
            let ni = cur - 1;
            if parent_dir[ni] == 0 && grid[ni] != b'#' {
                parent_dir[ni] = 3;
                if ni == goal { break; }
                q.push_back(ni);
            }
        }
        if x + 1 < w {
            let ni = cur + 1;
            if parent_dir[ni] == 0 && grid[ni] != b'#' {
                parent_dir[ni] = 4;
                if ni == goal { break; }
                q.push_back(ni);
            }
        }
    }

    if parent_dir[goal] == 0 {
        println!("0");
        return;
    }

    let mut path = Vec::new();
    let mut cur = goal;
    loop {
        path.push(cur);
        if cur == start {
            break;
        }
        cur = match parent_dir[cur] {
            1 => cur + w, // came from below
            2 => cur - w, // came from above
            3 => cur + 1, // came from right
            4 => cur - 1, // came from left
            _ => {
                println!("0");
                return;
            }
        };
    }
    path.reverse();

    let mut out = String::new();
    out.push_str(&format!("{}\n", path.len()));
    for idx in path {
        out.push_str(&format!("{} {}\n", idx % w, idx / w));
    }
    let _ = io::stdout().write_all(out.as_bytes());
}

fn read_usize(input: &[u8], p: &mut usize) -> usize {
    while *p < input.len() && input[*p].is_ascii_whitespace() {
        *p += 1;
    }
    let mut v = 0usize;
    while *p < input.len() && input[*p].is_ascii_digit() {
        v = v * 10 + (input[*p] - b'0') as usize;
        *p += 1;
    }
    v
}
