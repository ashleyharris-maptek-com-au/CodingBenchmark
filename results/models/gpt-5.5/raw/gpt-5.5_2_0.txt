using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    struct Edge
    {
        public int U, V, W;
        public Edge(int u, int v, int w) { U = u; V = v; W = w; }
    }

    struct Arc
    {
        public int To, EdgeId;
        public Arc(int to, int edgeId) { To = to; EdgeId = edgeId; }
    }

    sealed class FastScanner
    {
        readonly Stream s = Console.OpenStandardInput();
        readonly byte[] buf = new byte[1 << 20];
        int len, ptr;

        int Read()
        {
            if (ptr >= len)
            {
                len = s.Read(buf, 0, buf.Length);
                ptr = 0;
                if (len == 0) return -1;
            }
            return buf[ptr++];
        }

        public int NextInt()
        {
            int c, sign = 1, x = 0;
            do c = Read(); while (c <= 32 && c >= 0);
            if (c == '-') { sign = -1; c = Read(); }
            while (c > 32)
            {
                x = x * 10 + c - '0';
                c = Read();
            }
            return x * sign;
        }
    }

    sealed class MinHeap
    {
        long[] keys;
        int[] vals;
        public int Count;
        public MinHeap(int cap)
        {
            keys = new long[Math.Max(4, cap)];
            vals = new int[keys.Length];
        }
        public void Push(long key, int val)
        {
            if (Count == keys.Length)
            {
                Array.Resize(ref keys, keys.Length * 2);
                Array.Resize(ref vals, vals.Length * 2);
            }
            int i = Count++;
            while (i > 0)
            {
                int p = (i - 1) >> 1;
                if (keys[p] <= key) break;
                keys[i] = keys[p]; vals[i] = vals[p]; i = p;
            }
            keys[i] = key; vals[i] = val;
        }
        public bool TryPop(out long key, out int val)
        {
            if (Count == 0) { key = 0; val = 0; return false; }
            key = keys[0]; val = vals[0];
            long lk = keys[--Count]; int lv = vals[Count];
            int i = 0;
            while (true)
            {
                int l = i * 2 + 1;
                if (l >= Count) break;
                int r = l + 1, c = (r < Count && keys[r] < keys[l]) ? r : l;
                if (keys[c] >= lk) break;
                keys[i] = keys[c]; vals[i] = vals[c]; i = c;
            }
            if (Count > 0) { keys[i] = lk; vals[i] = lv; }
            return true;
        }
    }

    sealed class Dsu
    {
        readonly int[] p, r;
        public Dsu(int n)
        {
            p = new int[n]; r = new int[n];
            for (int i = 0; i < n; i++) p[i] = i;
        }
        public int Find(int x)
        {
            while (p[x] != x) { p[x] = p[p[x]]; x = p[x]; }
            return x;
        }
        public bool Union(int a, int b)
        {
            a = Find(a); b = Find(b);
            if (a == b) return false;
            if (r[a] < r[b]) { int t = a; a = b; b = t; }
            p[b] = a;
            if (r[a] == r[b]) r[a]++;
            return true;
        }
    }

    const long INF = long.MaxValue / 4;
    static int n, m;
    static Edge[] edges = null!;
    static List<Arc>[] original = null!;
    static int[] degree = null!;

    static void AddRoutePath(List<int> duplicatedEdges, int s, int t)
    {
        if (s == t) return;
        var dist = new long[n];
        var parentEdge = new int[n];
        Array.Fill(dist, INF);
        Array.Fill(parentEdge, -1);
        var heap = new MinHeap(64);
        dist[s] = 0;
        heap.Push(0, s);
        while (heap.TryPop(out long d, out int u))
        {
            if (d != dist[u]) continue;
            if (u == t) break;
            foreach (var a in original[u])
            {
                long nd = d + edges[a.EdgeId].W;
                if (nd < dist[a.To])
                {
                    dist[a.To] = nd;
                    parentEdge[a.To] = a.EdgeId;
                    heap.Push(nd, a.To);
                }
            }
        }
        if (parentEdge[t] < 0) return;
        for (int v = t; v != s;)
        {
            int eid = parentEdge[v];
            duplicatedEdges.Add(eid);
            int u = edges[eid].U ^ edges[eid].V ^ v;
            v = u;
        }
    }

    static void ExactOddMatching(List<int> odds, List<int> duplicatedEdges)
    {
        int k = odds.Count;
        var dist = new long[k][];
        var par = new int[k][];
        for (int i = 0; i < k; i++)
        {
            dist[i] = new long[n];
            par[i] = new int[n];
            Array.Fill(dist[i], INF);
            Array.Fill(par[i], -1);
            var heap = new MinHeap(n / 4 + 16);
            int src = odds[i];
            dist[i][src] = 0;
            heap.Push(0, src);
            while (heap.TryPop(out long d, out int u))
            {
                if (d != dist[i][u]) continue;
                foreach (var a in original[u])
                {
                    long nd = d + edges[a.EdgeId].W;
                    if (nd < dist[i][a.To])
                    {
                        dist[i][a.To] = nd;
                        par[i][a.To] = a.EdgeId;
                        heap.Push(nd, a.To);
                    }
                }
            }
        }

        int full = 1 << k;
        var dp = new long[full];
        var choice = new int[full];
        Array.Fill(dp, INF);
        dp[0] = 0;
        for (int mask = 1; mask < full; mask++)
        {
            int i = 0;
            while (((mask >> i) & 1) == 0) i++;
            int withoutI = mask ^ (1 << i);
            for (int j = i + 1; j < k; j++) if (((withoutI >> j) & 1) != 0)
            {
                int nm = withoutI ^ (1 << j);
                long cand = dp[nm] + dist[i][odds[j]];
                if (cand < dp[mask])
                {
                    dp[mask] = cand;
                    choice[mask] = (i << 16) | j;
                }
            }
        }

        for (int mask = full - 1; mask != 0;)
        {
            int ch = choice[mask], i = ch >> 16, j = ch & 65535;
            int s = odds[i], v = odds[j];
            while (v != s)
            {
                int eid = par[i][v];
                duplicatedEdges.Add(eid);
                v = edges[eid].U ^ edges[eid].V ^ v;
            }
            mask ^= (1 << i);
            mask ^= (1 << j);
        }
    }

    static void HeuristicPairing(List<int> odds, List<int> duplicatedEdges)
    {
        odds.Sort();
        var unmatched = new SortedSet<int>(odds);
        var rnd = new Random(1);
        while (unmatched.Count > 0)
        {
            int s = unmatched.Min;
            unmatched.Remove(s);
            if (unmatched.Count == 0) break;

            int best = -1, bestDirectW = int.MaxValue;
            foreach (var a in original[s])
            {
                if (unmatched.Contains(a.To) && edges[a.EdgeId].W < bestDirectW)
                {
                    bestDirectW = edges[a.EdgeId].W;
                    best = a.To;
                }
            }

            if (best < 0)
            {
                int targetIndex = rnd.Next(unmatched.Count);
                foreach (int x in unmatched)
                {
                    if (targetIndex-- == 0) { best = x; break; }
                }
            }

            unmatched.Remove(best);
            AddRoutePath(duplicatedEdges, s, best);
        }
    }

    static List<int> Euler(List<Arc>[] graph, int edgeCount)
    {
        var used = new bool[edgeCount];
        var it = new int[n];
        var stack = new List<int>(edgeCount + 1) { 0 };
        var route = new List<int>(edgeCount + 1);

        while (stack.Count > 0)
        {
            int v = stack[stack.Count - 1];
            var adj = graph[v];
            while (it[v] < adj.Count && used[adj[it[v]].EdgeId]) it[v]++;
            if (it[v] == adj.Count)
            {
                route.Add(v);
                stack.RemoveAt(stack.Count - 1);
            }
            else
            {
                var a = adj[it[v]++];
                if (used[a.EdgeId]) continue;
                used[a.EdgeId] = true;
                stack.Add(a.To);
            }
        }
        route.Reverse();
        return route;
    }

    public static void Main()
    {
        var fs = new FastScanner();
        n = fs.NextInt();
        m = fs.NextInt();
        edges = new Edge[m];
        original = new List<Arc>[n];
        degree = new int[n];
        for (int i = 0; i < n; i++) original[i] = new List<Arc>();

        var dsu = new Dsu(n);
        for (int i = 0; i < m; i++)
        {
            int u = fs.NextInt(), v = fs.NextInt(), w = fs.NextInt();
            edges[i] = new Edge(u, v, w);
            original[u].Add(new Arc(v, i));
            original[v].Add(new Arc(u, i));
            degree[u]++; degree[v]++;
            dsu.Union(u, v);
        }

        var odds = new List<int>();
        for (int i = 0; i < n; i++)
            if ((degree[i] & 1) != 0) odds.Add(i);

        var duplicated = new List<int>();
        if (odds.Count <= 22) ExactOddMatching(odds, duplicated);
        else HeuristicPairing(odds, duplicated);

        int totalEdges = m + duplicated.Count;
        var graph = new List<Arc>[n];
        for (int i = 0; i < n; i++) graph[i] = new List<Arc>(original[i].Count + 2);
        int id = 0;
        for (int i = 0; i < m; i++, id++)
        {
            int u = edges[i].U, v = edges[i].V;
            graph[u].Add(new Arc(v, id));
            graph[v].Add(new Arc(u, id));
        }
        foreach (int eid0 in duplicated)
        {
            int u = edges[eid0].U, v = edges[eid0].V;
            graph[u].Add(new Arc(v, id));
            graph[v].Add(new Arc(u, id));
            id++;
        }

        var route = Euler(graph, totalEdges);
        var sb = new StringBuilder(route.Count * 3);
        sb.Append(route.Count);
        for (int i = 0; i < route.Count; i++)
        {
            sb.Append(' ');
            sb.Append(route[i]);
        }
        Console.Write(sb.ToString());
    }
}
