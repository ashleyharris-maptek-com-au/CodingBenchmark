#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <vector>

struct Rect {
    int id;
    int w;
    int h;
};

struct Placement {
    int board = -1;
    int x = 0;
    int y = 0;
    int rotated = 0;
};

struct Node {
    int x;
    int y;
    int w;
};

struct Candidate {
    bool ok = false;
    int node = -1;
    int x = 0;
    int y = 0;
    int w = 0;
    int h = 0;
    int rotated = 0;
    int top = 0;
    long long waste = 0;
};

class SkylineBin {
public:
    SkylineBin(int bw, int bh) : bw_(bw), bh_(bh), usedArea_(0), height_(0) {
        nodes_.push_back({0, 0, bw_});
    }

    Candidate best(int rw, int rh) const {
        Candidate best;
        tryOrientation(rw, rh, 0, best);
        if (rw != rh) tryOrientation(rh, rw, 1, best);
        return best;
    }

    void place(const Candidate& c) {
        Node n{c.x, c.y + c.h, c.w};
        nodes_.insert(nodes_.begin() + c.node, n);

        for (std::size_t i = c.node + 1; i < nodes_.size(); ++i) {
            Node& cur = nodes_[i];
            Node& prev = nodes_[i - 1];
            int prevRight = prev.x + prev.w;
            if (cur.x < prevRight) {
                int shrink = prevRight - cur.x;
                cur.x += shrink;
                cur.w -= shrink;
                if (cur.w <= 0) {
                    nodes_.erase(nodes_.begin() + i);
                    --i;
                } else {
                    break;
                }
            } else {
                break;
            }
        }

        for (std::size_t i = 0; i + 1 < nodes_.size();) {
            if (nodes_[i].y == nodes_[i + 1].y) {
                nodes_[i].w += nodes_[i + 1].w;
                nodes_.erase(nodes_.begin() + i + 1);
            } else {
                ++i;
            }
        }

        usedArea_ += 1LL * c.w * c.h;
        height_ = std::max(height_, c.y + c.h);
    }

    long long usedArea() const { return usedArea_; }
    int height() const { return height_; }
    int nodeCount() const { return static_cast<int>(nodes_.size()); }

private:
    void tryOrientation(int rw, int rh, int rotated, Candidate& bestCand) const {
        if (rw > bw_ || rh > bh_) return;
        for (int i = 0; i < static_cast<int>(nodes_.size()); ++i) {
            int y = 0;
            if (!fits(i, rw, rh, y)) continue;

            int top = y + rh;
            long long waste = 0;
            int widthLeft = rw;
            for (std::size_t j = i; widthLeft > 0 && j < nodes_.size(); ++j) {
                int covered = std::min(widthLeft, nodes_[j].w);
                waste += 1LL * covered * (y - nodes_[j].y);
                widthLeft -= covered;
            }

            bool take = false;
            if (!bestCand.ok) {
                take = true;
            } else if (top != bestCand.top) {
                take = top < bestCand.top;
            } else if (waste != bestCand.waste) {
                take = waste < bestCand.waste;
            } else if (y != bestCand.y) {
                take = y < bestCand.y;
            } else {
                take = nodes_[i].w < nodes_[bestCand.node].w;
            }

            if (take) {
                bestCand.ok = true;
                bestCand.node = i;
                bestCand.x = nodes_[i].x;
                bestCand.y = y;
                bestCand.w = rw;
                bestCand.h = rh;
                bestCand.rotated = rotated;
                bestCand.top = top;
                bestCand.waste = waste;
            }
        }
    }

    bool fits(int start, int rw, int rh, int& y) const {
        int x = nodes_[start].x;
        if (x + rw > bw_) return false;

        int widthLeft = rw;
        y = nodes_[start].y;
        for (std::size_t i = start; widthLeft > 0; ++i) {
            if (i >= nodes_.size()) return false;
            y = std::max(y, nodes_[i].y);
            if (y + rh > bh_) return false;
            widthLeft -= nodes_[i].w;
        }
        return true;
    }

    int bw_;
    int bh_;
    long long usedArea_;
    int height_;
    std::vector<Node> nodes_;
};

static bool canFitBoard(const Rect& r, int bw, int bh) {
    return (r.w <= bw && r.h <= bh) || (r.h <= bw && r.w <= bh);
}

int main() {
    std::ios::sync_with_stdio(false);
    std::cin.tie(nullptr);

    int n, boardW, boardH;
    if (!(std::cin >> n >> boardW >> boardH)) return 0;

    std::vector<Rect> rects(n);
    for (int i = 0; i < n; ++i) {
        std::cin >> rects[i].w >> rects[i].h;
        rects[i].id = i;
    }

    std::vector<Placement> ans(n);
    std::vector<Rect> items;
    items.reserve(n);

    for (const Rect& r : rects) {
        if (canFitBoard(r, boardW, boardH)) {
            items.push_back(r);
        } else {
            ans[r.id] = {-1, 0, 0, 0};
        }
    }

    std::sort(items.begin(), items.end(), [](const Rect& a, const Rect& b) {
        long long aa = 1LL * a.w * a.h;
        long long ba = 1LL * b.w * b.h;
        if (aa != ba) return aa > ba;
        int amaj = std::max(a.w, a.h), bmaj = std::max(b.w, b.h);
        if (amaj != bmaj) return amaj > bmaj;
        int amin = std::min(a.w, a.h), bmin = std::min(b.w, b.h);
        return amin > bmin;
    });

    std::vector<SkylineBin> bins;
    bins.reserve(std::max(1, n / 8));

    const int exhaustiveLimit = 80;
    const int beamSmall = 96;
    const int beamLarge = 24;

    for (const Rect& r : items) {
        Candidate chosen;
        int chosenBin = -1;

        std::vector<int> order(bins.size());
        std::iota(order.begin(), order.end(), 0);
        if (static_cast<int>(bins.size()) > exhaustiveLimit) {
            std::partial_sort(order.begin(),
                              order.begin() + std::min<int>(order.size(), beamSmall),
                              order.end(),
                              [&](int a, int b) {
                                  if (bins[a].height() != bins[b].height()) return bins[a].height() < bins[b].height();
                                  return bins[a].usedArea() > bins[b].usedArea();
                              });
            order.resize(std::min<int>(order.size(), beamSmall));
        }

        int checked = 0;
        for (int bi : order) {
            Candidate c = bins[bi].best(r.w, r.h);
            if (!c.ok) continue;

            bool take = false;
            if (!chosen.ok) {
                take = true;
            } else if (c.top != chosen.top) {
                take = c.top < chosen.top;
            } else if (c.waste != chosen.waste) {
                take = c.waste < chosen.waste;
            } else {
                take = bins[bi].usedArea() > bins[chosenBin].usedArea();
            }

            if (take) {
                chosen = c;
                chosenBin = bi;
            }

            ++checked;
            if (static_cast<int>(bins.size()) > 1000 && checked >= beamLarge && chosen.ok) break;
        }

        if (!chosen.ok) {
            bins.emplace_back(boardW, boardH);
            chosenBin = static_cast<int>(bins.size()) - 1;
            chosen = bins.back().best(r.w, r.h);
        }

        bins[chosenBin].place(chosen);
        ans[r.id] = {chosenBin, chosen.x, chosen.y, chosen.rotated};
    }

    std::cout << bins.size() << '\n';
    for (int i = 0; i < n; ++i) {
        std::cout << ans[i].board << ' ' << ans[i].x << ' ' << ans[i].y << ' ' << ans[i].rotated << '\n';
    }

    return 0;
}
