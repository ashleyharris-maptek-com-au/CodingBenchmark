using namespace std;

struct City {
    double x, y;
    uint32_t id;
};

static inline double dist2(const City& a, const City& b) {
    double dx = a.x - b.x, dy = a.y - b.y;
    return dx * dx + dy * dy;
}

static inline bool better2opt(const City& a, const City& b, const City& c, const City& d) {
    double oldLen = sqrt(dist2(a, b)) + sqrt(dist2(c, d));
    double newLen = sqrt(dist2(a, c)) + sqrt(dist2(b, d));
    return newLen + 1e-12 < oldLen;
}

static vector<uint32_t> nearest_neighbor_small(const vector<City>& p) {
    const uint32_t n = (uint32_t)p.size();
    vector<uint32_t> tour;
    vector<unsigned char> used(n, 0);
    tour.reserve(n);
    uint32_t cur = 0;
    tour.push_back(cur);
    used[cur] = 1;
    for (uint32_t step = 1; step < n; ++step) {
        uint32_t best = 0;
        double bd = numeric_limits<double>::infinity();
        for (uint32_t i = 0; i < n; ++i) {
            if (!used[i]) {
                double d = dist2(p[cur], p[i]);
                if (d < bd) {
                    bd = d;
                    best = i;
                }
            }
        }
        cur = best;
        used[cur] = 1;
        tour.push_back(cur);
    }
    return tour;
}

static void limited_2opt(vector<uint32_t>& tour, const vector<City>& p, int passes, int window) {
    const int n = (int)tour.size();
    if (n < 4) return;
    for (int pass = 0; pass < passes; ++pass) {
        bool changed = false;
        for (int i = 0; i < n - 2; ++i) {
            int lim = min(n - 1, i + window);
            for (int k = i + 2; k <= lim; ++k) {
                if (i == 0 && k == n - 1) continue;
                const City& a = p[tour[i]];
                const City& b = p[tour[(i + 1) % n]];
                const City& c = p[tour[k]];
                const City& d = p[tour[(k + 1) % n]];
                if (better2opt(a, b, c, d)) {
                    reverse(tour.begin() + i + 1, tour.begin() + k + 1);
                    changed = true;
                }
            }
        }
        if (!changed) break;
    }
}

static uint64_t part1by1(uint32_t x) {
    uint64_t v = x;
    v = (v | (v << 16)) & 0x0000FFFF0000FFFFull;
    v = (v | (v << 8))  & 0x00FF00FF00FF00FFull;
    v = (v | (v << 4))  & 0x0F0F0F0F0F0F0F0Full;
    v = (v | (v << 2))  & 0x3333333333333333ull;
    v = (v | (v << 1))  & 0x5555555555555555ull;
    return v;
}

static uint64_t morton_code(double x, double y, double minx, double miny, double scale) {
    double sx = (x - minx) * scale;
    double sy = (y - miny) * scale;
    if (sx < 0) sx = 0; if (sx > 4294967295.0) sx = 4294967295.0;
    if (sy < 0) sy = 0; if (sy > 4294967295.0) sy = 4294967295.0;
    uint32_t ix = (uint32_t)sx, iy = (uint32_t)sy;
    return part1by1(ix) | (part1by1(iy) << 1);
}

static vector<uint32_t> spatial_sort_tour(vector<City>& p) {
    const size_t n = p.size();
    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (const auto& c : p) {
        minx = min(minx, c.x); maxx = max(maxx, c.x);
        miny = min(miny, c.y); maxy = max(maxy, c.y);
    }
    double span = max(maxx - minx, maxy - miny);
    double scale = span > 0 ? 4294967295.0 / span : 1.0;

    vector<pair<uint64_t, uint32_t>> key;
    key.reserve(n);
    for (const auto& c : p)
        key.push_back({morton_code(c.x, c.y, minx, miny, scale), c.id});
    sort(key.begin(), key.end(), [](const auto& a, const auto& b) {
        if (a.first != b.first) return a.first < b.first;
        return a.second < b.second;
    });

    vector<uint32_t> tour;
    tour.reserve(n);
    size_t pos0 = 0;
    for (size_t i = 0; i < key.size(); ++i) {
        if (key[i].second == 0) {
            pos0 = i;
            break;
        }
    }
    for (size_t i = 0; i < n; ++i)
        tour.push_back(key[(pos0 + i) % n].second);
    return tour;
}

static vector<uint32_t> grid_nearest_tour(const vector<City>& p) {
    const uint32_t n = (uint32_t)p.size();
    double minx = p[0].x, maxx = p[0].x, miny = p[0].y, maxy = p[0].y;
    for (const auto& c : p) {
        minx = min(minx, c.x); maxx = max(maxx, c.x);
        miny = min(miny, c.y); maxy = max(maxy, c.y);
    }

    uint32_t g = max<uint32_t>(1, (uint32_t)sqrt((double)n / 8.0));
    g = min<uint32_t>(g, 4096);
    double sx = (maxx > minx) ? (double)g / (maxx - minx) : 1.0;
    double sy = (maxy > miny) ? (double)g / (maxy - miny) : 1.0;

    vector<vector<uint32_t>> cells((size_t)g * g);
    auto cell_of = [&](const City& c) {
        uint32_t ix = min<uint32_t>(g - 1, (uint32_t)max(0.0, (c.x - minx) * sx));
        uint32_t iy = min<uint32_t>(g - 1, (uint32_t)max(0.0, (c.y - miny) * sy));
        return pair<uint32_t, uint32_t>(ix, iy);
    };
    for (uint32_t i = 0; i < n; ++i) {
        auto [ix, iy] = cell_of(p[i]);
        cells[(size_t)iy * g + ix].push_back(i);
    }

    vector<unsigned char> used(n, 0);
    vector<uint32_t> tour;
    tour.reserve(n);
    uint32_t cur = 0;
    used[cur] = 1;
    tour.push_back(cur);

    for (uint32_t step = 1; step < n; ++step) {
        auto [cx, cy] = cell_of(p[cur]);
        uint32_t best = UINT32_MAX;
        double bd = numeric_limits<double>::infinity();
        uint32_t maxr = max(g, g);

        for (uint32_t r = 0; r <= maxr && best == UINT32_MAX; ++r) {
            int x0 = max<int>(0, (int)cx - (int)r), x1 = min<int>((int)g - 1, (int)cx + (int)r);
            int y0 = max<int>(0, (int)cy - (int)r), y1 = min<int>((int)g - 1, (int)cy + (int)r);
            for (int y = y0; y <= y1; ++y) {
                for (int x = x0; x <= x1; ++x) {
                    if (x != x0 && x != x1 && y != y0 && y != y1) continue;
                    for (uint32_t idx : cells[(size_t)y * g + x]) {
                        if (!used[idx]) {
                            double d = dist2(p[cur], p[idx]);
                            if (d < bd) {
                                bd = d;
                                best = idx;
                            }
                        }
                    }
                }
            }
        }
        if (best == UINT32_MAX) {
            for (uint32_t i = 0; i < n; ++i) if (!used[i]) { best = i; break; }
        }
        cur = best;
        used[cur] = 1;
        tour.push_back(cur);
    }
    return tour;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    uint32_t n;
    if (!(cin >> n)) return 0;
    vector<City> p;
    p.reserve(n);
    for (uint32_t i = 0; i < n; ++i) {
        double x, y;
        cin >> x >> y;
        p.push_back({x, y, i});
    }
    if (n == 0) return 0;

    vector<uint32_t> tour;
    if (n <= 2500) {
        tour = nearest_neighbor_small(p);
        limited_2opt(tour, p, n <= 200 ? 8 : 3, n <= 200 ? 80 : 35);
    } else if (n <= 120000) {
        tour = grid_nearest_tour(p);
        limited_2opt(tour, p, 1, 18);
    } else {
        tour = spatial_sort_tour(p);
        if (n <= 1000000) limited_2opt(tour, p, 1, 8);
    }

    if (!tour.empty() && tour[0] != 0) {
        auto it = find(tour.begin(), tour.end(), 0);
        if (it != tour.end()) rotate(tour.begin(), it, tour.end());
    }

    for (uint32_t i = 0; i < n; ++i) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
