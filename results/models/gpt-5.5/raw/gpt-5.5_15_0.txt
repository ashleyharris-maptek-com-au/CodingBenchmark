use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Pt {
    x: f64,
    y: f64,
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn convex_hull(mut pts: Vec<Pt>) -> Vec<Pt> {
    pts.sort_by(|a, b| a.x.total_cmp(&b.x).then(a.y.total_cmp(&b.y)));
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1e-9 && (a.y - b.y).abs() < 1e-9);
    if pts.len() <= 1 {
        return pts;
    }
    let mut lower: Vec<Pt> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let n = lower.len();
            if cross(lower[n - 2], lower[n - 1], p) <= 0.0 {
                lower.pop();
            } else {
                break;
            }
        }
        lower.push(p);
    }
    let mut upper: Vec<Pt> = Vec::new();
    for &p in pts.iter().rev() {
        while upper.len() >= 2 {
            let n = upper.len();
            if cross(upper[n - 2], upper[n - 1], p) <= 0.0 {
                upper.pop();
            } else {
                break;
            }
        }
        upper.push(p);
    }
    lower.pop();
    upper.pop();
    lower.extend(upper);
    lower
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let nv = match it.next().and_then(|s| s.parse::<usize>().ok()) {
        Some(v) => v,
        None => return,
    };

    let mut pts = Vec::with_capacity(nv);
    for _ in 0..nv {
        let x = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        let y = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
        pts.push(Pt { x, y });
    }

    let sx = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sy = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(0.0);
    let sz = it.next().and_then(|s| s.parse::<f64>().ok()).unwrap_or(-1.0);
    let horiz = (sx * sx + sy * sy).sqrt();
    let vertical = sz.abs().max(1e-6);

    if pts.is_empty() {
        println!("0");
        return;
    }

    let mut min_x = pts[0].x;
    let mut max_x = pts[0].x;
    let mut min_y = pts[0].y;
    let mut max_y = pts[0].y;
    for p in &pts {
        min_x = min_x.min(p.x);
        max_x = max_x.max(p.x);
        min_y = min_y.min(p.y);
        max_y = max_y.max(p.y);
    }

    let width = (max_x - min_x).max(1e-6);
    let height = (max_y - min_y).max(1e-6);
    let span = width.max(height);
    let scale = 10.0_f64.min((span / 2.0).max(0.25));

    let hull = convex_hull(pts);
    let mut out: Vec<(f64, f64, f64, f64)> = Vec::new();

    if nv <= 64 && hull.len() <= 64 {
        for p in hull {
            out.push((p.x, p.y, 80.0 + out.len() as f64 * 25.0, scale));
        }
    } else {
        let step = (scale * 2.2).max(0.55);
        let nx = (width / step).ceil() as usize + 1;
        let ny = (height / step).ceil() as usize + 1;
        let max_cells = if nv < 1_000 { 2_500 } else if nv < 20_000 { 10_000 } else { 40_000 };
        let stride = (((nx * ny) as f64 / max_cells as f64).sqrt().ceil() as usize).max(1);
        let actual_step = step * stride as f64;
        let nx2 = (width / actual_step).ceil() as usize + 1;
        let ny2 = (height / actual_step).ceil() as usize + 1;
        for iy in 0..ny2 {
            for ix in 0..nx2 {
                let x = min_x + ix as f64 * actual_step;
                let y = min_y + iy as f64 * actual_step;
                let z = 80.0 + (ix + iy * nx2) as f64 * 25.0;
                out.push((x, y, z, scale));
            }
        }
    }

    if out.is_empty() {
        out.push(((min_x + max_x) * 0.5, (min_y + max_y) * 0.5, 80.0, scale));
    }

    let pad = scale * (3.0 + horiz / vertical);
    out.push(((min_x + max_x) * 0.5 - sx / vertical * pad, (min_y + max_y) * 0.5 - sy / vertical * pad, 80.0 + out.len() as f64 * 25.0, scale));

    println!("{}", out.len());
    for (x, y, z, s) in out {
        println!("{:.9} {:.9} {:.9} 1.0 0.0 0.0 0.0 {:.9}", x, y, z, s);
    }
}
