#include <algorithm>
#include <cstdint>
#include <iostream>
#include <limits>
#include <numeric>
#include <queue>
#include <random>
#include <utility>
#include <vector>

using namespace std;

struct Operation {
    int machine = 0;
    long long duration = 0;
};

struct Job {
    vector<Operation> ops;
};

struct ScheduledOp {
    long long start = 0;
    int machine = 0;
};

struct Candidate {
    long long key1 = 0;
    long long key2 = 0;
    int job = 0;
};

struct MinCandidate {
    bool operator()(const Candidate& a, const Candidate& b) const {
        if (a.key1 != b.key1) return a.key1 > b.key1;
        if (a.key2 != b.key2) return a.key2 > b.key2;
        return a.job > b.job;
    }
};

enum class Rule {
    EarliestReady,
    ShortestProcessing,
    LongestProcessing,
    MostWorkRemaining,
    LeastWorkRemaining
};

static long long ceil_div(long long a, long long b) {
    return b == 0 ? 0 : (a + b - 1) / b;
}

static long long machine_load_bound(const vector<Job>& jobs, int machines) {
    vector<long long> load(machines, 0);
    long long total = 0;
    for (const auto& job : jobs) {
        for (const auto& op : job.ops) {
            if (0 <= op.machine && op.machine < machines) load[op.machine] += op.duration;
            total += op.duration;
        }
    }
    long long best = machines ? ceil_div(total, machines) : 0;
    for (long long x : load) best = max(best, x);
    return best;
}

static long long job_load_bound(const vector<Job>& jobs) {
    long long best = 0;
    for (const auto& job : jobs) {
        long long s = 0;
        for (const auto& op : job.ops) s += op.duration;
        best = max(best, s);
    }
    return best;
}

static long long makespan_of(const vector<vector<ScheduledOp>>& sched,
                             const vector<Job>& jobs) {
    long long ms = 0;
    for (size_t j = 0; j < jobs.size(); ++j) {
        for (size_t k = 0; k < jobs[j].ops.size(); ++k) {
            ms = max(ms, sched[j][k].start + jobs[j].ops[k].duration);
        }
    }
    return ms;
}

static vector<vector<ScheduledOp>> list_schedule(const vector<Job>& jobs, int machines, Rule rule) {
    const int J = (int)jobs.size();
    vector<vector<ScheduledOp>> sched(J);
    vector<int> next_op(J, 0);
    vector<long long> job_ready(J, 0), machine_ready(max(0, machines), 0);
    vector<long long> remaining(J, 0);
    priority_queue<Candidate, vector<Candidate>, MinCandidate> pq;

    for (int j = 0; j < J; ++j) {
        sched[j].resize(jobs[j].ops.size());
        for (const auto& op : jobs[j].ops) remaining[j] += op.duration;
        if (!jobs[j].ops.empty()) {
            long long p = jobs[j].ops[0].duration;
            long long k1 = 0, k2 = 0;
            switch (rule) {
                case Rule::EarliestReady:      k1 = 0; k2 = j; break;
                case Rule::ShortestProcessing: k1 = p; k2 = j; break;
                case Rule::LongestProcessing:  k1 = -p; k2 = j; break;
                case Rule::MostWorkRemaining:  k1 = -remaining[j]; k2 = j; break;
                case Rule::LeastWorkRemaining: k1 = remaining[j]; k2 = j; break;
            }
            pq.push({k1, k2, j});
        }
    }

    while (!pq.empty()) {
        int j = pq.top().job;
        pq.pop();
        int k = next_op[j];
        if (k >= (int)jobs[j].ops.size()) continue;

        const Operation& op = jobs[j].ops[k];
        long long start = max(job_ready[j], machine_ready[op.machine]);
        sched[j][k] = {start, op.machine};
        long long finish = start + op.duration;
        job_ready[j] = finish;
        machine_ready[op.machine] = finish;
        remaining[j] -= op.duration;
        ++next_op[j];

        if (next_op[j] < (int)jobs[j].ops.size()) {
            const Operation& nx = jobs[j].ops[next_op[j]];
            long long ready = job_ready[j];
            long long p = nx.duration;
            long long k1 = ready, k2 = j;
            switch (rule) {
                case Rule::EarliestReady:      k1 = ready; k2 = j; break;
                case Rule::ShortestProcessing: k1 = ready * 4 + min(p, (long long)3); k2 = p; break;
                case Rule::LongestProcessing:  k1 = ready * 4 - min(p, (long long)3); k2 = -p; break;
                case Rule::MostWorkRemaining:  k1 = ready * 4 - min(remaining[j], (long long)3); k2 = -remaining[j]; break;
                case Rule::LeastWorkRemaining: k1 = ready * 4 + min(remaining[j], (long long)3); k2 = remaining[j]; break;
            }
            pq.push({k1, k2, j});
        }
    }
    return sched;
}

static vector<vector<ScheduledOp>> insertion_schedule(const vector<Job>& jobs, int machines, Rule rule) {
    const int J = (int)jobs.size();
    vector<vector<ScheduledOp>> sched(J);
    vector<vector<pair<long long, long long>>> intervals(max(0, machines));
    vector<int> next_op(J, 0);
    vector<long long> job_ready(J, 0), remaining(J, 0);
    priority_queue<Candidate, vector<Candidate>, MinCandidate> pq;

    for (int j = 0; j < J; ++j) {
        sched[j].resize(jobs[j].ops.size());
        for (const auto& op : jobs[j].ops) remaining[j] += op.duration;
        if (!jobs[j].ops.empty()) {
            long long p = jobs[j].ops[0].duration;
            long long primary = (rule == Rule::MostWorkRemaining) ? -remaining[j] :
                                (rule == Rule::LongestProcessing) ? -p :
                                (rule == Rule::ShortestProcessing) ? p : 0;
            pq.push({primary, j, j});
        }
    }

    while (!pq.empty()) {
        int j = pq.top().job;
        pq.pop();
        int k = next_op[j];
        if (k >= (int)jobs[j].ops.size()) continue;

        const Operation& op = jobs[j].ops[k];
        auto& v = intervals[op.machine];
        long long start = job_ready[j];
        size_t pos = 0;
        while (pos < v.size()) {
            if (start + op.duration <= v[pos].first) break;
            start = max(start, v[pos].second);
            ++pos;
        }
        v.insert(v.begin() + (ptrdiff_t)pos, {start, start + op.duration});
        sched[j][k] = {start, op.machine};
        job_ready[j] = start + op.duration;
        remaining[j] -= op.duration;
        ++next_op[j];

        if (next_op[j] < (int)jobs[j].ops.size()) {
            const Operation& nx = jobs[j].ops[next_op[j]];
            long long ready = job_ready[j];
            long long p = nx.duration;
            long long primary = ready;
            long long secondary = j;
            switch (rule) {
                case Rule::EarliestReady:      break;
                case Rule::ShortestProcessing: secondary = p; break;
                case Rule::LongestProcessing:  secondary = -p; break;
                case Rule::MostWorkRemaining:  secondary = -remaining[j]; break;
                case Rule::LeastWorkRemaining: secondary = remaining[j]; break;
            }
            pq.push({primary, secondary, j});
        }
    }
    return sched;
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int J, M;
    if (!(cin >> J >> M)) return 0;

    vector<Job> jobs(J);
    long long task_count = 0;
    for (int j = 0; j < J; ++j) {
        int T;
        cin >> T;
        jobs[j].ops.resize(T);
        task_count += T;
        for (int k = 0; k < T; ++k) {
            cin >> jobs[j].ops[k].machine >> jobs[j].ops[k].duration;
            if (jobs[j].ops[k].machine < 0) jobs[j].ops[k].machine = 0;
            if (M > 0 && jobs[j].ops[k].machine >= M) jobs[j].ops[k].machine %= M;
            if (jobs[j].ops[k].duration < 0) jobs[j].ops[k].duration = 0;
        }
    }

    vector<Rule> rules;
    if (task_count <= 5000) {
        rules = {Rule::MostWorkRemaining, Rule::LongestProcessing, Rule::EarliestReady,
                 Rule::ShortestProcessing, Rule::LeastWorkRemaining};
    } else if (task_count <= 50000) {
        rules = {Rule::MostWorkRemaining, Rule::LongestProcessing, Rule::EarliestReady};
    } else {
        rules = {Rule::MostWorkRemaining, Rule::EarliestReady};
    }

    vector<vector<ScheduledOp>> best;
    long long best_ms = numeric_limits<long long>::max();

    for (Rule r : rules) {
        auto sched = list_schedule(jobs, M, r);
        long long ms = makespan_of(sched, jobs);
        if (ms < best_ms) {
            best_ms = ms;
            best = move(sched);
        }
    }

    if (task_count <= 20000) {
        for (Rule r : rules) {
            auto sched = insertion_schedule(jobs, M, r);
            long long ms = makespan_of(sched, jobs);
            if (ms < best_ms) {
                best_ms = ms;
                best = move(sched);
            }
        }
    }

    (void)machine_load_bound;
    (void)job_load_bound;

    for (int j = 0; j < J; ++j) {
        for (size_t k = 0; k < jobs[j].ops.size(); ++k) {
            if (k) cout << ' ';
            cout << best[j][k].start << ' ' << best[j][k].machine;
        }
        cout << '\n';
    }
    return 0;
}
