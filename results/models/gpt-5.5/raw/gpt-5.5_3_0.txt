use std::cmp::Reverse;
use std::collections::VecDeque;
use std::f64::consts::PI;
use std::io::{self, Read};

#[derive(Clone, Copy)]
struct Edge {
    u: usize,
    v: usize,
}

struct Dsu {
    p: Vec<usize>,
    sz: Vec<usize>,
}

impl Dsu {
    fn new(n: usize) -> Self {
        let mut p = Vec::with_capacity(n);
        for i in 0..n {
            p.push(i);
        }
        Self { p, sz: vec![1; n] }
    }
    fn find(&mut self, x: usize) -> usize {
        let mut r = x;
        while self.p[r] != r {
            r = self.p[r];
        }
        let mut y = x;
        while self.p[y] != y {
            let z = self.p[y];
            self.p[y] = r;
            y = z;
        }
        r
    }
    fn union(&mut self, a: usize, b: usize) {
        let mut a = self.find(a);
        let mut b = self.find(b);
        if a == b {
            return;
        }
        if self.sz[a] < self.sz[b] {
            std::mem::swap(&mut a, &mut b);
        }
        self.p[b] = a;
        self.sz[a] += self.sz[b];
    }
}

fn next_u64(seed: &mut u64) -> u64 {
    *seed = seed.wrapping_mul(6364136223846793005).wrapping_add(1442695040888963407);
    let mut x = *seed;
    x ^= x >> 30;
    x = x.wrapping_mul(0xbf58476d1ce4e5b9);
    x ^= x >> 27;
    x = x.wrapping_mul(0x94d049bb133111eb);
    x ^ (x >> 31)
}

fn barycentric_smooth(order: &mut [usize], pos: &mut [usize], adj: &[Vec<usize>], rounds: usize) {
    let n = order.len();
    if n <= 2 {
        return;
    }
    let mut key = vec![0.0f64; pos.len()];
    for r in 0..rounds {
        for &v in order.iter() {
            if adj[v].is_empty() {
                key[v] = pos[v] as f64;
            } else {
                let mut s = 0usize;
                for &to in &adj[v] {
                    s += pos[to];
                }
                key[v] = (s as f64) / (adj[v].len() as f64);
            }
        }
        if r & 1 == 0 {
            order.sort_unstable_by(|&a, &b| {
                key[a].partial_cmp(&key[b]).unwrap()
                    .then_with(|| adj[b].len().cmp(&adj[a].len()))
                    .then_with(|| a.cmp(&b))
            });
        } else {
            order.sort_unstable_by(|&a, &b| {
                key[b].partial_cmp(&key[a]).unwrap()
                    .then_with(|| adj[b].len().cmp(&adj[a].len()))
                    .then_with(|| a.cmp(&b))
            });
        }
        for (i, &v) in order.iter().enumerate() {
            pos[v] = i;
        }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let n: usize = match it.next() {
        Some(s) => s.parse().unwrap_or(0),
        None => return,
    };
    let m: usize = it.next().unwrap_or("0").parse().unwrap_or(0);

    let mut edges = Vec::with_capacity(m.min(4_000_000));
    let mut degree = vec![0usize; n];
    let mut dsu = Dsu::new(n);
    for _ in 0..m {
        let u: usize = match it.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        let v: usize = match it.next() {
            Some(s) => s.parse().unwrap_or(0),
            None => break,
        };
        if u >= n || v >= n || u == v {
            continue;
        }
        edges.push(Edge { u, v });
        degree[u] += 1;
        degree[v] += 1;
        dsu.union(u, v);
    }

    if n == 0 {
        return;
    }

    let build_adj = n <= 120_000 && edges.len() <= 1_500_000;
    let mut adj = vec![Vec::<usize>::new(); if build_adj { n } else { 0 }];
    if build_adj {
        for e in &edges {
            adj[e.u].push(e.v);
            adj[e.v].push(e.u);
        }
        for a in &mut adj {
            if a.len() > 96 {
                a.sort_unstable_by_key(|&v| Reverse(degree[v]));
                a.truncate(96);
            }
        }
    }

    let mut roots = vec![0usize; n];
    for i in 0..n {
        roots[i] = dsu.find(i);
    }
    let mut verts: Vec<usize> = (0..n).collect();
    verts.sort_unstable_by_key(|&v| (roots[v], Reverse(degree[v]), v));

    let mut components: Vec<Vec<usize>> = Vec::new();
    for v in verts {
        if components.last().map_or(true, |c| roots[c[0]] != roots[v]) {
            components.push(Vec::new());
        }
        components.last_mut().unwrap().push(v);
    }
    components.sort_unstable_by_key(|c| Reverse(c.len()));

    let mut xy = vec![(0.0f64, 0.0f64); n];
    let mut comp_boxes = Vec::with_capacity(components.len());

    for comp in components.iter() {
        let k = comp.len();
        let mut order = Vec::with_capacity(k);
        if build_adj && k <= 20_000 {
            let mut in_comp = vec![false; n];
            for &v in comp {
                in_comp[v] = true;
            }
            let mut seen = vec![false; n];
            let mut starts = comp.clone();
            starts.sort_unstable_by_key(|&v| (Reverse(degree[v]), v));
            let mut q = VecDeque::new();
            for s in starts {
                if seen[s] {
                    continue;
                }
                seen[s] = true;
                q.push_back(s);
                while let Some(v) = q.pop_front() {
                    order.push(v);
                    let mut ns = adj[v].clone();
                    ns.sort_unstable_by_key(|&to| (Reverse(degree[to]), to));
                    for to in ns {
                        if in_comp[to] && !seen[to] {
                            seen[to] = true;
                            q.push_back(to);
                        }
                    }
                }
            }
        } else {
            order = comp.clone();
            order.sort_unstable_by_key(|&v| (Reverse(degree[v]), v));
        }

        let mut pos = vec![0usize; n];
        for (i, &v) in order.iter().enumerate() {
            pos[v] = i;
        }
        if build_adj && k <= 6000 {
            barycentric_smooth(&mut order, &mut pos, &adj, if k <= 200 { 8 } else { 4 });
        }

        let radius = (k as f64).sqrt().max(2.0) * 18.0;
        let spiral_gap = 8.0;
        let golden = PI * (3.0 - 5.0f64.sqrt());
        for (i, &v) in order.iter().enumerate() {
            let t = i as f64;
            let mut r = if k <= 64 {
                radius
            } else {
                10.0 + spiral_gap * t.sqrt()
            };
            let mut a = if k <= 64 {
                2.0 * PI * t / (k as f64)
            } else {
                t * golden
            };
            if degree[v] > 0 {
                r *= 1.0 + 0.035 * ((degree[v] as f64).ln_1p());
                a += 0.000_001 * (v as f64);
            }
            xy[v] = (r * a.cos(), r * a.sin());
        }

        if build_adj && k <= 1000 {
            let iterations = if k <= 60 { 160 } else if k <= 250 { 80 } else { 35 };
            let area = (k as f64).max(10.0) * 4000.0;
            let kk = (area / k as f64).sqrt();
            let mut temp = radius * 0.25;
            let mut disp = vec![(0.0f64, 0.0f64); n];
            for _ in 0..iterations {
                for &v in comp {
                    disp[v] = (0.0, 0.0);
                }
                for i in 0..k {
                    let v = comp[i];
                    for j in (i + 1)..k {
                        let u = comp[j];
                        let dx = xy[v].0 - xy[u].0;
                        let dy = xy[v].1 - xy[u].1;
                        let d2 = dx * dx + dy * dy + 0.01;
                        let d = d2.sqrt();
                        let f = kk * kk / d;
                        let fx = dx / d * f;
                        let fy = dy / d * f;
                        disp[v].0 += fx;
                        disp[v].1 += fy;
                        disp[u].0 -= fx;
                        disp[u].1 -= fy;
                    }
                }
                for e in &edges {
                    if roots[e.u] != roots[comp[0]] {
                        continue;
                    }
                    let dx = xy[e.u].0 - xy[e.v].0;
                    let dy = xy[e.u].1 - xy[e.v].1;
                    let d = (dx * dx + dy * dy + 0.01).sqrt();
                    let f = d * d / kk;
                    let fx = dx / d * f;
                    let fy = dy / d * f;
                    disp[e.u].0 -= fx;
                    disp[e.u].1 -= fy;
                    disp[e.v].0 += fx;
                    disp[e.v].1 += fy;
                }
                for &v in comp {
                    let (dx, dy) = disp[v];
                    let d = (dx * dx + dy * dy).sqrt();
                    if d > 1e-9 {
                        xy[v].0 += dx / d * d.min(temp);
                        xy[v].1 += dy / d * d.min(temp);
                    }
                }
                temp *= 0.94;
            }
        }

        let mut minx = f64::INFINITY;
        let mut maxx = f64::NEG_INFINITY;
        let mut miny = f64::INFINITY;
        let mut maxy = f64::NEG_INFINITY;
        for &v in comp {
            minx = minx.min(xy[v].0);
            maxx = maxx.max(xy[v].0);
            miny = miny.min(xy[v].1);
            maxy = maxy.max(xy[v].1);
        }
        comp_boxes.push((comp[0], minx, maxx, miny, maxy));
    }

    let total_area: f64 = comp_boxes
        .iter()
        .map(|&(_, minx, maxx, miny, maxy)| (maxx - minx + 80.0) * (maxy - miny + 80.0))
        .sum();
    let shelf_width = total_area.sqrt().max(400.0) * 1.5;
    let mut cursor_x = 0.0;
    let mut cursor_y = 0.0;
    let mut row_h = 0.0;
    for (ci, comp) in components.iter().enumerate() {
        let (_, minx, maxx, miny, maxy) = comp_boxes[ci];
        let w = maxx - minx + 100.0;
        let h = maxy - miny + 100.0;
        if cursor_x > 0.0 && cursor_x + w > shelf_width {
            cursor_x = 0.0;
            cursor_y += row_h;
            row_h = 0.0;
        }
        for &v in comp {
            xy[v].0 += cursor_x - minx + 50.0;
            xy[v].1 += cursor_y - miny + 50.0;
        }
        cursor_x += w;
        row_h = row_h.max(h);
    }

    let mut seed = 0x1234_5678_9abc_def0u64 ^ (n as u64).wrapping_mul(1_000_003) ^ edges.len() as u64;
    for v in 0..n {
        if !xy[v].0.is_finite() || !xy[v].1.is_finite() {
            let a = (next_u64(&mut seed) as f64 / u64::MAX as f64) * 2.0 * PI;
            let r = ((v + 1) as f64).sqrt() * 20.0;
            xy[v] = (r * a.cos(), r * a.sin());
        }
    }

    let mut out = String::with_capacity(n * 28);
    for (x, y) in xy {
        out.push_str(&format!("{:.6} {:.6}\n", x, y));
    }
    print!("{}", out);
}
