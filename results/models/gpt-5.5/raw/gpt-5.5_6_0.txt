using System;
using System.Collections.Generic;
using System.Globalization;
using System.Linq;

public static class Program
{
    struct Vec
    {
        public double X, Y, Z;
        public Vec(double x, double y, double z) { X = x; Y = y; Z = z; }
        public static Vec operator +(Vec a, Vec b) => new Vec(a.X + b.X, a.Y + b.Y, a.Z + b.Z);
        public static Vec operator -(Vec a, Vec b) => new Vec(a.X - b.X, a.Y - b.Y, a.Z - b.Z);
        public static Vec operator *(Vec a, double s) => new Vec(a.X * s, a.Y * s, a.Z * s);
        public double Norm() => Math.Sqrt(X * X + Y * Y + Z * Z);
        public static double Dot(Vec a, Vec b) => a.X * b.X + a.Y * b.Y + a.Z * b.Z;
        public static Vec Cross(Vec a, Vec b) => new Vec(a.Y * b.Z - a.Z * b.Y, a.Z * b.X - a.X * b.Z, a.X * b.Y - a.Y * b.X);
    }

    sealed class Orbit
    {
        public Vec R0, V0, H;
        public double R, V, Energy, A, Period, Phase, Radius, Speed, Inclination;
        public bool Circular;

        public Orbit(Vec r, Vec v, double mu)
        {
            R0 = r; V0 = v; R = r.Norm(); V = v.Norm();
            H = Vec.Cross(r, v);
            Energy = 0.5 * V * V - mu / Math.Max(R, 1e-9);
            A = Energy < -1e-9 ? -mu / (2.0 * Energy) : R;
            Period = A > 0 ? 2.0 * Math.PI * Math.Sqrt(A * A * A / mu) : 5400.0;
            Radius = R;
            Speed = V;
            Inclination = Math.Acos(Math.Max(-1.0, Math.Min(1.0, H.Z / Math.Max(H.Norm(), 1e-9))));
            double tangential = H.Norm() / Math.Max(R, 1e-9);
            Circular = Math.Abs(tangential - V) < 0.15 && Math.Abs(R - A) / Math.Max(R, 1.0) < 0.08;
            Phase = Math.Atan2(r.Y, r.X);
        }

        public Vec PositionAt(double t)
        {
            if (!Circular) return R0;
            double n = 2.0 * Math.PI / Math.Max(Period, 1.0);
            double ang = Phase + n * t;
            double c = Math.Cos(ang), s = Math.Sin(ang);
            double rho = Math.Sqrt(R0.X * R0.X + R0.Y * R0.Y);
            return new Vec(rho * c, rho * s, R0.Z);
        }
    }

    static double Mu;
    static Orbit Start = null!;
    static Orbit[] S = null!;
    static double[,] Cost = null!;
    static double[] StartCost = null!;

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        Mu = fs.NextDouble();
        if (n <= 0) { Console.WriteLine(); return; }

        Start = new Orbit(ReadVec(fs), ReadVec(fs), Mu);
        S = new Orbit[n];
        for (int i = 0; i < n; i++) S[i] = new Orbit(ReadVec(fs), ReadVec(fs), Mu);

        BuildCostMatrix(n);
        int[] route = n <= 10 ? ExactHeldKarp(n) : HeuristicRoute(n);
        Console.WriteLine(string.Join(" ", route));
    }

    static Vec ReadVec(FastScanner fs) => new Vec(fs.NextDouble(), fs.NextDouble(), fs.NextDouble());

    static void BuildCostMatrix(int n)
    {
        Cost = new double[n, n];
        StartCost = new double[n];
        for (int i = 0; i < n; i++)
        {
            StartCost[i] = TransferCost(Start, S[i]);
            for (int j = 0; j < n; j++) Cost[i, j] = i == j ? 0.0 : TransferCost(S[i], S[j]);
        }
    }

    static double TransferCost(Orbit a, Orbit b)
    {
        double dr = Math.Abs(a.Radius - b.Radius);
        double dvSpeed = Math.Abs(a.Speed - b.Speed);
        double plane = 2.0 * Math.Min(a.Speed, b.Speed) * Math.Sin(0.5 * Math.Abs(a.Inclination - b.Inclination));

        double dotH = Vec.Dot(a.H, b.H) / Math.Max(a.H.Norm() * b.H.Norm(), 1e-9);
        double planeFull = 2.0 * Math.Min(a.Speed, b.Speed) * Math.Sin(0.5 * Math.Acos(Math.Max(-1.0, Math.Min(1.0, dotH))));

        double phase = 0.00025 * (a.R0 - b.R0).Norm();
        double hohmann = 0.0;
        if (a.Radius > 1.0 && b.Radius > 1.0)
        {
            double r1 = a.Radius, r2 = b.Radius;
            double va = Math.Sqrt(Mu / r1), vb = Math.Sqrt(Mu / r2);
            double at = 0.5 * (r1 + r2);
            hohmann = Math.Abs(Math.Sqrt(Mu * (2 / r1 - 1 / at)) - va)
                    + Math.Abs(vb - Math.Sqrt(Mu * (2 / r2 - 1 / at)));
        }

        return hohmann + 0.35 * dvSpeed + 0.65 * planeFull + 0.20 * plane + phase + dr * 1e-5;
    }

    static int[] ExactHeldKarp(int n)
    {
        int masks = 1 << n;
        var dp = new double[masks, n];
        var prev = new int[masks, n];
        for (int m = 0; m < masks; m++)
            for (int i = 0; i < n; i++) { dp[m, i] = double.PositiveInfinity; prev[m, i] = -1; }

        for (int i = 0; i < n; i++) dp[1 << i, i] = StartCost[i];

        for (int mask = 1; mask < masks; mask++)
        {
            for (int last = 0; last < n; last++)
            {
                double cur = dp[mask, last];
                if (double.IsPositiveInfinity(cur)) continue;
                int rem = ((masks - 1) ^ mask);
                while (rem != 0)
                {
                    int bit = rem & -rem;
                    int nxt = BitIndex(bit);
                    double v = cur + Cost[last, nxt];
                    int nm = mask | bit;
                    if (v < dp[nm, nxt]) { dp[nm, nxt] = v; prev[nm, nxt] = last; }
                    rem -= bit;
                }
            }
        }

        int full = masks - 1, best = 0;
        for (int i = 1; i < n; i++) if (dp[full, i] < dp[full, best]) best = i;

        int[] route = new int[n];
        int mask2 = full, at = best;
        for (int p = n - 1; p >= 0; p--)
        {
            route[p] = at;
            int old = at;
            at = prev[mask2, at];
            mask2 ^= 1 << old;
        }
        return route;
    }

    static int BitIndex(int bit)
    {
        int i = 0;
        while ((bit >>= 1) != 0) i++;
        return i;
    }

    static int[] HeuristicRoute(int n)
    {
        int starts = n <= 25 ? Math.Min(n, 8) : Math.Min(n, 4);
        var candidates = Enumerable.Range(0, n).OrderBy(i => StartCost[i]).Take(starts).ToArray();
        int[] bestRoute = Array.Empty<int>();
        double bestScore = double.PositiveInfinity;

        foreach (int seed in candidates)
        {
            int[] r = NearestInsertion(n, seed);
            TwoOpt(r, n <= 25 ? 200000 : n <= 60 ? 120000 : 70000);
            RelocateImprove(r, n <= 25 ? 6 : 3);
            double sc = Score(r);
            if (sc < bestScore) { bestScore = sc; bestRoute = r; }
        }

        return bestRoute;
    }

    static int[] NearestInsertion(int n, int seed)
    {
        var used = new bool[n];
        var route = new List<int>(n) { seed };
        used[seed] = true;

        while (route.Count < n)
        {
            int bestNode = -1, bestPos = 0;
            double bestDelta = double.PositiveInfinity;
            for (int k = 0; k < n; k++)
            {
                if (used[k]) continue;
                for (int pos = 0; pos <= route.Count; pos++)
                {
                    double add;
                    if (pos == 0) add = StartCost[k] + Cost[k, route[0]] - StartCost[route[0]];
                    else if (pos == route.Count) add = Cost[route[pos - 1], k];
                    else add = Cost[route[pos - 1], k] + Cost[k, route[pos]] - Cost[route[pos - 1], route[pos]];
                    if (add < bestDelta) { bestDelta = add; bestNode = k; bestPos = pos; }
                }
            }
            route.Insert(bestPos, bestNode);
            used[bestNode] = true;
        }
        return route.ToArray();
    }

    static void TwoOpt(int[] r, int limit)
    {
        int n = r.Length, tries = 0;
        bool improved = true;
        while (improved && tries < limit)
        {
            improved = false;
            for (int i = 0; i < n - 2 && tries < limit; i++)
            {
                int a = i == 0 ? -1 : r[i - 1], b = r[i];
                for (int k = i + 1; k < n - 1 && tries < limit; k++, tries++)
                {
                    int c = r[k], d = r[k + 1];
                    double before = (a < 0 ? StartCost[b] : Cost[a, b]) + Cost[c, d];
                    double after = (a < 0 ? StartCost[c] : Cost[a, c]) + Cost[b, d];
                    if (after + 1e-10 < before)
                    {
                        Array.Reverse(r, i, k - i + 1);
                        improved = true;
                    }
                }
            }
        }
    }

    static void RelocateImprove(int[] r, int passes)
    {
        int n = r.Length;
        for (int pass = 0; pass < passes; pass++)
        {
            bool moved = false;
            for (int i = 0; i < n; i++)
            {
                int node = r[i];
                double remove = EdgeBefore(r, i, node) + EdgeAfter(r, i, node) - BridgeWithout(r, i);
                int bestJ = -1;
                double bestGain = 1e-10;
                for (int j = 0; j <= n; j++)
                {
                    if (j == i || j == i + 1) continue;
                    double add = InsertDelta(r, i, j, node);
                    double gain = remove - add;
                    if (gain > bestGain) { bestGain = gain; bestJ = j; }
                }
                if (bestJ >= 0)
                {
                    var list = r.ToList();
                    list.RemoveAt(i);
                    if (bestJ > i) bestJ--;
                    list.Insert(bestJ, node);
                    for (int t = 0; t < n; t++) r[t] = list[t];
                    moved = true;
                }
            }
            if (!moved) break;
        }
    }

    static double EdgeBefore(int[] r, int i, int node) => i == 0 ? StartCost[node] : Cost[r[i - 1], node];
    static double EdgeAfter(int[] r, int i, int node) => i == r.Length - 1 ? 0.0 : Cost[node, r[i + 1]];
    static double BridgeWithout(int[] r, int i) => i == r.Length - 1 ? 0.0 : (i == 0 ? StartCost[r[i + 1]] : Cost[r[i - 1], r[i + 1]]);

    static double InsertDelta(int[] r, int oldIndex, int pos, int node)
    {
        int n = r.Length;
        int left = pos - 1, right = pos;
        if (left == oldIndex) left--;
        if (right == oldIndex) right++;
        if (right >= n) right = -1;

        double oldEdge = right < 0 ? 0.0 : (left < 0 ? StartCost[r[right]] : Cost[r[left], r[right]]);
        double newEdges = (left < 0 ? StartCost[node] : Cost[r[left], node]) + (right < 0 ? 0.0 : Cost[node, r[right]]);
        return newEdges - oldEdge;
    }

    static double Score(int[] r)
    {
        if (r.Length == 0) return 0;
        double s = StartCost[r[0]];
        for (int i = 1; i < r.Length; i++) s += Cost[r[i - 1], r[i]];
        return s;
    }

    sealed class FastScanner
    {
        readonly string[] data = Console.In.ReadToEnd().Split((char[]?)null, StringSplitOptions.RemoveEmptyEntries);
        int idx;
        public int NextInt() => int.Parse(data[idx++], CultureInfo.InvariantCulture);
        public double NextDouble() => double.Parse(data[idx++], CultureInfo.InvariantCulture);
    }
}
