using System;
using System.Collections.Generic;
using System.Text;

public static class Program
{
    struct Clue
    {
        public int Need;
        public int[] Vars;
    }

    sealed class DSU
    {
        readonly int[] p, r;
        public DSU(int n) { p = new int[n]; r = new int[n]; for (int i = 0; i < n; i++) p[i] = i; }
        public int Find(int x) { while (p[x] != x) { p[x] = p[p[x]]; x = p[x]; } return x; }
        public void Union(int a, int b)
        {
            a = Find(a); b = Find(b); if (a == b) return;
            if (r[a] < r[b]) { int t = a; a = b; b = t; }
            p[b] = a; if (r[a] == r[b]) r[a]++;
        }
    }

    static int H, W;
    static char[][] board = Array.Empty<char[]>();
    static readonly int[] dr = { -1, -1, -1, 0, 0, 1, 1, 1 };
    static readonly int[] dc = { -1, 0, 1, -1, 1, -1, 0, 1 };

    static void Main()
    {
        string? first = Console.ReadLine();
        if (string.IsNullOrWhiteSpace(first)) { Console.WriteLine(0); return; }
        var hw = first.Split(new[] { ' ', '\t' }, StringSplitOptions.RemoveEmptyEntries);
        H = int.Parse(hw[0]); W = int.Parse(hw[1]);
        board = new char[H][];
        for (int r = 0; r < H; r++)
        {
            string? line = Console.ReadLine();
            line ??= "";
            if (line.Length < W) line = line.PadRight(W);
            board[r] = line.ToCharArray();
        }

        bool[,] safe = new bool[H, W];
        var clues = BuildCluesAndApplyLocalRules(safe, out int frontierCount);

        // Massive boards are usually sparse at the revealed frontier.  Run exact
        // reasoning only on connected frontier components small enough to finish.
        if (frontierCount > 0 && clues.Count > 0)
            RunAdaptiveCsp(clues, frontierCount, safe);

        var output = new StringBuilder();
        int count = 0;
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (safe[r, c]) count++;
        output.AppendLine(count.ToString());
        for (int r = 0; r < H; r++)
            for (int c = 0; c < W; c++)
                if (safe[r, c])
                    output.Append(c).Append(' ').Append(r).AppendLine();
        Console.Write(output.ToString());
    }

    static List<Clue> BuildCluesAndApplyLocalRules(bool[,] safe, out int frontierCount)
    {
        var varId = new Dictionary<long, int>(1024);
        var cells = new List<(int r, int c)>(1024);
        var clues = new List<Clue>();
        frontierCount = 0;

        for (int r = 0; r < H; r++)
        {
            for (int c = 0; c < W; c++)
            {
                char ch = board[r][c];
                if (ch < '0' || ch > '8') continue;
                int need = ch - '0';
                var vars = new List<int>(8);
                for (int k = 0; k < 8; k++)
                {
                    int nr = r + dr[k], nc = c + dc[k];
                    if ((uint)nr >= (uint)H || (uint)nc >= (uint)W) continue;
                    if (board[nr][nc] == '*') need--;
                    else if (board[nr][nc] == ' ')
                    {
                        long key = ((long)nr << 32) | (uint)nc;
                        if (!varId.TryGetValue(key, out int id))
                        {
                            id = frontierCount++;
                            varId[key] = id;
                            cells.Add((nr, nc));
                        }
                        vars.Add(id);
                    }
                }

                if (vars.Count == 0) continue;
                if (need <= 0)
                {
                    foreach (int id in vars)
                    {
                        var (rr, cc) = cells[id];
                        safe[rr, cc] = true;
                    }
                }
                else if (need < vars.Count)
                {
                    clues.Add(new Clue { Need = need, Vars = vars.ToArray() });
                }
            }
        }

        CellOfId = cells.ToArray();

        return clues;
    }

    static (int r, int c)[] CellOfId = Array.Empty<(int, int)>();

    static void RunAdaptiveCsp(List<Clue> clues, int varCount, bool[,] safe)
    {
        var dsu = new DSU(varCount);
        foreach (var cl in clues)
            for (int i = 1; i < cl.Vars.Length; i++)
                dsu.Union(cl.Vars[0], cl.Vars[i]);

        var compVars = new Dictionary<int, List<int>>();
        for (int i = 0; i < varCount; i++)
        {
            int root = dsu.Find(i);
            if (!compVars.TryGetValue(root, out var list)) compVars[root] = list = new List<int>();
            list.Add(i);
        }

        var compClues = new Dictionary<int, List<Clue>>();
        foreach (var cl in clues)
        {
            int root = dsu.Find(cl.Vars[0]);
            if (!compClues.TryGetValue(root, out var list)) compClues[root] = list = new List<Clue>();
            list.Add(cl);
        }

        foreach (var kv in compVars)
        {
            if (!compClues.TryGetValue(kv.Key, out var cls)) continue;
            var vars = kv.Value;
            if (vars.Count <= 24 || (vars.Count <= 64 && cls.Count <= 80))
                ExactComponent(vars, cls, safe);
            else
                SubsetRules(cls, safe);
        }
    }

    static void ExactComponent(List<int> vars, List<Clue> clues, bool[,] safe)
    {
        int n = vars.Count;
        var local = new Dictionary<int, int>(n);
        for (int i = 0; i < n; i++) local[vars[i]] = i;

        int m = clues.Count;
        int[][] clueVars = new int[m][];
        int[] need = new int[m], left = new int[m];
        var varToClues = new List<int>[n];
        for (int i = 0; i < n; i++) varToClues[i] = new List<int>();
        for (int i = 0; i < m; i++)
        {
            need[i] = clues[i].Need;
            clueVars[i] = Array.ConvertAll(clues[i].Vars, v => local[v]);
            left[i] = clueVars[i].Length;
            foreach (int v in clueVars[i]) varToClues[v].Add(i);
        }

        int solutions = 0;
        int[] mineInSolutions = new int[n];
        bool[] assign = new bool[n];

        void Dfs(int idx)
        {
            if (solutions > 200000) return; // adaptive cap for pathological cases
            if (idx == n)
            {
                for (int i = 0; i < m; i++) if (need[i] != 0) return;
                solutions++;
                for (int i = 0; i < n; i++) if (assign[i]) mineInSolutions[i]++;
                return;
            }

            foreach (bool val in new[] { false, true })
            {
                bool ok = true;
                assign[idx] = val;
                foreach (int ci in varToClues[idx])
                {
                    left[ci]--;
                    if (val) need[ci]--;
                    if (need[ci] < 0 || need[ci] > left[ci]) ok = false;
                }
                if (ok) Dfs(idx + 1);
                foreach (int ci in varToClues[idx])
                {
                    if (val) need[ci]++;
                    left[ci]++;
                }
            }
        }

        Dfs(0);
        if (solutions == 0 || solutions > 200000) return;
        for (int i = 0; i < n; i++)
        {
            if (mineInSolutions[i] == 0)
            {
                var cell = CellOfId[vars[i]];
                if (board[cell.r][cell.c] == ' ') safe[cell.r, cell.c] = true;
            }
        }
    }

    static void SubsetRules(List<Clue> clues, bool[,] safe)
    {
        for (int i = 0; i < clues.Count; i++)
        {
            var a = clues[i];
            var setA = new HashSet<int>(a.Vars);
            for (int j = 0; j < clues.Count; j++)
            {
                if (i == j) continue;
                var b = clues[j];
                if (a.Vars.Length > b.Vars.Length) continue;
                bool subset = true;
                foreach (int v in a.Vars) if (Array.IndexOf(b.Vars, v) < 0) { subset = false; break; }
                if (!subset) continue;
                int diffNeed = b.Need - a.Need;
                int diffCount = b.Vars.Length - a.Vars.Length;
                if (diffNeed == 0)
                {
                    foreach (int v in b.Vars)
                        if (!setA.Contains(v))
                        {
                            var cell = CellOfId[v];
                            if (board[cell.r][cell.c] == ' ') safe[cell.r, cell.c] = true;
                        }
                }
                // If diffNeed == diffCount, the difference is all mines; useful
                // internally, but flags are not part of the requested output.
            }
        }
    }
}
