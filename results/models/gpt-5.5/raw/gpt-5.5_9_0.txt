using System;
using System.Collections.Generic;
using System.IO;
using System.Text;

public static class Program
{
    struct Cut
    {
        public int Len;
        public int Idx;
        public Cut(int len, int idx) { Len = len; Idx = idx; }
    }

    sealed class FastScanner
    {
        readonly Stream _s = Console.OpenStandardInput();
        readonly byte[] _buf = new byte[1 << 20];
        int _len, _ptr;

        int Read()
        {
            if (_ptr >= _len)
            {
                _len = _s.Read(_buf, 0, _buf.Length);
                _ptr = 0;
                if (_len == 0) return -1;
            }
            return _buf[_ptr++];
        }

        public int NextInt()
        {
            int c;
            do { c = Read(); } while (c <= 32 && c >= 0);
            int sign = 1;
            if (c == '-') { sign = -1; c = Read(); }
            int v = 0;
            while (c > 32)
            {
                v = v * 10 + (c - '0');
                c = Read();
            }
            return v * sign;
        }
    }

    sealed class RemComparer : IComparer<int>
    {
        readonly List<int> _rem;
        public RemComparer(List<int> rem) { _rem = rem; }
        public int Compare(int a, int b)
        {
            int c = _rem[a].CompareTo(_rem[b]);
            return c != 0 ? c : a.CompareTo(b);
        }
    }

    public static void Main()
    {
        var fs = new FastScanner();
        int n = fs.NextInt();
        int stock = fs.NextInt();

        var cuts = new Cut[n];
        for (int i = 0; i < n; i++)
            cuts[i] = new Cut(fs.NextInt(), i);

        Array.Sort(cuts, (a, b) =>
        {
            int c = b.Len.CompareTo(a.Len);
            return c != 0 ? c : a.Idx.CompareTo(b.Idx);
        });

        var bins = new List<List<int>>();
        var rem = new List<int>();
        var tree = new SortedSet<int>(new RemComparer(rem));

        for (int i = 0; i < cuts.Length; i++)
        {
            int len = cuts[i].Len;
            int idx = cuts[i].Idx;

            int bin = -1;
            if (tree.Count != 0)
            {
                int probe = rem.Count;
                rem.Add(len);
                var view = tree.GetViewBetween(probe, int.MaxValue);
                if (view.Count != 0) bin = view.Min;
                rem.RemoveAt(rem.Count - 1);
            }

            if (bin < 0)
            {
                bin = bins.Count;
                bins.Add(new List<int>(4) { idx });
                rem.Add(stock - len);
                if (rem[bin] > 0) tree.Add(bin);
            }
            else
            {
                tree.Remove(bin);
                bins[bin].Add(idx);
                rem[bin] -= len;
                if (rem[bin] > 0) tree.Add(bin);
            }
        }

        using var output = new StreamWriter(Console.OpenStandardOutput(), new UTF8Encoding(false), 1 << 20);
        output.WriteLine(bins.Count);

        var sb = new StringBuilder(256);
        for (int i = 0; i < bins.Count; i++)
        {
            sb.Clear();
            var line = bins[i];
            for (int j = 0; j < line.Count; j++)
            {
                if (j != 0) sb.Append(' ');
                sb.Append(line[j]);
            }
            output.WriteLine(sb.ToString());
        }
    }
}
