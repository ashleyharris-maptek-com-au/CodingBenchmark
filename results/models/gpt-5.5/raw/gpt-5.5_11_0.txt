use std::cmp::Ordering;
use std::io::{self, Read};

const EPS: f64 = 1.0e-7;

#[derive(Clone, Copy, Debug)]
struct Pt {
    x: f64,
    y: f64,
}

#[derive(Clone)]
struct Piece {
    id: usize,
    pts: Vec<Pt>,
    area: f64,
    rots: Vec<RotShape>,
}

#[derive(Clone)]
struct RotShape {
    angle: f64,
    pts: Vec<Pt>,
    minx: f64,
    maxx: f64,
    miny: f64,
    maxy: f64,
    w: f64,
    h: f64,
    area: f64,
}

#[derive(Clone, Copy)]
struct Rect {
    x: f64,
    y: f64,
    w: f64,
    h: f64,
}

#[derive(Clone, Copy)]
struct Placement {
    stock: usize,
    x: f64,
    y: f64,
    rot: f64,
    rect: Rect,
}

struct StockInst {
    placed: Vec<Rect>,
    free: Vec<Rect>,
}

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn poly_area(p: &[Pt]) -> f64 {
    let mut s = 0.0;
    for i in 0..p.len() {
        let j = (i + 1) % p.len();
        s += p[i].x * p[j].y - p[j].x * p[i].y;
    }
    0.5 * s
}

fn bounds(p: &[Pt]) -> (f64, f64, f64, f64) {
    let mut minx = f64::INFINITY;
    let mut miny = f64::INFINITY;
    let mut maxx = f64::NEG_INFINITY;
    let mut maxy = f64::NEG_INFINITY;
    for q in p {
        minx = minx.min(q.x);
        miny = miny.min(q.y);
        maxx = maxx.max(q.x);
        maxy = maxy.max(q.y);
    }
    (minx, maxx, miny, maxy)
}

fn point_in_poly(p: Pt, poly: &[Pt]) -> bool {
    let mut inside = false;
    let n = poly.len();
    for i in 0..n {
        let a = poly[i];
        let b = poly[(i + 1) % n];
        if cross(a, b, p).abs() <= EPS
            && p.x >= a.x.min(b.x) - EPS && p.x <= a.x.max(b.x) + EPS
            && p.y >= a.y.min(b.y) - EPS && p.y <= a.y.max(b.y) + EPS {
            return true;
        }
        if (a.y > p.y) != (b.y > p.y) {
            let x = a.x + (p.y - a.y) * (b.x - a.x) / (b.y - a.y);
            if x >= p.x - EPS {
                inside = !inside;
            }
        }
    }
    inside
}

fn orient(a: Pt, b: Pt, c: Pt) -> f64 {
    cross(a, b, c)
}

fn seg_inter(a: Pt, b: Pt, c: Pt, d: Pt) -> bool {
    let o1 = orient(a, b, c);
    let o2 = orient(a, b, d);
    let o3 = orient(c, d, a);
    let o4 = orient(c, d, b);
    if o1.abs() <= EPS && c.x >= a.x.min(b.x) - EPS && c.x <= a.x.max(b.x) + EPS
        && c.y >= a.y.min(b.y) - EPS && c.y <= a.y.max(b.y) + EPS { return true; }
    if o2.abs() <= EPS && d.x >= a.x.min(b.x) - EPS && d.x <= a.x.max(b.x) + EPS
        && d.y >= a.y.min(b.y) - EPS && d.y <= a.y.max(b.y) + EPS { return true; }
    if o3.abs() <= EPS && a.x >= c.x.min(d.x) - EPS && a.x <= c.x.max(d.x) + EPS
        && a.y >= c.y.min(d.y) - EPS && a.y <= c.y.max(d.y) + EPS { return true; }
    if o4.abs() <= EPS && b.x >= c.x.min(d.x) - EPS && b.x <= c.x.max(d.x) + EPS
        && b.y >= c.y.min(d.y) - EPS && b.y <= c.y.max(d.y) + EPS { return true; }
    (o1 > EPS) != (o2 > EPS) && (o3 > EPS) != (o4 > EPS)
}

fn polys_overlap(a: &[Pt], b: &[Pt]) -> bool {
    for i in 0..a.len() {
        for j in 0..b.len() {
            if seg_inter(a[i], a[(i + 1) % a.len()], b[j], b[(j + 1) % b.len()]) {
                return true;
            }
        }
    }
    point_in_poly(a[0], b) || point_in_poly(b[0], a)
}

fn convex_hull(mut p: Vec<Pt>) -> Vec<Pt> {
    p.sort_by(|a, b| a.x.partial_cmp(&b.x).unwrap_or(Ordering::Equal)
        .then(a.y.partial_cmp(&b.y).unwrap_or(Ordering::Equal)));
    p.dedup_by(|a, b| (a.x - b.x).abs() < EPS && (a.y - b.y).abs() < EPS);
    if p.len() <= 2 { return p; }
    let mut lo = Vec::new();
    for &q in &p {
        while lo.len() >= 2 && cross(lo[lo.len()-2], lo[lo.len()-1], q) <= EPS { lo.pop(); }
        lo.push(q);
    }
    let mut up = Vec::new();
    for &q in p.iter().rev() {
        while up.len() >= 2 && cross(up[up.len()-2], up[up.len()-1], q) <= EPS { up.pop(); }
        up.push(q);
    }
    lo.pop();
    up.pop();
    lo.extend(up);
    lo
}

fn make_rot(pts: &[Pt], angle: f64, area: f64) -> RotShape {
    let c = angle.cos();
    let s = angle.sin();
    let r: Vec<Pt> = pts.iter().map(|p| Pt { x: p.x * c - p.y * s, y: p.x * s + p.y * c }).collect();
    let (minx, maxx, miny, maxy) = bounds(&r);
    RotShape { angle, pts: r, minx, maxx, miny, maxy, w: maxx - minx, h: maxy - miny, area }
}

fn rects_overlap(a: Rect, b: Rect) -> bool {
    a.x < b.x + b.w - EPS && a.x + a.w > b.x + EPS && a.y < b.y + b.h - EPS && a.y + a.h > b.y + EPS
}

fn rect_contained(a: Rect, b: Rect) -> bool {
    a.x >= b.x - EPS && a.y >= b.y - EPS && a.x + a.w <= b.x + b.w + EPS && a.y + a.h <= b.y + b.h + EPS
}

fn split_free(free: &mut Vec<Rect>, used: Rect) {
    let old = std::mem::take(free);
    for r in old {
        if !rects_overlap(r, used) {
            free.push(r);
            continue;
        }
        if used.x > r.x + EPS {
            free.push(Rect { x: r.x, y: r.y, w: used.x - r.x, h: r.h });
        }
        if used.x + used.w < r.x + r.w - EPS {
            let nx = used.x + used.w;
            free.push(Rect { x: nx, y: r.y, w: r.x + r.w - nx, h: r.h });
        }
        if used.y > r.y + EPS {
            free.push(Rect { x: r.x, y: r.y, w: r.w, h: used.y - r.y });
        }
        if used.y + used.h < r.y + r.h - EPS {
            let ny = used.y + used.h;
            free.push(Rect { x: r.x, y: ny, w: r.w, h: r.y + r.h - ny });
        }
    }
    let mut i = 0;
    while i < free.len() {
        let mut remove = false;
        for j in 0..free.len() {
            if i != j && rect_contained(free[i], free[j]) {
                remove = true;
                break;
            }
        }
        if remove { free.swap_remove(i); } else { i += 1; }
    }
}

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut it = input.split_whitespace();
    let sv: usize = it.next().unwrap().parse().unwrap();
    let np: usize = it.next().unwrap().parse().unwrap();
    let mut stock = Vec::with_capacity(sv);
    for _ in 0..sv {
        stock.push(Pt { x: it.next().unwrap().parse().unwrap(), y: it.next().unwrap().parse().unwrap() });
    }
    let stock_area = poly_area(&stock).abs().max(EPS);
    let hull = convex_hull(stock.clone());
    let (sx0, sx1, sy0, sy1) = bounds(&hull);
    let sw = sx1 - sx0;
    let sh = sy1 - sy0;
    let simple_fast = sv <= 4 && hull.len() <= 4 && np > 150;
    let rot_count = if np > 2500 { 2 } else if np > 700 { 4 } else { 12 };
    let mut pieces = Vec::with_capacity(np);
    for id in 0..np {
        let pv: usize = it.next().unwrap().parse().unwrap();
        let mut pts = Vec::with_capacity(pv);
        for _ in 0..pv {
            pts.push(Pt { x: it.next().unwrap().parse().unwrap(), y: it.next().unwrap().parse().unwrap() });
        }
        let area = poly_area(&pts).abs();
        let angles: Vec<f64> = if rot_count == 2 {
            vec![0.0, std::f64::consts::FRAC_PI_2]
        } else {
            (0..rot_count).map(|k| k as f64 * std::f64::consts::PI / rot_count as f64).collect()
        };
        let mut rots: Vec<RotShape> = angles.into_iter().map(|a| make_rot(&pts, a, area)).collect();
        rots.sort_by(|a, b| (a.w * a.h).partial_cmp(&(b.w * b.h)).unwrap_or(Ordering::Equal));
        rots.dedup_by(|a, b| (a.w - b.w).abs() < 1e-6 && (a.h - b.h).abs() < 1e-6);
        pieces.push(Piece { id, pts, area, rots });
    }
    pieces.sort_by(|a, b| b.area.partial_cmp(&a.area).unwrap_or(Ordering::Equal));

    let mut ans = vec![Placement { stock: 0, x: 0.0, y: 0.0, rot: 0.0, rect: Rect { x: 0.0, y: 0.0, w: 0.0, h: 0.0 } }; np];
    let mut stocks: Vec<StockInst> = Vec::new();

    for piece in &pieces {
        let mut best: Option<(usize, usize, usize, Rect, f64)> = None;
        for (si, st) in stocks.iter().enumerate() {
            for (ri, r) in piece.rots.iter().take(if np > 1000 { 3 } else { piece.rots.len() }).enumerate() {
                if r.w > sw + EPS || r.h > sh + EPS { continue; }
                for (fi, f) in st.free.iter().enumerate() {
                    if r.w <= f.w + EPS && r.h <= f.h + EPS {
                        let rect = Rect { x: f.x, y: f.y, w: r.w, h: r.h };
                        let score = rect.y * sw + rect.x + (f.w * f.h - r.w * r.h) * 0.001 + si as f64 * stock_area * 0.01;
                        if best.map_or(true, |b| score < b.4) {
                            best = Some((si, ri, fi, rect, score));
                        }
                    }
                }
            }
        }
        if best.is_none() {
            let si = stocks.len();
            stocks.push(StockInst {
                placed: Vec::new(),
                free: vec![Rect { x: sx0, y: sy0, w: sw, h: sh }],
            });
            let mut ri = 0;
            for (i, r) in piece.rots.iter().enumerate() {
                if r.w <= sw + EPS && r.h <= sh + EPS { ri = i; break; }
            }
            let r = &piece.rots[ri];
            best = Some((si, ri, 0, Rect { x: sx0, y: sy0, w: r.w, h: r.h }, 0.0));
        }
        let (si, ri, _fi, rect, _) = best.unwrap();
        let rot = &piece.rots[ri];
        let tx = rect.x - rot.minx;
        let ty = rect.y - rot.miny;
        let used = rect;
        if !simple_fast && piece.pts.len() <= 80 && stock.len() <= 80 {
            let placed_poly: Vec<Pt> = rot.pts.iter().map(|p| Pt { x: p.x + tx, y: p.y + ty }).collect();
            let inside = placed_poly.iter().all(|&p| point_in_poly(p, &stock));
            if inside {
                let _ = polys_overlap(&placed_poly, &placed_poly);
            }
        }
        ans[piece.id] = Placement { stock: si, x: tx, y: ty, rot: rot.angle, rect: used };
        stocks[si].placed.push(used);
        split_free(&mut stocks[si].free, used);
    }

    let mut out = String::new();
    out.push_str(&format!("{}\n", stocks.len()));
    for p in ans {
        out.push_str(&format!("{} {:.9} {:.9} {:.9}\n", p.stock, p.x, p.y, p.rot));
    }
    print!("{}", out);
}
