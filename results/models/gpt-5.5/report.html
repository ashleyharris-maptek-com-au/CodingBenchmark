static inline bool better2opt(const City& a, const City& b, const City& c, const City& d) {
    double oldLen = sqrt(dist2(a, b)) + sqrt(dist2(c, d));
    double newLen = sqrt(dist2(a, c)) + sqrt(dist2(b, d));

}

        let mut r = x;
        while self.p[r] != r {
            r = self.p[r];
        }
        let mut y = x;
        while self.p[y] != y {
            let z = self.p[y];
            self.p[y] = r;
            y = z;
        }
        r
    }

        let mut a = self.find(a);
        let mut b = self.find(b);

static bool rayIntersectsTri(const Vec3& orig, const Vec3& dir, const Vec3& v0, const Vec3& v1, const Vec3& v2, double& t) {
    const double eps = 1e-12;
    Vec3 e1 = v1 - v0, e2 = v2 - v0;
    Vec3 h = cross(dir, e2);
    double a = dot(e1, h);
    if (std::abs(a) < eps) return false;
    double f = 1.0 / a;
    Vec3 s = orig - v0;
    double u = f * dot(s, h);
    if (u < -eps || u > 1.0 + eps) return false;
    Vec3 q = cross(s, e1);
    double vv = f * dot(dir, q);
    if (vv < -eps || u + vv > 1.0 + eps) return false;
    t = f * dot(e2, q);
    return t > eps;
}

static bool allVerticesInside(const Mesh& inner, const Mesh& outer, const AABB& ob) {

        }
    };

    std::unordered_map<Key, int, Hash> map;
    map.reserve(in.v.size() * 2 + 1);
    Mesh out;

    for (size_t i = 0; i < in.v.size(); ++i) {
        Key k{llround(in.v[i].x * scale), llround(in.v[i].y * scale), llround(in.v[i].z * scale)};

            out.v.push_back(in.v[i]);

fn cross(a: Pt, b: Pt, c: Pt) -> f64 {
    (b.x - a.x) * (c.y - a.y) - (b.y - a.y) * (c.x - a.x)
}

fn convex_hull(mut pts: Vec<Pt>) -> Vec<Pt> {
    pts.sort_by(|a, b| a.x.total_cmp(&b.x).then(a.y.total_cmp(&b.y)));
    pts.dedup_by(|a, b| (a.x - b.x).abs() < 1e-9 && (a.y - b.y).abs() < 1e-9);
    if pts.len() <= 1 {
        return pts;
    }
    let mut lower: Vec<Pt> = Vec::new();
    for &p in &pts {
        while lower.len() >= 2 {
            let n = lower.len();
            if cross(lower[n - 2], lower[n - 1], p) <= 0.0 {
                lower.pop();
            } else {
                break;
            }

        lower.push(p);
