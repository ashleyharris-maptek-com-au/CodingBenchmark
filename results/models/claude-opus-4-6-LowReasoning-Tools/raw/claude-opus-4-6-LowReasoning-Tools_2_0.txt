using System;
using System.Collections.Generic;
using System.IO;

class ChinesePostman
{
    static int N, M;
    static List<(int to, int w, int idx)>[] adj;
    static int[] edgeU, edgeV, edgeW;

    static void Main()
    {
        var br = new StreamReader(Console.OpenStandardInput(), System.Text.Encoding.ASCII, false, 1 << 16);
        var first = br.ReadLine().Split();
        N = int.Parse(first[0]);
        M = int.Parse(first[1]);

        adj = new List<(int, int, int)>[N];
        for (int i = 0; i < N; i++) adj[i] = new List<(int, int, int)>();
        edgeU = new int[M];
        edgeV = new int[M];
        edgeW = new int[M];

        for (int i = 0; i < M; i++)
        {
            var parts = br.ReadLine().Split();
            int u = int.Parse(parts[0]), v = int.Parse(parts[1]), w = int.Parse(parts[2]);
            edgeU[i] = u; edgeV[i] = v; edgeW[i] = w;
            adj[u].Add((v, w, i));
            adj[v].Add((u, w, i));
        }

        int[] degree = new int[N];
        for (int i = 0; i < M; i++)
        {
            degree[edgeU[i]]++;
            degree[edgeV[i]]++;
        }

        List<int> oddVerts = new List<int>();
        for (int i = 0; i < N; i++)
            if (degree[i] % 2 == 1) oddVerts.Add(i);

        var extraEdges = new List<(int u, int v, int w)>();

        if (oddVerts.Count > 0)
        {
            int numOdd = oddVerts.Count;

            if (numOdd <= 20)
            {
                long[][] dist = new long[numOdd][];
                int[][] par = new int[numOdd][];

                for (int i = 0; i < numOdd; i++)
                    (dist[i], par[i]) = Dijkstra(oddVerts[i]);

                int full = (1 << numOdd) - 1;
                long[] dp = new long[full + 1];
                int[] dpMatch = new int[full + 1];
                Array.Fill(dp, long.MaxValue);
                dp[0] = 0;

                for (int mask = 0; mask < full; mask++)
                {
                    if (dp[mask] == long.MaxValue) continue;
                    int i = 0;
                    while ((mask & (1 << i)) != 0) i++;
                    for (int j = i + 1; j < numOdd; j++)
                    {
                        if ((mask & (1 << j)) != 0) continue;
                        int nmask = mask | (1 << i) | (1 << j);
                        long cost = dp[mask] + dist[i][oddVerts[j]];
                        if (cost < dp[nmask])
                        {
                            dp[nmask] = cost;
                            dpMatch[nmask] = (i << 16) | j;
                        }
                    }
                }

                int cur = full;
                while (cur != 0)
                {
                    int packed = dpMatch[cur];
                    int ii = packed >> 16, jj = packed & 0xFFFF;
                    cur ^= (1 << ii) | (1 << jj);
                    AddPath(par[ii], oddVerts[ii], oddVerts[jj], extraEdges);
                }
            }
            else
            {
                GreedyMatching(oddVerts, extraEdges);
            }
        }

        // Build multigraph and find Euler circuit
        int totalEdges = M + extraEdges.Count;
        var multigraph = new List<(int to, int edgeId)>[N];
        for (int i = 0; i < N; i++) multigraph[i] = new List<(int, int)>();

        for (int i = 0; i < M; i++)
        {
            multigraph[edgeU[i]].Add((edgeV[i], i));
            multigraph[edgeV[i]].Add((edgeU[i], i));
        }
        for (int i = 0; i < extraEdges.Count; i++)
        {
            int idx = M + i;
            multigraph[extraEdges[i].u].Add((extraEdges[i].v, idx));
            multigraph[extraEdges[i].v].Add((extraEdges[i].u, idx));
        }

        // Hierholzer's algorithm
        var usedEdge = new bool[totalEdges];
        var stackPos = new int[N];
        var stack = new Stack<int>();
        var circuit = new List<int>();

        stack.Push(0);
        while (stack.Count > 0)
        {
            int v = stack.Peek();
            bool found = false;
            while (stackPos[v] < multigraph[v].Count)
            {
                var (to, eid) = multigraph[v][stackPos[v]];
                stackPos[v]++;
                if (!usedEdge[eid])
                {
                    usedEdge[eid] = true;
                    stack.Push(to);
                    found = true;
                    break;
                }
            }
            if (!found)
            {
                stack.Pop();
                circuit.Add(v);
            }
        }

        var sw = new StreamWriter(Console.OpenStandardOutput(), System.Text.Encoding.ASCII, 1 << 16);
        sw.Write(circuit[0]);
        for (int i = 1; i < circuit.Count; i++)
        {
            sw.Write(' ');
            sw.Write(circuit[i]);
        }
        sw.WriteLine();
        sw.Flush();
    }

    static (long[] dist, int[] parent) Dijkstra(int src)
    {
        long[] dist = new long[N];
        int[] parent = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(parent, -1);
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            foreach (var (v, w, _) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parent[v] = u;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, parent);
    }

    static void AddPath(int[] parent, int src, int dst, List<(int, int, int)> extraEdges)
    {
        int cur = dst;
        while (cur != src)
        {
            int p = parent[cur];
            int w = 0;
            foreach (var (to, ww, _) in adj[p])
                if (to == cur) { w = ww; break; }
            extraEdges.Add((p, cur, w));
            cur = p;
        }
    }

    static void GreedyMatching(List<int> oddVerts, List<(int, int, int)> extraEdges)
    {
        int numOdd = oddVerts.Count;
        bool[] matched = new bool[numOdd];
        int remaining = numOdd;

        while (remaining > 0)
        {
            int first = -1;
            for (int i = 0; i < numOdd; i++)
                if (!matched[i]) { first = i; break; }

            var (dist, parent) = Dijkstra(oddVerts[first]);

            long bestDist = long.MaxValue;
            int bestJ = -1;
            for (int j = 0; j < numOdd; j++)
            {
                if (j == first || matched[j]) continue;
                if (dist[oddVerts[j]] < bestDist)
                {
                    bestDist = dist[oddVerts[j]];
                    bestJ = j;
                }
            }

            matched[first] = true;
            matched[bestJ] = true;
            remaining -= 2;

            AddPath(parent, oddVerts[first], oddVerts[bestJ], extraEdges);
        }
    }
}
