#include <cstdio>
#include <cmath>
#include <vector>
#include <algorithm>

using namespace std;

int main(){
    setvbuf(stdout, nullptr, _IOFBF, 1<<20);

    int V; double e;
    scanf("%d %lf", &V, &e);

    struct V3 { double x,y,z; };
    vector<V3> pts(V);
    for(int i=0;i<V;i++) scanf("%lf %lf %lf",&pts[i].x,&pts[i].y,&pts[i].z);

    struct Face { double nx,ny,nz,d; };
    vector<Face> faces;

    for(int i=0;i<V;i++) for(int j=i+1;j<V;j++) for(int k=j+1;k<V;k++){
        double ax=pts[j].x-pts[i].x, ay=pts[j].y-pts[i].y, az=pts[j].z-pts[i].z;
        double bx=pts[k].x-pts[i].x, by=pts[k].y-pts[i].y, bz=pts[k].z-pts[i].z;
        double nx=ay*bz-az*by, ny=az*bx-ax*bz, nz=ax*by-ay*bx;
        double len=sqrt(nx*nx+ny*ny+nz*nz);
        if(len<1e-10) continue;
        nx/=len; ny/=len; nz/=len;
        double d0=nx*pts[i].x+ny*pts[i].y+nz*pts[i].z;

        bool allBelow=true, allAbove=true;
        for(int l=0;l<V;l++){
            if(l==i||l==j||l==k) continue;
            double val=nx*pts[l].x+ny*pts[l].y+nz*pts[l].z-d0;
            if(val>1e-10) allBelow=false;
            if(val<-1e-10) allAbove=false;
        }
        if(!allBelow && !allAbove) continue;

        Face f;
        if(allAbove){
            f.nx=-nx; f.ny=-ny; f.nz=-nz; f.d=-d0;
        } else {
            f.nx=nx; f.ny=ny; f.nz=nz; f.d=d0;
        }
        faces.push_back(f);
    }

    vector<Face> ufaces;
    for(auto&f:faces){
        bool dup=false;
        for(auto&u:ufaces){
            if(fabs(f.nx-u.nx)<1e-8 && fabs(f.ny-u.ny)<1e-8 &&
               fabs(f.nz-u.nz)<1e-8 && fabs(f.d-u.d)<1e-8){
                dup=true; break;
            }
        }
        if(!dup) ufaces.push_back(f);
    }
    faces=ufaces;

    double R = e * sqrt(3.0/8.0);
    double spacing = e / sqrt(2.0);

    double bminx=1e18,bminy=1e18,bminz=1e18;
    double bmaxx=-1e18,bmaxy=-1e18,bmaxz=-1e18;
    for(auto&p:pts){
        bminx=min(bminx,p.x); bmaxx=max(bmaxx,p.x);
        bminy=min(bminy,p.y); bmaxy=max(bmaxy,p.y);
        bminz=min(bminz,p.z); bmaxz=max(bmaxz,p.z);
    }

    int nf=(int)faces.size();

    auto computeXRange = [&](double y, double z, double &xlo, double &xhi) -> bool {
        xlo=bminx; xhi=bmaxx;
        for(int fi=0;fi<nf;fi++){
            double rhs = faces[fi].d - R - faces[fi].ny*y - faces[fi].nz*z;
            double nx = faces[fi].nx;
            if(nx > 1e-12){
                xhi = min(xhi, rhs/nx);
            } else if(nx < -1e-12){
                xlo = max(xlo, rhs/nx);
            } else {
                if(rhs < -1e-10) return false;
            }
        }
        return xlo <= xhi;
    };

    double gox=bminx, goy=bminy, goz=bminz;

    long long count=0;
    for(double z=goz; z<=bmaxz; z+=spacing){
        for(double y=goy; y<=bmaxy; y+=spacing){
            double xlo,xhi;
            if(!computeXRange(y,z,xlo,xhi)) continue;
            long long ix0 = (long long)ceil((xlo - gox)/spacing);
            if(ix0<0) ix0=0;
            double xfirst = gox + ix0*spacing;
            if(xfirst > xhi) continue;
            long long n = (long long)((xhi - xfirst)/spacing) + 1;
            count += n;
        }
    }

    printf("%lld\n", count);

    for(double z=goz; z<=bmaxz; z+=spacing){
        for(double y=goy; y<=bmaxy; y+=spacing){
            double xlo,xhi;
            if(!computeXRange(y,z,xlo,xhi)) continue;
            long long ix0 = (long long)ceil((xlo - gox)/spacing);
            if(ix0<0) ix0=0;
            double xfirst = gox + ix0*spacing;
            for(double x=xfirst; x<=xhi; x+=spacing){
                printf("%.9f %.9f %.9f 0 0 0\n", x, y, z);
            }
        }
    }

    return 0;
}
