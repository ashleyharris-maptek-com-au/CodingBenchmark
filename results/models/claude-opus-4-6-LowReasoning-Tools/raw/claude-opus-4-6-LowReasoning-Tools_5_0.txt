use std::io::{self, Read};
use std::time::Instant;

fn main() {
    let mut input = String::new();
    io::stdin().read_to_string(&mut input).unwrap();
    let mut iter = input.split_whitespace();
    let w: usize = iter.next().unwrap().parse().unwrap();
    let h: usize = iter.next().unwrap().parse().unwrap();
    let obs_count: usize = iter.next().unwrap().parse().unwrap();
    let require_cycle: bool = iter.next().unwrap().parse::<u8>().unwrap() != 0;

    let n = w * h;
    let mut blocked = vec![false; n];
    for _ in 0..obs_count {
        let x: usize = iter.next().unwrap().parse().unwrap();
        let y: usize = iter.next().unwrap().parse().unwrap();
        blocked[y * w + x] = true;
    }

    if blocked[0] {
        println!("0");
        return;
    }

    let total_free: usize = blocked.iter().filter(|&&b| !b).count();
    if total_free == 0 {
        println!("0");
        return;
    }
    if total_free == 1 {
        if require_cycle {
            println!("0");
        } else {
            println!("1");
            println!("0 0");
        }
        return;
    }

    let dirs: [(i32, i32); 4] = [(1, 0), (-1, 0), (0, 1), (0, -1)];
    let mut adj: Vec<Vec<usize>> = vec![Vec::new(); n];
    for y in 0..h {
        for x in 0..w {
            if blocked[y * w + x] { continue; }
            for &(dx, dy) in &dirs {
                let nx = x as i32 + dx;
                let ny = y as i32 + dy;
                if nx >= 0 && nx < w as i32 && ny >= 0 && ny < h as i32 {
                    let ni = ny as usize * w + nx as usize;
                    if !blocked[ni] {
                        adj[y * w + x].push(ni);
                    }
                }
            }
        }
    }

    // Connectivity check
    let start = 0usize;
    {
        let mut vis = vec![false; n];
        let mut queue = std::collections::VecDeque::new();
        vis[start] = true;
        queue.push_back(start);
        let mut count = 0usize;
        while let Some(u) = queue.pop_front() {
            count += 1;
            for &v in &adj[u] {
                if !vis[v] {
                    vis[v] = true;
                    queue.push_back(v);
                }
            }
        }
        if count != total_free {
            println!("0");
            return;
        }
    }

    // Bipartite feasibility
    let mut color_count = [0usize; 2];
    for y in 0..h {
        for x in 0..w {
            if !blocked[y * w + x] {
                color_count[(x + y) % 2] += 1;
            }
        }
    }

    if require_cycle {
        if adj[start].len() < 2 || total_free % 2 != 0 || color_count[0] != color_count[1] {
            println!("0");
            return;
        }
    } else {
        if color_count[0] != color_count[1] && color_count[0] != color_count[1] + 1 {
            println!("0");
            return;
        }
    }

    let timer = Instant::now();
    let time_limit_ms = 280_000u128;

    // For small grids: exact backtracking with Warnsdorff + pruning
    // For large grids: randomized Warnsdorff
    let use_exact = total_free <= 64;

    if use_exact {
        if let Some(path) = exact_search(&adj, start, total_free, require_cycle, &timer, time_limit_ms) {
            print_path(&path, w);
            return;
        }
    }

    // Randomized Warnsdorff for large grids (and as fallback for small)
    if let Some(path) = random_warnsdorff(&adj, start, total_free, require_cycle, &timer, time_limit_ms, n) {
        print_path(&path, w);
        return;
    }

    println!("0");
}

fn print_path(path: &[usize], w: usize) {
    println!("{}", path.len());
    for &idx in path {
        println!("{} {}", idx % w, idx / w);
    }
}

fn exact_search(adj: &[Vec<usize>], start: usize, total_free: usize, require_cycle: bool, timer: &Instant, time_limit_ms: u128) -> Option<Vec<usize>> {
    let n = adj.len();
    let mut path = vec![0usize; total_free];
    path[0] = start;
    let mut visited = vec![false; n];
    visited[start] = true;

    let mut deg: Vec<u16> = adj.iter().map(|a| a.len() as u16).collect();
    for &v in &adj[start] {
        deg[v] -= 1;
    }

    let mut choices: Vec<Vec<usize>> = vec![Vec::new(); total_free];
    let mut choice_idx: Vec<usize> = vec![0; total_free];
    let mut depth = 1usize;

    // Build initial choices
    build_choices(adj, &deg, &visited, path[0], &mut choices[0], require_cycle, start, total_free, 1);
    choice_idx[0] = 0;

    let mut ops = 0u64;

    loop {
        ops += 1;
        if ops % 100_000 == 0 && timer.elapsed().as_millis() > time_limit_ms {
            return None;
        }

        if depth == total_free {
            if require_cycle {
                if adj[path[depth - 1]].contains(&start) {
                    return Some(path[..total_free].to_vec());
                }
            } else {
                return Some(path[..total_free].to_vec());
            }
            depth -= 1;
            let prev = path[depth];
            visited[prev] = false;
            for &u in &adj[prev] {
                if !visited[u] { deg[u] += 1; }
            }
            continue;
        }

        let d = depth - 1;
        if choice_idx[d] < choices[d].len() {
            let next = choices[d][choice_idx[d]];
            choice_idx[d] += 1;

            // Pruning: check for isolated unvisited neighbors
            let remaining = total_free - depth - 1;
            if remaining > 0 {
                let mut prune = false;
                for &u in &adj[next] {
                    if !visited[u] && deg[u] == 1 && remaining > 1 {
                        prune = true;
                        break;
                    }
                }
                if prune { continue; }
            }

            visited[next] = true;
            path[depth] = next;
            for &u in &adj[next] {
                if !visited[u] { deg[u] -= 1; }
            }
            depth += 1;

            if depth < total_free {
                build_choices(adj, &deg, &visited, next, &mut choices[depth - 1], require_cycle, start, total_free, depth);
                choice_idx[depth - 1] = 0;
            }
        } else {
            if d == 0 { return None; }
            depth -= 1;
            let prev = path[depth];
            visited[prev] = false;
            for &u in &adj[prev] {
                if !visited[u] { deg[u] += 1; }
            }
        }
    }
}

fn build_choices(adj: &[Vec<usize>], deg: &[u16], visited: &[bool], cur: usize, out: &mut Vec<usize>, require_cycle: bool, start: usize, total_free: usize, depth: usize) {
    out.clear();
    let mut nbuf: Vec<(u16, usize)> = Vec::with_capacity(4);
    for &v in &adj[cur] {
        if !visited[v] {
            nbuf.push((deg[v], v));
        }
    }
    nbuf.sort_unstable();

    if require_cycle && total_free - depth <= 3 {
        let mut front = Vec::new();
        let mut back = Vec::new();
        for &(_, v) in &nbuf {
            if adj[v].contains(&start) { front.push(v); } else { back.push(v); }
        }
        front.extend(back);
        *out = front;
    } else {
        *out = nbuf.iter().map(|&(_, v)| v).collect();
    }
}

fn random_warnsdorff(adj: &[Vec<usize>], start: usize, total_free: usize, require_cycle: bool, timer: &Instant, time_limit_ms: u128, n: usize) -> Option<Vec<usize>> {
    let mut rng: u64 = 0xdeadbeef12345678;

    loop {
        if timer.elapsed().as_millis() > time_limit_ms { return None; }

        let mut visited = vec![false; n];
        let mut path = vec![0usize; total_free];
        let mut deg: Vec<u16> = adj.iter().map(|a| a.len() as u16).collect();

        visited[start] = true;
        path[0] = start;
        for &u in &adj[start] { deg[u] -= 1; }
        let mut dp = 1usize;

        while dp < total_free {
            let cur = path[dp - 1];
            let mut best_deg = u16::MAX;
            let mut cands: Vec<usize> = Vec::with_capacity(4);
            for &v in &adj[cur] {
                if !visited[v] {
                    if deg[v] < best_deg {
                        best_deg = deg[v];
                        cands.clear();
                        cands.push(v);
                    } else if deg[v] == best_deg {
                        cands.push(v);
                    }
                }
            }
            if cands.is_empty() { break; }

            rng ^= rng << 13;
            rng ^= rng >> 7;
            rng ^= rng << 17;
            let chosen = cands[rng as usize % cands.len()];

            visited[chosen] = true;
            path[dp] = chosen;
            for &u in &adj[chosen] {
                if !visited[u] { deg[u] -= 1; }
            }
            dp += 1;
        }

        if dp == total_free {
            if require_cycle {
                if adj[path[dp - 1]].contains(&start) {
                    return Some(path);
                }
            } else {
                return Some(path);
            }
        }
    }
}
