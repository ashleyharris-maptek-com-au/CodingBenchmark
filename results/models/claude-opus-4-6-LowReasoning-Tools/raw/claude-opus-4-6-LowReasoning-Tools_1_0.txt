#include <iostream>
#include <vector>
#include <algorithm>
#include <cmath>
#include <numeric>
#include <chrono>
#include <cstring>

using namespace std;

struct Point { float x, y; };

static inline float dist2(const Point& a, const Point& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return dx*dx + dy*dy;
}
static inline float distf(const Point& a, const Point& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

vector<int> gridNN(const vector<Point>& pts, int N) {
    float minx = 1e18f, miny = 1e18f, maxx = -1e18f, maxy = -1e18f;
    for (int i = 0; i < N; i++) {
        if (pts[i].x < minx) minx = pts[i].x;
        if (pts[i].y < miny) miny = pts[i].y;
        if (pts[i].x > maxx) maxx = pts[i].x;
        if (pts[i].y > maxy) maxy = pts[i].y;
    }
    float rx = maxx - minx, ry = maxy - miny;
    if (rx < 1e-9f) rx = 1.0f;
    if (ry < 1e-9f) ry = 1.0f;

    int G = max(1, (int)sqrtf((float)N / 4.0f));
    if (G > 10000) G = 10000;

    vector<vector<int>> grid((long long)G * G);
    auto cellOf = [&](float x, float y, int& cx, int& cy) {
        cx = (int)((x - minx) / rx * (G - 1));
        cy = (int)((y - miny) / ry * (G - 1));
        if (cx < 0) cx = 0; if (cx >= G) cx = G - 1;
        if (cy < 0) cy = 0; if (cy >= G) cy = G - 1;
    };

    for (int i = 0; i < N; i++) {
        int cx, cy;
        cellOf(pts[i].x, pts[i].y, cx, cy);
        grid[(long long)cy * G + cx].push_back(i);
    }

    vector<bool> used(N, false);
    vector<int> tour;
    tour.reserve(N);
    tour.push_back(0);
    used[0] = true;

    for (int step = 1; step < N; step++) {
        int cur = tour.back();
        int gcx, gcy;
        cellOf(pts[cur].x, pts[cur].y, gcx, gcy);

        int best = -1;
        float bestD = 1e30f;

        for (int rad = 0; rad <= G; rad++) {
            int x0 = max(0, gcx - rad), x1 = min(G - 1, gcx + rad);
            int y0 = max(0, gcy - rad), y1 = min(G - 1, gcy + rad);
            for (int gy = y0; gy <= y1; gy++) {
                for (int gx = x0; gx <= x1; gx++) {
                    if (rad > 0 && gx > x0 && gx < x1 && gy > y0 && gy < y1) continue;
                    auto& cell = grid[(long long)gy * G + gx];
                    for (int idx : cell) {
                        if (!used[idx]) {
                            float d = dist2(pts[cur], pts[idx]);
                            if (d < bestD) { bestD = d; best = idx; }
                        }
                    }
                }
            }
            if (best >= 0) break;
        }
        tour.push_back(best);
        used[best] = true;
    }
    return tour;
}

vector<int> simpleNN(const vector<Point>& pts, int N) {
    vector<bool> used(N, false);
    vector<int> tour;
    tour.reserve(N);
    tour.push_back(0);
    used[0] = true;
    for (int i = 1; i < N; i++) {
        int cur = tour.back();
        int best = -1;
        float bestD = 1e30f;
        for (int j = 0; j < N; j++) {
            if (!used[j]) {
                float d = dist2(pts[cur], pts[j]);
                if (d < bestD) { bestD = d; best = j; }
            }
        }
        tour.push_back(best);
        used[best] = true;
    }
    return tour;
}

void twoOpt(vector<int>& tour, const vector<Point>& pts, double timeLimitSec) {
    int N = (int)tour.size();
    if (N < 4) return;
    auto start = chrono::steady_clock::now();
    bool improved = true;
    while (improved) {
        improved = false;
        for (int i = 0; i < N - 1; i++) {
            for (int j = i + 2; j < N; j++) {
                if (j == N - 1 && i == 0) continue;
                float d1 = distf(pts[tour[i]], pts[tour[i+1]]);
                float d2 = distf(pts[tour[j]], pts[tour[(j+1)%N]]);
                float d3 = distf(pts[tour[i]], pts[tour[j]]);
                float d4 = distf(pts[tour[i+1]], pts[tour[(j+1)%N]]);
                if (d3 + d4 < d1 + d2 - 1e-10f) {
                    reverse(tour.begin() + i + 1, tour.begin() + j + 1);
                    improved = true;
                }
            }
            if ((i & 127) == 0) {
                auto now = chrono::steady_clock::now();
                if (chrono::duration<double>(now - start).count() > timeLimitSec) return;
            }
        }
    }
}

int main() {
    ios::sync_with_stdio(false);
    cin.tie(nullptr);

    int N;
    cin >> N;

    vector<Point> pts(N);
    for (int i = 0; i < N; i++)
        cin >> pts[i].x >> pts[i].y;

    if (N <= 1) {
        for (int i = 0; i < N; i++) cout << i;
        cout << '\n';
        return 0;
    }

    auto globalStart = chrono::steady_clock::now();
    vector<int> tour;

    if (N <= 5000) {
        tour = simpleNN(pts, N);
        double elapsed = chrono::duration<double>(chrono::steady_clock::now() - globalStart).count();
        double remaining = 280.0 - elapsed;
        if (remaining > 0.5) {
            twoOpt(tour, pts, remaining * 0.9);
        }
    } else {
        tour = gridNN(pts, N);
        double elapsed = chrono::duration<double>(chrono::steady_clock::now() - globalStart).count();
        double remaining = 280.0 - elapsed;
        if (N <= 100000 && remaining > 2.0) {
            twoOpt(tour, pts, remaining * 0.8);
        }
    }

    if (tour[0] != 0) {
        int idx = 0;
        for (int i = 0; i < N; i++) if (tour[i] == 0) { idx = i; break; }
        rotate(tour.begin(), tour.begin() + idx, tour.end());
    }

    for (int i = 0; i < N; i++) {
        if (i) cout << ' ';
        cout << tour[i];
    }
    cout << '\n';
    return 0;
}
