use std::io::{self, BufRead, Write, BufWriter};
use std::collections::VecDeque;

fn main() {
    let stdin = io::stdin();
    let stdout = io::stdout();
    let mut out = BufWriter::new(stdout.lock());

    let mut lines = stdin.lock().lines();
    let first_line = lines.next().unwrap().unwrap();
    let mut it = first_line.split_whitespace();
    let n: usize = it.next().unwrap().parse().unwrap();
    let m: usize = it.next().unwrap().parse().unwrap();

    let mut adj: Vec<Vec<usize>> = vec![vec![]; n];

    for _ in 0..m {
        let line = lines.next().unwrap().unwrap();
        let mut it = line.split_whitespace();
        let u: usize = it.next().unwrap().parse().unwrap();
        let v: usize = it.next().unwrap().parse().unwrap();
        if u == v || u >= n || v >= n { continue; }
        adj[u].push(v);
        adj[v].push(u);
    }

    if n == 0 {
        return;
    }
    if n == 1 {
        writeln!(out, "1").unwrap();
        writeln!(out, "0.0 0.0").unwrap();
        return;
    }

    // Deduplicate adjacency
    for a in &mut adj {
        a.sort_unstable();
        a.dedup();
    }

    let mut x = vec![0.0f64; n];
    let mut y = vec![0.0f64; n];

    // Find connected components
    let mut comp = vec![usize::MAX; n];
    let mut comp_id = 0usize;
    let mut comp_nodes: Vec<Vec<usize>> = Vec::new();
    for start in 0..n {
        if comp[start] != usize::MAX { continue; }
        let mut queue = VecDeque::new();
        queue.push_back(start);
        comp[start] = comp_id;
        let mut nodes = Vec::new();
        while let Some(node) = queue.pop_front() {
            nodes.push(node);
            for &nb in &adj[node] {
                if comp[nb] == usize::MAX {
                    comp[nb] = comp_id;
                    queue.push_back(nb);
                }
            }
        }
        comp_nodes.push(nodes);
        comp_id += 1;
    }

    // Layout each component separately then pack them
    let mut offset_x = 0.0f64;

    for c in 0..comp_id {
        let nodes = &comp_nodes[c];
        let cn = nodes.len();

        if cn == 1 {
            x[nodes[0]] = offset_x;
            y[nodes[0]] = 0.0;
            offset_x += 2.0;
            continue;
        }

        // Build local adjacency
        let mut local_id = vec![0usize; n];
        for (i, &node) in nodes.iter().enumerate() {
            local_id[node] = i;
        }
        let mut local_adj: Vec<Vec<usize>> = vec![vec![]; cn];
        for &node in nodes {
            let li = local_id[node];
            for &nb in &adj[node] {
                if comp[nb] == c {
                    local_adj[li].push(local_id[nb]);
                }
            }
        }
        for a in &mut local_adj {
            a.sort_unstable();
            a.dedup();
        }

        let mut px = vec![0.0f64; cn];
        let mut py = vec![0.0f64; cn];

        if cn > 2000 {
            multilevel_layout(cn, &local_adj, &mut px, &mut py);
        } else {
            bfs_init(cn, &local_adj, &mut px, &mut py);
            let iters = if cn <= 50 { 500 } else if cn <= 200 { 300 } else if cn <= 500 { 200 } else { 150 };
            force_directed(cn, &local_adj, &mut px, &mut py, iters);
        }

        let mut min_x = f64::MAX;
        let mut max_x = f64::MIN;
        let mut min_y = f64::MAX;
        let mut _max_y = f64::MIN;
        for i in 0..cn {
            if px[i] < min_x { min_x = px[i]; }
            if px[i] > max_x { max_x = px[i]; }
            if py[i] < min_y { min_y = py[i]; }
            if py[i] > _max_y { _max_y = py[i]; }
        }

        for (i, &node) in nodes.iter().enumerate() {
            x[node] = px[i] - min_x + offset_x;
            y[node] = py[i] - min_y;
        }

        offset_x += (max_x - min_x) + 2.0 + (cn as f64).sqrt();
    }

    writeln!(out, "{}", n).unwrap();
    for i in 0..n {
        writeln!(out, "{:.6} {:.6}", x[i], y[i]).unwrap();
    }
}

fn bfs_init(n: usize, adj: &[Vec<usize>], px: &mut [f64], py: &mut [f64]) {
    let start = (0..n).max_by_key(|&i| adj[i].len()).unwrap_or(0);

    let mut dist = vec![usize::MAX; n];
    let mut order = Vec::with_capacity(n);
    let mut queue = VecDeque::new();
    dist[start] = 0;
    queue.push_back(start);

    while let Some(node) = queue.pop_front() {
        order.push(node);
        for &nb in &adj[node] {
            if dist[nb] == usize::MAX {
                dist[nb] = dist[node] + 1;
                queue.push_back(nb);
            }
        }
    }
    for i in 0..n {
        if dist[i] == usize::MAX {
            order.push(i);
            dist[i] = 0;
        }
    }

    let max_dist = *dist.iter().filter(|&&d| d != usize::MAX).max().unwrap_or(&1);
    let max_dist = max_dist.max(1);

    let mut count_at_dist = vec![0usize; max_dist + 1];
    let mut index_at_dist = vec![0usize; max_dist + 1];
    for &d in &dist {
        if d <= max_dist { count_at_dist[d] += 1; }
    }

    for &node in &order {
        let d = dist[node];
        if d > max_dist { continue; }
        let r = (d as f64) * 1.5;
        let count = count_at_dist[d];
        let idx = index_at_dist[d];
        index_at_dist[d] += 1;

        if d == 0 {
            px[node] = 0.0;
            py[node] = 0.0;
        } else {
            let angle = 2.0 * std::f64::consts::PI * (idx as f64) / (count as f64);
            px[node] = r * angle.cos();
            py[node] = r * angle.sin();
        }
    }
}

fn force_directed(n: usize, adj: &[Vec<usize>], px: &mut [f64], py: &mut [f64], iterations: usize) {
    let k = (1.0 / (n as f64).sqrt()) * 2.0;
    let k2 = k * k;

    let mut fx = vec![0.0f64; n];
    let mut fy = vec![0.0f64; n];

    let use_bh = n > 300;

    for iter in 0..iterations {
        let temp = k * (1.0 - iter as f64 / iterations as f64) * 0.5;
        if temp < 1e-6 { break; }

        for i in 0..n { fx[i] = 0.0; fy[i] = 0.0; }

        if use_bh {
            let tree = QuadTree::build(px, py, n);
            for i in 0..n {
                let (rfx, rfy) = tree.rf_node(0, px[i], py[i], k2, 1.0);
                fx[i] += rfx;
                fy[i] += rfy;
            }
        } else {
            for i in 0..n {
                for j in (i+1)..n {
                    let dx = px[i] - px[j];
                    let dy = py[i] - py[j];
                    let dist2 = dx * dx + dy * dy + 1e-10;
                    let dist = dist2.sqrt();
                    let force = k2 / dist;
                    let fdx = dx / dist * force;
                    let fdy = dy / dist * force;
                    fx[i] += fdx; fy[i] += fdy;
                    fx[j] -= fdx; fy[j] -= fdy;
                }
            }
        }

        for i in 0..n {
            for &j in &adj[i] {
                if j <= i { continue; }
                let dx = px[i] - px[j];
                let dy = py[i] - py[j];
                let dist = (dx * dx + dy * dy).sqrt().max(1e-10);
                let force = dist / k;
                let fdx = dx / dist * force;
                let fdy = dy / dist * force;
                fx[i] -= fdx; fy[i] -= fdy;
                fx[j] += fdx; fy[j] += fdy;
            }
        }

        for i in 0..n {
            let fm = (fx[i] * fx[i] + fy[i] * fy[i]).sqrt().max(1e-10);
            let scale = temp.min(fm) / fm;
            px[i] += fx[i] * scale;
            py[i] += fy[i] * scale;
        }
    }
}

struct QuadTree {
    nodes: Vec<QTNode>,
}

struct QTNode {
    cx: f64, cy: f64, mass: f64,
    x1: f64, y1: f64, x2: f64, y2: f64,
    children: [i32; 4],
    is_leaf: bool,
}

impl QuadTree {
    fn build(px: &[f64], py: &[f64], n: usize) -> Self {
        if n == 0 { return QuadTree { nodes: vec![] }; }
        let mut min_x = px[0]; let mut max_x = px[0];
        let mut min_y = py[0]; let mut max_y = py[0];
        for i in 1..n {
            if px[i] < min_x { min_x = px[i]; }
            if px[i] > max_x { max_x = px[i]; }
            if py[i] < min_y { min_y = py[i]; }
            if py[i] > max_y { max_y = py[i]; }
        }
        min_x -= 1.0; max_x += 1.0; min_y -= 1.0; max_y += 1.0;

        let mut tree = QuadTree { nodes: Vec::with_capacity(n * 2) };
        tree.nodes.push(QTNode {
            cx: 0.0, cy: 0.0, mass: 0.0,
            x1: min_x, y1: min_y, x2: max_x, y2: max_y,
            children: [-1; 4], is_leaf: true,
        });
        for i in 0..n { tree.insert(0, px[i], py[i], 0); }
        tree
    }

    fn insert(&mut self, idx: usize, px: f64, py: f64, depth: usize) {
        if depth > 50 { return; }
        let mass = self.nodes[idx].mass;
        let new_mass = mass + 1.0;
        self.nodes[idx].cx = (self.nodes[idx].cx * mass + px) / new_mass;
        self.nodes[idx].cy = (self.nodes[idx].cy * mass + py) / new_mass;
        self.nodes[idx].mass = new_mass;

        if self.nodes[idx].is_leaf && mass < 1.0 { return; }

        if self.nodes[idx].is_leaf {
            self.nodes[idx].is_leaf = false;
            let old_cx = (self.nodes[idx].cx * new_mass - px) / mass;
            let old_cy = (self.nodes[idx].cy * new_mass - py) / mass;
            self.insert_child(idx, old_cx, old_cy, depth);
        }
        self.insert_child(idx, px, py, depth);
    }

    fn insert_child(&mut self, idx: usize, px: f64, py: f64, depth: usize) {
        let mx = (self.nodes[idx].x1 + self.nodes[idx].x2) * 0.5;
        let my = (self.nodes[idx].y1 + self.nodes[idx].y2) * 0.5;
        let q = if px < mx { if py < my { 0 } else { 2 } } else { if py < my { 1 } else { 3 } };

        let child = self.nodes[idx].children[q];
        if child >= 0 {
            self.insert(child as usize, px, py, depth + 1);
        } else {
            let (x1, y1, x2, y2) = match q {
                0 => (self.nodes[idx].x1, self.nodes[idx].y1, mx, my),
                1 => (mx, self.nodes[idx].y1, self.nodes[idx].x2, my),
                2 => (self.nodes[idx].x1, my, mx, self.nodes[idx].y2),
                _ => (mx, my, self.nodes[idx].x2, self.nodes[idx].y2),
            };
            let new_idx = self.nodes.len();
            self.nodes.push(QTNode {
                cx: px, cy: py, mass: 1.0,
                x1, y1, x2, y2, children: [-1; 4], is_leaf: true,
            });
            self.nodes[idx].children[q] = new_idx as i32;
        }
    }

    fn rf_node(&self, idx: usize, px: f64, py: f64, k2: f64, theta: f64) -> (f64, f64) {
        let node = &self.nodes[idx];
        if node.mass < 0.5 { return (0.0, 0.0); }
        let dx = px - node.cx;
        let dy = py - node.cy;
        let dist2 = dx * dx + dy * dy;
        if dist2 < 1e-10 { return (0.0, 0.0); }
        let size = node.x2 - node.x1;
        if node.is_leaf || (size * size / dist2) < theta * theta {
            let dist = dist2.sqrt();
            let force = k2 * node.mass / dist;
            return (dx / dist * force, dy / dist * force);
        }
        let mut fx = 0.0; let mut fy = 0.0;
        for &child in &node.children {
            if child >= 0 {
                let (cfx, cfy) = self.rf_node(child as usize, px, py, k2, theta);
                fx += cfx; fy += cfy;
            }
        }
        (fx, fy)
    }
}

fn multilevel_layout(n: usize, adj: &[Vec<usize>], px: &mut [f64], py: &mut [f64]) {
    let max_levels = if n > 50000 { 8 } else if n > 10000 { 6 } else { 4 };
    let mut levels: Vec<(usize, Vec<Vec<usize>>, Vec<usize>)> = Vec::new();
    let mut cur_n = n;
    let mut cur_adj = adj.to_vec();

    for _ in 0..max_levels {
        if cur_n <= 100 { break; }
        let (cn, ca, map) = coarsen(cur_n, &cur_adj);
        if cn >= cur_n * 3 / 4 { break; }
        levels.push((cur_n, cur_adj, map));
        cur_n = cn;
        cur_adj = ca;
    }

    let mut cpx = vec![0.0f64; cur_n];
    let mut cpy = vec![0.0f64; cur_n];
    bfs_init(cur_n, &cur_adj, &mut cpx, &mut cpy);
    let iters = 300.min(500000 / cur_n.max(1));
    force_directed(cur_n, &cur_adj, &mut cpx, &mut cpy, iters);

    while let Some((fine_n, fine_adj, mapping)) = levels.pop() {
        let mut fpx = vec![0.0f64; fine_n];
        let mut fpy = vec![0.0f64; fine_n];
        for i in 0..fine_n {
            let c = mapping[i];
            let hash = (i.wrapping_mul(2654435761)) & 0xFFFF;
            fpx[i] = cpx[c] + (hash as f64 / 65536.0 - 0.5) * 0.01;
            fpy[i] = cpy[c] + ((hash.wrapping_mul(37)) as f64 / 65536.0 - 0.5) * 0.01;
        }
        cpx = fpx;
        cpy = fpy;
        let iters = 50.min(500000 / fine_n.max(1));
        force_directed(fine_n, &fine_adj, &mut cpx, &mut cpy, iters);
    }

    for i in 0..n { px[i] = cpx[i]; py[i] = cpy[i]; }
}

fn coarsen(n: usize, adj: &[Vec<usize>]) -> (usize, Vec<Vec<usize>>, Vec<usize>) {
    let mut mapping = vec![usize::MAX; n];
    let mut matched = vec![false; n];
    let mut cid = 0usize;

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by_key(|&i| adj[i].len());

    for &i in &order {
        if matched[i] { continue; }
        matched[i] = true;
        mapping[i] = cid;
        let mut best = None;
        for &j in &adj[i] {
            if !matched[j] { best = Some(j); break; }
        }
        if let Some(j) = best {
            matched[j] = true;
            mapping[j] = cid;
        }
        cid += 1;
    }

    let mut ca: Vec<Vec<usize>> = vec![vec![]; cid];
    for i in 0..n {
        let ci = mapping[i];
        for &j in &adj[i] {
            let cj = mapping[j];
            if ci != cj { ca[ci].push(cj); }
        }
    }
    for a in &mut ca { a.sort_unstable(); a.dedup(); }
    (cid, ca, mapping)
}
