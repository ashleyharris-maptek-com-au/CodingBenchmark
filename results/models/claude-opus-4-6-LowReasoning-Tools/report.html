struct Point { float x, y; };

static inline float dist2(const Point& a, const Point& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return dx*dx + dy*dy;
}
static inline float distf(const Point& a, const Point& b) {
    float dx = a.x - b.x, dy = a.y - b.y;
    return sqrtf(dx*dx + dy*dy);
}

vector<int> gridNN(const vector<Point>& pts, int N) {
    float minx = 1e18f, miny = 1e18f, maxx = -1e18f, maxy = -1e18f;

        if (pts[i].y < miny) miny = pts[i].y;

    if (ry < 1e-9f) ry = 1.0f;

    if (G > 10000) G = 10000;

        cy = (int)((y - miny) / ry * (G - 1));

    static (long[] dist, int[] parent) Dijkstra(int src)
    {
        long[] dist = new long[N];
        int[] parent = new int[N];
        Array.Fill(dist, long.MaxValue);
        Array.Fill(parent, -1);
        dist[src] = 0;
        var pq = new PriorityQueue<int, long>();
        pq.Enqueue(src, 0);
        while (pq.Count > 0)
        {
            pq.TryDequeue(out int u, out long d);
            if (d > dist[u]) continue;
            foreach (var (v, w, _) in adj[u])
            {
                long nd = d + w;
                if (nd < dist[v])
                {
                    dist[v] = nd;
                    parent[v] = u;
                    pq.Enqueue(v, nd);
                }
            }
        }
        return (dist, parent);
    }

    static void AddPath(int[] parent, int src, int dst, List<(int, int, int)> extraEdges)
    {

        while (cur != src)
        {
            int p = parent[cur];
            int w = 0;
            foreach (var (to, ww, _) in adj[p])
                if (to == cur) { w = ww; break; }
            extraEdges.Add((p, cur, w));
            cur = p;
        }

                queue.push_back(nb);

struct QTNode {
    cx: f64, cy: f64, mass: f64,
    x1: f64, y1: f64, x2: f64, y2: f64,
    children: [i32; 4],
    is_leaf: bool,
}

impl QuadTree {
    fn build(px: &[f64], py: &[f64], n: usize) -> Self {
        if n == 0 { return QuadTree { nodes: vec![] }; }
        let mut min_x = px[0]; let mut max_x = px[0];
        let mut min_y = py[0]; let mut max_y = py[0];
        for i in 1..n {
            if px[i] < min_x { min_x = px[i]; }
            if px[i] > max_x { max_x = px[i]; }
            if py[i] < min_y { min_y = py[i]; }
            if py[i] > max_y { max_y = py[i]; }
        }
        min_x -= 1.0; max_x += 1.0; min_y -= 1.0; max_y += 1.0;

        let mut tree = QuadTree { nodes: Vec::with_capacity(n * 2) };
        tree.nodes.push(QTNode {
            cx: 0.0, cy: 0.0, mass: 0.0,
            x1: min_x, y1: min_y, x2: max_x, y2: max_y,
            children: [-1; 4], is_leaf: true,
        });
        for i in 0..n { tree.insert(0, px[i], py[i], 0); }
        tree
    }

    fn insert(&mut self, idx: usize, px: f64, py: f64, depth: usize) {
        if depth > 50 { return; }
        let mass = self.nodes[idx].mass;
        let new_mass = mass + 1.0;
        self.nodes[idx].cx = (self.nodes[idx].cx * mass + px) / new_mass;
        self.nodes[idx].cy = (self.nodes[idx].cy * mass + py) / new_mass;
        self.nodes[idx].mass = new_mass;

        if self.nodes[idx].is_leaf && mass < 1.0 { return; }

        if self.nodes[idx].is_leaf {
            self.nodes[idx].is_leaf = false;
            let old_cx = (self.nodes[idx].cx * new_mass - px) / mass;
            let old_cy = (self.nodes[idx].cy * new_mass - py) / mass;
            self.insert_child(idx, old_cx, old_cy, depth);
        }
        self.insert_child(idx, px, py, depth);
    }

    fn insert_child(&mut self, idx: usize, px: f64, py: f64, depth: usize) {
        let mx = (self.nodes[idx].x1 + self.nodes[idx].x2) * 0.5;
        let my = (self.nodes[idx].y1 + self.nodes[idx].y2) * 0.5;
        let q = if px < mx { if py < my { 0 } else { 2 } } else { if py < my { 1 } else { 3 } };

        let child = self.nodes[idx].children[q];
        if child >= 0 {
            self.insert(child as usize, px, py, depth + 1);
        } else {
            let (x1, y1, x2, y2) = match q {
                0 => (self.nodes[idx].x1, self.nodes[idx].y1, mx, my),
                1 => (mx, self.nodes[idx].y1, self.nodes[idx].x2, my),
                2 => (self.nodes[idx].x1, my, mx, self.nodes[idx].y2),
                _ => (mx, my, self.nodes[idx].x2, self.nodes[idx].y2),
            };
            let new_idx = self.nodes.len();
            self.nodes.push(QTNode {
                cx: px, cy: py, mass: 1.0,
                x1, y1, x2, y2, children: [-1; 4], is_leaf: true,
            });
            self.nodes[idx].children[q] = new_idx as i32;
        }
    }

    fn rf_node(&self, idx: usize, px: f64, py: f64, k2: f64, theta: f64) -> (f64, f64) {
        let node = &self.nodes[idx];
        if node.mass < 0.5 { return (0.0, 0.0); }
        let dx = px - node.cx;
        let dy = py - node.cy;
        let dist2 = dx * dx + dy * dy;
        if dist2 < 1e-10 { return (0.0, 0.0); }
        let size = node.x2 - node.x1;
        if node.is_leaf || (size * size / dist2) < theta * theta {
            let dist = dist2.sqrt();
            let force = k2 * node.mass / dist;
            return (dx / dist * force, dy / dist * force);
        }
        let mut fx = 0.0; let mut fy = 0.0;
        for &child in &node.children {
            if child >= 0 {
                let (cfx, cfy) = self.rf_node(child as usize, px, py, k2, theta);
                fx += cfx; fy += cfy;
            }
        }
        (fx, fy)
    }
}

fn multilevel_layout(n: usize, adj: &[Vec<usize>], px: &mut [f64], py: &mut [f64]) {

    while let Some((fine_n, fine_adj, mapping)) = levels.pop() {

}

fn coarsen(n: usize, adj: &[Vec<usize>]) -> (usize, Vec<Vec<usize>>, Vec<usize>) {
    let mut mapping = vec![usize::MAX; n];
    let mut matched = vec![false; n];
    let mut cid = 0usize;

    let mut order: Vec<usize> = (0..n).collect();
    order.sort_unstable_by_key(|&i| adj[i].len());

    for &i in &order {
        if matched[i] { continue; }
        matched[i] = true;
        mapping[i] = cid;
        let mut best = None;
        for &j in &adj[i] {
            if !matched[j] { best = Some(j); break; }
        }
        if let Some(j) = best {
            matched[j] = true;
            mapping[j] = cid;
        }
        cid += 1;
    }

    let mut ca: Vec<Vec<usize>> = vec![vec![]; cid];
    for i in 0..n {
        let ci = mapping[i];
        for &j in &adj[i] {
            let cj = mapping[j];
            if ci != cj { ca[ci].push(cj); }
        }
    }
    for a in &mut ca { a.sort_unstable(); a.dedup(); }
    (cid, ca, mapping)
}

fn exact_search(adj: &[Vec<usize>], start: usize, total_free: usize, require_cycle: bool, timer: &Instant, time_limit_ms: u128) -> Option<Vec<usize>> {

    for &v in &adj[start] {
        deg[v] -= 1;
    }

    let mut choice_idx: Vec<usize> = vec![0; total_free];
    let mut depth = 1usize;

    build_choices(adj, &deg, &visited, path[0], &mut choices[0], require_cycle, start, total_free, 1);
    choice_idx[0] = 0;

        ops += 1;
        if ops % 100_000 == 0 && timer.elapsed().as_millis() > time_limit_ms {
            return None;
        }

            if require_cycle {
                if adj[path[depth - 1]].contains(&start) {
                    return Some(path[..total_free].to_vec());
                }
            } else {
                return Some(path[..total_free].to_vec());
            }
            depth -= 1;
            let prev = path[depth];
            visited[prev] = false;
            for &u in &adj[prev] {
                if !visited[u] { deg[u] += 1; }
            }
            continue;
        }

            choice_idx[d] += 1;

                    if !visited[u] && deg[u] == 1 && remaining > 1 {
                        prune = true;
                        break;
                    }
                }
                if prune { continue; }

            path[depth] = next;
            for &u in &adj[next] {

            }
            depth += 1;

fn build_choices(adj: &[Vec<usize>], deg: &[u16], visited: &[bool], cur: usize, out: &mut Vec<usize>, require_cycle: bool, start: usize, total_free: usize, depth: usize) {
    out.clear();
    let mut nbuf: Vec<(u16, usize)> = Vec::with_capacity(4);
    for &v in &adj[cur] {
        if !visited[v] {
            nbuf.push((deg[v], v));
        }
    }
    nbuf.sort_unstable();

    if require_cycle && total_free - depth <= 3 {
        let mut front = Vec::new();
        let mut back = Vec::new();
        for &(_, v) in &nbuf {
            if adj[v].contains(&start) { front.push(v); } else { back.push(v); }
        }
        front.extend(back);
        *out = front;
    } else {
        *out = nbuf.iter().map(|&(_, v)| v).collect();
    }
}

fn random_warnsdorff(adj: &[Vec<usize>], start: usize, total_free: usize, require_cycle: bool, timer: &Instant, time_limit_ms: u128, n: usize) -> Option<Vec<usize>> {
    let mut rng: u64 = 0xdeadbeef12345678;

    loop {
        if timer.elapsed().as_millis() > time_limit_ms { return None; }

        let mut visited = vec![false; n];
        let mut path = vec![0usize; total_free];
        let mut deg: Vec<u16> = adj.iter().map(|a| a.len() as u16).collect();

        visited[start] = true;
        path[0] = start;
        for &u in &adj[start] { deg[u] -= 1; }
        let mut dp = 1usize;

        while dp < total_free {
            let cur = path[dp - 1];
            let mut best_deg = u16::MAX;
            let mut cands: Vec<usize> = Vec::with_capacity(4);
            for &v in &adj[cur] {
                if !visited[v] {
                    if deg[v] < best_deg {
                        best_deg = deg[v];
                        cands.clear();
                        cands.push(v);
                    } else if deg[v] == best_deg {
                        cands.push(v);
                    }
                }
            }
            if cands.is_empty() { break; }

            rng ^= rng >> 7;
            rng ^= rng << 17;
            let chosen = cands[rng as usize % cands.len()];

            path[dp] = chosen;
            for &u in &adj[chosen] {
                if !visited[u] { deg[u] -= 1; }
            }
            dp += 1;
        }

            if require_cycle {
                if adj[path[dp - 1]].contains(&start) {
                    return Some(path);
                }
            } else {
                return Some(path);
            }
        }
    }
